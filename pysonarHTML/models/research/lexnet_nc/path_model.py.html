  53   def __init__(self, hparams, lemma_embeddings, instance):
  54     """Initialize the LexNET classifier.
  55 
  56     Args:
  57       hparams: the hyper-parameters.
  58       lemma_embeddings: word embeddings for the path-based component.
  59       instance: string tensor containing the input instance
  60     """
  61     self.hparams = hparams
  62     self.lemma_embeddings = lemma_embeddings
  63     self.instance = instance
  64     self.vocab_size, self.lemma_dim = self.lemma_embeddings.shape
  65 
  66     # Set the random seed
  67     if hparams.random_seed > 0:
  68       tf.set_random_seed(hparams.random_seed)
  69 
  70     # Create the network
  71     self.__create_computation_graph__()
  72 
  73   def __create_computation_graph__(self):
  74     """Initialize the model and define the graph."""
  75     self.lstm_input_dim = sum([self.hparams.lemma_dim, self.hparams.pos_dim,
  76                                self.hparams.dep_dim, self.hparams.dir_dim])
  77     self.lstm_output_dim = self.lstm_input_dim
  78 
  79     network_input = self.lstm_output_dim
  80     self.lemma_lookup = tf.get_variable(
  81         'lemma_lookup',
  82         initializer=self.lemma_embeddings,
  83         dtype=tf.float32,
  84         trainable=self.hparams.learn_lemmas)
  85     self.pos_lookup = tf.get_variable(
  86         'pos_lookup',
  87         shape=[self.hparams.num_pos, self.hparams.pos_dim],
  88         dtype=tf.float32)
  89     self.dep_lookup = tf.get_variable(
  90         'dep_lookup',
  91         shape=[self.hparams.num_dep, self.hparams.dep_dim],
  92         dtype=tf.float32)
  93     self.dir_lookup = tf.get_variable(
  94         'dir_lookup',
  95         shape=[self.hparams.num_directions, self.hparams.dir_dim],
  96         dtype=tf.float32)
  97 
  98     self.weights1 = tf.get_variable(
  99         'W1',
 100         shape=[network_input, self.hparams.num_classes],
 101         dtype=tf.float32)
 102     self.bias1 = tf.get_variable(
 103         'b1',
 104         shape=[self.hparams.num_classes],
 105         dtype=tf.float32)
 106 
 107     # Define the variables
 108     (self.batch_paths,
 109      self.path_counts,
 110      self.seq_lengths,
 111      self.path_strings,
 112      self.batch_labels) = _parse_tensorflow_example(
 113          self.instance, self.hparams.max_path_len, self.hparams.input_keep_prob)
 114 
 115     # Create the LSTM
 116     self.__lstm__()
 117 
 118     # Create the MLP
 119     self.__mlp__()
 120 
 121     self.instances_to_load = tf.placeholder(dtype=tf.string, shape=[None])
 122     self.labels_to_load = lexnet_common.load_all_labels(self.instances_to_load)
 123 
 124   def load_labels(self, session, batch_instances):

 127     Args:
 128       session: the current TensorFlow session.

 131     Returns:
 132       the labels.

 137   def run_one_epoch(self, session, num_steps):
 138     """Train the model.
 139 
 140     Args:
 141       session: The current TensorFlow session.
 142       num_steps: The number of steps in each epoch.
 143 
 144     Returns:
 145       The mean loss for the epoch.
 146 
 147     Raises:
 148       ArithmeticError: if the loss becomes non-finite.
 149     """
 150     losses = []
 151 
 152     for step in range(num_steps):
 153       curr_loss, _ = session.run([self.cost, self.train_op])
 154       if not np.isfinite(curr_loss):
 155         raise ArithmeticError('nan loss at step %d' % step)
 156 
 157       losses.append(curr_loss)
 158 
 159     return np.mean(losses)

 161   def predict(self, session, inputs):
 162     """Predict the classification of the test set.
 163 
 164     Args:
 165       session: The current TensorFlow session.
 166       inputs: the train paths, x, y and/or nc vectors
 167 
 168     Returns:
 169       The test predictions.
 170     """
 171     predictions, _ = zip(*self.predict_with_score(session, inputs))
 172     return np.array(predictions)

 174   def predict_with_score(self, session, inputs):
 175     """Predict the classification of the test set.
 176 
 177     Args:
 178       session: The current TensorFlow session.
 179       inputs: the test paths, x, y and/or nc vectors
 180 
 181     Returns:
 182       The test predictions along with their scores.
 183     """
 184     test_pred = [0] * len(inputs)
 185 
 186     for index, instance in enumerate(inputs):
 187 
 188       prediction, scores = session.run(
 189           [self.predictions, self.scores],
 190           feed_dict={self.instance: instance})
 191 
 192       test_pred[index] = (prediction, scores[prediction])
 193 
 194     return test_pred

 196   def __mlp__(self):
 197     """Performs the MLP operations.
 198 
 199     Returns: the prediction object to be computed in a Session
 200     """
 201     # Feed the paths to the MLP: path_embeddings is
 202     # [num_batch_paths, output_dim], and when we multiply it by W
 203     # ([output_dim, num_classes]), we get a matrix of class distributions:
 204     # [num_batch_paths, num_classes].
 205     self.distributions = tf.matmul(self.path_embeddings, self.weights1)
 206 
 207     # Now, compute weighted average on the class distributions, using the path
 208     # frequency as weights.
 209 
 210     # First, reshape path_freq to the same shape of distributions
 211     self.path_freq = tf.tile(tf.expand_dims(self.path_counts, -1),
 212                              [1, self.hparams.num_classes])
 213 
 214     # Second, multiply the distributions and frequencies element-wise.
 215     self.weighted = tf.multiply(self.path_freq, self.distributions)
 216 
 217     # Finally, take the average to get a tensor of shape [1, num_classes].
 218     self.weighted_sum = tf.reduce_sum(self.weighted, 0)
 219     self.num_paths = tf.clip_by_value(tf.reduce_sum(self.path_counts),
 220                                       1, np.inf)
 221     self.num_paths = tf.tile(tf.expand_dims(self.num_paths, -1),
 222                              [self.hparams.num_classes])
 223     self.scores = tf.div(self.weighted_sum, self.num_paths)
 224     self.predictions = tf.argmax(self.scores)
 225 
 226     # Define the loss function and the optimization algorithm
 227     self.cross_entropies = tf.nn.sparse_softmax_cross_entropy_with_logits(
 228         logits=self.scores, labels=tf.reduce_mean(self.batch_labels))
 229     self.cost = tf.reduce_sum(self.cross_entropies, name='cost')
 230     self.global_step = tf.Variable(0, name='global_step', trainable=False)
 231     self.optimizer = tf.train.AdamOptimizer()
 232     self.train_op = self.optimizer.minimize(self.cost,
 233                                             global_step=self.global_step)
 234 
 235   def __lstm__(self):
 236     """Defines the LSTM operations.
 237 
 238     Returns:
 239       A matrix of path embeddings.
 240     """
 241     lookup_tables = [self.lemma_lookup, self.pos_lookup,
 242                      self.dep_lookup, self.dir_lookup]
 243 
 244     # Split the edges to components: list of 4 tensors
 245     # [num_batch_paths, max_path_len, 1]
 246     self.edge_components = tf.split(self.batch_paths, 4, axis=2)
 247 
 248     # Look up the components embeddings and concatenate them back together
 249     self.path_matrix = tf.concat([
 250         tf.squeeze(tf.nn.embedding_lookup(lookup_table, component), 2)
 251         for lookup_table, component in
 252         zip(lookup_tables, self.edge_components)
 253     ], axis=2)
 254 
 255     self.sequence_lengths = tf.reshape(self.seq_lengths, [-1])
 256 
 257     # Define the LSTM.
 258     # The input is [num_batch_paths, max_path_len, input_dim].
 259     lstm_cell = tf.contrib.rnn.BasicLSTMCell(self.lstm_output_dim)
 260 
 261     # The output is [num_batch_paths, max_path_len, output_dim].
 262     self.lstm_outputs, _ = tf.nn.dynamic_rnn(
 263         lstm_cell, self.path_matrix, dtype=tf.float32,
 264         sequence_length=self.sequence_lengths)
 265 
 266     # Slice the last *relevant* output for each instance ->
 267     # [num_batch_paths, output_dim]
 268     self.path_embeddings = _extract_last_relevant(self.lstm_outputs,
 269                                                   self.sequence_lengths)

 272 def _parse_tensorflow_example(record, max_path_len, input_keep_prob):
 273   """Reads TensorFlow examples from a RecordReader.
 274 
 275   Args:
 276     record: a record with TensorFlow example.
 277     max_path_len: the maximum path length.
 278     input_keep_prob: 1 - the word dropout probability
 279 
 280   Returns:
 281     The paths and counts
 282   """
 283   features = tf.parse_single_example(record, {
 284       'lemmas':
 285           tf.FixedLenSequenceFeature(
 286               shape=(), dtype=tf.int64, allow_missing=True),
 287       'postags':
 288           tf.FixedLenSequenceFeature(
 289               shape=(), dtype=tf.int64, allow_missing=True),
 290       'deplabels':
 291           tf.FixedLenSequenceFeature(
 292               shape=(), dtype=tf.int64, allow_missing=True),
 293       'dirs':
 294           tf.FixedLenSequenceFeature(
 295               shape=(), dtype=tf.int64, allow_missing=True),
 296       'counts':
 297           tf.FixedLenSequenceFeature(
 298               shape=(), dtype=tf.int64, allow_missing=True),
 299       'pathlens':
 300           tf.FixedLenSequenceFeature(
 301               shape=(), dtype=tf.int64, allow_missing=True),
 302       'reprs':
 303           tf.FixedLenSequenceFeature(
 304               shape=(), dtype=tf.string, allow_missing=True),
 305       'rel_id':
 306           tf.FixedLenFeature([], dtype=tf.int64)
 307   })
 308 
 309   path_counts = tf.to_float(features['counts'])
 310   seq_lengths = features['pathlens']
 311 
 312   # Concatenate the edge components to create a path tensor:
 313   # [max_paths_per_ins, max_path_length, 4]
 314   lemmas = _word_dropout(
 315       tf.reshape(features['lemmas'], [-1, max_path_len]), input_keep_prob)
 316 
 317   paths = tf.stack(
 318       [lemmas] + [
 319           tf.reshape(features[f], [-1, max_path_len])
 320           for f in ('postags', 'deplabels', 'dirs')
 321       ],
 322       axis=-1)
 323 
 324   path_strings = features['reprs']
 325 
 326   # Add an empty path to pairs with no paths
 327   paths = tf.cond(
 328       tf.shape(paths)[0] > 0,
 329       lambda: paths,
 330       lambda: tf.zeros([1, max_path_len, 4], dtype=tf.int64))
 331 
 332   # Paths are left-padded. We reverse them to make them right-padded.
 333   #paths = tf.reverse(paths, axis=[1])
 334 
 335   path_counts = tf.cond(
 336       tf.shape(path_counts)[0] > 0,
 337       lambda: path_counts,
 338       lambda: tf.constant([1.0], dtype=tf.float32))
 339 
 340   seq_lengths = tf.cond(
 341       tf.shape(seq_lengths)[0] > 0,
 342       lambda: seq_lengths,
 343       lambda: tf.constant([1], dtype=tf.int64))
 344 
 345   # Duplicate the label for each path
 346   labels = tf.ones_like(path_counts, dtype=tf.int64) * features['rel_id']
 347 
 348   return paths, path_counts, seq_lengths, path_strings, labels

 351 def _extract_last_relevant(output, seq_lengths):
 352   """Get the last relevant LSTM output cell for each batch instance.
 353 
 354   Args:
 355     output: the LSTM outputs - a tensor with shape
 356     [num_paths, output_dim, max_path_len]
 357     seq_lengths: the sequences length per instance
 358 
 359   Returns:
 360     The last relevant LSTM output cell for each batch instance.
 361   """
 362   max_length = int(output.get_shape()[1])
 363   path_lengths = tf.clip_by_value(seq_lengths - 1, 0, max_length)
 364   relevant = tf.reduce_sum(tf.multiply(output, tf.expand_dims(
 365       tf.one_hot(path_lengths, max_length), -1)), 1)
 366   return relevant

 369 def _word_dropout(words, input_keep_prob):
 370   """Drops words with probability 1 - input_keep_prob.
 371 
 372   Args:
 373     words: a list of lemmas from the paths.
 374     input_keep_prob: the probability to keep the word.
 375 
 376   Returns:
 377     The revised list where some of the words are <UNK>ed.
 378   """
 379   # Create the mask: (-1) to drop, 1 to keep
 380   prob = tf.random_uniform(tf.shape(words), 0, 1)
 381   condition = tf.less(prob, (1 - input_keep_prob))
 382   mask = tf.where(condition,
 383                   tf.negative(tf.ones_like(words)), tf.ones_like(words))
 384 
 385   # We need to keep zeros (<PAD>), and change other numbers to 1 (<UNK>)
 386   # if their mask is -1. First, we multiply the mask and the words.
 387   # Zeros will stay zeros, and words to drop will become negative.
 388   # Then, we change negative values to 1.
 389   masked_words = tf.multiply(mask, words)
 390   condition = tf.less(masked_words, 0)
 391   dropped_words = tf.where(condition, tf.ones_like(words), words)
 392   return dropped_words

 395 def compute_path_embeddings(model, session, instances):
 396   """Compute the path embeddings for all the distinct paths.
 397 
 398   Args:
 399     model: The trained path-based model.
 400     session: The current TensorFlow session.
 401     instances: All the train, test and validation instances.
 402 
 403   Returns:
 404     The path to ID index and the path embeddings.
 405   """
 406   # Get an index for each distinct path
 407   path_index = collections.defaultdict(itertools.count(0).next)
 408   path_vectors = {}
 409 
 410   for instance in instances:
 411     curr_path_embeddings, curr_path_strings = session.run(
 412         [model.path_embeddings, model.path_strings],
 413         feed_dict={model.instance: instance})
 414 
 415     for i, path in enumerate(curr_path_strings):
 416       if not path:
 417         continue
 418 
 419       # Set a new/existing index for the path
 420       index = path_index[path]
 421 
 422       # Save its vector
 423       path_vectors[index] = curr_path_embeddings[i, :]
 424 
 425   print('Number of distinct paths: %d' % len(path_index))
 426   return path_index, path_vectors

 429 def save_path_embeddings(model, path_vectors, path_index, embeddings_base_path):
 430   """Saves the path embeddings.
 431 
 432   Args:
 433     model: The trained path-based model.
 434     path_vectors: The path embeddings.
 435     path_index: A map from path to ID.
 436     embeddings_base_path: The base directory where the embeddings are.
 437   """
 438   index_range = range(max(path_index.values()) + 1)
 439   path_matrix = [path_vectors[i] for i in index_range]
 440   path_matrix = np.vstack(path_matrix)
 441 
 442   # Save the path embeddings
 443   path_vector_filename = os.path.join(
 444       embeddings_base_path, '%d_path_vectors' % model.lstm_output_dim)
 445   with open(path_vector_filename, 'w') as f_out:
 446     np.save(f_out, path_matrix)
 447 
 448   index_to_path = {i: p for p, i in path_index.iteritems()}
 449   path_vocab = [index_to_path[i] for i in index_range]
 450 
 451   # Save the path vocabulary
 452   path_vocab_filename = os.path.join(
 453       embeddings_base_path, '%d_path_vocab' % model.lstm_output_dim)
 454   with open(path_vocab_filename, 'w') as f_out:
 455     f_out.write('\n'.join(path_vocab))
 456     f_out.write('\n')
 457 
 458   print('Saved path embeddings.')

 461 def load_path_embeddings(path_embeddings_dir, path_dim):
 462   """Loads pretrained path embeddings from a binary file and returns the matrix.
 463 
 464   Args:
 465     path_embeddings_dir: The directory for the path embeddings.
 466     path_dim: The dimension of the path embeddings, used as prefix to the
 467     path_vocab and path_vectors files.
 468 
 469   Returns:
 470     The path embeddings matrix and the path_to_index dictionary.
 471   """
 472   prefix = path_embeddings_dir + '/%d' % path_dim + '_'
 473   with open(prefix + 'path_vocab') as f_in:
 474     vocab = f_in.read().splitlines()
 475 
 476   vocab_size = len(vocab)
 477   embedding_file = prefix + 'path_vectors'
 478 
 479   print('Embedding file "%s" has %d paths' % (embedding_file, vocab_size))
 480 
 481   with open(embedding_file) as f_in:
 482     embeddings = np.load(f_in)
 483 
 484   path_to_index = {p: i for i, p in enumerate(vocab)}
 485   return embeddings, path_to_index

 488 def get_indicative_paths(model, session, path_index, path_vectors, classes,
 489                          save_dir, k=20, threshold=0.8):
 490   """Gets the most indicative paths for each class.
 491 
 492   Args:
 493     model: The trained path-based model.
 494     session: The current TensorFlow session.
 495     path_index: A map from path to ID.
 496     path_vectors: The path embeddings.
 497     classes: The class label names.
 498     save_dir: Where to save the paths.
 499     k: The k for top-k paths.
 500     threshold: The threshold above which to consider paths as indicative.
 501   """
 502   # Define graph variables for this operation
 503   p_path_embedding = tf.placeholder(dtype=tf.float32,
 504                                     shape=[1, model.lstm_output_dim])
 505   p_distributions = tf.nn.softmax(tf.matmul(p_path_embedding, model.weights1))
 506 
 507   # Treat each path as a pair instance with a single path, and get the
 508   # relation distribution for it. Then, take the top paths for each relation.
 509 
 510   # This dictionary contains a relation as a key, and the value is a list of
 511   # tuples of path index and score. A relation r will contain (p, s) if the
 512   # path p is classified to r with a confidence of s.
 513   prediction_per_relation = collections.defaultdict(list)
 514 
 515   index_to_path = {i: p for p, i in path_index.iteritems()}
 516 
 517   # Predict all the paths
 518   for index in range(len(path_index)):
 519     curr_path_vector = path_vectors[index]
 520 
 521     distribution = session.run(p_distributions,
 522                                feed_dict={
 523                                    p_path_embedding: np.reshape(
 524                                        curr_path_vector,
 525                                        [1, model.lstm_output_dim])})
 526 
 527     distribution = distribution[0, :]
 528     prediction = np.argmax(distribution)
 529     prediction_per_relation[prediction].append(
 530         (index, distribution[prediction]))
 531 
 532     if index % 10000 == 0:
 533       print('Classified %d/%d (%3.2f%%) of the paths' % (
 534           index, len(path_index), 100 * index / len(path_index)))
 535 
 536   # Retrieve k-best scoring paths for each relation
 537   for relation_index, relation in enumerate(classes):
 538     curr_paths = sorted(prediction_per_relation[relation_index],
 539                         key=lambda item: item[1], reverse=True)
 540     above_t = [(p, s) for (p, s) in curr_paths if s >= threshold]
 541     top_k = curr_paths[k+1]
 542     relation_paths = above_t if len(above_t) > len(top_k) else top_k
 543 
 544     paths_filename = os.path.join(save_dir, '%s.paths' % relation)
 545     with open(paths_filename, 'w') as f_out:
 546       for index, score in relation_paths:
 547         print('\t'.join([index_to_path[index], str(score)]), file=f_out)
