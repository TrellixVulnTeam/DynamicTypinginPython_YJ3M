  41 def _int64_features(value):
  42   return tf.train.Feature(int64_list=tf.train.Int64List(value=value))

  45 def _bytes_features(value):
  46   value = [v.encode('utf-8') if isinstance(v, unicode) else v for v in value]
  47   return tf.train.Feature(bytes_list=tf.train.BytesList(value=value))

  50 class CreateExampleFn(object):
  51 
  52   def __init__(self):
  53     # Read the vocabulary.  N.B. that 0 = PAD, 1 = UNK, 2 = <X>, 3 = <Y>, hence
  54     # the enumeration starting at 4.
  55     with tf.gfile.GFile(FLAGS.vocab) as fh:
  56       self.vocab = {w: ix for ix, w in enumerate(fh.read().splitlines(), start=4)}
  57 
  58     self.vocab.update({'<PAD>': 0, '<UNK>': 1, '<X>': 2, '<Y>': 3})
  59 
  60     # Read the relations.
  61     with tf.gfile.GFile(FLAGS.relations) as fh:
  62       self.relations = {r: ix for ix, r in enumerate(fh.read().splitlines())}
  63 
  64     # Some hackery to map from SpaCy postags to Google's.
  65     lexnet_common.POSTAG_TO_ID['PROPN'] = lexnet_common.POSTAG_TO_ID['NOUN']
  66     lexnet_common.POSTAG_TO_ID['PRON'] = lexnet_common.POSTAG_TO_ID['NOUN']
  67     lexnet_common.POSTAG_TO_ID['CCONJ'] = lexnet_common.POSTAG_TO_ID['CONJ']
  68     #lexnet_common.DEPLABEL_TO_ID['relcl'] = lexnet_common.DEPLABEL_TO_ID['rel']
  69     #lexnet_common.DEPLABEL_TO_ID['compound'] = lexnet_common.DEPLABEL_TO_ID['xcomp']
  70     #lexnet_common.DEPLABEL_TO_ID['oprd'] = lexnet_common.DEPLABEL_TO_ID['UNK']
  71 
  72   def __call__(self, mod, head, rel, raw_paths):
  73     # Drop any really long paths.
  74     paths = []
  75     counts = []
  76     for raw, count in raw_paths.most_common(FLAGS.max_paths):
  77       path = raw.split('::')
  78       if len(path) <= FLAGS.max_pathlen:
  79         paths.append(path)
  80         counts.append(count)
  81 
  82     if not paths:
  83       return None
  84 
  85     # Compute the true length.
  86     pathlens = [len(path) for path in paths]
  87 
  88     # Pad each path out to max_pathlen so the LSTM can eat it.
  89     paths = (
  90         itertools.islice(
  91             itertools.chain(path, itertools.repeat('<PAD>/PAD/PAD/_')),
  92             FLAGS.max_pathlen)
  93         for path in paths)
  94 
  95     # Split the lemma, POS, dependency label, and direction each into a
  96     # separate feature.
  97     lemmas, postags, deplabels, dirs = zip(
  98         *(part.split('/') for part in itertools.chain(*paths)))
  99 
 100     lemmas = [self.vocab.get(lemma, 1) for lemma in lemmas]
 101     postags = [lexnet_common.POSTAG_TO_ID[pos] for pos in postags]
 102     deplabels = [lexnet_common.DEPLABEL_TO_ID.get(dep, 1) for dep in deplabels]
 103     dirs = [lexnet_common.DIR_TO_ID.get(d, 0) for d in dirs]
 104 
 105     return tf.train.Example(features=tf.train.Features(feature={
 106         'pair': _bytes_features(['::'.join((mod, head))]),
 107         'rel': _bytes_features([rel]),
 108         'rel_id': _int64_features([self.relations[rel]]),
 109         'reprs': _bytes_features(raw_paths),
 110         'pathlens': _int64_features(pathlens),
 111         'counts': _int64_features(counts),
 112         'lemmas': _int64_features(lemmas),
 113         'dirs': _int64_features(dirs),
 114         'deplabels': _int64_features(deplabels),
 115         'postags': _int64_features(postags),
 116         'x_embedding_id': _int64_features([self.vocab[mod]]),
 117         'y_embedding_id': _int64_features([self.vocab[head]]),
 118     }))

 121 def main(_):
 122   # Read the splits file, if there is one.
 123   assignments = {}
 124   if FLAGS.splits:
 125     with tf.gfile.GFile(FLAGS.splits) as fh:
 126       parts = (line.split('\t') for line in fh.read().splitlines())
 127       assignments = {(mod, head): split for mod, head, split in parts}
 128 
 129   splits = set(assignments.itervalues())
 130   if FLAGS.default_split:
 131     default_split = FLAGS.default_split
 132     splits.add(FLAGS.default_split)
 133   elif splits:
 134     default_split = iter(splits).next()
 135   else:
 136     print('Please specify --splits, --default_split, or both', file=sys.stderr)
 137     return 1
 138 
 139   last_mod, last_head, last_label = None, None, None
 140   raw_paths = collections.Counter()
 141 
 142   # Keep track of pairs we've seen to ensure that we don't get unsorted data.
 143   seen_labeled_pairs = set()
 144 
 145   # Set up output compression
 146   compression_type = getattr(
 147       tf.python_io.TFRecordCompressionType, FLAGS.compression)
 148   options = tf.python_io.TFRecordOptions(compression_type=compression_type)
 149 
 150   writers = {
 151       split: tf.python_io.TFRecordWriter(
 152           os.path.join(FLAGS.output_dir, '%s.tfrecs.gz' % split),
 153           options=options)
 154       for split in splits}
 155 
 156   create_example = CreateExampleFn()
 157 
 158   in_fh = sys.stdin if not FLAGS.input else tf.gfile.GFile(FLAGS.input)
 159   for lineno, line in enumerate(in_fh, start=1):
 160     if lineno % 100 == 0:
 161       print('\rProcessed %d lines...' % lineno, end='', file=sys.stderr)
 162 
 163     parts = line.decode('utf-8').strip().split('\t')
 164     if len(parts) != 5:
 165       print('Skipping line %d: %d columns (expected 5)' % (
 166           lineno, len(parts)), file=sys.stderr)
 167 
 168       continue
 169 
 170     mod, head, label, raw_path, source = parts
 171     if mod == last_mod and head == last_head and label == last_label:
 172       raw_paths.update([raw_path])
 173       continue
 174 
 175     if last_mod and last_head and last_label and raw_paths:
 176       if (last_mod, last_head, last_label) in seen_labeled_pairs:
 177         print('It looks like the input data is not sorted; ignoring extra '
 178               'record for (%s::%s, %s) at line %d' % (
 179                   last_mod, last_head, last_label, lineno))
 180       else:
 181         ex = create_example(last_mod, last_head, last_label, raw_paths)
 182         if ex:
 183           split = assignments.get((last_mod, last_head), default_split)
 184           writers[split].write(ex.SerializeToString())
 185 
 186         seen_labeled_pairs.add((last_mod, last_head, last_label))
 187 
 188     last_mod, last_head, last_label = mod, head, label
 189     raw_paths = collections.Counter()
 190 
 191   if last_mod and last_head and last_label and raw_paths:
 192     ex = create_example(last_mod, last_head, last_label, raw_paths)
 193     if ex:
 194       split = assignments.get((last_mod, last_head), default_split)
 195       writers[split].write(ex.SerializeToString())
 196 
 197   for writer in writers.itervalues():
 198     writer.close()
 199 
 200 
 201 if __name__ == '__main__':
 202   tf.app.run()
