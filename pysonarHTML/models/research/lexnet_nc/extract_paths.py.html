  32 def get_path(mod_token, head_token):
  33   """Returns the path between a modifier token and a head token."""
  34   # Compute the path from the root to each token.
  35   mod_ancestors = list(reversed(list(mod_token.ancestors)))
  36   head_ancestors = list(reversed(list(head_token.ancestors)))
  37 
  38   # If the paths don't start at the same place (odd!) then there is no path at
  39   # all.
  40   if (not mod_ancestors or not head_ancestors
  41       or mod_ancestors[0] != head_ancestors[0]):
  42     return None
  43 
  44   # Eject elements from the common path until we reach the first differing
  45   # ancestor.
  46   ix = 1
  47   while (ix < len(mod_ancestors) and ix < len(head_ancestors)
  48          and mod_ancestors[ix] == head_ancestors[ix]):
  49     ix += 1
  50 
  51   # Construct the path.  TODO: add "satellites", possibly honor sentence
  52   # ordering between modifier and head rather than just always traversing from
  53   # the modifier to the head?
  54   path = ['/'.join(('<X>', mod_token.pos_, mod_token.dep_, '>'))]
  55 
  56   path += ['/'.join((tok.lemma_, tok.pos_, tok.dep_, '>'))
  57            for tok in reversed(mod_ancestors[ix:])]
  58 
  59   root_token = mod_ancestors[ix - 1]
  60   path += ['/'.join((root_token.lemma_, root_token.pos_, root_token.dep_, '^'))]
  61 
  62   path += ['/'.join((tok.lemma_, tok.pos_, tok.dep_, '<'))
  63            for tok in head_ancestors[ix:]]
  64 
  65   path += ['/'.join(('<Y>', head_token.pos_, head_token.dep_, '<'))]
  66 
  67   return '::'.join(path)

  70 def main(_):
  71   nlp = spacy.load('en_core_web_sm')
  72 
  73   # Grab the set of labeled pairs for which we wish to collect paths.
  74   with tf.gfile.GFile(FLAGS.labeled_pairs) as fh:
  75     parts = (l.decode('utf-8').split('\t') for l in fh.read().splitlines())
  76     labeled_pairs = {(mod, head): rel for mod, head, rel in parts}
  77 
  78   # Create a mapping from each head to the modifiers that are used with it.
  79   mods_for_head = {
  80       head: set(hm[1] for hm in head_mods)
  81       for head, head_mods in itertools.groupby(
  82           sorted((head, mod) for (mod, head) in labeled_pairs.iterkeys()),
  83           lambda (head, mod): head)}
  84 
  85   # Collect all the heads that we know about.
  86   heads = set(mods_for_head.keys())
  87 
  88   # For each sentence that contains a (head, modifier) pair that's in our set,
  89   # emit the dependency path that connects the pair.
  90   out_fh = sys.stdout if not FLAGS.output else tf.gfile.GFile(FLAGS.output, 'w')
  91   in_fh = sys.stdin if not FLAGS.corpus else tf.gfile.GFile(FLAGS.corpus)
  92 
  93   num_paths = 0
  94   for line, sen in enumerate(in_fh, start=1):
  95     if line % 100 == 0:
  96       print('\rProcessing line %d: %d paths' % (line, num_paths),
  97             end='', file=sys.stderr)
  98 
  99     sen = sen.decode('utf-8').strip()
 100     doc = nlp(sen)
 101 
 102     for head_token in doc:
 103       head_text = head_token.text.lower()
 104       if head_text in heads:
 105         mods = mods_for_head[head_text]
 106         for mod_token in doc:
 107           mod_text = mod_token.text.lower()
 108           if mod_text in mods:
 109             path = get_path(mod_token, head_token)
 110             if path:
 111               label = labeled_pairs[(mod_text, head_text)]
 112               line = '\t'.join((mod_text, head_text, label, path, sen))
 113               print(line.encode('utf-8'), file=out_fh)
 114               num_paths += 1
 115 
 116   out_fh.close()
 117 
 118 if __name__ == '__main__':
 119   tf.app.run()
