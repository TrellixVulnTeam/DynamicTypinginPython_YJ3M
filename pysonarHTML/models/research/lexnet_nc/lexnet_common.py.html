  58 def load_word_embeddings(embedding_filename):
  59   """Loads pretrained word embeddings from a binary file and returns the matrix.
  60 
  61   Adds the <PAD>, <UNK>, <X>, and <Y> tokens to the beginning of the vocab.
  62 
  63   Args:
  64     embedding_filename: filename of the binary NPY data
  65 
  66   Returns:
  67     The word embeddings matrix
  68   """
  69   embeddings = np.load(embedding_filename)
  70   dim = embeddings.shape[1]
  71 
  72   # Four initially random vectors for the special tokens: <PAD>, <UNK>, <X>, <Y>
  73   special_embeddings = np.random.normal(0, 0.1, (4, dim))
  74   embeddings = np.vstack((special_embeddings, embeddings))
  75   embeddings = embeddings.astype(np.float32)
  76 
  77   return embeddings

  80 def full_evaluation(model, session, instances, labels, set_name, classes):
  81   """Prints a full evaluation on the current set.
  82 
  83   Performance (recall, precision and F1), classification report (per
  84   class performance), and confusion matrix).
  85 
  86   Args:
  87     model: The currently trained path-based model.
  88     session: The current TensorFlow session.
  89     instances: The current set instances.
  90     labels: The current set labels.
  91     set_name: The current set name (train/validation/test).
  92     classes: The class label names.
  93 
  94   Returns:
  95     The model's prediction for the given instances.
  96   """
  97 
  98   # Predict the labels
  99   pred = model.predict(session, instances)
 100 
 101   # Print the performance
 102   precision, recall, f1, _ = metrics.precision_recall_fscore_support(
 103       labels, pred, average='weighted')
 104 
 105   print('%s set: Precision: %.3f, Recall: %.3f, F1: %.3f' % (
 106       set_name, precision, recall, f1))
 107 
 108   # Print a classification report
 109   print('%s classification report:' % set_name)
 110   print(metrics.classification_report(labels, pred, target_names=classes))
 111 
 112   # Print the confusion matrix
 113   print('%s confusion matrix:' % set_name)
 114   cm = metrics.confusion_matrix(labels, pred, labels=range(len(classes)))
 115   cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis] * 100
 116   print_cm(cm, labels=classes)
 117   return pred

 120 def print_cm(cm, labels):
 121   """Pretty print for confusion matrices.
 122 
 123   From: https://gist.github.com/zachguo/10296432.
 124 
 125   Args:
 126     cm: The confusion matrix.
 127     labels: The class names.
 128   """
 129   columnwidth = 10
 130   empty_cell = ' ' * columnwidth
 131   short_labels = [label[:12].rjust(10, ' ') for label in labels]
 132 
 133   # Print header
 134   header = empty_cell + ' '
 135   header += ''.join([' %{0}s '.format(columnwidth) % label
 136                      for label in short_labels])
 137 
 138   print(header)
 139 
 140   # Print rows
 141   for i, label1 in enumerate(short_labels):
 142     row = '%{0}s '.format(columnwidth) % label1[:10]
 143     for j in range(len(short_labels)):
 144       value = int(cm[i, j]) if not np.isnan(cm[i, j]) else 0
 145       cell = ' %{0}d '.format(10) % value
 146       row += cell + ' '
 147     print(row)

 150 def load_all_labels(records):
 151   """Reads TensorFlow examples from a RecordReader and returns only the labels.
 152 
 153   Args:
 154     records: a record list with TensorFlow examples.
 155 
 156   Returns:
 157     The labels
 158   """
 159   curr_features = tf.parse_example(records, {
 160       'rel_id': tf.FixedLenFeature([1], dtype=tf.int64),
 161   })
 162 
 163   labels = tf.squeeze(curr_features['rel_id'], [-1])
 164   return labels

 167 def load_all_pairs(records):
 168   """Reads TensorFlow examples from a RecordReader and returns the word pairs.
 169 
 170   Args:
 171     records: a record list with TensorFlow examples.
 172 
 173   Returns:
 174     The word pairs
 175   """
 176   curr_features = tf.parse_example(records, {
 177       'pair': tf.FixedLenFeature([1], dtype=tf.string)
 178   })
 179 
 180   word_pairs = curr_features['pair']
 181   return word_pairs

 184 def write_predictions(pairs, labels, predictions, classes, predictions_file):
 185   """Write the predictions to a file.
 186 
 187   Args:
 188     pairs: the word pairs (list of tuple of two strings).
 189     labels: the gold-standard labels for these pairs (array of rel ID).
 190     predictions: the predicted labels for these pairs (array of rel ID).
 191     classes: a list of relation names.
 192     predictions_file: where to save the predictions.
 193   """
 194   with open(predictions_file, 'w') as f_out:
 195     for pair, label, pred in zip(pairs, labels, predictions):
 196       w1, w2 = pair
 197       f_out.write('\t'.join([w1, w2, classes[label], classes[pred]]) + '\n')
