  60 def main(_):
  61   # Pick up any one-off hyper-parameters.
  62   hparams = lexnet_model.LexNETModel.default_hparams()
  63   hparams.corpus = FLAGS.corpus
  64   hparams.input = FLAGS.input
  65   hparams.path_embeddings_file = 'path_embeddings/%s/%s' % (
  66       FLAGS.dataset, FLAGS.corpus)
  67 
  68   input_dir = hparams.input if hparams.input != 'path' else 'path_classifier'
  69 
  70   # Set the number of classes
  71   classes_filename = os.path.join(
  72       FLAGS.dataset_dir, FLAGS.dataset, 'classes.txt')
  73   with open(classes_filename) as f_in:
  74     classes = f_in.read().splitlines()
  75 
  76   hparams.num_classes = len(classes)
  77   print('Model will predict into %d classes' % hparams.num_classes)
  78 
  79   # Get the datasets
  80   train_set, val_set, test_set = (
  81       os.path.join(
  82           FLAGS.dataset_dir, FLAGS.dataset, FLAGS.corpus,
  83           filename + '.tfrecs.gz')
  84       for filename in ['train', 'val', 'test'])
  85 
  86   print('Running with hyper-parameters: {}'.format(hparams))
  87 
  88   # Load the instances
  89   print('Loading instances...')
  90   opts = tf.python_io.TFRecordOptions(
  91       compression_type=tf.python_io.TFRecordCompressionType.GZIP)
  92   train_instances = list(tf.python_io.tf_record_iterator(train_set, opts))
  93   val_instances = list(tf.python_io.tf_record_iterator(val_set, opts))
  94   test_instances = list(tf.python_io.tf_record_iterator(test_set, opts))
  95 
  96   # Load the word embeddings
  97   print('Loading word embeddings...')
  98   relata_embeddings, path_embeddings, nc_embeddings, path_to_index = (
  99       None, None, None, None)
 100   if hparams.input in ['dist', 'dist-nc', 'integrated', 'integrated-nc']:
 101     relata_embeddings = lexnet_common.load_word_embeddings(
 102         FLAGS.embeddings_base_path, hparams.relata_embeddings_file)
 103 
 104   if hparams.input in ['path', 'integrated', 'integrated-nc']:
 105     path_embeddings, path_to_index = path_model.load_path_embeddings(
 106         os.path.join(FLAGS.embeddings_base_path, hparams.path_embeddings_file),
 107         hparams.path_dim)
 108 
 109   if hparams.input in ['dist-nc', 'integrated-nc']:
 110     nc_embeddings = lexnet_common.load_word_embeddings(
 111         FLAGS.embeddings_base_path, hparams.nc_embeddings_file)
 112 
 113   # Define the graph and the model
 114   with tf.Graph().as_default():
 115     model = lexnet_model.LexNETModel(
 116         hparams, relata_embeddings, path_embeddings,
 117         nc_embeddings, path_to_index)
 118 
 119     # Initialize a session and start training
 120     session = tf.Session()
 121     session.run(tf.global_variables_initializer())
 122 
 123     # Initalize the path mapping
 124     if hparams.input in ['path', 'integrated', 'integrated-nc']:
 125       session.run(tf.tables_initializer())
 126       session.run(model.initialize_path_op, {
 127           model.path_initial_value_t: path_embeddings
 128       })
 129 
 130     # Initialize the NC embeddings
 131     if hparams.input in ['dist-nc', 'integrated-nc']:
 132       session.run(model.initialize_nc_op, {
 133           model.nc_initial_value_t: nc_embeddings
 134       })
 135 
 136     # Load the labels
 137     print('Loading labels...')
 138     train_labels = model.load_labels(session, train_instances)
 139     val_labels = model.load_labels(session, val_instances)
 140     test_labels = model.load_labels(session, test_instances)
 141 
 142     save_path = '{logdir}/results/{dataset}/{input}/{corpus}'.format(
 143         logdir=FLAGS.logdir, dataset=FLAGS.dataset,
 144         corpus=model.hparams.corpus, input=input_dir)
 145 
 146     if not os.path.exists(save_path):
 147       os.makedirs(save_path)
 148 
 149     # Train the model
 150     print('Training the model...')
 151     model.fit(session, train_instances, epoch_completed,
 152               val_instances, val_labels, save_path)
 153 
 154     # Print the best performance on the validation set
 155     print('Best performance on the validation set: F1=%.3f' %
 156           epoch_completed.best_f1)
 157 
 158     # Evaluate on the train and validation sets
 159     lexnet_common.full_evaluation(model, session, train_instances, train_labels,
 160                                   'Train', classes)
 161     lexnet_common.full_evaluation(model, session, val_instances, val_labels,
 162                                   'Validation', classes)
 163     test_predictions = lexnet_common.full_evaluation(
 164         model, session, test_instances, test_labels, 'Test', classes)
 165 
 166     # Write the test predictions to a file
 167     predictions_file = os.path.join(save_path, 'test_predictions.tsv')
 168     print('Saving test predictions to %s' % save_path)
 169     test_pairs = model.load_pairs(session, test_instances)
 170     lexnet_common.write_predictions(test_pairs, test_labels, test_predictions,
 171                                     classes, predictions_file)

 174 def epoch_completed(model, session, epoch, epoch_loss,
 175                     val_instances, val_labels, save_path):
 176   """Runs every time an epoch completes.
 177 
 178   Print the performance on the validation set, and update the saved model if
 179   its performance is better on the previous ones. If the performance dropped,
 180   tell the training to stop.
 181 
 182   Args:
 183     model: The currently trained path-based model.
 184     session: The current TensorFlow session.
 185     epoch: The epoch number.
 186     epoch_loss: The current epoch loss.
 187     val_instances: The validation set instances (evaluation between epochs).
 188     val_labels: The validation set labels (for evaluation between epochs).
 189     save_path: Where to save the model.
 190 
 191   Returns:
 192     whether the training should stop.
 193   """
 194   stop_training = False
 195 
 196   # Evaluate on the validation set
 197   val_pred = model.predict(session, val_instances)
 198   precision, recall, f1, _ = metrics.precision_recall_fscore_support(
 199       val_labels, val_pred, average='weighted')
 200   print(
 201       'Epoch: %d/%d, Loss: %f, validation set: P: %.3f, R: %.3f, F1: %.3f\n' % (
 202           epoch + 1, model.hparams.num_epochs, epoch_loss,
 203           precision, recall, f1))
 204 
 205   # If the F1 is much smaller than the previous one, stop training. Else, if
 206   # it's bigger, save the model.
 207   if f1 < epoch_completed.best_f1 - 0.08:
 208     stop_training = True
 209 
 210   if f1 > epoch_completed.best_f1:
 211     saver = tf.train.Saver()
 212     checkpoint_filename = os.path.join(save_path, 'best.ckpt')
 213     print('Saving model in: %s' % checkpoint_filename)
 214     saver.save(session, checkpoint_filename)
 215     print('Model saved in file: %s' % checkpoint_filename)
 216     epoch_completed.best_f1 = f1
 217 
 218   return stop_training
 219 
 220 epoch_completed.best_f1 = 0
 221 
 222 if __name__ == '__main__':
 223   tf.app.run(main)
