  31   def default_hparams(cls):
  32     """Returns the default hyper-parameters."""
  33     return tf.contrib.training.HParams(
  34         batch_size=10,
  35         num_classes=37,
  36         num_epochs=30,
  37         input_keep_prob=0.9,
  38         input='integrated',  # dist/ dist-nc/ path/ integrated/ integrated-nc
  39         learn_relata=False,
  40         corpus='wiki_gigawords',
  41         random_seed=133,  # zero means no random seed
  42         relata_embeddings_file='glove/glove.6B.300d.bin',
  43         nc_embeddings_file='nc_glove/vecs.6B.300d.bin',
  44         path_embeddings_file='path_embeddings/tratz/fine_grained/wiki',
  45         hidden_layers=1,
  46         path_dim=60)

  48   def __init__(self, hparams, relata_embeddings, path_embeddings, nc_embeddings,
  49                path_to_index):
  50     """Initialize the LexNET classifier.
  51 
  52     Args:
  53       hparams: the hyper-parameters.
  54       relata_embeddings: word embeddings for the distributional component.
  55       path_embeddings: embeddings for the paths.
  56       nc_embeddings: noun compound embeddings.
  57       path_to_index: a mapping from string path to an index in the path
  58       embeddings matrix.
  59     """
  60     self.hparams = hparams
  61 
  62     self.path_embeddings = path_embeddings
  63     self.relata_embeddings = relata_embeddings
  64     self.nc_embeddings = nc_embeddings
  65 
  66     self.vocab_size, self.relata_dim = 0, 0
  67     self.path_to_index = None
  68     self.path_dim = 0
  69 
  70     # Set the random seed
  71     if hparams.random_seed > 0:
  72       tf.set_random_seed(hparams.random_seed)
  73 
  74     # Get the vocabulary size and relata dim
  75     if self.hparams.input in ['dist', 'dist-nc', 'integrated', 'integrated-nc']:
  76       self.vocab_size, self.relata_dim = self.relata_embeddings.shape
  77 
  78     # Create the mapping from string path to an index in the embeddings matrix
  79     if self.hparams.input in ['path', 'integrated', 'integrated-nc']:
  80       self.path_to_index = tf.contrib.lookup.HashTable(
  81           tf.contrib.lookup.KeyValueTensorInitializer(
  82               tf.constant(path_to_index.keys()),
  83               tf.constant(path_to_index.values()),
  84               key_dtype=tf.string, value_dtype=tf.int32), 0)
  85 
  86       self.path_dim = self.path_embeddings.shape[1]
  87 
  88     # Create the network
  89     self.__create_computation_graph__()
  90 
  91   def __create_computation_graph__(self):
  92     """Initialize the model and define the graph."""
  93     network_input = 0
  94 
  95     # Define the network inputs
  96     # Distributional x and y
  97     if self.hparams.input in ['dist', 'dist-nc', 'integrated', 'integrated-nc']:
  98       network_input += 2 * self.relata_dim
  99       self.relata_lookup = tf.get_variable(
 100           'relata_lookup',
 101           initializer=self.relata_embeddings,
 102           dtype=tf.float32,
 103           trainable=self.hparams.learn_relata)
 104 
 105     # Path-based
 106     if self.hparams.input in ['path', 'integrated', 'integrated-nc']:
 107       network_input += self.path_dim
 108 
 109       self.path_initial_value_t = tf.placeholder(tf.float32, None)
 110 
 111       self.path_lookup = tf.get_variable(
 112           name='path_lookup',
 113           dtype=tf.float32,
 114           trainable=False,
 115           shape=self.path_embeddings.shape)
 116 
 117       self.initialize_path_op = tf.assign(
 118           self.path_lookup, self.path_initial_value_t, validate_shape=False)
 119 
 120     # Distributional noun compound
 121     if self.hparams.input in ['dist-nc', 'integrated-nc']:
 122       network_input += self.relata_dim
 123 
 124       self.nc_initial_value_t = tf.placeholder(tf.float32, None)
 125 
 126       self.nc_lookup = tf.get_variable(
 127           name='nc_lookup',
 128           dtype=tf.float32,
 129           trainable=False,
 130           shape=self.nc_embeddings.shape)
 131 
 132       self.initialize_nc_op = tf.assign(
 133           self.nc_lookup, self.nc_initial_value_t, validate_shape=False)
 134 
 135     hidden_dim = network_input // 2
 136 
 137     # Define the MLP
 138     if self.hparams.hidden_layers == 0:
 139       self.weights1 = tf.get_variable(
 140           'W1',
 141           shape=[network_input, self.hparams.num_classes],
 142           dtype=tf.float32)
 143       self.bias1 = tf.get_variable(
 144           'b1',
 145           shape=[self.hparams.num_classes],
 146           dtype=tf.float32)
 147 
 148     elif self.hparams.hidden_layers == 1:
 149 
 150       self.weights1 = tf.get_variable(
 151           'W1',
 152           shape=[network_input, hidden_dim],
 153           dtype=tf.float32)
 154       self.bias1 = tf.get_variable(
 155           'b1',
 156           shape=[hidden_dim],
 157           dtype=tf.float32)
 158 
 159       self.weights2 = tf.get_variable(
 160           'W2',
 161           shape=[hidden_dim, self.hparams.num_classes],
 162           dtype=tf.float32)
 163       self.bias2 = tf.get_variable(
 164           'b2',
 165           shape=[self.hparams.num_classes],
 166           dtype=tf.float32)
 167 
 168     else:
 169       raise ValueError('Only 0 or 1 hidden layers are supported')
 170 
 171     # Define the variables
 172     self.instances = tf.placeholder(dtype=tf.string,
 173                                     shape=[self.hparams.batch_size])
 174 
 175     (self.x_embedding_id,
 176      self.y_embedding_id,
 177      self.nc_embedding_id,
 178      self.path_embedding_id,
 179      self.path_counts,
 180      self.labels) = parse_tensorflow_examples(
 181          self.instances, self.hparams.batch_size, self.path_to_index)
 182 
 183     # Create the MLP
 184     self.__mlp__()
 185 
 186     self.instances_to_load = tf.placeholder(dtype=tf.string, shape=[None])
 187     self.labels_to_load = lexnet_common.load_all_labels(self.instances_to_load)
 188     self.pairs_to_load = lexnet_common.load_all_pairs(self.instances_to_load)
 189 
 190   def load_labels(self, session, instances):
 191     """Loads the labels for these instances.
 192 
 193     Args:
 194       session: The current TensorFlow session,
 195       instances: The instances for which to load the labels.
 196 
 197     Returns:
 198       the labels of these instances.
 199     """
 200     return session.run(self.labels_to_load,
 201                        feed_dict={self.instances_to_load: instances})
 202 
 203   def load_pairs(self, session, instances):
 204     """Loads the word pairs for these instances.
 205 
 206     Args:
 207       session: The current TensorFlow session,
 208       instances: The instances for which to load the labels.
 209 
 210     Returns:
 211       the word pairs of these instances.
 212     """
 213     word_pairs = session.run(self.pairs_to_load,
 214                              feed_dict={self.instances_to_load: instances})
 215     return [pair[0].split('::') for pair in word_pairs]

 217   def __train_single_batch__(self, session, batch_instances):
 218     """Train a single batch.
 219 
 220     Args:
 221       session: The current TensorFlow session.
 222       batch_instances: TensorFlow examples containing the training intances
 223 
 224     Returns:
 225       The cost for the current batch.
 226     """
 227     cost, _ = session.run([self.cost, self.train_op],
 228                           feed_dict={self.instances: batch_instances})
 229 
 230     return cost

 232   def fit(self, session, inputs, on_epoch_completed, val_instances, val_labels,
 233           save_path):
 234     """Train the model.
 235 
 236     Args:
 237       session: The current TensorFlow session.
 238       inputs:
 239       on_epoch_completed: A method to call after each epoch.
 240       val_instances: The validation set instances (evaluation between epochs).
 241       val_labels: The validation set labels (for evaluation between epochs).
 242       save_path: Where to save the model.
 243     """
 244     for epoch in range(self.hparams.num_epochs):
 245 
 246       losses = []
 247       epoch_indices = list(np.random.permutation(len(inputs)))
 248 
 249       # If the number of instances doesn't divide by batch_size, enlarge it
 250       # by duplicating training examples
 251       mod = len(epoch_indices) % self.hparams.batch_size
 252       if mod > 0:
 253         epoch_indices.extend([np.random.randint(0, high=len(inputs))] * mod)
 254 
 255       # Define the batches
 256       n_batches = len(epoch_indices) // self.hparams.batch_size
 257 
 258       for minibatch in range(n_batches):
 259 
 260         batch_indices = epoch_indices[minibatch * self.hparams.batch_size:(
 261             minibatch + 1) * self.hparams.batch_size]
 262         batch_instances = [inputs[i] for i in batch_indices]
 263 
 264         loss = self.__train_single_batch__(session, batch_instances)
 265         losses.append(loss)
 266 
 267       epoch_loss = np.nanmean(losses)
 268 
 269       if on_epoch_completed:
 270         should_stop = on_epoch_completed(self, session, epoch, epoch_loss,
 271                                          val_instances, val_labels, save_path)
 272         if should_stop:
 273           print('Stopping training after %d epochs.' % epoch)
 274           return

 276   def predict(self, session, inputs):
 277     """Predict the classification of the test set.
 278 
 279     Args:
 280       session: The current TensorFlow session.
 281       inputs: the train paths, x, y and/or nc vectors
 282 
 283     Returns:
 284       The test predictions.
 285     """
 286     predictions, _ = zip(*self.predict_with_score(session, inputs))
 287     return np.array(predictions)

 289   def predict_with_score(self, session, inputs):
 290     """Predict the classification of the test set.
 291 
 292     Args:
 293       session: The current TensorFlow session.
 294       inputs: the test paths, x, y and/or nc vectors
 295 
 296     Returns:
 297       The test predictions along with their scores.
 298     """
 299     test_pred = [0] * len(inputs)
 300 
 301     for chunk in xrange(0, len(test_pred), self.hparams.batch_size):
 302 
 303       # Initialize the variables with the current batch data
 304       batch_indices = list(
 305           range(chunk, min(chunk + self.hparams.batch_size, len(test_pred))))
 306 
 307       # If the batch is too small, add a few other examples
 308       if len(batch_indices) < self.hparams.batch_size:
 309         batch_indices += [0] * (self.hparams.batch_size-len(batch_indices))
 310 
 311       batch_instances = [inputs[i] for i in batch_indices]
 312 
 313       predictions, scores = session.run(
 314           [self.predictions, self.scores],
 315           feed_dict={self.instances: batch_instances})
 316 
 317       for index_in_batch, index_in_dataset in enumerate(batch_indices):
 318         prediction = predictions[index_in_batch]
 319         score = scores[index_in_batch][prediction]
 320         test_pred[index_in_dataset] = (prediction, score)
 321 
 322     return test_pred

 324   def __mlp__(self):
 325     """Performs the MLP operations.
 326 
 327     Returns: the prediction object to be computed in a Session
 328     """
 329     # Define the operations
 330 
 331     # Network input
 332     vec_inputs = []
 333 
 334     # Distributional component
 335     if self.hparams.input in ['dist', 'dist-nc', 'integrated', 'integrated-nc']:
 336       for emb_id in [self.x_embedding_id, self.y_embedding_id]:
 337         vec_inputs.append(tf.nn.embedding_lookup(self.relata_lookup, emb_id))
 338 
 339     # Noun compound component
 340     if self.hparams.input in ['dist-nc', 'integrated-nc']:
 341       vec = tf.nn.embedding_lookup(self.nc_lookup, self.nc_embedding_id)
 342       vec_inputs.append(vec)
 343 
 344     # Path-based component
 345     if self.hparams.input in ['path', 'integrated', 'integrated-nc']:
 346 
 347       # Get the current paths for each batch instance
 348       self.path_embeddings = tf.nn.embedding_lookup(self.path_lookup,
 349                                                     self.path_embedding_id)
 350 
 351       # self.path_embeddings is of shape
 352       # [batch_size, max_path_per_instance, output_dim]
 353       # We need to multiply it by path counts
 354       # ([batch_size, max_path_per_instance]).
 355       # Start by duplicating path_counts along the output_dim axis.
 356       self.path_freq = tf.tile(tf.expand_dims(self.path_counts, -1),
 357                                [1, 1, self.path_dim])
 358 
 359       # Compute the averaged path vector for each instance.
 360       # First, multiply the path embeddings and frequencies element-wise.
 361       self.weighted = tf.multiply(self.path_freq, self.path_embeddings)
 362 
 363       # Second, take the sum to get a tensor of shape [batch_size, output_dim].
 364       self.pair_path_embeddings = tf.reduce_sum(self.weighted, 1)
 365 
 366       # Finally, divide by the total number of paths.
 367       # The number of paths for each pair has a shape [batch_size, 1],
 368       # We duplicate it output_dim times along the second axis.
 369       self.num_paths = tf.clip_by_value(
 370           tf.reduce_sum(self.path_counts, 1), 1, np.inf)
 371       self.num_paths = tf.tile(tf.expand_dims(self.num_paths, -1),
 372                                [1, self.path_dim])
 373 
 374       # And finally, divide pair_path_embeddings by num_paths element-wise.
 375       self.pair_path_embeddings = tf.div(
 376           self.pair_path_embeddings, self.num_paths)
 377       vec_inputs.append(self.pair_path_embeddings)
 378 
 379     # Concatenate the inputs and feed to the MLP
 380     self.input_vec = tf.nn.dropout(
 381         tf.concat(vec_inputs, 1),
 382         keep_prob=self.hparams.input_keep_prob)
 383 
 384     h = tf.matmul(self.input_vec, self.weights1)
 385     self.output = h
 386 
 387     if self.hparams.hidden_layers == 1:
 388       self.output = tf.matmul(tf.nn.tanh(h), self.weights2)
 389 
 390     self.scores = self.output
 391     self.predictions = tf.argmax(self.scores, axis=1)
 392 
 393     # Define the loss function and the optimization algorithm
 394     self.cross_entropies = tf.nn.sparse_softmax_cross_entropy_with_logits(
 395         logits=self.scores, labels=self.labels)
 396     self.cost = tf.reduce_sum(self.cross_entropies, name='cost')
 397     self.global_step = tf.Variable(0, name='global_step', trainable=False)
 398     self.optimizer = tf.train.AdamOptimizer()
 399     self.train_op = self.optimizer.minimize(
 400         self.cost, global_step=self.global_step)

 403 def parse_tensorflow_examples(record, batch_size, path_to_index):
 404   """Reads TensorFlow examples from a RecordReader.
 405 
 406   Args:
 407     record: a record with TensorFlow examples.
 408     batch_size: the number of instances in a minibatch
 409     path_to_index: mapping from string path to index in the embeddings matrix.
 410 
 411   Returns:
 412     The word embeddings IDs, paths and counts
 413   """
 414   features = tf.parse_example(
 415       record, {
 416           'x_embedding_id': tf.FixedLenFeature([1], dtype=tf.int64),
 417           'y_embedding_id': tf.FixedLenFeature([1], dtype=tf.int64),
 418           'nc_embedding_id': tf.FixedLenFeature([1], dtype=tf.int64),
 419           'reprs': tf.FixedLenSequenceFeature(
 420               shape=(), dtype=tf.string, allow_missing=True),
 421           'counts': tf.FixedLenSequenceFeature(
 422               shape=(), dtype=tf.int64, allow_missing=True),
 423           'rel_id': tf.FixedLenFeature([1], dtype=tf.int64)
 424       })
 425 
 426   x_embedding_id = tf.squeeze(features['x_embedding_id'], [-1])
 427   y_embedding_id = tf.squeeze(features['y_embedding_id'], [-1])
 428   nc_embedding_id = tf.squeeze(features['nc_embedding_id'], [-1])
 429   labels = tf.squeeze(features['rel_id'], [-1])
 430   path_counts = tf.to_float(tf.reshape(features['counts'], [batch_size, -1]))
 431 
 432   path_embedding_id = None
 433   if path_to_index:
 434     path_embedding_id = path_to_index.lookup(features['reprs'])
 435 
 436   return (
 437       x_embedding_id, y_embedding_id, nc_embedding_id,
 438       path_embedding_id, path_counts, labels)
