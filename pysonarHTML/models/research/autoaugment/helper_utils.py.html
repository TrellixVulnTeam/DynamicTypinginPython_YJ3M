  26 def setup_loss(logits, labels):
  27   """Returns the cross entropy for the given `logits` and `labels`."""
  28   predictions = tf.nn.softmax(logits)
  29   cost = tf.losses.softmax_cross_entropy(onehot_labels=labels,
  30                                          logits=logits)
  31   return predictions, cost

  34 def decay_weights(cost, weight_decay_rate):
  35   """Calculates the loss for l2 weight decay and adds it to `cost`."""
  36   costs = []
  37   for var in tf.trainable_variables():
  38     costs.append(tf.nn.l2_loss(var))
  39   cost += tf.multiply(weight_decay_rate, tf.add_n(costs))
  40   return cost

  43 def eval_child_model(session, model, data_loader, mode):
  44   """Evaluates `model` on held out data depending on `mode`.
  45 
  46   Args:
  47     session: TensorFlow session the model will be run with.
  48     model: TensorFlow model that will be evaluated.
  49     data_loader: DataSet object that contains data that `model` will
  50       evaluate.
  51     mode: Will `model` either evaluate validation or test data.
  52 
  53   Returns:
  54     Accuracy of `model` when evaluated on the specified dataset.
  55 
  56   Raises:
  57     ValueError: if invalid dataset `mode` is specified.
  58   """
  59   if mode == 'val':
  60     images = data_loader.val_images
  61     labels = data_loader.val_labels
  62   elif mode == 'test':
  63     images = data_loader.test_images
  64     labels = data_loader.test_labels
  65   else:
  66     raise ValueError('Not valid eval mode')
  67   assert len(images) == len(labels)
  68   tf.logging.info('model.batch_size is {}'.format(model.batch_size))
  69   assert len(images) % model.batch_size == 0
  70   eval_batches = int(len(images) / model.batch_size)
  71   for i in range(eval_batches):
  72     eval_images = images[i * model.batch_size:(i + 1) * model.batch_size]
  73     eval_labels = labels[i * model.batch_size:(i + 1) * model.batch_size]
  74     _ = session.run(
  75         model.eval_op,
  76         feed_dict={
  77             model.images: eval_images,
  78             model.labels: eval_labels,
  79         })
  80   return session.run(model.accuracy)

  83 def cosine_lr(learning_rate, epoch, iteration, batches_per_epoch, total_epochs):
  84   """Cosine Learning rate.
  85 
  86   Args:
  87     learning_rate: Initial learning rate.
  88     epoch: Current epoch we are one. This is one based.
  89     iteration: Current batch in this epoch.
  90     batches_per_epoch: Batches per epoch.
  91     total_epochs: Total epochs you are training for.
  92 
  93   Returns:
  94     The learning rate to be used for this current batch.
  95   """
  96   t_total = total_epochs * batches_per_epoch
  97   t_cur = float(epoch * batches_per_epoch + iteration)
  98   return 0.5 * learning_rate * (1 + np.cos(np.pi * t_cur / t_total))

 101 def get_lr(curr_epoch, hparams, iteration=None):
 102   """Returns the learning rate during training based on the current epoch."""
 103   assert iteration is not None
 104   batches_per_epoch = int(hparams.train_size / hparams.batch_size)
 105   lr = cosine_lr(hparams.lr, curr_epoch, iteration, batches_per_epoch,
 106                  hparams.num_epochs)
 107   return lr

 110 def run_epoch_training(session, model, data_loader, curr_epoch):
 111   """Runs one epoch of training for the model passed in.
 112 
 113   Args:
 114     session: TensorFlow session the model will be run with.
 115     model: TensorFlow model that will be evaluated.
 116     data_loader: DataSet object that contains data that `model` will
 117       evaluate.
 118     curr_epoch: How many of epochs of training have been done so far.
 119 
 120   Returns:
 121     The accuracy of 'model' on the training set
 122   """
 123   steps_per_epoch = int(model.hparams.train_size / model.hparams.batch_size)
 124   tf.logging.info('steps per epoch: {}'.format(steps_per_epoch))
 125   curr_step = session.run(model.global_step)
 126   assert curr_step % steps_per_epoch == 0
 127 
 128   # Get the current learning rate for the model based on the current epoch
 129   curr_lr = get_lr(curr_epoch, model.hparams, iteration=0)
 130   tf.logging.info('lr of {} for epoch {}'.format(curr_lr, curr_epoch))
 131 
 132   for step in xrange(steps_per_epoch):
 133     curr_lr = get_lr(curr_epoch, model.hparams, iteration=(step + 1))
 134     # Update the lr rate variable to the current LR.
 135     model.lr_rate_ph.load(curr_lr, session=session)
 136     if step % 20 == 0:
 137       tf.logging.info('Training {}/{}'.format(step, steps_per_epoch))
 138 
 139     train_images, train_labels = data_loader.next_batch()
 140     _, step, _ = session.run(
 141         [model.train_op, model.global_step, model.eval_op],
 142         feed_dict={
 143             model.images: train_images,
 144             model.labels: train_labels,
 145         })
 146 
 147   train_accuracy = session.run(model.accuracy)
 148   tf.logging.info('Train accuracy: {}'.format(train_accuracy))
 149   return train_accuracy
