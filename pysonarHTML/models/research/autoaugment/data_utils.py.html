  34 class DataSet(object):
  35   """Dataset object that produces augmented training and eval data."""
  36 
  37   def __init__(self, hparams):
  38     self.hparams = hparams
  39     self.epochs = 0
  40     self.curr_train_index = 0
  41 
  42     all_labels = []
  43 
  44     self.good_policies = found_policies.good_policies()
  45 
  46     # Determine how many databatched to load
  47     num_data_batches_to_load = 5
  48     total_batches_to_load = num_data_batches_to_load
  49     train_batches_to_load = total_batches_to_load
  50     assert hparams.train_size + hparams.validation_size <= 50000
  51     if hparams.eval_test:
  52       total_batches_to_load += 1
  53     # Determine how many images we have loaded
  54     total_dataset_size = 10000 * num_data_batches_to_load
  55     train_dataset_size = total_dataset_size
  56     if hparams.eval_test:
  57       total_dataset_size += 10000
  58 
  59     if hparams.dataset == 'cifar10':
  60       all_data = np.empty((total_batches_to_load, 10000, 3072), dtype=np.uint8)
  61     elif hparams.dataset == 'cifar100':
  62       assert num_data_batches_to_load == 5
  63       all_data = np.empty((1, 50000, 3072), dtype=np.uint8)
  64       if hparams.eval_test:
  65         test_data = np.empty((1, 10000, 3072), dtype=np.uint8)
  66     if hparams.dataset == 'cifar10':
  67       tf.logging.info('Cifar10')
  68       datafiles = [
  69           'data_batch_1', 'data_batch_2', 'data_batch_3', 'data_batch_4',
  70           'data_batch_5']
  71 
  72       datafiles = datafiles[:train_batches_to_load]
  73       if hparams.eval_test:
  74         datafiles.append('test_batch')
  75       num_classes = 10
  76     elif hparams.dataset == 'cifar100':
  77       datafiles = ['train']
  78       if hparams.eval_test:
  79         datafiles.append('test')
  80       num_classes = 100
  81     else:
  82       raise NotImplementedError('Unimplemented dataset: ', hparams.dataset)
  83     if hparams.dataset != 'test':
  84       for file_num, f in enumerate(datafiles):
  85         d = unpickle(os.path.join(hparams.data_path, f))
  86         if f == 'test':
  87           test_data[0] = copy.deepcopy(d['data'])
  88           all_data = np.concatenate([all_data, test_data], axis=1)
  89         else:
  90           all_data[file_num] = copy.deepcopy(d['data'])
  91         if hparams.dataset == 'cifar10':
  92           labels = np.array(d['labels'])
  93         else:
  94           labels = np.array(d['fine_labels'])
  95         nsamples = len(labels)
  96         for idx in range(nsamples):
  97           all_labels.append(labels[idx])
  98 
  99     all_data = all_data.reshape(total_dataset_size, 3072)
 100     all_data = all_data.reshape(-1, 3, 32, 32)
 101     all_data = all_data.transpose(0, 2, 3, 1).copy()
 102     all_data = all_data / 255.0
 103     mean = augmentation_transforms.MEANS
 104     std = augmentation_transforms.STDS
 105     tf.logging.info('mean:{}    std: {}'.format(mean, std))
 106 
 107     all_data = (all_data - mean) / std
 108     all_labels = np.eye(num_classes)[np.array(all_labels, dtype=np.int32)]
 109     assert len(all_data) == len(all_labels)
 110     tf.logging.info(
 111         'In CIFAR10 loader, number of images: {}'.format(len(all_data)))
 112 
 113     # Break off test data
 114     if hparams.eval_test:
 115       self.test_images = all_data[train_dataset_size:]
 116       self.test_labels = all_labels[train_dataset_size:]
 117 
 118     # Shuffle the rest of the data
 119     all_data = all_data[:train_dataset_size]
 120     all_labels = all_labels[:train_dataset_size]
 121     np.random.seed(0)
 122     perm = np.arange(len(all_data))
 123     np.random.shuffle(perm)
 124     all_data = all_data[perm]
 125     all_labels = all_labels[perm]
 126 
 127     # Break into train and val
 128     train_size, val_size = hparams.train_size, hparams.validation_size
 129     assert 50000 >= train_size + val_size
 130     self.train_images = all_data[:train_size]
 131     self.train_labels = all_labels[:train_size]
 132     self.val_images = all_data[train_size:train_size + val_size]
 133     self.val_labels = all_labels[train_size:train_size + val_size]
 134     self.num_train = self.train_images.shape[0]
 135 
 136   def next_batch(self):
 137     """Return the next minibatch of augmented data."""
 138     next_train_index = self.curr_train_index + self.hparams.batch_size
 139     if next_train_index > self.num_train:
 140       # Increase epoch number
 141       epoch = self.epochs + 1
 142       self.reset()
 143       self.epochs = epoch
 144     batched_data = (
 145         self.train_images[self.curr_train_index:
 146                           self.curr_train_index + self.hparams.batch_size],
 147         self.train_labels[self.curr_train_index:
 148                           self.curr_train_index + self.hparams.batch_size])
 149     final_imgs = []
 150 
 151     images, labels = batched_data
 152     for data in images:
 153       epoch_policy = self.good_policies[np.random.choice(
 154           len(self.good_policies))]
 155       final_img = augmentation_transforms.apply_policy(
 156           epoch_policy, data)
 157       final_img = augmentation_transforms.random_flip(
 158           augmentation_transforms.zero_pad_and_crop(final_img, 4))
 159       # Apply cutout
 160       final_img = augmentation_transforms.cutout_numpy(final_img)
 161       final_imgs.append(final_img)
 162     batched_data = (np.array(final_imgs, np.float32), labels)
 163     self.curr_train_index += self.hparams.batch_size
 164     return batched_data
 165 
 166   def reset(self):
 167     """Reset training data and index into the training data."""
 168     self.epochs = 0
 169     # Shuffle the training data
 170     perm = np.arange(self.num_train)
 171     np.random.shuffle(perm)
 172     assert self.num_train == self.train_images.shape[
 173         0], 'Error incorrect shuffling mask'
 174     self.train_images = self.train_images[perm]
 175     self.train_labels = self.train_labels[perm]
 176     self.curr_train_index = 0

 179 def unpickle(f):
 180   tf.logging.info('loading file: {}'.format(f))
 181   fo = tf.gfile.Open(f, 'r')
 182   d = cPickle.load(fo)
 183   fo.close()
 184   return d
