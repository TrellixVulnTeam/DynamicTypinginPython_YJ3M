  36 def random_flip(x):
  37   """Flip the input x horizontally with 50% probability."""
  38   if np.random.rand(1)[0] > 0.5:
  39     return np.fliplr(x)
  40   return x

  43 def zero_pad_and_crop(img, amount=4):
  44   """Zero pad by `amount` zero pixels on each side then take a random crop.
  45 
  46   Args:
  47     img: numpy image that will be zero padded and cropped.
  48     amount: amount of zeros to pad `img` with horizontally and verically.
  49 
  50   Returns:
  51     The cropped zero padded img. The returned numpy array will be of the same
  52     shape as `img`.
  53   """
  54   padded_img = np.zeros((img.shape[0] + amount * 2, img.shape[1] + amount * 2,
  55                          img.shape[2]))
  56   padded_img[amount:img.shape[0] + amount, amount:
  57              img.shape[1] + amount, :] = img
  58   top = np.random.randint(low=0, high=2 * amount)
  59   left = np.random.randint(low=0, high=2 * amount)
  60   new_img = padded_img[top:top + img.shape[0], left:left + img.shape[1], :]
  61   return new_img

  64 def create_cutout_mask(img_height, img_width, num_channels, size):
  65   """Creates a zero mask used for cutout of shape `img_height` x `img_width`.
  66 
  67   Args:
  68     img_height: Height of image cutout mask will be applied to.
  69     img_width: Width of image cutout mask will be applied to.
  70     num_channels: Number of channels in the image.
  71     size: Size of the zeros mask.
  72 
  73   Returns:
  74     A mask of shape `img_height` x `img_width` with all ones except for a
  75     square of zeros of shape `size` x `size`. This mask is meant to be
  76     elementwise multiplied with the original image. Additionally returns
  77     the `upper_coord` and `lower_coord` which specify where the cutout mask
  78     will be applied.
  79   """
  80   assert img_height == img_width
  81 
  82   # Sample center where cutout mask will be applied
  83   height_loc = np.random.randint(low=0, high=img_height)
  84   width_loc = np.random.randint(low=0, high=img_width)
  85 
  86   # Determine upper right and lower left corners of patch
  87   upper_coord = (max(0, height_loc - size // 2), max(0, width_loc - size // 2))
  88   lower_coord = (min(img_height, height_loc + size // 2),
  89                  min(img_width, width_loc + size // 2))
  90   mask_height = lower_coord[0] - upper_coord[0]
  91   mask_width = lower_coord[1] - upper_coord[1]
  92   assert mask_height > 0
  93   assert mask_width > 0
  94 
  95   mask = np.ones((img_height, img_width, num_channels))
  96   zeros = np.zeros((mask_height, mask_width, num_channels))
  97   mask[upper_coord[0]:lower_coord[0], upper_coord[1]:lower_coord[1], :] = (
  98       zeros)
  99   return mask, upper_coord, lower_coord

 102 def cutout_numpy(img, size=16):
 103   """Apply cutout with mask of shape `size` x `size` to `img`.
 104 
 105   The cutout operation is from the paper https://arxiv.org/abs/1708.04552.
 106   This operation applies a `size`x`size` mask of zeros to a random location
 107   within `img`.
 108 
 109   Args:
 110     img: Numpy image that cutout will be applied to.
 111     size: Height/width of the cutout mask that will be
 112 
 113   Returns:
 114     A numpy tensor that is the result of applying the cutout mask to `img`.
 115   """
 116   img_height, img_width, num_channels = (img.shape[0], img.shape[1],
 117                                          img.shape[2])
 118   assert len(img.shape) == 3
 119   mask, _, _ = create_cutout_mask(img_height, img_width, num_channels, size)
 120   return img * mask

 123 def float_parameter(level, maxval):
 124   """Helper function to scale `val` between 0 and maxval .
 125 
 126   Args:
 127     level: Level of the operation that will be between [0, `PARAMETER_MAX`].
 128     maxval: Maximum value that the operation can have. This will be scaled
 129       to level/PARAMETER_MAX.
 130 
 131   Returns:
 132     A float that results from scaling `maxval` according to `level`.
 133   """
 134   return float(level) * maxval / PARAMETER_MAX

 137 def int_parameter(level, maxval):
 138   """Helper function to scale `val` between 0 and maxval .
 139 
 140   Args:
 141     level: Level of the operation that will be between [0, `PARAMETER_MAX`].
 142     maxval: Maximum value that the operation can have. This will be scaled
 143       to level/PARAMETER_MAX.
 144 
 145   Returns:
 146     An int that results from scaling `maxval` according to `level`.
 147   """
 148   return int(level * maxval / PARAMETER_MAX)

 151 def pil_wrap(img):
 152   """Convert the `img` numpy tensor to a PIL Image."""
 153   return Image.fromarray(
 154       np.uint8((img * STDS + MEANS) * 255.0)).convert('RGBA')
 155 
 156 
 157 def pil_unwrap(pil_img):
 158   """Converts the PIL img to a numpy array."""
 159   pic_array = (np.array(pil_img.getdata()).reshape((32, 32, 4)) / 255.0)
 160   i1, i2 = np.where(pic_array[:, :, 3] == 0)
 161   pic_array = (pic_array[:, :, :3] - MEANS) / STDS
 162   pic_array[i1, i2] = [0, 0, 0]
 163   return pic_array

 166 def apply_policy(policy, img):
 167   """Apply the `policy` to the numpy `img`.
 168 
 169   Args:
 170     policy: A list of tuples with the form (name, probability, level) where
 171       `name` is the name of the augmentation operation to apply, `probability`
 172       is the probability of applying the operation and `level` is what strength
 173       the operation to apply.
 174     img: Numpy image that will have `policy` applied to it.
 175 
 176   Returns:
 177     The result of applying `policy` to `img`.
 178   """
 179   pil_img = pil_wrap(img)
 180   for xform in policy:
 181     assert len(xform) == 3
 182     name, probability, level = xform
 183     xform_fn = NAME_TO_TRANSFORM[name].pil_transformer(probability, level)
 184     pil_img = xform_fn(pil_img)
 185   return pil_unwrap(pil_img)

 188 class TransformFunction(object):
 189   """Wraps the Transform function for pretty printing options."""
 190 
 191   def __init__(self, func, name):
 192     self.f = func
 193     self.name = name
 194 
 195   def __repr__(self):
 196     return '<' + self.name + '>'
 197 
 198   def __call__(self, pil_img):
 199     return self.f(pil_img)
 200 
 201 
 202 class TransformT(object):
 203   """Each instance of this class represents a specific transform."""
 204 
 205   def __init__(self, name, xform_fn):
 206     self.name = name
 207     self.xform = xform_fn
 208 
 209   def pil_transformer(self, probability, level):
 210 
 211     def return_function(im):
 212       if random.random() < probability:
 213         im = self.xform(im, level)
 214       return im
 215 
 216     name = self.name + '({:.1f},{})'.format(probability, level)
 217     return TransformFunction(return_function, name)
 218 
 219   def do_transform(self, image, level):
 220     f = self.pil_transformer(PARAMETER_MAX, level)
 221     return pil_unwrap(f(pil_wrap(image)))

 253 def _rotate_impl(pil_img, level):
 254   """Rotates `pil_img` from -30 to 30 degrees depending on `level`."""
 255   degrees = int_parameter(level, 30)
 256   if random.random() > 0.5:
 257     degrees = -degrees
 258   return pil_img.rotate(degrees)

 273 def _shear_x_impl(pil_img, level):
 274   """Applies PIL ShearX to `pil_img`.
 275 
 276   The ShearX operation shears the image along the horizontal axis with `level`
 277   magnitude.
 278 
 279   Args:
 280     pil_img: Image in PIL object.
 281     level: Strength of the operation specified as an Integer from
 282       [0, `PARAMETER_MAX`].
 283 
 284   Returns:
 285     A PIL Image that has had ShearX applied to it.
 286   """
 287   level = float_parameter(level, 0.3)
 288   if random.random() > 0.5:
 289     level = -level
 290   return pil_img.transform((32, 32), Image.AFFINE, (1, level, 0, 0, 1, 0))

 365 def _crop_impl(pil_img, level, interpolation=Image.BILINEAR):
 366   """Applies a crop to `pil_img` with the size depending on the `level`."""
 367   cropped = pil_img.crop((level, level, IMAGE_SIZE - level, IMAGE_SIZE - level))
 368   resized = cropped.resize((IMAGE_SIZE, IMAGE_SIZE), interpolation)
 369   return resized

 396 def _cutout_pil_impl(pil_img, level):
 397   """Apply cutout to pil_img at the specified level."""
 398   size = int_parameter(level, 20)
 399   if size <= 0:
 400     return pil_img
 401   img_height, img_width, num_channels = (32, 32, 3)
 402   _, upper_coord, lower_coord = (
 403       create_cutout_mask(img_height, img_width, num_channels, size))
 404   pixels = pil_img.load()  # create the pixel map
 405   for i in range(upper_coord[0], lower_coord[0]):  # for every col:
 406     for j in range(upper_coord[1], lower_coord[1]):  # For every row
 407       pixels[i, j] = (125, 122, 113, 0)  # set the colour accordingly
 408   return pil_img
 409

 413 def _enhancer_impl(enhancer):
 414   """Sets level to be between 0.1 and 1.8 for ImageEnhance transforms of PIL."""
 415   def impl(pil_img, level):
 416     v = float_parameter(level, 1.8) + .1  # going to 0 just destroys it
 417     return enhancer(pil_img).enhance(v)
 418   return impl
