  33 def variable(name, shape, dtype, initializer, trainable):
  34   """Returns a TF variable with the passed in specifications."""
  35   var = tf.get_variable(
  36       name,
  37       shape=shape,
  38       dtype=dtype,
  39       initializer=initializer,
  40       trainable=trainable)
  41   return var

  44 def global_avg_pool(x, scope=None):
  45   """Average pools away spatial height and width dimension of 4D tensor."""
  46   assert x.get_shape().ndims == 4
  47   with tf.name_scope(scope, 'global_avg_pool', [x]):
  48     kernel_size = (1, int(x.shape[1]), int(x.shape[2]), 1)
  49     squeeze_dims = (1, 2)
  50     result = tf.nn.avg_pool(
  51         x,
  52         ksize=kernel_size,
  53         strides=(1, 1, 1, 1),
  54         padding='VALID',
  55         data_format='NHWC')
  56     return tf.squeeze(result, squeeze_dims)

  59 def zero_pad(inputs, in_filter, out_filter):
  60   """Zero pads `input` tensor to have `out_filter` number of filters."""
  61   outputs = tf.pad(inputs, [[0, 0], [0, 0], [0, 0],
  62                             [(out_filter - in_filter) // 2,
  63                              (out_filter - in_filter) // 2]])
  64   return outputs

  99 @tf.contrib.framework.add_arg_scope
 100 def conv2d(inputs,
 101            num_filters_out,
 102            kernel_size,
 103            stride=1,
 104            scope=None,
 105            reuse=None):
 106   """Adds a 2D convolution.
 107 
 108   conv2d creates a variable called 'weights', representing the convolutional
 109   kernel, that is convolved with the input.
 110 
 111   Args:
 112     inputs: a 4D tensor in NHWC format.
 113     num_filters_out: the number of output filters.
 114     kernel_size: an int specifying the kernel height and width size.
 115     stride: an int specifying the height and width stride.
 116     scope: Optional scope for variable_scope.
 117     reuse: whether or not the layer and its variables should be reused.
 118   Returns:
 119     a tensor that is the result of a convolution being applied to `inputs`.
 120   """
 121   with tf.variable_scope(scope, 'Conv', [inputs], reuse=reuse):
 122     num_filters_in = int(inputs.shape[3])
 123     weights_shape = [kernel_size, kernel_size, num_filters_in, num_filters_out]
 124 
 125     # Initialization
 126     n = int(weights_shape[0] * weights_shape[1] * weights_shape[3])
 127     weights_initializer = tf.random_normal_initializer(
 128         stddev=np.sqrt(2.0 / n))
 129 
 130     weights = variable(
 131         name='weights',
 132         shape=weights_shape,
 133         dtype=tf.float32,
 134         initializer=weights_initializer,
 135         trainable=True)
 136     strides = stride_arr(stride, stride)
 137     outputs = tf.nn.conv2d(
 138         inputs, weights, strides, padding='SAME', data_format='NHWC')
 139     return outputs

 142 @tf.contrib.framework.add_arg_scope
 143 def fc(inputs,
 144        num_units_out,
 145        scope=None,
 146        reuse=None):
 147   """Creates a fully connected layer applied to `inputs`.
 148 
 149   Args:
 150     inputs: a tensor that the fully connected layer will be applied to. It
 151       will be reshaped if it is not 2D.
 152     num_units_out: the number of output units in the layer.
 153     scope: Optional scope for variable_scope.
 154     reuse: whether or not the layer and its variables should be reused.
 155 
 156   Returns:
 157      a tensor that is the result of applying a linear matrix to `inputs`.
 158   """
 159   if len(inputs.shape) > 2:
 160     inputs = tf.reshape(inputs, [int(inputs.shape[0]), -1])
 161 
 162   with tf.variable_scope(scope, 'FC', [inputs], reuse=reuse):
 163     num_units_in = inputs.shape[1]
 164     weights_shape = [num_units_in, num_units_out]
 165     unif_init_range = 1.0 / (num_units_out)**(0.5)
 166     weights_initializer = tf.random_uniform_initializer(
 167         -unif_init_range, unif_init_range)
 168     weights = variable(
 169         name='weights',
 170         shape=weights_shape,
 171         dtype=tf.float32,
 172         initializer=weights_initializer,
 173         trainable=True)
 174     bias_initializer = tf.constant_initializer(0.0)
 175     biases = variable(
 176         name='biases',
 177         shape=[num_units_out,],
 178         dtype=tf.float32,
 179         initializer=bias_initializer,
 180         trainable=True)
 181     outputs = tf.nn.xw_plus_b(inputs, weights, biases)
 182     return outputs

 185 @tf.contrib.framework.add_arg_scope
 186 def avg_pool(inputs, kernel_size, stride=2, padding='VALID', scope=None):
 187   """Wrapper around tf.nn.avg_pool."""
 188   with tf.name_scope(scope, 'AvgPool', [inputs]):
 189     kernel = stride_arr(kernel_size, kernel_size)
 190     strides = stride_arr(stride, stride)
 191     return tf.nn.avg_pool(
 192         inputs,
 193         ksize=kernel,
 194         strides=strides,
 195         padding=padding,
 196         data_format='NHWC')
