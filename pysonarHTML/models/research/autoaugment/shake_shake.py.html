  26 def _shake_shake_skip_connection(x, output_filters, stride):
  27   """Adds a residual connection to the filter x for the shake-shake model."""
  28   curr_filters = int(x.shape[3])
  29   if curr_filters == output_filters:
  30     return x
  31   stride_spec = ops.stride_arr(stride, stride)
  32   # Skip path 1
  33   path1 = tf.nn.avg_pool(
  34       x, [1, 1, 1, 1], stride_spec, 'VALID', data_format='NHWC')
  35   path1 = ops.conv2d(path1, int(output_filters / 2), 1, scope='path1_conv')
  36 
  37   # Skip path 2
  38   # First pad with 0's then crop
  39   pad_arr = [[0, 0], [0, 1], [0, 1], [0, 0]]
  40   path2 = tf.pad(x, pad_arr)[:, 1:, 1:, :]
  41   concat_axis = 3
  42 
  43   path2 = tf.nn.avg_pool(
  44       path2, [1, 1, 1, 1], stride_spec, 'VALID', data_format='NHWC')
  45   path2 = ops.conv2d(path2, int(output_filters / 2), 1, scope='path2_conv')
  46 
  47   # Concat and apply BN
  48   final_path = tf.concat(values=[path1, path2], axis=concat_axis)
  49   final_path = ops.batch_norm(final_path, scope='final_path_bn')
  50   return final_path

  53 def _shake_shake_branch(x, output_filters, stride, rand_forward, rand_backward,
  54                         is_training):
  55   """Building a 2 branching convnet."""
  56   x = tf.nn.relu(x)
  57   x = ops.conv2d(x, output_filters, 3, stride=stride, scope='conv1')
  58   x = ops.batch_norm(x, scope='bn1')
  59   x = tf.nn.relu(x)
  60   x = ops.conv2d(x, output_filters, 3, scope='conv2')
  61   x = ops.batch_norm(x, scope='bn2')
  62   if is_training:
  63     x = x * rand_backward + tf.stop_gradient(x * rand_forward -
  64                                              x * rand_backward)
  65   else:
  66     x *= 1.0 / 2
  67   return x

  70 def _shake_shake_block(x, output_filters, stride, is_training):
  71   """Builds a full shake-shake sub layer."""
  72   batch_size = tf.shape(x)[0]
  73 
  74   # Generate random numbers for scaling the branches
  75   rand_forward = [
  76       tf.random_uniform(
  77           [batch_size, 1, 1, 1], minval=0, maxval=1, dtype=tf.float32)
  78       for _ in range(2)
  79   ]
  80   rand_backward = [
  81       tf.random_uniform(
  82           [batch_size, 1, 1, 1], minval=0, maxval=1, dtype=tf.float32)
  83       for _ in range(2)
  84   ]
  85   # Normalize so that all sum to 1
  86   total_forward = tf.add_n(rand_forward)
  87   total_backward = tf.add_n(rand_backward)
  88   rand_forward = [samp / total_forward for samp in rand_forward]
  89   rand_backward = [samp / total_backward for samp in rand_backward]
  90   zipped_rand = zip(rand_forward, rand_backward)
  91 
  92   branches = []
  93   for branch, (r_forward, r_backward) in enumerate(zipped_rand):
  94     with tf.variable_scope('branch_{}'.format(branch)):
  95       b = _shake_shake_branch(x, output_filters, stride, r_forward, r_backward,
  96                               is_training)
  97       branches.append(b)
  98   res = _shake_shake_skip_connection(x, output_filters, stride)
  99   return res + tf.add_n(branches)

 102 def _shake_shake_layer(x, output_filters, num_blocks, stride,
 103                        is_training):
 104   """Builds many sub layers into one full layer."""
 105   for block_num in range(num_blocks):
 106     curr_stride = stride if (block_num == 0) else 1
 107     with tf.variable_scope('layer_{}'.format(block_num)):
 108       x = _shake_shake_block(x, output_filters, curr_stride,
 109                              is_training)
 110   return x

 113 def build_shake_shake_model(images, num_classes, hparams, is_training):
 114   """Builds the Shake-Shake model.
 115 
 116   Build the Shake-Shake model from https://arxiv.org/abs/1705.07485.
 117 
 118   Args:
 119     images: Tensor of images that will be fed into the Wide ResNet Model.
 120     num_classes: Number of classed that the model needs to predict.
 121     hparams: tf.HParams object that contains additional hparams needed to
 122       construct the model. In this case it is the `shake_shake_widen_factor`
 123       that is used to determine how many filters the model has.
 124     is_training: Is the model training or not.
 125 
 126   Returns:
 127     The logits of the Shake-Shake model.
 128   """
 129   depth = 26
 130   k = hparams.shake_shake_widen_factor  # The widen factor
 131   n = int((depth - 2) / 6)
 132   x = images
 133 
 134   x = ops.conv2d(x, 16, 3, scope='init_conv')
 135   x = ops.batch_norm(x, scope='init_bn')
 136   with tf.variable_scope('L1'):
 137     x = _shake_shake_layer(x, 16 * k, n, 1, is_training)
 138   with tf.variable_scope('L2'):
 139     x = _shake_shake_layer(x, 32 * k, n, 2, is_training)
 140   with tf.variable_scope('L3'):
 141     x = _shake_shake_layer(x, 64 * k, n, 2, is_training)
 142   x = tf.nn.relu(x)
 143   x = ops.global_avg_pool(x)
 144 
 145   # Fully connected
 146   logits = ops.fc(x, num_classes)
 147   return logits
