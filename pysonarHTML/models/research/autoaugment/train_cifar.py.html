  51 def setup_arg_scopes(is_training):
  52   """Sets up the argscopes that will be used when building an image model.
  53 
  54   Args:
  55     is_training: Is the model training or not.
  56 
  57   Returns:
  58     Arg scopes to be put around the model being constructed.
  59   """
  60 
  61   batch_norm_decay = 0.9
  62   batch_norm_epsilon = 1e-5
  63   batch_norm_params = {
  64       # Decay for the moving averages.
  65       'decay': batch_norm_decay,
  66       # epsilon to prevent 0s in variance.
  67       'epsilon': batch_norm_epsilon,
  68       'scale': True,
  69       # collection containing the moving mean and moving variance.
  70       'is_training': is_training,
  71   }
  72 
  73   scopes = []
  74 
  75   scopes.append(arg_scope([ops.batch_norm], **batch_norm_params))
  76   return scopes

  79 def build_model(inputs, num_classes, is_training, hparams):
  80   """Constructs the vision model being trained/evaled.
  81 
  82   Args:
  83     inputs: input features/images being fed to the image model build built.
  84     num_classes: number of output classes being predicted.
  85     is_training: is the model training or not.
  86     hparams: additional hyperparameters associated with the image model.
  87 
  88   Returns:
  89     The logits of the image model.
  90   """
  91   scopes = setup_arg_scopes(is_training)
  92   with contextlib.nested(*scopes):
  93     if hparams.model_name == 'pyramid_net':
  94       logits = build_shake_drop_model(
  95           inputs, num_classes, is_training)
  96     elif hparams.model_name == 'wrn':
  97       logits = build_wrn_model(
  98           inputs, num_classes, hparams.wrn_size)
  99     elif hparams.model_name == 'shake_shake':
 100       logits = build_shake_shake_model(
 101           inputs, num_classes, hparams, is_training)
 102   return logits

 105 class CifarModel(object):
 106   """Builds an image model for Cifar10/Cifar100."""
 107 
 108   def __init__(self, hparams):
 109     self.hparams = hparams
 110 
 111   def build(self, mode):
 112     """Construct the cifar model."""
 113     assert mode in ['train', 'eval']
 114     self.mode = mode
 115     self._setup_misc(mode)
 116     self._setup_images_and_labels()
 117     self._build_graph(self.images, self.labels, mode)
 118 
 119     self.init = tf.group(tf.global_variables_initializer(),
 120                          tf.local_variables_initializer())
 121 
 122   def _setup_misc(self, mode):
 123     """Sets up miscellaneous in the cifar model constructor."""
 124     self.lr_rate_ph = tf.Variable(0.0, name='lrn_rate', trainable=False)
 125     self.reuse = None if (mode == 'train') else True
 126     self.batch_size = self.hparams.batch_size
 127     if mode == 'eval':
 128       self.batch_size = 25
 129 
 130   def _setup_images_and_labels(self):
 131     """Sets up image and label placeholders for the cifar model."""
 132     if FLAGS.dataset == 'cifar10':
 133       self.num_classes = 10
 134     else:
 135       self.num_classes = 100
 136     self.images = tf.placeholder(tf.float32, [self.batch_size, 32, 32, 3])
 137     self.labels = tf.placeholder(tf.float32,
 138                                  [self.batch_size, self.num_classes])
 139 
 140   def assign_epoch(self, session, epoch_value):
 141     session.run(self._epoch_update, feed_dict={self._new_epoch: epoch_value})
 142 
 143   def _build_graph(self, images, labels, mode):
 144     """Constructs the TF graph for the cifar model.
 145 
 146     Args:
 147       images: A 4-D image Tensor
 148       labels: A 2-D labels Tensor.
 149       mode: string indicating training mode ( e.g., 'train', 'valid', 'test').
 150     """
 151     is_training = 'train' in mode
 152     if is_training:
 153       self.global_step = tf.train.get_or_create_global_step()
 154 
 155     logits = build_model(
 156         images,
 157         self.num_classes,
 158         is_training,
 159         self.hparams)
 160     self.predictions, self.cost = helper_utils.setup_loss(
 161         logits, labels)
 162     self.accuracy, self.eval_op = tf.metrics.accuracy(
 163         tf.argmax(labels, 1), tf.argmax(self.predictions, 1))
 164     self._calc_num_trainable_params()
 165 
 166     # Adds L2 weight decay to the cost
 167     self.cost = helper_utils.decay_weights(self.cost,
 168                                            self.hparams.weight_decay_rate)
 169 
 170     if is_training:
 171       self._build_train_op()
 172 
 173     # Setup checkpointing for this child model
 174     # Keep 2 or more checkpoints around during training.
 175     with tf.device('/cpu:0'):
 176       self.saver = tf.train.Saver(max_to_keep=2)
 177 
 178     self.init = tf.group(tf.global_variables_initializer(),
 179                          tf.local_variables_initializer())
 180 
 181   def _calc_num_trainable_params(self):
 182     self.num_trainable_params = np.sum([
 183         np.prod(var.get_shape().as_list()) for var in tf.trainable_variables()
 184     ])
 185     tf.logging.info('number of trainable params: {}'.format(
 186         self.num_trainable_params))
 187 
 188   def _build_train_op(self):
 189     """Builds the train op for the cifar model."""
 190     hparams = self.hparams
 191     tvars = tf.trainable_variables()
 192     grads = tf.gradients(self.cost, tvars)
 193     if hparams.gradient_clipping_by_global_norm > 0.0:
 194       grads, norm = tf.clip_by_global_norm(
 195           grads, hparams.gradient_clipping_by_global_norm)
 196       tf.summary.scalar('grad_norm', norm)
 197 
 198     # Setup the initial learning rate
 199     initial_lr = self.lr_rate_ph
 200     optimizer = tf.train.MomentumOptimizer(
 201         initial_lr,
 202         0.9,
 203         use_nesterov=True)
 204 
 205     self.optimizer = optimizer
 206     apply_op = optimizer.apply_gradients(
 207         zip(grads, tvars), global_step=self.global_step, name='train_step')
 208     train_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS)
 209     with tf.control_dependencies([apply_op]):
 210       self.train_op = tf.group(*train_ops)

 213 class CifarModelTrainer(object):
 214   """Trains an instance of the CifarModel class."""
 215 
 216   def __init__(self, hparams):
 217     self._session = None
 218     self.hparams = hparams
 219 
 220     self.model_dir = os.path.join(FLAGS.checkpoint_dir, 'model')
 221     self.log_dir = os.path.join(FLAGS.checkpoint_dir, 'log')
 222     # Set the random seed to be sure the same validation set
 223     # is used for each model
 224     np.random.seed(0)
 225     self.data_loader = data_utils.DataSet(hparams)
 226     np.random.seed()  # Put the random seed back to random
 227     self.data_loader.reset()
 228 
 229   def save_model(self, step=None):
 230     """Dumps model into the backup_dir.
 231 
 232     Args:
 233       step: If provided, creates a checkpoint with the given step
 234         number, instead of overwriting the existing checkpoints.
 235     """
 236     model_save_name = os.path.join(self.model_dir, 'model.ckpt')
 237     if not tf.gfile.IsDirectory(self.model_dir):
 238       tf.gfile.MakeDirs(self.model_dir)
 239     self.saver.save(self.session, model_save_name, global_step=step)
 240     tf.logging.info('Saved child model')
 241 
 242   def extract_model_spec(self):
 243     """Loads a checkpoint with the architecture structure stored in the name."""
 244     checkpoint_path = tf.train.latest_checkpoint(self.model_dir)
 245     if checkpoint_path is not None:
 246       self.saver.restore(self.session, checkpoint_path)
 247       tf.logging.info('Loaded child model checkpoint from %s',
 248                       checkpoint_path)
 249     else:
 250       self.save_model(step=0)
 251 
 252   def eval_child_model(self, model, data_loader, mode):
 253     """Evaluate the child model.
 254 
 255     Args:
 256       model: image model that will be evaluated.
 257       data_loader: dataset object to extract eval data from.
 258       mode: will the model be evalled on train, val or test.
 259 
 260     Returns:
 261       Accuracy of the model on the specified dataset.
 262     """
 263     tf.logging.info('Evaluating child model in mode %s', mode)
 264     while True:
 265       try:
 266         with self._new_session(model):
 267           accuracy = helper_utils.eval_child_model(
 268               self.session,
 269               model,
 270               data_loader,
 271               mode)
 272           tf.logging.info('Eval child model accuracy: {}'.format(accuracy))
 273           # If epoch trained without raising the below errors, break
 274           # from loop.
 275           break
 276       except (tf.errors.AbortedError, tf.errors.UnavailableError) as e:
 277         tf.logging.info('Retryable error caught: %s.  Retrying.', e)
 278 
 279     return accuracy
 280 
 281   @contextlib.contextmanager
 282   def _new_session(self, m):
 283     """Creates a new session for model m."""
 284     # Create a new session for this model, initialize
 285     # variables, and save / restore from
 286     # checkpoint.
 287     self._session = tf.Session(
 288         '',
 289         config=tf.ConfigProto(
 290             allow_soft_placement=True, log_device_placement=False))
 291     self.session.run(m.init)
 292 
 293     # Load in a previous checkpoint, or save this one
 294     self.extract_model_spec()
 295     try:
 296       yield
 297     finally:
 298       tf.Session.reset('')
 299       self._session = None
 300 
 301   def _build_models(self):
 302     """Builds the image models for train and eval."""
 303     # Determine if we should build the train and eval model. When using
 304     # distributed training we only want to build one or the other and not both.
 305     with tf.variable_scope('model', use_resource=False):
 306       m = CifarModel(self.hparams)
 307       m.build('train')
 308       self._num_trainable_params = m.num_trainable_params
 309       self._saver = m.saver
 310     with tf.variable_scope('model', reuse=True, use_resource=False):
 311       meval = CifarModel(self.hparams)
 312       meval.build('eval')
 313     return m, meval
 314 
 315   def _calc_starting_epoch(self, m):
 316     """Calculates the starting epoch for model m based on global step."""
 317     hparams = self.hparams
 318     batch_size = hparams.batch_size
 319     steps_per_epoch = int(hparams.train_size / batch_size)
 320     with self._new_session(m):
 321       curr_step = self.session.run(m.global_step)
 322     total_steps = steps_per_epoch * hparams.num_epochs
 323     epochs_left = (total_steps - curr_step) // steps_per_epoch
 324     starting_epoch = hparams.num_epochs - epochs_left
 325     return starting_epoch
 326 
 327   def _run_training_loop(self, m, curr_epoch):
 328     """Trains the cifar model `m` for one epoch."""
 329     start_time = time.time()
 330     while True:
 331       try:
 332         with self._new_session(m):
 333           train_accuracy = helper_utils.run_epoch_training(
 334               self.session, m, self.data_loader, curr_epoch)
 335           tf.logging.info('Saving model after epoch')
 336           self.save_model(step=curr_epoch)
 337           break
 338       except (tf.errors.AbortedError, tf.errors.UnavailableError) as e:
 339         tf.logging.info('Retryable error caught: %s.  Retrying.', e)
 340     tf.logging.info('Finished epoch: {}'.format(curr_epoch))
 341     tf.logging.info('Epoch time(min): {}'.format(
 342         (time.time() - start_time) / 60.0))
 343     return train_accuracy
 344 
 345   def _compute_final_accuracies(self, meval):
 346     """Run once training is finished to compute final val/test accuracies."""
 347     valid_accuracy = self.eval_child_model(meval, self.data_loader, 'val')
 348     if self.hparams.eval_test:
 349       test_accuracy = self.eval_child_model(meval, self.data_loader, 'test')
 350     else:
 351       test_accuracy = 0
 352     tf.logging.info('Test Accuracy: {}'.format(test_accuracy))
 353     return valid_accuracy, test_accuracy
 354 
 355   def run_model(self):
 356     """Trains and evalutes the image model."""
 357     hparams = self.hparams
 358 
 359     # Build the child graph
 360     with tf.Graph().as_default(), tf.device(
 361         '/cpu:0' if FLAGS.use_cpu else '/gpu:0'):
 362       m, meval = self._build_models()
 363 
 364       # Figure out what epoch we are on
 365       starting_epoch = self._calc_starting_epoch(m)
 366 
 367       # Run the validation error right at the beginning
 368       valid_accuracy = self.eval_child_model(
 369           meval, self.data_loader, 'val')
 370       tf.logging.info('Before Training Epoch: {}     Val Acc: {}'.format(
 371           starting_epoch, valid_accuracy))
 372       training_accuracy = None
 373 
 374       for curr_epoch in xrange(starting_epoch, hparams.num_epochs):
 375 
 376         # Run one training epoch
 377         training_accuracy = self._run_training_loop(m, curr_epoch)
 378 
 379         valid_accuracy = self.eval_child_model(
 380             meval, self.data_loader, 'val')
 381         tf.logging.info('Epoch: {}    Valid Acc: {}'.format(
 382             curr_epoch, valid_accuracy))
 383 
 384       valid_accuracy, test_accuracy = self._compute_final_accuracies(
 385           meval)
 386 
 387     tf.logging.info(
 388         'Train Acc: {}    Valid Acc: {}     Test Acc: {}'.format(
 389             training_accuracy, valid_accuracy, test_accuracy))
 390

 392   def saver(self):
 393     return self._saver

 397     return self._session

 401     return self._num_trainable_params

 404 def main(_):
 405   if FLAGS.dataset not in ['cifar10', 'cifar100']:
 406     raise ValueError('Invalid dataset: %s' % FLAGS.dataset)
 407   hparams = tf.contrib.training.HParams(
 408       train_size=50000,
 409       validation_size=0,
 410       eval_test=1,
 411       dataset=FLAGS.dataset,
 412       data_path=FLAGS.data_path,
 413       batch_size=128,
 414       gradient_clipping_by_global_norm=5.0)
 415   if FLAGS.model_name == 'wrn':
 416     hparams.add_hparam('model_name', 'wrn')
 417     hparams.add_hparam('num_epochs', 200)
 418     hparams.add_hparam('wrn_size', 160)
 419     hparams.add_hparam('lr', 0.1)
 420     hparams.add_hparam('weight_decay_rate', 5e-4)
 421   elif FLAGS.model_name == 'shake_shake_32':
 422     hparams.add_hparam('model_name', 'shake_shake')
 423     hparams.add_hparam('num_epochs', 1800)
 424     hparams.add_hparam('shake_shake_widen_factor', 2)
 425     hparams.add_hparam('lr', 0.01)
 426     hparams.add_hparam('weight_decay_rate', 0.001)
 427   elif FLAGS.model_name == 'shake_shake_96':
 428     hparams.add_hparam('model_name', 'shake_shake')
 429     hparams.add_hparam('num_epochs', 1800)
 430     hparams.add_hparam('shake_shake_widen_factor', 6)
 431     hparams.add_hparam('lr', 0.01)
 432     hparams.add_hparam('weight_decay_rate', 0.001)
 433   elif FLAGS.model_name == 'shake_shake_112':
 434     hparams.add_hparam('model_name', 'shake_shake')
 435     hparams.add_hparam('num_epochs', 1800)
 436     hparams.add_hparam('shake_shake_widen_factor', 7)
 437     hparams.add_hparam('lr', 0.01)
 438     hparams.add_hparam('weight_decay_rate', 0.001)
 439   elif FLAGS.model_name == 'pyramid_net':
 440     hparams.add_hparam('model_name', 'pyramid_net')
 441     hparams.add_hparam('num_epochs', 1800)
 442     hparams.add_hparam('lr', 0.05)
 443     hparams.add_hparam('weight_decay_rate', 5e-5)
 444     hparams.batch_size = 64
 445   else:
 446     raise ValueError('Not Valid Model Name: %s' % FLAGS.model_name)
 447   cifar_trainer = CifarModelTrainer(hparams)
 448   cifar_trainer.run_model()
 449 
 450 if __name__ == '__main__':
 451   tf.logging.set_verbosity(tf.logging.INFO)
 452   tf.app.run()
