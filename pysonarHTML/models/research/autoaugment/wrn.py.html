  28 def residual_block(
  29     x, in_filter, out_filter, stride, activate_before_residual=False):
  30   """Adds residual connection to `x` in addition to applying BN->ReLU->3x3 Conv.
  31 
  32   Args:
  33     x: Tensor that is the output of the previous layer in the model.
  34     in_filter: Number of filters `x` has.
  35     out_filter: Number of filters that the output of this layer will have.
  36     stride: Integer that specified what stride should be applied to `x`.
  37     activate_before_residual: Boolean on whether a BN->ReLU should be applied
  38       to x before the convolution is applied.
  39 
  40   Returns:
  41     A Tensor that is the result of applying two sequences of BN->ReLU->3x3 Conv
  42     and then adding that Tensor to `x`.
  43   """
  44 
  45   if activate_before_residual:  # Pass up RELU and BN activation for resnet
  46     with tf.variable_scope('shared_activation'):
  47       x = ops.batch_norm(x, scope='init_bn')
  48       x = tf.nn.relu(x)
  49       orig_x = x
  50   else:
  51     orig_x = x
  52 
  53   block_x = x
  54   if not activate_before_residual:
  55     with tf.variable_scope('residual_only_activation'):
  56       block_x = ops.batch_norm(block_x, scope='init_bn')
  57       block_x = tf.nn.relu(block_x)
  58 
  59   with tf.variable_scope('sub1'):
  60     block_x = ops.conv2d(
  61         block_x, out_filter, 3, stride=stride, scope='conv1')
  62 
  63   with tf.variable_scope('sub2'):
  64     block_x = ops.batch_norm(block_x, scope='bn2')
  65     block_x = tf.nn.relu(block_x)
  66     block_x = ops.conv2d(
  67         block_x, out_filter, 3, stride=1, scope='conv2')
  68 
  69   with tf.variable_scope(
  70       'sub_add'):  # If number of filters do not agree then zero pad them
  71     if in_filter != out_filter:
  72       orig_x = ops.avg_pool(orig_x, stride, stride)
  73       orig_x = ops.zero_pad(orig_x, in_filter, out_filter)
  74   x = orig_x + block_x
  75   return x

  78 def _res_add(in_filter, out_filter, stride, x, orig_x):
  79   """Adds `x` with `orig_x`, both of which are layers in the model.
  80 
  81   Args:
  82     in_filter: Number of filters in `orig_x`.
  83     out_filter: Number of filters in `x`.
  84     stride: Integer specifying the stide that should be applied `orig_x`.
  85     x: Tensor that is the output of the previous layer.
  86     orig_x: Tensor that is the output of an earlier layer in the network.
  87 
  88   Returns:
  89     A Tensor that is the result of `x` and `orig_x` being added after
  90     zero padding and striding are applied to `orig_x` to get the shapes
  91     to match.
  92   """
  93   if in_filter != out_filter:
  94     orig_x = ops.avg_pool(orig_x, stride, stride)
  95     orig_x = ops.zero_pad(orig_x, in_filter, out_filter)
  96   x = x + orig_x
  97   orig_x = x
  98   return x, orig_x

 101 def build_wrn_model(images, num_classes, wrn_size):
 102   """Builds the WRN model.
 103 
 104   Build the Wide ResNet model from https://arxiv.org/abs/1605.07146.
 105 
 106   Args:
 107     images: Tensor of images that will be fed into the Wide ResNet Model.
 108     num_classes: Number of classed that the model needs to predict.
 109     wrn_size: Parameter that scales the number of filters in the Wide ResNet
 110       model.
 111 
 112   Returns:
 113     The logits of the Wide ResNet model.
 114   """
 115   kernel_size = wrn_size
 116   filter_size = 3
 117   num_blocks_per_resnet = 4
 118   filters = [
 119       min(kernel_size, 16), kernel_size, kernel_size * 2, kernel_size * 4
 120   ]
 121   strides = [1, 2, 2]  # stride for each resblock
 122 
 123   # Run the first conv
 124   with tf.variable_scope('init'):
 125     x = images
 126     output_filters = filters[0]
 127     x = ops.conv2d(x, output_filters, filter_size, scope='init_conv')
 128 
 129   first_x = x  # Res from the beginning
 130   orig_x = x  # Res from previous block
 131 
 132   for block_num in range(1, 4):
 133     with tf.variable_scope('unit_{}_0'.format(block_num)):
 134       activate_before_residual = True if block_num == 1 else False
 135       x = residual_block(
 136           x,
 137           filters[block_num - 1],
 138           filters[block_num],
 139           strides[block_num - 1],
 140           activate_before_residual=activate_before_residual)
 141     for i in range(1, num_blocks_per_resnet):
 142       with tf.variable_scope('unit_{}_{}'.format(block_num, i)):
 143         x = residual_block(
 144             x,
 145             filters[block_num],
 146             filters[block_num],
 147             1,
 148             activate_before_residual=False)
 149     x, orig_x = _res_add(filters[block_num - 1], filters[block_num],
 150                          strides[block_num - 1], x, orig_x)
 151   final_stride_val = np.prod(strides)
 152   x, _ = _res_add(filters[0], filters[3], final_stride_val, x, first_x)
 153   with tf.variable_scope('unit_last'):
 154     x = ops.batch_norm(x, scope='final_bn')
 155     x = tf.nn.relu(x)
 156     x = ops.global_avg_pool(x)
 157     logits = ops.fc(x, num_classes)
 158   return logits
