  27 def round_int(x):
  28   """Rounds `x` and then converts to an int."""
  29   return int(math.floor(x + 0.5))

  32 def shortcut(x, output_filters, stride):
  33   """Applies strided avg pool or zero padding to make output_filters match x."""
  34   num_filters = int(x.shape[3])
  35   if stride == 2:
  36     x = ops.avg_pool(x, 2, stride=stride, padding='SAME')
  37   if num_filters != output_filters:
  38     diff = output_filters - num_filters
  39     assert diff > 0
  40     # Zero padd diff zeros
  41     padding = [[0, 0], [0, 0], [0, 0], [0, diff]]
  42     x = tf.pad(x, padding)
  43   return x

  46 def calc_prob(curr_layer, total_layers, p_l):
  47   """Calculates drop prob depending on the current layer."""
  48   return 1 - (float(curr_layer) / total_layers) * p_l

  51 def bottleneck_layer(x, n, stride, prob, is_training, alpha, beta):
  52   """Bottleneck layer for shake drop model."""
  53   assert alpha[1] > alpha[0]
  54   assert beta[1] > beta[0]
  55   with tf.variable_scope('bottleneck_{}'.format(prob)):
  56     input_layer = x
  57     x = ops.batch_norm(x, scope='bn_1_pre')
  58     x = ops.conv2d(x, n, 1, scope='1x1_conv_contract')
  59     x = ops.batch_norm(x, scope='bn_1_post')
  60     x = tf.nn.relu(x)
  61     x = ops.conv2d(x, n, 3, stride=stride, scope='3x3')
  62     x = ops.batch_norm(x, scope='bn_2')
  63     x = tf.nn.relu(x)
  64     x = ops.conv2d(x, n * 4, 1, scope='1x1_conv_expand')
  65     x = ops.batch_norm(x, scope='bn_3')
  66 
  67     # Apply regularization here
  68     # Sample bernoulli with prob
  69     if is_training:
  70       batch_size = tf.shape(x)[0]
  71       bern_shape = [batch_size, 1, 1, 1]
  72       random_tensor = prob
  73       random_tensor += tf.random_uniform(bern_shape, dtype=tf.float32)
  74       binary_tensor = tf.floor(random_tensor)
  75 
  76       alpha_values = tf.random_uniform(
  77           [batch_size, 1, 1, 1], minval=alpha[0], maxval=alpha[1],
  78           dtype=tf.float32)
  79       beta_values = tf.random_uniform(
  80           [batch_size, 1, 1, 1], minval=beta[0], maxval=beta[1],
  81           dtype=tf.float32)
  82       rand_forward = (
  83           binary_tensor + alpha_values - binary_tensor * alpha_values)
  84       rand_backward = (
  85           binary_tensor + beta_values - binary_tensor * beta_values)
  86       x = x * rand_backward + tf.stop_gradient(x * rand_forward -
  87                                                x * rand_backward)
  88     else:
  89       expected_alpha = (alpha[1] + alpha[0])/2
  90       # prob is the expectation of the bernoulli variable
  91       x = (prob + expected_alpha - prob * expected_alpha) * x
  92 
  93     res = shortcut(input_layer, n * 4, stride)
  94     return x + res

  97 def build_shake_drop_model(images, num_classes, is_training):
  98   """Builds the PyramidNet Shake-Drop model.
  99 
 100   Build the PyramidNet Shake-Drop model from https://arxiv.org/abs/1802.02375.
 101 
 102   Args:
 103     images: Tensor of images that will be fed into the Wide ResNet Model.
 104     num_classes: Number of classed that the model needs to predict.
 105     is_training: Is the model training or not.
 106 
 107   Returns:
 108     The logits of the PyramidNet Shake-Drop model.
 109   """
 110   # ShakeDrop Hparams
 111   p_l = 0.5
 112   alpha_shake = [-1, 1]
 113   beta_shake = [0, 1]
 114 
 115   # PyramidNet Hparams
 116   alpha = 200
 117   depth = 272
 118   # This is for the bottleneck architecture specifically
 119   n = int((depth - 2) / 9)
 120   start_channel = 16
 121   add_channel = alpha / (3 * n)
 122 
 123   # Building the models
 124   x = images
 125   x = ops.conv2d(x, 16, 3, scope='init_conv')
 126   x = ops.batch_norm(x, scope='init_bn')
 127 
 128   layer_num = 1
 129   total_layers = n * 3
 130   start_channel += add_channel
 131   prob = calc_prob(layer_num, total_layers, p_l)
 132   x = bottleneck_layer(
 133       x, round_int(start_channel), 1, prob, is_training, alpha_shake,
 134       beta_shake)
 135   layer_num += 1
 136   for _ in range(1, n):
 137     start_channel += add_channel
 138     prob = calc_prob(layer_num, total_layers, p_l)
 139     x = bottleneck_layer(
 140         x, round_int(start_channel), 1, prob, is_training, alpha_shake,
 141         beta_shake)
 142     layer_num += 1
 143 
 144   start_channel += add_channel
 145   prob = calc_prob(layer_num, total_layers, p_l)
 146   x = bottleneck_layer(
 147       x, round_int(start_channel), 2, prob, is_training, alpha_shake,
 148       beta_shake)
 149   layer_num += 1
 150   for _ in range(1, n):
 151     start_channel += add_channel
 152     prob = calc_prob(layer_num, total_layers, p_l)
 153     x = bottleneck_layer(
 154         x, round_int(start_channel), 1, prob, is_training, alpha_shake,
 155         beta_shake)
 156     layer_num += 1
 157 
 158   start_channel += add_channel
 159   prob = calc_prob(layer_num, total_layers, p_l)
 160   x = bottleneck_layer(
 161       x, round_int(start_channel), 2, prob, is_training, alpha_shake,
 162       beta_shake)
 163   layer_num += 1
 164   for _ in range(1, n):
 165     start_channel += add_channel
 166     prob = calc_prob(layer_num, total_layers, p_l)
 167     x = bottleneck_layer(
 168         x, round_int(start_channel), 1, prob, is_training, alpha_shake,
 169         beta_shake)
 170     layer_num += 1
 171 
 172   assert layer_num - 1 == total_layers
 173   x = ops.batch_norm(x, scope='final_bn')
 174   x = tf.nn.relu(x)
 175   x = ops.global_avg_pool(x)
 176   # Fully connected
 177   logits = ops.fc(x, num_classes)
 178   return logits
