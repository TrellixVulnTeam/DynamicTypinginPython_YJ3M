  28 def tanh_sample_info(mu, logsigma, stop_action_gradient=False, n_samples=1):
  29     if n_samples > 1:
  30       mu = tf.expand_dims(mu, 2)
  31       logsigma = tf.expand_dims(logsigma, 2)
  32       sample_shape = tf.concat([tf.shape(mu), n_samples], 0)
  33     else:
  34       sample_shape = tf.shape(mu)
  35 
  36     flat_act = mu + tf.random_normal(sample_shape) * tf.exp(logsigma)
  37     if stop_action_gradient: flat_act = tf.stop_gradient(flat_act)
  38     normalized_dist_t = (flat_act - mu) * tf.exp(-logsigma)  # ... x D
  39     quadratic = - 0.5 * tf.reduce_sum(normalized_dist_t ** 2, axis=-1) # ... x (None)
  40     log_z = tf.reduce_sum(logsigma, axis=-1)  # ... x (None)
  41     D_t = tf.cast(tf.shape(mu)[-1], tf.float32)
  42     log_z += 0.5 * D_t * np.log(2 * np.pi)
  43     flat_ll = quadratic - log_z
  44 
  45     scaled_act = tf.tanh(flat_act)
  46     corr = tf.reduce_sum(tf.log(1. - tf.square(scaled_act) + 1e-6), axis=-1)
  47     scaled_ll = flat_ll - corr
  48     return flat_act, flat_ll, scaled_act, scaled_ll

  50 def tf_cheating_contcartpole(state, action):
  51     gravity = 9.8
  52     masscart = 1.0
  53     masspole = 0.1
  54     total_mass = (masspole + masscart)
  55     length = 0.5 # actually half the pole's length
  56     polemass_length = (masspole * length)
  57     force_mag = 10.0
  58     tau = 0.02  # seconds between state updates
  59 
  60     # Angle at which to fail the episode
  61     theta_threshold_radians = 12 * 2 * math.pi / 360
  62     x_threshold = 2.4
  63 
  64     x, x_dot, theta, theta_dot = tf.split(state, 4, axis=-1)
  65     done =  tf.logical_or(x < -x_threshold,
  66                           tf.logical_or(x > x_threshold,
  67                           tf.logical_or(theta < -theta_threshold_radians,
  68                                         theta > theta_threshold_radians)))
  69 
  70     force = force_mag * action
  71     costheta = tf.cos(theta)
  72     sintheta = tf.sin(theta)
  73     temp = old_div((force + polemass_length * theta_dot * theta_dot * sintheta), total_mass)
  74     thetaacc = old_div((gravity * sintheta - costheta* temp), (length * (old_div(4.0,3.0) - masspole * costheta * costheta / total_mass)))
  75     xacc  = temp - polemass_length * thetaacc * costheta / total_mass
  76     x  = x + tau * x_dot
  77     x_dot = x_dot + tau * xacc
  78     theta = theta + tau * theta_dot
  79     theta_dot = theta_dot + tau * thetaacc
  80     state = tf.concat([x,x_dot,theta,theta_dot], -1)
  81     done = tf.squeeze(tf.cast(done, tf.float32), -1)
  82     reward = 1.0 - done
  83     done *= 0.
  84     return state, reward, done

  86 def create_directory(dir):
  87     dir_chunks = dir.split("/")
  88     for i in range(len(dir_chunks)):
  89         partial_dir = "/".join(dir_chunks[:i+1])
  90         try:
  91             os.makedirs(partial_dir)
  92         except OSError:
  93             pass
  94     return dir

 105 def get_largest_epoch_in_dir(dir, saveid):
 106     reg_matches = [re.findall('\d+_%s'%saveid,filename) for filename in os.listdir(dir)]
 107     epoch_labels = [int(regmatch[0].split("_")[0]) for regmatch in reg_matches if regmatch]
 108     if len(epoch_labels) == 0: return False
 109     return max(epoch_labels)
 110 
 111 def wipe_all_but_largest_epoch_in_dir(dir, saveid):
 112     largest = get_largest_epoch_in_dir(dir, saveid)
 113     reg_matches = [(filename, re.findall('\d+_%s'%saveid,filename)) for filename in os.listdir(dir)]
 114     for filename, regmatch in reg_matches:
 115         if regmatch and int(regmatch[0].split("_")[0]) != largest:
 116             os.remove(os.path.join(dir,filename))

 118 class ConfigDict(dict):
 119     def __init__(self, loc=None, ghost=False):
 120         self._dict = defaultdict(lambda :False)
 121         self.ghost = ghost
 122         if loc:
 123             with open(loc) as f: raw = json.load(f)
 124             if "inherits" in raw and raw["inherits"]:
 125                 for dep_loc in raw["inherits"]:
 126                     self.update(ConfigDict(dep_loc))
 127             if "updates" in raw and raw["updates"]:
 128                 self.update(raw["updates"], include_all=True)
 129 
 130     def __getitem__(self, key):
 131         return self._dict[key]
 132 
 133     def __setitem__(self, key, value):
 134         self._dict[key] = value
 135 
 136     def __str__(self):
 137         return str(dict(self._dict))
 138 
 139     def __repr__(self):
 140         return str(dict(self._dict))
 141 
 142     def __iter__(self):
 143         return self._dict.__iter__()
 144 
 145     def __bool__(self):
 146         return bool(self._dict)
 147 
 148     def __nonzero__(self):
 149         return bool(self._dict)
 150 
 151     def update(self, dictlike, include_all=False):
 152         for key in dictlike:
 153             value = dictlike[key]
 154             if isinstance(value, dict):
 155                 if key[0] == "*": # this means only override, do not set
 156                     key = key[1:]
 157                     ghost = True
 158                 else:
 159                     ghost = False
 160                 if not include_all and isinstance(value, ConfigDict) and key not in self._dict and value.ghost: continue
 161                 if key not in self._dict: self._dict[key] = ConfigDict(ghost=ghost)
 162                 self._dict[key].update(value)
 163             else:
 164                 self._dict[key] = value
