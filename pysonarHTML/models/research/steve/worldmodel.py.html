  22 class DeterministicWorldModel(CoreModel):
  23   """
  24   A simple feed-forward neural network world model, with an option for an ensemble.
  25   """
  26 
  27   @property
  28   def saveid(self):
  29     return "worldmodel"
  30 
  31   def create_params(self, env_config, learner_config):
  32     self.obs_dim = np.prod(env_config["obs_dims"])
  33     self.action_dim = env_config["action_dim"]
  34     self.reward_scale = env_config["reward_scale"]
  35     self.discount = env_config["discount"]
  36 
  37     self.aux_hidden_dim = self.learner_config["aux_hidden_dim"]
  38     self.transition_hidden_dim = self.learner_config["transition_hidden_dim"]
  39     self.bayesian_config = self.learner_config["bayesian"]
  40 
  41     with tf.variable_scope(self.name):
  42       if self.bayesian_config:
  43         self.transition_predictor = nn.EnsembleFeedForwardNet('transition_predictor', self.obs_dim + self.action_dim, [self.obs_dim], layers=8, hidden_dim=self.transition_hidden_dim, get_uncertainty=True, ensemble_size=self.bayesian_config["transition"]["ensemble_size"], train_sample_count=self.bayesian_config["transition"]["train_sample_count"], eval_sample_count=self.bayesian_config["transition"]["eval_sample_count"])
  44         self.done_predictor =       nn.EnsembleFeedForwardNet('done_predictor', self.obs_dim + self.obs_dim + self.action_dim, [], layers=4, hidden_dim=self.aux_hidden_dim, get_uncertainty=True, ensemble_size=self.bayesian_config["transition"]["ensemble_size"], train_sample_count=self.bayesian_config["transition"]["train_sample_count"], eval_sample_count=self.bayesian_config["transition"]["eval_sample_count"])
  45         self.reward_predictor =     nn.EnsembleFeedForwardNet('reward_predictor', self.obs_dim + self.obs_dim + self.action_dim, [], layers=4, hidden_dim=self.aux_hidden_dim, get_uncertainty=True, ensemble_size=self.bayesian_config["reward"]["ensemble_size"], train_sample_count=self.bayesian_config["reward"]["train_sample_count"], eval_sample_count=self.bayesian_config["reward"]["eval_sample_count"])
  46       else:
  47         self.transition_predictor = nn.FeedForwardNet('transition_predictor', self.obs_dim + self.action_dim, [self.obs_dim], layers=8, hidden_dim=self.transition_hidden_dim, get_uncertainty=True)
  48         self.done_predictor =       nn.FeedForwardNet('done_predictor',   self.obs_dim + self.obs_dim + self.action_dim, [], layers=4, hidden_dim=self.aux_hidden_dim, get_uncertainty=True)
  49         self.reward_predictor =     nn.FeedForwardNet('reward_predictor', self.obs_dim + self.obs_dim + self.action_dim, [], layers=4, hidden_dim=self.aux_hidden_dim, get_uncertainty=True)
  50 
  51   def get_ensemble_idx_info(self):
  52     if self.bayesian_config is not False:
  53       ensemble_idxs = tf.random_shuffle(tf.range(self.transition_predictor.ensemble_size))
  54       transition_ensemble_sample_n = self.transition_predictor.eval_sample_count
  55       reward_ensemble_sample_n = self.reward_predictor.eval_sample_count
  56       ensemble_idxs = ensemble_idxs[:transition_ensemble_sample_n]
  57       return ensemble_idxs, transition_ensemble_sample_n, reward_ensemble_sample_n
  58     else:
  59       return None, 1, 1
  60 
  61   def build_training_graph(self, obs, next_obs, actions, rewards, dones, data_size):
  62     info = tf.concat([obs, actions], -1)
  63     predicted_next_obs = self.transition_predictor(info, is_eval=False, reduce_mode="random") + obs
  64     next_info = tf.concat([next_obs, info], -1)
  65     predicted_dones = self.done_predictor(next_info, is_eval=False, reduce_mode="random")
  66     predicted_rewards = self.reward_predictor(next_info, is_eval=False, reduce_mode="random")
  67 
  68     done_losses = tf.nn.sigmoid_cross_entropy_with_logits(labels=dones, logits=predicted_dones)
  69     reward_losses = .5 * tf.square(rewards - predicted_rewards)
  70     next_obs_losses = .5 * tf.reduce_sum(tf.square(next_obs - predicted_next_obs), -1)
  71 
  72     done_loss = tf.reduce_mean(done_losses)
  73     reward_loss = tf.reduce_mean(reward_losses)
  74     next_obs_loss = tf.reduce_mean(next_obs_losses)
  75     reg_loss = .0001 * (self.done_predictor.l2_loss() +
  76                         self.reward_predictor.l2_loss() +
  77                         self.transition_predictor.l2_loss())
  78 
  79     total_loss = done_loss + reward_loss + next_obs_loss + reg_loss
  80 
  81     inspect = (total_loss, done_loss, reward_loss, next_obs_loss, reg_loss)
  82 
  83     return total_loss, inspect
  84 
  85   def init_extra_info(self, obs):
  86     return tf.zeros_like(obs)
  87 
  88   def transition(self, obs, action, extra_info, ensemble_idxs=None, pre_expanded=None):
  89     info = tf.concat([obs, action], -1)
  90     next_obs_delta = self.transition_predictor(info, reduce_mode="none", ensemble_idxs=ensemble_idxs, pre_expanded=pre_expanded)
  91     if ensemble_idxs is None:
  92       next_obs = tf.expand_dims(obs,-2) + next_obs_delta
  93       next_info = tf.concat([next_obs, tf.expand_dims(info,-2)], -1)
  94     else:
  95       next_obs = obs + next_obs_delta
  96       next_info = tf.concat([next_obs, info], -1)
  97     done = tf.nn.sigmoid(self.done_predictor(next_info, reduce_mode="none", ensemble_idxs=ensemble_idxs, pre_expanded=True))
  98     extra_info = tf.zeros_like(obs)
  99     return next_obs, done, extra_info

 101   def get_rewards(self, obs, action, next_obs):
 102     next_info = tf.concat([next_obs, obs, action], -1)
 103     reward = self.reward_predictor(next_info, reduce_mode="none")
 104     return reward
