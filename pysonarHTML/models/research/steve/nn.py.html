  22 class FeedForwardNet(object):
  23     """Custom feed-forward network layer."""
  24     def __init__(self, name, in_size, out_shape, layers=1, hidden_dim=32, final_nonlinearity=None, get_uncertainty=False):
  25         self.name = name
  26         self.in_size = in_size
  27         self.out_shape = out_shape
  28         self.out_size = np.prod(out_shape)
  29         self.layers = layers
  30         self.hidden_dim = hidden_dim
  31         self.final_nonlinearity = (lambda x:x) if final_nonlinearity is None else final_nonlinearity
  32         self.get_uncertainty = get_uncertainty
  33 
  34         self.weights = [None] * layers
  35         self.biases = [None] * layers
  36 
  37         self.params_list = []
  38 
  39         with tf.variable_scope(name):
  40             for layer_i in range(self.layers):
  41                 in_size = self.hidden_dim
  42                 out_size = self.hidden_dim
  43                 if layer_i == 0: in_size = self.in_size
  44                 if layer_i == self.layers - 1: out_size = self.out_size
  45                 self.weights[layer_i] = tf.get_variable("weights%d" % layer_i, [in_size, out_size], initializer=tf.contrib.layers.xavier_initializer())
  46                 self.biases[layer_i] = tf.get_variable("bias%d" % layer_i, [1, out_size], initializer=tf.constant_initializer(0.0))
  47                 self.params_list += [self.weights[layer_i], self.biases[layer_i]]
  48 
  49     def __call__(self, x, stop_params_gradient=False, is_eval=True, ensemble_idxs=None, pre_expanded=None, reduce_mode="none"):
  50         original_shape = tf.shape(x)
  51         h = tf.reshape(x, [-1, self.in_size])
  52         for layer_i in range(self.layers):
  53             nonlinearity = tf.nn.relu if layer_i + 1 < self.layers else self.final_nonlinearity
  54             if stop_params_gradient: h = nonlinearity(tf.matmul(h, tf.stop_gradient(self.weights[layer_i])) + tf.stop_gradient(self.biases[layer_i]))
  55             else:             h = nonlinearity(tf.matmul(h, self.weights[layer_i]) + self.biases[layer_i])
  56         if len(self.out_shape) > 0: h = tf.reshape(h, tf.concat([original_shape[:-1], tf.constant(self.out_shape)], -1))
  57         else:                       h = tf.reshape(h, original_shape[:-1])
  58         if pre_expanded is None: pre_expanded = ensemble_idxs is not None
  59         if reduce_mode == "none" and not pre_expanded and self.get_uncertainty:
  60             if len(self.out_shape) > 0: h = tf.expand_dims(h, -2)
  61             else:                       h = tf.expand_dims(h, -1)
  62         return h
  63 
  64     def l2_loss(self):
  65         return tf.add_n([tf.reduce_sum(.5 * tf.square(mu)) for mu in self.params_list])

  67 class BayesianDropoutFeedForwardNet(FeedForwardNet):
  68     """Custom feed-forward network layer, with dropout as a Bayesian approximation."""
  69     def __init__(self, name, in_size, out_shape, layers=1, hidden_dim=32, final_nonlinearity=None, get_uncertainty=False, keep_prob=.5, eval_sample_count=2, consistent_random_seed=False):
  70         super(BayesianDropoutFeedForwardNet, self).__init__(name, in_size, out_shape, layers=layers, hidden_dim=hidden_dim,
  71                                                             final_nonlinearity=final_nonlinearity, get_uncertainty=get_uncertainty)
  72         self.keep_prob = keep_prob
  73         self.eval_sample_count = eval_sample_count
  74         if eval_sample_count < 2: raise Exception("eval_sample_count must be at least 2 to estimate uncertainty")
  75         self.dropout_seed = tf.random_uniform([layers], maxval=1e18, dtype=tf.int64) if consistent_random_seed else [None] * layers
  76 
  77     def __call__(self, x, stop_params_gradient=False, is_eval=True, pre_expanded=False, ensemble_idxs=None, reduce_mode="none"):
  78         if is_eval:
  79             x = tf.tile(tf.expand_dims(x,0), tf.concat([tf.constant([self.eval_sample_count]), tf.ones_like(tf.shape(x))], 0))
  80         original_shape = tf.shape(x)
  81         h = tf.reshape(x, [-1, self.in_size])
  82         for layer_i in range(self.layers):
  83             nonlinearity = tf.nn.relu if layer_i + 1 < self.layers else self.final_nonlinearity
  84             if layer_i > 0: h = tf.nn.dropout(h, keep_prob=self.keep_prob, seed=self.dropout_seed[layer_i])
  85             if stop_params_gradient: h = nonlinearity(tf.matmul(h, tf.stop_gradient(self.weights[layer_i])) + tf.stop_gradient(self.biases[layer_i]))
  86             else:                    h = nonlinearity(tf.matmul(h, self.weights[layer_i]) + self.biases[layer_i])
  87         if len(self.out_shape) > 0: h = tf.reshape(h, tf.concat([original_shape[:-1], tf.constant(self.out_shape)], -1))
  88         else:                       h = tf.reshape(h, original_shape[:-1])
  89         if is_eval:
  90             h, uncertainty = tf.nn.moments(h, 0)
  91             if self.get_uncertainty: return h, uncertainty
  92             else:                    return h
  93         else:
  94             return h

  97 class EnsembleFeedForwardNet(FeedForwardNet):
  98     """Custom feed-forward network layer with an ensemble."""
  99     def __init__(self, name, in_size, out_shape, layers=1, hidden_dim=32, final_nonlinearity=None, get_uncertainty=False, ensemble_size=2, train_sample_count=2, eval_sample_count=2):
 100         if train_sample_count > ensemble_size: raise Exception("train_sample_count cannot be larger than ensemble size")
 101         if eval_sample_count > ensemble_size: raise Exception("eval_sample_count cannot be larger than ensemble size")
 102         self.name = name
 103         self.in_size = in_size
 104         self.out_shape = out_shape
 105         self.out_size = np.prod(out_shape)
 106         self.layers = layers
 107         self.hidden_dim = hidden_dim
 108         self.final_nonlinearity = (lambda x:x) if final_nonlinearity is None else final_nonlinearity
 109         self.get_uncertainty = get_uncertainty
 110         self.ensemble_size = ensemble_size
 111         self.train_sample_count = train_sample_count
 112         self.eval_sample_count = eval_sample_count
 113 
 114         self.weights = [None] * layers
 115         self.biases = [None] * layers
 116 
 117         self.params_list = []
 118 
 119         with tf.variable_scope(name):
 120             for layer_i in range(self.layers):
 121                 in_size = self.hidden_dim
 122                 out_size = self.hidden_dim
 123                 if layer_i == 0: in_size = self.in_size
 124                 if layer_i == self.layers - 1: out_size = self.out_size
 125                 self.weights[layer_i] = tf.get_variable("weights%d" % layer_i, [ensemble_size, in_size, out_size], initializer=tf.contrib.layers.xavier_initializer())
 126                 self.biases[layer_i] = tf.get_variable("bias%d" % layer_i, [ensemble_size, out_size], initializer=tf.constant_initializer(0.0))
 127                 self.params_list += [self.weights[layer_i], self.biases[layer_i]]
 128 
 129     def __call__(self, x, stop_params_gradient=False, is_eval=True, ensemble_idxs=None, pre_expanded=None, reduce_mode="none"):
 130         if pre_expanded is None: pre_expanded = ensemble_idxs is not None
 131         if ensemble_idxs is None:
 132             ensemble_idxs = tf.random_shuffle(tf.range(self.ensemble_size))
 133             ensemble_sample_n = self.eval_sample_count if is_eval else self.train_sample_count
 134             ensemble_idxs = ensemble_idxs[:ensemble_sample_n]
 135         else:
 136             ensemble_sample_n = tf.shape(ensemble_idxs)[0]
 137 
 138         weights = [tf.gather(w, ensemble_idxs, axis=0) for w in self.weights]
 139         biases = [tf.expand_dims(tf.gather(b, ensemble_idxs, axis=0),0) for b in self.biases]
 140 
 141         original_shape = tf.shape(x)
 142         if pre_expanded: h = tf.reshape(x, [-1, ensemble_sample_n, self.in_size])
 143         else:            h = tf.tile(tf.reshape(x, [-1, 1, self.in_size]), [1, ensemble_sample_n, 1])
 144         for layer_i in range(self.layers):
 145             nonlinearity = tf.nn.relu if layer_i + 1 < self.layers else self.final_nonlinearity
 146             if stop_params_gradient: h = nonlinearity(tf.einsum('bri,rij->brj', h, tf.stop_gradient(weights[layer_i])) + tf.stop_gradient(biases[layer_i]))
 147             else:                    h = nonlinearity(tf.einsum('bri,rij->brj', h, weights[layer_i]) + biases[layer_i])
 148 
 149         if pre_expanded:
 150             if len(self.out_shape) > 0: h = tf.reshape(h, tf.concat([original_shape[:-1], tf.constant(self.out_shape)], -1))
 151             else:                       h = tf.reshape(h, original_shape[:-1])
 152         else:
 153             if len(self.out_shape) > 0: h = tf.reshape(h, tf.concat([original_shape[:-1], tf.constant([ensemble_sample_n]), tf.constant(self.out_shape)], -1))
 154             else:                       h = tf.reshape(h, tf.concat([original_shape[:-1], tf.constant([ensemble_sample_n])], -1))
 155 
 156         if reduce_mode == "none":
 157             pass
 158         elif reduce_mode == "random":
 159             if len(self.out_shape) > 0: h = tf.reduce_sum(h * tf.reshape(tf.one_hot(tf.random_uniform([tf.shape(h)[0]], 0, ensemble_sample_n, dtype=tf.int64), ensemble_sample_n), tf.concat([tf.shape(h)[:1], tf.ones_like(tf.shape(h)[1:-2]), tf.constant([ensemble_sample_n]), tf.constant([1])], 0)), -2)
 160             else:                       h = tf.reduce_sum(h * tf.reshape(tf.one_hot(tf.random_uniform([tf.shape(h)[0]], 0, ensemble_sample_n, dtype=tf.int64), ensemble_sample_n), tf.concat([tf.shape(h)[:1], tf.ones_like(tf.shape(h)[1:-1]), tf.constant([ensemble_sample_n])], 0)), -1)
 161         elif reduce_mode == "mean":
 162             if len(self.out_shape) > 0: h = tf.reduce_mean(h, -2)
 163             else:                       h = tf.reduce_mean(h, -1)
 164         else: raise Exception("use a valid reduce mode: none, random, or mean")
 165 
 166         return h

 169 class ReparamNormal(object):
 170     """Wrapper to make a feedforward network that outputs both mu and logsigma,
 171     for use in the reparameterization trick."""
 172     def __init__(self, base_net, name, in_size, out_shape, layers=2, hidden_dim=32, final_nonlinearity=None, ls_start_bias=0.0, final_net=FeedForwardNet, logsigma_min=-5., logsigma_max=2., **kwargs):
 173         assert layers > 1
 174         self.main_encoder = base_net(name+"_base", in_size, [hidden_dim], layers, hidden_dim, final_nonlinearity=tf.nn.relu, **kwargs)
 175         self.mu = final_net(name+"_mu", hidden_dim, out_shape, layers=1, final_nonlinearity=final_nonlinearity, **kwargs)
 176         self.logsigma = final_net(name+"_logsigma", hidden_dim, out_shape, layers=1, final_nonlinearity=None, **kwargs)
 177         self.ls_start_bias = ls_start_bias
 178         self.params_list = self.main_encoder.params_list + self.mu.params_list + self.logsigma.params_list
 179         self.logsigma_min = logsigma_min
 180         self.logsigma_max = logsigma_max
 181 
 182     def __call__(self, x):
 183         encoded = self.main_encoder(x)
 184         mu = self.mu(encoded)
 185         logsigma = tf.clip_by_value(self.logsigma(encoded) + self.ls_start_bias, self.logsigma_min, self.logsigma_max)
 186         return mu, logsigma
 187 
 188     def l2_loss(self):
 189         return self.main_encoder.l2_loss() + self.mu.l2_loss() + self.logsigma.l2_loss()
