  25 ### Hyperparameters
  26 
  27 NONTERMINAL_STATE_COUNT = 100
  28 NOISE_AMOUNT = 0.1
  29 TRAIN_STEPS = 10000
  30 Q_ENSEMBLE_SIZE = 8
  31 MODEL_ENSEMBLE_SIZE = 8
  32 HORIZON = 5
  33 TRIAL_N = 10
  34 
  35 ### Helper functions
  36 
  37 initial_state = 0
  38 terminal_state = NONTERMINAL_STATE_COUNT + 1
  39 nonterminal_state_count = NONTERMINAL_STATE_COUNT
  40 state_count = NONTERMINAL_STATE_COUNT + 1
  41 final_reward = NONTERMINAL_STATE_COUNT
  42 colors = sns.color_palette('husl', 4)
  43 plt.rcParams["figure.figsize"] = (6,5)
  44 
  45 def step(state):
  46   if state == terminal_state: next_state = terminal_state
  47   else:                       next_state = state + 1
  48 
  49   if state == terminal_state:     reward = 0
  50   elif state+1 == terminal_state: reward = final_reward
  51   else:                           reward = -1
  52 
  53   return next_state, reward
  54 
  55 def noisy_step(state):
  56   if state == terminal_state:               next_state = terminal_state
  57   elif np.random.random([]) < NOISE_AMOUNT: next_state = np.random.randint(0, state_count)
  58   else:                                     next_state = state + 1
  59 
  60   if state == terminal_state:     reward = 0
  61   elif state+1 == terminal_state: reward = final_reward
  62   else:                           reward = -1
  63 
  64   return next_state, reward
  65 
  66 def get_error(Q):
  67   losses = np.square(np.arange(state_count) - Q[:-1])
  68   return np.mean(losses)
  69 
  70 def downsample(array, factor):
  71   pad_size = np.ceil(old_div(float(array.size),factor))*factor - array.size
  72   array_padded = np.append(array, np.zeros([pad_size.astype(np.int64)])*np.NaN)
  73   return scipy.nanmean(array_padded.reshape(-1,factor), axis=1)

 214 # STEVE-Oracle
 215 if True:
 216   print("Running ensemble oracle STEVE.")
 217   trial_results = []
 218 
 219   oracle_q_estimate_errors = []
 220   oracle_mve_estimate_errors = []
 221   oracle_steve_estimate_errors = []
 222   oracle_opt_estimate_errors = []
 223 
 224 
 225   for run_i in range(TRIAL_N):
 226     print("Trial %d" % run_i)
 227     Q = np.random.randint(0,state_count,[Q_ENSEMBLE_SIZE, state_count+1]).astype(np.float64)
 228     Q[:, state_count] = 0
 229     losses = []
 230 
 231     q_estimate_errors = []
 232     mve_estimate_errors = []
 233     steve_estimate_errors = []
 234     opt_estimate_errors = []
 235     steve_beat_freq= []
 236 
 237     for step_i in range(TRAIN_STEPS):
 238       _q_estimate_errors = []
 239       _mve_estimate_errors = []
 240       _steve_estimate_errors = []
 241       _opt_estimate_errors = []
 242       _steve_beat_freq = []
 243 
 244       for q_ensemble_i in range(Q_ENSEMBLE_SIZE):
 245         state = np.random.randint(0,state_count)
 246         next_state, reward = step(state)
 247 
 248         # STEVE rollout
 249         Q_est_mat = np.zeros([HORIZON + 1, Q_ENSEMBLE_SIZE])
 250         reward_est_mat = np.zeros([HORIZON + 1, 1])
 251         first_next_state, first_reward = next_state, reward
 252         next_state, reward = first_next_state, first_reward
 253         Q_est_mat[0, :] = Q[:, next_state]
 254         reward_est_mat[0, 0] = reward
 255         for timestep_i in range(1,HORIZON+1):
 256           next_state, reward = step(next_state)
 257           Q_est_mat[timestep_i, :] = Q[:, next_state]
 258           reward_est_mat[timestep_i, 0] = reward
 259         all_targets = Q_est_mat + np.cumsum(reward_est_mat, axis=0)
 260 
 261         # STEVE weight calculation
 262         estimates = np.mean(all_targets, axis=1)
 263         confidences = old_div(1., (np.var(all_targets, axis=1) + 1e-8))
 264         coefficients = old_div(confidences, np.sum(confidences))
 265         target = np.sum(estimates * coefficients)
 266 
 267         Q[q_ensemble_i, state] = target
 268 
 269         true_target = state + 1. if state != terminal_state else 0.
 270         _q_estimate_errors.append(np.square(estimates[0] - true_target))
 271         _mve_estimate_errors.append(np.square(estimates[-1] - true_target))
 272         _steve_estimate_errors.append(np.square(np.sum(estimates * coefficients) - true_target))
 273         _opt_estimate_errors.append(np.min(np.square(estimates - true_target)))
 274 
 275       losses.append(get_error(np.mean(Q, axis=0)))
 276       q_estimate_errors.append(np.mean(_q_estimate_errors))
 277       mve_estimate_errors.append(np.mean(_mve_estimate_errors))
 278       steve_estimate_errors.append(np.mean(_steve_estimate_errors))
 279       opt_estimate_errors.append(np.mean(_opt_estimate_errors))
 280     trial_results.append(losses)
 281     oracle_q_estimate_errors.append(q_estimate_errors)
 282     oracle_mve_estimate_errors.append(mve_estimate_errors)
 283     oracle_steve_estimate_errors.append(steve_estimate_errors)
 284     oracle_opt_estimate_errors.append(opt_estimate_errors)
 285   print("...complete.\n")
 286 
 287   result = np.stack(trial_results, axis=1)
 288   means = np.mean(result, axis=1)
 289   stdevs = np.std(result, axis=1)
 290   plt.plot(means, label="STEVE-oracle", color=colors[3])
 291   plt.fill_between(np.arange(TRAIN_STEPS), means - stdevs, means + stdevs, alpha=.2, color=colors[3])

 360         _opt_estimate_errors.append(np.min(np.square(estimates - true_target)))
 361         _steve_beat_freq.append(float(np.square(estimates[0] - true_target) > np.square(target - true_target)))
