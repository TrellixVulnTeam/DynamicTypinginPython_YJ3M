  16 import tensorflow as tf
  17 import numpy as np
  18 import os
  19 
  20 from learner import Learner
  21 from valuerl import ValueRL
  22 from worldmodel import DeterministicWorldModel
  23 
  24 class ValueRLLearner(Learner):
  25   """
  26   ValueRL-specific training loop details.
  27   """
  28 
  29   def learner_name(self): return "valuerl"
  30 
  31   def make_loader_placeholders(self):
  32     self.obs_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"], np.prod(self.env_config["obs_dims"])])
  33     self.next_obs_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"], np.prod(self.env_config["obs_dims"])])
  34     self.action_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"], self.env_config["action_dim"]])
  35     self.reward_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"]])
  36     self.done_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"]])
  37     self.datasize_loader = tf.placeholder(tf.float64, [])
  38     return [self.obs_loader, self.next_obs_loader, self.action_loader, self.reward_loader, self.done_loader, self.datasize_loader]
  39 
  40   def make_core_model(self):
  41     if self.config["model_config"] is not False:
  42         self.worldmodel = DeterministicWorldModel(self.config["name"], self.env_config, self.config["model_config"])
  43     else:
  44         self.worldmodel = None
  45 
  46     valuerl = ValueRL(self.config["name"], self.env_config, self.learner_config)
  47     (policy_loss, Q_loss), inspect_losses = valuerl.build_training_graph(*self.current_batch, worldmodel=self.worldmodel)
  48 
  49     policy_optimizer = tf.train.AdamOptimizer(3e-4)
  50     policy_gvs = policy_optimizer.compute_gradients(policy_loss, var_list=valuerl.policy_params)
  51     capped_policy_gvs = policy_gvs
  52     policy_train_op = policy_optimizer.apply_gradients(capped_policy_gvs)
  53 
  54     Q_optimizer = tf.train.AdamOptimizer(3e-4)
  55     Q_gvs = Q_optimizer.compute_gradients(Q_loss, var_list=valuerl.Q_params)
  56     capped_Q_gvs = Q_gvs
  57     Q_train_op = Q_optimizer.apply_gradients(capped_Q_gvs)
  58 
  59     return valuerl, (policy_loss, Q_loss), (policy_train_op, Q_train_op), inspect_losses
  60 
  61   ## Optional functions to override
  62   def initialize(self):
  63       if self.config["model_config"] is not False:
  64           while not self.load_worldmodel(): pass
  65 
  66   def resume_from_checkpoint(self, epoch):
  67       if self.config["model_config"] is not False:
  68           with self.bonus_kwargs["model_lock"]: self.worldmodel.load(self.sess, self.save_path, epoch)
  69 
  70   def checkpoint(self):
  71       self.core.copy_to_old(self.sess)
  72       if self.config["model_config"] is not False:
  73           self.load_worldmodel()
  74 
  75   def backup(self): pass
  76 
  77   # Other functions
  78   def load_worldmodel(self):
  79       if not os.path.exists("%s/%s.params.index" % (self.save_path, self.worldmodel.saveid)): return False
  80       with self.bonus_kwargs["model_lock"]: self.worldmodel.load(self.sess, self.save_path)
  81       return True
