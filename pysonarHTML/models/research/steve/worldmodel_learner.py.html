  16 import tensorflow as tf
  17 import numpy as np
  18 from learner import Learner
  19 from worldmodel import DeterministicWorldModel
  20 
  21 class WorldmodelLearner(Learner):
  22     """
  23     Worldmodel-specific training loop details.
  24     """
  25     def learner_name(self): return "worldmodel"
  26 
  27     def make_loader_placeholders(self):
  28         self.obs_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"], np.prod(self.env_config["obs_dims"])])
  29         self.next_obs_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"], np.prod(self.env_config["obs_dims"])])
  30         self.action_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"], self.env_config["action_dim"]])
  31         self.reward_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"]])
  32         self.done_loader = tf.placeholder(tf.float32, [self.learner_config["batch_size"]])
  33         self.datasize_loader = tf.placeholder(tf.float64, [])
  34         return [self.obs_loader, self.next_obs_loader, self.action_loader, self.reward_loader, self.done_loader, self.datasize_loader]
  35 
  36     def make_core_model(self):
  37         worldmodel = DeterministicWorldModel(self.config["name"], self.env_config, self.learner_config)
  38         worldmodel_loss, inspect_losses = worldmodel.build_training_graph(*self.current_batch)
  39 
  40         model_optimizer = tf.train.AdamOptimizer(3e-4)
  41         model_gvs = model_optimizer.compute_gradients(worldmodel_loss, var_list=worldmodel.model_params)
  42         capped_model_gvs = model_gvs
  43         worldmodel_train_op = model_optimizer.apply_gradients(capped_model_gvs)
  44 
  45         return worldmodel, (worldmodel_loss,), (worldmodel_train_op,), inspect_losses
  46 
  47     ## Optional functions to override
  48     def initialize(self): pass
  49     def resume_from_checkpoint(self, epoch): pass
  50     def checkpoint(self): pass
  51     def backup(self): pass
