  30 def get_env(env_name, *args, **kwargs):
  31   MAPPING = {
  32     "CartPole-v0": CartPoleWrapper,
  33   }
  34   if env_name in MAPPING: return MAPPING[env_name](env_name, *args, **kwargs)
  35   else: return NoTimeLimitMujocoWrapper(env_name, *args, **kwargs)

  37 class GymWrapper(object):
  38   """
  39   Generic wrapper for OpenAI gym environments.
  40   """
  41   def __init__(self, env_name):
  42     self.internal_env = gym.make(env_name)
  43     self.observation_space = self.internal_env.observation_space
  44     self.action_space = self.internal_env.action_space
  45     self.custom_init()
  46 
  47   def custom_init(self):
  48     pass
  49 
  50   def reset(self):
  51     self.clock = 0
  52     return self.preprocess_obs(self.internal_env.reset())
  53 
  54   # returns normalized actions
  55   def sample(self):
  56     return self.action_space.sample()
  57 
  58   # this is used for converting continuous approximations back to the original domain
  59   def normalize_actions(self, actions):
  60     return actions
  61 
  62   # puts actions into a form where they can be predicted. by default, called after sample()
  63   def unnormalize_actions(self, actions):
  64     return actions
  65 
  66   def preprocess_obs(self, obs):
  67     # return np.append(obs, [self.clock/float(MAX_FRAMES)])
  68     return obs
  69 
  70   def step(self, normalized_action):
  71     out = self.internal_env.step(normalized_action)
  72     self.clock += 1
  73     obs, reward, done = self.preprocess_obs(out[0]), out[1], float(out[2])
  74     reset = done == 1. or self.clock == MAX_FRAMES
  75     return obs, reward, done, reset
  76 
  77   def render_rollout(self, states):
  78     ## states is numpy array of size [timesteps, state]
  79     self.internal_env.reset()
  80     for state in states:
  81       self.internal_env.env.state = state
  82       self.internal_env.render()
  83 
  84 class CartPoleWrapper(GymWrapper):
  85   """
  86   Wrap CartPole.
  87   """
  88   def sample(self):
  89     return np.array([np.random.uniform(0., 1.)])
  90 
  91   def normalize_actions(self, action):
  92     return 1 if action[0] >= 0 else 0
  93 
  94   def unnormalize_actions(self, action):
  95     return 2. * action - 1.

  97 class NoTimeLimitMujocoWrapper(GymWrapper):
  98   """
  99   Wrap Mujoco-style environments, removing the termination condition after time.
 100   This is needed to keep it Markovian.
 101   """
 102   def __init__(self, env_name):
 103     self.internal_env = gym.make(env_name).env
 104     self.observation_space = self.internal_env.observation_space
 105     self.action_space = self.internal_env.action_space
 106     self.custom_init()
