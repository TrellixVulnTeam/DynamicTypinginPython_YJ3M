  22 import numpy as np
  23 import pickle
  24 import multiprocessing
  25 
  26 class ReplayBuffer(object):
  27     """
  28     Stores frames sampled from the environment, with the ability to sample a batch
  29     for training.
  30     """
  31 
  32     def __init__(self, max_size, obs_dim, action_dim, roundrobin=True):
  33         self.max_size = max_size
  34         self.obs_dim = obs_dim
  35         self.action_dim = action_dim
  36         self.roundrobin = roundrobin
  37 
  38         self.obs_buffer = np.zeros([max_size, obs_dim])
  39         self.next_obs_buffer = np.zeros([max_size, obs_dim])
  40         self.action_buffer = np.zeros([max_size, action_dim])
  41         self.reward_buffer = np.zeros([max_size])
  42         self.done_buffer = np.zeros([max_size])
  43 
  44         self.count = 0
  45 
  46     def random_batch(self, batch_size):
  47         indices = np.random.randint(0, min(self.count, self.max_size), batch_size)
  48 
  49         return (
  50             self.obs_buffer[indices],
  51             self.next_obs_buffer[indices],
  52             self.action_buffer[indices],
  53             self.reward_buffer[indices],
  54             self.done_buffer[indices],
  55             self.count
  56         )
  57 
  58     def add_replay(self, obs, next_obs, action, reward, done):
  59         if self.count >= self.max_size:
  60             if self.roundrobin: index = self.count % self.max_size
  61             else:               index = np.random.randint(0, self.max_size)
  62         else:
  63             index = self.count
  64 
  65         self.obs_buffer[index] = obs
  66         self.next_obs_buffer[index] = next_obs
  67         self.action_buffer[index] = action
  68         self.reward_buffer[index] = reward
  69         self.done_buffer[index] = done
  70 
  71         self.count += 1
  72 
  73     def save(self, path, name):
  74         def _save(datas, fnames):
  75             print("saving replay buffer...")
  76             for data, fname in zip(datas, fnames):
  77                 with open("%s.npz"%fname, "wb") as f:
  78                     pickle.dump(data, f)
  79             with open("%s/%s.count" % (path,name), "wb") as f:
  80                 f.write(str(self.count))
  81             print("...done saving.")
  82 
  83         datas = [
  84             self.obs_buffer,
  85             self.next_obs_buffer,
  86             self.action_buffer,
  87             self.reward_buffer,
  88             self.done_buffer
  89         ]
  90 
  91         fnames = [
  92             "%s/%s.obs_buffer" % (path, name),
  93             "%s/%s.next_obs_buffer" % (path, name),
  94             "%s/%s.action_buffer" % (path, name),
  95             "%s/%s.reward_buffer" % (path, name),
  96             "%s/%s.done_buffer" % (path, name)
  97          ]
  98 
  99         proc = multiprocessing.Process(target=_save, args=(datas, fnames))
 100         proc.start()
 101 
 102     def load(self, path, name):
 103         print("Loading %s replay buffer (may take a while...)" % name)
 104         with open("%s/%s.obs_buffer.npz" % (path,name)) as f: self.obs_buffer = pickle.load(f)
 105         with open("%s/%s.next_obs_buffer.npz" % (path,name)) as f: self.next_obs_buffer = pickle.load(f)
 106         with open("%s/%s.action_buffer.npz" % (path,name)) as f: self.action_buffer = pickle.load(f)
 107         with open("%s/%s.reward_buffer.npz" % (path,name)) as f: self.reward_buffer = pickle.load(f)
 108         with open("%s/%s.done_buffer.npz" % (path,name)) as f: self.done_buffer = pickle.load(f)
 109         with open("%s/%s.count" % (path,name), "r") as f: self.count = int(f.read())
