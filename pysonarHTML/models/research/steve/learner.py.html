  28 class Learner(object):
  29     """
  30     Generic object which runs the main training loop of anything that trains using
  31     a replay buffer. Handles updating, logging, saving/loading, batching, etc.
  32     """
  33     def __init__(self, interactor_queue, lock, config, env_config, learner_config, **bonus_kwargs):
  34         self.learner_name = self.learner_name()
  35         self.interactor_queue = interactor_queue
  36         self.learner_lock = lock
  37         self.config = config
  38         self.env_config = env_config
  39         self.learner_config = learner_config
  40         self.bonus_kwargs = bonus_kwargs
  41         self.kill_threads = False
  42         self.permit_desync = False
  43         self.need_frames_notification = threading.Condition()
  44         self._reset_inspections()
  45         self.total_frames = 0
  46 
  47         self.save_path = util.create_directory("%s/%s/%s/%s" % (self.config["output_root"], self.config["env"]["name"], self.config["name"], self.config["save_model_path"]))
  48         self.log_path = util.create_directory("%s/%s/%s/%s" % (self.config["output_root"], self.config["env"]["name"], self.config["name"],  self.config["log_path"])) + "/%s.log" % self.learner_name
  49 
  50         # replay buffer to store data
  51         self.replay_buffer_lock = threading.RLock()
  52         self.replay_buffer = ReplayBuffer(self.learner_config["replay_size"],
  53                                           np.prod(self.env_config["obs_dims"]),
  54                                           self.env_config["action_dim"])
  55 
  56         # data loaders pull data from the replay buffer and put it into the tfqueue for model usage
  57         self.data_loaders = self.make_loader_placeholders()
  58         queue_capacity = np.ceil(1./self.learner_config["frames_per_update"]) if self.learner_config["frames_per_update"] else 100
  59         self.tf_queue = tf.FIFOQueue(capacity=queue_capacity, dtypes=[dl.dtype for dl in self.data_loaders])
  60         self.enqueue_op = self.tf_queue.enqueue(self.data_loaders)
  61         self.current_batch = self.tf_queue.dequeue()
  62 
  63         # build the TF graph for the actual model to train
  64         self.core, self.train_losses, self.train_ops, self.inspect_losses = self.make_core_model()
  65         self.sess = tf.Session()
  66         self.sess.run(tf.global_variables_initializer())

  79     ## Internal functions
  80     def _start(self):
  81         # fetch data from the interactors to pre-fill the replay buffer
  82         self.prefetch_thread = threading.Thread(target=self._poll_interactors, args=(True, self.learner_config["frames_before_learning"],))
  83         self.prefetch_thread.start()
  84         self.prefetch_thread.join()
  85 
  86         # start the interactor and data loader
  87         self.data_load_thread = threading.Thread(target=self._run_enqueue_data)
  88         self.data_load_thread.start()
  89 
  90         # initialize the learner, pretraining if needed
  91         if self.config["resume"]: self._resume_from_checkpoint()
  92         else:                     self._initialize()
  93 
  94         # re-sync everything, and start up interactions with the environment
  95         self.interactor_poll_thread = threading.Thread(target=self._poll_interactors)
  96         self.interactor_poll_thread.start()
  97 
  98         # start the clock
  99         self._last_checkpoint_time = time.time()
 100 
 101     def _learn(self, permit_desync=False, log=True, checkpoint=True, backup=True):
 102         # this is to keep the frames/update synced properly
 103         if self.learner_config["frames_per_update"] is not False and not permit_desync:
 104             if not self._have_enough_frames():
 105                 with self.need_frames_notification:
 106                     self.need_frames_notification.notify()
 107                 return
 108 
 109         # log
 110         if log and (self.update_i + 1) % self.learner_config["log_every_n"] == 0:
 111             self._log()
 112 
 113         # checkpoint
 114         if checkpoint and (self.update_i + 1) % self.learner_config["epoch_every_n"] == 0:
 115             self._checkpoint()
 116 
 117         # backup
 118         if backup and (self.update_i + 1) % self.learner_config["backup_every_n"] == 0:
 119             self._backup()
 120 
 121         # train
 122         self._training_step()
 123 
 124     def _have_enough_frames(self):
 125         gathered_frames = self.total_frames - self.learner_config["frames_before_learning"]
 126         return gathered_frames > self.learner_config["frames_per_update"] * self.update_i
 127 
 128     def _initialize(self):
 129         self.epoch = 0
 130         self.update_i = 0
 131         self.hours = 0
 132         self._last_checkpoint_time = time.time()
 133 
 134         self.initialize()
 135 
 136         if self.learner_config["pretrain_n"]: self._pretrain()
 137         self._checkpoint()
 138 
 139     def _pretrain(self):
 140         for _ in range(self.learner_config["pretrain_n"]):
 141             self._learn(permit_desync=True, checkpoint=False, backup=False)
 142         self.epoch = 0
 143         self.update_i = 0
 144 
 145     def _resume_from_checkpoint(self):
 146         epoch = util.get_largest_epoch_in_dir(self.save_path, self.core.saveid)
 147         if not self.config['keep_all_replay_buffers']: util.wipe_all_but_largest_epoch_in_dir(self.save_path, self.core.saveid)
 148         if epoch is False:
 149             raise Exception("Tried to reload but no model found")
 150         with self.learner_lock:
 151             self.core.load(self.sess, self.save_path, epoch)
 152             self.epoch, self.update_i, self.total_frames, self.hours = self.sess.run([self.core.epoch_n, self.core.update_n, self.core.frame_n, self.core.hours])
 153         with self.replay_buffer_lock:
 154             self.replay_buffer.load(self.save_path, '%09d_%s' % (epoch, self.learner_name))
 155         self.resume_from_checkpoint(epoch)
 156 
 157     def _log(self):
 158         if self.denom > 0:
 159             logstring = "(%3.2f sec) h%-8.2f e%-8d s%-8d f%-8d\t" % (time.time() - self._log_time, self.hours, self.epoch, self.update_i + 1, self.total_frames) + ', '.join(["%8f" % x for x in (self.running_total / self.denom).tolist()])
 160             print("%s\t%s" % (self.learner_name, logstring))
 161             with open(self.log_path, "a") as f: f.write(logstring + "\n")
 162         self._reset_inspections()
 163 
 164     def _reset_inspections(self):
 165         self.running_total = 0.
 166         self.denom = 0.
 167         self._log_time = time.time()
 168 
 169     def _checkpoint(self):
 170         self.checkpoint()
 171         self.epoch += 1
 172         self.hours += (time.time() - self._last_checkpoint_time) / 3600.
 173         self._last_checkpoint_time = time.time()
 174         self.core.update_epoch(self.sess, self.epoch, self.update_i, self.total_frames, self.hours)
 175         with self.learner_lock: self.core.save(self.sess, self.save_path)
 176 
 177     def _backup(self):
 178         self.backup()
 179         if not self.learner_config['keep_all_replay_buffers']: util.wipe_all_but_largest_epoch_in_dir(self.save_path, self.core.saveid)
 180         with self.learner_lock:
 181             self.core.save(self.sess, self.save_path, self.epoch)
 182         with self.replay_buffer_lock:
 183             self.replay_buffer.save(self.save_path, '%09d_%s' % (self.epoch, self.learner_name))
 184 
 185     def _training_step(self):
 186         train_ops = tuple([op for op, loss in zip(self.train_ops,
 187                                                   self.train_losses)
 188                            if loss is not None])
 189         outs = self.sess.run(train_ops + self.inspect_losses)
 190         self.running_total += np.array(outs[len(train_ops):])
 191         self.denom += 1.
 192         self.update_i += 1
 193 
 194     def _poll_interactors(self, continuous_poll=False, frames_before_terminate=None):
 195         # poll the interactors for new frames.
 196         # the synced_condition semaphore prevents this from consuming too much CPU
 197         while not self.kill_threads:
 198             if self.learner_config["frames_per_update"] is not False and not continuous_poll:
 199                 with self.need_frames_notification: self.need_frames_notification.wait()
 200             while not self.interactor_queue.empty():
 201                 new_frames = self.interactor_queue.get()
 202                 self._add_frames(new_frames)
 203                 if frames_before_terminate and self.total_frames >= frames_before_terminate: return
 204 
 205     def _add_frames(self, frames):
 206         with self.replay_buffer_lock:
 207             for frame in frames:
 208                 self.replay_buffer.add_replay(*frame)
 209             self.total_frames = self.replay_buffer.count
 210         return self.total_frames
 211 
 212     def _run_enqueue_data(self):
 213         while not self.kill_threads:
 214             data = self.replay_buffer.random_batch(self.learner_config["batch_size"])
 215             self.sess.run(self.enqueue_op, feed_dict=dict(list(zip(self.data_loaders, data))))
 216 
 217     def _kill_threads(self):
 218         self.kill_threads = True

 221 class CoreModel(object):
 222     """The base class for the "core" of learners."""
 223     def __init__(self, name, env_config, learner_config):
 224         self.name = self.saveid + "/" + name
 225         self.env_config = env_config
 226         self.learner_config = learner_config
 227 
 228         with tf.variable_scope(self.name):
 229             self.epoch_n = tf.get_variable('epoch_n', [], initializer=tf.constant_initializer(0), dtype=tf.int64, trainable=False)
 230             self.update_n = tf.get_variable('update_n', [], initializer=tf.constant_initializer(0), dtype=tf.int64, trainable=False)
 231             self.frame_n = tf.get_variable('frame_n', [], initializer=tf.constant_initializer(0), dtype=tf.int64, trainable=False)
 232             self.hours = tf.get_variable('hours', [], initializer=tf.constant_initializer(0.), dtype=tf.float64, trainable=False)
 233             self.epoch_n_placeholder = tf.placeholder(tf.int64, [])
 234             self.update_n_placeholder = tf.placeholder(tf.int64, [])
 235             self.frame_n_placeholder = tf.placeholder(tf.int64, [])
 236             self.hours_placeholder = tf.placeholder(tf.float64, [])
 237         self.assign_epoch_op = [tf.assign(self.epoch_n, self.epoch_n_placeholder), tf.assign(self.update_n, self.update_n_placeholder), tf.assign(self.frame_n, self.frame_n_placeholder), tf.assign(self.hours, self.hours_placeholder)]
 238 
 239         self.create_params(env_config, learner_config)
 240         self.model_params = tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES, scope=self.name)
 241         self.saver = tf.train.Saver(self.model_params)
 242 
 243     @property
 244     def saveid(self):
 245         raise Exception("specify a save ID")

 247     def create_params(self, env_config, learner_config):
 248         raise Exception("unimplemented")

 250     def update_epoch(self, sess, epoch, updates, frames, hours):
 251         sess.run(self.assign_epoch_op, feed_dict={self.epoch_n_placeholder: int(epoch), self.update_n_placeholder: int(updates), self.frame_n_placeholder: int(frames), self.hours_placeholder: float(hours)})
 252 
 253     def save(self, sess, path, epoch=None):
 254         if epoch is None:  self.saver.save(sess, path + "/%s.params" % self.saveid)
 255         else:              self.saver.save(sess, path + "/%09d_%s.params" % (epoch, self.saveid))
 256 
 257     def load(self, sess, path, epoch=None):
 258         if epoch is None:  self.saver.restore(sess, path + "/%s.params" % self.saveid)
 259         else:              self.saver.restore(sess, path + "/%09d_%s.params" % (epoch, self.saveid))

 261 def run_learner(learner_subclass, queue, lock, config, env_config, learner_config, **bonus_kwargs):
 262     learner = learner_subclass(queue, lock, config, env_config, learner_config, **bonus_kwargs)
 263     try:
 264         learner._start()
 265         while True: learner._learn()
 266 
 267     except Exception as e:
 268         print('Caught exception in learner process')
 269         traceback.print_exc()
 270         learner._kill_threads()
 271         print()
 272         raise e
