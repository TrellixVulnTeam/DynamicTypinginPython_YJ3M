  26 def _print_debug_ios(history, goal, output):
  27   """Prints sizes of history, goal and outputs."""
  28   if history is not None:
  29     shape = history.get_shape().as_list()
  30     # logging.info('history embedding shape ')
  31     # logging.info(shape)
  32   if len(shape) != 3:
  33       raise ValueError('history Tensor must have rank=3')

  35      logging.info('goal embedding shape ')
  36      logging.info(goal.get_shape().as_list())
  37   if output is not None:
  38      logging.info('targets shape ')
  39      logging.info(output.get_shape().as_list())

  42 class Policy(object):
  43   """Represents the policy of the agent for navigation tasks.
  44 
  45   Instantiates a policy that takes embedders for each modality and builds a
  46   model to infer the actions.
  47   """
  48   __metaclass__ = abc.ABCMeta
  49 
  50   def __init__(self, embedders_dict, action_size):
  51     """Instantiates the policy.
  52 
  53     Args:
  54       embedders_dict: Dictionary of embedders for different modalities. Keys
  55         should be identical to keys of observation modality.
  56       action_size: Number of possible actions.
  57     """
  58     self._embedders = embedders_dict
  59     self._action_size = action_size
  60 
  61   @abc.abstractmethod
  62   def build(self, observations, prev_state):
  63     """Builds the model that represents the policy of the agent.
  64 
  65     Args:
  66       observations: Dictionary of observations from different modalities. Keys
  67         are the name of the modalities.
  68       prev_state: The tensor of the previous state of the model. Should be set
  69         to None if the policy is stateless
  70     Returns:
  71       Tuple of (action, state) where action is the action logits and state is
  72       the state of the model after taking new observation.
  73     """
  74     raise NotImplementedError(
  75         'Needs implementation as part of Policy interface')

  78 class LSTMPolicy(Policy):
  79   """Represents the implementation of the LSTM based policy.
  80 
  81   The architecture of the model is as follows. It embeds all the observations
  82   using the embedders, concatenates the embeddings of all the modalities. Feed
  83   them through two fully connected layers. The lstm takes the features from
  84   fully connected layer and the previous action and success of previous action
  85   and feed them to LSTM. The value for each action is predicted afterwards.
  86   Although the class name has the word LSTM in it, it also supports a mode that
  87   builds the network without LSTM just for comparison purposes.
  88   """
  89 
  90   def __init__(self,
  91                modality_names,
  92                embedders_dict,
  93                action_size,
  94                params,
  95                max_episode_length,
  96                feedforward_mode=False):
  97     """Instantiates the LSTM policy.
  98 
  99     Args:
 100       modality_names: List of modality names. Makes sure the ordering in
 101         concatenation remains the same as modality_names list. Each modality
 102         needs to be in the embedders_dict.
 103       embedders_dict: Dictionary of embedders for different modalities. Keys
 104         should be identical to keys of observation modality. Values should be
 105         instance of Embedder class. All the observations except PREV_ACTION
 106         requires embedder.
 107       action_size: Number of possible actions.
 108       params: is instance of tf.hparams and contains the hyperparameters for the
 109         policy network.
 110       max_episode_length: integer, specifying the maximum length of each
 111         episode.
 112       feedforward_mode: If True, it does not add LSTM to the model. It should
 113         only be set True for comparison between LSTM and feedforward models.
 114     """
 115     super(LSTMPolicy, self).__init__(embedders_dict, action_size)
 116 
 117     self._modality_names = modality_names
 118 
 119     self._lstm_state_size = params.lstm_state_size
 120     self._fc_channels = params.fc_channels
 121     self._weight_decay = params.weight_decay
 122     self._target_embedding_size = params.target_embedding_size
 123     self._max_episode_length = max_episode_length
 124     self._feedforward_mode = feedforward_mode
 125 
 126   def _build_lstm(self, encoded_inputs, prev_state, episode_length,
 127                   prev_action=None):
 128     """Builds an LSTM on top of the encoded inputs.
 129 
 130     If prev_action is not None then it concatenates them to the input of LSTM.
 131 
 132     Args:
 133       encoded_inputs: The embedding of the observations and goal.
 134       prev_state: previous state of LSTM.
 135       episode_length: The tensor that contains the length of the sequence for
 136         each element of the batch.
 137       prev_action: tensor to previous chosen action and additional bit for
 138         indicating whether the previous action was successful or not.
 139 
 140     Returns:
 141       a tuple of (lstm output, lstm state).
 142     """
 143 
 144     # Adding prev action and success in addition to the embeddings of the
 145     # modalities.
 146     if prev_action is not None:
 147       encoded_inputs = tf.concat([encoded_inputs, prev_action], axis=-1)
 148 
 149     with tf.variable_scope('LSTM'):
 150       lstm_cell = tf.nn.rnn_cell.BasicLSTMCell(self._lstm_state_size)
 151       if prev_state is None:
 152         # If prev state is set to None, a state of all zeros will be
 153         # passed as a previous value for the cell. Should be used for the
 154         # first step of each episode.
 155         tf_prev_state = lstm_cell.zero_state(
 156             encoded_inputs.get_shape().as_list()[0], dtype=tf.float32)
 157       else:
 158         tf_prev_state = tf.nn.rnn_cell.LSTMStateTuple(prev_state[0],
 159                                                       prev_state[1])
 160 
 161       lstm_outputs, lstm_state = tf.nn.dynamic_rnn(
 162           cell=lstm_cell,
 163           inputs=encoded_inputs,
 164           sequence_length=episode_length,
 165           initial_state=tf_prev_state,
 166           dtype=tf.float32,
 167       )
 168     lstm_outputs = tf.reshape(lstm_outputs, [-1, lstm_cell.output_size])
 169     return lstm_outputs, lstm_state
 170 
 171   def build(
 172       self,
 173       observations,
 174       prev_state,
 175   ):
 176     """Builds the model that represents the policy of the agent.
 177 
 178     Args:
 179       observations: Dictionary of observations from different modalities. Keys
 180         are the name of the modalities. Observation should have the following
 181         key-values.
 182           observations['goal']: One-hot tensor that indicates the semantic
 183             category of the goal. The shape should be
 184             (batch_size x max_sequence_length x goals).
 185           observations[task_env.ModalityTypes.PREV_ACTION]: has action_size + 1
 186             elements where the first action_size numbers are the one hot vector
 187             of the previous action and the last element indicates whether the
 188             previous action was successful or not. If
 189             task_env.ModalityTypes.PREV_ACTION is not in the observation, it
 190             will not be used in the policy.
 191       prev_state: Previous state of the model. It should be a tuple of (c,h)
 192         where c and h are the previous cell value and hidden state of the lstm.
 193         Each element of tuple has shape of (batch_size x lstm_cell_size).
 194         If it is set to None, then it initializes the state of the lstm with all
 195         zeros.
 196 
 197     Returns:
 198       Tuple of (action, state) where action is the action logits and state is
 199       the state of the model after taking new observation.
 200     Raises:
 201       ValueError: If any of the modality names is not in observations or
 202         embedders_dict.
 203       ValueError: If 'goal' is not in the observations.
 204     """
 205 
 206     for modality_name in self._modality_names:
 207       if modality_name not in observations:
 208         raise ValueError('modality name does not exist in observations: {} not '
 209                          'in {}'.format(modality_name, observations.keys()))
 210       if modality_name not in self._embedders:
 211         if modality_name == task_env.ModalityTypes.PREV_ACTION:
 212           continue
 213         raise ValueError('modality name does not have corresponding embedder'
 214                          ' {} not in {}'.format(modality_name,
 215                                                 self._embedders.keys()))
 216 
 217     if task_env.ModalityTypes.GOAL not in observations:
 218       raise ValueError('goal should be provided in the observations')
 219 
 220     goal = observations[task_env.ModalityTypes.GOAL]
 221     prev_action = None
 222     if task_env.ModalityTypes.PREV_ACTION in observations:
 223       prev_action = observations[task_env.ModalityTypes.PREV_ACTION]
 224 
 225     with tf.variable_scope('policy'):
 226       with slim.arg_scope(
 227           [slim.fully_connected],
 228           activation_fn=tf.nn.relu,
 229           weights_initializer=tf.truncated_normal_initializer(stddev=0.01),
 230           weights_regularizer=slim.l2_regularizer(self._weight_decay)):
 231         all_inputs = []
 232 
 233         # Concatenating the embedding of each modality by applying the embedders
 234         # to corresponding observations.
 235         def embed(name):
 236           with tf.variable_scope('embed_{}'.format(name)):
 237             # logging.info('Policy uses embedding %s', name)
 238             return self._embedders[name].build(observations[name])
 239 
 240         all_inputs = map(embed, [
 241             x for x in self._modality_names
 242             if x != task_env.ModalityTypes.PREV_ACTION
 243         ])
 244 
 245         # Computing goal embedding.
 246         shape = goal.get_shape().as_list()
 247         with tf.variable_scope('embed_goal'):
 248           encoded_goal = tf.reshape(goal, [shape[0] * shape[1], -1])
 249           encoded_goal = slim.fully_connected(encoded_goal,
 250                                               self._target_embedding_size)
 251           encoded_goal = tf.reshape(encoded_goal, [shape[0], shape[1], -1])
 252           all_inputs.append(encoded_goal)
 253 
 254         # Concatenating all the modalities and goal.
 255         all_inputs = tf.concat(all_inputs, axis=-1, name='concat_embeddings')
 256 
 257         shape = all_inputs.get_shape().as_list()
 258         all_inputs = tf.reshape(all_inputs, [shape[0] * shape[1], shape[2]])
 259 
 260         # Applying fully connected layers.
 261         encoded_inputs = slim.fully_connected(all_inputs, self._fc_channels)
 262         encoded_inputs = slim.fully_connected(encoded_inputs, self._fc_channels)
 263 
 264         if not self._feedforward_mode:
 265           encoded_inputs = tf.reshape(encoded_inputs,
 266                                       [shape[0], shape[1], self._fc_channels])
 267           lstm_outputs, lstm_state = self._build_lstm(
 268               encoded_inputs=encoded_inputs,
 269               prev_state=prev_state,
 270               episode_length=tf.ones((shape[0],), dtype=tf.float32) *
 271               self._max_episode_length,
 272               prev_action=prev_action,
 273           )
 274         else:
 275           # If feedforward_mode=True, directly compute bypass the whole LSTM
 276           # computations.
 277           lstm_outputs = encoded_inputs
 278 
 279         lstm_outputs = slim.fully_connected(lstm_outputs, self._fc_channels)
 280         action_values = slim.fully_connected(
 281             lstm_outputs, self._action_size, activation_fn=None)
 282         action_values = tf.reshape(action_values, [shape[0], shape[1], -1])
 283         if not self._feedforward_mode:
 284           return action_values, lstm_state
 285         else:
 286           return action_values, None

 289 class TaskPolicy(Policy):
 290   """A covenience abstract class providing functionality to deal with Tasks."""
 291 
 292   def __init__(self,
 293                task_config,
 294                model_hparams=None,
 295                embedder_hparams=None,
 296                train_hparams=None):
 297     """Constructs a policy which knows how to work with tasks (see tasks.py).
 298 
 299     It allows to read task history, goal and outputs in consistency with the
 300     task config.
 301 
 302     Args:
 303       task_config: an object of type tasks.TaskIOConfig (see tasks.py)
 304       model_hparams: a tf.HParams object containing parameter pertaining to
 305         model (these are implementation specific)
 306       embedder_hparams: a tf.HParams object containing parameter pertaining to
 307         history, goal embedders (these are implementation specific)
 308       train_hparams: a tf.HParams object containing parameter pertaining to
 309         trainin (these are implementation specific)`
 310     """
 311     super(TaskPolicy, self).__init__(None, None)
 312     self._model_hparams = model_hparams
 313     self._embedder_hparams = embedder_hparams
 314     self._train_hparams = train_hparams
 315     self._task_config = task_config
 316     self._extra_train_ops = []
 317 
 318   @property
 319   def extra_train_ops(self):
 320     """Training ops in addition to the loss, e.g. batch norm updates.
 321 
 322     Returns:
 323       A list of tf ops.
 324     """
 325     return self._extra_train_ops
 326 
 327   def _embed_task_ios(self, streams):
 328     """Embeds a list of heterogenous streams.
 329 
 330     These streams correspond to task history, goal and output. The number of
 331     streams is equal to the total number of history, plus one for the goal if
 332     present, plus one for the output. If the number of history is k, then the
 333     first k streams are the history.
 334 
 335     The used embedders depend on the input (or goal) types. If an input is an
 336     image, then a ResNet embedder is used, otherwise
 337     MLPEmbedder (see embedders.py).
 338 
 339     Args:
 340       streams: a list of Tensors.
 341     Returns:
 342       Three float Tensors history, goal, output. If there are no history, or no
 343       goal, then the corresponding returned values are None. The shape of the
 344       embedded history is batch_size x sequence_length x sum of all embedding
 345       dimensions for all history. The shape of the goal is embedding dimension.
 346     """
 347     # EMBED history.
 348     index = 0
 349     inps = []
 350     scopes = []
 351     for c in self._task_config.inputs:
 352       if c == task_env.ModalityTypes.IMAGE:
 353         scope_name = 'image_embedder/image'
 354         reuse = scope_name in scopes
 355         scopes.append(scope_name)
 356         with tf.variable_scope(scope_name, reuse=reuse):
 357           resnet_embedder = embedders.ResNet(self._embedder_hparams.image)
 358           image_embeddings = resnet_embedder.build(streams[index])
 359           # Uncover batch norm ops.
 360           if self._embedder_hparams.image.is_train:
 361             self._extra_train_ops += resnet_embedder.extra_train_ops
 362           inps.append(image_embeddings)
 363           index += 1
 364       else:
 365         scope_name = 'input_embedder/vector'
 366         reuse = scope_name in scopes
 367         scopes.append(scope_name)
 368         with tf.variable_scope(scope_name, reuse=reuse):
 369           input_vector_embedder = embedders.MLPEmbedder(
 370               layers=self._embedder_hparams.vector)
 371           vector_embedder = input_vector_embedder.build(streams[index])
 372           inps.append(vector_embedder)
 373           index += 1
 374     history = tf.concat(inps, axis=2) if inps else None
 375 
 376     # EMBED goal.
 377     goal = None
 378     if self._task_config.query is not None:
 379       scope_name = 'image_embedder/query'
 380       reuse = scope_name in scopes
 381       scopes.append(scope_name)
 382       with tf.variable_scope(scope_name, reuse=reuse):
 383         resnet_goal_embedder = embedders.ResNet(self._embedder_hparams.goal)
 384         goal = resnet_goal_embedder.build(streams[index])
 385         if self._embedder_hparams.goal.is_train:
 386           self._extra_train_ops += resnet_goal_embedder.extra_train_ops
 387         index += 1
 388 
 389     # Embed true targets if needed (tbd).
 390     true_target = streams[index]
 391 
 392     return history, goal, true_target

 399 class ReactivePolicy(TaskPolicy):
 400   """A policy which ignores history.
 401 
 402   It processes only the current observation (last element in history) and the
 403   goal to output a prediction.
 404   """
 405 
 406   def __init__(self, *args, **kwargs):
 407     super(ReactivePolicy, self).__init__(*args, **kwargs)
 408 
 409   # The current implementation ignores the prev_state as it is purely reactive.
 410   # It returns None for the current state.
 411   def build(self, feeds, prev_state):
 412     history, goal, _ = self._embed_task_ios(feeds)
 413     _print_debug_ios(history, goal, None)
 414 
 415     with tf.variable_scope('output_decoder'):
 416       # Concatenate the embeddings of the current observation and the goal.
 417       reactive_input = tf.concat([tf.squeeze(history[:, -1, :]), goal], axis=1)
 418       oconfig = self._task_config.output.shape
 419       assert len(oconfig) == 1
 420       decoder = embedders.MLPEmbedder(
 421           layers=self._embedder_hparams.predictions.layer_sizes + oconfig)
 422       predictions = decoder.build(reactive_input)
 423 
 424     return predictions, None

 427 class RNNPolicy(TaskPolicy):
 428   """A policy which takes into account the full history via RNN.
 429 
 430   The implementation might and will change.
 431   The history, together with the goal, is processed using a stacked LSTM. The
 432   output of the last LSTM step is used to produce a prediction. Currently, only
 433   a single step output is supported.
 434   """
 435 
 436   def __init__(self, lstm_hparams, *args, **kwargs):
 437     super(RNNPolicy, self).__init__(*args, **kwargs)
 438     self._lstm_hparams = lstm_hparams
 439 
 440   # The prev_state is ignored as for now the full history is specified as first
 441   # element of the feeds. It might turn out to be beneficial to keep the state
 442   # as part of the policy object.
 443   def build(self, feeds, state):
 444     history, goal, _ = self._embed_task_ios(feeds)
 445     _print_debug_ios(history, goal, None)
 446 
 447     params = self._lstm_hparams
 448     cell = lambda: tf.contrib.rnn.BasicLSTMCell(params.cell_size)
 449     stacked_lstm = tf.contrib.rnn.MultiRNNCell(
 450         [cell() for _ in range(params.num_layers)])
 451     # history is of shape batch_size x seq_len x embedding_dimension
 452     batch_size, seq_len, _ = tuple(history.get_shape().as_list())
 453 
 454     if state is None:
 455       state = stacked_lstm.zero_state(batch_size, tf.float32)
 456     for t in range(seq_len):
 457       if params.concat_goal_everywhere:
 458         lstm_input = tf.concat([tf.squeeze(history[:, t, :]), goal], axis=1)
 459       else:
 460         lstm_input = tf.squeeze(history[:, t, :])
 461       output, state = stacked_lstm(lstm_input, state)
 462 
 463     with tf.variable_scope('output_decoder'):
 464       oconfig = self._task_config.output.shape
 465       assert len(oconfig) == 1
 466       features = tf.concat([output, goal], axis=1)
 467       assert len(output.get_shape().as_list()) == 2
 468       assert len(goal.get_shape().as_list()) == 2
 469       decoder = embedders.MLPEmbedder(
 470           layers=self._embedder_hparams.predictions.layer_sizes + oconfig)
 471       # Prediction is done off the last step lstm output and the goal.
 472       predictions = decoder.build(features)
 473 
 474     return predictions, state
