  79 def benchmark(env, targets):
  80   """Benchmarks the speed of sequence generation by env.
  81 
  82   Args:
  83     env: environment.
  84     targets: list of target classes.
  85   """
  86   episode_lengths = {}
  87   all_init_configs = {}
  88   all_actions = dict([(a, 0.) for a in env.actions])
  89   for i in range(FLAGS.benchmark_iter):
  90     path, actions, _, _ = env.random_step_sequence()
  91     selected_actions = np.argmax(actions, axis=-1)
  92     new_actions = dict([(a, 0.) for a in env.actions])
  93     for a in selected_actions:
  94       new_actions[env.actions[a]] += 1. / selected_actions.shape[0]
  95     for a in new_actions:
  96       all_actions[a] += new_actions[a] / FLAGS.benchmark_iter
  97     start_image_id, world, goal = env.get_init_config(path)
  98     print world
  99     if world not in all_init_configs:
 100       all_init_configs[world] = set()
 101     all_init_configs[world].add((start_image_id, goal, len(actions)))
 102     if env.goal_index not in episode_lengths:
 103       episode_lengths[env.goal_index] = []
 104     episode_lengths[env.goal_index].append(len(actions))
 105   for i, cls in enumerate(episode_lengths):
 106     plt.subplot(231 + i)
 107     plt.hist(episode_lengths[cls])
 108     plt.title(targets[cls])
 109   plt.show()

 112 def human(env, targets):
 113   """Lets user play around the env manually."""
 114   string_key_map = {
 115       'a': 'left',
 116       'd': 'right',
 117       'w': 'forward',
 118       's': 'backward',
 119       'j': 'rotate_ccw',
 120       'l': 'rotate_cw',
 121       'n': 'stop'
 122   }
 123   integer_key_map = {
 124       'a': env.actions.index('left'),
 125       'd': env.actions.index('right'),
 126       'w': env.actions.index('forward'),
 127       's': env.actions.index('backward'),
 128       'j': env.actions.index('rotate_ccw'),
 129       'l': env.actions.index('rotate_cw'),
 130       'n': env.actions.index('stop')
 131   }
 132   for k in integer_key_map:
 133     integer_key_map[k] = np.int32(integer_key_map[k])
 134   plt.ion()
 135   for _ in range(20):
 136     obs = env.reset()
 137     steps = -1
 138     action = None
 139     while True:
 140       print 'distance = ', obs[task_env.ModalityTypes.DISTANCE]
 141       steps += 1
 142       depth_value = obs[task_env.ModalityTypes.DEPTH][:, :, 0]
 143       depth_mask = obs[task_env.ModalityTypes.DEPTH][:, :, 1]
 144       seg_mask = np.squeeze(obs[task_env.ModalityTypes.SEMANTIC_SEGMENTATION])
 145       det_mask = np.argmax(
 146           obs[task_env.ModalityTypes.OBJECT_DETECTION], axis=-1)
 147       img = obs[task_env.ModalityTypes.IMAGE]
 148       plt.subplot(231)
 149       plt.title('steps = {}'.format(steps))
 150       plt.imshow(img.astype(np.uint8))
 151       plt.subplot(232)
 152       plt.imshow(depth_value)
 153       plt.title('depth value')
 154       plt.subplot(233)
 155       plt.imshow(depth_mask)
 156       plt.title('depth mask')
 157       plt.subplot(234)
 158       plt.imshow(seg_mask)
 159       plt.title('seg')
 160       plt.subplot(235)
 161       plt.imshow(det_mask)
 162       plt.title('det')
 163       plt.subplot(236)
 164       plt.title('goal={}'.format(targets[env.goal_index]))
 165       plt.draw()
 166       while True:
 167         s = raw_input('key = ')
 168         if np.random.rand() > 0.5:
 169           key_map = string_key_map
 170         else:
 171           key_map = integer_key_map
 172         if s in key_map:
 173           action = key_map[s]
 174           break
 175         else:
 176           print 'invalid action'
 177       print 'action = {}'.format(action)
 178       if action == 'stop':
 179         print 'dist to goal: {}'.format(len(env.path_to_goal()) - 2)
 180         break
 181       obs, reward, done, info = env.step(action)
 182       print 'reward = {}, done = {}, success = {}'.format(
 183           reward, done, info['success'])

 223 def visualize(env, input_folder, output_root_folder):
 224   """visualizes images for sequence of steps from the evals folder."""
 225   def which_env(file_name):
 226     img_name = file_name.split('_')[0][2:5]
 227     env_dict = {'161': 'Home_016_1', '131': 'Home_013_1', '111': 'Home_011_1'}
 228     if img_name in env_dict:
 229       return env_dict[img_name]
 230     else:
 231       raise ValueError('could not resolve env: {} {}'.format(
 232           img_name, file_name))
 233 
 234   def which_goal(file_name):
 235     return file_name[file_name.find('_')+1:]
 236 
 237   output_images_folder = os.path.join(output_root_folder, 'images')
 238   output_gifs_folder = os.path.join(output_root_folder, 'gifs')
 239   if not tf.gfile.IsDirectory(output_images_folder):
 240     tf.gfile.MakeDirs(output_images_folder)
 241   if not tf.gfile.IsDirectory(output_gifs_folder):
 242     tf.gfile.MakeDirs(output_gifs_folder)
 243   npy_files = [
 244       os.path.join(input_folder, name)
 245       for name in tf.gfile.ListDirectory(input_folder)
 246       if name.find('npy') >= 0
 247   ]
 248   for i, npy_file in enumerate(npy_files):
 249     print 'saving images {}/{}'.format(i, len(npy_files))
 250     pure_name = npy_file[npy_file.rfind('/') + 1:-4]
 251     output_folder = os.path.join(output_images_folder, pure_name)
 252     if not tf.gfile.IsDirectory(output_folder):
 253       tf.gfile.MakeDirs(output_folder)
 254     print '*******'
 255     print pure_name[0:pure_name.find('_')]
 256     env.reset_for_eval(which_env(pure_name),
 257                        which_goal(pure_name),
 258                        pure_name[0:pure_name.find('_')],
 259                       )
 260     with tf.gfile.Open(npy_file) as h:
 261       states = np.load(h).item()['states']
 262       images = [
 263           env.observation(state)[mt.IMAGE] for state in states
 264       ]
 265       for j, img in enumerate(images):
 266         cv2.imwrite(os.path.join(output_folder, '{0:03d}'.format(j) + '.jpg'),
 267                     img[:, :, ::-1])
 268       print 'converting to gif'
 269       os.system(
 270           'convert -set delay 20 -colors 256 -dispose 1 {}/*.jpg {}.gif'.format(
 271               output_folder,
 272               os.path.join(output_gifs_folder, pure_name + '.gif')
 273           )
 274       )

 276 def evaluate_folder(env, folder_path):
 277   """Evaluates the performance from the evals folder."""
 278   targets = ['fridge', 'dining_table', 'microwave', 'tv', 'couch']
 279 
 280   def compute_acc(npy_file):
 281     with tf.gfile.Open(npy_file) as h:
 282       data = np.load(h).item()
 283     if npy_file.find('dining_table') >= 0:
 284       category = 'dining_table'
 285     else:
 286       category = npy_file[npy_file.rfind('_') + 1:-4]
 287     return category, data['distance'][-1] - 2
 288 
 289   def evaluate_iteration(folder):
 290     """Evaluates the data from the folder of certain eval iteration."""
 291     print folder
 292     npy_files = [
 293         os.path.join(folder, name)
 294         for name in tf.gfile.ListDirectory(folder)
 295         if name.find('npy') >= 0
 296     ]
 297     eval_stats = {c: [] for c in targets}
 298     for npy_file in npy_files:
 299       try:
 300         category, dist = compute_acc(npy_file)
 301       except:  # pylint: disable=bare-except
 302         continue
 303       eval_stats[category].append(float(dist <= 5))
 304     for c in eval_stats:
 305       if not eval_stats[c]:
 306         print 'incomplete eval {}: empty class {}'.format(folder_path, c)
 307         return None
 308       eval_stats[c] = np.mean(eval_stats[c])
 309 
 310     eval_stats['mean'] = np.mean(eval_stats.values())
 311     return eval_stats
 312 
 313   checkpoint_folders = [
 314       folder_path + x
 315       for x in tf.gfile.ListDirectory(folder_path)
 316       if tf.gfile.IsDirectory(folder_path + x)
 317   ]
 318 
 319   print '{} folders found'.format(len(checkpoint_folders))
 320   print '------------------------'
 321   all_iters = []
 322   all_accs = []
 323   for i, folder in enumerate(checkpoint_folders):
 324     print 'processing {}/{}'.format(i, len(checkpoint_folders))
 325     eval_stats = evaluate_iteration(folder)
 326     if eval_stats is None:
 327       continue
 328     else:
 329       iter_no = int(folder[folder.rfind('/') + 1:])
 330       print 'result ', iter_no, eval_stats['mean']
 331       all_accs.append(eval_stats['mean'])
 332       all_iters.append(iter_no)
 333 
 334   all_accs = np.asarray(all_accs)
 335   all_iters = np.asarray(all_iters)
 336   idx = np.argmax(all_accs)
 337   print 'best result at iteration {} was {}'.format(all_iters[idx],
 338                                                     all_accs[idx])
 339   order = np.argsort(all_iters)
 340   all_iters = all_iters[order]
 341   all_accs = all_accs[order]
 342   #plt.plot(all_iters, all_accs)
 343   #plt.show()
 344   #print 'done plotting'
 345 
 346   best_iteration_folder = os.path.join(folder_path, str(all_iters[idx]))
 347 
 348   print 'generating gifs and images for {}'.format(best_iteration_folder)
 349   visualize(env, best_iteration_folder, FLAGS.output_folder)

 352 def main(_):
 353   gin.parse_config_files_and_bindings(FLAGS.gin_config, FLAGS.gin_params)
 354   print('********')
 355   print(FLAGS.mode)
 356   print(FLAGS.gin_config)
 357   print(FLAGS.gin_params)
 358 
 359   env = active_vision_dataset_env.ActiveVisionDatasetEnv(modality_types=[
 360       task_env.ModalityTypes.IMAGE,
 361       task_env.ModalityTypes.SEMANTIC_SEGMENTATION,
 362       task_env.ModalityTypes.OBJECT_DETECTION, task_env.ModalityTypes.DEPTH,
 363       task_env.ModalityTypes.DISTANCE
 364   ])
 365 
 366   if FLAGS.mode == BENCHMARK_MODE:
 367     benchmark(env, env.possible_targets)
 368   elif FLAGS.mode == GRAPH_MODE:
 369     for loc in env.worlds:
 370       env.check_scene_graph(loc, 'fridge')
 371   elif FLAGS.mode == HUMAN_MODE:
 372     human(env, env.possible_targets)
 373   elif FLAGS.mode == VIS_MODE:
 374     visualize_random_step_sequence(env)
 375   elif FLAGS.mode == EVAL_MODE:
 376     evaluate_folder(env, FLAGS.eval_folder)
 377 
 378 if __name__ == '__main__':
 379   app.run(main)
