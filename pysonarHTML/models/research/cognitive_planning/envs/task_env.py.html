  52 class TaskEnvInterface(object):
  53   """Interface for an environment topology.
  54 
  55   An environment can implement this interface if there is a topological graph
  56   underlying this environment. All paths below are defined as paths in this
  57   graph. Using path_to_actions function one can translate a topological path
  58   to a geometric path in the environment.
  59   """
  60 
  61   __metaclass__ = abc.ABCMeta
  62 
  63   @abc.abstractmethod
  64   def random_step_sequence(self, min_len=None, max_len=None):
  65     """Generates a random sequence of actions and executes them.
  66 
  67     Args:
  68       min_len: integer, minimum length of a step sequence.
  69       max_len: integer, if it is set to non-None, the method returns only
  70         the first n steps of a random sequence. If the environment is
  71         computationally heavy this argument should be set to speed up the
  72         training and avoid unnecessary computations by the environment.
  73 
  74     Returns:
  75       A path, defined as a list of vertex indices, a list of actions, a list of
  76       states, and a list of step() return tuples.
  77     """
  78     raise NotImplementedError(
  79         'Needs implementation as part of EnvTopology interface.')
  80 
  81   @abc.abstractmethod
  82   def targets(self):
  83     """A list of targets in the environment.
  84 
  85     Returns:
  86       A list of target locations.
  87     """
  88     raise NotImplementedError(
  89         'Needs implementation as part of EnvTopology interface.')
  90 
  91   @abc.abstractproperty
  92   def state(self):
  93     """Returns the position for the current location of agent."""
  94     raise NotImplementedError(
  95         'Needs implementation as part of EnvTopology interface.')
  96 
  97   @abc.abstractproperty
  98   def graph(self):
  99     """Returns a graph representing the environment topology.
 100 
 101     Returns:
 102       nx.Graph object.
 103     """
 104     raise NotImplementedError(
 105         'Needs implementation as part of EnvTopology interface.')
 106 
 107   @abc.abstractmethod
 108   def vertex_to_pose(self, vertex_index):
 109     """Maps a vertex index to a pose in the environment.
 110 
 111     Pose of the camera can be represented by (x,y,theta) or (x,y,z,theta).
 112     Args:
 113       vertex_index: index of a vertex in the topology graph.
 114 
 115     Returns:
 116       A np.array of floats of size 3 or 4 representing the pose of the vertex.
 117     """
 118     raise NotImplementedError(
 119         'Needs implementation as part of EnvTopology interface.')
 120 
 121   @abc.abstractmethod
 122   def pose_to_vertex(self, pose):
 123     """Maps a coordinate in the maze to the closest vertex in topology graph.
 124 
 125     Args:
 126       pose: np.array of floats containing a the pose of the view.
 127 
 128     Returns:
 129       index of a vertex.
 130     """
 131     raise NotImplementedError(
 132         'Needs implementation as part of EnvTopology interface.')
 133 
 134   @abc.abstractmethod
 135   def observation(self, state):
 136     """Returns observation at location xy and orientation theta.
 137 
 138     Args:
 139       state: a np.array of floats containing coordinates of a location and
 140         orientation.
 141 
 142     Returns:
 143       Dictionary of observations in the case of multiple observations.
 144       The keys are the modality names and the values are the np.array of float
 145       of observations for corresponding modality.
 146     """
 147     raise NotImplementedError(
 148         'Needs implementation as part of EnvTopology interface.')
 149 
 150   def action(self, init_state, final_state):
 151     """Computes the transition action from state1 to state2.
 152 
 153     If the environment is discrete and the views are not adjacent in the
 154     environment. i.e. it is not possible to move from the first view to the
 155     second view with one action it should return None. In the continuous case,
 156     it will be the continuous difference of first view and second view.
 157 
 158     Args:
 159       init_state: numpy array, the initial view of the agent.
 160       final_state: numpy array, the final view of the agent.
 161     """
 162     raise NotImplementedError(
 163         'Needs implementation as part of EnvTopology interface.')

 166 @gin.configurable
 167 class TaskEnv(gym.Env, TaskEnvInterface):
 168   """An environment which uses a Task to compute reward.
 169 
 170   The environment implements a a gym interface, as well as EnvTopology. The
 171   former makes sure it can be used within an RL training, while the latter
 172   makes sure it can be used by a Task.
 173 
 174   This environment requires _step_no_reward to be implemented, which steps
 175   through it but does not return reward. Instead, the reward calculation is
 176   delegated to the Task object, which in return can access needed properties
 177   of the environment. These properties are exposed via the EnvTopology
 178   interface.
 179   """
 180 
 181   def __init__(self, task=None):
 182     self._task = task
 183 
 184   def set_task(self, task):
 185     self._task = task
 186 
 187   @abc.abstractmethod
 188   def _step_no_reward(self, action):
 189     """Same as _step without returning reward.
 190 
 191     Args:
 192       action: see _step.
 193 
 194     Returns:
 195       state, done, info as defined in _step.
 196     """
 197     raise NotImplementedError('Implement step.')
 198 
 199   @abc.abstractmethod
 200   def _reset_env(self):
 201     """Resets the environment. Returns initial observation."""
 202     raise NotImplementedError('Implement _reset. Must call super!')
 203 
 204   def step(self, action):
 205     obs, done, info = self._step_no_reward(action)
 206 
 207     reward = 0.0
 208     if self._task is not None:
 209       obs, reward, done, info = self._task.reward(obs, done, info)
 210 
 211     return obs, reward, done, info
 212 
 213   def reset(self):
 214     """Resets the environment. Gym API."""
 215     obs = self._reset_env()
 216     if self._task is not None:
 217       self._task.reset(obs)
 218     return obs
