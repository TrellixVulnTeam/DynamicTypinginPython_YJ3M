 113 def generate_detection_image(detections,
 114                              image_size,
 115                              category_map,
 116                              num_classes,
 117                              is_binary=True):
 118   """Generates one_hot vector of the image using the detection boxes.
 119 
 120   Args:
 121     detections: 2D object detections from the image. It's a dictionary that
 122       contains detection_boxes, detection_classes, and detection_scores with
 123       dimensions of nx4, nx1, nx1 where n is the number of detections.
 124     image_size: The resolution of the output image.
 125     category_map: dictionary that maps label names to index.
 126     num_classes: Number of classes.
 127     is_binary: If true, it sets the corresponding channels to 0 and 1.
 128       Otherwise, sets the score in the corresponding channel.
 129   Returns:
 130     Returns image_size x image_size x num_classes image for the detection boxes.
 131   """
 132   res = np.zeros((image_size, image_size, num_classes), dtype=np.float32)
 133   boxes = detections['detection_boxes']
 134   labels = detections['detection_classes']
 135   scores = detections['detection_scores']
 136   for box, label, score in zip(boxes, labels, scores):
 137     transformed_boxes = [int(round(t)) for t in box * image_size]
 138     y1, x1, y2, x2 = transformed_boxes
 139     # Detector returns fixed number of detections. Boxes with area of zero
 140     # are equivalent of boxes that don't correspond to any detection box.
 141     # So, we need to skip the boxes with area 0.
 142     if (y2 - y1) * (x2 - x1) == 0:
 143       continue
 144     assert category_map[label] < num_classes, 'label = {}'.format(label)
 145     value = score
 146     if is_binary:
 147       value = 1
 148     res[y1:y2, x1:x2, category_map[label]] = value
 149   return res

 186 def read_all_poses(dataset_root, world):
 187   """Reads all the poses for each world.
 188 
 189   Args:
 190     dataset_root: the path to the root of the dataset.
 191     world: string, name of the world.
 192 
 193   Returns:
 194     Dictionary of poses for all the images in each world. The key is the image
 195     id of each view and the values are tuple of (x, z, R, scale). Where x and z
 196     are the first and third coordinate of translation. R is the 3x3 rotation
 197     matrix and scale is a float scalar that indicates the scale that needs to
 198     be multipled to x and z in order to get the real world coordinates.
 199 
 200   Raises:
 201     ValueError: if the number of images do not match the number of poses read.
 202   """
 203   path = os.path.join(dataset_root, world, 'image_structs.mat')
 204   with tf.gfile.Open(path) as f:
 205     data = sio.loadmat(f)
 206   xyz = data['image_structs']['world_pos']
 207   image_names = data['image_structs']['image_name'][0]
 208   rot = data['image_structs']['R'][0]
 209   scale = data['scale'][0][0]
 210   n = xyz.shape[1]
 211   x = [xyz[0][i][0][0] for i in range(n)]
 212   z = [xyz[0][i][2][0] for i in range(n)]
 213   names = [name[0][:-4] for name in image_names]
 214   if len(names) != len(x):
 215     raise ValueError('number of image names are not equal to the number of '
 216                      'poses {} != {}'.format(len(names), len(x)))
 217   output = {}
 218   for i in range(n):
 219     if rot[i].shape[0] != 0:
 220       assert rot[i].shape[0] == 3
 221       assert rot[i].shape[1] == 3
 222       output[names[i]] = (x[i], z[i], rot[i], scale)
 223     else:
 224       output[names[i]] = (x[i], z[i], None, scale)
 225 
 226   return output

 229 def read_cached_data(should_load_images, dataset_root, segmentation_file_name,
 230                      targets_file_name, output_size):
 231   """Reads all the necessary cached data.
 232 
 233   Args:
 234     should_load_images: whether to load the images or not.
 235     dataset_root: path to the root of the dataset.
 236     segmentation_file_name: The name of the file that contains semantic
 237       segmentation annotations.
 238     targets_file_name: The name of the file the contains targets annotated for
 239       each world.
 240     output_size: Size of the output images. This is used for pre-processing the
 241       loaded images.
 242   Returns:
 243     Dictionary of all the cached data.
 244   """
 245 
 246   load_start = time.time()
 247   result_data = {}
 248 
 249   annotated_target_path = os.path.join(dataset_root, 'Meta',
 250                                        targets_file_name + '.npy')
 251 
 252   logging.info('loading targets: %s', annotated_target_path)
 253   with tf.gfile.Open(annotated_target_path) as f:
 254     result_data['targets'] = np.load(f).item()
 255 
 256   depth_image_path = os.path.join(dataset_root, 'Meta/depth_imgs.npy')
 257   logging.info('loading depth: %s', depth_image_path)
 258   with tf.gfile.Open(depth_image_path) as f:
 259     depth_data = np.load(f).item()
 260 
 261   logging.info('processing depth')
 262   for home_id in depth_data:
 263     images = depth_data[home_id]
 264     for image_id in images:
 265       depth = images[image_id]
 266       depth = cv2.resize(
 267           depth / _MAX_DEPTH_VALUE, (output_size, output_size),
 268           interpolation=cv2.INTER_NEAREST)
 269       depth_mask = (depth > 0).astype(np.float32)
 270       depth = np.dstack((depth, depth_mask))
 271       images[image_id] = depth
 272   result_data[task_env.ModalityTypes.DEPTH] = depth_data
 273 
 274   sseg_path = os.path.join(dataset_root, 'Meta',
 275                            segmentation_file_name + '.npy')
 276   logging.info('loading sseg: %s', sseg_path)
 277   with tf.gfile.Open(sseg_path) as f:
 278     sseg_data = np.load(f).item()
 279 
 280   logging.info('processing sseg')
 281   for home_id in sseg_data:
 282     images = sseg_data[home_id]
 283     for image_id in images:
 284       sseg = images[image_id]
 285       sseg = cv2.resize(
 286           sseg, (output_size, output_size), interpolation=cv2.INTER_NEAREST)
 287       images[image_id] = np.expand_dims(sseg, axis=-1).astype(np.float32)
 288   result_data[task_env.ModalityTypes.SEMANTIC_SEGMENTATION] = sseg_data
 289 
 290   if should_load_images:
 291     image_path = os.path.join(dataset_root, 'Meta/imgs.npy')
 292     logging.info('loading imgs: %s', image_path)
 293     with tf.gfile.Open(image_path) as f:
 294       image_data = np.load(f).item()
 295 
 296     result_data[task_env.ModalityTypes.IMAGE] = image_data
 297 
 298   with tf.gfile.Open(os.path.join(dataset_root, 'Meta/world_id_dict.npy')) as f:
 299     result_data['world_id_dict'] = np.load(f).item()
 300 
 301   logging.info('logging done in %f seconds', time.time() - load_start)
 302   return result_data

 315   def __init__(
 316       self,
 317       episode_length,
 318       modality_types,
 319       confidence_threshold,
 320       output_size,
 321       worlds,
 322       targets,
 323       compute_distance,
 324       should_draw_detections,
 325       dataset_root,
 326       labelmap_path,
 327       reward_collision,
 328       reward_goal_range,
 329       num_detection_classes,
 330       segmentation_file_name,
 331       detection_folder_name,
 332       actions,
 333       targets_file_name,
 334       eval_init_points_file_name=None,
 335       shaped_reward=False,
 336   ):
 337     """Instantiates the environment for ActiveVision Dataset.
 338 
 339     Args:
 340       episode_length: the length of each episode.
 341       modality_types: a list of the strings where each entry indicates the name
 342         of the modalities to be loaded. Valid entries are "sseg", "det",
 343         "depth", "image", "distance", and "prev_action". "distance" should be
 344         used for computing metrics in tf agents.
 345       confidence_threshold: Consider detections more than confidence_threshold
 346         for potential targets.
 347       output_size: Resolution of the output image.
 348       worlds: List of the name of the worlds.
 349       targets: List of the target names. Each entry is a string label of the
 350         target category (e.g. 'fridge', 'microwave', so on).
 351       compute_distance: If True, outputs the distance of the view to the goal.
 352       should_draw_detections (bool): If True, the image returned for the
 353         observation will contains the bounding boxes.
 354       dataset_root: the path to the root folder of the dataset.
 355       labelmap_path: path to the dictionary that converts label strings to
 356         indexes.
 357       reward_collision: the reward the agents get after hitting an obstacle.
 358         It should be a non-positive number.
 359       reward_goal_range: the number of steps from goal, such that the agent is
 360         considered to have reached the goal. If the agent's distance is less
 361         than the specified goal range, the episode is also finishes by setting
 362         done = True.
 363       num_detection_classes: number of classes that detector outputs.
 364       segmentation_file_name: the name of the file that contains the semantic
 365         information. The file should be in the dataset_root/Meta/ folder.
 366       detection_folder_name: Name of the folder that contains the detections
 367         for each world. The folder should be under dataset_root/Meta/ folder.
 368       actions: The list of the action names. Valid entries are listed in
 369         SUPPORTED_ACTIONS.
 370       targets_file_name: the name of the file that contains the annotated
 371         targets. The file should be in the dataset_root/Meta/Folder
 372       eval_init_points_file_name: The name of the file that contains the initial
 373         points for evaluating the performance of the agent. If set to None,
 374         episodes start at random locations. Should be only set for evaluation.
 375       shaped_reward: Whether to add delta goal distance to the reward each step.
 376 
 377     Raises:
 378       ValueError: If one of the targets are not available in the annotated
 379         targets or the modality names are not from the domain specified above.
 380       ValueError: If one of the actions is not in SUPPORTED_ACTIONS.
 381       ValueError: If the reward_collision is a positive number.
 382       ValueError: If there is no action other than stop provided.
 383     """
 384     if reward_collision > 0:
 385       raise ValueError('"reward" for collision should be non positive')
 386 
 387     if reward_goal_range < 0:
 388       logging.warning('environment does not terminate the episode if the agent '
 389                       'is too close to the environment')
 390 
 391     if not modality_types:
 392       raise ValueError('modality names can not be empty')
 393 
 394     for name in modality_types:
 395       if name not in SUPPORTED_MODALITIES:
 396         raise ValueError('invalid modality type: {}'.format(name))
 397 
 398     actions_other_than_stop_found = False
 399     for a in actions:
 400       if a != 'stop':
 401         actions_other_than_stop_found = True
 402       if a not in SUPPORTED_ACTIONS:
 403         raise ValueError('invalid action %s', a)
 404 
 405     if not actions_other_than_stop_found:
 406       raise ValueError('environment needs to have actions other than stop.')
 407 
 408     super(ActiveVisionDatasetEnv, self).__init__()
 409 
 410     self._episode_length = episode_length
 411     self._modality_types = set(modality_types)
 412     self._confidence_threshold = confidence_threshold
 413     self._output_size = output_size
 414     self._dataset_root = dataset_root
 415     self._worlds = worlds
 416     self._targets = targets
 417     self._all_graph = {}
 418     for world in self._worlds:
 419       with tf.gfile.Open(_get_json_path(self._dataset_root, world), 'r') as f:
 420         file_content = f.read()
 421         file_content = file_content.replace('.jpg', '')
 422         io = StringIO(file_content)
 423         self._all_graph[world] = json.load(io)
 424 
 425     self._cur_world = ''
 426     self._cur_image_id = ''
 427     self._cur_graph = None  # Loaded by _update_graph
 428     self._steps_taken = 0
 429     self._last_action_success = True
 430     self._category_index = _init_category_index(labelmap_path)
 431     self._category_map = dict(
 432         [(c, i) for i, c in enumerate(self._category_index)])
 433     self._detection_cache = {}
 434     if not ActiveVisionDatasetEnv.cached_data:
 435       ActiveVisionDatasetEnv.cached_data = read_cached_data(
 436           True, self._dataset_root, segmentation_file_name, targets_file_name,
 437           self._output_size)
 438     cached_data = ActiveVisionDatasetEnv.cached_data
 439 
 440     self._world_id_dict = cached_data['world_id_dict']
 441     self._depth_images = cached_data[task_env.ModalityTypes.DEPTH]
 442     self._semantic_segmentations = cached_data[
 443         task_env.ModalityTypes.SEMANTIC_SEGMENTATION]
 444     self._annotated_targets = cached_data['targets']
 445     self._cached_imgs = cached_data[task_env.ModalityTypes.IMAGE]
 446     self._graph_cache = {}
 447     self._compute_distance = compute_distance
 448     self._should_draw_detections = should_draw_detections
 449     self._reward_collision = reward_collision
 450     self._reward_goal_range = reward_goal_range
 451     self._num_detection_classes = num_detection_classes
 452     self._actions = actions
 453     self._detection_folder_name = detection_folder_name
 454     self._shaped_reward = shaped_reward
 455 
 456     self._eval_init_points = None
 457     if eval_init_points_file_name is not None:
 458       self._eval_init_index = 0
 459       init_points_path = os.path.join(self._dataset_root, 'Meta',
 460                                       eval_init_points_file_name + '.npy')
 461       with tf.gfile.Open(init_points_path) as points_file:
 462         data = np.load(points_file).item()
 463       self._eval_init_points = []
 464       for world in self._worlds:
 465         for goal in self._targets:
 466           if world in self._annotated_targets[goal]:
 467             for image_id in data[world]:
 468               self._eval_init_points.append((world, image_id[0], goal))
 469         logging.info('loaded %d eval init points', len(self._eval_init_points))
 470 
 471     self.action_space = gym.spaces.Discrete(len(self._actions))
 472 
 473     obs_shapes = {}
 474     if task_env.ModalityTypes.SEMANTIC_SEGMENTATION in self._modality_types:
 475       obs_shapes[task_env.ModalityTypes.SEMANTIC_SEGMENTATION] = gym.spaces.Box(
 476           low=0, high=255, shape=(self._output_size, self._output_size, 1))
 477     if task_env.ModalityTypes.OBJECT_DETECTION in self._modality_types:
 478       obs_shapes[task_env.ModalityTypes.OBJECT_DETECTION] = gym.spaces.Box(
 479           low=0,
 480           high=255,
 481           shape=(self._output_size, self._output_size,
 482                  self._num_detection_classes))
 483     if task_env.ModalityTypes.DEPTH in self._modality_types:
 484       obs_shapes[task_env.ModalityTypes.DEPTH] = gym.spaces.Box(
 485           low=0,
 486           high=_MAX_DEPTH_VALUE,
 487           shape=(self._output_size, self._output_size, 2))
 488     if task_env.ModalityTypes.IMAGE in self._modality_types:
 489       obs_shapes[task_env.ModalityTypes.IMAGE] = gym.spaces.Box(
 490           low=0, high=255, shape=(self._output_size, self._output_size, 3))
 491     if task_env.ModalityTypes.GOAL in self._modality_types:
 492       obs_shapes[task_env.ModalityTypes.GOAL] = gym.spaces.Box(
 493           low=0, high=1., shape=(len(self._targets),))
 494     if task_env.ModalityTypes.PREV_ACTION in self._modality_types:
 495       obs_shapes[task_env.ModalityTypes.PREV_ACTION] = gym.spaces.Box(
 496           low=0, high=1., shape=(len(self._actions) + 1,))
 497     if task_env.ModalityTypes.DISTANCE in self._modality_types:
 498       obs_shapes[task_env.ModalityTypes.DISTANCE] = gym.spaces.Box(
 499           low=0, high=255, shape=(1,))
 500     self.observation_space = gym.spaces.Dict(obs_shapes)
 501 
 502     self._prev_action = np.zeros((len(self._actions) + 1), dtype=np.float32)
 503 
 504     # Loading all the poses.
 505     all_poses = {}
 506     for world in self._worlds:
 507       all_poses[world] = read_all_poses(self._dataset_root, world)
 508     self._cached_poses = all_poses
 509     self._vertex_to_pose = {}
 510     self._pose_to_vertex = {}

 517   def _next_image(self, image_id, action):
 518     """Given the action, returns the name of the image that agent ends up in.
 519 
 520     Args:
 521       image_id: The image id of the current view.
 522       action: valid actions are ['right', 'rotate_cw', 'rotate_ccw',
 523       'forward', 'left']. Each rotation is 30 degrees.
 524 
 525     Returns:
 526       The image name for the next location of the agent. If the action results
 527       in collision or it is not possible for the agent to execute that action,
 528       returns empty string.
 529     """
 530     assert action in self._actions, 'invalid action : {}'.format(action)
 531     assert self._cur_world in self._all_graph, 'invalid world {}'.format(
 532         self._cur_world)
 533     assert image_id in self._all_graph[
 534         self._cur_world], 'image_id {} is not in {}'.format(
 535             image_id, self._cur_world)
 536     return self._all_graph[self._cur_world][image_id][action]

 538   def _largest_detection_for_image(self, image_id, detections_dict):
 539     """Assigns area of the largest box for the view with given image id.
 540 
 541     Args:
 542       image_id: Image id of the view.
 543       detections_dict: Detections for the view.
 544     """
 545     for cls, box, score in zip(detections_dict['detection_classes'],
 546                                detections_dict['detection_boxes'],
 547                                detections_dict['detection_scores']):
 548       if cls not in self._targets:
 549         continue
 550       if score < self._confidence_threshold:
 551         continue
 552       ymin, xmin, ymax, xmax = box
 553       area = (ymax - ymin) * (xmax - xmin)
 554       if abs(area) < 1e-5:
 555         continue
 556       if image_id not in self._detection_area:
 557         self._detection_area[image_id] = area
 558       else:
 559         self._detection_area[image_id] = max(self._detection_area[image_id],
 560                                              area)

 563     """Computes the goal indexes for the environment.
 564 
 565     Returns:
 566       The indexes of the goals that are closest to target categories. A vertex
 567       is goal vertice if the desired objects are detected in the image and the
 568       target categories are not seen by moving forward from that vertice.
 569     """
 570     for image_id in self._world_id_dict[self._cur_world]:
 571       detections_dict = self._detection_table[image_id]
 572       self._largest_detection_for_image(image_id, detections_dict)
 573     goal_indexes = []
 574     for image_id in self._world_id_dict[self._cur_world]:
 575       if image_id not in self._detection_area:
 576         continue
 577       # Detection box is large enough.
 578       if self._detection_area[image_id] < 0.01:
 579         continue
 580       ok = True
 581       next_image_id = self._next_image(image_id, 'forward')
 582       if next_image_id:
 583         if next_image_id in self._detection_area:
 584           ok = False
 585       if ok:
 586         goal_indexes.append(self._cur_graph.id_to_index[image_id])
 587     return goal_indexes

 604

 607 
 608     Returns:
 609       Observation at the given view point.

 624              SEMANTIC_SEGMENTATION] = self._semantic_segmentations[

 626

 628     need_det = (
 629         task_env.ModalityTypes.OBJECT_DETECTION in self._modality_types or
 630         (task_env.ModalityTypes.IMAGE in self._modality_types and
 631          self._should_draw_detections))
 632     if need_det:
 633       detection = self._detection_table[image_id]
 634       detection_image = generate_detection_image(
 635           detection,
 636           self._output_size,
 637           self._category_map,
 638           num_classes=self._num_detection_classes)
 639 
 640     if task_env.ModalityTypes.OBJECT_DETECTION in self._modality_types:
 641       output[task_env.ModalityTypes.OBJECT_DETECTION] = detection_image
 642 
 643     if task_env.ModalityTypes.DEPTH in self._modality_types:
 644       output[task_env.ModalityTypes.DEPTH] = self._depth_images[

 658 
 659     if task_env.ModalityTypes.PREV_ACTION in self._modality_types:
 660       output[task_env.ModalityTypes.PREV_ACTION] = self._prev_action
 661 
 662     if task_env.ModalityTypes.DISTANCE in self._modality_types:
 663       output[task_env.ModalityTypes.DISTANCE] = np.asarray(
 664           [self.gt_value(self._cur_goal, vertex)], dtype=np.float32)
 665

 668   def _step_no_reward(self, action):
 669     """Performs a step in the environment with given action.
 670 
 671     Args:
 672       action: Action that is used to step in the environment. Action can be
 673         string or integer. If the type is integer then it uses the ith element
 674         from self._actions list. Otherwise, uses the string value as the action.
 675 
 676     Returns:
 677       observation, done, info
 678       observation: dictonary that contains all the observations specified in
 679         modality_types.
 680         observation[task_env.ModalityTypes.OBJECT_DETECTION]: contains the
 681         detection of the current view.
 682         observation[task_env.ModalityTypes.IMAGE]: contains the
 683           image of the current view. Note that if using the images for training,
 684           should_load_images should be set to false.
 685         observation[task_env.ModalityTypes.SEMANTIC_SEGMENTATION]: contains the
 686           semantic segmentation of the current view.
 687         observation[task_env.ModalityTypes.DEPTH]: If selected, returns the
 688           depth map for the current view.
 689         observation[task_env.ModalityTypes.PREV_ACTION]: If selected, returns
 690           a numpy of (action_size + 1,). The first action_size elements indicate
 691           the action and the last element indicates whether the previous action
 692           was successful or not.
 693       done: True after episode_length steps have been taken, False otherwise.
 694       info: Empty dictionary.
 695 
 696     Raises:
 697       ValueError: for invalid actions.
 698     """
 699     # Primarily used for gym interface.
 700     if not isinstance(action, str):
 701       if not self.action_space.contains(action):
 702         raise ValueError('Not a valid actions: %d', action)
 703 
 704       action = self._actions[action]
 705 
 706     if action not in self._actions:
 707       raise ValueError('Not a valid action: %s', action)
 708 
 709     action_index = self._actions.index(action)
 710 
 711     if action == 'stop':
 712       next_image_id = self._cur_image_id
 713       done = True
 714       success = True
 715     else:
 716       next_image_id = self._next_image(self._cur_image_id, action)
 717       self._steps_taken += 1
 718       done = False
 719       success = True
 720     if not next_image_id:
 721       success = False
 722     else:
 723       self._cur_image_id = next_image_id
 724 
 725     if self._steps_taken >= self._episode_length:
 726       done = True
 727 
 728     cur_vertex = self._cur_graph.id_to_index[self._cur_image_id]
 729     observation = self.observation(self.vertex_to_pose(cur_vertex))
 730 
 731     # Concatenation of one-hot prev action + a binary number for success of
 732     # previous actions.
 733     self._prev_action = np.zeros((len(self._actions) + 1,), dtype=np.float32)
 734     self._prev_action[action_index] = 1.
 735     self._prev_action[-1] = float(success)
 736 
 737     distance_to_goal = self.gt_value(self._cur_goal, cur_vertex)
 738     if success:
 739       if distance_to_goal <= self._reward_goal_range:
 740         done = True
 741 
 742     return observation, done, {'success': success}

 766   def _update_graph(self):
 767     """Creates the graph for each environment and updates the _cur_graph."""
 768     if self._cur_world not in self._graph_cache:
 769       graph = nx.DiGraph()
 770       id_to_index = {}
 771       index_to_id = {}
 772       image_list = self._world_id_dict[self._cur_world]
 773       for i, image_id in enumerate(image_list):
 774         id_to_index[image_id] = i
 775         index_to_id[i] = image_id
 776         graph.add_node(i)
 777 
 778       for image_id in image_list:
 779         for action in self._actions:
 780           if action == 'stop':
 781             continue
 782           next_image = self._all_graph[self._cur_world][image_id][action]
 783           if next_image:
 784             graph.add_edge(
 785                 id_to_index[image_id], id_to_index[next_image], action=action)
 786       target_indexes = {}
 787       number_of_nodes_without_targets = graph.number_of_nodes()
 788       distance_to_goal = {}
 789       for goal in self._targets:
 790         if self._cur_world not in self._annotated_targets[goal]:
 791           continue
 792         goal_indexes = [
 793             id_to_index[i]
 794             for i in self._annotated_targets[goal][self._cur_world]
 795             if i
 796         ]
 797         super_source_index = graph.number_of_nodes()
 798         target_indexes[goal] = super_source_index
 799         graph.add_node(super_source_index)
 800         index_to_id[super_source_index] = goal
 801         id_to_index[goal] = super_source_index
 802         for v in goal_indexes:
 803           graph.add_edge(v, super_source_index, action='stop')
 804           graph.add_edge(super_source_index, v, action='stop')
 805         distance_to_goal[goal] = {}
 806         for v in range(number_of_nodes_without_targets):
 807           distance_to_goal[goal][v] = len(
 808               nx.shortest_path(graph, v, super_source_index)) - 2
 809 
 810       self._graph_cache[self._cur_world] = _Graph(
 811           graph, id_to_index, index_to_id, target_indexes, distance_to_goal)
 812     self._cur_graph = self._graph_cache[self._cur_world]

 829   def _reset_env(
 830       self,
 831       new_world=None,
 832       new_goal=None,
 833       new_image_id=None,
 834   ):
 835     """Resets the agent in a random world and random id.
 836 
 837     Args:
 838       new_world: If not None, sets the new world to new_world.
 839       new_goal: If not None, sets the new goal to new_goal.
 840       new_image_id: If not None, sets the first image id to new_image_id.
 841 
 842     Returns:
 843       observation: dictionary of the observations. Content of the observation
 844       is similar to that of the step function.
 845     Raises:
 846       ValueError: if it can't find a world and annotated goal.
 847     """
 848     self._steps_taken = 0
 849     # The first prev_action is special all zero vector + success=1.
 850     self._prev_action = np.zeros((len(self._actions) + 1,), dtype=np.float32)
 851     self._prev_action[len(self._actions)] = 1.
 852     if self._eval_init_points is not None:
 853       if self._eval_init_index >= len(self._eval_init_points):
 854         self._eval_init_index = 0
 855       a = self._eval_init_points[self._eval_init_index]
 856       self._cur_world, self._cur_image_id, self._cur_goal = a
 857       self._eval_init_index += 1
 858     elif not new_world:
 859       attempts = 100
 860       found = False
 861       while attempts >= 0:
 862         attempts -= 1
 863         self._cur_goal = np.random.choice(self._targets)
 864         available_worlds = list(
 865             set(self._annotated_targets[self._cur_goal].keys()).intersection(
 866                 set(self._worlds)))
 867         if available_worlds:
 868           found = True
 869           break
 870       if not found:
 871         raise ValueError('could not find a world that has a target annotated')
 872       self._cur_world = np.random.choice(available_worlds)
 873     else:
 874       self._cur_world = new_world
 875       self._cur_goal = new_goal
 876       if new_world not in self._annotated_targets[new_goal]:
 877         return None
 878 
 879     self._cur_goal_index = self._targets.index(self._cur_goal)
 880     if new_image_id:
 881       self._cur_image_id = new_image_id
 882     else:
 883       self._cur_image_id = np.random.choice(
 884           self._world_id_dict[self._cur_world])
 885     if self._cur_world not in self._detection_cache:
 886       with tf.gfile.Open(
 887           _get_detection_path(self._dataset_root, self._detection_folder_name,
 888                               self._cur_world)) as f:
 889         # Each file contains a dictionary with image ids as keys and detection
 890         # dicts as values.
 891         self._detection_cache[self._cur_world] = np.load(f).item()
 892     self._detection_table = self._detection_cache[self._cur_world]
 893     self._detection_area = {}
 894     self._update_graph()
 895     if self._cur_world not in self._vertex_to_pose:
 896       # adding fake pose for the super nodes of each target categories.
 897       self._vertex_to_pose[self._cur_world] = {
 898           index: (-index,) for index in self._cur_graph.target_indexes.values()
 899       }
 900       # Calling vetex_to_pose for each vertex results in filling out the
 901       # dictionaries that contain pose related data.
 902       for image_id in self._world_id_dict[self._cur_world]:
 903         self.vertex_to_pose(self.to_vertex(image_id))
 904 
 905       # Filling out pose_to_vertex from vertex_to_pose.
 906       self._pose_to_vertex[self._cur_world] = {
 907           tuple(v): k
 908           for k, v in self._vertex_to_pose[self._cur_world].iteritems()
 909       }
 910 
 911     cur_vertex = self._cur_graph.id_to_index[self._cur_image_id]
 912     observation = self.observation(self.vertex_to_pose(cur_vertex))
 913     return observation

 947   def vertex_to_pose(self, v):
 948     """Returns pose of the view for a given vertex.
 949 
 950     Args:
 951       v: integer, vertex index.
 952 
 953     Returns:
 954       (x, z, dir_x, dir_z) where x and z are the tranlation and dir_x, dir_z are
 955         a vector giving direction of the view.
 956     """
 957     if v in self._vertex_to_pose[self._cur_world]:
 958       return np.copy(self._vertex_to_pose[self._cur_world][v])
 959 
 960     x, z, rot, scale = self._cached_poses[self._cur_world][self.to_image_id(
 961         v)]
 962     if rot is None:  # if rotation is not provided for the given vertex.
 963       self._vertex_to_pose[self._cur_world][v] = np.asarray(
 964           [x * scale, z * scale, v])
 965       return np.copy(self._vertex_to_pose[self._cur_world][v])
 966     # Multiply rotation matrix by [0,0,1] to get a vector of length 1 in the
 967     # direction of the ray.
 968     direction = np.zeros((3, 1), dtype=np.float32)
 969     direction[2][0] = 1
 970     direction = np.matmul(np.transpose(rot), direction)
 971     direction = [direction[0][0], direction[2][0]]
 972     self._vertex_to_pose[self._cur_world][v] = np.asarray(
 973         [x * scale, z * scale, direction[0], direction[1]])
 974     return np.copy(self._vertex_to_pose[self._cur_world][v])

1053   def random_step_sequence(self, min_len=None, max_len=None):
1054     """Generates random step sequence that takes agent to the goal.
1055 
1056     Args:
1057       min_len: integer, minimum length of a step sequence. Not yet implemented.
1058       max_len: integer, should be set to an integer and it is the maximum number
1059         of observations and path length to be max_len.
1060     Returns:
1061       Tuple of (path, actions, states, step_outputs).
1062         path: a random path from a random starting point and random environment.
1063         actions: actions of the returned path.
1064         states: viewpoints of all the states in between.
1065         step_outputs: list of step() return tuples.
1066     Raises:
1067       ValueError: if first_n is not greater than zero; if min_len is different
1068         from None.
1069     """
1070     if max_len is None:
1071       raise ValueError('max_len can not be set as None')
1072     if max_len < 1:
1073       raise ValueError('first_n must be greater or equal to 1.')
1074     if min_len is not None:
1075       raise ValueError('min_len is not yet implemented.')
1076 
1077     path = []
1078     actions = []
1079     states = []
1080     step_outputs = []
1081     obs = self.reset()
1082     last_obs_tuple = [obs, 0, False, {}]
1083     for _ in xrange(max_len):
1084       action = np.random.choice(self._actions)
1085       # We don't want to sample stop action because stop does not add new
1086       # information.
1087       while action == 'stop':
1088         action = np.random.choice(self._actions)
1089       path.append(self.to_vertex(self._cur_image_id))
1090       onehot = np.zeros((len(self._actions),), dtype=np.float32)
1091       onehot[self._actions.index(action)] = 1.
1092       actions.append(onehot)
1093       states.append(self.vertex_to_pose(path[-1]))
1094       step_outputs.append(copy.deepcopy(last_obs_tuple))
1095       last_obs_tuple = self.step(action)
1096 
1097     return path, actions, states, step_outputs
