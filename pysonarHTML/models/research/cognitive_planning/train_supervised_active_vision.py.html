 138 def create_modality_types():
 139   """Parses the modality_types and returns a list of task_env.ModalityType."""
 140   if not FLAGS.modality_types:
 141     raise ValueError('there needs to be at least one modality type')
 142   modality_types = FLAGS.modality_types.split('_')
 143   for x in modality_types:
 144     if x not in ['image', 'sseg', 'det', 'depth']:
 145       raise ValueError('invalid modality type: {}'.format(x))
 146 
 147   conversion_dict = {
 148       'image': task_env.ModalityTypes.IMAGE,
 149       'sseg': task_env.ModalityTypes.SEMANTIC_SEGMENTATION,
 150       'depth': task_env.ModalityTypes.DEPTH,
 151       'det': task_env.ModalityTypes.OBJECT_DETECTION,
 152   }
 153   return [conversion_dict[k] for k in modality_types]

 156 def create_task_io_config(
 157     modality_types,
 158     goal_category_count,
 159     action_size,
 160     sequence_length,
 161 ):
 162   """Generates task io config."""
 163   shape_prefix = [sequence_length, FLAGS.obs_resolution, FLAGS.obs_resolution]
 164   shapes = {
 165       task_env.ModalityTypes.IMAGE: [sequence_length, 224, 224, 3],
 166       task_env.ModalityTypes.DEPTH: shape_prefix + [
 167           2,
 168       ],
 169       task_env.ModalityTypes.SEMANTIC_SEGMENTATION: shape_prefix + [
 170           1,
 171       ],
 172       task_env.ModalityTypes.OBJECT_DETECTION: shape_prefix + [
 173           90,
 174       ]
 175   }
 176   types = {k: tf.float32 for k in shapes}
 177   types[task_env.ModalityTypes.IMAGE] = tf.uint8
 178   inputs = collections.OrderedDict(
 179       [[mtype, (types[mtype], shapes[mtype])] for mtype in modality_types])
 180   inputs[task_env.ModalityTypes.GOAL] = (tf.float32,
 181                                          [sequence_length, goal_category_count])
 182   inputs[task_env.ModalityTypes.PREV_ACTION] = (tf.float32, [
 183       sequence_length, action_size + 1
 184   ])
 185   print inputs
 186   return tasks.UnrolledTaskIOConfig(
 187       inputs=inputs,
 188       output=(tf.float32, [sequence_length, action_size]),
 189       query=None)

 192 def map_to_embedder(modality_type):
 193   """Maps modality_type to its corresponding embedder."""
 194   if modality_type == task_env.ModalityTypes.PREV_ACTION:
 195     return None
 196   if modality_type == task_env.ModalityTypes.GOAL:
 197     return embedders.IdentityEmbedder()
 198   if modality_type == task_env.ModalityTypes.IMAGE:
 199     return embedders.ResNet50Embedder()
 200   conv_window_sizes = [int(x) for x in FLAGS.conv_window_sizes.split('_')]
 201   conv_channels = [int(x) for x in FLAGS.conv_channels.split('_')]
 202   conv_strides = [int(x) for x in FLAGS.conv_strides.split('_')]
 203   params = tf.contrib.training.HParams(
 204       to_one_hot=modality_type == task_env.ModalityTypes.SEMANTIC_SEGMENTATION,
 205       one_hot_length=10,
 206       conv_sizes=conv_window_sizes,
 207       conv_strides=conv_strides,
 208       conv_channels=conv_channels,
 209       embedding_size=FLAGS.embedding_fc_size,
 210       weight_decay_rate=FLAGS.weight_decay,
 211   )
 212   return embedders.SmallNetworkEmbedder(params)

 215 def create_train_and_init_ops(policy, task):
 216   """Creates training ops given the arguments.
 217 
 218   Args:
 219     policy: the policy for the task.
 220     task: the task instance.
 221 
 222   Returns:
 223     train_op: the op that needs to be runned at each step.
 224     summaries_op: the summary op that is executed.
 225     init_fn: the op that initializes the variables if there is no previous
 226       checkpoint. If Resnet50 is not used in the model it is None, otherwise
 227       it reads the weights from FLAGS.resnet50_path and sets the init_fn
 228       to the op that initializes the ResNet50 with the pre-trained weights.
 229   """
 230   assert isinstance(task, tasks.GotoStaticXNoExplorationTask)
 231   assert isinstance(policy, policies.Policy)
 232 
 233   inputs, _, gt_outputs, masks = task.tf_episode_batch(FLAGS.batch_size)
 234   outputs, _ = policy.build(inputs, None)
 235   loss = task.target_loss(gt_outputs, outputs, masks)
 236 
 237   init_fn = None
 238 
 239   # If resnet is added to the graph, init_fn should initialize resnet weights
 240   # if there is no previous checkpoint.
 241   variables_assign_dict = {}
 242   vars_list = []
 243   for v in slim.get_model_variables():
 244     if v.name.find('resnet') >= 0:
 245       if not FLAGS.freeze_resnet_weights:
 246         vars_list.append(v)
 247       variables_assign_dict[v.name[v.name.find('resnet'):-2]] = v
 248     else:
 249       vars_list.append(v)
 250   
 251   global_step = tf.train.get_or_create_global_step()
 252   learning_rate = tf.train.exponential_decay(
 253       FLAGS.learning_rate,
 254       global_step,
 255       decay_steps=FLAGS.decay_steps,
 256       decay_rate=0.98,
 257       staircase=True)
 258   optimizer = tf.train.AdamOptimizer(learning_rate)
 259   train_op = slim.learning.create_train_op(
 260       loss,
 261       optimizer,
 262       global_step=global_step,
 263       variables_to_train=vars_list,
 264   )
 265 
 266   if variables_assign_dict:
 267     init_fn = slim.assign_from_checkpoint_fn(
 268         FLAGS.resnet50_path,
 269         variables_assign_dict,
 270         ignore_missing_vars=False)
 271   scalar_summaries = {}
 272   scalar_summaries['LR'] = learning_rate
 273   scalar_summaries['loss'] = loss
 274 
 275   for name, summary in scalar_summaries.iteritems():
 276     tf.summary.scalar(name, summary)
 277  
 278   return train_op, init_fn

 281 def create_eval_ops(policy, config, possible_targets):
 282   """Creates the necessary ops for evaluation."""
 283   inputs_feed = collections.OrderedDict([[
 284       mtype,
 285       tf.placeholder(config.inputs[mtype].type,
 286                      [1] + config.inputs[mtype].shape)
 287   ] for mtype in config.inputs])
 288   inputs_feed[task_env.ModalityTypes.PREV_ACTION] = tf.placeholder(
 289       tf.float32, [1, 1] + [
 290           config.output.shape[-1] + 1,
 291       ])
 292   prev_state_feed = [
 293       tf.placeholder(
 294           tf.float32, [1, FLAGS.lstm_cell_size], name='prev_state_{}'.format(i))
 295       for i in range(2)
 296   ]
 297   policy_outputs = policy.build(inputs_feed, prev_state_feed)
 298   summary_feed = {}
 299   for c in possible_targets + ['mean']:
 300     summary_feed[c] = tf.placeholder(
 301         tf.float32, [], name='eval_in_range_{}_input'.format(c))
 302     tf.summary.scalar('eval_in_range_{}'.format(c), summary_feed[c])
 303 
 304   return inputs_feed, prev_state_feed, policy_outputs, (tf.summary.merge_all(),
 305                                                         summary_feed)

 308 def unroll_policy_for_eval(
 309     sess,
 310     env,
 311     inputs_feed,
 312     prev_state_feed,
 313     policy_outputs,
 314     number_of_steps,
 315     output_folder,
 316 ):
 317   """unrolls the policy for testing.
 318 
 319   Args:
 320     sess: tf.Session
 321     env: The environment.
 322     inputs_feed: dictionary of placeholder for the input modalities.
 323     prev_state_feed: placeholder for the input to the prev_state of the model.
 324     policy_outputs: tensor that contains outputs of the policy.
 325     number_of_steps: maximum number of unrolling steps.
 326     output_folder: output_folder where the function writes a dictionary of
 327       detailed information about the path. The dictionary keys are 'states' and
 328       'distance'. The value for 'states' is the list of states that the agent
 329       goes along the path. The value for 'distance' contains the length of
 330       shortest path to the goal at each step.
 331 
 332   Returns:
 333     states: list of states along the path.
 334     distance: list of distances along the path.
 335   """
 336   prev_state = [
 337       np.zeros((1, FLAGS.lstm_cell_size), dtype=np.float32) for _ in range(2)
 338   ]
 339   prev_action = np.zeros((1, 1, FLAGS.action_size + 1), dtype=np.float32)
 340   obs = env.reset()
 341   distances_to_goal = []
 342   states = []
 343   unique_id = '{}_{}'.format(env.cur_image_id(), env.goal_string)
 344   for _ in range(number_of_steps):
 345     distances_to_goal.append(
 346         np.min([
 347             len(
 348                 nx.shortest_path(env.graph, env.pose_to_vertex(env.state()),
 349                                  env.pose_to_vertex(target_view)))
 350             for target_view in env.targets()
 351         ]))
 352     states.append(env.state())
 353     feed_dict = {inputs_feed[mtype]: [[obs[mtype]]] for mtype in inputs_feed}
 354     feed_dict[prev_state_feed[0]] = prev_state[0]
 355     feed_dict[prev_state_feed[1]] = prev_state[1]
 356     action_values, prev_state = sess.run(policy_outputs, feed_dict=feed_dict)
 357     chosen_action = np.argmax(action_values[0])
 358     obs, _, done, info = env.step(np.int32(chosen_action))
 359     prev_action[0][0][chosen_action] = 1.
 360     prev_action[0][0][-1] = float(info['success'])
 361     # If the agent chooses action stop or the number of steps exceeeded
 362     # env._episode_length.
 363     if done:
 364       break
 365 
 366   # logging.info('distance = %d, id = %s, #steps = %d', distances_to_goal[-1],
 367   output_path = os.path.join(output_folder, unique_id + '.npy')
 368   with tf.gfile.Open(output_path, 'w') as f:
 369     print 'saving path information to {}'.format(output_path)
 370     np.save(f, {'states': states, 'distance': distances_to_goal})
 371   return states, distances_to_goal

 374 def init(sequence_length, eval_init_points_file_name, worlds):
 375   """Initializes the common operations between train and test."""
 376   modality_types = create_modality_types()
 377   logging.info('modality types: %r', modality_types)
 378   # negative reward_goal_range prevents the env from terminating early when the
 379   # agent is close to the goal. The policy should keep the agent until the end
 380   # of the 100 steps either through chosing stop action or oscilating around
 381   # the target.
 382 
 383   env = active_vision_dataset_env.ActiveVisionDatasetEnv(
 384       modality_types=modality_types +
 385       [task_env.ModalityTypes.GOAL, task_env.ModalityTypes.PREV_ACTION],
 386       reward_goal_range=-1,
 387       eval_init_points_file_name=eval_init_points_file_name,
 388       worlds=worlds,
 389       output_size=FLAGS.obs_resolution,
 390   )
 391 
 392   config = create_task_io_config(
 393       modality_types=modality_types,
 394       goal_category_count=FLAGS.goal_category_count,
 395       action_size=FLAGS.action_size,
 396       sequence_length=sequence_length,
 397   )
 398   task = tasks.GotoStaticXNoExplorationTask(env=env, config=config)
 399   embedders_dict = {mtype: map_to_embedder(mtype) for mtype in config.inputs}
 400   policy_params = tf.contrib.training.HParams(
 401       lstm_state_size=FLAGS.lstm_cell_size,
 402       fc_channels=FLAGS.policy_fc_size,
 403       weight_decay=FLAGS.weight_decay,
 404       target_embedding_size=FLAGS.embedding_fc_size,
 405   )
 406   policy = policies.LSTMPolicy(
 407       modality_names=config.inputs.keys(),
 408       embedders_dict=embedders_dict,
 409       action_size=FLAGS.action_size,
 410       params=policy_params,
 411       max_episode_length=sequence_length)
 412   return env, config, task, policy

 415 def test():
 416   """Contains all the operations for testing policies."""
 417   env, config, _, policy = init(1, 'all_init_configs', TEST_WORLDS)
 418   inputs_feed, prev_state_feed, policy_outputs, summary_op = create_eval_ops(
 419       policy, config, env.possible_targets)
 420   sv = tf.train.Supervisor(logdir=FLAGS.logdir)
 421   prev_checkpoint = None
 422   with sv.managed_session(
 423       start_standard_services=False,
 424       config=tf.ConfigProto(allow_soft_placement=True)) as sess:
 425     while not sv.should_stop():
 426       while True:
 427         new_checkpoint = tf.train.latest_checkpoint(FLAGS.logdir)
 428         print 'new_checkpoint ', new_checkpoint
 429         if not new_checkpoint:
 430           time.sleep(1)
 431           continue
 432         if prev_checkpoint is None:
 433           prev_checkpoint = new_checkpoint
 434           break
 435         if prev_checkpoint != new_checkpoint:
 436           prev_checkpoint = new_checkpoint
 437           break
 438         else:  # if prev_checkpoint == new_checkpoint, we have to wait more.
 439           time.sleep(1)
 440 
 441       checkpoint_step = int(new_checkpoint[new_checkpoint.rfind('-') + 1:])
 442       sv.saver.restore(sess, new_checkpoint)
 443       print '--------------------'
 444       print 'evaluating checkpoint {}'.format(new_checkpoint)
 445       folder_path = os.path.join(FLAGS.logdir, 'evals', str(checkpoint_step))
 446       if not tf.gfile.Exists(folder_path):
 447         tf.gfile.MakeDirs(folder_path)
 448       eval_stats = {c: [] for c in env.possible_targets}
 449       for test_iter in range(FLAGS.test_iters):
 450         print 'evaluating {} of {}'.format(test_iter, FLAGS.test_iters)
 451         _, distance_to_goal = unroll_policy_for_eval(
 452             sess,
 453             env,
 454             inputs_feed,
 455             prev_state_feed,
 456             policy_outputs,
 457             FLAGS.max_eval_episode_length,
 458             folder_path,
 459         )
 460         print 'goal = {}'.format(env.goal_string)
 461         eval_stats[env.goal_string].append(float(distance_to_goal[-1] <= 7))
 462       eval_stats = {k: np.mean(v) for k, v in eval_stats.iteritems()}
 463       eval_stats['mean'] = np.mean(eval_stats.values())
 464       print eval_stats
 465       feed_dict = {summary_op[1][c]: eval_stats[c] for c in eval_stats}
 466       summary_str = sess.run(summary_op[0], feed_dict=feed_dict)
 467       writer = sv.summary_writer
 468       writer.add_summary(summary_str, checkpoint_step)
 469       writer.flush()

 472 def train():
 473   _, _, task, policy = init(FLAGS.sequence_length, None, TRAIN_WORLDS)
 474   print(FLAGS.save_summaries_secs)
 475   print(FLAGS.save_interval_secs)
 476   print(FLAGS.logdir)
 477 
 478   with tf.device(
 479       tf.train.replica_device_setter(ps_tasks=FLAGS.ps_tasks, merge_devices=True)):
 480     train_op, init_fn = create_train_and_init_ops(policy=policy, task=task)
 481     print(FLAGS.logdir)
 482     slim.learning.train(
 483         train_op=train_op,
 484         init_fn=init_fn,
 485         logdir=FLAGS.logdir,
 486         is_chief=FLAGS.task_id == 0,
 487         number_of_steps=FLAGS.train_iters,
 488         save_summaries_secs=FLAGS.save_summaries_secs,
 489         save_interval_secs=FLAGS.save_interval_secs,
 490         session_config=tf.ConfigProto(allow_soft_placement=True),
 491     )
