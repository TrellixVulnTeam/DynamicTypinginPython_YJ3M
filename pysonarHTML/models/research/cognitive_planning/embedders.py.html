  27 class Embedder(object):
  28   """Represents the embedder for different modalities.
  29 
  30   Modalities can be semantic segmentation, depth channel, object detection and
  31   so on, which require specific embedder for them.
  32   """
  33   __metaclass__ = abc.ABCMeta
  34 
  35   @abc.abstractmethod
  36   def build(self, observation):
  37     """Builds the model to embed the observation modality.
  38 
  39     Args:
  40       observation: tensor that contains the raw observation from modality.
  41     Returns:
  42       Embedding tensor for the given observation tensor.
  43     """
  44     raise NotImplementedError(
  45         'Needs to be implemented as part of Embedder Interface')
  46

  48 class DetectionBoxEmbedder(Embedder):
  49   """Represents the model that encodes the detection boxes from images."""
  50 
  51   def __init__(self, rnn_state_size, scope=None):
  52     self._rnn_state_size = rnn_state_size
  53     self._scope = scope
  54 
  55   def build(self, observations):
  56     """Builds the model to embed object detection observations.
  57 
  58     Args:
  59       observations: a tuple of (dets, det_num).
  60         dets is a tensor of BxTxLxE that has the detection boxes in all the
  61           images of the batch. B is the batch size, T is the maximum length of
  62           episode, L is the maximum number of detections per image in the batch
  63           and E is the size of each detection embedding.
  64         det_num is a tensor of BxT that contains the number of detected boxes
  65           each image of each sequence in the batch.
  66     Returns:
  67       For each image in the batch, returns the accumulative embedding of all the
  68       detection boxes in that image.
  69     """
  70     with tf.variable_scope(self._scope, default_name=''):
  71       shape = observations[0].shape
  72       dets = tf.reshape(observations[0], [-1, shape[-2], shape[-1]])
  73       det_num = tf.reshape(observations[1], [-1])
  74       lstm_cell = tf.nn.rnn_cell.BasicLSTMCell(self._rnn_state_size)
  75       batch_size = tf.shape(dets)[0]
  76       lstm_outputs, _ = tf.nn.dynamic_rnn(
  77           cell=lstm_cell,
  78           inputs=dets,
  79           sequence_length=det_num,
  80           initial_state=lstm_cell.zero_state(batch_size, dtype=tf.float32),
  81           dtype=tf.float32)
  82       # Gathering the last state of each sequence in the batch.
  83       batch_range = tf.range(batch_size)
  84       indices = tf.stack([batch_range, det_num - 1], axis=1)
  85       last_lstm_outputs = tf.gather_nd(lstm_outputs, indices)
  86       last_lstm_outputs = tf.reshape(last_lstm_outputs,
  87                                      [-1, shape[1], self._rnn_state_size])
  88     return last_lstm_outputs

  91 class ResNet(Embedder):
  92   """Residual net embedder for image data."""
  93 
  94   def __init__(self, params, *args, **kwargs):
  95     super(ResNet, self).__init__(*args, **kwargs)
  96     self._params = params
  97     self._extra_train_ops = []
  98 
  99   def build(self, images):
 100     shape = images.get_shape().as_list()
 101     if len(shape) == 5:
 102       images = tf.reshape(images,
 103                           [shape[0] * shape[1], shape[2], shape[3], shape[4]])
 104     embedding = self._build_model(images)
 105     if len(shape) == 5:
 106       embedding = tf.reshape(embedding, [shape[0], shape[1], -1])
 107 
 108     return embedding
 109 
 110   @property
 111   def extra_train_ops(self):
 112     return self._extra_train_ops
 113 
 114   def _build_model(self, images):
 115     """Builds the model."""
 116 
 117     # Convert images to floats and normalize them.
 118     images = tf.to_float(images)
 119     bs = images.get_shape().as_list()[0]
 120     images = [
 121         tf.image.per_image_standardization(tf.squeeze(i))
 122         for i in tf.split(images, bs)
 123     ]
 124     images = tf.concat([tf.expand_dims(i, axis=0) for i in images], axis=0)
 125 
 126     with tf.variable_scope('init'):
 127       x = self._conv('init_conv', images, 3, 3, 16, self._stride_arr(1))
 128 
 129     strides = [1, 2, 2]
 130     activate_before_residual = [True, False, False]
 131     if self._params.use_bottleneck:
 132       res_func = self._bottleneck_residual
 133       filters = [16, 64, 128, 256]
 134     else:
 135       res_func = self._residual
 136       filters = [16, 16, 32, 128]
 137 
 138     with tf.variable_scope('unit_1_0'):
 139       x = res_func(x, filters[0], filters[1], self._stride_arr(strides[0]),
 140                    activate_before_residual[0])
 141     for i in xrange(1, self._params.num_residual_units):
 142       with tf.variable_scope('unit_1_%d' % i):
 143         x = res_func(x, filters[1], filters[1], self._stride_arr(1), False)
 144 
 145     with tf.variable_scope('unit_2_0'):
 146       x = res_func(x, filters[1], filters[2], self._stride_arr(strides[1]),
 147                    activate_before_residual[1])
 148     for i in xrange(1, self._params.num_residual_units):
 149       with tf.variable_scope('unit_2_%d' % i):
 150         x = res_func(x, filters[2], filters[2], self._stride_arr(1), False)
 151 
 152     with tf.variable_scope('unit_3_0'):
 153       x = res_func(x, filters[2], filters[3], self._stride_arr(strides[2]),
 154                    activate_before_residual[2])
 155     for i in xrange(1, self._params.num_residual_units):
 156       with tf.variable_scope('unit_3_%d' % i):
 157         x = res_func(x, filters[3], filters[3], self._stride_arr(1), False)
 158 
 159     with tf.variable_scope('unit_last'):
 160       x = self._batch_norm('final_bn', x)
 161       x = self._relu(x, self._params.relu_leakiness)
 162 
 163     with tf.variable_scope('pool_logit'):
 164       x = self._global_avg_pooling(x)
 165 
 166     return x
 167 
 168   def _stride_arr(self, stride):
 169     return [1, stride, stride, 1]
 170 
 171   def _batch_norm(self, name, x):
 172     """batch norm implementation."""
 173     with tf.variable_scope(name):
 174       params_shape = [x.shape[-1]]
 175 
 176       beta = tf.get_variable(
 177           'beta',
 178           params_shape,
 179           tf.float32,
 180           initializer=tf.constant_initializer(0.0, tf.float32))
 181       gamma = tf.get_variable(
 182           'gamma',
 183           params_shape,
 184           tf.float32,
 185           initializer=tf.constant_initializer(1.0, tf.float32))
 186 
 187       if self._params.is_train:
 188         mean, variance = tf.nn.moments(x, [0, 1, 2], name='moments')
 189 
 190         moving_mean = tf.get_variable(
 191             'moving_mean',
 192             params_shape,
 193             tf.float32,
 194             initializer=tf.constant_initializer(0.0, tf.float32),
 195             trainable=False)
 196         moving_variance = tf.get_variable(
 197             'moving_variance',
 198             params_shape,
 199             tf.float32,
 200             initializer=tf.constant_initializer(1.0, tf.float32),
 201             trainable=False)
 202 
 203         self._extra_train_ops.append(
 204             tf.assign_moving_average(moving_mean, mean, 0.9))
 205         self._extra_train_ops.append(
 206             tf.assign_moving_average(moving_variance, variance, 0.9))
 207       else:
 208         mean = tf.get_variable(
 209             'moving_mean',
 210             params_shape,
 211             tf.float32,
 212             initializer=tf.constant_initializer(0.0, tf.float32),
 213             trainable=False)
 214         variance = tf.get_variable(
 215             'moving_variance',
 216             params_shape,
 217             tf.float32,
 218             initializer=tf.constant_initializer(1.0, tf.float32),
 219             trainable=False)
 220         tf.summary.histogram(mean.op.name, mean)
 221         tf.summary.histogram(variance.op.name, variance)
 222       # elipson used to be 1e-5. Maybe 0.001 solves NaN problem in deeper net.
 223       y = tf.nn.batch_normalization(x, mean, variance, beta, gamma, 0.001)
 224       y.set_shape(x.shape)
 225       return y
 226 
 227   def _residual(self,
 228                 x,
 229                 in_filter,
 230                 out_filter,
 231                 stride,
 232                 activate_before_residual=False):
 233     """Residual unit with 2 sub layers."""
 234 
 235     if activate_before_residual:
 236       with tf.variable_scope('shared_activation'):
 237         x = self._batch_norm('init_bn', x)
 238         x = self._relu(x, self._params.relu_leakiness)
 239         orig_x = x
 240     else:
 241       with tf.variable_scope('residual_only_activation'):
 242         orig_x = x
 243         x = self._batch_norm('init_bn', x)
 244         x = self._relu(x, self._params.relu_leakiness)
 245 
 246     with tf.variable_scope('sub1'):
 247       x = self._conv('conv1', x, 3, in_filter, out_filter, stride)
 248 
 249     with tf.variable_scope('sub2'):
 250       x = self._batch_norm('bn2', x)
 251       x = self._relu(x, self._params.relu_leakiness)
 252       x = self._conv('conv2', x, 3, out_filter, out_filter, [1, 1, 1, 1])
 253 
 254     with tf.variable_scope('sub_add'):
 255       if in_filter != out_filter:
 256         orig_x = tf.nn.avg_pool(orig_x, stride, stride, 'VALID')
 257         orig_x = tf.pad(
 258             orig_x, [[0, 0], [0, 0], [0, 0], [(out_filter - in_filter) // 2,
 259                                               (out_filter - in_filter) // 2]])
 260       x += orig_x
 261 
 262     return x
 263 
 264   def _bottleneck_residual(self,
 265                            x,
 266                            in_filter,
 267                            out_filter,
 268                            stride,
 269                            activate_before_residual=False):
 270     """A residual convolutional layer with a bottleneck.
 271 
 272     The layer is a composite of three convolutional layers with a ReLU non-
 273     linearity and batch normalization after each linear convolution. The depth
 274     if the second and third layer is out_filter / 4 (hence it is a bottleneck).
 275 
 276     Args:
 277       x: a float 4 rank Tensor representing the input to the layer.
 278       in_filter: a python integer representing depth of the input.
 279       out_filter: a python integer representing depth of the output.
 280       stride: a python integer denoting the stride of the layer applied before
 281         the first convolution.
 282       activate_before_residual: a python boolean. If True, then a ReLU is
 283         applied as a first operation on the input x before everything else.
 284     Returns:
 285       A 4 rank Tensor with batch_size = batch size of input, width and height =
 286       width / stride and height / stride of the input and depth = out_filter.
 287     """
 288     if activate_before_residual:
 289       with tf.variable_scope('common_bn_relu'):
 290         x = self._batch_norm('init_bn', x)
 291         x = self._relu(x, self._params.relu_leakiness)
 292         orig_x = x
 293     else:
 294       with tf.variable_scope('residual_bn_relu'):
 295         orig_x = x
 296         x = self._batch_norm('init_bn', x)
 297         x = self._relu(x, self._params.relu_leakiness)
 298 
 299     with tf.variable_scope('sub1'):
 300       x = self._conv('conv1', x, 1, in_filter, out_filter / 4, stride)
 301 
 302     with tf.variable_scope('sub2'):
 303       x = self._batch_norm('bn2', x)
 304       x = self._relu(x, self._params.relu_leakiness)
 305       x = self._conv('conv2', x, 3, out_filter / 4, out_filter / 4,
 306                      [1, 1, 1, 1])
 307 
 308     with tf.variable_scope('sub3'):
 309       x = self._batch_norm('bn3', x)
 310       x = self._relu(x, self._params.relu_leakiness)
 311       x = self._conv('conv3', x, 1, out_filter / 4, out_filter, [1, 1, 1, 1])
 312 
 313     with tf.variable_scope('sub_add'):
 314       if in_filter != out_filter:
 315         orig_x = self._conv('project', orig_x, 1, in_filter, out_filter, stride)
 316       x += orig_x
 317 
 318     return x
 319 
 320   def _decay(self):
 321     costs = []
 322     for var in tf.trainable_variables():
 323       if var.op.name.find(r'DW') > 0:
 324         costs.append(tf.nn.l2_loss(var))
 325 
 326     return tf.mul(self._params.weight_decay_rate, tf.add_n(costs))
 327 
 328   def _conv(self, name, x, filter_size, in_filters, out_filters, strides):
 329     """Convolution."""
 330     with tf.variable_scope(name):
 331       n = filter_size * filter_size * out_filters
 332       kernel = tf.get_variable(
 333           'DW', [filter_size, filter_size, in_filters, out_filters],
 334           tf.float32,
 335           initializer=tf.random_normal_initializer(stddev=np.sqrt(2.0 / n)))
 336       return tf.nn.conv2d(x, kernel, strides, padding='SAME')
 337 
 338   def _relu(self, x, leakiness=0.0):
 339     return tf.where(tf.less(x, 0.0), leakiness * x, x, name='leaky_relu')
 340 
 341   def _fully_connected(self, x, out_dim):
 342     x = tf.reshape(x, [self._params.batch_size, -1])
 343     w = tf.get_variable(
 344         'DW', [x.get_shape()[1], out_dim],
 345         initializer=tf.uniform_unit_scaling_initializer(factor=1.0))
 346     b = tf.get_variable(
 347         'biases', [out_dim], initializer=tf.constant_initializer())
 348     return tf.nn.xw_plus_b(x, w, b)
 349 
 350   def _global_avg_pooling(self, x):
 351     assert x.get_shape().ndims == 4
 352     return tf.reduce_mean(x, [1, 2])

 355 class MLPEmbedder(Embedder):
 356   """Embedder of vectorial data.
 357 
 358   The net is a multi-layer perceptron, with ReLU nonlinearities in all layers
 359   except the last one.
 360   """
 361 
 362   def __init__(self, layers, *args, **kwargs):
 363     """Constructs MLPEmbedder.
 364 
 365     Args:
 366       layers: a list of python integers representing layer sizes.
 367       *args: arguments for super constructor.
 368       **kwargs: keyed arguments for super constructor.
 369     """
 370     super(MLPEmbedder, self).__init__(*args, **kwargs)
 371     self._layers = layers
 372 
 373   def build(self, features):
 374     shape = features.get_shape().as_list()
 375     if len(shape) == 3:
 376       features = tf.reshape(features, [shape[0] * shape[1], shape[2]])
 377     x = features
 378     for i, dim in enumerate(self._layers):
 379       with tf.variable_scope('layer_%i' % i):
 380         x = self._fully_connected(x, dim)
 381         if i < len(self._layers) - 1:
 382           x = self._relu(x)
 383 
 384     if len(shape) == 3:
 385       x = tf.reshape(x, shape[:-1] + [self._layers[-1]])
 386     return x
 387 
 388   def _fully_connected(self, x, out_dim):
 389     w = tf.get_variable(
 390         'DW', [x.get_shape()[1], out_dim],
 391         initializer=tf.variance_scaling_initializer(distribution='uniform'))
 392     b = tf.get_variable(
 393         'biases', [out_dim], initializer=tf.constant_initializer())
 394     return tf.nn.xw_plus_b(x, w, b)
 395 
 396   def _relu(self, x, leakiness=0.0):
 397     return tf.where(tf.less(x, 0.0), leakiness * x, x, name='leaky_relu')

 400 class SmallNetworkEmbedder(Embedder):
 401   """Embedder for image like observations.
 402 
 403   The network is comprised of multiple conv layers and a fully connected layer
 404   at the end. The number of conv layers and the parameters are configured from
 405   params.
 406   """
 407 
 408   def __init__(self, params, *args, **kwargs):
 409     """Constructs the small network.
 410 
 411     Args:
 412       params: params should be tf.hparams type. params need to have a list of
 413         conv_sizes, conv_strides, conv_channels. The length of these lists
 414         should be equal to each other and to the number of conv layers in the
 415         network. Plus, it also needs to have boolean variable named to_one_hot
 416         which indicates whether the input should be converted to one hot or not.
 417         The size of the fully connected layer is specified by
 418         params.embedding_size.
 419 
 420       *args: The rest of the parameters.
 421       **kwargs: the reset of the parameters.
 422 
 423     Raises:
 424       ValueError: If the length of params.conv_strides, params.conv_sizes, and
 425         params.conv_channels are not equal.
 426 
 427     """
 428 
 429     super(SmallNetworkEmbedder, self).__init__(*args, **kwargs)
 430     self._params = params
 431     if len(self._params.conv_sizes) != len(self._params.conv_strides):
 432       raise ValueError(
 433           'Conv sizes and strides should have the same length: {} != {}'.format(
 434               len(self._params.conv_sizes), len(self._params.conv_strides)))
 435 
 436     if len(self._params.conv_sizes) != len(self._params.conv_channels):
 437       raise ValueError(
 438           'Conv sizes and channels should have the same length: {} != {}'.
 439           format(len(self._params.conv_sizes), len(self._params.conv_channels)))
 440 
 441   def build(self, images):
 442     """Builds the embedder with the given speicifcation.
 443 
 444     Args:
 445       images: a tensor that contains the input images which has the shape of
 446         NxTxHxWxC where N is the batch size, T is the maximum length of the
 447         sequence, H and W are the height and width of the images and C is the
 448         number of channels.
 449 
 450     Returns:
 451       A tensor that is the embedding of the images.
 452     """
 453 
 454     shape = images.get_shape().as_list()
 455     images = tf.reshape(images,
 456                         [shape[0] * shape[1], shape[2], shape[3], shape[4]])
 457 
 458     with slim.arg_scope(
 459         [slim.conv2d, slim.fully_connected],
 460         activation_fn=tf.nn.relu,
 461         weights_regularizer=slim.l2_regularizer(self._params.weight_decay_rate),
 462         biases_initializer=tf.zeros_initializer()):
 463       with slim.arg_scope([slim.conv2d], padding='SAME'):
 464         # convert the image to one hot if needed.
 465         if self._params.to_one_hot:
 466           net = tf.one_hot(
 467               tf.squeeze(tf.to_int32(images), axis=[-1]),
 468               self._params.one_hot_length)
 469         else:
 470           net = images
 471 
 472         p = self._params
 473         # Adding conv layers with the specified configurations.
 474         for conv_id, kernel_stride_channel in enumerate(
 475             zip(p.conv_sizes, p.conv_strides, p.conv_channels)):
 476           kernel_size, stride, channels = kernel_stride_channel
 477           net = slim.conv2d(
 478               net,
 479               channels, [kernel_size, kernel_size],
 480               stride,
 481               scope='conv_{}'.format(conv_id + 1))
 482 
 483         net = slim.flatten(net)
 484         net = slim.fully_connected(net, self._params.embedding_size, scope='fc')
 485 
 486         output = tf.reshape(net, [shape[0], shape[1], -1])
 487         return output

 490 class ResNet50Embedder(Embedder):
 491   """Uses ResNet50 to embed input images."""
 492 
 493   def build(self, images):
 494     """Builds a ResNet50 embedder for the input images.
 495 
 496     It assumes that the range of the pixel values in the images tensor is
 497       [0,255] and should be castable to tf.uint8.
 498 
 499     Args:
 500       images: a tensor that contains the input images which has the shape of
 501           NxTxHxWx3 where N is the batch size, T is the maximum length of the
 502           sequence, H and W are the height and width of the images and C is the
 503           number of channels.
 504     Returns:
 505       The embedding of the input image with the shape of NxTxL where L is the
 506         embedding size of the output.
 507 
 508     Raises:
 509       ValueError: if the shape of the input does not agree with the expected
 510       shape explained in the Args section.
 511     """
 512     shape = images.get_shape().as_list()
 513     if len(shape) != 5:
 514       raise ValueError(
 515           'The tensor shape should have 5 elements, {} is provided'.format(
 516               len(shape)))
 517     if shape[4] != 3:
 518       raise ValueError('Three channels are expected for the input image')
 519 
 520     images = tf.cast(images, tf.uint8)
 521     images = tf.reshape(images,
 522                         [shape[0] * shape[1], shape[2], shape[3], shape[4]])
 523     with slim.arg_scope(resnet_v2.resnet_arg_scope()):
 524 
 525       def preprocess_fn(x):
 526         x = tf.expand_dims(x, 0)
 527         x = tf.image.resize_bilinear(x, [299, 299],
 528                                        align_corners=False)
 529         return(tf.squeeze(x, [0]))
 530 
 531       images = tf.map_fn(preprocess_fn, images, dtype=tf.float32)
 532 
 533       net, _ = resnet_v2.resnet_v2_50(
 534           images, is_training=False, global_pool=True)
 535       output = tf.reshape(net, [shape[0], shape[1], -1])
 536       return output

 539 class IdentityEmbedder(Embedder):
 540   """This embedder just returns the input as the output.
 541 
 542   Used for modalitites that the embedding of the modality is the same as the
 543   modality itself. For example, it can be used for one_hot goal.
 544   """
 545 
 546   def build(self, images):
 547     return images
