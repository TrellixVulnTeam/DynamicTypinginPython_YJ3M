  36 def _pad_or_clip_array(np_arr, arr_len, is_front_clip=True, output_mask=False):
  37   """Make np_arr array to have length arr_len.
  38 
  39   If the array is shorter than arr_len, then it is padded from the front with
  40   zeros. If it is longer, then it is clipped either from the back or from the
  41   front. Only the first dimension is modified.
  42 
  43   Args:
  44     np_arr: numpy array.
  45     arr_len: integer scalar.
  46     is_front_clip: a boolean. If true then clipping is done in the front,
  47       otherwise in the back.
  48     output_mask: If True, outputs a numpy array of rank 1 which represents
  49       a mask of which values have been added (0 - added, 1 - actual output).
  50 
  51   Returns:
  52     A numpy array and the size of padding (as a python int32). This size is
  53     negative is the array is clipped.
  54   """
  55   shape = list(np_arr.shape)
  56   pad_size = arr_len - shape[0]
  57   padded_or_clipped = None
  58   if pad_size < 0:
  59     if is_front_clip:
  60       padded_or_clipped = np_arr[-pad_size:, :]
  61     else:
  62       padded_or_clipped = np_arr[:arr_len, :]
  63   elif pad_size > 0:
  64     padding = np.zeros([pad_size] + shape[1:], dtype=np_arr.dtype)
  65     padded_or_clipped = np.concatenate([np_arr, padding], axis=0)
  66   else:
  67     padded_or_clipped = np_arr
  68 
  69   if output_mask:
  70     mask = np.ones((arr_len,), dtype=np.int)
  71     if pad_size > 0:
  72       mask[-pad_size:] = 0
  73     return padded_or_clipped, pad_size, mask
  74   else:
  75     return padded_or_clipped, pad_size

  78 def classification_loss(truth, predicted, weights=None, is_one_hot=True):
  79   """A cross entropy loss.
  80 
  81   Computes the mean of cross entropy losses for all pairs of true labels and
  82   predictions. It wraps around a tf implementation of the cross entropy loss
  83   with additional reformating of the inputs. If the truth and predicted are
  84   n-rank Tensors with n > 2, then these are reshaped to 2-rank Tensors. It
  85   allows for truth to be specified as one hot vector or class indices. Finally,
  86   a weight can be specified for each element in truth and predicted.
  87 
  88   Args:
  89     truth: an n-rank or (n-1)-rank Tensor containing labels. If is_one_hot is
  90       True, then n-rank Tensor is expected, otherwise (n-1) rank one.
  91     predicted: an n-rank float Tensor containing prediction probabilities.
  92     weights: an (n-1)-rank float Tensor of weights
  93     is_one_hot: a boolean.
  94 
  95   Returns:
  96     A TF float scalar.
  97   """
  98   num_labels = predicted.get_shape().as_list()[-1]
  99   if not is_one_hot:
 100     truth = tf.reshape(truth, [-1])
 101     truth = tf.one_hot(
 102         truth, depth=num_labels, on_value=1.0, off_value=0.0, axis=-1)
 103   else:
 104     truth = tf.reshape(truth, [-1, num_labels])
 105   predicted = tf.reshape(predicted, [-1, num_labels])
 106   losses = tf.nn.softmax_cross_entropy_with_logits(
 107       labels=truth, logits=predicted)
 108   if weights is not None:
 109     losses = tf.boolean_mask(losses,
 110                              tf.cast(tf.reshape(weights, [-1]), dtype=tf.bool))
 111   return tf.reduce_mean(losses)

 114 class UnrolledTaskIOConfig(object):
 115   """Configuration of task inputs and outputs.
 116 
 117   A task can have multiple inputs, which define the context, and a task query
 118   which defines what is to be executed in this context. The desired execution
 119   is encoded in an output. The config defines the shapes of the inputs, the
 120   query and the outputs.
 121   """
 122 
 123   def __init__(self, inputs, output, query=None):
 124     """Constructs a Task input/output config.
 125 
 126     Args:
 127       inputs: a list of tuples. Each tuple represents the configuration of an
 128         input, with first element being the type (a string value) and the second
 129         element the shape.
 130       output: a tuple representing the configuration of the output.
 131       query: a tuple representing the configuration of the query. If no query,
 132         then None.
 133     """
 134     # A configuration of a single input, output or query. Consists of the type,
 135     # which can be one of the three specified above, and a shape. The shape must
 136     # be consistent with the type, e.g. if type == 'image', then shape is a 3
 137     # valued list.
 138     io_config = collections.namedtuple('IOConfig', ['type', 'shape'])
 139 
 140     def assert_config(config):
 141       if not isinstance(config, tuple):
 142         raise ValueError('config must be a tuple. Received {}'.format(
 143             type(config)))
 144       if len(config) != 2:
 145         raise ValueError('config must have 2 elements, has %d' % len(config))
 146       if not isinstance(config[0], tf.DType):
 147         raise ValueError('First element of config must be a tf.DType.')
 148       if not isinstance(config[1], list):
 149         raise ValueError('Second element of config must be a list.')
 150 
 151     assert isinstance(inputs, collections.OrderedDict)
 152     for modality_type in inputs:
 153       assert_config(inputs[modality_type])
 154     self._inputs = collections.OrderedDict(
 155         [(k, io_config(*value)) for k, value in inputs.iteritems()])
 156 
 157     if query is not None:
 158       assert_config(query)
 159       self._query = io_config(*query)
 160     else:
 161       self._query = None
 162 
 163     assert_config(output)
 164     self._output = io_config(*output)
 165 
 166   @property
 167   def inputs(self):
 168     return self._inputs
 169 
 170   @property
 171   def output(self):
 172     return self._output
 173 
 174   @property
 175   def query(self):
 176     return self._query

 179 class UnrolledTask(object):
 180   """An interface for a Task which can be unrolled during training.
 181 
 182   Each example is called episode and consists of inputs and target output, where
 183   the output can be considered as desired unrolled sequence of actions for the
 184   inputs. For the specified tasks, these action sequences are to be
 185   unambiguously definable.
 186   """
 187   __metaclass__ = abc.ABCMeta
 188 
 189   def __init__(self, config):
 190     assert isinstance(config, UnrolledTaskIOConfig)
 191     self._config = config
 192     # A dict of bookkeeping variables.
 193     self.info = {}
 194     # Tensorflow input is multithreaded and this lock is needed to prevent
 195     # race condition in the environment. Without the lock, non-thread safe
 196     # environments crash.
 197     self._lock = threading.Lock()
 198 
 199   @property
 200   def config(self):
 201     return self._config

 203   @abc.abstractmethod
 204   def episode(self):
 205     """Returns data needed to train and test a single episode.
 206 
 207     Each episode consists of inputs, which define the context of the task, a
 208     query which defines the task, and a target output, which defines a
 209     sequence of actions to be executed for this query. This sequence should not
 210     require feedback, i.e. can be predicted purely from input and query.]
 211 
 212     Returns:
 213       inputs, query, output, where inputs is a list of numpy arrays and query
 214       and output are numpy arrays. These arrays must be of shape and type as
 215       specified in the task configuration.
 216     """
 217     pass
 218 
 219   def reset(self, observation):
 220     """Called after the environment is reset."""
 221     pass

 223   def episode_batch(self, batch_size):
 224     """Returns a batch of episodes.
 225 
 226     Args:
 227       batch_size: size of batch.
 228 
 229     Returns:
 230       (inputs, query, output, masks) where inputs is list of numpy arrays and
 231       query, output, and mask are numpy arrays. These arrays must be of shape
 232       and type as specified in the task configuration with one additional
 233       preceding dimension corresponding to the batch.
 234 
 235     Raises:
 236       ValueError: if self.episode() returns illegal values.
 237     """
 238     batched_inputs = collections.OrderedDict(
 239         [[mtype, []] for mtype in self.config.inputs])
 240     batched_queries = []
 241     batched_outputs = []
 242     batched_masks = []
 243     for _ in range(int(batch_size)):
 244       with self._lock:
 245         # The episode function needs to be thread-safe. Since the current
 246         # implementation for the envs are not thread safe we need to have lock
 247         # the operations here.
 248         inputs, query, outputs = self.episode()
 249       if not isinstance(outputs, tuple):
 250         raise ValueError('Outputs return value must be tuple.')
 251       if len(outputs) != 2:
 252         raise ValueError('Output tuple must be of size 2.')
 253       if inputs is not None:
 254         for modality_type in batched_inputs:
 255           batched_inputs[modality_type].append(
 256               np.expand_dims(inputs[modality_type], axis=0))
 257 
 258       if query is not None:
 259         batched_queries.append(np.expand_dims(query, axis=0))
 260       batched_outputs.append(np.expand_dims(outputs[0], axis=0))
 261       if outputs[1] is not None:
 262         batched_masks.append(np.expand_dims(outputs[1], axis=0))
 263 
 264     batched_inputs = {
 265         k: np.concatenate(i, axis=0) for k, i in batched_inputs.iteritems()
 266     }
 267     if batched_queries:
 268       batched_queries = np.concatenate(batched_queries, axis=0)
 269     batched_outputs = np.concatenate(batched_outputs, axis=0)
 270     if batched_masks:
 271       batched_masks = np.concatenate(batched_masks, axis=0).astype(np.float32)
 272     else:
 273       # When the array is empty, the default np.dtype is float64 which causes
 274       # py_func to crash in the tests.
 275       batched_masks = np.array([], dtype=np.float32)
 276     batched_inputs = [batched_inputs[k] for k in self._config.inputs]
 277     return batched_inputs, batched_queries, batched_outputs, batched_masks
 278 
 279   def tf_episode_batch(self, batch_size):
 280     """A batch of episodes as TF Tensors.
 281 
 282     Same as episode_batch with the difference that the return values are TF
 283     Tensors.
 284 
 285     Args:
 286       batch_size: a python float for the batch size.
 287 
 288     Returns:
 289       inputs, query, output, mask where inputs is a dictionary of tf.Tensor
 290       where the keys are the modality types specified in the config.inputs.
 291       query, output, and mask are TF Tensors. These tensors must
 292       be of shape and type as specified in the task configuration with one
 293       additional preceding  dimension corresponding to the batch. Both mask and
 294       output have the same shape as output.
 295     """
 296 
 297     # Define TF outputs.
 298     touts = []
 299     shapes = []
 300     for _, i in self._config.inputs.iteritems():
 301       touts.append(i.type)
 302       shapes.append(i.shape)
 303     if self._config.query is not None:
 304       touts.append(self._config.query.type)
 305       shapes.append(self._config.query.shape)
 306     # Shapes and types for batched_outputs.
 307     touts.append(self._config.output.type)
 308     shapes.append(self._config.output.shape)
 309     # Shapes and types for batched_masks.
 310     touts.append(self._config.output.type)
 311     shapes.append(self._config.output.shape[0:1])
 312 
 313     def episode_batch_func():
 314       if self.config.query is None:
 315         inp, _, output, masks = self.episode_batch(int(batch_size))
 316         return tuple(inp) + (output, masks)
 317       else:
 318         inp, query, output, masks = self.episode_batch(int(batch_size))
 319         return tuple(inp) + (query, output, masks)
 320 
 321     tf_episode_batch = tf.py_func(episode_batch_func, [], touts,
 322                                   stateful=True, name='taskdata')
 323     for episode, shape in zip(tf_episode_batch, shapes):
 324       episode.set_shape([batch_size] + shape)
 325 
 326     tf_episode_batch_dict = collections.OrderedDict([
 327         (mtype, episode)
 328         for mtype, episode in zip(self.config.inputs.keys(), tf_episode_batch)
 329     ])
 330     cur_index = len(self.config.inputs.keys())
 331     tf_query = None
 332     if self.config.query is not None:
 333       tf_query = tf_episode_batch[cur_index]
 334       cur_index += 1
 335     tf_outputs = tf_episode_batch[cur_index]
 336     tf_masks = tf_episode_batch[cur_index + 1]
 337 
 338     return tf_episode_batch_dict, tf_query, tf_outputs, tf_masks

 380 class RandomExplorationBasedTask(UnrolledTask):
 381   """A Task which starts with a random exploration of the environment."""
 382 
 383   def __init__(self,
 384                env,
 385                seed,
 386                add_query_noise=False,
 387                query_noise_var=0.0,
 388                *args,
 389                **kwargs):  # pylint: disable=keyword-arg-before-vararg
 390     """Initializes a Task using a random exploration runs.
 391 
 392     Args:
 393       env: an instance of type TaskEnv and gym.Env.
 394       seed: a random seed.
 395       add_query_noise: boolean, if True then whatever queries are generated,
 396         they are randomly perturbed. The semantics of the queries depends on the
 397         concrete task implementation.
 398       query_noise_var: float, the variance of Gaussian noise used for query
 399         perturbation. Used iff add_query_noise==True.
 400       *args: see super class.
 401       **kwargs: see super class.
 402     """
 403     super(RandomExplorationBasedTask, self).__init__(*args, **kwargs)
 404     assert isinstance(env, task_env.TaskEnv)
 405     self._env = env
 406     self._env.set_task(self)
 407     self._rng = np.random.RandomState(seed)
 408     self._add_query_noise = add_query_noise
 409     self._query_noise_var = query_noise_var
 410 
 411     # GoToStaticXTask can also take empty config but for the rest of the classes
 412     # the number of modality types is 1.
 413     if len(self.config.inputs.keys()) > 1:
 414       raise NotImplementedError('current implementation supports input '
 415                                 'with only one modality type or less.')
 416 
 417   def _exploration(self):
 418     """Generates a random exploration run.
 419 
 420     The function uses the environment to generate a run.
 421 
 422     Returns:
 423       A tuple of numpy arrays. The i-th array contains observation of type and
 424       shape as specified in config.inputs[i].
 425       A list of states along the exploration path.
 426       A list of vertex indices corresponding to the path of the exploration.
 427     """
 428     in_seq_len = self._config.inputs.values()[0].shape[0]
 429     path, _, states, step_outputs = self._env.random_step_sequence(
 430         min_len=in_seq_len)
 431     obs = {modality_type: [] for modality_type in self._config.inputs}
 432     for o in step_outputs:
 433       step_obs, _, done, _ = o
 434       # It is expected that each value of step_obs is a dict of observations,
 435       # whose dimensions are consistent with the config.inputs sizes.
 436       for modality_type in self._config.inputs:
 437         assert modality_type in step_obs, '{}'.format(type(step_obs))
 438         o = step_obs[modality_type]
 439         i = self._config.inputs[modality_type]
 440         assert len(o.shape) == len(i.shape) - 1
 441         for dim_o, dim_i in zip(o.shape, i.shape[1:]):
 442           assert dim_o == dim_i, '{} != {}'.format(dim_o, dim_i)
 443         obs[modality_type].append(o)
 444       if done:
 445         break
 446 
 447     if not obs:
 448       return obs, states, path
 449 
 450     max_path_len = int(
 451         round(in_seq_len * float(len(path)) / float(len(obs.values()[0]))))
 452     path = path[-max_path_len:]
 453     states = states[-in_seq_len:]
 454 
 455     # The above obs is a list of tuples of np,array. Re-format them as tuple of
 456     # np.array, each array containing all observations from all steps.
 457     def regroup(obs, i):
 458       """Regroups observations.
 459 
 460       Args:
 461         obs: a list of tuples of same size. The k-th tuple contains all the
 462           observations from k-th step. Each observation is a numpy array.
 463         i: the index of the observation in each tuple to be grouped.
 464 
 465       Returns:
 466         A numpy array of shape config.inputs[i] which contains all i-th
 467         observations from all steps. These are concatenated along the first
 468         dimension. In addition, if the number of observations is different from
 469         the one specified in config.inputs[i].shape[0], then the array is either
 470         padded from front or clipped.
 471       """
 472       grouped_obs = np.concatenate(
 473           [np.expand_dims(o, axis=0) for o in obs[i]], axis=0)
 474       in_seq_len = self._config.inputs[i].shape[0]
 475       # pylint: disable=unbalanced-tuple-unpacking
 476       grouped_obs, _ = _pad_or_clip_array(
 477           grouped_obs, in_seq_len, is_front_clip=True)
 478       return grouped_obs
 479 
 480     all_obs = {i: regroup(obs, i) for i in self._config.inputs}
 481 
 482     return all_obs, states, path

 484   def _obs_to_state(self, path, states):
 485     """Computes mapping between path nodes and states."""
 486     # Generate a numpy array of locations corresponding to the path vertices.
 487     path_coordinates = map(self._env.vertex_to_pose, path)
 488     path_coordinates = np.concatenate(
 489         [np.reshape(p, [1, 2]) for p in path_coordinates])
 490 
 491     # The observations are taken along a smoothed trajectory following the path.
 492     # We compute a mapping between the obeservations and the map vertices.
 493     path_to_obs = collections.defaultdict(list)
 494     obs_to_state = []
 495     for i, s in enumerate(states):
 496       location = np.reshape(s[0:2], [1, 2])
 497       index = np.argmin(
 498           np.reshape(
 499               np.sum(np.power(path_coordinates - location, 2), axis=1), [-1]))
 500       index = path[index]
 501       path_to_obs[index].append(i)
 502       obs_to_state.append(index)
 503     return path_to_obs, obs_to_state
 504 
 505   def _perturb_state(self, state, noise_var):
 506     """Perturbes the state.
 507 
 508     The location are purturbed using a Gaussian noise with variance
 509     noise_var. The orientation is uniformly sampled.
 510 
 511     Args:
 512       state: a numpy array containing an env state (x, y locations).
 513       noise_var: float
 514     Returns:
 515       The perturbed state.
 516     """
 517 
 518     def normal(v, std):
 519       if std > 0:
 520         n = self._rng.normal(0.0, std)
 521         n = min(n, 2.0 * std)
 522         n = max(n, -2.0 * std)
 523         return v + n
 524       else:
 525         return v
 526 
 527     state = state.copy()
 528     state[0] = normal(state[0], noise_var)
 529     state[1] = normal(state[1], noise_var)
 530     if state.size > 2:
 531       state[2] = self._rng.uniform(-math.pi, math.pi)
 532     return state
 533 
 534   def _sample_obs(self,
 535                   indices,
 536                   observations,
 537                   observation_states,
 538                   path_to_obs,
 539                   max_obs_index=None,
 540                   use_exploration_obs=True):
 541     """Samples one observation which corresponds to vertex_index in path.
 542 
 543     In addition, the sampled observation must have index in observations less
 544     than max_obs_index. If these two conditions cannot be satisfied the
 545     function returns None.
 546 
 547     Args:
 548       indices: a list of integers.
 549       observations: a list of numpy arrays containing all the observations.
 550       observation_states: a list of numpy arrays, each array representing the
 551         state of the observation.
 552       path_to_obs: a dict of path indices to lists of observation indices.
 553       max_obs_index: an integer.
 554       use_exploration_obs: if True, then the observation is sampled among the
 555         specified observations, otherwise it is obtained from the environment.
 556     Returns:
 557       A tuple of:
 558         -- A numpy array of size width x height x 3 representing the sampled
 559           observation.
 560         -- The index of the sampld observation among the input observations.
 561         -- The state at which the observation is captured.
 562     Raises:
 563       ValueError: if the observation and observation_states lists are of
 564         different lengths.
 565     """
 566     if len(observations) != len(observation_states):
 567       raise ValueError('observation and observation_states lists must have '
 568                        'equal lengths')
 569     if not indices:
 570       return None, None, None
 571     vertex_index = self._rng.choice(indices)
 572     if use_exploration_obs:
 573       obs_indices = path_to_obs[vertex_index]
 574 
 575       if max_obs_index is not None:
 576         obs_indices = [i for i in obs_indices if i < max_obs_index]
 577 
 578       if obs_indices:
 579         index = self._rng.choice(obs_indices)
 580         if self._add_query_noise:
 581           xytheta = self._perturb_state(observation_states[index],
 582                                         self._query_noise_var)
 583           return self._env.observation(xytheta), index, xytheta
 584         else:
 585           return observations[index], index, observation_states[index]
 586       else:
 587         return None, None, None
 588     else:
 589       xy = self._env.vertex_to_pose(vertex_index)
 590       xytheta = np.array([xy[0], xy[1], 0.0])
 591       xytheta = self._perturb_state(xytheta, self._query_noise_var)
 592       return self._env.observation(xytheta), None, xytheta

 595 class AreNearbyTask(RandomExplorationBasedTask):
 596   """A task of identifying whether a query is nearby current location or not.
 597

 599   i.e. close to one of the observations. For each observation we have one

 601   """

 604       self,
 605       max_distance=0,
 606       *args,
 607       **kwargs):  # pylint: disable=keyword-arg-before-vararg
 608     super(AreNearbyTask, self).__init__(*args, **kwargs)
 609     self._max_distance = max_distance

 612       raise NotImplementedError('current implementation supports input '

 615   def episode(self):
 616     """Episode data.
 617 
 618     Returns:
 619       observations: a tuple with one element. This element is a numpy array of
 620         size in_seq_len x observation_size x observation_size x 3 containing
 621         in_seq_len images.
 622       query: a numpy array of size
 623         in_seq_len x observation_size X observation_size x 3 containing a query
 624         image.
 625       A tuple of size two. First element is a in_seq_len x 2 numpy array of
 626         either 1.0 or 0.0. The i-th element denotes whether the i-th query
 627         image is neraby (value 1.0) or not (value 0.0) to the i-th observation.
 628         The second element in the tuple is a mask, a numpy array of size
 629         in_seq_len x 1 and values 1.0 or 0.0 denoting whether the query is
 630         valid or not (it can happen that the query is not valid, e.g. there are
 631         not enough observations to have a meaningful queries).
 632     """
 633     observations, states, path = self._exploration()
 634     assert len(observations.values()[0]) == len(states)
 635 
 636     # The observations are taken along a smoothed trajectory following the path.
 637     # We compute a mapping between the obeservations and the map vertices.
 638     path_to_obs, obs_to_path = self._obs_to_state(path, states)
 639 
 640     # Go over all observations, and sample a query. With probability 0.5 this
 641     # query is a nearby observation (defined as belonging to the same vertex
 642     # in path).
 643     g = self._env.graph
 644     queries = []
 645     labels = []
 646     validity_masks = []
 647     query_index_in_observations = []
 648     for i, curr_o in enumerate(observations.values()[0]):
 649       p = obs_to_path[i]
 650       low = max(0, i - self._max_distance)
 651 
 652       # A list of lists of vertex indices. Each list in this group corresponds
 653       # to one possible label.
 654       index_groups = [[], [], []]
 655       # Nearby visited indices, label 1.
 656       nearby_visited = [
 657           ii for ii in path[low:i + 1] + g[p].keys() if ii in obs_to_path[:i]
 658       ]
 659       nearby_visited = [ii for ii in index_groups[1] if ii in path_to_obs]
 660       # NOT Nearby visited indices, label 0.
 661       not_nearby_visited = [ii for ii in path[:low] if ii not in g[p].keys()]
 662       not_nearby_visited = [ii for ii in index_groups[0] if ii in path_to_obs]
 663       # NOT visited indices, label 2.
 664       not_visited = [
 665           ii for ii in range(g.number_of_nodes()) if ii not in path[:i + 1]
 666       ]
 667 
 668       index_groups = [not_nearby_visited, nearby_visited, not_visited]
 669 
 670       # Consider only labels for which there are indices.
 671       allowed_labels = [ii for ii, group in enumerate(index_groups) if group]
 672       label = self._rng.choice(allowed_labels)
 673 
 674       indices = list(set(index_groups[label]))
 675       max_obs_index = None if label == 2 else i
 676       use_exploration_obs = False if label == 2 else True
 677       o, obs_index, _ = self._sample_obs(
 678           indices=indices,
 679           observations=observations.values()[0],
 680           observation_states=states,
 681           path_to_obs=path_to_obs,
 682           max_obs_index=max_obs_index,
 683           use_exploration_obs=use_exploration_obs)
 684       query_index_in_observations.append(obs_index)
 685 
 686       # If we cannot sample a valid query, we mark it as not valid in mask.
 687       if o is None:
 688         label = 0.0
 689         o = curr_o
 690         validity_masks.append(0)
 691       else:
 692         validity_masks.append(1)
 693 
 694       queries.append(o.values()[0])
 695       labels.append(label)
 696 
 697     query = np.concatenate([np.expand_dims(q, axis=0) for q in queries], axis=0)
 698 
 699     def one_hot(label, num_labels=3):
 700       a = np.zeros((num_labels,), dtype=np.float)
 701       a[int(label)] = 1.0
 702       return a
 703 
 704     outputs = np.stack([one_hot(l) for l in labels], axis=0)
 705     validity_mask = np.reshape(
 706         np.array(validity_masks, dtype=np.int32), [-1, 1])
 707 
 708     self.info['query_index_in_observations'] = query_index_in_observations
 709     self.info['observation_states'] = states
 710 
 711     return observations, query, (outputs, validity_mask)
 712

 714     pass

 940   def __init__(self, num_labels, *args, **kwargs):
 941     """Initializes a relative location task.
 942 
 943     Args:
 944       num_labels: integer, number of orientations to bin the relative
 945         orientation into.
 946       *args: see super class.
 947       **kwargs: see super class.
 948     """

 950     self._num_labels = num_labels

 957

 996 class LocationClassificationTask(UnrolledTask):
 997   """A task of classifying a location as one of several classes.
 998 
 999   The task does not have an input, but just a query and an output. The query
1000   is an observation of the current location, e.g. an image taken from the

1003

1009   """

1011   def __init__(self, env, seed, *args, **kwargs):
1012     super(LocationClassificationTask, self).__init__(*args, **kwargs)

1042       vertex = self._env.pose_to_vertex(location)

1048     def rotate(xy, theta):
1049       """Rotates a vector around the origin by angle theta.
1050 
1051       Args:
1052         xy: a numpy darray of shape (2, ) of floats containing the x and y
1053           coordinates of a vector.
1054         theta: a python float containing the rotation angle in radians.
1055 
1056       Returns:
1057         A numpy darray of floats of shape (2,) containing the x and y
1058           coordinates rotated xy.
1059       """
1060       rotated_x = np.cos(theta) * xy[0] - np.sin(theta) * xy[1]
1061       rotated_y = np.sin(theta) * xy[0] + np.cos(theta) * xy[1]
1062       return np.array([rotated_x, rotated_y])

1073       raise ValueError('Output shape should be of rank 1.')

1084     one_hot_label[int(label)] = 1.0
1085 
1086     query = self._env.observation(location).values()[0]
1087     return [], query, (one_hot_label, None)
