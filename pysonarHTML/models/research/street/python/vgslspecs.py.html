  31 # Class that builds a set of ops to manipulate variable-sized images.
  32 class VGSLSpecs(object):
  33   """Layers that can be built from a string definition."""

  35   def __init__(self, widths, heights, is_training):
  36     """Constructs a VGSLSpecs.
  37 
  38     Args:
  39       widths:  Tensor of size batch_size of the widths of the inputs.
  40       heights: Tensor of size batch_size of the heights of the inputs.
  41       is_training: True if the graph should be build for training.
  42     """
  43     # The string that was used to build this model.
  44     self.model_str = None
  45     # True if we are training
  46     self.is_training = is_training
  47     # Tensor for the size of the images, of size batch_size.
  48     self.widths = widths
  49     self.heights = heights
  50     # Overall reduction factors of this model so far for each dimension.
  51     # TODO(rays) consider building a graph from widths and heights instead of
  52     # computing a scale factor.
  53     self.reduction_factors = [1.0, 1.0, 1.0, 1.0]
  54     # List of Op parsers.
  55     # TODO(rays) add more Op types as needed.
  56     self.valid_ops = [self.AddSeries, self.AddParallel, self.AddConvLayer,
  57                       self.AddMaxPool, self.AddDropout, self.AddReShape,
  58                       self.AddFCLayer, self.AddLSTMLayer]
  59     # Translation table to convert unacceptable characters that may occur
  60     # in op strings that cannot be used as names.
  61     self.transtab = maketrans('(,)', '___')
  62 
  63   def Build(self, prev_layer, model_str):
  64     """Builds a network with input prev_layer from a VGSLSpecs description.
  65 
  66     Args:
  67       prev_layer: The input tensor.
  68       model_str:  Model definition similar to Tesseract as follows:
  69         ============ FUNCTIONAL OPS ============
  70         C(s|t|r|l|m)[{name}]<y>,<x>,<d> Convolves using a y,x window, with no
  71           shrinkage, SAME infill, d outputs, with s|t|r|l|m non-linear layer.
  72           (s|t|r|l|m) specifies the type of non-linearity:
  73           s = sigmoid
  74           t = tanh
  75           r = relu
  76           l = linear (i.e., None)
  77           m = softmax
  78         F(s|t|r|l|m)[{name}]<d> Fully-connected with s|t|r|l|m non-linearity and
  79           d outputs. Reduces height, width to 1. Input height and width must be
  80           constant.
  81         L(f|r|b)(x|y)[s][{name}]<n> LSTM cell with n outputs.
  82           f runs the LSTM forward only.
  83           r runs the LSTM reversed only.
  84           b runs the LSTM bidirectionally.
  85           x runs the LSTM in the x-dimension (on data with or without the
  86              y-dimension).
  87           y runs the LSTM in the y-dimension (data must have a y dimension).
  88           s (optional) summarizes the output in the requested dimension,
  89              outputting only the final step, collapsing the dimension to a
  90              single element.
  91           Examples:
  92           Lfx128 runs a forward-only LSTM in the x-dimension with 128
  93                  outputs, treating any y dimension independently.
  94           Lfys64 runs a forward-only LSTM in the y-dimension with 64 outputs
  95                  and collapses the y-dimension to 1 element.
  96           NOTE that Lbxsn is implemented as (LfxsnLrxsn) since the summaries
  97           need to be taken from opposite ends of the output
  98         Do[{name}] Insert a dropout layer.
  99         ============ PLUMBING OPS ============
 100         [...] Execute ... networks in series (layers).
 101         (...) Execute ... networks in parallel, with their output concatenated
 102           in depth.
 103         S[{name}]<d>(<a>x<b>)<e>,<f> Splits one dimension, moves one part to
 104           another dimension.
 105           Splits input dimension d into a x b, sending the high part (a) to the
 106           high side of dimension e, and the low part (b) to the high side of
 107           dimension f. Exception: if d=e=f, then then dimension d is internally
 108           transposed to bxa.
 109           Either a or b can be zero, meaning whatever is left after taking out
 110           the other, allowing dimensions to be of variable size.
 111           Eg. S3(3x50)2,3 will split the 150-element depth into 3x50, with the 3
 112           going to the most significant part of the width, and the 50 part
 113           staying in depth.
 114           This will rearrange a 3x50 output parallel operation to spread the 3
 115           output sets over width.
 116         Mp[{name}]<y>,<x> Maxpool the input, reducing the (y,x) rectangle to a
 117           single vector value.
 118 
 119     Returns:
 120       Output tensor
 121     """
 122     self.model_str = model_str
 123     final_layer, _ = self.BuildFromString(prev_layer, 0)
 124     return final_layer

 126   def GetLengths(self, dim=2, factor=1):
 127     """Returns the lengths of the batch of elements in the given dimension.
 128 
 129     WARNING: The returned sizes may not exactly match TF's calculation.
 130     Args:
 131       dim: dimension to get the sizes of, in [1,2]. batch, depth not allowed.
 132       factor: A scalar value to multiply by.
 133 
 134     Returns:
 135       The original heights/widths scaled by the current scaling of the model and
 136       the given factor.
 137 
 138     Raises:
 139       ValueError: If the args are invalid.
 140     """
 141     if dim == 1:
 142       lengths = self.heights
 143     elif dim == 2:
 144       lengths = self.widths
 145     else:
 146       raise ValueError('Invalid dimension given to GetLengths')
 147     lengths = tf.cast(lengths, tf.float32)
 148     if self.reduction_factors[dim] is not None:
 149       lengths = tf.div(lengths, self.reduction_factors[dim])
 150     else:
 151       lengths = tf.ones_like(lengths)
 152     if factor != 1:
 153       lengths = tf.multiply(lengths, tf.cast(factor, tf.float32))
 154     return tf.cast(lengths, tf.int32)
 155 
 156   def BuildFromString(self, prev_layer, index):
 157     """Adds the layers defined by model_str[index:] to the model.
 158 
 159     Args:
 160       prev_layer: Input tensor.
 161       index:      Position in model_str to start parsing
 162 
 163     Returns:
 164       Output tensor, next model_str index.
 165 
 166     Raises:
 167       ValueError: If the model string is unrecognized.
 168     """
 169     index = self._SkipWhitespace(index)
 170     for op in self.valid_ops:
 171       output_layer, next_index = op(prev_layer, index)
 172       if output_layer is not None:
 173         return output_layer, next_index
 174     if output_layer is not None:
 175       return output_layer, next_index
 176     raise ValueError('Unrecognized model string:' + self.model_str[index:])
 177 
 178   def AddSeries(self, prev_layer, index):
 179     """Builds a sequence of layers for a VGSLSpecs model.
 180 
 181     Args:
 182       prev_layer: Input tensor.
 183       index:      Position in model_str to start parsing
 184 
 185     Returns:
 186       Output tensor of the series, end index in model_str.
 187 
 188     Raises:
 189       ValueError: If [] are unbalanced.
 190     """
 191     if self.model_str[index] != '[':
 192       return None, None
 193     index += 1
 194     while index < len(self.model_str) and self.model_str[index] != ']':
 195       prev_layer, index = self.BuildFromString(prev_layer, index)
 196     if index == len(self.model_str):
 197       raise ValueError('Missing ] at end of series!' + self.model_str)
 198     return prev_layer, index + 1

 200   def AddParallel(self, prev_layer, index):
 201     """tf.concats outputs of layers that run on the same inputs.
 202 
 203     Args:
 204       prev_layer: Input tensor.
 205       index:      Position in model_str to start parsing
 206 
 207     Returns:
 208       Output tensor of the parallel,  end index in model_str.
 209 
 210     Raises:
 211       ValueError: If () are unbalanced or the elements don't match.
 212     """
 213     if self.model_str[index] != '(':
 214       return None, None
 215     index += 1
 216     layers = []
 217     num_dims = 0
 218     # Each parallel must output the same, including any reduction factor, in
 219     # all dimensions except depth.
 220     # We have to save the starting factors, so they don't get reduced by all
 221     # the elements of the parallel, only once.
 222     original_factors = self.reduction_factors
 223     final_factors = None
 224     while index < len(self.model_str) and self.model_str[index] != ')':
 225       self.reduction_factors = original_factors
 226       layer, index = self.BuildFromString(prev_layer, index)
 227       if num_dims == 0:
 228         num_dims = len(layer.get_shape())
 229       elif num_dims != len(layer.get_shape()):
 230         raise ValueError('All elements of parallel must return same num dims')
 231       layers.append(layer)
 232       if final_factors:
 233         if final_factors != self.reduction_factors:
 234           raise ValueError('All elements of parallel must scale the same')
 235       else:
 236         final_factors = self.reduction_factors
 237     if index == len(self.model_str):
 238       raise ValueError('Missing ) at end of parallel!' + self.model_str)
 239     return tf.concat(axis=num_dims - 1, values=layers), index + 1

 241   def AddConvLayer(self, prev_layer, index):
 242     """Add a single standard convolutional layer.
 243 
 244     Args:
 245       prev_layer: Input tensor.
 246       index:      Position in model_str to start parsing
 247 
 248     Returns:
 249       Output tensor, end index in model_str.
 250     """
 251     pattern = re.compile(R'(C)(s|t|r|l|m)({\w+})?(\d+),(\d+),(\d+)')
 252     m = pattern.match(self.model_str, index)
 253     if m is None:
 254       return None, None
 255     name = self._GetLayerName(m.group(0), index, m.group(3))
 256     width = int(m.group(4))
 257     height = int(m.group(5))
 258     depth = int(m.group(6))
 259     fn = self._NonLinearity(m.group(2))
 260     return slim.conv2d(
 261         prev_layer, depth, [height, width], activation_fn=fn,
 262         scope=name), m.end()
 263 
 264   def AddMaxPool(self, prev_layer, index):
 265     """Add a maxpool layer.
 266 
 267     Args:
 268       prev_layer: Input tensor.
 269       index:      Position in model_str to start parsing
 270 
 271     Returns:
 272       Output tensor, end index in model_str.
 273     """
 274     pattern = re.compile(R'(Mp)({\w+})?(\d+),(\d+)(?:,(\d+),(\d+))?')
 275     m = pattern.match(self.model_str, index)
 276     if m is None:
 277       return None, None
 278     name = self._GetLayerName(m.group(0), index, m.group(2))
 279     height = int(m.group(3))
 280     width = int(m.group(4))
 281     y_stride = height if m.group(5) is None else m.group(5)
 282     x_stride = width if m.group(6) is None else m.group(6)
 283     self.reduction_factors[1] *= y_stride
 284     self.reduction_factors[2] *= x_stride
 285     return slim.max_pool2d(
 286         prev_layer, [height, width], [y_stride, x_stride],
 287         padding='SAME',
 288         scope=name), m.end()
 289 
 290   def AddDropout(self, prev_layer, index):
 291     """Adds a dropout layer.
 292 
 293     Args:
 294       prev_layer: Input tensor.
 295       index:      Position in model_str to start parsing
 296 
 297     Returns:
 298       Output tensor, end index in model_str.
 299     """
 300     pattern = re.compile(R'(Do)({\w+})?')
 301     m = pattern.match(self.model_str, index)
 302     if m is None:
 303       return None, None
 304     name = self._GetLayerName(m.group(0), index, m.group(2))
 305     layer = slim.dropout(
 306         prev_layer, 0.5, is_training=self.is_training, scope=name)
 307     return layer, m.end()

 309   def AddReShape(self, prev_layer, index):
 310     """Reshapes the input tensor by moving each (x_scale,y_scale) rectangle to.
 311 
 312        the depth dimension. NOTE that the TF convention is that inputs are
 313        [batch, y, x, depth].
 314 
 315     Args:
 316       prev_layer: Input tensor.
 317       index:      Position in model_str to start parsing
 318 
 319     Returns:
 320       Output tensor, end index in model_str.
 321     """
 322     pattern = re.compile(R'(S)(?:{(\w)})?(\d+)\((\d+)x(\d+)\)(\d+),(\d+)')
 323     m = pattern.match(self.model_str, index)
 324     if m is None:
 325       return None, None
 326     name = self._GetLayerName(m.group(0), index, m.group(2))
 327     src_dim = int(m.group(3))
 328     part_a = int(m.group(4))
 329     part_b = int(m.group(5))
 330     dest_dim_a = int(m.group(6))
 331     dest_dim_b = int(m.group(7))
 332     if part_a == 0:
 333       part_a = -1
 334     if part_b == 0:
 335       part_b = -1
 336     prev_shape = tf.shape(prev_layer)
 337     layer = shapes.transposing_reshape(
 338         prev_layer, src_dim, part_a, part_b, dest_dim_a, dest_dim_b, name=name)
 339     # Compute scale factors.
 340     result_shape = tf.shape(layer)
 341     for i in xrange(len(self.reduction_factors)):
 342       if self.reduction_factors[i] is not None:
 343         factor1 = tf.cast(self.reduction_factors[i], tf.float32)
 344         factor2 = tf.cast(prev_shape[i], tf.float32)
 345         divisor = tf.cast(result_shape[i], tf.float32)
 346         self.reduction_factors[i] = tf.div(tf.multiply(factor1, factor2), divisor)
 347     return layer, m.end()

 349   def AddFCLayer(self, prev_layer, index):
 350     """Parse expression and add Fully Connected Layer.
 351 
 352     Args:
 353       prev_layer: Input tensor.
 354       index:      Position in model_str to start parsing
 355 
 356     Returns:
 357       Output tensor, end index in model_str.
 358     """
 359     pattern = re.compile(R'(F)(s|t|r|l|m)({\w+})?(\d+)')
 360     m = pattern.match(self.model_str, index)
 361     if m is None:
 362       return None, None
 363     fn = self._NonLinearity(m.group(2))
 364     name = self._GetLayerName(m.group(0), index, m.group(3))
 365     depth = int(m.group(4))
 366     input_depth = shapes.tensor_dim(prev_layer, 1) * shapes.tensor_dim(
 367         prev_layer, 2) * shapes.tensor_dim(prev_layer, 3)
 368     # The slim fully connected is actually a 1x1 conv, so we have to crush the
 369     # dimensions on input.
 370     # Everything except batch goes to depth, and therefore has to be known.
 371     shaped = tf.reshape(
 372         prev_layer, [-1, input_depth], name=name + '_reshape_in')
 373     output = slim.fully_connected(shaped, depth, activation_fn=fn, scope=name)
 374     # Width and height are collapsed to 1.
 375     self.reduction_factors[1] = None
 376     self.reduction_factors[2] = None
 377     return tf.reshape(
 378         output, [shapes.tensor_dim(prev_layer, 0), 1, 1, depth],
 379         name=name + '_reshape_out'), m.end()
 380 
 381   def AddLSTMLayer(self, prev_layer, index):
 382     """Parse expression and add LSTM Layer.
 383 
 384     Args:
 385       prev_layer: Input tensor.
 386       index:      Position in model_str to start parsing
 387 
 388     Returns:
 389       Output tensor, end index in model_str.
 390     """
 391     pattern = re.compile(R'(L)(f|r|b)(x|y)(s)?({\w+})?(\d+)')
 392     m = pattern.match(self.model_str, index)
 393     if m is None:
 394       return None, None
 395     direction = m.group(2)
 396     dim = m.group(3)
 397     summarize = m.group(4) == 's'
 398     name = self._GetLayerName(m.group(0), index, m.group(5))
 399     depth = int(m.group(6))
 400     if direction == 'b' and summarize:
 401       fwd = self._LSTMLayer(prev_layer, 'forward', dim, True, depth,
 402                             name + '_forward')
 403       back = self._LSTMLayer(prev_layer, 'backward', dim, True, depth,
 404                              name + '_reverse')
 405       return tf.concat(axis=3, values=[fwd, back], name=name + '_concat'), m.end()
 406     if direction == 'f':
 407       direction = 'forward'
 408     elif direction == 'r':
 409       direction = 'backward'
 410     else:
 411       direction = 'bidirectional'
 412     outputs = self._LSTMLayer(prev_layer, direction, dim, summarize, depth,
 413                               name)
 414     if summarize:
 415       # The x or y dimension is getting collapsed.
 416       if dim == 'x':
 417         self.reduction_factors[2] = None
 418       else:
 419         self.reduction_factors[1] = None
 420     return outputs, m.end()

 422   def _LSTMLayer(self, prev_layer, direction, dim, summarize, depth, name):
 423     """Adds an LSTM layer with the given pre-parsed attributes.
 424 
 425     Always maps 4-D to 4-D regardless of summarize.
 426     Args:
 427       prev_layer: Input tensor.
 428       direction:  'forward' 'backward' or 'bidirectional'
 429       dim:        'x' or 'y', dimension to consider as time.
 430       summarize:  True if we are to return only the last timestep.
 431       depth:      Output depth.
 432       name:       Some string naming the op.
 433 
 434     Returns:
 435       Output tensor.
 436     """
 437     # If the target dimension is y, we need to transpose.
 438     if dim == 'x':
 439       lengths = self.GetLengths(2, 1)
 440       inputs = prev_layer
 441     else:
 442       lengths = self.GetLengths(1, 1)
 443       inputs = tf.transpose(prev_layer, [0, 2, 1, 3], name=name + '_ytrans_in')
 444     input_batch = shapes.tensor_dim(inputs, 0)
 445     num_slices = shapes.tensor_dim(inputs, 1)
 446     num_steps = shapes.tensor_dim(inputs, 2)
 447     input_depth = shapes.tensor_dim(inputs, 3)
 448     # Reshape away the other dimension.
 449     inputs = tf.reshape(
 450         inputs, [-1, num_steps, input_depth], name=name + '_reshape_in')
 451     # We need to replicate the lengths by the size of the other dimension, and
 452     # any changes that have been made to the batch dimension.
 453     tile_factor = tf.to_float(input_batch *
 454                               num_slices) / tf.to_float(tf.shape(lengths)[0])
 455     lengths = tf.tile(lengths, [tf.cast(tile_factor, tf.int32)])
 456     lengths = tf.cast(lengths, tf.int64)
 457     outputs = nn_ops.rnn_helper(
 458         inputs,
 459         lengths,
 460         cell_type='lstm',
 461         num_nodes=depth,
 462         direction=direction,
 463         name=name,
 464         stddev=0.1)
 465     # Output depth is doubled if bi-directional.
 466     if direction == 'bidirectional':
 467       output_depth = depth * 2
 468     else:
 469       output_depth = depth
 470     # Restore the other dimension.
 471     if summarize:
 472       outputs = tf.slice(
 473           outputs, [0, num_steps - 1, 0], [-1, 1, -1], name=name + '_sum_slice')
 474       outputs = tf.reshape(
 475           outputs, [input_batch, num_slices, 1, output_depth],
 476           name=name + '_reshape_out')
 477     else:
 478       outputs = tf.reshape(
 479           outputs, [input_batch, num_slices, num_steps, output_depth],
 480           name=name + '_reshape_out')
 481     if dim == 'y':
 482       outputs = tf.transpose(outputs, [0, 2, 1, 3], name=name + '_ytrans_out')
 483     return outputs

 485   def _NonLinearity(self, code):
 486     """Returns the non-linearity function pointer for the given string code.
 487 
 488     For forwards compatibility, allows the full names for stand-alone
 489     non-linearities, as well as the single-letter names used in ops like C,F.
 490     Args:
 491       code: String code representing a non-linearity function.
 492     Returns:
 493       non-linearity function represented by the code.
 494     """
 495     if code in ['s', 'Sig']:
 496       return tf.sigmoid
 497     elif code in ['t', 'Tanh']:
 498       return tf.tanh
 499     elif code in ['r', 'Relu']:
 500       return tf.nn.relu
 501     elif code in ['m', 'Smax']:
 502       return tf.nn.softmax
 503     return None
 504 
 505   def _GetLayerName(self, op_str, index, name_str):
 506     """Generates a name for the op, using a user-supplied name if possible.
 507 
 508     Args:
 509       op_str:     String representing the parsed op.
 510       index:      Position in model_str of the start of the op.
 511       name_str:   User-supplied {name} with {} that need removing or None.
 512 
 513     Returns:
 514       Selected name.
 515     """
 516     if name_str:
 517       return name_str[1:-1]
 518     else:
 519       return op_str.translate(self.transtab) + '_' + str(index)

 521   def _SkipWhitespace(self, index):
 522     """Skips any leading whitespace in the model description.
 523 
 524     Args:
 525       index:      Position in model_str to start parsing
 526 
 527     Returns:
 528       end index in model_str of whitespace.
 529     """
 530     pattern = re.compile(R'([ \t\n]+)')
 531     m = pattern.match(self.model_str, index)
 532     if m is None:
 533       return index
 534     return m.end()
