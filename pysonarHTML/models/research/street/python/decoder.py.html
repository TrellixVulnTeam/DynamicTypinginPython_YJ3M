  16 """Basic CTC+recoder decoder.
  17 
  18 Decodes a sequence of class-ids into UTF-8 text.
  19 For basic information on CTC See:
  20 Alex Graves et al. Connectionist Temporal Classification: Labelling Unsegmented
  21 Sequence Data with Recurrent Neural Networks.
  22 http://www.cs.toronto.edu/~graves/icml_2006.pdf
  23 """
  24 import collections
  25 import re
  26 
  27 import errorcounter as ec
  28 from six.moves import xrange
  29 import tensorflow as tf
  30 
  31 # Named tuple Part describes a part of a multi (1 or more) part code that
  32 # represents a utf-8 string. For example, Chinese character 'x' might be
  33 # represented by 3 codes of which (utf8='x', index=1, num_codes3) would be the
  34 # middle part. (The actual code is not stored in the tuple).
  35 Part = collections.namedtuple('Part', 'utf8 index, num_codes')
  36 
  37 
  38 # Class that decodes a sequence of class-ids into UTF-8 text.
  39 class Decoder(object):
  40   """Basic CTC+recoder decoder."""
  41 
  42   def __init__(self, filename):
  43     r"""Constructs a Decoder.
  44 
  45     Reads the text file describing the encoding and build the encoder.
  46     The text file contains lines of the form:
  47     <code>[,<code>]*\t<string>
  48     Each line defines a mapping from a sequence of one or more integer codes to
  49     a corresponding utf-8 string.
  50     Args:
  51       filename:   Name of file defining the decoding sequences.
  52     """
  53     # self.decoder is a list of lists of Part(utf8, index, num_codes).
  54     # The index to the top-level list is a code. The list given by the code
  55     # index is a list of the parts represented by that code, Eg if the code 42
  56     # represents the 2nd (index 1) out of 3 part of Chinese character 'x', then
  57     # self.decoder[42] = [..., (utf8='x', index=1, num_codes3), ...] where ...
  58     # means all other uses of the code 42.
  59     self.decoder = []
  60     if filename:
  61       self._InitializeDecoder(filename)
  62 
  63   def SoftmaxEval(self, sess, model, num_steps):
  64     """Evaluate a model in softmax mode.
  65 
  66     Adds char, word recall and sequence error rate events to the sw summary
  67     writer, and returns them as well
  68     TODO(rays) Add LogisticEval.
  69     Args:
  70       sess:  A tensor flow Session.
  71       model: The model to run in the session. Requires a VGSLImageModel or any
  72         other class that has a using_ctc attribute and a RunAStep(sess) method
  73         that reurns a softmax result with corresponding labels.
  74       num_steps: Number of steps to evaluate for.
  75     Returns:
  76       ErrorRates named tuple.
  77     Raises:
  78       ValueError: If an unsupported number of dimensions is used.
  79     """
  80     coord = tf.train.Coordinator()
  81     threads = tf.train.start_queue_runners(sess=sess, coord=coord)
  82     # Run the requested number of evaluation steps, gathering the outputs of the
  83     # softmax and the true labels of the evaluation examples.
  84     total_label_counts = ec.ErrorCounts(0, 0, 0, 0)
  85     total_word_counts = ec.ErrorCounts(0, 0, 0, 0)
  86     sequence_errors = 0
  87     for _ in xrange(num_steps):
  88       softmax_result, labels = model.RunAStep(sess)
  89       # Collapse softmax to same shape as labels.
  90       predictions = softmax_result.argmax(axis=-1)
  91       # Exclude batch from num_dims.
  92       num_dims = len(predictions.shape) - 1
  93       batch_size = predictions.shape[0]
  94       null_label = softmax_result.shape[-1] - 1
  95       for b in xrange(batch_size):
  96         if num_dims == 2:
  97           # TODO(rays) Support 2-d data.
  98           raise ValueError('2-d label data not supported yet!')
  99         else:
 100           if num_dims == 1:
 101             pred_batch = predictions[b, :]
 102             labels_batch = labels[b, :]
 103           else:
 104             pred_batch = [predictions[b]]
 105             labels_batch = [labels[b]]
 106           text = self.StringFromCTC(pred_batch, model.using_ctc, null_label)
 107           truth = self.StringFromCTC(labels_batch, False, null_label)
 108           # Note that recall_errs is false negatives (fn) aka drops/deletions.
 109           # Actual recall would be 1-fn/truth_words.
 110           # Likewise precision_errs is false positives (fp) aka adds/insertions.
 111           # Actual precision would be 1-fp/ocr_words.
 112           total_word_counts = ec.AddErrors(total_word_counts,
 113                                            ec.CountWordErrors(text, truth))
 114           total_label_counts = ec.AddErrors(total_label_counts,
 115                                             ec.CountErrors(text, truth))
 116           if text != truth:
 117             sequence_errors += 1
 118 
 119     coord.request_stop()
 120     coord.join(threads)
 121     return ec.ComputeErrorRates(total_label_counts, total_word_counts,
 122                                 sequence_errors, num_steps * batch_size)

 124   def StringFromCTC(self, ctc_labels, merge_dups, null_label):
 125     """Decodes CTC output to a string.
 126 
 127     Extracts only sequences of codes that are allowed by self.decoder.
 128     Labels that make illegal code sequences are dropped.
 129     Note that, by its nature of taking only top choices, this is much weaker
 130     than a full-blown beam search that considers all the softmax outputs.
 131     For languages without many multi-code sequences, this doesn't make much
 132     difference, but for complex scripts the accuracy will be much lower.
 133     Args:
 134       ctc_labels: List of class labels including null characters to remove.
 135       merge_dups: If True, Duplicate labels will be merged
 136       null_label: Label value to ignore.
 137 
 138     Returns:
 139       Labels decoded to a string.
 140     """
 141     # Run regular ctc on the labels, extracting a list of codes.
 142     codes = self._CodesFromCTC(ctc_labels, merge_dups, null_label)
 143     length = len(codes)
 144     if length == 0:
 145       return ''
 146     # strings and partials are both indexed by the same index as codes.
 147     # strings[i] is the best completed string upto position i, and
 148     # partials[i] is a list of partial code sequences at position i.
 149     # Warning: memory is squared-order in length.
 150     strings = []
 151     partials = []
 152     for pos in xrange(length):
 153       code = codes[pos]
 154       parts = self.decoder[code]
 155       partials.append([])
 156       strings.append('')
 157       # Iterate over the parts that this code can represent.
 158       for utf8, index, num_codes in parts:
 159         if index > pos:
 160           continue
 161         # We can use code if it is an initial code (index==0) or continues a
 162         # sequence in the partials list at the previous position.
 163         if index == 0 or partials[pos - 1].count(
 164             Part(utf8, index - 1, num_codes)) > 0:
 165           if index < num_codes - 1:
 166             # Save the partial sequence.
 167             partials[-1].append(Part(utf8, index, num_codes))
 168           elif not strings[-1]:
 169             # A code sequence is completed. Append to the best string that we
 170             # had where it started.
 171             if pos >= num_codes:
 172               strings[-1] = strings[pos - num_codes] + utf8
 173             else:
 174               strings[-1] = utf8
 175       if not strings[-1] and pos > 0:
 176         # We didn't get anything here so copy the previous best string, skipping
 177         # the current code, but it may just be a partial anyway.
 178         strings[-1] = strings[-2]
 179     return strings[-1]

 181   def _InitializeDecoder(self, filename):
 182     """Reads the decoder file and initializes self.decoder from it.
 183 
 184     Args:
 185       filename: Name of text file mapping codes to utf8 strings.
 186     Raises:
 187       ValueError: if the input file is not parsed correctly.
 188     """
 189     line_re = re.compile(r'(?P<codes>\d+(,\d+)*)\t(?P<utf8>.+)')
 190     with tf.gfile.GFile(filename) as f:
 191       for line in f:
 192         m = line_re.match(line)
 193         if m is None:
 194           raise ValueError('Unmatched line:', line)
 195         # codes is the sequence that maps to the string.
 196         str_codes = m.groupdict()['codes'].split(',')
 197         codes = []
 198         for code in str_codes:
 199           codes.append(int(code))
 200         utf8 = m.groupdict()['utf8']
 201         num_codes = len(codes)
 202         for index, code in enumerate(codes):
 203           while code >= len(self.decoder):
 204             self.decoder.append([])
 205           self.decoder[code].append(Part(utf8, index, num_codes))
 206 
 207   def _CodesFromCTC(self, ctc_labels, merge_dups, null_label):
 208     """Collapses CTC output to regular output.
 209 
 210     Args:
 211       ctc_labels: List of class labels including null characters to remove.
 212       merge_dups: If True, Duplicate labels will be merged.
 213       null_label: Label value to ignore.
 214 
 215     All trailing zeros are removed!!
 216     TODO(rays) This may become a problem with non-CTC models.
 217     If using charset, this should not be a problem as zero is always space.
 218     tf.pad can only append zero, so we have to be able to drop them, as a
 219     non-ctc will have learned to output trailing zeros instead of trailing
 220     nulls. This is awkward, as the stock ctc loss function requires that the
 221     null character be num_classes-1.
 222     Returns:
 223       (List of) Labels with null characters removed.
 224     """
 225     out_labels = []
 226     prev_label = -1
 227     zeros_needed = 0
 228     for label in ctc_labels:
 229       if label == null_label:
 230         prev_label = -1
 231       elif label != prev_label or not merge_dups:
 232         if label == 0:
 233           # Count zeros and only emit them when it is clear there is a non-zero
 234           # after, so as to truncate away all trailing zeros.
 235           zeros_needed += 1
 236         else:
 237           if merge_dups and zeros_needed > 0:
 238             out_labels.append(0)
 239           else:
 240             out_labels += [0] * zeros_needed
 241           zeros_needed = 0
 242           out_labels.append(label)
 243         prev_label = label
 244     return out_labels
