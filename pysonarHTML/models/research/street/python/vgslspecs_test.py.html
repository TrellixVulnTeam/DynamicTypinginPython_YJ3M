  15 """Tests for vgslspecs."""
  16 
  17 import numpy as np
  18 import tensorflow as tf
  19 import vgslspecs
  20 
  21 
  22 def _rand(*size):
  23   return np.random.uniform(size=size).astype('f')
  24 
  25 
  26 class VgslspecsTest(tf.test.TestCase):
  27 
  28   def __init__(self, other):
  29     super(VgslspecsTest, self).__init__(other)
  30     self.max_width = 36
  31     self.max_height = 24
  32     self.batch_size = 4
  33 
  34   def SetupInputs(self):
  35     # Make placeholders for standard inputs.
  36     # Everything is variable in the input, except the depth.
  37     self.ph_image = tf.placeholder(
  38         tf.float32, shape=(None, None, None, 3), name='inputs')
  39     self.ph_widths = tf.placeholder(tf.int64, shape=(None,), name='w')
  40     self.ph_heights = tf.placeholder(tf.int64, shape=(None,), name='h')
  41     # Make actual inputs.
  42     self.in_image = _rand(self.batch_size, self.max_height, self.max_width, 3)
  43     self.in_widths = [24, 12, self.max_width, 30]
  44     self.in_heights = [self.max_height, 18, 12, 6]
  45 
  46   def ExpectScaledSize(self, spec, target_shape, factor=1):
  47     """Tests that the output of the graph of the given spec has target_shape."""
  48     with tf.Graph().as_default():
  49       with self.test_session() as sess:
  50         self.SetupInputs()
  51         # Only the placeholders are given at construction time.
  52         vgsl = vgslspecs.VGSLSpecs(self.ph_widths, self.ph_heights, True)
  53         outputs = vgsl.Build(self.ph_image, spec)
  54         # Compute the expected output widths from the given scale factor.
  55         target_widths = tf.div(self.in_widths, factor).eval()
  56         target_heights = tf.div(self.in_heights, factor).eval()
  57         # Run with the 'real' data.
  58         tf.global_variables_initializer().run()
  59         res_image, res_widths, res_heights = sess.run(
  60             [outputs, vgsl.GetLengths(2), vgsl.GetLengths(1)],
  61             feed_dict={self.ph_image: self.in_image,
  62                        self.ph_widths: self.in_widths,
  63                        self.ph_heights: self.in_heights})
  64         self.assertEqual(tuple(res_image.shape), target_shape)
  65         if target_shape[1] > 1:
  66           self.assertEqual(tuple(res_heights), tuple(target_heights))
  67         if target_shape[2] > 1:
  68           self.assertEqual(tuple(res_widths), tuple(target_widths))
  69 
  70   def testSameSizeConv(self):
  71     """Test all types of Conv. There is no scaling."""
  72     self.ExpectScaledSize(
  73         '[Cs{MyConv}5,5,16 Ct3,3,12 Cr4,4,24 Cl5,5,64]',
  74         (self.batch_size, self.max_height, self.max_width, 64))
  75 
  76   def testSameSizeLSTM(self):
  77     """Test all non-reducing LSTMs. Output depth is doubled with BiDi."""
  78     self.ExpectScaledSize('[Lfx16 Lrx8 Do Lbx24 Lfy12 Do{MyDo} Lry7 Lby32]',
  79                           (self.batch_size, self.max_height, self.max_width,
  80                            64))
  81 
  82   def testSameSizeParallel(self):
  83     """Parallel affects depth, but not scale."""
  84     self.ExpectScaledSize('[Cs5,5,16 (Lfx{MyLSTM}32 Lrx32 Lbx16)]',
  85                           (self.batch_size, self.max_height, self.max_width,
  86                            96))
  87 
  88   def testScalingOps(self):
  89     """Test a heterogeneous series with scaling."""
  90     self.ExpectScaledSize('[Cs5,5,16 Mp{MyPool}2,2 Ct3,3,32 Mp3,3 Lfx32 Lry64]',
  91                           (self.batch_size, self.max_height / 6,
  92                            self.max_width / 6, 64), 6)
  93 
  94   def testXReduction(self):
  95     """Test a heterogeneous series with reduction of x-dimension."""
  96     self.ExpectScaledSize('[Cr5,5,16 Mp2,2 Ct3,3,32 Mp3,3 Lfxs32 Lry64]',
  97                           (self.batch_size, self.max_height / 6, 1, 64), 6)
  98 
  99   def testYReduction(self):
 100     """Test a heterogeneous series with reduction of y-dimension."""
 101     self.ExpectScaledSize('[Cl5,5,16 Mp2,2 Ct3,3,32 Mp3,3 Lfys32 Lfx64]',
 102                           (self.batch_size, 1, self.max_width / 6, 64), 6)
 103 
 104   def testXYReduction(self):
 105     """Test a heterogeneous series with reduction to 0-d."""
 106     self.ExpectScaledSize(
 107         '[Cr5,5,16 Lfys32 Lfxs64 Fr{MyFC}16 Ft20 Fl12 Fs32 Fm40]',
 108         (self.batch_size, 1, 1, 40))
 109 
 110   def testReshapeTile(self):
 111     """Tests that a tiled input can be reshaped to the batch dimension."""
 112     self.ExpectScaledSize('[S2(3x0)0,2 Cr5,5,16 Lfys16]',
 113                           (self.batch_size * 3, 1, self.max_width / 3, 16), 3)
 114 
 115   def testReshapeDepth(self):
 116     """Tests that depth can be reshaped to the x dimension."""
 117     self.ExpectScaledSize('[Cl5,5,16 Mp3,3 (Lrys32 Lbys16 Lfys32) S3(3x0)2,3]',
 118                           (self.batch_size, 1, self.max_width, 32))
