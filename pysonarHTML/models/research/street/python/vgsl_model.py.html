  33 # Parameters for rate decay.
  34 # We divide the learning_rate_halflife by DECAY_STEPS_FACTOR and use DECAY_RATE
  35 # as the decay factor for the learning rate, ie we use the DECAY_STEPS_FACTORth
  36 # root of 2 as the decay rate every halflife/DECAY_STEPS_FACTOR to achieve the
  37 # desired halflife.
  38 DECAY_STEPS_FACTOR = 16
  39 DECAY_RATE = pow(0.5, 1.0 / DECAY_STEPS_FACTOR)

  42 def Train(train_dir,
  43           model_str,
  44           train_data,
  45           max_steps,
  46           master='',
  47           task=0,
  48           ps_tasks=0,
  49           initial_learning_rate=0.001,
  50           final_learning_rate=0.001,
  51           learning_rate_halflife=160000,
  52           optimizer_type='Adam',
  53           num_preprocess_threads=1,
  54           reader=None):
  55   """Testable trainer with no dependence on FLAGS.
  56 
  57   Args:
  58     train_dir: Directory to write checkpoints.
  59     model_str: Network specification string.
  60     train_data: Training data file pattern.
  61     max_steps: Number of training steps to run.
  62     master: Name of the TensorFlow master to use.
  63     task: Task id of this replica running the training. (0 will be master).
  64     ps_tasks: Number of tasks in ps job, or 0 if no ps job.
  65     initial_learning_rate: Learing rate at start of training.
  66     final_learning_rate: Asymptotic minimum learning rate.
  67     learning_rate_halflife: Number of steps over which to halve the difference
  68       between initial and final learning rate.
  69     optimizer_type: One of 'GradientDescent', 'AdaGrad', 'Momentum', 'Adam'.
  70     num_preprocess_threads: Number of input threads.
  71     reader: Function that returns an actual reader to read Examples from input
  72       files. If None, uses tf.TFRecordReader().
  73   """
  74   if master.startswith('local'):
  75     device = tf.ReplicaDeviceSetter(ps_tasks)
  76   else:
  77     device = '/cpu:0'
  78   with tf.Graph().as_default():
  79     with tf.device(device):
  80       model = InitNetwork(train_data, model_str, 'train', initial_learning_rate,
  81                           final_learning_rate, learning_rate_halflife,
  82                           optimizer_type, num_preprocess_threads, reader)
  83 
  84       # Create a Supervisor.  It will take care of initialization, summaries,
  85       # checkpoints, and recovery.
  86       #
  87       # When multiple replicas of this program are running, the first one,
  88       # identified by --task=0 is the 'chief' supervisor.  It is the only one
  89       # that takes case of initialization, etc.
  90       sv = tf.train.Supervisor(
  91           logdir=train_dir,
  92           is_chief=(task == 0),
  93           saver=model.saver,
  94           save_summaries_secs=10,
  95           save_model_secs=30,
  96           recovery_wait_secs=5)
  97 
  98       step = 0
  99       while step < max_steps:
 100         try:
 101           # Get an initialized, and possibly recovered session.  Launch the
 102           # services: Checkpointing, Summaries, step counting.
 103           with sv.managed_session(master) as sess:
 104             while step < max_steps:
 105               _, step = model.TrainAStep(sess)
 106               if sv.coord.should_stop():
 107                 break
 108         except tf.errors.AbortedError as e:
 109           logging.error('Received error:%s', e)
 110           continue

 113 def Eval(train_dir,
 114          eval_dir,
 115          model_str,
 116          eval_data,
 117          decoder_file,
 118          num_steps,
 119          graph_def_file=None,
 120          eval_interval_secs=0,
 121          reader=None):
 122   """Restores a model from a checkpoint and evaluates it.
 123 
 124   Args:
 125     train_dir: Directory to find checkpoints.
 126     eval_dir: Directory to write summary events.
 127     model_str: Network specification string.
 128     eval_data: Evaluation data file pattern.
 129     decoder_file: File to read to decode the labels.
 130     num_steps: Number of eval steps to run.
 131     graph_def_file: File to write graph definition to for freezing.
 132     eval_interval_secs: How often to run evaluations, or once if 0.
 133     reader: Function that returns an actual reader to read Examples from input
 134       files. If None, uses tf.TFRecordReader().
 135   Returns:
 136     (char error rate, word recall error rate, sequence error rate) as percent.
 137   Raises:
 138     ValueError: If unimplemented feature is used.
 139   """
 140   decode = None
 141   if decoder_file:
 142     decode = decoder.Decoder(decoder_file)
 143 
 144   # Run eval.
 145   rates = ec.ErrorRates(
 146       label_error=None,
 147       word_recall_error=None,
 148       word_precision_error=None,
 149       sequence_error=None)
 150   with tf.Graph().as_default():
 151     model = InitNetwork(eval_data, model_str, 'eval', reader=reader)
 152     sw = tf.summary.FileWriter(eval_dir)
 153 
 154     while True:
 155       sess = tf.Session('')
 156       if graph_def_file is not None:
 157         # Write the eval version of the graph to a file for freezing.
 158         if not tf.gfile.Exists(graph_def_file):
 159           with tf.gfile.FastGFile(graph_def_file, 'w') as f:
 160             f.write(
 161                 sess.graph.as_graph_def(add_shapes=True).SerializeToString())
 162       ckpt = tf.train.get_checkpoint_state(train_dir)
 163       if ckpt and ckpt.model_checkpoint_path:
 164         step = model.Restore(ckpt.model_checkpoint_path, sess)
 165         if decode:
 166           rates = decode.SoftmaxEval(sess, model, num_steps)
 167           _AddRateToSummary('Label error rate', rates.label_error, step, sw)
 168           _AddRateToSummary('Word recall error rate', rates.word_recall_error,
 169                             step, sw)
 170           _AddRateToSummary('Word precision error rate',
 171                             rates.word_precision_error, step, sw)
 172           _AddRateToSummary('Sequence error rate', rates.sequence_error, step,
 173                             sw)
 174           sw.flush()
 175           print('Error rates=', rates)
 176         else:
 177           raise ValueError('Non-softmax decoder evaluation not implemented!')
 178       if eval_interval_secs:
 179         time.sleep(eval_interval_secs)
 180       else:
 181         break
 182   return rates

 185 def InitNetwork(input_pattern,
 186                 model_spec,
 187                 mode='eval',
 188                 initial_learning_rate=0.00005,
 189                 final_learning_rate=0.00005,
 190                 halflife=1600000,
 191                 optimizer_type='Adam',
 192                 num_preprocess_threads=1,
 193                 reader=None):
 194   """Constructs a python tensor flow model defined by model_spec.
 195 
 196   Args:
 197     input_pattern: File pattern of the data in tfrecords of Example.
 198     model_spec: Concatenation of input spec, model spec and output spec.
 199       See Build below for input/output spec. For model spec, see vgslspecs.py
 200     mode: One of 'train', 'eval'
 201     initial_learning_rate: Initial learning rate for the network.
 202     final_learning_rate: Final learning rate for the network.
 203     halflife: Number of steps over which to halve the difference between
 204               initial and final learning rate for the network.
 205     optimizer_type: One of 'GradientDescent', 'AdaGrad', 'Momentum', 'Adam'.
 206     num_preprocess_threads: Number of threads to use for image processing.
 207     reader: Function that returns an actual reader to read Examples from input
 208       files. If None, uses tf.TFRecordReader().
 209     Eval tasks need only specify input_pattern and model_spec.
 210 
 211   Returns:
 212     A VGSLImageModel class.
 213 
 214   Raises:
 215     ValueError: if the model spec syntax is incorrect.
 216   """
 217   model = VGSLImageModel(mode, model_spec, initial_learning_rate,
 218                          final_learning_rate, halflife)
 219   left_bracket = model_spec.find('[')
 220   right_bracket = model_spec.rfind(']')
 221   if left_bracket < 0 or right_bracket < 0:
 222     raise ValueError('Failed to find [] in model spec! ', model_spec)
 223   input_spec = model_spec[:left_bracket]
 224   layer_spec = model_spec[left_bracket:right_bracket + 1]
 225   output_spec = model_spec[right_bracket + 1:]
 226   model.Build(input_pattern, input_spec, layer_spec, output_spec,
 227               optimizer_type, num_preprocess_threads, reader)
 228   return model

 231 class VGSLImageModel(object):
 232   """Class that builds a tensor flow model for training or evaluation.
 233   """
 234 
 235   def __init__(self, mode, model_spec, initial_learning_rate,
 236                final_learning_rate, halflife):
 237     """Constructs a VGSLImageModel.
 238 
 239     Args:
 240       mode:        One of "train", "eval"
 241       model_spec:  Full model specification string, for reference only.
 242       initial_learning_rate: Initial learning rate for the network.
 243       final_learning_rate: Final learning rate for the network.
 244       halflife: Number of steps over which to halve the difference between
 245                 initial and final learning rate for the network.
 246     """
 247     # The string that was used to build this model.
 248     self.model_spec = model_spec
 249     # The layers between input and output.
 250     self.layers = None
 251     # The train/eval mode.
 252     self.mode = mode
 253     # The initial learning rate.
 254     self.initial_learning_rate = initial_learning_rate
 255     self.final_learning_rate = final_learning_rate
 256     self.decay_steps = halflife / DECAY_STEPS_FACTOR
 257     self.decay_rate = DECAY_RATE
 258     # Tensor for the labels.
 259     self.labels = None
 260     self.sparse_labels = None
 261     # Debug data containing the truth text.
 262     self.truths = None
 263     # Tensor for loss
 264     self.loss = None
 265     # Train operation
 266     self.train_op = None
 267     # Tensor for the global step counter
 268     self.global_step = None
 269     # Tensor for the output predictions (usually softmax)
 270     self.output = None
 271     # True if we are using CTC training mode.
 272     self.using_ctc = False
 273     # Saver object to load or restore the variables.
 274     self.saver = None
 275 
 276   def Build(self, input_pattern, input_spec, model_spec, output_spec,
 277             optimizer_type, num_preprocess_threads, reader):
 278     """Builds the model from the separate input/layers/output spec strings.
 279 
 280     Args:
 281       input_pattern: File pattern of the data in tfrecords of TF Example format.
 282       input_spec: Specification of the input layer:
 283         batchsize,height,width,depth (4 comma-separated integers)
 284           Training will run with batches of batchsize images, but runtime can
 285           use any batch size.
 286           height and/or width can be 0 or -1, indicating variable size,
 287           otherwise all images must be the given size.
 288           depth must be 1 or 3 to indicate greyscale or color.
 289           NOTE 1-d image input, treating the y image dimension as depth, can
 290           be achieved using S1(1x0)1,3 as the first op in the model_spec, but
 291           the y-size of the input must then be fixed.
 292       model_spec: Model definition. See vgslspecs.py
 293       output_spec: Output layer definition:
 294         O(2|1|0)(l|s|c)n output layer with n classes.
 295           2 (heatmap) Output is a 2-d vector map of the input (possibly at
 296             different scale).
 297           1 (sequence) Output is a 1-d sequence of vector values.
 298           0 (value) Output is a 0-d single vector value.
 299           l uses a logistic non-linearity on the output, allowing multiple
 300             hot elements in any output vector value.
 301           s uses a softmax non-linearity, with one-hot output in each value.
 302           c uses a softmax with CTC. Can only be used with s (sequence).
 303           NOTE Only O1s and O1c are currently supported.
 304       optimizer_type: One of 'GradientDescent', 'AdaGrad', 'Momentum', 'Adam'.
 305       num_preprocess_threads: Number of threads to use for image processing.
 306       reader: Function that returns an actual reader to read Examples from input
 307         files. If None, uses tf.TFRecordReader().
 308     """
 309     self.global_step = tf.Variable(0, name='global_step', trainable=False)
 310     shape = _ParseInputSpec(input_spec)
 311     out_dims, out_func, num_classes = _ParseOutputSpec(output_spec)
 312     self.using_ctc = out_func == 'c'
 313     images, heights, widths, labels, sparse, _ = vgsl_input.ImageInput(
 314         input_pattern, num_preprocess_threads, shape, self.using_ctc, reader)
 315     self.labels = labels
 316     self.sparse_labels = sparse
 317     self.layers = vgslspecs.VGSLSpecs(widths, heights, self.mode == 'train')
 318     last_layer = self.layers.Build(images, model_spec)
 319     self._AddOutputs(last_layer, out_dims, out_func, num_classes)
 320     if self.mode == 'train':
 321       self._AddOptimizer(optimizer_type)
 322 
 323     # For saving the model across training and evaluation
 324     self.saver = tf.train.Saver()
 325 
 326   def TrainAStep(self, sess):
 327     """Runs a training step in the session.
 328 
 329     Args:
 330       sess: Session in which to train the model.
 331     Returns:
 332       loss, global_step.
 333     """
 334     _, loss, step = sess.run([self.train_op, self.loss, self.global_step])
 335     return loss, step
 336 
 337   def Restore(self, checkpoint_path, sess):
 338     """Restores the model from the given checkpoint path into the session.
 339 
 340     Args:
 341       checkpoint_path: File pathname of the checkpoint.
 342       sess:            Session in which to restore the model.
 343     Returns:
 344       global_step of the model.
 345     """
 346     self.saver.restore(sess, checkpoint_path)
 347     return tf.train.global_step(sess, self.global_step)

 349   def RunAStep(self, sess):
 350     """Runs a step for eval in the session.
 351 
 352     Args:
 353       sess:            Session in which to run the model.
 354     Returns:
 355       output tensor result, labels tensor result.
 356     """
 357     return sess.run([self.output, self.labels])

 359   def _AddOutputs(self, prev_layer, out_dims, out_func, num_classes):
 360     """Adds the output layer and loss function.
 361 
 362     Args:
 363       prev_layer:  Output of last layer of main network.
 364       out_dims:    Number of output dimensions, 0, 1 or 2.
 365       out_func:    Output non-linearity. 's' or 'c'=softmax, 'l'=logistic.
 366       num_classes: Number of outputs/size of last output dimension.
 367     """
 368     height_in = shapes.tensor_dim(prev_layer, dim=1)
 369     logits, outputs = self._AddOutputLayer(prev_layer, out_dims, out_func,
 370                                            num_classes)
 371     if self.mode == 'train':
 372       # Setup loss for training.
 373       self.loss = self._AddLossFunction(logits, height_in, out_dims, out_func)
 374       tf.summary.scalar('loss', self.loss)
 375     elif out_dims == 0:
 376       # Be sure the labels match the output, even in eval mode.
 377       self.labels = tf.slice(self.labels, [0, 0], [-1, 1])
 378       self.labels = tf.reshape(self.labels, [-1])
 379 
 380     logging.info('Final output=%s', outputs)
 381     logging.info('Labels tensor=%s', self.labels)
 382     self.output = outputs
 383 
 384   def _AddOutputLayer(self, prev_layer, out_dims, out_func, num_classes):
 385     """Add the fully-connected logits and SoftMax/Logistic output Layer.
 386 
 387     Args:
 388       prev_layer:  Output of last layer of main network.
 389       out_dims:    Number of output dimensions, 0, 1 or 2.
 390       out_func:    Output non-linearity. 's' or 'c'=softmax, 'l'=logistic.
 391       num_classes: Number of outputs/size of last output dimension.
 392 
 393     Returns:
 394       logits:  Pre-softmax/logistic fully-connected output shaped to out_dims.
 395       outputs: Post-softmax/logistic shaped to out_dims.
 396 
 397     Raises:
 398       ValueError: if syntax is incorrect.
 399     """
 400     # Reduce dimensionality appropriate to the output dimensions.
 401     batch_in = shapes.tensor_dim(prev_layer, dim=0)
 402     height_in = shapes.tensor_dim(prev_layer, dim=1)
 403     width_in = shapes.tensor_dim(prev_layer, dim=2)
 404     depth_in = shapes.tensor_dim(prev_layer, dim=3)
 405     if out_dims:
 406       # Combine any remaining height and width with batch and unpack after.
 407       shaped = tf.reshape(prev_layer, [-1, depth_in])
 408     else:
 409       # Everything except batch goes to depth, and therefore has to be known.
 410       shaped = tf.reshape(prev_layer, [-1, height_in * width_in * depth_in])
 411     logits = slim.fully_connected(shaped, num_classes, activation_fn=None)
 412     if out_func == 'l':
 413       raise ValueError('Logistic not yet supported!')
 414     else:
 415       output = tf.nn.softmax(logits)
 416     # Reshape to the dessired output.
 417     if out_dims == 2:
 418       output_shape = [batch_in, height_in, width_in, num_classes]
 419     elif out_dims == 1:
 420       output_shape = [batch_in, height_in * width_in, num_classes]
 421     else:
 422       output_shape = [batch_in, num_classes]
 423     output = tf.reshape(output, output_shape, name='Output')
 424     logits = tf.reshape(logits, output_shape)
 425     return logits, output
 426 
 427   def _AddLossFunction(self, logits, height_in, out_dims, out_func):
 428     """Add the appropriate loss function.
 429 
 430     Args:
 431       logits:  Pre-softmax/logistic fully-connected output shaped to out_dims.
 432       height_in:  Height of logits before going into the softmax layer.
 433       out_dims:   Number of output dimensions, 0, 1 or 2.
 434       out_func:   Output non-linearity. 's' or 'c'=softmax, 'l'=logistic.
 435 
 436     Returns:
 437       loss: That which is to be minimized.
 438 
 439     Raises:
 440       ValueError: if logistic is used.
 441     """
 442     if out_func == 'c':
 443       # Transpose batch to the middle.
 444       ctc_input = tf.transpose(logits, [1, 0, 2])
 445       # Compute the widths of each batch element from the input widths.
 446       widths = self.layers.GetLengths(dim=2, factor=height_in)
 447       cross_entropy = tf.nn.ctc_loss(ctc_input, self.sparse_labels, widths)
 448     elif out_func == 's':
 449       if out_dims == 2:
 450         self.labels = _PadLabels3d(logits, self.labels)
 451       elif out_dims == 1:
 452         self.labels = _PadLabels2d(
 453             shapes.tensor_dim(
 454                 logits, dim=1), self.labels)
 455       else:
 456         self.labels = tf.slice(self.labels, [0, 0], [-1, 1])
 457         self.labels = tf.reshape(self.labels, [-1])
 458       cross_entropy = tf.nn.sparse_softmax_cross_entropy_with_logits(
 459           logits=logits, labels=self.labels, name='xent')
 460     else:
 461       # TODO(rays) Labels need an extra dimension for logistic, so different
 462       # padding functions are needed, as well as a different loss function.
 463       raise ValueError('Logistic not yet supported!')
 464     return tf.reduce_sum(cross_entropy)
 465 
 466   def _AddOptimizer(self, optimizer_type):
 467     """Adds an optimizer with learning rate decay to minimize self.loss.
 468 
 469     Args:
 470       optimizer_type: One of 'GradientDescent', 'AdaGrad', 'Momentum', 'Adam'.
 471     Raises:
 472       ValueError: if the optimizer type is unrecognized.
 473     """
 474     learn_rate_delta = self.initial_learning_rate - self.final_learning_rate
 475     learn_rate_dec = tf.add(
 476         tf.train.exponential_decay(learn_rate_delta, self.global_step,
 477                                    self.decay_steps, self.decay_rate),
 478         self.final_learning_rate)
 479     if optimizer_type == 'GradientDescent':
 480       opt = tf.train.GradientDescentOptimizer(learn_rate_dec)
 481     elif optimizer_type == 'AdaGrad':
 482       opt = tf.train.AdagradOptimizer(learn_rate_dec)
 483     elif optimizer_type == 'Momentum':
 484       opt = tf.train.MomentumOptimizer(learn_rate_dec, momentum=0.9)
 485     elif optimizer_type == 'Adam':
 486       opt = tf.train.AdamOptimizer(learning_rate=learn_rate_dec)
 487     else:
 488       raise ValueError('Invalid optimizer type: ' + optimizer_type)
 489     tf.summary.scalar('learn_rate', learn_rate_dec)
 490 
 491     self.train_op = opt.minimize(
 492         self.loss, global_step=self.global_step, name='train')

 495 def _PadLabels3d(logits, labels):
 496   """Pads or slices 3-d labels to match logits.
 497 
 498   Covers the case of 2-d softmax output, when labels is [batch, height, width]
 499   and logits is [batch, height, width, onehot]
 500   Args:
 501     logits: 4-d Pre-softmax fully-connected output.
 502     labels: 3-d, but not necessarily matching in size.
 503 
 504   Returns:
 505     labels: Resized by padding or clipping to match logits.
 506   """
 507   logits_shape = shapes.tensor_shape(logits)
 508   labels_shape = shapes.tensor_shape(labels)
 509   labels = tf.reshape(labels, [-1, labels_shape[2]])
 510   labels = _PadLabels2d(logits_shape[2], labels)
 511   labels = tf.reshape(labels, [labels_shape[0], -1])
 512   labels = _PadLabels2d(logits_shape[1] * logits_shape[2], labels)
 513   return tf.reshape(labels, [labels_shape[0], logits_shape[1], logits_shape[2]])
 514 
 515 
 516 def _PadLabels2d(logits_size, labels):
 517   """Pads or slices the 2nd dimension of 2-d labels to match logits_size.
 518 
 519   Covers the case of 1-d softmax output, when labels is [batch, seq] and
 520   logits is [batch, seq, onehot]
 521   Args:
 522     logits_size: Tensor returned from tf.shape giving the target size.
 523     labels:      2-d, but not necessarily matching in size.
 524 
 525   Returns:
 526     labels: Resized by padding or clipping the last dimension to logits_size.
 527   """
 528   pad = logits_size - tf.shape(labels)[1]
 529 
 530   def _PadFn():
 531     return tf.pad(labels, [[0, 0], [0, pad]])
 532 
 533   def _SliceFn():
 534     return tf.slice(labels, [0, 0], [-1, logits_size])
 535 
 536   return tf.cond(tf.greater(pad, 0), _PadFn, _SliceFn)

 539 def _ParseInputSpec(input_spec):
 540   """Parses input_spec and returns the numbers obtained therefrom.
 541 
 542   Args:
 543     input_spec:  Specification of the input layer. See Build.
 544 
 545   Returns:
 546     shape:      ImageShape with the desired shape of the input.
 547 
 548   Raises:
 549     ValueError: if syntax is incorrect.
 550   """
 551   pattern = re.compile(R'(\d+),(\d+),(\d+),(\d+)')
 552   m = pattern.match(input_spec)
 553   if m is None:
 554     raise ValueError('Failed to parse input spec:' + input_spec)
 555   batch_size = int(m.group(1))
 556   y_size = int(m.group(2)) if int(m.group(2)) > 0 else None
 557   x_size = int(m.group(3)) if int(m.group(3)) > 0 else None
 558   depth = int(m.group(4))
 559   if depth not in [1, 3]:
 560     raise ValueError('Depth must be 1 or 3, had:', depth)
 561   return vgsl_input.ImageShape(batch_size, y_size, x_size, depth)
 562

 564 def _ParseOutputSpec(output_spec):
 565   """Parses the output spec.
 566 
 567   Args:
 568     output_spec: Output layer definition. See Build.
 569 
 570   Returns:
 571     out_dims:     2|1|0 for 2-d, 1-d, 0-d.
 572     out_func:     l|s|c for logistic, softmax, softmax+CTC
 573     num_classes:  Number of classes in output.
 574 
 575   Raises:
 576     ValueError: if syntax is incorrect.
 577   """
 578   pattern = re.compile(R'(O)(0|1|2)(l|s|c)(\d+)')
 579   m = pattern.match(output_spec)
 580   if m is None:
 581     raise ValueError('Failed to parse output spec:' + output_spec)
 582   out_dims = int(m.group(2))
 583   out_func = m.group(3)
 584   if out_func == 'c' and out_dims != 1:
 585     raise ValueError('CTC can only be used with a 1-D sequence!')
 586   num_classes = int(m.group(4))
 587   return out_dims, out_func, num_classes
