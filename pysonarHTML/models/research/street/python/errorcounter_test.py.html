  15 """Tests for errorcounter."""
  16 import tensorflow as tf
  17 import errorcounter as ec

  20 class ErrorcounterTest(tf.test.TestCase):

  22   def testComputeErrorRate(self):
  23     """Tests that the percent calculation works as expected.
  24     """
  25     rate = ec.ComputeErrorRate(error_count=0, truth_count=0)
  26     self.assertEqual(rate, 100.0)
  27     rate = ec.ComputeErrorRate(error_count=1, truth_count=0)
  28     self.assertEqual(rate, 100.0)
  29     rate = ec.ComputeErrorRate(error_count=10, truth_count=1)
  30     self.assertEqual(rate, 100.0)
  31     rate = ec.ComputeErrorRate(error_count=0, truth_count=1)
  32     self.assertEqual(rate, 0.0)
  33     rate = ec.ComputeErrorRate(error_count=3, truth_count=12)
  34     self.assertEqual(rate, 25.0)

  36   def testCountErrors(self):
  37     """Tests that the error counter works as expected.
  38     """
  39     truth_str = 'farm barn'
  40     counts = ec.CountErrors(ocr_text=truth_str, truth_text=truth_str)
  41     self.assertEqual(
  42         counts, ec.ErrorCounts(
  43             fn=0, fp=0, truth_count=9, test_count=9))
  44     # With a period on the end, we get a char error.
  45     dot_str = 'farm barn.'
  46     counts = ec.CountErrors(ocr_text=dot_str, truth_text=truth_str)
  47     self.assertEqual(
  48         counts, ec.ErrorCounts(
  49             fn=0, fp=1, truth_count=9, test_count=10))
  50     counts = ec.CountErrors(ocr_text=truth_str, truth_text=dot_str)
  51     self.assertEqual(
  52         counts, ec.ErrorCounts(
  53             fn=1, fp=0, truth_count=10, test_count=9))
  54     # Space is just another char.
  55     no_space = 'farmbarn'
  56     counts = ec.CountErrors(ocr_text=no_space, truth_text=truth_str)
  57     self.assertEqual(
  58         counts, ec.ErrorCounts(
  59             fn=1, fp=0, truth_count=9, test_count=8))
  60     counts = ec.CountErrors(ocr_text=truth_str, truth_text=no_space)
  61     self.assertEqual(
  62         counts, ec.ErrorCounts(
  63             fn=0, fp=1, truth_count=8, test_count=9))
  64     # Lose them all.
  65     counts = ec.CountErrors(ocr_text='', truth_text=truth_str)
  66     self.assertEqual(
  67         counts, ec.ErrorCounts(
  68             fn=9, fp=0, truth_count=9, test_count=0))
  69     counts = ec.CountErrors(ocr_text=truth_str, truth_text='')
  70     self.assertEqual(
  71         counts, ec.ErrorCounts(
  72             fn=0, fp=9, truth_count=0, test_count=9))
  73 
  74   def testCountWordErrors(self):
  75     """Tests that the error counter works as expected.
  76     """
  77     truth_str = 'farm barn'
  78     counts = ec.CountWordErrors(ocr_text=truth_str, truth_text=truth_str)
  79     self.assertEqual(
  80         counts, ec.ErrorCounts(
  81             fn=0, fp=0, truth_count=2, test_count=2))
  82     # With a period on the end, we get a word error.
  83     dot_str = 'farm barn.'
  84     counts = ec.CountWordErrors(ocr_text=dot_str, truth_text=truth_str)
  85     self.assertEqual(
  86         counts, ec.ErrorCounts(
  87             fn=1, fp=1, truth_count=2, test_count=2))
  88     counts = ec.CountWordErrors(ocr_text=truth_str, truth_text=dot_str)
  89     self.assertEqual(
  90         counts, ec.ErrorCounts(
  91             fn=1, fp=1, truth_count=2, test_count=2))
  92     # Space is special.
  93     no_space = 'farmbarn'
  94     counts = ec.CountWordErrors(ocr_text=no_space, truth_text=truth_str)
  95     self.assertEqual(
  96         counts, ec.ErrorCounts(
  97             fn=2, fp=1, truth_count=2, test_count=1))
  98     counts = ec.CountWordErrors(ocr_text=truth_str, truth_text=no_space)
  99     self.assertEqual(
 100         counts, ec.ErrorCounts(
 101             fn=1, fp=2, truth_count=1, test_count=2))
 102     # Lose them all.
 103     counts = ec.CountWordErrors(ocr_text='', truth_text=truth_str)
 104     self.assertEqual(
 105         counts, ec.ErrorCounts(
 106             fn=2, fp=0, truth_count=2, test_count=0))
 107     counts = ec.CountWordErrors(ocr_text=truth_str, truth_text='')
 108     self.assertEqual(
 109         counts, ec.ErrorCounts(
 110             fn=0, fp=2, truth_count=0, test_count=2))
 111     # With a space in ba rn, there is an extra add.
 112     sp_str = 'farm ba rn'
 113     counts = ec.CountWordErrors(ocr_text=sp_str, truth_text=truth_str)
 114     self.assertEqual(
 115         counts, ec.ErrorCounts(
 116             fn=1, fp=2, truth_count=2, test_count=3))
 117     counts = ec.CountWordErrors(ocr_text=truth_str, truth_text=sp_str)
 118     self.assertEqual(
 119         counts, ec.ErrorCounts(
 120             fn=2, fp=1, truth_count=3, test_count=2))
