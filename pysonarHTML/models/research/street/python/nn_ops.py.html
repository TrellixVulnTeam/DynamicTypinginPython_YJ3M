  25 def rnn_helper(inp,
  26                length,
  27                cell_type=None,
  28                direction="forward",
  29                name=None,
  30                *args,
  31                **kwargs):
  32   """Adds ops for a recurrent neural network layer.
  33 
  34   This function calls an actual implementation of a recurrent neural network
  35   based on `cell_type`.
  36 
  37   There are three modes depending on the value of `direction`:
  38 
  39     forward: Adds a forward RNN.
  40     backward: Adds a backward RNN.
  41     bidirectional: Adds both forward and backward RNNs and creates a
  42                    bidirectional RNN.
  43 
  44   Args:
  45     inp: A 3-D tensor of shape [`batch_size`, `max_length`, `feature_dim`].
  46     length: A 1-D tensor of shape [`batch_size`] and type int64. Each element
  47             represents the length of the corresponding sequence in `inp`.
  48     cell_type: Cell type of RNN. Currently can only be "lstm".
  49     direction: One of "forward", "backward", "bidirectional".
  50     name: Name of the op.
  51     *args: Other arguments to the layer.
  52     **kwargs: Keyword arugments to the layer.
  53 
  54   Returns:
  55     A 3-D tensor of shape [`batch_size`, `max_length`, `num_nodes`].
  56   """
  57 
  58   assert cell_type is not None
  59   rnn_func = None
  60   if cell_type == "lstm":
  61     rnn_func = lstm_layer
  62   assert rnn_func is not None
  63   assert direction in ["forward", "backward", "bidirectional"]
  64 
  65   with tf.variable_scope(name):
  66     if direction in ["forward", "bidirectional"]:
  67       forward = rnn_func(
  68           inp=inp,
  69           length=length,
  70           backward=False,
  71           name="forward",
  72           *args,
  73           **kwargs)
  74       if isinstance(forward, tuple):
  75         # lstm_layer returns a tuple (output, memory). We only need the first
  76         # element.
  77         forward = forward[0]
  78     if direction in ["backward", "bidirectional"]:
  79       backward = rnn_func(
  80           inp=inp,
  81           length=length,
  82           backward=True,
  83           name="backward",
  84           *args,
  85           **kwargs)
  86       if isinstance(backward, tuple):
  87         # lstm_layer returns a tuple (output, memory). We only need the first
  88         # element.
  89         backward = backward[0]
  90     if direction == "forward":
  91       out = forward
  92     elif direction == "backward":
  93       out = backward
  94     else:
  95       out = tf.concat(axis=2, values=[forward, backward])
  96   return out

  99 @tf.RegisterShape("VariableLSTM")
 100 def _variable_lstm_shape(op):
 101   """Shape function for the VariableLSTM op."""
 102   input_shape = op.inputs[0].get_shape().with_rank(4)
 103   state_shape = op.inputs[1].get_shape().with_rank(2)
 104   memory_shape = op.inputs[2].get_shape().with_rank(2)
 105   w_m_m_shape = op.inputs[3].get_shape().with_rank(3)
 106   batch_size = input_shape[0].merge_with(state_shape[0])
 107   batch_size = input_shape[0].merge_with(memory_shape[0])
 108   seq_len = input_shape[1]
 109   gate_num = input_shape[2].merge_with(w_m_m_shape[1])
 110   output_dim = input_shape[3].merge_with(state_shape[1])
 111   output_dim = output_dim.merge_with(memory_shape[1])
 112   output_dim = output_dim.merge_with(w_m_m_shape[0])
 113   output_dim = output_dim.merge_with(w_m_m_shape[2])
 114   return [[batch_size, seq_len, output_dim],
 115           [batch_size, seq_len, gate_num, output_dim],
 116           [batch_size, seq_len, output_dim]]

 119 @tf.RegisterGradient("VariableLSTM")
 120 def _variable_lstm_grad(op, act_grad, gate_grad, mem_grad):
 121   """Gradient function for the VariableLSTM op."""
 122   initial_state = op.inputs[1]
 123   initial_memory = op.inputs[2]
 124   w_m_m = op.inputs[3]
 125   act = op.outputs[0]
 126   gate_raw_act = op.outputs[1]
 127   memory = op.outputs[2]
 128   return rnn.variable_lstm_grad(initial_state, initial_memory, w_m_m, act,
 129                                 gate_raw_act, memory, act_grad, gate_grad,
 130                                 mem_grad)

 133 def lstm_layer(inp,
 134                length=None,
 135                state=None,
 136                memory=None,
 137                num_nodes=None,
 138                backward=False,
 139                clip=50.0,
 140                reg_func=tf.nn.l2_loss,
 141                weight_reg=False,
 142                weight_collection="LSTMWeights",
 143                bias_reg=False,
 144                stddev=None,
 145                seed=None,
 146                decode=False,
 147                use_native_weights=False,
 148                name=None):
 149   """Adds ops for an LSTM layer.
 150 
 151   This adds ops for the following operations:
 152 
 153     input => (forward-LSTM|backward-LSTM) => output
 154 
 155   The direction of the LSTM is determined by `backward`. If it is false, the
 156   forward LSTM is used, the backward one otherwise.
 157 
 158   Args:
 159     inp: A 3-D tensor of shape [`batch_size`, `max_length`, `feature_dim`].
 160     length: A 1-D tensor of shape [`batch_size`] and type int64. Each element
 161             represents the length of the corresponding sequence in `inp`.
 162     state: If specified, uses it as the initial state.
 163     memory: If specified, uses it as the initial memory.
 164     num_nodes: The number of LSTM cells.
 165     backward: If true, reverses the `inp` before adding the ops. The output is
 166               also reversed so that the direction is the same as `inp`.
 167     clip: Value used to clip the cell values.
 168     reg_func: Function used for the weight regularization such as
 169               `tf.nn.l2_loss`.
 170     weight_reg: If true, regularize the filter weights with `reg_func`.
 171     weight_collection: Collection to add the weights to for regularization.
 172     bias_reg: If true, regularize the bias vector with `reg_func`.
 173     stddev: Standard deviation used to initialize the variables.
 174     seed: Seed used to initialize the variables.
 175     decode: If true, does not add ops which are not used for inference.
 176     use_native_weights: If true, uses weights in the same format as the native
 177                         implementations.
 178     name: Name of the op.
 179 
 180   Returns:
 181     A 3-D tensor of shape [`batch_size`, `max_length`, `num_nodes`].
 182   """
 183   with tf.variable_scope(name):
 184     if backward:
 185       if length is None:
 186         inp = tf.reverse(inp, [1])
 187       else:
 188         inp = tf.reverse_sequence(inp, length, 1, 0)
 189 
 190     num_prev = inp.get_shape()[2]
 191     if stddev:
 192       initializer = tf.truncated_normal_initializer(stddev=stddev, seed=seed)
 193     else:
 194       initializer = tf.uniform_unit_scaling_initializer(seed=seed)
 195 
 196     if use_native_weights:
 197       with tf.variable_scope("LSTMCell"):
 198         w = tf.get_variable(
 199             "W_0",
 200             shape=[num_prev + num_nodes, 4 * num_nodes],
 201             initializer=initializer,
 202             dtype=tf.float32)
 203         w_i_m = tf.slice(w, [0, 0], [num_prev, 4 * num_nodes], name="w_i_m")
 204         w_m_m = tf.reshape(
 205             tf.slice(w, [num_prev, 0], [num_nodes, 4 * num_nodes]),
 206             [num_nodes, 4, num_nodes],
 207             name="w_m_m")
 208     else:
 209       w_i_m = tf.get_variable("w_i_m", [num_prev, 4 * num_nodes],
 210                               initializer=initializer)
 211       w_m_m = tf.get_variable("w_m_m", [num_nodes, 4, num_nodes],
 212                               initializer=initializer)
 213 
 214     if not decode and weight_reg:
 215       tf.add_to_collection(weight_collection, reg_func(w_i_m, name="w_i_m_reg"))
 216       tf.add_to_collection(weight_collection, reg_func(w_m_m, name="w_m_m_reg"))
 217 
 218     batch_size = shapes.tensor_dim(inp, dim=0)
 219     num_frames = shapes.tensor_dim(inp, dim=1)
 220     prev = tf.reshape(inp, tf.stack([batch_size * num_frames, num_prev]))
 221 
 222     if use_native_weights:
 223       with tf.variable_scope("LSTMCell"):
 224         b = tf.get_variable(
 225             "B",
 226             shape=[4 * num_nodes],
 227             initializer=tf.zeros_initializer(),
 228             dtype=tf.float32)
 229       biases = tf.identity(b, name="biases")
 230     else:
 231       biases = tf.get_variable(
 232           "biases", [4 * num_nodes], initializer=tf.constant_initializer(0.0))
 233     if not decode and bias_reg:
 234       tf.add_to_collection(
 235           weight_collection, reg_func(
 236               biases, name="biases_reg"))
 237     prev = tf.nn.xw_plus_b(prev, w_i_m, biases)
 238 
 239     prev = tf.reshape(prev, tf.stack([batch_size, num_frames, 4, num_nodes]))
 240     if state is None:
 241       state = tf.fill(tf.stack([batch_size, num_nodes]), 0.0)
 242     if memory is None:
 243       memory = tf.fill(tf.stack([batch_size, num_nodes]), 0.0)
 244 
 245     out, _, mem = rnn.variable_lstm(prev, state, memory, w_m_m, clip=clip)
 246 
 247     if backward:
 248       if length is None:
 249         out = tf.reverse(out, [1])
 250       else:
 251         out = tf.reverse_sequence(out, length, 1, 0)
 252 
 253   return out, mem
