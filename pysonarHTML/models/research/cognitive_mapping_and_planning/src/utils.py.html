  24 class Timer():
  25   def __init__(self):
  26     self.calls = 0.
  27     self.start_time = 0.
  28     self.time_per_call = 0.
  29     self.total_time = 0.
  30     self.last_log_time = 0.
  31 
  32   def tic(self):
  33     self.start_time = time.time()
  34 
  35   def toc(self, average=True, log_at=-1, log_str='', type='calls'):
  36     if self.start_time == 0:
  37       logging.error('Timer not started by calling tic().')
  38     t = time.time()
  39     diff = time.time() - self.start_time
  40     self.total_time += diff
  41     self.calls += 1.
  42     self.time_per_call = self.total_time/self.calls
  43 
  44     if type == 'calls' and log_at > 0 and np.mod(self.calls, log_at) == 0:
  45       _ = []
  46       logging.info('%s: %f seconds.', log_str, self.time_per_call)
  47     elif type == 'time' and log_at > 0 and t - self.last_log_time >= log_at:
  48       _ = []
  49       logging.info('%s: %f seconds.', log_str, self.time_per_call)
  50       self.last_log_time = t
  51 
  52     if average:
  53       return self.time_per_call
  54     else:
  55       return diff

  66         str__ = str__.replace('\n', '\n  ')

  74 def dict_equal(dict1, dict2):
  75   assert(set(dict1.keys()) == set(dict2.keys())), "Sets of keys between 2 dictionaries are different."
  76   for k in dict1.keys():
  77     assert(type(dict1[k]) == type(dict2[k])), "Type of key '{:s}' if different.".format(k)
  78     if type(dict1[k]) == np.ndarray:
  79       assert(dict1[k].dtype == dict2[k].dtype), "Numpy Type of key '{:s}' if different.".format(k)
  80       assert(np.allclose(dict1[k], dict2[k])), "Value for key '{:s}' do not match.".format(k)
  81     else:
  82       assert(dict1[k] == dict2[k]), "Value for key '{:s}' do not match.".format(k)
  83   return True

 108 def save_variables(pickle_file_name, var, info, overwrite = False):
 109   if fu.exists(pickle_file_name) and overwrite == False:
 110     raise Exception('{:s} exists and over write is false.'.format(pickle_file_name))
 111   # Construct the dictionary
 112   assert(type(var) == list); assert(type(info) == list);
 113   d = {}
 114   for i in xrange(len(var)):
 115     d[info[i]] = var[i]
 116   with fu.fopen(pickle_file_name, 'w') as f:
 117     cPickle.dump(d, f, cPickle.HIGHEST_PROTOCOL)

 127 def voc_ap(rec, prec):
 128   rec = rec.reshape((-1,1))
 129   prec = prec.reshape((-1,1))
 130   z = np.zeros((1,1))
 131   o = np.ones((1,1))
 132   mrec = np.vstack((z, rec, o))
 133   mpre = np.vstack((z, prec, z))
 134   for i in range(len(mpre)-2, -1, -1):
 135     mpre[i] = max(mpre[i], mpre[i+1])
 136 
 137   I = np.where(mrec[1:] != mrec[0:-1])[0]+1;
 138   ap = 0;
 139   for i in I:
 140     ap = ap + (mrec[i] - mrec[i-1])*mpre[i];
 141   return ap

 150 def calc_pr(gt, out, wt=None):
 151   if wt is None:
 152     wt = np.ones((gt.size,1))
 153 
 154   gt = gt.astype(np.float64).reshape((-1,1))
 155   wt = wt.astype(np.float64).reshape((-1,1))
 156   out = out.astype(np.float64).reshape((-1,1))
 157 
 158   gt = gt*wt
 159   tog = np.concatenate([gt, wt, out], axis=1)*1.
 160   ind = np.argsort(tog[:,2], axis=0)[::-1]
 161   tog = tog[ind,:]
 162   cumsumsortgt = np.cumsum(tog[:,0])
 163   cumsumsortwt = np.cumsum(tog[:,1])
 164   prec = cumsumsortgt / cumsumsortwt
 165   rec = cumsumsortgt / np.sum(tog[:,0])
 166 
 167   ap = voc_ap(rec, prec)
 168   return ap, rec, prec
