  29 # Compute shortest path from all nodes to or from all source nodes
  30 def get_distance_node_list(gtG, source_nodes, direction, weights=None):
  31   gtG_ = gt.Graph(gtG)
  32   v = gtG_.add_vertex()
  33 
  34   if weights is not None:
  35     weights = gtG_.edge_properties[weights]
  36 
  37   for s in source_nodes:
  38     e = gtG_.add_edge(s, int(v))
  39     if weights is not None:
  40       weights[e] = 0.
  41 
  42   if direction == 'to':
  43     dist = gt.topology.shortest_distance(
  44         gt.GraphView(gtG_, reversed=True), source=gtG_.vertex(int(v)),
  45         target=None, weights=weights)
  46   elif direction == 'from':
  47     dist = gt.topology.shortest_distance(
  48         gt.GraphView(gtG_, reversed=False), source=gtG_.vertex(int(v)),
  49         target=None, weights=weights)
  50   dist = np.array(dist.get_array())
  51   dist = dist[:-1]
  52   if weights is None:
  53     dist = dist-1
  54   return dist

  56 # Functions for semantically labelling nodes in the traversal graph.
  57 def generate_lattice(sz_x, sz_y):
  58   """Generates a lattice with sz_x vertices along x and sz_y vertices along y
  59   direction Each of these vertices is step_size distance apart. Origin is at
  60   (0,0).  """
  61   g = gt.generation.lattice([sz_x, sz_y])
  62   x, y = np.meshgrid(np.arange(sz_x), np.arange(sz_y))
  63   x = np.reshape(x, [-1,1]); y = np.reshape(y, [-1,1]);
  64   nodes = np.concatenate((x,y), axis=1)
  65   return g, nodes

  67 def add_diagonal_edges(g, nodes, sz_x, sz_y, edge_len):
  68   offset = [sz_x+1, sz_x-1]
  69   for o in offset:
  70     s = np.arange(nodes.shape[0]-o-1)
  71     t = s + o
  72     ind = np.all(np.abs(nodes[s,:] - nodes[t,:]) == np.array([[1,1]]), axis=1)
  73     s = s[ind][:,np.newaxis]
  74     t = t[ind][:,np.newaxis]
  75     st = np.concatenate((s,t), axis=1)
  76     for i in range(st.shape[0]):
  77       e = g.add_edge(st[i,0], st[i,1], add_missing=False)
  78       g.ep['wts'][e] = edge_len

  80 def convert_traversible_to_graph(traversible, ff_cost=1., fo_cost=1.,
  81                                  oo_cost=1., connectivity=4):
  82   assert(connectivity == 4 or connectivity == 8)
  83 
  84   sz_x = traversible.shape[1]
  85   sz_y = traversible.shape[0]
  86   g, nodes = generate_lattice(sz_x, sz_y)
  87 
  88   # Assign costs.
  89   edge_wts = g.new_edge_property('float')
  90   g.edge_properties['wts'] = edge_wts
  91   wts = np.ones(g.num_edges(), dtype=np.float32)
  92   edge_wts.get_array()[:] = wts
  93 
  94   if connectivity == 8:
  95     add_diagonal_edges(g, nodes, sz_x, sz_y, np.sqrt(2.))
  96 
  97   se = np.array([[int(e.source()), int(e.target())] for e in g.edges()])
  98   s_xy = nodes[se[:,0]]
  99   t_xy = nodes[se[:,1]]
 100   s_t = np.ravel_multi_index((s_xy[:,1], s_xy[:,0]), traversible.shape)
 101   t_t = np.ravel_multi_index((t_xy[:,1], t_xy[:,0]), traversible.shape)
 102   s_t = traversible.ravel()[s_t]
 103   t_t = traversible.ravel()[t_t]
 104 
 105   wts = np.zeros(g.num_edges(), dtype=np.float32)
 106   wts[np.logical_and(s_t == True, t_t == True)] = ff_cost
 107   wts[np.logical_and(s_t == False, t_t == False)] = oo_cost
 108   wts[np.logical_xor(s_t, t_t)] = fo_cost
 109 
 110   edge_wts = g.edge_properties['wts']
 111   for i, e in enumerate(g.edges()):
 112     edge_wts[e] = edge_wts[e] * wts[i]
 113   # d = edge_wts.get_array()*1.
 114   # edge_wts.get_array()[:] = d*wts
 115   return g, nodes

 117 def label_nodes_with_class(nodes_xyt, class_maps, pix):
 118   """
 119   Returns:
 120     class_maps__: one-hot class_map for each class.
 121     node_class_label: one-hot class_map for each class, nodes_xyt.shape[0] x n_classes
 122   """
 123   # Assign each pixel to a node.
 124   selem = skimage.morphology.disk(pix)
 125   class_maps_ = class_maps*1.
 126   for i in range(class_maps.shape[2]):
 127     class_maps_[:,:,i] = skimage.morphology.dilation(class_maps[:,:,i]*1, selem)
 128   class_maps__ = np.argmax(class_maps_, axis=2)
 129   class_maps__[np.max(class_maps_, axis=2) == 0] = -1
 130 
 131   # For each node pick out the label from this class map.
 132   x = np.round(nodes_xyt[:,[0]]).astype(np.int32)
 133   y = np.round(nodes_xyt[:,[1]]).astype(np.int32)
 134   ind = np.ravel_multi_index((y,x), class_maps__.shape)
 135   node_class_label = class_maps__.ravel()[ind][:,0]
 136 
 137   # Convert to one hot versions.
 138   class_maps_one_hot = np.zeros(class_maps.shape, dtype=np.bool)
 139   node_class_label_one_hot = np.zeros((node_class_label.shape[0], class_maps.shape[2]), dtype=np.bool)
 140   for i in range(class_maps.shape[2]):
 141     class_maps_one_hot[:,:,i] = class_maps__ == i
 142     node_class_label_one_hot[:,i] = node_class_label == i
 143   return class_maps_one_hot, node_class_label_one_hot

 145 def label_nodes_with_class_geodesic(nodes_xyt, class_maps, pix, traversible,
 146                                     ff_cost=1., fo_cost=1., oo_cost=1.,
 147                                     connectivity=4):
 148   """Labels nodes in nodes_xyt with class labels using geodesic distance as
 149   defined by traversible from class_maps.
 150   Inputs:
 151     nodes_xyt
 152     class_maps: counts for each class.
 153     pix: distance threshold to consider close enough to target.
 154     traversible: binary map of whether traversible or not.
 155   Output:
 156     labels: For each node in nodes_xyt returns a label of the class or -1 is
 157     unlabelled.
 158   """
 159   g, nodes = convert_traversible_to_graph(traversible, ff_cost=ff_cost,
 160                                           fo_cost=fo_cost, oo_cost=oo_cost,
 161                                           connectivity=connectivity)
 162 
 163   class_dist = np.zeros_like(class_maps*1.)
 164   n_classes = class_maps.shape[2]
 165   if False:
 166     # Assign each pixel to a class based on number of points.
 167     selem = skimage.morphology.disk(pix)
 168     class_maps_ = class_maps*1.
 169     class_maps__ = np.argmax(class_maps_, axis=2)
 170     class_maps__[np.max(class_maps_, axis=2) == 0] = -1
 171 
 172   # Label nodes with classes.
 173   for i in range(n_classes):
 174     # class_node_ids = np.where(class_maps__.ravel() == i)[0]
 175     class_node_ids = np.where(class_maps[:,:,i].ravel() > 0)[0]
 176     dist_i = get_distance_node_list(g, class_node_ids, 'to', weights='wts')
 177     class_dist[:,:,i] = np.reshape(dist_i, class_dist[:,:,i].shape)
 178   class_map_geodesic = (class_dist <= pix)
 179   class_map_geodesic = np.reshape(class_map_geodesic, [-1, n_classes])
 180 
 181   # For each node pick out the label from this class map.
 182   x = np.round(nodes_xyt[:,[0]]).astype(np.int32)
 183   y = np.round(nodes_xyt[:,[1]]).astype(np.int32)
 184   ind = np.ravel_multi_index((y,x), class_dist[:,:,0].shape)
 185   node_class_label = class_map_geodesic[ind[:,0],:]
 186   class_map_geodesic = class_dist <= pix
 187   return class_map_geodesic, node_class_label

 189 def _get_next_nodes_undirected(n, sc, n_ori):
 190   nodes_to_add = []
 191   nodes_to_validate = []
 192   (p, q, r) = n
 193   nodes_to_add.append((n, (p, q, r), 0))
 194   if n_ori == 4:
 195     for _ in [1, 2, 3, 4]:
 196       if _ == 1:
 197         v = (p - sc, q, r)
 198       elif _ == 2:
 199         v = (p + sc, q, r)
 200       elif _ == 3:
 201         v = (p, q - sc, r)
 202       elif _ == 4:
 203         v = (p, q + sc, r)
 204       nodes_to_validate.append((n, v, _))
 205   return nodes_to_add, nodes_to_validate

 207 def _get_next_nodes(n, sc, n_ori):
 208   nodes_to_add = []
 209   nodes_to_validate = []
 210   (p, q, r) = n
 211   for r_, a_ in zip([-1, 0, 1], [1, 0, 2]):
 212     nodes_to_add.append((n, (p, q, np.mod(r+r_, n_ori)), a_))
 213 
 214   if n_ori == 6:
 215     if r == 0:
 216       v = (p + sc, q, r)
 217     elif r == 1:
 218       v = (p + sc, q + sc, r)
 219     elif r == 2:
 220       v = (p, q + sc, r)
 221     elif r == 3:
 222       v = (p - sc, q, r)
 223     elif r == 4:
 224       v = (p - sc, q - sc, r)
 225     elif r == 5:
 226       v = (p, q - sc, r)
 227   elif n_ori == 4:
 228     if r == 0:
 229       v = (p + sc, q, r)
 230     elif r == 1:
 231       v = (p, q + sc, r)
 232     elif r == 2:
 233       v = (p - sc, q, r)
 234     elif r == 3:
 235       v = (p, q - sc, r)
 236   nodes_to_validate.append((n,v,3))
 237 
 238   return nodes_to_add, nodes_to_validate

 240 def generate_graph(valid_fn_vec=None, sc=1., n_ori=6,
 241                    starting_location=(0, 0, 0), vis=False, directed=True):
 242   timer = utils.Timer()
 243   timer.tic()
 244   if directed: G = nx.DiGraph(directed=True)
 245   else: G = nx.Graph()
 246   G.add_node(starting_location)
 247   new_nodes = G.nodes()
 248   while len(new_nodes) != 0:
 249     nodes_to_add = []
 250     nodes_to_validate = []
 251     for n in new_nodes:
 252       if directed:
 253         na, nv = _get_next_nodes(n, sc, n_ori)
 254       else:
 255         na, nv = _get_next_nodes_undirected(n, sc, n_ori)
 256       nodes_to_add = nodes_to_add + na
 257       if valid_fn_vec is not None:
 258         nodes_to_validate = nodes_to_validate + nv
 259       else:
 260         node_to_add = nodes_to_add + nv
 261 
 262     # Validate nodes.
 263     vs = [_[1] for _ in nodes_to_validate]
 264     valids = valid_fn_vec(vs)
 265 
 266     for nva, valid in zip(nodes_to_validate, valids):
 267       if valid:
 268         nodes_to_add.append(nva)
 269 
 270     new_nodes = []
 271     for n,v,a in nodes_to_add:
 272       if not G.has_node(v):
 273         new_nodes.append(v)
 274       G.add_edge(n, v, action=a)
 275 
 276   timer.toc(average=True, log_at=1, log_str='src.graph_utils.generate_graph')
 277   return (G)

 295 def convert_to_graph_tool(G):
 296   timer = utils.Timer()
 297   timer.tic()
 298   gtG = gt.Graph(directed=G.is_directed())
 299   gtG.ep['action'] = gtG.new_edge_property('int')
 300 
 301   nodes_list = G.nodes()
 302   nodes_array = np.array(nodes_list)
 303 
 304   nodes_id = np.zeros((nodes_array.shape[0],), dtype=np.int64)
 305 
 306   for i in range(nodes_array.shape[0]):
 307     v = gtG.add_vertex()
 308     nodes_id[i] = int(v)
 309 
 310   # d = {key: value for (key, value) in zip(nodes_list, nodes_id)}
 311   d = dict(itertools.izip(nodes_list, nodes_id))
 312 
 313   for src, dst, data in G.edges_iter(data=True):
 314     e = gtG.add_edge(d[src], d[dst])
 315     gtG.ep['action'][e] = data['action']
 316   nodes_to_id = d
 317   timer.toc(average=True, log_at=1, log_str='src.graph_utils.convert_to_graph_tool')
 318   return gtG, nodes_array, nodes_to_id

 321 def _rejection_sampling(rng, sampling_d, target_d, bins, hardness, M):
 322   bin_ind = np.digitize(hardness, bins)-1
 323   i = 0
 324   ratio = target_d[bin_ind] / (M*sampling_d[bin_ind])
 325   while i < ratio.size and rng.rand() > ratio[i]:
 326     i = i+1
 327   return i

 329 def heuristic_fn_vec(n1, n2, n_ori, step_size):
 330   # n1 is a vector and n2 is a single point.
 331   dx = (n1[:,0] - n2[0,0])/step_size
 332   dy = (n1[:,1] - n2[0,1])/step_size
 333   dt = n1[:,2] - n2[0,2]
 334   dt = np.mod(dt, n_ori)
 335   dt = np.minimum(dt, n_ori-dt)
 336 
 337   if n_ori == 6:
 338     if dx*dy > 0:
 339       d = np.maximum(np.abs(dx), np.abs(dy))
 340     else:
 341       d = np.abs(dy-dx)
 342   elif n_ori == 4:
 343     d = np.abs(dx) + np.abs(dy)
 344 
 345   return (d + dt).reshape((-1,1))

 347 def get_hardness_distribution(gtG, max_dist, min_dist, rng, trials, bins, nodes,
 348                               n_ori, step_size):
 349   heuristic_fn = lambda node_ids, node_id: \
 350     heuristic_fn_vec(nodes[node_ids, :], nodes[[node_id], :], n_ori, step_size)
 351   num_nodes = gtG.num_vertices()
 352   gt_dists = []; h_dists = [];
 353   for i in range(trials):
 354     end_node_id = rng.choice(num_nodes)
 355     gt_dist = gt.topology.shortest_distance(gt.GraphView(gtG, reversed=True),
 356                                             source=gtG.vertex(end_node_id),
 357                                             target=None, max_dist=max_dist)
 358     gt_dist = np.array(gt_dist.get_array())
 359     ind = np.where(np.logical_and(gt_dist <= max_dist, gt_dist >= min_dist))[0]
 360     gt_dist = gt_dist[ind]
 361     h_dist = heuristic_fn(ind, end_node_id)[:,0]
 362     gt_dists.append(gt_dist)
 363     h_dists.append(h_dist)
 364   gt_dists = np.concatenate(gt_dists)
 365   h_dists = np.concatenate(h_dists)
 366   hardness = 1. - h_dists*1./gt_dists
 367   hist, _ = np.histogram(hardness, bins)
 368   hist = hist.astype(np.float64)
 369   hist = hist / np.sum(hist)
 370   return hist

 372 def rng_next_goal_rejection_sampling(start_node_ids, batch_size, gtG, rng,
 373                                      max_dist, min_dist, max_dist_to_compute,
 374                                      sampling_d, target_d,
 375                                      nodes, n_ori, step_size, bins, M):
 376   sample_start_nodes = start_node_ids is None
 377   dists = []; pred_maps = []; end_node_ids = []; start_node_ids_ = [];
 378   hardnesss = []; gt_dists = [];
 379   num_nodes = gtG.num_vertices()
 380   for i in range(batch_size):
 381     done = False
 382     while not done:
 383       if sample_start_nodes:
 384         start_node_id = rng.choice(num_nodes)
 385       else:
 386         start_node_id = start_node_ids[i]
 387 
 388       gt_dist = gt.topology.shortest_distance(
 389           gt.GraphView(gtG, reversed=False), source=start_node_id, target=None,
 390           max_dist=max_dist)
 391       gt_dist = np.array(gt_dist.get_array())
 392       ind = np.where(np.logical_and(gt_dist <= max_dist, gt_dist >= min_dist))[0]
 393       ind = rng.permutation(ind)
 394       gt_dist = gt_dist[ind]*1.
 395       h_dist = heuristic_fn_vec(nodes[ind, :], nodes[[start_node_id], :],
 396                                 n_ori, step_size)[:,0]
 397       hardness = 1. - h_dist / gt_dist
 398       sampled_ind = _rejection_sampling(rng, sampling_d, target_d, bins,
 399                                         hardness, M)
 400       if sampled_ind < ind.size:
 401         # print sampled_ind
 402         end_node_id = ind[sampled_ind]
 403         hardness = hardness[sampled_ind]
 404         gt_dist = gt_dist[sampled_ind]
 405         done = True
 406 
 407     # Compute distance from end node to all nodes, to return.
 408     dist, pred_map = gt.topology.shortest_distance(
 409         gt.GraphView(gtG, reversed=True), source=end_node_id, target=None,
 410         max_dist=max_dist_to_compute, pred_map=True)
 411     dist = np.array(dist.get_array())
 412     pred_map = np.array(pred_map.get_array())
 413 
 414     hardnesss.append(hardness); dists.append(dist); pred_maps.append(pred_map);
 415     start_node_ids_.append(start_node_id); end_node_ids.append(end_node_id);
 416     gt_dists.append(gt_dist);
 417     paths = None
 418   return start_node_ids_, end_node_ids, dists, pred_maps, paths, hardnesss, gt_dists

 421 def rng_next_goal(start_node_ids, batch_size, gtG, rng, max_dist,
 422                   max_dist_to_compute, node_room_ids, nodes=None,
 423                   compute_path=False, dists_from_start_node=None):
 424   # Compute the distance field from the starting location, and then pick a
 425   # destination in another room if possible otherwise anywhere outside this
 426   # room.
 427   dists = []; pred_maps = []; paths = []; end_node_ids = [];
 428   for i in range(batch_size):
 429     room_id = node_room_ids[start_node_ids[i]]
 430     # Compute distances.
 431     if dists_from_start_node == None:
 432       dist, pred_map = gt.topology.shortest_distance(
 433         gt.GraphView(gtG, reversed=False), source=gtG.vertex(start_node_ids[i]),
 434         target=None, max_dist=max_dist_to_compute, pred_map=True)
 435       dist = np.array(dist.get_array())
 436     else:
 437       dist = dists_from_start_node[i]
 438 
 439     # Randomly sample nodes which are within max_dist.
 440     near_ids = dist <= max_dist
 441     near_ids = near_ids[:, np.newaxis]
 442     # Check to see if there is a non-negative node which is close enough.
 443     non_same_room_ids = node_room_ids != room_id
 444     non_hallway_ids = node_room_ids != -1
 445     good1_ids = np.logical_and(near_ids, np.logical_and(non_same_room_ids, non_hallway_ids))
 446     good2_ids = np.logical_and(near_ids, non_hallway_ids)
 447     good3_ids = near_ids
 448     if np.any(good1_ids):
 449       end_node_id = rng.choice(np.where(good1_ids)[0])
 450     elif np.any(good2_ids):
 451       end_node_id = rng.choice(np.where(good2_ids)[0])
 452     elif np.any(good3_ids):
 453       end_node_id = rng.choice(np.where(good3_ids)[0])
 454     else:
 455       logging.error('Did not find any good nodes.')
 456 
 457     # Compute distance to this new goal for doing distance queries.
 458     dist, pred_map = gt.topology.shortest_distance(
 459         gt.GraphView(gtG, reversed=True), source=gtG.vertex(end_node_id),
 460         target=None, max_dist=max_dist_to_compute, pred_map=True)
 461     dist = np.array(dist.get_array())
 462     pred_map = np.array(pred_map.get_array())
 463 
 464     dists.append(dist)
 465     pred_maps.append(pred_map)
 466     end_node_ids.append(end_node_id)
 467 
 468     path = None
 469     if compute_path:
 470       path = get_path_ids(start_node_ids[i], end_node_ids[i], pred_map)
 471     paths.append(path)
 472 
 473   return start_node_ids, end_node_ids, dists, pred_maps, paths

 476 def rng_room_to_room(batch_size, gtG, rng, max_dist, max_dist_to_compute,
 477                      node_room_ids, nodes=None, compute_path=False):
 478   # Sample one of the rooms, compute the distance field. Pick a destination in
 479   # another room if possible otherwise anywhere outside this room.
 480   dists = []; pred_maps = []; paths = []; start_node_ids = []; end_node_ids = [];
 481   room_ids = np.unique(node_room_ids[node_room_ids[:,0] >= 0, 0])
 482   for i in range(batch_size):
 483     room_id = rng.choice(room_ids)
 484     end_node_id = rng.choice(np.where(node_room_ids[:,0] == room_id)[0])
 485     end_node_ids.append(end_node_id)
 486 
 487     # Compute distances.
 488     dist, pred_map = gt.topology.shortest_distance(
 489         gt.GraphView(gtG, reversed=True), source=gtG.vertex(end_node_id),
 490         target=None, max_dist=max_dist_to_compute, pred_map=True)
 491     dist = np.array(dist.get_array())
 492     pred_map = np.array(pred_map.get_array())
 493     dists.append(dist)
 494     pred_maps.append(pred_map)
 495 
 496     # Randomly sample nodes which are within max_dist.
 497     near_ids = dist <= max_dist
 498     near_ids = near_ids[:, np.newaxis]
 499 
 500     # Check to see if there is a non-negative node which is close enough.
 501     non_same_room_ids = node_room_ids != room_id
 502     non_hallway_ids = node_room_ids != -1
 503     good1_ids = np.logical_and(near_ids, np.logical_and(non_same_room_ids, non_hallway_ids))
 504     good2_ids = np.logical_and(near_ids, non_hallway_ids)
 505     good3_ids = near_ids
 506     if np.any(good1_ids):
 507       start_node_id = rng.choice(np.where(good1_ids)[0])
 508     elif np.any(good2_ids):
 509       start_node_id = rng.choice(np.where(good2_ids)[0])
 510     elif np.any(good3_ids):
 511       start_node_id = rng.choice(np.where(good3_ids)[0])
 512     else:
 513       logging.error('Did not find any good nodes.')
 514 
 515     start_node_ids.append(start_node_id)
 516 
 517     path = None
 518     if compute_path:
 519       path = get_path_ids(start_node_ids[i], end_node_ids[i], pred_map)
 520     paths.append(path)
 521 
 522   return start_node_ids, end_node_ids, dists, pred_maps, paths

 525 def rng_target_dist_field(batch_size, gtG, rng, max_dist, max_dist_to_compute,
 526                           nodes=None, compute_path=False):
 527   # Sample a single node, compute distance to all nodes less than max_dist,
 528   # sample nodes which are a particular distance away.
 529   dists = []; pred_maps = []; paths = []; start_node_ids = []
 530   end_node_ids = rng.choice(gtG.num_vertices(), size=(batch_size,),
 531                             replace=False).tolist()
 532 
 533   for i in range(batch_size):
 534     dist, pred_map = gt.topology.shortest_distance(
 535         gt.GraphView(gtG, reversed=True), source=gtG.vertex(end_node_ids[i]),
 536         target=None, max_dist=max_dist_to_compute, pred_map=True)
 537     dist = np.array(dist.get_array())
 538     pred_map = np.array(pred_map.get_array())
 539     dists.append(dist)
 540     pred_maps.append(pred_map)
 541 
 542     # Randomly sample nodes which are withing max_dist
 543     near_ids = np.where(dist <= max_dist)[0]
 544     start_node_id = rng.choice(near_ids, size=(1,), replace=False)[0]
 545     start_node_ids.append(start_node_id)
 546 
 547     path = None
 548     if compute_path:
 549       path = get_path_ids(start_node_ids[i], end_node_ids[i], pred_map)
 550     paths.append(path)
 551 
 552   return start_node_ids, end_node_ids, dists, pred_maps, paths
