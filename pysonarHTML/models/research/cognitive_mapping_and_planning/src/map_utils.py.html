  29 def _get_xy_bounding_box(vertex, padding):
  30   """Returns the xy bounding box of the environment."""
  31   min_ = np.floor(np.min(vertex[:, :2], axis=0) - padding).astype(np.int)
  32   max_ = np.ceil(np.max(vertex[:, :2], axis=0) + padding).astype(np.int)
  33   return min_, max_

  35 def _project_to_map(map, vertex, wt=None, ignore_points_outside_map=False):
  36   """Projects points to map, returns how many points are present at each
  37   location."""
  38   num_points = np.zeros((map.size[1], map.size[0]))
  39   vertex_ = vertex[:, :2] - map.origin
  40   vertex_ = np.round(vertex_ / map.resolution).astype(np.int)
  41   if ignore_points_outside_map:
  42     good_ind = np.all(np.array([vertex_[:,1] >= 0, vertex_[:,1] < map.size[1],
  43                                 vertex_[:,0] >= 0, vertex_[:,0] < map.size[0]]),
  44                       axis=0)
  45     vertex_ = vertex_[good_ind, :]
  46     if wt is not None:
  47       wt = wt[good_ind, :]
  48   if wt is None:
  49     np.add.at(num_points, (vertex_[:, 1], vertex_[:, 0]), 1)
  50   else:
  51     assert(wt.shape[0] == vertex.shape[0]), \
  52       'number of weights should be same as vertices.'
  53     np.add.at(num_points, (vertex_[:, 1], vertex_[:, 0]), wt)
  54   return num_points

  56 def make_map(padding, resolution, vertex=None, sc=1.):
  57   """Returns a map structure."""
  58   min_, max_ = _get_xy_bounding_box(vertex*sc, padding=padding)
  59   sz = np.ceil((max_ - min_ + 1) / resolution).astype(np.int32)
  60   max_ = min_ + sz * resolution - 1
  61   map = utils.Foo(origin=min_, size=sz, max=max_, resolution=resolution,
  62                   padding=padding)
  63   return map

  65 def _fill_holes(img, thresh):
  66   """Fills holes less than thresh area (assumes 4 connectivity when computing
  67   hole area."""
  68   l, n = scipy.ndimage.label(np.logical_not(img))
  69   img_ = img == True
  70   cnts = np.bincount(l.reshape(-1))
  71   for i, cnt in enumerate(cnts):
  72     if cnt < thresh:
  73       l[l == i] = -1
  74   img_[l == -1] = True
  75   return img_

  77 def compute_traversibility(map, robot_base, robot_height, robot_radius,
  78                            valid_min, valid_max, num_point_threshold, shapess,
  79                            sc=100., n_samples_per_face=200):
  80   """Returns a bit map with pixels that are traversible or not as long as the
  81   robot center is inside this volume we are good colisions can be detected by
  82   doing a line search on things, or walking from current location to final
  83   location in the bitmap, or doing bwlabel on the traversibility map."""
  84 
  85   tt = utils.Timer()
  86   tt.tic()
  87   num_obstcale_points = np.zeros((map.size[1], map.size[0]))
  88   num_points = np.zeros((map.size[1], map.size[0]))
  89 
  90   for i, shapes in enumerate(shapess):
  91     for j in range(shapes.get_number_of_meshes()):
  92       p, face_areas, face_idx = shapes.sample_points_on_face_of_shape(
  93           j, n_samples_per_face, sc)
  94       wt = face_areas[face_idx]/n_samples_per_face
  95 
  96       ind = np.all(np.concatenate(
  97         (p[:, [2]] > robot_base,
  98          p[:, [2]] < robot_base + robot_height), axis=1),axis=1)
  99       num_obstcale_points += _project_to_map(map, p[ind, :], wt[ind])
 100 
 101       ind = np.all(np.concatenate(
 102         (p[:, [2]] > valid_min,
 103          p[:, [2]] < valid_max), axis=1),axis=1)
 104       num_points += _project_to_map(map, p[ind, :], wt[ind])
 105 
 106   selem = skimage.morphology.disk(robot_radius / map.resolution)
 107   obstacle_free = skimage.morphology.binary_dilation(
 108       _fill_holes(num_obstcale_points > num_point_threshold, 20), selem) != True
 109   valid_space = _fill_holes(num_points > num_point_threshold, 20)
 110   traversible = np.all(np.concatenate((obstacle_free[...,np.newaxis],
 111                                        valid_space[...,np.newaxis]), axis=2),
 112                        axis=2)
 113   # plt.imshow(np.concatenate((obstacle_free, valid_space, traversible), axis=1))
 114   # plt.show()
 115 
 116   map_out = copy.deepcopy(map)
 117   map_out.num_obstcale_points = num_obstcale_points
 118   map_out.num_points = num_points
 119   map_out.traversible = traversible
 120   map_out.obstacle_free = obstacle_free
 121   map_out.valid_space = valid_space
 122   tt.toc(log_at=1, log_str='src.map_utils.compute_traversibility: ')
 123   return map_out

 126 def resize_maps(map, map_scales, resize_method):
 127   scaled_maps = []
 128   for i, sc in enumerate(map_scales):
 129     if resize_method == 'antialiasing':
 130       # Resize using open cv so that we can compute the size.
 131       # Use PIL resize to use anti aliasing feature.
 132       map_ = cv2.resize(map*1, None, None, fx=sc, fy=sc, interpolation=cv2.INTER_LINEAR)
 133       w = map_.shape[1]; h = map_.shape[0]
 134 
 135       map_img = PIL.Image.fromarray((map*255).astype(np.uint8))
 136       map__img = map_img.resize((w,h), PIL.Image.ANTIALIAS)
 137       map_ = np.asarray(map__img).astype(np.float32)
 138       map_ = map_/255.
 139       map_ = np.minimum(map_, 1.0)
 140       map_ = np.maximum(map_, 0.0)
 141     elif resize_method == 'linear_noantialiasing':
 142       map_ = cv2.resize(map*1, None, None, fx=sc, fy=sc, interpolation=cv2.INTER_LINEAR)
 143     else:
 144       logging.error('Unknown resizing method')
 145     scaled_maps.append(map_)
 146   return scaled_maps

 149 def pick_largest_cc(traversible):
 150   out = scipy.ndimage.label(traversible)[0]
 151   cnt = np.bincount(out.reshape(-1))[1:]
 152   return out == np.argmax(cnt) + 1

 154 def get_graph_origin_loc(rng, traversible):
 155   """Erode the traversibility mask so that we get points in the bulk of the
 156   graph, and not end up with a situation where the graph is localized in the
 157   corner of a cramped room. Output Locs is in the coordinate frame of the
 158   map."""
 159 
 160   aa = pick_largest_cc(skimage.morphology.binary_erosion(traversible == True,
 161                                                          selem=np.ones((15,15))))
 162   y, x = np.where(aa > 0)
 163   ind = rng.choice(y.size)
 164   locs = np.array([x[ind], y[ind]])
 165   locs = locs + rng.rand(*(locs.shape)) - 0.5
 166   return locs

 169 def generate_egocentric_maps(scaled_maps, map_scales, map_crop_sizes, loc,
 170                              x_axis, y_axis, theta):
 171   maps = []
 172   for i, (map_, sc, map_crop_size) in enumerate(zip(scaled_maps, map_scales, map_crop_sizes)):
 173     maps_i = np.array(get_map_to_predict(loc*sc, x_axis, y_axis, map_,
 174                                          map_crop_size,
 175                                          interpolation=cv2.INTER_LINEAR)[0])
 176     maps_i[np.isnan(maps_i)] = 0
 177     maps.append(maps_i)
 178   return maps

 180 def generate_goal_images(map_scales, map_crop_sizes, n_ori, goal_dist,
 181                          goal_theta, rel_goal_orientation):
 182   goal_dist = goal_dist[:,0]
 183   goal_theta = goal_theta[:,0]
 184   rel_goal_orientation = rel_goal_orientation[:,0]
 185 
 186   goals = [];
 187   # Generate the map images.
 188   for i, (sc, map_crop_size) in enumerate(zip(map_scales, map_crop_sizes)):
 189     goal_i = np.zeros((goal_dist.shape[0], map_crop_size, map_crop_size, n_ori),
 190                       dtype=np.float32)
 191     x = goal_dist*np.cos(goal_theta)*sc + (map_crop_size-1.)/2.
 192     y = goal_dist*np.sin(goal_theta)*sc + (map_crop_size-1.)/2.
 193 
 194     for j in range(goal_dist.shape[0]):
 195       gc = rel_goal_orientation[j]
 196       x0 = np.floor(x[j]).astype(np.int32); x1 = x0 + 1;
 197       y0 = np.floor(y[j]).astype(np.int32); y1 = y0 + 1;
 198       if x0 >= 0 and x0 <= map_crop_size-1:
 199         if y0 >= 0 and y0 <= map_crop_size-1:
 200           goal_i[j, y0, x0, gc] = (x1-x[j])*(y1-y[j])
 201         if y1 >= 0 and y1 <= map_crop_size-1:
 202           goal_i[j, y1, x0, gc] = (x1-x[j])*(y[j]-y0)
 203 
 204       if x1 >= 0 and x1 <= map_crop_size-1:
 205         if y0 >= 0 and y0 <= map_crop_size-1:
 206           goal_i[j, y0, x1, gc] = (x[j]-x0)*(y1-y[j])
 207         if y1 >= 0 and y1 <= map_crop_size-1:
 208           goal_i[j, y1, x1, gc] = (x[j]-x0)*(y[j]-y0)
 209 
 210     goals.append(goal_i)
 211   return goals

 213 def get_map_to_predict(src_locs, src_x_axiss, src_y_axiss, map, map_size,
 214                        interpolation=cv2.INTER_LINEAR):
 215   fss = []
 216   valids = []
 217 
 218   center = (map_size-1.0)/2.0
 219   dst_theta = np.pi/2.0
 220   dst_loc = np.array([center, center])
 221   dst_x_axis = np.array([np.cos(dst_theta), np.sin(dst_theta)])
 222   dst_y_axis = np.array([np.cos(dst_theta+np.pi/2), np.sin(dst_theta+np.pi/2)])
 223 
 224   def compute_points(center, x_axis, y_axis):
 225     points = np.zeros((3,2),dtype=np.float32)
 226     points[0,:] = center
 227     points[1,:] = center + x_axis
 228     points[2,:] = center + y_axis
 229     return points
 230 
 231   dst_points = compute_points(dst_loc, dst_x_axis, dst_y_axis)
 232   for i in range(src_locs.shape[0]):
 233     src_loc = src_locs[i,:]
 234     src_x_axis = src_x_axiss[i,:]
 235     src_y_axis = src_y_axiss[i,:]
 236     src_points = compute_points(src_loc, src_x_axis, src_y_axis)
 237     M = cv2.getAffineTransform(src_points, dst_points)
 238 
 239     fs = cv2.warpAffine(map, M, (map_size, map_size), None, flags=interpolation,
 240                         borderValue=np.NaN)
 241     valid = np.invert(np.isnan(fs))
 242     valids.append(valid)
 243     fss.append(fs)
 244   return fss, valids
