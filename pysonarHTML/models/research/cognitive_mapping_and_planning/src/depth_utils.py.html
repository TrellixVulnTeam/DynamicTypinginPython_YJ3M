  22 def get_camera_matrix(width, height, fov):
  23   """Returns a camera matrix from image size and fov."""
  24   xc = (width-1.) / 2.
  25   zc = (height-1.) / 2.
  26   f = (width / 2.) / np.tan(np.deg2rad(fov / 2.))
  27   camera_matrix = utils.Foo(xc=xc, zc=zc, f=f)
  28   return camera_matrix

  30 def get_point_cloud_from_z(Y, camera_matrix):
  31   """Projects the depth image Y into a 3D point cloud.
  32   Inputs:
  33     Y is ...xHxW
  34     camera_matrix
  35   Outputs:
  36     X is positive going right
  37     Y is positive into the image
  38     Z is positive up in the image
  39     XYZ is ...xHxWx3
  40   """
  41   x, z = np.meshgrid(np.arange(Y.shape[-1]),
  42                      np.arange(Y.shape[-2]-1, -1, -1))
  43   for i in range(Y.ndim-2):
  44     x = np.expand_dims(x, axis=0)
  45     z = np.expand_dims(z, axis=0)
  46   X = (x-camera_matrix.xc) * Y / camera_matrix.f
  47   Z = (z-camera_matrix.zc) * Y / camera_matrix.f
  48   XYZ = np.concatenate((X[...,np.newaxis], Y[...,np.newaxis],
  49                         Z[...,np.newaxis]), axis=X.ndim)
  50   return XYZ

  52 def make_geocentric(XYZ, sensor_height, camera_elevation_degree):
  53   """Transforms the point cloud into geocentric coordinate frame.
  54   Input:
  55     XYZ                     : ...x3
  56     sensor_height           : height of the sensor
  57     camera_elevation_degree : camera elevation to rectify.
  58   Output:
  59     XYZ : ...x3
  60   """
  61   R = ru.get_r_matrix([1.,0.,0.], angle=np.deg2rad(camera_elevation_degree))
  62   XYZ = np.matmul(XYZ.reshape(-1,3), R.T).reshape(XYZ.shape)
  63   XYZ[...,2] = XYZ[...,2] + sensor_height
  64   return XYZ

  66 def bin_points(XYZ_cms, map_size, z_bins, xy_resolution):
  67   """Bins points into xy-z bins
  68   XYZ_cms is ... x H x W x3
  69   Outputs is ... x map_size x map_size x (len(z_bins)+1)
  70   """
  71   sh = XYZ_cms.shape
  72   XYZ_cms = XYZ_cms.reshape([-1, sh[-3], sh[-2], sh[-1]])
  73   n_z_bins = len(z_bins)+1
  74   map_center = (map_size-1.)/2.
  75   counts = []
  76   isvalids = []
  77   for XYZ_cm in XYZ_cms:
  78     isnotnan = np.logical_not(np.isnan(XYZ_cm[:,:,0]))
  79     X_bin = np.round(XYZ_cm[:,:,0] / xy_resolution + map_center).astype(np.int32)
  80     Y_bin = np.round(XYZ_cm[:,:,1] / xy_resolution + map_center).astype(np.int32)
  81     Z_bin = np.digitize(XYZ_cm[:,:,2], bins=z_bins).astype(np.int32)
  82 
  83     isvalid = np.array([X_bin >= 0, X_bin < map_size, Y_bin >= 0, Y_bin < map_size,
  84                         Z_bin >= 0, Z_bin < n_z_bins, isnotnan])
  85     isvalid = np.all(isvalid, axis=0)
  86 
  87     ind = (Y_bin * map_size + X_bin) * n_z_bins + Z_bin
  88     ind[np.logical_not(isvalid)] = 0
  89     count = np.bincount(ind.ravel(), isvalid.ravel().astype(np.int32),
  90                          minlength=map_size*map_size*n_z_bins)
  91     count = np.reshape(count, [map_size, map_size, n_z_bins])
  92     counts.append(count)
  93     isvalids.append(isvalid)
  94   counts = np.array(counts).reshape(list(sh[:-3]) + [map_size, map_size, n_z_bins])
  95   isvalids = np.array(isvalids).reshape(list(sh[:-3]) + [sh[-3], sh[-2], 1])
  96   return counts, isvalids
