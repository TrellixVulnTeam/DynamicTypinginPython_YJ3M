  17   Shape: Class that exposes high level functions for loading and manipulating
  18     shapes. This currently is bound to assimp

  41 def sample_points_on_faces(vs, fs, rng, n_samples_per_face):
  42   idx = np.repeat(np.arange(fs.shape[0]), n_samples_per_face)
  43   
  44   r = rng.rand(idx.size, 2)
  45   r1 = r[:,:1]; r2 = r[:,1:]; sqrt_r1 = np.sqrt(r1);
  46   
  47   v1 = vs[fs[idx, 0], :]; v2 = vs[fs[idx, 1], :]; v3 = vs[fs[idx, 2], :];
  48   pts = (1-sqrt_r1)*v1 + sqrt_r1*(1-r2)*v2 + sqrt_r1*r2*v3
  49   
  50   v1 = vs[fs[:,0], :]; v2 = vs[fs[:, 1], :]; v3 = vs[fs[:, 2], :];
  51   ar = 0.5*np.sqrt(np.sum(np.cross(v1-v3, v2-v3)**2, 1))
  52   
  53   return pts, ar, idx

  56   def get_pyassimp_load_options(self):
  57     load_flags = assimp.postprocess.aiProcess_Triangulate;
  58     load_flags = load_flags | assimp.postprocess.aiProcess_SortByPType;
  59     load_flags = load_flags | assimp.postprocess.aiProcess_OptimizeMeshes;
  60     load_flags = load_flags | assimp.postprocess.aiProcess_RemoveRedundantMaterials;
  61     load_flags = load_flags | assimp.postprocess.aiProcess_FindDegenerates;
  62     load_flags = load_flags | assimp.postprocess.aiProcess_GenSmoothNormals;
  63     load_flags = load_flags | assimp.postprocess.aiProcess_JoinIdenticalVertices;
  64     load_flags = load_flags | assimp.postprocess.aiProcess_ImproveCacheLocality;
  65     load_flags = load_flags | assimp.postprocess.aiProcess_GenUVCoords;
  66     load_flags = load_flags | assimp.postprocess.aiProcess_FindInvalidData;
  67     return load_flags

  69   def __init__(self, obj_file, material_file=None, load_materials=True,
  70                name_prefix='', name_suffix=''):
  71     if material_file is not None:
  72       logging.error('Ignoring material file input, reading them off obj file.')
  73     load_flags = self.get_pyassimp_load_options()
  74     scene = assimp.load(obj_file, processing=load_flags)
  75     filter_ind = self._filter_triangles(scene.meshes)
  76     self.meshes = [scene.meshes[i] for i in filter_ind]
  77     for m in self.meshes:
  78       m.name = name_prefix + m.name + name_suffix
  79 
  80     dir_name = os.path.dirname(obj_file)
  81     # Load materials
  82     materials = None
  83     if load_materials:
  84       materials = []
  85       for m in self.meshes:
  86         file_name = os.path.join(dir_name, m.material.properties[('file', 1)])
  87         assert(os.path.exists(file_name)), \
  88             'Texture file {:s} foes not exist.'.format(file_name)
  89         img_rgb = cv2.imread(file_name)[::-1,:,::-1]
  90         if img_rgb.shape[0] != img_rgb.shape[1]:
  91           logging.warn('Texture image not square.')
  92           sz = np.maximum(img_rgb.shape[0], img_rgb.shape[1])
  93           sz = int(np.power(2., np.ceil(np.log2(sz))))
  94           img_rgb = cv2.resize(img_rgb, (sz,sz), interpolation=cv2.INTER_LINEAR)
  95         else:
  96           sz = img_rgb.shape[0]
  97           sz_ = int(np.power(2., np.ceil(np.log2(sz))))
  98           if sz != sz_:
  99             logging.warn('Texture image not square of power of 2 size. ' +
 100                          'Changing size from %d to %d.', sz, sz_)
 101             sz = sz_
 102             img_rgb = cv2.resize(img_rgb, (sz,sz), interpolation=cv2.INTER_LINEAR)
 103         materials.append(img_rgb)
 104     self.scene = scene
 105     self.materials = materials

 107   def _filter_triangles(self, meshes):
 108     select = []
 109     for i in range(len(meshes)):
 110       if meshes[i].primitivetypes == 4:
 111         select.append(i)
 112     return select

 114   def flip_shape(self):
 115     for m in self.meshes:
 116       m.vertices[:,1] = -m.vertices[:,1]
 117       bb = m.faces*1
 118       bb[:,1] = m.faces[:,2]
 119       bb[:,2] = m.faces[:,1]
 120       m.faces = bb
 121       # m.vertices[:,[0,1]] = m.vertices[:,[1,0]]
 122 
 123   def get_vertices(self):
 124     vs = []
 125     for m in self.meshes:
 126       vs.append(m.vertices)
 127     vss = np.concatenate(vs, axis=0)
 128     return vss, vs
 129 
 130   def get_faces(self):
 131     vs = []
 132     for m in self.meshes:
 133       v = m.faces
 134       vs.append(v)
 135     return vs

 137   def get_number_of_meshes(self):
 138     return len(self.meshes)

 143   def sample_points_on_face_of_shape(self, i, n_samples_per_face, sc):
 144     v = self.meshes[i].vertices*sc
 145     f = self.meshes[i].faces
 146     p, face_areas, face_idx = sample_points_on_faces(
 147         v, f, np.random.RandomState(0), n_samples_per_face)
 148     return p, face_areas, face_idx

 180   def init_renderer_egl(self, width, height):
 181     major,minor = ctypes.c_long(),ctypes.c_long()
 182     logging.info('init_renderer_egl: EGL_DEFAULT_DISPLAY: %s', EGL_DEFAULT_DISPLAY)
 183 
 184     egl_display = eglGetDisplay(EGL_DEFAULT_DISPLAY)
 185     logging.info('init_renderer_egl: egl_display: %s', egl_display)
 186 
 187     eglInitialize(egl_display, major, minor)
 188     logging.info('init_renderer_egl: EGL_OPENGL_API, EGL_OPENGL_ES_API: %s, %s',
 189                  EGL_OPENGL_API, EGL_OPENGL_ES_API)
 190     eglBindAPI(EGL_OPENGL_ES_API)
 191 
 192     num_configs = ctypes.c_long()
 193     configs = (EGLConfig*1)()
 194     local_attributes = [EGL_RED_SIZE, 8, EGL_GREEN_SIZE, 8, EGL_BLUE_SIZE, 8,
 195                         EGL_DEPTH_SIZE, 16, EGL_SURFACE_TYPE, EGL_PBUFFER_BIT,
 196                         EGL_RENDERABLE_TYPE, EGL_OPENGL_ES2_BIT, EGL_NONE,]
 197     logging.error('init_renderer_egl: local attributes: %s', local_attributes)
 198     local_attributes = arrays.GLintArray.asArray(local_attributes)
 199     success = eglChooseConfig(egl_display, local_attributes, configs, 1, num_configs)
 200     logging.error('init_renderer_egl: eglChooseConfig success, num_configs: %d, %d', success, num_configs.value)
 201     egl_config = configs[0]
 202 
 203 
 204     context_attributes = [EGL_CONTEXT_CLIENT_VERSION, 2, EGL_NONE]
 205     context_attributes = arrays.GLintArray.asArray(context_attributes)
 206     egl_context = eglCreateContext(egl_display, egl_config, EGL_NO_CONTEXT, context_attributes)
 207 
 208     buffer_attributes = [EGL_WIDTH, width, EGL_HEIGHT, height, EGL_NONE]
 209     buffer_attributes = arrays.GLintArray.asArray(buffer_attributes)
 210     egl_surface = eglCreatePbufferSurface(egl_display, egl_config, buffer_attributes)

 214     logging.error("init_renderer_egl: egl_display: %s egl_surface: %s, egl_config: %s", egl_display, egl_surface, egl_context)

 227   def create_shaders(self, v_shader_file, f_shader_file):
 228     v_shader = glCreateShader(GL_VERTEX_SHADER)
 229     with open(v_shader_file, 'r') as f:
 230       ls = ''
 231       for l in f:
 232         ls = ls + l
 233     glShaderSource(v_shader, ls)
 234     glCompileShader(v_shader);
 235     assert(glGetShaderiv(v_shader, GL_COMPILE_STATUS) == 1)
 236 
 237     f_shader = glCreateShader(GL_FRAGMENT_SHADER)
 238     with open(f_shader_file, 'r') as f:
 239       ls = ''
 240       for l in f:
 241         ls = ls + l
 242     glShaderSource(f_shader, ls)
 243     glCompileShader(f_shader);
 244     assert(glGetShaderiv(f_shader, GL_COMPILE_STATUS) == 1)
 245 
 246     egl_program = glCreateProgram();
 247     assert(egl_program)
 248     glAttachShader(egl_program, v_shader)
 249     glAttachShader(egl_program, f_shader)
 250     glLinkProgram(egl_program);
 251     assert(glGetProgramiv(egl_program, GL_LINK_STATUS) == 1)
 252     glUseProgram(egl_program)
 253 
 254     glBindAttribLocation(egl_program, 0, "aPosition")
 255     glBindAttribLocation(egl_program, 1, "aColor")
 256     glBindAttribLocation(egl_program, 2, "aTextureCoord")
 257 
 258     self.egl_program = egl_program
 259     self.egl_mapping['vertexs'] = 0
 260     self.egl_mapping['vertexs_color'] = 1
 261     self.egl_mapping['vertexs_tc'] = 2
 262     
 263     glClearColor(0.0, 0.0, 0.0, 1.0);
 264     # glEnable(GL_CULL_FACE); glCullFace(GL_BACK);
 265     glEnable(GL_DEPTH_TEST);
 266     
 267     glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT)

 269   def set_camera(self, fov_vertical, z_near, z_far, aspect):
 270     width = 2*np.tan(np.deg2rad(fov_vertical)/2.0)*z_near*aspect;
 271     height = 2*np.tan(np.deg2rad(fov_vertical)/2.0)*z_near;
 272     egl_program = self.egl_program
 273     c = np.eye(4, dtype=np.float32)
 274     c[3,3] = 0
 275     c[3,2] = -1
 276     c[2,2] = -(z_near+z_far)/(z_far-z_near)
 277     c[2,3] = -2.0*(z_near*z_far)/(z_far-z_near)
 278     c[0,0] = 2.0*z_near/width
 279     c[1,1] = 2.0*z_near/height
 280     c = c.T
 281     
 282     projection_matrix_o = glGetUniformLocation(egl_program, 'uProjectionMatrix')
 283     projection_matrix = np.eye(4, dtype=np.float32)
 284     projection_matrix[...] = c
 285     projection_matrix = np.reshape(projection_matrix, (-1))
 286     glUniformMatrix4fv(projection_matrix_o, 1, GL_FALSE, projection_matrix)

 289   def load_default_object(self):
 290     v = np.array([[0.0, 0.5, 0.0, 1.0, 1.0, 0.0, 1.0],
 291                   [-0.5, -0.5, 0.0, 1.0, 0.0, 1.0, 1.0],
 292                   [0.5, -0.5, 0.0, 1.0, 1.0, 1.0, 1.0]], dtype=np.float32)
 293     v = np.concatenate((v,v+0.1), axis=0)
 294     v = np.ascontiguousarray(v, dtype=np.float32)
 295 
 296     vbo = glGenBuffers(1)
 297     glBindBuffer (GL_ARRAY_BUFFER, vbo)
 298     glBufferData (GL_ARRAY_BUFFER, v.dtype.itemsize*v.size, v, GL_STATIC_DRAW)
 299     glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 28, ctypes.c_void_p(0))
 300     glVertexAttribPointer(1, 4, GL_FLOAT, GL_FALSE, 28, ctypes.c_void_p(12))
 301     glEnableVertexAttribArray(0);
 302     glEnableVertexAttribArray(1);
 303 
 304     self.num_to_render = 6;

 306   def _actual_render(self):
 307     for entity_id, entity in self.entities.iteritems():
 308       if entity['visible']:
 309         vbo = entity['vbo']
 310         tbo = entity['tbo']
 311         num = entity['num']
 312 
 313         glBindBuffer(GL_ARRAY_BUFFER, vbo)
 314         glVertexAttribPointer(self.egl_mapping['vertexs'], 3, GL_FLOAT, GL_FALSE,
 315                               20, ctypes.c_void_p(0))
 316         glVertexAttribPointer(self.egl_mapping['vertexs_tc'], 2, GL_FLOAT,
 317                               GL_FALSE, 20, ctypes.c_void_p(12))
 318         glEnableVertexAttribArray(self.egl_mapping['vertexs']);
 319         glEnableVertexAttribArray(self.egl_mapping['vertexs_tc']);
 320 
 321         glBindTexture(GL_TEXTURE_2D, tbo)
 322         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
 323         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
 324         glDrawArrays(GL_TRIANGLES, 0, num)

 326   def render(self, take_screenshot=False, output_type=0):
 327     # self.render_timer.tic()
 328     self._actual_render()
 329     # self.render_timer.toc(log_at=1000, log_str='render timer', type='time')
 330 
 331     np_rgb_img = None
 332     np_d_img = None
 333     c = 1000.
 334     if take_screenshot:
 335       if self.modality == 'rgb':
 336         screenshot_rgba = np.zeros((self.height, self.width, 4), dtype=np.uint8)
 337         glReadPixels(0, 0, self.width, self.height, GL_RGBA, GL_UNSIGNED_BYTE, screenshot_rgba)
 338         np_rgb_img = screenshot_rgba[::-1,:,:3];
 339 
 340       if self.modality == 'depth': 
 341         screenshot_d = np.zeros((self.height, self.width, 4), dtype=np.uint8)
 342         glReadPixels(0, 0, self.width, self.height, GL_RGBA, GL_UNSIGNED_BYTE, screenshot_d)
 343         np_d_img = screenshot_d[::-1,:,:3];
 344         np_d_img = np_d_img[:,:,2]*(255.*255./c) + np_d_img[:,:,1]*(255./c) + np_d_img[:,:,0]*(1./c)
 345         np_d_img = np_d_img.astype(np.float32)
 346         np_d_img[np_d_img == 0] = np.NaN
 347         np_d_img = np_d_img[:,:,np.newaxis]
 348 
 349     glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT)
 350     return np_rgb_img, np_d_img

 352   def _load_mesh_into_gl(self, mesh, material):
 353     vvt = np.concatenate((mesh.vertices, mesh.texturecoords[0,:,:2]), axis=1)
 354     vvt = np.ascontiguousarray(vvt[mesh.faces.reshape((-1)),:], dtype=np.float32)
 355     num = vvt.shape[0]
 356     vvt = np.reshape(vvt, (-1))
 357 
 358     vbo = glGenBuffers(1)
 359     glBindBuffer(GL_ARRAY_BUFFER, vbo)
 360     glBufferData(GL_ARRAY_BUFFER, vvt.dtype.itemsize*vvt.size, vvt, GL_STATIC_DRAW)
 361 
 362     tbo = glGenTextures(1)
 363     glBindTexture(GL_TEXTURE_2D, tbo)
 364     glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, material.shape[1],
 365                  material.shape[0], 0, GL_RGB, GL_UNSIGNED_BYTE,
 366                  np.reshape(material, (-1)))
 367     return num, vbo, tbo

 369   def load_shapes(self, shapes):
 370     entities = self.entities
 371     entity_ids = []
 372     for i, shape in enumerate(shapes):
 373       for j in range(len(shape.meshes)):
 374         name = shape.meshes[j].name
 375         assert name not in entities, '{:s} entity already exists.'.format(name)
 376         num, vbo, tbo = self._load_mesh_into_gl(shape.meshes[j], shape.materials[j])
 377         entities[name] = {'num': num, 'vbo': vbo, 'tbo': tbo, 'visible': False}
 378         entity_ids.append(name)
 379     return entity_ids

 385   def position_camera(self, camera_xyz, lookat_xyz, up):
 386     camera_xyz = np.array(camera_xyz)
 387     lookat_xyz = np.array(lookat_xyz)
 388     up = np.array(up)
 389     lookat_to = lookat_xyz - camera_xyz
 390     lookat_from = np.array([0, 1., 0.])
 391     up_from = np.array([0, 0., 1.])
 392     up_to = up * 1.
 393     # np.set_printoptions(precision=2, suppress=True)
 394     # print up_from, lookat_from, up_to, lookat_to
 395     r = ru.rotate_camera_to_point_at(up_from, lookat_from, up_to, lookat_to)
 396     R = np.eye(4, dtype=np.float32)
 397     R[:3,:3] = r
 398 
 399     t = np.eye(4, dtype=np.float32)
 400     t[:3,3] = -camera_xyz
 401 
 402     view_matrix = np.dot(R.T, t)
 403     flip_yz = np.eye(4, dtype=np.float32)
 404     flip_yz[1,1] = 0; flip_yz[2,2] = 0; flip_yz[1,2] = 1; flip_yz[2,1] = -1;
 405     view_matrix = np.dot(flip_yz, view_matrix)
 406     view_matrix = view_matrix.T
 407     # print np.concatenate((R, t, view_matrix), axis=1)
 408     view_matrix = np.reshape(view_matrix, (-1))
 409     view_matrix_o = glGetUniformLocation(self.egl_program, 'uViewMatrix')
 410     glUniformMatrix4fv(view_matrix_o, 1, GL_FALSE, view_matrix)
 411     return None, None #camera_xyz, q
