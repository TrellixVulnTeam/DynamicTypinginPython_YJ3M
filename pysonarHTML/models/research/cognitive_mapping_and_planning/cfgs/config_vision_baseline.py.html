  62 def get_arch_vars(arch_str):
  63   if arch_str == '': vals = []
  64   else: vals = arch_str.split('_')
  65   
  66   ks = ['ver', 'lstm_dim', 'dropout']
  67   
  68   # Exp Ver
  69   if len(vals) == 0: vals.append('v0')
  70   # LSTM dimentsions
  71   if len(vals) == 1: vals.append('lstm2048')
  72   # Dropout
  73   if len(vals) == 2: vals.append('noDO')
  74   
  75   assert(len(vals) == 3)
  76   
  77   vars = utils.Foo()
  78   for k, v in zip(ks, vals):
  79     setattr(vars, k, v)
  80   
  81   logging.error('arch_vars: %s', vars)
  82   return vars

  84 def process_arch_str(args, arch_str):
  85   # This function modifies args.
  86   args.arch = get_default_baseline_args()
  87   arch_vars = get_arch_vars(arch_str)
  88 
  89   args.navtask.task_params.outputs.rel_goal_loc = True
  90   args.navtask.task_params.input_type = 'vision'
  91   args.navtask.task_params.outputs.images = True
  92   
  93   if args.navtask.camera_param.modalities[0] == 'rgb':
  94     args.solver.pretrained_path = rgb_resnet_v2_50_path
  95   elif args.navtask.camera_param.modalities[0] == 'depth':
  96     args.solver.pretrained_path = d_resnet_v2_50_path
  97   else:
  98     logging.fatal('Neither of rgb or d')
  99 
 100   if arch_vars.dropout == 'DO': 
 101     args.arch.fc_dropout = 0.5
 102 
 103   args.tfcode = 'B'
 104   
 105   exp_ver = arch_vars.ver
 106   if exp_ver == 'v0':
 107     # Multiplicative interaction between goal loc and image features.
 108     args.arch.combine_type = 'multiply'
 109     args.arch.pred_neurons = [256, 256]
 110     args.arch.goal_embed_neurons = [64, 8]
 111     args.arch.img_embed_neurons = [1024, 512, 256*8]
 112   
 113   elif exp_ver == 'v1':
 114     # Additive interaction between goal and image features.
 115     args.arch.combine_type = 'add'
 116     args.arch.pred_neurons = [256, 256]
 117     args.arch.goal_embed_neurons = [64, 256]
 118     args.arch.img_embed_neurons = [1024, 512, 256]
 119   
 120   elif exp_ver == 'v2':
 121     # LSTM at the output on top of multiple interactions.
 122     args.arch.combine_type = 'multiply'
 123     args.arch.goal_embed_neurons = [64, 8]
 124     args.arch.img_embed_neurons = [1024, 512, 256*8]
 125     args.arch.lstm_output = True
 126     args.arch.lstm_output_dim = int(arch_vars.lstm_dim[4:])
 127     args.arch.pred_neurons = [256] # The other is inside the LSTM.
 128   
 129   elif exp_ver == 'v0blind':
 130     # LSTM only on the goal location.
 131     args.arch.combine_type = 'goalonly'
 132     args.arch.goal_embed_neurons = [64, 256]
 133     args.arch.img_embed_neurons = [2] # I dont know what it will do otherwise.
 134     args.arch.lstm_output = True
 135     args.arch.lstm_output_dim = 256
 136     args.arch.pred_neurons = [256] # The other is inside the LSTM.
 137   
 138   else:
 139     logging.fatal('exp_ver: %s undefined', exp_ver)
 140     assert(False)
 141 
 142   # Log the arguments
 143   logging.error('%s', args)
 144   return args

 146 def get_args_for_config(config_name):
 147   args = utils.Foo()
 148 
 149   args.summary, args.control = get_default_args()
 150 
 151   exp_name, mode_str = config_name.split('+')
 152   arch_str, solver_str, navtask_str = exp_name.split('.')
 153   logging.error('config_name: %s', config_name)
 154   logging.error('arch_str: %s', arch_str)
 155   logging.error('navtask_str: %s', navtask_str)
 156   logging.error('solver_str: %s', solver_str)
 157   logging.error('mode_str: %s', mode_str)
 158 
 159   args.solver = cc.process_solver_str(solver_str)
 160   args.navtask = cc.process_navtask_str(navtask_str)
 161 
 162   args = process_arch_str(args, arch_str)
 163   args.arch.isd_k = args.solver.isd_k
 164 
 165   # Train, test, etc.
 166   mode, imset = mode_str.split('_')
 167   args = cc.adjust_args_for_mode(args, mode)
 168   args.navtask.building_names = args.navtask.dataset.get_split(imset)
 169   args.control.test_name = '{:s}_on_{:s}'.format(mode, imset)
 170 
 171   # Log the arguments
 172   logging.error('%s', args)
 173   return args
