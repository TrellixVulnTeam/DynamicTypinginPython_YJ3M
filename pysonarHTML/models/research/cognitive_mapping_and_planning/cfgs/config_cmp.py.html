  77 def get_arch_vars(arch_str):
  78   if arch_str == '': vals = []
  79   else: vals = arch_str.split('_')
  80   ks = ['var1', 'var2', 'var3']
  81   ks = ks[:len(vals)]
  82   
  83   # Exp Ver.
  84   if len(vals) == 0: ks.append('var1'); vals.append('v0')
  85   # custom arch.
  86   if len(vals) == 1: ks.append('var2'); vals.append('')
  87   # map scape for projection baseline.
  88   if len(vals) == 2: ks.append('var3'); vals.append('fr2')
  89 
  90   assert(len(vals) == 3)
  91 
  92   vars = utils.Foo()
  93   for k, v in zip(ks, vals):
  94     setattr(vars, k, v)
  95 
  96   logging.error('arch_vars: %s', vars)
  97   return vars

  99 def process_arch_str(args, arch_str):
 100   # This function modifies args.
 101   args.arch, args.mapper_arch = get_default_cmp_args()
 102 
 103   arch_vars = get_arch_vars(arch_str)
 104 
 105   args.navtask.task_params.outputs.ego_maps = True
 106   args.navtask.task_params.outputs.ego_goal_imgs = True
 107   args.navtask.task_params.outputs.egomotion = True
 108   args.navtask.task_params.toy_problem = False
 109 
 110   if arch_vars.var1 == 'lmap':
 111     args = process_arch_learned_map(args, arch_vars)
 112 
 113   elif arch_vars.var1 == 'pmap':
 114     args = process_arch_projected_map(args, arch_vars)
 115 
 116   else:
 117     logging.fatal('arch_vars.var1 should be lmap or pmap, but is %s', arch_vars.var1)
 118     assert(False)
 119 
 120   return args

 122 def process_arch_learned_map(args, arch_vars):
 123   # Multiscale vision based system.
 124   args.navtask.task_params.input_type = 'vision'
 125   args.navtask.task_params.outputs.images = True
 126   
 127   if args.navtask.camera_param.modalities[0] == 'rgb':
 128     args.solver.pretrained_path = rgb_resnet_v2_50_path
 129   elif args.navtask.camera_param.modalities[0] == 'depth':
 130     args.solver.pretrained_path = d_resnet_v2_50_path
 131 
 132   if arch_vars.var2 == 'Ssc':
 133     sc = 1./args.navtask.task_params.step_size
 134     args.arch.vin_num_iters = 40
 135     args.navtask.task_params.map_scales = [sc]
 136     max_dist = args.navtask.task_params.max_dist * \
 137         args.navtask.task_params.num_goals
 138     args.navtask.task_params.map_crop_sizes = [2*max_dist]
 139 
 140     args.arch.fr_stride = 1
 141     args.arch.vin_action_neurons = 8
 142     args.arch.vin_val_neurons = 3
 143     args.arch.fr_inside_neurons = 32
 144 
 145     args.mapper_arch.pad_map_with_zeros_each = [24]
 146     args.mapper_arch.deconv_neurons = [64, 32, 16]
 147     args.mapper_arch.deconv_strides = [1, 2, 1]
 148 
 149   elif (arch_vars.var2 == 'Msc' or arch_vars.var2 == 'MscROMms' or
 150         arch_vars.var2 == 'MscROMss' or arch_vars.var2 == 'MscNoVin'):
 151     # Code for multi-scale planner.
 152     args.arch.vin_num_iters = 8
 153     args.arch.crop_remove_each = 4
 154     args.arch.value_crop_size = 8
 155 
 156     sc = 1./args.navtask.task_params.step_size
 157     max_dist = args.navtask.task_params.max_dist * \
 158         args.navtask.task_params.num_goals
 159     n_scales = np.log2(float(max_dist) / float(args.arch.vin_num_iters))
 160     n_scales = int(np.ceil(n_scales)+1)
 161 
 162     args.navtask.task_params.map_scales = \
 163         list(sc*(0.5**(np.arange(n_scales))[::-1]))
 164     args.navtask.task_params.map_crop_sizes = [16 for x in range(n_scales)]
 165 
 166     args.arch.fr_stride = 1
 167     args.arch.vin_action_neurons = 8
 168     args.arch.vin_val_neurons = 3
 169     args.arch.fr_inside_neurons = 32
 170 
 171     args.mapper_arch.pad_map_with_zeros_each = [0 for _ in range(n_scales)]
 172     args.mapper_arch.deconv_neurons = [64*n_scales, 32*n_scales, 16*n_scales]
 173     args.mapper_arch.deconv_strides = [1, 2, 1]
 174 
 175     if arch_vars.var2 == 'MscNoVin':
 176       # No planning version.
 177       args.arch.fr_stride = [1, 2, 1, 2]
 178       args.arch.vin_action_neurons = None
 179       args.arch.vin_val_neurons = 16
 180       args.arch.fr_inside_neurons = 32
 181 
 182       args.arch.crop_remove_each = 0
 183       args.arch.value_crop_size = 4
 184       args.arch.vin_num_iters = 0
 185 
 186     elif arch_vars.var2 == 'MscROMms' or arch_vars.var2 == 'MscROMss':
 187       # Code with read outs, MscROMms flattens and reads out,
 188       # MscROMss does not flatten and produces output at multiple scales.
 189       args.navtask.task_params.outputs.readout_maps = True
 190       args.navtask.task_params.map_resize_method = 'antialiasing'
 191       args.arch.readout_maps = True
 192 
 193       if arch_vars.var2 == 'MscROMms':
 194         args.arch.rom_arch.num_neurons = [64, 1]
 195         args.arch.rom_arch.kernel_size = 4
 196         args.arch.rom_arch.strides = [2,2]
 197         args.arch.rom_arch.layers_per_block = 2
 198 
 199         args.navtask.task_params.readout_maps_crop_sizes = [64]
 200         args.navtask.task_params.readout_maps_scales = [sc]
 201 
 202       elif arch_vars.var2 == 'MscROMss':
 203         args.arch.rom_arch.num_neurons = \
 204             [64, len(args.navtask.task_params.map_scales)]
 205         args.arch.rom_arch.kernel_size = 4
 206         args.arch.rom_arch.strides = [1,1]
 207         args.arch.rom_arch.layers_per_block = 1
 208 
 209         args.navtask.task_params.readout_maps_crop_sizes = \
 210             args.navtask.task_params.map_crop_sizes
 211         args.navtask.task_params.readout_maps_scales = \
 212             args.navtask.task_params.map_scales
 213 
 214   else:
 215     logging.fatal('arch_vars.var2 not one of Msc, MscROMms, MscROMss, MscNoVin.')
 216     assert(False)
 217 
 218   map_channels = args.mapper_arch.deconv_neurons[-1] / \
 219     (2*len(args.navtask.task_params.map_scales))
 220   args.navtask.task_params.map_channels = map_channels
 221   
 222   return args

 224 def process_arch_projected_map(args, arch_vars):
 225   # Single scale vision based system which does not use a mapper but instead
 226   # uses an analytically estimated map.
 227   ds = int(arch_vars.var3[2])
 228   args.navtask.task_params.input_type = 'analytical_counts'
 229   args.navtask.task_params.outputs.analytical_counts = True
 230 
 231   assert(args.navtask.task_params.modalities[0] == 'depth')
 232   args.navtask.camera_param.img_channels = None
 233 
 234   analytical_counts = utils.Foo(map_sizes=[512/ds],
 235                                 xy_resolution=[5.*ds],
 236                                 z_bins=[[-10, 10, 150, 200]],
 237                                 non_linearity=[arch_vars.var2])
 238   args.navtask.task_params.analytical_counts = analytical_counts
 239 
 240   sc = 1./ds
 241   args.arch.vin_num_iters = 36
 242   args.navtask.task_params.map_scales = [sc]
 243   args.navtask.task_params.map_crop_sizes = [512/ds]
 244 
 245   args.arch.fr_stride = [1,2]
 246   args.arch.vin_action_neurons = 8
 247   args.arch.vin_val_neurons = 3
 248   args.arch.fr_inside_neurons = 32
 249 
 250   map_channels = len(analytical_counts.z_bins[0]) + 1
 251   args.navtask.task_params.map_channels = map_channels
 252   args.solver.freeze_conv = False
 253 
 254   return args

 256 def get_args_for_config(config_name):
 257   args = utils.Foo()
 258 
 259   args.summary, args.control = get_default_args()
 260 
 261   exp_name, mode_str = config_name.split('+')
 262   arch_str, solver_str, navtask_str = exp_name.split('.')
 263   logging.error('config_name: %s', config_name)
 264   logging.error('arch_str: %s', arch_str)
 265   logging.error('navtask_str: %s', navtask_str)
 266   logging.error('solver_str: %s', solver_str)
 267   logging.error('mode_str: %s', mode_str)
 268 
 269   args.solver = cc.process_solver_str(solver_str)
 270   args.navtask = cc.process_navtask_str(navtask_str)
 271 
 272   args = process_arch_str(args, arch_str)
 273   args.arch.isd_k = args.solver.isd_k
 274 
 275   # Train, test, etc.
 276   mode, imset = mode_str.split('_')
 277   args = cc.adjust_args_for_mode(args, mode)
 278   args.navtask.building_names = args.navtask.dataset.get_split(imset)
 279   args.control.test_name = '{:s}_on_{:s}'.format(mode, imset)
 280 
 281   # Log the arguments
 282   logging.error('%s', args)
 283   return args
