  76 def get_vars(config_name):
  77   vars = config_name.split('_')
  78   if len(vars) == 1: # All data or not.
  79     vars.append('noall')
  80   if len(vars) == 2: # n_ori
  81     vars.append('4')
  82   logging.error('vars: %s', vars)
  83   return vars
  84 
  85 def get_args_for_config(config_name):
  86   args = get_default_args()
  87   config_name, mode = config_name.split('+')
  88   vars = get_vars(config_name)
  89   
  90   logging.info('config_name: %s, mode: %s', config_name, mode)
  91   
  92   args.buildinger.task_params.n_ori = int(vars[2])
  93   args.solver.freeze_conv = True
  94   args.solver.pretrained_path = rgb_resnet_v2_50_path
  95   args.buildinger.task_params.img_channels = 5
  96   args.solver.data_loss_wt = 0.00001
  97  
  98   if vars[0] == 'v0':
  99     None
 100   else:
 101     logging.error('config_name: %s undefined', config_name)
 102 
 103   args.buildinger.task_params.height = args.buildinger.camera_param.height
 104   args.buildinger.task_params.width = args.buildinger.camera_param.width
 105   args.buildinger.task_params.modalities = args.buildinger.camera_param.modalities
 106   
 107   if vars[1] == 'all':
 108     args = cc.get_args_for_mode_building_all(args, mode)
 109   elif vars[1] == 'noall':
 110     args = cc.get_args_for_mode_building(args, mode)
 111   
 112   # Log the arguments
 113   logging.error('%s', args)
 114   return args
