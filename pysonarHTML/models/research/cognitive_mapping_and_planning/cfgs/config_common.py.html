  23 def adjust_args_for_mode(args, mode):
  24   if mode == 'train':
  25     args.control.train = True
  26   
  27   elif mode == 'val1':
  28     # Same settings as for training, to make sure nothing wonky is happening
  29     # there.
  30     args.control.test = True
  31     args.control.test_mode = 'val'
  32     args.navtask.task_params.batch_size = 32
  33 
  34   elif mode == 'val2':
  35     # No data augmentation, not sampling but taking the argmax action, not
  36     # sampling from the ground truth at all.
  37     args.control.test = True
  38     args.arch.action_sample_type = 'argmax'
  39     args.arch.sample_gt_prob_type = 'zero'
  40     args.navtask.task_params.data_augment = \
  41       utils.Foo(lr_flip=0, delta_angle=0, delta_xy=0, relight=False,
  42                 relight_fast=False, structured=False)
  43     args.control.test_mode = 'val'
  44     args.navtask.task_params.batch_size = 32
  45 
  46   elif mode == 'bench':
  47     # Actually testing the agent in settings that are kept same between
  48     # different runs.
  49     args.navtask.task_params.batch_size = 16
  50     args.control.test = True
  51     args.arch.action_sample_type = 'argmax'
  52     args.arch.sample_gt_prob_type = 'zero'
  53     args.navtask.task_params.data_augment = \
  54       utils.Foo(lr_flip=0, delta_angle=0, delta_xy=0, relight=False,
  55                 relight_fast=False, structured=False)
  56     args.summary.test_iters = 250
  57     args.control.only_eval_when_done = True
  58     args.control.reset_rng_seed = True
  59     args.control.test_mode = 'test'
  60   else:
  61     logging.fatal('Unknown mode: %s.', mode)
  62     assert(False)
  63   return args

  65 def get_solver_vars(solver_str):
  66   if solver_str == '': vals = []; 
  67   else: vals = solver_str.split('_')
  68   ks = ['clip', 'dlw', 'long', 'typ', 'isdk', 'adam_eps', 'init_lr'];
  69   ks = ks[:len(vals)]
  70 
  71   # Gradient clipping or not.
  72   if len(vals) == 0: ks.append('clip'); vals.append('noclip');
  73   # data loss weight.
  74   if len(vals) == 1: ks.append('dlw');  vals.append('dlw20')
  75   # how long to train for.
  76   if len(vals) == 2: ks.append('long');  vals.append('nolong')
  77   # Adam
  78   if len(vals) == 3: ks.append('typ');  vals.append('adam2')
  79   # reg loss wt
  80   if len(vals) == 4: ks.append('rlw');  vals.append('rlw1')
  81   # isd_k
  82   if len(vals) == 5: ks.append('isdk');  vals.append('isdk415') # 415, inflexion at 2.5k.
  83   # adam eps
  84   if len(vals) == 6: ks.append('adam_eps');  vals.append('aeps1en8')
  85   # init lr
  86   if len(vals) == 7: ks.append('init_lr');  vals.append('lr1en3')
  87 
  88   assert(len(vals) == 8)
  89   
  90   vars = utils.Foo()
  91   for k, v in zip(ks, vals):
  92     setattr(vars, k, v)
  93   logging.error('solver_vars: %s', vars)
  94   return vars

  96 def process_solver_str(solver_str):
  97   solver = utils.Foo(
  98       seed=0, learning_rate_decay=None, clip_gradient_norm=None, max_steps=None,
  99       initial_learning_rate=None, momentum=None, steps_per_decay=None,
 100       logdir=None, sync=False, adjust_lr_sync=True, wt_decay=0.0001,
 101       data_loss_wt=None, reg_loss_wt=None, freeze_conv=True, num_workers=1,
 102       task=0, ps_tasks=0, master='local', typ=None, momentum2=None,
 103       adam_eps=None)
 104 
 105   # Clobber with overrides from solver str.
 106   solver_vars = get_solver_vars(solver_str)
 107 
 108   solver.data_loss_wt          = float(solver_vars.dlw[3:].replace('x', '.'))
 109   solver.adam_eps              = float(solver_vars.adam_eps[4:].replace('x', '.').replace('n', '-'))
 110   solver.initial_learning_rate = float(solver_vars.init_lr[2:].replace('x', '.').replace('n', '-'))
 111   solver.reg_loss_wt           = float(solver_vars.rlw[3:].replace('x', '.'))
 112   solver.isd_k                 = float(solver_vars.isdk[4:].replace('x', '.'))
 113 
 114   long = solver_vars.long
 115   if long == 'long':
 116     solver.steps_per_decay = 40000
 117     solver.max_steps = 120000
 118   elif long == 'long2':
 119     solver.steps_per_decay = 80000
 120     solver.max_steps = 120000
 121   elif long == 'nolong' or long == 'nol':
 122     solver.steps_per_decay = 20000
 123     solver.max_steps = 60000
 124   else:
 125     logging.fatal('solver_vars.long should be long, long2, nolong or nol.')
 126     assert(False)
 127 
 128   clip = solver_vars.clip
 129   if clip == 'noclip' or clip == 'nocl':
 130     solver.clip_gradient_norm = 0
 131   elif clip[:4] == 'clip':
 132     solver.clip_gradient_norm = float(clip[4:].replace('x', '.'))
 133   else:
 134     logging.fatal('Unknown solver_vars.clip: %s', clip)
 135     assert(False)
 136 
 137   typ = solver_vars.typ
 138   if typ == 'adam':
 139     solver.typ = 'adam'
 140     solver.momentum = 0.9
 141     solver.momentum2 = 0.999
 142     solver.learning_rate_decay = 1.0
 143   elif typ == 'adam2':
 144     solver.typ = 'adam'
 145     solver.momentum = 0.9
 146     solver.momentum2 = 0.999
 147     solver.learning_rate_decay = 0.1
 148   elif typ == 'sgd':
 149     solver.typ = 'sgd'
 150     solver.momentum = 0.99
 151     solver.momentum2 = None
 152     solver.learning_rate_decay = 0.1
 153   else:
 154     logging.fatal('Unknown solver_vars.typ: %s', typ)
 155     assert(False)
 156 
 157   logging.error('solver: %s', solver)
 158   return solver

 160 def get_navtask_vars(navtask_str):
 161   if navtask_str == '': vals = []
 162   else: vals = navtask_str.split('_')
 163 
 164   ks_all = ['dataset_name', 'modality', 'task', 'history', 'max_dist',
 165             'num_steps', 'step_size', 'n_ori', 'aux_views', 'data_aug']
 166   ks = ks_all[:len(vals)]
 167 
 168   # All data or not.
 169   if len(vals) == 0: ks.append('dataset_name'); vals.append('sbpd')
 170   # modality
 171   if len(vals) == 1: ks.append('modality'); vals.append('rgb')
 172   # semantic task?
 173   if len(vals) == 2: ks.append('task'); vals.append('r2r')
 174   # number of history frames.
 175   if len(vals) == 3: ks.append('history'); vals.append('h0')
 176   # max steps
 177   if len(vals) == 4: ks.append('max_dist'); vals.append('32')
 178   # num steps
 179   if len(vals) == 5: ks.append('num_steps'); vals.append('40')
 180   # step size
 181   if len(vals) == 6: ks.append('step_size'); vals.append('8')
 182   # n_ori
 183   if len(vals) == 7: ks.append('n_ori'); vals.append('4')
 184   # Auxiliary views.
 185   if len(vals) == 8: ks.append('aux_views'); vals.append('nv0')
 186   # Normal data augmentation as opposed to structured data augmentation (if set
 187   # to straug.
 188   if len(vals) == 9: ks.append('data_aug'); vals.append('straug')
 189 
 190   assert(len(vals) == 10)
 191   for i in range(len(ks)):
 192     assert(ks[i] == ks_all[i])
 193 
 194   vars = utils.Foo()
 195   for k, v in zip(ks, vals):
 196     setattr(vars, k, v)
 197   logging.error('navtask_vars: %s', vals)
 198   return vars

 200 def process_navtask_str(navtask_str):
 201   navtask = nec.nav_env_base_config()
 202   
 203   # Clobber with overrides from strings.
 204   navtask_vars = get_navtask_vars(navtask_str)
 205 
 206   navtask.task_params.n_ori = int(navtask_vars.n_ori)
 207   navtask.task_params.max_dist = int(navtask_vars.max_dist)
 208   navtask.task_params.num_steps = int(navtask_vars.num_steps)
 209   navtask.task_params.step_size = int(navtask_vars.step_size)
 210   navtask.task_params.data_augment.delta_xy = int(navtask_vars.step_size)/2.
 211   n_aux_views_each = int(navtask_vars.aux_views[2])
 212   aux_delta_thetas = np.concatenate((np.arange(n_aux_views_each) + 1,
 213                                      -1 -np.arange(n_aux_views_each)))
 214   aux_delta_thetas = aux_delta_thetas*np.deg2rad(navtask.camera_param.fov)
 215   navtask.task_params.aux_delta_thetas = aux_delta_thetas
 216   
 217   if navtask_vars.data_aug == 'aug':
 218     navtask.task_params.data_augment.structured = False
 219   elif navtask_vars.data_aug == 'straug':
 220     navtask.task_params.data_augment.structured = True
 221   else:
 222     logging.fatal('Unknown navtask_vars.data_aug %s.', navtask_vars.data_aug)
 223     assert(False)
 224 
 225   navtask.task_params.num_history_frames = int(navtask_vars.history[1:])
 226   navtask.task_params.n_views = 1+navtask.task_params.num_history_frames
 227   
 228   navtask.task_params.goal_channels = int(navtask_vars.n_ori)
 229   
 230   if navtask_vars.task == 'hard': 
 231     navtask.task_params.type = 'rng_rejection_sampling_many'
 232     navtask.task_params.rejection_sampling_M = 2000
 233     navtask.task_params.min_dist = 10
 234   elif navtask_vars.task == 'r2r':
 235     navtask.task_params.type = 'room_to_room_many'
 236   elif navtask_vars.task == 'ST':
 237     # Semantic task at hand.
 238     navtask.task_params.goal_channels = \
 239         len(navtask.task_params.semantic_task.class_map_names)
 240     navtask.task_params.rel_goal_loc_dim = \
 241         len(navtask.task_params.semantic_task.class_map_names)
 242     navtask.task_params.type = 'to_nearest_obj_acc'
 243   else:
 244     logging.fatal('navtask_vars.task: should be hard or r2r, ST')
 245     assert(False)
 246   
 247   if navtask_vars.modality == 'rgb':
 248     navtask.camera_param.modalities = ['rgb']
 249     navtask.camera_param.img_channels = 3
 250   elif navtask_vars.modality == 'd':
 251     navtask.camera_param.modalities = ['depth']
 252     navtask.camera_param.img_channels = 2
 253   
 254   navtask.task_params.img_height   = navtask.camera_param.height
 255   navtask.task_params.img_width    = navtask.camera_param.width
 256   navtask.task_params.modalities   = navtask.camera_param.modalities
 257   navtask.task_params.img_channels = navtask.camera_param.img_channels
 258   navtask.task_params.img_fov      = navtask.camera_param.fov
 259   
 260   navtask.dataset = factory.get_dataset(navtask_vars.dataset_name)
 261   return navtask
