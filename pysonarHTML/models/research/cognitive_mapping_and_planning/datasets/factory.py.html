  31 def get_dataset(dataset_name):

  38 class Loader():
  39   def get_data_dir():
  40     pass
  41 
  42   def get_meta_data(self, file_name, data_dir=None):
  43     if data_dir is None:
  44       data_dir = self.get_data_dir()
  45     full_file_name = os.path.join(data_dir, 'meta', file_name)
  46     assert(fu.exists(full_file_name)), \
  47       '{:s} does not exist'.format(full_file_name)
  48     ext = os.path.splitext(full_file_name)[1]
  49     if ext == '.txt':
  50       ls = []
  51       with fu.fopen(full_file_name, 'r') as f:
  52         for l in f:
  53           ls.append(l.rstrip())
  54     elif ext == '.pkl':
  55       ls = utils.load_variables(full_file_name)
  56     return ls
  57 
  58   def load_building(self, name, data_dir=None):
  59     if data_dir is None:
  60       data_dir = self.get_data_dir()
  61     out = {}
  62     out['name'] = name
  63     out['data_dir'] = data_dir
  64     out['room_dimension_file'] = os.path.join(data_dir, 'room-dimension',
  65                                               name+'.pkl')
  66     out['class_map_folder'] = os.path.join(data_dir, 'class-maps')
  67     return out
  68 
  69   def load_building_meshes(self, building):
  70     dir_name = os.path.join(building['data_dir'], 'mesh', building['name'])
  71     mesh_file_name = glob.glob1(dir_name, '*.obj')[0]
  72     mesh_file_name_full = os.path.join(dir_name, mesh_file_name)
  73     logging.error('Loading building from obj file: %s', mesh_file_name_full)
  74     shape = renderer.Shape(mesh_file_name_full, load_materials=True, 
  75                            name_prefix=building['name']+'_')
  76     return [shape]
  77 
  78 class StanfordBuildingParserDataset(Loader):
  79   def __init__(self, ver):
  80     self.ver = ver
  81     self.data_dir = None
  82   
  83   def get_data_dir(self):
  84     if self.data_dir is None:
  85       self.data_dir = 'data/stanford_building_parser_dataset/'
  86     return self.data_dir
  87 
  88   def get_benchmark_sets(self):
  89     return self._get_benchmark_sets()
  90 
  91   def get_split(self, split_name):
  92     if self.ver == 'sbpd':
  93       return self._get_split(split_name)
  94     else:
  95       logging.fatal('Unknown version.')
  96 
  97   def _get_benchmark_sets(self):
  98     sets = ['train1', 'val', 'test']
  99     return sets
 100 
 101   def _get_split(self, split_name):
 102     train = ['area1', 'area5a', 'area5b', 'area6']
 103     train1 = ['area1']
 104     val = ['area3']
 105     test = ['area4']
 106 
 107     sets = {}
 108     sets['train'] = train
 109     sets['train1'] = train1
 110     sets['val'] = val
 111     sets['test'] = test
 112     sets['all'] = sorted(list(set(train + val + test)))
 113     return sets[split_name]
