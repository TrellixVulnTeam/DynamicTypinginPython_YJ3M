  35 def value_iteration_network(
  36     fr, num_iters, val_neurons, action_neurons, kernel_size, share_wts=False,
  37     name='vin', wt_decay=0.0001, activation_fn=None, shape_aware=False):
  38   """
  39   Constructs a Value Iteration Network, convolutions and max pooling across
  40   channels.
  41   Input:
  42     fr:             NxWxHxC
  43     val_neurons:    Number of channels for maintaining the value.
  44     action_neurons: Computes action_neurons * val_neurons at each iteration to
  45                     max pool over.
  46   Output:
  47     value image:  NxHxWx(val_neurons)
  48   """
  49   init_var = np.sqrt(2.0/(kernel_size**2)/(val_neurons*action_neurons))
  50   vals = []
  51   with tf.variable_scope(name) as varscope:
  52     if shape_aware == False:
  53       fr_shape = tf.unstack(tf.shape(fr))
  54       val_shape = tf.stack(fr_shape[:-1] + [val_neurons])
  55       val = tf.zeros(val_shape, name='val_init')
  56     else:
  57       val = tf.expand_dims(tf.zeros_like(fr[:,:,:,0]), dim=-1) * \
  58           tf.constant(0., dtype=tf.float32, shape=[1,1,1,val_neurons])
  59       val_shape = tf.shape(val)
  60     vals.append(val)
  61     for i in range(num_iters):
  62       if share_wts:
  63         # The first Value Iteration maybe special, so it can have its own
  64         # paramterss.
  65         scope = 'conv'
  66         if i == 0: scope = 'conv_0'
  67         if i > 1: varscope.reuse_variables()
  68       else:
  69         scope = 'conv_{:d}'.format(i)
  70       val = slim.conv2d(tf.concat([val, fr], 3, name='concat_{:d}'.format(i)),
  71                         num_outputs=action_neurons*val_neurons,
  72                         kernel_size=kernel_size, stride=1, activation_fn=activation_fn,
  73                         scope=scope, normalizer_fn=None,
  74                         weights_regularizer=slim.l2_regularizer(wt_decay),
  75                         weights_initializer=tf.random_normal_initializer(stddev=init_var),
  76                         biases_initializer=tf.zeros_initializer())
  77       val = tf.reshape(val, [-1, action_neurons*val_neurons, 1, 1],
  78                        name='re_{:d}'.format(i))
  79       val = slim.max_pool2d(val, kernel_size=[action_neurons,1],
  80                             stride=[action_neurons,1], padding='VALID',
  81                             scope='val_{:d}'.format(i))
  82       val = tf.reshape(val, val_shape, name='unre_{:d}'.format(i))
  83       vals.append(val)
  84   return val, vals

  87 def rotate_preds(loc_on_map, relative_theta, map_size, preds,
  88                  output_valid_mask):
  89   with tf.name_scope('rotate'):
  90     flow_op = tf_utils.get_flow(loc_on_map, relative_theta, map_size=map_size)
  91     if type(preds) != list:
  92       rotated_preds, valid_mask_warps = tf_utils.dense_resample(preds, flow_op,
  93                                                                 output_valid_mask)
  94     else:
  95       rotated_preds = [] ;valid_mask_warps = []
  96       for pred in preds:
  97         rotated_pred, valid_mask_warp = tf_utils.dense_resample(pred, flow_op,
  98                                                                 output_valid_mask)
  99         rotated_preds.append(rotated_pred)
 100         valid_mask_warps.append(valid_mask_warp)
 101   return rotated_preds, valid_mask_warps

 103 def get_visual_frustum(map_size, shape_like, expand_dims=[0,0]):
 104   with tf.name_scope('visual_frustum'):
 105     l = np.tril(np.ones(map_size)) ;l = l + l[:,::-1]
 106     l = (l == 2).astype(np.float32)
 107     for e in expand_dims:
 108       l = np.expand_dims(l, axis=e)
 109     confs_probs = tf.constant(l, dtype=tf.float32)
 110     confs_probs = tf.ones_like(shape_like, dtype=tf.float32) * confs_probs
 111   return confs_probs

 113 def deconv(x, is_training, wt_decay, neurons, strides, layers_per_block,
 114             kernel_size, conv_fn, name, offset=0):
 115   """Generates a up sampling network with residual connections. 
 116   """
 117   batch_norm_param = {'center': True, 'scale': True,
 118                       'activation_fn': tf.nn.relu,
 119                       'is_training': is_training}
 120   outs = []
 121   for i, (neuron, stride) in enumerate(zip(neurons, strides)):
 122     for s in range(layers_per_block):
 123       scope = '{:s}_{:d}_{:d}'.format(name, i+1+offset,s+1)
 124       x = custom_residual_block(x, neuron, kernel_size, stride, scope,
 125                                 is_training, wt_decay, use_residual=True,
 126                                 residual_stride_conv=True, conv_fn=conv_fn,
 127                                 batch_norm_param=batch_norm_param)
 128       stride = 1
 129     outs.append((x,True))
 130   return x, outs

 132 def fr_v2(x, output_neurons, inside_neurons, is_training, name='fr',
 133           wt_decay=0.0001, stride=1, updates_collections=tf.GraphKeys.UPDATE_OPS):
 134   """Performs fusion of information between the map and the reward map.
 135   Inputs
 136     x:   NxHxWxC1
 137 
 138   Outputs
 139     fr map:     NxHxWx(output_neurons)
 140   """
 141   if type(stride) != list:
 142     stride = [stride]
 143   with slim.arg_scope(resnet_v2.resnet_utils.resnet_arg_scope(
 144       is_training=is_training, weight_decay=wt_decay)):
 145     with slim.arg_scope([slim.batch_norm], updates_collections=updates_collections) as arg_sc:
 146       # Change the updates_collections for the conv normalizer_params to None
 147       for i in range(len(arg_sc.keys())):
 148         if 'convolution' in arg_sc.keys()[i]:
 149           arg_sc.values()[i]['normalizer_params']['updates_collections'] = updates_collections
 150       with slim.arg_scope(arg_sc):
 151         bottleneck = resnet_v2.bottleneck
 152         blocks = []
 153         for i, s in enumerate(stride):
 154           b = resnet_v2.resnet_utils.Block(
 155               'block{:d}'.format(i + 1), bottleneck, [{
 156                   'depth': output_neurons,
 157                   'depth_bottleneck': inside_neurons,
 158                   'stride': stride[i]
 159               }])
 160           blocks.append(b)
 161         x, outs = resnet_v2.resnet_v2(x, blocks, num_classes=None, global_pool=False,
 162                                      output_stride=None, include_root_block=False,
 163                                      reuse=False, scope=name)
 164   return x, outs
