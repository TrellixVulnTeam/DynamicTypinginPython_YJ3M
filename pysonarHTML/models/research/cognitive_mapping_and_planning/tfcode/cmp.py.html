  66 def _inputs(problem):
  67   # Set up inputs.
  68   with tf.name_scope('inputs'):
  69     inputs = []
  70     inputs.append(('orig_maps', tf.float32, 
  71                    (problem.batch_size, 1, None, None, 1)))
  72     inputs.append(('goal_loc', tf.float32, 
  73                    (problem.batch_size, problem.num_goals, 2)))
  74     common_input_data, _ = tf_utils.setup_inputs(inputs)
  75 
  76     inputs = []
  77     if problem.input_type == 'vision':
  78       # Multiple images from an array of cameras.
  79       inputs.append(('imgs', tf.float32, 
  80                      (problem.batch_size, None, len(problem.aux_delta_thetas)+1,
  81                       problem.img_height, problem.img_width,
  82                       problem.img_channels)))
  83     elif problem.input_type == 'analytical_counts':
  84       for i in range(len(problem.map_crop_sizes)):
  85         inputs.append(('analytical_counts_{:d}'.format(i), tf.float32, 
  86                       (problem.batch_size, None, problem.map_crop_sizes[i],
  87                        problem.map_crop_sizes[i], problem.map_channels)))
  88 
  89     if problem.outputs.readout_maps: 
  90       for i in range(len(problem.readout_maps_crop_sizes)):
  91         inputs.append(('readout_maps_{:d}'.format(i), tf.float32, 
  92                       (problem.batch_size, None,
  93                        problem.readout_maps_crop_sizes[i],
  94                        problem.readout_maps_crop_sizes[i],
  95                        problem.readout_maps_channels)))
  96 
  97     for i in range(len(problem.map_crop_sizes)):
  98       inputs.append(('ego_goal_imgs_{:d}'.format(i), tf.float32, 
  99                     (problem.batch_size, None, problem.map_crop_sizes[i],
 100                      problem.map_crop_sizes[i], problem.goal_channels)))
 101       for s in ['sum_num', 'sum_denom', 'max_denom']:
 102         inputs.append(('running_'+s+'_{:d}'.format(i), tf.float32,
 103                        (problem.batch_size, 1, problem.map_crop_sizes[i],
 104                         problem.map_crop_sizes[i], problem.map_channels)))
 105 
 106     inputs.append(('incremental_locs', tf.float32, 
 107                    (problem.batch_size, None, 2)))
 108     inputs.append(('incremental_thetas', tf.float32, 
 109                    (problem.batch_size, None, 1)))
 110     inputs.append(('step_number', tf.int32, (1, None, 1)))
 111     inputs.append(('node_ids', tf.int32, (problem.batch_size, None,
 112                                           problem.node_ids_dim)))
 113     inputs.append(('perturbs', tf.float32, (problem.batch_size, None,
 114                                             problem.perturbs_dim)))
 115     
 116     # For plotting result plots
 117     inputs.append(('loc_on_map', tf.float32, (problem.batch_size, None, 2)))
 118     inputs.append(('gt_dist_to_goal', tf.float32, (problem.batch_size, None, 1)))
 119 
 120     step_input_data, _ = tf_utils.setup_inputs(inputs)
 121 
 122     inputs = []
 123     inputs.append(('action', tf.int32, (problem.batch_size, None, problem.num_actions)))
 124     train_data, _ = tf_utils.setup_inputs(inputs)
 125     train_data.update(step_input_data)
 126     train_data.update(common_input_data)
 127   return common_input_data, step_input_data, train_data

 129 def readout_general(multi_scale_belief, num_neurons, strides, layers_per_block,
 130                     kernel_size, batch_norm_is_training_op, wt_decay):
 131   multi_scale_belief = tf.stop_gradient(multi_scale_belief)
 132   with tf.variable_scope('readout_maps_deconv'):
 133     x, outs = deconv(multi_scale_belief, batch_norm_is_training_op,
 134                      wt_decay=wt_decay, neurons=num_neurons, strides=strides,
 135                      layers_per_block=layers_per_block, kernel_size=kernel_size,
 136                      conv_fn=slim.conv2d_transpose, offset=0,
 137                      name='readout_maps_deconv')
 138     probs = tf.sigmoid(x)
 139   return x, probs

 142 def running_combine(fss_logits, confs_probs, incremental_locs,
 143                     incremental_thetas, previous_sum_num, previous_sum_denom,
 144                     previous_max_denom, map_size, num_steps):
 145   # fss_logits is B x N x H x W x C
 146   # confs_logits is B x N x H x W x C
 147   # incremental_locs is B x N x 2
 148   # incremental_thetas is B x N x 1
 149   # previous_sum_num etc is B x 1 x H x W x C
 150 
 151   with tf.name_scope('combine_{:d}'.format(num_steps)):
 152     running_sum_nums_ = []; running_sum_denoms_ = [];
 153     running_max_denoms_ = [];
 154 
 155     fss_logits_ = tf.unstack(fss_logits, axis=1, num=num_steps)
 156     confs_probs_ = tf.unstack(confs_probs, axis=1, num=num_steps)
 157     incremental_locs_ = tf.unstack(incremental_locs, axis=1, num=num_steps)
 158     incremental_thetas_ = tf.unstack(incremental_thetas, axis=1, num=num_steps)
 159     running_sum_num = tf.unstack(previous_sum_num, axis=1, num=1)[0]
 160     running_sum_denom = tf.unstack(previous_sum_denom, axis=1, num=1)[0]
 161     running_max_denom = tf.unstack(previous_max_denom, axis=1, num=1)[0]
 162 
 163     for i in range(num_steps):
 164       # Rotate the previous running_num and running_denom
 165       running_sum_num, running_sum_denom, running_max_denom = rotate_preds(
 166           incremental_locs_[i], incremental_thetas_[i], map_size,
 167           [running_sum_num, running_sum_denom, running_max_denom],
 168           output_valid_mask=False)[0]
 169       # print i, num_steps, running_sum_num.get_shape().as_list()
 170       running_sum_num = running_sum_num + fss_logits_[i] * confs_probs_[i]
 171       running_sum_denom = running_sum_denom + confs_probs_[i]
 172       running_max_denom = tf.maximum(running_max_denom, confs_probs_[i])
 173       running_sum_nums_.append(running_sum_num)
 174       running_sum_denoms_.append(running_sum_denom)
 175       running_max_denoms_.append(running_max_denom)
 176 
 177     running_sum_nums = tf.stack(running_sum_nums_, axis=1)
 178     running_sum_denoms = tf.stack(running_sum_denoms_, axis=1)
 179     running_max_denoms = tf.stack(running_max_denoms_, axis=1)
 180     return running_sum_nums, running_sum_denoms, running_max_denoms

 182 def get_map_from_images(imgs, mapper_arch, task_params, freeze_conv, wt_decay,
 183                         is_training, batch_norm_is_training_op, num_maps,
 184                         split_maps=True):
 185   # Hit image with a resnet.
 186   n_views = len(task_params.aux_delta_thetas) + 1
 187   out = utils.Foo()
 188 
 189   images_reshaped = tf.reshape(imgs, 
 190       shape=[-1, task_params.img_height,
 191              task_params.img_width,
 192              task_params.img_channels], name='re_image')
 193 
 194   x, out.vars_to_restore = get_repr_from_image(
 195       images_reshaped, task_params.modalities, task_params.data_augment,
 196       mapper_arch.encoder, freeze_conv, wt_decay, is_training)
 197 
 198   # Reshape into nice things so that these can be accumulated over time steps
 199   # for faster backprop.
 200   sh_before = x.get_shape().as_list()
 201   out.encoder_output = tf.reshape(x, shape=[task_params.batch_size, -1, n_views] + sh_before[1:])
 202   x = tf.reshape(out.encoder_output, shape=[-1] + sh_before[1:])
 203 
 204   # Add a layer to reduce dimensions for a fc layer.
 205   if mapper_arch.dim_reduce_neurons > 0:
 206     ks = 1; neurons = mapper_arch.dim_reduce_neurons;
 207     init_var = np.sqrt(2.0/(ks**2)/neurons)
 208     batch_norm_param = mapper_arch.batch_norm_param
 209     batch_norm_param['is_training'] = batch_norm_is_training_op
 210     out.conv_feat = slim.conv2d(x, neurons, kernel_size=ks, stride=1,
 211                     normalizer_fn=slim.batch_norm, normalizer_params=batch_norm_param,
 212                     padding='SAME', scope='dim_reduce',
 213                     weights_regularizer=slim.l2_regularizer(wt_decay),
 214                     weights_initializer=tf.random_normal_initializer(stddev=init_var))
 215     reshape_conv_feat = slim.flatten(out.conv_feat)
 216     sh = reshape_conv_feat.get_shape().as_list()
 217     out.reshape_conv_feat = tf.reshape(reshape_conv_feat, shape=[-1, sh[1]*n_views])
 218 
 219   with tf.variable_scope('fc'):
 220     # Fully connected layers to compute the representation in top-view space.
 221     fc_batch_norm_param = {'center': True, 'scale': True, 
 222                            'activation_fn':tf.nn.relu,
 223                            'is_training': batch_norm_is_training_op}
 224     f = out.reshape_conv_feat
 225     out_neurons = (mapper_arch.fc_out_size**2)*mapper_arch.fc_out_neurons
 226     neurons = mapper_arch.fc_neurons + [out_neurons]
 227     f, _ = tf_utils.fc_network(f, neurons=neurons, wt_decay=wt_decay,
 228                                name='fc', offset=0,
 229                                batch_norm_param=fc_batch_norm_param,
 230                                is_training=is_training,
 231                                dropout_ratio=mapper_arch.fc_dropout)
 232     f = tf.reshape(f, shape=[-1, mapper_arch.fc_out_size,
 233                              mapper_arch.fc_out_size,
 234                              mapper_arch.fc_out_neurons], name='re_fc')
 235 
 236   # Use pool5 to predict the free space map via deconv layers.
 237   with tf.variable_scope('deconv'):
 238     x, outs = deconv(f, batch_norm_is_training_op, wt_decay=wt_decay,
 239                      neurons=mapper_arch.deconv_neurons,
 240                      strides=mapper_arch.deconv_strides,
 241                      layers_per_block=mapper_arch.deconv_layers_per_block,
 242                      kernel_size=mapper_arch.deconv_kernel_size,
 243                      conv_fn=slim.conv2d_transpose, offset=0, name='deconv')
 244 
 245   # Reshape x the right way.
 246   sh = x.get_shape().as_list()
 247   x = tf.reshape(x, shape=[task_params.batch_size, -1] + sh[1:])
 248   out.deconv_output = x
 249 
 250   # Separate out the map and the confidence predictions, pass the confidence
 251   # through a sigmoid.
 252   if split_maps:
 253     with tf.name_scope('split'):
 254       out_all = tf.split(value=x, axis=4, num_or_size_splits=2*num_maps)
 255       out.fss_logits = out_all[:num_maps]
 256       out.confs_logits = out_all[num_maps:]
 257     with tf.name_scope('sigmoid'):
 258       out.confs_probs = [tf.nn.sigmoid(x) for x in out.confs_logits]
 259   return out

 261 def setup_to_run(m, args, is_training, batch_norm_is_training, summary_mode):
 262   assert(args.arch.multi_scale), 'removed support for old single scale code.'
 263   # Set up the model.
 264   tf.set_random_seed(args.solver.seed)
 265   task_params = args.navtask.task_params
 266 
 267   batch_norm_is_training_op = \
 268       tf.placeholder_with_default(batch_norm_is_training, shape=[],
 269                                   name='batch_norm_is_training_op') 
 270 
 271   # Setup the inputs
 272   m.input_tensors = {}
 273   m.train_ops = {}
 274   m.input_tensors['common'], m.input_tensors['step'], m.input_tensors['train'] = \
 275       _inputs(task_params)
 276 
 277   m.init_fn = None
 278 
 279   if task_params.input_type == 'vision':
 280     m.vision_ops = get_map_from_images(
 281         m.input_tensors['step']['imgs'], args.mapper_arch,
 282         task_params, args.solver.freeze_conv,
 283         args.solver.wt_decay, is_training, batch_norm_is_training_op,
 284         num_maps=len(task_params.map_crop_sizes))
 285 
 286     # Load variables from snapshot if needed.
 287     if args.solver.pretrained_path is not None:
 288       m.init_fn = slim.assign_from_checkpoint_fn(args.solver.pretrained_path,
 289                                                  m.vision_ops.vars_to_restore)
 290 
 291     # Set up caching of vision features if needed.
 292     if args.solver.freeze_conv:
 293       m.train_ops['step_data_cache'] = [m.vision_ops.encoder_output]
 294     else:
 295       m.train_ops['step_data_cache'] = []
 296 
 297     # Set up blobs that are needed for the computation in rest of the graph.
 298     m.ego_map_ops = m.vision_ops.fss_logits
 299     m.coverage_ops = m.vision_ops.confs_probs
 300     
 301     # Zero pad these to make them same size as what the planner expects.
 302     for i in range(len(m.ego_map_ops)):
 303       if args.mapper_arch.pad_map_with_zeros_each[i] > 0:
 304         paddings = np.zeros((5,2), dtype=np.int32)
 305         paddings[2:4,:] = args.mapper_arch.pad_map_with_zeros_each[i]
 306         paddings_op = tf.constant(paddings, dtype=tf.int32)
 307         m.ego_map_ops[i] = tf.pad(m.ego_map_ops[i], paddings=paddings_op)
 308         m.coverage_ops[i] = tf.pad(m.coverage_ops[i], paddings=paddings_op)
 309   
 310   elif task_params.input_type == 'analytical_counts':
 311     m.ego_map_ops = []; m.coverage_ops = []
 312     for i in range(len(task_params.map_crop_sizes)):
 313       ego_map_op = m.input_tensors['step']['analytical_counts_{:d}'.format(i)]
 314       coverage_op = tf.cast(tf.greater_equal(
 315           tf.reduce_max(ego_map_op, reduction_indices=[4],
 316                         keep_dims=True), 1), tf.float32)
 317       coverage_op = tf.ones_like(ego_map_op) * coverage_op
 318       m.ego_map_ops.append(ego_map_op)
 319       m.coverage_ops.append(coverage_op)
 320       m.train_ops['step_data_cache'] = []
 321   
 322   num_steps = task_params.num_steps
 323   num_goals = task_params.num_goals
 324 
 325   map_crop_size_ops = []
 326   for map_crop_size in task_params.map_crop_sizes:
 327     map_crop_size_ops.append(tf.constant(map_crop_size, dtype=tf.int32, shape=(2,)))
 328 
 329   with tf.name_scope('check_size'):
 330     is_single_step = tf.equal(tf.unstack(tf.shape(m.ego_map_ops[0]), num=5)[1], 1)
 331 
 332   fr_ops = []; value_ops = [];
 333   fr_intermediate_ops = []; value_intermediate_ops = [];
 334   crop_value_ops = [];
 335   resize_crop_value_ops = [];
 336   confs = []; occupancys = [];
 337 
 338   previous_value_op = None
 339   updated_state = []; state_names = [];
 340 
 341   for i in range(len(task_params.map_crop_sizes)):
 342     map_crop_size = task_params.map_crop_sizes[i]
 343     with tf.variable_scope('scale_{:d}'.format(i)): 
 344       # Accumulate the map.
 345       fn = lambda ns: running_combine(
 346              m.ego_map_ops[i],
 347              m.coverage_ops[i],
 348              m.input_tensors['step']['incremental_locs'] * task_params.map_scales[i],
 349              m.input_tensors['step']['incremental_thetas'],
 350              m.input_tensors['step']['running_sum_num_{:d}'.format(i)],
 351              m.input_tensors['step']['running_sum_denom_{:d}'.format(i)],
 352              m.input_tensors['step']['running_max_denom_{:d}'.format(i)],
 353              map_crop_size, ns)
 354 
 355       running_sum_num, running_sum_denom, running_max_denom = \
 356           tf.cond(is_single_step, lambda: fn(1), lambda: fn(num_steps*num_goals))
 357       updated_state += [running_sum_num, running_sum_denom, running_max_denom]
 358       state_names += ['running_sum_num_{:d}'.format(i),
 359                       'running_sum_denom_{:d}'.format(i),
 360                       'running_max_denom_{:d}'.format(i)]
 361 
 362       # Concat the accumulated map and goal
 363       occupancy = running_sum_num / tf.maximum(running_sum_denom, 0.001)
 364       conf = running_max_denom
 365       # print occupancy.get_shape().as_list()
 366 
 367       # Concat occupancy, how much occupied and goal.
 368       with tf.name_scope('concat'):
 369         sh = [-1, map_crop_size, map_crop_size, task_params.map_channels]
 370         occupancy = tf.reshape(occupancy, shape=sh)
 371         conf = tf.reshape(conf, shape=sh)
 372 
 373         sh = [-1, map_crop_size, map_crop_size, task_params.goal_channels]
 374         goal = tf.reshape(m.input_tensors['step']['ego_goal_imgs_{:d}'.format(i)], shape=sh)
 375         to_concat = [occupancy, conf, goal]
 376 
 377         if previous_value_op is not None:
 378           to_concat.append(previous_value_op)
 379 
 380         x = tf.concat(to_concat, 3)
 381 
 382       # Pass the map, previous rewards and the goal through a few convolutional
 383       # layers to get fR.
 384       fr_op, fr_intermediate_op = fr_v2(
 385          x, output_neurons=args.arch.fr_neurons,
 386          inside_neurons=args.arch.fr_inside_neurons,
 387          is_training=batch_norm_is_training_op, name='fr',
 388          wt_decay=args.solver.wt_decay, stride=args.arch.fr_stride)
 389 
 390       # Do Value Iteration on the fR
 391       if args.arch.vin_num_iters > 0:
 392         value_op, value_intermediate_op = value_iteration_network(
 393             fr_op, num_iters=args.arch.vin_num_iters,
 394             val_neurons=args.arch.vin_val_neurons,
 395             action_neurons=args.arch.vin_action_neurons,
 396             kernel_size=args.arch.vin_ks, share_wts=args.arch.vin_share_wts,
 397             name='vin', wt_decay=args.solver.wt_decay)
 398       else:
 399         value_op = fr_op
 400         value_intermediate_op = []
 401 
 402       # Crop out and upsample the previous value map.
 403       remove = args.arch.crop_remove_each
 404       if remove > 0:
 405         crop_value_op = value_op[:, remove:-remove, remove:-remove,:]
 406       else:
 407         crop_value_op = value_op
 408       crop_value_op = tf.reshape(crop_value_op, shape=[-1, args.arch.value_crop_size,
 409                                                        args.arch.value_crop_size,
 410                                                        args.arch.vin_val_neurons])
 411       if i < len(task_params.map_crop_sizes)-1:
 412         # Reshape it to shape of the next scale.
 413         previous_value_op = tf.image.resize_bilinear(crop_value_op,
 414                                                      map_crop_size_ops[i+1],
 415                                                      align_corners=True)
 416         resize_crop_value_ops.append(previous_value_op)
 417       
 418       occupancys.append(occupancy)
 419       confs.append(conf)
 420       value_ops.append(value_op)
 421       crop_value_ops.append(crop_value_op)
 422       fr_ops.append(fr_op)
 423       fr_intermediate_ops.append(fr_intermediate_op)
 424   
 425   m.value_ops = value_ops
 426   m.value_intermediate_ops = value_intermediate_ops
 427   m.fr_ops = fr_ops
 428   m.fr_intermediate_ops = fr_intermediate_ops
 429   m.final_value_op = crop_value_op
 430   m.crop_value_ops = crop_value_ops
 431   m.resize_crop_value_ops = resize_crop_value_ops
 432   m.confs = confs
 433   m.occupancys = occupancys
 434 
 435   sh = [-1, args.arch.vin_val_neurons*((args.arch.value_crop_size)**2)]
 436   m.value_features_op = tf.reshape(m.final_value_op, sh, name='reshape_value_op')
 437   
 438   # Determine what action to take.
 439   with tf.variable_scope('action_pred'):
 440     batch_norm_param = args.arch.pred_batch_norm_param
 441     if batch_norm_param is not None:
 442       batch_norm_param['is_training'] = batch_norm_is_training_op
 443     m.action_logits_op, _ = tf_utils.fc_network(
 444         m.value_features_op, neurons=args.arch.pred_neurons,
 445         wt_decay=args.solver.wt_decay, name='pred', offset=0,
 446         num_pred=task_params.num_actions,
 447         batch_norm_param=batch_norm_param) 
 448     m.action_prob_op = tf.nn.softmax(m.action_logits_op)
 449 
 450   init_state = tf.constant(0., dtype=tf.float32, shape=[
 451       task_params.batch_size, 1, map_crop_size, map_crop_size,
 452       task_params.map_channels])
 453 
 454   m.train_ops['state_names'] = state_names
 455   m.train_ops['updated_state'] = updated_state
 456   m.train_ops['init_state'] = [init_state for _ in updated_state]
 457 
 458   m.train_ops['step'] = m.action_prob_op
 459   m.train_ops['common'] = [m.input_tensors['common']['orig_maps'],
 460                            m.input_tensors['common']['goal_loc']]
 461   m.train_ops['batch_norm_is_training_op'] = batch_norm_is_training_op
 462   m.loss_ops = []; m.loss_ops_names = [];
 463 
 464   if args.arch.readout_maps:
 465     with tf.name_scope('readout_maps'):
 466       all_occupancys = tf.concat(m.occupancys + m.confs, 3)

 468           all_occupancys, num_neurons=args.arch.rom_arch.num_neurons,
 469           strides=args.arch.rom_arch.strides, 
 470           layers_per_block=args.arch.rom_arch.layers_per_block, 
 471           kernel_size=args.arch.rom_arch.kernel_size,
 472           batch_norm_is_training_op=batch_norm_is_training_op,
 473           wt_decay=args.solver.wt_decay)
 474 
 475       gt_ego_maps = [m.input_tensors['step']['readout_maps_{:d}'.format(i)]
 476                      for i in range(len(task_params.readout_maps_crop_sizes))]
 477       m.readout_maps_gt = tf.concat(gt_ego_maps, 4)
 478       gt_shape = tf.shape(m.readout_maps_gt)
 479       m.readout_maps_logits = tf.reshape(readout_maps, gt_shape)
 480       m.readout_maps_probs = tf.reshape(probs, gt_shape)
 481 
 482       # Add a loss op
 483       m.readout_maps_loss_op = tf.losses.sigmoid_cross_entropy(
 484           tf.reshape(m.readout_maps_gt, [-1, len(task_params.readout_maps_crop_sizes)]), 
 485           tf.reshape(readout_maps, [-1, len(task_params.readout_maps_crop_sizes)]),
 486           scope='loss')
 487       m.readout_maps_loss_op = 10.*m.readout_maps_loss_op
 488 
 489   ewma_decay = 0.99 if is_training else 0.0
 490   weight = tf.ones_like(m.input_tensors['train']['action'], dtype=tf.float32,
 491                         name='weight')
 492   m.reg_loss_op, m.data_loss_op, m.total_loss_op, m.acc_ops = \
 493     compute_losses_multi_or(m.action_logits_op,
 494                             m.input_tensors['train']['action'], weights=weight,
 495                             num_actions=task_params.num_actions,
 496                             data_loss_wt=args.solver.data_loss_wt,
 497                             reg_loss_wt=args.solver.reg_loss_wt,
 498                             ewma_decay=ewma_decay)
 499   
 500   if args.arch.readout_maps:
 501     m.total_loss_op = m.total_loss_op + m.readout_maps_loss_op
 502     m.loss_ops += [m.readout_maps_loss_op]
 503     m.loss_ops_names += ['readout_maps_loss']
 504 
 505   m.loss_ops += [m.reg_loss_op, m.data_loss_op, m.total_loss_op]
 506   m.loss_ops_names += ['reg_loss', 'data_loss', 'total_loss']
 507 
 508   if args.solver.freeze_conv:
 509     vars_to_optimize = list(set(tf.trainable_variables()) -
 510                             set(m.vision_ops.vars_to_restore))
 511   else:
 512     vars_to_optimize = None
 513 
 514   m.lr_op, m.global_step_op, m.train_op, m.should_stop_op, m.optimizer, \
 515   m.sync_optimizer = tf_utils.setup_training(
 516       m.total_loss_op, 
 517       args.solver.initial_learning_rate, 
 518       args.solver.steps_per_decay,
 519       args.solver.learning_rate_decay, 
 520       args.solver.momentum,
 521       args.solver.max_steps, 
 522       args.solver.sync, 
 523       args.solver.adjust_lr_sync,
 524       args.solver.num_workers, 
 525       args.solver.task,
 526       vars_to_optimize=vars_to_optimize,
 527       clip_gradient_norm=args.solver.clip_gradient_norm,
 528       typ=args.solver.typ, momentum2=args.solver.momentum2,
 529       adam_eps=args.solver.adam_eps)
 530 
 531   if args.arch.sample_gt_prob_type == 'inverse_sigmoid_decay':
 532     m.sample_gt_prob_op = tf_utils.inverse_sigmoid_decay(args.arch.isd_k,
 533                                                          m.global_step_op)
 534   elif args.arch.sample_gt_prob_type == 'zero':
 535     m.sample_gt_prob_op = tf.constant(-1.0, dtype=tf.float32)
 536 
 537   elif args.arch.sample_gt_prob_type.split('_')[0] == 'step':
 538     step = int(args.arch.sample_gt_prob_type.split('_')[1])
 539     m.sample_gt_prob_op = tf_utils.step_gt_prob(
 540         step, m.input_tensors['step']['step_number'][0,0,0])
 541 
 542   m.sample_action_type = args.arch.action_sample_type
 543   m.sample_action_combine_type = args.arch.action_sample_combine_type
 544 
 545   m.summary_ops = {
 546       summary_mode: _add_summaries(m, args, summary_mode,
 547                                    args.summary.arop_full_summary_iters)}
 548 
 549   m.init_op = tf.group(tf.global_variables_initializer(),
 550                        tf.local_variables_initializer())
 551   m.saver_op = tf.train.Saver(keep_checkpoint_every_n_hours=4,
 552                               write_version=tf.train.SaverDef.V2)
 553   return m
