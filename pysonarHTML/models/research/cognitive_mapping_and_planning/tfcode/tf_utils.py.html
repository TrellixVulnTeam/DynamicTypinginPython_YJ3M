  36 def custom_residual_block(x, neurons, kernel_size, stride, name, is_training,
  37                           wt_decay=0.0001, use_residual=True,
  38                           residual_stride_conv=True, conv_fn=slim.conv2d,
  39                           batch_norm_param=None):
  40   
  41   # batch norm x and relu
  42   init_var = np.sqrt(2.0/(kernel_size**2)/neurons)
  43   with arg_scope([conv_fn], 
  44                  weights_regularizer=slim.l2_regularizer(wt_decay),
  45                  weights_initializer=tf.random_normal_initializer(stddev=init_var),
  46                  biases_initializer=tf.zeros_initializer()): 
  47     
  48     if batch_norm_param is None:
  49       batch_norm_param = {'center': True, 'scale': False, 
  50                           'activation_fn':tf.nn.relu, 
  51                           'is_training': is_training}
  52     
  53     y = slim.batch_norm(x, scope=name+'_bn', **batch_norm_param)
  54 
  55     y = conv_fn(y, num_outputs=neurons, kernel_size=kernel_size, stride=stride,
  56                 activation_fn=None, scope=name+'_1',
  57                 normalizer_fn=slim.batch_norm,
  58                 normalizer_params=batch_norm_param)
  59     
  60     y = conv_fn(y, num_outputs=neurons, kernel_size=kernel_size,
  61                     stride=1, activation_fn=None, scope=name+'_2')
  62 
  63     if use_residual:
  64       if stride != 1 or x.get_shape().as_list()[-1] != neurons:
  65         batch_norm_param_ = dict(batch_norm_param)
  66         batch_norm_param_['activation_fn'] = None
  67         x = conv_fn(x, num_outputs=neurons, kernel_size=1,
  68                         stride=stride if residual_stride_conv else 1,
  69                         activation_fn=None, scope=name+'_0_1x1',
  70                         normalizer_fn=slim.batch_norm,
  71                         normalizer_params=batch_norm_param_)
  72         if not residual_stride_conv:
  73           x = slim.avg_pool2d(x, 1, stride=stride, scope=name+'_0_avg')
  74   
  75       y = tf.add(x, y, name=name+'_add')
  76     
  77     return y

  79 def step_gt_prob(step, step_number_op):
  80   # Change samping probability from 1 to -1 at step steps.
  81   with tf.name_scope('step_gt_prob'):
  82     out = tf.cond(tf.less(step_number_op, step),
  83             lambda: tf.constant(1.), lambda: tf.constant(-1.))
  84     return out

  86 def inverse_sigmoid_decay(k, global_step_op):
  87   with tf.name_scope('inverse_sigmoid_decay'):
  88     k = tf.constant(k, dtype=tf.float32)
  89     tmp = k*tf.exp(-tf.cast(global_step_op, tf.float32)/k)
  90     tmp = tmp / (1. + tmp)
  91   return tmp

  93 def dense_resample(im, flow_im, output_valid_mask, name='dense_resample'):
  94   """ Resample reward at particular locations.
  95   Args:
  96     im:      ...xHxWxC matrix to sample from.
  97     flow_im: ...xHxWx2 matrix, samples the image using absolute offsets as given
  98              by the flow_im.
  99   """
 100   with tf.name_scope(name):
 101     valid_mask = None
 102     
 103     x, y = tf.unstack(flow_im, axis=-1)
 104     x = tf.cast(tf.reshape(x, [-1]), tf.float32)
 105     y = tf.cast(tf.reshape(y, [-1]), tf.float32)
 106 
 107     # constants
 108     shape = tf.unstack(tf.shape(im))
 109     channels = shape[-1]
 110     width = shape[-2]
 111     height = shape[-3]
 112     num_batch = tf.cast(tf.reduce_prod(tf.stack(shape[:-3])), 'int32')
 113     zero = tf.constant(0, dtype=tf.int32)
 114 
 115     # Round up and down.
 116     x0 = tf.cast(tf.floor(x), 'int32'); x1 = x0 + 1;
 117     y0 = tf.cast(tf.floor(y), 'int32'); y1 = y0 + 1;
 118     
 119     if output_valid_mask:
 120       valid_mask = tf.logical_and(
 121           tf.logical_and(tf.less_equal(x, tf.cast(width, tf.float32)-1.), tf.greater_equal(x, 0.)),
 122           tf.logical_and(tf.less_equal(y, tf.cast(height, tf.float32)-1.), tf.greater_equal(y, 0.)))
 123       valid_mask = tf.reshape(valid_mask, shape=shape[:-1] + [1])
 124   
 125     x0 = tf.clip_by_value(x0, zero, width-1)
 126     x1 = tf.clip_by_value(x1, zero, width-1)
 127     y0 = tf.clip_by_value(y0, zero, height-1)
 128     y1 = tf.clip_by_value(y1, zero, height-1)
 129 
 130     dim2 = width; dim1 = width * height;
 131 
 132     # Create base index
 133     base = tf.reshape(tf.range(num_batch) * dim1, shape=[-1,1])
 134     base = tf.reshape(tf.tile(base, [1, height*width]), shape=[-1])
 135 
 136     base_y0 = base + y0 * dim2
 137     base_y1 = base + y1 * dim2
 138     idx_a = base_y0 + x0
 139     idx_b = base_y1 + x0
 140     idx_c = base_y0 + x1
 141     idx_d = base_y1 + x1
 142 
 143     # use indices to lookup pixels in the flat image and restore channels dim
 144     sh = tf.stack([tf.constant(-1,dtype=tf.int32), channels])
 145     im_flat = tf.cast(tf.reshape(im, sh), dtype=tf.float32)
 146     pixel_a = tf.gather(im_flat, idx_a)
 147     pixel_b = tf.gather(im_flat, idx_b)
 148     pixel_c = tf.gather(im_flat, idx_c)
 149     pixel_d = tf.gather(im_flat, idx_d)
 150 
 151     # and finally calculate interpolated values
 152     x1_f = tf.to_float(x1)
 153     y1_f = tf.to_float(y1)
 154 
 155     wa = tf.expand_dims(((x1_f - x) * (y1_f - y)), 1)
 156     wb = tf.expand_dims((x1_f - x) * (1.0 - (y1_f - y)), 1)
 157     wc = tf.expand_dims(((1.0 - (x1_f - x)) * (y1_f - y)), 1)
 158     wd = tf.expand_dims(((1.0 - (x1_f - x)) * (1.0 - (y1_f - y))), 1)
 159 
 160     output = tf.add_n([wa * pixel_a, wb * pixel_b, wc * pixel_c, wd * pixel_d])
 161     output = tf.reshape(output, shape=tf.shape(im))
 162     return output, valid_mask

 164 def get_flow(t, theta, map_size, name_scope='gen_flow'):
 165   """
 166   Rotates the map by theta and translates the rotated map by t.
 167   
 168   Assume that the robot rotates by an angle theta and then moves forward by
 169   translation t. This function returns the flow field field. For every pixel in
 170   the new image it tells us which pixel in the original image it came from:
 171   NewI(x, y) = OldI(flow_x(x,y), flow_y(x,y)).
 172 
 173   Assume there is a point p in the original image. Robot rotates by R and moves
 174   forward by t.  p1 = Rt*p; p2 = p1 - t; (the world moves in opposite direction.
 175   So, p2 = Rt*p - t, thus p2 came from R*(p2+t), which is what this function
 176   calculates.
 177 
 178     t:      ... x 2 (translation for B batches of N motions each).
 179     theta:  ... x 1 (rotation for B batches of N motions each).
 180     
 181     Output: ... x map_size x map_size x 2
 182   """
 183 
 184   with tf.name_scope(name_scope):
 185     tx, ty = tf.unstack(tf.reshape(t, shape=[-1, 1, 1, 1, 2]), axis=4)
 186     theta = tf.reshape(theta, shape=[-1, 1, 1, 1])
 187     c = tf.constant((map_size-1.)/2., dtype=tf.float32)
 188 
 189     x, y = np.meshgrid(np.arange(map_size), np.arange(map_size))
 190     x = tf.constant(x[np.newaxis, :, :, np.newaxis], dtype=tf.float32, name='x', 
 191                     shape=[1, map_size, map_size, 1])
 192     y = tf.constant(y[np.newaxis, :, :, np.newaxis], dtype=tf.float32, name='y',
 193                     shape=[1,map_size, map_size, 1])
 194 
 195     x = x-(-tx+c)
 196     y = y-(-ty+c)
 197 
 198     sin_theta = tf.sin(theta)
 199     cos_theta = tf.cos(theta)
 200     xr = cos_theta*x - sin_theta*y
 201     yr = sin_theta*x + cos_theta*y
 202 
 203     xr = xr + c
 204     yr = yr + c
 205     
 206     flow = tf.stack([xr, yr], axis=-1)
 207     sh = tf.unstack(tf.shape(t), axis=0)
 208     sh = tf.stack(sh[:-1]+[tf.constant(_, dtype=tf.int32) for _ in [map_size, map_size, 2]])
 209     flow = tf.reshape(flow, shape=sh)
 210     return flow

 212 def distort_image(im, fast_mode=False):
 213   # All images in the same batch are transformed the same way, but over
 214   # iterations you see different distortions.
 215   # im should be float with values between 0 and 1.
 216   im_ = tf.reshape(im, shape=(-1,1,3))
 217   im_ = ip.apply_with_random_selector(
 218       im_, lambda x, ordering: ip.distort_color(x, ordering, fast_mode),
 219       num_cases=4)
 220   im_ = tf.reshape(im_, tf.shape(im))
 221   return im_

 223 def fc_network(x, neurons, wt_decay, name, num_pred=None, offset=0,
 224                batch_norm_param=None, dropout_ratio=0.0, is_training=None): 
 225   if dropout_ratio > 0:
 226     assert(is_training is not None), \
 227       'is_training needs to be defined when trainnig with dropout.'
 228   
 229   repr = []
 230   for i, neuron in enumerate(neurons):
 231     init_var = np.sqrt(2.0/neuron)
 232     if batch_norm_param is not None:
 233       x = slim.fully_connected(x, neuron, activation_fn=None,
 234                                weights_initializer=tf.random_normal_initializer(stddev=init_var),
 235                                weights_regularizer=slim.l2_regularizer(wt_decay),
 236                                normalizer_fn=slim.batch_norm,
 237                                normalizer_params=batch_norm_param,
 238                                biases_initializer=tf.zeros_initializer(),
 239                                scope='{:s}_{:d}'.format(name, offset+i))
 240     else:
 241       x = slim.fully_connected(x, neuron, activation_fn=tf.nn.relu,
 242                                weights_initializer=tf.random_normal_initializer(stddev=init_var),
 243                                weights_regularizer=slim.l2_regularizer(wt_decay),
 244                                biases_initializer=tf.zeros_initializer(),
 245                                scope='{:s}_{:d}'.format(name, offset+i))
 246     if dropout_ratio > 0:
 247        x = slim.dropout(x, keep_prob=1-dropout_ratio, is_training=is_training,
 248                         scope='{:s}_{:d}'.format('dropout_'+name, offset+i))
 249     repr.append(x)
 250   
 251   if num_pred is not None:
 252     init_var = np.sqrt(2.0/num_pred)
 253     x = slim.fully_connected(x, num_pred,
 254                              weights_regularizer=slim.l2_regularizer(wt_decay),
 255                              weights_initializer=tf.random_normal_initializer(stddev=init_var),
 256                              biases_initializer=tf.zeros_initializer(),
 257                              activation_fn=None,
 258                              scope='{:s}_pred'.format(name))
 259   return x, repr

 274 def sample_action(rng, action_probs, optimal_action, sample_gt_prob,
 275                   type='sample', combine_type='one_or_other'):
 276   optimal_action_ = optimal_action/np.sum(optimal_action+0., 1, keepdims=True)
 277   action_probs_ = action_probs/np.sum(action_probs+0.001, 1, keepdims=True)
 278   batch_size = action_probs_.shape[0]
 279 
 280   action = np.zeros((batch_size), dtype=np.int32)
 281   action_sample_wt = np.zeros((batch_size), dtype=np.float32)
 282   if combine_type == 'add':
 283     sample_gt_prob_ = np.minimum(np.maximum(sample_gt_prob, 0.), 1.)
 284 
 285   for i in range(batch_size):
 286     if combine_type == 'one_or_other':
 287       sample_gt = rng.rand() < sample_gt_prob
 288       if sample_gt: distr_ = optimal_action_[i,:]*1.
 289       else: distr_ = action_probs_[i,:]*1.
 290     elif combine_type == 'add':
 291       distr_ = optimal_action_[i,:]*sample_gt_prob_ + \
 292           (1.-sample_gt_prob_)*action_probs_[i,:]
 293       distr_ = distr_ / np.sum(distr_)
 294 
 295     if type == 'sample':
 296       action[i] = np.argmax(rng.multinomial(1, distr_, size=1))
 297     elif type == 'argmax':
 298       action[i] = np.argmax(distr_)
 299     action_sample_wt[i] = action_probs_[i, action[i]] / distr_[action[i]]
 300   return action, action_sample_wt

 334                    v[i].name, np.any(np.isnan(v_op_value[i])),
 335                    np.linalg.norm(v_op_value[i]))
 336 
 337   tt = utils.Timer()
 338   for i in range(iters):
 339     tt.tic()
 340     # Sample a room.
 341     e = obj.sample_env(rng_data)
 342 
 343     # Initialize the agent.
 344     init_env_state = e.reset(rng_data)
 345 
 346     # Get and process the common data.

 351       feed_dict[m.train_ops['batch_norm_is_training_op']] = False

 355     state_features = []
 356     state_targets = []
 357     net_state_to_input = []
 358     step_data_cache = []
 359     executed_actions = []
 360     rewards = []

 366     net_state_to_input.append(net_state)
 367     for j in range(num_steps):
 368       f = e.get_features(states[j], j)
 369       f = e.pre_features(f)
 370       f.update(net_state)
 371       f['step_number'] = np.ones((1,1,1), dtype=np.int32)*j
 372       state_features.append(f)
 373 
 374       feed_dict = prepare_feed_dict(m.input_tensors['step'], state_features[-1])
 375       optimal_action = e.get_optimal_action(states[j], j)
 376       for x, v in zip(m.train_ops['common'], common_data):
 377         feed_dict[x] = v
 378       if dagger_sample_bn_false:
 379         feed_dict[m.train_ops['batch_norm_is_training_op']] = False
 380       outs = sess.run([m.train_ops['step'], m.sample_gt_prob_op,
 381                        m.train_ops['step_data_cache'],
 382                        m.train_ops['updated_state'],
 383                        m.train_ops['outputs']], feed_dict=feed_dict)

 385       sample_gt_prob = outs[1]
 386       step_data_cache.append(dict(zip(m.train_ops['step_data_cache'], outs[2])))
 387       net_state = outs[3]

 390         outputs = dict(zip(m.train_ops['output_names'], outputs))

 397             rng_action, action_probs, optimal_action, sample_gt_prob,

 400         executed_actions.append(action)
 401         states.append(next_state)
 402         rewards.append(reward)
 403         action_sample_wts.append(action_sample_wt)
 404         net_state = dict(zip(m.train_ops['state_names'], net_state))
 405         net_state_to_input.append(net_state)
 406     
 407     # Concatenate things together for training.
 408     rewards = np.array(rewards).T
 409     action_sample_wts = np.array(action_sample_wts).T
 410     executed_actions = np.array(executed_actions).T
 411     all_state_targets = concat_state_x(state_targets, e.get_targets_name())
 412     all_state_features = concat_state_x(state_features,
 413                                         e.get_features_name()+['step_number'])
 414     # all_state_net = concat_state_x(net_state_to_input,
 415     # m.train_ops['state_names'])
 416     all_step_data_cache = concat_state_x(step_data_cache,
 417                                          m.train_ops['step_data_cache'])
 418 
 419     dict_train = dict(input)
 420     dict_train.update(all_state_features)
 421     dict_train.update(all_state_targets)
 422     # dict_train.update(all_state_net)
 423     dict_train.update(net_state_to_input[0])
 424     dict_train.update(all_step_data_cache)
 425     dict_train.update({'rewards': rewards, 
 426                        'action_sample_wts': action_sample_wts,
 427                        'executed_actions': executed_actions})
 428     feed_dict = prepare_feed_dict(m.input_tensors['train'], dict_train)
 429     for x in m.train_ops['step_data_cache']:
 430       feed_dict[x] = all_step_data_cache[x]
 431     if mode == 'train':
 432       n_step = sess.run(global_step)
 433 
 434       if np.mod(n_step, train_display_interval) == 0:
 435         total_loss, np_global_step, summary, print_summary = sess.run(
 436             [train_op, global_step, s_ops.summary_ops, s_ops.print_summary_ops],
 437             feed_dict=feed_dict)
 438         logging.error("")
 439       else:
 440         total_loss, np_global_step, summary = sess.run(
 441             [train_op, global_step, s_ops.summary_ops], feed_dict=feed_dict)
 442 
 443       if writer is not None and summary is not None:
 444         writer.add_summary(summary, np_global_step)
 445 
 446       should_stop = sess.run(m.should_stop_op)
 447 
 448     if mode != 'train':
 449       arop = [[] for j in range(len(s_ops.additional_return_ops))]
 450       for j in range(len(s_ops.additional_return_ops)):
 451         if s_ops.arop_summary_iters[j] < 0 or i < s_ops.arop_summary_iters[j]:
 452           arop[j] = s_ops.additional_return_ops[j]
 453       val = sess.run(arop, feed_dict=feed_dict)
 454       val_additional_ops.append(val)
 455       tt.toc(log_at=60, log_str='val timer {:d} / {:d}: '.format(i, iters), 
 456              type='time')
 457 
 458   if mode != 'train':
 459     # Write the default val summaries.
 460     summary, print_summary, np_global_step = sess.run(
 461         [s_ops.summary_ops, s_ops.print_summary_ops, global_step]) 
 462     if writer is not None and summary is not None:
 463       writer.add_summary(summary, np_global_step)
 464 
 465     # write custom validation ops
 466     val_summarys = []
 467     val_additional_ops = zip(*val_additional_ops)
 468     if len(s_ops.arop_eval_fns) > 0:
 469       val_metric_summary = tf.summary.Summary()
 470       for i in range(len(s_ops.arop_eval_fns)):
 471         val_summary = None
 472         if s_ops.arop_eval_fns[i] is not None:
 473           val_summary = s_ops.arop_eval_fns[i](val_additional_ops[i],
 474                                                np_global_step, logdir,
 475                                                val_metric_summary,
 476                                                s_ops.arop_summary_iters[i])
 477         val_summarys.append(val_summary)
 478       if writer is not None:
 479         writer.add_summary(val_metric_summary, np_global_step)
 480 
 481     # Return the additional val_ops
 482     total_loss = (val_additional_ops, val_summarys)
 483     should_stop = None
 484   
 485   return total_loss, should_stop

 651 def setup_training(loss_op, initial_learning_rate, steps_per_decay,
 652                    learning_rate_decay, momentum, max_steps,
 653                    sync=False, adjust_lr_sync=True,
 654                    num_workers=1, replica_id=0, vars_to_optimize=None, 
 655                    clip_gradient_norm=0, typ=None, momentum2=0.999,
 656                    adam_eps=1e-8):
 657   if sync and adjust_lr_sync:
 658     initial_learning_rate = initial_learning_rate * num_workers
 659     max_steps = np.int(max_steps / num_workers)
 660     steps_per_decay = np.int(steps_per_decay / num_workers)
 661 
 662   global_step_op = slim.get_or_create_global_step()
 663   lr_op          = tf.train.exponential_decay(initial_learning_rate,
 664     global_step_op, steps_per_decay, learning_rate_decay, staircase=True)
 665   if typ == 'sgd':
 666     optimizer      = tf.train.MomentumOptimizer(lr_op, momentum)
 667   elif typ == 'adam':
 668     optimizer      = tf.train.AdamOptimizer(learning_rate=lr_op, beta1=momentum,
 669                                             beta2=momentum2, epsilon=adam_eps)
 670   
 671   if sync:
 672     
 673     sync_optimizer = tf.train.SyncReplicasOptimizer(optimizer, 
 674                                                replicas_to_aggregate=num_workers, 
 675                                                replica_id=replica_id, 
 676                                                total_num_replicas=num_workers)
 677     train_op       = slim.learning.create_train_op(loss_op, sync_optimizer,
 678                                                    variables_to_train=vars_to_optimize,
 679                                                    clip_gradient_norm=clip_gradient_norm)
 680   else:
 681     sync_optimizer = None
 682     train_op       = slim.learning.create_train_op(loss_op, optimizer,
 683                                                    variables_to_train=vars_to_optimize,
 684                                                    clip_gradient_norm=clip_gradient_norm)
 685     should_stop_op = tf.greater_equal(global_step_op, max_steps)
 686   return lr_op, global_step_op, train_op, should_stop_op, optimizer, sync_optimizer

 699 def add_scalar_summary_op(tensor, name=None, 
 700     summary_key='summaries', print_summary_key='print_summaries', prefix=''):
 701   collections = []
 702   op = tf.summary.scalar(name, tensor, collections=collections)
 703   if summary_key != print_summary_key:
 704     tf.add_to_collection(summary_key, op)
 705   
 706   op = tf.Print(op, [tensor], '    {:-<25s}: '.format(name) + prefix)
 707   tf.add_to_collection(print_summary_key, op)
 708   return op

 735 def add_summary_ops(m, summarize_ops, summarize_names, to_aggregate=None,
 736                     summary_key='summaries',
 737                     print_summary_key='print_summaries', prefix=''):

 748         # agg_ops.append(tf.contrib.metrics.streaming_mean(op, return_reset_op=True))
 749       else:
 750         agg_ops.append([None, None, None])

 775 def accum_val_ops(outputs, names, global_step, output_dir, metric_summary, N):
 776   """Processes the collected outputs to compute AP for action prediction.
 777   
 778   Args:
 779     outputs        : List of scalar ops to summarize.
 780     names          : Name of the scalar ops.
 781     global_step    : global_step.
 782     output_dir     : where to store results.
 783     metric_summary : summary object to add summaries to.
 784     N              : number of outputs to process.
 785   """
 786   outs = []
 787   if N >= 0:
 788     outputs = outputs[:N]
 789   for i in range(len(outputs[0])):
 790     scalar = np.array(map(lambda x: x[i], outputs))
 791     assert(scalar.ndim == 1)
 792     add_value_to_summary(metric_summary, names[i], np.mean(scalar),
 793                          tag_str='{:>27s}:  [{:s}]: %f'.format(names[i], ''))
 794     outs.append(np.mean(scalar))
 795   return outs

 803 def simple_summaries(summarize_ops, summarize_names, mode, to_aggregate=False,
 804                      scope_name='summary'):
 805 
 806   if type(to_aggregate) != list:
 807     to_aggregate = [to_aggregate for _ in summarize_ops]
 808   
 809   summary_key = '{:s}_summaries'.format(mode)
 810   print_summary_key = '{:s}_print_summaries'.format(mode)
 811   prefix=' [{:s}]: '.format(mode)
 812   
 813   # Default ops for things that dont need to be aggregated.
 814   if not np.all(to_aggregate):
 815     for op, name, to_agg in zip(summarize_ops, summarize_names, to_aggregate):
 816       if not to_agg:
 817         add_scalar_summary_op(op, name, summary_key, print_summary_key, prefix)
 818     summary_ops = tf.summary.merge_all(summary_key)
 819     print_summary_ops = tf.summary.merge_all(print_summary_key)
 820   else:
 821     summary_ops = tf.no_op()
 822     print_summary_ops = tf.no_op()
 823  
 824   # Default ops for things that dont need to be aggregated.
 825   if np.any(to_aggregate):
 826     additional_return_ops = [[summarize_ops[i] 
 827                               for i, x in enumerate(to_aggregate )if x]]
 828     arop_summary_iters = [-1]
 829     s_names = ['{:s}/{:s}'.format(scope_name, summarize_names[i]) 
 830                for i, x in enumerate(to_aggregate) if x]
 831     fn = lambda outputs, global_step, output_dir, metric_summary, N: \
 832       accum_val_ops(outputs, s_names, global_step, output_dir, metric_summary,
 833                     N)
 834     arop_eval_fns = [fn]
 835   else:
 836     additional_return_ops = []
 837     arop_summary_iters = []
 838     arop_eval_fns = []
 839   return summary_ops, print_summary_ops, additional_return_ops, \
 840     arop_summary_iters, arop_eval_fns
