  41 def lstm_online(cell_fn, num_steps, inputs, state, varscope):
  42   # inputs is B x num_steps x C, C channels.
  43   # state is 2 tuple with B x 1 x C1, B x 1 x C2 
  44   # Output state is always B x 1 x C
  45   inputs = tf.unstack(inputs, axis=1, num=num_steps)
  46   state = tf.unstack(state, axis=1, num=1)[0]
  47   outputs = [] 
  48   
  49   if num_steps > 1: 
  50     varscope.reuse_variables()
  51   
  52   for s in range(num_steps):
  53     output, state = cell_fn(inputs[s], state)
  54     outputs.append(output)
  55   outputs = tf.stack(outputs, axis=1)
  56   state = tf.stack([state], axis=1)
  57   return outputs, state

  59 def _inputs(problem, lstm_states, lstm_state_dims):
  60   # Set up inputs.
  61   with tf.name_scope('inputs'):
  62     n_views = problem.n_views
  63 
  64     inputs = []
  65     inputs.append(('orig_maps', tf.float32,
  66                    (problem.batch_size, 1, None, None, 1)))
  67     inputs.append(('goal_loc', tf.float32,
  68                    (problem.batch_size, problem.num_goals, 2)))
  69 
  70     # For initing LSTM.
  71     inputs.append(('rel_goal_loc_at_start', tf.float32,
  72                    (problem.batch_size, problem.num_goals,
  73                     problem.rel_goal_loc_dim)))
  74     common_input_data, _ = tf_utils.setup_inputs(inputs)
  75 
  76     inputs = []
  77     inputs.append(('imgs', tf.float32, (problem.batch_size, None, n_views,
  78                                         problem.img_height, problem.img_width,
  79                                         problem.img_channels)))
  80     # Goal location as a tuple of delta location and delta theta.
  81     inputs.append(('rel_goal_loc', tf.float32, (problem.batch_size, None,
  82                                                 problem.rel_goal_loc_dim)))
  83     if problem.outputs.visit_count:
  84       inputs.append(('visit_count', tf.int32, (problem.batch_size, None, 1)))
  85       inputs.append(('last_visit', tf.int32, (problem.batch_size, None, 1)))
  86 
  87     for i, (state, dim) in enumerate(zip(lstm_states, lstm_state_dims)):
  88       inputs.append((state, tf.float32, (problem.batch_size, 1, dim)))
  89 
  90     if problem.outputs.egomotion:
  91       inputs.append(('incremental_locs', tf.float32,
  92                      (problem.batch_size, None, 2)))
  93       inputs.append(('incremental_thetas', tf.float32,
  94                      (problem.batch_size, None, 1)))
  95 
  96     inputs.append(('step_number', tf.int32, (1, None, 1)))
  97     inputs.append(('node_ids', tf.int32, (problem.batch_size, None,
  98                                           problem.node_ids_dim)))
  99     inputs.append(('perturbs', tf.float32, (problem.batch_size, None,
 100                                             problem.perturbs_dim)))
 101 
 102     # For plotting result plots
 103     inputs.append(('loc_on_map', tf.float32, (problem.batch_size, None, 2)))
 104     inputs.append(('gt_dist_to_goal', tf.float32, (problem.batch_size, None, 1)))
 105     step_input_data, _ = tf_utils.setup_inputs(inputs)
 106 
 107     inputs = []
 108     inputs.append(('executed_actions', tf.int32, (problem.batch_size, None)))
 109     inputs.append(('rewards', tf.float32, (problem.batch_size, None)))
 110     inputs.append(('action_sample_wts', tf.float32, (problem.batch_size, None)))
 111     inputs.append(('action', tf.int32, (problem.batch_size, None,
 112                                         problem.num_actions)))
 113     train_data, _ = tf_utils.setup_inputs(inputs)
 114     train_data.update(step_input_data)
 115     train_data.update(common_input_data)
 116   return common_input_data, step_input_data, train_data

 119 def _add_summaries(m, summary_mode, arop_full_summary_iters):
 120   summarize_ops = [m.lr_op, m.global_step_op, m.sample_gt_prob_op,
 121                    m.total_loss_op, m.data_loss_op, m.reg_loss_op] + m.acc_ops
 122   summarize_names = ['lr', 'global_step', 'sample_gt_prob_op', 'total_loss',
 123                      'data_loss', 'reg_loss'] + \
 124                     ['acc_{:d}'.format(i) for i in range(len(m.acc_ops))]
 125   to_aggregate = [0, 0, 0, 1, 1, 1] + [1]*len(m.acc_ops)
 126 
 127   scope_name = 'summary'
 128   with tf.name_scope(scope_name):
 129     s_ops = nu.add_default_summaries(summary_mode, arop_full_summary_iters,
 130                                      summarize_ops, summarize_names,
 131                                      to_aggregate, m.action_prob_op,
 132                                      m.input_tensors, scope_name=scope_name)
 133     m.summary_ops = {summary_mode: s_ops}

 135 def visit_count_fc(visit_count, last_visit, embed_neurons, wt_decay, fc_dropout):
 136   with tf.variable_scope('embed_visit_count'):
 137     visit_count = tf.reshape(visit_count, shape=[-1])
 138     last_visit = tf.reshape(last_visit, shape=[-1])
 139     
 140     visit_count = tf.clip_by_value(visit_count, clip_value_min=-1,
 141                                    clip_value_max=15)
 142     last_visit = tf.clip_by_value(last_visit, clip_value_min=-1,
 143                                    clip_value_max=15)
 144     visit_count = tf.one_hot(visit_count, depth=16, axis=1, dtype=tf.float32,
 145                              on_value=10., off_value=0.)
 146     last_visit = tf.one_hot(last_visit, depth=16, axis=1, dtype=tf.float32,
 147                              on_value=10., off_value=0.)
 148     f = tf.concat([visit_count, last_visit], 1)
 149     x, _ = tf_utils.fc_network(
 150         f, neurons=embed_neurons, wt_decay=wt_decay, name='visit_count_embed',
 151         offset=0, batch_norm_param=None, dropout_ratio=fc_dropout,
 152         is_training=is_training)
 153   return x

 155 def lstm_setup(name, x, batch_size, is_single_step, lstm_dim, lstm_out,
 156                num_steps, state_input_op):
 157   # returns state_name, state_init_op, updated_state_op, out_op 
 158   with tf.name_scope('reshape_'+name):
 159     sh = x.get_shape().as_list()
 160     x = tf.reshape(x, shape=[batch_size, -1, sh[-1]])
 161 
 162   with tf.variable_scope(name) as varscope:
 163     cell = tf.contrib.rnn.LSTMCell(
 164       num_units=lstm_dim, forget_bias=1.0, state_is_tuple=False,
 165       num_proj=lstm_out, use_peepholes=True,
 166       initializer=tf.random_uniform_initializer(-0.01, 0.01, seed=0),
 167       cell_clip=None, proj_clip=None)
 168 
 169     sh = [batch_size, 1, lstm_dim+lstm_out]
 170     state_init_op = tf.constant(0., dtype=tf.float32, shape=sh)
 171 
 172     fn = lambda ns: lstm_online(cell, ns, x, state_input_op, varscope)
 173     out_op, updated_state_op = tf.cond(is_single_step, lambda: fn(1), lambda:
 174                                        fn(num_steps))
 175 
 176   return name, state_init_op, updated_state_op, out_op

 178 def combine_setup(name, combine_type, embed_img, embed_goal, num_img_neuorons=None,
 179                   num_goal_neurons=None):
 180   with tf.name_scope(name + '_' + combine_type):
 181     if combine_type == 'add':
 182       # Simple concat features from goal and image
 183       out = embed_img + embed_goal
 184 
 185     elif combine_type == 'multiply':
 186       # Multiply things together
 187       re_embed_img = tf.reshape(
 188           embed_img, shape=[-1, num_img_neuorons / num_goal_neurons,
 189                             num_goal_neurons])
 190       re_embed_goal = tf.reshape(embed_goal, shape=[-1, num_goal_neurons, 1])
 191       x = tf.matmul(re_embed_img, re_embed_goal, transpose_a=False, transpose_b=False)
 192       out = slim.flatten(x)
 193     elif combine_type == 'none' or combine_type == 'imgonly':
 194       out = embed_img
 195     elif combine_type == 'goalonly':
 196       out = embed_goal
 197     else:
 198       logging.fatal('Undefined combine_type: %s', combine_type)
 199   return out

 202 def preprocess_egomotion(locs, thetas):
 203   with tf.name_scope('pre_ego'):
 204     pre_ego = tf.concat([locs, tf.sin(thetas), tf.cos(thetas)], 2)
 205     sh = pre_ego.get_shape().as_list()
 206     pre_ego = tf.reshape(pre_ego, [-1, sh[-1]])
 207   return pre_ego

 209 def setup_to_run(m, args, is_training, batch_norm_is_training, summary_mode):
 210   # Set up the model.
 211   tf.set_random_seed(args.solver.seed)
 212   task_params = args.navtask.task_params
 213   num_steps = task_params.num_steps
 214   num_goals = task_params.num_goals
 215   num_actions = task_params.num_actions
 216   num_actions_ = num_actions
 217 
 218   n_views = task_params.n_views
 219 
 220   batch_norm_is_training_op = \
 221       tf.placeholder_with_default(batch_norm_is_training, shape=[],
 222                                   name='batch_norm_is_training_op') 
 223   # Setup the inputs
 224   m.input_tensors = {}
 225   lstm_states = []; lstm_state_dims = [];
 226   state_names = []; updated_state_ops = []; init_state_ops = [];
 227   if args.arch.lstm_output:
 228     lstm_states += ['lstm_output']
 229     lstm_state_dims += [args.arch.lstm_output_dim+task_params.num_actions]
 230   if args.arch.lstm_ego:
 231     lstm_states += ['lstm_ego']
 232     lstm_state_dims += [args.arch.lstm_ego_dim + args.arch.lstm_ego_out]
 233     lstm_states += ['lstm_img']
 234     lstm_state_dims += [args.arch.lstm_img_dim + args.arch.lstm_img_out]
 235   elif args.arch.lstm_img:
 236     # An LSTM only on the image
 237     lstm_states += ['lstm_img']
 238     lstm_state_dims += [args.arch.lstm_img_dim + args.arch.lstm_img_out]
 239   else:
 240     # No LSTMs involved here.
 241     None
 242 
 243   m.input_tensors['common'], m.input_tensors['step'], m.input_tensors['train'] = \
 244       _inputs(task_params, lstm_states, lstm_state_dims)
 245 
 246   with tf.name_scope('check_size'):
 247     is_single_step = tf.equal(tf.unstack(tf.shape(m.input_tensors['step']['imgs']), 
 248                                         num=6)[1], 1)
 249 
 250   images_reshaped = tf.reshape(m.input_tensors['step']['imgs'], 
 251       shape=[-1, task_params.img_height, task_params.img_width,
 252              task_params.img_channels], name='re_image')
 253 
 254   rel_goal_loc_reshaped = tf.reshape(m.input_tensors['step']['rel_goal_loc'], 
 255       shape=[-1, task_params.rel_goal_loc_dim], name='re_rel_goal_loc')
 256 
 257   x, vars_ = get_repr_from_image(
 258       images_reshaped, task_params.modalities, task_params.data_augment,
 259       args.arch.encoder, args.solver.freeze_conv, args.solver.wt_decay,
 260       is_training)
 261 
 262   # Reshape into nice things so that these can be accumulated over time steps
 263   # for faster backprop.
 264   sh_before = x.get_shape().as_list()
 265   m.encoder_output = tf.reshape(
 266       x, shape=[task_params.batch_size, -1, n_views] + sh_before[1:])
 267   x = tf.reshape(m.encoder_output, shape=[-1] + sh_before[1:])
 268 
 269   # Add a layer to reduce dimensions for a fc layer.
 270   if args.arch.dim_reduce_neurons > 0:
 271     ks = 1; neurons = args.arch.dim_reduce_neurons;
 272     init_var = np.sqrt(2.0/(ks**2)/neurons)
 273     batch_norm_param = args.arch.batch_norm_param
 274     batch_norm_param['is_training'] = batch_norm_is_training_op
 275     m.conv_feat = slim.conv2d(
 276         x, neurons, kernel_size=ks, stride=1, normalizer_fn=slim.batch_norm,
 277         normalizer_params=batch_norm_param, padding='SAME', scope='dim_reduce',
 278         weights_regularizer=slim.l2_regularizer(args.solver.wt_decay),
 279         weights_initializer=tf.random_normal_initializer(stddev=init_var))
 280     reshape_conv_feat = slim.flatten(m.conv_feat)
 281     sh = reshape_conv_feat.get_shape().as_list()
 282     m.reshape_conv_feat = tf.reshape(reshape_conv_feat, 
 283                                      shape=[-1, sh[1]*n_views])
 284 
 285   # Restore these from a checkpoint.
 286   if args.solver.pretrained_path is not None:
 287     m.init_fn = slim.assign_from_checkpoint_fn(args.solver.pretrained_path,
 288                                                vars_)
 289   else:
 290     m.init_fn = None
 291 
 292   # Hit the goal_location with a bunch of fully connected layers, to embed it
 293   # into some space.
 294   with tf.variable_scope('embed_goal'):
 295     batch_norm_param = args.arch.batch_norm_param
 296     batch_norm_param['is_training'] = batch_norm_is_training_op
 297     m.embed_goal, _ = tf_utils.fc_network(
 298         rel_goal_loc_reshaped, neurons=args.arch.goal_embed_neurons,
 299         wt_decay=args.solver.wt_decay, name='goal_embed', offset=0,
 300         batch_norm_param=batch_norm_param, dropout_ratio=args.arch.fc_dropout,
 301         is_training=is_training)
 302   
 303   if args.arch.embed_goal_for_state:
 304     with tf.variable_scope('embed_goal_for_state'):
 305       batch_norm_param = args.arch.batch_norm_param
 306       batch_norm_param['is_training'] = batch_norm_is_training_op
 307       m.embed_goal_for_state, _ = tf_utils.fc_network(
 308           m.input_tensors['common']['rel_goal_loc_at_start'][:,0,:],
 309           neurons=args.arch.goal_embed_neurons, wt_decay=args.solver.wt_decay,
 310           name='goal_embed', offset=0, batch_norm_param=batch_norm_param,
 311           dropout_ratio=args.arch.fc_dropout, is_training=is_training)
 312 
 313   # Hit the goal_location with a bunch of fully connected layers, to embed it
 314   # into some space.
 315   with tf.variable_scope('embed_img'):
 316     batch_norm_param = args.arch.batch_norm_param
 317     batch_norm_param['is_training'] = batch_norm_is_training_op
 318     m.embed_img, _ = tf_utils.fc_network(
 319         m.reshape_conv_feat, neurons=args.arch.img_embed_neurons,
 320         wt_decay=args.solver.wt_decay, name='img_embed', offset=0,
 321         batch_norm_param=batch_norm_param, dropout_ratio=args.arch.fc_dropout,
 322         is_training=is_training)
 323 
 324   # For lstm_ego, and lstm_image, embed the ego motion, accumulate it into an
 325   # LSTM, combine with image features and accumulate those in an LSTM. Finally
 326   # combine what you get from the image LSTM with the goal to output an action.
 327   if args.arch.lstm_ego:
 328     ego_reshaped = preprocess_egomotion(m.input_tensors['step']['incremental_locs'], 
 329                                         m.input_tensors['step']['incremental_thetas'])
 330     with tf.variable_scope('embed_ego'):
 331       batch_norm_param = args.arch.batch_norm_param
 332       batch_norm_param['is_training'] = batch_norm_is_training_op
 333       m.embed_ego, _ = tf_utils.fc_network(
 334           ego_reshaped, neurons=args.arch.ego_embed_neurons,
 335           wt_decay=args.solver.wt_decay, name='ego_embed', offset=0,
 336           batch_norm_param=batch_norm_param, dropout_ratio=args.arch.fc_dropout,
 337           is_training=is_training)
 338 
 339     state_name, state_init_op, updated_state_op, out_op = lstm_setup(
 340         'lstm_ego', m.embed_ego, task_params.batch_size, is_single_step, 
 341         args.arch.lstm_ego_dim, args.arch.lstm_ego_out, num_steps*num_goals,
 342         m.input_tensors['step']['lstm_ego'])
 343     state_names += [state_name]
 344     init_state_ops += [state_init_op]
 345     updated_state_ops += [updated_state_op]
 346 
 347     # Combine the output with the vision features.
 348     m.img_ego_op = combine_setup('img_ego', args.arch.combine_type_ego,
 349                                  m.embed_img, out_op,
 350                                  args.arch.img_embed_neurons[-1],
 351                                  args.arch.lstm_ego_out)
 352 
 353     # LSTM on these vision features.
 354     state_name, state_init_op, updated_state_op, out_op = lstm_setup(
 355         'lstm_img', m.img_ego_op, task_params.batch_size, is_single_step, 
 356         args.arch.lstm_img_dim, args.arch.lstm_img_out, num_steps*num_goals,
 357         m.input_tensors['step']['lstm_img'])
 358     state_names += [state_name]
 359     init_state_ops += [state_init_op]
 360     updated_state_ops += [updated_state_op]
 361 
 362     m.img_for_goal = out_op
 363     num_img_for_goal_neurons = args.arch.lstm_img_out
 364 
 365   elif args.arch.lstm_img:
 366     # LSTM on just the image features.
 367     state_name, state_init_op, updated_state_op, out_op = lstm_setup(
 368         'lstm_img', m.embed_img, task_params.batch_size, is_single_step,
 369         args.arch.lstm_img_dim, args.arch.lstm_img_out, num_steps*num_goals,
 370         m.input_tensors['step']['lstm_img'])
 371     state_names += [state_name]
 372     init_state_ops += [state_init_op]
 373     updated_state_ops += [updated_state_op]
 374     m.img_for_goal = out_op
 375     num_img_for_goal_neurons = args.arch.lstm_img_out
 376 
 377   else:
 378     m.img_for_goal = m.embed_img
 379     num_img_for_goal_neurons = args.arch.img_embed_neurons[-1]
 380 
 381 
 382   if args.arch.use_visit_count:
 383     m.embed_visit_count = visit_count_fc(
 384         m.input_tensors['step']['visit_count'],
 385         m.input_tensors['step']['last_visit'], args.arch.goal_embed_neurons,
 386         args.solver.wt_decay, args.arch.fc_dropout, is_training=is_training)
 387     m.embed_goal = m.embed_goal + m.embed_visit_count
 388   
 389   m.combined_f = combine_setup('img_goal', args.arch.combine_type,
 390                                m.img_for_goal, m.embed_goal,
 391                                num_img_for_goal_neurons,
 392                                args.arch.goal_embed_neurons[-1])
 393 
 394   # LSTM on the combined representation.
 395   if args.arch.lstm_output:
 396     name = 'lstm_output'
 397     # A few fully connected layers here.
 398     with tf.variable_scope('action_pred'):
 399       batch_norm_param = args.arch.batch_norm_param
 400       batch_norm_param['is_training'] = batch_norm_is_training_op
 401       x, _ = tf_utils.fc_network(
 402           m.combined_f, neurons=args.arch.pred_neurons,
 403           wt_decay=args.solver.wt_decay, name='pred', offset=0,
 404           batch_norm_param=batch_norm_param, dropout_ratio=args.arch.fc_dropout)
 405 
 406     if args.arch.lstm_output_init_state_from_goal:
 407       # Use the goal embedding to initialize the LSTM state.
 408       # UGLY CLUGGY HACK: if this is doing computation for a single time step
 409       # then this will not involve back prop, so we can use the state input from
 410       # the feed dict, otherwise we compute the state representation from the
 411       # goal and feed that in. Necessary for using goal location to generate the
 412       # state representation.
 413       m.embed_goal_for_state = tf.expand_dims(m.embed_goal_for_state, dim=1)
 414       state_op = tf.cond(is_single_step, lambda: m.input_tensors['step'][name],
 415                          lambda: m.embed_goal_for_state)
 416       state_name, state_init_op, updated_state_op, out_op = lstm_setup(
 417           name, x, task_params.batch_size, is_single_step,
 418           args.arch.lstm_output_dim,
 419           num_actions_,
 420           num_steps*num_goals, state_op)
 421       init_state_ops += [m.embed_goal_for_state]
 422     else:
 423       state_op = m.input_tensors['step'][name]
 424       state_name, state_init_op, updated_state_op, out_op = lstm_setup(
 425           name, x, task_params.batch_size, is_single_step,
 426           args.arch.lstm_output_dim,
 427           num_actions_, num_steps*num_goals, state_op)
 428       init_state_ops += [state_init_op]
 429 
 430     state_names += [state_name]
 431     updated_state_ops += [updated_state_op]
 432 
 433     out_op = tf.reshape(out_op, shape=[-1, num_actions_])
 434     if num_actions_ > num_actions:
 435       m.action_logits_op = out_op[:,:num_actions]
 436       m.baseline_op = out_op[:,num_actions:]
 437     else:
 438       m.action_logits_op = out_op
 439       m.baseline_op = None
 440     m.action_prob_op = tf.nn.softmax(m.action_logits_op)
 441 
 442   else:
 443     # A few fully connected layers here.
 444     with tf.variable_scope('action_pred'):
 445       batch_norm_param = args.arch.batch_norm_param
 446       batch_norm_param['is_training'] = batch_norm_is_training_op
 447       out_op, _ = tf_utils.fc_network(
 448           m.combined_f, neurons=args.arch.pred_neurons,
 449           wt_decay=args.solver.wt_decay, name='pred', offset=0,
 450           num_pred=num_actions_,
 451           batch_norm_param=batch_norm_param,
 452           dropout_ratio=args.arch.fc_dropout, is_training=is_training)
 453       if num_actions_ > num_actions:
 454         m.action_logits_op = out_op[:,:num_actions]
 455         m.baseline_op = out_op[:,num_actions:]
 456       else:
 457         m.action_logits_op = out_op 
 458         m.baseline_op = None
 459       m.action_prob_op = tf.nn.softmax(m.action_logits_op)
 460 
 461   m.train_ops = {}
 462   m.train_ops['step'] = m.action_prob_op
 463   m.train_ops['common'] = [m.input_tensors['common']['orig_maps'],
