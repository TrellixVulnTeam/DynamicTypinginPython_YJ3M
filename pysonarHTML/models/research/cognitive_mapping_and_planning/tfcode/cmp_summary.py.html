  35 def _vis_readout_maps(outputs, global_step, output_dir, metric_summary, N):
  36   # outputs is [gt_map, pred_map]:
  37   if N >= 0:
  38     outputs = outputs[:N]
  39   N = len(outputs)
  40 
  41   plt.set_cmap('jet')
  42   fig, axes = utils.subplot(plt, (N, outputs[0][0].shape[4]*2), (5,5))
  43   axes = axes.ravel()[::-1].tolist()
  44   for i in range(N):
  45     gt_map, pred_map = outputs[i]
  46     for j in [0]:
  47       for k in range(gt_map.shape[4]):
  48         # Display something like the midpoint of the trajectory.
  49         id = np.int(gt_map.shape[1]/2)
  50 
  51         ax = axes.pop();
  52         ax.imshow(gt_map[j,id,:,:,k], origin='lower', interpolation='none',
  53                   vmin=0., vmax=1.)
  54         ax.set_axis_off();
  55         if i == 0: ax.set_title('gt_map')
  56 
  57         ax = axes.pop();
  58         ax.imshow(pred_map[j,id,:,:,k], origin='lower', interpolation='none',
  59                   vmin=0., vmax=1.)
  60         ax.set_axis_off();
  61         if i == 0: ax.set_title('pred_map')
  62 
  63   file_name = os.path.join(output_dir, 'readout_map_{:d}.png'.format(global_step))
  64   with fu.fopen(file_name, 'w') as f:
  65     fig.savefig(f, bbox_inches='tight', transparent=True, pad_inches=0)
  66   plt.close(fig)

  68 def _vis(outputs, global_step, output_dir, metric_summary, N):
  69   # Plot the value map, goal for various maps to see what if the model is
  70   # learning anything useful.
  71   #
  72   # outputs is [values, goals, maps, occupancy, conf].
  73   #
  74   if N >= 0:
  75     outputs = outputs[:N]
  76   N = len(outputs)
  77 
  78   plt.set_cmap('jet')
  79   fig, axes = utils.subplot(plt, (N, outputs[0][0].shape[4]*5), (5,5))
  80   axes = axes.ravel()[::-1].tolist()
  81   for i in range(N):
  82     values, goals, maps, occupancy, conf = outputs[i]
  83     for j in [0]:
  84       for k in range(values.shape[4]):
  85         # Display something like the midpoint of the trajectory.
  86         id = np.int(values.shape[1]/2)
  87 
  88         ax = axes.pop();
  89         ax.imshow(goals[j,id,:,:,k], origin='lower', interpolation='none')
  90         ax.set_axis_off();
  91         if i == 0: ax.set_title('goal')
  92 
  93         ax = axes.pop();
  94         ax.imshow(occupancy[j,id,:,:,k], origin='lower', interpolation='none')
  95         ax.set_axis_off();
  96         if i == 0: ax.set_title('occupancy')
  97 
  98         ax = axes.pop();
  99         ax.imshow(conf[j,id,:,:,k], origin='lower', interpolation='none',
 100                   vmin=0., vmax=1.)
 101         ax.set_axis_off();
 102         if i == 0: ax.set_title('conf')
 103 
 104         ax = axes.pop();
 105         ax.imshow(values[j,id,:,:,k], origin='lower', interpolation='none')
 106         ax.set_axis_off();
 107         if i == 0: ax.set_title('value')
 108 
 109         ax = axes.pop();
 110         ax.imshow(maps[j,id,:,:,k], origin='lower', interpolation='none')
 111         ax.set_axis_off();
 112         if i == 0: ax.set_title('incr map')
 113 
 114   file_name = os.path.join(output_dir, 'value_vis_{:d}.png'.format(global_step))
 115   with fu.fopen(file_name, 'w') as f:
 116     fig.savefig(f, bbox_inches='tight', transparent=True, pad_inches=0)
 117   plt.close(fig)

 119 def _summary_vis(m, batch_size, num_steps, arop_full_summary_iters):
 120   arop = []; arop_summary_iters = []; arop_eval_fns = [];
 121   vis_value_ops = []; vis_goal_ops = []; vis_map_ops = []; 
 122   vis_occupancy_ops = []; vis_conf_ops = [];
 123   for i, val_op in enumerate(m.value_ops):
 124     vis_value_op = tf.reduce_mean(tf.abs(val_op), axis=3, keep_dims=True)
 125     vis_value_ops.append(vis_value_op)
 126     
 127     vis_occupancy_op = tf.reduce_mean(tf.abs(m.occupancys[i]), 3, True)
 128     vis_occupancy_ops.append(vis_occupancy_op)
 129     
 130     vis_conf_op = tf.reduce_max(tf.abs(m.confs[i]), axis=3, keep_dims=True)
 131     vis_conf_ops.append(vis_conf_op)
 132     
 133     ego_goal_imgs_i_op = m.input_tensors['step']['ego_goal_imgs_{:d}'.format(i)]
 134     vis_goal_op = tf.reduce_max(ego_goal_imgs_i_op, 4, True)
 135     vis_goal_ops.append(vis_goal_op)
 136     
 137     vis_map_op = tf.reduce_mean(tf.abs(m.ego_map_ops[i]), 4, True)
 138     vis_map_ops.append(vis_map_op)
 139 
 140   vis_goal_ops = tf.concat(vis_goal_ops, 4)
 141   vis_map_ops = tf.concat(vis_map_ops, 4)
 142   vis_value_ops = tf.concat(vis_value_ops, 3)
 143   vis_occupancy_ops = tf.concat(vis_occupancy_ops, 3)
 144   vis_conf_ops = tf.concat(vis_conf_ops, 3)
 145 
 146   sh = tf.unstack(tf.shape(vis_value_ops))[1:]
 147   vis_value_ops = tf.reshape(vis_value_ops, shape=[batch_size, -1] + sh)
 148 
 149   sh = tf.unstack(tf.shape(vis_conf_ops))[1:]
 150   vis_conf_ops = tf.reshape(vis_conf_ops, shape=[batch_size, -1] + sh)
 151 
 152   sh = tf.unstack(tf.shape(vis_occupancy_ops))[1:]
 153   vis_occupancy_ops = tf.reshape(vis_occupancy_ops, shape=[batch_size,-1] + sh)
 154 
 155   # Save memory, only return time steps that need to be visualized, factor of
 156   # 32 CPU memory saving.
 157   id = np.int(num_steps/2)
 158   vis_goal_ops = tf.expand_dims(vis_goal_ops[:,id,:,:,:], axis=1)
 159   vis_map_ops = tf.expand_dims(vis_map_ops[:,id,:,:,:], axis=1)
 160   vis_value_ops = tf.expand_dims(vis_value_ops[:,id,:,:,:], axis=1)
 161   vis_conf_ops = tf.expand_dims(vis_conf_ops[:,id,:,:,:], axis=1)
 162   vis_occupancy_ops = tf.expand_dims(vis_occupancy_ops[:,id,:,:,:], axis=1)
 163 
 164   arop += [[vis_value_ops, vis_goal_ops, vis_map_ops, vis_occupancy_ops,
 165             vis_conf_ops]]
 166   arop_summary_iters += [arop_full_summary_iters]
 167   arop_eval_fns += [_vis]
 168   return arop, arop_summary_iters, arop_eval_fns

 170 def _summary_readout_maps(m, num_steps, arop_full_summary_iters):
 171   arop = []; arop_summary_iters = []; arop_eval_fns = [];
 172   id = np.int(num_steps-1)
 173   vis_readout_maps_gt = m.readout_maps_gt
 174   vis_readout_maps_prob = tf.reshape(m.readout_maps_probs,
 175                                      shape=tf.shape(vis_readout_maps_gt))
 176   vis_readout_maps_gt = tf.expand_dims(vis_readout_maps_gt[:,id,:,:,:], 1)
 177   vis_readout_maps_prob = tf.expand_dims(vis_readout_maps_prob[:,id,:,:,:], 1)
 178   arop += [[vis_readout_maps_gt, vis_readout_maps_prob]]
 179   arop_summary_iters += [arop_full_summary_iters]
 180   arop_eval_fns += [_vis_readout_maps]
 181   return arop, arop_summary_iters, arop_eval_fns

 183 def _add_summaries(m, args, summary_mode, arop_full_summary_iters):
 184   task_params = args.navtask.task_params
 185   
 186   summarize_ops = [m.lr_op, m.global_step_op, m.sample_gt_prob_op] + \
 187       m.loss_ops + m.acc_ops
 188   summarize_names = ['lr', 'global_step', 'sample_gt_prob_op'] + \
 189       m.loss_ops_names + ['acc_{:d}'.format(i) for i in range(len(m.acc_ops))]
 190   to_aggregate = [0, 0, 0] + [1]*len(m.loss_ops_names) + [1]*len(m.acc_ops)
 191 
 192   scope_name = 'summary'
 193   with tf.name_scope(scope_name):
 194     s_ops = nu.add_default_summaries(summary_mode, arop_full_summary_iters,
 195                                      summarize_ops, summarize_names,
 196                                      to_aggregate, m.action_prob_op,
 197                                      m.input_tensors, scope_name=scope_name)
 198     if summary_mode == 'val':
 199       arop, arop_summary_iters, arop_eval_fns = _summary_vis(
 200           m, task_params.batch_size, task_params.num_steps,
 201           arop_full_summary_iters)
 202       s_ops.additional_return_ops += arop
 203       s_ops.arop_summary_iters += arop_summary_iters
 204       s_ops.arop_eval_fns += arop_eval_fns
 205       
 206       if args.arch.readout_maps:
 207         arop, arop_summary_iters, arop_eval_fns = _summary_readout_maps(
 208             m, task_params.num_steps, arop_full_summary_iters)
 209         s_ops.additional_return_ops += arop
 210         s_ops.arop_summary_iters += arop_summary_iters
 211         s_ops.arop_eval_fns += arop_eval_fns
 212   
 213   return s_ops
