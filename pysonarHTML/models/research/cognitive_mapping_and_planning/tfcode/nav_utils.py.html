  38 def compute_losses_multi_or(logits, actions_one_hot, weights=None,
  39                             num_actions=-1, data_loss_wt=1., reg_loss_wt=1.,
  40                             ewma_decay=0.99, reg_loss_op=None):
  41   assert(num_actions > 0), 'num_actions must be specified and must be > 0.'
  42   
  43   with tf.name_scope('loss'):
  44     if weights is None:
  45       weight = tf.ones_like(actions_one_hot, dtype=tf.float32, name='weight')
  46     
  47     actions_one_hot = tf.cast(tf.reshape(actions_one_hot, [-1, num_actions],
  48                                          're_actions_one_hot'), tf.float32)
  49     weights = tf.reduce_sum(tf.reshape(weights, [-1, num_actions], 're_weight'),
  50                             reduction_indices=1)
  51     total = tf.reduce_sum(weights)
  52 
  53     action_prob = tf.nn.softmax(logits)
  54     action_prob = tf.reduce_sum(tf.multiply(action_prob, actions_one_hot),
  55                                 reduction_indices=1)
  56     example_loss = -tf.log(tf.maximum(tf.constant(1e-4), action_prob))
  57 
  58     data_loss_op = tf.reduce_sum(example_loss * weights) / total
  59     if reg_loss_op is None:
  60       if reg_loss_wt > 0:
  61         reg_loss_op = tf.add_n(tf.losses.get_regularization_losses())
  62       else:
  63         reg_loss_op = tf.constant(0.)
  64     
  65     if reg_loss_wt > 0:
  66       total_loss_op = data_loss_wt*data_loss_op + reg_loss_wt*reg_loss_op 
  67     else:
  68       total_loss_op = data_loss_wt*data_loss_op
  69 
  70     is_correct = tf.cast(tf.greater(action_prob, 0.5, name='pred_class'), tf.float32)
  71     acc_op = tf.reduce_sum(is_correct*weights) / total
  72 
  73     ewma_acc_op = moving_averages.weighted_moving_average(
  74         acc_op, ewma_decay, weight=total, name='ewma_acc')
  75 
  76     acc_ops = [ewma_acc_op]
  77 
  78   return reg_loss_op, data_loss_op, total_loss_op, acc_ops

  81 def get_repr_from_image(images_reshaped, modalities, data_augment, encoder,
  82                         freeze_conv, wt_decay, is_training):
  83   # Pass image through lots of convolutional layers, to obtain pool5
  84   if modalities == ['rgb']:
  85     with tf.name_scope('pre_rgb'):
  86       x = (images_reshaped + 128.) / 255. # Convert to brightness between 0 and 1.
  87       if data_augment.relight and is_training:
  88         x = tf_utils.distort_image(x, fast_mode=data_augment.relight_fast)
  89       x = (x-0.5)*2.0
  90     scope_name = encoder
  91   elif modalities == ['depth']:
  92     with tf.name_scope('pre_d'):
  93       d_image = images_reshaped
  94       x = 2*(d_image[...,0] - 80.0)/100.0
  95       y = d_image[...,1]
  96       d_image = tf.concat([tf.expand_dims(x, -1), tf.expand_dims(y, -1)], 3)
  97       x = d_image
  98     scope_name = 'd_'+encoder
  99 
 100   resnet_is_training = is_training and (not freeze_conv)
 101   with slim.arg_scope(resnet_v2.resnet_utils.resnet_arg_scope(resnet_is_training)):
 102     fn = getattr(tf_utils, encoder)
 103     x, end_points = fn(x, num_classes=None, global_pool=False,
 104                        output_stride=None, reuse=None,
 105                        scope=scope_name)
 106   vars_ = slim.get_variables_to_restore()
 107 
 108   conv_feat = x
 109   return conv_feat, vars_

 111 def default_train_step_kwargs(m, obj, logdir, rng_seed, is_chief, num_steps,
 112                               iters, train_display_interval,
 113                               dagger_sample_bn_false):
 114   train_step_kwargs = {}
 115   train_step_kwargs['obj'] = obj 
 116   train_step_kwargs['m'] = m
 117   
 118   # rng_data has 2 independent rngs, one for sampling episodes and one for
 119   # sampling perturbs (so that we can make results reproducible.
 120   train_step_kwargs['rng_data'] = [np.random.RandomState(rng_seed), 
 121                                    np.random.RandomState(rng_seed)]
 122   train_step_kwargs['rng_action'] = np.random.RandomState(rng_seed)
 123   if is_chief: 
 124     train_step_kwargs['writer'] = tf.summary.FileWriter(logdir) #, m.tf_graph)
 125   else:
 126     train_step_kwargs['writer'] = None
 127   train_step_kwargs['iters'] = iters
 128   train_step_kwargs['train_display_interval'] = train_display_interval 
 129   train_step_kwargs['num_steps'] = num_steps
 130   train_step_kwargs['logdir'] = logdir
 131   train_step_kwargs['dagger_sample_bn_false'] = dagger_sample_bn_false 
 132   return train_step_kwargs

 135 def save_d_at_t(outputs, global_step, output_dir, metric_summary, N):

 160 def save_all(outputs, global_step, output_dir, metric_summary, N):
 161   """Save numerous statistics.
 162   
 163   Args:
 164     outputs        : [locs, goal_loc, gt_dist_to_goal, node_ids, perturbs]
 165     global_step    : number of iterations.
 166     output_dir     : output directory.
 167     metric_summary : to append scalars to summary.
 168     N              : number of outputs to process.
 169   """
 170   all_locs = np.concatenate(map(lambda x: x[0], outputs), axis=0)
 171   all_goal_locs = np.concatenate(map(lambda x: x[1], outputs), axis=0)
 172   all_d_at_t = np.concatenate(map(lambda x: x[2][:,:,0]*1, outputs), axis=0)
 173   all_node_ids = np.concatenate(map(lambda x: x[3], outputs), axis=0)
 174   all_perturbs = np.concatenate(map(lambda x: x[4], outputs), axis=0)
 175   
 176   file_name = os.path.join(output_dir, 'all_locs_at_t_{:d}.pkl'.format(global_step))
 177   vars = [all_locs, all_goal_locs, all_d_at_t, all_node_ids, all_perturbs]
 178   var_names = ['all_locs', 'all_goal_locs', 'all_d_at_t', 'all_node_ids', 'all_perturbs']
 179   utils.save_variables(file_name, vars, var_names,  overwrite=True)
 180   return None

 182 def eval_ap(outputs, global_step, output_dir, metric_summary, N, num_classes=4):
 183   """Processes the collected outputs to compute AP for action prediction.
 184   
 185   Args:
 186     outputs        : [logits, labels]
 187     global_step    : global_step.
 188     output_dir     : where to store results.
 189     metric_summary : summary object to add summaries to.
 190     N              : number of outputs to process.
 191     num_classes    : number of classes to compute AP over, and to reshape tensors.
 192   """
 193   if N >= 0:
 194     outputs = outputs[:N]
 195   logits = np.concatenate(map(lambda x: x[0], outputs), axis=0).reshape((-1, num_classes))
 196   labels = np.concatenate(map(lambda x: x[1], outputs), axis=0).reshape((-1, num_classes))
 197   aps = []
 198   for i in range(logits.shape[1]):
 199     ap, rec, prec = utils.calc_pr(labels[:,i], logits[:,i])
 200     ap = ap[0]
 201     tf_utils.add_value_to_summary(metric_summary, 'aps/ap_{:d}: '.format(i), ap)
 202     aps.append(ap)
 203   return aps

 205 def eval_dist(outputs, global_step, output_dir, metric_summary, N):
 206   """Processes the collected outputs during validation to 
 207   1. Plot the distance over time curve.
 208   2. Compute mean and median distances.
 209   3. Plots histogram of end distances.
 210   
 211   Args:
 212     outputs        : [locs, goal_loc, gt_dist_to_goal].
 213     global_step    : global_step.
 214     output_dir     : where to store results.
 215     metric_summary : summary object to add summaries to.
 216     N              : number of outputs to process.
 217   """
 218   SUCCESS_THRESH = 3
 219   if N >= 0:
 220     outputs = outputs[:N]
 221   
 222   # Plot distance at time t.
 223   d_at_t = []
 224   for i in range(len(outputs)):
 225     locs, goal_loc, gt_dist_to_goal = outputs[i]
 226     d_at_t.append(gt_dist_to_goal[:,:,0]*1)
 227 
 228   # Plot the distance.
 229   fig, axes = utils.subplot(plt, (1,1), (5,5))
 230   d_at_t = np.concatenate(d_at_t, axis=0)
 231   axes.plot(np.arange(d_at_t.shape[1]), np.mean(d_at_t, axis=0), 'r.')
 232   axes.set_xlabel('time step')
 233   axes.set_ylabel('dist to next goal')
 234   axes.grid('on')
 235   file_name = os.path.join(output_dir, 'dist_at_t_{:d}.png'.format(global_step))
 236   with fu.fopen(file_name, 'w') as f:
 237     fig.savefig(f, bbox_inches='tight', transparent=True, pad_inches=0)
 238   file_name = os.path.join(output_dir, 'dist_at_t_{:d}.pkl'.format(global_step))
 239   utils.save_variables(file_name, [d_at_t], ['d_at_t'], overwrite=True)
 240   plt.close(fig)
 241 
 242   # Plot the trajectories and the init_distance and final distance.
 243   d_inits = []
 244   d_ends = []
 245   for i in range(len(outputs)):
 246     locs, goal_loc, gt_dist_to_goal = outputs[i]
 247     d_inits.append(gt_dist_to_goal[:,0,0]*1)
 248     d_ends.append(gt_dist_to_goal[:,-1,0]*1)
 249 
 250   # Plot the distance.
 251   fig, axes = utils.subplot(plt, (1,1), (5,5))
 252   d_inits = np.concatenate(d_inits, axis=0)
 253   d_ends = np.concatenate(d_ends, axis=0)
 254   axes.plot(d_inits+np.random.rand(*(d_inits.shape))-0.5,
 255             d_ends+np.random.rand(*(d_ends.shape))-0.5, '.', mec='red', mew=1.0)
 256   axes.set_xlabel('init dist'); axes.set_ylabel('final dist'); 
 257   axes.grid('on'); axes.axis('equal');
 258   title_str = 'mean: {:0.1f}, 50: {:0.1f}, 75: {:0.2f}, s: {:0.1f}'
 259   title_str = title_str.format(
 260       np.mean(d_ends), np.median(d_ends), np.percentile(d_ends, q=75),
 261       100*(np.mean(d_ends <= SUCCESS_THRESH)))
 262   axes.set_title(title_str)
 263   file_name = os.path.join(output_dir, 'dist_{:d}.png'.format(global_step))
 264   with fu.fopen(file_name, 'w') as f:
 265     fig.savefig(f, bbox_inches='tight', transparent=True, pad_inches=0)
 266 
 267   file_name = os.path.join(output_dir, 'dist_{:d}.pkl'.format(global_step))
 268   utils.save_variables(file_name, [d_inits, d_ends], ['d_inits', 'd_ends'],
 269                        overwrite=True)
 270   plt.close(fig)
 271 
 272   # Plot the histogram of the end_distance.
 273   with plt.style.context('seaborn-white'):
 274     d_ends_ = np.sort(d_ends)
 275     d_inits_ = np.sort(d_inits)
 276     leg = [];
 277     fig, ax = utils.subplot(plt, (1,1), (5,5))
 278     ax.grid('on')
 279     ax.set_xlabel('Distance from goal'); ax.xaxis.label.set_fontsize(16);
 280     ax.set_ylabel('Fraction of data'); ax.yaxis.label.set_fontsize(16);
 281     ax.plot(d_ends_, np.arange(d_ends_.size)*1./d_ends_.size, 'r')
 282     ax.plot(d_inits_, np.arange(d_inits_.size)*1./d_inits_.size, 'k')
 283     leg.append('Final'); leg.append('Init');
 284     ax.legend(leg, fontsize='x-large');
 285     ax.set_axis_on()
 286     title_str = 'mean: {:0.1f}, 50: {:0.1f}, 75: {:0.2f}, s: {:0.1f}'
 287     title_str = title_str.format(
 288         np.mean(d_ends), np.median(d_ends), np.percentile(d_ends, q=75),
 289         100*(np.mean(d_ends <= SUCCESS_THRESH)))
 290     ax.set_title(title_str)
 291     file_name = os.path.join(output_dir, 'dist_hist_{:d}.png'.format(global_step))
 292     with fu.fopen(file_name, 'w') as f:
 293       fig.savefig(f, bbox_inches='tight', transparent=True, pad_inches=0)
 294   
 295   # Log distance metrics.
 296   tf_utils.add_value_to_summary(metric_summary, 'dists/success_init: ',
 297                                 100*(np.mean(d_inits <= SUCCESS_THRESH)))
 298   tf_utils.add_value_to_summary(metric_summary, 'dists/success_end: ',
 299                                 100*(np.mean(d_ends <= SUCCESS_THRESH)))
 300   tf_utils.add_value_to_summary(metric_summary, 'dists/dist_init (75): ',
 301                                 np.percentile(d_inits, q=75))
 302   tf_utils.add_value_to_summary(metric_summary, 'dists/dist_end (75): ',
 303                                 np.percentile(d_ends, q=75))
 304   tf_utils.add_value_to_summary(metric_summary, 'dists/dist_init (median): ',
 305                                 np.median(d_inits))
 306   tf_utils.add_value_to_summary(metric_summary, 'dists/dist_end (median): ',
 307                                 np.median(d_ends))
 308   tf_utils.add_value_to_summary(metric_summary, 'dists/dist_init (mean): ',
 309                                 np.mean(d_inits))
 310   tf_utils.add_value_to_summary(metric_summary, 'dists/dist_end (mean): ',
 311                                 np.mean(d_ends))
 312   return np.median(d_inits), np.median(d_ends), np.mean(d_inits), np.mean(d_ends), \
 313       np.percentile(d_inits, q=75), np.percentile(d_ends, q=75), \
 314       100*(np.mean(d_inits) <= SUCCESS_THRESH), 100*(np.mean(d_ends) <= SUCCESS_THRESH)

 316 def plot_trajectories(outputs, global_step, output_dir, metric_summary, N):
 317   """Processes the collected outputs during validation to plot the trajectories
 318   in the top view.
 319   
 320   Args:
 321     outputs        : [locs, orig_maps, goal_loc].
 322     global_step    : global_step.
 323     output_dir     : where to store results.
 324     metric_summary : summary object to add summaries to.
 325     N              : number of outputs to process.
 326   """
 327   if N >= 0:
 328     outputs = outputs[:N]
 329   N = len(outputs)
 330 
 331   plt.set_cmap('gray')
 332   fig, axes = utils.subplot(plt, (N, outputs[0][1].shape[0]), (5,5))
 333   axes = axes.ravel()[::-1].tolist()
 334   for i in range(N):
 335     locs, orig_maps, goal_loc = outputs[i]
 336     is_semantic = np.isnan(goal_loc[0,0,1])
 337     for j in range(orig_maps.shape[0]):
 338       ax = axes.pop();
 339       ax.plot(locs[j,0,0], locs[j,0,1], 'ys')
 340       # Plot one by one, so that they come in different colors.
 341       for k in range(goal_loc.shape[1]):
 342         if not is_semantic:
 343           ax.plot(goal_loc[j,k,0], goal_loc[j,k,1], 's')
 344       if False:
 345         ax.plot(locs[j,:,0], locs[j,:,1], 'r.', ms=3)
 346         ax.imshow(orig_maps[j,0,:,:,0], origin='lower')
 347         ax.set_axis_off();
 348       else:
 349         ax.scatter(locs[j,:,0], locs[j,:,1], c=np.arange(locs.shape[1]),
 350                    cmap='jet', s=10, lw=0)
 351         ax.imshow(orig_maps[j,0,:,:,0], origin='lower', vmin=-1.0, vmax=2.0)
 352         if not is_semantic:
 353           xymin = np.minimum(np.min(goal_loc[j,:,:], axis=0), np.min(locs[j,:,:], axis=0))
 354           xymax = np.maximum(np.max(goal_loc[j,:,:], axis=0), np.max(locs[j,:,:], axis=0))
 355         else:
 356           xymin = np.min(locs[j,:,:], axis=0)
 357           xymax = np.max(locs[j,:,:], axis=0)
 358         xy1 = (xymax+xymin)/2. - np.maximum(np.max(xymax-xymin), 12)
 359         xy2 = (xymax+xymin)/2. + np.maximum(np.max(xymax-xymin), 12)
 360         ax.set_xlim([xy1[0], xy2[0]])
 361         ax.set_ylim([xy1[1], xy2[1]])
 362         ax.set_axis_off()
 363   file_name = os.path.join(output_dir, 'trajectory_{:d}.png'.format(global_step))
 364   with fu.fopen(file_name, 'w') as f:
 365     fig.savefig(f, bbox_inches='tight', transparent=True, pad_inches=0)
 366   plt.close(fig)
 367   return None

 369 def add_default_summaries(mode, arop_full_summary_iters, summarize_ops,
 370                           summarize_names, to_aggregate, action_prob_op,
 371                           input_tensors, scope_name):
 372   assert(mode == 'train' or mode == 'val' or mode == 'test'), \
 373     'add_default_summaries mode is neither train or val or test.'
 374   
 375   s_ops = tf_utils.get_default_summary_ops()
 376   
 377   if mode == 'train':
 378     s_ops.summary_ops, s_ops.print_summary_ops, additional_return_ops, \
 379     arop_summary_iters, arop_eval_fns = tf_utils.simple_summaries(
 380             summarize_ops, summarize_names, mode, to_aggregate=False,
 381             scope_name=scope_name)
 382     s_ops.additional_return_ops += additional_return_ops
 383     s_ops.arop_summary_iters += arop_summary_iters
 384     s_ops.arop_eval_fns += arop_eval_fns
 385   elif mode == 'val':
 386     s_ops.summary_ops, s_ops.print_summary_ops, additional_return_ops, \
 387     arop_summary_iters, arop_eval_fns = tf_utils.simple_summaries(
 388             summarize_ops, summarize_names, mode, to_aggregate=to_aggregate,
 389             scope_name=scope_name)
 390     s_ops.additional_return_ops += additional_return_ops
 391     s_ops.arop_summary_iters += arop_summary_iters
 392     s_ops.arop_eval_fns += arop_eval_fns
 393   
 394   elif mode == 'test':
 395     s_ops.summary_ops, s_ops.print_summary_ops, additional_return_ops, \
 396     arop_summary_iters, arop_eval_fns = tf_utils.simple_summaries(
 397         [], [], mode, to_aggregate=[], scope_name=scope_name)
 398     s_ops.additional_return_ops += additional_return_ops
 399     s_ops.arop_summary_iters += arop_summary_iters
 400     s_ops.arop_eval_fns += arop_eval_fns
 401 
 402   
 403   if mode == 'val':
 404     arop = s_ops.additional_return_ops
 405     arop += [[action_prob_op, input_tensors['train']['action']]]
 406     arop += [[input_tensors['step']['loc_on_map'],
 407               input_tensors['common']['goal_loc'],
 408               input_tensors['step']['gt_dist_to_goal']]]
 409     arop += [[input_tensors['step']['loc_on_map'],
 410               input_tensors['common']['orig_maps'],
 411               input_tensors['common']['goal_loc']]]
 412     s_ops.arop_summary_iters += [-1, arop_full_summary_iters,
 413                                  arop_full_summary_iters]
 414     s_ops.arop_eval_fns += [eval_ap, eval_dist, plot_trajectories]
 415   
 416   elif mode == 'test':
 417     arop = s_ops.additional_return_ops
 418     arop += [[input_tensors['step']['loc_on_map'],
 419               input_tensors['common']['goal_loc'],
 420               input_tensors['step']['gt_dist_to_goal']]]
 421     arop += [[input_tensors['step']['gt_dist_to_goal']]]
 422     arop += [[input_tensors['step']['loc_on_map'],
 423               input_tensors['common']['goal_loc'],
 424               input_tensors['step']['gt_dist_to_goal'],
 425               input_tensors['step']['node_ids'],
 426               input_tensors['step']['perturbs']]]
 427     arop += [[input_tensors['step']['loc_on_map'],
 428               input_tensors['common']['orig_maps'],
 429               input_tensors['common']['goal_loc']]]
 430     s_ops.arop_summary_iters += [-1, -1, -1, arop_full_summary_iters]
 431     s_ops.arop_eval_fns += [eval_dist, save_d_at_t, save_all,
 432                             plot_trajectories]
 433   return s_ops
