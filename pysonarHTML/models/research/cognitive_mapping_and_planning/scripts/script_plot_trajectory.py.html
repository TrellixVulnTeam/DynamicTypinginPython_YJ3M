  61 def _load_trajectory():
  62   base_dir = FLAGS.base_dir
  63   config_name = FLAGS.config_name+_get_suffix_str()
  64 
  65   dir_name = os.path.join(base_dir, FLAGS.type, config_name)
  66   logging.info('Waiting for snapshot in directory %s.', dir_name)
  67   last_checkpoint = slim.evaluation.wait_for_new_checkpoint(dir_name, None)
  68   checkpoint_iter = int(os.path.basename(last_checkpoint).split('-')[1])
  69 
  70   # Load the distances.
  71   a = utils.load_variables(os.path.join(dir_name, 'bench_on_'+FLAGS.imset,
  72                                         'all_locs_at_t_{:d}.pkl'.format(checkpoint_iter)))
  73   return a

  75 def _compute_hardness():
  76   # Load the stanford data to compute the hardness.
  77   if FLAGS.type == '':
  78     args = sna.get_args_for_config(FLAGS.config_name+'+bench_'+FLAGS.imset)
  79   else:
  80     args = sna.get_args_for_config(FLAGS.type+'.'+FLAGS.config_name+'+bench_'+FLAGS.imset)
  81 
  82   args.navtask.logdir = None
  83   R = lambda: nav_env.get_multiplexer_class(args.navtask, 0)
  84   R = R()
  85 
  86   rng_data = [np.random.RandomState(0), np.random.RandomState(0)]
  87 
  88   # Sample a room.
  89   h_dists = []
  90   gt_dists = []
  91   for i in range(250):
  92     e = R.sample_env(rng_data)
  93     nodes = e.task.nodes
  94 
  95     # Initialize the agent.
  96     init_env_state = e.reset(rng_data)
  97 
  98     gt_dist_to_goal = [e.episode.dist_to_goal[0][j][s]
  99                        for j, s in enumerate(e.episode.start_node_ids)]
 100 
 101     for j in range(args.navtask.task_params.batch_size):
 102       start_node_id = e.episode.start_node_ids[j]
 103       end_node_id =e.episode.goal_node_ids[0][j]
 104       h_dist = graph_utils.heuristic_fn_vec(
 105           nodes[[start_node_id],:], nodes[[end_node_id], :],
 106           n_ori=args.navtask.task_params.n_ori,
 107           step_size=args.navtask.task_params.step_size)[0][0]
 108       gt_dist = e.episode.dist_to_goal[0][j][start_node_id]
 109       h_dists.append(h_dist)
 110       gt_dists.append(gt_dist)
 111 
 112   h_dists = np.array(h_dists)
 113   gt_dists = np.array(gt_dists)
 114   e = R.sample_env([np.random.RandomState(0), np.random.RandomState(0)])
 115   input = e.get_common_data()
 116   orig_maps = input['orig_maps'][0,0,:,:,0]
 117   return h_dists, gt_dists, orig_maps

 119 def plot_trajectory_first_person(dt, orig_maps, out_dir):
 120   out_dir = os.path.join(out_dir, FLAGS.config_name+_get_suffix_str(),
 121                          FLAGS.imset)
 122   fu.makedirs(out_dir)
 123 
 124   # Load the model so that we can render.
 125   plt.set_cmap('gray')
 126   samples_per_action = 8; wait_at_action = 0;
 127 
 128   Writer = animation.writers['mencoder']
 129   writer = Writer(fps=3*(samples_per_action+wait_at_action),
 130                   metadata=dict(artist='anonymous'), bitrate=1800)
 131 
 132   args = sna.get_args_for_config(FLAGS.config_name + '+bench_'+FLAGS.imset)
 133   args.navtask.logdir = None
 134   navtask_ = copy.deepcopy(args.navtask)
 135   navtask_.camera_param.modalities = ['rgb']
 136   navtask_.task_params.modalities = ['rgb']
 137   sz = 512
 138   navtask_.camera_param.height = sz
 139   navtask_.camera_param.width = sz
 140   navtask_.task_params.img_height = sz
 141   navtask_.task_params.img_width = sz
 142   R = lambda: nav_env.get_multiplexer_class(navtask_, 0)
 143   R = R()
 144   b = R.buildings[0]
 145 
 146   f = [0 for _ in range(wait_at_action)] + \
 147       [float(_)/samples_per_action for _ in range(samples_per_action)];
 148 
 149   # Generate things for it to render.
 150   inds_to_do = []
 151   inds_to_do += [1, 4, 10] #1291, 1268, 1273, 1289, 1302, 1426, 1413, 1449, 1399, 1390]
 152 
 153   for i in inds_to_do:
 154     fig = plt.figure(figsize=(10,8))
 155     gs = GridSpec(3,4)
 156     gs.update(wspace=0.05, hspace=0.05, left=0.0, top=0.97, right=1.0, bottom=0.)
 157     ax = fig.add_subplot(gs[:,:-1])
 158     ax1 = fig.add_subplot(gs[0,-1])
 159     ax2 = fig.add_subplot(gs[1,-1])
 160     ax3 = fig.add_subplot(gs[2,-1])
 161     axes = [ax, ax1, ax2, ax3]
 162     # ax = fig.add_subplot(gs[:,:])
 163     # axes = [ax]
 164     for ax in axes:
 165       ax.set_axis_off()
 166 
 167     node_ids = dt['all_node_ids'][i, :, 0]*1
 168     # Prune so that last node is not repeated more than 3 times?
 169     if np.all(node_ids[-4:] == node_ids[-1]):
 170       while node_ids[-4] == node_ids[-1]:
 171         node_ids = node_ids[:-1]
 172     num_steps = np.minimum(FLAGS.num_steps, len(node_ids))
 173 
 174     xyt = b.to_actual_xyt_vec(b.task.nodes[node_ids])
 175     xyt_diff = xyt[1:,:] - xyt[:-1:,:]
 176     xyt_diff[:,2] = np.mod(xyt_diff[:,2], 4)
 177     ind = np.where(xyt_diff[:,2] == 3)[0]
 178     xyt_diff[ind, 2] = -1
 179     xyt_diff = np.expand_dims(xyt_diff, axis=1)
 180     to_cat = [xyt_diff*_ for _ in f]
 181     perturbs_all = np.concatenate(to_cat, axis=1)
 182     perturbs_all = np.concatenate([perturbs_all, np.zeros_like(perturbs_all[:,:,:1])], axis=2)
 183     node_ids_all = np.expand_dims(node_ids, axis=1)*1
 184     node_ids_all = np.concatenate([node_ids_all for _ in f], axis=1)
 185     node_ids_all = np.reshape(node_ids_all[:-1,:], -1)
 186     perturbs_all = np.reshape(perturbs_all, [-1, 4])
 187     imgs = b.render_nodes(b.task.nodes[node_ids_all,:], perturb=perturbs_all)
 188 
 189     # Get action at each node.
 190     actions = []
 191     _, action_to_nodes = b.get_feasible_actions(node_ids)
 192     for j in range(num_steps-1):
 193       action_to_node = action_to_nodes[j]
 194       node_to_action = dict(zip(action_to_node.values(), action_to_node.keys()))
 195       actions.append(node_to_action[node_ids[j+1]])
 196 
 197     def init_fn():
 198       return fig,
 199     gt_dist_to_goal = []
 200 
 201     # Render trajectories.
 202     def worker(j):
 203       # Plot the image.
 204       step_number = j/(samples_per_action + wait_at_action)
 205       img = imgs[j]; ax = axes[0]; ax.clear(); ax.set_axis_off();
 206       img = img.astype(np.uint8); ax.imshow(img);
 207       tt = ax.set_title(
 208           "First Person View\n" +
 209           "Top corners show diagnostics (distance, agents' action) not input to agent.",
 210           fontsize=12)
 211       plt.setp(tt, color='white')
 212 
 213       # Distance to goal.
 214       t = 'Dist to Goal:\n{:2d} steps'.format(int(dt['all_d_at_t'][i, step_number]))
 215       t = ax.text(0.01, 0.99, t,
 216           horizontalalignment='left',
 217           verticalalignment='top',
 218           fontsize=20, color='red',
 219           transform=ax.transAxes, alpha=1.0)
 220       t.set_bbox(dict(color='white', alpha=0.85, pad=-0.1))
 221 
 222       # Action to take.
 223       action_latex = ['$\odot$ ', '$\curvearrowright$ ', '$\curvearrowleft$ ', r'$\Uparrow$ ']
 224       t = ax.text(0.99, 0.99, action_latex[actions[step_number]],
 225           horizontalalignment='right',
 226           verticalalignment='top',
 227           fontsize=40, color='green',
 228           transform=ax.transAxes, alpha=1.0)
 229       t.set_bbox(dict(color='white', alpha=0.85, pad=-0.1))
 230 
 231 
 232       # Plot the map top view.
 233       ax = axes[-1]
 234       if j == 0:
 235         # Plot the map
 236         locs = dt['all_locs'][i,:num_steps,:]
 237         goal_loc = dt['all_goal_locs'][i,:,:]
 238         xymin = np.minimum(np.min(goal_loc, axis=0), np.min(locs, axis=0))
 239         xymax = np.maximum(np.max(goal_loc, axis=0), np.max(locs, axis=0))
 240         xy1 = (xymax+xymin)/2. - 0.7*np.maximum(np.max(xymax-xymin), 24)
 241         xy2 = (xymax+xymin)/2. + 0.7*np.maximum(np.max(xymax-xymin), 24)
 242 
 243         ax.set_axis_on()
 244         ax.patch.set_facecolor((0.333, 0.333, 0.333))
 245         ax.set_xticks([]); ax.set_yticks([]);
 246         ax.imshow(orig_maps, origin='lower', vmin=-1.0, vmax=2.0)
 247         ax.plot(goal_loc[:,0], goal_loc[:,1], 'g*', markersize=12)
 248 
 249         locs = dt['all_locs'][i,:1,:]
 250         ax.plot(locs[:,0], locs[:,1], 'b.', markersize=12)
 251 
 252         ax.set_xlim([xy1[0], xy2[0]])
 253         ax.set_ylim([xy1[1], xy2[1]])
 254 
 255       locs = dt['all_locs'][i,step_number,:]
 256       locs = np.expand_dims(locs, axis=0)
 257       ax.plot(locs[:,0], locs[:,1], 'r.', alpha=1.0, linewidth=0, markersize=4)
 258       tt = ax.set_title('Trajectory in topview', fontsize=14)
 259       plt.setp(tt, color='white')
 260       return fig,
 261 
 262     line_ani = animation.FuncAnimation(fig, worker,
 263                                        (num_steps-1)*(wait_at_action+samples_per_action),
 264                                        interval=500, blit=True, init_func=init_fn)
 265     tmp_file_name = 'tmp.mp4'
 266     line_ani.save(tmp_file_name, writer=writer, savefig_kwargs={'facecolor':'black'})
 267     out_file_name = os.path.join(out_dir, 'vis_{:04d}.mp4'.format(i))
 268     print(out_file_name)

 270     if fu.exists(out_file_name):
 271       gfile.Remove(out_file_name)
 272     gfile.Copy(tmp_file_name, out_file_name)
 273     gfile.Remove(tmp_file_name)
 274     plt.close(fig)

 276 def plot_trajectory(dt, hardness, orig_maps, out_dir):
 277   out_dir = os.path.join(out_dir, FLAGS.config_name+_get_suffix_str(),
 278                          FLAGS.imset)
 279   fu.makedirs(out_dir)
 280   out_file = os.path.join(out_dir, 'all_locs_at_t.pkl')
 281   dt['hardness'] = hardness
 282   utils.save_variables(out_file, dt.values(), dt.keys(), overwrite=True)
 283 
 284   #Plot trajectories onto the maps
 285   plt.set_cmap('gray')
 286   for i in range(4000):
 287     goal_loc = dt['all_goal_locs'][i, :, :]
 288     locs = np.concatenate((dt['all_locs'][i,:,:],
 289                            dt['all_locs'][i,:,:]), axis=0)
 290     xymin = np.minimum(np.min(goal_loc, axis=0), np.min(locs, axis=0))
 291     xymax = np.maximum(np.max(goal_loc, axis=0), np.max(locs, axis=0))
 292     xy1 = (xymax+xymin)/2. - 1.*np.maximum(np.max(xymax-xymin), 24)
 293     xy2 = (xymax+xymin)/2. + 1.*np.maximum(np.max(xymax-xymin), 24)
 294 
 295     fig, ax = utils.tight_imshow_figure(plt, figsize=(6,6))
 296     ax.set_axis_on()
 297     ax.patch.set_facecolor((0.333, 0.333, 0.333))
 298     ax.set_xticks([])
 299     ax.set_yticks([])
 300 
 301     all_locs = dt['all_locs'][i,:,:]*1
 302     uniq = np.where(np.any(all_locs[1:,:] != all_locs[:-1,:], axis=1))[0]+1
 303     uniq = np.sort(uniq).tolist()
 304     uniq.insert(0,0)
 305     uniq = np.array(uniq)
 306     all_locs = all_locs[uniq, :]
 307 
 308     ax.plot(dt['all_locs'][i, 0, 0],
 309             dt['all_locs'][i, 0, 1], 'b.', markersize=24)
 310     ax.plot(dt['all_goal_locs'][i, 0, 0],
 311             dt['all_goal_locs'][i, 0, 1], 'g*', markersize=19)
 312     ax.plot(all_locs[:,0], all_locs[:,1], 'r', alpha=0.4, linewidth=2)
 313     ax.scatter(all_locs[:,0], all_locs[:,1],
 314                c=5+np.arange(all_locs.shape[0])*1./all_locs.shape[0],
 315                cmap='Reds', s=30, linewidth=0)
 316     ax.imshow(orig_maps, origin='lower', vmin=-1.0, vmax=2.0, aspect='equal')
 317     ax.set_xlim([xy1[0], xy2[0]])
 318     ax.set_ylim([xy1[1], xy2[1]])
 319 
 320     file_name = os.path.join(out_dir, 'trajectory_{:04d}.png'.format(i))
 321     print(file_name)
 322     with fu.fopen(file_name, 'w') as f:
 323       plt.savefig(f)
 324     plt.close(fig)

 327 def main(_):
 328   a = _load_trajectory()
 329   h_dists, gt_dists, orig_maps = _compute_hardness()
 330   hardness = 1.-h_dists*1./ gt_dists
 331 
 332   if FLAGS.top_view:
 333     plot_trajectory(a, hardness, orig_maps, out_dir=FLAGS.out_dir)
 334 
 335   if FLAGS.first_person:
 336     plot_trajectory_first_person(a, orig_maps, out_dir=FLAGS.out_dir)
 337 
 338 if __name__ == '__main__':
 339   app.run()
