  91 def _launcher(config_name, logdir):
  92   args = _setup_args(config_name, logdir)
  93 
  94   fu.makedirs(args.logdir)
  95 
  96   if args.control.train:
  97     _train(args)
  98 
  99   if args.control.test:
 100     _test(args)

 102 def get_args_for_config(config_name):
 103   configs = config_name.split('.')
 104   type = configs[0]
 105   config_name = '.'.join(configs[1:])
 106   if type == 'cmp':
 107     args = config_cmp.get_args_for_config(config_name)
 108     args.setup_to_run = cmp.setup_to_run
 109     args.setup_train_step_kwargs = cmp.setup_train_step_kwargs
 110 
 111   elif type == 'bl':
 112     args = config_vision_baseline.get_args_for_config(config_name)
 113     args.setup_to_run = vision_baseline_lstm.setup_to_run
 114     args.setup_train_step_kwargs = vision_baseline_lstm.setup_train_step_kwargs
 115 
 116   else:
 117     logging.fatal('Unknown type: {:s}'.format(type))
 118   return args
 119 
 120 def _setup_args(config_name, logdir):
 121   args = get_args_for_config(config_name)
 122   args.solver.num_workers = FLAGS.num_workers
 123   args.solver.task = FLAGS.task
 124   args.solver.ps_tasks = FLAGS.ps_tasks
 125   args.solver.master = FLAGS.master
 126   args.solver.seed = FLAGS.solver_seed
 127   args.logdir = logdir
 128   args.navtask.logdir = None
 129   return args

 131 def _train(args):
 132   container_name = ""
 133 
 134   R = lambda: nav_env.get_multiplexer_class(args.navtask, args.solver.task)
 135   m = utils.Foo()
 136   m.tf_graph = tf.Graph()
 137 
 138   config = tf.ConfigProto()
 139   config.device_count['GPU'] = 1
 140 
 141   with m.tf_graph.as_default():
 142     with tf.device(tf.train.replica_device_setter(args.solver.ps_tasks,
 143                                           merge_devices=True)):
 144       with tf.container(container_name):
 145         m = args.setup_to_run(m, args, is_training=True,
 146                              batch_norm_is_training=True, summary_mode='train')
 147 
 148         train_step_kwargs = args.setup_train_step_kwargs(
 149             m, R(), os.path.join(args.logdir, 'train'), rng_seed=args.solver.task,
 150             is_chief=args.solver.task==0,
 151             num_steps=args.navtask.task_params.num_steps*args.navtask.task_params.num_goals, iters=1,
 152             train_display_interval=args.summary.display_interval,
 153             dagger_sample_bn_false=args.arch.dagger_sample_bn_false)
 154 
 155         delay_start = (args.solver.task*(args.solver.task+1))/2 * FLAGS.delay_start_iters
 156         logging.error('delaying start for task %d by %d steps.',
 157                       args.solver.task, delay_start)
 158 
 159         additional_args = {}
 160         final_loss = slim.learning.train(
 161             train_op=m.train_op,
 162             logdir=args.logdir,
 163             master=args.solver.master,
 164             is_chief=args.solver.task == 0,
 165             number_of_steps=args.solver.max_steps,
 166             train_step_fn=tf_utils.train_step_custom_online_sampling,
 167             train_step_kwargs=train_step_kwargs,
 168             global_step=m.global_step_op,
 169             init_op=m.init_op,
 170             init_fn=m.init_fn,
 171             sync_optimizer=m.sync_optimizer,
 172             saver=m.saver_op,
 173             startup_delay_steps=delay_start,
 174             summary_op=None, session_config=config, **additional_args)
 175 
 176 def _test(args):
 177   args.solver.master = ''
 178   container_name = ""
 179   checkpoint_dir = os.path.join(format(args.logdir))
 180   logging.error('Checkpoint_dir: %s', args.logdir)
 181 
 182   config = tf.ConfigProto();
 183   config.device_count['GPU'] = 1;
 184 
 185   m = utils.Foo()
 186   m.tf_graph = tf.Graph()
 187 
 188   rng_data_seed = 0; rng_action_seed = 0;
 189   R = lambda: nav_env.get_multiplexer_class(args.navtask, rng_data_seed)
 190   with m.tf_graph.as_default():
 191     with tf.container(container_name):
 192       m = args.setup_to_run(
 193         m, args, is_training=False,
 194         batch_norm_is_training=args.control.force_batchnorm_is_training_at_test,
 195         summary_mode=args.control.test_mode)
 196       train_step_kwargs = args.setup_train_step_kwargs(
 197         m, R(), os.path.join(args.logdir, args.control.test_name),
 198         rng_seed=rng_data_seed, is_chief=True,
 199         num_steps=args.navtask.task_params.num_steps*args.navtask.task_params.num_goals,
 200         iters=args.summary.test_iters, train_display_interval=None,
 201         dagger_sample_bn_false=args.arch.dagger_sample_bn_false)
 202 
 203       saver = slim.learning.tf_saver.Saver(variables.get_variables_to_restore())
 204 
 205       sv = slim.learning.supervisor.Supervisor(
 206           graph=ops.get_default_graph(), logdir=None, init_op=m.init_op,
 207           summary_op=None, summary_writer=None, global_step=None, saver=m.saver_op)
 208 
 209       last_checkpoint = None
 210       reported = False
 211       while True:
 212         last_checkpoint_ = None
 213         while last_checkpoint_ is None:
 214           last_checkpoint_ = slim.evaluation.wait_for_new_checkpoint(
 215             checkpoint_dir, last_checkpoint, seconds_to_sleep=10, timeout=60)
 216         if last_checkpoint_ is None: break
 217 
 218         last_checkpoint = last_checkpoint_
 219         checkpoint_iter = int(os.path.basename(last_checkpoint).split('-')[1])
 220 
 221         logging.info('Starting evaluation at %s using checkpoint %s.',
 222                      time.strftime('%Y-%m-%d-%H:%M:%S', time.localtime()),
 223                      last_checkpoint)
 224 
 225         if (args.control.only_eval_when_done == False or 
 226             checkpoint_iter >= args.solver.max_steps):
 227           start = time.time()
 228           logging.info('Starting evaluation at %s using checkpoint %s.', 
 229                        time.strftime('%Y-%m-%d-%H:%M:%S', time.localtime()),
 230                        last_checkpoint)
 231 
 232           with sv.managed_session(args.solver.master, config=config,
 233                                   start_standard_services=False) as sess:
 234             sess.run(m.init_op)
 235             sv.saver.restore(sess, last_checkpoint)
 236             sv.start_queue_runners(sess)
 237             if args.control.reset_rng_seed:
 238               train_step_kwargs['rng_data'] = [np.random.RandomState(rng_data_seed),
 239                                                np.random.RandomState(rng_data_seed)]
 240               train_step_kwargs['rng_action'] = np.random.RandomState(rng_action_seed)
 241             vals, _ = tf_utils.train_step_custom_online_sampling(
 242                 sess, None, m.global_step_op, train_step_kwargs,
 243                 mode=args.control.test_mode)
 244             should_stop = False
 245 
 246             if checkpoint_iter >= args.solver.max_steps: 
 247               should_stop = True
 248 
 249             if should_stop:
 250               break
 251 
 252 if __name__ == '__main__':
 253   app.run()
