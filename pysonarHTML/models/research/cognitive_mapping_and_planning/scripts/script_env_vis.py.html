  45 def get_args():
  46   navtask = nec.nav_env_base_config()
  47   navtask.task_params.type = 'rng_rejection_sampling_many'
  48   navtask.task_params.rejection_sampling_M = 2000
  49   navtask.task_params.min_dist = 10
  50   sz = FLAGS.image_size
  51   navtask.camera_param.fov = FLAGS.fov
  52   navtask.camera_param.height = sz
  53   navtask.camera_param.width = sz
  54   navtask.task_params.img_height = sz
  55   navtask.task_params.img_width = sz
  56 
  57   # navtask.task_params.semantic_task.class_map_names = ['chair', 'door', 'table']
  58   # navtask.task_params.type = 'to_nearest_obj_acc'
  59 
  60   logging.info('navtask: %s', navtask)
  61   return navtask

  63 def load_building(dataset_name, building_name):
  64   dataset = factory.get_dataset(dataset_name)
  65 
  66   navtask = get_args()
  67   cp = navtask.camera_param
  68   rgb_shader, d_shader = renderer.get_shaders(cp.modalities)
  69   r_obj = SwiftshaderRenderer()
  70   r_obj.init_display(width=cp.width, height=cp.height,
  71                      fov=cp.fov, z_near=cp.z_near, z_far=cp.z_far,
  72                      rgb_shader=rgb_shader, d_shader=d_shader)
  73   r_obj.clear_scene()
  74   b = VisualNavigationEnv(robot=navtask.robot, env=navtask.env,
  75                           task_params=navtask.task_params,
  76                           building_name=building_name, flip=False,
  77                           logdir=None, building_loader=dataset,
  78                           r_obj=r_obj)
  79   b.load_building_into_scene()
  80   b.set_building_visibility(False)
  81   return b

  83 def walk_through(b):
  84   # init agent at a random location in the environment.
  85   init_env_state = b.reset([np.random.RandomState(0), np.random.RandomState(0)])
  86 
  87   global current_node
  88   rng = np.random.RandomState(0)
  89   current_node = rng.choice(b.task.nodes.shape[0])
  90 
  91   root = tk.Tk()
  92   image = b.render_nodes(b.task.nodes[[current_node],:])[0]
  93   print(image.shape)
  94   image = image.astype(np.uint8)
  95   im = Image.fromarray(image)
  96   im = ImageTk.PhotoImage(im)
  97   panel = tk.Label(root, image=im)
  98 
  99   map_size = b.traversible.shape
 100   sc = np.max(map_size)/256.
 101   loc = np.array([[map_size[1]/2., map_size[0]/2.]])
 102   x_axis = np.zeros_like(loc); x_axis[:,1] = sc
 103   y_axis = np.zeros_like(loc); y_axis[:,0] = -sc
 104   cum_fs, cum_valid = nav_env.get_map_to_predict(loc, x_axis, y_axis,
 105                                                    map=b.traversible*1.,
 106                                                    map_size=256)
 107   cum_fs = cum_fs[0]
 108   cum_fs = cv2.applyColorMap((cum_fs*255).astype(np.uint8), cv2.COLORMAP_JET)
 109   im = Image.fromarray(cum_fs)
 110   im = ImageTk.PhotoImage(im)
 111   panel_overhead = tk.Label(root, image=im)
 112 
 113   def refresh():
 114     global current_node
 115     image = b.render_nodes(b.task.nodes[[current_node],:])[0]
 116     image = image.astype(np.uint8)
 117     im = Image.fromarray(image)
 118     im = ImageTk.PhotoImage(im)
 119     panel.configure(image=im)
 120     panel.image = im
 121 
 122   def left_key(event):
 123     global current_node
 124     current_node = b.take_action([current_node], [2], 1)[0][0]
 125     refresh()
 126 
 127   def up_key(event):
 128     global current_node
 129     current_node = b.take_action([current_node], [3], 1)[0][0]
 130     refresh()
 131 
 132   def right_key(event):
 133     global current_node
 134     current_node = b.take_action([current_node], [1], 1)[0][0]
 135     refresh()
 136 
 137   def quit(event):
 138     root.destroy()
 139 
 140   panel_overhead.grid(row=4, column=5, rowspan=1, columnspan=1,
 141                       sticky=tk.W+tk.E+tk.N+tk.S)
 142   panel.bind('<Left>', left_key)
 143   panel.bind('<Up>', up_key)
 144   panel.bind('<Right>', right_key)
 145   panel.bind('q', quit)
 146   panel.focus_set()
 147   panel.grid(row=0, column=0, rowspan=5, columnspan=5,
 148              sticky=tk.W+tk.E+tk.N+tk.S)
 149   root.mainloop()

 151 def simple_window():
 152   root = tk.Tk()
 153 
 154   image = np.zeros((128, 128, 3), dtype=np.uint8)
 155   image[32:96, 32:96, 0] = 255
 156   im = Image.fromarray(image)
 157   im = ImageTk.PhotoImage(im)
 158 
 159   image = np.zeros((128, 128, 3), dtype=np.uint8)
 160   image[32:96, 32:96, 1] = 255
 161   im2 = Image.fromarray(image)
 162   im2 = ImageTk.PhotoImage(im2)
 163 
 164   panel = tk.Label(root, image=im)
 165 
 166   def left_key(event):
 167     panel.configure(image=im2)
 168     panel.image = im2
 169 
 170   def quit(event):
 171     sys.exit()
 172 
 173   panel.bind('<Left>', left_key)
 174   panel.bind('<Up>', left_key)
 175   panel.bind('<Down>', left_key)
 176   panel.bind('q', quit)
 177   panel.focus_set()
 178   panel.pack(side = "bottom", fill = "both", expand = "yes")
 179   root.mainloop()
