  68 def main(_):
  69   args = config_distill.get_args_for_config(FLAGS.config_name)
  70   args.logdir = FLAGS.logdir
  71   args.solver.num_workers = FLAGS.num_workers
  72   args.solver.task = FLAGS.task

  74   args.solver.master = FLAGS.master

  76   args.buildinger.env_class = nav_env.MeshMapper

  79   R = nav_env.get_multiplexor_class(args.buildinger, args.solver.task)
  80   
  81   if False:
  82     pr = cProfile.Profile()
  83     pr.enable()
  84     rng = np.random.RandomState(0)

  86       b, instances_perturbs = R.sample_building(rng)

  90         img = inputs['imgs'][j,0,:,:,:3]*1
  91         img = (img).astype(np.uint8)

  93       print(inputs['imgs'].shape)

 100       gfile.MakeDirs(args.logdir)

 112         m = distill.setup_to_run(m, args, is_training=True,

 115         train_step_kwargs = distill.setup_train_step_kwargs_mesh(

 143       train_step_kwargs = distill.setup_train_step_kwargs_mesh(

 148       sv = slim.learning.supervisor.Supervisor(

 160         
 161         config = tf.ConfigProto()

 172               sess, None, m.global_step_op, train_step_kwargs, mode='val')
 173           if checkpoint_iter >= args.solver.max_steps:
