  32 def _get_semantic_maps(building_name, transform, map_, flip, cats):
  33   rooms = get_room_in_building(building_name)
  34   maps = []
  35   for cat in cats:
  36     maps.append(np.zeros((map_.size[1], map_.size[0])))
  37   
  38   for r in rooms:
  39     room = load_room(building_name, r, category_list=cats)
  40     classes = room['class_id']
  41     for i, cat in enumerate(cats):
  42       c_ind = cats.index(cat)
  43       ind = [_ for _, c in enumerate(classes) if c == c_ind]
  44       if len(ind) > 0:
  45         vs = [room['vertexs'][x]*1 for x in ind]
  46         vs = np.concatenate(vs, axis=0)
  47         if transform:
  48           vs = np.array([vs[:,1], vs[:,0], vs[:,2]]).T
  49           vs[:,0] = -vs[:,0]
  50           vs[:,1] += 4.20
  51           vs[:,0] += 6.20
  52         vs = vs*100.
  53         if flip:
  54           vs[:,1] = -vs[:,1]
  55         maps[i] = maps[i] + \
  56             mu._project_to_map(map_, vs, ignore_points_outside_map=True)
  57   return maps

  59 def _map_building_name(building_name):
  60   b = int(building_name.split('_')[0][4])
  61   out_name = 'Area_{:d}'.format(b)
  62   if b == 5:
  63     if int(building_name.split('_')[0][5]) == 1:
  64       transform = True
  65     else:
  66       transform = False
  67   else:
  68     transform = False
  69   return out_name, transform

  73           'door', 'floor', 'sofa', 'table', 'wall', 'window']

  76 def _write_map_files(b_in, b_out, transform):
  77   cats = get_categories()
  78 
  79   env = utils.Foo(padding=10, resolution=5, num_point_threshold=2,
  80                   valid_min=-10, valid_max=200, n_samples_per_face=200)
  81   robot = utils.Foo(radius=15, base=10, height=140, sensor_height=120,
  82                     camera_elevation_degree=-15)
  83   
  84   building_loader = factory.get_dataset('sbpd')
  85   for flip in [False, True]:
  86     b = nav_env.Building(b_out, robot, env, flip=flip,
  87                          building_loader=building_loader)
  88     logging.info("building_in: %s, building_out: %s, transform: %d", b_in,
  89                  b_out, transform)
  90     maps = _get_semantic_maps(b_in, transform, b.map, flip, cats)
  91     maps = np.transpose(np.array(maps), axes=[1,2,0])
  92 
  93     #  Load file from the cache.
  94     file_name = '{:s}_{:d}_{:d}_{:d}_{:d}_{:d}_{:d}.pkl'
  95     file_name = file_name.format(b.building_name, b.map.size[0], b.map.size[1],
  96                                  b.map.origin[0], b.map.origin[1],
  97                                  b.map.resolution, flip)
  98     out_file = os.path.join(DATA_DIR, 'processing', 'class-maps', file_name)
  99     logging.info('Writing semantic maps to %s.', out_file)
 100     save_variables(out_file, [maps, cats], ['maps', 'cats'], overwrite=True)

 102 def _transform_area5b(room_dimension):
 103   for a in room_dimension.keys():
 104     r = room_dimension[a]*1
 105     r[[0,1,3,4]] = r[[1,0,4,3]]
 106     r[[0,3]] = -r[[3,0]]
 107     r[[1,4]] += 4.20
 108     r[[0,3]] += 6.20
 109     room_dimension[a] = r
 110   return room_dimension

 112 def collect_room(building_name, room_name):
 113   room_dir = os.path.join(DATA_DIR, 'Stanford3dDataset_v1.2', building_name,
 114                           room_name, 'Annotations')
 115   files = glob.glob1(room_dir, '*.txt')
 116   files = sorted(files, key=lambda s: s.lower())
 117   vertexs = []; colors = [];
 118   for f in files:
 119     file_name = os.path.join(room_dir, f)
 120     logging.info('  %s', file_name)
 121     a = np.loadtxt(file_name)
 122     vertex = a[:,:3]*1.
 123     color = a[:,3:]*1
 124     color = color.astype(np.uint8)
 125     vertexs.append(vertex)
 126     colors.append(color)
 127   files = [f.split('.')[0] for f in files]
 128   out = {'vertexs': vertexs, 'colors': colors, 'names': files}
 129   return out

 131 def load_room(building_name, room_name, category_list=None):
 132   room = collect_room(building_name, room_name)
 133   room['building_name'] = building_name
 134   room['room_name']     = room_name
 135   instance_id = range(len(room['names']))
 136   room['instance_id'] = instance_id
 137   if category_list is not None:
 138     name = [r.split('_')[0] for r in room['names']]
 139     class_id = []
 140     for n in name:
 141       if n in category_list:
 142         class_id.append(category_list.index(n))
 143       else:
 144         class_id.append(len(category_list))
 145     room['class_id'] = class_id
 146     room['category_list'] = category_list
 147   return room

 149 def get_room_in_building(building_name):
 150   building_dir = os.path.join(DATA_DIR, 'Stanford3dDataset_v1.2', building_name)
 151   rn = os.listdir(building_dir)
 152   rn = [x for x in rn if os.path.isdir(os.path.join(building_dir, x))]
 153   rn = sorted(rn, key=lambda s: s.lower())
 154   return rn

 156 def write_room_dimensions(b_in, b_out, transform):
 157   rooms = get_room_in_building(b_in)
 158   room_dimension = {}
 159   for r in rooms:
 160     room = load_room(b_in, r, category_list=None)
 161     vertex = np.concatenate(room['vertexs'], axis=0)
 162     room_dimension[r] = np.concatenate((np.min(vertex, axis=0), np.max(vertex, axis=0)), axis=0)
 163   if transform == 1:
 164     room_dimension = _transform_area5b(room_dimension)
 165   
 166   out_file = os.path.join(DATA_DIR, 'processing', 'room-dimension', b_out+'.pkl')
 167   save_variables(out_file, [room_dimension], ['room_dimension'], overwrite=True)
