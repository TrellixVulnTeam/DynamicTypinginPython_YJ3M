  43 def look_at(obj_camera, point):
  44   loc_camera = obj_camera.location
  45   direction = point - loc_camera
  46   # point the cameras '-Z' and use its 'Y' as up
  47   rot_quat = direction.to_track_quat('-Z', 'Y')
  48 
  49   obj_camera.rotation_euler = rot_quat.to_euler()
  50 
  51 
  52 def roll_camera(obj_camera):
  53   roll_rotate = mathutils.Euler(
  54       (0, 0, random.random() * math.pi - math.pi * 0.5), 'XYZ')
  55   obj_camera.rotation_euler = (obj_camera.rotation_euler.to_matrix() *
  56       roll_rotate.to_matrix()).to_euler()

  59 def norm(x):
  60   return math.sqrt(x[0] * x[0] + x[1] * x[1] + x[2] * x[2])
  61 
  62 
  63 def normalize(x):
  64   n = norm(x)
  65   x[0] /= n
  66   x[1] /= n
  67   x[2] /= n
  68

  70 def random_top_sphere():
  71   xyz = [random.normalvariate(0, 1) for x in range(3)]
  72   normalize(xyz)
  73 
  74   if xyz[2] < 0:
  75     xyz[2] *= -1
  76   return xyz
  77 
  78 
  79 def perturb_sphere(loc, size):
  80   while True:
  81     xyz = [random.normalvariate(0, 1) for x in range(3)]
  82     normalize(xyz)
  83 
  84     nloc = [loc[i] + xyz[i] * random.random() * size for i in range(3)]
  85     normalize(nloc)
  86 
  87     if nloc[2] >= 0:
  88       return nloc
  89 
  90 
  91 def perturb(loc, size):
  92   while True:
  93     nloc = [loc[i] + random.random() * size * 2 - size for i in range(3)]
  94     if nloc[2] >= 0:
  95       return nloc
  96 
  97     bpy.ops.object.mode_set()

 100 def delete_all_objects():
 101   bpy.ops.object.select_by_type(type="MESH")
 102   bpy.ops.object.delete(use_global=False)

 105 def set_scene(render_size, fov, alpha=False):
 106   """Set up default scene properties."""
 107   delete_all_objects()
 108 
 109   cam = bpy.data.cameras["Camera"]
 110   cam.angle = fov * pi / 180
 111 
 112   light = bpy.data.objects["Lamp"]
 113   light.location = (0, 0, 1)
 114   look_at(light, Vector((0.0, 0, 0)))
 115   bpy.data.lamps['Lamp'].type = "HEMI"
 116   bpy.data.lamps['Lamp'].energy = 1
 117   bpy.data.lamps['Lamp'].use_specular = False
 118   bpy.data.lamps['Lamp'].use_diffuse = True
 119 
 120   bpy.context.scene.world.horizon_color = (
 121       BG_LUMINANCE, BG_LUMINANCE, BG_LUMINANCE)
 122 
 123   bpy.context.scene.render.resolution_x = render_size
 124   bpy.context.scene.render.resolution_y = render_size
 125   bpy.context.scene.render.resolution_percentage = 100
 126 
 127   bpy.context.scene.render.use_antialiasing = True
 128   bpy.context.scene.render.antialiasing_samples = '5'

 131 def get_modelview_matrix():
 132   cam = bpy.data.objects["Camera"]
 133   bpy.context.scene.update()
 134 
 135   # when apply to object with CV coordinate i.e. to_blender * obj
 136   # this gives object in blender coordinate
 137   to_blender = mathutils.Matrix(
 138       ((1., 0., 0., 0.),
 139        (0., 0., -1., 0.),
 140        (0., 1., 0., 0.),
 141        (0., 0., 0., 1.)))
 142   return cam.matrix_world.inverted() * to_blender

 145 def print_matrix(f, mat):
 146   for i in range(4):
 147     for j in range(4):
 148       f.write("%lf " % mat[i][j])
 149     f.write("\n")

 152 def mul(loc, v):
 153   return [loc[i] * v for i in range(3)]

 156 def merge_all():
 157   bpy.ops.object.select_by_type(type="MESH")
 158   bpy.context.scene.objects.active = bpy.context.selected_objects[0]
 159   bpy.ops.object.join()
 160   obj = bpy.context.scene.objects.active
 161   bpy.ops.object.origin_set(type="ORIGIN_CENTER_OF_MASS")
 162   return obj

 165 def insert_frame(obj, frame_number):
 166   obj.keyframe_insert(data_path="location", frame=frame_number)
 167   obj.keyframe_insert(data_path="rotation_euler", frame=frame_number)
 168   obj.keyframe_insert(data_path="scale", frame=frame_number)

 171 def render(output_prefix):
 172   bpy.context.scene.render.filepath = output_prefix
 173   bpy.context.scene.render.image_settings.file_format = "PNG"
 174   bpy.context.scene.render.alpha_mode = "TRANSPARENT"
 175   bpy.context.scene.render.image_settings.color_mode = "RGBA"
 176   bpy.ops.render.render(write_still=True, animation=True)

 179 def render_obj(
 180     obj_fn, save_dir, n, perturb_size, rotate=False, roll=False, scale=1.0):
 181 
 182   # Load object.
 183   bpy.ops.import_scene.obj(filepath=obj_fn)
 184   cur_obj = merge_all()
 185 
 186   scale = 2.0 / max(cur_obj.dimensions) * scale
 187   cur_obj.scale = (scale, scale, scale)
 188   # Using the center of mass as the origin doesn't really work, because Blender
 189   # assumes the object is a solid shell. This seems to generate better-looking
 190   # rotations.
 191 
 192   bpy.ops.object.origin_set(type='ORIGIN_GEOMETRY', center='BOUNDS')
 193 
 194   # bpy.ops.mesh.primitive_cube_add(location=(0, 0, 1))
 195   # cube = bpy.data.objects["Cube"]
 196   # cube.scale = (0.2, 0.2, 0.2)
 197 
 198   for polygon in cur_obj.data.polygons:
 199     polygon.use_smooth = True
 200 
 201   bpy.ops.object.select_all(action="DESELECT")
 202 
 203   camera = bpy.data.objects["Camera"]
 204 
 205   # os.system("mkdir " + save_dir)
 206   for i in range(n):
 207     fo = open(save_dir + "/%06d.txt" % i, "w")
 208     d = 30
 209     shift = 0.2
 210     if rotate:
 211       t = 1.0 * i / (n-1) * 2 * math.pi
 212       loc = [math.sin(t), math.cos(t), 1]
 213 
 214       normalize(loc)
 215       camera.location = mul(loc, d)
 216       look_at(camera, Vector((0.0, 0, 0)))
 217 
 218       print_matrix(fo, get_modelview_matrix())
 219       print_matrix(fo, get_modelview_matrix())
 220 
 221       insert_frame(camera, 2 * i)
 222       insert_frame(camera, 2 * i + 1)
 223 
 224     else:
 225       loc = random_top_sphere()
 226 
 227       camera.location = mul(loc, d)
 228       look_at(camera, Vector((0.0, 0, 0)))
 229 
 230       if roll:
 231         roll_camera(camera)
 232       camera.location = perturb(mul(loc, d), shift)
 233 
 234       print_matrix(fo, get_modelview_matrix())
 235       insert_frame(camera, 2 * i)
 236 
 237       if perturb_size > 0:
 238         loc = perturb_sphere(loc, perturb_size)
 239       else:
 240         loc = random_top_sphere()
 241 
 242       camera.location = mul(loc, d)
 243       look_at(camera, Vector((0.0, 0, 0)))
 244       if roll:
 245         roll_camera(camera)
 246       camera.location = perturb(mul(loc, d), shift)
 247 
 248       print_matrix(fo, get_modelview_matrix())
 249       insert_frame(camera, 2 * i + 1)
 250 
 251     fo.close()
 252 
 253   # Create a bunch of views of the object
 254   bpy.context.scene.frame_start = 0
 255   bpy.context.scene.frame_end = 2 * n - 1
 256 
 257   stem = os.path.join(save_dir, '######')
 258   render(stem)

 261 def main():
 262   parser = argparse.ArgumentParser()
 263   parser.add_argument('-m', '--model', dest='model',
 264                       required=True,
 265                       help='Path to model obj file.')
 266   parser.add_argument('-o', '--output_dir', dest='output_dir',
 267                       required=True,
 268                       help='Where to output files.')
 269   parser.add_argument('-s', '--output_size', dest='output_size',
 270                       required=True,
 271                       help='Width and height of output in pixels, e.g. 32x32.')
 272   parser.add_argument('-n', '--num_frames', dest='n', type=int,
 273                       required=True,
 274                       help='Number of frames to generate per clip.')
 275 
 276   parser.add_argument('-scale', '--scale', dest='scale', type=float,
 277                       help='object scaling', default=1)
 278 
 279   parser.add_argument('-perturb', '--perturb', dest='perturb', type=float,
 280                       help='sphere perturbation', default=0)
 281 
 282   parser.add_argument('-rotate', '--rotate', dest='rotate', action='store_true',
 283                       help='render rotating test set')
 284 
 285   parser.add_argument('-roll', '--roll', dest='roll', action='store_true',
 286                       help='add roll')
 287 
 288   parser.add_argument(
 289       '-fov', '--fov', dest='fov', type=float, required=True,
 290       help='field of view')
 291 
 292   if '--' not in sys.argv:
 293     parser.print_help()
 294     exit(1)
 295 
 296   argv = sys.argv[sys.argv.index('--') + 1:]
 297   args, _ = parser.parse_known_args(argv)
 298 
 299   random.seed(args.model + str(time.time()) + str(os.getpid()))
 300   # random.seed(0)
 301 
 302   set_scene(int(args.output_size), args.fov)
 303   render_obj(
 304       args.model, args.output_dir, args.n, args.perturb, args.rotate,
 305       args.roll, args.scale)
 306   exit()
 307
