  26 FLAGS = tf.app.flags.FLAGS
  27 tf.app.flags.DEFINE_integer('max_decode_steps', 1000000,
  28                             'Number of decoding steps.')
  29 tf.app.flags.DEFINE_integer('decode_batches_per_ckpt', 8000,
  30                             'Number of batches to decode before restoring next '
  31                             'checkpoint')
  32 
  33 DECODE_LOOP_DELAY_SECS = 60
  34 DECODE_IO_FLUSH_INTERVAL = 100
  35 
  36 
  37 class DecodeIO(object):
  38   """Writes the decoded and references to RKV files for Rouge score.
  39 
  40     See nlp/common/utils/internal/rkv_parser.py for detail about rkv file.
  41   """
  42 
  43   def __init__(self, outdir):
  44     self._cnt = 0
  45     self._outdir = outdir
  46     if not os.path.exists(self._outdir):
  47       os.mkdir(self._outdir)
  48     self._ref_file = None
  49     self._decode_file = None
  50 
  51   def Write(self, reference, decode):
  52     """Writes the reference and decoded outputs to RKV files.
  53 
  54     Args:
  55       reference: The human (correct) result.
  56       decode: The machine-generated result
  57     """
  58     self._ref_file.write('output=%s\n' % reference)
  59     self._decode_file.write('output=%s\n' % decode)
  60     self._cnt += 1
  61     if self._cnt % DECODE_IO_FLUSH_INTERVAL == 0:
  62       self._ref_file.flush()
  63       self._decode_file.flush()
  64 
  65   def ResetFiles(self):
  66     """Resets the output files. Must be called once before Write()."""
  67     if self._ref_file: self._ref_file.close()
  68     if self._decode_file: self._decode_file.close()
  69     timestamp = int(time.time())
  70     self._ref_file = open(
  71         os.path.join(self._outdir, 'ref%d'%timestamp), 'w')
  72     self._decode_file = open(
  73         os.path.join(self._outdir, 'decode%d'%timestamp), 'w')
  74

  76 class BSDecoder(object):
  77   """Beam search decoder."""
  78 
  79   def __init__(self, model, batch_reader, hps, vocab):
  80     """Beam search decoding.
  81 
  82     Args:
  83       model: The seq2seq attentional model.
  84       batch_reader: The batch data reader.
  85       hps: Hyperparamters.
  86       vocab: Vocabulary
  87     """
  88     self._model = model
  89     self._model.build_graph()
  90     self._batch_reader = batch_reader
  91     self._hps = hps
  92     self._vocab = vocab
  93     self._saver = tf.train.Saver()
  94     self._decode_io = DecodeIO(FLAGS.decode_dir)
  95 
  96   def DecodeLoop(self):
  97     """Decoding loop for long running process."""
  98     sess = tf.Session(config=tf.ConfigProto(allow_soft_placement=True))
  99     step = 0
 100     while step < FLAGS.max_decode_steps:
 101       time.sleep(DECODE_LOOP_DELAY_SECS)
 102       if not self._Decode(self._saver, sess):
 103         continue
 104       step += 1
 105 
 106   def _Decode(self, saver, sess):
 107     """Restore a checkpoint and decode it.
 108 
 109     Args:
 110       saver: Tensorflow checkpoint saver.
 111       sess: Tensorflow session.
 112     Returns:
 113       If success, returns true, otherwise, false.
 114     """
 115     ckpt_state = tf.train.get_checkpoint_state(FLAGS.log_root)
 116     if not (ckpt_state and ckpt_state.model_checkpoint_path):
 117       tf.logging.info('No model to decode yet at %s', FLAGS.log_root)
 118       return False
 119 
 120     tf.logging.info('checkpoint path %s', ckpt_state.model_checkpoint_path)
 121     ckpt_path = os.path.join(
 122         FLAGS.log_root, os.path.basename(ckpt_state.model_checkpoint_path))
 123     tf.logging.info('renamed checkpoint path %s', ckpt_path)
 124     saver.restore(sess, ckpt_path)
 125 
 126     self._decode_io.ResetFiles()
 127     for _ in xrange(FLAGS.decode_batches_per_ckpt):
 128       (article_batch, _, _, article_lens, _, _, origin_articles,
 129        origin_abstracts) = self._batch_reader.NextBatch()
 130       for i in xrange(self._hps.batch_size):
 131         bs = beam_search.BeamSearch(
 132             self._model, self._hps.batch_size,
 133             self._vocab.WordToId(data.SENTENCE_START),
 134             self._vocab.WordToId(data.SENTENCE_END),
 135             self._hps.dec_timesteps)
 136 
 137         article_batch_cp = article_batch.copy()
 138         article_batch_cp[:] = article_batch[i:i+1]
 139         article_lens_cp = article_lens.copy()
 140         article_lens_cp[:] = article_lens[i:i+1]
 141         best_beam = bs.BeamSearch(sess, article_batch_cp, article_lens_cp)[0]
 142         decode_output = [int(t) for t in best_beam.tokens[1:]]
 143         self._DecodeBatch(
 144             origin_articles[i], origin_abstracts[i], decode_output)
 145     return True
 146 
 147   def _DecodeBatch(self, article, abstract, output_ids):
 148     """Convert id to words and writing results.
 149 
 150     Args:
 151       article: The original article string.
 152       abstract: The human (correct) abstract string.
 153       output_ids: The abstract word ids output by machine.
 154     """
 155     decoded_output = ' '.join(data.Ids2Words(output_ids, self._vocab))
 156     end_p = decoded_output.find(data.SENTENCE_END, 0)
 157     if end_p != -1:
 158       decoded_output = decoded_output[:end_p]
 159     tf.logging.info('article:  %s', article)
 160     tf.logging.info('abstract: %s', abstract)
 161     tf.logging.info('decoded:  %s', decoded_output)
 162     self._decode_io.Write(abstract, decoded_output.strip())
