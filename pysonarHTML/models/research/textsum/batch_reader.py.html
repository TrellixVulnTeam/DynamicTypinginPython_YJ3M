  39 class Batcher(object):
  40   """Batch reader with shuffling and bucketing support."""

  42   def __init__(self, data_path, vocab, hps,
  43                article_key, abstract_key, max_article_sentences,
  44                max_abstract_sentences, bucketing=True, truncate_input=False):
  45     """Batcher constructor.
  46 
  47     Args:
  48       data_path: tf.Example filepattern.
  49       vocab: Vocabulary.
  50       hps: Seq2SeqAttention model hyperparameters.
  51       article_key: article feature key in tf.Example.
  52       abstract_key: abstract feature key in tf.Example.
  53       max_article_sentences: Max number of sentences used from article.
  54       max_abstract_sentences: Max number of sentences used from abstract.
  55       bucketing: Whether bucket articles of similar length into the same batch.
  56       truncate_input: Whether to truncate input that is too long. Alternative is
  57         to discard such examples.
  58     """
  59     self._data_path = data_path
  60     self._vocab = vocab
  61     self._hps = hps
  62     self._article_key = article_key
  63     self._abstract_key = abstract_key
  64     self._max_article_sentences = max_article_sentences
  65     self._max_abstract_sentences = max_abstract_sentences
  66     self._bucketing = bucketing
  67     self._truncate_input = truncate_input
  68     self._input_queue = Queue.Queue(QUEUE_NUM_BATCH * self._hps.batch_size)
  69     self._bucket_input_queue = Queue.Queue(QUEUE_NUM_BATCH)
  70     self._input_threads = []
  71     for _ in xrange(16):
  72       self._input_threads.append(Thread(target=self._FillInputQueue))
  73       self._input_threads[-1].daemon = True
  74       self._input_threads[-1].start()
  75     self._bucketing_threads = []
  76     for _ in xrange(4):
  77       self._bucketing_threads.append(Thread(target=self._FillBucketInputQueue))
  78       self._bucketing_threads[-1].daemon = True
  79       self._bucketing_threads[-1].start()
  80 
  81     self._watch_thread = Thread(target=self._WatchThreads)
  82     self._watch_thread.daemon = True
  83     self._watch_thread.start()
  84 
  85   def NextBatch(self):
  86     """Returns a batch of inputs for seq2seq attention model.
  87 
  88     Returns:
  89       enc_batch: A batch of encoder inputs [batch_size, hps.enc_timestamps].
  90       dec_batch: A batch of decoder inputs [batch_size, hps.dec_timestamps].
  91       target_batch: A batch of targets [batch_size, hps.dec_timestamps].
  92       enc_input_len: encoder input lengths of the batch.
  93       dec_input_len: decoder input lengths of the batch.
  94       loss_weights: weights for loss function, 1 if not padded, 0 if padded.
  95       origin_articles: original article words.
  96       origin_abstracts: original abstract words.
  97     """
  98     enc_batch = np.zeros(
  99         (self._hps.batch_size, self._hps.enc_timesteps), dtype=np.int32)
 100     enc_input_lens = np.zeros(
 101         (self._hps.batch_size), dtype=np.int32)
 102     dec_batch = np.zeros(
 103         (self._hps.batch_size, self._hps.dec_timesteps), dtype=np.int32)
 104     dec_output_lens = np.zeros(
 105         (self._hps.batch_size), dtype=np.int32)
 106     target_batch = np.zeros(
 107         (self._hps.batch_size, self._hps.dec_timesteps), dtype=np.int32)
 108     loss_weights = np.zeros(
 109         (self._hps.batch_size, self._hps.dec_timesteps), dtype=np.float32)
 110     origin_articles = ['None'] * self._hps.batch_size
 111     origin_abstracts = ['None'] * self._hps.batch_size
 112 
 113     buckets = self._bucket_input_queue.get()
 114     for i in xrange(self._hps.batch_size):
 115       (enc_inputs, dec_inputs, targets, enc_input_len, dec_output_len,
 116        article, abstract) = buckets[i]
 117 
 118       origin_articles[i] = article
 119       origin_abstracts[i] = abstract
 120       enc_input_lens[i] = enc_input_len
 121       dec_output_lens[i] = dec_output_len
 122       enc_batch[i, :] = enc_inputs[:]
 123       dec_batch[i, :] = dec_inputs[:]
 124       target_batch[i, :] = targets[:]
 125       for j in xrange(dec_output_len):
 126         loss_weights[i][j] = 1
 127     return (enc_batch, dec_batch, target_batch, enc_input_lens, dec_output_lens,
 128             loss_weights, origin_articles, origin_abstracts)

 130   def _FillInputQueue(self):
 131     """Fill input queue with ModelInput."""
 132     start_id = self._vocab.WordToId(data.SENTENCE_START)
 133     end_id = self._vocab.WordToId(data.SENTENCE_END)
 134     pad_id = self._vocab.WordToId(data.PAD_TOKEN)
 135     input_gen = self._TextGenerator(data.ExampleGen(self._data_path))
 136     while True:
 137       (article, abstract) = six.next(input_gen)
 138       article_sentences = [sent.strip() for sent in
 139                            data.ToSentences(article, include_token=False)]
 140       abstract_sentences = [sent.strip() for sent in
 141                             data.ToSentences(abstract, include_token=False)]
 142 
 143       enc_inputs = []
 144       # Use the <s> as the <GO> symbol for decoder inputs.
 145       dec_inputs = [start_id]
 146 
 147       # Convert first N sentences to word IDs, stripping existing <s> and </s>.
 148       for i in xrange(min(self._max_article_sentences,
 149                           len(article_sentences))):
 150         enc_inputs += data.GetWordIds(article_sentences[i], self._vocab)
 151       for i in xrange(min(self._max_abstract_sentences,
 152                           len(abstract_sentences))):
 153         dec_inputs += data.GetWordIds(abstract_sentences[i], self._vocab)
 154 
 155       # Filter out too-short input
 156       if (len(enc_inputs) < self._hps.min_input_len or
 157           len(dec_inputs) < self._hps.min_input_len):
 158         tf.logging.warning('Drop an example - too short.\nenc:%d\ndec:%d',
 159                            len(enc_inputs), len(dec_inputs))
 160         continue
 161 
 162       # If we're not truncating input, throw out too-long input
 163       if not self._truncate_input:
 164         if (len(enc_inputs) > self._hps.enc_timesteps or
 165             len(dec_inputs) > self._hps.dec_timesteps):
 166           tf.logging.warning('Drop an example - too long.\nenc:%d\ndec:%d',
 167                              len(enc_inputs), len(dec_inputs))
 168           continue
 169       # If we are truncating input, do so if necessary
 170       else:
 171         if len(enc_inputs) > self._hps.enc_timesteps:
 172           enc_inputs = enc_inputs[:self._hps.enc_timesteps]
 173         if len(dec_inputs) > self._hps.dec_timesteps:
 174           dec_inputs = dec_inputs[:self._hps.dec_timesteps]
 175 
 176       # targets is dec_inputs without <s> at beginning, plus </s> at end
 177       targets = dec_inputs[1:]
 178       targets.append(end_id)
 179 
 180       # Now len(enc_inputs) should be <= enc_timesteps, and
 181       # len(targets) = len(dec_inputs) should be <= dec_timesteps
 182 
 183       enc_input_len = len(enc_inputs)
 184       dec_output_len = len(targets)
 185 
 186       # Pad if necessary
 187       while len(enc_inputs) < self._hps.enc_timesteps:
 188         enc_inputs.append(pad_id)
 189       while len(dec_inputs) < self._hps.dec_timesteps:
 190         dec_inputs.append(end_id)
 191       while len(targets) < self._hps.dec_timesteps:
 192         targets.append(end_id)
 193 
 194       element = ModelInput(enc_inputs, dec_inputs, targets, enc_input_len,
 195                            dec_output_len, ' '.join(article_sentences),
 196                            ' '.join(abstract_sentences))
 197       self._input_queue.put(element)
 198 
 199   def _FillBucketInputQueue(self):
 200     """Fill bucketed batches into the bucket_input_queue."""
 201     while True:
 202       inputs = []
 203       for _ in xrange(self._hps.batch_size * BUCKET_CACHE_BATCH):
 204         inputs.append(self._input_queue.get())
 205       if self._bucketing:
 206         inputs = sorted(inputs, key=lambda inp: inp.enc_len)
 207 
 208       batches = []
 209       for i in xrange(0, len(inputs), self._hps.batch_size):
 210         batches.append(inputs[i:i+self._hps.batch_size])
 211       shuffle(batches)
 212       for b in batches:
 213         self._bucket_input_queue.put(b)
 214 
 215   def _WatchThreads(self):
 216     """Watch the daemon input threads and restart if dead."""
 217     while True:
 218       time.sleep(60)
 219       input_threads = []
 220       for t in self._input_threads:
 221         if t.is_alive():
 222           input_threads.append(t)
 223         else:
 224           tf.logging.error('Found input thread dead.')
 225           new_t = Thread(target=self._FillInputQueue)
 226           input_threads.append(new_t)
 227           input_threads[-1].daemon = True
 228           input_threads[-1].start()
 229       self._input_threads = input_threads
 230 
 231       bucketing_threads = []
 232       for t in self._bucketing_threads:
 233         if t.is_alive():
 234           bucketing_threads.append(t)
 235         else:
 236           tf.logging.error('Found bucketing thread dead.')
 237           new_t = Thread(target=self._FillBucketInputQueue)
 238           bucketing_threads.append(new_t)
 239           bucketing_threads[-1].daemon = True
 240           bucketing_threads[-1].start()
 241       self._bucketing_threads = bucketing_threads

 243   def _TextGenerator(self, example_gen):
 244     """Generates article and abstract text from tf.Example."""
 245     while True:
 246       e = six.next(example_gen)
 247       try:
 248         article_text = self._GetExFeatureText(e, self._article_key)
 249         abstract_text = self._GetExFeatureText(e, self._abstract_key)
 250       except ValueError:
 251         tf.logging.error('Failed to get article or abstract from example')
 252         continue
 253 
 254       yield (article_text, abstract_text)
 255 
 256   def _GetExFeatureText(self, ex, key):
 257     """Extract text for a feature from td.Example.
 258 
 259     Args:
 260       ex: tf.Example.
 261       key: key of the feature to be extracted.
 262     Returns:
 263       feature: a feature text extracted.
 264     """
 265     return ex.features.feature[key].bytes_list.value[0]
