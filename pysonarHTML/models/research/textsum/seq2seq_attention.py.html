  68 def _RunningAvgLoss(loss, running_avg_loss, summary_writer, step, decay=0.999):
  69   """Calculate the running average of losses."""
  70   if running_avg_loss == 0:
  71     running_avg_loss = loss
  72   else:
  73     running_avg_loss = running_avg_loss * decay + (1 - decay) * loss
  74   running_avg_loss = min(running_avg_loss, 12)
  75   loss_sum = tf.Summary()
  76   loss_sum.value.add(tag='running_avg_loss', simple_value=running_avg_loss)
  77   summary_writer.add_summary(loss_sum, step)
  78   sys.stdout.write('running_avg_loss: %f\n' % running_avg_loss)
  79   return running_avg_loss

  82 def _Train(model, data_batcher):
  83   """Runs model training."""
  84   with tf.device('/cpu:0'):
  85     model.build_graph()
  86     saver = tf.train.Saver()
  87     # Train dir is different from log_root to avoid summary directory
  88     # conflict with Supervisor.
  89     summary_writer = tf.summary.FileWriter(FLAGS.train_dir)
  90     sv = tf.train.Supervisor(logdir=FLAGS.log_root,
  91                              is_chief=True,
  92                              saver=saver,
  93                              summary_op=None,
  94                              save_summaries_secs=60,
  95                              save_model_secs=FLAGS.checkpoint_secs,
  96                              global_step=model.global_step)
  97     sess = sv.prepare_or_wait_for_session(config=tf.ConfigProto(
  98         allow_soft_placement=True))
  99     running_avg_loss = 0
 100     step = 0
 101     while not sv.should_stop() and step < FLAGS.max_run_steps:
 102       (article_batch, abstract_batch, targets, article_lens, abstract_lens,
 103        loss_weights, _, _) = data_batcher.NextBatch()
 104       (_, summaries, loss, train_step) = model.run_train_step(
 105           sess, article_batch, abstract_batch, targets, article_lens,
 106           abstract_lens, loss_weights)
 107 
 108       summary_writer.add_summary(summaries, train_step)
 109       running_avg_loss = _RunningAvgLoss(
 110           running_avg_loss, loss, summary_writer, train_step)
 111       step += 1
 112       if step % 100 == 0:
 113         summary_writer.flush()
 114     sv.Stop()
 115     return running_avg_loss

 118 def _Eval(model, data_batcher, vocab=None):
 119   """Runs model eval."""
 120   model.build_graph()
 121   saver = tf.train.Saver()
 122   summary_writer = tf.summary.FileWriter(FLAGS.eval_dir)
 123   sess = tf.Session(config=tf.ConfigProto(allow_soft_placement=True))
 124   running_avg_loss = 0
 125   step = 0
 126   while True:
 127     time.sleep(FLAGS.eval_interval_secs)
 128     try:
 129       ckpt_state = tf.train.get_checkpoint_state(FLAGS.log_root)
 130     except tf.errors.OutOfRangeError as e:
 131       tf.logging.error('Cannot restore checkpoint: %s', e)
 132       continue
 133 
 134     if not (ckpt_state and ckpt_state.model_checkpoint_path):
 135       tf.logging.info('No model to eval yet at %s', FLAGS.train_dir)
 136       continue
 137 
 138     tf.logging.info('Loading checkpoint %s', ckpt_state.model_checkpoint_path)
 139     saver.restore(sess, ckpt_state.model_checkpoint_path)
 140 
 141     (article_batch, abstract_batch, targets, article_lens, abstract_lens,
 142      loss_weights, _, _) = data_batcher.NextBatch()
 143     (summaries, loss, train_step) = model.run_eval_step(
 144         sess, article_batch, abstract_batch, targets, article_lens,
 145         abstract_lens, loss_weights)
 146     tf.logging.info(
 147         'article:  %s',
 148         ' '.join(data.Ids2Words(article_batch[0][:].tolist(), vocab)))
 149     tf.logging.info(
 150         'abstract: %s',
 151         ' '.join(data.Ids2Words(abstract_batch[0][:].tolist(), vocab)))
 152 
 153     summary_writer.add_summary(summaries, train_step)
 154     running_avg_loss = _RunningAvgLoss(
 155         running_avg_loss, loss, summary_writer, train_step)
 156     if step % 100 == 0:
 157       summary_writer.flush()

 160 def main(unused_argv):
 161   vocab = data.Vocab(FLAGS.vocab_path, 1000000)
 162   # Check for presence of required special tokens.
 163   assert vocab.CheckVocab(data.PAD_TOKEN) > 0
 164   assert vocab.CheckVocab(data.UNKNOWN_TOKEN) >= 0
 165   assert vocab.CheckVocab(data.SENTENCE_START) > 0
 166   assert vocab.CheckVocab(data.SENTENCE_END) > 0
 167 
 168   batch_size = 4
 169   if FLAGS.mode == 'decode':
 170     batch_size = FLAGS.beam_size
 171 
 172   hps = seq2seq_attention_model.HParams(
 173       mode=FLAGS.mode,  # train, eval, decode
 174       min_lr=0.01,  # min learning rate.
 175       lr=0.15,  # learning rate
 176       batch_size=batch_size,
 177       enc_layers=4,
 178       enc_timesteps=120,
 179       dec_timesteps=30,
 180       min_input_len=2,  # discard articles/summaries < than this
 181       num_hidden=256,  # for rnn cell
 182       emb_dim=128,  # If 0, don't use embedding
 183       max_grad_norm=2,
 184       num_softmax_samples=4096)  # If 0, no sampled softmax.
 185 
 186   batcher = batch_reader.Batcher(
 187       FLAGS.data_path, vocab, hps, FLAGS.article_key,
 188       FLAGS.abstract_key, FLAGS.max_article_sentences,
 189       FLAGS.max_abstract_sentences, bucketing=FLAGS.use_bucketing,
 190       truncate_input=FLAGS.truncate_input)
 191   tf.set_random_seed(FLAGS.random_seed)
 192 
 193   if hps.mode == 'train':
 194     model = seq2seq_attention_model.Seq2SeqAttentionModel(
 195         hps, vocab, num_gpus=FLAGS.num_gpus)
 196     _Train(model, batcher)
 197   elif hps.mode == 'eval':
 198     model = seq2seq_attention_model.Seq2SeqAttentionModel(
 199         hps, vocab, num_gpus=FLAGS.num_gpus)
 200     _Eval(model, batcher, vocab=vocab)
 201   elif hps.mode == 'decode':
 202     decode_mdl_hps = hps
 203     # Only need to restore the 1st step and reuse it since
 204     # we keep and feed in state for each step's output.
 205     decode_mdl_hps = hps._replace(dec_timesteps=1)
 206     model = seq2seq_attention_model.Seq2SeqAttentionModel(
 207         decode_mdl_hps, vocab, num_gpus=FLAGS.num_gpus)
 208     decoder = seq2seq_attention_decode.BSDecoder(model, batcher, hps, vocab)
 209     decoder.DecodeLoop()
 210 
 211 
 212 if __name__ == '__main__':
 213   tf.app.run()
