  16 """Beam search module.
  17 
  18 Beam search takes the top K results from the model, predicts the K results for
  19 each of the previous K result, getting K*K results. Pick the top K results from
  20 K*K results, and start over again until certain number of results are fully
  21 decoded.
  22 """
  23 
  24 from six.moves import xrange
  25 import tensorflow as tf
  26 
  27 FLAGS = tf.flags.FLAGS
  28 tf.flags.DEFINE_bool('normalize_by_length', True, 'Whether to normalize')
  29 
  30 
  31 class Hypothesis(object):
  32   """Defines a hypothesis during beam search."""
  33 
  34   def __init__(self, tokens, log_prob, state):
  35     """Hypothesis constructor.
  36 
  37     Args:
  38       tokens: start tokens for decoding.
  39       log_prob: log prob of the start tokens, usually 1.
  40       state: decoder initial states.
  41     """
  42     self.tokens = tokens
  43     self.log_prob = log_prob
  44     self.state = state
  45 
  46   def Extend(self, token, log_prob, new_state):
  47     """Extend the hypothesis with result from latest step.
  48 
  49     Args:
  50       token: latest token from decoding.
  51       log_prob: log prob of the latest decoded tokens.
  52       new_state: decoder output state. Fed to the decoder for next step.
  53     Returns:
  54       New Hypothesis with the results from latest step.
  55     """
  56     return Hypothesis(self.tokens + [token], self.log_prob + log_prob,
  57                       new_state)
  58 
  59   @property
  60   def latest_token(self):
  61     return self.tokens[-1]
  62 
  63   def __str__(self):
  64     return ('Hypothesis(log prob = %.4f, tokens = %s)' % (self.log_prob,
  65                                                           self.tokens))

  68 class BeamSearch(object):
  69   """Beam search."""
  70 
  71   def __init__(self, model, beam_size, start_token, end_token, max_steps):
  72     """Creates BeamSearch object.
  73 
  74     Args:
  75       model: Seq2SeqAttentionModel.
  76       beam_size: int.
  77       start_token: int, id of the token to start decoding with
  78       end_token: int, id of the token that completes an hypothesis
  79       max_steps: int, upper limit on the size of the hypothesis
  80     """
  81     self._model = model
  82     self._beam_size = beam_size
  83     self._start_token = start_token
  84     self._end_token = end_token
  85     self._max_steps = max_steps
  86 
  87   def BeamSearch(self, sess, enc_inputs, enc_seqlen):
  88     """Performs beam search for decoding.
  89 
  90     Args:
  91       sess: tf.Session, session
  92       enc_inputs: ndarray of shape (enc_length, 1), the document ids to encode
  93       enc_seqlen: ndarray of shape (1), the length of the sequnce
  94 
  95     Returns:
  96       hyps: list of Hypothesis, the best hypotheses found by beam search,
  97           ordered by score
  98     """
  99 
 100     # Run the encoder and extract the outputs and final state.
 101     enc_top_states, dec_in_state = self._model.encode_top_state(
 102         sess, enc_inputs, enc_seqlen)
 103     # Replicate the initial states K times for the first step.
 104     hyps = [Hypothesis([self._start_token], 0.0, dec_in_state)
 105            ] * self._beam_size
 106     results = []
 107 
 108     steps = 0
 109     while steps < self._max_steps and len(results) < self._beam_size:
 110       latest_tokens = [h.latest_token for h in hyps]
 111       states = [h.state for h in hyps]
 112 
 113       topk_ids, topk_log_probs, new_states = self._model.decode_topk(
 114           sess, latest_tokens, enc_top_states, states)
 115       # Extend each hypothesis.
 116       all_hyps = []
 117       # The first step takes the best K results from first hyps. Following
 118       # steps take the best K results from K*K hyps.
 119       num_beam_source = 1 if steps == 0 else len(hyps)
 120       for i in xrange(num_beam_source):
 121         h, ns = hyps[i], new_states[i]
 122         for j in xrange(self._beam_size*2):
 123           all_hyps.append(h.Extend(topk_ids[i, j], topk_log_probs[i, j], ns))
 124 
 125       # Filter and collect any hypotheses that have the end token.
 126       hyps = []
 127       for h in self._BestHyps(all_hyps):
 128         if h.latest_token == self._end_token:
 129           # Pull the hypothesis off the beam if the end token is reached.
 130           results.append(h)
 131         else:
 132           # Otherwise continue to the extend the hypothesis.
 133           hyps.append(h)
 134         if len(hyps) == self._beam_size or len(results) == self._beam_size:
 135           break
 136 
 137       steps += 1
 138 
 139     if steps == self._max_steps:
 140       results.extend(hyps)
 141 
 142     return self._BestHyps(results)
 143 
 144   def _BestHyps(self, hyps):
 145     """Sort the hyps based on log probs and length.
 146 
 147     Args:
 148       hyps: A list of hypothesis.
 149     Returns:
 150       hyps: A list of sorted hypothesis in reverse log_prob order.
 151     """
 152     # This length normalization is only effective for the final results.
 153     if FLAGS.normalize_by_length:
 154       return sorted(hyps, key=lambda h: h.log_prob/len(h.tokens), reverse=True)
 155     else:
 156       return sorted(hyps, key=lambda h: h.log_prob, reverse=True)
