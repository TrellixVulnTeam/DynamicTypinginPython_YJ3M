  32 def _extract_argmax_and_embed(embedding, output_projection=None,
  33                               update_embedding=True):
  34   """Get a loop_function that extracts the previous symbol and embeds it.
  35 
  36   Args:
  37     embedding: embedding tensor for symbols.
  38     output_projection: None or a pair (W, B). If provided, each fed previous
  39       output will first be multiplied by W and added B.
  40     update_embedding: Boolean; if False, the gradients will not propagate
  41       through the embeddings.
  42 
  43   Returns:
  44     A loop function.
  45   """
  46   def loop_function(prev, _):
  47     """function that feed previous model output rather than ground truth."""
  48     if output_projection is not None:
  49       prev = tf.nn.xw_plus_b(
  50           prev, output_projection[0], output_projection[1])
  51     prev_symbol = tf.argmax(prev, 1)
  52     # Note that gradients will not propagate through the second parameter of
  53     # embedding_lookup.
  54     emb_prev = tf.nn.embedding_lookup(embedding, prev_symbol)
  55     if not update_embedding:
  56       emb_prev = tf.stop_gradient(emb_prev)
  57     return emb_prev
  58   return loop_function

  61 class Seq2SeqAttentionModel(object):
  62   """Wrapper for Tensorflow model graph for text sum vectors."""
  63 
  64   def __init__(self, hps, vocab, num_gpus=0):
  65     self._hps = hps
  66     self._vocab = vocab
  67     self._num_gpus = num_gpus
  68     self._cur_gpu = 0
  69 
  70   def run_train_step(self, sess, article_batch, abstract_batch, targets,
  71                      article_lens, abstract_lens, loss_weights):
  72     to_return = [self._train_op, self._summaries, self._loss, self.global_step]
  73     return sess.run(to_return,
  74                     feed_dict={self._articles: article_batch,
  75                                self._abstracts: abstract_batch,
  76                                self._targets: targets,
  77                                self._article_lens: article_lens,
  78                                self._abstract_lens: abstract_lens,
  79                                self._loss_weights: loss_weights})
  80 
  81   def run_eval_step(self, sess, article_batch, abstract_batch, targets,
  82                     article_lens, abstract_lens, loss_weights):
  83     to_return = [self._summaries, self._loss, self.global_step]
  84     return sess.run(to_return,
  85                     feed_dict={self._articles: article_batch,
  86                                self._abstracts: abstract_batch,
  87                                self._targets: targets,
  88                                self._article_lens: article_lens,
  89                                self._abstract_lens: abstract_lens,
  90                                self._loss_weights: loss_weights})
  91 
  92   def run_decode_step(self, sess, article_batch, abstract_batch, targets,
  93                       article_lens, abstract_lens, loss_weights):
  94     to_return = [self._outputs, self.global_step]
  95     return sess.run(to_return,
  96                     feed_dict={self._articles: article_batch,
  97                                self._abstracts: abstract_batch,
  98                                self._targets: targets,
  99                                self._article_lens: article_lens,
 100                                self._abstract_lens: abstract_lens,
 101                                self._loss_weights: loss_weights})
 102 
 103   def _next_device(self):
 104     """Round robin the gpu device. (Reserve last gpu for expensive op)."""
 105     if self._num_gpus == 0:
 106       return ''
 107     dev = '/gpu:%d' % self._cur_gpu
 108     if self._num_gpus > 1:
 109       self._cur_gpu = (self._cur_gpu + 1) % (self._num_gpus-1)
 110     return dev
 111 
 112   def _get_gpu(self, gpu_id):
 113     if self._num_gpus <= 0 or gpu_id >= self._num_gpus:
 114       return ''
 115     return '/gpu:%d' % gpu_id

 117   def _add_placeholders(self):
 118     """Inputs to be fed to the graph."""
 119     hps = self._hps
 120     self._articles = tf.placeholder(tf.int32,
 121                                     [hps.batch_size, hps.enc_timesteps],
 122                                     name='articles')
 123     self._abstracts = tf.placeholder(tf.int32,
 124                                      [hps.batch_size, hps.dec_timesteps],
 125                                      name='abstracts')
 126     self._targets = tf.placeholder(tf.int32,
 127                                    [hps.batch_size, hps.dec_timesteps],
 128                                    name='targets')
 129     self._article_lens = tf.placeholder(tf.int32, [hps.batch_size],
 130                                         name='article_lens')
 131     self._abstract_lens = tf.placeholder(tf.int32, [hps.batch_size],
 132                                          name='abstract_lens')
 133     self._loss_weights = tf.placeholder(tf.float32,
 134                                         [hps.batch_size, hps.dec_timesteps],
 135                                         name='loss_weights')
 136 
 137   def _add_seq2seq(self):
 138     hps = self._hps
 139     vsize = self._vocab.NumIds()
 140 
 141     with tf.variable_scope('seq2seq'):
 142       encoder_inputs = tf.unstack(tf.transpose(self._articles))
 143       decoder_inputs = tf.unstack(tf.transpose(self._abstracts))
 144       targets = tf.unstack(tf.transpose(self._targets))
 145       loss_weights = tf.unstack(tf.transpose(self._loss_weights))
 146       article_lens = self._article_lens
 147 
 148       # Embedding shared by the input and outputs.
 149       with tf.variable_scope('embedding'), tf.device('/cpu:0'):
 150         embedding = tf.get_variable(
 151             'embedding', [vsize, hps.emb_dim], dtype=tf.float32,
 152             initializer=tf.truncated_normal_initializer(stddev=1e-4))
 153         emb_encoder_inputs = [tf.nn.embedding_lookup(embedding, x)
 154                               for x in encoder_inputs]
 155         emb_decoder_inputs = [tf.nn.embedding_lookup(embedding, x)
 156                               for x in decoder_inputs]
 157 
 158       for layer_i in xrange(hps.enc_layers):
 159         with tf.variable_scope('encoder%d'%layer_i), tf.device(
 160             self._next_device()):
 161           cell_fw = tf.contrib.rnn.LSTMCell(
 162               hps.num_hidden,
 163               initializer=tf.random_uniform_initializer(-0.1, 0.1, seed=123),
 164               state_is_tuple=False)
 165           cell_bw = tf.contrib.rnn.LSTMCell(
 166               hps.num_hidden,
 167               initializer=tf.random_uniform_initializer(-0.1, 0.1, seed=113),
 168               state_is_tuple=False)
 169           (emb_encoder_inputs, fw_state, _) = tf.contrib.rnn.static_bidirectional_rnn(
 170               cell_fw, cell_bw, emb_encoder_inputs, dtype=tf.float32,
 171               sequence_length=article_lens)
 172       encoder_outputs = emb_encoder_inputs
 173 
 174       with tf.variable_scope('output_projection'):
 175         w = tf.get_variable(
 176             'w', [hps.num_hidden, vsize], dtype=tf.float32,
 177             initializer=tf.truncated_normal_initializer(stddev=1e-4))
 178         w_t = tf.transpose(w)
 179         v = tf.get_variable(
 180             'v', [vsize], dtype=tf.float32,
 181             initializer=tf.truncated_normal_initializer(stddev=1e-4))
 182 
 183       with tf.variable_scope('decoder'), tf.device(self._next_device()):
 184         # When decoding, use model output from the previous step
 185         # for the next step.
 186         loop_function = None
 187         if hps.mode == 'decode':
 188           loop_function = _extract_argmax_and_embed(
 189               embedding, (w, v), update_embedding=False)
 190 
 191         cell = tf.contrib.rnn.LSTMCell(
 192             hps.num_hidden,
 193             initializer=tf.random_uniform_initializer(-0.1, 0.1, seed=113),
 194             state_is_tuple=False)
 195 
 196         encoder_outputs = [tf.reshape(x, [hps.batch_size, 1, 2*hps.num_hidden])
 197                            for x in encoder_outputs]
 198         self._enc_top_states = tf.concat(axis=1, values=encoder_outputs)
 199         self._dec_in_state = fw_state
 200         # During decoding, follow up _dec_in_state are fed from beam_search.
 201         # dec_out_state are stored by beam_search for next step feeding.
 202         initial_state_attention = (hps.mode == 'decode')
 203         decoder_outputs, self._dec_out_state = tf.contrib.legacy_seq2seq.attention_decoder(
 204             emb_decoder_inputs, self._dec_in_state, self._enc_top_states,
 205             cell, num_heads=1, loop_function=loop_function,
 206             initial_state_attention=initial_state_attention)
 207 
 208       with tf.variable_scope('output'), tf.device(self._next_device()):
 209         model_outputs = []
 210         for i in xrange(len(decoder_outputs)):
 211           if i > 0:
 212             tf.get_variable_scope().reuse_variables()
 213           model_outputs.append(
 214               tf.nn.xw_plus_b(decoder_outputs[i], w, v))
 215 
 216       if hps.mode == 'decode':
 217         with tf.variable_scope('decode_output'), tf.device('/cpu:0'):
 218           best_outputs = [tf.argmax(x, 1) for x in model_outputs]
 219           tf.logging.info('best_outputs%s', best_outputs[0].get_shape())
 220           self._outputs = tf.concat(
 221               axis=1, values=[tf.reshape(x, [hps.batch_size, 1]) for x in best_outputs])
 222 
 223           self._topk_log_probs, self._topk_ids = tf.nn.top_k(
 224               tf.log(tf.nn.softmax(model_outputs[-1])), hps.batch_size*2)
 225 
 226       with tf.variable_scope('loss'), tf.device(self._next_device()):
 227         def sampled_loss_func(inputs, labels):
 228           with tf.device('/cpu:0'):  # Try gpu.
 229             labels = tf.reshape(labels, [-1, 1])
 230             return tf.nn.sampled_softmax_loss(
 231                 weights=w_t, biases=v, labels=labels, inputs=inputs,
 232                 num_sampled=hps.num_softmax_samples, num_classes=vsize)
 233 
 234         if hps.num_softmax_samples != 0 and hps.mode == 'train':
 235           self._loss = seq2seq_lib.sampled_sequence_loss(
 236               decoder_outputs, targets, loss_weights, sampled_loss_func)
 237         else:
 238           self._loss = tf.contrib.legacy_seq2seq.sequence_loss(
 239               model_outputs, targets, loss_weights)
 240         tf.summary.scalar('loss', tf.minimum(12.0, self._loss))
 241 
 242   def _add_train_op(self):
 243     """Sets self._train_op, op to run for training."""
 244     hps = self._hps
 245 
 246     self._lr_rate = tf.maximum(
 247         hps.min_lr,  # min_lr_rate.
 248         tf.train.exponential_decay(hps.lr, self.global_step, 30000, 0.98))
 249 
 250     tvars = tf.trainable_variables()
 251     with tf.device(self._get_gpu(self._num_gpus-1)):
 252       grads, global_norm = tf.clip_by_global_norm(
 253           tf.gradients(self._loss, tvars), hps.max_grad_norm)
 254     tf.summary.scalar('global_norm', global_norm)
 255     optimizer = tf.train.GradientDescentOptimizer(self._lr_rate)
 256     tf.summary.scalar('learning rate', self._lr_rate)
 257     self._train_op = optimizer.apply_gradients(
 258         zip(grads, tvars), global_step=self.global_step, name='train_step')
 259 
 260   def encode_top_state(self, sess, enc_inputs, enc_len):
 261     """Return the top states from encoder for decoder.
 262 
 263     Args:
 264       sess: tensorflow session.
 265       enc_inputs: encoder inputs of shape [batch_size, enc_timesteps].
 266       enc_len: encoder input length of shape [batch_size]
 267     Returns:
 268       enc_top_states: The top level encoder states.
 269       dec_in_state: The decoder layer initial state.
 270     """
 271     results = sess.run([self._enc_top_states, self._dec_in_state],
 272                        feed_dict={self._articles: enc_inputs,
 273                                   self._article_lens: enc_len})
 274     return results[0], results[1][0]
 275 
 276   def decode_topk(self, sess, latest_tokens, enc_top_states, dec_init_states):
 277     """Return the topK results and new decoder states."""
 278     feed = {
 279         self._enc_top_states: enc_top_states,
 280         self._dec_in_state:
 281             np.squeeze(np.array(dec_init_states)),
 282         self._abstracts:
 283             np.transpose(np.array([latest_tokens])),
 284         self._abstract_lens: np.ones([len(dec_init_states)], np.int32)}
 285 
 286     results = sess.run(
 287         [self._topk_ids, self._topk_log_probs, self._dec_out_state],
 288         feed_dict=feed)
 289 
 290     ids, probs, states = results[0], results[1], results[2]
 291     new_states = [s for s in states]
 292     return ids, probs, new_states
 293 
 294   def build_graph(self):
 295     self._add_placeholders()
 296     self._add_seq2seq()
 297     self.global_step = tf.Variable(0, name='global_step', trainable=False)
 298     if self._hps.mode == 'train':
 299       self._add_train_op()
 300     self._summaries = tf.summary.merge_all()
