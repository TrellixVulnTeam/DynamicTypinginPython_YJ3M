  37 class Vocab(object):
  38   """Vocabulary class for mapping words and ids."""
  39 
  40   def __init__(self, vocab_file, max_size):
  41     self._word_to_id = {}
  42     self._id_to_word = {}
  43     self._count = 0
  44 
  45     with open(vocab_file, 'r') as vocab_f:
  46       for line in vocab_f:
  47         pieces = line.split()
  48         if len(pieces) != 2:
  49           sys.stderr.write('Bad line: %s\n' % line)
  50           continue
  51         if pieces[0] in self._word_to_id:
  52           raise ValueError('Duplicated word: %s.' % pieces[0])
  53         self._word_to_id[pieces[0]] = self._count
  54         self._id_to_word[self._count] = pieces[0]
  55         self._count += 1
  56         if self._count > max_size:
  57           raise ValueError('Too many words: >%d.' % max_size)
  58 
  59   def CheckVocab(self, word):
  60     if word not in self._word_to_id:
  61       return None
  62     return self._word_to_id[word]
  63   
  64   def WordToId(self, word):
  65     if word not in self._word_to_id:
  66       return self._word_to_id[UNKNOWN_TOKEN]
  67     return self._word_to_id[word]
  68 
  69   def IdToWord(self, word_id):
  70     if word_id not in self._id_to_word:
  71       raise ValueError('id not found in vocab: %d.' % word_id)
  72     return self._id_to_word[word_id]
  73 
  74   def NumIds(self):
  75     return self._count

  78 def ExampleGen(data_path, num_epochs=None):
  79   """Generates tf.Examples from path of data files.
  80 
  81     Binary data format: <length><blob>. <length> represents the byte size
  82     of <blob>. <blob> is serialized tf.Example proto. The tf.Example contains
  83     the tokenized article text and summary.
  84 
  85   Args:
  86     data_path: path to tf.Example data files.
  87     num_epochs: Number of times to go through the data. None means infinite.
  88 
  89   Yields:
  90     Deserialized tf.Example.
  91 
  92   If there are multiple files specified, they accessed in a random order.
  93   """
  94   epoch = 0
  95   while True:
  96     if num_epochs is not None and epoch >= num_epochs:
  97       break
  98     filelist = glob.glob(data_path)
  99     assert filelist, 'Empty filelist.'
 100     random.shuffle(filelist)
 101     for f in filelist:
 102       reader = open(f, 'rb')
 103       while True:
 104         len_bytes = reader.read(8)
 105         if not len_bytes: break
 106         str_len = struct.unpack('q', len_bytes)[0]
 107         example_str = struct.unpack('%ds' % str_len, reader.read(str_len))[0]
 108         yield example_pb2.Example.FromString(example_str)
 109 
 110     epoch += 1

 113 def Pad(ids, pad_id, length):
 114   """Pad or trim list to len length.
 115 
 116   Args:
 117     ids: list of ints to pad
 118     pad_id: what to pad with
 119     length: length to pad or trim to
 120 
 121   Returns:
 122     ids trimmed or padded with pad_id
 123   """
 124   assert pad_id is not None
 125   assert length is not None
 126 
 127   if len(ids) < length:
 128     a = [pad_id] * (length - len(ids))
 129     return ids + a
 130   else:
 131     return ids[:length]

 134 def GetWordIds(text, vocab, pad_len=None, pad_id=None):
 135   """Get ids corresponding to words in text.
 136 
 137   Assumes tokens separated by space.
 138 
 139   Args:
 140     text: a string
 141     vocab: TextVocabularyFile object
 142     pad_len: int, length to pad to
 143     pad_id: int, word id for pad symbol
 144 
 145   Returns:
 146     A list of ints representing word ids.
 147   """
 148   ids = []
 149   for w in text.split():
 150     i = vocab.WordToId(w)
 151     if i >= 0:
 152       ids.append(i)
 153     else:
 154       ids.append(vocab.WordToId(UNKNOWN_TOKEN))
 155   if pad_len is not None:
 156     return Pad(ids, pad_id, pad_len)
 157   return ids

 160 def Ids2Words(ids_list, vocab):
 161   """Get words from ids.
 162 
 163   Args:
 164     ids_list: list of int32
 165     vocab: TextVocabulary object
 166 
 167   Returns:
 168     List of words corresponding to ids.
 169   """
 170   assert isinstance(ids_list, list), '%s  is not a list' % ids_list
 171   return [vocab.IdToWord(i) for i in ids_list]

 174 def SnippetGen(text, start_tok, end_tok, inclusive=True):
 175   """Generates consecutive snippets between start and end tokens.
 176 
 177   Args:
 178     text: a string
 179     start_tok: a string denoting the start of snippets
 180     end_tok: a string denoting the end of snippets
 181     inclusive: Whether include the tokens in the returned snippets.
 182 
 183   Yields:
 184     String snippets
 185   """
 186   cur = 0
 187   while True:
 188     try:
 189       start_p = text.index(start_tok, cur)
 190       end_p = text.index(end_tok, start_p + 1)
 191       cur = end_p + len(end_tok)
 192       if inclusive:
 193         yield text[start_p:cur]
 194       else:
 195         yield text[start_p+len(start_tok):end_p]
 196     except ValueError as e:
 197       raise StopIteration('no more snippets in text: %s' % e)

 204 def ToSentences(paragraph, include_token=True):
 205   """Takes tokens of a paragraph and returns list of sentences.
 206 
 207   Args:
 208     paragraph: string, text of paragraph
 209     include_token: Whether include the sentence separation tokens result.
 210 
 211   Returns:
 212     List of sentence strings.
 213   """
 214   s_gen = SnippetGen(paragraph, SENTENCE_START, SENTENCE_END, include_token)
 215   return [s for s in s_gen]
