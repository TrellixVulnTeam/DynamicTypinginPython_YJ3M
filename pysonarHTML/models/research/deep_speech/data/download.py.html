  54 def download_and_extract(directory, url):
  55   """Download and extract the given split of dataset.
  56 
  57   Args:
  58     directory: the directory where to extract the tarball.
  59     url: the url to download the data file.
  60   """
  61 
  62   if not tf.gfile.Exists(directory):
  63     tf.gfile.MakeDirs(directory)
  64 
  65   _, tar_filepath = tempfile.mkstemp(suffix=".tar.gz")
  66 
  67   try:
  68     tf.logging.info("Downloading %s to %s" % (url, tar_filepath))
  69 
  70     def _progress(count, block_size, total_size):
  71       sys.stdout.write("\r>> Downloading {} {:.1f}%".format(
  72           tar_filepath, 100.0 * count * block_size / total_size))
  73       sys.stdout.flush()
  74 
  75     urllib.request.urlretrieve(url, tar_filepath, _progress)
  76     print()
  77     statinfo = os.stat(tar_filepath)
  78     tf.logging.info(
  79         "Successfully downloaded %s, size(bytes): %d" % (url, statinfo.st_size))
  80     with tarfile.open(tar_filepath, "r") as tar:
  81       tar.extractall(directory)
  82   finally:
  83     tf.gfile.Remove(tar_filepath)

  86 def convert_audio_and_split_transcript(input_dir, source_name, target_name,
  87                                        output_dir, output_file):
  88   """Convert FLAC to WAV and split the transcript.
  89 
  90   For audio file, convert the format from FLAC to WAV using the sox.Transformer
  91   library.
  92   For transcripts, each line contains the sequence id and the corresponding
  93   transcript (separated by space):
  94   Input data format: seq-id transcript_of_seq-id
  95   For example:
  96    1-2-0 transcript_of_1-2-0.flac
  97    1-2-1 transcript_of_1-2-1.flac
  98    ...
  99 
 100   Each sequence id has a corresponding .flac file.
 101   Parse the transcript file and generate a new csv file which has three columns:
 102   "wav_filename": the absolute path to a wav file.
 103   "wav_filesize": the size of the corresponding wav file.
 104   "transcript": the transcript for this audio segement.
 105 
 106   Args:
 107     input_dir: the directory which holds the input dataset.
 108     source_name: the name of the specified dataset. e.g. test-clean
 109     target_name: the directory name for the newly generated audio files.
 110                  e.g. test-clean-wav
 111     output_dir: the directory to place the newly generated csv files.
 112     output_file: the name of the newly generated csv file. e.g. test-clean.csv
 113   """
 114 
 115   tf.logging.info("Preprocessing audio and transcript for %s" % source_name)
 116   source_dir = os.path.join(input_dir, source_name)
 117   target_dir = os.path.join(input_dir, target_name)
 118 
 119   if not tf.gfile.Exists(target_dir):
 120     tf.gfile.MakeDirs(target_dir)
 121 
 122   files = []
 123   tfm = Transformer()
 124   # Convert all FLAC file into WAV format. At the same time, generate the csv
 125   # file.
 126   for root, _, filenames in tf.gfile.Walk(source_dir):
 127     for filename in fnmatch.filter(filenames, "*.trans.txt"):
 128       trans_file = os.path.join(root, filename)
 129       with codecs.open(trans_file, "r", "utf-8") as fin:
 130         for line in fin:
 131           seqid, transcript = line.split(" ", 1)
 132           # We do a encode-decode transformation here because the output type
 133           # of encode is a bytes object, we need convert it to string.
 134           transcript = unicodedata.normalize("NFKD", transcript).encode(
 135               "ascii", "ignore").decode("ascii", "ignore").strip().lower()
 136 
 137           # Convert FLAC to WAV.
 138           flac_file = os.path.join(root, seqid + ".flac")
 139           wav_file = os.path.join(target_dir, seqid + ".wav")
 140           if not tf.gfile.Exists(wav_file):
 141             tfm.build(flac_file, wav_file)
 142           wav_filesize = os.path.getsize(wav_file)
 143 
 144           files.append((os.path.abspath(wav_file), wav_filesize, transcript))
 145 
 146   # Write to CSV file which contains three columns:
 147   # "wav_filename", "wav_filesize", "transcript".
 148   csv_file_path = os.path.join(output_dir, output_file)
 149   df = pandas.DataFrame(
 150       data=files, columns=["wav_filename", "wav_filesize", "transcript"])
 151   df.to_csv(csv_file_path, index=False, sep="\t")
 152   tf.logging.info("Successfully generated csv file {}".format(csv_file_path))

 155 def download_and_process_datasets(directory, datasets):
 156   """Download and pre-process the specified list of LibriSpeech dataset.
 157 
 158   Args:
 159     directory: the directory to put all the downloaded and preprocessed data.
 160     datasets: list of dataset names that will be downloaded and processed.
 161   """
 162 
 163   tf.logging.info("Preparing LibriSpeech dataset: {}".format(
 164       ",".join(datasets)))
 165   for dataset in datasets:
 166     tf.logging.info("Preparing dataset %s", dataset)
 167     dataset_dir = os.path.join(directory, dataset)
 168     download_and_extract(dataset_dir, LIBRI_SPEECH_URLS[dataset])
 169     convert_audio_and_split_transcript(
 170         dataset_dir + "/LibriSpeech", dataset, dataset + "-wav",
 171         dataset_dir + "/LibriSpeech", dataset + ".csv")

 174 def define_data_download_flags():
 175   """Define flags for data downloading."""
 176   absl_flags.DEFINE_string(
 177       "data_dir", "/tmp/librispeech_data",
 178       "Directory to download data and extract the tarball")
 179   absl_flags.DEFINE_bool("train_only", False,
 180                          "If true, only download the training set")
 181   absl_flags.DEFINE_bool("dev_only", False,
 182                          "If true, only download the dev set")
 183   absl_flags.DEFINE_bool("test_only", False,
 184                          "If true, only download the test set")
 185 
 186 
 187 def main(_):
 188   if not tf.gfile.Exists(FLAGS.data_dir):
 189     tf.gfile.MakeDirs(FLAGS.data_dir)
 190 
 191   if FLAGS.train_only:
 192     download_and_process_datasets(
 193         FLAGS.data_dir,
 194         ["train-clean-100", "train-clean-360", "train-other-500"])
 195   elif FLAGS.dev_only:
 196     download_and_process_datasets(FLAGS.data_dir, ["dev-clean", "dev-other"])
 197   elif FLAGS.test_only:
 198     download_and_process_datasets(FLAGS.data_dir, ["test-clean", "test-other"])
 199   else:
 200     # By default we download the entire dataset.
 201     download_and_process_datasets(FLAGS.data_dir, LIBRI_SPEECH_URLS.keys())
 202 
 203 
 204 if __name__ == "__main__":
 205   tf.logging.set_verbosity(tf.logging.INFO)
 206   define_data_download_flags()
 207   FLAGS = absl_flags.FLAGS
 208   absl_app.run(main)
