  32 class AudioConfig(object):
  33   """Configs for spectrogram extraction from audio."""
  34 
  35   def __init__(self,
  36                sample_rate,
  37                window_ms,
  38                stride_ms,
  39                normalize=False):
  40     """Initialize the AudioConfig class.
  41 
  42     Args:
  43       sample_rate: an integer denoting the sample rate of the input waveform.
  44       window_ms: an integer for the length of a spectrogram frame, in ms.
  45       stride_ms: an integer for the frame stride, in ms.
  46       normalize: a boolean for whether apply normalization on the audio feature.
  47     """
  48 
  49     self.sample_rate = sample_rate
  50     self.window_ms = window_ms
  51     self.stride_ms = stride_ms
  52     self.normalize = normalize

  55 class DatasetConfig(object):
  56   """Config class for generating the DeepSpeechDataset."""
  57 
  58   def __init__(self, audio_config, data_path, vocab_file_path, sortagrad):
  59     """Initialize the configs for deep speech dataset.
  60 
  61     Args:
  62       audio_config: AudioConfig object specifying the audio-related configs.
  63       data_path: a string denoting the full path of a manifest file.
  64       vocab_file_path: a string specifying the vocabulary file path.
  65       sortagrad: a boolean, if set to true, audio sequences will be fed by
  66                 increasing length in the first training epoch, which will
  67                 expedite network convergence.
  68 
  69     Raises:
  70       RuntimeError: file path not exist.
  71     """
  72 
  73     self.audio_config = audio_config
  74     assert tf.gfile.Exists(data_path)
  75     assert tf.gfile.Exists(vocab_file_path)
  76     self.data_path = data_path
  77     self.vocab_file_path = vocab_file_path
  78     self.sortagrad = sortagrad

  81 def _normalize_audio_feature(audio_feature):
  82   """Perform mean and variance normalization on the spectrogram feature.
  83 
  84   Args:
  85     audio_feature: a numpy array for the spectrogram feature.
  86 
  87   Returns:
  88     a numpy array of the normalized spectrogram.
  89   """
  90   mean = np.mean(audio_feature, axis=0)
  91   var = np.var(audio_feature, axis=0)
  92   normalized = (audio_feature - mean) / (np.sqrt(var) + 1e-6)
  93 
  94   return normalized

  97 def _preprocess_audio(audio_file_path, audio_featurizer, normalize):
  98   """Load the audio file and compute spectrogram feature."""
  99   data, _ = soundfile.read(audio_file_path)
 100   feature = featurizer.compute_spectrogram_feature(
 101       data, audio_featurizer.sample_rate, audio_featurizer.stride_ms,
 102       audio_featurizer.window_ms)
 103   # Feature normalization
 104   if normalize:
 105     feature = _normalize_audio_feature(feature)
 106 
 107   # Adding Channel dimension for conv2D input.
 108   feature = np.expand_dims(feature, axis=2)
 109   return feature

 112 def _preprocess_data(file_path):
 113   """Generate a list of tuples (wav_filename, wav_filesize, transcript).
 114 
 115   Each dataset file contains three columns: "wav_filename", "wav_filesize",
 116   and "transcript". This function parses the csv file and stores each example
 117   by the increasing order of audio length (indicated by wav_filesize).
 118   AS the waveforms are ordered in increasing length, audio samples in a
 119   mini-batch have similar length.
 120 
 121   Args:
 122     file_path: a string specifying the csv file path for a dataset.
 123 
 124   Returns:
 125     A list of tuples (wav_filename, wav_filesize, transcript) sorted by
 126     file_size.
 127   """
 128   tf.logging.info("Loading data set {}".format(file_path))
 129   with tf.gfile.Open(file_path, "r") as f:
 130     lines = f.read().splitlines()
 131   # Skip the csv header in lines[0].
 132   lines = lines[1:]
 133   # The metadata file is tab separated.
 134   lines = [line.split("\t", 2) for line in lines]
 135   # Sort input data by the length of audio sequence.
 136   lines.sort(key=lambda item: int(item[1]))
 137 
 138   return [tuple(line) for line in lines]

 141 class DeepSpeechDataset(object):
 142   """Dataset class for training/evaluation of DeepSpeech model."""
 143 
 144   def __init__(self, dataset_config):
 145     """Initialize the DeepSpeechDataset class.
 146 
 147     Args:
 148       dataset_config: DatasetConfig object.
 149     """
 150     self.config = dataset_config
 151     # Instantiate audio feature extractor.
 152     self.audio_featurizer = featurizer.AudioFeaturizer(
 153         sample_rate=self.config.audio_config.sample_rate,
 154         window_ms=self.config.audio_config.window_ms,
 155         stride_ms=self.config.audio_config.stride_ms)
 156     # Instantiate text feature extractor.
 157     self.text_featurizer = featurizer.TextFeaturizer(
 158         vocab_file=self.config.vocab_file_path)
 159 
 160     self.speech_labels = self.text_featurizer.speech_labels
 161     self.entries = _preprocess_data(self.config.data_path)
 162     # The generated spectrogram will have 161 feature bins.
 163     self.num_feature_bins = 161

 166 def batch_wise_dataset_shuffle(entries, epoch_index, sortagrad, batch_size):
 167   """Batch-wise shuffling of the data entries.
 168 
 169   Each data entry is in the format of (audio_file, file_size, transcript).
 170   If epoch_index is 0 and sortagrad is true, we don't perform shuffling and
 171   return entries in sorted file_size order. Otherwise, do batch_wise shuffling.
 172 
 173   Args:
 174     entries: a list of data entries.
 175     epoch_index: an integer of epoch index
 176     sortagrad: a boolean to control whether sorting the audio in the first
 177       training epoch.
 178     batch_size: an integer for the batch size.
 179 
 180   Returns:
 181     The shuffled data entries.
 182   """
 183   shuffled_entries = []
 184   if epoch_index == 0 and sortagrad:
 185     # No need to shuffle.
 186     shuffled_entries = entries
 187   else:
 188     # Shuffle entries batch-wise.
 189     max_buckets = int(math.floor(len(entries) / batch_size))
 190     total_buckets = [i for i in xrange(max_buckets)]
 191     random.shuffle(total_buckets)
 192     shuffled_entries = []
 193     for i in total_buckets:
 194       shuffled_entries.extend(entries[i * batch_size : (i + 1) * batch_size])
 195     # If the last batch doesn't contain enough batch_size examples,
 196     # just append it to the shuffled_entries.
 197     shuffled_entries.extend(entries[max_buckets * batch_size:])
 198 
 199   return shuffled_entries

 202 def input_fn(batch_size, deep_speech_dataset, repeat=1):
 203   """Input function for model training and evaluation.
 204 
 205   Args:
 206     batch_size: an integer denoting the size of a batch.
 207     deep_speech_dataset: DeepSpeechDataset object.
 208     repeat: an integer for how many times to repeat the dataset.
 209 
 210   Returns:
 211     a tf.data.Dataset object for model to consume.
 212   """
 213   # Dataset properties
 214   data_entries = deep_speech_dataset.entries
 215   num_feature_bins = deep_speech_dataset.num_feature_bins
 216   audio_featurizer = deep_speech_dataset.audio_featurizer
 217   feature_normalize = deep_speech_dataset.config.audio_config.normalize
 218   text_featurizer = deep_speech_dataset.text_featurizer
 219 
 220   def _gen_data():
 221     """Dataset generator function."""
 222     for audio_file, _, transcript in data_entries:
 223       features = _preprocess_audio(
 224           audio_file, audio_featurizer, feature_normalize)
 225       labels = featurizer.compute_label_feature(
 226           transcript, text_featurizer.token_to_index)
 227       input_length = [features.shape[0]]
 228       label_length = [len(labels)]
 229       # Yield a tuple of (features, labels) where features is a dict containing
 230       # all info about the actual data features.
 231       yield (
 232           {
 233               "features": features,
 234               "input_length": input_length,
 235               "label_length": label_length
 236           },
 237           labels)
 238 
 239   dataset = tf.data.Dataset.from_generator(
 240       _gen_data,
 241       output_types=(
 242           {
 243               "features": tf.float32,
 244               "input_length": tf.int32,
 245               "label_length": tf.int32
 246           },
 247           tf.int32),
 248       output_shapes=(
 249           {
 250               "features": tf.TensorShape([None, num_feature_bins, 1]),
 251               "input_length": tf.TensorShape([1]),
 252               "label_length": tf.TensorShape([1])
 253           },
 254           tf.TensorShape([None]))
 255   )
 256 
 257   # Repeat and batch the dataset
 258   dataset = dataset.repeat(repeat)
 259 
 260   # Padding the features to its max length dimensions.
 261   dataset = dataset.padded_batch(
 262       batch_size=batch_size,
 263       padded_shapes=(
 264           {
 265               "features": tf.TensorShape([None, num_feature_bins, 1]),
 266               "input_length": tf.TensorShape([1]),
 267               "label_length": tf.TensorShape([1])
 268           },
 269           tf.TensorShape([None]))
 270   )
 271 
 272   # Prefetch to improve speed of input pipeline.
 273   dataset = dataset.prefetch(buffer_size=tf.data.experimental.AUTOTUNE)
 274   return dataset
