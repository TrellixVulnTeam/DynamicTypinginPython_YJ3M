  24 def compute_spectrogram_feature(samples, sample_rate, stride_ms=10.0,
  25                                 window_ms=20.0, max_freq=None, eps=1e-14):
  26   """Compute the spectrograms for the input samples(waveforms).
  27 
  28   More about spectrogram computation, please refer to:
  29   https://en.wikipedia.org/wiki/Short-time_Fourier_transform.
  30   """
  31   if max_freq is None:
  32     max_freq = sample_rate / 2
  33   if max_freq > sample_rate / 2:
  34     raise ValueError("max_freq must not be greater than half of sample rate.")
  35 
  36   if stride_ms > window_ms:
  37     raise ValueError("Stride size must not be greater than window size.")
  38 
  39   stride_size = int(0.001 * sample_rate * stride_ms)
  40   window_size = int(0.001 * sample_rate * window_ms)
  41 
  42   # Extract strided windows
  43   truncate_size = (len(samples) - window_size) % stride_size
  44   samples = samples[:len(samples) - truncate_size]
  45   nshape = (window_size, (len(samples) - window_size) // stride_size + 1)
  46   nstrides = (samples.strides[0], samples.strides[0] * stride_size)
  47   windows = np.lib.stride_tricks.as_strided(
  48       samples, shape=nshape, strides=nstrides)
  49   assert np.all(
  50       windows[:, 1] == samples[stride_size:(stride_size + window_size)])
  51 
  52   # Window weighting, squared Fast Fourier Transform (fft), scaling
  53   weighting = np.hanning(window_size)[:, None]
  54   fft = np.fft.rfft(windows * weighting, axis=0)
  55   fft = np.absolute(fft)
  56   fft = fft**2
  57   scale = np.sum(weighting**2) * sample_rate
  58   fft[1:-1, :] *= (2.0 / scale)
  59   fft[(0, -1), :] /= scale
  60   # Prepare fft frequency list
  61   freqs = float(sample_rate) / window_size * np.arange(fft.shape[0])
  62 
  63   # Compute spectrogram feature
  64   ind = np.where(freqs <= max_freq)[0][-1] + 1
  65   specgram = np.log(fft[:ind, :] + eps)
  66   return np.transpose(specgram, (1, 0))

  69 class AudioFeaturizer(object):
  70   """Class to extract spectrogram features from the audio input."""
  71 
  72   def __init__(self,
  73                sample_rate=16000,
  74                window_ms=20.0,
  75                stride_ms=10.0):
  76     """Initialize the audio featurizer class according to the configs.
  77 
  78     Args:
  79       sample_rate: an integer specifying the sample rate of the input waveform.
  80       window_ms: an integer for the length of a spectrogram frame, in ms.
  81       stride_ms: an integer for the frame stride, in ms.
  82     """
  83     self.sample_rate = sample_rate
  84     self.window_ms = window_ms
  85     self.stride_ms = stride_ms

  88 def compute_label_feature(text, token_to_idx):
  89   """Convert string to a list of integers."""
  90   tokens = list(text.strip().lower())
  91   feats = [token_to_idx[token] for token in tokens]
  92   return feats

  95 class TextFeaturizer(object):
  96   """Extract text feature based on char-level granularity.
  97 
  98   By looking up the vocabulary table, each input string (one line of transcript)
  99   will be converted to a sequence of integer indexes.
 100   """
 101 
 102   def __init__(self, vocab_file):
 103     lines = []
 104     with codecs.open(vocab_file, "r", "utf-8") as fin:
 105       lines.extend(fin.readlines())
 106     self.token_to_index = {}
 107     self.index_to_token = {}
 108     self.speech_labels = ""
 109     index = 0
 110     for line in lines:
 111       line = line[:-1]  # Strip the '\n' char.
 112       if line.startswith("#"):
 113         # Skip from reading comment line.
 114         continue
 115       self.token_to_index[line] = index
 116       self.index_to_token[index] = line
 117       self.speech_labels += line
 118       index += 1
