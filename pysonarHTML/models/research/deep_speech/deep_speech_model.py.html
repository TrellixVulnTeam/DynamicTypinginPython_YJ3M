  16 from __future__ import absolute_import
  17 from __future__ import division

  20 from six.moves import xrange  # pylint: disable=redefined-builtin
  21 import tensorflow as tf

  23 # Supported rnn cells.
  24 SUPPORTED_RNNS = {
  25     "lstm": tf.contrib.rnn.BasicLSTMCell,
  26     "rnn": tf.contrib.rnn.RNNCell,
  27     "gru": tf.contrib.rnn.GRUCell,
  28 }
  29 
  30 # Parameters for batch normalization.
  31 _BATCH_NORM_EPSILON = 1e-5
  32 _BATCH_NORM_DECAY = 0.997
  33 
  34 # Filters of convolution layer
  35 _CONV_FILTERS = 32
  36

  38 def batch_norm(inputs, training):
  39   """Batch normalization layer.
  40 
  41   Note that the momentum to use will affect validation accuracy over time.
  42   Batch norm has different behaviors during training/evaluation. With a large
  43   momentum, the model takes longer to get a near-accurate estimation of the
  44   moving mean/variance over the entire training dataset, which means we need
  45   more iterations to see good evaluation results. If the training data is evenly
  46   distributed over the feature space, we can also try setting a smaller momentum
  47   (such as 0.1) to get good evaluation result sooner.
  48 
  49   Args:
  50     inputs: input data for batch norm layer.
  51     training: a boolean to indicate if it is in training stage.
  52 
  53   Returns:
  54     tensor output from batch norm layer.
  55   """
  56   return tf.layers.batch_normalization(
  57       inputs=inputs, momentum=_BATCH_NORM_DECAY, epsilon=_BATCH_NORM_EPSILON,
  58       fused=True, training=training)

  61 def _conv_bn_layer(inputs, padding, filters, kernel_size, strides, layer_id,
  62                    training):
  63   """Defines 2D convolutional + batch normalization layer.
  64 
  65   Args:
  66     inputs: input data for convolution layer.
  67     padding: padding to be applied before convolution layer.
  68     filters: an integer, number of output filters in the convolution.
  69     kernel_size: a tuple specifying the height and width of the 2D convolution
  70       window.
  71     strides: a tuple specifying the stride length of the convolution.
  72     layer_id: an integer specifying the layer index.
  73     training: a boolean to indicate which stage we are in (training/eval).
  74 
  75   Returns:
  76     tensor output from the current layer.
  77   """
  78   # Perform symmetric padding on the feature dimension of time_step
  79   # This step is required to avoid issues when RNN output sequence is shorter
  80   # than the label length.
  81   inputs = tf.pad(
  82       inputs,
  83       [[0, 0], [padding[0], padding[0]], [padding[1], padding[1]], [0, 0]])
  84   inputs = tf.layers.conv2d(
  85       inputs=inputs, filters=filters, kernel_size=kernel_size, strides=strides,
  86       padding="valid", use_bias=False, activation=tf.nn.relu6,
  87       name="cnn_{}".format(layer_id))
  88   return batch_norm(inputs, training)

  91 def _rnn_layer(inputs, rnn_cell, rnn_hidden_size, layer_id, is_batch_norm,
  92                is_bidirectional, training):
  93   """Defines a batch normalization + rnn layer.
  94 
  95   Args:
  96     inputs: input tensors for the current layer.
  97     rnn_cell: RNN cell instance to use.
  98     rnn_hidden_size: an integer for the dimensionality of the rnn output space.
  99     layer_id: an integer for the index of current layer.
 100     is_batch_norm: a boolean specifying whether to perform batch normalization
 101       on input states.
 102     is_bidirectional: a boolean specifying whether the rnn layer is
 103       bi-directional.
 104     training: a boolean to indicate which stage we are in (training/eval).
 105 
 106   Returns:
 107     tensor output for the current layer.
 108   """
 109   if is_batch_norm:
 110     inputs = batch_norm(inputs, training)
 111 
 112   # Construct forward/backward RNN cells.
 113   fw_cell = rnn_cell(num_units=rnn_hidden_size,
 114                      name="rnn_fw_{}".format(layer_id))
 115   bw_cell = rnn_cell(num_units=rnn_hidden_size,
 116                      name="rnn_bw_{}".format(layer_id))
 117 
 118   if is_bidirectional:
 119     outputs, _ = tf.nn.bidirectional_dynamic_rnn(
 120         cell_fw=fw_cell, cell_bw=bw_cell, inputs=inputs, dtype=tf.float32,
 121         swap_memory=True)
 122     rnn_outputs = tf.concat(outputs, -1)
 123   else:
 124     rnn_outputs = tf.nn.dynamic_rnn(
 125         fw_cell, inputs, dtype=tf.float32, swap_memory=True)
 126 
 127   return rnn_outputs

 130 class DeepSpeech2(object):
 131   """Define DeepSpeech2 model."""
 132 
 133   def __init__(self, num_rnn_layers, rnn_type, is_bidirectional,
 134                rnn_hidden_size, num_classes, use_bias):
 135     """Initialize DeepSpeech2 model.
 136 
 137     Args:
 138       num_rnn_layers: an integer, the number of rnn layers. By default, it's 5.
 139       rnn_type: a string, one of the supported rnn cells: gru, rnn and lstm.
 140       is_bidirectional: a boolean to indicate if the rnn layer is bidirectional.
 141       rnn_hidden_size: an integer for the number of hidden states in each unit.
 142       num_classes: an integer, the number of output classes/labels.
 143       use_bias: a boolean specifying whether to use bias in the last fc layer.
 144     """
 145     self.num_rnn_layers = num_rnn_layers
 146     self.rnn_type = rnn_type
 147     self.is_bidirectional = is_bidirectional
 148     self.rnn_hidden_size = rnn_hidden_size
 149     self.num_classes = num_classes
 150     self.use_bias = use_bias
 151 
 152   def __call__(self, inputs, training):
 153     # Two cnn layers.
 154     inputs = _conv_bn_layer(
 155         inputs, padding=(20, 5), filters=_CONV_FILTERS, kernel_size=(41, 11),
 156         strides=(2, 2), layer_id=1, training=training)
 157 
 158     inputs = _conv_bn_layer(
 159         inputs, padding=(10, 5), filters=_CONV_FILTERS, kernel_size=(21, 11),
 160         strides=(2, 1), layer_id=2, training=training)
 161 
 162     # output of conv_layer2 with the shape of
 163     # [batch_size (N), times (T), features (F), channels (C)].
 164     # Convert the conv output to rnn input.
 165     batch_size = tf.shape(inputs)[0]
 166     feat_size = inputs.get_shape().as_list()[2]
 167     inputs = tf.reshape(
 168         inputs,
 169         [batch_size, -1, feat_size * _CONV_FILTERS])
 170 
 171     # RNN layers.
 172     rnn_cell = SUPPORTED_RNNS[self.rnn_type]
 173     for layer_counter in xrange(self.num_rnn_layers):
 174       # No batch normalization on the first layer.
 175       is_batch_norm = (layer_counter != 0)
 176       inputs = _rnn_layer(
 177           inputs, rnn_cell, self.rnn_hidden_size, layer_counter + 1,
 178           is_batch_norm, self.is_bidirectional, training)
 179 
 180     # FC layer with batch norm.
 181     inputs = batch_norm(inputs, training)
 182     logits = tf.layers.dense(inputs, self.num_classes, use_bias=self.use_bias)
 183 
 184     return logits
