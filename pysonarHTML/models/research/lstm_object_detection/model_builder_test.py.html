  28   def create_train_model(self, model_config, lstm_config):
  29     """Builds a DetectionModel based on the model config.
  30 
  31     Args:
  32       model_config: A model.proto object containing the config for the desired
  33         DetectionModel.
  34       lstm_config: LstmModel config proto that specifies LSTM train/eval
  35         configs.
  36 
  37     Returns:
  38       DetectionModel based on the config.
  39     """
  40     return model_builder.build(model_config, lstm_config, is_training=True)
  41

 241   def test_model_creation_from_valid_configs(self):
 242     configs = self.get_model_configs_from_proto()
 243     # Test model properties.
 244     self.assertEqual(configs['model'].ssd.negative_class_weight, 2.0)
 245     self.assertTrue(configs['model'].ssd.normalize_loc_loss_by_codesize)
 246     self.assertEqual(configs['model'].ssd.feature_extractor.type,
 247                      'lstm_ssd_mobilenet_v1')
 248 
 249     model = self.create_train_model(configs['model'], configs['lstm_model'])
 250     # Test architechture type.
 251     self.assertIsInstance(model, lstm_ssd_meta_arch.LSTMSSDMetaArch)
 252     # Test LSTM unroll length.
 253     self.assertEqual(model.unroll_length, 4)
 254 
 255     model = self.create_eval_model(configs['model'], configs['lstm_model'])
 256     # Test architechture type.
 257     self.assertIsInstance(model, lstm_ssd_meta_arch.LSTMSSDMetaArch)
 258     # Test LSTM configs.
 259     self.assertEqual(model.unroll_length, 4)
 260 
 261   def test_interleaved_model_creation_from_valid_configs(self):
 262     configs = self.get_interleaved_model_configs_from_proto()
 263     # Test model properties.
 264     self.assertEqual(configs['model'].ssd.negative_class_weight, 2.0)
 265     self.assertTrue(configs['model'].ssd.normalize_loc_loss_by_codesize)
 266     self.assertEqual(configs['model'].ssd.feature_extractor.type,
 267                      'lstm_ssd_interleaved_mobilenet_v2')
 268 
 269     model = self.create_train_model(configs['model'], configs['lstm_model'])
 270     # Test architechture type.
 271     self.assertIsInstance(model, lstm_ssd_meta_arch.LSTMSSDMetaArch)
 272     # Test LSTM configs.
 273     self.assertEqual(model.unroll_length, 4)
 274     self.assertEqual(model._feature_extractor.lstm_state_depth, 320)
 275     self.assertAllClose(model._feature_extractor.depth_multipliers, (1.4, 0.35))
 276     self.assertTrue(model._feature_extractor.pre_bottleneck)
 277     self.assertTrue(model._feature_extractor.low_res)
 278     self.assertEqual(model._feature_extractor.interleave_method,
 279                      'RANDOM_SKIP_SMALL')
 280 
 281     model = self.create_eval_model(configs['model'], configs['lstm_model'])
 282     # Test architechture type.
 283     self.assertIsInstance(model, lstm_ssd_meta_arch.LSTMSSDMetaArch)
 284     # Test LSTM configs.
 285     self.assertEqual(model.unroll_length, 10)
 286     self.assertEqual(model._feature_extractor.lstm_state_depth, 320)
 287     self.assertAllClose(model._feature_extractor.depth_multipliers, (1.4, 0.35))
 288     self.assertTrue(model._feature_extractor.pre_bottleneck)
 289     self.assertTrue(model._feature_extractor.low_res)
 290     self.assertEqual(model._feature_extractor.interleave_method, 'SKIP3')
 291 
 292   def test_model_creation_from_invalid_configs(self):
 293     configs = self.get_model_configs_from_proto()
 294     # Test model build failure with wrong input configs.
 295     with self.assertRaises(AttributeError):
 296       _ = self.create_train_model(configs['model'], configs['model'])
 297     with self.assertRaises(AttributeError):
 298       _ = self.create_eval_model(configs['model'], configs['model'])
