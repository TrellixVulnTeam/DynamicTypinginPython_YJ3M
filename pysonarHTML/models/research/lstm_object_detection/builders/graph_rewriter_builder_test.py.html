  16 """Tests for graph_rewriter_builder."""
  17 import mock
  18 import tensorflow.compat.v1 as tf

  28 class QuantizationBuilderTest(tf.test.TestCase):

  30   def testQuantizationBuilderSetsUpCorrectTrainArguments(self):
  31     with mock.patch.object(
  32         contrib_quantize,
  33         'experimental_create_training_graph') as mock_quant_fn:
  34       with mock.patch.object(contrib_layers,
  35                              'summarize_collection') as mock_summarize_col:
  36         graph_rewriter_proto = graph_rewriter_pb2.GraphRewriter()
  37         graph_rewriter_proto.quantization.delay = 10
  38         graph_rewriter_proto.quantization.weight_bits = 8
  39         graph_rewriter_proto.quantization.activation_bits = 8
  40         graph_rewrite_fn = graph_rewriter_builder.build(
  41             graph_rewriter_proto, is_training=True)
  42         graph_rewrite_fn()
  43         _, kwargs = mock_quant_fn.call_args
  44         self.assertEqual(kwargs['input_graph'], tf.get_default_graph())
  45         self.assertEqual(kwargs['quant_delay'], 10)
  46         mock_summarize_col.assert_called_with('quant_vars')
  47 
  48   def testQuantizationBuilderSetsUpCorrectEvalArguments(self):

  60         mock_summarize_col.assert_called_with('quant_vars')

  62   def testQuantizationBuilderAddsQuantOverride(self):
  63     graph = ops.Graph()
  64     with graph.as_default():
  65       self._buildGraph()
  66 
  67       quant_overrides_proto = quant_overrides_pb2.QuantOverrides()
  68       quant_config = quant_overrides_proto.quant_configs.add()
  69       quant_config.op_name = 'test_graph/add_ab'
  70       quant_config.quant_op_name = 'act_quant'
  71       quant_config.fixed_range = True
  72       quant_config.min = 0
  73       quant_config.max = 6
  74       quant_config.delay = 100
  75 
  76       graph_rewriter_proto = graph_rewriter_pb2.GraphRewriter()
  77       graph_rewriter_proto.quantization.delay = 10
  78       graph_rewriter_proto.quantization.weight_bits = 8
  79       graph_rewriter_proto.quantization.activation_bits = 8
  80 
  81       graph_rewrite_fn = graph_rewriter_builder.build(
  82           graph_rewriter_proto,
  83           quant_overrides_config=quant_overrides_proto,
  84           is_training=True)
  85       graph_rewrite_fn()
  86 
  87       act_quant_found = False
  88       quant_delay_found = False
  89       for op in graph.get_operations():
  90         if (quant_config.quant_op_name in op.name and
  91             op.type == 'FakeQuantWithMinMaxArgs'):
  92           act_quant_found = True
  93           min_val = op.get_attr('min')
  94           max_val = op.get_attr('max')
  95           self.assertEqual(min_val, quant_config.min)
  96           self.assertEqual(max_val, quant_config.max)
  97         if ('activate_quant' in op.name and
  98             quant_config.quant_op_name in op.name and op.type == 'Const'):
  99           tensor = op.get_attr('value')
 100           if tensor.int64_val[0] == quant_config.delay:
 101             quant_delay_found = True
 102 
 103       self.assertTrue(act_quant_found)
 104       self.assertTrue(quant_delay_found)

 106   def _buildGraph(self, scope='test_graph'):
 107     with ops.name_scope(scope):
 108       a = tf.constant(10, dtype=dtypes.float32, name='input_a')
 109       b = tf.constant(20, dtype=dtypes.float32, name='input_b')
 110       ab = tf.add(a, b, name='add_ab')
 111       c = tf.constant(30, dtype=dtypes.float32, name='input_c')
 112       abc = tf.multiply(ab, c, name='mul_ab_c')
 113       return abc
