  26 class BottleneckConvLSTMCell(contrib_rnn.RNNCell):
  27   """Basic LSTM recurrent network cell using separable convolutions.
  28 
  29   The implementation is based on:
  30   Mobile Video Object Detection with Temporally-Aware Feature Maps
  31   https://arxiv.org/abs/1711.06368.
  32 
  33   We add forget_bias (default: 1) to the biases of the forget gate in order to
  34   reduce the scale of forgetting in the beginning of the training.
  35 
  36   This LSTM first projects inputs to the size of the output before doing gate
  37   computations. This saves params unless the input is less than a third of the
  38   state size channel-wise.
  39   """
  40 
  41   def __init__(self,
  42                filter_size,
  43                output_size,
  44                num_units,
  45                forget_bias=1.0,
  46                activation=tf.tanh,
  47                flatten_state=False,
  48                clip_state=False,
  49                output_bottleneck=False,
  50                pre_bottleneck=False,
  51                visualize_gates=False):
  52     """Initializes the basic LSTM cell.
  53 
  54     Args:
  55       filter_size: collection, conv filter size.
  56       output_size: collection, the width/height dimensions of the cell/output.
  57       num_units: int, The number of channels in the LSTM cell.
  58       forget_bias: float, The bias added to forget gates (see above).
  59       activation: Activation function of the inner states.
  60       flatten_state: if True, state tensor will be flattened and stored as a 2-d
  61         tensor. Use for exporting the model to tfmini.
  62       clip_state: if True, clip state between [-6, 6].
  63       output_bottleneck: if True, the cell bottleneck will be concatenated to
  64         the cell output.
  65       pre_bottleneck: if True, cell assumes that bottlenecking was performing
  66         before the function was called.
  67       visualize_gates: if True, add histogram summaries of all gates and outputs
  68         to tensorboard.
  69     """
  70     self._filter_size = list(filter_size)
  71     self._output_size = list(output_size)
  72     self._num_units = num_units
  73     self._forget_bias = forget_bias
  74     self._activation = activation
  75     self._viz_gates = visualize_gates
  76     self._flatten_state = flatten_state
  77     self._clip_state = clip_state
  78     self._output_bottleneck = output_bottleneck
  79     self._pre_bottleneck = pre_bottleneck
  80     self._param_count = self._num_units
  81     for dim in self._output_size:
  82       self._param_count *= dim
  83 
  84   @property
  85   def state_size(self):
  86     return contrib_rnn.LSTMStateTuple(self._output_size + [self._num_units],
  87                                       self._output_size + [self._num_units])
  88 
  89   @property
  90   def state_size_flat(self):
  91     return contrib_rnn.LSTMStateTuple([self._param_count], [self._param_count])
  92 
  93   @property
  94   def output_size(self):
  95     return self._output_size + [self._num_units]
  96 
  97   def __call__(self, inputs, state, scope=None):
  98     """Long short-term memory cell (LSTM) with bottlenecking.
  99 
 100     Args:
 101       inputs: Input tensor at the current timestep.
 102       state: Tuple of tensors, the state and output at the previous timestep.
 103       scope: Optional scope.
 104 
 105     Returns:
 106       A tuple where the first element is the LSTM output and the second is
 107       a LSTMStateTuple of the state at the current timestep.
 108     """
 109     scope = scope or 'conv_lstm_cell'
 110     with tf.variable_scope(scope, reuse=tf.AUTO_REUSE):
 111       c, h = state
 112 
 113       # unflatten state if necessary
 114       if self._flatten_state:
 115         c = tf.reshape(c, [-1] + self.output_size)
 116         h = tf.reshape(h, [-1] + self.output_size)
 117 
 118       # summary of input passed into cell
 119       if self._viz_gates:
 120         slim.summaries.add_histogram_summary(inputs, 'cell_input')
 121       if self._pre_bottleneck:
 122         bottleneck = inputs
 123       else:
 124         bottleneck = slim.separable_conv2d(
 125             tf.concat([inputs, h], 3),
 126             self._num_units,
 127             self._filter_size,
 128             depth_multiplier=1,
 129             activation_fn=self._activation,
 130             normalizer_fn=None,
 131             scope='bottleneck')
 132 
 133         if self._viz_gates:
 134           slim.summaries.add_histogram_summary(bottleneck, 'bottleneck')
 135 
 136       concat = slim.separable_conv2d(
 137           bottleneck,
 138           4 * self._num_units,
 139           self._filter_size,
 140           depth_multiplier=1,
 141           activation_fn=None,
 142           normalizer_fn=None,
 143           scope='gates')
 144 
 145       i, j, f, o = tf.split(concat, 4, 3)
 146 
 147       new_c = (
 148           c * tf.sigmoid(f + self._forget_bias) +
 149           tf.sigmoid(i) * self._activation(j))
 150       if self._clip_state:
 151         new_c = tf.clip_by_value(new_c, -6, 6)
 152       new_h = self._activation(new_c) * tf.sigmoid(o)
 153       # summary of cell output and new state
 154       if self._viz_gates:
 155         slim.summaries.add_histogram_summary(new_h, 'cell_output')
 156         slim.summaries.add_histogram_summary(new_c, 'cell_state')
 157 
 158       output = new_h
 159       if self._output_bottleneck:
 160         output = tf.concat([new_h, bottleneck], axis=3)
 161 
 162       # reflatten state to store it
 163       if self._flatten_state:
 164         new_c = tf.reshape(new_c, [-1, self._param_count])
 165         new_h = tf.reshape(new_h, [-1, self._param_count])
 166 
 167       return output, contrib_rnn.LSTMStateTuple(new_c, new_h)
 168 
 169   def init_state(self, state_name, batch_size, dtype, learned_state=False):
 170     """Creates an initial state compatible with this cell.
 171 
 172     Args:
 173       state_name: name of the state tensor
 174       batch_size: model batch size
 175       dtype: dtype for the tensor values i.e. tf.float32
 176       learned_state: whether the initial state should be learnable. If false,
 177         the initial state is set to all 0's
 178 
 179     Returns:
 180       The created initial state.
 181     """
 182     state_size = (
 183         self.state_size_flat if self._flatten_state else self.state_size)
 184     # list of 2 zero tensors or variables tensors, depending on if
 185     # learned_state is true
 186     # pylint: disable=g-long-ternary,g-complex-comprehension
 187     ret_flat = [(contrib_variables.model_variable(
 188         state_name + str(i),
 189         shape=s,
 190         dtype=dtype,
 191         initializer=tf.truncated_normal_initializer(stddev=0.03))
 192                  if learned_state else tf.zeros(
 193                      [batch_size] + s, dtype=dtype, name=state_name))
 194                 for i, s in enumerate(state_size)]
 195 
 196     # duplicates initial state across the batch axis if it's learned
 197     if learned_state:
 198       ret_flat = [
 199           tf.stack([tensor
 200                     for i in range(int(batch_size))])
 201           for tensor in ret_flat
 202       ]
 203     for s, r in zip(state_size, ret_flat):
 204       r.set_shape([None] + s)
 205     return tf.nest.pack_sequence_as(structure=[1, 1], flat_sequence=ret_flat)
 206 
 207   def pre_bottleneck(self, inputs, state, input_index):
 208     """Apply pre-bottleneck projection to inputs.
 209 
 210     Pre-bottleneck operation maps features of different channels into the same
 211     dimension. The purpose of this op is to share the features from both large
 212     and small models in the same LSTM cell.
 213 
 214     Args:
 215       inputs: 4D Tensor with shape [batch_size x width x height x input_size].
 216       state: 4D Tensor with shape [batch_size x width x height x state_size].
 217       input_index: integer index indicating which base features the inputs
 218         correspoding to.
 219 
 220     Returns:
 221       inputs: pre-bottlenecked inputs.
 222     Raises:
 223       ValueError: If pre_bottleneck is not set or inputs is not rank 4.
 224     """
 225     # Sometimes state is a tuple, in which case it cannot be modified, e.g.
 226     # during training, tf.contrib.training.SequenceQueueingStateSaver
 227     # returns the state as a tuple. This should not be an issue since we
 228     # only need to modify state[1] during export, when state should be a
 229     # list.
 230     if len(inputs.shape) != 4:
 231       raise ValueError('Expect rank 4 feature tensor.')
 232     if not self._flatten_state and len(state.shape) != 4:
 233       raise ValueError('Expect rank 4 state tensor.')
 234     if self._flatten_state and len(state.shape) != 2:
 235       raise ValueError('Expect rank 2 state tensor when flatten_state is set.')
 236 
 237     with tf.name_scope(None):
 238       state = tf.identity(state, name='raw_inputs/init_lstm_h')
 239     if self._flatten_state:
 240       batch_size = inputs.shape[0]
 241       height = inputs.shape[1]
 242       width = inputs.shape[2]
 243       state = tf.reshape(state, [batch_size, height, width, -1])
 244     with tf.variable_scope('conv_lstm_cell', reuse=tf.AUTO_REUSE):
 245       scope_name = 'bottleneck_%d' % input_index
 246       inputs = slim.separable_conv2d(
 247           tf.concat([inputs, state], 3),
 248           self.output_size[-1],
 249           self._filter_size,
 250           depth_multiplier=1,
 251           activation_fn=tf.nn.relu6,
 252           normalizer_fn=None,
 253           scope=scope_name)
 254       # For exporting inference graph, we only mark the first timestep.
 255       with tf.name_scope(None):
 256         inputs = tf.identity(
 257             inputs, name='raw_outputs/base_endpoint_%d' % (input_index + 1))
 258     return inputs

 261 class GroupedConvLSTMCell(contrib_rnn.RNNCell):
 262   """Basic LSTM recurrent network cell using separable convolutions.
 263 
 264   The implementation is based on: https://arxiv.org/abs/1903.10172.
 265 
 266   We add forget_bias (default: 1) to the biases of the forget gate in order to
 267   reduce the scale of forgetting in the beginning of the training.
 268 
 269   This LSTM first projects inputs to the size of the output before doing gate
 270   computations. This saves params unless the input is less than a third of the
 271   state size channel-wise. Computation of bottlenecks and gates is divided
 272   into independent groups for further savings.
 273   """
 274 
 275   def __init__(self,
 276                filter_size,
 277                output_size,
 278                num_units,
 279                is_training,
 280                forget_bias=1.0,
 281                activation=tf.tanh,
 282                use_batch_norm=False,
 283                flatten_state=False,
 284                groups=4,
 285                clip_state=False,
 286                scale_state=False,
 287                output_bottleneck=False,
 288                pre_bottleneck=False,
 289                is_quantized=False,
 290                visualize_gates=False,
 291                conv_op_overrides=None):
 292     """Initialize the basic LSTM cell.
 293 
 294     Args:
 295       filter_size: collection, conv filter size
 296       output_size: collection, the width/height dimensions of the cell/output
 297       num_units: int, The number of channels in the LSTM cell.
 298       is_training: Whether the LSTM is in training mode.
 299       forget_bias: float, The bias added to forget gates (see above).
 300       activation: Activation function of the inner states.
 301       use_batch_norm: if True, use batch norm after convolution
 302       flatten_state: if True, state tensor will be flattened and stored as a 2-d
 303         tensor. Use for exporting the model to tfmini
 304       groups: Number of groups to split the state into. Must evenly divide
 305         num_units.
 306       clip_state: if True, clips state between [-6, 6].
 307       scale_state: if True, scales state so that all values are under 6 at all
 308         times.
 309       output_bottleneck: if True, the cell bottleneck will be concatenated to
 310         the cell output.
 311       pre_bottleneck: if True, cell assumes that bottlenecking was performing
 312         before the function was called.
 313       is_quantized: if True, the model is in quantize mode, which requires
 314         quantization friendly concat and separable_conv2d ops.
 315       visualize_gates: if True, add histogram summaries of all gates and outputs
 316         to tensorboard
 317       conv_op_overrides: A list of convolutional operations that override the
 318         'bottleneck' and 'convolution' layers before lstm gates. If None, the
 319         original implementation of seperable_conv will be used. The length of
 320         the list should be two.
 321 
 322     Raises:
 323       ValueError: when both clip_state and scale_state are enabled.
 324     """
 325     if clip_state and scale_state:
 326       raise ValueError('clip_state and scale_state cannot both be enabled.')
 327 
 328     self._filter_size = list(filter_size)
 329     self._output_size = list(output_size)
 330     self._num_units = num_units
 331     self._is_training = is_training
 332     self._forget_bias = forget_bias
 333     self._activation = activation
 334     self._use_batch_norm = use_batch_norm
 335     self._viz_gates = visualize_gates
 336     self._flatten_state = flatten_state
 337     self._param_count = self._num_units
 338     self._groups = groups
 339     self._scale_state = scale_state
 340     self._clip_state = clip_state
 341     self._output_bottleneck = output_bottleneck
 342     self._pre_bottleneck = pre_bottleneck
 343     self._is_quantized = is_quantized
 344     for dim in self._output_size:
 345       self._param_count *= dim
 346     self._conv_op_overrides = conv_op_overrides
 347     if self._conv_op_overrides and len(self._conv_op_overrides) != 2:
 348       raise ValueError('Bottleneck and Convolutional layer should be overriden'
 349                        'together')
 350 
 351   @property
 352   def state_size(self):
 353     return contrib_rnn.LSTMStateTuple(self._output_size + [self._num_units],
 354                                       self._output_size + [self._num_units])
 355 
 356   @property
 357   def state_size_flat(self):
 358     return contrib_rnn.LSTMStateTuple([self._param_count], [self._param_count])
 359 
 360   @property
 361   def output_size(self):
 362     return self._output_size + [self._num_units]
 363 
 364   @property
 365   def filter_size(self):
 366     return self._filter_size
 367 
 368   @property
 369   def num_groups(self):
 370     return self._groups
 371 
 372   def __call__(self, inputs, state, scope=None):
 373     """Long short-term memory cell (LSTM) with bottlenecking.
 374 
 375     Includes logic for quantization-aware training. Note that all concats and
 376     activations use fixed ranges unless stated otherwise.
 377 
 378     Args:
 379       inputs: Input tensor at the current timestep.
 380       state: Tuple of tensors, the state at the previous timestep.
 381       scope: Optional scope.
 382 
 383     Returns:
 384       A tuple where the first element is the LSTM output and the second is
 385       a LSTMStateTuple of the state at the current timestep.
 386     """
 387     scope = scope or 'conv_lstm_cell'
 388     with tf.variable_scope(scope, reuse=tf.AUTO_REUSE):
 389       c, h = state
 390 
 391       # Set nodes to be under raw_inputs/ name scope for tfmini export.
 392       with tf.name_scope(None):
 393         c = tf.identity(c, name='raw_inputs/init_lstm_c')
 394         # When pre_bottleneck is enabled, input h handle is in rnn_decoder.py
 395         if not self._pre_bottleneck:
 396           h = tf.identity(h, name='raw_inputs/init_lstm_h')
 397 
 398       # unflatten state if necessary
 399       if self._flatten_state:
 400         c = tf.reshape(c, [-1] + self.output_size)
 401         h = tf.reshape(h, [-1] + self.output_size)
 402 
 403       c_list = tf.split(c, self._groups, axis=3)
 404       if self._pre_bottleneck:
 405         inputs_list = tf.split(inputs, self._groups, axis=3)
 406       else:
 407         h_list = tf.split(h, self._groups, axis=3)
 408       out_bottleneck = []
 409       out_c = []
 410       out_h = []
 411       # summary of input passed into cell
 412       if self._viz_gates:
 413         slim.summaries.add_histogram_summary(inputs, 'cell_input')
 414 
 415       for k in range(self._groups):
 416         if self._pre_bottleneck:
 417           bottleneck = inputs_list[k]
 418         else:
 419           if self._conv_op_overrides:
 420             bottleneck_fn = self._conv_op_overrides[0]
 421           else:
 422             bottleneck_fn = functools.partial(
 423                 lstm_utils.quantizable_separable_conv2d,
 424                 kernel_size=self._filter_size,
 425                 activation_fn=self._activation)
 426           if self._use_batch_norm:
 427             b_x = bottleneck_fn(
 428                 inputs=inputs,
 429                 num_outputs=self._num_units // self._groups,
 430                 is_quantized=self._is_quantized,
 431                 depth_multiplier=1,
 432                 normalizer_fn=None,
 433                 scope='bottleneck_%d_x' % k)
 434             b_h = bottleneck_fn(
 435                 inputs=h_list[k],
 436                 num_outputs=self._num_units // self._groups,
 437                 is_quantized=self._is_quantized,
 438                 depth_multiplier=1,
 439                 normalizer_fn=None,
 440                 scope='bottleneck_%d_h' % k)
 441             b_x = slim.batch_norm(
 442                 b_x,
 443                 scale=True,
 444                 is_training=self._is_training,
 445                 scope='BatchNorm_%d_X' % k)
 446             b_h = slim.batch_norm(
 447                 b_h,
 448                 scale=True,
 449                 is_training=self._is_training,
 450                 scope='BatchNorm_%d_H' % k)
 451             bottleneck = b_x + b_h
 452           else:
 453             # All concats use fixed quantization ranges to prevent rescaling
 454             # at inference. Both |inputs| and |h_list| are tensors resulting
 455             # from Relu6 operations so we fix the ranges to [0, 6].
 456             bottleneck_concat = lstm_utils.quantizable_concat(
 457                 [inputs, h_list[k]],
 458                 axis=3,
 459                 is_training=False,
 460                 is_quantized=self._is_quantized,
 461                 scope='bottleneck_%d/quantized_concat' % k)
 462             bottleneck = bottleneck_fn(
 463                 inputs=bottleneck_concat,
 464                 num_outputs=self._num_units // self._groups,
 465                 is_quantized=self._is_quantized,
 466                 depth_multiplier=1,
 467                 normalizer_fn=None,
 468                 scope='bottleneck_%d' % k)
 469 
 470         if self._conv_op_overrides:
 471           conv_fn = self._conv_op_overrides[1]
 472         else:
 473           conv_fn = functools.partial(
 474               lstm_utils.quantizable_separable_conv2d,
 475               kernel_size=self._filter_size,
 476               activation_fn=None)
 477         concat = conv_fn(
 478             inputs=bottleneck,
 479             num_outputs=4 * self._num_units // self._groups,
 480             is_quantized=self._is_quantized,
 481             depth_multiplier=1,
 482             normalizer_fn=None,
 483             scope='concat_conv_%d' % k)
 484 
 485         # Since there is no activation in the previous separable conv, we
 486         # quantize here. A starting range of [-6, 6] is used because the
 487         # tensors are input to a Sigmoid function that saturates at these
 488         # ranges.
 489         concat = lstm_utils.quantize_op(
 490             concat,
 491             is_training=self._is_training,
 492             default_min=-6,
 493             default_max=6,
 494             is_quantized=self._is_quantized,
 495             scope='gates_%d/act_quant' % k)
 496 
 497         # i = input_gate, j = new_input, f = forget_gate, o = output_gate
 498         i, j, f, o = tf.split(concat, 4, 3)
 499 
 500         f_add = f + self._forget_bias
 501         f_add = lstm_utils.quantize_op(
 502             f_add,
 503             is_training=self._is_training,
 504             default_min=-6,
 505             default_max=6,
 506             is_quantized=self._is_quantized,
 507             scope='forget_gate_%d/add_quant' % k)
 508         f_act = tf.sigmoid(f_add)
 509 
 510         a = c_list[k] * f_act
 511         a = lstm_utils.quantize_op(
 512             a,
 513             is_training=self._is_training,
 514             is_quantized=self._is_quantized,
 515             scope='forget_gate_%d/mul_quant' % k)
 516 
 517         i_act = tf.sigmoid(i)
 518 
 519         j_act = self._activation(j)
 520         # The quantization range is fixed for the relu6 to ensure that zero
 521         # is exactly representable.
 522         j_act = lstm_utils.fixed_quantize_op(
 523             j_act,
 524             fixed_min=0.0,
 525             fixed_max=6.0,
 526             is_quantized=self._is_quantized,
 527             scope='new_input_%d/act_quant' % k)
 528 
 529         b = i_act * j_act
 530         b = lstm_utils.quantize_op(
 531             b,
 532             is_training=self._is_training,
 533             is_quantized=self._is_quantized,
 534             scope='input_gate_%d/mul_quant' % k)
 535 
 536         new_c = a + b
 537         # The quantization range is fixed to [0, 6] due to an optimization in
 538         # TFLite. The order of operations is as fllows:
 539         #     Add -> FakeQuant -> Relu6 -> FakeQuant -> Concat.
 540         # The fakequant ranges to the concat must be fixed to ensure all inputs
 541         # to the concat have the same range, removing the need for rescaling.
 542         # The quantization ranges input to the relu6 are propagated to its
 543         # output. Any mismatch between these two ranges will cause an error.
 544         new_c = lstm_utils.fixed_quantize_op(
 545             new_c,
 546             fixed_min=0.0,
 547             fixed_max=6.0,
 548             is_quantized=self._is_quantized,
 549             scope='new_c_%d/add_quant' % k)
 550 
 551         if not self._is_quantized:
 552           if self._scale_state:
 553             normalizer = tf.maximum(1.0,
 554                                     tf.reduce_max(new_c, axis=(1, 2, 3)) / 6)
 555             new_c /= tf.reshape(normalizer, [tf.shape(new_c)[0], 1, 1, 1])
 556           elif self._clip_state:
 557             new_c = tf.clip_by_value(new_c, -6, 6)
 558 
 559         new_c_act = self._activation(new_c)
 560         # The quantization range is fixed for the relu6 to ensure that zero
 561         # is exactly representable.
 562         new_c_act = lstm_utils.fixed_quantize_op(
 563             new_c_act,
 564             fixed_min=0.0,
 565             fixed_max=6.0,
 566             is_quantized=self._is_quantized,
 567             scope='new_c_%d/act_quant' % k)
 568 
 569         o_act = tf.sigmoid(o)
 570 
 571         new_h = new_c_act * o_act
 572         # The quantization range is fixed since it is input to a concat.
 573         # A range of [0, 6] is used since |new_h| is a product of ranges [0, 6]
 574         # and [0, 1].
 575         new_h_act = lstm_utils.fixed_quantize_op(
 576             new_h,
 577             fixed_min=0.0,
 578             fixed_max=6.0,
 579             is_quantized=self._is_quantized,
 580             scope='new_h_%d/act_quant' % k)
 581 
 582         out_bottleneck.append(bottleneck)
 583         out_c.append(new_c_act)
 584         out_h.append(new_h_act)
 585 
 586       # Since all inputs to the below concats are already quantized, we can use
 587       # a regular concat operation.
 588       new_c = tf.concat(out_c, axis=3)
 589       new_h = tf.concat(out_h, axis=3)
 590 
 591       # |bottleneck| is input to a concat with |new_h|. We must use
 592       # quantizable_concat() with a fixed range that matches |new_h|.
 593       bottleneck = lstm_utils.quantizable_concat(
 594           out_bottleneck,
 595           axis=3,
 596           is_training=False,
 597           is_quantized=self._is_quantized,
 598           scope='out_bottleneck/quantized_concat')
 599 
 600       # summary of cell output and new state
 601       if self._viz_gates:
 602         slim.summaries.add_histogram_summary(new_h, 'cell_output')
 603         slim.summaries.add_histogram_summary(new_c, 'cell_state')
 604 
 605       output = new_h
 606       if self._output_bottleneck:
 607         output = lstm_utils.quantizable_concat(
 608             [new_h, bottleneck],
 609             axis=3,
 610             is_training=False,
 611             is_quantized=self._is_quantized,
 612             scope='new_output/quantized_concat')
 613 
 614       # reflatten state to store it
 615       if self._flatten_state:
 616         new_c = tf.reshape(new_c, [-1, self._param_count], name='lstm_c')
 617         new_h = tf.reshape(new_h, [-1, self._param_count], name='lstm_h')
 618 
 619       # Set nodes to be under raw_outputs/ name scope for tfmini export.
 620       with tf.name_scope(None):
 621         new_c = tf.identity(new_c, name='raw_outputs/lstm_c')
 622         new_h = tf.identity(new_h, name='raw_outputs/lstm_h')
 623       states_and_output = contrib_rnn.LSTMStateTuple(new_c, new_h)
 624 
 625       return output, states_and_output
 626 
 627   def init_state(self, state_name, batch_size, dtype, learned_state=False):
 628     """Creates an initial state compatible with this cell.
 629 
 630     Args:
 631       state_name: name of the state tensor
 632       batch_size: model batch size
 633       dtype: dtype for the tensor values i.e. tf.float32
 634       learned_state: whether the initial state should be learnable. If false,
 635         the initial state is set to all 0's
 636 
 637     Returns:
 638       ret: the created initial state
 639     """
 640     state_size = (
 641         self.state_size_flat if self._flatten_state else self.state_size)
 642     # list of 2 zero tensors or variables tensors,
 643     # depending on if learned_state is true
 644     # pylint: disable=g-long-ternary,g-complex-comprehension
 645     ret_flat = [(contrib_variables.model_variable(
 646         state_name + str(i),
 647         shape=s,
 648         dtype=dtype,
 649         initializer=tf.truncated_normal_initializer(stddev=0.03))
 650                  if learned_state else tf.zeros(
 651                      [batch_size] + s, dtype=dtype, name=state_name))
 652                 for i, s in enumerate(state_size)]
 653 
 654     # duplicates initial state across the batch axis if it's learned
 655     if learned_state:
 656       ret_flat = [tf.stack([tensor for i in range(int(batch_size))])
 657                   for tensor in ret_flat]
 658     for s, r in zip(state_size, ret_flat):
 659       r = tf.reshape(r, [-1] + s)
 660     ret = tf.nest.pack_sequence_as(structure=[1, 1], flat_sequence=ret_flat)
 661     return ret

 663   def pre_bottleneck(self, inputs, state, input_index):
 664     """Apply pre-bottleneck projection to inputs.
 665 
 666     Pre-bottleneck operation maps features of different channels into the same
 667     dimension. The purpose of this op is to share the features from both large
 668     and small models in the same LSTM cell.
 669 
 670     Args:
 671       inputs: 4D Tensor with shape [batch_size x width x height x input_size].
 672       state: 4D Tensor with shape [batch_size x width x height x state_size].
 673       input_index: integer index indicating which base features the inputs
 674         correspoding to.
 675 
 676     Returns:
 677       inputs: pre-bottlenecked inputs.
 678     Raises:
 679       ValueError: If pre_bottleneck is not set or inputs is not rank 4.
 680     """
 681     # Sometimes state is a tuple, in which case it cannot be modified, e.g.
 682     # during training, tf.contrib.training.SequenceQueueingStateSaver
 683     # returns the state as a tuple. This should not be an issue since we
 684     # only need to modify state[1] during export, when state should be a
 685     # list.
 686     if not self._pre_bottleneck:
 687       raise ValueError('Only applied when pre_bottleneck is set to true.')
 688     if len(inputs.shape) != 4:
 689       raise ValueError('Expect a rank 4 feature tensor.')
 690     if not self._flatten_state and len(state.shape) != 4:
 691       raise ValueError('Expect rank 4 state tensor.')
 692     if self._flatten_state and len(state.shape) != 2:
 693       raise ValueError('Expect rank 2 state tensor when flatten_state is set.')
 694 
 695     with tf.name_scope(None):
 696       state = tf.identity(
 697           state, name='raw_inputs/init_lstm_h_%d' % (input_index + 1))
 698     if self._flatten_state:
 699       batch_size = inputs.shape[0]
 700       height = inputs.shape[1]
 701       width = inputs.shape[2]
 702       state = tf.reshape(state, [batch_size, height, width, -1])
 703     with tf.variable_scope('conv_lstm_cell', reuse=tf.AUTO_REUSE):
 704       state_split = tf.split(state, self._groups, axis=3)
 705       with tf.variable_scope('bottleneck_%d' % input_index):
 706         bottleneck_out = []
 707         for k in range(self._groups):
 708           with tf.variable_scope('group_%d' % k):
 709             bottleneck_out.append(
 710                 lstm_utils.quantizable_separable_conv2d(
 711                     lstm_utils.quantizable_concat(
 712                         [inputs, state_split[k]],
 713                         axis=3,
 714                         is_training=self._is_training,
 715                         is_quantized=self._is_quantized,
 716                         scope='quantized_concat'),
 717                     self.output_size[-1] / self._groups,
 718                     self._filter_size,
 719                     is_quantized=self._is_quantized,
 720                     depth_multiplier=1,
 721                     activation_fn=tf.nn.relu6,
 722                     normalizer_fn=None,
 723                     scope='project'))
 724         inputs = lstm_utils.quantizable_concat(
 725             bottleneck_out,
 726             axis=3,
 727             is_training=self._is_training,
 728             is_quantized=self._is_quantized,
 729             scope='bottleneck_out/quantized_concat')
 730       # For exporting inference graph, we only mark the first timestep.
 731       with tf.name_scope(None):
 732         inputs = tf.identity(
 733             inputs, name='raw_outputs/base_endpoint_%d' % (input_index + 1))
 734     return inputs
