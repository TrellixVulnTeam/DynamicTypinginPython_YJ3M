  31 def rnn_decoder(decoder_inputs,
  32                 initial_state,
  33                 cell,
  34                 loop_function=None,
  35                 scope=None):
  36   """RNN decoder for the LSTM-SSD model.
  37 
  38   This decoder returns a list of all states, rather than only the final state.
  39   Args:
  40     decoder_inputs: A list of 4D Tensors with shape [batch_size x input_size].
  41     initial_state: 2D Tensor with shape [batch_size x cell.state_size].
  42     cell: rnn_cell.RNNCell defining the cell function and size.
  43     loop_function: If not None, this function will be applied to the i-th output
  44       in order to generate the i+1-st input, and decoder_inputs will be ignored,
  45       except for the first element ("GO" symbol). This can be used for decoding,
  46       but also for training to emulate http://arxiv.org/abs/1506.03099.
  47       Signature -- loop_function(prev, i) = next
  48         * prev is a 2D Tensor of shape [batch_size x output_size],
  49         * i is an integer, the step number (when advanced control is needed),
  50         * next is a 2D Tensor of shape [batch_size x input_size].
  51     scope: optional VariableScope for the created subgraph.
  52   Returns:
  53     A tuple of the form (outputs, state), where:
  54       outputs: A list of the same length as decoder_inputs of 4D Tensors with
  55         shape [batch_size x output_size] containing generated outputs.
  56       states: A list of the same length as decoder_inputs of the state of each
  57         cell at each time-step. It is a 2D Tensor of shape
  58         [batch_size x cell.state_size].
  59   """
  60   with tf.variable_scope(scope) if scope else _NoVariableScope():
  61     state_tuple = initial_state
  62     outputs = []
  63     states = []
  64     prev = None
  65     for local_step, decoder_input in enumerate(decoder_inputs):
  66       if loop_function is not None and prev is not None:
  67         with tf.variable_scope('loop_function', reuse=True):
  68           decoder_input = loop_function(prev, local_step)
  69       output, state_tuple = cell(decoder_input, state_tuple)
  70       outputs.append(output)
  71       states.append(state_tuple)
  72       if loop_function is not None:
  73         prev = output
  74   return outputs, states

  76 def multi_input_rnn_decoder(decoder_inputs,
  77                             initial_state,
  78                             cell,
  79                             sequence_step,
  80                             selection_strategy='RANDOM',
  81                             is_training=None,
  82                             is_quantized=False,
  83                             preprocess_fn_list=None,
  84                             pre_bottleneck=False,
  85                             flatten_state=False,
  86                             scope=None):
  87   """RNN decoder for the Interleaved LSTM-SSD model.
  88 
  89   This decoder takes multiple sequences of inputs and selects the input to feed
  90   to the rnn at each timestep using its selection_strategy, which can be random,
  91   learned, or deterministic.
  92   This decoder returns a list of all states, rather than only the final state.
  93   Args:
  94     decoder_inputs: A list of lists of 2D Tensors [batch_size x input_size].
  95     initial_state: 2D Tensor with shape [batch_size x cell.state_size].
  96     cell: rnn_cell.RNNCell defining the cell function and size.
  97     sequence_step: Tensor [batch_size] of the step number of the first elements
  98       in the sequence.
  99     selection_strategy: Method for picking the decoder_input to use at each
 100       timestep. Must be 'RANDOM', 'SKIPX' for integer X,  where X is the number
 101       of times to use the second input before using the first.
 102     is_training: boolean, whether the network is training. When using learned
 103       selection, attempts exploration if training.
 104     is_quantized: flag to enable/disable quantization mode.
 105     preprocess_fn_list: List of functions accepting two tensor arguments: one
 106       timestep of decoder_inputs and the lstm state. If not None,
 107       decoder_inputs[i] will be updated with preprocess_fn[i] at the start of
 108       each timestep.
 109     pre_bottleneck: if True, use separate bottleneck weights for each sequence.
 110       Useful when input sequences have differing numbers of channels. Final
 111       bottlenecks will have the same dimension.
 112     flatten_state: Whether the LSTM state is flattened.
 113     scope: optional VariableScope for the created subgraph.
 114   Returns:
 115     A tuple of the form (outputs, state), where:
 116       outputs: A list of the same length as decoder_inputs of 2D Tensors with
 117         shape [batch_size x output_size] containing generated outputs.
 118       states: A list of the same length as decoder_inputs of the state of each
 119         cell at each time-step. It is a 2D Tensor of shape
 120         [batch_size x cell.state_size].
 121   Raises:
 122     ValueError: If selection_strategy is not recognized or unexpected unroll
 123       length.
 124   """
 125   if flatten_state and len(decoder_inputs[0]) > 1:
 126     raise ValueError('In export mode, unroll length should not be more than 1')
 127   with tf.variable_scope(scope) if scope else _NoVariableScope():
 128     state_tuple = initial_state
 129     outputs = []
 130     states = []
 131     batch_size = decoder_inputs[0][0].shape[0].value
 132     num_sequences = len(decoder_inputs)
 133     sequence_length = len(decoder_inputs[0])
 134 
 135     for local_step in range(sequence_length):
 136       for sequence_index in range(num_sequences):
 137         if preprocess_fn_list is not None:
 138           decoder_inputs[sequence_index][local_step] = (
 139               preprocess_fn_list[sequence_index](
 140                   decoder_inputs[sequence_index][local_step], state_tuple[0]))
 141         if pre_bottleneck:
 142           decoder_inputs[sequence_index][local_step] = cell.pre_bottleneck(
 143               inputs=decoder_inputs[sequence_index][local_step],
 144               state=state_tuple[1],
 145               input_index=sequence_index)
 146 
 147       action = generate_action(selection_strategy, local_step, sequence_step,
 148                                [batch_size, 1, 1, 1])
 149       inputs, _ = (
 150           select_inputs(decoder_inputs, action, local_step, is_training,
 151                         is_quantized))
 152       # Mark base network endpoints under raw_inputs/
 153       with tf.name_scope(None):
 154         inputs = tf.identity(inputs, 'raw_inputs/base_endpoint')
 155       output, state_tuple_out = cell(inputs, state_tuple)
 156       state_tuple = select_state(state_tuple, state_tuple_out, action)
 157 
 158       outputs.append(output)
 159       states.append(state_tuple)
 160   return outputs, states

 163 def generate_action(selection_strategy, local_step, sequence_step,
 164                     action_shape):
 165   """Generate current (binary) action based on selection strategy.
 166 
 167   Args:
 168     selection_strategy: Method for picking the decoder_input to use at each
 169       timestep. Must be 'RANDOM', 'SKIPX' for integer X,  where X is the number
 170       of times to use the second input before using the first.
 171     local_step: Tensor [batch_size] of the step number within the current
 172       unrolled batch.
 173     sequence_step: Tensor [batch_size] of the step number of the first elements
 174       in the sequence.
 175     action_shape: The shape of action tensor to be generated.
 176 
 177   Returns:
 178     A tensor of shape action_shape, each element is an individual action.
 179 
 180   Raises:
 181     ValueError: if selection_strategy is not supported or if 'SKIP' is not
 182       followed by numerics.
 183   """
 184   if selection_strategy.startswith('RANDOM'):
 185     action = tf.random.uniform(action_shape, maxval=2, dtype=tf.int32)
 186     action = tf.minimum(action, 1)
 187 
 188     # First step always runs large network.
 189     if local_step == 0 and sequence_step is not None:
 190       action *= tf.minimum(
 191           tf.reshape(tf.cast(sequence_step, tf.int32), action_shape), 1)
 192   elif selection_strategy.startswith('SKIP'):
 193     inter_count = int(selection_strategy[4:])
 194     if local_step % (inter_count + 1) == 0:
 195       action = tf.zeros(action_shape)
 196     else:
 197       action = tf.ones(action_shape)
 198   else:
 199     raise ValueError('Selection strategy %s not recognized' %
 200                      selection_strategy)
 201   return tf.cast(action, tf.int32)

 204 def select_inputs(decoder_inputs, action, local_step, is_training, is_quantized,
 205                   get_alt_inputs=False):
 206   """Selects sequence from decoder_inputs based on 1D actions.
 207 
 208   Given multiple input batches, creates a single output batch by
 209   selecting from the action[i]-ith input for the i-th batch element.
 210 
 211   Args:
 212     decoder_inputs: A 2-D list of tensor inputs.
 213     action: A tensor of shape [batch_size]. Each element corresponds to an index
 214       of decoder_inputs to choose.
 215     local_step: The current timestep.
 216     is_training: boolean, whether the network is training. When using learned
 217       selection, attempts exploration if training.
 218     is_quantized: flag to enable/disable quantization mode.
 219     get_alt_inputs: Whether the non-chosen inputs should also be returned.
 220 
 221   Returns:
 222     The constructed output. Also outputs the elements that were not chosen
 223     if get_alt_inputs is True, otherwise None.
 224 
 225   Raises:
 226     ValueError: if the decoder inputs contains other than two sequences.
 227   """
 228   num_seqs = len(decoder_inputs)
 229   if not num_seqs == 2:
 230     raise ValueError('Currently only supports two sets of inputs.')
 231   stacked_inputs = tf.stack(
 232       [decoder_inputs[seq_index][local_step] for seq_index in range(num_seqs)],
 233       axis=-1)
 234   action_index = tf.one_hot(action, num_seqs)
 235   selected_inputs = (
 236       lstm_utils.quantize_op(stacked_inputs * action_index, is_training,
 237                              is_quantized, scope='quant_selected_inputs'))
 238   inputs = tf.reduce_sum(selected_inputs, axis=-1)
 239   inputs_alt = None
 240   # Only works for 2 models.
 241   if get_alt_inputs:
 242     # Reverse of action_index.
 243     action_index_alt = tf.one_hot(action, num_seqs, on_value=0.0, off_value=1.0)
 244     selected_inputs = (
 245         lstm_utils.quantize_op(stacked_inputs * action_index_alt, is_training,
 246                                is_quantized, scope='quant_selected_inputs_alt'))
 247     inputs_alt = tf.reduce_sum(selected_inputs, axis=-1)
 248   return inputs, inputs_alt

 250 def select_state(previous_state, new_state, action):
 251   """Select state given action.
 252 
 253   Currently only supports binary action. If action is 0, it means the state is
 254   generated from the large model, and thus we will update the state. Otherwise,
 255   if the action is 1, it means the state is generated from the small model, and
 256   in interleaved model, we skip this state update.
 257 
 258   Args:
 259     previous_state: A state tuple representing state from previous step.
 260     new_state: A state tuple representing newly computed state.
 261     action: A tensor the same shape as state.
 262 
 263   Returns:
 264     A state tuple selected based on the given action.
 265   """
 266   action = tf.cast(action, tf.float32)
 267   state_c = previous_state[0] * action + new_state[0] * (1 - action)
 268   state_h = previous_state[1] * action + new_state[1] * (1 - action)
 269   return (state_c, state_h)
