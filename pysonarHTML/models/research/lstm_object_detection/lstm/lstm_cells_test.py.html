  29   def test_run_lstm_cell(self):
  30     filter_size = [3, 3]
  31     output_size = [10, 10]
  32     num_units = 15
  33     state_name = 'lstm_state'
  34     batch_size = 4
  35     dtype = tf.float32
  36     learned_state = False
  37 
  38     inputs = tf.zeros([4, 10, 10, 3], dtype=tf.float32)
  39     cell = lstm_cells.BottleneckConvLSTMCell(
  40         filter_size=filter_size,
  41         output_size=output_size,
  42         num_units=num_units)
  43     init_state = cell.init_state(
  44         state_name, batch_size, dtype, learned_state)
  45     output, state_tuple = cell(inputs, init_state)
  46     self.assertAllEqual([4, 10, 10, 15], output.shape.as_list())
  47     self.assertAllEqual([4, 10, 10, 15], state_tuple[0].shape.as_list())
  48     self.assertAllEqual([4, 10, 10, 15], state_tuple[1].shape.as_list())
