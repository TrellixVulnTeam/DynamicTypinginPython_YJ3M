  30 class MockRnnCell(contrib_rnn.RNNCell):
  31 
  32   def __init__(self, input_size, num_units):
  33     self._input_size = input_size
  34     self._num_units = num_units
  35     self._filter_size = [3, 3]
  36 
  37   def __call__(self, inputs, state_tuple):
  38     outputs = tf.concat([inputs, state_tuple[0]], axis=3)
  39     new_state_tuple = (tf.multiply(state_tuple[0], 2), state_tuple[1])
  40     return outputs, new_state_tuple
  41 
  42   def state_size(self):
  43     return self._num_units
  44 
  45   def output_size(self):
  46     return self._input_size + self._num_units
  47 
  48   def pre_bottleneck(self, inputs, state, input_index):
  49     with tf.variable_scope('bottleneck_%d' % input_index, reuse=tf.AUTO_REUSE):
  50       inputs = contrib_layers.separable_conv2d(
  51           tf.concat([inputs, state], 3),
  52           self._input_size,
  53           self._filter_size,
  54           depth_multiplier=1,
  55           activation_fn=tf.nn.relu6,
  56           normalizer_fn=None)
  57     return inputs

  60 class RnnDecoderTest(tf.test.TestCase):

  62   def test_rnn_decoder_single_unroll(self):
  63     batch_size = 2
  64     num_unroll = 1
  65     num_units = 64
  66     width = 8
  67     height = 10
  68     input_channels = 128
  69 
  70     initial_state = tf.random_normal((batch_size, width, height, num_units))
  71     inputs = tf.random_normal([batch_size, width, height, input_channels])
  72 
  73     rnn_cell = MockRnnCell(input_channels, num_units)
  74     outputs, states = rnn_decoder.rnn_decoder(
  75         decoder_inputs=[inputs] * num_unroll,
  76         initial_state=(initial_state, initial_state),
  77         cell=rnn_cell)
  78 
  79     self.assertEqual(len(outputs), num_unroll)
  80     self.assertEqual(len(states), num_unroll)
  81     with tf.Session() as sess:
  82       sess.run(tf.global_variables_initializer())
  83       results = sess.run((outputs, states, inputs, initial_state))
  84       outputs_results = results[0]
  85       states_results = results[1]
  86       inputs_results = results[2]
  87       initial_states_results = results[3]
  88       self.assertEqual(outputs_results[0].shape,
  89                        (batch_size, width, height, input_channels + num_units))
  90       self.assertAllEqual(
  91           outputs_results[0],
  92           np.concatenate((inputs_results, initial_states_results), axis=3))
  93       self.assertEqual(states_results[0][0].shape,
  94                        (batch_size, width, height, num_units))
  95       self.assertEqual(states_results[0][1].shape,
  96                        (batch_size, width, height, num_units))
  97       self.assertAllEqual(states_results[0][0],
  98                           np.multiply(initial_states_results, 2.0))
  99       self.assertAllEqual(states_results[0][1], initial_states_results)

 245   def test_rnn_decoder_multiple_unroll_with_skip(self):
 246     batch_size = 2
 247     num_unroll = 5
 248     num_units = 12
 249     width = 8
 250     height = 10
 251     input_channels_large = 24
 252     input_channels_small = 12
 253     bottleneck_channels = 20
 254     skip = 2
 255 
 256     initial_state_c = tf.random_normal((batch_size, width, height, num_units))
 257     initial_state_h = tf.random_normal((batch_size, width, height, num_units))
 258     initial_state = (initial_state_c, initial_state_h)
 259     inputs_large = tf.random_normal(
 260         [batch_size, width, height, input_channels_large])
 261     inputs_small = tf.random_normal(
 262         [batch_size, width, height, input_channels_small])
 263 
 264     rnn_cell = MockRnnCell(bottleneck_channels, num_units)
 265     outputs, states = rnn_decoder.multi_input_rnn_decoder(
 266         decoder_inputs=[[inputs_large] * num_unroll,
 267                         [inputs_small] * num_unroll],
 268         initial_state=initial_state,
 269         cell=rnn_cell,
 270         sequence_step=tf.zeros([batch_size]),
 271         pre_bottleneck=True,
 272         selection_strategy='SKIP%d' % skip)
 273 
 274     self.assertEqual(len(outputs), num_unroll)
 275     self.assertEqual(len(states), num_unroll)
 276     with tf.Session() as sess:
 277       sess.run(tf.global_variables_initializer())
 278       results = sess.run(
 279           (outputs, states, inputs_large, inputs_small, initial_state))
 280       outputs_results = results[0]
 281       states_results = results[1]
 282       initial_states_results = results[4]
 283 
 284       for i in range(num_unroll):
 285         self.assertEqual(
 286             outputs_results[i].shape,
 287             (batch_size, width, height, bottleneck_channels + num_units))
 288         self.assertEqual(states_results[i][0].shape,
 289                          (batch_size, width, height, num_units))
 290         self.assertEqual(states_results[i][1].shape,
 291                          (batch_size, width, height, num_units))
 292 
 293         previous_state = (
 294             initial_states_results if i == 0 else states_results[i - 1])
 295         # State only updates during key frames
 296         if i % (skip + 1) == 0:
 297           self.assertAllEqual(states_results[i][0],
 298                               np.multiply(previous_state[0], 2))
 299           self.assertAllEqual(states_results[i][1], previous_state[1])
 300         else:
 301           self.assertAllEqual(states_results[i][0], previous_state[0])
 302           self.assertAllEqual(states_results[i][1], previous_state[1])
