  25 class QuantizableUtilsTest(tf.test.TestCase):
  26 
  27   def test_quantizable_concat_is_training(self):
  28     inputs_1 = tf.zeros([4, 10, 10, 1], dtype=tf.float32)
  29     inputs_2 = tf.ones([4, 10, 10, 2], dtype=tf.float32)
  30     concat_in_train = utils.quantizable_concat([inputs_1, inputs_2],
  31                                                axis=3,
  32                                                is_training=True)
  33     self.assertAllEqual([4, 10, 10, 3], concat_in_train.shape.as_list())
  34     self._check_min_max_ema(tf.get_default_graph())
  35     self._check_min_max_vars(tf.get_default_graph())
  36 
  37   def test_quantizable_concat_inference(self):
  38     inputs_1 = tf.zeros([4, 10, 10, 1], dtype=tf.float32)
  39     inputs_2 = tf.ones([4, 10, 10, 2], dtype=tf.float32)
  40     concat_in_train = utils.quantizable_concat([inputs_1, inputs_2],
  41                                                axis=3,
  42                                                is_training=False)
  43     self.assertAllEqual([4, 10, 10, 3], concat_in_train.shape.as_list())
  44     self._check_no_min_max_ema(tf.get_default_graph())
  45     self._check_min_max_vars(tf.get_default_graph())
  46 
  47   def test_quantizable_concat_not_quantized_is_training(self):
  48     inputs_1 = tf.zeros([4, 10, 10, 1], dtype=tf.float32)
  49     inputs_2 = tf.ones([4, 10, 10, 2], dtype=tf.float32)
  50     concat_in_train = utils.quantizable_concat([inputs_1, inputs_2],
  51                                                axis=3,
  52                                                is_training=True,
  53                                                is_quantized=False)
  54     self.assertAllEqual([4, 10, 10, 3], concat_in_train.shape.as_list())
  55     self._check_no_min_max_ema(tf.get_default_graph())
  56     self._check_no_min_max_vars(tf.get_default_graph())
  57 
  58   def test_quantizable_concat_not_quantized_inference(self):
  59     inputs_1 = tf.zeros([4, 10, 10, 1], dtype=tf.float32)
  60     inputs_2 = tf.ones([4, 10, 10, 2], dtype=tf.float32)
  61     concat_in_train = utils.quantizable_concat([inputs_1, inputs_2],
  62                                                axis=3,
  63                                                is_training=False,
  64                                                is_quantized=False)
  65     self.assertAllEqual([4, 10, 10, 3], concat_in_train.shape.as_list())
  66     self._check_no_min_max_ema(tf.get_default_graph())
  67     self._check_no_min_max_vars(tf.get_default_graph())
  68 
  69   def test_quantize_op_is_training(self):
  70     inputs = tf.zeros([4, 10, 10, 128], dtype=tf.float32)
  71     outputs = utils.quantize_op(inputs)
  72     self.assertAllEqual(inputs.shape.as_list(), outputs.shape.as_list())
  73     self._check_min_max_ema(tf.get_default_graph())
  74     self._check_min_max_vars(tf.get_default_graph())
  75 
  76   def test_quantize_op_inference(self):
  77     inputs = tf.zeros([4, 10, 10, 128], dtype=tf.float32)
  78     outputs = utils.quantize_op(inputs, is_training=False)
  79     self.assertAllEqual(inputs.shape.as_list(), outputs.shape.as_list())
  80     self._check_no_min_max_ema(tf.get_default_graph())
  81     self._check_min_max_vars(tf.get_default_graph())
  82 
  83   def test_fixed_quantize_op(self):
  84     inputs = tf.zeros([4, 10, 10, 128], dtype=tf.float32)
  85     outputs = utils.fixed_quantize_op(inputs)
  86     self.assertAllEqual(inputs.shape.as_list(), outputs.shape.as_list())
  87     self._check_no_min_max_ema(tf.get_default_graph())
  88     self._check_no_min_max_vars(tf.get_default_graph())
  89 
  90   def _check_min_max_vars(self, graph):
  91     op_types = [op.type for op in graph.get_operations()]
  92     self.assertTrue(
  93         any('FakeQuantWithMinMaxVars' in op_type for op_type in op_types))
  94 
  95   def _check_min_max_ema(self, graph):
  96     op_names = [op.name for op in graph.get_operations()]
  97     self.assertTrue(any('AssignMinEma' in name for name in op_names))
  98     self.assertTrue(any('AssignMaxEma' in name for name in op_names))
  99     self.assertTrue(any('SafeQuantRangeMin' in name for name in op_names))
 100     self.assertTrue(any('SafeQuantRangeMax' in name for name in op_names))
 101 
 102   def _check_no_min_max_vars(self, graph):
 103     op_types = [op.type for op in graph.get_operations()]
 104     self.assertFalse(
 105         any('FakeQuantWithMinMaxVars' in op_type for op_type in op_types))
 106 
 107   def _check_no_min_max_ema(self, graph):
 108     op_names = [op.name for op in graph.get_operations()]
 109     self.assertFalse(any('AssignMinEma' in name for name in op_names))
 110     self.assertFalse(any('AssignMaxEma' in name for name in op_names))
 111     self.assertFalse(any('SafeQuantRangeMin' in name for name in op_names))
 112     self.assertFalse(any('SafeQuantRangeMax' in name for name in op_names))

 115 class QuantizableSeparableConv2dTest(tf.test.TestCase):
 116 
 117   def test_quantizable_separable_conv2d(self):
 118     inputs = tf.zeros([4, 10, 10, 128], dtype=tf.float32)
 119     num_outputs = 64
 120     kernel_size = [3, 3]
 121     scope = 'QuantSeparable'
 122     outputs = utils.quantizable_separable_conv2d(
 123         inputs, num_outputs, kernel_size, scope=scope)
 124     self.assertAllEqual([4, 10, 10, num_outputs], outputs.shape.as_list())
 125     self._check_depthwise_bias_add(tf.get_default_graph(), scope)
 126 
 127   def test_quantizable_separable_conv2d_not_quantized(self):
 128     inputs = tf.zeros([4, 10, 10, 128], dtype=tf.float32)
 129     num_outputs = 64
 130     kernel_size = [3, 3]
 131     scope = 'QuantSeparable'
 132     outputs = utils.quantizable_separable_conv2d(
 133         inputs, num_outputs, kernel_size, is_quantized=False, scope=scope)
 134     self.assertAllEqual([4, 10, 10, num_outputs], outputs.shape.as_list())
 135     self._check_no_depthwise_bias_add(tf.get_default_graph(), scope)
 136 
 137   def _check_depthwise_bias_add(self, graph, scope):
 138     op_names = [op.name for op in graph.get_operations()]
 139     self.assertTrue(
 140         any('%s_bias/BiasAdd' % scope in name for name in op_names))
 141 
 142   def _check_no_depthwise_bias_add(self, graph, scope):
 143     op_names = [op.name for op in graph.get_operations()]
 144     self.assertFalse(
 145         any('%s_bias/BiasAdd' % scope in name for name in op_names))
