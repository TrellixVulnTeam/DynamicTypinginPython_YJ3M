  27 def _quant_var(
  28     name,
  29     initializer_val,
  30     vars_collection=tf.GraphKeys.MOVING_AVERAGE_VARIABLES,
  31 ):
  32   """Create an var for storing the min/max quantization range."""
  33   return contrib_framework.model_variable(
  34       name,
  35       shape=[],
  36       initializer=tf.constant_initializer(initializer_val),
  37       collections=[vars_collection],
  38       trainable=False)

  41 def quantizable_concat(inputs,
  42                        axis,
  43                        is_training,
  44                        is_quantized=True,
  45                        default_min=0,
  46                        default_max=6,
  47                        ema_decay=0.999,
  48                        scope='quantized_concat'):
  49   """Concat replacement with quantization option.
  50 
  51   Allows concat inputs to share the same min max ranges,
  52   from experimental/gazelle/synthetic/model/tpu/utils.py.
  53 
  54   Args:
  55     inputs: list of tensors to concatenate.
  56     axis: dimension along which to concatenate.
  57     is_training: true if the graph is a training graph.
  58     is_quantized: flag to enable/disable quantization.
  59     default_min: default min value for fake quant op.
  60     default_max: default max value for fake quant op.
  61     ema_decay: the moving average decay for the quantization variables.
  62     scope: Optional scope for variable_scope.
  63 
  64   Returns:
  65     Tensor resulting from concatenation of input tensors
  66   """
  67   if is_quantized:
  68     with tf.variable_scope(scope):
  69       tf.logging.info('inputs: {}'.format(inputs))
  70       for t in inputs:
  71         tf.logging.info(t)
  72 
  73       min_var = _quant_var('min', default_min)
  74       max_var = _quant_var('max', default_max)
  75       if not is_training:
  76         # If we are building an eval graph just use the values in the variables.
  77         quant_inputs = [
  78             tf.fake_quant_with_min_max_vars(t, min_var, max_var) for t in inputs
  79         ]
  80         tf.logging.info('min_val: {}'.format(min_var))
  81         tf.logging.info('max_val: {}'.format(max_var))
  82       else:
  83         concat_tensors = tf.concat(inputs, axis=axis)
  84         tf.logging.info('concat_tensors: {}'.format(concat_tensors))
  85         # TFLite requires that 0.0 is always in the [min; max] range.
  86         range_min = tf.minimum(
  87             tf.reduce_min(concat_tensors), 0.0, name='SafeQuantRangeMin')
  88         range_max = tf.maximum(
  89             tf.reduce_max(concat_tensors), 0.0, name='SafeQuantRangeMax')
  90         # Otherwise we need to keep track of the moving averages of the min and
  91         # of the elements of the input tensor max.
  92         min_val = moving_averages.assign_moving_average(
  93             min_var,
  94             range_min,
  95             ema_decay,
  96             name='AssignMinEma')
  97         max_val = moving_averages.assign_moving_average(
  98             max_var,
  99             range_max,
 100             ema_decay,
 101             name='AssignMaxEma')
 102         tf.logging.info('min_val: {}'.format(min_val))
 103         tf.logging.info('max_val: {}'.format(max_val))
 104         quant_inputs = [
 105             tf.fake_quant_with_min_max_vars(t, min_val, max_val) for t in inputs
 106         ]
 107       tf.logging.info('quant_inputs: {}'.format(quant_inputs))
 108       outputs = tf.concat(quant_inputs, axis=axis)
 109       tf.logging.info('outputs: {}'.format(outputs))
 110   else:
 111     outputs = tf.concat(inputs, axis=axis)
 112   return outputs

 115 def quantizable_separable_conv2d(inputs,
 116                                  num_outputs,
 117                                  kernel_size,
 118                                  is_quantized=True,
 119                                  depth_multiplier=1,
 120                                  stride=1,
 121                                  activation_fn=tf.nn.relu6,
 122                                  normalizer_fn=None,
 123                                  weights_initializer=None,
 124                                  pointwise_initializer=None,
 125                                  scope=None):
 126   """Quantization friendly backward compatible separable conv2d.
 127 
 128   This op has the same API is separable_conv2d. The main difference is that an
 129   additional BiasAdd is manually inserted after the depthwise conv, such that
 130   the depthwise bias will not have name conflict with pointwise bias. The
 131   motivation of this op is that quantization script need BiasAdd in order to
 132   recognize the op, in which a native call to separable_conv2d do not create
 133   for the depthwise conv.
 134 
 135   Args:
 136     inputs: A tensor of size [batch_size, height, width, channels].
 137     num_outputs: The number of pointwise convolution output filters. If is
 138       None, then we skip the pointwise convolution stage.
 139     kernel_size: A list of length 2: [kernel_height, kernel_width] of the
 140       filters. Can be an int if both values are the same.
 141     is_quantized: flag to enable/disable quantization.
 142     depth_multiplier: The number of depthwise convolution output channels for
 143       each input channel. The total number of depthwise convolution output
 144       channels will be equal to num_filters_in * depth_multiplier.
 145     stride: A list of length 2: [stride_height, stride_width], specifying the
 146       depthwise convolution stride. Can be an int if both strides are the same.
 147     activation_fn: Activation function. The default value is a ReLU function.
 148       Explicitly set it to None to skip it and maintain a linear activation.
 149     normalizer_fn: Normalization function to use instead of biases.
 150     weights_initializer: An initializer for the depthwise weights.
 151     pointwise_initializer: An initializer for the pointwise weights.
 152     scope: Optional scope for variable_scope.
 153 
 154   Returns:
 155     Tensor resulting from concatenation of input tensors
 156   """
 157   if is_quantized:
 158     outputs = contrib_layers.separable_conv2d(
 159         inputs,
 160         None,
 161         kernel_size,
 162         depth_multiplier=depth_multiplier,
 163         stride=1,
 164         activation_fn=None,
 165         normalizer_fn=None,
 166         biases_initializer=None,
 167         weights_initializer=weights_initializer,
 168         pointwise_initializer=None,
 169         scope=scope)
 170     outputs = contrib_layers.bias_add(
 171         outputs, trainable=True, scope='%s_bias' % scope)
 172     outputs = contrib_layers.conv2d(
 173         outputs,
 174         num_outputs, [1, 1],
 175         activation_fn=activation_fn,
 176         stride=stride,
 177         normalizer_fn=normalizer_fn,
 178         weights_initializer=pointwise_initializer,
 179         scope=scope)
 180   else:
 181     outputs = contrib_layers.separable_conv2d(
 182         inputs,
 183         num_outputs,
 184         kernel_size,
 185         depth_multiplier=depth_multiplier,
 186         stride=stride,
 187         activation_fn=activation_fn,
 188         normalizer_fn=normalizer_fn,
 189         weights_initializer=weights_initializer,
 190         pointwise_initializer=pointwise_initializer,
 191         scope=scope)
 192   return outputs

 195 def quantize_op(inputs,
 196                 is_training=True,
 197                 is_quantized=True,
 198                 default_min=0,
 199                 default_max=6,
 200                 ema_decay=0.999,
 201                 scope='quant'):
 202   """Inserts a fake quantization op after inputs.
 203 
 204   Args:
 205     inputs: A tensor of size [batch_size, height, width, channels].
 206     is_training: true if the graph is a training graph.
 207     is_quantized: flag to enable/disable quantization.
 208     default_min: default min value for fake quant op.
 209     default_max: default max value for fake quant op.
 210     ema_decay: the moving average decay for the quantization variables.
 211     scope: Optional scope for variable_scope.
 212 
 213   Returns:
 214     Tensor resulting from quantizing the input tensors.
 215   """
 216   if not is_quantized:
 217     return inputs
 218 
 219   with tf.variable_scope(scope):
 220     min_var = _quant_var('min', default_min)
 221     max_var = _quant_var('max', default_max)
 222     if not is_training:
 223       # Just use variables in the checkpoint.
 224       return tf.fake_quant_with_min_max_vars(inputs, min_var, max_var)
 225 
 226     # While training, collect EMAs of ranges seen, store in min_var, max_var.
 227     # TFLite requires that 0.0 is always in the [min; max] range.
 228     range_min = tf.minimum(tf.reduce_min(inputs), 0.0, 'SafeQuantRangeMin')
 229     # We set the lower_bound of max_range to prevent range collapse.
 230     range_max = tf.maximum(tf.reduce_max(inputs), 1e-5, 'SafeQuantRangeMax')
 231     min_val = moving_averages.assign_moving_average(
 232         min_var, range_min, ema_decay, name='AssignMinEma')
 233     max_val = moving_averages.assign_moving_average(
 234         max_var, range_max, ema_decay, name='AssignMaxEma')
 235     return tf.fake_quant_with_min_max_vars(inputs, min_val, max_val)

 238 def fixed_quantize_op(inputs, is_quantized=True,
 239                       fixed_min=0.0, fixed_max=6.0, scope='quant'):
 240   """Inserts a fake quantization op with fixed range after inputs.
 241 
 242   Args:
 243     inputs: A tensor of size [batch_size, height, width, channels].
 244     is_quantized: flag to enable/disable quantization.
 245     fixed_min: fixed min value for fake quant op.
 246     fixed_max: fixed max value for fake quant op.
 247     scope: Optional scope for variable_scope.
 248 
 249   Returns:
 250     Tensor resulting from quantizing the input tensors.
 251   """
 252   if not is_quantized:
 253     return inputs
 254 
 255   with tf.variable_scope(scope):
 256     # Just use fixed quantization range.
 257     return tf.fake_quant_with_min_max_args(inputs, fixed_min, fixed_max)
