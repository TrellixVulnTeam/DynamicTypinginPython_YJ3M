  16 """Test a tflite model using random input data."""
  17 
  18 from __future__ import print_function
  19 from absl import flags
  20 import numpy as np
  21 import tensorflow.compat.v1 as tf
  22 
  23 flags.DEFINE_string('model_path', None, 'Path to model.')
  24 FLAGS = flags.FLAGS
  25 
  26 
  27 def main(_):
  28 
  29   flags.mark_flag_as_required('model_path')
  30 
  31   # Load TFLite model and allocate tensors.
  32   interpreter = tf.lite.Interpreter(model_path=FLAGS.model_path)
  33   interpreter.allocate_tensors()
  34 
  35   # Get input and output tensors.
  36   input_details = interpreter.get_input_details()
  37   print('input_details:', input_details)
  38   output_details = interpreter.get_output_details()
  39   print('output_details:', output_details)
  40 
  41   # Test model on random input data.
  42   input_shape = input_details[0]['shape']
  43   # change the following line to feed into your own data.
  44   input_data = np.array(np.random.random_sample(input_shape), dtype=np.float32)
  45   interpreter.set_tensor(input_details[0]['index'], input_data)
  46 
  47   interpreter.invoke()
  48   output_data = interpreter.get_tensor(output_details[0]['index'])
  49   print(output_data)
  50 
  51 
  52 if __name__ == '__main__':
  53   tf.app.run()
