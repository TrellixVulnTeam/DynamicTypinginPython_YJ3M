  39 def create_input_queue(create_tensor_dict_fn):
  40   """Sets up reader, prefetcher and returns input queue.
  41 
  42   Args:
  43     create_tensor_dict_fn: function to create tensor dictionary.
  44 
  45   Returns:
  46     all_dict: A dictionary holds tensors for images, boxes, and targets.
  47   """
  48   tensor_dict = create_tensor_dict_fn()
  49   all_dict = {}
  50 
  51   num_images = len(tensor_dict[fields.InputDataFields.image])
  52   all_dict['batch'] = tensor_dict['batch']
  53   del tensor_dict['batch']
  54 
  55   for i in range(num_images):
  56     suffix = str(i)
  57     for key, val in tensor_dict.items():
  58       all_dict[key + suffix] = val[i]
  59 
  60     all_dict[fields.InputDataFields.image + suffix] = tf.to_float(
  61         tf.expand_dims(all_dict[fields.InputDataFields.image + suffix], 0))
  62 
  63   return all_dict

  66 def get_inputs(input_queue, num_classes, merge_multiple_label_boxes=False):
  67   """Dequeues batch and constructs inputs to object detection model.
  68 
  69   Args:
  70     input_queue: BatchQueue object holding enqueued tensor_dicts.
  71     num_classes: Number of classes.
  72     merge_multiple_label_boxes: Whether to merge boxes with multiple labels
  73       or not. Defaults to false. Merged boxes are represented with a single
  74       box and a k-hot encoding of the multiple labels associated with the
  75       boxes.
  76 
  77   Returns:
  78     images: a list of 3-D float tensor of images.
  79     image_keys: a list of string keys for the images.
  80     locations: a list of tensors of shape [num_boxes, 4] containing the corners
  81       of the groundtruth boxes.
  82     classes: a list of padded one-hot tensors containing target classes.
  83     masks: a list of 3-D float tensors of shape [num_boxes, image_height,
  84       image_width] containing instance masks for objects if present in the
  85       input_queue. Else returns None.
  86     keypoints: a list of 3-D float tensors of shape [num_boxes, num_keypoints,
  87       2] containing keypoints for objects if present in the
  88       input queue. Else returns None.
  89   """
  90   read_data_list = input_queue
  91   label_id_offset = 1
  92 
  93   def extract_images_and_targets(read_data):
  94     """Extract images and targets from the input dict."""
  95     suffix = 0
  96 
  97     images = []
  98     keys = []
  99     locations = []
 100     classes = []
 101     masks = []
 102     keypoints = []
 103 
 104     while fields.InputDataFields.image + str(suffix) in read_data:
 105       image = read_data[fields.InputDataFields.image + str(suffix)]
 106       key = ''
 107       if fields.InputDataFields.source_id in read_data:
 108         key = read_data[fields.InputDataFields.source_id + str(suffix)]
 109       location_gt = (
 110           read_data[fields.InputDataFields.groundtruth_boxes + str(suffix)])
 111       classes_gt = tf.cast(
 112           read_data[fields.InputDataFields.groundtruth_classes + str(suffix)],
 113           tf.int32)
 114       classes_gt -= label_id_offset
 115       masks_gt = read_data.get(
 116           fields.InputDataFields.groundtruth_instance_masks + str(suffix))
 117       keypoints_gt = read_data.get(
 118           fields.InputDataFields.groundtruth_keypoints + str(suffix))
 119 
 120       if merge_multiple_label_boxes:
 121         location_gt, classes_gt, _ = util_ops.merge_boxes_with_multiple_labels(
 122             location_gt, classes_gt, num_classes)
 123       else:
 124         classes_gt = util_ops.padded_one_hot_encoding(
 125             indices=classes_gt, depth=num_classes, left_pad=0)
 126 
 127       # Batch read input data and groundtruth. Images and locations, classes by
 128       # default should have the same number of items.
 129       images.append(image)
 130       keys.append(key)
 131       locations.append(location_gt)
 132       classes.append(classes_gt)
 133       masks.append(masks_gt)
 134       keypoints.append(keypoints_gt)
 135 
 136       suffix += 1
 137 
 138     return (images, keys, locations, classes, masks, keypoints)
 139 
 140   return extract_images_and_targets(read_data_list)

 143 def _create_losses(input_queue, create_model_fn, train_config):
 144   """Creates loss function for a DetectionModel.
 145 
 146   Args:
 147     input_queue: BatchQueue object holding enqueued tensor_dicts.
 148     create_model_fn: A function to create the DetectionModel.
 149     train_config: a train_pb2.TrainConfig protobuf.
 150   """
 151 
 152   detection_model = create_model_fn()
 153   (images, _, groundtruth_boxes_list, groundtruth_classes_list,
 154    groundtruth_masks_list, groundtruth_keypoints_list) = get_inputs(
 155        input_queue, detection_model.num_classes,
 156        train_config.merge_multiple_label_boxes)
 157 
 158   preprocessed_images = []
 159   true_image_shapes = []
 160   for image in images:
 161     resized_image, true_image_shape = detection_model.preprocess(image)
 162     preprocessed_images.append(resized_image)
 163     true_image_shapes.append(true_image_shape)
 164 
 165   images = tf.concat(preprocessed_images, 0)
 166   true_image_shapes = tf.concat(true_image_shapes, 0)
 167 
 168   if any(mask is None for mask in groundtruth_masks_list):
 169     groundtruth_masks_list = None
 170   if any(keypoints is None for keypoints in groundtruth_keypoints_list):
 171     groundtruth_keypoints_list = None
 172 
 173   detection_model.provide_groundtruth(
 174       groundtruth_boxes_list, groundtruth_classes_list, groundtruth_masks_list,
 175       groundtruth_keypoints_list)
 176   prediction_dict = detection_model.predict(images, true_image_shapes,
 177                                             input_queue['batch'])
 178 
 179   losses_dict = detection_model.loss(prediction_dict, true_image_shapes)
 180   for loss_tensor in losses_dict.values():
 181     tf.losses.add_loss(loss_tensor)

 184 def get_restore_checkpoint_ops(restore_checkpoints, detection_model,
 185                                train_config):
 186   """Restore checkpoint from saved checkpoints.
 187 
 188   Args:
 189     restore_checkpoints: loaded checkpoints.
 190     detection_model: Object detection model built from config file.
 191     train_config: a train_pb2.TrainConfig protobuf.
 192 
 193   Returns:
 194     restorers: A list ops to init the model from checkpoints.
 195 
 196   """
 197   restorers = []
 198   vars_restored = []
 199   for restore_checkpoint in restore_checkpoints:
 200     var_map = detection_model.restore_map(
 201         fine_tune_checkpoint_type=train_config.fine_tune_checkpoint_type)
 202     available_var_map = (
 203         variables_helper.get_variables_available_in_checkpoint(
 204             var_map, restore_checkpoint))
 205     for var_name, var in six.iteritems(available_var_map):
 206       if var in vars_restored:
 207         tf.logging.info('Variable %s contained in multiple checkpoints',
 208                      var.op.name)
 209         del available_var_map[var_name]
 210       else:
 211         vars_restored.append(var)
 212 
 213     # Initialize from ExponentialMovingAverages if possible.
 214     available_ema_var_map = {}
 215     ckpt_reader = tf.train.NewCheckpointReader(restore_checkpoint)
 216     ckpt_vars_to_shape_map = ckpt_reader.get_variable_to_shape_map()
 217     for var_name, var in six.iteritems(available_var_map):
 218       var_name_ema = var_name + '/ExponentialMovingAverage'
 219       if var_name_ema in ckpt_vars_to_shape_map:
 220         available_ema_var_map[var_name_ema] = var
 221       else:
 222         available_ema_var_map[var_name] = var
 223     available_var_map = available_ema_var_map
 224     init_saver = tf.train.Saver(available_var_map)
 225     if list(available_var_map.keys()):
 226       restorers.append(init_saver)
 227     else:
 228       tf.logging.info('WARNING: Checkpoint %s has no restorable variables',
 229                    restore_checkpoint)
 230 
 231     return restorers

 234 def train(create_tensor_dict_fn,
 235           create_model_fn,
 236           train_config,
 237           master,
 238           task,
 239           num_clones,
 240           worker_replicas,
 241           clone_on_cpu,
 242           ps_tasks,
 243           worker_job_name,
 244           is_chief,
 245           train_dir,
 246           graph_hook_fn=None):
 247   """Training function for detection models.
 248 
 249   Args:
 250     create_tensor_dict_fn: a function to create a tensor input dictionary.
 251     create_model_fn: a function that creates a DetectionModel and generates
 252                      losses.
 253     train_config: a train_pb2.TrainConfig protobuf.
 254     master: BNS name of the TensorFlow master to use.
 255     task: The task id of this training instance.
 256     num_clones: The number of clones to run per machine.
 257     worker_replicas: The number of work replicas to train with.
 258     clone_on_cpu: True if clones should be forced to run on CPU.
 259     ps_tasks: Number of parameter server tasks.
 260     worker_job_name: Name of the worker job.
 261     is_chief: Whether this replica is the chief replica.
 262     train_dir: Directory to write checkpoints and training summaries to.
 263     graph_hook_fn: Optional function that is called after the training graph is
 264       completely built. This is helpful to perform additional changes to the
 265       training graph such as optimizing batchnorm. The function should modify
 266       the default graph.
 267   """
 268 
 269   detection_model = create_model_fn()
 270 
 271   with tf.Graph().as_default():
 272     # Build a configuration specifying multi-GPU and multi-replicas.
 273     deploy_config = model_deploy.DeploymentConfig(
 274         num_clones=num_clones,
 275         clone_on_cpu=clone_on_cpu,
 276         replica_id=task,
 277         num_replicas=worker_replicas,
 278         num_ps_tasks=ps_tasks,
 279         worker_job_name=worker_job_name)
 280 
 281     # Place the global step on the device storing the variables.
 282     with tf.device(deploy_config.variables_device()):
 283       global_step = slim.create_global_step()
 284 
 285     with tf.device(deploy_config.inputs_device()):
 286       input_queue = create_input_queue(create_tensor_dict_fn)
 287 
 288     # Gather initial summaries.
 289     # TODO(rathodv): See if summaries can be added/extracted from global tf
 290     # collections so that they don't have to be passed around.
 291     summaries = set(tf.get_collection(tf.GraphKeys.SUMMARIES))
 292     global_summaries = set([])
 293 
 294     model_fn = functools.partial(
 295         _create_losses,
 296         create_model_fn=create_model_fn,
 297         train_config=train_config)
 298     clones = model_deploy.create_clones(deploy_config, model_fn, [input_queue])
 299     first_clone_scope = clones[0].scope
 300 
 301     # Gather update_ops from the first clone. These contain, for example,
 302     # the updates for the batch_norm variables created by model_fn.
 303     update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS, first_clone_scope)
 304 
 305     with tf.device(deploy_config.optimizer_device()):
 306       training_optimizer, optimizer_summary_vars = optimizer_builder.build(
 307           train_config.optimizer)
 308       for var in optimizer_summary_vars:
 309         tf.summary.scalar(var.op.name, var)
 310 
 311     sync_optimizer = None
 312     if train_config.sync_replicas:
 313       training_optimizer = tf.train.SyncReplicasOptimizer(
 314           training_optimizer,
 315           replicas_to_aggregate=train_config.replicas_to_aggregate,
 316           total_num_replicas=train_config.worker_replicas)
 317       sync_optimizer = training_optimizer
 318 
 319     # Create ops required to initialize the model from a given checkpoint.
 320     init_fn = None
 321     if train_config.fine_tune_checkpoint:
 322       restore_checkpoints = [
 323           path.strip() for path in train_config.fine_tune_checkpoint.split(',')
 324       ]
 325 
 326       restorers = get_restore_checkpoint_ops(restore_checkpoints,
 327                                              detection_model, train_config)
 328 
 329       def initializer_fn(sess):
 330         for i, restorer in enumerate(restorers):
 331           restorer.restore(sess, restore_checkpoints[i])
 332 
 333       init_fn = initializer_fn
 334 
 335     with tf.device(deploy_config.optimizer_device()):
 336       regularization_losses = (
 337           None if train_config.add_regularization_loss else [])
 338       total_loss, grads_and_vars = model_deploy.optimize_clones(
 339           clones,
 340           training_optimizer,
 341           regularization_losses=regularization_losses)
 342       total_loss = tf.check_numerics(total_loss, 'LossTensor is inf or nan.')
 343 
 344       # Optionally multiply bias gradients by train_config.bias_grad_multiplier.
 345       if train_config.bias_grad_multiplier:
 346         biases_regex_list = ['.*/biases']
 347         grads_and_vars = variables_helper.multiply_gradients_matching_regex(
 348             grads_and_vars,
 349             biases_regex_list,
 350             multiplier=train_config.bias_grad_multiplier)
 351 
 352       # Optionally clip gradients
 353       if train_config.gradient_clipping_by_norm > 0:
 354         with tf.name_scope('clip_grads'):
 355           grads_and_vars = slim.learning.clip_gradient_norms(
 356               grads_and_vars, train_config.gradient_clipping_by_norm)
 357 
 358       moving_average_variables = slim.get_model_variables()
 359       variable_averages = tf.train.ExponentialMovingAverage(0.9999, global_step)
 360       update_ops.append(variable_averages.apply(moving_average_variables))
 361 
 362       # Create gradient updates.
 363       grad_updates = training_optimizer.apply_gradients(
 364           grads_and_vars, global_step=global_step)
 365       update_ops.append(grad_updates)
 366       update_op = tf.group(*update_ops, name='update_barrier')
 367       with tf.control_dependencies([update_op]):
 368         train_tensor = tf.identity(total_loss, name='train_op')
 369 
 370     if graph_hook_fn:
 371       with tf.device(deploy_config.variables_device()):
 372         graph_hook_fn()
 373 
 374     # Add summaries.
 375     for model_var in slim.get_model_variables():
 376       global_summaries.add(tf.summary.histogram(model_var.op.name, model_var))
 377     for loss_tensor in tf.losses.get_losses():
 378       global_summaries.add(tf.summary.scalar(loss_tensor.op.name, loss_tensor))
 379     global_summaries.add(
 380         tf.summary.scalar('TotalLoss', tf.losses.get_total_loss()))
 381 
 382     # Add the summaries from the first clone. These contain the summaries
 383     # created by model_fn and either optimize_clones() or _gather_clone_loss().
 384     summaries |= set(
 385         tf.get_collection(tf.GraphKeys.SUMMARIES, first_clone_scope))
 386     summaries |= set(tf.get_collection(tf.GraphKeys.SUMMARIES, 'critic_loss'))
 387     summaries |= global_summaries
 388 
 389     # Merge all summaries together.
 390     summary_op = tf.summary.merge(list(summaries), name='summary_op')
 391 
 392     # Soft placement allows placing on CPU ops without GPU implementation.
 393     session_config = tf.ConfigProto(
 394         allow_soft_placement=True, log_device_placement=False)
 395 
 396     # Save checkpoints regularly.
 397     keep_checkpoint_every_n_hours = train_config.keep_checkpoint_every_n_hours
 398     saver = tf.train.Saver(
 399         keep_checkpoint_every_n_hours=keep_checkpoint_every_n_hours)
 400 
 401     slim.learning.train(
 402         train_tensor,
 403         logdir=train_dir,
 404         master=master,
 405         is_chief=is_chief,
 406         session_config=session_config,
 407         startup_delay_steps=train_config.startup_delay_steps,
 408         init_fn=init_fn,
 409         summary_op=summary_op,
 410         number_of_steps=(train_config.num_steps
 411                          if train_config.num_steps else None),
 412         save_summaries_secs=120,
 413         sync_optimizer=sync_optimizer,
 414         saver=saver)
