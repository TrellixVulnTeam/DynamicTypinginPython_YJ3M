  16 r"""Training executable for detection models.
  17 
  18 This executable is used to train DetectionModels. There are two ways of
  19 configuring the training job:
  20 
  21 1) A single pipeline_pb2.TrainEvalPipelineConfig configuration file
  22 can be specified by --pipeline_config_path.
  23 
  24 Example usage:
  25     ./train \
  26         --logtostderr \
  27         --train_dir=path/to/train_dir \
  28         --pipeline_config_path=pipeline_config.pbtxt
  29 
  30 2) Three configuration files can be provided: a model_pb2.DetectionModel
  31 configuration file to define what type of DetectionModel is being trained, an
  32 input_reader_pb2.InputReader file to specify what training data will be used and
  33 a train_pb2.TrainConfig file to configure training parameters.
  34 
  35 Example usage:
  36     ./train \
  37         --logtostderr \
  38         --train_dir=path/to/train_dir \
  39         --model_config_path=model_config.pbtxt \
  40         --train_config_path=train_config.pbtxt \
  41         --input_config_path=train_input_config.pbtxt
  42 
  43 """
  44 
  45 import functools
  46 import json
  47 import os
  48 from absl import flags
  49 import tensorflow.compat.v1 as tf
  50 from lstm_object_detection import model_builder
  51 from lstm_object_detection import trainer
  52 from lstm_object_detection.inputs import seq_dataset_builder
  53 from lstm_object_detection.utils import config_util
  54 from object_detection.builders import preprocessor_builder
  55 
  56 flags.DEFINE_string('master', '', 'Name of the TensorFlow master to use.')
  57 flags.DEFINE_integer('task', 0, 'task id')
  58 flags.DEFINE_integer('num_clones', 1, 'Number of clones to deploy per worker.')
  59 flags.DEFINE_boolean(
  60     'clone_on_cpu', False,
  61     'Force clones to be deployed on CPU.  Note that even if '
  62     'set to False (allowing ops to run on gpu), some ops may '
  63     'still be run on the CPU if they have no GPU kernel.')
  64 flags.DEFINE_integer('worker_replicas', 1, 'Number of worker+trainer '
  65                      'replicas.')
  66 flags.DEFINE_integer(
  67     'ps_tasks', 0, 'Number of parameter server tasks. If None, does not use '
  68     'a parameter server.')
  69 flags.DEFINE_string(
  70     'train_dir', '',
  71     'Directory to save the checkpoints and training summaries.')
  72 
  73 flags.DEFINE_string(
  74     'pipeline_config_path', '',
  75     'Path to a pipeline_pb2.TrainEvalPipelineConfig config '
  76     'file. If provided, other configs are ignored')
  77 
  78 flags.DEFINE_string('train_config_path', '',
  79                     'Path to a train_pb2.TrainConfig config file.')
  80 flags.DEFINE_string('input_config_path', '',
  81                     'Path to an input_reader_pb2.InputReader config file.')
  82 flags.DEFINE_string('model_config_path', '',
  83                     'Path to a model_pb2.DetectionModel config file.')
  84 
  85 FLAGS = flags.FLAGS
  86 
  87 
  88 def main(_):
  89   assert FLAGS.train_dir, '`train_dir` is missing.'
  90   if FLAGS.task == 0:
  91     tf.gfile.MakeDirs(FLAGS.train_dir)
  92   if FLAGS.pipeline_config_path:
  93     configs = config_util.get_configs_from_pipeline_file(
  94         FLAGS.pipeline_config_path)
  95     if FLAGS.task == 0:
  96       tf.gfile.Copy(
  97           FLAGS.pipeline_config_path,
  98           os.path.join(FLAGS.train_dir, 'pipeline.config'),
  99           overwrite=True)
 100   else:
 101     configs = config_util.get_configs_from_multiple_files(
 102         model_config_path=FLAGS.model_config_path,
 103         train_config_path=FLAGS.train_config_path,
 104         train_input_config_path=FLAGS.input_config_path)
 105     if FLAGS.task == 0:
 106       for name, config in [('model.config', FLAGS.model_config_path),
 107                            ('train.config', FLAGS.train_config_path),
 108                            ('input.config', FLAGS.input_config_path)]:
 109         tf.gfile.Copy(
 110             config, os.path.join(FLAGS.train_dir, name), overwrite=True)
 111 
 112   model_config = configs['model']
 113   lstm_config = configs['lstm_model']
 114   train_config = configs['train_config']
 115   input_config = configs['train_input_config']
 116 
 117   model_fn = functools.partial(
 118       model_builder.build,
 119       model_config=model_config,
 120       lstm_config=lstm_config,
 121       is_training=True)
 122 
 123   def get_next(config, model_config, lstm_config, unroll_length):
 124     data_augmentation_options = [
 125         preprocessor_builder.build(step)
 126         for step in train_config.data_augmentation_options
 127     ]
 128     return seq_dataset_builder.build(
 129         config,
 130         model_config,
 131         lstm_config,
 132         unroll_length,
 133         data_augmentation_options,
 134         batch_size=train_config.batch_size)
 135 
 136   create_input_dict_fn = functools.partial(get_next, input_config, model_config,
 137                                            lstm_config,
 138                                            lstm_config.train_unroll_length)
 139 
 140   env = json.loads(os.environ.get('TF_CONFIG', '{}'))
 141   cluster_data = env.get('cluster', None)
 142   cluster = tf.train.ClusterSpec(cluster_data) if cluster_data else None
 143   task_data = env.get('task', None) or {'type': 'master', 'index': 0}
 144   task_info = type('TaskSpec', (object,), task_data)
 145 
 146   # Parameters for a single worker.
 147   ps_tasks = 0
 148   worker_replicas = 1
 149   worker_job_name = 'lonely_worker'
 150   task = 0
 151   is_chief = True
 152   master = ''
 153 
 154   if cluster_data and 'worker' in cluster_data:
 155     # Number of total worker replicas include "worker"s and the "master".
 156     worker_replicas = len(cluster_data['worker']) + 1
 157   if cluster_data and 'ps' in cluster_data:
 158     ps_tasks = len(cluster_data['ps'])
 159 
 160   if worker_replicas > 1 and ps_tasks < 1:
 161     raise ValueError('At least 1 ps task is needed for distributed training.')
 162 
 163   if worker_replicas >= 1 and ps_tasks > 0:
 164     # Set up distributed training.
 165     server = tf.train.Server(
 166         tf.train.ClusterSpec(cluster),
 167         protocol='grpc',
 168         job_name=task_info.type,
 169         task_index=task_info.index)
 170     if task_info.type == 'ps':
 171       server.join()
 172       return
 173 
 174     worker_job_name = '%s/task:%d' % (task_info.type, task_info.index)
 175     task = task_info.index
 176     is_chief = (task_info.type == 'master')
 177     master = server.target
 178 
 179   trainer.train(create_input_dict_fn, model_fn, train_config, master, task,
 180                 FLAGS.num_clones, worker_replicas, FLAGS.clone_on_cpu, ps_tasks,
 181                 worker_job_name, is_chief, FLAGS.train_dir)
 182 
 183 
 184 if __name__ == '__main__':
 185   tf.app.run()
