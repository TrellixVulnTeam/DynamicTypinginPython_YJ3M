  87   def predict(self, preprocessed_inputs, true_image_shapes, states=None,
  88               state_name='lstm_state', feature_scope=None):
  89     with tf.variable_scope(self._extract_features_scope,
  90                            values=[preprocessed_inputs], reuse=tf.AUTO_REUSE):
  91       feature_maps = self._feature_extractor.extract_features(
  92           preprocessed_inputs, states, state_name,
  93           unroll_length=self._unroll_length, scope=feature_scope)
  94     feature_map_spatial_dims = self._get_feature_map_spatial_dims(feature_maps)
  95     image_shape = shape_utils.combined_static_and_dynamic_shape(
  96         preprocessed_inputs)
  97     self._batch_size = preprocessed_inputs.shape[0].value / self._unroll_length
  98     self._states = states
  99     anchors = self._anchor_generator.generate(feature_map_spatial_dims,
 100                                               im_height=image_shape[1],
 101                                               im_width=image_shape[2])
 102     with tf.variable_scope('MultipleGridAnchorGenerator', reuse=tf.AUTO_REUSE):
 103       self._anchors = box_list_ops.concatenate(anchors)
 104     prediction_dict = self._box_predictor.predict(
 105         feature_maps, self._anchor_generator.num_anchors_per_location())
 106     with tf.variable_scope('Loss', reuse=tf.AUTO_REUSE):
 107       box_encodings = tf.concat(prediction_dict['box_encodings'], axis=1)
 108       if box_encodings.shape.ndims == 4 and box_encodings.shape[2] == 1:
 109         box_encodings = tf.squeeze(box_encodings, axis=2)
 110       class_predictions_with_background = tf.concat(
 111           prediction_dict['class_predictions_with_background'], axis=1)
 112     predictions_dict = {
 113         'preprocessed_inputs': preprocessed_inputs,
 114         'box_encodings': box_encodings,
 115         'class_predictions_with_background': class_predictions_with_background,
 116         'feature_maps': feature_maps,
 117         'anchors': self._anchors.get(),
 118         'states_and_outputs': self._feature_extractor.states_and_outputs,
 119     }
 120     # In cases such as exporting the model, the states is always zero. Thus the
 121     # step should be ignored.
 122     if states is not None:
 123       predictions_dict['step'] = self._feature_extractor.step
 124     return predictions_dict

 126   def loss(self, prediction_dict, true_image_shapes, scope=None):
 127     """Computes scalar loss tensors with respect to provided groundtruth.
 128 
 129     Calling this function requires that groundtruth tensors have been
 130     provided via the provide_groundtruth function.
 131 
 132     Args:
 133       prediction_dict: a dictionary holding prediction tensors with
 134         1) box_encodings: 3-D float tensor of shape [batch_size, num_anchors,
 135           box_code_dimension] containing predicted boxes.
 136         2) class_predictions_with_background: 3-D float tensor of shape
 137           [batch_size, num_anchors, num_classes+1] containing class predictions
 138           (logits) for each of the anchors. Note that this tensor *includes*
 139           background class predictions.
 140       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 141         of the form [height, width, channels] indicating the shapes
 142         of true images in the resized images, as resized images can be padded
 143         with zeros.
 144       scope: Optional scope name.
 145 
 146     Returns:
 147       a dictionary mapping loss keys (`localization_loss` and
 148         `classification_loss`) to scalar tensors representing corresponding loss
 149         values.
 150     """
 151     with tf.name_scope(scope, 'Loss', prediction_dict.values()):
 152       keypoints = None
 153       if self.groundtruth_has_field(fields.BoxListFields.keypoints):
 154         keypoints = self.groundtruth_lists(fields.BoxListFields.keypoints)
 155       weights = None
 156       if self.groundtruth_has_field(fields.BoxListFields.weights):
 157         weights = self.groundtruth_lists(fields.BoxListFields.weights)
 158       (batch_cls_targets, batch_cls_weights, batch_reg_targets,
 159        batch_reg_weights, batch_match) = self._assign_targets(
 160            self.groundtruth_lists(fields.BoxListFields.boxes),
 161            self.groundtruth_lists(fields.BoxListFields.classes),
 162            keypoints, weights)
 163       match_list = [matcher.Match(match) for match in tf.unstack(batch_match)]
 164       if self._add_summaries:
 165         self._summarize_target_assignment(
 166             self.groundtruth_lists(fields.BoxListFields.boxes), match_list)
 167       location_losses = self._localization_loss(
 168           prediction_dict['box_encodings'],
 169           batch_reg_targets,
 170           ignore_nan_targets=True,
 171           weights=batch_reg_weights)
 172       cls_losses = ops.reduce_sum_trailing_dimensions(
 173           self._classification_loss(
 174               prediction_dict['class_predictions_with_background'],
 175               batch_cls_targets,
 176               weights=batch_cls_weights),
 177           ndims=2)
 178 
 179       if self._hard_example_miner:
 180         (loc_loss_list, cls_loss_list) = self._apply_hard_mining(
 181             location_losses, cls_losses, prediction_dict, match_list)
 182         localization_loss = tf.reduce_sum(tf.stack(loc_loss_list))
 183         classification_loss = tf.reduce_sum(tf.stack(cls_loss_list))
 184 
 185         if self._add_summaries:
 186           self._hard_example_miner.summarize()
 187       else:
 188         if self._add_summaries:
 189           class_ids = tf.argmax(batch_cls_targets, axis=2)
 190           flattened_class_ids = tf.reshape(class_ids, [-1])
 191           flattened_classification_losses = tf.reshape(cls_losses, [-1])
 192           self._summarize_anchor_classification_loss(
 193               flattened_class_ids, flattened_classification_losses)
 194         localization_loss = tf.reduce_sum(location_losses)
 195         classification_loss = tf.reduce_sum(cls_losses)
 196 
 197       # Optionally normalize by number of positive matches
 198       normalizer = tf.constant(1.0, dtype=tf.float32)
 199       if self._normalize_loss_by_num_matches:
 200         normalizer = tf.maximum(tf.to_float(tf.reduce_sum(batch_reg_weights)),
 201                                 1.0)
 202 
 203       with tf.name_scope('localization_loss'):
 204         localization_loss_normalizer = normalizer
 205         if self._normalize_loc_loss_by_codesize:
 206           localization_loss_normalizer *= self._box_coder.code_size
 207         localization_loss = ((self._localization_loss_weight / (
 208             localization_loss_normalizer)) * localization_loss)
 209       with tf.name_scope('classification_loss'):
 210         classification_loss = ((self._classification_loss_weight / normalizer) *
 211                                classification_loss)
 212 
 213       loss_dict = {
 214           'localization_loss': localization_loss,
 215           'classification_loss': classification_loss
 216       }
 217     return loss_dict

 219   def restore_map(self, fine_tune_checkpoint_type='lstm'):
 220     """Returns a map of variables to load from a foreign checkpoint.
 221 
 222     See parent class for details.
 223 
 224     Args:
 225       fine_tune_checkpoint_type: the type of checkpoint to restore from, either
 226         SSD/LSTM detection checkpoint (with compatible variable names)
 227         classification checkpoint for initialization prior to training.
 228         Available options: `classification`, `detection`, `interleaved`,
 229         and `lstm`.
 230 
 231     Returns:
 232       A dict mapping variable names (to load from a checkpoint) to variables in
 233       the model graph.
 234     Raises:
 235       ValueError: if fine_tune_checkpoint_type is not among
 236       `classification`/`detection`/`interleaved`/`lstm`.
 237     """
 238     if fine_tune_checkpoint_type not in [
 239         'classification', 'detection', 'interleaved', 'lstm',
 240         'interleaved_pretrain'
 241     ]:
 242       raise ValueError('Not supported fine_tune_checkpoint_type: {}'.format(
 243           fine_tune_checkpoint_type))
 244 
 245     self._restored_networks += 1
 246     base_network_scope = self.get_base_network_scope()
 247     if base_network_scope:
 248       scope_to_replace = '{0}_{1}'.format(base_network_scope,
 249                                           self._restored_networks)
 250 
 251     interleaved_model = False
 252     for variable in tf.global_variables():
 253       if scope_to_replace in variable.op.name:
 254         interleaved_model = True
 255         break
 256 
 257     variables_to_restore = {}
 258     for variable in tf.global_variables():
 259       var_name = variable.op.name
 260       if 'global_step' in var_name:
 261         continue
 262 
 263       # Remove FeatureExtractor prefix for classification checkpoints.
 264       if (fine_tune_checkpoint_type == 'classification' or
 265           fine_tune_checkpoint_type == 'interleaved_pretrain'):
 266         var_name = (
 267             re.split('^' + self._extract_features_scope + '/', var_name)[-1])
 268 
 269       # When loading from single frame detection checkpoints, we need to
 270       # remap FeatureMaps variable names.
 271       if ('FeatureMaps' in var_name and
 272           fine_tune_checkpoint_type == 'detection'):
 273         var_name = var_name.replace('FeatureMaps',
 274                                     self.get_base_network_scope())
 275 
 276       # Load interleaved checkpoint specifically.
 277       if interleaved_model:  # Interleaved LSTD.
 278         if 'interleaved' in fine_tune_checkpoint_type:
 279           variables_to_restore[var_name] = variable
 280         else:
 281           # Restore non-base layers from the first checkpoint only.
 282           if self._restored_networks == 1:
 283             if base_network_scope + '_' not in var_name:  # LSTM and FeatureMap
 284               variables_to_restore[var_name] = variable
 285           if scope_to_replace in var_name:
 286             var_name = var_name.replace(scope_to_replace, base_network_scope)
 287             variables_to_restore[var_name] = variable
 288       else:
 289         # Restore from the first model of interleaved checkpoints
 290         if 'interleaved' in fine_tune_checkpoint_type:
 291           var_name = var_name.replace(self.get_base_network_scope(),
 292                                       self.get_base_network_scope() + '_1', 1)
 293 
 294         variables_to_restore[var_name] = variable
 295 
 296     return variables_to_restore
 297

 348   @property
 349   def states_and_outputs(self):
 350     """LSTM states and outputs.
 351 
 352     This variable includes both LSTM states {C_t} and outputs {h_t}.
 353 
 354     Returns:
 355       states_and_outputs: A list of 4-D float tensors, including the lstm state
 356         and output at each timestep.
 357     """
 358     return self._states_out
 359 
 360   @property
 361   def step(self):
 362     return self._step
 363 
 364   def preprocess(self, resized_inputs):
 365     """SSD preprocessing.
 366 
 367     Maps pixel values to the range [-1, 1].
 368 
 369     Args:
 370       resized_inputs: a [batch, height, width, channels] float tensor
 371         representing a batch of images.
 372 
 373     Returns:
 374       preprocessed_inputs: a [batch, height, width, channels] float tensor
 375         representing a batch of images.
 376     """
 377     return (2.0 / 255.0) * resized_inputs - 1.0
 378 
 379   def get_base_network_scope(self):
 380     """Returns the variable scope of the base network.
 381 
 382     Returns:
 383       The variable scope of the base network, e.g. MobilenetV1
 384     """
 385     return self._base_network_scope

 388   def create_lstm_cell(self, batch_size, output_size, state_saver, state_name):
 389     """Create the LSTM cell, and initialize state if necessary.
 390 
 391     Args:
 392       batch_size: input batch size.
 393       output_size: output size of the lstm cell, [width, height].
 394       state_saver: a state saver object with methods `state` and `save_state`.
 395       state_name: string, the name to use with the state_saver.
 396     Returns:
 397       lstm_cell: the lstm cell unit.
 398       init_state: initial state representations.
 399       step: the step
 400     """
 401     pass
