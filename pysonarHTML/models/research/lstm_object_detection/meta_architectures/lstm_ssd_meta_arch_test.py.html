 170   def _create_model(self,
 171                     interleaved=False,
 172                     apply_hard_mining=True,
 173                     normalize_loc_loss_by_codesize=False,
 174                     add_background_class=True,
 175                     random_example_sampling=False,
 176                     use_expected_classification_loss_under_sampling=False,
 177                     min_num_negative_samples=1,
 178                     desired_negative_sampling_ratio=3,
 179                     unroll_length=1):
 180     num_classes = NUM_CLASSES
 181     is_training = False
 182     mock_anchor_generator = MockAnchorGenerator2x2()
 183     mock_box_predictor = test_utils.MockBoxPredictor(is_training, num_classes)
 184     mock_box_coder = test_utils.MockBoxCoder()
 185     if interleaved:
 186       fake_feature_extractor = FakeLSTMInterleavedFeatureExtractor()
 187     else:
 188       fake_feature_extractor = FakeLSTMFeatureExtractor()
 189     mock_matcher = test_utils.MockMatcher()
 190     region_similarity_calculator = sim_calc.IouSimilarity()
 191     encode_background_as_zeros = False
 192     def image_resizer_fn(image):
 193       return [tf.identity(image), tf.shape(image)]
 194 
 195     classification_loss = losses.WeightedSigmoidClassificationLoss()
 196     localization_loss = losses.WeightedSmoothL1LocalizationLoss()
 197     non_max_suppression_fn = functools.partial(
 198         post_processing.batch_multiclass_non_max_suppression,
 199         score_thresh=-20.0,
 200         iou_thresh=1.0,
 201         max_size_per_class=5,
 202         max_total_size=MAX_TOTAL_NUM_BOXES)
 203     classification_loss_weight = 1.0
 204     localization_loss_weight = 1.0
 205     negative_class_weight = 1.0
 206     normalize_loss_by_num_matches = False
 207 
 208     hard_example_miner = None
 209     if apply_hard_mining:
 210       # This hard example miner is expected to be a no-op.
 211       hard_example_miner = losses.HardExampleMiner(
 212           num_hard_examples=None,
 213           iou_threshold=1.0)
 214 
 215     target_assigner_instance = target_assigner.TargetAssigner(
 216         region_similarity_calculator,
 217         mock_matcher,
 218         mock_box_coder,
 219         negative_class_weight=negative_class_weight)
 220 
 221     code_size = 4
 222     model = lstm_ssd_meta_arch.LSTMSSDMetaArch(
 223         is_training=is_training,
 224         anchor_generator=mock_anchor_generator,
 225         box_predictor=mock_box_predictor,
 226         box_coder=mock_box_coder,
 227         feature_extractor=fake_feature_extractor,
 228         encode_background_as_zeros=encode_background_as_zeros,
 229         image_resizer_fn=image_resizer_fn,
 230         non_max_suppression_fn=non_max_suppression_fn,
 231         score_conversion_fn=tf.identity,
 232         classification_loss=classification_loss,
 233         localization_loss=localization_loss,
 234         classification_loss_weight=classification_loss_weight,
 235         localization_loss_weight=localization_loss_weight,
 236         normalize_loss_by_num_matches=normalize_loss_by_num_matches,
 237         hard_example_miner=hard_example_miner,
 238         unroll_length=unroll_length,
 239         target_assigner_instance=target_assigner_instance,
 240         add_summaries=False)
 241     return model, num_classes, mock_anchor_generator.num_anchors(), code_size

 243   def _get_value_for_matching_key(self, dictionary, suffix):
 244     for key in dictionary.keys():
 245       if key.endswith(suffix):
 246         return dictionary[key]
 247     raise ValueError('key not found {}'.format(suffix))

 249   def test_predict_returns_correct_items_and_sizes(self):
 250     batch_size = 3
 251     height = width = 2
 252     num_unroll = 1
 253 
 254     graph = tf.Graph()
 255     with graph.as_default():
 256       model, num_classes, num_anchors, code_size = self._create_model()
 257       preprocessed_images = tf.random_uniform(
 258           [batch_size * num_unroll, height, width, 3],
 259           minval=-1.,
 260           maxval=1.)
 261       true_image_shapes = tf.tile(
 262           [[height, width, 3]], [batch_size, 1])
 263       prediction_dict = model.predict(preprocessed_images, true_image_shapes)
 264 
 265 
 266       self.assertIn('preprocessed_inputs', prediction_dict)
 267       self.assertIn('box_encodings', prediction_dict)
 268       self.assertIn('class_predictions_with_background', prediction_dict)
 269       self.assertIn('feature_maps', prediction_dict)
 270       self.assertIn('anchors', prediction_dict)
 271       self.assertAllEqual(
 272           [batch_size * num_unroll, height, width, 3],
 273           prediction_dict['preprocessed_inputs'].shape.as_list())
 274       self.assertAllEqual(
 275           [batch_size * num_unroll, num_anchors, code_size],
 276           prediction_dict['box_encodings'].shape.as_list())
 277       self.assertAllEqual(
 278           [batch_size * num_unroll, num_anchors, num_classes + 1],
 279           prediction_dict['class_predictions_with_background'].shape.as_list())
 280       self.assertAllEqual(
 281           [num_anchors, code_size],
 282           prediction_dict['anchors'].shape.as_list())
