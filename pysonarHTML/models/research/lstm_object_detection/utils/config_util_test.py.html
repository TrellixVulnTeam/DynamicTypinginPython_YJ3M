  31 def _write_config(config, config_path):
  32   """Writes a config object to disk."""
  33   config_text = text_format.MessageToString(config)
  34   with tf.gfile.Open(config_path, "wb") as f:
  35     f.write(config_text)

  38 class ConfigUtilTest(tf.test.TestCase):
  39 
  40   def test_get_configs_from_pipeline_file(self):
  41     """Test that proto configs can be read from pipeline config file."""
  42     pipeline_config_path = os.path.join(self.get_temp_dir(), "pipeline.config")
  43     pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
  44     pipeline_config.model.ssd.num_classes = 10
  45     pipeline_config.train_config.batch_size = 32
  46     pipeline_config.train_input_reader.label_map_path = "path/to/label_map"
  47     pipeline_config.eval_config.num_examples = 20
  48     pipeline_config.eval_input_reader.add().queue_capacity = 100
  49 
  50     pipeline_config.Extensions[
  51         internal_pipeline_pb2.lstm_model].train_unroll_length = 5
  52     pipeline_config.Extensions[
  53         internal_pipeline_pb2.lstm_model].eval_unroll_length = 10
  54 
  55     _write_config(pipeline_config, pipeline_config_path)
  56 
  57     configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
  58     self.assertProtoEquals(pipeline_config.model, configs["model"])
  59     self.assertProtoEquals(pipeline_config.train_config,
  60                            configs["train_config"])
  61     self.assertProtoEquals(pipeline_config.train_input_reader,
  62                            configs["train_input_config"])
  63     self.assertProtoEquals(pipeline_config.eval_config, configs["eval_config"])
  64     self.assertProtoEquals(pipeline_config.eval_input_reader,
  65                            configs["eval_input_configs"])
  66     self.assertProtoEquals(
  67         pipeline_config.Extensions[internal_pipeline_pb2.lstm_model],
  68         configs["lstm_model"])
  69 
  70   def test_create_pipeline_proto_from_configs(self):
  71     """Tests that proto can be reconstructed from configs dictionary."""
  72     pipeline_config_path = os.path.join(self.get_temp_dir(), "pipeline.config")
  73 
  74     pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
  75     pipeline_config.model.ssd.num_classes = 10
  76     pipeline_config.train_config.batch_size = 32
  77     pipeline_config.train_input_reader.label_map_path = "path/to/label_map"
  78     pipeline_config.eval_config.num_examples = 20
  79     pipeline_config.eval_input_reader.add().queue_capacity = 100
  80 
  81     pipeline_config.Extensions[
  82         internal_pipeline_pb2.lstm_model].train_unroll_length = 5
  83     pipeline_config.Extensions[
  84         internal_pipeline_pb2.lstm_model].eval_unroll_length = 10
  85     _write_config(pipeline_config, pipeline_config_path)
  86 
  87     configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
  88     pipeline_config_reconstructed = (
  89         config_util.create_pipeline_proto_from_configs(configs))
  90     self.assertEqual(pipeline_config, pipeline_config_reconstructed)
