  58 def _create_detection_op(model, input_dict, batch):
  59   """Create detection ops.
  60 
  61   Args:
  62     model: model to perform predictions with.
  63     input_dict: A dict holds input data.
  64     batch: batch size for evaluation.
  65 
  66   Returns:
  67     Detection tensor ops.
  68   """
  69   video_tensor = tf.stack(list(input_dict[fields.InputDataFields.image]))
  70   preprocessed_video, true_image_shapes = model.preprocess(
  71       tf.to_float(video_tensor))
  72   if batch is not None:
  73     prediction_dict = model.predict(preprocessed_video, true_image_shapes,
  74                                     batch)
  75   else:
  76     prediction_dict = model.predict(preprocessed_video, true_image_shapes)
  77 
  78   return model.postprocess(prediction_dict, true_image_shapes)

  81 def _extract_prediction_tensors(model,
  82                                 create_input_dict_fn,
  83                                 ignore_groundtruth=False):
  84   """Restores the model in a tensorflow session.
  85 
  86   Args:
  87     model: model to perform predictions with.
  88     create_input_dict_fn: function to create input tensor dictionaries.
  89     ignore_groundtruth: whether groundtruth should be ignored.
  90 
  91 
  92   Returns:
  93     tensor_dict: A tensor dictionary with evaluations.
  94   """
  95   input_dict = create_input_dict_fn()
  96   batch = None
  97   if 'batch' in input_dict:
  98     batch = input_dict.pop('batch')
  99   else:
 100     prefetch_queue = prefetcher.prefetch(input_dict, capacity=500)
 101     input_dict = prefetch_queue.dequeue()
 102     # consistent format for images and videos
 103     for key, value in input_dict.iteritems():
 104       input_dict[key] = (value,)
 105 
 106   detections = _create_detection_op(model, input_dict, batch)
 107 
 108   # Print out anaylsis of the model.
 109   contrib_tfprof.model_analyzer.print_model_analysis(
 110       tf.get_default_graph(),
 111       tfprof_options=contrib_tfprof.model_analyzer
 112       .TRAINABLE_VARS_PARAMS_STAT_OPTIONS)
 113   contrib_tfprof.model_analyzer.print_model_analysis(
 114       tf.get_default_graph(),
 115       tfprof_options=contrib_tfprof.model_analyzer.FLOAT_OPS_OPTIONS)
 116 
 117   num_frames = len(input_dict[fields.InputDataFields.image])
 118   ret = []
 119   for i in range(num_frames):
 120     original_image = tf.expand_dims(input_dict[fields.InputDataFields.image][i],
 121                                     0)
 122     groundtruth = None
 123     if not ignore_groundtruth:
 124       groundtruth = {
 125           fields.InputDataFields.groundtruth_boxes:
 126               input_dict[fields.InputDataFields.groundtruth_boxes][i],
 127           fields.InputDataFields.groundtruth_classes:
 128               input_dict[fields.InputDataFields.groundtruth_classes][i],
 129       }
 130       optional_keys = (
 131           fields.InputDataFields.groundtruth_area,
 132           fields.InputDataFields.groundtruth_is_crowd,
 133           fields.InputDataFields.groundtruth_difficult,
 134           fields.InputDataFields.groundtruth_group_of,
 135       )
 136       for opt_key in optional_keys:
 137         if opt_key in input_dict:
 138           groundtruth[opt_key] = input_dict[opt_key][i]
 139       if fields.DetectionResultFields.detection_masks in detections:
 140         groundtruth[fields.InputDataFields.groundtruth_instance_masks] = (
 141             input_dict[fields.InputDataFields.groundtruth_instance_masks][i])
 142 
 143     detections_frame = {
 144         key: tf.expand_dims(value[i], 0)
 145         for key, value in detections.iteritems()
 146     }
 147 
 148     source_id = (
 149         batch.key[0] if batch is not None else
 150         input_dict[fields.InputDataFields.source_id][i])
 151     ret.append(
 152         eval_util.result_dict_for_single_example(
 153             original_image,
 154             source_id,
 155             detections_frame,
 156             groundtruth,
 157             class_agnostic=(fields.DetectionResultFields.detection_classes
 158                             not in detections),
 159             scale_to_absolute=True))
 160   return ret

 163 def get_evaluators(eval_config, categories):
 164   """Returns the evaluator class according to eval_config, valid for categories.
 165 
 166   Args:
 167     eval_config: evaluation configurations.
 168     categories: a list of categories to evaluate.
 169   Returns:
 170     An list of instances of DetectionEvaluator.
 171 
 172   Raises:
 173     ValueError: if metric is not in the metric class dictionary.
 174   """
 175   eval_metric_fn_keys = eval_config.metrics_set
 176   if not eval_metric_fn_keys:
 177     eval_metric_fn_keys = [EVAL_DEFAULT_METRIC]
 178   evaluators_list = []
 179   for eval_metric_fn_key in eval_metric_fn_keys:
 180     if eval_metric_fn_key not in EVAL_METRICS_CLASS_DICT:
 181       raise ValueError('Metric not found: {}'.format(eval_metric_fn_key))
 182     else:
 183       evaluators_list.append(
 184           EVAL_METRICS_CLASS_DICT[eval_metric_fn_key](categories=categories))
 185   return evaluators_list

 188 def evaluate(create_input_dict_fn,
 189              create_model_fn,
 190              eval_config,
 191              categories,
 192              checkpoint_dir,
 193              eval_dir,
 194              graph_hook_fn=None):
 195   """Evaluation function for detection models.
 196 
 197   Args:
 198     create_input_dict_fn: a function to create a tensor input dictionary.
 199     create_model_fn: a function that creates a DetectionModel.
 200     eval_config: a eval_pb2.EvalConfig protobuf.
 201     categories: a list of category dictionaries. Each dict in the list should
 202                 have an integer 'id' field and string 'name' field.
 203     checkpoint_dir: directory to load the checkpoints to evaluate from.
 204     eval_dir: directory to write evaluation metrics summary to.
 205     graph_hook_fn: Optional function that is called after the training graph is
 206       completely built. This is helpful to perform additional changes to the
 207       training graph such as optimizing batchnorm. The function should modify
 208       the default graph.
 209 
 210   Returns:
 211     metrics: A dictionary containing metric names and values from the latest
 212       run.
 213   """
 214 
 215   model = create_model_fn()
 216 
 217   if eval_config.ignore_groundtruth and not eval_config.export_path:
 218     tf.logging.fatal('If ignore_groundtruth=True then an export_path is '
 219                   'required. Aborting!!!')
 220 
 221   tensor_dicts = _extract_prediction_tensors(
 222       model=model,
 223       create_input_dict_fn=create_input_dict_fn,
 224       ignore_groundtruth=eval_config.ignore_groundtruth)
 225 
 226   def _process_batch(tensor_dicts,
 227                      sess,
 228                      batch_index,
 229                      counters,
 230                      losses_dict=None):
 231     """Evaluates tensors in tensor_dicts, visualizing the first K examples.
 232 
 233     This function calls sess.run on tensor_dicts, evaluating the original_image
 234     tensor only on the first K examples and visualizing detections overlaid
 235     on this original_image.
 236 
 237     Args:
 238       tensor_dicts: a dictionary of tensors
 239       sess: tensorflow session
 240       batch_index: the index of the batch amongst all batches in the run.
 241       counters: a dictionary holding 'success' and 'skipped' fields which can
 242         be updated to keep track of number of successful and failed runs,
 243         respectively.  If these fields are not updated, then the success/skipped
 244         counter values shown at the end of evaluation will be incorrect.
 245       losses_dict: Optional dictonary of scalar loss tensors. Necessary only
 246         for matching function signiture in third_party eval_util.py.
 247 
 248     Returns:
 249       result_dict: a dictionary of numpy arrays
 250       result_losses_dict: a dictionary of scalar losses. This is empty if input
 251         losses_dict is None. Necessary only for matching function signiture in
 252         third_party eval_util.py.
 253     """
 254     if batch_index % 10 == 0:
 255       tf.logging.info('Running eval ops batch %d', batch_index)
 256     if not losses_dict:
 257       losses_dict = {}
 258     try:
 259       result_dicts, result_losses_dict = sess.run([tensor_dicts, losses_dict])
 260       counters['success'] += 1
 261     except tf.errors.InvalidArgumentError:
 262       tf.logging.info('Skipping image')
 263       counters['skipped'] += 1
 264       return {}
 265     num_images = len(tensor_dicts)
 266     for i in range(num_images):
 267       result_dict = result_dicts[i]
 268       global_step = tf.train.global_step(sess, tf.train.get_global_step())
 269       tag = 'image-%d' % (batch_index * num_images + i)
 270       if batch_index < eval_config.num_visualizations / num_images:
 271         eval_util.visualize_detection_results(
 272             result_dict,
 273             tag,
 274             global_step,
 275             categories=categories,
 276             summary_dir=eval_dir,
 277             export_dir=eval_config.visualization_export_dir,
 278             show_groundtruth=eval_config.visualize_groundtruth_boxes,
 279             groundtruth_box_visualization_color=eval_config.
 280             groundtruth_box_visualization_color,
 281             min_score_thresh=eval_config.min_score_threshold,
 282             max_num_predictions=eval_config.max_num_boxes_to_visualize,
 283             skip_scores=eval_config.skip_scores,
 284             skip_labels=eval_config.skip_labels,
 285             keep_image_id_for_visualization_export=eval_config.
 286             keep_image_id_for_visualization_export)
 287     if num_images > 1:
 288       return result_dicts, result_losses_dict
 289     else:
 290       return result_dicts[0], result_losses_dict
 291 
 292   variables_to_restore = tf.global_variables()
 293   global_step = tf.train.get_or_create_global_step()
 294   variables_to_restore.append(global_step)
 295 
 296   if graph_hook_fn:
 297     graph_hook_fn()
 298 
 299   if eval_config.use_moving_averages:
 300     variable_averages = tf.train.ExponentialMovingAverage(0.0)
 301     variables_to_restore = variable_averages.variables_to_restore()
 302     for key in variables_to_restore.keys():
 303       if 'moving_mean' in key:
 304         variables_to_restore[key.replace(
 305             'moving_mean', 'moving_mean/ExponentialMovingAverage')] = (
 306                 variables_to_restore[key])
 307         del variables_to_restore[key]
 308       if 'moving_variance' in key:
 309         variables_to_restore[key.replace(
 310             'moving_variance', 'moving_variance/ExponentialMovingAverage')] = (
 311                 variables_to_restore[key])
 312         del variables_to_restore[key]
 313 
 314   saver = tf.train.Saver(variables_to_restore)
 315 
 316   def _restore_latest_checkpoint(sess):
 317     latest_checkpoint = tf.train.latest_checkpoint(checkpoint_dir)
 318     saver.restore(sess, latest_checkpoint)
 319 
 320   metrics = eval_util.repeated_checkpoint_run(
 321       tensor_dict=tensor_dicts,
 322       summary_dir=eval_dir,
 323       evaluators=get_evaluators(eval_config, categories),
 324       batch_processor=_process_batch,
 325       checkpoint_dirs=[checkpoint_dir],
 326       variables_to_restore=None,
 327       restore_fn=_restore_latest_checkpoint,
 328       num_batches=eval_config.num_examples,
 329       eval_interval_secs=eval_config.eval_interval_secs,
 330       max_number_of_evaluations=(1 if eval_config.ignore_groundtruth else
 331                                  eval_config.max_evals
 332                                  if eval_config.max_evals else None),
 333       master=eval_config.eval_master,
 334       save_graph=eval_config.save_graph,
 335       save_graph_dir=(eval_dir if eval_config.save_graph else ''))
 336 
 337   return metrics
