  39 def get_const_center_size_encoded_anchors(anchors):
  40   """Exports center-size encoded anchors as a constant tensor.
  41 
  42   Args:
  43     anchors: a float32 tensor of shape [num_anchors, 4] containing the anchor
  44       boxes
  45 
  46   Returns:
  47     encoded_anchors: a float32 constant tensor of shape [num_anchors, 4]
  48     containing the anchor boxes.
  49   """
  50   anchor_boxlist = box_list.BoxList(anchors)
  51   y, x, h, w = anchor_boxlist.get_center_coordinates_and_sizes()
  52   num_anchors = y.get_shape().as_list()
  53 
  54   with tf.Session() as sess:
  55     y_out, x_out, h_out, w_out = sess.run([y, x, h, w])
  56   encoded_anchors = tf.constant(
  57       np.transpose(np.stack((y_out, x_out, h_out, w_out))),
  58       dtype=tf.float32,
  59       shape=[num_anchors[0], _DEFAULT_NUM_COORD_BOX],
  60       name='anchors')
  61   return encoded_anchors

  64 def append_postprocessing_op(frozen_graph_def,
  65                              max_detections,
  66                              max_classes_per_detection,
  67                              nms_score_threshold,
  68                              nms_iou_threshold,
  69                              num_classes,
  70                              scale_values,
  71                              detections_per_class=100,
  72                              use_regular_nms=False):
  73   """Appends postprocessing custom op.
  74 
  75   Args:
  76     frozen_graph_def: Frozen GraphDef for SSD model after freezing the
  77       checkpoint
  78     max_detections: Maximum number of detections (boxes) to show
  79     max_classes_per_detection: Number of classes to display per detection
  80     nms_score_threshold: Score threshold used in Non-maximal suppression in
  81       post-processing
  82     nms_iou_threshold: Intersection-over-union threshold used in Non-maximal
  83       suppression in post-processing
  84     num_classes: number of classes in SSD detector
  85     scale_values: scale values is a dict with following key-value pairs
  86       {y_scale: 10, x_scale: 10, h_scale: 5, w_scale: 5} that are used in decode
  87         centersize boxes
  88     detections_per_class: In regular NonMaxSuppression, number of anchors used
  89       for NonMaxSuppression per class
  90     use_regular_nms: Flag to set postprocessing op to use Regular NMS instead of
  91       Fast NMS.
  92 
  93   Returns:
  94     transformed_graph_def: Frozen GraphDef with postprocessing custom op
  95     appended
  96     TFLite_Detection_PostProcess custom op node has four outputs:
  97     detection_boxes: a float32 tensor of shape [1, num_boxes, 4] with box
  98     locations
  99     detection_classes: a float32 tensor of shape [1, num_boxes]
 100     with class indices
 101     detection_scores: a float32 tensor of shape [1, num_boxes]
 102     with class scores
 103     num_boxes: a float32 tensor of size 1 containing the number of detected
 104     boxes
 105   """
 106   new_output = frozen_graph_def.node.add()
 107   new_output.op = 'TFLite_Detection_PostProcess'
 108   new_output.name = 'TFLite_Detection_PostProcess'
 109   new_output.attr['_output_quantized'].CopyFrom(
 110       attr_value_pb2.AttrValue(b=True))
 111   new_output.attr['_output_types'].list.type.extend([
 112       types_pb2.DT_FLOAT, types_pb2.DT_FLOAT, types_pb2.DT_FLOAT,
 113       types_pb2.DT_FLOAT
 114   ])
 115   new_output.attr['_support_output_type_float_in_quantized_op'].CopyFrom(
 116       attr_value_pb2.AttrValue(b=True))
 117   new_output.attr['max_detections'].CopyFrom(
 118       attr_value_pb2.AttrValue(i=max_detections))
 119   new_output.attr['max_classes_per_detection'].CopyFrom(
 120       attr_value_pb2.AttrValue(i=max_classes_per_detection))
 121   new_output.attr['nms_score_threshold'].CopyFrom(
 122       attr_value_pb2.AttrValue(f=nms_score_threshold.pop()))
 123   new_output.attr['nms_iou_threshold'].CopyFrom(
 124       attr_value_pb2.AttrValue(f=nms_iou_threshold.pop()))
 125   new_output.attr['num_classes'].CopyFrom(
 126       attr_value_pb2.AttrValue(i=num_classes))
 127 
 128   new_output.attr['y_scale'].CopyFrom(
 129       attr_value_pb2.AttrValue(f=scale_values['y_scale'].pop()))
 130   new_output.attr['x_scale'].CopyFrom(
 131       attr_value_pb2.AttrValue(f=scale_values['x_scale'].pop()))
 132   new_output.attr['h_scale'].CopyFrom(
 133       attr_value_pb2.AttrValue(f=scale_values['h_scale'].pop()))
 134   new_output.attr['w_scale'].CopyFrom(
 135       attr_value_pb2.AttrValue(f=scale_values['w_scale'].pop()))
 136   new_output.attr['detections_per_class'].CopyFrom(
 137       attr_value_pb2.AttrValue(i=detections_per_class))
 138   new_output.attr['use_regular_nms'].CopyFrom(
 139       attr_value_pb2.AttrValue(b=use_regular_nms))
 140 
 141   new_output.input.extend(
 142       ['raw_outputs/box_encodings', 'raw_outputs/class_predictions', 'anchors'])
 143   # Transform the graph to append new postprocessing op
 144   input_names = []
 145   output_names = ['TFLite_Detection_PostProcess']
 146   transforms = ['strip_unused_nodes']
 147   transformed_graph_def = TransformGraph(frozen_graph_def, input_names,
 148                                          output_names, transforms)
 149   return transformed_graph_def

 152 def export_tflite_graph(pipeline_config,
 153                         trained_checkpoint_prefix,
 154                         output_dir,
 155                         add_postprocessing_op,
 156                         max_detections,
 157                         max_classes_per_detection,
 158                         detections_per_class=100,
 159                         use_regular_nms=False,
 160                         binary_graph_name='tflite_graph.pb',
 161                         txt_graph_name='tflite_graph.pbtxt'):
 162   """Exports a tflite compatible graph and anchors for ssd detection model.
 163 
 164   Anchors are written to a tensor and tflite compatible graph
 165   is written to output_dir/tflite_graph.pb.
 166 
 167   Args:
 168     pipeline_config: Dictionary of configuration objects. Keys are `model`,
 169       `train_config`, `train_input_config`, `eval_config`, `eval_input_config`,
 170       `lstm_model`. Value are the corresponding config objects.
 171     trained_checkpoint_prefix: a file prefix for the checkpoint containing the
 172       trained parameters of the SSD model.
 173     output_dir: A directory to write the tflite graph and anchor file to.
 174     add_postprocessing_op: If add_postprocessing_op is true: frozen graph adds a
 175       TFLite_Detection_PostProcess custom op
 176     max_detections: Maximum number of detections (boxes) to show
 177     max_classes_per_detection: Number of classes to display per detection
 178     detections_per_class: In regular NonMaxSuppression, number of anchors used
 179       for NonMaxSuppression per class
 180     use_regular_nms: Flag to set postprocessing op to use Regular NMS instead of
 181       Fast NMS.
 182     binary_graph_name: Name of the exported graph file in binary format.
 183     txt_graph_name: Name of the exported graph file in text format.
 184 
 185   Raises:
 186     ValueError: if the pipeline config contains models other than ssd or uses an
 187       fixed_shape_resizer and provides a shape as well.
 188   """
 189   model_config = pipeline_config['model']
 190   lstm_config = pipeline_config['lstm_model']
 191   eval_config = pipeline_config['eval_config']
 192   tf.gfile.MakeDirs(output_dir)
 193   if model_config.WhichOneof('model') != 'ssd':
 194     raise ValueError('Only ssd models are supported in tflite. '
 195                      'Found {} in config'.format(
 196                          model_config.WhichOneof('model')))
 197 
 198   num_classes = model_config.ssd.num_classes
 199   nms_score_threshold = {
 200       model_config.ssd.post_processing.batch_non_max_suppression.score_threshold
 201   }
 202   nms_iou_threshold = {
 203       model_config.ssd.post_processing.batch_non_max_suppression.iou_threshold
 204   }
 205   scale_values = {}
 206   scale_values['y_scale'] = {
 207       model_config.ssd.box_coder.faster_rcnn_box_coder.y_scale
 208   }
 209   scale_values['x_scale'] = {
 210       model_config.ssd.box_coder.faster_rcnn_box_coder.x_scale
 211   }
 212   scale_values['h_scale'] = {
 213       model_config.ssd.box_coder.faster_rcnn_box_coder.height_scale
 214   }
 215   scale_values['w_scale'] = {
 216       model_config.ssd.box_coder.faster_rcnn_box_coder.width_scale
 217   }
 218 
 219   image_resizer_config = model_config.ssd.image_resizer
 220   image_resizer = image_resizer_config.WhichOneof('image_resizer_oneof')
 221   num_channels = _DEFAULT_NUM_CHANNELS
 222   if image_resizer == 'fixed_shape_resizer':
 223     height = image_resizer_config.fixed_shape_resizer.height
 224     width = image_resizer_config.fixed_shape_resizer.width
 225     if image_resizer_config.fixed_shape_resizer.convert_to_grayscale:
 226       num_channels = 1
 227 
 228     shape = [lstm_config.eval_unroll_length, height, width, num_channels]
 229   else:
 230     raise ValueError(
 231         'Only fixed_shape_resizer'
 232         'is supported with tflite. Found {}'.format(
 233             image_resizer_config.WhichOneof('image_resizer_oneof')))
 234 
 235   video_tensor = tf.placeholder(
 236       tf.float32, shape=shape, name='input_video_tensor')
 237 
 238   detection_model = model_builder.build(
 239       model_config, lstm_config, is_training=False)
 240   preprocessed_video, true_image_shapes = detection_model.preprocess(
 241       tf.to_float(video_tensor))
 242   predicted_tensors = detection_model.predict(preprocessed_video,
 243                                               true_image_shapes)
 244   # predicted_tensors = detection_model.postprocess(predicted_tensors,
 245   #   true_image_shapes)
 246   # The score conversion occurs before the post-processing custom op
 247   _, score_conversion_fn = post_processing_builder.build(
 248       model_config.ssd.post_processing)
 249   class_predictions = score_conversion_fn(
 250       predicted_tensors['class_predictions_with_background'])
 251 
 252   with tf.name_scope('raw_outputs'):
 253     # 'raw_outputs/box_encodings': a float32 tensor of shape [1, num_anchors, 4]
 254     #  containing the encoded box predictions. Note that these are raw
 255     #  predictions and no Non-Max suppression is applied on them and
 256     #  no decode center size boxes is applied to them.
 257     tf.identity(predicted_tensors['box_encodings'], name='box_encodings')
 258     # 'raw_outputs/class_predictions': a float32 tensor of shape
 259     #  [1, num_anchors, num_classes] containing the class scores for each anchor
 260     #  after applying score conversion.
 261     tf.identity(class_predictions, name='class_predictions')
 262   # 'anchors': a float32 tensor of shape
 263   #   [4, num_anchors] containing the anchors as a constant node.
 264   tf.identity(
 265       get_const_center_size_encoded_anchors(predicted_tensors['anchors']),
 266       name='anchors')
 267 
 268   # Add global step to the graph, so we know the training step number when we
 269   # evaluate the model.
 270   tf.train.get_or_create_global_step()
 271 
 272   # graph rewriter
 273   is_quantized = ('graph_rewriter' in pipeline_config)
 274   if is_quantized:
 275     graph_rewriter_config = pipeline_config['graph_rewriter']
 276     graph_rewriter_fn = graph_rewriter_builder.build(
 277         graph_rewriter_config, is_training=False, is_export=True)
 278     graph_rewriter_fn()
 279 
 280   if model_config.ssd.feature_extractor.HasField('fpn'):
 281     exporter.rewrite_nn_resize_op(is_quantized)
 282 
 283   # freeze the graph
 284   saver_kwargs = {}
 285   if eval_config.use_moving_averages:
 286     saver_kwargs['write_version'] = saver_pb2.SaverDef.V1
 287     moving_average_checkpoint = tempfile.NamedTemporaryFile()
 288     exporter.replace_variable_values_with_moving_averages(
 289         tf.get_default_graph(), trained_checkpoint_prefix,
 290         moving_average_checkpoint.name)
 291     checkpoint_to_use = moving_average_checkpoint.name
 292   else:
 293     checkpoint_to_use = trained_checkpoint_prefix
 294 
 295   saver = tf.train.Saver(**saver_kwargs)
 296   input_saver_def = saver.as_saver_def()
 297   frozen_graph_def = exporter.freeze_graph_with_def_protos(
 298       input_graph_def=tf.get_default_graph().as_graph_def(),
 299       input_saver_def=input_saver_def,
 300       input_checkpoint=checkpoint_to_use,
 301       output_node_names=','.join([
 302           'raw_outputs/box_encodings', 'raw_outputs/class_predictions',
 303           'anchors'
 304       ]),
 305       restore_op_name='save/restore_all',
 306       filename_tensor_name='save/Const:0',
 307       clear_devices=True,
 308       output_graph='',
 309       initializer_nodes='')
 310 
 311   # Add new operation to do post processing in a custom op (TF Lite only)
 312 
 313   if add_postprocessing_op:
 314     transformed_graph_def = append_postprocessing_op(
 315         frozen_graph_def, max_detections, max_classes_per_detection,
 316         nms_score_threshold, nms_iou_threshold, num_classes, scale_values,
 317         detections_per_class, use_regular_nms)
 318   else:
 319     # Return frozen without adding post-processing custom op
 320     transformed_graph_def = frozen_graph_def
 321 
 322   binary_graph = os.path.join(output_dir, binary_graph_name)
 323   with tf.gfile.GFile(binary_graph, 'wb') as f:
 324     f.write(transformed_graph_def.SerializeToString())
 325   txt_graph = os.path.join(output_dir, txt_graph_name)
 326   with tf.gfile.GFile(txt_graph, 'w') as f:
 327     f.write(str(transformed_graph_def))
