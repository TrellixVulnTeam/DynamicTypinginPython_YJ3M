  16 r"""tf.data.Dataset builder.
  17 
  18 Creates data sources for DetectionModels from an InputReader config. See
  19 input_reader.proto for options.
  20 
  21 Note: If users wishes to also use their own InputReaders with the Object
  22 Detection configuration framework, they should define their own builder function
  23 that wraps the build function.

  40 # frame has a total number of objects less than _PADDING_SIZE.
  41 _PADDING_SIZE = 30

  44 def _build_training_batch_dict(batch_sequences_with_states, unroll_length,
  45                                batch_size):
  46   """Builds training batch samples.
  47 
  48   Args:
  49     batch_sequences_with_states: A batch_sequences_with_states object.
  50     unroll_length: Unrolled length for LSTM training.
  51     batch_size: Batch size for queue outputs.
  52 
  53   Returns:
  54     A dictionary of tensors based on items in input_reader_config.
  55   """
  56   seq_tensors_dict = {
  57       fields.InputDataFields.image: [],
  58       fields.InputDataFields.groundtruth_boxes: [],
  59       fields.InputDataFields.groundtruth_classes: [],
  60       'batch': batch_sequences_with_states,
  61   }
  62   for i in range(unroll_length):
  63     for j in range(batch_size):
  64       filtered_dict = util_ops.filter_groundtruth_with_nan_box_coordinates({
  65           fields.InputDataFields.groundtruth_boxes: (
  66               batch_sequences_with_states.sequences['groundtruth_boxes'][j][i]),
  67           fields.InputDataFields.groundtruth_classes: (
  68               batch_sequences_with_states.sequences['groundtruth_classes'][j][i]
  69           ),
  70       })
  71       filtered_dict = util_ops.retain_groundtruth_with_positive_classes(
  72           filtered_dict)
  73       seq_tensors_dict[fields.InputDataFields.image].append(
  74           batch_sequences_with_states.sequences['image'][j][i])
  75       seq_tensors_dict[fields.InputDataFields.groundtruth_boxes].append(
  76           filtered_dict[fields.InputDataFields.groundtruth_boxes])
  77       seq_tensors_dict[fields.InputDataFields.groundtruth_classes].append(
  78           filtered_dict[fields.InputDataFields.groundtruth_classes])
  79   seq_tensors_dict[fields.InputDataFields.image] = tuple(
  80       seq_tensors_dict[fields.InputDataFields.image])
  81   seq_tensors_dict[fields.InputDataFields.groundtruth_boxes] = tuple(
  82       seq_tensors_dict[fields.InputDataFields.groundtruth_boxes])
  83   seq_tensors_dict[fields.InputDataFields.groundtruth_classes] = tuple(
  84       seq_tensors_dict[fields.InputDataFields.groundtruth_classes])
  85 
  86   return seq_tensors_dict

  89 def build(input_reader_config,
  90           model_config,
  91           lstm_config,
  92           unroll_length,
  93           data_augmentation_options=None,
  94           batch_size=1):
  95   """Builds a tensor dictionary based on the InputReader config.
  96 
  97   Args:
  98     input_reader_config: An input_reader_builder.InputReader object.
  99     model_config: A model.proto object containing the config for the desired
 100       DetectionModel.
 101     lstm_config: LSTM specific configs.
 102     unroll_length: Unrolled length for LSTM training.
 103     data_augmentation_options: A list of tuples, where each tuple contains a
 104       data augmentation function and a dictionary containing arguments and their
 105       values (see preprocessor.py).
 106     batch_size: Batch size for queue outputs.
 107 
 108   Returns:
 109     A dictionary of tensors based on items in the input_reader_config.
 110 
 111   Raises:
 112     ValueError: On invalid input reader proto.
 113     ValueError: If no input paths are specified.
 114   """
 115   if not isinstance(input_reader_config, input_reader_pb2.InputReader):
 116     raise ValueError('input_reader_config not of type '
 117                      'input_reader_pb2.InputReader.')
 118 
 119   external_reader_config = input_reader_config.external_input_reader
 120   external_input_reader_config = external_reader_config.Extensions[
 121       input_reader_google_pb2.GoogleInputReader.google_input_reader]
 122   input_reader_type = external_input_reader_config.WhichOneof('input_reader')
 123 
 124   if input_reader_type == 'tf_record_video_input_reader':
 125     config = external_input_reader_config.tf_record_video_input_reader
 126     reader_type_class = tf.TFRecordReader
 127   else:
 128     raise ValueError(
 129         'Unsupported reader in input_reader_config: %s' % input_reader_type)
 130 
 131   if not config.input_path:
 132     raise ValueError('At least one input path must be specified in '
 133                      '`input_reader_config`.')
 134   key, value = parallel_reader.parallel_read(
 135       config.input_path[:],  # Convert `RepeatedScalarContainer` to list.
 136       reader_class=reader_type_class,
 137       num_epochs=(input_reader_config.num_epochs
 138                   if input_reader_config.num_epochs else None),
 139       num_readers=input_reader_config.num_readers,
 140       shuffle=input_reader_config.shuffle,
 141       dtypes=[tf.string, tf.string],
 142       capacity=input_reader_config.queue_capacity,
 143       min_after_dequeue=input_reader_config.min_after_dequeue)
 144 
 145   # TODO(yinxiao): Add loading instance mask option.
 146   decoder = tf_sequence_example_decoder.TFSequenceExampleDecoder()
 147 
 148   keys_to_decode = [
 149       fields.InputDataFields.image, fields.InputDataFields.groundtruth_boxes,
 150       fields.InputDataFields.groundtruth_classes
 151   ]
 152   tensor_dict = decoder.decode(value, items=keys_to_decode)
 153 
 154   tensor_dict['image'].set_shape([None, None, None, 3])
 155   tensor_dict['groundtruth_boxes'].set_shape([None, None, 4])
 156 
 157   height = model_config.ssd.image_resizer.fixed_shape_resizer.height
 158   width = model_config.ssd.image_resizer.fixed_shape_resizer.width
 159 
 160   # If data augmentation is specified in the config file, the preprocessor
 161   # will be called here to augment the data as specified. Most common
 162   # augmentations include horizontal flip and cropping.
 163   if data_augmentation_options:
 164     images_pre = tf.split(tensor_dict['image'], config.video_length, axis=0)
 165     bboxes_pre = tf.split(
 166         tensor_dict['groundtruth_boxes'], config.video_length, axis=0)
 167     labels_pre = tf.split(
 168         tensor_dict['groundtruth_classes'], config.video_length, axis=0)
 169     images_proc, bboxes_proc, labels_proc = [], [], []
 170     cache = preprocessor_cache.PreprocessorCache()
 171 
 172     for i, _ in enumerate(images_pre):
 173       image_dict = {
 174           fields.InputDataFields.image:
 175               images_pre[i],
 176           fields.InputDataFields.groundtruth_boxes:
 177               tf.squeeze(bboxes_pre[i], axis=0),
 178           fields.InputDataFields.groundtruth_classes:
 179               tf.squeeze(labels_pre[i], axis=0),
 180       }
 181       image_dict = preprocessor.preprocess(
 182           image_dict,
 183           data_augmentation_options,
 184           func_arg_map=preprocessor.get_default_func_arg_map(),
 185           preprocess_vars_cache=cache)
 186       # Pads detection count to _PADDING_SIZE.
 187       image_dict[fields.InputDataFields.groundtruth_boxes] = tf.pad(
 188           image_dict[fields.InputDataFields.groundtruth_boxes],
 189           [[0, _PADDING_SIZE], [0, 0]])
 190       image_dict[fields.InputDataFields.groundtruth_boxes] = tf.slice(
 191           image_dict[fields.InputDataFields.groundtruth_boxes], [0, 0],
 192           [_PADDING_SIZE, -1])
 193       image_dict[fields.InputDataFields.groundtruth_classes] = tf.pad(
 194           image_dict[fields.InputDataFields.groundtruth_classes],
 195           [[0, _PADDING_SIZE]])
 196       image_dict[fields.InputDataFields.groundtruth_classes] = tf.slice(
 197           image_dict[fields.InputDataFields.groundtruth_classes], [0],
 198           [_PADDING_SIZE])
 199       images_proc.append(image_dict[fields.InputDataFields.image])
 200       bboxes_proc.append(image_dict[fields.InputDataFields.groundtruth_boxes])
 201       labels_proc.append(image_dict[fields.InputDataFields.groundtruth_classes])
 202     tensor_dict['image'] = tf.concat(images_proc, axis=0)
 203     tensor_dict['groundtruth_boxes'] = tf.stack(bboxes_proc, axis=0)
 204     tensor_dict['groundtruth_classes'] = tf.stack(labels_proc, axis=0)
 205   else:
 206     # Pads detection count to _PADDING_SIZE per frame.
 207     tensor_dict['groundtruth_boxes'] = tf.pad(
 208         tensor_dict['groundtruth_boxes'], [[0, 0], [0, _PADDING_SIZE], [0, 0]])
 209     tensor_dict['groundtruth_boxes'] = tf.slice(
 210         tensor_dict['groundtruth_boxes'], [0, 0, 0], [-1, _PADDING_SIZE, -1])
 211     tensor_dict['groundtruth_classes'] = tf.pad(
 212         tensor_dict['groundtruth_classes'], [[0, 0], [0, _PADDING_SIZE]])
 213     tensor_dict['groundtruth_classes'] = tf.slice(
 214         tensor_dict['groundtruth_classes'], [0, 0], [-1, _PADDING_SIZE])
 215 
 216   tensor_dict['image'], _ = preprocessor.resize_image(
 217       tensor_dict['image'], new_height=height, new_width=width)
 218 
 219   num_steps = config.video_length / unroll_length
 220 
 221   init_states = {
 222       'lstm_state_c':
 223           tf.zeros([height / 32, width / 32, lstm_config.lstm_state_depth]),
 224       'lstm_state_h':
 225           tf.zeros([height / 32, width / 32, lstm_config.lstm_state_depth]),
 226       'lstm_state_step':
 227           tf.constant(num_steps, shape=[]),
 228   }
 229 
 230   batch = sqss.batch_sequences_with_states(
 231       input_key=key,
 232       input_sequences=tensor_dict,
 233       input_context={},
 234       input_length=None,
 235       initial_states=init_states,
 236       num_unroll=unroll_length,
 237       batch_size=batch_size,
 238       num_threads=batch_size,
 239       make_keys_unique=True,
 240       capacity=batch_size * batch_size)
 241 
 242   return _build_training_batch_dict(batch, unroll_length, batch_size)
