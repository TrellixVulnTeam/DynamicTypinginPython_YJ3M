  29 class BoundingBoxSequence(tfexample_decoder.ItemHandler):
  30   """An ItemHandler that concatenates SparseTensors to Bounding Boxes.
  31   """
  32 
  33   def __init__(self, keys=None, prefix=None, return_dense=True,
  34                default_value=-1.0):
  35     """Initialize the bounding box handler.
  36 
  37     Args:
  38       keys: A list of four key names representing the ymin, xmin, ymax, xmax
  39         in the Example or SequenceExample.
  40       prefix: An optional prefix for each of the bounding box keys in the
  41         Example or SequenceExample. If provided, `prefix` is prepended to each
  42         key in `keys`.
  43       return_dense: if True, returns a dense tensor; if False, returns as
  44         sparse tensor.
  45       default_value: The value used when the `tensor_key` is not found in a
  46         particular `TFExample`.
  47 
  48     Raises:
  49       ValueError: if keys is not `None` and also not a list of exactly 4 keys
  50     """
  51     if keys is None:
  52       keys = ['ymin', 'xmin', 'ymax', 'xmax']
  53     elif len(keys) != 4:
  54       raise ValueError('BoundingBoxSequence expects 4 keys but got {}'.format(
  55           len(keys)))
  56     self._prefix = prefix
  57     self._keys = keys
  58     self._full_keys = [prefix + k for k in keys]
  59     self._return_dense = return_dense
  60     self._default_value = default_value
  61     super(BoundingBoxSequence, self).__init__(self._full_keys)
  62 
  63   def tensors_to_item(self, keys_to_tensors):
  64     """Maps the given dictionary of tensors to a concatenated list of bboxes.
  65 
  66     Args:
  67       keys_to_tensors: a mapping of TF-Example keys to parsed tensors.
  68 
  69     Returns:
  70       [time, num_boxes, 4] tensor of bounding box coordinates, in order
  71           [y_min, x_min, y_max, x_max]. Whether the tensor is a SparseTensor
  72           or a dense Tensor is determined by the return_dense parameter. Empty
  73           positions in the sparse tensor are filled with -1.0 values.
  74     """
  75     sides = []
  76     for key in self._full_keys:
  77       value = keys_to_tensors[key]
  78       expanded_dims = tf.concat(
  79           [tf.to_int64(tf.shape(value)),
  80            tf.constant([1], dtype=tf.int64)], 0)
  81       side = tf.sparse_reshape(value, expanded_dims)
  82       sides.append(side)
  83     bounding_boxes = tf.sparse_concat(2, sides)
  84     if self._return_dense:
  85       bounding_boxes = tf.sparse_tensor_to_dense(
  86           bounding_boxes, default_value=self._default_value)
  87     return bounding_boxes

  90 class TFSequenceExampleDecoder(data_decoder.DataDecoder):
  91   """Tensorflow Sequence Example proto decoder."""
  92 
  93   def __init__(self):
  94     """Constructor sets keys_to_features and items_to_handlers."""
  95     self.keys_to_context_features = {
  96         'image/format':
  97             tf.FixedLenFeature((), tf.string, default_value='jpeg'),
  98         'image/filename':
  99             tf.FixedLenFeature((), tf.string, default_value=''),
 100         'image/key/sha256':
 101             tf.FixedLenFeature((), tf.string, default_value=''),
 102         'image/source_id':
 103             tf.FixedLenFeature((), tf.string, default_value=''),
 104         'image/height':
 105             tf.FixedLenFeature((), tf.int64, 1),
 106         'image/width':
 107             tf.FixedLenFeature((), tf.int64, 1),
 108     }
 109     self.keys_to_features = {
 110         'image/encoded': tf.FixedLenSequenceFeature((), tf.string),
 111         'bbox/xmin': tf.VarLenFeature(dtype=tf.float32),
 112         'bbox/xmax': tf.VarLenFeature(dtype=tf.float32),
 113         'bbox/ymin': tf.VarLenFeature(dtype=tf.float32),
 114         'bbox/ymax': tf.VarLenFeature(dtype=tf.float32),
 115         'bbox/label/index': tf.VarLenFeature(dtype=tf.int64),
 116         'bbox/label/string': tf.VarLenFeature(tf.string),
 117         'area': tf.VarLenFeature(tf.float32),
 118         'is_crowd': tf.VarLenFeature(tf.int64),
 119         'difficult': tf.VarLenFeature(tf.int64),
 120         'group_of': tf.VarLenFeature(tf.int64),
 121     }
 122     self.items_to_handlers = {
 123         fields.InputDataFields.image:
 124             tfexample_decoder.Image(
 125                 image_key='image/encoded',
 126                 format_key='image/format',
 127                 channels=3,
 128                 repeated=True),
 129         fields.InputDataFields.source_id: (
 130             tfexample_decoder.Tensor('image/source_id')),
 131         fields.InputDataFields.key: (
 132             tfexample_decoder.Tensor('image/key/sha256')),
 133         fields.InputDataFields.filename: (
 134             tfexample_decoder.Tensor('image/filename')),
 135         # Object boxes and classes.
 136         fields.InputDataFields.groundtruth_boxes:
 137             BoundingBoxSequence(prefix='bbox/'),
 138         fields.InputDataFields.groundtruth_classes: (
 139             tfexample_decoder.Tensor('bbox/label/index')),
 140         fields.InputDataFields.groundtruth_area:
 141             tfexample_decoder.Tensor('area'),
 142         fields.InputDataFields.groundtruth_is_crowd: (
 143             tfexample_decoder.Tensor('is_crowd')),
 144         fields.InputDataFields.groundtruth_difficult: (
 145             tfexample_decoder.Tensor('difficult')),
 146         fields.InputDataFields.groundtruth_group_of: (
 147             tfexample_decoder.Tensor('group_of'))
 148     }
 149 
 150   def decode(self, tf_seq_example_string_tensor, items=None):
 151     """Decodes serialized tf.SequenceExample and returns a tensor dictionary.
 152 
 153     Args:
 154       tf_seq_example_string_tensor: A string tensor holding a serialized
 155         tensorflow example proto.
 156       items: The list of items to decode. These must be a subset of the item
 157         keys in self._items_to_handlers. If `items` is left as None, then all
 158         of the items in self._items_to_handlers are decoded.
 159 
 160     Returns:
 161       A dictionary of the following tensors.
 162       fields.InputDataFields.image - 3D uint8 tensor of shape [None, None, seq]
 163         containing image(s).
 164       fields.InputDataFields.source_id - string tensor containing original
 165         image id.
 166       fields.InputDataFields.key - string tensor with unique sha256 hash key.
 167       fields.InputDataFields.filename - string tensor with original dataset
 168         filename.
 169       fields.InputDataFields.groundtruth_boxes - 2D float32 tensor of shape
 170         [None, 4] containing box corners.
 171       fields.InputDataFields.groundtruth_classes - 1D int64 tensor of shape
 172         [None] containing classes for the boxes.
 173       fields.InputDataFields.groundtruth_area - 1D float32 tensor of shape
 174         [None] containing object mask area in pixel squared.
 175       fields.InputDataFields.groundtruth_is_crowd - 1D bool tensor of shape
 176         [None] indicating if the boxes enclose a crowd.
 177       fields.InputDataFields.groundtruth_difficult - 1D bool tensor of shape
 178         [None] indicating if the boxes represent `difficult` instances.
 179     """
 180     serialized_example = tf.reshape(tf_seq_example_string_tensor, shape=[])
 181     decoder = TFSequenceExampleDecoderHelper(self.keys_to_context_features,
 182                                              self.keys_to_features,
 183                                              self.items_to_handlers)
 184     if not items:
 185       items = decoder.list_items()
 186     tensors = decoder.decode(serialized_example, items=items)
 187     tensor_dict = dict(zip(items, tensors))
 188 
 189     return tensor_dict

 192 class TFSequenceExampleDecoderHelper(data_decoder.DataDecoder):
 193   """A decoder helper class for TensorFlow SequenceExamples.
 194 
 195   To perform this decoding operation, a SequenceExampleDecoder is given a list
 196   of ItemHandlers. Each ItemHandler indicates the set of features.
 197   """
 198 
 199   def __init__(self, keys_to_context_features, keys_to_sequence_features,
 200                items_to_handlers):
 201     """Constructs the decoder.
 202 
 203     Args:
 204       keys_to_context_features: A dictionary from TF-SequenceExample context
 205         keys to either tf.VarLenFeature or tf.FixedLenFeature instances.
 206         See tensorflow's parsing_ops.py.
 207       keys_to_sequence_features: A dictionary from TF-SequenceExample sequence
 208         keys to either tf.VarLenFeature or tf.FixedLenSequenceFeature instances.
 209       items_to_handlers: A dictionary from items (strings) to ItemHandler
 210         instances. Note that the ItemHandler's are provided the keys that they
 211         use to return the final item Tensors.
 212     Raises:
 213       ValueError: If the same key is present for context features and sequence
 214         features.
 215     """
 216     unique_keys = set()
 217     unique_keys.update(keys_to_context_features)
 218     unique_keys.update(keys_to_sequence_features)
 219     if len(unique_keys) != (
 220         len(keys_to_context_features) + len(keys_to_sequence_features)):
 221       # This situation is ambiguous in the decoder's keys_to_tensors variable.
 222       raise ValueError('Context and sequence keys are not unique. \n'
 223                        ' Context keys: %s \n Sequence keys: %s' %
 224                        (list(keys_to_context_features.keys()),
 225                         list(keys_to_sequence_features.keys())))
 226     self._keys_to_context_features = keys_to_context_features
 227     self._keys_to_sequence_features = keys_to_sequence_features
 228     self._items_to_handlers = items_to_handlers
 229 
 230   def list_items(self):
 231     """Returns keys of items."""
 232     return self._items_to_handlers.keys()
 233 
 234   def decode(self, serialized_example, items=None):
 235     """Decodes the given serialized TF-SequenceExample.
 236 
 237     Args:
 238       serialized_example: A serialized TF-SequenceExample tensor.
 239       items: The list of items to decode. These must be a subset of the item
 240         keys in self._items_to_handlers. If `items` is left as None, then all
 241         of the items in self._items_to_handlers are decoded.
 242     Returns:
 243       The decoded items, a list of tensor.
 244     """
 245     context, feature_list = tf.parse_single_sequence_example(
 246         serialized_example, self._keys_to_context_features,
 247         self._keys_to_sequence_features)
 248     # Reshape non-sparse elements just once:
 249     for k in self._keys_to_context_features:
 250       v = self._keys_to_context_features[k]
 251       if isinstance(v, tf.FixedLenFeature):
 252         context[k] = tf.reshape(context[k], v.shape)
 253     if not items:
 254       items = self._items_to_handlers.keys()
 255     outputs = []
 256     for item in items:
 257       handler = self._items_to_handlers[item]
 258       keys_to_tensors = {
 259           key: context[key] if key in context else feature_list[key]
 260           for key in handler.keys
 261       }
 262       outputs.append(handler.tensors_to_item(keys_to_tensors))
 263     return outputs
