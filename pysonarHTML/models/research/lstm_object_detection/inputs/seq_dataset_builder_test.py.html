 195   def _create_training_dict(self, tensor_dict):
 196     image_dict = {}
 197     all_dict = {}
 198     all_dict['batch'] = tensor_dict.pop('batch')
 199     for i, _ in enumerate(tensor_dict[fields.InputDataFields.image]):
 200       for key, val in tensor_dict.items():
 201         image_dict[key] = val[i]
 202 
 203       image_dict[fields.InputDataFields.image] = tf.to_float(
 204           tf.expand_dims(image_dict[fields.InputDataFields.image], 0))
 205       suffix = str(i)
 206       for key, val in image_dict.items():
 207         all_dict[key + suffix] = val
 208     return all_dict

 210   def _get_input_proto(self, input_reader):
 211     return """
 212         external_input_reader {
 213           [lstm_object_detection.protos.GoogleInputReader.google_input_reader] {
 214             %s: {
 215               input_path: '{0}'
 216               data_type: TF_SEQUENCE_EXAMPLE
 217               video_length: 4
 218             }
 219           }
 220         }
 221       """ % input_reader

 223   def test_video_input_reader(self):
 224     input_reader_proto = input_reader_pb2.InputReader()
 225     text_format.Merge(
 226         self._get_input_proto('tf_record_video_input_reader'),
 227         input_reader_proto)
 228 
 229     configs = self._get_model_configs_from_proto()
 230     tensor_dict = seq_dataset_builder.build(
 231         input_reader_proto,
 232         configs['model'],
 233         configs['lstm_model'],
 234         unroll_length=1)
 235 
 236     all_dict = self._create_training_dict(tensor_dict)
 237 
 238     self.assertEqual((1, 32, 32, 3), all_dict['image0'].shape)
 239     self.assertEqual(4, all_dict['groundtruth_boxes0'].shape[1])

 241   def test_build_with_data_augmentation(self):
 242     input_reader_proto = input_reader_pb2.InputReader()
 243     text_format.Merge(
 244         self._get_input_proto('tf_record_video_input_reader'),
 245         input_reader_proto)
 246 
 247     configs = self._get_model_configs_from_proto()
 248     data_augmentation_options = [
 249         preprocessor_builder.build(
 250             self._get_data_augmentation_preprocessor_proto())
 251     ]
 252     tensor_dict = seq_dataset_builder.build(
 253         input_reader_proto,
 254         configs['model'],
 255         configs['lstm_model'],
 256         unroll_length=1,
 257         data_augmentation_options=data_augmentation_options)
 258 
 259     all_dict = self._create_training_dict(tensor_dict)
 260     self.assertEqual((1, 32, 32, 3), all_dict['image0'].shape)
 261     self.assertEqual(4, all_dict['groundtruth_boxes0'].shape[1])

 263   def test_raises_error_without_input_paths(self):
 264     input_reader_text_proto = """
 265       shuffle: false
 266       num_readers: 1
 267       load_instance_masks: true
 268     """
 269     input_reader_proto = input_reader_pb2.InputReader()
 270     text_format.Merge(input_reader_text_proto, input_reader_proto)
 271 
 272     configs = self._get_model_configs_from_proto()
 273     with self.assertRaises(ValueError):
 274       _ = seq_dataset_builder.build(
 275           input_reader_proto,
 276           configs['model'],
 277           configs['lstm_model'],
 278           unroll_length=1)
