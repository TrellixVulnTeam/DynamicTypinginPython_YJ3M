  30   def _create_feature_extractor(self,
  31                                 depth_multiplier,
  32                                 pad_to_multiple,
  33                                 is_quantized=False):
  34     """Constructs a new feature extractor.
  35 
  36     Args:
  37       depth_multiplier: float depth multiplier for feature extractor
  38       pad_to_multiple: the nearest multiple to zero pad the input height and
  39         width dimensions to.
  40       is_quantized: whether to quantize the graph.
  41     Returns:
  42       an ssd_meta_arch.SSDFeatureExtractor object.
  43     """
  44     min_depth = 32
  45     def conv_hyperparams_fn():
  46       with slim.arg_scope([slim.conv2d], normalizer_fn=slim.batch_norm), \
  47         slim.arg_scope([slim.batch_norm], is_training=False) as sc:
  48         return sc
  49     feature_extractor = (
  50         lstm_ssd_interleaved_mobilenet_v2_feature_extractor
  51         .LSTMSSDInterleavedMobilenetV2FeatureExtractor(False, depth_multiplier,
  52                                                        min_depth,
  53                                                        pad_to_multiple,
  54                                                        conv_hyperparams_fn))
  55     feature_extractor.lstm_state_depth = int(320 * depth_multiplier)
  56     feature_extractor.depth_multipliers = [
  57         depth_multiplier, depth_multiplier / 4.0
  58     ]
  59     feature_extractor.is_quantized = is_quantized
  60     return feature_extractor
  61 
  62   def test_feature_extractor_construct_with_expected_params(self):
  63     def conv_hyperparams_fn():
  64       with (slim.arg_scope([slim.conv2d], normalizer_fn=slim.batch_norm) and
  65             slim.arg_scope([slim.batch_norm], decay=0.97, epsilon=1e-3)) as sc:
  66         return sc
  67 
  68     params = {
  69         'is_training': True,
  70         'depth_multiplier': .55,
  71         'min_depth': 9,
  72         'pad_to_multiple': 3,
  73         'conv_hyperparams_fn': conv_hyperparams_fn,
  74         'reuse_weights': False,
  75         'use_explicit_padding': True,
  76         'use_depthwise': False,
  77         'override_base_feature_extractor_hyperparams': True}
  78 
  79     feature_extractor = (
  80         lstm_ssd_interleaved_mobilenet_v2_feature_extractor
  81         .LSTMSSDInterleavedMobilenetV2FeatureExtractor(**params))
  82 
  83     self.assertEqual(params['is_training'],
  84                      feature_extractor._is_training)
  85     self.assertEqual(params['depth_multiplier'],
  86                      feature_extractor._depth_multiplier)
  87     self.assertEqual(params['min_depth'],
  88                      feature_extractor._min_depth)
  89     self.assertEqual(params['pad_to_multiple'],
  90                      feature_extractor._pad_to_multiple)
  91     self.assertEqual(params['conv_hyperparams_fn'],
  92                      feature_extractor._conv_hyperparams_fn)
  93     self.assertEqual(params['reuse_weights'],
  94                      feature_extractor._reuse_weights)
  95     self.assertEqual(params['use_explicit_padding'],
  96                      feature_extractor._use_explicit_padding)
  97     self.assertEqual(params['use_depthwise'],
  98                      feature_extractor._use_depthwise)
  99     self.assertEqual(params['override_base_feature_extractor_hyperparams'],
 100                      (feature_extractor.
 101                       _override_base_feature_extractor_hyperparams))

 255   def test_lstm_states(self):
 256     image_height = 256
 257     image_width = 256
 258     depth_multiplier = 1
 259     pad_to_multiple = 1
 260     state_channel = 320
 261     init_state1 = {
 262         'lstm_state_c': tf.zeros(
 263             [image_height // 32, image_width // 32, state_channel]),
 264         'lstm_state_h': tf.zeros(
 265             [image_height // 32, image_width // 32, state_channel]),
 266         'lstm_state_step': tf.zeros([1])
 267     }
 268     init_state2 = {
 269         'lstm_state_c': tf.random_uniform(
 270             [image_height // 32, image_width // 32, state_channel]),
 271         'lstm_state_h': tf.random_uniform(
 272             [image_height // 32, image_width // 32, state_channel]),
 273         'lstm_state_step': tf.zeros([1])
 274     }
 275     seq = {'dummy': tf.random_uniform([2, 1, 1, 1])}
 276     stateful_reader1 = contrib_training.SequenceQueueingStateSaver(
 277         batch_size=1,
 278         num_unroll=1,
 279         input_length=2,
 280         input_key='',
 281         input_sequences=seq,
 282         input_context={},
 283         initial_states=init_state1,
 284         capacity=1)
 285     stateful_reader2 = contrib_training.SequenceQueueingStateSaver(
 286         batch_size=1,
 287         num_unroll=1,
 288         input_length=2,
 289         input_key='',
 290         input_sequences=seq,
 291         input_context={},
 292         initial_states=init_state2,
 293         capacity=1)
 294     image = tf.random_uniform([1, image_height, image_width, 3])
 295     feature_extractor = self._create_feature_extractor(depth_multiplier,
 296                                                        pad_to_multiple)
 297     with tf.variable_scope('zero_state'):
 298       feature_maps1 = feature_extractor.extract_features(
 299           image, stateful_reader1.next_batch, unroll_length=1)
 300     with tf.variable_scope('random_state'):
 301       feature_maps2 = feature_extractor.extract_features(
 302           image, stateful_reader2.next_batch, unroll_length=1)
 303     with tf.Session() as sess:
 304       sess.run(tf.global_variables_initializer())
 305       sess.run(tf.local_variables_initializer())
 306       sess.run(tf.get_collection(tf.GraphKeys.TABLE_INITIALIZERS))
 307       sess.run([stateful_reader1.prefetch_op, stateful_reader2.prefetch_op])
 308       maps1, maps2 = sess.run([feature_maps1, feature_maps2])
 309       state = sess.run(stateful_reader1.next_batch.state('lstm_state_c'))
 310     # feature maps should be different because states are different
 311     self.assertFalse(np.all(np.equal(maps1[0], maps2[0])))
 312     # state should no longer be zero after update
 313     self.assertTrue(state.any())

 315   def check_extract_features_returns_correct_shape(
 316       self, batch_size, image_height, image_width, depth_multiplier,
 317       pad_to_multiple, expected_feature_map_shapes, unroll_length=1):
 318     def graph_fn(image_tensor):
 319       feature_extractor = self._create_feature_extractor(depth_multiplier,
 320                                                          pad_to_multiple)
 321       feature_maps = feature_extractor.extract_features(
 322           image_tensor, unroll_length=unroll_length)
 323       return feature_maps
 324 
 325     image_tensor = np.random.rand(batch_size, image_height, image_width,
 326                                   3).astype(np.float32)
 327     feature_maps = self.execute(graph_fn, [image_tensor])
 328     for feature_map, expected_shape in zip(
 329         feature_maps, expected_feature_map_shapes):
 330       self.assertAllEqual(feature_map.shape, expected_shape)

 332   def check_feature_extractor_variables_under_scopes(
 333       self, depth_multiplier, pad_to_multiple, scope_names):
 334     g = tf.Graph()
 335     with g.as_default():
 336       feature_extractor = self._create_feature_extractor(
 337           depth_multiplier, pad_to_multiple)
 338       preprocessed_inputs = tf.placeholder(tf.float32, (4, 320, 320, 3))
 339       feature_extractor.extract_features(
 340           preprocessed_inputs, unroll_length=1)
 341       variables = g.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)
 342       for variable in variables:
 343         self.assertTrue(
 344             any([
 345                 variable.name.startswith(scope_name)
 346                 for scope_name in scope_names
 347             ]), 'Variable name: ' + variable.name +
 348             ' is not under any provided scopes: ' + ','.join(scope_names))
