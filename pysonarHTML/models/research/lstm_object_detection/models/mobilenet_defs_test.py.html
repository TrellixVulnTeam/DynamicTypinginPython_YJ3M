  74 class MobilenetV2DefsTest(tf.test.TestCase):
  75 
  76   def test_mobilenet_v2_lite_def(self):
  77     net, features = mobilenet_v2.mobilenet_base(
  78         tf.placeholder(tf.float32, (10, 320, 320, 3)),
  79         min_depth=8,
  80         depth_multiplier=1.0,
  81         conv_defs=mobilenet_defs.mobilenet_v2_lite_def(),
  82         use_explicit_padding=True,
  83         scope='MobilenetV2')
  84     self.assertEqual(net.get_shape().as_list(), [10, 10, 10, 320])
  85     self._assert_contains_op('MobilenetV2/expanded_conv_16/project/Identity')
  86     self.assertEqual(
  87         features['layer_3/expansion_output'].get_shape().as_list(),
  88         [10, 160, 160, 96])
  89     self.assertEqual(
  90         features['layer_4/expansion_output'].get_shape().as_list(),
  91         [10, 80, 80, 144])
  92 
  93   def test_mobilenet_v2_lite_def_is_quantized(self):
  94     net, _ = mobilenet_v2.mobilenet_base(
  95         tf.placeholder(tf.float32, (10, 320, 320, 3)),
  96         min_depth=8,
  97         depth_multiplier=1.0,
  98         conv_defs=mobilenet_defs.mobilenet_v2_lite_def(is_quantized=True),
  99         use_explicit_padding=True,
 100         scope='MobilenetV2')
 101     self.assertEqual(net.get_shape().as_list(), [10, 10, 10, 320])
 102     self._assert_contains_op('MobilenetV2/expanded_conv_16/project/Relu6')
 103 
 104   def test_mobilenet_v2_lite_def_low_res(self):
 105     net, _ = mobilenet_v2.mobilenet_base(
 106         tf.placeholder(tf.float32, (10, 320, 320, 3)),
 107         min_depth=8,
 108         depth_multiplier=1.0,
 109         conv_defs=mobilenet_defs.mobilenet_v2_lite_def(low_res=True),
 110         use_explicit_padding=True,
 111         scope='MobilenetV2')
 112     self.assertEqual(net.get_shape().as_list(), [10, 20, 20, 320])
 113 
 114   def test_mobilenet_v2_lite_def_reduced(self):
 115     net, features = mobilenet_v2.mobilenet_base(
 116         tf.placeholder(tf.float32, (10, 320, 320, 3)),
 117         min_depth=8,
 118         depth_multiplier=1.0,
 119         conv_defs=mobilenet_defs.mobilenet_v2_lite_def(reduced=True),
 120         use_explicit_padding=True,
 121         scope='MobilenetV2')
 122     self.assertEqual(net.get_shape().as_list(), [10, 10, 10, 320])
 123     self.assertEqual(
 124         features['layer_3/expansion_output'].get_shape().as_list(),
 125         [10, 160, 160, 48])
 126     self.assertEqual(
 127         features['layer_4/expansion_output'].get_shape().as_list(),
 128         [10, 80, 80, 72])
 129 
 130   def _assert_contains_op(self, op_name):
 131     op_names = [op.name for op in tf.get_default_graph().get_operations()]
 132     self.assertIn(op_name, op_names)
