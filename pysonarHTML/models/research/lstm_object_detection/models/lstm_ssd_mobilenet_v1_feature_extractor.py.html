  31 class LSTMSSDMobileNetV1FeatureExtractor(
  32     lstm_ssd_meta_arch.LSTMSSDFeatureExtractor):
  33   """LSTM Feature Extractor using MobilenetV1 features."""
  34 
  35   def __init__(self,
  36                is_training,
  37                depth_multiplier,
  38                min_depth,
  39                pad_to_multiple,
  40                conv_hyperparams_fn,
  41                reuse_weights=None,
  42                use_explicit_padding=False,
  43                use_depthwise=True,
  44                override_base_feature_extractor_hyperparams=False,
  45                lstm_state_depth=256):
  46     """Initializes instance of MobileNetV1 Feature Extractor for LSTMSSD Models.
  47 
  48     Args:
  49       is_training: A boolean whether the network is in training mode.
  50       depth_multiplier: A float depth multiplier for feature extractor.
  51       min_depth: A number representing minimum feature extractor depth.
  52       pad_to_multiple: The nearest multiple to zero pad the input height and
  53         width dimensions to.
  54       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  55         and separable_conv2d ops in the layers that are added on top of the
  56         base feature extractor.
  57       reuse_weights: Whether to reuse variables. Default is None.
  58       use_explicit_padding: Whether to use explicit padding when extracting
  59         features. Default is False.
  60       use_depthwise: Whether to use depthwise convolutions. Default is True.
  61       override_base_feature_extractor_hyperparams: Whether to override
  62         hyperparameters of the base feature extractor with the one from
  63         `conv_hyperparams_fn`.
  64       lstm_state_depth: An integter of the depth of the lstm state.
  65     """
  66     super(LSTMSSDMobileNetV1FeatureExtractor, self).__init__(
  67         is_training=is_training,
  68         depth_multiplier=depth_multiplier,
  69         min_depth=min_depth,
  70         pad_to_multiple=pad_to_multiple,
  71         conv_hyperparams_fn=conv_hyperparams_fn,
  72         reuse_weights=reuse_weights,
  73         use_explicit_padding=use_explicit_padding,
  74         use_depthwise=use_depthwise,
  75         override_base_feature_extractor_hyperparams=
  76         override_base_feature_extractor_hyperparams)
  77     self._feature_map_layout = {
  78         'from_layer': ['Conv2d_13_pointwise_lstm', '', '', '', ''],
  79         'layer_depth': [-1, 512, 256, 256, 128],
  80         'use_explicit_padding': self._use_explicit_padding,
  81         'use_depthwise': self._use_depthwise,
  82     }
  83     self._base_network_scope = 'MobilenetV1'
  84     self._lstm_state_depth = lstm_state_depth
  85 
  86   def create_lstm_cell(self, batch_size, output_size, state_saver, state_name,
  87                        dtype=tf.float32):
  88     """Create the LSTM cell, and initialize state if necessary.
  89 
  90     Args:
  91       batch_size: input batch size.
  92       output_size: output size of the lstm cell, [width, height].
  93       state_saver: a state saver object with methods `state` and `save_state`.
  94       state_name: string, the name to use with the state_saver.
  95       dtype: dtype to initialize lstm state.
  96 
  97     Returns:
  98       lstm_cell: the lstm cell unit.
  99       init_state: initial state representations.
 100       step: the step
 101     """
 102     lstm_cell = lstm_cells.BottleneckConvLSTMCell(
 103         filter_size=(3, 3),
 104         output_size=output_size,
 105         num_units=max(self._min_depth, self._lstm_state_depth),
 106         activation=tf.nn.relu6,
 107         visualize_gates=False)
 108 
 109     if state_saver is None:
 110       init_state = lstm_cell.init_state(state_name, batch_size, dtype)
 111       step = None
 112     else:
 113       step = state_saver.state(state_name + '_step')
 114       c = state_saver.state(state_name + '_c')
 115       h = state_saver.state(state_name + '_h')
 116       init_state = (c, h)
 117     return lstm_cell, init_state, step
 118 
 119   def extract_features(self,
 120                        preprocessed_inputs,
 121                        state_saver=None,
 122                        state_name='lstm_state',
 123                        unroll_length=5,
 124                        scope=None):
 125     """Extracts features from preprocessed inputs.
 126 
 127     The features include the base network features, lstm features and SSD
 128     features, organized in the following name scope:
 129 
 130     <parent scope>/MobilenetV1/...
 131     <parent scope>/LSTM/...
 132     <parent scope>/FeatureMaps/...
 133 
 134     Args:
 135       preprocessed_inputs: A [batch, height, width, channels] float tensor
 136         representing a batch of consecutive frames from video clips.
 137       state_saver: A state saver object with methods `state` and `save_state`.
 138       state_name: A python string for the name to use with the state_saver.
 139       unroll_length: The number of steps to unroll the lstm.
 140       scope: The scope for the base network of the feature extractor.
 141 
 142     Returns:
 143       A list of tensors where the ith tensor has shape [batch, height_i,
 144       width_i, depth_i]
 145     """
 146     preprocessed_inputs = shape_utils.check_min_image_dim(
 147         33, preprocessed_inputs)
 148     with slim.arg_scope(
 149         mobilenet_v1.mobilenet_v1_arg_scope(is_training=self._is_training)):
 150       with (slim.arg_scope(self._conv_hyperparams_fn())
 151             if self._override_base_feature_extractor_hyperparams else
 152             context_manager.IdentityContextManager()):
 153         with slim.arg_scope([slim.batch_norm], fused=False):
 154           # Base network.
 155           with tf.variable_scope(
 156               scope, self._base_network_scope,
 157               reuse=self._reuse_weights) as scope:
 158             net, image_features = mobilenet_v1.mobilenet_v1_base(
 159                 ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple),
 160                 final_endpoint='Conv2d_13_pointwise',
 161                 min_depth=self._min_depth,
 162                 depth_multiplier=self._depth_multiplier,
 163                 scope=scope)
 164 
 165     with slim.arg_scope(self._conv_hyperparams_fn()):
 166       with slim.arg_scope(
 167           [slim.batch_norm], fused=False, is_training=self._is_training):
 168         # ConvLSTM layers.
 169         batch_size = net.shape[0].value // unroll_length
 170         with tf.variable_scope('LSTM', reuse=self._reuse_weights) as lstm_scope:
 171           lstm_cell, init_state, _ = self.create_lstm_cell(
 172               batch_size,
 173               (net.shape[1].value, net.shape[2].value),
 174               state_saver,
 175               state_name,
 176               dtype=preprocessed_inputs.dtype)
 177           net_seq = list(tf.split(net, unroll_length))
 178 
 179           # Identities added for inputing state tensors externally.
 180           c_ident = tf.identity(init_state[0], name='lstm_state_in_c')
 181           h_ident = tf.identity(init_state[1], name='lstm_state_in_h')
 182           init_state = (c_ident, h_ident)
 183 
 184           net_seq, states_out = rnn_decoder.rnn_decoder(
 185               net_seq, init_state, lstm_cell, scope=lstm_scope)
 186           batcher_ops = None
 187           self._states_out = states_out
 188           if state_saver is not None:
 189             self._step = state_saver.state('%s_step' % state_name)
 190             batcher_ops = [
 191                 state_saver.save_state('%s_c' % state_name, states_out[-1][0]),
 192                 state_saver.save_state('%s_h' % state_name, states_out[-1][1]),
 193                 state_saver.save_state('%s_step' % state_name, self._step + 1)
 194             ]
 195           with tf_ops.control_dependencies(batcher_ops):
 196             image_features['Conv2d_13_pointwise_lstm'] = tf.concat(net_seq, 0)
 197 
 198           # Identities added for reading output states, to be reused externally.
 199           tf.identity(states_out[-1][0], name='lstm_state_out_c')
 200           tf.identity(states_out[-1][1], name='lstm_state_out_h')
 201 
 202         # SSD layers.
 203         with tf.variable_scope('FeatureMaps', reuse=self._reuse_weights):
 204           feature_maps = feature_map_generators.multi_resolution_feature_maps(
 205               feature_map_layout=self._feature_map_layout,
 206               depth_multiplier=(self._depth_multiplier),
 207               min_depth=self._min_depth,
 208               insert_1x1_conv=True,
 209               image_features=image_features)
 210 
 211     return list(feature_maps.values())
