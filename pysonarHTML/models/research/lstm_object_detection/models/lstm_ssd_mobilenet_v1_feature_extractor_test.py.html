  30   def _create_feature_extractor(self,
  31                                 depth_multiplier=1.0,
  32                                 pad_to_multiple=1,
  33                                 is_training=True,
  34                                 use_explicit_padding=False):
  35     """Constructs a new feature extractor.
  36 
  37     Args:
  38       depth_multiplier: A float depth multiplier for feature extractor.
  39       pad_to_multiple: The nearest multiple to zero pad the input height and
  40         width dimensions to.
  41       is_training: A boolean whether the network is in training mode.
  42       use_explicit_padding: A boolean whether to use explicit padding.
  43 
  44     Returns:
  45       An lstm_ssd_meta_arch.LSTMSSDMobileNetV1FeatureExtractor object.
  46     """
  47     min_depth = 32
  48     extractor = (
  49         feature_extractor.LSTMSSDMobileNetV1FeatureExtractor(
  50             is_training,
  51             depth_multiplier,
  52             min_depth,
  53             pad_to_multiple,
  54             self.conv_hyperparams_fn,
  55             use_explicit_padding=use_explicit_padding))
  56     extractor.lstm_state_depth = int(256 * depth_multiplier)
  57     return extractor

  59   def test_feature_extractor_construct_with_expected_params(self):
  60     def conv_hyperparams_fn():
  61       with (slim.arg_scope([slim.conv2d], normalizer_fn=slim.batch_norm) and
  62             slim.arg_scope([slim.batch_norm], decay=0.97, epsilon=1e-3)) as sc:
  63         return sc
  64 
  65     params = {
  66         'is_training': True,
  67         'depth_multiplier': .55,
  68         'min_depth': 9,
  69         'pad_to_multiple': 3,
  70         'conv_hyperparams_fn': conv_hyperparams_fn,
  71         'reuse_weights': False,
  72         'use_explicit_padding': True,
  73         'use_depthwise': False,
  74         'override_base_feature_extractor_hyperparams': True}
  75 
  76     extractor = (
  77         feature_extractor.LSTMSSDMobileNetV1FeatureExtractor(**params))
  78 
  79     self.assertEqual(params['is_training'],
  80                      extractor._is_training)
  81     self.assertEqual(params['depth_multiplier'],
  82                      extractor._depth_multiplier)
  83     self.assertEqual(params['min_depth'],
  84                      extractor._min_depth)
  85     self.assertEqual(params['pad_to_multiple'],
  86                      extractor._pad_to_multiple)
  87     self.assertEqual(params['conv_hyperparams_fn'],
  88                      extractor._conv_hyperparams_fn)
  89     self.assertEqual(params['reuse_weights'],
  90                      extractor._reuse_weights)
  91     self.assertEqual(params['use_explicit_padding'],
  92                      extractor._use_explicit_padding)
  93     self.assertEqual(params['use_depthwise'],
  94                      extractor._use_depthwise)
  95     self.assertEqual(params['override_base_feature_extractor_hyperparams'],

 126   def test_preprocess_returns_correct_value_range(self):
 127     test_image = np.random.rand(5, 128, 128, 3)
 128     extractor = self._create_feature_extractor()
 129     preprocessed_image = extractor.preprocess(test_image)
 130     self.assertTrue(np.all(np.less_equal(np.abs(preprocessed_image), 1.0)))

 132   def test_variables_only_created_in_scope(self):
 133     scope_name = 'MobilenetV1'
 134     g = tf.Graph()
 135     with g.as_default():
 136       preprocessed_inputs = tf.placeholder(tf.float32, (5, 256, 256, 3))
 137       extractor = self._create_feature_extractor()
 138       extractor.extract_features(preprocessed_inputs)
 139       variables = g.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)
 140       find_scope = False
 141       for variable in variables:
 142         if scope_name in variable.name:
 143           find_scope = True
 144           break
 145       self.assertTrue(find_scope)

 147   def test_lstm_non_zero_state(self):
 148     init_state = {
 149         'lstm_state_c': tf.zeros([8, 8, 256]),
 150         'lstm_state_h': tf.zeros([8, 8, 256]),
 151         'lstm_state_step': tf.zeros([1])
 152     }
 153     seq = {'test': tf.random_uniform([3, 1, 1, 1])}
 154     stateful_reader = contrib_training.SequenceQueueingStateSaver(
 155         batch_size=1,
 156         num_unroll=1,
 157         input_length=2,
 158         input_key='',
 159         input_sequences=seq,
 160         input_context={},
 161         initial_states=init_state,
 162         capacity=1)
 163     extractor = self._create_feature_extractor()
 164     image = tf.random_uniform([5, 256, 256, 3])
 165     with tf.variable_scope('zero_state'):
 166       feature_map = extractor.extract_features(
 167           image, stateful_reader.next_batch)
 168     with tf.Session() as sess:
 169       sess.run(tf.global_variables_initializer())
 170       sess.run([stateful_reader.prefetch_op])
 171       _ = sess.run([feature_map])
 172       # Update states with the next batch.
 173       state = sess.run(stateful_reader.next_batch.state('lstm_state_c'))
 174     # State should no longer be zero after update.
 175     self.assertTrue(state.any())
