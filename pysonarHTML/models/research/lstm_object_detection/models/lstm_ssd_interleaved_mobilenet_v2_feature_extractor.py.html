  33 class LSTMSSDInterleavedMobilenetV2FeatureExtractor(
  34     lstm_ssd_meta_arch.LSTMSSDInterleavedFeatureExtractor):
  35   """LSTM-SSD Interleaved Feature Extractor using MobilenetV2 features."""
  36 
  37   def __init__(self,
  38                is_training,
  39                depth_multiplier,
  40                min_depth,
  41                pad_to_multiple,
  42                conv_hyperparams_fn,
  43                reuse_weights=None,
  44                use_explicit_padding=False,
  45                use_depthwise=True,
  46                override_base_feature_extractor_hyperparams=False):
  47     """Interleaved Feature Extractor for LSTD Models with MobileNet v2.
  48 
  49     Args:
  50       is_training: whether the network is in training mode.
  51       depth_multiplier: float depth multiplier for feature extractor.
  52       min_depth: minimum feature extractor depth.
  53       pad_to_multiple: the nearest multiple to zero pad the input height and
  54         width dimensions to.
  55       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  56         and separable_conv2d ops in the layers that are added on top of the
  57         base feature extractor.
  58       reuse_weights: Whether to reuse variables. Default is None.
  59       use_explicit_padding: Whether to use explicit padding when extracting
  60         features. Default is False.
  61       use_depthwise: Whether to use depthwise convolutions. Default is True.
  62       override_base_feature_extractor_hyperparams: Whether to override
  63         hyperparameters of the base feature extractor with the one from
  64         `conv_hyperparams_fn`.
  65     """
  66     super(LSTMSSDInterleavedMobilenetV2FeatureExtractor, self).__init__(
  67         is_training=is_training,
  68         depth_multiplier=depth_multiplier,
  69         min_depth=min_depth,
  70         pad_to_multiple=pad_to_multiple,
  71         conv_hyperparams_fn=conv_hyperparams_fn,
  72         reuse_weights=reuse_weights,
  73         use_explicit_padding=use_explicit_padding,
  74         use_depthwise=use_depthwise,
  75         override_base_feature_extractor_hyperparams=
  76         override_base_feature_extractor_hyperparams)
  77     # RANDOM_SKIP_SMALL means the training policy is random and the small model
  78     # does not update state during training.
  79     if self._is_training:
  80       self._interleave_method = 'RANDOM_SKIP_SMALL'
  81     else:
  82       self._interleave_method = 'SKIP9'
  83 
  84     self._flatten_state = False
  85     self._scale_state = False
  86     self._clip_state = True
  87     self._pre_bottleneck = True
  88     self._feature_map_layout = {
  89         'from_layer': ['layer_19', '', '', '', ''],
  90         'layer_depth': [-1, 256, 256, 256, 256],
  91         'use_depthwise': self._use_depthwise,
  92         'use_explicit_padding': self._use_explicit_padding,
  93     }
  94     self._low_res = True
  95     self._base_network_scope = 'MobilenetV2'
  96 
  97   def extract_base_features_large(self, preprocessed_inputs):
  98     """Extract the large base model features.
  99 
 100     Variables are created under the scope of <scope>/MobilenetV2_1/
 101 
 102     Args:
 103       preprocessed_inputs: preprocessed input images of shape:
 104         [batch, width, height, depth].
 105 
 106     Returns:
 107       net: the last feature map created from the base feature extractor.
 108       end_points: a dictionary of feature maps created.
 109     """
 110     scope_name = self._base_network_scope + '_1'
 111     with tf.variable_scope(scope_name, reuse=self._reuse_weights) as base_scope:
 112       net, end_points = mobilenet_v2.mobilenet_base(
 113           preprocessed_inputs,
 114           depth_multiplier=self._depth_multipliers[0],
 115           conv_defs=mobilenet_defs.mobilenet_v2_lite_def(
 116               is_quantized=self._is_quantized),
 117           use_explicit_padding=self._use_explicit_padding,
 118           scope=base_scope)
 119       return net, end_points
 120 
 121   def extract_base_features_small(self, preprocessed_inputs):
 122     """Extract the small base model features.
 123 
 124     Variables are created under the scope of <scope>/MobilenetV2_2/
 125 
 126     Args:
 127       preprocessed_inputs: preprocessed input images of shape:
 128         [batch, width, height, depth].
 129 
 130     Returns:
 131       net: the last feature map created from the base feature extractor.
 132       end_points: a dictionary of feature maps created.
 133     """
 134     scope_name = self._base_network_scope + '_2'
 135     with tf.variable_scope(scope_name, reuse=self._reuse_weights) as base_scope:
 136       if self._low_res:
 137         height_small = preprocessed_inputs.get_shape().as_list()[1] // 2
 138         width_small = preprocessed_inputs.get_shape().as_list()[2] // 2
 139         inputs_small = tf.image.resize_images(preprocessed_inputs,
 140                                               [height_small, width_small])
 141         # Create end point handle for tflite deployment.
 142         with tf.name_scope(None):
 143           inputs_small = tf.identity(
 144               inputs_small, name='normalized_input_image_tensor_small')
 145       else:
 146         inputs_small = preprocessed_inputs
 147       net, end_points = mobilenet_v2.mobilenet_base(
 148           inputs_small,
 149           depth_multiplier=self._depth_multipliers[1],
 150           conv_defs=mobilenet_defs.mobilenet_v2_lite_def(
 151               is_quantized=self._is_quantized, low_res=self._low_res),
 152           use_explicit_padding=self._use_explicit_padding,
 153           scope=base_scope)
 154       return net, end_points
 155 
 156   def create_lstm_cell(self, batch_size, output_size, state_saver, state_name,
 157                        dtype=tf.float32):
 158     """Create the LSTM cell, and initialize state if necessary.
 159 
 160     Args:
 161       batch_size: input batch size.
 162       output_size: output size of the lstm cell, [width, height].
 163       state_saver: a state saver object with methods `state` and `save_state`.
 164       state_name: string, the name to use with the state_saver.
 165       dtype: dtype to initialize lstm state.
 166 
 167     Returns:
 168       lstm_cell: the lstm cell unit.
 169       init_state: initial state representations.
 170       step: the step
 171     """
 172     lstm_cell = lstm_cells.GroupedConvLSTMCell(
 173         filter_size=(3, 3),
 174         output_size=output_size,
 175         num_units=max(self._min_depth, self._lstm_state_depth),
 176         is_training=self._is_training,
 177         activation=tf.nn.relu6,
 178         flatten_state=self._flatten_state,
 179         scale_state=self._scale_state,
 180         clip_state=self._clip_state,
 181         output_bottleneck=True,
 182         pre_bottleneck=self._pre_bottleneck,
 183         is_quantized=self._is_quantized,
 184         visualize_gates=False)
 185 
 186     if state_saver is None:
 187       init_state = lstm_cell.init_state('lstm_state', batch_size, dtype)
 188       step = None
 189     else:
 190       step = state_saver.state(state_name + '_step')
 191       c = state_saver.state(state_name + '_c')
 192       h = state_saver.state(state_name + '_h')
 193       c.set_shape([batch_size] + c.get_shape().as_list()[1:])
 194       h.set_shape([batch_size] + h.get_shape().as_list()[1:])
 195       init_state = (c, h)
 196     return lstm_cell, init_state, step

 198   def extract_features(self, preprocessed_inputs, state_saver=None,
 199                        state_name='lstm_state', unroll_length=10, scope=None):
 200     """Extract features from preprocessed inputs.
 201 
 202     The features include the base network features, lstm features and SSD
 203     features, organized in the following name scope:
 204 
 205     <scope>/MobilenetV2_1/...
 206     <scope>/MobilenetV2_2/...
 207     <scope>/LSTM/...
 208     <scope>/FeatureMap/...
 209 
 210     Args:
 211       preprocessed_inputs: a [batch, height, width, channels] float tensor
 212         representing a batch of consecutive frames from video clips.
 213       state_saver: A state saver object with methods `state` and `save_state`.
 214       state_name: Python string, the name to use with the state_saver.
 215       unroll_length: number of steps to unroll the lstm.
 216       scope: Scope for the base network of the feature extractor.
 217 
 218     Returns:
 219       feature_maps: a list of tensors where the ith tensor has shape
 220         [batch, height_i, width_i, depth_i]
 221     Raises:
 222       ValueError: if interleave_method not recognized or large and small base
 223         network output feature maps of different sizes.
 224     """
 225     preprocessed_inputs = shape_utils.check_min_image_dim(
 226         33, preprocessed_inputs)
 227     preprocessed_inputs = ops.pad_to_multiple(
 228         preprocessed_inputs, self._pad_to_multiple)
 229     batch_size = preprocessed_inputs.shape[0].value // unroll_length
 230     batch_axis = 0
 231     nets = []
 232 
 233     # Batch processing of mobilenet features.
 234     with slim.arg_scope(mobilenet_v2.training_scope(
 235         is_training=self._is_training,
 236         bn_decay=0.9997)), \
 237         slim.arg_scope([mobilenet.depth_multiplier],
 238                        min_depth=self._min_depth, divisible_by=8):
 239       # Big model.
 240       net, _ = self.extract_base_features_large(preprocessed_inputs)
 241       nets.append(net)
 242       large_base_feature_shape = net.shape
 243 
 244       # Small models
 245       net, _ = self.extract_base_features_small(preprocessed_inputs)
 246       nets.append(net)
 247       small_base_feature_shape = net.shape
 248       if not (large_base_feature_shape[1] == small_base_feature_shape[1] and
 249               large_base_feature_shape[2] == small_base_feature_shape[2]):
 250         raise ValueError('Large and Small base network feature map dimension '
 251                          'not equal!')
 252 
 253     with slim.arg_scope(self._conv_hyperparams_fn()):
 254       with tf.variable_scope('LSTM', reuse=self._reuse_weights):
 255         output_size = (large_base_feature_shape[1], large_base_feature_shape[2])
 256         lstm_cell, init_state, step = self.create_lstm_cell(
 257             batch_size, output_size, state_saver, state_name,
 258             dtype=preprocessed_inputs.dtype)
 259 
 260         nets_seq = [
 261             tf.split(net, unroll_length, axis=batch_axis) for net in nets
 262         ]
 263 
 264         net_seq, states_out = rnn_decoder.multi_input_rnn_decoder(
 265             nets_seq,
 266             init_state,
 267             lstm_cell,
 268             step,
 269             selection_strategy=self._interleave_method,
 270             is_training=self._is_training,
 271             is_quantized=self._is_quantized,
 272             pre_bottleneck=self._pre_bottleneck,
 273             flatten_state=self._flatten_state,
 274             scope=None)
 275         self._states_out = states_out
 276 
 277       image_features = {}
 278       if state_saver is not None:
 279         self._step = state_saver.state(state_name + '_step')
 280         batcher_ops = [
 281             state_saver.save_state(state_name + '_c', states_out[-1][0]),
 282             state_saver.save_state(state_name + '_h', states_out[-1][1]),
 283             state_saver.save_state(state_name + '_step', self._step + 1)]
 284         with tf_ops.control_dependencies(batcher_ops):
 285           image_features['layer_19'] = tf.concat(net_seq, 0)
 286       else:
 287         image_features['layer_19'] = tf.concat(net_seq, 0)
 288 
 289       # SSD layers.
 290       with tf.variable_scope('FeatureMap'):
 291         feature_maps = feature_map_generators.multi_resolution_feature_maps(
 292             feature_map_layout=self._feature_map_layout,
 293             depth_multiplier=self._depth_multiplier,
 294             min_depth=self._min_depth,
 295             insert_1x1_conv=True,
 296             image_features=image_features,
 297             pool_residual=True)
 298     return list(feature_maps.values())
