  43 def build(model_config, lstm_config, is_training):
  44   """Builds a DetectionModel based on the model config.
  45 
  46   Args:
  47     model_config: A model.proto object containing the config for the desired
  48       DetectionModel.
  49     lstm_config: LstmModel config proto that specifies LSTM train/eval configs.
  50     is_training: True if this model is being built for training purposes.
  51 
  52   Returns:
  53     DetectionModel based on the config.
  54 
  55   Raises:
  56     ValueError: On invalid meta architecture or model.
  57   """
  58   return _build_lstm_model(model_config.ssd, lstm_config, is_training)

  61 def _build_lstm_feature_extractor(feature_extractor_config,
  62                                   is_training,
  63                                   lstm_config,
  64                                   reuse_weights=None):
  65   """Builds a ssd_meta_arch.SSDFeatureExtractor based on config.
  66 
  67   Args:
  68     feature_extractor_config: A SSDFeatureExtractor proto config from ssd.proto.
  69     is_training: True if this feature extractor is being built for training.
  70     lstm_config: LSTM-SSD specific configs.
  71     reuse_weights: If the feature extractor should reuse weights.
  72 
  73   Returns:
  74     ssd_meta_arch.SSDFeatureExtractor based on config.
  75 
  76   Raises:
  77     ValueError: On invalid feature extractor type.
  78   """
  79 
  80   feature_type = feature_extractor_config.type
  81   depth_multiplier = feature_extractor_config.depth_multiplier
  82   min_depth = feature_extractor_config.min_depth
  83   pad_to_multiple = feature_extractor_config.pad_to_multiple
  84   use_explicit_padding = feature_extractor_config.use_explicit_padding
  85   use_depthwise = feature_extractor_config.use_depthwise
  86   conv_hyperparams = hyperparams_builder.build(
  87       feature_extractor_config.conv_hyperparams, is_training)
  88   override_base_feature_extractor_hyperparams = (
  89       feature_extractor_config.override_base_feature_extractor_hyperparams)
  90 
  91   if feature_type not in SSD_FEATURE_EXTRACTOR_CLASS_MAP:
  92     raise ValueError('Unknown ssd feature_extractor: {}'.format(feature_type))
  93 
  94   feature_extractor_class = SSD_FEATURE_EXTRACTOR_CLASS_MAP[feature_type]
  95   feature_extractor = feature_extractor_class(
  96       is_training, depth_multiplier, min_depth, pad_to_multiple,
  97       conv_hyperparams, reuse_weights, use_explicit_padding, use_depthwise,
  98       override_base_feature_extractor_hyperparams)
  99 
 100   # Extra configs for LSTM-SSD.
 101   feature_extractor.lstm_state_depth = lstm_config.lstm_state_depth
 102   feature_extractor.flatten_state = lstm_config.flatten_state
 103   feature_extractor.clip_state = lstm_config.clip_state
 104   feature_extractor.scale_state = lstm_config.scale_state
 105   feature_extractor.is_quantized = lstm_config.is_quantized
 106   feature_extractor.low_res = lstm_config.low_res
 107   # Extra configs for interleaved LSTM-SSD.
 108   if 'interleaved' in feature_extractor_config.type:
 109     feature_extractor.pre_bottleneck = lstm_config.pre_bottleneck
 110     feature_extractor.depth_multipliers = lstm_config.depth_multipliers
 111     if is_training:
 112       feature_extractor.interleave_method = lstm_config.train_interleave_method
 113     else:
 114       feature_extractor.interleave_method = lstm_config.eval_interleave_method
 115   return feature_extractor

 118 def _build_lstm_model(ssd_config, lstm_config, is_training):
 119   """Builds an LSTM detection model based on the model config.
 120 
 121   Args:
 122     ssd_config: A ssd.proto object containing the config for the desired
 123       LSTMSSDMetaArch.
 124     lstm_config: LstmModel config proto that specifies LSTM train/eval configs.
 125     is_training: True if this model is being built for training purposes.
 126 
 127   Returns:
 128     LSTMSSDMetaArch based on the config.
 129   Raises:
 130     ValueError: If ssd_config.type is not recognized (i.e. not registered in
 131       model_class_map), or if lstm_config.interleave_strategy is not recognized.
 132     ValueError: If unroll_length is not specified in the config file.
 133   """
 134   feature_extractor = _build_lstm_feature_extractor(
 135       ssd_config.feature_extractor, is_training, lstm_config)
 136 
 137   box_coder = box_coder_builder.build(ssd_config.box_coder)
 138   matcher = matcher_builder.build(ssd_config.matcher)
 139   region_similarity_calculator = sim_calc.build(
 140       ssd_config.similarity_calculator)
 141 
 142   num_classes = ssd_config.num_classes
 143   ssd_box_predictor = box_predictor_builder.build(hyperparams_builder.build,
 144                                                   ssd_config.box_predictor,
 145                                                   is_training, num_classes)
 146   anchor_generator = anchor_generator_builder.build(ssd_config.anchor_generator)
 147   image_resizer_fn = image_resizer_builder.build(ssd_config.image_resizer)
 148   non_max_suppression_fn, score_conversion_fn = post_processing_builder.build(
 149       ssd_config.post_processing)
 150   (classification_loss, localization_loss, classification_weight,
 151    localization_weight, miner, _, _) = losses_builder.build(ssd_config.loss)
 152 
 153   normalize_loss_by_num_matches = ssd_config.normalize_loss_by_num_matches
 154   encode_background_as_zeros = ssd_config.encode_background_as_zeros
 155   negative_class_weight = ssd_config.negative_class_weight
 156 
 157   # Extra configs for lstm unroll length.
 158   unroll_length = None
 159   if 'lstm' in ssd_config.feature_extractor.type:
 160     if is_training:
 161       unroll_length = lstm_config.train_unroll_length
 162     else:
 163       unroll_length = lstm_config.eval_unroll_length
 164   if unroll_length is None:
 165     raise ValueError('No unroll length found in the config file')
 166 
 167   target_assigner_instance = target_assigner.TargetAssigner(
 168       region_similarity_calculator,
 169       matcher,
 170       box_coder,
 171       negative_class_weight=negative_class_weight)
 172 
 173   lstm_model = lstm_ssd_meta_arch.LSTMSSDMetaArch(
 174       is_training=is_training,
 175       anchor_generator=anchor_generator,
 176       box_predictor=ssd_box_predictor,
 177       box_coder=box_coder,
 178       feature_extractor=feature_extractor,
 179       encode_background_as_zeros=encode_background_as_zeros,
 180       image_resizer_fn=image_resizer_fn,
 181       non_max_suppression_fn=non_max_suppression_fn,
 182       score_conversion_fn=score_conversion_fn,
 183       classification_loss=classification_loss,
 184       localization_loss=localization_loss,
 185       classification_loss_weight=classification_weight,
 186       localization_loss_weight=localization_weight,
 187       normalize_loss_by_num_matches=normalize_loss_by_num_matches,
 188       hard_example_miner=miner,
 189       unroll_length=unroll_length,
 190       target_assigner_instance=target_assigner_instance)
 191 
 192   return lstm_model
