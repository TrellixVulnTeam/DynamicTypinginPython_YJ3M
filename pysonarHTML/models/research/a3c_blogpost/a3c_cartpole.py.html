  35 class ActorCriticModel(keras.Model):
  36   def __init__(self, state_size, action_size):
  37     super(ActorCriticModel, self).__init__()
  38     self.state_size = state_size
  39     self.action_size = action_size
  40     self.dense1 = layers.Dense(100, activation='relu')
  41     self.policy_logits = layers.Dense(action_size)
  42     self.dense2 = layers.Dense(100, activation='relu')
  43     self.values = layers.Dense(1)
  44 
  45   def call(self, inputs):
  46     # Forward pass
  47     x = self.dense1(inputs)
  48     logits = self.policy_logits(x)
  49     v1 = self.dense2(inputs)
  50     values = self.values(v1)
  51     return logits, values

  53 def record(episode,
  54            episode_reward,
  55            worker_idx,
  56            global_ep_reward,
  57            result_queue,
  58            total_loss,
  59            num_steps):
  60   """Helper function to store score and print statistics.
  61 
  62   Arguments:
  63     episode: Current episode
  64     episode_reward: Reward accumulated over the current episode
  65     worker_idx: Which thread (worker)
  66     global_ep_reward: The moving average of the global reward
  67     result_queue: Queue storing the moving average of the scores
  68     total_loss: The total loss accumualted over the current episode
  69     num_steps: The number of steps the episode took to complete
  70   """
  71   if global_ep_reward == 0:
  72     global_ep_reward = episode_reward
  73   else:
  74     global_ep_reward = global_ep_reward * 0.99 + episode_reward * 0.01
  75   print(
  76       f"Episode: {episode} | "
  77       f"Moving Average Reward: {int(global_ep_reward)} | "
  78       f"Episode Reward: {int(episode_reward)} | "
  79       f"Loss: {int(total_loss / float(num_steps) * 1000) / 1000} | "
  80       f"Steps: {num_steps} | "
  81       f"Worker: {worker_idx}"
  82   )
  83   result_queue.put(global_ep_reward)
  84   return global_ep_reward

  87 class RandomAgent:
  88   """Random Agent that will play the specified game
  89 
  90     Arguments:
  91       env_name: Name of the environment to be played
  92       max_eps: Maximum number of episodes to run agent for.
  93   """
  94   def __init__(self, env_name, max_eps):
  95     self.env = gym.make(env_name)
  96     self.max_episodes = max_eps
  97     self.global_moving_average_reward = 0
  98     self.res_queue = Queue()
  99 
 100   def run(self):
 101     reward_avg = 0
 102     for episode in range(self.max_episodes):
 103       done = False
 104       self.env.reset()
 105       reward_sum = 0.0
 106       steps = 0
 107       while not done:
 108         # Sample randomly from the action space and step
 109         _, reward, done, _ = self.env.step(self.env.action_space.sample())
 110         steps += 1
 111         reward_sum += reward
 112       # Record statistics
 113       self.global_moving_average_reward = record(episode,
 114                                                  reward_sum,
 115                                                  0,
 116                                                  self.global_moving_average_reward,
 117                                                  self.res_queue, 0, steps)
 118 
 119       reward_avg += reward_sum
 120     final_avg = reward_avg / float(self.max_episodes)
 121     print("Average score across {} episodes: {}".format(self.max_episodes, final_avg))
 122     return final_avg

 221 class Worker(threading.Thread):
 222   # Set up global variables across different threads
 223   global_episode = 0
 224   # Moving average reward
 225   global_moving_average_reward = 0
 226   best_score = 0
 227   save_lock = threading.Lock()
 228 
 229   def __init__(self,
 230                state_size,
 231                action_size,
 232                global_model,
 233                opt,
 234                result_queue,
 235                idx,
 236                game_name='CartPole-v0',
 237                save_dir='/tmp'):
 238     super(Worker, self).__init__()
 239     self.state_size = state_size
 240     self.action_size = action_size
 241     self.result_queue = result_queue
 242     self.global_model = global_model
 243     self.opt = opt
 244     self.local_model = ActorCriticModel(self.state_size, self.action_size)
 245     self.worker_idx = idx
 246     self.game_name = game_name
 247     self.env = gym.make(self.game_name).unwrapped
 248     self.save_dir = save_dir
 249     self.ep_loss = 0.0
 250 
 251   def run(self):
 252     total_step = 1
 253     mem = Memory()
 254     while Worker.global_episode < args.max_eps:
 255       current_state = self.env.reset()
 256       mem.clear()
 257       ep_reward = 0.
 258       ep_steps = 0
 259       self.ep_loss = 0
 260 
 261       time_count = 0
 262       done = False
 263       while not done:
 264         logits, _ = self.local_model(
 265             tf.convert_to_tensor(current_state[None, :],
 266                                  dtype=tf.float32))
 267         probs = tf.nn.softmax(logits)
 268 
 269         action = np.random.choice(self.action_size, p=probs.numpy()[0])
 270         new_state, reward, done, _ = self.env.step(action)
 271         if done:
 272           reward = -1
 273         ep_reward += reward
 274         mem.store(current_state, action, reward)
 275 
 276         if time_count == args.update_freq or done:
 277           # Calculate gradient wrt to local model. We do so by tracking the
 278           # variables involved in computing the loss by using tf.GradientTape
 279           with tf.GradientTape() as tape:
 280             total_loss = self.compute_loss(done,
 281                                            new_state,
 282                                            mem,
 283                                            args.gamma)
 284           self.ep_loss += total_loss
 285           # Calculate local gradients
 286           grads = tape.gradient(total_loss, self.local_model.trainable_weights)
 287           # Push local gradients to global model
 288           self.opt.apply_gradients(zip(grads,
 289                                        self.global_model.trainable_weights))
 290           # Update local model with new weights
 291           self.local_model.set_weights(self.global_model.get_weights())
 292 
 293           mem.clear()
 294           time_count = 0
 295 
 296           if done:  # done and print information
 297             Worker.global_moving_average_reward = \
 298               record(Worker.global_episode, ep_reward, self.worker_idx,
 299                      Worker.global_moving_average_reward, self.result_queue,
 300                      self.ep_loss, ep_steps)
 301             # We must use a lock to save our model and to print to prevent data races.
 302             if ep_reward > Worker.best_score:
 303               with Worker.save_lock:
 304                 print("Saving best model to {}, "
 305                       "episode score: {}".format(self.save_dir, ep_reward))
 306                 self.global_model.save_weights(
 307                     os.path.join(self.save_dir,
 308                                  'model_{}.h5'.format(self.game_name))
 309                 )
 310                 Worker.best_score = ep_reward
 311             Worker.global_episode += 1
 312         ep_steps += 1
 313 
 314         time_count += 1
 315         current_state = new_state
 316         total_step += 1
 317     self.result_queue.put(None)
 318 
 319   def compute_loss(self,
 320                    done,
 321                    new_state,
 322                    memory,
 323                    gamma=0.99):
 324     if done:
 325       reward_sum = 0.  # terminal
 326     else:
 327       reward_sum = self.local_model(
 328           tf.convert_to_tensor(new_state[None, :],
 329                                dtype=tf.float32))[-1].numpy()[0]
 330 
 331     # Get discounted rewards
 332     discounted_rewards = []
 333     for reward in memory.rewards[::-1]:  # reverse buffer r
 334       reward_sum = reward + gamma * reward_sum
 335       discounted_rewards.append(reward_sum)
 336     discounted_rewards.reverse()
 337 
 338     logits, values = self.local_model(
 339         tf.convert_to_tensor(np.vstack(memory.states),
 340                              dtype=tf.float32))
 341     # Get our advantages
 342     advantage = tf.convert_to_tensor(np.array(discounted_rewards)[:, None],
 343                             dtype=tf.float32) - values
 344     # Value loss
 345     value_loss = advantage ** 2
 346 
 347     # Calculate our policy loss
 348     policy = tf.nn.softmax(logits)
 349     entropy = tf.nn.softmax_cross_entropy_with_logits(labels=policy, logits=logits)
 350 
 351     policy_loss = tf.nn.sparse_softmax_cross_entropy_with_logits(labels=memory.actions,
 352                                                                  logits=logits)
 353     policy_loss *= tf.stop_gradient(advantage)
 354     policy_loss -= 0.01 * entropy
 355     total_loss = tf.reduce_mean((0.5 * value_loss + policy_loss))
 356     return total_loss
