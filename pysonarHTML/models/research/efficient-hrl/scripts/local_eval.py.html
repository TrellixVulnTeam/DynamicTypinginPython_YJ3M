  16 """Script to run run_eval.py locally.
  17 """
  18 
  19 from __future__ import absolute_import
  20 from __future__ import division
  21 from __future__ import print_function
  22 import os
  23 from subprocess import call
  24 import sys
  25 
  26 CONFIGS_PATH = 'configs'
  27 CONTEXT_CONFIGS_PATH = 'context/configs'
  28 
  29 def main():
  30   bb = './'
  31   base_num_args = 6
  32   if len(sys.argv) < base_num_args:
  33     print(
  34         "usage: python %s <exp_name> <context_setting_gin> <context_gin> "
  35         "<agent_gin> <suite> [params...]"
  36         % sys.argv[0])
  37     sys.exit(0)
  38   exp = sys.argv[1]
  39   context_setting = sys.argv[2]
  40   context = sys.argv[3]
  41   agent = sys.argv[4]
  42   assert sys.argv[5] in ["suite"], "args[5] must be `suite'"
  43   suite = ""
  44   binary = "python {bb}/run_eval{suite}.py ".format(bb=bb, suite=suite)
  45 
  46   h = os.environ["HOME"]
  47   ucp = CONFIGS_PATH
  48   ccp = CONTEXT_CONFIGS_PATH
  49   extra = ''
  50   command_str = ("{binary} "
  51                  "--logtostderr "
  52                  "--checkpoint_dir={h}/tmp/{context_setting}/{context}/{agent}/{exp}/train "
  53                  "--eval_dir={h}/tmp/{context_setting}/{context}/{agent}/{exp}/eval "
  54                  "--config_file={ucp}/{agent}.gin "
  55                  "--config_file={ucp}/eval_{extra}uvf.gin "
  56                  "--config_file={ccp}/{context_setting}.gin "
  57                  "--config_file={ccp}/{context}.gin ").format(
  58                      h=h,
  59                      ucp=ucp,
  60                      ccp=ccp,
  61                      context_setting=context_setting,
  62                      context=context,
  63                      agent=agent,
  64                      extra=extra,
  65                      suite=suite,
  66                      exp=exp,
  67                      binary=binary)
  68   for extra_arg in sys.argv[base_num_args:]:
  69     command_str += "--params='%s' " % extra_arg
  70 
  71   print(command_str)
  72   call(command_str, shell=True)
  73 
  74 
  75 if __name__ == "__main__":
  76   main()
