  35 def get_goal_sample_fn(env_name):
  36   if env_name == 'AntMaze':
  37     # NOTE: When evaluating (i.e. the metrics shown in the paper,
  38     # we use the commented out goal sampling function.  The uncommented
  39     # one is only used for training.
  40     #return lambda: np.array([0., 16.])
  41     return lambda: np.random.uniform((-4, -4), (20, 20))
  42   elif env_name == 'AntPush':
  43     return lambda: np.array([0., 19.])
  44   elif env_name == 'AntFall':
  45     return lambda: np.array([0., 27., 4.5])
  46   else:
  47     assert False, 'Unknown env'
  48 
  49 
  50 def get_reward_fn(env_name):
  51   if env_name == 'AntMaze':
  52     return lambda obs, goal: -np.sum(np.square(obs[:2] - goal)) ** 0.5
  53   elif env_name == 'AntPush':
  54     return lambda obs, goal: -np.sum(np.square(obs[:2] - goal)) ** 0.5
  55   elif env_name == 'AntFall':
  56     return lambda obs, goal: -np.sum(np.square(obs[:3] - goal)) ** 0.5
  57   else:
  58     assert False, 'Unknown env'
  59 
  60 
  61 def success_fn(last_reward):
  62   return last_reward > -5.0
  63 
  64 
  65 class EnvWithGoal(object):
  66 
  67   def __init__(self, base_env, env_name):
  68     self.base_env = base_env
  69     self.goal_sample_fn = get_goal_sample_fn(env_name)
  70     self.reward_fn = get_reward_fn(env_name)
  71     self.goal = None
  72 
  73   def reset(self):
  74     obs = self.base_env.reset()
  75     self.goal = self.goal_sample_fn()
  76     return np.concatenate([obs, self.goal])
  77 
  78   def step(self, a):
  79     obs, _, done, info = self.base_env.step(a)
  80     reward = self.reward_fn(obs, self.goal)
  81     return np.concatenate([obs, self.goal]), reward, done, info
  82 
  83   @property
  84   def action_space(self):
  85     return self.base_env.action_space
  86 
  87 
  88 def run_environment(env_name, episode_length, num_episodes):
  89   env = EnvWithGoal(
  90       create_maze_env.create_maze_env(env_name).gym,
  91       env_name)
  92 
  93   def action_fn(obs):
  94     action_space = env.action_space
  95     action_space_mean = (action_space.low + action_space.high) / 2.0
  96     action_space_magn = (action_space.high - action_space.low) / 2.0
  97     random_action = (action_space_mean +
  98                      action_space_magn *
  99                      np.random.uniform(low=-1.0, high=1.0,
 100                                        size=action_space.shape))
 101     return random_action
 102 
 103   rewards = []
 104   successes = []
 105   for ep in range(num_episodes):
 106     rewards.append(0.0)
 107     successes.append(False)
 108     obs = env.reset()
 109     for _ in range(episode_length):
 110       obs, reward, done, _ = env.step(action_fn(obs))
 111       rewards[-1] += reward
 112       successes[-1] = success_fn(reward)
 113       if done:
 114         break
 115     logging.info('Episode %d reward: %.2f, Success: %d', ep + 1, rewards[-1], successes[-1])
 116 
 117   logging.info('Average Reward over %d episodes: %.2f',
 118                num_episodes, np.mean(rewards))
 119   logging.info('Average Success over %d episodes: %.2f',
 120                num_episodes, np.mean(successes))
