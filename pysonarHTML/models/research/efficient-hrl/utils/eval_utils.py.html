  30 def evaluate_checkpoint_repeatedly(checkpoint_dir,
  31                                    evaluate_checkpoint_fn,
  32                                    eval_interval_secs=600,
  33                                    max_number_of_evaluations=None,
  34                                    checkpoint_timeout=None,
  35                                    timeout_fn=None):
  36   """Evaluates a checkpointed model at a set interval."""
  37   if max_number_of_evaluations is not None and max_number_of_evaluations <= 0:
  38     raise ValueError(
  39         '`max_number_of_evaluations` must be either None or a positive number.')
  40 
  41   number_of_evaluations = 0
  42   for checkpoint_path in tf.contrib.training.checkpoints_iterator(
  43       checkpoint_dir,
  44       min_interval_secs=eval_interval_secs,
  45       timeout=checkpoint_timeout,
  46       timeout_fn=timeout_fn):
  47     retries = 3
  48     for _ in range(retries):
  49       try:
  50         should_stop = evaluate_checkpoint_fn(checkpoint_path)
  51         break
  52       except tf.errors.DataLossError as e:

  55             'can happen when reading a checkpoint before it is fully written. '
  56             'Retrying...'

  61 def compute_model_loss(sess, model_rollout_fn, states, actions):
  62   """Computes model loss."""
  63   preds, losses = [], []
  64   preds.append(states[0])
  65   losses.append(0)
  66   for state, action in zip(states[1:], actions[1:]):
  67     pred = model_rollout_fn(sess, preds[-1], action)
  68     loss = np.sqrt(np.sum((state - pred) ** 2))
  69     preds.append(pred)
  70     losses.append(loss)
  71   return preds, losses

  74 def compute_average_reward(sess, env_base, step_fn, gamma, num_steps,
  75                            num_episodes):
  76   """Computes the discounted reward for a given number of steps.
  77 
  78   Args:
  79     sess: The tensorflow session.
  80     env_base: A python environment.
  81     step_fn: A function that takes in `sess` and returns a list of
  82       [state, action, reward, discount, transition_type] values.
  83     gamma: discounting factor to apply to the reward.
  84     num_steps: number of steps to compute the reward over.
  85     num_episodes: number of episodes to average the reward over.
  86   Returns:
  87     average_reward: a scalar of discounted reward.
  88     last_reward: last reward received.
  89   """
  90   average_reward = 0
  91   average_last_reward = 0
  92   average_meta_reward = 0
  93   average_last_meta_reward = 0
  94   average_success = 0.
  95   states, actions = None, None
  96   for i in range(num_episodes):
  97     env_base.end_episode()
  98     env_base.begin_episode()
  99     (reward, last_reward, meta_reward, last_meta_reward,
 100      states, actions) = compute_reward(
 101         sess, step_fn, gamma, num_steps)
 102     s_reward = last_meta_reward  # Navigation
 103     success = (s_reward > -5.0)  # When using diff=False
 104     logging.info('Episode = %d, reward = %s, meta_reward = %f, '
 105                  'last_reward = %s, last meta_reward = %f, success = %s',
 106                  i, reward, meta_reward, last_reward, last_meta_reward,
 107                  success)
 108     average_reward += reward
 109     average_last_reward += last_reward
 110     average_meta_reward += meta_reward
 111     average_last_meta_reward += last_meta_reward
 112     average_success += success
 113   average_reward /= num_episodes
 114   average_last_reward /= num_episodes
 115   average_meta_reward /= num_episodes
 116   average_last_meta_reward /= num_episodes
 117   average_success /= num_episodes
 118   return (average_reward, average_last_reward,
 119           average_meta_reward, average_last_meta_reward,
 120           average_success,
 121           states, actions)

 124 def compute_reward(sess, step_fn, gamma, num_steps):
 125   """Computes the discounted reward for a given number of steps.
 126 
 127   Args:
 128     sess: The tensorflow session.
 129     step_fn: A function that takes in `sess` and returns a list of
 130       [state, action, reward, discount, transition_type] values.
 131     gamma: discounting factor to apply to the reward.
 132     num_steps: number of steps to compute the reward over.
 133   Returns:
 134     reward: cumulative discounted reward.
 135     last_reward: reward received at final step.
 136   """
 137 
 138   total_reward = 0
 139   total_meta_reward = 0
 140   gamma_step = 1
 141   states = []
 142   actions = []
 143   for _ in range(num_steps):
 144     state, action, transition_type, reward, meta_reward, discount, _, _ = step_fn(sess)
 145     total_reward += reward * gamma_step * discount
 146     total_meta_reward += meta_reward * gamma_step * discount
 147     gamma_step *= gamma
 148     states.append(state)
 149     actions.append(action)
 150   return (total_reward, reward, total_meta_reward, meta_reward,
 151           states, actions)
