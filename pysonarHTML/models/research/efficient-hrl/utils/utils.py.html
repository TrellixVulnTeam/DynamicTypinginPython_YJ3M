  34 def get_contextual_env_base(env_base, begin_ops=None, end_ops=None):
  35   """Wrap env_base with additional tf ops."""
  36   # pylint: disable=protected-access
  37   def init(self_, env_base):
  38     self_._env_base = env_base
  39     attribute_list = ["_render_mode", "_gym_env"]
  40     for attribute in attribute_list:
  41       if hasattr(env_base, attribute):
  42         setattr(self_, attribute, getattr(env_base, attribute))
  43     if hasattr(env_base, "physics"):
  44       self_._physics = env_base.physics
  45     elif hasattr(env_base, "gym"):
  46       class Physics(object):
  47         def render(self, *args, **kwargs):
  48           return env_base.gym.render("rgb_array")
  49       physics = Physics()
  50       self_._physics = physics
  51       self_.physics = physics
  52   def set_sess(self_, sess):
  53     self_._sess = sess
  54     if hasattr(self_._env_base, "set_sess"):
  55       self_._env_base.set_sess(sess)
  56   def begin_episode(self_):
  57     self_._env_base.reset()
  58     if begin_ops is not None:
  59       self_._sess.run(begin_ops)
  60   def end_episode(self_):
  61     self_._env_base.reset()
  62     if end_ops is not None:
  63       self_._sess.run(end_ops)
  64   return type("ContextualEnvBase", (env_base.__class__,), dict(
  65       __init__=init,
  66       set_sess=set_sess,
  67       begin_episode=begin_episode,
  68       end_episode=end_episode,
  69   ))(env_base)
  70   # pylint: enable=protected-access

  73 def merge_specs(specs_):
  74   """Merge TensorSpecs.
  75 
  76   Args:
  77     specs_: List of TensorSpecs to be merged.
  78   Returns:
  79     a TensorSpec: a merged TensorSpec.
  80   """
  81   shape = specs_[0].shape
  82   dtype = specs_[0].dtype
  83   name = specs_[0].name
  84   for spec in specs_[1:]:
  85     assert shape[1:] == spec.shape[1:], "incompatible shapes: %s, %s" % (
  86         shape, spec.shape)
  87     assert dtype == spec.dtype, "incompatible dtypes: %s, %s" % (
  88         dtype, spec.dtype)
  89     shape = merge_shapes((shape, spec.shape), axis=0)
  90   return specs.TensorSpec(
  91       shape=shape,
  92       dtype=dtype,
  93       name=name,
  94   )

  97 def merge_shapes(shapes, axis=0):
  98   """Merge TensorShapes.
  99 
 100   Args:
 101     shapes: List of TensorShapes to be merged.
 102     axis: optional, the axis to merge shaped.
 103   Returns:
 104     a TensorShape: a merged TensorShape.
 105   """
 106   assert len(shapes) > 1
 107   dims = deepcopy(shapes[0].dims)
 108   for shape in shapes[1:]:
 109     assert shapes[0].ndims == shape.ndims
 110     dims[axis] += shape.dims[axis]
 111   return tf.TensorShape(dims=dims)

 114 def get_all_vars(ignore_scopes=None):
 115   """Get all tf variables in scope.
 116 
 117   Args:
 118     ignore_scopes: A list of scope names to ignore.
 119   Returns:
 120     A list of all tf variables in scope.
 121   """
 122   all_vars = tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)
 123   all_vars = [var for var in all_vars if ignore_scopes is None or not
 124               any(var.name.startswith(scope) for scope in ignore_scopes)]
 125   return all_vars

 128 def clip(tensor, range_=None):
 129   """Return a tf op which clips tensor according to range_.
 130 
 131   Args:
 132     tensor: A Tensor to be clipped.
 133     range_: None, or a tuple representing (minval, maxval)
 134   Returns:
 135     A clipped Tensor.
 136   """
 137   if range_ is None:
 138     return tf.identity(tensor)
 139   elif isinstance(range_, (tuple, list)):
 140     assert len(range_) == 2
 141     return tf.clip_by_value(tensor, range_[0], range_[1])
 142   else: raise NotImplementedError("Unacceptable range input: %r" % range_)

 145 def clip_to_bounds(value, minimum, maximum):
 146   """Clips value to be between minimum and maximum.
 147 
 148   Args:
 149     value: (tensor) value to be clipped.
 150     minimum: (numpy float array) minimum value to clip to.
 151     maximum: (numpy float array) maximum value to clip to.
 152   Returns:
 153     clipped_value: (tensor) `value` clipped to between `minimum` and `maximum`.
 154   """
 155   value = tf.minimum(value, maximum)
 156   return tf.maximum(value, minimum)

 172 join_scope = common.join_scope
 173 def _join_scope(parent_scope, child_scope):
 174   """Joins a parent and child scope using `/`, checking for empty/none.
 175 
 176   Args:
 177     parent_scope: (string) parent/prefix scope.
 178     child_scope: (string) child/suffix scope.
 179   Returns:
 180     joined scope: (string) parent and child scopes joined by /.
 181   """
 182   if not parent_scope:
 183     return child_scope
 184   if not child_scope:
 185     return parent_scope
 186   return '/'.join([parent_scope, child_scope])

 189 def assign_vars(vars_, values):
 190   """Returns the update ops for assigning a list of vars.
 191 
 192   Args:
 193     vars_: A list of variables.
 194     values: A list of tensors representing new values.
 195   Returns:
 196     A list of update ops for the variables.
 197   """
 198   return [var.assign(value) for var, value in zip(vars_, values)]

 212 def tile(var, batch_size=1):
 213   """Return tiled tensor.
 214 
 215   Args:
 216     var: A tensor representing the state.
 217     batch_size: Batch size.
 218   Returns:
 219     A tensor with shape [batch_size,] + var.shape.
 220   """
 221   batch_var = tf.tile(
 222       tf.expand_dims(var, 0),
 223       (batch_size,) + (1,) * var.get_shape().ndims)
 224   return batch_var

 227 def batch_list(vars_list):
 228   """Batch a list of variables.
 229 
 230   Args:
 231     vars_list: A list of tensor variables.
 232   Returns:
 233     A list of tensor variables with additional first dimension.
 234   """
 235   return [tf.expand_dims(var, 0) for var in vars_list]

 238 def tf_print(op,
 239              tensors,
 240              message="",
 241              first_n=-1,
 242              name=None,
 243              sub_messages=None,
 244              print_freq=-1,
 245              include_count=True):
 246   """tf.Print, but to stdout."""
 247   # TODO(shanegu): `name` is deprecated. Remove from the rest of codes.
 248   global _tf_print_ids
 249   _tf_print_ids += 1
 250   name = _tf_print_ids
 251   _tf_print_counts[name] = 0
 252   if print_freq > 0:
 253     _tf_print_running_sums[name] = [0 for _ in tensors]
 254     _tf_print_running_counts[name] = 0
 255   def print_message(*xs):
 256     """print message fn."""
 257     _tf_print_counts[name] += 1
 258     if print_freq > 0:
 259       for i, x in enumerate(xs):
 260         _tf_print_running_sums[name][i] += x
 261       _tf_print_running_counts[name] += 1
 262     if (print_freq <= 0 or _tf_print_running_counts[name] >= print_freq) and (
 263         first_n < 0 or _tf_print_counts[name] <= first_n):
 264       for i, x in enumerate(xs):
 265         if print_freq > 0:
 266           del x
 267           x = _tf_print_running_sums[name][i]/_tf_print_running_counts[name]
 268         if sub_messages is None:
 269           sub_message = str(i)
 270         else:
 271           sub_message = sub_messages[i]
 272         log_message = "%s, %s" % (message, sub_message)
 273         if include_count:
 274           log_message += ", count=%d" % _tf_print_counts[name]
 275         tf.logging.info("[%s]: %s" % (log_message, x))
 276       if print_freq > 0:
 277         for i, x in enumerate(xs):
 278           _tf_print_running_sums[name][i] = 0
 279         _tf_print_running_counts[name] = 0
 280     return xs[0]
 281 
 282   print_op = tf.py_func(print_message, tensors, tensors[0].dtype)
 283   with tf.control_dependencies([print_op]):
 284     op = tf.identity(op)
 285   return op

 288 periodically = common.periodically
 289 def _periodically(body, period, name='periodically'):
 290   """Periodically performs a tensorflow op."""
 291   if period is None or period == 0:
 292     return tf.no_op()
 293 
 294   if period < 0:
 295     raise ValueError("period cannot be less than 0.")
 296 
 297   if period == 1:
 298     return body()
 299 
 300   with tf.variable_scope(None, default_name=name):
 301     counter = tf.get_variable(
 302         "counter",
 303         shape=[],
 304         dtype=tf.int64,
 305         trainable=False,
 306         initializer=tf.constant_initializer(period, dtype=tf.int64))
 307 
 308     def _wrapped_body():
 309       with tf.control_dependencies([body()]):
 310         return counter.assign(1)
 311 
 312     update = tf.cond(
 313         tf.equal(counter, period), _wrapped_body,
 314         lambda: counter.assign_add(1))
 315 
 316   return update
