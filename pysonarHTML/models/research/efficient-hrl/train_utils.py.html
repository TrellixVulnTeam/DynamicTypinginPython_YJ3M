  56 class TrainStep(object):
  57   """Handles training step."""
  58 
  59   def __init__(self,
  60                max_number_of_steps=0,
  61                num_updates_per_observation=1,
  62                num_collect_per_update=1,
  63                num_collect_per_meta_update=1,
  64                log_every_n_steps=1,
  65                policy_save_fn=None,
  66                save_policy_every_n_steps=0,
  67                should_stop_early=None):
  68     """Returns a function that is executed at each step of slim training.
  69 
  70     Args:
  71       max_number_of_steps: Optional maximum number of train steps to take.
  72       num_updates_per_observation: Number of updates per observation.
  73       log_every_n_steps: The frequency, in terms of global steps, that the loss
  74       and global step and logged.
  75       policy_save_fn: A tf.Saver().save function to save the policy.
  76       save_policy_every_n_steps: How frequently to save the policy.
  77       should_stop_early: Optional hook to report whether training should stop.
  78     Raises:
  79       ValueError: If policy_save_fn is not provided when
  80         save_policy_every_n_steps > 0.
  81     """
  82     if save_policy_every_n_steps and policy_save_fn is None:
  83       raise ValueError(
  84           'policy_save_fn is required when save_policy_every_n_steps > 0')
  85     self.max_number_of_steps = max_number_of_steps
  86     self.num_updates_per_observation = num_updates_per_observation
  87     self.num_collect_per_update = num_collect_per_update
  88     self.num_collect_per_meta_update = num_collect_per_meta_update
  89     self.log_every_n_steps = log_every_n_steps
  90     self.policy_save_fn = policy_save_fn
  91     self.save_policy_every_n_steps = save_policy_every_n_steps
  92     self.should_stop_early = should_stop_early
  93     self.last_global_step_val = 0
  94     self.train_op_fn = None
  95     self.collect_and_train_fn = None
  96     tf.logging.info('Training for %d max_number_of_steps',
  97                     self.max_number_of_steps)
  98 
  99   def train_step(self, sess, train_ops, global_step, _):
 100     """This function will be called at each step of training.
 101 
 102     This represents one step of the DDPG algorithm and can include:
 103     1. collect a <state, action, reward, next_state> transition
 104     2. update the target network
 105     3. train the actor
 106     4. train the critic
 107 
 108     Args:
 109       sess: A Tensorflow session.
 110       train_ops: A DdpgTrainOps tuple of train ops to run.
 111       global_step: The global step.
 112 
 113     Returns:
 114       A scalar total loss.
 115       A boolean should stop.
 116     """
 117     start_time = time.time()
 118     if self.train_op_fn is None:
 119       self.train_op_fn = sess.make_callable([train_ops.train_op, global_step])
 120       self.meta_train_op_fn = sess.make_callable([train_ops.meta_train_op, global_step])
 121       self.collect_fn = sess.make_callable([train_ops.collect_experience_op, global_step])
 122       self.collect_and_train_fn = sess.make_callable(
 123           [train_ops.train_op, global_step, train_ops.collect_experience_op])
 124       self.collect_and_meta_train_fn = sess.make_callable(
 125           [train_ops.meta_train_op, global_step, train_ops.collect_experience_op])
 126     for _ in range(self.num_collect_per_update - 1):
 127       self.collect_fn()
 128     for _ in range(self.num_updates_per_observation - 1):
 129       self.train_op_fn()
 130 
 131     total_loss, global_step_val, _ = self.collect_and_train_fn()
 132     if (global_step_val // self.num_collect_per_meta_update !=
 133         self.last_global_step_val // self.num_collect_per_meta_update):
 134       self.meta_train_op_fn()
 135 
 136     time_elapsed = time.time() - start_time
 137     should_stop = False
 138     if self.max_number_of_steps:
 139       should_stop = global_step_val >= self.max_number_of_steps
 140     if global_step_val != self.last_global_step_val:
 141       if (self.save_policy_every_n_steps and
 142           global_step_val // self.save_policy_every_n_steps !=
 143           self.last_global_step_val // self.save_policy_every_n_steps):
 144         self.policy_save_fn(sess)
 145 
 146       if (self.log_every_n_steps and
 147           global_step_val % self.log_every_n_steps == 0):
 148         tf.logging.info(
 149             'global step %d: loss = %.4f (%.3f sec/step) (%d steps/sec)',
 150             global_step_val, total_loss, time_elapsed, 1 / time_elapsed)
 151 
 152     self.last_global_step_val = global_step_val
 153     stop_early = bool(self.should_stop_early and self.should_stop_early())
 154     return total_loss, should_stop or stop_early

 157 def create_counter_summaries(counters):
 158   """Add named summaries to counters, a list of tuples (name, counter)."""
 159   if counters:
 160     with tf.name_scope('Counters/'):
 161       for name, counter in counters:
 162         tf.summary.scalar(name, counter)

 165 def gen_debug_batch_summaries(batch):
 166   """Generates summaries for the sampled replay batch."""
 167   states, actions, rewards, _, next_states = batch
 168   with tf.name_scope('batch'):
 169     for s in range(states.get_shape()[-1]):
 170       tf.summary.histogram('states_%d' % s, states[:, s])
 171     for s in range(states.get_shape()[-1]):
 172       tf.summary.histogram('next_states_%d' % s, next_states[:, s])
 173     for a in range(actions.get_shape()[-1]):
 174       tf.summary.histogram('actions_%d' % a, actions[:, a])
 175     tf.summary.histogram('rewards', rewards)
