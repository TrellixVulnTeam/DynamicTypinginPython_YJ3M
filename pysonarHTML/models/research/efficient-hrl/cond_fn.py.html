  23 @gin.configurable
  24 def env_transition(agent, state, action, transition_type, environment_steps,
  25                    num_episodes):
  26   """True if the transition_type is TRANSITION or FINAL_TRANSITION.
  27 
  28   Args:
  29     agent: RL agent.
  30     state: A [num_state_dims] tensor representing a state.
  31     action: Action performed.
  32     transition_type: Type of transition after action
  33     environment_steps: Number of steps performed by environment.
  34     num_episodes: Number of episodes.
  35   Returns:
  36     cond: Returns an op that evaluates to true if the transition type is
  37     not RESTARTING
  38   """
  39   del agent, state, action, num_episodes, environment_steps
  40   cond = tf.logical_not(transition_type)
  41   return cond

  65 @gin.configurable
  66 def every_n_steps(agent,
  67                   state,
  68                   action,
  69                   transition_type,
  70                   environment_steps,
  71                   num_episodes,
  72                   n=150):
  73   """True once every n steps.
  74 
  75   Args:
  76     agent: RL agent.
  77     state: A [num_state_dims] tensor representing a state.
  78     action: Action performed.
  79     transition_type: Type of transition after action
  80     environment_steps: Number of steps performed by environment.
  81     num_episodes: Number of episodes.
  82     n: Return true once every n steps.
  83   Returns:
  84     cond: Returns an op that evaluates to true if environment_steps
  85     equals 0 mod n. We increment the step before checking this condition, so
  86     we do not need to add one to environment_steps.
  87   """
  88   del agent, state, action, transition_type, num_episodes
  89   cond = tf.equal(tf.mod(environment_steps, n), 0)
  90   return cond

  93 @gin.configurable
  94 def every_n_episodes(agent,
  95                      state,
  96                      action,
  97                      transition_type,
  98                      environment_steps,
  99                      num_episodes,
 100                      n=2,
 101                      steps_per_episode=None):
 102   """True once every n episodes.
 103 
 104   Specifically, evaluates to True on the 0th step of every nth episode.
 105   Unlike environment_steps, num_episodes starts at 0, so we do want to add
 106   one to ensure it does not reset on the first call.
 107 
 108   Args:
 109     agent: RL agent.
 110     state: A [num_state_dims] tensor representing a state.
 111     action: Action performed.
 112     transition_type: Type of transition after action
 113     environment_steps: Number of steps performed by environment.
 114     num_episodes: Number of episodes.
 115     n: Return true once every n episodes.
 116     steps_per_episode: How many steps per episode. Needed to determine when a
 117     new episode starts.
 118   Returns:
 119     cond: Returns an op that evaluates to true on the last step of the episode
 120       (i.e. if num_episodes equals 0 mod n).
 121   """
 122   assert steps_per_episode is not None
 123   del agent, action, transition_type
 124   ant_fell = tf.logical_or(state[2] < 0.2, state[2] > 1.0)
 125   cond = tf.logical_and(
 126       tf.logical_or(
 127           ant_fell,
 128           tf.equal(tf.mod(num_episodes + 1, n), 0)),
 129       tf.equal(tf.mod(environment_steps, steps_per_episode), 0))
 130   return cond

 133 @gin.configurable
 134 def failed_reset_after_n_episodes(agent,
 135                                   state,
 136                                   action,
 137                                   transition_type,
 138                                   environment_steps,
 139                                   num_episodes,
 140                                   steps_per_episode=None,
 141                                   reset_state=None,
 142                                   max_dist=1.0,
 143                                   epsilon=1e-10):
 144   """Every n episodes, returns True if the reset agent fails to return.
 145 
 146   Specifically, evaluates to True if the distance between the state and the
 147   reset state is greater than max_dist at the end of the episode.
 148 
 149   Args:
 150     agent: RL agent.
 151     state: A [num_state_dims] tensor representing a state.
 152     action: Action performed.
 153     transition_type: Type of transition after action
 154     environment_steps: Number of steps performed by environment.
 155     num_episodes: Number of episodes.
 156     steps_per_episode: How many steps per episode. Needed to determine when a
 157     new episode starts.
 158     reset_state: State to which the reset controller should return.
 159     max_dist: Agent is considered to have successfully reset if its distance
 160     from the reset_state is less than max_dist.
 161     epsilon: small offset to ensure non-negative/zero distance.
 162   Returns:
 163     cond: Returns an op that evaluates to true if num_episodes+1 equals 0
 164     mod n. We add one to the num_episodes so the environment is not reset after
 165     the 0th step.
 166   """
 167   assert steps_per_episode is not None
 168   assert reset_state is not None
 169   del agent, state, action, transition_type, num_episodes
 170   dist = tf.sqrt(
 171       tf.reduce_sum(tf.squared_difference(state, reset_state)) + epsilon)
 172   cond = tf.logical_and(
 173       tf.greater(dist, tf.constant(max_dist)),
 174       tf.equal(tf.mod(environment_steps, steps_per_episode), 0))
 175   return cond

 178 @gin.configurable
 179 def q_too_small(agent,
 180                 state,
 181                 action,
 182                 transition_type,
 183                 environment_steps,
 184                 num_episodes,
 185                 q_min=0.5):
 186   """True of q is too small.
 187 
 188   Args:
 189     agent: RL agent.
 190     state: A [num_state_dims] tensor representing a state.
 191     action: Action performed.
 192     transition_type: Type of transition after action
 193     environment_steps: Number of steps performed by environment.
 194     num_episodes: Number of episodes.
 195     q_min: Returns true if the qval is less than q_min
 196   Returns:
 197     cond: Returns an op that evaluates to true if qval is less than q_min.
 198   """
 199   del transition_type, environment_steps, num_episodes
 200   state_for_reset_agent = tf.stack(state[:-1], tf.constant([0], dtype=tf.float))
 201   qval = agent.BASE_AGENT_CLASS.critic_net(
 202       tf.expand_dims(state_for_reset_agent, 0), tf.expand_dims(action, 0))[0, :]
 203   cond = tf.greater(tf.constant(q_min), qval)
 204   return cond

 207 @gin.configurable
 208 def true_fn(agent, state, action, transition_type, environment_steps,
 209             num_episodes):
 210   """Returns an op that evaluates to true.

 215     action: Action performed.
 216     transition_type: Type of transition after action
 217     environment_steps: Number of steps performed by environment.
 218     num_episodes: Number of episodes.
 219   Returns:
 220     cond: op that always evaluates to True.
 221   """
 222   del agent, state, action, transition_type, environment_steps, num_episodes
 223   cond = tf.constant(True, dtype=tf.bool)
 224   return cond
