  49 def collect_experience(tf_env, agent, meta_agent, state_preprocess,
  50                        replay_buffer, meta_replay_buffer,
  51                        action_fn, meta_action_fn,
  52                        environment_steps, num_episodes, num_resets,
  53                        episode_rewards, episode_meta_rewards,
  54                        store_context,
  55                        disable_agent_reset):
  56   """Collect experience in a tf_env into a replay_buffer using action_fn.
  57 
  58   Args:
  59     tf_env: A TFEnvironment.
  60     agent: A UVF agent.
  61     meta_agent: A Meta Agent.
  62     replay_buffer: A Replay buffer to collect experience in.
  63     meta_replay_buffer: A Replay buffer to collect meta agent experience in.
  64     action_fn: A function to produce actions given current state.
  65     meta_action_fn: A function to produce meta actions given current state.
  66     environment_steps: A variable to count the number of steps in the tf_env.
  67     num_episodes: A variable to count the number of episodes.
  68     num_resets: A variable to count the number of resets.
  69     store_context: A boolean to check if store context in replay.
  70     disable_agent_reset: A boolean that disables agent from resetting.
  71 
  72   Returns:
  73     A collect_experience_op that excute an action and store into the
  74     replay_buffers
  75   """
  76   tf_env.start_collect()
  77   state = tf_env.current_obs()
  78   state_repr = state_preprocess(state)
  79   action = action_fn(state, context=None)
  80 
  81   with tf.control_dependencies([state]):
  82     transition_type, reward, discount = tf_env.step(action)
  83 
  84   def increment_step():
  85     return environment_steps.assign_add(1)
  86 
  87   def increment_episode():
  88     return num_episodes.assign_add(1)
  89 
  90   def increment_reset():
  91     return num_resets.assign_add(1)
  92 
  93   def update_episode_rewards(context_reward, meta_reward, reset):
  94     new_episode_rewards = tf.concat(
  95         [episode_rewards[:1] + context_reward, episode_rewards[1:]], 0)
  96     new_episode_meta_rewards = tf.concat(
  97         [episode_meta_rewards[:1] + meta_reward,
  98          episode_meta_rewards[1:]], 0)
  99     return tf.group(
 100         episode_rewards.assign(
 101             tf.cond(reset,
 102                     lambda: tf.concat([[0.], episode_rewards[:-1]], 0),
 103                     lambda: new_episode_rewards)),
 104         episode_meta_rewards.assign(
 105             tf.cond(reset,
 106                     lambda: tf.concat([[0.], episode_meta_rewards[:-1]], 0),
 107                     lambda: new_episode_meta_rewards)))
 108 
 109   def no_op_int():
 110     return tf.constant(0, dtype=tf.int64)
 111 
 112   step_cond = agent.step_cond_fn(state, action,
 113                                  transition_type,
 114                                  environment_steps, num_episodes)
 115   reset_episode_cond = agent.reset_episode_cond_fn(
 116       state, action,
 117       transition_type, environment_steps, num_episodes)
 118   reset_env_cond = agent.reset_env_cond_fn(state, action,
 119                                            transition_type,
 120                                            environment_steps, num_episodes)
 121 
 122   increment_step_op = tf.cond(step_cond, increment_step, no_op_int)
 123   increment_episode_op = tf.cond(reset_episode_cond, increment_episode,
 124                                  no_op_int)
 125   increment_reset_op = tf.cond(reset_env_cond, increment_reset, no_op_int)
 126   increment_op = tf.group(increment_step_op, increment_episode_op,
 127                           increment_reset_op)
 128 
 129   with tf.control_dependencies([increment_op, reward, discount]):
 130     next_state = tf_env.current_obs()
 131     next_state_repr = state_preprocess(next_state)
 132     next_reset_episode_cond = tf.logical_or(
 133         agent.reset_episode_cond_fn(
 134             state, action,
 135             transition_type, environment_steps, num_episodes),
 136         tf.equal(discount, 0.0))
 137 
 138   if store_context:
 139     context = [tf.identity(var) + tf.zeros_like(var) for var in agent.context_vars]
 140     meta_context = [tf.identity(var) + tf.zeros_like(var) for var in meta_agent.context_vars]
 141   else:
 142     context = []
 143     meta_context = []
 144   with tf.control_dependencies([next_state] + context + meta_context):
 145     if disable_agent_reset:
 146       collect_experience_ops = [tf.no_op()]  # don't reset agent
 147     else:
 148       collect_experience_ops = agent.cond_begin_episode_op(
 149           tf.logical_not(reset_episode_cond),
 150           [state, action, reward, next_state,
 151            state_repr, next_state_repr],
 152           mode='explore', meta_action_fn=meta_action_fn)
 153       context_reward, meta_reward = collect_experience_ops
 154       collect_experience_ops = list(collect_experience_ops)
 155       collect_experience_ops.append(
 156           update_episode_rewards(tf.reduce_sum(context_reward), meta_reward,
 157                                  reset_episode_cond))
 158 
 159   meta_action_every_n = agent.tf_context.meta_action_every_n
 160   with tf.control_dependencies(collect_experience_ops):
 161     transition = [state, action, reward, discount, next_state]
 162 
 163     meta_action = tf.to_float(
 164         tf.concat(context, -1))  # Meta agent action is low-level context
 165 
 166     meta_end = tf.logical_and(  # End of meta-transition.
 167         tf.equal(agent.tf_context.t % meta_action_every_n, 1),
 168         agent.tf_context.t > 1)
 169     with tf.variable_scope(tf.get_variable_scope(), reuse=tf.AUTO_REUSE):
 170       states_var = tf.get_variable('states_var',
 171                                    [meta_action_every_n, state.shape[-1]],
 172                                    state.dtype)
 173       actions_var = tf.get_variable('actions_var',
 174                                     [meta_action_every_n, action.shape[-1]],
 175                                     action.dtype)
 176       state_var = tf.get_variable('state_var', state.shape, state.dtype)
 177       reward_var = tf.get_variable('reward_var', reward.shape, reward.dtype)
 178       meta_action_var = tf.get_variable('meta_action_var',
 179                                         meta_action.shape, meta_action.dtype)
 180       meta_context_var = [
 181           tf.get_variable('meta_context_var%d' % idx,
 182                           meta_context[idx].shape, meta_context[idx].dtype)
 183           for idx in range(len(meta_context))]
 184 
 185     actions_var_upd = tf.scatter_update(
 186         actions_var, (agent.tf_context.t - 2) % meta_action_every_n, action)
 187     with tf.control_dependencies([actions_var_upd]):
 188       actions = tf.identity(actions_var) + tf.zeros_like(actions_var)
 189       meta_reward = tf.identity(meta_reward) + tf.zeros_like(meta_reward)
 190       meta_reward = tf.reshape(meta_reward, reward.shape)
 191 
 192     reward = 0.1 * meta_reward
 193     meta_transition = [state_var, meta_action_var,
 194                        reward_var + reward,
 195                        discount * (1 - tf.to_float(next_reset_episode_cond)),
 196                        next_state]
 197     meta_transition.extend([states_var, actions])
 198     if store_context:  # store current and next context into replay
 199       transition += context + list(agent.context_vars)
 200       meta_transition += meta_context_var + list(meta_agent.context_vars)
 201 
 202     meta_step_cond = tf.squeeze(tf.logical_and(step_cond, tf.logical_or(next_reset_episode_cond, meta_end)))
 203 
 204     collect_experience_op = tf.group(
 205         replay_buffer.maybe_add(transition, step_cond),
 206         meta_replay_buffer.maybe_add(meta_transition, meta_step_cond),
 207     )
 208 
 209   with tf.control_dependencies([collect_experience_op]):
 210     collect_experience_op = tf.cond(reset_env_cond,
 211                                     tf_env.reset,
 212                                     tf_env.current_time_step)
 213 
 214     meta_period = tf.equal(agent.tf_context.t % meta_action_every_n, 1)
 215     states_var_upd = tf.scatter_update(
 216         states_var, (agent.tf_context.t - 1) % meta_action_every_n,
 217         next_state)
 218     state_var_upd = tf.assign(
 219         state_var,
 220         tf.cond(meta_period, lambda: next_state, lambda: state_var))
 221     reward_var_upd = tf.assign(
 222         reward_var,
 223         tf.cond(meta_period,
 224                 lambda: tf.zeros_like(reward_var),
 225                 lambda: reward_var + reward))
 226     meta_action = tf.to_float(tf.concat(agent.context_vars, -1))
 227     meta_action_var_upd = tf.assign(
 228         meta_action_var,
 229         tf.cond(meta_period, lambda: meta_action, lambda: meta_action_var))
 230     meta_context_var_upd = [
 231         tf.assign(
 232             meta_context_var[idx],
 233             tf.cond(meta_period,
 234                     lambda: meta_agent.context_vars[idx],
 235                     lambda: meta_context_var[idx]))
 236         for idx in range(len(meta_context))]
 237 
 238   return tf.group(
 239       collect_experience_op,
 240       states_var_upd,
 241       state_var_upd,
 242       reward_var_upd,
 243       meta_action_var_upd,
 244       *meta_context_var_upd)

 247 def sample_best_meta_actions(state_reprs, next_state_reprs, prev_meta_actions,
 248                              low_states, low_actions, low_state_reprs,
 249                              inverse_dynamics, uvf_agent, k=10):
 250   """Return meta-actions which approximately maximize low-level log-probs."""
 251   sampled_actions = inverse_dynamics.sample(state_reprs, next_state_reprs, k, prev_meta_actions)
 252   sampled_actions = tf.stop_gradient(sampled_actions)
 253   sampled_log_probs = tf.reshape(uvf_agent.log_probs(
 254       tf.tile(low_states, [k, 1, 1]),
 255       tf.tile(low_actions, [k, 1, 1]),
 256       tf.tile(low_state_reprs, [k, 1, 1]),
 257       [tf.reshape(sampled_actions, [-1, sampled_actions.shape[-1]])]),
 258                                  [k, low_states.shape[0],
 259                                   low_states.shape[1], -1])
 260   fitness = tf.reduce_sum(sampled_log_probs, [2, 3])
 261   best_actions = tf.argmax(fitness, 0)
 262   actions = tf.gather_nd(
 263       sampled_actions,
 264       tf.stack([best_actions,
 265                 tf.range(prev_meta_actions.shape[0], dtype=tf.int64)], -1))
 266   return actions

 269 @gin.configurable
 270 def train_uvf(train_dir,
 271               environment=None,
 272               num_bin_actions=3,
 273               agent_class=None,
 274               meta_agent_class=None,
 275               state_preprocess_class=None,
 276               inverse_dynamics_class=None,
 277               exp_action_wrapper=None,
 278               replay_buffer=None,
 279               meta_replay_buffer=None,
 280               replay_num_steps=1,
 281               meta_replay_num_steps=1,
 282               critic_optimizer=None,
 283               actor_optimizer=None,
 284               meta_critic_optimizer=None,
 285               meta_actor_optimizer=None,
 286               repr_optimizer=None,
 287               relabel_contexts=False,
 288               meta_relabel_contexts=False,
 289               batch_size=64,
 290               repeat_size=0,
 291               num_episodes_train=2000,
 292               initial_episodes=2,
 293               initial_steps=None,
 294               num_updates_per_observation=1,
 295               num_collect_per_update=1,
 296               num_collect_per_meta_update=1,
 297               gamma=1.0,
 298               meta_gamma=1.0,
 299               reward_scale_factor=1.0,
 300               target_update_period=1,
 301               should_stop_early=None,
 302               clip_gradient_norm=0.0,
 303               summarize_gradients=False,
 304               debug_summaries=False,
 305               log_every_n_steps=100,
 306               prefetch_queue_capacity=2,
 307               policy_save_dir='policy',
 308               save_policy_every_n_steps=1000,
 309               save_policy_interval_secs=0,
 310               replay_context_ratio=0.0,
 311               next_state_as_context_ratio=0.0,
 312               state_index=0,
 313               zero_timer_ratio=0.0,
 314               timer_index=-1,
 315               debug=False,
 316               max_policies_to_save=None,
 317               max_steps_per_episode=None,
 318               load_path=LOAD_PATH):
 319   """Train an agent."""
 320   tf_env = create_maze_env.TFPyEnvironment(environment)
 321   observation_spec = [tf_env.observation_spec()]
 322   action_spec = [tf_env.action_spec()]
 323 
 324   max_steps_per_episode = max_steps_per_episode or tf_env.pyenv.max_episode_steps
 325 
 326   assert max_steps_per_episode, 'max_steps_per_episode need to be set'
 327 
 328   if initial_steps is None:
 329     initial_steps = initial_episodes * max_steps_per_episode
 330 
 331   if agent_class.ACTION_TYPE == 'discrete':
 332     assert False
 333   else:
 334     assert agent_class.ACTION_TYPE == 'continuous'
 335 
 336   assert agent_class.ACTION_TYPE == meta_agent_class.ACTION_TYPE
 337   with tf.variable_scope('meta_agent'):
 338     meta_agent = meta_agent_class(
 339         observation_spec,
 340         action_spec,
 341         tf_env,
 342         debug_summaries=debug_summaries)
 343   meta_agent.set_replay(replay=meta_replay_buffer)
 344 
 345   with tf.variable_scope('uvf_agent'):
 346     uvf_agent = agent_class(
 347         observation_spec,
 348         action_spec,
 349         tf_env,
 350         debug_summaries=debug_summaries)
 351     uvf_agent.set_meta_agent(agent=meta_agent)
 352     uvf_agent.set_replay(replay=replay_buffer)
 353 
 354   with tf.variable_scope('state_preprocess'):
 355     state_preprocess = state_preprocess_class()
 356 
 357   with tf.variable_scope('inverse_dynamics'):
 358     inverse_dynamics = inverse_dynamics_class(
 359         meta_agent.sub_context_as_action_specs[0])
 360 
 361   # Create counter variables
 362   global_step = tf.contrib.framework.get_or_create_global_step()
 363   num_episodes = tf.Variable(0, dtype=tf.int64, name='num_episodes')
 364   num_resets = tf.Variable(0, dtype=tf.int64, name='num_resets')
 365   num_updates = tf.Variable(0, dtype=tf.int64, name='num_updates')
 366   num_meta_updates = tf.Variable(0, dtype=tf.int64, name='num_meta_updates')
 367   episode_rewards = tf.Variable([0.] * 100, name='episode_rewards')
 368   episode_meta_rewards = tf.Variable([0.] * 100, name='episode_meta_rewards')
 369 
 370   # Create counter variables summaries
 371   train_utils.create_counter_summaries([
 372       ('environment_steps', global_step),
 373       ('num_episodes', num_episodes),
 374       ('num_resets', num_resets),
 375       ('num_updates', num_updates),
 376       ('num_meta_updates', num_meta_updates),
 377       ('replay_buffer_adds', replay_buffer.get_num_adds()),
 378       ('meta_replay_buffer_adds', meta_replay_buffer.get_num_adds()),
 379   ])
 380 
 381   tf.summary.scalar('avg_episode_rewards',
 382                     tf.reduce_mean(episode_rewards[1:]))
 383   tf.summary.scalar('avg_episode_meta_rewards',
 384                     tf.reduce_mean(episode_meta_rewards[1:]))
 385   tf.summary.histogram('episode_rewards', episode_rewards[1:])
 386   tf.summary.histogram('episode_meta_rewards', episode_meta_rewards[1:])
 387 
 388   # Create init ops
 389   action_fn = uvf_agent.action
 390   action_fn = uvf_agent.add_noise_fn(action_fn, global_step=None)
 391   meta_action_fn = meta_agent.action
 392   meta_action_fn = meta_agent.add_noise_fn(meta_action_fn, global_step=None)
 393   meta_actions_fn = meta_agent.actions
 394   meta_actions_fn = meta_agent.add_noise_fn(meta_actions_fn, global_step=None)
 395   init_collect_experience_op = collect_experience(
 396       tf_env,
 397       uvf_agent,
 398       meta_agent,
 399       state_preprocess,
 400       replay_buffer,
 401       meta_replay_buffer,
 402       action_fn,
 403       meta_action_fn,
 404       environment_steps=global_step,
 405       num_episodes=num_episodes,
 406       num_resets=num_resets,
 407       episode_rewards=episode_rewards,
 408       episode_meta_rewards=episode_meta_rewards,
 409       store_context=True,
 410       disable_agent_reset=False,
 411   )
 412 
 413   # Create train ops
 414   collect_experience_op = collect_experience(
 415       tf_env,
 416       uvf_agent,
 417       meta_agent,
 418       state_preprocess,
 419       replay_buffer,
 420       meta_replay_buffer,
 421       action_fn,
 422       meta_action_fn,
 423       environment_steps=global_step,
 424       num_episodes=num_episodes,
 425       num_resets=num_resets,
 426       episode_rewards=episode_rewards,
 427       episode_meta_rewards=episode_meta_rewards,
 428       store_context=True,
 429       disable_agent_reset=False,
 430   )
 431 
 432   train_op_list = []
 433   repr_train_op = tf.constant(0.0)
 434   for mode in ['meta', 'nometa']:
 435     if mode == 'meta':
 436       agent = meta_agent
 437       buff = meta_replay_buffer
 438       critic_opt = meta_critic_optimizer
 439       actor_opt = meta_actor_optimizer
 440       relabel = meta_relabel_contexts
 441       num_steps = meta_replay_num_steps
 442       my_gamma = meta_gamma,
 443       n_updates = num_meta_updates
 444     else:
 445       agent = uvf_agent
 446       buff = replay_buffer
 447       critic_opt = critic_optimizer
 448       actor_opt = actor_optimizer
 449       relabel = relabel_contexts
 450       num_steps = replay_num_steps
 451       my_gamma = gamma
 452       n_updates = num_updates
 453 
 454     with tf.name_scope(mode):
 455       batch = buff.get_random_batch(batch_size, num_steps=num_steps)
 456       states, actions, rewards, discounts, next_states = batch[:5]
 457       with tf.name_scope('Reward'):
 458         tf.summary.scalar('average_step_reward', tf.reduce_mean(rewards))
 459       rewards *= reward_scale_factor
 460       batch_queue = slim.prefetch_queue.prefetch_queue(
 461           [states, actions, rewards, discounts, next_states] + batch[5:],
 462           capacity=prefetch_queue_capacity,
 463           name='batch_queue')
 464 
 465       batch_dequeue = batch_queue.dequeue()
 466       if repeat_size > 0:
 467         batch_dequeue = [
 468             tf.tile(batch, (repeat_size+1,) + (1,) * (batch.shape.ndims - 1))
 469             for batch in batch_dequeue
 470         ]
 471         batch_size *= (repeat_size + 1)
 472       states, actions, rewards, discounts, next_states = batch_dequeue[:5]
 473       if mode == 'meta':
 474         low_states = batch_dequeue[5]
 475         low_actions = batch_dequeue[6]
 476         low_state_reprs = state_preprocess(low_states)
 477       state_reprs = state_preprocess(states)
 478       next_state_reprs = state_preprocess(next_states)
 479 
 480       if mode == 'meta':  # Re-label meta-action
 481         prev_actions = actions
 482         if FLAGS.goal_sample_strategy == 'None':
 483           pass
 484         elif FLAGS.goal_sample_strategy == 'FuN':
 485           actions = inverse_dynamics.sample(state_reprs, next_state_reprs, 1, prev_actions, sc=0.1)
 486           actions = tf.stop_gradient(actions)
 487         elif FLAGS.goal_sample_strategy == 'sample':
 488           actions = sample_best_meta_actions(state_reprs, next_state_reprs, prev_actions,
 489                                              low_states, low_actions, low_state_reprs,
 490                                              inverse_dynamics, uvf_agent, k=10)
 491         else:
 492           assert False
 493 
 494       if state_preprocess.trainable and mode == 'meta':
 495         # Representation learning is based on meta-transitions, but is trained
 496         # along with low-level policy updates.
 497         repr_loss, _, _ = state_preprocess.loss(states, next_states, low_actions, low_states)
 498         repr_train_op = slim.learning.create_train_op(
 499             repr_loss,
 500             repr_optimizer,
 501             global_step=None,
 502             update_ops=None,
 503             summarize_gradients=summarize_gradients,
 504             clip_gradient_norm=clip_gradient_norm,
 505             variables_to_train=state_preprocess.get_trainable_vars(),)
 506 
 507       # Get contexts for training
 508       contexts, next_contexts = agent.sample_contexts(
 509           mode='train', batch_size=batch_size,
 510           state=states, next_state=next_states,
 511       )
 512       if not relabel:  # Re-label context (in the style of TDM or HER).
 513         contexts, next_contexts = (
 514             batch_dequeue[-2*len(contexts):-1*len(contexts)],
 515             batch_dequeue[-1*len(contexts):])
 516 
 517       merged_states = agent.merged_states(states, contexts)
 518       merged_next_states = agent.merged_states(next_states, next_contexts)
 519       if mode == 'nometa':
 520         context_rewards, context_discounts = agent.compute_rewards(
 521             'train', state_reprs, actions, rewards, next_state_reprs, contexts)
 522       elif mode == 'meta': # Meta-agent uses sum of rewards, not context-specific rewards.
 523         _, context_discounts = agent.compute_rewards(
 524             'train', states, actions, rewards, next_states, contexts)
 525         context_rewards = rewards
 526 
 527       if agent.gamma_index is not None:
 528         context_discounts *= tf.cast(
 529             tf.reshape(contexts[agent.gamma_index], (-1,)),
 530             dtype=context_discounts.dtype)
 531       else: context_discounts *= my_gamma
 532 
 533       critic_loss = agent.critic_loss(merged_states, actions,
 534                                       context_rewards, context_discounts,
 535                                       merged_next_states)
 536 
 537       critic_loss = tf.reduce_mean(critic_loss)
 538 
 539       actor_loss = agent.actor_loss(merged_states, actions,
 540                                     context_rewards, context_discounts,
 541                                     merged_next_states)
 542       actor_loss *= tf.to_float(  # Only update actor every N steps.
 543           tf.equal(n_updates % target_update_period, 0))
 544 
 545       critic_train_op = slim.learning.create_train_op(
 546           critic_loss,
 547           critic_opt,
 548           global_step=n_updates,
 549           update_ops=None,
 550           summarize_gradients=summarize_gradients,
 551           clip_gradient_norm=clip_gradient_norm,
 552           variables_to_train=agent.get_trainable_critic_vars(),)
 553       critic_train_op = uvf_utils.tf_print(
 554           critic_train_op, [critic_train_op],
 555           message='critic_loss',
 556           print_freq=1000,
 557           name='critic_loss')
 558       train_op_list.append(critic_train_op)
 559       if actor_loss is not None:
 560         actor_train_op = slim.learning.create_train_op(
 561             actor_loss,
 562             actor_opt,
 563             global_step=None,
 564             update_ops=None,
 565             summarize_gradients=summarize_gradients,
 566             clip_gradient_norm=clip_gradient_norm,
 567             variables_to_train=agent.get_trainable_actor_vars(),)
 568         actor_train_op = uvf_utils.tf_print(
 569             actor_train_op, [actor_train_op],
 570             message='actor_loss',
 571             print_freq=1000,
 572             name='actor_loss')
 573         train_op_list.append(actor_train_op)
 574 
 575   assert len(train_op_list) == 4
 576   # Update targets should happen after the networks have been updated.
 577   with tf.control_dependencies(train_op_list[2:]):
 578     update_targets_op = uvf_utils.periodically(
 579         uvf_agent.update_targets, target_update_period, 'update_targets')
 580   if meta_agent is not None:
 581     with tf.control_dependencies(train_op_list[:2]):
 582       update_meta_targets_op = uvf_utils.periodically(
 583           meta_agent.update_targets, target_update_period, 'update_targets')
 584 
 585   assert_op = tf.Assert(  # Hack to get training to stop.
 586       tf.less_equal(global_step, 200 + num_episodes_train * max_steps_per_episode),
 587       [global_step])
 588   with tf.control_dependencies([update_targets_op, assert_op]):
 589     train_op = tf.add_n(train_op_list[2:], name='post_update_targets')
 590     # Representation training steps on every low-level policy training step.
 591     train_op += repr_train_op
 592   with tf.control_dependencies([update_meta_targets_op, assert_op]):
 593     meta_train_op = tf.add_n(train_op_list[:2],
 594                              name='post_update_meta_targets')
 595 
 596   if debug_summaries:
 597     train_.gen_debug_batch_summaries(batch)
 598     slim.summaries.add_histogram_summaries(
 599         uvf_agent.get_trainable_critic_vars(), 'critic_vars')
 600     slim.summaries.add_histogram_summaries(
 601         uvf_agent.get_trainable_actor_vars(), 'actor_vars')
 602 
 603   train_ops = train_utils.TrainOps(train_op, meta_train_op,
 604                                    collect_experience_op)
 605 
 606   policy_save_path = os.path.join(train_dir, policy_save_dir, 'model.ckpt')
 607   policy_vars = uvf_agent.get_actor_vars() + meta_agent.get_actor_vars() + [
 608       global_step, num_episodes, num_resets
 609   ] + list(uvf_agent.context_vars) + list(meta_agent.context_vars) + state_preprocess.get_trainable_vars()
 610   # add critic vars, since some test evaluation depends on them
 611   policy_vars += uvf_agent.get_trainable_critic_vars() + meta_agent.get_trainable_critic_vars()
 612   policy_saver = tf.train.Saver(
 613       policy_vars, max_to_keep=max_policies_to_save, sharded=False)
 614 
 615   lowlevel_vars = (uvf_agent.get_actor_vars() +
 616                    uvf_agent.get_trainable_critic_vars() +
 617                    state_preprocess.get_trainable_vars())
 618   lowlevel_saver = tf.train.Saver(lowlevel_vars)
 619 
 620   def policy_save_fn(sess):
 621     policy_saver.save(
 622         sess, policy_save_path, global_step=global_step, write_meta_graph=False)
 623     if save_policy_interval_secs > 0:
 624       tf.logging.info(
 625           'Wait %d secs after save policy.' % save_policy_interval_secs)
 626       time.sleep(save_policy_interval_secs)
 627 
 628   train_step_fn = train_utils.TrainStep(
 629       max_number_of_steps=num_episodes_train * max_steps_per_episode + 100,
 630       num_updates_per_observation=num_updates_per_observation,
 631       num_collect_per_update=num_collect_per_update,
 632       num_collect_per_meta_update=num_collect_per_meta_update,
 633       log_every_n_steps=log_every_n_steps,
 634       policy_save_fn=policy_save_fn,
 635       save_policy_every_n_steps=save_policy_every_n_steps,
 636       should_stop_early=should_stop_early).train_step
 637 
 638   local_init_op = tf.local_variables_initializer()
 639   init_targets_op = tf.group(uvf_agent.update_targets(1.0),
 640                              meta_agent.update_targets(1.0))
 641 
 642   def initialize_training_fn(sess):
 643     """Initialize training function."""
 644     sess.run(local_init_op)
 645     sess.run(init_targets_op)
 646     if load_path:
 647       tf.logging.info('Restoring low-level from %s' % load_path)
 648       lowlevel_saver.restore(sess, load_path)
 649     global_step_value = sess.run(global_step)
 650     assert global_step_value == 0, 'Global step should be zero.'
 651     collect_experience_call = sess.make_callable(
 652         init_collect_experience_op)
 653 
 654     for _ in range(initial_steps):
 655       collect_experience_call()
 656 
 657   train_saver = tf.train.Saver(max_to_keep=2, sharded=True)
 658   tf.logging.info('train dir: %s', train_dir)
 659   return slim.learning.train(
 660       train_ops,
 661       train_dir,
 662       train_step_fn=train_step_fn,
 663       save_interval_secs=FLAGS.save_interval_secs,
 664       saver=train_saver,
 665       log_every_n_steps=0,
 666       global_step=global_step,
 667       master="",
 668       is_chief=(FLAGS.task == 0),
 669       save_summaries_secs=FLAGS.save_summaries_secs,
 670       init_fn=initialize_training_fn)
