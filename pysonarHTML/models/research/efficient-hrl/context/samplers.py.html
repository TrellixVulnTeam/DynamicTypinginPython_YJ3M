  31 @gin.configurable
  32 class BaseSampler(object):
  33   """Base sampler."""
  34 
  35   def __init__(self, context_spec, context_range=None, k=2, scope='sampler'):
  36     """Construct a base sampler.
  37 
  38     Args:
  39       context_spec: A context spec.
  40       context_range: A tuple of (minval, max), where minval, maxval are floats
  41         or Numpy arrays with the same shape as the context.
  42       scope: A string denoting scope.
  43     """
  44     self._context_spec = context_spec
  45     self._context_range = context_range
  46     self._k = k
  47     self._scope = scope
  48 
  49   def __call__(self, batch_size, **kwargs):
  50     raise NotImplementedError
  51 
  52   def set_replay(self, replay=None):
  53     pass
  54 
  55   def _validate_contexts(self, contexts):
  56     """Validate if contexts have right spec.
  57 
  58     Args:
  59       contexts: A [batch_size, num_contexts_dim] tensor.
  60     Raises:
  61       ValueError: If shape or dtype mismatches that of spec.
  62     """
  63     if contexts[0].shape != self._context_spec.shape:
  64       raise ValueError('contexts has invalid shape %s wrt spec shape %s' %
  65                        (contexts[0].shape, self._context_spec.shape))
  66     if contexts.dtype != self._context_spec.dtype:
  67       raise ValueError('contexts has invalid dtype %s wrt spec dtype %s' %
  68                        (contexts.dtype, self._context_spec.dtype))

  71 @gin.configurable
  72 class ZeroSampler(BaseSampler):
  73   """Zero sampler."""
  74 
  75   def __call__(self, batch_size, **kwargs):
  76     """Sample a batch of context.
  77 
  78     Args:
  79       batch_size: Batch size.
  80     Returns:
  81       Two [batch_size, num_context_dims] tensors.
  82     """
  83     contexts = tf.zeros(
  84         dtype=self._context_spec.dtype,
  85         shape=[
  86             batch_size,
  87         ] + self._context_spec.shape.as_list())
  88     return contexts, contexts

  91 @gin.configurable
  92 class BinarySampler(BaseSampler):
  93   """Binary sampler."""
  94 
  95   def __init__(self, probs=0.5, *args, **kwargs):
  96     """Constructor."""
  97     super(BinarySampler, self).__init__(*args, **kwargs)
  98     self._probs = probs
  99 
 100   def __call__(self, batch_size, **kwargs):
 101     """Sample a batch of context."""
 102     spec = self._context_spec
 103     contexts = tf.random_uniform(
 104         shape=[
 105             batch_size,
 106         ] + spec.shape.as_list(), dtype=tf.float32)
 107     contexts = tf.cast(tf.greater(contexts, self._probs), dtype=spec.dtype)
 108     return contexts, contexts

 161 @gin.configurable
 162 class ScheduledSampler(BaseSampler):
 163   """Scheduled sampler."""
 164 
 165   def __init__(self,
 166                scope='default',
 167                values=None,
 168                scheduler='cycle',
 169                scheduler_params=None,
 170                *args, **kwargs):
 171     """Construct sampler.
 172 
 173     Args:
 174       scope: Scope name.
 175       values: A list of numbers or [num_context_dim] Numpy arrays
 176         representing the values to cycle.
 177       scheduler: scheduler type.
 178       scheduler_params: scheduler parameters.
 179       *args: arguments.
 180       **kwargs: keyword arguments.
 181     """
 182     super(ScheduledSampler, self).__init__(*args, **kwargs)
 183     self._scope = scope
 184     self._values = values
 185     self._scheduler = scheduler
 186     self._scheduler_params = scheduler_params or {}
 187     assert self._values is not None and len(
 188         self._values), 'must provide non-empty values.'
 189     self._n = len(self._values)
 190     # TODO(shanegu): move variable creation outside. resolve tf.cond problem.
 191     self._count = 0
 192     self._i = tf.Variable(
 193         tf.zeros(shape=(), dtype=tf.int32),
 194         name='%s-scheduled_sampler_%d' % (self._scope, self._count))
 195     self._values = tf.constant(self._values, dtype=self._context_spec.dtype)
 196 
 197   def __call__(self, batch_size, **kwargs):
 198     """Sample a batch of context.
 199 
 200     Args:
 201       batch_size: Batch size.
 202     Returns:
 203       Two [batch_size, num_context_dims] tensors.
 204     """
 205     spec = self._context_spec
 206     next_op = self._next(self._i)
 207     with tf.control_dependencies([next_op]):
 208       value = self._values[self._i]
 209       if value.get_shape().as_list():
 210         values = tf.tile(
 211             tf.expand_dims(value, 0), (batch_size,) + (1,) * spec.shape.ndims)
 212       else:
 213         values = value + tf.zeros(
 214             shape=[
 215                 batch_size,
 216             ] + spec.shape.as_list(), dtype=spec.dtype)
 217     self._validate_contexts(values)
 218     self._count += 1
 219     return values, values
 220 
 221   def _next(self, i):
 222     """Return op that increments pointer to next value.
 223 
 224     Args:
 225       i: A tensorflow integer variable.
 226     Returns:
 227       Op that increments pointer.
 228     """
 229     if self._scheduler == 'cycle':
 230       inc = ('inc' in self._scheduler_params and
 231              self._scheduler_params['inc']) or 1
 232       return tf.assign(i, tf.mod(i+inc, self._n))
 233     else:
 234       raise NotImplementedError(self._scheduler)

 237 @gin.configurable
 238 class ReplaySampler(BaseSampler):
 239   """Replay sampler."""
 240 
 241   def __init__(self,
 242                prefetch_queue_capacity=2,
 243                override_indices=None,
 244                state_indices=None,
 245                *args,
 246                **kwargs):
 247     """Construct sampler.
 248 
 249     Args:
 250       prefetch_queue_capacity: Capacity for prefetch queue.
 251       override_indices: Override indices.
 252       state_indices: Select certain indices from state dimension.
 253       *args: arguments.
 254       **kwargs: keyword arguments.
 255     """
 256     super(ReplaySampler, self).__init__(*args, **kwargs)
 257     self._prefetch_queue_capacity = prefetch_queue_capacity
 258     self._override_indices = override_indices
 259     self._state_indices = state_indices
 260 
 261   def set_replay(self, replay):
 262     """Set replay.
 263 
 264     Args:
 265       replay: A replay buffer.
 266     """
 267     self._replay = replay
 268 
 269   def __call__(self, batch_size, **kwargs):
 270     """Sample a batch of context.
 271 
 272     Args:
 273       batch_size: Batch size.
 274     Returns:
 275       Two [batch_size, num_context_dims] tensors.
 276     """
 277     batch = self._replay.GetRandomBatch(batch_size)
 278     next_states = batch[4]
 279     if self._prefetch_queue_capacity > 0:
 280       batch_queue = slim.prefetch_queue.prefetch_queue(
 281           [next_states],
 282           capacity=self._prefetch_queue_capacity,
 283           name='%s/batch_context_queue' % self._scope)
 284       next_states = batch_queue.dequeue()
 285     if self._override_indices is not None:
 286       assert self._context_range is not None and isinstance(
 287           self._context_range[0], (int, long, float))
 288       next_states = tf.concat(
 289           [
 290               tf.random_uniform(
 291                   shape=next_states[:, :1].shape,
 292                   minval=self._context_range[0],
 293                   maxval=self._context_range[1],
 294                   dtype=next_states.dtype)
 295               if i in self._override_indices else next_states[:, i:i + 1]
 296               for i in range(self._context_spec.shape.as_list()[0])
 297           ],
 298           axis=1)
 299     if self._state_indices is not None:
 300       next_states = tf.concat(
 301           [
 302               next_states[:, i:i + 1]
 303               for i in range(self._context_spec.shape.as_list()[0])
 304           ],
 305           axis=1)
 306     self._validate_contexts(next_states)
 307     return next_states, next_states

 310 @gin.configurable
 311 class TimeSampler(BaseSampler):
 312   """Time Sampler."""
 313 
 314   def __init__(self, minval=0, maxval=1, timestep=-1, *args, **kwargs):
 315     """Construct sampler.
 316 
 317     Args:
 318       minval: Min value integer.
 319       maxval: Max value integer.
 320       timestep: Time step between states and next_states.
 321       *args: arguments.
 322       **kwargs: keyword arguments.
 323     """
 324     super(TimeSampler, self).__init__(*args, **kwargs)
 325     assert self._context_spec.shape.as_list() == [1]
 326     self._minval = minval
 327     self._maxval = maxval
 328     self._timestep = timestep
 329 
 330   def __call__(self, batch_size, **kwargs):
 331     """Sample a batch of context.
 332 
 333     Args:
 334       batch_size: Batch size.
 335     Returns:
 336       Two [batch_size, num_context_dims] tensors.
 337     """
 338     if self._maxval == self._minval:
 339       contexts = tf.constant(
 340           self._maxval, shape=[batch_size, 1], dtype=tf.int32)
 341     else:
 342       contexts = tf.random_uniform(
 343           shape=[batch_size, 1],
 344           dtype=tf.int32,
 345           maxval=self._maxval,
 346           minval=self._minval)
 347     next_contexts = tf.maximum(contexts + self._timestep, 0)
 348 
 349     return tf.cast(
 350         contexts, dtype=self._context_spec.dtype), tf.cast(
 351             next_contexts, dtype=self._context_spec.dtype)

 354 @gin.configurable
 355 class ConstantSampler(BaseSampler):
 356   """Constant sampler."""
 357 
 358   def __init__(self, value=None, *args, **kwargs):
 359     """Construct sampler.
 360 
 361     Args:
 362       value: A list or Numpy array for values of the constant.
 363       *args: arguments.
 364       **kwargs: keyword arguments.
 365     """
 366     super(ConstantSampler, self).__init__(*args, **kwargs)
 367     self._value = value
 368 
 369   def __call__(self, batch_size, **kwargs):
 370     """Sample a batch of context.
 371 
 372     Args:
 373       batch_size: Batch size.
 374     Returns:
 375       Two [batch_size, num_context_dims] tensors.
 376     """
 377     spec = self._context_spec
 378     value_ = tf.constant(self._value, shape=spec.shape, dtype=spec.dtype)
 379     values = tf.tile(
 380         tf.expand_dims(value_, 0), (batch_size,) + (1,) * spec.shape.ndims)
 381     self._validate_contexts(values)
 382     return values, values

 385 @gin.configurable
 386 class DirectionSampler(RandomSampler):
 387   """Direction sampler."""
 388 
 389   def __call__(self, batch_size, **kwargs):
 390     """Sample a batch of context.
 391 
 392     Args:
 393       batch_size: Batch size.
 394     Returns:
 395       Two [batch_size, num_context_dims] tensors.
 396     """
 397     spec = self._context_spec
 398     context_range = self._context_range
 399     if isinstance(context_range[0], (int, float)):
 400       contexts = tf.random_uniform(
 401           shape=[
 402               batch_size,
 403           ] + spec.shape.as_list(),
 404           minval=context_range[0],
 405           maxval=context_range[1],
 406           dtype=spec.dtype)
 407     elif isinstance(context_range[0], (list, tuple, np.ndarray)):
 408       assert len(spec.shape.as_list()) == 1
 409       assert spec.shape.as_list()[0] == len(context_range[0])
 410       assert spec.shape.as_list()[0] == len(context_range[1])
 411       contexts = tf.concat(
 412           [
 413               tf.random_uniform(
 414                   shape=[
 415                       batch_size, 1,
 416                   ] + spec.shape.as_list()[1:],
 417                   minval=context_range[0][i],
 418                   maxval=context_range[1][i],
 419                   dtype=spec.dtype) for i in range(spec.shape.as_list()[0])
 420           ],
 421           axis=1)
 422     else: raise NotImplementedError(context_range)
 423     self._validate_contexts(contexts)
 424     if 'sampler_fn' in kwargs:
 425       other_contexts = kwargs['sampler_fn']()
 426     else:
 427       other_contexts = contexts
 428     state, next_state = kwargs['state'], kwargs['next_state']
 429     if state is not None and next_state is not None:
 430       my_context_range = (np.array(context_range[1]) - np.array(context_range[0])) / 2 * np.ones(spec.shape.as_list())
 431       contexts = tf.concat(
 432           [0.1 * my_context_range[:self._k] *
 433            tf.random_normal(tf.shape(state[:, :self._k]), dtype=state.dtype) +
 434            tf.random_shuffle(state[:, :self._k]) - state[:, :self._k],
 435            other_contexts[:, self._k:]], 1)
 436       #contexts = tf.Print(contexts,
 437       #                    [contexts, tf.reduce_max(contexts, 0),
 438       #                     tf.reduce_min(state, 0), tf.reduce_max(state, 0)], 'contexts', summarize=15)
 439       next_contexts = tf.concat( #LALA
 440           [state[:, :self._k] + contexts[:, :self._k] - next_state[:, :self._k],
 441            other_contexts[:, self._k:]], 1)
 442       next_contexts = contexts  #LALA cosine
 443     else:
 444       next_contexts = contexts
 445     return tf.stop_gradient(contexts), tf.stop_gradient(next_contexts)
