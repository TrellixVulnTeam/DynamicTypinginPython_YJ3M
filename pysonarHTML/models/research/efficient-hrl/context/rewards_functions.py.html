  33 def summarize_stats(stats):
  34   """Summarize a dictionary of variables.
  35 
  36   Args:
  37     stats: a dictionary of {name: tensor} to compute stats over.
  38   """
  39   for name, stat in stats.items():
  40     mean = tf.reduce_mean(stat)
  41     tf.summary.scalar('mean_%s' % name, mean)
  42     tf.summary.scalar('max_%s' % name, tf.reduce_max(stat))
  43     tf.summary.scalar('min_%s' % name, tf.reduce_min(stat))
  44     std = tf.sqrt(tf.reduce_mean(tf.square(stat)) - tf.square(mean) + 1e-10)
  45     tf.summary.scalar('std_%s' % name, std)
  46     tf.summary.histogram(name, stat)

  49 def index_states(states, indices):
  50   """Return indexed states.
  51 
  52   Args:
  53     states: A [batch_size, num_state_dims] Tensor representing a batch
  54         of states.
  55     indices: (a list of Numpy integer array) Indices of states dimensions
  56       to be mapped.
  57   Returns:
  58     A [batch_size, num_indices] Tensor representing the batch of indexed states.
  59   """
  60   if indices is None:
  61     return states
  62   indices = tf.constant(indices, dtype=tf.int32)
  63   return tf.gather(states, indices=indices, axis=1)

  66 def record_tensor(tensor, indices, stats, name='states'):
  67   """Record specified tensor dimensions into stats.
  68 
  69   Args:
  70     tensor: A [batch_size, num_dims] Tensor.
  71     indices: (a list of integers) Indices of dimensions to record.
  72     stats: A dictionary holding stats.
  73     name: (string) Name of tensor.
  74   """
  75   if indices is None:
  76     indices = range(tensor.shape.as_list()[1])
  77   for index in indices:
  78     stats['%s_%02d' % (name, index)] = tensor[:, index]

  82 def potential_rewards(states,
  83                       actions,
  84                       rewards,
  85                       next_states,
  86                       contexts,
  87                       gamma=1.0,
  88                       reward_fn=None):
  89   """Return the potential-based rewards.
  90 
  91   Args:
  92     states: A [batch_size, num_state_dims] Tensor representing a batch
  93         of states.
  94     actions: A [batch_size, num_action_dims] Tensor representing a batch
  95       of actions.
  96     rewards: A [batch_size] Tensor representing a batch of rewards.
  97     next_states: A [batch_size, num_state_dims] Tensor representing a batch
  98       of next states.
  99     contexts: A list of [batch_size, num_context_dims] Tensor representing
 100       a batch of contexts.
 101     gamma: Reward discount.
 102     reward_fn: A reward function.
 103   Returns:
 104     A new tf.float32 [batch_size] rewards Tensor, and
 105       tf.float32 [batch_size] discounts tensor.
 106   """
 107   del actions  # unused args
 108   gamma = tf.to_float(gamma)
 109   rewards_tp1, discounts = reward_fn(None, None, rewards, next_states, contexts)
 110   rewards, _ = reward_fn(None, None, rewards, states, contexts)
 111   return -rewards + gamma * rewards_tp1, discounts

 114 @gin.configurable
 115 def timed_rewards(states,
 116                   actions,
 117                   rewards,
 118                   next_states,
 119                   contexts,
 120                   reward_fn=None,
 121                   dense=False,
 122                   timer_index=-1):
 123   """Return the timed rewards.
 124 
 125   Args:
 126     states: A [batch_size, num_state_dims] Tensor representing a batch
 127         of states.
 128     actions: A [batch_size, num_action_dims] Tensor representing a batch
 129       of actions.
 130     rewards: A [batch_size] Tensor representing a batch of rewards.
 131     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 132       of next states.
 133     contexts: A list of [batch_size, num_context_dims] Tensor representing
 134       a batch of contexts.
 135     reward_fn: A reward function.
 136     dense: (boolean) Provide dense rewards or sparse rewards at time = 0.
 137     timer_index: (integer) The context list index that specifies timer.
 138   Returns:
 139     A new tf.float32 [batch_size] rewards Tensor, and
 140       tf.float32 [batch_size] discounts tensor.
 141   """
 142   assert contexts[timer_index].get_shape().as_list()[1] == 1
 143   timers = contexts[timer_index][:, 0]
 144   rewards, discounts = reward_fn(states, actions, rewards, next_states,
 145                                  contexts)
 146   terminates = tf.to_float(timers <= 0)  # if terminate set 1, else set 0
 147   for _ in range(rewards.shape.ndims - 1):
 148     terminates = tf.expand_dims(terminates, axis=-1)
 149   if not dense:
 150     rewards *= terminates  # if terminate, return rewards, else return 0
 151   discounts *= (tf.to_float(1.0) - terminates)
 152   return rewards, discounts

 155 @gin.configurable
 156 def reset_rewards(states,
 157                   actions,
 158                   rewards,
 159                   next_states,
 160                   contexts,
 161                   reset_index=0,
 162                   reset_state=None,
 163                   reset_reward_function=None,
 164                   include_forward_rewards=True,
 165                   include_reset_rewards=True):
 166   """Returns the rewards for a forward/reset agent.
 167 
 168   Args:
 169     states: A [batch_size, num_state_dims] Tensor representing a batch
 170         of states.
 171     actions: A [batch_size, num_action_dims] Tensor representing a batch
 172       of actions.
 173     rewards: A [batch_size] Tensor representing a batch of rewards.
 174     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 175       of next states.
 176     contexts: A list of [batch_size, num_context_dims] Tensor representing
 177       a batch of contexts.
 178     reset_index: (integer) The context list index that specifies reset.
 179     reset_state: Reset state.
 180     reset_reward_function: Reward function for reset step.
 181     include_forward_rewards: Include the rewards from the forward pass.
 182     include_reset_rewards: Include the rewards from the reset pass.
 183 
 184   Returns:
 185     A new tf.float32 [batch_size] rewards Tensor, and
 186       tf.float32 [batch_size] discounts tensor.
 187   """
 188   reset_state = tf.constant(
 189       reset_state, dtype=next_states.dtype, shape=next_states.shape)
 190   reset_states = tf.expand_dims(reset_state, 0)
 191 
 192   def true_fn():
 193     if include_reset_rewards:
 194       return reset_reward_function(states, actions, rewards, next_states,
 195                                    [reset_states] + contexts[1:])
 196     else:
 197       return tf.zeros_like(rewards), tf.ones_like(rewards)
 198 
 199   def false_fn():
 200     if include_forward_rewards:
 201       return plain_rewards(states, actions, rewards, next_states, contexts)
 202     else:
 203       return tf.zeros_like(rewards), tf.ones_like(rewards)
 204 
 205   rewards, discounts = tf.cond(
 206       tf.cast(contexts[reset_index][0, 0], dtype=tf.bool), true_fn, false_fn)
 207   return rewards, discounts

 210 @gin.configurable
 211 def tanh_similarity(states,
 212                     actions,
 213                     rewards,
 214                     next_states,
 215                     contexts,
 216                     mse_scale=1.0,
 217                     state_scales=1.0,
 218                     goal_scales=1.0,
 219                     summarize=False):
 220   """Returns the similarity between next_states and contexts using tanh and mse.
 221 
 222   Args:
 223     states: A [batch_size, num_state_dims] Tensor representing a batch
 224         of states.
 225     actions: A [batch_size, num_action_dims] Tensor representing a batch
 226       of actions.
 227     rewards: A [batch_size] Tensor representing a batch of rewards.
 228     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 229       of next states.
 230     contexts: A list of [batch_size, num_context_dims] Tensor representing
 231       a batch of contexts.
 232     mse_scale: A float, to scale mse before tanh.
 233     state_scales: multiplicative scale for (next) states. A scalar or 1D tensor,
 234       must be broadcastable to number of state dimensions.
 235     goal_scales: multiplicative scale for contexts. A scalar or 1D tensor,
 236       must be broadcastable to number of goal dimensions.
 237     summarize: (boolean) enable summary ops.
 238 
 239 
 240   Returns:
 241     A new tf.float32 [batch_size] rewards Tensor, and
 242       tf.float32 [batch_size] discounts tensor.
 243   """
 244   del states, actions, rewards  # Unused
 245   mse = tf.reduce_mean(tf.squared_difference(next_states * state_scales,
 246                                              contexts[0] * goal_scales), -1)
 247   tanh = tf.tanh(mse_scale * mse)
 248   if summarize:
 249     with tf.name_scope('RewardFn/'):
 250       tf.summary.scalar('mean_mse', tf.reduce_mean(mse))
 251       tf.summary.histogram('mse', mse)
 252       tf.summary.scalar('mean_tanh', tf.reduce_mean(tanh))
 253       tf.summary.histogram('tanh', tanh)
 254   rewards = tf.to_float(1 - tanh)
 255   return rewards, tf.ones_like(rewards)

 300 @gin.configurable
 301 def negative_distance(states,
 302                       actions,
 303                       rewards,
 304                       next_states,
 305                       contexts,
 306                       state_scales=1.0,
 307                       goal_scales=1.0,
 308                       reward_scales=1.0,
 309                       weight_index=None,
 310                       weight_vector=None,
 311                       summarize=False,
 312                       termination_epsilon=1e-4,
 313                       state_indices=None,
 314                       goal_indices=None,
 315                       vectorize=False,
 316                       relative_context=False,
 317                       diff=False,
 318                       norm='L2',
 319                       epsilon=1e-10,
 320                       bonus_epsilon=0., #5.,
 321                       offset=0.0):
 322   """Returns the negative euclidean distance between next_states and contexts.
 323 
 324   Args:
 325     states: A [batch_size, num_state_dims] Tensor representing a batch
 326         of states.
 327     actions: A [batch_size, num_action_dims] Tensor representing a batch
 328       of actions.
 329     rewards: A [batch_size] Tensor representing a batch of rewards.
 330     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 331       of next states.
 332     contexts: A list of [batch_size, num_context_dims] Tensor representing
 333       a batch of contexts.
 334     state_scales: multiplicative scale for (next) states. A scalar or 1D tensor,
 335       must be broadcastable to number of state dimensions.
 336     goal_scales: multiplicative scale for goals. A scalar or 1D tensor,
 337       must be broadcastable to number of goal dimensions.
 338     reward_scales: multiplicative scale for rewards. A scalar or 1D tensor,
 339       must be broadcastable to number of reward dimensions.
 340     weight_index: (integer) The context list index that specifies weight.
 341     weight_vector: (a number or a list or Numpy array) The weighting vector,
 342       broadcastable to `next_states`.
 343     summarize: (boolean) enable summary ops.
 344     termination_epsilon: terminate if dist is less than this quantity.
 345     state_indices: (a list of integers) list of state indices to select.
 346     goal_indices: (a list of integers) list of goal indices to select.
 347     vectorize: Return a vectorized form.
 348     norm: L1 or L2.
 349     epsilon: small offset to ensure non-negative/zero distance.
 350 
 351   Returns:
 352     A new tf.float32 [batch_size] rewards Tensor, and
 353       tf.float32 [batch_size] discounts tensor.
 354   """
 355   del actions, rewards  # Unused
 356   stats = {}
 357   record_tensor(next_states, state_indices, stats, 'next_states')
 358   states = index_states(states, state_indices)
 359   next_states = index_states(next_states, state_indices)
 360   goals = index_states(contexts[0], goal_indices)
 361   if relative_context:
 362     goals = states + goals
 363   sq_dists = tf.squared_difference(next_states * state_scales,
 364                                    goals * goal_scales)
 365   old_sq_dists = tf.squared_difference(states * state_scales,
 366                                        goals * goal_scales)
 367   record_tensor(sq_dists, None, stats, 'sq_dists')
 368   if weight_vector is not None:
 369     sq_dists *= tf.convert_to_tensor(weight_vector, dtype=next_states.dtype)
 370     old_sq_dists *= tf.convert_to_tensor(weight_vector, dtype=next_states.dtype)
 371   if weight_index is not None:
 372     #sq_dists *= contexts[weight_index]
 373     weights = tf.abs(index_states(contexts[0], weight_index))
 374     #weights /= tf.reduce_sum(weights, -1, keepdims=True)
 375     sq_dists *= weights
 376     old_sq_dists *= weights
 377   if norm == 'L1':
 378     dist = tf.sqrt(sq_dists + epsilon)
 379     old_dist = tf.sqrt(old_sq_dists + epsilon)
 380     if not vectorize:
 381       dist = tf.reduce_sum(dist, -1)
 382       old_dist = tf.reduce_sum(old_dist, -1)
 383   elif norm == 'L2':
 384     if vectorize:
 385       dist = sq_dists
 386       old_dist = old_sq_dists
 387     else:
 388       dist = tf.reduce_sum(sq_dists, -1)
 389       old_dist = tf.reduce_sum(old_sq_dists, -1)
 390     dist = tf.sqrt(dist + epsilon)  # tf.gradients fails when tf.sqrt(-0.0)
 391     old_dist = tf.sqrt(old_dist + epsilon)  # tf.gradients fails when tf.sqrt(-0.0)
 392   else:
 393     raise NotImplementedError(norm)
 394   discounts = dist > termination_epsilon
 395   if summarize:
 396     with tf.name_scope('RewardFn/'):
 397       tf.summary.scalar('mean_dist', tf.reduce_mean(dist))
 398       tf.summary.histogram('dist', dist)
 399       summarize_stats(stats)
 400   bonus = tf.to_float(dist < bonus_epsilon)
 401   dist *= reward_scales
 402   old_dist *= reward_scales
 403   if diff:
 404     return bonus + offset + tf.to_float(old_dist - dist), tf.to_float(discounts)
 405   return bonus + offset + tf.to_float(-dist), tf.to_float(discounts)
 406

 408 @gin.configurable
 409 def cosine_similarity(states,
 410                       actions,
 411                       rewards,
 412                       next_states,
 413                       contexts,
 414                       state_scales=1.0,
 415                       goal_scales=1.0,
 416                       reward_scales=1.0,
 417                       normalize_states=True,
 418                       normalize_goals=True,
 419                       weight_index=None,
 420                       weight_vector=None,
 421                       summarize=False,
 422                       state_indices=None,
 423                       goal_indices=None,
 424                       offset=0.0):
 425   """Returns the cosine similarity between next_states - states and contexts.
 426 
 427   Args:
 428     states: A [batch_size, num_state_dims] Tensor representing a batch
 429         of states.
 430     actions: A [batch_size, num_action_dims] Tensor representing a batch
 431       of actions.
 432     rewards: A [batch_size] Tensor representing a batch of rewards.
 433     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 434       of next states.
 435     contexts: A list of [batch_size, num_context_dims] Tensor representing
 436       a batch of contexts.
 437     state_scales: multiplicative scale for (next) states. A scalar or 1D tensor,
 438       must be broadcastable to number of state dimensions.
 439     goal_scales: multiplicative scale for goals. A scalar or 1D tensor,
 440       must be broadcastable to number of goal dimensions.
 441     reward_scales: multiplicative scale for rewards. A scalar or 1D tensor,
 442       must be broadcastable to number of reward dimensions.
 443     weight_index: (integer) The context list index that specifies weight.
 444     weight_vector: (a number or a list or Numpy array) The weighting vector,
 445       broadcastable to `next_states`.
 446     summarize: (boolean) enable summary ops.
 447     termination_epsilon: terminate if dist is less than this quantity.
 448     state_indices: (a list of integers) list of state indices to select.
 449     goal_indices: (a list of integers) list of goal indices to select.
 450     vectorize: Return a vectorized form.
 451     norm: L1 or L2.
 452     epsilon: small offset to ensure non-negative/zero distance.
 453 
 454   Returns:
 455     A new tf.float32 [batch_size] rewards Tensor, and
 456       tf.float32 [batch_size] discounts tensor.
 457   """
 458   del actions, rewards  # Unused
 459   stats = {}
 460   record_tensor(next_states, state_indices, stats, 'next_states')
 461   states = index_states(states, state_indices)
 462   next_states = index_states(next_states, state_indices)
 463   goals = index_states(contexts[0], goal_indices)
 464 
 465   if weight_vector is not None:
 466     goals *= tf.convert_to_tensor(weight_vector, dtype=next_states.dtype)
 467   if weight_index is not None:
 468     weights = tf.abs(index_states(contexts[0], weight_index))
 469     goals *= weights
 470 
 471   direction_vec = next_states - states
 472   if normalize_states:
 473     direction_vec = tf.nn.l2_normalize(direction_vec, -1)
 474   goal_vec = goals
 475   if normalize_goals:
 476     goal_vec = tf.nn.l2_normalize(goal_vec, -1)
 477 
 478   similarity = tf.reduce_sum(goal_vec * direction_vec, -1)
 479   discounts = tf.ones_like(similarity)
 480   return offset + tf.to_float(similarity), tf.to_float(discounts)

 483 @gin.configurable
 484 def diff_distance(states,
 485                   actions,
 486                   rewards,
 487                   next_states,
 488                   contexts,
 489                   state_scales=1.0,
 490                   goal_scales=1.0,
 491                   reward_scales=1.0,
 492                   weight_index=None,
 493                   weight_vector=None,
 494                   summarize=False,
 495                   termination_epsilon=1e-4,
 496                   state_indices=None,
 497                   goal_indices=None,
 498                   norm='L2',
 499                   epsilon=1e-10):
 500   """Returns the difference in euclidean distance between states/next_states and contexts.
 501 
 502   Args:
 503     states: A [batch_size, num_state_dims] Tensor representing a batch
 504         of states.
 505     actions: A [batch_size, num_action_dims] Tensor representing a batch
 506       of actions.
 507     rewards: A [batch_size] Tensor representing a batch of rewards.
 508     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 509       of next states.
 510     contexts: A list of [batch_size, num_context_dims] Tensor representing
 511       a batch of contexts.
 512     state_scales: multiplicative scale for (next) states. A scalar or 1D tensor,
 513       must be broadcastable to number of state dimensions.
 514     goal_scales: multiplicative scale for goals. A scalar or 1D tensor,
 515       must be broadcastable to number of goal dimensions.
 516     reward_scales: multiplicative scale for rewards. A scalar or 1D tensor,
 517       must be broadcastable to number of reward dimensions.
 518     weight_index: (integer) The context list index that specifies weight.
 519     weight_vector: (a number or a list or Numpy array) The weighting vector,
 520       broadcastable to `next_states`.
 521     summarize: (boolean) enable summary ops.
 522     termination_epsilon: terminate if dist is less than this quantity.
 523     state_indices: (a list of integers) list of state indices to select.
 524     goal_indices: (a list of integers) list of goal indices to select.
 525     vectorize: Return a vectorized form.
 526     norm: L1 or L2.
 527     epsilon: small offset to ensure non-negative/zero distance.
 528 
 529   Returns:
 530     A new tf.float32 [batch_size] rewards Tensor, and
 531       tf.float32 [batch_size] discounts tensor.
 532   """
 533   del actions, rewards  # Unused
 534   stats = {}
 535   record_tensor(next_states, state_indices, stats, 'next_states')
 536   next_states = index_states(next_states, state_indices)
 537   states = index_states(states, state_indices)
 538   goals = index_states(contexts[0], goal_indices)
 539   next_sq_dists = tf.squared_difference(next_states * state_scales,
 540                                         goals * goal_scales)
 541   sq_dists = tf.squared_difference(states * state_scales,
 542                                    goals * goal_scales)
 543   record_tensor(sq_dists, None, stats, 'sq_dists')
 544   if weight_vector is not None:
 545     next_sq_dists *= tf.convert_to_tensor(weight_vector, dtype=next_states.dtype)
 546     sq_dists *= tf.convert_to_tensor(weight_vector, dtype=next_states.dtype)
 547   if weight_index is not None:
 548     next_sq_dists *= contexts[weight_index]
 549     sq_dists *= contexts[weight_index]
 550   if norm == 'L1':
 551     next_dist = tf.sqrt(next_sq_dists + epsilon)
 552     dist = tf.sqrt(sq_dists + epsilon)
 553     next_dist = tf.reduce_sum(next_dist, -1)
 554     dist = tf.reduce_sum(dist, -1)
 555   elif norm == 'L2':
 556     next_dist = tf.reduce_sum(next_sq_dists, -1)
 557     next_dist = tf.sqrt(next_dist + epsilon)  # tf.gradients fails when tf.sqrt(-0.0)
 558     dist = tf.reduce_sum(sq_dists, -1)
 559     dist = tf.sqrt(dist + epsilon)  # tf.gradients fails when tf.sqrt(-0.0)
 560   else:
 561     raise NotImplementedError(norm)
 562   discounts = next_dist > termination_epsilon
 563   if summarize:
 564     with tf.name_scope('RewardFn/'):
 565       tf.summary.scalar('mean_dist', tf.reduce_mean(dist))
 566       tf.summary.histogram('dist', dist)
 567       summarize_stats(stats)
 568   diff = dist - next_dist
 569   diff *= reward_scales
 570   return tf.to_float(diff), tf.to_float(discounts)

 595       a batch of contexts.
 596     termination_epsilon: terminate if dist is less than this quantity.
 597     offset: Offset the rewards.
 598     epsilon: small offset to ensure non-negative/zero distance.
 599 
 600   Returns:
 601     A new tf.float32 [batch_size] rewards Tensor, and
 602       tf.float32 [batch_size] discounts tensor.
 603   """
 604   del states, actions  # unused args
 605   next_states = index_states(next_states, state_indices)
 606   dist = tf.reduce_sum(tf.squared_difference(next_states, contexts[0]), -1)
 607   dist = tf.sqrt(dist + epsilon)
 608   discounts = dist > termination_epsilon
 609   rewards = tf.logical_not(discounts)
 610   rewards = tf.to_float(rewards) + offset
 611   return tf.to_float(rewards), tf.ones_like(tf.to_float(discounts)) #tf.to_float(discounts)

 638 def ctrl_rewards(states,
 639                  actions,
 640                  rewards,
 641                  next_states,
 642                  contexts,
 643                  reward_scales=1.0):
 644   """Returns the negative control cost.
 645 
 646   Args:
 647     states: A [batch_size, num_state_dims] Tensor representing a batch
 648         of states.
 649     actions: A [batch_size, num_action_dims] Tensor representing a batch
 650       of actions.
 651     rewards: A [batch_size] Tensor representing a batch of rewards.
 652     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 653       of next states.
 654     contexts: A list of [batch_size, num_context_dims] Tensor representing
 655       a batch of contexts.
 656     reward_scales: multiplicative scale for rewards. A scalar or 1D tensor,
 657       must be broadcastable to number of reward dimensions.
 658 
 659   Returns:
 660     A new tf.float32 [batch_size] rewards Tensor, and
 661       tf.float32 [batch_size] discounts tensor.
 662   """
 663   del states, rewards, contexts  # Unused
 664   if actions is None:
 665     rewards = tf.to_float(tf.zeros(shape=next_states.shape[:1]))
 666   else:
 667     rewards = -tf.reduce_sum(tf.square(actions), axis=1)
 668     rewards *= reward_scales
 669     rewards = tf.to_float(rewards)
 670   return rewards, tf.ones_like(rewards)

 690 @gin.configurable
 691 def state_rewards(states,
 692                   actions,
 693                   rewards,
 694                   next_states,
 695                   contexts,
 696                   weight_index=None,
 697                   state_indices=None,
 698                   weight_vector=1.0,
 699                   offset_vector=0.0,
 700                   summarize=False):
 701   """Returns the rewards that are linear mapping of next_states.
 702 
 703   Args:
 704     states: A [batch_size, num_state_dims] Tensor representing a batch
 705         of states.
 706     actions: A [batch_size, num_action_dims] Tensor representing a batch
 707       of actions.
 708     rewards: A [batch_size] Tensor representing a batch of rewards.
 709     next_states: A [batch_size, num_state_dims] Tensor representing a batch
 710       of next states.
 711     contexts: A list of [batch_size, num_context_dims] Tensor representing
 712       a batch of contexts.
 713     weight_index: (integer) Index of contexts lists that specify weighting.
 714     state_indices: (a list of Numpy integer array) Indices of states dimensions
 715       to be mapped.
 716     weight_vector: (a number or a list or Numpy array) The weighting vector,
 717       broadcastable to `next_states`.
 718     offset_vector: (a number or a list of Numpy array) The off vector.
 719     summarize: (boolean) enable summary ops.
 720 
 721   Returns:
 722     A new tf.float32 [batch_size] rewards Tensor, and
 723       tf.float32 [batch_size] discounts tensor.
 724   """
 725   del states, actions, rewards  # unused args
 726   stats = {}
 727   record_tensor(next_states, state_indices, stats)
 728   next_states = index_states(next_states, state_indices)
 729   weight = tf.constant(
 730       weight_vector, dtype=next_states.dtype, shape=next_states[0].shape)
 731   weights = tf.expand_dims(weight, 0)
 732   offset = tf.constant(
 733       offset_vector, dtype=next_states.dtype, shape=next_states[0].shape)
 734   offsets = tf.expand_dims(offset, 0)
 735   if weight_index is not None:
 736     weights *= contexts[weight_index]
 737   rewards = tf.to_float(tf.reduce_sum(weights * (next_states+offsets), axis=1))
 738   if summarize:
 739     with tf.name_scope('RewardFn/'):
 740       summarize_stats(stats)
 741   return rewards, tf.ones_like(rewards)
