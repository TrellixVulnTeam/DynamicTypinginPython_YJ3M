  19 from __future__ import absolute_import
  20 from __future__ import division
  21 from __future__ import print_function

  23 import numpy as np
  24 import gin.tf

  26 
  27 @gin.configurable
  28 def gin_sparse_array(size, values, indices, fill_value=0):
  29   arr = np.zeros(size)
  30   arr.fill(fill_value)
  31   arr[indices] = values
  32   return arr
  33 
  34 
  35 @gin.configurable
  36 def gin_sum(values):
  37   result = values[0]
  38   for value in values[1:]:
  39     result += value
  40   return result
  41

  43 @gin.configurable
  44 def gin_range(n):
  45   return range(n)
