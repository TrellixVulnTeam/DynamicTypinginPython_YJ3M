  16 """Context for Universal Value Function agents.
  17 
  18 A context specifies a list of contextual variables, each with
  19   own sampling and reward computation methods.

  21 Examples of contextual variables include

  39   def __init__(self,
  40                tf_env,
  41                context_ranges=None,
  42                context_shapes=None,
  43                state_indices=None,
  44                variable_indices=None,
  45                gamma_index=None,
  46                settable_context=False,
  47                timers=None,
  48                samplers=None,
  49                reward_weights=None,
  50                reward_fn=None,
  51                random_sampler_mode='random',
  52                normalizers=None,
  53                context_transition_fn=None,
  54                context_multi_transition_fn=None,
  55                meta_action_every_n=None):
  56     self._tf_env = tf_env
  57     self.variable_indices = variable_indices
  58     self.gamma_index = gamma_index
  59     self._settable_context = settable_context
  60     self.timers = timers
  61     self._context_transition_fn = context_transition_fn
  62     self._context_multi_transition_fn = context_multi_transition_fn
  63     self._random_sampler_mode = random_sampler_mode
  64 
  65     # assign specs
  66     self._obs_spec = self._tf_env.observation_spec()
  67     self._context_shapes = tuple([
  68         shape if shape is not None else self._obs_spec.shape
  69         for shape in context_shapes
  70     ])
  71     self.context_specs = tuple([
  72         specs.TensorSpec(dtype=self._obs_spec.dtype, shape=shape)
  73         for shape in self._context_shapes
  74     ])
  75     if context_ranges is not None:
  76       self.context_ranges = context_ranges
  77     else:
  78       self.context_ranges = [None] * len(self._context_shapes)
  79 
  80     self.context_as_action_specs = tuple([
  81         specs.BoundedTensorSpec(
  82             shape=shape,
  83             dtype=(tf.float32 if self._obs_spec.dtype in
  84                    [tf.float32, tf.float64] else self._obs_spec.dtype),
  85             minimum=context_range[0],
  86             maximum=context_range[-1])
  87         for shape, context_range in zip(self._context_shapes, self.context_ranges)
  88     ])
  89 
  90     if state_indices is not None:
  91       self.state_indices = state_indices
  92     else:
  93       self.state_indices = [None] * len(self._context_shapes)
  94     if self.variable_indices is not None and self.n != len(
  95         self.variable_indices):
  96       raise ValueError(
  97           'variable_indices (%s) must have the same length as contexts (%s).' %
  98           (self.variable_indices, self.context_specs))
  99     assert self.n == len(self.context_ranges)
 100     assert self.n == len(self.state_indices)
 101 
 102     # assign reward/sampler fns
 103     self._sampler_fns = dict()
 104     self._samplers = dict()
 105     self._reward_fns = dict()
 106 
 107     # assign reward fns
 108     self._add_custom_reward_fns()
 109     reward_weights = reward_weights or None
 110     self._reward_fn = self._make_reward_fn(reward_fn, reward_weights)
 111 
 112     # assign samplers
 113     self._add_custom_sampler_fns()
 114     for mode, sampler_fns in samplers.items():
 115       self._make_sampler_fn(sampler_fns, mode)
 116 
 117     # create normalizers
 118     if normalizers is None:
 119       self._normalizers = [None] * len(self.context_specs)
 120     else:
 121       self._normalizers = [
 122           normalizer(tf.zeros(shape=spec.shape, dtype=spec.dtype))
 123           if normalizer is not None else None
 124           for normalizer, spec in zip(normalizers, self.context_specs)
 125       ]
 126     assert self.n == len(self._normalizers)
 127 
 128     self.meta_action_every_n = meta_action_every_n
 129 
 130     # create vars
 131     self.context_vars = {}
 132     self.timer_vars = {}
 133     self.create_vars(self.VAR_NAME)
 134     self.t = tf.Variable(
 135         tf.zeros(shape=(), dtype=tf.int32), name='num_timer_steps')

 143   def sample_random_contexts(self, batch_size):
 144     """Sample random batch contexts."""
 145     assert self._random_sampler_mode is not None
 146     return self.sample_contexts(self._random_sampler_mode, batch_size)[0]

 148   def sample_contexts(self, mode, batch_size, state=None, next_state=None,
 149                       **kwargs):
 150     """Sample a batch of contexts.
 151 
 152     Args:
 153       mode: A string representing the mode [`train`, `explore`, `eval`].
 154       batch_size: Batch size.
 155     Returns:
 156       Two lists of [batch_size, num_context_dims] contexts.
 157     """
 158     contexts, next_contexts = self._sampler_fns[mode](
 159         batch_size, state=state, next_state=next_state,
 160         **kwargs)
 161     self._validate_contexts(contexts)
 162     self._validate_contexts(next_contexts)
 163     return contexts, next_contexts

 165   def compute_rewards(self, mode, states, actions, rewards, next_states,
 166                       contexts):
 167     """Compute context-based rewards.
 168 
 169     Args:
 170       mode: A string representing the mode ['uvf', 'task'].
 171       states: A [batch_size, num_state_dims] tensor.
 172       actions: A [batch_size, num_action_dims] tensor.
 173       rewards: A [batch_size] tensor representing unmodified rewards.
 174       next_states: A [batch_size, num_state_dims] tensor.
 175       contexts: A list of [batch_size, num_context_dims] tensors.
 176     Returns:
 177       A [batch_size] tensor representing rewards.
 178     """
 179     return self._reward_fn(states, actions, rewards, next_states,
 180                            contexts)
 181 
 182   def _make_reward_fn(self, reward_fns_list, reward_weights):
 183     """Returns a fn that computes rewards.
 184 
 185     Args:
 186       reward_fns_list: A fn or a list of reward fns.
 187       mode: A string representing the operating mode.
 188       reward_weights: A list of reward weights.
 189     """
 190     if not isinstance(reward_fns_list, (list, tuple)):
 191       reward_fns_list = [reward_fns_list]
 192     if reward_weights is None:
 193       reward_weights = [1.0] * len(reward_fns_list)
 194     assert len(reward_fns_list) == len(reward_weights)
 195 
 196     reward_fns_list = [
 197         self._custom_reward_fns[fn] if isinstance(fn, (str,)) else fn
 198         for fn in reward_fns_list
 199     ]
 200 
 201     def reward_fn(*args, **kwargs):
 202       """Returns rewards, discounts."""
 203       reward_tuples = [
 204           reward_fn(*args, **kwargs) for reward_fn in reward_fns_list
 205       ]
 206       rewards_list = [reward_tuple[0] for reward_tuple in reward_tuples]
 207       discounts_list = [reward_tuple[1] for reward_tuple in reward_tuples]
 208       ndims = max([r.shape.ndims for r in rewards_list])
 209       if ndims > 1:  # expand reward shapes to allow broadcasting
 210         for i in range(len(rewards_list)):
 211           for _ in range(rewards_list[i].shape.ndims - ndims):
 212             rewards_list[i] = tf.expand_dims(rewards_list[i], axis=-1)
 213           for _ in range(discounts_list[i].shape.ndims - ndims):
 214             discounts_list[i] = tf.expand_dims(discounts_list[i], axis=-1)
 215       rewards = tf.add_n(
 216           [r * tf.to_float(w) for r, w in zip(rewards_list, reward_weights)])
 217       discounts = discounts_list[0]
 218       for d in discounts_list[1:]:
 219         discounts *= d
 220 
 221       return rewards, discounts
 222 
 223     return reward_fn

 225   def _make_sampler_fn(self, sampler_cls_list, mode):
 226     """Returns a fn that samples a list of context vars.
 227 
 228     Args:
 229       sampler_cls_list: A list of sampler classes.
 230       mode: A string representing the operating mode.
 231     """
 232     if not isinstance(sampler_cls_list, (list, tuple)):
 233       sampler_cls_list = [sampler_cls_list]
 234 
 235     self._samplers[mode] = []
 236     sampler_fns = []
 237     for spec, sampler in zip(self.context_specs, sampler_cls_list):
 238       if isinstance(sampler, (str,)):
 239         sampler_fn = self._custom_sampler_fns[sampler]
 240       else:
 241         sampler_fn = sampler(context_spec=spec)
 242         self._samplers[mode].append(sampler_fn)
 243       sampler_fns.append(sampler_fn)
 244 
 245     def batch_sampler_fn(batch_size, state=None, next_state=None, **kwargs):
 246       """Sampler fn."""
 247       contexts_tuples = [
 248           sampler(batch_size, state=state, next_state=next_state, **kwargs)
 249           for sampler in sampler_fns]
 250       contexts = [c[0] for c in contexts_tuples]
 251       next_contexts = [c[1] for c in contexts_tuples]
 252       contexts = [
 253           normalizer.update_apply(c) if normalizer is not None else c
 254           for normalizer, c in zip(self._normalizers, contexts)
 255       ]
 256       next_contexts = [
 257           normalizer.apply(c) if normalizer is not None else c
 258           for normalizer, c in zip(self._normalizers, next_contexts)
 259       ]
 260       return contexts, next_contexts
 261 
 262     self._sampler_fns[mode] = batch_sampler_fn

 264   def set_env_context_op(self, context, disable_unnormalizer=False):
 265     """Returns a TensorFlow op that sets the environment context.
 266 
 267     Args:
 268       context: A list of context Tensor variables.
 269       disable_unnormalizer: Disable unnormalization.
 270     Returns:
 271       A TensorFlow op that sets the environment context.
 272     """
 273     ret_val = np.array(1.0, dtype=np.float32)
 274     if not self._settable_context:
 275       return tf.identity(ret_val)
 276 
 277     if not disable_unnormalizer:
 278       context = [
 279           normalizer.unapply(tf.expand_dims(c, 0))[0]
 280           if normalizer is not None else c
 281           for normalizer, c in zip(self._normalizers, context)
 282       ]
 283 
 284     def set_context_func(*env_context_values):
 285       tf.logging.info('[set_env_context_op] Setting gym environment context.')
 286       # pylint: disable=protected-access
 287       self.gym_env.set_context(*env_context_values)
 288       return ret_val
 289       # pylint: enable=protected-access
 290 
 291     with tf.name_scope('set_env_context'):
 292       set_op = tf.py_func(set_context_func, context, tf.float32,
 293                           name='set_env_context_py_func')
 294       set_op.set_shape([])
 295     return set_op

 297   def set_replay(self, replay):
 298     """Set replay buffer for samplers.
 299 
 300     Args:
 301       replay: A replay buffer.
 302     """
 303     for _, samplers in self._samplers.items():
 304       for sampler in samplers:
 305         sampler.set_replay(replay)

 307   def get_clip_fns(self):
 308     """Returns a list of clip fns for contexts.
 309 
 310     Returns:
 311       A list of fns that clip context tensors.
 312     """
 313     clip_fns = []
 314     for context_range in self.context_ranges:
 315       def clip_fn(var_, range_=context_range):
 316         """Clip a tensor."""
 317         if range_ is None:
 318           clipped_var = tf.identity(var_)
 319         elif isinstance(range_[0], (int, long, float, list, np.ndarray)):
 320           clipped_var = tf.clip_by_value(
 321               var_,
 322               range_[0],
 323               range_[1],)
 324         else: raise NotImplementedError(range_)
 325         return clipped_var
 326       clip_fns.append(clip_fn)
 327     return clip_fns

 329   def _validate_contexts(self, contexts):
 330     """Validate if contexts have right specs.
 331 
 332     Args:
 333       contexts: A list of [batch_size, num_context_dim] tensors.
 334     Raises:
 335       ValueError: If shape or dtype mismatches that of spec.
 336     """
 337     for i, (context, spec) in enumerate(zip(contexts, self.context_specs)):
 338       if context[0].shape != spec.shape:
 339         raise ValueError('contexts[%d] has invalid shape %s wrt spec shape %s' %
 340                          (i, context[0].shape, spec.shape))
 341       if context.dtype != spec.dtype:
 342         raise ValueError('contexts[%d] has invalid dtype %s wrt spec dtype %s' %
 343                          (i, context.dtype, spec.dtype))

 350   def step(self, mode, agent=None, action_fn=None, **kwargs):
 351     """Returns [next_contexts..., next_timer] list of ops.
 352 
 353     Args:
 354       mode: a string representing the mode=[train, explore, eval].
 355       **kwargs: kwargs for context_transition_fn.
 356     Returns:
 357       a list of ops that set the context.
 358     """
 359     if agent is None:
 360       ops = []
 361       if self._context_transition_fn is not None:
 362         def sampler_fn():
 363           samples = self.sample_contexts(mode, 1)[0]
 364           return [s[0] for s in samples]
 365         values = self._context_transition_fn(self.vars, self.t, sampler_fn, **kwargs)
 366         ops += [tf.assign(var, value) for var, value in zip(self.vars, values)]
 367       ops.append(tf.assign_add(self.t, 1))  # increment timer
 368       return ops
 369     else:
 370       ops = agent.tf_context.step(mode, **kwargs)
 371       state = kwargs['state']
 372       next_state = kwargs['next_state']
 373       state_repr = kwargs['state_repr']
 374       next_state_repr = kwargs['next_state_repr']
 375       with tf.control_dependencies(ops):  # Step high level context before computing low level one.
 376         # Get the context transition function output.
 377         values = self._context_transition_fn(self.vars, self.t, None,
 378                                              state=state_repr,
 379                                              next_state=next_state_repr)
 380         # Select a new goal every C steps, otherwise use context transition.
 381         low_level_context = [
 382             tf.cond(tf.equal(self.t % self.meta_action_every_n, 0),
 383                     lambda: tf.cast(action_fn(next_state, context=None), tf.float32),
 384                     lambda: values)]
 385         ops = [tf.assign(var, value)
 386                for var, value in zip(self.vars, low_level_context)]
 387         with tf.control_dependencies(ops):
 388           return [tf.assign_add(self.t, 1)]  # increment timer
 389         return ops

 391   def reset(self, mode, agent=None, action_fn=None, state=None):
 392     """Returns ops that reset the context.
 393 
 394     Args:
 395       mode: a string representing the mode=[train, explore, eval].
 396     Returns:
 397       a list of ops that reset the context.
 398     """
 399     if agent is None:
 400       values = self.sample_contexts(mode=mode, batch_size=1)[0]
 401       if values is None:
 402         return []
 403       values = [value[0] for value in values]
 404       values[0] = uvf_utils.tf_print(
 405           values[0],
 406           values,
 407           message='context:reset, mode=%s' % mode,
 408           first_n=10,
 409           name='context:reset:%s' % mode)
 410       all_ops = []
 411       for _, context_vars in sorted(self.context_vars.items()):
 412         ops = [tf.assign(var, value) for var, value in zip(context_vars, values)]
 413       all_ops += ops
 414       all_ops.append(self.set_env_context_op(values))
 415       all_ops.append(tf.assign(self.t, 0))  # reset timer
 416       return all_ops
 417     else:
 418       ops = agent.tf_context.reset(mode)
 419       # NOTE: The code is currently written in such a way that the higher level
 420       # policy does not provide a low-level context until the second
 421       # observation.  Insead, we just zero-out low-level contexts.
 422       for key, context_vars in sorted(self.context_vars.items()):
 423         ops += [tf.assign(var, tf.zeros_like(var)) for var, meta_var in
 424                 zip(context_vars, agent.tf_context.context_vars[key])]
 425 
 426       ops.append(tf.assign(self.t, 0))  # reset timer
 427       return ops

 429   def create_vars(self, name, agent=None):
 430     """Create tf variables for contexts.
 431 
 432     Args:
 433       name: Name of the variables.
 434     Returns:
 435       A list of [num_context_dims] tensors.
 436     """
 437     if agent is not None:
 438       meta_vars = agent.create_vars(name)
 439     else:
 440       meta_vars = {}
 441     assert name not in self.context_vars, ('Conflict! %s is already '
 442                                            'initialized.') % name
 443     self.context_vars[name] = tuple([
 444         tf.Variable(
 445             tf.zeros(shape=spec.shape, dtype=spec.dtype),
 446             name='%s_context_%d' % (name, i))
 447         for i, spec in enumerate(self.context_specs)
 448     ])
 449     return self.context_vars[name], meta_vars
