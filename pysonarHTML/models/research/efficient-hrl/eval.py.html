  48 def get_evaluate_checkpoint_fn(master, output_dir, eval_step_fns,
  49                                model_rollout_fn, gamma, max_steps_per_episode,
  50                                num_episodes_eval, num_episodes_videos,
  51                                tuner_hook, generate_videos,
  52                                generate_summaries, video_settings):
  53   """Returns a function that evaluates a given checkpoint.
  54 
  55   Args:
  56     master: BNS name of the TensorFlow master
  57     output_dir: The output directory to which the metric summaries are written.
  58     eval_step_fns: A dictionary of a functions that return a list of
  59       [state, action, reward, discount, transition_type] tensors,
  60       indexed by summary tag name.
  61     model_rollout_fn: Model rollout fn.
  62     gamma: Discount factor for the reward.
  63     max_steps_per_episode: Maximum steps to run each episode for.
  64     num_episodes_eval: Number of episodes to evaluate and average reward over.
  65     num_episodes_videos: Number of episodes to record for video.
  66     tuner_hook: A callable(average reward, global step) that updates a Vizier
  67       tuner trial.
  68     generate_videos: Whether to generate videos of the agent in action.
  69     generate_summaries: Whether to generate summaries.
  70     video_settings: Settings for generating videos of the agent.
  71 
  72   Returns:
  73     A function that evaluates a checkpoint.
  74   """
  75   sess = tf.Session(master, graph=tf.get_default_graph())
  76   sess.run(tf.global_variables_initializer())
  77   sess.run(tf.local_variables_initializer())
  78   summary_writer = tf.summary.FileWriter(output_dir)
  79 
  80   def evaluate_checkpoint(checkpoint_path):
  81     """Performs a one-time evaluation of the given checkpoint.
  82 
  83     Args:
  84       checkpoint_path: Checkpoint to evaluate.
  85     Returns:
  86       True if the evaluation process should stop
  87     """
  88     restore_fn = tf.contrib.framework.assign_from_checkpoint_fn(
  89         checkpoint_path,
  90         uvf_utils.get_all_vars(),
  91         ignore_missing_vars=True,
  92         reshape_variables=False)
  93     assert restore_fn is not None, 'cannot restore %s' % checkpoint_path
  94     restore_fn(sess)
  95     global_step = sess.run(slim.get_global_step())
  96     should_stop = False
  97     max_reward = -1e10
  98     max_meta_reward = -1e10
  99 
 100     for eval_tag, (eval_step, env_base,) in sorted(eval_step_fns.items()):
 101       if hasattr(env_base, 'set_sess'):
 102         env_base.set_sess(sess)  # set session
 103 
 104       if generate_summaries:
 105         tf.logging.info(
 106             '[%s] Computing average reward over %d episodes at global step %d.',
 107             eval_tag, num_episodes_eval, global_step)
 108         (average_reward, last_reward,
 109          average_meta_reward, last_meta_reward, average_success,
 110          states, actions) = eval_utils.compute_average_reward(
 111              sess, env_base, eval_step, gamma, max_steps_per_episode,
 112              num_episodes_eval)
 113         tf.logging.info('[%s] Average reward = %f', eval_tag, average_reward)
 114         tf.logging.info('[%s] Last reward = %f', eval_tag, last_reward)
 115         tf.logging.info('[%s] Average meta reward = %f', eval_tag, average_meta_reward)
 116         tf.logging.info('[%s] Last meta reward = %f', eval_tag, last_meta_reward)
 117         tf.logging.info('[%s] Average success = %f', eval_tag, average_success)
 118         if model_rollout_fn is not None:
 119           preds, model_losses = eval_utils.compute_model_loss(
 120               sess, model_rollout_fn, states, actions)
 121           for i, (pred, state, model_loss) in enumerate(
 122               zip(preds, states, model_losses)):
 123             tf.logging.info('[%s] Model rollout step %d: loss=%f', eval_tag, i,
 124                             model_loss)
 125             tf.logging.info('[%s] Model rollout step %d: pred=%s', eval_tag, i,
 126                             str(pred.tolist()))
 127             tf.logging.info('[%s] Model rollout step %d: state=%s', eval_tag, i,
 128                             str(state.tolist()))
 129 
 130         # Report the eval stats to the tuner.
 131         if average_reward > max_reward:
 132           max_reward = average_reward
 133         if average_meta_reward > max_meta_reward:
 134           max_meta_reward = average_meta_reward
 135 
 136         for (tag, value) in [('Reward/average_%s_reward', average_reward),
 137                              ('Reward/last_%s_reward', last_reward),
 138                              ('Reward/average_%s_meta_reward', average_meta_reward),
 139                              ('Reward/last_%s_meta_reward', last_meta_reward),
 140                              ('Reward/average_%s_success', average_success)]:
 141           summary_str = tf.Summary(value=[
 142               tf.Summary.Value(
 143                   tag=tag % eval_tag,
 144                   simple_value=value)
 145           ])
 146           summary_writer.add_summary(summary_str, global_step)
 147           summary_writer.flush()
 148 
 149       if generate_videos or should_stop:
 150         # Do a manual reset before generating the video to see the initial
 151         # pose of the robot, towards which the reset controller is moving.
 152         if hasattr(env_base, '_gym_env'):
 153           tf.logging.info('Resetting before recording video')
 154           if hasattr(env_base._gym_env, 'reset_model'):
 155             env_base._gym_env.reset_model()  # pylint: disable=protected-access
 156           else:
 157             env_base._gym_env.wrapped_env.reset_model()
 158         video_filename = os.path.join(output_dir, 'videos',
 159                                       '%s_step_%d.mp4' % (eval_tag,
 160                                                           global_step))
 161         eval_utils.capture_video(sess, eval_step, env_base,
 162                                 max_steps_per_episode * num_episodes_videos,
 163                                 video_filename, video_settings,
 164                                 reset_every=max_steps_per_episode)
 165 
 166       should_stop = should_stop or (generate_summaries and tuner_hook and
 167                                     tuner_hook(max_reward, global_step))
 168     return bool(should_stop)
 169 
 170   return evaluate_checkpoint

 173 def get_model_rollout(uvf_agent, tf_env):
 174   """Model rollout function."""
 175   state_spec = tf_env.observation_spec()[0]
 176   action_spec = tf_env.action_spec()[0]
 177   state_ph = tf.placeholder(dtype=state_spec.dtype, shape=state_spec.shape)
 178   action_ph = tf.placeholder(dtype=action_spec.dtype, shape=action_spec.shape)
 179 
 180   merged_state = uvf_agent.merged_state(state_ph)
 181   diff_value = uvf_agent.critic_net(tf.expand_dims(merged_state, 0),
 182                                     tf.expand_dims(action_ph, 0))[0]
 183   diff_value = tf.cast(diff_value, dtype=state_ph.dtype)
 184   state_ph.shape.assert_is_compatible_with(diff_value.shape)
 185   next_state = state_ph + diff_value
 186 
 187   def model_rollout_fn(sess, state, action):
 188     return sess.run(next_state, feed_dict={state_ph: state, action_ph: action})
 189 
 190   return model_rollout_fn

 193 def get_eval_step(uvf_agent,
 194                   state_preprocess,
 195                   tf_env,
 196                   action_fn,
 197                   meta_action_fn,
 198                   environment_steps,
 199                   num_episodes,
 200                   mode='eval'):
 201   """Get one-step policy/env stepping ops.
 202 
 203   Args:
 204     uvf_agent: A UVF agent.
 205     tf_env: A TFEnvironment.
 206     action_fn: A function to produce actions given current state.
 207     meta_action_fn: A function to produce meta actions given current state.
 208     environment_steps: A variable to count the number of steps in the tf_env.
 209     num_episodes: A variable to count the number of episodes.
 210     mode: a string representing the mode=[train, explore, eval].
 211 
 212   Returns:
 213     A collect_experience_op that excute an action and store into the
 214     replay_buffer
 215   """
 216 
 217   tf_env.start_collect()
 218   state = tf_env.current_obs()
 219   action = action_fn(state, context=None)
 220   state_repr = state_preprocess(state)
 221 
 222   action_spec = tf_env.action_spec()
 223   action_ph = tf.placeholder(dtype=action_spec.dtype, shape=action_spec.shape)
 224   with tf.control_dependencies([state]):
 225     transition_type, reward, discount = tf_env.step(action_ph)
 226 
 227   def increment_step():
 228     return environment_steps.assign_add(1)
 229 
 230   def increment_episode():
 231     return num_episodes.assign_add(1)
 232 
 233   def no_op_int():
 234     return tf.constant(0, dtype=tf.int64)
 235 
 236   step_cond = uvf_agent.step_cond_fn(state, action,
 237                                      transition_type,
 238                                      environment_steps, num_episodes)
 239   reset_episode_cond = uvf_agent.reset_episode_cond_fn(
 240       state, action,
 241       transition_type, environment_steps, num_episodes)
 242   reset_env_cond = uvf_agent.reset_env_cond_fn(state, action,
 243                                                transition_type,
 244                                                environment_steps, num_episodes)
 245 
 246   increment_step_op = tf.cond(step_cond, increment_step, no_op_int)
 247   with tf.control_dependencies([increment_step_op]):
 248     increment_episode_op = tf.cond(reset_episode_cond, increment_episode,
 249                                    no_op_int)
 250 
 251   with tf.control_dependencies([reward, discount]):
 252     next_state = tf_env.current_obs()
 253     next_state_repr = state_preprocess(next_state)
 254 
 255   with tf.control_dependencies([increment_episode_op]):
 256     post_reward, post_meta_reward = uvf_agent.cond_begin_episode_op(
 257         tf.logical_not(reset_episode_cond),
 258         [state, action_ph, reward, next_state,
 259          state_repr, next_state_repr],
 260         mode=mode, meta_action_fn=meta_action_fn)
 261 
 262   # Important: do manual reset after getting the final reward from the
 263   # unreset environment.
 264   with tf.control_dependencies([post_reward, post_meta_reward]):
 265     cond_reset_op = tf.cond(reset_env_cond,
 266                             tf_env.reset,
 267                             tf_env.current_time_step)
 268 
 269   # Add a dummy control dependency to force the reset_op to run
 270   with tf.control_dependencies(cond_reset_op):
 271     post_reward, post_meta_reward = map(tf.identity, [post_reward, post_meta_reward])
 272 
 273   eval_step = [next_state, action_ph, transition_type, post_reward, post_meta_reward, discount, uvf_agent.context_vars, state_repr]
 274 
 275   if callable(action):
 276     def step_fn(sess):
 277       action_value = action(sess)
 278       return sess.run(eval_step, feed_dict={action_ph: action_value})
 279   else:
 280     action = uvf_utils.clip_to_spec(action, action_spec)
 281     def step_fn(sess):
 282       action_value = sess.run(action)
 283       return sess.run(eval_step, feed_dict={action_ph: action_value})
 284 
 285   return step_fn

 288 @gin.configurable
 289 def evaluate(checkpoint_dir,
 290              eval_dir,
 291              environment=None,
 292              num_bin_actions=3,
 293              agent_class=None,
 294              meta_agent_class=None,
 295              state_preprocess_class=None,
 296              gamma=1.0,
 297              num_episodes_eval=10,
 298              eval_interval_secs=60,
 299              max_number_of_evaluations=None,
 300              checkpoint_timeout=None,
 301              timeout_fn=None,
 302              tuner_hook=None,
 303              generate_videos=False,
 304              generate_summaries=True,
 305              num_episodes_videos=5,
 306              video_settings=None,
 307              eval_modes=('eval',),
 308              eval_model_rollout=False,
 309              policy_save_dir='policy',
 310              checkpoint_range=None,
 311              checkpoint_path=None,
 312              max_steps_per_episode=None,
 313              evaluate_nohrl=False):
 314   """Loads and repeatedly evaluates a checkpointed model at a set interval.
 315 
 316   Args:
 317     checkpoint_dir: The directory where the checkpoints reside.
 318     eval_dir: Directory to save the evaluation summary results.
 319     environment: A BaseEnvironment to evaluate.
 320     num_bin_actions: Number of bins for discretizing continuous actions.
 321     agent_class: An RL agent class.
 322     meta_agent_class: A Meta agent class.
 323     gamma: Discount factor for the reward.
 324     num_episodes_eval: Number of episodes to evaluate and average reward over.
 325     eval_interval_secs: The number of seconds between each evaluation run.
 326     max_number_of_evaluations: The max number of evaluations. If None the
 327       evaluation continues indefinitely.
 328     checkpoint_timeout: The maximum amount of time to wait between checkpoints.
 329       If left as `None`, then the process will wait indefinitely.
 330     timeout_fn: Optional function to call after a timeout.
 331     tuner_hook: A callable that takes the average reward and global step and
 332       updates a Vizier tuner trial.
 333     generate_videos: Whether to generate videos of the agent in action.
 334     generate_summaries: Whether to generate summaries.
 335     num_episodes_videos: Number of episodes to evaluate for generating videos.
 336     video_settings: Settings for generating videos of the agent.
 337       optimal action based on the critic.
 338     eval_modes: A tuple of eval modes.
 339     eval_model_rollout: Evaluate model rollout.
 340     policy_save_dir: Optional sub-directory where the policies are
 341       saved.
 342     checkpoint_range: Optional. If provided, evaluate all checkpoints in
 343       the range.
 344     checkpoint_path: Optional sub-directory specifying which checkpoint to
 345       evaluate. If None, will evaluate the most recent checkpoint.
 346   """
 347   tf_env = create_maze_env.TFPyEnvironment(environment)
 348   observation_spec = [tf_env.observation_spec()]
 349   action_spec = [tf_env.action_spec()]
 350 
 351   assert max_steps_per_episode, 'max_steps_per_episode need to be set'
 352 
 353   if agent_class.ACTION_TYPE == 'discrete':
 354     assert False
 355   else:
 356     assert agent_class.ACTION_TYPE == 'continuous'
 357 
 358   if meta_agent_class is not None:
 359     assert agent_class.ACTION_TYPE == meta_agent_class.ACTION_TYPE
 360     with tf.variable_scope('meta_agent'):
 361       meta_agent = meta_agent_class(
 362         observation_spec,
 363         action_spec,
 364         tf_env,
 365       )
 366   else:
 367     meta_agent = None
 368 
 369   with tf.variable_scope('uvf_agent'):
 370     uvf_agent = agent_class(
 371         observation_spec,
 372         action_spec,
 373         tf_env,
 374     )
 375     uvf_agent.set_meta_agent(agent=meta_agent)
 376 
 377   with tf.variable_scope('state_preprocess'):
 378     state_preprocess = state_preprocess_class()
 379 
 380   # run both actor and critic once to ensure networks are initialized
 381   # and gin configs will be saved
 382   # pylint: disable=protected-access
 383   temp_states = tf.expand_dims(
 384       tf.zeros(
 385           dtype=uvf_agent._observation_spec.dtype,
 386           shape=uvf_agent._observation_spec.shape), 0)
 387   # pylint: enable=protected-access
 388   temp_actions = uvf_agent.actor_net(temp_states)
 389   uvf_agent.critic_net(temp_states, temp_actions)
 390 
 391   # create eval_step_fns for each action function
 392   eval_step_fns = dict()
 393   meta_agent = uvf_agent.meta_agent
 394   for meta in [True] + [False] * evaluate_nohrl:
 395     meta_tag = 'hrl' if meta else 'nohrl'
 396     uvf_agent.set_meta_agent(meta_agent if meta else None)
 397     for mode in eval_modes:
 398       # wrap environment
 399       wrapped_environment = uvf_agent.get_env_base_wrapper(
 400           environment, mode=mode)
 401       action_wrapper = lambda agent_: agent_.action
 402       action_fn = action_wrapper(uvf_agent)
 403       meta_action_fn = action_wrapper(meta_agent)
 404       eval_step_fns['%s_%s' % (mode, meta_tag)] = (get_eval_step(
 405           uvf_agent=uvf_agent,
 406           state_preprocess=state_preprocess,
 407           tf_env=tf_env,
 408           action_fn=action_fn,
 409           meta_action_fn=meta_action_fn,
 410           environment_steps=tf.Variable(
 411               0, dtype=tf.int64, name='environment_steps'),
 412           num_episodes=tf.Variable(0, dtype=tf.int64, name='num_episodes'),
 413           mode=mode), wrapped_environment,)
 414 
 415   model_rollout_fn = None
 416   if eval_model_rollout:
 417     model_rollout_fn = get_model_rollout(uvf_agent, tf_env)
 418 
 419   tf.train.get_or_create_global_step()
 420 
 421   if policy_save_dir:
 422     checkpoint_dir = os.path.join(checkpoint_dir, policy_save_dir)
 423 
 424   tf.logging.info('Evaluating policies at %s', checkpoint_dir)
 425   tf.logging.info('Running episodes for max %d steps', max_steps_per_episode)
 426 
 427   evaluate_checkpoint_fn = get_evaluate_checkpoint_fn(
 428       '', eval_dir, eval_step_fns, model_rollout_fn, gamma,
 429       max_steps_per_episode, num_episodes_eval, num_episodes_videos, tuner_hook,
 430       generate_videos, generate_summaries, video_settings)
 431 
 432   if checkpoint_path is not None:
 433     checkpoint_path = os.path.join(checkpoint_dir, checkpoint_path)
 434     evaluate_checkpoint_fn(checkpoint_path)
 435   elif checkpoint_range is not None:
 436     model_files = tf.gfile.Glob(
 437         os.path.join(checkpoint_dir, 'model.ckpt-*.index'))
 438     tf.logging.info('Found %s policies at %s', len(model_files), checkpoint_dir)
 439     model_files = {
 440         int(f.split('model.ckpt-', 1)[1].split('.', 1)[0]):
 441         os.path.splitext(f)[0]
 442         for f in model_files
 443     }
 444     model_files = {
 445         k: v
 446         for k, v in model_files.items()
 447         if k >= checkpoint_range[0] and k <= checkpoint_range[1]
 448     }
 449     tf.logging.info('Evaluating %d policies at %s',
 450                     len(model_files), checkpoint_dir)
 451     for _, checkpoint_path in sorted(model_files.items()):
 452       evaluate_checkpoint_fn(checkpoint_path)
 453   else:
 454     eval_utils.evaluate_checkpoint_repeatedly(
 455         checkpoint_dir,
 456         evaluate_checkpoint_fn,
 457         eval_interval_secs=eval_interval_secs,
 458         max_number_of_evaluations=max_number_of_evaluations,
 459         checkpoint_timeout=checkpoint_timeout,
 460         timeout_fn=timeout_fn)
