  25 def q_inv(a):
  26   return [a[0], -a[1], -a[2], -a[3]]
  27 
  28 
  29 def q_mult(a, b): # multiply two quaternion
  30   w = a[0] * b[0] - a[1] * b[1] - a[2] * b[2] - a[3] * b[3]
  31   i = a[0] * b[1] + a[1] * b[0] + a[2] * b[3] - a[3] * b[2]
  32   j = a[0] * b[2] - a[1] * b[3] + a[2] * b[0] + a[3] * b[1]
  33   k = a[0] * b[3] + a[1] * b[2] - a[2] * b[1] + a[3] * b[0]
  34   return [w, i, j, k]

  37 class AntEnv(mujoco_env.MujocoEnv, utils.EzPickle):
  38   FILE = "ant.xml"
  39   ORI_IND = 3

  41   def __init__(self, file_path=None, expose_all_qpos=True,
  42                expose_body_coms=None, expose_body_comvels=None):
  43     self._expose_all_qpos = expose_all_qpos
  44     self._expose_body_coms = expose_body_coms
  45     self._expose_body_comvels = expose_body_comvels
  46     self._body_com_indices = {}
  47     self._body_comvel_indices = {}
  48 
  49     mujoco_env.MujocoEnv.__init__(self, file_path, 5)
  50     utils.EzPickle.__init__(self)
  51 
  52   @property
  53   def physics(self):
  54     # check mujoco version is greater than version 1.50 to call correct physics
  55     # model containing PyMjData object for getting and setting position/velocity
  56     # check https://github.com/openai/mujoco-py/issues/80 for updates to api
  57     if mujoco_py.get_version() >= '1.50':
  58       return self.sim
  59     else:
  60       return self.model
  61 
  62   def _step(self, a):
  63     return self.step(a)
  64 
  65   def step(self, a):
  66     xposbefore = self.get_body_com("torso")[0]
  67     self.do_simulation(a, self.frame_skip)
  68     xposafter = self.get_body_com("torso")[0]
  69     forward_reward = (xposafter - xposbefore) / self.dt
  70     ctrl_cost = .5 * np.square(a).sum()
  71     survive_reward = 1.0
  72     reward = forward_reward - ctrl_cost + survive_reward
  73     state = self.state_vector()
  74     done = False
  75     ob = self._get_obs()
  76     return ob, reward, done, dict(
  77         reward_forward=forward_reward,
  78         reward_ctrl=-ctrl_cost,
  79         reward_survive=survive_reward)
  80 
  81   def _get_obs(self):
  82     # No cfrc observation
  83     if self._expose_all_qpos:
  84       obs = np.concatenate([
  85           self.physics.data.qpos.flat[:15],  # Ensures only ant obs.
  86           self.physics.data.qvel.flat[:14],
  87       ])
  88     else:
  89       obs = np.concatenate([
  90           self.physics.data.qpos.flat[2:15],
  91           self.physics.data.qvel.flat[:14],
  92       ])
  93 
  94     if self._expose_body_coms is not None:
  95       for name in self._expose_body_coms:
  96         com = self.get_body_com(name)
  97         if name not in self._body_com_indices:
  98           indices = range(len(obs), len(obs) + len(com))
  99           self._body_com_indices[name] = indices
 100         obs = np.concatenate([obs, com])
 101 
 102     if self._expose_body_comvels is not None:
 103       for name in self._expose_body_comvels:
 104         comvel = self.get_body_comvel(name)
 105         if name not in self._body_comvel_indices:
 106           indices = range(len(obs), len(obs) + len(comvel))
 107           self._body_comvel_indices[name] = indices
 108         obs = np.concatenate([obs, comvel])
 109     return obs

 111   def reset_model(self):
 112     qpos = self.init_qpos + self.np_random.uniform(
 113         size=self.model.nq, low=-.1, high=.1)
 114     qvel = self.init_qvel + self.np_random.randn(self.model.nv) * .1
 115 
 116     # Set everything other than ant to original position and 0 velocity.
 117     qpos[15:] = self.init_qpos[15:]
 118     qvel[14:] = 0.
 119     self.set_state(qpos, qvel)
 120     return self._get_obs()
 121 
 122   def viewer_setup(self):
 123     self.viewer.cam.distance = self.model.stat.extent * 0.5
 124 
 125   def get_ori(self):
 126     ori = [0, 1, 0, 0]
 127     rot = self.physics.data.qpos[self.__class__.ORI_IND:self.__class__.ORI_IND + 4]  # take the quaternion
 128     ori = q_mult(q_mult(rot, ori), q_inv(rot))[1:3]  # project onto x-y plane
 129     ori = math.atan2(ori[1], ori[0])
 130     return ori
 131 
 132   def set_xy(self, xy):
 133     qpos = np.copy(self.physics.data.qpos)
 134     qpos[0] = xy[0]
 135     qpos[1] = xy[1]
 136 
 137     qvel = self.physics.data.qvel
 138     self.set_state(qpos, qvel)
 139 
 140   def get_xy(self):
 141     return self.physics.data.qpos[:2]
