  37   def __init__(
  38       self,
  39       maze_id=None,
  40       maze_height=0.5,
  41       maze_size_scaling=8,
  42       n_bins=0,
  43       sensor_range=3.,
  44       sensor_span=2 * math.pi,
  45       observe_blocks=False,
  46       put_spin_near_agent=False,
  47       top_down_view=False,
  48       manual_collision=False,
  49       *args,
  50       **kwargs):
  51     self._maze_id = maze_id
  52 
  53     model_cls = self.__class__.MODEL_CLASS
  54     if model_cls is None:
  55       raise "MODEL_CLASS unspecified!"
  56     xml_path = os.path.join(MODEL_DIR, model_cls.FILE)
  57     tree = ET.parse(xml_path)
  58     worldbody = tree.find(".//worldbody")
  59 
  60     self.MAZE_HEIGHT = height = maze_height
  61     self.MAZE_SIZE_SCALING = size_scaling = maze_size_scaling
  62     self._n_bins = n_bins
  63     self._sensor_range = sensor_range * size_scaling
  64     self._sensor_span = sensor_span
  65     self._observe_blocks = observe_blocks
  66     self._put_spin_near_agent = put_spin_near_agent
  67     self._top_down_view = top_down_view
  68     self._manual_collision = manual_collision
  69 
  70     self.MAZE_STRUCTURE = structure = maze_env_utils.construct_maze(maze_id=self._maze_id)
  71     self.elevated = any(-1 in row for row in structure)  # Elevate the maze to allow for falling.
  72     self.blocks = any(
  73         any(maze_env_utils.can_move(r) for r in row)
  74         for row in structure)  # Are there any movable blocks?
  75 
  76     torso_x, torso_y = self._find_robot()
  77     self._init_torso_x = torso_x
  78     self._init_torso_y = torso_y
  79     self._init_positions = [
  80         (x - torso_x, y - torso_y)
  81         for x, y in self._find_all_robots()]
  82 
  83     self._xy_to_rowcol = lambda x, y: (2 + (y + size_scaling / 2) / size_scaling,
  84                                        2 + (x + size_scaling / 2) / size_scaling)
  85     self._view = np.zeros([5, 5, 3])  # walls (immovable), chasms (fall), movable blocks
  86 
  87     height_offset = 0.
  88     if self.elevated:
  89       # Increase initial z-pos of ant.
  90       height_offset = height * size_scaling
  91       torso = tree.find(".//body[@name='torso']")
  92       torso.set('pos', '0 0 %.2f' % (0.75 + height_offset))
  93     if self.blocks:
  94       # If there are movable blocks, change simulation settings to perform
  95       # better contact detection.
  96       default = tree.find(".//default")
  97       default.find('.//geom').set('solimp', '.995 .995 .01')
  98 
  99     self.movable_blocks = []
 100     for i in range(len(structure)):
 101       for j in range(len(structure[0])):
 102         struct = structure[i][j]
 103         if struct == 'r' and self._put_spin_near_agent:
 104           struct = maze_env_utils.Move.SpinXY
 105         if self.elevated and struct not in [-1]:
 106           # Create elevated platform.
 107           ET.SubElement(
 108               worldbody, "geom",
 109               name="elevated_%d_%d" % (i, j),
 110               pos="%f %f %f" % (j * size_scaling - torso_x,
 111                                 i * size_scaling - torso_y,
 112                                 height / 2 * size_scaling),
 113               size="%f %f %f" % (0.5 * size_scaling,
 114                                  0.5 * size_scaling,
 115                                  height / 2 * size_scaling),
 116               type="box",
 117               material="",
 118               contype="1",
 119               conaffinity="1",
 120               rgba="0.9 0.9 0.9 1",
 121           )
 122         if struct == 1:  # Unmovable block.
 123           # Offset all coordinates so that robot starts at the origin.
 124           ET.SubElement(
 125               worldbody, "geom",
 126               name="block_%d_%d" % (i, j),
 127               pos="%f %f %f" % (j * size_scaling - torso_x,
 128                                 i * size_scaling - torso_y,
 129                                 height_offset +
 130                                 height / 2 * size_scaling),
 131               size="%f %f %f" % (0.5 * size_scaling,
 132                                  0.5 * size_scaling,
 133                                  height / 2 * size_scaling),
 134               type="box",
 135               material="",
 136               contype="1",
 137               conaffinity="1",
 138               rgba="0.4 0.4 0.4 1",
 139           )
 140         elif maze_env_utils.can_move(struct):  # Movable block.
 141           # The "falling" blocks are shrunk slightly and increased in mass to
 142           # ensure that it can fall easily through a gap in the platform blocks.
 143           name = "movable_%d_%d" % (i, j)
 144           self.movable_blocks.append((name, struct))
 145           falling = maze_env_utils.can_move_z(struct)
 146           spinning = maze_env_utils.can_spin(struct)
 147           x_offset = 0.25 * size_scaling if spinning else 0.0
 148           y_offset = 0.0
 149           shrink = 0.1 if spinning else 0.99 if falling else 1.0
 150           height_shrink = 0.1 if spinning else 1.0
 151           movable_body = ET.SubElement(
 152               worldbody, "body",
 153               name=name,
 154               pos="%f %f %f" % (j * size_scaling - torso_x + x_offset,
 155                                 i * size_scaling - torso_y + y_offset,
 156                                 height_offset +
 157                                 height / 2 * size_scaling * height_shrink),
 158           )
 159           ET.SubElement(
 160               movable_body, "geom",
 161               name="block_%d_%d" % (i, j),
 162               pos="0 0 0",
 163               size="%f %f %f" % (0.5 * size_scaling * shrink,
 164                                  0.5 * size_scaling * shrink,
 165                                  height / 2 * size_scaling * height_shrink),
 166               type="box",
 167               material="",
 168               mass="0.001" if falling else "0.0002",
 169               contype="1",
 170               conaffinity="1",
 171               rgba="0.9 0.1 0.1 1"
 172           )
 173           if maze_env_utils.can_move_x(struct):
 174             ET.SubElement(
 175                 movable_body, "joint",
 176                 armature="0",
 177                 axis="1 0 0",
 178                 damping="0.0",
 179                 limited="true" if falling else "false",
 180                 range="%f %f" % (-size_scaling, size_scaling),
 181                 margin="0.01",
 182                 name="movable_x_%d_%d" % (i, j),
 183                 pos="0 0 0",
 184                 type="slide"
 185             )
 186           if maze_env_utils.can_move_y(struct):
 187             ET.SubElement(
 188                 movable_body, "joint",
 189                 armature="0",
 190                 axis="0 1 0",
 191                 damping="0.0",
 192                 limited="true" if falling else "false",
 193                 range="%f %f" % (-size_scaling, size_scaling),
 194                 margin="0.01",
 195                 name="movable_y_%d_%d" % (i, j),
 196                 pos="0 0 0",
 197                 type="slide"
 198             )
 199           if maze_env_utils.can_move_z(struct):
 200             ET.SubElement(
 201                 movable_body, "joint",
 202                 armature="0",
 203                 axis="0 0 1",
 204                 damping="0.0",
 205                 limited="true",
 206                 range="%f 0" % (-height_offset),
 207                 margin="0.01",
 208                 name="movable_z_%d_%d" % (i, j),
 209                 pos="0 0 0",
 210                 type="slide"
 211             )
 212           if maze_env_utils.can_spin(struct):
 213             ET.SubElement(
 214                 movable_body, "joint",
 215                 armature="0",
 216                 axis="0 0 1",
 217                 damping="0.0",
 218                 limited="false",
 219                 name="spinable_%d_%d" % (i, j),
 220                 pos="0 0 0",
 221                 type="ball"
 222             )
 223 
 224     torso = tree.find(".//body[@name='torso']")
 225     geoms = torso.findall(".//geom")
 226     for geom in geoms:
 227       if 'name' not in geom.attrib:
 228         raise Exception("Every geom of the torso must have a name "
 229                         "defined")
 230 
 231     _, file_path = tempfile.mkstemp(text=True, suffix='.xml')
 232     tree.write(file_path)
 233 
 234     self.wrapped_env = model_cls(*args, file_path=file_path, **kwargs)

 239   def get_top_down_view(self):
 240     self._view = np.zeros_like(self._view)
 241 
 242     def valid(row, col):
 243       return self._view.shape[0] > row >= 0 and self._view.shape[1] > col >= 0
 244 
 245     def update_view(x, y, d, row=None, col=None):
 246       if row is None or col is None:
 247         x = x - self._robot_x
 248         y = y - self._robot_y
 249         th = self._robot_ori
 250 
 251         row, col = self._xy_to_rowcol(x, y)
 252         update_view(x, y, d, row=row, col=col)
 253         return
 254 
 255       row, row_frac, col, col_frac = int(row), row % 1, int(col), col % 1
 256       if row_frac < 0:
 257         row_frac += 1
 258       if col_frac < 0:
 259         col_frac += 1
 260 
 261       if valid(row, col):
 262         self._view[row, col, d] += (
 263             (min(1., row_frac + 0.5) - max(0., row_frac - 0.5)) *
 264             (min(1., col_frac + 0.5) - max(0., col_frac - 0.5)))
 265       if valid(row - 1, col):
 266         self._view[row - 1, col, d] += (
 267             (max(0., 0.5 - row_frac)) *
 268             (min(1., col_frac + 0.5) - max(0., col_frac - 0.5)))
 269       if valid(row + 1, col):
 270         self._view[row + 1, col, d] += (
 271             (max(0., row_frac - 0.5)) *
 272             (min(1., col_frac + 0.5) - max(0., col_frac - 0.5)))
 273       if valid(row, col - 1):
 274         self._view[row, col - 1, d] += (
 275             (min(1., row_frac + 0.5) - max(0., row_frac - 0.5)) *
 276             (max(0., 0.5 - col_frac)))
 277       if valid(row, col + 1):
 278         self._view[row, col + 1, d] += (
 279             (min(1., row_frac + 0.5) - max(0., row_frac - 0.5)) *
 280             (max(0., col_frac - 0.5)))
 281       if valid(row - 1, col - 1):
 282         self._view[row - 1, col - 1, d] += (
 283             (max(0., 0.5 - row_frac)) * max(0., 0.5 - col_frac))
 284       if valid(row - 1, col + 1):
 285         self._view[row - 1, col + 1, d] += (
 286             (max(0., 0.5 - row_frac)) * max(0., col_frac - 0.5))
 287       if valid(row + 1, col + 1):
 288         self._view[row + 1, col + 1, d] += (
 289             (max(0., row_frac - 0.5)) * max(0., col_frac - 0.5))
 290       if valid(row + 1, col - 1):
 291         self._view[row + 1, col - 1, d] += (
 292             (max(0., row_frac - 0.5)) * max(0., 0.5 - col_frac))
 293 
 294     # Draw ant.
 295     robot_x, robot_y = self.wrapped_env.get_body_com("torso")[:2]
 296     self._robot_x = robot_x
 297     self._robot_y = robot_y
 298     self._robot_ori = self.get_ori()
 299 
 300     structure = self.MAZE_STRUCTURE
 301     size_scaling = self.MAZE_SIZE_SCALING
 302     height = self.MAZE_HEIGHT
 303 
 304     # Draw immovable blocks and chasms.
 305     for i in range(len(structure)):
 306       for j in range(len(structure[0])):
 307         if structure[i][j] == 1:  # Wall.
 308           update_view(j * size_scaling - self._init_torso_x,
 309                       i * size_scaling - self._init_torso_y,
 310                       0)
 311         if structure[i][j] == -1:  # Chasm.
 312           update_view(j * size_scaling - self._init_torso_x,
 313                       i * size_scaling - self._init_torso_y,
 314                       1)
 315 
 316     # Draw movable blocks.
 317     for block_name, block_type in self.movable_blocks:
 318       block_x, block_y = self.wrapped_env.get_body_com(block_name)[:2]
 319       update_view(block_x, block_y, 2)
 320 
 321     return self._view

 323   def get_range_sensor_obs(self):
 324     """Returns egocentric range sensor observations of maze."""
 325     robot_x, robot_y, robot_z = self.wrapped_env.get_body_com("torso")[:3]
 326     ori = self.get_ori()
 327 
 328     structure = self.MAZE_STRUCTURE
 329     size_scaling = self.MAZE_SIZE_SCALING
 330     height = self.MAZE_HEIGHT
 331 
 332     segments = []
 333     # Get line segments (corresponding to outer boundary) of each immovable
 334     # block or drop-off.
 335     for i in range(len(structure)):
 336       for j in range(len(structure[0])):
 337         if structure[i][j] in [1, -1]:  # There's a wall or drop-off.
 338           cx = j * size_scaling - self._init_torso_x
 339           cy = i * size_scaling - self._init_torso_y
 340           x1 = cx - 0.5 * size_scaling
 341           x2 = cx + 0.5 * size_scaling
 342           y1 = cy - 0.5 * size_scaling
 343           y2 = cy + 0.5 * size_scaling
 344           struct_segments = [
 345               ((x1, y1), (x2, y1)),
 346               ((x2, y1), (x2, y2)),
 347               ((x2, y2), (x1, y2)),
 348               ((x1, y2), (x1, y1)),
 349           ]
 350           for seg in struct_segments:
 351             segments.append(dict(
 352                 segment=seg,
 353                 type=structure[i][j],
 354             ))
 355     # Get line segments (corresponding to outer boundary) of each movable
 356     # block within the agent's z-view.
 357     for block_name, block_type in self.movable_blocks:
 358       block_x, block_y, block_z = self.wrapped_env.get_body_com(block_name)[:3]
 359       if (block_z + height * size_scaling / 2 >= robot_z and
 360           robot_z >= block_z - height * size_scaling / 2):  # Block in view.
 361         x1 = block_x - 0.5 * size_scaling
 362         x2 = block_x + 0.5 * size_scaling
 363         y1 = block_y - 0.5 * size_scaling
 364         y2 = block_y + 0.5 * size_scaling
 365         struct_segments = [
 366             ((x1, y1), (x2, y1)),
 367             ((x2, y1), (x2, y2)),
 368             ((x2, y2), (x1, y2)),
 369             ((x1, y2), (x1, y1)),
 370         ]
 371         for seg in struct_segments:
 372           segments.append(dict(
 373               segment=seg,
 374               type=block_type,
 375           ))
 376 
 377     sensor_readings = np.zeros((self._n_bins, 3))  # 3 for wall, drop-off, block
 378     for ray_idx in range(self._n_bins):
 379       ray_ori = (ori - self._sensor_span * 0.5 +
 380                  (2 * ray_idx + 1.0) / (2 * self._n_bins) * self._sensor_span)
 381       ray_segments = []
 382       # Get all segments that intersect with ray.
 383       for seg in segments:
 384         p = maze_env_utils.ray_segment_intersect(
 385             ray=((robot_x, robot_y), ray_ori),
 386             segment=seg["segment"])
 387         if p is not None:
 388           ray_segments.append(dict(
 389               segment=seg["segment"],
 390               type=seg["type"],
 391               ray_ori=ray_ori,
 392               distance=maze_env_utils.point_distance(p, (robot_x, robot_y)),
 393           ))
 394       if len(ray_segments) > 0:
 395         # Find out which segment is intersected first.
 396         first_seg = sorted(ray_segments, key=lambda x: x["distance"])[0]
 397         seg_type = first_seg["type"]
 398         idx = (0 if seg_type == 1 else  # Wall.
 399                1 if seg_type == -1 else  # Drop-off.
 400                2 if maze_env_utils.can_move(seg_type) else  # Block.
 401                None)
 402         if first_seg["distance"] <= self._sensor_range:
 403           sensor_readings[ray_idx][idx] = (self._sensor_range - first_seg["distance"]) / self._sensor_range
 404 
 405     return sensor_readings

 407   def _get_obs(self):
 408     wrapped_obs = self.wrapped_env._get_obs()
 409     if self._top_down_view:
 410       view = [self.get_top_down_view().flat]
 411     else:
 412       view = []
 413 
 414     if self._observe_blocks:
 415       additional_obs = []
 416       for block_name, block_type in self.movable_blocks:
 417         additional_obs.append(self.wrapped_env.get_body_com(block_name))
 418       wrapped_obs = np.concatenate([wrapped_obs[:3]] + additional_obs +
 419                                    [wrapped_obs[3:]])
 420 
 421     range_sensor_obs = self.get_range_sensor_obs()
 422     return np.concatenate([wrapped_obs,
 423                            range_sensor_obs.flat] +
 424                            view + [[self.t * 0.001]])

 426   def reset(self):
 427     self.t = 0
 428     self.trajectory = []

 430     if len(self._init_positions) > 1:

 442   @property

 447     return gym.spaces.Box(low, high)

 453   def _find_robot(self):
 454     structure = self.MAZE_STRUCTURE
 455     size_scaling = self.MAZE_SIZE_SCALING
 456     for i in range(len(structure)):
 457       for j in range(len(structure[0])):
 458         if structure[i][j] == 'r':
 459           return j * size_scaling, i * size_scaling
 460     assert False, 'No robot in maze specification.'
 461 
 462   def _find_all_robots(self):
 463     structure = self.MAZE_STRUCTURE
 464     size_scaling = self.MAZE_SIZE_SCALING
 465     coords = []
 466     for i in range(len(structure)):
 467       for j in range(len(structure[0])):
 468         if structure[i][j] == 'r':
 469           coords.append((j * size_scaling, i * size_scaling))
 470     return coords

 472   def _is_in_collision(self, pos):
 473     x, y = pos
 474     structure = self.MAZE_STRUCTURE
 475     size_scaling = self.MAZE_SIZE_SCALING
 476     for i in range(len(structure)):
 477       for j in range(len(structure[0])):
 478         if structure[i][j] == 1:
 479           minx = j * size_scaling - size_scaling * 0.5 - self._init_torso_x
 480           maxx = j * size_scaling + size_scaling * 0.5 - self._init_torso_x
 481           miny = i * size_scaling - size_scaling * 0.5 - self._init_torso_y
 482           maxy = i * size_scaling + size_scaling * 0.5 - self._init_torso_y
 483           if minx <= x <= maxx and miny <= y <= maxy:
 484             return True
 485     return False

 487   def step(self, action):
 488     self.t += 1
 489     if self._manual_collision:
 490       old_pos = self.wrapped_env.get_xy()
 491       inner_next_obs, inner_reward, done, info = self.wrapped_env.step(action)
 492       new_pos = self.wrapped_env.get_xy()
 493       if self._is_in_collision(new_pos):
 494         self.wrapped_env.set_xy(old_pos)
 495     else:
 496       inner_next_obs, inner_reward, done, info = self.wrapped_env.step(action)
 497     next_obs = self._get_obs()
 498     done = False
 499     return next_obs, inner_reward, done, info
