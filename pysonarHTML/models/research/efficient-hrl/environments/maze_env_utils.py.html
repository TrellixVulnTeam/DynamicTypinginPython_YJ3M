  32 def can_move_x(movable):
  33   return movable in [Move.X, Move.XY, Move.XZ, Move.XYZ,
  34                      Move.SpinXY]

  39                      Move.SpinXY]

  43   return movable in [Move.Z, Move.XZ, Move.YZ, Move.XYZ]

  46 def can_spin(movable):
  47   return movable in [Move.SpinXY]

  50 def can_move(movable):
  51   return can_move_x(movable) or can_move_y(movable) or can_move_z(movable)

  54 def construct_maze(maze_id='Maze'):
  55   if maze_id == 'Maze':
  56     structure = [
  57         [1, 1, 1, 1, 1],
  58         [1, 'r', 0, 0, 1],
  59         [1, 1, 1, 0, 1],
  60         [1, 0, 0, 0, 1],
  61         [1, 1, 1, 1, 1],
  62     ]
  63   elif maze_id == 'Push':
  64     structure = [
  65         [1, 1,  1,  1,   1],
  66         [1, 0, 'r', 1,   1],
  67         [1, 0,  Move.XY, 0,  1],
  68         [1, 1,  0,  1,   1],
  69         [1, 1,  1,  1,   1],
  70     ]
  71   elif maze_id == 'Fall':
  72     structure = [
  73         [1, 1,   1,  1],
  74         [1, 'r', 0,  1],
  75         [1, 0,   Move.YZ,  1],
  76         [1, -1, -1,  1],
  77         [1, 0,   0,  1],
  78         [1, 1,   1,  1],
  79     ]
  80   elif maze_id == 'Block':
  81     O = 'r'
  82     structure = [
  83         [1, 1, 1, 1, 1],
  84         [1, O, 0, 0, 1],
  85         [1, 0, 0, 0, 1],
  86         [1, 0, 0, 0, 1],
  87         [1, 1, 1, 1, 1],
  88     ]
  89   elif maze_id == 'BlockMaze':
  90     O = 'r'
  91     structure = [
  92         [1, 1, 1, 1],
  93         [1, O, 0, 1],
  94         [1, 1, 0, 1],
  95         [1, 0, 0, 1],
  96         [1, 1, 1, 1],
  97     ]
  98   else:
  99       raise NotImplementedError('The provided MazeId %s is not recognized' % maze_id)
 100 
 101   return structure

 104 def line_intersect(pt1, pt2, ptA, ptB):
 105   """
 106   Taken from https://www.cs.hmc.edu/ACM/lectures/intersections.html
 107 
 108   this returns the intersection of Line(pt1,pt2) and Line(ptA,ptB)
 109   """
 110 
 111   DET_TOLERANCE = 0.00000001
 112 
 113   # the first line is pt1 + r*(pt2-pt1)
 114   # in component form:
 115   x1, y1 = pt1
 116   x2, y2 = pt2
 117   dx1 = x2 - x1
 118   dy1 = y2 - y1
 119 
 120   # the second line is ptA + s*(ptB-ptA)
 121   x, y = ptA
 122   xB, yB = ptB
 123   dx = xB - x
 124   dy = yB - y
 125 
 126   DET = (-dx1 * dy + dy1 * dx)
 127 
 128   if math.fabs(DET) < DET_TOLERANCE: return (0, 0, 0, 0, 0)
 129 
 130   # now, the determinant should be OK
 131   DETinv = 1.0 / DET
 132 
 133   # find the scalar amount along the "self" segment
 134   r = DETinv * (-dy * (x - x1) + dx * (y - y1))
 135 
 136   # find the scalar amount along the input line
 137   s = DETinv * (-dy1 * (x - x1) + dx1 * (y - y1))
 138 
 139   # return the average of the two descriptions
 140   xi = (x1 + r * dx1 + x + s * dx) / 2.0
 141   yi = (y1 + r * dy1 + y + s * dy) / 2.0
 142   return (xi, yi, 1, r, s)

 145 def ray_segment_intersect(ray, segment):
 146   """
 147   Check if the ray originated from (x, y) with direction theta intersects the line segment (x1, y1) -- (x2, y2),
 148   and return the intersection point if there is one
 149   """
 150   (x, y), theta = ray
 151   # (x1, y1), (x2, y2) = segment
 152   pt1 = (x, y)
 153   len = 1
 154   pt2 = (x + len * math.cos(theta), y + len * math.sin(theta))
 155   xo, yo, valid, r, s = line_intersect(pt1, pt2, *segment)
 156   if valid and r >= 0 and 0 <= s <= 1:
 157     return (xo, yo)
 158   return None

 161 def point_distance(p1, p2):
 162   x1, y1 = p1
 163   x2, y2 = p2
 164   return ((x1 - x2) ** 2 + (y1 - y2) ** 2) ** 0.5
