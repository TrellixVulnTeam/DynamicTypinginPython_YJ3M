  16 """Wrapper for creating the ant environment in gym_mujoco."""
  17 
  18 import math
  19 import numpy as np
  20 import mujoco_py
  21 from gym import utils
  22 from gym.envs.mujoco import mujoco_env
  23 
  24 
  25 class PointEnv(mujoco_env.MujocoEnv, utils.EzPickle):
  26   FILE = "point.xml"
  27   ORI_IND = 2
  28 
  29   def __init__(self, file_path=None, expose_all_qpos=True):
  30     self._expose_all_qpos = expose_all_qpos
  31 
  32     mujoco_env.MujocoEnv.__init__(self, file_path, 1)
  33     utils.EzPickle.__init__(self)
  34 
  35   @property
  36   def physics(self):
  37     # check mujoco version is greater than version 1.50 to call correct physics
  38     # model containing PyMjData object for getting and setting position/velocity
  39     # check https://github.com/openai/mujoco-py/issues/80 for updates to api
  40     if mujoco_py.get_version() >= '1.50':  
  41       return self.sim
  42     else:
  43       return self.model
  44 
  45   def _step(self, a):
  46     return self.step(a)
  47 
  48   def step(self, action):
  49     action[0] = 0.2 * action[0]
  50     qpos = np.copy(self.physics.data.qpos)
  51     qpos[2] += action[1]
  52     ori = qpos[2]
  53     # compute increment in each direction
  54     dx = math.cos(ori) * action[0]
  55     dy = math.sin(ori) * action[0]
  56     # ensure that the robot is within reasonable range
  57     qpos[0] = np.clip(qpos[0] + dx, -100, 100)
  58     qpos[1] = np.clip(qpos[1] + dy, -100, 100)
  59     qvel = self.physics.data.qvel
  60     self.set_state(qpos, qvel)
  61     for _ in range(0, self.frame_skip):
  62       self.physics.step()
  63     next_obs = self._get_obs()
  64     reward = 0
  65     done = False
  66     info = {}
  67     return next_obs, reward, done, info
  68 
  69   def _get_obs(self):
  70     if self._expose_all_qpos:
  71       return np.concatenate([
  72           self.physics.data.qpos.flat[:3],  # Only point-relevant coords.
  73           self.physics.data.qvel.flat[:3]])
  74     return np.concatenate([
  75         self.physics.data.qpos.flat[2:3],
  76         self.physics.data.qvel.flat[:3]])
  77 
  78   def reset_model(self):
  79     qpos = self.init_qpos + self.np_random.uniform(
  80         size=self.physics.model.nq, low=-.1, high=.1)
  81     qvel = self.init_qvel + self.np_random.randn(self.physics.model.nv) * .1
  82 
  83     # Set everything other than point to original position and 0 velocity.
  84     qpos[3:] = self.init_qpos[3:]
  85     qvel[3:] = 0.
  86     self.set_state(qpos, qvel)
  87     return self._get_obs()
  88 
  89   def get_ori(self):
  90     return self.physics.data.qpos[self.__class__.ORI_IND]
  91 
  92   def set_xy(self, xy):
  93     qpos = np.copy(self.physics.data.qpos)
  94     qpos[0] = xy[0]
  95     qpos[1] = xy[1]
  96 
  97     qvel = self.physics.data.qvel
