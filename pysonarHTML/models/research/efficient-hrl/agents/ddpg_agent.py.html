  16 """A DDPG/NAF agent.
  17 
  18 Implements the Deep Deterministic Policy Gradient (DDPG) algorithm from
  19 "Continuous control with deep reinforcement learning" - Lilicrap et al.
  20 https://arxiv.org/abs/1509.02971, and the Normalized Advantage Functions (NAF)
  21 algorithm "Continuous Deep Q-Learning with Model-based Acceleration" - Gu et al.
  22 https://arxiv.org/pdf/1603.00748.
  23 """

  32 @gin.configurable
  33 class DdpgAgent(object):
  34   """An RL agent that learns using the DDPG algorithm.
  35 
  36   Example usage:
  37 
  38   def critic_net(states, actions):
  39     ...
  40   def actor_net(states, num_action_dims):
  41     ...
  42 
  43   Given a tensorflow environment tf_env,
  44   (of type learning.deepmind.rl.environments.tensorflow.python.tfpyenvironment)
  45 
  46   obs_spec = tf_env.observation_spec()
  47   action_spec = tf_env.action_spec()
  48 
  49   ddpg_agent = agent.DdpgAgent(obs_spec,
  50                                action_spec,
  51                                actor_net=actor_net,
  52                                critic_net=critic_net)
  53 
  54   we can perform actions on the environment as follows:
  55 
  56   state = tf_env.observations()[0]
  57   action = ddpg_agent.actor_net(tf.expand_dims(state, 0))[0, :]
  58   transition_type, reward, discount = tf_env.step([action])
  59 
  60   Train:
  61 
  62   critic_loss = ddpg_agent.critic_loss(states, actions, rewards, discounts,
  63                                        next_states)
  64   actor_loss = ddpg_agent.actor_loss(states)
  65 
  66   critic_train_op = slim.learning.create_train_op(
  67       critic_loss,
  68       critic_optimizer,
  69       variables_to_train=ddpg_agent.get_trainable_critic_vars(),
  70   )
  71 
  72   actor_train_op = slim.learning.create_train_op(
  73       actor_loss,
  74       actor_optimizer,
  75       variables_to_train=ddpg_agent.get_trainable_actor_vars(),
  76   )
  77   """
  78 
  79   ACTOR_NET_SCOPE = 'actor_net'
  80   CRITIC_NET_SCOPE = 'critic_net'
  81   TARGET_ACTOR_NET_SCOPE = 'target_actor_net'
  82   TARGET_CRITIC_NET_SCOPE = 'target_critic_net'
  83 
  84   def __init__(self,
  85                observation_spec,
  86                action_spec,
  87                actor_net=networks.actor_net,
  88                critic_net=networks.critic_net,
  89                td_errors_loss=tf.losses.huber_loss,
  90                dqda_clipping=0.,
  91                actions_regularizer=0.,
  92                target_q_clipping=None,
  93                residual_phi=0.0,
  94                debug_summaries=False):
  95     """Constructs a DDPG agent.
  96 
  97     Args:
  98       observation_spec: A TensorSpec defining the observations.
  99       action_spec: A BoundedTensorSpec defining the actions.
 100       actor_net: A callable that creates the actor network. Must take the
 101         following arguments: states, num_actions. Please see networks.actor_net
 102         for an example.
 103       critic_net: A callable that creates the critic network. Must take the
 104         following arguments: states, actions. Please see networks.critic_net
 105         for an example.
 106       td_errors_loss: A callable defining the loss function for the critic
 107         td error.
 108       dqda_clipping: (float) clips the gradient dqda element-wise between
 109         [-dqda_clipping, dqda_clipping]. Does not perform clipping if
 110         dqda_clipping == 0.
 111       actions_regularizer: A scalar, when positive penalizes the norm of the
 112         actions. This can prevent saturation of actions for the actor_loss.
 113       target_q_clipping: (tuple of floats) clips target q values within
 114         (low, high) values when computing the critic loss.
 115       residual_phi: (float) [0.0, 1.0] Residual algorithm parameter that
 116         interpolates between Q-learning and residual gradient algorithm.
 117         http://www.leemon.com/papers/1995b.pdf
 118       debug_summaries: If True, add summaries to help debug behavior.
 119     Raises:
 120       ValueError: If 'dqda_clipping' is < 0.
 121     """
 122     self._observation_spec = observation_spec[0]
 123     self._action_spec = action_spec[0]
 124     self._state_shape = tf.TensorShape([None]).concatenate(
 125         self._observation_spec.shape)
 126     self._action_shape = tf.TensorShape([None]).concatenate(
 127         self._action_spec.shape)
 128     self._num_action_dims = self._action_spec.shape.num_elements()
 129 
 130     self._scope = tf.get_variable_scope().name
 131     self._actor_net = tf.make_template(
 132         self.ACTOR_NET_SCOPE, actor_net, create_scope_now_=True)
 133     self._critic_net = tf.make_template(
 134         self.CRITIC_NET_SCOPE, critic_net, create_scope_now_=True)
 135     self._target_actor_net = tf.make_template(
 136         self.TARGET_ACTOR_NET_SCOPE, actor_net, create_scope_now_=True)
 137     self._target_critic_net = tf.make_template(
 138         self.TARGET_CRITIC_NET_SCOPE, critic_net, create_scope_now_=True)
 139     self._td_errors_loss = td_errors_loss
 140     if dqda_clipping < 0:
 141       raise ValueError('dqda_clipping must be >= 0.')
 142     self._dqda_clipping = dqda_clipping
 143     self._actions_regularizer = actions_regularizer
 144     self._target_q_clipping = target_q_clipping
 145     self._residual_phi = residual_phi
 146     self._debug_summaries = debug_summaries
 147 
 148   def _batch_state(self, state):
 149     """Convert state to a batched state.
 150 
 151     Args:
 152       state: Either a list/tuple with an state tensor [num_state_dims].
 153     Returns:
 154       A tensor [1, num_state_dims]
 155     """
 156     if isinstance(state, (tuple, list)):
 157       state = state[0]
 158     if state.get_shape().ndims == 1:
 159       state = tf.expand_dims(state, 0)
 160     return state
 161 
 162   def action(self, state):
 163     """Returns the next action for the state.
 164 
 165     Args:
 166       state: A [num_state_dims] tensor representing a state.
 167     Returns:
 168       A [num_action_dims] tensor representing the action.
 169     """
 170     return self.actor_net(self._batch_state(state), stop_gradients=True)[0, :]

 172   @gin.configurable('ddpg_sample_action')
 173   def sample_action(self, state, stddev=1.0):
 174     """Returns the action for the state with additive noise.
 175 
 176     Args:
 177       state: A [num_state_dims] tensor representing a state.
 178       stddev: stddev for the Ornstein-Uhlenbeck noise.
 179     Returns:
 180       A [num_action_dims] action tensor.
 181     """
 182     agent_action = self.action(state)
 183     agent_action += tf.random_normal(tf.shape(agent_action)) * stddev
 184     return utils.clip_to_spec(agent_action, self._action_spec)

 186   def actor_net(self, states, stop_gradients=False):
 187     """Returns the output of the actor network.
 188 
 189     Args:
 190       states: A [batch_size, num_state_dims] tensor representing a batch
 191         of states.
 192       stop_gradients: (boolean) if true, gradients cannot be propogated through
 193         this operation.
 194     Returns:
 195       A [batch_size, num_action_dims] tensor of actions.
 196     Raises:
 197       ValueError: If `states` does not have the expected dimensions.
 198     """
 199     self._validate_states(states)
 200     actions = self._actor_net(states, self._action_spec)
 201     if stop_gradients:
 202       actions = tf.stop_gradient(actions)
 203     return actions
 204 
 205   def critic_net(self, states, actions, for_critic_loss=False):
 206     """Returns the output of the critic network.
 207 
 208     Args:
 209       states: A [batch_size, num_state_dims] tensor representing a batch
 210         of states.
 211       actions: A [batch_size, num_action_dims] tensor representing a batch
 212         of actions.
 213     Returns:
 214       q values: A [batch_size] tensor of q values.
 215     Raises:
 216       ValueError: If `states` or `actions' do not have the expected dimensions.
 217     """
 218     self._validate_states(states)
 219     self._validate_actions(actions)
 220     return self._critic_net(states, actions,
 221                             for_critic_loss=for_critic_loss)
 222 
 223   def target_actor_net(self, states):
 224     """Returns the output of the target actor network.
 225 
 226     The target network is used to compute stable targets for training.
 227 
 228     Args:
 229       states: A [batch_size, num_state_dims] tensor representing a batch
 230         of states.
 231     Returns:
 232       A [batch_size, num_action_dims] tensor of actions.
 233     Raises:
 234       ValueError: If `states` does not have the expected dimensions.
 235     """
 236     self._validate_states(states)
 237     actions = self._target_actor_net(states, self._action_spec)
 238     return tf.stop_gradient(actions)
 239 
 240   def target_critic_net(self, states, actions, for_critic_loss=False):
 241     """Returns the output of the target critic network.
 242 
 243     The target network is used to compute stable targets for training.
 244 
 245     Args:
 246       states: A [batch_size, num_state_dims] tensor representing a batch
 247         of states.
 248       actions: A [batch_size, num_action_dims] tensor representing a batch
 249         of actions.
 250     Returns:
 251       q values: A [batch_size] tensor of q values.
 252     Raises:
 253       ValueError: If `states` or `actions' do not have the expected dimensions.
 254     """
 255     self._validate_states(states)
 256     self._validate_actions(actions)
 257     return tf.stop_gradient(
 258         self._target_critic_net(states, actions,
 259                                 for_critic_loss=for_critic_loss))
 260 
 261   def value_net(self, states, for_critic_loss=False):
 262     """Returns the output of the critic evaluated with the actor.
 263 
 264     Args:
 265       states: A [batch_size, num_state_dims] tensor representing a batch
 266         of states.
 267     Returns:
 268       q values: A [batch_size] tensor of q values.
 269     """
 270     actions = self.actor_net(states)
 271     return self.critic_net(states, actions,
 272                            for_critic_loss=for_critic_loss)
 273 
 274   def target_value_net(self, states, for_critic_loss=False):
 275     """Returns the output of the target critic evaluated with the target actor.
 276 
 277     Args:
 278       states: A [batch_size, num_state_dims] tensor representing a batch
 279         of states.
 280     Returns:
 281       q values: A [batch_size] tensor of q values.
 282     """
 283     target_actions = self.target_actor_net(states)
 284     return self.target_critic_net(states, target_actions,
 285                                   for_critic_loss=for_critic_loss)

 287   def critic_loss(self, states, actions, rewards, discounts,
 288                   next_states):
 289     """Computes a loss for training the critic network.
 290 
 291     The loss is the mean squared error between the Q value predictions of the
 292     critic and Q values estimated using TD-lambda.
 293 
 294     Args:
 295       states: A [batch_size, num_state_dims] tensor representing a batch
 296         of states.
 297       actions: A [batch_size, num_action_dims] tensor representing a batch
 298         of actions.
 299       rewards: A [batch_size, ...] tensor representing a batch of rewards,
 300         broadcastable to the critic net output.
 301       discounts: A [batch_size, ...] tensor representing a batch of discounts,
 302         broadcastable to the critic net output.
 303       next_states: A [batch_size, num_state_dims] tensor representing a batch
 304         of next states.
 305     Returns:
 306       A rank-0 tensor representing the critic loss.
 307     Raises:
 308       ValueError: If any of the inputs do not have the expected dimensions, or
 309         if their batch_sizes do not match.
 310     """
 311     self._validate_states(states)
 312     self._validate_actions(actions)
 313     self._validate_states(next_states)
 314 
 315     target_q_values = self.target_value_net(next_states, for_critic_loss=True)
 316     td_targets = target_q_values * discounts + rewards
 317     if self._target_q_clipping is not None:
 318       td_targets = tf.clip_by_value(td_targets, self._target_q_clipping[0],
 319                                     self._target_q_clipping[1])
 320     q_values = self.critic_net(states, actions, for_critic_loss=True)
 321     td_errors = td_targets - q_values
 322     if self._debug_summaries:
 323       gen_debug_td_error_summaries(
 324           target_q_values, q_values, td_targets, td_errors)
 325 
 326     loss = self._td_errors_loss(td_targets, q_values)
 327 
 328     if self._residual_phi > 0.0:  # compute residual gradient loss
 329       residual_q_values = self.value_net(next_states, for_critic_loss=True)
 330       residual_td_targets = residual_q_values * discounts + rewards
 331       if self._target_q_clipping is not None:
 332         residual_td_targets = tf.clip_by_value(residual_td_targets,
 333                                                self._target_q_clipping[0],
 334                                                self._target_q_clipping[1])
 335       residual_td_errors = residual_td_targets - q_values
 336       residual_loss = self._td_errors_loss(
 337           residual_td_targets, residual_q_values)
 338       loss = (loss * (1.0 - self._residual_phi) +
 339               residual_loss * self._residual_phi)
 340     return loss

 342   def actor_loss(self, states):
 343     """Computes a loss for training the actor network.
 344 
 345     Note that output does not represent an actual loss. It is called a loss only
 346     in the sense that its gradient w.r.t. the actor network weights is the
 347     correct gradient for training the actor network,
 348     i.e. dloss/dweights = (dq/da)*(da/dweights)
 349     which is the gradient used in Algorithm 1 of Lilicrap et al.
 350 
 351     Args:
 352       states: A [batch_size, num_state_dims] tensor representing a batch
 353         of states.
 354     Returns:
 355       A rank-0 tensor representing the actor loss.
 356     Raises:
 357       ValueError: If `states` does not have the expected dimensions.
 358     """
 359     self._validate_states(states)
 360     actions = self.actor_net(states, stop_gradients=False)
 361     critic_values = self.critic_net(states, actions)
 362     q_values = self.critic_function(critic_values, states)
 363     dqda = tf.gradients([q_values], [actions])[0]
 364     dqda_unclipped = dqda
 365     if self._dqda_clipping > 0:
 366       dqda = tf.clip_by_value(dqda, -self._dqda_clipping, self._dqda_clipping)
 367 
 368     actions_norm = tf.norm(actions)
 369     if self._debug_summaries:
 370       with tf.name_scope('dqda'):
 371         tf.summary.scalar('actions_norm', actions_norm)
 372         tf.summary.histogram('dqda', dqda)
 373         tf.summary.histogram('dqda_unclipped', dqda_unclipped)
 374         tf.summary.histogram('actions', actions)
 375         for a in range(self._num_action_dims):
 376           tf.summary.histogram('dqda_unclipped_%d' % a, dqda_unclipped[:, a])
 377           tf.summary.histogram('dqda_%d' % a, dqda[:, a])
 378 
 379     actions_norm *= self._actions_regularizer
 380     return slim.losses.mean_squared_error(tf.stop_gradient(dqda + actions),
 381                                           actions,
 382                                           scope='actor_loss') + actions_norm
 383 
 384   @gin.configurable('ddpg_critic_function')
 385   def critic_function(self, critic_values, states, weights=None):
 386     """Computes q values based on critic_net outputs, states, and weights.
 387 
 388     Args:
 389       critic_values: A tf.float32 [batch_size, ...] tensor representing outputs
 390         from the critic net.
 391       states: A [batch_size, num_state_dims] tensor representing a batch
 392         of states.
 393       weights: A list or Numpy array or tensor with a shape broadcastable to
 394         `critic_values`.
 395     Returns:
 396       A tf.float32 [batch_size] tensor representing q values.
 397     """
 398     del states  # unused args
 399     if weights is not None:
 400       weights = tf.convert_to_tensor(weights, dtype=critic_values.dtype)
 401       critic_values *= weights
 402     if critic_values.shape.ndims > 1:
 403       critic_values = tf.reduce_sum(critic_values,
 404                                     range(1, critic_values.shape.ndims))
 405     critic_values.shape.assert_has_rank(1)
 406     return critic_values

 408   @gin.configurable('ddpg_update_targets')
 409   def update_targets(self, tau=1.0):
 410     """Performs a soft update of the target network parameters.
 411 
 412     For each weight w_s in the actor/critic networks, and its corresponding
 413     weight w_t in the target actor/critic networks, a soft update is:
 414     w_t = (1- tau) x w_t + tau x ws
 415 
 416     Args:
 417       tau: A float scalar in [0, 1]
 418     Returns:
 419       An operation that performs a soft update of the target network parameters.
 420     Raises:
 421       ValueError: If `tau` is not in [0, 1].
 422     """
 423     if tau < 0 or tau > 1:
 424       raise ValueError('Input `tau` should be in [0, 1].')
 425     update_actor = utils.soft_variables_update(
 426         slim.get_trainable_variables(
 427             utils.join_scope(self._scope, self.ACTOR_NET_SCOPE)),
 428         slim.get_trainable_variables(
 429             utils.join_scope(self._scope, self.TARGET_ACTOR_NET_SCOPE)),
 430         tau)
 431     update_critic = utils.soft_variables_update(
 432         slim.get_trainable_variables(
 433             utils.join_scope(self._scope, self.CRITIC_NET_SCOPE)),
 434         slim.get_trainable_variables(
 435             utils.join_scope(self._scope, self.TARGET_CRITIC_NET_SCOPE)),
 436         tau)
 437     return tf.group(update_actor, update_critic, name='update_targets')

 475   def _validate_states(self, states):
 476     """Raises a value error if `states` does not have the expected shape.
 477 
 478     Args:
 479       states: A tensor.
 480     Raises:
 481       ValueError: If states.shape or states.dtype are not compatible with
 482         observation_spec.
 483     """
 484     states.shape.assert_is_compatible_with(self._state_shape)
 485     if not states.dtype.is_compatible_with(self._observation_spec.dtype):
 486       raise ValueError('states.dtype={} is not compatible with'
 487                        ' observation_spec.dtype={}'.format(
 488                            states.dtype, self._observation_spec.dtype))
 489 
 490   def _validate_actions(self, actions):
 491     """Raises a value error if `actions` does not have the expected shape.
 492 
 493     Args:
 494       actions: A tensor.
 495     Raises:
 496       ValueError: If actions.shape or actions.dtype are not compatible with
 497         action_spec.
 498     """
 499     actions.shape.assert_is_compatible_with(self._action_shape)
 500     if not actions.dtype.is_compatible_with(self._action_spec.dtype):
 501       raise ValueError('actions.dtype={} is not compatible with'
 502                        ' action_spec.dtype={}'.format(
 503                            actions.dtype, self._action_spec.dtype))

 565         self.ACTOR_NET_SCOPE, actor_net, create_scope_now_=True)
 566     self._critic_net = tf.make_template(
 567         self.CRITIC_NET_SCOPE, critic_net, create_scope_now_=True)
 568     self._critic_net2 = tf.make_template(
 569         self.CRITIC_NET2_SCOPE, critic_net, create_scope_now_=True)
 570     self._target_actor_net = tf.make_template(
 571         self.TARGET_ACTOR_NET_SCOPE, actor_net, create_scope_now_=True)
 572     self._target_critic_net = tf.make_template(
 573         self.TARGET_CRITIC_NET_SCOPE, critic_net, create_scope_now_=True)
 574     self._target_critic_net2 = tf.make_template(
 575         self.TARGET_CRITIC_NET2_SCOPE, critic_net, create_scope_now_=True)
 576     self._td_errors_loss = td_errors_loss
 577     if dqda_clipping < 0:
 578       raise ValueError('dqda_clipping must be >= 0.')
 579     self._dqda_clipping = dqda_clipping
 580     self._actions_regularizer = actions_regularizer
 581     self._target_q_clipping = target_q_clipping
 582     self._residual_phi = residual_phi
 583     self._debug_summaries = debug_summaries

 596   def critic_net(self, states, actions, for_critic_loss=False):
 597     """Returns the output of the critic network.
 598 
 599     Args:
 600       states: A [batch_size, num_state_dims] tensor representing a batch
 601         of states.
 602       actions: A [batch_size, num_action_dims] tensor representing a batch
 603         of actions.
 604     Returns:
 605       q values: A [batch_size] tensor of q values.
 606     Raises:
 607       ValueError: If `states` or `actions' do not have the expected dimensions.
 608     """
 609     values1 = self._critic_net(states, actions,
 610                                for_critic_loss=for_critic_loss)
 611     values2 = self._critic_net2(states, actions,
 612                                 for_critic_loss=for_critic_loss)
 613     if for_critic_loss:
 614       return values1, values2
 615     return values1
 616 
 617   def target_critic_net(self, states, actions, for_critic_loss=False):
 618     """Returns the output of the target critic network.
 619 
 620     The target network is used to compute stable targets for training.
 621 
 622     Args:
 623       states: A [batch_size, num_state_dims] tensor representing a batch
 624         of states.
 625       actions: A [batch_size, num_action_dims] tensor representing a batch
 626         of actions.
 627     Returns:
 628       q values: A [batch_size] tensor of q values.
 629     Raises:
 630       ValueError: If `states` or `actions' do not have the expected dimensions.
 631     """
 632     self._validate_states(states)
 633     self._validate_actions(actions)
 634     values1 = tf.stop_gradient(
 635         self._target_critic_net(states, actions,
 636                                 for_critic_loss=for_critic_loss))
 637     values2 = tf.stop_gradient(
 638         self._target_critic_net2(states, actions,
 639                                  for_critic_loss=for_critic_loss))
 640     if for_critic_loss:
 641       return values1, values2
 642     return values1

 644   def value_net(self, states, for_critic_loss=False):
 645     """Returns the output of the critic evaluated with the actor.
 646 
 647     Args:
 648       states: A [batch_size, num_state_dims] tensor representing a batch
 649         of states.
 650     Returns:
 651       q values: A [batch_size] tensor of q values.
 652     """
 653     actions = self.actor_net(states)
 654     return self.critic_net(states, actions,
 655                            for_critic_loss=for_critic_loss)
 656 
 657   def target_value_net(self, states, for_critic_loss=False):
 658     """Returns the output of the target critic evaluated with the target actor.
 659 
 660     Args:
 661       states: A [batch_size, num_state_dims] tensor representing a batch
 662         of states.
 663     Returns:
 664       q values: A [batch_size] tensor of q values.
 665     """
 666     target_actions = self.target_actor_net(states)
 667     noise = tf.clip_by_value(
 668         tf.random_normal(tf.shape(target_actions), stddev=0.2), -0.5, 0.5)
 669     values1, values2 = self.target_critic_net(
 670         states, target_actions + noise,
 671         for_critic_loss=for_critic_loss)
 672     values = tf.minimum(values1, values2)
 673     return values, values

 675   @gin.configurable('td3_update_targets')
 676   def update_targets(self, tau=1.0):
 677     """Performs a soft update of the target network parameters.
 678 
 679     For each weight w_s in the actor/critic networks, and its corresponding
 680     weight w_t in the target actor/critic networks, a soft update is:
 681     w_t = (1- tau) x w_t + tau x ws
 682 
 683     Args:
 684       tau: A float scalar in [0, 1]
 685     Returns:
 686       An operation that performs a soft update of the target network parameters.
 687     Raises:
 688       ValueError: If `tau` is not in [0, 1].
 689     """
 690     if tau < 0 or tau > 1:
 691       raise ValueError('Input `tau` should be in [0, 1].')
 692     update_actor = utils.soft_variables_update(
 693         slim.get_trainable_variables(
 694             utils.join_scope(self._scope, self.ACTOR_NET_SCOPE)),
 695         slim.get_trainable_variables(
 696             utils.join_scope(self._scope, self.TARGET_ACTOR_NET_SCOPE)),
 697         tau)
 698     # NOTE: This updates both critic networks.
 699     update_critic = utils.soft_variables_update(
 700         slim.get_trainable_variables(
 701             utils.join_scope(self._scope, self.CRITIC_NET_SCOPE)),
 702         slim.get_trainable_variables(
 703             utils.join_scope(self._scope, self.TARGET_CRITIC_NET_SCOPE)),
 704         tau)
 705     return tf.group(update_actor, update_critic, name='update_targets')

 708 def gen_debug_td_error_summaries(
 709     target_q_values, q_values, td_targets, td_errors):
 710   """Generates debug summaries for critic given a set of batch samples.
 711 
 712   Args:
 713     target_q_values: set of predicted next stage values.
 714     q_values: current predicted value for the critic network.
 715     td_targets: discounted target_q_values with added next stage reward.
 716     td_errors: the different between td_targets and q_values.
 717   """
 718   with tf.name_scope('td_errors'):
 719     tf.summary.histogram('td_targets', td_targets)
 720     tf.summary.histogram('q_values', q_values)
 721     tf.summary.histogram('target_q_values', target_q_values)
 722     tf.summary.histogram('td_errors', td_errors)
 723     with tf.name_scope('td_targets'):
 724       tf.summary.scalar('mean', tf.reduce_mean(td_targets))
 725       tf.summary.scalar('max', tf.reduce_max(td_targets))
 726       tf.summary.scalar('min', tf.reduce_min(td_targets))
 727     with tf.name_scope('q_values'):
 728       tf.summary.scalar('mean', tf.reduce_mean(q_values))
 729       tf.summary.scalar('max', tf.reduce_max(q_values))
 730       tf.summary.scalar('min', tf.reduce_min(q_values))
 731     with tf.name_scope('target_q_values'):
 732       tf.summary.scalar('mean', tf.reduce_mean(target_q_values))
 733       tf.summary.scalar('max', tf.reduce_max(target_q_values))
 734       tf.summary.scalar('min', tf.reduce_min(target_q_values))
 735     with tf.name_scope('td_errors'):
 736       tf.summary.scalar('mean', tf.reduce_mean(td_errors))
 737       tf.summary.scalar('max', tf.reduce_max(td_errors))
 738       tf.summary.scalar('min', tf.reduce_min(td_errors))
 739       tf.summary.scalar('mean_abs', tf.reduce_mean(tf.abs(td_errors)))
