  16 """A circular buffer where each element is a list of tensors.
  17 
  18 Each element of the buffer is a list of tensors. An example use case is a replay
  19 buffer in reinforcement learning, where each element is a list of tensors
  20 representing the state, action, reward etc.
  21 
  22 New elements are added sequentially, and once the buffer is full, we
  23 start overwriting them in a circular fashion. Reading does not remove any
  24 elements, only adding new elements does.
  25 """
  26 
  27 import collections
  28 import numpy as np
  29 import tensorflow as tf
  30 
  31 import gin.tf
  32 
  33 
  34 @gin.configurable
  35 class CircularBuffer(object):
  36   """A circular buffer where each element is a list of tensors."""
  37 
  38   def __init__(self, buffer_size=1000, scope='replay_buffer'):
  39     """Circular buffer of list of tensors.
  40 
  41     Args:
  42       buffer_size: (integer) maximum number of tensor lists the buffer can hold.
  43       scope: (string) variable scope for creating the variables.
  44     """
  45     self._buffer_size = np.int64(buffer_size)
  46     self._scope = scope
  47     self._tensors = collections.OrderedDict()
  48     with tf.variable_scope(self._scope):
  49       self._num_adds = tf.Variable(0, dtype=tf.int64, name='num_adds')
  50     self._num_adds_cs = tf.CriticalSection(name='num_adds')
  51 
  52   @property
  53   def buffer_size(self):
  54     return self._buffer_size
  55 
  56   @property
  57   def scope(self):
  58     return self._scope
  59 
  60   @property
  61   def num_adds(self):
  62     return self._num_adds
  63 
  64   def _create_variables(self, tensors):
  65     with tf.variable_scope(self._scope):
  66       for name in tensors.keys():
  67         tensor = tensors[name]
  68         self._tensors[name] = tf.get_variable(
  69             name='BufferVariable_' + name,
  70             shape=[self._buffer_size] + tensor.get_shape().as_list(),
  71             dtype=tensor.dtype,
  72             trainable=False)
  73 
  74   def _validate(self, tensors):
  75     """Validate shapes of tensors."""
  76     if len(tensors) != len(self._tensors):
  77       raise ValueError('Expected tensors to have %d elements. Received %d '
  78                        'instead.' % (len(self._tensors), len(tensors)))
  79     if self._tensors.keys() != tensors.keys():
  80       raise ValueError('The keys of tensors should be the always the same.'
  81                        'Received %s instead %s.' %
  82                        (tensors.keys(), self._tensors.keys()))
  83     for name, tensor in tensors.items():
  84       if tensor.get_shape().as_list() != self._tensors[
  85           name].get_shape().as_list()[1:]:
  86         raise ValueError('Tensor %s has incorrect shape.' % name)
  87       if not tensor.dtype.is_compatible_with(self._tensors[name].dtype):
  88         raise ValueError(
  89             'Tensor %s has incorrect data type. Expected %s, received %s' %
  90             (name, self._tensors[name].read_value().dtype, tensor.dtype))
  91 
  92   def add(self, tensors):
  93     """Adds an element (list/tuple/dict of tensors) to the buffer.
  94 
  95     Args:
  96       tensors: (list/tuple/dict of tensors) to be added to the buffer.
  97     Returns:
  98       An add operation that adds the input `tensors` to the buffer. Similar to
  99         an enqueue_op.
 100     Raises:
 101       ValueError: If the shapes and data types of input `tensors' are not the
 102         same across calls to the add function.
 103     """
 104     return self.maybe_add(tensors, True)
 105 
 106   def maybe_add(self, tensors, condition):
 107     """Adds an element (tensors) to the buffer based on the condition..
 108 
 109     Args:
 110       tensors: (list/tuple of tensors) to be added to the buffer.
 111       condition: A boolean Tensor controlling whether the tensors would be added
 112         to the buffer or not.
 113     Returns:
 114       An add operation that adds the input `tensors` to the buffer. Similar to
 115         an maybe_enqueue_op.
 116     Raises:
 117       ValueError: If the shapes and data types of input `tensors' are not the
 118         same across calls to the add function.
 119     """
 120     if not isinstance(tensors, dict):
 121       names = [str(i) for i in range(len(tensors))]
 122       tensors = collections.OrderedDict(zip(names, tensors))
 123     if not isinstance(tensors, collections.OrderedDict):
 124       tensors = collections.OrderedDict(
 125           sorted(tensors.items(), key=lambda t: t[0]))
 126     if not self._tensors:
 127       self._create_variables(tensors)
 128     else:
 129       self._validate(tensors)
 130 
 131     #@tf.critical_section(self._position_mutex)
 132     def _increment_num_adds():
 133       # Adding 0 to the num_adds variable is a trick to read the value of the
 134       # variable and return a read-only tensor. Doing this in a critical
 135       # section allows us to capture a snapshot of the variable that will
 136       # not be affected by other threads updating num_adds.
 137       return self._num_adds.assign_add(1) + 0
 138     def _add():
 139       num_adds_inc = self._num_adds_cs.execute(_increment_num_adds)
 140       current_pos = tf.mod(num_adds_inc - 1, self._buffer_size)
 141       update_ops = []
 142       for name in self._tensors.keys():
 143         update_ops.append(
 144             tf.scatter_update(self._tensors[name], current_pos, tensors[name]))
 145       return tf.group(*update_ops)
 146 
 147     return tf.contrib.framework.smart_cond(condition, _add, tf.no_op)
 148 
 149   def get_random_batch(self, batch_size, keys=None, num_steps=1):
 150     """Samples a batch of tensors from the buffer with replacement.
 151 
 152     Args:
 153       batch_size: (integer) number of elements to sample.
 154       keys: List of keys of tensors to retrieve. If None retrieve all.
 155       num_steps: (integer) length of trajectories to return. If > 1 will return
 156         a list of lists, where each internal list represents a trajectory of
 157         length num_steps.
 158     Returns:
 159       A list of tensors, where each element in the list is a batch sampled from
 160         one of the tensors in the buffer.
 161     Raises:
 162       ValueError: If get_random_batch is called before calling the add function.
 163       tf.errors.InvalidArgumentError: If this operation is executed before any
 164         items are added to the buffer.
 165     """
 166     if not self._tensors:
 167       raise ValueError('The add function must be called before get_random_batch.')
 168     if keys is None:
 169       keys = self._tensors.keys()
 170 
 171     latest_start_index = self.get_num_adds() - num_steps + 1
 172     empty_buffer_assert = tf.Assert(
 173         tf.greater(latest_start_index, 0),
 174         ['Not enough elements have been added to the buffer.'])
 175     with tf.control_dependencies([empty_buffer_assert]):
 176       max_index = tf.minimum(self._buffer_size, latest_start_index)
 177       indices = tf.random_uniform(
 178           [batch_size],
 179           minval=0,
 180           maxval=max_index,
 181           dtype=tf.int64)
 182       if num_steps == 1:
 183         return self.gather(indices, keys)
 184       else:
 185         return self.gather_nstep(num_steps, indices, keys)
 186 
 187   def gather(self, indices, keys=None):
 188     """Returns elements at the specified indices from the buffer.
 189 
 190     Args:
 191       indices: (list of integers or rank 1 int Tensor) indices in the buffer to
 192         retrieve elements from.
 193       keys: List of keys of tensors to retrieve. If None retrieve all.
 194     Returns:
 195       A list of tensors, where each element in the list is obtained by indexing
 196         one of the tensors in the buffer.
 197     Raises:
 198       ValueError: If gather is called before calling the add function.
 199       tf.errors.InvalidArgumentError: If indices are bigger than the number of
 200         items in the buffer.
 201     """
 202     if not self._tensors:
 203       raise ValueError('The add function must be called before calling gather.')
 204     if keys is None:
 205       keys = self._tensors.keys()
 206     with tf.name_scope('Gather'):
 207       index_bound_assert = tf.Assert(
 208           tf.less(
 209               tf.to_int64(tf.reduce_max(indices)),
 210               tf.minimum(self.get_num_adds(), self._buffer_size)),
 211           ['Index out of bounds.'])
 212       with tf.control_dependencies([index_bound_assert]):
 213         indices = tf.convert_to_tensor(indices)
 214 
 215       batch = []
 216       for key in keys:
 217         batch.append(tf.gather(self._tensors[key], indices, name=key))
 218       return batch
 219 
 220   def gather_nstep(self, num_steps, indices, keys=None):
 221     """Returns elements at the specified indices from the buffer.
 222 
 223     Args:
 224       num_steps: (integer) length of trajectories to return.
 225       indices: (list of rank num_steps int Tensor) indices in the buffer to
 226         retrieve elements from for multiple trajectories. Each Tensor in the
 227         list represents the indices for a trajectory.
 228       keys: List of keys of tensors to retrieve. If None retrieve all.
 229     Returns:
 230       A list of list-of-tensors, where each element in the list is obtained by
 231         indexing one of the tensors in the buffer.
 232     Raises:
 233       ValueError: If gather is called before calling the add function.
 234       tf.errors.InvalidArgumentError: If indices are bigger than the number of
 235         items in the buffer.
 236     """
 237     if not self._tensors:
 238       raise ValueError('The add function must be called before calling gather.')
 239     if keys is None:
 240       keys = self._tensors.keys()
 241     with tf.name_scope('Gather'):
 242       index_bound_assert = tf.Assert(
 243           tf.less_equal(
 244               tf.to_int64(tf.reduce_max(indices) + num_steps),
 245               self.get_num_adds()),
 246           ['Trajectory indices go out of bounds.'])
 247       with tf.control_dependencies([index_bound_assert]):
 248         indices = tf.map_fn(
 249             lambda x: tf.mod(tf.range(x, x + num_steps), self._buffer_size),
 250             indices,
 251             dtype=tf.int64)
 252 
 253       batch = []
 254       for key in keys:
 255 
 256         def SampleTrajectories(trajectory_indices, key=key,
 257                                num_steps=num_steps):
 258           trajectory_indices.set_shape([num_steps])
 259           return tf.gather(self._tensors[key], trajectory_indices, name=key)
 260 
 261         batch.append(tf.map_fn(SampleTrajectories, indices,
 262                                dtype=self._tensors[key].dtype))
 263       return batch
 264 
 265   def get_position(self):
 266     """Returns the position at which the last element was added.
 267 
 268     Returns:
 269       An int tensor representing the index at which the last element was added
 270         to the buffer or -1 if no elements were added.
 271     """
 272     return tf.cond(self.get_num_adds() < 1,
 273                    lambda: self.get_num_adds() - 1,
 274                    lambda: tf.mod(self.get_num_adds() - 1, self._buffer_size))
 275 
 276   def get_num_adds(self):
 277     """Returns the number of additions to the buffer.
 278 
 279     Returns:
 280       An int tensor representing the number of elements that were added.
 281     """
 282     def num_adds():
 283       return self._num_adds.value()
 284 
 285     return self._num_adds_cs.execute(num_adds)
 286 
 287   def get_num_tensors(self):
 288     """Returns the number of tensors (slots) in the buffer."""
 289     return len(self._tensors)
