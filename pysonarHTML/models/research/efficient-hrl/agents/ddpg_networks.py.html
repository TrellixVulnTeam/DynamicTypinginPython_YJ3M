  16 """Sample actor(policy) and critic(q) networks to use with DDPG/NAF agents.
  17 
  18 The DDPG networks are defined in "Section 7: Experiment Details" of
  19 "Continuous control with deep reinforcement learning" - Lilicrap et al.
  20 https://arxiv.org/abs/1509.02971
  21 
  22 The NAF critic network is based on "Section 4" of "Continuous deep Q-learning
  23 with model-based acceleration" - Gu et al. https://arxiv.org/pdf/1603.00748.
  24 """
  25 
  26 import tensorflow as tf
  27 slim = tf.contrib.slim
  28 import gin.tf
  29 
  30 
  31 @gin.configurable('ddpg_critic_net')
  32 def critic_net(states, actions,
  33                for_critic_loss=False,
  34                num_reward_dims=1,
  35                states_hidden_layers=(400,),
  36                actions_hidden_layers=None,
  37                joint_hidden_layers=(300,),
  38                weight_decay=0.0001,
  39                normalizer_fn=None,
  40                activation_fn=tf.nn.relu,
  41                zero_obs=False,
  42                images=False):
  43   """Creates a critic that returns q values for the given states and actions.
  44 
  45   Args:
  46     states: (castable to tf.float32) a [batch_size, num_state_dims] tensor
  47       representing a batch of states.
  48     actions: (castable to tf.float32) a [batch_size, num_action_dims] tensor
  49       representing a batch of actions.
  50     num_reward_dims: Number of reward dimensions.
  51     states_hidden_layers: tuple of hidden layers units for states.
  52     actions_hidden_layers: tuple of hidden layers units for actions.
  53     joint_hidden_layers: tuple of hidden layers units after joining states
  54       and actions using tf.concat().
  55     weight_decay: Weight decay for l2 weights regularizer.
  56     normalizer_fn: Normalizer function, i.e. slim.layer_norm,
  57     activation_fn: Activation function, i.e. tf.nn.relu, slim.leaky_relu, ...
  58   Returns:
  59     A tf.float32 [batch_size] tensor of q values, or a tf.float32
  60       [batch_size, num_reward_dims] tensor of vector q values if
  61       num_reward_dims > 1.
  62   """
  63   with slim.arg_scope(
  64       [slim.fully_connected],
  65       activation_fn=activation_fn,
  66       normalizer_fn=normalizer_fn,
  67       weights_regularizer=slim.l2_regularizer(weight_decay),
  68       weights_initializer=slim.variance_scaling_initializer(
  69           factor=1.0/3.0, mode='FAN_IN', uniform=True)):
  70 
  71     orig_states = tf.to_float(states)
  72     #states = tf.to_float(states)
  73     states = tf.concat([tf.to_float(states), tf.to_float(actions)], -1)  #TD3
  74     if images or zero_obs:
  75       states *= tf.constant([0.0] * 2 + [1.0] * (states.shape[1] - 2))  #LALA
  76     actions = tf.to_float(actions)
  77     if states_hidden_layers:
  78       states = slim.stack(states, slim.fully_connected, states_hidden_layers,
  79                           scope='states')
  80     if actions_hidden_layers:
  81       actions = slim.stack(actions, slim.fully_connected, actions_hidden_layers,
  82                            scope='actions')
  83     joint = tf.concat([states, actions], 1)
  84     if joint_hidden_layers:
  85       joint = slim.stack(joint, slim.fully_connected, joint_hidden_layers,
  86                          scope='joint')
  87     with slim.arg_scope([slim.fully_connected],
  88                         weights_regularizer=None,
  89                         weights_initializer=tf.random_uniform_initializer(
  90                             minval=-0.003, maxval=0.003)):
  91       value = slim.fully_connected(joint, num_reward_dims,
  92                                    activation_fn=None,
  93                                    normalizer_fn=None,
  94                                    scope='q_value')
  95     if num_reward_dims == 1:
  96       value = tf.reshape(value, [-1])
  97     if not for_critic_loss and num_reward_dims > 1:
  98       value = tf.reduce_sum(
  99           value * tf.abs(orig_states[:, -num_reward_dims:]), -1)
 100   return value

 103 @gin.configurable('ddpg_actor_net')
 104 def actor_net(states, action_spec,
 105               hidden_layers=(400, 300),
 106               normalizer_fn=None,
 107               activation_fn=tf.nn.relu,
 108               zero_obs=False,
 109               images=False):
 110   """Creates an actor that returns actions for the given states.
 111 
 112   Args:
 113     states: (castable to tf.float32) a [batch_size, num_state_dims] tensor
 114       representing a batch of states.
 115     action_spec: (BoundedTensorSpec) A tensor spec indicating the shape
 116       and range of actions.
 117     hidden_layers: tuple of hidden layers units.
 118     normalizer_fn: Normalizer function, i.e. slim.layer_norm,
 119     activation_fn: Activation function, i.e. tf.nn.relu, slim.leaky_relu, ...
 120   Returns:
 121     A tf.float32 [batch_size, num_action_dims] tensor of actions.
 122   """
 123 
 124   with slim.arg_scope(
 125       [slim.fully_connected],
 126       activation_fn=activation_fn,
 127       normalizer_fn=normalizer_fn,
 128       weights_initializer=slim.variance_scaling_initializer(
 129           factor=1.0/3.0, mode='FAN_IN', uniform=True)):
 130 
 131     states = tf.to_float(states)
 132     orig_states = states
 133     if images or zero_obs:  # Zero-out x, y position. Hacky.
 134       states *= tf.constant([0.0] * 2 + [1.0] * (states.shape[1] - 2))
 135     if hidden_layers:
 136       states = slim.stack(states, slim.fully_connected, hidden_layers,
 137                           scope='states')
 138     with slim.arg_scope([slim.fully_connected],
 139                         weights_initializer=tf.random_uniform_initializer(
 140                             minval=-0.003, maxval=0.003)):
 141       actions = slim.fully_connected(states,
 142                                      action_spec.shape.num_elements(),
 143                                      scope='actions',
 144                                      normalizer_fn=None,
 145                                      activation_fn=tf.nn.tanh)
 146       action_means = (action_spec.maximum + action_spec.minimum) / 2.0
 147       action_magnitudes = (action_spec.maximum - action_spec.minimum) / 2.0
 148       actions = action_means + action_magnitudes * actions
 149 
 150   return actions
