 135   def log_probs(self, states, actions, state_reprs, contexts=None):
 136     assert contexts is not None
 137     batch_dims = [tf.shape(states)[0], tf.shape(states)[1]]
 138     contexts = self.tf_context.context_multi_transition_fn(
 139         contexts, states=tf.to_float(state_reprs))
 140 
 141     flat_states = tf.reshape(states,
 142                              [batch_dims[0] * batch_dims[1], states.shape[-1]])
 143     flat_contexts = [tf.reshape(tf.cast(context, states.dtype),
 144                                 [batch_dims[0] * batch_dims[1], context.shape[-1]])
 145                      for context in contexts]
 146     flat_pred_actions = self.actions(flat_states, flat_contexts)
 147     pred_actions = tf.reshape(flat_pred_actions,
 148                               batch_dims + [flat_pred_actions.shape[-1]])
 149 
 150     error = tf.square(actions - pred_actions)
 151     spec_range = (self._action_spec.maximum - self._action_spec.minimum) / 2
 152     normalized_error = error / tf.constant(spec_range) ** 2
 153     return -normalized_error

 155   @gin.configurable('uvf_add_noise_fn')
 156   def add_noise_fn(self, action_fn, stddev=1.0, debug=False,
 157                    clip=True, global_step=None):
 158     """Returns the action_fn with additive Gaussian noise.
 159 
 160     Args:
 161       action_fn: A callable(`state`, `context`) which returns a
 162         [num_action_dims] tensor representing a action.
 163       stddev: stddev for the Ornstein-Uhlenbeck noise.
 164       debug: Print debug messages.
 165     Returns:
 166       A [num_action_dims] action tensor.
 167     """
 168     if global_step is not None:
 169       stddev *= tf.maximum(  # Decay exploration during training.
 170           tf.train.exponential_decay(1.0, global_step, 1e6, 0.8), 0.5)
 171     def noisy_action_fn(state, context=None):
 172       """Noisy action fn."""
 173       action = action_fn(state, context)
 174       if debug:
 175         action = uvf_utils.tf_print(
 176             action, [action],
 177             message='[add_noise_fn] pre-noise action',
 178             first_n=100)
 179       noise_dist = tf.distributions.Normal(tf.zeros_like(action),
 180                                            tf.ones_like(action) * stddev)
 181       noise = noise_dist.sample()
 182       action += noise
 183       if debug:
 184         action = uvf_utils.tf_print(
 185             action, [action],
 186             message='[add_noise_fn] post-noise action',
 187             first_n=100)
 188       if clip:
 189         action = uvf_utils.clip_to_spec(action, self._action_spec)
 190       return action
 191     return noisy_action_fn

 193   def merged_state(self, state, context=None):
 194     """Returns the merged state from the environment state and contexts.
 195 
 196     Args:
 197       state: A [num_state_dims] tensor representing a state.
 198       context: A list of [num_context_dims] tensor representing a context.
 199         If None, use the internal context.
 200     Returns:
 201       A [num_merged_state_dims] tensor representing the merged state.
 202     """
 203     if context is None:
 204       context = list(self.context_vars)
 205     state = tf.concat([state,] + context, axis=-1)
 206     self._validate_states(self._batch_state(state))
 207     return state

 209   def merged_states(self, states, contexts=None):
 210     """Returns the batch merged state from the batch env state and contexts.
 211 
 212     Args:
 213       states: A [batch_size, num_state_dims] tensor representing a batch
 214         of states.
 215       contexts: A list of [batch_size, num_context_dims] tensor
 216         representing a batch of contexts. If None,
 217         use the internal context.
 218     Returns:
 219       A [batch_size, num_merged_state_dims] tensor representing the batch
 220         of merged states.
 221     """
 222     if contexts is None:
 223       contexts = [tf.tile(tf.expand_dims(context, axis=0),
 224                           (tf.shape(states)[0], 1)) for
 225                   context in self.context_vars]
 226     states = tf.concat([states,] + contexts, axis=-1)
 227     self._validate_states(states)
 228     return states

 230   def unmerged_states(self, merged_states):
 231     """Returns the batch state and contexts from the batch merged state.
 232 
 233     Args:
 234       merged_states: A [batch_size, num_merged_state_dims] tensor
 235         representing a batch of merged states.
 236     Returns:
 237       A [batch_size, num_state_dims] tensor and a list of
 238         [batch_size, num_context_dims] tensors representing the batch state
 239         and contexts respectively.
 240     """
 241     self._validate_states(merged_states)
 242     num_state_dims = self.env_observation_spec.shape.as_list()[0]
 243     num_context_dims_list = [c.shape.as_list()[0] for c in self.context_specs]
 244     states = merged_states[:, :num_state_dims]
 245     contexts = []
 246     i = num_state_dims
 247     for num_context_dims in num_context_dims_list:
 248       contexts.append(merged_states[:, i: i+num_context_dims])
 249       i += num_context_dims
 250     return states, contexts

 252   def sample_random_actions(self, batch_size=1):
 253     """Return random actions.
 254 
 255     Args:
 256       batch_size: Batch size.
 257     Returns:
 258       A [batch_size, num_action_dims] tensor representing the batch of actions.
 259     """
 260     actions = tf.concat(
 261         [
 262             tf.random_uniform(
 263                 shape=(batch_size, 1),
 264                 minval=self._action_spec.minimum[i],
 265                 maxval=self._action_spec.maximum[i])
 266             for i in range(self._action_spec.shape[0].value)
 267         ],
 268         axis=1)
 269     return actions

 271   def clip_actions(self, actions):
 272     """Clip actions to spec.
 273 
 274     Args:
 275       actions: A [batch_size, num_action_dims] tensor representing
 276       the batch of actions.
 277     Returns:
 278       A [batch_size, num_action_dims] tensor representing the batch
 279       of clipped actions.
 280     """
 281     actions = tf.concat(
 282         [
 283             tf.clip_by_value(
 284                 actions[:, i:i+1],
 285                 self._action_spec.minimum[i],
 286                 self._action_spec.maximum[i])
 287             for i in range(self._action_spec.shape[0].value)
 288         ],
 289         axis=1)
 290     return actions

 292   def mix_contexts(self, contexts, insert_contexts, indices):
 293     """Mix two contexts based on indices.
 294 
 295     Args:
 296       contexts: A list of [batch_size, num_context_dims] tensor representing
 297       the batch of contexts.
 298       insert_contexts: A list of [batch_size, num_context_dims] tensor
 299       representing the batch of contexts to be inserted.
 300       indices: A list of a list of integers denoting indices to replace.
 301     Returns:
 302       A list of resulting contexts.
 303     """
 304     if indices is None: indices = [[]] * len(contexts)
 305     assert len(contexts) == len(indices)
 306     assert all([spec.shape.ndims == 1 for spec in self.context_specs])
 307     mix_contexts = []
 308     for contexts_, insert_contexts_, indices_, spec in zip(
 309         contexts, insert_contexts, indices, self.context_specs):
 310       mix_contexts.append(
 311           tf.concat(
 312               [
 313                   insert_contexts_[:, i:i + 1] if i in indices_ else
 314                   contexts_[:, i:i + 1] for i in range(spec.shape.as_list()[0])
 315               ],
 316               axis=1))
 317     return mix_contexts

 319   def begin_episode_ops(self, mode, action_fn=None, state=None):
 320     """Returns ops that reset agent at beginning of episodes.
 321 
 322     Args:
 323       mode: a string representing the mode=[train, explore, eval].
 324     Returns:
 325       A list of ops.
 326     """
 327     all_ops = []
 328     for _, action_var in sorted(self._action_vars.items()):
 329       sample_action = self.sample_random_actions(1)[0]
 330       all_ops.append(tf.assign(action_var, sample_action))
 331     all_ops += self.tf_context.reset(mode=mode, agent=self._meta_agent,
 332                                      action_fn=action_fn, state=state)
 333     return all_ops

 335   def cond_begin_episode_op(self, cond, input_vars, mode, meta_action_fn):
 336     """Returns op that resets agent at beginning of episodes.
 337 
 338     A new episode is begun if the cond op evalues to `False`.
 339 
 340     Args:
 341       cond: a Boolean tensor variable.
 342       input_vars: A list of tensor variables.
 343       mode: a string representing the mode=[train, explore, eval].
 344     Returns:
 345       Conditional begin op.
 346     """
 347     (state, action, reward, next_state,
 348      state_repr, next_state_repr) = input_vars
 349     def continue_fn():
 350       """Continue op fn."""
 351       items = [state, action, reward, next_state,
 352                state_repr, next_state_repr] + list(self.context_vars)
 353       batch_items = [tf.expand_dims(item, 0) for item in items]
 354       (states, actions, rewards, next_states,
 355        state_reprs, next_state_reprs) = batch_items[:6]
 356       context_reward = self.compute_rewards(
 357           mode, state_reprs, actions, rewards, next_state_reprs,
 358           batch_items[6:])[0][0]
 359       context_reward = tf.cast(context_reward, dtype=reward.dtype)
 360       if self.meta_agent is not None:
 361         meta_action = tf.concat(self.context_vars, -1)
 362         items = [state, meta_action, reward, next_state,
 363                  state_repr, next_state_repr] + list(self.meta_agent.context_vars)
 364         batch_items = [tf.expand_dims(item, 0) for item in items]
 365         (states, meta_actions, rewards, next_states,
 366          state_reprs, next_state_reprs) = batch_items[:6]
 367         meta_reward = self.meta_agent.compute_rewards(
 368             mode, states, meta_actions, rewards,
 369             next_states, batch_items[6:])[0][0]
 370         meta_reward = tf.cast(meta_reward, dtype=reward.dtype)
 371       else:
 372         meta_reward = tf.constant(0, dtype=reward.dtype)
 373 
 374       with tf.control_dependencies([context_reward, meta_reward]):
 375         step_ops = self.tf_context.step(mode=mode, agent=self._meta_agent,
 376                                         state=state,
 377                                         next_state=next_state,
 378                                         state_repr=state_repr,
 379                                         next_state_repr=next_state_repr,
 380                                         action_fn=meta_action_fn)
 381       with tf.control_dependencies(step_ops):
 382         context_reward, meta_reward = map(tf.identity, [context_reward, meta_reward])
 383       return context_reward, meta_reward
 384     def begin_episode_fn():
 385       """Begin op fn."""
 386       begin_ops = self.begin_episode_ops(mode=mode, action_fn=meta_action_fn, state=state)
 387       with tf.control_dependencies(begin_ops):
 388         return tf.zeros_like(reward), tf.zeros_like(reward)
 389     with tf.control_dependencies(input_vars):
 390       cond_begin_episode_op = tf.cond(cond, continue_fn, begin_episode_fn)
 391     return cond_begin_episode_op

 405   def init_action_vars(self, name, i=None):
 406     """Create and return a tensorflow Variable holding an action.
 407 
 408     Args:
 409       name: Name of the variables.
 410       i: Integer id.
 411     Returns:
 412       A [num_action_dims] tensor.
 413     """
 414     if i is not None:
 415       name += '_%d' % i
 416     assert name not in self._action_vars, ('Conflict! %s is already '
 417                                            'initialized.') % name
 418     self._action_vars[name] = tf.Variable(
 419         self.sample_random_actions(1)[0], name='%s_action' % (name))
 420     self._validate_actions(tf.expand_dims(self._action_vars[name], 0))
 421     return self._action_vars[name]

 423   @gin.configurable('uvf_critic_function')
 424   def critic_function(self, critic_vals, states, critic_fn=None):
 425     """Computes q values based on outputs from the critic net.
 426 
 427     Args:
 428       critic_vals: A tf.float32 [batch_size, ...] tensor representing outputs
 429         from the critic net.
 430       states: A [batch_size, num_state_dims] tensor representing a batch
 431         of states.
 432       critic_fn: A callable that process outputs from critic_net and
 433         outputs a [batch_size] tensor representing q values.
 434     Returns:
 435       A tf.float32 [batch_size] tensor representing q values.
 436     """
 437     if critic_fn is not None:
 438       env_states, contexts = self.unmerged_states(states)
 439       critic_vals = critic_fn(critic_vals, env_states, contexts)
 440     critic_vals.shape.assert_has_rank(1)
 441     return critic_vals

 547       context: A list of [num_context_dims] tensor representing a context.
 548     Returns:
 549       A [num_action_dims] tensor representing the action.
 550     """
 551     actions = self.actor_net(states, stop_gradients=False)
 552     regularizer = self._actions_reg * tf.reduce_mean(
 553         tf.reduce_sum(tf.abs(actions[:, self._k:]), -1), 0)
 554     loss = self.BASE_AGENT_CLASS.actor_loss(self, states)
 555     return regularizer + loss

 581 def state_preprocess_net(
 582     states,
 583     num_output_dims=2,
 584     states_hidden_layers=(100,),
 585     normalizer_fn=None,
 586     activation_fn=tf.nn.relu,
 587     zero_time=True,
 588     images=False):
 589   """Creates a simple feed forward net for embedding states.
 590   """
 591   with slim.arg_scope(
 592       [slim.fully_connected],
 593       activation_fn=activation_fn,
 594       normalizer_fn=normalizer_fn,
 595       weights_initializer=slim.variance_scaling_initializer(
 596           factor=1.0/3.0, mode='FAN_IN', uniform=True)):
 597 
 598     states_shape = tf.shape(states)
 599     states_dtype = states.dtype
 600     states = tf.to_float(states)
 601     if images:  # Zero-out x-y
 602       states *= tf.constant([0.] * 2 + [1.] * (states.shape[-1] - 2), dtype=states.dtype)
 603     if zero_time:
 604       states *= tf.constant([1.] * (states.shape[-1] - 1) + [0.], dtype=states.dtype)
 605     orig_states = states
 606     embed = states
 607     if states_hidden_layers:
 608       embed = slim.stack(embed, slim.fully_connected, states_hidden_layers,
 609                          scope='states')
 610 
 611     with slim.arg_scope([slim.fully_connected],
 612                         weights_regularizer=None,
 613                         weights_initializer=tf.random_uniform_initializer(
 614                             minval=-0.003, maxval=0.003)):
 615       embed = slim.fully_connected(embed, num_output_dims,
 616                                    activation_fn=None,
 617                                    normalizer_fn=None,
 618                                    scope='value')
 619 
 620     output = embed
 621     output = tf.cast(output, states_dtype)
 622     return output

 631     normalizer_fn=None,
 632     activation_fn=tf.nn.relu,

 634     images=False):
 635   """Creates a simple feed forward net for embedding actions.
 636   """
 637   with slim.arg_scope(
 638       [slim.fully_connected],

 641       weights_initializer=slim.variance_scaling_initializer(
 642           factor=1.0/3.0, mode='FAN_IN', uniform=True)):
 643 
 644     actions = tf.to_float(actions)
 645     if states is not None:
 646       if images:  # Zero-out x-y

 651

 662                                    activation_fn=None,
 663                                    normalizer_fn=None,
 664                                    scope='value')
 665       if num_output_dims == 1:
 666         return embed[:, 0, ...]
 667       else:
 668         return embed

 671 def huber(x, kappa=0.1):
 672   return (0.5 * tf.square(x) * tf.to_float(tf.abs(x) <= kappa) +
 673           kappa * (tf.abs(x) - 0.5 * kappa) * tf.to_float(tf.abs(x) > kappa)
 674           ) / kappa

 677 @gin.configurable()
 678 class StatePreprocess(object):
 679   STATE_PREPROCESS_NET_SCOPE = 'state_process_net'
 680   ACTION_EMBED_NET_SCOPE = 'action_embed_net'
 681 
 682   def __init__(self, trainable=False,
 683                state_preprocess_net=lambda states: states,
 684                action_embed_net=lambda actions, *args, **kwargs: actions,
 685                ndims=None):
 686     self.trainable = trainable
 687     self._scope = tf.get_variable_scope().name
 688     self._ndims = ndims
 689     self._state_preprocess_net = tf.make_template(
 690         self.STATE_PREPROCESS_NET_SCOPE, state_preprocess_net,
 691         create_scope_now_=True)
 692     self._action_embed_net = tf.make_template(
 693         self.ACTION_EMBED_NET_SCOPE, action_embed_net,
 694         create_scope_now_=True)
 695 
 696   def __call__(self, states):
 697     batched = states.get_shape().ndims != 1
 698     if not batched:
 699       states = tf.expand_dims(states, 0)
 700     embedded = self._state_preprocess_net(states)
 701     if self._ndims is not None:
 702       embedded = embedded[..., :self._ndims]
 703     if not batched:
 704       return embedded[0]
 705     return embedded
 706 
 707   def loss(self, states, next_states, low_actions, low_states):
 708     batch_size = tf.shape(states)[0]
 709     d = int(low_states.shape[1])
 710     # Sample indices into meta-transition to train on.
 711     probs = 0.99 ** tf.range(d, dtype=tf.float32)
 712     probs *= tf.constant([1.0] * (d - 1) + [1.0 / (1 - 0.99)],
 713                          dtype=tf.float32)
 714     probs /= tf.reduce_sum(probs)
 715     index_dist = tf.distributions.Categorical(probs=probs, dtype=tf.int64)
 716     indices = index_dist.sample(batch_size)
 717     batch_size = tf.cast(batch_size, tf.int64)
 718     next_indices = tf.concat(
 719         [tf.range(batch_size, dtype=tf.int64)[:, None],
 720          (1 + indices[:, None]) % d], -1)
 721     new_next_states = tf.where(indices < d - 1,
 722                                tf.gather_nd(low_states, next_indices),
 723                                next_states)
 724     next_states = new_next_states
 725 
 726     embed1 = tf.to_float(self._state_preprocess_net(states))
 727     embed2 = tf.to_float(self._state_preprocess_net(next_states))
 728     action_embed = self._action_embed_net(
 729         tf.layers.flatten(low_actions), states=states)
 730 
 731     tau = 2.0
 732     fn = lambda z: tau * tf.reduce_sum(huber(z), -1)
 733     all_embed = tf.get_variable('all_embed', [1024, int(embed1.shape[-1])],
 734                                 initializer=tf.zeros_initializer())
 735     upd = all_embed.assign(tf.concat([all_embed[batch_size:], embed2], 0))
 736     with tf.control_dependencies([upd]):
 737       close = 1 * tf.reduce_mean(fn(embed1 + action_embed - embed2))
 738       prior_log_probs = tf.reduce_logsumexp(
 739           -fn((embed1 + action_embed)[:, None, :] - all_embed[None, :, :]),
 740           axis=-1) - tf.log(tf.to_float(all_embed.shape[0]))
 741       far = tf.reduce_mean(tf.exp(-fn((embed1 + action_embed)[1:] - embed2[:-1])
 742                                   - tf.stop_gradient(prior_log_probs[1:])))
 743       repr_log_probs = tf.stop_gradient(
 744           -fn(embed1 + action_embed - embed2) - prior_log_probs) / tau
 745     return close + far, repr_log_probs, indices
 746 
 747   def get_trainable_vars(self):
 748     return (
 749         slim.get_trainable_variables(
 750             uvf_utils.join_scope(self._scope, self.STATE_PREPROCESS_NET_SCOPE)) +
 751         slim.get_trainable_variables(
 752             uvf_utils.join_scope(self._scope, self.ACTION_EMBED_NET_SCOPE)))

 755 @gin.configurable()
 756 class InverseDynamics(object):
 757   INVERSE_DYNAMICS_NET_SCOPE = 'inverse_dynamics'
 758 
 759   def __init__(self, spec):
 760     self._spec = spec
 761 
 762   def sample(self, states, next_states, num_samples, orig_goals, sc=0.5):
 763     goal_dim = orig_goals.shape[-1]
 764     spec_range = (self._spec.maximum - self._spec.minimum) / 2 * tf.ones([goal_dim])
 765     loc = tf.cast(next_states - states, tf.float32)[:, :goal_dim]
 766     scale = sc * tf.tile(tf.reshape(spec_range, [1, goal_dim]),
 767                          [tf.shape(states)[0], 1])
 768     dist = tf.distributions.Normal(loc, scale)
 769     if num_samples == 1:
 770       return dist.sample()
 771     samples = tf.concat([dist.sample(num_samples - 2),
 772                          tf.expand_dims(loc, 0),
 773                          tf.expand_dims(orig_goals, 0)], 0)
 774     return uvf_utils.clip_to_spec(samples, self._spec)
