  16 """Converts Tiny Imagenet dataset into TFRecord format.
  17 
  18 As an output this program generates following files in TFRecord format:
  19 - train.tfrecord
  20 - validation.tfrecord
  21 - test.tfrecord
  22 
  23 Generated train and validation files will contain tf.Example entries with
  24 following features:
  25 - image/encoded - encoded image
  26 - image/format - image format
  27 - label/wnid - label WordNet ID
  28 - label/imagenet - imagenet label [1 ... 1000]
  29 - label/tiny_imagenet - tiny imagenet label [0 ... 199]
  30 - bbox/xmin
  31 - bbox/ymin
  32 - bbox/xmax
  33 - bbox/ymax
  34 
  35 Test file will contain entries with 'image/encoded' and 'image/format' features.
  36 """

  67 class WnIdToNodeIdConverter(object):
  68   """Converts WordNet IDs to numerical labels."""
  69 
  70   def __init__(self, wnids_path, background_class):
  71     self._wnid_to_node_id = {}
  72     self._node_id_to_wnid = {}
  73     with tf.gfile.Open(wnids_path) as f:
  74       wnids_sequence = [wnid.strip() for wnid in f.readlines() if wnid.strip()]
  75     node_id_offset = 1 if background_class else 0
  76     for i, label in enumerate(wnids_sequence):
  77       self._wnid_to_node_id[label] = i + node_id_offset
  78       self._node_id_to_wnid[i + node_id_offset] = label
  79 
  80   def to_node_id(self, wnid):
  81     return self._wnid_to_node_id[wnid]
  82 
  83   def to_wnid(self, node_id):
  84     return self._node_id_to_wnid[node_id]
  85 
  86   def all_wnids(self):
  87     return self._wnid_to_node_id.keys()

  90 def read_tiny_imagenet_annotations(annotations_filename,
  91                                    images_dir,
  92                                    one_label=None):
  93   """Reads one file with Tiny Imagenet annotations."""
  94   result = []
  95   if one_label:
  96     column_names = ['filename', 'x1', 'y1', 'x2', 'y2']
  97   else:
  98     column_names = ['filename', 'label', 'x1', 'y1', 'x2', 'y2']
  99   with tf.gfile.Open(annotations_filename) as f:
 100     data = pd.read_csv(f, sep='\t', names=column_names)
 101   for row in data.itertuples():
 102     label = one_label if one_label else getattr(row, 'label')
 103     full_filename = os.path.join(images_dir, getattr(row, 'filename'))
 104     result.append((full_filename,
 105                    ImageMetadata(label=label,
 106                                  x1=getattr(row, 'x1'),
 107                                  y1=getattr(row, 'y1'),
 108                                  x2=getattr(row, 'x2'),
 109                                  y2=getattr(row, 'y2'))))
 110   return result

 113 def read_validation_annotations(validation_dir):
 114   """Reads validation data annotations."""
 115   return read_tiny_imagenet_annotations(
 116       os.path.join(validation_dir, 'val_annotations.txt'),
 117       os.path.join(validation_dir, 'images'))

 120 def read_training_annotations(training_dir):
 121   """Reads training data annotations."""
 122   result = []
 123   sub_dirs = tf.gfile.ListDirectory(training_dir)
 124   for sub_dir in sub_dirs:
 125     if not sub_dir.startswith('n'):
 126       logging.warning('Found non-class directory in training dir: %s', sub_dir)
 127       continue
 128     sub_dir_results = read_tiny_imagenet_annotations(
 129         os.path.join(training_dir, sub_dir, sub_dir + '_boxes.txt'),
 130         os.path.join(training_dir, sub_dir, 'images'),
 131         one_label=sub_dir)
 132     result.extend(sub_dir_results)
 133   return result

 136 def read_test_annotations(test_dir):
 137   """Reads test data annotations."""
 138   files = tf.gfile.ListDirectory(os.path.join(test_dir, 'images'))
 139   return [(os.path.join(test_dir, 'images', f), None)
 140           for f in files if f.endswith('.JPEG')]

 143 def get_image_format(filename):
 144   """Returns image format from filename."""
 145   filename = filename.lower()
 146   if filename.endswith('jpeg') or filename.endswith('jpg'):
 147     return 'jpeg'
 148   elif filename.endswith('png'):
 149     return 'png'
 150   else:
 151     raise ValueError('Unrecognized file format: %s' % filename)

 154 class TinyImagenetWriter(object):
 155   """Helper class which writes Tiny Imagenet dataset into TFRecord file."""
 156 
 157   def __init__(self, tiny_imagenet_wnid_conveter, imagenet_wnid_converter):
 158     self.tiny_imagenet_wnid_conveter = tiny_imagenet_wnid_conveter
 159     self.imagenet_wnid_converter = imagenet_wnid_converter
 160 
 161   def write_tf_record(self,
 162                       annotations,
 163                       output_file):
 164     """Generates TFRecord file from given list of annotations."""
 165     with tf.python_io.TFRecordWriter(output_file) as writer:
 166       for image_filename, image_metadata in annotations:
 167         with tf.gfile.Open(image_filename) as f:
 168           image_buffer = f.read()
 169         image_format = get_image_format(image_filename)
 170         features = {
 171             'image/encoded': tf.train.Feature(
 172                 bytes_list=tf.train.BytesList(value=[image_buffer])),
 173             'image/format': tf.train.Feature(
 174                 bytes_list=tf.train.BytesList(value=[image_format]))
 175         }
 176         if image_metadata:
 177           # bounding box features
 178           features['bbox/xmin'] = tf.train.Feature(
 179               int64_list=tf.train.Int64List(value=[image_metadata.x1]))
 180           features['bbox/ymin'] = tf.train.Feature(
 181               int64_list=tf.train.Int64List(value=[image_metadata.y1]))
 182           features['bbox/xmax'] = tf.train.Feature(
 183               int64_list=tf.train.Int64List(value=[image_metadata.x2]))
 184           features['bbox/ymax'] = tf.train.Feature(
 185               int64_list=tf.train.Int64List(value=[image_metadata.y2]))
 186           # tiny imagenet label, from [0, 200) iterval
 187           tiny_imagenet_label = self.tiny_imagenet_wnid_conveter.to_node_id(
 188               image_metadata.label)
 189           features['label/wnid'] = tf.train.Feature(
 190               bytes_list=tf.train.BytesList(value=image_metadata.label))
 191           features['label/tiny_imagenet'] = tf.train.Feature(
 192               int64_list=tf.train.Int64List(value=[tiny_imagenet_label]))
 193           # full imagenet label, from [1, 1001) interval
 194           if self.imagenet_wnid_converter:
 195             imagenet_label = self.imagenet_wnid_converter.to_node_id(
 196                 image_metadata.label)
 197             features['label/imagenet'] = tf.train.Feature(
 198                 int64_list=tf.train.Int64List(value=[imagenet_label]))
 199         example = tf.train.Example(features=tf.train.Features(feature=features))
 200         writer.write(example.SerializeToString())

 203 def main(_):
 204   assert FLAGS.input_dir, 'Input directory must be provided'
 205   assert FLAGS.output_dir, 'Output directory must be provided'
 206 
 207   # Create WordNet ID conveters for tiny imagenet and possibly for imagenet
 208   tiny_imagenet_wnid_conveter = WnIdToNodeIdConverter(
 209       os.path.join(FLAGS.input_dir, 'wnids.txt'),
 210       background_class=False)
 211   if FLAGS.imagenet_synsets_path:
 212     imagenet_wnid_converter = WnIdToNodeIdConverter(FLAGS.imagenet_synsets_path,
 213                                                     background_class=True)
 214   else:
 215     imagenet_wnid_converter = None
 216 
 217   # read tiny imagenet annotations
 218   train_annotations = read_training_annotations(
 219       os.path.join(FLAGS.input_dir, 'train'))
 220   random.shuffle(train_annotations)
 221   val_annotations = read_validation_annotations(
 222       os.path.join(FLAGS.input_dir, 'val'))
 223   test_filenames = read_test_annotations(os.path.join(FLAGS.input_dir, 'test'))
 224 
 225   # Generate TFRecord files
 226   writer = TinyImagenetWriter(tiny_imagenet_wnid_conveter,
 227                               imagenet_wnid_converter)
 228   tf.logging.info('Converting %d training images', len(train_annotations))
 229   writer.write_tf_record(train_annotations,
 230                          os.path.join(FLAGS.output_dir, 'train.tfrecord'))
 231   tf.logging.info('Converting %d validation images ', len(val_annotations))
 232   writer.write_tf_record(val_annotations,
 233                          os.path.join(FLAGS.output_dir, 'validation.tfrecord'))
 234   tf.logging.info('Converting %d test images', len(test_filenames))
 235   writer.write_tf_record(test_filenames,
 236                          os.path.join(FLAGS.output_dir, 'test.tfrecord'))
 237   tf.logging.info('All files are converted')
 238 
 239 
 240 if __name__ == '__main__':
 241   app.run(main)
