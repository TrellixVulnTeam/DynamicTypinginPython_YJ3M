  16 """Program which runs evaluation of Imagenet 64x64 and TinyImagenet models."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import os
  23 
  24 from absl import app
  25 from absl import flags
  26 
  27 import tensorflow as tf
  28 
  29 import adversarial_attack
  30 import model_lib
  31 from datasets import dataset_factory
  32 
  33 FLAGS = flags.FLAGS
  34 
  35 
  36 flags.DEFINE_string('train_dir', None,
  37                     'Training directory. If specified then this program '
  38                     'runs in continuous evaluation mode.')
  39 
  40 flags.DEFINE_string('checkpoint_path', None,
  41                     'Path to the file with checkpoint. If specified then '
  42                     'this program evaluates only provided checkpoint one time.')
  43 
  44 flags.DEFINE_string('output_file', None,
  45                     'Name of output file. Used only in single evaluation mode.')
  46 
  47 flags.DEFINE_string('eval_name', 'default', 'Name for eval subdirectory.')
  48 
  49 flags.DEFINE_string('master', '', 'Tensorflow master.')
  50 
  51 flags.DEFINE_string('model_name', 'resnet_v2_50', 'Name of the model.')
  52 
  53 flags.DEFINE_string('adv_method', 'clean',
  54                     'Method which is used to generate adversarial examples.')
  55 
  56 flags.DEFINE_string('dataset', 'imagenet',
  57                     'Dataset: "tiny_imagenet" or "imagenet".')
  58 
  59 flags.DEFINE_integer('dataset_image_size', 64,
  60                      'Size of the images in the dataset.')
  61 
  62 flags.DEFINE_string('hparams', '', 'Hyper parameters.')
  63 
  64 flags.DEFINE_string('split_name', 'validation', 'Name of the split.')
  65 
  66 flags.DEFINE_float('moving_average_decay', 0.9999,
  67                    'The decay to use for the moving average.')
  68 
  69 flags.DEFINE_integer('eval_interval_secs', 120,
  70                      'The frequency, in seconds, with which evaluation is run.')
  71 
  72 flags.DEFINE_integer(
  73     'num_examples', -1,
  74     'If positive - maximum number of example to use for evaluation.')
  75 
  76 flags.DEFINE_bool('eval_once', False,
  77                   'If true then evaluate model only once.')
  78 
  79 flags.DEFINE_string('trainable_scopes', None,
  80                     'If set then it defines list of variable scopes for '
  81                     'trainable variables.')
  82 
  83 
  84 def main(_):
  85   if not FLAGS.train_dir and not FLAGS.checkpoint_path:
  86     print('Either --train_dir or --checkpoint_path flags has to be provided.')
  87   if FLAGS.train_dir and FLAGS.checkpoint_path:
  88     print('Only one of --train_dir or --checkpoint_path should be provided.')
  89   params = model_lib.default_hparams()
  90   params.parse(FLAGS.hparams)
  91   tf.logging.info('User provided hparams: %s', FLAGS.hparams)
  92   tf.logging.info('All hyper parameters: %s', params)
  93   batch_size = params.eval_batch_size
  94   graph = tf.Graph()
  95   with graph.as_default():
  96     # dataset
  97     dataset, num_examples, num_classes, bounds = dataset_factory.get_dataset(
  98         FLAGS.dataset,
  99         FLAGS.split_name,
 100         batch_size,
 101         FLAGS.dataset_image_size,
 102         is_training=False)
 103     dataset_iterator = dataset.make_one_shot_iterator()
 104     images, labels = dataset_iterator.get_next()
 105     if FLAGS.num_examples > 0:
 106       num_examples = min(num_examples, FLAGS.num_examples)
 107 
 108     # setup model
 109     global_step = tf.train.get_or_create_global_step()
 110     model_fn_two_args = model_lib.get_model(FLAGS.model_name, num_classes)
 111     model_fn = lambda x: model_fn_two_args(x, is_training=False)
 112     if not FLAGS.adv_method or FLAGS.adv_method == 'clean':
 113       logits = model_fn(images)
 114     else:
 115       adv_examples = adversarial_attack.generate_adversarial_examples(
 116           images, bounds, model_fn, FLAGS.adv_method)
 117       logits = model_fn(adv_examples)
 118 
 119     # update trainable variables if fine tuning is used
 120     model_lib.filter_trainable_variables(FLAGS.trainable_scopes)
 121 
 122     # Setup the moving averages
 123     if FLAGS.moving_average_decay and (FLAGS.moving_average_decay > 0):
 124       variable_averages = tf.train.ExponentialMovingAverage(
 125           FLAGS.moving_average_decay, global_step)
 126       variables_to_restore = variable_averages.variables_to_restore(
 127           tf.contrib.framework.get_model_variables())
 128       variables_to_restore[global_step.op.name] = global_step
 129     else:
 130       variables_to_restore = tf.contrib.framework.get_variables_to_restore()
 131 
 132     # Setup evaluation metric
 133     with tf.name_scope('Eval'):
 134       names_to_values, names_to_updates = (
 135           tf.contrib.metrics.aggregate_metric_map({
 136               'Accuracy': tf.metrics.accuracy(labels, tf.argmax(logits, 1)),
 137               'Top5': tf.metrics.recall_at_k(tf.to_int64(labels), logits, 5)
 138           }))
 139 
 140       for name, value in names_to_values.iteritems():
 141         tf.summary.scalar(name, value)
 142 
 143     # Run evaluation
 144     num_batches = int(num_examples / batch_size)
 145     if FLAGS.train_dir:
 146       output_dir = os.path.join(FLAGS.train_dir, FLAGS.eval_name)
 147       if not tf.gfile.Exists(output_dir):
 148         tf.gfile.MakeDirs(output_dir)
 149       tf.contrib.training.evaluate_repeatedly(
 150           FLAGS.train_dir,
 151           master=FLAGS.master,
 152           scaffold=tf.train.Scaffold(
 153               saver=tf.train.Saver(variables_to_restore)),
 154           eval_ops=names_to_updates.values(),
 155           eval_interval_secs=FLAGS.eval_interval_secs,
 156           hooks=[
 157               tf.contrib.training.StopAfterNEvalsHook(num_batches),
 158               tf.contrib.training.SummaryAtEndHook(output_dir),
 159               tf.train.LoggingTensorHook(names_to_values, at_end=True),
 160           ],
 161           max_number_of_evaluations=1 if FLAGS.eval_once else None)
 162     else:
 163       result = tf.contrib.training.evaluate_once(
 164           FLAGS.checkpoint_path,
 165           master=FLAGS.master,
 166           scaffold=tf.train.Scaffold(
 167               saver=tf.train.Saver(variables_to_restore)),
 168           eval_ops=names_to_updates.values(),
 169           final_ops=names_to_values,
 170           hooks=[
 171               tf.contrib.training.StopAfterNEvalsHook(num_batches),
 172               tf.train.LoggingTensorHook(names_to_values, at_end=True),
 173           ])
 174       if FLAGS.output_file:
 175         with tf.gfile.Open(FLAGS.output_file, 'a') as f:
 176           f.write('%s,%.3f,%.3f\n'
 177                   % (FLAGS.eval_name, result['Accuracy'], result['Top5']))
 178 
 179 
 180 if __name__ == '__main__':
 181   app.run(main)
