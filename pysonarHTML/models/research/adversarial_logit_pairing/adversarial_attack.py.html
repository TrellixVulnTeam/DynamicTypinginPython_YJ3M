  16 """Library with adversarial attacks.
  17 
  18 This library designed to be self-contained and have no dependencies other
  19 than TensorFlow. It only contains PGD / Iterative FGSM attacks,
  20 see https://arxiv.org/abs/1706.06083 and https://arxiv.org/abs/1607.02533
  21 for details.
  22 
  23 For wider set of adversarial attacks refer to Cleverhans library:
  24 https://github.com/tensorflow/cleverhans
  25 """

  27 from __future__ import absolute_import
  28 from __future__ import division
  29 from __future__ import print_function

  31 import tensorflow as tf

  34 def generate_pgd_common(x,
  35                         bounds,
  36                         model_fn,
  37                         attack_params,
  38                         one_hot_labels,
  39                         perturbation_multiplier):
  40   """Common code for generating PGD adversarial examples.
  41 
  42   Args:
  43     x: original examples.
  44     bounds: tuple with bounds of image values, bounds[0] < bounds[1].
  45     model_fn: model function with signature model_fn(images).
  46     attack_params: parameters of the attack.
  47     one_hot_labels: one hot label vector to use in the loss.
  48     perturbation_multiplier: multiplier of adversarial perturbation,
  49       either +1.0 or -1.0.
  50 
  51   Returns:
  52     Tensor with adversarial examples.
  53 
  54   Raises:
  55     ValueError: if attack parameters are invalid.
  56   """
  57   # parse attack_params
  58   # Format of attack_params: 'EPS_STEP_NITER'
  59   # where EPS - epsilon, STEP - step size, NITER - number of iterations
  60   params_list = attack_params.split('_')
  61   if len(params_list) != 3:
  62     raise ValueError('Invalid parameters of PGD attack: %s' % attack_params)
  63   epsilon = int(params_list[0])
  64   step_size = int(params_list[1])
  65   niter = int(params_list[2])
  66 
  67   # rescale epsilon and step size to image bounds
  68   epsilon = float(epsilon) / 255.0 * (bounds[1] - bounds[0])
  69   step_size = float(step_size) / 255.0 * (bounds[1] - bounds[0])
  70 
  71   # clipping boundaries
  72   clip_min = tf.maximum(x - epsilon, bounds[0])
  73   clip_max = tf.minimum(x + epsilon, bounds[1])
  74 
  75   # compute starting point
  76   start_x = x + tf.random_uniform(tf.shape(x), -epsilon, epsilon)
  77   start_x = tf.clip_by_value(start_x, clip_min, clip_max)
  78 
  79   # main iteration of PGD
  80   loop_vars = [0, start_x]
  81 
  82   def loop_cond(index, _):
  83     return index < niter
  84 
  85   def loop_body(index, adv_images):
  86     logits = model_fn(adv_images)
  87     loss = tf.reduce_sum(
  88         tf.nn.softmax_cross_entropy_with_logits_v2(
  89             labels=one_hot_labels,
  90             logits=logits))
  91     perturbation = step_size * tf.sign(tf.gradients(loss, adv_images)[0])
  92     new_adv_images = adv_images + perturbation_multiplier * perturbation
  93     new_adv_images = tf.clip_by_value(new_adv_images, clip_min, clip_max)
  94     return index + 1, new_adv_images
  95 
  96   with tf.control_dependencies([start_x]):
  97     _, result = tf.while_loop(
  98         loop_cond,
  99         loop_body,
 100         loop_vars,
 101         back_prop=False,
 102         parallel_iterations=1)
 103     return result

 106 def generate_pgd_ll(x, bounds, model_fn, attack_params):
 107   # pylint: disable=g-doc-args
 108   """Generats targeted PGD adversarial examples with least likely target class.
 109 
 110   See generate_pgd_common for description of arguments.
 111 
 112   Returns:
 113     Tensor with adversarial examples.
 114   """
 115   # pylint: enable=g-doc-args
 116 
 117   # compute one hot least likely class
 118   logits = model_fn(x)
 119   num_classes = tf.shape(logits)[1]
 120   one_hot_labels = tf.one_hot(tf.argmin(model_fn(x), axis=1), num_classes)
 121 
 122   return generate_pgd_common(x, bounds, model_fn, attack_params,
 123                              one_hot_labels=one_hot_labels,
 124                              perturbation_multiplier=-1.0)

 127 def generate_pgd_rand(x, bounds, model_fn, attack_params):
 128   # pylint: disable=g-doc-args
 129   """Generats targeted PGD adversarial examples with random target class.
 130 
 131   See generate_pgd_common for description of arguments.
 132 
 133   Returns:
 134     Tensor with adversarial examples.
 135   """
 136   # pylint: enable=g-doc-args
 137 
 138   # compute one hot random class
 139   logits = model_fn(x)
 140   batch_size = tf.shape(logits)[0]
 141   num_classes = tf.shape(logits)[1]
 142   random_labels = tf.random_uniform(shape=[batch_size],
 143                                     minval=0,
 144                                     maxval=num_classes,
 145                                     dtype=tf.int32)
 146   one_hot_labels = tf.one_hot(random_labels, num_classes)
 147 
 148   return generate_pgd_common(x, bounds, model_fn, attack_params,
 149                              one_hot_labels=one_hot_labels,
 150                              perturbation_multiplier=-1.0)

 153 def generate_pgd(x, bounds, model_fn, attack_params):
 154   # pylint: disable=g-doc-args
 155   """Generats non-targeted PGD adversarial examples.
 156 
 157   See generate_pgd_common for description of arguments.
 158 
 159   Returns:
 160     tensor with adversarial examples.
 161   """
 162   # pylint: enable=g-doc-args
 163 
 164   # compute one hot predicted class
 165   logits = model_fn(x)
 166   num_classes = tf.shape(logits)[1]
 167   one_hot_labels = tf.one_hot(tf.argmax(model_fn(x), axis=1), num_classes)
 168 
 169   return generate_pgd_common(x, bounds, model_fn, attack_params,
 170                              one_hot_labels=one_hot_labels,
 171                              perturbation_multiplier=1.0)

 174 def generate_adversarial_examples(x, bounds, model_fn, attack_description):
 175   """Generates adversarial examples.
 176 
 177   Args:
 178     x: original examples.
 179     bounds: tuple with bounds of image values, bounds[0] < bounds[1]
 180     model_fn: model function with signature model_fn(images).
 181     attack_description: string which describes an attack, see notes below for
 182       details.
 183 
 184   Returns:
 185     Tensor with adversarial examples.
 186 
 187   Raises:
 188     ValueError: if attack description is invalid.
 189 
 190 
 191   Attack description could be one of the following strings:
 192   - "clean" - no attack, return original images.
 193   - "pgd_EPS_STEP_NITER" - non-targeted PGD attack.
 194   - "pgdll_EPS_STEP_NITER" - tageted PGD attack with least likely target class.
 195   - "pgdrnd_EPS_STEP_NITER" - targetd PGD attack with random target class.
 196 
 197   Meaning of attack parameters is following:
 198   - EPS - maximum size of adversarial perturbation, between 0 and 255.
 199   - STEP - step size of one iteration of PGD, between 0 and 255.
 200   - NITER - number of iterations.
 201   """
 202   if attack_description == 'clean':
 203     return x
 204   idx = attack_description.find('_')
 205   if idx < 0:
 206     raise ValueError('Invalid value of attack description %s'
 207                      % attack_description)
 208   attack_name = attack_description[:idx]
 209   attack_params = attack_description[idx+1:]
 210   if attack_name == 'pgdll':
 211     return generate_pgd_ll(x, bounds, model_fn, attack_params)
 212   elif attack_name == 'pgdrnd':
 213     return generate_pgd_rand(x, bounds, model_fn, attack_params)
 214   elif attack_name == 'pgd':
 215     return generate_pgd(x, bounds, model_fn, attack_params)
 216   else:
 217     raise ValueError('Invalid value of attack description %s'
 218                      % attack_description)
