  93 def _get_finetuning_init_fn(variable_averages):
  94   """Returns an init functions, used for fine tuning."""
  95   if not FLAGS.finetune_checkpoint_path:
  96     return None
  97 
  98   if tf.train.latest_checkpoint(FLAGS.output_dir):
  99     return None
 100 
 101   if tf.gfile.IsDirectory(FLAGS.finetune_checkpoint_path):
 102     checkpoint_path = tf.train.latest_checkpoint(FLAGS.finetune_checkpoint_path)
 103   else:
 104     checkpoint_path = FLAGS.finetune_checkpoint_path
 105 
 106   if not checkpoint_path:
 107     tf.logging.warning('Not doing fine tuning, can not find checkpoint in %s',
 108                        FLAGS.finetune_checkpoint_path)
 109     return None
 110 
 111   tf.logging.info('Fine-tuning from %s', checkpoint_path)
 112 
 113   if FLAGS.finetune_exclude_pretrained_scopes:
 114     exclusions = {
 115         scope.strip()
 116         for scope in FLAGS.finetune_exclude_pretrained_scopes.split(',')
 117     }
 118   else:
 119     exclusions = set()
 120 
 121   filtered_model_variables = [
 122       v for v in tf.contrib.framework.get_model_variables()
 123       if not any([v.op.name.startswith(e) for e in exclusions])
 124   ]
 125 
 126   if variable_averages:
 127     variables_to_restore = {}
 128     for v in filtered_model_variables:
 129       # variables_to_restore[variable_averages.average_name(v)] = v
 130       if v in tf.trainable_variables():
 131         variables_to_restore[variable_averages.average_name(v)] = v
 132       else:
 133         variables_to_restore[v.op.name] = v
 134   else:
 135     variables_to_restore = {v.op.name: v for v in filtered_model_variables}
 136 
 137   assign_fn = tf.contrib.framework.assign_from_checkpoint_fn(
 138       checkpoint_path,
 139       variables_to_restore)
 140   if assign_fn:
 141     return lambda _, sess: assign_fn(sess)
 142   else:
 143     return None

 146 def main(_):
 147   assert FLAGS.output_dir, '--output_dir has to be provided'
 148   if not tf.gfile.Exists(FLAGS.output_dir):
 149     tf.gfile.MakeDirs(FLAGS.output_dir)
 150   params = model_lib.default_hparams()
 151   params.parse(FLAGS.hparams)
 152   tf.logging.info('User provided hparams: %s', FLAGS.hparams)
 153   tf.logging.info('All hyper parameters: %s', params)
 154   batch_size = params.batch_size
 155   graph = tf.Graph()
 156   with graph.as_default():
 157     with tf.device(tf.train.replica_device_setter(ps_tasks=FLAGS.ps_tasks)):
 158       # dataset
 159       dataset, examples_per_epoch, num_classes, bounds = (
 160           dataset_factory.get_dataset(
 161               FLAGS.dataset,
 162               'train',
 163               batch_size,
 164               FLAGS.dataset_image_size,
 165               is_training=True))
 166       dataset_iterator = dataset.make_one_shot_iterator()
 167       images, labels = dataset_iterator.get_next()
 168       one_hot_labels = tf.one_hot(labels, num_classes)
 169 
 170       # set up model
 171       global_step = tf.train.get_or_create_global_step()
 172       model_fn = model_lib.get_model(FLAGS.model_name, num_classes)
 173       if params.train_adv_method == 'clean':
 174         logits = model_fn(images, is_training=True)
 175         adv_examples = None
 176       else:
 177         model_fn_eval_mode = lambda x: model_fn(x, is_training=False)
 178         adv_examples = adversarial_attack.generate_adversarial_examples(
 179             images, bounds, model_fn_eval_mode, params.train_adv_method)
 180         all_examples = tf.concat([images, adv_examples], axis=0)
 181         logits = model_fn(all_examples, is_training=True)
 182         one_hot_labels = tf.concat([one_hot_labels, one_hot_labels], axis=0)
 183 
 184       # update trainable variables if fine tuning is used
 185       model_lib.filter_trainable_variables(
 186           FLAGS.finetune_trainable_scopes)
 187 
 188       # set up losses
 189       total_loss = tf.losses.softmax_cross_entropy(
 190           onehot_labels=one_hot_labels,
 191           logits=logits,
 192           label_smoothing=params.label_smoothing)
 193       tf.summary.scalar('loss_xent', total_loss)
 194 
 195       if params.train_lp_weight > 0:
 196         images1, images2 = tf.split(logits, 2)
 197         loss_lp = tf.losses.mean_squared_error(
 198             images1, images2, weights=params.train_lp_weight)
 199         tf.summary.scalar('loss_lp', loss_lp)
 200         total_loss += loss_lp
 201 
 202       if params.weight_decay > 0:
 203         loss_wd = (
 204             params.weight_decay
 205             * tf.add_n([tf.nn.l2_loss(v) for v in tf.trainable_variables()])
 206         )
 207         tf.summary.scalar('loss_wd', loss_wd)
 208         total_loss += loss_wd
 209 
 210       # Setup the moving averages:
 211       if FLAGS.moving_average_decay and (FLAGS.moving_average_decay > 0):
 212         with tf.name_scope('moving_average'):
 213           moving_average_variables = tf.contrib.framework.get_model_variables()
 214           variable_averages = tf.train.ExponentialMovingAverage(
 215               FLAGS.moving_average_decay, global_step)
 216       else:
 217         moving_average_variables = None
 218         variable_averages = None
 219 
 220       # set up optimizer and training op
 221       learning_rate, steps_per_epoch = model_lib.get_lr_schedule(
 222           params, examples_per_epoch, FLAGS.replicas_to_aggregate)
 223 
 224       optimizer = model_lib.get_optimizer(params, learning_rate)
 225 
 226       optimizer = tf.train.SyncReplicasOptimizer(
 227           opt=optimizer,
 228           replicas_to_aggregate=FLAGS.replicas_to_aggregate,
 229           total_num_replicas=FLAGS.worker_replicas,
 230           variable_averages=variable_averages,
 231           variables_to_average=moving_average_variables)
 232 
 233       train_op = tf.contrib.training.create_train_op(
 234           total_loss, optimizer,
 235           update_ops=tf.get_collection(tf.GraphKeys.UPDATE_OPS))
 236 
 237       tf.summary.image('images', images[0:FLAGS.num_summary_images])
 238       if adv_examples is not None:
 239         tf.summary.image('adv_images', adv_examples[0:FLAGS.num_summary_images])
 240       tf.summary.scalar('total_loss', total_loss)
 241       tf.summary.scalar('learning_rate', learning_rate)
 242       tf.summary.scalar('current_epoch',
 243                         tf.to_double(global_step) / steps_per_epoch)
 244 
 245       # Training
 246       is_chief = FLAGS.task == 0
 247 
 248       scaffold = tf.train.Scaffold(
 249           init_fn=_get_finetuning_init_fn(variable_averages))
 250       hooks = [
 251           tf.train.LoggingTensorHook({'total_loss': total_loss,
 252                                       'global_step': global_step},
 253                                      every_n_iter=1),
 254           tf.train.NanTensorHook(total_loss),
 255       ]
 256       chief_only_hooks = [
 257           tf.train.SummarySaverHook(save_steps=FLAGS.save_summaries_steps,
 258                                     save_secs=FLAGS.save_summaries_secs,
 259                                     output_dir=FLAGS.output_dir,
 260                                     scaffold=scaffold),
 261           tf.train.CheckpointSaverHook(FLAGS.output_dir,
 262                                        save_steps=FLAGS.save_model_steps,
 263                                        scaffold=scaffold),
 264       ]
 265 
 266       if FLAGS.max_steps > 0:
 267         hooks.append(
 268             tf.train.StopAtStepHook(last_step=FLAGS.max_steps))
 269 
 270       # hook for sync replica training
 271       hooks.append(optimizer.make_session_run_hook(is_chief))
 272 
 273       with tf.train.MonitoredTrainingSession(
 274           master=FLAGS.master,
 275           is_chief=is_chief,
 276           checkpoint_dir=FLAGS.output_dir,
 277           scaffold=scaffold,
 278           hooks=hooks,
 279           chief_only_hooks=chief_only_hooks,
 280           save_checkpoint_secs=None,
 281           save_summaries_steps=None,
 282           save_summaries_secs=None) as session:
 283         while not session.should_stop():
 284           session.run([train_op])
 285 
 286 
 287 if __name__ == '__main__':
 288   app.run(main)
