  44 # TODO(adai): IMDB labels placeholder to model.
  45 def create_generator(hparams,
  46                      inputs,
  47                      targets,
  48                      present,
  49                      is_training,
  50                      is_validating,
  51                      reuse=None):
  52   """Create the Generator model specified by the FLAGS and hparams.
  53 
  54   Args;
  55     hparams:  Hyperparameters for the MaskGAN.
  56     inputs:  tf.int32 Tensor of the sequence input of shape [batch_size,
  57       sequence_length].
  58     present:  tf.bool Tensor indicating the presence or absence of the token
  59       of shape [batch_size, sequence_length].
  60     is_training:  Whether the model is training.
  61     is_validating:  Whether the model is being run in validation mode for
  62       calculating the perplexity.
  63     reuse (Optional):  Whether to reuse the model.
  64 
  65   Returns:
  66     Tuple of the (sequence, logits, log_probs) of the Generator.   Sequence
  67       and logits have shape [batch_size, sequence_length, vocab_size].  The
  68       log_probs will have shape [batch_size, sequence_length].  Log_probs
  69       corresponds to the log probability of selecting the words.
  70   """
  71   if FLAGS.generator_model == 'rnn':
  72     (sequence, logits, log_probs, initial_state, final_state) = rnn.generator(
  73         hparams,
  74         inputs,
  75         targets,
  76         present,
  77         is_training=is_training,
  78         is_validating=is_validating,
  79         reuse=reuse)
  80   elif FLAGS.generator_model == 'rnn_zaremba':
  81     (sequence, logits, log_probs, initial_state,
  82      final_state) = rnn_zaremba.generator(
  83          hparams,
  84          inputs,
  85          targets,
  86          present,
  87          is_training=is_training,
  88          is_validating=is_validating,
  89          reuse=reuse)
  90   elif FLAGS.generator_model == 'seq2seq':
  91     (sequence, logits, log_probs, initial_state,
  92      final_state) = seq2seq.generator(
  93          hparams,
  94          inputs,
  95          targets,
  96          present,
  97          is_training=is_training,
  98          is_validating=is_validating,
  99          reuse=reuse)
 100   elif FLAGS.generator_model == 'seq2seq_zaremba':
 101     (sequence, logits, log_probs, initial_state,
 102      final_state) = seq2seq_zaremba.generator(
 103          hparams,
 104          inputs,
 105          targets,
 106          present,
 107          is_training=is_training,
 108          is_validating=is_validating,
 109          reuse=reuse)
 110   elif FLAGS.generator_model == 'rnn_nas':
 111     (sequence, logits, log_probs, initial_state,
 112      final_state) = rnn_nas.generator(
 113          hparams,
 114          inputs,
 115          targets,
 116          present,
 117          is_training=is_training,
 118          is_validating=is_validating,
 119          reuse=reuse)
 120   elif FLAGS.generator_model == 'seq2seq_nas':
 121     (sequence, logits, log_probs, initial_state,
 122      final_state) = seq2seq_nas.generator(
 123          hparams,
 124          inputs,
 125          targets,
 126          present,
 127          is_training=is_training,
 128          is_validating=is_validating,
 129          reuse=reuse)
 130   elif FLAGS.generator_model == 'seq2seq_vd':
 131     (sequence, logits, log_probs, initial_state, final_state,
 132      encoder_states) = seq2seq_vd.generator(
 133          hparams,
 134          inputs,
 135          targets,
 136          present,
 137          is_training=is_training,
 138          is_validating=is_validating,
 139          reuse=reuse)
 140   else:
 141     raise NotImplementedError
 142   return (sequence, logits, log_probs, initial_state, final_state,
 143           encoder_states)

 146 def create_discriminator(hparams,
 147                          sequence,
 148                          is_training,
 149                          reuse=None,
 150                          initial_state=None,
 151                          inputs=None,
 152                          present=None):
 153   """Create the Discriminator model specified by the FLAGS and hparams.
 154 
 155   Args:
 156     hparams:  Hyperparameters for the MaskGAN.
 157     sequence:  tf.int32 Tensor sequence of shape [batch_size, sequence_length]
 158     is_training:  Whether the model is training.
 159     reuse (Optional):  Whether to reuse the model.
 160 
 161   Returns:
 162     predictions:  tf.float32 Tensor of predictions of shape [batch_size,
 163       sequence_length]
 164   """
 165   if FLAGS.discriminator_model == 'cnn':
 166     predictions = cnn.discriminator(
 167         hparams, sequence, is_training=is_training, reuse=reuse)
 168   elif FLAGS.discriminator_model == 'fnn':
 169     predictions = feedforward.discriminator(
 170         hparams, sequence, is_training=is_training, reuse=reuse)
 171   elif FLAGS.discriminator_model == 'rnn':
 172     predictions = rnn.discriminator(
 173         hparams, sequence, is_training=is_training, reuse=reuse)
 174   elif FLAGS.discriminator_model == 'bidirectional':
 175     predictions = bidirectional.discriminator(
 176         hparams, sequence, is_training=is_training, reuse=reuse)
 177   elif FLAGS.discriminator_model == 'bidirectional_zaremba':
 178     predictions = bidirectional_zaremba.discriminator(
 179         hparams, sequence, is_training=is_training, reuse=reuse)
 180   elif FLAGS.discriminator_model == 'seq2seq_vd':
 181     predictions = seq2seq_vd.discriminator(
 182         hparams,
 183         inputs,
 184         present,
 185         sequence,
 186         is_training=is_training,
 187         reuse=reuse)
 188   elif FLAGS.discriminator_model == 'rnn_zaremba':
 189     predictions = rnn_zaremba.discriminator(
 190         hparams, sequence, is_training=is_training, reuse=reuse)
 191   elif FLAGS.discriminator_model == 'rnn_nas':
 192     predictions = rnn_nas.discriminator(
 193         hparams, sequence, is_training=is_training, reuse=reuse)
 194   elif FLAGS.discriminator_model == 'rnn_vd':
 195     predictions = rnn_vd.discriminator(
 196         hparams,
 197         sequence,
 198         is_training=is_training,
 199         reuse=reuse,
 200         initial_state=initial_state)
 201   elif FLAGS.discriminator_model == 'bidirectional_vd':
 202     predictions = bidirectional_vd.discriminator(
 203         hparams,
 204         sequence,
 205         is_training=is_training,
 206         reuse=reuse,
 207         initial_state=initial_state)
 208   else:
 209     raise NotImplementedError
 210   return predictions

 213 def create_critic(hparams, sequence, is_training, reuse=None):
 214   """Create the Critic model specified by the FLAGS and hparams.
 215 
 216   Args:
 217     hparams:  Hyperparameters for the MaskGAN.
 218     sequence:  tf.int32 Tensor sequence of shape [batch_size, sequence_length]
 219     is_training:  Whether the model is training.
 220     reuse (Optional):  Whether to reuse the model.
 221 
 222   Returns:
 223     values:  tf.float32 Tensor of predictions of shape [batch_size,
 224       sequence_length]
 225   """
 226   if FLAGS.baseline_method == 'critic':
 227     if FLAGS.discriminator_model == 'seq2seq_vd':
 228       values = critic_vd.critic_seq2seq_vd_derivative(
 229           hparams, sequence, is_training, reuse=reuse)
 230     else:
 231       raise NotImplementedError
 232   else:
 233     raise NotImplementedError
 234   return values
