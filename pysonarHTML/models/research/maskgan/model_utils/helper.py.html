  29 def variable_summaries(var, name):
  30   """Attach a lot of summaries to a Tensor."""
  31   mean = tf.reduce_mean(var)
  32   tf.summary.scalar('mean/' + name, mean)
  33   with tf.name_scope('stddev'):
  34     stddev = tf.sqrt(tf.reduce_sum(tf.square(var - mean)))
  35   tf.summary.scalar('sttdev/' + name, stddev)
  36   tf.summary.scalar('max/' + name, tf.reduce_max(var))
  37   tf.summary.scalar('min/' + name, tf.reduce_min(var))
  38   tf.summary.histogram(name, var)

  41 def zip_seq_pred_crossent(id_to_word, sequences, predictions, cross_entropy):
  42   """Zip together the sequences, predictions, cross entropy."""
  43   indices = convert_to_indices(sequences)
  44 
  45   batch_of_metrics = []
  46 
  47   for ind_batch, pred_batch, crossent_batch in zip(indices, predictions,
  48                                                    cross_entropy):
  49     metrics = []
  50 
  51     for index, pred, crossent in zip(ind_batch, pred_batch, crossent_batch):
  52       metrics.append([str(id_to_word[index]), pred, crossent])
  53 
  54     batch_of_metrics.append(metrics)
  55   return batch_of_metrics

  58 def print_and_log(log, id_to_word, sequence_eval, max_num_to_print=5):
  59   """Helper function for printing and logging evaluated sequences."""
  60   indices_eval = convert_to_indices(sequence_eval)
  61   indices_arr = np.asarray(indices_eval)
  62   samples = convert_to_human_readable(id_to_word, indices_arr, max_num_to_print)
  63 
  64   for i, sample in enumerate(samples):
  65     print('Sample', i, '. ', sample)
  66     log.write('\nSample ' + str(i) + '. ' + sample)
  67   log.write('\n')
  68   print('\n')
  69   log.flush()

  72 def convert_to_human_readable(id_to_word, arr, max_num_to_print):
  73   """Convert a np.array of indices into words using id_to_word dictionary.
  74   Return max_num_to_print results.
  75   """
  76   assert arr.ndim == 2
  77 
  78   samples = []
  79   for sequence_id in xrange(min(len(arr), max_num_to_print)):
  80     buffer_str = ' '.join(
  81         [str(id_to_word[index]) for index in arr[sequence_id, :]])
  82     samples.append(buffer_str)
  83   return samples

  86 def index_to_vocab_array(indices, vocab_size, sequence_length):
  87   """Convert the indices into an array with vocab_size one-hot encoding."""
  88 
  89   # Extract properties of the indices.
  90   num_batches = len(indices)
  91   shape = list(indices.shape)
  92   shape.append(vocab_size)
  93 
  94   # Construct the vocab_size array.
  95   new_arr = np.zeros(shape)
  96 
  97   for n in xrange(num_batches):
  98     indices_batch = indices[n]
  99     new_arr_batch = new_arr[n]
 100 
 101     # We map all indices greater than the vocabulary size to an unknown
 102     # character.
 103     indices_batch = np.where(indices_batch < vocab_size, indices_batch,
 104                              vocab_size - 1)
 105 
 106     # Convert indices to vocab_size dimensions.
 107     new_arr_batch[np.arange(sequence_length), indices_batch] = 1
 108   return new_arr

 111 def convert_to_indices(sequences):
 112   """Convert a list of size [batch_size, sequence_length, vocab_size] to
 113   a list of size [batch_size, sequence_length] where the vocab element is
 114   denoted by the index.
 115   """
 116   batch_of_indices = []
 117 
 118   for sequence in sequences:
 119     indices = []
 120     for embedding in sequence:
 121       indices.append(np.argmax(embedding))
 122     batch_of_indices.append(indices)
 123   return batch_of_indices

 126 def convert_and_zip(id_to_word, sequences, predictions):
 127   """Helper function for printing or logging.  Retrieves list of sequences
 128   and predictions and zips them together.
 129   """
 130   indices = convert_to_indices(sequences)
 131 
 132   batch_of_indices_predictions = []
 133 
 134   for index_batch, pred_batch in zip(indices, predictions):
 135     indices_predictions = []
 136 
 137     for index, pred in zip(index_batch, pred_batch):
 138       indices_predictions.append([str(id_to_word[index]), pred])
 139     batch_of_indices_predictions.append(indices_predictions)
 140   return batch_of_indices_predictions

 143 def recursive_length(item):
 144   """Recursively determine the total number of elements in nested list."""
 145   if type(item) == list:
 146     return sum(recursive_length(subitem) for subitem in item)
 147   else:
 148     return 1.

 151 def percent_correct(real_sequence, fake_sequences):
 152   """Determine the percent of tokens correctly generated within a batch."""
 153   identical = 0.
 154   for fake_sequence in fake_sequences:
 155     for real, fake in zip(real_sequence, fake_sequence):
 156       if real == fake:
 157         identical += 1.
 158   return identical / recursive_length(fake_sequences)
