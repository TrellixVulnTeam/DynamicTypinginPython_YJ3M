  31 def generate_mask():
  32   """Generate the mask to be fed into the model."""
  33   if FLAGS.mask_strategy == 'random':
  34     p = np.random.choice(
  35         [True, False],
  36         size=[FLAGS.batch_size, FLAGS.sequence_length],
  37         p=[FLAGS.is_present_rate, 1. - FLAGS.is_present_rate])
  38 
  39   elif FLAGS.mask_strategy == 'contiguous':
  40     masked_length = int((1 - FLAGS.is_present_rate) * FLAGS.sequence_length) - 1
  41     # Determine location to start masking.
  42     start_mask = np.random.randint(
  43         1, FLAGS.sequence_length - masked_length + 1, size=FLAGS.batch_size)
  44     p = np.full([FLAGS.batch_size, FLAGS.sequence_length], True, dtype=bool)
  45 
  46     # Create contiguous masked section to be False.
  47     for i, index in enumerate(start_mask):
  48       p[i, index:index + masked_length] = False
  49 
  50   else:
  51     raise NotImplementedError
  52 
  53   return p

  56 def assign_percent_real(session, percent_real_update, new_rate, current_rate):
  57   """Run assign operation where the we load the current_rate of percent
  58   real into a Tensorflow variable.
  59 
  60   Args:
  61     session:  Current tf.Session.
  62     percent_real_update: tf.assign operation.
  63     new_rate: tf.placeholder for the new rate.
  64     current_rate: Percent of tokens that are currently real.  Fake tokens
  65       are the ones being imputed by the Generator.
  66   """
  67   session.run(percent_real_update, feed_dict={new_rate: current_rate})

  83 def clip_weights(variables, c_lower, c_upper):
  84   """Clip a list of weights to be within a certain range.
  85 
  86   Args:
  87     variables: List of tf.Variable weights.
  88     c_lower: Lower bound for weights.
  89     c_upper: Upper bound for weights.
  90   """
  91   clip_ops = []
  92 
  93   for var in variables:
  94     clipped_var = tf.clip_by_value(var, c_lower, c_upper)
  95 
  96     clip_ops.append(tf.assign(var, clipped_var))
  97   return tf.group(*clip_ops)

 100 def retrieve_init_savers(hparams):
 101   """Retrieve a dictionary of all the initial savers for the models.
 102 
 103   Args:
 104     hparams:  MaskGAN hyperparameters.
 105   """
 106   ## Dictionary of init savers.
 107   init_savers = {}
 108 
 109   ## Load Generator weights from MaskGAN checkpoint.
 110   if FLAGS.maskgan_ckpt:
 111     gen_vars = [
 112         v for v in tf.trainable_variables() if v.op.name.startswith('gen')
 113     ]
 114     init_saver = tf.train.Saver(var_list=gen_vars)
 115     init_savers['init_saver'] = init_saver
 116 
 117     ## Load the Discriminator weights from the MaskGAN checkpoint if
 118     # the weights are compatible.
 119     if FLAGS.discriminator_model == 'seq2seq_vd':
 120       dis_variable_maps = variable_mapping.dis_seq2seq_vd(hparams)
 121       dis_init_saver = tf.train.Saver(var_list=dis_variable_maps)
 122       init_savers['dis_init_saver'] = dis_init_saver
 123 
 124   ## Load weights from language model checkpoint.
 125   if FLAGS.language_model_ckpt_dir:
 126     if FLAGS.maskgan_ckpt is None:
 127       ## Generator Variables/Savers.
 128       if FLAGS.generator_model == 'rnn_nas':
 129         gen_variable_maps = variable_mapping.rnn_nas(hparams, model='gen')
 130         gen_init_saver = tf.train.Saver(var_list=gen_variable_maps)
 131         init_savers['gen_init_saver'] = gen_init_saver
 132 
 133       elif FLAGS.generator_model == 'seq2seq_nas':
 134         # Encoder.
 135         gen_encoder_variable_maps = variable_mapping.gen_encoder_seq2seq_nas(
 136             hparams)
 137         gen_encoder_init_saver = tf.train.Saver(
 138             var_list=gen_encoder_variable_maps)
 139         # Decoder.
 140         gen_decoder_variable_maps = variable_mapping.gen_decoder_seq2seq_nas(
 141             hparams)
 142         gen_decoder_init_saver = tf.train.Saver(
 143             var_list=gen_decoder_variable_maps)
 144         init_savers['gen_encoder_init_saver'] = gen_encoder_init_saver
 145         init_savers['gen_decoder_init_saver'] = gen_decoder_init_saver
 146 
 147       # seq2seq_vd derived from the same code base as seq2seq_zaremba.
 148       elif (FLAGS.generator_model == 'seq2seq_zaremba' or
 149             FLAGS.generator_model == 'seq2seq_vd'):
 150         # Encoder.
 151         gen_encoder_variable_maps = variable_mapping.gen_encoder_seq2seq(
 152             hparams)
 153         gen_encoder_init_saver = tf.train.Saver(
 154             var_list=gen_encoder_variable_maps)
 155         # Decoder.
 156         gen_decoder_variable_maps = variable_mapping.gen_decoder_seq2seq(
 157             hparams)
 158         gen_decoder_init_saver = tf.train.Saver(
 159             var_list=gen_decoder_variable_maps)
 160         init_savers['gen_encoder_init_saver'] = gen_encoder_init_saver
 161         init_savers['gen_decoder_init_saver'] = gen_decoder_init_saver
 162 
 163       else:
 164         raise NotImplementedError
 165 
 166     ## Discriminator Variables/Savers.
 167     if FLAGS.discriminator_model == 'rnn_nas':
 168       dis_variable_maps = variable_mapping.rnn_nas(hparams, model='dis')
 169       dis_init_saver = tf.train.Saver(var_list=dis_variable_maps)
 170       init_savers['dis_init_saver'] = dis_init_saver
 171 
 172     # rnn_vd derived from the same code base as rnn_zaremba.
 173     elif (FLAGS.discriminator_model == 'rnn_zaremba' or
 174           FLAGS.discriminator_model == 'rnn_vd'):
 175       dis_variable_maps = variable_mapping.rnn_zaremba(hparams, model='dis')
 176       dis_init_saver = tf.train.Saver(var_list=dis_variable_maps)
 177       init_savers['dis_init_saver'] = dis_init_saver
 178 
 179     elif (FLAGS.discriminator_model == 'bidirectional_zaremba' or
 180           FLAGS.discriminator_model == 'bidirectional_vd'):
 181       dis_fwd_variable_maps = variable_mapping.dis_fwd_bidirectional(hparams)
 182       dis_bwd_variable_maps = variable_mapping.dis_bwd_bidirectional(hparams)
 183       # Savers for the forward/backward Discriminator components.
 184       dis_fwd_init_saver = tf.train.Saver(var_list=dis_fwd_variable_maps)
 185       dis_bwd_init_saver = tf.train.Saver(var_list=dis_bwd_variable_maps)
 186       init_savers['dis_fwd_init_saver'] = dis_fwd_init_saver
 187       init_savers['dis_bwd_init_saver'] = dis_bwd_init_saver
 188 
 189     elif FLAGS.discriminator_model == 'cnn':
 190       dis_variable_maps = variable_mapping.cnn()
 191       dis_init_saver = tf.train.Saver(var_list=dis_variable_maps)
 192       init_savers['dis_init_saver'] = dis_init_saver
 193 
 194     elif FLAGS.discriminator_model == 'seq2seq_vd':
 195       # Encoder.
 196       dis_encoder_variable_maps = variable_mapping.dis_encoder_seq2seq(hparams)
 197       dis_encoder_init_saver = tf.train.Saver(
 198           var_list=dis_encoder_variable_maps)
 199       # Decoder.
 200       dis_decoder_variable_maps = variable_mapping.dis_decoder_seq2seq(hparams)
 201       dis_decoder_init_saver = tf.train.Saver(
 202           var_list=dis_decoder_variable_maps)
 203       init_savers['dis_encoder_init_saver'] = dis_encoder_init_saver
 204       init_savers['dis_decoder_init_saver'] = dis_decoder_init_saver
 205 
 206   return init_savers

 209 def init_fn(init_savers, sess):
 210   """The init_fn to be passed to the Supervisor.
 211 
 212   Args:
 213     init_savers:  Dictionary of init_savers.  'init_saver_name': init_saver.
 214     sess:  tf.Session.
 215   """
 216   ## Load Generator weights from MaskGAN checkpoint.
 217   if FLAGS.maskgan_ckpt:
 218     print('Restoring Generator from %s.' % FLAGS.maskgan_ckpt)
 219     tf.logging.info('Restoring Generator from %s.' % FLAGS.maskgan_ckpt)
 220     print('Asserting Generator is a seq2seq-variant.')
 221     tf.logging.info('Asserting Generator is a seq2seq-variant.')
 222     assert FLAGS.generator_model.startswith('seq2seq')
 223     init_saver = init_savers['init_saver']
 224     init_saver.restore(sess, FLAGS.maskgan_ckpt)
 225 
 226     ## Load the Discriminator weights from the MaskGAN checkpoint if
 227     # the weights are compatible.
 228     if FLAGS.discriminator_model == 'seq2seq_vd':
 229       print('Restoring Discriminator from %s.' % FLAGS.maskgan_ckpt)
 230       tf.logging.info('Restoring Discriminator from %s.' % FLAGS.maskgan_ckpt)
 231       dis_init_saver = init_savers['dis_init_saver']
 232       dis_init_saver.restore(sess, FLAGS.maskgan_ckpt)
 233 
 234   ## Load weights from language model checkpoint.
 235   if FLAGS.language_model_ckpt_dir:
 236     if FLAGS.maskgan_ckpt is None:
 237       ## Generator Models.
 238       if FLAGS.generator_model == 'rnn_nas':
 239         load_ckpt = tf.train.latest_checkpoint(FLAGS.language_model_ckpt_dir)
 240         print('Restoring Generator from %s.' % load_ckpt)
 241         tf.logging.info('Restoring Generator from %s.' % load_ckpt)
 242         gen_init_saver = init_savers['gen_init_saver']
 243         gen_init_saver.restore(sess, load_ckpt)
 244 
 245       elif FLAGS.generator_model.startswith('seq2seq'):
 246         load_ckpt = tf.train.latest_checkpoint(FLAGS.language_model_ckpt_dir)
 247         print('Restoring Generator from %s.' % load_ckpt)
 248         tf.logging.info('Restoring Generator from %s.' % load_ckpt)
 249         gen_encoder_init_saver = init_savers['gen_encoder_init_saver']
 250         gen_decoder_init_saver = init_savers['gen_decoder_init_saver']
 251         gen_encoder_init_saver.restore(sess, load_ckpt)
 252         gen_decoder_init_saver.restore(sess, load_ckpt)
 253 
 254     ## Discriminator Models.
 255     if (FLAGS.discriminator_model == 'rnn_nas' or
 256         FLAGS.discriminator_model == 'rnn_zaremba' or
 257         FLAGS.discriminator_model == 'rnn_vd' or
 258         FLAGS.discriminator_model == 'cnn'):
 259       load_ckpt = tf.train.latest_checkpoint(FLAGS.language_model_ckpt_dir)
 260       print('Restoring Discriminator from %s.' % load_ckpt)
 261       tf.logging.info('Restoring Discriminator from %s.' % load_ckpt)
 262       dis_init_saver = init_savers['dis_init_saver']
 263       dis_init_saver.restore(sess, load_ckpt)
 264 
 265     elif (FLAGS.discriminator_model == 'bidirectional_zaremba' or
 266           FLAGS.discriminator_model == 'bidirectional_vd'):
 267       assert FLAGS.language_model_ckpt_dir_reversed is not None, (
 268           'Need a reversed directory to fill in the backward components.')
 269       load_fwd_ckpt = tf.train.latest_checkpoint(FLAGS.language_model_ckpt_dir)
 270       load_bwd_ckpt = tf.train.latest_checkpoint(
 271           FLAGS.language_model_ckpt_dir_reversed)
 272       print('Restoring Discriminator from %s and %s.' % (load_fwd_ckpt,
 273                                                          load_bwd_ckpt))
 274       tf.logging.info('Restoring Discriminator from %s and %s.' %
 275                       (load_fwd_ckpt, load_bwd_ckpt))
 276       dis_fwd_init_saver = init_savers['dis_fwd_init_saver']
 277       dis_bwd_init_saver = init_savers['dis_bwd_init_saver']
 278       dis_fwd_init_saver.restore(sess, load_fwd_ckpt)
 279       dis_bwd_init_saver.restore(sess, load_bwd_ckpt)
 280 
 281     elif FLAGS.discriminator_model == 'seq2seq_vd':
 282       load_ckpt = tf.train.latest_checkpoint(FLAGS.language_model_ckpt_dir)
 283       print('Restoring Discriminator from %s.' % load_ckpt)
 284       tf.logging.info('Restoring Discriminator from %s.' % load_ckpt)
 285       dis_encoder_init_saver = init_savers['dis_encoder_init_saver']
 286       dis_decoder_init_saver = init_savers['dis_decoder_init_saver']
 287       dis_encoder_init_saver.restore(sess, load_ckpt)
 288       dis_decoder_init_saver.restore(sess, load_ckpt)
 289 
 290   else:
 291     return
