  27 def rnn_nas(hparams, model):
  28   assert model == 'gen' or model == 'dis'
  29 
  30   # This logic is only valid for rnn_zaremba
  31   if model == 'gen':
  32     assert FLAGS.generator_model == 'rnn_nas'
  33     assert hparams.gen_num_layers == 2
  34 
  35   if model == 'dis':
  36     assert FLAGS.discriminator_model == 'rnn_nas'
  37     assert hparams.dis_num_layers == 2
  38 
  39   # Output variables only for the Generator.  Discriminator output biases
  40   # will begin randomly initialized.
  41   if model == 'gen':
  42     softmax_b = [
  43         v for v in tf.trainable_variables() if v.op.name == 'gen/rnn/softmax_b'
  44     ][0]
  45 
  46   # Common elements to Generator and Discriminator.
  47   embedding = [
  48       v for v in tf.trainable_variables()
  49       if v.op.name == str(model) + '/rnn/embedding'
  50   ][0]
  51   lstm_w_0 = [
  52       v for v in tf.trainable_variables()
  53       if v.op.name ==
  54       str(model) + '/rnn/GenericMultiRNNCell/Cell0/Alien/rnn_builder/big_h_mat'
  55   ][0]
  56   lstm_b_0 = [
  57       v for v in tf.trainable_variables()
  58       if v.op.name == str(model) +
  59       '/rnn/GenericMultiRNNCell/Cell0/Alien/rnn_builder/big_inputs_mat'
  60   ][0]
  61   lstm_w_1 = [
  62       v for v in tf.trainable_variables()
  63       if v.op.name ==
  64       str(model) + '/rnn/GenericMultiRNNCell/Cell1/Alien/rnn_builder/big_h_mat'
  65   ][0]
  66   lstm_b_1 = [
  67       v for v in tf.trainable_variables()
  68       if v.op.name == str(model) +
  69       '/rnn/GenericMultiRNNCell/Cell1/Alien/rnn_builder/big_inputs_mat'
  70   ][0]
  71 
  72   # Dictionary mapping.
  73   if model == 'gen':
  74     variable_mapping = {
  75         'Model/embeddings/input_embedding':
  76             embedding,
  77         'Model/RNN/GenericMultiRNNCell/Cell0/Alien/rnn_builder/big_h_mat':
  78             lstm_w_0,
  79         'Model/RNN/GenericMultiRNNCell/Cell0/Alien/rnn_builder/big_inputs_mat':
  80             lstm_b_0,
  81         'Model/RNN/GenericMultiRNNCell/Cell1/Alien/rnn_builder/big_h_mat':
  82             lstm_w_1,
  83         'Model/RNN/GenericMultiRNNCell/Cell1/Alien/rnn_builder/big_inputs_mat':
  84             lstm_b_1,
  85         'Model/softmax_b':
  86             softmax_b
  87     }
  88   else:
  89     variable_mapping = {
  90         'Model/embeddings/input_embedding':
  91             embedding,
  92         'Model/RNN/GenericMultiRNNCell/Cell0/Alien/rnn_builder/big_h_mat':
  93             lstm_w_0,
  94         'Model/RNN/GenericMultiRNNCell/Cell0/Alien/rnn_builder/big_inputs_mat':
  95             lstm_b_0,
  96         'Model/RNN/GenericMultiRNNCell/Cell1/Alien/rnn_builder/big_h_mat':
  97             lstm_w_1,
  98         'Model/RNN/GenericMultiRNNCell/Cell1/Alien/rnn_builder/big_inputs_mat':
  99             lstm_b_1
 100     }
 101

 125 def rnn_zaremba(hparams, model):
 126   """Returns the PTB Variable name to MaskGAN Variable dictionary mapping.  This
 127   is a highly restrictive function just for testing.  This will need to be
 128   generalized.
 129 
 130   Args:
 131     hparams:  Hyperparameters for the MaskGAN.
 132     model:  Model type, one of ['gen', 'dis'].
 133 
 134   Returns:
 135     variable_mapping:  Dictionary with Key: ckpt_name, Value: model_var.
 136   """
 137   assert model == 'gen' or model == 'dis'
 138 
 139   # This logic is only valid for rnn_zaremba
 140   if model == 'gen':
 141     assert FLAGS.generator_model == 'rnn_zaremba'
 142     assert hparams.gen_num_layers == 2
 143 
 144   if model == 'dis':
 145     assert (FLAGS.discriminator_model == 'rnn_zaremba' or
 146             FLAGS.discriminator_model == 'rnn_vd')
 147     assert hparams.dis_num_layers == 2
 148 
 149   # Output variables only for the Generator.  Discriminator output weights
 150   # and biases will begin randomly initialized.
 151   if model == 'gen':
 152     softmax_w = [
 153         v for v in tf.trainable_variables() if v.op.name == 'gen/rnn/softmax_w'
 154     ][0]
 155     softmax_b = [
 156         v for v in tf.trainable_variables() if v.op.name == 'gen/rnn/softmax_b'
 157     ][0]
 158 
 159   # Common elements to Generator and Discriminator.
 160   if not FLAGS.dis_share_embedding or model != 'dis':
 161     embedding = [
 162         v for v in tf.trainable_variables()
 163         if v.op.name == str(model) + '/rnn/embedding'
 164     ][0]
 165   lstm_w_0 = [
 166       v for v in tf.trainable_variables() if v.op.name == str(model) +
 167       '/rnn/multi_rnn_cell/cell_0/basic_lstm_cell/kernel'
 168   ][0]
 169   lstm_b_0 = [
 170       v for v in tf.trainable_variables() if v.op.name == str(model) +
 171       '/rnn/multi_rnn_cell/cell_0/basic_lstm_cell/bias'
 172   ][0]
 173   lstm_w_1 = [
 174       v for v in tf.trainable_variables() if v.op.name == str(model) +
 175       '/rnn/multi_rnn_cell/cell_1/basic_lstm_cell/kernel'
 176   ][0]
 177   lstm_b_1 = [
 178       v for v in tf.trainable_variables() if v.op.name == str(model) +
 179       '/rnn/multi_rnn_cell/cell_1/basic_lstm_cell/bias'
 180   ][0]
 181 
 182   # Dictionary mapping.
 183   if model == 'gen':
 184     variable_mapping = {
 185         'Model/embedding': embedding,
 186         'Model/RNN/multi_rnn_cell/cell_0/basic_lstm_cell/kernel': lstm_w_0,
 187         'Model/RNN/multi_rnn_cell/cell_0/basic_lstm_cell/bias': lstm_b_0,
 188         'Model/RNN/multi_rnn_cell/cell_1/basic_lstm_cell/kernel': lstm_w_1,
 189         'Model/RNN/multi_rnn_cell/cell_1/basic_lstm_cell/bias': lstm_b_1,
 190         'Model/softmax_w': softmax_w,
 191         'Model/softmax_b': softmax_b
 192     }
 193   else:
 194     if FLAGS.dis_share_embedding:
 195       variable_mapping = {
 196           'Model/RNN/multi_rnn_cell/cell_0/basic_lstm_cell/kernel': lstm_w_0,
 197           'Model/RNN/multi_rnn_cell/cell_0/basic_lstm_cell/bias': lstm_b_0,
 198           'Model/RNN/multi_rnn_cell/cell_1/basic_lstm_cell/kernel': lstm_w_1,
 199           'Model/RNN/multi_rnn_cell/cell_1/basic_lstm_cell/bias': lstm_b_1
 200       }
 201     else:
 202       variable_mapping = {
 203           'Model/embedding': embedding,
 204           'Model/RNN/multi_rnn_cell/cell_0/basic_lstm_cell/kernel': lstm_w_0,
 205           'Model/RNN/multi_rnn_cell/cell_0/basic_lstm_cell/bias': lstm_b_0,
 206           'Model/RNN/multi_rnn_cell/cell_1/basic_lstm_cell/kernel': lstm_w_1,
 207           'Model/RNN/multi_rnn_cell/cell_1/basic_lstm_cell/bias': lstm_b_1
 208       }
 209 
 210   return variable_mapping
