  33 def create_dis_loss(fake_predictions, real_predictions, targets_present):
  34   """Compute Discriminator loss across real/fake."""
  35 
  36   missing = tf.cast(targets_present, tf.int32)
  37   missing = 1 - missing
  38   missing = tf.cast(missing, tf.bool)
  39 
  40   real_labels = tf.ones([FLAGS.batch_size, FLAGS.sequence_length])
  41   dis_loss_real = tf.losses.sigmoid_cross_entropy(
  42       real_labels, real_predictions, weights=missing)
  43   dis_loss_fake = tf.losses.sigmoid_cross_entropy(
  44       targets_present, fake_predictions, weights=missing)
  45 
  46   dis_loss = (dis_loss_fake + dis_loss_real) / 2.
  47   return dis_loss, dis_loss_fake, dis_loss_real

  50 def create_critic_loss(cumulative_rewards, estimated_values, present):
  51   """Compute Critic loss in estimating the value function.  This should be an
  52   estimate only for the missing elements."""
  53   missing = tf.cast(present, tf.int32)
  54   missing = 1 - missing
  55   missing = tf.cast(missing, tf.bool)
  56 
  57   loss = tf.losses.mean_squared_error(
  58       labels=cumulative_rewards, predictions=estimated_values, weights=missing)
  59   return loss

  62 def create_masked_cross_entropy_loss(targets, present, logits):
  63   """Calculate the cross entropy loss matrices for the masked tokens."""
  64   cross_entropy_losses = losses.cross_entropy_loss_matrix(targets, logits)
  65 
  66   # Zeros matrix.
  67   zeros_losses = tf.zeros(
  68       shape=[FLAGS.batch_size, FLAGS.sequence_length], dtype=tf.float32)
  69 
  70   missing_ce_loss = tf.where(present, zeros_losses, cross_entropy_losses)
  71 
  72   return missing_ce_loss

  75 def calculate_reinforce_objective(hparams,
  76                                   log_probs,
  77                                   dis_predictions,
  78                                   present,
  79                                   estimated_values=None):
  80   """Calculate the REINFORCE objectives.  The REINFORCE objective should
  81   only be on the tokens that were missing.  Specifically, the final Generator
  82   reward should be based on the Discriminator predictions on missing tokens.
  83   The log probaibilities should be only for missing tokens and the baseline
  84   should be calculated only on the missing tokens.
  85 
  86   For this model, we optimize the reward is the log of the *conditional*
  87   probability the Discriminator assigns to the distribution.  Specifically, for
  88   a Discriminator D which outputs probability of real, given the past context,
  89 
  90     r_t = log D(x_t|x_0,x_1,...x_{t-1})
  91 
  92   And the policy for Generator G is the log-probability of taking action x2
  93   given the past context.
  94 
  95 
  96   Args:
  97     hparams:  MaskGAN hyperparameters.
  98     log_probs:  tf.float32 Tensor of log probailities of the tokens selected by
  99       the Generator.  Shape [batch_size, sequence_length].
 100     dis_predictions:  tf.float32 Tensor of the predictions from the
 101       Discriminator.  Shape [batch_size, sequence_length].
 102     present:  tf.bool Tensor indicating which tokens are present.  Shape
 103       [batch_size, sequence_length].
 104     estimated_values:  tf.float32 Tensor of estimated state values of tokens.
 105       Shape [batch_size, sequence_length]
 106 
 107   Returns:
 108     final_gen_objective:  Final REINFORCE objective for the sequence.
 109     rewards:  tf.float32 Tensor of rewards for sequence of shape [batch_size,
 110       sequence_length]
 111     advantages: tf.float32 Tensor of advantages for sequence of shape
 112       [batch_size, sequence_length]
 113     baselines:  tf.float32 Tensor of baselines for sequence of shape
 114       [batch_size, sequence_length]
 115     maintain_averages_op:  ExponentialMovingAverage apply average op to
 116       maintain the baseline.
 117   """
 118   # Final Generator objective.
 119   final_gen_objective = 0.
 120   gamma = hparams.rl_discount_rate
 121   eps = 1e-7
 122 
 123   # Generator rewards are log-probabilities.
 124   eps = tf.constant(1e-7, tf.float32)
 125   dis_predictions = tf.nn.sigmoid(dis_predictions)
 126   rewards = tf.log(dis_predictions + eps)
 127 
 128   # Apply only for missing elements.
 129   zeros = tf.zeros_like(present, dtype=tf.float32)
 130   log_probs = tf.where(present, zeros, log_probs)
 131   rewards = tf.where(present, zeros, rewards)
 132 
 133   # Unstack Tensors into lists.
 134   rewards_list = tf.unstack(rewards, axis=1)
 135   log_probs_list = tf.unstack(log_probs, axis=1)
 136   missing = 1. - tf.cast(present, tf.float32)
 137   missing_list = tf.unstack(missing, axis=1)
 138 
 139   # Cumulative Discounted Returns.  The true value function V*(s).
 140   cumulative_rewards = []
 141   for t in xrange(FLAGS.sequence_length):
 142     cum_value = tf.zeros(shape=[FLAGS.batch_size])
 143     for s in xrange(t, FLAGS.sequence_length):
 144       cum_value += missing_list[s] * np.power(gamma, (s - t)) * rewards_list[s]
 145     cumulative_rewards.append(cum_value)
 146   cumulative_rewards = tf.stack(cumulative_rewards, axis=1)
 147 
 148   ## REINFORCE with different baselines.
 149   # We create a separate critic functionality for the Discriminator.  This
 150   # will need to operate unidirectionally and it may take in the past context.
 151   if FLAGS.baseline_method == 'critic':
 152 
 153     # Critic loss calculated from the estimated value function \hat{V}(s)
 154     # versus the true value function V*(s).
 155     critic_loss = create_critic_loss(cumulative_rewards, estimated_values,
 156                                      present)
 157 
 158     # Baselines are coming from the critic's estimated state values.
 159     baselines = tf.unstack(estimated_values, axis=1)
 160 
 161     ## Calculate the Advantages, A(s,a) = Q(s,a) - \hat{V}(s).
 162     advantages = []
 163     for t in xrange(FLAGS.sequence_length):
 164       log_probability = log_probs_list[t]
 165       cum_advantage = tf.zeros(shape=[FLAGS.batch_size])
 166 
 167       for s in xrange(t, FLAGS.sequence_length):
 168         cum_advantage += missing_list[s] * np.power(gamma,
 169                                                     (s - t)) * rewards_list[s]
 170       cum_advantage -= baselines[t]
 171       # Clip advantages.
 172       cum_advantage = tf.clip_by_value(cum_advantage, -FLAGS.advantage_clipping,
 173                                        FLAGS.advantage_clipping)
 174       advantages.append(missing_list[t] * cum_advantage)
 175       final_gen_objective += tf.multiply(
 176           log_probability, missing_list[t] * tf.stop_gradient(cum_advantage))
 177 
 178     maintain_averages_op = None
 179     baselines = tf.stack(baselines, axis=1)
 180     advantages = tf.stack(advantages, axis=1)
 181 
 182   # Split the batch into half.  Use half for MC estimates for REINFORCE.
 183   # Use the other half to establish a baseline.
 184   elif FLAGS.baseline_method == 'dis_batch':
 185     # TODO(liamfedus):  Recheck.
 186     [rewards_half, baseline_half] = tf.split(
 187         rewards, num_or_size_splits=2, axis=0)
 188     [log_probs_half, _] = tf.split(log_probs, num_or_size_splits=2, axis=0)
 189     [reward_present_half, baseline_present_half] = tf.split(
 190         present, num_or_size_splits=2, axis=0)
 191 
 192     # Unstack to lists.
 193     baseline_list = tf.unstack(baseline_half, axis=1)
 194     baseline_missing = 1. - tf.cast(baseline_present_half, tf.float32)
 195     baseline_missing_list = tf.unstack(baseline_missing, axis=1)
 196 
 197     baselines = []
 198     for t in xrange(FLAGS.sequence_length):
 199       # Calculate baseline only for missing tokens.
 200       num_missing = tf.reduce_sum(baseline_missing_list[t])
 201 
 202       avg_baseline = tf.reduce_sum(
 203           baseline_missing_list[t] * baseline_list[t], keep_dims=True) / (
 204               num_missing + eps)
 205       baseline = tf.tile(avg_baseline, multiples=[FLAGS.batch_size / 2])
 206       baselines.append(baseline)
 207 
 208     # Unstack to lists.
 209     rewards_list = tf.unstack(rewards_half, axis=1)
 210     log_probs_list = tf.unstack(log_probs_half, axis=1)
 211     reward_missing = 1. - tf.cast(reward_present_half, tf.float32)
 212     reward_missing_list = tf.unstack(reward_missing, axis=1)
 213 
 214     ## Calculate the Advantages, A(s,a) = Q(s,a) - \hat{V}(s).
 215     advantages = []
 216     for t in xrange(FLAGS.sequence_length):
 217       log_probability = log_probs_list[t]
 218       cum_advantage = tf.zeros(shape=[FLAGS.batch_size / 2])
 219 
 220       for s in xrange(t, FLAGS.sequence_length):
 221         cum_advantage += reward_missing_list[s] * np.power(gamma, (s - t)) * (
 222             rewards_list[s] - baselines[s])
 223       # Clip advantages.
 224       cum_advantage = tf.clip_by_value(cum_advantage, -FLAGS.advantage_clipping,
 225                                        FLAGS.advantage_clipping)
 226       advantages.append(reward_missing_list[t] * cum_advantage)
 227       final_gen_objective += tf.multiply(
 228           log_probability,
 229           reward_missing_list[t] * tf.stop_gradient(cum_advantage))
 230 
 231     # Cumulative Discounted Returns.  The true value function V*(s).
 232     cumulative_rewards = []
 233     for t in xrange(FLAGS.sequence_length):
 234       cum_value = tf.zeros(shape=[FLAGS.batch_size / 2])
 235       for s in xrange(t, FLAGS.sequence_length):
 236         cum_value += reward_missing_list[s] * np.power(gamma, (
 237             s - t)) * rewards_list[s]
 238       cumulative_rewards.append(cum_value)
 239     cumulative_rewards = tf.stack(cumulative_rewards, axis=1)
 240 
 241     rewards = rewards_half
 242     critic_loss = None
 243     maintain_averages_op = None
 244     baselines = tf.stack(baselines, axis=1)
 245     advantages = tf.stack(advantages, axis=1)
 246 
 247   # Exponential Moving Average baseline.
 248   elif FLAGS.baseline_method == 'ema':
 249     # TODO(liamfedus): Recheck.
 250     # Lists of rewards and Log probabilities of the actions taken only for
 251     # missing tokens.
 252     ema = tf.train.ExponentialMovingAverage(decay=hparams.baseline_decay)
 253     maintain_averages_op = ema.apply(rewards_list)
 254 
 255     baselines = []
 256     for r in rewards_list:
 257       baselines.append(ema.average(r))
 258 
 259     ## Calculate the Advantages, A(s,a) = Q(s,a) - \hat{V}(s).
 260     advantages = []
 261     for t in xrange(FLAGS.sequence_length):
 262       log_probability = log_probs_list[t]
 263 
 264       # Calculate the forward advantage only on the missing tokens.
 265       cum_advantage = tf.zeros(shape=[FLAGS.batch_size])
 266       for s in xrange(t, FLAGS.sequence_length):
 267         cum_advantage += missing_list[s] * np.power(gamma, (s - t)) * (
 268             rewards_list[s] - baselines[s])
 269       # Clip advantages.
 270       cum_advantage = tf.clip_by_value(cum_advantage, -FLAGS.advantage_clipping,
 271                                        FLAGS.advantage_clipping)
 272       advantages.append(missing_list[t] * cum_advantage)
 273       final_gen_objective += tf.multiply(
 274           log_probability, missing_list[t] * tf.stop_gradient(cum_advantage))
 275 
 276     critic_loss = None
 277     baselines = tf.stack(baselines, axis=1)
 278     advantages = tf.stack(advantages, axis=1)
 279 
 280   elif FLAGS.baseline_method is None:
 281     num_missing = tf.reduce_sum(missing)
 282     final_gen_objective += tf.reduce_sum(rewards) / (num_missing + eps)
 283     baselines = tf.zeros_like(rewards)
 284     critic_loss = None
 285     maintain_averages_op = None
 286     advantages = cumulative_rewards
 287 
 288   else:
 289     raise NotImplementedError
 290 
 291   return [
 292       final_gen_objective, log_probs, rewards, advantages, baselines,
 293       maintain_averages_op, critic_loss, cumulative_rewards
 294   ]

 297 def calculate_log_perplexity(logits, targets, present):
 298   """Calculate the average log perplexity per *missing* token.
 299 
 300   Args:
 301     logits:  tf.float32 Tensor of the logits of shape [batch_size,
 302       sequence_length, vocab_size].
 303     targets:  tf.int32 Tensor of the sequence target of shape [batch_size,
 304       sequence_length].
 305     present:  tf.bool Tensor indicating the presence or absence of the token
 306       of shape [batch_size, sequence_length].
 307 
 308   Returns:
 309     avg_log_perplexity:  Scalar indicating the average log perplexity per
 310       missing token in the batch.
 311   """
 312   # logits = tf.Print(logits, [logits], message='logits:', summarize=50)
 313   # targets = tf.Print(targets, [targets], message='targets:', summarize=50)
 314   eps = 1e-12
 315   logits = tf.reshape(logits, [-1, FLAGS.vocab_size])
 316 
 317   # Only calculate log-perplexity on missing tokens.
 318   weights = tf.cast(present, tf.float32)
 319   weights = 1. - weights
 320   weights = tf.reshape(weights, [-1])
 321   num_missing = tf.reduce_sum(weights)
 322 
 323   log_perplexity = tf.contrib.legacy_seq2seq.sequence_loss_by_example(
 324       [logits], [tf.reshape(targets, [-1])], [weights])
 325 
 326   avg_log_perplexity = tf.reduce_sum(log_perplexity) / (num_missing + eps)
 327   return avg_log_perplexity
