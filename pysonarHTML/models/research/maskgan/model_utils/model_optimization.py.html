  29 def create_dis_pretrain_op(hparams, dis_loss, global_step):
  30   """Create a train op for pretraining."""
  31   with tf.name_scope('pretrain_generator'):
  32     optimizer = tf.train.AdamOptimizer(hparams.dis_pretrain_learning_rate)
  33     dis_vars = [
  34         v for v in tf.trainable_variables() if v.op.name.startswith('dis')
  35     ]
  36     if FLAGS.dis_update_share_embedding and FLAGS.dis_share_embedding:
  37       shared_embedding = [
  38           v for v in tf.trainable_variables()
  39           if v.op.name == 'gen/decoder/rnn/embedding'
  40       ][0]
  41       dis_vars.append(shared_embedding)
  42     dis_grads = tf.gradients(dis_loss, dis_vars)
  43     dis_grads_clipped, _ = tf.clip_by_global_norm(dis_grads,
  44                                                   FLAGS.grad_clipping)
  45     dis_pretrain_op = optimizer.apply_gradients(
  46         zip(dis_grads_clipped, dis_vars), global_step=global_step)
  47     return dis_pretrain_op

  65 def create_gen_train_op(hparams, learning_rate, gen_loss, global_step, mode):
  66   """Create Generator train op."""
  67   del hparams
  68   with tf.name_scope('train_generator'):
  69     if FLAGS.generator_optimizer == 'sgd':
  70       gen_optimizer = tf.train.GradientDescentOptimizer(learning_rate)
  71     elif FLAGS.generator_optimizer == 'adam':
  72       gen_optimizer = tf.train.AdamOptimizer(learning_rate)
  73     else:
  74       raise NotImplementedError
  75     gen_vars = [
  76         v for v in tf.trainable_variables() if v.op.name.startswith('gen')
  77     ]
  78     print('Optimizing Generator vars.')
  79     for v in gen_vars:
  80       print(v)
  81     if mode == 'MINIMIZE':
  82       gen_grads = tf.gradients(gen_loss, gen_vars)
  83     elif mode == 'MAXIMIZE':
  84       gen_grads = tf.gradients(-gen_loss, gen_vars)
  85     else:
  86       raise ValueError("Must be one of 'MINIMIZE' or 'MAXIMIZE'")
  87     gen_grads_clipped, _ = tf.clip_by_global_norm(gen_grads,
  88                                                   FLAGS.grad_clipping)
  89     gen_train_op = gen_optimizer.apply_gradients(
  90         zip(gen_grads_clipped, gen_vars), global_step=global_step)
  91     return gen_train_op, gen_grads_clipped, gen_vars

  94 def create_reinforce_gen_train_op(hparams, learning_rate, final_gen_reward,
  95                                   averages_op, global_step):
  96   """Create the Generator train_op when using REINFORCE.
  97 
  98   Args:
  99     hparams:  MaskGAN hyperparameters.
 100     learning_rate:  tf.Variable scalar learning rate.
 101     final_gen_objective:  Scalar final REINFORCE objective for the sequence.
 102     averages_op:  ExponentialMovingAverage apply average op to
 103       maintain the baseline.
 104     global_step:  global_step tf.Variable.
 105 
 106   Returns:
 107     gen_train_op: Generator training op.
 108   """
 109   del hparams
 110   with tf.name_scope('train_generator'):
 111     if FLAGS.generator_optimizer == 'sgd':
 112       gen_optimizer = tf.train.GradientDescentOptimizer(learning_rate)
 113     elif FLAGS.generator_optimizer == 'adam':
 114       gen_optimizer = tf.train.AdamOptimizer(learning_rate)
 115     else:
 116       raise NotImplementedError
 117     gen_vars = [
 118         v for v in tf.trainable_variables() if v.op.name.startswith('gen')
 119     ]
 120     print('\nOptimizing Generator vars:')
 121     for v in gen_vars:
 122       print(v)
 123 
 124     # Maximize reward.
 125     gen_grads = tf.gradients(-final_gen_reward, gen_vars)
 126     gen_grads_clipped, _ = tf.clip_by_global_norm(gen_grads,
 127                                                   FLAGS.grad_clipping)
 128     maximize_op = gen_optimizer.apply_gradients(
 129         zip(gen_grads_clipped, gen_vars), global_step=global_step)
 130 
 131     # Group maintain averages op.
 132     if averages_op:
 133       gen_train_op = tf.group(maximize_op, averages_op)
 134     else:
 135       gen_train_op = maximize_op
 136 
 137     return [gen_train_op, gen_grads, gen_vars]

 164 def create_critic_train_op(hparams, critic_loss, global_step):
 165   """Create Discriminator train op."""
 166   with tf.name_scope('train_critic'):
 167     critic_optimizer = tf.train.AdamOptimizer(hparams.critic_learning_rate)
 168     output_vars = [
 169         v for v in tf.trainable_variables() if v.op.name.startswith('critic')
 170     ]
 171 
 172     if FLAGS.critic_update_dis_vars:
 173       if FLAGS.discriminator_model == 'bidirectional_vd':
 174         critic_vars = [
 175             v for v in tf.trainable_variables()
 176             if v.op.name.startswith('dis/rnn')
 177         ]
 178       elif FLAGS.discriminator_model == 'seq2seq_vd':
 179         critic_vars = [
 180             v for v in tf.trainable_variables()
 181             if v.op.name.startswith('dis/decoder/rnn/multi_rnn_cell')
 182         ]
 183       critic_vars.extend(output_vars)
 184     else:
 185       critic_vars = output_vars
 186     print('\nOptimizing Critic vars:')
 187     for v in critic_vars:
 188       print(v)
 189     critic_grads = tf.gradients(critic_loss, critic_vars)
 190     critic_grads_clipped, _ = tf.clip_by_global_norm(critic_grads,
 191                                                      FLAGS.grad_clipping)
 192     critic_train_op = critic_optimizer.apply_gradients(
 193         zip(critic_grads_clipped, critic_vars), global_step=global_step)
 194     return critic_train_op, critic_grads_clipped, critic_vars
