  25 def discriminator_loss(predictions, labels, missing_tokens):
  26   """Discriminator loss based on predictions and labels.
  27 
  28   Args:
  29     predictions:  Discriminator linear predictions Tensor of shape [batch_size,
  30       sequence_length]
  31     labels: Labels for predictions, Tensor of shape [batch_size,
  32       sequence_length]
  33     missing_tokens:  Indicator for the missing tokens.  Evaluate the loss only
  34       on the tokens that were missing.
  35 
  36   Returns:
  37     loss:  Scalar tf.float32 loss.
  38 
  39   """
  40   loss = tf.losses.sigmoid_cross_entropy(labels,
  41                                          predictions,
  42                                          weights=missing_tokens)
  43   loss = tf.Print(
  44       loss, [loss, labels, missing_tokens],
  45       message='loss, labels, missing_tokens',
  46       summarize=25,
  47       first_n=25)
  48   return loss

  51 def cross_entropy_loss_matrix(gen_labels, gen_logits):
  52   """Computes the cross entropy loss for G.
  53 
  54   Args:
  55     gen_labels:  Labels for the correct token.
  56     gen_logits: Generator logits.
  57 
  58   Returns:
  59     loss_matrix:  Loss matrix of shape [batch_size, sequence_length].
  60   """
  61   cross_entropy_loss = tf.nn.sparse_softmax_cross_entropy_with_logits(
  62       labels=gen_labels, logits=gen_logits)
  63   return cross_entropy_loss

  66 def GAN_loss_matrix(dis_predictions):
  67   """Computes the cross entropy loss for G.
  68 
  69   Args:
  70     dis_predictions:  Discriminator predictions.
  71 
  72   Returns:
  73     loss_matrix: Loss matrix of shape [batch_size, sequence_length].
  74   """
  75   eps = tf.constant(1e-7, tf.float32)
  76   gan_loss_matrix = -tf.log(dis_predictions + eps)
  77   return gan_loss_matrix

  80 def generator_GAN_loss(predictions):
  81   """Generator GAN loss based on Discriminator predictions."""
  82   return -tf.log(tf.reduce_mean(predictions))

  85 def generator_blended_forward_loss(gen_logits, gen_labels, dis_predictions,
  86                                    is_real_input):
  87   """Computes the masked-loss for G.  This will be a blend of cross-entropy
  88   loss where the true label is known and GAN loss where the true label has been
  89   masked.
  90 
  91   Args:
  92     gen_logits: Generator logits.
  93     gen_labels:  Labels for the correct token.
  94     dis_predictions:  Discriminator predictions.
  95     is_real_input:  Tensor indicating whether the label is present.
  96 
  97   Returns:
  98     loss: Scalar tf.float32 total loss.
  99   """
 100   cross_entropy_loss = tf.nn.sparse_softmax_cross_entropy_with_logits(
 101       labels=gen_labels, logits=gen_logits)
 102   gan_loss = -tf.log(dis_predictions)
 103   loss_matrix = tf.where(is_real_input, cross_entropy_loss, gan_loss)
 104   return tf.reduce_mean(loss_matrix)

 107 def wasserstein_generator_loss(gen_logits, gen_labels, dis_values,
 108                                is_real_input):
 109   """Computes the masked-loss for G.  This will be a blend of cross-entropy
 110   loss where the true label is known and GAN loss where the true label is
 111   missing.
 112 
 113   Args:
 114     gen_logits:  Generator logits.
 115     gen_labels:  Labels for the correct token.
 116     dis_values:  Discriminator values Tensor of shape [batch_size,
 117       sequence_length].
 118     is_real_input:  Tensor indicating whether the label is present.
 119 
 120   Returns:
 121     loss: Scalar tf.float32 total loss.
 122   """
 123   cross_entropy_loss = tf.nn.sparse_softmax_cross_entropy_with_logits(
 124       labels=gen_labels, logits=gen_logits)
 125   # Maximize the dis_values (minimize the negative)
 126   gan_loss = -dis_values
 127   loss_matrix = tf.where(is_real_input, cross_entropy_loss, gan_loss)
 128   loss = tf.reduce_mean(loss_matrix)
 129   return loss

 132 def wasserstein_discriminator_loss(real_values, fake_values):
 133   """Wasserstein discriminator loss.
 134 
 135   Args:
 136     real_values: Value given by the Wasserstein Discriminator to real data.
 137     fake_values: Value given by the Wasserstein Discriminator to fake data.
 138 
 139   Returns:
 140     loss:  Scalar tf.float32 loss.
 141 
 142   """
 143   real_avg = tf.reduce_mean(real_values)
 144   fake_avg = tf.reduce_mean(fake_values)
 145 
 146   wasserstein_loss = real_avg - fake_avg
 147   return wasserstein_loss

 150 def wasserstein_discriminator_loss_intrabatch(values, is_real_input):
 151   """Wasserstein discriminator loss.  This is an odd variant where the value
 152   difference is between the real tokens and the fake tokens within a single
 153   batch.
 154 
 155   Args:
 156     values: Value given by the Wasserstein Discriminator of shape [batch_size,
 157       sequence_length] to an imputed batch (real and fake).
 158     is_real_input: tf.bool Tensor of shape [batch_size, sequence_length]. If
 159       true, it indicates that the label is known.
 160 
 161   Returns:
 162     wasserstein_loss:  Scalar tf.float32 loss.
 163 
 164   """
 165   zero_tensor = tf.constant(0., dtype=tf.float32, shape=[])
 166 
 167   present = tf.cast(is_real_input, tf.float32)
 168   missing = tf.cast(1 - present, tf.float32)
 169 
 170   # Counts for real and fake tokens.
 171   real_count = tf.reduce_sum(present)
 172   fake_count = tf.reduce_sum(missing)
 173 
 174   # Averages for real and fake token values.
 175   real = tf.mul(values, present)
 176   fake = tf.mul(values, missing)
 177   real_avg = tf.reduce_sum(real) / real_count
 178   fake_avg = tf.reduce_sum(fake) / fake_count
 179 
 180   # If there are no real or fake entries in the batch, we assign an average
 181   # value of zero.
 182   real_avg = tf.where(tf.equal(real_count, 0), zero_tensor, real_avg)
 183   fake_avg = tf.where(tf.equal(fake_count, 0), zero_tensor, fake_avg)
 184 
 185   wasserstein_loss = real_avg - fake_avg
 186   return wasserstein_loss
