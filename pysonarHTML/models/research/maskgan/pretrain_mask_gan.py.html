  45 def pretrain_generator(sv, sess, model, data, log, id_to_word,
  46                        data_ngram_counts, is_chief):
  47   """Pretrain the generator with classic language modeling training."""
  48   print('\nPretraining generator for %d steps.' % FLAGS.gen_pretrain_steps)
  49   log.write(
  50       '\nPretraining generator for %d steps.\n' % FLAGS.gen_pretrain_steps)
  51 
  52   is_pretraining = True
  53 
  54   while is_pretraining:
  55 
  56     costs = 0.
  57     iters = 0
  58     if FLAGS.data_set == 'ptb':
  59       iterator = ptb_loader.ptb_iterator(data, FLAGS.batch_size,
  60                                          FLAGS.sequence_length,
  61                                          FLAGS.epoch_size_override)
  62     elif FLAGS.data_set == 'imdb':
  63       iterator = imdb_loader.imdb_iterator(data, FLAGS.batch_size,
  64                                            FLAGS.sequence_length)
  65 
  66     for x, y, _ in iterator:
  67 
  68       # For pretraining with cross entropy loss, we have all tokens in the
  69       # forward sequence present (all True).
  70       model_utils.assign_percent_real(sess, model.percent_real_update,
  71                                       model.new_rate, 1.0)
  72       p = np.ones(shape=[FLAGS.batch_size, FLAGS.sequence_length], dtype=bool)
  73 
  74       pretrain_feed = {model.inputs: x, model.targets: y, model.present: p}
  75 
  76       [losses, cost_eval, _, step] = sess.run(
  77           [
  78               model.fake_cross_entropy_losses, model.avg_log_perplexity,
  79               model.gen_pretrain_op, model.global_step
  80           ],
  81           feed_dict=pretrain_feed)
  82 
  83       costs += cost_eval
  84       iters += FLAGS.sequence_length
  85 
  86       # Calulate rolling perplexity.
  87       perplexity = np.exp(costs / iters)
  88 
  89       # Summaries.
  90       if is_chief and step % FLAGS.summaries_every == 0:
  91         # Graph summaries.
  92         summary_str = sess.run(
  93             model.merge_summaries_op, feed_dict=pretrain_feed)
  94         sv.SummaryComputed(sess, summary_str)
  95 
  96         # Additional summary.
  97         for n, data_ngram_count in data_ngram_counts.iteritems():
  98           avg_percent_captured = evaluation_utils.sequence_ngram_evaluation(
  99               sess, model.fake_sequence, log, pretrain_feed, data_ngram_count,
 100               int(n))
 101           summary_percent_str = tf.Summary(value=[
 102               tf.Summary.Value(
 103                   tag='general/%s-grams_percent_correct' % n,
 104                   simple_value=avg_percent_captured)
 105           ])
 106           sv.SummaryComputed(sess, summary_percent_str, global_step=step)
 107 
 108         summary_perplexity_str = tf.Summary(value=[
 109             tf.Summary.Value(tag='general/perplexity', simple_value=perplexity)
 110         ])
 111         sv.SummaryComputed(sess, summary_perplexity_str, global_step=step)
 112 
 113       # Printing and logging
 114       if is_chief and step % FLAGS.print_every == 0:
 115         print('global_step: %d' % step)
 116         print(' generator loss: %.3f' % np.mean(losses))
 117         print(' perplexity: %.3f' % perplexity)
 118         log.write('global_step: %d\n' % step)
 119         log.write(' generator loss: %.3f\n' % np.mean(losses))
 120         log.write(' perplexity: %.3f\n' % perplexity)
 121 
 122         for n, data_ngram_count in data_ngram_counts.iteritems():
 123           avg_percent_captured = evaluation_utils.sequence_ngram_evaluation(
 124               sess, model.fake_sequence, log, pretrain_feed, data_ngram_count,
 125               int(n))
 126           print(' percent of %s-grams captured: %.3f.\n' %
 127                 (n, avg_percent_captured))
 128           log.write(' percent of %s-grams captured: %.3f.\n\n' %
 129                     (n, avg_percent_captured))
 130 
 131         evaluation_utils.generate_logs(sess, model, log, id_to_word,
 132                                        pretrain_feed)
 133 
 134       if step >= FLAGS.gen_pretrain_steps:
 135         is_pretraining = False
 136         break
 137   return

 140 def pretrain_discriminator(sv, sess, model, data, log, id_to_word,
 141                            data_ngram_counts, is_chief):
 142   print('\nPretraining discriminator for %d steps.' % FLAGS.dis_pretrain_steps)
 143   log.write(
 144       '\nPretraining discriminator for %d steps.\n' % FLAGS.dis_pretrain_steps)
 145 
 146   is_pretraining = True
 147 
 148   while is_pretraining:
 149 
 150     cumulative_costs = 0.
 151     iters = 0
 152     if FLAGS.data_set == 'ptb':
 153       iterator = ptb_loader.ptb_iterator(data, FLAGS.batch_size,
 154                                          FLAGS.sequence_length,
 155                                          FLAGS.epoch_size_override)
 156     elif FLAGS.data_set == 'imdb':
 157       iterator = imdb_loader.imdb_iterator(data, FLAGS.batch_size,
 158                                            FLAGS.sequence_length)
 159 
 160     for x, y, _ in iterator:
 161       is_present_rate = FLAGS.is_present_rate
 162       # is_present_rate = np.random.uniform(low=0.0, high=1.0)
 163       model_utils.assign_percent_real(sess, model.percent_real_update,
 164                                       model.new_rate, is_present_rate)
 165       # Randomly mask out tokens.
 166       p = model_utils.generate_mask()
 167 
 168       pretrain_feed = {model.inputs: x, model.targets: y, model.present: p}
 169 
 170       [_, dis_loss_eval, gen_log_perplexity_eval, step] = sess.run(
 171           [
 172               model.dis_pretrain_op, model.dis_loss, model.avg_log_perplexity,
 173               model.global_step
 174           ],
 175           feed_dict=pretrain_feed)
 176 
 177       cumulative_costs += gen_log_perplexity_eval
 178       iters += 1
 179 
 180       # Calulate rolling perplexity.
 181       perplexity = np.exp(cumulative_costs / iters)
 182 
 183       # Summaries.
 184       if is_chief and step % FLAGS.summaries_every == 0:
 185         # Graph summaries.
 186         summary_str = sess.run(
 187             model.merge_summaries_op, feed_dict=pretrain_feed)
 188         sv.SummaryComputed(sess, summary_str)
 189 
 190         # Additional summary.
 191         for n, data_ngram_count in data_ngram_counts.iteritems():
 192           avg_percent_captured = evaluation_utils.sequence_ngram_evaluation(
 193               sess, model.fake_sequence, log, pretrain_feed, data_ngram_count,
 194               int(n))
 195           summary_percent_str = tf.Summary(value=[
 196               tf.Summary.Value(
 197                   tag='general/%s-grams_percent_correct' % n,
 198                   simple_value=avg_percent_captured)
 199           ])
 200           sv.SummaryComputed(sess, summary_percent_str, global_step=step)
 201 
 202         summary_perplexity_str = tf.Summary(value=[
 203             tf.Summary.Value(tag='general/perplexity', simple_value=perplexity)
 204         ])
 205         sv.SummaryComputed(sess, summary_perplexity_str, global_step=step)
 206 
 207       # Printing and logging
 208       if is_chief and step % FLAGS.print_every == 0:
 209         print('global_step: %d' % step)
 210         print(' discriminator loss: %.3f' % dis_loss_eval)
 211         print(' perplexity: %.3f' % perplexity)
 212         log.write('global_step: %d\n' % step)
 213         log.write(' discriminator loss: %.3f\n' % dis_loss_eval)
 214         log.write(' perplexity: %.3f\n' % perplexity)
 215 
 216         for n, data_ngram_count in data_ngram_counts.iteritems():
 217           avg_percent_captured = evaluation_utils.sequence_ngram_evaluation(
 218               sess, model.fake_sequence, log, pretrain_feed, data_ngram_count,
 219               int(n))
 220           print(' percent of %s-grams captured: %.3f.\n' %
 221                 (n, avg_percent_captured))
 222           log.write(' percent of %s-grams captured: %.3f.\n\n' %
 223                     (n, avg_percent_captured))
 224 
 225         evaluation_utils.generate_logs(sess, model, log, id_to_word,
 226                                        pretrain_feed)
 227 
 228       if step >= FLAGS.dis_pretrain_steps + int(FLAGS.gen_pretrain_steps or 0):
 229         is_pretraining = False
 230         break
 231   return
