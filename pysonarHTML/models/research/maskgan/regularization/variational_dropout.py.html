  16 """Variational Dropout Wrapper."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 
  24 
  25 class VariationalDropoutWrapper(tf.contrib.rnn.RNNCell):
  26   """Add variational dropout to a RNN cell."""
  27 
  28   def __init__(self, cell, batch_size, input_size, recurrent_keep_prob,
  29                input_keep_prob):
  30     self._cell = cell
  31     self._recurrent_keep_prob = recurrent_keep_prob
  32     self._input_keep_prob = input_keep_prob
  33 
  34     def make_mask(keep_prob, units):
  35       random_tensor = keep_prob
  36       # 0. if [keep_prob, 1.0) and 1. if [1.0, 1.0 + keep_prob)
  37       random_tensor += tf.random_uniform(tf.stack([batch_size, units]))
  38       return tf.floor(random_tensor) / keep_prob
  39 
  40     self._recurrent_mask = make_mask(recurrent_keep_prob,
  41                                      self._cell.state_size[0])
  42     self._input_mask = self._recurrent_mask
  43 
  44   @property
  45   def state_size(self):
  46     return self._cell.state_size
  47 
  48   @property
  49   def output_size(self):
  50     return self._cell.output_size
  51 
  52   def __call__(self, inputs, state, scope=None):
  53     dropped_inputs = inputs * self._input_mask
  54     dropped_state = (state[0], state[1] * self._recurrent_mask)
  55     new_h, new_state = self._cell(dropped_inputs, dropped_state, scope)
  56     return new_h, new_state
