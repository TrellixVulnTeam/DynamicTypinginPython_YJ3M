  16 """Zoneout Wrapper"""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 
  24 
  25 class ZoneoutWrapper(tf.contrib.rnn.RNNCell):
  26   """Add Zoneout to a RNN cell."""
  27 
  28   def __init__(self, cell, zoneout_drop_prob, is_training=True):
  29     self._cell = cell
  30     self._zoneout_prob = zoneout_drop_prob
  31     self._is_training = is_training
  32 
  33   @property
  34   def state_size(self):
  35     return self._cell.state_size
  36 
  37   @property
  38   def output_size(self):
  39     return self._cell.output_size
  40 
  41   def __call__(self, inputs, state, scope=None):
  42     output, new_state = self._cell(inputs, state, scope)
  43     if not isinstance(self._cell.state_size, tuple):
  44       new_state = tf.split(value=new_state, num_or_size_splits=2, axis=1)
  45       state = tf.split(value=state, num_or_size_splits=2, axis=1)
  46     final_new_state = [new_state[0], new_state[1]]
  47     if self._is_training:
  48       for i, state_element in enumerate(state):
  49         random_tensor = 1 - self._zoneout_prob  # keep probability
  50         random_tensor += tf.random_uniform(tf.shape(state_element))
  51         # 0. if [zoneout_prob, 1.0) and 1. if [1.0, 1.0 + zoneout_prob)
  52         binary_tensor = tf.floor(random_tensor)
  53         final_new_state[
  54             i] = (new_state[i] - state_element) * binary_tensor + state_element
  55     else:
  56       for i, state_element in enumerate(state):
  57         final_new_state[
  58             i] = state_element * self._zoneout_prob + new_state[i] * (
  59                 1 - self._zoneout_prob)
  60     if isinstance(self._cell.state_size, tuple):
  61       return output, tf.contrib.rnn.LSTMStateTuple(
  62           final_new_state[0], final_new_state[1])
  63 
  64     return output, tf.concat([final_new_state[0], final_new_state[1]], 1)
