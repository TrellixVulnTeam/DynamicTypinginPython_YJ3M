  29 def transform_input_with_is_missing_token(inputs, targets_present):
  30   """Transforms the inputs to have missing tokens when it's masked out.  The
  31   mask is for the targets, so therefore, to determine if an input at time t is
  32   masked, we have to check if the target at time t - 1 is masked out.
  33 
  34   e.g.
  35     inputs = [a, b, c, d]
  36     targets = [b, c, d, e]
  37     targets_present = [1, 0, 1, 0]
  38 
  39   then,
  40     transformed_input = [a, b, <missing>, d]
  41 
  42   Args:
  43     inputs:  tf.int32 Tensor of shape [batch_size, sequence_length] with tokens
  44       up to, but not including, vocab_size.
  45     targets_present:  tf.bool Tensor of shape [batch_size, sequence_length] with
  46       True representing the presence of the word.
  47 
  48   Returns:
  49     transformed_input:  tf.int32 Tensor of shape [batch_size, sequence_length]
  50       which takes on value of inputs when the input is present and takes on
  51       value=vocab_size to indicate a missing token.
  52   """
  53   # To fill in if the input is missing.
  54   input_missing = tf.constant(FLAGS.vocab_size,
  55                               dtype=tf.int32,
  56                               shape=[FLAGS.batch_size, FLAGS.sequence_length])
  57 
  58   # The 0th input will always be present to MaskGAN.
  59   zeroth_input_present = tf.constant(True, tf.bool, shape=[FLAGS.batch_size, 1])
  60 
  61   # Input present mask.
  62   inputs_present = tf.concat(
  63       [zeroth_input_present, targets_present[:, :-1]], axis=1)
  64 
  65   transformed_input = tf.where(inputs_present, inputs, input_missing)
  66   return transformed_input

  69 def gen_encoder(hparams, inputs, targets_present, is_training, reuse=None):
  70   """Define the Encoder graph.
  71 
  72 
  73   Args:
  74     hparams:  Hyperparameters for the MaskGAN.
  75     inputs:  tf.int32 Tensor of shape [batch_size, sequence_length] with tokens
  76       up to, but not including, vocab_size.
  77     targets_present:  tf.bool Tensor of shape [batch_size, sequence_length] with
  78       True representing the presence of the target.
  79     is_training:  Boolean indicating operational mode (train/inference).
  80     reuse (Optional):   Whether to reuse the variables.
  81 
  82   Returns:
  83     Tuple of (hidden_states, final_state).
  84   """
  85   with tf.variable_scope('encoder', reuse=reuse):
  86 
  87     def lstm_cell():
  88       return tf.contrib.rnn.BasicLSTMCell(hparams.gen_rnn_size,
  89                                           forget_bias=0.0,
  90                                           state_is_tuple=True,
  91                                           reuse=reuse)
  92 
  93     attn_cell = lstm_cell
  94     if is_training and FLAGS.keep_prob < 1:
  95 
  96       def attn_cell():
  97         return tf.contrib.rnn.DropoutWrapper(
  98             lstm_cell(), output_keep_prob=FLAGS.keep_prob)
  99 
 100     cell = tf.contrib.rnn.MultiRNNCell(
 101         [attn_cell() for _ in range(hparams.gen_num_layers)],
 102         state_is_tuple=True)
 103 
 104     initial_state = cell.zero_state(FLAGS.batch_size, tf.float32)
 105 
 106     # Add a missing token for inputs not present.
 107     real_inputs = inputs
 108     masked_inputs = transform_input_with_is_missing_token(inputs,
 109                                                           targets_present)
 110 
 111     with tf.variable_scope('rnn'):
 112       hidden_states = []
 113 
 114       # Split the embedding into two parts so that we can load the PTB
 115       # weights into one part of the Variable.
 116       embedding = tf.get_variable('embedding',
 117                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 118       missing_embedding = tf.get_variable('missing_embedding',
 119                                           [1, hparams.gen_rnn_size])
 120       embedding = tf.concat([embedding, missing_embedding], axis=0)
 121 
 122       real_rnn_inputs = tf.nn.embedding_lookup(embedding, real_inputs)
 123       masked_rnn_inputs = tf.nn.embedding_lookup(embedding, masked_inputs)
 124 
 125       if is_training and FLAGS.keep_prob < 1:
 126         masked_rnn_inputs = tf.nn.dropout(masked_rnn_inputs, FLAGS.keep_prob)
 127 
 128       state = initial_state
 129       for t in xrange(FLAGS.sequence_length):
 130         if t > 0:
 131           tf.get_variable_scope().reuse_variables()
 132 
 133         rnn_inp = masked_rnn_inputs[:, t]
 134         rnn_out, state = cell(rnn_inp, state)
 135         hidden_states.append(rnn_out)
 136       final_masked_state = state
 137       hidden_states = tf.stack(hidden_states, axis=1)
 138 
 139       # Produce the RNN state had the model operated only
 140       # over real data.
 141       real_state = initial_state
 142       for t in xrange(FLAGS.sequence_length):
 143         tf.get_variable_scope().reuse_variables()
 144 
 145         # RNN.
 146         rnn_inp = real_rnn_inputs[:, t]
 147         rnn_out, real_state = cell(rnn_inp, real_state)
 148       final_state = real_state
 149 
 150   return (hidden_states, final_masked_state), initial_state, final_state

 153 def gen_decoder(hparams,
 154                 inputs,
 155                 targets,
 156                 targets_present,
 157                 encoding_state,
 158                 is_training,
 159                 is_validating,
 160                 reuse=None):
 161   """Define the Decoder graph. The Decoder will now impute tokens that
 162       have been masked from the input seqeunce.
 163   """
 164   gen_decoder_rnn_size = hparams.gen_rnn_size
 165 
 166   with tf.variable_scope('decoder', reuse=reuse):
 167 
 168     def lstm_cell():
 169       return tf.contrib.rnn.BasicLSTMCell(gen_decoder_rnn_size,
 170                                           forget_bias=0.0,
 171                                           state_is_tuple=True,
 172                                           reuse=reuse)
 173 
 174     attn_cell = lstm_cell
 175     if is_training and FLAGS.keep_prob < 1:
 176 
 177       def attn_cell():
 178         return tf.contrib.rnn.DropoutWrapper(
 179             lstm_cell(), output_keep_prob=FLAGS.keep_prob)
 180 
 181     cell_gen = tf.contrib.rnn.MultiRNNCell(
 182         [attn_cell() for _ in range(hparams.gen_num_layers)],
 183         state_is_tuple=True)
 184 
 185     # Hidden encoder states.
 186     hidden_vector_encodings = encoding_state[0]
 187 
 188     # Carry forward the final state tuple from the encoder.
 189     # State tuples.
 190     state_gen = encoding_state[1]
 191 
 192     if FLAGS.attention_option is not None:
 193       (attention_keys, attention_values, _,
 194        attention_construct_fn) = attention_utils.prepare_attention(
 195            hidden_vector_encodings,
 196            FLAGS.attention_option,
 197            num_units=gen_decoder_rnn_size,
 198            reuse=reuse)
 199 
 200     with tf.variable_scope('rnn'):
 201       sequence, logits, log_probs = [], [], []
 202 
 203       embedding = tf.get_variable('embedding',
 204                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 205       softmax_w = tf.matrix_transpose(embedding)
 206       softmax_b = tf.get_variable('softmax_b', [FLAGS.vocab_size])
 207 
 208       rnn_inputs = tf.nn.embedding_lookup(embedding, inputs)
 209 
 210       if is_training and FLAGS.keep_prob < 1:
 211         rnn_inputs = tf.nn.dropout(rnn_inputs, FLAGS.keep_prob)
 212 
 213       rnn_outs = []
 214 
 215       fake = None
 216       for t in xrange(FLAGS.sequence_length):
 217         if t > 0:
 218           tf.get_variable_scope().reuse_variables()
 219 
 220         # Input to the Decoder.
 221         if t == 0:
 222           # Always provide the real input at t = 0.
 223           rnn_inp = rnn_inputs[:, t]
 224 
 225         # If the input is present, read in the input at t.
 226         # If the input is not present, read in the previously generated.
 227         else:
 228           real_rnn_inp = rnn_inputs[:, t]
 229 
 230           # While validating, the decoder should be operating in teacher
 231           # forcing regime.  Also, if we're just training with cross_entropy
 232           # use teacher forcing.
 233           if is_validating or FLAGS.gen_training_strategy == 'cross_entropy':
 234             rnn_inp = real_rnn_inp
 235           else:
 236             fake_rnn_inp = tf.nn.embedding_lookup(embedding, fake)
 237             rnn_inp = tf.where(targets_present[:, t - 1], real_rnn_inp,
 238                                fake_rnn_inp)
 239 
 240         # RNN.
 241         rnn_out, state_gen = cell_gen(rnn_inp, state_gen)
 242 
 243         if FLAGS.attention_option is not None:
 244           rnn_out = attention_construct_fn(rnn_out, attention_keys,
 245                                            attention_values)
 246         rnn_outs.append(rnn_out)
 247         if FLAGS.gen_training_strategy != 'cross_entropy':
 248           logit = tf.nn.bias_add(tf.matmul(rnn_out, softmax_w), softmax_b)
 249 
 250           # Output for Decoder.
 251           # If input is present:   Return real at t+1.
 252           # If input is not present:  Return fake for t+1.
 253           real = targets[:, t]
 254 
 255           categorical = tf.contrib.distributions.Categorical(logits=logit)
 256           fake = categorical.sample()
 257           log_prob = categorical.log_prob(fake)
 258 
 259           output = tf.where(targets_present[:, t], real, fake)
 260 
 261         else:
 262           batch_size = tf.shape(rnn_out)[0]
 263           logit = tf.zeros(tf.stack([batch_size, FLAGS.vocab_size]))
 264           log_prob = tf.zeros(tf.stack([batch_size]))
 265           output = targets[:, t]
 266 
 267         # Add to lists.
 268         sequence.append(output)
 269         log_probs.append(log_prob)
 270         logits.append(logit)
 271       if FLAGS.gen_training_strategy == 'cross_entropy':
 272         logits = tf.nn.bias_add(
 273             tf.matmul(
 274                 tf.reshape(tf.stack(rnn_outs, 1), [-1, gen_decoder_rnn_size]),
 275                 softmax_w), softmax_b)
 276         logits = tf.reshape(logits,
 277                             [-1, FLAGS.sequence_length, FLAGS.vocab_size])
 278       else:
 279         logits = tf.stack(logits, axis=1)
 280 
 281   return (tf.stack(sequence, axis=1), logits, tf.stack(log_probs, axis=1))

 284 def generator(hparams,
 285               inputs,
 286               targets,
 287               targets_present,
 288               is_training,
 289               is_validating,
 290               reuse=None):
 291   """Define the Generator graph."""
 292   with tf.variable_scope('gen', reuse=reuse):
 293     encoder_states, initial_state, final_state = gen_encoder(
 294         hparams, inputs, targets_present, is_training=is_training, reuse=reuse)
 295     stacked_sequence, stacked_logits, stacked_log_probs = gen_decoder(
 296         hparams,
 297         inputs,
 298         targets,
 299         targets_present,
 300         encoder_states,
 301         is_training=is_training,
 302         is_validating=is_validating,
 303         reuse=reuse)
 304     return (stacked_sequence, stacked_logits, stacked_log_probs, initial_state,
 305             final_state)
