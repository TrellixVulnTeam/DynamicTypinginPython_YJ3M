  28 def generator(hparams,
  29               inputs,
  30               targets,
  31               targets_present,
  32               is_training,
  33               is_validating,
  34               reuse=None):
  35   """Define the Generator graph.
  36 
  37     G will now impute tokens that have been masked from the input seqeunce.
  38   """
  39   tf.logging.warning(
  40       'Undirectional generative model is not a useful model for this MaskGAN '
  41       'because future context is needed.  Use only for debugging purposes.')
  42   init_scale = 0.05
  43   initializer = tf.random_uniform_initializer(-init_scale, init_scale)
  44   with tf.variable_scope('gen', reuse=reuse, initializer=initializer):
  45 
  46     def lstm_cell():
  47       return tf.contrib.rnn.BasicLSTMCell(hparams.gen_rnn_size,
  48                                           forget_bias=0.0,
  49                                           state_is_tuple=True,
  50                                           reuse=reuse)
  51 
  52     attn_cell = lstm_cell
  53     if is_training and FLAGS.keep_prob < 1:
  54 
  55       def attn_cell():
  56         return tf.contrib.rnn.DropoutWrapper(
  57             lstm_cell(), output_keep_prob=FLAGS.keep_prob)
  58 
  59     cell_gen = tf.contrib.rnn.MultiRNNCell(
  60         [attn_cell() for _ in range(hparams.gen_num_layers)],
  61         state_is_tuple=True)
  62 
  63     initial_state = cell_gen.zero_state(FLAGS.batch_size, tf.float32)
  64 
  65     with tf.variable_scope('rnn'):
  66       sequence, logits, log_probs = [], [], []
  67       embedding = tf.get_variable('embedding',
  68                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
  69       softmax_w = tf.get_variable('softmax_w',
  70                                   [hparams.gen_rnn_size, FLAGS.vocab_size])
  71       softmax_b = tf.get_variable('softmax_b', [FLAGS.vocab_size])
  72 
  73       rnn_inputs = tf.nn.embedding_lookup(embedding, inputs)
  74 
  75       if is_training and FLAGS.keep_prob < 1:
  76         rnn_inputs = tf.nn.dropout(rnn_inputs, FLAGS.keep_prob)
  77 
  78       fake = None
  79       for t in xrange(FLAGS.sequence_length):
  80         if t > 0:
  81           tf.get_variable_scope().reuse_variables()
  82 
  83         # Input to the model is the first token to provide context.  The
  84         # model will then predict token t > 0.
  85         if t == 0:
  86           # Always provide the real input at t = 0.
  87           state_gen = initial_state
  88           rnn_inp = rnn_inputs[:, t]
  89 
  90         # If the input is present, read in the input at t.
  91         # If the input is not present, read in the previously generated.
  92         else:
  93           real_rnn_inp = rnn_inputs[:, t]
  94           fake_rnn_inp = tf.nn.embedding_lookup(embedding, fake)
  95 
  96           # While validating, the decoder should be operating in teacher
  97           # forcing regime.  Also, if we're just training with cross_entropy
  98           # use teacher forcing.
  99           if is_validating or (is_training and
 100                                FLAGS.gen_training_strategy == 'cross_entropy'):
 101             rnn_inp = real_rnn_inp
 102           else:
 103             rnn_inp = tf.where(targets_present[:, t - 1], real_rnn_inp,
 104                                fake_rnn_inp)
 105 
 106         # RNN.
 107         rnn_out, state_gen = cell_gen(rnn_inp, state_gen)
 108         logit = tf.matmul(rnn_out, softmax_w) + softmax_b
 109 
 110         # Real sample.
 111         real = targets[:, t]
 112 
 113         categorical = tf.contrib.distributions.Categorical(logits=logit)
 114         fake = categorical.sample()
 115         log_prob = categorical.log_prob(fake)
 116 
 117         # Output for Generator will either be generated or the input.
 118         #
 119         # If present:   Return real.
 120         # If not present:  Return fake.
 121         output = tf.where(targets_present[:, t], real, fake)
 122 
 123         # Add to lists.
 124         sequence.append(output)
 125         log_probs.append(log_prob)
 126         logits.append(logit)
 127 
 128       # Produce the RNN state had the model operated only
 129       # over real data.
 130       real_state_gen = initial_state
 131       for t in xrange(FLAGS.sequence_length):
 132         tf.get_variable_scope().reuse_variables()
 133 
 134         rnn_inp = rnn_inputs[:, t]
 135 
 136         # RNN.
 137         rnn_out, real_state_gen = cell_gen(rnn_inp, real_state_gen)
 138 
 139       final_state = real_state_gen
 140 
 141   return (tf.stack(sequence, axis=1), tf.stack(logits, axis=1), tf.stack(
 142       log_probs, axis=1), initial_state, final_state)

 145 def discriminator(hparams, sequence, is_training, reuse=None):
 146   """Define the Discriminator graph."""
 147   tf.logging.warning(
 148       'Undirectional Discriminative model is not a useful model for this '
 149       'MaskGAN because future context is needed.  Use only for debugging '
 150       'purposes.')
 151   sequence = tf.cast(sequence, tf.int32)
 152 
 153   with tf.variable_scope('dis', reuse=reuse):
 154 
 155     def lstm_cell():
 156       return tf.contrib.rnn.BasicLSTMCell(hparams.dis_rnn_size,
 157                                           forget_bias=0.0,
 158                                           state_is_tuple=True,
 159                                           reuse=reuse)
 160 
 161     attn_cell = lstm_cell
 162     if is_training and FLAGS.keep_prob < 1:
 163 
 164       def attn_cell():
 165         return tf.contrib.rnn.DropoutWrapper(
 166             lstm_cell(), output_keep_prob=FLAGS.keep_prob)
 167 
 168     cell_dis = tf.contrib.rnn.MultiRNNCell(
 169         [attn_cell() for _ in range(hparams.dis_num_layers)],
 170         state_is_tuple=True)
 171 
 172     state_dis = cell_dis.zero_state(FLAGS.batch_size, tf.float32)
 173 
 174     with tf.variable_scope('rnn') as vs:
 175       predictions = []
 176       embedding = tf.get_variable('embedding',
 177                                   [FLAGS.vocab_size, hparams.dis_rnn_size])
 178 
 179       rnn_inputs = tf.nn.embedding_lookup(embedding, sequence)
 180 
 181       if is_training and FLAGS.keep_prob < 1:
 182         rnn_inputs = tf.nn.dropout(rnn_inputs, FLAGS.keep_prob)
 183 
 184       for t in xrange(FLAGS.sequence_length):
 185         if t > 0:
 186           tf.get_variable_scope().reuse_variables()
 187 
 188         rnn_in = rnn_inputs[:, t]
 189         rnn_out, state_dis = cell_dis(rnn_in, state_dis)
 190 
 191         # Prediction is linear output for Discriminator.
 192         pred = tf.contrib.layers.linear(rnn_out, 1, scope=vs)
 193 
 194         predictions.append(pred)
 195   predictions = tf.stack(predictions, axis=1)
 196   return tf.squeeze(predictions, axis=2)
