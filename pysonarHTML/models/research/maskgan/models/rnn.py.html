  31 def generator(hparams,
  32               inputs,
  33               targets,
  34               targets_present,
  35               is_training,
  36               is_validating,
  37               reuse=None):
  38   """Define the Generator graph.
  39 
  40     G will now impute tokens that have been masked from the input seqeunce.
  41   """
  42   tf.logging.warning(
  43       'Undirectional generative model is not a useful model for this MaskGAN '
  44       'because future context is needed.  Use only for debugging purposes.')
  45   init_scale = 0.05
  46   initializer = tf.random_uniform_initializer(-init_scale, init_scale)
  47 
  48   with tf.variable_scope('gen', reuse=reuse, initializer=initializer):
  49 
  50     def lstm_cell():
  51       return tf.contrib.rnn.LayerNormBasicLSTMCell(
  52           hparams.gen_rnn_size, reuse=reuse)
  53 
  54     attn_cell = lstm_cell
  55     if FLAGS.zoneout_drop_prob > 0.0:
  56 
  57       def attn_cell():
  58         return zoneout.ZoneoutWrapper(
  59             lstm_cell(),
  60             zoneout_drop_prob=FLAGS.zoneout_drop_prob,
  61             is_training=is_training)
  62 
  63     cell_gen = tf.contrib.rnn.MultiRNNCell(
  64         [attn_cell() for _ in range(hparams.gen_num_layers)],
  65         state_is_tuple=True)
  66 
  67     initial_state = cell_gen.zero_state(FLAGS.batch_size, tf.float32)
  68 
  69     with tf.variable_scope('rnn'):
  70       sequence, logits, log_probs = [], [], []
  71       embedding = tf.get_variable('embedding',
  72                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
  73       softmax_w = tf.get_variable('softmax_w',
  74                                   [hparams.gen_rnn_size, FLAGS.vocab_size])
  75       softmax_b = tf.get_variable('softmax_b', [FLAGS.vocab_size])
  76 
  77       rnn_inputs = tf.nn.embedding_lookup(embedding, inputs)
  78 
  79       for t in xrange(FLAGS.sequence_length):
  80         if t > 0:
  81           tf.get_variable_scope().reuse_variables()
  82 
  83         # Input to the model is the first token to provide context.  The
  84         # model will then predict token t > 0.
  85         if t == 0:
  86           # Always provide the real input at t = 0.
  87           state_gen = initial_state
  88           rnn_inp = rnn_inputs[:, t]
  89 
  90         # If the target at the last time-step was present, read in the real.
  91         # If the target at the last time-step was not present, read in the fake.
  92         else:
  93           real_rnn_inp = rnn_inputs[:, t]
  94           fake_rnn_inp = tf.nn.embedding_lookup(embedding, fake)
  95 
  96           # Use teacher forcing.
  97           if (is_training and
  98               FLAGS.gen_training_strategy == 'cross_entropy') or is_validating:
  99             rnn_inp = real_rnn_inp
 100           else:
 101             # Note that targets_t-1 == inputs_(t)
 102             rnn_inp = tf.where(targets_present[:, t - 1], real_rnn_inp,
 103                                fake_rnn_inp)
 104 
 105         # RNN.
 106         rnn_out, state_gen = cell_gen(rnn_inp, state_gen)
 107         logit = tf.matmul(rnn_out, softmax_w) + softmax_b
 108 
 109         # Real sample.
 110         real = targets[:, t]
 111 
 112         # Fake sample.
 113         categorical = tf.contrib.distributions.Categorical(logits=logit)
 114         fake = categorical.sample()
 115         log_prob = categorical.log_prob(fake)
 116 
 117         # Output for Generator will either be generated or the target.
 118         # If present:   Return real.
 119         # If not present:  Return fake.
 120         output = tf.where(targets_present[:, t], real, fake)
 121 
 122         # Append to lists.
 123         sequence.append(output)
 124         logits.append(logit)
 125         log_probs.append(log_prob)
 126 
 127       # Produce the RNN state had the model operated only
 128       # over real data.
 129       real_state_gen = initial_state
 130       for t in xrange(FLAGS.sequence_length):
 131         tf.get_variable_scope().reuse_variables()
 132 
 133         rnn_inp = rnn_inputs[:, t]
 134 
 135         # RNN.
 136         rnn_out, real_state_gen = cell_gen(rnn_inp, real_state_gen)
 137 
 138       final_state = real_state_gen
 139 
 140   return (tf.stack(sequence, axis=1), tf.stack(logits, axis=1), tf.stack(
 141       log_probs, axis=1), initial_state, final_state)

 144 def discriminator(hparams, sequence, is_training, reuse=None):
 145   """Define the Discriminator graph.
 146 
 147   Args:
 148     hparams:  Hyperparameters for the MaskGAN.
 149     FLAGS: Current flags.
 150     sequence:  [FLAGS.batch_size, FLAGS.sequence_length]
 151     is_training:
 152     reuse
 153 
 154   Returns:
 155     predictions:
 156   """
 157   tf.logging.warning(
 158       'Undirectional Discriminative model is not a useful model for this '
 159       'MaskGAN because future context is needed.  Use only for debugging '
 160       'purposes.')
 161   sequence = tf.cast(sequence, tf.int32)
 162 
 163   if FLAGS.dis_share_embedding:
 164     assert hparams.dis_rnn_size == hparams.gen_rnn_size, (
 165         'If you wish to share Discriminator/Generator embeddings, they must be'
 166         ' same dimension.')
 167     with tf.variable_scope('gen/rnn', reuse=True):
 168       embedding = tf.get_variable('embedding',
 169                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 170 
 171   with tf.variable_scope('dis', reuse=reuse):
 172 
 173     def lstm_cell():
 174       return tf.contrib.rnn.LayerNormBasicLSTMCell(
 175           hparams.dis_rnn_size, reuse=reuse)
 176 
 177     attn_cell = lstm_cell
 178     if FLAGS.zoneout_drop_prob > 0.0:
 179 
 180       def attn_cell():
 181         return zoneout.ZoneoutWrapper(
 182             lstm_cell(),
 183             zoneout_drop_prob=FLAGS.zoneout_drop_prob,
 184             is_training=is_training)
 185 
 186     cell_dis = tf.contrib.rnn.MultiRNNCell(
 187         [attn_cell() for _ in range(hparams.dis_num_layers)],
 188         state_is_tuple=True)
 189     state_dis = cell_dis.zero_state(FLAGS.batch_size, tf.float32)
 190 
 191     with tf.variable_scope('rnn') as vs:
 192       predictions = []
 193       if not FLAGS.dis_share_embedding:
 194         embedding = tf.get_variable('embedding',
 195                                     [FLAGS.vocab_size, hparams.dis_rnn_size])
 196 
 197       rnn_inputs = tf.nn.embedding_lookup(embedding, sequence)
 198 
 199       for t in xrange(FLAGS.sequence_length):
 200         if t > 0:
 201           tf.get_variable_scope().reuse_variables()
 202 
 203         rnn_in = rnn_inputs[:, t]
 204         rnn_out, state_dis = cell_dis(rnn_in, state_dis)
 205 
 206         # Prediction is linear output for Discriminator.
 207         pred = tf.contrib.layers.linear(rnn_out, 1, scope=vs)
 208 
 209         predictions.append(pred)
 210   predictions = tf.stack(predictions, axis=1)
 211   return tf.squeeze(predictions, axis=2)
