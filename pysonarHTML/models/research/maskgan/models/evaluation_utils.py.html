  35 def print_and_log_losses(log, step, is_present_rate, avg_dis_loss,
  36                          avg_gen_loss):
  37   """Prints and logs losses to the log file.
  38 
  39   Args:
  40     log: GFile for logs.
  41     step: Global step.
  42     is_present_rate: Current masking rate.
  43     avg_dis_loss: List of Discriminator losses.
  44     avg_gen_loss: List of Generator losses.
  45   """
  46   print('global_step: %d' % step)
  47   print(' is_present_rate: %.3f' % is_present_rate)
  48   print(' D train loss: %.5f' % np.mean(avg_dis_loss))
  49   print(' G train loss: %.5f' % np.mean(avg_gen_loss))
  50   log.write('\nglobal_step: %d\n' % step)
  51   log.write((' is_present_rate: %.3f\n' % is_present_rate))
  52   log.write(' D train loss: %.5f\n' % np.mean(avg_dis_loss))
  53   log.write(' G train loss: %.5f\n' % np.mean(avg_gen_loss))

  56 def print_and_log(log, id_to_word, sequence_eval, max_num_to_print=5):
  57   """Helper function for printing and logging evaluated sequences."""
  58   indices_arr = np.asarray(sequence_eval)
  59   samples = helper.convert_to_human_readable(id_to_word, indices_arr,
  60                                              max_num_to_print)
  61 
  62   for i, sample in enumerate(samples):
  63     print('Sample', i, '. ', sample)
  64     log.write('\nSample ' + str(i) + '. ' + sample)
  65   log.write('\n')
  66   print('\n')
  67   log.flush()
  68   return samples

  71 def zip_seq_pred_crossent(id_to_word, sequences, predictions, cross_entropy):
  72   """Zip together the sequences, predictions, cross entropy."""
  73   indices = np.asarray(sequences)
  74 
  75   batch_of_metrics = []
  76 
  77   for ind_batch, pred_batch, crossent_batch in zip(indices, predictions,
  78                                                    cross_entropy):
  79     metrics = []
  80 
  81     for index, pred, crossent in zip(ind_batch, pred_batch, crossent_batch):
  82       metrics.append([str(id_to_word[index]), pred, crossent])
  83 
  84     batch_of_metrics.append(metrics)
  85   return batch_of_metrics
  86

  88 def zip_metrics(indices, *args):
  89   """Zip together the indices matrices with the provided metrics matrices."""
  90   batch_of_metrics = []
  91   for metrics_batch in zip(indices, *args):
  92 
  93     metrics = []
  94     for m in zip(*metrics_batch):
  95       metrics.append(m)
  96     batch_of_metrics.append(metrics)
  97   return batch_of_metrics

 100 def print_formatted(present, id_to_word, log, batch_of_tuples):
 101   """Print and log metrics."""
 102   num_cols = len(batch_of_tuples[0][0])
 103   repeat_float_format = '{:<12.3f} '
 104   repeat_str_format = '{:<13}'
 105 
 106   format_str = ''.join(
 107       ['[{:<1}]  {:<20}',
 108        str(repeat_float_format * (num_cols - 1))])
 109 
 110   # TODO(liamfedus): Generalize the logging. This is sloppy.
 111   header_format_str = ''.join(
 112       ['[{:<1}]  {:<20}',
 113        str(repeat_str_format * (num_cols - 1))])
 114   header_str = header_format_str.format('p', 'Word', 'p(real)', 'log-perp',
 115                                         'log(p(a))', 'r', 'R=V*(s)', 'b=V(s)',
 116                                         'A(a,s)')
 117 
 118   for i, batch in enumerate(batch_of_tuples):
 119     print(' Sample: %d' % i)
 120     log.write(' Sample %d.\n' % i)
 121     print('  ', header_str)
 122     log.write('  ' + str(header_str) + '\n')
 123 
 124     for j, t in enumerate(batch):
 125       t = list(t)
 126       t[0] = id_to_word[t[0]]
 127       buffer_str = format_str.format(int(present[i][j]), *t)
 128       print('  ', buffer_str)
 129       log.write('  ' + str(buffer_str) + '\n')
 130   log.flush()

 133 def generate_RL_logs(sess, model, log, id_to_word, feed):
 134   """Generate complete logs while running with REINFORCE."""
 135   # Impute Sequences.
 136   [
 137       p,
 138       fake_sequence_eval,
 139       fake_predictions_eval,
 140       _,
 141       fake_cross_entropy_losses_eval,
 142       _,
 143       fake_log_probs_eval,
 144       fake_rewards_eval,
 145       fake_baselines_eval,
 146       cumulative_rewards_eval,
 147       fake_advantages_eval,
 148   ] = sess.run(
 149       [
 150           model.present,
 151           model.fake_sequence,
 152           model.fake_predictions,
 153           model.real_predictions,
 154           model.fake_cross_entropy_losses,
 155           model.fake_logits,
 156           model.fake_log_probs,
 157           model.fake_rewards,
 158           model.fake_baselines,
 159           model.cumulative_rewards,
 160           model.fake_advantages,
 161       ],
 162       feed_dict=feed)
 163 
 164   indices = np.asarray(fake_sequence_eval)
 165 
 166   # Convert Discriminator linear layer to probability.
 167   fake_prob_eval = expit(fake_predictions_eval)
 168 
 169   # Add metrics.
 170   fake_tuples = zip_metrics(indices, fake_prob_eval,
 171                             fake_cross_entropy_losses_eval, fake_log_probs_eval,
 172                             fake_rewards_eval, cumulative_rewards_eval,
 173                             fake_baselines_eval, fake_advantages_eval)
 174 
 175   # real_tuples = zip_metrics(indices, )
 176 
 177   # Print forward sequences.
 178   tuples_to_print = fake_tuples[:FLAGS.max_num_to_print]
 179   print_formatted(p, id_to_word, log, tuples_to_print)
 180 
 181   print('Samples')
 182   log.write('Samples\n')
 183   samples = print_and_log(log, id_to_word, fake_sequence_eval,
 184                           FLAGS.max_num_to_print)
 185   return samples

 188 def generate_logs(sess, model, log, id_to_word, feed):
 189   """Impute Sequences using the model for a particular feed and send it to
 190   logs."""
 191   # Impute Sequences.
 192   [
 193       p, sequence_eval, fake_predictions_eval, fake_cross_entropy_losses_eval,
 194       fake_logits_eval
 195   ] = sess.run(
 196       [
 197           model.present, model.fake_sequence, model.fake_predictions,
 198           model.fake_cross_entropy_losses, model.fake_logits
 199       ],
 200       feed_dict=feed)
 201 
 202   # Convert Discriminator linear layer to probability.
 203   fake_prob_eval = expit(fake_predictions_eval)
 204 
 205   # Forward Masked Tuples.
 206   fake_tuples = zip_seq_pred_crossent(id_to_word, sequence_eval, fake_prob_eval,
 207                                       fake_cross_entropy_losses_eval)
 208 
 209   tuples_to_print = fake_tuples[:FLAGS.max_num_to_print]
 210 
 211   if FLAGS.print_verbose:
 212     print('fake_logits_eval')
 213     print(fake_logits_eval)
 214 
 215   for i, batch in enumerate(tuples_to_print):
 216     print(' Sample %d.' % i)
 217     log.write(' Sample %d.\n' % i)
 218     for j, pred in enumerate(batch):
 219       buffer_str = ('[{:<1}]  {:<20}  {:<7.3f} {:<7.3f}').format(
 220           int(p[i][j]), pred[0], pred[1], pred[2])
 221       print('  ', buffer_str)
 222       log.write('  ' + str(buffer_str) + '\n')
 223   log.flush()
 224 
 225   print('Samples')
 226   log.write('Samples\n')
 227   samples = print_and_log(log, id_to_word, sequence_eval,
 228                           FLAGS.max_num_to_print)
 229   return samples

 232 def create_merged_ngram_dictionaries(indices, n):
 233   """Generate a single dictionary for the full batch.
 234 
 235   Args:
 236     indices:  List of lists of indices.
 237     n:  Degree of n-grams.
 238 
 239   Returns:
 240     Dictionary of hashed(n-gram tuples) to counts in the batch of indices.
 241   """
 242   ngram_dicts = []
 243 
 244   for ind in indices:
 245     ngrams = n_gram.find_all_ngrams(ind, n=n)
 246     ngram_counts = n_gram.construct_ngrams_dict(ngrams)
 247     ngram_dicts.append(ngram_counts)
 248 
 249   merged_gen_dict = Counter()
 250   for ngram_dict in ngram_dicts:
 251     merged_gen_dict += Counter(ngram_dict)
 252   return merged_gen_dict

 255 def sequence_ngram_evaluation(sess, sequence, log, feed, data_ngram_count, n):
 256   """Calculates the percent of ngrams produced in the sequence is present in
 257   data_ngram_count.
 258 
 259   Args:
 260     sess: tf.Session.
 261     sequence: Sequence Tensor from the MaskGAN model.
 262     log:  gFile log.
 263     feed: Feed to evaluate.
 264     data_ngram_count:  Dictionary of hashed(n-gram tuples) to counts in the
 265       data_set.
 266 
 267   Returns:
 268     avg_percent_captured: Percent of produced ngrams that appear in the
 269       data_ngram_count.
 270   """
 271   del log
 272   # Impute sequence.
 273   [sequence_eval] = sess.run([sequence], feed_dict=feed)
 274   indices = sequence_eval
 275 
 276   # Retrieve the counts across the batch of indices.
 277   gen_ngram_counts = create_merged_ngram_dictionaries(
 278       indices, n=n)
 279   return n_gram.percent_unique_ngrams_in_train(data_ngram_count,
 280                                                gen_ngram_counts)
