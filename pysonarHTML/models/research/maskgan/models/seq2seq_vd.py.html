  31 def transform_input_with_is_missing_token(inputs, targets_present):
  32   """Transforms the inputs to have missing tokens when it's masked out.  The
  33   mask is for the targets, so therefore, to determine if an input at time t is
  34   masked, we have to check if the target at time t - 1 is masked out.
  35 
  36   e.g.
  37     inputs = [a, b, c, d]
  38     targets = [b, c, d, e]
  39     targets_present = [1, 0, 1, 0]
  40 
  41   which computes,
  42     inputs_present = [1, 1, 0, 1]
  43 
  44   and outputs,
  45     transformed_input = [a, b, <missing>, d]
  46 
  47   Args:
  48     inputs:  tf.int32 Tensor of shape [batch_size, sequence_length] with tokens
  49       up to, but not including, vocab_size.
  50     targets_present:  tf.bool Tensor of shape [batch_size, sequence_length] with
  51       True representing the presence of the word.
  52 
  53   Returns:
  54     transformed_input:  tf.int32 Tensor of shape [batch_size, sequence_length]
  55       which takes on value of inputs when the input is present and takes on
  56       value=vocab_size to indicate a missing token.
  57   """
  58   # To fill in if the input is missing.
  59   input_missing = tf.constant(
  60       FLAGS.vocab_size,
  61       dtype=tf.int32,
  62       shape=[FLAGS.batch_size, FLAGS.sequence_length])
  63 
  64   # The 0th input will always be present to MaskGAN.
  65   zeroth_input_present = tf.constant(True, tf.bool, shape=[FLAGS.batch_size, 1])
  66 
  67   # Input present mask.
  68   inputs_present = tf.concat(
  69       [zeroth_input_present, targets_present[:, :-1]], axis=1)
  70 
  71   transformed_input = tf.where(inputs_present, inputs, input_missing)
  72   return transformed_input

  76 def gen_encoder(hparams, inputs, targets_present, is_training, reuse=None):
  77   """Define the Encoder graph.
  78 
  79   Args:
  80     hparams:  Hyperparameters for the MaskGAN.
  81     inputs:  tf.int32 Tensor of shape [batch_size, sequence_length] with tokens
  82       up to, but not including, vocab_size.
  83     targets_present:  tf.bool Tensor of shape [batch_size, sequence_length] with
  84       True representing the presence of the target.
  85     is_training:  Boolean indicating operational mode (train/inference).
  86     reuse (Optional):   Whether to reuse the variables.
  87 
  88   Returns:
  89     Tuple of (hidden_states, final_state).
  90   """
  91   # We will use the same variable from the decoder.
  92   if FLAGS.seq2seq_share_embedding:
  93     with tf.variable_scope('decoder/rnn'):
  94       embedding = tf.get_variable('embedding',
  95                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
  96 
  97   with tf.variable_scope('encoder', reuse=reuse):
  98 
  99     def lstm_cell():
 100       return tf.contrib.rnn.BasicLSTMCell(
 101           hparams.gen_rnn_size,
 102           forget_bias=0.0,
 103           state_is_tuple=True,
 104           reuse=reuse)
 105 
 106     attn_cell = lstm_cell
 107     if is_training and hparams.gen_vd_keep_prob < 1:
 108 
 109       def attn_cell():
 110         return variational_dropout.VariationalDropoutWrapper(
 111             lstm_cell(), FLAGS.batch_size, hparams.gen_rnn_size,
 112             hparams.gen_vd_keep_prob, hparams.gen_vd_keep_prob)
 113 
 114     cell = tf.contrib.rnn.MultiRNNCell(
 115         [attn_cell() for _ in range(hparams.gen_num_layers)],
 116         state_is_tuple=True)
 117 
 118     initial_state = cell.zero_state(FLAGS.batch_size, tf.float32)
 119 
 120     # Add a missing token for inputs not present.
 121     real_inputs = inputs
 122     masked_inputs = transform_input_with_is_missing_token(
 123         inputs, targets_present)
 124 
 125     with tf.variable_scope('rnn') as scope:
 126       hidden_states = []
 127 
 128       # Split the embedding into two parts so that we can load the PTB
 129       # weights into one part of the Variable.
 130       if not FLAGS.seq2seq_share_embedding:
 131         embedding = tf.get_variable('embedding',
 132                                     [FLAGS.vocab_size, hparams.gen_rnn_size])
 133       missing_embedding = tf.get_variable('missing_embedding',
 134                                           [1, hparams.gen_rnn_size])
 135       embedding = tf.concat([embedding, missing_embedding], axis=0)
 136 
 137       # TODO(adai): Perhaps append IMDB labels placeholder to input at
 138       # each time point.
 139       real_rnn_inputs = tf.nn.embedding_lookup(embedding, real_inputs)
 140       masked_rnn_inputs = tf.nn.embedding_lookup(embedding, masked_inputs)
 141 
 142       state = initial_state
 143 
 144       def make_mask(keep_prob, units):
 145         random_tensor = keep_prob
 146         # 0. if [keep_prob, 1.0) and 1. if [1.0, 1.0 + keep_prob)
 147         random_tensor += tf.random_uniform(
 148             tf.stack([FLAGS.batch_size, 1, units]))
 149         return tf.floor(random_tensor) / keep_prob
 150 
 151       if is_training:
 152         output_mask = make_mask(hparams.gen_vd_keep_prob, hparams.gen_rnn_size)
 153 
 154       hidden_states, state = tf.nn.dynamic_rnn(
 155           cell, masked_rnn_inputs, initial_state=state, scope=scope)
 156       if is_training:
 157         hidden_states *= output_mask
 158 
 159       final_masked_state = state
 160 
 161       # Produce the RNN state had the model operated only
 162       # over real data.
 163       real_state = initial_state
 164       _, real_state = tf.nn.dynamic_rnn(
 165           cell, real_rnn_inputs, initial_state=real_state, scope=scope)
 166       final_state = real_state
 167 
 168   return (hidden_states, final_masked_state), initial_state, final_state

 172 def gen_encoder_cnn(hparams, inputs, targets_present, is_training, reuse=None):
 173   """Define the CNN Encoder graph."""
 174   del reuse
 175   sequence = transform_input_with_is_missing_token(inputs, targets_present)
 176 
 177   # TODO(liamfedus): Make this a hyperparameter.
 178   dis_filter_sizes = [3, 4, 5, 6, 7, 8, 9, 10, 15, 20]
 179 
 180   # Keeping track of l2 regularization loss (optional)
 181   # l2_loss = tf.constant(0.0)
 182 
 183   with tf.variable_scope('encoder', reuse=True):
 184     with tf.variable_scope('rnn'):
 185       embedding = tf.get_variable('embedding',
 186                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 187 
 188   cnn_inputs = tf.nn.embedding_lookup(embedding, sequence)
 189 
 190   # Create a convolution layer for each filter size
 191   conv_outputs = []
 192   for filter_size in dis_filter_sizes:
 193     with tf.variable_scope('conv-%s' % filter_size):
 194       # Convolution Layer
 195       filter_shape = [
 196           filter_size, hparams.gen_rnn_size, hparams.dis_num_filters
 197       ]
 198       W = tf.get_variable(
 199           name='W', initializer=tf.truncated_normal(filter_shape, stddev=0.1))
 200       b = tf.get_variable(
 201           name='b',
 202           initializer=tf.constant(0.1, shape=[hparams.dis_num_filters]))
 203       conv = tf.nn.conv1d(cnn_inputs, W, stride=1, padding='SAME', name='conv')
 204 
 205       # Apply nonlinearity
 206       h = tf.nn.relu(tf.nn.bias_add(conv, b), name='relu')
 207 
 208       conv_outputs.append(h)
 209 
 210   # Combine all the pooled features
 211   dis_num_filters_total = hparams.dis_num_filters * len(dis_filter_sizes)
 212 
 213   h_conv = tf.concat(conv_outputs, axis=2)
 214   h_conv_flat = tf.reshape(h_conv, [-1, dis_num_filters_total])
 215 
 216   # Add dropout
 217   if is_training:
 218     with tf.variable_scope('dropout'):
 219       h_conv_flat = tf.nn.dropout(h_conv_flat, hparams.gen_vd_keep_prob)
 220 
 221   # Final (unnormalized) scores and predictions
 222   with tf.variable_scope('output'):
 223     W = tf.get_variable(
 224         'W',
 225         shape=[dis_num_filters_total, hparams.gen_rnn_size],
 226         initializer=tf.contrib.layers.xavier_initializer())
 227     b = tf.get_variable(
 228         name='b', initializer=tf.constant(0.1, shape=[hparams.gen_rnn_size]))
 229     # l2_loss += tf.nn.l2_loss(W)
 230     # l2_loss += tf.nn.l2_loss(b)
 231     predictions = tf.nn.xw_plus_b(h_conv_flat, W, b, name='predictions')
 232     predictions = tf.reshape(
 233         predictions,
 234         shape=[FLAGS.batch_size, FLAGS.sequence_length, hparams.gen_rnn_size])
 235   final_state = tf.reduce_mean(predictions, 1)
 236   return predictions, (final_state, final_state)

 239 # TODO(adai): IMDB labels placeholder to decoder.
 240 def gen_decoder(hparams,
 241                 inputs,
 242                 targets,
 243                 targets_present,
 244                 encoding_state,
 245                 is_training,
 246                 is_validating,
 247                 reuse=None):
 248   """Define the Decoder graph. The Decoder will now impute tokens that
 249       have been masked from the input seqeunce.
 250   """
 251   gen_decoder_rnn_size = hparams.gen_rnn_size
 252 
 253   targets = tf.Print(targets, [targets], message='targets', summarize=50)
 254   if FLAGS.seq2seq_share_embedding:
 255     with tf.variable_scope('decoder/rnn', reuse=True):
 256       embedding = tf.get_variable('embedding',
 257                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 258 
 259   with tf.variable_scope('decoder', reuse=reuse):
 260 
 261     def lstm_cell():
 262       return tf.contrib.rnn.BasicLSTMCell(
 263           gen_decoder_rnn_size,
 264           forget_bias=0.0,
 265           state_is_tuple=True,
 266           reuse=reuse)
 267 
 268     attn_cell = lstm_cell
 269     if is_training and hparams.gen_vd_keep_prob < 1:
 270 
 271       def attn_cell():
 272         return variational_dropout.VariationalDropoutWrapper(
 273             lstm_cell(), FLAGS.batch_size, hparams.gen_rnn_size,
 274             hparams.gen_vd_keep_prob, hparams.gen_vd_keep_prob)
 275 
 276     cell_gen = tf.contrib.rnn.MultiRNNCell(
 277         [attn_cell() for _ in range(hparams.gen_num_layers)],
 278         state_is_tuple=True)
 279 
 280     # Hidden encoder states.
 281     hidden_vector_encodings = encoding_state[0]
 282 
 283     # Carry forward the final state tuple from the encoder.
 284     # State tuples.
 285     state_gen = encoding_state[1]
 286 
 287     if FLAGS.attention_option is not None:
 288       (attention_keys, attention_values, _,
 289        attention_construct_fn) = attention_utils.prepare_attention(
 290            hidden_vector_encodings,
 291            FLAGS.attention_option,
 292            num_units=gen_decoder_rnn_size,
 293            reuse=reuse)
 294 
 295     def make_mask(keep_prob, units):
 296       random_tensor = keep_prob
 297       # 0. if [keep_prob, 1.0) and 1. if [1.0, 1.0 + keep_prob)
 298       random_tensor += tf.random_uniform(tf.stack([FLAGS.batch_size, units]))
 299       return tf.floor(random_tensor) / keep_prob
 300 
 301     if is_training:
 302       output_mask = make_mask(hparams.gen_vd_keep_prob, hparams.gen_rnn_size)
 303 
 304     with tf.variable_scope('rnn'):
 305       sequence, logits, log_probs = [], [], []
 306 
 307       if not FLAGS.seq2seq_share_embedding:
 308         embedding = tf.get_variable('embedding',
 309                                     [FLAGS.vocab_size, hparams.gen_rnn_size])
 310       softmax_w = tf.matrix_transpose(embedding)
 311       softmax_b = tf.get_variable('softmax_b', [FLAGS.vocab_size])
 312 
 313       rnn_inputs = tf.nn.embedding_lookup(embedding, inputs)
 314       # TODO(adai): Perhaps append IMDB labels placeholder to input at
 315       # each time point.
 316 
 317       rnn_outs = []
 318 
 319       fake = None
 320       for t in xrange(FLAGS.sequence_length):
 321         if t > 0:
 322           tf.get_variable_scope().reuse_variables()
 323 
 324         # Input to the Decoder.
 325         if t == 0:
 326           # Always provide the real input at t = 0.
 327           rnn_inp = rnn_inputs[:, t]
 328 
 329         # If the input is present, read in the input at t.
 330         # If the input is not present, read in the previously generated.
 331         else:
 332           real_rnn_inp = rnn_inputs[:, t]
 333 
 334           # While validating, the decoder should be operating in teacher
 335           # forcing regime.  Also, if we're just training with cross_entropy
 336           # use teacher forcing.
 337           if is_validating or FLAGS.gen_training_strategy == 'cross_entropy':
 338             rnn_inp = real_rnn_inp
 339           else:
 340             fake_rnn_inp = tf.nn.embedding_lookup(embedding, fake)
 341             rnn_inp = tf.where(targets_present[:, t - 1], real_rnn_inp,
 342                                fake_rnn_inp)
 343 
 344         # RNN.
 345         rnn_out, state_gen = cell_gen(rnn_inp, state_gen)
 346 
 347         if FLAGS.attention_option is not None:
 348           rnn_out = attention_construct_fn(rnn_out, attention_keys,
 349                                            attention_values)
 350         if is_training:
 351           rnn_out *= output_mask
 352 
 353         rnn_outs.append(rnn_out)
 354         if FLAGS.gen_training_strategy != 'cross_entropy':
 355           logit = tf.nn.bias_add(tf.matmul(rnn_out, softmax_w), softmax_b)
 356 
 357           # Output for Decoder.
 358           # If input is present:   Return real at t+1.
 359           # If input is not present:  Return fake for t+1.
 360           real = targets[:, t]
 361 
 362           categorical = tf.contrib.distributions.Categorical(logits=logit)
 363           if FLAGS.use_gen_mode:
 364             fake = categorical.mode()
 365           else:
 366             fake = categorical.sample()
 367           log_prob = categorical.log_prob(fake)
 368           output = tf.where(targets_present[:, t], real, fake)
 369 
 370         else:
 371           real = targets[:, t]
 372           logit = tf.zeros(tf.stack([FLAGS.batch_size, FLAGS.vocab_size]))
 373           log_prob = tf.zeros(tf.stack([FLAGS.batch_size]))
 374           output = real
 375 
 376         # Add to lists.
 377         sequence.append(output)
 378         log_probs.append(log_prob)
 379         logits.append(logit)
 380 
 381       if FLAGS.gen_training_strategy == 'cross_entropy':
 382         logits = tf.nn.bias_add(
 383             tf.matmul(
 384                 tf.reshape(tf.stack(rnn_outs, 1), [-1, gen_decoder_rnn_size]),
 385                 softmax_w), softmax_b)
 386         logits = tf.reshape(logits,
 387                             [-1, FLAGS.sequence_length, FLAGS.vocab_size])
 388       else:
 389         logits = tf.stack(logits, axis=1)
 390 
 391   return (tf.stack(sequence, axis=1), logits, tf.stack(log_probs, axis=1))

 394 def dis_encoder(hparams, masked_inputs, is_training, reuse=None,
 395                 embedding=None):
 396   """Define the Discriminator encoder.  Reads in the masked inputs for context
 397   and produces the hidden states of the encoder."""
 398   with tf.variable_scope('encoder', reuse=reuse):
 399 
 400     def lstm_cell():
 401       return tf.contrib.rnn.BasicLSTMCell(
 402           hparams.dis_rnn_size,
 403           forget_bias=0.0,
 404           state_is_tuple=True,
 405           reuse=reuse)
 406 
 407     attn_cell = lstm_cell
 408     if is_training and hparams.dis_vd_keep_prob < 1:
 409 
 410       def attn_cell():
 411         return variational_dropout.VariationalDropoutWrapper(
 412             lstm_cell(), FLAGS.batch_size, hparams.dis_rnn_size,
 413             hparams.dis_vd_keep_prob, hparams.dis_vd_keep_prob)
 414 
 415     cell_dis = tf.contrib.rnn.MultiRNNCell(
 416         [attn_cell() for _ in range(hparams.dis_num_layers)],
 417         state_is_tuple=True)
 418 
 419     state_dis = cell_dis.zero_state(FLAGS.batch_size, tf.float32)
 420 
 421     with tf.variable_scope('rnn'):
 422       hidden_states = []
 423 
 424       missing_embedding = tf.get_variable('missing_embedding',
 425                                           [1, hparams.dis_rnn_size])
 426       embedding = tf.concat([embedding, missing_embedding], axis=0)
 427       masked_rnn_inputs = tf.nn.embedding_lookup(embedding, masked_inputs)
 428 
 429       def make_mask(keep_prob, units):
 430         random_tensor = keep_prob
 431         # 0. if [keep_prob, 1.0) and 1. if [1.0, 1.0 + keep_prob)
 432         random_tensor += tf.random_uniform(tf.stack([FLAGS.batch_size, units]))
 433         return tf.floor(random_tensor) / keep_prob
 434 
 435       if is_training:
 436         output_mask = make_mask(hparams.dis_vd_keep_prob, hparams.dis_rnn_size)
 437 
 438       for t in xrange(FLAGS.sequence_length):
 439         if t > 0:
 440           tf.get_variable_scope().reuse_variables()
 441 
 442         rnn_in = masked_rnn_inputs[:, t]
 443         rnn_out, state_dis = cell_dis(rnn_in, state_dis)
 444         if is_training:
 445           rnn_out *= output_mask
 446         hidden_states.append(rnn_out)
 447       final_state = state_dis
 448 
 449   return (tf.stack(hidden_states, axis=1), final_state)

 452 def dis_decoder(hparams,
 453                 sequence,
 454                 encoding_state,
 455                 is_training,
 456                 reuse=None,
 457                 embedding=None):
 458   """Define the Discriminator decoder.  Read in the sequence and predict
 459     at each time point."""
 460   sequence = tf.cast(sequence, tf.int32)
 461 
 462   with tf.variable_scope('decoder', reuse=reuse):
 463 
 464     def lstm_cell():
 465       return tf.contrib.rnn.BasicLSTMCell(
 466           hparams.dis_rnn_size,
 467           forget_bias=0.0,
 468           state_is_tuple=True,
 469           reuse=reuse)
 470 
 471     attn_cell = lstm_cell
 472     if is_training and hparams.dis_vd_keep_prob < 1:
 473 
 474       def attn_cell():
 475         return variational_dropout.VariationalDropoutWrapper(
 476             lstm_cell(), FLAGS.batch_size, hparams.dis_rnn_size,
 477             hparams.dis_vd_keep_prob, hparams.dis_vd_keep_prob)
 478 
 479     cell_dis = tf.contrib.rnn.MultiRNNCell(
 480         [attn_cell() for _ in range(hparams.dis_num_layers)],
 481         state_is_tuple=True)
 482 
 483     # Hidden encoder states.
 484     hidden_vector_encodings = encoding_state[0]
 485 
 486     # Carry forward the final state tuple from the encoder.
 487     # State tuples.
 488     state = encoding_state[1]
 489 
 490     if FLAGS.attention_option is not None:
 491       (attention_keys, attention_values, _,
 492        attention_construct_fn) = attention_utils.prepare_attention(
 493            hidden_vector_encodings,
 494            FLAGS.attention_option,
 495            num_units=hparams.dis_rnn_size,
 496            reuse=reuse)
 497 
 498     def make_mask(keep_prob, units):
 499       random_tensor = keep_prob
 500       # 0. if [keep_prob, 1.0) and 1. if [1.0, 1.0 + keep_prob)
 501       random_tensor += tf.random_uniform(tf.stack([FLAGS.batch_size, units]))
 502       return tf.floor(random_tensor) / keep_prob
 503 
 504     if is_training:
 505       output_mask = make_mask(hparams.dis_vd_keep_prob, hparams.dis_rnn_size)
 506 
 507     with tf.variable_scope('rnn') as vs:
 508       predictions = []
 509 
 510       rnn_inputs = tf.nn.embedding_lookup(embedding, sequence)
 511 
 512       for t in xrange(FLAGS.sequence_length):
 513         if t > 0:
 514           tf.get_variable_scope().reuse_variables()
 515 
 516         rnn_in = rnn_inputs[:, t]
 517         rnn_out, state = cell_dis(rnn_in, state)
 518 
 519         if FLAGS.attention_option is not None:
 520           rnn_out = attention_construct_fn(rnn_out, attention_keys,
 521                                            attention_values)
 522         if is_training:
 523           rnn_out *= output_mask
 524 
 525         # Prediction is linear output for Discriminator.
 526         pred = tf.contrib.layers.linear(rnn_out, 1, scope=vs)
 527         predictions.append(pred)
 528 
 529   predictions = tf.stack(predictions, axis=1)
 530   return tf.squeeze(predictions, axis=2)

 533 def discriminator(hparams,
 534                   inputs,
 535                   targets_present,
 536                   sequence,
 537                   is_training,
 538                   reuse=None):
 539   """Define the Discriminator graph."""
 540   if FLAGS.dis_share_embedding:
 541     assert hparams.dis_rnn_size == hparams.gen_rnn_size, (
 542         'If you wish to share Discriminator/Generator embeddings, they must be'
 543         ' same dimension.')
 544     with tf.variable_scope('gen/decoder/rnn', reuse=True):
 545       embedding = tf.get_variable('embedding',
 546                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 547   else:
 548     # Explicitly share the embedding.
 549     with tf.variable_scope('dis/decoder/rnn', reuse=reuse):
 550       embedding = tf.get_variable('embedding',
 551                                   [FLAGS.vocab_size, hparams.dis_rnn_size])
 552 
 553   # Mask the input sequence.
 554   masked_inputs = transform_input_with_is_missing_token(inputs, targets_present)
 555 
 556   # Confirm masking.
 557   masked_inputs = tf.Print(
 558       masked_inputs, [inputs, targets_present, masked_inputs, sequence],
 559       message='inputs, targets_present, masked_inputs, sequence',
 560       summarize=10)
 561 
 562   with tf.variable_scope('dis', reuse=reuse):
 563     encoder_states = dis_encoder(
 564         hparams,
 565         masked_inputs,
 566         is_training=is_training,
 567         reuse=reuse,
 568         embedding=embedding)
 569     predictions = dis_decoder(
 570         hparams,
 571         sequence,
 572         encoder_states,
 573         is_training=is_training,
 574         reuse=reuse,
 575         embedding=embedding)
 576 
 577   # if FLAGS.baseline_method == 'critic':
 578   #   with tf.variable_scope('critic', reuse=reuse) as critic_scope:
 579   #     values = tf.contrib.layers.linear(rnn_outs, 1, scope=critic_scope)
 580   #     values = tf.squeeze(values, axis=2)
 581   # else:
 582   #   values = None
 583 
 584   return predictions

 588 def generator(hparams,
 589               inputs,
 590               targets,
 591               targets_present,
 592               is_training,
 593               is_validating,
 594               reuse=None):
 595   """Define the Generator graph."""
 596   with tf.variable_scope('gen', reuse=reuse):
 597     encoder_states, initial_state, final_state = gen_encoder(
 598         hparams, inputs, targets_present, is_training=is_training, reuse=reuse)
 599     stacked_sequence, stacked_logits, stacked_log_probs = gen_decoder(
 600         hparams,
 601         inputs,
 602         targets,
 603         targets_present,
 604         encoder_states,
 605         is_training=is_training,
 606         is_validating=is_validating,
