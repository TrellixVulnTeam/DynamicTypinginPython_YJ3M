  41 def generator(hparams,
  42               inputs,
  43               targets,
  44               targets_present,
  45               is_training,
  46               is_validating,
  47               reuse=None):
  48   """Define the Generator graph.
  49 
  50     G will now impute tokens that have been masked from the input seqeunce.
  51   """
  52   tf.logging.info(
  53       'Undirectional generative model is not a useful model for this MaskGAN '
  54       'because future context is needed.  Use only for debugging purposes.')
  55   config = get_config()
  56   config.keep_prob = [hparams.gen_nas_keep_prob_0, hparams.gen_nas_keep_prob_1]
  57   configs.print_config(config)
  58 
  59   init_scale = config.init_scale
  60   initializer = tf.random_uniform_initializer(-init_scale, init_scale)
  61 
  62   with tf.variable_scope('gen', reuse=reuse, initializer=initializer):
  63     # Neural architecture search cell.
  64     cell = custom_cell.Alien(config.hidden_size)
  65 
  66     if is_training:
  67       [h2h_masks, _, _,
  68        output_mask] = variational_dropout.generate_variational_dropout_masks(
  69            hparams, config.keep_prob)
  70     else:
  71       output_mask = None
  72 
  73     cell_gen = custom_cell.GenericMultiRNNCell([cell] * config.num_layers)
  74     initial_state = cell_gen.zero_state(FLAGS.batch_size, tf.float32)
  75 
  76     with tf.variable_scope('rnn'):
  77       sequence, logits, log_probs = [], [], []
  78       embedding = tf.get_variable('embedding',
  79                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
  80       softmax_w = tf.matrix_transpose(embedding)
  81       softmax_b = tf.get_variable('softmax_b', [FLAGS.vocab_size])
  82 
  83       rnn_inputs = tf.nn.embedding_lookup(embedding, inputs)
  84 
  85       if is_training and FLAGS.keep_prob < 1:
  86         rnn_inputs = tf.nn.dropout(rnn_inputs, FLAGS.keep_prob)
  87 
  88       for t in xrange(FLAGS.sequence_length):
  89         if t > 0:
  90           tf.get_variable_scope().reuse_variables()
  91 
  92         # Input to the model is the first token to provide context.  The
  93         # model will then predict token t > 0.
  94         if t == 0:
  95           # Always provide the real input at t = 0.
  96           state_gen = initial_state
  97           rnn_inp = rnn_inputs[:, t]
  98 
  99         # If the input is present, read in the input at t.
 100         # If the input is not present, read in the previously generated.
 101         else:
 102           real_rnn_inp = rnn_inputs[:, t]
 103           fake_rnn_inp = tf.nn.embedding_lookup(embedding, fake)
 104 
 105           # While validating, the decoder should be operating in teacher
 106           # forcing regime.  Also, if we're just training with cross_entropy
 107           # use teacher forcing.
 108           if is_validating or (is_training and
 109                                FLAGS.gen_training_strategy == 'cross_entropy'):
 110             rnn_inp = real_rnn_inp
 111           else:
 112             rnn_inp = tf.where(targets_present[:, t - 1], real_rnn_inp,
 113                                fake_rnn_inp)
 114 
 115         if is_training:
 116           state_gen = list(state_gen)
 117           for layer_num, per_layer_state in enumerate(state_gen):
 118             per_layer_state = LSTMTuple(
 119                 per_layer_state[0], per_layer_state[1] * h2h_masks[layer_num])
 120             state_gen[layer_num] = per_layer_state
 121 
 122         # RNN.
 123         rnn_out, state_gen = cell_gen(rnn_inp, state_gen)
 124 
 125         if is_training:
 126           rnn_out = output_mask * rnn_out
 127 
 128         logit = tf.matmul(rnn_out, softmax_w) + softmax_b
 129 
 130         # Real sample.
 131         real = targets[:, t]
 132 
 133         categorical = tf.contrib.distributions.Categorical(logits=logit)
 134         fake = categorical.sample()
 135         log_prob = categorical.log_prob(fake)
 136 
 137         # Output for Generator will either be generated or the input.
 138         #
 139         # If present:   Return real.
 140         # If not present:  Return fake.
 141         output = tf.where(targets_present[:, t], real, fake)
 142 
 143         # Add to lists.
 144         sequence.append(output)
 145         log_probs.append(log_prob)
 146         logits.append(logit)
 147 
 148       # Produce the RNN state had the model operated only
 149       # over real data.
 150       real_state_gen = initial_state
 151       for t in xrange(FLAGS.sequence_length):
 152         tf.get_variable_scope().reuse_variables()
 153 
 154         rnn_inp = rnn_inputs[:, t]
 155 
 156         # RNN.
 157         rnn_out, real_state_gen = cell_gen(rnn_inp, real_state_gen)
 158 
 159       final_state = real_state_gen
 160 
 161   return (tf.stack(sequence, axis=1), tf.stack(logits, axis=1), tf.stack(
 162       log_probs, axis=1), initial_state, final_state)

 165 def discriminator(hparams, sequence, is_training, reuse=None):
 166   """Define the Discriminator graph."""
 167   tf.logging.info(
 168       'Undirectional Discriminative model is not a useful model for this '
 169       'MaskGAN because future context is needed.  Use only for debugging '
 170       'purposes.')
 171   sequence = tf.cast(sequence, tf.int32)
 172 
 173   if FLAGS.dis_share_embedding:
 174     assert hparams.dis_rnn_size == hparams.gen_rnn_size, (
 175         'If you wish to share Discriminator/Generator embeddings, they must be'
 176         ' same dimension.')
 177     with tf.variable_scope('gen/rnn', reuse=True):
 178       embedding = tf.get_variable('embedding',
 179                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 180 
 181   config = get_config()
 182   config.keep_prob = [hparams.dis_nas_keep_prob_0, hparams.dis_nas_keep_prob_1]
 183   configs.print_config(config)
 184 
 185   with tf.variable_scope('dis', reuse=reuse):
 186     # Neural architecture search cell.
 187     cell = custom_cell.Alien(config.hidden_size)
 188 
 189     if is_training:
 190       [h2h_masks, _, _,
 191        output_mask] = variational_dropout.generate_variational_dropout_masks(
 192            hparams, config.keep_prob)
 193     else:
 194       output_mask = None
 195 
 196     cell_dis = custom_cell.GenericMultiRNNCell([cell] * config.num_layers)
 197     state_dis = cell_dis.zero_state(FLAGS.batch_size, tf.float32)
 198 
 199     with tf.variable_scope('rnn') as vs:
 200       predictions = []
 201       if not FLAGS.dis_share_embedding:
 202         embedding = tf.get_variable('embedding',
 203                                     [FLAGS.vocab_size, hparams.dis_rnn_size])
 204 
 205       rnn_inputs = tf.nn.embedding_lookup(embedding, sequence)
 206 
 207       if is_training and FLAGS.keep_prob < 1:
 208         rnn_inputs = tf.nn.dropout(rnn_inputs, FLAGS.keep_prob)
 209 
 210       for t in xrange(FLAGS.sequence_length):
 211         if t > 0:
 212           tf.get_variable_scope().reuse_variables()
 213 
 214         rnn_in = rnn_inputs[:, t]
 215 
 216         if is_training:
 217           state_dis = list(state_dis)
 218           for layer_num, per_layer_state in enumerate(state_dis):
 219             per_layer_state = LSTMTuple(
 220                 per_layer_state[0], per_layer_state[1] * h2h_masks[layer_num])
 221             state_dis[layer_num] = per_layer_state
 222 
 223         # RNN.
 224         rnn_out, state_dis = cell_dis(rnn_in, state_dis)
 225 
 226         if is_training:
 227           rnn_out = output_mask * rnn_out
 228 
 229         # Prediction is linear output for Discriminator.
 230         pred = tf.contrib.layers.linear(rnn_out, 1, scope=vs)
 231 
 232         predictions.append(pred)
 233   predictions = tf.stack(predictions, axis=1)
 234   return tf.squeeze(predictions, axis=2)
