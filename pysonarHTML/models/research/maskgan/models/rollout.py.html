  36 def create_rollout_MaskGAN(hparams, is_training):
  37   """Create the MaskGAN model.
  38 
  39   Args:
  40     hparams:  Hyperparameters for the MaskGAN.
  41     is_training:  Boolean indicating operational mode (train/inference).
  42       evaluated with a teacher forcing regime.
  43 
  44   Return:
  45     model:  Namedtuple for specifying the MaskGAN."""
  46   global_step = tf.Variable(0, name='global_step', trainable=False)
  47 
  48   new_learning_rate = tf.placeholder(tf.float32, [], name='new_learning_rate')
  49   learning_rate = tf.Variable(0.0, name='learning_rate', trainable=False)
  50   learning_rate_update = tf.assign(learning_rate, new_learning_rate)
  51 
  52   new_rate = tf.placeholder(tf.float32, [], name='new_rate')
  53   percent_real_var = tf.Variable(0.0, trainable=False)
  54   percent_real_update = tf.assign(percent_real_var, new_rate)
  55 
  56   ## Placeholders.
  57   inputs = tf.placeholder(
  58       tf.int32, shape=[FLAGS.batch_size, FLAGS.sequence_length])
  59   present = tf.placeholder(
  60       tf.bool, shape=[FLAGS.batch_size, FLAGS.sequence_length])
  61   inv_present = tf.placeholder(
  62       tf.bool, shape=[FLAGS.batch_size, FLAGS.sequence_length])
  63 
  64   ## Rollout Generator.
  65   fwd_gen_rollouts = rollout_generator(
  66       hparams, inputs, present, is_training=is_training, is_validating=False)
  67   inv_gen_rollouts = rollout_generator(
  68       hparams,
  69       inputs,
  70       inv_present,
  71       is_training=is_training,
  72       is_validating=False,
  73       reuse=True)
  74 
  75   ## Rollout Discriminator.
  76   fwd_dis_rollouts = rollout_discriminator(
  77       hparams, fwd_gen_rollouts, is_training=is_training)
  78   inv_dis_rollouts = rollout_discriminator(
  79       hparams, inv_gen_rollouts, is_training=is_training, reuse=True)
  80 
  81   ## Discriminator Loss.
  82   [dis_loss, dis_loss_pred, dis_loss_inv_pred] = rollout_discriminator_loss(
  83       fwd_dis_rollouts, present, inv_dis_rollouts, inv_present)
  84 
  85   ## Average log-perplexity for only missing words.  However, to do this,
  86   # the logits are still computed using teacher forcing, that is, the ground
  87   # truth tokens are fed in at each time point to be valid.
  88   # TODO(liamfedus): Fix the naming convention.
  89   with tf.variable_scope('gen_rollout'):
  90     _, fwd_eval_logits, _ = model_construction.create_generator(
  91         hparams,
  92         inputs,
  93         present,
  94         is_training=False,
  95         is_validating=True,
  96         reuse=True)
  97 
  98   avg_log_perplexity = model_losses.calculate_log_perplexity(
  99       fwd_eval_logits, inputs, present)
 100 
 101   ## Generator Loss.
 102   # 1.  Cross Entropy losses on missing tokens.
 103   [fwd_cross_entropy_losses,
 104    inv_cross_entropy_losses] = rollout_masked_cross_entropy_loss(
 105        inputs, present, inv_present, fwd_gen_rollouts, inv_gen_rollouts)
 106 
 107   # 2.  GAN losses on missing tokens.
 108   [fwd_RL_loss,
 109    fwd_RL_statistics, fwd_averages_op] = rollout_reinforce_objective(
 110        hparams, fwd_gen_rollouts, fwd_dis_rollouts, present)
 111   [inv_RL_loss,
 112    inv_RL_statistics, inv_averages_op] = rollout_reinforce_objective(
 113        hparams, inv_gen_rollouts, inv_dis_rollouts, inv_present)
 114 
 115   # TODO(liamfedus):  Generalize this to use all logs.
 116   [fwd_sequence, fwd_logits, fwd_log_probs] = fwd_gen_rollouts[-1]
 117   [inv_sequence, inv_logits, inv_log_probs] = inv_gen_rollouts[-1]
 118 
 119   # TODO(liamfedus):  Generalize this to use all logs.
 120   fwd_predictions = fwd_dis_rollouts[-1]
 121   inv_predictions = inv_dis_rollouts[-1]
 122 
 123   # TODO(liamfedus):  Generalize this to use all logs.
 124   [fwd_log_probs, fwd_rewards, fwd_advantages,
 125    fwd_baselines] = fwd_RL_statistics[-1]
 126   [inv_log_probs, inv_rewards, inv_advantages,
 127    inv_baselines] = inv_RL_statistics[-1]
 128 
 129   ## Pre-training.
 130   if FLAGS.gen_pretrain_steps:
 131     # TODO(liamfedus): Rewrite this.
 132     fwd_cross_entropy_loss = tf.reduce_mean(fwd_cross_entropy_losses)
 133     gen_pretrain_op = model_optimization.create_gen_pretrain_op(
 134         hparams, fwd_cross_entropy_loss, global_step)
 135   else:
 136     gen_pretrain_op = tf.no_op('gen_pretrain_no_op')
 137   if FLAGS.dis_pretrain_steps:
 138     dis_pretrain_op = model_optimization.create_dis_pretrain_op(
 139         hparams, dis_loss, global_step)
 140   else:
 141     dis_pretrain_op = tf.no_op('dis_pretrain_no_op')
 142 
 143   ##  Generator Train Op.
 144   # 1.  Cross-Entropy.
 145   if FLAGS.gen_training_strategy == 'cross_entropy':
 146     gen_loss = tf.reduce_mean(
 147         fwd_cross_entropy_losses + inv_cross_entropy_losses) / 2.
 148     [gen_train_op, gen_grads,
 149      gen_vars] = model_optimization.create_gen_train_op(
 150          hparams, learning_rate, gen_loss, global_step, mode='MINIMIZE')
 151 
 152   # 2.  GAN (REINFORCE)
 153   elif FLAGS.gen_training_strategy == 'reinforce':
 154     gen_loss = (fwd_RL_loss + inv_RL_loss) / 2.
 155     [gen_train_op, gen_grads,
 156      gen_vars] = model_optimization.create_reinforce_gen_train_op(
 157          hparams, learning_rate, gen_loss, fwd_averages_op, inv_averages_op,
 158          global_step)
 159 
 160   else:
 161     raise NotImplementedError
 162 
 163   ## Discriminator Train Op.
 164   dis_train_op, dis_grads, dis_vars = model_optimization.create_dis_train_op(
 165       hparams, dis_loss, global_step)
 166 
 167   ## Summaries.
 168   with tf.name_scope('general'):
 169     tf.summary.scalar('percent_real', percent_real_var)
 170     tf.summary.scalar('learning_rate', learning_rate)
 171 
 172   with tf.name_scope('generator_losses'):
 173     tf.summary.scalar('gen_loss', tf.reduce_mean(gen_loss))
 174     tf.summary.scalar('gen_loss_fwd_cross_entropy',
 175                       tf.reduce_mean(fwd_cross_entropy_losses))
 176     tf.summary.scalar('gen_loss_inv_cross_entropy',
 177                       tf.reduce_mean(inv_cross_entropy_losses))
 178 
 179   with tf.name_scope('REINFORCE'):
 180     with tf.name_scope('objective'):
 181       tf.summary.scalar('fwd_RL_loss', tf.reduce_mean(fwd_RL_loss))
 182       tf.summary.scalar('inv_RL_loss', tf.reduce_mean(inv_RL_loss))
 183 
 184     with tf.name_scope('rewards'):
 185       helper.variable_summaries(fwd_rewards, 'fwd_rewards')
 186       helper.variable_summaries(inv_rewards, 'inv_rewards')
 187 
 188     with tf.name_scope('advantages'):
 189       helper.variable_summaries(fwd_advantages, 'fwd_advantages')
 190       helper.variable_summaries(inv_advantages, 'inv_advantages')
 191 
 192     with tf.name_scope('baselines'):
 193       helper.variable_summaries(fwd_baselines, 'fwd_baselines')
 194       helper.variable_summaries(inv_baselines, 'inv_baselines')
 195 
 196     with tf.name_scope('log_probs'):
 197       helper.variable_summaries(fwd_log_probs, 'fwd_log_probs')
 198       helper.variable_summaries(inv_log_probs, 'inv_log_probs')
 199 
 200   with tf.name_scope('discriminator_losses'):
 201     tf.summary.scalar('dis_loss', dis_loss)
 202     tf.summary.scalar('dis_loss_fwd_sequence', dis_loss_pred)
 203     tf.summary.scalar('dis_loss_inv_sequence', dis_loss_inv_pred)
 204 
 205   with tf.name_scope('logits'):
 206     helper.variable_summaries(fwd_logits, 'fwd_logits')
 207     helper.variable_summaries(inv_logits, 'inv_logits')
 208 
 209   for v, g in zip(gen_vars, gen_grads):
 210     helper.variable_summaries(v, v.op.name)
 211     helper.variable_summaries(g, 'grad/' + v.op.name)
 212 
 213   for v, g in zip(dis_vars, dis_grads):
 214     helper.variable_summaries(v, v.op.name)
 215     helper.variable_summaries(g, 'grad/' + v.op.name)
 216 
 217   merge_summaries_op = tf.summary.merge_all()
 218 
 219   # Model saver.
 220   saver = tf.train.Saver(keep_checkpoint_every_n_hours=1, max_to_keep=5)
 221 
 222   # Named tuple that captures elements of the MaskGAN model.
 223   Model = collections.namedtuple('Model', [
 224       'inputs', 'present', 'inv_present', 'percent_real_update', 'new_rate',
 225       'fwd_sequence', 'fwd_logits', 'fwd_rewards', 'fwd_advantages',
 226       'fwd_log_probs', 'fwd_predictions', 'fwd_cross_entropy_losses',
 227       'inv_sequence', 'inv_logits', 'inv_rewards', 'inv_advantages',
 228       'inv_log_probs', 'inv_predictions', 'inv_cross_entropy_losses',
 229       'avg_log_perplexity', 'dis_loss', 'gen_loss', 'dis_train_op',
 230       'gen_train_op', 'gen_pretrain_op', 'dis_pretrain_op',
 231       'merge_summaries_op', 'global_step', 'new_learning_rate',
 232       'learning_rate_update', 'saver'
 233   ])
 234 
 235   model = Model(
 236       inputs, present, inv_present, percent_real_update, new_rate, fwd_sequence,
 237       fwd_logits, fwd_rewards, fwd_advantages, fwd_log_probs, fwd_predictions,
 238       fwd_cross_entropy_losses, inv_sequence, inv_logits, inv_rewards,
 239       inv_advantages, inv_log_probs, inv_predictions, inv_cross_entropy_losses,
 240       avg_log_perplexity, dis_loss, gen_loss, dis_train_op, gen_train_op,
 241       gen_pretrain_op, dis_pretrain_op, merge_summaries_op, global_step,
 242       new_learning_rate, learning_rate_update, saver)
 243   return model

 246 def rollout_generator(hparams,
 247                       inputs,
 248                       input_present,
 249                       is_training,
 250                       is_validating,
 251                       reuse=None):
 252   """Define the Generator graph which does rollouts.
 253 
 254     G will now impute tokens that have been masked from the input seqeunce.
 255   """
 256   rollouts = []
 257 
 258   with tf.variable_scope('gen_rollout'):
 259     for n in xrange(FLAGS.num_rollouts):
 260       if n > 0:
 261         # TODO(liamfedus): Why is it necessary here to manually set reuse?
 262         reuse = True
 263         tf.get_variable_scope().reuse_variables()
 264 
 265       [sequence, logits, log_probs] = model_construction.create_generator(
 266           hparams,
 267           inputs,
 268           input_present,
 269           is_training,
 270           is_validating,
 271           reuse=reuse)
 272 
 273       rollouts.append([sequence, logits, log_probs])
 274 
 275   # Length assertion.
 276   assert len(rollouts) == FLAGS.num_rollouts
 277 
 278   return rollouts

 281 def rollout_discriminator(hparams, gen_rollouts, is_training, reuse=None):
 282   """Define the Discriminator graph which does rollouts.
 283 
 284     G will now impute tokens that have been masked from the input seqeunce.
 285   """
 286   rollout_predictions = []
 287 
 288   with tf.variable_scope('dis_rollout'):
 289     for n, rollout in enumerate(gen_rollouts):
 290       if n > 0:
 291         # TODO(liamfedus): Why is it necessary here to manually set reuse?
 292         reuse = True
 293         tf.get_variable_scope().reuse_variables()
 294 
 295       [sequence, _, _] = rollout
 296 
 297       predictions = model_construction.create_discriminator(
 298           hparams, sequence, is_training=is_training, reuse=reuse)
 299 
 300       # Predictions for each rollout.
 301       rollout_predictions.append(predictions)
 302 
 303   # Length assertion.
 304   assert len(rollout_predictions) == FLAGS.num_rollouts
 305 
 306   return rollout_predictions

 309 def rollout_reinforce_objective(hparams, gen_rollouts, dis_rollouts, present):
 310   cumulative_gen_objective = 0.
 311   cumulative_averages_op = []
 312   cumulative_statistics = []
 313 
 314   assert len(gen_rollouts) == len(dis_rollouts)
 315 
 316   for gen_rollout, dis_rollout in zip(gen_rollouts, dis_rollouts):
 317     [_, _, log_probs] = gen_rollout
 318     dis_predictions = dis_rollout
 319 
 320     [
 321         final_gen_objective, log_probs, rewards, advantages, baselines,
 322         maintain_averages_op
 323     ] = model_losses.calculate_reinforce_objective(hparams, log_probs,
 324                                                    dis_predictions, present)
 325 
 326     # Accumulate results.
 327     cumulative_gen_objective += final_gen_objective
 328     cumulative_averages_op.append(maintain_averages_op)
 329     cumulative_statistics.append([log_probs, rewards, advantages, baselines])
 330 
 331   # Group all the averaging operations.
 332   cumulative_averages_op = tf.group(*cumulative_averages_op)
 333   cumulative_gen_objective /= FLAGS.num_rollouts
 334   [log_probs, rewards, advantages, baselines] = cumulative_statistics[-1]
 335 
 336   # Length assertion.
 337   assert len(cumulative_statistics) == FLAGS.num_rollouts
 338 
 339   return [
 340       cumulative_gen_objective, cumulative_statistics, cumulative_averages_op
 341   ]

 344 def rollout_masked_cross_entropy_loss(inputs, present, inv_present,
 345                                       fwd_rollouts, inv_rollouts):
 346   cumulative_fwd_cross_entropy_losses = tf.zeros(
 347       shape=[FLAGS.batch_size, FLAGS.sequence_length])
 348   cumulative_inv_cross_entropy_losses = tf.zeros(
 349       shape=[FLAGS.batch_size, FLAGS.sequence_length])
 350 
 351   for fwd_rollout, inv_rollout in zip(fwd_rollouts, inv_rollouts):
 352     [_, fwd_logits, _] = fwd_rollout
 353     [_, inv_logits, _] = inv_rollout
 354 
 355     [fwd_cross_entropy_losses,
 356      inv_cross_entropy_losses] = model_losses.create_masked_cross_entropy_loss(
 357          inputs, present, inv_present, fwd_logits, inv_logits)
 358 
 359     cumulative_fwd_cross_entropy_losses = tf.add(
 360         cumulative_fwd_cross_entropy_losses, fwd_cross_entropy_losses)
 361     cumulative_inv_cross_entropy_losses = tf.add(
 362         cumulative_inv_cross_entropy_losses, inv_cross_entropy_losses)
 363 
 364   return [
 365       cumulative_fwd_cross_entropy_losses, cumulative_inv_cross_entropy_losses
 366   ]

 369 def rollout_discriminator_loss(fwd_rollouts, present, inv_rollouts,
 370                                inv_present):
 371 
 372   dis_loss = 0
 373   dis_loss_pred = 0
 374   dis_loss_inv_pred = 0
 375 
 376   for fwd_predictions, inv_predictions in zip(fwd_rollouts, inv_rollouts):
 377     dis_loss_pred += losses.discriminator_loss(fwd_predictions, present)
 378     dis_loss_inv_pred += losses.discriminator_loss(inv_predictions, inv_present)
 379 
 380   dis_loss_pred /= FLAGS.num_rollouts
 381   dis_loss_inv_pred /= FLAGS.num_rollouts
 382 
 383   dis_loss = (dis_loss_pred + dis_loss_inv_pred) / 2.
 384   return [dis_loss, dis_loss_pred, dis_loss_inv_pred]
