  43 def transform_input_with_is_missing_token(inputs, targets_present):
  44   """Transforms the inputs to have missing tokens when it's masked out.  The
  45   mask is for the targets, so therefore, to determine if an input at time t is
  46   masked, we have to check if the target at time t - 1 is masked out.
  47 
  48   e.g.
  49     inputs = [a, b, c, d]
  50     targets = [b, c, d, e]
  51     targets_present = [1, 0, 1, 0]
  52 
  53   then,
  54     transformed_input = [a, b, <missing>, d]
  55 
  56   Args:
  57     inputs:  tf.int32 Tensor of shape [batch_size, sequence_length] with tokens
  58       up to, but not including, vocab_size.
  59     targets_present:  tf.bool Tensor of shape [batch_size, sequence_length] with
  60       True representing the presence of the word.
  61 
  62   Returns:
  63     transformed_input:  tf.int32 Tensor of shape [batch_size, sequence_length]
  64       which takes on value of inputs when the input is present and takes on
  65       value=vocab_size to indicate a missing token.
  66   """
  67   # To fill in if the input is missing.
  68   input_missing = tf.constant(
  69       FLAGS.vocab_size,
  70       dtype=tf.int32,
  71       shape=[FLAGS.batch_size, FLAGS.sequence_length])
  72 
  73   # The 0th input will always be present to MaskGAN.
  74   zeroth_input_present = tf.constant(True, tf.bool, shape=[FLAGS.batch_size, 1])
  75 
  76   # Input present mask.
  77   inputs_present = tf.concat(
  78       [zeroth_input_present, targets_present[:, :-1]], axis=1)
  79 
  80   transformed_input = tf.where(inputs_present, inputs, input_missing)
  81   return transformed_input

  84 def gen_encoder(hparams, inputs, targets_present, is_training, reuse=None):
  85   """Define the Encoder graph.
  86 
  87 
  88   Args:
  89     hparams:  Hyperparameters for the MaskGAN.
  90     inputs:  tf.int32 Tensor of shape [batch_size, sequence_length] with tokens
  91       up to, but not including, vocab_size.
  92     targets_present:  tf.bool Tensor of shape [batch_size, sequence_length] with
  93       True representing the presence of the target.
  94     is_training:  Boolean indicating operational mode (train/inference).
  95     reuse (Optional):   Whether to reuse the variables.
  96 
  97   Returns:
  98     Tuple of (hidden_states, final_state).
  99   """
 100   config = get_config()
 101   configs.print_config(config)
 102   # We will use the same variable from the decoder.
 103   if FLAGS.seq2seq_share_embedding:
 104     with tf.variable_scope('decoder/rnn'):
 105       embedding = tf.get_variable('embedding',
 106                                   [FLAGS.vocab_size, hparams.gen_rnn_size])
 107 
 108   with tf.variable_scope('encoder', reuse=reuse):
 109     # Neural architecture search cell.
 110     cell = custom_cell.Alien(config.hidden_size)
 111 
 112     if is_training:
 113       [h2h_masks, h2i_masks, _,
 114        output_mask] = variational_dropout.generate_variational_dropout_masks(
 115            hparams, config.keep_prob)
 116     else:
 117       h2i_masks, output_mask = None, None
 118 
 119     cell = custom_cell.GenericMultiRNNCell([cell] * config.num_layers)
 120 
 121     initial_state = cell.zero_state(FLAGS.batch_size, tf.float32)
 122 
 123     # Add a missing token for inputs not present.
 124     real_inputs = inputs
 125     masked_inputs = transform_input_with_is_missing_token(
 126         inputs, targets_present)
 127 
 128     with tf.variable_scope('rnn'):
 129       hidden_states = []
 130 
 131       # Split the embedding into two parts so that we can load the PTB
 132       # weights into one part of the Variable.
 133       if not FLAGS.seq2seq_share_embedding:
 134         embedding = tf.get_variable('embedding',
 135                                     [FLAGS.vocab_size, hparams.gen_rnn_size])
 136       missing_embedding = tf.get_variable('missing_embedding',
 137                                           [1, hparams.gen_rnn_size])
 138       embedding = tf.concat([embedding, missing_embedding], axis=0)
 139 
 140       real_rnn_inputs = tf.nn.embedding_lookup(embedding, real_inputs)
 141       masked_rnn_inputs = tf.nn.embedding_lookup(embedding, masked_inputs)
 142 
 143       if is_training and FLAGS.keep_prob < 1:
 144         masked_rnn_inputs = tf.nn.dropout(masked_rnn_inputs, FLAGS.keep_prob)
 145 
 146       state = initial_state
 147       for t in xrange(FLAGS.sequence_length):
 148         if t > 0:
 149           tf.get_variable_scope().reuse_variables()
 150 
 151         rnn_inp = masked_rnn_inputs[:, t]
 152 
 153         if is_training:
 154           state = list(state)
 155           for layer_num, per_layer_state in enumerate(state):
 156             per_layer_state = LSTMTuple(
 157                 per_layer_state[0], per_layer_state[1] * h2h_masks[layer_num])
 158             state[layer_num] = per_layer_state
 159 
 160         rnn_out, state = cell(rnn_inp, state, h2i_masks)
 161 
 162         if is_training:
 163           rnn_out = output_mask * rnn_out
 164 
 165         hidden_states.append(rnn_out)
 166       final_masked_state = state
 167       hidden_states = tf.stack(hidden_states, axis=1)
 168 
 169       # Produce the RNN state had the model operated only
 170       # over real data.
 171       real_state = initial_state
 172       for t in xrange(FLAGS.sequence_length):
 173         tf.get_variable_scope().reuse_variables()
 174 
 175         # RNN.
 176         rnn_inp = real_rnn_inputs[:, t]
 177         rnn_out, real_state = cell(rnn_inp, real_state)
 178       final_state = real_state
 179 
 180   return (hidden_states, final_masked_state), initial_state, final_state

 183 def gen_decoder(hparams,
 184                 inputs,
 185                 targets,
 186                 targets_present,
 187                 encoding_state,
 188                 is_training,
 189                 is_validating,
 190                 reuse=None):
 191   """Define the Decoder graph. The Decoder will now impute tokens that
 192       have been masked from the input seqeunce.
 193   """
 194   config = get_config()
 195   gen_decoder_rnn_size = hparams.gen_rnn_size
 196 
 197   if FLAGS.seq2seq_share_embedding:
 198     with tf.variable_scope('decoder/rnn', reuse=True):
 199       embedding = tf.get_variable('embedding',
 200                                   [FLAGS.vocab_size, gen_decoder_rnn_size])
 201 
 202   with tf.variable_scope('decoder', reuse=reuse):
 203     # Neural architecture search cell.
 204     cell = custom_cell.Alien(config.hidden_size)
 205 
 206     if is_training:
 207       [h2h_masks, _, _,
 208        output_mask] = variational_dropout.generate_variational_dropout_masks(
 209            hparams, config.keep_prob)
 210     else:
 211       output_mask = None
 212 
 213     cell_gen = custom_cell.GenericMultiRNNCell([cell] * config.num_layers)
 214 
 215     # Hidden encoder states.
 216     hidden_vector_encodings = encoding_state[0]
 217 
 218     # Carry forward the final state tuple from the encoder.
 219     # State tuples.
 220     state_gen = encoding_state[1]
 221 
 222     if FLAGS.attention_option is not None:
 223       (attention_keys, attention_values, _,
 224        attention_construct_fn) = attention_utils.prepare_attention(
 225            hidden_vector_encodings,
 226            FLAGS.attention_option,
 227            num_units=gen_decoder_rnn_size,
 228            reuse=reuse)
 229 
 230     with tf.variable_scope('rnn'):
 231       sequence, logits, log_probs = [], [], []
 232 
 233       if not FLAGS.seq2seq_share_embedding:
 234         embedding = tf.get_variable('embedding',
 235                                     [FLAGS.vocab_size, gen_decoder_rnn_size])
 236       softmax_w = tf.matrix_transpose(embedding)
 237       softmax_b = tf.get_variable('softmax_b', [FLAGS.vocab_size])
 238 
 239       rnn_inputs = tf.nn.embedding_lookup(embedding, inputs)
 240 
 241       if is_training and FLAGS.keep_prob < 1:
 242         rnn_inputs = tf.nn.dropout(rnn_inputs, FLAGS.keep_prob)
 243 
 244       for t in xrange(FLAGS.sequence_length):
 245         if t > 0:
 246           tf.get_variable_scope().reuse_variables()
 247 
 248         # Input to the Decoder.
 249         if t == 0:
 250           # Always provide the real input at t = 0.
 251           rnn_inp = rnn_inputs[:, t]
 252 
 253         # If the input is present, read in the input at t.
 254         # If the input is not present, read in the previously generated.
 255         else:
 256           real_rnn_inp = rnn_inputs[:, t]
 257           fake_rnn_inp = tf.nn.embedding_lookup(embedding, fake)
 258 
 259           # While validating, the decoder should be operating in teacher
 260           # forcing regime.  Also, if we're just training with cross_entropy
 261           # use teacher forcing.
 262           if is_validating or (is_training and
 263                                FLAGS.gen_training_strategy == 'cross_entropy'):
 264             rnn_inp = real_rnn_inp
 265           else:
 266             rnn_inp = tf.where(targets_present[:, t - 1], real_rnn_inp,
 267                                fake_rnn_inp)
 268 
 269         if is_training:
 270           state_gen = list(state_gen)
 271           for layer_num, per_layer_state in enumerate(state_gen):
 272             per_layer_state = LSTMTuple(
 273                 per_layer_state[0], per_layer_state[1] * h2h_masks[layer_num])
 274             state_gen[layer_num] = per_layer_state
 275 
 276         # RNN.
 277         rnn_out, state_gen = cell_gen(rnn_inp, state_gen)
 278 
 279         if is_training:
 280           rnn_out = output_mask * rnn_out
 281 
 282         if FLAGS.attention_option is not None:
 283           rnn_out = attention_construct_fn(rnn_out, attention_keys,
 284                                            attention_values)
 285         #   # TODO(liamfedus): Assert not "monotonic" attention_type.
 286         #   # TODO(liamfedus): FLAGS.attention_type.
 287         #   context_state = revised_attention_utils._empty_state()
 288         #   rnn_out, context_state = attention_construct_fn(
 289         #       rnn_out, attention_keys, attention_values, context_state, t)
 290         logit = tf.matmul(rnn_out, softmax_w) + softmax_b
 291 
 292         # Output for Decoder.
 293         # If input is present:   Return real at t+1.
 294         # If input is not present:  Return fake for t+1.
 295         real = targets[:, t]
 296 
 297         categorical = tf.contrib.distributions.Categorical(logits=logit)
 298         fake = categorical.sample()
 299         log_prob = categorical.log_prob(fake)
 300 
 301         output = tf.where(targets_present[:, t], real, fake)
 302 
 303         # Add to lists.
 304         sequence.append(output)
 305         log_probs.append(log_prob)
 306         logits.append(logit)
 307 
 308   return (tf.stack(sequence, axis=1), tf.stack(logits, axis=1), tf.stack(
 309       log_probs, axis=1))

 312 def generator(hparams,
 313               inputs,
 314               targets,
 315               targets_present,
 316               is_training,
 317               is_validating,
 318               reuse=None):
 319   """Define the Generator graph."""
 320   with tf.variable_scope('gen', reuse=reuse):
 321     encoder_states, initial_state, final_state = gen_encoder(
 322         hparams, inputs, targets_present, is_training=is_training, reuse=reuse)
 323     stacked_sequence, stacked_logits, stacked_log_probs = gen_decoder(
 324         hparams,
 325         inputs,
 326         targets,
 327         targets_present,
 328         encoder_states,
 329         is_training=is_training,
 330         is_validating=is_validating,
 331         reuse=reuse)
 332     return (stacked_sequence, stacked_logits, stacked_log_probs, initial_state,
 333             final_state)
