  69 def get_iterator(data):
  70   """Return the data iterator."""
  71   if FLAGS.data_set == 'ptb':
  72     iterator = ptb_loader.ptb_iterator(data, FLAGS.batch_size,
  73                                        FLAGS.sequence_length,
  74                                        FLAGS.epoch_size_override)
  75   elif FLAGS.data_set == 'imdb':
  76     iterator = imdb_loader.imdb_iterator(data, FLAGS.batch_size,
  77                                          FLAGS.sequence_length)
  78   return iterator

  81 def convert_to_human_readable(id_to_word, arr, p, max_num_to_print):
  82   """Convert a np.array of indices into words using id_to_word dictionary.
  83   Return max_num_to_print results.
  84   """
  85 
  86   assert arr.ndim == 2
  87 
  88   samples = []
  89   for sequence_id in xrange(min(len(arr), max_num_to_print)):
  90     sample = []
  91     for i, index in enumerate(arr[sequence_id, :]):
  92       if p[sequence_id, i] == 1:
  93         sample.append(str(id_to_word[index]))
  94       else:
  95         sample.append('*' + str(id_to_word[index]))
  96     buffer_str = ' '.join(sample)
  97     samples.append(buffer_str)
  98   return samples

 101 def write_unmasked_log(log, id_to_word, sequence_eval):
 102   """Helper function for logging evaluated sequences without mask."""
 103   indices_arr = np.asarray(sequence_eval)
 104   samples = helper.convert_to_human_readable(id_to_word, indices_arr,
 105                                              FLAGS.batch_size)
 106   for sample in samples:
 107     log.write(sample + '\n')
 108   log.flush()
 109   return samples

 112 def write_masked_log(log, id_to_word, sequence_eval, present_eval):
 113   indices_arr = np.asarray(sequence_eval)
 114   samples = convert_to_human_readable(id_to_word, indices_arr, present_eval,
 115                                       FLAGS.batch_size)
 116   for sample in samples:
 117     log.write(sample + '\n')
 118   log.flush()
 119   return samples

 122 def generate_logs(sess, model, log, id_to_word, feed):
 123   """Impute Sequences using the model for a particular feed and send it to
 124   logs.
 125   """
 126   # Impute Sequences.
 127   [p, inputs_eval, sequence_eval] = sess.run(
 128       [model.present, model.inputs, model.fake_sequence], feed_dict=feed)
 129 
 130   # Add the 0th time-step for coherence.
 131   first_token = np.expand_dims(inputs_eval[:, 0], axis=1)
 132   sequence_eval = np.concatenate((first_token, sequence_eval), axis=1)
 133 
 134   # 0th token always present.
 135   p = np.concatenate((np.ones((FLAGS.batch_size, 1)), p), axis=1)
 136 
 137   if FLAGS.output_masked_logs:
 138     samples = write_masked_log(log, id_to_word, sequence_eval, p)
 139   else:
 140     samples = write_unmasked_log(log, id_to_word, sequence_eval)
 141   return samples

 144 def generate_samples(hparams, data, id_to_word, log_dir, output_file):
 145   """"Generate samples.
 146 
 147     Args:
 148       hparams:  Hyperparameters for the MaskGAN.
 149       data: Data to evaluate.
 150       id_to_word: Dictionary of indices to words.
 151       log_dir: Log directory.
 152       output_file:  Output file for the samples.
 153   """
 154   # Boolean indicating operational mode.
 155   is_training = False
 156 
 157   # Set a random seed to keep fixed mask.
 158   np.random.seed(0)
 159 
 160   with tf.Graph().as_default():
 161     # Construct the model.
 162     model = train_mask_gan.create_MaskGAN(hparams, is_training)
 163 
 164     ## Retrieve the initial savers.
 165     init_savers = model_utils.retrieve_init_savers(hparams)
 166 
 167     ## Initial saver function to supervisor.
 168     init_fn = partial(model_utils.init_fn, init_savers)
 169 
 170     is_chief = FLAGS.task == 0
 171 
 172     # Create the supervisor.  It will take care of initialization, summaries,
 173     # checkpoints, and recovery.
 174     sv = tf.Supervisor(
 175         logdir=log_dir,
 176         is_chief=is_chief,
 177         saver=model.saver,
 178         global_step=model.global_step,
 179         recovery_wait_secs=30,
 180         summary_op=None,
 181         init_fn=init_fn)
 182 
 183     # Get an initialized, and possibly recovered session.  Launch the
 184     # services: Checkpointing, Summaries, step counting.
 185     #
 186     # When multiple replicas of this program are running the services are
 187     # only launched by the 'chief' replica.
 188     with sv.managed_session(
 189         FLAGS.master, start_standard_services=False) as sess:
 190 
 191       # Generator statefulness over the epoch.
 192       [gen_initial_state_eval, fake_gen_initial_state_eval] = sess.run(
 193           [model.eval_initial_state, model.fake_gen_initial_state])
 194 
 195       for n in xrange(FLAGS.number_epochs):
 196         print('Epoch number: %d' % n)
 197         # print('Percent done: %.2f' % float(n) / float(FLAGS.number_epochs))
 198         iterator = get_iterator(data)
 199         for x, y, _ in iterator:
 200           if FLAGS.eval_language_model:
 201             is_present_rate = 0.
 202           else:
 203             is_present_rate = FLAGS.is_present_rate
 204           tf.logging.info(
 205               'Evaluating on is_present_rate=%.3f.' % is_present_rate)
 206 
 207           model_utils.assign_percent_real(sess, model.percent_real_update,
 208                                           model.new_rate, is_present_rate)
 209 
 210           # Randomly mask out tokens.
 211           p = model_utils.generate_mask()
 212 
 213           eval_feed = {model.inputs: x, model.targets: y, model.present: p}
 214 
 215           if FLAGS.data_set == 'ptb':
 216             # Statefulness for *evaluation* Generator.
 217             for i, (c, h) in enumerate(model.eval_initial_state):
 218               eval_feed[c] = gen_initial_state_eval[i].c
 219               eval_feed[h] = gen_initial_state_eval[i].h
 220 
 221             # Statefulness for the Generator.
 222             for i, (c, h) in enumerate(model.fake_gen_initial_state):
 223               eval_feed[c] = fake_gen_initial_state_eval[i].c
 224               eval_feed[h] = fake_gen_initial_state_eval[i].h
 225 
 226           [gen_initial_state_eval, fake_gen_initial_state_eval, _] = sess.run(
 227               [
 228                   model.eval_final_state, model.fake_gen_final_state,
 229                   model.global_step
 230               ],
 231               feed_dict=eval_feed)
 232 
 233           generate_logs(sess, model, output_file, id_to_word, eval_feed)
 234       output_file.close()
 235       print('Closing output_file.')
 236       return

 239 def main(_):
 240   hparams = train_mask_gan.create_hparams()
 241   log_dir = FLAGS.base_directory
 242 
 243   tf.gfile.MakeDirs(FLAGS.output_path)
 244   output_file = tf.gfile.GFile(
 245       os.path.join(FLAGS.output_path, 'reviews.txt'), mode='w')
 246 
 247   # Load data set.
 248   if FLAGS.data_set == 'ptb':
 249     raw_data = ptb_loader.ptb_raw_data(FLAGS.data_dir)
 250     train_data, valid_data, _, _ = raw_data
 251   elif FLAGS.data_set == 'imdb':
 252     raw_data = imdb_loader.imdb_raw_data(FLAGS.data_dir)
 253     train_data, valid_data = raw_data
 254   else:
 255     raise NotImplementedError
 256 
 257   # Generating more data on train set.
 258   if FLAGS.sample_mode == SAMPLE_TRAIN:
 259     data_set = train_data
 260   elif FLAGS.sample_mode == SAMPLE_VALIDATION:
 261     data_set = valid_data
 262   else:
 263     raise NotImplementedError
 264 
 265   # Dictionary and reverse dictionry.
 266   if FLAGS.data_set == 'ptb':
 267     word_to_id = ptb_loader.build_vocab(
 268         os.path.join(FLAGS.data_dir, 'ptb.train.txt'))
 269   elif FLAGS.data_set == 'imdb':
 270     word_to_id = imdb_loader.build_vocab(
 271         os.path.join(FLAGS.data_dir, 'vocab.txt'))
 272   id_to_word = {v: k for k, v in word_to_id.iteritems()}
 273 
 274   FLAGS.vocab_size = len(id_to_word)
 275   print('Vocab size: %d' % FLAGS.vocab_size)
 276 
 277   generate_samples(hparams, data_set, id_to_word, log_dir, output_file)
 278 
 279 
 280 if __name__ == '__main__':
 281   tf.app.run()
