  29 def cell_depth(num):
  30   num /= 2
  31   val = np.log2(1 + num)
  32   assert abs(val - int(val)) == 0
  33   return int(val)

  36 class GenericMultiRNNCell(tf.contrib.rnn.RNNCell):
  37   """More generic version of MultiRNNCell that allows you to pass in a dropout mask"""
  38 
  39   def __init__(self, cells):
  40     """Create a RNN cell composed sequentially of a number of RNNCells.
  41 
  42     Args:
  43       cells: list of RNNCells that will be composed in this order.
  44       state_is_tuple: If True, accepted and returned states are n-tuples, where
  45         `n = len(cells)`.  If False, the states are all
  46         concatenated along the column axis.  This latter behavior will soon be
  47         deprecated.
  48 
  49     Raises:
  50       ValueError: if cells is empty (not allowed), or at least one of the cells
  51         returns a state tuple but the flag `state_is_tuple` is `False`.
  52     """
  53     self._cells = cells
  54 
  55   @property
  56   def state_size(self):
  57     return tuple(cell.state_size for cell in self._cells)
  58 
  59   @property
  60   def output_size(self):
  61     return self._cells[-1].output_size
  62 
  63   def __call__(self, inputs, state, input_masks=None, scope=None):
  64     """Run this multi-layer cell on inputs, starting from state."""
  65     with tf.variable_scope(scope or type(self).__name__):
  66       cur_inp = inputs
  67       new_states = []
  68       for i, cell in enumerate(self._cells):
  69         with tf.variable_scope('Cell%d' % i):
  70           cur_state = state[i]
  71           if input_masks is not None:
  72             cur_inp *= input_masks[i]
  73           cur_inp, new_state = cell(cur_inp, cur_state)
  74           new_states.append(new_state)
  75     new_states = tuple(new_states)
  76     return cur_inp, new_states

  79 class AlienRNNBuilder(tf.contrib.rnn.RNNCell):
  80 
  81   def __init__(self, num_units, params, additional_params, base_size):
  82     self.num_units = num_units
  83     self.cell_create_index = additional_params[0]
  84     self.cell_inject_index = additional_params[1]
  85     self.base_size = base_size
  86     self.cell_params = params[
  87         -2:]  # Cell injection parameters are always the last two
  88     params = params[:-2]
  89     self.depth = cell_depth(len(params))
  90     self.params = params
  91     self.units_per_layer = [2**i for i in range(self.depth)
  92                            ][::-1]  # start with the biggest layer
  93 
  94   def __call__(self, inputs, state, scope=None):
  95     with tf.variable_scope(scope or type(self).__name__):
  96       definition1 = ['add', 'elem_mult', 'max']
  97       definition2 = [tf.identity, tf.tanh, tf.sigmoid, tf.nn.relu, tf.sin]
  98       layer_outputs = [[] for _ in range(self.depth)]
  99       with tf.variable_scope('rnn_builder'):
 100         curr_index = 0
 101         c, h = state
 102 
 103         # Run all dense matrix multiplications at once
 104         big_h_mat = tf.get_variable(
 105             'big_h_mat', [self.num_units,
 106                           self.base_size * self.num_units], tf.float32)
 107         big_inputs_mat = tf.get_variable(
 108             'big_inputs_mat', [self.num_units,
 109                                self.base_size * self.num_units], tf.float32)
 110         big_h_output = tf.matmul(h, big_h_mat)
 111         big_inputs_output = tf.matmul(inputs, big_inputs_mat)
 112         h_splits = tf.split(big_h_output, self.base_size, axis=1)
 113         inputs_splits = tf.split(big_inputs_output, self.base_size, axis=1)
 114 
 115         for layer_num, units in enumerate(self.units_per_layer):
 116           for unit_num in range(units):
 117             with tf.variable_scope(
 118                 'layer_{}_unit_{}'.format(layer_num, unit_num)):
 119               if layer_num == 0:
 120                 prev1_mat = h_splits[unit_num]
 121                 prev2_mat = inputs_splits[unit_num]
 122               else:
 123                 prev1_mat = layer_outputs[layer_num - 1][2 * unit_num]
 124                 prev2_mat = layer_outputs[layer_num - 1][2 * unit_num + 1]
 125               if definition1[self.params[curr_index]] == 'add':
 126                 output = prev1_mat + prev2_mat
 127               elif definition1[self.params[curr_index]] == 'elem_mult':
 128                 output = prev1_mat * prev2_mat
 129               elif definition1[self.params[curr_index]] == 'max':
 130                 output = tf.maximum(prev1_mat, prev2_mat)
 131               if curr_index / 2 == self.cell_create_index:  # Take the new cell before the activation
 132                 new_c = tf.identity(output)
 133               output = definition2[self.params[curr_index + 1]](output)
 134               if curr_index / 2 == self.cell_inject_index:
 135                 if definition1[self.cell_params[0]] == 'add':
 136                   output += c
 137                 elif definition1[self.cell_params[0]] == 'elem_mult':
 138                   output *= c
 139                 elif definition1[self.cell_params[0]] == 'max':
 140                   output = tf.maximum(output, c)
 141                 output = definition2[self.cell_params[1]](output)
 142               layer_outputs[layer_num].append(output)
 143               curr_index += 2
 144         new_h = layer_outputs[-1][-1]
 145         return new_h, LSTMTuple(new_c, new_h)
 146 
 147   @property
 148   def state_size(self):
 149     return LSTMTuple(self.num_units, self.num_units)
 150 
 151   @property
 152   def output_size(self):
 153     return self.num_units
