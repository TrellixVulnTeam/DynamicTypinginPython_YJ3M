  27 def generate_dropout_masks(keep_prob, shape, amount):
  28   masks = []
  29   for _ in range(amount):
  30     dropout_mask = tf.random_uniform(shape) + (keep_prob)
  31     dropout_mask = tf.floor(dropout_mask) / (keep_prob)
  32     masks.append(dropout_mask)
  33   return masks

  36 def generate_variational_dropout_masks(hparams, keep_prob):
  37   [batch_size, num_steps, size, num_layers] = [
  38       FLAGS.batch_size, FLAGS.sequence_length, hparams.gen_rnn_size,
  39       hparams.gen_num_layers
  40   ]
  41   if len(keep_prob) == 2:
  42     emb_keep_prob = keep_prob[0]  # keep prob for embedding matrix
  43     h2h_keep_prob = emb_keep_prob  # keep prob for hidden to hidden connections
  44     h2i_keep_prob = keep_prob[1]  # keep prob for hidden to input connections
  45     out_keep_prob = h2i_keep_prob  # keep probability for output state
  46   else:
  47     emb_keep_prob = keep_prob[0]  # keep prob for embedding matrix
  48     h2h_keep_prob = keep_prob[1]  # keep prob for hidden to hidden connections
  49     h2i_keep_prob = keep_prob[2]  # keep prob for hidden to input connections
  50     out_keep_prob = keep_prob[3]  # keep probability for output state
  51   h2i_masks = []  # Masks for input to recurrent connections
  52   h2h_masks = []  # Masks for recurrent to recurrent connections
  53 
  54   # Input word dropout mask
  55   emb_masks = generate_dropout_masks(emb_keep_prob, [num_steps, 1], batch_size)
  56   output_mask = generate_dropout_masks(out_keep_prob, [batch_size, size], 1)[0]
  57   h2i_masks = generate_dropout_masks(h2i_keep_prob, [batch_size, size],
  58                                      num_layers)
  59   h2h_masks = generate_dropout_masks(h2h_keep_prob, [batch_size, size],
  60                                      num_layers)
  61   return h2h_masks, h2i_masks, emb_masks, output_mask
