  51 INFERENCE_MODE_SINGLE = 'single'  # Take plain single-frame input.
  52 INFERENCE_MODE_TRIPLETS = 'triplets'  # Take image triplets as input.

  54 # Cityscapes, the car hood and more image content has been cropped in order
  55 # to fit aspect ratio, and remove static content from the images. This has to be
  56 # kept at inference time.

  58 INFERENCE_CROP_CITYSCAPES = 'cityscapes'

 128 def _run_inference(output_dir=None,
 129                    file_extension='png',
 130                    depth=True,
 131                    egomotion=False,
 132                    model_ckpt=None,
 133                    input_dir=None,
 134                    input_list_file=None,
 135                    batch_size=1,
 136                    img_height=128,
 137                    img_width=416,
 138                    seq_length=3,
 139                    architecture=nets.RESNET,
 140                    imagenet_norm=True,
 141                    use_skip=True,
 142                    joint_encoder=True,
 143                    shuffle=False,
 144                    flip_for_depth=False,
 145                    inference_mode=INFERENCE_MODE_SINGLE,
 146                    inference_crop=INFERENCE_CROP_NONE,
 147                    use_masks=False):
 148   """Runs inference. Refer to flags in inference.py for details."""
 149   inference_model = model.Model(is_training=False,
 150                                 batch_size=batch_size,
 151                                 img_height=img_height,
 152                                 img_width=img_width,
 153                                 seq_length=seq_length,
 154                                 architecture=architecture,
 155                                 imagenet_norm=imagenet_norm,
 156                                 use_skip=use_skip,
 157                                 joint_encoder=joint_encoder)
 158   vars_to_restore = util.get_vars_to_save_and_restore(model_ckpt)
 159   saver = tf.train.Saver(vars_to_restore)
 160   sv = tf.train.Supervisor(logdir='/tmp/', saver=None)
 161   with sv.managed_session() as sess:
 162     saver.restore(sess, model_ckpt)
 163     if not gfile.Exists(output_dir):
 164       gfile.MakeDirs(output_dir)
 165     logging.info('Predictions will be saved in %s.', output_dir)
 166 
 167     # Collect all images to run inference on.
 168     im_files, basepath_in = collect_input_images(input_dir, input_list_file,
 169                                                  file_extension)
 170     if shuffle:
 171       logging.info('Shuffling data...')
 172       np.random.shuffle(im_files)
 173     logging.info('Running inference on %d files.', len(im_files))
 174 
 175     # Create missing output folders and pre-compute target directories.
 176     output_dirs = create_output_dirs(im_files, basepath_in, output_dir)
 177 
 178     # Run depth prediction network.
 179     if depth:
 180       im_batch = []
 181       for i in range(len(im_files)):
 182         if i % 100 == 0:
 183           logging.info('%s of %s files processed.', i, len(im_files))
 184 
 185         # Read image and run inference.
 186         if inference_mode == INFERENCE_MODE_SINGLE:
 187           if inference_crop == INFERENCE_CROP_NONE:
 188             im = util.load_image(im_files[i], resize=(img_width, img_height))
 189           elif inference_crop == INFERENCE_CROP_CITYSCAPES:
 190             im = util.crop_cityscapes(util.load_image(im_files[i]),
 191                                       resize=(img_width, img_height))
 192         elif inference_mode == INFERENCE_MODE_TRIPLETS:
 193           im = util.load_image(im_files[i], resize=(img_width * 3, img_height))
 194           im = im[:, img_width:img_width*2]
 195         if flip_for_depth:
 196           im = np.flip(im, axis=1)
 197         im_batch.append(im)
 198 
 199         if len(im_batch) == batch_size or i == len(im_files) - 1:
 200           # Call inference on batch.
 201           for _ in range(batch_size - len(im_batch)):  # Fill up batch.
 202             im_batch.append(np.zeros(shape=(img_height, img_width, 3),
 203                                      dtype=np.float32))
 204           im_batch = np.stack(im_batch, axis=0)
 205           est_depth = inference_model.inference_depth(im_batch, sess)
 206           if flip_for_depth:
 207             est_depth = np.flip(est_depth, axis=2)
 208             im_batch = np.flip(im_batch, axis=2)
 209 
 210           for j in range(len(im_batch)):
 211             color_map = util.normalize_depth_for_display(
 212                 np.squeeze(est_depth[j]))
 213             visualization = np.concatenate((im_batch[j], color_map), axis=0)
 214             # Save raw prediction and color visualization. Extract filename
 215             # without extension from full path: e.g. path/to/input_dir/folder1/
 216             # file1.png -> file1
 217             k = i - len(im_batch) + 1 + j
 218             filename_root = os.path.splitext(os.path.basename(im_files[k]))[0]
 219             pref = '_flip' if flip_for_depth else ''
 220             output_raw = os.path.join(
 221                 output_dirs[k], filename_root + pref + '.npy')
 222             output_vis = os.path.join(
 223                 output_dirs[k], filename_root + pref + '.png')
 224             with gfile.Open(output_raw, 'wb') as f:
 225               np.save(f, est_depth[j])
 226             util.save_image(output_vis, visualization, file_extension)
 227           im_batch = []
 228 
 229     # Run egomotion network.
 230     if egomotion:
 231       if inference_mode == INFERENCE_MODE_SINGLE:
 232         # Run regular egomotion inference loop.
 233         input_image_seq = []
 234         input_seg_seq = []
 235         current_sequence_dir = None
 236         current_output_handle = None
 237         for i in range(len(im_files)):
 238           sequence_dir = os.path.dirname(im_files[i])
 239           if sequence_dir != current_sequence_dir:
 240             # Assume start of a new sequence, since this image lies in a
 241             # different directory than the previous ones.
 242             # Clear egomotion input buffer.
 243             output_filepath = os.path.join(output_dirs[i], 'egomotion.txt')
 244             if current_output_handle is not None:
 245               current_output_handle.close()
 246             current_sequence_dir = sequence_dir
 247             logging.info('Writing egomotion sequence to %s.', output_filepath)
 248             current_output_handle = gfile.Open(output_filepath, 'w')
 249             input_image_seq = []
 250           im = util.load_image(im_files[i], resize=(img_width, img_height))
 251           input_image_seq.append(im)
 252           if use_masks:
 253             im_seg_path = im_files[i].replace('.%s' % file_extension,
 254                                               '-seg.%s' % file_extension)
 255             if not gfile.Exists(im_seg_path):
 256               raise ValueError('No segmentation mask %s has been found for '
 257                                'image %s. If none are available, disable '
 258                                'use_masks.' % (im_seg_path, im_files[i]))
 259             input_seg_seq.append(util.load_image(im_seg_path,
 260                                                  resize=(img_width, img_height),
 261                                                  interpolation='nn'))
 262 
 263           if len(input_image_seq) < seq_length:  # Buffer not filled yet.
 264             continue
 265           if len(input_image_seq) > seq_length:  # Remove oldest entry.
 266             del input_image_seq[0]
 267             if use_masks:
 268               del input_seg_seq[0]
 269 
 270           input_image_stack = np.concatenate(input_image_seq, axis=2)
 271           input_image_stack = np.expand_dims(input_image_stack, axis=0)
 272           if use_masks:
 273             input_image_stack = mask_image_stack(input_image_stack,
 274                                                  input_seg_seq)
 275           est_egomotion = np.squeeze(inference_model.inference_egomotion(
 276               input_image_stack, sess))
 277           egomotion_str = []
 278           for j in range(seq_length - 1):
 279             egomotion_str.append(','.join([str(d) for d in est_egomotion[j]]))
 280           current_output_handle.write(
 281               str(i) + ' ' + ' '.join(egomotion_str) + '\n')
 282         if current_output_handle is not None:
 283           current_output_handle.close()
 284       elif inference_mode == INFERENCE_MODE_TRIPLETS:
 285         written_before = []
 286         for i in range(len(im_files)):
 287           im = util.load_image(im_files[i], resize=(img_width * 3, img_height))
 288           input_image_stack = np.concatenate(
 289               [im[:, :img_width], im[:, img_width:img_width*2],
 290                im[:, img_width*2:]], axis=2)
 291           input_image_stack = np.expand_dims(input_image_stack, axis=0)
 292           if use_masks:
 293             im_seg_path = im_files[i].replace('.%s' % file_extension,
 294                                               '-seg.%s' % file_extension)
 295             if not gfile.Exists(im_seg_path):
 296               raise ValueError('No segmentation mask %s has been found for '
 297                                'image %s. If none are available, disable '
 298                                'use_masks.' % (im_seg_path, im_files[i]))
 299             seg = util.load_image(im_seg_path,
 300                                   resize=(img_width * 3, img_height),
 301                                   interpolation='nn')
 302             input_seg_seq = [seg[:, :img_width], seg[:, img_width:img_width*2],
 303                              seg[:, img_width*2:]]
 304             input_image_stack = mask_image_stack(input_image_stack,
 305                                                  input_seg_seq)
 306           est_egomotion = inference_model.inference_egomotion(
 307               input_image_stack, sess)
 308           est_egomotion = np.squeeze(est_egomotion)
 309           egomotion_1_2 = ','.join([str(d) for d in est_egomotion[0]])
 310           egomotion_2_3 = ','.join([str(d) for d in est_egomotion[1]])
 311 
 312           output_filepath = os.path.join(output_dirs[i], 'egomotion.txt')
 313           file_mode = 'w' if output_filepath not in written_before else 'a'
 314           with gfile.Open(output_filepath, file_mode) as current_output_handle:
 315             current_output_handle.write(str(i) + ' ' + egomotion_1_2 + ' ' +
 316                                         egomotion_2_3 + '\n')
 317           written_before.append(output_filepath)
 318       logging.info('Done.')

 321 def mask_image_stack(input_image_stack, input_seg_seq):
 322   """Masks out moving image contents by using the segmentation masks provided.
 323 
 324   This can lead to better odometry accuracy for motion models, but is optional
 325   to use. Is only called if use_masks is enabled.
 326   Args:
 327     input_image_stack: The input image stack of shape (1, H, W, seq_length).
 328     input_seg_seq: List of segmentation masks with seq_length elements of shape
 329                    (H, W, C) for some number of channels C.
 330 
 331   Returns:
 332     Input image stack with detections provided by segmentation mask removed.
 333   """
 334   background = [mask == 0 for mask in input_seg_seq]
 335   background = reduce(lambda m1, m2: m1 & m2, background)
 336   # If masks are RGB, assume all channels to be the same. Reduce to the first.
 337   if background.ndim == 3 and background.shape[2] > 1:
 338     background = np.expand_dims(background[:, :, 0], axis=2)
 339   elif background.ndim == 2:  # Expand.
 340     background = np.expand_dism(background, axis=2)
 341   # background is now of shape (H, W, 1).
 342   background_stack = np.tile(background, [1, 1, input_image_stack.shape[3]])
 343   return np.multiply(input_image_stack, background_stack)

 346 def collect_input_images(input_dir, input_list_file, file_extension):
 347   """Collects all input images that are to be processed."""
 348   if input_dir is not None:
 349     im_files = _recursive_glob(input_dir, '*.' + file_extension)
 350     basepath_in = os.path.normpath(input_dir)
 351   elif input_list_file is not None:
 352     im_files = util.read_text_lines(input_list_file)
 353     basepath_in = os.path.dirname(input_list_file)
 354     im_files = [os.path.join(basepath_in, f) for f in im_files]
 355   im_files = [f for f in im_files if 'disp' not in f and '-seg' not in f and
 356               '-fseg' not in f and '-flip' not in f]
 357   return sorted(im_files), basepath_in

 360 def create_output_dirs(im_files, basepath_in, output_dir):
 361   """Creates required directories, and returns output dir for each file."""
 362   output_dirs = []
 363   for i in range(len(im_files)):
 364     relative_folder_in = os.path.relpath(
 365         os.path.dirname(im_files[i]), basepath_in)
 366     absolute_folder_out = os.path.join(output_dir, relative_folder_in)
 367     if not gfile.IsDirectory(absolute_folder_out):
 368       gfile.MakeDirs(absolute_folder_out)
 369     output_dirs.append(absolute_folder_out)
 370   return output_dirs

 373 def _recursive_glob(treeroot, pattern):
 374   results = []
 375   for base, _, files in os.walk(treeroot):
 376     files = fnmatch.filter(files, pattern)
 377     results.extend(os.path.join(base, f) for f in files)
 378   return results
