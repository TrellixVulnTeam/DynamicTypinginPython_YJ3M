  36 CMAP_DEFAULT = 'plasma'
  37 # Defines the cropping that is applied to the Cityscapes dataset with respect to
  38 # the original raw input resolution.
  39 CITYSCAPES_CROP = [256, 768, 192, 1856]

  42 def crop_cityscapes(im, resize=None):
  43   ymin, ymax, xmin, xmax = CITYSCAPES_CROP
  44   im = im[ymin:ymax, xmin:xmax]
  45   if resize is not None:
  46     im = cv2.resize(im, resize)
  47   return im

  50 def gray2rgb(im, cmap=CMAP_DEFAULT):
  51   cmap = plt.get_cmap(cmap)
  52   result_img = cmap(im.astype(np.float32))
  53   if result_img.shape[2] > 3:
  54     result_img = np.delete(result_img, 3, 2)
  55   return result_img

  58 def load_image(img_file, resize=None, interpolation='linear'):
  59   """Load image from disk. Output value range: [0,1]."""
  60   im_data = np.fromstring(gfile.Open(img_file).read(), np.uint8)
  61   im = cv2.imdecode(im_data, cv2.IMREAD_COLOR)
  62   im = cv2.cvtColor(im, cv2.COLOR_BGR2RGB)
  63   if resize and resize != im.shape[:2]:
  64     ip = cv2.INTER_LINEAR if interpolation == 'linear' else cv2.INTER_NEAREST
  65     im = cv2.resize(im, resize, interpolation=ip)
  66   return np.array(im, dtype=np.float32) / 255.0

  69 def save_image(img_file, im, file_extension):
  70   """Save image from disk. Expected input value range: [0,1]."""
  71   im = (im * 255.0).astype(np.uint8)
  72   with gfile.Open(img_file, 'w') as f:
  73     im = cv2.cvtColor(im, cv2.COLOR_RGB2BGR)
  74     _, im_data = cv2.imencode('.%s' % file_extension, im)
  75     f.write(im_data.tostring())

  78 def normalize_depth_for_display(depth, pc=95, crop_percent=0, normalizer=None,
  79                                 cmap=CMAP_DEFAULT):
  80   """Converts a depth map to an RGB image."""
  81   # Convert to disparity.
  82 
  83   disp = 1.0 / (depth + 1e-6)
  84   if normalizer is not None:
  85     disp /= normalizer
  86   else:
  87     disp /= (np.percentile(disp, pc) + 1e-6)
  88   disp = np.clip(disp, 0, 1)
  89   disp = gray2rgb(disp, cmap=cmap)
  90   keep_h = int(disp.shape[0] * (1 - crop_percent))
  91   disp = disp[:keep_h]
  92   return disp

  95 def get_seq_start_end(target_index, seq_length, sample_every=1):
  96   """Returns absolute seq start and end indices for a given target frame."""
  97   half_offset = int((seq_length - 1) / 2) * sample_every
  98   end_index = target_index + half_offset
  99   start_index = end_index - (seq_length - 1) * sample_every
 100   return start_index, end_index

 103 def get_seq_middle(seq_length):
 104   """Returns relative index for the middle frame in sequence."""
 105   half_offset = int((seq_length - 1) / 2)
 106   return seq_length - 1 - half_offset

 109 def info(obj):
 110   """Return info on shape and dtype of a numpy array or TensorFlow tensor."""
 111   if obj is None:
 112     return 'None.'
 113   elif isinstance(obj, list):
 114     if obj:
 115       return 'List of %d... %s' % (len(obj), info(obj[0]))
 116     else:
 117       return 'Empty list.'
 118   elif isinstance(obj, tuple):
 119     if obj:
 120       return 'Tuple of %d... %s' % (len(obj), info(obj[0]))
 121     else:
 122       return 'Empty tuple.'
 123   else:
 124     if is_a_numpy_array(obj):
 125       return 'Array with shape: %s, dtype: %s' % (obj.shape, obj.dtype)
 126     else:
 127       return str(obj)

 135 def count_parameters(also_print=True):
 136   """Cound the number of parameters in the model.
 137 
 138   Args:
 139     also_print: Boolean.  If True also print the numbers.
 140 
 141   Returns:
 142     The total number of parameters.
 143   """
 144   total = 0
 145   if also_print:
 146     logging.info('Model Parameters:')
 147   for (_, v) in get_vars_to_save_and_restore().items():
 148     shape = v.get_shape()
 149     if also_print:
 150       logging.info('%s %s: %s', v.op.name, shape,
 151                    format_number(shape.num_elements()))
 152     total += shape.num_elements()
 153   if also_print:
 154     logging.info('Total: %s', format_number(total))
 155   return total

 158 def get_vars_to_save_and_restore(ckpt=None):
 159   """Returns list of variables that should be saved/restored.
 160 
 161   Args:
 162     ckpt: Path to existing checkpoint.  If present, returns only the subset of
 163         variables that exist in given checkpoint.
 164 
 165   Returns:
 166     List of all variables that need to be saved/restored.
 167   """
 168   model_vars = tf.trainable_variables()
 169   # Add batchnorm variables.
 170   bn_vars = [v for v in tf.global_variables()
 171              if 'moving_mean' in v.op.name or 'moving_variance' in v.op.name or
 172              'mu' in v.op.name or 'sigma' in v.op.name or
 173              'global_scale_var' in v.op.name]
 174   model_vars.extend(bn_vars)
 175   model_vars = sorted(model_vars, key=lambda x: x.op.name)
 176   mapping = {}
 177   if ckpt is not None:
 178     ckpt_var = tf.contrib.framework.list_variables(ckpt)
 179     ckpt_var_names = [name for (name, unused_shape) in ckpt_var]
 180     ckpt_var_shapes = [shape for (unused_name, shape) in ckpt_var]
 181     not_loaded = list(ckpt_var_names)
 182     for v in model_vars:
 183       if v.op.name not in ckpt_var_names:
 184         # For backward compatibility, try additional matching.
 185         v_additional_name = v.op.name.replace('egomotion_prediction/', '')
 186         if v_additional_name in ckpt_var_names:
 187           # Check if shapes match.
 188           ind = ckpt_var_names.index(v_additional_name)
 189           if ckpt_var_shapes[ind] == v.get_shape():
 190             mapping[v_additional_name] = v
 191             not_loaded.remove(v_additional_name)
 192             continue
 193           else:
 194             logging.warn('Shape mismatch, will not restore %s.', v.op.name)
 195         logging.warn('Did not find var %s in checkpoint: %s', v.op.name,
 196                      os.path.basename(ckpt))
 197       else:
 198         # Check if shapes match.
 199         ind = ckpt_var_names.index(v.op.name)
 200         if ckpt_var_shapes[ind] == v.get_shape():
 201           mapping[v.op.name] = v
 202           not_loaded.remove(v.op.name)
 203         else:
 204           logging.warn('Shape mismatch, will not restore %s.', v.op.name)
 205     if not_loaded:
 206       logging.warn('The following variables in the checkpoint were not loaded:')
 207       for varname_not_loaded in not_loaded:
 208         logging.info('%s', varname_not_loaded)
 209   else:  # just get model vars.
 210     for v in model_vars:
 211       mapping[v.op.name] = v
 212   return mapping

 215 def get_imagenet_vars_to_restore(imagenet_ckpt):
 216   """Returns dict of variables to restore from ImageNet-checkpoint."""
 217   vars_to_restore_imagenet = {}
 218   ckpt_var_names = tf.contrib.framework.list_variables(imagenet_ckpt)
 219   ckpt_var_names = [name for (name, unused_shape) in ckpt_var_names]
 220   model_vars = tf.global_variables()
 221   for v in model_vars:
 222     if 'global_step' in v.op.name: continue
 223     mvname_noprefix = v.op.name.replace('depth_prediction/', '')
 224     mvname_noprefix = mvname_noprefix.replace('moving_mean', 'mu')
 225     mvname_noprefix = mvname_noprefix.replace('moving_variance', 'sigma')
 226     if mvname_noprefix in ckpt_var_names:
 227       vars_to_restore_imagenet[mvname_noprefix] = v
 228     else:
 229       logging.info('The following variable will not be restored from '
 230                    'pretrained ImageNet-checkpoint: %s', mvname_noprefix)
 231   return vars_to_restore_imagenet

 234 def format_number(n):
 235   """Formats number with thousands commas."""
 236   locale.setlocale(locale.LC_ALL, 'en_US')
 237   return locale.format('%d', n, grouping=True)

 248 def read_text_lines(filepath):
 249   with tf.gfile.Open(filepath, 'r') as f:
 250     lines = f.readlines()
 251   lines = [l.rstrip() for l in lines]
 252   return lines
