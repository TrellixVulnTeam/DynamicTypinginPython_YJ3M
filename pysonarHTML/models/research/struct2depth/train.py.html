 115 def main(_):
 116   # Fixed seed for repeatability
 117   seed = 8964
 118   tf.set_random_seed(seed)
 119   np.random.seed(seed)
 120   random.seed(seed)
 121 
 122   if FLAGS.handle_motion and FLAGS.joint_encoder:
 123     raise ValueError('Using a joint encoder is currently not supported when '
 124                      'modeling object motion.')
 125   if FLAGS.handle_motion and FLAGS.seq_length != 3:
 126     raise ValueError('The current motion model implementation only supports '
 127                      'using a sequence length of three.')
 128   if FLAGS.handle_motion and not FLAGS.compute_minimum_loss:
 129     raise ValueError('Computing the minimum photometric loss is required when '
 130                      'enabling object motion handling.')
 131   if FLAGS.size_constraint_weight > 0 and not FLAGS.handle_motion:
 132     raise ValueError('To enforce object size constraints, enable motion '
 133                      'handling.')
 134   if FLAGS.imagenet_ckpt and not FLAGS.imagenet_norm:
 135     logging.warn('When initializing with an ImageNet-pretrained model, it is '
 136                  'recommended to normalize the image inputs accordingly using '
 137                  'imagenet_norm.')
 138   if FLAGS.compute_minimum_loss and FLAGS.seq_length % 2 != 1:
 139     raise ValueError('Compute minimum loss requires using an odd number of '
 140                      'images in a sequence.')
 141   if FLAGS.architecture != nets.RESNET and FLAGS.imagenet_ckpt:
 142     raise ValueError('Can only load weights from pre-trained ImageNet model '
 143                      'when using ResNet-architecture.')
 144   if FLAGS.compute_minimum_loss and FLAGS.exhaustive_mode:
 145     raise ValueError('Exhaustive mode has no effect when compute_minimum_loss '
 146                      'is enabled.')
 147   if FLAGS.img_width % (2 ** 5) != 0 or FLAGS.img_height % (2 ** 5) != 0:
 148     logging.warn('Image size is not divisible by 2^5. For the architecture '
 149                  'employed, this could cause artefacts caused by resizing in '
 150                  'lower dimensions.')
 151   if FLAGS.icp_weight > 0.0:
 152     # TODO(casser): Change ICP interface to take matrix instead of vector.
 153     raise ValueError('ICP is currently not supported.')
 154 
 155   if not gfile.Exists(FLAGS.checkpoint_dir):
 156     gfile.MakeDirs(FLAGS.checkpoint_dir)
 157 
 158   train_model = model.Model(data_dir=FLAGS.data_dir,
 159                             file_extension=FLAGS.file_extension,
 160                             is_training=True,
 161                             learning_rate=FLAGS.learning_rate,
 162                             beta1=FLAGS.beta1,
 163                             reconstr_weight=FLAGS.reconstr_weight,
 164                             smooth_weight=FLAGS.smooth_weight,
 165                             ssim_weight=FLAGS.ssim_weight,
 166                             icp_weight=FLAGS.icp_weight,
 167                             batch_size=FLAGS.batch_size,
 168                             img_height=FLAGS.img_height,
 169                             img_width=FLAGS.img_width,
 170                             seq_length=FLAGS.seq_length,
 171                             architecture=FLAGS.architecture,
 172                             imagenet_norm=FLAGS.imagenet_norm,
 173                             weight_reg=FLAGS.weight_reg,
 174                             exhaustive_mode=FLAGS.exhaustive_mode,
 175                             random_scale_crop=FLAGS.random_scale_crop,
 176                             flipping_mode=FLAGS.flipping_mode,
 177                             depth_upsampling=FLAGS.depth_upsampling,
 178                             depth_normalization=FLAGS.depth_normalization,
 179                             compute_minimum_loss=FLAGS.compute_minimum_loss,
 180                             use_skip=FLAGS.use_skip,
 181                             joint_encoder=FLAGS.joint_encoder,
 182                             handle_motion=FLAGS.handle_motion,
 183                             equal_weighting=FLAGS.equal_weighting,
 184                             size_constraint_weight=FLAGS.size_constraint_weight)
 185 
 186   train(train_model, FLAGS.pretrained_ckpt, FLAGS.imagenet_ckpt,
 187         FLAGS.checkpoint_dir, FLAGS.train_steps, FLAGS.summary_freq)

 190 def train(train_model, pretrained_ckpt, imagenet_ckpt, checkpoint_dir,
 191           train_steps, summary_freq):
 192   """Train model."""
 193   vars_to_restore = None
 194   if pretrained_ckpt is not None:
 195     vars_to_restore = util.get_vars_to_save_and_restore(pretrained_ckpt)
 196     ckpt_path = pretrained_ckpt
 197   elif imagenet_ckpt:
 198     vars_to_restore = util.get_imagenet_vars_to_restore(imagenet_ckpt)
 199     ckpt_path = imagenet_ckpt
 200   pretrain_restorer = tf.train.Saver(vars_to_restore)
 201   vars_to_save = util.get_vars_to_save_and_restore()
 202   vars_to_save[train_model.global_step.op.name] = train_model.global_step
 203   saver = tf.train.Saver(vars_to_save, max_to_keep=MAX_TO_KEEP)
 204   sv = tf.train.Supervisor(logdir=checkpoint_dir, save_summaries_secs=0,
 205                            saver=None)
 206   config = tf.ConfigProto()
 207   config.gpu_options.allow_growth = True
 208   with sv.managed_session(config=config) as sess:
 209     if pretrained_ckpt is not None or imagenet_ckpt:
 210       logging.info('Restoring pretrained weights from %s', ckpt_path)
 211       pretrain_restorer.restore(sess, ckpt_path)
 212 
 213     logging.info('Attempting to resume training from %s...', checkpoint_dir)
 214     checkpoint = tf.train.latest_checkpoint(checkpoint_dir)
 215     logging.info('Last checkpoint found: %s', checkpoint)
 216     if checkpoint:
 217       saver.restore(sess, checkpoint)
 218 
 219     logging.info('Training...')
 220     start_time = time.time()
 221     last_summary_time = time.time()
 222     steps_per_epoch = train_model.reader.steps_per_epoch
 223     step = 1
 224     while step <= train_steps:
 225       fetches = {
 226           'train': train_model.train_op,
 227           'global_step': train_model.global_step,
 228           'incr_global_step': train_model.incr_global_step
 229       }
 230       if step % summary_freq == 0:
 231         fetches['loss'] = train_model.total_loss
 232         fetches['summary'] = sv.summary_op
 233 
 234       results = sess.run(fetches)
 235       global_step = results['global_step']
 236 
 237       if step % summary_freq == 0:
 238         sv.summary_writer.add_summary(results['summary'], global_step)
 239         train_epoch = math.ceil(global_step / steps_per_epoch)
 240         train_step = global_step - (train_epoch - 1) * steps_per_epoch
 241         this_cycle = time.time() - last_summary_time
 242         last_summary_time += this_cycle
 243         logging.info(
 244             'Epoch: [%2d] [%5d/%5d] time: %4.2fs (%ds total) loss: %.3f',
 245             train_epoch, train_step, steps_per_epoch, this_cycle,
 246             time.time() - start_time, results['loss'])
 247 
 248       if step % steps_per_epoch == 0:
 249         logging.info('[*] Saving checkpoint to %s...', checkpoint_dir)
 250         saver.save(sess, os.path.join(checkpoint_dir, 'model'),
 251                    global_step=global_step)
 252 
 253       # Setting step to global_step allows for training for a total of
 254       # train_steps even if the program is restarted during training.
 255       step = global_step + 1
 256 
 257 
 258 if __name__ == '__main__':
 259   app.run(main)
