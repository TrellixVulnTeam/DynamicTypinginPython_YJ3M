  40 def get_line(file, start):
  41     file = open(file, 'r')
  42     lines = file.readlines()
  43     lines = [line.rstrip() for line in lines]
  44     ret = None
  45     for line in lines:
  46         nline = line.split(': ')
  47         if nline[0]==start:
  48             ret = nline[1].split(' ')
  49             ret = np.array([float(r) for r in ret], dtype=float)
  50             ret = ret.reshape((3,4))[0:3, 0:3]
  51             break
  52     file.close()
  53     return ret

  56 def crop(img, segimg, fx, fy, cx, cy):
  57     # Perform center cropping, preserving 50% vertically.
  58     middle_perc = 0.50
  59     left = 1-middle_perc
  60     half = left/2
  61     a = img[int(img.shape[0]*(half)):int(img.shape[0]*(1-half)), :]
  62     aseg = segimg[int(segimg.shape[0]*(half)):int(segimg.shape[0]*(1-half)), :]
  63     cy /= (1/middle_perc)
  64 
  65     # Resize to match target height while preserving aspect ratio.
  66     wdt = int((128*a.shape[1]/a.shape[0]))
  67     x_scaling = float(wdt)/a.shape[1]
  68     y_scaling = 128.0/a.shape[0]
  69     b = cv2.resize(a, (wdt, 128))
  70     bseg = cv2.resize(aseg, (wdt, 128))
  71 
  72     # Adjust intrinsics.
  73     fx*=x_scaling
  74     fy*=y_scaling
  75     cx*=x_scaling
  76     cy*=y_scaling
  77 
  78     # Perform center cropping horizontally.
  79     remain = b.shape[1] - 416
  80     cx /= (b.shape[1]/416)
  81     c = b[:, int(remain/2):b.shape[1]-int(remain/2)]
  82     cseg = bseg[:, int(remain/2):b.shape[1]-int(remain/2)]
  83 
  84     return c, cseg, fx, fy, cx, cy

  87 def run_all():
  88   ct = 0
  89 if not OUTPUT_DIR.endswith('/'):
  90     OUTPUT_DIR = OUTPUT_DIR + '/'
  91 
  92 for d in glob.glob(INPUT_DIR + '/*/'):
  93     date = d.split('/')[-2]
  94     file_calibration = d + 'calib_cam_to_cam.txt'
  95     calib_raw = [get_line(file_calibration, 'P_rect_02'), get_line(file_calibration, 'P_rect_03')]
  96 
  97     for d2 in glob.glob(d + '*/'):
  98         seqname = d2.split('/')[-2]
  99         print('Processing sequence', seqname)
 100         for subfolder in ['image_02/data', 'image_03/data']:
 101             ct = 1
 102             seqname = d2.split('/')[-2] + subfolder.replace('image', '').replace('/data', '')
 103             if not os.path.exists(OUTPUT_DIR + seqname):
 104                 os.mkdir(OUTPUT_DIR + seqname)
 105 
 106             calib_camera = calib_raw[0] if subfolder=='image_02/data' else calib_raw[1]
 107             folder = d2 + subfolder
 108             files = glob.glob(folder + '/*.png')
 109             files = [file for file in files if not 'disp' in file and not 'flip' in file and not 'seg' in file]
 110             files = sorted(files)
 111             for i in range(SEQ_LENGTH, len(files)+1, STEPSIZE):
 112                 imgnum = str(ct).zfill(10)
 113                 if os.path.exists(OUTPUT_DIR + seqname + '/' + imgnum + '.png'):
 114                     ct+=1
 115                     continue
 116                 big_img = np.zeros(shape=(HEIGHT, WIDTH*SEQ_LENGTH, 3))
 117                 wct = 0
 118 
 119                 for j in range(i-SEQ_LENGTH, i):  # Collect frames for this sample.
 120                     img = cv2.imread(files[j])
 121                     ORIGINAL_HEIGHT, ORIGINAL_WIDTH, _ = img.shape
 122 
 123                     zoom_x = WIDTH/ORIGINAL_WIDTH
 124                     zoom_y = HEIGHT/ORIGINAL_HEIGHT
 125 
 126                     # Adjust intrinsics.
 127                     calib_current = calib_camera.copy()
 128                     calib_current[0, 0] *= zoom_x
 129                     calib_current[0, 2] *= zoom_x
 130                     calib_current[1, 1] *= zoom_y
 131                     calib_current[1, 2] *= zoom_y
 132 
 133                     calib_representation = ','.join([str(c) for c in calib_current.flatten()])
 134 
 135                     img = cv2.resize(img, (WIDTH, HEIGHT))
 136                     big_img[:,wct*WIDTH:(wct+1)*WIDTH] = img
 137                     wct+=1
 138                 cv2.imwrite(OUTPUT_DIR + seqname + '/' + imgnum + '.png', big_img)
 139                 f = open(OUTPUT_DIR + seqname + '/' + imgnum + '_cam.txt', 'w')
 140                 f.write(calib_representation)
 141                 f.close()
 142                 ct+=1
 143 
 144 def main(_):
 145   run_all()
