  32 QUEUE_SIZE = 2000
  33 QUEUE_BUFFER = 3
  34 # See nets.encoder_resnet as reference for below input-normalizing constants.
  35 IMAGENET_MEAN = (0.485, 0.456, 0.406)
  36 IMAGENET_SD = (0.229, 0.224, 0.225)
  37 FLIP_RANDOM = 'random'  # Always perform random flipping.
  38 FLIP_ALWAYS = 'always'  # Always flip image input, used for test augmentation.
  39 FLIP_NONE = 'none'  # Always disables flipping.

  45   def __init__(self, data_dir, batch_size, img_height, img_width, seq_length,
  46                num_scales, file_extension, random_scale_crop, flipping_mode,
  47                random_color, imagenet_norm, shuffle, input_file='train'):
  48     self.data_dir = data_dir
  49     self.batch_size = batch_size
  50     self.img_height = img_height
  51     self.img_width = img_width
  52     self.seq_length = seq_length
  53     self.num_scales = num_scales
  54     self.file_extension = file_extension
  55     self.random_scale_crop = random_scale_crop
  56     self.flipping_mode = flipping_mode
  57     self.random_color = random_color
  58     self.imagenet_norm = imagenet_norm
  59     self.shuffle = shuffle
  60     self.input_file = input_file

  62   def read_data(self):
  63     """Provides images and camera intrinsics."""
  64     with tf.name_scope('data_loading'):
  65       with tf.name_scope('enqueue_paths'):
  66         seed = random.randint(0, 2**31 - 1)
  67         self.file_lists = self.compile_file_list(self.data_dir, self.input_file)
  68         image_paths_queue = tf.train.string_input_producer(
  69             self.file_lists['image_file_list'], seed=seed,
  70             shuffle=self.shuffle,
  71             num_epochs=(1 if not self.shuffle else None)
  72         )
  73         seg_paths_queue = tf.train.string_input_producer(
  74             self.file_lists['segment_file_list'], seed=seed,
  75             shuffle=self.shuffle,
  76             num_epochs=(1 if not self.shuffle else None))
  77         cam_paths_queue = tf.train.string_input_producer(
  78             self.file_lists['cam_file_list'], seed=seed,
  79             shuffle=self.shuffle,
  80             num_epochs=(1 if not self.shuffle else None))
  81         img_reader = tf.WholeFileReader()
  82         _, image_contents = img_reader.read(image_paths_queue)
  83         seg_reader = tf.WholeFileReader()
  84         _, seg_contents = seg_reader.read(seg_paths_queue)
  85         if self.file_extension == 'jpg':
  86           image_seq = tf.image.decode_jpeg(image_contents)
  87           seg_seq = tf.image.decode_jpeg(seg_contents, channels=3)
  88         elif self.file_extension == 'png':
  89           image_seq = tf.image.decode_png(image_contents, channels=3)
  90           seg_seq = tf.image.decode_png(seg_contents, channels=3)
  91 
  92       with tf.name_scope('load_intrinsics'):
  93         cam_reader = tf.TextLineReader()
  94         _, raw_cam_contents = cam_reader.read(cam_paths_queue)
  95         rec_def = []
  96         for _ in range(9):
  97           rec_def.append([1.0])
  98         raw_cam_vec = tf.decode_csv(raw_cam_contents, record_defaults=rec_def)
  99         raw_cam_vec = tf.stack(raw_cam_vec)
 100         intrinsics = tf.reshape(raw_cam_vec, [3, 3])
 101 
 102       with tf.name_scope('convert_image'):
 103         image_seq = self.preprocess_image(image_seq)  # Converts to float.
 104 
 105       if self.random_color:
 106         with tf.name_scope('image_augmentation'):
 107           image_seq = self.augment_image_colorspace(image_seq)
 108 
 109       image_stack = self.unpack_images(image_seq)
 110       seg_stack = self.unpack_images(seg_seq)
 111 
 112       if self.flipping_mode != FLIP_NONE:
 113         random_flipping = (self.flipping_mode == FLIP_RANDOM)
 114         with tf.name_scope('image_augmentation_flip'):
 115           image_stack, seg_stack, intrinsics = self.augment_images_flip(
 116               image_stack, seg_stack, intrinsics,
 117               randomized=random_flipping)
 118 
 119       if self.random_scale_crop:
 120         with tf.name_scope('image_augmentation_scale_crop'):
 121           image_stack, seg_stack, intrinsics = self.augment_images_scale_crop(
 122               image_stack, seg_stack, intrinsics, self.img_height,
 123               self.img_width)
 124 
 125       with tf.name_scope('multi_scale_intrinsics'):
 126         intrinsic_mat = self.get_multi_scale_intrinsics(intrinsics,
 127                                                         self.num_scales)
 128         intrinsic_mat.set_shape([self.num_scales, 3, 3])
 129         intrinsic_mat_inv = tf.matrix_inverse(intrinsic_mat)
 130         intrinsic_mat_inv.set_shape([self.num_scales, 3, 3])
 131 
 132       if self.imagenet_norm:
 133         im_mean = tf.tile(
 134             tf.constant(IMAGENET_MEAN), multiples=[self.seq_length])
 135         im_sd = tf.tile(
 136             tf.constant(IMAGENET_SD), multiples=[self.seq_length])
 137         image_stack_norm = (image_stack - im_mean) / im_sd
 138       else:
 139         image_stack_norm = image_stack
 140 
 141       with tf.name_scope('batching'):
 142         if self.shuffle:
 143           (image_stack, image_stack_norm, seg_stack, intrinsic_mat,
 144            intrinsic_mat_inv) = tf.train.shuffle_batch(
 145                [image_stack, image_stack_norm, seg_stack, intrinsic_mat,
 146                 intrinsic_mat_inv],
 147                batch_size=self.batch_size,
 148                capacity=QUEUE_SIZE + QUEUE_BUFFER * self.batch_size,
 149                min_after_dequeue=QUEUE_SIZE)
 150         else:
 151           (image_stack, image_stack_norm, seg_stack, intrinsic_mat,
 152            intrinsic_mat_inv) = tf.train.batch(
 153                [image_stack, image_stack_norm, seg_stack, intrinsic_mat,
 154                 intrinsic_mat_inv],
 155                batch_size=self.batch_size,
 156                num_threads=1,
 157                capacity=QUEUE_SIZE + QUEUE_BUFFER * self.batch_size)
 158         logging.info('image_stack: %s', util.info(image_stack))
 159     return (image_stack, image_stack_norm, seg_stack, intrinsic_mat,
 160             intrinsic_mat_inv)
 161 
 162   def unpack_images(self, image_seq):
 163     """[h, w * seq_length, 3] -> [h, w, 3 * seq_length]."""
 164     with tf.name_scope('unpack_images'):
 165       image_list = [
 166           image_seq[:, i * self.img_width:(i + 1) * self.img_width, :]
 167           for i in range(self.seq_length)
 168       ]
 169       image_stack = tf.concat(image_list, axis=2)
 170       image_stack.set_shape(
 171           [self.img_height, self.img_width, self.seq_length * 3])
 172     return image_stack

 174   @classmethod
 175   def preprocess_image(cls, image):
 176     # Convert from uint8 to float.
 177     return tf.image.convert_image_dtype(image, dtype=tf.float32)

 179   @classmethod
 180   def augment_image_colorspace(cls, image_stack):
 181     """Apply data augmentation to inputs."""
 182     image_stack_aug = image_stack
 183     # Randomly shift brightness.
 184     apply_brightness = tf.less(tf.random_uniform(
 185         shape=[], minval=0.0, maxval=1.0, dtype=tf.float32), 0.5)
 186     image_stack_aug = tf.cond(
 187         apply_brightness,
 188         lambda: tf.image.random_brightness(image_stack_aug, max_delta=0.1),
 189         lambda: image_stack_aug)
 190 
 191     # Randomly shift contrast.
 192     apply_contrast = tf.less(tf.random_uniform(
 193         shape=[], minval=0.0, maxval=1.0, dtype=tf.float32), 0.5)
 194     image_stack_aug = tf.cond(
 195         apply_contrast,
 196         lambda: tf.image.random_contrast(image_stack_aug, 0.85, 1.15),
 197         lambda: image_stack_aug)
 198 
 199     # Randomly change saturation.
 200     apply_saturation = tf.less(tf.random_uniform(
 201         shape=[], minval=0.0, maxval=1.0, dtype=tf.float32), 0.5)
 202     image_stack_aug = tf.cond(
 203         apply_saturation,
 204         lambda: tf.image.random_saturation(image_stack_aug, 0.85, 1.15),
 205         lambda: image_stack_aug)
 206 
 207     # Randomly change hue.
 208     apply_hue = tf.less(tf.random_uniform(
 209         shape=[], minval=0.0, maxval=1.0, dtype=tf.float32), 0.5)
 210     image_stack_aug = tf.cond(
 211         apply_hue,
 212         lambda: tf.image.random_hue(image_stack_aug, max_delta=0.1),
 213         lambda: image_stack_aug)
 214 
 215     image_stack_aug = tf.clip_by_value(image_stack_aug, 0, 1)
 216     return image_stack_aug

 218   @classmethod
 219   def augment_images_flip(cls, image_stack, seg_stack, intrinsics,
 220                           randomized=True):
 221     """Randomly flips the image horizontally."""
 222 
 223     def flip(cls, image_stack, seg_stack, intrinsics):
 224       _, in_w, _ = image_stack.get_shape().as_list()
 225       fx = intrinsics[0, 0]
 226       fy = intrinsics[1, 1]
 227       cx = in_w - intrinsics[0, 2]
 228       cy = intrinsics[1, 2]
 229       intrinsics = cls.make_intrinsics_matrix(fx, fy, cx, cy)
 230       return (tf.image.flip_left_right(image_stack),
 231               tf.image.flip_left_right(seg_stack), intrinsics)
 232 
 233     if randomized:
 234       prob = tf.random_uniform(shape=[], minval=0.0, maxval=1.0,
 235                                dtype=tf.float32)
 236       predicate = tf.less(prob, 0.5)
 237       return tf.cond(predicate,
 238                      lambda: flip(cls, image_stack, seg_stack, intrinsics),
 239                      lambda: (image_stack, seg_stack, intrinsics))
 240     else:
 241       return flip(cls, image_stack, seg_stack, intrinsics)

 243   @classmethod
 244   def augment_images_scale_crop(cls, im, seg, intrinsics, out_h, out_w):
 245     """Randomly scales and crops image."""
 246 
 247     def scale_randomly(im, seg, intrinsics):
 248       """Scales image and adjust intrinsics accordingly."""
 249       in_h, in_w, _ = im.get_shape().as_list()
 250       scaling = tf.random_uniform([2], 1, 1.15)
 251       x_scaling = scaling[0]
 252       y_scaling = scaling[1]
 253       out_h = tf.cast(in_h * y_scaling, dtype=tf.int32)
 254       out_w = tf.cast(in_w * x_scaling, dtype=tf.int32)
 255       # Add batch.
 256       im = tf.expand_dims(im, 0)
 257       im = tf.image.resize_area(im, [out_h, out_w])
 258       im = im[0]
 259       seg = tf.expand_dims(seg, 0)
 260       seg = tf.image.resize_area(seg, [out_h, out_w])
 261       seg = seg[0]
 262       fx = intrinsics[0, 0] * x_scaling
 263       fy = intrinsics[1, 1] * y_scaling
 264       cx = intrinsics[0, 2] * x_scaling
 265       cy = intrinsics[1, 2] * y_scaling
 266       intrinsics = cls.make_intrinsics_matrix(fx, fy, cx, cy)
 267       return im, seg, intrinsics
 268 
 269     # Random cropping
 270     def crop_randomly(im, seg, intrinsics, out_h, out_w):
 271       """Crops image and adjust intrinsics accordingly."""
 272       # batch_size, in_h, in_w, _ = im.get_shape().as_list()
 273       in_h, in_w, _ = tf.unstack(tf.shape(im))
 274       offset_y = tf.random_uniform([1], 0, in_h - out_h + 1, dtype=tf.int32)[0]
 275       offset_x = tf.random_uniform([1], 0, in_w - out_w + 1, dtype=tf.int32)[0]
 276       im = tf.image.crop_to_bounding_box(im, offset_y, offset_x, out_h, out_w)
 277       seg = tf.image.crop_to_bounding_box(seg, offset_y, offset_x, out_h, out_w)
 278       fx = intrinsics[0, 0]
 279       fy = intrinsics[1, 1]
 280       cx = intrinsics[0, 2] - tf.cast(offset_x, dtype=tf.float32)
 281       cy = intrinsics[1, 2] - tf.cast(offset_y, dtype=tf.float32)
 282       intrinsics = cls.make_intrinsics_matrix(fx, fy, cx, cy)
 283       return im, seg, intrinsics
 284 
 285     im, seg, intrinsics = scale_randomly(im, seg, intrinsics)
 286     im, seg, intrinsics = crop_randomly(im, seg, intrinsics, out_h, out_w)
 287     return im, seg, intrinsics

 289   def compile_file_list(self, data_dir, split, load_pose=False):
 290     """Creates a list of input files."""
 291     logging.info('data_dir: %s', data_dir)
 292     with gfile.Open(os.path.join(data_dir, '%s.txt' % split), 'r') as f:
 293       frames = f.readlines()
 294       frames = [k.rstrip() for k in frames]
 295     subfolders = [x.split(' ')[0] for x in frames]
 296     frame_ids = [x.split(' ')[1] for x in frames]
 297     image_file_list = [
 298         os.path.join(data_dir, subfolders[i], frame_ids[i] + '.' +
 299                      self.file_extension)
 300         for i in range(len(frames))
 301     ]
 302     segment_file_list = [
 303         os.path.join(data_dir, subfolders[i], frame_ids[i] + '-fseg.' +
 304                      self.file_extension)
 305         for i in range(len(frames))
 306     ]
 307     cam_file_list = [
 308         os.path.join(data_dir, subfolders[i], frame_ids[i] + '_cam.txt')
 309         for i in range(len(frames))
 310     ]
 311     file_lists = {}
 312     file_lists['image_file_list'] = image_file_list
 313     file_lists['segment_file_list'] = segment_file_list
 314     file_lists['cam_file_list'] = cam_file_list
 315     if load_pose:
 316       pose_file_list = [
 317           os.path.join(data_dir, subfolders[i], frame_ids[i] + '_pose.txt')
 318           for i in range(len(frames))
 319       ]
 320       file_lists['pose_file_list'] = pose_file_list
 321     self.steps_per_epoch = len(image_file_list) // self.batch_size
 322     return file_lists

 324   @classmethod
 325   def make_intrinsics_matrix(cls, fx, fy, cx, cy):
 326     r1 = tf.stack([fx, 0, cx])
 327     r2 = tf.stack([0, fy, cy])
 328     r3 = tf.constant([0., 0., 1.])
 329     intrinsics = tf.stack([r1, r2, r3])
 330     return intrinsics

 332   @classmethod
 333   def get_multi_scale_intrinsics(cls, intrinsics, num_scales):
 334     """Returns multiple intrinsic matrices for different scales."""
 335     intrinsics_multi_scale = []
 336     # Scale the intrinsics accordingly for each scale
 337     for s in range(num_scales):
 338       fx = intrinsics[0, 0] / (2**s)
 339       fy = intrinsics[1, 1] / (2**s)
 340       cx = intrinsics[0, 2] / (2**s)
 341       cy = intrinsics[1, 2] / (2**s)
 342       intrinsics_multi_scale.append(cls.make_intrinsics_matrix(fx, fy, cx, cy))
 343     intrinsics_multi_scale = tf.stack(intrinsics_multi_scale)
 344     return intrinsics_multi_scale
