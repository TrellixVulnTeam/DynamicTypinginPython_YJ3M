  31 SCALE_TRANSLATION = 0.001
  32 SCALE_ROTATION = 0.01
  33 
  34 # Disparity (inverse depth) values range from 0.01 to 10. Note that effectively,
  35 # this is undone if depth normalization is used, which scales the values to
  36 # have a mean of 1.
  37 DISP_SCALING = 10
  38 MIN_DISP = 0.01
  39 WEIGHT_DECAY_KEY = 'WEIGHT_DECAY'
  40 EGOMOTION_VEC_SIZE = 6

  43 def egomotion_net(image_stack, disp_bottleneck_stack, joint_encoder, seq_length,
  44                   weight_reg):
  45   """Predict ego-motion vectors from a stack of frames or embeddings.
  46 
  47   Args:
  48     image_stack: Input tensor with shape [B, h, w, seq_length * 3] in order.
  49     disp_bottleneck_stack: Input tensor with shape [B, h_hidden, w_hidden,
  50         seq_length * c_hidden] in order.
  51     joint_encoder: Determines if the same encoder is used for computing the
  52         bottleneck layer of both the egomotion and the depth prediction
  53         network. If enabled, disp_bottleneck_stack is used as input, and the
  54         encoding steps are skipped. If disabled, a separate encoder is defined
  55         on image_stack.
  56     seq_length: The sequence length used.
  57     weight_reg: The amount of weight regularization.
  58 
  59   Returns:
  60     Egomotion vectors with shape [B, seq_length - 1, 6].
  61   """
  62   num_egomotion_vecs = seq_length - 1
  63   with tf.variable_scope('pose_exp_net') as sc:
  64     end_points_collection = sc.original_name_scope + '_end_points'
  65     with slim.arg_scope([slim.conv2d, slim.conv2d_transpose],
  66                         normalizer_fn=None,
  67                         weights_regularizer=slim.l2_regularizer(weight_reg),
  68                         normalizer_params=None,
  69                         activation_fn=tf.nn.relu,
  70                         outputs_collections=end_points_collection):
  71       if not joint_encoder:
  72         # Define separate encoder. If sharing, we can skip the encoding step,
  73         # as the bottleneck layer will already be passed as input.
  74         cnv1 = slim.conv2d(image_stack, 16, [7, 7], stride=2, scope='cnv1')
  75         cnv2 = slim.conv2d(cnv1, 32, [5, 5], stride=2, scope='cnv2')
  76         cnv3 = slim.conv2d(cnv2, 64, [3, 3], stride=2, scope='cnv3')
  77         cnv4 = slim.conv2d(cnv3, 128, [3, 3], stride=2, scope='cnv4')
  78         cnv5 = slim.conv2d(cnv4, 256, [3, 3], stride=2, scope='cnv5')
  79 
  80       with tf.variable_scope('pose'):
  81         inputs = disp_bottleneck_stack if joint_encoder else cnv5
  82         cnv6 = slim.conv2d(inputs, 256, [3, 3], stride=2, scope='cnv6')
  83         cnv7 = slim.conv2d(cnv6, 256, [3, 3], stride=2, scope='cnv7')
  84         pred_channels = EGOMOTION_VEC_SIZE * num_egomotion_vecs
  85         egomotion_pred = slim.conv2d(cnv7, pred_channels, [1, 1], scope='pred',
  86                                      stride=1, normalizer_fn=None,
  87                                      activation_fn=None)
  88         egomotion_avg = tf.reduce_mean(egomotion_pred, [1, 2])
  89         egomotion_res = tf.reshape(
  90             egomotion_avg, [-1, num_egomotion_vecs, EGOMOTION_VEC_SIZE])
  91         # Tinghui found that scaling by a small constant facilitates training.
  92         egomotion_scaled = tf.concat([egomotion_res[:, 0:3] * SCALE_TRANSLATION,
  93                                       egomotion_res[:, 3:6] * SCALE_ROTATION],
  94                                      axis=1)
  95     return egomotion_scaled

  98 def objectmotion_net(image_stack, disp_bottleneck_stack, joint_encoder,
  99                      seq_length, weight_reg):
 100   """Predict object-motion vectors from a stack of frames or embeddings.
 101 
 102   Args:
 103     image_stack: Input tensor with shape [B, h, w, seq_length * 3] in order.
 104     disp_bottleneck_stack: Input tensor with shape [B, h_hidden, w_hidden,
 105         seq_length * c_hidden] in order.
 106     joint_encoder: Determines if the same encoder is used for computing the
 107         bottleneck layer of both the egomotion and the depth prediction
 108         network. If enabled, disp_bottleneck_stack is used as input, and the
 109         encoding steps are skipped. If disabled, a separate encoder is defined
 110         on image_stack.
 111     seq_length: The sequence length used.
 112     weight_reg: The amount of weight regularization.
 113 
 114   Returns:
 115     Egomotion vectors with shape [B, seq_length - 1, 6].
 116   """
 117   num_egomotion_vecs = seq_length - 1
 118   with tf.variable_scope('pose_exp_net') as sc:
 119     end_points_collection = sc.original_name_scope + '_end_points'
 120     with slim.arg_scope([slim.conv2d, slim.conv2d_transpose],
 121                         normalizer_fn=None,
 122                         weights_regularizer=slim.l2_regularizer(weight_reg),
 123                         normalizer_params=None,
 124                         activation_fn=tf.nn.relu,
 125                         outputs_collections=end_points_collection):
 126       if not joint_encoder:
 127         # Define separate encoder. If sharing, we can skip the encoding step,
 128         # as the bottleneck layer will already be passed as input.
 129         cnv1 = slim.conv2d(image_stack, 16, [7, 7], stride=2, scope='cnv1')
 130         cnv2 = slim.conv2d(cnv1, 32, [5, 5], stride=2, scope='cnv2')
 131         cnv3 = slim.conv2d(cnv2, 64, [3, 3], stride=2, scope='cnv3')
 132         cnv4 = slim.conv2d(cnv3, 128, [3, 3], stride=2, scope='cnv4')
 133         cnv5 = slim.conv2d(cnv4, 256, [3, 3], stride=2, scope='cnv5')
 134 
 135       with tf.variable_scope('pose'):
 136         inputs = disp_bottleneck_stack if joint_encoder else cnv5
 137         cnv6 = slim.conv2d(inputs, 256, [3, 3], stride=2, scope='cnv6')
 138         cnv7 = slim.conv2d(cnv6, 256, [3, 3], stride=2, scope='cnv7')
 139         pred_channels = EGOMOTION_VEC_SIZE * num_egomotion_vecs
 140         egomotion_pred = slim.conv2d(cnv7, pred_channels, [1, 1], scope='pred',
 141                                      stride=1, normalizer_fn=None,
 142                                      activation_fn=None)
 143         egomotion_avg = tf.reduce_mean(egomotion_pred, [1, 2])
 144         egomotion_res = tf.reshape(
 145             egomotion_avg, [-1, num_egomotion_vecs, EGOMOTION_VEC_SIZE])
 146         # Tinghui found that scaling by a small constant facilitates training.
 147         egomotion_scaled = tf.concat([egomotion_res[:, 0:3] * SCALE_TRANSLATION,
 148                                       egomotion_res[:, 3:6] * SCALE_ROTATION],
 149                                      axis=1)
 150     return egomotion_scaled

 153 def disp_net(architecture, image, use_skip, weight_reg, is_training):
 154   """Defines an encoder-decoder architecture for depth prediction."""
 155   if architecture not in ARCHITECTURES:
 156     raise ValueError('Unknown architecture.')
 157   encoder_selected = encoder(architecture)
 158   decoder_selected = decoder(architecture)
 159 
 160   # Encode image.
 161   bottleneck, skip_connections = encoder_selected(image, weight_reg,
 162                                                   is_training)
 163   # Decode to depth.
 164   multiscale_disps_i = decoder_selected(target_image=image,
 165                                         bottleneck=bottleneck,
 166                                         weight_reg=weight_reg,
 167                                         use_skip=use_skip,
 168                                         skip_connections=skip_connections)
 169   return multiscale_disps_i, bottleneck

 180 def encoder_simple(target_image, weight_reg, is_training):
 181   """Defines the old encoding architecture."""
 182   del is_training
 183   with slim.arg_scope([slim.conv2d],
 184                       normalizer_fn=None,
 185                       normalizer_params=None,
 186                       weights_regularizer=slim.l2_regularizer(weight_reg),
 187                       activation_fn=tf.nn.relu):
 188     # Define (joint) encoder.
 189     cnv1 = slim.conv2d(target_image, 32, [7, 7], stride=2, scope='cnv1')
 190     cnv1b = slim.conv2d(cnv1, 32, [7, 7], stride=1, scope='cnv1b')
 191     cnv2 = slim.conv2d(cnv1b, 64, [5, 5], stride=2, scope='cnv2')
 192     cnv2b = slim.conv2d(cnv2, 64, [5, 5], stride=1, scope='cnv2b')
 193     cnv3 = slim.conv2d(cnv2b, 128, [3, 3], stride=2, scope='cnv3')
 194     cnv3b = slim.conv2d(cnv3, 128, [3, 3], stride=1, scope='cnv3b')
 195     cnv4 = slim.conv2d(cnv3b, 256, [3, 3], stride=2, scope='cnv4')
 196     cnv4b = slim.conv2d(cnv4, 256, [3, 3], stride=1, scope='cnv4b')
 197     cnv5 = slim.conv2d(cnv4b, 512, [3, 3], stride=2, scope='cnv5')
 198     cnv5b = slim.conv2d(cnv5, 512, [3, 3], stride=1, scope='cnv5b')
 199     cnv6 = slim.conv2d(cnv5b, 512, [3, 3], stride=2, scope='cnv6')
 200     cnv6b = slim.conv2d(cnv6, 512, [3, 3], stride=1, scope='cnv6b')
 201     cnv7 = slim.conv2d(cnv6b, 512, [3, 3], stride=2, scope='cnv7')
 202     cnv7b = slim.conv2d(cnv7, 512, [3, 3], stride=1, scope='cnv7b')
 203   return cnv7b, (cnv6b, cnv5b, cnv4b, cnv3b, cnv2b, cnv1b)

 206 def decoder_simple(target_image, bottleneck, weight_reg, use_skip,
 207                    skip_connections):
 208   """Defines the old depth decoder architecture."""
 209   h = target_image.get_shape()[1].value
 210   w = target_image.get_shape()[2].value
 211   (cnv6b, cnv5b, cnv4b, cnv3b, cnv2b, cnv1b) = skip_connections
 212   with slim.arg_scope([slim.conv2d, slim.conv2d_transpose],
 213                       normalizer_fn=None,
 214                       normalizer_params=None,
 215                       weights_regularizer=slim.l2_regularizer(weight_reg),
 216                       activation_fn=tf.nn.relu):
 217     up7 = slim.conv2d_transpose(bottleneck, 512, [3, 3], stride=2,
 218                                 scope='upcnv7')
 219     up7 = _resize_like(up7, cnv6b)
 220     if use_skip:
 221       i7_in = tf.concat([up7, cnv6b], axis=3)
 222     else:
 223       i7_in = up7
 224     icnv7 = slim.conv2d(i7_in, 512, [3, 3], stride=1, scope='icnv7')
 225 
 226     up6 = slim.conv2d_transpose(icnv7, 512, [3, 3], stride=2, scope='upcnv6')
 227     up6 = _resize_like(up6, cnv5b)
 228     if use_skip:
 229       i6_in = tf.concat([up6, cnv5b], axis=3)
 230     else:
 231       i6_in = up6
 232     icnv6 = slim.conv2d(i6_in, 512, [3, 3], stride=1, scope='icnv6')
 233 
 234     up5 = slim.conv2d_transpose(icnv6, 256, [3, 3], stride=2, scope='upcnv5')
 235     up5 = _resize_like(up5, cnv4b)
 236     if use_skip:
 237       i5_in = tf.concat([up5, cnv4b], axis=3)
 238     else:
 239       i5_in = up5
 240     icnv5 = slim.conv2d(i5_in, 256, [3, 3], stride=1, scope='icnv5')
 241 
 242     up4 = slim.conv2d_transpose(icnv5, 128, [3, 3], stride=2, scope='upcnv4')
 243     up4 = _resize_like(up4, cnv3b)
 244     if use_skip:
 245       i4_in = tf.concat([up4, cnv3b], axis=3)
 246     else:
 247       i4_in = up4
 248     icnv4 = slim.conv2d(i4_in, 128, [3, 3], stride=1, scope='icnv4')
 249     disp4 = (slim.conv2d(icnv4, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 250                          normalizer_fn=None, scope='disp4')
 251              * DISP_SCALING + MIN_DISP)
 252     disp4_up = tf.image.resize_bilinear(disp4, [np.int(h / 4), np.int(w / 4)],
 253                                         align_corners=True)
 254 
 255     up3 = slim.conv2d_transpose(icnv4, 64, [3, 3], stride=2, scope='upcnv3')
 256     up3 = _resize_like(up3, cnv2b)
 257     if use_skip:
 258       i3_in = tf.concat([up3, cnv2b, disp4_up], axis=3)
 259     else:
 260       i3_in = tf.concat([up3, disp4_up])
 261     icnv3 = slim.conv2d(i3_in, 64, [3, 3], stride=1, scope='icnv3')
 262     disp3 = (slim.conv2d(icnv3, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 263                          normalizer_fn=None, scope='disp3')
 264              * DISP_SCALING + MIN_DISP)
 265     disp3_up = tf.image.resize_bilinear(disp3, [np.int(h / 2), np.int(w / 2)],
 266                                         align_corners=True)
 267 
 268     up2 = slim.conv2d_transpose(icnv3, 32, [3, 3], stride=2, scope='upcnv2')
 269     up2 = _resize_like(up2, cnv1b)
 270     if use_skip:
 271       i2_in = tf.concat([up2, cnv1b, disp3_up], axis=3)
 272     else:
 273       i2_in = tf.concat([up2, disp3_up])
 274     icnv2 = slim.conv2d(i2_in, 32, [3, 3], stride=1, scope='icnv2')
 275     disp2 = (slim.conv2d(icnv2, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 276                          normalizer_fn=None, scope='disp2')
 277              * DISP_SCALING + MIN_DISP)
 278     disp2_up = tf.image.resize_bilinear(disp2, [h, w], align_corners=True)
 279 
 280     up1 = slim.conv2d_transpose(icnv2, 16, [3, 3], stride=2, scope='upcnv1')
 281     i1_in = tf.concat([up1, disp2_up], axis=3)
 282     icnv1 = slim.conv2d(i1_in, 16, [3, 3], stride=1, scope='icnv1')
 283     disp1 = (slim.conv2d(icnv1, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 284                          normalizer_fn=None, scope='disp1')
 285              * DISP_SCALING + MIN_DISP)
 286   return [disp1, disp2, disp3, disp4]

 289 def encoder_resnet(target_image, weight_reg, is_training):
 290   """Defines a ResNet18-based encoding architecture.
 291 
 292   This implementation follows Juyong Kim's implementation of ResNet18 on GitHub:
 293   https://github.com/dalgu90/resnet-18-tensorflow
 294 
 295   Args:
 296     target_image: Input tensor with shape [B, h, w, 3] to encode.
 297     weight_reg: Parameter ignored.
 298     is_training: Whether the model is being trained or not.
 299 
 300   Returns:
 301     Tuple of tensors, with the first being the bottleneck layer as tensor of
 302     size [B, h_hid, w_hid, c_hid], and others being intermediate layers
 303     for building skip-connections.
 304   """
 305   del weight_reg
 306   encoder_filters = [64, 64, 128, 256, 512]
 307   stride = 2
 308 
 309   # conv1
 310   with tf.variable_scope('conv1'):
 311     x = _conv(target_image, 7, encoder_filters[0], stride)
 312     x = _bn(x, is_train=is_training)
 313     econv1 = _relu(x)
 314     x = tf.nn.max_pool(econv1, [1, 3, 3, 1], [1, 2, 2, 1], 'SAME')
 315 
 316   # conv2_x
 317   x = _residual_block(x, is_training, name='conv2_1')
 318   econv2 = _residual_block(x, is_training, name='conv2_2')
 319 
 320   # conv3_x
 321   x = _residual_block_first(econv2, is_training, encoder_filters[2], stride,
 322                             name='conv3_1')
 323   econv3 = _residual_block(x, is_training, name='conv3_2')
 324 
 325   # conv4_x
 326   x = _residual_block_first(econv3, is_training, encoder_filters[3], stride,
 327                             name='conv4_1')
 328   econv4 = _residual_block(x, is_training, name='conv4_2')
 329 
 330   # conv5_x
 331   x = _residual_block_first(econv4, is_training, encoder_filters[4], stride,
 332                             name='conv5_1')
 333   econv5 = _residual_block(x, is_training, name='conv5_2')
 334   return econv5, (econv4, econv3, econv2, econv1)

 337 def decoder_resnet(target_image, bottleneck, weight_reg, use_skip,
 338                    skip_connections):
 339   """Defines the depth decoder architecture.
 340 
 341   Args:
 342     target_image: The original encoder input tensor with shape [B, h, w, 3].
 343                   Just the shape information is used here.
 344     bottleneck: Bottleneck layer to be decoded.
 345     weight_reg: The amount of weight regularization.
 346     use_skip: Whether the passed skip connections econv1, econv2, econv3 and
 347               econv4 should be used.
 348     skip_connections: Tensors for building skip-connections.
 349 
 350   Returns:
 351     Disparities at 4 different scales.
 352   """
 353   (econv4, econv3, econv2, econv1) = skip_connections
 354   decoder_filters = [16, 32, 64, 128, 256]
 355   default_pad = tf.constant([[0, 0], [1, 1], [1, 1], [0, 0]])
 356   reg = slim.l2_regularizer(weight_reg) if weight_reg > 0.0 else None
 357   with slim.arg_scope([slim.conv2d, slim.conv2d_transpose],
 358                       normalizer_fn=None,
 359                       normalizer_params=None,
 360                       activation_fn=tf.nn.relu,
 361                       weights_regularizer=reg):
 362     upconv5 = slim.conv2d_transpose(bottleneck, decoder_filters[4], [3, 3],
 363                                     stride=2, scope='upconv5')
 364     upconv5 = _resize_like(upconv5, econv4)
 365     if use_skip:
 366       i5_in = tf.concat([upconv5, econv4], axis=3)
 367     else:
 368       i5_in = upconv5
 369     i5_in = tf.pad(i5_in, default_pad, mode='REFLECT')
 370     iconv5 = slim.conv2d(i5_in, decoder_filters[4], [3, 3], stride=1,
 371                          scope='iconv5', padding='VALID')
 372 
 373     upconv4 = slim.conv2d_transpose(iconv5, decoder_filters[3], [3, 3],
 374                                     stride=2, scope='upconv4')
 375     upconv4 = _resize_like(upconv4, econv3)
 376     if use_skip:
 377       i4_in = tf.concat([upconv4, econv3], axis=3)
 378     else:
 379       i4_in = upconv4
 380     i4_in = tf.pad(i4_in, default_pad, mode='REFLECT')
 381     iconv4 = slim.conv2d(i4_in, decoder_filters[3], [3, 3], stride=1,
 382                          scope='iconv4', padding='VALID')
 383 
 384     disp4_input = tf.pad(iconv4, default_pad, mode='REFLECT')
 385     disp4 = (slim.conv2d(disp4_input, 1, [3, 3], stride=1,
 386                          activation_fn=tf.sigmoid, normalizer_fn=None,
 387                          scope='disp4', padding='VALID')
 388              * DISP_SCALING + MIN_DISP)
 389 
 390     upconv3 = slim.conv2d_transpose(iconv4, decoder_filters[2], [3, 3],
 391                                     stride=2, scope='upconv3')
 392     upconv3 = _resize_like(upconv3, econv2)
 393     if use_skip:
 394       i3_in = tf.concat([upconv3, econv2], axis=3)
 395     else:
 396       i3_in = upconv3
 397     i3_in = tf.pad(i3_in, default_pad, mode='REFLECT')
 398     iconv3 = slim.conv2d(i3_in, decoder_filters[2], [3, 3], stride=1,
 399                          scope='iconv3', padding='VALID')
 400     disp3_input = tf.pad(iconv3, default_pad, mode='REFLECT')
 401     disp3 = (slim.conv2d(disp3_input, 1, [3, 3], stride=1,
 402                          activation_fn=tf.sigmoid, normalizer_fn=None,
 403                          scope='disp3', padding='VALID')
 404              * DISP_SCALING + MIN_DISP)
 405 
 406     upconv2 = slim.conv2d_transpose(iconv3, decoder_filters[1], [3, 3],
 407                                     stride=2, scope='upconv2')
 408     upconv2 = _resize_like(upconv2, econv1)
 409     if use_skip:
 410       i2_in = tf.concat([upconv2, econv1], axis=3)
 411     else:
 412       i2_in = upconv2
 413     i2_in = tf.pad(i2_in, default_pad, mode='REFLECT')
 414     iconv2 = slim.conv2d(i2_in, decoder_filters[1], [3, 3], stride=1,
 415                          scope='iconv2', padding='VALID')
 416     disp2_input = tf.pad(iconv2, default_pad, mode='REFLECT')
 417     disp2 = (slim.conv2d(disp2_input, 1, [3, 3], stride=1,
 418                          activation_fn=tf.sigmoid, normalizer_fn=None,
 419                          scope='disp2', padding='VALID')
 420              * DISP_SCALING + MIN_DISP)
 421 
 422     upconv1 = slim.conv2d_transpose(iconv2, decoder_filters[0], [3, 3],
 423                                     stride=2, scope='upconv1')
 424     upconv1 = _resize_like(upconv1, target_image)
 425     upconv1 = tf.pad(upconv1, default_pad, mode='REFLECT')
 426     iconv1 = slim.conv2d(upconv1, decoder_filters[0], [3, 3], stride=1,
 427                          scope='iconv1', padding='VALID')
 428     disp1_input = tf.pad(iconv1, default_pad, mode='REFLECT')
 429     disp1 = (slim.conv2d(disp1_input, 1, [3, 3], stride=1,
 430                          activation_fn=tf.sigmoid, normalizer_fn=None,
 431                          scope='disp1', padding='VALID')
 432              * DISP_SCALING + MIN_DISP)
 433 
 434   return [disp1, disp2, disp3, disp4]

 437 def _residual_block_first(x, is_training, out_channel, strides, name='unit'):
 438   """Helper function for defining ResNet architecture."""
 439   in_channel = x.get_shape().as_list()[-1]
 440   with tf.variable_scope(name):
 441     # Shortcut connection
 442     if in_channel == out_channel:
 443       if strides == 1:
 444         shortcut = tf.identity(x)
 445       else:
 446         shortcut = tf.nn.max_pool(x, [1, strides, strides, 1],
 447                                   [1, strides, strides, 1], 'VALID')
 448     else:
 449       shortcut = _conv(x, 1, out_channel, strides, name='shortcut')
 450     # Residual
 451     x = _conv(x, 3, out_channel, strides, name='conv_1')
 452     x = _bn(x, is_train=is_training, name='bn_1')
 453     x = _relu(x, name='relu_1')
 454     x = _conv(x, 3, out_channel, 1, name='conv_2')
 455     x = _bn(x, is_train=is_training, name='bn_2')
 456     # Merge
 457     x = x + shortcut
 458     x = _relu(x, name='relu_2')
 459   return x

 462 def _residual_block(x, is_training, input_q=None, output_q=None, name='unit'):
 463   """Helper function for defining ResNet architecture."""
 464   num_channel = x.get_shape().as_list()[-1]
 465   with tf.variable_scope(name):
 466     shortcut = x  # Shortcut connection
 467     # Residual
 468     x = _conv(x, 3, num_channel, 1, input_q=input_q, output_q=output_q,
 469               name='conv_1')
 470     x = _bn(x, is_train=is_training, name='bn_1')
 471     x = _relu(x, name='relu_1')
 472     x = _conv(x, 3, num_channel, 1, input_q=output_q, output_q=output_q,
 473               name='conv_2')
 474     x = _bn(x, is_train=is_training, name='bn_2')
 475     # Merge
 476     x = x + shortcut
 477     x = _relu(x, name='relu_2')
 478   return x

 481 def _conv(x, filter_size, out_channel, stride, pad='SAME', input_q=None,
 482           output_q=None, name='conv'):
 483   """Helper function for defining ResNet architecture."""
 484   if (input_q is None) ^ (output_q is None):
 485     raise ValueError('Input/Output splits are not correctly given.')
 486 
 487   in_shape = x.get_shape()
 488   with tf.variable_scope(name):
 489     # Main operation: conv2d
 490     with tf.device('/CPU:0'):
 491       kernel = tf.get_variable(
 492           'kernel', [filter_size, filter_size, in_shape[3], out_channel],
 493           tf.float32, initializer=tf.random_normal_initializer(
 494               stddev=np.sqrt(2.0/filter_size/filter_size/out_channel)))
 495     if kernel not in tf.get_collection(WEIGHT_DECAY_KEY):
 496       tf.add_to_collection(WEIGHT_DECAY_KEY, kernel)
 497     conv = tf.nn.conv2d(x, kernel, [1, stride, stride, 1], pad)
 498   return conv

 501 def _bn(x, is_train, name='bn'):
 502   """Helper function for defining ResNet architecture."""
 503   bn = tf.layers.batch_normalization(x, training=is_train, name=name)
 504   return bn

 507 def _relu(x, name=None, leakness=0.0):
 508   """Helper function for defining ResNet architecture."""
 509   if leakness > 0.0:
 510     name = 'lrelu' if name is None else name
 511     return tf.maximum(x, x*leakness, name='lrelu')
 512   else:
 513     name = 'relu' if name is None else name
 514     return tf.nn.relu(x, name='relu')

 517 def _resize_like(inputs, ref):
 518   i_h, i_w = inputs.get_shape()[1], inputs.get_shape()[2]
 519   r_h, r_w = ref.get_shape()[1], ref.get_shape()[2]
 520   if i_h == r_h and i_w == r_w:
 521     return inputs
 522   else:
 523     # TODO(casser): Other interpolation methods could be explored here.
 524     return tf.image.resize_bilinear(inputs, [r_h.value, r_w.value],
 525                                     align_corners=True)
