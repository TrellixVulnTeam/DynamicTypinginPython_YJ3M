  17 """Applies online refinement while running inference.
  18 
  19 Instructions: Run static inference first before calling this script. Make sure
  20 to point output_dir to the same folder where static inference results were
  21 saved previously.
  22 
  23 For example use, please refer to README.
  24 """

  44 gfile = tf.gfile
  45 SAVE_EVERY = 1  # Defines the interval that predictions should be saved at.
  46 SAVE_PREVIEWS = True  # If set, while save image previews of depth predictions.
  47 FIXED_SEED = 8964  # Fixed seed for repeatability.
  48 
  49 flags.DEFINE_string('output_dir', None, 'Directory to store predictions. '
  50                     'Assumes that regular inference has been executed before '
  51                     'and results were stored in this folder.')
  52 flags.DEFINE_string('data_dir', None, 'Folder pointing to preprocessed '
  53                     'triplets to fine-tune on.')
  54 flags.DEFINE_string('triplet_list_file', None, 'Text file containing paths to '
  55                     'image files to process. Paths should be relative with '
  56                     'respect to the list file location. Every line should be '
  57                     'of the form [input_folder_name] [input_frame_num] '
  58                     '[output_path], where [output_path] is optional to specify '
  59                     'a different path to store the prediction.')
  60 flags.DEFINE_string('triplet_list_file_remains', None, 'Optional text file '
  61                     'containing relative paths to image files which should not '
  62                     'be fine-tuned, e.g. because of missing adjacent frames. '
  63                     'For all files listed, the static prediction will be '
  64                     'copied instead. File can be empty. If not, every line '
  65                     'should be of the form [input_folder_name] '
  66                     '[input_frame_num] [output_path], where [output_path] is '
  67                     'optional to specify a different path to take and store '
  68                     'the unrefined prediction from/to.')
  69 flags.DEFINE_string('model_ckpt', None, 'Model checkpoint to optimize.')
  70 flags.DEFINE_string('ft_name', '', 'Optional prefix for temporary files.')
  71 flags.DEFINE_string('file_extension', 'png', 'Image data file extension.')
  72 flags.DEFINE_float('learning_rate', 0.0001, 'Adam learning rate.')
  73 flags.DEFINE_float('beta1', 0.9, 'Adam momentum.')
  74 flags.DEFINE_float('reconstr_weight', 0.85, 'Frame reconstruction loss weight.')
  75 flags.DEFINE_float('ssim_weight', 0.15, 'SSIM loss weight.')
  76 flags.DEFINE_float('smooth_weight', 0.01, 'Smoothness loss weight.')
  77 flags.DEFINE_float('icp_weight', 0.0, 'ICP loss weight.')
  78 flags.DEFINE_float('size_constraint_weight', 0.0005, 'Weight of the object '
  79                    'size constraint loss. Use only with motion handling.')
  80 flags.DEFINE_integer('batch_size', 1, 'The size of a sample batch')
  81 flags.DEFINE_integer('img_height', 128, 'Input frame height.')
  82 flags.DEFINE_integer('img_width', 416, 'Input frame width.')
  83 flags.DEFINE_integer('seq_length', 3, 'Number of frames in sequence.')
  84 flags.DEFINE_enum('architecture', nets.RESNET, nets.ARCHITECTURES,
  85                   'Defines the architecture to use for the depth prediction '
  86                   'network. Defaults to ResNet-based encoder and accompanying '
  87                   'decoder.')
  88 flags.DEFINE_boolean('imagenet_norm', True, 'Whether to normalize the input '
  89                      'images channel-wise so that they match the distribution '
  90                      'most ImageNet-models were trained on.')
  91 flags.DEFINE_float('weight_reg', 0.05, 'The amount of weight regularization to '
  92                    'apply. This has no effect on the ResNet-based encoder '
  93                    'architecture.')
  94 flags.DEFINE_boolean('exhaustive_mode', False, 'Whether to exhaustively warp '
  95                      'from any frame to any other instead of just considering '
  96                      'adjacent frames. Where necessary, multiple egomotion '
  97                      'estimates will be applied. Does not have an effect if '
  98                      'compute_minimum_loss is enabled.')
  99 flags.DEFINE_boolean('random_scale_crop', False, 'Whether to apply random '
 100                      'image scaling and center cropping during training.')
 101 flags.DEFINE_bool('depth_upsampling', True, 'Whether to apply depth '
 102                   'upsampling of lower-scale representations before warping to '
 103                   'compute reconstruction loss on full-resolution image.')
 104 flags.DEFINE_bool('depth_normalization', True, 'Whether to apply depth '
 105                   'normalization, that is, normalizing inverse depth '
 106                   'prediction maps by their mean to avoid degeneration towards '
 107                   'small values.')
 108 flags.DEFINE_bool('compute_minimum_loss', True, 'Whether to take the '
 109                   'element-wise minimum of the reconstruction/SSIM error in '
 110                   'order to avoid overly penalizing dis-occlusion effects.')
 111 flags.DEFINE_bool('use_skip', True, 'Whether to use skip connections in the '
 112                   'encoder-decoder architecture.')
 113 flags.DEFINE_bool('joint_encoder', False, 'Whether to share parameters '
 114                   'between the depth and egomotion networks by using a joint '
 115                   'encoder architecture. The egomotion network is then '
 116                   'operating only on the hidden representation provided by the '
 117                   'joint encoder.')
 118 flags.DEFINE_float('egomotion_threshold', 0.01, 'Minimum egomotion magnitude '
 119                    'to apply finetuning. If lower, just forwards the ordinary '
 120                    'prediction.')
 121 flags.DEFINE_integer('num_steps', 20, 'Number of optimization steps to run.')
 122 flags.DEFINE_boolean('handle_motion', True, 'Whether the checkpoint was '
 123                      'trained with motion handling.')
 124 flags.DEFINE_bool('flip', False, 'Whether images should be flipped as well as '
 125                   'resulting predictions (for test-time augmentation). This '
 126                   'currently applies to the depth network only.')
 127 
 128 FLAGS = flags.FLAGS
 129 flags.mark_flag_as_required('output_dir')
 130 flags.mark_flag_as_required('data_dir')
 131 flags.mark_flag_as_required('model_ckpt')
 132 flags.mark_flag_as_required('triplet_list_file')

 135 def main(_):
 136   """Runs fine-tuning and inference.
 137 
 138   There are three categories of images.
 139   1) Images where we have previous and next frame, and that are not filtered
 140      out by the heuristic. For them, we will use the fine-tuned predictions.
 141   2) Images where we have previous and next frame, but that were filtered out
 142      by our heuristic. For them, we will use the ordinary prediction instead.
 143   3) Images where we have at least one missing adjacent frame. For them, we will
 144      use the ordinary prediction as indicated by triplet_list_file_remains (if
 145      provided). They will also not be part of the generated inference list in
 146      the first place.
 147 
 148   Raises:
 149      ValueError: Invalid parameters have been passed.
 150   """
 151 
 152   if FLAGS.handle_motion and FLAGS.joint_encoder:
 153     raise ValueError('Using a joint encoder is currently not supported when '
 154                      'modeling object motion.')
 155   if FLAGS.handle_motion and FLAGS.seq_length != 3:
 156     raise ValueError('The current motion model implementation only supports '
 157                      'using a sequence length of three.')
 158   if FLAGS.handle_motion and not FLAGS.compute_minimum_loss:
 159     raise ValueError('Computing the minimum photometric loss is required when '
 160                      'enabling object motion handling.')
 161   if FLAGS.size_constraint_weight > 0 and not FLAGS.handle_motion:
 162     raise ValueError('To enforce object size constraints, enable motion '
 163                      'handling.')
 164   if FLAGS.icp_weight > 0.0:
 165     raise ValueError('ICP is currently not supported.')
 166   if FLAGS.compute_minimum_loss and FLAGS.seq_length % 2 != 1:
 167     raise ValueError('Compute minimum loss requires using an odd number of '
 168                      'images in a sequence.')
 169   if FLAGS.compute_minimum_loss and FLAGS.exhaustive_mode:
 170     raise ValueError('Exhaustive mode has no effect when compute_minimum_loss '
 171                      'is enabled.')
 172   if FLAGS.img_width % (2 ** 5) != 0 or FLAGS.img_height % (2 ** 5) != 0:
 173     logging.warn('Image size is not divisible by 2^5. For the architecture '
 174                  'employed, this could cause artefacts caused by resizing in '
 175                  'lower dimensions.')
 176 
 177   if FLAGS.output_dir.endswith('/'):
 178     FLAGS.output_dir = FLAGS.output_dir[:-1]
 179 
 180   # Create file lists to prepare fine-tuning, save it to unique_file.
 181   unique_file_name = (str(datetime.datetime.now().date()) + '_' +
 182                       str(datetime.datetime.now().time()).replace(':', '_'))
 183   unique_file = os.path.join(FLAGS.data_dir, unique_file_name + '.txt')
 184   with gfile.FastGFile(FLAGS.triplet_list_file, 'r') as f:
 185     files_to_process = f.readlines()
 186     files_to_process = [line.rstrip() for line in files_to_process]
 187     files_to_process = [line for line in files_to_process if len(line)]
 188   logging.info('Creating unique file list %s with %s entries.', unique_file,
 189                len(files_to_process))
 190   with gfile.FastGFile(unique_file, 'w') as f_out:
 191     fetches_network = FLAGS.num_steps * FLAGS.batch_size
 192     fetches_saves = FLAGS.batch_size * int(np.floor(FLAGS.num_steps/SAVE_EVERY))
 193     repetitions = fetches_network + 3 * fetches_saves
 194     for i in range(len(files_to_process)):
 195       for _ in range(repetitions):
 196         f_out.write(files_to_process[i] + '\n')
 197 
 198   # Read remaining files.
 199   remaining = []
 200   if gfile.Exists(FLAGS.triplet_list_file_remains):
 201     with gfile.FastGFile(FLAGS.triplet_list_file_remains, 'r') as f:
 202       remaining = f.readlines()
 203       remaining = [line.rstrip() for line in remaining]
 204       remaining = [line for line in remaining if len(line)]
 205   logging.info('Running fine-tuning on %s files, %s files are remaining.',
 206                len(files_to_process), len(remaining))
 207 
 208   # Run fine-tuning process and save predictions in id-folders.
 209   tf.set_random_seed(FIXED_SEED)
 210   np.random.seed(FIXED_SEED)
 211   random.seed(FIXED_SEED)
 212   flipping_mode = reader.FLIP_ALWAYS if FLAGS.flip else reader.FLIP_NONE
 213   train_model = model.Model(data_dir=FLAGS.data_dir,
 214                             file_extension=FLAGS.file_extension,
 215                             is_training=True,
 216                             learning_rate=FLAGS.learning_rate,
 217                             beta1=FLAGS.beta1,
 218                             reconstr_weight=FLAGS.reconstr_weight,
 219                             smooth_weight=FLAGS.smooth_weight,
 220                             ssim_weight=FLAGS.ssim_weight,
 221                             icp_weight=FLAGS.icp_weight,
 222                             batch_size=FLAGS.batch_size,
 223                             img_height=FLAGS.img_height,
 224                             img_width=FLAGS.img_width,
 225                             seq_length=FLAGS.seq_length,
 226                             architecture=FLAGS.architecture,
 227                             imagenet_norm=FLAGS.imagenet_norm,
 228                             weight_reg=FLAGS.weight_reg,
 229                             exhaustive_mode=FLAGS.exhaustive_mode,
 230                             random_scale_crop=FLAGS.random_scale_crop,
 231                             flipping_mode=flipping_mode,
 232                             random_color=False,
 233                             depth_upsampling=FLAGS.depth_upsampling,
 234                             depth_normalization=FLAGS.depth_normalization,
 235                             compute_minimum_loss=FLAGS.compute_minimum_loss,
 236                             use_skip=FLAGS.use_skip,
 237                             joint_encoder=FLAGS.joint_encoder,
 238                             build_sum=False,
 239                             shuffle=False,
 240                             input_file=unique_file_name,
 241                             handle_motion=FLAGS.handle_motion,
 242                             size_constraint_weight=FLAGS.size_constraint_weight,
 243                             train_global_scale_var=False)
 244 
 245   failed_heuristic_ids = finetune_inference(train_model, FLAGS.model_ckpt,
 246                                             FLAGS.output_dir + '_ft')
 247   logging.info('Fine-tuning completed, %s files were filtered out by '
 248                'heuristic.', len(failed_heuristic_ids))
 249   for failed_id in failed_heuristic_ids:
 250     failed_entry = files_to_process[failed_id]
 251     remaining.append(failed_entry)
 252   logging.info('In total, %s images were fine-tuned, while %s were not.',
 253                len(files_to_process)-len(failed_heuristic_ids), len(remaining))
 254 
 255   # Copy all results to have the same structural output as running ordinary
 256   # inference.
 257   for i in range(len(files_to_process)):
 258     if files_to_process[i] not in remaining:  # Use fine-tuned result.
 259       elements = files_to_process[i].split(' ')
 260       source_file = os.path.join(FLAGS.output_dir + '_ft', FLAGS.ft_name +
 261                                  'id_' + str(i),
 262                                  str(FLAGS.num_steps).zfill(10) +
 263                                  ('_flip' if FLAGS.flip else ''))
 264       if len(elements) == 2:  # No differing mapping defined.
 265         target_dir = os.path.join(FLAGS.output_dir + '_ft', elements[0])
 266         target_file = os.path.join(
 267             target_dir, elements[1] + ('_flip' if FLAGS.flip else ''))
 268       else:  # Other mapping for file defined, copy to this location instead.
 269         target_dir = os.path.join(
 270             FLAGS.output_dir + '_ft', os.path.dirname(elements[2]))
 271         target_file = os.path.join(
 272             target_dir,
 273             os.path.basename(elements[2]) + ('_flip' if FLAGS.flip else ''))
 274       if not gfile.Exists(target_dir):
 275         gfile.MakeDirs(target_dir)
 276       logging.info('Copy refined result %s to %s.', source_file, target_file)
 277       gfile.Copy(source_file + '.npy', target_file + '.npy', overwrite=True)
 278       gfile.Copy(source_file + '.txt', target_file + '.txt', overwrite=True)
 279       gfile.Copy(source_file + '.%s' % FLAGS.file_extension,
 280                  target_file + '.%s' % FLAGS.file_extension, overwrite=True)
 281   for j in range(len(remaining)):
 282     elements = remaining[j].split(' ')
 283     if len(elements) == 2:  # No differing mapping defined.
 284       target_dir = os.path.join(FLAGS.output_dir + '_ft', elements[0])
 285       target_file = os.path.join(
 286           target_dir, elements[1] + ('_flip' if FLAGS.flip else ''))
 287     else:  # Other mapping for file defined, copy to this location instead.
 288       target_dir = os.path.join(
 289           FLAGS.output_dir + '_ft', os.path.dirname(elements[2]))
 290       target_file = os.path.join(
 291           target_dir,
 292           os.path.basename(elements[2]) + ('_flip' if FLAGS.flip else ''))
 293     if not gfile.Exists(target_dir):
 294       gfile.MakeDirs(target_dir)
 295     source_file = target_file.replace('_ft', '')
 296     logging.info('Copy unrefined result %s to %s.', source_file, target_file)
 297     gfile.Copy(source_file + '.npy', target_file + '.npy', overwrite=True)
 298     gfile.Copy(source_file + '.%s' % FLAGS.file_extension,
 299                target_file + '.%s' % FLAGS.file_extension, overwrite=True)
 300   logging.info('Done, predictions saved in %s.', FLAGS.output_dir + '_ft')
 301

 303 def finetune_inference(train_model, model_ckpt, output_dir):
 304   """Train model."""
 305   vars_to_restore = None
 306   if model_ckpt is not None:
 307     vars_to_restore = util.get_vars_to_save_and_restore(model_ckpt)
 308     ckpt_path = model_ckpt
 309   pretrain_restorer = tf.train.Saver(vars_to_restore)
 310   sv = tf.train.Supervisor(logdir=None, save_summaries_secs=0, saver=None,
 311                            summary_op=None)
 312   config = tf.ConfigProto()
 313   config.gpu_options.allow_growth = True
 314   img_nr = 0
 315   failed_heuristic = []
 316   with sv.managed_session(config=config) as sess:
 317     # TODO(casser): Caching the weights would be better to avoid I/O bottleneck.
 318     while True:  # Loop terminates when all examples have been processed.
 319       if model_ckpt is not None:
 320         logging.info('Restored weights from %s', ckpt_path)
 321         pretrain_restorer.restore(sess, ckpt_path)
 322       logging.info('Running fine-tuning, image %s...', img_nr)
 323       img_pred_folder = os.path.join(
 324           output_dir, FLAGS.ft_name + 'id_' + str(img_nr))
 325       if not gfile.Exists(img_pred_folder):
 326         gfile.MakeDirs(img_pred_folder)
 327       step = 1
 328 
 329       # Run fine-tuning.
 330       while step <= FLAGS.num_steps:
 331         logging.info('Running step %s of %s.', step, FLAGS.num_steps)
 332         fetches = {
 333             'train': train_model.train_op,
 334             'global_step': train_model.global_step,
 335             'incr_global_step': train_model.incr_global_step
 336         }
 337         _ = sess.run(fetches)
 338         if step % SAVE_EVERY == 0:
 339           # Get latest prediction for middle frame, highest scale.
 340           pred = train_model.depth[1][0].eval(session=sess)
 341           if FLAGS.flip:
 342             pred = np.flip(pred, axis=2)
 343           input_img = train_model.image_stack.eval(session=sess)
 344           input_img_prev = input_img[0, :, :, 0:3]
 345           input_img_center = input_img[0, :, :, 3:6]
 346           input_img_next = input_img[0, :, :, 6:]
 347           img_pred_file = os.path.join(
 348               img_pred_folder,
 349               str(step).zfill(10) + ('_flip' if FLAGS.flip else '') + '.npy')
 350           motion = np.squeeze(train_model.egomotion.eval(session=sess))
 351           # motion of shape (seq_length - 1, 6).
 352           motion = np.mean(motion, axis=0)  # Average egomotion across frames.
 353 
 354           if SAVE_PREVIEWS or step == FLAGS.num_steps:
 355             # Also save preview of depth map.
 356             color_map = util.normalize_depth_for_display(
 357                 np.squeeze(pred[0, :, :]))
 358             visualization = np.concatenate(
 359                 (input_img_prev, input_img_center, input_img_next, color_map))
 360             motion_s = [str(m) for m in motion]
 361             s_rep = ','.join(motion_s)
 362             with gfile.Open(img_pred_file.replace('.npy', '.txt'), 'w') as f:
 363               f.write(s_rep)
 364             util.save_image(
 365                 img_pred_file.replace('.npy', '.%s' % FLAGS.file_extension),
 366                 visualization, FLAGS.file_extension)
 367 
 368           with gfile.Open(img_pred_file, 'wb') as f:
 369             np.save(f, pred)
 370 
 371         # Apply heuristic to not finetune if egomotion magnitude is too low.
 372         ego_magnitude = np.linalg.norm(motion[:3], ord=2)
 373         heuristic = ego_magnitude >= FLAGS.egomotion_threshold
 374         if not heuristic and step == FLAGS.num_steps:
 375           failed_heuristic.append(img_nr)
 376 
 377         step += 1
 378       img_nr += 1
 379   return failed_heuristic
