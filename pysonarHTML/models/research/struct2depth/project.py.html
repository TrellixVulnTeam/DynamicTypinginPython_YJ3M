  32 def inverse_warp(img, depth, egomotion_mat, intrinsic_mat,
  33                  intrinsic_mat_inv):
  34   """Inverse warp a source image to the target image plane.
  35 
  36   Args:
  37     img: The source image (to sample pixels from) -- [B, H, W, 3].
  38     depth: Depth map of the target image -- [B, H, W].
  39     egomotion_mat: Matrix defining egomotion transform -- [B, 4, 4].
  40     intrinsic_mat: Camera intrinsic matrix -- [B, 3, 3].
  41     intrinsic_mat_inv: Inverse of the intrinsic matrix -- [B, 3, 3].
  42   Returns:
  43     Projected source image
  44   """
  45   dims = tf.shape(img)
  46   batch_size, img_height, img_width = dims[0], dims[1], dims[2]
  47   depth = tf.reshape(depth, [batch_size, 1, img_height * img_width])
  48   grid = _meshgrid_abs(img_height, img_width)
  49   grid = tf.tile(tf.expand_dims(grid, 0), [batch_size, 1, 1])
  50   cam_coords = _pixel2cam(depth, grid, intrinsic_mat_inv)
  51   ones = tf.ones([batch_size, 1, img_height * img_width])
  52   cam_coords_hom = tf.concat([cam_coords, ones], axis=1)
  53 
  54   # Get projection matrix for target camera frame to source pixel frame
  55   hom_filler = tf.constant([0.0, 0.0, 0.0, 1.0], shape=[1, 1, 4])
  56   hom_filler = tf.tile(hom_filler, [batch_size, 1, 1])
  57   intrinsic_mat_hom = tf.concat(
  58       [intrinsic_mat, tf.zeros([batch_size, 3, 1])], axis=2)
  59   intrinsic_mat_hom = tf.concat([intrinsic_mat_hom, hom_filler], axis=1)
  60   proj_target_cam_to_source_pixel = tf.matmul(intrinsic_mat_hom, egomotion_mat)
  61   source_pixel_coords = _cam2pixel(cam_coords_hom,
  62                                    proj_target_cam_to_source_pixel)
  63   source_pixel_coords = tf.reshape(source_pixel_coords,
  64                                    [batch_size, 2, img_height, img_width])
  65   source_pixel_coords = tf.transpose(source_pixel_coords, perm=[0, 2, 3, 1])
  66   projected_img, mask = _spatial_transformer(img, source_pixel_coords)
  67   return projected_img, mask

  70 def get_transform_mat(egomotion_vecs, i, j):
  71   """Returns a transform matrix defining the transform from frame i to j."""
  72   egomotion_transforms = []
  73   batchsize = tf.shape(egomotion_vecs)[0]
  74   if i == j:
  75     return tf.tile(tf.expand_dims(tf.eye(4, 4), axis=0), [batchsize, 1, 1])
  76   for k in range(min(i, j), max(i, j)):
  77     transform_matrix = _egomotion_vec2mat(egomotion_vecs[:, k, :], batchsize)
  78     if i > j:  # Going back in sequence, need to invert egomotion.
  79       egomotion_transforms.insert(0, tf.linalg.inv(transform_matrix))
  80     else:  # Going forward in sequence
  81       egomotion_transforms.append(transform_matrix)
  82 
  83   # Multiply all matrices.
  84   egomotion_mat = egomotion_transforms[0]
  85   for i in range(1, len(egomotion_transforms)):
  86     egomotion_mat = tf.matmul(egomotion_mat, egomotion_transforms[i])
  87   return egomotion_mat

  90 def _pixel2cam(depth, pixel_coords, intrinsic_mat_inv):
  91   """Transform coordinates in the pixel frame to the camera frame."""
  92   cam_coords = tf.matmul(intrinsic_mat_inv, pixel_coords) * depth
  93   return cam_coords

  96 def _cam2pixel(cam_coords, proj_c2p):
  97   """Transform coordinates in the camera frame to the pixel frame."""
  98   pcoords = tf.matmul(proj_c2p, cam_coords)
  99   x = tf.slice(pcoords, [0, 0, 0], [-1, 1, -1])
 100   y = tf.slice(pcoords, [0, 1, 0], [-1, 1, -1])
 101   z = tf.slice(pcoords, [0, 2, 0], [-1, 1, -1])
 102   # Not tested if adding a small number is necessary
 103   x_norm = x / (z + 1e-10)
 104   y_norm = y / (z + 1e-10)
 105   pixel_coords = tf.concat([x_norm, y_norm], axis=1)
 106   return pixel_coords

 109 def _meshgrid_abs(height, width):
 110   """Meshgrid in the absolute coordinates."""
 111   x_t = tf.matmul(
 112       tf.ones(shape=tf.stack([height, 1])),
 113       tf.transpose(tf.expand_dims(tf.linspace(-1.0, 1.0, width), 1), [1, 0]))
 114   y_t = tf.matmul(
 115       tf.expand_dims(tf.linspace(-1.0, 1.0, height), 1),
 116       tf.ones(shape=tf.stack([1, width])))
 117   x_t = (x_t + 1.0) * 0.5 * tf.cast(width - 1, tf.float32)
 118   y_t = (y_t + 1.0) * 0.5 * tf.cast(height - 1, tf.float32)
 119   x_t_flat = tf.reshape(x_t, (1, -1))
 120   y_t_flat = tf.reshape(y_t, (1, -1))
 121   ones = tf.ones_like(x_t_flat)
 122   grid = tf.concat([x_t_flat, y_t_flat, ones], axis=0)
 123   return grid

 126 def _euler2mat(z, y, x):
 127   """Converts euler angles to rotation matrix.
 128 
 129    From:
 130    https://github.com/pulkitag/pycaffe-utils/blob/master/rot_utils.py#L174
 131 
 132    TODO: Remove the dimension for 'N' (deprecated for converting all source
 133    poses altogether).
 134 
 135   Args:
 136     z: rotation angle along z axis (in radians) -- size = [B, n]
 137     y: rotation angle along y axis (in radians) -- size = [B, n]
 138     x: rotation angle along x axis (in radians) -- size = [B, n]
 139 
 140   Returns:
 141     Rotation matrix corresponding to the euler angles, with shape [B, n, 3, 3].
 142   """
 143   batch_size = tf.shape(z)[0]
 144   n = 1
 145   z = tf.clip_by_value(z, -np.pi, np.pi)
 146   y = tf.clip_by_value(y, -np.pi, np.pi)
 147   x = tf.clip_by_value(x, -np.pi, np.pi)
 148 
 149   # Expand to B x N x 1 x 1
 150   z = tf.expand_dims(tf.expand_dims(z, -1), -1)
 151   y = tf.expand_dims(tf.expand_dims(y, -1), -1)
 152   x = tf.expand_dims(tf.expand_dims(x, -1), -1)
 153 
 154   zeros = tf.zeros([batch_size, n, 1, 1])
 155   ones = tf.ones([batch_size, n, 1, 1])
 156 
 157   cosz = tf.cos(z)
 158   sinz = tf.sin(z)
 159   rotz_1 = tf.concat([cosz, -sinz, zeros], axis=3)
 160   rotz_2 = tf.concat([sinz, cosz, zeros], axis=3)
 161   rotz_3 = tf.concat([zeros, zeros, ones], axis=3)
 162   zmat = tf.concat([rotz_1, rotz_2, rotz_3], axis=2)
 163 
 164   cosy = tf.cos(y)
 165   siny = tf.sin(y)
 166   roty_1 = tf.concat([cosy, zeros, siny], axis=3)
 167   roty_2 = tf.concat([zeros, ones, zeros], axis=3)
 168   roty_3 = tf.concat([-siny, zeros, cosy], axis=3)
 169   ymat = tf.concat([roty_1, roty_2, roty_3], axis=2)
 170 
 171   cosx = tf.cos(x)
 172   sinx = tf.sin(x)
 173   rotx_1 = tf.concat([ones, zeros, zeros], axis=3)
 174   rotx_2 = tf.concat([zeros, cosx, -sinx], axis=3)
 175   rotx_3 = tf.concat([zeros, sinx, cosx], axis=3)
 176   xmat = tf.concat([rotx_1, rotx_2, rotx_3], axis=2)
 177 
 178   return tf.matmul(tf.matmul(xmat, ymat), zmat)

 181 def _egomotion_vec2mat(vec, batch_size):
 182   """Converts 6DoF transform vector to transformation matrix.
 183 
 184   Args:
 185     vec: 6DoF parameters [tx, ty, tz, rx, ry, rz] -- [B, 6].
 186     batch_size: Batch size.
 187 
 188   Returns:
 189     A transformation matrix -- [B, 4, 4].
 190   """
 191   translation = tf.slice(vec, [0, 0], [-1, 3])
 192   translation = tf.expand_dims(translation, -1)
 193   rx = tf.slice(vec, [0, 3], [-1, 1])
 194   ry = tf.slice(vec, [0, 4], [-1, 1])
 195   rz = tf.slice(vec, [0, 5], [-1, 1])
 196   rot_mat = _euler2mat(rz, ry, rx)
 197   rot_mat = tf.squeeze(rot_mat, squeeze_dims=[1])
 198   filler = tf.constant([0.0, 0.0, 0.0, 1.0], shape=[1, 1, 4])
 199   filler = tf.tile(filler, [batch_size, 1, 1])
 200   transform_mat = tf.concat([rot_mat, translation], axis=2)
 201   transform_mat = tf.concat([transform_mat, filler], axis=1)
 202   return transform_mat

 205 def _bilinear_sampler(im, x, y, name='blinear_sampler'):
 206   """Perform bilinear sampling on im given list of x, y coordinates.
 207 
 208   Implements the differentiable sampling mechanism with bilinear kernel
 209   in https://arxiv.org/abs/1506.02025.
 210 
 211   x,y are tensors specifying normalized coordinates [-1, 1] to be sampled on im.
 212   For example, (-1, -1) in (x, y) corresponds to pixel location (0, 0) in im,
 213   and (1, 1) in (x, y) corresponds to the bottom right pixel in im.
 214 
 215   Args:
 216     im: Batch of images with shape [B, h, w, channels].
 217     x: Tensor of normalized x coordinates in [-1, 1], with shape [B, h, w, 1].
 218     y: Tensor of normalized y coordinates in [-1, 1], with shape [B, h, w, 1].
 219     name: Name scope for ops.
 220 
 221   Returns:
 222     Sampled image with shape [B, h, w, channels].
 223     Principled mask with shape [B, h, w, 1], dtype:float32.  A value of 1.0
 224       in the mask indicates that the corresponding coordinate in the sampled
 225       image is valid.
 226   """
 227   with tf.variable_scope(name):
 228     x = tf.reshape(x, [-1])
 229     y = tf.reshape(y, [-1])
 230 
 231     # Constants.
 232     batch_size = tf.shape(im)[0]
 233     _, height, width, channels = im.get_shape().as_list()
 234 
 235     x = tf.to_float(x)
 236     y = tf.to_float(y)
 237     height_f = tf.cast(height, 'float32')
 238     width_f = tf.cast(width, 'float32')
 239     zero = tf.constant(0, dtype=tf.int32)
 240     max_y = tf.cast(tf.shape(im)[1] - 1, 'int32')
 241     max_x = tf.cast(tf.shape(im)[2] - 1, 'int32')
 242 
 243     # Scale indices from [-1, 1] to [0, width - 1] or [0, height - 1].
 244     x = (x + 1.0) * (width_f - 1.0) / 2.0
 245     y = (y + 1.0) * (height_f - 1.0) / 2.0
 246 
 247     # Compute the coordinates of the 4 pixels to sample from.
 248     x0 = tf.cast(tf.floor(x), 'int32')
 249     x1 = x0 + 1
 250     y0 = tf.cast(tf.floor(y), 'int32')
 251     y1 = y0 + 1
 252 
 253     mask = tf.logical_and(
 254         tf.logical_and(x0 >= zero, x1 <= max_x),
 255         tf.logical_and(y0 >= zero, y1 <= max_y))
 256     mask = tf.to_float(mask)
 257 
 258     x0 = tf.clip_by_value(x0, zero, max_x)
 259     x1 = tf.clip_by_value(x1, zero, max_x)
 260     y0 = tf.clip_by_value(y0, zero, max_y)
 261     y1 = tf.clip_by_value(y1, zero, max_y)
 262     dim2 = width
 263     dim1 = width * height
 264 
 265     # Create base index.
 266     base = tf.range(batch_size) * dim1
 267     base = tf.reshape(base, [-1, 1])
 268     base = tf.tile(base, [1, height * width])
 269     base = tf.reshape(base, [-1])
 270 
 271     base_y0 = base + y0 * dim2
 272     base_y1 = base + y1 * dim2
 273     idx_a = base_y0 + x0
 274     idx_b = base_y1 + x0
 275     idx_c = base_y0 + x1
 276     idx_d = base_y1 + x1
 277 
 278     # Use indices to lookup pixels in the flat image and restore channels dim.
 279     im_flat = tf.reshape(im, tf.stack([-1, channels]))
 280     im_flat = tf.to_float(im_flat)
 281     pixel_a = tf.gather(im_flat, idx_a)
 282     pixel_b = tf.gather(im_flat, idx_b)
 283     pixel_c = tf.gather(im_flat, idx_c)
 284     pixel_d = tf.gather(im_flat, idx_d)
 285 
 286     x1_f = tf.to_float(x1)
 287     y1_f = tf.to_float(y1)
 288 
 289     # And finally calculate interpolated values.
 290     wa = tf.expand_dims(((x1_f - x) * (y1_f - y)), 1)
 291     wb = tf.expand_dims((x1_f - x) * (1.0 - (y1_f - y)), 1)
 292     wc = tf.expand_dims(((1.0 - (x1_f - x)) * (y1_f - y)), 1)
 293     wd = tf.expand_dims(((1.0 - (x1_f - x)) * (1.0 - (y1_f - y))), 1)
 294 
 295     output = tf.add_n([wa * pixel_a, wb * pixel_b, wc * pixel_c, wd * pixel_d])
 296     output = tf.reshape(output, tf.stack([batch_size, height, width, channels]))
 297     mask = tf.reshape(mask, tf.stack([batch_size, height, width, 1]))
 298     return output, mask

 301 def _spatial_transformer(img, coords):
 302   """A wrapper over binlinear_sampler(), taking absolute coords as input."""
 303   img_height = tf.cast(tf.shape(img)[1], tf.float32)
 304   img_width = tf.cast(tf.shape(img)[2], tf.float32)
 305   px = coords[:, :, :, :1]
 306   py = coords[:, :, :, 1:]
 307   # Normalize coordinates to [-1, 1] to send to _bilinear_sampler.
 308   px = px / (img_width - 1) * 2.0 - 1.0
 309   py = py / (img_height - 1) * 2.0 - 1.0
 310   output_img, mask = _bilinear_sampler(img, px, py)
 311   return output_img, mask

 314 def get_cloud(depth, intrinsics_inv, name=None):
 315   """Convert depth map to 3D point cloud."""
 316   with tf.name_scope(name):
 317     dims = depth.shape.as_list()
 318     batch_size, img_height, img_width = dims[0], dims[1], dims[2]
 319     depth = tf.reshape(depth, [batch_size, 1, img_height * img_width])
 320     grid = _meshgrid_abs(img_height, img_width)
 321     grid = tf.tile(tf.expand_dims(grid, 0), [batch_size, 1, 1])
 322     cam_coords = _pixel2cam(depth, grid, intrinsics_inv)
 323     cam_coords = tf.transpose(cam_coords, [0, 2, 1])
 324     cam_coords = tf.reshape(cam_coords, [batch_size, img_height, img_width, 3])
 325     logging.info('depth -> cloud: %s', cam_coords)
 326     return cam_coords
