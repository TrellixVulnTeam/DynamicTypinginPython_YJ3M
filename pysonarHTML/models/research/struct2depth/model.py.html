 175   def build_inference_for_training(self):
 176     """Invokes depth and ego-motion networks and computes clouds if needed."""
 177     (self.image_stack, self.image_stack_norm, self.seg_stack,
 178      self.intrinsic_mat, self.intrinsic_mat_inv) = self.reader.read_data()
 179     with tf.variable_scope('depth_prediction'):
 180       # Organized by ...[i][scale].  Note that the order is flipped in
 181       # variables in build_loss() below.
 182       self.disp = {}
 183       self.depth = {}
 184       self.depth_upsampled = {}
 185       self.inf_loss = 0.0
 186       # Organized by [i].
 187       disp_bottlenecks = [None] * self.seq_length
 188 
 189       if self.icp_weight > 0:
 190         self.cloud = {}
 191       for i in range(self.seq_length):
 192         image = self.image_stack_norm[:, :, :, 3 * i:3 * (i + 1)]
 193 
 194         multiscale_disps_i, disp_bottlenecks[i] = nets.disp_net(
 195             self.architecture, image, self.use_skip,
 196             self.weight_reg, True)
 197         multiscale_depths_i = [1.0 / d for d in multiscale_disps_i]
 198         self.disp[i] = multiscale_disps_i
 199         self.depth[i] = multiscale_depths_i
 200         if self.depth_upsampling:
 201           self.depth_upsampled[i] = []
 202           # Upsample low-resolution depth maps using differentiable bilinear
 203           # interpolation.
 204           for s in range(len(multiscale_depths_i)):
 205             self.depth_upsampled[i].append(tf.image.resize_bilinear(
 206                 multiscale_depths_i[s], [self.img_height, self.img_width],
 207                 align_corners=True))
 208 
 209         if self.icp_weight > 0:
 210           multiscale_clouds_i = [
 211               project.get_cloud(d,
 212                                 self.intrinsic_mat_inv[:, s, :, :],
 213                                 name='cloud%d_%d' % (s, i))
 214               for (s, d) in enumerate(multiscale_depths_i)
 215           ]
 216           self.cloud[i] = multiscale_clouds_i
 217         # Reuse the same depth graph for all images.
 218         tf.get_variable_scope().reuse_variables()
 219 
 220     if self.handle_motion:
 221       # Define egomotion network. This network can see the whole scene except
 222       # for any moving objects as indicated by the provided segmentation masks.
 223       # To avoid the network getting clues of motion by tracking those masks, we
 224       # define the segmentation masks as the union temporally.
 225       with tf.variable_scope('egomotion_prediction'):
 226         base_input = self.image_stack_norm  # (B, H, W, 9)
 227         seg_input = self.seg_stack  # (B, H, W, 9)
 228         ref_zero = tf.constant(0, dtype=tf.uint8)
 229         # Motion model is currently defined for three-frame sequences.
 230         object_mask1 = tf.equal(seg_input[:, :, :, 0], ref_zero)
 231         object_mask2 = tf.equal(seg_input[:, :, :, 3], ref_zero)
 232         object_mask3 = tf.equal(seg_input[:, :, :, 6], ref_zero)
 233         mask_complete = tf.expand_dims(tf.logical_and(  # (B, H, W, 1)
 234             tf.logical_and(object_mask1, object_mask2), object_mask3), axis=3)
 235         mask_complete = tf.tile(mask_complete, (1, 1, 1, 9))  # (B, H, W, 9)
 236         # Now mask out base_input.
 237         self.mask_complete = tf.to_float(mask_complete)
 238         self.base_input_masked = base_input * self.mask_complete
 239         self.egomotion = nets.egomotion_net(
 240             image_stack=self.base_input_masked,
 241             disp_bottleneck_stack=None,
 242             joint_encoder=False,
 243             seq_length=self.seq_length,
 244             weight_reg=self.weight_reg)
 245 
 246       # Define object motion network for refinement. This network only sees
 247       # one object at a time over the whole sequence, and tries to estimate its
 248       # motion. The sequence of images are the respective warped frames.
 249 
 250       # For each scale, contains batch_size elements of shape (N, 2, 6).
 251       self.object_transforms = {}
 252       # For each scale, contains batch_size elements of shape (N, H, W, 9).
 253       self.object_masks = {}
 254       self.object_masks_warped = {}
 255       # For each scale, contains batch_size elements of size N.
 256       self.object_ids = {}
 257 
 258       self.egomotions_seq = {}
 259       self.warped_seq = {}
 260       self.inputs_objectmotion_net = {}
 261       with tf.variable_scope('objectmotion_prediction'):
 262         # First, warp raw images according to overall egomotion.
 263         for s in range(NUM_SCALES):
 264           self.warped_seq[s] = []
 265           self.egomotions_seq[s] = []
 266           for source_index in range(self.seq_length):
 267             egomotion_mat_i_1 = project.get_transform_mat(
 268                 self.egomotion, source_index, 1)
 269             warped_image_i_1, _ = (
 270                 project.inverse_warp(
 271                     self.image_stack[
 272                         :, :, :, source_index*3:(source_index+1)*3],
 273                     self.depth_upsampled[1][s],
 274                     egomotion_mat_i_1,
 275                     self.intrinsic_mat[:, 0, :, :],
 276                     self.intrinsic_mat_inv[:, 0, :, :]))
 277 
 278             self.warped_seq[s].append(warped_image_i_1)
 279             self.egomotions_seq[s].append(egomotion_mat_i_1)
 280 
 281           # Second, for every object in the segmentation mask, take its mask and
 282           # warp it according to the egomotion estimate. Then put a threshold to
 283           # binarize the warped result. Use this mask to mask out background and
 284           # other objects, and pass the filtered image to the object motion
 285           # network.
 286           self.object_transforms[s] = []
 287           self.object_masks[s] = []
 288           self.object_ids[s] = []
 289           self.object_masks_warped[s] = []
 290           self.inputs_objectmotion_net[s] = {}
 291 
 292           for i in range(self.batch_size):
 293             seg_sequence = self.seg_stack[i]  # (H, W, 9=3*3)
 294             object_ids = tf.unique(tf.reshape(seg_sequence, [-1]))[0]
 295             self.object_ids[s].append(object_ids)
 296             color_stack = []
 297             mask_stack = []
 298             mask_stack_warped = []
 299             for j in range(self.seq_length):
 300               current_image = self.warped_seq[s][j][i]  # (H, W, 3)
 301               current_seg = seg_sequence[:, :, j * 3:(j+1) * 3]  # (H, W, 3)
 302 
 303               def process_obj_mask_warp(obj_id):
 304                 """Performs warping of the individual object masks."""
 305                 obj_mask = tf.to_float(tf.equal(current_seg, obj_id))
 306                 # Warp obj_mask according to overall egomotion.
 307                 obj_mask_warped, _ = (
 308                     project.inverse_warp(
 309                         tf.expand_dims(obj_mask, axis=0),
 310                         # Middle frame, highest scale, batch element i:
 311                         tf.expand_dims(self.depth_upsampled[1][s][i], axis=0),
 312                         # Matrix for warping j into middle frame, batch elem. i:
 313                         tf.expand_dims(self.egomotions_seq[s][j][i], axis=0),
 314                         tf.expand_dims(self.intrinsic_mat[i, 0, :, :], axis=0),
 315                         tf.expand_dims(self.intrinsic_mat_inv[i, 0, :, :],
 316                                        axis=0)))
 317                 obj_mask_warped = tf.squeeze(obj_mask_warped)
 318                 obj_mask_binarized = tf.greater(  # Threshold to binarize mask.
 319                     obj_mask_warped, tf.constant(0.5))
 320                 return tf.to_float(obj_mask_binarized)
 321 
 322               def process_obj_mask(obj_id):
 323                 """Returns the individual object masks separately."""
 324                 return tf.to_float(tf.equal(current_seg, obj_id))
 325               object_masks = tf.map_fn(  # (N, H, W, 3)
 326                   process_obj_mask, object_ids, dtype=tf.float32)
 327 
 328               if self.size_constraint_weight > 0:
 329                 # The object segmentation masks are all in object_masks.
 330                 # We need to measure the height of every of them, and get the
 331                 # approximate distance.
 332 
 333                 # self.depth_upsampled of shape (seq_length, scale, B, H, W).
 334                 depth_pred = self.depth_upsampled[j][s][i]  # (H, W)
 335                 def get_losses(obj_mask):
 336                   """Get motion constraint loss."""
 337                   # Find height of segment.
 338                   coords = tf.where(tf.greater(  # Shape (num_true, 2=yx)
 339                       obj_mask[:, :, 0], tf.constant(0.5, dtype=tf.float32)))
 340                   y_max = tf.reduce_max(coords[:, 0])
 341                   y_min = tf.reduce_min(coords[:, 0])
 342                   seg_height = y_max - y_min
 343                   f_y = self.intrinsic_mat[i, 0, 1, 1]
 344                   approx_depth = ((f_y * self.global_scale_var) /
 345                                   tf.to_float(seg_height))
 346                   reference_pred = tf.boolean_mask(
 347                       depth_pred, tf.greater(
 348                           tf.reshape(obj_mask[:, :, 0],
 349                                      (self.img_height, self.img_width, 1)),
 350                           tf.constant(0.5, dtype=tf.float32)))
 351 
 352                   # Establish loss on approx_depth, a scalar, and
 353                   # reference_pred, our dense prediction. Normalize both to
 354                   # prevent degenerative depth shrinking.
 355                   global_mean_depth_pred = tf.reduce_mean(depth_pred)
 356                   reference_pred /= global_mean_depth_pred
 357                   approx_depth /= global_mean_depth_pred
 358                   spatial_err = tf.abs(reference_pred - approx_depth)
 359                   mean_spatial_err = tf.reduce_mean(spatial_err)
 360                   return mean_spatial_err
 361 
 362                 losses = tf.map_fn(
 363                     get_losses, object_masks, dtype=tf.float32)
 364                 self.inf_loss += tf.reduce_mean(losses)
 365               object_masks_warped = tf.map_fn(  # (N, H, W, 3)
 366                   process_obj_mask_warp, object_ids, dtype=tf.float32)
 367               filtered_images = tf.map_fn(
 368                   lambda mask: current_image * mask, object_masks_warped,
 369                   dtype=tf.float32)  # (N, H, W, 3)
 370               color_stack.append(filtered_images)
 371               mask_stack.append(object_masks)
 372               mask_stack_warped.append(object_masks_warped)
 373 
 374             # For this batch-element, if there are N moving objects,
 375             # color_stack, mask_stack and mask_stack_warped contain both
 376             # seq_length elements of shape (N, H, W, 3).
 377             # We can now concatenate them on the last axis, creating a tensor of
 378             # (N, H, W, 3*3 = 9), and, assuming N does not get too large so that
 379             # we have enough memory, pass them in a single batch to the object
 380             # motion network.
 381             mask_stack = tf.concat(mask_stack, axis=3)  # (N, H, W, 9)
 382             mask_stack_warped = tf.concat(mask_stack_warped, axis=3)
 383             color_stack = tf.concat(color_stack, axis=3)  # (N, H, W, 9)
 384             all_transforms = nets.objectmotion_net(
 385                 # We cut the gradient flow here as the object motion gradient
 386                 # should have no saying in how the egomotion network behaves.
 387                 # One could try just stopping the gradient for egomotion, but
 388                 # not for the depth prediction network.
 389                 image_stack=tf.stop_gradient(color_stack),
 390                 disp_bottleneck_stack=None,
 391                 joint_encoder=False,  # Joint encoder not supported.
 392                 seq_length=self.seq_length,
 393                 weight_reg=self.weight_reg)
 394             # all_transforms of shape (N, 2, 6).
 395             self.object_transforms[s].append(all_transforms)
 396             self.object_masks[s].append(mask_stack)
 397             self.object_masks_warped[s].append(mask_stack_warped)
 398             self.inputs_objectmotion_net[s][i] = color_stack
 399             tf.get_variable_scope().reuse_variables()
 400     else:
 401       # Don't handle motion, classic model formulation.
 402       with tf.name_scope('egomotion_prediction'):
 403         if self.joint_encoder:
 404           # Re-arrange disp_bottleneck_stack to be of shape
 405           # [B, h_hid, w_hid, c_hid * seq_length]. Currently, it is a list with
 406           # seq_length elements, each of dimension [B, h_hid, w_hid, c_hid].
 407           disp_bottleneck_stack = tf.concat(disp_bottlenecks, axis=3)
 408         else:
 409           disp_bottleneck_stack = None
 410         self.egomotion = nets.egomotion_net(
 411             image_stack=self.image_stack_norm,
 412             disp_bottleneck_stack=disp_bottleneck_stack,
 413             joint_encoder=self.joint_encoder,
 414             seq_length=self.seq_length,
 415             weight_reg=self.weight_reg)
 416 
 417   def build_loss(self):
 418     """Adds ops for computing loss."""
 419     with tf.name_scope('compute_loss'):
 420       self.reconstr_loss = 0
 421       self.smooth_loss = 0
 422       self.ssim_loss = 0
 423       self.icp_transform_loss = 0
 424       self.icp_residual_loss = 0
 425 
 426       # self.images is organized by ...[scale][B, h, w, seq_len * 3].
 427       self.images = [None for _ in range(NUM_SCALES)]
 428       # Following nested lists are organized by ...[scale][source-target].
 429       self.warped_image = [{} for _ in range(NUM_SCALES)]
 430       self.warp_mask = [{} for _ in range(NUM_SCALES)]
 431       self.warp_error = [{} for _ in range(NUM_SCALES)]
 432       self.ssim_error = [{} for _ in range(NUM_SCALES)]
 433       self.icp_transform = [{} for _ in range(NUM_SCALES)]
 434       self.icp_residual = [{} for _ in range(NUM_SCALES)]
 435 
 436       self.middle_frame_index = util.get_seq_middle(self.seq_length)
 437 
 438       # Compute losses at each scale.
 439       for s in range(NUM_SCALES):
 440         # Scale image stack.
 441         if s == 0:  # Just as a precaution. TF often has interpolation bugs.
 442           self.images[s] = self.image_stack
 443         else:
 444           height_s = int(self.img_height / (2**s))
 445           width_s = int(self.img_width / (2**s))
 446           self.images[s] = tf.image.resize_bilinear(
 447               self.image_stack, [height_s, width_s], align_corners=True)
 448 
 449         # Smoothness.
 450         if self.smooth_weight > 0:
 451           for i in range(self.seq_length):
 452             # When computing minimum loss, use the depth map from the middle
 453             # frame only.
 454             if not self.compute_minimum_loss or i == self.middle_frame_index:
 455               disp_smoothing = self.disp[i][s]
 456               if self.depth_normalization:
 457                 # Perform depth normalization, dividing by the mean.
 458                 mean_disp = tf.reduce_mean(disp_smoothing, axis=[1, 2, 3],
 459                                            keep_dims=True)
 460                 disp_input = disp_smoothing / mean_disp
 461               else:
 462                 disp_input = disp_smoothing
 463               scaling_f = (1.0 if self.equal_weighting else 1.0 / (2**s))
 464               self.smooth_loss += scaling_f * self.depth_smoothness(
 465                   disp_input, self.images[s][:, :, :, 3 * i:3 * (i + 1)])
 466 
 467         self.debug_all_warped_image_batches = []
 468         for i in range(self.seq_length):
 469           for j in range(self.seq_length):
 470             if i == j:
 471               continue
 472 
 473             # When computing minimum loss, only consider the middle frame as
 474             # target.
 475             if self.compute_minimum_loss and j != self.middle_frame_index:
 476               continue
 477             # We only consider adjacent frames, unless either
 478             # compute_minimum_loss is on (where the middle frame is matched with
 479             # all other frames) or exhaustive_mode is on (where all frames are
 480             # matched with each other).
 481             if (not self.compute_minimum_loss and not self.exhaustive_mode and
 482                 abs(i - j) != 1):
 483               continue
 484 
 485             selected_scale = 0 if self.depth_upsampling else s
 486             source = self.images[selected_scale][:, :, :, 3 * i:3 * (i + 1)]
 487             target = self.images[selected_scale][:, :, :, 3 * j:3 * (j + 1)]
 488 
 489             if self.depth_upsampling:
 490               target_depth = self.depth_upsampled[j][s]
 491             else:
 492               target_depth = self.depth[j][s]
 493 
 494             key = '%d-%d' % (i, j)
 495 
 496             if self.handle_motion:
 497               # self.seg_stack of shape (B, H, W, 9).
 498               # target_depth corresponds to middle frame, of shape (B, H, W, 1).
 499 
 500               # Now incorporate the other warping results, performed according
 501               # to the object motion network's predictions.
 502               # self.object_masks batch_size elements of (N, H, W, 9).
 503               # self.object_masks_warped batch_size elements of (N, H, W, 9).
 504               # self.object_transforms batch_size elements of (N, 2, 6).
 505               self.all_batches = []
 506               for batch_s in range(self.batch_size):
 507                 # To warp i into j, first take the base warping (this is the
 508                 # full image i warped into j using only the egomotion estimate).
 509                 base_warping = self.warped_seq[s][i][batch_s]
 510                 transform_matrices_thisbatch = tf.map_fn(
 511                     lambda transform: project.get_transform_mat(
 512                         tf.expand_dims(transform, axis=0), i, j)[0],
 513                     self.object_transforms[0][batch_s])
 514 
 515                 def inverse_warp_wrapper(matrix):
 516                   """Wrapper for inverse warping method."""
 517                   warp_image, _ = (
 518                       project.inverse_warp(
 519                           tf.expand_dims(base_warping, axis=0),
 520                           tf.expand_dims(target_depth[batch_s], axis=0),
 521                           tf.expand_dims(matrix, axis=0),
 522                           tf.expand_dims(self.intrinsic_mat[
 523                               batch_s, selected_scale, :, :], axis=0),
 524                           tf.expand_dims(self.intrinsic_mat_inv[
 525                               batch_s, selected_scale, :, :], axis=0)))
 526                   return warp_image
 527                 warped_images_thisbatch = tf.map_fn(
 528                     inverse_warp_wrapper, transform_matrices_thisbatch,
 529                     dtype=tf.float32)
 530                 warped_images_thisbatch = warped_images_thisbatch[:, 0, :, :, :]
 531                 # warped_images_thisbatch is now of shape (N, H, W, 9).
 532 
 533                 # Combine warped frames into a single one, using the object
 534                 # masks. Result should be (1, 128, 416, 3).
 535                 # Essentially, we here want to sum them all up, filtered by the
 536                 # respective object masks.
 537                 mask_base_valid_source = tf.equal(
 538                     self.seg_stack[batch_s, :, :, i*3:(i+1)*3],
 539                     tf.constant(0, dtype=tf.uint8))
 540                 mask_base_valid_target = tf.equal(
 541                     self.seg_stack[batch_s, :, :, j*3:(j+1)*3],
 542                     tf.constant(0, dtype=tf.uint8))
 543                 mask_valid = tf.logical_and(
 544                     mask_base_valid_source, mask_base_valid_target)
 545                 self.base_warping = base_warping * tf.to_float(mask_valid)
 546                 background = tf.expand_dims(self.base_warping, axis=0)
 547                 def construct_const_filter_tensor(obj_id):
 548                   return tf.fill(
 549                       dims=[self.img_height, self.img_width, 3],
 550                       value=tf.sign(obj_id)) * tf.to_float(
 551                           tf.equal(self.seg_stack[batch_s, :, :, 3:6],
 552                                    tf.cast(obj_id, dtype=tf.uint8)))
 553                 filter_tensor = tf.map_fn(
 554                     construct_const_filter_tensor,
 555                     tf.to_float(self.object_ids[s][batch_s]))
 556                 filter_tensor = tf.stack(filter_tensor, axis=0)
 557                 objects_to_add = tf.reduce_sum(
 558                     tf.multiply(warped_images_thisbatch, filter_tensor),
 559                     axis=0, keepdims=True)
 560                 combined = background + objects_to_add
 561                 self.all_batches.append(combined)
 562                # Now of shape (B, 128, 416, 3).
 563               self.warped_image[s][key] = tf.concat(self.all_batches, axis=0)
 564 
 565             else:
 566               # Don't handle motion, classic model formulation.
 567               egomotion_mat_i_j = project.get_transform_mat(
 568                   self.egomotion, i, j)
 569               # Inverse warp the source image to the target image frame for
 570               # photometric consistency loss.
 571               self.warped_image[s][key], self.warp_mask[s][key] = (
 572                   project.inverse_warp(
 573                       source,
 574                       target_depth,
 575                       egomotion_mat_i_j,
 576                       self.intrinsic_mat[:, selected_scale, :, :],
 577                       self.intrinsic_mat_inv[:, selected_scale, :, :]))
 578 
 579             # Reconstruction loss.
 580             self.warp_error[s][key] = tf.abs(self.warped_image[s][key] - target)
 581             if not self.compute_minimum_loss:
 582               self.reconstr_loss += tf.reduce_mean(
 583                   self.warp_error[s][key] * self.warp_mask[s][key])
 584             # SSIM.
 585             if self.ssim_weight > 0:
 586               self.ssim_error[s][key] = self.ssim(self.warped_image[s][key],
 587                                                   target)
 588               # TODO(rezama): This should be min_pool2d().
 589               if not self.compute_minimum_loss:
 590                 ssim_mask = slim.avg_pool2d(self.warp_mask[s][key], 3, 1,
 591                                             'VALID')
 592                 self.ssim_loss += tf.reduce_mean(
 593                     self.ssim_error[s][key] * ssim_mask)
 594 
 595         # If the minimum loss should be computed, the loss calculation has been
 596         # postponed until here.
 597         if self.compute_minimum_loss:
 598           for frame_index in range(self.middle_frame_index):
 599             key1 = '%d-%d' % (frame_index, self.middle_frame_index)
 600             key2 = '%d-%d' % (self.seq_length - frame_index - 1,
 601                               self.middle_frame_index)
 602             logging.info('computing min error between %s and %s', key1, key2)
 603             min_error = tf.minimum(self.warp_error[s][key1],
 604                                    self.warp_error[s][key2])
 605             self.reconstr_loss += tf.reduce_mean(min_error)
 606             if self.ssim_weight > 0:  # Also compute the minimum SSIM loss.
 607               min_error_ssim = tf.minimum(self.ssim_error[s][key1],
 608                                           self.ssim_error[s][key2])
 609               self.ssim_loss += tf.reduce_mean(min_error_ssim)
 610 
 611       # Build the total loss as composed of L1 reconstruction, SSIM, smoothing
 612       # and object size constraint loss as appropriate.
 613       self.reconstr_loss *= self.reconstr_weight
 614       self.total_loss = self.reconstr_loss
 615       if self.smooth_weight > 0:
 616         self.smooth_loss *= self.smooth_weight
 617         self.total_loss += self.smooth_loss
 618       if self.ssim_weight > 0:
 619         self.ssim_loss *= self.ssim_weight
 620         self.total_loss += self.ssim_loss
 621       if self.size_constraint_weight > 0:
 622         self.inf_loss *= self.size_constraint_weight
 623         self.total_loss += self.inf_loss

 625   def gradient_x(self, img):
 626     return img[:, :, :-1, :] - img[:, :, 1:, :]
 627 
 628   def gradient_y(self, img):
 629     return img[:, :-1, :, :] - img[:, 1:, :, :]

 631   def depth_smoothness(self, depth, img):
 632     """Computes image-aware depth smoothness loss."""
 633     depth_dx = self.gradient_x(depth)
 634     depth_dy = self.gradient_y(depth)
 635     image_dx = self.gradient_x(img)
 636     image_dy = self.gradient_y(img)
 637     weights_x = tf.exp(-tf.reduce_mean(tf.abs(image_dx), 3, keepdims=True))
 638     weights_y = tf.exp(-tf.reduce_mean(tf.abs(image_dy), 3, keepdims=True))
 639     smoothness_x = depth_dx * weights_x
 640     smoothness_y = depth_dy * weights_y
 641     return tf.reduce_mean(abs(smoothness_x)) + tf.reduce_mean(abs(smoothness_y))

 643   def ssim(self, x, y):
 644     """Computes a differentiable structured image similarity measure."""
 645     c1 = 0.01**2  # As defined in SSIM to stabilize div. by small denominator.
 646     c2 = 0.03**2
 647     mu_x = slim.avg_pool2d(x, 3, 1, 'VALID')
 648     mu_y = slim.avg_pool2d(y, 3, 1, 'VALID')
 649     sigma_x = slim.avg_pool2d(x**2, 3, 1, 'VALID') - mu_x**2
 650     sigma_y = slim.avg_pool2d(y**2, 3, 1, 'VALID') - mu_y**2
 651     sigma_xy = slim.avg_pool2d(x * y, 3, 1, 'VALID') - mu_x * mu_y
 652     ssim_n = (2 * mu_x * mu_y + c1) * (2 * sigma_xy + c2)
 653     ssim_d = (mu_x**2 + mu_y**2 + c1) * (sigma_x + sigma_y + c2)
 654     ssim = ssim_n / ssim_d
 655     return tf.clip_by_value((1 - ssim) / 2, 0, 1)

 775   def build_egomotion_test_graph(self):
 776     """Builds egomotion model reading from placeholders."""
 777     input_image_stack = tf.placeholder(
 778         tf.float32,
 779         [1, self.img_height, self.img_width, self.seq_length * 3],
 780         name='raw_input')
 781     input_bottleneck_stack = None
 782 
 783     if self.imagenet_norm:
 784       im_mean = tf.tile(
 785           tf.constant(reader.IMAGENET_MEAN), multiples=[self.seq_length])
 786       im_sd = tf.tile(
 787           tf.constant(reader.IMAGENET_SD), multiples=[self.seq_length])

 799               target_image=input_image,
 800               weight_reg=self.weight_reg,
 801               is_training=True)

 803         input_bottleneck_stack = tf.concat(input_bottleneck_stack, axis=3)

 806       est_egomotion = nets.egomotion_net(
 807           image_stack=input_image_stack,
 808           disp_bottleneck_stack=input_bottleneck_stack,
