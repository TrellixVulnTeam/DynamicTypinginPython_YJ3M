  28 def generate_rnn(rng, N, g, tau, dt, max_firing_rate):
  29   """Create a (vanilla) RNN with a bunch of hyper parameters for generating
  30 chaotic data.
  31   Args:
  32     rng: numpy random number generator
  33     N: number of hidden units
  34     g: scaling of recurrent weight matrix in g W, with W ~ N(0,1/N)
  35     tau: time scale of individual unit dynamics
  36     dt: time step for equation updates
  37     max_firing_rate: how to resecale the -1,1 firing rates
  38   Returns:
  39     the dictionary of these parameters, plus some others.
  40 """
  41   rnn = {}
  42   rnn['N'] = N
  43   rnn['W'] = rng.randn(N,N)/np.sqrt(N)
  44   rnn['Bin'] = rng.randn(N)/np.sqrt(1.0)
  45   rnn['Bin2'] = rng.randn(N)/np.sqrt(1.0)
  46   rnn['b'] = np.zeros(N)
  47   rnn['g'] = g
  48   rnn['tau'] = tau
  49   rnn['dt'] = dt
  50   rnn['max_firing_rate'] = max_firing_rate
  51   mfr = rnn['max_firing_rate']                   # spikes / sec
  52   nbins_per_sec = 1.0/rnn['dt']                  # bins / sec
  53   # Used for plotting in LFADS
  54   rnn['conversion_factor'] = mfr / nbins_per_sec # spikes / bin
  55   return rnn

  58 def generate_data(rnn, T, E, x0s=None, P_sxn=None, input_magnitude=0.0,
  59                   input_times=None):
  60   """ Generates data from an randomly initialized RNN.
  61   Args:
  62     rnn: the rnn
  63     T: Time in seconds to run (divided by rnn['dt'] to get steps, rounded down.
  64     E: total number of examples
  65     S: number of samples (subsampling N)
  66   Returns:
  67     A list of length E of NxT tensors of the network being run.
  68   """
  69   N = rnn['N']
  70   def run_rnn(rnn, x0, ntime_steps, input_time=None):
  71     rs = np.zeros([N,ntime_steps])
  72     x_tm1 = x0
  73     r_tm1 = np.tanh(x0)
  74     tau = rnn['tau']
  75     dt = rnn['dt']
  76     alpha = (1.0-dt/tau)
  77     W = dt/tau*rnn['W']*rnn['g']
  78     Bin = dt/tau*rnn['Bin']
  79     Bin2 = dt/tau*rnn['Bin2']
  80     b = dt/tau*rnn['b']
  81 
  82     us = np.zeros([1, ntime_steps])
  83     for t in range(ntime_steps):
  84       x_t = alpha*x_tm1 + np.dot(W,r_tm1) + b
  85       if input_time is not None and t == input_time:
  86         us[0,t] = input_magnitude
  87         x_t += Bin * us[0,t] # DCS is this what was used?
  88       r_t = np.tanh(x_t)
  89       x_tm1 = x_t
  90       r_tm1 = r_t
  91       rs[:,t] = r_t
  92     return rs, us
  93 
  94   if P_sxn is None:
  95     P_sxn = np.eye(N)
  96   ntime_steps = int(T / rnn['dt'])
  97   data_e = []
  98   inputs_e = []
  99   for e in range(E):
 100     input_time = input_times[e] if input_times is not None else None
 101     r_nxt, u_uxt = run_rnn(rnn, x0s[:,e], ntime_steps, input_time)
 102     r_sxt = np.dot(P_sxn, r_nxt)
 103     inputs_e.append(u_uxt)
 104     data_e.append(r_sxt)
 105 
 106   S = P_sxn.shape[0]
 107   data_e = normalize_rates(data_e, E, S)
 108 
 109   return data_e, x0s, inputs_e

 112 def normalize_rates(data_e, E, S):
 113   # Normalization, made more complex because of the P matrices.
 114   # Normalize by min and max in each channel.  This normalization will
 115   # cause offset differences between identical rnn runs, but different
 116   # t hits.
 117   for e in range(E):
 118     r_sxt = data_e[e]
 119     for i in range(S):
 120       rmin = np.min(r_sxt[i,:])
 121       rmax = np.max(r_sxt[i,:])
 122       assert rmax - rmin != 0, 'Something wrong'
 123       r_sxt[i,:] = (r_sxt[i,:] - rmin)/(rmax-rmin)
 124     data_e[e] = r_sxt
 125   return data_e

 128 def spikify_data(data_e, rng, dt=1.0, max_firing_rate=100):
 129   """ Apply spikes to a continuous dataset whose values are between 0.0 and 1.0
 130   Args:
 131     data_e: nexamples length list of NxT trials
 132     dt: how often the data are sampled
 133     max_firing_rate: the firing rate that is associated with a value of 1.0
 134   Returns:
 135     spikified_e: a list of length b of the data represented as spikes,
 136     sampled from the underlying poisson process.
 137     """
 138 
 139   E = len(data_e)
 140   spikes_e = []
 141   for e in range(E):
 142     data = data_e[e]
 143     N,T = data.shape
 144     data_s = np.zeros([N,T]).astype(np.int)
 145     for n in range(N):
 146       f = data[n,:]
 147       s = rng.poisson(f*max_firing_rate*dt, size=T)
 148       data_s[n,:] = s
 149     spikes_e.append(data_s)
 150 
 151   return spikes_e

 154 def gaussify_data(data_e, rng, dt=1.0, max_firing_rate=100):
 155   """ Apply gaussian noise to a continuous dataset whose values are between
 156   0.0 and 1.0
 157 
 158   Args:
 159     data_e: nexamples length list of NxT trials
 160     dt: how often the data are sampled
 161     max_firing_rate: the firing rate that is associated with a value of 1.0
 162   Returns:
 163     gauss_e: a list of length b of the data with noise.
 164     """
 165 
 166   E = len(data_e)
 167   mfr = max_firing_rate
 168   gauss_e = []
 169   for e in range(E):
 170     data = data_e[e]
 171     N,T = data.shape
 172     noisy_data = data * mfr + np.random.randn(N,T) * (5.0*mfr) * np.sqrt(dt)
 173     gauss_e.append(noisy_data)
 174 
 175   return gauss_e

 179 def get_train_n_valid_inds(num_trials, train_fraction, nreplications):
 180   """Split the numbers between 0 and num_trials-1 into two portions for
 181   training and validation, based on the train fraction.
 182   Args:
 183     num_trials: the number of trials
 184     train_fraction: (e.g. .80)
 185     nreplications: the number of spiking trials per initial condition
 186   Returns:
 187     a 2-tuple of two lists: the training indices and validation indices
 188     """
 189   train_inds = []
 190   valid_inds = []
 191   for i in range(num_trials):
 192     # This line divides up the trials so that within one initial condition,
 193     # the randomness of spikifying the condition is shared among both
 194     # training and validation data splits.
 195     if (i % nreplications)+1 > train_fraction * nreplications:
 196       valid_inds.append(i)
 197     else:
 198       train_inds.append(i)
 199 
 200   return train_inds, valid_inds

 203 def split_list_by_inds(data, inds1, inds2):
 204   """Take the data, a list, and split it up based on the indices in inds1 and
 205   inds2.
 206   Args:
 207     data: the list of data to split
 208     inds1, the first list of indices
 209     inds2, the second list of indices
 210   Returns: a 2-tuple of two lists.
 211   """
 212   if data is None or len(data) == 0:
 213     return [], []
 214   else:
 215     dout1 = [data[i] for i in inds1]
 216     dout2 = [data[i] for i in inds2]
 217     return dout1, dout2

 220 def nparray_and_transpose(data_a_b_c):
 221   """Convert the list of items in data to a numpy array, and transpose it
 222   Args:
 223     data: data_asbsc: a nested, nested list of length a, with sublist length
 224       b, with sublist length c.
 225   Returns:
 226     a numpy 3-tensor with dimensions a x c x b
 227 """
 228   data_axbxc = np.array([datum_b_c for datum_b_c in data_a_b_c])
 229   data_axcxb = np.transpose(data_axbxc, axes=[0,2,1])
 230   return data_axcxb

 233 def add_alignment_projections(datasets, npcs, ntime=None, nsamples=None):
 234   """Create a matrix that aligns the datasets a bit, under
 235   the assumption that each dataset is observing the same underlying dynamical
 236   system.
 237 
 238   Args:
 239     datasets: The dictionary of dataset structures.
 240     npcs:  The number of pcs for each, basically like lfads factors.
 241     nsamples (optional): Number of samples to take for each dataset.
 242     ntime (optional): Number of time steps to take in each sample.
 243 
 244   Returns:
 245     The dataset structures, with the field alignment_matrix_cxf added.
 246     This is # channels x npcs dimension
 247 """
 248   nchannels_all = 0
 249   channel_idxs = {}
 250   conditions_all = {}
 251   nconditions_all = 0
 252   for name, dataset in datasets.items():
 253     cidxs = np.where(dataset['P_sxn'])[1] # non-zero entries in columns
 254     channel_idxs[name] = [cidxs[0], cidxs[-1]+1]
 255     nchannels_all += cidxs[-1]+1 - cidxs[0]
 256     conditions_all[name] = np.unique(dataset['condition_labels_train'])
 257 
 258   all_conditions_list = \
 259       np.unique(np.ndarray.flatten(np.array(conditions_all.values())))
 260   nconditions_all = all_conditions_list.shape[0]
 261 
 262   if ntime is None:
 263     ntime = dataset['train_data'].shape[1]
 264   if nsamples is None:
 265     nsamples = dataset['train_data'].shape[0]
 266 
 267   # In the data workup in the paper, Chethan did intra condition
 268   # averaging, so let's do that here.
 269   avg_data_all = {}
 270   for name, conditions in conditions_all.items():
 271     dataset = datasets[name]
 272     avg_data_all[name] = {}
 273     for cname in conditions:
 274       td_idxs = np.argwhere(np.array(dataset['condition_labels_train'])==cname)
 275       data = np.squeeze(dataset['train_data'][td_idxs,:,:], axis=1)
 276       avg_data = np.mean(data, axis=0)
 277       avg_data_all[name][cname] = avg_data
 278 
 279   # Visualize this in the morning.
 280   all_data_nxtc = np.zeros([nchannels_all, ntime * nconditions_all])
 281   for name, dataset in datasets.items():
 282     cidx_s = channel_idxs[name][0]
 283     cidx_f = channel_idxs[name][1]
 284     for cname in conditions_all[name]:
 285       cidxs = np.argwhere(all_conditions_list == cname)
 286       if cidxs.shape[0] > 0:
 287         cidx = cidxs[0][0]
 288         all_tidxs = np.arange(0, ntime+1) + cidx*ntime
 289         all_data_nxtc[cidx_s:cidx_f, all_tidxs[0]:all_tidxs[-1]] = \
 290             avg_data_all[name][cname].T
 291 
 292   # A bit of filtering. We don't care about spectral properties, or
 293   # filtering artifacts, simply correlate time steps a bit.
 294   filt_len = 6
 295   bc_filt = np.ones([filt_len])/float(filt_len)
 296   for c in range(nchannels_all):
 297     all_data_nxtc[c,:] = scipy.signal.filtfilt(bc_filt, [1.0], all_data_nxtc[c,:])
 298 
 299   # Compute the PCs.
 300   all_data_mean_nx1 = np.mean(all_data_nxtc, axis=1, keepdims=True)
 301   all_data_zm_nxtc = all_data_nxtc - all_data_mean_nx1
 302   corr_mat_nxn = np.dot(all_data_zm_nxtc, all_data_zm_nxtc.T)
 303   evals_n, evecs_nxn = np.linalg.eigh(corr_mat_nxn)
 304   sidxs = np.flipud(np.argsort(evals_n)) # sort such that 0th is highest
 305   evals_n = evals_n[sidxs]
 306   evecs_nxn = evecs_nxn[:,sidxs]
 307 
 308   # Project all the channels data onto the low-D PCA basis, where
 309   # low-d is the npcs parameter.
 310   all_data_pca_pxtc = np.dot(evecs_nxn[:, 0:npcs].T, all_data_zm_nxtc)
 311 
 312   # Now for each dataset, we regress the channel data onto the top
 313   # pcs, and this will be our alignment matrix for that dataset.
 314   # |B - A*W|^2
 315   for name, dataset in datasets.items():
 316     cidx_s = channel_idxs[name][0]
 317     cidx_f = channel_idxs[name][1]
 318     all_data_zm_chxtc = all_data_zm_nxtc[cidx_s:cidx_f,:] # ch for channel
 319     W_chxp, _, _, _ = \
 320         np.linalg.lstsq(all_data_zm_chxtc.T, all_data_pca_pxtc.T)
 321     dataset['alignment_matrix_cxf'] = W_chxp
 322     alignment_bias_cx1 = all_data_mean_nx1[cidx_s:cidx_f]
 323     dataset['alignment_bias_c'] = np.squeeze(alignment_bias_cx1, axis=1)
 324 
 325   do_debug_plot = False
 326   if do_debug_plot:
 327     pc_vecs = evecs_nxn[:,0:npcs]
 328     ntoplot = 400
 329 
 330     plt.figure()
 331     plt.plot(np.log10(evals_n), '-x')
 332     plt.figure()
 333     plt.subplot(311)
 334     plt.imshow(all_data_pca_pxtc)
 335     plt.colorbar()
 336 
 337     plt.subplot(312)
 338     plt.imshow(np.dot(W_chxp.T, all_data_zm_chxtc))
 339     plt.colorbar()
 340 
 341     plt.subplot(313)
 342     plt.imshow(np.dot(all_data_zm_chxtc.T, W_chxp).T - all_data_pca_pxtc)
 343     plt.colorbar()
 344 
 345     import pdb
 346     pdb.set_trace()
 347 
 348   return datasets
