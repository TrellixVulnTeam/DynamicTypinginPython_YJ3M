  16 from __future__ import print_function
  17 
  18 import os
  19 import h5py
  20 import numpy as np
  21 from six.moves import xrange
  22 
  23 from synthetic_data_utils import generate_data, generate_rnn
  24 from synthetic_data_utils import get_train_n_valid_inds
  25 from synthetic_data_utils import nparray_and_transpose
  26 from synthetic_data_utils import spikify_data, split_list_by_inds
  27 import tensorflow as tf
  28 from utils import write_datasets
  29 
  30 DATA_DIR = "rnn_synth_data_v1.0"
  31 
  32 flags = tf.app.flags
  33 flags.DEFINE_string("save_dir", "/tmp/" + DATA_DIR + "/",
  34                     "Directory for saving data.")
  35 flags.DEFINE_string("datafile_name", "conditioned_rnn_data",
  36                     "Name of data file for input case.")
  37 flags.DEFINE_integer("synth_data_seed", 5, "Random seed for RNN generation.")
  38 flags.DEFINE_float("T", 1.0, "Time in seconds to generate.")
  39 flags.DEFINE_integer("C", 400, "Number of conditions")
  40 flags.DEFINE_integer("N", 50, "Number of units for the RNN")
  41 flags.DEFINE_float("train_percentage", 4.0/5.0,
  42                    "Percentage of train vs validation trials")
  43 flags.DEFINE_integer("nreplications", 10,
  44                      "Number of spikifications of the same underlying rates.")
  45 flags.DEFINE_float("g", 1.5, "Complexity of dynamics")
  46 flags.DEFINE_float("x0_std", 1.0,
  47                    "Volume from which to pull initial conditions (affects diversity of dynamics.")
  48 flags.DEFINE_float("tau", 0.025, "Time constant of RNN")
  49 flags.DEFINE_float("dt", 0.010, "Time bin")
  50 flags.DEFINE_float("max_firing_rate", 30.0, "Map 1.0 of RNN to a spikes per second")
  51 FLAGS = flags.FLAGS
  52 
  53 rng = np.random.RandomState(seed=FLAGS.synth_data_seed)
  54 rnn_rngs = [np.random.RandomState(seed=FLAGS.synth_data_seed+1),
  55             np.random.RandomState(seed=FLAGS.synth_data_seed+2)]
  56 T = FLAGS.T
  57 C = FLAGS.C
  58 N = FLAGS.N
  59 nreplications = FLAGS.nreplications
  60 E = nreplications * C
  61 train_percentage = FLAGS.train_percentage
  62 ntimesteps = int(T / FLAGS.dt)
  63 
  64 rnn_a = generate_rnn(rnn_rngs[0], N, FLAGS.g, FLAGS.tau, FLAGS.dt,
  65                      FLAGS.max_firing_rate)
  66 rnn_b = generate_rnn(rnn_rngs[1], N, FLAGS.g, FLAGS.tau, FLAGS.dt,
  67                      FLAGS.max_firing_rate)
  68 rnns = [rnn_a, rnn_b]
  69 
  70 # pick which RNN is used on each trial
  71 rnn_to_use = rng.randint(2, size=E)
  72 ext_input = np.repeat(np.expand_dims(rnn_to_use, axis=1), ntimesteps, axis=1)
  73 ext_input = np.expand_dims(ext_input, axis=2)  # these are "a's" in the paper
  74 
  75 x0s = []
  76 condition_labels = []
  77 condition_number = 0
  78 for c in range(C):
  79   x0 = FLAGS.x0_std * rng.randn(N, 1)
  80   x0s.append(np.tile(x0, nreplications))
  81   for ns in range(nreplications):
  82     condition_labels.append(condition_number)
  83   condition_number += 1
  84 x0s = np.concatenate(x0s, axis=1)
  85 
  86 P_nxn = rng.randn(N, N) / np.sqrt(N)
  87 
  88 # generate trials for both RNNs
  89 rates_a, x0s_a, _ = generate_data(rnn_a, T=T, E=E, x0s=x0s, P_sxn=P_nxn,
  90                                   input_magnitude=0.0, input_times=None)
  91 spikes_a = spikify_data(rates_a, rng, rnn_a['dt'], rnn_a['max_firing_rate'])
  92 
  93 rates_b, x0s_b, _ = generate_data(rnn_b, T=T, E=E, x0s=x0s, P_sxn=P_nxn,
  94                                   input_magnitude=0.0, input_times=None)
  95 spikes_b = spikify_data(rates_b, rng, rnn_b['dt'], rnn_b['max_firing_rate'])
  96 
  97 # not the best way to do this but E is small enough
  98 rates = []
  99 spikes = []
 100 for trial in xrange(E):
 101   if rnn_to_use[trial] == 0:
 102     rates.append(rates_a[trial])
 103     spikes.append(spikes_a[trial])
 104   else:
 105     rates.append(rates_b[trial])
 106     spikes.append(spikes_b[trial])
 107 
 108 # split into train and validation sets
 109 train_inds, valid_inds = get_train_n_valid_inds(E, train_percentage,
 110                                                 nreplications)
 111 
 112 rates_train, rates_valid = split_list_by_inds(rates, train_inds, valid_inds)
 113 spikes_train, spikes_valid = split_list_by_inds(spikes, train_inds, valid_inds)
 114 condition_labels_train, condition_labels_valid = split_list_by_inds(
 115     condition_labels, train_inds, valid_inds)
 116 ext_input_train, ext_input_valid = split_list_by_inds(
 117     ext_input, train_inds, valid_inds)
 118 
 119 rates_train = nparray_and_transpose(rates_train)
 120 rates_valid = nparray_and_transpose(rates_valid)
 121 spikes_train = nparray_and_transpose(spikes_train)
 122 spikes_valid = nparray_and_transpose(spikes_valid)
 123 
 124 # add train_ext_input and valid_ext input
 125 data = {'train_truth': rates_train,
 126         'valid_truth': rates_valid,
 127         'train_data' : spikes_train,
 128         'valid_data' : spikes_valid,
 129         'train_ext_input' : np.array(ext_input_train),
 130         'valid_ext_input': np.array(ext_input_valid),
 131         'train_percentage' : train_percentage,
 132         'nreplications' : nreplications,
 133         'dt' : FLAGS.dt,
 134         'P_sxn' : P_nxn,
 135         'condition_labels_train' : condition_labels_train,
 136         'condition_labels_valid' : condition_labels_valid,
 137         'conversion_factor': 1.0 / rnn_a['conversion_factor']}
 138 
 139 # just one dataset here
 140 datasets = {}
 141 dataset_name = 'dataset_N' + str(N)
 142 datasets[dataset_name] = data
 143 
 144 # write out the dataset
 145 write_datasets(FLAGS.save_dir, FLAGS.datafile_name, datasets)
 146 print ('Saved to ', os.path.join(FLAGS.save_dir,
 147                                  FLAGS.datafile_name + '_' + dataset_name))
