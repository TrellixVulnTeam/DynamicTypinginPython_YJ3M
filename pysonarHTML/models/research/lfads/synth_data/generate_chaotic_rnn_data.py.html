  33 DATA_DIR = "rnn_synth_data_v1.0"

  35 flags = tf.app.flags
  36 flags.DEFINE_string("save_dir", "/tmp/" + DATA_DIR + "/",
  37                     "Directory for saving data.")
  38 flags.DEFINE_string("datafile_name", "thits_data",
  39                     "Name of data file for input case.")
  40 flags.DEFINE_string("noise_type", "poisson", "Noise type for data.")
  41 flags.DEFINE_integer("synth_data_seed", 5, "Random seed for RNN generation.")
  42 flags.DEFINE_float("T", 1.0, "Time in seconds to generate.")
  43 flags.DEFINE_integer("C", 100, "Number of conditions")
  44 flags.DEFINE_integer("N", 50, "Number of units for the RNN")
  45 flags.DEFINE_integer("S", 50, "Number of sampled units from RNN")
  46 flags.DEFINE_integer("npcs", 10, "Number of PCS for multi-session case.")
  47 flags.DEFINE_float("train_percentage", 4.0/5.0,
  48                    "Percentage of train vs validation trials")
  49 flags.DEFINE_integer("nreplications", 40,
  50                      "Number of noise replications of the same underlying rates.")
  51 flags.DEFINE_float("g", 1.5, "Complexity of dynamics")
  52 flags.DEFINE_float("x0_std", 1.0,
  53                    "Volume from which to pull initial conditions (affects diversity of dynamics.")
  54 flags.DEFINE_float("tau", 0.025, "Time constant of RNN")
  55 flags.DEFINE_float("dt", 0.010, "Time bin")
  56 flags.DEFINE_float("input_magnitude", 20.0,
  57                    "For the input case, what is the value of the input?")
  58 flags.DEFINE_float("max_firing_rate", 30.0, "Map 1.0 of RNN to a spikes per second")
  59 FLAGS = flags.FLAGS

  62 # Note that with N small, (as it is 25 above), the finite size effects
  63 # will have pretty dramatic effects on the dynamics of the random RNN.
  64 # If you want more complex dynamics, you'll have to run the script a
  65 # lot, or increase N (or g).
  66 
  67 # Getting hard vs. easy data can be a little stochastic, so we set the seed.
  68 
  69 # Pull out some commonly used parameters.
  70 # These are user parameters (configuration)
  71 rng = np.random.RandomState(seed=FLAGS.synth_data_seed)
  72 T = FLAGS.T
  73 C = FLAGS.C
  74 N = FLAGS.N
  75 S = FLAGS.S
  76 input_magnitude = FLAGS.input_magnitude
  77 nreplications = FLAGS.nreplications
  78 E = nreplications * C         # total number of trials
  79 # S is the number of measurements in each datasets, w/ each
  80 # dataset having a different set of observations.
  81 ndatasets = N/S                 # ok if rounded down
  82 train_percentage = FLAGS.train_percentage
  83 ntime_steps = int(T / FLAGS.dt)
  84 # End of user parameters
  85 
  86 rnn = generate_rnn(rng, N, FLAGS.g, FLAGS.tau, FLAGS.dt, FLAGS.max_firing_rate)
  87 
  88 # Check to make sure the RNN is the one we used in the paper.
  89 if N == 50:
  90   assert abs(rnn['W'][0,0] - 0.06239899) < 1e-8, 'Error in random seed?'
  91   rem_check = nreplications * train_percentage
  92   assert  abs(rem_check - int(rem_check)) < 1e-8, \
  93     'Train percentage  * nreplications should be integral number.'
  94 
  95 
  96 # Initial condition generation, and condition label generation.  This
  97 # happens outside of the dataset loop, so that all datasets have the
  98 # same conditions, which is similar to a neurophys setup.
  99 condition_number = 0
 100 x0s = []
 101 condition_labels = []
 102 for c in range(C):
 103   x0 = FLAGS.x0_std * rng.randn(N, 1)
 104   x0s.append(np.tile(x0, nreplications)) # replicate x0 nreplications times
 105   # replicate the condition label nreplications times
 106   for ns in range(nreplications):
 107     condition_labels.append(condition_number)
 108   condition_number += 1
 109 x0s = np.concatenate(x0s, axis=1)
 110 
 111 # Containers for storing data across data.
 112 datasets = {}
 113 for n in range(ndatasets):
 114   print(n+1, " of ", ndatasets)
 115 
 116   # First generate all firing rates. in the next loop, generate all
 117   # replications this allows the random state for rate generation to be
 118   # independent of n_replications.
 119   dataset_name = 'dataset_N' + str(N) + '_S' + str(S)
 120   if S < N:
 121     dataset_name += '_n' + str(n+1)
 122 
 123   # Sample neuron subsets.  The assumption is the PC axes of the RNN
 124   # are not unit aligned, so sampling units is adequate to sample all
 125   # the high-variance PCs.
 126   P_sxn = np.eye(S,N)
 127   for m in range(n):
 128     P_sxn = np.roll(P_sxn, S, axis=1)
 129 
 130   if input_magnitude > 0.0:
 131     # time of "hits" randomly chosen between [1/4 and 3/4] of total time
 132     input_times = rng.choice(int(ntime_steps/2), size=[E]) + int(ntime_steps/4)
 133   else:
 134     input_times = None
 135 
 136   rates, x0s, inputs = \
 137       generate_data(rnn, T=T, E=E, x0s=x0s, P_sxn=P_sxn,
 138                     input_magnitude=input_magnitude,
 139                     input_times=input_times)
 140 
 141   if FLAGS.noise_type == "poisson":
 142     noisy_data = spikify_data(rates, rng, rnn['dt'], rnn['max_firing_rate'])
 143   elif FLAGS.noise_type == "gaussian":
 144     noisy_data = gaussify_data(rates, rng, rnn['dt'], rnn['max_firing_rate'])
 145   else:
 146     raise ValueError("Only noise types supported are poisson or gaussian")
 147 
 148     # split into train and validation sets
 149   train_inds, valid_inds = get_train_n_valid_inds(E, train_percentage,
 150                                                   nreplications)
 151 
 152   # Split the data, inputs, labels and times into train vs. validation.
 153   rates_train, rates_valid = \
 154       split_list_by_inds(rates, train_inds, valid_inds)
 155   noisy_data_train, noisy_data_valid = \
 156       split_list_by_inds(noisy_data, train_inds, valid_inds)
 157   input_train, inputs_valid = \
 158       split_list_by_inds(inputs, train_inds, valid_inds)
 159   condition_labels_train, condition_labels_valid = \
 160       split_list_by_inds(condition_labels, train_inds, valid_inds)
 161   input_times_train, input_times_valid = \
 162       split_list_by_inds(input_times, train_inds, valid_inds)
 163 
 164   # Turn rates, noisy_data, and input into numpy arrays.
 165   rates_train = nparray_and_transpose(rates_train)
 166   rates_valid = nparray_and_transpose(rates_valid)
 167   noisy_data_train = nparray_and_transpose(noisy_data_train)
 168   noisy_data_valid = nparray_and_transpose(noisy_data_valid)
 169   input_train = nparray_and_transpose(input_train)
 170   inputs_valid = nparray_and_transpose(inputs_valid)
 171 
 172   # Note that we put these 'truth' rates and input into this
 173   # structure, the only data that is used in LFADS are the noisy
 174   # data e.g. spike trains.  The rest is either for printing or posterity.
 175   data = {'train_truth': rates_train,
 176           'valid_truth': rates_valid,
 177           'input_train_truth' : input_train,
 178           'input_valid_truth' : inputs_valid,
 179           'train_data' : noisy_data_train,
 180           'valid_data' : noisy_data_valid,
 181           'train_percentage' : train_percentage,
 182           'nreplications' : nreplications,
 183           'dt' : rnn['dt'],
 184           'input_magnitude' : input_magnitude,
 185           'input_times_train' : input_times_train,
 186           'input_times_valid' : input_times_valid,
 187           'P_sxn' : P_sxn,
 188           'condition_labels_train' : condition_labels_train,
 189           'condition_labels_valid' : condition_labels_valid,
 190           'conversion_factor': 1.0 / rnn['conversion_factor']}
 191   datasets[dataset_name] = data
 192 
 193 if S < N:
 194   # Note that this isn't necessary for this synthetic example, but
 195   # it's useful to see how the input factor matrices were initialized
 196   # for actual neurophysiology data.
 197   datasets = add_alignment_projections(datasets, npcs=FLAGS.npcs)
 198 
 199 # Write out the datasets.
 200 write_datasets(FLAGS.save_dir, FLAGS.datafile_name, datasets)
