  61 class IntegrationToBoundModel:
  62   def __init__(self, N):
  63     scale = 0.8 / float(N**0.5)
  64     self.N = N
  65     self.Wh_nxn = tf.Variable(tf.random_normal([N, N], stddev=scale))
  66     self.b_1xn = tf.Variable(tf.zeros([1, N]))
  67     self.Bu_1xn = tf.Variable(tf.zeros([1, N]))
  68     self.Wro_nxo = tf.Variable(tf.random_normal([N, 1], stddev=scale))
  69     self.bro_o = tf.Variable(tf.zeros([1]))
  70 
  71   def call(self, h_tm1_bxn, u_bx1):
  72     act_t_bxn = tf.matmul(h_tm1_bxn, self.Wh_nxn) + self.b_1xn + u_bx1 * self.Bu_1xn
  73     h_t_bxn = tf.nn.tanh(act_t_bxn)
  74     z_t = tf.nn.xw_plus_b(h_t_bxn, self.Wro_nxo, self.bro_o)
  75     return z_t, h_t_bxn

  77 def get_data_batch(batch_size, T, rng, u_std):
  78   u_bxt = rng.randn(batch_size, T) * u_std
  79   running_sum_b = np.zeros([batch_size])
  80   labels_bxt = np.zeros([batch_size, T])
  81   for t in xrange(T):
  82     running_sum_b += u_bxt[:, t]
  83     labels_bxt[:, t] += running_sum_b
  84   labels_bxt = np.clip(labels_bxt, -1, 1)
  85   return u_bxt, labels_bxt

  88 rng = np.random.RandomState(seed=FLAGS.synth_data_seed)
  89 u_rng = np.random.RandomState(seed=FLAGS.synth_data_seed+1)
  90 T = FLAGS.T
  91 C = FLAGS.C
  92 N = FLAGS.N  # must be same N as in trained model (provided example is N = 50)
  93 nreplications = FLAGS.nreplications
  94 E = nreplications * C  # total number of trials
  95 train_percentage = FLAGS.train_percentage
  96 ntimesteps = int(T / FLAGS.dt)
  97 batch_size = 1  # gives one example per ntrial
  98 
  99 model = IntegrationToBoundModel(N)
 100 inputs_ph_t = [tf.placeholder(tf.float32,
 101                               shape=[None, 1]) for _ in range(ntimesteps)]
 102 state = tf.zeros([batch_size, N])
 103 saver = tf.train.Saver()
 104 
 105 P_nxn = rng.randn(N,N) / np.sqrt(N)  # random projections
 106 
 107 # unroll RNN for T timesteps
 108 outputs_t = []
 109 states_t = []
 110 
 111 for inp in inputs_ph_t:
 112   output, state = model.call(state, inp)
 113   outputs_t.append(output)
 114   states_t.append(state)
 115 
 116 with tf.Session() as sess:
 117   # restore the latest model ckpt
 118   if FLAGS.checkpoint_path == "SAMPLE_CHECKPOINT":
 119     dir_path = os.path.dirname(os.path.realpath(__file__))
 120     model_checkpoint_path = os.path.join(dir_path, "trained_itb/model-65000")
 121   else:
 122     model_checkpoint_path = FLAGS.checkpoint_path
 123   try:
 124     saver.restore(sess, model_checkpoint_path)
 125     print ('Model restored from', model_checkpoint_path)
 126   except:
 127     assert False, ("No checkpoints to restore from, is the path %s correct?"
 128                    %model_checkpoint_path)
 129 
 130   # generate data for trials
 131   data_e = []
 132   u_e = []
 133   outs_e = []
 134   for c in range(C):
 135     u_1xt, outs_1xt = get_data_batch(batch_size, ntimesteps, u_rng, FLAGS.u_std)
 136 
 137     feed_dict = {}
 138     for t in xrange(ntimesteps):
 139       feed_dict[inputs_ph_t[t]] = np.reshape(u_1xt[:,t], (batch_size,-1))
 140 
 141     states_t_bxn, outputs_t_bxn = sess.run([states_t, outputs_t],
 142                                            feed_dict=feed_dict)
 143     states_nxt = np.transpose(np.squeeze(np.asarray(states_t_bxn)))
 144     outputs_t_bxn = np.squeeze(np.asarray(outputs_t_bxn))
 145     r_sxt = np.dot(P_nxn, states_nxt)
 146 
 147     for s in xrange(nreplications):
 148       data_e.append(r_sxt)
 149       u_e.append(u_1xt)
 150       outs_e.append(outputs_t_bxn)
 151 
 152   truth_data_e = normalize_rates(data_e, E, N)
 153 
 154 spiking_data_e = spikify_data(truth_data_e, rng, dt=FLAGS.dt,
 155                               max_firing_rate=FLAGS.max_firing_rate)
 156 train_inds, valid_inds = get_train_n_valid_inds(E, train_percentage,
 157                                                 nreplications)
 158 
 159 data_train_truth, data_valid_truth = split_list_by_inds(truth_data_e,
 160                                                         train_inds,
 161                                                         valid_inds)
 162 data_train_spiking, data_valid_spiking = split_list_by_inds(spiking_data_e,
 163                                                             train_inds,
 164                                                             valid_inds)
 165 
 166 data_train_truth = nparray_and_transpose(data_train_truth)
 167 data_valid_truth = nparray_and_transpose(data_valid_truth)
 168 data_train_spiking = nparray_and_transpose(data_train_spiking)
 169 data_valid_spiking = nparray_and_transpose(data_valid_spiking)
 170 
 171 # save down the inputs used to generate this data
 172 train_inputs_u, valid_inputs_u = split_list_by_inds(u_e,
 173                                                     train_inds,
 174                                                     valid_inds)
 175 train_inputs_u = nparray_and_transpose(train_inputs_u)
 176 valid_inputs_u = nparray_and_transpose(valid_inputs_u)
 177 
 178 # save down the network outputs (may be useful later)
 179 train_outputs_u, valid_outputs_u = split_list_by_inds(outs_e,
 180                                                       train_inds,
 181                                                       valid_inds)
 182 train_outputs_u = np.array(train_outputs_u)
 183 valid_outputs_u = np.array(valid_outputs_u)
 184
