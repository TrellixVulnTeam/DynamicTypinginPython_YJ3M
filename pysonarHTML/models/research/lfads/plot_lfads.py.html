  26 def _plot_item(W, name, full_name, nspaces):
  27   plt.figure()
  28   if W.shape == ():
  29     print(name, ": ", W)
  30   elif W.shape[0] == 1:
  31     plt.stem(W.T)
  32     plt.title(full_name)
  33   elif W.shape[1] == 1:
  34     plt.stem(W)
  35     plt.title(full_name)
  36   else:
  37     plt.imshow(np.abs(W), interpolation='nearest', cmap='jet');
  38     plt.colorbar()
  39     plt.title(full_name)

  42 def all_plot(d, full_name="", exclude="", nspaces=0):
  43   """Recursively plot all the LFADS model parameters in the nested
  44   dictionary."""
  45   for k, v in d.iteritems():
  46     this_name = full_name+"/"+k
  47     if isinstance(v, dict):
  48       all_plot(v, full_name=this_name, exclude=exclude, nspaces=nspaces+4)
  49     else:
  50       if exclude == "" or exclude not in this_name:
  51         _plot_item(v, name=k, full_name=full_name+"/"+k, nspaces=nspaces+4)

  55 def plot_time_series(vals_bxtxn, bidx=None, n_to_plot=np.inf, scale=1.0,
  56                      color='r', title=None):
  57 
  58   if bidx is None:
  59     vals_txn = np.mean(vals_bxtxn, axis=0)
  60   else:
  61     vals_txn = vals_bxtxn[bidx,:,:]
  62 
  63   T, N = vals_txn.shape
  64   if n_to_plot > N:
  65     n_to_plot = N
  66 
  67   plt.plot(vals_txn[:,0:n_to_plot] + scale*np.array(range(n_to_plot)),
  68            color=color, lw=1.0)
  69   plt.axis('tight')
  70   if title:
  71     plt.title(title)

  74 def plot_lfads_timeseries(data_bxtxn, model_vals, ext_input_bxtxi=None,
  75                           truth_bxtxn=None, bidx=None, output_dist="poisson",
  76                           conversion_factor=1.0, subplot_cidx=0,
  77                           col_title=None):
  78 
  79   n_to_plot = 10
  80   scale = 1.0
  81   nrows = 7
  82   plt.subplot(nrows,2,1+subplot_cidx)
  83 
  84   if output_dist == 'poisson':
  85     rates = means = conversion_factor * model_vals['output_dist_params']
  86     plot_time_series(rates, bidx, n_to_plot=n_to_plot, scale=scale,
  87                      title=col_title + " rates (LFADS - red, Truth - black)")
  88   elif output_dist == 'gaussian':
  89     means_vars = model_vals['output_dist_params']
  90     means, vars = np.split(means_vars,2, axis=2) # bxtxn
  91     stds = np.sqrt(vars)
  92     plot_time_series(means, bidx, n_to_plot=n_to_plot, scale=scale,
  93                      title=col_title + " means (LFADS - red, Truth - black)")
  94     plot_time_series(means+stds, bidx, n_to_plot=n_to_plot, scale=scale,
  95                      color='c')
  96     plot_time_series(means-stds, bidx, n_to_plot=n_to_plot, scale=scale,
  97                      color='c')
  98   else:
  99     assert 'NIY'
 100 
 101 
 102   if truth_bxtxn is not None:
 103     plot_time_series(truth_bxtxn, bidx, n_to_plot=n_to_plot, color='k',
 104                      scale=scale)
 105 
 106   input_title = ""
 107   if "controller_outputs" in model_vals.keys():
 108     input_title += " Controller Output"
 109     plt.subplot(nrows,2,3+subplot_cidx)
 110     u_t = model_vals['controller_outputs'][0:-1]
 111     plot_time_series(u_t, bidx, n_to_plot=n_to_plot, color='c', scale=1.0,
 112                      title=col_title + input_title)
 113 
 114   if ext_input_bxtxi is not None:
 115     input_title += " External Input"
 116     plot_time_series(ext_input_bxtxi, n_to_plot=n_to_plot, color='b',
 117                      scale=scale, title=col_title + input_title)
 118 
 119   plt.subplot(nrows,2,5+subplot_cidx)
 120   plot_time_series(means, bidx,
 121                    n_to_plot=n_to_plot, scale=1.0,
 122                    title=col_title + " Spikes (LFADS - red, Spikes - black)")
 123   plot_time_series(data_bxtxn, bidx, n_to_plot=n_to_plot, color='k', scale=1.0)
 124 
 125   plt.subplot(nrows,2,7+subplot_cidx)
 126   plot_time_series(model_vals['factors'], bidx, n_to_plot=n_to_plot, color='b',
 127                    scale=2.0, title=col_title + " Factors")
 128 
 129   plt.subplot(nrows,2,9+subplot_cidx)
 130   plot_time_series(model_vals['gen_states'], bidx, n_to_plot=n_to_plot,
 131                    color='g', scale=1.0, title=col_title + " Generator State")
 132 
 133   if bidx is not None:
 134     data_nxt = data_bxtxn[bidx,:,:].T
 135     params_nxt = model_vals['output_dist_params'][bidx,:,:].T
 136   else:
 137     data_nxt = np.mean(data_bxtxn, axis=0).T
 138     params_nxt = np.mean(model_vals['output_dist_params'], axis=0).T
 139   if output_dist == 'poisson':
 140     means_nxt = params_nxt
 141   elif output_dist == 'gaussian': # (means+vars) x time
 142     means_nxt = np.vsplit(params_nxt,2)[0] # get means
 143   else:
 144     assert "NIY"
 145 
 146   plt.subplot(nrows,2,11+subplot_cidx)
 147   plt.imshow(data_nxt, aspect='auto', interpolation='nearest')
 148   plt.title(col_title + ' Data')
 149 
 150   plt.subplot(nrows,2,13+subplot_cidx)
 151   plt.imshow(means_nxt, aspect='auto', interpolation='nearest')
 152   plt.title(col_title + ' Means')

 155 def plot_lfads(train_bxtxd, train_model_vals,
 156                train_ext_input_bxtxi=None, train_truth_bxtxd=None,
 157                valid_bxtxd=None, valid_model_vals=None,
 158                valid_ext_input_bxtxi=None, valid_truth_bxtxd=None,
 159                bidx=None, cf=1.0, output_dist='poisson'):
 160 
 161   # Plotting
 162   f = plt.figure(figsize=(18,20), tight_layout=True)
 163   plot_lfads_timeseries(train_bxtxd, train_model_vals,
 164                         train_ext_input_bxtxi,
 165                         truth_bxtxn=train_truth_bxtxd,
 166                         conversion_factor=cf, bidx=bidx,
 167                         output_dist=output_dist, col_title='Train')
 168   plot_lfads_timeseries(valid_bxtxd, valid_model_vals,
 169                         valid_ext_input_bxtxi,
 170                         truth_bxtxn=valid_truth_bxtxd,
 171                         conversion_factor=cf, bidx=bidx,
 172                         output_dist=output_dist,
 173                         subplot_cidx=1, col_title='Valid')
 174 
 175   # Convert from figure to an numpy array width x height x 3 (last for RGB)
 176   f.canvas.draw()
 177   data = np.fromstring(f.canvas.tostring_rgb(), dtype=np.uint8, sep='')
 178   data_wxhx3 = data.reshape(f.canvas.get_width_height()[::-1] + (3,))
 179   plt.close()
 180 
 181   return data_wxhx3
