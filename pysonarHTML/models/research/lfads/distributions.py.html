  20 class Poisson(object):
  21   """Poisson distributon
  22 
  23   Computes the log probability under the model.
  24 
  25   """
  26   def __init__(self, log_rates):
  27     """ Create Poisson distributions with log_rates parameters.
  28 
  29     Args:
  30       log_rates: a tensor-like list of log rates underlying the Poisson dist.
  31     """
  32     self.logr = log_rates
  33 
  34   def logp(self, bin_counts):
  35     """Compute the log probability for the counts in the bin, under the model.
  36 
  37     Args:
  38       bin_counts: array-like integer counts
  39 
  40     Returns:
  41       The log-probability under the Poisson models for each element of
  42       bin_counts.
  43     """
  44     k = tf.to_float(bin_counts)
  45     # log poisson(k, r) = log(r^k * e^(-r) / k!) = k log(r) - r - log k!
  46     # log poisson(k, r=exp(x)) = k * x - exp(x) - lgamma(k + 1)
  47     return k * self.logr - tf.exp(self.logr) - tf.lgamma(k + 1)

  50 def diag_gaussian_log_likelihood(z, mu=0.0, logvar=0.0):
  51   """Log-likelihood under a Gaussian distribution with diagonal covariance.
  52     Returns the log-likelihood for each dimension.  One should sum the
  53     results for the log-likelihood under the full multidimensional model.
  54 
  55   Args:
  56     z: The value to compute the log-likelihood.
  57     mu: The mean of the Gaussian
  58     logvar: The log variance of the Gaussian.
  59 
  60   Returns:
  61     The log-likelihood under the Gaussian model.
  62   """
  63 
  64   return -0.5 * (logvar + np.log(2*np.pi) + \
  65                  tf.square((z-mu)/tf.exp(0.5*logvar)))
  66

  68 def gaussian_pos_log_likelihood(unused_mean, logvar, noise):
  69   """Gaussian log-likelihood function for a posterior in VAE
  70 
  71   Note: This function is specialized for a posterior distribution, that has the
  72   form of z = mean + sigma * noise.
  73 
  74   Args:
  75     unused_mean: ignore
  76     logvar: The log variance of the distribution
  77     noise: The noise used in the sampling of the posterior.
  78 
  79   Returns:
  80     The log-likelihood under the Gaussian model.
  81   """
  82   # ln N(z; mean, sigma) = - ln(sigma) - 0.5 ln 2pi - noise^2 / 2
  83   return - 0.5 * (logvar + np.log(2 * np.pi) + tf.square(noise))

  91 class DiagonalGaussian(Gaussian):
  92   """Diagonal Gaussian with different constant mean and variances in each
  93   dimension.
  94   """
  95 
  96   def __init__(self, batch_size, z_size, mean, logvar):
  97     """Create a diagonal gaussian distribution.
  98 
  99     Args:
 100       batch_size: The size of the batch, i.e. 0th dim in 2D tensor of samples.
 101       z_size: The dimension of the distribution, i.e. 1st dim in 2D tensor.
 102       mean: The N-D mean of the distribution.
 103       logvar: The N-D log variance of the diagonal distribution.
 104     """
 105     size__xz = [None, z_size]
 106     self.mean = mean            # bxn already
 107     self.logvar = logvar        # bxn already
 108     self.noise = noise = tf.random_normal(tf.shape(logvar))
 109     self.sample = mean + tf.exp(0.5 * logvar) * noise
 110     mean.set_shape(size__xz)
 111     logvar.set_shape(size__xz)
 112     self.sample.set_shape(size__xz)
 113 
 114   def logp(self, z=None):
 115     """Compute the log-likelihood under the distribution.
 116 
 117     Args:
 118       z (optional): value to compute likelihood for, if None, use sample.
 119 
 120     Returns:
 121       The likelihood of z under the model.
 122     """
 123     if z is None:
 124       z = self.sample
 125 
 126     # This is needed to make sure that the gradients are simple.
 127     # The value of the function shouldn't change.
 128     if z == self.sample:
 129       return gaussian_pos_log_likelihood(self.mean, self.logvar, self.noise)
 130 
 131     return diag_gaussian_log_likelihood(z, self.mean, self.logvar)

 134 class LearnableDiagonalGaussian(Gaussian):
 135   """Diagonal Gaussian whose mean and variance are learned parameters."""
 136 
 137   def __init__(self, batch_size, z_size, name, mean_init=0.0,
 138                var_init=1.0, var_min=0.0, var_max=1000000.0):
 139     """Create a learnable diagonal gaussian distribution.
 140 
 141     Args:
 142       batch_size: The size of the batch, i.e. 0th dim in 2D tensor of samples.
 143       z_size: The dimension of the distribution, i.e. 1st dim in 2D tensor.
 144       name: prefix name for the mean and log TF variables.
 145       mean_init (optional): The N-D mean initialization of the distribution.
 146       var_init (optional): The N-D variance initialization of the diagonal
 147         distribution.
 148       var_min (optional): The minimum value the learned variance can take in any
 149         dimension.
 150       var_max (optional): The maximum value the learned variance can take in any
 151         dimension.
 152     """
 153 
 154     size_1xn = [1, z_size]
 155     size__xn = [None, z_size]
 156     size_bx1 = tf.stack([batch_size, 1])
 157     assert var_init > 0.0, "Problems"
 158     assert var_max >= var_min, "Problems"
 159     assert var_init >= var_min, "Problems"
 160     assert var_max >= var_init, "Problems"
 161 
 162 
 163     z_mean_1xn = tf.get_variable(name=name+"/mean", shape=size_1xn,
 164                                  initializer=tf.constant_initializer(mean_init))
 165     self.mean_bxn = mean_bxn = tf.tile(z_mean_1xn, size_bx1)
 166     mean_bxn.set_shape(size__xn) # tile loses shape
 167 
 168     log_var_init = np.log(var_init)
 169     if var_max > var_min:
 170       var_is_trainable = True
 171     else:
 172       var_is_trainable = False
 173 
 174     z_logvar_1xn = \
 175         tf.get_variable(name=(name+"/logvar"), shape=size_1xn,
 176                         initializer=tf.constant_initializer(log_var_init),
 177                         trainable=var_is_trainable)
 178 
 179     if var_is_trainable:
 180       z_logit_var_1xn = tf.exp(z_logvar_1xn)
 181       z_var_1xn = tf.nn.sigmoid(z_logit_var_1xn)*(var_max-var_min) + var_min
 182       z_logvar_1xn = tf.log(z_var_1xn)
 183 
 184     logvar_bxn = tf.tile(z_logvar_1xn, size_bx1)
 185     self.logvar_bxn = logvar_bxn
 186     self.noise_bxn = noise_bxn = tf.random_normal(tf.shape(logvar_bxn))
 187     self.sample_bxn = mean_bxn + tf.exp(0.5 * logvar_bxn) * noise_bxn
 188 
 189   def logp(self, z=None):
 190     """Compute the log-likelihood under the distribution.
 191 
 192     Args:
 193       z (optional): value to compute likelihood for, if None, use sample.
 194 
 195     Returns:
 196       The likelihood of z under the model.
 197     """
 198     if z is None:
 199       z = self.sample
 200 
 201     # This is needed to make sure that the gradients are simple.
 202     # The value of the function shouldn't change.
 203     if z == self.sample_bxn:
 204       return gaussian_pos_log_likelihood(self.mean_bxn, self.logvar_bxn,
 205                                          self.noise_bxn)
 206 
 207     return diag_gaussian_log_likelihood(z, self.mean_bxn, self.logvar_bxn)
 208 
 209   @property
 210   def mean(self):
 211     return self.mean_bxn
 212 
 213   @property
 214   def logvar(self):
 215     return self.logvar_bxn
 216 
 217   @property
 218   def sample(self):
 219     return self.sample_bxn

 222 class DiagonalGaussianFromInput(Gaussian):
 223   """Diagonal Gaussian whose mean and variance are conditioned on other
 224   variables.
 225 
 226   Note: the parameters to convert from input to the learned mean and log
 227   variance are held in this class.
 228   """
 229 
 230   def __init__(self, x_bxu, z_size, name, var_min=0.0):
 231     """Create an input dependent diagonal Gaussian distribution.
 232 
 233     Args:
 234       x: The input tensor from which the mean and variance are computed,
 235         via a linear transformation of x.  I.e.
 236           mu = Wx + b, log(var) = Mx + c
 237       z_size: The size of the distribution.
 238       name:  The name to prefix to learned variables.
 239       var_min (optional): Minimal variance allowed.  This is an additional
 240         way to control the amount of information getting through the stochastic
 241         layer.
 242     """
 243     size_bxn = tf.stack([tf.shape(x_bxu)[0], z_size])
 244     self.mean_bxn = mean_bxn = linear(x_bxu, z_size, name=(name+"/mean"))
 245     logvar_bxn = linear(x_bxu, z_size, name=(name+"/logvar"))
 246     if var_min > 0.0:
 247       logvar_bxn = tf.log(tf.exp(logvar_bxn) + var_min)
 248     self.logvar_bxn = logvar_bxn
 249 
 250     self.noise_bxn = noise_bxn = tf.random_normal(size_bxn)
 251     self.noise_bxn.set_shape([None, z_size])
 252     self.sample_bxn = mean_bxn + tf.exp(0.5 * logvar_bxn) * noise_bxn
 253 
 254   def logp(self, z=None):
 255     """Compute the log-likelihood under the distribution.
 256 
 257     Args:
 258       z (optional): value to compute likelihood for, if None, use sample.
 259 
 260     Returns:
 261       The likelihood of z under the model.
 262     """
 263 
 264     if z is None:
 265       z = self.sample
 266 
 267     # This is needed to make sure that the gradients are simple.
 268     # The value of the function shouldn't change.
 269     if z == self.sample_bxn:
 270       return gaussian_pos_log_likelihood(self.mean_bxn,
 271                                          self.logvar_bxn, self.noise_bxn)
 272 
 273     return diag_gaussian_log_likelihood(z, self.mean_bxn, self.logvar_bxn)
 274 
 275   @property
 276   def mean(self):
 277     return self.mean_bxn
 278 
 279   @property
 280   def logvar(self):
 281     return self.logvar_bxn
 282 
 283   @property
 284   def sample(self):
 285     return self.sample_bxn

 293 class LearnableAutoRegressive1Prior(GaussianProcess):
 294   """AR(1) model where autocorrelation and process variance are learned
 295   parameters.  Assumed zero mean.
 296 
 297   """
 298 
 299   def __init__(self, batch_size, z_size,
 300                autocorrelation_taus, noise_variances,
 301                do_train_prior_ar_atau, do_train_prior_ar_nvar,
 302                num_steps, name):
 303     """Create a learnable autoregressive (1) process.
 304 
 305     Args:
 306       batch_size: The size of the batch, i.e. 0th dim in 2D tensor of samples.
 307       z_size: The dimension of the distribution, i.e. 1st dim in 2D tensor.
 308       autocorrelation_taus: The auto correlation time constant of the AR(1)
 309       process.
 310         A value of 0 is uncorrelated gaussian noise.
 311       noise_variances: The variance of the additive noise, *not* the process
 312         variance.
 313       do_train_prior_ar_atau: Train or leave as constant, the autocorrelation?
 314       do_train_prior_ar_nvar: Train or leave as constant, the noise variance?
 315       num_steps: Number of steps to run the process.
 316       name: The name to prefix to learned TF variables.
 317     """
 318 
 319     # Note the use of the plural in all of these quantities.  This is intended
 320     # to mark that even though a sample z_t from the posterior is thought of a
 321     # single sample of a multidimensional gaussian, the prior is actually
 322     # thought of as U AR(1) processes, where U is the dimension of the inferred
 323     # input.
 324     size_bx1 = tf.stack([batch_size, 1])
 325     size__xu = [None, z_size]
 326     # process variance, the variance at time t over all instantiations of AR(1)
 327     # with these parameters.
 328     log_evar_inits_1xu = tf.expand_dims(tf.log(noise_variances), 0)
 329     self.logevars_1xu = logevars_1xu = \
 330         tf.Variable(log_evar_inits_1xu, name=name+"/logevars", dtype=tf.float32,
 331                     trainable=do_train_prior_ar_nvar)
 332     self.logevars_bxu = logevars_bxu = tf.tile(logevars_1xu, size_bx1)
 333     logevars_bxu.set_shape(size__xu) # tile loses shape
 334 
 335     # \tau, which is the autocorrelation time constant of the AR(1) process
 336     log_atau_inits_1xu = tf.expand_dims(tf.log(autocorrelation_taus), 0)
 337     self.logataus_1xu = logataus_1xu = \
 338         tf.Variable(log_atau_inits_1xu, name=name+"/logatau", dtype=tf.float32,
 339                     trainable=do_train_prior_ar_atau)
 340 
 341     # phi in x_t = \mu + phi x_tm1 + \eps
 342     # phi = exp(-1/tau)
 343     # phi = exp(-1/exp(logtau))
 344     # phi = exp(-exp(-logtau))
 345     phis_1xu = tf.exp(-tf.exp(-logataus_1xu))
 346     self.phis_bxu = phis_bxu = tf.tile(phis_1xu, size_bx1)
 347     phis_bxu.set_shape(size__xu)
 348 
 349     # process noise
 350     # pvar = evar / (1- phi^2)
 351     # logpvar = log ( exp(logevar) / (1 - phi^2) )
 352     # logpvar = logevar - log(1-phi^2)
 353     # logpvar = logevar - (log(1-phi) + log(1+phi))
 354     self.logpvars_1xu = \
 355         logevars_1xu - tf.log(1.0-phis_1xu) - tf.log(1.0+phis_1xu)
 356     self.logpvars_bxu = logpvars_bxu = tf.tile(self.logpvars_1xu, size_bx1)
 357     logpvars_bxu.set_shape(size__xu)
 358 
 359     # process mean (zero but included in for completeness)
 360     self.pmeans_bxu = pmeans_bxu = tf.zeros_like(phis_bxu)
 361 
 362     # For sampling from the prior during de-novo generation.
 363     self.means_t = means_t = [None] * num_steps
 364     self.logvars_t = logvars_t = [None] * num_steps
 365     self.samples_t = samples_t = [None] * num_steps
 366     self.gaussians_t = gaussians_t = [None] * num_steps
 367     sample_bxu = tf.zeros_like(phis_bxu)
 368     for t in range(num_steps):
 369       # process variance used here to make process completely stationary
 370       if t == 0:
 371         logvar_pt_bxu = self.logpvars_bxu
 372       else:
 373         logvar_pt_bxu = self.logevars_bxu
 374 
 375       z_mean_pt_bxu = pmeans_bxu + phis_bxu * sample_bxu
 376       gaussians_t[t] = DiagonalGaussian(batch_size, z_size,
 377                                         mean=z_mean_pt_bxu,
 378                                         logvar=logvar_pt_bxu)
 379       sample_bxu = gaussians_t[t].sample
 380       samples_t[t] = sample_bxu
 381       logvars_t[t] = logvar_pt_bxu
 382       means_t[t] = z_mean_pt_bxu
 383 
 384   def logp_t(self, z_t_bxu, z_tm1_bxu=None):
 385     """Compute the log-likelihood under the distribution for a given time t,
 386     not the whole sequence.
 387 
 388     Args:
 389       z_t_bxu: sample to compute likelihood for at time t.
 390       z_tm1_bxu (optional): sample condition probability of z_t upon.
 391 
 392     Returns:
 393       The likelihood of p_t under the model at time t. i.e.
 394         p(z_t|z_tm1_bxu) = N(z_tm1_bxu * phis, eps^2)
 395 
 396     """
 397     if z_tm1_bxu is None:
 398       return diag_gaussian_log_likelihood(z_t_bxu, self.pmeans_bxu,
 399                                           self.logpvars_bxu)
 400     else:
 401       means_t_bxu = self.pmeans_bxu + self.phis_bxu * z_tm1_bxu
 402       logp_tgtm1_bxu = diag_gaussian_log_likelihood(z_t_bxu,
 403                                                     means_t_bxu,
 404                                                     self.logevars_bxu)
 405       return logp_tgtm1_bxu

 408 class KLCost_GaussianGaussian(object):
 409   """log p(x|z) + KL(q||p) terms for Gaussian posterior and Gaussian prior. See
 410   eqn 10 and Appendix B in VAE for latter term,
 411   http://arxiv.org/abs/1312.6114
 412 
 413   The log p(x|z) term is the reconstruction error under the model.
 414   The KL term represents the penalty for passing information from the encoder
 415   to the decoder.
 416   To sample KL(q||p), we simply sample
 417         ln q - ln p
 418   by drawing samples from q and averaging.
 419   """
 420 
 421   def __init__(self, zs, prior_zs):
 422     """Create a lower bound in three parts, normalized reconstruction
 423     cost, normalized KL divergence cost, and their sum.
 424 
 425     E_q[ln p(z_i | z_{i+1}) / q(z_i | x)
 426        \int q(z) ln p(z) dz = - 0.5 ln(2pi) - 0.5 \sum (ln(sigma_p^2) + \
 427           sigma_q^2 / sigma_p^2 + (mean_p - mean_q)^2 / sigma_p^2)
 428 
 429        \int q(z) ln q(z) dz = - 0.5 ln(2pi) - 0.5 \sum (ln(sigma_q^2) + 1)
 430 
 431     Args:
 432       zs: posterior z ~ q(z|x)
 433       prior_zs: prior zs
 434     """
 435     # L = -KL + log p(x|z), to maximize bound on likelihood
 436     # -L = KL - log p(x|z), to minimize bound on NLL
 437     # so 'KL cost' is postive KL divergence
 438     kl_b = 0.0
 439     for z, prior_z in zip(zs, prior_zs):
 440       assert isinstance(z, Gaussian)
 441       assert isinstance(prior_z, Gaussian)
 442       # ln(2pi) terms cancel
 443       kl_b += 0.5 * tf.reduce_sum(
 444           prior_z.logvar - z.logvar
 445           + tf.exp(z.logvar - prior_z.logvar)
 446           + tf.square((z.mean - prior_z.mean) / tf.exp(0.5 * prior_z.logvar))
 447           - 1.0, [1])
 448 
 449     self.kl_cost_b = kl_b
 450     self.kl_cost = tf.reduce_mean(kl_b)
 451

 453 class KLCost_GaussianGaussianProcessSampled(object):
 454   """ log p(x|z) + KL(q||p) terms for Gaussian posterior and Gaussian process
 455   prior via sampling.
 456 
 457   The log p(x|z) term is the reconstruction error under the model.
 458   The KL term represents the penalty for passing information from the encoder
 459   to the decoder.
 460   To sample KL(q||p), we simply sample
 461         ln q - ln p
 462   by drawing samples from q and averaging.
 463   """
 464 
 465   def __init__(self, post_zs, prior_z_process):
 466     """Create a lower bound in three parts, normalized reconstruction
 467     cost, normalized KL divergence cost, and their sum.
 468 
 469     Args:
 470       post_zs: posterior z ~ q(z|x)
 471       prior_z_process: prior AR(1) process
 472     """
 473     assert len(post_zs) > 1, "GP is for time, need more than 1 time step."
 474     assert isinstance(prior_z_process, GaussianProcess), "Must use GP."
 475 
 476     # L = -KL + log p(x|z), to maximize bound on likelihood
 477     # -L = KL - log p(x|z), to minimize bound on NLL
 478     # so 'KL cost' is postive KL divergence
 479     z0_bxu = post_zs[0].sample
 480     logq_bxu = post_zs[0].logp(z0_bxu)
 481     logp_bxu = prior_z_process.logp_t(z0_bxu)
 482     z_tm1_bxu = z0_bxu
 483     for z_t in post_zs[1:]:
 484       # posterior is independent in time, prior is not
 485       z_t_bxu = z_t.sample
 486       logq_bxu += z_t.logp(z_t_bxu)
 487       logp_bxu += prior_z_process.logp_t(z_t_bxu, z_tm1_bxu)
 488       z_tm1_bxu = z_t_bxu
 489 
 490     kl_bxu = logq_bxu - logp_bxu
 491     kl_b = tf.reduce_sum(kl_bxu, [1])
 492     self.kl_cost_b = kl_b
 493     self.kl_cost = tf.reduce_mean(kl_b)
