  16 """
  17 LFADS - Latent Factor Analysis via Dynamical Systems.
  18 
  19 LFADS is an unsupervised method to decompose time series data into
  20 various factors, such as an initial condition, a generative
  21 dynamical system, control inputs to that generator, and a low
  22 dimensional description of the observed data, called the factors.
  23 Additionally, the observations have a noise model (in this case
  24 Poisson), so a denoised version of the observations is also created
  25 (e.g. underlying rates of a Poisson distribution given the observed
  26 event counts).
  27 
  28 The main data structure being passed around is a dataset.  This is a dictionary
  29 of data dictionaries.
  30 
  31 DATASET: The top level dictionary is simply name (string -> dictionary).
  32 The nested dictionary is the DATA DICTIONARY, which has the following keys:
  33   'train_data' and 'valid_data', whose values are the corresponding training
  34     and validation data with shape
  35     ExTxD, E - # examples, T - # time steps, D - # dimensions in data.
  36   The data dictionary also has a few more keys:
  37     'train_ext_input' and 'valid_ext_input', if there are know external inputs
  38       to the system being modeled, these take on dimensions:
  39       ExTxI, E - # examples, T - # time steps, I = # dimensions in input.
  40    'alignment_matrix_cxf' - If you are using multiple days data, it's possible
  41      that one can align the channels (see manuscript).  If so each dataset will
  42      contain this matrix, which will be used for both the input adapter and the
  43      output adapter for each dataset. These matrices, if provided, must be of
  44      size [data_dim x factors] where data_dim is the number of neurons recorded
  45      on that day, and factors is chosen and set through the '--factors' flag.
  46    'alignment_bias_c' - See alignment_matrix_cxf.  This bias will used to
  47      the offset for the alignment transformation.  It will *subtract* off the
  48      bias from the data, so pca style inits can align factors across sessions.
  49 
  50 
  51   If one runs LFADS on data where the true rates are known for some trials,
  52   (say simulated, testing data, as in the example shipped with the paper), then
  53   one can add three more fields for plotting purposes.  These are 'train_truth'
  54   and 'valid_truth', and 'conversion_factor'.  These have the same dimensions as
  55   'train_data', and 'valid_data' but represent the underlying rates of the
  56   observations.  Finally, if one needs to convert scale for plotting the true
  57   underlying firing rates, there is the 'conversion_factor' key.
  58 """

  79 class GRU(object):
  80   """Gated Recurrent Unit cell (cf. http://arxiv.org/abs/1406.1078).
  81 
  82   """
  83   def __init__(self, num_units, forget_bias=1.0, weight_scale=1.0,
  84                clip_value=np.inf, collections=None):
  85     """Create a GRU object.
  86 
  87     Args:
  88       num_units: Number of units in the GRU
  89       forget_bias (optional): Hack to help learning.
  90       weight_scale (optional): weights are scaled by ws/sqrt(#inputs), with
  91        ws being the weight scale.
  92       clip_value (optional): if the recurrent values grow above this value,
  93         clip them.
  94       collections (optional): List of additonal collections variables should
  95         belong to.
  96     """
  97     self._num_units = num_units
  98     self._forget_bias = forget_bias
  99     self._weight_scale = weight_scale
 100     self._clip_value = clip_value
 101     self._collections = collections
 102 
 103   @property
 104   def state_size(self):
 105     return self._num_units
 106 
 107   @property
 108   def output_size(self):
 109     return self._num_units
 110 
 111   @property
 112   def state_multiplier(self):
 113     return 1
 114 
 115   def output_from_state(self, state):
 116     """Return the output portion of the state."""
 117     return state
 118 
 119   def __call__(self, inputs, state, scope=None):
 120     """Gated recurrent unit (GRU) function.
 121 
 122     Args:
 123       inputs: A 2D batch x input_dim tensor of inputs.
 124       state: The previous state from the last time step.
 125       scope (optional): TF variable scope for defined GRU variables.
 126 
 127     Returns:
 128       A tuple (state, state), where state is the newly computed state at time t.
 129       It is returned twice to respect an interface that works for LSTMs.
 130     """
 131 
 132     x = inputs
 133     h = state
 134     if inputs is not None:
 135       xh = tf.concat(axis=1, values=[x, h])
 136     else:
 137       xh = h
 138 
 139     with tf.variable_scope(scope or type(self).__name__):  # "GRU"
 140       with tf.variable_scope("Gates"):  # Reset gate and update gate.
 141         # We start with bias of 1.0 to not reset and not update.
 142         r, u = tf.split(axis=1, num_or_size_splits=2, value=linear(xh,
 143                                      2 * self._num_units,
 144                                      alpha=self._weight_scale,
 145                                      name="xh_2_ru",
 146                                      collections=self._collections))
 147         r, u = tf.sigmoid(r), tf.sigmoid(u + self._forget_bias)
 148       with tf.variable_scope("Candidate"):
 149         xrh = tf.concat(axis=1, values=[x, r * h])
 150         c = tf.tanh(linear(xrh, self._num_units, name="xrh_2_c",
 151                            collections=self._collections))
 152       new_h = u * h + (1 - u) * c
 153       new_h = tf.clip_by_value(new_h, -self._clip_value, self._clip_value)
 154 
 155     return new_h, new_h

 158 class GenGRU(object):
 159   """Gated Recurrent Unit cell (cf. http://arxiv.org/abs/1406.1078).
 160 
 161   This version is specialized for the generator, but isn't as fast, so
 162   we have two.  Note this allows for l2 regularization on the recurrent
 163   weights, but also implicitly rescales the inputs via the 1/sqrt(input)
 164   scaling in the linear helper routine to be large magnitude, if there are
 165   fewer inputs than recurrent state.
 166 
 167   """
 168   def __init__(self, num_units, forget_bias=1.0,
 169                input_weight_scale=1.0, rec_weight_scale=1.0, clip_value=np.inf,
 170                input_collections=None, recurrent_collections=None):
 171     """Create a GRU object.
 172 
 173     Args:
 174       num_units: Number of units in the GRU
 175       forget_bias (optional): Hack to help learning.
 176       input_weight_scale (optional): weights are scaled ws/sqrt(#inputs), with
 177         ws being the weight scale.
 178       rec_weight_scale (optional): weights are scaled ws/sqrt(#inputs),
 179         with ws being the weight scale.
 180       clip_value (optional): if the recurrent values grow above this value,
 181         clip them.
 182       input_collections (optional): List of additonal collections variables
 183         that input->rec weights should belong to.
 184       recurrent_collections (optional): List of additonal collections variables
 185         that rec->rec weights should belong to.
 186     """
 187     self._num_units = num_units
 188     self._forget_bias = forget_bias
 189     self._input_weight_scale = input_weight_scale
 190     self._rec_weight_scale = rec_weight_scale
 191     self._clip_value = clip_value
 192     self._input_collections = input_collections
 193     self._rec_collections = recurrent_collections
 194 
 195   @property
 196   def state_size(self):
 197     return self._num_units
 198 
 199   @property
 200   def output_size(self):
 201     return self._num_units
 202 
 203   @property
 204   def state_multiplier(self):
 205     return 1
 206 
 207   def output_from_state(self, state):
 208     """Return the output portion of the state."""
 209     return state
 210 
 211   def __call__(self, inputs, state, scope=None):
 212     """Gated recurrent unit (GRU) function.
 213 
 214     Args:
 215       inputs: A 2D batch x input_dim tensor of inputs.
 216       state: The previous state from the last time step.
 217       scope (optional): TF variable scope for defined GRU variables.
 218 
 219     Returns:
 220       A tuple (state, state), where state is the newly computed state at time t.
 221       It is returned twice to respect an interface that works for LSTMs.
 222     """
 223 
 224     x = inputs
 225     h = state
 226     with tf.variable_scope(scope or type(self).__name__):  # "GRU"
 227       with tf.variable_scope("Gates"):  # Reset gate and update gate.
 228         # We start with bias of 1.0 to not reset and not update.
 229         r_x = u_x = 0.0
 230         if x is not None:
 231           r_x, u_x = tf.split(axis=1, num_or_size_splits=2, value=linear(x,
 232                                            2 * self._num_units,
 233                                            alpha=self._input_weight_scale,
 234                                            do_bias=False,
 235                                            name="x_2_ru",
 236                                            normalized=False,
 237                                            collections=self._input_collections))
 238 
 239         r_h, u_h = tf.split(axis=1, num_or_size_splits=2, value=linear(h,
 240                                          2 * self._num_units,
 241                                          do_bias=True,
 242                                          alpha=self._rec_weight_scale,
 243                                          name="h_2_ru",
 244                                          collections=self._rec_collections))
 245         r = r_x + r_h
 246         u = u_x + u_h
 247         r, u = tf.sigmoid(r), tf.sigmoid(u + self._forget_bias)
 248 
 249       with tf.variable_scope("Candidate"):
 250         c_x = 0.0
 251         if x is not None:
 252           c_x = linear(x, self._num_units, name="x_2_c", do_bias=False,
 253                        alpha=self._input_weight_scale,
 254                        normalized=False,
 255                        collections=self._input_collections)
 256         c_rh = linear(r*h, self._num_units, name="rh_2_c", do_bias=True,
 257                      alpha=self._rec_weight_scale,
 258                      collections=self._rec_collections)
 259         c = tf.tanh(c_x + c_rh)
 260 
 261       new_h = u * h + (1 - u) * c
 262       new_h = tf.clip_by_value(new_h, -self._clip_value, self._clip_value)
 263 
 264     return new_h, new_h

 267 class LFADS(object):
 268   """LFADS - Latent Factor Analysis via Dynamical Systems.
 269 
 270   LFADS is an unsupervised method to decompose time series data into
 271   various factors, such as an initial condition, a generative
 272   dynamical system, inferred inputs to that generator, and a low
 273   dimensional description of the observed data, called the factors.
 274   Additoinally, the observations have a noise model (in this case
 275   Poisson), so a denoised version of the observations is also created
 276   (e.g. underlying rates of a Poisson distribution given the observed
 277   event counts).
 278   """
 279 
 280   def __init__(self, hps, kind="train", datasets=None):
 281     """Create an LFADS model.
 282 
 283        train - a model for training, sampling of posteriors is used
 284        posterior_sample_and_average - sample from the posterior, this is used
 285          for evaluating the expected value of the outputs of LFADS, given a
 286          specific input, by averaging over multiple samples from the approx
 287          posterior.  Also used for the lower bound on the negative
 288          log-likelihood using IWAE error (Importance Weighed Auto-encoder).
 289          This is the denoising operation.
 290        prior_sample - a model for generation - sampling from priors is used
 291 
 292     Args:
 293       hps: The dictionary of hyper parameters.
 294       kind: the type of model to build (see above).
 295       datasets: a dictionary of named data_dictionaries, see top of lfads.py
 296     """
 297     print("Building graph...")
 298     all_kinds = ['train', 'posterior_sample_and_average', 'posterior_push_mean',
 299                  'prior_sample']
 300     assert kind in all_kinds, 'Wrong kind'
 301     if hps.feedback_factors_or_rates == "rates":
 302       assert len(hps.dataset_names) == 1, \
 303       "Multiple datasets not supported for rate feedback."
 304     num_steps = hps.num_steps
 305     ic_dim = hps.ic_dim
 306     co_dim = hps.co_dim
 307     ext_input_dim = hps.ext_input_dim
 308     cell_class = GRU
 309     gen_cell_class = GenGRU
 310 
 311     def makelambda(v):          # Used with tf.case
 312       return lambda: v
 313 
 314     # Define the data placeholder, and deal with all parts of the graph
 315     # that are dataset dependent.
 316     self.dataName = tf.placeholder(tf.string, shape=())
 317     # The batch_size to be inferred from data, as normal.
 318     # Additionally, the data_dim will be inferred as well, allowing for a
 319     # single placeholder for all datasets, regardless of data dimension.
 320     if hps.output_dist == 'poisson':
 321       # Enforce correct dtype
 322       assert np.issubdtype(
 323           datasets[hps.dataset_names[0]]['train_data'].dtype, int), \
 324           "Data dtype must be int for poisson output distribution"
 325       data_dtype = tf.int32
 326     elif hps.output_dist == 'gaussian':
 327       assert np.issubdtype(
 328           datasets[hps.dataset_names[0]]['train_data'].dtype, float), \
 329           "Data dtype must be float for gaussian output dsitribution"
 330       data_dtype = tf.float32
 331     else:
 332       assert False, "NIY"
 333     self.dataset_ph = dataset_ph = tf.placeholder(data_dtype,
 334                                                   [None, num_steps, None],
 335                                                   name="data")
 336     self.train_step = tf.get_variable("global_step", [], tf.int64,
 337                                       tf.zeros_initializer(),
 338                                       trainable=False)
 339     self.hps = hps
 340     ndatasets = hps.ndatasets
 341     factors_dim = hps.factors_dim
 342     self.preds = preds = [None] * ndatasets
 343     self.fns_in_fac_Ws = fns_in_fac_Ws = [None] * ndatasets
 344     self.fns_in_fatcor_bs = fns_in_fac_bs = [None] * ndatasets
 345     self.fns_out_fac_Ws = fns_out_fac_Ws = [None] * ndatasets
 346     self.fns_out_fac_bs = fns_out_fac_bs = [None] * ndatasets
 347     self.datasetNames = dataset_names = hps.dataset_names
 348     self.ext_inputs = ext_inputs = None
 349 
 350     if len(dataset_names) == 1:  # single session
 351       if 'alignment_matrix_cxf' in datasets[dataset_names[0]].keys():
 352         used_in_factors_dim = factors_dim
 353         in_identity_if_poss = False
 354       else:
 355         used_in_factors_dim = hps.dataset_dims[dataset_names[0]]
 356         in_identity_if_poss = True
 357     else:  # multisession
 358       used_in_factors_dim = factors_dim
 359       in_identity_if_poss = False
 360 
 361     for d, name in enumerate(dataset_names):
 362       data_dim = hps.dataset_dims[name]
 363       in_mat_cxf = None
 364       in_bias_1xf = None
 365       align_bias_1xc = None
 366 
 367       if datasets and 'alignment_matrix_cxf' in datasets[name].keys():
 368         dataset = datasets[name]
 369         if hps.do_train_readin:
 370             print("Initializing trainable readin matrix with alignment matrix" \
 371                   " provided for dataset:", name)
 372         else:
 373             print("Setting non-trainable readin matrix to alignment matrix" \
 374                   " provided for dataset:", name)
 375         in_mat_cxf = dataset['alignment_matrix_cxf'].astype(np.float32)
 376         if in_mat_cxf.shape != (data_dim, factors_dim):
 377           raise ValueError("""Alignment matrix must have dimensions %d x %d

 379                            (data_dim, factors_dim, in_mat_cxf.shape[0],
 380                             in_mat_cxf.shape[1]))
 381       if datasets and 'alignment_bias_c' in datasets[name].keys():
 382         dataset = datasets[name]
 383         if hps.do_train_readin:
 384           print("Initializing trainable readin bias with alignment bias " \

 386         else:
 387           print("Setting non-trainable readin bias to alignment bias " \

 402           # only add to IO transformations collection only if we want it to be

 404           # when do_train_io_only
 405           collections_readin=['IO_transformations']

 407           collections_readin=None

 411                                mat_init_value=in_mat_cxf,
 412                                bias_init_value=in_bias_1xf,
 413                                identity_if_possible=in_identity_if_poss,
 414                                normalized=False, name="x_2_infac_"+name,
 415                                collections=collections_readin,
 416                                trainable=hps.do_train_readin)
 417       in_fac_W, in_fac_b = in_fac_lin
 418       fns_in_fac_Ws[d] = makelambda(in_fac_W)
 419       fns_in_fac_bs[d] = makelambda(in_fac_b)

 421     with tf.variable_scope("glm"):
 422       out_identity_if_poss = False
 423       if len(dataset_names) == 1 and \
 424           factors_dim == hps.dataset_dims[dataset_names[0]]:
 425         out_identity_if_poss = True
 426       for d, name in enumerate(dataset_names):
 427         data_dim = hps.dataset_dims[name]
 428         in_mat_cxf = None
 429         if datasets and 'alignment_matrix_cxf' in datasets[name].keys():
 430           dataset = datasets[name]
 431           in_mat_cxf = dataset['alignment_matrix_cxf'].astype(np.float32)
 432 
 433         if datasets and 'alignment_bias_c' in datasets[name].keys():
 434           dataset = datasets[name]
 435           align_bias_c = dataset['alignment_bias_c'].astype(np.float32)
 436           align_bias_1xc = np.expand_dims(align_bias_c, axis=0)
 437 
 438         out_mat_fxc = None
 439         out_bias_1xc = None
 440         if in_mat_cxf is not None:
 441             out_mat_fxc = in_mat_cxf.T
 442         if align_bias_1xc is not None:
 443           out_bias_1xc = align_bias_1xc
 444 
 445         if hps.output_dist == 'poisson':
 446           out_fac_lin = init_linear(factors_dim, data_dim, do_bias=True,
 447                                     mat_init_value=out_mat_fxc,
 448                                     bias_init_value=out_bias_1xc,
 449                                     identity_if_possible=out_identity_if_poss,
 450                                     normalized=False,
 451                                     name="fac_2_logrates_"+name,
 452                                     collections=['IO_transformations'])
 453           out_fac_W, out_fac_b = out_fac_lin
 454 
 455         elif hps.output_dist == 'gaussian':
 456           out_fac_lin_mean = \
 457               init_linear(factors_dim, data_dim, do_bias=True,
 458                           mat_init_value=out_mat_fxc,
 459                           bias_init_value=out_bias_1xc,
 460                           normalized=False,
 461                           name="fac_2_means_"+name,
 462                           collections=['IO_transformations'])
 463           out_fac_W_mean, out_fac_b_mean = out_fac_lin_mean
 464 
 465           mat_init_value = np.zeros([factors_dim, data_dim]).astype(np.float32)
 466           bias_init_value = np.ones([1, data_dim]).astype(np.float32)
 467           out_fac_lin_logvar = \
 468               init_linear(factors_dim, data_dim, do_bias=True,
 469                           mat_init_value=mat_init_value,
 470                           bias_init_value=bias_init_value,
 471                           normalized=False,
 472                           name="fac_2_logvars_"+name,
 473                           collections=['IO_transformations'])
 474           out_fac_W_mean, out_fac_b_mean = out_fac_lin_mean
 475           out_fac_W_logvar, out_fac_b_logvar = out_fac_lin_logvar
 476           out_fac_W = tf.concat(
 477               axis=1, values=[out_fac_W_mean, out_fac_W_logvar])
 478           out_fac_b = tf.concat(
 479               axis=1, values=[out_fac_b_mean, out_fac_b_logvar])
 480         else:
 481           assert False, "NIY"
 482 
 483         preds[d] = tf.equal(tf.constant(name), self.dataName)
 484         data_dim = hps.dataset_dims[name]
 485         fns_out_fac_Ws[d] = makelambda(out_fac_W)
 486         fns_out_fac_bs[d] =  makelambda(out_fac_b)
 487 
 488     pf_pairs_in_fac_Ws = zip(preds, fns_in_fac_Ws)
 489     pf_pairs_in_fac_bs = zip(preds, fns_in_fac_bs)
 490     pf_pairs_out_fac_Ws = zip(preds, fns_out_fac_Ws)
 491     pf_pairs_out_fac_bs = zip(preds, fns_out_fac_bs)
 492 
 493     this_in_fac_W = tf.case(pf_pairs_in_fac_Ws, exclusive=True)
 494     this_in_fac_b = tf.case(pf_pairs_in_fac_bs, exclusive=True)
 495     this_out_fac_W = tf.case(pf_pairs_out_fac_Ws, exclusive=True)
 496     this_out_fac_b = tf.case(pf_pairs_out_fac_bs, exclusive=True)
 497 
 498     # External inputs (not changing by dataset, by definition).
 499     if hps.ext_input_dim > 0:
 500       self.ext_input = tf.placeholder(tf.float32,
 501                                       [None, num_steps, ext_input_dim],
 502                                       name="ext_input")
 503     else:
 504       self.ext_input = None
 505     ext_input_bxtxi = self.ext_input
 506 
 507     self.keep_prob = keep_prob = tf.placeholder(tf.float32, [], "keep_prob")
 508     self.batch_size = batch_size = int(hps.batch_size)
 509     self.learning_rate = tf.Variable(float(hps.learning_rate_init),
 510                                      trainable=False, name="learning_rate")
 511     self.learning_rate_decay_op = self.learning_rate.assign(
 512         self.learning_rate * hps.learning_rate_decay_factor)
 513 
 514     # Dropout the data.
 515     dataset_do_bxtxd = tf.nn.dropout(tf.to_float(dataset_ph), keep_prob)
 516     if hps.ext_input_dim > 0:
 517       ext_input_do_bxtxi = tf.nn.dropout(ext_input_bxtxi, keep_prob)
 518     else:
 519       ext_input_do_bxtxi = None

 522     def encode_data(dataset_bxtxd, enc_cell, name, forward_or_reverse,
 523                 num_steps_to_encode):
 524       """Encode data for LFADS
 525       Args:
 526         dataset_bxtxd - the data to encode, as a 3 tensor, with dims
 527           time x batch x data dims.
 528         enc_cell: encoder cell
 529         name: name of encoder
 530         forward_or_reverse: string, encode in forward or reverse direction
 531         num_steps_to_encode: number of steps to  encode, 0:num_steps_to_encode
 532       Returns:
 533         encoded data as a list with num_steps_to_encode items, in order
 534       """
 535       if forward_or_reverse == "forward":
 536         dstr = "_fwd"
 537         time_fwd_or_rev = range(num_steps_to_encode)
 538       else:
 539         dstr = "_rev"
 540         time_fwd_or_rev = reversed(range(num_steps_to_encode))
 541 
 542       with tf.variable_scope(name+"_enc"+dstr, reuse=False):
 543         enc_state = tf.tile(
 544             tf.Variable(tf.zeros([1, enc_cell.state_size]),
 545                         name=name+"_enc_t0"+dstr), tf.stack([batch_size, 1]))
 546         enc_state.set_shape([None, enc_cell.state_size]) # tile loses shape
 547 
 548       enc_outs = [None] * num_steps_to_encode
 549       for i, t in enumerate(time_fwd_or_rev):
 550         with tf.variable_scope(name+"_enc"+dstr, reuse=True if i > 0 else None):
 551           dataset_t_bxd = dataset_bxtxd[:,t,:]
 552           in_fac_t_bxf = tf.matmul(dataset_t_bxd, this_in_fac_W) + this_in_fac_b
 553           in_fac_t_bxf.set_shape([None, used_in_factors_dim])
 554           if ext_input_dim > 0 and not hps.inject_ext_input_to_gen:
 555             ext_input_t_bxi = ext_input_do_bxtxi[:,t,:]
 556             enc_input_t_bxfpe = tf.concat(
 557                 axis=1, values=[in_fac_t_bxf, ext_input_t_bxi])
 558           else:
 559             enc_input_t_bxfpe = in_fac_t_bxf
 560           enc_out, enc_state = enc_cell(enc_input_t_bxfpe, enc_state)
 561           enc_outs[t] = enc_out
 562 
 563       return enc_outs
 564 
 565     # Encode initial condition means and variances
 566     # ([x_T, x_T-1, ... x_0] and [x_0, x_1, ... x_T] -> g0/c0)
 567     self.ic_enc_fwd = [None] * num_steps
 568     self.ic_enc_rev = [None] * num_steps
 569     if ic_dim > 0:
 570       enc_ic_cell = cell_class(hps.ic_enc_dim,
 571                                weight_scale=hps.cell_weight_scale,
 572                                clip_value=hps.cell_clip_value)
 573       ic_enc_fwd = encode_data(dataset_do_bxtxd, enc_ic_cell,
 574                                "ic", "forward",
 575                                hps.num_steps_for_gen_ic)
 576       ic_enc_rev = encode_data(dataset_do_bxtxd, enc_ic_cell,
 577                                "ic", "reverse",
 578                                hps.num_steps_for_gen_ic)
 579       self.ic_enc_fwd = ic_enc_fwd
 580       self.ic_enc_rev = ic_enc_rev
 581 
 582     # Encoder control input means and variances, bi-directional encoding so:
 583     # ([x_T, x_T-1, ..., x_0] and [x_0, x_1 ... x_T] -> u_t)
 584     self.ci_enc_fwd = [None] * num_steps
 585     self.ci_enc_rev = [None] * num_steps
 586     if co_dim > 0:
 587       enc_ci_cell = cell_class(hps.ci_enc_dim,
 588                                weight_scale=hps.cell_weight_scale,
 589                                clip_value=hps.cell_clip_value)
 590       ci_enc_fwd = encode_data(dataset_do_bxtxd, enc_ci_cell,
 591                                "ci", "forward",
 592                                hps.num_steps)
 593       if hps.do_causal_controller:
 594         ci_enc_rev = None
 595       else:
 596         ci_enc_rev = encode_data(dataset_do_bxtxd, enc_ci_cell,
 597                                  "ci", "reverse",
 598                                  hps.num_steps)
 599       self.ci_enc_fwd = ci_enc_fwd
 600       self.ci_enc_rev = ci_enc_rev
 601 
 602     # STOCHASTIC LATENT VARIABLES, priors and posteriors
 603     # (initial conditions g0, and control inputs, u_t)
 604     # Note that zs represent all the stochastic latent variables.
 605     with tf.variable_scope("z", reuse=False):
 606       self.prior_zs_g0 = None
 607       self.posterior_zs_g0 = None
 608       self.g0s_val = None
 609       if ic_dim > 0:
 610         self.prior_zs_g0 = \
 611             LearnableDiagonalGaussian(batch_size, ic_dim, name="prior_g0",
 612                                       mean_init=0.0,
 613                                       var_min=hps.ic_prior_var_min,
 614                                       var_init=hps.ic_prior_var_scale,
 615                                       var_max=hps.ic_prior_var_max)
 616         ic_enc = tf.concat(axis=1, values=[ic_enc_fwd[-1], ic_enc_rev[0]])
 617         ic_enc = tf.nn.dropout(ic_enc, keep_prob)
 618         self.posterior_zs_g0 = \
 619             DiagonalGaussianFromInput(ic_enc, ic_dim, "ic_enc_2_post_g0",
 620                                       var_min=hps.ic_post_var_min)
 621         if kind in ["train", "posterior_sample_and_average",
 622                     "posterior_push_mean"]:
 623           zs_g0 = self.posterior_zs_g0
 624         else:
 625           zs_g0 = self.prior_zs_g0
 626         if kind in ["train", "posterior_sample_and_average", "prior_sample"]:
 627           self.g0s_val = zs_g0.sample
 628         else:
 629           self.g0s_val = zs_g0.mean
 630 
 631       # Priors for controller, 'co' for controller output
 632       self.prior_zs_co = prior_zs_co = [None] * num_steps
 633       self.posterior_zs_co = posterior_zs_co = [None] * num_steps
 634       self.zs_co = zs_co = [None] * num_steps
 635       self.prior_zs_ar_con = None
 636       if co_dim > 0:
 637         # Controller outputs
 638         autocorrelation_taus = [hps.prior_ar_atau for x in range(hps.co_dim)]
 639         noise_variances = [hps.prior_ar_nvar for x in range(hps.co_dim)]
 640         self.prior_zs_ar_con = prior_zs_ar_con = \
 641             LearnableAutoRegressive1Prior(batch_size, hps.co_dim,
 642                                           autocorrelation_taus,
 643                                           noise_variances,
 644                                           hps.do_train_prior_ar_atau,
 645                                           hps.do_train_prior_ar_nvar,
 646                                           num_steps, "u_prior_ar1")
 647 
 648     # CONTROLLER -> GENERATOR -> RATES
 649     # (u(t) -> gen(t) -> factors(t) -> rates(t) -> p(x_t|z_t) )
 650     self.controller_outputs = u_t = [None] * num_steps
 651     self.con_ics = con_state = None
 652     self.con_states = con_states = [None] * num_steps
 653     self.con_outs = con_outs = [None] * num_steps
 654     self.gen_inputs = gen_inputs = [None] * num_steps
 655     if co_dim > 0:
 656       # gen_cell_class here for l2 penalty recurrent weights
 657       # didn't split the cell_weight scale here, because I doubt it matters
 658       con_cell = gen_cell_class(hps.con_dim,
 659                                 input_weight_scale=hps.cell_weight_scale,
 660                                 rec_weight_scale=hps.cell_weight_scale,
 661                                 clip_value=hps.cell_clip_value,
 662                                 recurrent_collections=['l2_con_reg'])
 663       with tf.variable_scope("con", reuse=False):
 664         self.con_ics = tf.tile(
 665             tf.Variable(tf.zeros([1, hps.con_dim*con_cell.state_multiplier]),
 666                         name="c0"),
 667             tf.stack([batch_size, 1]))
 668         self.con_ics.set_shape([None, con_cell.state_size]) # tile loses shape
 669         con_states[-1] = self.con_ics
 670 
 671     gen_cell = gen_cell_class(hps.gen_dim,
 672                               input_weight_scale=hps.gen_cell_input_weight_scale,
 673                               rec_weight_scale=hps.gen_cell_rec_weight_scale,
 674                               clip_value=hps.cell_clip_value,
 675                               recurrent_collections=['l2_gen_reg'])
 676     with tf.variable_scope("gen", reuse=False):
 677       if ic_dim == 0:
 678         self.gen_ics = tf.tile(
 679               tf.Variable(tf.zeros([1, gen_cell.state_size]), name="g0"),
 680               tf.stack([batch_size, 1]))
 681       else:
 682         self.gen_ics = linear(self.g0s_val, gen_cell.state_size,
 683                               identity_if_possible=True,
 684                               name="g0_2_gen_ic")
 685 
 686       self.gen_states = gen_states = [None] * num_steps
 687       self.gen_outs = gen_outs = [None] * num_steps
 688       gen_states[-1] = self.gen_ics
 689       gen_outs[-1] = gen_cell.output_from_state(gen_states[-1])
 690       self.factors = factors = [None] * num_steps
 691       factors[-1] = linear(gen_outs[-1], factors_dim, do_bias=False,
 692                            normalized=True, name="gen_2_fac")
 693 
 694     self.rates = rates = [None] * num_steps
 695     # rates[-1] is collected to potentially feed back to controller
 696     with tf.variable_scope("glm", reuse=False):
 697       if hps.output_dist == 'poisson':
 698         log_rates_t0 = tf.matmul(factors[-1], this_out_fac_W) + this_out_fac_b
 699         log_rates_t0.set_shape([None, None])
 700         rates[-1] = tf.exp(log_rates_t0) # rate
 701         rates[-1].set_shape([None, hps.dataset_dims[hps.dataset_names[0]]])
 702       elif hps.output_dist == 'gaussian':
 703         mean_n_logvars = tf.matmul(factors[-1],this_out_fac_W) + this_out_fac_b
 704         mean_n_logvars.set_shape([None, None])
 705         means_t_bxd, logvars_t_bxd = tf.split(axis=1, num_or_size_splits=2,
 706                                               value=mean_n_logvars)
 707         rates[-1] = means_t_bxd
 708       else:
 709         assert False, "NIY"
 710 
 711     # We support multiple output distributions, for example Poisson, and also
 712     # Gaussian. In these two cases respectively, there are one and two
 713     # parameters (rates vs. mean and variance).  So the output_dist_params
 714     # tensor will variable sizes via tf.concat and tf.split, along the 1st
 715     # dimension. So in the case of gaussian, for example, it'll be
 716     # batch x (D+D), where each D dims is the mean, and then variances,
 717     # respectively. For a distribution with 3 parameters, it would be
 718     # batch x (D+D+D).
 719     self.output_dist_params = dist_params = [None] * num_steps
 720     self.log_p_xgz_b = log_p_xgz_b = 0.0  # log P(x|z)
 721     for t in range(num_steps):
 722       # Controller
 723       if co_dim > 0:
 724         # Build inputs for controller
 725         tlag = t - hps.controller_input_lag
 726         if tlag < 0:
 727           con_in_f_t = tf.zeros_like(ci_enc_fwd[0])
 728         else:
 729           con_in_f_t = ci_enc_fwd[tlag]
 730         if hps.do_causal_controller:
 731           # If controller is causal (wrt to data generation process), then it
 732           # cannot see future data.  Thus, excluding ci_enc_rev[t] is obvious.
 733           # Less obvious is the need to exclude factors[t-1].  This arises
 734           # because information flows from g0 through factors to the controller
 735           # input.  The g0 encoding is backwards, so we must necessarily exclude
 736           # the factors in order to keep the controller input purely from a
 737           # forward encoding (however unlikely it is that
 738           # g0->factors->controller channel might actually be used in this way).
 739           con_in_list_t = [con_in_f_t]
 740         else:
 741           tlag_rev = t + hps.controller_input_lag
 742           if tlag_rev >= num_steps:
 743             # better than zeros
 744             con_in_r_t = tf.zeros_like(ci_enc_rev[0])
 745           else:
 746             con_in_r_t = ci_enc_rev[tlag_rev]
 747           con_in_list_t = [con_in_f_t, con_in_r_t]
 748 
 749         if hps.do_feed_factors_to_controller:
 750           if hps.feedback_factors_or_rates == "factors":
 751             con_in_list_t.append(factors[t-1])
 752           elif hps.feedback_factors_or_rates == "rates":
 753             con_in_list_t.append(rates[t-1])
 754           else:
 755             assert False, "NIY"
 756 
 757         con_in_t = tf.concat(axis=1, values=con_in_list_t)
 758         con_in_t = tf.nn.dropout(con_in_t, keep_prob)
 759         with tf.variable_scope("con", reuse=True if t > 0 else None):
 760           con_outs[t], con_states[t] = con_cell(con_in_t, con_states[t-1])
 761           posterior_zs_co[t] = \
 762             DiagonalGaussianFromInput(con_outs[t], co_dim,
 763                                       name="con_to_post_co")
 764         if kind == "train":
 765           u_t[t] = posterior_zs_co[t].sample
 766         elif kind == "posterior_sample_and_average":
 767           u_t[t] = posterior_zs_co[t].sample
 768         elif kind == "posterior_push_mean":
 769           u_t[t] = posterior_zs_co[t].mean
 770         else:
 771           u_t[t] = prior_zs_ar_con.samples_t[t]
 772 
 773       # Inputs to the generator (controller output + external input)
 774       if ext_input_dim > 0 and hps.inject_ext_input_to_gen:
 775         ext_input_t_bxi = ext_input_do_bxtxi[:,t,:]
 776         if co_dim > 0:
 777           gen_inputs[t] = tf.concat(axis=1, values=[u_t[t], ext_input_t_bxi])
 778         else:
 779           gen_inputs[t] = ext_input_t_bxi
 780       else:
 781         gen_inputs[t] = u_t[t]
 782 
 783       # Generator
 784       data_t_bxd = dataset_ph[:,t,:]
 785       with tf.variable_scope("gen", reuse=True if t > 0 else None):
 786         gen_outs[t], gen_states[t] = gen_cell(gen_inputs[t], gen_states[t-1])
 787         gen_outs[t] = tf.nn.dropout(gen_outs[t], keep_prob)
 788       with tf.variable_scope("gen", reuse=True): # ic defined it above
 789         factors[t] = linear(gen_outs[t], factors_dim, do_bias=False,
 790                             normalized=True, name="gen_2_fac")
 791       with tf.variable_scope("glm", reuse=True if t > 0 else None):
 792         if hps.output_dist == 'poisson':
 793           log_rates_t = tf.matmul(factors[t], this_out_fac_W) + this_out_fac_b
 794           log_rates_t.set_shape([None, None])
 795           rates[t] = dist_params[t] = tf.exp(tf.clip_by_value(log_rates_t, -hps._clip_value, hps._clip_value)) # rates feed back
 796           rates[t].set_shape([None, hps.dataset_dims[hps.dataset_names[0]]])
 797           loglikelihood_t = Poisson(log_rates_t).logp(data_t_bxd)
 798 
 799         elif hps.output_dist == 'gaussian':
 800           mean_n_logvars = tf.matmul(factors[t],this_out_fac_W) + this_out_fac_b
 801           mean_n_logvars.set_shape([None, None])
 802           means_t_bxd, logvars_t_bxd = tf.split(axis=1, num_or_size_splits=2,
 803                                                 value=mean_n_logvars)
 804           rates[t] = means_t_bxd # rates feed back to controller
 805           dist_params[t] = tf.concat(
 806               axis=1, values=[means_t_bxd, tf.exp(tf.clip_by_value(logvars_t_bxd, -hps._clip_value, hps._clip_value))])
 807           loglikelihood_t = \
 808               diag_gaussian_log_likelihood(data_t_bxd,
 809                                            means_t_bxd, logvars_t_bxd)
 810         else:
 811           assert False, "NIY"
 812 
 813         log_p_xgz_b += tf.reduce_sum(loglikelihood_t, [1])
 814 
 815     # Correlation of inferred inputs cost.
 816     self.corr_cost = tf.constant(0.0)
 817     if hps.co_mean_corr_scale > 0.0:
 818       all_sum_corr = []
 819       for i in range(hps.co_dim):
 820         for j in range(i+1, hps.co_dim):
 821           sum_corr_ij = tf.constant(0.0)
 822           for t in range(num_steps):
 823             u_mean_t = posterior_zs_co[t].mean
 824             sum_corr_ij += u_mean_t[:,i]*u_mean_t[:,j]
 825           all_sum_corr.append(0.5 * tf.square(sum_corr_ij))
 826       self.corr_cost = tf.reduce_mean(all_sum_corr) # div by batch and by n*(n-1)/2 pairs
 827 
 828     # Variational Lower Bound on posterior, p(z|x), plus reconstruction cost.
 829     # KL and reconstruction costs are normalized only by batch size, not by
 830     # dimension, or by time steps.
 831     kl_cost_g0_b = tf.zeros_like(batch_size, dtype=tf.float32)
 832     kl_cost_co_b = tf.zeros_like(batch_size, dtype=tf.float32)
 833     self.kl_cost = tf.constant(0.0) # VAE KL cost
 834     self.recon_cost = tf.constant(0.0) # VAE reconstruction cost
 835     self.nll_bound_vae = tf.constant(0.0)
 836     self.nll_bound_iwae = tf.constant(0.0) # for eval with IWAE cost.
 837     if kind in ["train", "posterior_sample_and_average", "posterior_push_mean"]:
 838       kl_cost_g0_b = 0.0
 839       kl_cost_co_b = 0.0
 840       if ic_dim > 0:
 841         g0_priors = [self.prior_zs_g0]
 842         g0_posts = [self.posterior_zs_g0]
 843         kl_cost_g0_b = KLCost_GaussianGaussian(g0_posts, g0_priors).kl_cost_b
 844         kl_cost_g0_b = hps.kl_ic_weight * kl_cost_g0_b
 845       if co_dim > 0:
 846         kl_cost_co_b = \
 847             KLCost_GaussianGaussianProcessSampled(
 848                 posterior_zs_co, prior_zs_ar_con).kl_cost_b
 849         kl_cost_co_b = hps.kl_co_weight * kl_cost_co_b
 850 
 851       # L = -KL + log p(x|z), to maximize bound on likelihood
 852       # -L = KL - log p(x|z), to minimize bound on NLL
 853       # so 'reconstruction cost' is negative log likelihood
 854       self.recon_cost = - tf.reduce_mean(log_p_xgz_b)
 855       self.kl_cost = tf.reduce_mean(kl_cost_g0_b + kl_cost_co_b)
 856 
 857       lb_on_ll_b = log_p_xgz_b - kl_cost_g0_b - kl_cost_co_b
 858 
 859       # VAE error averages outside the log
 860       self.nll_bound_vae = -tf.reduce_mean(lb_on_ll_b)
 861 
 862       # IWAE error averages inside the log
 863       k = tf.cast(tf.shape(log_p_xgz_b)[0], tf.float32)
 864       iwae_lb_on_ll = -tf.log(k) + log_sum_exp(lb_on_ll_b)
 865       self.nll_bound_iwae = -iwae_lb_on_ll
 866 
 867     # L2 regularization on the generator, normalized by number of parameters.
 868     self.l2_cost = tf.constant(0.0)
 869     if self.hps.l2_gen_scale > 0.0 or self.hps.l2_con_scale > 0.0:
 870       l2_costs = []
 871       l2_numels = []
 872       l2_reg_var_lists = [tf.get_collection('l2_gen_reg'),
 873                           tf.get_collection('l2_con_reg')]
 874       l2_reg_scales = [self.hps.l2_gen_scale, self.hps.l2_con_scale]
 875       for l2_reg_vars, l2_scale in zip(l2_reg_var_lists, l2_reg_scales):
 876         for v in l2_reg_vars:
 877           numel = tf.reduce_prod(tf.concat(axis=0, values=tf.shape(v)))
 878           numel_f = tf.cast(numel, tf.float32)
 879           l2_numels.append(numel_f)
 880           v_l2 = tf.reduce_sum(v*v)
 881           l2_costs.append(0.5 * l2_scale * v_l2)
 882       self.l2_cost = tf.add_n(l2_costs) / tf.add_n(l2_numels)
 883 
 884     # Compute the cost for training, part of the graph regardless.
 885     # The KL cost can be problematic at the beginning of optimization,
 886     # so we allow an exponential increase in weighting the KL from 0
 887     # to 1.
 888     self.kl_decay_step = tf.maximum(self.train_step - hps.kl_start_step, 0)
 889     self.l2_decay_step = tf.maximum(self.train_step - hps.l2_start_step, 0)
 890     kl_decay_step_f = tf.cast(self.kl_decay_step, tf.float32)
 891     l2_decay_step_f = tf.cast(self.l2_decay_step, tf.float32)
 892     kl_increase_steps_f = tf.cast(hps.kl_increase_steps, tf.float32)
 893     l2_increase_steps_f = tf.cast(hps.l2_increase_steps, tf.float32)
 894     self.kl_weight = kl_weight = \
 895         tf.minimum(kl_decay_step_f / kl_increase_steps_f, 1.0)
 896     self.l2_weight = l2_weight = \
 897         tf.minimum(l2_decay_step_f / l2_increase_steps_f, 1.0)
 898 
 899     self.timed_kl_cost = kl_weight * self.kl_cost
 900     self.timed_l2_cost = l2_weight * self.l2_cost
 901     self.weight_corr_cost = hps.co_mean_corr_scale * self.corr_cost
 902     self.cost = self.recon_cost + self.timed_kl_cost + \
 903         self.timed_l2_cost + self.weight_corr_cost
 904 
 905     if kind != "train":
 906       # save every so often
 907       self.seso_saver = tf.train.Saver(tf.global_variables(),
 908                                        max_to_keep=hps.max_ckpt_to_keep)
 909       # lowest validation error
 910       self.lve_saver = tf.train.Saver(tf.global_variables(),
 911                                       max_to_keep=hps.max_ckpt_to_keep_lve)
 912 
 913       return
 914 
 915     # OPTIMIZATION
 916     # train the io matrices only
 917     if self.hps.do_train_io_only:
 918       self.train_vars = tvars = \
 919         tf.get_collection('IO_transformations',
 920                           scope=tf.get_variable_scope().name)
 921     # train the encoder only
 922     elif self.hps.do_train_encoder_only:
 923       tvars1 = \
 924         tf.get_collection(tf.GraphKeys.TRAINABLE_VARIABLES,
 925                           scope='LFADS/ic_enc_*')
 926       tvars2 = \
 927         tf.get_collection(tf.GraphKeys.TRAINABLE_VARIABLES,
 928                           scope='LFADS/z/ic_enc_*')
 929 
 930       self.train_vars = tvars = tvars1 + tvars2
 931     # train all variables
 932     else:
 933       self.train_vars = tvars = \
 934         tf.get_collection(tf.GraphKeys.TRAINABLE_VARIABLES,
 935                           scope=tf.get_variable_scope().name)
 936     print("done.")
 937     print("Model Variables (to be optimized): ")
 938     total_params = 0
 939     for i in range(len(tvars)):
 940       shape = tvars[i].get_shape().as_list()
 941       print("    ", i, tvars[i].name, shape)
 942       total_params += np.prod(shape)
 943     print("Total model parameters: ", total_params)
 944 
 945     grads = tf.gradients(self.cost, tvars)
 946     grads, grad_global_norm = tf.clip_by_global_norm(grads, hps.max_grad_norm)
 947     opt = tf.train.AdamOptimizer(self.learning_rate, beta1=0.9, beta2=0.999,
 948                                  epsilon=1e-01)
 949     self.grads = grads

1090   @staticmethod
1091   def example_idxs_mod_batch_size(nexamples, batch_size):
1092     """Given a number of examples, E, and a batch_size, B, generate indices
1093     [0, 1, 2, ... B-1;
1094     [B, B+1, ... 2*B-1;
1095     ...
1096     ]
1097     returning those indices as a 2-dim tensor shaped like E/B x B.  Note that
1098     shape is only correct if E % B == 0.  If not, then an extra row is generated
1099     so that the remainder of examples is included. The extra examples are
1100     explicitly to to the zero index (see randomize_example_idxs_mod_batch_size)
1101     for randomized behavior.
1102 
1103     Args:
1104       nexamples: The number of examples to batch up.
1105       batch_size: The size of the batch.
1106     Returns:
1107       2-dim tensor as described above.
1108     """
1109     bmrem = batch_size - (nexamples % batch_size)
1110     bmrem_examples = []
1111     if bmrem < batch_size:
1112       #bmrem_examples = np.zeros(bmrem, dtype=np.int32)
1113       ridxs = np.random.permutation(nexamples)[0:bmrem].astype(np.int32)
1114       bmrem_examples = np.sort(ridxs)
1115     example_idxs = range(nexamples) + list(bmrem_examples)
1116     example_idxs_e_x_edivb = np.reshape(example_idxs, [-1, batch_size])
1117     return example_idxs_e_x_edivb, bmrem

1119   @staticmethod
1120   def randomize_example_idxs_mod_batch_size(nexamples, batch_size):
1121     """Indices 1:nexamples, randomized, in 2D form of
1122     shape = (nexamples / batch_size) x batch_size.  The remainder
1123     is managed by drawing randomly from 1:nexamples.
1124 
1125     Args:
1126       nexamples: number of examples to randomize
1127       batch_size: number of elements in batch
1128 
1129     Returns:
1130       The randomized, properly shaped indicies.
1131     """
1132     assert nexamples > batch_size, "Problems"
1133     bmrem = batch_size - nexamples % batch_size
1134     bmrem_examples = []
1135     if bmrem < batch_size:
1136       bmrem_examples = np.random.choice(range(nexamples),
1137                                         size=bmrem, replace=False)
1138     example_idxs = range(nexamples) + list(bmrem_examples)
1139     mixed_example_idxs = np.random.permutation(example_idxs)
1140     example_idxs_e_x_edivb = np.reshape(mixed_example_idxs, [-1, batch_size])
1141     return example_idxs_e_x_edivb, bmrem

1143   def shuffle_spikes_in_time(self, data_bxtxd):
1144     """Shuffle the spikes in the temporal dimension.  This is useful to
1145     help the LFADS system avoid overfitting to individual spikes or fast
1146     oscillations found in the data that are irrelevant to behavior. A
1147     pure 'tabula rasa' approach would avoid this, but LFADS is sensitive
1148     enough to pick up dynamics that you may not want.
1149 
1150     Args:
1151       data_bxtxd: numpy array of spike count data to be shuffled.
1152     Returns:
1153     S_bxtxd, a numpy array with the same dimensions and contents as
1154       data_bxtxd, but shuffled appropriately.
1155 
1156     """
1157 
1158     B, T, N = data_bxtxd.shape
1159     w = self.hps.temporal_spike_jitter_width
1160 
1161     if w == 0:
1162       return data_bxtxd
1163 
1164     max_counts = np.max(data_bxtxd)
1165     S_bxtxd = np.zeros([B,T,N])
1166 
1167     # Intuitively, shuffle spike occurances, 0 or 1, but since we have counts,
1168     # Do it over and over again up to the max count.
1169     for mc in range(1,max_counts+1):
1170       idxs = np.nonzero(data_bxtxd >= mc)
1171 
1172       data_ones = np.zeros_like(data_bxtxd)
1173       data_ones[data_bxtxd >= mc] = 1
1174 
1175       nfound = len(idxs[0])
1176       shuffles_incrs_in_time = np.random.randint(-w, w, size=nfound)
1177 
1178       shuffle_tidxs = idxs[1].copy()
1179       shuffle_tidxs += shuffles_incrs_in_time
1180 
1181       # Reflect on the boundaries to not lose mass.
1182       shuffle_tidxs[shuffle_tidxs < 0] = -shuffle_tidxs[shuffle_tidxs < 0]
1183       shuffle_tidxs[shuffle_tidxs > T-1] = \
1184           (T-1)-(shuffle_tidxs[shuffle_tidxs > T-1] -(T-1))
1185 
1186       for iii in zip(idxs[0], shuffle_tidxs, idxs[2]):
1187         S_bxtxd[iii] += 1
1188 
1189     return S_bxtxd

1191   def shuffle_and_flatten_datasets(self, datasets, kind='train'):
1192     """Since LFADS supports multiple datasets in the same dynamical model,
1193     we have to be careful to use all the data in a single training epoch.  But
1194     since the datasets my have different data dimensionality, we cannot batch
1195     examples from data dictionaries together.  Instead, we generate random
1196     batches within each data dictionary, and then randomize these batches
1197     while holding onto the dataname, so that when it's time to feed
1198     the graph, the correct in/out matrices can be selected, per batch.
1199 
1200     Args:
1201       datasets: A dict of data dicts.  The dataset dict is simply a
1202         name(string)-> data dictionary mapping (See top of lfads.py).
1203       kind: 'train' or 'valid'
1204 
1205     Returns:
1206       A flat list, in which each element is a pair ('name', indices).
1207     """
1208     batch_size = self.hps.batch_size
1209     ndatasets = len(datasets)
1210     random_example_idxs = {}
1211     epoch_idxs = {}
1212     all_name_example_idx_pairs = []
1213     kind_data = kind + '_data'
1214     for name, data_dict in datasets.items():
1215       nexamples, ntime, data_dim = data_dict[kind_data].shape
1216       epoch_idxs[name] = 0
1217       random_example_idxs, _ = \
1218         self.randomize_example_idxs_mod_batch_size(nexamples, batch_size)
1219 
1220       epoch_size = random_example_idxs.shape[0]
1221       names = [name] * epoch_size
1222       all_name_example_idx_pairs += zip(names, random_example_idxs)
1223 
1224     np.random.shuffle(all_name_example_idx_pairs) # shuffle in place
1225 
1226     return all_name_example_idx_pairs

1472   def train_model(self, datasets):
1473     """Train the model, print per-epoch information, and save checkpoints.
1474 
1475     Loop over training epochs. The function that actually does the
1476     training is train_epoch.  This function iterates over the training
1477     data, one epoch at a time.  The learning rate schedule is such
1478     that it will stay the same until the cost goes up in comparison to
1479     the last few values, then it will drop.
1480 
1481     Args:
1482       datasets: A dict of data dicts.  The dataset dict is simply a
1483         name(string)-> data dictionary mapping (See top of lfads.py).
1484     """
1485     hps = self.hps
1486     has_any_valid_set = False
1487     for data_dict in datasets.values():
1488       if data_dict['valid_data'] is not None:
1489         has_any_valid_set = True
1490         break
1491 
1492     session = tf.get_default_session()
1493     lr = session.run(self.learning_rate)
1494     lr_stop = hps.learning_rate_stop
1495     i = -1
1496     train_costs = []
1497     valid_costs = []
1498     ev_total_cost = ev_recon_cost = ev_kl_cost = 0.0
1499     lowest_ev_cost = np.Inf
1500     while True:
1501       i += 1
1502       do_save_ckpt = True if i % 10 ==0 else False
1503       tr_total_cost, tr_recon_cost, tr_kl_cost, kl_weight, l2_cost, l2_weight = \
1504                 self.train_epoch(datasets, do_save_ckpt=do_save_ckpt)
1505 
1506       # Evaluate the validation cost, and potentially save.  Note that this
1507       # routine will not save a validation checkpoint until the kl weight and
1508       # l2 weights are equal to 1.0.
1509       if has_any_valid_set:
1510         ev_total_cost, ev_recon_cost, ev_kl_cost = \
1511             self.eval_cost_epoch(datasets, kind='valid')
1512         valid_costs.append(ev_total_cost)
1513 
1514         # > 1 may give more consistent results, but not the actual lowest vae.
1515         # == 1 gives the lowest vae seen so far.
1516         n_lve = 1
1517         run_avg_lve = np.mean(valid_costs[-n_lve:])
1518 
1519         # conditions for saving checkpoints:
1520         #   KL weight must have finished stepping (>=1.0), AND
1521         #   L2 weight must have finished stepping OR L2 is not being used, AND
1522         #   the current run has a lower LVE than previous runs AND
1523         #     len(valid_costs > n_lve) (not sure what that does)
1524         if kl_weight >= 1.0 and \
1525           (l2_weight >= 1.0 or \
1526            (self.hps.l2_gen_scale == 0.0 and self.hps.l2_con_scale == 0.0)) \
1527            and (len(valid_costs) > n_lve and run_avg_lve < lowest_ev_cost):
1528 
1529           lowest_ev_cost = run_avg_lve
1530           checkpoint_path = os.path.join(self.hps.lfads_save_dir,
1531                                          self.hps.checkpoint_name + '_lve.ckpt')
1532           self.lve_saver.save(session, checkpoint_path,
1533                               global_step=self.train_step,
1534                               latest_filename='checkpoint_lve')
1535 
1536       # Plot and summarize.
1537       values = {'nepochs':i, 'has_any_valid_set': has_any_valid_set,
1538                 'tr_total_cost':tr_total_cost, 'ev_total_cost':ev_total_cost,
1539                 'tr_recon_cost':tr_recon_cost, 'ev_recon_cost':ev_recon_cost,
1540                 'tr_kl_cost':tr_kl_cost, 'ev_kl_cost':ev_kl_cost,
1541                 'l2_weight':l2_weight, 'kl_weight':kl_weight,
1542                 'l2_cost':l2_cost}
1543       self.summarize_all(datasets, values)
1544       self.plot_single_example(datasets)
1545 
1546       # Manage learning rate.
1547       train_res = tr_total_cost
1548       n_lr = hps.learning_rate_n_to_compare
1549       if len(train_costs) > n_lr and train_res > np.max(train_costs[-n_lr:]):
1550         _ = session.run(self.learning_rate_decay_op)
1551         lr = session.run(self.learning_rate)
1552         print("     Decreasing learning rate to %f." % lr)
1553         # Force the system to run n_lr times while at this lr.
1554         train_costs.append(np.inf)
1555       else:
1556         train_costs.append(train_res)
1557 
1558       if lr < lr_stop:
1559         print("Stopping optimization based on learning rate criteria.")
1560         break

1562   def eval_cost_epoch(self, datasets, kind='train', ext_input_extxi=None,
1563                       batch_size=None):
1564     """Evaluate the cost of the epoch.
1565 
1566     Args:
1567       data_dict: The dictionary of data (training and validation) used for
1568         training and evaluation of the model, respectively.
1569 
1570     Returns:
1571       a 3 tuple of costs:
1572         (epoch total cost, epoch reconstruction cost, epoch KL cost)
1573     """
1574     ops_to_eval = [self.cost, self.recon_cost, self.kl_cost]
1575     collected_op_values = self.run_epoch(datasets, ops_to_eval, kind=kind,
1576                                          keep_prob=1.0)
1577 
1578     total_cost = total_recon_cost = total_kl_cost = 0.0
1579     # normalizing by batch done in distributions.py
1580     epoch_size = len(collected_op_values)
1581     for op_values in collected_op_values:
1582       total_cost += op_values[0]
1583       total_recon_cost += op_values[1]
1584       total_kl_cost += op_values[2]
1585 
1586     epoch_total_cost = total_cost / epoch_size
1587     epoch_recon_cost = total_recon_cost / epoch_size
1588     epoch_kl_cost = total_kl_cost / epoch_size
1589 
1590     return epoch_total_cost, epoch_recon_cost, epoch_kl_cost
