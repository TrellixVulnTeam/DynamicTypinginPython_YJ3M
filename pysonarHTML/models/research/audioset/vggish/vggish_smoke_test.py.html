  16 """A smoke test for VGGish.
  17 
  18 This is a simple smoke test of a local install of VGGish and its associated
  19 downloaded files. We create a synthetic sound, extract log mel spectrogram
  20 features, run them through VGGish, post-process the embedding ouputs, and
  21 check some simple statistics of the results, allowing for variations that
  22 might occur due to platform/version differences in the libraries we use.
  23 
  24 Usage:
  25 - Download the VGGish checkpoint and PCA parameters into the same directory as
  26   the VGGish source code. If you keep them elsewhere, update the checkpoint_path
  27   and pca_params_path variables below.
  28 - Run:
  29   $ python vggish_smoke_test.py
  30 """
  31 
  32 from __future__ import print_function
  33 
  34 import numpy as np
  35 import tensorflow.compat.v1 as tf
  36 tf.disable_v2_behavior()
  37 
  38 import vggish_input
  39 import vggish_params
  40 import vggish_postprocess
  41 import vggish_slim
  42 
  43 print('\nTesting your install of VGGish\n')
  44 
  45 # Paths to downloaded VGGish files.
  46 checkpoint_path = 'vggish_model.ckpt'
  47 pca_params_path = 'vggish_pca_params.npz'
  48 
  49 # Relative tolerance of errors in mean and standard deviation of embeddings.
  50 rel_error = 0.1  # Up to 10%
  51 
  52 # Generate a 1 kHz sine wave at 44.1 kHz (we use a high sampling rate
  53 # to test resampling to 16 kHz during feature extraction).
  54 num_secs = 3
  55 freq = 1000
  56 sr = 44100
  57 t = np.linspace(0, num_secs, int(num_secs * sr))
  58 x = np.sin(2 * np.pi * freq * t)
  59 
  60 # Produce a batch of log mel spectrogram examples.
  61 input_batch = vggish_input.waveform_to_examples(x, sr)
  62 print('Log Mel Spectrogram example: ', input_batch[0])
  63 np.testing.assert_equal(
  64     input_batch.shape,
  65     [num_secs, vggish_params.NUM_FRAMES, vggish_params.NUM_BANDS])
  66 
  67 # Define VGGish, load the checkpoint, and run the batch through the model to
  68 # produce embeddings.
  69 with tf.Graph().as_default(), tf.Session() as sess:
  70   vggish_slim.define_vggish_slim()
  71   vggish_slim.load_vggish_slim_checkpoint(sess, checkpoint_path)
  72 
  73   features_tensor = sess.graph.get_tensor_by_name(
  74       vggish_params.INPUT_TENSOR_NAME)
  75   embedding_tensor = sess.graph.get_tensor_by_name(
  76       vggish_params.OUTPUT_TENSOR_NAME)
  77   [embedding_batch] = sess.run([embedding_tensor],
  78                                feed_dict={features_tensor: input_batch})
  79   print('VGGish embedding: ', embedding_batch[0])
  80   expected_embedding_mean = 0.131
  81   expected_embedding_std = 0.238
  82   np.testing.assert_allclose(
  83       [np.mean(embedding_batch), np.std(embedding_batch)],
  84       [expected_embedding_mean, expected_embedding_std],
  85       rtol=rel_error)
  86 
  87 # Postprocess the results to produce whitened quantized embeddings.
  88 pproc = vggish_postprocess.Postprocessor(pca_params_path)
  89 postprocessed_batch = pproc.postprocess(embedding_batch)
  90 print('Postprocessed VGGish embedding: ', postprocessed_batch[0])
  91 expected_postprocessed_mean = 123.0
  92 expected_postprocessed_std = 75.0
  93 np.testing.assert_allclose(
  94     [np.mean(postprocessed_batch), np.std(postprocessed_batch)],
  95     [expected_postprocessed_mean, expected_postprocessed_std],
  96     rtol=rel_error)
  97 
  98 print('\nLooks Good To Me!\n')
