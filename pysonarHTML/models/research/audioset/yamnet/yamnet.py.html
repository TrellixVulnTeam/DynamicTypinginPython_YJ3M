  28 def _batch_norm(name):
  29   def _bn_layer(layer_input):
  30     return layers.BatchNormalization(
  31       name=name,
  32       center=params.BATCHNORM_CENTER,
  33       scale=params.BATCHNORM_SCALE,
  34       epsilon=params.BATCHNORM_EPSILON)(layer_input)
  35   return _bn_layer

  38 def _conv(name, kernel, stride, filters):
  39   def _conv_layer(layer_input):
  40     output = layers.Conv2D(name='{}/conv'.format(name),
  41                            filters=filters,
  42                            kernel_size=kernel,
  43                            strides=stride,
  44                            padding=params.CONV_PADDING,
  45                            use_bias=False,
  46                            activation=None)(layer_input)
  47     output = _batch_norm(name='{}/conv/bn'.format(name))(output)
  48     output = layers.ReLU(name='{}/relu'.format(name))(output)
  49     return output
  50   return _conv_layer
  51 
  52 
  53 def _separable_conv(name, kernel, stride, filters):
  54   def _separable_conv_layer(layer_input):
  55     output = layers.DepthwiseConv2D(name='{}/depthwise_conv'.format(name),
  56                                     kernel_size=kernel,
  57                                     strides=stride,
  58                                     depth_multiplier=1,
  59                                     padding=params.CONV_PADDING,
  60                                     use_bias=False,
  61                                     activation=None)(layer_input)
  62     output = _batch_norm(name='{}/depthwise_conv/bn'.format(name))(output)
  63     output = layers.ReLU(name='{}/depthwise_conv/relu'.format(name))(output)
  64     output = layers.Conv2D(name='{}/pointwise_conv'.format(name),
  65                            filters=filters,
  66                            kernel_size=(1, 1),
  67                            strides=1,
  68                            padding=params.CONV_PADDING,
  69                            use_bias=False,
  70                            activation=None)(output)
  71     output = _batch_norm(name='{}/pointwise_conv/bn'.format(name))(output)
  72     output = layers.ReLU(name='{}/pointwise_conv/relu'.format(name))(output)
  73     return output
  74   return _separable_conv_layer

  96 def yamnet(features):
  97   """Define the core YAMNet mode in Keras."""
  98   net = layers.Reshape(
  99     (params.PATCH_FRAMES, params.PATCH_BANDS, 1),
 100     input_shape=(params.PATCH_FRAMES, params.PATCH_BANDS))(features)
 101   for (i, (layer_fun, kernel, stride, filters)) in enumerate(_YAMNET_LAYER_DEFS):
 102     net = layer_fun('layer{}'.format(i + 1), kernel, stride, filters)(net)
 103   net = layers.GlobalAveragePooling2D()(net)
 104   logits = layers.Dense(units=params.NUM_CLASSES, use_bias=True)(net)
 105   predictions = layers.Activation(
 106     name=params.EXAMPLE_PREDICTIONS_LAYER_NAME,
 107     activation=params.CLASSIFIER_ACTIVATION)(logits)
 108   return predictions

 111 def yamnet_frames_model(feature_params):
 112   """Defines the YAMNet waveform-to-class-scores model.
 113 
 114   Args:
 115     feature_params: An object with parameter fields to control the feature
 116     calculation.
 117 
 118   Returns:
 119     A model accepting (1, num_samples) waveform input and emitting a
 120     (num_patches, num_classes) matrix of class scores per time frame as
 121     well as a (num_spectrogram_frames, num_mel_bins) spectrogram feature
 122     matrix.
 123   """
 124   waveform = layers.Input(batch_shape=(1, None))
 125   # Store the intermediate spectrogram features to use in visualization.
 126   spectrogram = features_lib.waveform_to_log_mel_spectrogram(
 127     tf.squeeze(waveform, axis=0), feature_params)
 128   patches = features_lib.spectrogram_to_patches(spectrogram, feature_params)
 129   predictions = yamnet(patches)
 130   frames_model = Model(name='yamnet_frames', 
 131                        inputs=waveform, outputs=[predictions, spectrogram])
 132   return frames_model

 135 def class_names(class_map_csv):
 136   """Read the class name definition file and return a list of strings."""
 137   with open(class_map_csv) as csv_file:
 138     reader = csv.reader(csv_file)
 139     next(reader)   # Skip header
 140     return np.array([display_name for (_, _, display_name) in reader])
