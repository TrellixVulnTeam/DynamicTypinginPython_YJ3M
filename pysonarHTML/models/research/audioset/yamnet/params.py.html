  16 """Hyperparameters for YAMNet."""
  17 
  18 # The following hyperparameters (except PATCH_HOP_SECONDS) were used to train YAMNet,
  19 # so expect some variability in performance if you change these. The patch hop can
  20 # be changed arbitrarily: a smaller hop should give you more patches from the same
  21 # clip and possibly better performance at a larger computational cost.
  22 SAMPLE_RATE = 16000
  23 STFT_WINDOW_SECONDS = 0.025
  24 STFT_HOP_SECONDS = 0.010
  25 MEL_BANDS = 64
  26 MEL_MIN_HZ = 125
  27 MEL_MAX_HZ = 7500
  28 LOG_OFFSET = 0.001
  29 PATCH_WINDOW_SECONDS = 0.96
  30 PATCH_HOP_SECONDS = 0.48
  31 
  32 PATCH_FRAMES = int(round(PATCH_WINDOW_SECONDS / STFT_HOP_SECONDS))
  33 PATCH_BANDS = MEL_BANDS
  34 NUM_CLASSES = 521
  35 CONV_PADDING = 'same'
  36 BATCHNORM_CENTER = True
  37 BATCHNORM_SCALE = False
  38 BATCHNORM_EPSILON = 1e-4
  39 CLASSIFIER_ACTIVATION = 'sigmoid'
  40 
  41 FEATURES_LAYER_NAME = 'features'
  42 EXAMPLE_PREDICTIONS_LAYER_NAME = 'predictions'
