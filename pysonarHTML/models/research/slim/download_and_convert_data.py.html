  15 r"""Downloads and converts a particular dataset.
  16 
  17 Usage:
  18 ```shell
  19 
  20 $ python download_and_convert_data.py \
  21     --dataset_name=flowers \
  22     --dataset_dir=/tmp/flowers
  23 
  24 $ python download_and_convert_data.py \
  25     --dataset_name=cifar10 \
  26     --dataset_dir=/tmp/cifar10
  27 
  28 $ python download_and_convert_data.py \
  29     --dataset_name=mnist \
  30     --dataset_dir=/tmp/mnist
  31 
  32 $ python download_and_convert_data.py \
  33     --dataset_name=visualwakewords \
  34     --dataset_dir=/tmp/visualwakewords
  35 
  36 ```
  37 """
  38 from __future__ import absolute_import
  39 from __future__ import division
  40 from __future__ import print_function
  41 
  42 import tensorflow.compat.v1 as tf
  43 
  44 from datasets import download_and_convert_cifar10
  45 from datasets import download_and_convert_flowers
  46 from datasets import download_and_convert_mnist
  47 from datasets import download_and_convert_visualwakewords
  48 
  49 FLAGS = tf.app.flags.FLAGS
  50 
  51 tf.app.flags.DEFINE_string(
  52     'dataset_name',
  53     None,
  54     'The name of the dataset to convert, one of "flowers", "cifar10", "mnist", "visualwakewords"'
  55     )
  56 
  57 tf.app.flags.DEFINE_string(
  58     'dataset_dir',
  59     None,
  60     'The directory where the output TFRecords and temporary files are saved.')
  61 
  62 tf.flags.DEFINE_float(
  63     'small_object_area_threshold', 0.005,
  64     'For --dataset_name=visualwakewords only. Threshold of fraction of image '
  65     'area below which small objects are filtered')
  66 
  67 tf.flags.DEFINE_string(
  68     'foreground_class_of_interest', 'person',
  69     'For --dataset_name=visualwakewords only. Build a binary classifier based '
  70     'on the presence or absence of this object in the image.')
  71 
  72 
  73 def main(_):
  74   if not FLAGS.dataset_name:
  75     raise ValueError('You must supply the dataset name with --dataset_name')
  76   if not FLAGS.dataset_dir:
  77     raise ValueError('You must supply the dataset directory with --dataset_dir')
  78 
  79   if FLAGS.dataset_name == 'flowers':
  80     download_and_convert_flowers.run(FLAGS.dataset_dir)
  81   elif FLAGS.dataset_name == 'cifar10':
  82     download_and_convert_cifar10.run(FLAGS.dataset_dir)
  83   elif FLAGS.dataset_name == 'mnist':
  84     download_and_convert_mnist.run(FLAGS.dataset_dir)
  85   elif FLAGS.dataset_name == 'visualwakewords':
  86     download_and_convert_visualwakewords.run(
  87         FLAGS.dataset_dir, FLAGS.small_object_area_threshold,
  88         FLAGS.foreground_class_of_interest)
  89   else:
  90     raise ValueError(
  91         'dataset_name [%s] was not recognized.' % FLAGS.dataset_name)
  92 
  93 if __name__ == '__main__':
  94   tf.app.run()
