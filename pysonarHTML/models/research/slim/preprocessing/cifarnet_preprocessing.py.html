  29 def preprocess_for_train(image,
  30                          output_height,
  31                          output_width,
  32                          padding=_PADDING,
  33                          add_image_summaries=True,
  34                          use_grayscale=False):
  35   """Preprocesses the given image for training.
  36 
  37   Note that the actual resizing scale is sampled from
  38     [`resize_size_min`, `resize_size_max`].
  39 
  40   Args:
  41     image: A `Tensor` representing an image of arbitrary size.
  42     output_height: The height of the image after preprocessing.
  43     output_width: The width of the image after preprocessing.
  44     padding: The amound of padding before and after each dimension of the image.
  45     add_image_summaries: Enable image summaries.
  46     use_grayscale: Whether to convert the image from RGB to grayscale.
  47 
  48   Returns:
  49     A preprocessed image.
  50   """
  51   if add_image_summaries:
  52     tf.summary.image('image', tf.expand_dims(image, 0))
  53 
  54   # Transform the image to floats.
  55   image = tf.to_float(image)
  56   if use_grayscale:
  57     image = tf.image.rgb_to_grayscale(image)
  58   if padding > 0:
  59     image = tf.pad(image, [[padding, padding], [padding, padding], [0, 0]])
  60   # Randomly crop a [height, width] section of the image.
  61   distorted_image = tf.random_crop(image,
  62                                    [output_height, output_width, 3])
  63 
  64   # Randomly flip the image horizontally.
  65   distorted_image = tf.image.random_flip_left_right(distorted_image)
  66 
  67   if add_image_summaries:
  68     tf.summary.image('distorted_image', tf.expand_dims(distorted_image, 0))
  69 
  70   # Because these operations are not commutative, consider randomizing
  71   # the order their operation.
  72   distorted_image = tf.image.random_brightness(distorted_image,
  73                                                max_delta=63)
  74   distorted_image = tf.image.random_contrast(distorted_image,
  75                                              lower=0.2, upper=1.8)
  76   # Subtract off the mean and divide by the variance of the pixels.
  77   return tf.image.per_image_standardization(distorted_image)

  80 def preprocess_for_eval(image,
  81                         output_height,
  82                         output_width,
  83                         add_image_summaries=True,
  84                         use_grayscale=False):
  85   """Preprocesses the given image for evaluation.
  86 
  87   Args:
  88     image: A `Tensor` representing an image of arbitrary size.
  89     output_height: The height of the image after preprocessing.
  90     output_width: The width of the image after preprocessing.
  91     add_image_summaries: Enable image summaries.
  92     use_grayscale: Whether to convert the image from RGB to grayscale.
  93 
  94   Returns:
  95     A preprocessed image.
  96   """
  97   if add_image_summaries:
  98     tf.summary.image('image', tf.expand_dims(image, 0))
  99   # Transform the image to floats.
 100   image = tf.to_float(image)
 101   if use_grayscale:
 102     image = tf.image.rgb_to_grayscale(image)
 103 
 104   # Resize and crop if needed.
 105   resized_image = tf.image.resize_image_with_crop_or_pad(image,
 106                                                          output_width,
 107                                                          output_height)
 108   if add_image_summaries:
 109     tf.summary.image('resized_image', tf.expand_dims(resized_image, 0))
 110 
 111   # Subtract off the mean and divide by the variance of the pixels.
 112   return tf.image.per_image_standardization(resized_image)

 115 def preprocess_image(image,
 116                      output_height,
 117                      output_width,
 118                      is_training=False,
 119                      add_image_summaries=True,
 120                      use_grayscale=False):
 121   """Preprocesses the given image.
 122 
 123   Args:
 124     image: A `Tensor` representing an image of arbitrary size.
 125     output_height: The height of the image after preprocessing.
 126     output_width: The width of the image after preprocessing.
 127     is_training: `True` if we're preprocessing the image for training and
 128       `False` otherwise.
 129     add_image_summaries: Enable image summaries.
 130     use_grayscale: Whether to convert the image from RGB to grayscale.
 131 
 132   Returns:
 133     A preprocessed image.
 134   """
 135   if is_training:
 136     return preprocess_for_train(
 137         image,
 138         output_height,
 139         output_width,
 140         add_image_summaries=add_image_summaries,
 141         use_grayscale=use_grayscale)
 142   else:
 143     return preprocess_for_eval(
 144         image,
 145         output_height,
 146         output_width,
 147         add_image_summaries=add_image_summaries,
 148         use_grayscale=use_grayscale)
