  87                          'Whether to convert input images to grayscale.')

  89 FLAGS = tf.app.flags.FLAGS

  92 def main(_):
  93   if not FLAGS.dataset_dir:
  94     raise ValueError('You must supply the dataset directory with --dataset_dir')
  95 
  96   tf.logging.set_verbosity(tf.logging.INFO)
  97   with tf.Graph().as_default():
  98     tf_global_step = slim.get_or_create_global_step()
  99 
 100     ######################
 101     # Select the dataset #
 102     ######################
 103     dataset = dataset_factory.get_dataset(
 104         FLAGS.dataset_name, FLAGS.dataset_split_name, FLAGS.dataset_dir)
 105 
 106     ####################
 107     # Select the model #
 108     ####################
 109     network_fn = nets_factory.get_network_fn(
 110         FLAGS.model_name,
 111         num_classes=(dataset.num_classes - FLAGS.labels_offset),
 112         is_training=False)
 113 
 114     ##############################################################
 115     # Create a dataset provider that loads data from the dataset #
 116     ##############################################################
 117     provider = slim.dataset_data_provider.DatasetDataProvider(
 118         dataset,
 119         shuffle=False,
 120         common_queue_capacity=2 * FLAGS.batch_size,
 121         common_queue_min=FLAGS.batch_size)
 122     [image, label] = provider.get(['image', 'label'])
 123     label -= FLAGS.labels_offset
 124 
 125     #####################################
 126     # Select the preprocessing function #
 127     #####################################
 128     preprocessing_name = FLAGS.preprocessing_name or FLAGS.model_name
 129     image_preprocessing_fn = preprocessing_factory.get_preprocessing(
 130         preprocessing_name,
 131         is_training=False,
 132         use_grayscale=FLAGS.use_grayscale)
 133 
 134     eval_image_size = FLAGS.eval_image_size or network_fn.default_image_size
 135 
 136     image = image_preprocessing_fn(image, eval_image_size, eval_image_size)
 137 
 138     images, labels = tf.train.batch(
 139         [image, label],
 140         batch_size=FLAGS.batch_size,
 141         num_threads=FLAGS.num_preprocessing_threads,
 142         capacity=5 * FLAGS.batch_size)
 143 
 144     ####################
 145     # Define the model #
 146     ####################
 147     logits, _ = network_fn(images)
 148 
 149     if FLAGS.quantize:
 150       contrib_quantize.create_eval_graph()
 151 
 152     if FLAGS.moving_average_decay:
 153       variable_averages = tf.train.ExponentialMovingAverage(
 154           FLAGS.moving_average_decay, tf_global_step)
 155       variables_to_restore = variable_averages.variables_to_restore(
 156           slim.get_model_variables())
 157       variables_to_restore[tf_global_step.op.name] = tf_global_step
 158     else:
 159       variables_to_restore = slim.get_variables_to_restore()
 160 
 161     predictions = tf.argmax(logits, 1)
 162     labels = tf.squeeze(labels)
 163 
 164     # Define the metrics:
 165     names_to_values, names_to_updates = slim.metrics.aggregate_metric_map({
 166         'Accuracy': slim.metrics.streaming_accuracy(predictions, labels),
 167         'Recall_5': slim.metrics.streaming_recall_at_k(
 168             logits, labels, 5),
 169     })
 170 
 171     # Print the summaries to screen.
 172     for name, value in names_to_values.items():
 173       summary_name = 'eval/%s' % name
 174       op = tf.summary.scalar(summary_name, value, collections=[])
 175       op = tf.Print(op, [value], summary_name)
 176       tf.add_to_collection(tf.GraphKeys.SUMMARIES, op)
 177 
 178     # TODO(sguada) use num_epochs=1
 179     if FLAGS.max_num_batches:
 180       num_batches = FLAGS.max_num_batches
 181     else:
 182       # This ensures that we make a single pass over all of the data.
 183       num_batches = math.ceil(dataset.num_samples / float(FLAGS.batch_size))
 184 
 185     if tf.gfile.IsDirectory(FLAGS.checkpoint_path):
 186       checkpoint_path = tf.train.latest_checkpoint(FLAGS.checkpoint_path)
 187     else:
 188       checkpoint_path = FLAGS.checkpoint_path
 189 
 190     tf.logging.info('Evaluating %s' % checkpoint_path)
 191 
 192     slim.evaluation.evaluate_once(
 193         master=FLAGS.master,
 194         checkpoint_path=checkpoint_path,
 195         logdir=FLAGS.eval_dir,
 196         num_evals=num_batches,
 197         eval_op=list(names_to_updates.values()),
 198         variables_to_restore=variables_to_restore)
 199 
 200 
 201 if __name__ == '__main__':
 202   tf.app.run()
