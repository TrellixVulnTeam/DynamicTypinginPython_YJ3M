  31 def create_test_input(batch_size, height, width, channels):
  32   """Create test input tensor.
  33 
  34   Args:
  35     batch_size: The number of images per batch or `None` if unknown.
  36     height: The height of each image or `None` if unknown.
  37     width: The width of each image or `None` if unknown.
  38     channels: The number of channels per image or `None` if unknown.
  39 
  40   Returns:
  41     Either a placeholder `Tensor` of dimension
  42       [batch_size, height, width, channels] if any of the inputs are `None` or a
  43     constant `Tensor` with the mesh grid values along the spatial dimensions.
  44   """
  45   if None in [batch_size, height, width, channels]:
  46     return tf.placeholder(tf.float32, (batch_size, height, width, channels))
  47   else:
  48     return tf.cast(
  49         np.tile(
  50             np.reshape(
  51                 np.reshape(np.arange(height), [height, 1]) +
  52                 np.reshape(np.arange(width), [1, width]),
  53                 [1, height, width, 1]), [batch_size, 1, 1, channels]),
  54         dtype=tf.float32)

  74     n, n2 = 4, 2
  75 
  76     # Input image.
  77     x = create_test_input(1, n, n, 1)
  78 
  79     # Convolution kernel.
  80     w = create_test_input(1, 3, 3, 1)
  81     w = tf.reshape(w, [3, 3, 1, 1])
  82 
  83     tf.get_variable('Conv/weights', initializer=w)
  84     tf.get_variable('Conv/biases', initializer=tf.zeros([1]))
  85     tf.get_variable_scope().reuse_variables()
  86 
  87     y1 = slim.conv2d(x, 1, [3, 3], stride=1, scope='Conv')
  88     y1_expected = tf.cast([[14, 28, 43, 26], [28, 48, 66, 37], [43, 66, 84, 46],

  90                           dtype=tf.float32)

  92 
  93     y2 = resnet_utils.subsample(y1, 2)
  94     y2_expected = tf.cast([[14, 43], [43, 84]], dtype=tf.float32)
  95     y2_expected = tf.reshape(y2_expected, [1, n2, n2, 1])
  96 
  97     y3 = resnet_utils.conv2d_same(x, 1, 3, stride=2, scope='Conv')
  98     y3_expected = y2_expected
  99 
 100     y4 = slim.conv2d(x, 1, [3, 3], stride=2, scope='Conv')
 101     y4_expected = tf.cast([[48, 37], [37, 22]], dtype=tf.float32)
 102     y4_expected = tf.reshape(y4_expected, [1, n2, n2, 1])

 150   def _resnet_plain(self, inputs, blocks, output_stride=None, scope=None):
 151     """A plain ResNet without extra layers before or after the ResNet blocks."""
 152     with tf.variable_scope(scope, values=[inputs]):
 153       with slim.arg_scope([slim.conv2d], outputs_collections='end_points'):
 154         net = resnet_utils.stack_blocks_dense(inputs, blocks, output_stride)
 155         end_points = slim.utils.convert_collection_to_dict('end_points')
 156         return net, end_points

 186   def _stack_blocks_nondense(self, net, blocks):
 187     """A simplified ResNet Block stacker without output stride control."""
 188     for block in blocks:
 189       with tf.variable_scope(block.scope, 'block', [net]):
 190         for i, unit in enumerate(block.args):
 191           with tf.variable_scope('unit_%d' % (i + 1), values=[net]):
 192             net = block.unit_fn(net, rate=1, **unit)
 193     return net

 195   def testAtrousValuesBottleneck(self):
 196     """Verify the values of dense feature extraction by atrous convolution.
 197 
 198     Make sure that dense feature extraction by stack_blocks_dense() followed by
 199     subsampling gives identical results to feature extraction at the nominal
 200     network output stride using the simple self._stack_blocks_nondense() above.
 201     """
 202     block = resnet_v1.resnet_v1_block
 203     blocks = [
 204         block('block1', base_depth=1, num_units=2, stride=2),
 205         block('block2', base_depth=2, num_units=2, stride=2),
 206         block('block3', base_depth=4, num_units=2, stride=2),
 207         block('block4', base_depth=8, num_units=2, stride=1),
 208     ]
 209     nominal_stride = 8
 210 
 211     # Test both odd and even input dimensions.
 212     height = 30
 213     width = 31
 214     with slim.arg_scope(resnet_utils.resnet_arg_scope()):
 215       with slim.arg_scope([slim.batch_norm], is_training=False):
 216         for output_stride in [1, 2, 4, 8, None]:
 217           with tf.Graph().as_default():
 218             with self.test_session() as sess:
 219               tf.set_random_seed(0)
 220               inputs = create_test_input(1, height, width, 3)
 221               # Dense feature extraction followed by subsampling.
 222               output = resnet_utils.stack_blocks_dense(inputs,
 223                                                        blocks,
 224                                                        output_stride)
 225               if output_stride is None:
 226                 factor = 1
 227               else:
 228                 factor = nominal_stride // output_stride
 229 
 230               output = resnet_utils.subsample(output, factor)
 231               # Make the two networks use the same weights.
 232               tf.get_variable_scope().reuse_variables()
 233               # Feature extraction at the nominal network rate.
 234               expected = self._stack_blocks_nondense(inputs, blocks)
 235               sess.run(tf.global_variables_initializer())
 236               output, expected = sess.run([output, expected])
 237               self.assertAllClose(output, expected, atol=1e-4, rtol=1e-4)

 239   def testStridingLastUnitVsSubsampleBlockEnd(self):
 240     """Compares subsampling at the block's last unit or block's end.
 241 
 242     Makes sure that the final output is the same when we use a stride at the
 243     last unit of a block vs. we subsample activations at the end of a block.
 244     """
 245     block = resnet_v1.resnet_v1_block
 246 
 247     blocks = [
 248         block('block1', base_depth=1, num_units=2, stride=2),
 249         block('block2', base_depth=2, num_units=2, stride=2),
 250         block('block3', base_depth=4, num_units=2, stride=2),
 251         block('block4', base_depth=8, num_units=2, stride=1),
 252     ]
 253 
 254     # Test both odd and even input dimensions.
 255     height = 30
 256     width = 31
 257     with slim.arg_scope(resnet_utils.resnet_arg_scope()):
 258       with slim.arg_scope([slim.batch_norm], is_training=False):
 259         for output_stride in [1, 2, 4, 8, None]:
 260           with tf.Graph().as_default():
 261             with self.test_session() as sess:
 262               tf.set_random_seed(0)
 263               inputs = create_test_input(1, height, width, 3)
 264 
 265               # Subsampling at the last unit of the block.
 266               output = resnet_utils.stack_blocks_dense(
 267                   inputs, blocks, output_stride,
 268                   store_non_strided_activations=False,
 269                   outputs_collections='output')
 270               output_end_points = slim.utils.convert_collection_to_dict(
 271                   'output')
 272 
 273               # Make the two networks use the same weights.
 274               tf.get_variable_scope().reuse_variables()
 275 
 276               # Subsample activations at the end of the blocks.
 277               expected = resnet_utils.stack_blocks_dense(
 278                   inputs, blocks, output_stride,
 279                   store_non_strided_activations=True,
 280                   outputs_collections='expected')
 281               expected_end_points = slim.utils.convert_collection_to_dict(
 282                   'expected')
 283 
 284               sess.run(tf.global_variables_initializer())
 285 
 286               # Make sure that the final output is the same.
 287               output, expected = sess.run([output, expected])
 288               self.assertAllClose(output, expected, atol=1e-4, rtol=1e-4)
 289 
 290               # Make sure that intermediate block activations in
 291               # output_end_points are subsampled versions of the corresponding
 292               # ones in expected_end_points.
 293               for i, block in enumerate(blocks[:-1:]):
 294                 output = output_end_points[block.scope]
 295                 expected = expected_end_points[block.scope]
 296                 atrous_activated = (output_stride is not None and
 297                                     2 ** i >= output_stride)
 298                 if not atrous_activated:
 299                   expected = resnet_utils.subsample(expected, 2)
 300                 output, expected = sess.run([output, expected])
 301                 self.assertAllClose(output, expected, atol=1e-4, rtol=1e-4)

 549   def testDepthMultiplier(self):
 550     resnets = [
 551         resnet_v1.resnet_v1_50, resnet_v1.resnet_v1_101,
 552         resnet_v1.resnet_v1_152, resnet_v1.resnet_v1_200
 553     ]
 554     resnet_names = [
 555         'resnet_v1_50', 'resnet_v1_101', 'resnet_v1_152', 'resnet_v1_200'
 556     ]
 557     for resnet, resnet_name in zip(resnets, resnet_names):
 558       depth_multiplier = 0.25
 559       global_pool = True
 560       num_classes = 10
 561       inputs = create_test_input(2, 224, 224, 3)
 562       with slim.arg_scope(resnet_utils.resnet_arg_scope()):
 563         scope_base = resnet_name + '_base'
 564         _, end_points_base = resnet(
 565             inputs,
 566             num_classes,
 567             global_pool=global_pool,
 568             min_base_depth=1,
 569             scope=scope_base)
 570         scope_test = resnet_name + '_test'
 571         _, end_points_test = resnet(
 572             inputs,
 573             num_classes,
 574             global_pool=global_pool,
 575             min_base_depth=1,
 576             depth_multiplier=depth_multiplier,
 577             scope=scope_test)
 578         for block in ['block1', 'block2', 'block3', 'block4']:
 579           block_name_base = scope_base + '/' + block
 580           block_name_test = scope_test + '/' + block
 581           self.assertTrue(block_name_base in end_points_base)
 582           self.assertTrue(block_name_test in end_points_test)
 583           self.assertEqual(
 584               len(end_points_base[block_name_base].get_shape().as_list()), 4)
 585           self.assertEqual(
 586               len(end_points_test[block_name_test].get_shape().as_list()), 4)
 587           self.assertListEqual(
 588               end_points_base[block_name_base].get_shape().as_list()[:3],
 589               end_points_test[block_name_test].get_shape().as_list()[:3])
 590           self.assertEqual(
 591               int(depth_multiplier *
 592                   end_points_base[block_name_base].get_shape().as_list()[3]),
 593               end_points_test[block_name_test].get_shape().as_list()[3])
