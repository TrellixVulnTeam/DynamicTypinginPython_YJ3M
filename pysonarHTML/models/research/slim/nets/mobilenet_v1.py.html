 114 # Conv and DepthSepConv namedtuple define layers of the MobileNet architecture
 115 # Conv defines 3x3 convolution layers
 116 # DepthSepConv defines 3x3 depthwise convolution followed by 1x1 convolution.
 117 # stride is the stride of the convolution
 118 # depth is the number of channels or filters in a layer
 119 Conv = namedtuple('Conv', ['kernel', 'stride', 'depth'])
 120 DepthSepConv = namedtuple('DepthSepConv', ['kernel', 'stride', 'depth'])

 141 def _fixed_padding(inputs, kernel_size, rate=1):
 142   """Pads the input along the spatial dimensions independently of input size.
 143 
 144   Pads the input such that if it was used in a convolution with 'VALID' padding,
 145   the output would have the same dimensions as if the unpadded input was used
 146   in a convolution with 'SAME' padding.
 147 
 148   Args:
 149     inputs: A tensor of size [batch, height_in, width_in, channels].
 150     kernel_size: The kernel to be used in the conv2d or max_pool2d operation.
 151     rate: An integer, rate for atrous convolution.
 152 
 153   Returns:
 154     output: A tensor of size [batch, height_out, width_out, channels] with the
 155       input, either intact (if kernel_size == 1) or padded (if kernel_size > 1).
 156   """
 157   kernel_size_effective = [kernel_size[0] + (kernel_size[0] - 1) * (rate - 1),
 158                            kernel_size[0] + (kernel_size[0] - 1) * (rate - 1)]
 159   pad_total = [kernel_size_effective[0] - 1, kernel_size_effective[1] - 1]
 160   pad_beg = [pad_total[0] // 2, pad_total[1] // 2]
 161   pad_end = [pad_total[0] - pad_beg[0], pad_total[1] - pad_beg[1]]
 162   padded_inputs = tf.pad(
 163       tensor=inputs,
 164       paddings=[[0, 0], [pad_beg[0], pad_end[0]], [pad_beg[1], pad_end[1]],
 165                 [0, 0]])
 166   return padded_inputs

 169 def mobilenet_v1_base(inputs,
 170                       final_endpoint='Conv2d_13_pointwise',
 171                       min_depth=8,
 172                       depth_multiplier=1.0,
 173                       conv_defs=None,
 174                       output_stride=None,
 175                       use_explicit_padding=False,
 176                       scope=None):
 177   """Mobilenet v1.
 178 
 179   Constructs a Mobilenet v1 network from inputs to the given final endpoint.
 180 
 181   Args:
 182     inputs: a tensor of shape [batch_size, height, width, channels].
 183     final_endpoint: specifies the endpoint to construct the network up to. It
 184       can be one of ['Conv2d_0', 'Conv2d_1_pointwise', 'Conv2d_2_pointwise',
 185       'Conv2d_3_pointwise', 'Conv2d_4_pointwise', 'Conv2d_5'_pointwise,
 186       'Conv2d_6_pointwise', 'Conv2d_7_pointwise', 'Conv2d_8_pointwise',
 187       'Conv2d_9_pointwise', 'Conv2d_10_pointwise', 'Conv2d_11_pointwise',
 188       'Conv2d_12_pointwise', 'Conv2d_13_pointwise'].
 189     min_depth: Minimum depth value (number of channels) for all convolution ops.
 190       Enforced when depth_multiplier < 1, and not an active constraint when
 191       depth_multiplier >= 1.
 192     depth_multiplier: Float multiplier for the depth (number of channels)
 193       for all convolution ops. The value must be greater than zero. Typical
 194       usage will be to set this value in (0, 1) to reduce the number of
 195       parameters or computation cost of the model.
 196     conv_defs: A list of ConvDef namedtuples specifying the net architecture.
 197     output_stride: An integer that specifies the requested ratio of input to
 198       output spatial resolution. If not None, then we invoke atrous convolution
 199       if necessary to prevent the network from reducing the spatial resolution
 200       of the activation maps. Allowed values are 8 (accurate fully convolutional
 201       mode), 16 (fast fully convolutional mode), 32 (classification mode).
 202     use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
 203       inputs so that the output dimensions are the same as if 'SAME' padding
 204       were used.
 205     scope: Optional variable_scope.
 206 
 207   Returns:
 208     tensor_out: output tensor corresponding to the final_endpoint.
 209     end_points: a set of activations for external use, for example summaries or
 210                 losses.
 211 
 212   Raises:
 213     ValueError: if final_endpoint is not set to one of the predefined values,
 214                 or depth_multiplier <= 0, or the target output_stride is not
 215                 allowed.
 216   """
 217   depth = lambda d: max(int(d * depth_multiplier), min_depth)
 218   end_points = {}
 219 
 220   # Used to find thinned depths for each layer.
 221   if depth_multiplier <= 0:
 222     raise ValueError('depth_multiplier is not greater than zero.')
 223 
 224   if conv_defs is None:
 225     conv_defs = MOBILENETV1_CONV_DEFS
 226 
 227   if output_stride is not None and output_stride not in [8, 16, 32]:
 228     raise ValueError('Only allowed output_stride values are 8, 16, 32.')
 229 
 230   padding = 'SAME'
 231   if use_explicit_padding:
 232     padding = 'VALID'
 233   with tf.variable_scope(scope, 'MobilenetV1', [inputs]):
 234     with slim.arg_scope([slim.conv2d, slim.separable_conv2d], padding=padding):
 235       # The current_stride variable keeps track of the output stride of the
 236       # activations, i.e., the running product of convolution strides up to the
 237       # current network layer. This allows us to invoke atrous convolution
 238       # whenever applying the next convolution would result in the activations
 239       # having output stride larger than the target output_stride.
 240       current_stride = 1
 241 
 242       # The atrous convolution rate parameter.
 243       rate = 1
 244 
 245       net = inputs
 246       for i, conv_def in enumerate(conv_defs):
 247         end_point_base = 'Conv2d_%d' % i
 248 
 249         if output_stride is not None and current_stride == output_stride:
 250           # If we have reached the target output_stride, then we need to employ
 251           # atrous convolution with stride=1 and multiply the atrous rate by the
 252           # current unit's stride for use in subsequent layers.
 253           layer_stride = 1
 254           layer_rate = rate
 255           rate *= conv_def.stride
 256         else:
 257           layer_stride = conv_def.stride
 258           layer_rate = 1
 259           current_stride *= conv_def.stride
 260 
 261         if isinstance(conv_def, Conv):
 262           end_point = end_point_base
 263           if use_explicit_padding:
 264             net = _fixed_padding(net, conv_def.kernel)
 265           net = slim.conv2d(net, depth(conv_def.depth), conv_def.kernel,
 266                             stride=conv_def.stride,
 267                             scope=end_point)
 268           end_points[end_point] = net
 269           if end_point == final_endpoint:
 270             return net, end_points
 271 
 272         elif isinstance(conv_def, DepthSepConv):
 273           end_point = end_point_base + '_depthwise'
 274 
 275           # By passing filters=None
 276           # separable_conv2d produces only a depthwise convolution layer
 277           if use_explicit_padding:
 278             net = _fixed_padding(net, conv_def.kernel, layer_rate)
 279           net = slim.separable_conv2d(net, None, conv_def.kernel,
 280                                       depth_multiplier=1,
 281                                       stride=layer_stride,
 282                                       rate=layer_rate,
 283                                       scope=end_point)
 284 
 285           end_points[end_point] = net
 286           if end_point == final_endpoint:
 287             return net, end_points
 288 
 289           end_point = end_point_base + '_pointwise'
 290 
 291           net = slim.conv2d(net, depth(conv_def.depth), [1, 1],
 292                             stride=1,
 293                             scope=end_point)
 294 
 295           end_points[end_point] = net
 296           if end_point == final_endpoint:
 297             return net, end_points
 298         else:
 299           raise ValueError('Unknown convolution type %s for layer %d'
 300                            % (conv_def.ltype, i))
 301   raise ValueError('Unknown final endpoint %s' % final_endpoint)

 304 def mobilenet_v1(inputs,
 305                  num_classes=1000,
 306                  dropout_keep_prob=0.999,
 307                  is_training=True,
 308                  min_depth=8,
 309                  depth_multiplier=1.0,
 310                  conv_defs=None,
 311                  prediction_fn=slim.softmax,
 312                  spatial_squeeze=True,
 313                  reuse=None,
 314                  scope='MobilenetV1',
 315                  global_pool=False):
 316   """Mobilenet v1 model for classification.
 317 
 318   Args:
 319     inputs: a tensor of shape [batch_size, height, width, channels].
 320     num_classes: number of predicted classes. If 0 or None, the logits layer
 321       is omitted and the input features to the logits layer (before dropout)
 322       are returned instead.
 323     dropout_keep_prob: the percentage of activation values that are retained.
 324     is_training: whether is training or not.
 325     min_depth: Minimum depth value (number of channels) for all convolution ops.
 326       Enforced when depth_multiplier < 1, and not an active constraint when
 327       depth_multiplier >= 1.
 328     depth_multiplier: Float multiplier for the depth (number of channels)
 329       for all convolution ops. The value must be greater than zero. Typical
 330       usage will be to set this value in (0, 1) to reduce the number of
 331       parameters or computation cost of the model.
 332     conv_defs: A list of ConvDef namedtuples specifying the net architecture.
 333     prediction_fn: a function to get predictions out of logits.
 334     spatial_squeeze: if True, logits is of shape is [B, C], if false logits is
 335         of shape [B, 1, 1, C], where B is batch_size and C is number of classes.
 336     reuse: whether or not the network and its variables should be reused. To be
 337       able to reuse 'scope' must be given.
 338     scope: Optional variable_scope.
 339     global_pool: Optional boolean flag to control the avgpooling before the
 340       logits layer. If false or unset, pooling is done with a fixed window
 341       that reduces default-sized inputs to 1x1, while larger inputs lead to
 342       larger outputs. If true, any input size is pooled down to 1x1.
 343 
 344   Returns:
 345     net: a 2D Tensor with the logits (pre-softmax activations) if num_classes
 346       is a non-zero integer, or the non-dropped-out input to the logits layer
 347       if num_classes is 0 or None.
 348     end_points: a dictionary from components of the network to the corresponding
 349       activation.
 350 
 351   Raises:
 352     ValueError: Input rank is invalid.
 353   """
 354   input_shape = inputs.get_shape().as_list()
 355   if len(input_shape) != 4:
 356     raise ValueError('Invalid input tensor rank, expected 4, was: %d' %
 357                      len(input_shape))
 358 
 359   with tf.variable_scope(
 360       scope, 'MobilenetV1', [inputs], reuse=reuse) as scope:
 361     with slim.arg_scope([slim.batch_norm, slim.dropout],
 362                         is_training=is_training):
 363       net, end_points = mobilenet_v1_base(inputs, scope=scope,
 364                                           min_depth=min_depth,
 365                                           depth_multiplier=depth_multiplier,
 366                                           conv_defs=conv_defs)
 367       with tf.variable_scope('Logits'):
 368         if global_pool:
 369           # Global average pooling.
 370           net = tf.reduce_mean(
 371               input_tensor=net, axis=[1, 2], keepdims=True, name='global_pool')
 372           end_points['global_pool'] = net
 373         else:
 374           # Pooling with a fixed kernel size.
 375           kernel_size = _reduced_kernel_size_for_small_input(net, [7, 7])
 376           net = slim.avg_pool2d(net, kernel_size, padding='VALID',
 377                                 scope='AvgPool_1a')
 378           end_points['AvgPool_1a'] = net
 379         if not num_classes:
 380           return net, end_points
 381         # 1 x 1 x 1024
 382         net = slim.dropout(net, keep_prob=dropout_keep_prob, scope='Dropout_1b')
 383         logits = slim.conv2d(net, num_classes, [1, 1], activation_fn=None,
 384                              normalizer_fn=None, scope='Conv2d_1c_1x1')
 385         if spatial_squeeze:
 386           logits = tf.squeeze(logits, [1, 2], name='SpatialSqueeze')
 387       end_points['Logits'] = logits
 388       if prediction_fn:
 389         end_points['Predictions'] = prediction_fn(logits, scope='Predictions')
 390   return logits, end_points

 395 def wrapped_partial(func, *args, **kwargs):

 406 def _reduced_kernel_size_for_small_input(input_tensor, kernel_size):
 407   """Define kernel size which is automatically reduced for small input.
 408 
 409   If the shape of the input images is unknown at graph construction time this
 410   function assumes that the input images are large enough.
 411 
 412   Args:
 413     input_tensor: input tensor of size [batch_size, height, width, channels].
 414     kernel_size: desired kernel size of length 2: [kernel_height, kernel_width]
 415 
 416   Returns:
 417     a tensor with the kernel size.
 418   """
 419   shape = input_tensor.get_shape().as_list()
 420   if shape[1] is None or shape[2] is None:
 421     kernel_size_out = kernel_size
 422   else:
 423     kernel_size_out = [min(shape[1], kernel_size[0]),
 424                        min(shape[2], kernel_size[1])]
 425   return kernel_size_out

 428 def mobilenet_v1_arg_scope(
 429     is_training=True,
 430     weight_decay=0.00004,
 431     stddev=0.09,
 432     regularize_depthwise=False,
 433     batch_norm_decay=0.9997,
 434     batch_norm_epsilon=0.001,
 435     batch_norm_updates_collections=tf.GraphKeys.UPDATE_OPS,
 436     normalizer_fn=slim.batch_norm):
 437   """Defines the default MobilenetV1 arg scope.
 438 
 439   Args:
 440     is_training: Whether or not we're training the model. If this is set to
 441       None, the parameter is not added to the batch_norm arg_scope.
 442     weight_decay: The weight decay to use for regularizing the model.
 443     stddev: The standard deviation of the trunctated normal weight initializer.
 444     regularize_depthwise: Whether or not apply regularization on depthwise.
 445     batch_norm_decay: Decay for batch norm moving average.
 446     batch_norm_epsilon: Small float added to variance to avoid dividing by zero
 447       in batch norm.
 448     batch_norm_updates_collections: Collection for the update ops for
 449       batch norm.
 450     normalizer_fn: Normalization function to apply after convolution.
 451 
 452   Returns:
 453     An `arg_scope` to use for the mobilenet v1 model.
 454   """
 455   batch_norm_params = {
 456       'center': True,
 457       'scale': True,
 458       'decay': batch_norm_decay,
 459       'epsilon': batch_norm_epsilon,
 460       'updates_collections': batch_norm_updates_collections,
 461   }
 462   if is_training is not None:
 463     batch_norm_params['is_training'] = is_training
 464 
 465   # Set weight_decay for weights in Conv and DepthSepConv layers.
 466   weights_init = tf.truncated_normal_initializer(stddev=stddev)
 467   regularizer = slim.l2_regularizer(weight_decay)
 468   if regularize_depthwise:
 469     depthwise_regularizer = regularizer
 470   else:
 471     depthwise_regularizer = None
 472   with slim.arg_scope([slim.conv2d, slim.separable_conv2d],
 473                       weights_initializer=weights_init,
 474                       activation_fn=tf.nn.relu6, normalizer_fn=normalizer_fn):
 475     with slim.arg_scope([slim.batch_norm], **batch_norm_params):
 476       with slim.arg_scope([slim.conv2d], weights_regularizer=regularizer):
 477         with slim.arg_scope([slim.separable_conv2d],
 478                             weights_regularizer=depthwise_regularizer) as sc:
 479           return sc
