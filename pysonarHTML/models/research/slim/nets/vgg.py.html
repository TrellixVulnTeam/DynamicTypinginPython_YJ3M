  15 """Contains model definitions for versions of the Oxford VGG network.
  16 
  17 These model definitions were introduced in the following technical report:
  18 
  19   Very Deep Convolutional Networks For Large-Scale Image Recognition
  20   Karen Simonyan and Andrew Zisserman
  21   arXiv technical report, 2015
  22   PDF: http://arxiv.org/pdf/1409.1556.pdf
  23   ILSVRC 2014 Slides: http://www.robots.ox.ac.uk/~karen/pdf/ILSVRC_2014.pdf
  24   CC-BY-4.0
  25 
  26 More information can be obtained from the VGG website:
  27 www.robots.ox.ac.uk/~vgg/research/very_deep/
  28 
  29 Usage:
  30   with slim.arg_scope(vgg.vgg_arg_scope()):
  31     outputs, end_points = vgg.vgg_a(inputs)
  32 
  33   with slim.arg_scope(vgg.vgg_arg_scope()):
  34     outputs, end_points = vgg.vgg_16(inputs)
  35 
  36 @@vgg_a
  37 @@vgg_16

  39 """

  48 def vgg_arg_scope(weight_decay=0.0005):
  49   """Defines the VGG arg scope.
  50 
  51   Args:
  52     weight_decay: The l2 regularization coefficient.
  53 
  54   Returns:
  55     An arg_scope.
  56   """
  57   with slim.arg_scope([slim.conv2d, slim.fully_connected],
  58                       activation_fn=tf.nn.relu,
  59                       weights_regularizer=slim.l2_regularizer(weight_decay),
  60                       biases_initializer=tf.zeros_initializer()):
  61     with slim.arg_scope([slim.conv2d], padding='SAME') as arg_sc:
  62       return arg_sc

  65 def vgg_a(inputs,
  66           num_classes=1000,
  67           is_training=True,
  68           dropout_keep_prob=0.5,
  69           spatial_squeeze=True,
  70           reuse=None,
  71           scope='vgg_a',
  72           fc_conv_padding='VALID',
  73           global_pool=False):
  74   """Oxford Net VGG 11-Layers version A Example.
  75 
  76   Note: All the fully_connected layers have been transformed to conv2d layers.
  77         To use in classification mode, resize input to 224x224.
  78 
  79   Args:
  80     inputs: a tensor of size [batch_size, height, width, channels].
  81     num_classes: number of predicted classes. If 0 or None, the logits layer is
  82       omitted and the input features to the logits layer are returned instead.
  83     is_training: whether or not the model is being trained.
  84     dropout_keep_prob: the probability that activations are kept in the dropout
  85       layers during training.
  86     spatial_squeeze: whether or not should squeeze the spatial dimensions of the
  87       outputs. Useful to remove unnecessary dimensions for classification.
  88     reuse: whether or not the network and its variables should be reused. To be
  89       able to reuse 'scope' must be given.
  90     scope: Optional scope for the variables.
  91     fc_conv_padding: the type of padding to use for the fully connected layer
  92       that is implemented as a convolutional layer. Use 'SAME' padding if you
  93       are applying the network in a fully convolutional manner and want to
  94       get a prediction map downsampled by a factor of 32 as an output.
  95       Otherwise, the output prediction map will be (input / 32) - 6 in case of
  96       'VALID' padding.
  97     global_pool: Optional boolean flag. If True, the input to the classification
  98       layer is avgpooled to size 1x1, for any input size. (This is not part
  99       of the original VGG architecture.)
 100 
 101   Returns:
 102     net: the output of the logits layer (if num_classes is a non-zero integer),
 103       or the input to the logits layer (if num_classes is 0 or None).
 104     end_points: a dict of tensors with intermediate activations.
 105   """
 106   with tf.variable_scope(scope, 'vgg_a', [inputs], reuse=reuse) as sc:
 107     end_points_collection = sc.original_name_scope + '_end_points'
 108     # Collect outputs for conv2d, fully_connected and max_pool2d.
 109     with slim.arg_scope([slim.conv2d, slim.max_pool2d],
 110                         outputs_collections=end_points_collection):
 111       net = slim.repeat(inputs, 1, slim.conv2d, 64, [3, 3], scope='conv1')
 112       net = slim.max_pool2d(net, [2, 2], scope='pool1')
 113       net = slim.repeat(net, 1, slim.conv2d, 128, [3, 3], scope='conv2')
 114       net = slim.max_pool2d(net, [2, 2], scope='pool2')
 115       net = slim.repeat(net, 2, slim.conv2d, 256, [3, 3], scope='conv3')
 116       net = slim.max_pool2d(net, [2, 2], scope='pool3')
 117       net = slim.repeat(net, 2, slim.conv2d, 512, [3, 3], scope='conv4')
 118       net = slim.max_pool2d(net, [2, 2], scope='pool4')
 119       net = slim.repeat(net, 2, slim.conv2d, 512, [3, 3], scope='conv5')
 120       net = slim.max_pool2d(net, [2, 2], scope='pool5')
 121 
 122       # Use conv2d instead of fully_connected layers.
 123       net = slim.conv2d(net, 4096, [7, 7], padding=fc_conv_padding, scope='fc6')
 124       net = slim.dropout(net, dropout_keep_prob, is_training=is_training,
 125                          scope='dropout6')
 126       net = slim.conv2d(net, 4096, [1, 1], scope='fc7')
 127       # Convert end_points_collection into a end_point dict.
 128       end_points = slim.utils.convert_collection_to_dict(end_points_collection)
 129       if global_pool:
 130         net = tf.reduce_mean(
 131             input_tensor=net, axis=[1, 2], keepdims=True, name='global_pool')
 132         end_points['global_pool'] = net
 133       if num_classes:
 134         net = slim.dropout(net, dropout_keep_prob, is_training=is_training,
 135                            scope='dropout7')
 136         net = slim.conv2d(net, num_classes, [1, 1],
 137                           activation_fn=None,
 138                           normalizer_fn=None,
 139                           scope='fc8')
 140         if spatial_squeeze:
 141           net = tf.squeeze(net, [1, 2], name='fc8/squeezed')
 142         end_points[sc.name + '/fc8'] = net
 143       return net, end_points
 144 vgg_a.default_image_size = 224
