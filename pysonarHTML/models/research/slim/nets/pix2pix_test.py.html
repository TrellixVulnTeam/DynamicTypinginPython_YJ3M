  26 class GeneratorTest(tf.test.TestCase):
  27 
  28   def _reduced_default_blocks(self):
  29     """Returns the default blocks, scaled down to make test run faster."""
  30     return [pix2pix.Block(b.num_filters // 32, b.decoder_keep_prob)
  31             for b in pix2pix._default_generator_blocks()]
  32 
  33   def test_output_size_nn_upsample_conv(self):
  34     batch_size = 2
  35     height, width = 256, 256
  36     num_outputs = 4
  37 
  38     images = tf.ones((batch_size, height, width, 3))
  39     with slim.arg_scope(pix2pix.pix2pix_arg_scope()):
  40       logits, _ = pix2pix.pix2pix_generator(
  41           images, num_outputs, blocks=self._reduced_default_blocks(),
  42           upsample_method='nn_upsample_conv')
  43 
  44     with self.test_session() as session:
  45       session.run(tf.global_variables_initializer())
  46       np_outputs = session.run(logits)
  47       self.assertListEqual([batch_size, height, width, num_outputs],
  48                            list(np_outputs.shape))
  49

  67   def test_block_number_dictates_number_of_layers(self):
  68     batch_size = 2
  69     height, width = 256, 256
  70     num_outputs = 4
  71 
  72     images = tf.ones((batch_size, height, width, 3))
  73     blocks = [
  74         pix2pix.Block(64, 0.5),
  75         pix2pix.Block(128, 0),
  76     ]
  77     with slim.arg_scope(pix2pix.pix2pix_arg_scope()):
  78       _, end_points = pix2pix.pix2pix_generator(
  79           images, num_outputs, blocks)
  80 
  81     num_encoder_layers = 0
  82     num_decoder_layers = 0
  83     for end_point in end_points:
  84       if end_point.startswith('encoder'):
  85         num_encoder_layers += 1
  86       elif end_point.startswith('decoder'):
  87         num_decoder_layers += 1
  88 
  89     self.assertEqual(num_encoder_layers, len(blocks))
  90     self.assertEqual(num_decoder_layers, len(blocks))

  93 class DiscriminatorTest(tf.test.TestCase):
  94 
  95   def _layer_output_size(self, input_size, kernel_size=4, stride=2, pad=2):
  96     return (input_size + pad * 2 - kernel_size) // stride + 1
  97 
  98   def test_four_layers(self):
  99     batch_size = 2
 100     input_size = 256
 101 
 102     output_size = self._layer_output_size(input_size)
 103     output_size = self._layer_output_size(output_size)
 104     output_size = self._layer_output_size(output_size)
 105     output_size = self._layer_output_size(output_size, stride=1)
 106     output_size = self._layer_output_size(output_size, stride=1)
 107 
 108     images = tf.ones((batch_size, input_size, input_size, 3))
 109     with slim.arg_scope(pix2pix.pix2pix_arg_scope()):
 110       logits, end_points = pix2pix.pix2pix_discriminator(
 111           images, num_filters=[64, 128, 256, 512])
 112     self.assertListEqual([batch_size, output_size, output_size, 1],
 113                          logits.shape.as_list())
 114     self.assertListEqual([batch_size, output_size, output_size, 1],
 115                          end_points['predictions'].shape.as_list())
 116 
 117   def test_four_layers_no_padding(self):
 118     batch_size = 2
 119     input_size = 256
 120 
 121     output_size = self._layer_output_size(input_size, pad=0)
 122     output_size = self._layer_output_size(output_size, pad=0)
 123     output_size = self._layer_output_size(output_size, pad=0)
 124     output_size = self._layer_output_size(output_size, stride=1, pad=0)
 125     output_size = self._layer_output_size(output_size, stride=1, pad=0)
 126 
 127     images = tf.ones((batch_size, input_size, input_size, 3))
 128     with slim.arg_scope(pix2pix.pix2pix_arg_scope()):
 129       logits, end_points = pix2pix.pix2pix_discriminator(
 130           images, num_filters=[64, 128, 256, 512], padding=0)
 131     self.assertListEqual([batch_size, output_size, output_size, 1],
 132                          logits.shape.as_list())
 133     self.assertListEqual([batch_size, output_size, output_size, 1],
 134                          end_points['predictions'].shape.as_list())
