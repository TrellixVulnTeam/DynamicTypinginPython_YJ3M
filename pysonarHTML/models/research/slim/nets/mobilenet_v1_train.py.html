  55 _LEARNING_RATE_DECAY_FACTOR = 0.94

  58 def get_learning_rate():
  59   if FLAGS.fine_tune_checkpoint:
  60     # If we are fine tuning a checkpoint we need to start at a lower learning
  61     # rate since we are farther along on training.
  62     return 1e-4
  63   else:
  64     return 0.045
  65 
  66 
  67 def get_quant_delay():
  68   if FLAGS.fine_tune_checkpoint:
  69     # We can start quantizing immediately if we are finetuning.
  70     return 0
  71   else:
  72     # We need to wait for the model to train a bit before we quantize if we are
  73     # training from scratch.
  74     return 250000

  77 def imagenet_input(is_training):
  78   """Data reader for imagenet.
  79 
  80   Reads in imagenet data and performs pre-processing on the images.
  81 
  82   Args:
  83      is_training: bool specifying if train or validation dataset is needed.
  84   Returns:
  85      A batch of images and labels.
  86   """
  87   if is_training:
  88     dataset = dataset_factory.get_dataset('imagenet', 'train',
  89                                           FLAGS.dataset_dir)
  90   else:
  91     dataset = dataset_factory.get_dataset('imagenet', 'validation',
  92                                           FLAGS.dataset_dir)
  93 
  94   provider = slim.dataset_data_provider.DatasetDataProvider(
  95       dataset,
  96       shuffle=is_training,
  97       common_queue_capacity=2 * FLAGS.batch_size,
  98       common_queue_min=FLAGS.batch_size)
  99   [image, label] = provider.get(['image', 'label'])
 100 
 101   image_preprocessing_fn = preprocessing_factory.get_preprocessing(
 102       'mobilenet_v1', is_training=is_training)
 103 
 104   image = image_preprocessing_fn(image, FLAGS.image_size, FLAGS.image_size)
 105 
 106   images, labels = tf.train.batch([image, label],
 107                                   batch_size=FLAGS.batch_size,
 108                                   num_threads=4,
 109                                   capacity=5 * FLAGS.batch_size)
 110   labels = slim.one_hot_encoding(labels, FLAGS.num_classes)
 111   return images, labels

 114 def build_model():
 115   """Builds graph for model to train with rewrites for quantization.
 116 
 117   Returns:
 118     g: Graph with fake quantization ops and batch norm folding suitable for
 119     training quantized weights.
 120     train_tensor: Train op for execution during training.
 121   """
 122   g = tf.Graph()
 123   with g.as_default(), tf.device(
 124       tf.train.replica_device_setter(FLAGS.ps_tasks)):
 125     inputs, labels = imagenet_input(is_training=True)
 126     with slim.arg_scope(mobilenet_v1.mobilenet_v1_arg_scope(is_training=True)):
 127       logits, _ = mobilenet_v1.mobilenet_v1(
 128           inputs,
 129           is_training=True,
 130           depth_multiplier=FLAGS.depth_multiplier,
 131           num_classes=FLAGS.num_classes)
 132 
 133     tf.losses.softmax_cross_entropy(labels, logits)
 134 
 135     # Call rewriter to produce graph with fake quant ops and folded batch norms
 136     # quant_delay delays start of quantization till quant_delay steps, allowing
 137     # for better model accuracy.
 138     if FLAGS.quantize:
 139       contrib_quantize.create_training_graph(quant_delay=get_quant_delay())
 140 
 141     total_loss = tf.losses.get_total_loss(name='total_loss')
 142     # Configure the learning rate using an exponential decay.
 143     num_epochs_per_decay = 2.5
 144     imagenet_size = 1271167
 145     decay_steps = int(imagenet_size / FLAGS.batch_size * num_epochs_per_decay)
 146 
 147     learning_rate = tf.train.exponential_decay(
 148         get_learning_rate(),
 149         tf.train.get_or_create_global_step(),
 150         decay_steps,
 151         _LEARNING_RATE_DECAY_FACTOR,
 152         staircase=True)
 153     opt = tf.train.GradientDescentOptimizer(learning_rate)
 154 
 155     train_tensor = slim.learning.create_train_op(
 156         total_loss,
 157         optimizer=opt)
 158 
 159   slim.summaries.add_scalar_summary(total_loss, 'total_loss', 'losses')
 160   slim.summaries.add_scalar_summary(learning_rate, 'learning_rate', 'training')
 161   return g, train_tensor

 164 def get_checkpoint_init_fn():
 165   """Returns the checkpoint init_fn if the checkpoint is provided."""
 166   if FLAGS.fine_tune_checkpoint:
 167     variables_to_restore = slim.get_variables_to_restore()
 168     global_step_reset = tf.assign(
 169         tf.train.get_or_create_global_step(), 0)
 170     # When restoring from a floating point model, the min/max values for
 171     # quantized weights and activations are not present.
 172     # We instruct slim to ignore variables that are missing during restoration
 173     # by setting ignore_missing_vars=True
 174     slim_init_fn = slim.assign_from_checkpoint_fn(
 175         FLAGS.fine_tune_checkpoint,
 176         variables_to_restore,
 177         ignore_missing_vars=True)
 178 
 179     def init_fn(sess):
 180       slim_init_fn(sess)
 181       # If we are restoring from a floating point model, we need to initialize
 182       # the global step to zero for the exponential decay to result in
 183       # reasonable learning rates.
 184       sess.run(global_step_reset)
 185     return init_fn
 186   else:
 187     return None

 190 def train_model():
 191   """Trains mobilenet_v1."""
 192   g, train_tensor = build_model()
 193   with g.as_default():
 194     slim.learning.train(
 195         train_tensor,
 196         FLAGS.checkpoint_dir,
 197         is_chief=(FLAGS.task == 0),
 198         master=FLAGS.master,
 199         log_every_n_steps=FLAGS.log_every_n_steps,
 200         graph=g,
 201         number_of_steps=FLAGS.number_of_steps,
 202         save_summaries_secs=FLAGS.save_summaries_secs,
 203         save_interval_secs=FLAGS.save_interval_secs,
 204         init_fn=get_checkpoint_init_fn(),
 205         global_step=tf.train.get_global_step())
 206

 208 def main(unused_arg):
 209   train_model()
