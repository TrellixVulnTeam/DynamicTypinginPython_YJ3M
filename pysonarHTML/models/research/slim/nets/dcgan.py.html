  27 def _validate_image_inputs(inputs):
  28   inputs.get_shape().assert_has_rank(4)
  29   inputs.get_shape()[1:3].assert_is_fully_defined()
  30   if inputs.get_shape()[1] != inputs.get_shape()[2]:
  31     raise ValueError('Input tensor does not have equal width and height: ',
  32                      inputs.get_shape()[1:3])
  33   width = inputs.get_shape().as_list()[1]
  34   if log(width, 2) != int(log(width, 2)):
  35     raise ValueError('Input tensor `width` is not a power of 2: ', width)

  38 # TODO(joelshor): Use fused batch norm by default. Investigate why some GAN
  39 # setups need the gradient of gradient FusedBatchNormGrad.
  40 def discriminator(inputs,
  41                   depth=64,
  42                   is_training=True,
  43                   reuse=None,
  44                   scope='Discriminator',
  45                   fused_batch_norm=False):
  46   """Discriminator network for DCGAN.
  47 
  48   Construct discriminator network from inputs to the final endpoint.
  49 
  50   Args:
  51     inputs: A tensor of size [batch_size, height, width, channels]. Must be
  52       floating point.
  53     depth: Number of channels in first convolution layer.
  54     is_training: Whether the network is for training or not.
  55     reuse: Whether or not the network variables should be reused. `scope`
  56       must be given to be reused.
  57     scope: Optional variable_scope.
  58     fused_batch_norm: If `True`, use a faster, fused implementation of
  59       batch norm.
  60 
  61   Returns:
  62     logits: The pre-softmax activations, a tensor of size [batch_size, 1]
  63     end_points: a dictionary from components of the network to their activation.
  64 
  65   Raises:
  66     ValueError: If the input image shape is not 4-dimensional, if the spatial
  67       dimensions aren't defined at graph construction time, if the spatial
  68       dimensions aren't square, or if the spatial dimensions aren't a power of
  69       two.
  70   """
  71 
  72   normalizer_fn = slim.batch_norm
  73   normalizer_fn_args = {
  74       'is_training': is_training,
  75       'zero_debias_moving_mean': True,
  76       'fused': fused_batch_norm,
  77   }
  78 
  79   _validate_image_inputs(inputs)
  80   inp_shape = inputs.get_shape().as_list()[1]
  81 
  82   end_points = {}
  83   with tf.variable_scope(
  84       scope, values=[inputs], reuse=reuse) as scope:
  85     with slim.arg_scope([normalizer_fn], **normalizer_fn_args):
  86       with slim.arg_scope([slim.conv2d],
  87                           stride=2,
  88                           kernel_size=4,
  89                           activation_fn=tf.nn.leaky_relu):
  90         net = inputs
  91         for i in xrange(int(log(inp_shape, 2))):
  92           scope = 'conv%i' % (i + 1)
  93           current_depth = depth * 2**i
  94           normalizer_fn_ = None if i == 0 else normalizer_fn
  95           net = slim.conv2d(
  96               net, current_depth, normalizer_fn=normalizer_fn_, scope=scope)
  97           end_points[scope] = net
  98 
  99         logits = slim.conv2d(net, 1, kernel_size=1, stride=1, padding='VALID',
 100                              normalizer_fn=None, activation_fn=None)
 101         logits = tf.reshape(logits, [-1, 1])
 102         end_points['logits'] = logits
 103 
 104         return logits, end_points

 109 def generator(inputs,
 110               depth=64,
 111               final_size=32,
 112               num_outputs=3,
 113               is_training=True,
 114               reuse=None,
 115               scope='Generator',
 116               fused_batch_norm=False):
 117   """Generator network for DCGAN.
 118 
 119   Construct generator network from inputs to the final endpoint.
 120 
 121   Args:
 122     inputs: A tensor with any size N. [batch_size, N]
 123     depth: Number of channels in last deconvolution layer.
 124     final_size: The shape of the final output.
 125     num_outputs: Number of output features. For images, this is the number of
 126       channels.
 127     is_training: whether is training or not.
 128     reuse: Whether or not the network has its variables should be reused. scope
 129       must be given to be reused.
 130     scope: Optional variable_scope.
 131     fused_batch_norm: If `True`, use a faster, fused implementation of
 132       batch norm.
 133 
 134   Returns:
 135     logits: the pre-softmax activations, a tensor of size
 136       [batch_size, 32, 32, channels]
 137     end_points: a dictionary from components of the network to their activation.
 138 
 139   Raises:
 140     ValueError: If `inputs` is not 2-dimensional.
 141     ValueError: If `final_size` isn't a power of 2 or is less than 8.
 142   """
 143   normalizer_fn = slim.batch_norm
 144   normalizer_fn_args = {
 145       'is_training': is_training,
 146       'zero_debias_moving_mean': True,
 147       'fused': fused_batch_norm,
 148   }
 149 
 150   inputs.get_shape().assert_has_rank(2)
 151   if log(final_size, 2) != int(log(final_size, 2)):
 152     raise ValueError('`final_size` (%i) must be a power of 2.' % final_size)
 153   if final_size < 8:
 154     raise ValueError('`final_size` (%i) must be greater than 8.' % final_size)
 155 
 156   end_points = {}
 157   num_layers = int(log(final_size, 2)) - 1
 158   with tf.variable_scope(
 159       scope, values=[inputs], reuse=reuse) as scope:
 160     with slim.arg_scope([normalizer_fn], **normalizer_fn_args):
 161       with slim.arg_scope([slim.conv2d_transpose],
 162                           normalizer_fn=normalizer_fn,
 163                           stride=2,
 164                           kernel_size=4):
 165         net = tf.expand_dims(tf.expand_dims(inputs, 1), 1)
 166 
 167         # First upscaling is different because it takes the input vector.
 168         current_depth = depth * 2 ** (num_layers - 1)
 169         scope = 'deconv1'
 170         net = slim.conv2d_transpose(
 171             net, current_depth, stride=1, padding='VALID', scope=scope)
 172         end_points[scope] = net
 173 
 174         for i in xrange(2, num_layers):
 175           scope = 'deconv%i' % (i)
 176           current_depth = depth * 2 ** (num_layers - i)
 177           net = slim.conv2d_transpose(net, current_depth, scope=scope)
 178           end_points[scope] = net
 179 
 180         # Last layer has different normalizer and activation.
 181         scope = 'deconv%i' % (num_layers)
 182         net = slim.conv2d_transpose(
 183             net, depth, normalizer_fn=None, activation_fn=None, scope=scope)
 184         end_points[scope] = net
 185 
 186         # Convert to proper channels.
 187         scope = 'logits'
 188         logits = slim.conv2d(
 189             net,
 190             num_outputs,
 191             normalizer_fn=None,
 192             activation_fn=None,
 193             kernel_size=1,
 194             stride=1,
 195             padding='VALID',
 196             scope=scope)
 197         end_points[scope] = logits
 198 
 199         logits.get_shape().assert_has_rank(4)
 200         logits.get_shape().assert_is_compatible_with(
 201             [None, final_size, final_size, num_outputs])
 202 
 203         return logits, end_points
