  29   def testBuildClassificationNetwork(self):
  30     batch_size = 5
  31     num_frames = 64
  32     height, width = 224, 224
  33     num_classes = 1000
  34 
  35     inputs = tf.random.uniform((batch_size, num_frames, height, width, 3))
  36     logits, end_points = s3dg.s3dg(inputs, num_classes)
  37     self.assertTrue(logits.op.name.startswith('InceptionV1/Logits'))
  38     self.assertListEqual(logits.get_shape().as_list(),
  39                          [batch_size, num_classes])
  40     self.assertTrue('Predictions' in end_points)
  41     self.assertListEqual(end_points['Predictions'].get_shape().as_list(),
  42                          [batch_size, num_classes])

  44   def testBuildBaseNetwork(self):
  45     batch_size = 5
  46     num_frames = 64
  47     height, width = 224, 224
  48 
  49     inputs = tf.random.uniform((batch_size, num_frames, height, width, 3))
  50     mixed_6c, end_points = s3dg.s3dg_base(inputs)
  51     self.assertTrue(mixed_6c.op.name.startswith('InceptionV1/Mixed_5c'))

  55                           'Conv2d_2c_3x3', 'MaxPool_3a_3x3', 'Mixed_3b',
  56                           'Mixed_3c', 'MaxPool_4a_3x3', 'Mixed_4b', 'Mixed_4c',
  57                           'Mixed_4d', 'Mixed_4e', 'Mixed_4f', 'MaxPool_5a_2x2',
  58                           'Mixed_5b', 'Mixed_5c']
  59     self.assertItemsEqual(list(end_points.keys()), expected_endpoints)

  61   def testBuildOnlyUptoFinalEndpointNoGating(self):
  62     batch_size = 5
  63     num_frames = 64
  64     height, width = 224, 224
  65     endpoints = ['Conv2d_1a_7x7', 'MaxPool_2a_3x3', 'Conv2d_2b_1x1',
  66                  'Conv2d_2c_3x3', 'MaxPool_3a_3x3', 'Mixed_3b', 'Mixed_3c',
  67                  'MaxPool_4a_3x3', 'Mixed_4b', 'Mixed_4c', 'Mixed_4d',
  68                  'Mixed_4e', 'Mixed_4f', 'MaxPool_5a_2x2', 'Mixed_5b',
  69                  'Mixed_5c']
  70     for index, endpoint in enumerate(endpoints):
  71       with tf.Graph().as_default():
  72         inputs = tf.random.uniform((batch_size, num_frames, height, width, 3))
  73         out_tensor, end_points = s3dg.s3dg_base(
  74             inputs, final_endpoint=endpoint, gating_startat=None)
  75         print(endpoint, out_tensor.op.name)
  76         self.assertTrue(out_tensor.op.name.startswith(
  77             'InceptionV1/' + endpoint))
  78         self.assertItemsEqual(endpoints[:index+1], end_points)

  80   def testBuildAndCheckAllEndPointsUptoMixed5c(self):
  81     batch_size = 5
  82     num_frames = 64
  83     height, width = 224, 224
  84 
  85     inputs = tf.random.uniform((batch_size, num_frames, height, width, 3))
  86     _, end_points = s3dg.s3dg_base(inputs,
  87                                    final_endpoint='Mixed_5c')
  88     endpoints_shapes = {'Conv2d_1a_7x7': [5, 32, 112, 112, 64],
  89                         'MaxPool_2a_3x3': [5, 32, 56, 56, 64],
  90                         'Conv2d_2b_1x1': [5, 32, 56, 56, 64],
  91                         'Conv2d_2c_3x3': [5, 32, 56, 56, 192],
  92                         'MaxPool_3a_3x3': [5, 32, 28, 28, 192],
  93                         'Mixed_3b': [5, 32, 28, 28, 256],
  94                         'Mixed_3c': [5, 32, 28, 28, 480],
  95                         'MaxPool_4a_3x3': [5, 16, 14, 14, 480],
  96                         'Mixed_4b': [5, 16, 14, 14, 512],
  97                         'Mixed_4c': [5, 16, 14, 14, 512],
  98                         'Mixed_4d': [5, 16, 14, 14, 512],
  99                         'Mixed_4e': [5, 16, 14, 14, 528],
 100                         'Mixed_4f': [5, 16, 14, 14, 832],
 101                         'MaxPool_5a_2x2': [5, 8, 7, 7, 832],
 102                         'Mixed_5b': [5, 8, 7, 7, 832],
 103                         'Mixed_5c': [5, 8, 7, 7, 1024]}
 104 
 105     self.assertItemsEqual(
 106         list(endpoints_shapes.keys()), list(end_points.keys()))
 107     for endpoint_name, expected_shape in six.iteritems(endpoints_shapes):
 108       self.assertTrue(endpoint_name in end_points)
 109       self.assertListEqual(end_points[endpoint_name].get_shape().as_list(),
 110                            expected_shape)

 134   def testEvaluation(self):
 135     batch_size = 2
 136     num_frames = 64
 137     height, width = 224, 224
 138     num_classes = 1000
 139 
 140     eval_inputs = tf.random.uniform((batch_size, num_frames, height, width, 3))
 141     logits, _ = s3dg.s3dg(eval_inputs, num_classes,
 142                           is_training=False)
 143     predictions = tf.argmax(input=logits, axis=1)
 144 
 145     with self.test_session() as sess:
 146       sess.run(tf.global_variables_initializer())
 147       output = sess.run(predictions)
 148       self.assertEquals(output.shape, (batch_size,))
