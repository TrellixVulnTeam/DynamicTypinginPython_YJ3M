  30   def testBuildClassificationNetwork(self):

  42     self.assertListEqual(end_points['Predictions'].get_shape().as_list(),
  43                          [batch_size, num_classes])

  74   def testBuildOnlyUptoFinalEndpoint(self):
  75     batch_size = 5
  76     height, width = 299, 299
  77     endpoints = ['Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3',
  78                  'MaxPool_3a_3x3', 'Conv2d_3b_1x1', 'Conv2d_4a_3x3',
  79                  'MaxPool_5a_3x3', 'Mixed_5b', 'Mixed_5c', 'Mixed_5d',
  80                  'Mixed_6a', 'Mixed_6b', 'Mixed_6c', 'Mixed_6d',
  81                  'Mixed_6e', 'Mixed_7a', 'Mixed_7b', 'Mixed_7c']
  82 
  83     for index, endpoint in enumerate(endpoints):
  84       with tf.Graph().as_default():
  85         inputs = tf.random.uniform((batch_size, height, width, 3))

  90         self.assertItemsEqual(endpoints[:index + 1], end_points.keys())

 134   def testBuildEndPoints(self):
 135     batch_size = 5
 136     height, width = 299, 299
 137     num_classes = 1000
 138 
 139     inputs = tf.random.uniform((batch_size, height, width, 3))
 140     _, end_points = inception.inception_v3(inputs, num_classes)
 141     self.assertTrue('Logits' in end_points)
 142     logits = end_points['Logits']
 143     self.assertListEqual(logits.get_shape().as_list(),
 144                          [batch_size, num_classes])
 145     self.assertTrue('AuxLogits' in end_points)
 146     aux_logits = end_points['AuxLogits']
 147     self.assertListEqual(aux_logits.get_shape().as_list(),
 148                          [batch_size, num_classes])
 149     self.assertTrue('Mixed_7c' in end_points)
 150     pre_pool = end_points['Mixed_7c']
 151     self.assertListEqual(pre_pool.get_shape().as_list(),
 152                          [batch_size, 8, 8, 2048])
 153     self.assertTrue('PreLogits' in end_points)
 154     pre_logits = end_points['PreLogits']
 155     self.assertListEqual(pre_logits.get_shape().as_list(),
 156                          [batch_size, 1, 1, 2048])

 158   def testBuildEndPointsWithDepthMultiplierLessThanOne(self):
 159     batch_size = 5
 160     height, width = 299, 299
 161     num_classes = 1000
 162 
 163     inputs = tf.random.uniform((batch_size, height, width, 3))
 164     _, end_points = inception.inception_v3(inputs, num_classes)
 165 
 166     endpoint_keys = [key for key in end_points.keys()
 167                      if key.startswith('Mixed') or key.startswith('Conv')]
 168 
 169     _, end_points_with_multiplier = inception.inception_v3(
 170         inputs, num_classes, scope='depth_multiplied_net',
 171         depth_multiplier=0.5)
 172 
 173     for key in endpoint_keys:
 174       original_depth = end_points[key].get_shape().as_list()[3]
 175       new_depth = end_points_with_multiplier[key].get_shape().as_list()[3]
 176       self.assertEqual(0.5 * original_depth, new_depth)

 223   def testUnknownImageShape(self):
 224     tf.reset_default_graph()
 225     batch_size = 2
 226     height, width = 299, 299
 227     num_classes = 1000
 228     input_np = np.random.uniform(0, 1, (batch_size, height, width, 3))
 229     with self.test_session() as sess:
 230       inputs = tf.placeholder(
 231           tf.float32, shape=(batch_size, None, None, 3))
 232       logits, end_points = inception.inception_v3(inputs, num_classes)
 233       self.assertListEqual(logits.get_shape().as_list(),
 234                            [batch_size, num_classes])
 235       pre_pool = end_points['Mixed_7c']
 236       feed_dict = {inputs: input_np}
 237       tf.global_variables_initializer().run()
 238       pre_pool_out = sess.run(pre_pool, feed_dict=feed_dict)
 239       self.assertListEqual(list(pre_pool_out.shape), [batch_size, 8, 8, 2048])

 260   def testUnknowBatchSize(self):
 261     batch_size = 1
 262     height, width = 299, 299
 263     num_classes = 1000
 264 
 265     inputs = tf.placeholder(tf.float32, (None, height, width, 3))
 266     logits, _ = inception.inception_v3(inputs, num_classes)
 267     self.assertTrue(logits.op.name.startswith('InceptionV3/Logits'))
 268     self.assertListEqual(logits.get_shape().as_list(),
 269                          [None, num_classes])
 270     images = tf.random.uniform((batch_size, height, width, 3))
 271 
 272     with self.test_session() as sess:
 273       sess.run(tf.global_variables_initializer())
 274       output = sess.run(logits, {inputs: images.eval()})
 275       self.assertEquals(output.shape, (batch_size, num_classes))

 331   def testBatchNormScale(self):
 332     height, width = 299, 299
 333     num_classes = 1000
 334     inputs = tf.placeholder(tf.float32, (1, height, width, 3))
 335     with slim.arg_scope(
 336         inception.inception_v3_arg_scope(batch_norm_scale=True)):
 337       inception.inception_v3(inputs, num_classes, is_training=False)
 338 
 339     gamma_names = set(
 340         v.op.name
 341         for v in tf.global_variables('.*/BatchNorm/gamma:0$'))
 342     self.assertGreater(len(gamma_names), 0)
 343     for v in tf.global_variables('.*/BatchNorm/moving_mean:0$'):
 344       self.assertIn(v.op.name[:-len('moving_mean')] + 'gamma', gamma_names)
