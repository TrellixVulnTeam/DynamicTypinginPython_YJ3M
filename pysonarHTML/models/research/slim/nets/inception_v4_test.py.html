  28   def testBuildLogits(self):
  29     batch_size = 5
  30     height, width = 299, 299
  31     num_classes = 1000
  32     inputs = tf.random.uniform((batch_size, height, width, 3))
  33     logits, end_points = inception.inception_v4(inputs, num_classes)
  34     auxlogits = end_points['AuxLogits']
  35     predictions = end_points['Predictions']
  36     self.assertTrue(auxlogits.op.name.startswith('InceptionV4/AuxLogits'))
  37     self.assertListEqual(auxlogits.get_shape().as_list(),
  38                          [batch_size, num_classes])
  39     self.assertTrue(logits.op.name.startswith('InceptionV4/Logits'))
  40     self.assertListEqual(logits.get_shape().as_list(),
  41                          [batch_size, num_classes])
  42     self.assertTrue(predictions.op.name.startswith(
  43         'InceptionV4/Logits/Predictions'))
  44     self.assertListEqual(predictions.get_shape().as_list(),
  45                          [batch_size, num_classes])

  70   def testAllEndPointsShapes(self):
  71     batch_size = 5
  72     height, width = 299, 299
  73     num_classes = 1000
  74     inputs = tf.random.uniform((batch_size, height, width, 3))
  75     _, end_points = inception.inception_v4(inputs, num_classes)
  76     endpoints_shapes = {'Conv2d_1a_3x3': [batch_size, 149, 149, 32],
  77                         'Conv2d_2a_3x3': [batch_size, 147, 147, 32],
  78                         'Conv2d_2b_3x3': [batch_size, 147, 147, 64],
  79                         'Mixed_3a': [batch_size, 73, 73, 160],
  80                         'Mixed_4a': [batch_size, 71, 71, 192],
  81                         'Mixed_5a': [batch_size, 35, 35, 384],
  82                         # 4 x Inception-A blocks
  83                         'Mixed_5b': [batch_size, 35, 35, 384],
  84                         'Mixed_5c': [batch_size, 35, 35, 384],
  85                         'Mixed_5d': [batch_size, 35, 35, 384],
  86                         'Mixed_5e': [batch_size, 35, 35, 384],
  87                         # Reduction-A block
  88                         'Mixed_6a': [batch_size, 17, 17, 1024],
  89                         # 7 x Inception-B blocks
  90                         'Mixed_6b': [batch_size, 17, 17, 1024],
  91                         'Mixed_6c': [batch_size, 17, 17, 1024],
  92                         'Mixed_6d': [batch_size, 17, 17, 1024],
  93                         'Mixed_6e': [batch_size, 17, 17, 1024],
  94                         'Mixed_6f': [batch_size, 17, 17, 1024],
  95                         'Mixed_6g': [batch_size, 17, 17, 1024],
  96                         'Mixed_6h': [batch_size, 17, 17, 1024],
  97                         # Reduction-A block
  98                         'Mixed_7a': [batch_size, 8, 8, 1536],
  99                         # 3 x Inception-C blocks
 100                         'Mixed_7b': [batch_size, 8, 8, 1536],
 101                         'Mixed_7c': [batch_size, 8, 8, 1536],
 102                         'Mixed_7d': [batch_size, 8, 8, 1536],
 103                         # Logits and predictions
 104                         'AuxLogits': [batch_size, num_classes],
 105                         'global_pool': [batch_size, 1, 1, 1536],
 106                         'PreLogitsFlatten': [batch_size, 1536],
 107                         'Logits': [batch_size, num_classes],
 108                         'Predictions': [batch_size, num_classes]}
 109     self.assertItemsEqual(endpoints_shapes.keys(), end_points.keys())
 110     for endpoint_name in endpoints_shapes:
 111       expected_shape = endpoints_shapes[endpoint_name]
 112       self.assertTrue(endpoint_name in end_points)
 113       self.assertListEqual(end_points[endpoint_name].get_shape().as_list(),
 114                            expected_shape)

 116   def testBuildBaseNetwork(self):
 117     batch_size = 5
 118     height, width = 299, 299
 119     inputs = tf.random.uniform((batch_size, height, width, 3))
 120     net, end_points = inception.inception_v4_base(inputs)
 121     self.assertTrue(net.op.name.startswith(
 122         'InceptionV4/Mixed_7d'))
 123     self.assertListEqual(net.get_shape().as_list(), [batch_size, 8, 8, 1536])
 124     expected_endpoints = [
 125         'Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3', 'Mixed_3a',
 126         'Mixed_4a', 'Mixed_5a', 'Mixed_5b', 'Mixed_5c', 'Mixed_5d',
 127         'Mixed_5e', 'Mixed_6a', 'Mixed_6b', 'Mixed_6c', 'Mixed_6d',
 128         'Mixed_6e', 'Mixed_6f', 'Mixed_6g', 'Mixed_6h', 'Mixed_7a',
 129         'Mixed_7b', 'Mixed_7c', 'Mixed_7d']
 130     self.assertItemsEqual(end_points.keys(), expected_endpoints)
 131     for name, op in end_points.items():
 132       self.assertTrue(op.name.startswith('InceptionV4/' + name))

 134   def testBuildOnlyUpToFinalEndpoint(self):
 135     batch_size = 5
 136     height, width = 299, 299
 137     all_endpoints = [
 138         'Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3', 'Mixed_3a',
 139         'Mixed_4a', 'Mixed_5a', 'Mixed_5b', 'Mixed_5c', 'Mixed_5d',
 140         'Mixed_5e', 'Mixed_6a', 'Mixed_6b', 'Mixed_6c', 'Mixed_6d',
 141         'Mixed_6e', 'Mixed_6f', 'Mixed_6g', 'Mixed_6h', 'Mixed_7a',
 142         'Mixed_7b', 'Mixed_7c', 'Mixed_7d']
 143     for index, endpoint in enumerate(all_endpoints):
 144       with tf.Graph().as_default():
 145         inputs = tf.random.uniform((batch_size, height, width, 3))
 146         out_tensor, end_points = inception.inception_v4_base(
 147             inputs, final_endpoint=endpoint)
 148         self.assertTrue(out_tensor.op.name.startswith(
 149             'InceptionV4/' + endpoint))
 150         self.assertItemsEqual(all_endpoints[:index + 1], end_points.keys())

 152   def testVariablesSetDevice(self):
 153     batch_size = 5
 154     height, width = 299, 299
 155     num_classes = 1000
 156     inputs = tf.random.uniform((batch_size, height, width, 3))
 157     # Force all Variables to reside on the device.
 158     with tf.variable_scope('on_cpu'), tf.device('/cpu:0'):
 159       inception.inception_v4(inputs, num_classes)
 160     with tf.variable_scope('on_gpu'), tf.device('/gpu:0'):
 161       inception.inception_v4(inputs, num_classes)
 162     for v in tf.get_collection(
 163         tf.GraphKeys.GLOBAL_VARIABLES, scope='on_cpu'):
 164       self.assertDeviceEqual(v.device, '/cpu:0')
 165     for v in tf.get_collection(
 166         tf.GraphKeys.GLOBAL_VARIABLES, scope='on_gpu'):
 167       self.assertDeviceEqual(v.device, '/gpu:0')

 195   def testGlobalPoolUnknownImageShape(self):
 196     batch_size = 1
 197     height, width = 350, 400
 198     num_classes = 1000
 199     with self.test_session() as sess:
 200       inputs = tf.placeholder(tf.float32, (batch_size, None, None, 3))
 201       logits, end_points = inception.inception_v4(
 202           inputs, num_classes, create_aux_logits=False)
 203       self.assertTrue(logits.op.name.startswith('InceptionV4/Logits'))
 204       self.assertListEqual(logits.get_shape().as_list(),
 205                            [batch_size, num_classes])
 206       pre_pool = end_points['Mixed_7d']
 207       images = tf.random.uniform((batch_size, height, width, 3))
 208       sess.run(tf.global_variables_initializer())
 209       logits_out, pre_pool_out = sess.run([logits, pre_pool],
 210                                           {inputs: images.eval()})
 211       self.assertTupleEqual(logits_out.shape, (batch_size, num_classes))
 212       self.assertTupleEqual(pre_pool_out.shape, (batch_size, 9, 11, 1536))

 214   def testUnknownBatchSize(self):
 215     batch_size = 1
 216     height, width = 299, 299
 217     num_classes = 1000
 218     with self.test_session() as sess:
 219       inputs = tf.placeholder(tf.float32, (None, height, width, 3))
 220       logits, _ = inception.inception_v4(inputs, num_classes)

 229   def testEvaluation(self):
 230     batch_size = 2
 231     height, width = 299, 299
 232     num_classes = 1000
 233     with self.test_session() as sess:
 234       eval_inputs = tf.random.uniform((batch_size, height, width, 3))
 235       logits, _ = inception.inception_v4(eval_inputs,
 236                                          num_classes,
 237                                          is_training=False)
 238       predictions = tf.argmax(input=logits, axis=1)
 239       sess.run(tf.global_variables_initializer())
 240       output = sess.run(predictions)
 241       self.assertEquals(output.shape, (batch_size,))

 270   def testBatchNormScale(self):
 271     height, width = 299, 299
 272     num_classes = 1000
 273     inputs = tf.placeholder(tf.float32, (1, height, width, 3))
 274     with slim.arg_scope(
 275         inception.inception_v4_arg_scope(batch_norm_scale=True)):
 276       inception.inception_v4(inputs, num_classes, is_training=False)
 277 
 278     gamma_names = set(
 279         v.op.name
 280         for v in tf.global_variables('.*/BatchNorm/gamma:0$'))
 281     self.assertGreater(len(gamma_names), 0)
 282     for v in tf.global_variables('.*/BatchNorm/moving_mean:0$'):
 283       self.assertIn(v.op.name[:-len('moving_mean')] + 'gamma', gamma_names)
