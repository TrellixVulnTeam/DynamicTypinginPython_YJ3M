  54 # Mean and standard deviation used for normalizing the image tensor.
  55 _MEAN_RGB = 127.5
  56 _STD_RGB = 127.5
  57

  59 def _preprocess_for_quantization(image_data, image_size, crop_padding=32):
  60   """Crops to center of image with padding then scales, normalizes image_size.
  61 
  62   Args:
  63     image_data: A 3D Tensor representing the RGB image data. Image can be of
  64       arbitrary height and width.
  65     image_size: image height/width dimension.
  66     crop_padding: the padding size to use when centering the crop.
  67 
  68   Returns:
  69     A decoded and cropped image Tensor. Image is normalized to [-1,1].
  70 
  71   """
  72 
  73   shape = tf.shape(image_data)
  74   image_height = shape[0]
  75   image_width = shape[1]
  76 
  77   padded_center_crop_size = tf.cast(
  78       (image_size * 1.0 / (image_size + crop_padding)) *
  79       tf.cast(tf.minimum(image_height, image_width), tf.float32), tf.int32)
  80 
  81   offset_height = ((image_height - padded_center_crop_size) + 1) // 2
  82   offset_width = ((image_width - padded_center_crop_size) + 1) // 2
  83 
  84   image = tf.image.crop_to_bounding_box(
  85       image_data,
  86       offset_height=offset_height,
  87       offset_width=offset_width,
  88       target_height=padded_center_crop_size,
  89       target_width=padded_center_crop_size)
  90 
  91   image = tf.image.resize([image], [image_size, image_size],
  92                           method=tf.image.ResizeMethod.BICUBIC)[0]
  93   image = tf.cast(image, tf.float32)
  94   image -= tf.constant(_MEAN_RGB)
  95   image /= tf.constant(_STD_RGB)
  96   return image

  99 def restore_model(sess, checkpoint_path, enable_ema=True):
 100   """Restore variables from the checkpoint into the provided session.
 101 
 102   Args:
 103     sess: A tensorflow session where the checkpoint will be loaded.
 104     checkpoint_path: Path to the trained checkpoint.
 105     enable_ema: (optional) Whether to load the exponential moving average (ema)
 106       version of the tensorflow variables. Defaults to True.
 107   """
 108   if enable_ema:
 109     ema = tf.train.ExponentialMovingAverage(decay=0.0)
 110     ema_vars = tf.trainable_variables() + tf.get_collection("moving_vars")
 111     for v in tf.global_variables():
 112       if "moving_mean" in v.name or "moving_variance" in v.name:
 113         ema_vars.append(v)
 114     ema_vars = list(set(ema_vars))
 115     var_dict = ema.variables_to_restore(ema_vars)
 116   else:
 117     var_dict = None
 118 
 119   sess.run(tf.global_variables_initializer())
 120   saver = tf.train.Saver(var_dict, max_to_keep=1)
 121   saver.restore(sess, checkpoint_path)

 124 def _representative_dataset_gen():
 125   """Gets a python generator of numpy arrays for the given dataset."""
 126   image_size = FLAGS.image_size
 127   dataset = tfds.builder(FLAGS.dataset_name, data_dir=FLAGS.dataset_dir)
 128   dataset.download_and_prepare()
 129   data = dataset.as_dataset()[FLAGS.dataset_split]
 130   iterator = tf.data.make_one_shot_iterator(data)
 131   if FLAGS.use_model_specific_preprocessing:
 132     preprocess_fn = functools.partial(
 133         preprocessing_factory.get_preprocessing(name=FLAGS.model_name),
 134         output_height=image_size,
 135         output_width=image_size)
 136   else:
 137     preprocess_fn = functools.partial(
 138         _preprocess_for_quantization, image_size=image_size)
 139   features = iterator.get_next()
 140   image = features["image"]
 141   image = preprocess_fn(image)
 142   image = tf.reshape(image, [1, image_size, image_size, 3])
 143   for _ in range(FLAGS.num_steps):
 144     yield [image.eval()]

 147 def main(_):
 148   with tf.Graph().as_default(), tf.Session() as sess:
 149     network_fn = nets_factory.get_network_fn(
 150         FLAGS.model_name, num_classes=FLAGS.num_classes, is_training=False)
 151     image_size = FLAGS.image_size
 152     images = tf.placeholder(
 153         tf.float32, shape=(1, image_size, image_size, 3), name="images")
 154 
 155     logits, _ = network_fn(images)
 156 
 157     output_tensor = tf.nn.softmax(logits)
 158     restore_model(sess, FLAGS.checkpoint_path, enable_ema=FLAGS.enable_ema)
 159 
 160     converter = tf.lite.TFLiteConverter.from_session(sess, [images],
 161                                                      [output_tensor])
 162 
 163     converter.representative_dataset = tf.lite.RepresentativeDataset(
 164         _representative_dataset_gen)
 165     converter.optimizations = [tf.lite.Optimize.DEFAULT]
 166     converter.inference_input_type = tf.int8
 167     converter.inference_output_type = tf.int8
 168     converter.target_spec.supported_ops = [tf.lite.OpsSet.TFLITE_BUILTINS_INT8]
 169 
 170     tflite_buffer = converter.convert()
 171     with tf.gfile.GFile(FLAGS.output_tflite, "wb") as output_tflite:
 172       output_tflite.write(tflite_buffer)
 173   print("tflite model written to %s" % FLAGS.output_tflite)
