  31 def inception_v1_base(inputs,
  32                       final_endpoint='Mixed_5c',
  33                       include_root_block=True,
  34                       scope='InceptionV1'):
  35   """Defines the Inception V1 base architecture.
  36 
  37   This architecture is defined in:
  38     Going deeper with convolutions
  39     Christian Szegedy, Wei Liu, Yangqing Jia, Pierre Sermanet, Scott Reed,
  40     Dragomir Anguelov, Dumitru Erhan, Vincent Vanhoucke, Andrew Rabinovich.
  41     http://arxiv.org/pdf/1409.4842v1.pdf.
  42 
  43   Args:
  44     inputs: a tensor of size [batch_size, height, width, channels].
  45     final_endpoint: specifies the endpoint to construct the network up to. It
  46       can be one of ['Conv2d_1a_7x7', 'MaxPool_2a_3x3', 'Conv2d_2b_1x1',
  47       'Conv2d_2c_3x3', 'MaxPool_3a_3x3', 'Mixed_3b', 'Mixed_3c',
  48       'MaxPool_4a_3x3', 'Mixed_4b', 'Mixed_4c', 'Mixed_4d', 'Mixed_4e',
  49       'Mixed_4f', 'MaxPool_5a_2x2', 'Mixed_5b', 'Mixed_5c']. If
  50       include_root_block is False, ['Conv2d_1a_7x7', 'MaxPool_2a_3x3',
  51       'Conv2d_2b_1x1', 'Conv2d_2c_3x3', 'MaxPool_3a_3x3'] will not be available.
  52     include_root_block: If True, include the convolution and max-pooling layers
  53       before the inception modules. If False, excludes those layers.
  54     scope: Optional variable_scope.
  55 
  56   Returns:
  57     A dictionary from components of the network to the corresponding activation.
  58 
  59   Raises:
  60     ValueError: if final_endpoint is not set to one of the predefined values.
  61   """
  62   end_points = {}
  63   with tf.variable_scope(scope, 'InceptionV1', [inputs]):
  64     with slim.arg_scope(
  65         [slim.conv2d, slim.fully_connected],
  66         weights_initializer=trunc_normal(0.01)):
  67       with slim.arg_scope([slim.conv2d, slim.max_pool2d],
  68                           stride=1, padding='SAME'):
  69         net = inputs
  70         if include_root_block:
  71           end_point = 'Conv2d_1a_7x7'
  72           net = slim.conv2d(inputs, 64, [7, 7], stride=2, scope=end_point)
  73           end_points[end_point] = net
  74           if final_endpoint == end_point:
  75             return net, end_points
  76           end_point = 'MaxPool_2a_3x3'
  77           net = slim.max_pool2d(net, [3, 3], stride=2, scope=end_point)
  78           end_points[end_point] = net
  79           if final_endpoint == end_point:
  80             return net, end_points
  81           end_point = 'Conv2d_2b_1x1'
  82           net = slim.conv2d(net, 64, [1, 1], scope=end_point)
  83           end_points[end_point] = net
  84           if final_endpoint == end_point:
  85             return net, end_points
  86           end_point = 'Conv2d_2c_3x3'
  87           net = slim.conv2d(net, 192, [3, 3], scope=end_point)
  88           end_points[end_point] = net
  89           if final_endpoint == end_point:
  90             return net, end_points
  91           end_point = 'MaxPool_3a_3x3'
  92           net = slim.max_pool2d(net, [3, 3], stride=2, scope=end_point)
  93           end_points[end_point] = net
  94           if final_endpoint == end_point:
  95             return net, end_points
  96 
  97         end_point = 'Mixed_3b'
  98         with tf.variable_scope(end_point):
  99           with tf.variable_scope('Branch_0'):
 100             branch_0 = slim.conv2d(net, 64, [1, 1], scope='Conv2d_0a_1x1')
 101           with tf.variable_scope('Branch_1'):
 102             branch_1 = slim.conv2d(net, 96, [1, 1], scope='Conv2d_0a_1x1')
 103             branch_1 = slim.conv2d(branch_1, 128, [3, 3], scope='Conv2d_0b_3x3')
 104           with tf.variable_scope('Branch_2'):
 105             branch_2 = slim.conv2d(net, 16, [1, 1], scope='Conv2d_0a_1x1')
 106             branch_2 = slim.conv2d(branch_2, 32, [3, 3], scope='Conv2d_0b_3x3')
 107           with tf.variable_scope('Branch_3'):
 108             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 109             branch_3 = slim.conv2d(branch_3, 32, [1, 1], scope='Conv2d_0b_1x1')
 110           net = tf.concat(
 111               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 112         end_points[end_point] = net
 113         if final_endpoint == end_point: return net, end_points
 114 
 115         end_point = 'Mixed_3c'
 116         with tf.variable_scope(end_point):
 117           with tf.variable_scope('Branch_0'):
 118             branch_0 = slim.conv2d(net, 128, [1, 1], scope='Conv2d_0a_1x1')
 119           with tf.variable_scope('Branch_1'):
 120             branch_1 = slim.conv2d(net, 128, [1, 1], scope='Conv2d_0a_1x1')
 121             branch_1 = slim.conv2d(branch_1, 192, [3, 3], scope='Conv2d_0b_3x3')
 122           with tf.variable_scope('Branch_2'):
 123             branch_2 = slim.conv2d(net, 32, [1, 1], scope='Conv2d_0a_1x1')
 124             branch_2 = slim.conv2d(branch_2, 96, [3, 3], scope='Conv2d_0b_3x3')
 125           with tf.variable_scope('Branch_3'):
 126             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 127             branch_3 = slim.conv2d(branch_3, 64, [1, 1], scope='Conv2d_0b_1x1')
 128           net = tf.concat(
 129               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 130         end_points[end_point] = net
 131         if final_endpoint == end_point: return net, end_points
 132 
 133         end_point = 'MaxPool_4a_3x3'
 134         net = slim.max_pool2d(net, [3, 3], stride=2, scope=end_point)
 135         end_points[end_point] = net
 136         if final_endpoint == end_point: return net, end_points
 137 
 138         end_point = 'Mixed_4b'
 139         with tf.variable_scope(end_point):
 140           with tf.variable_scope('Branch_0'):
 141             branch_0 = slim.conv2d(net, 192, [1, 1], scope='Conv2d_0a_1x1')
 142           with tf.variable_scope('Branch_1'):
 143             branch_1 = slim.conv2d(net, 96, [1, 1], scope='Conv2d_0a_1x1')
 144             branch_1 = slim.conv2d(branch_1, 208, [3, 3], scope='Conv2d_0b_3x3')
 145           with tf.variable_scope('Branch_2'):
 146             branch_2 = slim.conv2d(net, 16, [1, 1], scope='Conv2d_0a_1x1')
 147             branch_2 = slim.conv2d(branch_2, 48, [3, 3], scope='Conv2d_0b_3x3')
 148           with tf.variable_scope('Branch_3'):
 149             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 150             branch_3 = slim.conv2d(branch_3, 64, [1, 1], scope='Conv2d_0b_1x1')
 151           net = tf.concat(
 152               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 153         end_points[end_point] = net
 154         if final_endpoint == end_point: return net, end_points
 155 
 156         end_point = 'Mixed_4c'
 157         with tf.variable_scope(end_point):
 158           with tf.variable_scope('Branch_0'):
 159             branch_0 = slim.conv2d(net, 160, [1, 1], scope='Conv2d_0a_1x1')
 160           with tf.variable_scope('Branch_1'):
 161             branch_1 = slim.conv2d(net, 112, [1, 1], scope='Conv2d_0a_1x1')
 162             branch_1 = slim.conv2d(branch_1, 224, [3, 3], scope='Conv2d_0b_3x3')
 163           with tf.variable_scope('Branch_2'):
 164             branch_2 = slim.conv2d(net, 24, [1, 1], scope='Conv2d_0a_1x1')
 165             branch_2 = slim.conv2d(branch_2, 64, [3, 3], scope='Conv2d_0b_3x3')
 166           with tf.variable_scope('Branch_3'):
 167             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 168             branch_3 = slim.conv2d(branch_3, 64, [1, 1], scope='Conv2d_0b_1x1')
 169           net = tf.concat(
 170               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 171         end_points[end_point] = net
 172         if final_endpoint == end_point: return net, end_points
 173 
 174         end_point = 'Mixed_4d'
 175         with tf.variable_scope(end_point):
 176           with tf.variable_scope('Branch_0'):
 177             branch_0 = slim.conv2d(net, 128, [1, 1], scope='Conv2d_0a_1x1')
 178           with tf.variable_scope('Branch_1'):
 179             branch_1 = slim.conv2d(net, 128, [1, 1], scope='Conv2d_0a_1x1')
 180             branch_1 = slim.conv2d(branch_1, 256, [3, 3], scope='Conv2d_0b_3x3')
 181           with tf.variable_scope('Branch_2'):
 182             branch_2 = slim.conv2d(net, 24, [1, 1], scope='Conv2d_0a_1x1')
 183             branch_2 = slim.conv2d(branch_2, 64, [3, 3], scope='Conv2d_0b_3x3')
 184           with tf.variable_scope('Branch_3'):
 185             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 186             branch_3 = slim.conv2d(branch_3, 64, [1, 1], scope='Conv2d_0b_1x1')
 187           net = tf.concat(
 188               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 189         end_points[end_point] = net
 190         if final_endpoint == end_point: return net, end_points
 191 
 192         end_point = 'Mixed_4e'
 193         with tf.variable_scope(end_point):
 194           with tf.variable_scope('Branch_0'):
 195             branch_0 = slim.conv2d(net, 112, [1, 1], scope='Conv2d_0a_1x1')
 196           with tf.variable_scope('Branch_1'):
 197             branch_1 = slim.conv2d(net, 144, [1, 1], scope='Conv2d_0a_1x1')
 198             branch_1 = slim.conv2d(branch_1, 288, [3, 3], scope='Conv2d_0b_3x3')
 199           with tf.variable_scope('Branch_2'):
 200             branch_2 = slim.conv2d(net, 32, [1, 1], scope='Conv2d_0a_1x1')
 201             branch_2 = slim.conv2d(branch_2, 64, [3, 3], scope='Conv2d_0b_3x3')
 202           with tf.variable_scope('Branch_3'):
 203             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 204             branch_3 = slim.conv2d(branch_3, 64, [1, 1], scope='Conv2d_0b_1x1')
 205           net = tf.concat(
 206               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 207         end_points[end_point] = net
 208         if final_endpoint == end_point: return net, end_points
 209 
 210         end_point = 'Mixed_4f'
 211         with tf.variable_scope(end_point):
 212           with tf.variable_scope('Branch_0'):
 213             branch_0 = slim.conv2d(net, 256, [1, 1], scope='Conv2d_0a_1x1')
 214           with tf.variable_scope('Branch_1'):
 215             branch_1 = slim.conv2d(net, 160, [1, 1], scope='Conv2d_0a_1x1')
 216             branch_1 = slim.conv2d(branch_1, 320, [3, 3], scope='Conv2d_0b_3x3')
 217           with tf.variable_scope('Branch_2'):
 218             branch_2 = slim.conv2d(net, 32, [1, 1], scope='Conv2d_0a_1x1')
 219             branch_2 = slim.conv2d(branch_2, 128, [3, 3], scope='Conv2d_0b_3x3')
 220           with tf.variable_scope('Branch_3'):
 221             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 222             branch_3 = slim.conv2d(branch_3, 128, [1, 1], scope='Conv2d_0b_1x1')
 223           net = tf.concat(
 224               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 225         end_points[end_point] = net
 226         if final_endpoint == end_point: return net, end_points
 227 
 228         end_point = 'MaxPool_5a_2x2'
 229         net = slim.max_pool2d(net, [2, 2], stride=2, scope=end_point)
 230         end_points[end_point] = net
 231         if final_endpoint == end_point: return net, end_points
 232 
 233         end_point = 'Mixed_5b'
 234         with tf.variable_scope(end_point):
 235           with tf.variable_scope('Branch_0'):
 236             branch_0 = slim.conv2d(net, 256, [1, 1], scope='Conv2d_0a_1x1')
 237           with tf.variable_scope('Branch_1'):
 238             branch_1 = slim.conv2d(net, 160, [1, 1], scope='Conv2d_0a_1x1')
 239             branch_1 = slim.conv2d(branch_1, 320, [3, 3], scope='Conv2d_0b_3x3')
 240           with tf.variable_scope('Branch_2'):
 241             branch_2 = slim.conv2d(net, 32, [1, 1], scope='Conv2d_0a_1x1')
 242             branch_2 = slim.conv2d(branch_2, 128, [3, 3], scope='Conv2d_0a_3x3')
 243           with tf.variable_scope('Branch_3'):
 244             branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 245             branch_3 = slim.conv2d(branch_3, 128, [1, 1], scope='Conv2d_0b_1x1')
 246           net = tf.concat(
 247               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 248         end_points[end_point] = net
 249         if final_endpoint == end_point: return net, end_points
 250 
 251         end_point = 'Mixed_5c'
 252         with tf.variable_scope(end_point):
 253           with tf.variable_scope('Branch_0'):
 254             branch_0 = slim.conv2d(net, 384, [1, 1], scope='Conv2d_0a_1x1')
 255           with tf.variable_scope('Branch_1'):
 256             branch_1 = slim.conv2d(net, 192, [1, 1], scope='Conv2d_0a_1x1')
 257             branch_1 = slim.conv2d(branch_1, 384, [3, 3], scope='Conv2d_0b_3x3')
 258           with tf.variable_scope('Branch_2'):
 259             branch_2 = slim.conv2d(net, 48, [1, 1], scope='Conv2d_0a_1x1')
 260             branch_2 = slim.conv2d(branch_2, 128, [3, 3], scope='Conv2d_0b_3x3')

 265               axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 266         end_points[end_point] = net
 267         if final_endpoint == end_point: return net, end_points
 268     raise ValueError('Unknown final endpoint %s' % final_endpoint)

 271 def inception_v1(inputs,
 272                  num_classes=1000,
 273                  is_training=True,
 274                  dropout_keep_prob=0.8,
 275                  prediction_fn=slim.softmax,
 276                  spatial_squeeze=True,
 277                  reuse=None,
 278                  scope='InceptionV1',
 279                  global_pool=False):
 280   """Defines the Inception V1 architecture.
 281 
 282   This architecture is defined in:
 283 
 284     Going deeper with convolutions
 285     Christian Szegedy, Wei Liu, Yangqing Jia, Pierre Sermanet, Scott Reed,
 286     Dragomir Anguelov, Dumitru Erhan, Vincent Vanhoucke, Andrew Rabinovich.
 287     http://arxiv.org/pdf/1409.4842v1.pdf.
 288 
 289   The default image size used to train this network is 224x224.
 290 
 291   Args:
 292     inputs: a tensor of size [batch_size, height, width, channels].
 293     num_classes: number of predicted classes. If 0 or None, the logits layer
 294       is omitted and the input features to the logits layer (before dropout)
 295       are returned instead.
 296     is_training: whether is training or not.
 297     dropout_keep_prob: the percentage of activation values that are retained.
 298     prediction_fn: a function to get predictions out of logits.
 299     spatial_squeeze: if True, logits is of shape [B, C], if false logits is of
 300         shape [B, 1, 1, C], where B is batch_size and C is number of classes.
 301     reuse: whether or not the network and its variables should be reused. To be
 302       able to reuse 'scope' must be given.
 303     scope: Optional variable_scope.
 304     global_pool: Optional boolean flag to control the avgpooling before the
 305       logits layer. If false or unset, pooling is done with a fixed window
 306       that reduces default-sized inputs to 1x1, while larger inputs lead to
 307       larger outputs. If true, any input size is pooled down to 1x1.
 308 
 309   Returns:
 310     net: a Tensor with the logits (pre-softmax activations) if num_classes
 311       is a non-zero integer, or the non-dropped-out input to the logits layer
 312       if num_classes is 0 or None.
 313     end_points: a dictionary from components of the network to the corresponding
 314       activation.
 315   """
 316   # Final pooling and prediction
 317   with tf.variable_scope(
 318       scope, 'InceptionV1', [inputs], reuse=reuse) as scope:
 319     with slim.arg_scope([slim.batch_norm, slim.dropout],
 320                         is_training=is_training):
 321       net, end_points = inception_v1_base(inputs, scope=scope)
 322       with tf.variable_scope('Logits'):
 323         if global_pool:
 324           # Global average pooling.
 325           net = tf.reduce_mean(
 326               input_tensor=net, axis=[1, 2], keepdims=True, name='global_pool')
 327           end_points['global_pool'] = net
 328         else:
 329           # Pooling with a fixed kernel size.
 330           net = slim.avg_pool2d(net, [7, 7], stride=1, scope='AvgPool_0a_7x7')
 331           end_points['AvgPool_0a_7x7'] = net
 332         if not num_classes:
 333           return net, end_points
 334         net = slim.dropout(net, dropout_keep_prob, scope='Dropout_0b')
 335         logits = slim.conv2d(net, num_classes, [1, 1], activation_fn=None,
 336                              normalizer_fn=None, scope='Conv2d_0c_1x1')
 337         if spatial_squeeze:
 338           logits = tf.squeeze(logits, [1, 2], name='SpatialSqueeze')
 339 
 340         end_points['Logits'] = logits
 341         end_points['Predictions'] = prediction_fn(logits, scope='Predictions')
 342   return logits, end_points
 343 inception_v1.default_image_size = 224
