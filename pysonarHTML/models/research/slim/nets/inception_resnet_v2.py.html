  32 def block35(net, scale=1.0, activation_fn=tf.nn.relu, scope=None, reuse=None):
  33   """Builds the 35x35 resnet block."""
  34   with tf.variable_scope(scope, 'Block35', [net], reuse=reuse):
  35     with tf.variable_scope('Branch_0'):
  36       tower_conv = slim.conv2d(net, 32, 1, scope='Conv2d_1x1')
  37     with tf.variable_scope('Branch_1'):
  38       tower_conv1_0 = slim.conv2d(net, 32, 1, scope='Conv2d_0a_1x1')
  39       tower_conv1_1 = slim.conv2d(tower_conv1_0, 32, 3, scope='Conv2d_0b_3x3')
  40     with tf.variable_scope('Branch_2'):
  41       tower_conv2_0 = slim.conv2d(net, 32, 1, scope='Conv2d_0a_1x1')
  42       tower_conv2_1 = slim.conv2d(tower_conv2_0, 48, 3, scope='Conv2d_0b_3x3')
  43       tower_conv2_2 = slim.conv2d(tower_conv2_1, 64, 3, scope='Conv2d_0c_3x3')
  44     mixed = tf.concat(axis=3, values=[tower_conv, tower_conv1_1, tower_conv2_2])
  45     up = slim.conv2d(mixed, net.get_shape()[3], 1, normalizer_fn=None,
  46                      activation_fn=None, scope='Conv2d_1x1')
  47     scaled_up = up * scale
  48     if activation_fn == tf.nn.relu6:
  49       # Use clip_by_value to simulate bandpass activation.
  50       scaled_up = tf.clip_by_value(scaled_up, -6.0, 6.0)
  51 
  52     net += scaled_up
  53     if activation_fn:
  54       net = activation_fn(net)
  55   return net

  58 def block17(net, scale=1.0, activation_fn=tf.nn.relu, scope=None, reuse=None):

  60   with tf.variable_scope(scope, 'Block17', [net], reuse=reuse):
  61     with tf.variable_scope('Branch_0'):
  62       tower_conv = slim.conv2d(net, 192, 1, scope='Conv2d_1x1')
  63     with tf.variable_scope('Branch_1'):
  64       tower_conv1_0 = slim.conv2d(net, 128, 1, scope='Conv2d_0a_1x1')
  65       tower_conv1_1 = slim.conv2d(tower_conv1_0, 160, [1, 7],
  66                                   scope='Conv2d_0b_1x7')
  67       tower_conv1_2 = slim.conv2d(tower_conv1_1, 192, [7, 1],
  68                                   scope='Conv2d_0c_7x1')

  76       scaled_up = tf.clip_by_value(scaled_up, -6.0, 6.0)
  77

 110 def inception_resnet_v2_base(inputs,
 111                              final_endpoint='Conv2d_7b_1x1',
 112                              output_stride=16,
 113                              align_feature_maps=False,
 114                              scope=None,
 115                              activation_fn=tf.nn.relu):
 116   """Inception model from  http://arxiv.org/abs/1602.07261.
 117 
 118   Constructs an Inception Resnet v2 network from inputs to the given final
 119   endpoint. This method can construct the network up to the final inception
 120   block Conv2d_7b_1x1.
 121 
 122   Args:
 123     inputs: a tensor of size [batch_size, height, width, channels].
 124     final_endpoint: specifies the endpoint to construct the network up to. It
 125       can be one of ['Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3',
 126       'MaxPool_3a_3x3', 'Conv2d_3b_1x1', 'Conv2d_4a_3x3', 'MaxPool_5a_3x3',
 127       'Mixed_5b', 'Mixed_6a', 'PreAuxLogits', 'Mixed_7a', 'Conv2d_7b_1x1']
 128     output_stride: A scalar that specifies the requested ratio of input to
 129       output spatial resolution. Only supports 8 and 16.
 130     align_feature_maps: When true, changes all the VALID paddings in the network
 131       to SAME padding so that the feature maps are aligned.
 132     scope: Optional variable_scope.
 133     activation_fn: Activation function for block scopes.
 134 
 135   Returns:
 136     tensor_out: output tensor corresponding to the final_endpoint.
 137     end_points: a set of activations for external use, for example summaries or
 138                 losses.
 139 
 140   Raises:
 141     ValueError: if final_endpoint is not set to one of the predefined values,
 142       or if the output_stride is not 8 or 16, or if the output_stride is 8 and
 143       we request an end point after 'PreAuxLogits'.
 144   """
 145   if output_stride != 8 and output_stride != 16:
 146     raise ValueError('output_stride must be 8 or 16.')
 147 
 148   padding = 'SAME' if align_feature_maps else 'VALID'
 149 
 150   end_points = {}
 151 
 152   def add_and_check_final(name, net):
 153     end_points[name] = net
 154     return name == final_endpoint
 155 
 156   with tf.variable_scope(scope, 'InceptionResnetV2', [inputs]):
 157     with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 158                         stride=1, padding='SAME'):
 159       # 149 x 149 x 32
 160       net = slim.conv2d(inputs, 32, 3, stride=2, padding=padding,
 161                         scope='Conv2d_1a_3x3')
 162       if add_and_check_final('Conv2d_1a_3x3', net): return net, end_points
 163 
 164       # 147 x 147 x 32
 165       net = slim.conv2d(net, 32, 3, padding=padding,
 166                         scope='Conv2d_2a_3x3')
 167       if add_and_check_final('Conv2d_2a_3x3', net): return net, end_points
 168       # 147 x 147 x 64
 169       net = slim.conv2d(net, 64, 3, scope='Conv2d_2b_3x3')
 170       if add_and_check_final('Conv2d_2b_3x3', net): return net, end_points
 171       # 73 x 73 x 64
 172       net = slim.max_pool2d(net, 3, stride=2, padding=padding,
 173                             scope='MaxPool_3a_3x3')
 174       if add_and_check_final('MaxPool_3a_3x3', net): return net, end_points
 175       # 73 x 73 x 80
 176       net = slim.conv2d(net, 80, 1, padding=padding,
 177                         scope='Conv2d_3b_1x1')
 178       if add_and_check_final('Conv2d_3b_1x1', net): return net, end_points
 179       # 71 x 71 x 192
 180       net = slim.conv2d(net, 192, 3, padding=padding,
 181                         scope='Conv2d_4a_3x3')
 182       if add_and_check_final('Conv2d_4a_3x3', net): return net, end_points
 183       # 35 x 35 x 192
 184       net = slim.max_pool2d(net, 3, stride=2, padding=padding,
 185                             scope='MaxPool_5a_3x3')
 186       if add_and_check_final('MaxPool_5a_3x3', net): return net, end_points
 187 
 188       # 35 x 35 x 320
 189       with tf.variable_scope('Mixed_5b'):
 190         with tf.variable_scope('Branch_0'):
 191           tower_conv = slim.conv2d(net, 96, 1, scope='Conv2d_1x1')
 192         with tf.variable_scope('Branch_1'):
 193           tower_conv1_0 = slim.conv2d(net, 48, 1, scope='Conv2d_0a_1x1')
 194           tower_conv1_1 = slim.conv2d(tower_conv1_0, 64, 5,
 195                                       scope='Conv2d_0b_5x5')
 196         with tf.variable_scope('Branch_2'):
 197           tower_conv2_0 = slim.conv2d(net, 64, 1, scope='Conv2d_0a_1x1')
 198           tower_conv2_1 = slim.conv2d(tower_conv2_0, 96, 3,
 199                                       scope='Conv2d_0b_3x3')
 200           tower_conv2_2 = slim.conv2d(tower_conv2_1, 96, 3,
 201                                       scope='Conv2d_0c_3x3')
 202         with tf.variable_scope('Branch_3'):
 203           tower_pool = slim.avg_pool2d(net, 3, stride=1, padding='SAME',
 204                                        scope='AvgPool_0a_3x3')
 205           tower_pool_1 = slim.conv2d(tower_pool, 64, 1,
 206                                      scope='Conv2d_0b_1x1')
 207         net = tf.concat(
 208             [tower_conv, tower_conv1_1, tower_conv2_2, tower_pool_1], 3)
 209 
 210       if add_and_check_final('Mixed_5b', net): return net, end_points
 211       # TODO(alemi): Register intermediate endpoints
 212       net = slim.repeat(net, 10, block35, scale=0.17,
 213                         activation_fn=activation_fn)
 214 
 215       # 17 x 17 x 1088 if output_stride == 8,
 216       # 33 x 33 x 1088 if output_stride == 16
 217       use_atrous = output_stride == 8
 218 
 219       with tf.variable_scope('Mixed_6a'):
 220         with tf.variable_scope('Branch_0'):
 221           tower_conv = slim.conv2d(net, 384, 3, stride=1 if use_atrous else 2,
 222                                    padding=padding,
 223                                    scope='Conv2d_1a_3x3')
 224         with tf.variable_scope('Branch_1'):
 225           tower_conv1_0 = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 226           tower_conv1_1 = slim.conv2d(tower_conv1_0, 256, 3,
 227                                       scope='Conv2d_0b_3x3')
 228           tower_conv1_2 = slim.conv2d(tower_conv1_1, 384, 3,
 229                                       stride=1 if use_atrous else 2,
 230                                       padding=padding,
 231                                       scope='Conv2d_1a_3x3')
 232         with tf.variable_scope('Branch_2'):
 233           tower_pool = slim.max_pool2d(net, 3, stride=1 if use_atrous else 2,
 234                                        padding=padding,
 235                                        scope='MaxPool_1a_3x3')
 236         net = tf.concat([tower_conv, tower_conv1_2, tower_pool], 3)
 237 
 238       if add_and_check_final('Mixed_6a', net): return net, end_points
 239 
 240       # TODO(alemi): register intermediate endpoints
 241       with slim.arg_scope([slim.conv2d], rate=2 if use_atrous else 1):
 242         net = slim.repeat(net, 20, block17, scale=0.10,
 243                           activation_fn=activation_fn)
 244       if add_and_check_final('PreAuxLogits', net): return net, end_points
 245 
 246       if output_stride == 8:
 247         # TODO(gpapan): Properly support output_stride for the rest of the net.
 248         raise ValueError('output_stride==8 is only supported up to the '
 249                          'PreAuxlogits end_point for now.')
 250 
 251       # 8 x 8 x 2080
 252       with tf.variable_scope('Mixed_7a'):
 253         with tf.variable_scope('Branch_0'):
 254           tower_conv = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 255           tower_conv_1 = slim.conv2d(tower_conv, 384, 3, stride=2,
 256                                      padding=padding,
 257                                      scope='Conv2d_1a_3x3')
 258         with tf.variable_scope('Branch_1'):
 259           tower_conv1 = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 260           tower_conv1_1 = slim.conv2d(tower_conv1, 288, 3, stride=2,
 261                                       padding=padding,
 262                                       scope='Conv2d_1a_3x3')
 263         with tf.variable_scope('Branch_2'):
 264           tower_conv2 = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 265           tower_conv2_1 = slim.conv2d(tower_conv2, 288, 3,
 266                                       scope='Conv2d_0b_3x3')
 267           tower_conv2_2 = slim.conv2d(tower_conv2_1, 320, 3, stride=2,
 268                                       padding=padding,
 269                                       scope='Conv2d_1a_3x3')
 270         with tf.variable_scope('Branch_3'):
 271           tower_pool = slim.max_pool2d(net, 3, stride=2,
 272                                        padding=padding,
 273                                        scope='MaxPool_1a_3x3')
 274         net = tf.concat(
 275             [tower_conv_1, tower_conv1_1, tower_conv2_2, tower_pool], 3)
 276 
 277       if add_and_check_final('Mixed_7a', net): return net, end_points
 278 
 279       # TODO(alemi): register intermediate endpoints
 280       net = slim.repeat(net, 9, block8, scale=0.20, activation_fn=activation_fn)
 281       net = block8(net, activation_fn=None)
 282 
 283       # 8 x 8 x 1536
 284       net = slim.conv2d(net, 1536, 1, scope='Conv2d_7b_1x1')
 285       if add_and_check_final('Conv2d_7b_1x1', net): return net, end_points
 286 
 287     raise ValueError('final_endpoint (%s) not recognized', final_endpoint)

 290 def inception_resnet_v2(inputs, num_classes=1001, is_training=True,
 291                         dropout_keep_prob=0.8,
 292                         reuse=None,
 293                         scope='InceptionResnetV2',
 294                         create_aux_logits=True,
 295                         activation_fn=tf.nn.relu):
 296   """Creates the Inception Resnet V2 model.
 297 
 298   Args:
 299     inputs: a 4-D tensor of size [batch_size, height, width, 3].
 300       Dimension batch_size may be undefined. If create_aux_logits is false,
 301       also height and width may be undefined.
 302     num_classes: number of predicted classes. If 0 or None, the logits layer
 303       is omitted and the input features to the logits layer (before  dropout)
 304       are returned instead.
 305     is_training: whether is training or not.
 306     dropout_keep_prob: float, the fraction to keep before final layer.
 307     reuse: whether or not the network and its variables should be reused. To be
 308       able to reuse 'scope' must be given.
 309     scope: Optional variable_scope.
 310     create_aux_logits: Whether to include the auxilliary logits.
 311     activation_fn: Activation function for conv2d.
 312 
 313   Returns:
 314     net: the output of the logits layer (if num_classes is a non-zero integer),
 315       or the non-dropped-out input to the logits layer (if num_classes is 0 or
 316       None).
 317     end_points: the set of end_points from the inception model.
 318   """
 319   end_points = {}
 320 
 321   with tf.variable_scope(
 322       scope, 'InceptionResnetV2', [inputs], reuse=reuse) as scope:
 323     with slim.arg_scope([slim.batch_norm, slim.dropout],
 324                         is_training=is_training):
 325 
 326       net, end_points = inception_resnet_v2_base(inputs, scope=scope,
 327                                                  activation_fn=activation_fn)
 328 
 329       if create_aux_logits and num_classes:
 330         with tf.variable_scope('AuxLogits'):
 331           aux = end_points['PreAuxLogits']
 332           aux = slim.avg_pool2d(aux, 5, stride=3, padding='VALID',
 333                                 scope='Conv2d_1a_3x3')
 334           aux = slim.conv2d(aux, 128, 1, scope='Conv2d_1b_1x1')
 335           aux = slim.conv2d(aux, 768, aux.get_shape()[1:3],
 336                             padding='VALID', scope='Conv2d_2a_5x5')
 337           aux = slim.flatten(aux)
 338           aux = slim.fully_connected(aux, num_classes, activation_fn=None,
 339                                      scope='Logits')
 340           end_points['AuxLogits'] = aux
 341 
 342       with tf.variable_scope('Logits'):
 343         # TODO(sguada,arnoegw): Consider adding a parameter global_pool which
 344         # can be set to False to disable pooling here (as in resnet_*()).
 345         kernel_size = net.get_shape()[1:3]
 346         if kernel_size.is_fully_defined():
 347           net = slim.avg_pool2d(net, kernel_size, padding='VALID',
 348                                 scope='AvgPool_1a_8x8')
 349         else:
 350           net = tf.reduce_mean(
 351               input_tensor=net, axis=[1, 2], keepdims=True, name='global_pool')
 352         end_points['global_pool'] = net
 353         if not num_classes:
 354           return net, end_points
 355         net = slim.flatten(net)
 356         net = slim.dropout(net, dropout_keep_prob, is_training=is_training,
 357                            scope='Dropout')
 358         end_points['PreLogitsFlatten'] = net
 359         logits = slim.fully_connected(net, num_classes, activation_fn=None,
 360                                       scope='Logits')
 361         end_points['Logits'] = logits
 362         end_points['Predictions'] = tf.nn.softmax(logits, name='Predictions')
 363 
 364     return logits, end_points
 365 inception_resnet_v2.default_image_size = 299

 368 def inception_resnet_v2_arg_scope(
 369     weight_decay=0.00004,
 370     batch_norm_decay=0.9997,
 371     batch_norm_epsilon=0.001,
 372     activation_fn=tf.nn.relu,
 373     batch_norm_updates_collections=tf.GraphKeys.UPDATE_OPS,
 374     batch_norm_scale=False):
 375   """Returns the scope with the default parameters for inception_resnet_v2.
 376 
 377   Args:
 378     weight_decay: the weight decay for weights variables.
 379     batch_norm_decay: decay for the moving average of batch_norm momentums.
 380     batch_norm_epsilon: small float added to variance to avoid dividing by zero.
 381     activation_fn: Activation function for conv2d.
 382     batch_norm_updates_collections: Collection for the update ops for
 383       batch norm.
 384     batch_norm_scale: If True, uses an explicit `gamma` multiplier to scale the
 385       activations in the batch normalization layer.
 386 
 387   Returns:
 388     a arg_scope with the parameters needed for inception_resnet_v2.
 389   """
 390   # Set weight_decay for weights in conv2d and fully_connected layers.
 391   with slim.arg_scope([slim.conv2d, slim.fully_connected],
 392                       weights_regularizer=slim.l2_regularizer(weight_decay),
 393                       biases_regularizer=slim.l2_regularizer(weight_decay)):
 394 
 395     batch_norm_params = {
 396         'decay': batch_norm_decay,
 397         'epsilon': batch_norm_epsilon,
 398         'updates_collections': batch_norm_updates_collections,
 399         'fused': None,  # Use fused batch norm if possible.
 400         'scale': batch_norm_scale,
 401     }
 402     # Set activation_fn and parameters for batch_norm.
 403     with slim.arg_scope([slim.conv2d], activation_fn=activation_fn,
 404                         normalizer_fn=slim.batch_norm,
 405                         normalizer_params=batch_norm_params) as scope:
 406       return scope
