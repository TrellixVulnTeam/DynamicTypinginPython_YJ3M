  23 def _fixed_padding(inputs, kernel_size, rate=1):
  24   """Pads the input along the spatial dimensions independently of input size.
  25 
  26   Pads the input such that if it was used in a convolution with 'VALID' padding,
  27   the output would have the same dimensions as if the unpadded input was used
  28   in a convolution with 'SAME' padding.
  29 
  30   Args:
  31     inputs: A tensor of size [batch, height_in, width_in, channels].
  32     kernel_size: The kernel to be used in the conv2d or max_pool2d operation.
  33     rate: An integer, rate for atrous convolution.
  34 
  35   Returns:
  36     output: A tensor of size [batch, height_out, width_out, channels] with the
  37       input, either intact (if kernel_size == 1) or padded (if kernel_size > 1).
  38   """
  39   kernel_size_effective = [kernel_size[0] + (kernel_size[0] - 1) * (rate - 1),
  40                            kernel_size[0] + (kernel_size[0] - 1) * (rate - 1)]
  41   pad_total = [kernel_size_effective[0] - 1, kernel_size_effective[1] - 1]
  42   pad_beg = [pad_total[0] // 2, pad_total[1] // 2]
  43   pad_end = [pad_total[0] - pad_beg[0], pad_total[1] - pad_beg[1]]
  44   padded_inputs = tf.pad(inputs, [[0, 0], [pad_beg[0], pad_end[0]],
  45                                   [pad_beg[1], pad_end[1]], [0, 0]])
  46   return padded_inputs

  49 def _make_divisible(v, divisor, min_value=None):
  50   if min_value is None:
  51     min_value = divisor
  52   new_v = max(min_value, int(v + divisor / 2) // divisor * divisor)
  53   # Make sure that round down does not go down by more than 10%.
  54   if new_v < 0.9 * v:
  55     new_v += divisor
  56   return new_v

  59 def _split_divisible(num, num_ways, divisible_by=8):
  60   """Evenly splits num, num_ways so each piece is a multiple of divisible_by."""
  61   assert num % divisible_by == 0
  62   assert num / num_ways >= divisible_by
  63   # Note: want to round down, we adjust each split to match the total.
  64   base = num // num_ways // divisible_by * divisible_by
  65   result = []
  66   accumulated = 0
  67   for i in range(num_ways):
  68     r = base
  69     while accumulated + r < num * (i + 1) / num_ways:
  70       r += divisible_by
  71     result.append(r)
  72     accumulated += r
  73   assert accumulated == num
  74   return result

  77 @contextlib.contextmanager
  78 def _v1_compatible_scope_naming(scope):
  79   """v1 compatible scope naming."""
  80   if scope is None:  # Create uniqified separable blocks.
  81     with tf.variable_scope(None, default_name='separable') as s, \
  82          tf.name_scope(s.original_name_scope):
  83       yield ''
  84   else:
  85     # We use scope_depthwise, scope_pointwise for compatibility with V1 ckpts.
  86     # which provide numbered scopes.
  87     scope += '_'
  88     yield scope

  92 def split_separable_conv2d(input_tensor,
  93                            num_outputs,
  94                            scope=None,
  95                            normalizer_fn=None,
  96                            stride=1,
  97                            rate=1,
  98                            endpoints=None,
  99                            use_explicit_padding=False):
 100   """Separable mobilenet V1 style convolution.
 101 
 102   Depthwise convolution, with default non-linearity,
 103   followed by 1x1 depthwise convolution.  This is similar to
 104   slim.separable_conv2d, but differs in tha it applies batch
 105   normalization and non-linearity to depthwise. This  matches
 106   the basic building of Mobilenet Paper
 107   (https://arxiv.org/abs/1704.04861)
 108 
 109   Args:
 110     input_tensor: input
 111     num_outputs: number of outputs
 112     scope: optional name of the scope. Note if provided it will use
 113     scope_depthwise for deptwhise, and scope_pointwise for pointwise.
 114     normalizer_fn: which normalizer function to use for depthwise/pointwise
 115     stride: stride
 116     rate: output rate (also known as dilation rate)
 117     endpoints: optional, if provided, will export additional tensors to it.
 118     use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
 119       inputs so that the output dimensions are the same as if 'SAME' padding
 120       were used.
 121 
 122   Returns:
 123     output tesnor
 124   """
 125 
 126   with _v1_compatible_scope_naming(scope) as scope:
 127     dw_scope = scope + 'depthwise'
 128     endpoints = endpoints if endpoints is not None else {}
 129     kernel_size = [3, 3]
 130     padding = 'SAME'
 131     if use_explicit_padding:
 132       padding = 'VALID'
 133       input_tensor = _fixed_padding(input_tensor, kernel_size, rate)
 134     net = slim.separable_conv2d(
 135         input_tensor,
 136         None,
 137         kernel_size,
 138         depth_multiplier=1,
 139         stride=stride,
 140         rate=rate,
 141         normalizer_fn=normalizer_fn,
 142         padding=padding,
 143         scope=dw_scope)
 144 
 145     endpoints[dw_scope] = net
 146 
 147     pw_scope = scope + 'pointwise'
 148     net = slim.conv2d(
 149         net,
 150         num_outputs, [1, 1],
 151         stride=1,
 152         normalizer_fn=normalizer_fn,
 153         scope=pw_scope)
 154     endpoints[pw_scope] = net
 155   return net

 158 def expand_input_by_factor(n, divisible_by=8):
 159   return lambda num_inputs, **_: _make_divisible(num_inputs * n, divisible_by)

 162 def split_conv(input_tensor,
 163                num_outputs,
 164                num_ways,
 165                scope,
 166                divisible_by=8,
 167                **kwargs):
 168   """Creates a split convolution.
 169 
 170   Split convolution splits the input and output into
 171   'num_blocks' blocks of approximately the same size each,
 172   and only connects $i$-th input to $i$ output.
 173 
 174   Args:
 175     input_tensor: input tensor
 176     num_outputs: number of output filters
 177     num_ways: num blocks to split by.
 178     scope: scope for all the operators.
 179     divisible_by: make sure that every part is divisiable by this.
 180     **kwargs: will be passed directly into conv2d operator
 181   Returns:
 182     tensor
 183   """
 184   b = input_tensor.get_shape().as_list()[3]
 185 
 186   if num_ways == 1 or min(b // num_ways,
 187                           num_outputs // num_ways) < divisible_by:
 188     # Don't do any splitting if we end up with less than 8 filters
 189     # on either side.
 190     return slim.conv2d(input_tensor, num_outputs, [1, 1], scope=scope, **kwargs)
 191 
 192   outs = []
 193   input_splits = _split_divisible(b, num_ways, divisible_by=divisible_by)
 194   output_splits = _split_divisible(
 195       num_outputs, num_ways, divisible_by=divisible_by)
 196   inputs = tf.split(input_tensor, input_splits, axis=3, name='split_' + scope)
 197   base = scope
 198   for i, (input_tensor, out_size) in enumerate(zip(inputs, output_splits)):
 199     scope = base + '_part_%d' % (i,)
 200     n = slim.conv2d(input_tensor, out_size, [1, 1], scope=scope, **kwargs)
 201     n = tf.identity(n, scope + '_output')
 202     outs.append(n)
 203   return tf.concat(outs, 3, name=scope + '_concat')

 206 @slim.add_arg_scope
 207 def expanded_conv(input_tensor,
 208                   num_outputs,
 209                   expansion_size=expand_input_by_factor(6),
 210                   stride=1,
 211                   rate=1,
 212                   kernel_size=(3, 3),
 213                   residual=True,
 214                   normalizer_fn=None,
 215                   split_projection=1,
 216                   split_expansion=1,
 217                   split_divisible_by=8,
 218                   expansion_transform=None,
 219                   depthwise_location='expansion',
 220                   depthwise_channel_multiplier=1,
 221                   endpoints=None,
 222                   use_explicit_padding=False,
 223                   padding='SAME',
 224                   inner_activation_fn=None,
 225                   depthwise_activation_fn=None,
 226                   project_activation_fn=tf.identity,
 227                   depthwise_fn=slim.separable_conv2d,
 228                   expansion_fn=split_conv,
 229                   projection_fn=split_conv,
 230                   scope=None):
 231   """Depthwise Convolution Block with expansion.
 232 
 233   Builds a composite convolution that has the following structure
 234   expansion (1x1) -> depthwise (kernel_size) -> projection (1x1)
 235 
 236   Args:
 237     input_tensor: input
 238     num_outputs: number of outputs in the final layer.
 239     expansion_size: the size of expansion, could be a constant or a callable.
 240       If latter it will be provided 'num_inputs' as an input. For forward
 241       compatibility it should accept arbitrary keyword arguments.
 242       Default will expand the input by factor of 6.
 243     stride: depthwise stride
 244     rate: depthwise rate
 245     kernel_size: depthwise kernel
 246     residual: whether to include residual connection between input
 247       and output.
 248     normalizer_fn: batchnorm or otherwise
 249     split_projection: how many ways to split projection operator
 250       (that is conv expansion->bottleneck)
 251     split_expansion: how many ways to split expansion op
 252       (that is conv bottleneck->expansion) ops will keep depth divisible
 253       by this value.
 254     split_divisible_by: make sure every split group is divisible by this number.
 255     expansion_transform: Optional function that takes expansion
 256       as a single input and returns output.
 257     depthwise_location: where to put depthwise covnvolutions supported
 258       values None, 'input', 'output', 'expansion'
 259     depthwise_channel_multiplier: depthwise channel multiplier:
 260     each input will replicated (with different filters)
 261     that many times. So if input had c channels,
 262     output will have c x depthwise_channel_multpilier.
 263     endpoints: An optional dictionary into which intermediate endpoints are
 264       placed. The keys "expansion_output", "depthwise_output",
 265       "projection_output" and "expansion_transform" are always populated, even
 266       if the corresponding functions are not invoked.
 267     use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
 268       inputs so that the output dimensions are the same as if 'SAME' padding
 269       were used.
 270     padding: Padding type to use if `use_explicit_padding` is not set.
 271     inner_activation_fn: activation function to use in all inner convolutions.
 272     If none, will rely on slim default scopes.
 273     depthwise_activation_fn: activation function to use for deptwhise only.
 274       If not provided will rely on slim default scopes. If both
 275       inner_activation_fn and depthwise_activation_fn are provided,
 276       depthwise_activation_fn takes precedence over inner_activation_fn.
 277     project_activation_fn: activation function for the project layer.
 278     (note this layer is not affected by inner_activation_fn)
 279     depthwise_fn: Depthwise convolution function.
 280     expansion_fn: Expansion convolution function. If use custom function then
 281       "split_expansion" and "split_divisible_by" will be ignored.
 282     projection_fn: Projection convolution function. If use custom function then
 283       "split_projection" and "split_divisible_by" will be ignored.
 284 
 285     scope: optional scope.
 286 
 287   Returns:
 288     Tensor of depth num_outputs
 289 
 290   Raises:
 291     TypeError: on inval
 292   """
 293   conv_defaults = {}
 294   dw_defaults = {}
 295   if inner_activation_fn is not None:
 296     conv_defaults['activation_fn'] = inner_activation_fn
 297     dw_defaults['activation_fn'] = inner_activation_fn
 298   if depthwise_activation_fn is not None:
 299     dw_defaults['activation_fn'] = depthwise_activation_fn
 300   # pylint: disable=g-backslash-continuation
 301   with tf.variable_scope(scope, default_name='expanded_conv') as s, \
 302        tf.name_scope(s.original_name_scope), \
 303       slim.arg_scope((slim.conv2d,), **conv_defaults), \
 304        slim.arg_scope((slim.separable_conv2d,), **dw_defaults):
 305     prev_depth = input_tensor.get_shape().as_list()[3]
 306     if  depthwise_location not in [None, 'input', 'output', 'expansion']:
 307       raise TypeError('%r is unknown value for depthwise_location' %
 308                       depthwise_location)
 309     if use_explicit_padding:
 310       if padding != 'SAME':
 311         raise TypeError('`use_explicit_padding` should only be used with '
 312                         '"SAME" padding.')
 313       padding = 'VALID'
 314     depthwise_func = functools.partial(
 315         depthwise_fn,
 316         num_outputs=None,
 317         kernel_size=kernel_size,
 318         depth_multiplier=depthwise_channel_multiplier,
 319         stride=stride,
 320         rate=rate,
 321         normalizer_fn=normalizer_fn,
 322         padding=padding,
 323         scope='depthwise')
 324     # b1 -> b2 * r -> b2
 325     #   i -> (o * r) (bottleneck) -> o
 326     input_tensor = tf.identity(input_tensor, 'input')
 327     net = input_tensor
 328 
 329     if depthwise_location == 'input':
 330       if use_explicit_padding:
 331         net = _fixed_padding(net, kernel_size, rate)
 332       net = depthwise_func(net, activation_fn=None)
 333       net = tf.identity(net, name='depthwise_output')
 334       if endpoints is not None:
 335         endpoints['depthwise_output'] = net
 336 
 337     if callable(expansion_size):
 338       inner_size = expansion_size(num_inputs=prev_depth)
 339     else:
 340       inner_size = expansion_size
 341 
 342     if inner_size > net.shape[3]:
 343       if expansion_fn == split_conv:
 344         expansion_fn = functools.partial(
 345             expansion_fn,
 346             num_ways=split_expansion,
 347             divisible_by=split_divisible_by,
 348             stride=1)
 349       net = expansion_fn(
 350           net,
 351           inner_size,
 352           scope='expand',
 353           normalizer_fn=normalizer_fn)
 354       net = tf.identity(net, 'expansion_output')
 355       if endpoints is not None:
 356         endpoints['expansion_output'] = net
 357 
 358     if depthwise_location == 'expansion':
 359       if use_explicit_padding:
 360         net = _fixed_padding(net, kernel_size, rate)
 361       net = depthwise_func(net)
 362       net = tf.identity(net, name='depthwise_output')
 363       if endpoints is not None:
 364         endpoints['depthwise_output'] = net
 365 
 366     if expansion_transform:
 367       net = expansion_transform(expansion_tensor=net, input_tensor=input_tensor)
 368     # Note in contrast with expansion, we always have
 369     # projection to produce the desired output size.
 370     if projection_fn == split_conv:
 371       projection_fn = functools.partial(
 372           projection_fn,
 373           num_ways=split_projection,
 374           divisible_by=split_divisible_by,
 375           stride=1)
 376     net = projection_fn(
 377         net,
 378         num_outputs,
 379         scope='project',
 380         normalizer_fn=normalizer_fn,
 381         activation_fn=project_activation_fn)
 382     if endpoints is not None:
 383       endpoints['projection_output'] = net
 384     if depthwise_location == 'output':
 385       if use_explicit_padding:
 386         net = _fixed_padding(net, kernel_size, rate)
 387       net = depthwise_func(net, activation_fn=None)
 388       net = tf.identity(net, name='depthwise_output')
 389       if endpoints is not None:
 390         endpoints['depthwise_output'] = net
 391 
 392     if callable(residual):  # custom residual
 393       net = residual(input_tensor=input_tensor, output_tensor=net)
 394     elif (residual and
 395           # stride check enforces that we don't add residuals when spatial
 396           # dimensions are None
 397           stride == 1 and
 398           # Depth matches
 399           net.get_shape().as_list()[3] ==
 400           input_tensor.get_shape().as_list()[3]):
 401       net += input_tensor
 402     return tf.identity(net, name='output')

 405 @slim.add_arg_scope
 406 def squeeze_excite(input_tensor,
 407                    divisible_by=8,
 408                    squeeze_factor=3,
 409                    inner_activation_fn=tf.nn.relu,
 410                    gating_fn=tf.sigmoid,
 411                    squeeze_input_tensor=None,
 412                    pool=None):
 413   """Squeeze excite block for Mobilenet V3.
 414 
 415   If the squeeze_input_tensor - or the input_tensor if squeeze_input_tensor is
 416   None - contains variable dimensions (Nonetype in tensor shape), perform
 417   average pooling (as the first step in the squeeze operation) by calling
 418   reduce_mean across the H/W of the input tensor.
 419 
 420   Args:
 421     input_tensor: input tensor to apply SE block to.
 422     divisible_by: ensures all inner dimensions are divisible by this number.
 423     squeeze_factor: the factor of squeezing in the inner fully connected layer
 424     inner_activation_fn: non-linearity to be used in inner layer.
 425     gating_fn: non-linearity to be used for final gating function
 426     squeeze_input_tensor: custom tensor to use for computing gating activation.
 427      If provided the result will be input_tensor * SE(squeeze_input_tensor)
 428      instead of input_tensor * SE(input_tensor).
 429     pool: if number is  provided will average pool with that kernel size
 430       to compute inner tensor, followed by bilinear upsampling.
 431 
 432   Returns:
 433     Gated input_tensor. (e.g. X * SE(X))
 434   """
 435   with tf.variable_scope('squeeze_excite'):
 436     if squeeze_input_tensor is None:
 437       squeeze_input_tensor = input_tensor
 438     input_size = input_tensor.shape.as_list()[1:3]
 439     pool_height, pool_width = squeeze_input_tensor.shape.as_list()[1:3]
 440     stride = 1
 441     if pool is not None and pool_height >= pool:
 442       pool_height, pool_width, stride = pool, pool, pool
 443     input_channels = squeeze_input_tensor.shape.as_list()[3]
 444     output_channels = input_tensor.shape.as_list()[3]
 445     squeeze_channels = _make_divisible(
 446         input_channels / squeeze_factor, divisor=divisible_by)
 447 
 448     if pool is None:
 449       pooled = tf.reduce_mean(squeeze_input_tensor, axis=[1, 2], keepdims=True)
 450     else:
 451       pooled = tf.nn.avg_pool(
 452           squeeze_input_tensor, (1, pool_height, pool_width, 1),
 453           strides=(1, stride, stride, 1),
 454           padding='VALID')
 455     squeeze = slim.conv2d(
 456         pooled,
 457         kernel_size=(1, 1),
 458         num_outputs=squeeze_channels,
 459         normalizer_fn=None,
 460         activation_fn=inner_activation_fn)
 461     excite_outputs = output_channels
 462     excite = slim.conv2d(squeeze, num_outputs=excite_outputs,
 463                          kernel_size=[1, 1],
 464                          normalizer_fn=None,
 465                          activation_fn=gating_fn)
 466     if pool is not None:
 467       # Note: As of 03/20/2019 only BILINEAR (the default) with
 468       # align_corners=True has gradients implemented in TPU.
 469       excite = tf.image.resize_images(
 470           excite, input_size,
 471           align_corners=True)
 472     result = input_tensor * excite
 473   return result
