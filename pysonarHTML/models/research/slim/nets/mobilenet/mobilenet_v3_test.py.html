  29   # pylint: disable = g-unreachable-test-method
  30   def assertVariablesHaveNormalizerFn(self, use_groupnorm):
  31     global_variables = [v.name for v in tf.global_variables()]
  32     has_batch_norm = False
  33     has_group_norm = False
  34     for global_variable in global_variables:
  35       if 'BatchNorm' in global_variable:
  36         has_batch_norm = True
  37       if 'GroupNorm' in global_variable:
  38         has_group_norm = True
  39     if use_groupnorm:
  40       self.assertFalse(has_batch_norm)
  41       self.assertTrue(has_group_norm)
  42     else:
  43       self.assertTrue(has_batch_norm)
  44       self.assertFalse(has_group_norm)
  45 
  46   @parameterized.named_parameters(('without_groupnorm', False),
  47                                   ('with_groupnorm', True))
  48   def testMobilenetV3Large(self, use_groupnorm):
  49     logits, endpoints = mobilenet_v3.mobilenet(
  50         tf.placeholder(tf.float32, (1, 224, 224, 3)),
  51         use_groupnorm=use_groupnorm)
  52     self.assertEqual(endpoints['layer_19'].shape, [1, 1, 1, 1280])
  53     self.assertEqual(logits.shape, [1, 1001])
  54     self.assertVariablesHaveNormalizerFn(use_groupnorm)
  55 
  56   @parameterized.named_parameters(('without_groupnorm', False),
  57                                   ('with_groupnorm', True))
  58   def testMobilenetV3Small(self, use_groupnorm):
  59     _, endpoints = mobilenet_v3.mobilenet(
  60         tf.placeholder(tf.float32, (1, 224, 224, 3)),
  61         conv_defs=mobilenet_v3.V3_SMALL,
  62         use_groupnorm=use_groupnorm)
  63     self.assertEqual(endpoints['layer_15'].shape, [1, 1, 1, 1024])
  64     self.assertVariablesHaveNormalizerFn(use_groupnorm)
  65 
  66   @parameterized.named_parameters(('without_groupnorm', False),
  67                                   ('with_groupnorm', True))
  68   def testMobilenetEdgeTpu(self, use_groupnorm):
  69     _, endpoints = mobilenet_v3.edge_tpu(
  70         tf.placeholder(tf.float32, (1, 224, 224, 3)),
  71         use_groupnorm=use_groupnorm)
  72     self.assertIn('Inference mode is created by default',
  73                   mobilenet_v3.edge_tpu.__doc__)
  74     self.assertEqual(endpoints['layer_24'].shape, [1, 7, 7, 1280])
  75     self.assertStartsWith(
  76         endpoints['layer_24'].name, 'MobilenetEdgeTPU')
  77     self.assertVariablesHaveNormalizerFn(use_groupnorm)
  78 
  79   def testMobilenetEdgeTpuChangeScope(self):
  80     _, endpoints = mobilenet_v3.edge_tpu(
  81         tf.placeholder(tf.float32, (1, 224, 224, 3)), scope='Scope')
  82     self.assertStartsWith(
  83         endpoints['layer_24'].name, 'Scope')
  84 
  85   @parameterized.named_parameters(('without_groupnorm', False),
  86                                   ('with_groupnorm', True))
  87   def testMobilenetV3BaseOnly(self, use_groupnorm):
  88     result, endpoints = mobilenet_v3.mobilenet(
  89         tf.placeholder(tf.float32, (1, 224, 224, 3)),
  90         conv_defs=mobilenet_v3.V3_LARGE,
  91         use_groupnorm=use_groupnorm,
  92         base_only=True,
  93         final_endpoint='layer_17')
  94     # Get the latest layer before average pool.
  95     self.assertEqual(endpoints['layer_17'].shape, [1, 7, 7, 960])
  96     self.assertEqual(result, endpoints['layer_17'])
  97     self.assertVariablesHaveNormalizerFn(use_groupnorm)
  98 
  99   def testMobilenetV3BaseOnly_VariableInput(self):
 100     result, endpoints = mobilenet_v3.mobilenet(
 101         tf.placeholder(tf.float32, (None, None, None, 3)),
 102         conv_defs=mobilenet_v3.V3_LARGE,
 103         base_only=True,
 104         final_endpoint='layer_17')
 105     # Get the latest layer before average pool.
 106     self.assertEqual(endpoints['layer_17'].shape.as_list(),
 107                      [None, None, None, 960])
 108     self.assertEqual(result, endpoints['layer_17'])

 110   # Use reduce mean for pooling and check for operation 'ReduceMean' in graph
 111   @parameterized.named_parameters(('without_groupnorm', False),
 112                                   ('with_groupnorm', True))
 113   def testMobilenetV3WithReduceMean(self, use_groupnorm):
 114     _, _ = mobilenet_v3.mobilenet(
 115         tf.placeholder(tf.float32, (1, 224, 224, 3)),
 116         conv_defs=mobilenet_v3.V3_SMALL,
 117         use_groupnorm=use_groupnorm,
 118         use_reduce_mean_for_pooling=True)
 119     g = tf.get_default_graph()
 120     reduce_mean = [v for v in g.get_operations() if 'ReduceMean' in v.name]
 121     self.assertNotEmpty(reduce_mean)
 122     self.assertVariablesHaveNormalizerFn(use_groupnorm)
 123 
 124   @parameterized.named_parameters(('without_groupnorm', False),
 125                                   ('with_groupnorm', True))
 126   def testMobilenetV3WithOutReduceMean(self, use_groupnorm):
