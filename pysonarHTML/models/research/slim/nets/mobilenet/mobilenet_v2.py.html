 103 @slim.add_arg_scope
 104 def mobilenet(input_tensor,
 105               num_classes=1001,
 106               depth_multiplier=1.0,
 107               scope='MobilenetV2',
 108               conv_defs=None,
 109               finegrain_classification_mode=False,
 110               min_depth=None,
 111               divisible_by=None,
 112               activation_fn=None,
 113               **kwargs):
 114   """Creates mobilenet V2 network.
 115 
 116   Inference mode is created by default. To create training use training_scope
 117   below.
 118 
 119   with slim.arg_scope(mobilenet_v2.training_scope()):
 120      logits, endpoints = mobilenet_v2.mobilenet(input_tensor)
 121 
 122   Args:
 123     input_tensor: The input tensor
 124     num_classes: number of classes
 125     depth_multiplier: The multiplier applied to scale number of
 126     channels in each layer.
 127     scope: Scope of the operator
 128     conv_defs: Allows to override default conv def.
 129     finegrain_classification_mode: When set to True, the model
 130     will keep the last layer large even for small multipliers. Following
 131     https://arxiv.org/abs/1801.04381
 132     suggests that it improves performance for ImageNet-type of problems.
 133       *Note* ignored if final_endpoint makes the builder exit earlier.
 134     min_depth: If provided, will ensure that all layers will have that
 135     many channels after application of depth multiplier.
 136     divisible_by: If provided will ensure that all layers # channels
 137     will be divisible by this number.
 138     activation_fn: Activation function to use, defaults to tf.nn.relu6 if not
 139       specified.
 140     **kwargs: passed directly to mobilenet.mobilenet:
 141       prediction_fn- what prediction function to use.
 142       reuse-: whether to reuse variables (if reuse set to true, scope
 143       must be given).
 144   Returns:
 145     logits/endpoints pair
 146 
 147   Raises:
 148     ValueError: On invalid arguments
 149   """
 150   if conv_defs is None:
 151     conv_defs = V2_DEF
 152   if 'multiplier' in kwargs:
 153     raise ValueError('mobilenetv2 doesn\'t support generic '
 154                      'multiplier parameter use "depth_multiplier" instead.')
 155   if finegrain_classification_mode:
 156     conv_defs = copy.deepcopy(conv_defs)
 157     if depth_multiplier < 1:
 158       conv_defs['spec'][-1].params['num_outputs'] /= depth_multiplier
 159   if activation_fn:
 160     conv_defs = copy.deepcopy(conv_defs)
 161     defaults = conv_defs['defaults']
 162     conv_defaults = (
 163         defaults[(slim.conv2d, slim.fully_connected, slim.separable_conv2d)])
 164     conv_defaults['activation_fn'] = activation_fn
 165 
 166   depth_args = {}
 167   # NB: do not set depth_args unless they are provided to avoid overriding
 168   # whatever default depth_multiplier might have thanks to arg_scope.
 169   if min_depth is not None:
 170     depth_args['min_depth'] = min_depth
 171   if divisible_by is not None:
 172     depth_args['divisible_by'] = divisible_by
 173 
 174   with slim.arg_scope((lib.depth_multiplier,), **depth_args):
 175     return lib.mobilenet(
 176         input_tensor,
 177         num_classes=num_classes,
 178         conv_defs=conv_defs,
 179         scope=scope,
 180         multiplier=depth_multiplier,
 181         **kwargs)
 182 
 183 mobilenet.default_image_size = 224

 186 def wrapped_partial(func, *args, **kwargs):
 187   partial_func = functools.partial(func, *args, **kwargs)
 188   functools.update_wrapper(partial_func, func)
 189   return partial_func

 192 # Wrappers for mobilenet v2 with depth-multipliers. Be noticed that
 193 # 'finegrain_classification_mode' is set to True, which means the embedding
 194 # layer will not be shrinked when given a depth-multiplier < 1.0.
 195 mobilenet_v2_140 = wrapped_partial(mobilenet, depth_multiplier=1.4)
 196 mobilenet_v2_050 = wrapped_partial(mobilenet, depth_multiplier=0.50,
 197                                    finegrain_classification_mode=True)
 198 mobilenet_v2_035 = wrapped_partial(mobilenet, depth_multiplier=0.35,
 199                                    finegrain_classification_mode=True)

 202 @slim.add_arg_scope
 203 def mobilenet_base(input_tensor, depth_multiplier=1.0, **kwargs):
 204   """Creates base of the mobilenet (no pooling and no logits) ."""
 205   return mobilenet(input_tensor,
 206                    depth_multiplier=depth_multiplier,
 207                    base_only=True, **kwargs)

 210 @slim.add_arg_scope
 211 def mobilenet_base_group_norm(input_tensor, depth_multiplier=1.0, **kwargs):
 212   """Creates base of the mobilenet (no pooling and no logits) ."""
 213   kwargs['conv_defs'] = V2_DEF_GROUP_NORM
 214   kwargs['conv_defs']['defaults'].update({
 215       (slim.group_norm,): {
 216           'groups': kwargs.pop('groups', 8)
 217       }
 218   })
 219   return mobilenet(
 220       input_tensor, depth_multiplier=depth_multiplier, base_only=True, **kwargs)

 223 def training_scope(**kwargs):
 224   """Defines MobilenetV2 training scope.
 225 
 226   Usage:
 227      with slim.arg_scope(mobilenet_v2.training_scope()):
 228        logits, endpoints = mobilenet_v2.mobilenet(input_tensor)
 229 
 230   Args:
 231     **kwargs: Passed to mobilenet.training_scope. The following parameters
 232     are supported:
 233       weight_decay- The weight decay to use for regularizing the model.
 234       stddev-  Standard deviation for initialization, if negative uses xavier.
 235       dropout_keep_prob- dropout keep probability
 236       bn_decay- decay for the batch norm moving averages.
 237 
 238   Returns:
 239     An `arg_scope` to use for the mobilenet v2 model.
 240   """
 241   return lib.training_scope(**kwargs)
