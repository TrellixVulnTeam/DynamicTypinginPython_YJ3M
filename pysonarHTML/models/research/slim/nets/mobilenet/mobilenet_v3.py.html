 615 @slim.add_arg_scope
 616 def mobilenet(input_tensor,
 617               num_classes=1001,
 618               depth_multiplier=1.0,
 619               scope='MobilenetV3',
 620               conv_defs=None,
 621               finegrain_classification_mode=False,
 622               use_groupnorm=False,
 623               **kwargs):
 624   """Creates mobilenet V3 network.
 625 
 626   Inference mode is created by default. To create training use training_scope
 627   below.
 628 
 629   with slim.arg_scope(mobilenet_v3.training_scope()):
 630      logits, endpoints = mobilenet_v3.mobilenet(input_tensor)
 631 
 632   Args:
 633     input_tensor: The input tensor
 634     num_classes: number of classes
 635     depth_multiplier: The multiplier applied to scale number of
 636     channels in each layer.
 637     scope: Scope of the operator
 638     conv_defs: Which version to create. Could be large/small or
 639     any conv_def (see mobilenet_v3.py for examples).
 640     finegrain_classification_mode: When set to True, the model
 641     will keep the last layer large even for small multipliers. Following
 642     https://arxiv.org/abs/1801.04381
 643     it improves performance for ImageNet-type of problems.
 644       *Note* ignored if final_endpoint makes the builder exit earlier.
 645     use_groupnorm: When set to True, use group_norm as normalizer_fn.
 646 
 647     **kwargs: passed directly to mobilenet.mobilenet:
 648       prediction_fn- what prediction function to use.
 649       reuse-: whether to reuse variables (if reuse set to true, scope
 650       must be given).
 651   Returns:
 652     logits/endpoints pair
 653 
 654   Raises:
 655     ValueError: On invalid arguments
 656   """
 657   if conv_defs is None:
 658     conv_defs = V3_LARGE
 659   if 'multiplier' in kwargs:
 660     raise ValueError('mobilenetv2 doesn\'t support generic '
 661                      'multiplier parameter use "depth_multiplier" instead.')
 662 
 663   if use_groupnorm:
 664     conv_defs = copy.deepcopy(conv_defs)
 665     conv_defs['defaults'] = dict(DEFAULTS_GROUP_NORM)
 666     conv_defs['defaults'].update({
 667         (slim.group_norm,): {
 668             'groups': kwargs.pop('groups', 8)
 669         }
 670     })
 671 
 672   if finegrain_classification_mode:
 673     conv_defs = copy.deepcopy(conv_defs)
 674     conv_defs['spec'][-1] = conv_defs['spec'][-1]._replace(
 675         multiplier_func=lambda params, multiplier: params)
 676   depth_args = {}
 677   with slim.arg_scope((lib.depth_multiplier,), **depth_args):
 678     return lib.mobilenet(
 679         input_tensor,
 680         num_classes=num_classes,
 681         conv_defs=conv_defs,
 682         scope=scope,
 683         multiplier=depth_multiplier,
 684         **kwargs)

 697 def wrapped_partial(func, new_defaults=None,
 698                     **kwargs):
 699   """Partial function with new default parameters and updated docstring."""
 700   if not new_defaults:
 701     new_defaults = {}
 702   def func_wrapper(*f_args, **f_kwargs):
 703     new_kwargs = dict(new_defaults)
 704     new_kwargs.update(f_kwargs)
 705     return func(*f_args, **new_kwargs)
 706   functools.update_wrapper(func_wrapper, func)
 707   partial_func = functools.partial(func_wrapper, **kwargs)
 708   functools.update_wrapper(partial_func, func)
 709   return partial_func

 731 def _reduce_consecutive_layers(conv_defs, start_id, end_id, multiplier=0.5):
 732   """Reduce the outputs of consecutive layers with multiplier.
 733 
 734   Args:
 735     conv_defs: Mobilenet conv_defs.
 736     start_id: 0-based index of the starting conv_def to be reduced.
 737     end_id: 0-based index of the last conv_def to be reduced.
 738     multiplier: The multiplier by which to reduce the conv_defs.
 739 
 740   Returns:
 741     Mobilenet conv_defs where the output sizes from layers [start_id, end_id],
 742     inclusive, are reduced by multiplier.
 743 
 744   Raises:
 745     ValueError if any layer to be reduced does not have the 'num_outputs'
 746     attribute.
 747   """
 748   defs = copy.deepcopy(conv_defs)
 749   for d in defs['spec'][start_id:end_id+1]:
 750     d.params.update({
 751         'num_outputs': np.int(np.round(d.params['num_outputs'] * multiplier))
 752     })
 753   return defs
