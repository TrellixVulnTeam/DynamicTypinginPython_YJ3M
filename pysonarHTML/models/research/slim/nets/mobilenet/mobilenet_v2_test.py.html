  29 def find_ops(optype):
  30   """Find ops of a given type in graphdef or a graph.
  31 
  32   Args:
  33     optype: operation type (e.g. Conv2D)
  34   Returns:
  35      List of operations.
  36   """
  37   gd = tf.get_default_graph()
  38   return [var for var in gd.get_operations() if var.type == optype]

  43   def testCreation(self):
  44     spec = dict(mobilenet_v2.V2_DEF)
  45     _, ep = mobilenet.mobilenet(
  46         tf.placeholder(tf.float32, (10, 224, 224, 16)),
  47         conv_defs=spec)
  48     num_convs = len(find_ops('Conv2D'))
  49 
  50     # This is mostly a sanity test. No deep reason for these particular
  51     # constants.
  52     #
  53     # All but first 2 and last one have  two convolutions, and there is one
  54     # extra conv that is not in the spec. (logits)
  55     self.assertEqual(num_convs, len(spec['spec']) * 2 - 2)
  56     # Check that depthwise are exposed.
  57     for i in range(2, 17):
  58       self.assertIn('layer_%d/depthwise_output' % i, ep)

  60   def testCreationNoClasses(self):
  61     spec = copy.deepcopy(mobilenet_v2.V2_DEF)
  62     net, ep = mobilenet.mobilenet(
  63         tf.placeholder(tf.float32, (10, 224, 224, 16)),
  64         conv_defs=spec,
  65         num_classes=None)
  66     self.assertIs(net, ep['global_pool'])

  68   def testImageSizes(self):
  69     for input_size, output_size in [(224, 7), (192, 6), (160, 5),
  70                                     (128, 4), (96, 3)]:
  71       tf.reset_default_graph()
  72       _, ep = mobilenet_v2.mobilenet(
  73           tf.placeholder(tf.float32, (10, input_size, input_size, 3)))
  74 
  75       self.assertEqual(ep['layer_18/output'].get_shape().as_list()[1:3],
  76                        [output_size] * 2)

  78   def testWithSplits(self):
  79     spec = copy.deepcopy(mobilenet_v2.V2_DEF)
  80     spec['overrides'] = {
  81         (ops.expanded_conv,): dict(split_expansion=2),
  82     }
  83     _, _ = mobilenet.mobilenet(
  84         tf.placeholder(tf.float32, (10, 224, 224, 16)),
  85         conv_defs=spec)
  86     num_convs = len(find_ops('Conv2D'))
  87     # All but 3 op has 3 conv operatore, the remainign 3 have one
  88     # and there is one unaccounted.
  89     self.assertEqual(num_convs, len(spec['spec']) * 3 - 5)

  99   def testDivisibleBy(self):
 100     tf.reset_default_graph()
 101     mobilenet_v2.mobilenet(
 102         tf.placeholder(tf.float32, (10, 224, 224, 16)),
 103         conv_defs=mobilenet_v2.V2_DEF,
 104         divisible_by=16,
 105         min_depth=32)
 106     s = [op.outputs[0].get_shape().as_list()[-1] for op in find_ops('Conv2D')]
 107     s = set(s)
 108     self.assertSameElements([32, 64, 96, 160, 192, 320, 384, 576, 960, 1280,
 109                              1001], s)

 111   def testDivisibleByWithArgScope(self):
 112     tf.reset_default_graph()
 113     # Verifies that depth_multiplier arg scope actually works
 114     # if no default min_depth is provided.
 115     with slim.arg_scope((mobilenet.depth_multiplier,), min_depth=32):
 116       mobilenet_v2.mobilenet(
 117           tf.placeholder(tf.float32, (10, 224, 224, 2)),
 118           conv_defs=mobilenet_v2.V2_DEF,
 119           depth_multiplier=0.1)
 120       s = [op.outputs[0].get_shape().as_list()[-1] for op in find_ops('Conv2D')]
 121       s = set(s)
 122       self.assertSameElements(s, [32, 192, 128, 1001])

 124   def testFineGrained(self):
 125     tf.reset_default_graph()
 126     # Verifies that depth_multiplier arg scope actually works
 127     # if no default min_depth is provided.
 128 
 129     mobilenet_v2.mobilenet(
 130         tf.placeholder(tf.float32, (10, 224, 224, 2)),
 131         conv_defs=mobilenet_v2.V2_DEF,
 132         depth_multiplier=0.01,
 133         finegrain_classification_mode=True)
 134     s = [op.outputs[0].get_shape().as_list()[-1] for op in find_ops('Conv2D')]
 135     s = set(s)
 136     # All convolutions will be 8->48, except for the last one.
 137     self.assertSameElements(s, [8, 48, 1001, 1280])

 157   def testMultiplier(self):
 158     op = mobilenet.op
 159     new_def = copy.deepcopy(mobilenet_v2.V2_DEF)
 160 
 161     def inverse_multiplier(output_params, multiplier):
 162       output_params['num_outputs'] = int(
 163           output_params['num_outputs'] / multiplier)
 164 
 165     new_def['spec'][0] = op(
 166         slim.conv2d,
 167         kernel_size=(3, 3),
 168         multiplier_func=inverse_multiplier,
 169         num_outputs=16)
 170     _ = mobilenet_v2.mobilenet_base(
 171         tf.placeholder(tf.float32, (10, 224, 224, 16)),
 172         conv_defs=new_def,
 173         depth_multiplier=0.1)
 174     s = [op.outputs[0].get_shape().as_list()[-1] for op in find_ops('Conv2D')]
 175     # Expect first layer to be 160 (16 / 0.1), and other layers
 176     # their max(original size * 0.1, 8)
 177     self.assertEqual([160, 8, 48, 8, 48], s[:5])

 198   def testBatchNormScopeDoesNotHaveIsTrainingWhenItsSetToNone(self):
 199     sc = mobilenet.training_scope(is_training=None)
 200     self.assertNotIn('is_training', sc[slim.arg_scope_func_key(
 201         slim.batch_norm)])
 202 
 203   def testBatchNormScopeDoesHasIsTrainingWhenItsNotNone(self):
 204     sc = mobilenet.training_scope(is_training=False)
 205     self.assertIn('is_training', sc[slim.arg_scope_func_key(slim.batch_norm)])
 206     sc = mobilenet.training_scope(is_training=True)
 207     self.assertIn('is_training', sc[slim.arg_scope_func_key(slim.batch_norm)])
 208     sc = mobilenet.training_scope()
 209     self.assertIn('is_training', sc[slim.arg_scope_func_key(slim.batch_norm)])
