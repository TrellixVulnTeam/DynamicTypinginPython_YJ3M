  28   def testBuildLogitsCifarModel(self):
  29     batch_size = 5
  30     height, width = 32, 32
  31     num_classes = 10
  32     inputs = tf.random.uniform((batch_size, height, width, 3))
  33     tf.train.create_global_step()
  34     with slim.arg_scope(nasnet.nasnet_cifar_arg_scope()):
  35       logits, end_points = nasnet.build_nasnet_cifar(inputs, num_classes)
  36     auxlogits = end_points['AuxLogits']
  37     predictions = end_points['Predictions']
  38     self.assertListEqual(auxlogits.get_shape().as_list(),
  39                          [batch_size, num_classes])
  40     self.assertListEqual(logits.get_shape().as_list(),
  41                          [batch_size, num_classes])
  42     self.assertListEqual(predictions.get_shape().as_list(),
  43                          [batch_size, num_classes])

 137                         'Cell_10': [batch_size, 16, 16, 384],
 138                         'Cell_11': [batch_size, 16, 16, 384],
 139                         'Cell_12': [batch_size, 8, 8, 768],
 140                         'Cell_13': [batch_size, 8, 8, 768],
 141                         'Cell_14': [batch_size, 8, 8, 768],

 283   def testVariablesSetDeviceMobileModel(self):
 284     batch_size = 5
 285     height, width = 224, 224
 286     num_classes = 1000
 287     inputs = tf.random.uniform((batch_size, height, width, 3))
 288     tf.train.create_global_step()
 289     # Force all Variables to reside on the device.
 290     with tf.variable_scope('on_cpu'), tf.device('/cpu:0'):
 291       with slim.arg_scope(nasnet.nasnet_mobile_arg_scope()):
 292         nasnet.build_nasnet_mobile(inputs, num_classes)
 293     with tf.variable_scope('on_gpu'), tf.device('/gpu:0'):
 294       with slim.arg_scope(nasnet.nasnet_mobile_arg_scope()):
 295         nasnet.build_nasnet_mobile(inputs, num_classes)
 296     for v in tf.get_collection(
 297         tf.GraphKeys.GLOBAL_VARIABLES, scope='on_cpu'):
 298       self.assertDeviceEqual(v.device, '/cpu:0')
 299     for v in tf.get_collection(
 300         tf.GraphKeys.GLOBAL_VARIABLES, scope='on_gpu'):
 301       self.assertDeviceEqual(v.device, '/gpu:0')

 303   def testUnknownBatchSizeMobileModel(self):
 304     batch_size = 1
 305     height, width = 224, 224
 306     num_classes = 1000
 307     with self.test_session() as sess:
 308       inputs = tf.placeholder(tf.float32, (None, height, width, 3))
 309       with slim.arg_scope(nasnet.nasnet_mobile_arg_scope()):
 310         logits, _ = nasnet.build_nasnet_mobile(inputs, num_classes)
 311       self.assertListEqual(logits.get_shape().as_list(),
 312                            [None, num_classes])
 313       images = tf.random.uniform((batch_size, height, width, 3))
 314       sess.run(tf.global_variables_initializer())
 315       output = sess.run(logits, {inputs: images.eval()})
 316       self.assertEquals(output.shape, (batch_size, num_classes))

 318   def testEvaluationMobileModel(self):
 319     batch_size = 2
 320     height, width = 224, 224
 321     num_classes = 1000
 322     with self.test_session() as sess:
 323       eval_inputs = tf.random.uniform((batch_size, height, width, 3))
 324       with slim.arg_scope(nasnet.nasnet_mobile_arg_scope()):
 325         logits, _ = nasnet.build_nasnet_mobile(eval_inputs,
 326                                                num_classes,
 327                                                is_training=False)
 328       predictions = tf.argmax(input=logits, axis=1)
 329       sess.run(tf.global_variables_initializer())
 330       output = sess.run(predictions)
 331       self.assertEquals(output.shape, (batch_size,))

 375   def testCurrentStepCifarModel(self):
 376     batch_size = 5
 377     height, width = 32, 32
 378     num_classes = 10
 379     inputs = tf.random.uniform((batch_size, height, width, 3))
 380     global_step = tf.train.create_global_step()
 381     with slim.arg_scope(nasnet.nasnet_cifar_arg_scope()):
 382       logits, end_points = nasnet.build_nasnet_cifar(inputs,
 383                                                      num_classes,
 384                                                      current_step=global_step)
 385     auxlogits = end_points['AuxLogits']
 386     predictions = end_points['Predictions']
 387     self.assertListEqual(auxlogits.get_shape().as_list(),
 388                          [batch_size, num_classes])
 389     self.assertListEqual(logits.get_shape().as_list(),
 390                          [batch_size, num_classes])
 391     self.assertListEqual(predictions.get_shape().as_list(),
 392                          [batch_size, num_classes])

 394   def testUseBoundedAcitvationCifarModel(self):
 395     batch_size = 1
 396     height, width = 32, 32
 397     num_classes = 10
 398     for use_bounded_activation in (True, False):
 399       tf.reset_default_graph()
 400       inputs = tf.random.uniform((batch_size, height, width, 3))
 401       config = nasnet.cifar_config()
 402       config.set_hparam('use_bounded_activation', use_bounded_activation)
 403       with slim.arg_scope(nasnet.nasnet_cifar_arg_scope()):
 404         _, _ = nasnet.build_nasnet_cifar(
 405             inputs, num_classes, config=config)
 406       for node in tf.get_default_graph().as_graph_def().node:
 407         if node.op.startswith('Relu'):
 408           self.assertEqual(node.op == 'Relu6', use_bounded_activation)
