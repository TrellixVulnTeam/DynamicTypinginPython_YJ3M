  47 def calc_reduction_layers(num_cells, num_reduction_layers):
  48   """Figure out what layers should have reductions."""
  49   reduction_layers = []
  50   for pool_num in range(1, num_reduction_layers + 1):
  51     layer_num = (float(pool_num) / (num_reduction_layers + 1)) * num_cells
  52     layer_num = int(layer_num)
  53     reduction_layers.append(layer_num)
  54   return reduction_layers

  76 @slim.add_arg_scope
  77 def global_avg_pool(x, data_format=INVALID):
  78   """Average pool away the height and width spatial dimensions of x."""
  79   assert data_format != INVALID
  80   assert data_format in ['NHWC', 'NCHW']
  81   assert x.shape.ndims == 4
  82   if data_format == 'NHWC':
  83     return tf.reduce_mean(input_tensor=x, axis=[1, 2])
  84   else:
  85     return tf.reduce_mean(input_tensor=x, axis=[2, 3])

  88 @slim.add_arg_scope
  89 def factorized_reduction(net, output_filters, stride, data_format=INVALID):
  90   """Reduces the shape of net without information loss due to striding."""
  91   assert data_format != INVALID
  92   if stride == 1:
  93     net = slim.conv2d(net, output_filters, 1, scope='path_conv')
  94     net = slim.batch_norm(net, scope='path_bn')
  95     return net
  96   if data_format == 'NHWC':
  97     stride_spec = [1, stride, stride, 1]
  98   else:
  99     stride_spec = [1, 1, stride, stride]
 100 
 101   # Skip path 1
 102   path1 = tf.nn.avg_pool2d(
 103       net,
 104       ksize=[1, 1, 1, 1],
 105       strides=stride_spec,
 106       padding='VALID',
 107       data_format=data_format)
 108   path1 = slim.conv2d(path1, int(output_filters / 2), 1, scope='path1_conv')
 109 
 110   # Skip path 2
 111   # First pad with 0's on the right and bottom, then shift the filter to
 112   # include those 0's that were added.
 113   if data_format == 'NHWC':
 114     pad_arr = [[0, 0], [0, 1], [0, 1], [0, 0]]
 115     path2 = tf.pad(tensor=net, paddings=pad_arr)[:, 1:, 1:, :]
 116     concat_axis = 3
 117   else:
 118     pad_arr = [[0, 0], [0, 0], [0, 1], [0, 1]]
 119     path2 = tf.pad(tensor=net, paddings=pad_arr)[:, :, 1:, 1:]
 120     concat_axis = 1
 121   path2 = tf.nn.avg_pool2d(
 122       path2,
 123       ksize=[1, 1, 1, 1],
 124       strides=stride_spec,
 125       padding='VALID',
 126       data_format=data_format)
 127 
 128   # If odd number of filters, add an additional one to the second path.
 129   final_filter_size = int(output_filters / 2) + int(output_filters % 2)
 130   path2 = slim.conv2d(path2, final_filter_size, 1, scope='path2_conv')
 131 
 132   # Concat and apply BN
 133   final_path = tf.concat(values=[path1, path2], axis=concat_axis)
 134   final_path = slim.batch_norm(final_path, scope='final_path_bn')
 135   return final_path

 138 @slim.add_arg_scope
 139 def drop_path(net, keep_prob, is_training=True):
 140   """Drops out a whole example hiddenstate with the specified probability."""
 141   if is_training:
 142     batch_size = tf.shape(input=net)[0]
 143     noise_shape = [batch_size, 1, 1, 1]
 144     random_tensor = keep_prob
 145     random_tensor += tf.random.uniform(noise_shape, dtype=tf.float32)
 146     binary_tensor = tf.cast(tf.floor(random_tensor), net.dtype)
 147     keep_prob_inv = tf.cast(1.0 / keep_prob, net.dtype)
 148     net = net * keep_prob_inv * binary_tensor
 149 
 150   return net

 153 def _operation_to_filter_shape(operation):
 154   splitted_operation = operation.split('x')
 155   filter_shape = int(splitted_operation[0][-1])
 156   assert filter_shape == int(
 157       splitted_operation[1][0]), 'Rectangular filters not supported.'
 158   return filter_shape
 159

 161 def _operation_to_num_layers(operation):
 162   splitted_operation = operation.split('_')
 163   if 'x' in splitted_operation[-1]:
 164     return 1
 165   return int(splitted_operation[-1])

 168 def _operation_to_info(operation):
 169   """Takes in operation name and returns meta information.
 170 
 171   An example would be 'separable_3x3_4' -> (3, 4).
 172 
 173   Args:
 174     operation: String that corresponds to convolution operation.
 175 
 176   Returns:
 177     Tuple of (filter shape, num layers).
 178   """
 179   num_layers = _operation_to_num_layers(operation)
 180   filter_shape = _operation_to_filter_shape(operation)
 181   return num_layers, filter_shape

 184 def _stacked_separable_conv(net, stride, operation, filter_size,
 185                             use_bounded_activation):
 186   """Takes in an operations and parses it to the correct sep operation."""
 187   num_layers, kernel_size = _operation_to_info(operation)
 188   activation_fn = tf.nn.relu6 if use_bounded_activation else tf.nn.relu
 189   for layer_num in range(num_layers - 1):
 190     net = activation_fn(net)
 191     net = slim.separable_conv2d(
 192         net,
 193         filter_size,
 194         kernel_size,
 195         depth_multiplier=1,
 196         scope='separable_{0}x{0}_{1}'.format(kernel_size, layer_num + 1),
 197         stride=stride)
 198     net = slim.batch_norm(
 199         net, scope='bn_sep_{0}x{0}_{1}'.format(kernel_size, layer_num + 1))
 200     stride = 1
 201   net = activation_fn(net)
 202   net = slim.separable_conv2d(
 203       net,
 204       filter_size,
 205       kernel_size,
 206       depth_multiplier=1,
 207       scope='separable_{0}x{0}_{1}'.format(kernel_size, num_layers),
 208       stride=stride)
 209   net = slim.batch_norm(
 210       net, scope='bn_sep_{0}x{0}_{1}'.format(kernel_size, num_layers))
 211   return net

 214 def _operation_to_pooling_type(operation):
 215   """Takes in the operation string and returns the pooling type."""
 216   splitted_operation = operation.split('_')
 217   return splitted_operation[0]
 218 
 219 
 220 def _operation_to_pooling_shape(operation):
 221   """Takes in the operation string and returns the pooling kernel shape."""
 222   splitted_operation = operation.split('_')
 223   shape = splitted_operation[-1]
 224   assert 'x' in shape
 225   filter_height, filter_width = shape.split('x')
 226   assert filter_height == filter_width
 227   return int(filter_height)
 228 
 229 
 230 def _operation_to_pooling_info(operation):
 231   """Parses the pooling operation string to return its type and shape."""
 232   pooling_type = _operation_to_pooling_type(operation)
 233   pooling_shape = _operation_to_pooling_shape(operation)
 234   return pooling_type, pooling_shape

 237 def _pooling(net, stride, operation, use_bounded_activation):
 238   """Parses operation and performs the correct pooling operation on net."""
 239   padding = 'SAME'
 240   pooling_type, pooling_shape = _operation_to_pooling_info(operation)
 241   if use_bounded_activation:
 242     net = tf.nn.relu6(net)
 243   if pooling_type == 'avg':
 244     net = slim.avg_pool2d(net, pooling_shape, stride=stride, padding=padding)
 245   elif pooling_type == 'max':
 246     net = slim.max_pool2d(net, pooling_shape, stride=stride, padding=padding)
 247   else:
 248     raise NotImplementedError('Unimplemented pooling type: ', pooling_type)
 249   return net

 252 class NasNetABaseCell(object):
 253   """NASNet Cell class that is used as a 'layer' in image architectures.
 254 
 255   Args:
 256     num_conv_filters: The number of filters for each convolution operation.
 257     operations: List of operations that are performed in the NASNet Cell in
 258       order.
 259     used_hiddenstates: Binary array that signals if the hiddenstate was used
 260       within the cell. This is used to determine what outputs of the cell
 261       should be concatenated together.
 262     hiddenstate_indices: Determines what hiddenstates should be combined
 263       together with the specified operations to create the NASNet cell.
 264     use_bounded_activation: Whether or not to use bounded activations. Bounded
 265       activations better lend themselves to quantized inference.
 266   """
 267 
 268   def __init__(self, num_conv_filters, operations, used_hiddenstates,
 269                hiddenstate_indices, drop_path_keep_prob, total_num_cells,
 270                total_training_steps, use_bounded_activation=False):
 271     self._num_conv_filters = num_conv_filters
 272     self._operations = operations
 273     self._used_hiddenstates = used_hiddenstates
 274     self._hiddenstate_indices = hiddenstate_indices
 275     self._drop_path_keep_prob = drop_path_keep_prob
 276     self._total_num_cells = total_num_cells
 277     self._total_training_steps = total_training_steps
 278     self._use_bounded_activation = use_bounded_activation
 279 
 280   def _reduce_prev_layer(self, prev_layer, curr_layer):
 281     """Matches dimension of prev_layer to the curr_layer."""
 282     # Set the prev layer to the current layer if it is none
 283     if prev_layer is None:
 284       return curr_layer
 285     curr_num_filters = self._filter_size
 286     prev_num_filters = get_channel_dim(prev_layer.shape)
 287     curr_filter_shape = int(curr_layer.shape[2])
 288     prev_filter_shape = int(prev_layer.shape[2])
 289     activation_fn = tf.nn.relu6 if self._use_bounded_activation else tf.nn.relu
 290     if curr_filter_shape != prev_filter_shape:
 291       prev_layer = activation_fn(prev_layer)
 292       prev_layer = factorized_reduction(
 293           prev_layer, curr_num_filters, stride=2)
 294     elif curr_num_filters != prev_num_filters:
 295       prev_layer = activation_fn(prev_layer)
 296       prev_layer = slim.conv2d(
 297           prev_layer, curr_num_filters, 1, scope='prev_1x1')
 298       prev_layer = slim.batch_norm(prev_layer, scope='prev_bn')
 299     return prev_layer
 300 
 301   def _cell_base(self, net, prev_layer):
 302     """Runs the beginning of the conv cell before the predicted ops are run."""
 303     num_filters = self._filter_size
 304 
 305     # Check to be sure prev layer stuff is setup correctly
 306     prev_layer = self._reduce_prev_layer(prev_layer, net)
 307 
 308     net = tf.nn.relu6(net) if self._use_bounded_activation else tf.nn.relu(net)
 309     net = slim.conv2d(net, num_filters, 1, scope='1x1')
 310     net = slim.batch_norm(net, scope='beginning_bn')
 311     # num_or_size_splits=1
 312     net = [net]
 313     net.append(prev_layer)
 314     return net
 315 
 316   def __call__(self, net, scope=None, filter_scaling=1, stride=1,
 317                prev_layer=None, cell_num=-1, current_step=None):
 318     """Runs the conv cell."""
 319     self._cell_num = cell_num
 320     self._filter_scaling = filter_scaling
 321     self._filter_size = int(self._num_conv_filters * filter_scaling)
 322 
 323     i = 0
 324     with tf.variable_scope(scope):
 325       net = self._cell_base(net, prev_layer)
 326       for iteration in range(5):
 327         with tf.variable_scope('comb_iter_{}'.format(iteration)):
 328           left_hiddenstate_idx, right_hiddenstate_idx = (
 329               self._hiddenstate_indices[i],
 330               self._hiddenstate_indices[i + 1])
 331           original_input_left = left_hiddenstate_idx < 2
 332           original_input_right = right_hiddenstate_idx < 2
 333           h1 = net[left_hiddenstate_idx]
 334           h2 = net[right_hiddenstate_idx]
 335 
 336           operation_left = self._operations[i]
 337           operation_right = self._operations[i+1]
 338           i += 2
 339           # Apply conv operations
 340           with tf.variable_scope('left'):
 341             h1 = self._apply_conv_operation(h1, operation_left,
 342                                             stride, original_input_left,
 343                                             current_step)
 344           with tf.variable_scope('right'):
 345             h2 = self._apply_conv_operation(h2, operation_right,
 346                                             stride, original_input_right,
 347                                             current_step)
 348 
 349           # Combine hidden states using 'add'.
 350           with tf.variable_scope('combine'):
 351             h = h1 + h2
 352             if self._use_bounded_activation:
 353               h = tf.nn.relu6(h)
 354 
 355           # Add hiddenstate to the list of hiddenstates we can choose from
 356           net.append(h)
 357 
 358       with tf.variable_scope('cell_output'):
 359         net = self._combine_unused_states(net)
 360 
 361       return net
 362 
 363   def _apply_conv_operation(self, net, operation,
 364                             stride, is_from_original_input, current_step):
 365     """Applies the predicted conv operation to net."""
 366     # Dont stride if this is not one of the original hiddenstates
 367     if stride > 1 and not is_from_original_input:
 368       stride = 1
 369     input_filters = get_channel_dim(net.shape)
 370     filter_size = self._filter_size
 371     if 'separable' in operation:
 372       net = _stacked_separable_conv(net, stride, operation, filter_size,
 373                                     self._use_bounded_activation)
 374       if self._use_bounded_activation:
 375         net = tf.clip_by_value(net, -CLIP_BY_VALUE_CAP, CLIP_BY_VALUE_CAP)
 376     elif operation in ['none']:
 377       if self._use_bounded_activation:
 378         net = tf.nn.relu6(net)
 379       # Check if a stride is needed, then use a strided 1x1 here
 380       if stride > 1 or (input_filters != filter_size):
 381         if not self._use_bounded_activation:
 382           net = tf.nn.relu(net)
 383         net = slim.conv2d(net, filter_size, 1, stride=stride, scope='1x1')
 384         net = slim.batch_norm(net, scope='bn_1')
 385         if self._use_bounded_activation:
 386           net = tf.clip_by_value(net, -CLIP_BY_VALUE_CAP, CLIP_BY_VALUE_CAP)
 387     elif 'pool' in operation:
 388       net = _pooling(net, stride, operation, self._use_bounded_activation)
 389       if input_filters != filter_size:
 390         net = slim.conv2d(net, filter_size, 1, stride=1, scope='1x1')
 391         net = slim.batch_norm(net, scope='bn_1')
 392       if self._use_bounded_activation:
 393         net = tf.clip_by_value(net, -CLIP_BY_VALUE_CAP, CLIP_BY_VALUE_CAP)
 394     else:
 395       raise ValueError('Unimplemented operation', operation)
 396 
 397     if operation != 'none':
 398       net = self._apply_drop_path(net, current_step=current_step)
 399     return net
 400 
 401   def _combine_unused_states(self, net):
 402     """Concatenate the unused hidden states of the cell."""
 403     used_hiddenstates = self._used_hiddenstates
 404 
 405     final_height = int(net[-1].shape[2])
 406     final_num_filters = get_channel_dim(net[-1].shape)
 407     assert len(used_hiddenstates) == len(net)
 408     for idx, used_h in enumerate(used_hiddenstates):
 409       curr_height = int(net[idx].shape[2])
 410       curr_num_filters = get_channel_dim(net[idx].shape)
 411 
 412       # Determine if a reduction should be applied to make the number of
 413       # filters match.
 414       should_reduce = final_num_filters != curr_num_filters
 415       should_reduce = (final_height != curr_height) or should_reduce
 416       should_reduce = should_reduce and not used_h
 417       if should_reduce:
 418         stride = 2 if final_height != curr_height else 1
 419         with tf.variable_scope('reduction_{}'.format(idx)):
 420           net[idx] = factorized_reduction(
 421               net[idx], final_num_filters, stride)
 422 
 423     states_to_combine = (
 424         [h for h, is_used in zip(net, used_hiddenstates) if not is_used])
 425 
 426     # Return the concat of all the states
 427     concat_axis = get_channel_index()
 428     net = tf.concat(values=states_to_combine, axis=concat_axis)
 429     return net

 431   @slim.add_arg_scope  # No public API. For internal use only.
 432   def _apply_drop_path(self, net, current_step=None,
 433                        use_summaries=False, drop_connect_version='v3'):
 434     """Apply drop_path regularization.
 435 
 436     Args:
 437       net: the Tensor that gets drop_path regularization applied.
 438       current_step: a float32 Tensor with the current global_step value,
 439         to be divided by hparams.total_training_steps. Usually None, which
 440         defaults to tf.train.get_or_create_global_step() properly casted.
 441       use_summaries: a Python boolean. If set to False, no summaries are output.
 442       drop_connect_version: one of 'v1', 'v2', 'v3', controlling whether
 443         the dropout rate is scaled by current_step (v1), layer (v2), or
 444         both (v3, the default).
 445 
 446     Returns:
 447       The dropped-out value of `net`.
 448     """
 449     drop_path_keep_prob = self._drop_path_keep_prob
 450     if drop_path_keep_prob < 1.0:
 451       assert drop_connect_version in ['v1', 'v2', 'v3']
 452       if drop_connect_version in ['v2', 'v3']:
 453         # Scale keep prob by layer number
 454         assert self._cell_num != -1
 455         # The added 2 is for the reduction cells
 456         num_cells = self._total_num_cells
 457         layer_ratio = (self._cell_num + 1)/float(num_cells)
 458         if use_summaries:
 459           with tf.device('/cpu:0'):
 460             tf.summary.scalar('layer_ratio', layer_ratio)
 461         drop_path_keep_prob = 1 - layer_ratio * (1 - drop_path_keep_prob)
 462       if drop_connect_version in ['v1', 'v3']:
 463         # Decrease the keep probability over time
 464         if current_step is None:
 465           current_step = tf.train.get_or_create_global_step()
 466         current_step = tf.cast(current_step, tf.float32)
 467         drop_path_burn_in_steps = self._total_training_steps
 468         current_ratio = current_step / drop_path_burn_in_steps
 469         current_ratio = tf.minimum(1.0, current_ratio)
 470         if use_summaries:
 471           with tf.device('/cpu:0'):
 472             tf.summary.scalar('current_ratio', current_ratio)
 473         drop_path_keep_prob = (1 - current_ratio * (1 - drop_path_keep_prob))
 474       if use_summaries:
 475         with tf.device('/cpu:0'):
 476           tf.summary.scalar('drop_path_keep_prob', drop_path_keep_prob)
 477       net = drop_path(net, drop_path_keep_prob)
 478     return net
