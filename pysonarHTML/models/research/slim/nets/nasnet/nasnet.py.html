  33 # Notes for training NASNet Cifar Model
  34 # -------------------------------------
  35 # batch_size: 32
  36 # learning rate: 0.025
  37 # cosine (single period) learning rate decay
  38 # auxiliary head loss weighting: 0.4
  39 # clip global norm of all gradients by 5
  40 def cifar_config():
  41   return contrib_training.HParams(
  42       stem_multiplier=3.0,
  43       drop_path_keep_prob=0.6,
  44       num_cells=18,
  45       use_aux_head=1,
  46       num_conv_filters=32,
  47       dense_dropout_keep_prob=1.0,
  48       filter_scaling_rate=2.0,
  49       num_reduction_layers=2,
  50       data_format='NHWC',
  51       skip_reduction_layer_input=0,
  52       # 600 epochs with a batch size of 32
  53       # This is used for the drop path probabilities since it needs to increase
  54       # the drop out probability over the course of training.
  55       total_training_steps=937500,
  56       use_bounded_activation=False,
  57   )
  58

  60 # Notes for training large NASNet model on ImageNet
  61 # -------------------------------------
  62 # batch size (per replica): 16
  63 # learning rate: 0.015 * 100
  64 # learning rate decay factor: 0.97
  65 # num epochs per decay: 2.4
  66 # sync sgd with 100 replicas
  67 # auxiliary head loss weighting: 0.4
  68 # label smoothing: 0.1
  69 # clip global norm of all gradients by 10
  70 def large_imagenet_config():
  71   return contrib_training.HParams(
  72       stem_multiplier=3.0,
  73       dense_dropout_keep_prob=0.5,

 120 def nasnet_cifar_arg_scope(weight_decay=5e-4,
 121                            batch_norm_decay=0.9,
 122                            batch_norm_epsilon=1e-5):
 123   """Defines the default arg scope for the NASNet-A Cifar model.
 124 
 125   Args:
 126     weight_decay: The weight decay to use for regularizing the model.
 127     batch_norm_decay: Decay for batch norm moving average.
 128     batch_norm_epsilon: Small float added to variance to avoid dividing by zero
 129       in batch norm.
 130 
 131   Returns:
 132     An `arg_scope` to use for the NASNet Cifar Model.
 133   """
 134   batch_norm_params = {
 135       # Decay for the moving averages.
 136       'decay': batch_norm_decay,
 137       # epsilon to prevent 0s in variance.
 138       'epsilon': batch_norm_epsilon,
 139       'scale': True,
 140       'fused': True,
 141   }
 142   weights_regularizer = slim.l2_regularizer(weight_decay)
 143   weights_initializer = slim.variance_scaling_initializer(mode='FAN_OUT')
 144   with arg_scope([slim.fully_connected, slim.conv2d, slim.separable_conv2d],
 145                  weights_regularizer=weights_regularizer,
 146                  weights_initializer=weights_initializer):
 147     with arg_scope([slim.fully_connected],
 148                    activation_fn=None, scope='FC'):
 149       with arg_scope([slim.conv2d, slim.separable_conv2d],
 150                      activation_fn=None, biases_initializer=None):
 151         with arg_scope([slim.batch_norm], **batch_norm_params) as sc:
 152           return sc

 225 def _build_aux_head(net, end_points, num_classes, hparams, scope):
 226   """Auxiliary head used for all models across all datasets."""
 227   activation_fn = tf.nn.relu6 if hparams.use_bounded_activation else tf.nn.relu
 228   with tf.variable_scope(scope):
 229     aux_logits = tf.identity(net)
 230     with tf.variable_scope('aux_logits'):
 231       aux_logits = slim.avg_pool2d(
 232           aux_logits, [5, 5], stride=3, padding='VALID')
 233       aux_logits = slim.conv2d(aux_logits, 128, [1, 1], scope='proj')
 234       aux_logits = slim.batch_norm(aux_logits, scope='aux_bn0')
 235       aux_logits = activation_fn(aux_logits)
 236       # Shape of feature map before the final layer.
 237       shape = aux_logits.shape
 238       if hparams.data_format == 'NHWC':
 239         shape = shape[1:3]
 240       else:
 241         shape = shape[2:4]
 242       aux_logits = slim.conv2d(aux_logits, 768, shape, padding='VALID')
 243       aux_logits = slim.batch_norm(aux_logits, scope='aux_bn1')
 244       aux_logits = activation_fn(aux_logits)
 245       aux_logits = slim.flatten(aux_logits)
 246       aux_logits = slim.fully_connected(aux_logits, num_classes)
 247       end_points['AuxLogits'] = aux_logits

 250 def _imagenet_stem(inputs, hparams, stem_cell, current_step=None):
 251   """Stem used for models trained on ImageNet."""
 252   num_stem_cells = 2
 253 
 254   # 149 x 149 x 32
 255   num_stem_filters = int(32 * hparams.stem_multiplier)
 256   net = slim.conv2d(
 257       inputs, num_stem_filters, [3, 3], stride=2, scope='conv0',
 258       padding='VALID')
 259   net = slim.batch_norm(net, scope='conv0_bn')
 260 
 261   # Run the reduction cells
 262   cell_outputs = [None, net]
 263   filter_scaling = 1.0 / (hparams.filter_scaling_rate**num_stem_cells)
 264   for cell_num in range(num_stem_cells):
 265     net = stem_cell(
 266         net,
 267         scope='cell_stem_{}'.format(cell_num),
 268         filter_scaling=filter_scaling,
 269         stride=2,
 270         prev_layer=cell_outputs[-2],
 271         cell_num=cell_num,
 272         current_step=current_step)
 273     cell_outputs.append(net)
 274     filter_scaling *= hparams.filter_scaling_rate
 275   return net, cell_outputs

 278 def _cifar_stem(inputs, hparams):
 279   """Stem used for models trained on Cifar."""
 280   num_stem_filters = int(hparams.num_conv_filters * hparams.stem_multiplier)
 281   net = slim.conv2d(
 282       inputs,
 283       num_stem_filters,
 284       3,
 285       scope='l1_stem_3x3')
 286   net = slim.batch_norm(net, scope='l1_stem_bn')
 287   return net, [None, net]

 290 def build_nasnet_cifar(images, num_classes,
 291                        is_training=True,
 292                        config=None,
 293                        current_step=None):
 294   """Build NASNet model for the Cifar Dataset."""
 295   hparams = cifar_config() if config is None else copy.deepcopy(config)
 296   _update_hparams(hparams, is_training)
 297 
 298   if tf.test.is_gpu_available() and hparams.data_format == 'NHWC':
 299     tf.logging.info(
 300         'A GPU is available on the machine, consider using NCHW '
 301         'data format for increased speed on GPU.')
 302 
 303   if hparams.data_format == 'NCHW':
 304     images = tf.transpose(a=images, perm=[0, 3, 1, 2])
 305 
 306   # Calculate the total number of cells in the network
 307   # Add 2 for the reduction cells
 308   total_num_cells = hparams.num_cells + 2
 309 
 310   normal_cell = nasnet_utils.NasNetANormalCell(
 311       hparams.num_conv_filters, hparams.drop_path_keep_prob,
 312       total_num_cells, hparams.total_training_steps,
 313       hparams.use_bounded_activation)
 314   reduction_cell = nasnet_utils.NasNetAReductionCell(
 315       hparams.num_conv_filters, hparams.drop_path_keep_prob,
 316       total_num_cells, hparams.total_training_steps,
 317       hparams.use_bounded_activation)
 318   with arg_scope([slim.dropout, nasnet_utils.drop_path, slim.batch_norm],
 319                  is_training=is_training):
 320     with arg_scope([slim.avg_pool2d,
 321                     slim.max_pool2d,
 322                     slim.conv2d,
 323                     slim.batch_norm,
 324                     slim.separable_conv2d,
 325                     nasnet_utils.factorized_reduction,
 326                     nasnet_utils.global_avg_pool,
 327                     nasnet_utils.get_channel_index,
 328                     nasnet_utils.get_channel_dim],
 329                    data_format=hparams.data_format):
 330       return _build_nasnet_base(images,
 331                                 normal_cell=normal_cell,
 332                                 reduction_cell=reduction_cell,
 333                                 num_classes=num_classes,
 334                                 hparams=hparams,
 335                                 is_training=is_training,
 336                                 stem_type='cifar',
 337                                 current_step=current_step)
 338 build_nasnet_cifar.default_image_size = 32

 453 def _build_nasnet_base(images,
 454                        normal_cell,
 455                        reduction_cell,
 456                        num_classes,
 457                        hparams,
 458                        is_training,
 459                        stem_type,
 460                        final_endpoint=None,
 461                        current_step=None):
 462   """Constructs a NASNet image model."""
 463 
 464   end_points = {}
 465   def add_and_check_endpoint(endpoint_name, net):
 466     end_points[endpoint_name] = net
 467     return final_endpoint and (endpoint_name == final_endpoint)
 468 
 469   # Find where to place the reduction cells or stride normal cells
 470   reduction_indices = nasnet_utils.calc_reduction_layers(
 471       hparams.num_cells, hparams.num_reduction_layers)
 472   stem_cell = reduction_cell
 473 
 474   if stem_type == 'imagenet':
 475     stem = lambda: _imagenet_stem(images, hparams, stem_cell)
 476   elif stem_type == 'cifar':
 477     stem = lambda: _cifar_stem(images, hparams)
 478   else:
 479     raise ValueError('Unknown stem_type: ', stem_type)
 480   net, cell_outputs = stem()
 481   if add_and_check_endpoint('Stem', net): return net, end_points
 482 
 483   # Setup for building in the auxiliary head.
 484   aux_head_cell_idxes = []
 485   if len(reduction_indices) >= 2:
 486     aux_head_cell_idxes.append(reduction_indices[1] - 1)
 487 
 488   # Run the cells
 489   filter_scaling = 1.0
 490   # true_cell_num accounts for the stem cells
 491   true_cell_num = 2 if stem_type == 'imagenet' else 0
 492   activation_fn = tf.nn.relu6 if hparams.use_bounded_activation else tf.nn.relu
 493   for cell_num in range(hparams.num_cells):
 494     stride = 1
 495     if hparams.skip_reduction_layer_input:
 496       prev_layer = cell_outputs[-2]
 497     if cell_num in reduction_indices:
 498       filter_scaling *= hparams.filter_scaling_rate
 499       net = reduction_cell(
 500           net,
 501           scope='reduction_cell_{}'.format(reduction_indices.index(cell_num)),
 502           filter_scaling=filter_scaling,
 503           stride=2,
 504           prev_layer=cell_outputs[-2],
 505           cell_num=true_cell_num,
 506           current_step=current_step)
 507       if add_and_check_endpoint(
 508           'Reduction_Cell_{}'.format(reduction_indices.index(cell_num)), net):
 509         return net, end_points
 510       true_cell_num += 1
 511       cell_outputs.append(net)
 512     if not hparams.skip_reduction_layer_input:
 513       prev_layer = cell_outputs[-2]
 514     net = normal_cell(
 515         net,
 516         scope='cell_{}'.format(cell_num),
 517         filter_scaling=filter_scaling,
 518         stride=stride,
 519         prev_layer=prev_layer,
 520         cell_num=true_cell_num,
 521         current_step=current_step)
 522 
 523     if add_and_check_endpoint('Cell_{}'.format(cell_num), net):
 524       return net, end_points
 525     true_cell_num += 1
 526     if (hparams.use_aux_head and cell_num in aux_head_cell_idxes and
 527         num_classes and is_training):
 528       aux_net = activation_fn(net)
 529       _build_aux_head(aux_net, end_points, num_classes, hparams,
 530                       scope='aux_{}'.format(cell_num))
 531     cell_outputs.append(net)
 532 
 533   # Final softmax layer
 534   with tf.variable_scope('final_layer'):
 535     net = activation_fn(net)
 536     net = nasnet_utils.global_avg_pool(net)
 537     if add_and_check_endpoint('global_pool', net) or not num_classes:
 538       return net, end_points
 539     net = slim.dropout(net, hparams.dense_dropout_keep_prob, scope='dropout')
 540     logits = slim.fully_connected(net, num_classes)
 541 
 542     if add_and_check_endpoint('Logits', logits):
 543       return net, end_points
 544 
 545     predictions = tf.nn.softmax(logits, name='predictions')
 546     if add_and_check_endpoint('Predictions', predictions):
 547       return net, end_points
 548   return logits, end_points
