  86 def _build_pnasnet_base(images,
  87                         normal_cell,
  88                         num_classes,
  89                         hparams,
  90                         is_training,
  91                         final_endpoint=None):
  92   """Constructs a PNASNet image model."""
  93 
  94   end_points = {}
  95 
  96   def add_and_check_endpoint(endpoint_name, net):
  97     end_points[endpoint_name] = net
  98     return final_endpoint and (endpoint_name == final_endpoint)
  99 
 100   # Find where to place the reduction cells or stride normal cells
 101   reduction_indices = nasnet_utils.calc_reduction_layers(
 102       hparams.num_cells, hparams.num_reduction_layers)
 103 
 104   # pylint: disable=protected-access
 105   stem = lambda: nasnet._imagenet_stem(images, hparams, normal_cell)
 106   # pylint: enable=protected-access
 107   net, cell_outputs = stem()
 108   if add_and_check_endpoint('Stem', net):
 109     return net, end_points
 110 
 111   # Setup for building in the auxiliary head.
 112   aux_head_cell_idxes = []
 113   if len(reduction_indices) >= 2:
 114     aux_head_cell_idxes.append(reduction_indices[1] - 1)
 115 
 116   # Run the cells
 117   filter_scaling = 1.0
 118   # true_cell_num accounts for the stem cells
 119   true_cell_num = 2
 120   activation_fn = tf.nn.relu6 if hparams.use_bounded_activation else tf.nn.relu
 121   for cell_num in range(hparams.num_cells):
 122     is_reduction = cell_num in reduction_indices
 123     stride = 2 if is_reduction else 1
 124     if is_reduction: filter_scaling *= hparams.filter_scaling_rate
 125     if hparams.skip_reduction_layer_input or not is_reduction:
 126       prev_layer = cell_outputs[-2]
 127     net = normal_cell(
 128         net,
 129         scope='cell_{}'.format(cell_num),
 130         filter_scaling=filter_scaling,
 131         stride=stride,
 132         prev_layer=prev_layer,
 133         cell_num=true_cell_num)
 134     if add_and_check_endpoint('Cell_{}'.format(cell_num), net):
 135       return net, end_points
 136     true_cell_num += 1
 137     cell_outputs.append(net)
 138 
 139     if (hparams.use_aux_head and cell_num in aux_head_cell_idxes and
 140         num_classes and is_training):
 141       aux_net = activation_fn(net)
 142       # pylint: disable=protected-access
 143       nasnet._build_aux_head(aux_net, end_points, num_classes, hparams,
 144                              scope='aux_{}'.format(cell_num))
 145       # pylint: enable=protected-access
 146 
 147   # Final softmax layer
 148   with tf.variable_scope('final_layer'):
 149     net = activation_fn(net)
 150     net = nasnet_utils.global_avg_pool(net)
 151     if add_and_check_endpoint('global_pool', net) or not num_classes:
 152       return net, end_points
 153     net = slim.dropout(net, hparams.dense_dropout_keep_prob, scope='dropout')
 154     logits = slim.fully_connected(net, num_classes)
 155 
 156     if add_and_check_endpoint('Logits', logits):
 157       return net, end_points
 158 
 159     predictions = tf.nn.softmax(logits, name='predictions')
 160     if add_and_check_endpoint('Predictions', predictions):
 161       return net, end_points
 162   return logits, end_points

 165 def build_pnasnet_large(images,
 166                         num_classes,
 167                         is_training=True,
 168                         final_endpoint=None,
 169                         config=None):
 170   """Build PNASNet Large model for the ImageNet Dataset."""
 171   hparams = copy.deepcopy(config) if config else large_imagenet_config()
 172   # pylint: disable=protected-access
 173   nasnet._update_hparams(hparams, is_training)
 174   # pylint: enable=protected-access
 175 
 176   if tf.test.is_gpu_available() and hparams.data_format == 'NHWC':
 177     tf.logging.info(
 178         'A GPU is available on the machine, consider using NCHW '
 179         'data format for increased speed on GPU.')
 180 
 181   if hparams.data_format == 'NCHW':
 182     images = tf.transpose(a=images, perm=[0, 3, 1, 2])
 183 
 184   # Calculate the total number of cells in the network.
 185   # There is no distinction between reduction and normal cells in PNAS so the
 186   # total number of cells is equal to the number normal cells plus the number
 187   # of stem cells (two by default).
 188   total_num_cells = hparams.num_cells + 2
 189 
 190   normal_cell = PNasNetNormalCell(hparams.num_conv_filters,
 191                                   hparams.drop_path_keep_prob, total_num_cells,
 192                                   hparams.total_training_steps,
 193                                   hparams.use_bounded_activation)
 194   with arg_scope(
 195       [slim.dropout, nasnet_utils.drop_path, slim.batch_norm],
 196       is_training=is_training):
 197     with arg_scope([slim.avg_pool2d, slim.max_pool2d, slim.conv2d,
 198                     slim.batch_norm, slim.separable_conv2d,
 199                     nasnet_utils.factorized_reduction,
 200                     nasnet_utils.global_avg_pool,
 201                     nasnet_utils.get_channel_index,
 202                     nasnet_utils.get_channel_dim],
 203                    data_format=hparams.data_format):
 204       return _build_pnasnet_base(
 205           images,
 206           normal_cell=normal_cell,
 207           num_classes=num_classes,
 208           hparams=hparams,
 209           is_training=is_training,
 210           final_endpoint=final_endpoint)
 211 build_pnasnet_large.default_image_size = 331

 215                          num_classes,
 216                          is_training=True,
 217                          final_endpoint=None,

 219   """Build PNASNet Mobile model for the ImageNet Dataset."""

 223   # pylint: enable=protected-access
 224 
 225   if tf.test.is_gpu_available() and hparams.data_format == 'NHWC':
 226     tf.logging.info(
 227         'A GPU is available on the machine, consider using NCHW '
 228         'data format for increased speed on GPU.')
 229 
 230   if hparams.data_format == 'NCHW':
 231     images = tf.transpose(a=images, perm=[0, 3, 1, 2])
 232 
 233   # Calculate the total number of cells in the network.
 234   # There is no distinction between reduction and normal cells in PNAS so the
 235   # total number of cells is equal to the number normal cells plus the number
 236   # of stem cells (two by default).
 237   total_num_cells = hparams.num_cells + 2
 238 
 239   normal_cell = PNasNetNormalCell(hparams.num_conv_filters,
 240                                   hparams.drop_path_keep_prob, total_num_cells,
 241                                   hparams.total_training_steps,
 242                                   hparams.use_bounded_activation)
 243   with arg_scope(
 244       [slim.dropout, nasnet_utils.drop_path, slim.batch_norm],
 245       is_training=is_training):
 246     with arg_scope(
 247         [
 248             slim.avg_pool2d, slim.max_pool2d, slim.conv2d, slim.batch_norm,
 249             slim.separable_conv2d, nasnet_utils.factorized_reduction,
 250             nasnet_utils.global_avg_pool, nasnet_utils.get_channel_index,

 260           final_endpoint=final_endpoint)

 266 class PNasNetNormalCell(nasnet_utils.NasNetABaseCell):
 267   """PNASNet Normal Cell."""
 268 
 269   def __init__(self, num_conv_filters, drop_path_keep_prob, total_num_cells,
 270                total_training_steps, use_bounded_activation=False):
 271     # Configuration for the PNASNet-5 model.
 272     operations = [
 273         'separable_5x5_2', 'max_pool_3x3', 'separable_7x7_2', 'max_pool_3x3',
 274         'separable_5x5_2', 'separable_3x3_2', 'separable_3x3_2', 'max_pool_3x3',
 275         'separable_3x3_2', 'none'
 276     ]
 277     used_hiddenstates = [1, 1, 0, 0, 0, 0, 0]
 278     hiddenstate_indices = [1, 1, 0, 0, 0, 0, 4, 0, 1, 0]
 279 
 280     super(PNasNetNormalCell, self).__init__(
 281         num_conv_filters, operations, used_hiddenstates, hiddenstate_indices,
 282         drop_path_keep_prob, total_num_cells, total_training_steps,
 283         use_bounded_activation)
