  28   def testBuildLogitsLargeModel(self):
  29     batch_size = 5
  30     height, width = 331, 331
  31     num_classes = 1000
  32     inputs = tf.random.uniform((batch_size, height, width, 3))
  33     tf.train.create_global_step()
  34     with slim.arg_scope(pnasnet.pnasnet_large_arg_scope()):
  35       logits, end_points = pnasnet.build_pnasnet_large(inputs, num_classes)
  36     auxlogits = end_points['AuxLogits']
  37     predictions = end_points['Predictions']
  38     self.assertListEqual(auxlogits.get_shape().as_list(),
  39                          [batch_size, num_classes])
  40     self.assertListEqual(logits.get_shape().as_list(),
  41                          [batch_size, num_classes])
  42     self.assertListEqual(predictions.get_shape().as_list(),
  43                          [batch_size, num_classes])

  62   def testBuildNonExistingLayerLargeModel(self):
  63     """Tests that the model is built correctly without unnecessary layers."""
  64     inputs = tf.random.uniform((5, 331, 331, 3))
  65     tf.train.create_global_step()
  66     with slim.arg_scope(pnasnet.pnasnet_large_arg_scope()):
  67       pnasnet.build_pnasnet_large(inputs, 1000)
  68     vars_names = [x.op.name for x in tf.trainable_variables()]
  69     self.assertIn('cell_stem_0/1x1/weights', vars_names)
  70     self.assertNotIn('cell_stem_1/comb_iter_0/right/1x1/weights', vars_names)

  82   def testBuildPreLogitsLargeModel(self):
  83     batch_size = 5
  84     height, width = 331, 331
  85     num_classes = None
  86     inputs = tf.random.uniform((batch_size, height, width, 3))
  87     tf.train.create_global_step()
  88     with slim.arg_scope(pnasnet.pnasnet_large_arg_scope()):
  89       net, end_points = pnasnet.build_pnasnet_large(inputs, num_classes)
  90     self.assertFalse('AuxLogits' in end_points)
  91     self.assertFalse('Predictions' in end_points)
  92     self.assertTrue(net.op.name.startswith('final_layer/Mean'))
  93     self.assertListEqual(net.get_shape().as_list(), [batch_size, 4320])

 108   def testAllEndPointsShapesLargeModel(self):
 109     batch_size = 5
 110     height, width = 331, 331
 111     num_classes = 1000
 112     inputs = tf.random.uniform((batch_size, height, width, 3))
 113     tf.train.create_global_step()
 114     with slim.arg_scope(pnasnet.pnasnet_large_arg_scope()):
 115       _, end_points = pnasnet.build_pnasnet_large(inputs, num_classes)
 116 
 117     endpoints_shapes = {'Stem': [batch_size, 42, 42, 540],
 118                         'Cell_0': [batch_size, 42, 42, 1080],
 119                         'Cell_1': [batch_size, 42, 42, 1080],
 120                         'Cell_2': [batch_size, 42, 42, 1080],
 121                         'Cell_3': [batch_size, 42, 42, 1080],
 122                         'Cell_4': [batch_size, 21, 21, 2160],
 123                         'Cell_5': [batch_size, 21, 21, 2160],
 124                         'Cell_6': [batch_size, 21, 21, 2160],
 125                         'Cell_7': [batch_size, 21, 21, 2160],
 126                         'Cell_8': [batch_size, 11, 11, 4320],
 127                         'Cell_9': [batch_size, 11, 11, 4320],
 128                         'Cell_10': [batch_size, 11, 11, 4320],
 129                         'Cell_11': [batch_size, 11, 11, 4320],
 130                         'global_pool': [batch_size, 4320],
 131                         # Logits and predictions
 132                         'AuxLogits': [batch_size, 1000],
 133                         'Predictions': [batch_size, 1000],
 134                         'Logits': [batch_size, 1000],
 135                        }
 136     self.assertEqual(len(end_points), 17)
 137     self.assertItemsEqual(endpoints_shapes.keys(), end_points.keys())
 138     for endpoint_name in endpoints_shapes:
 139       tf.logging.info('Endpoint name: {}'.format(endpoint_name))
 140       expected_shape = endpoints_shapes[endpoint_name]
 141       self.assertIn(endpoint_name, end_points)
 142       self.assertListEqual(end_points[endpoint_name].get_shape().as_list(),
 143                            expected_shape)

 180   def testNoAuxHeadLargeModel(self):
 181     batch_size = 5
 182     height, width = 331, 331
 183     num_classes = 1000
 184     for use_aux_head in (True, False):
 185       tf.reset_default_graph()
 186       inputs = tf.random.uniform((batch_size, height, width, 3))
 187       tf.train.create_global_step()
 188       config = pnasnet.large_imagenet_config()
 189       config.set_hparam('use_aux_head', int(use_aux_head))
 190       with slim.arg_scope(pnasnet.pnasnet_large_arg_scope()):
 191         _, end_points = pnasnet.build_pnasnet_large(inputs, num_classes,
 192                                                     config=config)
 193       self.assertEqual('AuxLogits' in end_points, use_aux_head)

 238   def testUseBoundedAcitvationMobileModel(self):
 239     batch_size = 1
 240     height, width = 224, 224
 241     num_classes = 1000
 242     for use_bounded_activation in (True, False):
 243       tf.reset_default_graph()
 244       inputs = tf.random.uniform((batch_size, height, width, 3))
 245       config = pnasnet.mobile_imagenet_config()
 246       config.set_hparam('use_bounded_activation', use_bounded_activation)
 247       with slim.arg_scope(pnasnet.pnasnet_mobile_arg_scope()):
 248         _, _ = pnasnet.build_pnasnet_mobile(
 249             inputs, num_classes, config=config)
 250       for node in tf.get_default_graph().as_graph_def().node:
 251         if node.op.startswith('Relu'):
 252           self.assertEqual(node.op == 'Relu6', use_bounded_activation)
