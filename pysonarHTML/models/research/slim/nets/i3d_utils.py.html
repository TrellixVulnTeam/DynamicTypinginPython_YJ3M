  29 def center_initializer():
  30   """Centering Initializer for I3D.
  31 
  32   This initializer allows identity mapping for temporal convolution at the
  33   initialization, which is critical for a desired convergence behavior
  34   for training a seprable I3D model.
  35 
  36   The centering behavior of this initializer requires an odd-sized kernel,
  37   typically set to 3.
  38 
  39   Returns:
  40     A weight initializer op used in temporal convolutional layers.
  41 
  42   Raises:
  43     ValueError: Input tensor data type has to be tf.float32.
  44     ValueError: If input tensor is not a 5-D tensor.
  45     ValueError: If input and output channel dimensions are different.
  46     ValueError: If spatial kernel sizes are not 1.
  47     ValueError: If temporal kernel size is even.
  48   """
  49 
  50   def _initializer(shape, dtype=tf.float32, partition_info=None):  # pylint: disable=unused-argument
  51     """Initializer op."""
  52 
  53     if dtype != tf.float32 and dtype != tf.bfloat16:
  54       raise ValueError(
  55           'Input tensor data type has to be tf.float32 or tf.bfloat16.')
  56     if len(shape) != 5:
  57       raise ValueError('Input tensor has to be 5-D.')
  58     if shape[3] != shape[4]:
  59       raise ValueError('Input and output channel dimensions must be the same.')
  60     if shape[1] != 1 or shape[2] != 1:
  61       raise ValueError('Spatial kernel sizes must be 1 (pointwise conv).')
  62     if shape[0] % 2 == 0:
  63       raise ValueError('Temporal kernel size has to be odd.')
  64 
  65     center_pos = int(shape[0] / 2)
  66     init_mat = np.zeros(
  67         [shape[0], shape[1], shape[2], shape[3], shape[4]], dtype=np.float32)
  68     for i in range(0, shape[3]):
  69       init_mat[center_pos, 0, 0, i, i] = 1.0
  70 
  71     init_op = tf.constant(init_mat, dtype=dtype)
  72     return init_op
  73 
  74   return _initializer

  77 @add_arg_scope
  78 def conv3d_spatiotemporal(inputs,
  79                           num_outputs,
  80                           kernel_size,
  81                           stride=1,
  82                           padding='SAME',
  83                           activation_fn=None,
  84                           normalizer_fn=None,
  85                           normalizer_params=None,
  86                           weights_regularizer=None,
  87                           separable=False,
  88                           data_format='NDHWC',
  89                           scope=''):
  90   """A wrapper for conv3d to model spatiotemporal representations.
  91 
  92   This allows switching between original 3D convolution and separable 3D
  93   convolutions for spatial and temporal features respectively. On Kinetics,
  94   seprable 3D convolutions yields better classification performance.
  95 
  96   Args:
  97     inputs: a 5-D tensor  `[batch_size, depth, height, width, channels]`.
  98     num_outputs: integer, the number of output filters.
  99     kernel_size: a list of length 3
 100       `[kernel_depth, kernel_height, kernel_width]` of the filters. Can be an
 101       int if all values are the same.
 102     stride: a list of length 3 `[stride_depth, stride_height, stride_width]`.
 103       Can be an int if all strides are the same.
 104     padding: one of `VALID` or `SAME`.
 105     activation_fn: activation function.
 106     normalizer_fn: normalization function to use instead of `biases`.
 107     normalizer_params: dictionary of normalization function parameters.
 108     weights_regularizer: Optional regularizer for the weights.
 109     separable: If `True`, use separable spatiotemporal convolutions.
 110     data_format: An optional string from: "NDHWC", "NCDHW". Defaults to "NDHWC".
 111       The data format of the input and output data. With the default format
 112       "NDHWC", the data is stored in the order of: [batch, in_depth, in_height,
 113       in_width, in_channels]. Alternatively, the format could be "NCDHW", the
 114       data storage order is:
 115       [batch, in_channels, in_depth, in_height, in_width].
 116     scope: scope for `variable_scope`.
 117 
 118   Returns:
 119     A tensor representing the output of the (separable) conv3d operation.
 120 
 121   """
 122   assert len(kernel_size) == 3
 123   if separable and kernel_size[0] != 1:
 124     spatial_kernel_size = [1, kernel_size[1], kernel_size[2]]
 125     temporal_kernel_size = [kernel_size[0], 1, 1]
 126     if isinstance(stride, list) and len(stride) == 3:
 127       spatial_stride = [1, stride[1], stride[2]]
 128       temporal_stride = [stride[0], 1, 1]
 129     else:
 130       spatial_stride = [1, stride, stride]
 131       temporal_stride = [stride, 1, 1]
 132     net = layers.conv3d(
 133         inputs,
 134         num_outputs,
 135         spatial_kernel_size,
 136         stride=spatial_stride,
 137         padding=padding,
 138         activation_fn=activation_fn,
 139         normalizer_fn=normalizer_fn,
 140         normalizer_params=normalizer_params,
 141         weights_regularizer=weights_regularizer,
 142         data_format=data_format,
 143         scope=scope)
 144     net = layers.conv3d(
 145         net,
 146         num_outputs,
 147         temporal_kernel_size,
 148         stride=temporal_stride,
 149         padding=padding,
 150         scope=scope + '/temporal',
 151         activation_fn=activation_fn,
 152         normalizer_fn=None,
 153         data_format=data_format,
 154         weights_initializer=center_initializer())
 155     return net
 156   else:
 157     return layers.conv3d(
 158         inputs,
 159         num_outputs,
 160         kernel_size,
 161         stride=stride,
 162         padding=padding,
 163         activation_fn=activation_fn,
 164         normalizer_fn=normalizer_fn,
 165         normalizer_params=normalizer_params,
 166         weights_regularizer=weights_regularizer,
 167         data_format=data_format,
 168         scope=scope)

 171 @add_arg_scope
 172 def inception_block_v1_3d(inputs,
 173                           num_outputs_0_0a,
 174                           num_outputs_1_0a,
 175                           num_outputs_1_0b,
 176                           num_outputs_2_0a,
 177                           num_outputs_2_0b,
 178                           num_outputs_3_0b,
 179                           temporal_kernel_size=3,
 180                           self_gating_fn=None,
 181                           data_format='NDHWC',
 182                           scope=''):
 183   """A 3D Inception v1 block.
 184 
 185   This allows use of separable 3D convolutions and self-gating, as
 186   described in:
 187   Saining Xie, Chen Sun, Jonathan Huang, Zhuowen Tu and Kevin Murphy,
 188     Rethinking Spatiotemporal Feature Learning For Video Understanding.
 189     https://arxiv.org/abs/1712.04851.
 190 
 191   Args:
 192     inputs: a 5-D tensor  `[batch_size, depth, height, width, channels]`.
 193     num_outputs_0_0a: integer, the number of output filters for Branch 0,
 194       operation Conv2d_0a_1x1.
 195     num_outputs_1_0a: integer, the number of output filters for Branch 1,
 196       operation Conv2d_0a_1x1.
 197     num_outputs_1_0b: integer, the number of output filters for Branch 1,
 198       operation Conv2d_0b_3x3.
 199     num_outputs_2_0a: integer, the number of output filters for Branch 2,
 200       operation Conv2d_0a_1x1.
 201     num_outputs_2_0b: integer, the number of output filters for Branch 2,
 202       operation Conv2d_0b_3x3.
 203     num_outputs_3_0b: integer, the number of output filters for Branch 3,
 204       operation Conv2d_0b_1x1.
 205     temporal_kernel_size: integer, the size of the temporal convolutional
 206       filters in the conv3d_spatiotemporal blocks.
 207     self_gating_fn: function which optionally performs self-gating.
 208       Must have two arguments, `inputs` and `scope`, and return one output
 209       tensor the same size as `inputs`. If `None`, no self-gating is
 210       applied.
 211     data_format: An optional string from: "NDHWC", "NCDHW". Defaults to "NDHWC".
 212       The data format of the input and output data. With the default format
 213       "NDHWC", the data is stored in the order of: [batch, in_depth, in_height,
 214       in_width, in_channels]. Alternatively, the format could be "NCDHW", the
 215       data storage order is:
 216       [batch, in_channels, in_depth, in_height, in_width].
 217     scope: scope for `variable_scope`.
 218 
 219   Returns:
 220     A 5-D tensor `[batch_size, depth, height, width, out_channels]`, where
 221     `out_channels = num_outputs_0_0a + num_outputs_1_0b + num_outputs_2_0b
 222     + num_outputs_3_0b`.
 223 
 224   """
 225   use_gating = self_gating_fn is not None
 226 
 227   with tf.variable_scope(scope):
 228     with tf.variable_scope('Branch_0'):
 229       branch_0 = layers.conv3d(
 230           inputs, num_outputs_0_0a, [1, 1, 1], scope='Conv2d_0a_1x1')
 231       if use_gating:
 232         branch_0 = self_gating_fn(branch_0, scope='Conv2d_0a_1x1')
 233     with tf.variable_scope('Branch_1'):
 234       branch_1 = layers.conv3d(
 235           inputs, num_outputs_1_0a, [1, 1, 1], scope='Conv2d_0a_1x1')
 236       branch_1 = conv3d_spatiotemporal(
 237           branch_1, num_outputs_1_0b, [temporal_kernel_size, 3, 3],
 238           scope='Conv2d_0b_3x3')
 239       if use_gating:
 240         branch_1 = self_gating_fn(branch_1, scope='Conv2d_0b_3x3')
 241     with tf.variable_scope('Branch_2'):
 242       branch_2 = layers.conv3d(
 243           inputs, num_outputs_2_0a, [1, 1, 1], scope='Conv2d_0a_1x1')
 244       branch_2 = conv3d_spatiotemporal(
 245           branch_2, num_outputs_2_0b, [temporal_kernel_size, 3, 3],
 246           scope='Conv2d_0b_3x3')
 247       if use_gating:
 248         branch_2 = self_gating_fn(branch_2, scope='Conv2d_0b_3x3')
 249     with tf.variable_scope('Branch_3'):
 250       branch_3 = layers.max_pool3d(inputs, [3, 3, 3], scope='MaxPool_0a_3x3')
 251       branch_3 = layers.conv3d(
 252           branch_3, num_outputs_3_0b, [1, 1, 1], scope='Conv2d_0b_1x1')
 253       if use_gating:
 254         branch_3 = self_gating_fn(branch_3, scope='Conv2d_0b_1x1')
 255     index_c = data_format.index('C')
 256     assert 1 <= index_c <= 4, 'Cannot identify channel dimension.'
 257     output = tf.concat([branch_0, branch_1, branch_2, branch_3], index_c)
 258   return output

 261 def reduced_kernel_size_3d(input_tensor, kernel_size):
 262   """Define kernel size which is automatically reduced for small input.
 263 
 264   If the shape of the input images is unknown at graph construction time this
 265   function assumes that the input images are large enough.
 266 
 267   Args:
 268     input_tensor: input tensor of size
 269       [batch_size, time, height, width, channels].
 270     kernel_size: desired kernel size of length 3, corresponding to time,
 271       height and width.
 272 
 273   Returns:
 274     a tensor with the kernel size.
 275   """
 276   assert len(kernel_size) == 3
 277   shape = input_tensor.get_shape().as_list()
 278   assert len(shape) == 5
 279   if None in shape[1:4]:
 280     kernel_size_out = kernel_size
 281   else:
 282     kernel_size_out = [min(shape[1], kernel_size[0]),
 283                        min(shape[2], kernel_size[1]),
 284                        min(shape[3], kernel_size[2])]
 285   return kernel_size_out
