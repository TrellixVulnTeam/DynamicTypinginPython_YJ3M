  28   def testBuildLogits(self):
  29     batch_size = 5
  30     height, width = 299, 299
  31     num_classes = 1000
  32     with self.test_session():
  33       inputs = tf.random.uniform((batch_size, height, width, 3))
  34       logits, endpoints = inception.inception_resnet_v2(inputs, num_classes)
  35       self.assertTrue('AuxLogits' in endpoints)
  36       auxlogits = endpoints['AuxLogits']
  37       self.assertTrue(
  38           auxlogits.op.name.startswith('InceptionResnetV2/AuxLogits'))
  39       self.assertListEqual(auxlogits.get_shape().as_list(),
  40                            [batch_size, num_classes])
  41       self.assertTrue(logits.op.name.startswith('InceptionResnetV2/Logits'))
  42       self.assertListEqual(logits.get_shape().as_list(),
  43                            [batch_size, num_classes])

  71   def testBuildEndPoints(self):
  72     batch_size = 5
  73     height, width = 299, 299
  74     num_classes = 1000
  75     with self.test_session():
  76       inputs = tf.random.uniform((batch_size, height, width, 3))

  78       self.assertTrue('Logits' in end_points)

  81                            [batch_size, num_classes])
  82       self.assertTrue('AuxLogits' in end_points)

  87       self.assertListEqual(pre_pool.get_shape().as_list(),
  88                            [batch_size, 8, 8, 1536])

 105   def testBuildOnlyUptoFinalEndpoint(self):
 106     batch_size = 5
 107     height, width = 299, 299
 108     endpoints = ['Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3',
 109                  'MaxPool_3a_3x3', 'Conv2d_3b_1x1', 'Conv2d_4a_3x3',
 110                  'MaxPool_5a_3x3', 'Mixed_5b', 'Mixed_6a',
 111                  'PreAuxLogits', 'Mixed_7a', 'Conv2d_7b_1x1']
 112     for index, endpoint in enumerate(endpoints):
 113       with tf.Graph().as_default():
 114         inputs = tf.random.uniform((batch_size, height, width, 3))
 115         out_tensor, end_points = inception.inception_resnet_v2_base(
 116             inputs, final_endpoint=endpoint)
 117         if endpoint != 'PreAuxLogits':
 118           self.assertTrue(out_tensor.op.name.startswith(
 119               'InceptionResnetV2/' + endpoint))
 120         self.assertItemsEqual(endpoints[:index + 1], end_points.keys())

 200   def testVariablesSetDevice(self):
 201     batch_size = 5
 202     height, width = 299, 299
 203     num_classes = 1000
 204     with self.test_session():
 205       inputs = tf.random.uniform((batch_size, height, width, 3))
 206       # Force all Variables to reside on the device.
 207       with tf.variable_scope('on_cpu'), tf.device('/cpu:0'):
 208         inception.inception_resnet_v2(inputs, num_classes)
 209       with tf.variable_scope('on_gpu'), tf.device('/gpu:0'):
 210         inception.inception_resnet_v2(inputs, num_classes)
 211       for v in tf.get_collection(
 212           tf.GraphKeys.GLOBAL_VARIABLES, scope='on_cpu'):
 213         self.assertDeviceEqual(v.device, '/cpu:0')
 214       for v in tf.get_collection(
 215           tf.GraphKeys.GLOBAL_VARIABLES, scope='on_gpu'):
 216         self.assertDeviceEqual(v.device, '/gpu:0')

 246   def testGlobalPoolUnknownImageShape(self):
 247     batch_size = 1
 248     height, width = 330, 400
 249     num_classes = 1000
 250     with self.test_session() as sess:
 251       inputs = tf.placeholder(tf.float32, (batch_size, None, None, 3))
 252       logits, end_points = inception.inception_resnet_v2(
 253           inputs, num_classes, create_aux_logits=False)
 254       self.assertTrue(logits.op.name.startswith('InceptionResnetV2/Logits'))
 255       self.assertListEqual(logits.get_shape().as_list(),
 256                            [batch_size, num_classes])
 257       pre_pool = end_points['Conv2d_7b_1x1']
 258       images = tf.random.uniform((batch_size, height, width, 3))
 259       sess.run(tf.global_variables_initializer())
 260       logits_out, pre_pool_out = sess.run([logits, pre_pool],
 261                                           {inputs: images.eval()})
 262       self.assertTupleEqual(logits_out.shape, (batch_size, num_classes))
 263       self.assertTupleEqual(pre_pool_out.shape, (batch_size, 8, 11, 1536))

 266     batch_size = 1

 268     num_classes = 1000

 321   def testBatchNormScale(self):
 322     height, width = 299, 299
 323     num_classes = 1000
 324     inputs = tf.placeholder(tf.float32, (1, height, width, 3))
 325     with slim.arg_scope(
 326         inception.inception_resnet_v2_arg_scope(batch_norm_scale=True)):
 327       inception.inception_resnet_v2(inputs, num_classes, is_training=False)
 328 
 329     gamma_names = set(
 330         v.op.name
 331         for v in tf.global_variables('.*/BatchNorm/gamma:0$'))
 332     self.assertGreater(len(gamma_names), 0)
 333     for v in tf.global_variables('.*/BatchNorm/moving_mean:0$'):
 334       self.assertIn(v.op.name[:-len('moving_mean')] + 'gamma', gamma_names)
