  26 class AlexnetV2Test(tf.test.TestCase):

  28   def testBuild(self):
  29     batch_size = 5
  30     height, width = 224, 224
  31     num_classes = 1000
  32     with self.test_session():
  33       inputs = tf.random.uniform((batch_size, height, width, 3))
  34       logits, _ = alexnet.alexnet_v2(inputs, num_classes)
  35       self.assertEquals(logits.op.name, 'alexnet_v2/fc8/squeezed')
  36       self.assertListEqual(logits.get_shape().as_list(),
  37                            [batch_size, num_classes])
  38 
  39   def testFullyConvolutional(self):

 133   def testEvaluation(self):
 134     batch_size = 2
 135     height, width = 224, 224
 136     num_classes = 1000
 137     with self.test_session():
 138       eval_inputs = tf.random.uniform((batch_size, height, width, 3))
 139       logits, _ = alexnet.alexnet_v2(eval_inputs, is_training=False)
 140       self.assertListEqual(logits.get_shape().as_list(),
 141                            [batch_size, num_classes])
 142       predictions = tf.argmax(input=logits, axis=1)
 143       self.assertListEqual(predictions.get_shape().as_list(), [batch_size])

 145   def testTrainEvalWithReuse(self):
 146     train_batch_size = 2
 147     eval_batch_size = 1
 148     train_height, train_width = 224, 224
 149     eval_height, eval_width = 300, 400
 150     num_classes = 1000
 151     with self.test_session():
 152       train_inputs = tf.random.uniform(
 153           (train_batch_size, train_height, train_width, 3))
 154       logits, _ = alexnet.alexnet_v2(train_inputs)
 155       self.assertListEqual(logits.get_shape().as_list(),
 156                            [train_batch_size, num_classes])
 157       tf.get_variable_scope().reuse_variables()
 158       eval_inputs = tf.random.uniform(
 159           (eval_batch_size, eval_height, eval_width, 3))
 160       logits, _ = alexnet.alexnet_v2(eval_inputs, is_training=False,
 161                                      spatial_squeeze=False)
 162       self.assertListEqual(logits.get_shape().as_list(),
 163                            [eval_batch_size, 4, 7, num_classes])
 164       logits = tf.reduce_mean(input_tensor=logits, axis=[1, 2])
 165       predictions = tf.argmax(input=logits, axis=1)
 166       self.assertEquals(predictions.get_shape().as_list(), [eval_batch_size])

 168   def testForward(self):
