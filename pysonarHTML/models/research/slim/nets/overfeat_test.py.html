  26 class OverFeatTest(tf.test.TestCase):

  28   def testBuild(self):
  29     batch_size = 5
  30     height, width = 231, 231
  31     num_classes = 1000
  32     with self.test_session():
  33       inputs = tf.random.uniform((batch_size, height, width, 3))
  34       logits, _ = overfeat.overfeat(inputs, num_classes)
  35       self.assertEquals(logits.op.name, 'overfeat/fc8/squeezed')
  36       self.assertListEqual(logits.get_shape().as_list(),
  37                            [batch_size, num_classes])

 131   def testEvaluation(self):
 132     batch_size = 2
 133     height, width = 231, 231
 134     num_classes = 1000
 135     with self.test_session():
 136       eval_inputs = tf.random.uniform((batch_size, height, width, 3))
 137       logits, _ = overfeat.overfeat(eval_inputs, is_training=False)
 138       self.assertListEqual(logits.get_shape().as_list(),
 139                            [batch_size, num_classes])
 140       predictions = tf.argmax(input=logits, axis=1)
 141       self.assertListEqual(predictions.get_shape().as_list(), [batch_size])

 143   def testTrainEvalWithReuse(self):
 144     train_batch_size = 2
 145     eval_batch_size = 1
 146     train_height, train_width = 231, 231
 147     eval_height, eval_width = 281, 281
 148     num_classes = 1000
 149     with self.test_session():
 150       train_inputs = tf.random.uniform(
 151           (train_batch_size, train_height, train_width, 3))
 152       logits, _ = overfeat.overfeat(train_inputs)
 153       self.assertListEqual(logits.get_shape().as_list(),
 154                            [train_batch_size, num_classes])
 155       tf.get_variable_scope().reuse_variables()
 156       eval_inputs = tf.random.uniform(
 157           (eval_batch_size, eval_height, eval_width, 3))
 158       logits, _ = overfeat.overfeat(eval_inputs, is_training=False,
 159                                     spatial_squeeze=False)
 160       self.assertListEqual(logits.get_shape().as_list(),
 161                            [eval_batch_size, 2, 2, num_classes])
 162       logits = tf.reduce_mean(input_tensor=logits, axis=[1, 2])
 163       predictions = tf.argmax(input=logits, axis=1)
 164       self.assertEquals(predictions.get_shape().as_list(), [eval_batch_size])

 166   def testForward(self):
 167     batch_size = 1
 168     height, width = 231, 231
 169     with self.test_session() as sess:
 170       inputs = tf.random.uniform((batch_size, height, width, 3))
 171       logits, _ = overfeat.overfeat(inputs)
 172       sess.run(tf.global_variables_initializer())
 173       output = sess.run(logits)
 174       self.assertTrue(output.any())
