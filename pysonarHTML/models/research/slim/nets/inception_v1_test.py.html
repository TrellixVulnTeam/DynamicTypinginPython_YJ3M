  30   def testBuildClassificationNetwork(self):
  31     batch_size = 5
  32     height, width = 224, 224
  33     num_classes = 1000
  34 
  35     inputs = tf.random.uniform((batch_size, height, width, 3))
  36     logits, end_points = inception.inception_v1(inputs, num_classes)
  37     self.assertTrue(logits.op.name.startswith(
  38         'InceptionV1/Logits/SpatialSqueeze'))
  39     self.assertListEqual(logits.get_shape().as_list(),
  40                          [batch_size, num_classes])
  41     self.assertTrue('Predictions' in end_points)
  42     self.assertListEqual(end_points['Predictions'].get_shape().as_list(),
  43                          [batch_size, num_classes])

  57   def testBuildBaseNetwork(self):

  73   def testBuildOnlyUptoFinalEndpoint(self):
  74     batch_size = 5
  75     height, width = 224, 224
  76     endpoints = ['Conv2d_1a_7x7', 'MaxPool_2a_3x3', 'Conv2d_2b_1x1',
  77                  'Conv2d_2c_3x3', 'MaxPool_3a_3x3', 'Mixed_3b', 'Mixed_3c',
  78                  'MaxPool_4a_3x3', 'Mixed_4b', 'Mixed_4c', 'Mixed_4d',
  79                  'Mixed_4e', 'Mixed_4f', 'MaxPool_5a_2x2', 'Mixed_5b',
  80                  'Mixed_5c']
  81     for index, endpoint in enumerate(endpoints):
  82       with tf.Graph().as_default():
  83         inputs = tf.random.uniform((batch_size, height, width, 3))
  84         out_tensor, end_points = inception.inception_v1_base(
  85             inputs, final_endpoint=endpoint)
  86         self.assertTrue(out_tensor.op.name.startswith(
  87             'InceptionV1/' + endpoint))
  88         self.assertItemsEqual(endpoints[:index+1], end_points.keys())

 172   def testUnknownImageShape(self):
 173     tf.reset_default_graph()
 174     batch_size = 2
 175     height, width = 224, 224
 176     num_classes = 1000
 177     input_np = np.random.uniform(0, 1, (batch_size, height, width, 3))
 178     with self.test_session() as sess:
 179       inputs = tf.placeholder(
 180           tf.float32, shape=(batch_size, None, None, 3))
 181       logits, end_points = inception.inception_v1(inputs, num_classes)
 182       self.assertTrue(logits.op.name.startswith('InceptionV1/Logits'))
 183       self.assertListEqual(logits.get_shape().as_list(),
 184                            [batch_size, num_classes])
 185       pre_pool = end_points['Mixed_5c']
 186       feed_dict = {inputs: input_np}
 187       tf.global_variables_initializer().run()
 188       pre_pool_out = sess.run(pre_pool, feed_dict=feed_dict)
 189       self.assertListEqual(list(pre_pool_out.shape), [batch_size, 7, 7, 1024])

 211   def testUnknowBatchSize(self):
 212     batch_size = 1
 213     height, width = 224, 224
 214     num_classes = 1000
 215 
 216     inputs = tf.placeholder(tf.float32, (None, height, width, 3))
 217     logits, _ = inception.inception_v1(inputs, num_classes)
 218     self.assertTrue(logits.op.name.startswith('InceptionV1/Logits'))
 219     self.assertListEqual(logits.get_shape().as_list(),
 220                          [None, num_classes])
 221     images = tf.random.uniform((batch_size, height, width, 3))
 222 
 223     with self.test_session() as sess:
 224       sess.run(tf.global_variables_initializer())
 225       output = sess.run(logits, {inputs: images.eval()})
 226       self.assertEquals(output.shape, (batch_size, num_classes))

 281   def testBatchNormScale(self):
 282     height, width = 224, 224
 283     num_classes = 1000
 284     inputs = tf.placeholder(tf.float32, (1, height, width, 3))
 285     with slim.arg_scope(
 286         inception.inception_v1_arg_scope(batch_norm_scale=True)):
 287       inception.inception_v1(inputs, num_classes, is_training=False)
 288 
 289     gamma_names = set(
 290         v.op.name
 291         for v in tf.global_variables('.*/BatchNorm/gamma:0$'))
 292     self.assertGreater(len(gamma_names), 0)
 293     for v in tf.global_variables('.*/BatchNorm/moving_mean:0$'):
 294       self.assertIn(v.op.name[:-len('moving_mean')] + 'gamma', gamma_names)
