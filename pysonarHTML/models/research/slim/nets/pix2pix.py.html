  39 def pix2pix_arg_scope():
  40   """Returns a default argument scope for isola_net.
  41 
  42   Returns:
  43     An arg scope.
  44   """
  45   # These parameters come from the online port, which don't necessarily match
  46   # those in the paper.
  47   # TODO(nsilberman): confirm these values with Philip.
  48   instance_norm_params = {
  49       'center': True,
  50       'scale': True,
  51       'epsilon': 0.00001,
  52   }
  53 
  54   with slim.arg_scope(
  55       [slim.conv2d, slim.conv2d_transpose],
  56       normalizer_fn=slim.instance_norm,
  57       normalizer_params=instance_norm_params,
  58       weights_initializer=tf.random_normal_initializer(0, 0.02)) as sc:
  59     return sc

  62 def upsample(net, num_outputs, kernel_size, method='nn_upsample_conv'):
  63   """Upsamples the given inputs.
  64 
  65   Args:
  66     net: A `Tensor` of size [batch_size, height, width, filters].
  67     num_outputs: The number of output filters.
  68     kernel_size: A list of 2 scalars or a 1x2 `Tensor` indicating the scale,
  69       relative to the inputs, of the output dimensions. For example, if kernel
  70       size is [2, 3], then the output height and width will be twice and three
  71       times the input size.
  72     method: The upsampling method.
  73 
  74   Returns:
  75     An `Tensor` which was upsampled using the specified method.
  76 
  77   Raises:
  78     ValueError: if `method` is not recognized.
  79   """
  80   net_shape = tf.shape(input=net)
  81   height = net_shape[1]
  82   width = net_shape[2]
  83 
  84   if method == 'nn_upsample_conv':
  85     net = tf.image.resize(
  86         net, [kernel_size[0] * height, kernel_size[1] * width],
  87         method=tf.image.ResizeMethod.NEAREST_NEIGHBOR)
  88     net = slim.conv2d(net, num_outputs, [4, 4], activation_fn=None)
  89   elif method == 'conv2d_transpose':
  90     net = slim.conv2d_transpose(
  91         net, num_outputs, [4, 4], stride=kernel_size, activation_fn=None)
  92   else:
  93     raise ValueError('Unknown method: [%s]' % method)
  94 
  95   return net

  98 class Block(
  99     collections.namedtuple('Block', ['num_filters', 'decoder_keep_prob'])):
 100   """Represents a single block of encoder and decoder processing.
 101 
 102   The Image-to-Image translation paper works a bit differently than the original
 103   U-Net model. In particular, each block represents a single operation in the
 104   encoder which is concatenated with the corresponding decoder representation.
 105   A dropout layer follows the concatenation and convolution of the concatenated
 106   features.
 107   """
 108   pass

 111 def _default_generator_blocks():
 112   """Returns the default generator block definitions.
 113 
 114   Returns:
 115     A list of generator blocks.
 116   """
 117   return [
 118       Block(64, 0.5),
 119       Block(128, 0.5),
 120       Block(256, 0.5),
 121       Block(512, 0),
 122       Block(512, 0),
 123       Block(512, 0),
 124       Block(512, 0),

 128 def pix2pix_generator(net,
 129                       num_outputs,
 130                       blocks=None,
 131                       upsample_method='nn_upsample_conv',
 132                       is_training=False):  # pylint: disable=unused-argument
 133   """Defines the network architecture.
 134 
 135   Args:
 136     net: A `Tensor` of size [batch, height, width, channels]. Note that the
 137       generator currently requires square inputs (e.g. height=width).
 138     num_outputs: The number of (per-pixel) outputs.
 139     blocks: A list of generator blocks or `None` to use the default generator
 140       definition.
 141     upsample_method: The method of upsampling images, one of 'nn_upsample_conv'
 142       or 'conv2d_transpose'
 143     is_training: Whether or not we're in training or testing mode.
 144 
 145   Returns:
 146     A `Tensor` representing the model output and a dictionary of model end
 147       points.
 148 
 149   Raises:
 150     ValueError: if the input heights do not match their widths.
 151   """
 152   end_points = {}
 153 
 154   blocks = blocks or _default_generator_blocks()
 155 
 156   input_size = net.get_shape().as_list()
 157 
 158   input_size[3] = num_outputs
 159 
 160   upsample_fn = functools.partial(upsample, method=upsample_method)
 161 
 162   encoder_activations = []
 163 
 164   ###########
 165   # Encoder #
 166   ###########
 167   with tf.variable_scope('encoder'):
 168     with slim.arg_scope([slim.conv2d],
 169                         kernel_size=[4, 4],
 170                         stride=2,
 171                         activation_fn=tf.nn.leaky_relu):
 172 
 173       for block_id, block in enumerate(blocks):
 174         # No normalizer for the first encoder layers as per 'Image-to-Image',
 175         # Section 5.1.1
 176         if block_id == 0:
 177           # First layer doesn't use normalizer_fn
 178           net = slim.conv2d(net, block.num_filters, normalizer_fn=None)
 179         elif block_id < len(blocks) - 1:
 180           net = slim.conv2d(net, block.num_filters)
 181         else:
 182           # Last layer doesn't use activation_fn nor normalizer_fn
 183           net = slim.conv2d(
 184               net, block.num_filters, activation_fn=None, normalizer_fn=None)
 185 
 186         encoder_activations.append(net)
 187         end_points['encoder%d' % block_id] = net
 188 
 189   ###########
 190   # Decoder #
 191   ###########
 192   reversed_blocks = list(blocks)
 193   reversed_blocks.reverse()
 194 
 195   with tf.variable_scope('decoder'):
 196     # Dropout is used at both train and test time as per 'Image-to-Image',
 197     # Section 2.1 (last paragraph).
 198     with slim.arg_scope([slim.dropout], is_training=True):
 199 
 200       for block_id, block in enumerate(reversed_blocks):
 201         if block_id > 0:
 202           net = tf.concat([net, encoder_activations[-block_id - 1]], axis=3)
 203 
 204         # The Relu comes BEFORE the upsample op:
 205         net = tf.nn.relu(net)
 206         net = upsample_fn(net, block.num_filters, [2, 2])
 207         if block.decoder_keep_prob > 0:
 208           net = slim.dropout(net, keep_prob=block.decoder_keep_prob)
 209         end_points['decoder%d' % block_id] = net
 210 
 211   with tf.variable_scope('output'):
 212     # Explicitly set the normalizer_fn to None to override any default value
 213     # that may come from an arg_scope, such as pix2pix_arg_scope.
 214     logits = slim.conv2d(
 215         net, num_outputs, [4, 4], activation_fn=None, normalizer_fn=None)
 216     logits = tf.reshape(logits, input_size)
 217 
 218     end_points['logits'] = logits
 219     end_points['predictions'] = tf.tanh(logits)
 220 
 221   return logits, end_points

 224 def pix2pix_discriminator(net, num_filters, padding=2, pad_mode='REFLECT',
 225                           activation_fn=tf.nn.leaky_relu, is_training=False):
 226   """Creates the Image2Image Translation Discriminator.
 227 
 228   Args:
 229     net: A `Tensor` of size [batch_size, height, width, channels] representing
 230       the input.
 231     num_filters: A list of the filters in the discriminator. The length of the
 232       list determines the number of layers in the discriminator.
 233     padding: Amount of reflection padding applied before each convolution.
 234     pad_mode: mode for tf.pad, one of "CONSTANT", "REFLECT", or "SYMMETRIC".
 235     activation_fn: activation fn for slim.conv2d.
 236     is_training: Whether or not the model is training or testing.
 237 
 238   Returns:
 239     A logits `Tensor` of size [batch_size, N, N, 1] where N is the number of
 240     'patches' we're attempting to discriminate and a dictionary of model end
 241     points.
 242   """
 243   del is_training
 244   end_points = {}
 245 
 246   num_layers = len(num_filters)
 247 
 248   def padded(net, scope):
 249     if padding:
 250       with tf.variable_scope(scope):
 251         spatial_pad = tf.constant(
 252             [[0, 0], [padding, padding], [padding, padding], [0, 0]],
 253             dtype=tf.int32)
 254         return tf.pad(tensor=net, paddings=spatial_pad, mode=pad_mode)
 255     else:
 256       return net
 257 
 258   with slim.arg_scope([slim.conv2d],
 259                       kernel_size=[4, 4],
 260                       stride=2,
 261                       padding='valid',
 262                       activation_fn=activation_fn):
 263 
 264     # No normalization on the input layer.
 265     net = slim.conv2d(
 266         padded(net, 'conv0'), num_filters[0], normalizer_fn=None, scope='conv0')
 267 
 268     end_points['conv0'] = net
 269 
 270     for i in range(1, num_layers - 1):
 271       net = slim.conv2d(
 272           padded(net, 'conv%d' % i), num_filters[i], scope='conv%d' % i)
 273       end_points['conv%d' % i] = net
 274 
 275     # Stride 1 on the last layer.
 276     net = slim.conv2d(
 277         padded(net, 'conv%d' % (num_layers - 1)),
 278         num_filters[-1],
 279         stride=1,
 280         scope='conv%d' % (num_layers - 1))
 281     end_points['conv%d' % (num_layers - 1)] = net
 282 
 283     # 1-dim logits, stride 1, no activation, no normalization.
 284     logits = slim.conv2d(
 285         padded(net, 'conv%d' % num_layers),
 286         1,
 287         stride=1,
 288         activation_fn=None,
 289         normalizer_fn=None,
 290         scope='conv%d' % num_layers)
 291     end_points['logits'] = logits
 292     end_points['predictions'] = tf.sigmoid(logits)
 293   return logits, end_points
