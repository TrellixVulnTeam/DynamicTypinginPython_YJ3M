 129   def testBuildEndPointsWithDepthMultiplierLessThanOne(self):
 130     batch_size = 5
 131     height, width = 224, 224
 132     num_classes = 1000
 133 
 134     inputs = tf.random.uniform((batch_size, height, width, 3))
 135     _, end_points = inception.inception_v2(inputs, num_classes)
 136 
 137     endpoint_keys = [key for key in end_points.keys()
 138                      if key.startswith('Mixed') or key.startswith('Conv')]
 139 
 140     _, end_points_with_multiplier = inception.inception_v2(
 141         inputs, num_classes, scope='depth_multiplied_net',
 142         depth_multiplier=0.5)
 143 
 144     for key in endpoint_keys:
 145       original_depth = end_points[key].get_shape().as_list()[3]
 146       new_depth = end_points_with_multiplier[key].get_shape().as_list()[3]
 147       self.assertEqual(0.5 * original_depth, new_depth)

 180   def testBuildEndPointsWithUseSeparableConvolutionFalse(self):
 181     batch_size = 5
 182     height, width = 224, 224
 183 
 184     inputs = tf.random.uniform((batch_size, height, width, 3))
 185     _, end_points = inception.inception_v2_base(inputs)
 186 
 187     endpoint_keys = [
 188         key for key in end_points.keys()
 189         if key.startswith('Mixed') or key.startswith('Conv')
 190     ]
 191 
 192     _, end_points_with_replacement = inception.inception_v2_base(
 193         inputs, use_separable_conv=False)
 194 
 195     # The endpoint shapes must be equal to the original shape even when the
 196     # separable convolution is replaced with a normal convolution.
 197     for key in endpoint_keys:
 198       original_shape = end_points[key].get_shape().as_list()
 199       self.assertTrue(key in end_points_with_replacement)
 200       new_shape = end_points_with_replacement[key].get_shape().as_list()
 201       self.assertListEqual(original_shape, new_shape)

 203   def testBuildEndPointsNCHWDataFormat(self):
 204     batch_size = 5
 205     height, width = 224, 224
 206 
 207     inputs = tf.random.uniform((batch_size, height, width, 3))
 208     _, end_points = inception.inception_v2_base(inputs)
 209 
 210     endpoint_keys = [
 211         key for key in end_points.keys()
 212         if key.startswith('Mixed') or key.startswith('Conv')
 213     ]
 214 
 215     inputs_in_nchw = tf.random.uniform((batch_size, 3, height, width))
 216     _, end_points_with_replacement = inception.inception_v2_base(
 217         inputs_in_nchw, use_separable_conv=False, data_format='NCHW')
 218 
 219     # With the 'NCHW' data format, all endpoint activations have a transposed
 220     # shape from the original shape with the 'NHWC' layout.
 221     for key in endpoint_keys:
 222       transposed_original_shape = tf.transpose(
 223           a=end_points[key], perm=[0, 3, 1, 2]).get_shape().as_list()
 224       self.assertTrue(key in end_points_with_replacement)
 225       new_shape = end_points_with_replacement[key].get_shape().as_list()
 226       self.assertListEqual(transposed_original_shape, new_shape)

 284   def testUnknownImageShape(self):
 285     tf.reset_default_graph()
 286     batch_size = 2
 287     height, width = 224, 224
 288     num_classes = 1000
 289     input_np = np.random.uniform(0, 1, (batch_size, height, width, 3))
 290     with self.test_session() as sess:
 291       inputs = tf.placeholder(
 292           tf.float32, shape=(batch_size, None, None, 3))
 293       logits, end_points = inception.inception_v2(inputs, num_classes)
 294       self.assertTrue(logits.op.name.startswith('InceptionV2/Logits'))
 295       self.assertListEqual(logits.get_shape().as_list(),
 296                            [batch_size, num_classes])
 297       pre_pool = end_points['Mixed_5c']
 298       feed_dict = {inputs: input_np}
 299       tf.global_variables_initializer().run()
 300       pre_pool_out = sess.run(pre_pool, feed_dict=feed_dict)
 301       self.assertListEqual(list(pre_pool_out.shape), [batch_size, 7, 7, 1024])

 323   def testUnknowBatchSize(self):
 324     batch_size = 1
 325     height, width = 224, 224
 326     num_classes = 1000
 327 
 328     inputs = tf.placeholder(tf.float32, (None, height, width, 3))
 329     logits, _ = inception.inception_v2(inputs, num_classes)
 330     self.assertTrue(logits.op.name.startswith('InceptionV2/Logits'))
 331     self.assertListEqual(logits.get_shape().as_list(),
 332                          [None, num_classes])
 333     images = tf.random.uniform((batch_size, height, width, 3))
 334 
 335     with self.test_session() as sess:
 336       sess.run(tf.global_variables_initializer())
 337       output = sess.run(logits, {inputs: images.eval()})
 338       self.assertEquals(output.shape, (batch_size, num_classes))

 393   def testBatchNormScale(self):
 394     height, width = 224, 224
 395     num_classes = 1000
 396     inputs = tf.placeholder(tf.float32, (1, height, width, 3))
 397     with slim.arg_scope(
 398         inception.inception_v2_arg_scope(batch_norm_scale=True)):
 399       inception.inception_v2(inputs, num_classes, is_training=False)
 400 
 401     gamma_names = set(
 402         v.op.name
 403         for v in tf.global_variables('.*/BatchNorm/gamma:0$'))
 404     self.assertGreater(len(gamma_names), 0)
 405     for v in tf.global_variables('.*/BatchNorm/moving_mean:0$'):
 406       self.assertIn(v.op.name[:-len('moving_mean')] + 'gamma', gamma_names)
