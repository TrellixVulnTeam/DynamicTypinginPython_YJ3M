  42 def s3dg_arg_scope(weight_decay=1e-7,
  43                    batch_norm_decay=0.999,
  44                    batch_norm_epsilon=0.001):
  45   """Defines default arg_scope for S3D-G.
  46 
  47   Args:
  48     weight_decay: The weight decay to use for regularizing the model.
  49     batch_norm_decay: Decay for batch norm moving average.
  50     batch_norm_epsilon: Small float added to variance to avoid dividing by zero
  51       in batch norm.
  52 
  53   Returns:
  54     sc: An arg_scope to use for the models.
  55   """
  56   batch_norm_params = {
  57       # Decay for the moving averages.
  58       'decay': batch_norm_decay,
  59       # epsilon to prevent 0s in variance.
  60       'epsilon': batch_norm_epsilon,
  61       # Turns off fused batch norm.
  62       'fused': False,
  63       # collection containing the moving mean and moving variance.
  64       'variables_collections': {
  65           'beta': None,
  66           'gamma': None,
  67           'moving_mean': ['moving_vars'],
  68           'moving_variance': ['moving_vars'],
  69       }
  70   }
  71 
  72   with arg_scope([slim.conv3d, conv3d_spatiotemporal],
  73                  weights_regularizer=slim.l2_regularizer(weight_decay),
  74                  activation_fn=tf.nn.relu,
  75                  normalizer_fn=slim.batch_norm,
  76                  normalizer_params=batch_norm_params):
  77     with arg_scope([conv3d_spatiotemporal], separable=True) as sc:
  78       return sc

  81 def self_gating(input_tensor, scope, data_format='NDHWC'):
  82   """Feature gating as used in S3D-G.
  83 
  84   Transforms the input features by aggregating features from all
  85   spatial and temporal locations, and applying gating conditioned
  86   on the aggregated features. More details can be found at:
  87   https://arxiv.org/abs/1712.04851
  88 
  89   Args:
  90     input_tensor: A 5-D float tensor of size [batch_size, num_frames,
  91       height, width, channels].
  92     scope: scope for `variable_scope`.
  93     data_format: An optional string from: "NDHWC", "NCDHW". Defaults to "NDHWC".
  94       The data format of the input and output data. With the default format
  95       "NDHWC", the data is stored in the order of: [batch, in_depth, in_height,
  96       in_width, in_channels]. Alternatively, the format could be "NCDHW", the
  97       data storage order is:
  98       [batch, in_channels, in_depth, in_height, in_width].
  99 
 100   Returns:
 101     A tensor with the same shape as input_tensor.
 102   """
 103 
 104   index_c = data_format.index('C')
 105   index_d = data_format.index('D')
 106   index_h = data_format.index('H')
 107   index_w = data_format.index('W')
 108   input_shape = input_tensor.get_shape().as_list()
 109   t = input_shape[index_d]
 110   w = input_shape[index_w]
 111   h = input_shape[index_h]
 112   num_channels = input_shape[index_c]
 113 
 114   spatiotemporal_average = slim.avg_pool3d(
 115       input_tensor, [t, w, h],
 116       stride=1,
 117       data_format=data_format,
 118       scope=scope + '/self_gating/avg_pool3d')
 119 
 120   weights = slim.conv3d(
 121       spatiotemporal_average,
 122       num_channels, [1, 1, 1],
 123       activation_fn=None,
 124       normalizer_fn=None,
 125       biases_initializer=None,
 126       data_format=data_format,
 127       weights_initializer=trunc_normal(0.01),
 128       scope=scope + '/self_gating/transformer_W')
 129 
 130   tile_multiples = [1, t, w, h]
 131   tile_multiples.insert(index_c, 1)
 132   weights = tf.tile(weights, tile_multiples)
 133   weights = tf.nn.sigmoid(weights)
 134 
 135   return tf.multiply(weights, input_tensor)

 138 def s3dg_base(inputs,
 139               first_temporal_kernel_size=3,
 140               temporal_conv_startat='Conv2d_2c_3x3',
 141               gating_startat='Conv2d_2c_3x3',
 142               final_endpoint='Mixed_5c',
 143               min_depth=16,
 144               depth_multiplier=1.0,
 145               data_format='NDHWC',
 146               scope='InceptionV1'):
 147   """Defines the I3D/S3DG base architecture.
 148 
 149   Note that we use the names as defined in Inception V1 to facilitate checkpoint
 150   conversion from an image-trained Inception V1 checkpoint to I3D checkpoint.
 151 
 152   Args:
 153     inputs: A 5-D float tensor of size [batch_size, num_frames, height, width,
 154       channels].
 155     first_temporal_kernel_size: Specifies the temporal kernel size for the first
 156       conv3d filter. A larger value slows down the model but provides little
 157       accuracy improvement. The default is 7 in the original I3D and S3D-G but 3
 158       gives better performance. Must be set to one of 1, 3, 5 or 7.
 159     temporal_conv_startat: Specifies the first conv block to use 3D or separable
 160       3D convs rather than 2D convs (implemented as [1, k, k] 3D conv). This is
 161       used to construct the inverted pyramid models. 'Conv2d_2c_3x3' is the
 162       first valid block to use separable 3D convs. If provided block name is
 163       not present, all valid blocks will use separable 3D convs. Note that
 164       'Conv2d_1a_7x7' cannot be made into a separable 3D conv, but can be made
 165       into a 2D or 3D conv using the `first_temporal_kernel_size` option.
 166     gating_startat: Specifies the first conv block to use self gating.
 167       'Conv2d_2c_3x3' is the first valid block to use self gating. If provided
 168       block name is not present, all valid blocks will use separable 3D convs.
 169     final_endpoint: Specifies the endpoint to construct the network up to. It
 170       can be one of ['Conv2d_1a_7x7', 'MaxPool_2a_3x3', 'Conv2d_2b_1x1',
 171       'Conv2d_2c_3x3', 'MaxPool_3a_3x3', 'Mixed_3b', 'Mixed_3c',
 172       'MaxPool_4a_3x3', 'Mixed_4b', 'Mixed_4c', 'Mixed_4d', 'Mixed_4e',
 173       'Mixed_4f', 'MaxPool_5a_2x2', 'Mixed_5b', 'Mixed_5c']
 174     min_depth: Minimum depth value (number of channels) for all convolution ops.
 175       Enforced when depth_multiplier < 1, and not an active constraint when
 176       depth_multiplier >= 1.
 177     depth_multiplier: Float multiplier for the depth (number of channels)
 178       for all convolution ops. The value must be greater than zero. Typical
 179       usage will be to set this value in (0, 1) to reduce the number of
 180       parameters or computation cost of the model.
 181     data_format: An optional string from: "NDHWC", "NCDHW". Defaults to "NDHWC".
 182       The data format of the input and output data. With the default format
 183       "NDHWC", the data is stored in the order of: [batch, in_depth, in_height,
 184       in_width, in_channels]. Alternatively, the format could be "NCDHW", the
 185       data storage order is:
 186       [batch, in_channels, in_depth, in_height, in_width].
 187     scope: Optional variable_scope.
 188 
 189   Returns:
 190     A dictionary from components of the network to the corresponding activation.
 191 
 192   Raises:
 193     ValueError: if final_endpoint is not set to one of the predefined values, or
 194       if depth_multiplier <= 0.
 195   """
 196 
 197   assert data_format in ['NDHWC', 'NCDHW']
 198   end_points = {}
 199   t = 1
 200   # For inverted pyramid models, we start with gating switched off.
 201   use_gating = False
 202   self_gating_fn = None
 203   def gating_fn(inputs, scope):
 204     return self_gating(inputs, scope, data_format=data_format)
 205 
 206   if depth_multiplier <= 0:
 207     raise ValueError('depth_multiplier is not greater than zero.')
 208   depth = lambda d: max(int(d * depth_multiplier), min_depth)
 209 
 210   with tf.variable_scope(scope, 'InceptionV1', [inputs]):
 211     with arg_scope([slim.conv3d], weights_initializer=trunc_normal(0.01)):
 212       with arg_scope([slim.conv3d, slim.max_pool3d, conv3d_spatiotemporal],
 213                      stride=1,
 214                      data_format=data_format,
 215                      padding='SAME'):
 216         # batch_size x 32 x 112 x 112 x 64
 217         end_point = 'Conv2d_1a_7x7'
 218         if first_temporal_kernel_size not in [1, 3, 5, 7]:
 219           raise ValueError(
 220               'first_temporal_kernel_size can only be 1, 3, 5 or 7.')
 221         # Separable conv is slow when used at first conv layer.
 222         net = conv3d_spatiotemporal(
 223             inputs,
 224             depth(64), [first_temporal_kernel_size, 7, 7],
 225             stride=2,
 226             separable=False,
 227             scope=end_point)
 228         end_points[end_point] = net
 229         if final_endpoint == end_point:
 230           return net, end_points
 231         # batch_size x 32 x 56 x 56 x 64
 232         end_point = 'MaxPool_2a_3x3'
 233         net = slim.max_pool3d(net, [1, 3, 3], stride=[1, 2, 2], scope=end_point)
 234         end_points[end_point] = net
 235         if final_endpoint == end_point:
 236           return net, end_points
 237         # batch_size x 32 x 56 x 56 x 64
 238         end_point = 'Conv2d_2b_1x1'
 239         net = slim.conv3d(net, depth(64), [1, 1, 1], scope=end_point)
 240         end_points[end_point] = net
 241         if final_endpoint == end_point:
 242           return net, end_points
 243         # batch_size x 32 x 56 x 56 x 192
 244         end_point = 'Conv2d_2c_3x3'
 245         if temporal_conv_startat == end_point:
 246           t = 3
 247         if gating_startat == end_point:
 248           use_gating = True
 249           self_gating_fn = gating_fn
 250         net = conv3d_spatiotemporal(net, depth(192), [t, 3, 3], scope=end_point)
 251         if use_gating:
 252           net = self_gating(net, scope=end_point, data_format=data_format)
 253         end_points[end_point] = net
 254         if final_endpoint == end_point:
 255           return net, end_points
 256         # batch_size x 32 x 28 x 28 x 192
 257         end_point = 'MaxPool_3a_3x3'
 258         net = slim.max_pool3d(net, [1, 3, 3], stride=[1, 2, 2], scope=end_point)
 259         end_points[end_point] = net
 260         if final_endpoint == end_point:
 261           return net, end_points
 262 
 263         # batch_size x 32 x 28 x 28 x 256
 264         end_point = 'Mixed_3b'
 265         if temporal_conv_startat == end_point:
 266           t = 3
 267         if gating_startat == end_point:
 268           use_gating = True
 269           self_gating_fn = gating_fn
 270         net = inception_block_v1_3d(
 271             net,
 272             num_outputs_0_0a=depth(64),
 273             num_outputs_1_0a=depth(96),
 274             num_outputs_1_0b=depth(128),
 275             num_outputs_2_0a=depth(16),
 276             num_outputs_2_0b=depth(32),
 277             num_outputs_3_0b=depth(32),
 278             temporal_kernel_size=t,
 279             self_gating_fn=self_gating_fn,
 280             data_format=data_format,
 281             scope=end_point)
 282         end_points[end_point] = net
 283         if final_endpoint == end_point:
 284           return net, end_points
 285 
 286         end_point = 'Mixed_3c'
 287         if temporal_conv_startat == end_point:
 288           t = 3
 289         if gating_startat == end_point:
 290           use_gating = True
 291           self_gating_fn = gating_fn
 292         net = inception_block_v1_3d(
 293             net,
 294             num_outputs_0_0a=depth(128),
 295             num_outputs_1_0a=depth(128),
 296             num_outputs_1_0b=depth(192),
 297             num_outputs_2_0a=depth(32),
 298             num_outputs_2_0b=depth(96),
 299             num_outputs_3_0b=depth(64),
 300             temporal_kernel_size=t,
 301             self_gating_fn=self_gating_fn,
 302             data_format=data_format,
 303             scope=end_point)
 304         end_points[end_point] = net
 305         if final_endpoint == end_point:
 306           return net, end_points
 307 
 308         end_point = 'MaxPool_4a_3x3'
 309         net = slim.max_pool3d(net, [3, 3, 3], stride=[2, 2, 2], scope=end_point)
 310         end_points[end_point] = net
 311         if final_endpoint == end_point:
 312           return net, end_points
 313 
 314         # batch_size x 16 x 14 x 14 x 512
 315         end_point = 'Mixed_4b'
 316         if temporal_conv_startat == end_point:
 317           t = 3
 318         if gating_startat == end_point:
 319           use_gating = True
 320           self_gating_fn = gating_fn
 321         net = inception_block_v1_3d(
 322             net,
 323             num_outputs_0_0a=depth(192),
 324             num_outputs_1_0a=depth(96),
 325             num_outputs_1_0b=depth(208),
 326             num_outputs_2_0a=depth(16),
 327             num_outputs_2_0b=depth(48),
 328             num_outputs_3_0b=depth(64),
 329             temporal_kernel_size=t,
 330             self_gating_fn=self_gating_fn,
 331             data_format=data_format,
 332             scope=end_point)
 333         end_points[end_point] = net
 334         if final_endpoint == end_point:
 335           return net, end_points
 336 
 337         # batch_size x 16 x 14 x 14 x 512
 338         end_point = 'Mixed_4c'
 339         if temporal_conv_startat == end_point:
 340           t = 3
 341         if gating_startat == end_point:
 342           use_gating = True
 343           self_gating_fn = gating_fn
 344         net = inception_block_v1_3d(
 345             net,
 346             num_outputs_0_0a=depth(160),
 347             num_outputs_1_0a=depth(112),
 348             num_outputs_1_0b=depth(224),
 349             num_outputs_2_0a=depth(24),
 350             num_outputs_2_0b=depth(64),
 351             num_outputs_3_0b=depth(64),
 352             temporal_kernel_size=t,
 353             self_gating_fn=self_gating_fn,
 354             data_format=data_format,
 355             scope=end_point)
 356         end_points[end_point] = net
 357         if final_endpoint == end_point:
 358           return net, end_points
 359 
 360         # batch_size x 16 x 14 x 14 x 512
 361         end_point = 'Mixed_4d'
 362         if temporal_conv_startat == end_point:
 363           t = 3
 364         if gating_startat == end_point:
 365           use_gating = True
 366           self_gating_fn = gating_fn
 367         net = inception_block_v1_3d(
 368             net,
 369             num_outputs_0_0a=depth(128),
 370             num_outputs_1_0a=depth(128),
 371             num_outputs_1_0b=depth(256),
 372             num_outputs_2_0a=depth(24),
 373             num_outputs_2_0b=depth(64),
 374             num_outputs_3_0b=depth(64),
 375             temporal_kernel_size=t,
 376             self_gating_fn=self_gating_fn,
 377             data_format=data_format,
 378             scope=end_point)
 379         end_points[end_point] = net
 380         if final_endpoint == end_point:
 381           return net, end_points
 382 
 383         # batch_size x 16 x 14 x 14 x 528
 384         end_point = 'Mixed_4e'
 385         if temporal_conv_startat == end_point:
 386           t = 3
 387         if gating_startat == end_point:
 388           use_gating = True
 389           self_gating_fn = gating_fn
 390         net = inception_block_v1_3d(
 391             net,
 392             num_outputs_0_0a=depth(112),
 393             num_outputs_1_0a=depth(144),
 394             num_outputs_1_0b=depth(288),
 395             num_outputs_2_0a=depth(32),
 396             num_outputs_2_0b=depth(64),
 397             num_outputs_3_0b=depth(64),
 398             temporal_kernel_size=t,
 399             self_gating_fn=self_gating_fn,
 400             data_format=data_format,
 401             scope=end_point)
 402         end_points[end_point] = net
 403         if final_endpoint == end_point:
 404           return net, end_points
 405 
 406         # batch_size x 16 x 14 x 14 x 832
 407         end_point = 'Mixed_4f'
 408         if temporal_conv_startat == end_point:
 409           t = 3
 410         if gating_startat == end_point:
 411           use_gating = True
 412           self_gating_fn = gating_fn
 413         net = inception_block_v1_3d(
 414             net,
 415             num_outputs_0_0a=depth(256),
 416             num_outputs_1_0a=depth(160),
 417             num_outputs_1_0b=depth(320),
 418             num_outputs_2_0a=depth(32),
 419             num_outputs_2_0b=depth(128),
 420             num_outputs_3_0b=depth(128),
 421             temporal_kernel_size=t,
 422             self_gating_fn=self_gating_fn,
 423             data_format=data_format,
 424             scope=end_point)
 425         end_points[end_point] = net
 426         if final_endpoint == end_point:
 427           return net, end_points
 428 
 429         end_point = 'MaxPool_5a_2x2'
 430         net = slim.max_pool3d(net, [2, 2, 2], stride=[2, 2, 2], scope=end_point)
 431         end_points[end_point] = net
 432         if final_endpoint == end_point:
 433           return net, end_points
 434 
 435         # batch_size x 8 x 7 x 7 x 832
 436         end_point = 'Mixed_5b'
 437         if temporal_conv_startat == end_point:
 438           t = 3
 439         if gating_startat == end_point:
 440           use_gating = True
 441           self_gating_fn = gating_fn
 442         net = inception_block_v1_3d(
 443             net,
 444             num_outputs_0_0a=depth(256),
 445             num_outputs_1_0a=depth(160),
 446             num_outputs_1_0b=depth(320),
 447             num_outputs_2_0a=depth(32),
 448             num_outputs_2_0b=depth(128),
 449             num_outputs_3_0b=depth(128),
 450             temporal_kernel_size=t,
 451             self_gating_fn=self_gating_fn,
 452             data_format=data_format,
 453             scope=end_point)
 454         end_points[end_point] = net
 455         if final_endpoint == end_point:
 456           return net, end_points
 457 
 458         # batch_size x 8 x 7 x 7 x 1024
 459         end_point = 'Mixed_5c'
 460         if temporal_conv_startat == end_point:
 461           t = 3
 462         if gating_startat == end_point:
 463           use_gating = True
 464           self_gating_fn = gating_fn
 465         net = inception_block_v1_3d(
 466             net,
 467             num_outputs_0_0a=depth(384),
 468             num_outputs_1_0a=depth(192),
 469             num_outputs_1_0b=depth(384),
 470             num_outputs_2_0a=depth(48),
 471             num_outputs_2_0b=depth(128),
 472             num_outputs_3_0b=depth(128),
 473             temporal_kernel_size=t,
 474             self_gating_fn=self_gating_fn,
 475             data_format=data_format,
 476             scope=end_point)
 477         end_points[end_point] = net
 478         if final_endpoint == end_point:
 479           return net, end_points
 480     raise ValueError('Unknown final endpoint %s' % final_endpoint)

 483 def s3dg(inputs,
 484          num_classes=1000,
 485          first_temporal_kernel_size=3,
 486          temporal_conv_startat='Conv2d_2c_3x3',
 487          gating_startat='Conv2d_2c_3x3',
 488          final_endpoint='Mixed_5c',
 489          min_depth=16,
 490          depth_multiplier=1.0,
 491          dropout_keep_prob=0.8,
 492          is_training=True,
 493          prediction_fn=slim.softmax,
 494          spatial_squeeze=True,
 495          reuse=None,
 496          data_format='NDHWC',
 497          scope='InceptionV1'):
 498   """Defines the S3D-G architecture.
 499 
 500   The default image size used to train this network is 224x224.
 501 
 502   Args:
 503     inputs: A 5-D float tensor of size [batch_size, num_frames, height, width,
 504       channels].
 505     num_classes: number of predicted classes.
 506     first_temporal_kernel_size: Specifies the temporal kernel size for the first
 507       conv3d filter. A larger value slows down the model but provides little
 508       accuracy improvement. Must be set to one of 1, 3, 5 or 7.
 509     temporal_conv_startat: Specifies the first conv block to use separable 3D
 510       convs rather than 2D convs (implemented as [1, k, k] 3D conv). This is
 511       used to construct the inverted pyramid models. 'Conv2d_2c_3x3' is the
 512       first valid block to use separable 3D convs. If provided block name is
 513       not present, all valid blocks will use separable 3D convs.
 514     gating_startat: Specifies the first conv block to use self gating.
 515       'Conv2d_2c_3x3' is the first valid block to use self gating. If provided
 516       block name is not present, all valid blocks will use separable 3D convs.
 517     final_endpoint: Specifies the endpoint to construct the network up to. It
 518       can be one of ['Conv2d_1a_7x7', 'MaxPool_2a_3x3', 'Conv2d_2b_1x1',
 519       'Conv2d_2c_3x3', 'MaxPool_3a_3x3', 'Mixed_3b', 'Mixed_3c',
 520       'MaxPool_4a_3x3', 'Mixed_4b', 'Mixed_4c', 'Mixed_4d', 'Mixed_4e',
 521       'Mixed_4f', 'MaxPool_5a_2x2', 'Mixed_5b', 'Mixed_5c']
 522     min_depth: Minimum depth value (number of channels) for all convolution ops.
 523       Enforced when depth_multiplier < 1, and not an active constraint when
 524       depth_multiplier >= 1.
 525     depth_multiplier: Float multiplier for the depth (number of channels)
 526       for all convolution ops. The value must be greater than zero. Typical
 527       usage will be to set this value in (0, 1) to reduce the number of
 528       parameters or computation cost of the model.
 529     dropout_keep_prob: the percentage of activation values that are retained.
 530     is_training: whether is training or not.
 531     prediction_fn: a function to get predictions out of logits.
 532     spatial_squeeze: if True, logits is of shape is [B, C], if false logits is
 533         of shape [B, 1, 1, C], where B is batch_size and C is number of classes.
 534     reuse: whether or not the network and its variables should be reused. To be
 535       able to reuse 'scope' must be given.
 536     data_format: An optional string from: "NDHWC", "NCDHW". Defaults to "NDHWC".
 537       The data format of the input and output data. With the default format
 538       "NDHWC", the data is stored in the order of: [batch, in_depth, in_height,
 539       in_width, in_channels]. Alternatively, the format could be "NCDHW", the
 540       data storage order is:
 541       [batch, in_channels, in_depth, in_height, in_width].
 542     scope: Optional variable_scope.
 543 
 544   Returns:
 545     logits: the pre-softmax activations, a tensor of size
 546       [batch_size, num_classes]
 547     end_points: a dictionary from components of the network to the corresponding
 548       activation.
 549   """
 550   assert data_format in ['NDHWC', 'NCDHW']
 551   # Final pooling and prediction
 552   with tf.variable_scope(
 553       scope, 'InceptionV1', [inputs, num_classes], reuse=reuse) as scope:
 554     with arg_scope([slim.batch_norm, slim.dropout], is_training=is_training):
 555       net, end_points = s3dg_base(
 556           inputs,
 557           first_temporal_kernel_size=first_temporal_kernel_size,
 558           temporal_conv_startat=temporal_conv_startat,
 559           gating_startat=gating_startat,
 560           final_endpoint=final_endpoint,
 561           min_depth=min_depth,
 562           depth_multiplier=depth_multiplier,
 563           data_format=data_format,
 564           scope=scope)
 565       with tf.variable_scope('Logits'):
 566         if data_format.startswith('NC'):
 567           net = tf.transpose(a=net, perm=[0, 2, 3, 4, 1])
 568         kernel_size = i3d_utils.reduced_kernel_size_3d(net, [2, 7, 7])
 569         net = slim.avg_pool3d(
 570             net,
 571             kernel_size,
 572             stride=1,
 573             data_format='NDHWC',
 574             scope='AvgPool_0a_7x7')
 575         net = slim.dropout(net, dropout_keep_prob, scope='Dropout_0b')
 576         logits = slim.conv3d(
 577             net,
 578             num_classes, [1, 1, 1],
 579             activation_fn=None,
 580             normalizer_fn=None,
 581             data_format='NDHWC',
 582             scope='Conv2d_0c_1x1')
 583         # Temporal average pooling.
 584         logits = tf.reduce_mean(input_tensor=logits, axis=1)
 585         if spatial_squeeze:
 586           logits = tf.squeeze(logits, [1, 2], name='SpatialSqueeze')
 587 
 588         end_points['Logits'] = logits
 589         end_points['Predictions'] = prediction_fn(logits, scope='Predictions')
 590   return logits, end_points
