  27 def cyclegan_arg_scope(instance_norm_center=True,
  28                        instance_norm_scale=True,
  29                        instance_norm_epsilon=0.001,
  30                        weights_init_stddev=0.02,
  31                        weight_decay=0.0):
  32   """Returns a default argument scope for all generators and discriminators.
  33 
  34   Args:
  35     instance_norm_center: Whether instance normalization applies centering.
  36     instance_norm_scale: Whether instance normalization applies scaling.
  37     instance_norm_epsilon: Small float added to the variance in the instance
  38       normalization to avoid dividing by zero.
  39     weights_init_stddev: Standard deviation of the random values to initialize
  40       the convolution kernels with.
  41     weight_decay: Magnitude of weight decay applied to all convolution kernel
  42       variables of the generator.
  43 
  44   Returns:
  45     An arg-scope.
  46   """
  47   instance_norm_params = {
  48       'center': instance_norm_center,
  49       'scale': instance_norm_scale,
  50       'epsilon': instance_norm_epsilon,
  51   }
  52 
  53   weights_regularizer = None
  54   if weight_decay and weight_decay > 0.0:
  55     weights_regularizer = slim.l2_regularizer(weight_decay)
  56 
  57   with slim.arg_scope(
  58       [slim.conv2d],
  59       normalizer_fn=slim.instance_norm,
  60       normalizer_params=instance_norm_params,
  61       weights_initializer=tf.random_normal_initializer(
  62           0, weights_init_stddev),
  63       weights_regularizer=weights_regularizer) as sc:
  64     return sc

  67 def cyclegan_upsample(net, num_outputs, stride, method='conv2d_transpose',
  68                       pad_mode='REFLECT', align_corners=False):
  69   """Upsamples the given inputs.
  70 
  71   Args:
  72     net: A Tensor of size [batch_size, height, width, filters].
  73     num_outputs: The number of output filters.
  74     stride: A list of 2 scalars or a 1x2 Tensor indicating the scale,
  75       relative to the inputs, of the output dimensions. For example, if kernel
  76       size is [2, 3], then the output height and width will be twice and three
  77       times the input size.
  78     method: The upsampling method: 'nn_upsample_conv', 'bilinear_upsample_conv',
  79       or 'conv2d_transpose'.
  80     pad_mode: mode for tf.pad, one of "CONSTANT", "REFLECT", or "SYMMETRIC".
  81     align_corners: option for method, 'bilinear_upsample_conv'. If true, the
  82       centers of the 4 corner pixels of the input and output tensors are
  83       aligned, preserving the values at the corner pixels.
  84 
  85   Returns:
  86     A Tensor which was upsampled using the specified method.
  87 
  88   Raises:
  89     ValueError: if `method` is not recognized.
  90   """
  91   with tf.variable_scope('upconv'):
  92     net_shape = tf.shape(input=net)
  93     height = net_shape[1]
  94     width = net_shape[2]
  95 
  96     # Reflection pad by 1 in spatial dimensions (axes 1, 2 = h, w) to make a 3x3
  97     # 'valid' convolution produce an output with the same dimension as the
  98     # input.
  99     spatial_pad_1 = np.array([[0, 0], [1, 1], [1, 1], [0, 0]])
 100 
 101     if method == 'nn_upsample_conv':
 102       net = tf.image.resize(
 103           net, [stride[0] * height, stride[1] * width],
 104           method=tf.image.ResizeMethod.NEAREST_NEIGHBOR)
 105       net = tf.pad(tensor=net, paddings=spatial_pad_1, mode=pad_mode)
 106       net = slim.conv2d(net, num_outputs, kernel_size=[3, 3], padding='valid')
 107     elif method == 'bilinear_upsample_conv':
 108       net = tf.image.resize_bilinear(
 109           net, [stride[0] * height, stride[1] * width],
 110           align_corners=align_corners)
 111       net = tf.pad(tensor=net, paddings=spatial_pad_1, mode=pad_mode)
 112       net = slim.conv2d(net, num_outputs, kernel_size=[3, 3], padding='valid')
 113     elif method == 'conv2d_transpose':
 114       # This corrects 1 pixel offset for images with even width and height.
 115       # conv2d is left aligned and conv2d_transpose is right aligned for even
 116       # sized images (while doing 'SAME' padding).
 117       # Note: This doesn't reflect actual model in paper.
 118       net = slim.conv2d_transpose(
 119           net, num_outputs, kernel_size=[3, 3], stride=stride, padding='valid')
 120       net = net[:, 1:, 1:, :]
 121     else:
 122       raise ValueError('Unknown method: [%s]' % method)
 123 
 124     return net

 127 def _dynamic_or_static_shape(tensor):
 128   shape = tf.shape(input=tensor)
 129   static_shape = tensor_util.constant_value(shape)
 130   return static_shape if static_shape is not None else shape

 133 def cyclegan_generator_resnet(images,
 134                               arg_scope_fn=cyclegan_arg_scope,
 135                               num_resnet_blocks=6,
 136                               num_filters=64,
 137                               upsample_fn=cyclegan_upsample,
 138                               kernel_size=3,
 139                               tanh_linear_slope=0.0,
 140                               is_training=False):
 141   """Defines the cyclegan resnet network architecture.
 142 
 143   As closely as possible following
 144   https://github.com/junyanz/CycleGAN/blob/master/models/architectures.lua#L232
 145 
 146   FYI: This network requires input height and width to be divisible by 4 in
 147   order to generate an output with shape equal to input shape. Assertions will
 148   catch this if input dimensions are known at graph construction time, but
 149   there's no protection if unknown at graph construction time (you'll see an
 150   error).
 151 
 152   Args:
 153     images: Input image tensor of shape [batch_size, h, w, 3].
 154     arg_scope_fn: Function to create the global arg_scope for the network.
 155     num_resnet_blocks: Number of ResNet blocks in the middle of the generator.
 156     num_filters: Number of filters of the first hidden layer.
 157     upsample_fn: Upsampling function for the decoder part of the generator.
 158     kernel_size: Size w or list/tuple [h, w] of the filter kernels for all inner
 159       layers.
 160     tanh_linear_slope: Slope of the linear function to add to the tanh over the
 161       logits.
 162     is_training: Whether the network is created in training mode or inference
 163       only mode. Not actually needed, just for compliance with other generator
 164       network functions.
 165 
 166   Returns:
 167     A `Tensor` representing the model output and a dictionary of model end
 168       points.
 169 
 170   Raises:
 171     ValueError: If the input height or width is known at graph construction time
 172       and not a multiple of 4.
 173   """
 174   # Neither dropout nor batch norm -> dont need is_training
 175   del is_training
 176 
 177   end_points = {}
 178 
 179   input_size = images.shape.as_list()
 180   height, width = input_size[1], input_size[2]
 181   if height and height % 4 != 0:
 182     raise ValueError('The input height must be a multiple of 4.')
 183   if width and width % 4 != 0:
 184     raise ValueError('The input width must be a multiple of 4.')
 185   num_outputs = input_size[3]
 186 
 187   if not isinstance(kernel_size, (list, tuple)):
 188     kernel_size = [kernel_size, kernel_size]
 189 
 190   kernel_height = kernel_size[0]
 191   kernel_width = kernel_size[1]
 192   pad_top = (kernel_height - 1) // 2
 193   pad_bottom = kernel_height // 2
 194   pad_left = (kernel_width - 1) // 2
 195   pad_right = kernel_width // 2
 196   paddings = np.array(
 197       [[0, 0], [pad_top, pad_bottom], [pad_left, pad_right], [0, 0]],
 198       dtype=np.int32)
 199   spatial_pad_3 = np.array([[0, 0], [3, 3], [3, 3], [0, 0]])
 200 
 201   with slim.arg_scope(arg_scope_fn()):
 202 
 203     ###########
 204     # Encoder #
 205     ###########
 206     with tf.variable_scope('input'):
 207       # 7x7 input stage
 208       net = tf.pad(tensor=images, paddings=spatial_pad_3, mode='REFLECT')
 209       net = slim.conv2d(net, num_filters, kernel_size=[7, 7], padding='VALID')
 210       end_points['encoder_0'] = net
 211 
 212     with tf.variable_scope('encoder'):
 213       with slim.arg_scope([slim.conv2d],
 214                           kernel_size=kernel_size,
 215                           stride=2,
 216                           activation_fn=tf.nn.relu,
 217                           padding='VALID'):
 218 
 219         net = tf.pad(tensor=net, paddings=paddings, mode='REFLECT')
 220         net = slim.conv2d(net, num_filters * 2)
 221         end_points['encoder_1'] = net
 222         net = tf.pad(tensor=net, paddings=paddings, mode='REFLECT')
 223         net = slim.conv2d(net, num_filters * 4)
 224         end_points['encoder_2'] = net
 225 
 226     ###################
 227     # Residual Blocks #
 228     ###################
 229     with tf.variable_scope('residual_blocks'):
 230       with slim.arg_scope([slim.conv2d],
 231                           kernel_size=kernel_size,
 232                           stride=1,
 233                           activation_fn=tf.nn.relu,
 234                           padding='VALID'):
 235         for block_id in xrange(num_resnet_blocks):
 236           with tf.variable_scope('block_{}'.format(block_id)):
 237             res_net = tf.pad(tensor=net, paddings=paddings, mode='REFLECT')
 238             res_net = slim.conv2d(res_net, num_filters * 4)
 239             res_net = tf.pad(tensor=res_net, paddings=paddings, mode='REFLECT')
 240             res_net = slim.conv2d(res_net, num_filters * 4, activation_fn=None)
 241             net += res_net
 242 
 243             end_points['resnet_block_%d' % block_id] = net
 244 
 245     ###########
 246     # Decoder #
 247     ###########
 248     with tf.variable_scope('decoder'):
 249 
 250       with slim.arg_scope([slim.conv2d],
 251                           kernel_size=kernel_size,
 252                           stride=1,
 253                           activation_fn=tf.nn.relu):
 254 
 255         with tf.variable_scope('decoder1'):
 256           net = upsample_fn(net, num_outputs=num_filters * 2, stride=[2, 2])
 257         end_points['decoder1'] = net
 258 
 259         with tf.variable_scope('decoder2'):
 260           net = upsample_fn(net, num_outputs=num_filters, stride=[2, 2])
 261         end_points['decoder2'] = net
 262 
 263     with tf.variable_scope('output'):
 264       net = tf.pad(tensor=net, paddings=spatial_pad_3, mode='REFLECT')
 265       logits = slim.conv2d(
 266           net,
 267           num_outputs, [7, 7],
 268           activation_fn=None,
 269           normalizer_fn=None,
 270           padding='valid')
 271       logits = tf.reshape(logits, _dynamic_or_static_shape(images))
 272 
 273       end_points['logits'] = logits
 274       end_points['predictions'] = tf.tanh(logits) + logits * tanh_linear_slope
 275 
 276   return end_points['predictions'], end_points
