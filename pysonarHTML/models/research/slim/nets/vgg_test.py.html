 159   def testTrainEvalWithReuse(self):
 160     train_batch_size = 2
 161     eval_batch_size = 1
 162     train_height, train_width = 224, 224
 163     eval_height, eval_width = 256, 256
 164     num_classes = 1000
 165     with self.test_session():
 166       train_inputs = tf.random.uniform(
 167           (train_batch_size, train_height, train_width, 3))
 168       logits, _ = vgg.vgg_a(train_inputs)
 169       self.assertListEqual(logits.get_shape().as_list(),
 170                            [train_batch_size, num_classes])
 171       tf.get_variable_scope().reuse_variables()
 172       eval_inputs = tf.random.uniform(
 173           (eval_batch_size, eval_height, eval_width, 3))
 174       logits, _ = vgg.vgg_a(eval_inputs, is_training=False,
 175                             spatial_squeeze=False)
 176       self.assertListEqual(logits.get_shape().as_list(),
 177                            [eval_batch_size, 2, 2, num_classes])
 178       logits = tf.reduce_mean(input_tensor=logits, axis=[1, 2])
 179       predictions = tf.argmax(input=logits, axis=1)
 180       self.assertEquals(predictions.get_shape().as_list(), [eval_batch_size])
 181 
 182   def testForward(self):
 183     batch_size = 1
 184     height, width = 224, 224
 185     with self.test_session() as sess:
 186       inputs = tf.random.uniform((batch_size, height, width, 3))
 187       logits, _ = vgg.vgg_a(inputs)
 188       sess.run(tf.global_variables_initializer())
 189       output = sess.run(logits)
 190       self.assertTrue(output.any())

 548   def testTrainEvalWithReuse(self):
 549     train_batch_size = 2
 550     eval_batch_size = 1
 551     train_height, train_width = 224, 224
 552     eval_height, eval_width = 256, 256
 553     num_classes = 1000
 554     with self.test_session():
 555       train_inputs = tf.random.uniform(
 556           (train_batch_size, train_height, train_width, 3))
 557       logits, _ = vgg.vgg_19(train_inputs)
 558       self.assertListEqual(logits.get_shape().as_list(),
 559                            [train_batch_size, num_classes])
 560       tf.get_variable_scope().reuse_variables()
 561       eval_inputs = tf.random.uniform(
 562           (eval_batch_size, eval_height, eval_width, 3))
 563       logits, _ = vgg.vgg_19(eval_inputs, is_training=False,
 564                              spatial_squeeze=False)
 565       self.assertListEqual(logits.get_shape().as_list(),
 566                            [eval_batch_size, 2, 2, num_classes])
 567       logits = tf.reduce_mean(input_tensor=logits, axis=[1, 2])
 568       predictions = tf.argmax(input=logits, axis=1)
 569       self.assertEquals(predictions.get_shape().as_list(), [eval_batch_size])
