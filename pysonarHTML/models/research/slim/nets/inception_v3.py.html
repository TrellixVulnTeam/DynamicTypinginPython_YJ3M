  31 def inception_v3_base(inputs,
  32                       final_endpoint='Mixed_7c',
  33                       min_depth=16,
  34                       depth_multiplier=1.0,
  35                       scope=None):
  36   """Inception model from http://arxiv.org/abs/1512.00567.
  37 
  38   Constructs an Inception v3 network from inputs to the given final endpoint.
  39   This method can construct the network up to the final inception block
  40   Mixed_7c.
  41 
  42   Note that the names of the layers in the paper do not correspond to the names
  43   of the endpoints registered by this function although they build the same
  44   network.
  45 
  46   Here is a mapping from the old_names to the new names:
  47   Old name          | New name
  48   =======================================
  49   conv0             | Conv2d_1a_3x3
  50   conv1             | Conv2d_2a_3x3
  51   conv2             | Conv2d_2b_3x3
  52   pool1             | MaxPool_3a_3x3
  53   conv3             | Conv2d_3b_1x1
  54   conv4             | Conv2d_4a_3x3
  55   pool2             | MaxPool_5a_3x3
  56   mixed_35x35x256a  | Mixed_5b
  57   mixed_35x35x288a  | Mixed_5c
  58   mixed_35x35x288b  | Mixed_5d
  59   mixed_17x17x768a  | Mixed_6a
  60   mixed_17x17x768b  | Mixed_6b
  61   mixed_17x17x768c  | Mixed_6c
  62   mixed_17x17x768d  | Mixed_6d
  63   mixed_17x17x768e  | Mixed_6e
  64   mixed_8x8x1280a   | Mixed_7a
  65   mixed_8x8x2048a   | Mixed_7b
  66   mixed_8x8x2048b   | Mixed_7c
  67 
  68   Args:
  69     inputs: a tensor of size [batch_size, height, width, channels].
  70     final_endpoint: specifies the endpoint to construct the network up to. It
  71       can be one of ['Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3',
  72       'MaxPool_3a_3x3', 'Conv2d_3b_1x1', 'Conv2d_4a_3x3', 'MaxPool_5a_3x3',
  73       'Mixed_5b', 'Mixed_5c', 'Mixed_5d', 'Mixed_6a', 'Mixed_6b', 'Mixed_6c',
  74       'Mixed_6d', 'Mixed_6e', 'Mixed_7a', 'Mixed_7b', 'Mixed_7c'].
  75     min_depth: Minimum depth value (number of channels) for all convolution ops.
  76       Enforced when depth_multiplier < 1, and not an active constraint when
  77       depth_multiplier >= 1.
  78     depth_multiplier: Float multiplier for the depth (number of channels)
  79       for all convolution ops. The value must be greater than zero. Typical
  80       usage will be to set this value in (0, 1) to reduce the number of
  81       parameters or computation cost of the model.
  82     scope: Optional variable_scope.
  83 
  84   Returns:
  85     tensor_out: output tensor corresponding to the final_endpoint.
  86     end_points: a set of activations for external use, for example summaries or
  87                 losses.
  88 
  89   Raises:
  90     ValueError: if final_endpoint is not set to one of the predefined values,
  91                 or depth_multiplier <= 0
  92   """
  93   # end_points will collect relevant activations for external use, for example
  94   # summaries or losses.
  95   end_points = {}
  96 
  97   if depth_multiplier <= 0:
  98     raise ValueError('depth_multiplier is not greater than zero.')
  99   depth = lambda d: max(int(d * depth_multiplier), min_depth)
 100 
 101   with tf.variable_scope(scope, 'InceptionV3', [inputs]):
 102     with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 103                         stride=1, padding='VALID'):
 104       # 299 x 299 x 3
 105       end_point = 'Conv2d_1a_3x3'
 106       net = slim.conv2d(inputs, depth(32), [3, 3], stride=2, scope=end_point)
 107       end_points[end_point] = net
 108       if end_point == final_endpoint: return net, end_points
 109       # 149 x 149 x 32
 110       end_point = 'Conv2d_2a_3x3'
 111       net = slim.conv2d(net, depth(32), [3, 3], scope=end_point)
 112       end_points[end_point] = net
 113       if end_point == final_endpoint: return net, end_points
 114       # 147 x 147 x 32
 115       end_point = 'Conv2d_2b_3x3'
 116       net = slim.conv2d(net, depth(64), [3, 3], padding='SAME', scope=end_point)
 117       end_points[end_point] = net
 118       if end_point == final_endpoint: return net, end_points
 119       # 147 x 147 x 64
 120       end_point = 'MaxPool_3a_3x3'
 121       net = slim.max_pool2d(net, [3, 3], stride=2, scope=end_point)
 122       end_points[end_point] = net
 123       if end_point == final_endpoint: return net, end_points
 124       # 73 x 73 x 64
 125       end_point = 'Conv2d_3b_1x1'
 126       net = slim.conv2d(net, depth(80), [1, 1], scope=end_point)
 127       end_points[end_point] = net
 128       if end_point == final_endpoint: return net, end_points
 129       # 73 x 73 x 80.
 130       end_point = 'Conv2d_4a_3x3'
 131       net = slim.conv2d(net, depth(192), [3, 3], scope=end_point)
 132       end_points[end_point] = net
 133       if end_point == final_endpoint: return net, end_points
 134       # 71 x 71 x 192.
 135       end_point = 'MaxPool_5a_3x3'
 136       net = slim.max_pool2d(net, [3, 3], stride=2, scope=end_point)
 137       end_points[end_point] = net
 138       if end_point == final_endpoint: return net, end_points
 139       # 35 x 35 x 192.
 140 
 141     # Inception blocks
 142     with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 143                         stride=1, padding='SAME'):
 144       # mixed: 35 x 35 x 256.
 145       end_point = 'Mixed_5b'
 146       with tf.variable_scope(end_point):
 147         with tf.variable_scope('Branch_0'):
 148           branch_0 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 149         with tf.variable_scope('Branch_1'):
 150           branch_1 = slim.conv2d(net, depth(48), [1, 1], scope='Conv2d_0a_1x1')
 151           branch_1 = slim.conv2d(branch_1, depth(64), [5, 5],
 152                                  scope='Conv2d_0b_5x5')
 153         with tf.variable_scope('Branch_2'):
 154           branch_2 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 155           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 156                                  scope='Conv2d_0b_3x3')
 157           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 158                                  scope='Conv2d_0c_3x3')
 159         with tf.variable_scope('Branch_3'):
 160           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 161           branch_3 = slim.conv2d(branch_3, depth(32), [1, 1],
 162                                  scope='Conv2d_0b_1x1')
 163         net = tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 164       end_points[end_point] = net
 165       if end_point == final_endpoint: return net, end_points
 166 
 167       # mixed_1: 35 x 35 x 288.
 168       end_point = 'Mixed_5c'
 169       with tf.variable_scope(end_point):
 170         with tf.variable_scope('Branch_0'):
 171           branch_0 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 172         with tf.variable_scope('Branch_1'):
 173           branch_1 = slim.conv2d(net, depth(48), [1, 1], scope='Conv2d_0b_1x1')
 174           branch_1 = slim.conv2d(branch_1, depth(64), [5, 5],
 175                                  scope='Conv_1_0c_5x5')
 176         with tf.variable_scope('Branch_2'):
 177           branch_2 = slim.conv2d(net, depth(64), [1, 1],
 178                                  scope='Conv2d_0a_1x1')
 179           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 180                                  scope='Conv2d_0b_3x3')
 181           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 182                                  scope='Conv2d_0c_3x3')
 183         with tf.variable_scope('Branch_3'):
 184           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 185           branch_3 = slim.conv2d(branch_3, depth(64), [1, 1],
 186                                  scope='Conv2d_0b_1x1')
 187         net = tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 188       end_points[end_point] = net
 189       if end_point == final_endpoint: return net, end_points
 190 
 191       # mixed_2: 35 x 35 x 288.
 192       end_point = 'Mixed_5d'
 193       with tf.variable_scope(end_point):
 194         with tf.variable_scope('Branch_0'):
 195           branch_0 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 196         with tf.variable_scope('Branch_1'):
 197           branch_1 = slim.conv2d(net, depth(48), [1, 1], scope='Conv2d_0a_1x1')
 198           branch_1 = slim.conv2d(branch_1, depth(64), [5, 5],
 199                                  scope='Conv2d_0b_5x5')
 200         with tf.variable_scope('Branch_2'):
 201           branch_2 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 202           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 203                                  scope='Conv2d_0b_3x3')
 204           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 205                                  scope='Conv2d_0c_3x3')
 206         with tf.variable_scope('Branch_3'):
 207           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 208           branch_3 = slim.conv2d(branch_3, depth(64), [1, 1],
 209                                  scope='Conv2d_0b_1x1')
 210         net = tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 211       end_points[end_point] = net
 212       if end_point == final_endpoint: return net, end_points
 213 
 214       # mixed_3: 17 x 17 x 768.
 215       end_point = 'Mixed_6a'
 216       with tf.variable_scope(end_point):
 217         with tf.variable_scope('Branch_0'):
 218           branch_0 = slim.conv2d(net, depth(384), [3, 3], stride=2,
 219                                  padding='VALID', scope='Conv2d_1a_1x1')
 220         with tf.variable_scope('Branch_1'):
 221           branch_1 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 222           branch_1 = slim.conv2d(branch_1, depth(96), [3, 3],
 223                                  scope='Conv2d_0b_3x3')
 224           branch_1 = slim.conv2d(branch_1, depth(96), [3, 3], stride=2,
 225                                  padding='VALID', scope='Conv2d_1a_1x1')
 226         with tf.variable_scope('Branch_2'):
 227           branch_2 = slim.max_pool2d(net, [3, 3], stride=2, padding='VALID',
 228                                      scope='MaxPool_1a_3x3')
 229         net = tf.concat(axis=3, values=[branch_0, branch_1, branch_2])
 230       end_points[end_point] = net
 231       if end_point == final_endpoint: return net, end_points
 232 
 233       # mixed4: 17 x 17 x 768.
 234       end_point = 'Mixed_6b'
 235       with tf.variable_scope(end_point):
 236         with tf.variable_scope('Branch_0'):
 237           branch_0 = slim.conv2d(net, depth(192), [1, 1], scope='Conv2d_0a_1x1')
 238         with tf.variable_scope('Branch_1'):
 239           branch_1 = slim.conv2d(net, depth(128), [1, 1], scope='Conv2d_0a_1x1')
 240           branch_1 = slim.conv2d(branch_1, depth(128), [1, 7],
 241                                  scope='Conv2d_0b_1x7')
 242           branch_1 = slim.conv2d(branch_1, depth(192), [7, 1],
 243                                  scope='Conv2d_0c_7x1')
 244         with tf.variable_scope('Branch_2'):
 245           branch_2 = slim.conv2d(net, depth(128), [1, 1], scope='Conv2d_0a_1x1')
 246           branch_2 = slim.conv2d(branch_2, depth(128), [7, 1],
 247                                  scope='Conv2d_0b_7x1')
 248           branch_2 = slim.conv2d(branch_2, depth(128), [1, 7],
 249                                  scope='Conv2d_0c_1x7')
 250           branch_2 = slim.conv2d(branch_2, depth(128), [7, 1],
 251                                  scope='Conv2d_0d_7x1')
 252           branch_2 = slim.conv2d(branch_2, depth(192), [1, 7],
 253                                  scope='Conv2d_0e_1x7')
 254         with tf.variable_scope('Branch_3'):
 255           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 256           branch_3 = slim.conv2d(branch_3, depth(192), [1, 1],
 257                                  scope='Conv2d_0b_1x1')
 258         net = tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 259       end_points[end_point] = net
 260       if end_point == final_endpoint: return net, end_points
 261 
 262       # mixed_5: 17 x 17 x 768.
 263       end_point = 'Mixed_6c'
 264       with tf.variable_scope(end_point):
 265         with tf.variable_scope('Branch_0'):
 266           branch_0 = slim.conv2d(net, depth(192), [1, 1], scope='Conv2d_0a_1x1')
 267         with tf.variable_scope('Branch_1'):
 268           branch_1 = slim.conv2d(net, depth(160), [1, 1], scope='Conv2d_0a_1x1')
 269           branch_1 = slim.conv2d(branch_1, depth(160), [1, 7],
 270                                  scope='Conv2d_0b_1x7')
 271           branch_1 = slim.conv2d(branch_1, depth(192), [7, 1],
 272                                  scope='Conv2d_0c_7x1')
 273         with tf.variable_scope('Branch_2'):
 274           branch_2 = slim.conv2d(net, depth(160), [1, 1], scope='Conv2d_0a_1x1')
 275           branch_2 = slim.conv2d(branch_2, depth(160), [7, 1],
 276                                  scope='Conv2d_0b_7x1')
 277           branch_2 = slim.conv2d(branch_2, depth(160), [1, 7],
 278                                  scope='Conv2d_0c_1x7')
 279           branch_2 = slim.conv2d(branch_2, depth(160), [7, 1],
 280                                  scope='Conv2d_0d_7x1')
 281           branch_2 = slim.conv2d(branch_2, depth(192), [1, 7],
 282                                  scope='Conv2d_0e_1x7')
 283         with tf.variable_scope('Branch_3'):
 284           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 285           branch_3 = slim.conv2d(branch_3, depth(192), [1, 1],
 286                                  scope='Conv2d_0b_1x1')
 287         net = tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 288       end_points[end_point] = net
 289       if end_point == final_endpoint: return net, end_points
 290       # mixed_6: 17 x 17 x 768.
 291       end_point = 'Mixed_6d'
 292       with tf.variable_scope(end_point):
 293         with tf.variable_scope('Branch_0'):
 294           branch_0 = slim.conv2d(net, depth(192), [1, 1], scope='Conv2d_0a_1x1')
 295         with tf.variable_scope('Branch_1'):
 296           branch_1 = slim.conv2d(net, depth(160), [1, 1], scope='Conv2d_0a_1x1')
 297           branch_1 = slim.conv2d(branch_1, depth(160), [1, 7],
 298                                  scope='Conv2d_0b_1x7')
 299           branch_1 = slim.conv2d(branch_1, depth(192), [7, 1],
 300                                  scope='Conv2d_0c_7x1')
 301         with tf.variable_scope('Branch_2'):
 302           branch_2 = slim.conv2d(net, depth(160), [1, 1], scope='Conv2d_0a_1x1')
 303           branch_2 = slim.conv2d(branch_2, depth(160), [7, 1],
 304                                  scope='Conv2d_0b_7x1')

 403               slim.conv2d(branch_1, depth(384), [3, 1], scope='Conv2d_0c_3x1')])
 404         with tf.variable_scope('Branch_2'):
 405           branch_2 = slim.conv2d(net, depth(448), [1, 1], scope='Conv2d_0a_1x1')
 406           branch_2 = slim.conv2d(
 407               branch_2, depth(384), [3, 3], scope='Conv2d_0b_3x3')
 408           branch_2 = tf.concat(axis=3, values=[
 409               slim.conv2d(branch_2, depth(384), [1, 3], scope='Conv2d_0c_1x3'),
 410               slim.conv2d(branch_2, depth(384), [3, 1], scope='Conv2d_0d_3x1')])
 411         with tf.variable_scope('Branch_3'):
 412           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 413           branch_3 = slim.conv2d(
 414               branch_3, depth(192), [1, 1], scope='Conv2d_0b_1x1')
 415         net = tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])
 416       end_points[end_point] = net
 417       if end_point == final_endpoint: return net, end_points
 418     raise ValueError('Unknown final endpoint %s' % final_endpoint)

 421 def inception_v3(inputs,
 422                  num_classes=1000,
 423                  is_training=True,
 424                  dropout_keep_prob=0.8,
 425                  min_depth=16,
 426                  depth_multiplier=1.0,
 427                  prediction_fn=slim.softmax,
 428                  spatial_squeeze=True,
 429                  reuse=None,
 430                  create_aux_logits=True,
 431                  scope='InceptionV3',
 432                  global_pool=False):
 433   """Inception model from http://arxiv.org/abs/1512.00567.
 434 
 435   "Rethinking the Inception Architecture for Computer Vision"
 436 
 437   Christian Szegedy, Vincent Vanhoucke, Sergey Ioffe, Jonathon Shlens,
 438   Zbigniew Wojna.
 439 
 440   With the default arguments this method constructs the exact model defined in
 441   the paper. However, one can experiment with variations of the inception_v3
 442   network by changing arguments dropout_keep_prob, min_depth and
 443   depth_multiplier.
 444 
 445   The default image size used to train this network is 299x299.
 446 
 447   Args:
 448     inputs: a tensor of size [batch_size, height, width, channels].
 449     num_classes: number of predicted classes. If 0 or None, the logits layer
 450       is omitted and the input features to the logits layer (before dropout)
 451       are returned instead.
 452     is_training: whether is training or not.
 453     dropout_keep_prob: the percentage of activation values that are retained.
 454     min_depth: Minimum depth value (number of channels) for all convolution ops.
 455       Enforced when depth_multiplier < 1, and not an active constraint when
 456       depth_multiplier >= 1.
 457     depth_multiplier: Float multiplier for the depth (number of channels)
 458       for all convolution ops. The value must be greater than zero. Typical
 459       usage will be to set this value in (0, 1) to reduce the number of
 460       parameters or computation cost of the model.
 461     prediction_fn: a function to get predictions out of logits.
 462     spatial_squeeze: if True, logits is of shape [B, C], if false logits is of
 463         shape [B, 1, 1, C], where B is batch_size and C is number of classes.
 464     reuse: whether or not the network and its variables should be reused. To be
 465       able to reuse 'scope' must be given.
 466     create_aux_logits: Whether to create the auxiliary logits.
 467     scope: Optional variable_scope.
 468     global_pool: Optional boolean flag to control the avgpooling before the
 469       logits layer. If false or unset, pooling is done with a fixed window
 470       that reduces default-sized inputs to 1x1, while larger inputs lead to
 471       larger outputs. If true, any input size is pooled down to 1x1.
 472 
 473   Returns:
 474     net: a Tensor with the logits (pre-softmax activations) if num_classes
 475       is a non-zero integer, or the non-dropped-out input to the logits layer
 476       if num_classes is 0 or None.
 477     end_points: a dictionary from components of the network to the corresponding
 478       activation.
 479 
 480   Raises:
 481     ValueError: if 'depth_multiplier' is less than or equal to zero.
 482   """
 483   if depth_multiplier <= 0:
 484     raise ValueError('depth_multiplier is not greater than zero.')
 485   depth = lambda d: max(int(d * depth_multiplier), min_depth)
 486 
 487   with tf.variable_scope(
 488       scope, 'InceptionV3', [inputs], reuse=reuse) as scope:
 489     with slim.arg_scope([slim.batch_norm, slim.dropout],
 490                         is_training=is_training):
 491       net, end_points = inception_v3_base(
 492           inputs, scope=scope, min_depth=min_depth,
 493           depth_multiplier=depth_multiplier)
 494 
 495       # Auxiliary Head logits
 496       if create_aux_logits and num_classes:
 497         with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 498                             stride=1, padding='SAME'):
 499           aux_logits = end_points['Mixed_6e']
 500           with tf.variable_scope('AuxLogits'):
 501             aux_logits = slim.avg_pool2d(
 502                 aux_logits, [5, 5], stride=3, padding='VALID',
 503                 scope='AvgPool_1a_5x5')
 504             aux_logits = slim.conv2d(aux_logits, depth(128), [1, 1],
 505                                      scope='Conv2d_1b_1x1')
 506 
 507             # Shape of feature map before the final layer.
 508             kernel_size = _reduced_kernel_size_for_small_input(
 509                 aux_logits, [5, 5])
 510             aux_logits = slim.conv2d(
 511                 aux_logits, depth(768), kernel_size,
 512                 weights_initializer=trunc_normal(0.01),
 513                 padding='VALID', scope='Conv2d_2a_{}x{}'.format(*kernel_size))
 514             aux_logits = slim.conv2d(
 515                 aux_logits, num_classes, [1, 1], activation_fn=None,
 516                 normalizer_fn=None, weights_initializer=trunc_normal(0.001),
 517                 scope='Conv2d_2b_1x1')
 518             if spatial_squeeze:
 519               aux_logits = tf.squeeze(aux_logits, [1, 2], name='SpatialSqueeze')
 520             end_points['AuxLogits'] = aux_logits
 521 
 522       # Final pooling and prediction
 523       with tf.variable_scope('Logits'):
 524         if global_pool:
 525           # Global average pooling.
 526           net = tf.reduce_mean(
 527               input_tensor=net, axis=[1, 2], keepdims=True, name='GlobalPool')
 528           end_points['global_pool'] = net
 529         else:
 530           # Pooling with a fixed kernel size.
 531           kernel_size = _reduced_kernel_size_for_small_input(net, [8, 8])
 532           net = slim.avg_pool2d(net, kernel_size, padding='VALID',
 533                                 scope='AvgPool_1a_{}x{}'.format(*kernel_size))
 534           end_points['AvgPool_1a'] = net
 535         if not num_classes:
 536           return net, end_points
 537         # 1 x 1 x 2048
 538         net = slim.dropout(net, keep_prob=dropout_keep_prob, scope='Dropout_1b')
 539         end_points['PreLogits'] = net
 540         # 2048
 541         logits = slim.conv2d(net, num_classes, [1, 1], activation_fn=None,
 542                              normalizer_fn=None, scope='Conv2d_1c_1x1')
 543         if spatial_squeeze:
 544           logits = tf.squeeze(logits, [1, 2], name='SpatialSqueeze')
 545         # 1000
 546       end_points['Logits'] = logits
 547       end_points['Predictions'] = prediction_fn(logits, scope='Predictions')
 548   return logits, end_points
 549 inception_v3.default_image_size = 299

 552 def _reduced_kernel_size_for_small_input(input_tensor, kernel_size):
 553   """Define kernel size which is automatically reduced for small input.
 554 
 555   If the shape of the input images is unknown at graph construction time this
 556   function assumes that the input images are is large enough.
 557 
 558   Args:
 559     input_tensor: input tensor of size [batch_size, height, width, channels].
 560     kernel_size: desired kernel size of length 2: [kernel_height, kernel_width]
 561 
 562   Returns:
 563     a tensor with the kernel size.
 564 
 565   TODO(jrru): Make this function work with unknown shapes. Theoretically, this
 566   can be done with the code below. Problems are two-fold: (1) If the shape was
 567   known, it will be lost. (2) inception.slim.ops._two_element_tuple cannot
 568   handle tensors that define the kernel size.
 569       shape = tf.shape(input_tensor)
 570       return = tf.stack([tf.minimum(shape[1], kernel_size[0]),
 571                          tf.minimum(shape[2], kernel_size[1])])
 572 
 573   """
 574   shape = input_tensor.get_shape().as_list()
 575   if shape[1] is None or shape[2] is None:
 576     kernel_size_out = kernel_size
 577   else:
 578     kernel_size_out = [min(shape[1], kernel_size[0]),
 579                        min(shape[2], kernel_size[1])]
 580   return kernel_size_out
