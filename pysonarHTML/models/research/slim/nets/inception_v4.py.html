  33 def block_inception_a(inputs, scope=None, reuse=None):
  34   """Builds Inception-A block for Inception v4 network."""
  35   # By default use stride=1 and SAME padding
  36   with slim.arg_scope([slim.conv2d, slim.avg_pool2d, slim.max_pool2d],
  37                       stride=1, padding='SAME'):
  38     with tf.variable_scope(
  39         scope, 'BlockInceptionA', [inputs], reuse=reuse):
  40       with tf.variable_scope('Branch_0'):
  41         branch_0 = slim.conv2d(inputs, 96, [1, 1], scope='Conv2d_0a_1x1')
  42       with tf.variable_scope('Branch_1'):
  43         branch_1 = slim.conv2d(inputs, 64, [1, 1], scope='Conv2d_0a_1x1')
  44         branch_1 = slim.conv2d(branch_1, 96, [3, 3], scope='Conv2d_0b_3x3')
  45       with tf.variable_scope('Branch_2'):
  46         branch_2 = slim.conv2d(inputs, 64, [1, 1], scope='Conv2d_0a_1x1')
  47         branch_2 = slim.conv2d(branch_2, 96, [3, 3], scope='Conv2d_0b_3x3')
  48         branch_2 = slim.conv2d(branch_2, 96, [3, 3], scope='Conv2d_0c_3x3')
  49       with tf.variable_scope('Branch_3'):
  50         branch_3 = slim.avg_pool2d(inputs, [3, 3], scope='AvgPool_0a_3x3')
  51         branch_3 = slim.conv2d(branch_3, 96, [1, 1], scope='Conv2d_0b_1x1')
  52       return tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])

  55 def block_reduction_a(inputs, scope=None, reuse=None):
  56   """Builds Reduction-A block for Inception v4 network."""
  57   # By default use stride=1 and SAME padding
  58   with slim.arg_scope([slim.conv2d, slim.avg_pool2d, slim.max_pool2d],
  59                       stride=1, padding='SAME'):
  60     with tf.variable_scope(
  61         scope, 'BlockReductionA', [inputs], reuse=reuse):
  62       with tf.variable_scope('Branch_0'):
  63         branch_0 = slim.conv2d(inputs, 384, [3, 3], stride=2, padding='VALID',
  64                                scope='Conv2d_1a_3x3')
  65       with tf.variable_scope('Branch_1'):
  66         branch_1 = slim.conv2d(inputs, 192, [1, 1], scope='Conv2d_0a_1x1')
  67         branch_1 = slim.conv2d(branch_1, 224, [3, 3], scope='Conv2d_0b_3x3')
  68         branch_1 = slim.conv2d(branch_1, 256, [3, 3], stride=2,
  69                                padding='VALID', scope='Conv2d_1a_3x3')
  70       with tf.variable_scope('Branch_2'):
  71         branch_2 = slim.max_pool2d(inputs, [3, 3], stride=2, padding='VALID',
  72                                    scope='MaxPool_1a_3x3')
  73       return tf.concat(axis=3, values=[branch_0, branch_1, branch_2])

  76 def block_inception_b(inputs, scope=None, reuse=None):
  77   """Builds Inception-B block for Inception v4 network."""
  78   # By default use stride=1 and SAME padding
  79   with slim.arg_scope([slim.conv2d, slim.avg_pool2d, slim.max_pool2d],
  80                       stride=1, padding='SAME'):
  81     with tf.variable_scope(
  82         scope, 'BlockInceptionB', [inputs], reuse=reuse):
  83       with tf.variable_scope('Branch_0'):
  84         branch_0 = slim.conv2d(inputs, 384, [1, 1], scope='Conv2d_0a_1x1')
  85       with tf.variable_scope('Branch_1'):
  86         branch_1 = slim.conv2d(inputs, 192, [1, 1], scope='Conv2d_0a_1x1')
  87         branch_1 = slim.conv2d(branch_1, 224, [1, 7], scope='Conv2d_0b_1x7')
  88         branch_1 = slim.conv2d(branch_1, 256, [7, 1], scope='Conv2d_0c_7x1')
  89       with tf.variable_scope('Branch_2'):
  90         branch_2 = slim.conv2d(inputs, 192, [1, 1], scope='Conv2d_0a_1x1')
  91         branch_2 = slim.conv2d(branch_2, 192, [7, 1], scope='Conv2d_0b_7x1')
  92         branch_2 = slim.conv2d(branch_2, 224, [1, 7], scope='Conv2d_0c_1x7')
  93         branch_2 = slim.conv2d(branch_2, 224, [7, 1], scope='Conv2d_0d_7x1')
  94         branch_2 = slim.conv2d(branch_2, 256, [1, 7], scope='Conv2d_0e_1x7')
  95       with tf.variable_scope('Branch_3'):
  96         branch_3 = slim.avg_pool2d(inputs, [3, 3], scope='AvgPool_0a_3x3')
  97         branch_3 = slim.conv2d(branch_3, 128, [1, 1], scope='Conv2d_0b_1x1')
  98       return tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])

 101 def block_reduction_b(inputs, scope=None, reuse=None):
 102   """Builds Reduction-B block for Inception v4 network."""
 103   # By default use stride=1 and SAME padding
 104   with slim.arg_scope([slim.conv2d, slim.avg_pool2d, slim.max_pool2d],
 105                       stride=1, padding='SAME'):
 106     with tf.variable_scope(
 107         scope, 'BlockReductionB', [inputs], reuse=reuse):
 108       with tf.variable_scope('Branch_0'):
 109         branch_0 = slim.conv2d(inputs, 192, [1, 1], scope='Conv2d_0a_1x1')
 110         branch_0 = slim.conv2d(branch_0, 192, [3, 3], stride=2,
 111                                padding='VALID', scope='Conv2d_1a_3x3')
 112       with tf.variable_scope('Branch_1'):
 113         branch_1 = slim.conv2d(inputs, 256, [1, 1], scope='Conv2d_0a_1x1')
 114         branch_1 = slim.conv2d(branch_1, 256, [1, 7], scope='Conv2d_0b_1x7')
 115         branch_1 = slim.conv2d(branch_1, 320, [7, 1], scope='Conv2d_0c_7x1')
 116         branch_1 = slim.conv2d(branch_1, 320, [3, 3], stride=2,
 117                                padding='VALID', scope='Conv2d_1a_3x3')
 118       with tf.variable_scope('Branch_2'):
 119         branch_2 = slim.max_pool2d(inputs, [3, 3], stride=2, padding='VALID',
 120                                    scope='MaxPool_1a_3x3')
 121       return tf.concat(axis=3, values=[branch_0, branch_1, branch_2])

 124 def block_inception_c(inputs, scope=None, reuse=None):
 125   """Builds Inception-C block for Inception v4 network."""
 126   # By default use stride=1 and SAME padding
 127   with slim.arg_scope([slim.conv2d, slim.avg_pool2d, slim.max_pool2d],
 128                       stride=1, padding='SAME'):
 129     with tf.variable_scope(
 130         scope, 'BlockInceptionC', [inputs], reuse=reuse):
 131       with tf.variable_scope('Branch_0'):
 132         branch_0 = slim.conv2d(inputs, 256, [1, 1], scope='Conv2d_0a_1x1')
 133       with tf.variable_scope('Branch_1'):
 134         branch_1 = slim.conv2d(inputs, 384, [1, 1], scope='Conv2d_0a_1x1')
 135         branch_1 = tf.concat(axis=3, values=[
 136             slim.conv2d(branch_1, 256, [1, 3], scope='Conv2d_0b_1x3'),
 137             slim.conv2d(branch_1, 256, [3, 1], scope='Conv2d_0c_3x1')])
 138       with tf.variable_scope('Branch_2'):
 139         branch_2 = slim.conv2d(inputs, 384, [1, 1], scope='Conv2d_0a_1x1')
 140         branch_2 = slim.conv2d(branch_2, 448, [3, 1], scope='Conv2d_0b_3x1')
 141         branch_2 = slim.conv2d(branch_2, 512, [1, 3], scope='Conv2d_0c_1x3')
 142         branch_2 = tf.concat(axis=3, values=[
 143             slim.conv2d(branch_2, 256, [1, 3], scope='Conv2d_0d_1x3'),
 144             slim.conv2d(branch_2, 256, [3, 1], scope='Conv2d_0e_3x1')])
 145       with tf.variable_scope('Branch_3'):
 146         branch_3 = slim.avg_pool2d(inputs, [3, 3], scope='AvgPool_0a_3x3')
 147         branch_3 = slim.conv2d(branch_3, 256, [1, 1], scope='Conv2d_0b_1x1')
 148       return tf.concat(axis=3, values=[branch_0, branch_1, branch_2, branch_3])

 151 def inception_v4_base(inputs, final_endpoint='Mixed_7d', scope=None):
 152   """Creates the Inception V4 network up to the given final endpoint.
 153 
 154   Args:
 155     inputs: a 4-D tensor of size [batch_size, height, width, 3].
 156     final_endpoint: specifies the endpoint to construct the network up to.
 157       It can be one of [ 'Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3',
 158       'Mixed_3a', 'Mixed_4a', 'Mixed_5a', 'Mixed_5b', 'Mixed_5c', 'Mixed_5d',
 159       'Mixed_5e', 'Mixed_6a', 'Mixed_6b', 'Mixed_6c', 'Mixed_6d', 'Mixed_6e',
 160       'Mixed_6f', 'Mixed_6g', 'Mixed_6h', 'Mixed_7a', 'Mixed_7b', 'Mixed_7c',
 161       'Mixed_7d']
 162     scope: Optional variable_scope.
 163 
 164   Returns:
 165     logits: the logits outputs of the model.
 166     end_points: the set of end_points from the inception model.
 167 
 168   Raises:
 169     ValueError: if final_endpoint is not set to one of the predefined values,
 170   """
 171   end_points = {}
 172 
 173   def add_and_check_final(name, net):
 174     end_points[name] = net
 175     return name == final_endpoint
 176 
 177   with tf.variable_scope(scope, 'InceptionV4', [inputs]):
 178     with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 179                         stride=1, padding='SAME'):
 180       # 299 x 299 x 3
 181       net = slim.conv2d(inputs, 32, [3, 3], stride=2,
 182                         padding='VALID', scope='Conv2d_1a_3x3')
 183       if add_and_check_final('Conv2d_1a_3x3', net): return net, end_points
 184       # 149 x 149 x 32
 185       net = slim.conv2d(net, 32, [3, 3], padding='VALID',
 186                         scope='Conv2d_2a_3x3')
 187       if add_and_check_final('Conv2d_2a_3x3', net): return net, end_points
 188       # 147 x 147 x 32
 189       net = slim.conv2d(net, 64, [3, 3], scope='Conv2d_2b_3x3')
 190       if add_and_check_final('Conv2d_2b_3x3', net): return net, end_points
 191       # 147 x 147 x 64
 192       with tf.variable_scope('Mixed_3a'):
 193         with tf.variable_scope('Branch_0'):
 194           branch_0 = slim.max_pool2d(net, [3, 3], stride=2, padding='VALID',
 195                                      scope='MaxPool_0a_3x3')
 196         with tf.variable_scope('Branch_1'):
 197           branch_1 = slim.conv2d(net, 96, [3, 3], stride=2, padding='VALID',
 198                                  scope='Conv2d_0a_3x3')
 199         net = tf.concat(axis=3, values=[branch_0, branch_1])
 200         if add_and_check_final('Mixed_3a', net): return net, end_points
 201 
 202       # 73 x 73 x 160
 203       with tf.variable_scope('Mixed_4a'):
 204         with tf.variable_scope('Branch_0'):
 205           branch_0 = slim.conv2d(net, 64, [1, 1], scope='Conv2d_0a_1x1')
 206           branch_0 = slim.conv2d(branch_0, 96, [3, 3], padding='VALID',
 207                                  scope='Conv2d_1a_3x3')
 208         with tf.variable_scope('Branch_1'):
 209           branch_1 = slim.conv2d(net, 64, [1, 1], scope='Conv2d_0a_1x1')
 210           branch_1 = slim.conv2d(branch_1, 64, [1, 7], scope='Conv2d_0b_1x7')
 211           branch_1 = slim.conv2d(branch_1, 64, [7, 1], scope='Conv2d_0c_7x1')
 212           branch_1 = slim.conv2d(branch_1, 96, [3, 3], padding='VALID',
 213                                  scope='Conv2d_1a_3x3')
 214         net = tf.concat(axis=3, values=[branch_0, branch_1])
 215         if add_and_check_final('Mixed_4a', net): return net, end_points
 216 
 217       # 71 x 71 x 192
 218       with tf.variable_scope('Mixed_5a'):
 219         with tf.variable_scope('Branch_0'):
 220           branch_0 = slim.conv2d(net, 192, [3, 3], stride=2, padding='VALID',
 221                                  scope='Conv2d_1a_3x3')
 222         with tf.variable_scope('Branch_1'):
 223           branch_1 = slim.max_pool2d(net, [3, 3], stride=2, padding='VALID',
 224                                      scope='MaxPool_1a_3x3')
 225         net = tf.concat(axis=3, values=[branch_0, branch_1])
 226         if add_and_check_final('Mixed_5a', net): return net, end_points
 227 
 228       # 35 x 35 x 384
 229       # 4 x Inception-A blocks
 230       for idx in range(4):
 231         block_scope = 'Mixed_5' + chr(ord('b') + idx)
 232         net = block_inception_a(net, block_scope)
 233         if add_and_check_final(block_scope, net): return net, end_points
 234 
 235       # 35 x 35 x 384
 236       # Reduction-A block
 237       net = block_reduction_a(net, 'Mixed_6a')
 238       if add_and_check_final('Mixed_6a', net): return net, end_points
 239 
 240       # 17 x 17 x 1024
 241       # 7 x Inception-B blocks
 242       for idx in range(7):
 243         block_scope = 'Mixed_6' + chr(ord('b') + idx)
 244         net = block_inception_b(net, block_scope)
 245         if add_and_check_final(block_scope, net): return net, end_points
 246 
 247       # 17 x 17 x 1024
 248       # Reduction-B block
 249       net = block_reduction_b(net, 'Mixed_7a')
 250       if add_and_check_final('Mixed_7a', net): return net, end_points
 251 
 252       # 8 x 8 x 1536
 253       # 3 x Inception-C blocks
 254       for idx in range(3):
 255         block_scope = 'Mixed_7' + chr(ord('b') + idx)
 256         net = block_inception_c(net, block_scope)
 257         if add_and_check_final(block_scope, net): return net, end_points
 258   raise ValueError('Unknown final endpoint %s' % final_endpoint)

 261 def inception_v4(inputs, num_classes=1001, is_training=True,
 262                  dropout_keep_prob=0.8,
 263                  reuse=None,
 264                  scope='InceptionV4',
 265                  create_aux_logits=True):
 266   """Creates the Inception V4 model.
 267 
 268   Args:
 269     inputs: a 4-D tensor of size [batch_size, height, width, 3].
 270     num_classes: number of predicted classes. If 0 or None, the logits layer
 271       is omitted and the input features to the logits layer (before dropout)
 272       are returned instead.
 273     is_training: whether is training or not.
 274     dropout_keep_prob: float, the fraction to keep before final layer.
 275     reuse: whether or not the network and its variables should be reused. To be
 276       able to reuse 'scope' must be given.
 277     scope: Optional variable_scope.
 278     create_aux_logits: Whether to include the auxiliary logits.
 279 
 280   Returns:
 281     net: a Tensor with the logits (pre-softmax activations) if num_classes
 282       is a non-zero integer, or the non-dropped input to the logits layer
 283       if num_classes is 0 or None.
 284     end_points: the set of end_points from the inception model.
 285   """
 286   end_points = {}
 287   with tf.variable_scope(
 288       scope, 'InceptionV4', [inputs], reuse=reuse) as scope:
 289     with slim.arg_scope([slim.batch_norm, slim.dropout],
 290                         is_training=is_training):
 291       net, end_points = inception_v4_base(inputs, scope=scope)
 292 
 293       with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 294                           stride=1, padding='SAME'):
 295         # Auxiliary Head logits
 296         if create_aux_logits and num_classes:
 297           with tf.variable_scope('AuxLogits'):
 298             # 17 x 17 x 1024
 299             aux_logits = end_points['Mixed_6h']
 300             aux_logits = slim.avg_pool2d(aux_logits, [5, 5], stride=3,
 301                                          padding='VALID',
 302                                          scope='AvgPool_1a_5x5')
 303             aux_logits = slim.conv2d(aux_logits, 128, [1, 1],
 304                                      scope='Conv2d_1b_1x1')
 305             aux_logits = slim.conv2d(aux_logits, 768,
 306                                      aux_logits.get_shape()[1:3],
 307                                      padding='VALID', scope='Conv2d_2a')
 308             aux_logits = slim.flatten(aux_logits)
 309             aux_logits = slim.fully_connected(aux_logits, num_classes,
 310                                               activation_fn=None,
 311                                               scope='Aux_logits')
 312             end_points['AuxLogits'] = aux_logits
 313 
 314         # Final pooling and prediction
 315         # TODO(sguada,arnoegw): Consider adding a parameter global_pool which
 316         # can be set to False to disable pooling here (as in resnet_*()).
 317         with tf.variable_scope('Logits'):
 318           # 8 x 8 x 1536
 319           kernel_size = net.get_shape()[1:3]
 320           if kernel_size.is_fully_defined():
 321             net = slim.avg_pool2d(net, kernel_size, padding='VALID',
 322                                   scope='AvgPool_1a')
 323           else:
 324             net = tf.reduce_mean(
 325                 input_tensor=net,
 326                 axis=[1, 2],
 327                 keepdims=True,
 328                 name='global_pool')
 329           end_points['global_pool'] = net
 330           if not num_classes:
 331             return net, end_points
 332           # 1 x 1 x 1536
 333           net = slim.dropout(net, dropout_keep_prob, scope='Dropout_1b')
 334           net = slim.flatten(net, scope='PreLogitsFlatten')
 335           end_points['PreLogitsFlatten'] = net
 336           # 1536
 337           logits = slim.fully_connected(net, num_classes, activation_fn=None,
 338                                         scope='Logits')
 339           end_points['Logits'] = logits
 340           end_points['Predictions'] = tf.nn.softmax(logits, name='Predictions')
 341     return logits, end_points
 342 inception_v4.default_image_size = 299
