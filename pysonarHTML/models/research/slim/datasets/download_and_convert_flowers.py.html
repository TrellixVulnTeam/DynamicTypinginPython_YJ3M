  15 r"""Downloads and converts Flowers data to TFRecords of TF-Example protos.
  16 
  17 This module downloads the Flowers data, uncompresses it, reads the files
  18 that make up the Flowers data and creates two TFRecord datasets: one for train
  19 and one for test. Each TFRecord dataset is comprised of a set of TF-Example
  20 protocol buffers, each of which contain a single image and label.
  21 
  22 The script should take about a minute to run.
  23 
  24 """
  25 
  26 from __future__ import absolute_import
  27 from __future__ import division
  28 from __future__ import print_function
  29 
  30 import math
  31 import os
  32 import random
  33 import sys
  34 
  35 from six.moves import range
  36 from six.moves import zip
  37 import tensorflow.compat.v1 as tf
  38 
  39 from datasets import dataset_utils
  40 
  41 # The URL where the Flowers data can be downloaded.
  42 _DATA_URL = 'http://download.tensorflow.org/example_images/flower_photos.tgz'
  43 
  44 # The number of images in the validation set.
  45 _NUM_VALIDATION = 350
  46 
  47 # Seed for repeatability.
  48 _RANDOM_SEED = 0
  49 
  50 # The number of shards per dataset split.
  51 _NUM_SHARDS = 5
  52 
  53 
  54 class ImageReader(object):
  55   """Helper class that provides TensorFlow image coding utilities."""
  56 
  57   def __init__(self):
  58     # Initializes function that decodes RGB JPEG data.
  59     self._decode_jpeg_data = tf.placeholder(dtype=tf.string)
  60     self._decode_jpeg = tf.image.decode_jpeg(self._decode_jpeg_data, channels=3)
  61 
  62   def read_image_dims(self, sess, image_data):
  63     image = self.decode_jpeg(sess, image_data)
  64     return image.shape[0], image.shape[1]
  65 
  66   def decode_jpeg(self, sess, image_data):
  67     image = sess.run(self._decode_jpeg,
  68                      feed_dict={self._decode_jpeg_data: image_data})
  69     assert len(image.shape) == 3
  70     assert image.shape[2] == 3
  71     return image
  72

  74 def _get_filenames_and_classes(dataset_dir):
  75   """Returns a list of filenames and inferred class names.
  76 
  77   Args:
  78     dataset_dir: A directory containing a set of subdirectories representing
  79       class names. Each subdirectory should contain PNG or JPG encoded images.
  80 
  81   Returns:
  82     A list of image file paths, relative to `dataset_dir` and the list of
  83     subdirectories, representing class names.
  84   """
  85   flower_root = os.path.join(dataset_dir, 'flower_photos')
  86   directories = []
  87   class_names = []
  88   for filename in os.listdir(flower_root):
  89     path = os.path.join(flower_root, filename)
  90     if os.path.isdir(path):
  91       directories.append(path)
  92       class_names.append(filename)
  93 
  94   photo_filenames = []
  95   for directory in directories:
  96     for filename in os.listdir(directory):
  97       path = os.path.join(directory, filename)
  98       photo_filenames.append(path)
  99 
 100   return photo_filenames, sorted(class_names)

 103 def _get_dataset_filename(dataset_dir, split_name, shard_id):
 104   output_filename = 'flowers_%s_%05d-of-%05d.tfrecord' % (
 105       split_name, shard_id, _NUM_SHARDS)
 106   return os.path.join(dataset_dir, output_filename)
 107 
 108 
 109 def _convert_dataset(split_name, filenames, class_names_to_ids, dataset_dir):
 110   """Converts the given filenames to a TFRecord dataset.
 111 
 112   Args:
 113     split_name: The name of the dataset, either 'train' or 'validation'.
 114     filenames: A list of absolute paths to png or jpg images.
 115     class_names_to_ids: A dictionary from class names (strings) to ids
 116       (integers).
 117     dataset_dir: The directory where the converted datasets are stored.
 118   """
 119   assert split_name in ['train', 'validation']
 120 
 121   num_per_shard = int(math.ceil(len(filenames) / float(_NUM_SHARDS)))
 122 
 123   with tf.Graph().as_default():
 124     image_reader = ImageReader()
 125 
 126     with tf.Session('') as sess:
 127 
 128       for shard_id in range(_NUM_SHARDS):
 129         output_filename = _get_dataset_filename(
 130             dataset_dir, split_name, shard_id)
 131 
 132         with tf.python_io.TFRecordWriter(output_filename) as tfrecord_writer:
 133           start_ndx = shard_id * num_per_shard
 134           end_ndx = min((shard_id+1) * num_per_shard, len(filenames))
 135           for i in range(start_ndx, end_ndx):
 136             sys.stdout.write('\r>> Converting image %d/%d shard %d' % (
 137                 i+1, len(filenames), shard_id))
 138             sys.stdout.flush()
 139 
 140             # Read the filename:
 141             image_data = tf.gfile.GFile(filenames[i], 'rb').read()
 142             height, width = image_reader.read_image_dims(sess, image_data)
 143 
 144             class_name = os.path.basename(os.path.dirname(filenames[i]))
 145             class_id = class_names_to_ids[class_name]
 146 
 147             example = dataset_utils.image_to_tfexample(
 148                 image_data, b'jpg', height, width, class_id)
 149             tfrecord_writer.write(example.SerializeToString())
 150 
 151   sys.stdout.write('\n')
 152   sys.stdout.flush()
 153

 155 def _clean_up_temporary_files(dataset_dir):
 156   """Removes temporary files used to create the dataset.
 157 
 158   Args:
 159     dataset_dir: The directory where the temporary files are stored.
 160   """
 161   filename = _DATA_URL.split('/')[-1]
 162   filepath = os.path.join(dataset_dir, filename)
 163   tf.gfile.Remove(filepath)
 164 
 165   tmp_dir = os.path.join(dataset_dir, 'flower_photos')
 166   tf.gfile.DeleteRecursively(tmp_dir)
 167 
 168 
 169 def _dataset_exists(dataset_dir):
 170   for split_name in ['train', 'validation']:
 171     for shard_id in range(_NUM_SHARDS):
 172       output_filename = _get_dataset_filename(
 173           dataset_dir, split_name, shard_id)
 174       if not tf.gfile.Exists(output_filename):
 175         return False
 176   return True
 177 
 178 
 179 def run(dataset_dir):
 180   """Runs the download and conversion operation.
 181 
 182   Args:
 183     dataset_dir: The dataset directory where the dataset is stored.
 184   """
 185   if not tf.gfile.Exists(dataset_dir):
 186     tf.gfile.MakeDirs(dataset_dir)
 187 
 188   if _dataset_exists(dataset_dir):
 189     print('Dataset files already exist. Exiting without re-creating them.')
 190     return
 191 
 192   dataset_utils.download_and_uncompress_tarball(_DATA_URL, dataset_dir)
 193   photo_filenames, class_names = _get_filenames_and_classes(dataset_dir)
 194   class_names_to_ids = dict(
 195       list(zip(class_names, list(range(len(class_names))))))
 196 
 197   # Divide into train and test:
 198   random.seed(_RANDOM_SEED)
 199   random.shuffle(photo_filenames)
 200   training_filenames = photo_filenames[_NUM_VALIDATION:]
 201   validation_filenames = photo_filenames[:_NUM_VALIDATION]
 202 
 203   # First, convert the training and validation sets.
 204   _convert_dataset('train', training_filenames, class_names_to_ids,
 205                    dataset_dir)
 206   _convert_dataset('validation', validation_filenames, class_names_to_ids,
 207                    dataset_dir)
 208 
 209   # Finally, write the labels file:
 210   labels_to_class_names = dict(
 211       list(zip(list(range(len(class_names))), class_names)))
 212   dataset_utils.write_label_file(labels_to_class_names, dataset_dir)
 213 
 214   _clean_up_temporary_files(dataset_dir)
 215   print('\nFinished converting the Flowers dataset!')
