  15 """Provides data for the ImageNet ILSVRC 2012 Dataset plus some bounding boxes.
  16 
  17 Some images have one or more bounding boxes associated with the label of the
  18 image. See details here: http://image-net.org/download-bboxes
  19 
  20 ImageNet is based upon WordNet 3.0. To uniquely identify a synset, we use
  21 "WordNet ID" (wnid), which is a concatenation of POS ( i.e. part of speech )
  22 and SYNSET OFFSET of WordNet. For more information, please refer to the
  23 WordNet documentation[http://wordnet.princeton.edu/wordnet/documentation/].
  24 
  25 "There are bounding boxes for over 3000 popular synsets available.
  26 For each synset, there are on average 150 images with bounding boxes."
  27 
  28 WARNING: Don't use for object detection, in this case all the bounding boxes
  29 of the image belong to just one class.
  30 """
  31 from __future__ import absolute_import
  32 from __future__ import division
  33 from __future__ import print_function
  34 
  35 import os
  36 from six.moves import urllib
  37 import tensorflow.compat.v1 as tf
  38 import tf_slim as slim
  39 
  40 from datasets import dataset_utils
  41 
  42 # TODO(nsilberman): Add tfrecord file type once the script is updated.
  43 _FILE_PATTERN = '%s-*'
  44 
  45 _SPLITS_TO_SIZES = {
  46     'train': 1281167,
  47     'validation': 50000,
  48 }
  49 
  50 _ITEMS_TO_DESCRIPTIONS = {
  51     'image': 'A color image of varying height and width.',
  52     'label': 'The label id of the image, integer between 0 and 999',
  53     'label_text': 'The text of the label.',
  54     'object/bbox': 'A list of bounding boxes.',
  55     'object/label': 'A list of labels, one per each object.',
  56 }
  57 
  58 _NUM_CLASSES = 1001
  59 
  60 # If set to false, will not try to set label_to_names in dataset
  61 # by reading them from labels.txt or github.
  62 LOAD_READABLE_NAMES = True

  65 def create_readable_names_for_imagenet_labels():
  66   """Create a dict mapping label id to human readable string.
  67 
  68   Returns:
  69       labels_to_names: dictionary where keys are integers from to 1000
  70       and values are human-readable names.
  71 
  72   We retrieve a synset file, which contains a list of valid synset labels used
  73   by ILSVRC competition. There is one synset one per line, eg.
  74           #   n01440764
  75           #   n01443537
  76   We also retrieve a synset_to_human_file, which contains a mapping from synsets
  77   to human-readable names for every synset in Imagenet. These are stored in a
  78   tsv format, as follows:
  79           #   n02119247    black fox
  80           #   n02119359    silver fox
  81   We assign each synset (in alphabetical order) an integer, starting from 1
  82   (since 0 is reserved for the background class).
  83 
  84   Code is based on
  85   https://github.com/tensorflow/models/blob/master/research/inception/inception/data/build_imagenet_data.py#L463
  86   """
  87 
  88   # pylint: disable=g-line-too-long
  89   base_url = 'https://raw.githubusercontent.com/tensorflow/models/master/research/inception/inception/data/'
  90   synset_url = '{}/imagenet_lsvrc_2015_synsets.txt'.format(base_url)
  91   synset_to_human_url = '{}/imagenet_metadata.txt'.format(base_url)
  92 
  93   filename, _ = urllib.request.urlretrieve(synset_url)
  94   synset_list = [s.strip() for s in open(filename).readlines()]
  95   num_synsets_in_ilsvrc = len(synset_list)
  96   assert num_synsets_in_ilsvrc == 1000
  97 
  98   filename, _ = urllib.request.urlretrieve(synset_to_human_url)
  99   synset_to_human_list = open(filename).readlines()
 100   num_synsets_in_all_imagenet = len(synset_to_human_list)
 101   assert num_synsets_in_all_imagenet == 21842
 102 
 103   synset_to_human = {}
 104   for s in synset_to_human_list:
 105     parts = s.strip().split('\t')
 106     assert len(parts) == 2
 107     synset = parts[0]
 108     human = parts[1]
 109     synset_to_human[synset] = human
 110 
 111   label_index = 1
 112   labels_to_names = {0: 'background'}
 113   for synset in synset_list:
 114     name = synset_to_human[synset]
 115     labels_to_names[label_index] = name
 116     label_index += 1
 117 
 118   return labels_to_names

 121 def get_split(split_name, dataset_dir, file_pattern=None, reader=None):
 122   """Gets a dataset tuple with instructions for reading ImageNet.
 123 
 124   Args:
 125     split_name: A train/test split name.
 126     dataset_dir: The base directory of the dataset sources.
 127     file_pattern: The file pattern to use when matching the dataset sources.
 128       It is assumed that the pattern contains a '%s' string so that the split
 129       name can be inserted.
 130     reader: The TensorFlow reader type.
 131 
 132   Returns:
 133     A `Dataset` namedtuple.
 134 
 135   Raises:
 136     ValueError: if `split_name` is not a valid train/test split.
 137   """
 138   if split_name not in _SPLITS_TO_SIZES:
 139     raise ValueError('split name %s was not recognized.' % split_name)
 140 
 141   if not file_pattern:
 142     file_pattern = _FILE_PATTERN
 143   file_pattern = os.path.join(dataset_dir, file_pattern % split_name)
 144 
 145   # Allowing None in the signature so that dataset_factory can use the default.
 146   if reader is None:
 147     reader = tf.TFRecordReader
 148 
 149   keys_to_features = {
 150       'image/encoded': tf.FixedLenFeature(
 151           (), tf.string, default_value=''),
 152       'image/format': tf.FixedLenFeature(
 153           (), tf.string, default_value='jpeg'),
 154       'image/class/label': tf.FixedLenFeature(
 155           [], dtype=tf.int64, default_value=-1),
 156       'image/class/text': tf.FixedLenFeature(
 157           [], dtype=tf.string, default_value=''),
 158       'image/object/bbox/xmin': tf.VarLenFeature(
 159           dtype=tf.float32),
 160       'image/object/bbox/ymin': tf.VarLenFeature(
 161           dtype=tf.float32),
 162       'image/object/bbox/xmax': tf.VarLenFeature(
 163           dtype=tf.float32),
 164       'image/object/bbox/ymax': tf.VarLenFeature(
 165           dtype=tf.float32),
 166       'image/object/class/label': tf.VarLenFeature(
 167           dtype=tf.int64),
 168   }
 169 
 170   items_to_handlers = {
 171       'image': slim.tfexample_decoder.Image('image/encoded', 'image/format'),
 172       'label': slim.tfexample_decoder.Tensor('image/class/label'),
 173       'label_text': slim.tfexample_decoder.Tensor('image/class/text'),
 174       'object/bbox': slim.tfexample_decoder.BoundingBox(
 175           ['ymin', 'xmin', 'ymax', 'xmax'], 'image/object/bbox/'),
 176       'object/label': slim.tfexample_decoder.Tensor('image/object/class/label'),
 177   }
 178 
 179   decoder = slim.tfexample_decoder.TFExampleDecoder(
 180       keys_to_features, items_to_handlers)
 181 
 182   labels_to_names = None
 183   if LOAD_READABLE_NAMES:
 184     if dataset_utils.has_labels(dataset_dir):
 185       labels_to_names = dataset_utils.read_label_file(dataset_dir)
 186     else:
 187       labels_to_names = create_readable_names_for_imagenet_labels()
 188       dataset_utils.write_label_file(labels_to_names, dataset_dir)
 189 
 190   return slim.dataset.Dataset(
 191       data_sources=file_pattern,
 192       reader=reader,
 193       decoder=decoder,
 194       num_samples=_SPLITS_TO_SIZES[split_name],
 195       items_to_descriptions=_ITEMS_TO_DESCRIPTIONS,
 196       num_classes=_NUM_CLASSES,
 197       labels_to_names=labels_to_names)
