  31 def int64_feature(values):
  32   """Returns a TF-Feature of int64s.
  33 
  34   Args:
  35     values: A scalar or list of values.
  36 
  37   Returns:
  38     A TF-Feature.
  39   """
  40   if not isinstance(values, (tuple, list)):
  41     values = [values]
  42   return tf.train.Feature(int64_list=tf.train.Int64List(value=values))

  95 def image_to_tfexample(image_data, image_format, height, width, class_id):
  96   return tf.train.Example(features=tf.train.Features(feature={
  97       'image/encoded': bytes_feature(image_data),
  98       'image/format': bytes_feature(image_format),
  99       'image/class/label': int64_feature(class_id),
 100       'image/height': int64_feature(height),
 101       'image/width': int64_feature(width),
 102   }))

 105 def download_url(url, dataset_dir):
 106   """Downloads the tarball or zip file from url into filepath.
 107 
 108   Args:
 109     url: The URL of a tarball or zip file.
 110     dataset_dir: The directory where the temporary files are stored.
 111 
 112   Returns:
 113     filepath: path where the file is downloaded.
 114   """
 115   filename = url.split('/')[-1]
 116   filepath = os.path.join(dataset_dir, filename)
 117 
 118   def _progress(count, block_size, total_size):
 119     sys.stdout.write('\r>> Downloading %s %.1f%%' % (
 120         filename, float(count * block_size) / float(total_size) * 100.0))
 121     sys.stdout.flush()
 122 
 123   filepath, _ = urllib.request.urlretrieve(url, filepath, _progress)
 124   print()
 125   statinfo = os.stat(filepath)
 126   print('Successfully downloaded', filename, statinfo.st_size, 'bytes.')
 127   return filepath

 130 def download_and_uncompress_tarball(tarball_url, dataset_dir):
 131   """Downloads the `tarball_url` and uncompresses it locally.
 132 
 133   Args:
 134     tarball_url: The URL of a tarball file.
 135     dataset_dir: The directory where the temporary files are stored.
 136   """
 137   filepath = download_url(tarball_url, dataset_dir)
 138   tarfile.open(filepath, 'r:gz').extractall(dataset_dir)

 141 def download_and_uncompress_zipfile(zip_url, dataset_dir):
 142   """Downloads the `zip_url` and uncompresses it locally.
 143 
 144   Args:
 145     zip_url: The URL of a zip file.
 146     dataset_dir: The directory where the temporary files are stored.
 147   """
 148   filename = zip_url.split('/')[-1]
 149   filepath = os.path.join(dataset_dir, filename)
 150 
 151   if tf.gfile.Exists(filepath):
 152     print('File {filename} has been already downloaded at {filepath}. '
 153           'Unzipping it....'.format(filename=filename, filepath=filepath))
 154   else:
 155     filepath = download_url(zip_url, dataset_dir)
 156 
 157   with zipfile.ZipFile(filepath, 'r') as zip_file:
 158     for member in zip_file.namelist():
 159       memberpath = os.path.join(dataset_dir, member)
 160       # extract only if file doesn't exist
 161       if not (os.path.exists(memberpath) or os.path.isfile(memberpath)):
 162         zip_file.extract(member, dataset_dir)

 165 def write_label_file(labels_to_class_names,
 166                      dataset_dir,
 167                      filename=LABELS_FILENAME):
 168   """Writes a file with the list of class names.
 169 
 170   Args:
 171     labels_to_class_names: A map of (integer) labels to class names.
 172     dataset_dir: The directory in which the labels file should be written.
 173     filename: The filename where the class names are written.
 174   """
 175   labels_filename = os.path.join(dataset_dir, filename)
 176   with tf.gfile.Open(labels_filename, 'w') as f:
 177     for label in labels_to_class_names:
 178       class_name = labels_to_class_names[label]
 179       f.write('%d:%s\n' % (label, class_name))

 182 def has_labels(dataset_dir, filename=LABELS_FILENAME):
 183   """Specifies whether or not the dataset directory contains a label map file.
 184 
 185   Args:
 186     dataset_dir: The directory in which the labels file is found.
 187     filename: The filename where the class names are written.
 188 
 189   Returns:
 190     `True` if the labels file exists and `False` otherwise.
 191   """
 192   return tf.gfile.Exists(os.path.join(dataset_dir, filename))

 195 def read_label_file(dataset_dir, filename=LABELS_FILENAME):
 196   """Reads the labels file and returns a mapping from ID to class name.
 197 
 198   Args:
 199     dataset_dir: The directory in which the labels file is found.
 200     filename: The filename where the class names are written.
 201 
 202   Returns:
 203     A map from a label (integer) to class name.
 204   """
 205   labels_filename = os.path.join(dataset_dir, filename)
 206   with tf.gfile.Open(labels_filename, 'rb') as f:
 207     lines = f.read().decode()
 208   lines = lines.split('\n')
 209   lines = filter(None, lines)
 210 
 211   labels_to_class_names = {}
 212   for line in lines:
 213     index = line.index(':')
 214     labels_to_class_names[int(line[:index])] = line[index+1:]
 215   return labels_to_class_names

 218 def open_sharded_output_tfrecords(exit_stack, base_path, num_shards):
 219   """Opens all TFRecord shards for writing and adds them to an exit stack.
 220 
 221   Args:
 222     exit_stack: A context2.ExitStack used to automatically closed the TFRecords
 223       opened in this function.
 224     base_path: The base path for all shards
 225     num_shards: The number of shards
 226 
 227   Returns:
 228     The list of opened TFRecords. Position k in the list corresponds to shard k.
 229   """
 230   tf_record_output_filenames = [
 231       '{}-{:05d}-of-{:05d}'.format(base_path, idx, num_shards)
 232       for idx in range(num_shards)
 233   ]
 234 
 235   tfrecords = [
 236       exit_stack.enter_context(tf.python_io.TFRecordWriter(file_name))
 237       for file_name in tf_record_output_filenames
 238   ]
 239 
 240   return tfrecords
