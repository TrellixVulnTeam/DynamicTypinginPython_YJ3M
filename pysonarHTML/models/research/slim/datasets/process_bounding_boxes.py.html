  16 """Process the ImageNet Challenge bounding boxes for TensorFlow model training.
  17 
  18 This script is called as
  19 
  20 process_bounding_boxes.py <dir> [synsets-file]
  21 
  22 Where <dir> is a directory containing the downloaded and unpacked bounding box
  23 data. If [synsets-file] is supplied, then only the bounding boxes whose
  24 synstes are contained within this file are returned. Note that the
  25 [synsets-file] file contains synset ids, one per line.
  26 
  27 The script dumps out a CSV text file in which each line contains an entry.
  28   n00007846_64193.JPEG,0.0060,0.2620,0.7545,0.9940
  29 
  30 The entry can be read as:
  31   <JPEG file name>, <xmin>, <ymin>, <xmax>, <ymax>
  32 
  33 The bounding box for <JPEG file name> contains two points (xmin, ymin) and
  34 (xmax, ymax) specifying the lower-left corner and upper-right corner of a
  35 bounding box in *relative* coordinates.
  36 
  37 The user supplies a directory where the XML files reside. The directory
  38 structure in the directory <dir> is assumed to look like this:
  39 
  40 <dir>/nXXXXXXXX/nXXXXXXXX_YYYY.xml
  41 
  42 Each XML file contains a bounding box annotation. The script:
  43 
  44  (1) Parses the XML file and extracts the filename, label and bounding box info.
  45 
  46  (2) The bounding box is specified in the XML files as integer (xmin, ymin) and
  47     (xmax, ymax) *relative* to image size displayed to the human annotator. The
  48     size of the image displayed to the human annotator is stored in the XML file
  49     as integer (height, width).
  50 
  51     Note that the displayed size will differ from the actual size of the image
  52     downloaded from image-net.org. To make the bounding box annotation useable,
  53     we convert bounding box to floating point numbers relative to displayed
  54     height and width of the image.
  55 
  56     Note that each XML file might contain N bounding box annotations.
  57 
  58     Note that the points are all clamped at a range of [0.0, 1.0] because some
  59     human annotations extend outside the range of the supplied image.
  60 
  61     See details here: http://image-net.org/download-bboxes
  62 
  63 (3) By default, the script outputs all valid bounding boxes. If a
  64     [synsets-file] is supplied, only the subset of bounding boxes associated
  65     with those synsets are outputted. Importantly, one can supply a list of
  66     synsets in the ImageNet Challenge and output the list of bounding boxes
  67     associated with the training images of the ILSVRC.
  68 
  69     We use these bounding boxes to inform the random distortion of images
  70     supplied to the network.
  71 
  72 If you run this script successfully, you will see the following output
  73 to stderr:
  74 > Finished processing 544546 XML files.
  75 > Skipped 0 XML files not in ImageNet Challenge.
  76 > Skipped 0 bounding boxes not in ImageNet Challenge.
  77 > Wrote 615299 bounding boxes from 544546 annotated images.
  78 """
  79 
  80 from __future__ import absolute_import
  81 from __future__ import division
  82 from __future__ import print_function
  83 
  84 import glob
  85 import os.path
  86 import sys
  87 import xml.etree.ElementTree as ET
  88 from six.moves import xrange  # pylint: disable=redefined-builtin
  89 
  90 
  91 class BoundingBox(object):
  92   pass
  93 
  94 
  95 def GetItem(name, root, index=0):
  96   count = 0
  97   for item in root.iter(name):
  98     if count == index:
  99       return item.text
 100     count += 1
 101   # Failed to find "index" occurrence of item.
 102   return -1
 103 
 104 
 105 def GetInt(name, root, index=0):
 106   return int(GetItem(name, root, index))
 107 
 108 
 109 def FindNumberBoundingBoxes(root):
 110   index = 0
 111   while True:
 112     if GetInt('xmin', root, index) == -1:
 113       break
 114     index += 1
 115   return index
 116 
 117 
 118 def ProcessXMLAnnotation(xml_file):
 119   """Process a single XML file containing a bounding box."""
 120   # pylint: disable=broad-except
 121   try:
 122     tree = ET.parse(xml_file)
 123   except Exception:
 124     print('Failed to parse: ' + xml_file, file=sys.stderr)
 125     return None
 126   # pylint: enable=broad-except
 127   root = tree.getroot()
 128 
 129   num_boxes = FindNumberBoundingBoxes(root)
 130   boxes = []
 131 
 132   for index in xrange(num_boxes):
 133     box = BoundingBox()
 134     # Grab the 'index' annotation.
 135     box.xmin = GetInt('xmin', root, index)
 136     box.ymin = GetInt('ymin', root, index)
 137     box.xmax = GetInt('xmax', root, index)
 138     box.ymax = GetInt('ymax', root, index)
 139 
 140     box.width = GetInt('width', root)
 141     box.height = GetInt('height', root)
 142     box.filename = GetItem('filename', root) + '.JPEG'
 143     box.label = GetItem('name', root)
 144 
 145     xmin = float(box.xmin) / float(box.width)
 146     xmax = float(box.xmax) / float(box.width)
 147     ymin = float(box.ymin) / float(box.height)
 148     ymax = float(box.ymax) / float(box.height)
 149 
 150     # Some images contain bounding box annotations that
 151     # extend outside of the supplied image. See, e.g.
 152     # n03127925/n03127925_147.xml
 153     # Additionally, for some bounding boxes, the min > max
 154     # or the box is entirely outside of the image.
 155     min_x = min(xmin, xmax)
 156     max_x = max(xmin, xmax)
 157     box.xmin_scaled = min(max(min_x, 0.0), 1.0)
 158     box.xmax_scaled = min(max(max_x, 0.0), 1.0)
 159 
 160     min_y = min(ymin, ymax)
 161     max_y = max(ymin, ymax)
 162     box.ymin_scaled = min(max(min_y, 0.0), 1.0)
 163     box.ymax_scaled = min(max(max_y, 0.0), 1.0)
 164 
 165     boxes.append(box)
 166 
 167   return boxes
 168 
 169 if __name__ == '__main__':
 170   if len(sys.argv) < 2 or len(sys.argv) > 3:
 171     print('Invalid usage\n'
 172           'usage: process_bounding_boxes.py <dir> [synsets-file]',
 173           file=sys.stderr)
 174     sys.exit(-1)
 175 
 176   xml_files = glob.glob(sys.argv[1] + '/*/*.xml')
 177   print('Identified %d XML files in %s' % (len(xml_files), sys.argv[1]),
 178         file=sys.stderr)
 179 
 180   if len(sys.argv) == 3:
 181     labels = set([l.strip() for l in open(sys.argv[2]).readlines()])
 182     print('Identified %d synset IDs in %s' % (len(labels), sys.argv[2]),
 183           file=sys.stderr)
 184   else:
 185     labels = None
 186 
 187   skipped_boxes = 0
 188   skipped_files = 0
 189   saved_boxes = 0
 190   saved_files = 0
 191   for file_index, one_file in enumerate(xml_files):
 192     # Example: <...>/n06470073/n00141669_6790.xml
 193     label = os.path.basename(os.path.dirname(one_file))
 194 
 195     # Determine if the annotation is from an ImageNet Challenge label.
 196     if labels is not None and label not in labels:
 197       skipped_files += 1
 198       continue
 199 
 200     bboxes = ProcessXMLAnnotation(one_file)
 201     assert bboxes is not None, 'No bounding boxes found in ' + one_file
 202 
 203     found_box = False
 204     for bbox in bboxes:
 205       if labels is not None:
 206         if bbox.label != label:
 207           # Note: There is a slight bug in the bounding box annotation data.
 208           # Many of the dog labels have the human label 'Scottish_deerhound'
 209           # instead of the synset ID 'n02092002' in the bbox.label field. As a
 210           # simple hack to overcome this issue, we only exclude bbox labels
 211           # *which are synset ID's* that do not match original synset label for
 212           # the XML file.
 213           if bbox.label in labels:
 214             skipped_boxes += 1
 215             continue
 216 
 217       # Guard against improperly specified boxes.
 218       if (bbox.xmin_scaled >= bbox.xmax_scaled or
 219           bbox.ymin_scaled >= bbox.ymax_scaled):
 220         skipped_boxes += 1
 221         continue
 222 
 223       # Note bbox.filename occasionally contains '%s' in the name. This is
 224       # data set noise that is fixed by just using the basename of the XML file.
 225       image_filename = os.path.splitext(os.path.basename(one_file))[0]
 226       print('%s.JPEG,%.4f,%.4f,%.4f,%.4f' %
 227             (image_filename,
 228              bbox.xmin_scaled, bbox.ymin_scaled,
 229              bbox.xmax_scaled, bbox.ymax_scaled))
 230 
 231       saved_boxes += 1
 232       found_box = True
 233     if found_box:
 234       saved_files += 1
 235     else:
 236       skipped_files += 1
 237 
 238     if not file_index % 5000:
 239       print('--> processed %d of %d XML files.' %
 240             (file_index + 1, len(xml_files)),
 241             file=sys.stderr)
 242       print('--> skipped %d boxes and %d XML files.' %
 243             (skipped_boxes, skipped_files), file=sys.stderr)
 244 
 245   print('Finished processing %d XML files.' % len(xml_files), file=sys.stderr)
 246   print('Skipped %d XML files not in ImageNet Challenge.' % skipped_files,
 247         file=sys.stderr)
 248   print('Skipped %d bounding boxes not in ImageNet Challenge.' % skipped_boxes,
 249         file=sys.stderr)
 250   print('Wrote %d bounding boxes from %d annotated images.' %
 251         (saved_boxes, saved_files),
 252         file=sys.stderr)
 253   print('Finished.', file=sys.stderr)
