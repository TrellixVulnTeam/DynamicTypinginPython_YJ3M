 242 def _configure_learning_rate(num_samples_per_epoch, global_step):
 243   """Configures the learning rate.
 244 
 245   Args:
 246     num_samples_per_epoch: The number of samples in each epoch of training.
 247     global_step: The global_step tensor.
 248 
 249   Returns:
 250     A `Tensor` representing the learning rate.
 251 
 252   Raises:
 253     ValueError: if
 254   """
 255   # Note: when num_clones is > 1, this will actually have each clone to go
 256   # over each epoch FLAGS.num_epochs_per_decay times. This is different
 257   # behavior from sync replicas and is expected to produce different results.
 258   steps_per_epoch = num_samples_per_epoch / FLAGS.batch_size
 259   if FLAGS.sync_replicas:
 260     steps_per_epoch /= FLAGS.replicas_to_aggregate
 261 
 262   decay_steps = int(steps_per_epoch * FLAGS.num_epochs_per_decay)
 263 
 264   if FLAGS.learning_rate_decay_type == 'exponential':
 265     learning_rate = tf.train.exponential_decay(
 266         FLAGS.learning_rate,
 267         global_step,
 268         decay_steps,
 269         FLAGS.learning_rate_decay_factor,
 270         staircase=True,
 271         name='exponential_decay_learning_rate')
 272   elif FLAGS.learning_rate_decay_type == 'fixed':
 273     learning_rate = tf.constant(FLAGS.learning_rate, name='fixed_learning_rate')
 274   elif FLAGS.learning_rate_decay_type == 'polynomial':
 275     learning_rate = tf.train.polynomial_decay(
 276         FLAGS.learning_rate,
 277         global_step,
 278         decay_steps,
 279         FLAGS.end_learning_rate,
 280         power=1.0,
 281         cycle=False,
 282         name='polynomial_decay_learning_rate')
 283   else:
 284     raise ValueError('learning_rate_decay_type [%s] was not recognized' %
 285                      FLAGS.learning_rate_decay_type)
 286 
 287   if FLAGS.warmup_epochs:
 288     warmup_lr = (
 289         FLAGS.learning_rate * tf.cast(global_step, tf.float32) /
 290         (steps_per_epoch * FLAGS.warmup_epochs))
 291     learning_rate = tf.minimum(warmup_lr, learning_rate)
 292   return learning_rate

 295 def _configure_optimizer(learning_rate):
 296   """Configures the optimizer used for training.
 297 
 298   Args:
 299     learning_rate: A scalar or `Tensor` learning rate.
 300 
 301   Returns:
 302     An instance of an optimizer.
 303 
 304   Raises:
 305     ValueError: if FLAGS.optimizer is not recognized.
 306   """
 307   if FLAGS.optimizer == 'adadelta':
 308     optimizer = tf.train.AdadeltaOptimizer(
 309         learning_rate,
 310         rho=FLAGS.adadelta_rho,
 311         epsilon=FLAGS.opt_epsilon)
 312   elif FLAGS.optimizer == 'adagrad':
 313     optimizer = tf.train.AdagradOptimizer(
 314         learning_rate,
 315         initial_accumulator_value=FLAGS.adagrad_initial_accumulator_value)
 316   elif FLAGS.optimizer == 'adam':
 317     optimizer = tf.train.AdamOptimizer(
 318         learning_rate,
 319         beta1=FLAGS.adam_beta1,
 320         beta2=FLAGS.adam_beta2,
 321         epsilon=FLAGS.opt_epsilon)
 322   elif FLAGS.optimizer == 'ftrl':
 323     optimizer = tf.train.FtrlOptimizer(
 324         learning_rate,
 325         learning_rate_power=FLAGS.ftrl_learning_rate_power,
 326         initial_accumulator_value=FLAGS.ftrl_initial_accumulator_value,
 327         l1_regularization_strength=FLAGS.ftrl_l1,
 328         l2_regularization_strength=FLAGS.ftrl_l2)
 329   elif FLAGS.optimizer == 'momentum':
 330     optimizer = tf.train.MomentumOptimizer(
 331         learning_rate,
 332         momentum=FLAGS.momentum,
 333         name='Momentum')
 334   elif FLAGS.optimizer == 'rmsprop':
 335     optimizer = tf.train.RMSPropOptimizer(
 336         learning_rate,
 337         decay=FLAGS.rmsprop_decay,
 338         momentum=FLAGS.rmsprop_momentum,
 339         epsilon=FLAGS.opt_epsilon)
 340   elif FLAGS.optimizer == 'sgd':
 341     optimizer = tf.train.GradientDescentOptimizer(learning_rate)
 342   else:
 343     raise ValueError('Optimizer [%s] was not recognized' % FLAGS.optimizer)
 344   return optimizer

 347 def _get_init_fn():
 348   """Returns a function run by the chief worker to warm-start the training.
 349 
 350   Note that the init_fn is only run when initializing the model during the very
 351   first global step.
 352 
 353   Returns:
 354     An init function run by the supervisor.
 355   """
 356   if FLAGS.checkpoint_path is None:
 357     return None
 358 
 359   # Warn the user if a checkpoint exists in the train_dir. Then we'll be
 360   # ignoring the checkpoint anyway.
 361   if tf.train.latest_checkpoint(FLAGS.train_dir):
 362     tf.logging.info(
 363         'Ignoring --checkpoint_path because a checkpoint already exists in %s'
 364         % FLAGS.train_dir)
 365     return None
 366 
 367   exclusions = []
 368   if FLAGS.checkpoint_exclude_scopes:
 369     exclusions = [scope.strip()
 370                   for scope in FLAGS.checkpoint_exclude_scopes.split(',')]
 371 
 372   # TODO(sguada) variables.filter_variables()
 373   variables_to_restore = []
 374   for var in slim.get_model_variables():
 375     for exclusion in exclusions:
 376       if var.op.name.startswith(exclusion):
 377         break
 378     else:
 379       variables_to_restore.append(var)
 380 
 381   if tf.gfile.IsDirectory(FLAGS.checkpoint_path):
 382     checkpoint_path = tf.train.latest_checkpoint(FLAGS.checkpoint_path)
 383   else:
 384     checkpoint_path = FLAGS.checkpoint_path
 385 
 386   tf.logging.info('Fine-tuning from %s' % checkpoint_path)
 387 
 388   return slim.assign_from_checkpoint_fn(
 389       checkpoint_path,
 390       variables_to_restore,
 391       ignore_missing_vars=FLAGS.ignore_missing_vars)

 394 def _get_variables_to_train():
 395   """Returns a list of variables to train.
 396 
 397   Returns:
 398     A list of variables to train by the optimizer.
 399   """
 400   if FLAGS.trainable_scopes is None:
 401     return tf.trainable_variables()
 402   else:
 403     scopes = [scope.strip() for scope in FLAGS.trainable_scopes.split(',')]
 404 
 405   variables_to_train = []
 406   for scope in scopes:
 407     variables = tf.get_collection(tf.GraphKeys.TRAINABLE_VARIABLES, scope)
 408     variables_to_train.extend(variables)
 409   return variables_to_train

 412 def main(_):
 413   if not FLAGS.dataset_dir:
 414     raise ValueError('You must supply the dataset directory with --dataset_dir')
 415 
 416   tf.logging.set_verbosity(tf.logging.INFO)
 417   with tf.Graph().as_default():
 418     #######################
 419     # Config model_deploy #
 420     #######################
 421     deploy_config = model_deploy.DeploymentConfig(
 422         num_clones=FLAGS.num_clones,
 423         clone_on_cpu=FLAGS.clone_on_cpu,
 424         replica_id=FLAGS.task,
 425         num_replicas=FLAGS.worker_replicas,
 426         num_ps_tasks=FLAGS.num_ps_tasks)
 427 
 428     # Create global_step
 429     with tf.device(deploy_config.variables_device()):
 430       global_step = slim.create_global_step()
 431 
 432     ######################
 433     # Select the dataset #
 434     ######################
 435     dataset = dataset_factory.get_dataset(
 436         FLAGS.dataset_name, FLAGS.dataset_split_name, FLAGS.dataset_dir)
 437 
 438     ######################
 439     # Select the network #
 440     ######################
 441     network_fn = nets_factory.get_network_fn(
 442         FLAGS.model_name,
 443         num_classes=(dataset.num_classes - FLAGS.labels_offset),
 444         weight_decay=FLAGS.weight_decay,
 445         is_training=True)
 446 
 447     #####################################
 448     # Select the preprocessing function #
 449     #####################################
 450     preprocessing_name = FLAGS.preprocessing_name or FLAGS.model_name
 451     image_preprocessing_fn = preprocessing_factory.get_preprocessing(
 452         preprocessing_name,
 453         is_training=True,
 454         use_grayscale=FLAGS.use_grayscale)
 455 
 456     ##############################################################
 457     # Create a dataset provider that loads data from the dataset #
 458     ##############################################################
 459     with tf.device(deploy_config.inputs_device()):
 460       provider = slim.dataset_data_provider.DatasetDataProvider(
 461           dataset,
 462           num_readers=FLAGS.num_readers,
 463           common_queue_capacity=20 * FLAGS.batch_size,
 464           common_queue_min=10 * FLAGS.batch_size)
 465       [image, label] = provider.get(['image', 'label'])
 466       label -= FLAGS.labels_offset
 467 
 468       train_image_size = FLAGS.train_image_size or network_fn.default_image_size
 469 
 470       image = image_preprocessing_fn(image, train_image_size, train_image_size)
 471 
 472       images, labels = tf.train.batch(
 473           [image, label],
 474           batch_size=FLAGS.batch_size,
 475           num_threads=FLAGS.num_preprocessing_threads,
 476           capacity=5 * FLAGS.batch_size)
 477       labels = slim.one_hot_encoding(
 478           labels, dataset.num_classes - FLAGS.labels_offset)
 479       batch_queue = slim.prefetch_queue.prefetch_queue(
 480           [images, labels], capacity=2 * deploy_config.num_clones)
 481 
 482     ####################
 483     # Define the model #
 484     ####################
 485     def clone_fn(batch_queue):
 486       """Allows data parallelism by creating multiple clones of network_fn."""
 487       images, labels = batch_queue.dequeue()
 488       logits, end_points = network_fn(images)
 489 
 490       #############################
 491       # Specify the loss function #
 492       #############################
 493       if 'AuxLogits' in end_points:
 494         slim.losses.softmax_cross_entropy(
 495             end_points['AuxLogits'], labels,
 496             label_smoothing=FLAGS.label_smoothing, weights=0.4,
 497             scope='aux_loss')
 498       slim.losses.softmax_cross_entropy(
 499           logits, labels, label_smoothing=FLAGS.label_smoothing, weights=1.0)
 500       return end_points
 501 
 502     # Gather initial summaries.
 503     summaries = set(tf.get_collection(tf.GraphKeys.SUMMARIES))
 504 
 505     clones = model_deploy.create_clones(deploy_config, clone_fn, [batch_queue])
 506     first_clone_scope = deploy_config.clone_scope(0)
 507     # Gather update_ops from the first clone. These contain, for example,
 508     # the updates for the batch_norm variables created by network_fn.
 509     update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS, first_clone_scope)
 510 
 511     # Add summaries for end_points.
 512     end_points = clones[0].outputs
 513     for end_point in end_points:
 514       x = end_points[end_point]
 515       summaries.add(tf.summary.histogram('activations/' + end_point, x))
 516       summaries.add(tf.summary.scalar('sparsity/' + end_point,
 517                                       tf.nn.zero_fraction(x)))
 518 
 519     # Add summaries for losses.
 520     for loss in tf.get_collection(tf.GraphKeys.LOSSES, first_clone_scope):
 521       summaries.add(tf.summary.scalar('losses/%s' % loss.op.name, loss))
 522 
 523     # Add summaries for variables.
 524     for variable in slim.get_model_variables():
 525       summaries.add(tf.summary.histogram(variable.op.name, variable))
 526 
 527     #################################
 528     # Configure the moving averages #
 529     #################################
 530     if FLAGS.moving_average_decay:
 531       moving_average_variables = slim.get_model_variables()
 532       variable_averages = tf.train.ExponentialMovingAverage(
 533           FLAGS.moving_average_decay, global_step)
 534     else:
 535       moving_average_variables, variable_averages = None, None
 536 
 537     if FLAGS.quantize_delay >= 0:
 538       contrib_quantize.create_training_graph(quant_delay=FLAGS.quantize_delay)
 539 
 540     #########################################
 541     # Configure the optimization procedure. #
 542     #########################################
 543     with tf.device(deploy_config.optimizer_device()):
 544       learning_rate = _configure_learning_rate(dataset.num_samples, global_step)
 545       optimizer = _configure_optimizer(learning_rate)
 546       summaries.add(tf.summary.scalar('learning_rate', learning_rate))
 547 
 548     if FLAGS.sync_replicas:
 549       # If sync_replicas is enabled, the averaging will be done in the chief
 550       # queue runner.
 551       optimizer = tf.train.SyncReplicasOptimizer(
 552           opt=optimizer,
 553           replicas_to_aggregate=FLAGS.replicas_to_aggregate,
 554           total_num_replicas=FLAGS.worker_replicas,
 555           variable_averages=variable_averages,
 556           variables_to_average=moving_average_variables)
 557     elif FLAGS.moving_average_decay:
 558       # Update ops executed locally by trainer.
 559       update_ops.append(variable_averages.apply(moving_average_variables))
 560 
 561     # Variables to train.
 562     variables_to_train = _get_variables_to_train()
 563 
 564     #  and returns a train_tensor and summary_op
 565     total_loss, clones_gradients = model_deploy.optimize_clones(
 566         clones,
 567         optimizer,
 568         var_list=variables_to_train)
 569     # Add total_loss to summary.
 570     summaries.add(tf.summary.scalar('total_loss', total_loss))
 571 
 572     # Create gradient updates.
 573     grad_updates = optimizer.apply_gradients(clones_gradients,
 574                                              global_step=global_step)
 575     update_ops.append(grad_updates)
 576 
 577     update_op = tf.group(*update_ops)
 578     with tf.control_dependencies([update_op]):
 579       train_tensor = tf.identity(total_loss, name='train_op')
 580 
 581     # Add the summaries from the first clone. These contain the summaries
 582     # created by model_fn and either optimize_clones() or _gather_clone_loss().
 583     summaries |= set(tf.get_collection(tf.GraphKeys.SUMMARIES,
 584                                        first_clone_scope))
 585 
 586     # Merge all summaries together.
 587     summary_op = tf.summary.merge(list(summaries), name='summary_op')
 588 
 589     ###########################
 590     # Kicks off the training. #
 591     ###########################
 592     slim.learning.train(
 593         train_tensor,
 594         logdir=FLAGS.train_dir,
 595         master=FLAGS.master,
 596         is_chief=(FLAGS.task == 0),
 597         init_fn=_get_init_fn(),
 598         summary_op=summary_op,
 599         number_of_steps=FLAGS.max_number_of_steps,
 600         log_every_n_steps=FLAGS.log_every_n_steps,
 601         save_summaries_secs=FLAGS.save_summaries_secs,
 602         save_interval_secs=FLAGS.save_interval_secs,
 603         sync_optimizer=optimizer if FLAGS.sync_replicas else None)
 604 
 605 
 606 if __name__ == '__main__':
 607   tf.app.run()
