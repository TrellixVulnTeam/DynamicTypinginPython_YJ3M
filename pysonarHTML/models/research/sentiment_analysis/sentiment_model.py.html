   1 """Model for sentiment analysis.
   2 
   3 The model makes use of concatenation of two CNN layers with
   4 different kernel sizes.
   5 """
   6 
   7 from __future__ import absolute_import
   8 from __future__ import division
   9 from __future__ import print_function
  10 
  11 import tensorflow as tf
  12 
  13 
  14 class CNN(tf.keras.models.Model):
  15   """CNN for sentimental analysis."""
  16 
  17   def __init__(self, emb_dim, num_words, sentence_length, hid_dim,
  18                class_dim, dropout_rate):
  19     """Initialize CNN model.
  20 
  21     Args:
  22       emb_dim: The dimension of the Embedding layer.
  23       num_words: The number of the most frequent tokens
  24         to be used from the corpus.
  25       sentence_length: The number of words in each sentence.
  26         Longer sentences get cut, shorter ones padded.
  27       hid_dim: The dimension of the Embedding layer.
  28       class_dim: The number of the CNN layer filters.
  29       dropout_rate: The portion of kept value in the Dropout layer.
  30     Returns:
  31       tf.keras.models.Model: A Keras model.
  32     """
  33 
  34     input_layer = tf.keras.layers.Input(shape=(sentence_length,), dtype=tf.int32)
  35 
  36     layer = tf.keras.layers.Embedding(num_words, output_dim=emb_dim)(input_layer)
  37 
  38     layer_conv3 = tf.keras.layers.Conv1D(hid_dim, 3, activation="relu")(layer)
  39     layer_conv3 = tf.keras.layers.GlobalMaxPooling1D()(layer_conv3)
  40 
  41     layer_conv4 = tf.keras.layers.Conv1D(hid_dim, 2, activation="relu")(layer)
  42     layer_conv4 = tf.keras.layers.GlobalMaxPooling1D()(layer_conv4)
  43 
  44     layer = tf.keras.layers.concatenate([layer_conv4, layer_conv3], axis=1)
  45     layer = tf.keras.layers.BatchNormalization()(layer)
  46     layer = tf.keras.layers.Dropout(dropout_rate)(layer)
  47 
  48     output = tf.keras.layers.Dense(class_dim, activation="softmax")(layer)
  49 
  50     super(CNN, self).__init__(inputs=[input_layer], outputs=output)
