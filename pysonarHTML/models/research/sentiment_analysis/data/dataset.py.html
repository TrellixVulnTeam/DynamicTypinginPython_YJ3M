  15 def load(dataset, vocabulary_size, sentence_length):
  16   """Returns training and evaluation input.
  17 
  18   Args:
  19     dataset: Dataset to be trained and evaluated.
  20       Currently only imdb is supported.
  21     vocabulary_size: The number of the most frequent tokens
  22       to be used from the corpus.
  23     sentence_length: The number of words in each sentence.
  24       Longer sentences get cut, shorter ones padded.
  25   Raises:
  26     ValueError: if the dataset value is not valid.
  27   Returns:
  28     A tuple of length 4, for training sentences, labels,
  29     evaluation sentences, and evaluation labels,
  30     each being an numpy array.
  31   """
  32   if dataset == DATASET_IMDB:
  33     return imdb.load(vocabulary_size, sentence_length)
  34   else:
  35     raise ValueError("unsupported dataset: " + dataset)

  38 def get_num_class(dataset):
  39   """Returns an integer for the number of label classes.
  40 
  41   Args:
  42     dataset: Dataset to be trained and evaluated.
  43       Currently only imdb is supported.
  44   Raises:
  45     ValueError: if the dataset value is not valid.
  46   Returns:
  47     int: The number of label classes.
  48   """
  49   if dataset == DATASET_IMDB:
  50     return imdb.NUM_CLASS
  51   else:
  52     raise ValueError("unsupported dataset: " + dataset)
