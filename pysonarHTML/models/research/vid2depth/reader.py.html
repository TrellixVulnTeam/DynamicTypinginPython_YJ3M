  34 class DataReader(object):
  35   """Reads stored sequences which are produced by dataset/gen_data.py."""

  37   def __init__(self, data_dir, batch_size, img_height, img_width, seq_length,
  38                num_scales):
  39     self.data_dir = data_dir
  40     self.batch_size = batch_size
  41     self.img_height = img_height
  42     self.img_width = img_width
  43     self.seq_length = seq_length
  44     self.num_scales = num_scales
  45 
  46   def read_data(self):
  47     """Provides images and camera intrinsics."""
  48     with tf.name_scope('data_loading'):
  49       with tf.name_scope('enqueue_paths'):
  50         seed = random.randint(0, 2**31 - 1)
  51         self.file_lists = self.compile_file_list(self.data_dir, 'train')
  52         image_paths_queue = tf.train.string_input_producer(
  53             self.file_lists['image_file_list'], seed=seed, shuffle=True)
  54         cam_paths_queue = tf.train.string_input_producer(
  55             self.file_lists['cam_file_list'], seed=seed, shuffle=True)
  56         img_reader = tf.WholeFileReader()
  57         _, image_contents = img_reader.read(image_paths_queue)
  58         image_seq = tf.image.decode_jpeg(image_contents)
  59 
  60       with tf.name_scope('load_intrinsics'):
  61         cam_reader = tf.TextLineReader()
  62         _, raw_cam_contents = cam_reader.read(cam_paths_queue)
  63         rec_def = []
  64         for _ in range(9):
  65           rec_def.append([1.0])
  66         raw_cam_vec = tf.decode_csv(raw_cam_contents, record_defaults=rec_def)
  67         raw_cam_vec = tf.stack(raw_cam_vec)
  68         intrinsics = tf.reshape(raw_cam_vec, [3, 3])
  69 
  70       with tf.name_scope('convert_image'):
  71         image_seq = self.preprocess_image(image_seq)  # Converts to float.
  72 
  73       with tf.name_scope('image_augmentation'):
  74         image_seq = self.augment_image_colorspace(image_seq)
  75 
  76       image_stack = self.unpack_images(image_seq)
  77 
  78       with tf.name_scope('image_augmentation_scale_crop'):
  79         image_stack, intrinsics = self.augment_images_scale_crop(
  80             image_stack, intrinsics, self.img_height, self.img_width)
  81 
  82       with tf.name_scope('multi_scale_intrinsics'):
  83         intrinsic_mat = self.get_multi_scale_intrinsics(intrinsics,
  84                                                         self.num_scales)
  85         intrinsic_mat.set_shape([self.num_scales, 3, 3])
  86         intrinsic_mat_inv = tf.matrix_inverse(intrinsic_mat)
  87         intrinsic_mat_inv.set_shape([self.num_scales, 3, 3])
  88 
  89       with tf.name_scope('batching'):
  90         image_stack, intrinsic_mat, intrinsic_mat_inv = (
  91             tf.train.shuffle_batch(
  92                 [image_stack, intrinsic_mat, intrinsic_mat_inv],
  93                 batch_size=self.batch_size,
  94                 capacity=QUEUE_SIZE + QUEUE_BUFFER * self.batch_size,
  95                 min_after_dequeue=QUEUE_SIZE))
  96         logging.info('image_stack: %s', util.info(image_stack))
  97     return image_stack, intrinsic_mat, intrinsic_mat_inv
  98 
  99   def unpack_images(self, image_seq):
 100     """[h, w * seq_length, 3] -> [h, w, 3 * seq_length]."""
 101     with tf.name_scope('unpack_images'):
 102       image_list = [
 103           image_seq[:, i * self.img_width:(i + 1) * self.img_width, :]
 104           for i in range(self.seq_length)
 105       ]
 106       image_stack = tf.concat(image_list, axis=2)
 107       image_stack.set_shape(
 108           [self.img_height, self.img_width, self.seq_length * 3])
 109     return image_stack

 111   @classmethod
 112   def preprocess_image(cls, image):
 113     # Convert from uint8 to float.
 114     return tf.image.convert_image_dtype(image, dtype=tf.float32)

 116   # Source: https://github.com/mrharicot/monodepth.
 117   @classmethod
 118   def augment_image_colorspace(cls, image_seq):
 119     """Apply data augmentation to inputs."""
 120     # Randomly shift gamma.
 121     random_gamma = tf.random_uniform([], 0.8, 1.2)
 122     image_seq_aug = image_seq**random_gamma
 123     # Randomly shift brightness.
 124     random_brightness = tf.random_uniform([], 0.5, 2.0)
 125     image_seq_aug *= random_brightness
 126     # Randomly shift color.
 127     random_colors = tf.random_uniform([3], 0.8, 1.2)
 128     white = tf.ones([tf.shape(image_seq)[0], tf.shape(image_seq)[1]])
 129     color_image = tf.stack([white * random_colors[i] for i in range(3)], axis=2)
 130     image_seq_aug *= color_image
 131     # Saturate.
 132     image_seq_aug = tf.clip_by_value(image_seq_aug, 0, 1)
 133     return image_seq_aug

 135   @classmethod
 136   def augment_images_scale_crop(cls, im, intrinsics, out_h, out_w):
 137     """Randomly scales and crops image."""
 138 
 139     def scale_randomly(im, intrinsics):
 140       """Scales image and adjust intrinsics accordingly."""
 141       in_h, in_w, _ = im.get_shape().as_list()
 142       scaling = tf.random_uniform([2], 1, 1.15)
 143       x_scaling = scaling[0]
 144       y_scaling = scaling[1]
 145       out_h = tf.cast(in_h * y_scaling, dtype=tf.int32)
 146       out_w = tf.cast(in_w * x_scaling, dtype=tf.int32)
 147       # Add batch.
 148       im = tf.expand_dims(im, 0)
 149       im = tf.image.resize_area(im, [out_h, out_w])
 150       im = im[0]
 151       fx = intrinsics[0, 0] * x_scaling
 152       fy = intrinsics[1, 1] * y_scaling
 153       cx = intrinsics[0, 2] * x_scaling
 154       cy = intrinsics[1, 2] * y_scaling
 155       intrinsics = cls.make_intrinsics_matrix(fx, fy, cx, cy)
 156       return im, intrinsics
 157 
 158     # Random cropping
 159     def crop_randomly(im, intrinsics, out_h, out_w):
 160       """Crops image and adjust intrinsics accordingly."""
 161       # batch_size, in_h, in_w, _ = im.get_shape().as_list()
 162       in_h, in_w, _ = tf.unstack(tf.shape(im))
 163       offset_y = tf.random_uniform([1], 0, in_h - out_h + 1, dtype=tf.int32)[0]
 164       offset_x = tf.random_uniform([1], 0, in_w - out_w + 1, dtype=tf.int32)[0]
 165       im = tf.image.crop_to_bounding_box(im, offset_y, offset_x, out_h, out_w)
 166       fx = intrinsics[0, 0]
 167       fy = intrinsics[1, 1]
 168       cx = intrinsics[0, 2] - tf.cast(offset_x, dtype=tf.float32)
 169       cy = intrinsics[1, 2] - tf.cast(offset_y, dtype=tf.float32)
 170       intrinsics = cls.make_intrinsics_matrix(fx, fy, cx, cy)
 171       return im, intrinsics
 172 
 173     im, intrinsics = scale_randomly(im, intrinsics)
 174     im, intrinsics = crop_randomly(im, intrinsics, out_h, out_w)
 175     return im, intrinsics

 177   def compile_file_list(self, data_dir, split, load_pose=False):
 178     """Creates a list of input files."""
 179     logging.info('data_dir: %s', data_dir)
 180     with gfile.Open(os.path.join(data_dir, '%s.txt' % split), 'r') as f:
 181       frames = f.readlines()
 182     subfolders = [x.split(' ')[0] for x in frames]
 183     frame_ids = [x.split(' ')[1][:-1] for x in frames]
 184     image_file_list = [
 185         os.path.join(data_dir, subfolders[i], frame_ids[i] + '.jpg')
 186         for i in range(len(frames))
 187     ]
 188     cam_file_list = [
 189         os.path.join(data_dir, subfolders[i], frame_ids[i] + '_cam.txt')
 190         for i in range(len(frames))
 191     ]
 192     file_lists = {}
 193     file_lists['image_file_list'] = image_file_list
 194     file_lists['cam_file_list'] = cam_file_list
 195     if load_pose:
 196       pose_file_list = [
 197           os.path.join(data_dir, subfolders[i], frame_ids[i] + '_pose.txt')
 198           for i in range(len(frames))
 199       ]
 200       file_lists['pose_file_list'] = pose_file_list
 201     self.steps_per_epoch = len(image_file_list) // self.batch_size
 202     return file_lists

 204   @classmethod
 205   def make_intrinsics_matrix(cls, fx, fy, cx, cy):
 206     r1 = tf.stack([fx, 0, cx])
 207     r2 = tf.stack([0, fy, cy])
 208     r3 = tf.constant([0., 0., 1.])
 209     intrinsics = tf.stack([r1, r2, r3])
 210     return intrinsics
 211 
 212   @classmethod
 213   def get_multi_scale_intrinsics(cls, intrinsics, num_scales):
 214     """Returns multiple intrinsic matrices for different scales."""
 215     intrinsics_multi_scale = []
 216     # Scale the intrinsics accordingly for each scale
 217     for s in range(num_scales):
 218       fx = intrinsics[0, 0] / (2**s)
 219       fy = intrinsics[1, 1] / (2**s)
 220       cx = intrinsics[0, 2] / (2**s)
 221       cy = intrinsics[1, 2] / (2**s)
 222       intrinsics_multi_scale.append(cls.make_intrinsics_matrix(fx, fy, cx, cy))
 223     intrinsics_multi_scale = tf.stack(intrinsics_multi_scale)
 224     return intrinsics_multi_scale
