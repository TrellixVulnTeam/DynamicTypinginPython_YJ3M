  50 flags.DEFINE_enum('dataset_name', None, DATASETS, 'Dataset name.')

  65 # Process data in chunks for reporting progress.
  66 NUM_CHUNKS = 100
  67 
  68 
  69 def _generate_data():
  70   """Extract sequences from dataset_dir and store them in data_dir."""
  71   if not gfile.Exists(FLAGS.data_dir):
  72     gfile.MakeDirs(FLAGS.data_dir)
  73 
  74   global dataloader  # pylint: disable=global-variable-undefined
  75   if FLAGS.dataset_name == 'bike':
  76     dataloader = dataset_loader.Bike(FLAGS.dataset_dir,
  77                                      img_height=FLAGS.img_height,
  78                                      img_width=FLAGS.img_width,
  79                                      seq_length=FLAGS.seq_length)
  80   elif FLAGS.dataset_name == 'kitti_odom':
  81     dataloader = dataset_loader.KittiOdom(FLAGS.dataset_dir,
  82                                           img_height=FLAGS.img_height,
  83                                           img_width=FLAGS.img_width,
  84                                           seq_length=FLAGS.seq_length)
  85   elif FLAGS.dataset_name == 'kitti_raw_eigen':
  86     dataloader = dataset_loader.KittiRaw(FLAGS.dataset_dir,
  87                                          split='eigen',
  88                                          img_height=FLAGS.img_height,
  89                                          img_width=FLAGS.img_width,
  90                                          seq_length=FLAGS.seq_length)
  91   elif FLAGS.dataset_name == 'kitti_raw_stereo':
  92     dataloader = dataset_loader.KittiRaw(FLAGS.dataset_dir,
  93                                          split='stereo',
  94                                          img_height=FLAGS.img_height,
  95                                          img_width=FLAGS.img_width,
  96                                          seq_length=FLAGS.seq_length)
  97   elif FLAGS.dataset_name == 'cityscapes':
  98     dataloader = dataset_loader.Cityscapes(FLAGS.dataset_dir,
  99                                            img_height=FLAGS.img_height,
 100                                            img_width=FLAGS.img_width,
 101                                            seq_length=FLAGS.seq_length)
 102   else:
 103     raise ValueError('Unknown dataset')
 104 
 105   # The default loop below uses multiprocessing, which can make it difficult
 106   # to locate source of errors in data loader classes.
 107   # Uncomment this loop for easier debugging:
 108 
 109   # all_examples = {}
 110   # for i in range(dataloader.num_train):
 111   #   _gen_example(i, all_examples)
 112   #   logging.info('Generated: %d', len(all_examples))
 113 
 114   all_frames = range(dataloader.num_train)
 115   frame_chunks = np.array_split(all_frames, NUM_CHUNKS)
 116 
 117   manager = multiprocessing.Manager()
 118   all_examples = manager.dict()
 119   num_cores = multiprocessing.cpu_count()
 120   num_threads = num_cores if FLAGS.num_threads is None else FLAGS.num_threads
 121   pool = multiprocessing.Pool(num_threads)
 122 
 123   # Split into training/validation sets. Fixed seed for repeatability.
 124   np.random.seed(8964)
 125 
 126   if not gfile.Exists(FLAGS.data_dir):
 127     gfile.MakeDirs(FLAGS.data_dir)
 128 
 129   with gfile.Open(os.path.join(FLAGS.data_dir, 'train.txt'), 'w') as train_f:
 130     with gfile.Open(os.path.join(FLAGS.data_dir, 'val.txt'), 'w') as val_f:
 131       logging.info('Generating data...')
 132       for index, frame_chunk in enumerate(frame_chunks):
 133         all_examples.clear()
 134         pool.map(_gen_example_star,
 135                  itertools.izip(frame_chunk, itertools.repeat(all_examples)))
 136         logging.info('Chunk %d/%d: saving %s entries...', index + 1, NUM_CHUNKS,
 137                      len(all_examples))
 138         for _, example in all_examples.items():
 139           if example:
 140             s = example['folder_name']
 141             frame = example['file_name']
 142             if np.random.random() < 0.1:
 143               val_f.write('%s %s\n' % (s, frame))
 144             else:
 145               train_f.write('%s %s\n' % (s, frame))
 146   pool.close()
 147   pool.join()
 148

 150 def _gen_example(i, all_examples):
 151   """Saves one example to file.  Also adds it to all_examples dict."""
 152   example = dataloader.get_example_with_index(i)
 153   if not example:
 154     return
 155   image_seq_stack = _stack_image_seq(example['image_seq'])
 156   example.pop('image_seq', None)  # Free up memory.
 157   intrinsics = example['intrinsics']
 158   fx = intrinsics[0, 0]
 159   fy = intrinsics[1, 1]
 160   cx = intrinsics[0, 2]
 161   cy = intrinsics[1, 2]
 162   save_dir = os.path.join(FLAGS.data_dir, example['folder_name'])
 163   if not gfile.Exists(save_dir):
 164     gfile.MakeDirs(save_dir)
 165   img_filepath = os.path.join(save_dir, '%s.jpg' % example['file_name'])
 166   scipy.misc.imsave(img_filepath, image_seq_stack.astype(np.uint8))
 167   cam_filepath = os.path.join(save_dir, '%s_cam.txt' % example['file_name'])
 168   example['cam'] = '%f,0.,%f,0.,%f,%f,0.,0.,1.' % (fx, cx, fy, cy)
 169   with open(cam_filepath, 'w') as cam_f:
 170     cam_f.write(example['cam'])
 171 
 172   key = example['folder_name'] + '_' + example['file_name']
 173   all_examples[key] = example
 174

 176 def _gen_example_star(params):
 177   return _gen_example(*params)

 180 def _stack_image_seq(seq):
 181   for i, im in enumerate(seq):
 182     if i == 0:
 183       res = im
 184     else:
 185       res = np.hstack((res, im))
 186   return res
