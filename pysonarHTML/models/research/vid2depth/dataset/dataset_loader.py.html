  55   def collect_frames(self):
  56     """Create a list of unique ids for available frames."""
  57     video_list = os.listdir(self.dataset_dir)
  58     logging.info('video_list: %s', video_list)
  59     frames = []
  60     for video in video_list:
  61       im_files = glob.glob(os.path.join(self.dataset_dir, video, '*.jpg'))
  62       im_files = sorted(im_files, key=natural_keys)
  63       # Adding 3 crops of the video.
  64       frames.extend(['A' + video + '/' + os.path.basename(f) for f in im_files])
  65       frames.extend(['B' + video + '/' + os.path.basename(f) for f in im_files])
  66       frames.extend(['C' + video + '/' + os.path.basename(f) for f in im_files])
  67     return frames

  75   def load_intrinsics(self, unused_frame_idx, cy):
  76     """Load intrinsics."""
  77     # https://www.wired.com/2013/05/calculating-the-angular-view-of-an-iphone/
  78     # https://codeyarns.com/2015/09/08/how-to-compute-intrinsic-camera-matrix-for-a-camera/
  79     # https://stackoverflow.com/questions/39992968/how-to-calculate-field-of-view-of-the-camera-from-camera-intrinsic-matrix
  80     # # iPhone: These numbers are for images with resolution 720 x 1280.
  81     # Assuming FOV = 50.9 => fx = (1280 // 2) / math.tan(fov / 2) = 1344.8
  82     intrinsics = np.array([[1344.8, 0, 1280 // 2],
  83                            [0, 1344.8, cy],
  84                            [0, 0, 1.0]])
  85     return intrinsics

  87   def is_valid_sample(self, target_index):
  88     """Checks whether we can find a valid sequence around this frame."""
  89     target_video, _ = self.frames[target_index].split('/')
  90     start_index, end_index = get_seq_start_end(target_index,
  91                                                self.seq_length,

 101   def load_image_raw(self, frame_id):
 102     """Reads the image and crops it according to first letter of frame_id."""
 103     crop_type = frame_id[0]
 104     img_file = os.path.join(self.dataset_dir, frame_id[1:])
 105     img = scipy.misc.imread(img_file)
 106     allowed_height = int(img.shape[1] * self.img_height / self.img_width)
 107     # Starting height for the middle crop.
 108     mid_crop_top = int(img.shape[0] / 2 - allowed_height / 2)
 109     # How much to go up or down to get the other two crops.
 110     height_var = int(mid_crop_top / 3)
 111     if crop_type == 'A':
 112       crop_top = mid_crop_top - height_var
 113       cy = allowed_height / 2 + height_var
 114     elif crop_type == 'B':
 115       crop_top = mid_crop_top
 116       cy = allowed_height / 2
 117     elif crop_type == 'C':
 118       crop_top = mid_crop_top + height_var
 119       cy = allowed_height / 2 - height_var
 120     else:
 121       raise ValueError('Unknown crop_type: %s' % crop_type)
 122     crop_bottom = crop_top + allowed_height + 1
 123     return img[crop_top:crop_bottom, :, :], cy

 125   def load_image_sequence(self, target_index):
 126     """Returns a list of images around target index."""
 127     start_index, end_index = get_seq_start_end(target_index,
 128                                                self.seq_length,
 129                                                self.sample_every)
 130     image_seq = []
 131     for idx in range(start_index, end_index + 1, self.sample_every):
 132       frame_id = self.frames[idx]
 133       img, cy = self.load_image_raw(frame_id)
 134       if idx == target_index:
 135         zoom_y = self.img_height / img.shape[0]
 136         zoom_x = self.img_width / img.shape[1]
 137       img = scipy.misc.imresize(img, (self.img_height, self.img_width))
 138       image_seq.append(img)
 139     return image_seq, zoom_x, zoom_y, cy

 141   def load_example(self, target_index):
 142     """Returns a sequence with requested target frame."""
 143     image_seq, zoom_x, zoom_y, cy = self.load_image_sequence(target_index)
 144     target_video, target_filename = self.frames[target_index].split('/')
 145     # Put A, B, C at the end for better shuffling.
 146     target_video = target_video[1:] + target_video[0]
 147     intrinsics = self.load_intrinsics(target_index, cy)
 148     intrinsics = self.scale_intrinsics(intrinsics, zoom_x, zoom_y)
 149     example = {}
 150     example['intrinsics'] = intrinsics
 151     example['image_seq'] = image_seq
 152     example['folder_name'] = target_video
 153     example['file_name'] = target_filename.split('.')[0]
 154     return example

 156   def scale_intrinsics(self, mat, sx, sy):

 158     out[0, 0] *= sx
 159     out[0, 2] *= sx
 160     out[1, 1] *= sy
 161     out[1, 2] *= sy
 162     return out

 168   def __init__(self,
 169                dataset_dir,
 170                split,
 171                load_pose=False,
 172                img_height=128,
 173                img_width=416,
 174                seq_length=3):
 175     static_frames_file = 'dataset/kitti/static_frames.txt'
 176     test_scene_file = 'dataset/kitti/test_scenes_' + split + '.txt'
 177     with open(get_resource_path(test_scene_file), 'r') as f:
 178       test_scenes = f.readlines()
 179     self.test_scenes = [t[:-1] for t in test_scenes]
 180     self.dataset_dir = dataset_dir
 181     self.img_height = img_height
 182     self.img_width = img_width
 183     self.seq_length = seq_length
 184     self.load_pose = load_pose
 185     self.cam_ids = ['02', '03']
 186     self.date_list = [
 187         '2011_09_26', '2011_09_28', '2011_09_29', '2011_09_30', '2011_10_03'
 188     ]
 189     self.collect_static_frames(static_frames_file)
 190     self.collect_train_frames()

 192   def collect_static_frames(self, static_frames_file):
 193     with open(get_resource_path(static_frames_file), 'r') as f:
 194       frames = f.readlines()
 195     self.static_frames = []
 196     for fr in frames:
 197       if fr == '\n':
 198         continue
 199       unused_date, drive, frame_id = fr.split(' ')
 200       fid = '%.10d' % (np.int(frame_id[:-1]))
 201       for cam_id in self.cam_ids:
 202         self.static_frames.append(drive + ' ' + cam_id + ' ' + fid)
 203 
 204   def collect_train_frames(self):
 205     """Creates a list of training frames."""
 206     all_frames = []
 207     for date in self.date_list:
 208       date_dir = os.path.join(self.dataset_dir, date)
 209       drive_set = os.listdir(date_dir)
 210       for dr in drive_set:
 211         drive_dir = os.path.join(date_dir, dr)
 212         if os.path.isdir(drive_dir):
 213           if dr[:-5] in self.test_scenes:
 214             continue
 215           for cam in self.cam_ids:
 216             img_dir = os.path.join(drive_dir, 'image_' + cam, 'data')
 217             num_frames = len(glob.glob(img_dir + '/*[0-9].png'))
 218             for i in range(num_frames):
 219               frame_id = '%.10d' % i
 220               all_frames.append(dr + ' ' + cam + ' ' + frame_id)
 221 
 222     for s in self.static_frames:
 223       try:
 224         all_frames.remove(s)
 225       except ValueError:
 226         pass
 227 
 228     self.train_frames = all_frames
 229     self.num_train = len(self.train_frames)

 231   def is_valid_sample(self, frames, target_index):
 232     """Checks whether we can find a valid sequence around this frame."""
 233     num_frames = len(frames)
 234     target_drive, cam_id, _ = frames[target_index].split(' ')
 235     start_index, end_index = get_seq_start_end(target_index, self.seq_length)
 236     if start_index < 0 or end_index >= num_frames:
 237       return False
 238     start_drive, start_cam_id, _ = frames[start_index].split(' ')
 239     end_drive, end_cam_id, _ = frames[end_index].split(' ')
 240     if (target_drive == start_drive and target_drive == end_drive and
 241         cam_id == start_cam_id and cam_id == end_cam_id):
 242       return True
 243     return False

 251   def load_image_sequence(self, frames, target_index):
 252     """Returns a sequence with requested target frame."""
 253     start_index, end_index = get_seq_start_end(target_index, self.seq_length)
 254     image_seq = []
 255     for index in range(start_index, end_index + 1):
 256       drive, cam_id, frame_id = frames[index].split(' ')
 257       img = self.load_image_raw(drive, cam_id, frame_id)
 258       if index == target_index:
 259         zoom_y = self.img_height / img.shape[0]
 260         zoom_x = self.img_width / img.shape[1]
 261       img = scipy.misc.imresize(img, (self.img_height, self.img_width))
 262       image_seq.append(img)
 263     return image_seq, zoom_x, zoom_y

 265   def load_pose_sequence(self, frames, target_index):
 266     """Returns a sequence of pose vectors for frames around the target frame."""
 267     target_drive, _, target_frame_id = frames[target_index].split(' ')
 268     target_pose = self.load_pose_raw(target_drive, target_frame_id)
 269     start_index, end_index = get_seq_start_end(target_frame_id, self.seq_length)
 270     pose_seq = []
 271     for index in range(start_index, end_index + 1):
 272       if index == target_frame_id:
 273         continue
 274       drive, _, frame_id = frames[index].split(' ')
 275       pose = self.load_pose_raw(drive, frame_id)
 276       # From target to index.
 277       pose = np.dot(np.linalg.inv(pose), target_pose)
 278       pose_seq.append(pose)
 279     return pose_seq

 281   def load_example(self, frames, target_index):
 282     """Returns a sequence with requested target frame."""
 283     image_seq, zoom_x, zoom_y = self.load_image_sequence(frames, target_index)
 284     target_drive, target_cam_id, target_frame_id = (
 285         frames[target_index].split(' '))
 286     intrinsics = self.load_intrinsics_raw(target_drive, target_cam_id)
 287     intrinsics = self.scale_intrinsics(intrinsics, zoom_x, zoom_y)
 288     example = {}
 289     example['intrinsics'] = intrinsics
 290     example['image_seq'] = image_seq
 291     example['folder_name'] = target_drive + '_' + target_cam_id + '/'
 292     example['file_name'] = target_frame_id
 293     if self.load_pose:
 294       pose_seq = self.load_pose_sequence(frames, target_index)
 295       example['pose_seq'] = pose_seq
 296     return example

 298   def load_pose_raw(self, drive, frame_id):
 299     date = drive[:10]
 300     pose_file = os.path.join(self.dataset_dir, date, drive, 'poses',
 301                              frame_id + '.txt')
 302     with open(pose_file, 'r') as f:
 303       pose = f.readline()
 304     pose = np.array(pose.split(' ')).astype(np.float32).reshape(3, 4)
 305     pose = np.vstack((pose, np.array([0, 0, 0, 1]).reshape((1, 4))))
 306     return pose

 308   def load_image_raw(self, drive, cam_id, frame_id):
 309     date = drive[:10]
 310     img_file = os.path.join(self.dataset_dir, date, drive, 'image_' + cam_id,
 311                             'data', frame_id + '.png')
 312     img = scipy.misc.imread(img_file)
 313     return img

 315   def load_intrinsics_raw(self, drive, cam_id):
 316     date = drive[:10]
 317     calib_file = os.path.join(self.dataset_dir, date, 'calib_cam_to_cam.txt')
 318     filedata = self.read_raw_calib_file(calib_file)
 319     p_rect = np.reshape(filedata['P_rect_' + cam_id], (3, 4))
 320     intrinsics = p_rect[:3, :3]
 321     return intrinsics

 323   # From https://github.com/utiasSTARS/pykitti/blob/master/pykitti/utils.py
 324   def read_raw_calib_file(self, filepath):
 325     """Read in a calibration file and parse into a dictionary."""
 326     data = {}
 327     with open(filepath, 'r') as f:
 328       for line in f:
 329         key, value = line.split(':', 1)
 330         # The only non-float values in these files are dates, which we don't
 331         # care about.
 332         try:
 333           data[key] = np.array([float(x) for x in value.split()])
 334         except ValueError:
 335           pass
 336     return data

 437     calib_file = os.path.join(self.dataset_dir, 'sequences',
 438                               '%s/calib.txt' % drive)

 443   def read_calib_file(self, filepath, cam_id=2):
 444     """Read in a calibration file and parse into a dictionary."""
 445 
 446     def parse_line(line, shape):
 447       data = line.split()
 448       data = np.array(data[1:]).reshape(shape).astype(np.float32)
 449       return data
 450 
 451     with open(filepath, 'r') as f:
 452       mat = f.readlines()
 453     proj_c2p = parse_line(mat[cam_id], shape=(3, 4))
 454     proj_v2c = parse_line(mat[-1], shape=(3, 4))
 455     filler = np.array([0, 0, 0, 1]).reshape((1, 4))
 456     proj_v2c = np.concatenate((proj_v2c, filler), axis=0)
 457     return proj_c2p, proj_v2c

 461     out[0, 0] *= sx
 462     out[0, 2] *= sx
 463     out[1, 1] *= sy
 464     out[1, 2] *= sy

 509     if not self.is_valid_example(target_frame_id):
 510       return False
 511     example = self.load_example(self.frames[target_index])
 512     return example
 513 
 514   def load_intrinsics(self, frame_id, split):
 515     """Read intrinsics data for frame."""
 516     city, seq, _, _ = frame_id.split('_')
 517     camera_file = os.path.join(self.dataset_dir, 'camera', split, city,
 518                                city + '_' + seq + '_*_camera.json')
 519     camera_file = glob.glob(camera_file)[0]
 520     with open(camera_file, 'r') as f:
 521       camera = json.load(f)
 522     fx = camera['intrinsic']['fx']
 523     fy = camera['intrinsic']['fy']
 524     u0 = camera['intrinsic']['u0']
 525     v0 = camera['intrinsic']['v0']
 526     # Cropping the bottom of the image and then resizing it to the same
 527     # (height, width) amounts to stretching the image's height.
 528     if self.crop_bottom:
 529       fy *= 1.0 / self.crop_pct
 530     intrinsics = np.array([[fx, 0, u0],
 531                            [0, fy, v0],
 532                            [0, 0, 1]])
 533     return intrinsics
 534 
 535   def is_valid_example(self, target_frame_id):
 536     """Checks whether we can find a valid sequence around this frame."""
 537     city, snippet_id, target_local_frame_id, _ = target_frame_id.split('_')
 538     start_index, end_index = get_seq_start_end(
 539         int(target_local_frame_id), self.seq_length, self.sample_every)
 540     for index in range(start_index, end_index + 1, self.sample_every):
 541       local_frame_id = '%.6d' % index
 542       frame_id = '%s_%s_%s_' % (city, snippet_id, local_frame_id)
 543       image_filepath = os.path.join(self.dataset_dir, 'leftImg8bit_sequence',
 544                                     self.split, city,
 545                                     frame_id + 'leftImg8bit.png')
 546       if not os.path.exists(image_filepath):
 547         return False
 548     return True

 550   def load_image_sequence(self, target_frame_id):
 551     """Returns a sequence with requested target frame."""
 552     city, snippet_id, target_local_frame_id, _ = target_frame_id.split('_')
 553     start_index, end_index = get_seq_start_end(
 554         int(target_local_frame_id), self.seq_length, self.sample_every)
 555     image_seq = []
 556     for index in range(start_index, end_index + 1, self.sample_every):
 557       local_frame_id = '%.6d' % index
 558       frame_id = '%s_%s_%s_' % (city, snippet_id, local_frame_id)
 559       image_filepath = os.path.join(self.dataset_dir, 'leftImg8bit_sequence',
 560                                     self.split, city,
 561                                     frame_id + 'leftImg8bit.png')
 562       img = scipy.misc.imread(image_filepath)
 563       if self.crop_bottom:
 564         ymax = int(img.shape[0] * self.crop_pct)
 565         img = img[:ymax]
 566       raw_shape = img.shape
 567       if index == int(target_local_frame_id):
 568         zoom_y = self.img_height / raw_shape[0]
 569         zoom_x = self.img_width / raw_shape[1]
 570       img = scipy.misc.imresize(img, (self.img_height, self.img_width))
 571       image_seq.append(img)
 572     return image_seq, zoom_x, zoom_y
 573 
 574   def load_example(self, target_frame_id):
 575     """Returns a sequence with requested target frame."""
 576     image_seq, zoom_x, zoom_y = self.load_image_sequence(target_frame_id)
 577     intrinsics = self.load_intrinsics(target_frame_id, self.split)
 578     intrinsics = self.scale_intrinsics(intrinsics, zoom_x, zoom_y)
 579     example = {}
 580     example['intrinsics'] = intrinsics
 581     example['image_seq'] = image_seq
 582     example['folder_name'] = target_frame_id.split('_')[0]
 583     example['file_name'] = target_frame_id[:-1]
 584     return example

 586   def scale_intrinsics(self, mat, sx, sy):
 587     out = np.copy(mat)
 588     out[0, 0] *= sx
 589     out[0, 2] *= sx
 590     out[1, 1] *= sy
 591     out[1, 2] *= sy
 592     return out

 599 def get_seq_start_end(target_index, seq_length, sample_every=1):
 600   """Returns absolute seq start and end indices for a given target frame."""
 601   half_offset = int((seq_length - 1) / 2) * sample_every
 602   end_index = target_index + half_offset
 603   start_index = end_index - (seq_length - 1) * sample_every
 604   return start_index, end_index
