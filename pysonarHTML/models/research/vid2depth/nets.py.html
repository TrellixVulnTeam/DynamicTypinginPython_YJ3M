  43 def egomotion_net(image_stack, is_training=True, legacy_mode=False):
  44   """Predict ego-motion vectors from a stack of frames.
  45 
  46   Args:
  47     image_stack: Input tensor with shape [B, h, w, seq_length * 3].  Regardless
  48         of the value of legacy_mode, the input image sequence passed to the
  49         function should be in normal order, e.g. [1, 2, 3].
  50     is_training: Whether the model is being trained or not.
  51     legacy_mode: Setting legacy_mode to True enables compatibility with
  52         SfMLearner checkpoints.  When legacy_mode is on, egomotion_net()
  53         rearranges the input tensor to place the target (middle) frame first in
  54         sequence.  This is the arrangement of inputs that legacy models have
  55         received during training.  In legacy mode, the client program
  56         (model.Model.build_loss()) interprets the outputs of this network
  57         differently as well.  For example:
  58 
  59         When legacy_mode == True,
  60         Network inputs will be [2, 1, 3]
  61         Network outputs will be [1 -> 2, 3 -> 2]
  62 
  63         When legacy_mode == False,
  64         Network inputs will be [1, 2, 3]
  65         Network outputs will be [1 -> 2, 2 -> 3]
  66 
  67   Returns:
  68     Egomotion vectors with shape [B, seq_length - 1, 6].
  69   """
  70   seq_length = image_stack.get_shape()[3].value // 3  # 3 == RGB.
  71   if legacy_mode:
  72     # Put the target frame at the beginning of stack.
  73     with tf.name_scope('rearrange_stack'):
  74       mid_index = util.get_seq_middle(seq_length)
  75       left_subset = image_stack[:, :, :, :mid_index * 3]
  76       target_frame = image_stack[:, :, :, mid_index * 3:(mid_index + 1) * 3]
  77       right_subset = image_stack[:, :, :, (mid_index + 1) * 3:]
  78       image_stack = tf.concat([target_frame, left_subset, right_subset], axis=3)
  79   batch_norm_params = {'is_training': is_training}
  80   num_egomotion_vecs = seq_length - 1
  81   with tf.variable_scope('pose_exp_net') as sc:
  82     end_points_collection = sc.original_name_scope + '_end_points'
  83     normalizer_fn = slim.batch_norm if FLAGS.use_bn else None
  84     normalizer_params = batch_norm_params if FLAGS.use_bn else None
  85     with slim.arg_scope([slim.conv2d, slim.conv2d_transpose],
  86                         normalizer_fn=normalizer_fn,
  87                         weights_regularizer=slim.l2_regularizer(WEIGHT_REG),
  88                         normalizer_params=normalizer_params,
  89                         activation_fn=tf.nn.relu,
  90                         outputs_collections=end_points_collection):
  91       cnv1 = slim.conv2d(image_stack, 16, [7, 7], stride=2, scope='cnv1')
  92       cnv2 = slim.conv2d(cnv1, 32, [5, 5], stride=2, scope='cnv2')
  93       cnv3 = slim.conv2d(cnv2, 64, [3, 3], stride=2, scope='cnv3')
  94       cnv4 = slim.conv2d(cnv3, 128, [3, 3], stride=2, scope='cnv4')
  95       cnv5 = slim.conv2d(cnv4, 256, [3, 3], stride=2, scope='cnv5')
  96 
  97       # Ego-motion specific layers
  98       with tf.variable_scope('pose'):
  99         cnv6 = slim.conv2d(cnv5, 256, [3, 3], stride=2, scope='cnv6')
 100         cnv7 = slim.conv2d(cnv6, 256, [3, 3], stride=2, scope='cnv7')
 101         pred_channels = EGOMOTION_VEC_SIZE * num_egomotion_vecs
 102         egomotion_pred = slim.conv2d(cnv7,
 103                                      pred_channels,
 104                                      [1, 1],
 105                                      scope='pred',
 106                                      stride=1,
 107                                      normalizer_fn=None,
 108                                      activation_fn=None)
 109         egomotion_avg = tf.reduce_mean(egomotion_pred, [1, 2])
 110         # Tinghui found that scaling by a small constant facilitates training.
 111         egomotion_final = 0.01 * tf.reshape(
 112             egomotion_avg, [-1, num_egomotion_vecs, EGOMOTION_VEC_SIZE])
 113 
 114       end_points = slim.utils.convert_collection_to_dict(end_points_collection)
 115       return egomotion_final, end_points

 118 def disp_net(target_image, is_training=True):
 119   """Predict inverse of depth from a single image."""
 120   batch_norm_params = {'is_training': is_training}
 121   h = target_image.get_shape()[1].value
 122   w = target_image.get_shape()[2].value
 123   inputs = target_image
 124   with tf.variable_scope('depth_net') as sc:
 125     end_points_collection = sc.original_name_scope + '_end_points'
 126     normalizer_fn = slim.batch_norm if FLAGS.use_bn else None
 127     normalizer_params = batch_norm_params if FLAGS.use_bn else None
 128     with slim.arg_scope([slim.conv2d, slim.conv2d_transpose],
 129                         normalizer_fn=normalizer_fn,
 130                         normalizer_params=normalizer_params,
 131                         weights_regularizer=slim.l2_regularizer(WEIGHT_REG),
 132                         activation_fn=tf.nn.relu,
 133                         outputs_collections=end_points_collection):
 134       cnv1 = slim.conv2d(inputs, 32, [7, 7], stride=2, scope='cnv1')
 135       cnv1b = slim.conv2d(cnv1, 32, [7, 7], stride=1, scope='cnv1b')
 136       cnv2 = slim.conv2d(cnv1b, 64, [5, 5], stride=2, scope='cnv2')
 137       cnv2b = slim.conv2d(cnv2, 64, [5, 5], stride=1, scope='cnv2b')
 138 
 139       cnv3 = slim.conv2d(cnv2b, 128, [3, 3], stride=2, scope='cnv3')
 140       cnv3b = slim.conv2d(cnv3, 128, [3, 3], stride=1, scope='cnv3b')
 141       cnv4 = slim.conv2d(cnv3b, 256, [3, 3], stride=2, scope='cnv4')
 142       cnv4b = slim.conv2d(cnv4, 256, [3, 3], stride=1, scope='cnv4b')
 143       cnv5 = slim.conv2d(cnv4b, 512, [3, 3], stride=2, scope='cnv5')
 144       cnv5b = slim.conv2d(cnv5, 512, [3, 3], stride=1, scope='cnv5b')
 145       cnv6 = slim.conv2d(cnv5b, 512, [3, 3], stride=2, scope='cnv6')
 146       cnv6b = slim.conv2d(cnv6, 512, [3, 3], stride=1, scope='cnv6b')
 147       cnv7 = slim.conv2d(cnv6b, 512, [3, 3], stride=2, scope='cnv7')
 148       cnv7b = slim.conv2d(cnv7, 512, [3, 3], stride=1, scope='cnv7b')
 149 
 150       up7 = slim.conv2d_transpose(cnv7b, 512, [3, 3], stride=2, scope='upcnv7')
 151       # There might be dimension mismatch due to uneven down/up-sampling.
 152       up7 = _resize_like(up7, cnv6b)
 153       i7_in = tf.concat([up7, cnv6b], axis=3)
 154       icnv7 = slim.conv2d(i7_in, 512, [3, 3], stride=1, scope='icnv7')
 155 
 156       up6 = slim.conv2d_transpose(icnv7, 512, [3, 3], stride=2, scope='upcnv6')
 157       up6 = _resize_like(up6, cnv5b)
 158       i6_in = tf.concat([up6, cnv5b], axis=3)
 159       icnv6 = slim.conv2d(i6_in, 512, [3, 3], stride=1, scope='icnv6')
 160 
 161       up5 = slim.conv2d_transpose(icnv6, 256, [3, 3], stride=2, scope='upcnv5')
 162       up5 = _resize_like(up5, cnv4b)
 163       i5_in = tf.concat([up5, cnv4b], axis=3)
 164       icnv5 = slim.conv2d(i5_in, 256, [3, 3], stride=1, scope='icnv5')
 165 
 166       up4 = slim.conv2d_transpose(icnv5, 128, [3, 3], stride=2, scope='upcnv4')
 167       i4_in = tf.concat([up4, cnv3b], axis=3)
 168       icnv4 = slim.conv2d(i4_in, 128, [3, 3], stride=1, scope='icnv4')
 169       disp4 = (slim.conv2d(icnv4, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 170                            normalizer_fn=None, scope='disp4')
 171                * DISP_SCALING + MIN_DISP)
 172       disp4_up = tf.image.resize_bilinear(disp4, [np.int(h / 4), np.int(w / 4)])
 173 
 174       up3 = slim.conv2d_transpose(icnv4, 64, [3, 3], stride=2, scope='upcnv3')
 175       i3_in = tf.concat([up3, cnv2b, disp4_up], axis=3)
 176       icnv3 = slim.conv2d(i3_in, 64, [3, 3], stride=1, scope='icnv3')
 177       disp3 = (slim.conv2d(icnv3, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 178                            normalizer_fn=None, scope='disp3')
 179                * DISP_SCALING + MIN_DISP)
 180       disp3_up = tf.image.resize_bilinear(disp3, [np.int(h / 2), np.int(w / 2)])
 181 
 182       up2 = slim.conv2d_transpose(icnv3, 32, [3, 3], stride=2, scope='upcnv2')
 183       i2_in = tf.concat([up2, cnv1b, disp3_up], axis=3)
 184       icnv2 = slim.conv2d(i2_in, 32, [3, 3], stride=1, scope='icnv2')
 185       disp2 = (slim.conv2d(icnv2, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 186                            normalizer_fn=None, scope='disp2')
 187                * DISP_SCALING + MIN_DISP)
 188       disp2_up = tf.image.resize_bilinear(disp2, [h, w])
 189 
 190       up1 = slim.conv2d_transpose(icnv2, 16, [3, 3], stride=2, scope='upcnv1')
 191       i1_in = tf.concat([up1, disp2_up], axis=3)
 192       icnv1 = slim.conv2d(i1_in, 16, [3, 3], stride=1, scope='icnv1')
 193       disp1 = (slim.conv2d(icnv1, 1, [3, 3], stride=1, activation_fn=tf.sigmoid,
 194                            normalizer_fn=None, scope='disp1')
 195                * DISP_SCALING + MIN_DISP)
 196 
 197       end_points = slim.utils.convert_collection_to_dict(end_points_collection)
 198       return [disp1, disp2, disp3, disp4], end_points

 201 def _resize_like(inputs, ref):
 202   i_h, i_w = inputs.get_shape()[1], inputs.get_shape()[2]
 203   r_h, r_w = ref.get_shape()[1], ref.get_shape()[2]
 204   if i_h == r_h and i_w == r_w:
 205     return inputs
 206   else:
 207     return tf.image.resize_nearest_neighbor(inputs, [r_h.value, r_w.value])
