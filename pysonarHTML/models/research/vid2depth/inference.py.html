  50 gfile = tf.gfile

  52 HOME_DIR = os.path.expanduser('~')
  53 DEFAULT_OUTPUT_DIR = os.path.join(HOME_DIR, 'vid2depth/inference')

  56 flags.DEFINE_string('output_dir', DEFAULT_OUTPUT_DIR,
  57                     'Directory to store estimated depth maps.')
  58 flags.DEFINE_string('kitti_dir', DEFAULT_KITTI_DIR, 'KITTI dataset directory.')
  59 flags.DEFINE_string('model_ckpt', None, 'Model checkpoint to load.')

  63 flags.DEFINE_integer('img_width', 416, 'Image width.')

  65 FLAGS = flags.FLAGS

  73 def _run_inference():
  74   """Runs all images through depth model and saves depth maps."""
  75   ckpt_basename = os.path.basename(FLAGS.model_ckpt)
  76   ckpt_modelname = os.path.basename(os.path.dirname(FLAGS.model_ckpt))
  77   output_dir = os.path.join(FLAGS.output_dir,
  78                             FLAGS.kitti_video.replace('/', '_') + '_' +
  79                             ckpt_modelname + '_' + ckpt_basename)
  80   if not gfile.Exists(output_dir):
  81     gfile.MakeDirs(output_dir)
  82   inference_model = model.Model(is_training=False,
  83                                 seq_length=FLAGS.seq_length,
  84                                 batch_size=FLAGS.batch_size,
  85                                 img_height=FLAGS.img_height,
  86                                 img_width=FLAGS.img_width)
  87   vars_to_restore = util.get_vars_to_restore(FLAGS.model_ckpt)
  88   saver = tf.train.Saver(vars_to_restore)
  89   sv = tf.train.Supervisor(logdir='/tmp/', saver=None)
  90   with sv.managed_session() as sess:
  91     saver.restore(sess, FLAGS.model_ckpt)
  92     if FLAGS.kitti_video == 'test_files_eigen':
  93       im_files = util.read_text_lines(
  94           util.get_resource_path('dataset/kitti/test_files_eigen.txt'))
  95       im_files = [os.path.join(FLAGS.kitti_dir, f) for f in im_files]
  96     else:
  97       video_path = os.path.join(FLAGS.kitti_dir, FLAGS.kitti_video)
  98       im_files = gfile.Glob(os.path.join(video_path, 'image_02/data', '*.png'))
  99       im_files = [f for f in im_files if 'disp' not in f]
 100       im_files = sorted(im_files)
 101     for i in range(0, len(im_files), FLAGS.batch_size):
 102       if i % 100 == 0:
 103         logging.info('Generating from %s: %d/%d', ckpt_basename, i,
 104                      len(im_files))
 105       inputs = np.zeros(
 106           (FLAGS.batch_size, FLAGS.img_height, FLAGS.img_width, 3),
 107           dtype=np.uint8)
 108       for b in range(FLAGS.batch_size):
 109         idx = i + b
 110         if idx >= len(im_files):
 111           break
 112         im = scipy.misc.imread(im_files[idx])
 113         inputs[b] = scipy.misc.imresize(im, (FLAGS.img_height, FLAGS.img_width))
 114       results = inference_model.inference(inputs, sess, mode='depth')
 115       for b in range(FLAGS.batch_size):
 116         idx = i + b
 117         if idx >= len(im_files):
 118           break
 119         if FLAGS.kitti_video == 'test_files_eigen':
 120           depth_path = os.path.join(output_dir, '%03d.png' % idx)
 121         else:
 122           depth_path = os.path.join(output_dir, '%04d.png' % idx)
 123         depth_map = results['depth'][b]
 124         depth_map = np.squeeze(depth_map)
 125         colored_map = _normalize_depth_for_display(depth_map, cmap=CMAP)
 126         input_float = inputs[b].astype(np.float32) / 255.0
 127         vertical_stack = np.concatenate((input_float, colored_map), axis=0)
 128         scipy.misc.imsave(depth_path, vertical_stack)

 131 def _gray2rgb(im, cmap=CMAP):
 132   cmap = plt.get_cmap(cmap)
 133   rgba_img = cmap(im.astype(np.float32))
 134   rgb_img = np.delete(rgba_img, 3, 2)
 135   return rgb_img
 136

 138 def _normalize_depth_for_display(depth,
 139                                  pc=95,
 140                                  crop_percent=0,
 141                                  normalizer=None,
 142                                  cmap=CMAP):
 143   """Converts a depth map to an RGB image."""
 144   # Convert to disparity.
 145   disp = 1.0 / (depth + 1e-6)
 146   if normalizer is not None:
 147     disp /= normalizer
 148   else:
 149     disp /= (np.percentile(disp, pc) + 1e-6)
 150   disp = np.clip(disp, 0, 1)
 151   disp = _gray2rgb(disp, cmap=cmap)
 152   keep_h = int(disp.shape[0] * (1 - crop_percent))
 153   disp = disp[:keep_h]
 154   return disp
