  98   def build_inference_for_training(self):
  99     """Invokes depth and ego-motion networks and computes clouds if needed."""
 100     (self.image_stack, self.intrinsic_mat, self.intrinsic_mat_inv) = (
 101         self.reader.read_data())
 102     with tf.name_scope('egomotion_prediction'):
 103       self.egomotion, _ = nets.egomotion_net(self.image_stack, is_training=True,
 104                                              legacy_mode=self.legacy_mode)
 105     with tf.variable_scope('depth_prediction'):
 106       # Organized by ...[i][scale].  Note that the order is flipped in
 107       # variables in build_loss() below.
 108       self.disp = {}
 109       self.depth = {}
 110       if self.icp_weight > 0:
 111         self.cloud = {}
 112       for i in range(self.seq_length):
 113         image = self.image_stack[:, :, :, 3 * i:3 * (i + 1)]
 114         multiscale_disps_i, _ = nets.disp_net(image, is_training=True)
 115         multiscale_depths_i = [1.0 / d for d in multiscale_disps_i]
 116         self.disp[i] = multiscale_disps_i
 117         self.depth[i] = multiscale_depths_i
 118         if self.icp_weight > 0:
 119           multiscale_clouds_i = [
 120               project.get_cloud(d,
 121                                 self.intrinsic_mat_inv[:, s, :, :],
 122                                 name='cloud%d_%d' % (s, i))
 123               for (s, d) in enumerate(multiscale_depths_i)
 124           ]
 125           self.cloud[i] = multiscale_clouds_i
 126         # Reuse the same depth graph for all images.
 127         tf.get_variable_scope().reuse_variables()
 128     logging.info('disp: %s', util.info(self.disp))

 130   def build_loss(self):
 131     """Adds ops for computing loss."""
 132     with tf.name_scope('compute_loss'):
 133       self.reconstr_loss = 0
 134       self.smooth_loss = 0
 135       self.ssim_loss = 0
 136       self.icp_transform_loss = 0
 137       self.icp_residual_loss = 0
 138 
 139       # self.images is organized by ...[scale][B, h, w, seq_len * 3].
 140       self.images = [{} for _ in range(NUM_SCALES)]
 141       # Following nested lists are organized by ...[scale][source-target].
 142       self.warped_image = [{} for _ in range(NUM_SCALES)]
 143       self.warp_mask = [{} for _ in range(NUM_SCALES)]
 144       self.warp_error = [{} for _ in range(NUM_SCALES)]
 145       self.ssim_error = [{} for _ in range(NUM_SCALES)]
 146       self.icp_transform = [{} for _ in range(NUM_SCALES)]
 147       self.icp_residual = [{} for _ in range(NUM_SCALES)]
 148 
 149       self.middle_frame_index = util.get_seq_middle(self.seq_length)
 150 
 151       # Compute losses at each scale.
 152       for s in range(NUM_SCALES):
 153         # Scale image stack.
 154         height_s = int(self.img_height / (2**s))
 155         width_s = int(self.img_width / (2**s))
 156         self.images[s] = tf.image.resize_area(self.image_stack,
 157                                               [height_s, width_s])
 158 
 159         # Smoothness.
 160         if self.smooth_weight > 0:
 161           for i in range(self.seq_length):
 162             # In legacy mode, use the depth map from the middle frame only.
 163             if not self.legacy_mode or i == self.middle_frame_index:
 164               self.smooth_loss += 1.0 / (2**s) * self.depth_smoothness(
 165                   self.disp[i][s], self.images[s][:, :, :, 3 * i:3 * (i + 1)])
 166 
 167         for i in range(self.seq_length):
 168           for j in range(self.seq_length):
 169             # Only consider adjacent frames.
 170             if i == j or abs(i - j) != 1:
 171               continue
 172             # In legacy mode, only consider the middle frame as target.
 173             if self.legacy_mode and j != self.middle_frame_index:
 174               continue
 175             source = self.images[s][:, :, :, 3 * i:3 * (i + 1)]
 176             target = self.images[s][:, :, :, 3 * j:3 * (j + 1)]
 177             target_depth = self.depth[j][s]
 178             key = '%d-%d' % (i, j)
 179 
 180             # Extract ego-motion from i to j
 181             egomotion_index = min(i, j)
 182             egomotion_mult = 1
 183             if i > j:
 184               # Need to inverse egomotion when going back in sequence.
 185               egomotion_mult *= -1
 186             # For compatiblity with SfMLearner, interpret all egomotion vectors
 187             # as pointing toward the middle frame.  Note that unlike SfMLearner,
 188             # each vector captures the motion to/from its next frame, and not
 189             # the center frame.  Although with seq_length == 3, there is no
 190             # difference.
 191             if self.legacy_mode:
 192               if egomotion_index >= self.middle_frame_index:
 193                 egomotion_mult *= -1
 194             egomotion = egomotion_mult * self.egomotion[:, egomotion_index, :]
 195 
 196             # Inverse warp the source image to the target image frame for
 197             # photometric consistency loss.
 198             self.warped_image[s][key], self.warp_mask[s][key] = (
 199                 project.inverse_warp(source,
 200                                      target_depth,
 201                                      egomotion,
 202                                      self.intrinsic_mat[:, s, :, :],
 203                                      self.intrinsic_mat_inv[:, s, :, :]))
 204 
 205             # Reconstruction loss.
 206             self.warp_error[s][key] = tf.abs(self.warped_image[s][key] - target)
 207             self.reconstr_loss += tf.reduce_mean(
 208                 self.warp_error[s][key] * self.warp_mask[s][key])
 209             # SSIM.
 210             if self.ssim_weight > 0:
 211               self.ssim_error[s][key] = self.ssim(self.warped_image[s][key],
 212                                                   target)
 213               # TODO(rezama): This should be min_pool2d().
 214               ssim_mask = slim.avg_pool2d(self.warp_mask[s][key], 3, 1, 'VALID')
 215               self.ssim_loss += tf.reduce_mean(
 216                   self.ssim_error[s][key] * ssim_mask)
 217             # 3D loss.
 218             if self.icp_weight > 0:
 219               cloud_a = self.cloud[j][s]
 220               cloud_b = self.cloud[i][s]
 221               self.icp_transform[s][key], self.icp_residual[s][key] = icp(
 222                   cloud_a, egomotion, cloud_b)
 223               self.icp_transform_loss += 1.0 / (2**s) * tf.reduce_mean(
 224                   tf.abs(self.icp_transform[s][key]))
 225               self.icp_residual_loss += 1.0 / (2**s) * tf.reduce_mean(
 226                   tf.abs(self.icp_residual[s][key]))
 227 
 228       self.total_loss = self.reconstr_weight * self.reconstr_loss
 229       if self.smooth_weight > 0:
 230         self.total_loss += self.smooth_weight * self.smooth_loss
 231       if self.ssim_weight > 0:
 232         self.total_loss += self.ssim_weight * self.ssim_loss
 233       if self.icp_weight > 0:
 234         self.total_loss += self.icp_weight * (self.icp_transform_loss +
 235                                               self.icp_residual_loss)

 237   def gradient_x(self, img):
 238     return img[:, :, :-1, :] - img[:, :, 1:, :]
 239 
 240   def gradient_y(self, img):
 241     return img[:, :-1, :, :] - img[:, 1:, :, :]
 242 
 243   def depth_smoothness(self, depth, img):
 244     """Computes image-aware depth smoothness loss."""
 245     depth_dx = self.gradient_x(depth)
 246     depth_dy = self.gradient_y(depth)
 247     image_dx = self.gradient_x(img)
 248     image_dy = self.gradient_y(img)
 249     weights_x = tf.exp(-tf.reduce_mean(tf.abs(image_dx), 3, keepdims=True))
 250     weights_y = tf.exp(-tf.reduce_mean(tf.abs(image_dy), 3, keepdims=True))
 251     smoothness_x = depth_dx * weights_x
 252     smoothness_y = depth_dy * weights_y
 253     return tf.reduce_mean(abs(smoothness_x)) + tf.reduce_mean(abs(smoothness_y))
 254 
 255   def ssim(self, x, y):
 256     """Computes a differentiable structured image similarity measure."""
 257     c1 = 0.01**2
 258     c2 = 0.03**2
 259     mu_x = slim.avg_pool2d(x, 3, 1, 'VALID')
 260     mu_y = slim.avg_pool2d(y, 3, 1, 'VALID')
 261     sigma_x = slim.avg_pool2d(x**2, 3, 1, 'VALID') - mu_x**2
 262     sigma_y = slim.avg_pool2d(y**2, 3, 1, 'VALID') - mu_y**2
 263     sigma_xy = slim.avg_pool2d(x * y, 3, 1, 'VALID') - mu_x * mu_y
 264     ssim_n = (2 * mu_x * mu_y + c1) * (2 * sigma_xy + c2)
 265     ssim_d = (mu_x**2 + mu_y**2 + c1) * (sigma_x + sigma_y + c2)
 266     ssim = ssim_n / ssim_d
 267     return tf.clip_by_value((1 - ssim) / 2, 0, 1)

 269   def build_train_op(self):
 270     with tf.name_scope('train_op'):
 271       optim = tf.train.AdamOptimizer(self.learning_rate, self.beta1)
 272       self.train_op = slim.learning.create_train_op(self.total_loss, optim)
 273       self.global_step = tf.Variable(0, name='global_step', trainable=False)
 274       self.incr_global_step = tf.assign(self.global_step, self.global_step + 1)

 276   def build_summaries(self):
 277     """Adds scalar and image summaries for TensorBoard."""
 278     tf.summary.scalar('total_loss', self.total_loss)
 279     tf.summary.scalar('reconstr_loss', self.reconstr_loss)
 280     if self.smooth_weight > 0:
 281       tf.summary.scalar('smooth_loss', self.smooth_loss)
 282     if self.ssim_weight > 0:
 283       tf.summary.scalar('ssim_loss', self.ssim_loss)
 284     if self.icp_weight > 0:
 285       tf.summary.scalar('icp_transform_loss', self.icp_transform_loss)
 286       tf.summary.scalar('icp_residual_loss', self.icp_residual_loss)
 287 
 288     for i in range(self.seq_length - 1):
 289       tf.summary.histogram('tx%d' % i, self.egomotion[:, i, 0])
 290       tf.summary.histogram('ty%d' % i, self.egomotion[:, i, 1])
 291       tf.summary.histogram('tz%d' % i, self.egomotion[:, i, 2])
 292       tf.summary.histogram('rx%d' % i, self.egomotion[:, i, 3])
 293       tf.summary.histogram('ry%d' % i, self.egomotion[:, i, 4])
 294       tf.summary.histogram('rz%d' % i, self.egomotion[:, i, 5])
 295 
 296     for s in range(NUM_SCALES):
 297       for i in range(self.seq_length):
 298         tf.summary.image('scale%d_image%d' % (s, i),
 299                          self.images[s][:, :, :, 3 * i:3 * (i + 1)])
 300         if i in self.depth:
 301           tf.summary.histogram('scale%d_depth%d' % (s, i), self.depth[i][s])
 302           tf.summary.histogram('scale%d_disp%d' % (s, i), self.disp[i][s])
 303           tf.summary.image('scale%d_disparity%d' % (s, i), self.disp[i][s])
 304 
 305       for key in self.warped_image[s]:
 306         tf.summary.image('scale%d_warped_image%s' % (s, key),
 307                          self.warped_image[s][key])
 308         tf.summary.image('scale%d_warp_mask%s' % (s, key),
 309                          self.warp_mask[s][key])
 310         tf.summary.image('scale%d_warp_error%s' % (s, key),
 311                          self.warp_error[s][key])
 312         if self.ssim_weight > 0:
 313           tf.summary.image('scale%d_ssim_error%s' % (s, key),
 314                            self.ssim_error[s][key])
 315         if self.icp_weight > 0:
 316           tf.summary.image('scale%d_icp_residual%s' % (s, key),
 317                            self.icp_residual[s][key])
 318           transform = self.icp_transform[s][key]
 319           tf.summary.histogram('scale%d_icp_tx%s' % (s, key), transform[:, 0])
 320           tf.summary.histogram('scale%d_icp_ty%s' % (s, key), transform[:, 1])
 321           tf.summary.histogram('scale%d_icp_tz%s' % (s, key), transform[:, 2])
 322           tf.summary.histogram('scale%d_icp_rx%s' % (s, key), transform[:, 3])
 323           tf.summary.histogram('scale%d_icp_ry%s' % (s, key), transform[:, 4])
 324           tf.summary.histogram('scale%d_icp_rz%s' % (s, key), transform[:, 5])

 326   def build_depth_test_graph(self):
 327     """Builds depth model reading from placeholders."""
 328     with tf.name_scope('depth_prediction'):
 329       with tf.variable_scope('depth_prediction'):
 330         input_uint8 = tf.placeholder(
 331             tf.uint8, [self.batch_size, self.img_height, self.img_width, 3],
 332             name='raw_input')
 333         input_float = tf.image.convert_image_dtype(input_uint8, tf.float32)
 334         # TODO(rezama): Retrain published model with batchnorm params and set
 335         # is_training to False.
 336         est_disp, _ = nets.disp_net(input_float, is_training=True)
 337         est_depth = 1.0 / est_disp[0]
 338     self.inputs_depth = input_uint8
 339     self.est_depth = est_depth
 340 
 341   def build_egomotion_test_graph(self):
 342     """Builds egomotion model reading from placeholders."""
 343     input_uint8 = tf.placeholder(
 344         tf.uint8,
 345         [self.batch_size, self.img_height, self.img_width * self.seq_length, 3],
 346         name='raw_input')
 347     input_float = tf.image.convert_image_dtype(input_uint8, tf.float32)
 348     image_seq = input_float
 349     image_stack = self.unpack_image_batches(image_seq)
 350     with tf.name_scope('egomotion_prediction'):
 351         # TODO(rezama): Retrain published model with batchnorm params and set
 352         # is_training to False.
 353       egomotion, _ = nets.egomotion_net(image_stack, is_training=True,
 354                                         legacy_mode=self.legacy_mode)
 355     self.inputs_egomotion = input_uint8
 356     self.est_egomotion = egomotion
 357 
 358   def unpack_image_batches(self, image_seq):
 359     """[B, h, w * seq_length, 3] -> [B, h, w, 3 * seq_length]."""
 360     with tf.name_scope('unpack_images'):
 361       image_list = [
 362           image_seq[:, :, i * self.img_width:(i + 1) * self.img_width, :]
 363           for i in range(self.seq_length)
 364       ]
 365       image_stack = tf.concat(image_list, axis=3)
 366       image_stack.set_shape([
 367           self.batch_size, self.img_height, self.img_width, self.seq_length * 3
 368       ])
 369     return image_stack

 371   def inference(self, inputs, sess, mode):
 372     """Runs depth or egomotion inference from placeholders."""
 373     fetches = {}
 374     if mode == 'depth':
 375       fetches['depth'] = self.est_depth
 376       inputs_ph = self.inputs_depth
 377     if mode == 'egomotion':
 378       fetches['egomotion'] = self.est_egomotion
 379       inputs_ph = self.inputs_egomotion
 380     results = sess.run(fetches, feed_dict={inputs_ph: inputs})
 381     return results
