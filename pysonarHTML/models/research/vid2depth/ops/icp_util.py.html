  29 def get_transformation_matrix(transform):
  30   """Converts [tx, ty, tz, rx, ry, rz] to a transform matrix."""
  31   rx = transform[3]
  32   ry = transform[4]
  33   rz = transform[5]
  34 
  35   rz = tf.clip_by_value(rz, -np.pi, np.pi)
  36   ry = tf.clip_by_value(ry, -np.pi, np.pi)
  37   rx = tf.clip_by_value(rx, -np.pi, np.pi)
  38 
  39   cos_rx = tf.cos(rx)
  40   sin_rx = tf.sin(rx)
  41   rotx_1 = tf.stack([1.0, 0.0, 0.0])
  42   rotx_2 = tf.stack([0.0, cos_rx, -sin_rx])
  43   rotx_3 = tf.stack([0.0, sin_rx, cos_rx])
  44   xmat = tf.stack([rotx_1, rotx_2, rotx_3])
  45 
  46   cos_ry = tf.cos(ry)
  47   sin_ry = tf.sin(ry)
  48   roty_1 = tf.stack([cos_ry, 0.0, sin_ry])
  49   roty_2 = tf.stack([0.0, 1.0, 0.0])
  50   roty_3 = tf.stack([-sin_ry, 0.0, cos_ry])
  51   ymat = tf.stack([roty_1, roty_2, roty_3])
  52 
  53   cos_rz = tf.cos(rz)
  54   sin_rz = tf.sin(rz)
  55   rotz_1 = tf.stack([cos_rz, -sin_rz, 0.0])
  56   rotz_2 = tf.stack([sin_rz, cos_rz, 0.0])
  57   rotz_3 = tf.stack([0.0, 0.0, 1.0])
  58   zmat = tf.stack([rotz_1, rotz_2, rotz_3])
  59 
  60   rotate = tf.matmul(tf.matmul(xmat, ymat), zmat)
  61 
  62   translate = transform[:3]
  63   mat = tf.concat([rotate, tf.expand_dims(translate, 1)], axis=1)
  64 
  65   hom_filler = tf.constant([0.0, 0.0, 0.0, 1.0], shape=[1, 4], dtype=tf.float32)
  66   mat = tf.concat([mat, hom_filler], axis=0)
  67   return mat

  70 def np_get_transformation_matrix(transform):
  71   """Converts [tx, ty, tz, rx, ry, rz] to a transform matrix."""
  72   rx = transform[3]
  73   ry = transform[4]
  74   rz = transform[5]
  75 
  76   rz = np.clip(rz, -np.pi, np.pi)
  77   ry = np.clip(ry, -np.pi, np.pi)
  78   rx = np.clip(rx, -np.pi, np.pi)
  79 
  80   cos_rx = np.cos(rx)
  81   sin_rx = np.sin(rx)
  82   rotx_1 = np.stack([1.0, 0.0, 0.0])
  83   rotx_2 = np.stack([0.0, cos_rx, -sin_rx])
  84   rotx_3 = np.stack([0.0, sin_rx, cos_rx])
  85   xmat = np.stack([rotx_1, rotx_2, rotx_3])
  86 
  87   cos_ry = np.cos(ry)
  88   sin_ry = np.sin(ry)
  89   roty_1 = np.stack([cos_ry, 0.0, sin_ry])
  90   roty_2 = np.stack([0.0, 1.0, 0.0])
  91   roty_3 = np.stack([-sin_ry, 0.0, cos_ry])
  92   ymat = np.stack([roty_1, roty_2, roty_3])
  93 
  94   cos_rz = np.cos(rz)
  95   sin_rz = np.sin(rz)
  96   rotz_1 = np.stack([cos_rz, -sin_rz, 0.0])
  97   rotz_2 = np.stack([sin_rz, cos_rz, 0.0])
  98   rotz_3 = np.stack([0.0, 0.0, 1.0])
  99   zmat = np.stack([rotz_1, rotz_2, rotz_3])
 100 
 101   rotate = np.dot(np.dot(xmat, ymat), zmat)
 102 
 103   translate = transform[:3]
 104   mat = np.concatenate((rotate, np.expand_dims(translate, 1)), axis=1)
 105 
 106   hom_filler = np.array([[0.0, 0.0, 0.0, 1.0]], dtype=np.float32)
 107   mat = np.concatenate((mat, hom_filler), axis=0)
 108   return mat

 111 def transform_cloud_xyz(cloud, transform):
 112   num_points = cloud.shape.as_list()[0]
 113   ones = tf.ones(shape=[num_points, 1], dtype=tf.float32)
 114   hom_cloud = tf.concat([cloud, ones], axis=1)
 115   hom_cloud_t = tf.transpose(hom_cloud)
 116   mat = get_transformation_matrix(transform)
 117   transformed_cloud = tf.matmul(mat, hom_cloud_t)
 118   transformed_cloud = tf.transpose(transformed_cloud)
 119   transformed_cloud = transformed_cloud[:, :3]
 120   return transformed_cloud

 123 def np_transform_cloud_xyz(cloud, transform):
 124   num_points = cloud.shape[0]
 125   ones = np.ones(shape=[num_points, 1], dtype=np.float32)
 126   hom_cloud = np.concatenate((cloud, ones), axis=1)
 127   hom_cloud_t = np.transpose(hom_cloud)
 128   mat = np_get_transformation_matrix(transform)
 129   transformed_cloud = np.dot(mat, hom_cloud_t)
 130   transformed_cloud = np.transpose(transformed_cloud)
 131   transformed_cloud = transformed_cloud[:, :3]
 132   return transformed_cloud

 135 def batch_transform_cloud_xyz(cloud, transform):
 136   results = []
 137   cloud_items = tf.unstack(cloud)
 138   if len(transform.shape.as_list()) == 2:
 139     transform_items = tf.unstack(transform)
 140   else:
 141     transform_items = [transform] * len(cloud_items)
 142   for cloud_item, transform_item in zip(cloud_items, transform_items):
 143     results.append(transform_cloud_xyz(cloud_item, transform_item))
 144   return tf.stack(results)
