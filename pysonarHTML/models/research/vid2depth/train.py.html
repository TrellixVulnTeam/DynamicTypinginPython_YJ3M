  74 # Maximum number of checkpoints to keep.
  75 MAX_TO_KEEP = 100

  78 def main(_):
  79   # Fixed seed for repeatability
  80   seed = 8964
  81   tf.set_random_seed(seed)
  82   np.random.seed(seed)
  83   random.seed(seed)
  84 
  85   if FLAGS.legacy_mode and FLAGS.seq_length < 3:
  86     raise ValueError('Legacy mode supports sequence length > 2 only.')
  87 
  88   if not gfile.Exists(FLAGS.checkpoint_dir):
  89     gfile.MakeDirs(FLAGS.checkpoint_dir)
  90 
  91   train_model = model.Model(data_dir=FLAGS.data_dir,
  92                             is_training=True,
  93                             learning_rate=FLAGS.learning_rate,
  94                             beta1=FLAGS.beta1,
  95                             reconstr_weight=FLAGS.reconstr_weight,
  96                             smooth_weight=FLAGS.smooth_weight,
  97                             ssim_weight=FLAGS.ssim_weight,
  98                             icp_weight=FLAGS.icp_weight,
  99                             batch_size=FLAGS.batch_size,
 100                             img_height=FLAGS.img_height,
 101                             img_width=FLAGS.img_width,
 102                             seq_length=FLAGS.seq_length,
 103                             legacy_mode=FLAGS.legacy_mode)
 104 
 105   train(train_model, FLAGS.pretrained_ckpt, FLAGS.checkpoint_dir,
 106         FLAGS.train_steps, FLAGS.summary_freq)

 109 def train(train_model, pretrained_ckpt, checkpoint_dir, train_steps,
 110           summary_freq):
 111   """Train model."""
 112   if pretrained_ckpt is not None:
 113     vars_to_restore = util.get_vars_to_restore(pretrained_ckpt)
 114     pretrain_restorer = tf.train.Saver(vars_to_restore)
 115   vars_to_save = util.get_vars_to_restore()
 116   saver = tf.train.Saver(vars_to_save + [train_model.global_step],
 117                          max_to_keep=MAX_TO_KEEP)
 118   sv = tf.train.Supervisor(logdir=checkpoint_dir, save_summaries_secs=0,
 119                            saver=None)
 120   config = tf.ConfigProto()
 121   config.gpu_options.allow_growth = True
 122   with sv.managed_session(config=config) as sess:
 123     if pretrained_ckpt is not None:
 124       logging.info('Restoring pretrained weights from %s', pretrained_ckpt)
 125       pretrain_restorer.restore(sess, pretrained_ckpt)
 126     logging.info('Attempting to resume training from %s...', checkpoint_dir)
 127     checkpoint = tf.train.latest_checkpoint(checkpoint_dir)
 128     logging.info('Last checkpoint found: %s', checkpoint)
 129     if checkpoint:
 130       saver.restore(sess, checkpoint)
 131 
 132     logging.info('Training...')
 133     start_time = time.time()
 134     last_summary_time = time.time()
 135     steps_per_epoch = train_model.reader.steps_per_epoch
 136     step = 1
 137     while step <= train_steps:
 138       fetches = {
 139           'train': train_model.train_op,
 140           'global_step': train_model.global_step,
 141           'incr_global_step': train_model.incr_global_step
 142       }
 143 
 144       if step % summary_freq == 0:
 145         fetches['loss'] = train_model.total_loss
 146         fetches['summary'] = sv.summary_op
 147 
 148       results = sess.run(fetches)
 149       global_step = results['global_step']
 150 
 151       if step % summary_freq == 0:
 152         sv.summary_writer.add_summary(results['summary'], global_step)
 153         train_epoch = math.ceil(global_step / steps_per_epoch)
 154         train_step = global_step - (train_epoch - 1) * steps_per_epoch
 155         this_cycle = time.time() - last_summary_time
 156         last_summary_time += this_cycle
 157         logging.info(
 158             'Epoch: [%2d] [%5d/%5d] time: %4.2fs (%ds total) loss: %.3f',
 159             train_epoch, train_step, steps_per_epoch, this_cycle,
 160             time.time() - start_time, results['loss'])
 161 
 162       if step % steps_per_epoch == 0:
 163         logging.info('[*] Saving checkpoint to %s...', checkpoint_dir)
 164         saver.save(sess, os.path.join(checkpoint_dir, 'model'),
 165                    global_step=global_step)
 166 
 167       # Setting step to global_step allows for training for a total of
 168       # train_steps even if the program is restarted during training.
 169       step = global_step + 1
 170 
 171 
 172 if __name__ == '__main__':
 173   app.run(main)
