  24 class Vocabulary(object):
  25   """Class that holds a vocabulary for the dataset."""
  26 
  27   def __init__(self, filename):
  28     """Initialize vocabulary.
  29 
  30     Args:
  31       filename: Vocabulary file name.
  32     """
  33 
  34     self._id_to_word = []
  35     self._word_to_id = {}
  36     self._unk = -1
  37     self._bos = -1
  38     self._eos = -1
  39 
  40     with tf.gfile.Open(filename) as f:
  41       idx = 0
  42       for line in f:
  43         word_name = line.strip()
  44         if word_name == '<S>':
  45           self._bos = idx
  46         elif word_name == '</S>':
  47           self._eos = idx
  48         elif word_name == '<UNK>':
  49           self._unk = idx
  50         if word_name == '!!!MAXTERMID':
  51           continue
  52 
  53         self._id_to_word.append(word_name)
  54         self._word_to_id[word_name] = idx
  55         idx += 1
  56 
  57   @property
  58   def bos(self):
  59     return self._bos
  60 
  61   @property
  62   def eos(self):
  63     return self._eos
  64 
  65   @property
  66   def unk(self):
  67     return self._unk
  68 
  69   @property
  70   def size(self):
  71     return len(self._id_to_word)
  72 
  73   def word_to_id(self, word):
  74     if word in self._word_to_id:
  75       return self._word_to_id[word]
  76     return self.unk
  77 
  78   def id_to_word(self, cur_id):
  79     if cur_id < self.size:
  80       return self._id_to_word[cur_id]
  81     return 'ERROR'
  82 
  83   def decode(self, cur_ids):
  84     """Convert a list of ids to a sentence, with space inserted."""
  85     return ' '.join([self.id_to_word(cur_id) for cur_id in cur_ids])
  86 
  87   def encode(self, sentence):
  88     """Convert a sentence to a list of ids, with special tokens added."""
  89     word_ids = [self.word_to_id(cur_word) for cur_word in sentence.split()]
  90     return np.array([self.bos] + word_ids + [self.eos], dtype=np.int32)

  93 class CharsVocabulary(Vocabulary):
  94   """Vocabulary containing character-level information."""
  95 
  96   def __init__(self, filename, max_word_length):
  97     super(CharsVocabulary, self).__init__(filename)
  98     self._max_word_length = max_word_length
  99     chars_set = set()
 100 
 101     for word in self._id_to_word:
 102       chars_set |= set(word)
 103 
 104     free_ids = []
 105     for i in range(256):
 106       if chr(i) in chars_set:
 107         continue
 108       free_ids.append(chr(i))
 109 
 110     if len(free_ids) < 5:
 111       raise ValueError('Not enough free char ids: %d' % len(free_ids))
 112 
 113     self.bos_char = free_ids[0]  # <begin sentence>
 114     self.eos_char = free_ids[1]  # <end sentence>
 115     self.bow_char = free_ids[2]  # <begin word>
 116     self.eow_char = free_ids[3]  # <end word>
 117     self.pad_char = free_ids[4]  # <padding>
 118 
 119     chars_set |= {self.bos_char, self.eos_char, self.bow_char, self.eow_char,
 120                   self.pad_char}
 121 
 122     self._char_set = chars_set
 123     num_words = len(self._id_to_word)
 124 
 125     self._word_char_ids = np.zeros([num_words, max_word_length], dtype=np.int32)
 126 
 127     self.bos_chars = self._convert_word_to_char_ids(self.bos_char)
 128     self.eos_chars = self._convert_word_to_char_ids(self.eos_char)
 129 
 130     for i, word in enumerate(self._id_to_word):
 131       self._word_char_ids[i] = self._convert_word_to_char_ids(word)
 132 
 133   @property
 134   def word_char_ids(self):
 135     return self._word_char_ids
 136 
 137   @property
 138   def max_word_length(self):
 139     return self._max_word_length
 140 
 141   def _convert_word_to_char_ids(self, word):
 142     code = np.zeros([self.max_word_length], dtype=np.int32)
 143     code[:] = ord(self.pad_char)
 144 
 145     if len(word) > self.max_word_length - 2:
 146       word = word[:self.max_word_length-2]
 147     cur_word = self.bow_char + word + self.eow_char
 148     for j in range(len(cur_word)):
 149       code[j] = ord(cur_word[j])
 150     return code
 151 
 152   def word_to_char_ids(self, word):
 153     if word in self._word_to_id:
 154       return self._word_char_ids[self._word_to_id[word]]
 155     else:
 156       return self._convert_word_to_char_ids(word)
 157 
 158   def encode_chars(self, sentence):
 159     chars_ids = [self.word_to_char_ids(cur_word)
 160                  for cur_word in sentence.split()]
 161     return np.vstack([self.bos_chars] + chars_ids + [self.eos_chars])

 164 def get_batch(generator, batch_size, num_steps, max_word_length, pad=False):
 165   """Read batches of input."""
 166   cur_stream = [None] * batch_size
 167 
 168   inputs = np.zeros([batch_size, num_steps], np.int32)
 169   char_inputs = np.zeros([batch_size, num_steps, max_word_length], np.int32)
 170   global_word_ids = np.zeros([batch_size, num_steps], np.int32)
 171   targets = np.zeros([batch_size, num_steps], np.int32)
 172   weights = np.ones([batch_size, num_steps], np.float32)
 173 
 174   no_more_data = False
 175   while True:
 176     inputs[:] = 0
 177     char_inputs[:] = 0
 178     global_word_ids[:] = 0
 179     targets[:] = 0
 180     weights[:] = 0.0
 181 
 182     for i in range(batch_size):
 183       cur_pos = 0
 184 
 185       while cur_pos < num_steps:
 186         if cur_stream[i] is None or len(cur_stream[i][0]) <= 1:
 187           try:
 188             cur_stream[i] = list(generator.next())
 189           except StopIteration:
 190             # No more data, exhaust current streams and quit
 191             no_more_data = True
 192             break
 193 
 194         how_many = min(len(cur_stream[i][0]) - 1, num_steps - cur_pos)
 195         next_pos = cur_pos + how_many
 196 
 197         inputs[i, cur_pos:next_pos] = cur_stream[i][0][:how_many]
 198         char_inputs[i, cur_pos:next_pos] = cur_stream[i][1][:how_many]
 199         global_word_ids[i, cur_pos:next_pos] = cur_stream[i][2][:how_many]
 200         targets[i, cur_pos:next_pos] = cur_stream[i][0][1:how_many+1]
 201         weights[i, cur_pos:next_pos] = 1.0
 202 
 203         cur_pos = next_pos
 204         cur_stream[i][0] = cur_stream[i][0][how_many:]
 205         cur_stream[i][1] = cur_stream[i][1][how_many:]
 206         cur_stream[i][2] = cur_stream[i][2][how_many:]
 207 
 208         if pad:
 209           break
 210 
 211     if no_more_data and np.sum(weights) == 0:
 212       # There is no more data and this is an empty batch. Done!
 213       break
 214     yield inputs, char_inputs, global_word_ids, targets, weights

 217 class LM1BDataset(object):
 218   """Utility class for 1B word benchmark dataset.
 219 
 220   The current implementation reads the data from the tokenized text files.
 221   """
 222 
 223   def __init__(self, filepattern, vocab):
 224     """Initialize LM1BDataset reader.
 225 
 226     Args:
 227       filepattern: Dataset file pattern.
 228       vocab: Vocabulary.
 229     """
 230     self._vocab = vocab
 231     self._all_shards = tf.gfile.Glob(filepattern)
 232     tf.logging.info('Found %d shards at %s', len(self._all_shards), filepattern)
 233 
 234   def _load_random_shard(self):
 235     """Randomly select a file and read it."""
 236     return self._load_shard(random.choice(self._all_shards))
 237 
 238   def _load_shard(self, shard_name):
 239     """Read one file and convert to ids.
 240 
 241     Args:
 242       shard_name: file path.
 243 
 244     Returns:
 245       list of (id, char_id, global_word_id) tuples.
 246     """
 247     tf.logging.info('Loading data from: %s', shard_name)
 248     with tf.gfile.Open(shard_name) as f:
 249       sentences = f.readlines()
 250     chars_ids = [self.vocab.encode_chars(sentence) for sentence in sentences]
 251     ids = [self.vocab.encode(sentence) for sentence in sentences]
 252 
 253     global_word_ids = []
 254     current_idx = 0
 255     for word_ids in ids:
 256       current_size = len(word_ids) - 1  # without <BOS> symbol
 257       cur_ids = np.arange(current_idx, current_idx + current_size)
 258       global_word_ids.append(cur_ids)
 259       current_idx += current_size
 260 
 261     tf.logging.info('Loaded %d words.', current_idx)
 262     tf.logging.info('Finished loading')
 263     return zip(ids, chars_ids, global_word_ids)
 264 
 265   def _get_sentence(self, forever=True):
 266     while True:
 267       ids = self._load_random_shard()
 268       for current_ids in ids:
 269         yield current_ids
 270       if not forever:
 271         break
 272 
 273   def get_batch(self, batch_size, num_steps, pad=False, forever=True):
 274     return get_batch(self._get_sentence(forever), batch_size, num_steps,
 275                      self.vocab.max_word_length, pad=pad)
 276 
 277   @property
 278   def vocab(self):
 279     return self._vocab
