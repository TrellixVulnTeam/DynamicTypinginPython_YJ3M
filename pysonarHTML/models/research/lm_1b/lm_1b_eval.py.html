  74 def _LoadModel(gd_file, ckpt_file):
  75   """Load the model from GraphDef and Checkpoint.
  76 
  77   Args:
  78     gd_file: GraphDef proto text file.
  79     ckpt_file: TensorFlow Checkpoint file.
  80 
  81   Returns:
  82     TensorFlow session and tensors dict.
  83   """
  84   with tf.Graph().as_default():
  85     sys.stderr.write('Recovering graph.\n')
  86     with tf.gfile.FastGFile(gd_file, 'r') as f:
  87       s = f.read().decode()
  88       gd = tf.GraphDef()
  89       text_format.Merge(s, gd)
  90 
  91     tf.logging.info('Recovering Graph %s', gd_file)
  92     t = {}
  93     [t['states_init'], t['lstm/lstm_0/control_dependency'],
  94      t['lstm/lstm_1/control_dependency'], t['softmax_out'], t['class_ids_out'],
  95      t['class_weights_out'], t['log_perplexity_out'], t['inputs_in'],
  96      t['targets_in'], t['target_weights_in'], t['char_inputs_in'],
  97      t['all_embs'], t['softmax_weights'], t['global_step']
  98     ] = tf.import_graph_def(gd, {}, ['states_init',
  99                                      'lstm/lstm_0/control_dependency:0',
 100                                      'lstm/lstm_1/control_dependency:0',
 101                                      'softmax_out:0',
 102                                      'class_ids_out:0',
 103                                      'class_weights_out:0',
 104                                      'log_perplexity_out:0',
 105                                      'inputs_in:0',
 106                                      'targets_in:0',
 107                                      'target_weights_in:0',
 108                                      'char_inputs_in:0',
 109                                      'all_embs_out:0',
 110                                      'Reshape_3:0',
 111                                      'global_step:0'], name='')
 112 
 113     sys.stderr.write('Recovering checkpoint %s\n' % ckpt_file)
 114     sess = tf.Session(config=tf.ConfigProto(allow_soft_placement=True))
 115     sess.run('save/restore_all', {'save/Const:0': ckpt_file})
 116     sess.run(t['states_init'])
 117 
 118   return sess, t

 121 def _EvalModel(dataset):
 122   """Evaluate model perplexity using provided dataset.
 123 
 124   Args:
 125     dataset: LM1BDataset object.
 126   """
 127   sess, t = _LoadModel(FLAGS.pbtxt, FLAGS.ckpt)
 128 
 129   current_step = t['global_step'].eval(session=sess)
 130   sys.stderr.write('Loaded step %d.\n' % current_step)
 131 
 132   data_gen = dataset.get_batch(BATCH_SIZE, NUM_TIMESTEPS, forever=False)
 133   sum_num = 0.0
 134   sum_den = 0.0
 135   perplexity = 0.0
 136   for i, (inputs, char_inputs, _, targets, weights) in enumerate(data_gen):
 137     input_dict = {t['inputs_in']: inputs,
 138                   t['targets_in']: targets,
 139                   t['target_weights_in']: weights}
 140     if 'char_inputs_in' in t:
 141       input_dict[t['char_inputs_in']] = char_inputs
 142     log_perp = sess.run(t['log_perplexity_out'], feed_dict=input_dict)
 143 
 144     if np.isnan(log_perp):
 145       sys.stderr.error('log_perplexity is Nan.\n')
 146     else:
 147       sum_num += log_perp * weights.mean()
 148       sum_den += weights.mean()
 149     if sum_den > 0:
 150       perplexity = np.exp(sum_num / sum_den)
 151 
 152     sys.stderr.write('Eval Step: %d, Average Perplexity: %f.\n' %
 153                      (i, perplexity))
 154 
 155     if i > FLAGS.max_eval_steps:
 156       break

 159 def _SampleSoftmax(softmax):
 160   return min(np.sum(np.cumsum(softmax) < np.random.rand()), len(softmax) - 1)

 163 def _SampleModel(prefix_words, vocab):
 164   """Predict next words using the given prefix words.
 165 
 166   Args:
 167     prefix_words: Prefix words.
 168     vocab: Vocabulary. Contains max word chard id length and converts between
 169         words and ids.
 170   """
 171   targets = np.zeros([BATCH_SIZE, NUM_TIMESTEPS], np.int32)
 172   weights = np.ones([BATCH_SIZE, NUM_TIMESTEPS], np.float32)
 173 
 174   sess, t = _LoadModel(FLAGS.pbtxt, FLAGS.ckpt)
 175 
 176   if prefix_words.find('<S>') != 0:
 177     prefix_words = '<S> ' + prefix_words
 178 
 179   prefix = [vocab.word_to_id(w) for w in prefix_words.split()]
 180   prefix_char_ids = [vocab.word_to_char_ids(w) for w in prefix_words.split()]
 181   for _ in xrange(FLAGS.num_samples):
 182     inputs = np.zeros([BATCH_SIZE, NUM_TIMESTEPS], np.int32)
 183     char_ids_inputs = np.zeros(
 184         [BATCH_SIZE, NUM_TIMESTEPS, vocab.max_word_length], np.int32)
 185     samples = prefix[:]
 186     char_ids_samples = prefix_char_ids[:]
 187     sent = ''
 188     while True:
 189       inputs[0, 0] = samples[0]
 190       char_ids_inputs[0, 0, :] = char_ids_samples[0]
 191       samples = samples[1:]
 192       char_ids_samples = char_ids_samples[1:]
 193 
 194       softmax = sess.run(t['softmax_out'],
 195                          feed_dict={t['char_inputs_in']: char_ids_inputs,
 196                                     t['inputs_in']: inputs,
 197                                     t['targets_in']: targets,
 198                                     t['target_weights_in']: weights})
 199 
 200       sample = _SampleSoftmax(softmax[0])
 201       sample_char_ids = vocab.word_to_char_ids(vocab.id_to_word(sample))
 202 
 203       if not samples:
 204         samples = [sample]
 205         char_ids_samples = [sample_char_ids]
 206       sent += vocab.id_to_word(samples[0]) + ' '
 207       sys.stderr.write('%s\n' % sent)
 208 
 209       if (vocab.id_to_word(samples[0]) == '</S>' or
 210           len(sent) > FLAGS.max_sample_words):
 211         break

 214 def _DumpEmb(vocab):
 215   """Dump the softmax weights and word embeddings to files.
 216 
 217   Args:
 218     vocab: Vocabulary. Contains vocabulary size and converts word to ids.
 219   """
 220   assert FLAGS.save_dir, 'Must specify FLAGS.save_dir for dump_emb.'
 221   inputs = np.zeros([BATCH_SIZE, NUM_TIMESTEPS], np.int32)
 222   targets = np.zeros([BATCH_SIZE, NUM_TIMESTEPS], np.int32)
 223   weights = np.ones([BATCH_SIZE, NUM_TIMESTEPS], np.float32)
 224 
 225   sess, t = _LoadModel(FLAGS.pbtxt, FLAGS.ckpt)
 226 
 227   softmax_weights = sess.run(t['softmax_weights'])
 228   fname = FLAGS.save_dir + '/embeddings_softmax.npy'
 229   with tf.gfile.Open(fname, mode='w') as f:
 230     np.save(f, softmax_weights)
 231   sys.stderr.write('Finished softmax weights\n')
 232 
 233   all_embs = np.zeros([vocab.size, 1024])
 234   for i in xrange(vocab.size):
 235     input_dict = {t['inputs_in']: inputs,
 236                   t['targets_in']: targets,
 237                   t['target_weights_in']: weights}
 238     if 'char_inputs_in' in t:
 239       input_dict[t['char_inputs_in']] = (
 240           vocab.word_char_ids[i].reshape([-1, 1, MAX_WORD_LEN]))
 241     embs = sess.run(t['all_embs'], input_dict)
 242     all_embs[i, :] = embs
 243     sys.stderr.write('Finished word embedding %d/%d\n' % (i, vocab.size))
 244 
 245   fname = FLAGS.save_dir + '/embeddings_char_cnn.npy'
 246   with tf.gfile.Open(fname, mode='w') as f:
 247     np.save(f, all_embs)
 248   sys.stderr.write('Embedding file saved\n')

 251 def _DumpSentenceEmbedding(sentence, vocab):
 252   """Predict next words using the given prefix words.
 253 
 254   Args:
 255     sentence: Sentence words.
 256     vocab: Vocabulary. Contains max word chard id length and converts between
 257         words and ids.
 258   """
 259   targets = np.zeros([BATCH_SIZE, NUM_TIMESTEPS], np.int32)
 260   weights = np.ones([BATCH_SIZE, NUM_TIMESTEPS], np.float32)
 261 
 262   sess, t = _LoadModel(FLAGS.pbtxt, FLAGS.ckpt)
 263 
 264   if sentence.find('<S>') != 0:
 265     sentence = '<S> ' + sentence
 266 
 267   word_ids = [vocab.word_to_id(w) for w in sentence.split()]
 268   char_ids = [vocab.word_to_char_ids(w) for w in sentence.split()]
 269 
 270   inputs = np.zeros([BATCH_SIZE, NUM_TIMESTEPS], np.int32)
 271   char_ids_inputs = np.zeros(
 272       [BATCH_SIZE, NUM_TIMESTEPS, vocab.max_word_length], np.int32)
 273   for i in xrange(len(word_ids)):
 274     inputs[0, 0] = word_ids[i]
 275     char_ids_inputs[0, 0, :] = char_ids[i]
 276 
 277     # Add 'lstm/lstm_0/control_dependency' if you want to dump previous layer
 278     # LSTM.
 279     lstm_emb = sess.run(t['lstm/lstm_1/control_dependency'],
 280                         feed_dict={t['char_inputs_in']: char_ids_inputs,
 281                                    t['inputs_in']: inputs,
 282                                    t['targets_in']: targets,
 283                                    t['target_weights_in']: weights})
 284 
 285     fname = os.path.join(FLAGS.save_dir, 'lstm_emb_step_%d.npy' % i)
 286     with tf.gfile.Open(fname, mode='w') as f:
 287       np.save(f, lstm_emb)
 288     sys.stderr.write('LSTM embedding step %d file saved\n' % i)

 291 def main(unused_argv):
 292   vocab = data_utils.CharsVocabulary(FLAGS.vocab_file, MAX_WORD_LEN)
 293 
 294   if FLAGS.mode == 'eval':
 295     dataset = data_utils.LM1BDataset(FLAGS.input_data, vocab)
 296     _EvalModel(dataset)
 297   elif FLAGS.mode == 'sample':
 298     _SampleModel(FLAGS.prefix, vocab)
 299   elif FLAGS.mode == 'dump_emb':
 300     _DumpEmb(vocab)
 301   elif FLAGS.mode == 'dump_lstm_emb':
 302     _DumpSentenceEmbedding(FLAGS.sentence, vocab)
 303   else:
 304     raise Exception('Mode not supported.')
 305 
 306 
 307 if __name__ == '__main__':
 308   tf.app.run()
