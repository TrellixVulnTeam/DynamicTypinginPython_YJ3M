  39   Attributes:

  58   def __init__(self, hparams, dataset_info, targets, use_tpu=False):
  59     """Initializes the ConvolutionalNet according to provided hyperparameters.
  60 
  61     Does not build the graph---this is done by calling `build_graph` on the
  62     constructed object or using `model_fn`.
  63 
  64     Args:
  65       hparams: tf.contrib.training.Hparams object containing the model's
  66         hyperparamters; see configuration.py for hyperparameter definitions.
  67       dataset_info: a `Seq2LabelDatasetInfo` message reflecting the dataset
  68         metadata.
  69       targets: list of strings: the names of the prediction targets.
  70       use_tpu: whether we are running on TPU; if True, summaries will be
  71         disabled.
  72     """
  73     self._placeholders = {}
  74     self._targets = targets
  75     self._dataset_info = dataset_info
  76     self._hparams = hparams
  77     all_label_values = seq2label_utils.get_all_label_values(self.dataset_info)
  78     self._possible_labels = {
  79         target: all_label_values[target]
  80         for target in self.targets
  81     }
  82     self._use_tpu = use_tpu

 156   def _summary_scalar(self, name, scalar):
 157     """Adds a summary scalar, if the platform supports summaries."""
 158     if not self.use_tpu:
 159       return tf.summary.scalar(name, scalar)
 160     else:
 161       return None

 163   def _summary_histogram(self, name, values):
 164     """Adds a summary histogram, if the platform supports summaries."""
 165     if not self.use_tpu:
 166       return tf.summary.histogram(name, values)
 167     else:
 168       return None

 170   def _init_weights(self, shape, scale=1.0, name='weights'):
 171     """Randomly initializes a weight Tensor of the given shape.
 172 
 173     Args:
 174       shape: list; desired Tensor dimensions.
 175       scale: float; standard deviation scale with which to initialize weights.
 176       name: string name for the variable.
 177 
 178     Returns:
 179       TF Variable contining truncated random Normal initialized weights.
 180     """
 181     num_inputs = shape[0] if len(shape) < 3 else shape[0] * shape[1] * shape[2]
 182     stddev = scale / math.sqrt(num_inputs)
 183     return tf.get_variable(
 184         name,
 185         shape=shape,
 186         initializer=tf.truncated_normal_initializer(0., stddev))
 187 
 188   def _init_bias(self, size):
 189     """Initializes bias vector of given shape as zeros.
 190 
 191     Args:
 192       size: int; desired size of bias Tensor.
 193 
 194     Returns:
 195       TF Variable containing the initialized biases.
 196     """
 197     return tf.get_variable(
 198         name='b_{}'.format(size),
 199         shape=[size],
 200         initializer=tf.zeros_initializer())

 202   def _add_summaries(self, mode, gradient_norm, parameter_norm):
 203     """Defines TensorFlow operation for logging summaries to event files.
 204 
 205     Args:
 206       mode: the ModeKey string.
 207       gradient_norm: Tensor; norm of gradients produced during the current
 208         training operation.
 209       parameter_norm: Tensor; norm of the model parameters produced during the
 210         current training operation.
 211     """
 212     # Log summaries for TensorBoard.
 213     if mode == tf.estimator.ModeKeys.TRAIN:
 214       self._summary_scalar('norm_of_gradients', gradient_norm)
 215       self._summary_scalar('norm_of_parameters', parameter_norm)
 216       self._summary_scalar('total_loss', self.total_loss)
 217       self._summary_scalar('learning_rate', self._learn_rate)
 218       for target in self.targets:
 219         self._summary_scalar('per_read_weighted_accuracy/{}'.format(target),
 220                              self.weighted_accuracy[target])
 221         self._summary_scalar('per_read_accuracy/{}'.format(target),
 222                              self.accuracy[target])
 223         self._summary_histogram('prediction_frequency/{}'.format(target),
 224                                 self._predictions[target])
 225         self._summary_scalar('cross_entropy_loss/{}'.format(target),
 226                              self._loss[target])
 227       self._summary_op = tf.summary.merge_all()
 228     else:
 229       # Log average performance metrics over many batches using placeholders.
 230       summaries = []
 231       for target in self.targets:
 232         accuracy_ph = tf.placeholder(tf.float32, shape=())
 233         weighted_accuracy_ph = tf.placeholder(tf.float32, shape=())
 234         cross_entropy_ph = tf.placeholder(tf.float32, shape=())
 235         self._placeholders.update({
 236             'accuracy/{}'.format(target): accuracy_ph,
 237             'weighted_accuracy/{}'.format(target): weighted_accuracy_ph,
 238             'cross_entropy/{}'.format(target): cross_entropy_ph,
 239         })
 240         summaries += [
 241             self._summary_scalar('cross_entropy_loss/{}'.format(target),
 242                                  cross_entropy_ph),
 243             self._summary_scalar('per_read_accuracy/{}'.format(target),
 244                                  accuracy_ph),
 245             self._summary_scalar('per_read_weighted_accuracy/{}'.format(target),
 246                                  weighted_accuracy_ph)
 247         ]
 248 
 249       self._summary_op = tf.summary.merge(summaries)

 251   def _convolution(self,
 252                    inputs,
 253                    filter_dim,
 254                    pointwise_dim=None,
 255                    scale=1.0,
 256                    padding='SAME'):
 257     """Applies convolutional filter of given dimensions to given input Tensor.
 258 
 259     If a pointwise dimension is specified, a depthwise separable convolution is
 260     performed.
 261 
 262     Args:
 263       inputs: 4D Tensor of shape (# reads, 1, # basepairs, # bases).
 264       filter_dim: integer tuple of the form (width, depth).
 265       pointwise_dim: int; output dimension for pointwise convolution.
 266       scale: float; standard deviation scale with which to initialize weights.
 267       padding: string; type of padding to use. One of "SAME" or "VALID".
 268 
 269     Returns:
 270       4D Tensor result of applying the convolutional filter to the inputs.
 271     """
 272     in_channels = inputs.get_shape()[3].value
 273     filter_width, filter_depth = filter_dim
 274     filters = self._init_weights([1, filter_width, in_channels, filter_depth],
 275                                  scale)
 276     self._summary_histogram(filters.name.split(':')[0].split('/')[1], filters)
 277     if pointwise_dim is None:
 278       return tf.nn.conv2d(
 279           inputs,
 280           filters,
 281           strides=[1, 1, 1, 1],
 282           padding=padding,
 283           name='weights')
 284     pointwise_filters = self._init_weights(
 285         [1, 1, filter_depth * in_channels, pointwise_dim],
 286         scale,
 287         name='pointwise_weights')
 288     self._summary_histogram(
 289         pointwise_filters.name.split(':')[0].split('/')[1], pointwise_filters)
 290     return tf.nn.separable_conv2d(
 291         inputs,
 292         filters,
 293         pointwise_filters,
 294         strides=[1, 1, 1, 1],
 295         padding=padding)

 297   def _pool(self, inputs, pooling_type):
 298     """Performs pooling across width and height of the given inputs.
 299 
 300     Args:
 301       inputs: Tensor shaped (batch, height, width, channels) over which to pool.
 302         In our case, height is a unitary dimension and width can be thought of
 303         as the read dimension.
 304       pooling_type: string; one of "avg" or "max".
 305 
 306     Returns:
 307       Tensor result of performing pooling of the given pooling_type over the
 308       height and width dimensions of the given inputs.
 309     """
 310     if pooling_type == 'max':
 311       return tf.reduce_max(inputs, axis=[1, 2])
 312     if pooling_type == 'avg':
 313       return tf.reduce_sum(
 314           inputs, axis=[1, 2]) / tf.to_float(tf.shape(inputs)[2])
 315 
 316   def _leaky_relu(self, lrelu_slope, inputs):
 317     """Applies leaky ReLu activation to the given inputs with the given slope.
 318 
 319     Args:
 320       lrelu_slope: float; slope value for the activation function.
 321         A slope of 0.0 defines a standard ReLu activation, while a positive
 322         slope defines a leaky ReLu.
 323       inputs: Tensor upon which to apply the activation function.
 324 
 325     Returns:
 326       Tensor result of applying the activation function to the given inputs.
 327     """
 328     with tf.variable_scope('leaky_relu_activation'):
 329       return tf.maximum(lrelu_slope * inputs, inputs)
 330 
 331   def _dropout(self, inputs, keep_prob):
 332     """Applies dropout to the given inputs.
 333 
 334     Args:
 335       inputs: Tensor upon which to apply dropout.
 336       keep_prob: float; probability with which to randomly retain values in
 337         the given input.
 338 
 339     Returns:
 340       Tensor result of applying dropout to the given inputs.
 341     """
 342     with tf.variable_scope('dropout'):
 343       if keep_prob < 1.0:
 344         return tf.nn.dropout(inputs, keep_prob)
 345       return inputs

 347   def build_graph(self, features, labels, mode, batch_size):
 348     """Creates TensorFlow model graph.
 349 
 350     Args:
 351       features: a dict of input features Tensors.
 352       labels: a dict (by target name) of prediction labels.
 353       mode: the ModeKey string.
 354       batch_size: the integer batch size.
 355 
 356     Side Effect:
 357       Adds the following key Tensors and operations as class attributes:
 358         placeholders, global_step, train_op, summary_op, accuracy,
 359         weighted_accuracy, loss, logits, and predictions.
 360     """
 361     is_train = (mode == tf.estimator.ModeKeys.TRAIN)
 362     read = features['sequence']
 363 
 364     # Add a unitary dimension, so we can use conv2d.
 365     read = tf.expand_dims(read, 1)
 366     prev_out = read
 367 
 368     filters = zip(self.hparams.filter_widths, self.hparams.filter_depths)
 369     for i, f in enumerate(filters):
 370       with tf.variable_scope('convolution_' + str(i)):
 371         if self.hparams.use_depthwise_separable:
 372           p = self.hparams.pointwise_depths[i]
 373         else:
 374           p = None
 375         conv_out = self._convolution(
 376             prev_out, f, pointwise_dim=p, scale=self.hparams.weight_scale)
 377         conv_act_out = self._leaky_relu(self.hparams.lrelu_slope, conv_out)
 378         prev_out = (
 379             self._dropout(conv_act_out, self.hparams.keep_prob)
 380             if is_train else conv_act_out)
 381 
 382     for i in xrange(self.hparams.num_fc_layers):
 383       with tf.variable_scope('fully_connected_' + str(i)):
 384         # Create a convolutional layer which is equivalent to a fully-connected
 385         # layer when reads have length self.hparams.min_read_length.
 386         # The convolution will tile the layer appropriately for longer reads.
 387         biases = self._init_bias(self.hparams.num_fc_units)
 388         if i == 0:
 389           # Take entire min_read_length segment as input.
 390           # Output a single value per min_read_length_segment.
 391           filter_dimensions = (self.hparams.min_read_length,
 392                                self.hparams.num_fc_units)
 393         else:
 394           # Take single output value of previous layer as input.
 395           filter_dimensions = (1, self.hparams.num_fc_units)
 396         fc_out = biases + self._convolution(
 397             prev_out,
 398             filter_dimensions,
 399             scale=self.hparams.weight_scale,
 400             padding='VALID')
 401         self._summary_histogram(biases.name.split(':')[0].split('/')[1], biases)
 402         fc_act_out = self._leaky_relu(self.hparams.lrelu_slope, fc_out)
 403         prev_out = (
 404             self._dropout(fc_act_out, self.hparams.keep_prob)
 405             if is_train else fc_act_out)
 406 
 407     # Pool to collapse tiling for reads longer than hparams.min_read_length.
 408     with tf.variable_scope('pool'):
 409       pool_out = self._pool(prev_out, self.hparams.pooling_type)
 410 
 411     with tf.variable_scope('output'):
 412       self._logits = {}
 413       self._predictions = {}
 414       self._weighted_accuracy = {}
 415       self._accuracy = {}
 416       self._loss = collections.OrderedDict()
 417 
 418       for target in self.targets:
 419         with tf.variable_scope(target):
 420           label = labels[target]
 421           possible_labels = self.possible_labels[target]
 422           weights = self._init_weights(
 423               [pool_out.get_shape()[1].value,
 424                len(possible_labels)],
 425               self.hparams.weight_scale,
 426               name='weights')
 427           biases = self._init_bias(len(possible_labels))
 428           self._summary_histogram(
 429               weights.name.split(':')[0].split('/')[1], weights)
 430           self._summary_histogram(
 431               biases.name.split(':')[0].split('/')[1], biases)
 432           logits = tf.matmul(pool_out, weights) + biases
 433           predictions = tf.nn.softmax(logits)
 434 
 435           gather_inds = tf.stack([tf.range(batch_size), label], axis=1)
 436           self._weighted_accuracy[target] = tf.reduce_mean(
 437               tf.gather_nd(predictions, gather_inds))
 438           argmax_prediction = tf.cast(tf.argmax(predictions, axis=1), tf.int32)
 439           self._accuracy[target] = tf.reduce_mean(
 440               tf.to_float(tf.equal(label, argmax_prediction)))
 441 
 442           losses = tf.nn.sparse_softmax_cross_entropy_with_logits(
 443               labels=label, logits=logits)
 444           self._loss[target] = tf.reduce_mean(losses)
 445           self._logits[target] = logits
 446           self._predictions[target] = predictions
 447 
 448     # Compute total loss
 449     self._total_loss = tf.add_n(self._loss.values())
 450 
 451     # Define the optimizer.
 452 
 453     # tf.estimator framework builds the global_step for us, but if we aren't
 454     # using the framework we have to make it ourselves.
 455     self._global_step = tf.train.get_or_create_global_step()
 456     if self.hparams.lr_decay < 0:
 457       self._learn_rate = self.hparams.lr_init
 458     else:
 459       self._learn_rate = tf.train.exponential_decay(
 460           self.hparams.lr_init,
 461           self._global_step,
 462           int(self.hparams.train_steps),
 463           self.hparams.lr_decay,
 464           staircase=False)
 465     if self.hparams.optimizer == 'adam':
 466       opt = tf.train.AdamOptimizer(self._learn_rate, self.hparams.optimizer_hp)
 467     elif self.hparams.optimizer == 'momentum':
 468       opt = tf.train.MomentumOptimizer(self._learn_rate,
 469                                        self.hparams.optimizer_hp)
 470     if self.use_tpu:
 471       opt = tf.contrib.tpu.CrossShardOptimizer(opt)
 472 
 473     gradients, variables = zip(*opt.compute_gradients(self._total_loss))
 474     clipped_gradients, _ = tf.clip_by_global_norm(gradients,
 475                                                   self.hparams.grad_clip_norm)
 476     with tf.control_dependencies(tf.get_collection(tf.GraphKeys.UPDATE_OPS)):
 477       self._train_op = opt.apply_gradients(
 478           zip(clipped_gradients, variables), global_step=self._global_step)
 479 
 480     if not self.use_tpu:
 481       grad_norm = tf.global_norm(gradients) if is_train else None
 482       param_norm = tf.global_norm(variables) if is_train else None
 483       self._add_summaries(mode, grad_norm, param_norm)
 484 
 485   def model_fn(self, features, labels, mode, params):
 486     """Function fulfilling the tf.estimator model_fn interface.
 487 
 488     Args:
 489       features: a dict containing the input features for prediction.
 490       labels: a dict from target name to Tensor-value prediction.
 491       mode: the ModeKey string.
 492       params: a dictionary of parameters for building the model; current params
 493         are params["batch_size"]: the integer batch size.
 494 
 495     Returns:
 496       A tf.estimator.EstimatorSpec object ready for use in training, inference.
 497       or evaluation.
 498     """
 499     self.build_graph(features, labels, mode, params['batch_size'])
 500 
 501     return tf.estimator.EstimatorSpec(
 502         mode,
 503         predictions=self.predictions,
 504         loss=self.total_loss,
 505         train_op=self.train_op,
 506         eval_metric_ops={})
