  29 DNA_BASES = tuple('ACGT')
  30 NUM_DNA_BASES = len(DNA_BASES)
  31 # Possible FASTA characters/IUPAC ambiguity codes.
  32 # See https://en.wikipedia.org/wiki/Nucleic_acid_notation.
  33 AMBIGUITY_CODES = {
  34     'K': 'GT',
  35     'M': 'AC',
  36     'R': 'AG',
  37     'Y': 'CT',
  38     'S': 'CG',
  39     'W': 'AT',
  40     'B': 'CGT',
  41     'V': 'ACG',
  42     'H': 'ACT',
  43     'D': 'AGT',
  44     'X': 'ACGT',
  45     'N': 'ACGT'
  46 }

  49 def load_dataset_info(dataset_info_path):
  50   """Load a `Seq2LabelDatasetInfo` from a serialized text proto file."""
  51   dataset_info = seq2label_pb2.Seq2LabelDatasetInfo()
  52   with tf.gfile.Open(dataset_info_path, 'r') as f:
  53     text_format.Parse(f.read(), dataset_info)
  54   return dataset_info

  57 class _InputEncoding(object):
  58   """A helper class providing the graph operations needed to encode input.
  59 
  60   Instantiation of an _InputEncoding will write on the default TF graph, so it
  61   should only be instantiated inside the `input_fn`.
  62 
  63   Attributes:
  64     mode: `tf.estimator.ModeKeys`; the execution mode {TRAIN, EVAL, INFER}.
  65     targets: list of strings; the names of the labels of interest (e.g.
  66       "species").
  67     dna_bases: a tuple of the recognized DNA alphabet.
  68     n_bases: the size of the DNA alphabet.
  69     all_characters: list of recognized alphabet, including ambiguity codes.
  70     label_values: a tuple of strings, the possible label values of the
  71       prediction target.
  72     n_labels: the size of label_values
  73     fixed_read_length: an integer value of the statically-known read length, or
  74      None if the read length is to be determined dynamically.
  75   """
  76 
  77   def __init__(self,
  78                dataset_info,
  79                mode,
  80                targets,
  81                noise_rate=0.0,
  82                fixed_read_length=None):
  83     self.mode = mode
  84     self.targets = targets
  85     self.dna_bases = DNA_BASES
  86     self.n_bases = NUM_DNA_BASES
  87     self.all_characters = list(DNA_BASES) + sorted(AMBIGUITY_CODES.keys())
  88     self.character_encodings = np.concatenate(
  89         [[self._character_to_base_distribution(char)]
  90          for char in self.all_characters],
  91         axis=0)
  92     all_legal_label_values = seq2label_utils.get_all_label_values(dataset_info)
  93     # TF lookup tables.
  94     self.characters_table = tf.contrib.lookup.index_table_from_tensor(
  95         mapping=self.all_characters)
  96     self.label_tables = {
  97         target: tf.contrib.lookup.index_table_from_tensor(
  98             all_legal_label_values[target])
  99         for target in targets
 100     }
 101     self.fixed_read_length = fixed_read_length
 102     self.noise_rate = noise_rate
 103 
 104   def _character_to_base_distribution(self, char):
 105     """Maps the given character to a probability distribution over DNA bases.
 106 
 107     Args:
 108       char: character to be encoded as a probability distribution over bases.
 109 
 110     Returns:
 111       Array of size (self.n_bases,) representing the identity of the given
 112       character as a distribution over the possible DNA bases, self.dna_bases.
 113 
 114     Raises:
 115       ValueError: if the given character is not contained in the recognized
 116         alphabet, self.all_characters.
 117     """
 118     if char not in self.all_characters:
 119       raise ValueError(
 120           'Base distribution requested for unrecognized character %s.' % char)
 121     possible_bases = AMBIGUITY_CODES[char] if char in AMBIGUITY_CODES else char
 122     base_indices = [self.dna_bases.index(base) for base in possible_bases]
 123     probability_weight = 1.0 / len(possible_bases)
 124     distribution = np.zeros((self.n_bases))
 125     distribution[base_indices] = probability_weight
 126     return distribution
 127 
 128   def encode_read(self, string_seq):
 129     """Converts the input read sequence to one-hot encoding.
 130 
 131     Args:
 132       string_seq: tf.String; input read sequence.
 133 
 134     Returns:
 135       Input read sequence as a one-hot encoded Tensor, with depth and ordering
 136       of one-hot encoding determined by the given bases. Ambiguous characters
 137       such as "N" and "S" are encoded as a probability distribution over the
 138       possible bases they represent.
 139     """
 140     with tf.variable_scope('encode_read'):
 141       read = tf.string_split([string_seq], delimiter='').values
 142       read = self.characters_table.lookup(read)
 143       read = tf.cast(tf.gather(self.character_encodings, read), tf.float32)
 144       if self.fixed_read_length:
 145         read = tf.reshape(read, (self.fixed_read_length, self.n_bases))
 146       return read
 147 
 148   def encode_label(self, target, string_label):
 149     """Converts the label value to an integer encoding.
 150 
 151     Args:
 152       target: str; the target name.
 153       string_label: tf.String; value of the label for the current input read.
 154 
 155     Returns:
 156       Given label value as an index into the possible_target_values.
 157     """
 158     with tf.variable_scope('encode_label/{}'.format(target)):
 159       return tf.cast(self.label_tables[target].lookup(string_label), tf.int32)
 160 
 161   def _empty_label(self):
 162     return tf.constant((), dtype=tf.int32, shape=())
 163 
 164   def parse_single_tfexample(self, serialized_example):
 165     """Parses a tf.train.Example proto to a one-hot encoded read, label pair.
 166 
 167     Injects noise into the incoming tf.train.Example's read sequence
 168     when noise_rate is non-zero.
 169 
 170     Args:
 171       serialized_example: string; the serialized tf.train.Example proto
 172         containing the read sequence and label value of interest as
 173         tf.FixedLenFeatures.
 174 
 175     Returns:
 176       Tuple (features, labels) of dicts for the input features and prediction
 177       targets.
 178     """
 179     with tf.variable_scope('parse_single_tfexample'):
 180       features_spec = {'sequence': tf.FixedLenFeature([], tf.string)}
 181       for target in self.targets:
 182         features_spec[target] = tf.FixedLenFeature([], tf.string)
 183       features = tf.parse_single_example(
 184           serialized_example, features=features_spec)
 185       if self.noise_rate > 0.0:
 186         read_sequence = tf.py_func(seq2label_utils.add_read_noise,
 187                                    [features['sequence'], self.noise_rate],
 188                                    (tf.string))
 189       else:
 190         read_sequence = features['sequence']
 191       read_sequence = self.encode_read(read_sequence)
 192       read_features = {'sequence': read_sequence}
 193       if self.mode in (tf.estimator.ModeKeys.TRAIN, tf.estimator.ModeKeys.EVAL):
 194         label = {
 195             target: self.encode_label(target, features[target])
 196             for target in self.targets
 197         }
 198       else:
 199         label = {target: self._empty_label() for target in self.targets}
 200       return read_features, label

 203 class InputDataset(object):
 204   """A class providing access to input data for the Seq2Label model.
 205 
 206   Attributes:
 207     mode: `tf.estimator.ModeKeys`; the execution mode {TRAIN, EVAL, INFER}.
 208     targets: list of strings; the names of the labels of interest (e.g.
 209       "species").
 210     dataset_info: a `Seq2LabelDatasetInfo` message reflecting the dataset
 211       metadata.
 212     initializer: the TF initializer op for the underlying iterator, which
 213       will rewind the iterator.
 214     is_train: Boolean indicating whether or not the execution mode is TRAIN.
 215   """
 216 
 217   def __init__(self,
 218                mode,
 219                targets,
 220                dataset_info,
 221                train_epochs=None,
 222                noise_rate=0.0,
 223                random_seed=None,
 224                input_tfrecord_files=None,
 225                fixed_read_length=None,
 226                ensure_constant_batch_size=False,
 227                num_parallel_calls=32):
 228     """Constructor for InputDataset.
 229 
 230     Args:
 231       mode: `tf.estimator.ModeKeys`; the execution mode {TRAIN, EVAL, INFER}.
 232       targets: list of strings; the names of the labels of interest (e.g.
 233         "species").
 234       dataset_info: a `Seq2LabelDatasetInfo` message reflecting the dataset
 235         metadata.
 236       train_epochs: the number of training epochs to perform, if mode==TRAIN.
 237       noise_rate: float [0.0, 1.0] specifying rate at which to inject
 238         base-flipping noise into the read sequences.
 239       random_seed: seed to be used for shuffling, if mode==TRAIN.
 240       input_tfrecord_files: a list of filenames for TFRecords of TF examples.
 241       fixed_read_length: an integer value of the statically-known read length,
 242         or None if the read length is to be determined dynamically.  The read
 243         length must be known statically for TPU execution.
 244       ensure_constant_batch_size: ensure a constant batch size at the expense of
 245         discarding the last "short" batch.  This also gives us a statically
 246         constant batch size, which is essential for e.g. the TPU platform.
 247       num_parallel_calls: the number of dataset elements to process in parallel.
 248         If None, elements will be processed sequentially.
 249     """
 250     self.input_tfrecord_files = input_tfrecord_files
 251     self.mode = mode
 252     self.targets = targets
 253     self.dataset_info = dataset_info
 254     self._train_epochs = train_epochs
 255     self._noise_rate = noise_rate
 256     self._random_seed = random_seed
 257     if random_seed is not None:
 258       np.random.seed(random_seed)
 259     self._fixed_read_length = fixed_read_length
 260     self._ensure_constant_batch_size = ensure_constant_batch_size
 261     self._num_parallel_calls = num_parallel_calls
 262 
 263   @staticmethod
 264   def from_tfrecord_files(input_tfrecord_files, *args, **kwargs):
 265     return InputDataset(
 266         *args, input_tfrecord_files=input_tfrecord_files, **kwargs)
 267 
 268   @property
 269   def is_train(self):
 270     return self.mode == tf.estimator.ModeKeys.TRAIN
 271 
 272   def input_fn(self, params):
 273     """Supplies input for the model.
 274 
 275     This function supplies input to our model as a function of the mode.
 276 
 277     Args:
 278       params: a dictionary, containing:
 279         - params['batch_size']: the integer batch size.
 280 
 281     Returns:
 282       A tuple of two values as follows:
 283        1) the *features* dict, containing a tensor value for keys as follows:
 284             - "sequence" - the encoded read input sequence.
 285        2) the *labels* dict. containing a key for `target`, whose value is:
 286            - a string Tensor value (in TRAIN/EVAL mode), or
 287            - a blank Tensor (PREDICT mode).
 288     """
 289     randomize_input = self.is_train
 290     batch_size = params['batch_size']
 291 
 292     encoding = _InputEncoding(
 293         self.dataset_info,
 294         self.mode,
 295         self.targets,
 296         noise_rate=self._noise_rate,
 297         fixed_read_length=self._fixed_read_length)
 298 
 299     dataset = tf.data.TFRecordDataset(self.input_tfrecord_files)
 300     dataset = dataset.map(
 301         encoding.parse_single_tfexample,
 302         num_parallel_calls=self._num_parallel_calls)
 303 
 304     dataset = dataset.repeat(self._train_epochs if self.is_train else 1)
 305     if randomize_input:
 306       dataset = dataset.shuffle(
 307           buffer_size=max(1000, batch_size), seed=self._random_seed)
 308 
 309     if self._ensure_constant_batch_size:
 310       # Only take batches of *exactly* size batch_size; then we get a
 311       # statically knowable batch shape.
 312       dataset = dataset.batch(batch_size, drop_remainder=True)
 313     else:
 314       dataset = dataset.batch(batch_size)
 315 
 316     # Prefetch to allow infeed to be in parallel with model computations.
 317     dataset = dataset.prefetch(2)
 318 
 319     # Use initializable iterator to support table lookups.
 320     iterator = dataset.make_initializable_iterator()
 321     self.initializer = iterator.initializer
 322     tf.add_to_collection(tf.GraphKeys.TABLE_INITIALIZERS, iterator.initializer)
 323 
 324     features, labels = iterator.get_next()
 325     return (features, labels)
