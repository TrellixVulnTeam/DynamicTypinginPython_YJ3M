  36 def create_test_input(batch, height, width, channels):
  37   """Create test input tensor."""
  38   if None in [batch, height, width, channels]:
  39     return tf.placeholder(tf.float32, (batch, height, width, channels))
  40   else:
  41     return tf.to_float(
  42         np.tile(
  43             np.reshape(
  44                 np.reshape(np.arange(height), [height, 1]) +
  45                 np.reshape(np.arange(width), [1, width]),
  46                 [1, height, width, 1]),
  47             [batch, 1, 1, channels]))

  53   def _resnet_small_lite_bottleneck(self,
  54                                     inputs,
  55                                     num_classes=None,
  56                                     is_training=True,
  57                                     global_pool=True,
  58                                     output_stride=None,
  59                                     multi_grid=None,
  60                                     reuse=None,
  61                                     scope='resnet_v1_small'):
  62     """A shallow and thin ResNet v1 with lite_bottleneck."""
  63     if multi_grid is None:
  64       multi_grid = [1, 1]
  65     else:
  66       if len(multi_grid) != 2:
  67         raise ValueError('Expect multi_grid to have length 2.')
  68     block = resnet_v1_beta.resnet_v1_small_beta_block
  69     blocks = [
  70         block('block1', base_depth=1, num_units=1, stride=2),
  71         block('block2', base_depth=2, num_units=1, stride=2),
  72         block('block3', base_depth=4, num_units=1, stride=2),
  73         resnet_utils.Block('block4', resnet_v1_beta.lite_bottleneck, [
  74             {'depth': 8,
  75              'stride': 1,
  76              'unit_rate': rate} for rate in multi_grid])]
  77     return resnet_v1_beta.resnet_v1_beta(
  78         inputs,
  79         blocks,
  80         num_classes=num_classes,
  81         is_training=is_training,
  82         global_pool=global_pool,
  83         output_stride=output_stride,
  84         root_block_fn=functools.partial(
  85             resnet_v1_beta.root_block_fn_for_beta_variant,
  86             depth_multiplier=0.25),
  87         reuse=reuse,
  88         scope=scope)

  90   def _resnet_small(self,
  91                     inputs,
  92                     num_classes=None,
  93                     is_training=True,
  94                     global_pool=True,
  95                     output_stride=None,
  96                     multi_grid=None,
  97                     reuse=None,
  98                     scope='resnet_v1_small'):
  99     """A shallow and thin ResNet v1 for faster tests."""
 100     if multi_grid is None:
 101       multi_grid = [1, 1, 1]
 102     else:
 103       if len(multi_grid) != 3:
 104         raise ValueError('Expect multi_grid to have length 3.')
 105 
 106     block = resnet_v1_beta.resnet_v1_beta_block
 107     blocks = [
 108         block('block1', base_depth=1, num_units=1, stride=2),
 109         block('block2', base_depth=2, num_units=1, stride=2),
 110         block('block3', base_depth=4, num_units=1, stride=2),
 111         resnet_utils.Block('block4', resnet_v1_beta.bottleneck, [
 112             {'depth': 32, 'depth_bottleneck': 8, 'stride': 1,
 113              'unit_rate': rate} for rate in multi_grid])]
 114 
 115     return resnet_v1_beta.resnet_v1_beta(
 116         inputs,
 117         blocks,
 118         num_classes=num_classes,
 119         is_training=is_training,
 120         global_pool=global_pool,
 121         output_stride=output_stride,
 122         root_block_fn=functools.partial(
 123             resnet_v1_beta.root_block_fn_for_beta_variant),
 124         reuse=reuse,
 125         scope=scope)

 228   def testAtrousFullyConvolutionalValuesWithLiteBottleneck(self):
 229     """Verify dense feature extraction with atrous convolution."""
 230     nominal_stride = 32
 231     for output_stride in [4, 8, 16, 32, None]:
 232       with slim.arg_scope(resnet_utils.resnet_arg_scope()):
 233         with tf.Graph().as_default():
 234           with self.test_session() as sess:
 235             tf.set_random_seed(0)
 236             inputs = create_test_input(2, 81, 81, 3)
 237             # Dense feature extraction followed by subsampling.
 238             output, _ = self._resnet_small_lite_bottleneck(
 239                 inputs,
 240                 None,
 241                 is_training=False,
 242                 global_pool=False,
 243                 output_stride=output_stride)
 244             if output_stride is None:
 245               factor = 1
 246             else:
 247               factor = nominal_stride // output_stride
 248             output = resnet_utils.subsample(output, factor)
 249             # Make the two networks use the same weights.
 250             tf.get_variable_scope().reuse_variables()
 251             # Feature extraction at the nominal network rate.
 252             expected, _ = self._resnet_small_lite_bottleneck(
 253                 inputs,
 254                 None,
 255                 is_training=False,
 256                 global_pool=False)
 257             sess.run(tf.global_variables_initializer())
 258             self.assertAllClose(output.eval(), expected.eval(),
 259                                 atol=1e-4, rtol=1e-4)
