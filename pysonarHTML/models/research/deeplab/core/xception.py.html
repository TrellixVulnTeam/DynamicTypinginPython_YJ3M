  17 r"""Xception model.
  18 
  19 "Xception: Deep Learning with Depthwise Separable Convolutions"
  20 Fran{\c{c}}ois Chollet
  21 https://arxiv.org/abs/1610.02357
  22 
  23 We implement the modified version by Jifeng Dai et al. for their COCO 2017
  24 detection challenge submission, where the model is made deeper and has aligned
  25 features for dense prediction tasks. See their slides for details:
  26 
  27 "Deformable Convolutional Networks -- COCO Detection and Segmentation Challenge
  28 2017 Entry"
  29 Haozhi Qi, Zheng Zhang, Bin Xiao, Han Hu, Bowen Cheng, Yichen Wei and Jifeng Dai
  30 ICCV 2017 COCO Challenge workshop
  31 http://presentations.cocodataset.org/COCO17-Detect-MSRA.pdf
  32 
  33 We made a few more changes on top of MSRA's modifications:
  34 1. Fully convolutional: All the max-pooling layers are replaced with separable
  35   conv2d with stride = 2. This allows us to use atrous convolution to extract
  36   feature maps at any resolution.
  37 
  38 2. We support adding ReLU and BatchNorm after depthwise convolution, motivated
  39   by the design of MobileNetv1.
  40 
  41 "MobileNets: Efficient Convolutional Neural Networks for Mobile Vision
  42 Applications"
  43 Andrew G. Howard, Menglong Zhu, Bo Chen, Dmitry Kalenichenko, Weijun Wang,
  44 Tobias Weyand, Marco Andreetto, Hartwig Adam
  45 https://arxiv.org/abs/1704.04861

  68 class Block(collections.namedtuple('Block', ['scope', 'unit_fn', 'args'])):
  69   """A named tuple describing an Xception block.
  70 
  71   Its parts are:
  72     scope: The scope of the block.
  73     unit_fn: The Xception unit function which takes as input a tensor and
  74       returns another tensor with the output of the Xception unit.
  75     args: A list of length equal to the number of units in the block. The list
  76       contains one dictionary for each unit in the block to serve as argument to
  77       unit_fn.
  78   """

  81 def fixed_padding(inputs, kernel_size, rate=1):
  82   """Pads the input along the spatial dimensions independently of input size.
  83 
  84   Args:
  85     inputs: A tensor of size [batch, height_in, width_in, channels].
  86     kernel_size: The kernel to be used in the conv2d or max_pool2d operation.
  87                  Should be a positive integer.
  88     rate: An integer, rate for atrous convolution.
  89 
  90   Returns:
  91     output: A tensor of size [batch, height_out, width_out, channels] with the
  92       input, either intact (if kernel_size == 1) or padded (if kernel_size > 1).
  93   """
  94   kernel_size_effective = kernel_size + (kernel_size - 1) * (rate - 1)
  95   pad_total = kernel_size_effective - 1
  96   pad_beg = pad_total // 2
  97   pad_end = pad_total - pad_beg
  98   padded_inputs = tf.pad(inputs, [[0, 0], [pad_beg, pad_end],
  99                                   [pad_beg, pad_end], [0, 0]])
 100   return padded_inputs

 103 @slim.add_arg_scope
 104 def separable_conv2d_same(inputs,
 105                           num_outputs,
 106                           kernel_size,
 107                           depth_multiplier,
 108                           stride,
 109                           rate=1,
 110                           use_explicit_padding=True,
 111                           regularize_depthwise=False,
 112                           scope=None,
 113                           **kwargs):
 114   """Strided 2-D separable convolution with 'SAME' padding.
 115 
 116   If stride > 1 and use_explicit_padding is True, then we do explicit zero-
 117   padding, followed by conv2d with 'VALID' padding.
 118 
 119   Note that
 120 
 121      net = separable_conv2d_same(inputs, num_outputs, 3,
 122        depth_multiplier=1, stride=stride)
 123 
 124   is equivalent to
 125 
 126      net = slim.separable_conv2d(inputs, num_outputs, 3,
 127        depth_multiplier=1, stride=1, padding='SAME')
 128      net = resnet_utils.subsample(net, factor=stride)
 129 
 130   whereas
 131 
 132      net = slim.separable_conv2d(inputs, num_outputs, 3, stride=stride,
 133        depth_multiplier=1, padding='SAME')
 134 
 135   is different when the input's height or width is even, which is why we add the
 136   current function.
 137 
 138   Consequently, if the input feature map has even height or width, setting
 139   `use_explicit_padding=False` will result in feature misalignment by one pixel
 140   along the corresponding dimension.
 141 
 142   Args:
 143     inputs: A 4-D tensor of size [batch, height_in, width_in, channels].
 144     num_outputs: An integer, the number of output filters.
 145     kernel_size: An int with the kernel_size of the filters.
 146     depth_multiplier: The number of depthwise convolution output channels for
 147       each input channel. The total number of depthwise convolution output
 148       channels will be equal to `num_filters_in * depth_multiplier`.
 149     stride: An integer, the output stride.
 150     rate: An integer, rate for atrous convolution.
 151     use_explicit_padding: If True, use explicit padding to make the model fully
 152       compatible with the open source version, otherwise use the native
 153       Tensorflow 'SAME' padding.
 154     regularize_depthwise: Whether or not apply L2-norm regularization on the
 155       depthwise convolution weights.
 156     scope: Scope.
 157     **kwargs: additional keyword arguments to pass to slim.conv2d
 158 
 159   Returns:
 160     output: A 4-D tensor of size [batch, height_out, width_out, channels] with
 161       the convolution output.
 162   """
 163   def _separable_conv2d(padding):
 164     """Wrapper for separable conv2d."""
 165     return slim.separable_conv2d(inputs,
 166                                  num_outputs,
 167                                  kernel_size,
 168                                  depth_multiplier=depth_multiplier,
 169                                  stride=stride,
 170                                  rate=rate,
 171                                  padding=padding,
 172                                  scope=scope,
 173                                  **kwargs)
 174   def _split_separable_conv2d(padding):
 175     """Splits separable conv2d into depthwise and pointwise conv2d."""
 176     outputs = slim.separable_conv2d(inputs,
 177                                     None,
 178                                     kernel_size,
 179                                     depth_multiplier=depth_multiplier,
 180                                     stride=stride,
 181                                     rate=rate,
 182                                     padding=padding,
 183                                     scope=scope + '_depthwise',
 184                                     **kwargs)
 185     return slim.conv2d(outputs,
 186                        num_outputs,
 187                        1,
 188                        scope=scope + '_pointwise',
 189                        **kwargs)
 190   if stride == 1 or not use_explicit_padding:
 191     if regularize_depthwise:
 192       outputs = _separable_conv2d(padding='SAME')
 193     else:
 194       outputs = _split_separable_conv2d(padding='SAME')
 195   else:
 196     inputs = fixed_padding(inputs, kernel_size, rate)
 197     if regularize_depthwise:
 198       outputs = _separable_conv2d(padding='VALID')
 199     else:
 200       outputs = _split_separable_conv2d(padding='VALID')
 201   return outputs

 204 @slim.add_arg_scope
 205 def xception_module(inputs,
 206                     depth_list,
 207                     skip_connection_type,
 208                     stride,
 209                     kernel_size=3,
 210                     unit_rate_list=None,
 211                     rate=1,
 212                     activation_fn_in_separable_conv=False,
 213                     regularize_depthwise=False,
 214                     outputs_collections=None,
 215                     scope=None,
 216                     use_bounded_activation=False,
 217                     use_explicit_padding=True,
 218                     use_squeeze_excite=False,
 219                     se_pool_size=None):
 220   """An Xception module.
 221 
 222   The output of one Xception module is equal to the sum of `residual` and
 223   `shortcut`, where `residual` is the feature computed by three separable
 224   convolution. The `shortcut` is the feature computed by 1x1 convolution with
 225   or without striding. In some cases, the `shortcut` path could be a simple
 226   identity function or none (i.e, no shortcut).
 227 
 228   Note that we replace the max pooling operations in the Xception module with
 229   another separable convolution with striding, since atrous rate is not properly
 230   supported in current TensorFlow max pooling implementation.
 231 
 232   Args:
 233     inputs: A tensor of size [batch, height, width, channels].
 234     depth_list: A list of three integers specifying the depth values of one
 235       Xception module.
 236     skip_connection_type: Skip connection type for the residual path. Only
 237       supports 'conv', 'sum', or 'none'.
 238     stride: The block unit's stride. Determines the amount of downsampling of
 239       the units output compared to its input.
 240     kernel_size: Integer, convolution kernel size.
 241     unit_rate_list: A list of three integers, determining the unit rate for
 242       each separable convolution in the xception module.
 243     rate: An integer, rate for atrous convolution.
 244     activation_fn_in_separable_conv: Includes activation function in the
 245       separable convolution or not.
 246     regularize_depthwise: Whether or not apply L2-norm regularization on the
 247       depthwise convolution weights.
 248     outputs_collections: Collection to add the Xception unit output.
 249     scope: Optional variable_scope.
 250     use_bounded_activation: Whether or not to use bounded activations. Bounded
 251       activations better lend themselves to quantized inference.
 252     use_explicit_padding: If True, use explicit padding to make the model fully
 253       compatible with the open source version, otherwise use the native
 254       Tensorflow 'SAME' padding.
 255     use_squeeze_excite: Boolean, use squeeze-and-excitation or not.
 256     se_pool_size: None or integer specifying the pooling size used in SE module.
 257 
 258   Returns:
 259     The Xception module's output.
 260 
 261   Raises:
 262     ValueError: If depth_list and unit_rate_list do not contain three elements,
 263       or if stride != 1 for the third separable convolution operation in the
 264       residual path, or unsupported skip connection type.
 265   """
 266   if len(depth_list) != 3:
 267     raise ValueError('Expect three elements in depth_list.')
 268   if unit_rate_list:
 269     if len(unit_rate_list) != 3:
 270       raise ValueError('Expect three elements in unit_rate_list.')
 271 
 272   with tf.variable_scope(scope, 'xception_module', [inputs]) as sc:
 273     residual = inputs
 274 
 275     def _separable_conv(features, depth, kernel_size, depth_multiplier,
 276                         regularize_depthwise, rate, stride, scope):
 277       """Separable conv block."""
 278       if activation_fn_in_separable_conv:
 279         activation_fn = tf.nn.relu6 if use_bounded_activation else tf.nn.relu
 280       else:
 281         if use_bounded_activation:
 282           # When use_bounded_activation is True, we clip the feature values and
 283           # apply relu6 for activation.
 284           activation_fn = lambda x: tf.clip_by_value(x, -_CLIP_CAP, _CLIP_CAP)
 285           features = tf.nn.relu6(features)
 286         else:
 287           # Original network design.
 288           activation_fn = None
 289           features = tf.nn.relu(features)
 290       return separable_conv2d_same(features,
 291                                    depth,
 292                                    kernel_size,
 293                                    depth_multiplier=depth_multiplier,
 294                                    stride=stride,
 295                                    rate=rate,
 296                                    activation_fn=activation_fn,
 297                                    use_explicit_padding=use_explicit_padding,
 298                                    regularize_depthwise=regularize_depthwise,
 299                                    scope=scope)
 300     for i in range(3):
 301       residual = _separable_conv(residual,
 302                                  depth_list[i],
 303                                  kernel_size=kernel_size,
 304                                  depth_multiplier=1,
 305                                  regularize_depthwise=regularize_depthwise,
 306                                  rate=rate*unit_rate_list[i],
 307                                  stride=stride if i == 2 else 1,
 308                                  scope='separable_conv' + str(i+1))
 309     if use_squeeze_excite:
 310       residual = mobilenet_v3_ops.squeeze_excite(
 311           input_tensor=residual,
 312           squeeze_factor=16,
 313           inner_activation_fn=tf.nn.relu,
 314           gating_fn=lambda x: tf.nn.relu6(x+3)*0.16667,
 315           pool=se_pool_size)
 316 
 317     if skip_connection_type == 'conv':
 318       shortcut = slim.conv2d(inputs,
 319                              depth_list[-1],
 320                              [1, 1],
 321                              stride=stride,
 322                              activation_fn=None,
 323                              scope='shortcut')
 324       if use_bounded_activation:
 325         residual = tf.clip_by_value(residual, -_CLIP_CAP, _CLIP_CAP)
 326         shortcut = tf.clip_by_value(shortcut, -_CLIP_CAP, _CLIP_CAP)
 327       outputs = residual + shortcut
 328       if use_bounded_activation:
 329         outputs = tf.nn.relu6(outputs)
 330     elif skip_connection_type == 'sum':
 331       if use_bounded_activation:
 332         residual = tf.clip_by_value(residual, -_CLIP_CAP, _CLIP_CAP)
 333         inputs = tf.clip_by_value(inputs, -_CLIP_CAP, _CLIP_CAP)
 334       outputs = residual + inputs
 335       if use_bounded_activation:
 336         outputs = tf.nn.relu6(outputs)
 337     elif skip_connection_type == 'none':
 338       outputs = residual
 339     else:
 340       raise ValueError('Unsupported skip connection type.')
 341 
 342     return slim.utils.collect_named_outputs(outputs_collections,
 343                                             sc.name,
 344                                             outputs)

 347 @slim.add_arg_scope
 348 def stack_blocks_dense(net,
 349                        blocks,
 350                        output_stride=None,
 351                        outputs_collections=None):
 352   """Stacks Xception blocks and controls output feature density.
 353 
 354   First, this function creates scopes for the Xception in the form of
 355   'block_name/unit_1', 'block_name/unit_2', etc.
 356 
 357   Second, this function allows the user to explicitly control the output
 358   stride, which is the ratio of the input to output spatial resolution. This
 359   is useful for dense prediction tasks such as semantic segmentation or
 360   object detection.
 361 
 362   Control of the output feature density is implemented by atrous convolution.
 363 
 364   Args:
 365     net: A tensor of size [batch, height, width, channels].
 366     blocks: A list of length equal to the number of Xception blocks. Each
 367       element is an Xception Block object describing the units in the block.
 368     output_stride: If None, then the output will be computed at the nominal
 369       network stride. If output_stride is not None, it specifies the requested
 370       ratio of input to output spatial resolution, which needs to be equal to
 371       the product of unit strides from the start up to some level of Xception.
 372       For example, if the Xception employs units with strides 1, 2, 1, 3, 4, 1,
 373       then valid values for the output_stride are 1, 2, 6, 24 or None (which
 374       is equivalent to output_stride=24).
 375     outputs_collections: Collection to add the Xception block outputs.
 376 
 377   Returns:
 378     net: Output tensor with stride equal to the specified output_stride.
 379 
 380   Raises:
 381     ValueError: If the target output_stride is not valid.
 382   """
 383   # The current_stride variable keeps track of the effective stride of the
 384   # activations. This allows us to invoke atrous convolution whenever applying
 385   # the next residual unit would result in the activations having stride larger
 386   # than the target output_stride.
 387   current_stride = 1
 388 
 389   # The atrous convolution rate parameter.
 390   rate = 1
 391 
 392   for block in blocks:
 393     with tf.variable_scope(block.scope, 'block', [net]) as sc:
 394       for i, unit in enumerate(block.args):
 395         if output_stride is not None and current_stride > output_stride:
 396           raise ValueError('The target output_stride cannot be reached.')
 397         with tf.variable_scope('unit_%d' % (i + 1), values=[net]):
 398           # If we have reached the target output_stride, then we need to employ
 399           # atrous convolution with stride=1 and multiply the atrous rate by the
 400           # current unit's stride for use in subsequent layers.
 401           if output_stride is not None and current_stride == output_stride:
 402             net = block.unit_fn(net, rate=rate, **dict(unit, stride=1))
 403             rate *= unit.get('stride', 1)
 404           else:
 405             net = block.unit_fn(net, rate=1, **unit)
 406             current_stride *= unit.get('stride', 1)
 407 
 408       # Collect activations at the block's end before performing subsampling.
 409       net = slim.utils.collect_named_outputs(outputs_collections, sc.name, net)
 410 
 411   if output_stride is not None and current_stride != output_stride:
 412     raise ValueError('The target output_stride cannot be reached.')
 413 
 414   return net

 417 def xception(inputs,
 418              blocks,
 419              num_classes=None,
 420              is_training=True,
 421              global_pool=True,
 422              keep_prob=0.5,
 423              output_stride=None,
 424              reuse=None,
 425              scope=None,
 426              sync_batch_norm_method='None'):
 427   """Generator for Xception models.
 428 
 429   This function generates a family of Xception models. See the xception_*()
 430   methods for specific model instantiations, obtained by selecting different
 431   block instantiations that produce Xception of various depths.
 432 
 433   Args:
 434     inputs: A tensor of size [batch, height_in, width_in, channels]. Must be
 435       floating point. If a pretrained checkpoint is used, pixel values should be
 436       the same as during training (see go/slim-classification-models for
 437       specifics).
 438     blocks: A list of length equal to the number of Xception blocks. Each
 439       element is an Xception Block object describing the units in the block.
 440     num_classes: Number of predicted classes for classification tasks.
 441       If 0 or None, we return the features before the logit layer.
 442     is_training: whether batch_norm layers are in training mode.
 443     global_pool: If True, we perform global average pooling before computing the
 444       logits. Set to True for image classification, False for dense prediction.
 445     keep_prob: Keep probability used in the pre-logits dropout layer.
 446     output_stride: If None, then the output will be computed at the nominal
 447       network stride. If output_stride is not None, it specifies the requested
 448       ratio of input to output spatial resolution.
 449     reuse: whether or not the network and its variables should be reused. To be
 450       able to reuse 'scope' must be given.
 451     scope: Optional variable_scope.
 452     sync_batch_norm_method: String, sync batchnorm method. Currently only
 453       support `None`.
 454 
 455   Returns:
 456     net: A rank-4 tensor of size [batch, height_out, width_out, channels_out].
 457       If global_pool is False, then height_out and width_out are reduced by a
 458       factor of output_stride compared to the respective height_in and width_in,
 459       else both height_out and width_out equal one. If num_classes is 0 or None,
 460       then net is the output of the last Xception block, potentially after
 461       global average pooling. If num_classes is a non-zero integer, net contains
 462       the pre-softmax activations.
 463     end_points: A dictionary from components of the network to the corresponding
 464       activation.
 465 
 466   Raises:
 467     ValueError: If the target output_stride is not valid.
 468   """
 469   with tf.variable_scope(
 470       scope, 'xception', [inputs], reuse=reuse) as sc:
 471     end_points_collection = sc.original_name_scope + 'end_points'
 472     batch_norm = utils.get_batch_norm_fn(sync_batch_norm_method)
 473     with slim.arg_scope([slim.conv2d,
 474                          slim.separable_conv2d,
 475                          xception_module,
 476                          stack_blocks_dense],
 477                         outputs_collections=end_points_collection):
 478       with slim.arg_scope([batch_norm], is_training=is_training):
 479         net = inputs
 480         if output_stride is not None:
 481           if output_stride % 2 != 0:
 482             raise ValueError('The output_stride needs to be a multiple of 2.')
 483           output_stride //= 2
 484         # Root block function operated on inputs.
 485         net = resnet_utils.conv2d_same(net, 32, 3, stride=2,
 486                                        scope='entry_flow/conv1_1')
 487         net = resnet_utils.conv2d_same(net, 64, 3, stride=1,
 488                                        scope='entry_flow/conv1_2')
 489 
 490         # Extract features for entry_flow, middle_flow, and exit_flow.
 491         net = stack_blocks_dense(net, blocks, output_stride)
 492 
 493         # Convert end_points_collection into a dictionary of end_points.
 494         end_points = slim.utils.convert_collection_to_dict(
 495             end_points_collection, clear_collection=True)
 496 
 497         if global_pool:
 498           # Global average pooling.
 499           net = tf.reduce_mean(net, [1, 2], name='global_pool', keepdims=True)
 500           end_points['global_pool'] = net
 501         if num_classes:
 502           net = slim.dropout(net, keep_prob=keep_prob, is_training=is_training,
 503                              scope='prelogits_dropout')
 504           net = slim.conv2d(net, num_classes, [1, 1], activation_fn=None,
 505                             normalizer_fn=None, scope='logits')
 506           end_points[sc.name + '/logits'] = net
 507           end_points['predictions'] = slim.softmax(net, scope='predictions')
 508         return net, end_points

 511 def xception_block(scope,
 512                    depth_list,
 513                    skip_connection_type,
 514                    activation_fn_in_separable_conv,
 515                    regularize_depthwise,
 516                    num_units,
 517                    stride,
 518                    kernel_size=3,
 519                    unit_rate_list=None,
 520                    use_squeeze_excite=False,
 521                    se_pool_size=None):
 522   """Helper function for creating a Xception block.
 523 
 524   Args:
 525     scope: The scope of the block.
 526     depth_list: The depth of the bottleneck layer for each unit.
 527     skip_connection_type: Skip connection type for the residual path. Only
 528       supports 'conv', 'sum', or 'none'.
 529     activation_fn_in_separable_conv: Includes activation function in the
 530       separable convolution or not.
 531     regularize_depthwise: Whether or not apply L2-norm regularization on the
 532       depthwise convolution weights.
 533     num_units: The number of units in the block.
 534     stride: The stride of the block, implemented as a stride in the last unit.
 535       All other units have stride=1.
 536     kernel_size: Integer, convolution kernel size.
 537     unit_rate_list: A list of three integers, determining the unit rate in the
 538       corresponding xception block.
 539     use_squeeze_excite: Boolean, use squeeze-and-excitation or not.
 540     se_pool_size: None or integer specifying the pooling size used in SE module.
 541 
 542   Returns:
 543     An Xception block.
 544   """
 545   if unit_rate_list is None:
 546     unit_rate_list = _DEFAULT_MULTI_GRID
 547   return Block(scope, xception_module, [{
 548       'depth_list': depth_list,
 549       'skip_connection_type': skip_connection_type,
 550       'activation_fn_in_separable_conv': activation_fn_in_separable_conv,
 551       'regularize_depthwise': regularize_depthwise,
 552       'stride': stride,
 553       'kernel_size': kernel_size,
 554       'unit_rate_list': unit_rate_list,
 555       'use_squeeze_excite': use_squeeze_excite,
 556       'se_pool_size': se_pool_size,
 557   }] * num_units)

 560 def xception_41(inputs,
 561                 num_classes=None,
 562                 is_training=True,
 563                 global_pool=True,
 564                 keep_prob=0.5,
 565                 output_stride=None,
 566                 regularize_depthwise=False,
 567                 multi_grid=None,
 568                 reuse=None,
 569                 scope='xception_41',
 570                 sync_batch_norm_method='None'):
 571   """Xception-41 model."""
 572   blocks = [
 573       xception_block('entry_flow/block1',
 574                      depth_list=[128, 128, 128],
 575                      skip_connection_type='conv',
 576                      activation_fn_in_separable_conv=False,
 577                      regularize_depthwise=regularize_depthwise,
 578                      num_units=1,
 579                      stride=2),
 580       xception_block('entry_flow/block2',
 581                      depth_list=[256, 256, 256],
 582                      skip_connection_type='conv',
 583                      activation_fn_in_separable_conv=False,
 584                      regularize_depthwise=regularize_depthwise,
 585                      num_units=1,
 586                      stride=2),
 587       xception_block('entry_flow/block3',
 588                      depth_list=[728, 728, 728],
 589                      skip_connection_type='conv',
 590                      activation_fn_in_separable_conv=False,
 591                      regularize_depthwise=regularize_depthwise,
 592                      num_units=1,
 593                      stride=2),
 594       xception_block('middle_flow/block1',

 885 def xception_arg_scope(weight_decay=0.00004,
 886                        batch_norm_decay=0.9997,
 887                        batch_norm_epsilon=0.001,
 888                        batch_norm_scale=True,
 889                        weights_initializer_stddev=0.09,
 890                        regularize_depthwise=False,
 891                        use_batch_norm=True,
 892                        use_bounded_activation=False,
 893                        sync_batch_norm_method='None'):
 894   """Defines the default Xception arg scope.
 895 
 896   Args:
 897     weight_decay: The weight decay to use for regularizing the model.
 898     batch_norm_decay: The moving average decay when estimating layer activation
 899       statistics in batch normalization.
 900     batch_norm_epsilon: Small constant to prevent division by zero when
 901       normalizing activations by their variance in batch normalization.
 902     batch_norm_scale: If True, uses an explicit `gamma` multiplier to scale the
 903       activations in the batch normalization layer.
 904     weights_initializer_stddev: The standard deviation of the trunctated normal
 905       weight initializer.
 906     regularize_depthwise: Whether or not apply L2-norm regularization on the
 907       depthwise convolution weights.
 908     use_batch_norm: Whether or not to use batch normalization.
 909     use_bounded_activation: Whether or not to use bounded activations. Bounded
 910       activations better lend themselves to quantized inference.
 911     sync_batch_norm_method: String, sync batchnorm method. Currently only
 912       support `None`. Also, it is only effective for Xception.
 913 
 914   Returns:
 915     An `arg_scope` to use for the Xception models.
 916   """
 917   batch_norm_params = {
 918       'decay': batch_norm_decay,
 919       'epsilon': batch_norm_epsilon,
 920       'scale': batch_norm_scale,
 921   }
 922   if regularize_depthwise:
 923     depthwise_regularizer = slim.l2_regularizer(weight_decay)
 924   else:
 925     depthwise_regularizer = None
 926   activation_fn = tf.nn.relu6 if use_bounded_activation else tf.nn.relu
 927   batch_norm = utils.get_batch_norm_fn(sync_batch_norm_method)
 928   with slim.arg_scope(
 929       [slim.conv2d, slim.separable_conv2d],
 930       weights_initializer=tf.truncated_normal_initializer(
 931           stddev=weights_initializer_stddev),
 932       activation_fn=activation_fn,
 933       normalizer_fn=batch_norm if use_batch_norm else None):
 934     with slim.arg_scope([batch_norm], **batch_norm_params):
 935       with slim.arg_scope(
 936           [slim.conv2d],
 937           weights_regularizer=slim.l2_regularizer(weight_decay)):
 938         with slim.arg_scope(
 939             [slim.separable_conv2d],
 940             weights_regularizer=depthwise_regularizer):
 941           with slim.arg_scope(
 942               [xception_module],
 943               use_bounded_activation=use_bounded_activation,
 944               use_explicit_padding=not use_bounded_activation) as arg_sc:
 945             return arg_sc
