  40 class Conv2D(tf.keras.layers.Conv2D, tf.layers.Layer):
  41   """2D convolution layer (e.g. spatial convolution over images).
  42 
  43   This layer creates a convolution kernel that is convolved
  44   (actually cross-correlated) with the layer input to produce a tensor of
  45   outputs. If `use_bias` is True (and a `bias_initializer` is provided),
  46   a bias vector is created and added to the outputs. Finally, if
  47   `activation` is not `None`, it is applied to the outputs as well.
  48   """
  49 
  50   def __init__(self,
  51                filters,
  52                kernel_size,
  53                strides=(1, 1),
  54                padding='valid',
  55                data_format='channels_last',
  56                dilation_rate=(1, 1),
  57                activation=None,
  58                use_bias=True,
  59                kernel_initializer=None,
  60                bias_initializer=tf.zeros_initializer(),
  61                kernel_regularizer=None,
  62                bias_regularizer=None,
  63                use_weight_standardization=False,
  64                activity_regularizer=None,
  65                kernel_constraint=None,
  66                bias_constraint=None,
  67                trainable=True,
  68                name=None,
  69                **kwargs):
  70     """Constructs the 2D convolution layer.
  71 
  72     Args:
  73       filters: Integer, the dimensionality of the output space (i.e. the number
  74         of filters in the convolution).
  75       kernel_size: An integer or tuple/list of 2 integers, specifying the height
  76         and width of the 2D convolution window. Can be a single integer to
  77         specify the same value for all spatial dimensions.
  78       strides: An integer or tuple/list of 2 integers, specifying the strides of
  79         the convolution along the height and width. Can be a single integer to
  80         specify the same value for all spatial dimensions. Specifying any stride
  81         value != 1 is incompatible with specifying any `dilation_rate` value !=
  82         1.
  83       padding: One of `"valid"` or `"same"` (case-insensitive).
  84       data_format: A string, one of `channels_last` (default) or
  85         `channels_first`. The ordering of the dimensions in the inputs.
  86         `channels_last` corresponds to inputs with shape `(batch, height, width,
  87         channels)` while `channels_first` corresponds to inputs with shape
  88         `(batch, channels, height, width)`.
  89       dilation_rate: An integer or tuple/list of 2 integers, specifying the
  90         dilation rate to use for dilated convolution. Can be a single integer to
  91         specify the same value for all spatial dimensions. Currently, specifying
  92         any `dilation_rate` value != 1 is incompatible with specifying any
  93         stride value != 1.
  94       activation: Activation function. Set it to None to maintain a linear
  95         activation.
  96       use_bias: Boolean, whether the layer uses a bias.
  97       kernel_initializer: An initializer for the convolution kernel.
  98       bias_initializer: An initializer for the bias vector. If None, the default
  99         initializer will be used.
 100       kernel_regularizer: Optional regularizer for the convolution kernel.
 101       bias_regularizer: Optional regularizer for the bias vector.
 102       use_weight_standardization: Boolean, whether the layer uses weight
 103         standardization.
 104       activity_regularizer: Optional regularizer function for the output.
 105       kernel_constraint: Optional projection function to be applied to the
 106         kernel after being updated by an `Optimizer` (e.g. used to implement
 107         norm constraints or value constraints for layer weights). The function
 108         must take as input the unprojected variable and must return the
 109         projected variable (which must have the same shape). Constraints are not
 110         safe to use when doing asynchronous distributed training.
 111       bias_constraint: Optional projection function to be applied to the bias
 112         after being updated by an `Optimizer`.
 113       trainable: Boolean, if `True` also add variables to the graph collection
 114         `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 115       name: A string, the name of the layer.
 116       **kwargs: Arbitrary keyword arguments passed to tf.keras.layers.Conv2D
 117     """
 118 
 119     super(Conv2D, self).__init__(
 120         filters=filters,
 121         kernel_size=kernel_size,
 122         strides=strides,
 123         padding=padding,
 124         data_format=data_format,
 125         dilation_rate=dilation_rate,
 126         activation=activation,
 127         use_bias=use_bias,
 128         kernel_initializer=kernel_initializer,
 129         bias_initializer=bias_initializer,
 130         kernel_regularizer=kernel_regularizer,
 131         bias_regularizer=bias_regularizer,
 132         activity_regularizer=activity_regularizer,
 133         kernel_constraint=kernel_constraint,
 134         bias_constraint=bias_constraint,
 135         trainable=trainable,
 136         name=name,
 137         **kwargs)
 138     self.use_weight_standardization = use_weight_standardization
 139 
 140   def call(self, inputs):
 141     if self.use_weight_standardization:
 142       mean, var = tf.nn.moments(self.kernel, [0, 1, 2], keep_dims=True)
 143       kernel = (self.kernel - mean) / tf.sqrt(var + 1e-5)
 144       outputs = self._convolution_op(inputs, kernel)
 145     else:
 146       outputs = self._convolution_op(inputs, self.kernel)
 147 
 148     if self.use_bias:
 149       if self.data_format == 'channels_first':
 150         if self.rank == 1:
 151           # tf.nn.bias_add does not accept a 1D input tensor.
 152           bias = tf.reshape(self.bias, (1, self.filters, 1))
 153           outputs += bias
 154         else:
 155           outputs = tf.nn.bias_add(outputs, self.bias, data_format='NCHW')
 156       else:
 157         outputs = tf.nn.bias_add(outputs, self.bias, data_format='NHWC')
 158 
 159     if self.activation is not None:
 160       return self.activation(outputs)
 161     return outputs

 164 @contrib_framework.add_arg_scope
 165 def conv2d(inputs,
 166            num_outputs,
 167            kernel_size,
 168            stride=1,
 169            padding='SAME',
 170            data_format=None,
 171            rate=1,
 172            activation_fn=tf.nn.relu,
 173            normalizer_fn=None,
 174            normalizer_params=None,
 175            weights_initializer=contrib_layers.xavier_initializer(),
 176            weights_regularizer=None,
 177            biases_initializer=tf.zeros_initializer(),
 178            biases_regularizer=None,
 179            use_weight_standardization=False,
 180            reuse=None,
 181            variables_collections=None,
 182            outputs_collections=None,
 183            trainable=True,
 184            scope=None):
 185   """Adds a 2D convolution followed by an optional batch_norm layer.
 186 
 187   `convolution` creates a variable called `weights`, representing the
 188   convolutional kernel, that is convolved (actually cross-correlated) with the
 189   `inputs` to produce a `Tensor` of activations. If a `normalizer_fn` is
 190   provided (such as `batch_norm`), it is then applied. Otherwise, if
 191   `normalizer_fn` is None and a `biases_initializer` is provided then a `biases`
 192   variable would be created and added the activations. Finally, if
 193   `activation_fn` is not `None`, it is applied to the activations as well.
 194 
 195   Performs atrous convolution with input stride/dilation rate equal to `rate`
 196   if a value > 1 for any dimension of `rate` is specified.  In this case
 197   `stride` values != 1 are not supported.
 198 
 199   Args:
 200     inputs: A Tensor of rank N+2 of shape `[batch_size] + input_spatial_shape +
 201       [in_channels]` if data_format does not start with "NC" (default), or
 202       `[batch_size, in_channels] + input_spatial_shape` if data_format starts
 203       with "NC".
 204     num_outputs: Integer, the number of output filters.
 205     kernel_size: A sequence of N positive integers specifying the spatial
 206       dimensions of the filters.  Can be a single integer to specify the same
 207       value for all spatial dimensions.
 208     stride: A sequence of N positive integers specifying the stride at which to
 209       compute output.  Can be a single integer to specify the same value for all
 210       spatial dimensions.  Specifying any `stride` value != 1 is incompatible
 211       with specifying any `rate` value != 1.
 212     padding: One of `"VALID"` or `"SAME"`.
 213     data_format: A string or None.  Specifies whether the channel dimension of
 214       the `input` and output is the last dimension (default, or if `data_format`
 215       does not start with "NC"), or the second dimension (if `data_format`
 216       starts with "NC").  For N=1, the valid values are "NWC" (default) and
 217       "NCW".  For N=2, the valid values are "NHWC" (default) and "NCHW". For
 218       N=3, the valid values are "NDHWC" (default) and "NCDHW".
 219     rate: A sequence of N positive integers specifying the dilation rate to use
 220       for atrous convolution.  Can be a single integer to specify the same value
 221       for all spatial dimensions.  Specifying any `rate` value != 1 is
 222       incompatible with specifying any `stride` value != 1.
 223     activation_fn: Activation function. The default value is a ReLU function.
 224       Explicitly set it to None to skip it and maintain a linear activation.
 225     normalizer_fn: Normalization function to use instead of `biases`. If
 226       `normalizer_fn` is provided then `biases_initializer` and
 227       `biases_regularizer` are ignored and `biases` are not created nor added.
 228       default set to None for no normalizer function
 229     normalizer_params: Normalization function parameters.
 230     weights_initializer: An initializer for the weights.
 231     weights_regularizer: Optional regularizer for the weights.
 232     biases_initializer: An initializer for the biases. If None skip biases.
 233     biases_regularizer: Optional regularizer for the biases.
 234     use_weight_standardization: Boolean, whether the layer uses weight
 235       standardization.
 236     reuse: Whether or not the layer and its variables should be reused. To be
 237       able to reuse the layer scope must be given.
 238     variables_collections: Optional list of collections for all the variables or
 239       a dictionary containing a different list of collection per variable.
 240     outputs_collections: Collection to add the outputs.
 241     trainable: If `True` also add variables to the graph collection
 242       `GraphKeys.TRAINABLE_VARIABLES` (see tf.Variable).
 243     scope: Optional scope for `variable_scope`.
 244 
 245   Returns:
 246     A tensor representing the output of the operation.
 247 
 248   Raises:
 249     ValueError: If `data_format` is invalid.
 250     ValueError: Both 'rate' and `stride` are not uniformly 1.
 251   """
 252   if data_format not in [None, 'NWC', 'NCW', 'NHWC', 'NCHW', 'NDHWC', 'NCDHW']:
 253     raise ValueError('Invalid data_format: %r' % (data_format,))
 254 
 255   # pylint: disable=protected-access
 256   layer_variable_getter = layers._build_variable_getter({
 257       'bias': 'biases',
 258       'kernel': 'weights'
 259   })
 260   # pylint: enable=protected-access
 261   with tf.variable_scope(
 262       scope, 'Conv', [inputs], reuse=reuse,
 263       custom_getter=layer_variable_getter) as sc:
 264     inputs = tf.convert_to_tensor(inputs)
 265     input_rank = inputs.get_shape().ndims
 266 
 267     if input_rank != 4:
 268       raise ValueError('Convolution expects input with rank %d, got %d' %
 269                        (4, input_rank))
 270 
 271     data_format = ('channels_first' if data_format and
 272                    data_format.startswith('NC') else 'channels_last')
 273     layer = Conv2D(
 274         filters=num_outputs,
 275         kernel_size=kernel_size,
 276         strides=stride,
 277         padding=padding,
 278         data_format=data_format,
 279         dilation_rate=rate,
 280         activation=None,
 281         use_bias=not normalizer_fn and biases_initializer,
 282         kernel_initializer=weights_initializer,
 283         bias_initializer=biases_initializer,
 284         kernel_regularizer=weights_regularizer,
 285         bias_regularizer=biases_regularizer,
 286         use_weight_standardization=use_weight_standardization,
 287         activity_regularizer=None,
 288         trainable=trainable,
 289         name=sc.name,
 290         dtype=inputs.dtype.base_dtype,
 291         _scope=sc,
 292         _reuse=reuse)
 293     outputs = layer.apply(inputs)
 294 
 295     # Add variables to collections.
 296     # pylint: disable=protected-access
 297     layers._add_variable_to_collections(layer.kernel, variables_collections,
 298                                         'weights')
 299     if layer.use_bias:
 300       layers._add_variable_to_collections(layer.bias, variables_collections,
 301                                           'biases')
 302     # pylint: enable=protected-access
 303     if normalizer_fn is not None:
 304       normalizer_params = normalizer_params or {}
 305       outputs = normalizer_fn(outputs, **normalizer_params)
 306 
 307     if activation_fn is not None:
 308       outputs = activation_fn(outputs)
 309     return utils.collect_named_outputs(outputs_collections, sc.name, outputs)

 312 def conv2d_same(inputs, num_outputs, kernel_size, stride, rate=1, scope=None):
 313   """Strided 2-D convolution with 'SAME' padding.
 314 
 315   When stride > 1, then we do explicit zero-padding, followed by conv2d with
 316   'VALID' padding.
 317 
 318   Note that
 319 
 320      net = conv2d_same(inputs, num_outputs, 3, stride=stride)
 321 
 322   is equivalent to
 323 
 324      net = conv2d(inputs, num_outputs, 3, stride=1, padding='SAME')
 325      net = subsample(net, factor=stride)
 326 
 327   whereas
 328 
 329      net = conv2d(inputs, num_outputs, 3, stride=stride, padding='SAME')
 330 
 331   is different when the input's height or width is even, which is why we add the
 332   current function. For more details, see ResnetUtilsTest.testConv2DSameEven().
 333 
 334   Args:
 335     inputs: A 4-D tensor of size [batch, height_in, width_in, channels].
 336     num_outputs: An integer, the number of output filters.
 337     kernel_size: An int with the kernel_size of the filters.
 338     stride: An integer, the output stride.
 339     rate: An integer, rate for atrous convolution.
 340     scope: Scope.
 341 
 342   Returns:
 343     output: A 4-D tensor of size [batch, height_out, width_out, channels] with
 344       the convolution output.
 345   """
 346   if stride == 1:
 347     return conv2d(
 348         inputs,
 349         num_outputs,
 350         kernel_size,
 351         stride=1,
 352         rate=rate,
 353         padding='SAME',
 354         scope=scope)
 355   else:
 356     kernel_size_effective = kernel_size + (kernel_size - 1) * (rate - 1)
 357     pad_total = kernel_size_effective - 1
 358     pad_beg = pad_total // 2
 359     pad_end = pad_total - pad_beg
 360     inputs = tf.pad(inputs,
 361                     [[0, 0], [pad_beg, pad_end], [pad_beg, pad_end], [0, 0]])
 362     return conv2d(
 363         inputs,
 364         num_outputs,
 365         kernel_size,
 366         stride=stride,
 367         rate=rate,
 368         padding='VALID',
 369         scope=scope)
