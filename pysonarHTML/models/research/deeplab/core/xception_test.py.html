  29 def create_test_input(batch, height, width, channels):
  30   """Create test input tensor."""
  31   if None in [batch, height, width, channels]:
  32     return tf.placeholder(tf.float32, (batch, height, width, channels))
  33   else:
  34     return tf.cast(
  35         np.tile(
  36             np.reshape(
  37                 np.reshape(np.arange(height), [height, 1]) +
  38                 np.reshape(np.arange(width), [1, width]),
  39                 [1, height, width, 1]),
  40             [batch, 1, 1, channels]),
  41         tf.float32)

  46   def testSeparableConv2DSameWithInputEvenSize(self):
  47     n, n2 = 4, 2
  48 
  49     # Input image.
  50     x = create_test_input(1, n, n, 1)
  51 
  52     # Convolution kernel.
  53     dw = create_test_input(1, 3, 3, 1)
  54     dw = tf.reshape(dw, [3, 3, 1, 1])
  55 
  56     tf.get_variable('Conv/depthwise_weights', initializer=dw)
  57     tf.get_variable('Conv/pointwise_weights',
  58                     initializer=tf.ones([1, 1, 1, 1]))
  59     tf.get_variable('Conv/biases', initializer=tf.zeros([1]))
  60     tf.get_variable_scope().reuse_variables()
  61 
  62     y1 = slim.separable_conv2d(x, 1, [3, 3], depth_multiplier=1,
  63                                stride=1, scope='Conv')
  64     y1_expected = tf.cast([[14, 28, 43, 26],
  65                            [28, 48, 66, 37],
  66                            [43, 66, 84, 46],
  67                            [26, 37, 46, 22]], tf.float32)
  68     y1_expected = tf.reshape(y1_expected, [1, n, n, 1])
  69 
  70     y2 = resnet_utils.subsample(y1, 2)
  71     y2_expected = tf.cast([[14, 43],
  72                            [43, 84]], tf.float32)
  73     y2_expected = tf.reshape(y2_expected, [1, n2, n2, 1])
  74 
  75     y3 = xception.separable_conv2d_same(x, 1, 3, depth_multiplier=1,
  76                                         regularize_depthwise=True,
  77                                         stride=2, scope='Conv')
  78     y3_expected = y2_expected
  79 
  80     y4 = slim.separable_conv2d(x, 1, [3, 3], depth_multiplier=1,
  81                                stride=2, scope='Conv')
  82     y4_expected = tf.cast([[48, 37],
  83                            [37, 22]], tf.float32)
  84     y4_expected = tf.reshape(y4_expected, [1, n2, n2, 1])
  85 
  86     with self.test_session() as sess:
  87       sess.run(tf.global_variables_initializer())
  88       self.assertAllClose(y1.eval(), y1_expected.eval())
  89       self.assertAllClose(y2.eval(), y2_expected.eval())
  90       self.assertAllClose(y3.eval(), y3_expected.eval())
  91       self.assertAllClose(y4.eval(), y4_expected.eval())

 144   def _xception_small(self,
 145                       inputs,
 146                       num_classes=None,
 147                       is_training=True,
 148                       global_pool=True,
 149                       output_stride=None,
 150                       regularize_depthwise=True,
 151                       reuse=None,
 152                       scope='xception_small'):
 153     """A shallow and thin Xception for faster tests."""
 154     block = xception.xception_block
 155     blocks = [
 156         block('entry_flow/block1',
 157               depth_list=[1, 1, 1],
 158               skip_connection_type='conv',
 159               activation_fn_in_separable_conv=False,
 160               regularize_depthwise=regularize_depthwise,
 161               num_units=1,
 162               stride=2),
 163         block('entry_flow/block2',
 164               depth_list=[2, 2, 2],
 165               skip_connection_type='conv',
 166               activation_fn_in_separable_conv=False,
 167               regularize_depthwise=regularize_depthwise,
 168               num_units=1,
 169               stride=2),
 170         block('entry_flow/block3',
 171               depth_list=[4, 4, 4],
 172               skip_connection_type='conv',
 173               activation_fn_in_separable_conv=False,
 174               regularize_depthwise=regularize_depthwise,
 175               num_units=1,
 176               stride=1),
 177         block('entry_flow/block4',
 178               depth_list=[4, 4, 4],
 179               skip_connection_type='conv',
 180               activation_fn_in_separable_conv=False,
 181               regularize_depthwise=regularize_depthwise,
 182               num_units=1,
 183               stride=2),
 184         block('middle_flow/block1',
 185               depth_list=[4, 4, 4],
 186               skip_connection_type='sum',
 187               activation_fn_in_separable_conv=False,
 188               regularize_depthwise=regularize_depthwise,
 189               num_units=2,
 190               stride=1),
 191         block('exit_flow/block1',
 192               depth_list=[8, 8, 8],
 193               skip_connection_type='conv',
 194               activation_fn_in_separable_conv=False,
 195               regularize_depthwise=regularize_depthwise,
 196               num_units=1,
 197               stride=2),
 198         block('exit_flow/block2',
 199               depth_list=[16, 16, 16],
 200               skip_connection_type='none',
 201               activation_fn_in_separable_conv=True,
 202               regularize_depthwise=regularize_depthwise,
 203               num_units=1,
 204               stride=1),
 205     ]
 206     return xception.xception(inputs,
 207                              blocks=blocks,
 208                              num_classes=num_classes,
 209                              is_training=is_training,
 210                              global_pool=global_pool,
 211                              output_stride=output_stride,
 212                              reuse=reuse,
 213                              scope=scope)

 362   def testAtrousFullyConvolutionalValues(self):
 363     """Verify dense feature extraction with atrous convolution."""
 364     nominal_stride = 32
 365     for output_stride in [4, 8, 16, 32, None]:
 366       with slim.arg_scope(xception.xception_arg_scope()):
 367         with tf.Graph().as_default():
 368           with self.test_session() as sess:
 369             tf.set_random_seed(0)
 370             inputs = create_test_input(2, 96, 97, 3)
 371             # Dense feature extraction followed by subsampling.
 372             output, _ = self._xception_small(
 373                 inputs,
 374                 None,
 375                 is_training=False,
 376                 global_pool=False,
 377                 output_stride=output_stride)
 378             if output_stride is None:
 379               factor = 1
 380             else:
 381               factor = nominal_stride // output_stride
 382             output = resnet_utils.subsample(output, factor)
 383             # Make the two networks use the same weights.
 384             tf.get_variable_scope().reuse_variables()
 385             # Feature extraction at the nominal network rate.
 386             expected, _ = self._xception_small(
 387                 inputs,
 388                 None,
 389                 is_training=False,
 390                 global_pool=False)
 391             sess.run(tf.global_variables_initializer())
 392             self.assertAllClose(output.eval(), expected.eval(),
 393                                 atol=1e-5, rtol=1e-5)

 468   def testUseBoundedAcitvation(self):
 469     global_pool = False
 470     num_classes = 3
 471     output_stride = 16
 472     for use_bounded_activation in (True, False):
 473       tf.reset_default_graph()
 474       inputs = create_test_input(2, 65, 65, 3)
 475       with slim.arg_scope(xception.xception_arg_scope(
 476           use_bounded_activation=use_bounded_activation)):
 477         _, _ = self._xception_small(
 478             inputs,
 479             num_classes,
 480             global_pool=global_pool,
 481             output_stride=output_stride,
 482             scope='xception')

 484           if node.op.startswith('Relu'):
 485             self.assertEqual(node.op == 'Relu6', use_bounded_activation)
