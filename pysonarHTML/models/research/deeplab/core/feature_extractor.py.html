  43 def _mobilenet_v2(net,
  44                   depth_multiplier,
  45                   output_stride,
  46                   conv_defs=None,
  47                   divisible_by=None,
  48                   reuse=None,
  49                   scope=None,
  50                   final_endpoint=None):
  51   """Auxiliary function to add support for 'reuse' to mobilenet_v2.
  52 
  53   Args:
  54     net: Input tensor of shape [batch_size, height, width, channels].
  55     depth_multiplier: Float multiplier for the depth (number of channels)
  56       for all convolution ops. The value must be greater than zero. Typical
  57       usage will be to set this value in (0, 1) to reduce the number of
  58       parameters or computation cost of the model.
  59     output_stride: An integer that specifies the requested ratio of input to
  60       output spatial resolution. If not None, then we invoke atrous convolution
  61       if necessary to prevent the network from reducing the spatial resolution
  62       of the activation maps. Allowed values are 8 (accurate fully convolutional
  63       mode), 16 (fast fully convolutional mode), 32 (classification mode).
  64     conv_defs: MobileNet con def.
  65     divisible_by: None (use default setting) or an integer that ensures all
  66       layers # channels will be divisible by this number. Used in MobileNet.
  67     reuse: Reuse model variables.
  68     scope: Optional variable scope.
  69     final_endpoint: The endpoint to construct the network up to.
  70 
  71   Returns:
  72     Features extracted by MobileNetv2.
  73   """
  74   if divisible_by is None:
  75     divisible_by = 8 if depth_multiplier == 1.0 else 1
  76   if conv_defs is None:
  77     conv_defs = mobilenet_v2.V2_DEF
  78   with tf.variable_scope(
  79       scope, 'MobilenetV2', [net], reuse=reuse) as scope:
  80     return mobilenet_v2.mobilenet_base(
  81         net,
  82         conv_defs=conv_defs,
  83         depth_multiplier=depth_multiplier,
  84         min_depth=8 if depth_multiplier == 1.0 else 1,
  85         divisible_by=divisible_by,
  86         final_endpoint=final_endpoint or _MOBILENET_V2_FINAL_ENDPOINT,
  87         output_stride=output_stride,
  88         scope=scope)

  91 def _mobilenet_v3(net,
  92                   depth_multiplier,
  93                   output_stride,
  94                   conv_defs=None,
  95                   divisible_by=None,
  96                   reuse=None,
  97                   scope=None,
  98                   final_endpoint=None):
  99   """Auxiliary function to build mobilenet v3.
 100 
 101   Args:
 102     net: Input tensor of shape [batch_size, height, width, channels].
 103     depth_multiplier: Float multiplier for the depth (number of channels)
 104       for all convolution ops. The value must be greater than zero. Typical
 105       usage will be to set this value in (0, 1) to reduce the number of
 106       parameters or computation cost of the model.
 107     output_stride: An integer that specifies the requested ratio of input to
 108       output spatial resolution. If not None, then we invoke atrous convolution
 109       if necessary to prevent the network from reducing the spatial resolution
 110       of the activation maps. Allowed values are 8 (accurate fully convolutional
 111       mode), 16 (fast fully convolutional mode), 32 (classification mode).
 112     conv_defs: A list of ConvDef namedtuples specifying the net architecture.
 113     divisible_by: None (use default setting) or an integer that ensures all
 114       layers # channels will be divisible by this number. Used in MobileNet.
 115     reuse: Reuse model variables.
 116     scope: Optional variable scope.
 117     final_endpoint: The endpoint to construct the network up to.
 118 
 119   Returns:
 120     net: The output tensor.
 121     end_points: A set of activations for external use.
 122 
 123   Raises:
 124     ValueError: If conv_defs or final_endpoint is not specified.
 125   """
 126   del divisible_by
 127   with tf.variable_scope(
 128       scope, 'MobilenetV3', [net], reuse=reuse) as scope:
 129     if conv_defs is None:
 130       raise ValueError('conv_defs must be specified for mobilenet v3.')
 131     if final_endpoint is None:
 132       raise ValueError('Final endpoint must be specified for mobilenet v3.')
 133     net, end_points = mobilenet_v3.mobilenet_base(
 134         net,
 135         depth_multiplier=depth_multiplier,
 136         conv_defs=conv_defs,
 137         output_stride=output_stride,
 138         final_endpoint=final_endpoint,
 139         scope=scope)
 140 
 141     return net, end_points

 144 def mobilenet_v3_large_seg(net,
 145                            depth_multiplier,
 146                            output_stride,
 147                            divisible_by=None,
 148                            reuse=None,
 149                            scope=None,
 150                            final_endpoint=None):
 151   """Final mobilenet v3 large model for segmentation task."""
 152   del divisible_by
 153   del final_endpoint
 154   conv_defs = copy.deepcopy(mobilenet_v3.V3_LARGE)
 155 
 156   # Reduce the filters by a factor of 2 in the last block.
 157   for layer, expansion in [(13, 336), (14, 480), (15, 480), (16, None)]:
 158     conv_defs['spec'][layer].params['num_outputs'] /= 2
 159     # Update expansion size
 160     if expansion is not None:
 161       factor = expansion / conv_defs['spec'][layer - 1].params['num_outputs']
 162       conv_defs['spec'][layer].params[
 163           'expansion_size'] = mobilenet_v3.expand_input(factor)
 164 
 165   return _mobilenet_v3(
 166       net,
 167       depth_multiplier=depth_multiplier,
 168       output_stride=output_stride,
 169       divisible_by=8,
 170       conv_defs=conv_defs,
 171       reuse=reuse,
 172       scope=scope,
 173       final_endpoint=_MOBILENET_V3_LARGE_FINAL_ENDPOINT)

 251 def mobilenet_v2_arg_scope(is_training=True,
 252                            weight_decay=0.00004,
 253                            stddev=0.09,
 254                            activation=tf.nn.relu6,
 255                            bn_decay=0.997,
 256                            bn_epsilon=None,
 257                            bn_renorm=None):
 258   """Defines the default MobilenetV2 arg scope.
 259 
 260   Args:
 261     is_training: Whether or not we're training the model. If this is set to None
 262       is_training parameter in batch_norm is not set. Please note that this also
 263       sets the is_training parameter in dropout to None.
 264     weight_decay: The weight decay to use for regularizing the model.
 265     stddev: Standard deviation for initialization, if negative uses xavier.
 266     activation: If True, a modified activation is used (initialized ~ReLU6).
 267     bn_decay: decay for the batch norm moving averages.
 268     bn_epsilon: batch normalization epsilon.
 269     bn_renorm: whether to use batchnorm renormalization
 270 
 271   Returns:
 272     An `arg_scope` to use for the mobilenet v1 model.
 273   """
 274   batch_norm_params = {
 275       'center': True,
 276       'scale': True,
 277       'decay': bn_decay,
 278   }
 279   if bn_epsilon is not None:
 280     batch_norm_params['epsilon'] = bn_epsilon
 281   if is_training is not None:
 282     batch_norm_params['is_training'] = is_training
 283   if bn_renorm is not None:
 284     batch_norm_params['renorm'] = bn_renorm
 285   dropout_params = {}
 286   if is_training is not None:
 287     dropout_params['is_training'] = is_training
 288 
 289   instance_norm_params = {
 290       'center': True,
 291       'scale': True,
 292       'epsilon': 0.001,
 293   }
 294 
 295   if stddev < 0:
 296     weight_intitializer = slim.initializers.xavier_initializer()
 297   else:
 298     weight_intitializer = tf.truncated_normal_initializer(stddev=stddev)
 299 
 300   # Set weight_decay for weights in Conv and FC layers.
 301   with slim.arg_scope(
 302       [slim.conv2d, slim.fully_connected, slim.separable_conv2d],
 303       weights_initializer=weight_intitializer,
 304       activation_fn=activation,
 305       normalizer_fn=slim.batch_norm), \
 306       slim.arg_scope(
 307           [conv_blocks.expanded_conv], normalizer_fn=slim.batch_norm), \
 308       slim.arg_scope([mobilenet.apply_activation], activation_fn=activation),\
 309       slim.arg_scope([slim.batch_norm], **batch_norm_params), \
 310       slim.arg_scope([mobilenet.mobilenet_base, mobilenet.mobilenet],
 311                      is_training=is_training),\
 312       slim.arg_scope([slim.dropout], **dropout_params), \
 313       slim.arg_scope([slim.instance_norm], **instance_norm_params), \
 314       slim.arg_scope([slim.conv2d], \
 315                      weights_regularizer=slim.l2_regularizer(weight_decay)), \
 316       slim.arg_scope([slim.separable_conv2d], weights_regularizer=None), \
 317       slim.arg_scope([slim.conv2d, slim.separable_conv2d], padding='SAME') as s:
 318     return s
 319

 478 def _preprocess_subtract_imagenet_mean(inputs, dtype=tf.float32):
 479   """Subtract Imagenet mean RGB value."""
 480   mean_rgb = tf.reshape(_MEAN_RGB, [1, 1, 1, 3])
 481   num_channels = tf.shape(inputs)[-1]
 482   # We set mean pixel as 0 for the non-RGB channels.
 483   mean_rgb_extended = tf.concat(
 484       [mean_rgb, tf.zeros([1, 1, 1, num_channels - 3])], axis=3)
 485   return tf.cast(inputs - mean_rgb_extended, dtype=dtype)

 488 def _preprocess_zero_mean_unit_range(inputs, dtype=tf.float32):
 489   """Map image values from [0, 255] to [-1, 1]."""
 490   preprocessed_inputs = (2.0 / 255.0) * tf.to_float(inputs) - 1.0
 491   return tf.cast(preprocessed_inputs, dtype=dtype)

 513 def mean_pixel(model_variant=None):
 514   """Gets mean pixel value.
 515 
 516   This function returns different mean pixel value, depending on the input
 517   model_variant which adopts different preprocessing functions. We currently
 518   handle the following preprocessing functions:
 519   (1) _preprocess_subtract_imagenet_mean. We simply return mean pixel value.
 520   (2) _preprocess_zero_mean_unit_range. We return [127.5, 127.5, 127.5].
 521   The return values are used in a way that the padded regions after
 522   pre-processing will contain value 0.
 523 
 524   Args:
 525     model_variant: Model variant (string) for feature extraction. For
 526       backwards compatibility, model_variant=None returns _MEAN_RGB.
 527 
 528   Returns:
 529     Mean pixel value.
 530   """
 531   if model_variant in ['resnet_v1_50',
 532                        'resnet_v1_101'] or model_variant is None:
 533     return _MEAN_RGB
 534   else:
 535     return [127.5, 127.5, 127.5]

 539                      output_stride=8,
 540                      multi_grid=None,
 541                      depth_multiplier=1.0,
 542                      divisible_by=None,
 543                      final_endpoint=None,
 544                      model_variant=None,
 545                      weight_decay=0.0001,
 546                      reuse=None,
 547                      is_training=False,
 548                      fine_tune_batch_norm=False,
 549                      regularize_depthwise=False,
 550                      preprocess_images=True,
 551                      preprocessed_images_dtype=tf.float32,
 552                      num_classes=None,
 553                      global_pool=False,
 554                      nas_architecture_options=None,
 555                      nas_training_hyper_parameters=None,
 556                      use_bounded_activation=False):
 557   """Extracts features by the particular model_variant.
 558 
 559   Args:
 560     images: A tensor of size [batch, height, width, channels].
 561     output_stride: The ratio of input to output spatial resolution.
 562     multi_grid: Employ a hierarchy of different atrous rates within network.
 563     depth_multiplier: Float multiplier for the depth (number of channels)
 564       for all convolution ops used in MobileNet.
 565     divisible_by: None (use default setting) or an integer that ensures all
 566       layers # channels will be divisible by this number. Used in MobileNet.
 567     final_endpoint: The MobileNet endpoint to construct the network up to.
 568     model_variant: Model variant for feature extraction.
 569     weight_decay: The weight decay for model variables.
 570     reuse: Reuse the model variables or not.
 571     is_training: Is training or not.
 572     fine_tune_batch_norm: Fine-tune the batch norm parameters or not.
 573     regularize_depthwise: Whether or not apply L2-norm regularization on the
 574       depthwise convolution weights.
 575     preprocess_images: Performs preprocessing on images or not. Defaults to
 576       True. Set to False if preprocessing will be done by other functions. We
 577       supprot two types of preprocessing: (1) Mean pixel substraction and (2)
 578       Pixel values normalization to be [-1, 1].
 579     preprocessed_images_dtype: The type after the preprocessing function.
 580     num_classes: Number of classes for image classification task. Defaults
 581       to None for dense prediction tasks.
 582     global_pool: Global pooling for image classification task. Defaults to
 583       False, since dense prediction tasks do not use this.
 584     nas_architecture_options: A dictionary storing NAS architecture options.
 585       It is either None or its kerys are:
 586       - `nas_stem_output_num_conv_filters`: Number of filters of the NAS stem
 587         output tensor.
 588       - `nas_use_classification_head`: Boolean, use image classification head.
 589     nas_training_hyper_parameters: A dictionary storing hyper-parameters for
 590       training nas models. It is either None or its keys are:
 591       - `drop_path_keep_prob`: Probability to keep each path in the cell when
 592         training.
 593       - `total_training_steps`: Total training steps to help drop path
 594         probability calculation.
 595     use_bounded_activation: Whether or not to use bounded activations. Bounded
 596       activations better lend themselves to quantized inference. Currently,
 597       bounded activation is only used in xception model.
 598 
 599   Returns:
 600     features: A tensor of size [batch, feature_height, feature_width,
 601       feature_channels], where feature_height/feature_width are determined
 602       by the images height/width and output_stride.
 603     end_points: A dictionary from components of the network to the corresponding
 604       activation.
 605 
 606   Raises:
 607     ValueError: Unrecognized model variant.
 608   """
 609   if 'resnet' in model_variant:
 610     arg_scope = arg_scopes_map[model_variant](
 611         weight_decay=weight_decay,
 612         batch_norm_decay=0.95,
 613         batch_norm_epsilon=1e-5,
 614         batch_norm_scale=True)
 615     features, end_points = get_network(
 616         model_variant, preprocess_images, preprocessed_images_dtype, arg_scope)(
 617             inputs=images,
 618             num_classes=num_classes,
 619             is_training=(is_training and fine_tune_batch_norm),
 620             global_pool=global_pool,
 621             output_stride=output_stride,
 622             multi_grid=multi_grid,
 623             reuse=reuse,
 624             scope=name_scope[model_variant])
 625   elif 'xception' in model_variant:
 626     arg_scope = arg_scopes_map[model_variant](
 627         weight_decay=weight_decay,
 628         batch_norm_decay=0.9997,
 629         batch_norm_epsilon=1e-3,
 630         batch_norm_scale=True,
 631         regularize_depthwise=regularize_depthwise,
 632         use_bounded_activation=use_bounded_activation)
 633     features, end_points = get_network(
 634         model_variant, preprocess_images, preprocessed_images_dtype, arg_scope)(
 635             inputs=images,
 636             num_classes=num_classes,
 637             is_training=(is_training and fine_tune_batch_norm),
 638             global_pool=global_pool,
 639             output_stride=output_stride,
 640             regularize_depthwise=regularize_depthwise,
 641             multi_grid=multi_grid,
 642             reuse=reuse,
 643             scope=name_scope[model_variant])
 644   elif 'mobilenet' in model_variant or model_variant.startswith('mnas'):
 645     arg_scope = arg_scopes_map[model_variant](
 646         is_training=(is_training and fine_tune_batch_norm),
 647         weight_decay=weight_decay)
 648     features, end_points = get_network(
 649         model_variant, preprocess_images, preprocessed_images_dtype, arg_scope)(
 650             inputs=images,
 651             depth_multiplier=depth_multiplier,
 652             divisible_by=divisible_by,
 653             output_stride=output_stride,
 654             reuse=reuse,
 655             scope=name_scope[model_variant],
 656             final_endpoint=final_endpoint)
 657   elif model_variant.startswith('nas'):
 658     arg_scope = arg_scopes_map[model_variant](
 659         weight_decay=weight_decay,
 660         batch_norm_decay=0.9997,
 661         batch_norm_epsilon=1e-3)
 662     features, end_points = get_network(
 663         model_variant, preprocess_images, preprocessed_images_dtype, arg_scope)(
 664             inputs=images,
 665             num_classes=num_classes,
 666             is_training=(is_training and fine_tune_batch_norm),
 667             global_pool=global_pool,
 668             output_stride=output_stride,
 669             nas_architecture_options=nas_architecture_options,
 670             nas_training_hyper_parameters=nas_training_hyper_parameters,
 671             reuse=reuse,
 672             scope=name_scope[model_variant])
 673   else:
 674     raise ValueError('Unknown model variant %s.' % model_variant)
 675 
 676   return features, end_points

 679 def get_network(network_name, preprocess_images,
 680                 preprocessed_images_dtype=tf.float32, arg_scope=None):
 681   """Gets the network.
 682 
 683   Args:
 684     network_name: Network name.
 685     preprocess_images: Preprocesses the images or not.
 686     preprocessed_images_dtype: The type after the preprocessing function.
 687     arg_scope: Optional, arg_scope to build the network. If not provided the
 688       default arg_scope of the network would be used.
 689 
 690   Returns:
 691     A network function that is used to extract features.
 692 
 693   Raises:
 694     ValueError: network is not supported.
 695   """
 696   if network_name not in networks_map:
 697     raise ValueError('Unsupported network %s.' % network_name)
 698   arg_scope = arg_scope or arg_scopes_map[network_name]()
 699   def _identity_function(inputs, dtype=preprocessed_images_dtype):
 700     return tf.cast(inputs, dtype=dtype)
 701   if preprocess_images:
 702     preprocess_function = _PREPROCESS_FN[network_name]
 703   else:
 704     preprocess_function = _identity_function
 705   func = networks_map[network_name]
 706   @functools.wraps(func)
 707   def network_fn(inputs, *args, **kwargs):
 708     with slim.arg_scope(arg_scope):
 709       return func(preprocess_function(inputs, preprocessed_images_dtype),
 710                   *args, **kwargs)
 711   return network_fn
