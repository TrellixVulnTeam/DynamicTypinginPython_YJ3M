  41 class NASBaseCell(object):
  42   """NASNet Cell class that is used as a 'layer' in image architectures."""
  43 
  44   def __init__(self, num_conv_filters, operations, used_hiddenstates,
  45                hiddenstate_indices, drop_path_keep_prob, total_num_cells,
  46                total_training_steps, batch_norm_fn=slim.batch_norm):
  47     """Init function.
  48 
  49     For more details about NAS cell, see
  50     https://arxiv.org/abs/1707.07012 and https://arxiv.org/abs/1712.00559.
  51 
  52     Args:
  53       num_conv_filters: The number of filters for each convolution operation.
  54       operations: List of operations that are performed in the NASNet Cell in
  55         order.
  56       used_hiddenstates: Binary array that signals if the hiddenstate was used
  57         within the cell. This is used to determine what outputs of the cell
  58         should be concatenated together.
  59       hiddenstate_indices: Determines what hiddenstates should be combined
  60         together with the specified operations to create the NASNet cell.
  61       drop_path_keep_prob: Float, drop path keep probability.
  62       total_num_cells: Integer, total number of cells.
  63       total_training_steps: Integer, total training steps.
  64       batch_norm_fn: Function, batch norm function. Defaults to
  65         slim.batch_norm.
  66     """
  67     if len(hiddenstate_indices) != len(operations):
  68       raise ValueError(
  69           'Number of hiddenstate_indices and operations should be the same.')
  70     if len(operations) % 2:
  71       raise ValueError('Number of operations should be even.')
  72     self._num_conv_filters = num_conv_filters
  73     self._operations = operations
  74     self._used_hiddenstates = used_hiddenstates
  75     self._hiddenstate_indices = hiddenstate_indices
  76     self._drop_path_keep_prob = drop_path_keep_prob
  77     self._total_num_cells = total_num_cells
  78     self._total_training_steps = total_training_steps
  79     self._batch_norm_fn = batch_norm_fn
  80 
  81   def __call__(self, net, scope, filter_scaling, stride, prev_layer, cell_num):
  82     """Runs the conv cell."""
  83     self._cell_num = cell_num
  84     self._filter_scaling = filter_scaling
  85     self._filter_size = int(self._num_conv_filters * filter_scaling)
  86 
  87     with tf.variable_scope(scope):
  88       net = self._cell_base(net, prev_layer)
  89       for i in range(len(self._operations) // 2):
  90         with tf.variable_scope('comb_iter_{}'.format(i)):
  91           h1 = net[self._hiddenstate_indices[i * 2]]
  92           h2 = net[self._hiddenstate_indices[i * 2 + 1]]
  93           with tf.variable_scope('left'):
  94             h1 = self._apply_conv_operation(
  95                 h1, self._operations[i * 2], stride,
  96                 self._hiddenstate_indices[i * 2] < 2)
  97           with tf.variable_scope('right'):
  98             h2 = self._apply_conv_operation(
  99                 h2, self._operations[i * 2 + 1], stride,
 100                 self._hiddenstate_indices[i * 2 + 1] < 2)
 101           with tf.variable_scope('combine'):
 102             h = h1 + h2
 103           net.append(h)
 104 
 105       with tf.variable_scope('cell_output'):
 106         net = self._combine_unused_states(net)
 107 
 108       return net
 109 
 110   def _cell_base(self, net, prev_layer):
 111     """Runs the beginning of the conv cell before the chosen ops are run."""
 112     filter_size = self._filter_size
 113 
 114     if prev_layer is None:
 115       prev_layer = net
 116     else:
 117       if net.shape[2] != prev_layer.shape[2]:
 118         prev_layer = resize_bilinear(
 119             prev_layer, tf.shape(net)[1:3], prev_layer.dtype)
 120       if filter_size != prev_layer.shape[3]:
 121         prev_layer = tf.nn.relu(prev_layer)
 122         prev_layer = slim.conv2d(prev_layer, filter_size, 1, scope='prev_1x1')
 123         prev_layer = self._batch_norm_fn(prev_layer, scope='prev_bn')
 124 
 125     net = tf.nn.relu(net)
 126     net = slim.conv2d(net, filter_size, 1, scope='1x1')
 127     net = self._batch_norm_fn(net, scope='beginning_bn')
 128     net = tf.split(axis=3, num_or_size_splits=1, value=net)
 129     net.append(prev_layer)
 130     return net
 131 
 132   def _apply_conv_operation(self, net, operation, stride,
 133                             is_from_original_input):
 134     """Applies the predicted conv operation to net."""
 135     if stride > 1 and not is_from_original_input:
 136       stride = 1
 137     input_filters = net.shape[3]
 138     filter_size = self._filter_size
 139     if 'separable' in operation:
 140       num_layers = int(operation.split('_')[-1])
 141       kernel_size = int(operation.split('x')[0][-1])
 142       for layer_num in range(num_layers):
 143         net = tf.nn.relu(net)
 144         net = separable_conv2d_same(
 145             net,
 146             filter_size,
 147             kernel_size,
 148             depth_multiplier=1,
 149             scope='separable_{0}x{0}_{1}'.format(kernel_size, layer_num + 1),
 150             stride=stride)
 151         net = self._batch_norm_fn(
 152             net, scope='bn_sep_{0}x{0}_{1}'.format(kernel_size, layer_num + 1))
 153         stride = 1
 154     elif 'atrous' in operation:
 155       kernel_size = int(operation.split('x')[0][-1])
 156       net = tf.nn.relu(net)
 157       if stride == 2:
 158         scaled_height = scale_dimension(tf.shape(net)[1], 0.5)
 159         scaled_width = scale_dimension(tf.shape(net)[2], 0.5)
 160         net = resize_bilinear(net, [scaled_height, scaled_width], net.dtype)
 161         net = resnet_utils.conv2d_same(
 162             net, filter_size, kernel_size, rate=1, stride=1,
 163             scope='atrous_{0}x{0}'.format(kernel_size))
 164       else:
 165         net = resnet_utils.conv2d_same(
 166             net, filter_size, kernel_size, rate=2, stride=1,
 167             scope='atrous_{0}x{0}'.format(kernel_size))
 168       net = self._batch_norm_fn(net, scope='bn_atr_{0}x{0}'.format(kernel_size))
 169     elif operation in ['none']:
 170       if stride > 1 or (input_filters != filter_size):
 171         net = tf.nn.relu(net)
 172         net = slim.conv2d(net, filter_size, 1, stride=stride, scope='1x1')
 173         net = self._batch_norm_fn(net, scope='bn_1')
 174     elif 'pool' in operation:
 175       pooling_type = operation.split('_')[0]
 176       pooling_shape = int(operation.split('_')[-1].split('x')[0])
 177       if pooling_type == 'avg':
 178         net = slim.avg_pool2d(net, pooling_shape, stride=stride, padding='SAME')
 179       elif pooling_type == 'max':
 180         net = slim.max_pool2d(net, pooling_shape, stride=stride, padding='SAME')
 181       else:
 182         raise ValueError('Unimplemented pooling type: ', pooling_type)
 183       if input_filters != filter_size:
 184         net = slim.conv2d(net, filter_size, 1, stride=1, scope='1x1')
 185         net = self._batch_norm_fn(net, scope='bn_1')
 186     else:
 187       raise ValueError('Unimplemented operation', operation)
 188 
 189     if operation != 'none':
 190       net = self._apply_drop_path(net)
 191     return net
 192 
 193   def _combine_unused_states(self, net):
 194     """Concatenates the unused hidden states of the cell."""
 195     used_hiddenstates = self._used_hiddenstates
 196     states_to_combine = ([
 197         h for h, is_used in zip(net, used_hiddenstates) if not is_used])
 198     net = tf.concat(values=states_to_combine, axis=3)
 199     return net

 201   @contrib_framework.add_arg_scope
 202   def _apply_drop_path(self, net):
 203     """Apply drop_path regularization."""
 204     drop_path_keep_prob = self._drop_path_keep_prob
 205     if drop_path_keep_prob < 1.0:
 206       # Scale keep prob by layer number.
 207       assert self._cell_num != -1
 208       layer_ratio = (self._cell_num + 1) / float(self._total_num_cells)
 209       drop_path_keep_prob = 1 - layer_ratio * (1 - drop_path_keep_prob)
 210       # Decrease keep prob over time.
 211       current_step = tf.cast(tf.train.get_or_create_global_step(), tf.float32)
 212       current_ratio = tf.minimum(1.0, current_step / self._total_training_steps)
 213       drop_path_keep_prob = (1 - current_ratio * (1 - drop_path_keep_prob))
 214       # Drop path.
 215       noise_shape = [tf.shape(net)[0], 1, 1, 1]
 216       random_tensor = drop_path_keep_prob
 217       random_tensor += tf.random_uniform(noise_shape, dtype=tf.float32)
 218       binary_tensor = tf.cast(tf.floor(random_tensor), net.dtype)
 219       keep_prob_inv = tf.cast(1.0 / drop_path_keep_prob, net.dtype)
 220       net = net * keep_prob_inv * binary_tensor
 221     return net
