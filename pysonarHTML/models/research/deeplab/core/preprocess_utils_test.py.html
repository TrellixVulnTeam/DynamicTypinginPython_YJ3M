  31   def testNoFlipWhenProbIsZero(self):
  32     numpy_image = np.dstack([[[5., 6.],
  33                               [9., 0.]],
  34                              [[4., 3.],
  35                               [3., 5.]]])
  36     image = tf.convert_to_tensor(numpy_image)
  37 
  38     with self.test_session():
  39       actual, is_flipped = preprocess_utils.flip_dim([image], prob=0, dim=0)
  40       self.assertAllEqual(numpy_image, actual.eval())
  41       self.assertAllEqual(False, is_flipped.eval())
  42       actual, is_flipped = preprocess_utils.flip_dim([image], prob=0, dim=1)
  43       self.assertAllEqual(numpy_image, actual.eval())
  44       self.assertAllEqual(False, is_flipped.eval())
  45       actual, is_flipped = preprocess_utils.flip_dim([image], prob=0, dim=2)
  46       self.assertAllEqual(numpy_image, actual.eval())
  47       self.assertAllEqual(False, is_flipped.eval())

  49   def testFlipWhenProbIsOne(self):
  50     numpy_image = np.dstack([[[5., 6.],
  51                               [9., 0.]],
  52                              [[4., 3.],
  53                               [3., 5.]]])
  54     dim0_flipped = np.dstack([[[9., 0.],
  55                                [5., 6.]],
  56                               [[3., 5.],
  57                                [4., 3.]]])
  58     dim1_flipped = np.dstack([[[6., 5.],
  59                                [0., 9.]],
  60                               [[3., 4.],
  61                                [5., 3.]]])
  62     dim2_flipped = np.dstack([[[4., 3.],
  63                                [3., 5.]],
  64                               [[5., 6.],
  65                                [9., 0.]]])
  66     image = tf.convert_to_tensor(numpy_image)
  67 
  68     with self.test_session():
  69       actual, is_flipped = preprocess_utils.flip_dim([image], prob=1, dim=0)
  70       self.assertAllEqual(dim0_flipped, actual.eval())
  71       self.assertAllEqual(True, is_flipped.eval())
  72       actual, is_flipped = preprocess_utils.flip_dim([image], prob=1, dim=1)
  73       self.assertAllEqual(dim1_flipped, actual.eval())
  74       self.assertAllEqual(True, is_flipped.eval())
  75       actual, is_flipped = preprocess_utils.flip_dim([image], prob=1, dim=2)
  76       self.assertAllEqual(dim2_flipped, actual.eval())
  77       self.assertAllEqual(True, is_flipped.eval())

  79   def testFlipMultipleImagesConsistentlyWhenProbIsOne(self):
  80     numpy_image = np.dstack([[[5., 6.],
  81                               [9., 0.]],
  82                              [[4., 3.],
  83                               [3., 5.]]])
  84     numpy_label = np.dstack([[[0., 1.],
  85                               [2., 3.]]])
  86     image_dim1_flipped = np.dstack([[[6., 5.],
  87                                      [0., 9.]],
  88                                     [[3., 4.],
  89                                      [5., 3.]]])
  90     label_dim1_flipped = np.dstack([[[1., 0.],
  91                                      [3., 2.]]])
  92     image = tf.convert_to_tensor(numpy_image)
  93     label = tf.convert_to_tensor(numpy_label)
  94 
  95     with self.test_session() as sess:
  96       image, label, is_flipped = preprocess_utils.flip_dim(
  97           [image, label], prob=1, dim=1)
  98       actual_image, actual_label = sess.run([image, label])
  99       self.assertAllEqual(image_dim1_flipped, actual_image)
 100       self.assertAllEqual(label_dim1_flipped, actual_label)
 101       self.assertEqual(True, is_flipped.eval())

 103   def testReturnRandomFlipsOnMultipleEvals(self):
 104     numpy_image = np.dstack([[[5., 6.],
 105                               [9., 0.]],
 106                              [[4., 3.],
 107                               [3., 5.]]])
 108     dim1_flipped = np.dstack([[[6., 5.],
 109                                [0., 9.]],
 110                               [[3., 4.],
 111                                [5., 3.]]])
 112     image = tf.convert_to_tensor(numpy_image)
 113     tf.compat.v1.set_random_seed(53)
 114 
 115     with self.test_session() as sess:
 116       actual, is_flipped = preprocess_utils.flip_dim(
 117           [image], prob=0.5, dim=1)
 118       actual_image, actual_is_flipped = sess.run([actual, is_flipped])
 119       self.assertAllEqual(numpy_image, actual_image)
 120       self.assertEqual(False, actual_is_flipped)
 121       actual_image, actual_is_flipped = sess.run([actual, is_flipped])
 122       self.assertAllEqual(dim1_flipped, actual_image)
 123       self.assertEqual(True, actual_is_flipped)

 125   def testReturnCorrectCropOfSingleImage(self):
 126     np.random.seed(0)
 127 
 128     height, width = 10, 20
 129     image = np.random.randint(0, 256, size=(height, width, 3))
 130 
 131     crop_height, crop_width = 2, 4
 132 
 133     image_placeholder = tf.placeholder(tf.int32, shape=(None, None, 3))
 134     [cropped] = preprocess_utils.random_crop([image_placeholder],
 135                                              crop_height,
 136                                              crop_width)
 137 
 138     with self.test_session():
 139       cropped_image = cropped.eval(feed_dict={image_placeholder: image})
 140 
 141     # Ensure we can find the cropped image in the original:
 142     is_found = False
 143     for x in range(0, width - crop_width + 1):
 144       for y in range(0, height - crop_height + 1):
 145         if np.isclose(image[y:y+crop_height, x:x+crop_width, :],
 146                       cropped_image).all():
 147           is_found = True
 148           break
 149 
 150     self.assertTrue(is_found)

 181   def testReturnConsistenCropsOfImagesInTheList(self):
 182     tf.compat.v1.set_random_seed(0)
 183 
 184     height, width = 10, 20
 185     crop_height, crop_width = 2, 3
 186     labels = np.linspace(0, height * width-1, height * width)
 187     labels = labels.reshape((height, width, 1))
 188     image = np.tile(labels, (1, 1, 3))
 189

 191     label_placeholder = tf.placeholder(tf.int32, shape=(None, None, 1))
 192     [cropped_image, cropped_label] = preprocess_utils.random_crop(
 193         [image_placeholder, label_placeholder], crop_height, crop_width)
 194 
 195     with self.test_session() as sess:
 196       cropped_image, cropped_labels = sess.run([cropped_image, cropped_label],
 197                                                feed_dict={
 198                                                    image_placeholder: image,
 199                                                    label_placeholder: labels})
 200       for i in range(3):
 201         self.assertAllEqual(cropped_image[:, :, i], cropped_labels.squeeze())

 243   def testReturnPaddedImageWithNonZeroPadValue(self):
 244     for dtype in [np.int32, np.int64, np.float32, np.float64]:
 245       image = np.dstack([[[5, 6],
 246                           [9, 0]],
 247                          [[4, 3],
 248                           [3, 5]]]).astype(dtype)
 249       expected_image = np.dstack([[[255, 255, 255, 255, 255],
 250                                    [255, 255, 255, 255, 255],
 251                                    [255, 5, 6, 255, 255],
 252                                    [255, 9, 0, 255, 255],
 253                                    [255, 255, 255, 255, 255]],
 254                                   [[255, 255, 255, 255, 255],
 255                                    [255, 255, 255, 255, 255],
 256                                    [255, 4, 3, 255, 255],
 257                                    [255, 3, 5, 255, 255],
 258                                    [255, 255, 255, 255, 255]]]).astype(dtype)
 259 
 260       with self.session() as sess:
 261         padded_image = preprocess_utils.pad_to_bounding_box(
 262             image, 2, 1, 5, 5, 255)
 263         padded_image = sess.run(padded_image)
 264         self.assertAllClose(padded_image, expected_image)
 265         # Add batch size = 1 to image.
 266         padded_image = preprocess_utils.pad_to_bounding_box(
 267             np.expand_dims(image, 0), 2, 1, 5, 5, 255)
 268         padded_image = sess.run(padded_image)
 269         self.assertAllClose(padded_image, np.expand_dims(expected_image, 0))
