  29 def resize_bilinear(images, size, output_dtype=tf.float32):
  30   """Returns resized images as output_type.
  31 
  32   Args:
  33     images: A tensor of size [batch, height_in, width_in, channels].
  34     size: A 1-D int32 Tensor of 2 elements: new_height, new_width. The new size
  35       for the images.
  36     output_dtype: The destination type.
  37   Returns:
  38     A tensor of size [batch, height_out, width_out, channels] as a dtype of
  39       output_dtype.
  40   """
  41   images = tf.image.resize_bilinear(images, size, align_corners=True)
  42   return tf.cast(images, dtype=output_dtype)

  45 def scale_dimension(dim, scale):
  46   """Scales the input dimension.
  47 
  48   Args:
  49     dim: Input dimension (a scalar or a scalar Tensor).
  50     scale: The amount of scaling applied to the input.
  51 
  52   Returns:
  53     Scaled dimension.
  54   """
  55   if isinstance(dim, tf.Tensor):
  56     return tf.cast((tf.to_float(dim) - 1.0) * scale + 1.0, dtype=tf.int32)
  57   else:
  58     return int((float(dim) - 1.0) * scale + 1.0)

  61 def split_separable_conv2d(inputs,
  62                            filters,
  63                            kernel_size=3,
  64                            rate=1,
  65                            weight_decay=0.00004,
  66                            depthwise_weights_initializer_stddev=0.33,
  67                            pointwise_weights_initializer_stddev=0.06,
  68                            scope=None):
  69   """Splits a separable conv2d into depthwise and pointwise conv2d.
  70 
  71   This operation differs from `tf.layers.separable_conv2d` as this operation
  72   applies activation function between depthwise and pointwise conv2d.
  73 
  74   Args:
  75     inputs: Input tensor with shape [batch, height, width, channels].
  76     filters: Number of filters in the 1x1 pointwise convolution.
  77     kernel_size: A list of length 2: [kernel_height, kernel_width] of
  78       of the filters. Can be an int if both values are the same.
  79     rate: Atrous convolution rate for the depthwise convolution.
  80     weight_decay: The weight decay to use for regularizing the model.
  81     depthwise_weights_initializer_stddev: The standard deviation of the
  82       truncated normal weight initializer for depthwise convolution.
  83     pointwise_weights_initializer_stddev: The standard deviation of the
  84       truncated normal weight initializer for pointwise convolution.
  85     scope: Optional scope for the operation.
  86 
  87   Returns:
  88     Computed features after split separable conv2d.
  89   """
  90   outputs = slim.separable_conv2d(
  91       inputs,
  92       None,
  93       kernel_size=kernel_size,
  94       depth_multiplier=1,
  95       rate=rate,
  96       weights_initializer=tf.truncated_normal_initializer(
  97           stddev=depthwise_weights_initializer_stddev),
  98       weights_regularizer=None,
  99       scope=scope + '_depthwise')
 100   return slim.conv2d(
 101       outputs,
 102       filters,
 103       1,
 104       weights_initializer=tf.truncated_normal_initializer(
 105           stddev=pointwise_weights_initializer_stddev),
 106       weights_regularizer=slim.l2_regularizer(weight_decay),
 107       scope=scope + '_pointwise')

 110 def get_label_weight_mask(labels, ignore_label, num_classes, label_weights=1.0):
 111   """Gets the label weight mask.
 112 
 113   Args:
 114     labels: A Tensor of labels with the shape of [-1].
 115     ignore_label: Integer, label to ignore.
 116     num_classes: Integer, the number of semantic classes.
 117     label_weights: A float or a list of weights. If it is a float, it means all
 118       the labels have the same weight. If it is a list of weights, then each
 119       element in the list represents the weight for the label of its index, for
 120       example, label_weights = [0.1, 0.5] means the weight for label 0 is 0.1
 121       and the weight for label 1 is 0.5.
 122 
 123   Returns:
 124     A Tensor of label weights with the same shape of labels, each element is the
 125       weight for the label with the same index in labels and the element is 0.0
 126       if the label is to ignore.
 127 
 128   Raises:
 129     ValueError: If label_weights is neither a float nor a list, or if
 130       label_weights is a list and its length is not equal to num_classes.
 131   """
 132   if not isinstance(label_weights, (float, list)):
 133     raise ValueError(
 134         'The type of label_weights is invalid, it must be a float or a list.')
 135 
 136   if isinstance(label_weights, list) and len(label_weights) != num_classes:
 137     raise ValueError(
 138         'Length of label_weights must be equal to num_classes if it is a list, '
 139         'label_weights: %s, num_classes: %d.' % (label_weights, num_classes))
 140 
 141   not_ignore_mask = tf.not_equal(labels, ignore_label)
 142   not_ignore_mask = tf.cast(not_ignore_mask, tf.float32)
 143   if isinstance(label_weights, float):
 144     return not_ignore_mask * label_weights
 145 
 146   label_weights = tf.constant(label_weights, tf.float32)
 147   weight_mask = tf.einsum('...y,y->...',
 148                           tf.one_hot(labels, num_classes, dtype=tf.float32),
 149                           label_weights)
 150   return tf.multiply(not_ignore_mask, weight_mask)

 153 def get_batch_norm_fn(sync_batch_norm_method):
 154   """Gets batch norm function.
 155 
 156   Currently we only support the following methods:
 157     - `None` (no sync batch norm). We use slim.batch_norm in this case.
 158 
 159   Args:
 160     sync_batch_norm_method: String, method used to sync batch norm.
 161 
 162   Returns:
 163     Batchnorm function.
 164 
 165   Raises:
 166     ValueError: If sync_batch_norm_method is not supported.
 167   """
 168   if sync_batch_norm_method == 'None':
 169     return slim.batch_norm
 170   else:
 171     raise ValueError('Unsupported sync_batch_norm_method.')

 174 def get_batch_norm_params(decay=0.9997,
 175                           epsilon=1e-5,
 176                           center=True,
 177                           scale=True,
 178                           is_training=True,
 179                           sync_batch_norm_method='None',
 180                           initialize_gamma_as_zeros=False):
 181   """Gets batch norm parameters.
 182 
 183   Args:
 184     decay: Float, decay for the moving average.
 185     epsilon: Float, value added to variance to avoid dividing by zero.
 186     center: Boolean. If True, add offset of `beta` to normalized tensor. If
 187       False,`beta` is ignored.
 188     scale: Boolean. If True, multiply by `gamma`. If False, `gamma` is not used.
 189     is_training: Boolean, whether or not the layer is in training mode.
 190     sync_batch_norm_method: String, method used to sync batch norm.
 191     initialize_gamma_as_zeros: Boolean, initializing `gamma` as zeros or not.
 192 
 193   Returns:
 194     A dictionary for batchnorm parameters.
 195 
 196   Raises:
 197     ValueError: If sync_batch_norm_method is not supported.
 198   """
 199   batch_norm_params = {
 200       'is_training': is_training,
 201       'decay': decay,
 202       'epsilon': epsilon,
 203       'scale': scale,
 204       'center': center,
 205   }
 206   if initialize_gamma_as_zeros:
 207     if sync_batch_norm_method == 'None':
 208       # Slim-type gamma_initialier.
 209       batch_norm_params['param_initializers'] = {
 210           'gamma': tf.zeros_initializer(),
 211       }
 212     else:
 213       raise ValueError('Unsupported sync_batch_norm_method.')
 214   return batch_norm_params
