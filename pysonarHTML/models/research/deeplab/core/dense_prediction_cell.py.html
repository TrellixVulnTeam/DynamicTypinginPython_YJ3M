  17 """Dense Prediction Cell class that can be evolved in semantic segmentation.
  18 
  19 DensePredictionCell is used as a `layer` in semantic segmentation whose
  20 architecture is determined by the `config`, a dictionary specifying
  21 the architecture.
  22 """

  41 _KERNEL = 'kernel'
  42 _RATE = 'rate'
  43 _GRID_SIZE = 'grid_size'
  44 _TARGET_SIZE = 'target_size'

  48 def dense_prediction_cell_hparams():
  49   """DensePredictionCell HParams.
  50 
  51   Returns:
  52     A dictionary of hyper-parameters used for dense prediction cell with keys:
  53       - reduction_size: Integer, the number of output filters for each operation
  54           inside the cell.
  55       - dropout_on_concat_features: Boolean, apply dropout on the concatenated
  56           features or not.
  57       - dropout_on_projection_features: Boolean, apply dropout on the projection
  58           features or not.
  59       - dropout_keep_prob: Float, when `dropout_on_concat_features' or
  60           `dropout_on_projection_features' is True, the `keep_prob` value used
  61           in the dropout operation.
  62       - concat_channels: Integer, the concatenated features will be
  63           channel-reduced to `concat_channels` channels.
  64       - conv_rate_multiplier: Integer, used to multiply the convolution rates.
  65           This is useful in the case when the output_stride is changed from 16
  66           to 8, we need to double the convolution rates correspondingly.
  67   """
  68   return {
  69       'reduction_size': 256,
  70       'dropout_on_concat_features': True,
  71       'dropout_on_projection_features': False,
  72       'dropout_keep_prob': 0.9,
  73       'concat_channels': 256,
  74       'conv_rate_multiplier': 1,
  75   }

  78 class DensePredictionCell(object):
  79   """DensePredictionCell class used as a 'layer' in semantic segmentation."""
  80 
  81   def __init__(self, config, hparams=None):
  82     """Initializes the dense prediction cell.
  83 
  84     Args:
  85       config: A dictionary storing the architecture of a dense prediction cell.
  86       hparams: A dictionary of hyper-parameters, provided by users. This
  87         dictionary will be used to update the default dictionary returned by
  88         dense_prediction_cell_hparams().
  89 
  90     Raises:
  91        ValueError: If `conv_rate_multiplier` has value < 1.
  92     """
  93     self.hparams = dense_prediction_cell_hparams()
  94     if hparams is not None:
  95       self.hparams.update(hparams)
  96     self.config = config
  97 
  98     # Check values in hparams are valid or not.
  99     if self.hparams['conv_rate_multiplier'] < 1:
 100       raise ValueError('conv_rate_multiplier cannot have value < 1.')
 101 
 102   def _get_pyramid_pooling_arguments(
 103       self, crop_size, output_stride, image_grid, image_pooling_crop_size=None):
 104     """Gets arguments for pyramid pooling.
 105 
 106     Args:
 107       crop_size: A list of two integers, [crop_height, crop_width] specifying
 108         whole patch crop size.
 109       output_stride: Integer, output stride value for extracted features.
 110       image_grid: A list of two integers, [image_grid_height, image_grid_width],
 111         specifying the grid size of how the pyramid pooling will be performed.
 112       image_pooling_crop_size: A list of two integers, [crop_height, crop_width]
 113         specifying the crop size for image pooling operations. Note that we
 114         decouple whole patch crop_size and image_pooling_crop_size as one could
 115         perform the image_pooling with different crop sizes.
 116 
 117     Returns:
 118       A list of (resize_value, pooled_kernel)
 119     """
 120     resize_height = utils.scale_dimension(crop_size[0], 1. / output_stride)
 121     resize_width = utils.scale_dimension(crop_size[1], 1. / output_stride)
 122     # If image_pooling_crop_size is not specified, use crop_size.
 123     if image_pooling_crop_size is None:
 124       image_pooling_crop_size = crop_size
 125     pooled_height = utils.scale_dimension(
 126         image_pooling_crop_size[0], 1. / (output_stride * image_grid[0]))
 127     pooled_width = utils.scale_dimension(
 128         image_pooling_crop_size[1], 1. / (output_stride * image_grid[1]))
 129     return ([resize_height, resize_width], [pooled_height, pooled_width])
 130 
 131   def _parse_operation(self, config, crop_size, output_stride,
 132                        image_pooling_crop_size=None):
 133     """Parses one operation.
 134 
 135     When 'operation' is 'pyramid_pooling', we compute the required
 136     hyper-parameters and save in config.
 137 
 138     Args:
 139       config: A dictionary storing required hyper-parameters for one
 140         operation.
 141       crop_size: A list of two integers, [crop_height, crop_width] specifying
 142         whole patch crop size.
 143       output_stride: Integer, output stride value for extracted features.
 144       image_pooling_crop_size: A list of two integers, [crop_height, crop_width]
 145         specifying the crop size for image pooling operations. Note that we
 146         decouple whole patch crop_size and image_pooling_crop_size as one could
 147         perform the image_pooling with different crop sizes.
 148 
 149     Returns:
 150       A dictionary stores the related information for the operation.
 151     """
 152     if config[_OP] == _PYRAMID_POOLING:
 153       (config[_TARGET_SIZE],
 154        config[_KERNEL]) = self._get_pyramid_pooling_arguments(
 155            crop_size=crop_size,
 156            output_stride=output_stride,
 157            image_grid=config[_GRID_SIZE],
 158            image_pooling_crop_size=image_pooling_crop_size)
 159 
 160     return config

 162   def build_cell(self,
 163                  features,
 164                  output_stride=16,
 165                  crop_size=None,
 166                  image_pooling_crop_size=None,
 167                  weight_decay=0.00004,
 168                  reuse=None,
 169                  is_training=False,
 170                  fine_tune_batch_norm=False,
 171                  scope=None):
 172     """Builds the dense prediction cell based on the config.
 173 
 174     Args:
 175       features: Input feature map of size [batch, height, width, channels].
 176       output_stride: Int, output stride at which the features were extracted.
 177       crop_size: A list [crop_height, crop_width], determining the input
 178         features resolution.
 179       image_pooling_crop_size: A list of two integers, [crop_height, crop_width]
 180         specifying the crop size for image pooling operations. Note that we
 181         decouple whole patch crop_size and image_pooling_crop_size as one could
 182         perform the image_pooling with different crop sizes.
 183       weight_decay: Float, the weight decay for model variables.
 184       reuse: Reuse the model variables or not.
 185       is_training: Boolean, is training or not.
 186       fine_tune_batch_norm: Boolean, fine-tuning batch norm parameters or not.
 187       scope: Optional string, specifying the variable scope.
 188 
 189     Returns:
 190       Features after passing through the constructed dense prediction cell with
 191         shape = [batch, height, width, channels] where channels are determined
 192         by `reduction_size` returned by dense_prediction_cell_hparams().
 193 
 194     Raises:
 195       ValueError: Use Convolution with kernel size not equal to 1x1 or 3x3 or
 196         the operation is not recognized.
 197     """
 198     batch_norm_params = {
 199         'is_training': is_training and fine_tune_batch_norm,
 200         'decay': 0.9997,
 201         'epsilon': 1e-5,
 202         'scale': True,
 203     }
 204     hparams = self.hparams
 205     with slim.arg_scope(
 206         [slim.conv2d, slim.separable_conv2d],
 207         weights_regularizer=slim.l2_regularizer(weight_decay),
 208         activation_fn=tf.nn.relu,
 209         normalizer_fn=slim.batch_norm,
 210         padding='SAME',
 211         stride=1,
 212         reuse=reuse):
 213       with slim.arg_scope([slim.batch_norm], **batch_norm_params):
 214         with tf.variable_scope(scope, _META_ARCHITECTURE_SCOPE, [features]):
 215           depth = hparams['reduction_size']
 216           branch_logits = []
 217           for i, current_config in enumerate(self.config):
 218             scope = 'branch%d' % i
 219             current_config = self._parse_operation(
 220                 config=current_config,
 221                 crop_size=crop_size,
 222                 output_stride=output_stride,
 223                 image_pooling_crop_size=image_pooling_crop_size)
 224             tf.logging.info(current_config)
 225             if current_config[_INPUT] < 0:
 226               operation_input = features
 227             else:
 228               operation_input = branch_logits[current_config[_INPUT]]
 229             if current_config[_OP] == _CONV:
 230               if current_config[_KERNEL] == [1, 1] or current_config[
 231                   _KERNEL] == 1:
 232                 branch_logits.append(
 233                     slim.conv2d(operation_input, depth, 1, scope=scope))
 234               else:
 235                 conv_rate = [r * hparams['conv_rate_multiplier']
 236                              for r in current_config[_RATE]]
 237                 branch_logits.append(
 238                     utils.split_separable_conv2d(
 239                         operation_input,
 240                         filters=depth,
 241                         kernel_size=current_config[_KERNEL],
 242                         rate=conv_rate,
 243                         weight_decay=weight_decay,
 244                         scope=scope))
 245             elif current_config[_OP] == _PYRAMID_POOLING:
 246               pooled_features = slim.avg_pool2d(
 247                   operation_input,
 248                   kernel_size=current_config[_KERNEL],
 249                   stride=[1, 1],
 250                   padding='VALID')
 251               pooled_features = slim.conv2d(
 252                   pooled_features,
 253                   depth,
 254                   1,
 255                   scope=scope)
 256               pooled_features = tf.image.resize_bilinear(
 257                   pooled_features,
 258                   current_config[_TARGET_SIZE],
 259                   align_corners=True)
 260               # Set shape for resize_height/resize_width if they are not Tensor.
 261               resize_height = current_config[_TARGET_SIZE][0]
 262               resize_width = current_config[_TARGET_SIZE][1]
 263               if isinstance(resize_height, tf.Tensor):
 264                 resize_height = None
 265               if isinstance(resize_width, tf.Tensor):
 266                 resize_width = None
 267               pooled_features.set_shape(
 268                   [None, resize_height, resize_width, depth])
 269               branch_logits.append(pooled_features)
 270             else:
 271               raise ValueError('Unrecognized operation.')
 272           # Merge branch logits.
 273           concat_logits = tf.concat(branch_logits, 3)
 274           if self.hparams['dropout_on_concat_features']:
 275             concat_logits = slim.dropout(
 276                 concat_logits,
 277                 keep_prob=self.hparams['dropout_keep_prob'],
 278                 is_training=is_training,
 279                 scope=_CONCAT_PROJECTION_SCOPE + '_dropout')
 280           concat_logits = slim.conv2d(concat_logits,
 281                                       self.hparams['concat_channels'],
 282                                       1,
 283                                       scope=_CONCAT_PROJECTION_SCOPE)
 284           if self.hparams['dropout_on_projection_features']:
 285             concat_logits = slim.dropout(
 286                 concat_logits,
 287                 keep_prob=self.hparams['dropout_keep_prob'],
 288                 is_training=is_training,
 289                 scope=_CONCAT_PROJECTION_SCOPE + '_dropout')
 290           return concat_logits
