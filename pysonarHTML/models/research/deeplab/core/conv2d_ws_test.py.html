 196   def testCreateConvWithWD(self):
 197     height, width = 7, 9
 198     weight_decay = 0.01
 199     with self.cached_session() as sess:
 200       images = tf.random_uniform((5, height, width, 3), seed=1)
 201       regularizer = contrib_layers.l2_regularizer(weight_decay)
 202       conv2d_ws.conv2d(images, 32, [3, 3], weights_regularizer=regularizer)
 203       l2_loss = tf.nn.l2_loss(
 204           contrib_framework.get_variables_by_name('weights')[0])
 205       wd = tf.get_collection(tf.GraphKeys.REGULARIZATION_LOSSES)[0]
 206       self.assertEqual(wd.op.name, 'Conv/kernel/Regularizer/l2_regularizer')
 207       sess.run(tf.global_variables_initializer())
 208       self.assertAlmostEqual(sess.run(wd), weight_decay * l2_loss.eval())

 236   def testReuseConvWithWD(self):
 237     height, width = 7, 9
 238     with self.cached_session():
 239       images = tf.random_uniform((5, height, width, 3), seed=1)
 240       weight_decay = contrib_layers.l2_regularizer(0.01)
 241       with contrib_framework.arg_scope([conv2d_ws.conv2d],
 242                                        weights_regularizer=weight_decay):
 243         conv2d_ws.conv2d(images, 32, [3, 3], scope='conv1')
 244         self.assertEqual(len(contrib_framework.get_variables()), 2)
 245         self.assertEqual(
 246             len(tf.get_collection(tf.GraphKeys.REGULARIZATION_LOSSES)), 1)
 247         conv2d_ws.conv2d(images, 32, [3, 3], scope='conv1', reuse=True)
 248         self.assertEqual(len(contrib_framework.get_variables()), 2)
 249         self.assertEqual(
 250             len(tf.get_collection(tf.GraphKeys.REGULARIZATION_LOSSES)), 1)
