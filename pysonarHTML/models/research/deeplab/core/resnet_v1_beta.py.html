  44 @slim.add_arg_scope
  45 def bottleneck(inputs,
  46                depth,
  47                depth_bottleneck,
  48                stride,
  49                unit_rate=1,
  50                rate=1,
  51                outputs_collections=None,
  52                scope=None):
  53   """Bottleneck residual unit variant with BN after convolutions.
  54 
  55   This is the original residual unit proposed in [1]. See Fig. 1(a) of [2] for
  56   its definition. Note that we use here the bottleneck variant which has an
  57   extra bottleneck layer.
  58 
  59   When putting together two consecutive ResNet blocks that use this unit, one
  60   should use stride = 2 in the last unit of the first block.
  61 
  62   Args:
  63     inputs: A tensor of size [batch, height, width, channels].
  64     depth: The depth of the ResNet unit output.
  65     depth_bottleneck: The depth of the bottleneck layers.
  66     stride: The ResNet unit's stride. Determines the amount of downsampling of
  67       the units output compared to its input.
  68     unit_rate: An integer, unit rate for atrous convolution.
  69     rate: An integer, rate for atrous convolution.
  70     outputs_collections: Collection to add the ResNet unit output.
  71     scope: Optional variable_scope.
  72 
  73   Returns:
  74     The ResNet unit's output.
  75   """
  76   with tf.variable_scope(scope, 'bottleneck_v1', [inputs]) as sc:
  77     depth_in = slim.utils.last_dimension(inputs.get_shape(), min_rank=4)
  78     if depth == depth_in:
  79       shortcut = resnet_utils.subsample(inputs, stride, 'shortcut')
  80     else:
  81       shortcut = conv2d_ws.conv2d(
  82           inputs,
  83           depth,
  84           [1, 1],
  85           stride=stride,
  86           activation_fn=None,
  87           scope='shortcut')
  88 
  89     residual = conv2d_ws.conv2d(inputs, depth_bottleneck, [1, 1], stride=1,
  90                                 scope='conv1')
  91     residual = conv2d_ws.conv2d_same(residual, depth_bottleneck, 3, stride,
  92                                      rate=rate*unit_rate, scope='conv2')
  93     residual = conv2d_ws.conv2d(residual, depth, [1, 1], stride=1,
  94                                 activation_fn=None, scope='conv3')
  95     output = tf.nn.relu(shortcut + residual)
  96 
  97     return slim.utils.collect_named_outputs(outputs_collections, sc.name,
  98                                             output)

 101 @slim.add_arg_scope
 102 def lite_bottleneck(inputs,
 103                     depth,
 104                     stride,
 105                     unit_rate=1,
 106                     rate=1,
 107                     outputs_collections=None,
 108                     scope=None):
 109   """Bottleneck residual unit variant with BN after convolutions.
 110 
 111   This is the original residual unit proposed in [1]. See Fig. 1(a) of [2] for
 112   its definition. Note that we use here the bottleneck variant which has an
 113   extra bottleneck layer.
 114 
 115   When putting together two consecutive ResNet blocks that use this unit, one
 116   should use stride = 2 in the last unit of the first block.
 117 
 118   Args:
 119     inputs: A tensor of size [batch, height, width, channels].
 120     depth: The depth of the ResNet unit output.
 121     stride: The ResNet unit's stride. Determines the amount of downsampling of
 122       the units output compared to its input.
 123     unit_rate: An integer, unit rate for atrous convolution.
 124     rate: An integer, rate for atrous convolution.
 125     outputs_collections: Collection to add the ResNet unit output.
 126     scope: Optional variable_scope.
 127 
 128   Returns:
 129     The ResNet unit's output.
 130   """
 131   with tf.variable_scope(scope, 'lite_bottleneck_v1', [inputs]) as sc:
 132     depth_in = slim.utils.last_dimension(inputs.get_shape(), min_rank=4)
 133     if depth == depth_in:
 134       shortcut = resnet_utils.subsample(inputs, stride, 'shortcut')
 135     else:
 136       shortcut = conv2d_ws.conv2d(
 137           inputs,
 138           depth, [1, 1],
 139           stride=stride,
 140           activation_fn=None,
 141           scope='shortcut')
 142 
 143     residual = conv2d_ws.conv2d_same(
 144         inputs, depth, 3, 1, rate=rate * unit_rate, scope='conv1')
 145     with slim.arg_scope([conv2d_ws.conv2d], activation_fn=None):
 146       residual = conv2d_ws.conv2d_same(
 147           residual, depth, 3, stride, rate=rate * unit_rate, scope='conv2')
 148     output = tf.nn.relu(shortcut + residual)
 149 
 150     return slim.utils.collect_named_outputs(outputs_collections, sc.name,
 151                                             output)

 154 def root_block_fn_for_beta_variant(net, depth_multiplier=1.0):
 155   """Gets root_block_fn for beta variant.
 156 
 157   ResNet-v1 beta variant modifies the first original 7x7 convolution to three
 158   3x3 convolutions.
 159 
 160   Args:
 161     net: A tensor of size [batch, height, width, channels], input to the model.
 162     depth_multiplier: Controls the number of convolution output channels for
 163       each input channel. The total number of depthwise convolution output
 164       channels will be equal to `num_filters_out * depth_multiplier`.
 165 
 166   Returns:
 167     A tensor after three 3x3 convolutions.
 168   """
 169   net = conv2d_ws.conv2d_same(
 170       net, int(64 * depth_multiplier), 3, stride=2, scope='conv1_1')
 171   net = conv2d_ws.conv2d_same(
 172       net, int(64 * depth_multiplier), 3, stride=1, scope='conv1_2')
 173   net = conv2d_ws.conv2d_same(
 174       net, int(128 * depth_multiplier), 3, stride=1, scope='conv1_3')
 175 
 176   return net

 179 def resnet_v1_beta(inputs,
 180                    blocks,
 181                    num_classes=None,
 182                    is_training=None,
 183                    global_pool=True,
 184                    output_stride=None,
 185                    root_block_fn=None,
 186                    reuse=None,
 187                    scope=None,
 188                    sync_batch_norm_method='None'):
 189   """Generator for v1 ResNet models (beta variant).
 190 
 191   This function generates a family of modified ResNet v1 models. In particular,
 192   the first original 7x7 convolution is replaced with three 3x3 convolutions.
 193   See the resnet_v1_*() methods for specific model instantiations, obtained by
 194   selecting different block instantiations that produce ResNets of various
 195   depths.
 196 
 197   The code is modified from slim/nets/resnet_v1.py, and please refer to it for
 198   more details.
 199 
 200   Args:
 201     inputs: A tensor of size [batch, height_in, width_in, channels].
 202     blocks: A list of length equal to the number of ResNet blocks. Each element
 203       is a resnet_utils.Block object describing the units in the block.
 204     num_classes: Number of predicted classes for classification tasks. If None
 205       we return the features before the logit layer.
 206     is_training: Enable/disable is_training for batch normalization.
 207     global_pool: If True, we perform global average pooling before computing the
 208       logits. Set to True for image classification, False for dense prediction.
 209     output_stride: If None, then the output will be computed at the nominal
 210       network stride. If output_stride is not None, it specifies the requested
 211       ratio of input to output spatial resolution.
 212     root_block_fn: The function consisting of convolution operations applied to
 213       the root input. If root_block_fn is None, use the original setting of
 214       RseNet-v1, which is simply one convolution with 7x7 kernel and stride=2.
 215     reuse: whether or not the network and its variables should be reused. To be
 216       able to reuse 'scope' must be given.
 217     scope: Optional variable_scope.
 218     sync_batch_norm_method: String, sync batchnorm method.
 219 
 220   Returns:
 221     net: A rank-4 tensor of size [batch, height_out, width_out, channels_out].
 222       If global_pool is False, then height_out and width_out are reduced by a
 223       factor of output_stride compared to the respective height_in and width_in,
 224       else both height_out and width_out equal one. If num_classes is None, then
 225       net is the output of the last ResNet block, potentially after global
 226       average pooling. If num_classes is not None, net contains the pre-softmax
 227       activations.
 228     end_points: A dictionary from components of the network to the corresponding
 229       activation.
 230 
 231   Raises:
 232     ValueError: If the target output_stride is not valid.
 233   """
 234   if root_block_fn is None:
 235     root_block_fn = functools.partial(conv2d_ws.conv2d_same,
 236                                       num_outputs=64,
 237                                       kernel_size=7,
 238                                       stride=2,
 239                                       scope='conv1')
 240   batch_norm = utils.get_batch_norm_fn(sync_batch_norm_method)
 241   with tf.variable_scope(scope, 'resnet_v1', [inputs], reuse=reuse) as sc:
 242     end_points_collection = sc.original_name_scope + '_end_points'
 243     with slim.arg_scope([
 244         conv2d_ws.conv2d, bottleneck, lite_bottleneck,
 245         resnet_utils.stack_blocks_dense
 246     ],
 247                         outputs_collections=end_points_collection):
 248       if is_training is not None:
 249         arg_scope = slim.arg_scope([batch_norm], is_training=is_training)
 250       else:
 251         arg_scope = slim.arg_scope([])
 252       with arg_scope:
 253         net = inputs
 254         if output_stride is not None:
 255           if output_stride % 4 != 0:
 256             raise ValueError('The output_stride needs to be a multiple of 4.')
 257           output_stride //= 4
 258         net = root_block_fn(net)
 259         net = slim.max_pool2d(net, 3, stride=2, padding='SAME', scope='pool1')
 260         net = resnet_utils.stack_blocks_dense(net, blocks, output_stride)
 261 
 262         if global_pool:
 263           # Global average pooling.
 264           net = tf.reduce_mean(net, [1, 2], name='pool5', keepdims=True)
 265         if num_classes is not None:
 266           net = conv2d_ws.conv2d(net, num_classes, [1, 1], activation_fn=None,
 267                                  normalizer_fn=None, scope='logits',
 268                                  use_weight_standardization=False)
 269         # Convert end_points_collection into a dictionary of end_points.
 270         end_points = slim.utils.convert_collection_to_dict(
 271             end_points_collection)
 272         if num_classes is not None:
 273           end_points['predictions'] = slim.softmax(net, scope='predictions')
 274         return net, end_points

 277 def resnet_v1_beta_block(scope, base_depth, num_units, stride):
 278   """Helper function for creating a resnet_v1 beta variant bottleneck block.
 279 
 280   Args:
 281     scope: The scope of the block.
 282     base_depth: The depth of the bottleneck layer for each unit.
 283     num_units: The number of units in the block.
 284     stride: The stride of the block, implemented as a stride in the last unit.
 285       All other units have stride=1.
 286 
 287   Returns:
 288     A resnet_v1 bottleneck block.
 289   """
 290   return resnet_utils.Block(scope, bottleneck, [{
 291       'depth': base_depth * 4,
 292       'depth_bottleneck': base_depth,
 293       'stride': 1,
 294       'unit_rate': 1
 295   }] * (num_units - 1) + [{
 296       'depth': base_depth * 4,
 297       'depth_bottleneck': base_depth,
 298       'stride': stride,
 299       'unit_rate': 1
 300   }])

 303 def resnet_v1_small_beta_block(scope, base_depth, num_units, stride):
 304   """Helper function for creating a resnet_18 beta variant bottleneck block.
 305 
 306   Args:
 307     scope: The scope of the block.
 308     base_depth: The depth of the bottleneck layer for each unit.
 309     num_units: The number of units in the block.
 310     stride: The stride of the block, implemented as a stride in the last unit.
 311       All other units have stride=1.
 312 
 313   Returns:
 314     A resnet_18 bottleneck block.
 315   """
 316   block_args = []
 317   for _ in range(num_units - 1):
 318     block_args.append({'depth': base_depth, 'stride': 1, 'unit_rate': 1})
 319   block_args.append({'depth': base_depth, 'stride': stride, 'unit_rate': 1})
 320   return resnet_utils.Block(scope, lite_bottleneck, block_args)

 323 def resnet_v1_18(inputs,
 324                  num_classes=None,
 325                  is_training=None,
 326                  global_pool=False,
 327                  output_stride=None,
 328                  multi_grid=None,
 329                  reuse=None,
 330                  scope='resnet_v1_18',
 331                  sync_batch_norm_method='None'):
 332   """Resnet v1 18.
 333 
 334   Args:
 335     inputs: A tensor of size [batch, height_in, width_in, channels].
 336     num_classes: Number of predicted classes for classification tasks. If None
 337       we return the features before the logit layer.
 338     is_training: Enable/disable is_training for batch normalization.
 339     global_pool: If True, we perform global average pooling before computing the
 340       logits. Set to True for image classification, False for dense prediction.
 341     output_stride: If None, then the output will be computed at the nominal
 342       network stride. If output_stride is not None, it specifies the requested
 343       ratio of input to output spatial resolution.
 344     multi_grid: Employ a hierarchy of different atrous rates within network.
 345     reuse: whether or not the network and its variables should be reused. To be
 346       able to reuse 'scope' must be given.
 347     scope: Optional variable_scope.
 348     sync_batch_norm_method: String, sync batchnorm method.
 349 
 350   Returns:
 351     net: A rank-4 tensor of size [batch, height_out, width_out, channels_out].
 352       If global_pool is False, then height_out and width_out are reduced by a
 353       factor of output_stride compared to the respective height_in and width_in,
 354       else both height_out and width_out equal one. If num_classes is None, then
 355       net is the output of the last ResNet block, potentially after global
 356       average pooling. If num_classes is not None, net contains the pre-softmax
 357       activations.
 358     end_points: A dictionary from components of the network to the corresponding
 359       activation.
 360 
 361   Raises:
 362     ValueError: if multi_grid is not None and does not have length = 3.
 363   """
 364   if multi_grid is None:
 365     multi_grid = _DEFAULT_MULTI_GRID_RESNET_18
 366   else:
 367     if len(multi_grid) != 2:
 368       raise ValueError('Expect multi_grid to have length 2.')
 369 
 370   block4_args = []
 371   for rate in multi_grid:
 372     block4_args.append({'depth': 512, 'stride': 1, 'unit_rate': rate})
 373 
 374   blocks = [
 375       resnet_v1_small_beta_block(
 376           'block1', base_depth=64, num_units=2, stride=2),
 377       resnet_v1_small_beta_block(
 378           'block2', base_depth=128, num_units=2, stride=2),
 379       resnet_v1_small_beta_block(
 380           'block3', base_depth=256, num_units=2, stride=2),
 381       resnet_utils.Block('block4', lite_bottleneck, block4_args),
 382   ]
 383   return resnet_v1_beta(
 384       inputs,
 385       blocks=blocks,
 386       num_classes=num_classes,
 387       is_training=is_training,
 388       global_pool=global_pool,
 389       output_stride=output_stride,
 390       reuse=reuse,
 391       scope=scope,
 392       sync_batch_norm_method=sync_batch_norm_method)

 766 def resnet_arg_scope(weight_decay=0.0001,
 767                      batch_norm_decay=0.997,
 768                      batch_norm_epsilon=1e-5,
 769                      batch_norm_scale=True,
 770                      activation_fn=tf.nn.relu,
 771                      use_batch_norm=True,
 772                      sync_batch_norm_method='None',
 773                      normalization_method='unspecified',
 774                      use_weight_standardization=False):
 775   """Defines the default ResNet arg scope.
 776 
 777   Args:
 778     weight_decay: The weight decay to use for regularizing the model.
 779     batch_norm_decay: The moving average decay when estimating layer activation
 780       statistics in batch normalization.
 781     batch_norm_epsilon: Small constant to prevent division by zero when
 782       normalizing activations by their variance in batch normalization.
 783     batch_norm_scale: If True, uses an explicit `gamma` multiplier to scale the
 784       activations in the batch normalization layer.
 785     activation_fn: The activation function which is used in ResNet.
 786     use_batch_norm: Deprecated in favor of normalization_method.
 787     sync_batch_norm_method: String, sync batchnorm method.
 788     normalization_method: String, one of `batch`, `none`, or `group`, to use
 789       batch normalization, no normalization, or group normalization.
 790     use_weight_standardization: Boolean, whether to use weight standardization.
 791 
 792   Returns:
 793     An `arg_scope` to use for the resnet models.
 794   """
 795   batch_norm_params = {
 796       'decay': batch_norm_decay,
 797       'epsilon': batch_norm_epsilon,
 798       'scale': batch_norm_scale,
 799   }
 800   batch_norm = utils.get_batch_norm_fn(sync_batch_norm_method)
 801   if normalization_method == 'batch':
 802     normalizer_fn = batch_norm
 803   elif normalization_method == 'none':
 804     normalizer_fn = None
 805   elif normalization_method == 'group':
 806     normalizer_fn = slim.group_norm
 807   elif normalization_method == 'unspecified':
 808     normalizer_fn = batch_norm if use_batch_norm else None
 809   else:
 810     raise ValueError('Unrecognized normalization_method %s' %
 811                      normalization_method)
 812 
 813   with slim.arg_scope([conv2d_ws.conv2d],
 814                       weights_regularizer=slim.l2_regularizer(weight_decay),
 815                       weights_initializer=slim.variance_scaling_initializer(),
 816                       activation_fn=activation_fn,
 817                       normalizer_fn=normalizer_fn,
 818                       use_weight_standardization=use_weight_standardization):
 819     with slim.arg_scope([batch_norm], **batch_norm_params):
 820       # The following implies padding='SAME' for pool1, which makes feature
 821       # alignment easier for dense prediction tasks. This is also used in
 822       # https://github.com/facebook/fb.resnet.torch. However the accompanying
 823       # code of 'Deep Residual Learning for Image Recognition' uses
 824       # padding='VALID' for pool1. You can switch to that choice by setting
 825       # slim.arg_scope([slim.max_pool2d], padding='VALID').
 826       with slim.arg_scope([slim.max_pool2d], padding='SAME') as arg_sc:
 827         return arg_sc
