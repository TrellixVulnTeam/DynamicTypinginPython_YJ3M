  17 """Converts Cityscapes data to TFRecord file format with Example protos.
  18 
  19 The Cityscapes dataset is expected to have the following directory structure:
  20 
  21   + cityscapes
  22      - build_cityscapes_data.py (current working directiory).
  23      - build_data.py
  24      + cityscapesscripts
  25        + annotation
  26        + evaluation
  27        + helpers
  28        + preparation
  29        + viewer
  30      + gtFine
  31        + train
  32        + val
  33        + test
  34      + leftImg8bit
  35        + train
  36        + val
  37        + test
  38      + tfrecord
  39 
  40 This script converts data into sharded data files and save at tfrecord folder.
  41 
  42 Note that before running this script, the users should (1) register the
  43 Cityscapes dataset website at https://www.cityscapes-dataset.com to
  44 download the dataset, and (2) run the script provided by Cityscapes
  45 `preparation/createTrainIdLabelImgs.py` to generate the training groundtruth.
  46 
  47 Also note that the tensorflow model will be trained with `TrainId' instead
  48 of `EvalId' used on the evaluation server. Thus, the users need to convert
  49 the predicted labels to `EvalId` for evaluation on the server. See the
  50 vis.py for more details.
  51 
  52 The Example proto contains the following fields:
  53 
  54   image/encoded: encoded image content.
  55   image/filename: image filename.
  56   image/format: image file format.
  57   image/height: image height.
  58   image/width: image width.
  59   image/channels: image channels.
  60   image/segmentation/class/encoded: encoded semantic segmentation content.
  61   image/segmentation/class/format: semantic segmentation file format.
  62 """
  63 from __future__ import absolute_import
  64 from __future__ import division
  65 from __future__ import print_function
  66 import glob
  67 import math
  68 import os.path
  69 import re
  70 import sys
  71 import build_data
  72 from six.moves import range
  73 import tensorflow as tf
  74 
  75 FLAGS = tf.app.flags.FLAGS
  76 
  77 tf.app.flags.DEFINE_string('cityscapes_root',
  78                            './cityscapes',
  79                            'Cityscapes dataset root folder.')
  80 
  81 tf.app.flags.DEFINE_string(
  82     'output_dir',
  83     './tfrecord',
  84     'Path to save converted SSTable of TensorFlow examples.')
  85 
  86 
  87 _NUM_SHARDS = 10
  88 
  89 # A map from data type to folder name that saves the data.
  90 _FOLDERS_MAP = {
  91     'image': 'leftImg8bit',
  92     'label': 'gtFine',
  93 }
  94 
  95 # A map from data type to filename postfix.
  96 _POSTFIX_MAP = {
  97     'image': '_leftImg8bit',
  98     'label': '_gtFine_labelTrainIds',
  99 }
 100 
 101 # A map from data type to data format.
 102 _DATA_FORMAT_MAP = {
 103     'image': 'png',
 104     'label': 'png',
 105 }
 106 
 107 # Image file pattern.
 108 _IMAGE_FILENAME_RE = re.compile('(.+)' + _POSTFIX_MAP['image'])
 109 
 110 
 111 def _get_files(data, dataset_split):
 112   """Gets files for the specified data type and dataset split.
 113 
 114   Args:
 115     data: String, desired data ('image' or 'label').
 116     dataset_split: String, dataset split ('train', 'val', 'test')
 117 
 118   Returns:
 119     A list of sorted file names or None when getting label for
 120       test set.
 121   """
 122   if data == 'label' and dataset_split == 'test':
 123     return None
 124   pattern = '*%s.%s' % (_POSTFIX_MAP[data], _DATA_FORMAT_MAP[data])
 125   search_files = os.path.join(
 126       FLAGS.cityscapes_root, _FOLDERS_MAP[data], dataset_split, '*', pattern)
 127   filenames = glob.glob(search_files)
 128   return sorted(filenames)
 129 
 130 
 131 def _convert_dataset(dataset_split):
 132   """Converts the specified dataset split to TFRecord format.
 133 
 134   Args:
 135     dataset_split: The dataset split (e.g., train, val).
 136 
 137   Raises:
 138     RuntimeError: If loaded image and label have different shape, or if the
 139       image file with specified postfix could not be found.
 140   """
 141   image_files = _get_files('image', dataset_split)
 142   label_files = _get_files('label', dataset_split)
 143 
 144   num_images = len(image_files)
 145   num_per_shard = int(math.ceil(num_images / _NUM_SHARDS))
 146 
 147   image_reader = build_data.ImageReader('png', channels=3)
 148   label_reader = build_data.ImageReader('png', channels=1)
 149 
 150   for shard_id in range(_NUM_SHARDS):
 151     shard_filename = '%s-%05d-of-%05d.tfrecord' % (
 152         dataset_split, shard_id, _NUM_SHARDS)
 153     output_filename = os.path.join(FLAGS.output_dir, shard_filename)
 154     with tf.python_io.TFRecordWriter(output_filename) as tfrecord_writer:
 155       start_idx = shard_id * num_per_shard
 156       end_idx = min((shard_id + 1) * num_per_shard, num_images)
 157       for i in range(start_idx, end_idx):
 158         sys.stdout.write('\r>> Converting image %d/%d shard %d' % (
 159             i + 1, num_images, shard_id))
 160         sys.stdout.flush()
 161         # Read the image.
 162         image_data = tf.gfile.FastGFile(image_files[i], 'rb').read()
 163         height, width = image_reader.read_image_dims(image_data)
 164         # Read the semantic segmentation annotation.
 165         seg_data = tf.gfile.FastGFile(label_files[i], 'rb').read()
 166         seg_height, seg_width = label_reader.read_image_dims(seg_data)
 167         if height != seg_height or width != seg_width:
 168           raise RuntimeError('Shape mismatched between image and label.')
 169         # Convert to tf example.
 170         re_match = _IMAGE_FILENAME_RE.search(image_files[i])
 171         if re_match is None:
 172           raise RuntimeError('Invalid image filename: ' + image_files[i])
 173         filename = os.path.basename(re_match.group(1))
 174         example = build_data.image_seg_to_tfexample(
 175             image_data, filename, height, width, seg_data)
 176         tfrecord_writer.write(example.SerializeToString())
 177     sys.stdout.write('\n')
 178     sys.stdout.flush()
 179 
 180 
 181 def main(unused_argv):
 182   # Only support converting 'train' and 'val' sets for now.
 183   for dataset_split in ['train', 'val']:
 184     _convert_dataset(dataset_split)
 185 
 186 
 187 if __name__ == '__main__':
 188   tf.app.run()
