  16 """Wrapper for providing semantic segmentaion data.
  17 
  18 The SegmentationDataset class provides both images and annotations (semantic
  19 segmentation and/or instance segmentation) for TensorFlow. Currently, we
  20 support the following datasets:
  21 
  22 1. PASCAL VOC 2012 (http://host.robots.ox.ac.uk/pascal/VOC/voc2012/).
  23 
  24 PASCAL VOC 2012 semantic segmentation dataset annotates 20 foreground objects
  25 (e.g., bike, person, and so on) and leaves all the other semantic classes as
  26 one background class. The dataset contains 1464, 1449, and 1456 annotated
  27 images for the training, validation and test respectively.
  28 
  29 2. Cityscapes dataset (https://www.cityscapes-dataset.com)
  30 
  31 The Cityscapes dataset contains 19 semantic labels (such as road, person, car,
  32 and so on) for urban street scenes.
  33 
  34 3. ADE20K dataset (http://groups.csail.mit.edu/vision/datasets/ADE20K)
  35 
  36 The ADE20K dataset contains 150 semantic labels both urban street scenes and
  37 indoor scenes.
  38 
  39 References:
  40   M. Everingham, S. M. A. Eslami, L. V. Gool, C. K. I. Williams, J. Winn,
  41   and A. Zisserman, The pascal visual object classes challenge a retrospective.
  42   IJCV, 2014.
  43 
  44   M. Cordts, M. Omran, S. Ramos, T. Rehfeld, M. Enzweiler, R. Benenson,
  45   U. Franke, S. Roth, and B. Schiele, "The cityscapes dataset for semantic urban
  46   scene understanding," In Proc. of CVPR, 2016.
  47 
  48   B. Zhou, H. Zhao, X. Puig, S. Fidler, A. Barriuso, A. Torralba, "Scene Parsing
  49   through ADE20K dataset", In Proc. of CVPR, 2017.
  50 """
  51

  58 # Named tuple to describe the dataset properties.
  59 DatasetDescriptor = collections.namedtuple(
  60     'DatasetDescriptor',
  61     [
  62         'splits_to_sizes',  # Splits of the dataset into training, val and test.
  63         'num_classes',  # Number of semantic classes, including the
  64                         # background class (if exists). For example, there
  65                         # are 20 foreground classes + 1 background class in
  66                         # the PASCAL VOC 2012 dataset. Thus, we set
  67                         # num_classes=21.
  68         'ignore_label',  # Ignore label value.
  69     ])
  70 
  71 _CITYSCAPES_INFORMATION = DatasetDescriptor(
  72     splits_to_sizes={'train_fine': 2975,
  73                      'train_coarse': 22973,
  74                      'trainval_fine': 3475,
  75                      'trainval_coarse': 23473,
  76                      'val_fine': 500,
  77                      'test_fine': 1525},
  78     num_classes=19,
  79     ignore_label=255,
  80 )
  81 
  82 _PASCAL_VOC_SEG_INFORMATION = DatasetDescriptor(
  83     splits_to_sizes={
  84         'train': 1464,
  85         'train_aug': 10582,
  86         'trainval': 2913,
  87         'val': 1449,
  88     },
  89     num_classes=21,
  90     ignore_label=255,
  91 )
  92 
  93 _ADE20K_INFORMATION = DatasetDescriptor(
  94     splits_to_sizes={
  95         'train': 20210,  # num of samples in images/training
  96         'val': 2000,  # num of samples in images/validation
  97     },
  98     num_classes=151,
  99     ignore_label=0,
 100 )

 103     'cityscapes': _CITYSCAPES_INFORMATION,
 104     'pascal_voc_seg': _PASCAL_VOC_SEG_INFORMATION,

 116 class Dataset(object):
 117   """Represents input dataset for deeplab model."""
 118 
 119   def __init__(self,
 120                dataset_name,
 121                split_name,
 122                dataset_dir,
 123                batch_size,
 124                crop_size,
 125                min_resize_value=None,
 126                max_resize_value=None,
 127                resize_factor=None,
 128                min_scale_factor=1.,
 129                max_scale_factor=1.,
 130                scale_factor_step_size=0,
 131                model_variant=None,
 132                num_readers=1,
 133                is_training=False,
 134                should_shuffle=False,
 135                should_repeat=False):
 136     """Initializes the dataset.
 137 
 138     Args:
 139       dataset_name: Dataset name.
 140       split_name: A train/val Split name.
 141       dataset_dir: The directory of the dataset sources.
 142       batch_size: Batch size.
 143       crop_size: The size used to crop the image and label.
 144       min_resize_value: Desired size of the smaller image side.
 145       max_resize_value: Maximum allowed size of the larger image side.
 146       resize_factor: Resized dimensions are multiple of factor plus one.
 147       min_scale_factor: Minimum scale factor value.
 148       max_scale_factor: Maximum scale factor value.
 149       scale_factor_step_size: The step size from min scale factor to max scale
 150         factor. The input is randomly scaled based on the value of
 151         (min_scale_factor, max_scale_factor, scale_factor_step_size).
 152       model_variant: Model variant (string) for choosing how to mean-subtract
 153         the images. See feature_extractor.network_map for supported model
 154         variants.
 155       num_readers: Number of readers for data provider.
 156       is_training: Boolean, if dataset is for training or not.
 157       should_shuffle: Boolean, if should shuffle the input data.
 158       should_repeat: Boolean, if should repeat the input data.
 159 
 160     Raises:
 161       ValueError: Dataset name and split name are not supported.
 162     """
 163     if dataset_name not in _DATASETS_INFORMATION:
 164       raise ValueError('The specified dataset is not supported yet.')
 165     self.dataset_name = dataset_name
 166 
 167     splits_to_sizes = _DATASETS_INFORMATION[dataset_name].splits_to_sizes
 168 
 169     if split_name not in splits_to_sizes:
 170       raise ValueError('data split name %s not recognized' % split_name)
 171 
 172     if model_variant is None:
 173       tf.logging.warning('Please specify a model_variant. See '
 174                          'feature_extractor.network_map for supported model '
 175                          'variants.')
 176 
 177     self.split_name = split_name
 178     self.dataset_dir = dataset_dir
 179     self.batch_size = batch_size
 180     self.crop_size = crop_size
 181     self.min_resize_value = min_resize_value
 182     self.max_resize_value = max_resize_value
 183     self.resize_factor = resize_factor
 184     self.min_scale_factor = min_scale_factor
 185     self.max_scale_factor = max_scale_factor
 186     self.scale_factor_step_size = scale_factor_step_size
 187     self.model_variant = model_variant
 188     self.num_readers = num_readers
 189     self.is_training = is_training
 190     self.should_shuffle = should_shuffle
 191     self.should_repeat = should_repeat
 192 
 193     self.num_of_classes = _DATASETS_INFORMATION[self.dataset_name].num_classes
 194     self.ignore_label = _DATASETS_INFORMATION[self.dataset_name].ignore_label
 195 
 196   def _parse_function(self, example_proto):
 197     """Function to parse the example proto.
 198 
 199     Args:
 200       example_proto: Proto in the format of tf.Example.
 201 
 202     Returns:
 203       A dictionary with parsed image, label, height, width and image name.
 204 
 205     Raises:
 206       ValueError: Label is of wrong shape.
 207     """
 208 
 209     # Currently only supports jpeg and png.
 210     # Need to use this logic because the shape is not known for
 211     # tf.image.decode_image and we rely on this info to
 212     # extend label if necessary.
 213     def _decode_image(content, channels):
 214       return tf.cond(
 215           tf.image.is_jpeg(content),
 216           lambda: tf.image.decode_jpeg(content, channels),
 217           lambda: tf.image.decode_png(content, channels))
 218 
 219     features = {
 220         'image/encoded':
 221             tf.FixedLenFeature((), tf.string, default_value=''),
 222         'image/filename':
 223             tf.FixedLenFeature((), tf.string, default_value=''),
 224         'image/format':
 225             tf.FixedLenFeature((), tf.string, default_value='jpeg'),
 226         'image/height':
 227             tf.FixedLenFeature((), tf.int64, default_value=0),
 228         'image/width':
 229             tf.FixedLenFeature((), tf.int64, default_value=0),
 230         'image/segmentation/class/encoded':
 231             tf.FixedLenFeature((), tf.string, default_value=''),
 232         'image/segmentation/class/format':
 233             tf.FixedLenFeature((), tf.string, default_value='png'),
 234     }
 235 
 236     parsed_features = tf.parse_single_example(example_proto, features)
 237 
 238     image = _decode_image(parsed_features['image/encoded'], channels=3)
 239 
 240     label = None
 241     if self.split_name != common.TEST_SET:
 242       label = _decode_image(
 243           parsed_features['image/segmentation/class/encoded'], channels=1)
 244 
 245     image_name = parsed_features['image/filename']
 246     if image_name is None:
 247       image_name = tf.constant('')
 248 
 249     sample = {
 250         common.IMAGE: image,
 251         common.IMAGE_NAME: image_name,
 252         common.HEIGHT: parsed_features['image/height'],
 253         common.WIDTH: parsed_features['image/width'],
 254     }
 255 
 256     if label is not None:
 257       if label.get_shape().ndims == 2:
 258         label = tf.expand_dims(label, 2)
 259       elif label.get_shape().ndims == 3 and label.shape.dims[2] == 1:
 260         pass
 261       else:
 262         raise ValueError('Input label shape must be [height, width], or '
 263                          '[height, width, 1].')
 264 
 265       label.set_shape([None, None, 1])
 266 
 267       sample[common.LABELS_CLASS] = label
 268 
 269     return sample
 270 
 271   def _preprocess_image(self, sample):
 272     """Preprocesses the image and label.
 273 
 274     Args:
 275       sample: A sample containing image and label.
 276 
 277     Returns:
 278       sample: Sample with preprocessed image and label.
 279 
 280     Raises:
 281       ValueError: Ground truth label not provided during training.
 282     """
 283     image = sample[common.IMAGE]
 284     label = sample[common.LABELS_CLASS]
 285 
 286     original_image, image, label = input_preprocess.preprocess_image_and_label(
 287         image=image,
 288         label=label,
 289         crop_height=self.crop_size[0],
 290         crop_width=self.crop_size[1],
 291         min_resize_value=self.min_resize_value,
 292         max_resize_value=self.max_resize_value,
 293         resize_factor=self.resize_factor,
 294         min_scale_factor=self.min_scale_factor,
 295         max_scale_factor=self.max_scale_factor,
 296         scale_factor_step_size=self.scale_factor_step_size,
 297         ignore_label=self.ignore_label,
 298         is_training=self.is_training,
 299         model_variant=self.model_variant)
 300 
 301     sample[common.IMAGE] = image
 302 
 303     if not self.is_training:
 304       # Original image is only used during visualization.
 305       sample[common.ORIGINAL_IMAGE] = original_image
 306 
 307     if label is not None:
 308       sample[common.LABEL] = label
 309 
 310     # Remove common.LABEL_CLASS key in the sample since it is only used to
 311     # derive label and not used in training and evaluation.
 312     sample.pop(common.LABELS_CLASS, None)
 313 
 314     return sample
 315 
 316   def get_one_shot_iterator(self):
 317     """Gets an iterator that iterates across the dataset once.
 318 
 319     Returns:
 320       An iterator of type tf.data.Iterator.
 321     """
 322 
 323     files = self._get_all_files()
 324 
 325     dataset = (
 326         tf.data.TFRecordDataset(files, num_parallel_reads=self.num_readers)
 327         .map(self._parse_function, num_parallel_calls=self.num_readers)
 328         .map(self._preprocess_image, num_parallel_calls=self.num_readers))
 329 
 330     if self.should_shuffle:
 331       dataset = dataset.shuffle(buffer_size=100)
 332 
 333     if self.should_repeat:
 334       dataset = dataset.repeat()  # Repeat forever for training.
 335     else:
 336       dataset = dataset.repeat(1)
 337 
 338     dataset = dataset.batch(self.batch_size).prefetch(self.batch_size)
 339     return dataset.make_one_shot_iterator()
 340 
 341   def _get_all_files(self):
 342     """Gets all the files to read data from.
 343 
 344     Returns:
 345       A list of input files.
 346     """
 347     file_pattern = _FILE_PATTERN
 348     file_pattern = os.path.join(self.dataset_dir,
 349                                 file_pattern % self.split_name)
 350     return tf.gfile.Glob(file_pattern)
