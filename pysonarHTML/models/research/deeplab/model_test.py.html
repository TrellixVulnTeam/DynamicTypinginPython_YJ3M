  33   def testBuildDeepLabv2(self):
  34     batch_size = 2
  35     crop_size = [41, 41]
  36 
  37     # Test with two image_pyramids.
  38     image_pyramids = [[1], [0.5, 1]]
  39 
  40     # Test two model variants.
  41     model_variants = ['xception_65', 'mobilenet_v2']
  42 
  43     # Test with two output_types.
  44     outputs_to_num_classes = {'semantic': 3,
  45                               'direction': 2}
  46 
  47     expected_endpoints = [['merged_logits'],
  48                           ['merged_logits',
  49                            'logits_0.50',
  50                            'logits_1.00']]
  51     expected_num_logits = [1, 3]
  52 
  53     for model_variant in model_variants:
  54       model_options = common.ModelOptions(outputs_to_num_classes)._replace(
  55           add_image_level_feature=False,
  56           aspp_with_batch_norm=False,
  57           aspp_with_separable_conv=False,
  58           model_variant=model_variant)
  59 
  60       for i, image_pyramid in enumerate(image_pyramids):
  61         g = tf.Graph()
  62         with g.as_default():
  63           with self.test_session(graph=g):
  64             inputs = tf.random_uniform(
  65                 (batch_size, crop_size[0], crop_size[1], 3))
  66             outputs_to_scales_to_logits = model.multi_scale_logits(
  67                 inputs, model_options, image_pyramid=image_pyramid)
  68 
  69             # Check computed results for each output type.
  70             for output in outputs_to_num_classes:
  71               scales_to_logits = outputs_to_scales_to_logits[output]
  72               self.assertListEqual(sorted(scales_to_logits.keys()),
  73                                    sorted(expected_endpoints[i]))
  74 
  75               # Expected number of logits = len(image_pyramid) + 1, since the
  76               # last logits is merged from all the scales.
  77               self.assertEqual(len(scales_to_logits), expected_num_logits[i])

  79   def testForwardpassDeepLabv3plus(self):
  80     crop_size = [33, 33]
  81     outputs_to_num_classes = {'semantic': 3}
  82 
  83     model_options = common.ModelOptions(
  84         outputs_to_num_classes,
  85         crop_size,
  86         output_stride=16
  87     )._replace(
  88         add_image_level_feature=True,
  89         aspp_with_batch_norm=True,
  90         logits_kernel_size=1,
  91         decoder_output_stride=[4],
  92         model_variant='mobilenet_v2')  # Employ MobileNetv2 for fast test.
  93 
  94     g = tf.Graph()
  95     with g.as_default():
  96       with self.test_session(graph=g) as sess:
  97         inputs = tf.random_uniform(
  98             (1, crop_size[0], crop_size[1], 3))
  99         outputs_to_scales_to_logits = model.multi_scale_logits(
 100             inputs,
 101             model_options,
 102             image_pyramid=[1.0])
 103 
 104         sess.run(tf.global_variables_initializer())
 105         outputs_to_scales_to_logits = sess.run(outputs_to_scales_to_logits)
 106 
 107         # Check computed results for each output type.
 108         for output in outputs_to_num_classes:
 109           scales_to_logits = outputs_to_scales_to_logits[output]
 110           # Expect only one output.
 111           self.assertEqual(len(scales_to_logits), 1)
 112           for logits in scales_to_logits.values():
 113             self.assertTrue(logits.any())

 115   def testBuildDeepLabWithDensePredictionCell(self):
 116     batch_size = 1
 117     crop_size = [33, 33]
 118     outputs_to_num_classes = {'semantic': 2}
 119     expected_endpoints = ['merged_logits']
 120     dense_prediction_cell_config = [
 121         {'kernel': 3, 'rate': [1, 6], 'op': 'conv', 'input': -1},
 122         {'kernel': 3, 'rate': [18, 15], 'op': 'conv', 'input': 0},
 123     ]
 124     model_options = common.ModelOptions(
 125         outputs_to_num_classes,
 126         crop_size,
 127         output_stride=16)._replace(
 128             aspp_with_batch_norm=True,
 129             model_variant='mobilenet_v2',
 130             dense_prediction_cell_config=dense_prediction_cell_config)
 131     g = tf.Graph()
 132     with g.as_default():
 133       with self.test_session(graph=g):
 134         inputs = tf.random_uniform(
 135             (batch_size, crop_size[0], crop_size[1], 3))
 136         outputs_to_scales_to_model_results = model.multi_scale_logits(
 137             inputs,
 138             model_options,
 139             image_pyramid=[1.0])
 140         for output in outputs_to_num_classes:
 141           scales_to_model_results = outputs_to_scales_to_model_results[output]
 142           self.assertListEqual(
 143               list(scales_to_model_results), expected_endpoints)
 144           self.assertEqual(len(scales_to_model_results), 1)
