 174 class ModelOptions(
 175     collections.namedtuple('ModelOptions', [
 176         'outputs_to_num_classes',
 177         'crop_size',
 178         'atrous_rates',
 179         'output_stride',
 180         'preprocessed_images_dtype',
 181         'merge_method',
 182         'add_image_level_feature',
 183         'image_pooling_crop_size',
 184         'image_pooling_stride',
 185         'aspp_with_batch_norm',
 186         'aspp_with_separable_conv',
 187         'multi_grid',
 188         'decoder_output_stride',
 189         'decoder_use_separable_conv',
 190         'logits_kernel_size',
 191         'model_variant',
 192         'depth_multiplier',
 193         'divisible_by',
 194         'prediction_with_upsampled_logits',
 195         'dense_prediction_cell_config',
 196         'nas_architecture_options',
 197         'use_bounded_activation',
 198         'aspp_with_concat_projection',
 199         'aspp_with_squeeze_and_excitation',
 200         'aspp_convs_filters',
 201         'decoder_use_sum_merge',
 202         'decoder_filters',
 203         'decoder_output_is_logits',
 204         'image_se_uses_qsigmoid',
 205         'label_weights',
 206         'sync_batch_norm_method',
 207         'batch_norm_decay',
 208     ])):
 209   """Immutable class to hold model options."""
 210 
 211   __slots__ = ()
 212 
 213   def __new__(cls,
 214               outputs_to_num_classes,
 215               crop_size=None,
 216               atrous_rates=None,
 217               output_stride=8,
 218               preprocessed_images_dtype=tf.float32):
 219     """Constructor to set default values.
 220 
 221     Args:
 222       outputs_to_num_classes: A dictionary from output type to the number of
 223         classes. For example, for the task of semantic segmentation with 21
 224         semantic classes, we would have outputs_to_num_classes['semantic'] = 21.
 225       crop_size: A tuple [crop_height, crop_width].
 226       atrous_rates: A list of atrous convolution rates for ASPP.
 227       output_stride: The ratio of input to output spatial resolution.
 228       preprocessed_images_dtype: The type after the preprocessing function.
 229 
 230     Returns:
 231       A new ModelOptions instance.
 232     """
 233     dense_prediction_cell_config = None
 234     if FLAGS.dense_prediction_cell_json:
 235       with tf.gfile.Open(FLAGS.dense_prediction_cell_json, 'r') as f:
 236         dense_prediction_cell_config = json.load(f)
 237     decoder_output_stride = None
 238     if FLAGS.decoder_output_stride:
 239       decoder_output_stride = [
 240           int(x) for x in FLAGS.decoder_output_stride]
 241       if sorted(decoder_output_stride, reverse=True) != decoder_output_stride:
 242         raise ValueError('Decoder output stride need to be sorted in the '
 243                          'descending order.')
 244     image_pooling_crop_size = None
 245     if FLAGS.image_pooling_crop_size:
 246       image_pooling_crop_size = [int(x) for x in FLAGS.image_pooling_crop_size]
 247     image_pooling_stride = [1, 1]
 248     if FLAGS.image_pooling_stride:
 249       image_pooling_stride = [int(x) for x in FLAGS.image_pooling_stride]
 250     label_weights = FLAGS.label_weights
 251     if label_weights is None:
 252       label_weights = 1.0
 253     nas_architecture_options = {
 254         'nas_stem_output_num_conv_filters': (
 255             FLAGS.nas_stem_output_num_conv_filters),
 256         'nas_use_classification_head': FLAGS.nas_use_classification_head,
 257         'nas_remove_os32_stride': FLAGS.nas_remove_os32_stride,
 258     }
 259     return super(ModelOptions, cls).__new__(
 260         cls, outputs_to_num_classes, crop_size, atrous_rates, output_stride,
 261         preprocessed_images_dtype,
 262         FLAGS.merge_method,
 263         FLAGS.add_image_level_feature,
 264         image_pooling_crop_size,
 265         image_pooling_stride,
 266         FLAGS.aspp_with_batch_norm,
 267         FLAGS.aspp_with_separable_conv,
 268         FLAGS.multi_grid,
 269         decoder_output_stride,
 270         FLAGS.decoder_use_separable_conv,
 271         FLAGS.logits_kernel_size,
 272         FLAGS.model_variant,
 273         FLAGS.depth_multiplier,
 274         FLAGS.divisible_by,
 275         FLAGS.prediction_with_upsampled_logits,
 276         dense_prediction_cell_config,
 277         nas_architecture_options,
 278         FLAGS.use_bounded_activation,
 279         FLAGS.aspp_with_concat_projection,
 280         FLAGS.aspp_with_squeeze_and_excitation,
 281         FLAGS.aspp_convs_filters,
 282         FLAGS.decoder_use_sum_merge,
 283         FLAGS.decoder_filters,
 284         FLAGS.decoder_output_is_logits,
 285         FLAGS.image_se_uses_qsigmoid,
 286         label_weights,
 287         'None',
 288         FLAGS.batch_norm_decay)
 289 
 290   def __deepcopy__(self, memo):
 291     return ModelOptions(copy.deepcopy(self.outputs_to_num_classes),
 292                         self.crop_size,
 293                         self.atrous_rates,
 294                         self.output_stride,
 295                         self.preprocessed_images_dtype)
