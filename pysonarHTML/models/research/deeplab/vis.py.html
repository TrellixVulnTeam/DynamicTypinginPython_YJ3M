 100 # The folder where semantic segmentation predictions are saved.
 101 _SEMANTIC_PREDICTION_SAVE_FOLDER = 'segmentation_results'
 102 
 103 # The folder where raw semantic segmentation predictions are saved.
 104 _RAW_SEMANTIC_PREDICTION_SAVE_FOLDER = 'raw_segmentation_results'
 105 
 106 # The format to save image.
 107 _IMAGE_FORMAT = '%06d_image'
 108 
 109 # The format to save prediction
 110 _PREDICTION_FORMAT = '%06d_prediction'
 111 
 112 # To evaluate Cityscapes results on the evaluation server, the labels used
 113 # during training should be mapped to the labels for evaluation.
 114 _CITYSCAPES_TRAIN_ID_TO_EVAL_ID = [7, 8, 11, 12, 13, 17, 19, 20, 21, 22,
 115                                    23, 24, 25, 26, 27, 28, 31, 32, 33]

 118 def _convert_train_id_to_eval_id(prediction, train_id_to_eval_id):
 119   """Converts the predicted label for evaluation.
 120 
 121   There are cases where the training labels are not equal to the evaluation
 122   labels. This function is used to perform the conversion so that we could
 123   evaluate the results on the evaluation server.
 124 
 125   Args:
 126     prediction: Semantic segmentation prediction.
 127     train_id_to_eval_id: A list mapping from train id to evaluation id.
 128 
 129   Returns:
 130     Semantic segmentation prediction whose labels have been changed.
 131   """
 132   converted_prediction = prediction.copy()
 133   for train_id, eval_id in enumerate(train_id_to_eval_id):
 134     converted_prediction[prediction == train_id] = eval_id
 135 
 136   return converted_prediction

 139 def _process_batch(sess, original_images, semantic_predictions, image_names,
 140                    image_heights, image_widths, image_id_offset, save_dir,
 141                    raw_save_dir, train_id_to_eval_id=None):
 142   """Evaluates one single batch qualitatively.
 143 
 144   Args:
 145     sess: TensorFlow session.
 146     original_images: One batch of original images.
 147     semantic_predictions: One batch of semantic segmentation predictions.
 148     image_names: Image names.
 149     image_heights: Image heights.
 150     image_widths: Image widths.
 151     image_id_offset: Image id offset for indexing images.
 152     save_dir: The directory where the predictions will be saved.
 153     raw_save_dir: The directory where the raw predictions will be saved.
 154     train_id_to_eval_id: A list mapping from train id to eval id.
 155   """
 156   (original_images,
 157    semantic_predictions,
 158    image_names,
 159    image_heights,
 160    image_widths) = sess.run([original_images, semantic_predictions,
 161                              image_names, image_heights, image_widths])
 162 
 163   num_image = semantic_predictions.shape[0]
 164   for i in range(num_image):
 165     image_height = np.squeeze(image_heights[i])
 166     image_width = np.squeeze(image_widths[i])
 167     original_image = np.squeeze(original_images[i])
 168     semantic_prediction = np.squeeze(semantic_predictions[i])
 169     crop_semantic_prediction = semantic_prediction[:image_height, :image_width]
 170 
 171     # Save image.
 172     save_annotation.save_annotation(
 173         original_image, save_dir, _IMAGE_FORMAT % (image_id_offset + i),
 174         add_colormap=False)
 175 
 176     # Save prediction.
 177     save_annotation.save_annotation(
 178         crop_semantic_prediction, save_dir,
 179         _PREDICTION_FORMAT % (image_id_offset + i), add_colormap=True,
 180         colormap_type=FLAGS.colormap_type)
 181 
 182     if FLAGS.also_save_raw_predictions:
 183       image_filename = os.path.basename(image_names[i])
 184 
 185       if train_id_to_eval_id is not None:
 186         crop_semantic_prediction = _convert_train_id_to_eval_id(
 187             crop_semantic_prediction,
 188             train_id_to_eval_id)
 189       save_annotation.save_annotation(
 190           crop_semantic_prediction, raw_save_dir, image_filename,
 191           add_colormap=False)

 194 def main(unused_argv):
 195   tf.logging.set_verbosity(tf.logging.INFO)
 196 
 197   # Get dataset-dependent information.
 198   dataset = data_generator.Dataset(
 199       dataset_name=FLAGS.dataset,
 200       split_name=FLAGS.vis_split,
 201       dataset_dir=FLAGS.dataset_dir,
 202       batch_size=FLAGS.vis_batch_size,
 203       crop_size=[int(sz) for sz in FLAGS.vis_crop_size],
 204       min_resize_value=FLAGS.min_resize_value,
 205       max_resize_value=FLAGS.max_resize_value,
 206       resize_factor=FLAGS.resize_factor,
 207       model_variant=FLAGS.model_variant,
 208       is_training=False,
 209       should_shuffle=False,
 210       should_repeat=False)
 211 
 212   train_id_to_eval_id = None
 213   if dataset.dataset_name == data_generator.get_cityscapes_dataset_name():
 214     tf.logging.info('Cityscapes requires converting train_id to eval_id.')
 215     train_id_to_eval_id = _CITYSCAPES_TRAIN_ID_TO_EVAL_ID
 216 
 217   # Prepare for visualization.
 218   tf.gfile.MakeDirs(FLAGS.vis_logdir)
 219   save_dir = os.path.join(FLAGS.vis_logdir, _SEMANTIC_PREDICTION_SAVE_FOLDER)
 220   tf.gfile.MakeDirs(save_dir)
 221   raw_save_dir = os.path.join(
 222       FLAGS.vis_logdir, _RAW_SEMANTIC_PREDICTION_SAVE_FOLDER)
 223   tf.gfile.MakeDirs(raw_save_dir)
 224 
 225   tf.logging.info('Visualizing on %s set', FLAGS.vis_split)
 226 
 227   with tf.Graph().as_default():
 228     samples = dataset.get_one_shot_iterator().get_next()
 229 
 230     model_options = common.ModelOptions(
 231         outputs_to_num_classes={common.OUTPUT_TYPE: dataset.num_of_classes},
 232         crop_size=[int(sz) for sz in FLAGS.vis_crop_size],
 233         atrous_rates=FLAGS.atrous_rates,
 234         output_stride=FLAGS.output_stride)
 235 
 236     if tuple(FLAGS.eval_scales) == (1.0,):
 237       tf.logging.info('Performing single-scale test.')
 238       predictions = model.predict_labels(
 239           samples[common.IMAGE],
 240           model_options=model_options,
 241           image_pyramid=FLAGS.image_pyramid)
 242     else:
 243       tf.logging.info('Performing multi-scale test.')
 244       if FLAGS.quantize_delay_step >= 0:
 245         raise ValueError(
 246             'Quantize mode is not supported with multi-scale test.')
 247       predictions = model.predict_labels_multi_scale(
 248           samples[common.IMAGE],
 249           model_options=model_options,
 250           eval_scales=FLAGS.eval_scales,
 251           add_flipped_images=FLAGS.add_flipped_images)
 252     predictions = predictions[common.OUTPUT_TYPE]
 253 
 254     if FLAGS.min_resize_value and FLAGS.max_resize_value:
 255       # Only support batch_size = 1, since we assume the dimensions of original
 256       # image after tf.squeeze is [height, width, 3].
 257       assert FLAGS.vis_batch_size == 1
 258 
 259       # Reverse the resizing and padding operations performed in preprocessing.
 260       # First, we slice the valid regions (i.e., remove padded region) and then
 261       # we resize the predictions back.
 262       original_image = tf.squeeze(samples[common.ORIGINAL_IMAGE])
 263       original_image_shape = tf.shape(original_image)
 264       predictions = tf.slice(
 265           predictions,
 266           [0, 0, 0],
 267           [1, original_image_shape[0], original_image_shape[1]])
 268       resized_shape = tf.to_int32([tf.squeeze(samples[common.HEIGHT]),
 269                                    tf.squeeze(samples[common.WIDTH])])
 270       predictions = tf.squeeze(
 271           tf.image.resize_images(tf.expand_dims(predictions, 3),
 272                                  resized_shape,
 273                                  method=tf.image.ResizeMethod.NEAREST_NEIGHBOR,
 274                                  align_corners=True), 3)
 275 
 276     tf.train.get_or_create_global_step()
 277     if FLAGS.quantize_delay_step >= 0:
 278       contrib_quantize.create_eval_graph()
 279 
 280     num_iteration = 0
 281     max_num_iteration = FLAGS.max_number_of_iterations
 282 
 283     checkpoints_iterator = contrib_training.checkpoints_iterator(
 284         FLAGS.checkpoint_dir, min_interval_secs=FLAGS.eval_interval_secs)
 285     for checkpoint_path in checkpoints_iterator:
 286       num_iteration += 1
 287       tf.logging.info(
 288           'Starting visualization at ' + time.strftime('%Y-%m-%d-%H:%M:%S',
 289                                                        time.gmtime()))
 290       tf.logging.info('Visualizing with model %s', checkpoint_path)
 291 
 292       scaffold = tf.train.Scaffold(init_op=tf.global_variables_initializer())
 293       session_creator = tf.train.ChiefSessionCreator(
 294           scaffold=scaffold,
 295           master=FLAGS.master,
 296           checkpoint_filename_with_path=checkpoint_path)
 297       with tf.train.MonitoredSession(
 298           session_creator=session_creator, hooks=None) as sess:
 299         batch = 0
 300         image_id_offset = 0
 301 
 302         while not sess.should_stop():
 303           tf.logging.info('Visualizing batch %d', batch + 1)
 304           _process_batch(sess=sess,
 305                          original_images=samples[common.ORIGINAL_IMAGE],
 306                          semantic_predictions=predictions,
 307                          image_names=samples[common.IMAGE_NAME],
 308                          image_heights=samples[common.HEIGHT],
 309                          image_widths=samples[common.WIDTH],
 310                          image_id_offset=image_id_offset,
 311                          save_dir=save_dir,
 312                          raw_save_dir=raw_save_dir,
 313                          train_id_to_eval_id=train_id_to_eval_id)
 314           image_id_offset += FLAGS.vis_batch_size
 315           batch += 1
 316 
 317       tf.logging.info(
 318           'Finished visualization at ' + time.strftime('%Y-%m-%d-%H:%M:%S',
 319                                                        time.gmtime()))
 320       if max_num_iteration > 0 and num_iteration >= max_num_iteration:
 321         break
 322 
 323 if __name__ == '__main__':
 324   flags.mark_flag_as_required('checkpoint_dir')
 325   flags.mark_flag_as_required('vis_logdir')
 326   flags.mark_flag_as_required('dataset_dir')
 327   tf.app.run()
