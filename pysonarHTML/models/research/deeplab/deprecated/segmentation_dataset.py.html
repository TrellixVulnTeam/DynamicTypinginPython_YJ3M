  17 """Provides data from semantic segmentation datasets.
  18 
  19 The SegmentationDataset class provides both images and annotations (semantic
  20 segmentation and/or instance segmentation) for TensorFlow. Currently, we
  21 support the following datasets:
  22 
  23 1. PASCAL VOC 2012 (http://host.robots.ox.ac.uk/pascal/VOC/voc2012/).
  24 
  25 PASCAL VOC 2012 semantic segmentation dataset annotates 20 foreground objects
  26 (e.g., bike, person, and so on) and leaves all the other semantic classes as
  27 one background class. The dataset contains 1464, 1449, and 1456 annotated
  28 images for the training, validation and test respectively.
  29 
  30 2. Cityscapes dataset (https://www.cityscapes-dataset.com)
  31 
  32 The Cityscapes dataset contains 19 semantic labels (such as road, person, car,
  33 and so on) for urban street scenes.
  34 
  35 3. ADE20K dataset (http://groups.csail.mit.edu/vision/datasets/ADE20K)
  36 
  37 The ADE20K dataset contains 150 semantic labels both urban street scenes and
  38 indoor scenes.
  39 
  40 References:
  41   M. Everingham, S. M. A. Eslami, L. V. Gool, C. K. I. Williams, J. Winn,
  42   and A. Zisserman, The pascal visual object classes challenge a retrospective.
  43   IJCV, 2014.
  44 
  45   M. Cordts, M. Omran, S. Ramos, T. Rehfeld, M. Enzweiler, R. Benenson,
  46   U. Franke, S. Roth, and B. Schiele, "The cityscapes dataset for semantic urban
  47   scene understanding," In Proc. of CVPR, 2016.
  48 
  49   B. Zhou, H. Zhao, X. Puig, S. Fidler, A. Barriuso, A. Torralba, "Scene Parsing
  50   through ADE20K dataset", In Proc. of CVPR, 2017.
  51 """
  52 import collections
  53 import os.path
  54 import tensorflow as tf
  55 from tensorflow.contrib import slim as contrib_slim
  56 
  57 slim = contrib_slim
  58 
  59 dataset = slim.dataset
  60 
  61 tfexample_decoder = slim.tfexample_decoder
  62 
  63 
  64 _ITEMS_TO_DESCRIPTIONS = {
  65     'image': 'A color image of varying height and width.',
  66     'labels_class': ('A semantic segmentation label whose size matches image.'
  67                      'Its values range from 0 (background) to num_classes.'),
  68 }
  69 
  70 # Named tuple to describe the dataset properties.
  71 DatasetDescriptor = collections.namedtuple(
  72     'DatasetDescriptor',
  73     ['splits_to_sizes',   # Splits of the dataset into training, val, and test.
  74      'num_classes',   # Number of semantic classes, including the background
  75                       # class (if exists). For example, there are 20
  76                       # foreground classes + 1 background class in the PASCAL
  77                       # VOC 2012 dataset. Thus, we set num_classes=21.
  78      'ignore_label',  # Ignore label value.
  79     ]
  80 )
  81 
  82 _CITYSCAPES_INFORMATION = DatasetDescriptor(
  83     splits_to_sizes={
  84         'train': 2975,
  85         'val': 500,
  86     },
  87     num_classes=19,
  88     ignore_label=255,
  89 )
  90 
  91 _PASCAL_VOC_SEG_INFORMATION = DatasetDescriptor(
  92     splits_to_sizes={
  93         'train': 1464,
  94         'train_aug': 10582,
  95         'trainval': 2913,
  96         'val': 1449,
  97     },
  98     num_classes=21,
  99     ignore_label=255,
 100 )
 101 
 102 # These number (i.e., 'train'/'test') seems to have to be hard coded
 103 # You are required to figure it out for your training/testing example.
 104 _ADE20K_INFORMATION = DatasetDescriptor(
 105     splits_to_sizes={
 106         'train': 20210,  # num of samples in images/training
 107         'val': 2000,  # num of samples in images/validation
 108     },
 109     num_classes=151,
 110     ignore_label=0,
 111 )
 112 
 113 
 114 _DATASETS_INFORMATION = {
 115     'cityscapes': _CITYSCAPES_INFORMATION,
 116     'pascal_voc_seg': _PASCAL_VOC_SEG_INFORMATION,
 117     'ade20k': _ADE20K_INFORMATION,
 118 }
 119 
 120 # Default file pattern of TFRecord of TensorFlow Example.
 121 _FILE_PATTERN = '%s-*'
 122 
 123 
 124 def get_cityscapes_dataset_name():
 125   return 'cityscapes'

 128 def get_dataset(dataset_name, split_name, dataset_dir):
 129   """Gets an instance of slim Dataset.
 130 
 131   Args:
 132     dataset_name: Dataset name.
 133     split_name: A train/val Split name.
 134     dataset_dir: The directory of the dataset sources.
 135 
 136   Returns:
 137     An instance of slim Dataset.
 138 
 139   Raises:
 140     ValueError: if the dataset_name or split_name is not recognized.
 141   """
 142   if dataset_name not in _DATASETS_INFORMATION:
 143     raise ValueError('The specified dataset is not supported yet.')
 144 
 145   splits_to_sizes = _DATASETS_INFORMATION[dataset_name].splits_to_sizes
 146 
 147   if split_name not in splits_to_sizes:
 148     raise ValueError('data split name %s not recognized' % split_name)
 149 
 150   # Prepare the variables for different datasets.
 151   num_classes = _DATASETS_INFORMATION[dataset_name].num_classes
 152   ignore_label = _DATASETS_INFORMATION[dataset_name].ignore_label
 153 
 154   file_pattern = _FILE_PATTERN
 155   file_pattern = os.path.join(dataset_dir, file_pattern % split_name)
 156 
 157   # Specify how the TF-Examples are decoded.
 158   keys_to_features = {
 159       'image/encoded': tf.FixedLenFeature(
 160           (), tf.string, default_value=''),
 161       'image/filename': tf.FixedLenFeature(
 162           (), tf.string, default_value=''),
 163       'image/format': tf.FixedLenFeature(
 164           (), tf.string, default_value='jpeg'),
 165       'image/height': tf.FixedLenFeature(
 166           (), tf.int64, default_value=0),
 167       'image/width': tf.FixedLenFeature(
 168           (), tf.int64, default_value=0),
 169       'image/segmentation/class/encoded': tf.FixedLenFeature(
 170           (), tf.string, default_value=''),
 171       'image/segmentation/class/format': tf.FixedLenFeature(
 172           (), tf.string, default_value='png'),
 173   }
 174   items_to_handlers = {
 175       'image': tfexample_decoder.Image(
 176           image_key='image/encoded',
 177           format_key='image/format',
 178           channels=3),
 179       'image_name': tfexample_decoder.Tensor('image/filename'),
 180       'height': tfexample_decoder.Tensor('image/height'),
 181       'width': tfexample_decoder.Tensor('image/width'),
 182       'labels_class': tfexample_decoder.Image(
 183           image_key='image/segmentation/class/encoded',
 184           format_key='image/segmentation/class/format',
 185           channels=1),
 186   }
 187 
 188   decoder = tfexample_decoder.TFExampleDecoder(
 189       keys_to_features, items_to_handlers)
 190 
 191   return dataset.Dataset(
 192       data_sources=file_pattern,
 193       reader=tf.TFRecordReader,
 194       decoder=decoder,
 195       num_samples=splits_to_sizes[split_name],
 196       items_to_descriptions=_ITEMS_TO_DESCRIPTIONS,
 197       ignore_label=ignore_label,
 198       num_classes=num_classes,
 199       name=dataset_name,
 200       multi_label=True)
