  45   def test_perfect_match(self):
  46     categories = np.zeros([6, 6], np.uint16)
  47     instances = np.array([
  48         [1, 1, 1, 1, 1, 1],
  49         [1, 2, 2, 2, 2, 1],
  50         [1, 2, 2, 2, 2, 1],
  51         [1, 2, 2, 2, 2, 1],
  52         [1, 2, 2, 1, 1, 1],
  53         [1, 2, 1, 1, 1, 1],
  54     ],
  55                          dtype=np.uint16)
  56 
  57     pq = panoptic_quality.PanopticQuality(
  58         num_categories=1,
  59         ignored_label=2,
  60         max_instances_per_category=16,
  61         offset=16)
  62     pq.compare_and_accumulate(categories, instances, categories, instances)
  63     np.testing.assert_array_equal(pq.iou_per_class, [2.0])
  64     np.testing.assert_array_equal(pq.tp_per_class, [2])
  65     np.testing.assert_array_equal(pq.fn_per_class, [0])
  66     np.testing.assert_array_equal(pq.fp_per_class, [0])
  67     np.testing.assert_array_equal(pq.result_per_category(), [1.0])
  68     self.assertEqual(pq.result(), 1.0)

  70   def test_totally_wrong(self):
  71     det_categories = np.array([
  72         [0, 0, 0, 0, 0, 0],
  73         [0, 1, 0, 0, 1, 0],
  74         [0, 1, 1, 1, 1, 0],
  75         [0, 1, 1, 1, 1, 0],
  76         [0, 0, 0, 0, 0, 0],
  77         [0, 0, 0, 0, 0, 0],
  78     ],
  79                               dtype=np.uint16)
  80     gt_categories = 1 - det_categories
  81     instances = np.zeros([6, 6], np.uint16)

  83     pq = panoptic_quality.PanopticQuality(
  84         num_categories=2,
  85         ignored_label=2,

  91     np.testing.assert_array_equal(pq.tp_per_class, [0, 0])

  93     np.testing.assert_array_equal(pq.fp_per_class, [1, 1])

  97   def test_matches_by_iou(self):
  98     good_det_labels = np.array(
  99         [
 100             [1, 1, 1, 1, 1, 1],
 101             [1, 1, 1, 1, 1, 1],
 102             [1, 2, 2, 2, 2, 1],
 103             [1, 2, 2, 2, 1, 1],
 104             [1, 1, 1, 1, 1, 1],
 105             [1, 1, 1, 1, 1, 1],
 106         ],
 107         dtype=np.uint16)
 108     gt_labels = np.array(
 109         [
 110             [1, 1, 1, 1, 1, 1],
 111             [1, 1, 1, 1, 1, 1],
 112             [1, 1, 2, 2, 2, 1],
 113             [1, 2, 2, 2, 2, 1],
 114             [1, 1, 1, 1, 1, 1],
 115             [1, 1, 1, 1, 1, 1],
 116         ],
 117         dtype=np.uint16)
 118 
 119     pq = panoptic_quality.PanopticQuality(
 120         num_categories=1,
 121         ignored_label=2,
 122         max_instances_per_category=16,
 123         offset=16)
 124     pq.compare_and_accumulate(
 125         np.zeros_like(gt_labels), gt_labels, np.zeros_like(good_det_labels),
 126         good_det_labels)
 127 
 128     # iou(1, 1) = 28/30
 129     # iou(2, 2) = 6/8
 130     np.testing.assert_array_almost_equal(pq.iou_per_class, [28 / 30 + 6 / 8])
 131     np.testing.assert_array_equal(pq.tp_per_class, [2])
 132     np.testing.assert_array_equal(pq.fn_per_class, [0])
 133     np.testing.assert_array_equal(pq.fp_per_class, [0])
 134     self.assertAlmostEqual(pq.result(), (28 / 30 + 6 / 8) / 2)
 135 
 136     bad_det_labels = np.array(
 137         [
 138             [1, 1, 1, 1, 1, 1],
 139             [1, 1, 1, 1, 1, 1],
 140             [1, 1, 1, 2, 2, 1],
 141             [1, 1, 1, 2, 2, 1],
 142             [1, 1, 1, 2, 2, 1],
 143             [1, 1, 1, 1, 1, 1],
 144         ],
 145         dtype=np.uint16)
 146 
 147     pq.reset()
 148     pq.compare_and_accumulate(
 149         np.zeros_like(gt_labels), gt_labels, np.zeros_like(bad_det_labels),
 150         bad_det_labels)
 151 
 152     # iou(1, 1) = 27/32
 153     np.testing.assert_array_almost_equal(pq.iou_per_class, [27 / 32])
 154     np.testing.assert_array_equal(pq.tp_per_class, [1])
 155     np.testing.assert_array_equal(pq.fn_per_class, [1])
 156     np.testing.assert_array_equal(pq.fp_per_class, [1])
 157     self.assertAlmostEqual(pq.result(), (27 / 32) * (1 / 2))

 239   def test_matches_expected(self):
 240     pred_classes = test_utils.read_segmentation_with_rgb_color_map(
 241         'team_pred_class.png', _CLASS_COLOR_MAP)
 242     pred_instances = test_utils.read_test_image(
 243         'team_pred_instance.png', mode='L')
 244 
 245     instance_class_map = {
 246         0: 0,
 247         47: 1,
 248         97: 1,
 249         133: 1,
 250         150: 1,
 251         174: 1,
 252         198: 2,
 253         215: 1,
 254         244: 1,
 255         255: 1,
 256     }
 257     gt_instances, gt_classes = test_utils.panoptic_segmentation_with_class_map(
 258         'team_gt_instance.png', instance_class_map)
 259 
 260     pq = panoptic_quality.PanopticQuality(
 261         num_categories=3,
 262         ignored_label=0,
 263         max_instances_per_category=256,
 264         offset=256 * 256)
 265     pq.compare_and_accumulate(gt_classes, gt_instances, pred_classes,
 266                               pred_instances)
 267     np.testing.assert_array_almost_equal(
 268         pq.iou_per_class, [2.06104, 5.26827, 0.54069], decimal=4)
 269     np.testing.assert_array_equal(pq.tp_per_class, [1, 7, 1])
 270     np.testing.assert_array_equal(pq.fn_per_class, [0, 1, 0])
 271     np.testing.assert_array_equal(pq.fp_per_class, [0, 0, 0])
 272     np.testing.assert_array_almost_equal(pq.result_per_category(),
 273                                          [2.061038, 0.702436, 0.54069])
 274     self.assertAlmostEqual(pq.result(), 0.62156287)

 276   def test_merge_accumulates_all_across_instances(self):
 277     categories = np.zeros([6, 6], np.uint16)
 278     good_det_labels = np.array([
 279         [1, 1, 1, 1, 1, 1],
 280         [1, 1, 1, 1, 1, 1],
 281         [1, 2, 2, 2, 2, 1],
 282         [1, 2, 2, 2, 1, 1],
 283         [1, 1, 1, 1, 1, 1],
 284         [1, 1, 1, 1, 1, 1],
 285     ],
 286                                dtype=np.uint16)
 287     gt_labels = np.array([
 288         [1, 1, 1, 1, 1, 1],
 289         [1, 1, 1, 1, 1, 1],
 290         [1, 1, 2, 2, 2, 1],
 291         [1, 2, 2, 2, 2, 1],
 292         [1, 1, 1, 1, 1, 1],
 293         [1, 1, 1, 1, 1, 1],
 294     ],
 295                          dtype=np.uint16)
 296 
 297     good_pq = panoptic_quality.PanopticQuality(
 298         num_categories=1,
 299         ignored_label=2,
 300         max_instances_per_category=16,
 301         offset=16)
 302     for _ in six.moves.range(2):
 303       good_pq.compare_and_accumulate(categories, gt_labels, categories,
 304                                      good_det_labels)
 305 
 306     bad_det_labels = np.array([
 307         [1, 1, 1, 1, 1, 1],
 308         [1, 1, 1, 1, 1, 1],
 309         [1, 1, 1, 2, 2, 1],
 310         [1, 1, 1, 2, 2, 1],
 311         [1, 1, 1, 2, 2, 1],
 312         [1, 1, 1, 1, 1, 1],
 313     ],
 314                               dtype=np.uint16)
 315 
 316     bad_pq = panoptic_quality.PanopticQuality(
 317         num_categories=1,
 318         ignored_label=2,
 319         max_instances_per_category=16,
 320         offset=16)
 321     for _ in six.moves.range(2):
 322       bad_pq.compare_and_accumulate(categories, gt_labels, categories,
 323                                     bad_det_labels)
 324 
 325     good_pq.merge(bad_pq)
 326 
 327     np.testing.assert_array_almost_equal(
 328         good_pq.iou_per_class, [2 * (28 / 30 + 6 / 8) + 2 * (27 / 32)])
 329     np.testing.assert_array_equal(good_pq.tp_per_class, [2 * 2 + 2])
 330     np.testing.assert_array_equal(good_pq.fn_per_class, [2])
 331     np.testing.assert_array_equal(good_pq.fp_per_class, [2])
 332     self.assertAlmostEqual(good_pq.result(), 0.63177083)
