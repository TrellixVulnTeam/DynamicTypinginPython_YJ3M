  35 def _realdiv_maybe_zero(x, y):
  36   """Support tf.realdiv(x, y) where y may contain zeros."""
  37   return tf.where(tf.less(y, _EPSILON), tf.zeros_like(x), tf.realdiv(x, y))

  40 def _running_total(value, shape, name=None):
  41   """Maintains a running total of tensor `value` between calls."""
  42   with tf.variable_scope(name, 'running_total', [value]):
  43     total_var = tf.get_variable(
  44         'total',
  45         shape,
  46         value.dtype,
  47         initializer=tf.zeros_initializer(),
  48         trainable=False,
  49         collections=[
  50             tf.GraphKeys.LOCAL_VARIABLES, tf.GraphKeys.METRIC_VARIABLES
  51         ])
  52     updated_total = tf.assign_add(total_var, value, use_locking=True)
  53 
  54   return total_var, updated_total

  57 def _panoptic_quality_helper(
  58     groundtruth_category_array, groundtruth_instance_array,
  59     predicted_category_array, predicted_instance_array, num_classes,
  60     max_instances_per_category, ignored_label, offset):
  61   """Helper function to compute panoptic quality."""
  62   pq = panoptic_quality.PanopticQuality(num_classes, ignored_label,
  63                                         max_instances_per_category, offset)
  64   pq.compare_and_accumulate(groundtruth_category_array,
  65                             groundtruth_instance_array,
  66                             predicted_category_array, predicted_instance_array)
  67   return pq.iou_per_class, pq.tp_per_class, pq.fn_per_class, pq.fp_per_class

  70 def streaming_panoptic_quality(groundtruth_categories,
  71                                groundtruth_instances,
  72                                predicted_categories,
  73                                predicted_instances,
  74                                num_classes,
  75                                max_instances_per_category,
  76                                ignored_label,
  77                                offset,
  78                                name=None):
  79   """Aggregates the panoptic metric across calls with different input tensors.
  80 
  81   See tf.metrics.* functions for comparable functionality and usage.
  82 
  83   Args:
  84     groundtruth_categories: A 2D uint16 tensor of groundtruth category labels.
  85     groundtruth_instances: A 2D uint16 tensor of groundtruth instance labels.
  86     predicted_categories: A 2D uint16 tensor of predicted category labels.
  87     predicted_instances: A 2D uint16 tensor of predicted instance labels.
  88     num_classes: Number of classes in the dataset as an integer.
  89     max_instances_per_category: The maximum number of instances for each class
  90       as an integer or integer tensor.
  91     ignored_label: The class id to be ignored in evaluation as an integer or
  92       integer tensor.
  93     offset: The maximum number of unique labels as an integer or integer tensor.
  94     name: An optional variable_scope name.
  95 
  96   Returns:
  97     qualities: A tensor of shape `[6, num_classes]`, where (1) panoptic quality,
  98       (2) segmentation quality, (3) recognition quality, (4) total_tp,
  99       (5) total_fn and (6) total_fp are saved in the respective rows.
 100     update_ops: List of operations that update the running overall panoptic
 101       quality.
 102 
 103   Raises:
 104     RuntimeError: If eager execution is enabled.
 105   """
 106   if tf.executing_eagerly():
 107     raise RuntimeError('Cannot aggregate when eager execution is enabled.')
 108 
 109   input_args = [
 110       tf.convert_to_tensor(groundtruth_categories, tf.uint16),
 111       tf.convert_to_tensor(groundtruth_instances, tf.uint16),
 112       tf.convert_to_tensor(predicted_categories, tf.uint16),
 113       tf.convert_to_tensor(predicted_instances, tf.uint16),
 114       tf.convert_to_tensor(num_classes, tf.int32),
 115       tf.convert_to_tensor(max_instances_per_category, tf.int32),
 116       tf.convert_to_tensor(ignored_label, tf.int32),
 117       tf.convert_to_tensor(offset, tf.int32),
 118   ]
 119   return_types = [
 120       tf.float64,
 121       tf.float64,
 122       tf.float64,
 123       tf.float64,
 124   ]
 125   with tf.variable_scope(name, 'streaming_panoptic_quality', input_args):
 126     panoptic_results = tf.py_func(
 127         _panoptic_quality_helper, input_args, return_types, stateful=False)
 128     iou, tp, fn, fp = tuple(panoptic_results)
 129 
 130     total_iou, updated_iou = _running_total(
 131         iou, [num_classes], name='iou_total')
 132     total_tp, updated_tp = _running_total(tp, [num_classes], name='tp_total')
 133     total_fn, updated_fn = _running_total(fn, [num_classes], name='fn_total')
 134     total_fp, updated_fp = _running_total(fp, [num_classes], name='fp_total')
 135     update_ops = [updated_iou, updated_tp, updated_fn, updated_fp]
 136 
 137     sq = _realdiv_maybe_zero(total_iou, total_tp)
 138     rq = _realdiv_maybe_zero(total_tp,
 139                              total_tp + 0.5 * total_fn + 0.5 * total_fp)
 140     pq = tf.multiply(sq, rq)
 141     qualities = tf.stack([pq, sq, rq, total_tp, total_fn, total_fp], axis=0)
 142   return qualities, update_ops

 145 def _parsing_covering_helper(
 146     groundtruth_category_array, groundtruth_instance_array,
 147     predicted_category_array, predicted_instance_array, num_classes,
 148     max_instances_per_category, ignored_label, offset, normalize_by_image_size):
 149   """Helper function to compute parsing covering."""
 150   pc = parsing_covering.ParsingCovering(num_classes, ignored_label,
 151                                         max_instances_per_category, offset,
 152                                         normalize_by_image_size)
 153   pc.compare_and_accumulate(groundtruth_category_array,
 154                             groundtruth_instance_array,
 155                             predicted_category_array, predicted_instance_array)
 156   return pc.weighted_iou_per_class, pc.gt_area_per_class

 159 def streaming_parsing_covering(groundtruth_categories,
 160                                groundtruth_instances,
 161                                predicted_categories,
 162                                predicted_instances,
 163                                num_classes,
 164                                max_instances_per_category,
 165                                ignored_label,
 166                                offset,
 167                                normalize_by_image_size=True,
 168                                name=None):
 169   """Aggregates the covering across calls with different input tensors.
 170 
 171   See tf.metrics.* functions for comparable functionality and usage.
 172 
 173   Args:
 174     groundtruth_categories: A 2D uint16 tensor of groundtruth category labels.
 175     groundtruth_instances: A 2D uint16 tensor of groundtruth instance labels.
 176     predicted_categories: A 2D uint16 tensor of predicted category labels.
 177     predicted_instances: A 2D uint16 tensor of predicted instance labels.
 178     num_classes: Number of classes in the dataset as an integer.
 179     max_instances_per_category: The maximum number of instances for each class
 180       as an integer or integer tensor.
 181     ignored_label: The class id to be ignored in evaluation as an integer or
 182       integer tensor.
 183     offset: The maximum number of unique labels as an integer or integer tensor.
 184     normalize_by_image_size: Whether to normalize groundtruth region areas by
 185       image size. If True, groundtruth instance areas and weighted IoUs will be
 186       divided by the size of the corresponding image before accumulated across
 187       the dataset.
 188     name: An optional variable_scope name.
 189 
 190   Returns:
 191     coverings: A tensor of shape `[3, num_classes]`, where (1) per class
 192       coverings, (2) per class sum of weighted IoUs, and (3) per class sum of
 193       groundtruth region areas are saved in the perspective rows.
 194     update_ops: List of operations that update the running overall parsing
 195       covering.
 196 
 197   Raises:
 198     RuntimeError: If eager execution is enabled.
 199   """
 200   if tf.executing_eagerly():
 201     raise RuntimeError('Cannot aggregate when eager execution is enabled.')
 202 
 203   input_args = [
 204       tf.convert_to_tensor(groundtruth_categories, tf.uint16),
 205       tf.convert_to_tensor(groundtruth_instances, tf.uint16),
 206       tf.convert_to_tensor(predicted_categories, tf.uint16),
 207       tf.convert_to_tensor(predicted_instances, tf.uint16),
 208       tf.convert_to_tensor(num_classes, tf.int32),
 209       tf.convert_to_tensor(max_instances_per_category, tf.int32),
 210       tf.convert_to_tensor(ignored_label, tf.int32),
 211       tf.convert_to_tensor(offset, tf.int32),
 212       tf.convert_to_tensor(normalize_by_image_size, tf.bool),
 213   ]
 214   return_types = [
 215       tf.float64,
 216       tf.float64,
 217   ]
 218   with tf.variable_scope(name, 'streaming_parsing_covering', input_args):
 219     covering_results = tf.py_func(
 220         _parsing_covering_helper, input_args, return_types, stateful=False)
 221     weighted_iou_per_class, gt_area_per_class = tuple(covering_results)
 222 
 223     total_weighted_iou_per_class, updated_weighted_iou_per_class = (
 224         _running_total(
 225             weighted_iou_per_class, [num_classes],
 226             name='weighted_iou_per_class_total'))
 227     total_gt_area_per_class, updated_gt_area_per_class = _running_total(
 228         gt_area_per_class, [num_classes], name='gt_area_per_class_total')
 229 
 230     covering_per_class = _realdiv_maybe_zero(total_weighted_iou_per_class,
 231                                              total_gt_area_per_class)
 232     coverings = tf.stack([
 233         covering_per_class,
 234         total_weighted_iou_per_class,
 235         total_gt_area_per_class,
 236     ],
 237                          axis=0)
 238     update_ops = [updated_weighted_iou_per_class, updated_gt_area_per_class]
 239 
 240   return coverings, update_ops
