  37   def test_compare_pq_with_reference_eval(self):
  38     sample_data_dir = os.path.join(_TEST_DIR)
  39     gt_json_file = os.path.join(sample_data_dir, 'coco_gt.json')
  40     gt_folder = os.path.join(sample_data_dir, 'coco_gt')
  41     pred_json_file = os.path.join(sample_data_dir, 'coco_pred.json')
  42     pred_folder = os.path.join(sample_data_dir, 'coco_pred')
  43 
  44     panopticapi_results = panopticapi_eval.pq_compute(
  45         gt_json_file, pred_json_file, gt_folder, pred_folder)
  46     deeplab_results = eval_coco_format.eval_coco_format(
  47         gt_json_file,
  48         pred_json_file,
  49         gt_folder,
  50         pred_folder,
  51         metric='pq',
  52         num_categories=7,
  53         ignored_label=0,
  54         max_instances_per_category=256,
  55         intersection_offset=(256 * 256))
  56     self.assertCountEqual(
  57         list(deeplab_results.keys()), ['All', 'Things', 'Stuff'])
  58     for cat_group in ['All', 'Things', 'Stuff']:
  59       self.assertCountEqual(deeplab_results[cat_group], ['pq', 'sq', 'rq', 'n'])
  60       for metric in ['pq', 'sq', 'rq', 'n']:
  61         self.assertAlmostEqual(deeplab_results[cat_group][metric],
  62                                panopticapi_results[cat_group][metric])

  64   def test_compare_pc_with_golden_value(self):
  65     sample_data_dir = os.path.join(_TEST_DIR)
  66     gt_json_file = os.path.join(sample_data_dir, 'coco_gt.json')
  67     gt_folder = os.path.join(sample_data_dir, 'coco_gt')
  68     pred_json_file = os.path.join(sample_data_dir, 'coco_pred.json')
  69     pred_folder = os.path.join(sample_data_dir, 'coco_pred')
  70 
  71     deeplab_results = eval_coco_format.eval_coco_format(
  72         gt_json_file,
  73         pred_json_file,
  74         gt_folder,
  75         pred_folder,
  76         metric='pc',
  77         num_categories=7,
  78         ignored_label=0,
  79         max_instances_per_category=256,
  80         intersection_offset=(256 * 256),
  81         normalize_by_image_size=False)
  82     self.assertCountEqual(
  83         list(deeplab_results.keys()), ['All', 'Things', 'Stuff'])
  84     for cat_group in ['All', 'Things', 'Stuff']:
  85       self.assertCountEqual(deeplab_results[cat_group], ['pc', 'n'])
  86     self.assertAlmostEqual(deeplab_results['All']['pc'], 0.68210561)
  87     self.assertEqual(deeplab_results['All']['n'], 6)
  88     self.assertAlmostEqual(deeplab_results['Things']['pc'], 0.5890529)
  89     self.assertEqual(deeplab_results['Things']['n'], 4)
  90     self.assertAlmostEqual(deeplab_results['Stuff']['pc'], 0.86821097)
  91     self.assertEqual(deeplab_results['Stuff']['n'], 2)
