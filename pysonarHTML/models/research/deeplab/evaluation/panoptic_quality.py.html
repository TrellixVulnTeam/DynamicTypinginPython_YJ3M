  38 def _ids_to_counts(id_array):
  39   """Given a numpy array, a mapping from each unique entry to its count."""
  40   ids, counts = np.unique(id_array, return_counts=True)
  41   return dict(six.moves.zip(ids, counts))

  47   "Panoptic Segmentation" by Alexander Kirillov, Kaiming He, Ross Girshick,

  52   def compare_and_accumulate(
  53       self, groundtruth_category_array, groundtruth_instance_array,
  54       predicted_category_array, predicted_instance_array):
  55     """See base class."""
  56     # First, combine the category and instance labels so that every unique
  57     # value for (category, instance) is assigned a unique integer label.
  58     pred_segment_id = self._naively_combine_labels(predicted_category_array,
  59                                                    predicted_instance_array)
  60     gt_segment_id = self._naively_combine_labels(groundtruth_category_array,
  61                                                  groundtruth_instance_array)
  62 
  63     # Pre-calculate areas for all groundtruth and predicted segments.
  64     gt_segment_areas = _ids_to_counts(gt_segment_id)
  65     pred_segment_areas = _ids_to_counts(pred_segment_id)
  66 
  67     # We assume there is only one void segment and it has instance id = 0.
  68     void_segment_id = self.ignored_label * self.max_instances_per_category
  69 
  70     # There may be other ignored groundtruth segments with instance id > 0, find
  71     # those ids using the unique segment ids extracted with the area computation
  72     # above.
  73     ignored_segment_ids = {
  74         gt_segment_id for gt_segment_id in six.iterkeys(gt_segment_areas)
  75         if (gt_segment_id //
  76             self.max_instances_per_category) == self.ignored_label
  77     }
  78 
  79     # Next, combine the groundtruth and predicted labels. Dividing up the pixels
  80     # based on which groundtruth segment and which predicted segment they belong
  81     # to, this will assign a different 32-bit integer label to each choice
  82     # of (groundtruth segment, predicted segment), encoded as
  83     #   gt_segment_id * offset + pred_segment_id.
  84     intersection_id_array = (
  85         gt_segment_id.astype(np.uint32) * self.offset +
  86         pred_segment_id.astype(np.uint32))
  87 
  88     # For every combination of (groundtruth segment, predicted segment) with a
  89     # non-empty intersection, this counts the number of pixels in that
  90     # intersection.
  91     intersection_areas = _ids_to_counts(intersection_id_array)
  92 
  93     # Helper function that computes the area of the overlap between a predicted
  94     # segment and the ground-truth void/ignored segment.
  95     def prediction_void_overlap(pred_segment_id):
  96       void_intersection_id = void_segment_id * self.offset + pred_segment_id
  97       return intersection_areas.get(void_intersection_id, 0)
  98 
  99     # Compute overall ignored overlap.
 100     def prediction_ignored_overlap(pred_segment_id):
 101       total_ignored_overlap = 0
 102       for ignored_segment_id in ignored_segment_ids:
 103         intersection_id = ignored_segment_id * self.offset + pred_segment_id
 104         total_ignored_overlap += intersection_areas.get(intersection_id, 0)
 105       return total_ignored_overlap
 106 
 107     # Sets that are populated with which segments groundtruth/predicted segments
 108     # have been matched with overlapping predicted/groundtruth segments
 109     # respectively.
 110     gt_matched = set()
 111     pred_matched = set()
 112 
 113     # Calculate IoU per pair of intersecting segments of the same category.
 114     for intersection_id, intersection_area in six.iteritems(intersection_areas):
 115       gt_segment_id = intersection_id // self.offset
 116       pred_segment_id = intersection_id % self.offset
 117 
 118       gt_category = gt_segment_id // self.max_instances_per_category
 119       pred_category = pred_segment_id // self.max_instances_per_category
 120       if gt_category != pred_category:
 121         continue
 122 
 123       # Union between the groundtruth and predicted segments being compared does
 124       # not include the portion of the predicted segment that consists of
 125       # groundtruth "void" pixels.
 126       union = (
 127           gt_segment_areas[gt_segment_id] +
 128           pred_segment_areas[pred_segment_id] - intersection_area -
 129           prediction_void_overlap(pred_segment_id))
 130       iou = intersection_area / union
 131       if iou > 0.5:
 132         self.tp_per_class[gt_category] += 1
 133         self.iou_per_class[gt_category] += iou
 134         gt_matched.add(gt_segment_id)
 135         pred_matched.add(pred_segment_id)
 136 
 137     # Count false negatives for each category.
 138     for gt_segment_id in six.iterkeys(gt_segment_areas):
 139       if gt_segment_id in gt_matched:
 140         continue
 141       category = gt_segment_id // self.max_instances_per_category
 142       # Failing to detect a void segment is not a false negative.
 143       if category == self.ignored_label:
 144         continue
 145       self.fn_per_class[category] += 1
 146 
 147     # Count false positives for each category.
 148     for pred_segment_id in six.iterkeys(pred_segment_areas):
 149       if pred_segment_id in pred_matched:
 150         continue
 151       # A false positive is not penalized if is mostly ignored in the
 152       # groundtruth.
 153       if (prediction_ignored_overlap(pred_segment_id) /
 154           pred_segment_areas[pred_segment_id]) > 0.5:
 155         continue
 156       category = pred_segment_id // self.max_instances_per_category
 157       self.fp_per_class[category] += 1
 158 
 159     return self.result()
 160 
 161   def _valid_categories(self):
 162     """Categories with a "valid" value for the metric, have > 0 instances.
 163 
 164     We will ignore the `ignore_label` class and other classes which have
 165     `tp + fn + fp = 0`.
 166 
 167     Returns:
 168       Boolean array of shape `[num_categories]`.
 169     """
 170     valid_categories = np.not_equal(
 171         self.tp_per_class + self.fn_per_class + self.fp_per_class, 0)
 172     if self.ignored_label >= 0 and self.ignored_label < self.num_categories:
 173       valid_categories[self.ignored_label] = False
 174     return valid_categories
 175 
 176   def detailed_results(self, is_thing=None):
 177     """See base class."""
 178     valid_categories = self._valid_categories()
 179 
 180     # If known, break down which categories are valid _and_ things/stuff.
 181     category_sets = collections.OrderedDict()
 182     category_sets['All'] = valid_categories
 183     if is_thing is not None:
 184       category_sets['Things'] = np.logical_and(valid_categories, is_thing)
 185       category_sets['Stuff'] = np.logical_and(valid_categories,
 186                                               np.logical_not(is_thing))
 187 
 188     # Compute individual per-class metrics that constitute factors of PQ.
 189     sq = base_metric.realdiv_maybe_zero(self.iou_per_class, self.tp_per_class)
 190     rq = base_metric.realdiv_maybe_zero(
 191         self.tp_per_class,
 192         self.tp_per_class + 0.5 * self.fn_per_class + 0.5 * self.fp_per_class)
 193     pq = np.multiply(sq, rq)
 194 
 195     # Assemble detailed results dictionary.
 196     results = {}
 197     for category_set_name, in_category_set in six.iteritems(category_sets):
 198       if np.any(in_category_set):
 199         results[category_set_name] = {
 200             'pq': np.mean(pq[in_category_set]),
 201             'sq': np.mean(sq[in_category_set]),
 202             'rq': np.mean(rq[in_category_set]),
 203             # The number of categories in this subset.
 204             'n': np.sum(in_category_set.astype(np.int32)),
 205         }
 206       else:
 207         results[category_set_name] = {'pq': 0, 'sq': 0, 'rq': 0, 'n': 0}
 208 
 209     return results
 210 
 211   def result_per_category(self):
 212     """See base class."""
 213     sq = base_metric.realdiv_maybe_zero(self.iou_per_class, self.tp_per_class)
 214     rq = base_metric.realdiv_maybe_zero(
 215         self.tp_per_class,
 216         self.tp_per_class + 0.5 * self.fn_per_class + 0.5 * self.fp_per_class)
 217     return np.multiply(sq, rq)
 218 
 219   def print_detailed_results(self, is_thing=None, print_digits=3):
 220     """See base class."""
 221     results = self.detailed_results(is_thing=is_thing)
 222 
 223     tab = prettytable.PrettyTable()
 224 
 225     tab.add_column('', [], align='l')
 226     for fieldname in ['PQ', 'SQ', 'RQ', 'N']:
 227       tab.add_column(fieldname, [], align='r')
 228 
 229     for category_set, subset_results in six.iteritems(results):
 230       data_cols = [
 231           round(subset_results[col_key], print_digits) * 100
 232           for col_key in ['pq', 'sq', 'rq']
 233       ]
 234       data_cols += [subset_results['n']]
 235       tab.add_row([category_set] + data_cols)
 236 
 237     print(tab)
 238 
 239   def result(self):
 240     """See base class."""
 241     pq_per_class = self.result_per_category()
 242     valid_categories = self._valid_categories()
 243     if not np.any(valid_categories):
 244       return 0.
 245     return np.mean(pq_per_class[valid_categories])
 246 
 247   def merge(self, other_instance):
 248     """See base class."""
 249     self.iou_per_class += other_instance.iou_per_class
 250     self.tp_per_class += other_instance.tp_per_class
 251     self.fn_per_class += other_instance.fn_per_class
 252     self.fp_per_class += other_instance.fp_per_class
 253 
 254   def reset(self):
 255     """See base class."""
 256     self.iou_per_class = np.zeros(self.num_categories, dtype=np.float64)
 257     self.tp_per_class = np.zeros(self.num_categories, dtype=np.float64)
 258     self.fn_per_class = np.zeros(self.num_categories, dtype=np.float64)
 259     self.fp_per_class = np.zeros(self.num_categories, dtype=np.float64)
