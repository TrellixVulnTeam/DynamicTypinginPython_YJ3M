  16 """Computes evaluation metrics on groundtruth and predictions in COCO format.
  17 
  18 The Common Objects in Context (COCO) dataset defines a format for specifying
  19 combined semantic and instance segmentations as "panoptic" segmentations. This
  20 is done with the combination of JSON and image files as specified at:
  21 http://cocodataset.org/#format-results
  22 where the JSON file specifies the overall structure of the result,
  23 including the categories for each annotation, and the images specify the image
  24 region for each annotation in that image by its ID.
  25 
  26 This script computes additional metrics such as Parsing Covering on datasets and
  27 predictions in this format. An implementation of Panoptic Quality is also
  28 provided for convenience.
  29 """

  97 def _build_metric(metric,
  98                   num_categories,
  99                   ignored_label,
 100                   max_instances_per_category,
 101                   intersection_offset=None,
 102                   normalize_by_image_size=True):
 103   """Creates a metric aggregator objet of the given name."""
 104   if metric == 'pq':
 105     logging.warning('One should check Panoptic Quality results against the '
 106                     'official COCO API code. Small numerical differences '
 107                     '(< 0.1%) can be magnified by rounding.')
 108     return panoptic_quality.PanopticQuality(num_categories, ignored_label,
 109                                             max_instances_per_category,
 110                                             intersection_offset)
 111   elif metric == 'pc':
 112     return parsing_covering.ParsingCovering(
 113         num_categories, ignored_label, max_instances_per_category,
 114         intersection_offset, normalize_by_image_size)
 115   else:
 116     raise ValueError('No implementation for metric "%s"' % metric)

 119 def _matched_annotations(gt_json, pred_json):
 120   """Yields a set of (groundtruth, prediction) image annotation pairs.."""
 121   image_id_to_pred_ann = {
 122       annotation['image_id']: annotation
 123       for annotation in pred_json['annotations']
 124   }
 125   for gt_ann in gt_json['annotations']:
 126     image_id = gt_ann['image_id']
 127     pred_ann = image_id_to_pred_ann[image_id]
 128     yield gt_ann, pred_ann

 131 def _open_panoptic_id_image(image_path):
 132   """Loads a COCO-format panoptic ID image from file."""
 133   return panopticapi_utils.rgb2id(
 134       np.array(Image.open(image_path), dtype=np.uint32))

 137 def _split_panoptic(ann_json, id_array, ignored_label, allow_crowds):
 138   """Given the COCO JSON and ID map, splits into categories and instances."""
 139   category = np.zeros(id_array.shape, np.uint16)
 140   instance = np.zeros(id_array.shape, np.uint16)
 141   next_instance_id = collections.defaultdict(int)
 142   # Skip instance label 0 for ignored label. That is reserved for void.
 143   next_instance_id[ignored_label] = 1
 144   for segment_info in ann_json['segments_info']:
 145     if allow_crowds and segment_info['iscrowd']:
 146       category_id = ignored_label
 147     else:
 148       category_id = segment_info['category_id']
 149     mask = np.equal(id_array, segment_info['id'])
 150     category[mask] = category_id
 151     instance[mask] = next_instance_id[category_id]
 152     next_instance_id[category_id] += 1
 153   return category, instance

 156 def _category_and_instance_from_annotation(ann_json, folder, ignored_label,
 157                                            allow_crowds):
 158   """Given the COCO JSON annotations, finds maps of categories and instances."""
 159   panoptic_id_image = _open_panoptic_id_image(
 160       os.path.join(folder, ann_json['file_name']))
 161   return _split_panoptic(ann_json, panoptic_id_image, ignored_label,
 162                          allow_crowds)

 165 def _compute_metric(metric_aggregator, gt_folder, pred_folder,
 166                     annotation_pairs):
 167   """Iterates over matched annotation pairs and computes a metric over them."""
 168   for gt_ann, pred_ann in annotation_pairs:
 169     # We only expect "iscrowd" to appear in the ground-truth, and not in model
 170     # output. In predicted JSON it is simply ignored, as done in official code.
 171     gt_category, gt_instance = _category_and_instance_from_annotation(
 172         gt_ann, gt_folder, metric_aggregator.ignored_label, True)
 173     pred_category, pred_instance = _category_and_instance_from_annotation(
 174         pred_ann, pred_folder, metric_aggregator.ignored_label, False)
 175 
 176     metric_aggregator.compare_and_accumulate(gt_category, gt_instance,
 177                                              pred_category, pred_instance)
 178   return metric_aggregator

 181 def _iterate_work_queue(work_queue):
 182   """Creates an iterable that retrieves items from a queue until one is None."""
 183   task = work_queue.get(block=True)
 184   while task is not None:
 185     yield task
 186     task = work_queue.get(block=True)
 187

 189 def _run_metrics_worker(metric_aggregator, gt_folder, pred_folder, work_queue,
 190                         result_queue):
 191   result = _compute_metric(metric_aggregator, gt_folder, pred_folder,
 192                            _iterate_work_queue(work_queue))
 193   result_queue.put(result, block=True)

 196 def _is_thing_array(categories_json, ignored_label):
 197   """is_thing[category_id] is a bool on if category is "thing" or "stuff"."""
 198   is_thing_dict = {}
 199   for category_json in categories_json:
 200     is_thing_dict[category_json['id']] = bool(category_json['isthing'])
 201 
 202   # Check our assumption that the category ids are consecutive.
 203   # Usually metrics should be able to handle this case, but adding a warning
 204   # here.
 205   max_category_id = max(six.iterkeys(is_thing_dict))
 206   if len(is_thing_dict) != max_category_id + 1:
 207     seen_ids = six.viewkeys(is_thing_dict)
 208     all_ids = set(six.moves.range(max_category_id + 1))
 209     unseen_ids = all_ids.difference(seen_ids)
 210     if unseen_ids != {ignored_label}:
 211       logging.warning(
 212           'Nonconsecutive category ids or no category JSON specified for ids: '
 213           '%s', unseen_ids)
 214 
 215   is_thing_array = np.zeros(max_category_id + 1)
 216   for category_id, is_thing in six.iteritems(is_thing_dict):
 217     is_thing_array[category_id] = is_thing
 218 
 219   return is_thing_array

 222 def eval_coco_format(gt_json_file,
 223                      pred_json_file,
 224                      gt_folder=None,
 225                      pred_folder=None,
 226                      metric='pq',
 227                      num_categories=201,
 228                      ignored_label=0,
 229                      max_instances_per_category=256,
 230                      intersection_offset=None,
 231                      normalize_by_image_size=True,
 232                      num_workers=0,
 233                      print_digits=3):
 234   """Top-level code to compute metrics on a COCO-format result.
 235 
 236   Note that the default values are set for COCO panoptic segmentation dataset,
 237   and thus the users may want to change it for their own dataset evaluation.
 238 
 239   Args:
 240     gt_json_file: Path to a JSON file giving ground-truth annotations in COCO
 241       format.
 242     pred_json_file: Path to a JSON file for the predictions to evaluate.
 243     gt_folder: Folder containing panoptic-format ID images to match ground-truth
 244       annotations to image regions.
 245     pred_folder: Folder containing ID images for predictions.
 246     metric: Name of a metric to compute.
 247     num_categories: The number of segmentation categories (or "classes") in the
 248       dataset.
 249     ignored_label: A category id that is ignored in evaluation, e.g. the "void"
 250       label as defined in the COCO panoptic segmentation dataset.
 251     max_instances_per_category: The maximum number of instances for each
 252       category. Used in ensuring unique instance labels.
 253     intersection_offset: The maximum number of unique labels.
 254     normalize_by_image_size: Whether to normalize groundtruth instance region
 255       areas by image size. If True, groundtruth instance areas and weighted IoUs
 256       will be divided by the size of the corresponding image before accumulated
 257       across the dataset. Only used for Parsing Covering (pc) evaluation.
 258     num_workers: If set to a positive number, will spawn child processes to
 259       compute parts of the metric in parallel by splitting the images between
 260       the workers. If set to -1, will use the value of
 261       multiprocessing.cpu_count().
 262     print_digits: Number of significant digits to print in summary of computed
 263       metrics.
 264 
 265   Returns:
 266     The computed result of the metric as a float scalar.
 267   """
 268   with open(gt_json_file, 'r') as gt_json_fo:
 269     gt_json = json.load(gt_json_fo)
 270   with open(pred_json_file, 'r') as pred_json_fo:
 271     pred_json = json.load(pred_json_fo)
 272   if gt_folder is None:
 273     gt_folder = gt_json_file.replace('.json', '')
 274   if pred_folder is None:
 275     pred_folder = pred_json_file.replace('.json', '')
 276   if intersection_offset is None:
 277     intersection_offset = (num_categories + 1) * max_instances_per_category
 278 
 279   metric_aggregator = _build_metric(
 280       metric, num_categories, ignored_label, max_instances_per_category,
 281       intersection_offset, normalize_by_image_size)
 282 
 283   if num_workers == -1:
 284     logging.info('Attempting to get the CPU count to set # workers.')
 285     num_workers = multiprocessing.cpu_count()
 286 
 287   if num_workers > 0:
 288     logging.info('Computing metric in parallel with %d workers.', num_workers)
 289     work_queue = multiprocessing.Queue()
 290     result_queue = multiprocessing.Queue()
 291     workers = []
 292     worker_args = (metric_aggregator, gt_folder, pred_folder, work_queue,
 293                    result_queue)
 294     for _ in six.moves.range(num_workers):
 295       workers.append(
 296           multiprocessing.Process(target=_run_metrics_worker, args=worker_args))
 297     for worker in workers:
 298       worker.start()
 299     for ann_pair in _matched_annotations(gt_json, pred_json):
 300       work_queue.put(ann_pair, block=True)
 301 
 302     # Will cause each worker to return a result and terminate upon recieving a
 303     # None task.
 304     for _ in six.moves.range(num_workers):
 305       work_queue.put(None, block=True)
 306 
 307     # Retrieve results.
 308     for _ in six.moves.range(num_workers):
 309       metric_aggregator.merge(result_queue.get(block=True))
 310 
 311     for worker in workers:
 312       worker.join()
 313   else:
 314     logging.info('Computing metric in a single process.')
 315     annotation_pairs = _matched_annotations(gt_json, pred_json)
 316     _compute_metric(metric_aggregator, gt_folder, pred_folder, annotation_pairs)
 317 
 318   is_thing = _is_thing_array(gt_json['categories'], ignored_label)
 319   metric_aggregator.print_detailed_results(
 320       is_thing=is_thing, print_digits=print_digits)
 321   return metric_aggregator.detailed_results(is_thing=is_thing)
