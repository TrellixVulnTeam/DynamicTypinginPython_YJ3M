  48   def test_streaming_metric_on_single_image(self):
  49     offset = 256 * 256
  50 
  51     instance_class_map = {
  52         0: 0,
  53         47: 1,
  54         97: 1,
  55         133: 1,
  56         150: 1,
  57         174: 1,
  58         198: 2,
  59         215: 1,
  60         244: 1,
  61         255: 1,
  62     }
  63     gt_instances, gt_classes = test_utils.panoptic_segmentation_with_class_map(
  64         'team_gt_instance.png', instance_class_map)
  65 
  66     pred_classes = test_utils.read_segmentation_with_rgb_color_map(
  67         'team_pred_class.png', _CLASS_COLOR_MAP)
  68     pred_instances = test_utils.read_test_image(
  69         'team_pred_instance.png', mode='L')
  70 
  71     gt_class_tensor = tf.placeholder(tf.uint16)
  72     gt_instance_tensor = tf.placeholder(tf.uint16)
  73     pred_class_tensor = tf.placeholder(tf.uint16)
  74     pred_instance_tensor = tf.placeholder(tf.uint16)
  75     qualities, update_pq = streaming_metrics.streaming_panoptic_quality(
  76         gt_class_tensor,
  77         gt_instance_tensor,
  78         pred_class_tensor,
  79         pred_instance_tensor,
  80         num_classes=3,
  81         max_instances_per_category=256,
  82         ignored_label=0,
  83         offset=offset)
  84     pq, sq, rq, total_tp, total_fn, total_fp = tf.unstack(qualities, 6, axis=0)
  85     feed_dict = {
  86         gt_class_tensor: gt_classes,
  87         gt_instance_tensor: gt_instances,
  88         pred_class_tensor: pred_classes,
  89         pred_instance_tensor: pred_instances
  90     }
  91 
  92     with self.session() as sess:
  93       sess.run(tf.local_variables_initializer())
  94       sess.run(update_pq, feed_dict=feed_dict)
  95       (result_pq, result_sq, result_rq, result_total_tp, result_total_fn,
  96        result_total_fp) = sess.run([pq, sq, rq, total_tp, total_fn, total_fp],
  97                                    feed_dict=feed_dict)
  98     np.testing.assert_array_almost_equal(
  99         result_pq, [2.06104, 0.7024, 0.54069], decimal=4)
 100     np.testing.assert_array_almost_equal(
 101         result_sq, [2.06104, 0.7526, 0.54069], decimal=4)
 102     np.testing.assert_array_almost_equal(result_rq, [1., 0.9333, 1.], decimal=4)
 103     np.testing.assert_array_almost_equal(
 104         result_total_tp, [1., 7., 1.], decimal=4)
 105     np.testing.assert_array_almost_equal(
 106         result_total_fn, [0., 1., 0.], decimal=4)
 107     np.testing.assert_array_almost_equal(
 108         result_total_fp, [0., 0., 0.], decimal=4)

 110   def test_streaming_metric_on_multiple_images(self):
 111     num_classes = 7
 112     offset = 256 * 256
 113 
 114     bird_gt_instance_class_map = {
 115         92: 5,
 116         176: 3,
 117         255: 4,
 118     }
 119     cat_gt_instance_class_map = {
 120         0: 0,
 121         255: 6,
 122     }
 123     team_gt_instance_class_map = {
 124         0: 0,
 125         47: 1,
 126         97: 1,
 127         133: 1,
 128         150: 1,
 129         174: 1,
 130         198: 2,
 131         215: 1,
 132         244: 1,
 133         255: 1,
 134     }
 135     test_image = collections.namedtuple(
 136         'TestImage',
 137         ['gt_class_map', 'gt_path', 'pred_inst_path', 'pred_class_path'])
 138     test_images = [
 139         test_image(bird_gt_instance_class_map, 'bird_gt.png',
 140                    'bird_pred_instance.png', 'bird_pred_class.png'),
 141         test_image(cat_gt_instance_class_map, 'cat_gt.png',
 142                    'cat_pred_instance.png', 'cat_pred_class.png'),
 143         test_image(team_gt_instance_class_map, 'team_gt_instance.png',
 144                    'team_pred_instance.png', 'team_pred_class.png'),
 145     ]
 146 
 147     gt_classes = []
 148     gt_instances = []
 149     pred_classes = []
 150     pred_instances = []
 151     for test_image in test_images:
 152       (image_gt_instances,
 153        image_gt_classes) = test_utils.panoptic_segmentation_with_class_map(
 154            test_image.gt_path, test_image.gt_class_map)
 155       gt_classes.append(image_gt_classes)
 156       gt_instances.append(image_gt_instances)
 157 
 158       pred_classes.append(
 159           test_utils.read_segmentation_with_rgb_color_map(
 160               test_image.pred_class_path, _CLASS_COLOR_MAP))
 161       pred_instances.append(
 162           test_utils.read_test_image(test_image.pred_inst_path, mode='L'))
 163 
 164     gt_class_tensor = tf.placeholder(tf.uint16)
 165     gt_instance_tensor = tf.placeholder(tf.uint16)
 166     pred_class_tensor = tf.placeholder(tf.uint16)
 167     pred_instance_tensor = tf.placeholder(tf.uint16)
 168     qualities, update_pq = streaming_metrics.streaming_panoptic_quality(
 169         gt_class_tensor,
 170         gt_instance_tensor,
 171         pred_class_tensor,
 172         pred_instance_tensor,
 173         num_classes=num_classes,
 174         max_instances_per_category=256,
 175         ignored_label=0,
 176         offset=offset)
 177     pq, sq, rq, total_tp, total_fn, total_fp = tf.unstack(qualities, 6, axis=0)
 178     with self.session() as sess:
 179       sess.run(tf.local_variables_initializer())
 180       for pred_class, pred_instance, gt_class, gt_instance in six.moves.zip(
 181           pred_classes, pred_instances, gt_classes, gt_instances):
 182         sess.run(
 183             update_pq,
 184             feed_dict={
 185                 gt_class_tensor: gt_class,
 186                 gt_instance_tensor: gt_instance,
 187                 pred_class_tensor: pred_class,
 188                 pred_instance_tensor: pred_instance
 189             })
 190       (result_pq, result_sq, result_rq, result_total_tp, result_total_fn,
 191        result_total_fp) = sess.run(
 192            [pq, sq, rq, total_tp, total_fn, total_fp],
 193            feed_dict={
 194                gt_class_tensor: 0,
 195                gt_instance_tensor: 0,
 196                pred_class_tensor: 0,
 197                pred_instance_tensor: 0
 198            })
 199     np.testing.assert_array_almost_equal(
 200         result_pq,
 201         [4.3107, 0.7024, 0.54069, 0.745353, 0.85768, 0.99107, 0.77410],
 202         decimal=4)
 203     np.testing.assert_array_almost_equal(
 204         result_sq, [5.3883, 0.7526, 0.5407, 0.7454, 0.8577, 0.9911, 0.7741],
 205         decimal=4)
 206     np.testing.assert_array_almost_equal(
 207         result_rq, [0.8, 0.9333, 1., 1., 1., 1., 1.], decimal=4)
 208     np.testing.assert_array_almost_equal(
 209         result_total_tp, [2., 7., 1., 1., 1., 1., 1.], decimal=4)
 210     np.testing.assert_array_almost_equal(
 211         result_total_fn, [0., 1., 0., 0., 0., 0., 0.], decimal=4)
 212     np.testing.assert_array_almost_equal(
 213         result_total_fp, [1., 0., 0., 0., 0., 0., 0.], decimal=4)
