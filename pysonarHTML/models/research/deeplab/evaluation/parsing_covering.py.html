  16 """Implementation of the Parsing Covering metric.
  17 
  18 Parsing Covering is a region-based metric for evaluating the task of
  19 image parsing, aka panoptic segmentation.
  20 
  21 Please see the paper for details:
  22 "DeeperLab: Single-Shot Image Parser", Tien-Ju Yang, Maxwell D. Collins,
  23 Yukun Zhu, Jyh-Jing Hwang, Ting Liu, Xiao Zhang, Vivienne Sze,
  24 George Papandreou, Liang-Chieh Chen. arXiv: 1902.05093, 2019.
  25 """

  40 class ParsingCovering(base_metric.SegmentationMetric):
  41   r"""Metric class for Parsing Covering.
  42 
  43   Computes segmentation covering metric introduced in (Arbelaez, et al., 2010)
  44   with extension to handle multi-class semantic labels (a.k.a. parsing
  45   covering). Specifically, segmentation covering (SC) is defined in Eq. (8) in
  46   (Arbelaez et al., 2010) as:
  47 
  48   SC(c) = \sum_{R\in S}(|R| * \max_{R'\in S'}O(R,R')) / \sum_{R\in S}|R|,
  49 
  50   where S are the groundtruth instance regions and S' are the predicted
  51   instance regions. The parsing covering is simply:
  52 
  53   PC = \sum_{c=1}^{C}SC(c) / C,
  54 
  55   where C is the number of classes.
  56   """
  57 
  58   def __init__(self,
  59                num_categories,
  60                ignored_label,
  61                max_instances_per_category,
  62                offset,
  63                normalize_by_image_size=True):
  64     """Initialization for ParsingCovering.
  65 
  66     Args:
  67       num_categories: The number of segmentation categories (or "classes" in the
  68         dataset.
  69       ignored_label: A category id that is ignored in evaluation, e.g. the void
  70         label as defined in COCO panoptic segmentation dataset.
  71       max_instances_per_category: The maximum number of instances for each
  72         category. Used in ensuring unique instance labels.
  73       offset: The maximum number of unique labels. This is used, by multiplying
  74         the ground-truth labels, to generate unique ids for individual regions
  75         of overlap between groundtruth and predicted segments.
  76       normalize_by_image_size: Whether to normalize groundtruth instance region
  77         areas by image size. If True, groundtruth instance areas and weighted
  78         IoUs will be divided by the size of the corresponding image before
  79         accumulated across the dataset.
  80     """
  81     super(ParsingCovering, self).__init__(num_categories, ignored_label,
  82                                           max_instances_per_category, offset)
  83     self.normalize_by_image_size = normalize_by_image_size
  84 
  85   def compare_and_accumulate(
  86       self, groundtruth_category_array, groundtruth_instance_array,
  87       predicted_category_array, predicted_instance_array):
  88     """See base class."""
  89     # Allocate intermediate data structures.
  90     max_ious = np.zeros([self.num_categories, self.max_instances_per_category],
  91                         dtype=np.float64)
  92     gt_areas = np.zeros([self.num_categories, self.max_instances_per_category],
  93                         dtype=np.float64)
  94     pred_areas = np.zeros(
  95         [self.num_categories, self.max_instances_per_category],
  96         dtype=np.float64)
  97     # This is a dictionary in the format:
  98     #   {(category, gt_instance): [(pred_instance, intersection_area)]}.
  99     intersections = collections.defaultdict(list)
 100 
 101     # First, combine the category and instance labels so that every unique
 102     # value for (category, instance) is assigned a unique integer label.
 103     pred_segment_id = self._naively_combine_labels(predicted_category_array,
 104                                                    predicted_instance_array)
 105     gt_segment_id = self._naively_combine_labels(groundtruth_category_array,
 106                                                  groundtruth_instance_array)
 107 
 108     # Next, combine the groundtruth and predicted labels. Dividing up the pixels
 109     # based on which groundtruth segment and which predicted segment they belong
 110     # to, this will assign a different 32-bit integer label to each choice
 111     # of (groundtruth segment, predicted segment), encoded as
 112     #   gt_segment_id * offset + pred_segment_id.
 113     intersection_id_array = (
 114         gt_segment_id.astype(np.uint32) * self.offset +
 115         pred_segment_id.astype(np.uint32))
 116 
 117     # For every combination of (groundtruth segment, predicted segment) with a
 118     # non-empty intersection, this counts the number of pixels in that
 119     # intersection.
 120     intersection_ids, intersection_areas = np.unique(
 121         intersection_id_array, return_counts=True)
 122 
 123     # Find areas of all groundtruth and predicted instances, as well as of their
 124     # intersections.
 125     for intersection_id, intersection_area in six.moves.zip(
 126         intersection_ids, intersection_areas):
 127       gt_segment_id = intersection_id // self.offset
 128       gt_category = gt_segment_id // self.max_instances_per_category
 129       if gt_category == self.ignored_label:
 130         continue
 131       gt_instance = gt_segment_id % self.max_instances_per_category
 132       gt_areas[gt_category, gt_instance] += intersection_area
 133 
 134       pred_segment_id = intersection_id % self.offset
 135       pred_category = pred_segment_id // self.max_instances_per_category
 136       pred_instance = pred_segment_id % self.max_instances_per_category
 137       pred_areas[pred_category, pred_instance] += intersection_area
 138       if pred_category != gt_category:
 139         continue
 140 
 141       intersections[gt_category, gt_instance].append((pred_instance,
 142                                                       intersection_area))
 143 
 144     # Find maximum IoU for every groundtruth instance.
 145     for gt_label, instance_intersections in six.iteritems(intersections):
 146       category, gt_instance = gt_label
 147       gt_area = gt_areas[category, gt_instance]
 148       ious = []
 149       for pred_instance, intersection_area in instance_intersections:
 150         pred_area = pred_areas[category, pred_instance]
 151         union = gt_area + pred_area - intersection_area
 152         ious.append(intersection_area / union)
 153       max_ious[category, gt_instance] = max(ious)
 154 
 155     # Normalize groundtruth instance areas by image size if necessary.
 156     if self.normalize_by_image_size:
 157       gt_areas /= groundtruth_category_array.size
 158 
 159     # Compute per-class weighted IoUs and areas summed over all groundtruth
 160     # instances.
 161     self.weighted_iou_per_class += np.sum(max_ious * gt_areas, axis=-1)
 162     self.gt_area_per_class += np.sum(gt_areas, axis=-1)
 163 
 164     return self.result()
 165 
 166   def result_per_category(self):
 167     """See base class."""
 168     return base_metric.realdiv_maybe_zero(self.weighted_iou_per_class,
 169                                           self.gt_area_per_class)
 170 
 171   def _valid_categories(self):
 172     """Categories with a "valid" value for the metric, have > 0 instances.
 173 
 174     We will ignore the `ignore_label` class and other classes which have
 175     groundtruth area of 0.
 176 
 177     Returns:
 178       Boolean array of shape `[num_categories]`.
 179     """
 180     valid_categories = np.not_equal(self.gt_area_per_class, 0)
 181     if self.ignored_label >= 0 and self.ignored_label < self.num_categories:
 182       valid_categories[self.ignored_label] = False
 183     return valid_categories
 184 
 185   def detailed_results(self, is_thing=None):
 186     """See base class."""
 187     valid_categories = self._valid_categories()
 188 
 189     # If known, break down which categories are valid _and_ things/stuff.
 190     category_sets = collections.OrderedDict()
 191     category_sets['All'] = valid_categories
 192     if is_thing is not None:
 193       category_sets['Things'] = np.logical_and(valid_categories, is_thing)
 194       category_sets['Stuff'] = np.logical_and(valid_categories,
 195                                               np.logical_not(is_thing))
 196 
 197     covering_per_class = self.result_per_category()
 198     results = {}
 199     for category_set_name, in_category_set in six.iteritems(category_sets):
 200       if np.any(in_category_set):
 201         results[category_set_name] = {
 202             'pc': np.mean(covering_per_class[in_category_set]),
 203             # The number of valid categories in this subset.
 204             'n': np.sum(in_category_set.astype(np.int32)),
 205         }
 206       else:
 207         results[category_set_name] = {'pc': 0, 'n': 0}
 208 
 209     return results
 210 
 211   def print_detailed_results(self, is_thing=None, print_digits=3):
 212     """See base class."""
 213     results = self.detailed_results(is_thing=is_thing)
 214 
 215     tab = prettytable.PrettyTable()
 216 
 217     tab.add_column('', [], align='l')
 218     for fieldname in ['PC', 'N']:
 219       tab.add_column(fieldname, [], align='r')
 220 
 221     for category_set, subset_results in six.iteritems(results):
 222       data_cols = [
 223           round(subset_results['pc'], print_digits) * 100, subset_results['n']
 224       ]
 225       tab.add_row([category_set] + data_cols)
 226 
 227     print(tab)
 228 
 229   def result(self):
 230     """See base class."""
 231     covering_per_class = self.result_per_category()
 232     valid_categories = self._valid_categories()
 233     if not np.any(valid_categories):
 234       return 0.
 235     return np.mean(covering_per_class[valid_categories])
 236 
 237   def merge(self, other_instance):
 238     """See base class."""
 239     self.weighted_iou_per_class += other_instance.weighted_iou_per_class
 240     self.gt_area_per_class += other_instance.gt_area_per_class
 241 
 242   def reset(self):
 243     """See base class."""
 244     self.weighted_iou_per_class = np.zeros(
 245         self.num_categories, dtype=np.float64)
 246     self.gt_area_per_class = np.zeros(self.num_categories, dtype=np.float64)
