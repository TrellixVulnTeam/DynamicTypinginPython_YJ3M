  30 class TestUtilsTest(absltest.TestCase):
  31 
  32   def test_read_test_image(self):
  33     image_array = test_utils.read_test_image('team_pred_class.png')
  34     self.assertSequenceEqual(image_array.shape, (231, 345, 4))
  35 
  36   def test_reads_segmentation_with_color_map(self):
  37     rgb_to_semantic_label = {(0, 0, 0): 0, (0, 0, 255): 1, (255, 0, 0): 23}
  38     labels = test_utils.read_segmentation_with_rgb_color_map(
  39         'team_pred_class.png', rgb_to_semantic_label)
  40 
  41     input_image = test_utils.read_test_image('team_pred_class.png')
  42     np.testing.assert_array_equal(
  43         labels == 0,
  44         np.logical_and(input_image[:, :, 0] == 0, input_image[:, :, 2] == 0))
  45     np.testing.assert_array_equal(labels == 1, input_image[:, :, 2] == 255)
  46     np.testing.assert_array_equal(labels == 23, input_image[:, :, 0] == 255)
  47 
  48   def test_reads_gt_segmentation(self):
  49     instance_label_to_semantic_label = {
  50         0: 0,
  51         47: 1,
  52         97: 1,
  53         133: 1,
  54         150: 1,
  55         174: 1,
  56         198: 23,
  57         215: 1,
  58         244: 1,
  59         255: 1,
  60     }
  61     instances, classes = test_utils.panoptic_segmentation_with_class_map(
  62         'team_gt_instance.png', instance_label_to_semantic_label)
  63 
  64     expected_label_shape = (231, 345)
  65     self.assertSequenceEqual(instances.shape, expected_label_shape)
  66     self.assertSequenceEqual(classes.shape, expected_label_shape)
  67     np.testing.assert_array_equal(instances == 0, classes == 0)
  68     np.testing.assert_array_equal(instances == 198, classes == 23)
  69     np.testing.assert_array_equal(
  70         np.logical_and(instances != 0, instances != 198), classes == 1)
