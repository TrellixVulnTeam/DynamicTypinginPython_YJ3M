  20 
  21 (1) Atrous convolution to explicitly control the resolution at which
  22 feature responses are computed within Deep Convolutional Neural Networks.
  23 
  24 (2) Atrous spatial pyramid pooling (ASPP) to robustly segment objects at
  25 multiple scales with filters at multiple sampling rates and effective
  26 fields-of-views.
  27

 101 def predict_labels_multi_scale(images,
 102                                model_options,
 103                                eval_scales=(1.0,),
 104                                add_flipped_images=False):
 105   """Predicts segmentation labels.
 106 
 107   Args:
 108     images: A tensor of size [batch, height, width, channels].
 109     model_options: A ModelOptions instance to configure models.
 110     eval_scales: The scales to resize images for evaluation.
 111     add_flipped_images: Add flipped images for evaluation or not.
 112 
 113   Returns:
 114     A dictionary with keys specifying the output_type (e.g., semantic
 115       prediction) and values storing Tensors representing predictions (argmax
 116       over channels). Each prediction has size [batch, height, width].
 117   """
 118   outputs_to_predictions = {
 119       output: []
 120       for output in model_options.outputs_to_num_classes
 121   }
 122 
 123   for i, image_scale in enumerate(eval_scales):
 124     with tf.variable_scope(tf.get_variable_scope(), reuse=True if i else None):
 125       outputs_to_scales_to_logits = multi_scale_logits(
 126           images,
 127           model_options=model_options,
 128           image_pyramid=[image_scale],
 129           is_training=False,
 130           fine_tune_batch_norm=False)
 131 
 132     if add_flipped_images:
 133       with tf.variable_scope(tf.get_variable_scope(), reuse=True):
 134         outputs_to_scales_to_logits_reversed = multi_scale_logits(
 135             tf.reverse_v2(images, [2]),
 136             model_options=model_options,
 137             image_pyramid=[image_scale],
 138             is_training=False,
 139             fine_tune_batch_norm=False)
 140 
 141     for output in sorted(outputs_to_scales_to_logits):
 142       scales_to_logits = outputs_to_scales_to_logits[output]
 143       logits = _resize_bilinear(
 144           scales_to_logits[MERGED_LOGITS_SCOPE],
 145           tf.shape(images)[1:3],
 146           scales_to_logits[MERGED_LOGITS_SCOPE].dtype)
 147       outputs_to_predictions[output].append(
 148           tf.expand_dims(tf.nn.softmax(logits), 4))
 149 
 150       if add_flipped_images:
 151         scales_to_logits_reversed = (
 152             outputs_to_scales_to_logits_reversed[output])
 153         logits_reversed = _resize_bilinear(
 154             tf.reverse_v2(scales_to_logits_reversed[MERGED_LOGITS_SCOPE], [2]),
 155             tf.shape(images)[1:3],
 156             scales_to_logits_reversed[MERGED_LOGITS_SCOPE].dtype)
 157         outputs_to_predictions[output].append(
 158             tf.expand_dims(tf.nn.softmax(logits_reversed), 4))
 159 
 160   for output in sorted(outputs_to_predictions):
 161     predictions = outputs_to_predictions[output]
 162     # Compute average prediction across different scales and flipped images.
 163     predictions = tf.reduce_mean(tf.concat(predictions, 4), axis=4)
 164     outputs_to_predictions[output] = tf.argmax(predictions, 3)
 165     outputs_to_predictions[output + PROB_SUFFIX] = tf.nn.softmax(predictions)
 166 
 167   return outputs_to_predictions

 170 def predict_labels(images, model_options, image_pyramid=None):
 171   """Predicts segmentation labels.
 172 
 173   Args:
 174     images: A tensor of size [batch, height, width, channels].
 175     model_options: A ModelOptions instance to configure models.
 176     image_pyramid: Input image scales for multi-scale feature extraction.
 177 
 178   Returns:
 179     A dictionary with keys specifying the output_type (e.g., semantic
 180       prediction) and values storing Tensors representing predictions (argmax
 181       over channels). Each prediction has size [batch, height, width].
 182   """
 183   outputs_to_scales_to_logits = multi_scale_logits(
 184       images,
 185       model_options=model_options,
 186       image_pyramid=image_pyramid,
 187       is_training=False,
 188       fine_tune_batch_norm=False)
 189 
 190   predictions = {}
 191   for output in sorted(outputs_to_scales_to_logits):
 192     scales_to_logits = outputs_to_scales_to_logits[output]
 193     logits = scales_to_logits[MERGED_LOGITS_SCOPE]
 194     # There are two ways to obtain the final prediction results: (1) bilinear
 195     # upsampling the logits followed by argmax, or (2) argmax followed by
 196     # nearest neighbor upsampling. The second option may introduce the "blocking
 197     # effect" but is computationally efficient.
 198     if model_options.prediction_with_upsampled_logits:
 199       logits = _resize_bilinear(logits,
 200                                 tf.shape(images)[1:3],
 201                                 scales_to_logits[MERGED_LOGITS_SCOPE].dtype)
 202       predictions[output] = tf.argmax(logits, 3)
 203       predictions[output + PROB_SUFFIX] = tf.nn.softmax(logits)
 204     else:
 205       argmax_results = tf.argmax(logits, 3)
 206       argmax_results = tf.image.resize_nearest_neighbor(
 207           tf.expand_dims(argmax_results, 3),
 208           tf.shape(images)[1:3],
 209           align_corners=True,
 210           name='resize_prediction')
 211       predictions[output] = tf.squeeze(argmax_results, 3)
 212       predictions[output + PROB_SUFFIX] = tf.image.resize_bilinear(
 213           tf.nn.softmax(logits),
 214           tf.shape(images)[1:3],
 215           align_corners=True,
 216           name='resize_prob')
 217   return predictions

 220 def multi_scale_logits(images,
 221                        model_options,
 222                        image_pyramid,
 223                        weight_decay=0.0001,
 224                        is_training=False,
 225                        fine_tune_batch_norm=False,
 226                        nas_training_hyper_parameters=None):
 227   """Gets the logits for multi-scale inputs.
 228 
 229   The returned logits are all downsampled (due to max-pooling layers)
 230   for both training and evaluation.
 231 
 232   Args:
 233     images: A tensor of size [batch, height, width, channels].
 234     model_options: A ModelOptions instance to configure models.
 235     image_pyramid: Input image scales for multi-scale feature extraction.
 236     weight_decay: The weight decay for model variables.
 237     is_training: Is training or not.
 238     fine_tune_batch_norm: Fine-tune the batch norm parameters or not.
 239     nas_training_hyper_parameters: A dictionary storing hyper-parameters for
 240       training nas models. Its keys are:
 241       - `drop_path_keep_prob`: Probability to keep each path in the cell when
 242         training.
 243       - `total_training_steps`: Total training steps to help drop path
 244         probability calculation.
 245 
 246   Returns:
 247     outputs_to_scales_to_logits: A map of maps from output_type (e.g.,
 248       semantic prediction) to a dictionary of multi-scale logits names to
 249       logits. For each output_type, the dictionary has keys which
 250       correspond to the scales and values which correspond to the logits.
 251       For example, if `scales` equals [1.0, 1.5], then the keys would
 252       include 'merged_logits', 'logits_1.00' and 'logits_1.50'.
 253 
 254   Raises:
 255     ValueError: If model_options doesn't specify crop_size and its
 256       add_image_level_feature = True, since add_image_level_feature requires
 257       crop_size information.
 258   """
 259   # Setup default values.
 260   if not image_pyramid:
 261     image_pyramid = [1.0]
 262   crop_height = (
 263       model_options.crop_size[0]
 264       if model_options.crop_size else tf.shape(images)[1])
 265   crop_width = (
 266       model_options.crop_size[1]
 267       if model_options.crop_size else tf.shape(images)[2])
 268   if model_options.image_pooling_crop_size:
 269     image_pooling_crop_height = model_options.image_pooling_crop_size[0]
 270     image_pooling_crop_width = model_options.image_pooling_crop_size[1]
 271 
 272   # Compute the height, width for the output logits.
 273   if model_options.decoder_output_stride:
 274     logits_output_stride = min(model_options.decoder_output_stride)
 275   else:
 276     logits_output_stride = model_options.output_stride
 277 
 278   logits_height = scale_dimension(
 279       crop_height,
 280       max(1.0, max(image_pyramid)) / logits_output_stride)
 281   logits_width = scale_dimension(
 282       crop_width,
 283       max(1.0, max(image_pyramid)) / logits_output_stride)
 284 
 285   # Compute the logits for each scale in the image pyramid.
 286   outputs_to_scales_to_logits = {
 287       k: {}
 288       for k in model_options.outputs_to_num_classes
 289   }
 290 
 291   num_channels = images.get_shape().as_list()[-1]
 292 
 293   for image_scale in image_pyramid:
 294     if image_scale != 1.0:
 295       scaled_height = scale_dimension(crop_height, image_scale)
 296       scaled_width = scale_dimension(crop_width, image_scale)
 297       scaled_crop_size = [scaled_height, scaled_width]
 298       scaled_images = _resize_bilinear(images, scaled_crop_size, images.dtype)
 299       if model_options.crop_size:
 300         scaled_images.set_shape(
 301             [None, scaled_height, scaled_width, num_channels])
 302       # Adjust image_pooling_crop_size accordingly.
 303       scaled_image_pooling_crop_size = None
 304       if model_options.image_pooling_crop_size:
 305         scaled_image_pooling_crop_size = [
 306             scale_dimension(image_pooling_crop_height, image_scale),
 307             scale_dimension(image_pooling_crop_width, image_scale)]
 308     else:
 309       scaled_crop_size = model_options.crop_size
 310       scaled_images = images
 311       scaled_image_pooling_crop_size = model_options.image_pooling_crop_size
 312 
 313     updated_options = model_options._replace(
 314         crop_size=scaled_crop_size,
 315         image_pooling_crop_size=scaled_image_pooling_crop_size)
 316     outputs_to_logits = _get_logits(
 317         scaled_images,
 318         updated_options,
 319         weight_decay=weight_decay,
 320         reuse=tf.AUTO_REUSE,
 321         is_training=is_training,
 322         fine_tune_batch_norm=fine_tune_batch_norm,
 323         nas_training_hyper_parameters=nas_training_hyper_parameters)
 324 
 325     # Resize the logits to have the same dimension before merging.
 326     for output in sorted(outputs_to_logits):
 327       outputs_to_logits[output] = _resize_bilinear(
 328           outputs_to_logits[output], [logits_height, logits_width],
 329           outputs_to_logits[output].dtype)
 330 
 331     # Return when only one input scale.
 332     if len(image_pyramid) == 1:
 333       for output in sorted(model_options.outputs_to_num_classes):
 334         outputs_to_scales_to_logits[output][
 335             MERGED_LOGITS_SCOPE] = outputs_to_logits[output]
 336       return outputs_to_scales_to_logits
 337 
 338     # Save logits to the output map.
 339     for output in sorted(model_options.outputs_to_num_classes):
 340       outputs_to_scales_to_logits[output][
 341           'logits_%.2f' % image_scale] = outputs_to_logits[output]
 342 
 343   # Merge the logits from all the multi-scale inputs.
 344   for output in sorted(model_options.outputs_to_num_classes):
 345     # Concatenate the multi-scale logits for each output type.
 346     all_logits = [
 347         tf.expand_dims(logits, axis=4)
 348         for logits in outputs_to_scales_to_logits[output].values()
 349     ]
 350     all_logits = tf.concat(all_logits, 4)
 351     merge_fn = (
 352         tf.reduce_max
 353         if model_options.merge_method == 'max' else tf.reduce_mean)
 354     outputs_to_scales_to_logits[output][MERGED_LOGITS_SCOPE] = merge_fn(
 355         all_logits, axis=4)
 356 
 357   return outputs_to_scales_to_logits

 360 def extract_features(images,
 361                      model_options,
 362                      weight_decay=0.0001,
 363                      reuse=None,
 364                      is_training=False,
 365                      fine_tune_batch_norm=False,
 366                      nas_training_hyper_parameters=None):
 367   """Extracts features by the particular model_variant.
 368 
 369   Args:
 370     images: A tensor of size [batch, height, width, channels].
 371     model_options: A ModelOptions instance to configure models.
 372     weight_decay: The weight decay for model variables.
 373     reuse: Reuse the model variables or not.
 374     is_training: Is training or not.
 375     fine_tune_batch_norm: Fine-tune the batch norm parameters or not.
 376     nas_training_hyper_parameters: A dictionary storing hyper-parameters for
 377       training nas models. Its keys are:
 378       - `drop_path_keep_prob`: Probability to keep each path in the cell when
 379         training.
 380       - `total_training_steps`: Total training steps to help drop path
 381         probability calculation.
 382 
 383   Returns:
 384     concat_logits: A tensor of size [batch, feature_height, feature_width,
 385       feature_channels], where feature_height/feature_width are determined by
 386       the images height/width and output_stride.
 387     end_points: A dictionary from components of the network to the corresponding
 388       activation.
 389   """
 390   features, end_points = feature_extractor.extract_features(
 391       images,
 392       output_stride=model_options.output_stride,
 393       multi_grid=model_options.multi_grid,
 394       model_variant=model_options.model_variant,
 395       depth_multiplier=model_options.depth_multiplier,
 396       divisible_by=model_options.divisible_by,
 397       weight_decay=weight_decay,
 398       reuse=reuse,
 399       is_training=is_training,
 400       preprocessed_images_dtype=model_options.preprocessed_images_dtype,
 401       fine_tune_batch_norm=fine_tune_batch_norm,
 402       nas_architecture_options=model_options.nas_architecture_options,
 403       nas_training_hyper_parameters=nas_training_hyper_parameters,
 404       use_bounded_activation=model_options.use_bounded_activation)
 405 
 406   if not model_options.aspp_with_batch_norm:
 407     return features, end_points
 408   else:
 409     if model_options.dense_prediction_cell_config is not None:
 410       tf.logging.info('Using dense prediction cell config.')
 411       dense_prediction_layer = dense_prediction_cell.DensePredictionCell(
 412           config=model_options.dense_prediction_cell_config,
 413           hparams={
 414               'conv_rate_multiplier': 16 // model_options.output_stride,
 415           })
 416       concat_logits = dense_prediction_layer.build_cell(
 417           features,
 418           output_stride=model_options.output_stride,
 419           crop_size=model_options.crop_size,
 420           image_pooling_crop_size=model_options.image_pooling_crop_size,
 421           weight_decay=weight_decay,
 422           reuse=reuse,
 423           is_training=is_training,
 424           fine_tune_batch_norm=fine_tune_batch_norm)
 425       return concat_logits, end_points
 426     else:
 427       # The following codes employ the DeepLabv3 ASPP module. Note that we
 428       # could express the ASPP module as one particular dense prediction
 429       # cell architecture. We do not do so but leave the following codes
 430       # for backward compatibility.
 431       batch_norm_params = utils.get_batch_norm_params(
 432           decay=0.9997,
 433           epsilon=1e-5,
 434           scale=True,
 435           is_training=(is_training and fine_tune_batch_norm),
 436           sync_batch_norm_method=model_options.sync_batch_norm_method)
 437       batch_norm = utils.get_batch_norm_fn(
 438           model_options.sync_batch_norm_method)
 439       activation_fn = (
 440           tf.nn.relu6 if model_options.use_bounded_activation else tf.nn.relu)
 441       with slim.arg_scope(
 442           [slim.conv2d, slim.separable_conv2d],
 443           weights_regularizer=slim.l2_regularizer(weight_decay),
 444           activation_fn=activation_fn,
 445           normalizer_fn=batch_norm,
 446           padding='SAME',
 447           stride=1,
 448           reuse=reuse):
 449         with slim.arg_scope([batch_norm], **batch_norm_params):
 450           depth = model_options.aspp_convs_filters
 451           branch_logits = []
 452 
 453           if model_options.add_image_level_feature:
 454             if model_options.crop_size is not None:
 455               image_pooling_crop_size = model_options.image_pooling_crop_size
 456               # If image_pooling_crop_size is not specified, use crop_size.
 457               if image_pooling_crop_size is None:
 458                 image_pooling_crop_size = model_options.crop_size
 459               pool_height = scale_dimension(
 460                   image_pooling_crop_size[0],
 461                   1. / model_options.output_stride)
 462               pool_width = scale_dimension(
 463                   image_pooling_crop_size[1],
 464                   1. / model_options.output_stride)
 465               image_feature = slim.avg_pool2d(
 466                   features, [pool_height, pool_width],
 467                   model_options.image_pooling_stride, padding='VALID')
 468               resize_height = scale_dimension(
 469                   model_options.crop_size[0],
 470                   1. / model_options.output_stride)
 471               resize_width = scale_dimension(
 472                   model_options.crop_size[1],
 473                   1. / model_options.output_stride)
 474             else:
 475               # If crop_size is None, we simply do global pooling.
 476               pool_height = tf.shape(features)[1]
 477               pool_width = tf.shape(features)[2]
 478               image_feature = tf.reduce_mean(
 479                   features, axis=[1, 2], keepdims=True)
 480               resize_height = pool_height
 481               resize_width = pool_width
 482             image_feature_activation_fn = tf.nn.relu
 483             image_feature_normalizer_fn = batch_norm
 484             if model_options.aspp_with_squeeze_and_excitation:
 485               image_feature_activation_fn = tf.nn.sigmoid
 486               if model_options.image_se_uses_qsigmoid:
 487                 image_feature_activation_fn = utils.q_sigmoid
 488               image_feature_normalizer_fn = None
 489             image_feature = slim.conv2d(
 490                 image_feature, depth, 1,
 491                 activation_fn=image_feature_activation_fn,
 492                 normalizer_fn=image_feature_normalizer_fn,
 493                 scope=IMAGE_POOLING_SCOPE)
 494             image_feature = _resize_bilinear(
 495                 image_feature,
 496                 [resize_height, resize_width],
 497                 image_feature.dtype)
 498             # Set shape for resize_height/resize_width if they are not Tensor.
 499             if isinstance(resize_height, tf.Tensor):
 500               resize_height = None
 501             if isinstance(resize_width, tf.Tensor):
 502               resize_width = None
 503             image_feature.set_shape([None, resize_height, resize_width, depth])
 504             if not model_options.aspp_with_squeeze_and_excitation:
 505               branch_logits.append(image_feature)
 506 
 507           # Employ a 1x1 convolution.
 508           branch_logits.append(slim.conv2d(features, depth, 1,
 509                                            scope=ASPP_SCOPE + str(0)))
 510 
 511           if model_options.atrous_rates:
 512             # Employ 3x3 convolutions with different atrous rates.
 513             for i, rate in enumerate(model_options.atrous_rates, 1):
 514               scope = ASPP_SCOPE + str(i)
 515               if model_options.aspp_with_separable_conv:
 516                 aspp_features = split_separable_conv2d(
 517                     features,
 518                     filters=depth,
 519                     rate=rate,
 520                     weight_decay=weight_decay,
 521                     scope=scope)
 522               else:
 523                 aspp_features = slim.conv2d(
 524                     features, depth, 3, rate=rate, scope=scope)
 525               branch_logits.append(aspp_features)
 526 
 527           # Merge branch logits.
 528           concat_logits = tf.concat(branch_logits, 3)
 529           if model_options.aspp_with_concat_projection:
 530             concat_logits = slim.conv2d(
 531                 concat_logits, depth, 1, scope=CONCAT_PROJECTION_SCOPE)
 532             concat_logits = slim.dropout(
 533                 concat_logits,
 534                 keep_prob=0.9,
 535                 is_training=is_training,
 536                 scope=CONCAT_PROJECTION_SCOPE + '_dropout')
 537           if (model_options.add_image_level_feature and
 538               model_options.aspp_with_squeeze_and_excitation):
 539             concat_logits *= image_feature
 540 
 541           return concat_logits, end_points

 544 def _get_logits(images,
 545                 model_options,
 546                 weight_decay=0.0001,
 547                 reuse=None,
 548                 is_training=False,
 549                 fine_tune_batch_norm=False,
 550                 nas_training_hyper_parameters=None):
 551   """Gets the logits by atrous/image spatial pyramid pooling.
 552 
 553   Args:
 554     images: A tensor of size [batch, height, width, channels].
 555     model_options: A ModelOptions instance to configure models.
 556     weight_decay: The weight decay for model variables.
 557     reuse: Reuse the model variables or not.
 558     is_training: Is training or not.
 559     fine_tune_batch_norm: Fine-tune the batch norm parameters or not.
 560     nas_training_hyper_parameters: A dictionary storing hyper-parameters for
 561       training nas models. Its keys are:
 562       - `drop_path_keep_prob`: Probability to keep each path in the cell when
 563         training.
 564       - `total_training_steps`: Total training steps to help drop path
 565         probability calculation.
 566 
 567   Returns:
 568     outputs_to_logits: A map from output_type to logits.
 569   """
 570   features, end_points = extract_features(
 571       images,
 572       model_options,
 573       weight_decay=weight_decay,
 574       reuse=reuse,
 575       is_training=is_training,
 576       fine_tune_batch_norm=fine_tune_batch_norm,
 577       nas_training_hyper_parameters=nas_training_hyper_parameters)
 578 
 579   if model_options.decoder_output_stride:
 580     crop_size = model_options.crop_size
 581     if crop_size is None:
 582       crop_size = [tf.shape(images)[1], tf.shape(images)[2]]
 583     features = refine_by_decoder(
 584         features,
 585         end_points,
 586         crop_size=crop_size,
 587         decoder_output_stride=model_options.decoder_output_stride,
 588         decoder_use_separable_conv=model_options.decoder_use_separable_conv,
 589         decoder_use_sum_merge=model_options.decoder_use_sum_merge,
 590         decoder_filters=model_options.decoder_filters,
 591         decoder_output_is_logits=model_options.decoder_output_is_logits,
 592         model_variant=model_options.model_variant,
 593         weight_decay=weight_decay,
 594         reuse=reuse,
 595         is_training=is_training,
 596         fine_tune_batch_norm=fine_tune_batch_norm,
 597         use_bounded_activation=model_options.use_bounded_activation)
 598 
 599   outputs_to_logits = {}
 600   for output in sorted(model_options.outputs_to_num_classes):
 601     if model_options.decoder_output_is_logits:
 602       outputs_to_logits[output] = tf.identity(features,
 603                                               name=output)
 604     else:
 605       outputs_to_logits[output] = get_branch_logits(
 606           features,
 607           model_options.outputs_to_num_classes[output],
 608           model_options.atrous_rates,
 609           aspp_with_batch_norm=model_options.aspp_with_batch_norm,
 610           kernel_size=model_options.logits_kernel_size,
 611           weight_decay=weight_decay,
 612           reuse=reuse,
 613           scope_suffix=output)
 614 
 615   return outputs_to_logits

 618 def refine_by_decoder(features,
 619                       end_points,
 620                       crop_size=None,
 621                       decoder_output_stride=None,
 622                       decoder_use_separable_conv=False,
 623                       decoder_use_sum_merge=False,
 624                       decoder_filters=256,
 625                       decoder_output_is_logits=False,
 626                       model_variant=None,
 627                       weight_decay=0.0001,
 628                       reuse=None,
 629                       is_training=False,
 630                       fine_tune_batch_norm=False,
 631                       use_bounded_activation=False,
 632                       sync_batch_norm_method='None'):
 633   """Adds the decoder to obtain sharper segmentation results.
 634 
 635   Args:
 636     features: A tensor of size [batch, features_height, features_width,
 637       features_channels].
 638     end_points: A dictionary from components of the network to the corresponding
 639       activation.
 640     crop_size: A tuple [crop_height, crop_width] specifying whole patch crop
 641       size.
 642     decoder_output_stride: A list of integers specifying the output stride of
 643       low-level features used in the decoder module.
 644     decoder_use_separable_conv: Employ separable convolution for decoder or not.
 645     decoder_use_sum_merge: Boolean, decoder uses simple sum merge or not.
 646     decoder_filters: Integer, decoder filter size.
 647     decoder_output_is_logits: Boolean, using decoder output as logits or not.
 648     model_variant: Model variant for feature extraction.
 649     weight_decay: The weight decay for model variables.
 650     reuse: Reuse the model variables or not.
 651     is_training: Is training or not.
 652     fine_tune_batch_norm: Fine-tune the batch norm parameters or not.
 653     use_bounded_activation: Whether or not to use bounded activations. Bounded
 654       activations better lend themselves to quantized inference.
 655     sync_batch_norm_method: String, method used to sync batch norm. Currently
 656      only support `None` (no sync batch norm) and `tpu` (use tpu code to
 657      sync batch norm).
 658 
 659   Returns:
 660     Decoder output with size [batch, decoder_height, decoder_width,
 661       decoder_channels].
 662 
 663   Raises:
 664     ValueError: If crop_size is None.
 665   """
 666   if crop_size is None:
 667     raise ValueError('crop_size must be provided when using decoder.')
 668   batch_norm_params = utils.get_batch_norm_params(
 669       decay=0.9997,
 670       epsilon=1e-5,
 671       scale=True,
 672       is_training=(is_training and fine_tune_batch_norm),
 673       sync_batch_norm_method=sync_batch_norm_method)
 674   batch_norm = utils.get_batch_norm_fn(sync_batch_norm_method)
 675   decoder_depth = decoder_filters
 676   projected_filters = 48
 677   if decoder_use_sum_merge:
 678     # When using sum merge, the projected filters must be equal to decoder
 679     # filters.
 680     projected_filters = decoder_filters
 681   if decoder_output_is_logits:
 682     # Overwrite the setting when decoder output is logits.
 683     activation_fn = None
 684     normalizer_fn = None
 685     conv2d_kernel = 1
 686     # Use original conv instead of separable conv.
 687     decoder_use_separable_conv = False
 688   else:
 689     # Default setting when decoder output is not logits.
 690     activation_fn = tf.nn.relu6 if use_bounded_activation else tf.nn.relu
 691     normalizer_fn = batch_norm
 692     conv2d_kernel = 3
 693   with slim.arg_scope(
 694       [slim.conv2d, slim.separable_conv2d],
 695       weights_regularizer=slim.l2_regularizer(weight_decay),
 696       activation_fn=activation_fn,
 697       normalizer_fn=normalizer_fn,
 698       padding='SAME',
 699       stride=1,
 700       reuse=reuse):
 701     with slim.arg_scope([batch_norm], **batch_norm_params):
 702       with tf.variable_scope(DECODER_SCOPE, DECODER_SCOPE, [features]):
 703         decoder_features = features
 704         decoder_stage = 0
 705         scope_suffix = ''
 706         for output_stride in decoder_output_stride:
 707           feature_list = feature_extractor.networks_to_feature_maps[
 708               model_variant][
 709                   feature_extractor.DECODER_END_POINTS][output_stride]
 710           # If only one decoder stage, we do not change the scope name in
 711           # order for backward compactibility.
 712           if decoder_stage:
 713             scope_suffix = '_{}'.format(decoder_stage)
 714           for i, name in enumerate(feature_list):
 715             decoder_features_list = [decoder_features]
 716             # MobileNet and NAS variants use different naming convention.
 717             if ('mobilenet' in model_variant or
 718                 model_variant.startswith('mnas') or
 719                 model_variant.startswith('nas')):
 720               feature_name = name
 721             else:
 722               feature_name = '{}/{}'.format(
 723                   feature_extractor.name_scope[model_variant], name)
 724             decoder_features_list.append(
 725                 slim.conv2d(
 726                     end_points[feature_name],
 727                     projected_filters,
 728                     1,
 729                     scope='feature_projection' + str(i) + scope_suffix))
 730             # Determine the output size.
 731             decoder_height = scale_dimension(crop_size[0], 1.0 / output_stride)
 732             decoder_width = scale_dimension(crop_size[1], 1.0 / output_stride)
 733             # Resize to decoder_height/decoder_width.
 734             for j, feature in enumerate(decoder_features_list):
 735               decoder_features_list[j] = _resize_bilinear(
 736                   feature, [decoder_height, decoder_width], feature.dtype)
 737               h = (None if isinstance(decoder_height, tf.Tensor)
 738                    else decoder_height)
 739               w = (None if isinstance(decoder_width, tf.Tensor)
 740                    else decoder_width)
 741               decoder_features_list[j].set_shape([None, h, w, None])
 742             if decoder_use_sum_merge:
 743               decoder_features = _decoder_with_sum_merge(
 744                   decoder_features_list,
 745                   decoder_depth,
 746                   conv2d_kernel=conv2d_kernel,
 747                   decoder_use_separable_conv=decoder_use_separable_conv,
 748                   weight_decay=weight_decay,
 749                   scope_suffix=scope_suffix)
 750             else:
 751               if not decoder_use_separable_conv:
 752                 scope_suffix = str(i) + scope_suffix
 753               decoder_features = _decoder_with_concat_merge(
 754                   decoder_features_list,
 755                   decoder_depth,
 756                   decoder_use_separable_conv=decoder_use_separable_conv,
 757                   weight_decay=weight_decay,
 758                   scope_suffix=scope_suffix)
 759           decoder_stage += 1
 760         return decoder_features

 763 def _decoder_with_sum_merge(decoder_features_list,
 764                             decoder_depth,
 765                             conv2d_kernel=3,
 766                             decoder_use_separable_conv=True,
 767                             weight_decay=0.0001,
 768                             scope_suffix=''):
 769   """Decoder with sum to merge features.
 770 
 771   Args:
 772     decoder_features_list: A list of decoder features.
 773     decoder_depth: Integer, the filters used in the convolution.
 774     conv2d_kernel: Integer, the convolution kernel size.
 775     decoder_use_separable_conv: Boolean, use separable conv or not.
 776     weight_decay: Weight decay for the model variables.
 777     scope_suffix: String, used in the scope suffix.
 778 
 779   Returns:
 780     decoder features merged with sum.
 781 
 782   Raises:
 783     RuntimeError: If decoder_features_list have length not equal to 2.
 784   """
 785   if len(decoder_features_list) != 2:
 786     raise RuntimeError('Expect decoder_features has length 2.')
 787   # Only apply one convolution when decoder use sum merge.
 788   if decoder_use_separable_conv:
 789     decoder_features = split_separable_conv2d(
 790         decoder_features_list[0],
 791         filters=decoder_depth,
 792         rate=1,
 793         weight_decay=weight_decay,
 794         scope='decoder_split_sep_conv0'+scope_suffix) + decoder_features_list[1]
 795   else:
 796     decoder_features = slim.conv2d(
 797         decoder_features_list[0],
 798         decoder_depth,
 799         conv2d_kernel,
 800         scope='decoder_conv0'+scope_suffix) + decoder_features_list[1]
 801   return decoder_features

 804 def _decoder_with_concat_merge(decoder_features_list,
 805                                decoder_depth,
 806                                decoder_use_separable_conv=True,
 807                                weight_decay=0.0001,
 808                                scope_suffix=''):
 809   """Decoder with concatenation to merge features.
 810 
 811   This decoder method applies two convolutions to smooth the features obtained
 812   by concatenating the input decoder_features_list.
 813 
 814   This decoder module is proposed in the DeepLabv3+ paper.
 815 
 816   Args:
 817     decoder_features_list: A list of decoder features.
 818     decoder_depth: Integer, the filters used in the convolution.
 819     decoder_use_separable_conv: Boolean, use separable conv or not.
 820     weight_decay: Weight decay for the model variables.
 821     scope_suffix: String, used in the scope suffix.
 822 
 823   Returns:
 824     decoder features merged with concatenation.
 825   """
 826   if decoder_use_separable_conv:
 827     decoder_features = split_separable_conv2d(
 828         tf.concat(decoder_features_list, 3),
 829         filters=decoder_depth,
 830         rate=1,
 831         weight_decay=weight_decay,
 832         scope='decoder_conv0'+scope_suffix)
 833     decoder_features = split_separable_conv2d(
 834         decoder_features,
 835         filters=decoder_depth,
 836         rate=1,
 837         weight_decay=weight_decay,
 838         scope='decoder_conv1'+scope_suffix)
 839   else:
 840     num_convs = 2
 841     decoder_features = slim.repeat(
 842         tf.concat(decoder_features_list, 3),
 843         num_convs,
 844         slim.conv2d,
 845         decoder_depth,
 846         3,
 847         scope='decoder_conv'+scope_suffix)
 848   return decoder_features

 851 def get_branch_logits(features,
 852                       num_classes,
 853                       atrous_rates=None,
 854                       aspp_with_batch_norm=False,
 855                       kernel_size=1,
 856                       weight_decay=0.0001,
 857                       reuse=None,
 858                       scope_suffix=''):
 859   """Gets the logits from each model's branch.
 860 
 861   The underlying model is branched out in the last layer when atrous
 862   spatial pyramid pooling is employed, and all branches are sum-merged
 863   to form the final logits.
 864 
 865   Args:
 866     features: A float tensor of shape [batch, height, width, channels].
 867     num_classes: Number of classes to predict.
 868     atrous_rates: A list of atrous convolution rates for last layer.
 869     aspp_with_batch_norm: Use batch normalization layers for ASPP.
 870     kernel_size: Kernel size for convolution.
 871     weight_decay: Weight decay for the model variables.
 872     reuse: Reuse model variables or not.
 873     scope_suffix: Scope suffix for the model variables.
 874 
 875   Returns:
 876     Merged logits with shape [batch, height, width, num_classes].
 877 
 878   Raises:
 879     ValueError: Upon invalid input kernel_size value.
 880   """
 881   # When using batch normalization with ASPP, ASPP has been applied before
 882   # in extract_features, and thus we simply apply 1x1 convolution here.
 883   if aspp_with_batch_norm or atrous_rates is None:
 884     if kernel_size != 1:
 885       raise ValueError('Kernel size must be 1 when atrous_rates is None or '
 886                        'using aspp_with_batch_norm. Gets %d.' % kernel_size)
 887     atrous_rates = [1]
 888 
 889   with slim.arg_scope(
 890       [slim.conv2d],
 891       weights_regularizer=slim.l2_regularizer(weight_decay),
 892       weights_initializer=tf.truncated_normal_initializer(stddev=0.01),
 893       reuse=reuse):
 894     with tf.variable_scope(LOGITS_SCOPE_NAME, LOGITS_SCOPE_NAME, [features]):
 895       branch_logits = []
 896       for i, rate in enumerate(atrous_rates):
 897         scope = scope_suffix
 898         if i:
 899           scope += '_%d' % i
 900 
 901         branch_logits.append(
 902             slim.conv2d(
 903                 features,
 904                 num_classes,
 905                 kernel_size=kernel_size,
 906                 rate=rate,
 907                 activation_fn=None,
 908                 normalizer_fn=None,
 909                 scope=scope))
 910 
 911       return tf.add_n(branch_logits)
