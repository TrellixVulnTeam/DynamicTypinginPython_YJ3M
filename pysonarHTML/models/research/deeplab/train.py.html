 221 def _build_deeplab(iterator, outputs_to_num_classes, ignore_label):
 222   """Builds a clone of DeepLab.
 223 
 224   Args:
 225     iterator: An iterator of type tf.data.Iterator for images and labels.
 226     outputs_to_num_classes: A map from output type to the number of classes. For
 227       example, for the task of semantic segmentation with 21 semantic classes,
 228       we would have outputs_to_num_classes['semantic'] = 21.
 229     ignore_label: Ignore label.
 230   """
 231   samples = iterator.get_next()
 232 
 233   # Add name to input and label nodes so we can add to summary.
 234   samples[common.IMAGE] = tf.identity(samples[common.IMAGE], name=common.IMAGE)
 235   samples[common.LABEL] = tf.identity(samples[common.LABEL], name=common.LABEL)
 236 
 237   model_options = common.ModelOptions(
 238       outputs_to_num_classes=outputs_to_num_classes,
 239       crop_size=[int(sz) for sz in FLAGS.train_crop_size],
 240       atrous_rates=FLAGS.atrous_rates,
 241       output_stride=FLAGS.output_stride)
 242 
 243   outputs_to_scales_to_logits = model.multi_scale_logits(
 244       samples[common.IMAGE],
 245       model_options=model_options,
 246       image_pyramid=FLAGS.image_pyramid,
 247       weight_decay=FLAGS.weight_decay,
 248       is_training=True,
 249       fine_tune_batch_norm=FLAGS.fine_tune_batch_norm,
 250       nas_training_hyper_parameters={
 251           'drop_path_keep_prob': FLAGS.drop_path_keep_prob,
 252           'total_training_steps': FLAGS.training_number_of_steps,
 253       })
 254 
 255   # Add name to graph node so we can add to summary.
 256   output_type_dict = outputs_to_scales_to_logits[common.OUTPUT_TYPE]
 257   output_type_dict[model.MERGED_LOGITS_SCOPE] = tf.identity(
 258       output_type_dict[model.MERGED_LOGITS_SCOPE], name=common.OUTPUT_TYPE)
 259 
 260   for output, num_classes in six.iteritems(outputs_to_num_classes):
 261     train_utils.add_softmax_cross_entropy_loss_for_each_scale(
 262         outputs_to_scales_to_logits[output],
 263         samples[common.LABEL],
 264         num_classes,
 265         ignore_label,
 266         loss_weight=model_options.label_weights,
 267         upsample_logits=FLAGS.upsample_logits,
 268         hard_example_mining_step=FLAGS.hard_example_mining_step,
 269         top_k_percent_pixels=FLAGS.top_k_percent_pixels,
 270         scope=output)
 271

 273 def main(unused_argv):
 274   tf.logging.set_verbosity(tf.logging.INFO)
 275   # Set up deployment (i.e., multi-GPUs and/or multi-replicas).
 276   config = model_deploy.DeploymentConfig(
 277       num_clones=FLAGS.num_clones,
 278       clone_on_cpu=FLAGS.clone_on_cpu,
 279       replica_id=FLAGS.task,
 280       num_replicas=FLAGS.num_replicas,
 281       num_ps_tasks=FLAGS.num_ps_tasks)
 282 
 283   # Split the batch across GPUs.
 284   assert FLAGS.train_batch_size % config.num_clones == 0, (
 285       'Training batch size not divisble by number of clones (GPUs).')
 286 
 287   clone_batch_size = FLAGS.train_batch_size // config.num_clones
 288 
 289   tf.gfile.MakeDirs(FLAGS.train_logdir)
 290   tf.logging.info('Training on %s set', FLAGS.train_split)
 291 
 292   with tf.Graph().as_default() as graph:
 293     with tf.device(config.inputs_device()):
 294       dataset = data_generator.Dataset(
 295           dataset_name=FLAGS.dataset,
 296           split_name=FLAGS.train_split,
 297           dataset_dir=FLAGS.dataset_dir,
 298           batch_size=clone_batch_size,
 299           crop_size=[int(sz) for sz in FLAGS.train_crop_size],
 300           min_resize_value=FLAGS.min_resize_value,
 301           max_resize_value=FLAGS.max_resize_value,
 302           resize_factor=FLAGS.resize_factor,
 303           min_scale_factor=FLAGS.min_scale_factor,
 304           max_scale_factor=FLAGS.max_scale_factor,
 305           scale_factor_step_size=FLAGS.scale_factor_step_size,
 306           model_variant=FLAGS.model_variant,
 307           num_readers=4,
 308           is_training=True,
 309           should_shuffle=True,
 310           should_repeat=True)
 311 
 312     # Create the global step on the device storing the variables.
 313     with tf.device(config.variables_device()):
 314       global_step = tf.train.get_or_create_global_step()
 315 
 316       # Define the model and create clones.
 317       model_fn = _build_deeplab
 318       model_args = (dataset.get_one_shot_iterator(), {
 319           common.OUTPUT_TYPE: dataset.num_of_classes
 320       }, dataset.ignore_label)
 321       clones = model_deploy.create_clones(config, model_fn, args=model_args)
 322 
 323       # Gather update_ops from the first clone. These contain, for example,
 324       # the updates for the batch_norm variables created by model_fn.
 325       first_clone_scope = config.clone_scope(0)
 326       update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS, first_clone_scope)
 327 
 328     # Gather initial summaries.
 329     summaries = set(tf.get_collection(tf.GraphKeys.SUMMARIES))
 330 
 331     # Add summaries for model variables.
 332     for model_var in tf.model_variables():
 333       summaries.add(tf.summary.histogram(model_var.op.name, model_var))
 334 
 335     # Add summaries for images, labels, semantic predictions
 336     if FLAGS.save_summaries_images:
 337       summary_image = graph.get_tensor_by_name(
 338           ('%s/%s:0' % (first_clone_scope, common.IMAGE)).strip('/'))
 339       summaries.add(
 340           tf.summary.image('samples/%s' % common.IMAGE, summary_image))
 341 
 342       first_clone_label = graph.get_tensor_by_name(
 343           ('%s/%s:0' % (first_clone_scope, common.LABEL)).strip('/'))
 344       # Scale up summary image pixel values for better visualization.
 345       pixel_scaling = max(1, 255 // dataset.num_of_classes)
 346       summary_label = tf.cast(first_clone_label * pixel_scaling, tf.uint8)
 347       summaries.add(
 348           tf.summary.image('samples/%s' % common.LABEL, summary_label))
 349 
 350       first_clone_output = graph.get_tensor_by_name(
 351           ('%s/%s:0' % (first_clone_scope, common.OUTPUT_TYPE)).strip('/'))
 352       predictions = tf.expand_dims(tf.argmax(first_clone_output, 3), -1)
 353 
 354       summary_predictions = tf.cast(predictions * pixel_scaling, tf.uint8)
 355       summaries.add(
 356           tf.summary.image(
 357               'samples/%s' % common.OUTPUT_TYPE, summary_predictions))
 358 
 359     # Add summaries for losses.
 360     for loss in tf.get_collection(tf.GraphKeys.LOSSES, first_clone_scope):
 361       summaries.add(tf.summary.scalar('losses/%s' % loss.op.name, loss))
 362 
 363     # Build the optimizer based on the device specification.
 364     with tf.device(config.optimizer_device()):
 365       learning_rate = train_utils.get_model_learning_rate(
 366           FLAGS.learning_policy,
 367           FLAGS.base_learning_rate,
 368           FLAGS.learning_rate_decay_step,
 369           FLAGS.learning_rate_decay_factor,
 370           FLAGS.training_number_of_steps,
 371           FLAGS.learning_power,
 372           FLAGS.slow_start_step,
 373           FLAGS.slow_start_learning_rate,
 374           decay_steps=FLAGS.decay_steps,
 375           end_learning_rate=FLAGS.end_learning_rate)
 376 
 377       summaries.add(tf.summary.scalar('learning_rate', learning_rate))
 378 
 379       if FLAGS.optimizer == 'momentum':
 380         optimizer = tf.train.MomentumOptimizer(learning_rate, FLAGS.momentum)
 381       elif FLAGS.optimizer == 'adam':
 382         optimizer = tf.train.AdamOptimizer(
 383             learning_rate=FLAGS.adam_learning_rate, epsilon=FLAGS.adam_epsilon)
 384       else:
 385         raise ValueError('Unknown optimizer')
 386 
 387     if FLAGS.quantize_delay_step >= 0:
 388       if FLAGS.num_clones > 1:
 389         raise ValueError('Quantization doesn\'t support multi-clone yet.')
 390       contrib_quantize.create_training_graph(
 391           quant_delay=FLAGS.quantize_delay_step)
 392 
 393     startup_delay_steps = FLAGS.task * FLAGS.startup_delay_steps
 394 
 395     with tf.device(config.variables_device()):
 396       total_loss, grads_and_vars = model_deploy.optimize_clones(
 397           clones, optimizer)
 398       total_loss = tf.check_numerics(total_loss, 'Loss is inf or nan.')
 399       summaries.add(tf.summary.scalar('total_loss', total_loss))
 400 
 401       # Modify the gradients for biases and last layer variables.
 402       last_layers = model.get_extra_layer_scopes(
 403           FLAGS.last_layers_contain_logits_only)
 404       grad_mult = train_utils.get_model_gradient_multipliers(
 405           last_layers, FLAGS.last_layer_gradient_multiplier)
 406       if grad_mult:
 407         grads_and_vars = slim.learning.multiply_gradients(
 408             grads_and_vars, grad_mult)
 409 
 410       # Create gradient update op.
 411       grad_updates = optimizer.apply_gradients(
 412           grads_and_vars, global_step=global_step)
 413       update_ops.append(grad_updates)
 414       update_op = tf.group(*update_ops)
 415       with tf.control_dependencies([update_op]):
 416         train_tensor = tf.identity(total_loss, name='train_op')
 417 
 418     # Add the summaries from the first clone. These contain the summaries
 419     # created by model_fn and either optimize_clones() or _gather_clone_loss().
 420     summaries |= set(
 421         tf.get_collection(tf.GraphKeys.SUMMARIES, first_clone_scope))
 422 
 423     # Merge all summaries together.
 424     summary_op = tf.summary.merge(list(summaries))
 425 
 426     # Soft placement allows placing on CPU ops without GPU implementation.
 427     session_config = tf.ConfigProto(
 428         allow_soft_placement=True, log_device_placement=False)
 429 
 430     # Start the training.
 431     profile_dir = FLAGS.profile_logdir
 432     if profile_dir is not None:
 433       tf.gfile.MakeDirs(profile_dir)
 434 
 435     with contrib_tfprof.ProfileContext(
 436         enabled=profile_dir is not None, profile_dir=profile_dir):
 437       init_fn = None
 438       if FLAGS.tf_initial_checkpoint:
 439         init_fn = train_utils.get_model_init_fn(
 440             FLAGS.train_logdir,
 441             FLAGS.tf_initial_checkpoint,
 442             FLAGS.initialize_last_layer,
 443             last_layers,
 444             ignore_missing_vars=True)
 445 
 446       slim.learning.train(
 447           train_tensor,
 448           logdir=FLAGS.train_logdir,
 449           log_every_n_steps=FLAGS.log_steps,
 450           master=FLAGS.master,
 451           number_of_steps=FLAGS.training_number_of_steps,
 452           is_chief=(FLAGS.task == 0),
 453           session_config=session_config,
 454           startup_delay_steps=startup_delay_steps,
 455           init_fn=init_fn,
 456           summary_op=summary_op,
 457           save_summaries_secs=FLAGS.save_summaries_secs,
 458           save_interval_secs=FLAGS.save_interval_secs)
 459 
 460 
 461 if __name__ == '__main__':
 462   flags.mark_flag_as_required('train_logdir')
 463   flags.mark_flag_as_required('dataset_dir')
 464   tf.app.run()
