  17 """Tests for get_dataset_colormap.py."""
  18 
  19 import numpy as np
  20 import tensorflow as tf
  21 
  22 from deeplab.utils import get_dataset_colormap
  23 
  24 
  25 class VisualizationUtilTest(tf.test.TestCase):
  26 
  27   def testBitGet(self):
  28     """Test that if the returned bit value is correct."""
  29     self.assertEqual(1, get_dataset_colormap.bit_get(9, 0))
  30     self.assertEqual(0, get_dataset_colormap.bit_get(9, 1))
  31     self.assertEqual(0, get_dataset_colormap.bit_get(9, 2))
  32     self.assertEqual(1, get_dataset_colormap.bit_get(9, 3))
  33 
  34   def testPASCALLabelColorMapValue(self):
  35     """Test the getd color map value."""
  36     colormap = get_dataset_colormap.create_pascal_label_colormap()
  37 
  38     # Only test a few sampled entries in the color map.
  39     self.assertTrue(np.array_equal([128., 0., 128.], colormap[5, :]))
  40     self.assertTrue(np.array_equal([128., 192., 128.], colormap[23, :]))
  41     self.assertTrue(np.array_equal([128., 0., 192.], colormap[37, :]))
  42     self.assertTrue(np.array_equal([224., 192., 192.], colormap[127, :]))
  43     self.assertTrue(np.array_equal([192., 160., 192.], colormap[175, :]))
  44 
  45   def testLabelToPASCALColorImage(self):
  46     """Test the value of the converted label value."""
  47     label = np.array([[0, 16, 16], [52, 7, 52]])
  48     expected_result = np.array([
  49         [[0, 0, 0], [0, 64, 0], [0, 64, 0]],
  50         [[0, 64, 192], [128, 128, 128], [0, 64, 192]]
  51     ])
  52     colored_label = get_dataset_colormap.label_to_color_image(
  53         label, get_dataset_colormap.get_pascal_name())
  54     self.assertTrue(np.array_equal(expected_result, colored_label))
  55 
  56   def testUnExpectedLabelValueForLabelToPASCALColorImage(self):
  57     """Raise ValueError when input value exceeds range."""
  58     label = np.array([[120], [600]])
  59     with self.assertRaises(ValueError):
  60       get_dataset_colormap.label_to_color_image(
  61           label, get_dataset_colormap.get_pascal_name())
  62 
  63   def testUnExpectedLabelDimensionForLabelToPASCALColorImage(self):
  64     """Raise ValueError if input dimension is not correct."""
  65     label = np.array([120])
  66     with self.assertRaises(ValueError):
  67       get_dataset_colormap.label_to_color_image(
  68           label, get_dataset_colormap.get_pascal_name())
  69 
  70   def testGetColormapForUnsupportedDataset(self):
  71     with self.assertRaises(ValueError):
  72       get_dataset_colormap.create_label_colormap('unsupported_dataset')
  73 
  74   def testUnExpectedLabelDimensionForLabelToADE20KColorImage(self):
  75     label = np.array([250])

  78           label, get_dataset_colormap.get_ade20k_name())
  79 
  80   def testFirstColorInADE20KColorMap(self):
  81     label = np.array([[1, 3], [10, 20]])
  82     expected_result = np.array([
  83         [[120, 120, 120], [6, 230, 230]],
  84         [[4, 250, 7], [204, 70, 3]]
  85     ])
  86     colored_label = get_dataset_colormap.label_to_color_image(
  87         label, get_dataset_colormap.get_ade20k_name())
  88     self.assertTrue(np.array_equal(colored_label, expected_result))
  89 
  90   def testMapillaryVistasColorMapValue(self):
  91     colormap = get_dataset_colormap.create_mapillary_vistas_label_colormap()
  92     self.assertTrue(np.array_equal([190, 153, 153], colormap[3, :]))
  93     self.assertTrue(np.array_equal([102, 102, 156], colormap[6, :]))
