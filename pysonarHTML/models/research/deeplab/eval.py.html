  25 from tensorflow.contrib import quantize as contrib_quantize
  26 from tensorflow.contrib import tfprof as contrib_tfprof

  32 flags = tf.app.flags

  35 flags.DEFINE_string('master', '', 'BNS name of the tensorflow server')
  36

  38 
  39 flags.DEFINE_string('eval_logdir', None, 'Where to write the event logs.')

  46                      'The number of images in each batch during evaluation.')

  49                   'Image crop size [height, width] for evaluation.')

  54 # For `xception_65`, use atrous_rates = [12, 24, 36] if output_stride = 8, or
  55 # rates = [6, 12, 18] if output_stride = 16. For `mobilenet_v2`, use None. Note

  90 def main(unused_argv):
  91   tf.logging.set_verbosity(tf.logging.INFO)
  92 
  93   dataset = data_generator.Dataset(
  94       dataset_name=FLAGS.dataset,
  95       split_name=FLAGS.eval_split,
  96       dataset_dir=FLAGS.dataset_dir,
  97       batch_size=FLAGS.eval_batch_size,
  98       crop_size=[int(sz) for sz in FLAGS.eval_crop_size],
  99       min_resize_value=FLAGS.min_resize_value,
 100       max_resize_value=FLAGS.max_resize_value,
 101       resize_factor=FLAGS.resize_factor,
 102       model_variant=FLAGS.model_variant,
 103       num_readers=2,
 104       is_training=False,
 105       should_shuffle=False,
 106       should_repeat=False)
 107 
 108   tf.gfile.MakeDirs(FLAGS.eval_logdir)
 109   tf.logging.info('Evaluating on %s set', FLAGS.eval_split)
 110 
 111   with tf.Graph().as_default():
 112     samples = dataset.get_one_shot_iterator().get_next()
 113 
 114     model_options = common.ModelOptions(
 115         outputs_to_num_classes={common.OUTPUT_TYPE: dataset.num_of_classes},
 116         crop_size=[int(sz) for sz in FLAGS.eval_crop_size],
 117         atrous_rates=FLAGS.atrous_rates,
 118         output_stride=FLAGS.output_stride)
 119 
 120     # Set shape in order for tf.contrib.tfprof.model_analyzer to work properly.
 121     samples[common.IMAGE].set_shape(
 122         [FLAGS.eval_batch_size,
 123          int(FLAGS.eval_crop_size[0]),
 124          int(FLAGS.eval_crop_size[1]),
 125          3])
 126     if tuple(FLAGS.eval_scales) == (1.0,):
 127       tf.logging.info('Performing single-scale test.')
 128       predictions = model.predict_labels(samples[common.IMAGE], model_options,
 129                                          image_pyramid=FLAGS.image_pyramid)
 130     else:
 131       tf.logging.info('Performing multi-scale test.')
 132       if FLAGS.quantize_delay_step >= 0:
 133         raise ValueError(
 134             'Quantize mode is not supported with multi-scale test.')
 135 
 136       predictions = model.predict_labels_multi_scale(
 137           samples[common.IMAGE],
 138           model_options=model_options,
 139           eval_scales=FLAGS.eval_scales,
 140           add_flipped_images=FLAGS.add_flipped_images)
 141     predictions = predictions[common.OUTPUT_TYPE]
 142     predictions = tf.reshape(predictions, shape=[-1])
 143     labels = tf.reshape(samples[common.LABEL], shape=[-1])
 144     weights = tf.to_float(tf.not_equal(labels, dataset.ignore_label))
 145 
 146     # Set ignore_label regions to label 0, because metrics.mean_iou requires
 147     # range of labels = [0, dataset.num_classes). Note the ignore_label regions
 148     # are not evaluated since the corresponding regions contain weights = 0.
 149     labels = tf.where(
 150         tf.equal(labels, dataset.ignore_label), tf.zeros_like(labels), labels)
 151 
 152     predictions_tag = 'miou'
 153     for eval_scale in FLAGS.eval_scales:
 154       predictions_tag += '_' + str(eval_scale)
 155     if FLAGS.add_flipped_images:
 156       predictions_tag += '_flipped'
 157 
 158     # Define the evaluation metric.
 159     metric_map = {}
 160     num_classes = dataset.num_of_classes
 161     metric_map['eval/%s_overall' % predictions_tag] = tf.metrics.mean_iou(
 162         labels=labels, predictions=predictions, num_classes=num_classes,
 163         weights=weights)
 164     # IoU for each class.
 165     one_hot_predictions = tf.one_hot(predictions, num_classes)
 166     one_hot_predictions = tf.reshape(one_hot_predictions, [-1, num_classes])
 167     one_hot_labels = tf.one_hot(labels, num_classes)
 168     one_hot_labels = tf.reshape(one_hot_labels, [-1, num_classes])
 169     for c in range(num_classes):
 170       predictions_tag_c = '%s_class_%d' % (predictions_tag, c)
 171       tp, tp_op = tf.metrics.true_positives(
 172           labels=one_hot_labels[:, c], predictions=one_hot_predictions[:, c],
 173           weights=weights)
 174       fp, fp_op = tf.metrics.false_positives(
 175           labels=one_hot_labels[:, c], predictions=one_hot_predictions[:, c],
 176           weights=weights)
 177       fn, fn_op = tf.metrics.false_negatives(
 178           labels=one_hot_labels[:, c], predictions=one_hot_predictions[:, c],
 179           weights=weights)
 180       tp_fp_fn_op = tf.group(tp_op, fp_op, fn_op)
 181       iou = tf.where(tf.greater(tp + fn, 0.0),
 182                      tp / (tp + fn + fp),
 183                      tf.constant(np.NaN))
 184       metric_map['eval/%s' % predictions_tag_c] = (iou, tp_fp_fn_op)
 185 
 186     (metrics_to_values,
 187      metrics_to_updates) = contrib_metrics.aggregate_metric_map(metric_map)
 188 
 189     summary_ops = []
 190     for metric_name, metric_value in six.iteritems(metrics_to_values):
 191       op = tf.summary.scalar(metric_name, metric_value)
 192       op = tf.Print(op, [metric_value], metric_name)
 193       summary_ops.append(op)
 194 
 195     summary_op = tf.summary.merge(summary_ops)
 196     summary_hook = contrib_training.SummaryAtEndHook(
 197         log_dir=FLAGS.eval_logdir, summary_op=summary_op)
 198     hooks = [summary_hook]
 199 
 200     num_eval_iters = None
 201     if FLAGS.max_number_of_evaluations > 0:
 202       num_eval_iters = FLAGS.max_number_of_evaluations
 203 
 204     if FLAGS.quantize_delay_step >= 0:
 205       contrib_quantize.create_eval_graph()
 206 
 207     contrib_tfprof.model_analyzer.print_model_analysis(
 208         tf.get_default_graph(),
 209         tfprof_options=contrib_tfprof.model_analyzer
 210         .TRAINABLE_VARS_PARAMS_STAT_OPTIONS)
 211     contrib_tfprof.model_analyzer.print_model_analysis(
 212         tf.get_default_graph(),
 213         tfprof_options=contrib_tfprof.model_analyzer.FLOAT_OPS_OPTIONS)
 214     contrib_training.evaluate_repeatedly(
 215         checkpoint_dir=FLAGS.checkpoint_dir,
 216         master=FLAGS.master,
 217         eval_ops=list(metrics_to_updates.values()),
 218         max_number_of_evaluations=num_eval_iters,
 219         hooks=hooks,
 220         eval_interval_secs=FLAGS.eval_interval_secs)
 221 
 222 
 223 if __name__ == '__main__':
 224   flags.mark_flag_as_required('checkpoint_dir')
 225   flags.mark_flag_as_required('eval_logdir')
 226   flags.mark_flag_as_required('dataset_dir')
 227   tf.app.run()
