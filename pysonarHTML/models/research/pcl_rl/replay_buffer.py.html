  16 """Replay buffer.
  17 
  18 Implements replay buffer in Python.
  19 """
  20 
  21 import random
  22 import numpy as np
  23 from six.moves import xrange
  24 
  25 
  26 class ReplayBuffer(object):
  27 
  28   def __init__(self, max_size):
  29     self.max_size = max_size
  30     self.cur_size = 0
  31     self.buffer = {}
  32     self.init_length = 0
  33 
  34   def __len__(self):
  35     return self.cur_size
  36 
  37   def seed_buffer(self, episodes):
  38     self.init_length = len(episodes)
  39     self.add(episodes, np.ones(self.init_length))
  40 
  41   def add(self, episodes, *args):
  42     """Add episodes to buffer."""
  43     idx = 0
  44     while self.cur_size < self.max_size and idx < len(episodes):
  45       self.buffer[self.cur_size] = episodes[idx]
  46       self.cur_size += 1
  47       idx += 1
  48 
  49     if idx < len(episodes):
  50       remove_idxs = self.remove_n(len(episodes) - idx)
  51       for remove_idx in remove_idxs:
  52         self.buffer[remove_idx] = episodes[idx]
  53         idx += 1
  54 
  55     assert len(self.buffer) == self.cur_size
  56 
  57   def remove_n(self, n):
  58     """Get n items for removal."""
  59     # random removal
  60     idxs = random.sample(xrange(self.init_length, self.cur_size), n)
  61     return idxs
  62 
  63   def get_batch(self, n):
  64     """Get batch of episodes to train on."""
  65     # random batch
  66     idxs = random.sample(xrange(self.cur_size), n)
  67     return [self.buffer[idx] for idx in idxs], None
  68 
  69   def update_last_batch(self, delta):
  70     pass

  73 class PrioritizedReplayBuffer(ReplayBuffer):
  74 
  75   def __init__(self, max_size, alpha=0.2,
  76                eviction_strategy='rand'):
  77     self.max_size = max_size
  78     self.alpha = alpha
  79     self.eviction_strategy = eviction_strategy
  80     assert self.eviction_strategy in ['rand', 'fifo', 'rank']
  81     self.remove_idx = 0
  82 
  83     self.cur_size = 0
  84     self.buffer = {}
  85     self.priorities = np.zeros(self.max_size)
  86     self.init_length = 0
  87 
  88   def __len__(self):
  89     return self.cur_size
  90 
  91   def add(self, episodes, priorities, new_idxs=None):
  92     """Add episodes to buffer."""
  93     if new_idxs is None:
  94       idx = 0
  95       new_idxs = []
  96       while self.cur_size < self.max_size and idx < len(episodes):
  97         self.buffer[self.cur_size] = episodes[idx]
  98         new_idxs.append(self.cur_size)
  99         self.cur_size += 1
 100         idx += 1
 101 
 102       if idx < len(episodes):
 103         remove_idxs = self.remove_n(len(episodes) - idx)
 104         for remove_idx in remove_idxs:
 105           self.buffer[remove_idx] = episodes[idx]
 106           new_idxs.append(remove_idx)
 107           idx += 1
 108     else:
 109       assert len(new_idxs) == len(episodes)
 110       for new_idx, ep in zip(new_idxs, episodes):
 111         self.buffer[new_idx] = ep
 112 
 113     self.priorities[new_idxs] = priorities
 114     self.priorities[0:self.init_length] = np.max(
 115         self.priorities[self.init_length:])
 116 
 117     assert len(self.buffer) == self.cur_size
 118     return new_idxs
 119 
 120   def remove_n(self, n):
 121     """Get n items for removal."""
 122     assert self.init_length + n <= self.cur_size
 123 
 124     if self.eviction_strategy == 'rand':
 125       # random removal
 126       idxs = random.sample(xrange(self.init_length, self.cur_size), n)
 127     elif self.eviction_strategy == 'fifo':
 128       # overwrite elements in cyclical fashion
 129       idxs = [
 130           self.init_length +
 131           (self.remove_idx + i) % (self.max_size - self.init_length)
 132           for i in xrange(n)]
 133       self.remove_idx = idxs[-1] + 1 - self.init_length
 134     elif self.eviction_strategy == 'rank':
 135       # remove lowest-priority indices
 136       idxs = np.argpartition(self.priorities, n-1)[:n]
 137 
 138     return idxs
 139 
 140   def sampling_distribution(self):
 141     p = self.priorities[:self.cur_size]
 142     p = np.exp(self.alpha * (p - np.max(p)))
 143     norm = np.sum(p)
 144     if norm > 0:
 145       uniform = 0.0
 146       p = p / norm * (1 - uniform) + 1.0 / self.cur_size * uniform
 147     else:
 148       p = np.ones(self.cur_size) / self.cur_size
 149     return p
 150 
 151   def get_batch(self, n):
 152     """Get batch of episodes to train on."""
 153     p = self.sampling_distribution()
 154     idxs = np.random.choice(self.cur_size, size=int(n), replace=False, p=p)
 155     self.last_batch = idxs
 156     return [self.buffer[idx] for idx in idxs], p[idxs]
 157 
 158   def update_last_batch(self, delta):
 159     """Update last batch idxs with new priority."""
 160     self.priorities[self.last_batch] = np.abs(delta)
 161     self.priorities[0:self.init_length] = np.max(
 162         self.priorities[self.init_length:])
