  22 import numpy as np

  26 class spaces(object):
  27   discrete = 0
  28   box = 1
  29 
  30 
  31 def get_space(space):
  32   if hasattr(space, 'n'):
  33     return space.n, spaces.discrete, None
  34   elif hasattr(space, 'shape'):
  35     return np.prod(space.shape), spaces.box, (space.low, space.high)
  36 
  37 
  38 def get_spaces(spaces):
  39   if hasattr(spaces, 'spaces'):
  40     return zip(*[get_space(space) for space in spaces.spaces])
  41   else:
  42     return [(ret,) for ret in get_space(spaces)]

  45 class EnvSpec(object):
  46 
  47   def __init__(self, env, try_combining_actions=True,
  48                discretize_actions=None):
  49     self.discretize_actions = discretize_actions
  50 
  51     # figure out observation space
  52     self.obs_space = env.observation_space
  53     self.obs_dims, self.obs_types, self.obs_info = get_spaces(self.obs_space)
  54 
  55     # figure out action space
  56     self.act_space = env.action_space
  57     self.act_dims, self.act_types, self.act_info = get_spaces(self.act_space)
  58 
  59     if self.discretize_actions:
  60       self._act_dims = self.act_dims[:]
  61       self._act_types = self.act_types[:]
  62       self.act_dims = []
  63       self.act_types = []
  64       for i, (dim, typ) in enumerate(zip(self._act_dims, self._act_types)):
  65         if typ == spaces.discrete:
  66           self.act_dims.append(dim)
  67           self.act_types.append(spaces.discrete)
  68         elif typ == spaces.box:
  69           for _ in xrange(dim):
  70             self.act_dims.append(self.discretize_actions)
  71             self.act_types.append(spaces.discrete)
  72     else:
  73       self._act_dims = None
  74       self._act_types = None
  75 
  76     if (try_combining_actions and
  77         all(typ == spaces.discrete for typ in self.act_types)):
  78       self.combine_actions = True
  79       self.orig_act_dims = self.act_dims[:]
  80       self.orig_act_types = self.act_types[:]
  81       total_act_dim = 1
  82       for dim in self.act_dims:
  83         total_act_dim *= dim
  84       self.act_dims = [total_act_dim]
  85       self.act_types = [spaces.discrete]
  86     else:
  87       self.combine_actions = False
  88 
  89     self.obs_dims_and_types = tuple(zip(self.obs_dims, self.obs_types))
  90     self.act_dims_and_types = tuple(zip(self.act_dims, self.act_types))
  91 
  92     self.total_obs_dim = sum(self.obs_dims)
  93     self.total_sampling_act_dim = sum(self.sampling_dim(dim, typ)
  94                                       for dim, typ in self.act_dims_and_types)
  95     self.total_sampled_act_dim = sum(self.act_dims)
  96 
  97   def sampling_dim(self, dim, typ):
  98     if typ == spaces.discrete:
  99       return dim
 100     elif typ == spaces.box:
 101       return 2 * dim  # Gaussian mean and std
 102     else:
 103       assert False
 104 
 105   def convert_actions_to_env(self, actions):
 106     if self.combine_actions:
 107       new_actions = []
 108       actions = actions[0]
 109       for dim in self.orig_act_dims:
 110         new_actions.append(np.mod(actions, dim))
 111         actions = (actions / dim).astype('int32')
 112       actions = new_actions
 113 
 114     if self.discretize_actions:
 115       new_actions = []
 116       idx = 0
 117       for i, (dim, typ) in enumerate(zip(self._act_dims, self._act_types)):
 118         if typ == spaces.discrete:
 119           new_actions.append(actions[idx])
 120           idx += 1
 121         elif typ == spaces.box:
 122           low, high = self.act_info[i]
 123           cur_action = []
 124           for j in xrange(dim):
 125             cur_action.append(
 126                 low[j] + (high[j] - low[j]) * actions[idx] /
 127                 float(self.discretize_actions))
 128             idx += 1
 129           new_actions.append(np.hstack(cur_action))
 130       actions = new_actions
 131 
 132     return actions
 133 
 134   def convert_env_actions_to_actions(self, actions):
 135     if not self.combine_actions:
 136       return actions
 137 
 138     new_actions = 0
 139     base = 1
 140     for act, dim in zip(actions, self.orig_act_dims):
 141       new_actions = new_actions + base * act
 142       base *= dim
 143 
 144     return [new_actions]
 145 
 146   def convert_obs_to_list(self, obs):
 147     if len(self.obs_dims) == 1:
 148       return [obs]
 149     else:
 150       return list(obs)
 151 
 152   def convert_action_to_gym(self, action):
 153     if len(action) == 1:
 154       return action[0]
 155     else:
 156       return list(action)
 157     if ((not self.combine_actions or len(self.orig_act_dims) == 1) and
 158         (len(self.act_dims) == 1 or
 159          (self.discretize_actions and len(self._act_dims) == 1))):
 160       return action[0]
 161     else:
 162       return list(action)
 163 
 164   def initial_obs(self, batch_size):
 165     batched = batch_size is not None
 166     batch_size = batch_size or 1
 167 
 168     obs = []
 169     for dim, typ in self.obs_dims_and_types:
 170       if typ == spaces.discrete:
 171         obs.append(np.zeros(batch_size))
 172       elif typ == spaces.box:
 173         obs.append(np.zeros([batch_size, dim]))
 174 
 175     if batched:
 176       return obs
 177     else:
 178       return zip(*obs)[0]
 179 
 180   def initial_act(self, batch_size=None):
 181     batched = batch_size is not None
 182     batch_size = batch_size or 1
 183 
 184     act = []
 185     for dim, typ in self.act_dims_and_types:
 186       if typ == spaces.discrete:
 187         act.append(-np.ones(batch_size))
 188       elif typ == spaces.box:
 189         act.append(-np.ones([batch_size, dim]))
 190 
 191     if batched:
 192       return act
 193     else:
 194       return zip(*act)[0]
 195 
 196   def is_discrete(self, typ):
 197     return typ == spaces.discrete
 198 
 199   def is_box(self, typ):
 200     return typ == spaces.box
