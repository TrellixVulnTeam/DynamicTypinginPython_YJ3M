  16 """Wrapper around gym env.
  17 
  18 Allows for using batches of possibly identitically seeded environments.
  19 """
  20 
  21 import gym
  22 import numpy as np
  23 import random
  24 
  25 from six.moves import xrange
  26 import env_spec
  27 
  28 
  29 def get_env(env_str):
  30   return gym.make(env_str)
  31 
  32 
  33 class GymWrapper(object):
  34 
  35   def __init__(self, env_str, distinct=1, count=1, seeds=None):
  36     self.distinct = distinct
  37     self.count = count
  38     self.total = self.distinct * self.count
  39     self.seeds = seeds or [random.randint(0, 1e12)
  40                            for _ in xrange(self.distinct)]
  41 
  42     self.envs = []
  43     for seed in self.seeds:
  44       for _ in xrange(self.count):
  45         env = get_env(env_str)
  46         env.seed(seed)
  47         if hasattr(env, 'last'):
  48           env.last = 100  # for algorithmic envs
  49         self.envs.append(env)
  50 
  51     self.dones = [True] * self.total
  52     self.num_episodes_played = 0
  53 
  54     one_env = self.get_one()
  55     self.use_action_list = hasattr(one_env.action_space, 'spaces')
  56     self.env_spec = env_spec.EnvSpec(self.get_one())
  57 
  58   def get_seeds(self):
  59     return self.seeds
  60 
  61   def reset(self):
  62     self.dones = [False] * self.total
  63     self.num_episodes_played += len(self.envs)
  64 
  65     # reset seeds to be synchronized
  66     self.seeds = [random.randint(0, 1e12) for _ in xrange(self.distinct)]
  67     counter = 0
  68     for seed in self.seeds:
  69       for _ in xrange(self.count):
  70         self.envs[counter].seed(seed)
  71         counter += 1
  72 
  73     return [self.env_spec.convert_obs_to_list(env.reset())
  74             for env in self.envs]
  75 
  76   def reset_if(self, predicate=None):
  77     if predicate is None:
  78       predicate = self.dones
  79     if self.count != 1:
  80       assert np.all(predicate)
  81       return self.reset()
  82     self.num_episodes_played += sum(predicate)
  83     output = [self.env_spec.convert_obs_to_list(env.reset())
  84               if pred else None
  85               for env, pred in zip(self.envs, predicate)]
  86     for i, pred in enumerate(predicate):
  87       if pred:
  88         self.dones[i] = False
  89     return output
  90 
  91   def all_done(self):
  92     return all(self.dones)
  93 
  94   def step(self, actions):
  95 
  96     def env_step(env, action):
  97       action = self.env_spec.convert_action_to_gym(action)
  98       obs, reward, done, tt = env.step(action)
  99       obs = self.env_spec.convert_obs_to_list(obs)
 100       return obs, reward, done, tt
 101 
 102     actions = zip(*actions)
 103     outputs = [env_step(env, action)
 104                if not done else (self.env_spec.initial_obs(None), 0, True, None)
 105                for action, env, done in zip(actions, self.envs, self.dones)]
 106     for i, (_, _, done, _) in enumerate(outputs):
 107       self.dones[i] = self.dones[i] or done
 108     obs, reward, done, tt = zip(*outputs)
 109     obs = [list(oo) for oo in zip(*obs)]
 110     return [obs, reward, done, tt]
 111 
 112   def get_one(self):
 113     return random.choice(self.envs)
 114 
 115   def __len__(self):
 116     return len(self.envs)
