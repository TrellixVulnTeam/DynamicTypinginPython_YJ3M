  16 """Policy neural network.
  17 
  18 Implements network which takes in input and produces actions
  19 and log probabilities given a sampling distribution parameterization.
  20 """
  21 
  22 import tensorflow as tf
  23 import numpy as np

  26 class Policy(object):
  27   def __init__(self, env_spec, internal_dim,
  28                fixed_std=True, recurrent=True,
  29                input_prev_actions=True):
  30     self.env_spec = env_spec
  31     self.internal_dim = internal_dim
  32     self.rnn_state_dim = self.internal_dim
  33     self.fixed_std = fixed_std
  34     self.recurrent = recurrent
  35     self.input_prev_actions = input_prev_actions
  36 
  37     self.matrix_init = tf.truncated_normal_initializer(stddev=0.01)
  38     self.vector_init = tf.constant_initializer(0.0)
  39 
  40   @property
  41   def input_dim(self):
  42     return (self.env_spec.total_obs_dim +
  43             self.env_spec.total_sampled_act_dim * self.input_prev_actions)
  44 
  45   @property
  46   def output_dim(self):
  47     return self.env_spec.total_sampling_act_dim
  48 
  49   def get_cell(self):
  50     """Get RNN cell."""
  51     self.cell_input_dim = self.internal_dim // 2
  52     cell = tf.contrib.rnn.LSTMCell(self.cell_input_dim,
  53                                    state_is_tuple=False,
  54                                    reuse=tf.get_variable_scope().reuse)
  55 
  56     cell = tf.contrib.rnn.OutputProjectionWrapper(
  57         cell, self.output_dim,
  58         reuse=tf.get_variable_scope().reuse)
  59 
  60     return cell
  61 
  62   def core(self, obs, prev_internal_state, prev_actions):
  63     """Core neural network taking in inputs and outputting sampling
  64     distribution parameters."""
  65     batch_size = tf.shape(obs[0])[0]
  66     if not self.recurrent:
  67       prev_internal_state = tf.zeros([batch_size, self.rnn_state_dim])
  68 
  69     cell = self.get_cell()
  70 
  71     b = tf.get_variable('input_bias', [self.cell_input_dim],
  72                         initializer=self.vector_init)
  73     cell_input = tf.nn.bias_add(tf.zeros([batch_size, self.cell_input_dim]), b)
  74 
  75     for i, (obs_dim, obs_type) in enumerate(self.env_spec.obs_dims_and_types):
  76       w = tf.get_variable('w_state%d' % i, [obs_dim, self.cell_input_dim],
  77                           initializer=self.matrix_init)
  78       if self.env_spec.is_discrete(obs_type):
  79         cell_input += tf.matmul(tf.one_hot(obs[i], obs_dim), w)
  80       elif self.env_spec.is_box(obs_type):
  81         cell_input += tf.matmul(obs[i], w)
  82       else:
  83         assert False
  84 
  85     if self.input_prev_actions:
  86       if self.env_spec.combine_actions:  # TODO(ofir): clean this up
  87         prev_action = prev_actions[0]
  88         for i, action_dim in enumerate(self.env_spec.orig_act_dims):
  89           act = tf.mod(prev_action, action_dim)
  90           w = tf.get_variable('w_prev_action%d' % i, [action_dim, self.cell_input_dim],
  91                               initializer=self.matrix_init)
  92           cell_input += tf.matmul(tf.one_hot(act, action_dim), w)
  93           prev_action = tf.to_int32(prev_action / action_dim)
  94       else:
  95         for i, (act_dim, act_type) in enumerate(self.env_spec.act_dims_and_types):
  96           w = tf.get_variable('w_prev_action%d' % i, [act_dim, self.cell_input_dim],
  97                               initializer=self.matrix_init)
  98           if self.env_spec.is_discrete(act_type):
  99             cell_input += tf.matmul(tf.one_hot(prev_actions[i], act_dim), w)
 100           elif self.env_spec.is_box(act_type):
 101             cell_input += tf.matmul(prev_actions[i], w)
 102           else:
 103             assert False
 104 
 105     output, next_state = cell(cell_input, prev_internal_state)
 106 
 107     return output, next_state

 109   def sample_action(self, logits, sampling_dim,
 110                     act_dim, act_type, greedy=False):
 111     """Sample an action from a distribution."""
 112     if self.env_spec.is_discrete(act_type):
 113       if greedy:
 114         act = tf.argmax(logits, 1)
 115       else:
 116         act = tf.reshape(tf.multinomial(logits, 1), [-1])
 117     elif self.env_spec.is_box(act_type):
 118       means = logits[:, :sampling_dim / 2]
 119       std = logits[:, sampling_dim / 2:]
 120       if greedy:
 121         act = means
 122       else:
 123         batch_size = tf.shape(logits)[0]
 124         act = means + std * tf.random_normal([batch_size, act_dim])
 125     else:
 126       assert False
 127 
 128     return act
 129 
 130   def entropy(self, logits,
 131               sampling_dim, act_dim, act_type):
 132     """Calculate entropy of distribution."""
 133     if self.env_spec.is_discrete(act_type):
 134       entropy = tf.reduce_sum(
 135           -tf.nn.softmax(logits) * tf.nn.log_softmax(logits), -1)
 136     elif self.env_spec.is_box(act_type):
 137       means = logits[:, :sampling_dim / 2]
 138       std = logits[:, sampling_dim / 2:]
 139       entropy = tf.reduce_sum(
 140           0.5 * (1 + tf.log(2 * np.pi * tf.square(std))), -1)
 141     else:
 142       assert False
 143 
 144     return entropy
 145 
 146   def self_kl(self, logits,
 147               sampling_dim, act_dim, act_type):
 148     """Calculate KL of distribution with itself.
 149 
 150     Used layer only for the gradients.
 151     """
 152 
 153     if self.env_spec.is_discrete(act_type):
 154       probs = tf.nn.softmax(logits)
 155       log_probs = tf.nn.log_softmax(logits)
 156       self_kl = tf.reduce_sum(
 157           tf.stop_gradient(probs) *
 158           (tf.stop_gradient(log_probs) - log_probs), -1)
 159     elif self.env_spec.is_box(act_type):
 160       means = logits[:, :sampling_dim / 2]
 161       std = logits[:, sampling_dim / 2:]
 162       my_means = tf.stop_gradient(means)
 163       my_std = tf.stop_gradient(std)
 164       self_kl = tf.reduce_sum(
 165           tf.log(std / my_std) +
 166           (tf.square(my_std) + tf.square(my_means - means)) /
 167           (2.0 * tf.square(std)) - 0.5,
 168           -1)
 169     else:
 170       assert False
 171 
 172     return self_kl
 173 
 174   def log_prob_action(self, action, logits,
 175                       sampling_dim, act_dim, act_type):
 176     """Calculate log-prob of action sampled from distribution."""
 177     if self.env_spec.is_discrete(act_type):
 178       act_log_prob = tf.reduce_sum(
 179           tf.one_hot(action, act_dim) * tf.nn.log_softmax(logits), -1)
 180     elif self.env_spec.is_box(act_type):
 181       means = logits[:, :sampling_dim / 2]
 182       std = logits[:, sampling_dim / 2:]
 183       act_log_prob = (- 0.5 * tf.log(2 * np.pi * tf.square(std))
 184                       - 0.5 * tf.square(action - means) / tf.square(std))
 185       act_log_prob = tf.reduce_sum(act_log_prob, -1)
 186     else:
 187       assert False
 188 
 189     return act_log_prob
 190 
 191   def sample_actions(self, output, actions=None, greedy=False):
 192     """Sample all actions given output of core network."""
 193     sampled_actions = []
 194     logits = []
 195     log_probs = []
 196     entropy = []
 197     self_kl = []
 198 
 199     start_idx = 0
 200     for i, (act_dim, act_type) in enumerate(self.env_spec.act_dims_and_types):
 201       sampling_dim = self.env_spec.sampling_dim(act_dim, act_type)
 202       if self.fixed_std and self.env_spec.is_box(act_type):
 203         act_logits = output[:, start_idx:start_idx + act_dim]
 204 
 205         log_std = tf.get_variable('std%d' % i, [1, sampling_dim // 2])
 206         # fix standard deviations to variable
 207         act_logits = tf.concat(
 208             [act_logits,
 209              1e-6 + tf.exp(log_std) + 0 * act_logits], 1)
 210       else:
 211         act_logits = output[:, start_idx:start_idx + sampling_dim]
 212 
 213       if actions is None:
 214         act = self.sample_action(act_logits, sampling_dim,
 215                                  act_dim, act_type,
 216                                  greedy=greedy)
 217       else:
 218         act = actions[i]
 219 
 220       ent = self.entropy(act_logits, sampling_dim, act_dim, act_type)
 221       kl = self.self_kl(act_logits, sampling_dim, act_dim, act_type)
 222 
 223       act_log_prob = self.log_prob_action(
 224           act, act_logits,
 225           sampling_dim, act_dim, act_type)
 226 
 227       sampled_actions.append(act)
 228       logits.append(act_logits)
 229       log_probs.append(act_log_prob)
 230       entropy.append(ent)
 231       self_kl.append(kl)
 232 
 233       start_idx += sampling_dim
 234 
 235     assert start_idx == self.env_spec.total_sampling_act_dim
 236 
 237     return sampled_actions, logits, log_probs, entropy, self_kl
 238 
 239   def get_kl(self, my_logits, other_logits):
 240     """Calculate KL between one policy output and another."""
 241     kl = []
 242     for i, (act_dim, act_type) in enumerate(self.env_spec.act_dims_and_types):
 243       sampling_dim = self.env_spec.sampling_dim(act_dim, act_type)
 244       single_my_logits = my_logits[i]
 245       single_other_logits = other_logits[i]
 246       if self.env_spec.is_discrete(act_type):
 247         my_probs = tf.nn.softmax(single_my_logits)
 248         my_log_probs = tf.nn.log_softmax(single_my_logits)
 249         other_log_probs = tf.nn.log_softmax(single_other_logits)
 250         my_kl = tf.reduce_sum(my_probs * (my_log_probs - other_log_probs), -1)
 251       elif self.env_spec.is_box(act_type):
 252         my_means = single_my_logits[:, :sampling_dim / 2]
 253         my_std = single_my_logits[:, sampling_dim / 2:]
 254         other_means = single_other_logits[:, :sampling_dim / 2]
 255         other_std = single_other_logits[:, sampling_dim / 2:]
 256         my_kl = tf.reduce_sum(
 257             tf.log(other_std / my_std) +
 258             (tf.square(my_std) + tf.square(my_means - other_means)) /
 259             (2.0 * tf.square(other_std)) - 0.5,
 260             -1)
 261       else:
 262         assert False
 263 
 264       kl.append(my_kl)
 265 
 266     return kl

 268   def single_step(self, prev, cur, greedy=False):
 269     """Single RNN step.  Equivalently, single-time-step sampled actions."""
 270     prev_internal_state, prev_actions, _, _, _, _ = prev
 271     obs, actions = cur  # state observed and action taken at this time step
 272 
 273     # feed into RNN cell
 274     output, next_state = self.core(
 275         obs, prev_internal_state, prev_actions)
 276 
 277     # sample actions with values and log-probs
 278     (actions, logits, log_probs,
 279      entropy, self_kl) = self.sample_actions(
 280         output, actions=actions, greedy=greedy)
 281 
 282     return (next_state, tuple(actions), tuple(logits), tuple(log_probs),
 283             tuple(entropy), tuple(self_kl))
 284 
 285   def sample_step(self, obs, prev_internal_state, prev_actions, greedy=False):
 286     """Sample single step from policy."""
 287     (next_state, sampled_actions, logits, log_probs,
 288      entropies, self_kls) = self.single_step(
 289         (prev_internal_state, prev_actions, None, None, None, None),
 290         (obs, None), greedy=greedy)
 291     return next_state, sampled_actions
 292 
 293   def multi_step(self, all_obs, initial_state, all_actions):
 294     """Calculate log-probs and other calculations on batch of episodes."""
 295     batch_size = tf.shape(initial_state)[0]
 296     time_length = tf.shape(all_obs[0])[0]
 297     initial_actions = [act[0] for act in all_actions]
 298     all_actions = [tf.concat([act[1:], act[0:1]], 0)
 299                    for act in all_actions]  # "final" action is dummy
 300 
 301     (internal_states, _, logits, log_probs,
 302      entropies, self_kls) = tf.scan(
 303         self.single_step,
 304         (all_obs, all_actions),
 305         initializer=self.get_initializer(
 306             batch_size, initial_state, initial_actions))
 307 
 308     # remove "final" computations
 309     log_probs = [log_prob[:-1] for log_prob in log_probs]
 310     entropies = [entropy[:-1] for entropy in entropies]
 311     self_kls = [self_kl[:-1] for self_kl in self_kls]
 312 
 313     return internal_states, logits, log_probs, entropies, self_kls
 314 
 315   def get_initializer(self, batch_size, initial_state, initial_actions):
 316     """Get initializer for RNN."""
 317     logits_init = []
 318     log_probs_init = []
 319     for act_dim, act_type in self.env_spec.act_dims_and_types:
 320       sampling_dim = self.env_spec.sampling_dim(act_dim, act_type)
 321       logits_init.append(tf.zeros([batch_size, sampling_dim]))
 322       log_probs_init.append(tf.zeros([batch_size]))
 323     entropy_init = [tf.zeros([batch_size]) for _ in self.env_spec.act_dims]
 324     self_kl_init = [tf.zeros([batch_size]) for _ in self.env_spec.act_dims]
 325 
 326     return (initial_state,
 327             tuple(initial_actions),
 328             tuple(logits_init), tuple(log_probs_init),
 329             tuple(entropy_init),
 330             tuple(self_kl_init))
 331 
 332   def calculate_kl(self, my_logits, other_logits):
 333     """Calculate KL between one policy and another on batch of episodes."""
 334     batch_size = tf.shape(my_logits[0])[1]
 335     time_length = tf.shape(my_logits[0])[0]
 336 
 337     reshaped_my_logits = [
 338         tf.reshape(my_logit, [batch_size * time_length, -1])
 339         for my_logit in my_logits]
 340     reshaped_other_logits = [
 341         tf.reshape(other_logit, [batch_size * time_length, -1])
 342         for other_logit in other_logits]
 343 
 344     kl = self.get_kl(reshaped_my_logits, reshaped_other_logits)
 345     kl = [tf.reshape(kkl, [time_length, batch_size])
 346           for kkl in kl]
 347     return kl

 350 class MLPPolicy(Policy):
 351   """Non-recurrent policy."""
 352 
 353   def get_cell(self):
 354     self.cell_input_dim = self.internal_dim
 355 
 356     def mlp(cell_input, prev_internal_state):
 357       w1 = tf.get_variable('w1', [self.cell_input_dim, self.internal_dim])
 358       b1 = tf.get_variable('b1', [self.internal_dim])
 359 
 360       w2 = tf.get_variable('w2', [self.internal_dim, self.internal_dim])
 361       b2 = tf.get_variable('b2', [self.internal_dim])
 362 
 363       w3 = tf.get_variable('w3', [self.internal_dim, self.internal_dim])
 364       b3 = tf.get_variable('b3', [self.internal_dim])
 365 
 366       proj = tf.get_variable(
 367           'proj', [self.internal_dim, self.output_dim])
 368 
 369       hidden = cell_input
 370       hidden = tf.tanh(tf.nn.bias_add(tf.matmul(hidden, w1), b1))
 371       hidden = tf.tanh(tf.nn.bias_add(tf.matmul(hidden, w2), b2))
 372 
 373       output = tf.matmul(hidden, proj)
 374 
 375       return output, hidden
 376 
 377     return mlp
 378 
 379   def single_step(self, obs, actions, prev_actions, greedy=False):
 380     """Single step."""
 381     batch_size = tf.shape(obs[0])[0]
 382     prev_internal_state = tf.zeros([batch_size, self.internal_dim])
 383 
 384     output, next_state = self.core(
 385         obs, prev_internal_state, prev_actions)
 386 
 387     # sample actions with values and log-probs
 388     (actions, logits, log_probs,
 389      entropy, self_kl) = self.sample_actions(
 390         output, actions=actions, greedy=greedy)
 391 
 392     return (next_state, tuple(actions), tuple(logits), tuple(log_probs),
 393             tuple(entropy), tuple(self_kl))
 394 
 395   def sample_step(self, obs, prev_internal_state, prev_actions, greedy=False):
 396     """Sample single step from policy."""
 397     (next_state, sampled_actions, logits, log_probs,
 398      entropies, self_kls) = self.single_step(obs, None, prev_actions,
 399                                              greedy=greedy)
 400     return next_state, sampled_actions

 402   def multi_step(self, all_obs, initial_state, all_actions):
 403     """Calculate log-probs and other calculations on batch of episodes."""
 404     batch_size = tf.shape(initial_state)[0]
 405     time_length = tf.shape(all_obs[0])[0]
 406 
 407     # first reshape inputs as a single batch
 408     reshaped_obs = []
 409     for obs, (obs_dim, obs_type) in zip(all_obs, self.env_spec.obs_dims_and_types):
 410       if self.env_spec.is_discrete(obs_type):
 411         reshaped_obs.append(tf.reshape(obs, [time_length * batch_size]))
 412       elif self.env_spec.is_box(obs_type):
 413         reshaped_obs.append(tf.reshape(obs, [time_length * batch_size, obs_dim]))
 414 
 415     reshaped_act = []
 416     reshaped_prev_act = []
 417     for i, (act_dim, act_type) in enumerate(self.env_spec.act_dims_and_types):
 418       act = tf.concat([all_actions[i][1:], all_actions[i][0:1]], 0)
 419       prev_act = all_actions[i]
 420       if self.env_spec.is_discrete(act_type):
 421         reshaped_act.append(tf.reshape(act, [time_length * batch_size]))
 422         reshaped_prev_act.append(
 423             tf.reshape(prev_act, [time_length * batch_size]))
 424       elif self.env_spec.is_box(act_type):
 425         reshaped_act.append(
 426             tf.reshape(act, [time_length * batch_size, act_dim]))
 427         reshaped_prev_act.append(
 428             tf.reshape(prev_act, [time_length * batch_size, act_dim]))
 429 
 430     # now inputs go into single step as one large batch
 431     (internal_states, _, logits, log_probs,
 432      entropies, self_kls) = self.single_step(
 433          reshaped_obs, reshaped_act, reshaped_prev_act)
 434 
 435     # reshape the outputs back to original time-major format
 436     internal_states = tf.reshape(internal_states, [time_length, batch_size, -1])
 437     logits = [tf.reshape(logit, [time_length, batch_size, -1])
 438               for logit in logits]
 439     log_probs = [tf.reshape(log_prob, [time_length, batch_size])[:-1]
 440                  for log_prob in log_probs]
 441     entropies = [tf.reshape(ent, [time_length, batch_size])[:-1]
 442                  for ent in entropies]
 443     self_kls = [tf.reshape(self_kl, [time_length, batch_size])[:-1]
 444                 for self_kl in self_kls]
 445 
 446     return internal_states, logits, log_probs, entropies, self_kls
