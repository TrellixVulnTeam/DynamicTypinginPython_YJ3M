  34 def var_size(v):
  35   return int(np.prod([int(d) for d in v.shape]))
  36 
  37 
  38 def gradients(loss, var_list):
  39   grads = tf.gradients(loss, var_list)
  40   return [g if g is not None else tf.zeros(v.shape)
  41           for g, v in zip(grads, var_list)]
  42 
  43 def flatgrad(loss, var_list):
  44   grads = gradients(loss, var_list)
  45   return tf.concat([tf.reshape(grad, [-1])
  46                     for (v, grad) in zip(var_list, grads)
  47                     if grad is not None], 0)
  48 
  49 
  50 def get_flat(var_list):
  51   return tf.concat([tf.reshape(v, [-1]) for v in var_list], 0)
  52 
  53 
  54 def set_from_flat(var_list, flat_theta):
  55   assigns = []
  56   shapes = [v.shape for v in var_list]
  57   sizes = [var_size(v) for v in var_list]
  58 
  59   start = 0
  60   assigns = []
  61   for (shape, size, v) in zip(shapes, sizes, var_list):
  62     assigns.append(v.assign(
  63         tf.reshape(flat_theta[start:start + size], shape)))
  64     start += size
  65   assert start == sum(sizes)
  66 
  67   return tf.group(*assigns)

  70 class LbfgsOptimization(object):
  71 
  72   def __init__(self, max_iter=25, mix_frac=1.0):
  73     self.max_iter = max_iter
  74     self.mix_frac = mix_frac
  75 
  76   def setup_placeholders(self):
  77     self.flat_theta = tf.placeholder(tf.float32, [None], 'flat_theta')
  78     self.intended_values = tf.placeholder(tf.float32, [None], 'intended_values')
  79 
  80   def setup(self, var_list, values, targets, pads,
  81             inputs, regression_weight):
  82     self.setup_placeholders()
  83     self.values = values
  84     self.targets = targets
  85 
  86     self.raw_loss = (tf.reduce_sum((1 - pads) * tf.square(values - self.intended_values))
  87                      / tf.reduce_sum(1 - pads))
  88     self.loss_flat_gradient = flatgrad(self.raw_loss, var_list)
  89 
  90     self.flat_vars = get_flat(var_list)
  91     self.set_vars = set_from_flat(var_list, self.flat_theta)
  92 
  93   def optimize(self, sess, feed_dict):
  94     old_theta = sess.run(self.flat_vars)
  95 
  96     old_values, targets = sess.run([self.values, self.targets], feed_dict=feed_dict)
  97     intended_values = targets * self.mix_frac + old_values * (1 - self.mix_frac)
  98     feed_dict = dict(feed_dict)
  99     feed_dict[self.intended_values] = intended_values
 100 
 101     def calc_loss_and_grad(theta):
 102       sess.run(self.set_vars, feed_dict={self.flat_theta: theta})
 103       loss, grad = sess.run([self.raw_loss, self.loss_flat_gradient],
 104                             feed_dict=feed_dict)
 105       grad = grad.astype('float64')
 106       return loss, grad
 107 
 108     theta, _, _ = scipy.optimize.fmin_l_bfgs_b(
 109         calc_loss_and_grad, old_theta, maxiter=self.max_iter)
 110     sess.run(self.set_vars, feed_dict={self.flat_theta: theta})

 113 class GradOptimization(object):
 114 
 115   def __init__(self, learning_rate=0.001, max_iter=25, mix_frac=1.0):
 116     self.learning_rate = learning_rate
 117     self.max_iter = max_iter
 118     self.mix_frac = mix_frac
 119 
 120   def get_optimizer(self):
 121     return tf.train.AdamOptimizer(learning_rate=self.learning_rate,
 122                                   epsilon=2e-4)
 123 
 124   def setup_placeholders(self):
 125     self.flat_theta = tf.placeholder(tf.float32, [None], 'flat_theta')
 126     self.intended_values = tf.placeholder(tf.float32, [None], 'intended_values')
 127 
 128   def setup(self, var_list, values, targets, pads,
 129             inputs, regression_weight):
 130     self.setup_placeholders()
 131     self.values = values
 132     self.targets = targets
 133 
 134     self.raw_loss = (tf.reduce_sum((1 - pads) * tf.square(values - self.intended_values))
 135                      / tf.reduce_sum(1 - pads))
 136 
 137     opt = self.get_optimizer()
 138     params = var_list
 139     grads = tf.gradients(self.raw_loss, params)
 140     self.gradient_ops = opt.apply_gradients(zip(grads, params))
 141 
 142   def optimize(self, sess, feed_dict):
 143     old_values, targets = sess.run([self.values, self.targets], feed_dict=feed_dict)
 144     intended_values = targets * self.mix_frac + old_values * (1 - self.mix_frac)
 145 
 146     feed_dict = dict(feed_dict)
 147     feed_dict[self.intended_values] = intended_values
 148 
 149     for _ in xrange(self.max_iter):
 150       sess.run(self.gradient_ops, feed_dict=feed_dict)

 153 class BestFitOptimization(object):
 154 
 155   def __init__(self, mix_frac=1.0):
 156     self.mix_frac = mix_frac
 157 
 158   def setup_placeholders(self):
 159     self.new_regression_weight = tf.placeholder(
 160         tf.float32, self.regression_weight.shape)
 161 
 162   def setup(self, var_list, values, targets, pads,
 163             inputs, regression_weight):
 164     self.values = values
 165     self.targets = targets
 166 
 167     self.inputs = inputs
 168     self.regression_weight = regression_weight
 169 
 170     self.setup_placeholders()
 171 
 172     self.update_regression_weight = tf.assign(
 173         self.regression_weight, self.new_regression_weight)
 174 
 175   def optimize(self, sess, feed_dict):
 176     reg_input, reg_weight, old_values, targets = sess.run(
 177         [self.inputs, self.regression_weight, self.values, self.targets],
 178         feed_dict=feed_dict)
 179 
 180     intended_values = targets * self.mix_frac + old_values * (1 - self.mix_frac)
 181 
 182     # taken from rllab
 183     reg_coeff = 1e-5
 184     for _ in range(5):
 185       best_fit_weight = np.linalg.lstsq(
 186           reg_input.T.dot(reg_input) +
 187           reg_coeff * np.identity(reg_input.shape[1]),
 188           reg_input.T.dot(intended_values))[0]
 189       if not np.any(np.isnan(best_fit_weight)):
 190         break
 191       reg_coeff *= 10
 192 
 193     if len(best_fit_weight.shape) == 1:
 194       best_fit_weight = np.expand_dims(best_fit_weight, -1)
 195 
 196     sess.run(self.update_regression_weight,
 197              feed_dict={self.new_regression_weight: best_fit_weight})
