 144 class Trainer(object):
 145   """Coordinates single or multi-replica training."""
 146 
 147   def __init__(self):
 148     self.batch_size = FLAGS.batch_size
 149     self.replay_batch_size = FLAGS.replay_batch_size
 150     if self.replay_batch_size is None:
 151       self.replay_batch_size = self.batch_size
 152     self.num_samples = FLAGS.num_samples
 153 
 154     self.env_str = FLAGS.env
 155     self.env = gym_wrapper.GymWrapper(self.env_str,
 156                                       distinct=FLAGS.batch_size // self.num_samples,
 157                                       count=self.num_samples)
 158     self.eval_env = gym_wrapper.GymWrapper(
 159         self.env_str,
 160         distinct=FLAGS.batch_size // self.num_samples,
 161         count=self.num_samples)
 162     self.env_spec = env_spec.EnvSpec(self.env.get_one())
 163 
 164     self.max_step = FLAGS.max_step
 165     self.cutoff_agent = FLAGS.cutoff_agent
 166     self.num_steps = FLAGS.num_steps
 167     self.validation_frequency = FLAGS.validation_frequency
 168 
 169     self.target_network_lag = FLAGS.target_network_lag
 170     self.sample_from = FLAGS.sample_from
 171     assert self.sample_from in ['online', 'target']
 172 
 173     self.critic_weight = FLAGS.critic_weight
 174     self.objective = FLAGS.objective
 175     self.trust_region_p = FLAGS.trust_region_p
 176     self.value_opt = FLAGS.value_opt
 177     assert not self.trust_region_p or self.objective in ['pcl', 'trpo']
 178     assert self.objective != 'trpo' or self.trust_region_p
 179     assert self.value_opt is None or self.value_opt == 'None' or \
 180         self.critic_weight == 0.0
 181     self.max_divergence = FLAGS.max_divergence
 182 
 183     self.learning_rate = FLAGS.learning_rate
 184     self.clip_norm = FLAGS.clip_norm
 185     self.clip_adv = FLAGS.clip_adv
 186     self.tau = FLAGS.tau
 187     self.tau_decay = FLAGS.tau_decay
 188     self.tau_start = FLAGS.tau_start
 189     self.eps_lambda = FLAGS.eps_lambda
 190     self.update_eps_lambda = FLAGS.update_eps_lambda
 191     self.gamma = FLAGS.gamma
 192     self.rollout = FLAGS.rollout
 193     self.use_target_values = FLAGS.use_target_values
 194     self.fixed_std = FLAGS.fixed_std
 195     self.input_prev_actions = FLAGS.input_prev_actions
 196     self.recurrent = FLAGS.recurrent
 197     assert not self.trust_region_p or not self.recurrent
 198     self.input_time_step = FLAGS.input_time_step
 199     assert not self.input_time_step or (self.cutoff_agent <= self.max_step)
 200 
 201     self.use_online_batch = FLAGS.use_online_batch
 202     self.batch_by_steps = FLAGS.batch_by_steps
 203     self.unify_episodes = FLAGS.unify_episodes
 204     if self.unify_episodes:
 205       assert self.batch_size == 1
 206 
 207     self.replay_buffer_size = FLAGS.replay_buffer_size
 208     self.replay_buffer_alpha = FLAGS.replay_buffer_alpha
 209     self.replay_buffer_freq = FLAGS.replay_buffer_freq
 210     assert self.replay_buffer_freq in [-1, 0, 1]
 211     self.eviction = FLAGS.eviction
 212     self.prioritize_by = FLAGS.prioritize_by
 213     assert self.prioritize_by in ['rewards', 'step']
 214     self.num_expert_paths = FLAGS.num_expert_paths
 215 
 216     self.internal_dim = FLAGS.internal_dim
 217     self.value_hidden_layers = FLAGS.value_hidden_layers
 218     self.tf_seed = FLAGS.tf_seed
 219 
 220     self.save_trajectories_dir = FLAGS.save_trajectories_dir
 221     self.save_trajectories_file = (
 222         os.path.join(
 223             self.save_trajectories_dir, self.env_str.replace('-', '_'))
 224         if self.save_trajectories_dir else None)
 225     self.load_trajectories_file = FLAGS.load_trajectories_file
 226 
 227     self.hparams = dict((attr, getattr(self, attr))
 228                         for attr in dir(self)
 229                         if not attr.startswith('__') and
 230                         not callable(getattr(self, attr)))
 231 
 232   def hparams_string(self):
 233     return '\n'.join('%s: %s' % item for item in sorted(self.hparams.items()))
 234 
 235   def get_objective(self):
 236     tau = self.tau
 237     if self.tau_decay is not None:
 238       assert self.tau_start >= self.tau
 239       tau = tf.maximum(
 240           tf.train.exponential_decay(
 241               self.tau_start, self.global_step, 100, self.tau_decay),
 242           self.tau)
 243 
 244     if self.objective in ['pcl', 'a3c', 'trpo', 'upcl']:
 245       cls = (objective.PCL if self.objective in ['pcl', 'upcl'] else
 246              objective.TRPO if self.objective == 'trpo' else
 247              objective.ActorCritic)
 248       policy_weight = 1.0
 249 
 250       return cls(self.learning_rate,
 251                  clip_norm=self.clip_norm,
 252                  policy_weight=policy_weight,
 253                  critic_weight=self.critic_weight,
 254                  tau=tau, gamma=self.gamma, rollout=self.rollout,
 255                  eps_lambda=self.eps_lambda, clip_adv=self.clip_adv,
 256                  use_target_values=self.use_target_values)
 257     elif self.objective in ['reinforce', 'urex']:
 258       cls = (full_episode_objective.Reinforce
 259              if self.objective == 'reinforce' else
 260              full_episode_objective.UREX)
 261       return cls(self.learning_rate,
 262                  clip_norm=self.clip_norm,
 263                  num_samples=self.num_samples,
 264                  tau=tau, bonus_weight=1.0)  # TODO: bonus weight?
 265     else:
 266       assert False, 'Unknown objective %s' % self.objective
 267 
 268   def get_policy(self):
 269     if self.recurrent:
 270       cls = policy.Policy
 271     else:
 272       cls = policy.MLPPolicy
 273     return cls(self.env_spec, self.internal_dim,
 274                fixed_std=self.fixed_std,
 275                recurrent=self.recurrent,
 276                input_prev_actions=self.input_prev_actions)
 277 
 278   def get_baseline(self):
 279     cls = (baseline.UnifiedBaseline if self.objective == 'upcl' else
 280            baseline.Baseline)
 281     return cls(self.env_spec, self.internal_dim,
 282                input_prev_actions=self.input_prev_actions,
 283                input_time_step=self.input_time_step,
 284                input_policy_state=self.recurrent,  # may want to change this
 285                n_hidden_layers=self.value_hidden_layers,
 286                hidden_dim=self.internal_dim,
 287                tau=self.tau)
 288 
 289   def get_trust_region_p_opt(self):
 290     if self.trust_region_p:
 291       return trust_region.TrustRegionOptimization(
 292           max_divergence=self.max_divergence)
 293     else:
 294       return None
 295 
 296   def get_value_opt(self):
 297     if self.value_opt == 'grad':
 298       return optimizers.GradOptimization(
 299           learning_rate=self.learning_rate, max_iter=5, mix_frac=0.05)
 300     elif self.value_opt == 'lbfgs':
 301       return optimizers.LbfgsOptimization(max_iter=25, mix_frac=0.1)
 302     elif self.value_opt == 'best_fit':
 303       return optimizers.BestFitOptimization(mix_frac=1.0)
 304     else:
 305       return None
 306 
 307   def get_model(self):
 308     cls = model.Model
 309     return cls(self.env_spec, self.global_step,
 310                target_network_lag=self.target_network_lag,
 311                sample_from=self.sample_from,
 312                get_policy=self.get_policy,
 313                get_baseline=self.get_baseline,
 314                get_objective=self.get_objective,
 315                get_trust_region_p_opt=self.get_trust_region_p_opt,
 316                get_value_opt=self.get_value_opt)
 317 
 318   def get_replay_buffer(self):
 319     if self.replay_buffer_freq <= 0:
 320       return None
 321     else:
 322       assert self.objective in ['pcl', 'upcl'], 'Can\'t use replay buffer with %s' % (
 323           self.objective)
 324     cls = replay_buffer.PrioritizedReplayBuffer
 325     return cls(self.replay_buffer_size,
 326                alpha=self.replay_buffer_alpha,
 327                eviction_strategy=self.eviction)
 328 
 329   def get_buffer_seeds(self):
 330     return expert_paths.sample_expert_paths(
 331         self.num_expert_paths, self.env_str, self.env_spec,
 332         load_trajectories_file=self.load_trajectories_file)
 333 
 334   def get_controller(self, env):
 335     """Get controller."""
 336     cls = controller.Controller
 337     return cls(env, self.env_spec, self.internal_dim,
 338                use_online_batch=self.use_online_batch,
 339                batch_by_steps=self.batch_by_steps,
 340                unify_episodes=self.unify_episodes,
 341                replay_batch_size=self.replay_batch_size,
 342                max_step=self.max_step,
 343                cutoff_agent=self.cutoff_agent,
 344                save_trajectories_file=self.save_trajectories_file,
 345                use_trust_region=self.trust_region_p,
 346                use_value_opt=self.value_opt not in [None, 'None'],
 347                update_eps_lambda=self.update_eps_lambda,
 348                prioritize_by=self.prioritize_by,
 349                get_model=self.get_model,
 350                get_replay_buffer=self.get_replay_buffer,
 351                get_buffer_seeds=self.get_buffer_seeds)
 352 
 353   def do_before_step(self, step):
 354     pass
 355 
 356   def run(self):
 357     """Run training."""
 358     is_chief = FLAGS.task_id == 0 or not FLAGS.supervisor
 359     sv = None
 360 
 361     def init_fn(sess, saver):
 362       ckpt = None
 363       if FLAGS.save_dir and sv is None:
 364         load_dir = FLAGS.save_dir
 365         ckpt = tf.train.get_checkpoint_state(load_dir)
 366       if ckpt and ckpt.model_checkpoint_path:
 367         logging.info('restoring from %s', ckpt.model_checkpoint_path)
 368         saver.restore(sess, ckpt.model_checkpoint_path)
 369       elif FLAGS.load_path:
 370         logging.info('restoring from %s', FLAGS.load_path)
 371         saver.restore(sess, FLAGS.load_path)
 372 
 373     if FLAGS.supervisor:
 374       with tf.device(tf.ReplicaDeviceSetter(FLAGS.ps_tasks, merge_devices=True)):
 375         self.global_step = tf.contrib.framework.get_or_create_global_step()
 376         tf.set_random_seed(FLAGS.tf_seed)
 377         self.controller = self.get_controller(self.env)
 378         self.model = self.controller.model
 379         self.controller.setup()
 380         with tf.variable_scope(tf.get_variable_scope(), reuse=True):
 381           self.eval_controller = self.get_controller(self.eval_env)
 382           self.eval_controller.setup(train=False)
 383 
 384         saver = tf.train.Saver(max_to_keep=10)
 385         step = self.model.global_step
 386         sv = tf.Supervisor(logdir=FLAGS.save_dir,
 387                            is_chief=is_chief,
 388                            saver=saver,
 389                            save_model_secs=600,
 390                            summary_op=None,  # we define it ourselves
 391                            save_summaries_secs=60,
 392                            global_step=step,
 393                            init_fn=lambda sess: init_fn(sess, saver))
 394         sess = sv.PrepareSession(FLAGS.master)
 395     else:
 396       tf.set_random_seed(FLAGS.tf_seed)
 397       self.global_step = tf.contrib.framework.get_or_create_global_step()
 398       self.controller = self.get_controller(self.env)
 399       self.model = self.controller.model
 400       self.controller.setup()
 401       with tf.variable_scope(tf.get_variable_scope(), reuse=True):
 402         self.eval_controller = self.get_controller(self.eval_env)
 403         self.eval_controller.setup(train=False)
 404 
 405       saver = tf.train.Saver(max_to_keep=10)
 406       sess = tf.Session()
 407       sess.run(tf.initialize_all_variables())
 408       init_fn(sess, saver)
 409 
 410     self.sv = sv
 411     self.sess = sess
 412 
 413     logging.info('hparams:\n%s', self.hparams_string())
 414 
 415     model_step = sess.run(self.model.global_step)
 416     if model_step >= self.num_steps:
 417       logging.info('training has reached final step')
 418       return
 419 
 420     losses = []
 421     rewards = []
 422     all_ep_rewards = []
 423     for step in xrange(1 + self.num_steps):
 424 
 425       if sv is not None and sv.ShouldStop():
 426         logging.info('stopping supervisor')
 427         break
 428 
 429       self.do_before_step(step)
 430 
 431       (loss, summary,
 432        total_rewards, episode_rewards) = self.controller.train(sess)
 433       _, greedy_episode_rewards = self.eval_controller.eval(sess)
 434       self.controller.greedy_episode_rewards = greedy_episode_rewards
 435       losses.append(loss)
 436       rewards.append(total_rewards)
 437       all_ep_rewards.extend(episode_rewards)
 438 
 439       if (random.random() < 0.1 and summary and episode_rewards and
 440           is_chief and sv and sv._summary_writer):
 441         sv.summary_computed(sess, summary)
 442 
 443       model_step = sess.run(self.model.global_step)
 444       if is_chief and step % self.validation_frequency == 0:
 445         logging.info('at training step %d, model step %d: '
 446                      'avg loss %f, avg reward %f, '
 447                      'episode rewards: %f, greedy rewards: %f',
 448                      step, model_step,
 449                      np.mean(losses), np.mean(rewards),
 450                      np.mean(all_ep_rewards),
 451                      np.mean(greedy_episode_rewards))
 452 
 453         losses = []
 454         rewards = []
 455         all_ep_rewards = []
 456 
 457       if model_step >= self.num_steps:
 458         logging.info('training has reached final step')
 459         break
 460 
 461     if is_chief and sv is not None:
 462       logging.info('saving final model to %s', sv.save_path)
 463       sv.saver.save(sess, sv.save_path, global_step=sv.global_step)
 464 
 465 
 466 def main(unused_argv):
 467   logging.set_verbosity(logging.INFO)
 468   trainer = Trainer()
 469   trainer.run()

 473   app.run()
