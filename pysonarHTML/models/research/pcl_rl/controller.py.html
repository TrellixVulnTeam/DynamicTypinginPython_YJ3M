  35 def find_best_eps_lambda(rewards, lengths):
  36   """Find the best lambda given a desired epsilon = FLAGS.max_divergence."""
  37   # perhaps not the best way to do this
  38   desired_div = FLAGS.max_divergence * np.mean(lengths)
  39 
  40   def calc_divergence(eps_lambda):
  41     max_reward = np.max(rewards)
  42     logz = (max_reward / eps_lambda +
  43             np.log(np.mean(np.exp((rewards - max_reward) / eps_lambda))))
  44     exprr = np.mean(np.exp(rewards / eps_lambda - logz) *
  45                     rewards / eps_lambda)
  46     return exprr - logz
  47 
  48   left = 0.0
  49   right = 1000.0
  50 
  51   if len(rewards) <= 8:
  52     return (left + right) / 2
  53 
  54   num_iter = max(4, 1 + int(np.log((right - left) / 0.1) / np.log(2.0)))
  55   for _ in xrange(num_iter):
  56     mid = (left + right) / 2
  57     cur_div = calc_divergence(mid)
  58     if cur_div > desired_div:
  59       left = mid
  60     else:
  61       right = mid
  62 
  63   return (left + right) / 2

  68   def __init__(self, env, env_spec, internal_dim,
  69                use_online_batch=True,
  70                batch_by_steps=False,
  71                unify_episodes=False,
  72                replay_batch_size=None,
  73                max_step=None,
  74                cutoff_agent=1,
  75                save_trajectories_file=None,
  76                use_trust_region=False,
  77                use_value_opt=False,
  78                update_eps_lambda=False,
  79                prioritize_by='rewards',
  80                get_model=None,
  81                get_replay_buffer=None,
  82                get_buffer_seeds=None):
  83     self.env = env
  84     self.env_spec = env_spec
  85     self.internal_dim = internal_dim
  86     self.use_online_batch = use_online_batch
  87     self.batch_by_steps = batch_by_steps
  88     self.unify_episodes = unify_episodes
  89     self.replay_batch_size = replay_batch_size
  90     self.max_step = max_step
  91     self.cutoff_agent = cutoff_agent
  92     self.save_trajectories_file = save_trajectories_file
  93     self.use_trust_region = use_trust_region
  94     self.use_value_opt = use_value_opt
  95     self.update_eps_lambda = update_eps_lambda
  96     self.prioritize_by = prioritize_by
  97 
  98     self.model = get_model()
  99     self.replay_buffer = get_replay_buffer()
 100     self.seed_replay_buffer(get_buffer_seeds())
 101 
 102     self.internal_state = np.array([self.initial_internal_state()] *
 103                                    len(self.env))
 104     self.last_obs = self.env_spec.initial_obs(len(self.env))
 105     self.last_act = self.env_spec.initial_act(len(self.env))
 106     self.last_pad = np.zeros(len(self.env))
 107 
 108     self.start_episode = np.array([True] * len(self.env))
 109     self.step_count = np.array([0] * len(self.env))
 110     self.episode_running_rewards = np.zeros(len(self.env))
 111     self.episode_running_lengths = np.zeros(len(self.env))
 112     self.episode_rewards = []
 113     self.greedy_episode_rewards = []
 114     self.episode_lengths = []
 115     self.total_rewards = []
 116 
 117     self.best_batch_rewards = None

 122   def initial_internal_state(self):
 123     return np.zeros(self.model.policy.rnn_state_dim)
 124 
 125   def _sample_episodes(self, sess, greedy=False):
 126     """Sample episodes from environment using model."""
 127     # reset environments as necessary
 128     obs_after_reset = self.env.reset_if(self.start_episode)
 129 
 130     for i, obs in enumerate(obs_after_reset):
 131       if obs is not None:
 132         self.step_count[i] = 0
 133         self.internal_state[i] = self.initial_internal_state()
 134         for j in xrange(len(self.env_spec.obs_dims)):
 135           self.last_obs[j][i] = obs[j]
 136         for j in xrange(len(self.env_spec.act_dims)):
 137           self.last_act[j][i] = -1
 138         self.last_pad[i] = 0
 139 
 140     # maintain episode as a single unit if the last sampling
 141     # batch ended before the episode was terminated
 142     if self.unify_episodes:
 143       assert len(obs_after_reset) == 1
 144       new_ep = obs_after_reset[0] is not None
 145     else:
 146       new_ep = True
 147 
 148     self.start_id = 0 if new_ep else len(self.all_obs[:])
 149 
 150     initial_state = self.internal_state
 151     all_obs = [] if new_ep else self.all_obs[:]
 152     all_act = ([self.last_act] if new_ep else self.all_act[:])
 153     all_pad = [] if new_ep else self.all_pad[:]
 154     rewards = [] if new_ep else self.rewards[:]
 155 
 156     # start stepping in the environments
 157     step = 0
 158     while not self.env.all_done():
 159       self.step_count += 1 - np.array(self.env.dones)
 160 
 161       next_internal_state, sampled_actions = self.model.sample_step(
 162           sess, self.last_obs, self.internal_state, self.last_act,
 163           greedy=greedy)
 164 
 165       env_actions = self.env_spec.convert_actions_to_env(sampled_actions)
 166       next_obs, reward, next_dones, _ = self.env.step(env_actions)
 167 
 168       all_obs.append(self.last_obs)
 169       all_act.append(sampled_actions)
 170       all_pad.append(self.last_pad)
 171       rewards.append(reward)
 172 
 173       self.internal_state = next_internal_state
 174       self.last_obs = next_obs
 175       self.last_act = sampled_actions
 176       self.last_pad = np.array(next_dones).astype('float32')
 177 
 178       step += 1
 179       if self.max_step and step >= self.max_step:
 180         break
 181 
 182     self.all_obs = all_obs[:]
 183     self.all_act = all_act[:]
 184     self.all_pad = all_pad[:]
 185     self.rewards = rewards[:]
 186 
 187     # append final observation
 188     all_obs.append(self.last_obs)
 189 
 190     return initial_state, all_obs, all_act, rewards, all_pad
 191 
 192   def sample_episodes(self, sess, greedy=False):
 193     """Sample steps from the environment until we have enough for a batch."""
 194 
 195     # check if last batch ended with episode that was not terminated
 196     if self.unify_episodes:
 197       self.all_new_ep = self.start_episode[0]
 198 
 199     # sample episodes until we either have enough episodes or enough steps
 200     episodes = []
 201     total_steps = 0
 202     while total_steps < self.max_step * len(self.env):
 203       (initial_state,
 204        observations, actions, rewards,
 205        pads) = self._sample_episodes(sess, greedy=greedy)
 206 
 207       observations = zip(*observations)
 208       actions = zip(*actions)
 209 
 210       terminated = np.array(self.env.dones)
 211 
 212       self.total_rewards = np.sum(np.array(rewards[self.start_id:]) *
 213                                   (1 - np.array(pads[self.start_id:])), axis=0)
 214       self.episode_running_rewards *= 1 - self.start_episode
 215       self.episode_running_lengths *= 1 - self.start_episode
 216       self.episode_running_rewards += self.total_rewards
 217       self.episode_running_lengths += np.sum(1 - np.array(pads[self.start_id:]), axis=0)
 218 
 219       episodes.extend(self.convert_from_batched_episodes(
 220           initial_state, observations, actions, rewards,
 221           terminated, pads))
 222       total_steps += np.sum(1 - np.array(pads))
 223 
 224       # set next starting episodes
 225       self.start_episode = np.logical_or(terminated,
 226                                          self.step_count >= self.cutoff_agent)
 227       episode_rewards = self.episode_running_rewards[self.start_episode].tolist()
 228       self.episode_rewards.extend(episode_rewards)
 229       self.episode_lengths.extend(self.episode_running_lengths[self.start_episode].tolist())
 230       self.episode_rewards = self.episode_rewards[-100:]
 231       self.episode_lengths = self.episode_lengths[-100:]
 232 
 233       if (self.save_trajectories_file is not None and
 234           (self.best_batch_rewards is None or
 235            np.mean(self.total_rewards) > self.best_batch_rewards)):
 236         self.best_batch_rewards = np.mean(self.total_rewards)
 237         my_episodes = self.convert_from_batched_episodes(
 238           initial_state, observations, actions, rewards,
 239           terminated, pads)
 240         with gfile.GFile(self.save_trajectories_file, 'w') as f:
 241           pickle.dump(my_episodes, f)
 242 
 243       if not self.batch_by_steps:
 244         return (initial_state,
 245                 observations, actions, rewards,
 246                 terminated, pads)
 247 
 248     return self.convert_to_batched_episodes(episodes)
 249 
 250   def _train(self, sess,
 251              observations, initial_state, actions,
 252              rewards, terminated, pads):
 253     """Train model using batch."""
 254     avg_episode_reward = np.mean(self.episode_rewards)
 255     greedy_episode_reward = (np.mean(self.greedy_episode_rewards)
 256                              if self.greedy_episode_rewards else
 257                              avg_episode_reward)
 258     loss, summary = None, None
 259     if self.use_trust_region:
 260       # use trust region to optimize policy
 261       loss, _, summary = self.model.trust_region_step(
 262           sess,
 263           observations, initial_state, actions,
 264           rewards, terminated, pads,
 265           avg_episode_reward=avg_episode_reward,
 266           greedy_episode_reward=greedy_episode_reward)
 267     else:  # otherwise use simple gradient descent on policy
 268       loss, _, summary = self.model.train_step(
 269           sess,
 270           observations, initial_state, actions,
 271           rewards, terminated, pads,
 272           avg_episode_reward=avg_episode_reward,
 273           greedy_episode_reward=greedy_episode_reward)
 274 
 275     if self.use_value_opt:  # optionally perform specific value optimization
 276       self.model.fit_values(
 277           sess,
 278           observations, initial_state, actions,
 279           rewards, terminated, pads)
 280 
 281     return loss, summary

 283   def train(self, sess):
 284     """Sample some episodes and train on some episodes."""
 285     cur_step = sess.run(self.model.inc_global_step)
 286     self.cur_step = cur_step
 287 
 288     # on the first iteration, set target network close to online network
 289     if self.cur_step == 0:
 290       for _ in xrange(100):
 291         sess.run(self.model.copy_op)
 292     # on other iterations, just perform single target <-- online operation
 293     sess.run(self.model.copy_op)
 294 
 295     # sample from env
 296     (initial_state,
 297      observations, actions, rewards,
 298      terminated, pads) = self.sample_episodes(sess)
 299 
 300     # add to replay buffer
 301     self.add_to_replay_buffer(
 302         initial_state, observations, actions,
 303         rewards, terminated, pads)
 304 
 305     loss, summary = 0, None
 306     # train on online batch
 307     if self.use_online_batch:
 308       loss, summary = self._train(
 309           sess,
 310           observations, initial_state, actions,
 311           rewards, terminated, pads)
 312 
 313     # update relative entropy coefficient
 314     if self.update_eps_lambda:
 315       episode_rewards = np.array(self.episode_rewards)
 316       episode_lengths = np.array(self.episode_lengths)
 317       eps_lambda = find_best_eps_lambda(
 318           episode_rewards[-20:], episode_lengths[-20:])
 319       sess.run(self.model.objective.assign_eps_lambda,
 320                feed_dict={self.model.objective.new_eps_lambda: eps_lambda})
 321 
 322     # train on replay batch
 323     replay_batch, replay_probs = self.get_from_replay_buffer(
 324         self.replay_batch_size)
 325     if replay_batch:
 326       (initial_state,
 327        observations, actions, rewards,
 328        terminated, pads) = replay_batch
 329 
 330       loss, summary = self._train(
 331           sess,
 332           observations, initial_state, actions,
 333           rewards, terminated, pads)
 334 
 335     return loss, summary, self.total_rewards, self.episode_rewards
 336 
 337   def eval(self, sess):
 338     """Use greedy sampling."""
 339     (initial_state,
 340      observations, actions, rewards,
 341      pads, terminated) = self.sample_episodes(sess, greedy=True)
 342 
 343     total_rewards = np.sum(np.array(rewards) * (1 - np.array(pads)), axis=0)
 344     return total_rewards, self.episode_rewards

 346   def convert_from_batched_episodes(
 347       self, initial_state, observations, actions, rewards,
 348       terminated, pads):
 349     """Convert time-major batch of episodes to batch-major list of episodes."""
 350 
 351     rewards = np.array(rewards)
 352     pads = np.array(pads)
 353     observations = [np.array(obs) for obs in observations]
 354     actions = [np.array(act) for act in actions]
 355 
 356     total_rewards = np.sum(rewards * (1 - pads), axis=0)
 357     total_length = np.sum(1 - pads, axis=0).astype('int32')
 358 
 359     episodes = []
 360     num_episodes = rewards.shape[1]
 361     for i in xrange(num_episodes):
 362       length = total_length[i]
 363       ep_initial = initial_state[i]
 364       ep_obs = [obs[:length + 1, i, ...] for obs in observations]
 365       ep_act = [act[:length + 1, i, ...] for act in actions]
 366       ep_rewards = rewards[:length, i]
 367 
 368       episodes.append(
 369           [ep_initial, ep_obs, ep_act, ep_rewards, terminated[i]])
 370 
 371     return episodes

 373   def convert_to_batched_episodes(self, episodes, max_length=None):
 374     """Convert batch-major list of episodes to time-major batch of episodes."""
 375     lengths = [len(ep[-2]) for ep in episodes]
 376     max_length = max_length or max(lengths)
 377 
 378     new_episodes = []
 379     for ep, length in zip(episodes, lengths):
 380       initial, observations, actions, rewards, terminated = ep
 381       observations = [np.resize(obs, [max_length + 1] + list(obs.shape)[1:])
 382                       for obs in observations]
 383       actions = [np.resize(act, [max_length + 1] + list(act.shape)[1:])
 384                  for act in actions]
 385       pads = np.array([0] * length + [1] * (max_length - length))
 386       rewards = np.resize(rewards, [max_length]) * (1 - pads)
 387       new_episodes.append([initial, observations, actions, rewards,
 388                            terminated, pads])
 389 
 390     (initial, observations, actions, rewards,
 391      terminated, pads) = zip(*new_episodes)
 392     observations = [np.swapaxes(obs, 0, 1)
 393                     for obs in zip(*observations)]
 394     actions = [np.swapaxes(act, 0, 1)
 395                for act in zip(*actions)]
 396     rewards = np.transpose(rewards)
 397     pads = np.transpose(pads)
 398 
 399     return (initial, observations, actions, rewards, terminated, pads)

 401   def add_to_replay_buffer(self, initial_state,
 402                            observations, actions, rewards,
 403                            terminated, pads):
 404     """Add batch of episodes to replay buffer."""
 405     if self.replay_buffer is None:
 406       return
 407 
 408     rewards = np.array(rewards)
 409     pads = np.array(pads)
 410     total_rewards = np.sum(rewards * (1 - pads), axis=0)
 411 
 412     episodes = self.convert_from_batched_episodes(
 413       initial_state, observations, actions, rewards,
 414       terminated, pads)
 415 
 416     priorities = (total_rewards if self.prioritize_by == 'reward'
 417                   else self.cur_step)
 418 
 419     if not self.unify_episodes or self.all_new_ep:
 420       self.last_idxs = self.replay_buffer.add(
 421           episodes, priorities)
 422     else:
 423       # If we are unifying episodes, we attempt to
 424       # keep them unified in the replay buffer.
 425       # The first episode sampled in the current batch is a
 426       # continuation of the last episode from the previous batch
 427       self.replay_buffer.add(episodes[:1], priorities, self.last_idxs[-1:])
 428       if len(episodes) > 1:
 429         self.replay_buffer.add(episodes[1:], priorities)
 430 
 431   def get_from_replay_buffer(self, batch_size):
 432     """Sample a batch of episodes from the replay buffer."""
 433     if self.replay_buffer is None or len(self.replay_buffer) < 1 * batch_size:
 434       return None, None
 435 
 436     desired_count = batch_size * self.max_step
 437     # in the case of batch_by_steps, we sample larger and larger
 438     # amounts from the replay buffer until we have enough steps.
 439     while True:
 440       if batch_size > len(self.replay_buffer):
 441         batch_size = len(self.replay_buffer)
 442       episodes, probs = self.replay_buffer.get_batch(batch_size)
 443       count = sum(len(ep[-2]) for ep in episodes)
 444       if count >= desired_count or not self.batch_by_steps:
 445         break
 446       if batch_size == len(self.replay_buffer):
 447         return None, None
 448       batch_size *= 1.2
 449 
 450     return (self.convert_to_batched_episodes(episodes), probs)
 451 
 452   def seed_replay_buffer(self, episodes):
 453     """Seed the replay buffer with some episodes."""
 454     if self.replay_buffer is None:
 455       return
 456 
 457     # just need to add initial state
 458     for i in xrange(len(episodes)):
 459       episodes[i] = [self.initial_internal_state()] + episodes[i]
 460 
 461     self.replay_buffer.seed_buffer(episodes)
