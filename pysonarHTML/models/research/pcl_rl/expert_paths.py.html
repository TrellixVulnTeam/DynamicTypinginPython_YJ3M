  21 import tensorflow as tf
  22 import random
  23 import os
  24 import numpy as np
  25 from six.moves import xrange
  26 import pickle

  28 gfile = tf.gfile

  31 def sample_expert_paths(num, env_str, env_spec,
  32                         load_trajectories_file=None):
  33   """Sample a number of expert paths randomly."""
  34   if load_trajectories_file is not None:
  35     if not gfile.Exists(load_trajectories_file):
  36       assert False, 'trajectories file %s does not exist' % load_trajectories_file
  37 
  38     with gfile.GFile(load_trajectories_file, 'r') as f:
  39       episodes = pickle.load(f)
  40       episodes = random.sample(episodes, num)
  41       return [ep[1:] for ep in episodes]
  42 
  43   return [sample_expert_path(env_str, env_spec)
  44           for _ in xrange(num)]
  45

  47 def sample_expert_path(env_str, env_spec):
  48   """Algorithmic tasks have known distribution of expert paths we sample from."""
  49   t = random.randint(2, 10)  # sequence length
  50   observations = []
  51   actions = [env_spec.initial_act(None)]
  52   rewards = []
  53 
  54   if env_str in ['DuplicatedInput-v0', 'Copy-v0']:
  55     chars = 5
  56     random_ints = [int(random.random() * 1000) for _ in xrange(t)]
  57     for tt in xrange(t):
  58       char_idx = tt // 2 if env_str == 'DuplicatedInput-v0' else tt
  59       char = random_ints[char_idx] % chars
  60       observations.append([char])
  61       actions.append([1, (tt + 1) % 2, char])
  62       rewards.append((tt + 1) % 2)
  63   elif env_str in ['RepeatCopy-v0']:
  64     chars = 5
  65 
  66     random_ints = [int(random.random() * 1000) for _ in xrange(t)]
  67     for tt in xrange(3 * t + 2):
  68       char_idx = (tt if tt < t else
  69                   2 * t - tt if tt <= 2 * t else
  70                   tt - 2 * t - 2)
  71       if tt in [t, 2 * t + 1]:
  72         char = chars
  73       else:
  74         char = random_ints[char_idx] % chars
  75       observations.append([char])
  76       actions.append([1 if tt < t else 0 if tt <= 2 * t else 1,
  77                       tt not in [t, 2 * t + 1], char])
  78       rewards.append(actions[-1][-2])
  79   elif env_str in ['Reverse-v0']:
  80     chars = 2
  81     random_ints = [int(random.random() * 1000) for _ in xrange(t)]
  82     for tt in xrange(2 * t + 1):
  83       char_idx = tt if tt < t else 2 * t - tt
  84       if tt != t:
  85         char = random_ints[char_idx] % chars
  86       else:
  87         char = chars
  88       observations.append([char])
  89       actions.append([tt < t, tt > t, char])
  90       rewards.append(tt > t)
  91   elif env_str in ['ReversedAddition-v0']:
  92     chars = 3
  93     random_ints = [int(random.random() * 1000) for _ in xrange(1 + 2 * t)]
  94     carry = 0
  95     char_history = []
  96     move_map = {0: 3, 1: 1, 2: 2, 3: 1}
  97     for tt in xrange(2 * t + 1):
  98       char_idx = tt
  99       if tt >= 2 * t:
 100         char = chars
 101       else:
 102         char = random_ints[char_idx] % chars
 103       char_history.append(char)
 104       if tt % 2 == 1:
 105         tot = char_history[-2] + char_history[-1] + carry
 106         carry = tot // chars
 107         tot = tot % chars
 108       elif tt == 2 * t:
 109         tot = carry
 110       else:
 111         tot = 0
 112       observations.append([char])
 113       actions.append([move_map[tt % len(move_map)],
 114                       tt % 2 or tt == 2 * t, tot])
 115       rewards.append(tt % 2 or tt == 2 * t)
 116   elif env_str in ['ReversedAddition3-v0']:
 117     chars = 3
 118     random_ints = [int(random.random() * 1000) for _ in xrange(1 + 3 * t)]
 119     carry = 0
 120     char_history = []
 121     move_map = {0: 3, 1: 3, 2: 1, 3: 2, 4:2, 5: 1}
 122     for tt in xrange(3 * t + 1):
 123       char_idx = tt
 124       if tt >= 3 * t:
 125         char = chars
 126       else:
 127         char = random_ints[char_idx] % chars
 128       char_history.append(char)
 129       if tt % 3 == 2:
 130         tot = char_history[-3] + char_history[-2] + char_history[-1] + carry
 131         carry = tot // chars
 132         tot = tot % chars
 133       elif tt == 3 * t:
 134         tot = carry
 135       else:
 136         tot = 0
 137       observations.append([char])
 138       actions.append([move_map[tt % len(move_map)],
 139                       tt % 3 == 2 or tt == 3 * t, tot])
 140       rewards.append(tt % 3 == 2 or tt == 3 * t)
 141 
 142   else:
 143     assert False, 'No expert trajectories for env %s' % env_str
 144 
 145   actions = [
 146       env_spec.convert_env_actions_to_actions(act)
 147       for act in actions]
 148   observations.append([chars])
 149 
 150   observations = [np.array(obs) for obs in zip(*observations)]
 151   actions = [np.array(act) for act in zip(*actions)]
 152   rewards = np.array(rewards)
 153   return [observations, actions, rewards, True]
