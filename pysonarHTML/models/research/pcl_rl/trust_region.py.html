  32 def var_size(v):
  33   return int(np.prod([int(d) for d in v.shape]))
  34 
  35 
  36 def gradients(loss, var_list):
  37   grads = tf.gradients(loss, var_list)
  38   return [g if g is not None else tf.zeros(v.shape)
  39           for g, v in zip(grads, var_list)]
  40 
  41 def flatgrad(loss, var_list):
  42   grads = gradients(loss, var_list)
  43   return tf.concat([tf.reshape(grad, [-1])
  44                     for (v, grad) in zip(var_list, grads)
  45                     if grad is not None], 0)
  46 
  47 
  48 def get_flat(var_list):
  49   return tf.concat([tf.reshape(v, [-1]) for v in var_list], 0)
  50 
  51 
  52 def set_from_flat(var_list, flat_theta):
  53   assigns = []
  54   shapes = [v.shape for v in var_list]
  55   sizes = [var_size(v) for v in var_list]
  56 
  57   start = 0
  58   assigns = []
  59   for (shape, size, v) in zip(shapes, sizes, var_list):
  60     assigns.append(v.assign(
  61         tf.reshape(flat_theta[start:start + size], shape)))
  62     start += size
  63   assert start == sum(sizes)
  64 
  65   return tf.group(*assigns)

  68 class TrustRegionOptimization(object):
  69 
  70   def __init__(self, max_divergence=0.1, cg_damping=0.1):
  71     self.max_divergence = max_divergence
  72     self.cg_damping = cg_damping
  73 
  74   def setup_placeholders(self):
  75     self.flat_tangent = tf.placeholder(tf.float32, [None], 'flat_tangent')
  76     self.flat_theta = tf.placeholder(tf.float32, [None], 'flat_theta')
  77 
  78   def setup(self, var_list, raw_loss, self_divergence,
  79             divergence=None):
  80     self.setup_placeholders()
  81 
  82     self.raw_loss = raw_loss
  83     self.divergence = divergence
  84     self.loss_flat_gradient = flatgrad(raw_loss, var_list)
  85     self.divergence_gradient = gradients(self_divergence, var_list)
  86 
  87     shapes = [var.shape for var in var_list]
  88     sizes = [var_size(var) for var in var_list]
  89 
  90     start = 0
  91     tangents = []
  92     for shape, size in zip(shapes, sizes):
  93       param = tf.reshape(self.flat_tangent[start:start + size], shape)
  94       tangents.append(param)
  95       start += size
  96     assert start == sum(sizes)
  97 
  98     self.grad_vector_product = sum(
  99         tf.reduce_sum(g * t) for (g, t) in zip(self.divergence_gradient, tangents))
 100     self.fisher_vector_product = flatgrad(self.grad_vector_product, var_list)
 101 
 102     self.flat_vars = get_flat(var_list)
 103     self.set_vars = set_from_flat(var_list, self.flat_theta)
 104 
 105   def optimize(self, sess, feed_dict):
 106     old_theta = sess.run(self.flat_vars)
 107     loss_flat_grad = sess.run(self.loss_flat_gradient,
 108                               feed_dict=feed_dict)
 109 
 110     def calc_fisher_vector_product(tangent):
 111       feed_dict[self.flat_tangent] = tangent
 112       fvp = sess.run(self.fisher_vector_product,
 113                      feed_dict=feed_dict)
 114       fvp += self.cg_damping * tangent
 115       return fvp
 116 
 117     step_dir = conjugate_gradient(calc_fisher_vector_product, -loss_flat_grad)
 118 
 119     shs = 0.5 * step_dir.dot(calc_fisher_vector_product(step_dir))
 120     lm = np.sqrt(shs / self.max_divergence)
 121     fullstep = step_dir / lm
 122     neggdotstepdir = -loss_flat_grad.dot(step_dir)
 123 
 124     def calc_loss(theta):
 125       sess.run(self.set_vars, feed_dict={self.flat_theta: theta})
 126       if self.divergence is None:
 127         return sess.run(self.raw_loss, feed_dict=feed_dict), True
 128       else:
 129         raw_loss, divergence = sess.run(
 130             [self.raw_loss, self.divergence], feed_dict=feed_dict)
 131         return raw_loss, divergence < self.max_divergence
 132 
 133     # find optimal theta
 134     theta = linesearch(calc_loss, old_theta, fullstep, neggdotstepdir / lm)
 135     if self.divergence is not None:
 136       final_divergence = sess.run(self.divergence, feed_dict=feed_dict)
 137     else:
 138       final_divergence = None
 139 
 140     # set vars accordingly
 141     if final_divergence is None or final_divergence < self.max_divergence:
 142       sess.run(self.set_vars, feed_dict={self.flat_theta: theta})
 143     else:
 144       sess.run(self.set_vars, feed_dict={self.flat_theta: old_theta})

 147 def conjugate_gradient(f_Ax, b, cg_iters=10, residual_tol=1e-10):
 148   p = b.copy()
 149   r = b.copy()
 150   x = np.zeros_like(b)
 151   rdotr = r.dot(r)
 152   for i in xrange(cg_iters):
 153     z = f_Ax(p)
 154     v = rdotr / p.dot(z)
 155     x += v * p
 156     r -= v * z
 157     newrdotr = r.dot(r)
 158     mu = newrdotr / rdotr
 159     p = r + mu * p
 160     rdotr = newrdotr
 161     if rdotr < residual_tol:
 162       break
 163   return x

 166 def linesearch(f, x, fullstep, expected_improve_rate):
 167   accept_ratio = 0.1
 168   max_backtracks = 10
 169 
 170   fval, _ = f(x)
 171   for (_n_backtracks, stepfrac) in enumerate(.5 ** np.arange(max_backtracks)):
 172     xnew = x + stepfrac * fullstep
 173     newfval, valid = f(xnew)
 174     if not valid:
 175       continue
 176     actual_improve = fval - newfval
 177     expected_improve = expected_improve_rate * stepfrac
 178     ratio = actual_improve / expected_improve
 179     if ratio > accept_ratio and actual_improve > 0:
 180       return xnew
 181 
 182   return x
