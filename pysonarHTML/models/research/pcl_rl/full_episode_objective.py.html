  16 """Objectives for full-episode.
  17 
  18 Implementations of UREX & REINFORCE.  Note that these implementations
  19 use a non-parametric baseline to reduce variance.  Thus, multiple
  20 samples with the same seed must be taken from the environment.
  21 
  22 """
  23 
  24 import tensorflow as tf
  25 
  26 import objective
  27 
  28 
  29 class Reinforce(objective.Objective):
  30   def __init__(self, learning_rate, clip_norm, num_samples,
  31                tau=0.1, bonus_weight=1.0):
  32     super(Reinforce, self).__init__(learning_rate, clip_norm=clip_norm)
  33     self.num_samples = num_samples
  34     assert self.num_samples > 1
  35     self.tau = tau
  36     self.bonus_weight = bonus_weight
  37     self.eps_lambda = 0.0
  38 
  39   def get_bonus(self, total_rewards, total_log_probs):
  40     """Exploration bonus."""
  41     return -self.tau * total_log_probs
  42 
  43   def get(self, rewards, pads, values, final_values,
  44           log_probs, prev_log_probs, target_log_probs,
  45           entropies, logits,
  46           target_values, final_target_values):
  47     seq_length = tf.shape(rewards)[0]
  48 
  49     not_pad = tf.reshape(1 - pads, [seq_length, -1, self.num_samples])
  50     rewards = not_pad * tf.reshape(rewards, [seq_length, -1, self.num_samples])
  51     log_probs = not_pad * tf.reshape(sum(log_probs), [seq_length, -1, self.num_samples])
  52 
  53     total_rewards = tf.reduce_sum(rewards, 0)
  54     total_log_probs = tf.reduce_sum(log_probs, 0)
  55 
  56     rewards_and_bonus = (total_rewards +
  57                          self.bonus_weight *
  58                          self.get_bonus(total_rewards, total_log_probs))
  59 
  60     baseline = tf.reduce_mean(rewards_and_bonus, 1, keep_dims=True)
  61 
  62     loss = -tf.stop_gradient(rewards_and_bonus - baseline) * total_log_probs
  63     loss = tf.reduce_mean(loss)
  64     raw_loss = loss  # TODO
  65 
  66     gradient_ops = self.training_ops(
  67         loss, learning_rate=self.learning_rate)
  68 
  69     tf.summary.histogram('log_probs', total_log_probs)
  70     tf.summary.histogram('rewards', total_rewards)
  71     tf.summary.scalar('avg_rewards',
  72                       tf.reduce_mean(total_rewards))
  73     tf.summary.scalar('loss', loss)
  74 
  75     return loss, raw_loss, baseline, gradient_ops, tf.summary.merge_all()
  76 
  77 
  78 class UREX(Reinforce):
  79   def get_bonus(self, total_rewards, total_log_probs):
  80     """Exploration bonus."""
  81     discrepancy = total_rewards / self.tau - total_log_probs
  82     normalized_d = self.num_samples * tf.nn.softmax(discrepancy)
  83     return self.tau * normalized_d
