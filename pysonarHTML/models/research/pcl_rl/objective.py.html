  25 class Objective(object):
  26   def __init__(self, learning_rate, clip_norm):
  27     self.learning_rate = learning_rate
  28     self.clip_norm = clip_norm
  29 
  30   def get_optimizer(self, learning_rate):
  31     """Optimizer for gradient descent ops."""
  32     return tf.train.AdamOptimizer(learning_rate=learning_rate,
  33                                   epsilon=2e-4)
  34 
  35   def training_ops(self, loss, learning_rate=None):
  36     """Gradient ops."""
  37     opt = self.get_optimizer(learning_rate)
  38     params = tf.trainable_variables()
  39     grads = tf.gradients(loss, params)
  40 
  41     if self.clip_norm:
  42       grads, global_norm = tf.clip_by_global_norm(grads, self.clip_norm)
  43       tf.summary.scalar('grad_global_norm', global_norm)
  44 
  45     return opt.apply_gradients(zip(grads, params))
  46 
  47   def get(self, rewards, pads, values, final_values,
  48           log_probs, prev_log_probs, target_log_probs,
  49           entropies, logits,
  50           target_values, final_target_values):
  51     """Get objective calculations."""
  52     raise NotImplementedError()

  55 def discounted_future_sum(values, discount, rollout):
  56   """Discounted future sum of time-major values."""
  57   discount_filter = tf.reshape(
  58       discount ** tf.range(float(rollout)), [-1, 1, 1])
  59   expanded_values = tf.concat(
  60       [values, tf.zeros([rollout - 1, tf.shape(values)[1]])], 0)
  61 
  62   conv_values = tf.transpose(tf.squeeze(tf.nn.conv1d(
  63       tf.expand_dims(tf.transpose(expanded_values), -1), discount_filter,
  64       stride=1, padding='VALID'), -1))
  65 
  66   return conv_values
  67 
  68 
  69 def discounted_two_sided_sum(values, discount, rollout):
  70   """Discounted two-sided sum of time-major values."""
  71   roll = float(rollout)
  72   discount_filter = tf.reshape(
  73       discount ** tf.abs(tf.range(-roll + 1, roll)), [-1, 1, 1])
  74   expanded_values = tf.concat(
  75       [tf.zeros([rollout - 1, tf.shape(values)[1]]), values,
  76        tf.zeros([rollout - 1, tf.shape(values)[1]])], 0)
  77 
  78   conv_values = tf.transpose(tf.squeeze(tf.nn.conv1d(
  79       tf.expand_dims(tf.transpose(expanded_values), -1), discount_filter,
  80       stride=1, padding='VALID'), -1))
  81 
  82   return conv_values
  83

  85 def shift_values(values, discount, rollout, final_values=0.0):
  86   """Shift values up by some amount of time.
  87 
  88   Those values that shift from a value beyond the last value
  89   are calculated using final_values.
  90 
  91   """
  92   roll_range = tf.cumsum(tf.ones_like(values[:rollout, :]), 0,
  93                          exclusive=True, reverse=True)
  94   final_pad = tf.expand_dims(final_values, 0) * discount ** roll_range
  95   return tf.concat([discount ** rollout * values[rollout:, :],
  96                     final_pad], 0)

  99 class ActorCritic(Objective):
 100   """Standard Actor-Critic."""
 101 
 102   def __init__(self, learning_rate, clip_norm=5,
 103                policy_weight=1.0, critic_weight=0.1,
 104                tau=0.1, gamma=1.0, rollout=10,
 105                eps_lambda=0.0, clip_adv=None,
 106                use_target_values=False):
 107     super(ActorCritic, self).__init__(learning_rate, clip_norm=clip_norm)
 108     self.policy_weight = policy_weight
 109     self.critic_weight = critic_weight
 110     self.tau = tau
 111     self.gamma = gamma
 112     self.rollout = rollout
 113     self.clip_adv = clip_adv
 114 
 115     self.eps_lambda = tf.get_variable(  # TODO: need a better way
 116         'eps_lambda', [], initializer=tf.constant_initializer(eps_lambda),
 117         trainable=False)
 118     self.new_eps_lambda = tf.placeholder(tf.float32, [])
 119     self.assign_eps_lambda = self.eps_lambda.assign(
 120         0.99 * self.eps_lambda + 0.01 * self.new_eps_lambda)
 121     self.use_target_values = use_target_values
 122 
 123   def get(self, rewards, pads, values, final_values,
 124           log_probs, prev_log_probs, target_log_probs,
 125           entropies, logits,
 126           target_values, final_target_values):
 127     not_pad = 1 - pads
 128     batch_size = tf.shape(rewards)[1]
 129 
 130     entropy = not_pad * sum(entropies)
 131     rewards = not_pad * rewards
 132     value_estimates = not_pad * values
 133     log_probs = not_pad * sum(log_probs)
 134     target_values = not_pad * tf.stop_gradient(target_values)
 135     final_target_values = tf.stop_gradient(final_target_values)
 136 
 137     sum_rewards = discounted_future_sum(rewards, self.gamma, self.rollout)
 138     if self.use_target_values:
 139       last_values = shift_values(
 140           target_values, self.gamma, self.rollout,
 141           final_target_values)
 142     else:
 143       last_values = shift_values(value_estimates, self.gamma, self.rollout,
 144                                  final_values)
 145 
 146     future_values = sum_rewards + last_values
 147     baseline_values = value_estimates
 148 
 149     adv = tf.stop_gradient(-baseline_values + future_values)
 150     if self.clip_adv:
 151       adv = tf.minimum(self.clip_adv, tf.maximum(-self.clip_adv, adv))
 152     policy_loss = -adv * log_probs
 153     critic_loss = -adv * baseline_values
 154     regularizer = -self.tau * entropy
 155 
 156     policy_loss = tf.reduce_mean(
 157         tf.reduce_sum(policy_loss * not_pad, 0))
 158     critic_loss = tf.reduce_mean(
 159         tf.reduce_sum(critic_loss * not_pad, 0))
 160     regularizer = tf.reduce_mean(
 161         tf.reduce_sum(regularizer * not_pad, 0))
 162 
 163     # loss for gradient calculation
 164     loss = (self.policy_weight * policy_loss +
 165             self.critic_weight * critic_loss + regularizer)
 166 
 167     raw_loss = tf.reduce_mean(  # TODO
 168         tf.reduce_sum(not_pad * policy_loss, 0))
 169 
 170     gradient_ops = self.training_ops(
 171         loss, learning_rate=self.learning_rate)
 172 
 173     tf.summary.histogram('log_probs', tf.reduce_sum(log_probs, 0))
 174     tf.summary.histogram('rewards', tf.reduce_sum(rewards, 0))
 175     tf.summary.scalar('avg_rewards',
 176                       tf.reduce_mean(tf.reduce_sum(rewards, 0)))
 177     tf.summary.scalar('policy_loss',
 178                       tf.reduce_mean(tf.reduce_sum(not_pad * policy_loss)))
 179     tf.summary.scalar('critic_loss',
 180                       tf.reduce_mean(tf.reduce_sum(not_pad * policy_loss)))
 181     tf.summary.scalar('loss', loss)
 182     tf.summary.scalar('raw_loss', raw_loss)
 183 
 184     return (loss, raw_loss, future_values,
 185             gradient_ops, tf.summary.merge_all())

 188 class PCL(ActorCritic):
 189   """PCL implementation.
 190 
 191   Implements vanilla PCL, Unified PCL, and Trust PCL depending
 192   on provided inputs.
 193 
 194   """
 195 
 196   def get(self, rewards, pads, values, final_values,
 197           log_probs, prev_log_probs, target_log_probs,
 198           entropies, logits,
 199           target_values, final_target_values):
 200     not_pad = 1 - pads
 201     batch_size = tf.shape(rewards)[1]
 202 
 203     rewards = not_pad * rewards
 204     value_estimates = not_pad * values
 205     log_probs = not_pad * sum(log_probs)
 206     target_log_probs = not_pad * tf.stop_gradient(sum(target_log_probs))
 207     relative_log_probs = not_pad * (log_probs - target_log_probs)
 208     target_values = not_pad * tf.stop_gradient(target_values)
 209     final_target_values = tf.stop_gradient(final_target_values)
 210 
 211     # Prepend.
 212     not_pad = tf.concat([tf.ones([self.rollout - 1, batch_size]),
 213                          not_pad], 0)
 214     rewards = tf.concat([tf.zeros([self.rollout - 1, batch_size]),
 215                          rewards], 0)
 216     value_estimates = tf.concat(
 217         [self.gamma ** tf.expand_dims(
 218             tf.range(float(self.rollout - 1), 0, -1), 1) *
 219          tf.ones([self.rollout - 1, batch_size]) *
 220          value_estimates[0:1, :],
 221          value_estimates], 0)
 222     log_probs = tf.concat([tf.zeros([self.rollout - 1, batch_size]),
 223                            log_probs], 0)
 224     prev_log_probs = tf.concat([tf.zeros([self.rollout - 1, batch_size]),
 225                                 prev_log_probs], 0)
 226     relative_log_probs = tf.concat([tf.zeros([self.rollout - 1, batch_size]),
 227                                     relative_log_probs], 0)
 228     target_values = tf.concat(
 229         [self.gamma ** tf.expand_dims(
 230             tf.range(float(self.rollout - 1), 0, -1), 1) *
 231          tf.ones([self.rollout - 1, batch_size]) *
 232          target_values[0:1, :],
 233          target_values], 0)
 234 
 235     sum_rewards = discounted_future_sum(rewards, self.gamma, self.rollout)
 236     sum_log_probs = discounted_future_sum(log_probs, self.gamma, self.rollout)
 237     sum_prev_log_probs = discounted_future_sum(prev_log_probs, self.gamma, self.rollout)
 238     sum_relative_log_probs = discounted_future_sum(
 239         relative_log_probs, self.gamma, self.rollout)
 240 
 241     if self.use_target_values:
 242       last_values = shift_values(
 243           target_values, self.gamma, self.rollout,
 244           final_target_values)
 245     else:
 246       last_values = shift_values(value_estimates, self.gamma, self.rollout,
 247                                  final_values)
 248 
 249     future_values = (
 250         - self.tau * sum_log_probs
 251         - self.eps_lambda * sum_relative_log_probs
 252         + sum_rewards + last_values)
 253     baseline_values = value_estimates
 254 
 255     adv = tf.stop_gradient(-baseline_values + future_values)
 256     if self.clip_adv:
 257       adv = tf.minimum(self.clip_adv, tf.maximum(-self.clip_adv, adv))
 258     policy_loss = -adv * sum_log_probs
 259     critic_loss = -adv * (baseline_values - last_values)
 260 
 261     policy_loss = tf.reduce_mean(
 262         tf.reduce_sum(policy_loss * not_pad, 0))
 263     critic_loss = tf.reduce_mean(
 264         tf.reduce_sum(critic_loss * not_pad, 0))
 265 
 266     # loss for gradient calculation
 267     loss = (self.policy_weight * policy_loss +
 268             self.critic_weight * critic_loss)
 269 
 270     # actual quantity we're trying to minimize
 271     raw_loss = tf.reduce_mean(
 272         tf.reduce_sum(not_pad * adv * (-baseline_values + future_values), 0))
 273 
 274     gradient_ops = self.training_ops(
 275         loss, learning_rate=self.learning_rate)
 276 
 277     tf.summary.histogram('log_probs', tf.reduce_sum(log_probs, 0))
 278     tf.summary.histogram('rewards', tf.reduce_sum(rewards, 0))
 279     tf.summary.histogram('future_values', future_values)
 280     tf.summary.histogram('baseline_values', baseline_values)
 281     tf.summary.histogram('advantages', adv)
 282     tf.summary.scalar('avg_rewards',
 283                       tf.reduce_mean(tf.reduce_sum(rewards, 0)))
 284     tf.summary.scalar('policy_loss',
 285                       tf.reduce_mean(tf.reduce_sum(not_pad * policy_loss)))
 286     tf.summary.scalar('critic_loss',
 287                       tf.reduce_mean(tf.reduce_sum(not_pad * policy_loss)))
 288     tf.summary.scalar('loss', loss)
 289     tf.summary.scalar('raw_loss', tf.reduce_mean(raw_loss))
 290     tf.summary.scalar('eps_lambda', self.eps_lambda)
 291 
 292     return (loss, raw_loss,
 293             future_values[self.rollout - 1:, :],
 294             gradient_ops, tf.summary.merge_all())

 297 class TRPO(ActorCritic):
 298   """TRPO."""
 299 
 300   def get(self, rewards, pads, values, final_values,
 301           log_probs, prev_log_probs, target_log_probs,
 302           entropies, logits,
 303           target_values, final_target_values):
 304     not_pad = 1 - pads
 305     batch_size = tf.shape(rewards)[1]
 306 
 307     rewards = not_pad * rewards
 308     value_estimates = not_pad * values
 309     log_probs = not_pad * sum(log_probs)
 310     prev_log_probs = not_pad * prev_log_probs
 311     target_values = not_pad * tf.stop_gradient(target_values)
 312     final_target_values = tf.stop_gradient(final_target_values)
 313 
 314     sum_rewards = discounted_future_sum(rewards, self.gamma, self.rollout)
 315 
 316     if self.use_target_values:
 317       last_values = shift_values(
 318           target_values, self.gamma, self.rollout,
 319           final_target_values)
 320     else:
 321       last_values = shift_values(value_estimates, self.gamma, self.rollout,
 322                                  final_values)
 323 
 324     future_values = sum_rewards + last_values
 325     baseline_values = value_estimates
 326 
 327 
 328     adv = tf.stop_gradient(-baseline_values + future_values)
 329     if self.clip_adv:
 330       adv = tf.minimum(self.clip_adv, tf.maximum(-self.clip_adv, adv))
 331     policy_loss = -adv * tf.exp(log_probs - prev_log_probs)
 332     critic_loss = -adv * baseline_values
 333 
 334     policy_loss = tf.reduce_mean(
 335         tf.reduce_sum(policy_loss * not_pad, 0))
 336     critic_loss = tf.reduce_mean(
 337         tf.reduce_sum(critic_loss * not_pad, 0))
 338     raw_loss = policy_loss
 339 
 340     # loss for gradient calculation
 341     if self.policy_weight == 0:
 342       policy_loss = 0.0
 343     elif self.critic_weight == 0:
 344       critic_loss = 0.0
 345 
 346     loss = (self.policy_weight * policy_loss +
 347             self.critic_weight * critic_loss)
 348 
 349     gradient_ops = self.training_ops(
 350         loss, learning_rate=self.learning_rate)
 351 
 352     tf.summary.histogram('log_probs', tf.reduce_sum(log_probs, 0))
 353     tf.summary.histogram('rewards', tf.reduce_sum(rewards, 0))
 354     tf.summary.scalar('avg_rewards',
 355                       tf.reduce_mean(tf.reduce_sum(rewards, 0)))
 356     tf.summary.scalar('policy_loss',
 357                       tf.reduce_mean(tf.reduce_sum(not_pad * policy_loss)))
 358     tf.summary.scalar('critic_loss',
 359                       tf.reduce_mean(tf.reduce_sum(not_pad * policy_loss)))
 360     tf.summary.scalar('loss', loss)
 361     tf.summary.scalar('raw_loss', raw_loss)
 362 
 363     return (loss, raw_loss, future_values,
 364             gradient_ops, tf.summary.merge_all())
