  32   def __init__(self, env_spec, global_step,
  33                target_network_lag=0.95,
  34                sample_from='online',
  35                get_policy=None,
  36                get_baseline=None,
  37                get_objective=None,
  38                get_trust_region_p_opt=None,
  39                get_value_opt=None):

  41 
  42     self.global_step = global_step
  43     self.inc_global_step = self.global_step.assign_add(1)
  44 
  45     self.target_network_lag = target_network_lag
  46     self.sample_from = sample_from
  47 
  48     self.policy = get_policy()
  49     self.baseline = get_baseline()
  50     self.objective = get_objective()
  51     self.baseline.eps_lambda = self.objective.eps_lambda  # TODO: do this better

  55   def setup_placeholders(self):
  56     """Create the Tensorflow placeholders."""
  57     # summary placeholder
  58     self.avg_episode_reward = tf.placeholder(
  59         tf.float32, [], 'avg_episode_reward')
  60     self.greedy_episode_reward = tf.placeholder(
  61         tf.float32, [], 'greedy_episode_reward')
  62 
  63     # sampling placeholders
  64     self.internal_state = tf.placeholder(tf.float32,
  65                                          [None, self.policy.rnn_state_dim],
  66                                          'internal_state')
  67 
  68     self.single_observation = []
  69     for i, (obs_dim, obs_type) in enumerate(self.env_spec.obs_dims_and_types):
  70       if self.env_spec.is_discrete(obs_type):
  71         self.single_observation.append(
  72             tf.placeholder(tf.int32, [None], 'obs%d' % i))
  73       elif self.env_spec.is_box(obs_type):
  74         self.single_observation.append(
  75             tf.placeholder(tf.float32, [None, obs_dim], 'obs%d' % i))
  76       else:
  77         assert False
  78 
  79     self.single_action = []
  80     for i, (action_dim, action_type) in \
  81         enumerate(self.env_spec.act_dims_and_types):
  82       if self.env_spec.is_discrete(action_type):
  83         self.single_action.append(
  84             tf.placeholder(tf.int32, [None], 'act%d' % i))
  85       elif self.env_spec.is_box(action_type):
  86         self.single_action.append(
  87             tf.placeholder(tf.float32, [None, action_dim], 'act%d' % i))
  88       else:
  89         assert False
  90 
  91     # training placeholders
  92     self.observations = []
  93     for i, (obs_dim, obs_type) in enumerate(self.env_spec.obs_dims_and_types):
  94       if self.env_spec.is_discrete(obs_type):
  95         self.observations.append(
  96             tf.placeholder(tf.int32, [None, None], 'all_obs%d' % i))
  97       else:
  98         self.observations.append(
  99             tf.placeholder(tf.float32, [None, None, obs_dim], 'all_obs%d' % i))
 100 
 101     self.actions = []
 102     self.other_logits = []
 103     for i, (action_dim, action_type) in \
 104         enumerate(self.env_spec.act_dims_and_types):
 105       if self.env_spec.is_discrete(action_type):
 106         self.actions.append(
 107             tf.placeholder(tf.int32, [None, None], 'all_act%d' % i))
 108       if self.env_spec.is_box(action_type):
 109         self.actions.append(
 110             tf.placeholder(tf.float32, [None, None, action_dim],
 111                            'all_act%d' % i))
 112       self.other_logits.append(
 113           tf.placeholder(tf.float32, [None, None, None],
 114                          'other_logits%d' % i))
 115 
 116     self.rewards = tf.placeholder(tf.float32, [None, None], 'rewards')
 117     self.terminated = tf.placeholder(tf.float32, [None], 'terminated')
 118     self.pads = tf.placeholder(tf.float32, [None, None], 'pads')
 119 
 120     self.prev_log_probs = tf.placeholder(tf.float32, [None, None],
 121                                          'prev_log_probs')
 122 
 123   def setup(self, train=True):
 124     """Setup Tensorflow Graph."""
 125 
 126     self.setup_placeholders()
 127 
 128     tf.summary.scalar('avg_episode_reward', self.avg_episode_reward)
 129     tf.summary.scalar('greedy_episode_reward', self.greedy_episode_reward)
 130 
 131     with tf.variable_scope('model', reuse=None):
 132       # policy network
 133       with tf.variable_scope('policy_net'):
 134         (self.policy_internal_states, self.logits, self.log_probs,
 135          self.entropies, self.self_kls) = \
 136             self.policy.multi_step(self.observations,
 137                                    self.internal_state,
 138                                    self.actions)
 139         self.out_log_probs = sum(self.log_probs)
 140         self.kl = self.policy.calculate_kl(self.other_logits, self.logits)
 141         self.avg_kl = (tf.reduce_sum(sum(self.kl)[:-1] * (1 - self.pads)) /
 142                        tf.reduce_sum(1 - self.pads))
 143 
 144       # value network
 145       with tf.variable_scope('value_net'):
 146         (self.values,
 147          self.regression_input,
 148          self.regression_weight) = self.baseline.get_values(
 149             self.observations, self.actions,
 150             self.policy_internal_states, self.logits)
 151 
 152       # target policy network
 153       with tf.variable_scope('target_policy_net'):
 154         (self.target_policy_internal_states,
 155          self.target_logits, self.target_log_probs,
 156          _, _) = \
 157             self.policy.multi_step(self.observations,
 158                                    self.internal_state,
 159                                    self.actions)
 160 
 161       # target value network
 162       with tf.variable_scope('target_value_net'):
 163         (self.target_values, _, _) = self.baseline.get_values(
 164             self.observations, self.actions,
 165             self.target_policy_internal_states, self.target_logits)
 166 
 167       # construct copy op online --> target
 168       all_vars = tf.trainable_variables()
 169       online_vars = [p for p in all_vars if
 170                      '/policy_net' in p.name or '/value_net' in p.name]
 171       target_vars = [p for p in all_vars if
 172                      'target_policy_net' in p.name or 'target_value_net' in p.name]
 173       online_vars.sort(key=lambda p: p.name)
 174       target_vars.sort(key=lambda p: p.name)
 175       aa = self.target_network_lag
 176       self.copy_op = tf.group(*[
 177           target_p.assign(aa * target_p + (1 - aa) * online_p)
 178           for online_p, target_p in zip(online_vars, target_vars)])
 179 
 180       if train:
 181         # evaluate objective
 182         (self.loss, self.raw_loss, self.regression_target,
 183          self.gradient_ops, self.summary) = self.objective.get(
 184             self.rewards, self.pads,
 185             self.values[:-1, :],
 186             self.values[-1, :] * (1 - self.terminated),
 187             self.log_probs, self.prev_log_probs, self.target_log_probs,
 188             self.entropies, self.logits, self.target_values[:-1, :],
 189             self.target_values[-1, :] * (1 - self.terminated))
 190 
 191         self.regression_target = tf.reshape(self.regression_target, [-1])
 192 
 193         self.policy_vars = [
 194             v for v in tf.trainable_variables()
 195             if '/policy_net' in v.name]
 196         self.value_vars = [
 197             v for v in tf.trainable_variables()
 198             if '/value_net' in v.name]
 199 
 200         # trust region optimizer
 201         if self.trust_region_policy_opt is not None:
 202           with tf.variable_scope('trust_region_policy', reuse=None):
 203             avg_self_kl = (
 204                 tf.reduce_sum(sum(self.self_kls) * (1 - self.pads)) /
 205                 tf.reduce_sum(1 - self.pads))
 206 
 207             self.trust_region_policy_opt.setup(
 208                 self.policy_vars, self.raw_loss, avg_self_kl,
 209                 self.avg_kl)
 210 
 211         # value optimizer
 212         if self.value_opt is not None:
 213           with tf.variable_scope('trust_region_value', reuse=None):
 214             self.value_opt.setup(
 215                 self.value_vars,
 216                 tf.reshape(self.values[:-1, :], [-1]),
 217                 self.regression_target,
 218                 tf.reshape(self.pads, [-1]),
 219                 self.regression_input, self.regression_weight)
 220 
 221     # we re-use variables for the sampling operations
 222     with tf.variable_scope('model', reuse=True):
 223       scope = ('target_policy_net' if self.sample_from == 'target'
 224                else 'policy_net')
 225       with tf.variable_scope(scope):
 226         self.next_internal_state, self.sampled_actions = \
 227             self.policy.sample_step(self.single_observation,
 228                                     self.internal_state,
 229                                     self.single_action)
 230         self.greedy_next_internal_state, self.greedy_sampled_actions = \
 231             self.policy.sample_step(self.single_observation,
 232                                     self.internal_state,
 233                                     self.single_action,
 234                                     greedy=True)
 235 
 236   def sample_step(self, sess,
 237                   single_observation, internal_state, single_action,
 238                   greedy=False):
 239     """Sample batch of steps from policy."""
 240     if greedy:
 241       outputs = [self.greedy_next_internal_state, self.greedy_sampled_actions]
 242     else:
 243       outputs = [self.next_internal_state, self.sampled_actions]
 244 
 245     feed_dict = {self.internal_state: internal_state}
 246     for action_place, action in zip(self.single_action, single_action):
 247       feed_dict[action_place] = action
 248     for obs_place, obs in zip(self.single_observation, single_observation):
 249       feed_dict[obs_place] = obs
 250 
 251     return sess.run(outputs, feed_dict=feed_dict)
 252 
 253   def train_step(self, sess,
 254                  observations, internal_state, actions,
 255                  rewards, terminated, pads,
 256                  avg_episode_reward=0, greedy_episode_reward=0):
 257     """Train network using standard gradient descent."""
 258     outputs = [self.raw_loss, self.gradient_ops, self.summary]
 259     feed_dict = {self.internal_state: internal_state,
 260                  self.rewards: rewards,
 261                  self.terminated: terminated,
 262                  self.pads: pads,
 263                  self.avg_episode_reward: avg_episode_reward,
 264                  self.greedy_episode_reward: greedy_episode_reward}
 265     time_len = None
 266     for action_place, action in zip(self.actions, actions):
 267       if time_len is None:
 268         time_len = len(action)
 269       assert time_len == len(action)
 270       feed_dict[action_place] = action
 271     for obs_place, obs in zip(self.observations, observations):
 272       assert time_len == len(obs)
 273       feed_dict[obs_place] = obs
 274 
 275     assert len(rewards) == time_len - 1
 276 
 277     return sess.run(outputs, feed_dict=feed_dict)

 280   def trust_region_step(self, sess,
 281                         observations, internal_state, actions,
 282                         rewards, terminated, pads,
 283                         avg_episode_reward=0,
 284                         greedy_episode_reward=0):
 285     """Train policy using trust region step."""
 286     feed_dict = {self.internal_state: internal_state,
 287                  self.rewards: rewards,
 288                  self.terminated: terminated,
 289                  self.pads: pads,
 290                  self.avg_episode_reward: avg_episode_reward,
 291                  self.greedy_episode_reward: greedy_episode_reward}
 292     for action_place, action in zip(self.actions, actions):
 293       feed_dict[action_place] = action
 294     for obs_place, obs in zip(self.observations, observations):
 295       feed_dict[obs_place] = obs
 296 
 297     (prev_log_probs, prev_logits) = sess.run(
 298         [self.out_log_probs, self.logits], feed_dict=feed_dict)
 299     feed_dict[self.prev_log_probs] = prev_log_probs
 300     for other_logit, prev_logit in zip(self.other_logits, prev_logits):
 301       feed_dict[other_logit] = prev_logit
 302 
 303     # fit policy
 304     self.trust_region_policy_opt.optimize(sess, feed_dict)
 305 
 306     ret = sess.run([self.raw_loss, self.summary], feed_dict=feed_dict)
 307     ret = [ret[0], None, ret[1]]
 308     return ret

 310   def fit_values(self, sess,
 311                  observations, internal_state, actions,
 312                  rewards, terminated, pads):
 313     """Train value network using value-specific optimizer."""
 314     feed_dict = {self.internal_state: internal_state,
 315                  self.rewards: rewards,
 316                  self.terminated: terminated,
 317                  self.pads: pads}
 318     for action_place, action in zip(self.actions, actions):
 319       feed_dict[action_place] = action
 320     for obs_place, obs in zip(self.observations, observations):
 321       feed_dict[obs_place] = obs
 322 
 323     # fit values
 324     if self.value_opt is None:
 325       raise ValueError('Specific value optimizer does not exist')
 326     self.value_opt.optimize(sess, feed_dict)
