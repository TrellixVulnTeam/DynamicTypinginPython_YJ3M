  16 """Baseline model for value estimates.
  17 
  18 Implements the value component of the neural network.
  19 In some cases this is just an additional linear layer on the policy.
  20 In other cases, it is a completely separate neural network.
  21 """

  28 class Baseline(object):
  29   def __init__(self, env_spec, internal_policy_dim,
  30                input_prev_actions=True,
  31                input_time_step=False,
  32                input_policy_state=True,
  33                n_hidden_layers=0,
  34                hidden_dim=64,
  35                tau=0.0):
  36     self.env_spec = env_spec
  37     self.internal_policy_dim = internal_policy_dim
  38     self.input_prev_actions = input_prev_actions
  39     self.input_time_step = input_time_step
  40     self.input_policy_state = input_policy_state
  41     self.n_hidden_layers = n_hidden_layers
  42     self.hidden_dim = hidden_dim
  43     self.tau = tau
  44 
  45     self.matrix_init = tf.truncated_normal_initializer(stddev=0.01)
  46 
  47   def get_inputs(self, time_step, obs, prev_actions,
  48                  internal_policy_states):
  49     """Get inputs to network as single tensor."""
  50     inputs = [tf.ones_like(time_step)]
  51     input_dim = 1
  52 
  53     if not self.input_policy_state:
  54       for i, (obs_dim, obs_type) in enumerate(self.env_spec.obs_dims_and_types):
  55         if self.env_spec.is_discrete(obs_type):
  56           inputs.append(
  57               tf.one_hot(obs[i], obs_dim))
  58           input_dim += obs_dim
  59         elif self.env_spec.is_box(obs_type):
  60           cur_obs = obs[i]
  61           inputs.append(cur_obs)
  62           inputs.append(cur_obs ** 2)
  63           input_dim += obs_dim * 2
  64         else:
  65           assert False
  66 
  67       if self.input_prev_actions:
  68         for i, (act_dim, act_type) in enumerate(self.env_spec.act_dims_and_types):
  69           if self.env_spec.is_discrete(act_type):
  70             inputs.append(
  71                 tf.one_hot(prev_actions[i], act_dim))
  72             input_dim += act_dim
  73           elif self.env_spec.is_box(act_type):
  74             inputs.append(prev_actions[i])
  75             input_dim += act_dim
  76           else:
  77             assert False
  78 
  79     if self.input_policy_state:
  80       inputs.append(internal_policy_states)
  81       input_dim += self.internal_policy_dim
  82 
  83     if self.input_time_step:
  84       scaled_time = 0.01 * time_step
  85       inputs.extend([scaled_time, scaled_time ** 2, scaled_time ** 3])
  86       input_dim += 3
  87 
  88     return input_dim, tf.concat(inputs, 1)

  90   def reshape_batched_inputs(self, all_obs, all_actions,
  91                              internal_policy_states, policy_logits):
  92     """Reshape inputs from [time_length, batch_size, ...] to
  93     [time_length * batch_size, ...].
  94 
  95     This allows for computing the value estimate in one go.
  96     """
  97     batch_size = tf.shape(all_obs[0])[1]
  98     time_length = tf.shape(all_obs[0])[0]
  99 
 100     reshaped_obs = []
 101     for obs, (obs_dim, obs_type) in zip(all_obs, self.env_spec.obs_dims_and_types):
 102       if self.env_spec.is_discrete(obs_type):
 103         reshaped_obs.append(tf.reshape(obs, [time_length * batch_size]))
 104       elif self.env_spec.is_box(obs_type):
 105         reshaped_obs.append(tf.reshape(obs, [time_length * batch_size, obs_dim]))
 106 
 107     reshaped_prev_act = []
 108     reshaped_policy_logits = []
 109     for i, (act_dim, act_type) in enumerate(self.env_spec.act_dims_and_types):
 110       prev_act = all_actions[i]
 111       if self.env_spec.is_discrete(act_type):
 112         reshaped_prev_act.append(
 113             tf.reshape(prev_act, [time_length * batch_size]))
 114       elif self.env_spec.is_box(act_type):
 115         reshaped_prev_act.append(
 116             tf.reshape(prev_act, [time_length * batch_size, act_dim]))
 117 
 118       reshaped_policy_logits.append(
 119           tf.reshape(policy_logits[i], [time_length * batch_size, -1]))
 120 
 121     reshaped_internal_policy_states = tf.reshape(
 122         internal_policy_states,
 123         [time_length * batch_size, self.internal_policy_dim])
 124 
 125     time_step = (float(self.input_time_step) *
 126                  tf.expand_dims(
 127                      tf.to_float(tf.range(time_length * batch_size) /
 128                                  batch_size), -1))
 129 
 130     return (time_step, reshaped_obs, reshaped_prev_act,
 131             reshaped_internal_policy_states,
 132             reshaped_policy_logits)
 133 
 134   def get_values(self, all_obs, all_actions, internal_policy_states,
 135                  policy_logits):
 136     """Get value estimates given input."""
 137     batch_size = tf.shape(all_obs[0])[1]
 138     time_length = tf.shape(all_obs[0])[0]
 139 
 140     (time_step, reshaped_obs, reshaped_prev_act,
 141      reshaped_internal_policy_states,
 142      reshaped_policy_logits) = self.reshape_batched_inputs(
 143          all_obs, all_actions, internal_policy_states, policy_logits)
 144 
 145     input_dim, inputs = self.get_inputs(
 146         time_step, reshaped_obs, reshaped_prev_act,
 147         reshaped_internal_policy_states)
 148 
 149     for depth in xrange(self.n_hidden_layers):
 150       with tf.variable_scope('value_layer%d' % depth):
 151         w = tf.get_variable('w', [input_dim, self.hidden_dim])
 152         inputs = tf.nn.tanh(tf.matmul(inputs, w))
 153         input_dim = self.hidden_dim
 154 
 155     w_v = tf.get_variable('w_v', [input_dim, 1],
 156                           initializer=self.matrix_init)
 157     values = tf.matmul(inputs, w_v)
 158     values = tf.reshape(values, [time_length, batch_size])
 159 
 160     inputs = inputs[:-batch_size]  # remove "final vals"
 161     return values, inputs, w_v

 167   def get_values(self, all_obs, all_actions, internal_policy_states,
 168                  policy_logits):
 169     batch_size = tf.shape(all_obs[0])[1]
 170     time_length = tf.shape(all_obs[0])[0]
 171 
 172     (time_step, reshaped_obs, reshaped_prev_act,
 173      reshaped_internal_policy_states,
 174      reshaped_policy_logits) = self.reshape_batched_inputs(
 175          all_obs, all_actions, internal_policy_states, policy_logits)
 176 
 177     def f_transform(q_values, tau):
 178       max_q = tf.reduce_max(q_values, -1, keep_dims=True)
 179       return tf.squeeze(max_q, [-1]) + tau * tf.log(
 180           tf.reduce_sum(tf.exp((q_values - max_q) / tau), -1))
 181 
 182     assert len(reshaped_policy_logits) == 1
 183     values = f_transform((self.tau + self.eps_lambda) * reshaped_policy_logits[0],
 184                          (self.tau + self.eps_lambda))
 185     values = tf.reshape(values, [time_length, batch_size])
 186 
 187     # not used
 188     input_dim, inputs = self.get_inputs(
 189         time_step, reshaped_obs, reshaped_prev_act,
 190         reshaped_internal_policy_states)
 191 
 192     w_v = tf.get_variable('w_v', [input_dim, 1],
 193                           initializer=self.matrix_init)
 194 
 195     inputs = inputs[:-batch_size]  # remove "final vals"
 196 
 197     return values, inputs, w_v
