  25 def weighted_sigmoid_cross_entropy(targets, logits, weight):
  26   return (weight * targets * np.log(1.0 + np.exp(-logits)) + (
  27       (1.0 - targets) * np.log(1.0 + 1.0 / np.exp(-logits))))
  28

  30 def hinge_loss(labels, logits):
  31   # Mostly copied from tensorflow.python.ops.losses but with loss per datapoint.
  32   labels = tf.to_float(labels)
  33   all_ones = tf.ones_like(labels)
  34   labels = tf.subtract(2 * labels, all_ones)
  35   return tf.nn.relu(tf.subtract(all_ones, tf.multiply(labels, logits)))

  40   def testTrivialCompatibilityWithSigmoidCrossEntropy(self):
  41     """Tests compatibility with unweighted function with weight 1.0."""
  42     x_shape = [300, 10]
  43     targets = np.random.random_sample(x_shape).astype(np.float32)
  44     logits = np.random.randn(*x_shape).astype(np.float32)
  45     weighted_loss = util.weighted_sigmoid_cross_entropy_with_logits(
  46         targets,
  47         logits)
  48     expected_loss = (
  49         tf.contrib.nn.deprecated_flipped_sigmoid_cross_entropy_with_logits(
  50             logits, targets))
  51     with self.test_session():
  52       self.assertAllClose(expected_loss.eval(),
  53                           weighted_loss.eval(),
  54                           atol=0.000001)

  76   def testDifferentSizeWeightedSigmoidCrossEntropy(self):
  77     """Tests correctness on 3D tensors.
  78 
  79     Tests that the function works as expected when logits is a 3D tensor and
  80     targets is a 2D tensor.
  81     """
  82     targets_shape = [30, 4]
  83     logits_shape = [targets_shape[0], targets_shape[1], 3]
  84     targets = np.random.random_sample(targets_shape).astype(np.float32)
  85     logits = np.random.randn(*logits_shape).astype(np.float32)
  86 
  87     weight_vector = [2.0, 3.0, 13.0]
  88     loss = util.weighted_sigmoid_cross_entropy_with_logits(targets,
  89                                                            logits,
  90                                                            weight_vector)
  91 
  92     with self.test_session():
  93       loss = loss.eval()
  94       for i in range(0, len(weight_vector)):
  95         expected = weighted_sigmoid_cross_entropy(targets, logits[:, :, i],
  96                                                   weight_vector[i])
  97         self.assertAllClose(loss[:, :, i], expected, atol=0.000001)

 116   def testGradients(self):
 117     """Tests that weighted loss gradients behave as expected."""
 118     dummy_tensor = tf.constant(1.0)
 119 
 120     positives_shape = [10, 1]
 121     positives_logits = dummy_tensor * tf.Variable(
 122         tf.random_normal(positives_shape) + 1.0)
 123     positives_targets = tf.ones(positives_shape)
 124     positives_weight = 4.6
 125     positives_loss = (
 126         tf.contrib.nn.deprecated_flipped_sigmoid_cross_entropy_with_logits(
 127             positives_logits, positives_targets) * positives_weight)
 128 
 129     negatives_shape = [190, 1]
 130     negatives_logits = dummy_tensor * tf.Variable(
 131         tf.random_normal(negatives_shape))
 132     negatives_targets = tf.zeros(negatives_shape)
 133     negatives_weight = 0.9
 134     negatives_loss = (
 135         tf.contrib.nn.deprecated_flipped_sigmoid_cross_entropy_with_logits(
 136             negatives_logits, negatives_targets) * negatives_weight)
 137 
 138     all_logits = tf.concat([positives_logits, negatives_logits], 0)
 139     all_targets = tf.concat([positives_targets, negatives_targets], 0)
 140     weighted_loss = tf.reduce_sum(
 141         util.weighted_sigmoid_cross_entropy_with_logits(
 142             all_targets, all_logits, positives_weight, negatives_weight))
 143     weighted_gradients = tf.gradients(weighted_loss, dummy_tensor)
 144 
 145     expected_loss = tf.add(
 146         tf.reduce_sum(positives_loss),
 147         tf.reduce_sum(negatives_loss))
 148     expected_gradients = tf.gradients(expected_loss, dummy_tensor)
 149 
 150     with tf.Session() as session:
 151       tf.global_variables_initializer().run()
 152       grad, expected_grad = session.run(
 153           [weighted_gradients, expected_gradients])
 154       self.assertAllClose(grad, expected_grad)

 173   def testTrivialCompatibilityWithHinge(self):
 174     # Tests compatibility with unweighted hinge loss.
 175     x_shape = [55, 10]
 176     logits = tf.constant(np.random.randn(*x_shape).astype(np.float32))
 177     targets = tf.to_float(tf.constant(np.random.random_sample(x_shape) > 0.3))
 178     weighted_loss = util.weighted_hinge_loss(targets, logits)
 179     expected_loss = hinge_loss(targets, logits)
 180     with self.test_session():
 181       self.assertAllClose(expected_loss.eval(), weighted_loss.eval())
 182 
 183   def testLessTrivialCompatibilityWithHinge(self):
 184     # Tests compatibility with a constant weight for positives and negatives.
 185     x_shape = [56, 11]
 186     logits = tf.constant(np.random.randn(*x_shape).astype(np.float32))
 187     targets = tf.to_float(tf.constant(np.random.random_sample(x_shape) > 0.7))
 188     weight = 1.0 + 1.0/2 + 1.0/3 + 1.0/4 + 1.0/5 + 1.0/6 + 1.0/7
 189     weighted_loss = util.weighted_hinge_loss(targets, logits, weight, weight)
 190     expected_loss = hinge_loss(targets, logits) * weight
 191     with self.test_session():
 192       self.assertAllClose(expected_loss.eval(), weighted_loss.eval())
 193 
 194   def testNontrivialCompatibilityWithHinge(self):
 195     # Tests compatibility with different positive and negative weights.
 196     x_shape = [23, 8]
 197     logits_positives = tf.constant(np.random.randn(*x_shape).astype(np.float32))
 198     logits_negatives = tf.constant(np.random.randn(*x_shape).astype(np.float32))
 199     targets_positives = tf.ones(x_shape)
 200     targets_negatives = tf.zeros(x_shape)
 201     logits = tf.concat([logits_positives, logits_negatives], 0)
 202     targets = tf.concat([targets_positives, targets_negatives], 0)
 203 
 204     raw_loss = util.weighted_hinge_loss(targets,
 205                                         logits,
 206                                         positive_weights=3.4,
 207                                         negative_weights=1.2)
 208     loss = tf.reduce_sum(raw_loss, 0)
 209     positives_hinge = hinge_loss(targets_positives, logits_positives)
 210     negatives_hinge = hinge_loss(targets_negatives, logits_negatives)
 211     expected = tf.add(tf.reduce_sum(3.4 * positives_hinge, 0),
 212                       tf.reduce_sum(1.2 * negatives_hinge, 0))
 213 
 214     with self.test_session():
 215       self.assertAllClose(loss.eval(), expected.eval())

 234 class BuildLabelPriorsTest(tf.test.TestCase):
 235 
 236   def testLabelPriorConsistency(self):
 237     # Checks that, with zero pseudocounts, the returned label priors reproduce
 238     # label frequencies in the batch.
 239     batch_shape = [4, 10]
 240     labels = tf.Variable(
 241         tf.to_float(tf.greater(tf.random_uniform(batch_shape), 0.678)))
 242 
 243     label_priors_update = util.build_label_priors(
 244         labels=labels, positive_pseudocount=0, negative_pseudocount=0)
 245     expected_priors = tf.reduce_mean(labels, 0)
 246 
 247     with self.test_session():
 248       tf.global_variables_initializer().run()
 249       self.assertAllClose(label_priors_update.eval(), expected_priors.eval())
 250 
 251   def testLabelPriorsUpdate(self):
 252     # Checks that the update of label priors behaves as expected.
 253     batch_shape = [1, 5]
 254     labels = tf.Variable(
 255         tf.to_float(tf.greater(tf.random_uniform(batch_shape), 0.4)))
 256     label_priors_update = util.build_label_priors(labels)
 257 
 258     label_sum = np.ones(shape=batch_shape)
 259     weight_sum = 2.0 * np.ones(shape=batch_shape)
 260 
 261     with self.test_session() as session:
 262       tf.global_variables_initializer().run()
 263 
 264       for _ in range(3):
 265         label_sum += labels.eval()
 266         weight_sum += np.ones(shape=batch_shape)
 267         expected_posteriors = label_sum / weight_sum
 268         label_priors = label_priors_update.eval().reshape(batch_shape)
 269         self.assertAllClose(label_priors, expected_posteriors)
 270 
 271         # Re-initialize labels to get a new random sample.
 272         session.run(labels.initializer)
 273 
 274   def testLabelPriorsUpdateWithWeights(self):
 275     # Checks the update of label priors with per-example weights.
 276     batch_size = 6
 277     num_labels = 5
 278     batch_shape = [batch_size, num_labels]
 279     labels = tf.Variable(
 280         tf.to_float(tf.greater(tf.random_uniform(batch_shape), 0.6)))
 281     weights = tf.Variable(tf.random_uniform(batch_shape) * 6.2)
 282 
 283     update_op = util.build_label_priors(labels, weights=weights)
 284 
 285     expected_weighted_label_counts = 1.0 + tf.reduce_sum(weights * labels, 0)
 286     expected_weight_sum = 2.0 + tf.reduce_sum(weights, 0)
 287     expected_label_posteriors = tf.divide(expected_weighted_label_counts,
 288                                           expected_weight_sum)
 289 
 290     with self.test_session() as session:
 291       tf.global_variables_initializer().run()
 292 
 293       updated_priors, expected_posteriors = session.run(
 294           [update_op, expected_label_posteriors])
 295       self.assertAllClose(updated_priors, expected_posteriors)

 298 class WeightedSurrogateLossTest(parameterized.TestCase, tf.test.TestCase):

 300   @parameterized.parameters(
 301       ('hinge', util.weighted_hinge_loss),
 302       ('xent', util.weighted_sigmoid_cross_entropy_with_logits))
 303   def testCompatibilityLoss(self, loss_name, loss_fn):
 304     x_shape = [28, 4]
 305     logits = tf.constant(np.random.randn(*x_shape).astype(np.float32))
 306     targets = tf.to_float(tf.constant(np.random.random_sample(x_shape) > 0.5))
 307     positive_weights = 0.66
 308     negative_weights = 11.1
 309     expected_loss = loss_fn(
 310         targets,
 311         logits,
 312         positive_weights=positive_weights,
 313         negative_weights=negative_weights)
 314     computed_loss = util.weighted_surrogate_loss(
 315         targets,
 316         logits,
 317         loss_name,
 318         positive_weights=positive_weights,
 319         negative_weights=negative_weights)
 320     with self.test_session():
 321       self.assertAllClose(expected_loss.eval(), computed_loss.eval())
