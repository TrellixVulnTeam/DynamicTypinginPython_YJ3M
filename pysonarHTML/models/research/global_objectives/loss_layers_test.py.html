 117   def testSimpleScores(self):
 118     # Tests the loss on data with only one negative example with score zero.
 119     # In this case, the loss should equal the surrogate loss on the scores with

 123                                    shape=[num_positives, 1])
 124     labels = tf.constant([0.0] + [1.0] * num_positives,
 125                          shape=[num_positives + 1, 1])

 129         loss_layers.roc_auc_loss(labels, scores, surrogate_type='hinge')[0])

 131         tf.maximum(1.0 - scores_positives, 0)) / (num_positives + 1)
 132     with self.test_session():
 133       self.assertAllClose(expected_loss.eval(), loss.eval())

 285   def testEquivalenceBetweenSingleAndMultiplePrecisions(self):
 286     """Checks recall at precision with different precision values.
 287 
 288     Runs recall at precision with multiple precision values, and runs each label
 289     seperately with its own precision value as a scalar. Validates that the
 290     returned loss values are the same.
 291     """
 292     target_precision = [0.2, 0.9, 0.4]
 293     num_labels = 3
 294     batch_shape = [30, num_labels]
 295     logits = tf.Variable(tf.random_normal(batch_shape))
 296     targets = tf.Variable(
 297         tf.to_float(tf.greater(tf.random_uniform(batch_shape), 0.4)))
 298     label_priors = tf.constant([0.45, 0.8, 0.3], shape=[num_labels])
 299 
 300     multi_label_loss, _ = loss_layers.recall_at_precision_loss(
 301         targets, logits, target_precision, label_priors=label_priors,
 302     )
 303 
 304     single_label_losses = [
 305         loss_layers.recall_at_precision_loss(
 306             tf.expand_dims(targets[:, i], -1),
 307             tf.expand_dims(logits[:, i], -1),
 308             target_precision[i],
 309             label_priors=label_priors[i])[0]
 310         for i in range(num_labels)
 311     ]
 312 
 313     single_label_losses = tf.concat(single_label_losses, 1)
 314 
 315     with self.test_session() as session:
 316       tf.global_variables_initializer().run()
 317       multi_label_loss_val, single_label_loss_val = session.run(
 318           [multi_label_loss, single_label_losses])
