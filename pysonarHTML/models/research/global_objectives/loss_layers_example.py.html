  15 """Example for using global objectives.
  16 
  17 Illustrate, using synthetic data, how using the precision_at_recall loss
  18 significanly improves the performace of a linear classifier.
  19 """
  20 from __future__ import absolute_import
  21 from __future__ import division
  22 from __future__ import print_function
  23 
  24 # Dependency imports
  25 import numpy as np
  26 from sklearn.metrics import precision_score
  27 import tensorflow as tf
  28 from global_objectives import loss_layers
  29 
  30 # When optimizing using global_objectives, if set to True then the saddle point
  31 # optimization steps are performed internally by the Tensorflow optimizer,
  32 # otherwise by dedicated saddle-point steps as part of the optimization loop.
  33 USE_GO_SADDLE_POINT_OPT = False
  34 
  35 TARGET_RECALL = 0.98
  36 TRAIN_ITERATIONS = 150
  37 LEARNING_RATE = 1.0
  38 GO_DUAL_RATE_FACTOR = 15.0
  39 NUM_CHECKPOINTS = 6
  40 
  41 EXPERIMENT_DATA_CONFIG = {
  42     'positives_centers': [[0, 1.0], [1, -0.5]],
  43     'negatives_centers': [[0, -0.5], [1, 1.0]],
  44     'positives_variances': [0.15, 0.1],
  45     'negatives_variances': [0.15, 0.1],
  46     'positives_counts': [500, 50],
  47     'negatives_counts': [3000, 100]
  48 }
  49

  51 def create_training_and_eval_data_for_experiment(**data_config):
  52   """Creates train and eval data sets.
  53 
  54   Note: The synthesized binary-labeled data is a mixture of four Gaussians - two
  55     positives and two negatives. The centers, variances, and sizes for each of
  56     the two positives and negatives mixtures are passed in the respective keys
  57     of data_config:
  58 
  59   Args:
  60       **data_config: Dictionary with Array entries as follows:
  61         positives_centers - float [2,2] two centers of positives data sets.
  62         negatives_centers - float [2,2] two centers of negatives data sets.
  63         positives_variances - float [2] Variances for the positives sets.
  64         negatives_variances - float [2] Variances for the negatives sets.
  65         positives_counts - int [2] Counts for each of the two positives sets.
  66         negatives_counts - int [2] Counts for each of the two negatives sets.
  67 
  68   Returns:
  69     A dictionary with two shuffled data sets created - one for training and one
  70     for eval. The dictionary keys are 'train_data', 'train_labels', 'eval_data',
  71     and 'eval_labels'. The data points are two-dimentional floats, and the
  72     labels are in {0,1}.
  73   """
  74   def data_points(is_positives, index):
  75     variance = data_config['positives_variances'
  76                            if is_positives else 'negatives_variances'][index]
  77     center = data_config['positives_centers'
  78                          if is_positives else 'negatives_centers'][index]
  79     count = data_config['positives_counts'
  80                         if is_positives else 'negatives_counts'][index]
  81     return variance*np.random.randn(count, 2) + np.array([center])
  82 
  83   def create_data():
  84     return np.concatenate([data_points(False, 0),
  85                            data_points(True, 0),
  86                            data_points(True, 1),
  87                            data_points(False, 1)], axis=0)
  88 
  89   def create_labels():
  90     """Creates an array of 0.0 or 1.0 labels for the data_config batches."""
  91     return np.array([0.0]*data_config['negatives_counts'][0] +
  92                     [1.0]*data_config['positives_counts'][0] +
  93                     [1.0]*data_config['positives_counts'][1] +
  94                     [0.0]*data_config['negatives_counts'][1])
  95 
  96   permutation = np.random.permutation(
  97       sum(data_config['positives_counts'] + data_config['negatives_counts']))
  98 
  99   train_data = create_data()[permutation, :]
 100   eval_data = create_data()[permutation, :]
 101   train_labels = create_labels()[permutation]
 102   eval_labels = create_labels()[permutation]
 103 
 104   return {
 105       'train_data': train_data,
 106       'train_labels': train_labels,
 107       'eval_data': eval_data,
 108       'eval_labels': eval_labels
 109   }

 112 def train_model(data, use_global_objectives):
 113   """Trains a linear model for maximal accuracy or precision at given recall."""
 114 
 115   def precision_at_recall(scores, labels, target_recall):
 116     """Computes precision - at target recall - over data."""
 117     positive_scores = scores[labels == 1.0]
 118     threshold = np.percentile(positive_scores, 100 - target_recall*100)
 119     predicted = scores >= threshold
 120     return precision_score(labels, predicted)
 121 
 122   w = tf.Variable(tf.constant([-1.0, -1.0], shape=[2, 1]), trainable=True,
 123                   name='weights', dtype=tf.float32)
 124   b = tf.Variable(tf.zeros([1]), trainable=True, name='biases',
 125                   dtype=tf.float32)
 126 
 127   logits = tf.matmul(tf.cast(data['train_data'], tf.float32), w) + b
 128 
 129   labels = tf.constant(
 130       data['train_labels'],
 131       shape=[len(data['train_labels']), 1],
 132       dtype=tf.float32)
 133 
 134   if use_global_objectives:
 135     loss, other_outputs = loss_layers.precision_at_recall_loss(
 136         labels, logits,
 137         TARGET_RECALL,
 138         dual_rate_factor=GO_DUAL_RATE_FACTOR)
 139     loss = tf.reduce_mean(loss)
 140   else:
 141     loss = tf.reduce_mean(
 142         tf.nn.sigmoid_cross_entropy_with_logits(labels=labels, logits=logits))
 143 
 144   global_step = tf.Variable(0, trainable=False)
 145 
 146   learning_rate = tf.train.polynomial_decay(
 147       LEARNING_RATE,
 148       global_step,
 149       TRAIN_ITERATIONS, (LEARNING_RATE / TRAIN_ITERATIONS),
 150       power=1.0,
 151       cycle=False,
 152       name='learning_rate')
 153 
 154   optimizer = tf.train.GradientDescentOptimizer(learning_rate)
 155 
 156   if (not use_global_objectives) or USE_GO_SADDLE_POINT_OPT:
 157     training_op = optimizer.minimize(loss, global_step=global_step)
 158   else:
 159     lambdas = other_outputs['lambdas']
 160     primal_update_op = optimizer.minimize(loss, var_list=[w, b])
 161     dual_update_op = optimizer.minimize(
 162         loss, global_step=global_step, var_list=[lambdas])
 163 
 164   # Training loop:
 165   with tf.Session() as sess:
 166     checkpoint_step = TRAIN_ITERATIONS // NUM_CHECKPOINTS
 167     sess.run(tf.global_variables_initializer())
 168     step = sess.run(global_step)
 169 
 170     while step <= TRAIN_ITERATIONS:
 171       if (not use_global_objectives) or USE_GO_SADDLE_POINT_OPT:
 172         _, step, loss_value, w_value, b_value = sess.run(
 173             [training_op, global_step, loss, w, b])
 174       else:
 175         _, w_value, b_value = sess.run([primal_update_op, w, b])
 176         _, loss_value, step = sess.run([dual_update_op, loss, global_step])
 177 
 178       if use_global_objectives:
 179         go_outputs = sess.run(other_outputs.values())
 180 
 181       if step % checkpoint_step == 0:
 182         precision = precision_at_recall(
 183             np.dot(data['train_data'], w_value) + b_value,
 184             data['train_labels'], TARGET_RECALL)
 185 
 186         tf.logging.info('Loss = %f Precision = %f', loss_value, precision)
 187         if use_global_objectives:
 188           for i, output_name in enumerate(other_outputs.keys()):
 189             tf.logging.info('\t%s = %f', output_name, go_outputs[i])
 190 
 191     w_value, b_value = sess.run([w, b])
 192     return precision_at_recall(np.dot(data['eval_data'], w_value) + b_value,
 193                                data['eval_labels'],
 194                                TARGET_RECALL)

 197 def main(unused_argv):
 198   del unused_argv
 199   experiment_data = create_training_and_eval_data_for_experiment(
 200       **EXPERIMENT_DATA_CONFIG)
 201   global_objectives_loss_precision = train_model(experiment_data, True)
 202   tf.logging.info('global_objectives precision at requested recall is %f',
 203                   global_objectives_loss_precision)
 204   cross_entropy_loss_precision = train_model(experiment_data, False)
 205   tf.logging.info('cross_entropy precision at requested recall is %f',
 206                   cross_entropy_loss_precision)

 209 if __name__ == '__main__':
 210   tf.logging.set_verbosity(tf.logging.INFO)
 211   tf.app.run()
