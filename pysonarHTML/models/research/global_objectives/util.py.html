  21 def weighted_sigmoid_cross_entropy_with_logits(labels,
  22                                                logits,
  23                                                positive_weights=1.0,
  24                                                negative_weights=1.0,
  25                                                name=None):
  26   """Computes a weighting of sigmoid cross entropy given `logits`.
  27 
  28   Measures the weighted probability error in discrete classification tasks in
  29   which classes are independent and not mutually exclusive.  For instance, one
  30   could perform multilabel classification where a picture can contain both an
  31   elephant and a dog at the same time. The class weight multiplies the
  32   different types of errors.
  33   For brevity, let `x = logits`, `z = labels`, `c = positive_weights`,
  34   `d = negative_weights`  The
  35   weighed logistic loss is
  36 
  37   ```
  38   c * z * -log(sigmoid(x)) + d * (1 - z) * -log(1 - sigmoid(x))
  39   = c * z * -log(1 / (1 + exp(-x))) - d * (1 - z) * log(exp(-x) / (1 + exp(-x)))
  40   = c * z * log(1 + exp(-x)) + d * (1 - z) * (-log(exp(-x)) + log(1 + exp(-x)))
  41   = c * z * log(1 + exp(-x)) + d * (1 - z) * (x + log(1 + exp(-x)))
  42   = (1 - z) * x * d + (1 - z + c * z ) * log(1 + exp(-x))
  43   =  - d * x * z + d * x + (d - d * z + c * z ) * log(1 + exp(-x))
  44   ```
  45 
  46   To ensure stability and avoid overflow, the implementation uses the identity
  47       log(1 + exp(-x)) = max(0,-x) + log(1 + exp(-abs(x)))
  48   and the result is computed as
  49 
  50     ```
  51     = -d * x * z + d * x
  52       + (d - d * z + c * z ) * (max(0,-x) + log(1 + exp(-abs(x))))
  53     ```
  54 
  55   Note that the loss is NOT an upper bound on the 0-1 loss, unless it is divided
  56   by log(2).
  57 
  58   Args:
  59     labels: A `Tensor` of type `float32` or `float64`. `labels` can be a 2D
  60       tensor with shape [batch_size, num_labels] or a 3D tensor with shape
  61       [batch_size, num_labels, K].
  62     logits: A `Tensor` of the same type and shape as `labels`. If `logits` has
  63       shape [batch_size, num_labels, K], the loss is computed separately on each
  64       slice [:, :, k] of `logits`.
  65     positive_weights: A `Tensor` that holds positive weights and has the
  66       following semantics according to its shape:
  67         scalar - A global positive weight.
  68         1D tensor - must be of size K, a weight for each 'attempt'
  69         2D tensor - of size [num_labels, K'] where K' is either K or 1.
  70       The `positive_weights` will be expanded to the left to match the
  71       dimensions of logits and labels.
  72     negative_weights: A `Tensor` that holds positive weight and has the
  73       semantics identical to positive_weights.
  74     name: A name for the operation (optional).
  75 
  76   Returns:
  77     A `Tensor` of the same shape as `logits` with the componentwise
  78       weighted logistic losses.
  79   """
  80   with tf.name_scope(
  81       name,
  82       'weighted_logistic_loss',
  83       [logits, labels, positive_weights, negative_weights]) as name:
  84     labels, logits, positive_weights, negative_weights = prepare_loss_args(
  85         labels, logits, positive_weights, negative_weights)
  86 
  87     softplus_term = tf.add(tf.maximum(-logits, 0.0),
  88                            tf.log(1.0 + tf.exp(-tf.abs(logits))))
  89     weight_dependent_factor = (
  90         negative_weights + (positive_weights - negative_weights) * labels)
  91     return (negative_weights * (logits - labels * logits) +
  92             weight_dependent_factor * softplus_term)

  95 def weighted_hinge_loss(labels,
  96                         logits,
  97                         positive_weights=1.0,
  98                         negative_weights=1.0,
  99                         name=None):
 100   """Computes weighted hinge loss given logits `logits`.
 101 
 102   The loss applies to multi-label classification tasks where labels are
 103   independent and not mutually exclusive. See also
 104   `weighted_sigmoid_cross_entropy_with_logits`.
 105 
 106   Args:
 107     labels: A `Tensor` of type `float32` or `float64`. Each entry must be
 108       either 0 or 1. `labels` can be a 2D tensor with shape
 109       [batch_size, num_labels] or a 3D tensor with shape
 110       [batch_size, num_labels, K].
 111     logits: A `Tensor` of the same type and shape as `labels`. If `logits` has
 112       shape [batch_size, num_labels, K], the loss is computed separately on each
 113       slice [:, :, k] of `logits`.
 114     positive_weights: A `Tensor` that holds positive weights and has the
 115       following semantics according to its shape:
 116         scalar - A global positive weight.
 117         1D tensor - must be of size K, a weight for each 'attempt'
 118         2D tensor - of size [num_labels, K'] where K' is either K or 1.
 119       The `positive_weights` will be expanded to the left to match the
 120       dimensions of logits and labels.
 121     negative_weights: A `Tensor` that holds positive weight and has the
 122       semantics identical to positive_weights.
 123     name: A name for the operation (optional).
 124 
 125   Returns:
 126     A `Tensor` of the same shape as `logits` with the componentwise
 127       weighted hinge loss.
 128   """
 129   with tf.name_scope(
 130       name, 'weighted_hinge_loss',
 131       [logits, labels, positive_weights, negative_weights]) as name:
 132     labels, logits, positive_weights, negative_weights = prepare_loss_args(
 133         labels, logits, positive_weights, negative_weights)
 134 
 135     positives_term = positive_weights * labels * tf.maximum(1.0 - logits, 0)
 136     negatives_term = (negative_weights * (1.0 - labels)
 137                       * tf.maximum(1.0 + logits, 0))
 138     return positives_term + negatives_term

 141 def weighted_surrogate_loss(labels,
 142                             logits,
 143                             surrogate_type='xent',
 144                             positive_weights=1.0,
 145                             negative_weights=1.0,
 146                             name=None):
 147   """Returns either weighted cross-entropy or hinge loss.
 148 
 149   For example `surrogate_type` is 'xent' returns the weighted cross
 150   entropy loss.
 151 
 152   Args:
 153    labels: A `Tensor` of type `float32` or `float64`. Each entry must be
 154       between 0 and 1. `labels` can be a 2D tensor with shape
 155       [batch_size, num_labels] or a 3D tensor with shape
 156       [batch_size, num_labels, K].
 157     logits: A `Tensor` of the same type and shape as `labels`. If `logits` has
 158       shape [batch_size, num_labels, K], each slice [:, :, k] represents an
 159       'attempt' to predict `labels` and the loss is computed per slice.
 160     surrogate_type: A string that determines which loss to return, supports
 161     'xent' for cross-entropy and 'hinge' for hinge loss.
 162     positive_weights: A `Tensor` that holds positive weights and has the
 163       following semantics according to its shape:
 164         scalar - A global positive weight.
 165         1D tensor - must be of size K, a weight for each 'attempt'
 166         2D tensor - of size [num_labels, K'] where K' is either K or 1.
 167       The `positive_weights` will be expanded to the left to match the
 168       dimensions of logits and labels.
 169     negative_weights: A `Tensor` that holds positive weight and has the
 170       semantics identical to positive_weights.
 171     name: A name for the operation (optional).
 172 
 173   Returns:
 174     The weigthed loss.
 175 
 176   Raises:
 177     ValueError: If value of `surrogate_type` is not supported.
 178   """
 179   with tf.name_scope(
 180       name, 'weighted_loss',
 181       [logits, labels, surrogate_type, positive_weights,
 182        negative_weights]) as name:
 183     if surrogate_type == 'xent':
 184       return weighted_sigmoid_cross_entropy_with_logits(
 185           logits=logits,
 186           labels=labels,
 187           positive_weights=positive_weights,
 188           negative_weights=negative_weights,
 189           name=name)
 190     elif surrogate_type == 'hinge':
 191       return weighted_hinge_loss(
 192           logits=logits,
 193           labels=labels,
 194           positive_weights=positive_weights,
 195           negative_weights=negative_weights,
 196           name=name)
 197     raise ValueError('surrogate_type %s not supported.' % surrogate_type)

 200 def expand_outer(tensor, rank):
 201   """Expands the given `Tensor` outwards to a target rank.
 202 
 203   For example if rank = 3 and tensor.shape is [3, 4], this function will expand
 204   to such that the resulting shape will be  [1, 3, 4].
 205 
 206   Args:
 207     tensor: The tensor to expand.
 208     rank: The target dimension.
 209 
 210   Returns:
 211     The expanded tensor.
 212 
 213   Raises:
 214     ValueError: If rank of `tensor` is unknown, or if `rank` is smaller than
 215       the rank of `tensor`.
 216   """
 217   if tensor.get_shape().ndims is None:
 218     raise ValueError('tensor dimension must be known.')
 219   if len(tensor.get_shape()) > rank:
 220     raise ValueError(
 221         '`rank` must be at least the current tensor dimension: (%s vs %s).' %
 222         (rank, len(tensor.get_shape())))
 223   while len(tensor.get_shape()) < rank:
 224     tensor = tf.expand_dims(tensor, 0)
 225   return tensor

 228 def build_label_priors(labels,
 229                        weights=None,
 230                        positive_pseudocount=1.0,
 231                        negative_pseudocount=1.0,
 232                        variables_collections=None):
 233   """Creates an op to maintain and update label prior probabilities.
 234 
 235   For each label, the label priors are estimated as
 236       (P + sum_i w_i y_i) / (P + N + sum_i w_i),
 237   where y_i is the ith label, w_i is the ith weight, P is a pseudo-count of
 238   positive labels, and N is a pseudo-count of negative labels. The index i
 239   ranges over all labels observed during all evaluations of the returned op.
 240 
 241   Args:
 242     labels: A `Tensor` with shape [batch_size, num_labels]. Entries should be
 243       in [0, 1].
 244     weights: Coefficients representing the weight of each label. Must be either
 245       a Tensor of shape [batch_size, num_labels] or `None`, in which case each
 246       weight is treated as 1.0.
 247     positive_pseudocount: Number of positive labels used to initialize the label
 248       priors.
 249     negative_pseudocount: Number of negative labels used to initialize the label
 250       priors.
 251     variables_collections: Optional list of collections for created variables.
 252 
 253   Returns:
 254     label_priors: An op to update the weighted label_priors. Gives the
 255       current value of the label priors when evaluated.
 256   """
 257   dtype = labels.dtype.base_dtype
 258   num_labels = get_num_labels(labels)
 259 
 260   if weights is None:
 261     weights = tf.ones_like(labels)
 262 
 263   # We disable partitioning while constructing dual variables because they will
 264   # be updated with assign, which is not available for partitioned variables.
 265   partitioner = tf.get_variable_scope().partitioner
 266   try:
 267     tf.get_variable_scope().set_partitioner(None)
 268     # Create variable and update op for weighted label counts.
 269     weighted_label_counts = tf.contrib.framework.model_variable(
 270         name='weighted_label_counts',
 271         shape=[num_labels],
 272         dtype=dtype,
 273         initializer=tf.constant_initializer(
 274             [positive_pseudocount] * num_labels, dtype=dtype),
 275         collections=variables_collections,
 276         trainable=False)
 277     weighted_label_counts_update = weighted_label_counts.assign_add(
 278         tf.reduce_sum(weights * labels, 0))
 279 
 280     # Create variable and update op for the sum of the weights.
 281     weight_sum = tf.contrib.framework.model_variable(
 282         name='weight_sum',
 283         shape=[num_labels],
 284         dtype=dtype,
 285         initializer=tf.constant_initializer(
 286             [positive_pseudocount + negative_pseudocount] * num_labels,
 287             dtype=dtype),
 288         collections=variables_collections,
 289         trainable=False)
 290     weight_sum_update = weight_sum.assign_add(tf.reduce_sum(weights, 0))
 291 
 292   finally:
 293     tf.get_variable_scope().set_partitioner(partitioner)
 294 
 295   label_priors = tf.div(
 296       weighted_label_counts_update,
 297       weight_sum_update)
 298   return label_priors

 301 def convert_and_cast(value, name, dtype):
 302   """Convert input to tensor and cast to dtype.
 303 
 304   Args:
 305     value: An object whose type has a registered Tensor conversion function,
 306         e.g. python numerical type or numpy array.
 307     name: Name to use for the new Tensor, if one is created.
 308     dtype: Optional element type for the returned tensor.
 309 
 310   Returns:
 311     A tensor.
 312   """
 313   return tf.cast(tf.convert_to_tensor(value, name=name), dtype=dtype)

 316 def prepare_loss_args(labels, logits, positive_weights, negative_weights):
 317   """Prepare arguments for weighted loss functions.
 318 
 319   If needed, will convert given arguments to appropriate type and shape.
 320 
 321   Args:
 322     labels: labels or labels of the loss function.
 323     logits: Logits of the loss function.
 324     positive_weights: Weight on the positive examples.
 325     negative_weights: Weight on the negative examples.
 326 
 327   Returns:
 328     Converted labels, logits, positive_weights, negative_weights.
 329   """
 330   logits = tf.convert_to_tensor(logits, name='logits')
 331   labels = convert_and_cast(labels, 'labels', logits.dtype)
 332   if len(labels.get_shape()) == 2 and len(logits.get_shape()) == 3:
 333     labels = tf.expand_dims(labels, [2])
 334 
 335   positive_weights = convert_and_cast(positive_weights, 'positive_weights',
 336                                       logits.dtype)
 337   positive_weights = expand_outer(positive_weights, logits.get_shape().ndims)
 338   negative_weights = convert_and_cast(negative_weights, 'negative_weights',
 339                                       logits.dtype)
 340   negative_weights = expand_outer(negative_weights, logits.get_shape().ndims)
 341   return labels, logits, positive_weights, negative_weights

 344 def get_num_labels(labels_or_logits):
 345   """Returns the number of labels inferred from labels_or_logits."""
 346   if labels_or_logits.get_shape().ndims <= 1:
 347     return 1
 348   return labels_or_logits.get_shape()[1].value
