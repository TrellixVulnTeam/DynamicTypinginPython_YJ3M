  17 from __future__ import absolute_import

  25 from domain_adaptation.pixel_domain_adaptation import pixelda_preprocess

  28 class PixelDAPreprocessTest(tf.test.TestCase):
  29 
  30   def assert_preprocess_classification_is_centered(self, dtype, is_training):
  31     tf.set_random_seed(0)
  32 
  33     if dtype == tf.uint8:
  34       image = tf.random_uniform((100, 200, 3), maxval=255, dtype=tf.int64)
  35       image = tf.cast(image, tf.uint8)
  36     else:
  37       image = tf.random_uniform((100, 200, 3), maxval=1.0, dtype=dtype)
  38 
  39     labels = {}
  40     image, labels = pixelda_preprocess.preprocess_classification(
  41         image, labels, is_training=is_training)
  42 
  43     with self.test_session() as sess:
  44       np_image = sess.run(image)
  45 
  46       self.assertTrue(np_image.min() <= -0.95)
  47       self.assertTrue(np_image.min() >= -1.0)
  48       self.assertTrue(np_image.max() >= 0.95)
  49       self.assertTrue(np_image.max() <= 1.0)
  50 
  51   def testPreprocessClassificationZeroCentersUint8DuringTrain(self):
  52     self.assert_preprocess_classification_is_centered(
  53         tf.uint8, is_training=True)
  54 
  55   def testPreprocessClassificationZeroCentersUint8DuringTest(self):
  56     self.assert_preprocess_classification_is_centered(
  57         tf.uint8, is_training=False)
  58 
  59   def testPreprocessClassificationZeroCentersFloatDuringTrain(self):
  60     self.assert_preprocess_classification_is_centered(
  61         tf.float32, is_training=True)
  62 
  63   def testPreprocessClassificationZeroCentersFloatDuringTest(self):
  64     self.assert_preprocess_classification_is_centered(
  65         tf.float32, is_training=False)
