  15 """Define model HParams."""
  16 import tensorflow as tf

  19 def create_hparams(hparam_string=None):
  20   """Create model hyperparameters. Parse nondefault from given string."""
  21   hparams = tf.contrib.training.HParams(
  22       # The name of the architecture to use.
  23       arch='resnet',
  24       lrelu_leakiness=0.2,
  25       batch_norm_decay=0.9,
  26       weight_decay=1e-5,
  27       normal_init_std=0.02,
  28       generator_kernel_size=3,
  29       discriminator_kernel_size=3,
  30 
  31       # Stop training after this many examples are processed
  32       # If none, train indefinitely
  33       num_training_examples=0,
  34 
  35       # Apply data augmentation to datasets
  36       # Applies only in training job
  37       augment_source_images=False,
  38       augment_target_images=False,
  39 
  40       # Discriminator
  41       # Number of filters in first layer of discriminator
  42       num_discriminator_filters=64,
  43       discriminator_conv_block_size=1,  # How many convs to have at each size
  44       discriminator_filter_factor=2.0,  # Multiply # filters by this each layer
  45       # Add gaussian noise with this stddev to every hidden layer of D
  46       discriminator_noise_stddev=0.2,  # lmetz: Start seeing results at >= 0.1
  47       # If true, add this gaussian noise to input images to D as well
  48       discriminator_image_noise=False,
  49       discriminator_first_stride=1,  # Stride in first conv of discriminator
  50       discriminator_do_pooling=False,  # If true, replace stride 2 with avg pool
  51       discriminator_dropout_keep_prob=0.9,  # keep probability for dropout
  52 
  53       # DCGAN Generator
  54       # Number of filters in generator decoder last layer (repeatedly halved
  55       # from 1st layer)
  56       num_decoder_filters=64,
  57       # Number of filters in generator encoder 1st layer (repeatedly doubled
  58       # after 1st layer)
  59       num_encoder_filters=64,
  60 
  61       # This is the shape to which the noise vector is projected (if we're
  62       # transferring from noise).
  63       # Write this way instead of [4, 4, 64] for hparam search flexibility
  64       projection_shape_size=4,
  65       projection_shape_channels=64,
  66 
  67       # Indicates the method by which we enlarge the spatial representation
  68       # of an image. Possible values include:
  69       # - resize_conv: Performs a nearest neighbor resize followed by a conv.
  70       # - conv2d_transpose: Performs a conv2d_transpose.
  71       upsample_method='resize_conv',
  72 
  73       # Visualization
  74       summary_steps=500,  # Output image summary every N steps
  75 
  76       ###################################
  77       # Task Classifier Hyperparameters #
  78       ###################################
  79 
  80       # Which task-specific prediction tower to use. Possible choices are:
  81       #  none: No task tower.
  82       #  doubling_pose_estimator: classifier + quaternion regressor.
  83       #    [conv + pool]* + FC
  84       # Classifiers used in DSN paper:
  85       #  gtsrb: Classifier used for GTSRB
  86       #  svhn: Classifier used for SVHN
  87       #  mnist: Classifier used for MNIST
  88       #  pose_mini: Classifier + regressor used for pose_mini
  89       task_tower='doubling_pose_estimator',
  90       weight_decay_task_classifier=1e-5,
  91       source_task_loss_weight=1.0,
  92       transferred_task_loss_weight=1.0,
  93 
  94       # Number of private layers in doubling_pose_estimator task tower
  95       num_private_layers=2,
  96 
  97       # The weight for the log quaternion loss we use for source and transferred
  98       # samples of the cropped_linemod dataset.
  99       # In the DSN work, 1/8 of the classifier weight worked well for our log
 100       # quaternion loss
 101       source_pose_weight=0.125 * 2.0,
 102       transferred_pose_weight=0.125 * 1.0,
 103 
 104       # If set to True, the style transfer network also attempts to change its
 105       # weights to maximize the performance of the task tower. If set to False,
 106       # then the style transfer network only attempts to change its weights to
 107       # make the transferred images more likely according to the domain
 108       # classifier.
 109       task_tower_in_g_step=True,
 110       task_loss_in_g_weight=1.0,  # Weight of task loss in G
 111 
 112       #########################################
 113       # 'simple` generator arch model hparams #
 114       #########################################
 115       simple_num_conv_layers=1,
 116       simple_conv_filters=8,
 117 
 118       #########################
 119       # Resnet Hyperparameters#
 120       #########################
 121       resnet_blocks=6,  # Number of resnet blocks
 122       resnet_filters=64,  # Number of filters per conv in resnet blocks
 123       # If true, add original input back to result of convolutions inside the
 124       # resnet arch. If false, it turns into a simple stack of conv/relu/BN
 125       # layers.
 126       resnet_residuals=True,
 127 
 128       #######################################
 129       # The residual / interpretable model. #
 130       #######################################
 131       res_int_blocks=2,  # The number of residual blocks.
 132       res_int_convs=2,  # The number of conv calls inside each block.
 133       res_int_filters=64,  # The number of filters used by each convolution.
 134 
 135       ####################
 136       # Latent variables #
 137       ####################
 138       # if true, then generate random noise and project to input for generator
 139       noise_channel=True,
 140       # The number of dimensions in the input noise vector.
 141       noise_dims=10,
 142 
 143       # If true, then one hot encode source image class and project as an
 144       # additional channel for the input to generator. This gives the generator
 145       # access to the class, which may help generation performance.
 146       condition_on_source_class=False,
 147 
 148       ########################
 149       # Loss Hyperparameters #
 150       ########################
 151       domain_loss_weight=1.0,
 152       style_transfer_loss_weight=1.0,
 153 
 154       ########################################################################
 155       # Encourages the transferred images to be similar to the source images #
 156       # using a configurable metric.                                         #
 157       ########################################################################
 158 
 159       # The weight of the loss function encouraging the source and transferred
 160       # images to be similar. If set to 0, then the loss function is not used.
 161       transferred_similarity_loss_weight=0.0,
 162 
 163       # The type of loss used to encourage transferred and source image
 164       # similarity. Valid values include:
 165       #   mpse: Mean Pairwise Squared Error
 166       #   mse: Mean Squared Error
 167       #   hinged_mse: Computes the mean squared error using squared differences
 168       #     greater than hparams.transferred_similarity_max_diff
 169       #   hinged_mae: Computes the mean absolute error using absolute
 170       #     differences greater than hparams.transferred_similarity_max_diff.
 171       transferred_similarity_loss='mpse',
 172 
 173       # The maximum allowable difference between the source and target images.
 174       # This value is used, in effect, to produce a hinge loss. Note that the
 175       # range of values should be between 0 and 1.
 176       transferred_similarity_max_diff=0.4,
 177 
 178       ################################
 179       # Optimization Hyperparameters #
 180       ################################
 181       learning_rate=0.001,
 182       batch_size=32,
 183       lr_decay_steps=20000,
 184       lr_decay_rate=0.95,
 185 
 186       # Recomendation from the DCGAN paper:
 187       adam_beta1=0.5,
 188       clip_gradient_norm=5.0,
 189 
 190       # The number of times we run the discriminator train_op in a row.
 191       discriminator_steps=1,
 192 
 193       # The number of times we run the generator train_op in a row.
 194       generator_steps=1)
 195 
 196   if hparam_string:
 197     tf.logging.info('Parsing command line hparams: %s', hparam_string)
 198     hparams.parse(hparam_string)
 199 
 200   tf.logging.info('Final parsed hparams: %s', hparams.values())
 201   return hparams
