  28 def add_domain_classifier_losses(end_points, hparams):
  29   """Adds losses related to the domain-classifier.
  30 
  31   Args:
  32     end_points: A map of network end point names to `Tensors`.
  33     hparams: The hyperparameters struct.
  34 
  35   Returns:
  36     loss: A `Tensor` representing the total task-classifier loss.
  37   """
  38   if hparams.domain_loss_weight == 0:
  39     tf.logging.info(
  40         'Domain classifier loss weight is 0, so not creating losses.')
  41     return 0
  42 
  43   # The domain prediction loss is minimized with respect to the domain
  44   # classifier features only. Its aim is to predict the domain of the images.
  45   # Note: 1 = 'real image' label, 0 = 'fake image' label
  46   transferred_domain_loss = tf.losses.sigmoid_cross_entropy(
  47       multi_class_labels=tf.zeros_like(end_points['transferred_domain_logits']),
  48       logits=end_points['transferred_domain_logits'])
  49   tf.summary.scalar('Domain_loss_transferred', transferred_domain_loss)
  50 
  51   target_domain_loss = tf.losses.sigmoid_cross_entropy(
  52       multi_class_labels=tf.ones_like(end_points['target_domain_logits']),
  53       logits=end_points['target_domain_logits'])
  54   tf.summary.scalar('Domain_loss_target', target_domain_loss)
  55 
  56   # Compute the total domain loss:
  57   total_domain_loss = transferred_domain_loss + target_domain_loss
  58   total_domain_loss *= hparams.domain_loss_weight
  59   tf.summary.scalar('Domain_loss_total', total_domain_loss)
  60 
  61   return total_domain_loss

  63 def log_quaternion_loss_batch(predictions, labels, params):
  64   """A helper function to compute the error between quaternions.
  65 
  66   Args:
  67     predictions: A Tensor of size [batch_size, 4].
  68     labels: A Tensor of size [batch_size, 4].
  69     params: A dictionary of parameters. Expecting 'use_logging', 'batch_size'.
  70 
  71   Returns:
  72     A Tensor of size [batch_size], denoting the error between the quaternions.
  73   """
  74   use_logging = params['use_logging']
  75   assertions = []
  76   if use_logging:
  77     assertions.append(
  78         tf.Assert(
  79             tf.reduce_all(
  80                 tf.less(
  81                     tf.abs(tf.reduce_sum(tf.square(predictions), [1]) - 1),
  82                     1e-4)),
  83             ['The l2 norm of each prediction quaternion vector should be 1.']))
  84     assertions.append(
  85         tf.Assert(
  86             tf.reduce_all(
  87                 tf.less(
  88                     tf.abs(tf.reduce_sum(tf.square(labels), [1]) - 1), 1e-4)),
  89             ['The l2 norm of each label quaternion vector should be 1.']))
  90 
  91   with tf.control_dependencies(assertions):
  92     product = tf.multiply(predictions, labels)
  93   internal_dot_products = tf.reduce_sum(product, [1])
  94 
  95   if use_logging:
  96     internal_dot_products = tf.Print(internal_dot_products, [
  97         internal_dot_products,
  98         tf.shape(internal_dot_products)
  99     ], 'internal_dot_products:')
 100 
 101   logcost = tf.log(1e-4 + 1 - tf.abs(internal_dot_products))
 102   return logcost

 105 def log_quaternion_loss(predictions, labels, params):
 106   """A helper function to compute the mean error between batches of quaternions.
 107 
 108   The caller is expected to add the loss to the graph.
 109 
 110   Args:
 111     predictions: A Tensor of size [batch_size, 4].
 112     labels: A Tensor of size [batch_size, 4].
 113     params: A dictionary of parameters. Expecting 'use_logging', 'batch_size'.
 114 
 115   Returns:
 116     A Tensor of size 1, denoting the mean error between batches of quaternions.
 117   """
 118   use_logging = params['use_logging']
 119   logcost = log_quaternion_loss_batch(predictions, labels, params)
 120   logcost = tf.reduce_sum(logcost, [0])
 121   batch_size = params['batch_size']
 122   logcost = tf.multiply(logcost, 1.0 / batch_size, name='log_quaternion_loss')
 123   if use_logging:
 124     logcost = tf.Print(
 125         logcost, [logcost], '[logcost]', name='log_quaternion_loss_print')
 126   return logcost

 128 def _quaternion_loss(labels, predictions, weight, batch_size, domain,
 129                      add_summaries):
 130   """Creates a Quaternion Loss.
 131 
 132   Args:
 133     labels: The true quaternions.
 134     predictions: The predicted quaternions.
 135     weight: A scalar weight.
 136     batch_size: The size of the batches.
 137     domain: The name of the domain from which the labels were taken.
 138     add_summaries: Whether or not to add summaries for the losses.
 139 
 140   Returns:
 141     A `Tensor` representing the loss.
 142   """
 143   assert domain in ['Source', 'Transferred']
 144 
 145   params = {'use_logging': False, 'batch_size': batch_size}
 146   loss = weight * log_quaternion_loss(labels, predictions, params)
 147 
 148   if add_summaries:
 149     assert_op = tf.Assert(tf.is_finite(loss), [loss])
 150     with tf.control_dependencies([assert_op]):
 151       tf.summary.histogram(
 152           'Log_Quaternion_Loss_%s' % domain, loss, collections='losses')
 153       tf.summary.scalar(
 154           'Task_Quaternion_Loss_%s' % domain, loss, collections='losses')
 155 
 156   return loss

 159 def _add_task_specific_losses(end_points, source_labels, num_classes, hparams,
 160                               add_summaries=False):
 161   """Adds losses related to the task-classifier.
 162 
 163   Args:
 164     end_points: A map of network end point names to `Tensors`.
 165     source_labels: A dictionary of output labels to `Tensors`.
 166     num_classes: The number of classes used by the classifier.
 167     hparams: The hyperparameters struct.
 168     add_summaries: Whether or not to add the summaries.
 169 
 170   Returns:
 171     loss: A `Tensor` representing the total task-classifier loss.
 172   """
 173   # TODO(ddohan): Make sure the l2 regularization is added to the loss
 174 
 175   one_hot_labels = slim.one_hot_encoding(source_labels['class'], num_classes)
 176   total_loss = 0
 177 
 178   if 'source_task_logits' in end_points:
 179     loss = tf.losses.softmax_cross_entropy(
 180         onehot_labels=one_hot_labels,
 181         logits=end_points['source_task_logits'],
 182         weights=hparams.source_task_loss_weight)
 183     if add_summaries:
 184       tf.summary.scalar('Task_Classifier_Loss_Source', loss)
 185     total_loss += loss
 186 
 187   if 'transferred_task_logits' in end_points:
 188     loss = tf.losses.softmax_cross_entropy(
 189         onehot_labels=one_hot_labels,
 190         logits=end_points['transferred_task_logits'],
 191         weights=hparams.transferred_task_loss_weight)
 192     if add_summaries:
 193       tf.summary.scalar('Task_Classifier_Loss_Transferred', loss)
 194     total_loss += loss
 195 
 196   #########################
 197   # Pose specific losses. #
 198   #########################
 199   if 'quaternion' in source_labels:
 200     total_loss += _quaternion_loss(
 201         source_labels['quaternion'],
 202         end_points['source_quaternion'],
 203         hparams.source_pose_weight,
 204         hparams.batch_size,
 205         'Source',
 206         add_summaries)
 207 
 208     total_loss += _quaternion_loss(
 209         source_labels['quaternion'],
 210         end_points['transferred_quaternion'],
 211         hparams.transferred_pose_weight,
 212         hparams.batch_size,
 213         'Transferred',
 214         add_summaries)
 215 
 216   if add_summaries:
 217     tf.summary.scalar('Task_Loss_Total', total_loss)
 218 
 219   return total_loss

 222 def _transferred_similarity_loss(reconstructions,
 223                                  source_images,
 224                                  weight=1.0,
 225                                  method='mse',
 226                                  max_diff=0.4,
 227                                  name='similarity'):
 228   """Computes a loss encouraging similarity between source and transferred.
 229 
 230   Args:
 231     reconstructions: A `Tensor` of shape [batch_size, height, width, channels]
 232     source_images: A `Tensor` of shape [batch_size, height, width, channels].
 233     weight: Multiple similarity loss by this weight before returning
 234     method: One of:
 235       mpse = Mean Pairwise Squared Error
 236       mse = Mean Squared Error
 237       hinged_mse = Computes the mean squared error using squared differences
 238         greater than hparams.transferred_similarity_max_diff
 239       hinged_mae = Computes the mean absolute error using absolute
 240         differences greater than hparams.transferred_similarity_max_diff.
 241     max_diff: Maximum unpenalized difference for hinged losses
 242     name: Identifying name to use for creating summaries
 243 
 244 
 245   Returns:
 246     A `Tensor` representing the transferred similarity loss.
 247 
 248   Raises:
 249     ValueError: if `method` is not recognized.
 250   """
 251   if weight == 0:
 252     return 0
 253 
 254   source_channels = source_images.shape.as_list()[-1]
 255   reconstruction_channels = reconstructions.shape.as_list()[-1]
 256 
 257   # Convert grayscale source to RGB if target is RGB
 258   if source_channels == 1 and reconstruction_channels != 1:
 259     source_images = tf.tile(source_images, [1, 1, 1, reconstruction_channels])
 260   if reconstruction_channels == 1 and source_channels != 1:
 261     reconstructions = tf.tile(reconstructions, [1, 1, 1, source_channels])
 262 
 263   if method == 'mpse':
 264     reconstruction_similarity_loss_fn = (
 265         tf.contrib.losses.mean_pairwise_squared_error)
 266   elif method == 'masked_mpse':
 267 
 268     def masked_mpse(predictions, labels, weight):
 269       """Masked mpse assuming we have a depth to create a mask from."""
 270       assert labels.shape.as_list()[-1] == 4
 271       mask = tf.to_float(tf.less(labels[:, :, :, 3:4], 0.99))
 272       mask = tf.tile(mask, [1, 1, 1, 4])
 273       predictions *= mask
 274       labels *= mask
 275       tf.image_summary('masked_pred', predictions)
 276       tf.image_summary('masked_label', labels)
 277       return tf.contrib.losses.mean_pairwise_squared_error(
 278           predictions, labels, weight)
 279 
 280     reconstruction_similarity_loss_fn = masked_mpse
 281   elif method == 'mse':
 282     reconstruction_similarity_loss_fn = tf.contrib.losses.mean_squared_error
 283   elif method == 'hinged_mse':
 284 
 285     def hinged_mse(predictions, labels, weight):
 286       diffs = tf.square(predictions - labels)
 287       diffs = tf.maximum(0.0, diffs - max_diff)
 288       return tf.reduce_mean(diffs) * weight
 289 
 290     reconstruction_similarity_loss_fn = hinged_mse
 291   elif method == 'hinged_mae':
 292 
 293     def hinged_mae(predictions, labels, weight):
 294       diffs = tf.abs(predictions - labels)
 295       diffs = tf.maximum(0.0, diffs - max_diff)
 296       return tf.reduce_mean(diffs) * weight
 297 
 298     reconstruction_similarity_loss_fn = hinged_mae
 299   else:
 300     raise ValueError('Unknown reconstruction loss %s' % method)
 301 
 302   reconstruction_similarity_loss = reconstruction_similarity_loss_fn(
 303       reconstructions, source_images, weight)
 304 
 305   name = '%s_Similarity_(%s)' % (name, method)
 306   tf.summary.scalar(name, reconstruction_similarity_loss)
 307   return reconstruction_similarity_loss

 310 def g_step_loss(source_images, source_labels, end_points, hparams, num_classes):
 311   """Configures the loss function which runs during the g-step.
 312 
 313   Args:
 314     source_images: A `Tensor` of shape [batch_size, height, width, channels].
 315     source_labels: A dictionary of `Tensors` of shape [batch_size]. Valid keys
 316       are 'class' and 'quaternion'.
 317     end_points: A map of the network end points.
 318     hparams: The hyperparameters struct.
 319     num_classes: Number of classes for classifier loss
 320 
 321   Returns:
 322     A `Tensor` representing a loss function.
 323 
 324   Raises:
 325     ValueError: if hparams.transferred_similarity_loss_weight is non-zero but
 326       hparams.transferred_similarity_loss is invalid.
 327   """
 328   generator_loss = 0
 329 
 330   ################################################################
 331   # Adds a loss which encourages the discriminator probabilities #
 332   # to be high (near one).
 333   ################################################################
 334 
 335   # As per the GAN paper, maximize the log probs, instead of minimizing
 336   # log(1-probs). Since we're minimizing, we'll minimize -log(probs) which is
 337   # the same thing.
 338   style_transfer_loss = tf.losses.sigmoid_cross_entropy(
 339       logits=end_points['transferred_domain_logits'],
 340       multi_class_labels=tf.ones_like(end_points['transferred_domain_logits']),
 341       weights=hparams.style_transfer_loss_weight)
 342   tf.summary.scalar('Style_transfer_loss', style_transfer_loss)
 343   generator_loss += style_transfer_loss
 344 
 345   # Optimizes the style transfer network to produce transferred images similar
 346   # to the source images.
 347   generator_loss += _transferred_similarity_loss(
 348       end_points['transferred_images'],
 349       source_images,
 350       weight=hparams.transferred_similarity_loss_weight,
 351       method=hparams.transferred_similarity_loss,
 352       name='transferred_similarity')
 353 
 354   # Optimizes the style transfer network to maximize classification accuracy.
 355   if source_labels is not None and hparams.task_tower_in_g_step:
 356     generator_loss += _add_task_specific_losses(
 357         end_points, source_labels, num_classes,
 358         hparams) * hparams.task_loss_in_g_weight
 359 
 360   return generator_loss

 363 def d_step_loss(end_points, source_labels, num_classes, hparams):
 364   """Configures the losses during the D-Step.
 365 
 366   Note that during the D-step, the model optimizes both the domain (binary)
 367   classifier and the task classifier.
 368 
 369   Args:
 370     end_points: A map of the network end points.
 371     source_labels: A dictionary of output labels to `Tensors`.
 372     num_classes: The number of classes used by the classifier.
 373     hparams: The hyperparameters struct.
 374 
 375   Returns:
 376     A `Tensor` representing the value of the D-step loss.
 377   """
 378   domain_classifier_loss = add_domain_classifier_losses(end_points, hparams)
 379 
 380   task_classifier_loss = 0
 381   if source_labels is not None:
 382     task_classifier_loss = _add_task_specific_losses(
 383         end_points, source_labels, num_classes, hparams, add_summaries=True)
 384 
 385   return domain_classifier_loss + task_classifier_loss
