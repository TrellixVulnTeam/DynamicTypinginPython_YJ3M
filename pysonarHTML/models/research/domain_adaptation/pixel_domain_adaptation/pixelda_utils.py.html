  28 def remove_depth(images):
  29   """Takes a batch of images and remove depth channel if present."""
  30   if images.shape.as_list()[-1] == 4:
  31     return images[:, :, :, 0:3]
  32   return images

  35 def image_grid(images, max_grid_size=4):
  36   """Given images and N, return first N^2 images as an NxN image grid.
  37 
  38   Args:
  39     images: a `Tensor` of size [batch_size, height, width, channels]
  40     max_grid_size: Maximum image grid height/width
  41 
  42   Returns:
  43     Single image batch, of dim [1, h*n, w*n, c]
  44   """
  45   images = remove_depth(images)
  46   batch_size = images.shape.as_list()[0]
  47   grid_size = min(int(math.sqrt(batch_size)), max_grid_size)
  48   assert images.shape.as_list()[0] >= grid_size * grid_size
  49 
  50   # If we have a depth channel
  51   if images.shape.as_list()[-1] == 4:
  52     images = images[:grid_size * grid_size, :, :, 0:3]
  53     depth = tf.image.grayscale_to_rgb(images[:grid_size * grid_size, :, :, 3:4])
  54 
  55     images = tf.reshape(images, [-1, images.shape.as_list()[2], 3])
  56     split = tf.split(0, grid_size, images)
  57     depth = tf.reshape(depth, [-1, images.shape.as_list()[2], 3])
  58     depth_split = tf.split(0, grid_size, depth)
  59     grid = tf.concat(split + depth_split, 1)
  60     return tf.expand_dims(grid, 0)
  61   else:
  62     images = images[:grid_size * grid_size, :, :, :]
  63     images = tf.reshape(
  64         images, [-1, images.shape.as_list()[2],
  65                  images.shape.as_list()[3]])
  66     split = tf.split(images, grid_size, 0)
  67     grid = tf.concat(split, 1)
  68     return tf.expand_dims(grid, 0)

  71 def source_and_output_image_grid(output_images,
  72                                  source_images=None,
  73                                  max_grid_size=4):
  74   """Create NxN image grid for output, concatenate source grid if given.
  75 
  76   Makes grid out of output_images and, if provided, source_images, and
  77   concatenates them.
  78 
  79   Args:
  80     output_images: [batch_size, h, w, c] tensor of images
  81     source_images: optional[batch_size, h, w, c] tensor of images
  82     max_grid_size: Image grid height/width
  83 
  84   Returns:
  85     Single image batch, of dim [1, h*n, w*n, c]
  86 
  87 
  88   """
  89   output_grid = image_grid(output_images, max_grid_size=max_grid_size)
  90   if source_images is not None:
  91     source_grid = image_grid(source_images, max_grid_size=max_grid_size)
  92     # Make sure they have the same # of channels before concat
  93     # Assumes either 1 or 3 channels
  94     if output_grid.shape.as_list()[-1] != source_grid.shape.as_list()[-1]:
  95       if output_grid.shape.as_list()[-1] == 1:
  96         output_grid = tf.tile(output_grid, [1, 1, 1, 3])
  97       if source_grid.shape.as_list()[-1] == 1:
  98         source_grid = tf.tile(source_grid, [1, 1, 1, 3])
  99     output_grid = tf.concat([output_grid, source_grid], 1)
 100   return output_grid

 103 def summarize_model(end_points):
 104   """Summarizes the given model via its end_points.
 105 
 106   Args:
 107     end_points: A dictionary of end_point names to `Tensor`.
 108   """
 109   tf.summary.histogram('domain_logits_transferred',
 110                        tf.sigmoid(end_points['transferred_domain_logits']))
 111 
 112   tf.summary.histogram('domain_logits_target',
 113                        tf.sigmoid(end_points['target_domain_logits']))

 116 def summarize_transferred_grid(transferred_images,
 117                                source_images=None,
 118                                name='Transferred'):
 119   """Produces a visual grid summarization of the image transferrence.
 120 
 121   Args:
 122     transferred_images: A `Tensor` of size [batch_size, height, width, c].
 123     source_images: A `Tensor` of size [batch_size, height, width, c].
 124     name: Name to use in summary name
 125   """
 126   if source_images is not None:
 127     grid = source_and_output_image_grid(transferred_images, source_images)
 128   else:
 129     grid = image_grid(transferred_images)
 130   tf.summary.image('%s_Images_Grid' % name, grid, max_outputs=1)

 133 def summarize_transferred(source_images,
 134                           transferred_images,
 135                           max_images=20,
 136                           name='Transferred'):
 137   """Produces a visual summary of the image transferrence.
 138 
 139   This summary displays the source image, transferred image, and a grayscale
 140   difference image which highlights the differences between input and output.
 141 
 142   Args:
 143     source_images: A `Tensor` of size [batch_size, height, width, channels].
 144     transferred_images: A `Tensor` of size [batch_size, height, width, channels]
 145     max_images: The number of images to show.
 146     name: Name to use in summary name
 147 
 148   Raises:
 149     ValueError: If number of channels in source and target are incompatible
 150   """
 151   source_channels = source_images.shape.as_list()[-1]
 152   transferred_channels = transferred_images.shape.as_list()[-1]
 153   if source_channels < transferred_channels:
 154     if source_channels != 1:
 155       raise ValueError(
 156           'Source must be 1 channel or same # of channels as target')
 157     source_images = tf.tile(source_images, [1, 1, 1, transferred_channels])
 158   if transferred_channels < source_channels:
 159     if transferred_channels != 1:
 160       raise ValueError(
 161           'Target must be 1 channel or same # of channels as source')
 162     transferred_images = tf.tile(transferred_images, [1, 1, 1, source_channels])
 163   diffs = tf.abs(source_images - transferred_images)
 164   diffs = tf.reduce_max(diffs, reduction_indices=[3], keep_dims=True)
 165   diffs = tf.tile(diffs, [1, 1, 1, max(source_channels, transferred_channels)])
 166 
 167   transition_images = tf.concat([
 168       source_images,
 169       transferred_images,
 170       diffs,
 171   ], 2)
 172 
 173   tf.summary.image(
 174       '%s_difference' % name, transition_images, max_outputs=max_images)

 177 def summaries_color_distributions(images, name):
 178   """Produces a histogram of the color distributions of the images.

 182     name: The name of the images being summarized.

 187 def summarize_images(images, name):
 188   """Produces a visual summary of the given images.
 189 
 190   Args:
 191     images: A `Tensor` of size [batch_size, height, width, 3].
 192     name: The name of the images being summarized.
 193   """

 195   tf.summary.image('%s_Images' % name, grid, max_outputs=1)
