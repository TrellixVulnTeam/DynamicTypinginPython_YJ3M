  93 def run_eval(run_dir, checkpoint_dir, hparams):
  94   """Runs the eval loop.
  95 
  96   Args:
  97     run_dir: The directory where eval specific logs are placed
  98     checkpoint_dir: The directory where the checkpoints are stored
  99     hparams: The hyperparameters struct.
 100 
 101   Raises:
 102     ValueError: if hparams.arch is not recognized.
 103   """
 104   for checkpoint_path in slim.evaluation.checkpoints_iterator(
 105       checkpoint_dir, FLAGS.eval_interval_secs):
 106     with tf.Graph().as_default():
 107       #########################
 108       # Preprocess the inputs #
 109       #########################
 110       target_dataset = dataset_factory.get_dataset(
 111           FLAGS.target_dataset,
 112           split_name=FLAGS.target_split_name,
 113           dataset_dir=FLAGS.dataset_dir)
 114       target_images, target_labels = dataset_factory.provide_batch(
 115           FLAGS.target_dataset, FLAGS.target_split_name, FLAGS.dataset_dir,
 116           FLAGS.num_readers, hparams.batch_size,
 117           FLAGS.num_preprocessing_threads)
 118       num_target_classes = target_dataset.num_classes
 119       target_labels['class'] = tf.argmax(target_labels['classes'], 1)
 120       del target_labels['classes']
 121 
 122       if hparams.arch not in ['dcgan']:
 123         source_dataset = dataset_factory.get_dataset(
 124             FLAGS.source_dataset,
 125             split_name=FLAGS.source_split_name,
 126             dataset_dir=FLAGS.dataset_dir)
 127         num_source_classes = source_dataset.num_classes
 128         source_images, source_labels = dataset_factory.provide_batch(
 129             FLAGS.source_dataset, FLAGS.source_split_name, FLAGS.dataset_dir,
 130             FLAGS.num_readers, hparams.batch_size,
 131             FLAGS.num_preprocessing_threads)
 132         source_labels['class'] = tf.argmax(source_labels['classes'], 1)
 133         del source_labels['classes']
 134         if num_source_classes != num_target_classes:
 135           raise ValueError(
 136               'Input and output datasets must have same number of classes')
 137       else:
 138         source_images = None
 139         source_labels = None
 140 
 141       ####################
 142       # Define the model #
 143       ####################
 144       end_points = pixelda_model.create_model(
 145           hparams,
 146           target_images,
 147           source_images=source_images,
 148           source_labels=source_labels,
 149           is_training=False,
 150           num_classes=num_target_classes)
 151 
 152       #######################
 153       # Metrics & Summaries #
 154       #######################
 155       names_to_values, names_to_updates = create_metrics(end_points,
 156                                                          source_labels,
 157                                                          target_labels, hparams)
 158       pixelda_utils.summarize_model(end_points)
 159       pixelda_utils.summarize_transferred_grid(
 160           end_points['transferred_images'], source_images, name='Transferred')
 161       if 'source_images_recon' in end_points:
 162         pixelda_utils.summarize_transferred_grid(
 163             end_points['source_images_recon'],
 164             source_images,
 165             name='Source Reconstruction')
 166       pixelda_utils.summarize_images(target_images, 'Target')
 167 
 168       for name, value in names_to_values.iteritems():
 169         tf.summary.scalar(name, value)
 170 
 171       # Use the entire split by default
 172       num_examples = target_dataset.num_samples
 173 
 174       num_batches = math.ceil(num_examples / float(hparams.batch_size))
 175       global_step = slim.get_or_create_global_step()
 176 
 177       result = slim.evaluation.evaluate_once(
 178           master=FLAGS.master,
 179           checkpoint_path=checkpoint_path,
 180           logdir=run_dir,
 181           num_evals=num_batches,
 182           eval_op=names_to_updates.values(),
 183           final_op=names_to_values)
 184

 186 def to_degrees(log_quaternion_loss):
 187   """Converts a log quaternion distance to an angle.
 188 
 189   Args:
 190     log_quaternion_loss: The log quaternion distance between two
 191       unit quaternions (or a batch of pairs of quaternions).
 192 
 193   Returns:
 194     The angle in degrees of the implied angle-axis representation.
 195   """
 196   return tf.acos(-(tf.exp(log_quaternion_loss) - 1)) * 2 * 180 / math.pi

 199 def create_metrics(end_points, source_labels, target_labels, hparams):
 200   """Create metrics for the model.
 201 
 202   Args:
 203     end_points: A dictionary of end point name to tensor
 204     source_labels: Labels for source images. batch_size x 1
 205     target_labels: Labels for target images. batch_size x 1
 206     hparams: The hyperparameters struct.
 207 
 208   Returns:
 209     Tuple of (names_to_values, names_to_updates), dictionaries that map a metric
 210     name to its value and update op, respectively
 211 
 212   """
 213   ###########################################
 214   # Evaluate the Domain Prediction Accuracy #
 215   ###########################################
 216   batch_size = hparams.batch_size
 217   names_to_values, names_to_updates = slim.metrics.aggregate_metric_map({
 218       ('eval/Domain_Accuracy-Transferred'):
 219           tf.contrib.metrics.streaming_accuracy(
 220               tf.to_int32(
 221                   tf.round(tf.sigmoid(end_points[
 222                       'transferred_domain_logits']))),
 223               tf.zeros(batch_size, dtype=tf.int32)),
 224       ('eval/Domain_Accuracy-Target'):
 225           tf.contrib.metrics.streaming_accuracy(
 226               tf.to_int32(
 227                   tf.round(tf.sigmoid(end_points['target_domain_logits']))),
 228               tf.ones(batch_size, dtype=tf.int32))
 229   })
 230 
 231   ################################
 232   # Evaluate the task classifier #
 233   ################################
 234   if 'source_task_logits' in end_points:
 235     metric_name = 'eval/Task_Accuracy-Source'
 236     names_to_values[metric_name], names_to_updates[
 237         metric_name] = tf.contrib.metrics.streaming_accuracy(
 238             tf.argmax(end_points['source_task_logits'], 1),
 239             source_labels['class'])
 240 
 241   if 'transferred_task_logits' in end_points:
 242     metric_name = 'eval/Task_Accuracy-Transferred'
 243     names_to_values[metric_name], names_to_updates[
 244         metric_name] = tf.contrib.metrics.streaming_accuracy(
 245             tf.argmax(end_points['transferred_task_logits'], 1),
 246             source_labels['class'])
 247 
 248   if 'target_task_logits' in end_points:
 249     metric_name = 'eval/Task_Accuracy-Target'
 250     names_to_values[metric_name], names_to_updates[
 251         metric_name] = tf.contrib.metrics.streaming_accuracy(
 252             tf.argmax(end_points['target_task_logits'], 1),
 253             target_labels['class'])
 254 
 255   ##########################################################################
 256   # Pose data-specific losses.
 257   ##########################################################################
 258   if 'quaternion' in source_labels.keys():
 259     params = {}
 260     params['use_logging'] = False
 261     params['batch_size'] = batch_size
 262 
 263     angle_loss_source = to_degrees(
 264         pixelda_losses.log_quaternion_loss_batch(end_points[
 265             'source_quaternion'], source_labels['quaternion'], params))
 266     angle_loss_transferred = to_degrees(
 267         pixelda_losses.log_quaternion_loss_batch(end_points[
 268             'transferred_quaternion'], source_labels['quaternion'], params))
 269     angle_loss_target = to_degrees(
 270         pixelda_losses.log_quaternion_loss_batch(end_points[
 271             'target_quaternion'], target_labels['quaternion'], params))
 272 
 273     metric_name = 'eval/Angle_Loss-Source'
 274     names_to_values[metric_name], names_to_updates[
 275         metric_name] = slim.metrics.mean(angle_loss_source)
 276 
 277     metric_name = 'eval/Angle_Loss-Transferred'
 278     names_to_values[metric_name], names_to_updates[
 279         metric_name] = slim.metrics.mean(angle_loss_transferred)
 280 
 281     metric_name = 'eval/Angle_Loss-Target'
 282     names_to_values[metric_name], names_to_updates[
 283         metric_name] = slim.metrics.mean(angle_loss_target)
 284 
 285   return names_to_values, names_to_updates

 294       hparams=hparams)
