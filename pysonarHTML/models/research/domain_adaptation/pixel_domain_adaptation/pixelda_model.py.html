  15 """Contains the Domain Adaptation via Style Transfer (PixelDA) model components.
  16 
  17 A number of details in the implementation make reference to one of the following
  18 works:
  19 
  20 - "Unsupervised Representation Learning with Deep Convolutional
  21     Generative Adversarial Networks""
  22    https://arxiv.org/abs/1511.06434
  23 
  24 This paper makes several architecture recommendations:
  25 1. Use strided convs in discriminator, fractional-strided convs in generator
  26 2. batchnorm everywhere
  27 3. remove fully connected layers for deep models
  28 4. ReLu for all layers in generator, except tanh on output
  29 5. LeakyReLu for everything in discriminator
  30 """

  32 import math

  44 def create_model(hparams,
  45                  target_images,
  46                  source_images=None,
  47                  source_labels=None,
  48                  is_training=False,
  49                  noise=None,
  50                  num_classes=None):
  51   """Create a GAN model.
  52 
  53   Arguments:
  54     hparams: HParam object specifying model params
  55     target_images: A `Tensor` of size [batch_size, height, width, channels]. It
  56       is assumed that the images are [-1, 1] normalized.
  57     source_images: A `Tensor` of size [batch_size, height, width, channels]. It
  58       is assumed that the images are [-1, 1] normalized.
  59     source_labels: A `Tensor` of size [batch_size] of categorical labels between
  60       [0, num_classes]
  61     is_training: whether model is currently training
  62     noise: If None, model generates its own noise. Otherwise use provided.
  63     num_classes: Number of classes for classification
  64 
  65   Returns:
  66     end_points dict with model outputs
  67 
  68   Raises:
  69     ValueError: unknown hparams.arch setting
  70   """
  71   if num_classes is None and hparams.arch in ['resnet', 'simple']:
  72     raise ValueError('Num classes must be provided to create task classifier')
  73 
  74   if target_images.dtype != tf.float32:
  75     raise ValueError('target_images must be tf.float32 and [-1, 1] normalized.')
  76   if source_images is not None and source_images.dtype != tf.float32:
  77     raise ValueError('source_images must be tf.float32 and [-1, 1] normalized.')
  78 
  79   ###########################
  80   # Create latent variables #
  81   ###########################
  82   latent_vars = dict()
  83 
  84   if hparams.noise_channel:
  85     noise_shape = [hparams.batch_size, hparams.noise_dims]
  86     if noise is not None:
  87       assert noise.shape.as_list() == noise_shape
  88       tf.logging.info('Using provided noise')
  89     else:
  90       tf.logging.info('Using random noise')
  91       noise = tf.random_uniform(
  92           shape=noise_shape,
  93           minval=-1,
  94           maxval=1,
  95           dtype=tf.float32,
  96           name='random_noise')
  97     latent_vars['noise'] = noise
  98 
  99   ####################
 100   # Create generator #
 101   ####################
 102 
 103   with slim.arg_scope(
 104       [slim.conv2d, slim.conv2d_transpose, slim.fully_connected],
 105       normalizer_params=batch_norm_params(is_training,
 106                                           hparams.batch_norm_decay),
 107       weights_initializer=tf.random_normal_initializer(
 108           stddev=hparams.normal_init_std),
 109       weights_regularizer=tf.contrib.layers.l2_regularizer(
 110           hparams.weight_decay)):
 111     with slim.arg_scope([slim.conv2d], padding='SAME'):
 112       if hparams.arch == 'dcgan':
 113         end_points = dcgan(
 114             target_images, latent_vars, hparams, scope='generator')
 115       elif hparams.arch == 'resnet':
 116         end_points = resnet_generator(
 117             source_images,
 118             target_images.shape.as_list()[1:4],
 119             hparams=hparams,
 120             latent_vars=latent_vars)
 121       elif hparams.arch == 'residual_interpretation':
 122         end_points = residual_interpretation_generator(
 123             source_images, is_training=is_training, hparams=hparams)
 124       elif hparams.arch == 'simple':
 125         end_points = simple_generator(
 126             source_images,
 127             target_images,
 128             is_training=is_training,
 129             hparams=hparams,
 130             latent_vars=latent_vars)
 131       elif hparams.arch == 'identity':
 132         # Pass through unmodified, besides changing # channels
 133         # Used to calculate baseline numbers
 134         # Also set `generator_steps=0` for baseline
 135         if hparams.generator_steps:
 136           raise ValueError('Must set generator_steps=0 for identity arch. Is %s'
 137                            % hparams.generator_steps)
 138         transferred_images = source_images
 139         source_channels = source_images.shape.as_list()[-1]
 140         target_channels = target_images.shape.as_list()[-1]
 141         if source_channels == 1 and target_channels == 3:
 142           transferred_images = tf.tile(source_images, [1, 1, 1, 3])
 143         if source_channels == 3 and target_channels == 1:
 144           transferred_images = tf.image.rgb_to_grayscale(source_images)
 145         end_points = {'transferred_images': transferred_images}
 146       else:
 147         raise ValueError('Unknown architecture: %s' % hparams.arch)
 148 
 149       #####################
 150       # Domain Classifier #
 151       #####################
 152       if hparams.arch in [
 153           'dcgan', 'resnet', 'residual_interpretation', 'simple', 'identity',
 154       ]:
 155 
 156         # Add a discriminator for these architectures
 157         end_points['transferred_domain_logits'] = predict_domain(
 158             end_points['transferred_images'],
 159             hparams,
 160             is_training=is_training,
 161             reuse=False)
 162         end_points['target_domain_logits'] = predict_domain(
 163             target_images,
 164             hparams,
 165             is_training=is_training,
 166             reuse=True)
 167 
 168       ###################
 169       # Task Classifier #
 170       ###################
 171       if hparams.task_tower != 'none' and hparams.arch in [
 172           'resnet', 'residual_interpretation', 'simple', 'identity',
 173       ]:
 174         with tf.variable_scope('discriminator'):
 175           with tf.variable_scope('task_tower'):
 176             end_points['source_task_logits'], end_points[
 177                 'source_quaternion'] = pixelda_task_towers.add_task_specific_model(
 178                     source_images,
 179                     hparams,
 180                     num_classes=num_classes,
 181                     is_training=is_training,
 182                     reuse_private=False,
 183                     private_scope='source_task_classifier',
 184                     reuse_shared=False)
 185             end_points['transferred_task_logits'], end_points[
 186                 'transferred_quaternion'] = (
 187                     pixelda_task_towers.add_task_specific_model(
 188                         end_points['transferred_images'],
 189                         hparams,
 190                         num_classes=num_classes,
 191                         is_training=is_training,
 192                         reuse_private=False,
 193                         private_scope='transferred_task_classifier',
 194                         reuse_shared=True))
 195             end_points['target_task_logits'], end_points[
 196                 'target_quaternion'] = pixelda_task_towers.add_task_specific_model(
 197                     target_images,
 198                     hparams,
 199                     num_classes=num_classes,
 200                     is_training=is_training,
 201                     reuse_private=True,
 202                     private_scope='transferred_task_classifier',
 203                     reuse_shared=True)
 204   # Remove any endpoints with None values
 205   return dict((k, v) for k, v in end_points.iteritems() if v is not None)

 208 def batch_norm_params(is_training, batch_norm_decay):
 209   return {
 210       'is_training': is_training,
 211       # Decay for the moving averages.
 212       'decay': batch_norm_decay,
 213       # epsilon to prevent 0s in variance.
 214       'epsilon': 0.001,
 215   }

 218 def lrelu(x, leakiness=0.2):
 219   """Relu, with optional leaky support."""
 220   return tf.where(tf.less(x, 0.0), leakiness * x, x, name='leaky_relu')

 223 def upsample(net, num_filters, scale=2, method='resize_conv', scope=None):
 224   """Performs spatial upsampling of the given features.
 225 
 226   Args:
 227     net: A `Tensor` of shape [batch_size, height, width, filters].
 228     num_filters: The number of output filters.
 229     scale: The scale of the upsampling. Must be a positive integer greater or
 230       equal to two.
 231     method: The method by which the features are upsampled. Valid options
 232       include 'resize_conv' and 'conv2d_transpose'.
 233     scope: An optional variable scope.
 234 
 235   Returns:
 236     A new set of features of shape
 237       [batch_size, height*scale, width*scale, num_filters].
 238 
 239   Raises:
 240     ValueError: if `method` is not valid or
 241   """
 242   if scale < 2:
 243     raise ValueError('scale must be greater or equal to two.')
 244 
 245   with tf.variable_scope(scope, 'upsample', [net]):
 246     if method == 'resize_conv':
 247       net = tf.image.resize_nearest_neighbor(
 248           net, [net.shape.as_list()[1] * scale,
 249                 net.shape.as_list()[2] * scale],
 250           align_corners=True,
 251           name='resize')
 252       return slim.conv2d(net, num_filters, stride=1, scope='conv')
 253     elif method == 'conv2d_transpose':
 254       return slim.conv2d_transpose(net, num_filters, scope='deconv')
 255     else:
 256       raise ValueError('Upsample method [%s] was not recognized.' % method)

 259 def project_latent_vars(hparams, proj_shape, latent_vars, combine_method='sum'):
 260   """Generate noise and project to input volume size.
 261 
 262   Args:
 263     hparams: The hyperparameter HParams struct.
 264     proj_shape: Shape to project noise (not including batch size).
 265     latent_vars: dictionary of `'key': Tensor of shape [batch_size, N]`
 266     combine_method: How to combine the projected values.
 267       sum = project to volume then sum
 268       concat = concatenate along last dimension (i.e. channel)
 269 
 270   Returns:
 271     If combine_method=sum, a `Tensor` of size `hparams.projection_shape`
 272     If combine_method=concat and there are N latent vars, a `Tensor` of size
 273       `hparams.projection_shape`, with the last channel multiplied by N
 274 
 275 
 276   Raises:
 277     ValueError: combine_method is not one of sum/concat
 278   """
 279   values = []
 280   for var in latent_vars:
 281     with tf.variable_scope(var):
 282       # Project & reshape noise to a HxWxC input
 283       projected = slim.fully_connected(
 284           latent_vars[var],
 285           np.prod(proj_shape),
 286           activation_fn=tf.nn.relu,
 287           normalizer_fn=slim.batch_norm)
 288       values.append(tf.reshape(projected, [hparams.batch_size] + proj_shape))
 289 
 290   if combine_method == 'sum':
 291     result = values[0]
 292     for value in values[1:]:
 293       result += value
 294   elif combine_method == 'concat':
 295     # Concatenate along last axis
 296     result = tf.concat(values, len(proj_shape))
 297   else:
 298     raise ValueError('Unknown combine_method %s' % combine_method)
 299 
 300   tf.logging.info('Latent variables projected to size %s volume', result.shape)
 301 
 302   return result

 305 def resnet_block(net, hparams):
 306   """Create a resnet block."""
 307   net_in = net
 308   net = slim.conv2d(
 309       net,
 310       hparams.resnet_filters,
 311       stride=1,
 312       normalizer_fn=slim.batch_norm,
 313       activation_fn=tf.nn.relu)
 314   net = slim.conv2d(
 315       net,
 316       hparams.resnet_filters,
 317       stride=1,
 318       normalizer_fn=slim.batch_norm,
 319       activation_fn=None)
 320   if hparams.resnet_residuals:
 321     net += net_in
 322   return net

 325 def resnet_stack(images, output_shape, hparams, scope=None):
 326   """Create a resnet style transfer block.
 327 
 328   Args:
 329     images: [batch-size, height, width, channels] image tensor to feed as input
 330     output_shape: output image shape in form [height, width, channels]
 331     hparams: hparams objects
 332     scope: Variable scope
 333 
 334   Returns:
 335     Images after processing with resnet blocks.
 336   """
 337   end_points = {}
 338   if hparams.noise_channel:
 339     # separate the noise for visualization
 340     end_points['noise'] = images[:, :, :, -1]
 341   assert images.shape.as_list()[1:3] == output_shape[0:2]
 342 
 343   with tf.variable_scope(scope, 'resnet_style_transfer', [images]):
 344     with slim.arg_scope(
 345         [slim.conv2d],
 346         normalizer_fn=slim.batch_norm,
 347         kernel_size=[hparams.generator_kernel_size] * 2,
 348         stride=1):
 349       net = slim.conv2d(
 350           images,
 351           hparams.resnet_filters,
 352           normalizer_fn=None,
 353           activation_fn=tf.nn.relu)
 354       for block in range(hparams.resnet_blocks):
 355         net = resnet_block(net, hparams)
 356         end_points['resnet_block_{}'.format(block)] = net
 357 
 358       net = slim.conv2d(
 359           net,
 360           output_shape[-1],
 361           kernel_size=[1, 1],
 362           normalizer_fn=None,
 363           activation_fn=tf.nn.tanh,
 364           scope='conv_out')
 365       end_points['transferred_images'] = net
 366     return net, end_points

 369 def predict_domain(images,
 370                    hparams,
 371                    is_training=False,
 372                    reuse=False,
 373                    scope='discriminator'):
 374   """Creates a discriminator for a GAN.
 375 
 376   Args:
 377     images: A `Tensor` of size [batch_size, height, width, channels]. It is
 378       assumed that the images are centered between -1 and 1.
 379     hparams: hparam object with params for discriminator
 380     is_training: Specifies whether or not we're training or testing.
 381     reuse: Whether to reuse variable scope
 382     scope: An optional variable_scope.
 383 
 384   Returns:
 385     [batch size, 1] - logit output of discriminator.
 386   """
 387   with tf.variable_scope(scope, 'discriminator', [images], reuse=reuse):
 388     lrelu_partial = functools.partial(lrelu, leakiness=hparams.lrelu_leakiness)
 389     with slim.arg_scope(
 390         [slim.conv2d],
 391         kernel_size=[hparams.discriminator_kernel_size] * 2,
 392         activation_fn=lrelu_partial,
 393         stride=2,
 394         normalizer_fn=slim.batch_norm):
 395 
 396       def add_noise(hidden, scope_num=None):
 397         if scope_num:
 398           hidden = slim.dropout(
 399               hidden,
 400               hparams.discriminator_dropout_keep_prob,
 401               is_training=is_training,
 402               scope='dropout_%s' % scope_num)
 403         if hparams.discriminator_noise_stddev == 0:
 404           return hidden
 405         return hidden + tf.random_normal(
 406             hidden.shape.as_list(),
 407             mean=0.0,
 408             stddev=hparams.discriminator_noise_stddev)
 409 
 410       # As per the recommendation of the DCGAN paper, we don't use batch norm
 411       # on the discriminator input (https://arxiv.org/pdf/1511.06434v2.pdf).
 412       if hparams.discriminator_image_noise:
 413         images = add_noise(images)
 414       net = slim.conv2d(
 415           images,
 416           hparams.num_discriminator_filters,
 417           normalizer_fn=None,
 418           stride=hparams.discriminator_first_stride,
 419           scope='conv1_stride%s' % hparams.discriminator_first_stride)
 420       net = add_noise(net, 1)
 421 
 422       block_id = 2
 423       # Repeatedly stack
 424       # discriminator_conv_block_size-1 conv layers with stride 1
 425       #  followed by a stride 2 layer
 426       # Add (optional) noise at every point
 427       while net.shape.as_list()[1] > hparams.projection_shape_size:
 428         num_filters = int(hparams.num_discriminator_filters *
 429                           (hparams.discriminator_filter_factor**(block_id - 1)))
 430         for conv_id in range(1, hparams.discriminator_conv_block_size):
 431           net = slim.conv2d(
 432               net,
 433               num_filters,
 434               stride=1,
 435               scope='conv_%s_%s' % (block_id, conv_id))
 436         if hparams.discriminator_do_pooling:
 437           net = slim.conv2d(
 438               net, num_filters, scope='conv_%s_prepool' % block_id)
 439           net = slim.avg_pool2d(
 440               net, kernel_size=[2, 2], stride=2, scope='pool_%s' % block_id)
 441         else:
 442           net = slim.conv2d(
 443               net, num_filters, scope='conv_%s_stride2' % block_id)
 444         net = add_noise(net, block_id)
 445         block_id += 1
 446       net = slim.flatten(net)
 447       net = slim.fully_connected(
 448           net,
 449           1,
 450           # Models with BN here generally produce noise
 451           normalizer_fn=None,
 452           activation_fn=None,
 453           scope='fc_logit_out')  # Returns logits!
 454   return net

 457 def dcgan_generator(images, output_shape, hparams, scope=None):
 458   """Transforms the visual style of the input images.
 459 
 460   Args:
 461     images: A `Tensor` of shape [batch_size, height, width, channels].
 462     output_shape: A list or tuple of 3 elements: the output height, width and
 463       number of channels.
 464     hparams: hparams object with generator parameters
 465     scope: Scope to place generator inside
 466 
 467   Returns:
 468     A `Tensor` of shape [batch_size, height, width, output_channels] which
 469     represents the result of style transfer.
 470 
 471   Raises:
 472     ValueError: If `output_shape` is not a list or tuple or if it doesn't have
 473     three elements or if `output_shape` or `images` arent square.
 474   """
 475   if not isinstance(output_shape, (tuple, list)):
 476     raise ValueError('output_shape must be a tuple or list.')
 477   elif len(output_shape) != 3:
 478     raise ValueError('output_shape must have three elements.')
 479 
 480   if output_shape[0] != output_shape[1]:
 481     raise ValueError('output_shape must be square')
 482   if images.shape.as_list()[1] != images.shape.as_list()[2]:
 483     raise ValueError('images height and width must match.')
 484 
 485   outdim = output_shape[0]
 486   indim = images.shape.as_list()[1]
 487   num_iterations = int(math.ceil(math.log(float(outdim) / float(indim), 2.0)))
 488 
 489   with slim.arg_scope(
 490       [slim.conv2d, slim.conv2d_transpose],
 491       kernel_size=[hparams.generator_kernel_size] * 2,
 492       stride=2):
 493     with tf.variable_scope(scope or 'generator'):
 494 
 495       net = images
 496 
 497       # Repeatedly halve # filters until = hparams.decode_filters in last layer
 498       for i in range(num_iterations):
 499         num_filters = hparams.num_decoder_filters * 2**(num_iterations - i - 1)
 500         net = slim.conv2d_transpose(net, num_filters, scope='deconv_%s' % i)
 501 
 502       # Crop down to desired size (e.g. 32x32 -> 28x28)
 503       dif = net.shape.as_list()[1] - outdim
 504       low = dif / 2
 505       high = net.shape.as_list()[1] - low
 506       net = net[:, low:high, low:high, :]
 507 
 508       # No batch norm on generator output
 509       net = slim.conv2d(
 510           net,
 511           output_shape[2],
 512           kernel_size=[1, 1],
 513           stride=1,
 514           normalizer_fn=None,
 515           activation_fn=tf.tanh,
 516           scope='conv_out')
 517   return net

 520 def dcgan(target_images, latent_vars, hparams, scope='dcgan'):
 521   """Creates the PixelDA model.
 522 
 523   Args:
 524     target_images: A `Tensor` of shape [batch_size, height, width, 3]
 525       sampled from the image domain to which we want to transfer.
 526     latent_vars: dictionary of 'key': Tensor of shape [batch_size, N]
 527     hparams: The hyperparameter map.
 528     scope: Surround generator component with this scope
 529 
 530   Returns:
 531     A dictionary of model outputs.
 532   """
 533   proj_shape = [
 534       hparams.projection_shape_size, hparams.projection_shape_size,
 535       hparams.projection_shape_channels
 536   ]
 537   source_volume = project_latent_vars(
 538       hparams, proj_shape, latent_vars, combine_method='concat')
 539 
 540   ###################################################
 541   # Transfer the source images to the target style. #
 542   ###################################################
 543   with tf.variable_scope(scope, 'generator', [target_images]):
 544     transferred_images = dcgan_generator(
 545         source_volume,
 546         output_shape=target_images.shape.as_list()[1:4],
 547         hparams=hparams)
 548     assert transferred_images.shape.as_list() == target_images.shape.as_list()
 549 
 550   return {'transferred_images': transferred_images}

 553 def resnet_generator(images, output_shape, hparams, latent_vars=None):
 554   """Creates a ResNet-based generator.
 555 
 556   Args:
 557     images: A `Tensor` of shape [batch_size, height, width, num_channels]
 558       sampled from the image domain from which we want to transfer
 559     output_shape: A length-3 array indicating the height, width and channels of
 560       the output.
 561     hparams: The hyperparameter map.
 562     latent_vars: dictionary of 'key': Tensor of shape [batch_size, N]
 563 
 564   Returns:
 565     A dictionary of model outputs.
 566   """
 567   with tf.variable_scope('generator'):
 568     if latent_vars:
 569       noise_channel = project_latent_vars(
 570           hparams,
 571           proj_shape=images.shape.as_list()[1:3] + [1],
 572           latent_vars=latent_vars,
 573           combine_method='concat')
 574       images = tf.concat([images, noise_channel], 3)
 575 
 576     transferred_images, end_points = resnet_stack(
 577         images,
 578         output_shape=output_shape,
 579         hparams=hparams,
 580         scope='resnet_stack')
 581     end_points['transferred_images'] = transferred_images
 582 
 583   return end_points

 586 def residual_interpretation_block(images, hparams, scope):
 587   """Learns a residual image which is added to the incoming image.
 588 
 589   Args:
 590     images: A `Tensor` of size [batch_size, height, width, 3]
 591     hparams: The hyperparameters struct.
 592     scope: The name of the variable op scope.
 593 
 594   Returns:
 595     The updated images.
 596   """
 597   with tf.variable_scope(scope):
 598     with slim.arg_scope(
 599         [slim.conv2d],
 600         normalizer_fn=None,
 601         kernel_size=[hparams.generator_kernel_size] * 2):
 602 
 603       net = images
 604       for _ in range(hparams.res_int_convs):
 605         net = slim.conv2d(
 606             net, hparams.res_int_filters, activation_fn=tf.nn.relu)
 607       net = slim.conv2d(net, 3, activation_fn=tf.nn.tanh)
 608 
 609     # Add the residual
 610     images += net
 611 
 612     # Clip the output
 613     images = tf.maximum(images, -1.0)
 614     images = tf.minimum(images, 1.0)
 615     return images

 618 def residual_interpretation_generator(images,
 619                                       is_training,
 620                                       hparams,
 621                                       latent_vars=None):
 622   """Creates a generator producing purely residual transformations.
 623 
 624   A residual generator differs from the resnet generator in that each 'block' of
 625   the residual generator produces a residual image. Consequently, the 'progress'
 626   of the model generation process can be directly observed at inference time,
 627   making it easier to diagnose and understand.
 628 
 629   Args:
 630     images: A `Tensor` of shape [batch_size, height, width, num_channels]
 631       sampled from the image domain from which we want to transfer. It is
 632       assumed that the images are centered between -1 and 1.
 633     is_training: whether or not the model is training.
 634     hparams: The hyperparameter map.
 635     latent_vars: dictionary of 'key': Tensor of shape [batch_size, N]
 636 
 637   Returns:
 638     A dictionary of model outputs.
 639   """
 640   end_points = {}
 641 
 642   with tf.variable_scope('generator'):
 643     if latent_vars:
 644       projected_latent = project_latent_vars(
 645           hparams,
 646           proj_shape=images.shape.as_list()[1:3] + [images.shape.as_list()[-1]],
 647           latent_vars=latent_vars,
 648           combine_method='sum')
 649       images += projected_latent
 650     with tf.variable_scope(None, 'residual_style_transfer', [images]):
 651       for i in range(hparams.res_int_blocks):
 652         images = residual_interpretation_block(images, hparams,
 653                                                'residual_%d' % i)
 654         end_points['transferred_images_%d' % i] = images
 655 
 656       end_points['transferred_images'] = images
 657 
 658   return end_points

 661 def simple_generator(source_images, target_images, is_training, hparams,
 662                      latent_vars):
 663   """Simple generator architecture (stack of convs) for trying small models."""
 664   end_points = {}
 665   with tf.variable_scope('generator'):
 666     feed_source_images = source_images
 667 
 668     if latent_vars:
 669       projected_latent = project_latent_vars(
 670           hparams,
 671           proj_shape=source_images.shape.as_list()[1:3] + [1],
 672           latent_vars=latent_vars,
 673           combine_method='concat')
 674       feed_source_images = tf.concat([source_images, projected_latent], 3)
 675 
 676     end_points = {}
 677 
 678     ###################################################
 679     # Transfer the source images to the target style. #
 680     ###################################################
 681     with slim.arg_scope(
 682         [slim.conv2d],
 683         normalizer_fn=slim.batch_norm,
 684         stride=1,
 685         kernel_size=[hparams.generator_kernel_size] * 2):
 686       net = feed_source_images
 687 
 688       # N convolutions
 689       for i in range(1, hparams.simple_num_conv_layers):
 690         normalizer_fn = None
 691         if i != 0:
 692           normalizer_fn = slim.batch_norm
 693         net = slim.conv2d(
 694             net,
 695             hparams.simple_conv_filters,
 696             normalizer_fn=normalizer_fn,
 697             activation_fn=tf.nn.relu)
 698 
 699       # Project back to right # image channels
 700       net = slim.conv2d(
 701           net,
 702           target_images.shape.as_list()[-1],
 703           kernel_size=[1, 1],
 704           stride=1,
 705           normalizer_fn=None,
 706           activation_fn=tf.tanh,
 707           scope='conv_out')
 708 
 709     transferred_images = net
 710     assert transferred_images.shape.as_list() == target_images.shape.as_list()
 711     end_points['transferred_images'] = transferred_images
 712 
 713   return end_points
