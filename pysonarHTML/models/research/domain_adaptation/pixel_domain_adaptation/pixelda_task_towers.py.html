  21 def add_task_specific_model(images,
  22                             hparams,
  23                             num_classes=10,
  24                             is_training=False,
  25                             reuse_private=False,
  26                             private_scope=None,
  27                             reuse_shared=False,
  28                             shared_scope=None):
  29   """Create a classifier for the given images.
  30 
  31   The classifier is composed of a few 'private' layers followed by a few
  32   'shared' layers. This lets us account for different image 'style', while
  33   sharing the last few layers as 'content' layers.
  34 
  35   Args:
  36     images: A `Tensor` of size [batch_size, height, width, 3].
  37     hparams: model hparams
  38     num_classes: The number of output classes.
  39     is_training: whether model is training
  40     reuse_private: Whether or not to reuse the private weights, which are the
  41       first few layers in the classifier
  42     private_scope: The name of the variable_scope for the private (unshared)
  43       components of the classifier.
  44     reuse_shared: Whether or not to reuse the shared weights, which are the last
  45       few layers in the classifier
  46     shared_scope: The name of the variable_scope for the shared components of
  47       the classifier.
  48 
  49   Returns:
  50     The logits, a `Tensor` of shape [batch_size, num_classes].
  51 
  52   Raises:
  53     ValueError: If hparams.task_classifier is an unknown value
  54   """
  55 
  56   model = hparams.task_tower
  57   # Make sure the classifier name shows up in graph
  58   shared_scope = shared_scope or (model + '_shared')
  59   kwargs = {
  60       'num_classes': num_classes,
  61       'is_training': is_training,
  62       'reuse_private': reuse_private,
  63       'reuse_shared': reuse_shared,
  64   }
  65 
  66   if private_scope:
  67     kwargs['private_scope'] = private_scope
  68   if shared_scope:
  69     kwargs['shared_scope'] = shared_scope
  70 
  71   quaternion_pred = None
  72   with slim.arg_scope(
  73       [slim.conv2d, slim.fully_connected],
  74       activation_fn=tf.nn.relu,
  75       weights_regularizer=tf.contrib.layers.l2_regularizer(
  76           hparams.weight_decay_task_classifier)):
  77     with slim.arg_scope([slim.conv2d], padding='SAME'):
  78       if model == 'doubling_pose_estimator':
  79         logits, quaternion_pred = doubling_cnn_class_and_quaternion(
  80             images, num_private_layers=hparams.num_private_layers, **kwargs)
  81       elif model == 'mnist':
  82         logits, _ = mnist_classifier(images, **kwargs)
  83       elif model == 'svhn':
  84         logits, _ = svhn_classifier(images, **kwargs)
  85       elif model == 'gtsrb':
  86         logits, _ = gtsrb_classifier(images, **kwargs)
  87       elif model == 'pose_mini':
  88         logits, quaternion_pred = pose_mini_tower(images, **kwargs)
  89       else:
  90         raise ValueError('Unknown task classifier %s' % model)
  91 
  92   return logits, quaternion_pred

 100 def mnist_classifier(images,
 101                      is_training=False,
 102                      num_classes=10,
 103                      reuse_private=False,
 104                      private_scope='mnist',
 105                      reuse_shared=False,
 106                      shared_scope='task_model'):
 107   """Creates the convolutional MNIST model from the gradient reversal paper.
 108 
 109   Note that since the output is a set of 'logits', the values fall in the
 110   interval of (-infinity, infinity). Consequently, to convert the outputs to a
 111   probability distribution over the characters, one will need to convert them
 112   using the softmax function:
 113         logits, endpoints = conv_mnist(images, is_training=False)
 114         predictions = tf.nn.softmax(logits)
 115 
 116   Args:
 117     images: the MNIST digits, a tensor of size [batch_size, 28, 28, 1].
 118     is_training: specifies whether or not we're currently training the model.
 119       This variable will determine the behaviour of the dropout layer.
 120     num_classes: the number of output classes to use.
 121 
 122   Returns:
 123     the output logits, a tensor of size [batch_size, num_classes].
 124     a dictionary with key/values the layer names and tensors.
 125   """
 126 
 127   net = {}
 128 
 129   with tf.variable_scope(private_scope, reuse=reuse_private):
 130     net['conv1'] = slim.conv2d(images, 32, [5, 5], scope='conv1')
 131     net['pool1'] = slim.max_pool2d(net['conv1'], [2, 2], 2, scope='pool1')
 132 
 133   with tf.variable_scope(shared_scope, reuse=reuse_shared):
 134     net['conv2'] = slim.conv2d(net['pool1'], 48, [5, 5], scope='conv2')
 135     net['pool2'] = slim.max_pool2d(net['conv2'], [2, 2], 2, scope='pool2')
 136     net['fc3'] = slim.fully_connected(
 137         slim.flatten(net['pool2']), 100, scope='fc3')
 138     net['fc4'] = slim.fully_connected(
 139         slim.flatten(net['fc3']), 100, scope='fc4')
 140     logits = slim.fully_connected(
 141         net['fc4'], num_classes, activation_fn=None, scope='fc5')
 142   return logits, net

 145 def svhn_classifier(images,
 146                     is_training=False,
 147                     num_classes=10,
 148                     reuse_private=False,
 149                     private_scope=None,
 150                     reuse_shared=False,
 151                     shared_scope='task_model'):
 152   """Creates the convolutional SVHN model from the gradient reversal paper.
 153 
 154   Note that since the output is a set of 'logits', the values fall in the
 155   interval of (-infinity, infinity). Consequently, to convert the outputs to a
 156   probability distribution over the characters, one will need to convert them
 157   using the softmax function:
 158         logits = mnist.Mnist(images, is_training=False)
 159         predictions = tf.nn.softmax(logits)
 160 
 161   Args:
 162     images: the SVHN digits, a tensor of size [batch_size, 40, 40, 3].
 163     is_training: specifies whether or not we're currently training the model.
 164       This variable will determine the behaviour of the dropout layer.
 165     num_classes: the number of output classes to use.
 166 
 167   Returns:
 168     the output logits, a tensor of size [batch_size, num_classes].
 169     a dictionary with key/values the layer names and tensors.
 170   """
 171 
 172   net = {}
 173 
 174   with tf.variable_scope(private_scope, reuse=reuse_private):
 175     net['conv1'] = slim.conv2d(images, 64, [5, 5], scope='conv1')
 176     net['pool1'] = slim.max_pool2d(net['conv1'], [3, 3], 2, scope='pool1')
 177 
 178   with tf.variable_scope(shared_scope, reuse=reuse_shared):
 179     net['conv2'] = slim.conv2d(net['pool1'], 64, [5, 5], scope='conv2')
 180     net['pool2'] = slim.max_pool2d(net['conv2'], [3, 3], 2, scope='pool2')
 181     net['conv3'] = slim.conv2d(net['pool2'], 128, [5, 5], scope='conv3')
 182 
 183     net['fc3'] = slim.fully_connected(
 184         slim.flatten(net['conv3']), 3072, scope='fc3')
 185     net['fc4'] = slim.fully_connected(
 186         slim.flatten(net['fc3']), 2048, scope='fc4')
 187 
 188     logits = slim.fully_connected(
 189         net['fc4'], num_classes, activation_fn=None, scope='fc5')
 190 
 191   return logits, net

 194 def gtsrb_classifier(images,
 195                      is_training=False,
 196                      num_classes=43,
 197                      reuse_private=False,
 198                      private_scope='gtsrb',
 199                      reuse_shared=False,
 200                      shared_scope='task_model'):
 201   """Creates the convolutional GTSRB model from the gradient reversal paper.
 202 
 203   Note that since the output is a set of 'logits', the values fall in the
 204   interval of (-infinity, infinity). Consequently, to convert the outputs to a
 205   probability distribution over the characters, one will need to convert them
 206   using the softmax function:
 207         logits = mnist.Mnist(images, is_training=False)
 208         predictions = tf.nn.softmax(logits)
 209 
 210   Args:
 211     images: the SVHN digits, a tensor of size [batch_size, 40, 40, 3].
 212     is_training: specifies whether or not we're currently training the model.
 213       This variable will determine the behaviour of the dropout layer.
 214     num_classes: the number of output classes to use.
 215     reuse_private: Whether or not to reuse the private components of the model.
 216     private_scope: The name of the private scope.
 217     reuse_shared: Whether or not to reuse the shared components of the model.
 218     shared_scope: The name of the shared scope.
 219 
 220   Returns:
 221     the output logits, a tensor of size [batch_size, num_classes].
 222     a dictionary with key/values the layer names and tensors.
 223   """
 224 
 225   net = {}
 226 
 227   with tf.variable_scope(private_scope, reuse=reuse_private):
 228     net['conv1'] = slim.conv2d(images, 96, [5, 5], scope='conv1')
 229     net['pool1'] = slim.max_pool2d(net['conv1'], [2, 2], 2, scope='pool1')
 230   with tf.variable_scope(shared_scope, reuse=reuse_shared):
 231     net['conv2'] = slim.conv2d(net['pool1'], 144, [3, 3], scope='conv2')
 232     net['pool2'] = slim.max_pool2d(net['conv2'], [2, 2], 2, scope='pool2')
 233     net['conv3'] = slim.conv2d(net['pool2'], 256, [5, 5], scope='conv3')
 234     net['pool3'] = slim.max_pool2d(net['conv3'], [2, 2], 2, scope='pool3')
 235 
 236     net['fc3'] = slim.fully_connected(
 237         slim.flatten(net['pool3']), 512, scope='fc3')
 238     logits = slim.fully_connected(
 239         net['fc3'], num_classes, activation_fn=None, scope='fc4')
 240 
 241     return logits, net

 249 def pose_mini_tower(images,
 250                     num_classes=11,
 251                     is_training=False,
 252                     reuse_private=False,
 253                     private_scope='pose_mini',
 254                     reuse_shared=False,
 255                     shared_scope='task_model'):
 256   """Task tower for the pose_mini dataset."""
 257 
 258   with tf.variable_scope(private_scope, reuse=reuse_private):
 259     net = slim.conv2d(images, 32, [5, 5], scope='conv1')
 260     net = slim.max_pool2d(net, [2, 2], stride=2, scope='pool1')
 261   with tf.variable_scope(shared_scope, reuse=reuse_shared):
 262     net = slim.conv2d(net, 64, [5, 5], scope='conv2')
 263     net = slim.max_pool2d(net, [2, 2], stride=2, scope='pool2')
 264     net = slim.flatten(net)
 265 
 266     net = slim.fully_connected(net, 128, scope='fc3')
 267     net = slim.dropout(net, 0.5, is_training=is_training, scope='dropout')
 268     with tf.variable_scope('quaternion_prediction'):
 269       quaternion_pred = slim.fully_connected(
 270           net, 4, activation_fn=tf.tanh, scope='fc_q')
 271       quaternion_pred = tf.nn.l2_normalize(quaternion_pred, 1)
 272 
 273     logits = slim.fully_connected(
 274         net, num_classes, activation_fn=None, scope='fc4')
 275 
 276     return logits, quaternion_pred

 279 def doubling_cnn_class_and_quaternion(images,
 280                                       num_private_layers=1,
 281                                       num_classes=10,
 282                                       is_training=False,
 283                                       reuse_private=False,
 284                                       private_scope='doubling_cnn',
 285                                       reuse_shared=False,
 286                                       shared_scope='task_model'):
 287   """Alternate conv, pool while doubling filter count."""
 288   net = images
 289   depth = 32
 290   layer_id = 1
 291 
 292   with tf.variable_scope(private_scope, reuse=reuse_private):
 293     while num_private_layers > 0 and net.shape.as_list()[1] > 5:
 294       net = slim.conv2d(net, depth, [3, 3], scope='conv%s' % layer_id)
 295       net = slim.max_pool2d(net, [2, 2], stride=2, scope='pool%s' % layer_id)
 296       depth *= 2
 297       layer_id += 1
 298       num_private_layers -= 1
 299 
 300   with tf.variable_scope(shared_scope, reuse=reuse_shared):
 301     while net.shape.as_list()[1] > 5:
 302       net = slim.conv2d(net, depth, [3, 3], scope='conv%s' % layer_id)
 303       net = slim.max_pool2d(net, [2, 2], stride=2, scope='pool%s' % layer_id)
 304       depth *= 2
 305       layer_id += 1
 306 
 307     net = slim.flatten(net)
 308     net = slim.fully_connected(net, 100, scope='fc1')
 309     net = slim.dropout(net, 0.5, is_training=is_training, scope='dropout')
 310     quaternion_pred = slim.fully_connected(
 311         net, 4, activation_fn=tf.tanh, scope='fc_q')
 312     quaternion_pred = tf.nn.l2_normalize(quaternion_pred, 1)
 313 
 314     logits = slim.fully_connected(
 315         net, num_classes, activation_fn=None, scope='fc_logits')
 316 
 317     return logits, quaternion_pred
