  92 def _get_vars_and_update_ops(hparams, scope):
  93   """Returns the variables and update ops for a particular variable scope.
  94 
  95   Args:
  96     hparams: The hyperparameters struct.
  97     scope: The variable scope.
  98 
  99   Returns:
 100     A tuple consisting of trainable variables and update ops.
 101   """
 102   is_trainable = lambda x: x in tf.trainable_variables()
 103   var_list = filter(is_trainable, slim.get_model_variables(scope))
 104   global_step = slim.get_or_create_global_step()
 105 
 106   update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS, scope)
 107 
 108   tf.logging.info('All variables for scope: %s',
 109                   slim.get_model_variables(scope))
 110   tf.logging.info('Trainable variables for scope: %s', var_list)
 111 
 112   return var_list, update_ops

 115 def _train(discriminator_train_op,
 116            generator_train_op,
 117            logdir,
 118            master='',
 119            is_chief=True,
 120            scaffold=None,
 121            hooks=None,
 122            chief_only_hooks=None,
 123            save_checkpoint_secs=600,
 124            save_summaries_steps=100,
 125            hparams=None):
 126   """Runs the training loop.
 127 
 128   Args:
 129     discriminator_train_op: A `Tensor` that, when executed, will apply the
 130       gradients and return the loss value for the discriminator.
 131     generator_train_op: A `Tensor` that, when executed, will apply the
 132       gradients and return the loss value for the generator.
 133     logdir: The directory where the graph and checkpoints are saved.
 134     master: The URL of the master.
 135     is_chief: Specifies whether or not the training is being run by the primary
 136       replica during replica training.
 137     scaffold: An tf.train.Scaffold instance.
 138     hooks: List of `tf.train.SessionRunHook` callbacks which are run inside the
 139       training loop.
 140     chief_only_hooks: List of `tf.train.SessionRunHook` instances which are run
 141       inside the training loop for the chief trainer only.
 142     save_checkpoint_secs: The frequency, in seconds, that a checkpoint is saved
 143       using a default checkpoint saver. If `save_checkpoint_secs` is set to
 144       `None`, then the default checkpoint saver isn't used.
 145     save_summaries_steps: The frequency, in number of global steps, that the
 146       summaries are written to disk using a default summary saver. If
 147       `save_summaries_steps` is set to `None`, then the default summary saver
 148       isn't used.
 149     hparams: The hparams struct.
 150 
 151   Returns:
 152     the value of the loss function after training.
 153 
 154   Raises:
 155     ValueError: if `logdir` is `None` and either `save_checkpoint_secs` or
 156     `save_summaries_steps` are `None.
 157   """
 158   global_step = slim.get_or_create_global_step()
 159 
 160   scaffold = scaffold or tf.train.Scaffold()
 161 
 162   hooks = hooks or []
 163 
 164   if is_chief:
 165     session_creator = tf.train.ChiefSessionCreator(
 166         scaffold=scaffold, checkpoint_dir=logdir, master=master)
 167 
 168     if chief_only_hooks:
 169       hooks.extend(chief_only_hooks)
 170     hooks.append(tf.train.StepCounterHook(output_dir=logdir))
 171 
 172     if save_summaries_steps:
 173       if logdir is None:
 174         raise ValueError(
 175             'logdir cannot be None when save_summaries_steps is None')
 176       hooks.append(
 177           tf.train.SummarySaverHook(
 178               scaffold=scaffold,
 179               save_steps=save_summaries_steps,
 180               output_dir=logdir))
 181 
 182     if save_checkpoint_secs:
 183       if logdir is None:
 184         raise ValueError(
 185             'logdir cannot be None when save_checkpoint_secs is None')
 186       hooks.append(
 187           tf.train.CheckpointSaverHook(
 188               logdir, save_secs=save_checkpoint_secs, scaffold=scaffold))
 189   else:
 190     session_creator = tf.train.WorkerSessionCreator(
 191         scaffold=scaffold, master=master)
 192 
 193   with tf.train.MonitoredSession(
 194       session_creator=session_creator, hooks=hooks) as session:
 195     loss = None
 196     while not session.should_stop():
 197       # Run the domain classifier op X times.
 198       for _ in range(hparams.discriminator_steps):
 199         if session.should_stop():
 200           return loss
 201         loss, np_global_step = session.run(
 202             [discriminator_train_op, global_step])
 203         if np_global_step % FLAGS.print_loss_steps == 0:
 204           tf.logging.info('Step %d: Discriminator Loss = %.2f', np_global_step,
 205                           loss)
 206 
 207       # Run the generator op X times.
 208       for _ in range(hparams.generator_steps):
 209         if session.should_stop():
 210           return loss
 211         loss, np_global_step = session.run([generator_train_op, global_step])
 212         if np_global_step % FLAGS.print_loss_steps == 0:
 213           tf.logging.info('Step %d: Generator Loss = %.2f', np_global_step,
 214                           loss)
 215   return loss

 218 def run_training(run_dir, checkpoint_dir, hparams):
 219   """Runs the training loop.
 220 
 221   Args:
 222     run_dir: The directory where training specific logs are placed
 223     checkpoint_dir: The directory where the checkpoints and log files are
 224       stored.
 225     hparams: The hyperparameters struct.
 226 
 227   Raises:
 228     ValueError: if hparams.arch is not recognized.
 229   """
 230   for path in [run_dir, checkpoint_dir]:
 231     if not tf.gfile.Exists(path):
 232       tf.gfile.MakeDirs(path)
 233 
 234   # Serialize hparams to log dir
 235   hparams_filename = os.path.join(checkpoint_dir, 'hparams.json')
 236   with tf.gfile.FastGFile(hparams_filename, 'w') as f:
 237     f.write(hparams.to_json())
 238 
 239   with tf.Graph().as_default():
 240     with tf.device(tf.train.replica_device_setter(FLAGS.ps_tasks)):
 241       global_step = slim.get_or_create_global_step()
 242 
 243       #########################
 244       # Preprocess the inputs #
 245       #########################
 246       target_dataset = dataset_factory.get_dataset(
 247           FLAGS.target_dataset,
 248           split_name='train',
 249           dataset_dir=FLAGS.dataset_dir)
 250       target_images, _ = dataset_factory.provide_batch(
 251           FLAGS.target_dataset, 'train', FLAGS.dataset_dir, FLAGS.num_readers,
 252           hparams.batch_size, FLAGS.num_preprocessing_threads)
 253       num_target_classes = target_dataset.num_classes
 254 
 255       if hparams.arch not in ['dcgan']:
 256         source_dataset = dataset_factory.get_dataset(
 257             FLAGS.source_dataset,
 258             split_name='train',
 259             dataset_dir=FLAGS.dataset_dir)
 260         num_source_classes = source_dataset.num_classes
 261         source_images, source_labels = dataset_factory.provide_batch(
 262             FLAGS.source_dataset, 'train', FLAGS.dataset_dir, FLAGS.num_readers,
 263             hparams.batch_size, FLAGS.num_preprocessing_threads)
 264         # Data provider provides 1 hot labels, but we expect categorical.
 265         source_labels['class'] = tf.argmax(source_labels['classes'], 1)
 266         del source_labels['classes']
 267         if num_source_classes != num_target_classes:
 268           raise ValueError(
 269               'Source and Target datasets must have same number of classes. '
 270               'Are %d and %d' % (num_source_classes, num_target_classes))
 271       else:
 272         source_images = None
 273         source_labels = None
 274 
 275       ####################
 276       # Define the model #
 277       ####################
 278       end_points = pixelda_model.create_model(
 279           hparams,
 280           target_images,
 281           source_images=source_images,
 282           source_labels=source_labels,
 283           is_training=True,
 284           num_classes=num_target_classes)
 285 
 286       #################################
 287       # Get the variables to optimize #
 288       #################################
 289       generator_vars, generator_update_ops = _get_vars_and_update_ops(
 290           hparams, 'generator')
 291       discriminator_vars, discriminator_update_ops = _get_vars_and_update_ops(
 292           hparams, 'discriminator')
 293 
 294       ########################
 295       # Configure the losses #
 296       ########################
 297       generator_loss = pixelda_losses.g_step_loss(
 298           source_images,
 299           source_labels,
 300           end_points,
 301           hparams,
 302           num_classes=num_target_classes)
 303       discriminator_loss = pixelda_losses.d_step_loss(
 304           end_points, source_labels, num_target_classes, hparams)
 305 
 306       ###########################
 307       # Create the training ops #
 308       ###########################
 309       learning_rate = hparams.learning_rate
 310       if hparams.lr_decay_steps:
 311         learning_rate = tf.train.exponential_decay(
 312             learning_rate,
 313             slim.get_or_create_global_step(),
 314             decay_steps=hparams.lr_decay_steps,
 315             decay_rate=hparams.lr_decay_rate,
 316             staircase=True)
 317       tf.summary.scalar('Learning_rate', learning_rate)
 318 
 319 
 320       if hparams.discriminator_steps == 0:
 321         discriminator_train_op = tf.no_op()
 322       else:
 323         discriminator_optimizer = tf.train.AdamOptimizer(
 324             learning_rate, beta1=hparams.adam_beta1)
 325 
 326         discriminator_train_op = slim.learning.create_train_op(
 327             discriminator_loss,
 328             discriminator_optimizer,
 329             update_ops=discriminator_update_ops,
 330             variables_to_train=discriminator_vars,
 331             clip_gradient_norm=hparams.clip_gradient_norm,
 332             summarize_gradients=FLAGS.summarize_gradients)
 333 
 334       if hparams.generator_steps == 0:
 335         generator_train_op = tf.no_op()
 336       else:
 337         generator_optimizer = tf.train.AdamOptimizer(
 338             learning_rate, beta1=hparams.adam_beta1)
 339         generator_train_op = slim.learning.create_train_op(
 340             generator_loss,
 341             generator_optimizer,
 342             update_ops=generator_update_ops,
 343             variables_to_train=generator_vars,
 344             clip_gradient_norm=hparams.clip_gradient_norm,
 345             summarize_gradients=FLAGS.summarize_gradients)
 346 
 347       #############
 348       # Summaries #
 349       #############
 350       pixelda_utils.summarize_model(end_points)
 351       pixelda_utils.summarize_transferred_grid(
 352           end_points['transferred_images'], source_images, name='Transferred')
 353       if 'source_images_recon' in end_points:
 354         pixelda_utils.summarize_transferred_grid(
 355             end_points['source_images_recon'],
 356             source_images,
 357             name='Source Reconstruction')
 358       pixelda_utils.summaries_color_distributions(end_points['transferred_images'],
 359                                                'Transferred')
 360       pixelda_utils.summaries_color_distributions(target_images, 'Target')
 361 
 362       if source_images is not None:
 363         pixelda_utils.summarize_transferred(source_images,
 364                                          end_points['transferred_images'])
 365         pixelda_utils.summaries_color_distributions(source_images, 'Source')
 366         pixelda_utils.summaries_color_distributions(
 367             tf.abs(source_images - end_points['transferred_images']),
 368             'Abs(Source_minus_Transferred)')
 369 
 370       number_of_steps = None
 371       if hparams.num_training_examples:
 372         # Want to control by amount of data seen, not # steps
 373         number_of_steps = hparams.num_training_examples / hparams.batch_size
 374 
 375       hooks = [tf.train.StepCounterHook(),]
 376 
 377       chief_only_hooks = [
 378           tf.train.CheckpointSaverHook(
 379               saver=tf.train.Saver(),
 380               checkpoint_dir=run_dir,
 381               save_secs=FLAGS.save_interval_secs)
 382       ]
 383 
 384       if number_of_steps:
 385         hooks.append(tf.train.StopAtStepHook(last_step=number_of_steps))
 386 
 387       _train(
 388           discriminator_train_op,
 389           generator_train_op,
 390           logdir=run_dir,
 391           master=FLAGS.master,
 392           is_chief=FLAGS.task == 0,
 393           hooks=hooks,
 394           chief_only_hooks=chief_only_hooks,
 395           save_checkpoint_secs=None,
 396           save_summaries_steps=FLAGS.save_summaries_steps,
 397           hparams=hparams)
 398 
 399 def main(_):
 400   tf.logging.set_verbosity(tf.logging.INFO)
 401   hparams = create_hparams(FLAGS.hparams)
 402   run_training(
 403       run_dir=FLAGS.train_log_dir,
 404       checkpoint_dir=FLAGS.train_log_dir,
 405       hparams=hparams)

 408 if __name__ == '__main__':
 409   tf.app.run()
