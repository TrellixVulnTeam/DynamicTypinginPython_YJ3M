  30 def get_dataset(dataset_name,
  31                 split_name,
  32                 dataset_dir,
  33                 file_pattern=None,
  34                 reader=None):
  35   """Given a dataset name and a split_name returns a Dataset.
  36 
  37   Args:
  38     dataset_name: String, the name of the dataset.
  39     split_name: A train/test split name.
  40     dataset_dir: The directory where the dataset files are stored.
  41     file_pattern: The file pattern to use for matching the dataset source files.
  42     reader: The subclass of tf.ReaderBase. If left as `None`, then the default
  43       reader defined by each dataset is used.
  44 
  45   Returns:
  46     A tf-slim `Dataset` class.
  47 
  48   Raises:
  49     ValueError: if `dataset_name` isn't recognized.
  50   """
  51   dataset_name_to_module = {'mnist': mnist, 'mnist_m': mnist_m}
  52   if dataset_name not in dataset_name_to_module:
  53     raise ValueError('Name of dataset unknown %s.' % dataset_name)
  54 
  55   return dataset_name_to_module[dataset_name].get_split(split_name, dataset_dir,
  56                                                         file_pattern, reader)

  59 def provide_batch(dataset_name, split_name, dataset_dir, num_readers,
  60                   batch_size, num_preprocessing_threads):
  61   """Provides a batch of images and corresponding labels.
  62 
  63     Args:
  64     dataset_name: String, the name of the dataset.
  65     split_name: A train/test split name.
  66     dataset_dir: The directory where the dataset files are stored.
  67     num_readers: The number of readers used by DatasetDataProvider.
  68     batch_size: The size of the batch requested.
  69     num_preprocessing_threads: The number of preprocessing threads for
  70       tf.train.batch.
  71     file_pattern: The file pattern to use for matching the dataset source files.
  72     reader: The subclass of tf.ReaderBase. If left as `None`, then the default
  73       reader defined by each dataset is used.
  74 
  75   Returns:
  76     A batch of
  77       images: tensor of [batch_size, height, width, channels].
  78       labels: dictionary of labels.
  79   """
  80   dataset = get_dataset(dataset_name, split_name, dataset_dir)
  81   provider = slim.dataset_data_provider.DatasetDataProvider(
  82       dataset,
  83       num_readers=num_readers,
  84       common_queue_capacity=20 * batch_size,
  85       common_queue_min=10 * batch_size)
  86   [image, label] = provider.get(['image', 'label'])
  87 
  88   # Convert images to float32
  89   image = tf.image.convert_image_dtype(image, tf.float32)
  90   image -= 0.5
  91   image *= 2
  92 
  93   # Load the data.
  94   labels = {}
  95   images, labels['classes'] = tf.train.batch(
  96       [image, label],
  97       batch_size=batch_size,
  98       num_threads=num_preprocessing_threads,
  99       capacity=5 * batch_size)
 100   labels['classes'] = slim.one_hot_encoding(labels['classes'],
 101                                             dataset.num_classes)
 102 
 103   # Convert mnist to RGB and 32x32 so that it can match mnist_m.
 104   if dataset_name == 'mnist':
 105     images = tf.image.grayscale_to_rgb(images)
 106     images = tf.image.resize_images(images, [32, 32])
 107   return images, labels
