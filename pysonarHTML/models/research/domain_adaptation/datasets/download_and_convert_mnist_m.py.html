  15 r"""Downloads and converts MNIST-M data to TFRecords of TF-Example protos.
  16 
  17 This module downloads the MNIST-M data, uncompresses it, reads the files
  18 that make up the MNIST-M data and creates two TFRecord datasets: one for train
  19 and one for test. Each TFRecord dataset is comprised of a set of TF-Example
  20 protocol buffers, each of which contain a single image and label.
  21 
  22 The script should take about a minute to run.
  23 
  24 """

  76 class ImageReader(object):
  77   """Helper class that provides TensorFlow image coding utilities."""
  78 
  79   def __init__(self):
  80     # Initializes function that decodes RGB PNG data.
  81     self._decode_png_data = tf.placeholder(dtype=tf.string)
  82     self._decode_png = tf.image.decode_png(self._decode_png_data, channels=3)
  83 
  84   def read_image_dims(self, sess, image_data):
  85     image = self.decode_png(sess, image_data)
  86     return image.shape[0], image.shape[1]
  87 
  88   def decode_png(self, sess, image_data):
  89     image = sess.run(
  90         self._decode_png, feed_dict={self._decode_png_data: image_data})
  91     assert len(image.shape) == 3
  92     assert image.shape[2] == 3
  93     return image

  96 def _convert_dataset(split_name, filenames, filename_to_class_id, dataset_dir):
  97   """Converts the given filenames to a TFRecord dataset.
  98 
  99   Args:
 100     split_name: The name of the dataset, either 'train' or 'valid'.
 101     filenames: A list of absolute paths to png images.
 102     filename_to_class_id: A dictionary from filenames (strings) to class ids
 103       (integers).
 104     dataset_dir: The directory where the converted datasets are stored.
 105   """
 106   print('Converting the {} split.'.format(split_name))
 107   # Train and validation splits are both in the train directory.
 108   if split_name in ['train', 'valid']:
 109     png_directory = os.path.join(dataset_dir, 'mnist_m', 'mnist_m_train')
 110   elif split_name == 'test':
 111     png_directory = os.path.join(dataset_dir, 'mnist_m', 'mnist_m_test')
 112 
 113   with tf.Graph().as_default():
 114     image_reader = ImageReader()
 115 
 116     with tf.Session('') as sess:
 117       output_filename = _get_output_filename(dataset_dir, split_name)
 118 
 119       with tf.python_io.TFRecordWriter(output_filename) as tfrecord_writer:
 120         for filename in filenames:
 121           # Read the filename:
 122           image_data = tf.gfile.FastGFile(
 123               os.path.join(png_directory, filename), 'r').read()
 124           height, width = image_reader.read_image_dims(sess, image_data)
 125 
 126           class_id = filename_to_class_id[filename]
 127           example = dataset_utils.image_to_tfexample(image_data, 'png', height,
 128                                                      width, class_id)
 129           tfrecord_writer.write(example.SerializeToString())
 130 
 131   sys.stdout.write('\n')
 132   sys.stdout.flush()

 135 def _extract_labels(label_filename):
 136   """Extract the labels into a dict of filenames to int labels.
 137 
 138   Args:
 139     labels_filename: The filename of the MNIST-M labels.
 140 
 141   Returns:
 142     A dictionary of filenames to int labels.
 143   """
 144   print('Extracting labels from: ', label_filename)
 145   label_file = tf.gfile.FastGFile(label_filename, 'r').readlines()
 146   label_lines = [line.rstrip('\n').split() for line in label_file]
 147   labels = {}
 148   for line in label_lines:
 149     assert len(line) == 2
 150     labels[line[0]] = int(line[1])
 151   return labels

 154 def _get_output_filename(dataset_dir, split_name):
 155   """Creates the output filename.
 156 
 157   Args:
 158     dataset_dir: The directory where the temporary files are stored.
 159     split_name: The name of the train/test split.
 160 
 161   Returns:
 162     An absolute file path.
 163   """
 164   return '%s/mnist_m_%s.tfrecord' % (dataset_dir, split_name)

 167 def _get_filenames(dataset_dir):
 168   """Returns a list of filenames and inferred class names.
 169 
 170   Args:
 171     dataset_dir: A directory containing a set PNG encoded MNIST-M images.
 172 
 173   Returns:
 174     A list of image file paths, relative to `dataset_dir`.
 175   """
 176   photo_filenames = []
 177   for filename in os.listdir(dataset_dir):
 178     photo_filenames.append(filename)
 179   return photo_filenames

 182 def run(dataset_dir):
 183   """Runs the download and conversion operation.
 184 
 185   Args:
 186     dataset_dir: The dataset directory where the dataset is stored.
 187   """
 188   if not tf.gfile.Exists(dataset_dir):
 189     tf.gfile.MakeDirs(dataset_dir)
 190 
 191   train_filename = _get_output_filename(dataset_dir, 'train')
 192   testing_filename = _get_output_filename(dataset_dir, 'test')
 193 
 194   if tf.gfile.Exists(train_filename) and tf.gfile.Exists(testing_filename):
 195     print('Dataset files already exist. Exiting without re-creating them.')
 196     return
 197 
 198   # TODO(konstantinos): Add download and cleanup functionality
 199 
 200   train_validation_filenames = _get_filenames(
 201       os.path.join(dataset_dir, 'mnist_m', 'mnist_m_train'))
 202   test_filenames = _get_filenames(
 203       os.path.join(dataset_dir, 'mnist_m', 'mnist_m_test'))
 204 
 205   # Divide into train and validation:
 206   random.seed(_RANDOM_SEED)
 207   random.shuffle(train_validation_filenames)
 208   train_filenames = train_validation_filenames[_NUM_VALIDATION:]
 209   validation_filenames = train_validation_filenames[:_NUM_VALIDATION]
 210 
 211   train_validation_filenames_to_class_ids = _extract_labels(
 212       os.path.join(dataset_dir, 'mnist_m', 'mnist_m_train_labels.txt'))
 213   test_filenames_to_class_ids = _extract_labels(
 214       os.path.join(dataset_dir, 'mnist_m', 'mnist_m_test_labels.txt'))
 215 
 216   # Convert the train, validation, and test sets.
 217   _convert_dataset('train', train_filenames,
 218                    train_validation_filenames_to_class_ids, dataset_dir)
 219   _convert_dataset('valid', validation_filenames,
 220                    train_validation_filenames_to_class_ids, dataset_dir)
 221   _convert_dataset('test', test_filenames, test_filenames_to_class_ids,
 222                    dataset_dir)
 223 
 224   # Finally, write the labels file:
 225   labels_to_class_names = dict(zip(range(len(_CLASS_NAMES)), _CLASS_NAMES))
 226   dataset_utils.write_label_file(labels_to_class_names, dataset_dir)
 227 
 228   print('\nFinished converting the MNIST-M dataset!')
 229
