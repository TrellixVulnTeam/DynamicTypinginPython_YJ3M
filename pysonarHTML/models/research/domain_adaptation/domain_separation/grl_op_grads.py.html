  16 """Gradients for operators defined in grl_ops.py."""
  17 import tensorflow as tf

  20 @tf.RegisterGradient("GradientReversal")
  21 def _GradientReversalGrad(_, grad):
  22   """The gradients for `gradient_reversal`.
  23 
  24   Args:
  25     _: The `gradient_reversal` `Operation` that we are differentiating,
  26       which we can use to find the inputs and outputs of the original op.
  27     grad: Gradient with respect to the output of the `gradient_reversal` op.
  28 
  29   Returns:
  30     Gradient with respect to the input of `gradient_reversal`, which is simply
  31     the negative of the input gradient.
  32 
  33   """
  34   return tf.negative(grad)
