  40 def dsn_loss_coefficient(params):
  41   """The global_step-dependent weight that specifies when to kick in DSN losses.
  42 
  43   Args:
  44     params: A dictionary of parameters. Expecting 'domain_separation_startpoint'
  45 
  46   Returns:
  47     A weight to that effectively enables or disables the DSN-related losses,
  48     i.e. similarity, difference, and reconstruction losses.
  49   """
  50   return tf.where(
  51       tf.less(slim.get_or_create_global_step(),
  52               params['domain_separation_startpoint']), 1e-10, 1.0)

  58 def create_model(source_images, source_labels, domain_selection_mask,
  59                  target_images, target_labels, similarity_loss, params,
  60                  basic_tower_name):
  61   """Creates a DSN model.
  62 
  63   Args:
  64     source_images: images from the source domain, a tensor of size
  65       [batch_size, height, width, channels]
  66     source_labels: a dictionary with the name, tensor pairs. 'classes' is one-
  67       hot for the number of classes.
  68     domain_selection_mask: a boolean tensor of size [batch_size, ] which denotes
  69       the labeled images that belong to the source domain.
  70     target_images: images from the target domain, a tensor of size
  71       [batch_size, height width, channels].
  72     target_labels: a dictionary with the name, tensor pairs.
  73     similarity_loss: The type of method to use for encouraging
  74       the codes from the shared encoder to be similar.
  75     params: A dictionary of parameters. Expecting 'weight_decay',
  76       'layers_to_regularize', 'use_separation', 'domain_separation_startpoint',
  77       'alpha_weight', 'beta_weight', 'gamma_weight', 'recon_loss_name',
  78       'decoder_name', 'encoder_name'
  79     basic_tower_name: the name of the tower to use for the shared encoder.
  80 
  81   Raises:
  82     ValueError: if the arch is not one of the available architectures.
  83   """
  84   network = getattr(models, basic_tower_name)
  85   num_classes = source_labels['classes'].get_shape().as_list()[1]
  86 
  87   # Make sure we are using the appropriate number of classes.
  88   network = partial(network, num_classes=num_classes)
  89 
  90   # Add the classification/pose estimation loss to the source domain.
  91   source_endpoints = add_task_loss(source_images, source_labels, network,
  92                                    params)
  93 
  94   if similarity_loss == 'none':
  95     # No domain adaptation, we can stop here.
  96     return
  97 
  98   with tf.variable_scope('towers', reuse=True):
  99     target_logits, target_endpoints = network(
 100         target_images, weight_decay=params['weight_decay'], prefix='target')
 101 
 102   # Plot target accuracy of the train set.
 103   target_accuracy = utils.accuracy(
 104       tf.argmax(target_logits, 1), tf.argmax(target_labels['classes'], 1))
 105 
 106   if 'quaternions' in target_labels:
 107     target_quaternion_loss = losses.log_quaternion_loss(
 108         target_labels['quaternions'], target_endpoints['quaternion_pred'],
 109         params)
 110     tf.summary.scalar('eval/Target quaternions', target_quaternion_loss)
 111 
 112   tf.summary.scalar('eval/Target accuracy', target_accuracy)
 113 
 114   source_shared = source_endpoints[params['layers_to_regularize']]
 115   target_shared = target_endpoints[params['layers_to_regularize']]
 116 
 117   # When using the semisupervised model we include labeled target data in the
 118   # source classifier. We do not want to include these target domain when
 119   # we use the similarity loss.
 120   indices = tf.range(0, source_shared.get_shape().as_list()[0])
 121   indices = tf.boolean_mask(indices, domain_selection_mask)
 122   add_similarity_loss(similarity_loss,
 123                       tf.gather(source_shared, indices),
 124                       tf.gather(target_shared, indices), params)
 125 
 126   if params['use_separation']:
 127     add_autoencoders(
 128         source_images,
 129         source_shared,
 130         target_images,
 131         target_shared,
 132         params=params,)

 135 def add_similarity_loss(method_name,
 136                         source_samples,
 137                         target_samples,
 138                         params,
 139                         scope=None):
 140   """Adds a loss encouraging the shared encoding from each domain to be similar.
 141 
 142   Args:
 143     method_name: the name of the encoding similarity method to use. Valid
 144       options include `dann_loss', `mmd_loss' or `correlation_loss'.
 145     source_samples: a tensor of shape [num_samples, num_features].
 146     target_samples: a tensor of shape [num_samples, num_features].
 147     params: a dictionary of parameters. Expecting 'gamma_weight'.
 148     scope: optional name scope for summary tags.
 149   Raises:
 150     ValueError: if `method_name` is not recognized.
 151   """
 152   weight = dsn_loss_coefficient(params) * params['gamma_weight']
 153   method = getattr(losses, method_name)
 154   method(source_samples, target_samples, weight, scope)

 157 def add_reconstruction_loss(recon_loss_name, images, recons, weight, domain):
 158   """Adds a reconstruction loss.
 159 
 160   Args:
 161     recon_loss_name: The name of the reconstruction loss.
 162     images: A `Tensor` of size [batch_size, height, width, 3].
 163     recons: A `Tensor` whose size matches `images`.
 164     weight: A scalar coefficient for the loss.
 165     domain: The name of the domain being reconstructed.
 166 
 167   Raises:
 168     ValueError: If `recon_loss_name` is not recognized.
 169   """
 170   if recon_loss_name == 'sum_of_pairwise_squares':
 171     loss_fn = tf.contrib.losses.mean_pairwise_squared_error
 172   elif recon_loss_name == 'sum_of_squares':
 173     loss_fn = tf.contrib.losses.mean_squared_error
 174   else:
 175     raise ValueError('recon_loss_name value [%s] not recognized.' %
 176                      recon_loss_name)
 177 
 178   loss = loss_fn(recons, images, weight)
 179   assert_op = tf.Assert(tf.is_finite(loss), [loss])
 180   with tf.control_dependencies([assert_op]):
 181     tf.summary.scalar('losses/%s Recon Loss' % domain, loss)

 184 def add_autoencoders(source_data, source_shared, target_data, target_shared,
 185                      params):
 186   """Adds the encoders/decoders for our domain separation model w/ incoherence.
 187 
 188   Args:
 189     source_data: images from the source domain, a tensor of size
 190       [batch_size, height, width, channels]
 191     source_shared: a tensor with first dimension batch_size
 192     target_data: images from the target domain, a tensor of size
 193       [batch_size, height, width, channels]
 194     target_shared: a tensor with first dimension batch_size
 195     params: A dictionary of parameters. Expecting 'layers_to_regularize',
 196       'beta_weight', 'alpha_weight', 'recon_loss_name', 'decoder_name',
 197       'encoder_name', 'weight_decay'
 198   """
 199 
 200   def normalize_images(images):
 201     images -= tf.reduce_min(images)
 202     return images / tf.reduce_max(images)
 203 
 204   def concat_operation(shared_repr, private_repr):
 205     return shared_repr + private_repr
 206 
 207   mu = dsn_loss_coefficient(params)
 208 
 209   # The layer to concatenate the networks at.
 210   concat_layer = params['layers_to_regularize']
 211 
 212   # The coefficient for modulating the private/shared difference loss.
 213   difference_loss_weight = params['beta_weight'] * mu
 214 
 215   # The reconstruction weight.
 216   recon_loss_weight = params['alpha_weight'] * mu
 217 
 218   # The reconstruction loss to use.
 219   recon_loss_name = params['recon_loss_name']
 220 
 221   # The decoder/encoder to use.
 222   decoder_name = params['decoder_name']
 223   encoder_name = params['encoder_name']
 224 
 225   _, height, width, _ = source_data.get_shape().as_list()
 226   code_size = source_shared.get_shape().as_list()[-1]
 227   weight_decay = params['weight_decay']
 228 
 229   encoder_fn = getattr(models, encoder_name)
 230   # Target Auto-encoding.
 231   with tf.variable_scope('source_encoder'):
 232     source_endpoints = encoder_fn(
 233         source_data, code_size, weight_decay=weight_decay)
 234 
 235   with tf.variable_scope('target_encoder'):
 236     target_endpoints = encoder_fn(
 237         target_data, code_size, weight_decay=weight_decay)
 238 
 239   decoder_fn = getattr(models, decoder_name)
 240 
 241   decoder = partial(
 242       decoder_fn,
 243       height=height,
 244       width=width,
 245       channels=source_data.get_shape().as_list()[-1],
 246       weight_decay=weight_decay)
 247 
 248   # Source Auto-encoding.
 249   source_private = source_endpoints[concat_layer]
 250   target_private = target_endpoints[concat_layer]
 251   with tf.variable_scope('decoder'):
 252     source_recons = decoder(concat_operation(source_shared, source_private))
 253 
 254   with tf.variable_scope('decoder', reuse=True):
 255     source_private_recons = decoder(
 256         concat_operation(tf.zeros_like(source_private), source_private))
 257     source_shared_recons = decoder(
 258         concat_operation(source_shared, tf.zeros_like(source_shared)))
 259 
 260   with tf.variable_scope('decoder', reuse=True):
 261     target_recons = decoder(concat_operation(target_shared, target_private))
 262     target_shared_recons = decoder(
 263         concat_operation(target_shared, tf.zeros_like(target_shared)))
 264     target_private_recons = decoder(
 265         concat_operation(tf.zeros_like(target_private), target_private))
 266 
 267   losses.difference_loss(
 268       source_private,
 269       source_shared,
 270       weight=difference_loss_weight,
 271       name='Source')
 272   losses.difference_loss(
 273       target_private,
 274       target_shared,
 275       weight=difference_loss_weight,
 276       name='Target')
 277 
 278   add_reconstruction_loss(recon_loss_name, source_data, source_recons,
 279                           recon_loss_weight, 'source')
 280   add_reconstruction_loss(recon_loss_name, target_data, target_recons,
 281                           recon_loss_weight, 'target')
 282 
 283   # Add summaries
 284   source_reconstructions = tf.concat(
 285       axis=2,
 286       values=map(normalize_images, [
 287           source_data, source_recons, source_shared_recons,
 288           source_private_recons
 289       ]))
 290   target_reconstructions = tf.concat(
 291       axis=2,
 292       values=map(normalize_images, [
 293           target_data, target_recons, target_shared_recons,
 294           target_private_recons
 295       ]))
 296   tf.summary.image(
 297       'Source Images:Recons:RGB',
 298       source_reconstructions[:, :, :, :3],
 299       max_outputs=10)
 300   tf.summary.image(
 301       'Target Images:Recons:RGB',
 302       target_reconstructions[:, :, :, :3],
 303       max_outputs=10)
 304 
 305   if source_reconstructions.get_shape().as_list()[3] == 4:
 306     tf.summary.image(
 307         'Source Images:Recons:Depth',
 308         source_reconstructions[:, :, :, 3:4],
 309         max_outputs=10)
 310     tf.summary.image(
 311         'Target Images:Recons:Depth',
 312         target_reconstructions[:, :, :, 3:4],
 313         max_outputs=10)

 316 def add_task_loss(source_images, source_labels, basic_tower, params):
 317   """Adds a classification and/or pose estimation loss to the model.
 318 
 319   Args:
 320     source_images: images from the source domain, a tensor of size
 321       [batch_size, height, width, channels]
 322     source_labels: labels from the source domain, a tensor of size [batch_size].
 323       or a tuple of (quaternions, class_labels)
 324     basic_tower: a function that creates the single tower of the model.
 325     params: A dictionary of parameters. Expecting 'weight_decay', 'pose_weight'.
 326   Returns:
 327     The source endpoints.
 328 
 329   Raises:
 330     RuntimeError: if basic tower does not support pose estimation.
 331   """
 332   with tf.variable_scope('towers'):
 333     source_logits, source_endpoints = basic_tower(
 334         source_images, weight_decay=params['weight_decay'], prefix='Source')
 335 
 336   if 'quaternions' in source_labels:  # We have pose estimation as well
 337     if 'quaternion_pred' not in source_endpoints:
 338       raise RuntimeError('Please use a model for estimation e.g. pose_mini')
 339 
 340     loss = losses.log_quaternion_loss(source_labels['quaternions'],
 341                                       source_endpoints['quaternion_pred'],
 342                                       params)
 343 
 344     assert_op = tf.Assert(tf.is_finite(loss), [loss])
 345     with tf.control_dependencies([assert_op]):
 346       quaternion_loss = loss
 347       tf.summary.histogram('log_quaternion_loss_hist', quaternion_loss)
 348     slim.losses.add_loss(quaternion_loss * params['pose_weight'])
 349     tf.summary.scalar('losses/quaternion_loss', quaternion_loss)
 350 
 351   classification_loss = tf.losses.softmax_cross_entropy(
 352       source_labels['classes'], source_logits)
 353 
 354   tf.summary.scalar('losses/classification_loss', classification_loss)
 355   return source_endpoints
