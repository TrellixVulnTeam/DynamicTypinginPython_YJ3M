  23 class HelperFunctionsTest(tf.test.TestCase):

  25   def testBasicDomainSeparationStartPoint(self):
  26     with self.test_session() as sess:
  27       # Test for when global_step < domain_separation_startpoint
  28       step = tf.contrib.slim.get_or_create_global_step()
  29       sess.run(tf.global_variables_initializer())  # global_step = 0
  30       params = {'domain_separation_startpoint': 2}
  31       weight = dsn.dsn_loss_coefficient(params)
  32       weight_np = sess.run(weight)
  33       self.assertAlmostEqual(weight_np, 1e-10)
  34 
  35       step_op = tf.assign_add(step, 1)
  36       step_np = sess.run(step_op)  # global_step = 1
  37       weight = dsn.dsn_loss_coefficient(params)
  38       weight_np = sess.run(weight)
  39       self.assertAlmostEqual(weight_np, 1e-10)
  40 
  41       # Test for when global_step >= domain_separation_startpoint
  42       step_np = sess.run(step_op)  # global_step = 2
  43       tf.logging.info(step_np)
  44       weight = dsn.dsn_loss_coefficient(params)
  45       weight_np = sess.run(weight)
  46       self.assertAlmostEqual(weight_np, 1.0)

  49 class DsnModelAssemblyTest(tf.test.TestCase):
  50 
  51   def _testBuildDefaultModel(self):
  52     images = tf.to_float(np.random.rand(32, 28, 28, 1))
  53     labels = {}
  54     labels['classes'] = tf.one_hot(
  55         tf.to_int32(np.random.randint(0, 9, (32))), 10)
  56 
  57     params = {
  58         'use_separation': True,
  59         'layers_to_regularize': 'fc3',
  60         'weight_decay': 0.0,
  61         'ps_tasks': 1,
  62         'domain_separation_startpoint': 1,
  63         'alpha_weight': 1,
  64         'beta_weight': 1,
  65         'gamma_weight': 1,
  66         'recon_loss_name': 'sum_of_squares',
  67         'decoder_name': 'small_decoder',
  68         'encoder_name': 'default_encoder',
  69     }
  70     return images, labels, params
  71 
  72   def testBuildModelDann(self):
  73     images, labels, params = self._testBuildDefaultModel()
  74 
  75     with self.test_session():
  76       dsn.create_model(images, labels,
  77                        tf.cast(tf.ones([32,]), tf.bool), images, labels,
  78                        'dann_loss', params, 'dann_mnist')
  79       loss_tensors = tf.contrib.losses.get_losses()
  80     self.assertEqual(len(loss_tensors), 6)
