  31 def default_batch_norm_params(is_training=False):
  32   """Returns default batch normalization parameters for DSNs.
  33 
  34   Args:
  35     is_training: whether or not the model is training.
  36 
  37   Returns:
  38     a dictionary that maps batch norm parameter names (strings) to values.
  39   """
  40   return {
  41       # Decay for the moving averages.
  42       'decay': 0.5,
  43       # epsilon to prevent 0s in variance.
  44       'epsilon': 0.001,
  45       'is_training': is_training
  46   }

  52 def default_encoder(images, code_size, batch_norm_params=None,
  53                     weight_decay=0.0):
  54   """Encodes the given images to codes of the given size.
  55 
  56   Args:
  57     images: a tensor of size [batch_size, height, width, 1].
  58     code_size: the number of hidden units in the code layer of the classifier.
  59     batch_norm_params: a dictionary that maps batch norm parameter names to
  60       values.
  61     weight_decay: the value for the weight decay coefficient.
  62 
  63   Returns:
  64     end_points: the code of the input.
  65   """
  66   end_points = {}
  67   with slim.arg_scope(
  68       [slim.conv2d, slim.fully_connected],
  69       weights_regularizer=slim.l2_regularizer(weight_decay),
  70       activation_fn=tf.nn.relu,
  71       normalizer_fn=slim.batch_norm,
  72       normalizer_params=batch_norm_params):
  73     with slim.arg_scope([slim.conv2d], kernel_size=[5, 5], padding='SAME'):
  74       net = slim.conv2d(images, 32, scope='conv1')
  75       net = slim.max_pool2d(net, [2, 2], 2, scope='pool1')
  76       net = slim.conv2d(net, 64, scope='conv2')
  77       net = slim.max_pool2d(net, [2, 2], 2, scope='pool2')
  78 
  79       net = slim.flatten(net)
  80       end_points['flatten'] = net
  81       net = slim.fully_connected(net, code_size, scope='fc1')
  82       end_points['fc3'] = net
  83   return end_points

  89 def large_decoder(codes,
  90                   height,
  91                   width,
  92                   channels,
  93                   batch_norm_params=None,
  94                   weight_decay=0.0):
  95   """Decodes the codes to a fixed output size.
  96 
  97   Args:
  98     codes: a tensor of size [batch_size, code_size].
  99     height: the height of the output images.
 100     width: the width of the output images.
 101     channels: the number of the output channels.
 102     batch_norm_params: a dictionary that maps batch norm parameter names to
 103       values.
 104     weight_decay: the value for the weight decay coefficient.
 105 
 106   Returns:
 107     recons: the reconstruction tensor of shape [batch_size, height, width, 3].
 108   """
 109   with slim.arg_scope(
 110       [slim.conv2d, slim.fully_connected],
 111       weights_regularizer=slim.l2_regularizer(weight_decay),
 112       activation_fn=tf.nn.relu,
 113       normalizer_fn=slim.batch_norm,
 114       normalizer_params=batch_norm_params):
 115     net = slim.fully_connected(codes, 600, scope='fc1')
 116     batch_size = net.get_shape().as_list()[0]
 117     net = tf.reshape(net, [batch_size, 10, 10, 6])
 118 
 119     net = slim.conv2d(net, 32, [5, 5], scope='conv1_1')
 120 
 121     net = tf.image.resize_nearest_neighbor(net, (16, 16))
 122 
 123     net = slim.conv2d(net, 32, [5, 5], scope='conv2_1')
 124 
 125     net = tf.image.resize_nearest_neighbor(net, (32, 32))
 126 
 127     net = slim.conv2d(net, 32, [5, 5], scope='conv3_2')
 128 
 129     output_size = [height, width]
 130     net = tf.image.resize_nearest_neighbor(net, output_size)
 131 
 132     with slim.arg_scope([slim.conv2d], kernel_size=[3, 3]):
 133       net = slim.conv2d(net, channels, activation_fn=None, scope='conv4_1')
 134 
 135   return net

 138 def gtsrb_decoder(codes,
 139                   height,
 140                   width,
 141                   channels,
 142                   batch_norm_params=None,
 143                   weight_decay=0.0):
 144   """Decodes the codes to a fixed output size. This decoder is specific to GTSRB
 145 
 146   Args:
 147     codes: a tensor of size [batch_size, 100].
 148     height: the height of the output images.
 149     width: the width of the output images.
 150     channels: the number of the output channels.
 151     batch_norm_params: a dictionary that maps batch norm parameter names to
 152       values.
 153     weight_decay: the value for the weight decay coefficient.
 154 
 155   Returns:
 156     recons: the reconstruction tensor of shape [batch_size, height, width, 3].
 157 
 158   Raises:
 159     ValueError: When the input code size is not 100.
 160   """
 161   batch_size, code_size = codes.get_shape().as_list()
 162   if code_size != 100:
 163     raise ValueError('The code size used as an input to the GTSRB decoder is '
 164                      'expected to be 100.')
 165 
 166   with slim.arg_scope(
 167       [slim.conv2d, slim.fully_connected],
 168       weights_regularizer=slim.l2_regularizer(weight_decay),
 169       activation_fn=tf.nn.relu,
 170       normalizer_fn=slim.batch_norm,
 171       normalizer_params=batch_norm_params):
 172     net = codes
 173     net = tf.reshape(net, [batch_size, 10, 10, 1])
 174     net = slim.conv2d(net, 32, [3, 3], scope='conv1_1')
 175 
 176     # First upsampling 20x20
 177     net = tf.image.resize_nearest_neighbor(net, [20, 20])
 178 
 179     net = slim.conv2d(net, 32, [3, 3], scope='conv2_1')
 180 
 181     output_size = [height, width]
 182     # Final upsampling 40 x 40
 183     net = tf.image.resize_nearest_neighbor(net, output_size)
 184 
 185     with slim.arg_scope([slim.conv2d], kernel_size=[3, 3]):
 186       net = slim.conv2d(net, 16, scope='conv3_1')
 187       net = slim.conv2d(net, channels, activation_fn=None, scope='conv3_2')
 188 
 189   return net

 238 def dann_mnist(images,
 239                weight_decay=0.0,
 240                prefix='model',
 241                num_classes=10,
 242                **kwargs):
 243   """Creates a convolution MNIST model.
 244 
 245   Note that this model implements the architecture for MNIST proposed in:
 246    Y. Ganin et al., Domain-Adversarial Training of Neural Networks (DANN),
 247    JMLR 2015
 248 
 249   Args:
 250     images: the MNIST digits, a tensor of size [batch_size, 28, 28, 1].
 251     weight_decay: the value for the weight decay coefficient.
 252     prefix: name of the model to use when prefixing tags.
 253     num_classes: the number of output classes to use.
 254     **kwargs: Placeholder for keyword arguments used by other shared encoders.
 255 
 256   Returns:
 257     the output logits, a tensor of size [batch_size, num_classes].
 258     a dictionary with key/values the layer names and tensors.
 259   """
 260   end_points = {}
 261 
 262   with slim.arg_scope(
 263       [slim.conv2d, slim.fully_connected],
 264       weights_regularizer=slim.l2_regularizer(weight_decay),
 265       activation_fn=tf.nn.relu,):
 266     with slim.arg_scope([slim.conv2d], padding='SAME'):
 267       end_points['conv1'] = slim.conv2d(images, 32, [5, 5], scope='conv1')
 268       end_points['pool1'] = slim.max_pool2d(
 269           end_points['conv1'], [2, 2], 2, scope='pool1')
 270       end_points['conv2'] = slim.conv2d(
 271           end_points['pool1'], 48, [5, 5], scope='conv2')
 272       end_points['pool2'] = slim.max_pool2d(
 273           end_points['conv2'], [2, 2], 2, scope='pool2')
 274       end_points['fc3'] = slim.fully_connected(
 275           slim.flatten(end_points['pool2']), 100, scope='fc3')
 276       end_points['fc4'] = slim.fully_connected(
 277           slim.flatten(end_points['fc3']), 100, scope='fc4')
 278 
 279   logits = slim.fully_connected(
 280       end_points['fc4'], num_classes, activation_fn=None, scope='fc5')
 281 
 282   return logits, end_points

 389 def dsn_cropped_linemod(images,
 390                         weight_decay=0.0,
 391                         prefix='model',
 392                         num_classes=11,
 393                         batch_norm_params=None,
 394                         is_training=False):
 395   """Creates the convolutional pose estimation model for Cropped Linemod.
 396 
 397   Args:
 398     images: the Cropped Linemod samples, a tensor of size
 399       [batch_size, 64, 64, 4].
 400     weight_decay: the value for the weight decay coefficient.
 401     prefix: name of the model to use when prefixing tags.
 402     num_classes: the number of output classes to use.
 403     batch_norm_params: a dictionary that maps batch norm parameter names to
 404       values.
 405     is_training: specifies whether or not we're currently training the model.
 406       This variable will determine the behaviour of the dropout layer.
 407 
 408   Returns:
 409     the output logits, a tensor of size [batch_size, num_classes].
 410     a dictionary with key/values the layer names and tensors.
 411   """
 412 
 413   end_points = {}
 414 
 415   tf.summary.image('{}/input_images'.format(prefix), images)
 416   with slim.arg_scope(
 417       [slim.conv2d, slim.fully_connected],
 418       weights_regularizer=slim.l2_regularizer(weight_decay),
 419       activation_fn=tf.nn.relu,
 420       normalizer_fn=slim.batch_norm if batch_norm_params else None,
 421       normalizer_params=batch_norm_params):
 422     with slim.arg_scope([slim.conv2d], padding='SAME'):
 423       end_points['conv1'] = slim.conv2d(images, 32, [5, 5], scope='conv1')
 424       end_points['pool1'] = slim.max_pool2d(
 425           end_points['conv1'], [2, 2], 2, scope='pool1')
 426       end_points['conv2'] = slim.conv2d(
 427           end_points['pool1'], 64, [5, 5], scope='conv2')
 428       end_points['pool2'] = slim.max_pool2d(
 429           end_points['conv2'], [2, 2], 2, scope='pool2')
 430       net = slim.flatten(end_points['pool2'])
 431       end_points['fc3'] = slim.fully_connected(net, 128, scope='fc3')
 432       net = slim.dropout(
 433           end_points['fc3'], 0.5, is_training=is_training, scope='dropout')
 434 
 435       with tf.variable_scope('quaternion_prediction'):
 436         predicted_quaternion = slim.fully_connected(
 437             net, 4, activation_fn=tf.nn.tanh)
 438         predicted_quaternion = tf.nn.l2_normalize(predicted_quaternion, 1)
 439       logits = slim.fully_connected(
 440           net, num_classes, activation_fn=None, scope='fc4')
 441   end_points['quaternion_pred'] = predicted_quaternion
 442 
 443   return logits, end_points
