  67 def quaternion_metric(predictions, labels):
  68   params = {'batch_size': FLAGS.batch_size, 'use_logging': False}
  69   logcost = losses.log_quaternion_loss_batch(predictions, labels, params)
  70   return slim.metrics.streaming_mean(logcost)
  71 
  72 
  73 def angle_diff(true_q, pred_q):
  74   angles = 2 * (
  75       180.0 /
  76       np.pi) * np.arccos(np.abs(np.sum(np.multiply(pred_q, true_q), axis=1)))
  77   return angles

  85 def main(_):
  86   g = tf.Graph()
  87   with g.as_default():
  88     # Load the data.
  89     images, labels = provide_batch_fn()(
  90         FLAGS.dataset, FLAGS.split, FLAGS.dataset_dir, 4, FLAGS.batch_size, 4)
  91 
  92     num_classes = labels['classes'].get_shape().as_list()[1]
  93 
  94     tf.summary.image('eval_images', images, max_outputs=3)
  95 
  96     # Define the model:
  97     with tf.variable_scope('towers'):
  98       basic_tower = getattr(models, FLAGS.basic_tower)
  99       predictions, endpoints = basic_tower(
 100           images,
 101           num_classes=num_classes,
 102           is_training=False,
 103           batch_norm_params=None)
 104     metric_names_to_values = {}
 105 
 106     # Define the metrics:
 107     if 'quaternions' in labels:  # Also have to evaluate pose estimation!
 108       quaternion_loss = quaternion_metric(labels['quaternions'],
 109                                           endpoints['quaternion_pred'])
 110 
 111       angle_errors, = tf.py_func(
 112           angle_diff, [labels['quaternions'], endpoints['quaternion_pred']],
 113           [tf.float32])
 114 
 115       metric_names_to_values[
 116           'Angular mean error'] = slim.metrics.streaming_mean(angle_errors)
 117       metric_names_to_values['Quaternion Loss'] = quaternion_loss
 118 
 119     accuracy = tf.contrib.metrics.streaming_accuracy(
 120         tf.argmax(predictions, 1), tf.argmax(labels['classes'], 1))
 121 
 122     predictions = tf.argmax(predictions, 1)
 123     labels = tf.argmax(labels['classes'], 1)
 124     metric_names_to_values['Accuracy'] = accuracy
 125 
 126     if FLAGS.enable_precision_recall:
 127       for i in xrange(num_classes):
 128         index_map = tf.one_hot(i, depth=num_classes)
 129         name = 'PR/Precision_{}'.format(i)
 130         metric_names_to_values[name] = slim.metrics.streaming_precision(
 131             tf.gather(index_map, predictions), tf.gather(index_map, labels))
 132         name = 'PR/Recall_{}'.format(i)
 133         metric_names_to_values[name] = slim.metrics.streaming_recall(
 134             tf.gather(index_map, predictions), tf.gather(index_map, labels))
 135 
 136     names_to_values, names_to_updates = slim.metrics.aggregate_metric_map(
 137         metric_names_to_values)
 138 
 139     # Create the summary ops such that they also print out to std output:
 140     summary_ops = []
 141     for metric_name, metric_value in names_to_values.iteritems():
 142       op = tf.summary.scalar(metric_name, metric_value)
 143       op = tf.Print(op, [metric_value], metric_name)
 144       summary_ops.append(op)
 145 
 146     # This ensures that we make a single pass over all of the data.
 147     num_batches = math.ceil(FLAGS.num_examples / float(FLAGS.batch_size))
 148 
 149     # Setup the global step.
 150     slim.get_or_create_global_step()
 151     slim.evaluation.evaluation_loop(
 152         FLAGS.master,
 153         checkpoint_dir=FLAGS.checkpoint_dir,
 154         logdir=FLAGS.eval_dir,
 155         num_evals=num_batches,
 156         eval_op=names_to_updates.values(),
 157         summary_op=tf.summary.merge(summary_ops))
 158 
 159 
 160 if __name__ == '__main__':
 161   tf.app.run()
