  25 def MaximumMeanDiscrepancySlow(x, y, sigmas):
  26   num_samples = x.get_shape().as_list()[0]
  27 
  28   def AverageGaussianKernel(x, y, sigmas):
  29     result = 0
  30     for sigma in sigmas:
  31       dist = tf.reduce_sum(tf.square(x - y))
  32       result += tf.exp((-1.0 / (2.0 * sigma)) * dist)
  33     return result / num_samples**2
  34 
  35   total = 0
  36 
  37   for i in range(num_samples):
  38     for j in range(num_samples):
  39       total += AverageGaussianKernel(x[i, :], x[j, :], sigmas)
  40       total += AverageGaussianKernel(y[i, :], y[j, :], sigmas)
  41       total += -2 * AverageGaussianKernel(x[i, :], y[j, :], sigmas)
  42 
  43   return total

  48   def test_log_quaternion_loss_batch(self):
  49     with self.test_session():
  50       predictions = tf.random_uniform((10, 4), seed=1)
  51       predictions = tf.nn.l2_normalize(predictions, 1)
  52       labels = tf.random_uniform((10, 4), seed=1)
  53       labels = tf.nn.l2_normalize(labels, 1)
  54       params = {'batch_size': 10, 'use_logging': False}
  55       x = losses.log_quaternion_loss_batch(predictions, labels, params)
  56       self.assertTrue(((10,) == tf.shape(x).eval()).all())

  59 class MaximumMeanDiscrepancyTest(tf.test.TestCase):
  60 
  61   def test_mmd_name(self):
  62     with self.test_session():
  63       x = tf.random_uniform((2, 3), seed=1)
  64       kernel = partial(utils.gaussian_kernel_matrix, sigmas=tf.constant([1.]))
  65       loss = losses.maximum_mean_discrepancy(x, x, kernel)
  66 
  67       self.assertEquals(loss.op.name, 'MaximumMeanDiscrepancy/value')
  68 
  69   def test_mmd_is_zero_when_inputs_are_same(self):
  70     with self.test_session():
  71       x = tf.random_uniform((2, 3), seed=1)
  72       kernel = partial(utils.gaussian_kernel_matrix, sigmas=tf.constant([1.]))
  73       self.assertEquals(0, losses.maximum_mean_discrepancy(x, x, kernel).eval())
  74 
  75   def test_fast_mmd_is_similar_to_slow_mmd(self):
  76     with self.test_session():
  77       x = tf.constant(np.random.normal(size=(2, 3)), tf.float32)
  78       y = tf.constant(np.random.rand(2, 3), tf.float32)
  79 
  80       cost_old = MaximumMeanDiscrepancySlow(x, y, [1.]).eval()
  81       kernel = partial(utils.gaussian_kernel_matrix, sigmas=tf.constant([1.]))
  82       cost_new = losses.maximum_mean_discrepancy(x, y, kernel).eval()
  83 
  84       self.assertAlmostEqual(cost_old, cost_new, delta=1e-5)
  85 
  86   def test_multiple_sigmas(self):
  87     with self.test_session():
  88       x = tf.constant(np.random.normal(size=(2, 3)), tf.float32)
  89       y = tf.constant(np.random.rand(2, 3), tf.float32)
  90 
  91       sigmas = tf.constant([2., 5., 10, 20, 30])
  92       kernel = partial(utils.gaussian_kernel_matrix, sigmas=sigmas)
  93       cost_old = MaximumMeanDiscrepancySlow(x, y, [2., 5., 10, 20, 30]).eval()
  94       cost_new = losses.maximum_mean_discrepancy(x, y, kernel=kernel).eval()
  95 
  96       self.assertAlmostEqual(cost_old, cost_new, delta=1e-5)
  97 
  98   def test_mmd_is_zero_when_distributions_are_same(self):
  99 
 100     with self.test_session():
 101       x = tf.random_uniform((1000, 10), seed=1)
 102       y = tf.random_uniform((1000, 10), seed=3)
 103 
 104       kernel = partial(utils.gaussian_kernel_matrix, sigmas=tf.constant([100.]))
 105       loss = losses.maximum_mean_discrepancy(x, y, kernel=kernel).eval()
 106 
 107       self.assertAlmostEqual(0, loss, delta=1e-4)
