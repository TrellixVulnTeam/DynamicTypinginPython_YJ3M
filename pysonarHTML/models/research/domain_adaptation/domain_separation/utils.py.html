  21 def create_summaries(end_points, prefix='', max_images=3, use_op_name=False):
  22   """Creates a tf summary per endpoint.
  23 
  24   If the endpoint is a 4 dimensional tensor it displays it as an image
  25   otherwise if it is a two dimensional one it creates a histogram summary.
  26 
  27   Args:
  28     end_points: a dictionary of name, tf tensor pairs.
  29     prefix: an optional string to prefix the summary with.
  30     max_images: the maximum number of images to display per summary.
  31     use_op_name: Use the op name as opposed to the shorter end_points key.
  32   """
  33   for layer_name in end_points:
  34     if use_op_name:
  35       name = end_points[layer_name].op.name
  36     else:
  37       name = layer_name
  38     if len(end_points[layer_name].get_shape().as_list()) == 4:
  39       # if it's an actual image do not attempt to reshape it
  40       if end_points[layer_name].get_shape().as_list()[-1] == 1 or end_points[
  41           layer_name].get_shape().as_list()[-1] == 3:
  42         visualization_image = end_points[layer_name]
  43       else:
  44         visualization_image = reshape_feature_maps(end_points[layer_name])
  45       tf.summary.image(
  46           '{}/{}'.format(prefix, name),
  47           visualization_image,
  48           max_outputs=max_images)
  49     elif len(end_points[layer_name].get_shape().as_list()) == 3:
  50       images = tf.expand_dims(end_points[layer_name], 3)
  51       tf.summary.image(
  52           '{}/{}'.format(prefix, name),
  53           images,
  54           max_outputs=max_images)
  55     elif len(end_points[layer_name].get_shape().as_list()) == 2:
  56       tf.summary.histogram('{}/{}'.format(prefix, name), end_points[layer_name])

  59 def reshape_feature_maps(features_tensor):
  60   """Reshape activations for tf.summary.image visualization.
  61 
  62   Arguments:
  63     features_tensor: a tensor of activations with a square number of feature
  64                      maps, eg 4, 9, 16, etc.
  65   Returns:
  66     A composite image with all the feature maps that can be passed as an
  67     argument to tf.summary.image.
  68   """
  69   assert len(features_tensor.get_shape().as_list()) == 4
  70   num_filters = features_tensor.get_shape().as_list()[-1]
  71   assert num_filters > 0
  72   num_filters_sqrt = math.sqrt(num_filters)
  73   assert num_filters_sqrt.is_integer(
  74   ), 'Number of filters should be a square number but got {}'.format(
  75       num_filters)
  76   num_filters_sqrt = int(num_filters_sqrt)
  77   conv_summary = tf.unstack(features_tensor, axis=3)
  78   conv_one_row = tf.concat(axis=2, values=conv_summary[0:num_filters_sqrt])
  79   ind = 1
  80   conv_final = conv_one_row
  81   for ind in range(1, num_filters_sqrt):
  82     conv_one_row = tf.concat(axis=2,
  83                              values=conv_summary[
  84         ind * num_filters_sqrt + 0:ind * num_filters_sqrt + num_filters_sqrt])
  85     conv_final = tf.concat(
  86         axis=1, values=[tf.squeeze(conv_final), tf.squeeze(conv_one_row)])
  87     conv_final = tf.expand_dims(conv_final, -1)
  88   return conv_final

  91 def accuracy(predictions, labels):
  92   """Calculates the classificaton accuracy.
  93 
  94   Args:
  95     predictions: the predicted values, a tensor whose size matches 'labels'.
  96     labels: the ground truth values, a tensor of any size.
  97 
  98   Returns:
  99     a tensor whose value on evaluation returns the total accuracy.
 100   """
 101   return tf.reduce_mean(tf.cast(tf.equal(predictions, labels), tf.float32))

 104 def compute_upsample_values(input_tensor, upsample_height, upsample_width):
 105   """Compute values for an upsampling op (ops.BatchCropAndResize).
 106 
 107   Args:
 108     input_tensor: image tensor with shape [batch, height, width, in_channels]
 109     upsample_height: integer
 110     upsample_width: integer
 111 
 112   Returns:
 113     grid_centers: tensor with shape [batch, 1]
 114     crop_sizes: tensor with shape [batch, 1]
 115     output_height: integer
 116     output_width: integer
 117   """
 118   batch, input_height, input_width, _ = input_tensor.shape
 119 
 120   height_half = input_height / 2.
 121   width_half = input_width / 2.
 122   grid_centers = tf.constant(batch * [[height_half, width_half]])
 123   crop_sizes = tf.constant(batch * [[input_height, input_width]])
 124   output_height = input_height * upsample_height
 125   output_width = input_width * upsample_width
 126 
 127   return grid_centers, tf.to_float(crop_sizes), output_height, output_width

 130 def compute_pairwise_distances(x, y):
 131   """Computes the squared pairwise Euclidean distances between x and y.
 132 
 133   Args:
 134     x: a tensor of shape [num_x_samples, num_features]
 135     y: a tensor of shape [num_y_samples, num_features]
 136 
 137   Returns:
 138     a distance matrix of dimensions [num_x_samples, num_y_samples].
 139 
 140   Raises:
 141     ValueError: if the inputs do no matched the specified dimensions.
 142   """
 143 
 144   if not len(x.get_shape()) == len(y.get_shape()) == 2:
 145     raise ValueError('Both inputs should be matrices.')
 146 
 147   if x.get_shape().as_list()[1] != y.get_shape().as_list()[1]:
 148     raise ValueError('The number of features should be the same.')
 149 
 150   norm = lambda x: tf.reduce_sum(tf.square(x), 1)
 151 
 152   # By making the `inner' dimensions of the two matrices equal to 1 using
 153   # broadcasting then we are essentially substracting every pair of rows
 154   # of x and y.
 155   # x will be num_samples x num_features x 1,
 156   # and y will be 1 x num_features x num_samples (after broadcasting).
 157   # After the substraction we will get a
 158   # num_x_samples x num_features x num_y_samples matrix.
 159   # The resulting dist will be of shape num_y_samples x num_x_samples.
 160   # and thus we need to transpose it again.
 161   return tf.transpose(norm(tf.expand_dims(x, 2) - tf.transpose(y)))

 164 def gaussian_kernel_matrix(x, y, sigmas):
 165   r"""Computes a Guassian Radial Basis Kernel between the samples of x and y.
 166 
 167   We create a sum of multiple gaussian kernels each having a width sigma_i.
 168 
 169   Args:
 170     x: a tensor of shape [num_samples, num_features]
 171     y: a tensor of shape [num_samples, num_features]
 172     sigmas: a tensor of floats which denote the widths of each of the
 173       gaussians in the kernel.
 174   Returns:
 175     A tensor of shape [num_samples{x}, num_samples{y}] with the RBF kernel.
 176   """
 177   beta = 1. / (2. * (tf.expand_dims(sigmas, 1)))
 178 
 179   dist = compute_pairwise_distances(x, y)
 180 
 181   s = tf.matmul(beta, tf.reshape(dist, (1, -1)))
 182 
 183   return tf.reshape(tf.reduce_sum(tf.exp(-s), 0), tf.shape(dist))
