 140 def provide_batch_fn():
 141   """ The provide_batch function to use. """
 142   return dataset_factory.provide_batch
 143 
 144 def main(_):
 145   model_params = {
 146       'use_separation': FLAGS.use_separation,
 147       'domain_separation_startpoint': FLAGS.domain_separation_startpoint,
 148       'layers_to_regularize': FLAGS.layers_to_regularize,
 149       'alpha_weight': FLAGS.alpha_weight,
 150       'beta_weight': FLAGS.beta_weight,
 151       'gamma_weight': FLAGS.gamma_weight,
 152       'pose_weight': FLAGS.pose_weight,
 153       'recon_loss_name': FLAGS.recon_loss_name,
 154       'decoder_name': FLAGS.decoder_name,
 155       'encoder_name': FLAGS.encoder_name,
 156       'weight_decay': FLAGS.weight_decay,
 157       'batch_size': FLAGS.batch_size,
 158       'use_logging': FLAGS.use_logging,
 159       'ps_tasks': FLAGS.ps_tasks,
 160       'task': FLAGS.task,
 161   }
 162   g = tf.Graph()
 163   with g.as_default():
 164     with tf.device(tf.train.replica_device_setter(FLAGS.ps_tasks)):
 165       # Load the data.
 166       source_images, source_labels = provide_batch_fn()(
 167           FLAGS.source_dataset, 'train', FLAGS.dataset_dir, FLAGS.num_readers,
 168           FLAGS.batch_size, FLAGS.num_preprocessing_threads)
 169       target_images, target_labels = provide_batch_fn()(
 170           FLAGS.target_dataset, 'train', FLAGS.dataset_dir, FLAGS.num_readers,
 171           FLAGS.batch_size, FLAGS.num_preprocessing_threads)
 172 
 173       # In the unsupervised case all the samples in the labeled
 174       # domain are from the source domain.
 175       domain_selection_mask = tf.fill((source_images.get_shape().as_list()[0],),
 176                                       True)
 177 
 178       # When using the semisupervised model we include labeled target data in
 179       # the source labelled data.
 180       if FLAGS.target_labeled_dataset != 'none':
 181         # 1000 is the maximum number of labelled target samples that exists in
 182         # the datasets.
 183         target_semi_images, target_semi_labels = provide_batch_fn()(
 184             FLAGS.target_labeled_dataset, 'train', FLAGS.batch_size)
 185 
 186         # Calculate the proportion of source domain samples in the semi-
 187         # supervised setting, so that the proportion is set accordingly in the
 188         # batches.
 189         proportion = float(source_labels['num_train_samples']) / (
 190             source_labels['num_train_samples'] +
 191             target_semi_labels['num_train_samples'])
 192 
 193         rnd_tensor = tf.random_uniform(
 194             (target_semi_images.get_shape().as_list()[0],))
 195 
 196         domain_selection_mask = rnd_tensor < proportion
 197         source_images = tf.where(domain_selection_mask, source_images,
 198                                  target_semi_images)
 199         source_class_labels = tf.where(domain_selection_mask,
 200                                        source_labels['classes'],
 201                                        target_semi_labels['classes'])
 202 
 203         if 'quaternions' in source_labels:
 204           source_pose_labels = tf.where(domain_selection_mask,
 205                                         source_labels['quaternions'],
 206                                         target_semi_labels['quaternions'])
 207           (source_images, source_class_labels, source_pose_labels,
 208            domain_selection_mask) = tf.train.shuffle_batch(
 209                [
 210                    source_images, source_class_labels, source_pose_labels,
 211                    domain_selection_mask
 212                ],
 213                FLAGS.batch_size,
 214                50000,
 215                5000,
 216                num_threads=1,
 217                enqueue_many=True)
 218 
 219         else:
 220           (source_images, source_class_labels,
 221            domain_selection_mask) = tf.train.shuffle_batch(
 222                [source_images, source_class_labels, domain_selection_mask],
 223                FLAGS.batch_size,
 224                50000,
 225                5000,
 226                num_threads=1,
 227                enqueue_many=True)
 228         source_labels = {}
 229         source_labels['classes'] = source_class_labels
 230         if 'quaternions' in source_labels:
 231           source_labels['quaternions'] = source_pose_labels
 232 
 233       slim.get_or_create_global_step()
 234       tf.summary.image('source_images', source_images, max_outputs=3)
 235       tf.summary.image('target_images', target_images, max_outputs=3)
 236 
 237       dsn.create_model(
 238           source_images,
 239           source_labels,
 240           domain_selection_mask,
 241           target_images,
 242           target_labels,
 243           FLAGS.similarity_loss,
 244           model_params,
 245           basic_tower_name=FLAGS.basic_tower)
 246 
 247       # Configure the optimization scheme:
 248       learning_rate = tf.train.exponential_decay(
 249           FLAGS.learning_rate,
 250           slim.get_or_create_global_step(),
 251           FLAGS.decay_steps,
 252           FLAGS.decay_rate,
 253           staircase=True,
 254           name='learning_rate')
 255 
 256       tf.summary.scalar('learning_rate', learning_rate)
 257       tf.summary.scalar('total_loss', tf.losses.get_total_loss())
 258 
 259       opt = tf.train.MomentumOptimizer(learning_rate, FLAGS.momentum)
 260       tf.logging.set_verbosity(tf.logging.INFO)
 261       # Run training.
 262       loss_tensor = slim.learning.create_train_op(
 263           slim.losses.get_total_loss(),
 264           opt,
 265           summarize_gradients=True,
 266           colocate_gradients_with_ops=True)
 267       slim.learning.train(
 268           train_op=loss_tensor,
 269           logdir=FLAGS.train_log_dir,
 270           master=FLAGS.master,
 271           is_chief=FLAGS.task == 0,
 272           number_of_steps=FLAGS.max_number_of_steps,
 273           save_summaries_secs=FLAGS.save_summaries_secs,
 274           save_interval_secs=FLAGS.save_interval_secs)
 275 
 276 
 277 if __name__ == '__main__':
 278   tf.app.run()
