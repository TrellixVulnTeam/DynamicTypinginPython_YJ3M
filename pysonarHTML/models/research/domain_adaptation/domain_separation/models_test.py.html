  24 class SharedEncodersTest(tf.test.TestCase):
  25 
  26   def _testSharedEncoder(self,
  27                          input_shape=[5, 28, 28, 1],
  28                          model=models.dann_mnist,
  29                          is_training=True):
  30     images = tf.to_float(np.random.rand(*input_shape))
  31 
  32     with self.test_session() as sess:
  33       logits, _ = model(images)
  34       sess.run(tf.global_variables_initializer())
  35       logits_np = sess.run(logits)
  36     return logits_np
  37 
  38   def testBuildGRLMnistModel(self):
  39     logits = self._testSharedEncoder(model=getattr(models,
  40                                                    'dann_mnist'))
  41     self.assertEqual(logits.shape, (5, 10))
  42     self.assertTrue(np.any(logits))
  43 
  44   def testBuildGRLSvhnModel(self):
  45     logits = self._testSharedEncoder(model=getattr(models,
  46                                                    'dann_svhn'))
  47     self.assertEqual(logits.shape, (5, 10))
  48     self.assertTrue(np.any(logits))
  49 
  50   def testBuildGRLGtsrbModel(self):
  51     logits = self._testSharedEncoder([5, 40, 40, 3],
  52                                      getattr(models, 'dann_gtsrb'))

  54     self.assertTrue(np.any(logits))
  55 
  56   def testBuildPoseModel(self):
  57     logits = self._testSharedEncoder([5, 64, 64, 4],
  58                                      getattr(models, 'dsn_cropped_linemod'))
  59     self.assertEqual(logits.shape, (5, 11))
  60     self.assertTrue(np.any(logits))
  61 
  62   def testBuildPoseModelWithBatchNorm(self):
  63     images = tf.to_float(np.random.rand(10, 64, 64, 4))
  64 
  65     with self.test_session() as sess:
  66       logits, _ = getattr(models, 'dsn_cropped_linemod')(
  67           images, batch_norm_params=models.default_batch_norm_params(True))
  68       sess.run(tf.global_variables_initializer())
  69       logits_np = sess.run(logits)
  70     self.assertEqual(logits_np.shape, (10, 11))
  71     self.assertTrue(np.any(logits_np))

  74 class EncoderTest(tf.test.TestCase):
  75 
  76   def _testEncoder(self, batch_norm_params=None, channels=1):
  77     images = tf.to_float(np.random.rand(10, 28, 28, channels))
  78 
  79     with self.test_session() as sess:
  80       end_points = models.default_encoder(
  81           images, 128, batch_norm_params=batch_norm_params)
  82       sess.run(tf.global_variables_initializer())
  83       private_code = sess.run(end_points['fc3'])
  84     self.assertEqual(private_code.shape, (10, 128))
  85     self.assertTrue(np.any(private_code))
  86     self.assertTrue(np.all(np.isfinite(private_code)))
  87 
  88   def testEncoder(self):
  89     self._testEncoder()
  90 
  91   def testEncoderMultiChannel(self):
  92     self._testEncoder(None, 4)
  93 
  94   def testEncoderIsTrainingBatchNorm(self):
  95     self._testEncoder(models.default_batch_norm_params(True))
  96 
  97   def testEncoderBatchNorm(self):
  98     self._testEncoder(models.default_batch_norm_params(False))
