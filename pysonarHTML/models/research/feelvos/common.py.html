 109 class VideoModelOptions(common.ModelOptions):
 110   """Internal version of immutable class to hold model options."""
 111 
 112   def __new__(cls,
 113               outputs_to_num_classes,
 114               crop_size=None,
 115               atrous_rates=None,
 116               output_stride=8):
 117     """Constructor to set default values.
 118 
 119     Args:
 120       outputs_to_num_classes: A dictionary from output type to the number of
 121         classes. For example, for the task of semantic segmentation with 21
 122         semantic classes, we would have outputs_to_num_classes['semantic'] = 21.
 123       crop_size: A tuple [crop_height, crop_width].
 124       atrous_rates: A list of atrous convolution rates for ASPP.
 125       output_stride: The ratio of input to output spatial resolution.
 126 
 127     Returns:
 128       A new VideoModelOptions instance.
 129     """
 130     self = super(VideoModelOptions, cls).__new__(
 131         cls,
 132         outputs_to_num_classes,
 133         crop_size,
 134         atrous_rates,
 135         output_stride)
 136     # Add internal flags.
 137     self.classification_loss = FLAGS.classification_loss
 138 
 139     return self

 142 def parse_decoder_output_stride():
 143   """Parses decoder output stride.
 144 
 145   FEELVOS assumes decoder_output_stride = 4. Thus, this function is created for
 146   this particular purpose.
 147 
 148   Returns:
 149     An integer specifying the decoder_output_stride.
 150 
 151   Raises:
 152     ValueError: If decoder_output_stride is None or contains more than one
 153       element.
 154   """
 155   if FLAGS.decoder_output_stride:
 156     decoder_output_stride = [
 157         int(x) for x in FLAGS.decoder_output_stride]
 158     if len(decoder_output_stride) != 1:
 159       raise ValueError('Expect decoder output stride has only one element.')
 160     decoder_output_stride = decoder_output_stride[0]
 161   else:
 162     raise ValueError('Expect flag decoder output stride not to be None.')
 163   return decoder_output_stride
