  31 def preprocess_image_and_label(image,
  32                                label,
  33                                crop_height,
  34                                crop_width,
  35                                min_resize_value=None,
  36                                max_resize_value=None,
  37                                resize_factor=None,
  38                                min_scale_factor=1.,
  39                                max_scale_factor=1.,
  40                                scale_factor_step_size=0,
  41                                ignore_label=255,
  42                                is_training=True,
  43                                model_variant=None):
  44   """Preprocesses the image and label.
  45 
  46   Args:
  47     image: Input image.
  48     label: Ground truth annotation label.
  49     crop_height: The height value used to crop the image and label.
  50     crop_width: The width value used to crop the image and label.
  51     min_resize_value: Desired size of the smaller image side.
  52     max_resize_value: Maximum allowed size of the larger image side.
  53     resize_factor: Resized dimensions are multiple of factor plus one.
  54     min_scale_factor: Minimum scale factor value.
  55     max_scale_factor: Maximum scale factor value.
  56     scale_factor_step_size: The step size from min scale factor to max scale
  57       factor. The input is randomly scaled based on the value of
  58       (min_scale_factor, max_scale_factor, scale_factor_step_size).
  59     ignore_label: The label value which will be ignored for training and
  60       evaluation.
  61     is_training: If the preprocessing is used for training or not.
  62     model_variant: Model variant (string) for choosing how to mean-subtract the
  63       images. See feature_extractor.network_map for supported model variants.
  64 
  65   Returns:
  66     original_image: Original image (could be resized).
  67     processed_image: Preprocessed image.
  68     label: Preprocessed ground truth segmentation label.
  69 
  70   Raises:
  71     ValueError: Ground truth label not provided during training.
  72   """
  73   if is_training and label is None:
  74     raise ValueError('During training, label must be provided.')
  75   if model_variant is None:
  76     tf.logging.warning('Default mean-subtraction is performed. Please specify '
  77                        'a model_variant. See feature_extractor.network_map for '
  78                        'supported model variants.')
  79 
  80   # Keep reference to original image.
  81   original_image = image
  82 
  83   processed_image = tf.cast(image, tf.float32)
  84 
  85   if label is not None:
  86     label = tf.cast(label, tf.int32)
  87 
  88   # Resize image and label to the desired range.
  89   if min_resize_value is not None or max_resize_value is not None:
  90     [processed_image, label] = (
  91         preprocess_utils.resize_to_range(
  92             image=processed_image,
  93             label=label,
  94             min_size=min_resize_value,
  95             max_size=max_resize_value,
  96             factor=resize_factor,
  97             align_corners=True))
  98     # The `original_image` becomes the resized image.
  99     original_image = tf.identity(processed_image)
 100 
 101   # Data augmentation by randomly scaling the inputs.
 102   scale = get_random_scale(
 103       min_scale_factor, max_scale_factor, scale_factor_step_size)
 104   processed_image, label = randomly_scale_image_and_label(
 105       processed_image, label, scale)
 106 
 107   processed_image.set_shape([None, None, 3])
 108 
 109   if crop_height is not None and crop_width is not None:
 110     # Pad image and label to have dimensions >= [crop_height, crop_width].
 111     image_shape = tf.shape(processed_image)
 112     image_height = image_shape[0]
 113     image_width = image_shape[1]
 114 
 115     target_height = image_height + tf.maximum(crop_height - image_height, 0)
 116     target_width = image_width + tf.maximum(crop_width - image_width, 0)
 117 
 118     # Pad image with mean pixel value.
 119     mean_pixel = tf.reshape(
 120         feature_extractor.mean_pixel(model_variant), [1, 1, 3])
 121     processed_image = preprocess_utils.pad_to_bounding_box(
 122         processed_image, 0, 0, target_height, target_width, mean_pixel)
 123 
 124     if label is not None:
 125       label = preprocess_utils.pad_to_bounding_box(
 126           label, 0, 0, target_height, target_width, ignore_label)
 127 
 128     # Randomly crop the image and label.
 129     if is_training and label is not None:
 130       processed_image, label = preprocess_utils.random_crop(
 131           [processed_image, label], crop_height, crop_width)
 132 
 133     processed_image.set_shape([crop_height, crop_width, 3])
 134 
 135     if label is not None:
 136       label.set_shape([crop_height, crop_width, 1])
 137 
 138   if is_training:
 139     # Randomly left-right flip the image and label.
 140     processed_image, label, _ = preprocess_utils.flip_dim(
 141         [processed_image, label], _PROB_OF_FLIP, dim=1)
 142 
 143   return original_image, processed_image, label

 146 def preprocess_images_and_labels_consistently(images,
 147                                               labels,
 148                                               crop_height,
 149                                               crop_width,
 150                                               min_resize_value=None,
 151                                               max_resize_value=None,
 152                                               resize_factor=None,
 153                                               min_scale_factor=1.,
 154                                               max_scale_factor=1.,
 155                                               scale_factor_step_size=0,
 156                                               ignore_label=255,
 157                                               is_training=True,
 158                                               model_variant=None):
 159   """Preprocesses images and labels in a consistent way.
 160 
 161   Similar to preprocess_image_and_label, but works on a list of images
 162   and a list of labels and uses the same crop coordinates and either flips
 163   all images and labels or none of them.
 164 
 165   Args:
 166     images: List of input images.
 167     labels: List of ground truth annotation labels.
 168     crop_height: The height value used to crop the image and label.
 169     crop_width: The width value used to crop the image and label.
 170     min_resize_value: Desired size of the smaller image side.
 171     max_resize_value: Maximum allowed size of the larger image side.
 172     resize_factor: Resized dimensions are multiple of factor plus one.
 173     min_scale_factor: Minimum scale factor value.
 174     max_scale_factor: Maximum scale factor value.
 175     scale_factor_step_size: The step size from min scale factor to max scale
 176       factor. The input is randomly scaled based on the value of
 177       (min_scale_factor, max_scale_factor, scale_factor_step_size).
 178     ignore_label: The label value which will be ignored for training and
 179       evaluation.
 180     is_training: If the preprocessing is used for training or not.
 181     model_variant: Model variant (string) for choosing how to mean-subtract the
 182       images. See feature_extractor.network_map for supported model variants.
 183 
 184   Returns:
 185     original_images: Original images (could be resized).
 186     processed_images: Preprocessed images.
 187     labels: Preprocessed ground truth segmentation labels.
 188 
 189   Raises:
 190     ValueError: Ground truth label not provided during training.
 191   """
 192   if is_training and labels is None:
 193     raise ValueError('During training, labels must be provided.')
 194   if model_variant is None:
 195     tf.logging.warning('Default mean-subtraction is performed. Please specify '
 196                        'a model_variant. See feature_extractor.network_map for '
 197                        'supported model variants.')
 198   if labels is not None:
 199     assert len(images) == len(labels)
 200   num_imgs = len(images)
 201 
 202   # Keep reference to original images.
 203   original_images = images
 204 
 205   processed_images = [tf.cast(image, tf.float32) for image in images]
 206 
 207   if labels is not None:
 208     labels = [tf.cast(label, tf.int32) for label in labels]
 209 
 210   # Resize images and labels to the desired range.
 211   if min_resize_value is not None or max_resize_value is not None:
 212     processed_images, labels = zip(*[
 213         preprocess_utils.resize_to_range(
 214             image=processed_image,
 215             label=label,
 216             min_size=min_resize_value,
 217             max_size=max_resize_value,
 218             factor=resize_factor,
 219             align_corners=True) for processed_image, label
 220         in zip(processed_images, labels)])
 221     # The `original_images` becomes the resized images.
 222     original_images = [tf.identity(processed_image)
 223                        for processed_image in processed_images]
 224 
 225   # Data augmentation by randomly scaling the inputs.
 226   scale = get_random_scale(
 227       min_scale_factor, max_scale_factor, scale_factor_step_size)
 228   processed_images, labels = zip(
 229       *[randomly_scale_image_and_label(processed_image, label, scale)
 230         for processed_image, label in zip(processed_images, labels)])
 231 
 232   for processed_image in processed_images:
 233     processed_image.set_shape([None, None, 3])
 234 
 235   if crop_height is not None and crop_width is not None:
 236     # Pad image and label to have dimensions >= [crop_height, crop_width].
 237     image_shape = tf.shape(processed_images[0])
 238     image_height = image_shape[0]
 239     image_width = image_shape[1]
 240 
 241     target_height = image_height + tf.maximum(crop_height - image_height, 0)
 242     target_width = image_width + tf.maximum(crop_width - image_width, 0)
 243 
 244     # Pad image with mean pixel value.
 245     mean_pixel = tf.reshape(
 246         feature_extractor.mean_pixel(model_variant), [1, 1, 3])
 247     processed_images = [preprocess_utils.pad_to_bounding_box(
 248         processed_image, 0, 0, target_height, target_width, mean_pixel)
 249                         for processed_image in processed_images]
 250 
 251     if labels is not None:
 252       labels = [preprocess_utils.pad_to_bounding_box(
 253           label, 0, 0, target_height, target_width, ignore_label)
 254                 for label in labels]
 255 
 256     # Randomly crop the images and labels.
 257     if is_training and labels is not None:
 258       cropped = preprocess_utils.random_crop(
 259           processed_images + labels, crop_height, crop_width)
 260       assert len(cropped) == 2 * num_imgs
 261       processed_images = cropped[:num_imgs]
 262       labels = cropped[num_imgs:]
 263 
 264     for processed_image in processed_images:
 265       processed_image.set_shape([crop_height, crop_width, 3])
 266 
 267     if labels is not None:
 268       for label in labels:
 269         label.set_shape([crop_height, crop_width, 1])
 270 
 271   if is_training:
 272     # Randomly left-right flip the image and label.
 273     res = preprocess_utils.flip_dim(
 274         list(processed_images + labels), _PROB_OF_FLIP, dim=1)
 275     maybe_flipped = res[:-1]
 276     assert len(maybe_flipped) == 2 * num_imgs
 277     processed_images = maybe_flipped[:num_imgs]
 278     labels = maybe_flipped[num_imgs:]
 279 
 280   return original_images, processed_images, labels
