  92 def _process_seq_data(segmentation_dir, embeddings_dir, seq_name,
  93                       predicted_labels, gt_labels, embeddings):
  94   """Calculates the sequence IoU and optionally save the segmentation masks.
  95 
  96   Args:
  97     segmentation_dir: Directory in which the segmentation results are stored.
  98     embeddings_dir: Directory in which the embeddings are stored.
  99     seq_name: String, the name of the sequence.
 100     predicted_labels: Int64 np.array of shape [n_frames, height, width].
 101     gt_labels: Ground truth labels, Int64 np.array of shape
 102       [n_frames, height, width].
 103     embeddings: Float32 np.array of embeddings of shape
 104       [n_frames, decoder_height, decoder_width, embedding_dim], or None.
 105 
 106   Returns:
 107     The IoU for the sequence (float).
 108   """
 109   sequence_dir = os.path.join(segmentation_dir, seq_name)
 110   tf.gfile.MakeDirs(sequence_dir)
 111   embeddings_seq_dir = os.path.join(embeddings_dir, seq_name)
 112   tf.gfile.MakeDirs(embeddings_seq_dir)
 113   label_set = np.unique(gt_labels[0])
 114   ious = []
 115   assert len(predicted_labels) == len(gt_labels)
 116   if embeddings is not None:
 117     assert len(predicted_labels) == len(embeddings)
 118   for t, (predicted_label, gt_label) in enumerate(
 119       zip(predicted_labels, gt_labels)):
 120     if FLAGS.save_segmentations:
 121       seg_filename = os.path.join(segmentation_dir, seq_name, '%05d.png' % t)
 122       eval_utils.save_segmentation_with_colormap(seg_filename, predicted_label)
 123     if FLAGS.save_embeddings:
 124       embedding_filename = os.path.join(embeddings_dir, seq_name,
 125                                         '%05d.npy' % t)
 126       assert embeddings is not None
 127       eval_utils.save_embeddings(embedding_filename, embeddings[t])
 128     object_ious_t = eval_utils.calculate_multi_object_ious(
 129         predicted_label, gt_label, label_set)
 130     ious.append(object_ious_t)
 131   # First and last frame are excluded in DAVIS eval.
 132   seq_ious = np.mean(ious[1:-1], axis=0)
 133   tf.logging.info('seq ious: %s %s', seq_name, seq_ious)
 134   return seq_ious

 137 def create_predictions(samples, reference_labels, first_frame_img,
 138                        model_options):
 139   """Predicts segmentation labels for each frame of the video.
 140 
 141   Slower version than create_predictions_fast, but does support more options.
 142 
 143   Args:
 144     samples: Dictionary of input samples.
 145     reference_labels: Int tensor of shape [1, height, width, 1].
 146     first_frame_img: Float32 tensor of shape [height, width, 3].
 147     model_options: An InternalModelOptions instance to configure models.
 148 
 149   Returns:
 150     predicted_labels: Int tensor of shape [time, height, width] of
 151       predicted labels for each frame.
 152     all_embeddings: Float32 tensor of shape
 153       [time, height, width, embedding_dim], or None.
 154   """
 155 
 156   def predict(args, imgs):
 157     """Predicts segmentation labels and softmax probabilities for each image.
 158 
 159     Args:
 160       args: A tuple of (predictions, softmax_probabilities), where predictions
 161         is an int tensor of shape [1, h, w] and softmax_probabilities is a
 162         float32 tensor of shape [1, h_decoder, w_decoder, n_objects].
 163       imgs: Either a one-tuple of the image to predict labels for of shape
 164         [h, w, 3], or pair of previous frame and current frame image.
 165 
 166     Returns:
 167       predictions: The predicted labels as int tensor of shape [1, h, w].
 168       softmax_probabilities: The softmax probabilities of shape
 169         [1, h_decoder, w_decoder, n_objects].
 170     """
 171     if FLAGS.save_embeddings:
 172       last_frame_predictions, last_softmax_probabilities, _ = args
 173     else:
 174       last_frame_predictions, last_softmax_probabilities = args
 175 
 176     if FLAGS.also_attend_to_previous_frame or FLAGS.use_softmax_feedback:
 177       ref_labels_to_use = tf.concat(
 178           [reference_labels, last_frame_predictions[..., tf.newaxis]],
 179           axis=0)
 180     else:
 181       ref_labels_to_use = reference_labels
 182 
 183     predictions, softmax_probabilities = model.predict_labels(
 184         tf.stack((first_frame_img,) + imgs),
 185         model_options=model_options,
 186         image_pyramid=FLAGS.image_pyramid,
 187         embedding_dimension=FLAGS.embedding_dimension,
 188         reference_labels=ref_labels_to_use,
 189         k_nearest_neighbors=FLAGS.k_nearest_neighbors,
 190         use_softmax_feedback=FLAGS.use_softmax_feedback,
 191         initial_softmax_feedback=last_softmax_probabilities,
 192         embedding_seg_feature_dimension=
 193         FLAGS.embedding_seg_feature_dimension,
 194         embedding_seg_n_layers=FLAGS.embedding_seg_n_layers,
 195         embedding_seg_kernel_size=FLAGS.embedding_seg_kernel_size,
 196         embedding_seg_atrous_rates=FLAGS.embedding_seg_atrous_rates,
 197         also_return_softmax_probabilities=True,
 198         num_frames_per_video=
 199         (3 if FLAGS.also_attend_to_previous_frame or
 200          FLAGS.use_softmax_feedback else 2),
 201         normalize_nearest_neighbor_distances=
 202         FLAGS.normalize_nearest_neighbor_distances,
 203         also_attend_to_previous_frame=FLAGS.also_attend_to_previous_frame,
 204         use_local_previous_frame_attention=
 205         FLAGS.use_local_previous_frame_attention,
 206         previous_frame_attention_window_size=
 207         FLAGS.previous_frame_attention_window_size,
 208         use_first_frame_matching=FLAGS.use_first_frame_matching
 209     )
 210     predictions = tf.cast(predictions[common.OUTPUT_TYPE], tf.int32)
 211 
 212     if FLAGS.save_embeddings:
 213       names = [n.name for n in tf.get_default_graph().as_graph_def().node]
 214       embedding_names = [x for x in names if 'embeddings' in x]
 215       # This will crash when multi-scale inference is used.
 216       assert len(embedding_names) == 1, len(embedding_names)
 217       embedding_name = embedding_names[0] + ':0'
 218       embeddings = tf.get_default_graph().get_tensor_by_name(embedding_name)
 219       return predictions, softmax_probabilities, embeddings
 220     else:
 221       return predictions, softmax_probabilities
 222 
 223   init_labels = tf.squeeze(reference_labels, axis=-1)
 224   init_softmax = embedding_utils.create_initial_softmax_from_labels(
 225       reference_labels, reference_labels, common.parse_decoder_output_stride(),
 226       reduce_labels=False)
 227   if FLAGS.save_embeddings:
 228     decoder_height = tf.shape(init_softmax)[1]
 229     decoder_width = tf.shape(init_softmax)[2]
 230     n_frames = (3 if FLAGS.also_attend_to_previous_frame
 231                 or FLAGS.use_softmax_feedback else 2)
 232     embeddings_init = tf.zeros((n_frames, decoder_height, decoder_width,
 233                                 FLAGS.embedding_dimension))
 234     init = (init_labels, init_softmax, embeddings_init)
 235   else:
 236     init = (init_labels, init_softmax)
 237   # Do not eval the first frame again but concat the first frame ground
 238   # truth instead.
 239   if FLAGS.also_attend_to_previous_frame or FLAGS.use_softmax_feedback:
 240     elems = (samples[common.IMAGE][:-1], samples[common.IMAGE][1:])
 241   else:
 242     elems = (samples[common.IMAGE][1:],)
 243   res = tf.scan(predict, elems,
 244                 initializer=init,
 245                 parallel_iterations=1,
 246                 swap_memory=True)
 247   if FLAGS.save_embeddings:
 248     predicted_labels, _, all_embeddings = res
 249     first_frame_embeddings = all_embeddings[0, 0, tf.newaxis]
 250     other_frame_embeddings = all_embeddings[:, -1]
 251     all_embeddings = tf.concat(
 252         [first_frame_embeddings, other_frame_embeddings], axis=0)
 253   else:
 254     predicted_labels, _ = res
 255     all_embeddings = None
 256   predicted_labels = tf.concat([reference_labels[..., 0],
 257                                 tf.squeeze(predicted_labels, axis=1)],
 258                                axis=0)
 259   return predicted_labels, all_embeddings

 262 def create_predictions_fast(samples, reference_labels, first_frame_img,
 263                             model_options):
 264   """Predicts segmentation labels for each frame of the video.
 265 
 266   Faster version than create_predictions, but does not support all options.
 267 
 268   Args:
 269     samples: Dictionary of input samples.
 270     reference_labels: Int tensor of shape [1, height, width, 1].
 271     first_frame_img: Float32 tensor of shape [height, width, 3].
 272     model_options: An InternalModelOptions instance to configure models.
 273 
 274   Returns:
 275     predicted_labels: Int tensor of shape [time, height, width] of
 276       predicted labels for each frame.
 277     all_embeddings: Float32 tensor of shape
 278       [time, height, width, embedding_dim], or None.
 279 
 280   Raises:
 281     ValueError: If FLAGS.save_embeddings is True, FLAGS.use_softmax_feedback is
 282       False, or FLAGS.also_attend_to_previous_frame is False.
 283   """
 284   if FLAGS.save_embeddings:
 285     raise ValueError('save_embeddings does not work with '
 286                      'create_predictions_fast. Use the slower '
 287                      'create_predictions instead.')
 288   if not FLAGS.use_softmax_feedback:
 289     raise ValueError('use_softmax_feedback must be True for '
 290                      'create_predictions_fast. Use the slower '
 291                      'create_predictions instead.')
 292   if not FLAGS.also_attend_to_previous_frame:
 293     raise ValueError('also_attend_to_previous_frame must be True for '
 294                      'create_predictions_fast. Use the slower '
 295                      'create_predictions instead.')
 296   # Extract embeddings for first frame and prepare initial predictions.
 297   first_frame_embeddings = embedding_utils.get_embeddings(
 298       first_frame_img[tf.newaxis], model_options, FLAGS.embedding_dimension)
 299   init_labels = tf.squeeze(reference_labels, axis=-1)
 300   init_softmax = embedding_utils.create_initial_softmax_from_labels(
 301       reference_labels, reference_labels, common.parse_decoder_output_stride(),
 302       reduce_labels=False)
 303   init = (init_labels, init_softmax, first_frame_embeddings)
 304 
 305   def predict(args, img):
 306     """Predicts segmentation labels and softmax probabilities for each image.
 307 
 308     Args:
 309       args: tuple of
 310         (predictions, softmax_probabilities, last_frame_embeddings), where
 311         predictions is an int tensor of shape [1, h, w],
 312         softmax_probabilities is a float32 tensor of shape
 313         [1, h_decoder, w_decoder, n_objects],
 314         and last_frame_embeddings is a float32 tensor of shape
 315         [h_decoder, w_decoder, embedding_dimension].
 316       img: Image to predict labels for of shape [h, w, 3].
 317 
 318     Returns:
 319       predictions: The predicted labels as int tensor of shape [1, h, w].
 320       softmax_probabilities: The softmax probabilities of shape
 321         [1, h_decoder, w_decoder, n_objects].
 322     """
 323     (last_frame_predictions, last_softmax_probabilities,
 324      prev_frame_embeddings) = args
 325     ref_labels_to_use = tf.concat(
 326         [reference_labels, last_frame_predictions[..., tf.newaxis]],
 327         axis=0)
 328 
 329     predictions, softmax_probabilities, embeddings = model.predict_labels(
 330         img[tf.newaxis],
 331         model_options=model_options,
 332         image_pyramid=FLAGS.image_pyramid,
 333         embedding_dimension=FLAGS.embedding_dimension,
 334         reference_labels=ref_labels_to_use,
 335         k_nearest_neighbors=FLAGS.k_nearest_neighbors,
 336         use_softmax_feedback=FLAGS.use_softmax_feedback,
 337         initial_softmax_feedback=last_softmax_probabilities,
 338         embedding_seg_feature_dimension=
 339         FLAGS.embedding_seg_feature_dimension,
 340         embedding_seg_n_layers=FLAGS.embedding_seg_n_layers,
 341         embedding_seg_kernel_size=FLAGS.embedding_seg_kernel_size,
 342         embedding_seg_atrous_rates=FLAGS.embedding_seg_atrous_rates,
 343         also_return_softmax_probabilities=True,
 344         num_frames_per_video=1,
 345         normalize_nearest_neighbor_distances=
 346         FLAGS.normalize_nearest_neighbor_distances,
 347         also_attend_to_previous_frame=FLAGS.also_attend_to_previous_frame,
 348         use_local_previous_frame_attention=
 349         FLAGS.use_local_previous_frame_attention,
 350         previous_frame_attention_window_size=
 351         FLAGS.previous_frame_attention_window_size,
 352         use_first_frame_matching=FLAGS.use_first_frame_matching,
 353         also_return_embeddings=True,
 354         ref_embeddings=(first_frame_embeddings, prev_frame_embeddings)
 355     )
 356     predictions = tf.cast(predictions[common.OUTPUT_TYPE], tf.int32)
 357     return predictions, softmax_probabilities, embeddings
 358 
 359   # Do not eval the first frame again but concat the first frame ground
 360   # truth instead.
 361   # If you have a lot of GPU memory, you can try to set swap_memory=False,
 362   # and/or parallel_iterations=2.
 363   elems = samples[common.IMAGE][1:]
 364   res = tf.scan(predict, elems,
 365                 initializer=init,
 366                 parallel_iterations=1,
 367                 swap_memory=True)
 368   predicted_labels, _, _ = res
 369   predicted_labels = tf.concat([reference_labels[..., 0],
 370                                 tf.squeeze(predicted_labels, axis=1)],
 371                                axis=0)
 372   return predicted_labels

 375 def main(unused_argv):
 376   if FLAGS.vis_batch_size != 1:
 377     raise ValueError('Only batch size 1 is supported for now')
 378 
 379   data_type = 'tf_sequence_example'
 380   # Get dataset-dependent information.
 381   dataset = video_dataset.get_dataset(
 382       FLAGS.dataset,
 383       FLAGS.vis_split,
 384       dataset_dir=FLAGS.dataset_dir,
 385       data_type=data_type)
 386 
 387   # Prepare for visualization.
 388   tf.gfile.MakeDirs(FLAGS.vis_logdir)
 389   segmentation_dir = os.path.join(FLAGS.vis_logdir, _SEGMENTATION_SAVE_FOLDER)
 390   tf.gfile.MakeDirs(segmentation_dir)
 391   embeddings_dir = os.path.join(FLAGS.vis_logdir, _EMBEDDINGS_SAVE_FOLDER)
 392   tf.gfile.MakeDirs(embeddings_dir)
 393   num_vis_examples = (dataset.num_videos if (FLAGS.num_vis_examples < 0)
 394                       else FLAGS.num_vis_examples)
 395   if FLAGS.first_frame_finetuning:
 396     num_vis_examples = 1
 397 
 398   tf.logging.info('Visualizing on %s set', FLAGS.vis_split)
 399   g = tf.Graph()
 400   with g.as_default():
 401     # Without setting device to CPU we run out of memory.
 402     with tf.device('cpu:0'):
 403       samples = video_input_generator.get(
 404           dataset,
 405           None,
 406           None,
 407           FLAGS.vis_batch_size,
 408           min_resize_value=FLAGS.min_resize_value,
 409           max_resize_value=FLAGS.max_resize_value,
 410           resize_factor=FLAGS.resize_factor,
 411           dataset_split=FLAGS.vis_split,
 412           is_training=False,
 413           model_variant=FLAGS.model_variant,
 414           preprocess_image_and_label=False,
 415           remap_labels_to_reference_frame=False)
 416       samples[common.IMAGE] = tf.cast(samples[common.IMAGE], tf.float32)
 417       samples[common.LABEL] = tf.cast(samples[common.LABEL], tf.int32)
 418       first_frame_img = samples[common.IMAGE][0]
 419       reference_labels = samples[common.LABEL][0, tf.newaxis]
 420       gt_labels = tf.squeeze(samples[common.LABEL], axis=-1)
 421       seq_name = samples[common.VIDEO_ID][0]
 422 
 423     model_options = common.VideoModelOptions(
 424         outputs_to_num_classes={common.OUTPUT_TYPE: dataset.num_classes},
 425         crop_size=None,
 426         atrous_rates=FLAGS.atrous_rates,
 427         output_stride=FLAGS.output_stride)
 428 
 429     all_embeddings = None
 430     predicted_labels = create_predictions_fast(
 431         samples, reference_labels, first_frame_img, model_options)
 432     # If you need more options like saving embeddings, replace the call to
 433     # create_predictions_fast with create_predictions.
 434 
 435     tf.train.get_or_create_global_step()
 436     saver = tf.train.Saver(slim.get_variables_to_restore())
 437     sv = tf.train.Supervisor(graph=g,
 438                              logdir=FLAGS.vis_logdir,
 439                              init_op=tf.global_variables_initializer(),
 440                              summary_op=None,
 441                              summary_writer=None,
 442                              global_step=None,
 443                              saver=saver)
 444     num_batches = int(
 445         math.ceil(num_vis_examples / float(FLAGS.vis_batch_size)))
 446     last_checkpoint = None
 447 
 448     # Infinite loop to visualize the results when new checkpoint is created.
 449     while True:
 450       last_checkpoint = slim.evaluation.wait_for_new_checkpoint(
 451           FLAGS.checkpoint_dir, last_checkpoint)
 452       start = time.time()
 453       tf.logging.info(
 454           'Starting visualization at ' + time.strftime('%Y-%m-%d-%H:%M:%S',
 455                                                        time.gmtime()))
 456       tf.logging.info('Visualizing with model %s', last_checkpoint)
 457 
 458       all_ious = []
 459       with sv.managed_session(FLAGS.master,
 460                               start_standard_services=False) as sess:
 461         sv.start_queue_runners(sess)
 462         sv.saver.restore(sess, last_checkpoint)
 463 
 464         for batch in range(num_batches):
 465           ops = [predicted_labels, gt_labels, seq_name]
 466           if FLAGS.save_embeddings:
 467             ops.append(all_embeddings)
 468           tf.logging.info('Visualizing batch %d / %d', batch + 1, num_batches)
 469           res = sess.run(ops)
 470           tf.logging.info('Forwarding done')
 471           pred_labels_val, gt_labels_val, seq_name_val = res[:3]
 472           if FLAGS.save_embeddings:
 473             all_embeddings_val = res[3]
 474           else:
 475             all_embeddings_val = None
 476           seq_ious = _process_seq_data(segmentation_dir, embeddings_dir,
 477                                        seq_name_val, pred_labels_val,
 478                                        gt_labels_val, all_embeddings_val)
 479           all_ious.append(seq_ious)
 480       all_ious = np.concatenate(all_ious, axis=0)
 481       tf.logging.info('n_seqs %s, mIoU %f', all_ious.shape, all_ious.mean())
 482       tf.logging.info(
 483           'Finished visualization at ' + time.strftime('%Y-%m-%d-%H:%M:%S',
 484                                                        time.gmtime()))
 485       result_dir = FLAGS.vis_logdir + '/results/'
 486       tf.gfile.MakeDirs(result_dir)
 487       with tf.gfile.GFile(result_dir + seq_name_val + '.txt', 'w') as f:
 488         f.write(str(all_ious))
 489       if FLAGS.first_frame_finetuning or FLAGS.eval_once_and_quit:
 490         break
 491       time_to_next_eval = start + FLAGS.eval_interval_secs - time.time()
 492       if time_to_next_eval > 0:
 493         time.sleep(time_to_next_eval)
 494 
 495 
 496 if __name__ == '__main__':
 497   flags.mark_flag_as_required('checkpoint_dir')
 498   flags.mark_flag_as_required('vis_logdir')
 499   tf.logging.set_verbosity(tf.logging.INFO)
 500   tf.app.run()
