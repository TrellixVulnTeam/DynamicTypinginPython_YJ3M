  16 r"""Provides DeepLab model definition and helper functions.
  17 
  18 DeepLab is a deep learning system for semantic image segmentation with
  19 the following features:
  20 
  21 (1) Atrous convolution to explicitly control the resolution at which
  22 feature responses are computed within Deep Convolutional Neural Networks.
  23 
  24 (2) Atrous spatial pyramid pooling (ASPP) to robustly segment objects at
  25 multiple scales with filters at multiple sampling rates and effective
  26 fields-of-views.
  27 
  28 (3) ASPP module augmented with image-level feature and batch normalization.
  29 
  30 (4) A simple yet effective decoder module to recover the object boundaries.
  31 
  32 See the following papers for more details:
  33 
  34 "Encoder-Decoder with Atrous Separable Convolution for Semantic Image
  35 Segmentation"
  36 Liang-Chieh Chen, Yukun Zhu, George Papandreou, Florian Schroff, Hartwig Adam.

  42

  79 def predict_labels(images,
  80                    model_options,
  81                    image_pyramid=None,
  82                    reference_labels=None,
  83                    k_nearest_neighbors=1,
  84                    embedding_dimension=None,
  85                    use_softmax_feedback=False,
  86                    initial_softmax_feedback=None,
  87                    embedding_seg_feature_dimension=256,
  88                    embedding_seg_n_layers=4,
  89                    embedding_seg_kernel_size=7,
  90                    embedding_seg_atrous_rates=None,
  91                    also_return_softmax_probabilities=False,
  92                    num_frames_per_video=None,
  93                    normalize_nearest_neighbor_distances=False,
  94                    also_attend_to_previous_frame=False,
  95                    use_local_previous_frame_attention=False,
  96                    previous_frame_attention_window_size=9,
  97                    use_first_frame_matching=True,
  98                    also_return_embeddings=False,
  99                    ref_embeddings=None):
 100   """Predicts segmentation labels.
 101 
 102   Args:
 103     images: A tensor of size [batch, height, width, channels].
 104     model_options: An InternalModelOptions instance to configure models.
 105     image_pyramid: Input image scales for multi-scale feature extraction.
 106     reference_labels: A tensor of size [batch, height, width, 1].
 107       ground truth labels used to perform a nearest neighbor query
 108     k_nearest_neighbors: Integer, the number of neighbors to use for nearest
 109       neighbor queries.
 110     embedding_dimension: Integer, the dimension used for the learned embedding.
 111     use_softmax_feedback: Boolean, whether to give the softmax predictions of
 112       the last frame as additional input to the segmentation head.
 113     initial_softmax_feedback: Float32 tensor, or None. Can be used to
 114       initialize the softmax predictions used for the feedback loop.
 115       Typically only useful for inference. Only has an effect if
 116       use_softmax_feedback is True.
 117     embedding_seg_feature_dimension: Integer, the dimensionality used in the
 118       segmentation head layers.
 119     embedding_seg_n_layers: Integer, the number of layers in the segmentation
 120       head.
 121     embedding_seg_kernel_size: Integer, the kernel size used in the
 122       segmentation head.
 123     embedding_seg_atrous_rates: List of integers of length
 124       embedding_seg_n_layers, the atrous rates to use for the segmentation head.
 125     also_return_softmax_probabilities: Boolean, if true, additionally return
 126       the softmax probabilities as second return value.
 127     num_frames_per_video: Integer, the number of frames per video.
 128     normalize_nearest_neighbor_distances: Boolean, whether to normalize the
 129       nearest neighbor distances to [0,1] using sigmoid, scale and shift.
 130     also_attend_to_previous_frame: Boolean, whether to also use nearest
 131       neighbor attention with respect to the previous frame.
 132     use_local_previous_frame_attention: Boolean, whether to restrict the
 133       previous frame attention to a local search window.
 134       Only has an effect, if also_attend_to_previous_frame is True.
 135     previous_frame_attention_window_size: Integer, the window size used for
 136       local previous frame attention, if use_local_previous_frame_attention
 137       is True.
 138     use_first_frame_matching: Boolean, whether to extract features by matching
 139       to the reference frame. This should always be true except for ablation
 140       experiments.
 141     also_return_embeddings: Boolean, whether to return the embeddings as well.
 142     ref_embeddings: Tuple of
 143       (first_frame_embeddings, previous_frame_embeddings),
 144       each of shape [batch, height, width, embedding_dimension], or None.
 145 
 146   Returns:
 147     A dictionary with keys specifying the output_type (e.g., semantic
 148       prediction) and values storing Tensors representing predictions (argmax
 149       over channels). Each prediction has size [batch, height, width].
 150     If also_return_softmax_probabilities is True, the second return value are
 151       the softmax probabilities.
 152     If also_return_embeddings is True, it will also return an embeddings
 153       tensor of shape [batch, height, width, embedding_dimension].
 154 
 155   Raises:
 156     ValueError: If classification_loss is not softmax, softmax_with_attention,
 157       nor triplet.
 158   """
 159   if (model_options.classification_loss == 'triplet' and
 160       reference_labels is None):
 161     raise ValueError('Need reference_labels for triplet loss')
 162 
 163   if model_options.classification_loss == 'softmax_with_attention':
 164     if embedding_dimension is None:
 165       raise ValueError('Need embedding_dimension for softmax_with_attention '
 166                        'loss')
 167     if reference_labels is None:
 168       raise ValueError('Need reference_labels for softmax_with_attention loss')
 169     res = (
 170         multi_scale_logits_with_nearest_neighbor_matching(
 171             images,
 172             model_options=model_options,
 173             image_pyramid=image_pyramid,
 174             is_training=False,
 175             reference_labels=reference_labels,
 176             clone_batch_size=1,
 177             num_frames_per_video=num_frames_per_video,
 178             embedding_dimension=embedding_dimension,
 179             max_neighbors_per_object=0,
 180             k_nearest_neighbors=k_nearest_neighbors,
 181             use_softmax_feedback=use_softmax_feedback,
 182             initial_softmax_feedback=initial_softmax_feedback,
 183             embedding_seg_feature_dimension=embedding_seg_feature_dimension,
 184             embedding_seg_n_layers=embedding_seg_n_layers,
 185             embedding_seg_kernel_size=embedding_seg_kernel_size,
 186             embedding_seg_atrous_rates=embedding_seg_atrous_rates,
 187             normalize_nearest_neighbor_distances=
 188             normalize_nearest_neighbor_distances,
 189             also_attend_to_previous_frame=also_attend_to_previous_frame,
 190             use_local_previous_frame_attention=
 191             use_local_previous_frame_attention,
 192             previous_frame_attention_window_size=
 193             previous_frame_attention_window_size,
 194             use_first_frame_matching=use_first_frame_matching,
 195             also_return_embeddings=also_return_embeddings,
 196             ref_embeddings=ref_embeddings
 197         ))
 198     if also_return_embeddings:
 199       outputs_to_scales_to_logits, embeddings = res
 200     else:
 201       outputs_to_scales_to_logits = res
 202       embeddings = None
 203   else:
 204     outputs_to_scales_to_logits = multi_scale_logits_v2(
 205         images,
 206         model_options=model_options,
 207         image_pyramid=image_pyramid,
 208         is_training=False,
 209         fine_tune_batch_norm=False)
 210 
 211   predictions = {}
 212   for output in sorted(outputs_to_scales_to_logits):
 213     scales_to_logits = outputs_to_scales_to_logits[output]
 214     original_logits = scales_to_logits[MERGED_LOGITS_SCOPE]
 215     if isinstance(original_logits, list):
 216       assert len(original_logits) == 1
 217       original_logits = original_logits[0]
 218     logits = tf.image.resize_bilinear(original_logits, tf.shape(images)[1:3],
 219                                       align_corners=True)
 220     if model_options.classification_loss in ('softmax',
 221                                              'softmax_with_attention'):
 222       predictions[output] = tf.argmax(logits, 3)
 223     elif model_options.classification_loss == 'triplet':
 224       # to keep this fast, we do the nearest neighbor assignment on the
 225       # resolution at which the embedding is extracted and scale the result up
 226       # afterwards
 227       embeddings = original_logits
 228       reference_labels_logits_size = tf.squeeze(
 229           tf.image.resize_nearest_neighbor(
 230               reference_labels[tf.newaxis],
 231               train_utils.resolve_shape(embeddings)[1:3],
 232               align_corners=True), axis=0)
 233       nn_labels = embedding_utils.assign_labels_by_nearest_neighbors(
 234           embeddings[0], embeddings[1:], reference_labels_logits_size,
 235           k_nearest_neighbors)
 236       predictions[common.OUTPUT_TYPE] = tf.image.resize_nearest_neighbor(
 237           nn_labels, tf.shape(images)[1:3], align_corners=True)
 238     else:
 239       raise ValueError(
 240           'Only support softmax, triplet, or softmax_with_attention for '
 241           'classification_loss.')
 242 
 243   if also_return_embeddings:
 244     assert also_return_softmax_probabilities
 245     return predictions, tf.nn.softmax(original_logits, axis=-1), embeddings
 246   elif also_return_softmax_probabilities:
 247     return predictions, tf.nn.softmax(original_logits, axis=-1)
 248   else:
 249     return predictions

 252 def multi_scale_logits_with_nearest_neighbor_matching(
 253     images,
 254     model_options,
 255     image_pyramid,
 256     clone_batch_size,
 257     reference_labels,
 258     num_frames_per_video,
 259     embedding_dimension,
 260     max_neighbors_per_object,
 261     weight_decay=0.0001,
 262     is_training=False,
 263     fine_tune_batch_norm=False,
 264     k_nearest_neighbors=1,
 265     use_softmax_feedback=False,
 266     initial_softmax_feedback=None,
 267     embedding_seg_feature_dimension=256,
 268     embedding_seg_n_layers=4,
 269     embedding_seg_kernel_size=7,
 270     embedding_seg_atrous_rates=None,
 271     normalize_nearest_neighbor_distances=False,
 272     also_attend_to_previous_frame=False,
 273     damage_initial_previous_frame_mask=False,
 274     use_local_previous_frame_attention=False,
 275     previous_frame_attention_window_size=9,
 276     use_first_frame_matching=True,
 277     also_return_embeddings=False,
 278     ref_embeddings=None):
 279   """Gets the logits for multi-scale inputs using nearest neighbor attention.
 280 
 281   Adjusted version of multi_scale_logits_v2 to support nearest neighbor
 282   attention and a variable number of classes for each element of the batch.
 283   The returned logits are all downsampled (due to max-pooling layers)
 284   for both training and evaluation.
 285 
 286   Args:
 287     images: A tensor of size [batch, height, width, channels].
 288     model_options: A ModelOptions instance to configure models.
 289     image_pyramid: Input image scales for multi-scale feature extraction.
 290     clone_batch_size: Integer, the number of videos on a batch.
 291     reference_labels: The segmentation labels of the reference frame on which
 292       attention is applied.
 293     num_frames_per_video: Integer, the number of frames per video.
 294     embedding_dimension: Integer, the dimension of the embedding.
 295     max_neighbors_per_object: Integer, the maximum number of candidates
 296       for the nearest neighbor query per object after subsampling.
 297       Can be 0 for no subsampling.
 298     weight_decay: The weight decay for model variables.
 299     is_training: Is training or not.
 300     fine_tune_batch_norm: Fine-tune the batch norm parameters or not.
 301     k_nearest_neighbors: Integer, the number of nearest neighbors to use.
 302     use_softmax_feedback: Boolean, whether to give the softmax predictions of
 303       the last frame as additional input to the segmentation head.
 304     initial_softmax_feedback: List of Float32 tensors, or None.
 305       Can be used to initialize the softmax predictions used for the feedback
 306       loop. Only has an effect if use_softmax_feedback is True.
 307     embedding_seg_feature_dimension: Integer, the dimensionality used in the
 308       segmentation head layers.
 309     embedding_seg_n_layers: Integer, the number of layers in the segmentation
 310       head.
 311     embedding_seg_kernel_size: Integer, the kernel size used in the
 312       segmentation head.
 313     embedding_seg_atrous_rates: List of integers of length
 314       embedding_seg_n_layers, the atrous rates to use for the segmentation head.
 315     normalize_nearest_neighbor_distances: Boolean, whether to normalize the
 316       nearest neighbor distances to [0,1] using sigmoid, scale and shift.
 317     also_attend_to_previous_frame: Boolean, whether to also use nearest
 318       neighbor attention with respect to the previous frame.
 319     damage_initial_previous_frame_mask: Boolean, whether to artificially damage
 320       the initial previous frame mask. Only has an effect if
 321       also_attend_to_previous_frame is True.
 322     use_local_previous_frame_attention: Boolean, whether to restrict the
 323       previous frame attention to a local search window.
 324       Only has an effect, if also_attend_to_previous_frame is True.
 325     previous_frame_attention_window_size: Integer, the window size used for
 326       local previous frame attention, if use_local_previous_frame_attention
 327       is True.
 328     use_first_frame_matching: Boolean, whether to extract features by matching
 329       to the reference frame. This should always be true except for ablation
 330       experiments.
 331     also_return_embeddings: Boolean, whether to return the embeddings as well.
 332     ref_embeddings: Tuple of
 333       (first_frame_embeddings, previous_frame_embeddings),
 334       each of shape [batch, height, width, embedding_dimension], or None.
 335 
 336   Returns:
 337     outputs_to_scales_to_logits: A map of maps from output_type (e.g.,
 338       semantic prediction) to a dictionary of multi-scale logits names to
 339       logits. For each output_type, the dictionary has keys which
 340       correspond to the scales and values which correspond to the logits.
 341       For example, if `scales` equals [1.0, 1.5], then the keys would
 342       include 'merged_logits', 'logits_1.00' and 'logits_1.50'.
 343     If also_return_embeddings is True, it will also return an embeddings
 344       tensor of shape [batch, height, width, embedding_dimension].
 345 
 346   Raises:
 347     ValueError: If model_options doesn't specify crop_size and its
 348       add_image_level_feature = True, since add_image_level_feature requires
 349       crop_size information.
 350   """
 351   # Setup default values.
 352   if not image_pyramid:
 353     image_pyramid = [1.0]
 354   crop_height = (
 355       model_options.crop_size[0]
 356       if model_options.crop_size else tf.shape(images)[1])
 357   crop_width = (
 358       model_options.crop_size[1]
 359       if model_options.crop_size else tf.shape(images)[2])
 360 
 361   # Compute the height, width for the output logits.
 362   if model_options.decoder_output_stride:
 363     logits_output_stride = min(model_options.decoder_output_stride)
 364   else:
 365     logits_output_stride = model_options.output_stride
 366   logits_height = scale_dimension(
 367       crop_height,
 368       max(1.0, max(image_pyramid)) / logits_output_stride)
 369   logits_width = scale_dimension(
 370       crop_width,
 371       max(1.0, max(image_pyramid)) / logits_output_stride)
 372 
 373   # Compute the logits for each scale in the image pyramid.
 374   outputs_to_scales_to_logits = {
 375       k: {}
 376       for k in model_options.outputs_to_num_classes
 377   }
 378 
 379   for image_scale in image_pyramid:
 380     if image_scale != 1.0:
 381       scaled_height = scale_dimension(crop_height, image_scale)
 382       scaled_width = scale_dimension(crop_width, image_scale)
 383       scaled_crop_size = [scaled_height, scaled_width]
 384       scaled_images = tf.image.resize_bilinear(
 385           images, scaled_crop_size, align_corners=True)
 386       scaled_reference_labels = tf.image.resize_nearest_neighbor(
 387           reference_labels, scaled_crop_size, align_corners=True
 388       )
 389       if model_options.crop_size is None:
 390         scaled_crop_size = None
 391       if model_options.crop_size:
 392         scaled_images.set_shape([None, scaled_height, scaled_width, 3])
 393     else:
 394       scaled_crop_size = model_options.crop_size
 395       scaled_images = images
 396       scaled_reference_labels = reference_labels
 397 
 398     updated_options = model_options._replace(crop_size=scaled_crop_size)
 399     res = embedding_utils.get_logits_with_matching(
 400         scaled_images,
 401         updated_options,
 402         weight_decay=weight_decay,
 403         reuse=tf.AUTO_REUSE,
 404         is_training=is_training,
 405         fine_tune_batch_norm=fine_tune_batch_norm,
 406         reference_labels=scaled_reference_labels,
 407         batch_size=clone_batch_size,
 408         num_frames_per_video=num_frames_per_video,
 409         embedding_dimension=embedding_dimension,
 410         max_neighbors_per_object=max_neighbors_per_object,
 411         k_nearest_neighbors=k_nearest_neighbors,
 412         use_softmax_feedback=use_softmax_feedback,
 413         initial_softmax_feedback=initial_softmax_feedback,
 414         embedding_seg_feature_dimension=embedding_seg_feature_dimension,
 415         embedding_seg_n_layers=embedding_seg_n_layers,
 416         embedding_seg_kernel_size=embedding_seg_kernel_size,
 417         embedding_seg_atrous_rates=embedding_seg_atrous_rates,
 418         normalize_nearest_neighbor_distances=
 419         normalize_nearest_neighbor_distances,
 420         also_attend_to_previous_frame=also_attend_to_previous_frame,
 421         damage_initial_previous_frame_mask=damage_initial_previous_frame_mask,
 422         use_local_previous_frame_attention=use_local_previous_frame_attention,
 423         previous_frame_attention_window_size=
 424         previous_frame_attention_window_size,
 425         use_first_frame_matching=use_first_frame_matching,
 426         also_return_embeddings=also_return_embeddings,
 427         ref_embeddings=ref_embeddings
 428     )
 429     if also_return_embeddings:
 430       outputs_to_logits, embeddings = res
 431     else:
 432       outputs_to_logits = res
 433       embeddings = None
 434 
 435     # Resize the logits to have the same dimension before merging.
 436     for output in sorted(outputs_to_logits):
 437       if isinstance(outputs_to_logits[output], collections.Sequence):
 438         outputs_to_logits[output] = [tf.image.resize_bilinear(
 439             x, [logits_height, logits_width], align_corners=True)
 440                                      for x in outputs_to_logits[output]]
 441       else:
 442         outputs_to_logits[output] = tf.image.resize_bilinear(
 443             outputs_to_logits[output], [logits_height, logits_width],
 444             align_corners=True)
 445 
 446     # Return when only one input scale.
 447     if len(image_pyramid) == 1:
 448       for output in sorted(model_options.outputs_to_num_classes):
 449         outputs_to_scales_to_logits[output][
 450             MERGED_LOGITS_SCOPE] = outputs_to_logits[output]
 451       if also_return_embeddings:
 452         return outputs_to_scales_to_logits, embeddings
 453       else:
 454         return outputs_to_scales_to_logits
 455 
 456     # Save logits to the output map.
 457     for output in sorted(model_options.outputs_to_num_classes):
 458       outputs_to_scales_to_logits[output][
 459           'logits_%.2f' % image_scale] = outputs_to_logits[output]
 460 
 461   # Merge the logits from all the multi-scale inputs.
 462   for output in sorted(model_options.outputs_to_num_classes):
 463     # Concatenate the multi-scale logits for each output type.
 464     all_logits = [
 465         [tf.expand_dims(l, axis=4)]
 466         for logits in outputs_to_scales_to_logits[output].values()
 467         for l in logits
 468     ]
 469     transposed = map(list, zip(*all_logits))
 470     all_logits = [tf.concat(t, 4) for t in transposed]
 471     merge_fn = (
 472         tf.reduce_max
 473         if model_options.merge_method == 'max' else tf.reduce_mean)
 474     outputs_to_scales_to_logits[output][MERGED_LOGITS_SCOPE] = [merge_fn(
 475         l, axis=4) for l in all_logits]
 476 
 477   if also_return_embeddings:
 478     return outputs_to_scales_to_logits, embeddings
 479   else:
 480     return outputs_to_scales_to_logits
