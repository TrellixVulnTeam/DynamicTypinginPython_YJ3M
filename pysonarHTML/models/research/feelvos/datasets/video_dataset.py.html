  16 """Provides data from video object segmentation datasets.
  17 
  18 This file provides both images and annotations (instance segmentations) for
  19 TensorFlow. Currently, we support the following datasets:
  20 
  21 1. DAVIS 2017 (https://davischallenge.org/davis2017/code.html).
  22 
  23 2. DAVIS 2016 (https://davischallenge.org/davis2016/code.html).
  24 
  25 3. YouTube-VOS (https://youtube-vos.org/dataset/download).
  26 """
  27 
  28 from __future__ import absolute_import
  29 from __future__ import division
  30 from __future__ import print_function
  31 
  32 import collections
  33 import os.path
  34 import tensorflow as tf
  35 from feelvos.datasets import tfsequence_example_decoder
  36 
  37 slim = tf.contrib.slim
  38 dataset = slim.dataset
  39 tfexample_decoder = slim.tfexample_decoder
  40 
  41 
  42 _ITEMS_TO_DESCRIPTIONS = {
  43     'image': 'A color image of varying height and width.',
  44     'labels_class': ('A semantic segmentation label whose size matches image.'
  45                      'Its values range from 0 (background) to num_classes.'),
  46 }
  47 
  48 # Named tuple to describe the dataset properties.
  49 DatasetDescriptor = collections.namedtuple(
  50     'DatasetDescriptor',
  51     ['splits_to_sizes',   # Splits of the dataset into training, val, and test.
  52      'num_classes',   # Number of semantic classes.
  53      'ignore_label',  # Ignore label value.
  54     ]
  55 )
  56 
  57 _DAVIS_2016_INFORMATION = DatasetDescriptor(
  58     splits_to_sizes={'train': [30, 1830],
  59                      'val': [20, 1376]},
  60     num_classes=2,
  61     ignore_label=255,
  62 )
  63 
  64 _DAVIS_2017_INFORMATION = DatasetDescriptor(
  65     splits_to_sizes={'train': [60, 4219],
  66                      'val': [30, 2023],
  67                      'test-dev': [30, 2037]},
  68     num_classes=None,  # Number of instances per videos differ.
  69     ignore_label=255,
  70 )
  71 
  72 _YOUTUBE_VOS_2018_INFORMATION = DatasetDescriptor(
  73     # Leave these sizes as None to allow for different splits into
  74     # training and validation sets.
  75     splits_to_sizes={'train': [None, None],
  76                      'val': [None, None]},
  77     num_classes=None,  # Number of instances per video differs.
  78     ignore_label=255,
  79 )
  80 
  81 _DATASETS_INFORMATION = {
  82     'davis_2016': _DAVIS_2016_INFORMATION,
  83     'davis_2017': _DAVIS_2017_INFORMATION,
  84     'youtube_vos_2018': _YOUTUBE_VOS_2018_INFORMATION,
  85 }
  86 
  87 # Default file pattern of SSTable. Note we include '-' to avoid the confusion
  88 # between `train-` and `trainval-` sets.
  89 _FILE_PATTERN = '%s-*'
  90 
  91 
  92 def get_dataset(dataset_name,
  93                 split_name,
  94                 dataset_dir,
  95                 file_pattern=None,
  96                 data_type='tf_sequence_example',
  97                 decode_video_frames=False):
  98   """Gets an instance of slim Dataset.
  99 
 100   Args:
 101     dataset_name: String, dataset name.
 102     split_name: String, the train/val Split name.
 103     dataset_dir: String, the directory of the dataset sources.
 104     file_pattern: String, file pattern of SSTable.
 105     data_type: String, data type. Currently supports 'tf_example' and
 106       'annotated_image'.
 107     decode_video_frames: Boolean, decode the images or not. Not decoding it here
 108         is useful if we subsample later
 109 
 110   Returns:
 111     An instance of slim Dataset.
 112 
 113   Raises:
 114     ValueError: If the dataset_name or split_name is not recognized, or if
 115       the dataset_type is not supported.
 116   """
 117   if dataset_name not in _DATASETS_INFORMATION:
 118     raise ValueError('The specified dataset is not supported yet.')
 119 
 120   splits_to_sizes = _DATASETS_INFORMATION[dataset_name].splits_to_sizes
 121 
 122   if split_name not in splits_to_sizes:
 123     raise ValueError('data split name %s not recognized' % split_name)
 124 
 125   # Prepare the variables for different datasets.
 126   num_classes = _DATASETS_INFORMATION[dataset_name].num_classes
 127   ignore_label = _DATASETS_INFORMATION[dataset_name].ignore_label
 128 
 129   if file_pattern is None:
 130     file_pattern = _FILE_PATTERN
 131   file_pattern = os.path.join(dataset_dir, file_pattern % split_name)
 132   if data_type == 'tf_sequence_example':
 133     keys_to_context_features = {
 134         'image/format': tf.FixedLenFeature((), tf.string, default_value='jpeg'),
 135         'image/height': tf.FixedLenFeature((), tf.int64, default_value=0),
 136         'image/width': tf.FixedLenFeature((), tf.int64, default_value=0),
 137         'segmentation/object/format': tf.FixedLenFeature(
 138             (), tf.string, default_value='png'),
 139         'video_id': tf.FixedLenFeature((), tf.string, default_value='unknown')
 140     }
 141     label_name = 'class' if dataset_name == 'davis_2016' else 'object'
 142     keys_to_sequence_features = {
 143         'image/encoded': tf.FixedLenSequenceFeature((), dtype=tf.string),
 144         'segmentation/{}/encoded'.format(label_name):
 145             tf.FixedLenSequenceFeature((), tf.string),
 146         'segmentation/{}/encoded'.format(label_name):
 147             tf.FixedLenSequenceFeature((), tf.string),
 148     }
 149     items_to_handlers = {
 150         'height': tfexample_decoder.Tensor('image/height'),
 151         'width': tfexample_decoder.Tensor('image/width'),
 152         'video_id': tfexample_decoder.Tensor('video_id')
 153     }
 154     if decode_video_frames:
 155       decode_image_handler = tfexample_decoder.Image(
 156           image_key='image/encoded',
 157           format_key='image/format',
 158           channels=3,
 159           repeated=True)
 160       items_to_handlers['image'] = decode_image_handler
 161       decode_label_handler = tfexample_decoder.Image(
 162           image_key='segmentation/{}/encoded'.format(label_name),
 163           format_key='segmentation/{}/format'.format(label_name),
 164           channels=1,
 165           repeated=True)
 166       items_to_handlers['labels_class'] = decode_label_handler
 167     else:
 168       items_to_handlers['image/encoded'] = tfexample_decoder.Tensor(
 169           'image/encoded')
 170       items_to_handlers[
 171           'segmentation/object/encoded'] = tfexample_decoder.Tensor(
 172               'segmentation/{}/encoded'.format(label_name))
 173     decoder = tfsequence_example_decoder.TFSequenceExampleDecoder(
 174         keys_to_context_features, keys_to_sequence_features, items_to_handlers)
 175   else:
 176     raise ValueError('Unknown data type.')
 177 
 178   size = splits_to_sizes[split_name]
 179   if isinstance(size, collections.Sequence):
 180     num_videos = size[0]
 181     num_samples = size[1]
 182   else:
 183     num_videos = 0
 184     num_samples = size
 185 
 186   return dataset.Dataset(
 187       data_sources=file_pattern,
 188       reader=tf.TFRecordReader,
 189       decoder=decoder,
 190       num_samples=num_samples,
 191       num_videos=num_videos,
 192       items_to_descriptions=_ITEMS_TO_DESCRIPTIONS,
 193       ignore_label=ignore_label,
 194       num_classes=num_classes,
 195       name=dataset_name,
 196       multi_label=True)
