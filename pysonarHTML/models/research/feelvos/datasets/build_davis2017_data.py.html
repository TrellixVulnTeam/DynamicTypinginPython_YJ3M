  43 def read_image(path):
  44   with open(path) as fid:
  45     image_str = fid.read()
  46     image = PIL.Image.open(io.BytesIO(image_str))
  47     w, h = image.size
  48   return image_str, (h, w)

  51 def read_annotation(path):
  52   """Reads a single image annotation from a png image.
  53 
  54   Args:
  55     path: Path to the png image.
  56 
  57   Returns:
  58     png_string: The png encoded as string.
  59     size: Tuple of (height, width).
  60   """
  61   with open(path) as fid:
  62     x = np.array(PIL.Image.open(fid))
  63     h, w = x.shape
  64     im = PIL.Image.fromarray(x)
  65 
  66   output = StringIO()
  67   im.save(output, format='png')
  68   png_string = output.getvalue()
  69   output.close()
  70 
  71   return png_string, (h, w)

  74 def process_video(key, input_dir, anno_dir):
  75   """Creates a SequenceExample for the video.
  76 
  77   Args:
  78     key: Name of the video.
  79     input_dir: Directory which contains the image files.
  80     anno_dir: Directory which contains the annotation files.
  81 
  82   Returns:
  83     The created SequenceExample.
  84   """
  85   frame_names = sorted(tf.gfile.ListDirectory(input_dir))
  86   anno_files = sorted(tf.gfile.ListDirectory(anno_dir))
  87   assert len(frame_names) == len(anno_files)
  88 
  89   sequence = tf.train.SequenceExample()
  90   context = sequence.context.feature
  91   features = sequence.feature_lists.feature_list
  92 
  93   for i, name in enumerate(frame_names):
  94     image_str, image_shape = read_image(
  95         os.path.join(input_dir, name))
  96     anno_str, anno_shape = read_annotation(
  97         os.path.join(anno_dir, name[:-4] + '.png'))
  98     image_encoded = features['image/encoded'].feature.add()
  99     image_encoded.bytes_list.value.append(image_str)
 100     segmentation_encoded = features['segmentation/object/encoded'].feature.add()
 101     segmentation_encoded.bytes_list.value.append(anno_str)
 102 
 103     np.testing.assert_array_equal(np.array(image_shape), np.array(anno_shape))
 104 
 105     if i == 0:
 106       first_shape = np.array(image_shape)
 107     else:
 108       np.testing.assert_array_equal(np.array(image_shape), first_shape)
 109 
 110   context['video_id'].bytes_list.value.append(key.encode('ascii'))
 111   context['clip/frames'].int64_list.value.append(len(frame_names))
 112   context['image/format'].bytes_list.value.append('JPEG')
 113   context['image/channels'].int64_list.value.append(3)
 114   context['image/height'].int64_list.value.append(first_shape[0])
 115   context['image/width'].int64_list.value.append(first_shape[1])
 116   context['segmentation/object/format'].bytes_list.value.append('PNG')
 117   context['segmentation/object/height'].int64_list.value.append(first_shape[0])
 118   context['segmentation/object/width'].int64_list.value.append(first_shape[1])
 119 
 120   return sequence

 123 def convert(data_folder, imageset, output_dir, num_shards):
 124   """Converts the specified subset of DAVIS 2017 to TFRecord format.
 125 
 126   Args:
 127     data_folder: The path to the DAVIS 2017 data.
 128     imageset: The subset to use, either train or val.
 129     output_dir: Where to store the TFRecords.
 130     num_shards: The number of shards used for storing the data.
 131   """
 132   sets_file = os.path.join(data_folder, 'ImageSets', '2017', imageset + '.txt')
 133   vids = [x.strip() for x in open(sets_file).readlines()]
 134   num_vids = len(vids)
 135   num_vids_per_shard = int(math.ceil(num_vids) / float(num_shards))
 136   for shard_id in range(num_shards):
 137     output_filename = os.path.join(
 138         output_dir,
 139         '%s-%05d-of-%05d.tfrecord' % (imageset, shard_id, num_shards))
 140     with tf.python_io.TFRecordWriter(output_filename) as tfrecord_writer:
 141       start_idx = shard_id * num_vids_per_shard
 142       end_idx = min((shard_id + 1) * num_vids_per_shard, num_vids)
 143       for i in range(start_idx, end_idx):
 144         print('Converting video %d/%d shard %d video %s' % (
 145             i + 1, num_vids, shard_id, vids[i]))
 146         img_dir = os.path.join(data_folder, 'JPEGImages', '480p', vids[i])
 147         anno_dir = os.path.join(data_folder, 'Annotations', '480p', vids[i])
 148         example = process_video(vids[i], img_dir, anno_dir)
 149         tfrecord_writer.write(example.SerializeToString())

 152 def main(unused_argv):
 153   imageset = FLAGS.imageset
 154   assert imageset in ('train', 'val')
 155   if imageset == 'train':
 156     num_shards = _NUM_SHARDS_TRAIN
 157   else:
 158     num_shards = _NUM_SHARDS_VAL
 159   convert(FLAGS.data_folder, FLAGS.imageset, FLAGS.output_dir, num_shards)

 163   tf.app.run()
