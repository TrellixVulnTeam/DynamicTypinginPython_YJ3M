  34 def add_triplet_loss_for_each_scale(batch_size, num_frames_per_video,
  35                                     embedding_dim, scales_to_embeddings,
  36                                     labels, scope):
  37   """Adds triplet loss for logits of each scale.
  38 
  39   Args:
  40     batch_size: Int, the number of video chunks sampled per batch
  41     num_frames_per_video: Int, the number of frames per video.
  42     embedding_dim: Int, the dimension of the learned embedding
  43     scales_to_embeddings: A map from embedding names for different scales to
  44       embeddings. The embeddings have shape [batch, embeddings_height,
  45       embeddings_width, embedding_dim].
  46     labels: Groundtruth labels with shape [batch, image_height, image_width, 1].
  47     scope: String, the scope for the loss.
  48 
  49   Raises:
  50     ValueError: labels is None.
  51   """
  52   if labels is None:
  53     raise ValueError('No label for triplet loss.')
  54   for scale, embeddings in scales_to_embeddings.iteritems():
  55     loss_scope = None
  56     if scope:
  57       loss_scope = '%s_%s' % (scope, scale)
  58     # Label is downsampled to the same size as logits.
  59     scaled_labels = tf.image.resize_nearest_neighbor(
  60         labels,
  61         resolve_shape(embeddings, 4)[1:3],
  62         align_corners=True)
  63     # Reshape from [batch * num_frames, ...] to [batch, num_frames, ...].
  64     h = tf.shape(embeddings)[1]
  65     w = tf.shape(embeddings)[2]
  66     new_labels_shape = tf.stack([batch_size, num_frames_per_video, h, w, 1])
  67     reshaped_labels = tf.reshape(scaled_labels, new_labels_shape)
  68     new_embeddings_shape = tf.stack([batch_size, num_frames_per_video, h, w,
  69                                      -1])
  70     reshaped_embeddings = tf.reshape(embeddings, new_embeddings_shape)
  71 
  72     with tf.name_scope(loss_scope):
  73       total_loss = tf.constant(0, dtype=tf.float32)
  74       for n in range(batch_size):
  75         embedding = reshaped_embeddings[n]
  76         label = reshaped_labels[n]
  77         n_pixels = h * w
  78         n_anchors_used = 256
  79         sampled_anchor_indices = tf.random_shuffle(tf.range(n_pixels))[
  80             :n_anchors_used]
  81         anchors_pool = tf.reshape(embedding[0], [-1, embedding_dim])
  82         anchors_pool_classes = tf.reshape(label[0], [-1])
  83         anchors = tf.gather(anchors_pool, sampled_anchor_indices)
  84         anchor_classes = tf.gather(anchors_pool_classes, sampled_anchor_indices)
  85 
  86         pos_neg_pool = tf.reshape(embedding[1:], [-1, embedding_dim])
  87         pos_neg_pool_classes = tf.reshape(label[1:], [-1])
  88         dists = embedding_utils.pairwise_distances(anchors, pos_neg_pool)
  89         pos_mask = tf.equal(anchor_classes[:, tf.newaxis],
  90                             pos_neg_pool_classes[tf.newaxis, :])
  91         neg_mask = tf.logical_not(pos_mask)
  92         pos_mask_f = tf.cast(pos_mask, tf.float32)
  93         neg_mask_f = tf.cast(neg_mask, tf.float32)
  94         pos_dists = pos_mask_f * dists + 1e20 * neg_mask_f
  95         neg_dists = neg_mask_f * dists + 1e20 * pos_mask_f
  96         pos_dists_min = tf.reduce_min(pos_dists, axis=1)
  97         neg_dists_min = tf.reduce_min(neg_dists, axis=1)
  98         margin = 1.0
  99         loss = tf.nn.relu(pos_dists_min - neg_dists_min + margin)
 100         # Handle case that no positive is present (per anchor).
 101         any_pos = tf.reduce_any(pos_mask, axis=1)
 102         loss *= tf.cast(any_pos, tf.float32)
 103         # Average over anchors
 104         loss = tf.reduce_mean(loss, axis=0)
 105         total_loss += loss
 106       total_loss /= batch_size
 107       # Scale the loss up a bit.
 108       total_loss *= 3.0
 109       tf.add_to_collection(tf.GraphKeys.LOSSES, total_loss)

 112 def add_dynamic_softmax_cross_entropy_loss_for_each_scale(
 113     scales_to_logits, labels, ignore_label, loss_weight=1.0,
 114     upsample_logits=True, scope=None, top_k_percent_pixels=1.0,
 115     hard_example_mining_step=100000):
 116   """Adds softmax cross entropy loss per scale for logits with varying classes.
 117 
 118   Also adds summaries for mIoU.
 119 
 120   Args:
 121     scales_to_logits: A map from logits names for different scales to logits.
 122       The logits are a list of length batch_size of tensors of shape
 123       [time, logits_height, logits_width, num_classes].
 124     labels: Groundtruth labels with shape [batch_size * time, image_height,
 125       image_width, 1].
 126     ignore_label: Integer, label to ignore.
 127     loss_weight: Float, loss weight.
 128     upsample_logits: Boolean, upsample logits or not.
 129     scope: String, the scope for the loss.
 130     top_k_percent_pixels: A float, the value lies in [0.0, 1.0]. When its
 131       value < 1.0, only compute the loss for the top k percent pixels (e.g.,
 132       the top 20% pixels). This is useful for hard pixel mining.
 133     hard_example_mining_step: An integer, the training step in which the
 134       hard exampling mining kicks off. Note that we gradually reduce the
 135       mining percent to the top_k_percent_pixels. For example, if
 136       hard_example_mining_step=100K and top_k_percent_pixels=0.25, then
 137       mining percent will gradually reduce from 100% to 25% until 100K steps
 138       after which we only mine top 25% pixels.
 139 
 140   Raises:
 141     ValueError: Label or logits is None.
 142   """
 143   if labels is None:
 144     raise ValueError('No label for softmax cross entropy loss.')
 145 
 146   if top_k_percent_pixels < 0 or top_k_percent_pixels > 1:
 147     raise ValueError('Unexpected value of top_k_percent_pixels.')
 148 
 149   for scale, logits in six.iteritems(scales_to_logits):
 150     loss_scope = None
 151     if scope:
 152       loss_scope = '%s_%s' % (scope, scale)
 153 
 154     if upsample_logits:
 155       # Label is not downsampled, and instead we upsample logits.
 156       assert isinstance(logits, collections.Sequence)
 157       logits = [tf.image.resize_bilinear(
 158           x,
 159           preprocess_utils.resolve_shape(labels, 4)[1:3],
 160           align_corners=True) for x in logits]
 161       scaled_labels = labels
 162     else:
 163       # Label is downsampled to the same size as logits.
 164       assert isinstance(logits, collections.Sequence)
 165       scaled_labels = tf.image.resize_nearest_neighbor(
 166           labels,
 167           preprocess_utils.resolve_shape(logits[0], 4)[1:3],
 168           align_corners=True)
 169 
 170     batch_size = len(logits)
 171     num_time = preprocess_utils.resolve_shape(logits[0])[0]
 172     reshaped_labels = tf.reshape(
 173         scaled_labels, ([batch_size, num_time] +
 174                         preprocess_utils.resolve_shape(scaled_labels)[1:]))
 175     for n, logits_n in enumerate(logits):
 176       labels_n = reshaped_labels[n]
 177       labels_n = tf.reshape(labels_n, shape=[-1])
 178       not_ignore_mask = tf.to_float(tf.not_equal(labels_n,
 179                                                  ignore_label)) * loss_weight
 180       num_classes_n = tf.shape(logits_n)[-1]
 181       one_hot_labels = slim.one_hot_encoding(
 182           labels_n, num_classes_n, on_value=1.0, off_value=0.0)
 183       logits_n_flat = tf.reshape(logits_n, shape=[-1, num_classes_n])
 184       if top_k_percent_pixels == 1.0:
 185         tf.losses.softmax_cross_entropy(
 186             one_hot_labels,
 187             logits_n_flat,
 188             weights=not_ignore_mask,
 189             scope=loss_scope)
 190       else:
 191         # Only compute the loss for top k percent pixels.
 192         # First, compute the loss for all pixels. Note we do not put the loss
 193         # to loss_collection and set reduction = None to keep the shape.
 194         num_pixels = tf.to_float(tf.shape(logits_n_flat)[0])
 195         pixel_losses = tf.losses.softmax_cross_entropy(
 196             one_hot_labels,
 197             logits_n_flat,
 198             weights=not_ignore_mask,
 199             scope='pixel_losses',
 200             loss_collection=None,
 201             reduction=tf.losses.Reduction.NONE)
 202         # Compute the top_k_percent pixels based on current training step.
 203         if hard_example_mining_step == 0:
 204           # Directly focus on the top_k pixels.
 205           top_k_pixels = tf.to_int32(top_k_percent_pixels * num_pixels)
 206         else:
 207           # Gradually reduce the mining percent to top_k_percent_pixels.
 208           global_step = tf.to_float(tf.train.get_or_create_global_step())
 209           ratio = tf.minimum(1.0, global_step / hard_example_mining_step)
 210           top_k_pixels = tf.to_int32(
 211               (ratio * top_k_percent_pixels + (1.0 - ratio)) * num_pixels)
 212         _, top_k_indices = tf.nn.top_k(pixel_losses,
 213                                        k=top_k_pixels,
 214                                        sorted=True,
 215                                        name='top_k_percent_pixels')
 216         # Compute the loss for the top k percent pixels.
 217         tf.losses.softmax_cross_entropy(
 218             tf.gather(one_hot_labels, top_k_indices),
 219             tf.gather(logits_n_flat, top_k_indices),
 220             weights=tf.gather(not_ignore_mask, top_k_indices),
 221             scope=loss_scope)
 222 
 223       pred_n = tf.argmax(logits_n, axis=-1, output_type=tf.int32)[
 224           ..., tf.newaxis]
 225       labels_n = labels[n * num_time: (n + 1) * num_time]
 226       miou = eval_utils.calculate_multi_object_miou_tf(pred_n, labels_n)
 227       tf.summary.scalar('miou', miou)

 230 def get_model_init_fn(train_logdir,
 231                       tf_initial_checkpoint,
 232                       initialize_last_layer,
 233                       last_layers,
 234                       ignore_missing_vars=False):
 235   """Gets the function initializing model variables from a checkpoint.
 236 
 237   Args:
 238     train_logdir: Log directory for training.
 239     tf_initial_checkpoint: TensorFlow checkpoint for initialization.
 240     initialize_last_layer: Initialize last layer or not.
 241     last_layers: Last layers of the model.
 242     ignore_missing_vars: Ignore missing variables in the checkpoint.
 243 
 244   Returns:
 245     Initialization function.
 246   """
 247   if tf_initial_checkpoint is None:
 248     tf.logging.info('Not initializing the model from a checkpoint.')
 249     return None
 250 
 251   if tf.train.latest_checkpoint(train_logdir):
 252     tf.logging.info('Ignoring initialization; other checkpoint exists')
 253     return None
 254 
 255   tf.logging.info('Initializing model from path: %s', tf_initial_checkpoint)
 256 
 257   # Variables that will not be restored.
 258   exclude_list = ['global_step']
 259   if not initialize_last_layer:
 260     exclude_list.extend(last_layers)
 261 
 262   variables_to_restore = slim.get_variables_to_restore(exclude=exclude_list)
 263 
 264   if variables_to_restore:
 265     return slim.assign_from_checkpoint_fn(
 266         tf_initial_checkpoint,
 267         variables_to_restore,
 268         ignore_missing_vars=ignore_missing_vars)
 269   return None
