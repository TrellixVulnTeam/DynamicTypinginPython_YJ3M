  43   @unittest.skipIf(not embedding_utils.USE_CORRELATION_COST,
  44                    'depends on correlation_cost')
  45   def test_correlation_cost_one_dimensional(self):
  46     a = np.array([[[[1.0], [2.0]], [[3.0], [4.0]]]])
  47     b = np.array([[[[2.0], [1.0]], [[4.0], [3.0]]]])
  48     g = tf.Graph()
  49     with g.as_default():
  50       with self.test_session(graph=g) as sess:
  51         c = correlation_cost_op.correlation_cost(
  52             a, b, kernel_size=1, max_displacement=1, stride_1=1, stride_2=1,
  53             pad=1)
  54         c = tf.squeeze(c, axis=0)
  55         c_val = sess.run(c)
  56         self.assertAllEqual(c_val.shape, (2, 2, 9))
  57         for y in range(2):
  58           for x in range(2):
  59             for dy in range(-1, 2):
  60               for dx in range(-1, 2):
  61                 a_slice = a[0, y, x, 0]
  62                 if y + dy < 0 or y + dy > 1 or x + dx < 0 or x + dx > 1:
  63                   b_slice = 0
  64                 else:
  65                   b_slice = b[0, y + dy, x + dx, 0]
  66                 expected = a_slice * b_slice
  67                 dy0 = dy + 1
  68                 dx0 = dx + 1
  69                 self.assertAlmostEqual(c_val[y, x, 3 * dy0 + dx0], expected)

  76     g = tf.Graph()
  77     with g.as_default():
  78       with self.test_session(graph=g) as sess:
  79         c = correlation_cost_op.correlation_cost(
  80             a, b, kernel_size=1, max_displacement=1, stride_1=1, stride_2=1,
  81             pad=1)
  82         c = tf.squeeze(c, axis=0)
  83         c_val = sess.run(c)
  84         self.assertAllEqual(c_val.shape, (2, 2, 9))
  85         for y in range(2):
  86           for x in range(2):
  87             for dy in range(-1, 2):
  88               for dx in range(-1, 2):
  89                 a_slice = a[0, y, x, :]
  90                 if y + dy < 0 or y + dy > 1 or x + dx < 0 or x + dx > 1:
  91                   b_slice = 0
  92                 else:
  93                   b_slice = b[0, y + dy, x + dx, :]

  96                 dx0 = dx + 1
  97                 self.assertAlmostEqual(c_val[y, x, 3 * dy0 + dx0], expected)

 101   def test_local_pairwise_distances_one_dimensional(self):
 102     a = np.array([[[1.0], [2.0]], [[3.0], [4.0]]])
 103     b = np.array([[[2.0], [1.0]], [[4.0], [3.0]]])
 104     g = tf.Graph()
 105     with g.as_default():
 106       with self.test_session(graph=g) as sess:
 107         a_tf = tf.constant(a, dtype=tf.float32)
 108         b_tf = tf.constant(b, dtype=tf.float32)
 109         d = embedding_utils.local_pairwise_distances(a_tf, b_tf,
 110                                                      max_distance=1)
 111         d_val = sess.run(d)
 112         for y in range(2):
 113           for x in range(2):
 114             for dy in range(-1, 2):
 115               for dx in range(-1, 2):
 116                 a_slice = a[y, x, 0]
 117                 if y + dy < 0 or y + dy > 1 or x + dx < 0 or x + dx > 1:
 118                   expected = np.float('inf')
 119                 else:
 120                   b_slice = b[y + dy, x + dx, 0]
 121                   expected = (a_slice - b_slice) ** 2
 122                 dy0 = dy + 1
 123                 dx0 = dx + 1
 124                 self.assertAlmostEqual(d_val[y, x, 3 * dy0 + dx0], expected)

 167   @unittest.skipIf(not embedding_utils.USE_CORRELATION_COST,
 168                    'depends on correlation_cost')
 169   def test_local_previous_frame_nearest_neighbor_features_per_object(self):
 170     prev_frame_embedding = np.array([[[1.0, -5.0], [7.0, 2.0]],
 171                                      [[1.0, 3.0], [3.0, 4.0]]]) / 10
 172     query_embedding = np.array([[[2.0, 1.0], [0.0, -9.0]],
 173                                 [[4.0, 3.0], [3.0, 1.0]]]) / 10
 174     prev_frame_labels = np.array([[[0], [1]], [[1], [0]]])
 175     gt_ids = np.array([0, 1])
 176     g = tf.Graph()
 177     with g.as_default():
 178       with self.test_session(graph=g) as sess:
 179         prev_frame_embedding_tf = tf.constant(prev_frame_embedding,
 180                                               dtype=tf.float32)
 181         query_embedding_tf = tf.constant(query_embedding, dtype=tf.float32)
 182         embu = embedding_utils
 183         dists = (
 184             embu.local_previous_frame_nearest_neighbor_features_per_object(
 185                 prev_frame_embedding_tf, query_embedding_tf,
 186                 prev_frame_labels, gt_ids, max_distance=1))
 187         dists = tf.squeeze(dists, axis=4)
 188         dists = tf.squeeze(dists, axis=0)
 189         dists_val = sess.run(dists)
 190         for obj_id in gt_ids:
 191           for y in range(2):
 192             for x in range(2):
 193               curr_min = 1.0
 194               for dy in range(-1, 2):
 195                 for dx in range(-1, 2):
 196                   # Attention: here we shift the prev frame embedding,
 197                   # not the query.
 198                   if y + dy < 0 or y + dy > 1 or x + dx < 0 or x + dx > 1:
 199                     continue
 200                   if prev_frame_labels[y + dy, x + dx, 0] != obj_id:
 201                     continue
 202                   prev_frame_slice = prev_frame_embedding[y + dy, x + dx, :]
 203                   query_frame_slice = query_embedding[y, x, :]
 204                   v_unnorm = ((prev_frame_slice - query_frame_slice) ** 2).sum()
 205                   v = ((1.0 / (1.0 + np.exp(-v_unnorm))) - 0.5) * 2
 206                   curr_min = min(curr_min, v)
 207               expected = curr_min
 208               self.assertAlmostEqual(dists_val[y, x, obj_id], expected,
 209                                      places=5)
