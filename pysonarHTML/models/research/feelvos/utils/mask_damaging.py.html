  21 from skimage import transform
  22 import tensorflow as tf

  25 def damage_masks(labels, shift=True, scale=True, rotate=True, dilate=True):
  26   """Damages segmentation masks by random transformations.
  27 
  28   Args:
  29     labels: Int32 labels tensor of shape (height, width, 1).
  30     shift: Boolean, whether to damage the masks by shifting.
  31     scale: Boolean, whether to damage the masks by scaling.
  32     rotate: Boolean, whether to damage the masks by rotation.
  33     dilate: Boolean, whether to damage the masks by dilation.
  34 
  35   Returns:
  36     The damaged version of labels.
  37   """
  38   def _damage_masks_np(labels_):
  39     return damage_masks_np(labels_, shift, scale, rotate, dilate)
  40   damaged_masks = tf.py_func(_damage_masks_np, [labels], tf.int32,
  41                              name='damage_masks')
  42   damaged_masks.set_shape(labels.get_shape())
  43   return damaged_masks
  44

  46 def damage_masks_np(labels, shift=True, scale=True, rotate=True, dilate=True):
  47   """Performs the actual mask damaging in numpy.
  48 
  49   Args:
  50     labels: Int32 numpy array of shape (height, width, 1).
  51     shift: Boolean, whether to damage the masks by shifting.
  52     scale: Boolean, whether to damage the masks by scaling.
  53     rotate: Boolean, whether to damage the masks by rotation.
  54     dilate: Boolean, whether to damage the masks by dilation.
  55 
  56   Returns:
  57     The damaged version of labels.
  58   """
  59   unique_labels = np.unique(labels)
  60   unique_labels = np.setdiff1d(unique_labels, [0])
  61   # Shuffle to get random depth ordering when combining together.
  62   np.random.shuffle(unique_labels)
  63   damaged_labels = np.zeros_like(labels)
  64   for l in unique_labels:
  65     obj_mask = (labels == l)
  66     damaged_obj_mask = _damage_single_object_mask(obj_mask, shift, scale,
  67                                                   rotate, dilate)
  68     damaged_labels[damaged_obj_mask] = l
  69   return damaged_labels
  70

  72 def _damage_single_object_mask(mask, shift, scale, rotate, dilate):
  73   """Performs mask damaging in numpy for a single object.
  74 
  75   Args:
  76     mask: Boolean numpy array of shape(height, width, 1).
  77     shift: Boolean, whether to damage the masks by shifting.
  78     scale: Boolean, whether to damage the masks by scaling.
  79     rotate: Boolean, whether to damage the masks by rotation.
  80     dilate: Boolean, whether to damage the masks by dilation.
  81 
  82   Returns:
  83     The damaged version of mask.
  84   """
  85   # For now we just do shifting and scaling. Better would be Affine or thin
  86   # spline plate transformations.
  87   if shift:
  88     mask = _shift_mask(mask)
  89   if scale:
  90     mask = _scale_mask(mask)
  91   if rotate:
  92     mask = _rotate_mask(mask)
  93   if dilate:
  94     mask = _dilate_mask(mask)
  95   return mask

  98 def _shift_mask(mask, max_shift_factor=0.05):
  99   """Damages a mask for a single object by randomly shifting it in numpy.
 100 
 101   Args:
 102     mask: Boolean numpy array of shape(height, width, 1).
 103     max_shift_factor: Float scalar, the maximum factor for random shifting.
 104 
 105   Returns:
 106     The shifted version of mask.
 107   """
 108   nzy, nzx, _ = mask.nonzero()
 109   h = nzy.max() - nzy.min()
 110   w = nzx.max() - nzx.min()
 111   size = np.sqrt(h * w)
 112   offset = np.random.uniform(-size * max_shift_factor, size * max_shift_factor,
 113                              2)
 114   shifted_mask = interpolation.shift(np.squeeze(mask, axis=2),
 115                                      offset, order=0).astype('bool')[...,
 116                                                                      np.newaxis]
 117   return shifted_mask

 120 def _scale_mask(mask, scale_amount=0.025):
 121   """Damages a mask for a single object by randomly scaling it in numpy.
 122 
 123   Args:
 124     mask: Boolean numpy array of shape(height, width, 1).
 125     scale_amount: Float scalar, the maximum factor for random scaling.
 126 
 127   Returns:
 128     The scaled version of mask.
 129   """
 130   nzy, nzx, _ = mask.nonzero()
 131   cy = 0.5 * (nzy.max() - nzy.min())
 132   cx = 0.5 * (nzx.max() - nzx.min())
 133   scale_factor = np.random.uniform(1.0 - scale_amount, 1.0 + scale_amount)
 134   shift = transform.SimilarityTransform(translation=[-cx, -cy])
 135   inv_shift = transform.SimilarityTransform(translation=[cx, cy])
 136   s = transform.SimilarityTransform(scale=[scale_factor, scale_factor])
 137   m = (shift + (s + inv_shift)).inverse
 138   scaled_mask = transform.warp(mask, m) > 0.5
 139   return scaled_mask

 142 def _rotate_mask(mask, max_rot_degrees=3.0):
 143   """Damages a mask for a single object by randomly rotating it in numpy.
 144 
 145   Args:
 146     mask: Boolean numpy array of shape(height, width, 1).
 147     max_rot_degrees: Float scalar, the maximum number of degrees to rotate.
 148 
 149   Returns:
 150     The scaled version of mask.
 151   """
 152   cy = 0.5 * mask.shape[0]
 153   cx = 0.5 * mask.shape[1]
 154   rot_degrees = np.random.uniform(-max_rot_degrees, max_rot_degrees)
 155   shift = transform.SimilarityTransform(translation=[-cx, -cy])
 156   inv_shift = transform.SimilarityTransform(translation=[cx, cy])
 157   r = transform.SimilarityTransform(rotation=np.deg2rad(rot_degrees))
 158   m = (shift + (r + inv_shift)).inverse
 159   scaled_mask = transform.warp(mask, m) > 0.5
 160   return scaled_mask

 163 def _dilate_mask(mask, dilation_radius=5):
 164   """Damages a mask for a single object by dilating it in numpy.
 165 
 166   Args:
 167     mask: Boolean numpy array of shape(height, width, 1).
 168     dilation_radius: Integer, the radius of the used disk structure element.
 169 
 170   Returns:
 171     The dilated version of mask.
 172   """
 173   disk = morphology.disk(dilation_radius, dtype=np.bool)
 174   dilated_mask = morphology.binary_dilation(
 175       np.squeeze(mask, axis=2), selem=disk)[..., np.newaxis]
 176   return dilated_mask
