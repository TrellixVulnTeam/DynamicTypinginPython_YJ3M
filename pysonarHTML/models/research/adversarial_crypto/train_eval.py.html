  16 """Adversarial training to learn trivial encryption functions,
  17 from the paper "Learning to Protect Communications with
  18 Adversarial Neural Cryptography", Abadi & Andersen, 2016.
  19 
  20 https://arxiv.org/abs/1610.06918
  21 
  22 This program creates and trains three neural networks,
  23 termed Alice, Bob, and Eve.  Alice takes inputs
  24 in_m (message), in_k (key) and outputs 'ciphertext'.
  25 
  26 Bob takes inputs in_k, ciphertext and tries to reconstruct
  27 the message.
  28 
  29 Eve is an adversarial network that takes input ciphertext
  30 and also tries to reconstruct the message.
  31 
  32 The main function attempts to train these networks and then
  33 evaluates them, all on random plaintext and key values.
  34 
  35 """

  49 flags.DEFINE_integer('batch_size', 4096, 'Batch size')

  53 # Input and output configuration.
  54 TEXT_SIZE = 16
  55 KEY_SIZE = 16
  56 
  57 # Training parameters.
  58 ITERS_PER_ACTOR = 1
  59 EVE_MULTIPLIER = 2  # Train Eve 2x for every step of Alice/Bob
  60 # Train until either max loops or Alice/Bob "good enough":
  61 MAX_TRAINING_LOOPS = 850000
  62 BOB_LOSS_THRESH = 0.02  # Exit when Bob loss < 0.02 and Eve > 7.7 bits
  63 EVE_LOSS_THRESH = 7.7
  64 
  65 # Logging and evaluation.
  66 PRINT_EVERY = 200  # In training, log every 200 steps.
  67 EVE_EXTRA_ROUNDS = 2000  # At end, train eve a bit more.
  68 RETRAIN_EVE_ITERS = 10000  # Retrain eve up to ITERS*LOOPS times.
  69 RETRAIN_EVE_LOOPS = 25  # With an evaluation each loop
  70 NUMBER_OF_EVE_RESETS = 5  # And do this up to 5 times with a fresh eve.
  71 # Use EVAL_BATCHES samples each time we check accuracy.
  72 EVAL_BATCHES = 1

  75 def batch_of_random_bools(batch_size, n):
  76   """Return a batch of random "boolean" numbers.
  77 
  78   Args:
  79     batch_size:  Batch size dimension of returned tensor.
  80     n:  number of entries per batch.
  81 
  82   Returns:
  83     A [batch_size, n] tensor of "boolean" numbers, where each number is
  84     preresented as -1 or 1.
  85   """
  86 
  87   as_int = tf.random.uniform(
  88       [batch_size, n], minval=0, maxval=2, dtype=tf.int32)
  89   expanded_range = (as_int * 2) - 1
  90   return tf.cast(expanded_range, tf.float32)

  93 class AdversarialCrypto(object):
  94   """Primary model implementation class for Adversarial Neural Crypto.
  95 
  96   This class contains the code for the model itself,
  97   and when created, plumbs the pathways from Alice to Bob and
  98   Eve, creates the optimizers and loss functions, etc.
  99 
 100   Attributes:
 101     eve_loss:  Eve's loss function.
 102     bob_loss:  Bob's loss function.  Different units from eve_loss.
 103     eve_optimizer:  A tf op that runs Eve's optimizer.
 104     bob_optimizer:  A tf op that runs Bob's optimizer.
 105     bob_reconstruction_loss:  Bob's message reconstruction loss,
 106       which is comparable to eve_loss.
 107     reset_eve_vars:  Execute this op to completely reset Eve.
 108   """
 109 
 110   def get_message_and_key(self):
 111     """Generate random pseudo-boolean key and message values."""
 112 
 113     batch_size = tf.compat.v1.placeholder_with_default(FLAGS.batch_size, shape=[])
 114 
 115     in_m = batch_of_random_bools(batch_size, TEXT_SIZE)
 116     in_k = batch_of_random_bools(batch_size, KEY_SIZE)
 117     return in_m, in_k
 118 
 119   def model(self, collection, message, key=None):
 120     """The model for Alice, Bob, and Eve.  If key=None, the first fully connected layer
 121     takes only the message as inputs.  Otherwise, it uses both the key
 122     and the message.
 123 
 124     Args:
 125       collection:  The graph keys collection to add new vars to.
 126       message:  The input message to process.
 127       key:  The input key (if any) to use.
 128     """
 129 
 130     if key is not None:
 131       combined_message = tf.concat(axis=1, values=[message, key])
 132     else:
 133       combined_message = message
 134 
 135     # Ensure that all variables created are in the specified collection.
 136     with tf.contrib.framework.arg_scope(
 137         [tf.contrib.layers.fully_connected, tf.contrib.layers.conv2d],
 138         variables_collections=[collection]):
 139 
 140       fc = tf.contrib.layers.fully_connected(
 141           combined_message,
 142           TEXT_SIZE + KEY_SIZE,
 143           biases_initializer=tf.constant_initializer(0.0),
 144           activation_fn=None)
 145 
 146       # Perform a sequence of 1D convolutions (by expanding the message out to 2D
 147       # and then squeezing it back down).
 148       fc = tf.expand_dims(fc, 2) # 2D
 149       fc = tf.expand_dims(fc, 3) # 3D -- conv2d needs a depth
 150       # 2,1 -> 1,2
 151       conv = tf.contrib.layers.conv2d(
 152           fc, 2, 2, 2, 'SAME', activation_fn=tf.nn.sigmoid)
 153       # 1,2 -> 1, 2
 154       conv = tf.contrib.layers.conv2d(
 155           conv, 2, 1, 1, 'SAME', activation_fn=tf.nn.sigmoid)
 156       # 1,2 -> 1, 1
 157       conv = tf.contrib.layers.conv2d(
 158           conv, 1, 1, 1, 'SAME', activation_fn=tf.nn.tanh)
 159       conv = tf.squeeze(conv, 3)
 160       conv = tf.squeeze(conv, 2)
 161       return conv
 162 
 163   def __init__(self):
 164     in_m, in_k = self.get_message_and_key()
 165     encrypted = self.model('alice', in_m, in_k)
 166     decrypted = self.model('bob', encrypted, in_k)
 167     eve_out = self.model('eve', encrypted, None)
 168 
 169     self.reset_eve_vars = tf.group(
 170         *[w.initializer for w in tf.compat.v1.get_collection('eve')])
 171 
 172     optimizer = tf.compat.v1.train.AdamOptimizer(learning_rate=FLAGS.learning_rate)
 173 
 174     # Eve's goal is to decrypt the entire message:
 175     eve_bits_wrong = tf.reduce_sum(
 176         tf.abs((eve_out + 1.0) / 2.0 - (in_m + 1.0) / 2.0), [1])
 177     self.eve_loss = tf.reduce_sum(eve_bits_wrong)
 178     self.eve_optimizer = optimizer.minimize(
 179         self.eve_loss, var_list=tf.compat.v1.get_collection('eve'))
 180 
 181     # Alice and Bob want to be accurate...
 182     self.bob_bits_wrong = tf.reduce_sum(
 183         tf.abs((decrypted + 1.0) / 2.0 - (in_m + 1.0) / 2.0), [1])
 184     # ... and to not let Eve do better than guessing.
 185     self.bob_reconstruction_loss = tf.reduce_sum(self.bob_bits_wrong)
 186     bob_eve_error_deviation = tf.abs(float(TEXT_SIZE) / 2.0 - eve_bits_wrong)
 187     # 7-9 bits wrong is OK too, so we squish the error function a bit.
 188     # Without doing this, we often tend to hang out at 0.25 / 7.5 error,
 189     # and it seems bad to have continued, high communication error.
 190     bob_eve_loss = tf.reduce_sum(
 191         tf.square(bob_eve_error_deviation) / (TEXT_SIZE / 2)**2)
 192 
 193     # Rescale the losses to [0, 1] per example and combine.
 194     self.bob_loss = (self.bob_reconstruction_loss / TEXT_SIZE + bob_eve_loss)
 195 
 196     self.bob_optimizer = optimizer.minimize(
 197         self.bob_loss,
 198         var_list=(tf.compat.v1.get_collection('alice') + tf.compat.v1.get_collection('bob')))
 199

 201 def doeval(s, ac, n, itercount):
 202   """Evaluate the current network on n batches of random examples.
 203 
 204   Args:
 205     s:  The current TensorFlow session
 206     ac: an instance of the AdversarialCrypto class
 207     n:  The number of iterations to run.
 208     itercount: Iteration count label for logging.
 209 
 210   Returns:
 211     Bob and Eve's loss, as a percent of bits incorrect.
 212   """
 213 
 214   bob_loss_accum = 0
 215   eve_loss_accum = 0
 216   for _ in xrange(n):
 217     bl, el = s.run([ac.bob_reconstruction_loss, ac.eve_loss])
 218     bob_loss_accum += bl
 219     eve_loss_accum += el
 220   bob_loss_percent = bob_loss_accum / (n * FLAGS.batch_size)
 221   eve_loss_percent = eve_loss_accum / (n * FLAGS.batch_size)
 222   print('%10d\t%20.2f\t%20.2f'%(itercount, bob_loss_percent, eve_loss_percent))
 223   sys.stdout.flush()
 224   return bob_loss_percent, eve_loss_percent

 227 def train_until_thresh(s, ac):
 228   for j in xrange(MAX_TRAINING_LOOPS):
 229     for _ in xrange(ITERS_PER_ACTOR):
 230       s.run(ac.bob_optimizer)
 231     for _ in xrange(ITERS_PER_ACTOR * EVE_MULTIPLIER):
 232       s.run(ac.eve_optimizer)
 233     if j % PRINT_EVERY == 0:
 234       bob_avg_loss, eve_avg_loss = doeval(s, ac, EVAL_BATCHES, j)
 235       if (bob_avg_loss < BOB_LOSS_THRESH and eve_avg_loss > EVE_LOSS_THRESH):
 236         print('Target losses achieved.')
 237         return True
 238   return False

 241 def train_and_evaluate():
 242   """Run the full training and evaluation loop."""
 243 
 244   ac = AdversarialCrypto()
 245   init = tf.compat.v1.global_variables_initializer()
 246 
 247   with tf.compat.v1.Session() as s:
 248     s.run(init)
 249     print('# Batch size: ', FLAGS.batch_size)
 250     print('# %10s\t%20s\t%20s'%("Iter","Bob_Recon_Error","Eve_Recon_Error"))
 251 
 252     if train_until_thresh(s, ac):
 253       for _ in xrange(EVE_EXTRA_ROUNDS):
 254         s.run(ac.eve_optimizer)
 255       print('Loss after eve extra training:')
 256       doeval(s, ac, EVAL_BATCHES * 2, 0)
 257       for _ in xrange(NUMBER_OF_EVE_RESETS):
 258         print('Resetting Eve')
 259         s.run(ac.reset_eve_vars)
 260         eve_counter = 0
 261         for _ in xrange(RETRAIN_EVE_LOOPS):
 262           for _ in xrange(RETRAIN_EVE_ITERS):
 263             eve_counter += 1
 264             s.run(ac.eve_optimizer)
 265           doeval(s, ac, EVAL_BATCHES, eve_counter)
 266         doeval(s, ac, EVAL_BATCHES, eve_counter)
