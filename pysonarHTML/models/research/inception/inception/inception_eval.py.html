  34 FLAGS = tf.app.flags.FLAGS

  36 tf.app.flags.DEFINE_string('eval_dir', '/tmp/imagenet_eval',
  37                            """Directory where to write event logs.""")
  38 tf.app.flags.DEFINE_string('checkpoint_dir', '/tmp/imagenet_train',

  42 tf.app.flags.DEFINE_integer('eval_interval_secs', 60 * 5,
  43                             """How often to run the eval.""")
  44 tf.app.flags.DEFINE_boolean('run_once', False,
  45                             """Whether to run eval only once.""")

  52                            """Either 'validation' or 'train'.""")

  55 def _eval_once(saver, summary_writer, top_1_op, top_5_op, summary_op):
  56   """Runs Eval once.
  57 
  58   Args:
  59     saver: Saver.
  60     summary_writer: Summary writer.
  61     top_1_op: Top 1 op.
  62     top_5_op: Top 5 op.
  63     summary_op: Summary op.
  64   """
  65   with tf.Session() as sess:
  66     ckpt = tf.train.get_checkpoint_state(FLAGS.checkpoint_dir)
  67     if ckpt and ckpt.model_checkpoint_path:
  68       if os.path.isabs(ckpt.model_checkpoint_path):
  69         # Restores from checkpoint with absolute path.
  70         saver.restore(sess, ckpt.model_checkpoint_path)
  71       else:
  72         # Restores from checkpoint with relative path.
  73         saver.restore(sess, os.path.join(FLAGS.checkpoint_dir,
  74                                          ckpt.model_checkpoint_path))
  75 
  76       # Assuming model_checkpoint_path looks something like:
  77       #   /my-favorite-path/imagenet_train/model.ckpt-0,
  78       # extract global_step from it.
  79       global_step = ckpt.model_checkpoint_path.split('/')[-1].split('-')[-1]
  80       print('Successfully loaded model from %s at step=%s.' %
  81             (ckpt.model_checkpoint_path, global_step))
  82     else:
  83       print('No checkpoint file found')
  84       return
  85 
  86     # Start the queue runners.
  87     coord = tf.train.Coordinator()
  88     try:
  89       threads = []
  90       for qr in tf.get_collection(tf.GraphKeys.QUEUE_RUNNERS):
  91         threads.extend(qr.create_threads(sess, coord=coord, daemon=True,
  92                                          start=True))
  93 
  94       num_iter = int(math.ceil(FLAGS.num_examples / FLAGS.batch_size))
  95       # Counts the number of correct predictions.
  96       count_top_1 = 0.0
  97       count_top_5 = 0.0
  98       total_sample_count = num_iter * FLAGS.batch_size
  99       step = 0
 100 
 101       print('%s: starting evaluation on (%s).' % (datetime.now(), FLAGS.subset))
 102       start_time = time.time()
 103       while step < num_iter and not coord.should_stop():
 104         top_1, top_5 = sess.run([top_1_op, top_5_op])
 105         count_top_1 += np.sum(top_1)
 106         count_top_5 += np.sum(top_5)
 107         step += 1
 108         if step % 20 == 0:
 109           duration = time.time() - start_time
 110           sec_per_batch = duration / 20.0
 111           examples_per_sec = FLAGS.batch_size / sec_per_batch
 112           print('%s: [%d batches out of %d] (%.1f examples/sec; %.3f'
 113                 'sec/batch)' % (datetime.now(), step, num_iter,
 114                                 examples_per_sec, sec_per_batch))
 115           start_time = time.time()
 116 
 117       # Compute precision @ 1.
 118       precision_at_1 = count_top_1 / total_sample_count
 119       recall_at_5 = count_top_5 / total_sample_count
 120       print('%s: precision @ 1 = %.4f recall @ 5 = %.4f [%d examples]' %
 121             (datetime.now(), precision_at_1, recall_at_5, total_sample_count))
 122 
 123       summary = tf.Summary()
 124       summary.ParseFromString(sess.run(summary_op))
 125       summary.value.add(tag='Precision @ 1', simple_value=precision_at_1)
 126       summary.value.add(tag='Recall @ 5', simple_value=recall_at_5)
 127       summary_writer.add_summary(summary, global_step)
 128 
 129     except Exception as e:  # pylint: disable=broad-except
 130       coord.request_stop(e)
 131 
 132     coord.request_stop()
 133     coord.join(threads, stop_grace_period_secs=10)

 136 def evaluate(dataset):
 137   """Evaluate model on Dataset for a number of steps."""
 138   with tf.Graph().as_default():
 139     # Get images and labels from the dataset.
 140     images, labels = image_processing.inputs(dataset)
 141 
 142     # Number of classes in the Dataset label set plus 1.
 143     # Label 0 is reserved for an (unused) background class.
 144     num_classes = dataset.num_classes() + 1
 145 
 146     # Build a Graph that computes the logits predictions from the
 147     # inference model.
 148     logits, _ = inception.inference(images, num_classes)
 149 
 150     # Calculate predictions.
 151     top_1_op = tf.nn.in_top_k(logits, labels, 1)
 152     top_5_op = tf.nn.in_top_k(logits, labels, 5)
 153 
 154     # Restore the moving average version of the learned variables for eval.
 155     variable_averages = tf.train.ExponentialMovingAverage(
 156         inception.MOVING_AVERAGE_DECAY)
 157     variables_to_restore = variable_averages.variables_to_restore()
 158     saver = tf.train.Saver(variables_to_restore)
 159 
 160     # Build the summary operation based on the TF collection of Summaries.
 161     summary_op = tf.summary.merge_all()
 162 
 163     graph_def = tf.get_default_graph().as_graph_def()
 164     summary_writer = tf.summary.FileWriter(FLAGS.eval_dir,
 165                                             graph_def=graph_def)
 166 
 167     while True:
 168       _eval_once(saver, summary_writer, top_1_op, top_5_op, summary_op)
 169       if FLAGS.run_once:
 170         break
 171       time.sleep(FLAGS.eval_interval_secs)
