  16 """Converts image data to TFRecords file format with Example protos.
  17 
  18 The image data set is expected to reside in JPEG files located in the
  19 following directory structure.
  20 
  21   data_dir/label_0/image0.jpeg
  22   data_dir/label_0/image1.jpg
  23   ...
  24   data_dir/label_1/weird-image.jpeg
  25   data_dir/label_1/my-image.jpeg
  26   ...
  27 
  28 where the sub-directory is the unique label associated with these images.
  29 
  30 This TensorFlow script converts the training and evaluation data into
  31 a sharded data set consisting of TFRecord files
  32 
  33   train_directory/train-00000-of-01024
  34   train_directory/train-00001-of-01024
  35   ...
  36   train_directory/train-01023-of-01024
  37 
  38 and
  39 
  40   validation_directory/validation-00000-of-00128
  41   validation_directory/validation-00001-of-00128
  42   ...
  43   validation_directory/validation-00127-of-00128
  44 
  45 where we have selected 1024 and 128 shards for each data set. Each record
  46 within the TFRecord file is a serialized Example proto. The Example proto
  47 contains the following fields:
  48 
  49   image/encoded: string containing JPEG encoded image in RGB colorspace
  50   image/height: integer, image height in pixels
  51   image/width: integer, image width in pixels
  52   image/colorspace: string, specifying the colorspace, always 'RGB'
  53   image/channels: integer, specifying the number of channels, always 3
  54   image/format: string, specifying the format, always 'JPEG'
  55 
  56   image/filename: string containing the basename of the image file
  57             e.g. 'n01440764_10026.JPEG' or 'ILSVRC2012_val_00000293.JPEG'
  58   image/class/label: integer specifying the index in a classification layer.
  59     The label ranges from [0, num_labels] where 0 is unused and left as
  60     the background class.
  61   image/class/text: string specifying the human-readable version of the label
  62     e.g. 'dog'
  63 
  64 If your data set involves bounding boxes, please look at build_imagenet_data.py.
  65 """

 107 def _int64_feature(value):
 108   """Wrapper for inserting int64 features into Example proto."""
 109   if not isinstance(value, list):
 110     value = [value]
 111   return tf.train.Feature(int64_list=tf.train.Int64List(value=value))
 112 
 113 
 114 def _bytes_feature(value):
 115   """Wrapper for inserting bytes features into Example proto."""
 116   return tf.train.Feature(bytes_list=tf.train.BytesList(value=[value]))

 119 def _convert_to_example(filename, image_buffer, label, text, height, width):
 120   """Build an Example proto for an example.
 121 
 122   Args:
 123     filename: string, path to an image file, e.g., '/path/to/example.JPG'
 124     image_buffer: string, JPEG encoding of RGB image
 125     label: integer, identifier for the ground truth for the network
 126     text: string, unique human-readable, e.g. 'dog'
 127     height: integer, image height in pixels
 128     width: integer, image width in pixels
 129   Returns:
 130     Example proto
 131   """
 132 
 133   colorspace = 'RGB'
 134   channels = 3
 135   image_format = 'JPEG'
 136 
 137   example = tf.train.Example(features=tf.train.Features(feature={
 138       'image/height': _int64_feature(height),
 139       'image/width': _int64_feature(width),
 140       'image/colorspace': _bytes_feature(tf.compat.as_bytes(colorspace)),
 141       'image/channels': _int64_feature(channels),
 142       'image/class/label': _int64_feature(label),
 143       'image/class/text': _bytes_feature(tf.compat.as_bytes(text)),
 144       'image/format': _bytes_feature(tf.compat.as_bytes(image_format)),
 145       'image/filename': _bytes_feature(tf.compat.as_bytes(os.path.basename(filename))),
 146       'image/encoded': _bytes_feature(tf.compat.as_bytes(image_buffer))}))
 147   return example

 150 class ImageCoder(object):
 151   """Helper class that provides TensorFlow image coding utilities."""
 152 
 153   def __init__(self):
 154     # Create a single Session to run all image coding calls.
 155     self._sess = tf.Session()
 156 
 157     # Initializes function that converts PNG to JPEG data.
 158     self._png_data = tf.placeholder(dtype=tf.string)
 159     image = tf.image.decode_png(self._png_data, channels=3)
 160     self._png_to_jpeg = tf.image.encode_jpeg(image, format='rgb', quality=100)
 161 
 162     # Initializes function that decodes RGB JPEG data.
 163     self._decode_jpeg_data = tf.placeholder(dtype=tf.string)
 164     self._decode_jpeg = tf.image.decode_jpeg(self._decode_jpeg_data, channels=3)
 165 
 166   def png_to_jpeg(self, image_data):
 167     return self._sess.run(self._png_to_jpeg,
 168                           feed_dict={self._png_data: image_data})
 169 
 170   def decode_jpeg(self, image_data):
 171     image = self._sess.run(self._decode_jpeg,
 172                            feed_dict={self._decode_jpeg_data: image_data})
 173     assert len(image.shape) == 3
 174     assert image.shape[2] == 3
 175     return image

 179   """Determine if a file contains a PNG format image.

 181   Args:

 184   Returns:

 190 def _process_image(filename, coder):
 191   """Process a single image file.
 192 
 193   Args:
 194     filename: string, path to an image file e.g., '/path/to/example.JPG'.
 195     coder: instance of ImageCoder to provide TensorFlow image coding utils.
 196   Returns:
 197     image_buffer: string, JPEG encoding of RGB image.
 198     height: integer, image height in pixels.
 199     width: integer, image width in pixels.
 200   """
 201   # Read the image file.
 202   with tf.gfile.FastGFile(filename, 'rb') as f:
 203     image_data = f.read()
 204 
 205   # Convert any PNG to JPEG's for consistency.
 206   if _is_png(filename):
 207     print('Converting PNG to JPEG for %s' % filename)
 208     image_data = coder.png_to_jpeg(image_data)
 209 
 210   # Decode the RGB JPEG.
 211   image = coder.decode_jpeg(image_data)
 212 
 213   # Check that image converted to RGB
 214   assert len(image.shape) == 3
 215   height = image.shape[0]
 216   width = image.shape[1]
 217   assert image.shape[2] == 3
 218 
 219   return image_data, height, width

 222 def _process_image_files_batch(coder, thread_index, ranges, name, filenames,
 223                                texts, labels, num_shards):
 224   """Processes and saves list of images as TFRecord in 1 thread.
 225 
 226   Args:
 227     coder: instance of ImageCoder to provide TensorFlow image coding utils.
 228     thread_index: integer, unique batch to run index is within [0, len(ranges)).
 229     ranges: list of pairs of integers specifying ranges of each batches to
 230       analyze in parallel.
 231     name: string, unique identifier specifying the data set
 232     filenames: list of strings; each string is a path to an image file
 233     texts: list of strings; each string is human readable, e.g. 'dog'
 234     labels: list of integer; each integer identifies the ground truth
 235     num_shards: integer number of shards for this data set.
 236   """
 237   # Each thread produces N shards where N = int(num_shards / num_threads).
 238   # For instance, if num_shards = 128, and the num_threads = 2, then the first
 239   # thread would produce shards [0, 64).
 240   num_threads = len(ranges)
 241   assert not num_shards % num_threads
 242   num_shards_per_batch = int(num_shards / num_threads)
 243 
 244   shard_ranges = np.linspace(ranges[thread_index][0],
 245                              ranges[thread_index][1],
 246                              num_shards_per_batch + 1).astype(int)
 247   num_files_in_thread = ranges[thread_index][1] - ranges[thread_index][0]
 248 
 249   counter = 0
 250   for s in range(num_shards_per_batch):
 251     # Generate a sharded version of the file name, e.g. 'train-00002-of-00010'
 252     shard = thread_index * num_shards_per_batch + s
 253     output_filename = '%s-%.5d-of-%.5d' % (name, shard, num_shards)
 254     output_file = os.path.join(FLAGS.output_directory, output_filename)
 255     writer = tf.python_io.TFRecordWriter(output_file)
 256 
 257     shard_counter = 0
 258     files_in_shard = np.arange(shard_ranges[s], shard_ranges[s + 1], dtype=int)
 259     for i in files_in_shard:
 260       filename = filenames[i]
 261       label = labels[i]
 262       text = texts[i]
 263 
 264       try:
 265         image_buffer, height, width = _process_image(filename, coder)
 266       except Exception as e:
 267         print(e)
 268         print('SKIPPED: Unexpected error while decoding %s.' % filename)
 269         continue
 270 
 271       example = _convert_to_example(filename, image_buffer, label,
 272                                     text, height, width)
 273       writer.write(example.SerializeToString())
 274       shard_counter += 1
 275       counter += 1
 276 
 277       if not counter % 1000:
 278         print('%s [thread %d]: Processed %d of %d images in thread batch.' %
 279               (datetime.now(), thread_index, counter, num_files_in_thread))
 280         sys.stdout.flush()
 281 
 282     writer.close()
 283     print('%s [thread %d]: Wrote %d images to %s' %
 284           (datetime.now(), thread_index, shard_counter, output_file))
 285     sys.stdout.flush()
 286     shard_counter = 0
 287   print('%s [thread %d]: Wrote %d images to %d shards.' %
 288         (datetime.now(), thread_index, counter, num_files_in_thread))
 289   sys.stdout.flush()

 292 def _process_image_files(name, filenames, texts, labels, num_shards):
 293   """Process and save list of images as TFRecord of Example protos.
 294 
 295   Args:
 296     name: string, unique identifier specifying the data set
 297     filenames: list of strings; each string is a path to an image file
 298     texts: list of strings; each string is human readable, e.g. 'dog'
 299     labels: list of integer; each integer identifies the ground truth
 300     num_shards: integer number of shards for this data set.
 301   """
 302   assert len(filenames) == len(texts)
 303   assert len(filenames) == len(labels)
 304 
 305   # Break all images into batches with a [ranges[i][0], ranges[i][1]].
 306   spacing = np.linspace(0, len(filenames), FLAGS.num_threads + 1).astype(np.int)
 307   ranges = []
 308   for i in range(len(spacing) - 1):
 309     ranges.append([spacing[i], spacing[i + 1]])
 310 
 311   # Launch a thread for each batch.
 312   print('Launching %d threads for spacings: %s' % (FLAGS.num_threads, ranges))
 313   sys.stdout.flush()
 314 
 315   # Create a mechanism for monitoring when all threads are finished.
 316   coord = tf.train.Coordinator()
 317 
 318   # Create a generic TensorFlow-based utility for converting all image codings.
 319   coder = ImageCoder()
 320 
 321   threads = []
 322   for thread_index in range(len(ranges)):
 323     args = (coder, thread_index, ranges, name, filenames,
 324             texts, labels, num_shards)
 325     t = threading.Thread(target=_process_image_files_batch, args=args)
 326     t.start()
 327     threads.append(t)
 328 
 329   # Wait for all the threads to terminate.
 330   coord.join(threads)
 331   print('%s: Finished writing all %d images in data set.' %
 332         (datetime.now(), len(filenames)))
 333   sys.stdout.flush()

 336 def _find_image_files(data_dir, labels_file):
 337   """Build a list of all images files and labels in the data set.
 338 
 339   Args:
 340     data_dir: string, path to the root directory of images.
 341 
 342       Assumes that the image data set resides in JPEG files located in
 343       the following directory structure.
 344 
 345         data_dir/dog/another-image.JPEG
 346         data_dir/dog/my-image.jpg
 347 
 348       where 'dog' is the label associated with these images.
 349 
 350     labels_file: string, path to the labels file.
 351 
 352       The list of valid labels are held in this file. Assumes that the file
 353       contains entries as such:
 354         dog
 355         cat
 356         flower
 357       where each line corresponds to a label. We map each label contained in
 358       the file to an integer starting with the integer 0 corresponding to the
 359       label contained in the first line.
 360 
 361   Returns:
 362     filenames: list of strings; each string is a path to an image file.
 363     texts: list of strings; each string is the class, e.g. 'dog'
 364     labels: list of integer; each integer identifies the ground truth.
 365   """
 366   print('Determining list of input files and labels from %s.' % data_dir)
 367   unique_labels = [l.strip() for l in tf.gfile.FastGFile(
 368       labels_file, 'r').readlines()]
 369 
 370   labels = []
 371   filenames = []
 372   texts = []
 373 
 374   # Leave label index 0 empty as a background class.
 375   label_index = 1
 376 
 377   # Construct the list of JPEG files and labels.
 378   for text in unique_labels:
 379     jpeg_file_path = '%s/%s/*' % (data_dir, text)
 380     matching_files = tf.gfile.Glob(jpeg_file_path)
 381 
 382     labels.extend([label_index] * len(matching_files))
 383     texts.extend([text] * len(matching_files))
 384     filenames.extend(matching_files)
 385 
 386     if not label_index % 100:
 387       print('Finished finding files in %d of %d classes.' % (
 388           label_index, len(labels)))
 389     label_index += 1
 390 
 391   # Shuffle the ordering of all image files in order to guarantee
 392   # random ordering of the images with respect to label in the
 393   # saved TFRecord files. Make the randomization repeatable.
 394   shuffled_index = list(range(len(filenames)))
 395   random.seed(12345)
 396   random.shuffle(shuffled_index)
 397 
 398   filenames = [filenames[i] for i in shuffled_index]
 399   texts = [texts[i] for i in shuffled_index]
 400   labels = [labels[i] for i in shuffled_index]
 401 
 402   print('Found %d JPEG files across %d labels inside %s.' %
 403         (len(filenames), len(unique_labels), data_dir))
 404   return filenames, texts, labels

 407 def _process_dataset(name, directory, num_shards, labels_file):
 408   """Process a complete data set and save it as a TFRecord.
 409 
 410   Args:
 411     name: string, unique identifier specifying the data set.
 412     directory: string, root path to the data set.
 413     num_shards: integer number of shards for this data set.
 414     labels_file: string, path to the labels file.
 415   """
 416   filenames, texts, labels = _find_image_files(directory, labels_file)
 417   _process_image_files(name, filenames, texts, labels, num_shards)
