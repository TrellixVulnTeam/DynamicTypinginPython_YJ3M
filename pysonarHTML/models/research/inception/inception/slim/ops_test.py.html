 591   def testComputeMovingVars(self):
 592     height, width = 3, 3
 593     with self.test_session() as sess:
 594       image_shape = (10, height, width, 3)
 595       image_values = np.random.rand(*image_shape)
 596       expected_mean = np.mean(image_values, axis=(0, 1, 2))
 597       expected_var = np.var(image_values, axis=(0, 1, 2))
 598       images = tf.constant(image_values, shape=image_shape, dtype=tf.float32)
 599       output = ops.batch_norm(images, decay=0.1)
 600       update_ops = tf.get_collection(ops.UPDATE_OPS_COLLECTION)
 601       with tf.control_dependencies(update_ops):
 602         output = tf.identity(output)
 603       # Initialize all variables
 604       sess.run(tf.global_variables_initializer())
 605       moving_mean = variables.get_variables('BatchNorm/moving_mean')[0]
 606       moving_variance = variables.get_variables('BatchNorm/moving_variance')[0]
 607       mean, variance = sess.run([moving_mean, moving_variance])
 608       # After initialization moving_mean == 0 and moving_variance == 1.
 609       self.assertAllClose(mean, [0] * 3)
 610       self.assertAllClose(variance, [1] * 3)
 611       for _ in range(10):
 612         sess.run([output])
 613       mean = moving_mean.eval()
 614       variance = moving_variance.eval()
 615       # After 10 updates with decay 0.1 moving_mean == expected_mean and
 616       # moving_variance == expected_var.
 617       self.assertAllClose(mean, expected_mean)
 618       self.assertAllClose(variance, expected_var)
