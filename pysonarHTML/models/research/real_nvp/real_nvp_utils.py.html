  29 def stable_var(input_, mean=None, axes=[0]):
  30     """Numerically more stable variance computation."""
  31     if mean is None:
  32         mean = tf.reduce_mean(input_, axes)
  33     res = tf.square(input_ - mean)
  34     max_sqr = tf.reduce_max(res, axes)
  35     res /= max_sqr
  36     res = tf.reduce_mean(res, axes)
  37     res *= max_sqr
  38 
  39     return res

  42 def variable_on_cpu(name, shape, initializer, trainable=True):
  43     """Helper to create a Variable stored on CPU memory.
  44 
  45     Args:
  46             name: name of the variable
  47             shape: list of ints
  48             initializer: initializer for Variable
  49             trainable: boolean defining if the variable is for training
  50     Returns:
  51             Variable Tensor
  52     """
  53     var = tf.get_variable(
  54         name, shape, initializer=initializer, trainable=trainable)
  55     return var

  59 def conv_layer(input_,
  60                filter_size,
  61                dim_in,
  62                dim_out,
  63                name,
  64                stddev=1e-2,
  65                strides=[1, 1, 1, 1],
  66                padding="SAME",
  67                nonlinearity=None,
  68                bias=False,
  69                weight_norm=False,
  70                scale=False):
  71     """Convolutional layer."""
  72     with tf.variable_scope(name) as scope:
  73         weights = variable_on_cpu(
  74             "weights",
  75             filter_size + [dim_in, dim_out],
  76             tf.random_uniform_initializer(
  77                 minval=-stddev, maxval=stddev))
  78         # weight normalization
  79         if weight_norm:
  80             weights /= tf.sqrt(tf.reduce_sum(tf.square(weights), [0, 1, 2]))
  81             if scale:
  82                 magnitude = variable_on_cpu(
  83                     "magnitude", [dim_out],
  84                     tf.constant_initializer(
  85                         stddev * numpy.sqrt(dim_in * numpy.prod(filter_size) / 12.)))
  86                 weights *= magnitude
  87         res = input_
  88         # handling filter size bigger than image size
  89         if hasattr(input_, "shape"):
  90             if input_.get_shape().as_list()[1] < filter_size[0]:
  91                 pad_1 = tf.zeros([
  92                     input_.get_shape().as_list()[0],
  93                     filter_size[0] - input_.get_shape().as_list()[1],
  94                     input_.get_shape().as_list()[2],
  95                     input_.get_shape().as_list()[3]
  96                 ])
  97                 pad_2 = tf.zeros([
  98                     input_.get_shape().as_list[0],
  99                     filter_size[0],
 100                     filter_size[1] - input_.get_shape().as_list()[2],
 101                     input_.get_shape().as_list()[3]
 102                 ])
 103                 res = tf.concat(axis=1, values=[pad_1, res])
 104                 res = tf.concat(axis=2, values=[pad_2, res])
 105         res = tf.nn.conv2d(
 106             input=res,
 107             filter=weights,
 108             strides=strides,
 109             padding=padding,
 110             name=scope.name)
 111 
 112         if hasattr(input_, "shape"):
 113             if input_.get_shape().as_list()[1] < filter_size[0]:
 114                 res = tf.slice(res, [
 115                     0, filter_size[0] - input_.get_shape().as_list()[1],
 116                     filter_size[1] - input_.get_shape().as_list()[2], 0
 117                 ], [-1, -1, -1, -1])
 118 
 119         if bias:
 120             biases = variable_on_cpu("biases", [dim_out], tf.constant_initializer(0.))
 121             res = tf.nn.bias_add(res, biases)
 122         if nonlinearity is not None:
 123             res = nonlinearity(res)
 124 
 125     return res

 134 def depool_2x2(input_, stride=2):
 135     """Depooling."""
 136     shape = input_.get_shape().as_list()
 137     batch_size = shape[0]
 138     height = shape[1]
 139     width = shape[2]
 140     channels = shape[3]
 141     res = tf.reshape(input_, [batch_size, height, 1, width, 1, channels])
 142     res = tf.concat(
 143         axis=2, values=[res, tf.zeros([batch_size, height, stride - 1, width, 1, channels])])
 144     res = tf.concat(axis=4, values=[
 145         res, tf.zeros([batch_size, height, stride, width, stride - 1, channels])
 146     ])
 147     res = tf.reshape(res, [batch_size, stride * height, stride * width, channels])
 148 
 149     return res

 152 # random flip on a batch of images
 153 def batch_random_flip(input_):
 154     """Simultaneous horizontal random flip."""
 155     if isinstance(input_, (float, int)):
 156         return input_
 157     shape = input_.get_shape().as_list()
 158     batch_size = shape[0]
 159     height = shape[1]
 160     width = shape[2]
 161     channels = shape[3]
 162     res = tf.split(axis=0, num_or_size_splits=batch_size, value=input_)
 163     res = [elem[0, :, :, :] for elem in res]
 164     res = [tf.image.random_flip_left_right(elem) for elem in res]
 165     res = [tf.reshape(elem, [1, height, width, channels]) for elem in res]
 166     res = tf.concat(axis=0, values=res)
 167 
 168     return res

 171 # build a one hot representation corresponding to the integer tensor
 172 # the one-hot dimension is appended to the integer tensor shape
 173 def as_one_hot(input_, n_indices):
 174     """Convert indices to one-hot."""
 175     shape = input_.get_shape().as_list()
 176     n_elem = numpy.prod(shape)
 177     indices = tf.range(n_elem)
 178     indices = tf.cast(indices, tf.int64)
 179     indices_input = tf.concat(axis=0, values=[indices, tf.reshape(input_, [-1])])
 180     indices_input = tf.reshape(indices_input, [2, -1])
 181     indices_input = tf.transpose(indices_input)
 182     res = tf.sparse_to_dense(
 183         indices_input, [n_elem, n_indices], 1., 0., name="flat_one_hot")
 184     res = tf.reshape(res, [elem for elem in shape] + [n_indices])
 185 
 186     return res

 194 def squeeze_nxn(input_, n_factor=2):
 195     """Squeezing operation: reshape to convert space to channels."""
 196     if isinstance(input_, (float, int)):
 197         return input_
 198     shape = input_.get_shape().as_list()
 199     batch_size = shape[0]
 200     height = shape[1]
 201     width = shape[2]
 202     channels = shape[3]
 203     if height % n_factor != 0:
 204         raise ValueError("Height not divisible by %d." % n_factor)
 205     if width % n_factor != 0:
 206         raise ValueError("Width not divisible by %d." % n_factor)
 207     res = tf.reshape(
 208         input_,
 209         [batch_size,
 210          height // n_factor,
 211          n_factor, width // n_factor,
 212          n_factor, channels])
 213     res = tf.transpose(res, [0, 1, 3, 5, 2, 4])
 214     res = tf.reshape(
 215         res,
 216         [batch_size,
 217          height // n_factor,
 218          width // n_factor,
 219          channels * n_factor * n_factor])
 220 
 221     return res

 224 def unsqueeze_2x2(input_):
 225     """Unsqueezing operation: reshape to convert channels into space."""
 226     if isinstance(input_, (float, int)):
 227         return input_
 228     shape = input_.get_shape().as_list()
 229     batch_size = shape[0]
 230     height = shape[1]
 231     width = shape[2]
 232     channels = shape[3]
 233     if channels % 4 != 0:
 234         raise ValueError("Number of channels not divisible by 4.")
 235     res = tf.reshape(input_, [batch_size, height, width, channels // 4, 2, 2])
 236     res = tf.transpose(res, [0, 1, 4, 2, 5, 3])
 237     res = tf.reshape(res, [batch_size, 2 * height, 2 * width, channels // 4])
 238 
 239     return res

 242 # batch norm
 243 def batch_norm(input_,
 244                dim,
 245                name,
 246                scale=True,
 247                train=True,
 248                epsilon=1e-8,
 249                decay=.1,
 250                axes=[0],
 251                bn_lag=DEFAULT_BN_LAG):
 252     """Batch normalization."""
 253     # create variables
 254     with tf.variable_scope(name):
 255         var = variable_on_cpu(
 256             "var", [dim], tf.constant_initializer(1.), trainable=False)
 257         mean = variable_on_cpu(
 258             "mean", [dim], tf.constant_initializer(0.), trainable=False)
 259         step = variable_on_cpu("step", [], tf.constant_initializer(0.), trainable=False)
 260         if scale:
 261             gamma = variable_on_cpu("gamma", [dim], tf.constant_initializer(1.))
 262         beta = variable_on_cpu("beta", [dim], tf.constant_initializer(0.))
 263     # choose the appropriate moments
 264     if train:
 265         used_mean, used_var = tf.nn.moments(input_, axes, name="batch_norm")
 266         cur_mean, cur_var = used_mean, used_var
 267         if bn_lag > 0.:
 268             used_mean -= (1. - bn_lag) * (used_mean - tf.stop_gradient(mean))
 269             used_var -= (1 - bn_lag) * (used_var - tf.stop_gradient(var))
 270             used_mean /= (1. - bn_lag**(step + 1))
 271             used_var /= (1. - bn_lag**(step + 1))
 272     else:
 273         used_mean, used_var = mean, var
 274         cur_mean, cur_var = used_mean, used_var
 275 
 276     # normalize
 277     res = (input_ - used_mean) / tf.sqrt(used_var + epsilon)
 278     # de-normalize
 279     if scale:
 280         res *= gamma
 281     res += beta
 282 
 283     # update variables
 284     if train:
 285         with tf.name_scope(name, "AssignMovingAvg", [mean, cur_mean, decay]):
 286             with ops.colocate_with(mean):
 287                 new_mean = tf.assign_sub(
 288                     mean,
 289                     tf.check_numerics(decay * (mean - cur_mean), "NaN in moving mean."))
 290         with tf.name_scope(name, "AssignMovingAvg", [var, cur_var, decay]):
 291             with ops.colocate_with(var):
 292                 new_var = tf.assign_sub(
 293                     var,
 294                     tf.check_numerics(decay * (var - cur_var),
 295                                       "NaN in moving variance."))
 296         with tf.name_scope(name, "IncrementTime", [step]):
 297             with ops.colocate_with(step):
 298                 new_step = tf.assign_add(step, 1.)
 299         res += 0. * new_mean * new_var * new_step
 300 
 301     return res

 304 # batch normalization taking into account the volume transformation
 305 def batch_norm_log_diff(input_,
 306                         dim,
 307                         name,
 308                         train=True,
 309                         epsilon=1e-8,
 310                         decay=.1,
 311                         axes=[0],
 312                         reuse=None,
 313                         bn_lag=DEFAULT_BN_LAG):
 314     """Batch normalization with corresponding log determinant Jacobian."""
 315     if reuse is None:
 316         reuse = not train
 317     # create variables
 318     with tf.variable_scope(name) as scope:
 319         if reuse:
 320             scope.reuse_variables()
 321         var = variable_on_cpu(
 322             "var", [dim], tf.constant_initializer(1.), trainable=False)
 323         mean = variable_on_cpu(
 324             "mean", [dim], tf.constant_initializer(0.), trainable=False)
 325         step = variable_on_cpu("step", [], tf.constant_initializer(0.), trainable=False)
 326     # choose the appropriate moments
 327     if train:
 328         used_mean, used_var = tf.nn.moments(input_, axes, name="batch_norm")
 329         cur_mean, cur_var = used_mean, used_var
 330         if bn_lag > 0.:
 331             used_var = stable_var(input_=input_, mean=used_mean, axes=axes)
 332             cur_var = used_var
 333             used_mean -= (1 - bn_lag) * (used_mean - tf.stop_gradient(mean))
 334             used_mean /= (1. - bn_lag**(step + 1))
 335             used_var -= (1 - bn_lag) * (used_var - tf.stop_gradient(var))
 336             used_var /= (1. - bn_lag**(step + 1))
 337     else:
 338         used_mean, used_var = mean, var
 339         cur_mean, cur_var = used_mean, used_var
 340 
 341     # update variables
 342     if train:
 343         with tf.name_scope(name, "AssignMovingAvg", [mean, cur_mean, decay]):
 344             with ops.colocate_with(mean):
 345                 new_mean = tf.assign_sub(
 346                     mean,
 347                     tf.check_numerics(
 348                         decay * (mean - cur_mean), "NaN in moving mean."))
 349         with tf.name_scope(name, "AssignMovingAvg", [var, cur_var, decay]):
 350             with ops.colocate_with(var):
 351                 new_var = tf.assign_sub(
 352                     var,
 353                     tf.check_numerics(decay * (var - cur_var),
 354                                       "NaN in moving variance."))
 355         with tf.name_scope(name, "IncrementTime", [step]):
 356             with ops.colocate_with(step):
 357                 new_step = tf.assign_add(step, 1.)
 358         used_var += 0. * new_mean * new_var * new_step
 359     used_var += epsilon
 360 
 361     return used_mean, used_var

 364 def convnet(input_,
 365             dim_in,
 366             dim_hid,
 367             filter_sizes,
 368             dim_out,
 369             name,
 370             use_batch_norm=True,
 371             train=True,
 372             nonlinearity=tf.nn.relu):
 373     """Chaining of convolutional layers."""
 374     dims_in = [dim_in] + dim_hid[:-1]
 375     dims_out = dim_hid
 376     res = input_
 377 
 378     bias = (not use_batch_norm)
 379     with tf.variable_scope(name):
 380         for layer_idx in xrange(len(dim_hid)):
 381             res = conv_layer(
 382                 input_=res,
 383                 filter_size=filter_sizes[layer_idx],
 384                 dim_in=dims_in[layer_idx],
 385                 dim_out=dims_out[layer_idx],
 386                 name="h_%d" % layer_idx,
 387                 stddev=1e-2,
 388                 nonlinearity=None,
 389                 bias=bias)
 390             if use_batch_norm:
 391                 res = batch_norm(
 392                     input_=res,
 393                     dim=dims_out[layer_idx],
 394                     name="bn_%d" % layer_idx,
 395                     scale=(nonlinearity == tf.nn.relu),
 396                     train=train,
 397                     epsilon=1e-8,
 398                     axes=[0, 1, 2])
 399             if nonlinearity is not None:
 400                 res = nonlinearity(res)
 401 
 402         res = conv_layer(
 403             input_=res,
 404             filter_size=filter_sizes[-1],
 405             dim_in=dims_out[-1],
 406             dim_out=dim_out,
 407             name="out",
 408             stddev=1e-2,
 409             nonlinearity=None)
 410 
 411     return res

 415 # log-likelihood estimation
 416 def standard_normal_ll(input_):
 417     """Log-likelihood of standard Gaussian distribution."""
 418     res = -.5 * (tf.square(input_) + numpy.log(2. * numpy.pi))
 419 
 420     return res

 428 SQUEEZE_MATRIX = numpy.array([[[[1., 0., 0., 0.]], [[0., 0., 1., 0.]]],
 429                               [[[0., 0., 0., 1.]], [[0., 1., 0., 0.]]]])

 432 def squeeze_2x2_ordered(input_, reverse=False):
 433     """Squeezing operation with a controlled ordering."""
 434     shape = input_.get_shape().as_list()
 435     batch_size = shape[0]
 436     height = shape[1]
 437     width = shape[2]
 438     channels = shape[3]
 439     if reverse:
 440         if channels % 4 != 0:
 441             raise ValueError("Number of channels not divisible by 4.")
 442         channels /= 4
 443     else:
 444         if height % 2 != 0:
 445             raise ValueError("Height not divisible by 2.")
 446         if width % 2 != 0:
 447             raise ValueError("Width not divisible by 2.")
 448     weights = numpy.zeros((2, 2, channels, 4 * channels))
 449     for idx_ch in xrange(channels):
 450         slice_2 = slice(idx_ch, (idx_ch + 1))
 451         slice_3 = slice((idx_ch * 4), ((idx_ch + 1) * 4))
 452         weights[:, :, slice_2, slice_3] = SQUEEZE_MATRIX
 453     shuffle_channels = [idx_ch * 4 for idx_ch in xrange(channels)]
 454     shuffle_channels += [idx_ch * 4 + 1 for idx_ch in xrange(channels)]
 455     shuffle_channels += [idx_ch * 4 + 2 for idx_ch in xrange(channels)]
 456     shuffle_channels += [idx_ch * 4 + 3 for idx_ch in xrange(channels)]
 457     shuffle_channels = numpy.array(shuffle_channels)
 458     weights = weights[:, :, :, shuffle_channels].astype("float32")
 459     if reverse:
 460         res = tf.nn.conv2d_transpose(
 461             value=input_,
 462             filter=weights,
 463             output_shape=[batch_size, height * 2, width * 2, channels],
 464             strides=[1, 2, 2, 1],
 465             padding="SAME",
 466             name="unsqueeze_2x2")
 467     else:
 468         res = tf.nn.conv2d(
 469             input=input_,
 470             filter=weights,
 471             strides=[1, 2, 2, 1],
 472             padding="SAME",
 473             name="squeeze_2x2")
 474 
 475     return res
