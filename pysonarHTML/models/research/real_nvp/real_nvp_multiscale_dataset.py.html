  84 class HParams(object):
  85     """Dictionary of hyperparameters."""
  86     def __init__(self, **kwargs):
  87         self.dict_ = kwargs
  88         self.__dict__.update(self.dict_)
  89 
  90     def update_config(self, in_string):
  91         """Update the dictionary with a comma separated list."""
  92         pairs = in_string.split(",")
  93         pairs = [pair.split("=") for pair in pairs]
  94         for key, val in pairs:
  95             self.dict_[key] = type(self.dict_[key])(val)
  96         self.__dict__.update(self.dict_)
  97         return self
  98 
  99     def __getitem__(self, key):
 100         return self.dict_[key]
 101 
 102     def __setitem__(self, key, val):
 103         self.dict_[key] = val
 104         self.__dict__.update(self.dict_)

 132 def residual_block(input_, dim, name, use_batch_norm=True,
 133                    train=True, weight_norm=True, bottleneck=False):
 134     """Residual convolutional block."""
 135     with tf.variable_scope(name):
 136         res = input_
 137         if use_batch_norm:
 138             res = batch_norm(
 139                 input_=res, dim=dim, name="bn_in", scale=False,
 140                 train=train, epsilon=1e-4, axes=[0, 1, 2])
 141         res = tf.nn.relu(res)
 142         if bottleneck:
 143             res = conv_layer(
 144                 input_=res, filter_size=[1, 1], dim_in=dim, dim_out=dim,
 145                 name="h_0", stddev=numpy.sqrt(2. / (dim)),
 146                 strides=[1, 1, 1, 1], padding="SAME",
 147                 nonlinearity=None, bias=(not use_batch_norm),
 148                 weight_norm=weight_norm, scale=False)
 149             if use_batch_norm:
 150                 res = batch_norm(
 151                     input_=res, dim=dim,
 152                     name="bn_0", scale=False, train=train,
 153                     epsilon=1e-4, axes=[0, 1, 2])
 154             res = tf.nn.relu(res)
 155             res = conv_layer(
 156                 input_=res, filter_size=[3, 3], dim_in=dim,
 157                 dim_out=dim, name="h_1", stddev=numpy.sqrt(2. / (1. * dim)),
 158                 strides=[1, 1, 1, 1], padding="SAME", nonlinearity=None,
 159                 bias=(not use_batch_norm),
 160                 weight_norm=weight_norm, scale=False)
 161             if use_batch_norm:
 162                 res = batch_norm(
 163                     input_=res, dim=dim, name="bn_1", scale=False,
 164                     train=train, epsilon=1e-4, axes=[0, 1, 2])
 165             res = tf.nn.relu(res)
 166             res = conv_layer(
 167                 input_=res, filter_size=[1, 1], dim_in=dim, dim_out=dim,
 168                 name="out", stddev=numpy.sqrt(2. / (1. * dim)),
 169                 strides=[1, 1, 1, 1], padding="SAME", nonlinearity=None,
 170                 bias=True, weight_norm=weight_norm, scale=True)
 171         else:
 172             res = conv_layer(
 173                 input_=res, filter_size=[3, 3], dim_in=dim, dim_out=dim,
 174                 name="h_0", stddev=numpy.sqrt(2. / (dim)),
 175                 strides=[1, 1, 1, 1], padding="SAME",
 176                 nonlinearity=None, bias=(not use_batch_norm),
 177                 weight_norm=weight_norm, scale=False)
 178             if use_batch_norm:
 179                 res = batch_norm(
 180                     input_=res, dim=dim, name="bn_0", scale=False,
 181                     train=train, epsilon=1e-4, axes=[0, 1, 2])
 182             res = tf.nn.relu(res)
 183             res = conv_layer(
 184                 input_=res, filter_size=[3, 3], dim_in=dim, dim_out=dim,
 185                 name="out", stddev=numpy.sqrt(2. / (1. * dim)),
 186                 strides=[1, 1, 1, 1], padding="SAME", nonlinearity=None,
 187                 bias=True, weight_norm=weight_norm, scale=True)
 188         res += input_
 189 
 190     return res

 193 def resnet(input_, dim_in, dim, dim_out, name, use_batch_norm=True,
 194            train=True, weight_norm=True, residual_blocks=5,
 195            bottleneck=False, skip=True):
 196     """Residual convolutional network."""
 197     with tf.variable_scope(name):
 198         res = input_
 199         if residual_blocks != 0:
 200             res = conv_layer(
 201                 input_=res, filter_size=[3, 3], dim_in=dim_in, dim_out=dim,
 202                 name="h_in", stddev=numpy.sqrt(2. / (dim_in)),
 203                 strides=[1, 1, 1, 1], padding="SAME",
 204                 nonlinearity=None, bias=True,
 205                 weight_norm=weight_norm, scale=False)
 206             if skip:
 207                 out = conv_layer(
 208                     input_=res, filter_size=[1, 1], dim_in=dim, dim_out=dim,
 209                     name="skip_in", stddev=numpy.sqrt(2. / (dim)),
 210                     strides=[1, 1, 1, 1], padding="SAME",
 211                     nonlinearity=None, bias=True,
 212                     weight_norm=weight_norm, scale=True)
 213 
 214             # residual blocks
 215             for idx_block in xrange(residual_blocks):
 216                 res = residual_block(res, dim, "block_%d" % idx_block,
 217                                      use_batch_norm=use_batch_norm, train=train,
 218                                      weight_norm=weight_norm,
 219                                      bottleneck=bottleneck)
 220                 if skip:
 221                     out += conv_layer(
 222                         input_=res, filter_size=[1, 1], dim_in=dim, dim_out=dim,
 223                         name="skip_%d" % idx_block, stddev=numpy.sqrt(2. / (dim)),
 224                         strides=[1, 1, 1, 1], padding="SAME",
 225                         nonlinearity=None, bias=True,
 226                         weight_norm=weight_norm, scale=True)
 227             # outputs
 228             if skip:
 229                 res = out
 230             if use_batch_norm:
 231                 res = batch_norm(
 232                     input_=res, dim=dim, name="bn_pre_out", scale=False,
 233                     train=train, epsilon=1e-4, axes=[0, 1, 2])
 234             res = tf.nn.relu(res)
 235             res = conv_layer(
 236                 input_=res, filter_size=[1, 1], dim_in=dim,
 237                 dim_out=dim_out,
 238                 name="out", stddev=numpy.sqrt(2. / (1. * dim)),
 239                 strides=[1, 1, 1, 1], padding="SAME",
 240                 nonlinearity=None, bias=True,
 241                 weight_norm=weight_norm, scale=True)
 242         else:
 243             if bottleneck:
 244                 res = conv_layer(
 245                     input_=res, filter_size=[1, 1], dim_in=dim_in, dim_out=dim,
 246                     name="h_0", stddev=numpy.sqrt(2. / (dim_in)),
 247                     strides=[1, 1, 1, 1], padding="SAME",
 248                     nonlinearity=None, bias=(not use_batch_norm),
 249                     weight_norm=weight_norm, scale=False)
 250                 if use_batch_norm:
 251                     res = batch_norm(
 252                         input_=res, dim=dim, name="bn_0", scale=False,
 253                         train=train, epsilon=1e-4, axes=[0, 1, 2])
 254                 res = tf.nn.relu(res)
 255                 res = conv_layer(
 256                     input_=res, filter_size=[3, 3], dim_in=dim,
 257                     dim_out=dim, name="h_1", stddev=numpy.sqrt(2. / (1. * dim)),
 258                     strides=[1, 1, 1, 1], padding="SAME",
 259                     nonlinearity=None,
 260                     bias=(not use_batch_norm),
 261                     weight_norm=weight_norm, scale=False)
 262                 if use_batch_norm:
 263                     res = batch_norm(
 264                         input_=res, dim=dim, name="bn_1", scale=False,
 265                         train=train, epsilon=1e-4, axes=[0, 1, 2])
 266                 res = tf.nn.relu(res)
 267                 res = conv_layer(
 268                     input_=res, filter_size=[1, 1], dim_in=dim, dim_out=dim_out,
 269                     name="out", stddev=numpy.sqrt(2. / (1. * dim)),
 270                     strides=[1, 1, 1, 1], padding="SAME",
 271                     nonlinearity=None, bias=True,
 272                     weight_norm=weight_norm, scale=True)
 273             else:
 274                 res = conv_layer(
 275                     input_=res, filter_size=[3, 3], dim_in=dim_in, dim_out=dim,
 276                     name="h_0", stddev=numpy.sqrt(2. / (dim_in)),
 277                     strides=[1, 1, 1, 1], padding="SAME",
 278                     nonlinearity=None, bias=(not use_batch_norm),
 279                     weight_norm=weight_norm, scale=False)
 280                 if use_batch_norm:
 281                     res = batch_norm(
 282                         input_=res, dim=dim, name="bn_0", scale=False,
 283                         train=train, epsilon=1e-4, axes=[0, 1, 2])
 284                 res = tf.nn.relu(res)
 285                 res = conv_layer(
 286                     input_=res, filter_size=[3, 3], dim_in=dim, dim_out=dim_out,
 287                     name="out", stddev=numpy.sqrt(2. / (1. * dim)),
 288                     strides=[1, 1, 1, 1], padding="SAME",
 289                     nonlinearity=None, bias=True,
 290                     weight_norm=weight_norm, scale=True)
 291         return res

 296 def masked_conv_aff_coupling(input_, mask_in, dim, name,
 297                              use_batch_norm=True, train=True, weight_norm=True,
 298                              reverse=False, residual_blocks=5,
 299                              bottleneck=False, use_width=1., use_height=1.,
 300                              mask_channel=0., skip=True):
 301     """Affine coupling with masked convolution."""
 302     with tf.variable_scope(name) as scope:
 303         if reverse or (not train):
 304             scope.reuse_variables()
 305         shape = input_.get_shape().as_list()
 306         batch_size = shape[0]
 307         height = shape[1]
 308         width = shape[2]
 309         channels = shape[3]
 310 
 311         # build mask
 312         mask = use_width * numpy.arange(width)
 313         mask = use_height * numpy.arange(height).reshape((-1, 1)) + mask
 314         mask = mask.astype("float32")
 315         mask = tf.mod(mask_in + mask, 2)
 316         mask = tf.reshape(mask, [-1, height, width, 1])
 317         if mask.get_shape().as_list()[0] == 1:
 318             mask = tf.tile(mask, [batch_size, 1, 1, 1])
 319         res = input_ * tf.mod(mask_channel + mask, 2)
 320 
 321         # initial input
 322         if use_batch_norm:
 323             res = batch_norm(
 324                 input_=res, dim=channels, name="bn_in", scale=False,
 325                 train=train, epsilon=1e-4, axes=[0, 1, 2])
 326             res *= 2.
 327         res = tf.concat([res, -res], 3)
 328         res = tf.concat([res, mask], 3)
 329         dim_in = 2. * channels + 1
 330         res = tf.nn.relu(res)
 331         res = resnet(input_=res, dim_in=dim_in, dim=dim,
 332                      dim_out=2 * channels,
 333                      name="resnet", use_batch_norm=use_batch_norm,
 334                      train=train, weight_norm=weight_norm,
 335                      residual_blocks=residual_blocks,
 336                      bottleneck=bottleneck, skip=skip)
 337         mask = tf.mod(mask_channel + mask, 2)
 338         res = tf.split(axis=3, num_or_size_splits=2, value=res)
 339         shift, log_rescaling = res[-2], res[-1]
 340         scale = variable_on_cpu(
 341             "rescaling_scale", [],
 342             tf.constant_initializer(0.))
 343         shift = tf.reshape(
 344             shift, [batch_size, height, width, channels])
 345         log_rescaling = tf.reshape(
 346             log_rescaling, [batch_size, height, width, channels])
 347         log_rescaling = scale * tf.tanh(log_rescaling)
 348         if not use_batch_norm:
 349             scale_shift = variable_on_cpu(
 350                 "scale_shift", [],
 351                 tf.constant_initializer(0.))
 352             log_rescaling += scale_shift
 353         shift *= (1. - mask)
 354         log_rescaling *= (1. - mask)
 355         if reverse:
 356             res = input_
 357             if use_batch_norm:
 358                 mean, var = batch_norm_log_diff(
 359                     input_=res * (1. - mask), dim=channels, name="bn_out",
 360                     train=False, epsilon=1e-4, axes=[0, 1, 2])
 361                 log_var = tf.log(var)
 362                 res *= tf.exp(.5 * log_var * (1. - mask))
 363                 res += mean * (1. - mask)
 364             res *= tf.exp(-log_rescaling)
 365             res -= shift
 366             log_diff = -log_rescaling
 367             if use_batch_norm:
 368                 log_diff += .5 * log_var * (1. - mask)
 369         else:
 370             res = input_
 371             res += shift
 372             res *= tf.exp(log_rescaling)
 373             log_diff = log_rescaling
 374             if use_batch_norm:
 375                 mean, var = batch_norm_log_diff(
 376                     input_=res * (1. - mask), dim=channels, name="bn_out",
 377                     train=train, epsilon=1e-4, axes=[0, 1, 2])
 378                 log_var = tf.log(var)
 379                 res -= mean * (1. - mask)
 380                 res *= tf.exp(-.5 * log_var * (1. - mask))
 381                 log_diff -= .5 * log_var * (1. - mask)
 382 
 383     return res, log_diff

 386 def masked_conv_add_coupling(input_, mask_in, dim, name,
 387                              use_batch_norm=True, train=True, weight_norm=True,
 388                              reverse=False, residual_blocks=5,
 389                              bottleneck=False, use_width=1., use_height=1.,
 390                              mask_channel=0., skip=True):
 391     """Additive coupling with masked convolution."""
 392     with tf.variable_scope(name) as scope:
 393         if reverse or (not train):
 394             scope.reuse_variables()
 395         shape = input_.get_shape().as_list()
 396         batch_size = shape[0]
 397         height = shape[1]
 398         width = shape[2]
 399         channels = shape[3]
 400 
 401         # build mask
 402         mask = use_width * numpy.arange(width)
 403         mask = use_height * numpy.arange(height).reshape((-1, 1)) + mask
 404         mask = mask.astype("float32")
 405         mask = tf.mod(mask_in + mask, 2)
 406         mask = tf.reshape(mask, [-1, height, width, 1])
 407         if mask.get_shape().as_list()[0] == 1:
 408             mask = tf.tile(mask, [batch_size, 1, 1, 1])
 409         res = input_ * tf.mod(mask_channel + mask, 2)
 410 
 411         # initial input
 412         if use_batch_norm:
 413             res = batch_norm(
 414                 input_=res, dim=channels, name="bn_in", scale=False,
 415                 train=train, epsilon=1e-4, axes=[0, 1, 2])
 416             res *= 2.
 417         res = tf.concat([res, -res], 3)
 418         res = tf.concat([res, mask], 3)
 419         dim_in = 2. * channels + 1
 420         res = tf.nn.relu(res)
 421         shift = resnet(input_=res, dim_in=dim_in, dim=dim, dim_out=channels,
 422                        name="resnet", use_batch_norm=use_batch_norm,
 423                        train=train, weight_norm=weight_norm,
 424                        residual_blocks=residual_blocks,
 425                        bottleneck=bottleneck, skip=skip)
 426         mask = tf.mod(mask_channel + mask, 2)
 427         shift *= (1. - mask)
 428         # use_batch_norm = False
 429         if reverse:
 430             res = input_
 431             if use_batch_norm:
 432                 mean, var = batch_norm_log_diff(
 433                     input_=res * (1. - mask),
 434                     dim=channels, name="bn_out", train=False, epsilon=1e-4)
 435                 log_var = tf.log(var)
 436                 res *= tf.exp(.5 * log_var * (1. - mask))
 437                 res += mean * (1. - mask)
 438             res -= shift
 439             log_diff = tf.zeros_like(res)
 440             if use_batch_norm:
 441                 log_diff += .5 * log_var * (1. - mask)
 442         else:
 443             res = input_
 444             res += shift
 445             log_diff = tf.zeros_like(res)
 446             if use_batch_norm:
 447                 mean, var = batch_norm_log_diff(
 448                     input_=res * (1. - mask), dim=channels,
 449                     name="bn_out", train=train, epsilon=1e-4, axes=[0, 1, 2])
 450                 log_var = tf.log(var)
 451                 res -= mean * (1. - mask)
 452                 res *= tf.exp(-.5 * log_var * (1. - mask))
 453                 log_diff -= .5 * log_var * (1. - mask)
 454 
 455     return res, log_diff

 482 def conv_ch_aff_coupling(input_, dim, name,
 483                          use_batch_norm=True, train=True, weight_norm=True,
 484                          reverse=False, residual_blocks=5,
 485                          bottleneck=False, change_bottom=True, skip=True):
 486     """Affine coupling with channel-wise splitting."""
 487     with tf.variable_scope(name) as scope:
 488         if reverse or (not train):
 489             scope.reuse_variables()
 490 
 491         if change_bottom:
 492             input_, canvas = tf.split(axis=3, num_or_size_splits=2, value=input_)
 493         else:
 494             canvas, input_ = tf.split(axis=3, num_or_size_splits=2, value=input_)
 495         shape = input_.get_shape().as_list()
 496         batch_size = shape[0]
 497         height = shape[1]
 498         width = shape[2]
 499         channels = shape[3]
 500         res = input_
 501 
 502         # initial input
 503         if use_batch_norm:
 504             res = batch_norm(
 505                 input_=res, dim=channels, name="bn_in", scale=False,
 506                 train=train, epsilon=1e-4, axes=[0, 1, 2])
 507         res = tf.concat([res, -res], 3)
 508         dim_in = 2. * channels
 509         res = tf.nn.relu(res)
 510         res = resnet(input_=res, dim_in=dim_in, dim=dim, dim_out=2 * channels,
 511                      name="resnet", use_batch_norm=use_batch_norm,
 512                      train=train, weight_norm=weight_norm,
 513                      residual_blocks=residual_blocks,
 514                      bottleneck=bottleneck, skip=skip)
 515         shift, log_rescaling = tf.split(axis=3, num_or_size_splits=2, value=res)
 516         scale = variable_on_cpu(
 517             "scale", [],
 518             tf.constant_initializer(1.))
 519         shift = tf.reshape(
 520             shift, [batch_size, height, width, channels])
 521         log_rescaling = tf.reshape(
 522             log_rescaling, [batch_size, height, width, channels])
 523         log_rescaling = scale * tf.tanh(log_rescaling)
 524         if not use_batch_norm:
 525             scale_shift = variable_on_cpu(
 526                 "scale_shift", [],
 527                 tf.constant_initializer(0.))
 528             log_rescaling += scale_shift
 529         if reverse:
 530             res = canvas
 531             if use_batch_norm:
 532                 mean, var = batch_norm_log_diff(
 533                     input_=res, dim=channels, name="bn_out", train=False,
 534                     epsilon=1e-4, axes=[0, 1, 2])
 535                 log_var = tf.log(var)
 536                 res *= tf.exp(.5 * log_var)
 537                 res += mean
 538             res *= tf.exp(-log_rescaling)
 539             res -= shift
 540             log_diff = -log_rescaling
 541             if use_batch_norm:
 542                 log_diff += .5 * log_var
 543         else:
 544             res = canvas
 545             res += shift
 546             res *= tf.exp(log_rescaling)
 547             log_diff = log_rescaling
 548             if use_batch_norm:
 549                 mean, var = batch_norm_log_diff(
 550                     input_=res, dim=channels, name="bn_out", train=train,
 551                     epsilon=1e-4, axes=[0, 1, 2])
 552                 log_var = tf.log(var)
 553                 res -= mean
 554                 res *= tf.exp(-.5 * log_var)
 555                 log_diff -= .5 * log_var
 556         if change_bottom:
 557             res = tf.concat([input_, res], 3)
 558             log_diff = tf.concat([tf.zeros_like(log_diff), log_diff], 3)
 559         else:
 560             res = tf.concat([res, input_], 3)
 561             log_diff = tf.concat([log_diff, tf.zeros_like(log_diff)], 3)
 562 
 563     return res, log_diff

 566 def conv_ch_add_coupling(input_, dim, name,
 567                          use_batch_norm=True, train=True, weight_norm=True,
 568                          reverse=False, residual_blocks=5,
 569                          bottleneck=False, change_bottom=True, skip=True):
 570     """Additive coupling with channel-wise splitting."""
 571     with tf.variable_scope(name) as scope:
 572         if reverse or (not train):
 573             scope.reuse_variables()
 574 
 575         if change_bottom:
 576             input_, canvas = tf.split(axis=3, num_or_size_splits=2, value=input_)
 577         else:
 578             canvas, input_ = tf.split(axis=3, num_or_size_splits=2, value=input_)
 579         shape = input_.get_shape().as_list()
 580         channels = shape[3]
 581         res = input_
 582 
 583         # initial input
 584         if use_batch_norm:
 585             res = batch_norm(
 586                 input_=res, dim=channels, name="bn_in", scale=False,
 587                 train=train, epsilon=1e-4, axes=[0, 1, 2])
 588         res = tf.concat([res, -res], 3)
 589         dim_in = 2. * channels
 590         res = tf.nn.relu(res)
 591         shift = resnet(input_=res, dim_in=dim_in, dim=dim, dim_out=channels,
 592                        name="resnet", use_batch_norm=use_batch_norm,
 593                        train=train, weight_norm=weight_norm,
 594                        residual_blocks=residual_blocks,
 595                        bottleneck=bottleneck, skip=skip)
 596         if reverse:
 597             res = canvas
 598             if use_batch_norm:
 599                 mean, var = batch_norm_log_diff(
 600                     input_=res, dim=channels, name="bn_out", train=False,
 601                     epsilon=1e-4, axes=[0, 1, 2])
 602                 log_var = tf.log(var)
 603                 res *= tf.exp(.5 * log_var)
 604                 res += mean
 605             res -= shift
 606             log_diff = tf.zeros_like(res)
 607             if use_batch_norm:
 608                 log_diff += .5 * log_var
 609         else:
 610             res = canvas
 611             res += shift
 612             log_diff = tf.zeros_like(res)
 613             if use_batch_norm:
 614                 mean, var = batch_norm_log_diff(
 615                     input_=res, dim=channels, name="bn_out", train=train,
 616                     epsilon=1e-4, axes=[0, 1, 2])
 617                 log_var = tf.log(var)
 618                 res -= mean
 619                 res *= tf.exp(-.5 * log_var)
 620                 log_diff -= .5 * log_var
 621         if change_bottom:
 622             res = tf.concat([input_, res], 3)
 623             log_diff = tf.concat([tf.zeros_like(log_diff), log_diff], 3)
 624         else:
 625             res = tf.concat([res, input_], 3)
 626             log_diff = tf.concat([log_diff, tf.zeros_like(log_diff)], 3)
 627 
 628     return res, log_diff

 652 def rec_masked_conv_coupling(input_, hps, scale_idx, n_scale,

 667         dim *= 2 ** scale_idx
 668     with tf.variable_scope("scale_%d" % scale_idx):
 669         # initial coupling layers
 670         res, inc_log_diff = masked_conv_coupling(
 671             input_=res,

 675             weight_norm=weight_norm,

 700     if scale_idx < (n_scale - 1):
 701         with tf.variable_scope("scale_%d" % scale_idx):
 702             res = squeeze_2x2(res)
 703             log_diff = squeeze_2x2(log_diff)

 705                 input_=res,
 706                 change_bottom=True, dim=2 * dim,
 707                 name="coupling_4",
 708                 use_batch_norm=use_batch_norm, train=train,
 709                 weight_norm=weight_norm,
 710                 reverse=False, residual_blocks=residual_blocks,
 711                 bottleneck=hps.bottleneck, use_aff=use_aff, skip=skip)
 712             log_diff += inc_log_diff
 713             res, inc_log_diff = conv_ch_coupling(
 714                 input_=res,
 715                 change_bottom=False, dim=2 * dim,
 716                 name="coupling_5",
 717                 use_batch_norm=use_batch_norm, train=train,
 718                 weight_norm=weight_norm,
 719                 reverse=False, residual_blocks=residual_blocks,
 720                 bottleneck=hps.bottleneck, use_aff=use_aff, skip=skip)
 721             log_diff += inc_log_diff
 722             res, inc_log_diff = conv_ch_coupling(
 723                 input_=res,
 724                 change_bottom=True, dim=2 * dim,
 725                 name="coupling_6",
 726                 use_batch_norm=use_batch_norm, train=train,
 727                 weight_norm=weight_norm,
 728                 reverse=False, residual_blocks=residual_blocks,
 729                 bottleneck=hps.bottleneck, use_aff=True, skip=skip)
 730             log_diff += inc_log_diff
 731             res = unsqueeze_2x2(res)
 732             log_diff = unsqueeze_2x2(log_diff)
 733         if FLAGS.recursion_type > 1:
 734             res = squeeze_2x2_ordered(res)
 735             log_diff = squeeze_2x2_ordered(log_diff)
 736             if FLAGS.recursion_type > 2:
 737                 res_1 = res[:, :, :, :channels]
 738                 res_2 = res[:, :, :, channels:]
 739                 log_diff_1 = log_diff[:, :, :, :channels]
 740                 log_diff_2 = log_diff[:, :, :, channels:]
 741             else:
 742                 res_1, res_2 = tf.split(axis=3, num_or_size_splits=2, value=res)
 743                 log_diff_1, log_diff_2 = tf.split(axis=3, num_or_size_splits=2, value=log_diff)
 744             res_1, inc_log_diff = rec_masked_conv_coupling(
 745                 input_=res_1, hps=hps, scale_idx=scale_idx + 1, n_scale=n_scale,
 746                 use_batch_norm=use_batch_norm, weight_norm=weight_norm,
 747                 train=train)
 748             res = tf.concat([res_1, res_2], 3)
 749             log_diff_1 += inc_log_diff
 750             log_diff = tf.concat([log_diff_1, log_diff_2], 3)
 751             res = squeeze_2x2_ordered(res, reverse=True)
 752             log_diff = squeeze_2x2_ordered(log_diff, reverse=True)
 753         else:
 754             res = squeeze_2x2_ordered(res)
 755             log_diff = squeeze_2x2_ordered(log_diff)
 756             res, inc_log_diff = rec_masked_conv_coupling(
 757                 input_=res, hps=hps, scale_idx=scale_idx + 1, n_scale=n_scale,
 758                 use_batch_norm=use_batch_norm, weight_norm=weight_norm,
 759                 train=train)
 760             log_diff += inc_log_diff
 761             res = squeeze_2x2_ordered(res, reverse=True)
 762             log_diff = squeeze_2x2_ordered(log_diff, reverse=True)
 763     else:
 764         with tf.variable_scope("scale_%d" % scale_idx):
 765             res, inc_log_diff = masked_conv_coupling(
 766                 input_=res,
 767                 mask_in=1. - mask, dim=dim,
 768                 name="coupling_3",
 769                 use_batch_norm=use_batch_norm, train=train,
 770                 weight_norm=weight_norm,
 771                 reverse=False, residual_blocks=residual_blocks,
 772                 bottleneck=hps.bottleneck, use_aff=True,
 773                 use_width=1., use_height=1., skip=skip)
 774             log_diff += inc_log_diff
 775     return res, log_diff

1448 def train_model(hps, logdir):
1449     """Training."""
1450     with tf.Graph().as_default():
1451         with tf.device(tf.train.replica_device_setter(0)):
1452             with tf.variable_scope("model"):
1453                 model = RealNVP(hps)
1454 
1455             saver = tf.train.Saver(tf.global_variables())
1456 
1457             # Build the summary operation from the last tower summaries.
1458             summary_op = tf.summary.merge_all()
1459 
1460             # Build an initialization operation to run below.
1461             init = tf.global_variables_initializer()
1462 
1463             # Start running operations on the Graph. allow_soft_placement must be set to
1464             # True to build towers on GPU, as some of the ops do not have GPU
1465             # implementations.
1466             sess = tf.Session(config=tf.ConfigProto(
1467                 allow_soft_placement=True,
1468                 log_device_placement=True))
1469             sess.run(init)
1470 
1471             ckpt_state = tf.train.get_checkpoint_state(logdir)
1472             if ckpt_state and ckpt_state.model_checkpoint_path:
1473                 print("Loading file %s" % ckpt_state.model_checkpoint_path)
1474                 saver.restore(sess, ckpt_state.model_checkpoint_path)
1475 
1476             # Start the queue runners.
1477             tf.train.start_queue_runners(sess=sess)
1478 
1479             summary_writer = tf.summary.FileWriter(
1480                 logdir,
1481                 graph=sess.graph)
1482 
1483             local_step = 0
1484             while True:
1485                 fetches = [model.step, model.bit_per_dim, model.train_step]
1486                 # The chief worker evaluates the summaries every 10 steps.
1487                 should_eval_summaries = local_step % 100 == 0
1488                 if should_eval_summaries:
1489                     fetches += [summary_op]
1490 
1491 
1492                 start_time = time.time()
1493                 outputs = sess.run(fetches)
1494                 global_step_val = outputs[0]
1495                 loss = outputs[1]
1496                 duration = time.time() - start_time
1497                 assert not numpy.isnan(
1498                     loss), 'Model diverged with loss = NaN'
1499 
1500                 if local_step % 10 == 0:
1501                     examples_per_sec = hps.batch_size / float(duration)
1502                     format_str = ('%s: step %d, loss = %.2f '
1503                                   '(%.1f examples/sec; %.3f '
1504                                   'sec/batch)')
1505                     print(format_str % (datetime.now(), global_step_val, loss,
1506                                         examples_per_sec, duration))
1507 
1508                 if should_eval_summaries:
1509                     summary_str = outputs[-1]
1510                     summary_writer.add_summary(summary_str, global_step_val)
1511 
1512                 # Save the model checkpoint periodically.
1513                 if local_step % 1000 == 0 or (local_step + 1) == FLAGS.train_steps:
1514                     checkpoint_path = os.path.join(logdir, 'model.ckpt')
1515                     saver.save(
1516                         sess,
1517                         checkpoint_path,
1518                         global_step=global_step_val)
1519 
1520                 if outputs[0] >= FLAGS.train_steps:
1521                     break
1522 
1523                 local_step += 1
