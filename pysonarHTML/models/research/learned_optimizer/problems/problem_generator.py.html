  45 class Problem(object):
  46   """Base class for optimization problems.
  47 
  48   This defines an interface for optimization problems, including objective and
  49   gradients functions and a feed_generator function that yields data to pass to
  50   feed_dict in tensorflow.
  51 
  52   Subclasses of Problem must (at the minimum) override the objective method,
  53   which computes the objective/loss/cost to minimize, and specify the desired
  54   shape of the parameters in a list in the param_shapes attribute.
  55   """
  56 
  57   def __init__(self, param_shapes, random_seed, noise_stdev, init_fn=None):
  58     """Initializes a global random seed for the problem.
  59 
  60     Args:
  61       param_shapes: A list of tuples defining the expected shapes of the
  62         parameters for this problem
  63       random_seed: Either an integer (or None, in which case the seed is
  64         randomly drawn)
  65       noise_stdev: Strength (standard deviation) of added gradient noise
  66       init_fn: A function taking a tf.Session object that is used to
  67         initialize the problem's variables.
  68 
  69     Raises:
  70       ValueError: If the random_seed is not an integer and not None
  71     """
  72     if random_seed is not None and not isinstance(random_seed, int):
  73       raise ValueError("random_seed must be an integer or None")
  74 
  75     # Pick a random seed.
  76     self.random_seed = (np.random.randint(MAX_SEED) if random_seed is None
  77                         else random_seed)
  78 
  79     # Store the noise level.
  80     self.noise_stdev = noise_stdev
  81 
  82     # Set the random seed to ensure any random data in the problem is the same.
  83     np.random.seed(self.random_seed)
  84 
  85     # Store the parameter shapes.
  86     self.param_shapes = param_shapes
  87 
  88     if init_fn is not None:
  89       self.init_fn = init_fn
  90     else:
  91       self.init_fn = lambda _: None
  92 
  93   def init_tensors(self, seed=None):
  94     """Returns a list of tensors with the given shape."""
  95     return [tf.random_normal(shape, seed=seed) for shape in self.param_shapes]
  96 
  97   def init_variables(self, seed=None):
  98     """Returns a list of variables with the given shape."""
  99     with tf.variable_scope(PARAMETER_SCOPE):
 100       params = [tf.Variable(param) for param in self.init_tensors(seed)]
 101     return params
 102 
 103   def objective(self, parameters, data=None, labels=None):
 104     """Computes the objective given a list of parameters.
 105 
 106     Args:
 107       parameters: The parameters to optimize (as a list of tensors)
 108       data: An optional batch of data for calculating objectives
 109       labels: An optional batch of corresponding labels
 110 
 111     Returns:
 112       A scalar tensor representing the objective value
 113     """
 114     raise NotImplementedError
 115 
 116   def gradients(self, objective, parameters):
 117     """Compute gradients of the objective with respect to the parameters.
 118 
 119     Args:
 120       objective: The objective op (e.g. output of self.objective())
 121       parameters: A list of tensors (the parameters to optimize)
 122 
 123     Returns:
 124       A list of tensors representing the gradient for each parameter,
 125         returned in the same order as the given list
 126     """
 127     grads = tf.gradients(objective, list(parameters))
 128     noisy_grads = []
 129 
 130     for grad in grads:
 131       if isinstance(grad, tf.IndexedSlices):
 132         noise = self.noise_stdev * tf.random_normal(tf.shape(grad.values))
 133         new_grad = tf.IndexedSlices(grad.values + noise, grad.indices)
 134       else:
 135         new_grad = grad + self.noise_stdev * tf.random_normal(grad.get_shape())
 136       noisy_grads.append(new_grad)
 137 
 138     return noisy_grads

 141 class Quadratic(Problem):
 142   """Optimizes a random quadratic function.
 143 
 144   The objective is: f(x) = (1/2) ||Wx - y||_2^2
 145   where W is a random Gaussian matrix and y is a random Gaussian vector.
 146   """
 147 
 148   def __init__(self, ndim, random_seed=None, noise_stdev=0.0):
 149     """Initializes a random quadratic problem."""
 150     param_shapes = [(ndim, 1)]
 151     super(Quadratic, self).__init__(param_shapes, random_seed, noise_stdev)
 152 
 153     # Generate a random problem instance.
 154     self.w = np.random.randn(ndim, ndim).astype("float32")
 155     self.y = np.random.randn(ndim, 1).astype("float32")
 156 
 157   def objective(self, params, data=None, labels=None):
 158     """Quadratic objective (see base class for details)."""
 159     return tf.nn.l2_loss(tf.matmul(self.w, params[0]) - self.y)

 162 class SoftmaxClassifier(Problem):
 163   """Helper functions for supervised softmax classification problems."""
 164 
 165   def init_tensors(self, seed=None):
 166     """Returns a list of tensors with the given shape."""
 167     return [tf.random_normal(shape, seed=seed) * 1.2 / np.sqrt(shape[0])
 168             for shape in self.param_shapes]
 169 
 170   def inference(self, params, data):
 171     """Computes logits given parameters and data.
 172 
 173     Args:
 174       params: List of parameter tensors or variables
 175       data: Batch of features with samples along the first dimension
 176 
 177     Returns:
 178       logits: Un-normalized logits with shape (num_samples, num_classes)
 179     """
 180     raise NotImplementedError
 181 
 182   def objective(self, params, data, labels):
 183     """Computes the softmax cross entropy.
 184 
 185     Args:
 186       params: List of parameter tensors or variables
 187       data: Batch of features with samples along the first dimension
 188       labels: Vector of labels with the same number of samples as the data
 189 
 190     Returns:
 191       loss: Softmax cross entropy loss averaged over the samples in the batch
 192 
 193     Raises:
 194       ValueError: If the objective is to be computed over >2 classes, because
 195         this operation is broken in tensorflow at the moment.
 196     """
 197     # Forward pass.
 198     logits = self.inference(params, data)
 199 
 200     # Compute the loss.
 201     l2reg = [tf.reduce_sum(param ** 2) for param in params]
 202     if int(logits.get_shape()[1]) == 2:
 203       labels = tf.cast(labels, tf.float32)
 204       losses = tf.nn.sigmoid_cross_entropy_with_logits(
 205           labels=labels, logits=logits[:, 0])
 206     else:
 207       raise ValueError("Unable to compute softmax cross entropy for more than"
 208                        " 2 classes.")
 209 
 210     return tf.reduce_mean(losses) + tf.reduce_mean(l2reg) * FLAGS.l2_reg_scale
 211 
 212   def argmax(self, logits):
 213     """Samples the most likely class label given the logits.
 214 
 215     Args:
 216       logits: Un-normalized logits with shape (num_samples, num_classes)
 217 
 218     Returns:
 219       predictions: Predicted class labels, has shape (num_samples,)
 220     """
 221     return tf.cast(tf.argmax(tf.nn.softmax(logits), 1), tf.int32)
 222 
 223   def accuracy(self, params, data, labels):
 224     """Computes the accuracy (fraction of correct classifications).
 225 
 226     Args:
 227       params: List of parameter tensors or variables
 228       data: Batch of features with samples along the first dimension
 229       labels: Vector of labels with the same number of samples as the data
 230 
 231     Returns:
 232       accuracy: Fraction of correct classifications across the batch
 233     """
 234     predictions = self.argmax(self.inference(params, data))
 235     return tf.contrib.metrics.accuracy(predictions, tf.cast(labels, tf.int32))

 238 class SoftmaxRegression(SoftmaxClassifier):
 239   """Builds a softmax regression problem."""
 240 
 241   def __init__(self, n_features, n_classes, activation=tf.identity,
 242                random_seed=None, noise_stdev=0.0):
 243     self.activation = activation
 244     self.n_features = n_features
 245     param_shapes = [(n_features, n_classes), (n_classes,)]
 246     super(SoftmaxRegression, self).__init__(param_shapes,
 247                                             random_seed,
 248                                             noise_stdev)
 249 
 250   def inference(self, params, data):
 251     features = tf.reshape(data, (-1, self.n_features))
 252     return tf.matmul(features, params[0]) + params[1]

 255 class SparseSoftmaxRegression(SoftmaxClassifier):
 256   """Builds a sparse input softmax regression problem."""
 257 
 258   def __init__(self,
 259                n_features,
 260                n_classes,
 261                activation=tf.identity,
 262                random_seed=None,
 263                noise_stdev=0.0):
 264     self.activation = activation
 265     self.n_features = n_features
 266     param_shapes = [(n_classes, n_features), (n_features, n_classes), (
 267         n_classes,)]
 268     super(SparseSoftmaxRegression, self).__init__(param_shapes, random_seed,
 269                                                   noise_stdev)
 270 
 271   def inference(self, params, data):
 272     all_embeddings, softmax_weights, softmax_bias = params
 273     embeddings = tf.nn.embedding_lookup(all_embeddings, tf.cast(data, tf.int32))
 274     embeddings = tf.reduce_sum(embeddings, 1)
 275     return tf.matmul(embeddings, softmax_weights) + softmax_bias

 278 class OneHotSparseSoftmaxRegression(SoftmaxClassifier):
 279   """Builds a sparse input softmax regression problem.
 280 
 281   This is identical to SparseSoftmaxRegression, but without using embedding
 282   ops.
 283   """
 284 
 285   def __init__(self,
 286                n_features,
 287                n_classes,
 288                activation=tf.identity,
 289                random_seed=None,
 290                noise_stdev=0.0):
 291     self.activation = activation
 292     self.n_features = n_features
 293     self.n_classes = n_classes
 294     param_shapes = [(n_classes, n_features), (n_features, n_classes), (
 295         n_classes,)]
 296     super(OneHotSparseSoftmaxRegression, self).__init__(param_shapes,
 297                                                         random_seed,
 298                                                         noise_stdev)
 299 
 300   def inference(self, params, data):
 301     all_embeddings, softmax_weights, softmax_bias = params
 302     num_ids = tf.shape(data)[1]
 303     one_hot_embeddings = tf.one_hot(tf.cast(data, tf.int32), self.n_classes)
 304     one_hot_embeddings = tf.reshape(one_hot_embeddings, [-1, self.n_classes])
 305     embeddings = tf.matmul(one_hot_embeddings, all_embeddings)
 306     embeddings = tf.reshape(embeddings, [-1, num_ids, self.n_features])
 307     embeddings = tf.reduce_sum(embeddings, 1)
 308     return tf.matmul(embeddings, softmax_weights) + softmax_bias

 311 class FullyConnected(SoftmaxClassifier):
 312   """Builds a multi-layer perceptron classifier."""
 313 
 314   def __init__(self, n_features, n_classes, hidden_sizes=(32, 64),
 315                activation=tf.nn.sigmoid, random_seed=None, noise_stdev=0.0):
 316     """Initializes an multi-layer perceptron classification problem."""
 317     # Store the number of features and activation function.
 318     self.n_features = n_features
 319     self.activation = activation
 320 
 321     # Define the network as a list of weight + bias shapes for each layer.
 322     param_shapes = []
 323     for ix, sz in enumerate(hidden_sizes + (n_classes,)):
 324 
 325       # The previous layer"s size (n_features if input).
 326       prev_size = n_features if ix == 0 else hidden_sizes[ix - 1]
 327 
 328       # Weight shape for this layer.
 329       param_shapes.append((prev_size, sz))
 330 
 331       # Bias shape for this layer.
 332       param_shapes.append((sz,))
 333 
 334     super(FullyConnected, self).__init__(param_shapes, random_seed, noise_stdev)
 335 
 336   def inference(self, params, data):
 337     # Flatten the features into a vector.
 338     features = tf.reshape(data, (-1, self.n_features))
 339 
 340     # Pass the data through the network.
 341     preactivations = tf.matmul(features, params[0]) + params[1]
 342 
 343     for layer in range(2, len(self.param_shapes), 2):
 344       net = self.activation(preactivations)
 345       preactivations = tf.matmul(net, params[layer]) + params[layer + 1]
 346 
 347     return preactivations
 348 
 349   def accuracy(self, params, data, labels):
 350     """Computes the accuracy (fraction of correct classifications).
 351 
 352     Args:
 353       params: List of parameter tensors or variables
 354       data: Batch of features with samples along the first dimension
 355       labels: Vector of labels with the same number of samples as the data
 356 
 357     Returns:
 358       accuracy: Fraction of correct classifications across the batch
 359     """
 360     predictions = self.argmax(self.activation(self.inference(params, data)))
 361     return tf.contrib.metrics.accuracy(predictions, tf.cast(labels, tf.int32))

 364 class ConvNet(SoftmaxClassifier):
 365   """Builds an N-layer convnet for image classification."""
 366 
 367   def __init__(self,
 368                image_shape,
 369                n_classes,
 370                filter_list,
 371                activation=tf.nn.relu,
 372                random_seed=None,
 373                noise_stdev=0.0):
 374     # Number of channels, number of pixels in x- and y- dimensions.
 375     n_channels, px, py = image_shape
 376 
 377     # Store the activation.
 378     self.activation = activation
 379 
 380     param_shapes = []
 381     input_size = n_channels
 382     for fltr in filter_list:
 383       # Add conv2d filters.
 384       param_shapes.append((fltr[0], fltr[1], input_size, fltr[2]))
 385       input_size = fltr[2]
 386 
 387     # Number of units in the final (dense) layer.
 388     self.affine_size = input_size * px * py
 389 
 390     param_shapes.append((self.affine_size, n_classes))  # affine weights
 391     param_shapes.append((n_classes,))  # affine bias
 392 
 393     super(ConvNet, self).__init__(param_shapes, random_seed, noise_stdev)
 394 
 395   def init_tensors(self, seed=None):
 396     """Returns a list of tensors with the given shape."""
 397     return [tf.random_normal(shape, mean=0., stddev=0.01, seed=seed)
 398             for shape in self.param_shapes]
 399 
 400   def inference(self, params, data):
 401 
 402     # Unpack.
 403     w_conv_list = params[:-2]
 404     output_w, output_b = params[-2:]
 405 
 406     conv_input = data
 407     for w_conv in w_conv_list:
 408       layer = tf.nn.conv2d(conv_input, w_conv, strides=[1] * 4, padding="SAME")
 409       output = self.activation(layer)
 410       conv_input = output
 411 
 412     # Flatten.
 413     flattened = tf.reshape(conv_input, (-1, self.affine_size))
 414 
 415     # Fully connected layer.
 416     return tf.matmul(flattened, output_w) + output_b

 419 class Bowl(Problem):
 420   """A 2D quadratic bowl."""
 421 
 422   def __init__(self, condition_number, angle=0.0,
 423                random_seed=None, noise_stdev=0.0):
 424     assert condition_number > 0, "Condition number must be positive."
 425 
 426     # Define parameter shapes.
 427     param_shapes = [(2, 1)]
 428     super(Bowl, self).__init__(param_shapes, random_seed, noise_stdev)
 429 
 430     self.condition_number = condition_number
 431     self.angle = angle
 432     self._build_matrix(condition_number, angle)
 433 
 434   def _build_matrix(self, condition_number, angle):
 435     """Builds the Hessian matrix."""
 436     hessian = np.array([[condition_number, 0.], [0., 1.]], dtype="float32")
 437 
 438     # Build the rotation matrix.
 439     rotation_matrix = np.array([
 440         [np.cos(angle), -np.sin(angle)],
 441         [np.sin(angle), np.cos(angle)]
 442     ])
 443 
 444     # The objective is 0.5 * || Ax ||_2^2
 445     # where the data matrix (A) is: sqrt(Hessian).dot(rotation_matrix).
 446     self.matrix = np.sqrt(hessian).dot(rotation_matrix)
 447 
 448   def objective(self, params, data=None, labels=None):
 449     mtx = tf.constant(self.matrix, dtype=tf.float32)
 450     return tf.nn.l2_loss(tf.matmul(mtx, params[0]))
 451 
 452   def surface(self, xlim=5, ylim=5, n=50):
 453     xm, ym = _mesh(xlim, ylim, n)
 454     pts = np.vstack([xm.ravel(), ym.ravel()])
 455     zm = 0.5 * np.linalg.norm(self.matrix.dot(pts), axis=0) ** 2
 456     return xm, ym, zm.reshape(n, n)

 459 class Problem2D(Problem):
 460 
 461   def __init__(self, random_seed=None, noise_stdev=0.0):
 462     param_shapes = [(2,)]
 463     super(Problem2D, self).__init__(param_shapes, random_seed, noise_stdev)
 464 
 465   def surface(self, n=50, xlim=5, ylim=5):
 466     """Computes the objective surface over a 2d mesh."""
 467 
 468     # Create a mesh over the given coordinate ranges.
 469     xm, ym = _mesh(xlim, ylim, n)
 470 
 471     with tf.Graph().as_default(), tf.Session() as sess:
 472 
 473       # Ops to compute the objective at every (x, y) point.
 474       x = tf.placeholder(tf.float32, shape=xm.shape)
 475       y = tf.placeholder(tf.float32, shape=ym.shape)
 476       obj = self.objective([[x, y]])
 477 
 478       # Run the computation.
 479       zm = sess.run(obj, feed_dict={x: xm, y: ym})
 480 
 481     return xm, ym, zm

 484 class Rosenbrock(Problem2D):
 485   """See https://en.wikipedia.org/wiki/Rosenbrock_function.
 486 
 487   This function has a single global minima at [1, 1]
 488   The objective value at this point is zero.
 489   """
 490 
 491   def init_tensors(self, seed=None):
 492     """Returns a list of tensors with the given shape."""
 493     return [tf.random_uniform(shape, minval=-5., maxval=10., seed=seed)
 494             for shape in self.param_shapes]
 495 
 496   def objective(self, params, data=None, labels=None):
 497     x, y = tf.split(params[0], 2, axis=0)
 498     obj = (1 - x)**2 + 100 * (y - x**2)**2
 499     return tf.squeeze(obj)

 502 def make_rosenbrock_loss_and_init(device=None):
 503   """A variable-backed version of Rosenbrock problem.
 504 
 505   See the Rosenbrock class for details.
 506 
 507   Args:
 508     device: Where to place the ops of this problem.
 509 
 510   Returns:
 511     A tuple of two callables, first of which creates the loss and the second
 512     creates the parameter initializer function.
 513   """
 514   def make_rosenbrock_loss():
 515     with tf.name_scope("optimizee"):
 516       with tf.device(device):
 517         x = tf.get_variable("x", [1])
 518         y = tf.get_variable("y", [1])
 519         c = tf.get_variable(
 520             "c", [1],
 521             initializer=tf.constant_initializer(100.0),
 522             trainable=False)
 523         obj = (1 - x)**2 + c * (y - x**2)**2
 524       return tf.squeeze(obj)
 525 
 526   def make_init_fn(parameters):
 527     with tf.device(device):
 528       init_op = tf.variables_initializer(parameters)
 529     def init_fn(sess):
 530       tf.logging.info("Initializing model parameters.")
 531       sess.run(init_op)
 532     return init_fn
 533 
 534   return make_rosenbrock_loss, make_init_fn

 546 class LogSumExp(Problem2D):
 547   """2D function defined by the log of the sum of exponentials."""
 548 
 549   def objective(self, params, data=None, labels=None):
 550     x, y = tf.split(params[0], 2, axis=0)
 551     obj = tf.log(tf.exp(x + 3. * y - 0.1) +
 552                  tf.exp(x - 3. * y - 0.1) +
 553                  tf.exp(-x - 0.1) + 1.0)
 554     return tf.squeeze(obj)

 557 class Ackley(Problem2D):
 558   """Ackley's function (contains many local minima)."""
 559 
 560   def init_tensors(self, seed=None):
 561     """Returns a list of tensors with the given shape."""
 562     return [tf.random_uniform(shape, minval=-32.768, maxval=32.768, seed=seed)
 563             for shape in self.param_shapes]
 564 
 565   def objective(self, params, data=None, labels=None):
 566     x, y = tf.split(params[0], 2, axis=0)
 567     obj = (-20 * tf.exp(-0.2 * tf.sqrt(0.5 * (x ** 2 + y ** 2))) -
 568            tf.exp(0.5 * (tf.cos(2 * np.pi * x) + tf.cos(2 * np.pi * y))) +
 569            tf.exp(1.0) + 20.)
 570     return tf.squeeze(obj)

 573 class Beale(Problem2D):
 574   """Beale function (a multimodal function with sharp peaks)."""
 575 
 576   def init_tensors(self, seed=None):
 577     """Returns a list of tensors with the given shape."""
 578     return [tf.random_uniform(shape, minval=-4.5, maxval=4.5, seed=seed)
 579             for shape in self.param_shapes]

 581   def objective(self, params, data=None, labels=None):

 698 class SumTask(Problem):
 699   """Takes a list of problems and modifies the objective to be their sum."""
 700 
 701   def __init__(self, problem_specs, noise_stdev=0.0):
 702     self.problems = [ps.build() for ps in problem_specs]
 703     self.param_shapes = []
 704     for prob in self.problems:
 705       self.param_shapes += prob.param_shapes
 706 
 707     super(SumTask, self).__init__(self.param_shapes, random_seed=None,
 708                                   noise_stdev=noise_stdev)
 709 
 710   def init_tensors(self, seed=None):
 711     tensors = []
 712     for prob in self.problems:
 713       tensors += prob.init_tensors(seed=seed)
 714     return tensors
 715 
 716   def objective(self, params, data=None, labels=None):
 717     obj = 0.
 718     index = 0
 719     for prob in self.problems:
 720       num_params = len(prob.param_shapes)
 721       obj += prob.objective(params[index:index + num_params])
 722       index += num_params
 723     return obj

 733 class Norm(Problem):
 734   """Takes an existing problem and modifies the objective to be its N-norm."""
 735 
 736   def __init__(self, ndim, random_seed=None, noise_stdev=0.0, norm_power=2.):
 737     param_shapes = [(ndim, 1)]
 738     super(Norm, self).__init__(param_shapes, random_seed, noise_stdev)
 739 
 740     # Generate a random problem instance.
 741     self.w = np.random.randn(ndim, ndim).astype("float32")
 742     self.y = np.random.randn(ndim, 1).astype("float32")
 743     self.norm_power = norm_power
 744 
 745   def objective(self, params, data=None, labels=None):
 746     diff = tf.matmul(self.w, params[0]) - self.y
 747     exp = 1. / self.norm_power
 748     loss = tf.reduce_sum((tf.abs(diff) + EPSILON) ** self.norm_power) ** exp
 749     return loss

 768 class SparseProblem(Problem):
 769   """Takes a problem and sets gradients to 0 with the given probability."""
 770 
 771   def __init__(self,
 772                problem_spec,
 773                zero_probability=0.99,
 774                random_seed=None,
 775                noise_stdev=0.0):
 776     self.problem = problem_spec.build()
 777     self.param_shapes = self.problem.param_shapes
 778     self.zero_prob = zero_probability
 779 
 780     super(SparseProblem, self).__init__(self.param_shapes,
 781                                         random_seed=random_seed,
 782                                         noise_stdev=noise_stdev)
 783 
 784   def objective(self, parameters, data=None, labels=None):
 785     return self.problem.objective(parameters, data, labels)
 786 
 787   def gradients(self, objective, parameters):
 788     grads = tf.gradients(objective, list(parameters))
 789 
 790     new_grads = []
 791     for grad in grads:
 792       mask = tf.greater(self.zero_prob, tf.random_uniform(grad.get_shape()))
 793       zero_grad = tf.zeros_like(grad, dtype=tf.float32)
 794       noisy_grad = grad + self.noise_stdev * tf.random_normal(grad.get_shape())
 795       new_grads.append(tf.where(mask, zero_grad, noisy_grad))
 796     return new_grads

 799 class DependencyChain(Problem):
 800   """A problem in which parameters must be optimized in order.
 801 
 802   A sequence of parameters which all need to be brought to 0, but where each
 803   parameter in the sequence can't be brought to 0 until the preceding one
 804   has been. This should take a long time to optimize, with steady
 805   (or accelerating) progress throughout the entire process.
 806   """

 808   def __init__(self, ndim, random_seed=None, noise_stdev=0.):

 812         param_shapes, random_seed, noise_stdev)
 813 
 814   def objective(self, params, data=None, labels=None):
 815     terms = params[0][0]**2 + params[0][1:]**2 / (params[0][:-1]**2 + EPSILON)
 816     return tf.reduce_sum(terms)

 819 class MinMaxWell(Problem):
 820   """Problem with global min when both the min and max (absolute) params are 1.
 821 
 822   The gradient for all but two parameters (the min and max) is zero. This
 823   should therefore encourage the optimizer to behave sensible even when
 824   parameters have zero gradients, as is common eg for some deep neural nets.
 825   """
 826 
 827   def __init__(self, ndim, random_seed=None, noise_stdev=0.):
 828     param_shapes = [(ndim,)]
 829     self.ndim = ndim
 830     super(MinMaxWell, self).__init__(param_shapes, random_seed, noise_stdev)
 831 
 832   def objective(self, params, data=None, labels=None):
 833     params_sqr = params[0]**2
 834     min_sqr = tf.reduce_min(params_sqr)
 835     max_sqr = tf.reduce_max(params_sqr)
 836     epsilon = 1e-12
 837 
 838     return max_sqr + 1./min_sqr - 2. + epsilon

 841 class OutwardSnake(Problem):

 843 
 844   Ideal step length stays constant along the entire path.
 845   """
 846 
 847   def __init__(self, ndim, random_seed=None, noise_stdev=0.):
 848     param_shapes = [(ndim,)]

 850     super(OutwardSnake, self).__init__(param_shapes, random_seed, noise_stdev)

 853     radius = tf.sqrt(tf.reduce_sum(params[0]**2))

 858 
 859     return rad_loss + sin_loss

 890 class MatMulAlgorithm(Problem):
 891   """A 6-th order polynomial optimization problem.
 892 
 893   This problem is parametrized by n and k. A solution to this problem with
 894   objective value exactly zero defines a matrix multiplication algorithm of
 895   n x n matrices using k multiplications between matrices. When applied
 896   recursively, such an algorithm has complexity O(n^(log_n(k))).
 897 
 898   Given n, it is not known in general which values of k in [n^2, n^3] have a
 899   solution. There is always a solution with k = n^3 (this is the naive
 900   algorithm).
 901 
 902   In the special case n = 2, it is known that there are solutions for k = {7, 8}
 903   but not for k <= 6. For n = 3, it is known that there are exact solutions for
 904   23 <= k <= 27, and there are asymptotic solutions for k = {21, 22}, but the
 905   other cases are unknown.
 906 
 907   For a given n and k, if one solution exists then infinitely many solutions
 908   exist due to permutation and scaling symmetries in the parameters.
 909 
 910   This is a very hard problem for some values of n and k (e.g. n = 3, k = 21),
 911   but very easy for other values (e.g. n = 2, k = 7).
 912 
 913   For a given n and k, the specific formulation of this problem is as follows.
 914   Let theta_a, theta_b, theta_c be parameter matrices with respective dimensions
 915   [n**2, k], [n**2, k], [k, n**2]. Then for any matrices a, b with shape [n, n],
 916   we can form the matrix c with shape [n, n] via the operation:
 917       ((vec(a) * theta_a) .* (vec(b) * theta_b)) * theta_c = vec(c),  (#)
 918   where vec(x) is the operator that flattens a matrix with shape [n, n] into a
 919   row vector with shape [1, n**2], * denotes matrix multiplication and .*
 920   denotes elementwise multiplication.
 921 
 922   This operation, parameterized by theta_a, theta_b, theta_c, is a matrix
 923   multiplication algorithm iff c = a*b for all [n, n] matrices a and b. But
 924   actually it suffices to verify all combinations of one-hot matrices a and b,
 925   of which there are n**4 such combinations. This gives a batch of n**4 matrix
 926   triplets (a, b, c) such that equation (#) must hold for each triplet. We solve
 927   for theta_a, theta_b, theta_c by minimizing the sum of squares of errors
 928   across this batch.
 929 
 930   Finally, theta_c can be computed from theta_a and theta_b. Therefore it
 931   suffices to learn theta_a and theta_b, from which theta_c and therefore the
 932   objective value can be computed.
 933   """
 934 
 935   def __init__(self, n, k):
 936     assert isinstance(n, int), "n must be an integer"
 937     assert isinstance(k, int), "k must be an integer"
 938     assert n >= 2, "Must have n >= 2"
 939     assert k >= n**2 and k <= n**3, "Must have n**2 <= k <= n**3"
 940 
 941     param_shapes = [(n**2, k), (n**2, k)]  # theta_a, theta_b
 942     super(MatMulAlgorithm, self).__init__(
 943         param_shapes, random_seed=None, noise_stdev=0.0)
 944 
 945     self.n = n
 946     self.k = k
 947 
 948     # Build a batch of all combinations of one-hot matrices a, b, and their
 949     # respective products c. Correctness on this batch is a necessary and
 950     # sufficient condition for the algorithm to be valid. The number of matrices
 951     # in {a, b, c}_3d is n**4 and each matrix is n x n.
 952     onehots = np.identity(n**2).reshape(n**2, n, n)
 953     a_3d = np.repeat(onehots, n**2, axis=0)
 954     b_3d = np.tile(onehots, [n**2, 1, 1])
 955     c_3d = np.matmul(a_3d, b_3d)
 956 
 957     # Convert the batch to 2D Tensors.
 958     self.a = tf.constant(a_3d.reshape(n**4, n**2), tf.float32, name="a")
 959     self.b = tf.constant(b_3d.reshape(n**4, n**2), tf.float32, name="b")
 960     self.c = tf.constant(c_3d.reshape(n**4, n**2), tf.float32, name="c")
 961 
 962   def init_tensors(self, seed=None):
 963     # Initialize params such that the columns of theta_a and theta_b have L2
 964     # norm 1.
 965     def _param_initializer(shape, seed=None):
 966       x = tf.random_normal(shape, dtype=tf.float32, seed=seed)
 967       return tf.transpose(tf.nn.l2_normalize(tf.transpose(x), 1))
 968 
 969     return [_param_initializer(shape, seed) for shape in self.param_shapes]
 970 
 971   def objective(self, parameters, data=None, labels=None):
 972     theta_a = parameters[0]
 973     theta_b = parameters[1]
 974 
 975     # Compute theta_c from theta_a and theta_b.
 976     p = tf.matmul(self.a, theta_a) * tf.matmul(self.b, theta_b)
 977     p_trans = tf.transpose(p, name="p_trans")
 978     p_inv = tf.matmul(
 979         tf.matrix_inverse(tf.matmul(p_trans, p)), p_trans, name="p_inv")
 980     theta_c = tf.matmul(p_inv, self.c, name="theta_c")
 981 
 982     # Compute the "predicted" value of c.
 983     c_hat = tf.matmul(p, theta_c, name="c_hat")
 984 
 985     # Compute the loss (sum of squared errors).
 986     loss = tf.reduce_sum((c_hat - self.c)**2, name="loss")
 987 
 988     return loss
 989

 993   return [(_Spec(MatMulAlgorithm, (n, k), {}), None, None)
 994           for k in range(k_min, k_max + 1)]
