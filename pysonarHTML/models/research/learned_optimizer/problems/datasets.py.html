  30 class Dataset(namedtuple("Dataset", "data labels")):
  31   """Helper class for managing a supervised learning dataset.
  32 
  33   Args:
  34     data: an array of type float32 with N samples, each of which is the set
  35       of features for that sample. (Shape (N, D_i), where N is the number of
  36       samples and D_i is the number of features for that sample.)
  37     labels: an array of type int32 or int64 with N elements, indicating the
  38       class label for the corresponding set of features in data.
  39   """
  40   # Since this is an immutable object, we don't need to reserve slots.
  41   __slots__ = ()
  42 
  43   @property
  44   def size(self):
  45     """Dataset size (number of samples)."""
  46     return len(self.data)
  47 
  48   def batch_indices(self, num_batches, batch_size):
  49     """Creates indices of shuffled minibatches.
  50 
  51     Args:
  52       num_batches: the number of batches to generate
  53       batch_size: the size of each batch
  54 
  55     Returns:
  56       batch_indices: a list of minibatch indices, arranged so that the dataset
  57           is randomly shuffled.
  58 
  59     Raises:
  60       ValueError: if the data and labels have different lengths
  61     """
  62     if len(self.data) != len(self.labels):
  63       raise ValueError("Labels and data must have the same number of samples.")
  64 
  65     batch_indices = []
  66 
  67     # Follows logic in mnist.py to ensure we cover the entire dataset.
  68     index_in_epoch = 0
  69     dataset_size = len(self.data)
  70     dataset_indices = np.arange(dataset_size)
  71     np.random.shuffle(dataset_indices)
  72 
  73     for _ in range(num_batches):
  74       start = index_in_epoch
  75       index_in_epoch += batch_size
  76       if index_in_epoch > dataset_size:
  77 
  78         # Finished epoch, reshuffle.
  79         np.random.shuffle(dataset_indices)
  80 
  81         # Start next epoch.
  82         start = 0
  83         index_in_epoch = batch_size
  84 
  85       end = index_in_epoch
  86       batch_indices.append(dataset_indices[start:end].tolist())
  87 
  88     return batch_indices

  91 def noisy_parity_class(n_samples,
  92                        n_classes=2,
  93                        n_context_ids=5,
  94                        noise_prob=0.25,
  95                        random_seed=None):
  96   """Returns a randomly generated sparse-to-sparse dataset.
  97 
  98   The label is a parity class of a set of context classes.
  99 
 100   Args:
 101     n_samples: number of samples (data points)
 102     n_classes: number of class labels (default: 2)
 103     n_context_ids: how many classes to take the parity of (default: 5).
 104     noise_prob: how often to corrupt the label (default: 0.25)
 105     random_seed: seed used for drawing the random data (default: None)
 106   Returns:
 107     dataset: A Dataset namedtuple containing the generated data and labels
 108   """
 109   np.random.seed(random_seed)
 110   x = np.random.randint(0, n_classes, [n_samples, n_context_ids])
 111   noise = np.random.binomial(1, noise_prob, [n_samples])
 112   y = (np.sum(x, 1) + noise) % n_classes
 113   return Dataset(x.astype("float32"), y.astype("int32"))

 116 def random(n_features, n_samples, n_classes=2, sep=1.0, random_seed=None):
 117   """Returns a randomly generated classification dataset.
 118 
 119   Args:
 120     n_features: number of features (dependent variables)
 121     n_samples: number of samples (data points)
 122     n_classes: number of class labels (default: 2)
 123     sep: separation of the two classes, a higher value corresponds to
 124       an easier classification problem (default: 1.0)
 125     random_seed: seed used for drawing the random data (default: None)
 126 
 127   Returns:
 128     dataset: A Dataset namedtuple containing the generated data and labels
 129   """
 130   # Generate the problem data.
 131   x, y = make_classification(n_samples=n_samples,
 132                              n_features=n_features,
 133                              n_informative=n_features,
 134                              n_redundant=0,
 135                              n_classes=n_classes,
 136                              class_sep=sep,
 137                              random_state=random_seed)
 138 
 139   return Dataset(x.astype("float32"), y.astype("int32"))

 142 def random_binary(n_features, n_samples, random_seed=None):
 143   """Returns a randomly generated dataset of binary values.
 144 
 145   Args:
 146     n_features: number of features (dependent variables)
 147     n_samples: number of samples (data points)
 148     random_seed: seed used for drawing the random data (default: None)
 149 
 150   Returns:
 151     dataset: A Dataset namedtuple containing the generated data and labels
 152   """
 153   random_seed = (np.random.randint(MAX_SEED) if random_seed is None
 154                  else random_seed)
 155   np.random.seed(random_seed)
 156 
 157   x = np.random.randint(2, size=(n_samples, n_features))
 158   y = np.zeros((n_samples, 1))
 159 
 160   return Dataset(x.astype("float32"), y.astype("int32"))

 163 def random_symmetric(n_features, n_samples, random_seed=None):
 164   """Returns a randomly generated dataset of values and their negatives.
 165 
 166   Args:
 167     n_features: number of features (dependent variables)
 168     n_samples: number of samples (data points)
 169     random_seed: seed used for drawing the random data (default: None)
 170 
 171   Returns:
 172     dataset: A Dataset namedtuple containing the generated data and labels
 173   """
 174   random_seed = (np.random.randint(MAX_SEED) if random_seed is None
 175                  else random_seed)
 176   np.random.seed(random_seed)
 177 
 178   x1 = np.random.normal(size=(int(n_samples/2), n_features))
 179   x = np.concatenate((x1, -x1), axis=0)
 180   y = np.zeros((n_samples, 1))
 181 
 182   return Dataset(x.astype("float32"), y.astype("int32"))

 185 def random_mlp(n_features, n_samples, random_seed=None, n_layers=6, width=20):
 186   """Returns a generated output of an MLP with random weights.
 187 
 188   Args:
 189     n_features: number of features (dependent variables)
 190     n_samples: number of samples (data points)
 191     random_seed: seed used for drawing the random data (default: None)
 192     n_layers: number of layers in random MLP
 193     width: width of the layers in random MLP
 194 
 195   Returns:
 196     dataset: A Dataset namedtuple containing the generated data and labels
 197   """
 198   random_seed = (np.random.randint(MAX_SEED) if random_seed is None
 199                  else random_seed)
 200   np.random.seed(random_seed)
 201 
 202   x = np.random.normal(size=(n_samples, n_features))
 203   y = x
 204   n_in = n_features
 205   scale_factor = np.sqrt(2.) / np.sqrt(n_features)
 206   for _ in range(n_layers):
 207     weights = np.random.normal(size=(n_in, width)) * scale_factor
 208     y = np.dot(y, weights).clip(min=0)
 209     n_in = width
 210 
 211   y = y[:, 0]
 212   y[y > 0] = 1
 213 
 214   return Dataset(x.astype("float32"), y.astype("int32"))
