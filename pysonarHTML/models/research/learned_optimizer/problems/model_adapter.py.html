  28 class ModelAdapter(pg.Problem):
  29   """Adapts Tensorflow models/graphs into a form suitable for meta-training.
  30 
  31   This class adapts an existing TensorFlow graph into a form suitable for
  32   meta-training a learned optimizer.
  33   """
  34 
  35   def __init__(self, make_loss_and_init_fn):
  36     """Wraps a model in the Problem interface.
  37 
  38     make_loss_and_init argument is a callable that returns a tuple of
  39     two other callables as follows.
  40 
  41     The first will construct most of the graph and return the problem loss. It
  42     is essential that this graph contains the totality of the model's variables,
  43     but none of its queues.
  44 
  45     The second will return construct the model initialization graph given a list
  46     of parameters and return a callable that is passed an instance of
  47     tf.Session, and should initialize the models' parameters.
  48 
  49     An argument value function would look like this:
  50 
  51     ```python
  52     def make_loss_and_init_fn():
  53       inputs = queued_reader()
  54 
  55       def make_loss():
  56         return create_model_with_variables(inputs)
  57 
  58       def make_init_fn(parameters):
  59         saver = tf.Saver(parameters)
  60         def init_fn(sess):
  61           sess.restore(sess, ...)
  62         return init_fn
  63 
  64       return make_loss, make_init_fn
  65     ```
  66 
  67     Args:
  68       make_loss_and_init_fn: a callable, as described aboce
  69     """
  70     make_loss_fn, make_init_fn = make_loss_and_init_fn()
  71 
  72     self.make_loss_fn = make_loss_fn
  73     self.parameters, self.constants = _get_variables(make_loss_fn)
  74 
  75     if make_init_fn is not None:
  76       init_fn = make_init_fn(self.parameters + self.constants)
  77     else:
  78       init_op = tf.initialize_variables(self.parameters + self.constants)
  79       init_fn = lambda sess: sess.run(init_op)
  80 
  81     tf.logging.info("ModelAdapter parameters: %s",
  82                     [op.name for op in self.parameters])
  83     tf.logging.info("ModelAdapter constants: %s",
  84                     [op.name for op in self.constants])
  85 
  86     super(ModelAdapter, self).__init__(
  87         [], random_seed=None, noise_stdev=0.0, init_fn=init_fn)
  88 
  89   def init_tensors(self, seed=None):
  90     """Returns a list of tensors with the given shape."""
  91     return self.parameters
  92 
  93   def init_variables(self, seed=None):
  94     """Returns a list of variables with the given shape."""
  95     # NOTE(siege): This is awkward, as these are not set as trainable.
  96     return self.parameters
  97 
  98   def objective(self, parameters, data=None, labels=None):
  99     """Computes the objective given a list of parameters.
 100 
 101     Args:
 102       parameters: The parameters to optimize (as a list of tensors)
 103       data: An optional batch of data for calculating objectives
 104       labels: An optional batch of corresponding labels
 105 
 106     Returns:
 107       A scalar tensor representing the objective value
 108     """
 109     # We need to set up a mapping based on the original parameter names, because
 110     # the parameters passed can be arbitrary tensors.
 111     parameter_mapping = {
 112         old_p.name: p
 113         for old_p, p in zip(self.parameters, parameters)
 114     }
 115 
 116     with tf.variable_scope(tf.get_variable_scope(), reuse=True):
 117       return _make_with_custom_variables(self.make_loss_fn, parameter_mapping)

 120 def _get_variables(func):
 121   """Calls func, returning any variables created.
 122 
 123   The created variables are modified to not be trainable, and are placed into
 124   the LOCAL_VARIABLES collection.
 125 
 126   Args:
 127     func: Function to be called.
 128 
 129   Returns:
 130     A tuple (variables, constants) where the first element is a list of
 131     trainable variables and the second is the non-trainable variables.
 132   """
 133   variables = []
 134   constants = []
 135 
 136   # We need to create these variables like normal, so grab the original
 137   # constructor before we mock it.
 138   original_init = tf.Variable.__init__
 139 
 140   def custom_init(self, *args, **kwargs):
 141     trainable = kwargs["trainable"]
 142     kwargs["trainable"] = False
 143     # Making these variables local keeps them out of the optimizer's checkpoints
 144     # somehow.
 145     kwargs["collections"] = [tf.GraphKeys.LOCAL_VARIABLES]
 146     original_init(self, *args, **kwargs)
 147     if trainable:
 148       variables.append(self)
 149     else:
 150       constants.append(self)
 151 
 152   # This name-scope is just a nicety for TensorBoard.
 153   with tf.name_scope("unused_graph"):
 154     with mock.patch.object(tf.Variable, "__init__", custom_init):
 155       func()
 156 
 157   return variables, constants

 160 def _make_with_custom_variables(func, variable_mapping):
 161   """Calls func and replaces the value of some variables created in it.
 162 
 163   Args:
 164     func: Function to be called.
 165     variable_mapping: A mapping of variable name to the replacement tensor or
 166       tf.Variable.
 167 
 168   Returns:
 169     The return value of func is returned.
 170   """
 171   original_value = tf.Variable.value
 172 
 173   def custom_value(self):
 174     if self.name in variable_mapping:
 175       replacement = variable_mapping[self.name]
 176       tf.logging.info("Replaced %s with %s" % (self.name, replacement))
 177 
 178       # value() method needs to return a tensor, we need to call value on it.
 179       # This has to be done manually like this otherwise we'll get an infinite
 180       # loop.
 181       if isinstance(replacement, tf.Variable):
 182         replacement = original_value(replacement)
 183 
 184       return replacement
 185     else:
 186       return original_value(self)
 187 
 188   with mock.patch.object(tf.Variable, "value", custom_value):
 189     with mock.patch.object(tf.Variable, "_AsTensor", custom_value):
 190       return func()
