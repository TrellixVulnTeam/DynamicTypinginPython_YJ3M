  29 class TrainableAdam(opt.TrainableOptimizer):
  30   """Adam optimizer with learnable scalar parameters.
  31 
  32   See Kingma et. al., 2014 for algorithm (http://arxiv.org/abs/1412.6980).
  33   """
  34 
  35   def __init__(self,
  36                learning_rate=1e-3,
  37                beta1=0.9,
  38                beta2=0.999,
  39                epsilon=1e-8,
  40                **kwargs):
  41     """Initializes the TrainableAdam optimizer with the given initial values.
  42 
  43     Args:
  44       learning_rate: The learning rate (default: 1e-3).
  45       beta1: The exponential decay rate for the 1st moment estimates.
  46       beta2: The exponential decay rate for the 2nd moment estimates.
  47       epsilon: A small constant for numerical stability.
  48       **kwargs: Any additional keyword arguments for TrainableOptimizer.
  49 
  50     Raises:
  51       ValueError: if the learning rate or epsilon is not positive
  52       ValueError: if beta1 or beta2 is not in (0, 1).
  53     """
  54     if learning_rate <= 0:
  55       raise ValueError("Learning rate must be positive.")
  56     if epsilon <= 0:
  57       raise ValueError("Epsilon must be positive.")
  58     if not 0 < beta1 < 1 or not 0 < beta2 < 1:
  59       raise ValueError("Beta values must be between 0 and 1, exclusive.")
  60 
  61     self._reuse_vars = False
  62 
  63     with tf.variable_scope(opt.OPTIMIZER_SCOPE):
  64       def inv_sigmoid(x):
  65         return np.log(x / (1.0 - x))
  66 
  67       self.log_learning_rate = tf.get_variable(
  68           "log_learning_rate",
  69           shape=[],
  70           initializer=tf.constant_initializer(np.log(learning_rate)))
  71       self.beta1_logit = tf.get_variable(
  72           "beta1_logit",
  73           shape=[],
  74           initializer=tf.constant_initializer(inv_sigmoid(beta1)))
  75       self.beta2_logit = tf.get_variable(
  76           "beta2_logit",
  77           shape=[],
  78           initializer=tf.constant_initializer(inv_sigmoid(beta2)))
  79       self.log_epsilon = tf.get_variable(
  80           "log_epsilon",
  81           shape=[],
  82           initializer=tf.constant_initializer(np.log(epsilon)))
  83 
  84     # Key names are derived from Algorithm 1 described in
  85     # https://arxiv.org/pdf/1412.6980.pdf
  86     state_keys = ["m", "v", "t"]
  87     super(TrainableAdam, self).__init__("Adam", state_keys, **kwargs)
  88 
  89   def _initialize_state(self, var):
  90     """Returns a dictionary mapping names of state variables to their values."""
  91     vectorized_shape = var.get_shape().num_elements(), 1
  92 
  93     return {key: tf.zeros(vectorized_shape) for key in self.state_keys}
  94 
  95   def _compute_update(self, param, grad, state):
  96     """Calculates the new internal state and parameters.
  97 
  98     If the gradient is sparse, updates the appropriate slices in the internal
  99     state and stacks the update tensor.
 100 
 101     Args:
 102       param: A tensor of parameters.
 103       grad: A tensor of gradients with the same shape as param.
 104       state: A dictionary containing any state for the optimizer.
 105 
 106     Returns:
 107       updated_param: The updated parameters.
 108       updated_state: The updated state variables in a dictionary.
 109     """
 110 
 111     with tf.variable_scope(opt.OPTIMIZER_SCOPE) as scope:
 112 
 113       if self._reuse_vars:
 114         scope.reuse_variables()
 115       else:
 116         self._reuse_vars = True
 117 
 118       (grad_values, first_moment, second_moment, timestep, grad_indices
 119       ) = self._extract_gradients_and_internal_state(
 120           grad, state, tf.shape(param))
 121 
 122       beta1 = tf.nn.sigmoid(self.beta1_logit)
 123       beta2 = tf.nn.sigmoid(self.beta2_logit)
 124       epsilon = tf.exp(self.log_epsilon) + 1e-10
 125       learning_rate = tf.exp(self.log_learning_rate)
 126 
 127       old_grad_shape = tf.shape(grad_values)
 128       grad_values = tf.reshape(grad_values, [-1, 1])
 129 
 130       new_timestep = timestep + 1
 131       new_first_moment = self._update_adam_estimate(
 132           first_moment, grad_values, beta1)
 133       new_second_moment = self._debias_adam_estimate(
 134           second_moment, tf.square(grad_values), beta2)
 135 
 136       debiased_first_moment = self._debias_adam_estimate(
 137           new_first_moment, beta1, new_timestep)
 138       debiased_second_moment = self._debias_adam_estimate(
 139           new_second_moment, beta2, new_timestep)
 140 
 141       # Propagating through the square root of 0 is very bad for stability.
 142       update = (learning_rate * debiased_first_moment /
 143                 (tf.sqrt(debiased_second_moment + 1e-10) + epsilon))
 144 
 145       update = tf.reshape(update, old_grad_shape)
 146 
 147       if grad_indices is not None:
 148         param_shape = tf.shape(param)
 149         update = utils.stack_tensor(
 150             update, grad_indices, param, param_shape[:1])
 151         new_first_moment = utils.update_slices(
 152             new_first_moment, grad_indices, state["m"], param_shape)
 153         new_second_moment = utils.update_slices(
 154             new_second_moment, grad_indices, state["v"], param_shape)
 155         new_timestep = utils.update_slices(
 156             new_timestep, grad_indices, state["t"], param_shape)
 157 
 158       new_param = param - update
 159 
 160       # collect the update and new state
 161       new_state = {
 162           "m": new_first_moment,
 163           "v": new_second_moment,
 164           "t": new_timestep
 165       }
 166 
 167     return new_param, new_state
 168 
 169   def _update_adam_estimate(self, estimate, value, beta):
 170     """Returns a beta-weighted average of estimate and value."""
 171     return (beta * estimate) + ((1 - beta) * value)
 172 
 173   def _debias_adam_estimate(self, estimate, beta, t_step):
 174     """Returns a debiased estimate based on beta and the timestep."""
 175     return estimate / (1 - tf.pow(beta, t_step))
 176 
 177   def _extract_gradients_and_internal_state(self, grad, state, param_shape):
 178     """Extracts the gradients and relevant internal state.
 179 
 180     If the gradient is sparse, extracts the appropriate slices from the state.
 181 
 182     Args:
 183       grad: The current gradient.
 184       state: The current state.
 185       param_shape: The shape of the parameter (used if gradient is sparse).
 186 
 187     Returns:
 188       grad_values: The gradient value tensor.
 189       first_moment: The first moment tensor (internal state).
 190       second_moment: The second moment tensor (internal state).
 191       timestep: The current timestep (internal state).
 192       grad_indices: The indices for the gradient tensor, if sparse.
 193           None otherwise.
 194     """
 195     grad_values = grad
 196     grad_indices = None
 197     first_moment = state["m"]
 198     second_moment = state["v"]
 199     timestep = state["t"]
 200 
 201     if isinstance(grad, tf.IndexedSlices):
 202       grad_indices, grad_values = utils.accumulate_sparse_gradients(grad)
 203       first_moment = utils.slice_tensor(
 204           first_moment, grad_indices, param_shape)
 205       second_moment = utils.slice_tensor(
 206           second_moment, grad_indices, param_shape)
 207       timestep = utils.slice_tensor(timestep, grad_indices, param_shape)
 208 
 209     return grad_values, first_moment, second_moment, timestep, grad_indices
