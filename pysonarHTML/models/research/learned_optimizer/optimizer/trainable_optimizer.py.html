  35 class TrainableOptimizer(tf.train.Optimizer):
  36   """Base class for trainable optimizers.
  37 
  38   A trainable optimizer is an optimizer that has parameters that can themselves
  39   be learned (meta-optimized).
  40 
  41   Subclasses must implement:
  42       _compute_update(self, param, grad, state)
  43   """
  44 
  45   def __init__(self, name, state_keys, use_attention=False,
  46                use_log_objective=False, obj_train_max_multiplier=-1,
  47                use_second_derivatives=True, use_numerator_epsilon=False,
  48                **kwargs):
  49     """Initializes the optimizer with the given name and settings.
  50 
  51     Args:
  52       name: The name string for this optimizer.
  53       state_keys: The names of any required state variables (list)
  54       use_attention: Whether this optimizer uses attention (Default: True)
  55       use_log_objective: Whether this optimizer uses the logarithm of the
  56           objective when computing the loss (Default: False)
  57       obj_train_max_multiplier: The maximum multiplier for the increase in the
  58           objective before meta-training is stopped. If <= 0, meta-training is
  59           not stopped early. (Default: -1)
  60       use_second_derivatives: Whether this optimizer uses second derivatives in
  61           meta-training. This should be set to False if some second derivatives
  62           in the meta-training problem set are not defined in Tensorflow.
  63           (Default: True)
  64       use_numerator_epsilon: Whether to use epsilon in the numerator when
  65           scaling the problem objective during meta-training. (Default: False)
  66       **kwargs: Any additional keyword arguments.
  67     """
  68     self.use_second_derivatives = use_second_derivatives
  69     self.state_keys = sorted(state_keys)
  70     self.use_attention = use_attention
  71     self.use_log_objective = use_log_objective
  72     self.obj_train_max_multiplier = obj_train_max_multiplier
  73     self.use_numerator_epsilon = use_numerator_epsilon
  74 
  75     use_locking = False
  76     super(TrainableOptimizer, self).__init__(use_locking, name)
  77 
  78   def _create_slots(self, var_list):
  79     """Creates all slots needed by the variables.
  80 
  81     Args:
  82       var_list: A list of `Variable` objects.
  83     """
  84     for var in var_list:
  85       init_states = self._initialize_state(var)
  86       for slot_name in sorted(init_states):
  87         slot_var_name = "{}_{}".format(self.get_name(), slot_name)
  88         value = init_states[slot_name]
  89         self._get_or_make_slot(var, value, slot_name, slot_var_name)
  90 
  91   def _initialize_state(self, var):
  92     """Initializes any state required for this variable.
  93 
  94     Args:
  95       var: a tensor containing parameters to be optimized
  96 
  97     Returns:
  98       state: a dictionary mapping state keys to initial state values (tensors)
  99     """
 100     return {}
 101 
 102   def _initialize_global_state(self):
 103     """Initializes any global state values."""
 104     return []

 106   def _apply_common(self, grad, var):
 107     """Applies the optimizer updates to the variables.
 108 
 109     Note: this should only get called via _apply_dense or _apply_sparse when
 110     using the optimizer via optimizer.minimize or optimizer.apply_gradients.
 111     During meta-training, the optimizer.train function should be used to
 112     construct an optimization path that is differentiable.
 113 
 114     Args:
 115       grad: A tensor representing the gradient.
 116       var: A tf.Variable with the same shape as grad.
 117 
 118     Returns:
 119       update_op: A tensorflow op that assigns new values to the variable, and
 120           also defines dependencies that update the state variables for the
 121           optimizer.
 122     """
 123     state = {key: self.get_slot(var, key) for key in self.get_slot_names()}
 124     new_var, new_state = self._compute_update(var, grad, state)
 125     state_assign_ops = [tf.assign(state_var, new_state[key])
 126                         for key, state_var in state.items()]
 127     with tf.control_dependencies(state_assign_ops):
 128       update_op = var.assign(new_var)
 129 
 130     return update_op
 131 
 132   def _apply_dense(self, grad, var):
 133     """Adds ops to apply dense gradients to 'var'."""
 134     return self._apply_common(grad, var)
 135 
 136   def _apply_sparse(self, grad, var):
 137     """Adds ops to apply sparse gradients to 'var'."""
 138     return self._apply_common(grad, var)
 139 
 140   def _compute_update(self, param, grad, state):
 141     """Computes the update step for optimization.
 142 
 143     Args:
 144       param: A tensor of parameters to optimize.
 145       grad: The gradient tensor of the objective with respect to the parameters.
 146           (It has the same shape as param.)
 147       state: A dictionary containing any extra state required by the optimizer.
 148 
 149     Returns:
 150       updated_params: The updated parameters.
 151       updated_state: The dictionary of updated state variable(s).
 152     """
 153     raise NotImplementedError
 154 
 155   def _compute_updates(self, params, grads, states, global_state):
 156     """Maps the compute update functions for each parameter.
 157 
 158     This function can be overriden by a subclass if the subclass wants to
 159     combine information across the different parameters in the list.
 160 
 161     Args:
 162       params: A list of parameter tensors.
 163       grads: A list of gradients corresponding to each parameter.
 164       states: A list of state variables corresponding to each parameter.
 165       global_state: A list of global state variables for the problem.
 166 
 167     Returns:
 168       new_params: The updated parameters.
 169       new_states: The updated states.
 170       new_global_state: The updated global state.
 171       attention_params: A list of attention parameters. This is the same as
 172           new_params if the optimizer does not use attention.
 173     """
 174     # Zip up the arguments to _compute_update.
 175     args = zip(params, grads, states)
 176 
 177     # Call compute_update on each set of parameter/gradient/state args.
 178     new_params, new_states = zip(*list(
 179         itertools.starmap(self._compute_update, args)))
 180 
 181     # Global state is unused in the basic case, just pass it through.
 182     return list(new_params), list(new_states), global_state, list(new_params)

 184   def train(self, problem, dataset):
 185     """Creates graph operations to train the optimizer.
 186 
 187     Args:
 188       problem: A problem_generator.Problem instance to train on.
 189       dataset: A datasets.Dataset tuple to use when training.
 190 
 191     Returns:
 192       meta_objective: A tensorflow operation for computing the meta-objective
 193       obj_weights: A tensor placeholder for feeding in the objective weights
 194       obj_values: The subproblem objective values during optimization
 195       batches: The batch indexes tensor for overriding with feed_dict
 196       first_unroll: A placeholder signifying if this is a first unroll
 197         (this will propagate the gradients slightly differently).
 198       reset_state: A placeholder signifying that the rnn state should be reset.
 199       output_state: The final state of the optimizer
 200       init_loop_vars_to_override: Local variables that can be assigned to
 201         propagate the optimizer and problem state for unrolling
 202       final_loop_vals: Final values of the loop variables that can be
 203         assigned to init_loop_vars_to_override.
 204     """
 205 
 206     # Placeholder for the objective weights
 207     obj_weights = tf.placeholder(tf.float32)
 208     num_iter = tf.shape(obj_weights)[0]
 209 
 210     # Unpack the dataset and generate the minibatches for training
 211     data, labels = dataset
 212     # Convert the ndarrays to tensors so we can pass them back in via feed_dict
 213     data = tf.constant(data)
 214     labels = tf.constant(labels)
 215     batches = tf.placeholder(tf.int32)
 216     first_unroll = tf.placeholder_with_default(False, [])
 217     reset_state = tf.placeholder_with_default(False, [])
 218 
 219     training_output = collections.namedtuple("TrainingOutput",
 220                                              ["metaobj",
 221                                               "obj_weights",
 222                                               "problem_objectives",
 223                                               "initial_obj",
 224                                               "batches",
 225                                               "first_unroll",
 226                                               "reset_state",
 227                                               "output_state",
 228                                               "init_loop_vars",
 229                                               "output_loop_vars"])
 230 
 231     def loop_body(itr, obj_accum, params, attend_params, flattened_states,
 232                   global_state, all_obj, unused_init_obj, data,
 233                   labels, batches):
 234       """Body of the meta-training while loop for optimizing a sub-problem.
 235 
 236       Args:
 237         itr: The current meta-training iteration.
 238         obj_accum: The accumulated objective over all training steps so far.
 239         params: The parameters of the sub-problem.
 240         attend_params: The parameters of the sub-problems at the attended
 241             location.
 242         flattened_states: The states of the trainable optimizer, sorted and
 243             flattened into a list (since a while loop can't handle nested lists
 244             or dictionaries).
 245         global_state: The global state of the optimizer.
 246         all_obj: The list of all objective values in the training process.
 247         unused_init_obj: The initial objective (unused here, but needed in the
 248             variable list because it's used in a stopping condition in the
 249             loop_cond.)
 250         data: The data for this problem.
 251         labels: The labels corresponding to the data.
 252         batches: The batch indexes needed for shuffled minibatch creation.
 253 
 254       Returns:
 255         itr: The updated meta-training iteration.
 256         obj_accum: The updated accumulated objective.
 257         params: The new parameters of the sub-problem.
 258         attend_params: The new parameters of the sub-problems at the attended
 259             location.
 260         flattened_states: The new states of the trainable optimizer.
 261         global_state: The updated global state.
 262         all_obj: The updates list of all objective values.
 263         unused_init_obj: The initial objective.
 264         data: The data for this problem.
 265         labels: The labels corresponding to the data.
 266         batches: The batch indexes needed for shuffled minibatch creation.
 267       """
 268       batch_indices = tf.gather(batches, itr)
 269       batch_data = tf.gather(data, batch_indices)
 270       batch_labels = tf.gather(labels, batch_indices)
 271 
 272       # Compute the objective over the entire dataset (full batch).
 273       obj = problem.objective(params, data, labels)
 274 
 275       # Compute the gradients on just the current batch
 276       if self.use_attention:
 277         current_obj = problem.objective(attend_params, batch_data, batch_labels)
 278         grads = problem.gradients(current_obj, attend_params)
 279       else:
 280         current_obj = problem.objective(params, batch_data, batch_labels)
 281         grads = problem.gradients(current_obj, params)
 282 
 283       if not self.use_second_derivatives:
 284         new_grads = []
 285         for grad in grads:
 286           if isinstance(grad, tf.IndexedSlices):
 287             new_grads.append(
 288                 tf.IndexedSlices(tf.stop_gradient(grad.values), grad.indices))
 289           else:
 290             new_grads.append(tf.stop_gradient(grad))
 291         grads = new_grads
 292 
 293       # store the objective value for the entire problem at each iteration
 294       all_obj = tf.concat([all_obj, tf.reshape(obj, (1,))], 0)
 295 
 296       # accumulate the weighted objective for the entire dataset
 297       acc = tf.gather(obj_weights, itr) * obj
 298 
 299       obj_accum = tf.add(obj_accum, acc)
 300       # Set the shape to keep the shape invariant for obj_accum. Without this,
 301       # the graph builder thinks the tensor shape is unknown on the 2nd iter.
 302       obj_accum.set_shape([])
 303 
 304       # convert flattened_states to dictionaries
 305       dict_states = [dict(zip(self.state_keys, flat_state))
 306                      for flat_state in flattened_states]
 307 
 308       # compute the new parameters and states
 309       args = (params, grads, dict_states, global_state)
 310       updates = self._compute_updates(*args)
 311       new_params, new_states, new_global_state, new_attend_params = updates
 312 
 313       # flatten the states
 314       new_flattened_states = map(flatten_and_sort, new_states)
 315 
 316       return [itr + 1, obj_accum, new_params, new_attend_params,
 317               new_flattened_states, new_global_state, all_obj, unused_init_obj,
 318               data, labels, batches]
 319 
 320     def loop_cond(itr, obj_accum, unused_params, unused_attend_params,
 321                   unused_flattened_states, unused_global_state, all_obj,
 322                   init_obj, *args):
 323       """Termination conditions of the sub-problem optimization loop."""
 324       del args  # unused
 325 
 326       cond1 = tf.less(itr, num_iter)  # We've run < num_iter times
 327       cond2 = tf.is_finite(obj_accum)  # The objective is still finite
 328 
 329       if self.obj_train_max_multiplier > 0:
 330         current_obj = tf.gather(all_obj, itr)
 331         # Account for negative init_obj too
 332         max_diff = (self.obj_train_max_multiplier - 1) * tf.abs(init_obj)
 333         max_obj = init_obj + max_diff
 334         # The objective is a reasonable multiplier of the original objective
 335         cond3 = tf.less(current_obj, max_obj)
 336 
 337         return tf.logical_and(tf.logical_and(cond1, cond2), cond3,
 338                               name="training_loop_cond")
 339       else:
 340         return tf.logical_and(cond1, cond2, name="training_loop_cond")
 341 
 342     init = self._initialize_training_loop_parameters(
 343         problem, data, labels, batches, first_unroll, reset_state)
 344     loop_vars, invariants, initial_obj, init_loop_vars_to_override = init
 345 
 346     loop_output = tf.while_loop(loop_cond, loop_body, loop_vars,
 347                                 swap_memory=True, shape_invariants=invariants)
 348     meta_obj, problem_objectives = loop_output[1], loop_output[6]
 349 
 350     # The meta objective is normalized by the initial objective at the start of
 351     # the series of partial unrolls.
 352     scaled_meta_objective = self.scale_objective(
 353         meta_obj, problem_objectives, initial_obj)
 354 
 355     final_loop_vals = (
 356         [initial_obj] + loop_output[2] + loop_output[3] + loop_output[5])
 357     final_loop_vals.extend(itertools.chain(*loop_output[4]))
 358 
 359     return training_output(scaled_meta_objective,
 360                            obj_weights,
 361                            problem_objectives,
 362                            initial_obj,
 363                            batches,
 364                            first_unroll,
 365                            reset_state,
 366                            loop_output[4],
 367                            init_loop_vars_to_override,
 368                            final_loop_vals)
 369 
 370   def _initialize_training_loop_parameters(
 371       self, problem, data, labels, batches, first_unroll, reset_state):
 372     """Initializes the vars and params needed for the training process.
 373 
 374     Args:
 375       problem: The problem being optimized.
 376       data: The data for the problem.
 377       labels: The corresponding labels for the data.
 378       batches: The indexes needed to create shuffled batches of the data.
 379       first_unroll: Whether this is the first unroll in a partial unrolling.
 380       reset_state: Whether RNN state variables should be reset.
 381 
 382     Returns:
 383       loop_vars: The while loop variables for training.
 384       invariants: The corresponding variable shapes (required by while loop).
 385       initial_obj: The initial objective (used later for scaling).
 386       init_loop_vars_to_override: The loop vars that can be overridden when
 387           performing training via partial unrolls.
 388     """
 389     # Extract these separately so we don't have to make inter-variable
 390     # dependencies.
 391     initial_tensors = problem.init_tensors()
 392 
 393     return_initial_tensor_values = first_unroll
 394     initial_params_vars, initial_params = local_state_variables(
 395         initial_tensors, return_initial_tensor_values)
 396     initial_attend_params_vars, initial_attend_params = local_state_variables(
 397         initial_tensors, return_initial_tensor_values)
 398     # Recalculate the initial objective for the list on each partial unroll with
 399     # the new initial_params. initial_obj holds the value from the very first
 400     # unroll.
 401     initial_obj_init = problem.objective(initial_params, data, labels)
 402     return_initial_obj_init = first_unroll
 403     [initial_obj_var], [initial_obj] = local_state_variables(
 404         [initial_obj_init], return_initial_obj_init)
 405 
 406     # Initialize the loop variables.
 407     initial_itr = tf.constant(0, dtype=tf.int32)
 408     initial_meta_obj = tf.constant(0, dtype=tf.float32)
 409     # N.B. the use of initial_obj_init here rather than initial_obj
 410     initial_problem_objectives = tf.reshape(initial_obj_init, (1,))
 411 
 412     # Initialize the extra state.
 413     initial_state_vars = []
 414     initial_state = []
 415     state_shapes = []
 416     return_initial_state_values = reset_state
 417     for param in initial_tensors:
 418       param_state_vars, param_state = local_state_variables(
 419           flatten_and_sort(self._initialize_state(param)),
 420           return_initial_state_values)
 421 
 422       initial_state_vars.append(param_state_vars)
 423       initial_state.append(param_state)
 424       state_shapes.append([f.get_shape() for f in param_state])
 425 
 426     # Initialize any global (problem-level) state.
 427     initial_global_state_vars, initial_global_state = local_state_variables(
 428         self._initialize_global_state(), return_initial_state_values)
 429 
 430     global_shapes = []
 431     for item in initial_global_state:
 432       global_shapes.append(item.get_shape())
 433 
 434     # build the list of loop variables:
 435     loop_vars = [
 436         initial_itr,
 437         initial_meta_obj,
 438         initial_params,         # Local variables.
 439         initial_attend_params,  # Local variables.
 440         initial_state,          # Local variables.
 441         initial_global_state,   # Local variables.
 442         initial_problem_objectives,
 443         initial_obj,            # Local variable.
 444         data,
 445         labels,
 446         batches,
 447     ]
 448 
 449     invariants = [
 450         initial_itr.get_shape(),
 451         initial_meta_obj.get_shape(),
 452         [t.get_shape() for t in initial_params],
 453         [t.get_shape() for t in initial_attend_params],
 454         state_shapes,
 455         global_shapes,
 456         tensor_shape.TensorShape([None]),   # The problem objectives list grows
 457         initial_obj.get_shape(),
 458         tensor_shape.unknown_shape(),  # Placeholder shapes are unknown
 459         tensor_shape.unknown_shape(),
 460         tensor_shape.unknown_shape(),
 461     ]
 462 
 463     # Initialize local variables that we will override with final tensors at the
 464     # next iter.
 465     init_loop_vars_to_override = (
 466         [initial_obj_var] + initial_params_vars + initial_attend_params_vars +
 467         initial_global_state_vars)
 468     init_loop_vars_to_override.extend(itertools.chain(*initial_state_vars))
 469 
 470     return loop_vars, invariants, initial_obj, init_loop_vars_to_override

 472   def scale_objective(self, total_obj, all_objs, initial_obj,
 473                       obj_scale_eps=1e-6):
 474     """Normalizes the objective based on the initial objective value.
 475 
 476     Args:
 477       total_obj: The total accumulated objective over the training run.
 478       all_objs: A list of all the individual objectives over the training run.
 479       initial_obj: The initial objective value.
 480       obj_scale_eps: The epsilon value to use in computations for stability.
 481 
 482     Returns:
 483       The scaled objective as a single value.
 484     """
 485     if self.use_log_objective:
 486       if self.use_numerator_epsilon:
 487         scaled_problem_obj = ((all_objs + obj_scale_eps) /
 488                               (initial_obj + obj_scale_eps))
 489         log_scaled_problem_obj = tf.log(scaled_problem_obj)
 490       else:
 491         scaled_problem_obj = all_objs / (initial_obj + obj_scale_eps)
 492         log_scaled_problem_obj = tf.log(scaled_problem_obj + obj_scale_eps)
 493       return tf.reduce_mean(log_scaled_problem_obj)
 494     else:
 495       return total_obj / (initial_obj + obj_scale_eps)

 498 def local_state_variables(init_values, return_init_values):
 499   """Create local variables initialized from init_values.
 500 
 501   This will create local variables from a list of init_values. Each variable
 502   will be named based on the value's shape and dtype.
 503 
 504   As a convenience, a boolean tensor allows you to return value from
 505   the created local variable or from the original init value.
 506 
 507   Args:
 508     init_values: iterable of tensors
 509     return_init_values: boolean tensor
 510 
 511   Returns:
 512     local_vars: list of the created local variables.
 513     vals: if return_init_values is true, then this returns the values of
 514       init_values. Otherwise it returns the values of the local_vars.
 515   """
 516   if not init_values:
 517     return [], []
 518 
 519   # This generates a harmless warning when saving the metagraph.
 520   variable_use_count = tf.get_collection_ref(_LOCAL_STATE_VARIABLE_COLLECTION)
 521   if not variable_use_count:
 522     variable_use_count.append(collections.defaultdict(int))
 523   variable_use_count = variable_use_count[0]
 524 
 525   local_vars = []
 526   with tf.variable_scope(OPTIMIZER_SCOPE):
 527     # We can't use the init_value as an initializer as init_value may
 528     # itself depend on some problem variables. This would produce
 529     # inter-variable initialization order dependence which TensorFlow
 530     # sucks at making easy.
 531     for init_value in init_values:
 532       name = create_local_state_variable_name(init_value)
 533       unique_name = name + "_" + str(variable_use_count[name])
 534       variable_use_count[name] += 1
 535       # The overarching idea here is to be able to reuse variables between
 536       # different sessions on the same TensorFlow master without errors. By
 537       # uniquifying based on the type and name we mirror the checks made inside
 538       # TensorFlow, while still allowing some memory reuse. Ultimately this is a
 539       # hack due to the broken Session.reset().
 540       local_vars.append(
 541           tf.get_local_variable(
 542               unique_name,
 543               initializer=tf.zeros(
 544                   init_value.get_shape(), dtype=init_value.dtype)))
 545 
 546   # It makes things a lot simpler if we use the init_value the first
 547   # iteration, instead of the variable itself. It allows us to propagate
 548   # gradients through it as well as simplifying initialization. The variable
 549   # ends up assigned to after the first iteration.
 550   vals = tf.cond(return_init_values, lambda: init_values, lambda: local_vars)
 551   if len(init_values) == 1:
 552     # tf.cond extracts elements from singleton lists.
 553     vals = [vals]
 554   return local_vars, vals

 557 def create_local_state_variable_name(tensor):
 558   """Create a name of the variable based on its type and shape."""
 559   if not tensor.get_shape().is_fully_defined():
 560     raise ValueError("Need a fully specified shape to create a local variable.")
 561 
 562   return (_LOCAL_VARIABLE_PREFIX + "_".join(
 563       map(str, tensor.get_shape().as_list())) + "_" + tensor.dtype.name)
 564

 566 def is_local_state_variable(op):
 567   """Returns if this op is a local state variable created for training."""
 568   return op.node_def.op in ["Variable", "VariableV2"] and op.name.startswith(
 569       OPTIMIZER_SCOPE + "/" + _LOCAL_VARIABLE_PREFIX)

 572 def flatten_and_sort(dictionary):
 573   """Flattens a dictionary into a list of values sorted by the keys."""
 574   return [dictionary[k] for k in sorted(dictionary.keys())]
