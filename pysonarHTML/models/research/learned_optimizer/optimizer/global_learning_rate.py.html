  16 """A trainable optimizer that learns a single global learning rate."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 
  24 from learned_optimizer.optimizer import trainable_optimizer
  25 
  26 
  27 class GlobalLearningRate(trainable_optimizer.TrainableOptimizer):
  28   """Optimizes for a single global learning rate."""
  29 
  30   def __init__(self, initial_rate=1e-3, **kwargs):
  31     """Initializes the global learning rate."""
  32     with tf.variable_scope(trainable_optimizer.OPTIMIZER_SCOPE):
  33       initializer = tf.constant_initializer(initial_rate)
  34       self.learning_rate = tf.get_variable("global_learning_rate", shape=(),
  35                                            initializer=initializer)
  36     super(GlobalLearningRate, self).__init__("GLR", [], **kwargs)
  37 
  38   def _compute_update(self, param, grad, state):
  39     return param - tf.scalar_mul(self.learning_rate, grad), state
