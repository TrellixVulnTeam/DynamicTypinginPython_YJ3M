  16 """Custom RNN cells for hierarchical RNNs."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 
  24 from learned_optimizer.optimizer import utils
  25 
  26 
  27 class BiasGRUCell(tf.contrib.rnn.RNNCell):
  28   """GRU cell (cf. http://arxiv.org/abs/1406.1078) with an additional bias."""
  29 
  30   def __init__(self, num_units, activation=tf.tanh, scale=0.1,
  31                gate_bias_init=0., random_seed=None):
  32     self._num_units = num_units
  33     self._activation = activation
  34     self._scale = scale
  35     self._gate_bias_init = gate_bias_init
  36     self._random_seed = random_seed
  37 
  38   @property
  39   def state_size(self):
  40     return self._num_units
  41 
  42   @property
  43   def output_size(self):
  44     return self._num_units
  45 
  46   def __call__(self, inputs, state, bias=None):
  47     # Split the injected bias vector into a bias for the r, u, and c updates.
  48     if bias is None:
  49       bias = tf.zeros((1, 3))
  50 
  51     r_bias, u_bias, c_bias = tf.split(bias, 3, 1)
  52 
  53     with tf.variable_scope(type(self).__name__):  # "BiasGRUCell"
  54       with tf.variable_scope("gates"):  # Reset gate and update gate.
  55         proj = utils.affine([inputs, state], 2 * self._num_units,
  56                             scale=self._scale, bias_init=self._gate_bias_init,
  57                             random_seed=self._random_seed)
  58         r_lin, u_lin = tf.split(proj, 2, 1)
  59         r, u = tf.nn.sigmoid(r_lin + r_bias), tf.nn.sigmoid(u_lin + u_bias)
  60 
  61       with tf.variable_scope("candidate"):
  62         proj = utils.affine([inputs, r * state], self._num_units,
  63                             scale=self._scale, random_seed=self._random_seed)
  64         c = self._activation(proj + c_bias)
  65 
  66       new_h = u * state + (1 - u) * c
  67 
  68     return new_h, new_h
