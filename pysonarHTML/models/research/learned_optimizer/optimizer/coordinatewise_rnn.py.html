  42 class CoordinatewiseRNN(opt.TrainableOptimizer):
  43   """RNN that operates on each coordinate of the problem independently."""
  44 
  45   def __init__(self,
  46                cell_sizes,
  47                cell_cls,
  48                init_lr_range=(1., 1.),
  49                dynamic_output_scale=True,
  50                learnable_decay=True,
  51                zero_init_lr_weights=False,
  52                **kwargs):
  53     """Initializes the RNN per-parameter optimizer.
  54 
  55     Args:
  56       cell_sizes: List of hidden state sizes for each RNN cell in the network
  57       cell_cls: tf.contrib.rnn class for specifying the RNN cell type
  58       init_lr_range: the range in which to initialize the learning rates.
  59       dynamic_output_scale: whether to learn weights that dynamically modulate
  60           the output scale (default: True)
  61       learnable_decay: whether to learn weights that dynamically modulate the
  62           input scale via RMS style decay (default: True)
  63       zero_init_lr_weights: whether to initialize the lr weights to zero
  64       **kwargs: args passed to TrainableOptimizer's constructor
  65 
  66     Raises:
  67       ValueError: If the init lr range is not of length 2.
  68       ValueError: If the init lr range is not a valid range (min > max).
  69     """
  70     if len(init_lr_range) != 2:
  71       raise ValueError(
  72           "Initial LR range must be len 2, was {}".format(len(init_lr_range)))
  73     if init_lr_range[0] > init_lr_range[1]:
  74       raise ValueError("Initial LR range min is greater than max.")
  75     self.init_lr_range = init_lr_range
  76 
  77     self.zero_init_lr_weights = zero_init_lr_weights
  78     self.reuse_vars = False
  79 
  80     # create the RNN cell
  81     with tf.variable_scope(opt.OPTIMIZER_SCOPE):
  82       self.component_cells = [cell_cls(sz) for sz in cell_sizes]
  83       self.cell = tf.contrib.rnn.MultiRNNCell(self.component_cells)
  84 
  85       # random normal initialization scaled by the output size
  86       scale_factor = FLAGS.crnn_rnn_readout_scale / math.sqrt(cell_sizes[-1])
  87       scaled_init = tf.random_normal_initializer(0., scale_factor)
  88 
  89       # weights for projecting the hidden state to a parameter update
  90       self.update_weights = tf.get_variable("update_weights",
  91                                             shape=(cell_sizes[-1], 1),
  92                                             initializer=scaled_init)
  93 
  94       self._initialize_decay(learnable_decay, (cell_sizes[-1], 1), scaled_init)
  95 
  96       self._initialize_lr(dynamic_output_scale, (cell_sizes[-1], 1),
  97                           scaled_init)
  98 
  99       state_size = sum([sum(state_size) for state_size in self.cell.state_size])
 100       self._init_vector = tf.get_variable(
 101           "init_vector", shape=[1, state_size],
 102           initializer=tf.random_uniform_initializer(-1., 1.))
 103 
 104     state_keys = ["rms", "rnn", "learning_rate", "decay"]
 105     super(CoordinatewiseRNN, self).__init__("cRNN", state_keys, **kwargs)
 106 
 107   def _initialize_decay(
 108       self, learnable_decay, weights_tensor_shape, scaled_init):
 109     """Initializes the decay weights and bias variables or tensors.
 110 
 111     Args:
 112       learnable_decay: Whether to use learnable decay.
 113       weights_tensor_shape: The shape the weight tensor should take.
 114       scaled_init: The scaled initialization for the weights tensor.
 115     """
 116     if learnable_decay:
 117 
 118       # weights for projecting the hidden state to the RMS decay term
 119       self.decay_weights = tf.get_variable("decay_weights",
 120                                            shape=weights_tensor_shape,
 121                                            initializer=scaled_init)
 122       self.decay_bias = tf.get_variable(
 123           "decay_bias", shape=(1,),
 124           initializer=tf.constant_initializer(
 125               FLAGS.crnn_default_decay_var_init))
 126     else:
 127       self.decay_weights = tf.zeros_like(self.update_weights)
 128       self.decay_bias = tf.constant(FLAGS.crnn_default_decay_var_init)
 129 
 130   def _initialize_lr(
 131       self, dynamic_output_scale, weights_tensor_shape, scaled_init):
 132     """Initializes the learning rate weights and bias variables or tensors.
 133 
 134     Args:
 135       dynamic_output_scale: Whether to use a dynamic output scale.
 136       weights_tensor_shape: The shape the weight tensor should take.
 137       scaled_init: The scaled initialization for the weights tensor.
 138     """
 139     if dynamic_output_scale:
 140       zero_init = tf.constant_initializer(0.)
 141       wt_init = zero_init if self.zero_init_lr_weights else scaled_init
 142       self.lr_weights = tf.get_variable("learning_rate_weights",
 143                                         shape=weights_tensor_shape,
 144                                         initializer=wt_init)
 145       self.lr_bias = tf.get_variable("learning_rate_bias", shape=(1,),
 146                                      initializer=zero_init)
 147     else:
 148       self.lr_weights = tf.zeros_like(self.update_weights)
 149       self.lr_bias = tf.zeros([1, 1])
 150 
 151   def _initialize_state(self, var):
 152     """Return a dictionary mapping names of state variables to their values."""
 153     vectorized_shape = [var.get_shape().num_elements(), 1]
 154 
 155     min_lr = self.init_lr_range[0]
 156     max_lr = self.init_lr_range[1]
 157     if min_lr == max_lr:
 158       init_lr = tf.constant(min_lr, shape=vectorized_shape)
 159     else:
 160       actual_vals = tf.random_uniform(vectorized_shape,
 161                                       np.log(min_lr),
 162                                       np.log(max_lr))
 163       init_lr = tf.exp(actual_vals)
 164 
 165     ones = tf.ones(vectorized_shape)
 166     rnn_init = ones * self._init_vector
 167 
 168     return {
 169         "rms": tf.ones(vectorized_shape),
 170         "learning_rate": init_lr,
 171         "rnn": rnn_init,
 172         "decay": tf.ones(vectorized_shape),
 173     }
 174 
 175   def _compute_update(self, param, grad, state):
 176     """Update parameters given the gradient and state.
 177 
 178     Args:
 179       param: tensor of parameters
 180       grad: tensor of gradients with the same shape as param
 181       state: a dictionary containing any state for the optimizer
 182 
 183     Returns:
 184       updated_param: updated parameters
 185       updated_state: updated state variables in a dictionary
 186     """
 187 
 188     with tf.variable_scope(opt.OPTIMIZER_SCOPE) as scope:
 189 
 190       if self.reuse_vars:
 191         scope.reuse_variables()
 192       else:
 193         self.reuse_vars = True
 194 
 195       param_shape = tf.shape(param)
 196 
 197       (grad_values, decay_state, rms_state, rnn_state, learning_rate_state,
 198        grad_indices) = self._extract_gradients_and_internal_state(
 199            grad, state, param_shape)
 200 
 201       # Vectorize and scale the gradients.
 202       grad_scaled, rms = utils.rms_scaling(grad_values, decay_state, rms_state)
 203 
 204       # Apply the RNN update.
 205       rnn_state_tuples = self._unpack_rnn_state_into_tuples(rnn_state)
 206       rnn_output, rnn_state_tuples = self.cell(grad_scaled, rnn_state_tuples)
 207       rnn_state = self._pack_tuples_into_rnn_state(rnn_state_tuples)
 208 
 209       # Compute the update direction (a linear projection of the RNN output).
 210       delta = utils.project(rnn_output, self.update_weights)
 211 
 212       # The updated decay is an affine projection of the hidden state
 213       decay = utils.project(rnn_output, self.decay_weights,
 214                             bias=self.decay_bias, activation=tf.nn.sigmoid)
 215 
 216       # Compute the change in learning rate (an affine projection of the RNN
 217       # state, passed through a 2x sigmoid, so the change is bounded).
 218       learning_rate_change = 2. * utils.project(rnn_output, self.lr_weights,
 219                                                 bias=self.lr_bias,
 220                                                 activation=tf.nn.sigmoid)
 221 
 222       # Update the learning rate.
 223       new_learning_rate = learning_rate_change * learning_rate_state
 224 
 225       # Apply the update to the parameters.
 226       update = tf.reshape(new_learning_rate * delta, tf.shape(grad_values))
 227 
 228       if isinstance(grad, tf.IndexedSlices):
 229         update = utils.stack_tensor(update, grad_indices, param,
 230                                     param_shape[:1])
 231         rms = utils.update_slices(rms, grad_indices, state["rms"], param_shape)
 232         new_learning_rate = utils.update_slices(new_learning_rate, grad_indices,
 233                                                 state["learning_rate"],
 234                                                 param_shape)
 235         rnn_state = utils.update_slices(rnn_state, grad_indices, state["rnn"],
 236                                         param_shape)
 237         decay = utils.update_slices(decay, grad_indices, state["decay"],
 238                                     param_shape)
 239 
 240       new_param = param - update
 241 
 242       # Collect the update and new state.
 243       new_state = {
 244           "rms": rms,
 245           "learning_rate": new_learning_rate,
 246           "rnn": rnn_state,
 247           "decay": decay,
 248       }
 249 
 250     return new_param, new_state

 252   def _extract_gradients_and_internal_state(self, grad, state, param_shape):
 253     """Extracts the gradients and relevant internal state.
 254 
 255     If the gradient is sparse, extracts the appropriate slices from the state.
 256 
 257     Args:
 258       grad: The current gradient.
 259       state: The current state.
 260       param_shape: The shape of the parameter (used if gradient is sparse).
 261 
 262     Returns:
 263       grad_values: The gradient value tensor.
 264       decay_state: The current decay state.
 265       rms_state: The current rms state.
 266       rnn_state: The current state of the internal rnns.
 267       learning_rate_state: The current learning rate state.
 268       grad_indices: The indices for the gradient tensor, if sparse.
 269           None otherwise.
 270     """
 271     if isinstance(grad, tf.IndexedSlices):
 272       grad_indices, grad_values = utils.accumulate_sparse_gradients(grad)
 273       decay_state = utils.slice_tensor(state["decay"], grad_indices,
 274                                        param_shape)
 275       rms_state = utils.slice_tensor(state["rms"], grad_indices, param_shape)
 276       rnn_state = utils.slice_tensor(state["rnn"], grad_indices, param_shape)
 277       learning_rate_state = utils.slice_tensor(state["learning_rate"],
 278                                                grad_indices, param_shape)
 279       decay_state.set_shape([None, 1])
 280       rms_state.set_shape([None, 1])
 281     else:
 282       grad_values = grad
 283       grad_indices = None
 284 
 285       decay_state = state["decay"]
 286       rms_state = state["rms"]
 287       rnn_state = state["rnn"]
 288       learning_rate_state = state["learning_rate"]
 289     return (grad_values, decay_state, rms_state, rnn_state, learning_rate_state,
 290             grad_indices)

 292   def _unpack_rnn_state_into_tuples(self, rnn_state):
 293     """Creates state tuples from the rnn state vector."""
 294     rnn_state_tuples = []
 295     cur_state_pos = 0
 296     for cell in self.component_cells:
 297       total_state_size = sum(cell.state_size)
 298       cur_state = tf.slice(rnn_state, [0, cur_state_pos],
 299                            [-1, total_state_size])
 300       cur_state_tuple = tf.split(value=cur_state, num_or_size_splits=2,
 301                                  axis=1)
 302       rnn_state_tuples.append(cur_state_tuple)
 303       cur_state_pos += total_state_size
 304     return rnn_state_tuples
 305 
 306   def _pack_tuples_into_rnn_state(self, rnn_state_tuples):
 307     """Creates a single state vector concatenated along column axis."""
 308     rnn_state = None
 309     for new_state_tuple in rnn_state_tuples:
 310       new_c, new_h = new_state_tuple
 311       if rnn_state is None:
 312         rnn_state = tf.concat([new_c, new_h], axis=1)
 313       else:
 314         rnn_state = tf.concat([rnn_state, tf.concat([new_c, new_h], 1)], axis=1)
 315     return rnn_state
