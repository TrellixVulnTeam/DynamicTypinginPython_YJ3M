  26 def make_finite(t, replacement):
  27   """Replaces non-finite tensor values with the replacement value."""
  28   return tf.where(tf.is_finite(t), t, replacement)

  31 def asinh(x):
  32   """Computes the inverse hyperbolic sine function (in tensorflow)."""
  33   return tf.log(x + tf.sqrt(1. + x ** 2))

  36 def affine(inputs, output_size, scope="Affine", scale=0.1, vec_mean=0.,
  37            include_bias=True, bias_init=0., random_seed=None):
  38   """Computes an affine function of the inputs.
  39 
  40   Creates or recalls tensorflow variables "Matrix" and "Bias"
  41   to generate an affine operation on the input.
  42 
  43   If the inputs are a list of tensors, they are concatenated together.
  44 
  45   Initial weights for the matrix are drawn from a Gaussian with zero
  46   mean and standard deviation that is the given scale divided by the
  47   square root of the input dimension. Initial weights for the bias are
  48   set to zero.
  49 
  50   Args:
  51     inputs: List of tensors with shape (batch_size, input_size)
  52     output_size: Size (dimension) of the output
  53     scope: Variable scope for these parameters (default: "Affine")
  54     scale: Initial weight scale for the matrix parameters (default: 0.1),
  55       this constant is divided by the sqrt of the input size to get the
  56       std. deviation of the initial weights
  57     vec_mean: The mean for the random initializer
  58     include_bias: Whether to include the bias term
  59     bias_init: The initializer bias (default 0.)
  60     random_seed: Random seed for random initializers. (Default: None)
  61 
  62   Returns:
  63     output: Tensor with shape (batch_size, output_size)
  64   """
  65 
  66   # Concatenate the input arguments.
  67   x = tf.concat(inputs, 1)
  68 
  69   with tf.variable_scope(scope):
  70     input_size = x.get_shape().as_list()[1]
  71 
  72     sigma = scale / np.sqrt(input_size)
  73     rand_init = tf.random_normal_initializer(mean=vec_mean, stddev=sigma,
  74                                              seed=random_seed)
  75 
  76     matrix = tf.get_variable("Matrix", [input_size, output_size],
  77                              dtype=tf.float32, initializer=rand_init)
  78 
  79     if include_bias:
  80       bias = tf.get_variable("Bias", [output_size], dtype=tf.float32,
  81                              initializer=tf.constant_initializer(bias_init,
  82                                                                  tf.float32))
  83     else:
  84       bias = 0.
  85     output = tf.matmul(x, matrix) + bias
  86 
  87   return output

  90 def project(inputs, weights, bias=0., activation=tf.identity):
  91   """Computes an affine or linear projection of the inputs.
  92 
  93   Projects the inputs onto the given weight vector and (optionally)
  94   adds a bias and passes the result through an activation function.
  95 
  96   Args:
  97     inputs: matrix of inputs with shape [batch_size, dim]
  98     weights: weight matrix with shape [dim, output_dim]
  99     bias: bias vector with shape [output_dim] (default: 0)
 100     activation: nonlinear activation function (default: tf.identity)
 101 
 102   Returns:
 103     outputs: an op which computes activation(inputs @ weights + bias)
 104   """
 105   return activation(tf.matmul(inputs, weights) + bias)

 108 def new_mean_squared(grad_vec, decay, ms):
 109   """Calculates the new accumulated mean squared of the gradient.
 110 
 111   Args:
 112     grad_vec: the vector for the current gradient
 113     decay: the decay term
 114     ms: the previous mean_squared value
 115 
 116   Returns:
 117     the new mean_squared value
 118   """
 119   decay_size = decay.get_shape().num_elements()
 120   decay_check_ops = [
 121       tf.assert_less_equal(decay, 1., summarize=decay_size),
 122       tf.assert_greater_equal(decay, 0., summarize=decay_size)]
 123 
 124   with tf.control_dependencies(decay_check_ops):
 125     grad_squared = tf.square(grad_vec)
 126 
 127   # If the previous mean_squared is the 0 vector, don't use the decay and just
 128   # return the full grad_squared. This should only happen on the first timestep.
 129   decay = tf.cond(tf.reduce_all(tf.equal(ms, 0.)),
 130                   lambda: tf.zeros_like(decay, dtype=tf.float32), lambda: decay)
 131 
 132   # Update the running average of squared gradients.
 133   epsilon = 1e-12
 134   return (1. - decay) * (grad_squared + epsilon) + decay * ms

 137 def rms_scaling(gradient, decay, ms, update_ms=True):
 138   """Vectorizes and scales a tensor of gradients.
 139 
 140   Args:
 141     gradient: the current gradient
 142     decay: the current decay value.
 143     ms: the previous mean squared value
 144     update_ms: Whether to update the mean squared value (default: True)
 145 
 146   Returns:
 147     The scaled gradient and the new ms value if update_ms is True,
 148     the old ms value otherwise.
 149   """
 150 
 151   # Vectorize the gradients and compute the squared gradients.
 152   grad_vec = tf.reshape(gradient, [-1, 1])
 153 
 154   if update_ms:
 155     ms = new_mean_squared(grad_vec, decay, ms)
 156 
 157   # Scale the current gradients by the RMS, squashed by the asinh function.
 158   scaled_gradient = asinh(grad_vec / tf.sqrt(ms + 1e-16))
 159 
 160   return scaled_gradient, ms

 163 def accumulate_sparse_gradients(grad):
 164   """Accumulates repeated indices of a sparse gradient update.
 165 
 166   Args:
 167     grad: a tf.IndexedSlices gradient
 168 
 169   Returns:
 170     grad_indices: unique indices
 171     grad_values: gradient values corresponding to the indices
 172   """
 173 
 174   grad_indices, grad_segments = tf.unique(grad.indices)
 175   grad_values = tf.unsorted_segment_sum(grad.values, grad_segments,
 176                                         tf.shape(grad_indices)[0])
 177   return grad_indices, grad_values

 180 def slice_tensor(dense_tensor, indices, head_dims):
 181   """Extracts slices from a partially flattened dense tensor.
 182 
 183   indices is assumed to index into the first dimension of head_dims.
 184   dense_tensor is assumed to have a shape [D_0, D_1, ...] such that
 185   prod(head_dims) == D_0. This function will extract slices along the
 186   first_dimension of head_dims.
 187 
 188   Example:
 189 
 190   Consider a tensor with shape head_dims = [100, 2] and a dense_tensor with
 191   shape [200, 3]. Note that the first dimension of dense_tensor equals the
 192   product of head_dims. This function will reshape dense_tensor such that
 193   its shape is now [100, 2, 3] (i.e. the first dimension became head-dims)
 194   and then slice it along the first dimension. After slicing, the slices will
 195   have their initial dimensions flattened just as they were in dense_tensor
 196   (e.g. if there are 4 indices, the return value will have a shape of [4, 3]).
 197 
 198   Args:
 199     dense_tensor: a N-D dense tensor. Shape: [D_0, D_1, ...]
 200     indices: a 1-D integer tensor. Shape: [K]
 201     head_dims: True dimensions of the dense_tensor's first dimension.
 202 
 203   Returns:
 204     Extracted slices. Shape [K, D_1, ...]
 205   """
 206 
 207   tail_dims = tf.shape(dense_tensor)[1:]
 208   dense_tensor = tf.reshape(dense_tensor,
 209                             tf.concat([head_dims, tail_dims], 0))
 210 
 211   slices = tf.gather(dense_tensor, indices)
 212   # NOTE(siege): This kills the shape annotation.
 213   return tf.reshape(slices, tf.concat([[-1], tail_dims], 0))

 216 def stack_tensor(slices, indices, dense_tensor, head_dims):
 217   """Reconsititutes a tensor from slices and corresponding indices.
 218 
 219   This is an inverse operation to slice_tensor. Missing slices are set to 0.
 220 
 221   Args:
 222     slices: a tensor. Shape [K, D_1, ...]
 223     indices: a 1-D integer tensor. Shape: [K]
 224     dense_tensor: the original tensor the slices were taken
 225       from. Shape: [D_0, D_1, ...]
 226     head_dims: True dimensions of the dense_tensor's first dimension.
 227 
 228   Returns:
 229     Reconsituted tensor. Shape: [D_0, D_1, ...]
 230   """
 231   # NOTE(siege): This cast shouldn't be necessary.
 232   indices = tf.cast(indices, tf.int32)
 233 
 234   tail_dims = tf.shape(dense_tensor)[1:]
 235   dense_shape = tf.concat([head_dims, tail_dims], 0)
 236 
 237   slices = tf.reshape(slices, tf.concat([[-1], dense_shape[1:]], 0))
 238   indices = tf.expand_dims(indices, -1)
 239 
 240   return tf.reshape(tf.scatter_nd(indices, slices, dense_shape),
 241                     tf.shape(dense_tensor))

 244 def update_slices(slices, indices, dense_tensor, head_dims):
 245   """Reconstitutes a tensor from slices and corresponding indices.
 246 
 247   Like _stack_tensor, but instead of setting missing slices to 0, sets them to
 248   what they were in the original tensor. The return value is reshaped to be
 249   the same as dense_tensor.
 250 
 251   Args:
 252     slices: a tensor. Shape [K, D_1, ...]
 253     indices: a 1-D integer tensor. Shape: [K]
 254     dense_tensor: the original tensor the slices were taken
 255       from. Shape: [D_0, D_1, ...]
 256     head_dims: True dimensions of the dense_tensor's first dimension.
 257 
 258   Returns:
 259     Reconsituted tensor. Shape: [D_0, D_1, ...]
 260   """
 261   # NOTE(siege): This cast shouldn't be necessary.
 262   indices = tf.cast(indices, tf.int32)
 263 
 264   tail_dims = tf.shape(dense_tensor)[1:]
 265   dense_shape = tf.concat([head_dims, tail_dims], 0)
 266 
 267   update_mask_vals = tf.fill(tf.shape(indices), 1)
 268   reshaped_indices = tf.expand_dims(indices, -1)
 269   update_mask = tf.equal(
 270       tf.scatter_nd(reshaped_indices, update_mask_vals, head_dims[:1]), 1)
 271 
 272   reshaped_dense_slices = tf.reshape(
 273       stack_tensor(slices, indices, dense_tensor, head_dims), dense_shape)
 274   reshaped_dense_tensor = tf.reshape(dense_tensor, dense_shape)
 275 
 276   return tf.reshape(
 277       tf.where(update_mask, reshaped_dense_slices, reshaped_dense_tensor),
 278       tf.shape(dense_tensor))
