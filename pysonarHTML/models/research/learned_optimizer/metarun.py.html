 222 def register_optimizers():
 223   opts = {}
 224   opts["CoordinatewiseRNN"] = coordinatewise_rnn.CoordinatewiseRNN
 225   opts["GlobalLearningRate"] = global_learning_rate.GlobalLearningRate
 226   opts["HierarchicalRNN"] = hierarchical_rnn.HierarchicalRNN
 227   opts["LearningRateSchedule"] = learning_rate_schedule.LearningRateSchedule
 228   opts["TrainableAdam"] = trainable_adam.TrainableAdam
 229   return opts
 230

 232 def main(unused_argv):
 233   """Runs the main script."""
 234 
 235   opts = register_optimizers()
 236 
 237   # Choose a set of problems to optimize. By default this includes quadratics,
 238   # 2-dimensional bowls, 2-class softmax problems, and non-noisy optimization
 239   # test problems (e.g. Rosenbrock, Beale)
 240   problems_and_data = []
 241 
 242   if FLAGS.include_sparse_softmax_problems:
 243     problems_and_data.extend(ps.sparse_softmax_2_class_sparse_problems())
 244 
 245   if FLAGS.include_one_hot_sparse_softmax_problems:
 246     problems_and_data.extend(
 247         ps.one_hot_sparse_softmax_2_class_sparse_problems())
 248 
 249   if FLAGS.include_quadratic_problems:
 250     problems_and_data.extend(ps.quadratic_problems())
 251 
 252   if FLAGS.include_noisy_quadratic_problems:
 253     problems_and_data.extend(ps.quadratic_problems_noisy())
 254 
 255   if FLAGS.include_large_quadratic_problems:
 256     problems_and_data.extend(ps.quadratic_problems_large())
 257 
 258   if FLAGS.include_bowl_problems:
 259     problems_and_data.extend(ps.bowl_problems())
 260 
 261   if FLAGS.include_noisy_bowl_problems:
 262     problems_and_data.extend(ps.bowl_problems_noisy())
 263 
 264   if FLAGS.include_softmax_2_class_problems:
 265     problems_and_data.extend(ps.softmax_2_class_problems())
 266 
 267   if FLAGS.include_noisy_softmax_2_class_problems:
 268     problems_and_data.extend(ps.softmax_2_class_problems_noisy())
 269 
 270   if FLAGS.include_optimization_test_problems:
 271     problems_and_data.extend(ps.optimization_test_problems())
 272 
 273   if FLAGS.include_noisy_optimization_test_problems:
 274     problems_and_data.extend(ps.optimization_test_problems_noisy())
 275 
 276   if FLAGS.include_fully_connected_random_2_class_problems:
 277     problems_and_data.extend(ps.fully_connected_random_2_class_problems())
 278 
 279   if FLAGS.include_matmul_problems:
 280     problems_and_data.extend(ps.matmul_problems())
 281 
 282   if FLAGS.include_log_objective_problems:
 283     problems_and_data.extend(ps.log_objective_problems())
 284 
 285   if FLAGS.include_rescale_problems:
 286     problems_and_data.extend(ps.rescale_problems())
 287 
 288   if FLAGS.include_norm_problems:
 289     problems_and_data.extend(ps.norm_problems())
 290 
 291   if FLAGS.include_noisy_norm_problems:
 292     problems_and_data.extend(ps.norm_problems_noisy())
 293 
 294   if FLAGS.include_sum_problems:
 295     problems_and_data.extend(ps.sum_problems())
 296 
 297   if FLAGS.include_noisy_sum_problems:
 298     problems_and_data.extend(ps.sum_problems_noisy())
 299 
 300   if FLAGS.include_sparse_gradient_problems:
 301     problems_and_data.extend(ps.sparse_gradient_problems())
 302     if FLAGS.include_fully_connected_random_2_class_problems:
 303       problems_and_data.extend(ps.sparse_gradient_problems_mlp())
 304 
 305   if FLAGS.include_min_max_well_problems:
 306     problems_and_data.extend(ps.min_max_well_problems())
 307 
 308   if FLAGS.include_sum_of_quadratics_problems:
 309     problems_and_data.extend(ps.sum_of_quadratics_problems())
 310 
 311   if FLAGS.include_projection_quadratic_problems:
 312     problems_and_data.extend(ps.projection_quadratic_problems())
 313 
 314   if FLAGS.include_outward_snake_problems:
 315     problems_and_data.extend(ps.outward_snake_problems())
 316 
 317   if FLAGS.include_dependency_chain_problems:
 318     problems_and_data.extend(ps.dependency_chain_problems())
 319 
 320   # log directory
 321   logdir = os.path.join(FLAGS.train_dir,
 322                         "{}_{}_{}_{}".format(FLAGS.optimizer,
 323                                              FLAGS.cell_cls,
 324                                              FLAGS.cell_size,
 325                                              FLAGS.num_cells))
 326 
 327   # get the optimizer class and arguments
 328   optimizer_cls = opts[FLAGS.optimizer]
 329 
 330   assert len(HRNN_CELL_SIZES) in [1, 2, 3]
 331   optimizer_args = (HRNN_CELL_SIZES,)
 332 
 333   optimizer_kwargs = {
 334       "init_lr_range": (FLAGS.min_lr, FLAGS.max_lr),
 335       "learnable_decay": FLAGS.learnable_decay,
 336       "dynamic_output_scale": FLAGS.dynamic_output_scale,
 337       "cell_cls": getattr(tf.contrib.rnn, FLAGS.cell_cls),
 338       "use_attention": FLAGS.use_attention,
 339       "use_log_objective": FLAGS.use_log_objective,
 340       "num_gradient_scales": FLAGS.num_gradient_scales,
 341       "zero_init_lr_weights": FLAGS.zero_init_lr_weights,
 342       "use_log_means_squared": FLAGS.use_log_means_squared,
 343       "use_relative_lr": FLAGS.use_relative_lr,
 344       "use_extreme_indicator": FLAGS.use_extreme_indicator,
 345       "max_log_lr": FLAGS.max_log_lr,
 346       "obj_train_max_multiplier": FLAGS.objective_training_max_multiplier,
 347       "use_problem_lr_mean": FLAGS.use_problem_lr_mean,
 348       "use_gradient_shortcut": FLAGS.use_gradient_shortcut,
 349       "use_second_derivatives": FLAGS.use_second_derivatives,
 350       "use_lr_shortcut": FLAGS.use_lr_shortcut,
 351       "use_grad_products": FLAGS.use_grad_products,
 352       "use_multiple_scale_decays": FLAGS.use_multiple_scale_decays,
 353       "use_numerator_epsilon": FLAGS.use_numerator_epsilon,
 354       "learnable_inp_decay": FLAGS.learnable_inp_decay,
 355       "learnable_rnn_init": FLAGS.learnable_rnn_init,
 356   }
 357   optimizer_spec = problem_spec.Spec(
 358       optimizer_cls, optimizer_args, optimizer_kwargs)
 359 
 360   # make log directory
 361   tf.gfile.MakeDirs(logdir)
 362 
 363   is_chief = FLAGS.task == 0
 364   # if this is a distributed run, make the chief run through problems in order
 365   select_random_problems = FLAGS.worker_tasks == 1 or not is_chief
 366 
 367   def num_unrolls():
 368     return metaopt.sample_numiter(FLAGS.num_unroll_scale, FLAGS.min_num_unrolls)
 369 
 370   def num_partial_unroll_itrs():
 371     return metaopt.sample_numiter(FLAGS.num_partial_unroll_itr_scale,
 372                                   FLAGS.min_num_itr_partial_unroll)
 373 
 374   # run it
 375   metaopt.train_optimizer(
 376       logdir,
 377       optimizer_spec,
 378       problems_and_data,
 379       FLAGS.num_problems,
 380       FLAGS.num_meta_iterations,
 381       num_unrolls,
 382       num_partial_unroll_itrs,
 383       learning_rate=FLAGS.meta_learning_rate,
 384       gradient_clip=FLAGS.gradient_clip_level,
 385       is_chief=is_chief,
 386       select_random_problems=select_random_problems,
 387       obj_train_max_multiplier=FLAGS.objective_training_max_multiplier,
 388       callbacks=[])
 389 
 390   return 0
 391 
 392 
 393 if __name__ == "__main__":
 394   tf.app.run()
