  58 def sigmoid_weights(n, slope=0.1, offset=5):
  59   """Generates a sigmoid, scaled to sum to 1.
  60 
  61   This function is used to generate weights that serve to mask out
  62   the early objective values of an optimization problem such that
  63   initial variation in the objective is phased out (hence the sigmoid
  64   starts at zero and ramps up to the maximum value, and the total
  65   weight is normalized to sum to one)
  66 
  67   Args:
  68     n: the number of samples
  69     slope: slope of the sigmoid (Default: 0.1)
  70     offset: threshold of the sigmoid (Default: 5)
  71 
  72   Returns:
  73     No
  74   """
  75   x = np.arange(n)
  76   y = 1. / (1. + np.exp(-slope * (x-offset)))
  77   y_normalized = y / np.sum(y)
  78   return y_normalized

  81 def sample_numiter(scale, min_steps=50):
  82   """Samples a number of iterations from an exponential distribution.
  83 
  84   Args:
  85     scale: parameter for the exponential distribution
  86     min_steps: minimum number of steps to run (additive)
  87 
  88   Returns:
  89     num_steps: An integer equal to a rounded sample from the exponential
  90                distribution + the value of min_steps.
  91   """
  92   return int(np.round(np.random.exponential(scale=scale)) + min_steps)

  95 def train_optimizer(logdir,
  96                     optimizer_spec,
  97                     problems_and_data,
  98                     num_problems,
  99                     num_meta_iterations,
 100                     num_unroll_func,
 101                     num_partial_unroll_itrs_func,
 102                     learning_rate=1e-4,
 103                     gradient_clip=5.,
 104                     is_chief=False,
 105                     select_random_problems=True,
 106                     callbacks=None,
 107                     obj_train_max_multiplier=-1,
 108                     out=sys.stdout):
 109   """Trains the meta-parameters of this optimizer.
 110 
 111   Args:
 112     logdir: a directory filepath for storing model checkpoints (must exist)
 113     optimizer_spec: specification for an Optimizer (see utils.Spec)
 114     problems_and_data: a list of tuples containing three elements: a problem
 115       specification (see utils.Spec), a dataset (see datasets.Dataset), and
 116       a batch_size (int) for generating a problem and corresponding dataset. If
 117       the problem doesn't have data, set dataset to None.
 118     num_problems: the number of problems to sample during meta-training
 119     num_meta_iterations: the number of iterations (steps) to run the
 120       meta-optimizer for on each subproblem.
 121     num_unroll_func: called once per meta iteration and returns the number of
 122       unrolls to do for that meta iteration.
 123     num_partial_unroll_itrs_func: called once per unroll and returns the number
 124       of iterations to do for that unroll.
 125     learning_rate: learning rate of the RMSProp meta-optimizer (Default: 1e-4)
 126     gradient_clip: value to clip gradients at (Default: 5.0)
 127     is_chief: whether this is the chief task (Default: False)
 128     select_random_problems: whether to select training problems randomly
 129         (Default: True)
 130     callbacks: a list of callback functions that is run after every random
 131         problem draw
 132     obj_train_max_multiplier: the maximum increase in the objective value over
 133         a single training run. Ignored if < 0.
 134     out: where to write output to, e.g. a file handle (Default: sys.stdout)
 135 
 136   Raises:
 137     ValueError: If one of the subproblems has a negative objective value.
 138   """
 139 
 140   if select_random_problems:
 141     # iterate over random draws of problem / dataset pairs
 142     sampler = (random.choice(problems_and_data) for _ in range(num_problems))
 143   else:
 144     # iterate over a random shuffle of problems, looping if necessary
 145     num_repeats = (num_problems / len(problems_and_data)) + 1
 146     random.shuffle(problems_and_data)
 147     sampler = (problems_and_data * num_repeats)[:num_problems]
 148 
 149   for problem_itr, (problem_spec, dataset, batch_size) in enumerate(sampler):
 150 
 151     # timer used to time how long it takes to initialize a problem
 152     problem_start_time = time.time()
 153 
 154     # if dataset is None, use the EMPTY_DATASET
 155     if dataset is None:
 156       dataset = datasets.EMPTY_DATASET
 157       batch_size = dataset.size
 158 
 159     # build a new graph for this problem
 160     graph = tf.Graph()
 161     real_device_setter = tf.train.replica_device_setter(FLAGS.ps_tasks)
 162 
 163     def custom_device_setter(op):
 164       # Places the local variables onto the workers.
 165       if trainable_optimizer.is_local_state_variable(op):
 166         return "/job:worker"
 167       else:
 168         return real_device_setter(op)
 169 
 170     if real_device_setter:
 171       device_setter = custom_device_setter
 172     else:
 173       device_setter = None
 174 
 175     with graph.as_default(), graph.device(device_setter):
 176 
 177       # initialize a problem
 178       problem = problem_spec.build()
 179 
 180       # build the optimizer
 181       opt = optimizer_spec.build()
 182 
 183       # get the meta-objective for training the optimizer
 184       train_output = opt.train(problem, dataset)
 185 
 186       state_keys = opt.state_keys
 187       for key, val in zip(state_keys, train_output.output_state[0]):
 188         finite_val = utils.make_finite(val, replacement=tf.zeros_like(val))
 189         tf.summary.histogram("State/{}".format(key), finite_val,
 190                              collections=[OPT_SUM_COLLECTION])
 191 
 192       tf.summary.scalar("MetaObjective", train_output.metaobj,
 193                         collections=[OPT_SUM_COLLECTION])
 194 
 195       # Per-problem meta-objective
 196       tf.summary.scalar(problem_spec.callable.__name__ + "_MetaObjective",
 197                         train_output.metaobj,
 198                         collections=[OPT_SUM_COLLECTION])
 199 
 200       # create the meta-train_op
 201       global_step = tf.Variable(0, name="global_step", trainable=False)
 202       meta_parameters = tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES,
 203                                           scope=OPTIMIZER_SCOPE)
 204       # parameter regularization
 205       reg_l2 = FLAGS.l2_reg * sum([tf.reduce_sum(param ** 2)
 206                                    for param in meta_parameters])
 207 
 208       # compute the meta-gradients
 209       meta_opt = tf.train.RMSPropOptimizer(learning_rate, decay=FLAGS.rms_decay,
 210                                            use_locking=True,
 211                                            epsilon=FLAGS.rms_epsilon)
 212       grads_and_vars = meta_opt.compute_gradients(train_output.metaobj + reg_l2,
 213                                                   meta_parameters)
 214 
 215       # clip the gradients
 216       clipped_grads_and_vars = []
 217       for grad, var in grads_and_vars:
 218         clipped_grad = tf.clip_by_value(
 219             utils.make_finite(grad, replacement=tf.zeros_like(var)),
 220             -gradient_clip, gradient_clip)
 221         clipped_grads_and_vars.append((clipped_grad, var))
 222 
 223       # histogram summary of grads and vars
 224       for grad, var in grads_and_vars:
 225         tf.summary.histogram(
 226             var.name + "_rawgrad",
 227             utils.make_finite(
 228                 grad, replacement=tf.zeros_like(grad)),
 229             collections=[OPT_SUM_COLLECTION])
 230       for grad, var in clipped_grads_and_vars:
 231         tf.summary.histogram(var.name + "_var", var,
 232                              collections=[OPT_SUM_COLLECTION])
 233         tf.summary.histogram(var.name + "_grad", grad,
 234                              collections=[OPT_SUM_COLLECTION])
 235 
 236       # builds the train and summary operations
 237       train_op = meta_opt.apply_gradients(clipped_grads_and_vars,
 238                                           global_step=global_step)
 239 
 240       # only grab summaries defined for LOL, not inside the problem
 241       summary_op = tf.summary.merge_all(key=OPT_SUM_COLLECTION)
 242 
 243       # make sure the state gets propagated after the gradients and summaries
 244       # were computed.
 245       with tf.control_dependencies([train_op, summary_op]):
 246         propagate_loop_state_ops = []
 247         for dest, src in zip(
 248             train_output.init_loop_vars, train_output.output_loop_vars):
 249           propagate_loop_state_ops.append(dest.assign(src))
 250         propagate_loop_state_op = tf.group(*propagate_loop_state_ops)
 251 
 252       # create the supervisor
 253       sv = tf.train.Supervisor(
 254           graph=graph,
 255           is_chief=is_chief,
 256           logdir=logdir,
 257           summary_op=None,
 258           save_model_secs=0,      # we save checkpoints manually
 259           global_step=global_step,
 260       )
 261 
 262       with sv.managed_session() as sess:
 263 
 264         init_time = time.time() - problem_start_time
 265         out.write("--------- Problem #{} ---------\n".format(problem_itr))
 266         out.write("{callable.__name__}{args}{kwargs}\n".format(
 267             **problem_spec.__dict__))
 268         out.write("Took {} seconds to initialize.\n".format(init_time))
 269         out.flush()
 270 
 271         # For profiling summaries
 272         if FLAGS.set_profiling:
 273           summary_writer = tf.summary.FileWriter(logdir, graph=sess.graph)
 274 
 275         # used to store information during training
 276         metadata = defaultdict(list)
 277 
 278         for k in range(num_meta_iterations):
 279 
 280           if sv.should_stop():
 281             break
 282 
 283           problem.init_fn(sess)
 284 
 285           # set run options (for profiling)
 286           full_trace_opt = tf.RunOptions(trace_level=tf.RunOptions.FULL_TRACE)
 287           run_options = full_trace_opt if FLAGS.set_profiling else None
 288           run_metadata = tf.RunMetadata() if FLAGS.set_profiling else None
 289 
 290           num_unrolls = num_unroll_func()
 291           partial_unroll_iters = [
 292               num_partial_unroll_itrs_func() for _ in xrange(num_unrolls)
 293           ]
 294           total_num_iter = sum(partial_unroll_iters)
 295 
 296           objective_weights = [np.ones(num) / float(num)
 297                                for num in partial_unroll_iters]
 298           db = dataset.batch_indices(total_num_iter, batch_size)
 299           dataset_batches = []
 300           last_index = 0
 301           for num in partial_unroll_iters:
 302             dataset_batches.append(db[last_index:last_index + num])
 303             last_index += num
 304 
 305           train_start_time = time.time()
 306 
 307           unroll_itr = 0
 308           additional_log_info = ""
 309 
 310           for unroll_itr in range(num_unrolls):
 311             first_unroll = unroll_itr == 0
 312             if FLAGS.reset_rnn_params:
 313               reset_state = first_unroll and k == 0
 314             else:
 315               reset_state = first_unroll
 316 
 317             feed = {
 318                 train_output.obj_weights: objective_weights[unroll_itr],
 319                 train_output.batches: dataset_batches[unroll_itr],
 320                 train_output.first_unroll: first_unroll,
 321                 train_output.reset_state: reset_state,
 322             }
 323 
 324             # run the train and summary ops
 325             # when a "save_diagnostics" flag is turned on
 326             fetches_list = [
 327                 train_output.metaobj,
 328                 train_output.problem_objectives,
 329                 train_output.initial_obj,
 330                 summary_op,
 331                 clipped_grads_and_vars,
 332                 train_op
 333             ]
 334             if unroll_itr + 1 < num_unrolls:
 335               fetches_list += [propagate_loop_state_op]
 336 
 337             fetched = sess.run(fetches_list, feed_dict=feed,
 338                                options=run_options, run_metadata=run_metadata)
 339             meta_obj = fetched[0]
 340             sub_obj = fetched[1]
 341             init_obj = fetched[2]
 342             summ = fetched[3]
 343             meta_grads_and_params = fetched[4]
 344 
 345             # assert that the subproblem objectives are non-negative
 346             # (this is so that we can rescale the objective by the initial value
 347             # and not worry about rescaling by a negative value)
 348             if np.any(sub_obj < 0):
 349               raise ValueError(
 350                   "Training problem objectives must be nonnegative.")
 351             # If the objective has increased more than we want, exit this
 352             # training run and start over on another meta iteration.
 353             if obj_train_max_multiplier > 0 and (
 354                 sub_obj[-1] > (init_obj +
 355                                abs(init_obj) * (obj_train_max_multiplier - 1))):
 356               msg = " Broke early at {} out of {} unrolls. ".format(
 357                   unroll_itr + 1, num_unrolls)
 358               additional_log_info += msg
 359               break
 360 
 361             # only the chief task is allowed to write the summary
 362             if is_chief:
 363               sv.summary_computed(sess, summ)
 364 
 365             metadata["subproblem_objs"].append(sub_obj)
 366             # store training metadata to pass to the callback
 367             metadata["meta_objs"].append(meta_obj)
 368             metadata["meta_grads_and_params"].append(meta_grads_and_params)
 369 
 370           optimization_time = time.time() - train_start_time
 371 
 372           if FLAGS.set_profiling:
 373             summary_name = "%02d_iter%04d_%02d" % (FLAGS.task, problem_itr, k)
 374             summary_writer.add_run_metadata(run_metadata, summary_name)
 375 
 376           metadata["global_step"].append(sess.run(global_step))
 377           metadata["runtimes"].append(optimization_time)
 378 
 379           # write a diagnostic message to the output
 380           args = (k, meta_obj, optimization_time,
 381                   sum(partial_unroll_iters[:unroll_itr+1]))
 382           out.write("  [{:02}] {}, {} seconds, {} iters ".format(*args))
 383           out.write("(unrolled {} steps)".format(
 384               ", ".join([str(s) for s in partial_unroll_iters[:unroll_itr+1]])))
 385           out.write("{}\n".format(additional_log_info))
 386           out.flush()
 387 
 388         if FLAGS.set_profiling:
 389           summary_writer.close()
 390 
 391         # force a checkpoint save before we load a new problem
 392         # only the chief task has the save_path and can write the checkpoint
 393         if is_chief:
 394           sv.saver.save(sess, sv.save_path, global_step=global_step)
 395 
 396     # run the callbacks on the chief
 397     if is_chief and callbacks is not None:
 398       for callback in callbacks:
 399         if hasattr(callback, "__call__"):
 400           problem_name = problem_spec.callable.__name__
 401           callback(problem_name, problem_itr, logdir, metadata)

 404 def test_optimizer(optimizer,
 405                    problem,
 406                    num_iter,
 407                    dataset=datasets.EMPTY_DATASET,
 408                    batch_size=None,
 409                    seed=None,
 410                    graph=None,
 411                    logdir=None,
 412                    record_every=None):
 413   """Tests an optimization algorithm on a given problem.
 414 
 415   Args:
 416     optimizer: Either a tf.train.Optimizer instance, or an Optimizer instance
 417                inheriting from trainable_optimizer.py
 418     problem: A Problem instance that defines an optimization problem to solve
 419     num_iter: The number of iterations of the optimizer to run
 420     dataset: The dataset to train the problem against
 421     batch_size: The number of samples per batch. If None (default), the
 422       batch size is set to the full batch (dataset.size)
 423     seed: A random seed used for drawing the initial parameters, or a list of
 424       numpy arrays used to explicitly initialize the parameters.
 425     graph: The tensorflow graph to execute (if None, uses the default graph)
 426     logdir: A directory containing model checkpoints. If given, then the
 427             parameters of the optimizer are loaded from the latest checkpoint
 428             in this folder.
 429     record_every: if an integer, stores the parameters, objective, and gradient
 430                   every recored_every iterations. If None, nothing is stored
 431 
 432   Returns:
 433     objective_values: A list of the objective values during optimization
 434     parameters: The parameters obtained after training
 435     records: A dictionary containing lists of the parameters and gradients
 436              during optimization saved every record_every iterations (empty if
 437              record_every is set to None)
 438   """
 439 
 440   if dataset is None:
 441     dataset = datasets.EMPTY_DATASET
 442     batch_size = dataset.size
 443   else:
 444     # default batch size is the entire dataset
 445     batch_size = dataset.size if batch_size is None else batch_size
 446 
 447   graph = tf.get_default_graph() if graph is None else graph
 448   with graph.as_default():
 449 
 450     # define the parameters of the optimization problem
 451     if isinstance(seed, (list, tuple)):
 452       # seed is a list of arrays
 453       params = problem_generator.init_fixed_variables(seed)
 454     else:
 455       # seed is an int or None
 456       params = problem.init_variables(seed)
 457 
 458     data_placeholder = tf.placeholder(tf.float32)
 459     labels_placeholder = tf.placeholder(tf.int32)
 460 
 461     # get the problem objective and gradient(s)
 462     obj = problem.objective(params, data_placeholder, labels_placeholder)
 463     gradients = problem.gradients(obj, params)
 464 
 465     vars_to_preinitialize = params
 466 
 467   with tf.Session(graph=graph) as sess:
 468     # initialize the parameter scope variables; necessary for apply_gradients
 469     sess.run(tf.variables_initializer(vars_to_preinitialize))
 470     coord = tf.train.Coordinator()
 471     threads = tf.train.start_queue_runners(sess=sess, coord=coord)
 472 
 473     # create the train operation and training variables
 474     try:
 475       train_op, real_params = optimizer.apply_gradients(zip(gradients, params))
 476       obj = problem.objective(real_params, data_placeholder, labels_placeholder)
 477     except TypeError:
 478       # If all goes well, this exception should only be thrown when we are using
 479       # a non-hrnn optimizer.
 480       train_op = optimizer.apply_gradients(zip(gradients, params))
 481 
 482     vars_to_restore = tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES,
 483                                         scope=OPTIMIZER_SCOPE)
 484     vars_to_initialize = list(
 485         set(tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)) -
 486         set(vars_to_restore) - set(vars_to_preinitialize))
 487     # load or initialize optimizer variables
 488     if logdir is not None:
 489       restorer = tf.Saver(var_list=vars_to_restore)
 490       ckpt = tf.train.latest_checkpoint(logdir)
 491       restorer.restore(sess, ckpt)
 492     else:
 493       sess.run(tf.variables_initializer(vars_to_restore))
 494     # initialize all the other variables
 495     sess.run(tf.variables_initializer(vars_to_initialize))
 496 
 497     problem.init_fn(sess)
 498 
 499     # generate the minibatch indices
 500     batch_inds = dataset.batch_indices(num_iter, batch_size)
 501 
 502     # run the train operation for n iterations and save the objectives
 503     records = defaultdict(list)
 504     objective_values = []
 505     for itr, batch in enumerate(batch_inds):
 506 
 507       # data to feed in
 508       feed = {data_placeholder: dataset.data[batch],
 509               labels_placeholder: dataset.labels[batch]}
 510       full_feed = {data_placeholder: dataset.data,
 511                    labels_placeholder: dataset.labels}
 512 
 513       # record stuff
 514       if record_every is not None and (itr % record_every) == 0:
 515         def grad_value(g):
 516           if isinstance(g, tf.IndexedSlices):
 517             return g.values
 518           else:
 519             return g
 520 
 521         records_fetch = {}
 522         for p in params:
 523           for key in optimizer.get_slot_names():
 524             v = optimizer.get_slot(p, key)
 525             records_fetch[p.name + "_" + key] = v
 526         gav_fetch = [(grad_value(g), v) for g, v in zip(gradients, params)]
 527 
 528         _, gav_eval, records_eval = sess.run(
 529             (obj, gav_fetch, records_fetch), feed_dict=feed)
 530         full_obj_eval = sess.run([obj], feed_dict=full_feed)
 531 
 532         records["objective"].append(full_obj_eval)
 533         records["grad_norm"].append([np.linalg.norm(g.ravel())
 534                                      for g, _ in gav_eval])
 535         records["param_norm"].append([np.linalg.norm(v.ravel())
 536                                       for _, v in gav_eval])
 537         records["grad"].append([g for g, _ in gav_eval])
 538         records["param"].append([v for _, v in gav_eval])
 539         records["iter"].append(itr)
 540 
 541         for k, v in records_eval.iteritems():
 542           records[k].append(v)
 543 
 544       # run the optimization train operation
 545       objective_values.append(sess.run([train_op, obj], feed_dict=feed)[1])
 546 
 547     # final parameters
 548     parameters = [sess.run(p) for p in params]
 549     coord.request_stop()
 550     coord.join(threads)
 551 
 552   return objective_values, parameters, records

 555 def run_wall_clock_test(optimizer,
 556                         problem,
 557                         num_steps,
 558                         dataset=datasets.EMPTY_DATASET,
 559                         seed=None,
 560                         logdir=None,
 561                         batch_size=None):
 562   """Runs optimization with the given parameters and return average iter time.
 563 
 564   Args:
 565     optimizer: The tf.train.Optimizer instance
 566     problem: The problem to optimize (a problem_generator.Problem)
 567     num_steps: The number of steps to run optimization for
 568     dataset: The dataset to train the problem against
 569     seed: The seed used for drawing the initial parameters, or a list of
 570       numpy arrays used to explicitly initialize the parameters
 571     logdir: A directory containing model checkpoints. If given, then the
 572             parameters of the optimizer are loaded from the latest checkpoint
 573             in this folder.
 574     batch_size: The number of samples per batch.
 575 
 576   Returns:
 577     The average time in seconds for a single optimization iteration.
 578   """
 579   if dataset is None:
 580     dataset = datasets.EMPTY_DATASET
 581     batch_size = dataset.size
 582   else:
 583     # default batch size is the entire dataset
 584     batch_size = dataset.size if batch_size is None else batch_size
 585 
 586   # define the parameters of the optimization problem
 587   if isinstance(seed, (list, tuple)):
 588     # seed is a list of arrays
 589     params = problem_generator.init_fixed_variables(seed)
 590   else:
 591     # seed is an int or None
 592     params = problem.init_variables(seed)
 593 
 594   data_placeholder = tf.placeholder(tf.float32)
 595   labels_placeholder = tf.placeholder(tf.int32)
 596 
 597   obj = problem.objective(params, data_placeholder, labels_placeholder)
 598   gradients = problem.gradients(obj, params)
 599   vars_to_preinitialize = params
 600 
 601   with tf.Session(graph=tf.get_default_graph()) as sess:
 602     # initialize the parameter scope variables; necessary for apply_gradients
 603     sess.run(tf.variables_initializer(vars_to_preinitialize))
 604     train_op = optimizer.apply_gradients(zip(gradients, params))
 605     if isinstance(train_op, tuple) or isinstance(train_op, list):
 606       # LOL apply_gradients returns a tuple. Regular optimizers do not.
 607       train_op = train_op[0]
 608     vars_to_restore = tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES,
 609                                         scope=OPTIMIZER_SCOPE)
 610     vars_to_initialize = list(
 611         set(tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)) -
 612         set(vars_to_restore) - set(vars_to_preinitialize))
 613     # load or initialize optimizer variables
 614     if logdir is not None:
 615       restorer = tf.Saver(var_list=vars_to_restore)
 616       ckpt = tf.train.latest_checkpoint(logdir)
 617       restorer.restore(sess, ckpt)
 618     else:
 619       sess.run(tf.variables_initializer(vars_to_restore))
 620     # initialize all the other variables
 621     sess.run(tf.variables_initializer(vars_to_initialize))
 622 
 623     problem.init_fn(sess)
 624 
 625     # generate the minibatch indices
 626     batch_inds = dataset.batch_indices(num_steps, batch_size)
 627 
 628     avg_iter_time = []
 629     for batch in batch_inds:
 630       # data to feed in
 631       feed = {data_placeholder: dataset.data[batch],
 632               labels_placeholder: dataset.labels[batch]}
 633 
 634       # run the optimization train operation
 635       start = time.time()
 636       sess.run([train_op], feed_dict=feed)
 637       avg_iter_time.append(time.time() - start)
 638 
 639   return np.median(np.array(avg_iter_time))
