  47 def manual_scalar_summary(name, value):
  48   value = tf.Summary.Value(tag=name, simple_value=value)
  49   summary_str = tf.Summary(value=[value])
  50   return summary_str

  52 def eval(sbn, eval_xs, n_samples=100, batch_size=5):
  53   n = eval_xs.shape[0]
  54   i = 0
  55   res = []
  56   while i < n:
  57     batch_xs = eval_xs[i:min(i+batch_size, n)]
  58     res.append(sbn.partial_eval(batch_xs, n_samples))
  59     i += batch_size
  60   res = np.mean(res, axis=0)
  61   return res

  63 def train(sbn, train_xs, valid_xs, test_xs, training_steps, debug=False):
  64   hparams = sorted(sbn.hparams.values().items())
  65   hparams = (map(str, x) for x in hparams)
  66   hparams = ('_'.join(x) for x in hparams)
  67   hparams_str = '.'.join(hparams)
  68 
  69   logger = L.Logger()
  70 
  71   # Create the experiment name from the hparams
  72   experiment_name = ([str(sbn.hparams.n_hidden) for i in xrange(sbn.hparams.n_layer)] +
  73                      [str(sbn.hparams.n_input)])
  74   if sbn.hparams.nonlinear:
  75     experiment_name = '~'.join(experiment_name)
  76   else:
  77     experiment_name = '-'.join(experiment_name)
  78   experiment_name = 'SBN_%s' % experiment_name
  79   rowkey = {'experiment': experiment_name,
  80             'model': hparams_str}
  81 
  82   # Create summary writer
  83   summ_dir = os.path.join(FLAGS.working_dir, hparams_str)
  84   summary_writer = tf.summary.FileWriter(
  85       summ_dir, flush_secs=15, max_queue=100)
  86 
  87   sv = tf.train.Supervisor(logdir=os.path.join(
  88       FLAGS.working_dir, hparams_str),
  89                      save_summaries_secs=0,
  90                      save_model_secs=1200,
  91                      summary_op=None,
  92                      recovery_wait_secs=30,
  93                      global_step=sbn.global_step)
  94   with sv.managed_session() as sess:
  95     # Dump hparams to file
  96     with gfile.Open(os.path.join(FLAGS.working_dir,
  97                                  hparams_str,
  98                                  'hparams.json'),
  99                     'w') as out:
 100       json.dump(sbn.hparams.values(), out)
 101 
 102     sbn.initialize(sess)
 103     batch_size = sbn.hparams.batch_size
 104     scores = []
 105     n = train_xs.shape[0]
 106     index = range(n)
 107 
 108     while not sv.should_stop():
 109       lHats = []
 110       grad_variances = []
 111       temperatures = []
 112       random.shuffle(index)
 113       i = 0
 114       while i < n:
 115         batch_index = index[i:min(i+batch_size, n)]
 116         batch_xs = train_xs[batch_index, :]
 117 
 118         if sbn.hparams.dynamic_b:
 119           # Dynamically binarize the batch data
 120           batch_xs = (np.random.rand(*batch_xs.shape) < batch_xs).astype(float)
 121 
 122         lHat, grad_variance, step, temperature = sbn.partial_fit(batch_xs,
 123                                                     sbn.hparams.n_samples)
 124         if debug:
 125           print(i, lHat)
 126           if i > 100:
 127             return
 128         lHats.append(lHat)
 129         grad_variances.append(grad_variance)
 130         temperatures.append(temperature)
 131         i += batch_size
 132 
 133       grad_variances = np.log(np.mean(grad_variances, axis=0)).tolist()
 134       summary_strings = []
 135       if isinstance(grad_variances, list):
 136         grad_variances = dict(zip([k for (k, v) in sbn.losses], map(float, grad_variances)))
 137         rowkey['step'] = step
 138         logger.log(rowkey, {'step': step,
 139                              'train': np.mean(lHats, axis=0)[0],
 140                              'grad_variances': grad_variances,
 141                              'temperature': np.mean(temperatures), })
 142         grad_variances = '\n'.join(map(str, sorted(grad_variances.iteritems())))
 143       else:
 144         rowkey['step'] = step
 145         logger.log(rowkey, {'step': step,
 146                              'train': np.mean(lHats, axis=0)[0],
 147                              'grad_variance': grad_variances,
 148                              'temperature': np.mean(temperatures), })
 149         summary_strings.append(manual_scalar_summary("log grad variance", grad_variances))
 150 
 151       print('Step %d: %s\n%s' % (step, str(np.mean(lHats, axis=0)), str(grad_variances)))
 152 
 153       # Every few epochs compute test and validation scores
 154       epoch = int(step / (train_xs.shape[0] / sbn.hparams.batch_size))
 155       if epoch % FLAGS.eval_freq == 0:
 156         valid_res = eval(sbn, valid_xs)
 157         test_res= eval(sbn, test_xs)
 158 
 159         print('\nValid %d: %s' % (step, str(valid_res)))
 160         print('Test %d: %s\n' % (step, str(test_res)))
 161         logger.log(rowkey, {'step': step,
 162                              'valid': valid_res[0],
 163                              'test': test_res[0]})
 164         logger.flush()  # Flush infrequently
 165 
 166       # Create summaries
 167       summary_strings.extend([
 168         manual_scalar_summary("Train ELBO", np.mean(lHats, axis=0)[0]),
 169         manual_scalar_summary("Temperature", np.mean(temperatures)),
 170       ])
 171       for summ_str in summary_strings:
 172         summary_writer.add_summary(summ_str, global_step=step)
 173       summary_writer.flush()
 174 
 175       sys.stdout.flush()
 176       scores.append(np.mean(lHats, axis=0))
 177 
 178       if step > training_steps:
 179         break
 180 
 181     return scores

 184 def main():
 185   # Parse hyperparams
 186   hparams = rebar.default_hparams
 187   hparams.parse(FLAGS.hparams)
 188   print(hparams.values())
 189 
 190   train_xs, valid_xs, test_xs = datasets.load_data(hparams)
 191   mean_xs = np.mean(train_xs, axis=0)  # Compute mean centering on training
 192 
 193   training_steps = 2000000
 194   model = getattr(rebar, hparams.model)
 195   sbn = model(hparams, mean_xs=mean_xs)
 196 
 197   scores = train(sbn, train_xs, valid_xs, test_xs,
 198                  training_steps=training_steps, debug=False)
 199 
 200 if __name__ == '__main__':
 201   main()
