  38 def load_mnist_float(local_filename):
  39   with open(local_filename, 'rb') as f:
  40     f.seek(4)
  41     nimages, rows, cols = struct.unpack('>iii', f.read(12))
  42     dim = rows*cols
  43 
  44     images = np.fromfile(f, dtype=np.dtype(np.ubyte))
  45     images = (images/255.0).astype('float32').reshape((nimages, dim))
  46 
  47   return images

  49 if __name__ == '__main__':
  50   if not os.path.exists(config.DATA_DIR):
  51     os.makedirs(config.DATA_DIR)

  53   # Get MNIST and convert to npy file
  54   local_filename = os.path.join(config.DATA_DIR, MNIST_FLOAT_TRAIN)
  55   if not os.path.exists(local_filename):
  56     urllib.urlretrieve("%s/%s.gz" % (MNIST_URL, MNIST_FLOAT_TRAIN), local_filename+'.gz')
  57     with gzip.open(local_filename+'.gz', 'rb') as f:
  58       file_content = f.read()
  59     with open(local_filename, 'wb') as f:
  60       f.write(file_content)
  61     os.remove(local_filename+'.gz')
  62 
  63   mnist_float_train = load_mnist_float(local_filename)[:-10000]
  64   # save in a nice format
  65   np.save(os.path.join(config.DATA_DIR, config.MNIST_FLOAT), mnist_float_train)
  66 
  67   # Get binarized MNIST
  68   splits = ['train', 'valid', 'test']
  69   mnist_binarized = []
  70   for split in splits:
  71     filename = 'binarized_mnist_%s.amat' % split
  72     url = '%s/binarized_mnist_%s.amat' % (MNIST_BINARIZED_URL, split)
  73     local_filename = os.path.join(config.DATA_DIR, filename)
  74     if not os.path.exists(local_filename):
  75       urllib.urlretrieve(url, local_filename)
  76 
  77     with open(local_filename, 'rb') as f:
  78       mnist_binarized.append((np.array([map(int, line.split()) for line in f.readlines()]).astype('float32'), None))
  79 
  80   # save in a nice format
  81   with open(os.path.join(config.DATA_DIR, config.MNIST_BINARIZED), 'w') as out:
  82     pickle.dump(mnist_binarized, out)
  83 
  84   # Get Omniglot
  85   local_filename = os.path.join(config.DATA_DIR, config.OMNIGLOT)
  86   if not os.path.exists(local_filename):
  87     urllib.urlretrieve(OMNIGLOT_URL,
