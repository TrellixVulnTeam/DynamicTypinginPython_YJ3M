  94   def _create_baseline(self, n_output=1, n_hidden=100,
  95                        is_zero_init=False,
  96                        collection='BASELINE'):
  97     # center input
  98     h = self._x
  99     if self.mean_xs is not None:
 100       h -= self.mean_xs
 101 
 102     if is_zero_init:
 103       initializer = init_ops.zeros_initializer()
 104     else:
 105       initializer = slim.variance_scaling_initializer()
 106 
 107     with slim.arg_scope([slim.fully_connected],
 108                         variables_collections=[collection, Q_COLLECTION],
 109                         trainable=False,
 110                         weights_initializer=initializer):
 111       h = slim.fully_connected(h, n_hidden, activation_fn=tf.nn.tanh)
 112       baseline = slim.fully_connected(h, n_output, activation_fn=None)
 113 
 114       if n_output == 1:
 115         baseline = tf.reshape(baseline, [-1])  # very important to reshape
 116     return baseline

 119   def _create_transformation(self, input, n_output, reuse, scope_prefix):
 120     """Create the deterministic transformation between stochastic layers.
 121 
 122     If self.hparam.nonlinear:
 123         2 x tanh layers
 124     Else:
 125         1 x linear layer
 126     """
 127     if self.hparams.nonlinear:
 128       h = slim.fully_connected(input,
 129                                self.hparams.n_hidden,
 130                                reuse=reuse,
 131                                activation_fn=tf.nn.tanh,
 132                                scope='%s_nonlinear_1' % scope_prefix)
 133       h = slim.fully_connected(h,
 134                                self.hparams.n_hidden,
 135                                reuse=reuse,
 136                                activation_fn=tf.nn.tanh,
 137                                scope='%s_nonlinear_2' % scope_prefix)
 138       h = slim.fully_connected(h,
 139                                n_output,
 140                                reuse=reuse,
 141                                activation_fn=None,
 142                                scope='%s' % scope_prefix)
 143     else:
 144       h = slim.fully_connected(input,
 145                                n_output,
 146                                reuse=reuse,
 147                                activation_fn=None,
 148                                scope='%s' % scope_prefix)
 149     return h

 151   def _recognition_network(self, sampler=None, log_likelihood_func=None):
 152     """x values -> samples from Q and return log Q(h|x)."""
 153     samples = {}
 154     reuse = None if not self.run_recognition_network else True
 155 
 156     # Set defaults
 157     if sampler is None:
 158       sampler = self._random_sample
 159 
 160     if log_likelihood_func is None:
 161       log_likelihood_func = lambda sample, log_params: (
 162         U.binary_log_likelihood(sample['activation'], log_params))
 163 
 164     logQ = []
 165 
 166 
 167     if self.hparams.task in ['sbn', 'omni']:
 168       # Initialize the edge case
 169       samples[-1] = {'activation': self._x}
 170       if self.mean_xs is not None:
 171         samples[-1]['activation'] -= self.mean_xs  # center the input
 172       samples[-1]['activation'] = (samples[-1]['activation'] + 1)/2.0
 173 
 174       with slim.arg_scope([slim.fully_connected],
 175                           weights_initializer=slim.variance_scaling_initializer(),
 176                           variables_collections=[Q_COLLECTION]):
 177         for i in xrange(self.hparams.n_layer):
 178           # Set up the input to the layer
 179           input = 2.0*samples[i-1]['activation'] - 1.0
 180 
 181           # Create the conditional distribution (output is the logits)
 182           h = self._create_transformation(input,
 183                                           n_output=self.hparams.n_hidden,
 184                                           reuse=reuse,
 185                                           scope_prefix='q_%d' % i)
 186 
 187           samples[i] = sampler(h, self.uniform_samples[i], i)
 188           logQ.append(log_likelihood_func(samples[i], h))
 189 
 190       self.run_recognition_network = True
 191       return logQ, samples
 192     elif self.hparams.task == 'sp':
 193       # Initialize the edge case
 194       samples[-1] = {'activation': tf.split(self._x,
 195                                             num_or_size_splits=2,
 196                                             axis=1)[0]}  # top half of digit
 197       if self.mean_xs is not None:
 198         samples[-1]['activation'] -= np.split(self.mean_xs, 2, 0)[0]  # center the input
 199       samples[-1]['activation'] = (samples[-1]['activation'] + 1)/2.0
 200 
 201       with slim.arg_scope([slim.fully_connected],
 202                           weights_initializer=slim.variance_scaling_initializer(),
 203                           variables_collections=[Q_COLLECTION]):
 204         for i in xrange(self.hparams.n_layer):
 205           # Set up the input to the layer
 206           input = 2.0*samples[i-1]['activation'] - 1.0
 207 
 208           # Create the conditional distribution (output is the logits)
 209           h = self._create_transformation(input,
 210                                           n_output=self.hparams.n_hidden,
 211                                           reuse=reuse,
 212                                           scope_prefix='q_%d' % i)
 213 
 214           samples[i] = sampler(h, self.uniform_samples[i], i)
 215           logQ.append(log_likelihood_func(samples[i], h))
 216 
 217       self.run_recognition_network = True
 218       return logQ, samples

 220   def _generator_network(self, samples, logQ, log_likelihood_func=None):
 221     '''Returns learning signal and function.
 222 
 223     This is the implementation for SBNs for the ELBO.
 224 
 225     Args:
 226       samples: dictionary of sampled latent variables
 227       logQ: list of log q(h_i) terms
 228       log_likelihood_func: function used to compute log probs for the latent
 229         variables
 230 
 231     Returns:
 232       learning_signal: the "reward" function
 233       function_term: part of the function that depends on the parameters
 234         and needs to have the gradient taken through
 235     '''
 236     reuse=None if not self.run_generator_network else True
 237 
 238     if self.hparams.task in ['sbn', 'omni']:
 239       if log_likelihood_func is None:
 240         log_likelihood_func = lambda sample, log_params: (
 241           U.binary_log_likelihood(sample['activation'], log_params))
 242 
 243       logPPrior = log_likelihood_func(
 244           samples[self.hparams.n_layer-1],
 245           tf.expand_dims(self.prior, 0))
 246 
 247       with slim.arg_scope([slim.fully_connected],
 248                           weights_initializer=slim.variance_scaling_initializer(),
 249                           variables_collections=[P_COLLECTION]):
 250 
 251         for i in reversed(xrange(self.hparams.n_layer)):
 252           if i == 0:
 253             n_output = self.hparams.n_input
 254           else:
 255             n_output = self.hparams.n_hidden
 256           input = 2.0*samples[i]['activation']-1.0
 257 
 258           h = self._create_transformation(input,
 259                                           n_output,
 260                                           reuse=reuse,
 261                                           scope_prefix='p_%d' % i)
 262 
 263           if i == 0:
 264             # Assume output is binary
 265             logP = U.binary_log_likelihood(self._x, h + self.train_bias)
 266           else:
 267             logPPrior += log_likelihood_func(samples[i-1], h)
 268 
 269       self.run_generator_network = True
 270       return logP + logPPrior - tf.add_n(logQ), logP + logPPrior
 271     elif self.hparams.task == 'sp':
 272       with slim.arg_scope([slim.fully_connected],
 273                           weights_initializer=slim.variance_scaling_initializer(),
 274                           variables_collections=[P_COLLECTION]):
 275         n_output = int(self.hparams.n_input/2)
 276         i = self.hparams.n_layer - 1  # use the last layer
 277         input = 2.0*samples[i]['activation']-1.0
 278 
 279         h = self._create_transformation(input,
 280                                         n_output,
 281                                         reuse=reuse,
 282                                         scope_prefix='p_%d' % i)
 283 
 284         # Predict on the lower half of the image
 285         logP = U.binary_log_likelihood(tf.split(self._x,
 286                                               num_or_size_splits=2,
 287                                               axis=1)[1],
 288                                      h + np.split(self.train_bias, 2, 0)[1])
 289 
 290       self.run_generator_network = True
 291       return logP, logP

 294   def _create_loss(self):
 295     # Hard loss
 296     logQHard, samples = self._recognition_network()
 297     reinforce_learning_signal, reinforce_model_grad = self._generator_network(samples, logQHard)
 298     logQHard = tf.add_n(logQHard)
 299 
 300     # REINFORCE
 301     learning_signal = tf.stop_gradient(U.center(reinforce_learning_signal))
 302     self.optimizerLoss = -(learning_signal*logQHard +
 303                            reinforce_model_grad)
 304     self.lHat = map(tf.reduce_mean, [
 305         reinforce_learning_signal,
 306         U.rms(learning_signal),
 307     ])
 308 
 309     return reinforce_learning_signal

 316   def compute_tensor_variance(self, t):
 317     """Compute the mean per component variance.
 318 
 319     Use a moving average to estimate the required moments.
 320     """
 321     t_sq = tf.reduce_mean(tf.square(t))
 322     self.maintain_ema_ops.append(self.ema.apply([t, t_sq]))
 323 
 324     # mean per component variance
 325     variance_estimator = (self.ema.average(t_sq) -
 326                           tf.reduce_mean(
 327                               tf.square(self.ema.average(t))))
 328 
 329     return variance_estimator

 331   def _create_train_op(self, grads_and_vars, extra_grads_and_vars=[]):
 332     '''
 333     Args:
 334       grads_and_vars: gradients to apply and compute running average variance
 335       extra_grads_and_vars: gradients to apply (not used to compute average variance)
 336     '''
 337     # Variance summaries
 338     first_moment = U.vectorize(grads_and_vars, skip_none=True)
 339     second_moment = tf.square(first_moment)
 340     self.maintain_ema_ops.append(self.ema.apply([first_moment, second_moment]))
 341 
 342     # Add baseline losses
 343     if len(self.baseline_loss) > 0:
 344       mean_baseline_loss = tf.reduce_mean(tf.add_n(self.baseline_loss))
 345       extra_grads_and_vars += self.optimizer_class.compute_gradients(
 346           mean_baseline_loss,
 347           var_list=tf.get_collection('BASELINE'))
 348 
 349     # Ensure that all required tensors are computed before updates are executed
 350     extra_optimizer = tf.train.AdamOptimizer(
 351         learning_rate=10*self.hparams.learning_rate,
 352         beta2=self.hparams.beta2)
 353     with tf.control_dependencies(
 354         [tf.group(*[g for g, _ in (grads_and_vars + extra_grads_and_vars) if g is not None])]):
 355 
 356       # Filter out the P_COLLECTION variables if we're in eval mode
 357       if self.eval_mode:
 358         grads_and_vars = [(g, v) for g, v in grads_and_vars
 359                           if v not in tf.get_collection(P_COLLECTION)]
 360 
 361       train_op = self.optimizer_class.apply_gradients(grads_and_vars,
 362                                                       global_step=self.global_step)
 363 
 364       if len(extra_grads_and_vars) > 0:
 365         extra_train_op = extra_optimizer.apply_gradients(extra_grads_and_vars)
 366       else:
 367         extra_train_op = tf.no_op()
 368 
 369       self.optimizer = tf.group(train_op, extra_train_op, *self.maintain_ema_ops)
 370 
 371     # per parameter variance
 372     variance_estimator = (self.ema.average(second_moment) -
 373         tf.square(self.ema.average(first_moment)))
 374     self.grad_variance = tf.reduce_mean(variance_estimator)
 375 
 376   def _create_network(self):
 377     logF = self._create_loss()
 378     self.optimizerLoss = tf.reduce_mean(self.optimizerLoss)
 379 
 380     # Setup optimizer
 381     grads_and_vars = self.optimizer_class.compute_gradients(self.optimizerLoss)
 382     self._create_train_op(grads_and_vars)
 383 
 384     # Create IWAE lower bound for evaluation
 385     self.logF = self._reshape(logF)
 386     self.iwae = tf.reduce_mean(U.logSumExp(self.logF, axis=1) -
 387                                tf.log(tf.to_float(self.n_samples)))
 388 
 389   def partial_fit(self, X, n_samples=1):
 390     if hasattr(self, 'grad_variances'):
 391       grad_variance_field_to_return = self.grad_variances
 392     else:
 393       grad_variance_field_to_return = self.grad_variance
 394     _, res, grad_variance, step, temperature = self.sess.run(
 395         (self.optimizer, self.lHat, grad_variance_field_to_return, self.global_step, self.temperature_variable),
 396         feed_dict={self.x: X, self.n_samples: n_samples})
 397     return res, grad_variance, step, temperature

 399   def partial_grad(self, X, n_samples=1):
 400     control_variate_grads, step = self.sess.run(
 401         (self.control_variate_grads, self.global_step),
 402         feed_dict={self.x: X, self.n_samples: n_samples})

 405   def partial_eval(self, X, n_samples=5):
 406     if n_samples < 1000:
 407       res, iwae = self.sess.run(
 408           (self.lHat, self.iwae),
 409           feed_dict={self.x: X, self.n_samples: n_samples})
 410       res = [iwae] + res
 411     else:  # special case to handle OOM
 412       assert n_samples % 100 == 0, "When using large # of samples, it must be divisble by 100"
 413       res = []
 414       for i in xrange(int(n_samples/100)):
 415         logF, = self.sess.run(
 416             (self.logF,),
 417             feed_dict={self.x: X, self.n_samples: 100})
 418         res.append(logsumexp(logF, axis=1))
 419       res = [np.mean(logsumexp(res, axis=0) - np.log(n_samples))]
 420     return res

 433   def _generate_randomness(self):
 434     for i in xrange(self.hparams.n_layer):
 435       self.uniform_samples[i] = tf.stop_gradient(tf.random_uniform(
 436           [self.batch_size, self.hparams.n_hidden]))

 438   def _u_to_v(self, log_alpha, u, eps = 1e-8):
 439     """Convert u to tied randomness in v."""
 440     u_prime = tf.nn.sigmoid(-log_alpha)  # g(u') = 0
 441 
 442     v_1 = (u - u_prime) / tf.clip_by_value(1 - u_prime, eps, 1)
 443     v_1 = tf.clip_by_value(v_1, 0, 1)
 444     v_1 = tf.stop_gradient(v_1)
 445     v_1 = v_1*(1 - u_prime) + u_prime
 446     v_0 = u / tf.clip_by_value(u_prime, eps, 1)
 447     v_0 = tf.clip_by_value(v_0, 0, 1)
 448     v_0 = tf.stop_gradient(v_0)
 449     v_0 = v_0 * u_prime
 450 
 451     v = tf.where(u > u_prime, v_1, v_0)
 452     v = tf.check_numerics(v, 'v sampling is not numerically stable.')
 453     v = v + tf.stop_gradient(-v + u)  # v and u are the same up to numerical errors
 454 
 455     return v

 457   def _random_sample(self, log_alpha, u, layer):
 458     """Returns sampled random variables parameterized by log_alpha."""
 459     # Generate tied randomness for later
 460     if layer not in self.uniform_samples_v:
 461       self.uniform_samples_v[layer] = self._u_to_v(log_alpha, u)
 462 
 463     # Sample random variable underlying softmax/argmax
 464     x = log_alpha + U.safe_log_prob(u) - U.safe_log_prob(1 - u)
 465     samples = tf.stop_gradient(tf.to_float(x > 0))
 466 
 467     return {
 468         'preactivation': x,
 469         'activation': samples,
 470         'log_param': log_alpha,
 471     }

 473   def _random_sample_soft(self, log_alpha, u, layer, temperature=None):
 474     """Returns sampled random variables parameterized by log_alpha."""
 475     if temperature is None:
 476       temperature = self.hparams.temperature
 477 
 478     # Sample random variable underlying softmax/argmax
 479     x = log_alpha + U.safe_log_prob(u) - U.safe_log_prob(1 - u)
 480     x /= tf.expand_dims(temperature, -1)
 481 
 482     if self.hparams.muprop_relaxation:
 483       y = tf.nn.sigmoid(x + log_alpha * tf.expand_dims(temperature/(temperature + 1), -1))
 484     else:
 485       y = tf.nn.sigmoid(x)
 486 
 487     return {
 488         'preactivation': x,
 489         'activation': y,
 490         'log_param': log_alpha
 491     }

 493   def _random_sample_soft_v(self, log_alpha, _, layer, temperature=None):
 494     """Returns sampled random variables parameterized by log_alpha."""
 495     v = self.uniform_samples_v[layer]
 496 
 497     return self._random_sample_soft(log_alpha, v, layer, temperature)

 499   def get_gumbel_gradient(self):
 500     logQ, softSamples = self._recognition_network(sampler=self._random_sample_soft)
 501     logQ = tf.add_n(logQ)
 502     logPPrior, logP = self._generator_network(softSamples)
 503 
 504     softELBO = logPPrior + logP - logQ
 505     gumbel_gradient = (self.optimizer_class.
 506                        compute_gradients(softELBO))
 507     debug = {
 508         'softELBO': softELBO,
 509     }
 510 
 511     return gumbel_gradient, debug

 513   # samplers used for quadratic version
 514   def _random_sample_switch(self, log_alpha, u, layer, switch_layer, temperature=None):
 515     """Run partial discrete, then continuous path.
 516 
 517        Args:
 518         switch_layer: this layer and beyond will be continuous
 519     """
 520     if layer < switch_layer:
 521       return self._random_sample(log_alpha, u, layer)
 522     else:
 523       return self._random_sample_soft(log_alpha, u, layer, temperature)

 525   def _random_sample_switch_v(self, log_alpha, u, layer, switch_layer, temperature=None):
 526     """Run partial discrete, then continuous path.
 527 
 528        Args:
 529         switch_layer: this layer and beyond will be continuous
 530     """
 531     if layer < switch_layer:
 532       return self._random_sample(log_alpha, u, layer)
 533     else:
 534       return self._random_sample_soft_v(log_alpha, u, layer, temperature)

 540   def get_nvil_gradient(self):
 541     """Compute the NVIL gradient."""
 542     # Hard loss
 543     logQHard, samples = self._recognition_network()
 544     ELBO, reinforce_model_grad = self._generator_network(samples, logQHard)
 545     logQHard = tf.add_n(logQHard)
 546 
 547     # Add baselines (no variance normalization)
 548     learning_signal = tf.stop_gradient(ELBO) - self._create_baseline()
 549 
 550     # Set up losses
 551     self.baseline_loss.append(tf.square(learning_signal))
 552     optimizerLoss = -(tf.stop_gradient(learning_signal)*logQHard +
 553                            reinforce_model_grad)
 554     optimizerLoss = tf.reduce_mean(optimizerLoss)
 555 
 556     nvil_gradient = self.optimizer_class.compute_gradients(optimizerLoss)
 557     debug = {
 558         'ELBO': ELBO,
 559         'RMS of centered learning signal': U.rms(learning_signal),
 560     }
 561

 565   def get_simple_muprop_gradient(self):
 566     """ Computes the simple muprop gradient.
 567 
 568     This muprop control variate does not include the linear term.
 569     """
 570     # Hard loss
 571     logQHard, hardSamples = self._recognition_network()
 572     hardELBO, reinforce_model_grad = self._generator_network(hardSamples, logQHard)
 573 
 574     # Soft loss
 575     logQ, muSamples = self._recognition_network(sampler=self._mean_sample)
 576     muELBO, _  = self._generator_network(muSamples, logQ)
 577 
 578     scaling_baseline = self._create_eta(collection='BASELINE')
 579     learning_signal = (hardELBO
 580                        - scaling_baseline * muELBO
 581                        - self._create_baseline())
 582     self.baseline_loss.append(tf.square(learning_signal))
 583 
 584     optimizerLoss = -(tf.stop_gradient(learning_signal) * tf.add_n(logQHard)
 585                       + reinforce_model_grad)
 586     optimizerLoss = tf.reduce_mean(optimizerLoss)
 587 
 588     simple_muprop_gradient = (self.optimizer_class.
 589                               compute_gradients(optimizerLoss))
 590     debug = {
 591         'ELBO': hardELBO,
 592         'muELBO': muELBO,
 593         'RMS': U.rms(learning_signal),
 594     }
 595 
 596     return simple_muprop_gradient, debug

 598   def get_muprop_gradient(self):
 599     """
 600     random sample function that actually returns mean
 601     new forward pass that returns logQ as a list
 602 
 603     can get x_i from samples
 604     """
 605 
 606     # Hard loss
 607     logQHard, hardSamples = self._recognition_network()
 608     hardELBO, reinforce_model_grad = self._generator_network(hardSamples, logQHard)
 609 
 610     # Soft loss
 611     logQ, muSamples = self._recognition_network(sampler=self._mean_sample)
 612     muELBO, _ = self._generator_network(muSamples, logQ)
 613 
 614     # Compute gradients
 615     muELBOGrads = tf.gradients(tf.reduce_sum(muELBO),
 616                                [ muSamples[i]['activation'] for
 617                                 i in xrange(self.hparams.n_layer) ])
 618 
 619     # Compute MuProp gradient estimates
 620     learning_signal = hardELBO
 621     optimizerLoss = 0.0
 622     learning_signals = []
 623     for i in xrange(self.hparams.n_layer):
 624       dfDiff = tf.reduce_sum(
 625           muELBOGrads[i] * (hardSamples[i]['activation'] -
 626                             muSamples[i]['activation']),
 627           axis=1)
 628       dfMu = tf.reduce_sum(
 629           tf.stop_gradient(muELBOGrads[i]) *
 630           tf.nn.sigmoid(hardSamples[i]['log_param']),
 631           axis=1)
 632 
 633       scaling_baseline_0 = self._create_eta(collection='BASELINE')
 634       scaling_baseline_1 = self._create_eta(collection='BASELINE')
 635       learning_signals.append(learning_signal - scaling_baseline_0 * muELBO - scaling_baseline_1 * dfDiff - self._create_baseline())
 636       self.baseline_loss.append(tf.square(learning_signals[i]))
 637 
 638       optimizerLoss += (
 639           logQHard[i] * tf.stop_gradient(learning_signals[i]) +
 640           tf.stop_gradient(scaling_baseline_1) * dfMu)
 641     optimizerLoss += reinforce_model_grad
 642     optimizerLoss *= -1
 643 
 644     optimizerLoss = tf.reduce_mean(optimizerLoss)
 645 
 646     muprop_gradient = self.optimizer_class.compute_gradients(optimizerLoss)
 647     debug = {
 648         'ELBO': hardELBO,
 649         'muELBO': muELBO,
 650     }
 651 
 652     debug.update(dict([
 653         ('RMS learning signal layer %d' % i, U.rms(learning_signal))
 654         for (i, learning_signal) in enumerate(learning_signals)]))
 655 
 656     return muprop_gradient, debug

 659   def _create_gumbel_control_variate(self, logQHard, temperature=None):
 660     '''Calculate gumbel control variate.
 661     '''
 662     if temperature is None:
 663       temperature = self.hparams.temperature
 664 
 665     logQ, softSamples = self._recognition_network(sampler=functools.partial(
 666         self._random_sample_soft, temperature=temperature))
 667     softELBO, _ = self._generator_network(softSamples, logQ)
 668     logQ = tf.add_n(logQ)
 669 
 670     # Generate the softELBO_v (should be the same value but different grads)
 671     logQ_v, softSamples_v = self._recognition_network(sampler=functools.partial(
 672         self._random_sample_soft_v, temperature=temperature))
 673     softELBO_v, _ = self._generator_network(softSamples_v, logQ_v)
 674     logQ_v = tf.add_n(logQ_v)
 675 
 676     # Compute losses
 677     learning_signal = tf.stop_gradient(softELBO_v)
 678 
 679     # Control variate
 680     h = (tf.stop_gradient(learning_signal) * tf.add_n(logQHard)
 681           - softELBO + softELBO_v)
 682 
 683     extra = (softELBO_v, -softELBO + softELBO_v)
 684 
 685     return h, extra

 687   def _create_gumbel_control_variate_quadratic(self, logQHard, temperature=None):
 688     '''Calculate gumbel control variate.
 689     '''
 690     if temperature is None:
 691       temperature = self.hparams.temperature
 692 
 693     h = 0
 694     extra = []
 695     for layer in xrange(self.hparams.n_layer):
 696       logQ, softSamples = self._recognition_network(sampler=functools.partial(
 697           self._random_sample_switch, switch_layer=layer, temperature=temperature))
 698       softELBO, _ = self._generator_network(softSamples, logQ)
 699 
 700       # Generate the softELBO_v (should be the same value but different grads)
 701       logQ_v, softSamples_v = self._recognition_network(sampler=functools.partial(
 702           self._random_sample_switch_v, switch_layer=layer, temperature=temperature))
 703       softELBO_v, _ = self._generator_network(softSamples_v, logQ_v)
 704 
 705       # Compute losses
 706       learning_signal = tf.stop_gradient(softELBO_v)
 707 
 708       # Control variate
 709       h += (tf.stop_gradient(learning_signal) * logQHard[layer]
 710             - softELBO + softELBO_v)
 711 
 712       extra.append((softELBO_v, -softELBO + softELBO_v))
 713 
 714     return h, extra

 716   def _create_hard_elbo(self):
 717     logQHard, hardSamples = self._recognition_network()
 718     hardELBO, reinforce_model_grad = self._generator_network(hardSamples, logQHard)
 719     reinforce_learning_signal = tf.stop_gradient(hardELBO)
 720 
 721     # Center learning signal
 722     baseline = self._create_baseline(collection='CV')
 723     reinforce_learning_signal = tf.stop_gradient(reinforce_learning_signal) - baseline
 724 
 725     nvil_gradient = (tf.stop_gradient(hardELBO) - baseline) * tf.add_n(logQHard) + reinforce_model_grad
 726 
 727     return hardELBO, nvil_gradient, logQHard

 729   def multiply_by_eta(self, h_grads, eta):
 730     # Modifies eta
 731     res = []

 733     for (g, v) in h_grads:

 736       else:
 737         if 'network' not in eta:
 738           eta['network'] = self._create_eta()
 739         res.append((g*eta['network'], v))
 740     eta_statistics.append(eta['network'])

 759   def multiply_by_eta_per_unit(self, h_grads, eta):
 760     # Modifies eta
 761     res = []
 762     eta_statistics = []
 763     for (g, v) in h_grads:
 764       if g is None:
 765         res.append((g, v))
 766       else:
 767         if v not in eta:
 768           g_shape = g.shape_as_list()
 769           assert len(g_shape) <= 2, 'Gradient has too many dimensions'
 770           if len(g_shape) == 1:
 771             eta[v] = self._create_eta(g_shape)
 772           else:
 773             eta[v] = self._create_eta([1, g_shape[1]])
 774         h_grads.append((g*eta[v], v))
 775         eta_statistics.extend(tf.nn.moments(tf.squeeze(eta[v]), axes=[0]))
 776     return res, eta_statistics

 778   def get_dynamic_rebar_gradient(self):
 779     """Get the dynamic rebar gradient (t, eta optimized)."""
 780     tiled_pre_temperature = tf.tile([self.pre_temperature_variable],
 781                                 [self.batch_size])
 782     temperature = tf.exp(tiled_pre_temperature)
 783 
 784     hardELBO, nvil_gradient, logQHard = self._create_hard_elbo()
 785     if self.hparams.quadratic:
 786       gumbel_cv, extra  = self._create_gumbel_control_variate_quadratic(logQHard, temperature=temperature)
 787     else:
 788       gumbel_cv, extra  = self._create_gumbel_control_variate(logQHard, temperature=temperature)
 789 
 790     f_grads = self.optimizer_class.compute_gradients(tf.reduce_mean(-nvil_gradient))
 791 
 792     eta = {}
 793     h_grads, eta_statistics = self.multiply_by_eta_per_layer(
 794         self.optimizer_class.compute_gradients(tf.reduce_mean(gumbel_cv)),
 795         eta)
 796 
 797     model_grads = U.add_grads_and_vars(f_grads, h_grads)
 798     total_grads = model_grads
 799 
 800     # Construct the variance objective
 801     g = U.vectorize(model_grads, set_none_to_zero=True)
 802     self.maintain_ema_ops.append(self.ema.apply([g]))
 803     gbar = 0  #tf.stop_gradient(self.ema.average(g))
 804     variance_objective = tf.reduce_mean(tf.square(g - gbar))
 805 
 806     reinf_g_t = 0
 807     if self.hparams.quadratic:
 808       for layer in xrange(self.hparams.n_layer):
 809         gumbel_learning_signal, _ = extra[layer]
 810         df_dt = tf.gradients(gumbel_learning_signal, tiled_pre_temperature)[0]
 811         reinf_g_t_i, _ = self.multiply_by_eta_per_layer(
 812             self.optimizer_class.compute_gradients(tf.reduce_mean(tf.stop_gradient(df_dt) * logQHard[layer])),
 813             eta)
 814         reinf_g_t += U.vectorize(reinf_g_t_i, set_none_to_zero=True)
 815 
 816       reparam = tf.add_n([reparam_i for _, reparam_i in extra])
 817     else:
 818       gumbel_learning_signal, reparam = extra
 819       df_dt = tf.gradients(gumbel_learning_signal, tiled_pre_temperature)[0]
 820       reinf_g_t, _ = self.multiply_by_eta_per_layer(
 821           self.optimizer_class.compute_gradients(tf.reduce_mean(tf.stop_gradient(df_dt) * tf.add_n(logQHard))),
 822           eta)
 823       reinf_g_t = U.vectorize(reinf_g_t, set_none_to_zero=True)
 824 
 825     reparam_g, _ = self.multiply_by_eta_per_layer(
 826         self.optimizer_class.compute_gradients(tf.reduce_mean(reparam)),
 827         eta)
 828     reparam_g = U.vectorize(reparam_g, set_none_to_zero=True)
 829     reparam_g_t = tf.gradients(tf.reduce_mean(2*tf.stop_gradient(g - gbar)*reparam_g), self.pre_temperature_variable)[0]
 830 
 831     variance_objective_grad = tf.reduce_mean(2*(g - gbar)*reinf_g_t) + reparam_g_t
 832 
 833     debug = { 'ELBO': hardELBO,
 834              'etas': eta_statistics,
 835              'variance_objective': variance_objective,
 836              }
 837     return total_grads, debug, variance_objective, variance_objective_grad

1052     """Returns sampled random variables parameterized by log_alpha."""
1053     if temperature is None:
1054       temperature = self.hparams.temperature
1055

1059

1061       x += temperature/(temperature + 1)*log_alpha
1062 
1063     y = tf.nn.sigmoid(x)
1064

1066         'preactivation': x,
1067         'activation': y,
1068         'log_param': log_alpha
1069     }
