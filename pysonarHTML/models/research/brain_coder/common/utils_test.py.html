 102   def testRecord(self):
 103     my_record = utils.make_record('my_record', ['a', 'b', 'c'], {'b': 55})
 104     inst = my_record(a=1, b=2, c=3)
 105     self.assertEqual(1, inst.a)
 106     self.assertEqual(2, inst.b)
 107     self.assertEqual(3, inst.c)
 108     self.assertEqual(1, inst[0])
 109     self.assertEqual(2, inst[1])
 110     self.assertEqual(3, inst[2])
 111     self.assertEqual([1, 2, 3], list(iter(inst)))
 112     self.assertEqual(3, len(inst))
 113 
 114     inst.b = 999
 115     self.assertEqual(999, inst.b)
 116     self.assertEqual(999, inst[1])
 117 
 118     inst2 = my_record(1, 999, 3)
 119     self.assertTrue(inst == inst2)
 120     inst2[1] = 3
 121     self.assertFalse(inst == inst2)
 122 
 123     inst3 = my_record(a=1, c=3)
 124     inst.b = 55
 125     self.assertEqual(inst, inst3)

 213   def testRouletteWheel(self):
 214     random.seed(12345678987654321)
 215     r = utils.RouletteWheel()
 216     self.assertTrue(r.is_empty())
 217     with self.assertRaises(RuntimeError):
 218       r.sample()  # Cannot sample when empty.
 219     self.assertEqual(0, r.total_weight)
 220     self.assertEqual(True, r.add('a', 0.1))
 221     self.assertFalse(r.is_empty())
 222     self.assertEqual(0.1, r.total_weight)
 223     self.assertEqual(True, r.add('b', 0.01))
 224     self.assertEqual(0.11, r.total_weight)
 225     self.assertEqual(True, r.add('c', 0.5))
 226     self.assertEqual(True, r.add('d', 0.1))
 227     self.assertEqual(True, r.add('e', 0.05))
 228     self.assertEqual(True, r.add('f', 0.03))
 229     self.assertEqual(True, r.add('g', 0.001))
 230     self.assertEqual(0.791, r.total_weight)
 231     self.assertFalse(r.is_empty())
 232 
 233     # Check that sampling is correct.
 234     obj, weight = r.sample()
 235     self.assertTrue(isinstance(weight, float), 'Type: %s' % type(weight))
 236     self.assertTrue((obj, weight) in r)
 237     for obj, weight in r.sample_many(100):
 238       self.assertTrue(isinstance(weight, float), 'Type: %s' % type(weight))
 239       self.assertTrue((obj, weight) in r)
 240 
 241     # Check that sampling distribution is correct.
 242     n = 1000000
 243     c = Counter(r.sample_many(n))
 244     for obj, w in r:
 245       estimated_w = c[(obj, w)] / float(n) * r.total_weight
 246       self.assertTrue(
 247           np.isclose(w, estimated_w, atol=1e-3),
 248           'Expected %s, got %s, for object %s' % (w, estimated_w, obj))

 296   def testRouletteWheel_UniqueMode(self):
 297     random.seed(12345678987654321)
 298     r = utils.RouletteWheel(unique_mode=True)
 299     self.assertEqual(True, r.add([1, 2, 3], 1, 'a'))
 300     self.assertEqual(True, r.add([4, 5], 0.5, 'b'))
 301     self.assertEqual(False, r.add([1, 2, 3], 1.5, 'a'))
 302     self.assertEqual(
 303         [([1, 2, 3], 1.0), ([4, 5], 0.5)],
 304         list(r))
 305     self.assertEqual(1.5, r.total_weight)
 306     self.assertEqual(
 307         2,
 308         r.add_many(
 309             [[5, 6, 2, 3], [1, 2, 3], [8], [1, 2, 3]],
 310             [0.1, 0.2, 0.1, 2.0],
 311             ['c', 'a', 'd', 'a']))
 312     self.assertEqual(
 313         [([1, 2, 3], 1.0), ([4, 5], 0.5), ([5, 6, 2, 3], 0.1), ([8], 0.1)],
 314         list(r))
 315     self.assertTrue(np.isclose(1.7, r.total_weight))
 316     self.assertEqual(0, r.add_many([], [], []))  # Adding no items is allowed.
 317     with self.assertRaises(ValueError):
 318       # Key not given.
 319       r.add([7, 8, 9], 2.0)
 320     with self.assertRaises(ValueError):
 321       # Keys not given.
 322       r.add_many([[7, 8, 9], [10]], [2.0, 2.0])
 323     self.assertEqual(True, r.has_key('a'))
 324     self.assertEqual(True, r.has_key('b'))
 325     self.assertEqual(False, r.has_key('z'))
 326     self.assertEqual(1.0, r.get_weight('a'))
 327     self.assertEqual(0.5, r.get_weight('b'))
 328 
 329     r = utils.RouletteWheel(unique_mode=False)
 330     self.assertEqual(True, r.add([1, 2, 3], 1))
 331     self.assertEqual(True, r.add([4, 5], 0.5))
 332     self.assertEqual(True, r.add([1, 2, 3], 1.5))
 333     self.assertEqual(
 334         [([1, 2, 3], 1.0), ([4, 5], 0.5), ([1, 2, 3], 1.5)],
 335         list(r))
 336     self.assertEqual(3, r.total_weight)
 337     self.assertEqual(
 338         4,
 339         r.add_many(
 340             [[5, 6, 2, 3], [1, 2, 3], [8], [1, 2, 3]],
 341             [0.1, 0.2, 0.1, 0.2]))
 342     self.assertEqual(
 343         [([1, 2, 3], 1.0), ([4, 5], 0.5), ([1, 2, 3], 1.5),
 344          ([5, 6, 2, 3], 0.1), ([1, 2, 3], 0.2), ([8], 0.1), ([1, 2, 3], 0.2)],
 345         list(r))
 346     self.assertTrue(np.isclose(3.6, r.total_weight))
 347     with self.assertRaises(ValueError):
 348       # Key is given.
 349       r.add([7, 8, 9], 2.0, 'a')
 350     with self.assertRaises(ValueError):
 351       # Keys are given.
 352       r.add_many([[7, 8, 9], [10]], [2.0, 2.0], ['a', 'b'])

 354   def testRouletteWheel_IncrementalSave(self):
 355     f = tempfile.NamedTemporaryFile()
 356     r = utils.RouletteWheel(unique_mode=True, save_file=f.name)
 357     entries = [
 358         ([1, 2, 3], 0.1, 'a'),
 359         ([4, 5], 0.2, 'b'),
 360         ([6], 0.3, 'c'),
 361         ([7, 8, 9, 10], 0.25, 'd'),
 362         ([-1, -2], 0.15, 'e'),
 363         ([-3, -4, -5], 0.5, 'f')]
 364 
 365     self.assertTrue(r.is_empty())
 366     for i in range(0, len(entries), 2):
 367       r.add(*entries[i])
 368       r.add(*entries[i + 1])
 369       r.incremental_save()
 370 
 371       r2 = utils.RouletteWheel(unique_mode=True, save_file=f.name)
 372       self.assertEqual(i + 2, len(r2))
 373       count = 0
 374       for j, (obj, weight) in enumerate(r2):
 375         self.assertEqual(entries[j][0], obj)
 376         self.assertEqual(entries[j][1], weight)
 377         self.assertEqual(weight, r2.get_weight(entries[j][2]))
 378         count += 1
 379       self.assertEqual(i + 2, count)
