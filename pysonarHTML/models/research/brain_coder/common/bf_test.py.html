  14   def assertCorrectOutput(self, target_output, eval_result):
  15     self.assertEqual(target_output, eval_result.output)
  16     self.assertTrue(eval_result.success)
  17     self.assertEqual(bf.Status.SUCCESS, eval_result.failure_reason)
  18 
  19   def testBasicOps(self):
  20     self.assertCorrectOutput(
  21         [3, 1, 2],
  22         bf.evaluate('+++.--.+.'))
  23     self.assertCorrectOutput(
  24         [1, 1, 2],
  25         bf.evaluate('+.<.>++.'))
  26     self.assertCorrectOutput(
  27         [0],
  28         bf.evaluate('+,.'))
  29     self.assertCorrectOutput(
  30         [ord(char) for char in 'Hello World!\n'],
  31         bf.evaluate(
  32             '>++++++++[-<+++++++++>]<.>>+>-[+]++>++>+++[>[->+++<<+++>]<<]>-----'
  33             '.>->+++..+++.>-.<<+[>[+>+]>>]<--------------.>>.+++.------.-------'
  34             '-.>+.>+.'))
  35 
  36   def testBase(self):
  37     self.assertCorrectOutput(
  38         [1, 4],
  39         bf.evaluate('+.--.', base=5, input_buffer=[]))
  40 
  41   def testInputBuffer(self):
  42     self.assertCorrectOutput(
  43         [2, 3, 4],
  44         bf.evaluate('>,[>,]<[.<]', input_buffer=[4, 3, 2]))

  46   def testBadChars(self):
  47     self.assertCorrectOutput(
  48         [2, 3, 4],
  49         bf.evaluate('>,[>,]hello<world[.<]comments',
  50                     input_buffer=[4, 3, 2]))
  51 
  52   def testUnmatchedBraces(self):
  53     self.assertCorrectOutput(
  54         [3, 6, 1],
  55         bf.evaluate('+++.]]]]>----.[[[[[>+.',
  56                     input_buffer=[],
  57                     base=10,
  58                     require_correct_syntax=False))
  59 
  60     eval_result = bf.evaluate(
  61         '+++.]]]]>----.[[[[[>+.',
  62         input_buffer=[],
  63         base=10,
  64         require_correct_syntax=True)
  65     self.assertEqual([], eval_result.output)
  66     self.assertFalse(eval_result.success)
  67     self.assertEqual(bf.Status.SYNTAX_ERROR,
  68                      eval_result.failure_reason)
  69 
  70   def testTimeout(self):
  71     er = bf.evaluate('+.[].', base=5, input_buffer=[], timeout=0.1)
  72     self.assertEqual(
  73         ([1], False, bf.Status.TIMEOUT),
  74         (er.output, er.success, er.failure_reason))
  75     self.assertTrue(0.07 < er.time < 0.21)
  76 
  77     er = bf.evaluate('+.[-].', base=5, input_buffer=[], timeout=0.1)
  78     self.assertEqual(
  79         ([1, 0], True, bf.Status.SUCCESS),
  80         (er.output, er.success, er.failure_reason))
  81     self.assertTrue(er.time < 0.15)
  82 
  83   def testMaxSteps(self):
  84     er = bf.evaluate('+.[].', base=5, input_buffer=[], timeout=None,
  85                      max_steps=100)
  86     self.assertEqual(
  87         ([1], False, bf.Status.STEP_LIMIT, 100),
  88         (er.output, er.success, er.failure_reason, er.steps))
  89 
  90     er = bf.evaluate('+.[-].', base=5, input_buffer=[], timeout=None,
  91                      max_steps=100)
  92     self.assertEqual(
  93         ([1, 0], True, bf.Status.SUCCESS),
  94         (er.output, er.success, er.failure_reason))
  95     self.assertTrue(er.steps < 100)
  96 
  97   def testOutputMemory(self):
  98     er = bf.evaluate('+>++>+++>++++.', base=256, input_buffer=[],
  99                      output_memory=True)
 100     self.assertEqual(
 101         ([4], True, bf.Status.SUCCESS),
 102         (er.output, er.success, er.failure_reason))
 103     self.assertEqual([1, 2, 3, 4], er.memory)
