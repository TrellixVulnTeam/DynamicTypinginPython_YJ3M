  14 class Schedule(object):
  15   """Schedule is a function which sets a hyperparameter's value over time.
  16 
  17   For example, a schedule can be used to decay an hparams, or oscillate it over
  18   time.
  19 
  20   This object is constructed with an instance of config_lib.Config (will be
  21   specific to each class implementation). For example if this is a decay
  22   schedule, the config may specify the rate of decay and decay start time. Then
  23   the object instance is called like a function, mapping global step (an integer
  24   counting how many calls to the train op have been made) to the hparam value.
  25 
  26   Properties of a schedule function f(t):
  27   0) Domain of t is the non-negative integers (t may be 0).
  28   1) Range of f is the reals.
  29   2) Schedule functions can assume that they will be called in time order. This
  30      allows schedules to be stateful.
  31   3) Schedule functions should be deterministic. Two schedule instances with the
  32      same config must always give the same value for each t, and regardless of
  33      what t's it was previously called on. Users may call f(t) on arbitrary
  34      (positive) time jumps. Essentially, multiple schedule instances used in
  35      replica training will behave the same.
  36   4) Duplicate successive calls on the same time are allowed.
  37   """
  38   __metaclass__ = ABCMeta
  39 
  40   @abstractmethod
  41   def __init__(self, config):
  42     """Construct this schedule with a config specific to each class impl.
  43 
  44     Args:
  45       config: An instance of config_lib.Config.
  46     """
  47     pass
  48 
  49   @abstractmethod
  50   def __call__(self, global_step):
  51     """Map `global_step` to a value.
  52 
  53     `global_step` is an integer counting how many calls to the train op have
  54     been made across all replicas (hence why it is global). Implementations
  55     may assume calls to be made in time order, i.e. `global_step` now >=
  56     previous `global_step` values.
  57 
  58     Args:
  59       global_step: Non-negative integer.
  60 
  61     Returns:
  62       Hparam value at this step. A number.
  63     """
  64     pass
  65

  67 class ConstSchedule(Schedule):
  68   """Constant function.
  69 
  70   config:
  71     const: Constant value at every step.
  72 
  73   f(t) = const.
  74   """
  75 
  76   def __init__(self, config):
  77     super(ConstSchedule, self).__init__(config)
  78     self.const = config.const
  79 
  80   def __call__(self, global_step):
  81     return self.const

  84 class LinearDecaySchedule(Schedule):
  85   """Linear decay function.
  86 
  87   config:
  88     initial: Decay starts from this value.
  89     final: Decay ends at this value.
  90     start_time: Step when decay starts. Constant before it.
  91     end_time: When decay ends. Constant after it.
  92 
  93   f(t) is a linear function when start_time <= t <= end_time, with slope of
  94   (final - initial) / (end_time - start_time). f(t) = initial
  95   when t <= start_time. f(t) = final when t >= end_time.
  96 
  97   If start_time == end_time, this becomes a step function.
  98   """
  99 
 100   def __init__(self, config):
 101     super(LinearDecaySchedule, self).__init__(config)
 102     self.initial = config.initial
 103     self.final = config.final
 104     self.start_time = config.start_time
 105     self.end_time = config.end_time
 106 
 107     if self.end_time < self.start_time:
 108       raise ValueError('start_time must be before end_time.')
 109 
 110     # Linear interpolation.
 111     self._time_diff = float(self.end_time - self.start_time)
 112     self._diff = float(self.final - self.initial)
 113     self._slope = (
 114         self._diff / self._time_diff if self._time_diff > 0 else float('inf'))
 115 
 116   def __call__(self, global_step):
 117     if global_step <= self.start_time:
 118       return self.initial
 119     if global_step > self.end_time:
 120       return self.final
 121     return self.initial + (global_step - self.start_time) * self._slope

 124 class ExponentialDecaySchedule(Schedule):
 125   """Exponential decay function.
 126 
 127   See https://en.wikipedia.org/wiki/Exponential_decay.
 128 
 129   Use this decay function to decay over orders of magnitude. For example, to
 130   decay learning rate from 1e-2 to 1e-6. Exponential decay will decay the
 131   exponent linearly.
 132 
 133   config:
 134     initial: Decay starts from this value.
 135     final: Decay ends at this value.
 136     start_time: Step when decay starts. Constant before it.
 137     end_time: When decay ends. Constant after it.
 138 
 139   f(t) is an exponential decay function when start_time <= t <= end_time. The
 140   decay rate and amplitude are chosen so that f(t) = initial when
 141   t = start_time, and f(t) = final when t = end_time. f(t) is constant for
 142   t < start_time or t > end_time. initial and final must be positive values.
 143 
 144   If start_time == end_time, this becomes a step function.
 145   """
 146 
 147   def __init__(self, config):
 148     super(ExponentialDecaySchedule, self).__init__(config)
 149     self.initial = config.initial
 150     self.final = config.final
 151     self.start_time = config.start_time
 152     self.end_time = config.end_time
 153 
 154     if self.initial <= 0 or self.final <= 0:
 155       raise ValueError('initial and final must be positive numbers.')
 156 
 157     # Linear interpolation in log space.
 158     self._linear_fn = LinearDecaySchedule(
 159         config_lib.Config(
 160             initial=math.log(self.initial),
 161             final=math.log(self.final),
 162             start_time=self.start_time,
 163             end_time=self.end_time))
 164 
 165   def __call__(self, global_step):
 166     return math.exp(self._linear_fn(global_step))

 169 class SmootherstepDecaySchedule(Schedule):
 170   """Smootherstep decay function.
 171 
 172   A sigmoidal like transition from initial to final values. A smoother
 173   transition than linear and exponential decays, hence the name.
 174   See https://en.wikipedia.org/wiki/Smoothstep.
 175 
 176   config:
 177     initial: Decay starts from this value.
 178     final: Decay ends at this value.
 179     start_time: Step when decay starts. Constant before it.
 180     end_time: When decay ends. Constant after it.
 181 
 182   f(t) is fully defined here:
 183   https://en.wikipedia.org/wiki/Smoothstep#Variations.
 184 
 185   f(t) is smooth, as in its first-derivative exists everywhere.
 186   """
 187 
 188   def __init__(self, config):
 189     super(SmootherstepDecaySchedule, self).__init__(config)
 190     self.initial = config.initial
 191     self.final = config.final
 192     self.start_time = config.start_time
 193     self.end_time = config.end_time
 194 
 195     if self.end_time < self.start_time:
 196       raise ValueError('start_time must be before end_time.')
 197 
 198     self._time_diff = float(self.end_time - self.start_time)
 199     self._diff = float(self.final - self.initial)
 200 
 201   def __call__(self, global_step):
 202     if global_step <= self.start_time:
 203       return self.initial
 204     if global_step > self.end_time:
 205       return self.final
 206     x = (global_step - self.start_time) / self._time_diff
 207 
 208     # Smootherstep
 209     return self.initial + x * x * x * (x * (x * 6 - 15) + 10) * self._diff

 212 class HardOscillatorSchedule(Schedule):
 213   """Hard oscillator function.
 214 
 215   config:
 216     high: Max value of the oscillator. Value at constant plateaus.
 217     low: Min value of the oscillator. Value at constant valleys.
 218     start_time: Global step when oscillation starts. Constant before this.
 219     period: Width of one oscillation, i.e. number of steps over which the
 220         oscillation takes place.
 221     transition_fraction: Fraction of the period spent transitioning between high
 222         and low values. 50% of this time is spent rising, and 50% of this time
 223         is spent falling. 50% of the remaining time is spent constant at the
 224         high value, and 50% of the remaining time is spent constant at the low
 225         value. transition_fraction = 1.0 means the entire period is spent
 226         rising and falling. transition_fraction = 0.0 means no time is spent
 227         rising and falling, i.e. the function jumps instantaneously between
 228         high and low.
 229 
 230   f(t) = high when t < start_time.
 231   f(t) is periodic when t >= start_time, with f(t + period) = f(t).
 232   f(t) is linear with positive slope when rising, and negative slope when
 233   falling. At the start of the period t0, f(t0) = high and begins to descend.
 234   At the middle of the period f is low and is constant until the ascension
 235   begins. f then rises from low to high and is constant again until the period
 236   repeats.
 237 
 238   Note: when transition_fraction is 0, f starts the period low and ends high.
 239   """
 240 
 241   def __init__(self, config):
 242     super(HardOscillatorSchedule, self).__init__(config)
 243     self.high = config.high
 244     self.low = config.low
 245     self.start_time = config.start_time
 246     self.period = float(config.period)
 247     self.transition_fraction = config.transition_fraction
 248     self.half_transition_fraction = config.transition_fraction / 2.0
 249 
 250     if self.transition_fraction < 0 or self.transition_fraction > 1.0:
 251       raise ValueError('transition_fraction must be between 0 and 1.0')
 252     if self.period <= 0:
 253       raise ValueError('period must be positive')
 254 
 255     self._slope = (
 256         float(self.high - self.low) / self.half_transition_fraction
 257         if self.half_transition_fraction > 0 else float('inf'))
 258 
 259   def __call__(self, global_step):
 260     if global_step < self.start_time:
 261       return self.high
 262     period_pos = ((global_step - self.start_time) / self.period) % 1.0
 263     if period_pos >= 0.5:
 264       # ascending
 265       period_pos -= 0.5
 266       if period_pos < self.half_transition_fraction:
 267         return self.low + period_pos * self._slope
 268       else:
 269         return self.high
 270     else:
 271       # descending
 272       if period_pos < self.half_transition_fraction:
 273         return self.high - period_pos * self._slope
 274       else:
 275         return self.low

 287 def make_schedule(config):
 288   """Schedule factory.
 289 
 290   Given `config` containing a `fn` property, a Schedule implementation is
 291   instantiated with `config`. See `_NAME_TO_CONFIG` for `fn` options.
 292 
 293   Args:
 294     config: Config with a `fn` option that specifies which Schedule
 295         implementation to use. `config` is passed into the constructor.
 296 
 297   Returns:
 298     A Schedule impl instance.
 299   """
 300   schedule_class = _NAME_TO_CONFIG[config.fn]
 301   return schedule_class(config)
