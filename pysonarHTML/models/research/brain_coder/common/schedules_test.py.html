  19   def ScheduleTestHelper(self, config, schedule_subtype, io_values):
  20     """Run common checks for schedules.
  21 
  22     Args:
  23       config: Config object which is passed into schedules.make_schedule.
  24       schedule_subtype: The expected schedule type to be instantiated.
  25       io_values: List of (input, output) pairs. Must be in ascending input
  26           order. No duplicate inputs.
  27     """
  28 
  29     # Check that make_schedule makes the correct type.
  30     f = schedules.make_schedule(config)
  31     self.assertTrue(isinstance(f, schedule_subtype))
  32 
  33     # Check that multiple instances returned from make_schedule behave the same.
  34     fns = [schedules.make_schedule(config) for _ in xrange(3)]
  35 
  36     # Check that all the inputs map to the right outputs.
  37     for i, o in io_values:
  38       for f in fns:
  39         f_out = f(i)
  40         self.assertTrue(
  41             np.isclose(o, f_out),
  42             'Wrong value at input %d. Expected %s, got %s' % (i, o, f_out))
  43 
  44     # Check that a subset of the io_values are still correct.
  45     f = schedules.make_schedule(config)
  46     subseq = [io_values[i**2] for i in xrange(int(sqrt(len(io_values))))]
  47     if subseq[-1] != io_values[-1]:
  48       subseq.append(io_values[-1])
  49     for i, o in subseq:
  50       f_out = f(i)
  51       self.assertTrue(
  52           np.isclose(o, f_out),
  53           'Wrong value at input %d. Expected %s, got %s' % (i, o, f_out))
  54 
  55     # Check duplicate calls.
  56     f = schedules.make_schedule(config)
  57     for i, o in io_values:
  58       for _ in xrange(3):
  59         f_out = f(i)
  60         self.assertTrue(
  61             np.isclose(o, f_out),
  62             'Duplicate calls at input %d are not equal. Expected %s, got %s'
  63             % (i, o, f_out))
