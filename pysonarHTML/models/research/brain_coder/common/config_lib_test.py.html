 176   def testConfigStrictUpdateFail(self):
 177     config = config_lib.Config(a=1, b=2, c=3, x=config_lib.Config(a=1, b=2))
 178     with self.assertRaises(KeyError):
 179       config.strict_update({'b': 10, 'c': 20, 'd': 50})
 180     with self.assertRaises(KeyError):
 181       config.strict_update(b=10, d=50)
 182     with self.assertRaises(KeyError):
 183       config.strict_update(x={'c': 3})
 184     with self.assertRaises(TypeError):
 185       config.strict_update(a='string')
 186     with self.assertRaises(TypeError):
 187       config.strict_update(x={'a': 'string'})
 188     with self.assertRaises(TypeError):
 189       config.strict_update(x=[1, 2, 3])

 198   def testConfigParse(self):
 199     config = config_lib.Config.parse(
 200         'hello="world",num=1234.5,lst=[10,20.5,True,"hi",("a","b","c")],'
 201         'dct={9:10,"stuff":"qwerty","subdict":{1:True,2:False}},'
 202         'subconfig=c(a=1,b=[1,2,[3,4]],c=c(f="f",g="g"))')
 203     self.assertEqual(
 204         {'hello': 'world', 'num': 1234.5,
 205          'lst': [10, 20.5, True, 'hi', ('a', 'b', 'c')],
 206          'dct': {9: 10, 'stuff': 'qwerty', 'subdict': {1: True, 2: False}},
 207          'subconfig': {'a': 1, 'b': [1, 2, [3, 4]], 'c': {'f': 'f', 'g': 'g'}}},
 208         config)
 209     self.assertTrue(isinstance(config, config_lib.Config))
 210     self.assertTrue(isinstance(config.subconfig, config_lib.Config))
 211     self.assertTrue(isinstance(config.subconfig.c, config_lib.Config))
 212     self.assertFalse(isinstance(config.dct, config_lib.Config))
 213     self.assertFalse(isinstance(config.dct['subdict'], config_lib.Config))
 214     self.assertTrue(isinstance(config.lst[4], tuple))

 216   def testConfigParseErrors(self):
 217     with self.assertRaises(SyntaxError):
 218       config_lib.Config.parse('a=[1,2,b="hello"')
 219     with self.assertRaises(SyntaxError):
 220       config_lib.Config.parse('a=1,b=c(x="a",y="b"')
 221     with self.assertRaises(SyntaxError):
 222       config_lib.Config.parse('a=1,b=c(x="a")y="b"')
 223     with self.assertRaises(SyntaxError):
 224       config_lib.Config.parse('a=1,b=c(x="a"),y="b",')

 226   def testOneOf(self):
 227     def make_config():
 228       return config_lib.Config(
 229           data=config_lib.OneOf(
 230               [config_lib.Config(task=1, a='hello'),
 231                config_lib.Config(task=2, a='world', b='stuff'),
 232                config_lib.Config(task=3, c=1234)],
 233               task=2),
 234           model=config_lib.Config(stuff=1))
 235 
 236     config = make_config()
 237     config.update(config_lib.Config.parse(
 238         'model=c(stuff=2),data=c(task=1,a="hi")'))
 239     self.assertEqual(
 240         config_lib.Config(
 241             data=config_lib.Config(task=1, a='hi'),
 242             model=config_lib.Config(stuff=2)),
 243         config)
 244 
 245     config = make_config()
 246     config.update(config_lib.Config.parse(
 247         'model=c(stuff=2),data=c(task=2,a="hi")'))
 248     self.assertEqual(
 249         config_lib.Config(
 250             data=config_lib.Config(task=2, a='hi', b='stuff'),
 251             model=config_lib.Config(stuff=2)),
 252         config)
 253 
 254     config = make_config()
 255     config.update(config_lib.Config.parse(
 256         'model=c(stuff=2),data=c(task=3)'))
 257     self.assertEqual(
 258         config_lib.Config(
 259             data=config_lib.Config(task=3, c=1234),
 260             model=config_lib.Config(stuff=2)),
 261         config)
 262 
 263     config = make_config()
 264     config.update(config_lib.Config.parse(
 265         'model=c(stuff=2)'))
 266     self.assertEqual(
 267         config_lib.Config(
 268             data=config_lib.Config(task=2, a='world', b='stuff'),
 269             model=config_lib.Config(stuff=2)),
 270         config)
 271 
 272     config = make_config()
 273     config.update(config_lib.Config.parse(
 274         'model=c(stuff=2),data=c(task=4,d=9999)'))
 275     self.assertEqual(
 276         config_lib.Config(
 277             data=config_lib.Config(task=4, d=9999),
 278             model=config_lib.Config(stuff=2)),
 279         config)
 280 
 281     config = make_config()
 282     config.update(config_lib.Config.parse(
 283         'model=c(stuff=2),data=5'))
 284     self.assertEqual(
 285         config_lib.Config(
 286             data=5,
 287             model=config_lib.Config(stuff=2)),
 288         config)
