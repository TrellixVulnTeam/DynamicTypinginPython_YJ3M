  18 class Rollout(object):
  19   """Holds a rollout for an episode.
  20 
  21   A rollout is a record of the states observed in some environment and actions
  22   taken by the agent to arrive at those states. Other information includes
  23   rewards received after each action, values estimated for each state, whether
  24   the rollout concluded the episide, and total reward received. Everything
  25   should be given in time order.
  26 
  27   At each time t, the agent sees state s_t, takes action a_t, and then receives
  28   reward r_t. The agent may optionally estimate a state value V(s_t) for each
  29   state.
  30 
  31   For an episode of length T:
  32   states = [s_0, ..., s_(T-1)]
  33   actions = [a_0, ..., a_(T-1)]
  34   rewards = [r_0, ..., r_(T-1)]
  35   values = [V(s_0), ..., V(s_(T-1))]
  36 
  37   Note that there is an extra state s_T observed after taking action a_(T-1),
  38   but this is not included in the rollout.
  39 
  40   Rollouts have an `terminated` attribute which is True when the rollout is
  41   "finalized", i.e. it holds a full episode. terminated will be False when
  42   time steps are still being added to it.
  43   """
  44 
  45   def __init__(self):
  46     self.states = []
  47     self.actions = []
  48     self.rewards = []
  49     self.values = []
  50     self.total_reward = 0.0
  51     self.terminated = False
  52 
  53   def add(self, state, action, reward, value=0.0, terminated=False):
  54     """Add the next timestep to this rollout.
  55 
  56     Args:
  57       state: The state observed at the start of this timestep.
  58       action: The action taken after observing the given state.
  59       reward: The reward received for taking the given action.
  60       value: The value estimated for the given state.
  61       terminated: Whether this timestep ends the episode.
  62 
  63     Raises:
  64       ValueError: If this.terminated is already True, meaning that the episode
  65           has already ended.
  66     """
  67     if self.terminated:
  68       raise ValueError(
  69           'Trying to add timestep to an already terminal rollout.')
  70     self.states += [state]
  71     self.actions += [action]
  72     self.rewards += [reward]
  73     self.values += [value]
  74     self.terminated = terminated
  75     self.total_reward += reward
  76 
  77   def add_many(self, states, actions, rewards, values=None, terminated=False):
  78     """Add many timesteps to this rollout.
  79 
  80     Arguments are the same as `add`, but are lists of equal size.
  81 
  82     Args:
  83       states: The states observed.
  84       actions: The actions taken.
  85       rewards: The rewards received.
  86       values: The values estimated for the given states.
  87       terminated: Whether this sequence ends the episode.
  88 
  89     Raises:
  90       ValueError: If the lengths of all the input lists are not equal.
  91       ValueError: If this.terminated is already True, meaning that the episode
  92           has already ended.
  93     """
  94     if len(states) != len(actions):
  95       raise ValueError(
  96           'Number of states and actions must be the same. Got %d states and '
  97           '%d actions' % (len(states), len(actions)))
  98     if len(states) != len(rewards):
  99       raise ValueError(
 100           'Number of states and rewards must be the same. Got %d states and '
 101           '%d rewards' % (len(states), len(rewards)))
 102     if values is not None and len(states) != len(values):
 103       raise ValueError(
 104           'Number of states and values must be the same. Got %d states and '
 105           '%d values' % (len(states), len(values)))
 106     if self.terminated:
 107       raise ValueError(
 108           'Trying to add timesteps to an already terminal rollout.')
 109     self.states += states
 110     self.actions += actions
 111     self.rewards += rewards
 112     self.values += values if values is not None else [0.0] * len(states)
 113     self.terminated = terminated
 114     self.total_reward += sum(rewards)
 115 
 116   def extend(self, other):
 117     """Append another rollout to this rollout."""
 118     assert not self.terminated
 119     self.states.extend(other.states)
 120     self.actions.extend(other.actions)
 121     self.rewards.extend(other.rewards)
 122     self.values.extend(other.values)
 123     self.terminated = other.terminated
 124     self.total_reward += other.total_reward

 127 def discount(x, gamma):
 128   """Returns discounted sums for each value in x, with discount factor gamma.
 129 
 130   This can be used to compute the return (discounted sum of rewards) at each
 131   timestep given a sequence of rewards. See the definitions for return and
 132   REINFORCE in section 3 of https://arxiv.org/pdf/1602.01783.pdf.
 133 
 134   Let g^k mean gamma ** k.
 135   For list [x_0, ..., x_N], the following list of discounted sums is computed:
 136   [x_0 + g^1 * x_1 + g^2 * x_2 + ... g^N * x_N,
 137    x_1 + g^1 * x_2 + g^2 * x_3 + ... g^(N-1) * x_N,
 138    x_2 + g^1 * x_3 + g^2 * x_4 + ... g^(N-2) * x_N,
 139    ...,
 140    x_(N-1) + g^1 * x_N,
 141    x_N]
 142 
 143   Args:
 144     x: List of numbers [x_0, ..., x_N].
 145     gamma: Float between 0 and 1 (inclusive). This is the discount factor.
 146 
 147   Returns:
 148     List of discounted sums.
 149   """
 150   return scipy.signal.lfilter([1], [1, -gamma], x[::-1], axis=0)[::-1]

 153 def discounted_advantage_and_rewards(rewards, values, gamma, lambda_=1.0):
 154   """Compute advantages and returns (discounted sum of rewards).
 155 
 156   For an episode of length T, rewards = [r_0, ..., r_(T-1)].
 157   Each reward r_t is observed after taking action a_t at state s_t. A final
 158   state s_T is observed but no reward is given at this state since no action
 159   a_T is taken (otherwise there would be a new state s_(T+1)).
 160 
 161   `rewards` and `values` are for a single episode. Return R_t is the discounted
 162   sum of future rewards starting at time t, where `gamma` is the discount
 163   factor.
 164   R_t = r_t + gamma * r_(t+1) + gamma**2 * r_(t+2) + ...
 165         + gamma**(T-1-t) * r_(T-1)
 166 
 167   Advantage A(a_t, s_t) is approximated by computing A(a_t, s_t) = R_t - V(s_t)
 168   where V(s_t) is an approximation of the value at that state, given in the
 169   `values` list. Returns R_t are needed for all REINFORCE algorithms. Advantage
 170   is used for the advantage actor critic variant of REINFORCE.
 171   See algorithm S3 in https://arxiv.org/pdf/1602.01783.pdf.
 172 
 173   Additionally another parameter `lambda_` controls the bias-variance tradeoff.
 174   See "Generalized Advantage Estimation": https://arxiv.org/abs/1506.02438.
 175   lambda_ = 1 reduces to regular advantage.
 176   0 <= lambda_ < 1 trades off variance for bias, with lambda_ = 0 being the
 177   most biased.
 178 
 179   Bootstrapping is also supported. If an episode does not end in a terminal
 180   state (either because the episode was ended early, or the environment does not
 181   have end states), the true return cannot be computed from the rewards alone.
 182   However, it can be estimated by computing the value (an approximation of
 183   return) of the last state s_T. Thus the `values` list will have an extra item:
 184   values = [V(s_0), ..., V(s_(T-1)), V(s_T)].
 185 
 186   Args:
 187     rewards: List of observed rewards [r_0, ..., r_(T-1)].
 188     values: List of estimated values [V(s_0), ..., V(s_(T-1))] with an optional
 189         extra V(s_T) item.
 190     gamma: Discount factor. Number between 0 and 1. 1 means no discount.
 191         If not 1, gamma is typically near 1, like 0.99.
 192     lambda_: Bias-variance tradeoff factor. Between 0 and 1.
 193 
 194   Returns:
 195     empirical_values: Returns at each timestep.
 196     generalized_advantage: Avantages at each timestep.
 197 
 198   Raises:
 199     ValueError: If shapes of `rewards` and `values` are not rank 1.
 200     ValueError: If len(values) not in (len(rewards), len(rewards) + 1).
 201   """
 202   rewards = np.asarray(rewards, dtype=np.float32)
 203   values = np.asarray(values, dtype=np.float32)
 204   if rewards.ndim != 1:
 205     raise ValueError('Single episode only. rewards must be rank 1.')
 206   if values.ndim != 1:
 207     raise ValueError('Single episode only. values must be rank 1.')
 208   if len(values) == len(rewards):
 209     # No bootstrapping.
 210     values = np.append(values, 0)
 211     empirical_values = discount(rewards, gamma)
 212   elif len(values) == len(rewards) + 1:
 213     # With bootstrapping.
 214     # Last value is for the terminal state (final state after last action was
 215     # taken).
 216     empirical_values = discount(np.append(rewards, values[-1]), gamma)[:-1]
 217   else:
 218     raise ValueError('values should contain the same number of items or one '
 219                      'more item than rewards')
 220   delta = rewards + gamma * values[1:] - values[:-1]
 221   generalized_advantage = discount(delta, gamma * lambda_)
 222 
 223   # empirical_values is the discounted sum of rewards into the future.
 224   # generalized_advantage is the target for each policy update.
 225   return empirical_values, generalized_advantage

 228 """Batch holds a minibatch of episodes.
 229 
 230 Let bi = batch_index, i.e. the index of each episode in the minibatch.
 231 Let t = time.
 232 
 233 Attributes:
 234   states: States for each timestep in each episode. Indexed by states[bi, t].
 235   actions: Actions for each timestep in each episode. Indexed by actions[bi, t].
 236   discounted_adv: Advantages (computed by discounted_advantage_and_rewards)
 237       for each timestep in each episode. Indexed by discounted_adv[bi, t].
 238   discounted_r: Returns (discounted sum of rewards computed by
 239       discounted_advantage_and_rewards) for each timestep in each episode.
 240       Indexed by discounted_r[bi, t].
 241   total_rewards: Total reward for each episode, i.e. sum of rewards across all
 242       timesteps (not discounted). Indexed by total_rewards[bi].
 243   episode_lengths: Number of timesteps in each episode. If an episode has
 244       N actions, N rewards, and N states, then its length is N. Indexed by
 245       episode_lengths[bi].
 246   batch_size: Number of episodes in this minibatch. An integer.
 247   max_time: Maximum episode length in the batch. An integer.
 248 """  # pylint: disable=pointless-string-statement
 249 Batch = namedtuple(
 250     'Batch',
 251     ['states', 'actions', 'discounted_adv', 'discounted_r', 'total_rewards',
 252      'episode_lengths', 'batch_size', 'max_time'])

 255 def process_rollouts(rollouts, gamma, lambda_=1.0):
 256   """Convert a batch of rollouts into tensors ready to be fed into a model.
 257 
 258   Lists from each episode are stacked into 2D tensors and padded with 0s up to
 259   the maximum timestep in the batch.
 260 
 261   Args:
 262     rollouts: A list of Rollout instances.
 263     gamma: The discount factor. A number between 0 and 1 (inclusive). See gamma
 264         argument in discounted_advantage_and_rewards.
 265     lambda_: See lambda_ argument in discounted_advantage_and_rewards.
 266 
 267   Returns:
 268     Batch instance. states, actions, discounted_adv, and discounted_r are
 269     numpy arrays with shape (batch_size, max_episode_length). episode_lengths
 270     is a list of ints. total_rewards is a list of floats (total reward in each
 271     episode). batch_size and max_time are ints.
 272 
 273   Raises:
 274     ValueError: If any of the rollouts are not terminal.
 275   """
 276   for ro in rollouts:
 277     if not ro.terminated:
 278       raise ValueError('Can only process terminal rollouts.')
 279 
 280   episode_lengths = [len(ro.states) for ro in rollouts]
 281   batch_size = len(rollouts)
 282   max_time = max(episode_lengths)
 283 
 284   states = utils.stack_pad([ro.states for ro in rollouts], 0, max_time)
 285   actions = utils.stack_pad([ro.actions for ro in rollouts], 0, max_time)
 286 
 287   discounted_rewards = [None] * batch_size
 288   discounted_adv = [None] * batch_size
 289   for i, ro in enumerate(rollouts):
 290     disc_r, disc_adv = discounted_advantage_and_rewards(
 291         ro.rewards, ro.values, gamma, lambda_)
 292     discounted_rewards[i] = disc_r
 293     discounted_adv[i] = disc_adv
 294   discounted_rewards = utils.stack_pad(discounted_rewards, 0, max_time)
 295   discounted_adv = utils.stack_pad(discounted_adv, 0, max_time)
 296 
 297   total_rewards = [sum(ro.rewards) for ro in rollouts]
 298 
 299   return Batch(states=states,
 300                actions=actions,
 301                discounted_adv=discounted_adv,
 302                discounted_r=discounted_rewards,
 303                total_rewards=total_rewards,
 304                episode_lengths=episode_lengths,
 305                batch_size=batch_size,
 306                max_time=max_time)
