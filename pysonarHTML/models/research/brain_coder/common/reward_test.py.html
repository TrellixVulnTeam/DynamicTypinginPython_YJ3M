 125   def testLogAbsoluteDistance(self):
 126     def log_diff(diff, base):
 127       return log(diff + 1) / log(base // 2 + 2)
 128 
 129     self.assertEqual(
 130         log_diff(1, 5),
 131         reward.log_absolute_distance([0], [4], 5))
 132     self.assertEqual(
 133         log_diff(2, 5),
 134         reward.log_absolute_distance([1], [4], 5))
 135     self.assertEqual(
 136         log_diff(2, 5),
 137         reward.log_absolute_distance([2], [4], 5))
 138     self.assertEqual(
 139         log_diff(1, 5),
 140         reward.log_absolute_distance([3], [4], 5))
 141     self.assertEqual(
 142         log_diff(3, 5),  # max_dist = base // 2 + 1 = 3
 143         reward.log_absolute_distance([], [4], 5))
 144     self.assertEqual(
 145         0 + log_diff(3, 5),  # max_dist = base // 2 + 1 = 3
 146         reward.log_absolute_distance([4, 4], [4], 5))
 147     self.assertEqual(
 148         0,
 149         reward.log_absolute_distance([4], [4], 5))
 150     self.assertEqual(
 151         0,
 152         reward.log_absolute_distance([], [], 5))
 153     self.assertEqual(
 154         1,
 155         reward.log_absolute_distance([1], [], 5))
 156     self.assertEqual(
 157         1,
 158         reward.log_absolute_distance([], [1], 5))
 159 
 160     self.assertEqual(
 161         0,
 162         reward.log_absolute_distance([1, 2, 3], [1, 2, 3], 5))
 163     self.assertEqual(
 164         log_diff(1, 5) / 3,  # divided by target length.
 165         reward.log_absolute_distance([1, 2, 4], [1, 2, 3], 5))
 166     self.assertEqual(
 167         log_diff(1, 5) / 3,
 168         reward.log_absolute_distance([1, 2, 2], [1, 2, 3], 5))
 169     self.assertEqual(
 170         log_diff(3, 5) / 3,  # max_dist
 171         reward.log_absolute_distance([1, 2], [1, 2, 3], 5))
 172     self.assertEqual(
 173         log_diff(3, 5) / 3,  # max_dist
 174         reward.log_absolute_distance([1, 2, 3, 4], [1, 2, 3], 5))
 175     # Add log differences for each position.
 176     self.assertEqual(
 177         (log_diff(2, 5) + log_diff(2, 5) + log_diff(1, 5)) / 3,
 178         reward.log_absolute_distance([4, 4, 4], [1, 2, 3], 5))

 180   def testAbsoluteDistanceReward(self):
 181     self.assertEqual(
 182         1,
 183         reward.absolute_distance_reward([1, 2, 3], [1, 2, 3], 5))
 184     self.assertEqual(
 185         1 - 1 / (5 * 3.),  # 1 - distance / (base * target_len)
 186         reward.absolute_distance_reward([1, 2, 4], [1, 2, 3], 5))
 187     self.assertEqual(
 188         1 - 1 / (5 * 3.),
 189         reward.absolute_distance_reward([1, 2, 2], [1, 2, 3], 5))
 190     self.assertTrue(np.isclose(
 191         1 - 5 / (5 * 3.),
 192         reward.absolute_distance_reward([1, 2], [1, 2, 3], 5)))
 193     self.assertTrue(np.isclose(
 194         1 - 5 / (5 * 3.),
 195         reward.absolute_distance_reward([1, 2, 3, 4], [1, 2, 3], 5)))
 196     # Add log differences for each position.
 197     self.assertEqual(
 198         1 - (3 + 2 + 1) / (5 * 3.),
 199         reward.absolute_distance_reward([4, 4, 4], [1, 2, 3], 5))
 200     self.assertEqual(
 201         1,
 202         reward.absolute_distance_reward([], [], 5))

 231   def testAbsoluteLogDistanceReward(self):
 232     def log_diff(diff, base):
 233       return log(diff + 1) / log(base // 2 + 2)
 234 
 235     self.assertEqual(
 236         1,
 237         reward.absolute_log_distance_reward([1, 2, 3], [1, 2, 3], 5))
 238     self.assertEqual(
 239         1 - log_diff(1, 5) / 3,  # divided by target length.
 240         reward.absolute_log_distance_reward([1, 2, 4], [1, 2, 3], 5))
 241     self.assertEqual(
 242         1 - log_diff(1, 5) / 3,
 243         reward.absolute_log_distance_reward([1, 2, 2], [1, 2, 3], 5))
 244     self.assertEqual(
 245         1 - log_diff(3, 5) / 3,  # max_dist
 246         reward.absolute_log_distance_reward([1, 2], [1, 2, 3], 5))
 247     self.assertEqual(
 248         1 - log_diff(3, 5) / 3,  # max_dist
 249         reward.absolute_log_distance_reward([1, 2, 3, 4], [1, 2, 3], 5))
 250     # Add log differences for each position.
 251     self.assertEqual(
 252         1 - (log_diff(2, 5) + log_diff(2, 5) + log_diff(1, 5)) / 3,
 253         reward.absolute_log_distance_reward([4, 4, 4], [1, 2, 3], 5))
 254     self.assertEqual(
 255         1 - (log_diff(1, 5) + log_diff(2, 5) + log_diff(2, 5)) / 3,
 256         reward.absolute_log_distance_reward([0, 1, 2], [4, 4, 4], 5))
 257     self.assertEqual(
 258         1,
 259         reward.absolute_log_distance_reward([], [], 5))

 261   def testDeltaRewardManager(self):

 263         [1, 2, 3, 4], base=5, distance_fn=reward.absolute_distance)
 264     self.assertEqual(-3, reward_manager([1]))
 265     self.assertEqual(0, reward_manager([1]))
 266     self.assertEqual(4 / 5., reward_manager([1, 3]))
 267     self.assertEqual(-4 / 5, reward_manager([1]))
 268     self.assertEqual(3, reward_manager([1, 2, 3, 4]))
 269     self.assertEqual(-1, reward_manager([1, 2, 3]))
 270     self.assertEqual(0, reward_manager([1, 2, 3, 4, 3]))
 271     self.assertEqual(-1, reward_manager([1, 2, 3, 4, 3, 2]))
 272     self.assertEqual(2, reward_manager([1, 2, 3, 4]))
 273     self.assertEqual(0, reward_manager([1, 2, 3, 4]))
 274     self.assertEqual(0, reward_manager([1, 2, 3, 4]))

 277     reward_manager = reward.FloorRewardManager(
 278         [1, 2, 3, 4], base=5, distance_fn=reward.absolute_distance)
 279     self.assertEqual(1, reward_manager([1]))
 280     self.assertEqual(0, reward_manager([1]))
 281     self.assertEqual(4 / 5., reward_manager([1, 3]))
 282     self.assertEqual(0, reward_manager([1]))
 283     self.assertEqual(1 / 5., reward_manager([1, 2]))
 284     self.assertEqual(0, reward_manager([0, 1]))
 285     self.assertEqual(0, reward_manager([]))
 286     self.assertEqual(0, reward_manager([1, 2]))
 287     self.assertEqual(2, reward_manager([1, 2, 3, 4]))
 288     self.assertEqual(0, reward_manager([1, 2, 3]))
 289     self.assertEqual(-1, reward_manager([1, 2, 3, 4, 3]))
 290     self.assertEqual(0, reward_manager([1, 2, 3, 4, 3, 2]))
 291     self.assertEqual(1, reward_manager([1, 2, 3, 4]))
 292     self.assertEqual(0, reward_manager([1, 2, 3, 4]))
 293     self.assertEqual(0, reward_manager([1, 2, 3, 4]))
 294 
 295     reward_manager = reward.FloorRewardManager(
 296         [1, 2, 3, 4], base=5, distance_fn=reward.absolute_distance)
 297     self.assertEqual(1, reward_manager([1]))
 298     self.assertEqual(-1, reward_manager([1, 0, 0, 0, 0, 0]))
 299     self.assertEqual(0, reward_manager([1, 2, 3, 4, 0, 0]))
 300     self.assertEqual(0, reward_manager([1, 2, 3, 4, 0]))
 301     self.assertEqual(1, reward_manager([]))
 302     self.assertEqual(0, reward_manager([]))
 303     self.assertEqual(0, reward_manager([1]))
 304     self.assertEqual(1, reward_manager([1, 2]))
 305     self.assertEqual(-1, reward_manager([1, 2, 3, 4, 0, 0]))
 306     self.assertEqual(0, reward_manager([1, 1, 1, 1, 1]))
 307     self.assertEqual(1 + 2, reward_manager([1, 2, 3, 4]))
