  25   def pretty_str(self, new_lines=True, indent=2, final_indent=0):
  26     prefix = (' ' * indent) if new_lines else ''
  27     final_prefix = (' ' * final_indent) if new_lines else ''
  28     kv = ['%s%s=%s' % (prefix, k,
  29                        (repr(v) if not isinstance(v, Config)
  30                         else v.pretty_str(new_lines=new_lines,
  31                                           indent=indent+2,
  32                                           final_indent=indent)))
  33           for k, v in self.items()]
  34     if new_lines:
  35       return 'Config(\n%s\n%s)' % (',\n'.join(kv), final_prefix)
  36     else:
  37       return 'Config(%s)' % ', '.join(kv)

  39   def _update_iterator(self, *args, **kwargs):
  40     """Convert mixed input into an iterator over (key, value) tuples.
  41 
  42     Follows the dict.update call signature.
  43 
  44     Args:
  45       *args: (Optional) Pass a dict or iterable of (key, value) 2-tuples as
  46           an unnamed argument. Only one unnamed argument allowed.
  47       **kwargs: (Optional) Pass (key, value) pairs as named arguments, where the
  48           argument name is the key and the argument value is the value.
  49 
  50     Returns:
  51       An iterator over (key, value) tuples given in the input.
  52 
  53     Raises:
  54       TypeError: If more than one unnamed argument is given.
  55     """
  56     if len(args) > 1:
  57       raise TypeError('Expected at most 1 unnamed arguments, got %d'
  58                       % len(args))
  59     obj = args[0] if args else dict()
  60     if isinstance(obj, dict):
  61       return itertools.chain(obj.items(), kwargs.items())
  62     # Assume obj is an iterable of 2-tuples.
  63     return itertools.chain(obj, kwargs.items())

  65   def make_default(self, keys=None):
  66     """Convert OneOf objects into their default configs.
  67 
  68     Recursively calls into Config objects.
  69 
  70     Args:
  71       keys: Iterable of key names to check. If None, all keys in self will be
  72           used.
  73     """
  74     if keys is None:
  75       keys = self.keys()
  76     for k in keys:
  77       # Replace OneOf with its default value.
  78       if isinstance(self[k], OneOf):
  79         self[k] = self[k].default()
  80       # Recursively call into all Config objects, even those that came from
  81       # OneOf objects in the previous code line (for nested OneOf objects).
  82       if isinstance(self[k], Config):
  83         self[k].make_default()

  85   def update(self, *args, **kwargs):
  86     """Same as dict.update except nested Config objects are updated.
  87 
  88     Args:
  89       *args: (Optional) Pass a dict or list of (key, value) 2-tuples as unnamed
  90           argument.
  91       **kwargs: (Optional) Pass (key, value) pairs as named arguments, where the
  92           argument name is the key and the argument value is the value.
  93     """
  94     key_set = set(self.keys())
  95     for k, v in self._update_iterator(*args, **kwargs):
  96       if k in key_set:
  97         key_set.remove(k)  # This key is updated so exclude from make_default.
  98       if k in self and isinstance(self[k], Config) and isinstance(v, dict):
  99         self[k].update(v)
 100       elif k in self and isinstance(self[k], OneOf) and isinstance(v, dict):
 101         # Replace OneOf with the chosen config.
 102         self[k] = self[k].update(v)
 103       else:
 104         self[k] = v
 105     self.make_default(key_set)

 107   def strict_update(self, *args, **kwargs):
 108     """Same as Config.update except keys and types are not allowed to change.
 109 
 110     If a given key is not already in this instance, an exception is raised. If a
 111     given value does not have the same type as the existing value for the same
 112     key, an exception is raised. Use this method to catch config mistakes.
 113 
 114     Args:
 115       *args: (Optional) Pass a dict or list of (key, value) 2-tuples as unnamed
 116           argument.
 117       **kwargs: (Optional) Pass (key, value) pairs as named arguments, where the
 118           argument name is the key and the argument value is the value.
 119 
 120     Raises:
 121       TypeError: If more than one unnamed argument is given.
 122       TypeError: If new value type does not match existing type.
 123       KeyError: If a given key is not already defined in this instance.
 124     """
 125     key_set = set(self.keys())
 126     for k, v in self._update_iterator(*args, **kwargs):
 127       if k in self:
 128         key_set.remove(k)  # This key is updated so exclude from make_default.
 129         if isinstance(self[k], Config):
 130           if not isinstance(v, dict):
 131             raise TypeError('dict required for Config value, got %s' % type(v))
 132           self[k].strict_update(v)
 133         elif isinstance(self[k], OneOf):
 134           if not isinstance(v, dict):
 135             raise TypeError('dict required for OneOf value, got %s' % type(v))
 136           # Replace OneOf with the chosen config.
 137           self[k] = self[k].strict_update(v)
 138         else:
 139           if not isinstance(v, type(self[k])):
 140             raise TypeError('Expecting type %s for key %s, got type %s'
 141                             % (type(self[k]), k, type(v)))
 142           self[k] = v
 143       else:
 144         raise KeyError(
 145             'Key %s does not exist. New key creation not allowed in '
 146             'strict_update.' % k)
 147     self.make_default(key_set)

 162   @staticmethod
 163   def parse(key_val_string):
 164     """Parse hyperparameter string into Config object.
 165 
 166     Format is 'key=val,key=val,...'
 167     Values can be any python literal, or another Config object encoded as
 168     'c(key=val,key=val,...)'.
 169     c(...) expressions can be arbitrarily nested.
 170 
 171     Example:
 172     'a=1,b=3e-5,c=[1,2,3],d="hello world",e={"a":1,"b":2},f=c(x=1,y=[10,20])'
 173 
 174     Args:
 175       key_val_string: The hyperparameter string.
 176 
 177     Returns:
 178       Config object parsed from the input string.
 179     """
 180     if not key_val_string.strip():
 181       return Config()
 182     def _pair_to_kv(pair):
 183       split_index = pair.find('=')
 184       key, val = pair[:split_index].strip(), pair[split_index+1:].strip()
 185       if val.startswith('c(') and val.endswith(')'):
 186         val = Config.parse(val[2:-1])
 187       else:
 188         val = ast.literal_eval(val)
 189       return key, val
 190     return Config(**dict([_pair_to_kv(pair)
 191                           for pair in _comma_iterator(key_val_string)]))

 194 class OneOf(object):
 195   """Stores branching config.
 196 
 197   In some cases there may be options which each have their own set of config
 198   params. For example, if specifying config for an environment, each environment
 199   can have custom config options. OneOf is a way to organize branching config.
 200 
 201   Usage example:
 202   one_of = OneOf(
 203       [Config(a=1, b=2),
 204        Config(a=2, c='hello'),
 205        Config(a=3, d=10, e=-10)],
 206       a=1)
 207   config = one_of.strict_update(Config(a=3, d=20))
 208   config == {'a': 3, 'd': 20, 'e': -10}
 209   """
 210 
 211   def __init__(self, choices, **kwargs):
 212     """Constructor.
 213 
 214     Usage: OneOf([Config(...), Config(...), ...], attribute=default_value)
 215 
 216     Args:
 217       choices: An iterable of Config objects. When update/strict_update is
 218           called on this OneOf, one of these Config will be selected.
 219       **kwargs: Give exactly one config attribute to branch on. The value of
 220           this attribute during update/strict_update will determine which
 221           Config is used.
 222 
 223     Raises:
 224       ValueError: If kwargs does not contain exactly one entry. Should give one
 225           named argument which is used as the attribute to condition on.
 226     """
 227     if len(kwargs) != 1:
 228       raise ValueError(
 229           'Incorrect usage. Must give exactly one named argument. The argument '
 230           'name is the config attribute to condition on, and the argument '
 231           'value is the default choice. Got %d named arguments.' % len(kwargs))
 232     key, default_value = kwargs.items()[0]
 233     self.key = key
 234     self.default_value = default_value
 235 
 236     # Make sure each choice is a Config object.
 237     for config in choices:
 238       if not isinstance(config, Config):
 239         raise TypeError('choices must be a list of Config objects. Got %s.'
 240                         % type(config))
 241 
 242     # Map value for key to the config with that value.
 243     self.value_map = {config[key]: config for config in choices}
 244     self.default_config = self.value_map[self.default_value]
 245 
 246     # Make sure there are no duplicate values.
 247     if len(self.value_map) != len(choices):
 248       raise ValueError('Multiple choices given for the same value of %s.' % key)
 249 
 250     # Check that the default value is valid.
 251     if self.default_value not in self.value_map:
 252       raise ValueError(
 253           'Default value is not an available choice. Got %s=%s. Choices are %s.'
 254           % (key, self.default_value, self.value_map.keys()))
 255 
 256   def default(self):
 257     return self.default_config
 258 
 259   def update(self, other):
 260     """Choose a config and update it.
 261 
 262     If `other` is a Config, one of the config choices is selected and updated.
 263     Otherwise `other` is returned.
 264 
 265     Args:
 266       other: Will update chosen config with this value by calling `update` on
 267           the config.
 268 
 269     Returns:
 270       The chosen config after updating it, or `other` if no config could be
 271       selected.
 272     """
 273     if not isinstance(other, Config):
 274       return other
 275     if self.key not in other or other[self.key] not in self.value_map:
 276       return other
 277     target = self.value_map[other[self.key]]
 278     target.update(other)
 279     return target
 280 
 281   def strict_update(self, config):
 282     """Choose a config and update it.
 283 
 284     `config` must be a Config object. `config` must have the key used to select
 285     among the config choices, and that key must have a value which one of the
 286     config choices has.
 287 
 288     Args:
 289       config: A Config object. the chosen config will be update by calling
 290            `strict_update`.
 291 
 292     Returns:
 293       The chosen config after updating it.
 294 
 295     Raises:
 296       TypeError: If `config` is not a Config instance.
 297       ValueError: If `config` does not have the branching key in its key set.
 298       ValueError: If the value of the config's branching key is not one of the
 299           valid choices.
 300     """
 301     if not isinstance(config, Config):
 302       raise TypeError('Expecting Config instance, got %s.' % type(config))
 303     if self.key not in config:
 304       raise ValueError(
 305           'Branching key %s required but not found in %s' % (self.key, config))
 306     if config[self.key] not in self.value_map:
 307       raise ValueError(
 308           'Value %s for key %s is not a possible choice. Choices are %s.'
 309           % (config[self.key], self.key, self.value_map.keys()))
 310     target = self.value_map[config[self.key]]
 311     target.strict_update(config)
 312     return target

 315 def _next_comma(string, start_index):
 316   """Finds the position of the next comma not used in a literal collection."""
 317   paren_count = 0
 318   for i in xrange(start_index, len(string)):
 319     c = string[i]
 320     if c == '(' or c == '[' or c == '{':
 321       paren_count += 1
 322     elif c == ')' or c == ']' or c == '}':
 323       paren_count -= 1
 324     if paren_count == 0 and c == ',':
 325       return i
 326   return -1

 329 def _comma_iterator(string):
 330   index = 0
 331   while 1:
 332     next_index = _next_comma(string, index)
 333     if next_index == -1:
 334       yield string[index:]
 335       return
 336     yield string[index:next_index]
 337     index = next_index + 1
