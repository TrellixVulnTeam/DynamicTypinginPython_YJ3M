  24 def make_record(type_name, attributes, defaults=None):
  25   """Factory for mutable record classes.
  26 
  27   A record acts just like a collections.namedtuple except slots are writable.
  28   One exception is that record classes are not equivalent to tuples or other
  29   record classes of the same length.
  30 
  31   Note, each call to `make_record` produces a unique type. Two calls will make
  32   different types even if `type_name` is the same each time.
  33 
  34   Args:
  35     type_name: Name of the record type to create.
  36     attributes: List of names of each record attribute. The order of the list
  37         is preserved.
  38     defaults: (optional) default values for attributes. A dict mapping attribute
  39         names to values.
  40 
  41   Returns:
  42     A new record type.
  43 
  44   Raises:
  45     ValueError: If,
  46         `defaults` is not a dict,
  47         `attributes` contains duplicate names,
  48         `defaults` keys are not contained in `attributes`.
  49   """
  50   if defaults is None:
  51     defaults = {}
  52   if not isinstance(defaults, dict):
  53     raise ValueError('defaults must be a dict.')
  54   attr_set = set(attributes)
  55   if len(attr_set) < len(attributes):
  56     raise ValueError('No duplicate attributes allowed.')
  57   if not set(defaults.keys()).issubset(attr_set):
  58     raise ValueError('Default attributes must be given in the attributes list.')
  59 
  60   class RecordClass(object):
  61     """A record type.
  62 
  63     Acts like mutable tuple with named slots.
  64     """
  65     __slots__ = list(attributes)
  66     _defaults = dict(defaults)
  67 
  68     def __init__(self, *args, **kwargs):
  69       if len(args) > len(self.__slots__):
  70         raise ValueError('Too many arguments. %s has length %d.'
  71                          % (type(self).__name__, len(self.__slots__)))
  72       for attr, val in self._defaults.items():
  73         setattr(self, attr, val)
  74       for i, arg in enumerate(args):
  75         setattr(self, self.__slots__[i], arg)
  76       for attr, val in kwargs.items():
  77         setattr(self, attr, val)
  78       for attr in self.__slots__:
  79         if not hasattr(self, attr):
  80           raise ValueError('Required attr "%s" is not set.' % attr)
  81 
  82     def __len__(self):
  83       return len(self.__slots__)
  84 
  85     def __iter__(self):
  86       for attr in self.__slots__:
  87         yield getattr(self, attr)
  88 
  89     def __getitem__(self, index):
  90       return getattr(self, self.__slots__[index])
  91 
  92     def __setitem__(self, index, value):
  93       return setattr(self, self.__slots__[index], value)
  94 
  95     def __eq__(self, other):
  96       # Types must be equal as well as values.
  97       return (isinstance(other, type(self))
  98               and all(a == b for a, b in zip(self, other)))
  99 
 100     def __str__(self):
 101       return '%s(%s)' % (
 102           type(self).__name__,
 103           ', '.join(attr + '=' + str(getattr(self, attr))
 104                     for attr in self.__slots__))
 105 
 106     def __repr__(self):
 107       return str(self)
 108 
 109   RecordClass.__name__ = type_name
 110   return RecordClass

 114 def stack_pad(tensors, pad_axes=None, pad_to_lengths=None, dtype=np.float32,
 115               pad_value=0):
 116   """Stack tensors along 0-th dim and pad them to be the same shape.
 117 
 118   Args:
 119     tensors: Any list of iterables (python list, numpy array, etc). Can be 1D
 120         or multi-D iterables.
 121     pad_axes: An int or list of ints. Axes to pad along.
 122     pad_to_lengths: Length in each dimension. If pad_axes was an int, this is an
 123         int or None. If pad_axes was a list of ints, this is a list of mixed int
 124         and None types with the same length, or None. A None length means the
 125         maximum length among the given tensors is used.
 126     dtype: Type of output numpy array. Defaults to np.float32.
 127     pad_value: Value to use for padding. Defaults to 0.
 128 
 129   Returns:
 130     Numpy array containing the tensors stacked along the 0-th dimension and
 131         padded along the specified dimensions.
 132 
 133   Raises:
 134     ValueError: If the tensors do not have equal shapes along non-padded
 135         dimensions.
 136   """
 137   tensors = [np.asarray(t) for t in tensors]
 138   max_lengths = [max(l) for l in zip(*[t.shape for t in tensors])]
 139   same_axes = dict(enumerate(max_lengths))
 140   if pad_axes is None:
 141     pad_axes = []
 142   if isinstance(pad_axes, six.integer_types):
 143     if pad_to_lengths is not None:
 144       max_lengths[pad_axes] = pad_to_lengths
 145     del same_axes[pad_axes]
 146   else:
 147     if pad_to_lengths is None:
 148       pad_to_lengths = [None] * len(pad_axes)
 149     for i, l in zip(pad_axes, pad_to_lengths):
 150       if l is not None:
 151         max_lengths[i] = l
 152       del same_axes[i]
 153   same_axes_items = same_axes.items()
 154   dest = np.full([len(tensors)] + max_lengths, pad_value, dtype=dtype)
 155   for i, t in enumerate(tensors):
 156     for j, l in same_axes_items:
 157       if t.shape[j] != l:
 158         raise ValueError(
 159             'Tensor at index %d does not have size %d along axis %d'
 160             % (i, l, j))
 161     dest[[i] + [slice(0, d) for d in t.shape]] = t
 162   return dest

 180 class MPQItemContainer(object):
 181   """Class for holding an item with its score.
 182 
 183   Defines a comparison function for use in the heap-queue.
 184   """
 185 
 186   def __init__(self, score, item, extra_data):
 187     self.item = item
 188     self.score = score
 189     self.extra_data = extra_data
 190 
 191   def __cmp__(self, other):
 192     assert isinstance(other, type(self))
 193     return cmp(self.score, other.score)
 194 
 195   def __iter__(self):
 196     """Allows unpacking like a tuple."""
 197     yield self.score
 198     yield self.item
 199     yield self.extra_data
 200 
 201   def __repr__(self):
 202     """String representation of this item.
 203 
 204     `extra_data` is not included in the representation. We are assuming that
 205     `extra_data` is not easily interpreted by a human (if it was, it should be
 206     hashable, like a string or tuple).
 207 
 208     Returns:
 209       String representation of `self`.
 210     """
 211     return str((self.score, self.item))
 212 
 213   def __str__(self):
 214     return repr(self)

 217 class MaxUniquePriorityQueue(object):
 218   """A maximum priority queue where duplicates are not added.
 219 
 220   The top items by score remain in the queue. When the capacity is reached,
 221   the lowest scored item in the queue will be dropped.
 222 
 223   This implementation differs from a typical priority queue, in that the minimum
 224   score is popped, instead of the maximum. Largest scores remain stuck in the
 225   queue. This is useful for accumulating the best known items from a population.
 226 
 227   The items used to determine uniqueness must be hashable, but additional
 228   non-hashable data may be stored with each item.
 229   """
 230 
 231   def __init__(self, capacity):
 232     self.capacity = capacity
 233     self.heap = []
 234     self.unique_items = set()
 235 
 236   def push(self, score, item, extra_data=None):
 237     """Push an item onto the queue.
 238 
 239     If the queue is at capacity, the item with the smallest score will be
 240     dropped. Note that it is assumed each item has exactly one score. The same
 241     item with a different score will still be dropped.
 242 
 243     Args:
 244       score: Number used to prioritize items in the queue. Largest scores are
 245           kept in the queue.
 246       item: A hashable item to be stored. Duplicates of this item will not be
 247           added to the queue.
 248       extra_data: An extra (possible not hashable) data to store with the item.
 249     """
 250     if item in self.unique_items:
 251       return
 252     if len(self.heap) >= self.capacity:
 253       _, popped_item, _ = heapq.heappushpop(
 254           self.heap, MPQItemContainer(score, item, extra_data))
 255       self.unique_items.add(item)
 256       self.unique_items.remove(popped_item)
 257     else:
 258       heapq.heappush(self.heap, MPQItemContainer(score, item, extra_data))
 259       self.unique_items.add(item)
 260 
 261   def pop(self):
 262     """Pop the item with the lowest score.
 263 
 264     Returns:
 265       score: Item's score.
 266       item: The item that was popped.
 267       extra_data: Any extra data stored with the item.
 268     """
 269     if not self.heap:
 270       return ()
 271     score, item, extra_data = heapq.heappop(self.heap)
 272     self.unique_items.remove(item)
 273     return score, item, extra_data
 274 
 275   def get_max(self):
 276     """Peek at the item with the highest score.
 277 
 278     Returns:
 279       Same as `pop`.
 280     """
 281     if not self.heap:
 282       return ()
 283     score, item, extra_data = heapq.nlargest(1, self.heap)[0]
 284     return score, item, extra_data
 285 
 286   def get_min(self):
 287     """Peek at the item with the lowest score.
 288 
 289     Returns:
 290       Same as `pop`.
 291     """
 292     if not self.heap:
 293       return ()
 294     score, item, extra_data = heapq.nsmallest(1, self.heap)[0]
 295     return score, item, extra_data
 296 
 297   def random_sample(self, sample_size):
 298     """Randomly select items from the queue.
 299 
 300     This does not modify the queue.
 301 
 302     Items are drawn from a uniform distribution, and not weighted by score.
 303 
 304     Args:
 305       sample_size: Number of random samples to draw. The same item can be
 306           sampled multiple times.
 307 
 308     Returns:
 309       List of sampled items (of length `sample_size`). Each element in the list
 310       is a tuple: (item, extra_data).
 311     """
 312     idx = np.random.choice(len(self.heap), sample_size)
 313     return [(self.heap[i].item, self.heap[i].extra_data) for i in idx]
 314 
 315   def iter_in_order(self):
 316     """Iterate over items in the queue from largest score to smallest.
 317 
 318     Yields:
 319       item: Hashable item.
 320       extra_data: Extra data stored with the item.
 321     """
 322     for _, item, extra_data in heapq.nlargest(len(self.heap), self.heap):
 323       yield item, extra_data
 324 
 325   def __len__(self):
 326     return len(self.heap)
 327 
 328   def __iter__(self):
 329     for _, item, _ in self.heap:
 330       yield item
 331 
 332   def __repr__(self):
 333     return '[' + ', '.join(repr(c) for c in self.heap) + ']'
 334 
 335   def __str__(self):
 336     return repr(self)

 339 class RouletteWheel(object):
 340   """Randomly samples stored objects proportionally to their given weights.
 341 
 342   Stores objects and weights. Acts like a roulette wheel where each object is
 343   given a slice of the roulette disk proportional to its weight.
 344 
 345   This can be used as a replay buffer where past experiences are sampled
 346   proportionally to their weights. A good choice of "weight" for reinforcement
 347   learning is exp(reward / temperature) where temperature -> inf makes the
 348   distribution more uniform and temperature -> 0 makes the distribution more
 349   peaky.
 350 
 351   To prevent experiences from being overweighted by appearing in the replay
 352   buffer multiple times, a "unique mode" is supported where duplicate
 353   experiences are ignored. In unique mode, weights can be quickly retrieved from
 354   keys.
 355   """
 356 
 357   def __init__(self, unique_mode=False, save_file=None):
 358     """Construct empty RouletteWheel.
 359 
 360     If `save_file` is not None, and the file already exists on disk, whatever
 361     is in the file will be loaded into this instance. This allows jobs using
 362     RouletteWheel to resume after preemption.
 363 
 364     Args:
 365       unique_mode: If True, puts this RouletteWheel into unique mode, where
 366           objects are added with hashable keys, so that duplicates are ignored.
 367       save_file: Optional file path to save to. Must be a string containing
 368           an absolute path to a file, or None. File will be Python pickle
 369           format.
 370     """
 371     self.unique_mode = unique_mode
 372     self.objects = []
 373     self.weights = []
 374     self.partial_sums = []
 375     if self.unique_mode:
 376       self.keys_to_weights = {}
 377     self.save_file = save_file
 378     self.save_to_disk_buffer = []
 379 
 380     if save_file is not None and tf.gfile.Exists(save_file):
 381       # Load from disk.
 382       with tf.gfile.OpenFast(save_file, 'r') as f:
 383         count = 0
 384         while 1:
 385           try:
 386             obj, weight, key = cPickle.load(f)
 387           except EOFError:
 388             break
 389           else:
 390             self.add(obj, weight, key)
 391             count += 1
 392       logging.info('Loaded %d samples from disk.', count)
 393       # Clear buffer since these items are already on disk.
 394       self.save_to_disk_buffer = []
 395 
 396   def __iter__(self):
 397     return iter(zip(self.objects, self.weights))
 398 
 399   def __len__(self):
 400     return len(self.objects)
 401 
 402   def is_empty(self):
 403     """Returns whether there is anything in the roulette wheel."""
 404     return not self.partial_sums
 405 
 406   @property
 407   def total_weight(self):
 408     """Total cumulative weight across all objects."""
 409     if self.partial_sums:
 410       return self.partial_sums[-1]
 411     return 0.0
 412 
 413   def has_key(self, key):
 414     if self.unique_mode:
 415       RuntimeError('has_key method can only be called in unique mode.')
 416     return key in self.keys_to_weights
 417 
 418   def get_weight(self, key):
 419     if self.unique_mode:
 420       RuntimeError('get_weight method can only be called in unique mode.')
 421     return self.keys_to_weights[key]
 422 
 423   def add(self, obj, weight, key=None):
 424     """Add one object and its weight to the roulette wheel.
 425 
 426     Args:
 427       obj: Any object to be stored.
 428       weight: A non-negative float. The given object will be drawn with
 429           probability proportional to this weight when sampling.
 430       key: This argument is only used when in unique mode. To allow `obj` to
 431           be an unhashable type, like list, a separate hashable key is given.
 432           Each `key` should be unique to each `obj`. `key` is used to check if
 433           `obj` has been added to the roulette wheel before.
 434 
 435     Returns:
 436       True if the object was added, False if it was not added due to it being
 437       a duplicate (this only happens in unique mode).
 438 
 439     Raises:
 440       ValueError: If `weight` is negative.
 441       ValueError: If `key` is not given when in unique mode, or if `key` is
 442           given when not in unique mode.
 443     """
 444     if weight < 0:
 445       raise ValueError('Weight must be non-negative')
 446     if self.unique_mode:
 447       if key is None:
 448         raise ValueError(
 449             'Hashable key required for objects when unique mode is enabled.')
 450       if key in self.keys_to_weights:
 451         # Weight updates are not allowed. Ignore the given value of `weight`.
 452         return False
 453       self.keys_to_weights[key] = weight
 454     elif key is not None:
 455       raise ValueError(
 456           'key argument should not be used when unique mode is disabled.')
 457     self.objects.append(obj)
 458     self.weights.append(weight)
 459     self.partial_sums.append(self.total_weight + weight)
 460     if self.save_file is not None:
 461       # Record new item in buffer.
 462       self.save_to_disk_buffer.append((obj, weight, key))
 463     return True
 464 
 465   def add_many(self, objs, weights, keys=None):
 466     """Add many object and their weights to the roulette wheel.
 467 
 468     Arguments are the same as the `add` method, except each is a list. Lists
 469     must all be the same length.
 470 
 471     Args:
 472       objs: List of objects to be stored.
 473       weights: List of non-negative floats. See `add` method.
 474       keys: List of hashable keys. This argument is only used when in unique
 475           mode. See `add` method.
 476 
 477     Returns:
 478       Number of objects added. This number will be less than the number of
 479       objects provided if we are in unique mode and some keys are already
 480       in the roulette wheel.
 481 
 482     Raises:
 483       ValueError: If `keys` argument is provided when unique_mode == False, or
 484           is not provided when unique_mode == True.
 485       ValueError: If any of the lists are not the same length.
 486       ValueError: If any of the weights are negative.
 487     """
 488     if keys is not None and not self.unique_mode:
 489       raise ValueError('Not in unique mode. Do not provide keys.')
 490     elif keys is None and self.unique_mode:
 491       raise ValueError('In unique mode. You must provide hashable keys.')
 492     if keys and len(objs) != len(keys):
 493       raise ValueError('Number of objects does not equal number of keys.')
 494     if len(objs) != len(weights):
 495       raise ValueError('Number of objects does not equal number of weights.')
 496     return sum([self.add(obj, weights[i], key=keys[i] if keys else None)
 497                 for i, obj in enumerate(objs)])
 498 
 499   def sample(self):
 500     """Spin the roulette wheel.
 501 
 502     Randomly select an object with probability proportional to its weight.
 503 
 504     Returns:
 505       object: The selected object.
 506       weight: The weight of the selected object.
 507 
 508     Raises:
 509       RuntimeError: If the roulette wheel is empty.
 510     """
 511     if self.is_empty():
 512       raise RuntimeError('Trying to sample from empty roulette wheel.')
 513     spin = random.random() * self.total_weight
 514 
 515     # Binary search.
 516     i = bisect.bisect_right(self.partial_sums, spin)
 517     if i == len(self.partial_sums):
 518       # This should not happen since random.random() will always be strictly
 519       # less than 1.0, and the last partial sum equals self.total_weight().
 520       # However it may happen due to rounding error. In that case it is easy to
 521       # handle this, just select the last object.
 522       i -= 1
 523 
 524     return self.objects[i], self.weights[i]
 525 
 526   def sample_many(self, count):
 527     """Spin the roulette wheel `count` times and return the results."""
 528     if self.is_empty():
 529       raise RuntimeError('Trying to sample from empty roulette wheel.')
 530     return [self.sample() for _ in xrange(count)]

 532   def incremental_save(self, log_info=False):
 533     """Write new entries to disk.
 534 
 535     This performs an append operation on the `save_file` given in the
 536     constructor. Any entries added since the last call to `incremental_save`
 537     will be appended to the file.
 538 
 539     If a new RouletteWheel is constructed with the same `save_file`, all the
 540     entries written there will be automatically loaded into the instance.
 541     This is useful when a job resumes after preemption.
 542 
 543     Args:
 544       log_info: If True, info about this operation will be logged.
 545 
 546     Raises:
 547       RuntimeError: If `save_file` given in the constructor is None.
 548     """
 549     if self.save_file is None:
 550       raise RuntimeError('Cannot call incremental_save. `save_file` is None.')
 551     if log_info:
 552       logging.info('Saving %d new samples to disk.',
 553                    len(self.save_to_disk_buffer))
 554     with tf.gfile.OpenFast(self.save_file, 'a') as f:
 555       for entry in self.save_to_disk_buffer:
 556         cPickle.dump(entry, f)
 557     # Clear the buffer.
 558     self.save_to_disk_buffer = []
