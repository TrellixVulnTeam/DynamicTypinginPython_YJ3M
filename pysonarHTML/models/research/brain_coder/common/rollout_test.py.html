  22   def testDiscount(self):
  23     discounted = np.array([1.0 / 2 ** n for n in range(4, -1, -1)])
  24     discounted[:2] += [1.0 / 2 ** n for n in range(1, -1, -1)]
  25 
  26     self.assertTrue(np.array_equal(
  27         rollout_lib.discount([0.0, 1.0, 0.0, 0.0, 1.0], 0.50),
  28         discounted))
  29     self.assertTrue(np.array_equal(
  30         rollout_lib.discount(np.array([0.0, 1.0, 0.0, 0.0, 1.0]), 0.50),
  31         discounted))

  33   def testDiscountedAdvantageAndRewards(self):
  34     # lambda=1, No bootstrapping.
  35     values = [0.1, 0.5, 0.5, 0.25]
  36     (empirical_values,
  37      generalized_advantage) = rollout_lib.discounted_advantage_and_rewards(
  38          [0.0, 0.0, 0.0, 1.0],
  39          values,
  40          gamma=0.75,
  41          lambda_=1.0)
  42     expected_discounted_r = (
  43         np.array([1.0 * 0.75 ** n for n in range(3, -1, -1)]))
  44     expected_adv = expected_discounted_r - values
  45     self.assertTrue(np.array_equal(empirical_values, expected_discounted_r))
  46     self.assertTrue(np.allclose(generalized_advantage, expected_adv))
  47 
  48     # lambda=1, With bootstrapping.
  49     values = [0.1, 0.5, 0.5, 0.25, 0.75]
  50     (empirical_values,
  51      generalized_advantage) = rollout_lib.discounted_advantage_and_rewards(
  52          [0.0, 0.0, 0.0, 1.0],
  53          values,
  54          gamma=0.75,
  55          lambda_=1.0)
  56     expected_discounted_r = (
  57         np.array([0.75 * 0.75 ** n for n in range(4, 0, -1)])
  58         + np.array([1.0 * 0.75 ** n for n in range(3, -1, -1)]))
  59     expected_adv = expected_discounted_r - values[:-1]
  60     self.assertTrue(np.array_equal(empirical_values, expected_discounted_r))
  61     self.assertTrue(np.allclose(generalized_advantage, expected_adv))
  62 
  63     # lambda=0.5, With bootstrapping.
  64     values = [0.1, 0.5, 0.5, 0.25, 0.75]
  65     rewards = [0.0, 0.0, 0.0, 1.0]
  66     l = 0.5  # lambda
  67     g = 0.75  # gamma
  68     (empirical_values,
  69      generalized_advantage) = rollout_lib.discounted_advantage_and_rewards(
  70          rewards,
  71          values,
  72          gamma=g,
  73          lambda_=l)
  74     expected_discounted_r = (
  75         np.array([0.75 * g ** n for n in range(4, 0, -1)])
  76         + np.array([1.0 * g ** n for n in range(3, -1, -1)]))
  77     expected_adv = [0.0] * len(values)
  78     for t in range(3, -1, -1):
  79       delta_t = rewards[t] + g * values[t + 1] - values[t]
  80       expected_adv[t] = delta_t + g * l * expected_adv[t + 1]
  81     expected_adv = expected_adv[:-1]
  82     self.assertTrue(np.array_equal(empirical_values, expected_discounted_r))
  83     self.assertTrue(np.allclose(generalized_advantage, expected_adv))

  85   def testProcessRollouts(self):
  86     g = 0.95
  87     rollouts = [
  88         self.MakeRollout(
  89             states=[3, 6, 9],
  90             actions=[1, 2, 3],
  91             rewards=[1.0, -1.0, 0.5],
  92             values=[0.5, 0.5, 0.1]),
  93         self.MakeRollout(
  94             states=[10],
  95             actions=[5],
  96             rewards=[1.0],
  97             values=[0.5])]
  98     batch = rollout_lib.process_rollouts(rollouts, gamma=g)
  99 
 100     self.assertEqual(2, batch.batch_size)
 101     self.assertEqual(3, batch.max_time)
 102     self.assertEqual([3, 1], batch.episode_lengths)
 103     self.assertEqual([0.5, 1.0], batch.total_rewards)
 104     self.assertEqual(
 105         [[3, 6, 9], [10, 0, 0]],
 106         batch.states.tolist())
 107     self.assertEqual(
 108         [[1, 2, 3], [5, 0, 0]],
 109         batch.actions.tolist())
 110 
 111     rew1, rew2 = rollouts[0].rewards, rollouts[1].rewards
 112     expected_discounted_rewards = [
 113         [rew1[0] + g * rew1[1] + g * g * rew1[2],
 114          rew1[1] + g * rew1[2],
 115          rew1[2]],
 116         [rew2[0], 0.0, 0.0]]
 117     expected_advantages = [
 118         [dr - v
 119          for dr, v
 120          in zip(expected_discounted_rewards[0], rollouts[0].values)],
 121         [expected_discounted_rewards[1][0] - rollouts[1].values[0], 0.0, 0.0]]
 122     self.assertTrue(
 123         np.allclose(expected_discounted_rewards, batch.discounted_r))
 124     self.assertTrue(
 125         np.allclose(expected_advantages, batch.discounted_adv))
