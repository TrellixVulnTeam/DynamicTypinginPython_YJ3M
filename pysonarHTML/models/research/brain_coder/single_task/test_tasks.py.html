  12 class BasicTaskManager(object):
  13   """Wraps a generic reward function."""
  14 
  15   def __init__(self, reward_fn):
  16     self.reward_fn = reward_fn
  17     self.good_reward = 1.0
  18 
  19   def _score_string(self, string):
  20     actions = misc.bf_string_to_tokens(string)
  21     reward, correct = self.reward_fn(actions)
  22     return misc.RewardInfo(
  23         episode_rewards=[0.0] * (len(string) - 1) + [reward],
  24         input_case=None,
  25         correct_output=None,
  26         code_output=actions,
  27         input_type=None,
  28         output_type=misc.IOType.integer,
  29         reason='correct' if correct else 'wrong')
  30 
  31   def rl_batch(self, batch_size):
  32     reward_fns = [self._score_string] * batch_size
  33     return reward_fns

  36 class Trie(object):
  37   """Trie for sequences."""
  38   EOS = ()
  39 
  40   def __init__(self):
  41     self.trie = {}
  42 
  43   def insert(self, sequence):
  44     d = self.trie
  45     for e in sequence:
  46       if e not in d:
  47         d[e] = {}
  48       d = d[e]
  49     d[self.EOS] = True   # Terminate sequence.
  50 
  51   def prefix_match(self, sequence):
  52     """Return prefix of `sequence` which exists in the trie."""
  53     d = self.trie
  54     index = 0
  55     for i, e in enumerate(sequence + [self.EOS]):
  56       index = i
  57       if e in d:
  58         d = d[e]
  59         if e == self.EOS:
  60           return sequence, True
  61       else:
  62         break
  63     return sequence[:index], False
  64 
  65   def next_choices(self, sequence):
  66     d = self.trie
  67     for e in sequence:
  68       if e in d:
  69         d = d[e]
  70       else:
  71         raise ValueError('Sequence not a prefix: %s' % (sequence,))
  72     return d.keys()

  75 class HillClimbingTask(object):
  76   """Simple task that tests reward hill climbing ability.
  77 
  78   There are a set of paths (sequences of tokens) which are rewarded. The total
  79   reward for a path is proportional to its length, so the longest path is the
  80   target. Shorter paths can be dead ends.
  81   """
  82 
  83   def __init__(self):
  84     # Paths are sequences of sub-sequences. Here we form unique sub-sequences
  85     # out of 3 arbitrary ints. We use sub-sequences instead of single entities
  86     # to make the task harder by making the episodes last longer, i.e. more
  87     # for the agent to remember.
  88     a = (1, 2, 3)
  89     b = (4, 5, 6)
  90     c = (7, 8, 7)
  91     d = (6, 5, 4)
  92     e = (3, 2, 1)
  93     f = (8, 5, 1)
  94     g = (6, 4, 2)
  95     h = (1, 8, 3)
  96     self.paths = Trie()
  97     self.paths.insert([a, b, h])
  98     self.paths.insert([a, b, c, d, e, f, g, h])
  99     self.paths.insert([a, b, c, d, e, b, a])
 100     self.paths.insert([a, b, g, h])
 101     self.paths.insert([a, e, f, g])
 102     self.correct_sequence = misc.flatten([a, b, c, d, e, f, g, h])
 103 
 104     def distance_fn(a, b):
 105       len_diff = abs(len(a) - len(b))
 106       return sum(reward_lib.mod_abs_diff(ai - 1, bi - 1, 8)
 107                  for ai, bi in zip(a, b)) + len_diff * 4  # 8 / 2 = 4
 108     self.distance_fn = distance_fn
 109 
 110   def __call__(self, actions):
 111     # Compute reward for action sequence.
 112     actions = [a for a in actions if a > 0]
 113     sequence = [tuple(actions[i: i + 3]) for i in xrange(0, len(actions), 3)]
 114     prefix, complete = self.paths.prefix_match(sequence)
 115     if complete:
 116       return float(len(prefix)), actions == self.correct_sequence
 117     if len(prefix) == len(sequence):
 118       return float(len(prefix)), False
 119     next_pred = sequence[len(prefix)]
 120     choices = self.paths.next_choices(prefix)
 121     if choices == [()]:
 122       return (len(prefix) - len(next_pred) / 3.0), False
 123     min_dist = min(self.distance_fn(c, next_pred) for c in choices)
 124     # +1 reward for each element in the sequence correct, plus fraction torwards
 125     # closest next element.
 126     # Maximum distance possible is num_actions * base / 2 = 3 * 8 / 2 = 12
 127     return (len(prefix) + (1 - min_dist / 12.0)), False
