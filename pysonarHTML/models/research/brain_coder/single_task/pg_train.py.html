  60 def define_tuner_hparam_space(hparam_space_type):
  61   """Define tunable hparams for grid search."""
  62   if hparam_space_type not in ('pg', 'pg-topk', 'topk', 'is'):
  63     raise ValueError('Hparam space is not valid: "%s"' % hparam_space_type)
  64 
  65   # Discrete hparam space is stored as a dict from hparam name to discrete
  66   # values.
  67   hparam_space = {}
  68 
  69   if hparam_space_type in ('pg', 'pg-topk', 'is'):
  70     # Add a floating point parameter named learning rate.
  71     hparam_space['lr'] = [1e-5, 1e-4, 1e-3]
  72     hparam_space['entropy_beta'] = [0.005, 0.01, 0.05, 0.10]
  73   else:  # 'topk'
  74     # Add a floating point parameter named learning rate.
  75     hparam_space['lr'] = [1e-5, 1e-4, 1e-3]
  76     hparam_space['entropy_beta'] = [0.0, 0.005, 0.01, 0.05, 0.10]
  77 
  78   if hparam_space_type in ('topk', 'pg-topk'):
  79     # topk tuning will be enabled.
  80     hparam_space['topk'] = [10]
  81     hparam_space['topk_loss_hparam'] = [1.0, 10.0, 50.0, 200.0]
  82 
  83   elif hparam_space_type == 'is':
  84     # importance sampling tuning will be enabled.
  85     hparam_space['replay_temperature'] = [0.25, 0.5, 1.0, 2.0]
  86     hparam_space['alpha'] = [0.5, 0.75, 63/64.]
  87 
  88   return hparam_space

  98

 104     # importance sampling tuning will be enabled.
 105     config.agent.replay_temperature = hparams.replay_temperature
 106     config.agent.alpha = hparams.alpha

 109 def make_initialized_variable(value, name, shape=None, dtype=tf.float32):
 110   """Create a tf.Variable with a constant initializer.
 111 
 112   Args:
 113     value: Constant value to initialize the variable with. This is the value
 114         that the variable starts with.
 115     name: Name of the variable in the TF graph.
 116     shape: Shape of the variable. If None, variable will be a scalar.
 117     dtype: Data type of the variable. Should be a TF dtype. Defaults to
 118         tf.float32.
 119 
 120   Returns:
 121     tf.Variable instance.
 122   """
 123   if shape is None:
 124     shape = []
 125   return tf.get_variable(
 126       name=name, shape=shape, initializer=tf.constant_initializer(value),
 127       dtype=dtype, trainable=False)

 130 class AsyncTrainer(object):
 131   """Manages graph creation and training.
 132 
 133   This async trainer creates a global model on the parameter server, and a local
 134   model (for this worker). Gradient updates are sent to the global model, and
 135   the updated weights are synced to the local copy.
 136   """
 137 
 138   def __init__(self, config, task_id, ps_tasks, num_workers, is_chief=True,
 139                summary_writer=None,
 140                dtype=tf.float32,
 141                summary_interval=1,
 142                run_number=0,
 143                logging_dir='/tmp', model_v=0):
 144     self.config = config
 145     self.data_manager = data.DataManager(
 146         config, run_number=run_number,
 147         do_code_simplification=not FLAGS.stop_on_success)
 148     self.task_id = task_id
 149     self.ps_tasks = ps_tasks
 150     self.is_chief = is_chief
 151     if ps_tasks == 0:
 152       assert task_id == 0, 'No parameter servers specified. Expecting 1 task.'
 153       assert num_workers == 1, (
 154           'No parameter servers specified. Expecting 1 task.')
 155       worker_device = '/job:localhost/replica:%d/task:0/cpu:0' % task_id
 156       # worker_device = '/cpu:0'
 157       # ps_device = '/cpu:0'
 158     else:
 159       assert num_workers > 0, 'There must be at least 1 training worker.'
 160       worker_device = '/job:worker/replica:%d/task:0/cpu:0' % task_id
 161       # ps_device = '/job:ps/replica:0/task:0/cpu:0'
 162     logging.info('worker_device: %s', worker_device)
 163 
 164     logging_file = os.path.join(
 165         logging_dir, 'solutions_%d.txt' % task_id)
 166     experience_replay_file = os.path.join(
 167         logging_dir, 'replay_buffer_%d.pickle' % task_id)
 168     self.topk_file = os.path.join(
 169         logging_dir, 'topk_buffer_%d.pickle' % task_id)
 170 
 171     tf.get_variable_scope().set_use_resource(True)
 172 
 173     # global model
 174     with tf.device(tf.train.replica_device_setter(ps_tasks,
 175                                                   ps_device='/job:ps/replica:0',
 176                                                   worker_device=worker_device)):
 177       with tf.variable_scope('global'):
 178         global_model = agent_lib.LMAgent(config, dtype=dtype, is_local=False)
 179         global_params_dict = {p.name: p
 180                               for p in global_model.sync_variables}
 181         self.global_model = global_model
 182         self.global_step = make_initialized_variable(
 183             0, 'global_step', dtype=tf.int64)
 184 
 185         self.global_best_reward = make_initialized_variable(
 186             -10.0, 'global_best_reward', dtype=tf.float64)
 187         self.is_best_model = make_initialized_variable(
 188             False, 'is_best_model', dtype=tf.bool)
 189         self.reset_is_best_model = self.is_best_model.assign(False)
 190         self.global_best_reward_placeholder = tf.placeholder(
 191             tf.float64, [], name='global_best_reward_placeholder')
 192         self.assign_global_best_reward_op = tf.group(
 193             self.global_best_reward.assign(
 194                 self.global_best_reward_placeholder),
 195             self.is_best_model.assign(True))
 196         def assign_global_best_reward_fn(session, reward):
 197           reward = round(reward, 10)
 198           best_reward = round(session.run(self.global_best_reward), 10)
 199           is_best = reward > best_reward
 200           if is_best:
 201             session.run(self.assign_global_best_reward_op,
 202                         {self.global_best_reward_placeholder: reward})
 203           return is_best
 204         self.assign_global_best_reward_fn = assign_global_best_reward_fn
 205 
 206         # Any worker will set to true when it finds a solution.
 207         self.found_solution_flag = make_initialized_variable(
 208             False, 'found_solution_flag', dtype=tf.bool)
 209         self.found_solution_op = self.found_solution_flag.assign(True)
 210 
 211         self.run_number = make_initialized_variable(
 212             run_number, 'run_number', dtype=tf.int32)
 213 
 214         # Store a solution when found.
 215         self.code_solution_variable = tf.get_variable(
 216             'code_solution', [], tf.string,
 217             initializer=tf.constant_initializer(''))
 218         self.code_solution_ph = tf.placeholder(
 219             tf.string, [], name='code_solution_ph')
 220         self.code_solution_assign_op = self.code_solution_variable.assign(
 221             self.code_solution_ph)
 222         def assign_code_solution_fn(session, code_solution_string):
 223           session.run(self.code_solution_assign_op,
 224                       {self.code_solution_ph: code_solution_string})
 225         self.assign_code_solution_fn = assign_code_solution_fn
 226 
 227         # Count all programs sampled from policy. This does not include
 228         # programs sampled from replay buffer.
 229         # This equals NPE (number of programs executed). Only programs sampled
 230         # from the policy need to be executed.
 231         self.program_count = make_initialized_variable(
 232             0, 'program_count', dtype=tf.int64)
 233 
 234     # local model
 235     with tf.device(worker_device):
 236       with tf.variable_scope('local'):
 237         self.model = model = agent_lib.LMAgent(
 238             config,
 239             task_id=task_id,
 240             logging_file=logging_file,
 241             experience_replay_file=experience_replay_file,
 242             dtype=dtype,
 243             global_best_reward_fn=self.assign_global_best_reward_fn,
 244             found_solution_op=self.found_solution_op,
 245             assign_code_solution_fn=self.assign_code_solution_fn,
 246             program_count=self.program_count,
 247             stop_on_success=FLAGS.stop_on_success,
 248             verbose_level=model_v)
 249         local_params = model.trainable_variables
 250         local_params_dict = {p.name: p for p in local_params}
 251 
 252     # Pull global params to local model.
 253     def _global_to_local_scope(name):
 254       assert name.startswith('global/')
 255       return 'local' + name[6:]
 256     sync_dict = {
 257         local_params_dict[_global_to_local_scope(p_name)]: p
 258         for p_name, p in global_params_dict.items()}
 259     self.sync_op = tf.group(*[v_local.assign(v_global)
 260                               for v_local, v_global
 261                               in sync_dict.items()])
 262 
 263     # Pair local gradients with global params.
 264     grad_var_dict = {
 265         gradient: sync_dict[local_var]
 266         for local_var, gradient in model.gradients_dict.items()}
 267 
 268     # local model
 269     model.make_summary_ops()  # Don't put summaries under 'local' scope.
 270     with tf.variable_scope('local'):
 271       self.train_op = model.optimizer.apply_gradients(
 272           grad_var_dict.items(), global_step=self.global_step)
 273       self.local_init_op = tf.variables_initializer(
 274           tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES,
 275                             tf.get_variable_scope().name))
 276 
 277     self.local_step = 0
 278     self.last_summary_time = time.time()
 279     self.summary_interval = summary_interval
 280     self.summary_writer = summary_writer
 281     self.cached_global_step = -1
 282     self.cached_global_npe = -1
 283 
 284     logging.info('summary_interval: %d', self.summary_interval)
 285 
 286     # Load top-k buffer.
 287     if self.model.top_episodes is not None and tf.gfile.Exists(self.topk_file):
 288       try:
 289         with tf.gfile.FastGFile(self.topk_file, 'r') as f:
 290           self.model.top_episodes = cPickle.loads(f.read())
 291         logging.info(
 292             'Loaded top-k buffer from disk with %d items. Location: "%s"',
 293             len(self.model.top_episodes), self.topk_file)
 294       except (cPickle.UnpicklingError, EOFError) as e:
 295         logging.warn(
 296             'Failed to load existing top-k buffer from disk. Removing bad file.'
 297             '\nLocation: "%s"\nException: %s', self.topk_file, str(e))
 298         tf.gfile.Remove(self.topk_file)
 299 
 300   def initialize(self, session):
 301     """Run initialization ops."""
 302     session.run(self.local_init_op)
 303     session.run(self.sync_op)
 304     self.cached_global_step, self.cached_global_npe = session.run(
 305         [self.global_step, self.program_count])

 307   def update_global_model(self, session):
 308     """Run an update step.
 309 
 310     1) Asynchronously copy global weights to local model.
 311     2) Call into local model's update_step method, which does the following:
 312         a) Sample batch of programs from policy.
 313         b) Compute rewards.
 314         c) Compute gradients and update the global model asynchronously.
 315     3) Write tensorboard summaries to disk.
 316 
 317     Args:
 318       session: tf.Session instance.
 319     """
 320     session.run(self.sync_op)  # Copy weights from global to local.
 321 
 322     with session.as_default():
 323       result = self.model.update_step(
 324           session, self.data_manager.sample_rl_batch(), self.train_op,
 325           self.global_step)
 326       global_step = result.global_step
 327       global_npe = result.global_npe
 328       summaries = result.summaries_list
 329     self.cached_global_step = global_step
 330     self.cached_global_npe = global_npe
 331     self.local_step += 1
 332 
 333     if self.summary_writer and self.local_step % self.summary_interval == 0:
 334       if not isinstance(summaries, (tuple, list)):
 335         summaries = [summaries]
 336       summaries.append(self._local_step_summary())
 337       if self.is_chief:
 338         (global_best_reward,
 339          found_solution_flag,
 340          program_count) = session.run(
 341              [self.global_best_reward,
 342               self.found_solution_flag,
 343               self.program_count])
 344         summaries.append(
 345             tf.Summary(
 346                 value=[tf.Summary.Value(
 347                     tag='model/best_reward',
 348                     simple_value=global_best_reward)]))
 349         summaries.append(
 350             tf.Summary(
 351                 value=[tf.Summary.Value(
 352                     tag='model/solution_found',
 353                     simple_value=int(found_solution_flag))]))
 354         summaries.append(
 355             tf.Summary(
 356                 value=[tf.Summary.Value(
 357                     tag='model/program_count',
 358                     simple_value=program_count)]))
 359       for s in summaries:
 360         self.summary_writer.add_summary(s, global_step)
 361       self.last_summary_time = time.time()
 362 
 363   def _local_step_summary(self):
 364     """Compute number of local steps per time increment."""
 365     dt = time.time() - self.last_summary_time
 366     steps_per_time = self.summary_interval / float(dt)
 367     return tf.Summary(value=[
 368         tf.Summary.Value(
 369             tag='local_step/per_sec',
 370             simple_value=steps_per_time),
 371         tf.Summary.Value(
 372             tag='local_step/step',
 373             simple_value=self.local_step)])

 375   def maybe_save_best_model(self, session, saver, checkpoint_file):
 376     """Check if this model got the highest reward and save to disk if so."""
 377     if self.is_chief and session.run(self.is_best_model):
 378       logging.info('Saving best model to "%s"', checkpoint_file)
 379       saver.save(session, checkpoint_file)
 380       session.run(self.reset_is_best_model)

 403   def save_topk_buffer(self):
 404     """Save top-k buffer to disk.
 405

 415 @contextlib.contextmanager
 416 def managed_session(sv, master='', config=None,
 417                     start_standard_services=True,
 418                     close_summary_writer=True,
 419                     max_wait_secs=7200):
 420   # Same as Supervisor.managed_session, but with configurable timeout.
 421   try:
 422     sess = sv.prepare_or_wait_for_session(
 423         master=master, config=config,
 424         start_standard_services=start_standard_services,
 425         max_wait_secs=max_wait_secs)
 426     yield sess
 427   except tf.errors.DeadlineExceededError:
 428     raise
 429   except Exception as e:  # pylint: disable=broad-except
 430     sv.request_stop(e)
 431   finally:
 432     try:
 433       # Request all the threads to stop and wait for them to do so.  Any
 434       # exception raised by the threads is raised again from stop().
 435       # Passing stop_grace_period_secs is for blocked enqueue/dequeue
 436       # threads which are not checking for `should_stop()`.  They
 437       # will be stopped when we close the session further down.
 438       sv.stop(close_summary_writer=close_summary_writer)
 439     finally:
 440       # Close the session to finish up all pending calls.  We do not care
 441       # about exceptions raised when closing.  This takes care of
 442       # blocked enqueue/dequeue calls.
 443       try:
 444         sess.close()
 445       except Exception:  # pylint: disable=broad-except
 446         # Silently ignore exceptions raised by close().
 447         pass

 450 def train(config, is_chief, tuner=None, run_dir=None, run_number=0,
 451           results_writer=None):
 452   """Run training loop.
 453 
 454   Args:
 455     config: config_lib.Config instance containing global config (agent and env).
 456     is_chief: True if this worker is chief. Chief worker manages writing some
 457         data to disk and initialization of the global model.
 458     tuner: A tuner instance. If not tuning, leave as None.
 459     run_dir: Directory where all data for this run will be written. If None,
 460         run_dir = FLAGS.logdir. Set this argument when doing multiple runs.
 461     run_number: Which run is this.
 462     results_writer: Managest writing training results to disk. Results are a
 463         dict of metric names and values.
 464 
 465   Returns:
 466     The trainer object used to run training updates.
 467   """
 468   logging.info('Will run asynchronous training.')
 469 
 470   if run_dir is None:
 471     run_dir = FLAGS.logdir
 472   train_dir = os.path.join(run_dir, 'train')
 473   best_model_checkpoint = os.path.join(train_dir, 'best.ckpt')
 474   events_dir = '%s/events_%d' % (run_dir, FLAGS.task_id)
 475   logging.info('Events directory: %s', events_dir)
 476 
 477   logging_dir = os.path.join(run_dir, 'logs')
 478   if not tf.gfile.Exists(logging_dir):
 479     tf.gfile.MakeDirs(logging_dir)
 480   status_file = os.path.join(logging_dir, 'status.txt')
 481 
 482   if FLAGS.summary_tasks and FLAGS.task_id < FLAGS.summary_tasks:
 483     summary_writer = tf.summary.FileWriter(events_dir)
 484   else:
 485     summary_writer = None
 486 
 487   # Only profile task 0.
 488   if FLAGS.do_profiling:
 489     logging.info('Profiling enabled')
 490     profiler = cProfile.Profile()
 491     profiler.enable()
 492   else:
 493     profiler = None
 494 
 495   trainer = AsyncTrainer(
 496       config, FLAGS.task_id, FLAGS.ps_tasks, FLAGS.num_workers,
 497       is_chief=is_chief,
 498       summary_interval=FLAGS.summary_interval,
 499       summary_writer=summary_writer,
 500       logging_dir=logging_dir,
 501       run_number=run_number,
 502       model_v=FLAGS.model_v)
 503 
 504   variables_to_save = [v for v in tf.global_variables()
 505                        if v.name.startswith('global')]
 506   global_init_op = tf.variables_initializer(variables_to_save)
 507   saver = tf.train.Saver(variables_to_save)
 508 
 509   var_list = tf.get_collection(tf.GraphKeys.TRAINABLE_VARIABLES,
 510                                tf.get_variable_scope().name)
 511   logging.info('Trainable vars:')
 512   for v in var_list:
 513     logging.info('  %s, %s, %s', v.name, v.device, v.get_shape())
 514 
 515   logging.info('All vars:')
 516   for v in tf.global_variables():
 517     logging.info('  %s, %s, %s', v.name, v.device, v.get_shape())
 518 
 519   def init_fn(unused_sess):
 520     logging.info('No checkpoint found. Initialized global params.')
 521 
 522   sv = tf.train.Supervisor(is_chief=is_chief,
 523                            logdir=train_dir,
 524                            saver=saver,
 525                            summary_op=None,
 526                            init_op=global_init_op,
 527                            init_fn=init_fn,
 528                            summary_writer=summary_writer,
 529                            ready_op=tf.report_uninitialized_variables(
 530                                variables_to_save),
 531                            ready_for_local_init_op=None,
 532                            global_step=trainer.global_step,
 533                            save_model_secs=30,
 534                            save_summaries_secs=30)
 535 
 536   # Add a thread that periodically checks if this Trial should stop
 537   # based on an early stopping policy.
 538   if tuner:
 539     sv.Loop(60, tuner.check_for_stop, (sv.coord,))
 540 
 541   last_replay_save_time = time.time()
 542 
 543   global_step = -1
 544   logging.info(
 545       'Starting session. '
 546       'If this hangs, we\'re mostly likely waiting to connect '
 547       'to the parameter server. One common cause is that the parameter '
 548       'server DNS name isn\'t resolving yet, or is misspecified.')
 549   should_retry = True
 550   supervisor_deadline_exceeded = False
 551   while should_retry:
 552     try:
 553       with managed_session(
 554           sv, FLAGS.master, max_wait_secs=60) as session, session.as_default():
 555         should_retry = False
 556         do_training = True
 557 
 558         try:
 559           trainer.initialize(session)
 560           if session.run(trainer.run_number) != run_number:
 561             # If we loaded existing model from disk, and the saved run number is
 562             # different, throw an exception.
 563             raise RuntimeError(
 564                 'Expecting to be on run %d, but is actually on run %d. '
 565                 'run_dir: "%s"'
 566                 % (run_number, session.run(trainer.run_number), run_dir))
 567           global_step = trainer.cached_global_step
 568           logging.info('Starting training at step=%d', global_step)
 569           while do_training:
 570             trainer.update_global_model(session)
 571 
 572             if is_chief:
 573               trainer.maybe_save_best_model(
 574                   session, saver, best_model_checkpoint)
 575             global_step = trainer.cached_global_step
 576             global_npe = trainer.cached_global_npe
 577 
 578             if time.time() - last_replay_save_time >= 30:
 579               trainer.save_replay_buffer()
 580               trainer.save_topk_buffer()
 581               last_replay_save_time = time.time()
 582 
 583             # Stopping conditions.
 584             if tuner and tuner.should_trial_stop():
 585               logging.info('Tuner requested early stopping. Finishing.')
 586               do_training = False
 587             if is_chief and FLAGS.stop_on_success:
 588               found_solution = session.run(trainer.found_solution_flag)
 589               if found_solution:
 590                 do_training = False
 591                 logging.info('Solution found. Finishing.')
 592             if FLAGS.max_npe and global_npe >= FLAGS.max_npe:
 593               # Max NPE (number of programs executed) reached.
 594               logging.info('Max NPE reached. Finishing.')
 595               do_training = False
 596             if sv.should_stop():
 597               logging.info('Supervisor issued stop. Finishing.')
 598               do_training = False
 599 
 600         except tf.errors.NotFoundError:
 601           # Catch "Error while reading resource variable".
 602           # The chief worker likely destroyed the container, so do not retry.
 603           logging.info('Caught NotFoundError. Quitting.')
 604           do_training = False
 605           should_retry = False
 606           break
 607         except tf.errors.InternalError as e:
 608           # Catch "Invalid variable reference."
 609           if str(e).startswith('Invalid variable reference.'):
 610             # The chief worker likely destroyed the container, so do not
 611             # retry.
 612             logging.info(
 613                 'Caught "InternalError: Invalid variable reference.". '
 614                 'Quitting.')
 615             do_training = False
 616             should_retry = False
 617             break
 618           else:
 619             # Pass exception through.
 620             raise
 621 
 622         # Exited training loop. Write results to disk.
 623         if is_chief and results_writer:
 624           assert not should_retry
 625           with tf.gfile.FastGFile(status_file, 'w') as f:
 626             f.write('done')
 627           (program_count,
 628            found_solution,
 629            code_solution,
 630            best_reward,
 631            global_step) = session.run(
 632                [trainer.program_count,
 633                 trainer.found_solution_flag,
 634                 trainer.code_solution_variable,
 635                 trainer.global_best_reward,
 636                 trainer.global_step])
 637           results_dict = {
 638               'max_npe': FLAGS.max_npe,
 639               'batch_size': config.batch_size,
 640               'max_batches': FLAGS.max_npe // config.batch_size,
 641               'npe': program_count,
 642               'max_global_repetitions': FLAGS.num_repetitions,
 643               'max_local_repetitions': FLAGS.num_repetitions,
 644               'code_solution': code_solution,
 645               'best_reward': best_reward,
 646               'num_batches': global_step,
 647               'found_solution': found_solution,
 648               'task': trainer.data_manager.task_name,
 649               'global_rep': run_number}
 650           logging.info('results_dict: %s', results_dict)
 651           results_writer.append(results_dict)
 652 
 653     except tf.errors.AbortedError:
 654       # Catch "Graph handle is not found" error due to preempted jobs.
 655       logging.info('Caught AbortedError. Retying.')
 656       should_retry = True
 657     except tf.errors.DeadlineExceededError:
 658       supervisor_deadline_exceeded = True
 659       should_retry = False
 660 
 661   if is_chief:
 662     logging.info('This is chief worker. Stopping all workers.')
 663     sv.stop()
 664 
 665   if supervisor_deadline_exceeded:
 666     logging.info('Supervisor timed out. Quitting.')
 667   else:
 668     logging.info('Reached %s steps. Worker stopped.', global_step)
 669 
 670   # Dump profiling.
 671   """
 672   How to use profiling data.
 673 
 674   Download the profiler dump to your local machine, say to PROF_FILE_PATH.
 675   In a separate script, run something like the following:
 676 
 677   import pstats
 678   p = pstats.Stats(PROF_FILE_PATH)
 679   p.strip_dirs().sort_stats('cumtime').print_stats()
 680 
 681   This will sort by 'cumtime', which "is the cumulative time spent in this and
 682   all subfunctions (from invocation till exit)."
 683   https://docs.python.org/2/library/profile.html#instant-user-s-manual
 684   """  # pylint: disable=pointless-string-statement
 685   if profiler:
 686     prof_file = os.path.join(run_dir, 'task_%d.prof' % FLAGS.task_id)
 687     logging.info('Done profiling.\nDumping to "%s".', prof_file)
 688     profiler.create_stats()
 689     with tf.gfile.Open(prof_file, 'w') as f:
 690       f.write(marshal.dumps(profiler.stats))
 691 
 692   return trainer
 693 
 694 
 695 def run_training(config=None, tuner=None, logdir=None, trial_name=None,
 696                  is_chief=True):
 697   """Do all training runs.
 698 
 699   This is the top level training function for policy gradient based models.
 700   Run this from the main function.
 701 
 702   Args:
 703     config: config_lib.Config instance containing global config (agent and
 704         environment hparams). If None, config will be parsed from FLAGS.config.
 705     tuner: A tuner instance. Leave as None if not tuning.
 706     logdir: Parent directory where all data from all runs will be written. If
 707         None, FLAGS.logdir will be used.
 708     trial_name: If tuning, set this to a unique string that identifies this
 709         trial. If `tuner` is not None, this also must be set.
 710     is_chief: True if this worker is the chief.
 711 
 712   Returns:
 713     List of results dicts which were written to disk. Each training run gets a
 714     results dict. Results dict contains metrics, i.e. (name, value) pairs which
 715     give information about the training run.
 716 
 717   Raises:
 718     ValueError: If results dicts read from disk contain invalid data.
 719   """
 720   if not config:
 721     # If custom config is not given, get it from flags.
 722     config = defaults.default_config_with_updates(FLAGS.config)
 723   if not logdir:
 724     logdir = FLAGS.logdir
 725   if not tf.gfile.Exists(logdir):
 726     tf.gfile.MakeDirs(logdir)
 727   assert FLAGS.num_repetitions > 0
 728   results = results_lib.Results(logdir)
 729   results_list, _ = results.read_all()
 730 
 731   logging.info('Starting experiment. Directory: "%s"', logdir)
 732 
 733   if results_list:
 734     if results_list[0]['max_npe'] != FLAGS.max_npe:
 735       raise ValueError(
 736           'Cannot resume training. Max-NPE changed. Was %s, now %s',
 737           results_list[0]['max_npe'], FLAGS.max_npe)
 738     if results_list[0]['max_global_repetitions'] != FLAGS.num_repetitions:
 739       raise ValueError(
 740           'Cannot resume training. Number of repetitions changed. Was %s, '
 741           'now %s',
 742           results_list[0]['max_global_repetitions'],
 743           FLAGS.num_repetitions)
 744 
 745   while len(results_list) < FLAGS.num_repetitions:
 746     run_number = len(results_list)
 747     rep_container_name = trial_name if trial_name else 'container'
 748     if FLAGS.num_repetitions > 1:
 749       rep_dir = os.path.join(logdir, 'run_%d' % run_number)
 750       rep_container_name = rep_container_name + '_run_' + str(run_number)
 751     else:
 752       rep_dir = logdir
 753 
 754     logging.info(
 755         'Starting repetition %d (%d out of %d)', run_number, run_number + 1,
 756         FLAGS.num_repetitions)
 757 
 758     # Train will write result to disk.
 759     with tf.container(rep_container_name):
 760       trainer = train(config, is_chief, tuner, rep_dir, run_number, results)
 761     logging.info('Done training.')
 762 
 763     if is_chief:
 764       # Destroy current container immediately (clears current graph).
 765       logging.info('Clearing shared variables.')
 766       tf.Session.reset(FLAGS.master, containers=[rep_container_name])
 767       logging.info('Shared variables cleared.')
 768 
 769       # Delete replay buffer on disk.
 770       assert trainer
 771       trainer.delete_replay_buffer()
 772     else:
 773       # Give chief worker time to clean up.
 774       sleep_sec = 30.0
 775       logging.info('Sleeping for %s sec.', sleep_sec)
 776       time.sleep(sleep_sec)
 777     tf.reset_default_graph()
 778     logging.info('Default graph reset.')
 779 
 780     # Expecting that train wrote new result to disk before returning.
 781     results_list, _ = results.read_all()
 782   return results_list
