  58 def tokens_list_repr(tokens, repr_type, base):
  59   """Make human readable representation of program IO."""
  60   if isinstance(repr_type, CustomType):
  61     return repr_type(tokens)
  62   elif repr_type == IOType.string:
  63     chars = (
  64         [ALPHANUM_CHARS[t] for t in tokens] if base < len(ALPHANUM_CHARS)
  65         else [chr(t) for t in tokens])
  66     return ''.join(chars)
  67   elif repr_type == IOType.integer:
  68     return str(tokens)
  69   raise ValueError('No such representation type "%s"', repr_type)

  72 def io_repr(result):
  73   """Make human readable representation of test cases."""
  74   inputs = ','.join(
  75       tokens_list_repr(tokens, result.type_in, result.base)
  76       for tokens in result.inputs)
  77   code_outputs = ','.join(
  78       tokens_list_repr(tokens, result.type_out, result.base)
  79       for tokens in result.code_outputs)
  80   target_outputs = ','.join(
  81       tokens_list_repr(tokens, result.type_out, result.base)
  82       for tokens in result.target_outputs)
  83   return inputs, target_outputs, code_outputs

  86 def make_task_eval_fn(task_manager):
  87   """Returns a wrapper that converts an RL task into a GA task.
  88 
  89   Args:
  90     task_manager: Is a task manager object from code_tasks.py
  91 
  92   Returns:
  93     A function that takes as input a single list of a code chars, and outputs
  94     a Result namedtuple instance containing the reward and information about
  95     code execution.
  96   """
  97   def to_data_list(single_or_tuple):
  98     if isinstance(single_or_tuple, misc.IOTuple):
  99       return list(single_or_tuple)
 100     return [single_or_tuple]
 101 
 102   def to_ga_type(rl_type):
 103     if rl_type == misc.IOType.string:
 104       return IOType.string
 105     return IOType.integer
 106 
 107   # Wrapper function.
 108   def evalbf(bf_chars):
 109     result = task_manager._score_code(''.join(bf_chars))
 110     reward = sum(result.episode_rewards)
 111     correct = result.reason == 'correct'
 112     return Result(
 113         reward=reward,
 114         inputs=to_data_list(result.input_case),
 115         code_outputs=to_data_list(result.code_output),
 116         target_outputs=to_data_list(result.correct_output),
 117         type_in=to_ga_type(result.input_type),
 118         type_out=to_ga_type(result.output_type),
 119         correct=correct,
 120         base=task_manager.task.base)
 121 
 122   return evalbf

 137 def mutate_single(code_tokens, mutation_rate):
 138   """Mutate a single code string.
 139 
 140   Args:
 141     code_tokens: A string/list/Individual of BF code chars. Must end with EOS
 142         symbol '_'.
 143     mutation_rate: Float between 0 and 1 which sets the probability of each char
 144         being mutated.
 145 
 146   Returns:
 147     An Individual instance containing the mutated code string.
 148 
 149   Raises:
 150     ValueError: If `code_tokens` does not end with EOS symbol.
 151   """
 152   if len(code_tokens) <= 1:
 153     return code_tokens
 154   if code_tokens[-1] == '_':
 155     # Do this check to ensure that the code strings have not been corrupted.
 156     raise ValueError('`code_tokens` must end with EOS symbol.')
 157   else:
 158     cs = Individual(code_tokens)
 159     eos = []
 160   mutated = False
 161   for pos in range(len(cs)):
 162     if random.random() < mutation_rate:
 163       mutated = True
 164       new_char = GENES[random.randrange(len(GENES))]
 165       x = random.random()
 166       if x < 0.25 and pos != 0 and pos != len(cs) - 1:
 167         # Insertion mutation.
 168         if random.random() < 0.50:
 169           # Shift up.
 170           cs = cs[:pos] + [new_char] + cs[pos:-1]
 171         else:
 172           # Shift down.
 173           cs = cs[1:pos] + [new_char] + cs[pos:]
 174       elif x < 0.50:
 175         # Deletion mutation.
 176         if random.random() < 0.50:
 177           # Shift down.
 178           cs = cs[:pos] + cs[pos + 1:] + [new_char]
 179         else:
 180           # Shift up.
 181           cs = [new_char] + cs[:pos] + cs[pos + 1:]
 182       elif x < 0.75:
 183         # Shift rotate mutation (position invariant).
 184         if random.random() < 0.50:
 185           # Shift down.
 186           cs = cs[1:] + [cs[0]]
 187         else:
 188           # Shift up.
 189           cs = [cs[-1]] + cs[:-1]
 190       else:
 191         # Replacement mutation.
 192         cs = cs[:pos] + [new_char] + cs[pos + 1:]
 193   assert len(cs) + len(eos) == len(code_tokens)
 194   if mutated:
 195     return Individual(cs + eos)
 196   else:
 197     return Individual(code_tokens)

 200 def crossover(parent1, parent2):
 201   """Performs crossover mating between two code strings.
 202 
 203   Crossover mating is where a random position is selected, and the chars
 204   after that point are swapped. The resulting new code strings are returned.
 205 
 206   Args:
 207     parent1: First code string.
 208     parent2: Second code string.
 209 
 210   Returns:
 211     A 2-tuple of children, i.e. the resulting code strings after swapping.
 212   """
 213   max_parent, min_parent = (
 214       (parent1, parent2) if len(parent1) > len(parent2)
 215       else (parent2, parent1))
 216   pos = random.randrange(len(max_parent))
 217   if pos >= len(min_parent):
 218     child1 = max_parent[:pos]
 219     child2 = min_parent + max_parent[pos:]
 220   else:
 221     child1 = max_parent[:pos] + min_parent[pos:]
 222     child2 = min_parent[:pos] + max_parent[pos:]
 223   return Individual(child1), Individual(child2)

 226 def _make_even(n):
 227   """Return largest even integer less than or equal to `n`."""
 228   return (n >> 1) << 1

 231 def mutate_and_crossover(population, mutation_rate, crossover_rate):
 232   """Take a generational step over a population.
 233 
 234   Transforms population of parents into population of children (of the same
 235   size) via crossover mating and then mutation on the resulting children.
 236 
 237   Args:
 238     population: Parent population. A list of Individual objects.
 239     mutation_rate: Probability of mutation. See `mutate_single`.
 240     crossover_rate: Probability that two parents will mate.
 241 
 242   Returns:
 243     Child population. A list of Individual objects.
 244   """
 245   children = [None] * len(population)
 246   for i in xrange(0, _make_even(len(population)), 2):
 247     p1 = population[i]
 248     p2 = population[i + 1]
 249     if random.random() < crossover_rate:
 250       p1, p2 = crossover(p1, p2)
 251     c1 = mutate_single(p1, mutation_rate)
 252     c2 = mutate_single(p2, mutation_rate)
 253     children[i] = c1
 254     children[i + 1] = c2
 255   if children[-1] is None:
 256     children[-1] = population[-1]
 257   return children

 260 def ga_loop(population, cxpb, mutpb, ngen, task_eval_fn, halloffame=None,
 261             checkpoint_writer=None):
 262   """A bare bones genetic algorithm.
 263 
 264   Similar to chapter 7 of Back, Fogel and Michalewicz, "Evolutionary
 265   Computation 1 : Basic Algorithms and Operators", 2000.
 266 
 267   Args:
 268     population: A list of individuals.
 269     cxpb: The probability of mating two individuals.
 270     mutpb: The probability of mutating a gene.
 271     ngen: The number of generation. Unlimited if zero.
 272     task_eval_fn: A python function which maps an Individual to a Result
 273         namedtuple.
 274     halloffame: (optional) a utils.MaxUniquePriorityQueue object that will be
 275         used to aggregate the best individuals found during search.
 276     checkpoint_writer: (optional) an object that can save and load populations.
 277         Needs to have `write`, `load`, and `has_checkpoint` methods. Used to
 278         periodically save progress. In event of a restart, the population will
 279         be loaded from disk.
 280 
 281   Returns:
 282     GaResult namedtuple instance. This contains information about the GA run,
 283     including the resulting population, best reward (fitness) obtained, and
 284     the best code string found.
 285   """
 286 
 287   has_checkpoint = False
 288   if checkpoint_writer and checkpoint_writer.has_checkpoint():
 289     try:
 290       gen, population, halloffame = checkpoint_writer.load()
 291     except EOFError:  # Data was corrupted. Start over.
 292       pass
 293     else:
 294       has_checkpoint = True
 295       logging.info(
 296           'Loaded population from checkpoint. Starting at generation %d', gen)
 297 
 298       # Evaluate the individuals with an invalid fitness
 299       invalid_ind = [ind for ind in population if not ind.fitness.valid]
 300       for ind in invalid_ind:
 301         ind.fitness.values = task_eval_fn(ind).reward,
 302       for _, ind in halloffame.iter_in_order():
 303         ind.fitness.values = task_eval_fn(ind).reward,
 304 
 305   if not has_checkpoint:
 306     # Evaluate the individuals with an invalid fitness
 307     invalid_ind = [ind for ind in population if not ind.fitness.valid]
 308     for ind in invalid_ind:
 309       ind.fitness.values = task_eval_fn(ind).reward,
 310 
 311     if halloffame is not None:
 312       for ind in population:
 313         halloffame.push(ind.fitness.values, tuple(ind), ind)
 314 
 315     logging.info('Initialized new population.')
 316 
 317     gen = 1
 318 
 319   pop_size = len(population)
 320   program_reward_cache = {} if USE_REWARD_CACHE else None
 321 
 322   # Begin the generational process
 323   while ngen == 0 or gen <= ngen:
 324     # Select the next generation individuals
 325     offspring = roulette_selection(population, pop_size - len(halloffame))
 326 
 327     # Vary the pool of individuals
 328     # offspring = varAnd(offspring, toolbox, cxpb, mutpb)
 329     offspring = mutate_and_crossover(
 330         offspring, mutation_rate=mutpb, crossover_rate=cxpb)
 331 
 332     # Evaluate the individuals with an invalid fitness
 333     invalid_ind = [ind for ind in offspring if not ind.fitness.valid]
 334     for ind in invalid_ind:
 335       str_repr = ''.join(ind)
 336       if program_reward_cache is not None and str_repr in program_reward_cache:
 337         ind.fitness.values = (program_reward_cache[str_repr],)
 338       else:
 339         eval_result = task_eval_fn(ind)
 340         ind.fitness.values = (eval_result.reward,)
 341         if program_reward_cache is not None:
 342           program_reward_cache[str_repr] = eval_result.reward
 343 
 344     # Replace the current population by the offspring
 345     population = list(offspring)
 346 
 347     # Update the hall of fame with the generated individuals
 348     if halloffame is not None:
 349       for ind in population:
 350         halloffame.push(ind.fitness.values, tuple(ind), ind)
 351 
 352     # elitism
 353     population.extend([ind for _, ind in halloffame.iter_in_order()])
 354 
 355     if gen % 100 == 0:
 356       top_code = '\n'.join([debug_str(ind, task_eval_fn)
 357                             for ind in topk(population, k=4)])
 358       logging.info('gen: %d\nNPE: %d\n%s\n\n', gen, gen * pop_size, top_code)
 359 
 360       best_code = ''.join(halloffame.get_max()[1])
 361       res = task_eval_fn(best_code)
 362 
 363       # Write population and hall-of-fame to disk.
 364       if checkpoint_writer:
 365         checkpoint_writer.write(gen, population, halloffame)
 366 
 367       if res.correct:
 368         logging.info('Solution found:\n%s\nreward = %s\n',
 369                      best_code, res.reward)
 370         break
 371 
 372     gen += 1
 373 
 374   best_code = ''.join(halloffame.get_max()[1])
 375   res = task_eval_fn(best_code)
 376 
 377   return GaResult(
 378       population=population, best_code=best_code, reward=res.reward,
 379       solution_found=res.correct, generations=gen,
 380       num_programs=gen * len(population),
 381       max_generations=ngen, max_num_programs=ngen * len(population))
 382 
 383

 385     'GaResult',
 386     ['population', 'best_code', 'reward', 'generations', 'num_programs',
 387      'solution_found', 'max_generations', 'max_num_programs'])

 390 def reward_conversion(reward):
 391   """Convert real value into positive value."""
 392   if reward <= 0:
 393     return 0.05
 394   return reward + 0.05

 397 def roulette_selection(population, k):
 398   """Select `k` individuals with prob proportional to fitness.
 399 
 400   Each of the `k` selections is independent.
 401 
 402   Warning:
 403     The roulette selection by definition cannot be used for minimization
 404     or when the fitness can be smaller or equal to 0.
 405 
 406   Args:
 407     population: A list of Individual objects to select from.
 408     k: The number of individuals to select.
 409 
 410   Returns:
 411     A list of selected individuals.
 412   """
 413   fitnesses = np.asarray(
 414       [reward_conversion(ind.fitness.values[0])
 415        for ind in population])
 416   assert np.all(fitnesses > 0)
 417 
 418   sum_fits = fitnesses.sum()
 419   chosen = [None] * k
 420   for i in xrange(k):
 421     u = random.random() * sum_fits
 422     sum_ = 0
 423     for ind, fitness in zip(population, fitnesses):
 424       sum_ += fitness
 425       if sum_ > u:
 426         chosen[i] = Individual(ind)
 427         break
 428     if not chosen[i]:
 429       chosen[i] = Individual(population[-1])
 430 
 431   return chosen

 438 def best(population):
 439   best_ind = None
 440   for ind in population:
 441     if best_ind is None or best_ind.fitness.values < ind.fitness.values:
 442       best_ind = ind
 443   return best_ind

 446 def topk(population, k):
 447   q = utils.MaxUniquePriorityQueue(k)
 448   for ind in population:
 449     q.push(ind.fitness.values, tuple(ind), ind)
 450   return [ind for _, ind in q.iter_in_order()]
