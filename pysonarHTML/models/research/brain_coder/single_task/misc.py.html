  58 def bf_tokens_to_string(bf_tokens, truncate=True):
  59   """Convert token list to code string. Will truncate at EOS token.
  60 
  61   Args:
  62     bf_tokens: Python list of ints representing the code string.
  63     truncate: If true, the output string will end at the first EOS token.
  64         If false, the entire token list is converted to string.
  65 
  66   Returns:
  67     String representation of the tokens.
  68 
  69   Raises:
  70     ValueError: If bf_tokens is not a python list.
  71   """
  72   if not isinstance(bf_tokens, list):
  73     raise ValueError('Only python list supported here.')
  74   if truncate:
  75     try:
  76       eos_index = bf_tokens.index(BF_EOS_INT)
  77     except ValueError:
  78       eos_index = len(bf_tokens)
  79   else:
  80     eos_index = len(bf_tokens)
  81   return ''.join([BF_INT_TO_CHAR[t] for t in bf_tokens[:eos_index]])

  84 def bf_string_to_tokens(bf_string):
  85   """Convert string to token list. Will strip and append EOS token."""
  86   tokens = [BF_CHAR_TO_INT[char] for char in bf_string.strip()]
  87   tokens.append(BF_EOS_INT)
  88   return tokens

  91 def tokens_to_text(tokens):
  92   """Convert token list to human readable text."""
  93   return ''.join(
  94       [TEXT_EOS_CHAR if t == 0 else chr(t - 1 + ord('A')) for t in tokens])

 102 # https://en.wikipedia.org/wiki/Metric_prefix
 103 si_magnitudes = {
 104     'k': 1e3,
 105     'm': 1e6,
 106     'g': 1e9}
 107 
 108 
 109 def si_to_int(s):
 110   """Convert string ending with SI magnitude to int.
 111 
 112   Examples: 5K ==> 5000, 12M ==> 12000000.
 113 
 114   Args:
 115     s: String in the form 'xx..xP' where x is a digit and P is an SI prefix.
 116 
 117   Returns:
 118     Integer equivalent to the string.
 119   """
 120   if isinstance(s, string_types) and s[-1].lower() in si_magnitudes.keys():
 121     return int(int(s[:-1]) * si_magnitudes[s[-1].lower()])
 122   return int(s)
 123

 125 def int_to_si(n):
 126   """Convert integer to string with SI magnitude.
 127 
 128   `n` will be truncated.
 129 
 130   Examples: 5432 ==> 5k, 12345678 ==> 12M
 131 
 132   Args:
 133     n: Integer to represent as a string.
 134 
 135   Returns:
 136     String representation of `n` containing SI magnitude.
 137   """
 138   m = abs(n)
 139   sign = -1 if n < 0 else 1
 140   if m < 1e3:
 141     return str(n)
 142   if m < 1e6:
 143     return '{0}K'.format(sign*int(m / 1e3))
 144   if m < 1e9:
 145     return '{0}M'.format(sign*int(m / 1e6))
 146   if m < 1e12:
 147     return '{0}G'.format(sign*int(m / 1e9))
 148   return str(m)
