  69 def make_csv_string(table):
  70   """Convert 2D list to CSV string."""
  71   s = StringIO.StringIO()
  72   writer = csv.writer(s)
  73   writer.writerows(table)
  74   value = s.getvalue()
  75   s.close()
  76   return value

  79 def process_results(metrics):
  80   """Extract useful information from given metrics.
  81 
  82   Args:
  83     metrics: List of results dicts. These should have been written to disk by
  84         training jobs.
  85 
  86   Returns:
  87     Dict mapping stats names to values.
  88 
  89   Raises:
  90     ValueError: If max_npe or max_global_repetitions values are inconsistant
  91         across dicts in the `metrics` list.
  92   """
  93   count = len(metrics)
  94   success_count = 0
  95   total_npe = 0  # Counting NPE across all runs.
  96   success_npe = 0  # Counting NPE in successful runs only.
  97   max_npe = 0
  98   max_repetitions = 0
  99   for metric_dict in metrics:
 100     if not max_npe:
 101       max_npe = metric_dict['max_npe']
 102     elif max_npe != metric_dict['max_npe']:
 103       raise ValueError(
 104           'Invalid experiment. Different reps have different max-NPE settings.')
 105     if not max_repetitions:
 106       max_repetitions = metric_dict['max_global_repetitions']
 107     elif max_repetitions != metric_dict['max_global_repetitions']:
 108       raise ValueError(
 109           'Invalid experiment. Different reps have different num-repetition '
 110           'settings.')
 111     if metric_dict['found_solution']:
 112       success_count += 1
 113       success_npe += metric_dict['npe']
 114     total_npe += metric_dict['npe']
 115   stats = {}
 116   stats['max_npe'] = max_npe
 117   stats['max_repetitions'] = max_repetitions
 118   stats['repetitions'] = count
 119   stats['successes'] = success_count  # successful reps
 120   stats['failures'] = count - success_count  # failed reps
 121   stats['success_npe'] = success_npe
 122   stats['total_npe'] = total_npe
 123   if success_count:
 124     # Only successful runs counted.
 125     stats['avg_success_npe'] = stats['success_npe'] / float(success_count)
 126   else:
 127     stats['avg_success_npe'] = 0.0
 128   if count:
 129     stats['success_rate'] = success_count / float(count)
 130     stats['avg_total_npe'] = stats['total_npe'] / float(count)
 131   else:
 132     stats['success_rate'] = 0.0
 133     stats['avg_total_npe'] = 0.0
 134 
 135   return stats

 141 def get_results_for_experiment(
 142     models_dir, task_name, model_type='pg', max_npe='5M', desc='v0',
 143     name_prefix='bf_rl_paper', extra_desc=''):
 144   """Get and process results for a given experiment.
 145 
 146   An experiment is a set of runs with the same hyperparameters and environment.
 147   It is uniquely specified by a (task_name, model_type, max_npe) triple, as
 148   well as an optional description.
 149 
 150   We assume that each experiment has a folder with the same name as the job that
 151   ran the experiment. The name is computed by
 152   "%name_prefix%.%desc%-%max_npe%_%task_name%".
 153 
 154   Args:
 155     models_dir: Parent directory containing experiment folders.
 156     task_name: String name of task (the coding env). See code_tasks.py or
 157         run_eval_tasks.py
 158     model_type: Name of the algorithm, such as 'pg', 'topk', 'ga', 'rand'.
 159     max_npe: String SI unit representation of the maximum NPE threshold for the
 160         experiment. For example, "5M" means 5 million.
 161     desc: Description.
 162     name_prefix: Prefix of job names. Normally leave this as default.
 163     extra_desc: Optional extra description at the end of the job name.
 164 
 165   Returns:
 166     ProcessedResults namedtuple instance, containing
 167     metrics: Raw dicts read from disk.
 168     processed: Stats computed by `process_results`.
 169 
 170   Raises:
 171     ValueError: If max_npe in the metrics does not match NPE in the experiment
 172         folder name.
 173   """
 174   folder = name_prefix + '.{0}.{1}-{2}_{3}'.format(desc, model_type, max_npe,
 175                                                    task_name)
 176   if extra_desc:
 177     folder += '.' + extra_desc
 178 
 179   results = results_lib.Results(os.path.join(models_dir, folder))
 180   metrics, _ = results.read_all()
 181   processed = process_results(metrics)
 182   if (not np.isclose(processed['max_npe'], misc.si_to_int(max_npe))
 183       and processed['repetitions']):
 184     raise ValueError(
 185         'Invalid experiment. Max-NPE setting does not match expected max-NPE '
 186         'in experiment name.')
 187   return ProcessedResults(metrics=metrics, processed=processed)

 201 def get_best_code_for_experiment(
 202     models_dir, task_name, model_type='pg', max_npe='5M', desc=0,
 203     name_prefix='bf_rl_paper', extra_desc=''):
 204   """Like `get_results_for_experiment`, but fetches the code solutions."""
 205   folder = name_prefix + '.{0}.{1}-{2}_{3}'.format(desc, model_type, max_npe,
 206                                                    task_name)
 207   if extra_desc:
 208     folder += '.' + extra_desc
 209 
 210   log_dir = os.path.join(models_dir, folder, 'logs')
 211   search_regex = r'^solutions_([0-9])+\.txt$'
 212   try:
 213     all_children = tf.gfile.ListDirectory(log_dir)
 214   except tf.errors.NotFoundError:
 215     return BestCodeResults(
 216         code=None, reward=0.0, npe=0, folder=folder, finished=False,
 217         error=BestCodeResultError.experiment_does_not_exist)
 218   solution_files = [
 219       fname for fname in all_children if re.search(search_regex, fname)]
 220   max_reward = 0.0
 221   npe = 0
 222   best_code = None
 223   for fname in solution_files:
 224     with tf.gfile.FastGFile(os.path.join(log_dir, fname), 'r') as reader:
 225       results = [ast.literal_eval(entry) for entry in reader]
 226     for res in results:
 227       if res['reward'] > max_reward:
 228         best_code = res['code']
 229         max_reward = res['reward']
 230         npe = res['npe']
 231   error = (
 232       BestCodeResultError.success if best_code
 233       else BestCodeResultError.no_solution_found)
 234   try:
 235     # If there is a status.txt file, check if it contains the status of the job.
 236     with tf.gfile.FastGFile(os.path.join(log_dir, 'status.txt'), 'r') as f:
 237       # Job is done, so mark this experiment as finished.
 238       finished = f.read().lower().strip() == 'done'
 239   except tf.errors.NotFoundError:
 240     # No status file has been written, so the experiment is not done. No need to
 241     # report an error here, because we do not require that experiment jobs write
 242     # out a status.txt file until they have finished.
 243     finished = False
 244   return BestCodeResults(
 245       code=best_code, reward=max_reward, npe=npe, folder=folder,
 246       finished=finished, error=error)

 249 def make_results_table(
 250     models=None,
 251     tasks=None,
 252     max_npe='5M',
 253     name_prefix='bf_rl_paper',
 254     extra_desc='',
 255     models_dir='/tmp'):
 256   """Creates a table of results: algorithm + version by tasks.
 257 
 258   Args:
 259     models: The table columns. A list of (algorithm, desc) tuples.
 260     tasks: The table rows. List of task names.
 261     max_npe: String SI unit representation of the maximum NPE threshold for the
 262         experiment. For example, "5M" means 5 million. All entries in the table
 263         share the same max-NPE.
 264     name_prefix: Name prefix used in logging directory for the experiment.
 265     extra_desc: Extra description added to name of logging directory for the
 266         experiment.
 267     models_dir: Parent directory containing all experiment folders.
 268 
 269   Returns:
 270     A 2D list holding the table cells.
 271   """
 272   if models is None:
 273     models = DEFAULT_MODELS
 274   if tasks is None:
 275     tasks = DEFAULT_TASKS
 276   model_results = {}
 277   for model_type, desc in models:
 278     model_results[model_type] = {
 279         tname: get_results_for_experiment(
 280             models_dir, tname, model_type, max_npe, desc,
 281             name_prefix=name_prefix, extra_desc=extra_desc
 282         ).processed
 283         for tname in tasks}
 284 
 285   def info(stats):
 286     return [str(stats['repetitions']),
 287             '%.2f' % stats['success_rate'],
 288             str(int(stats['avg_total_npe']))]
 289 
 290   rows = [['max NPE: ' + max_npe]
 291           + misc.flatten([['{0} ({1})'.format(m, d), '', '']
 292                           for m, d in models])]
 293   rows.append(
 294       [''] + misc.flatten([['reps', 'success rate', 'avg NPE']
 295                            for _ in models]))
 296   for tname in tasks:
 297     rows.append(
 298         [tname]
 299         + misc.flatten([info(model_results[model][tname])
 300                         for model, _ in models]))
 301 
 302   return rows

 305 def print_results_table(results_table):
 306   """Print human readable results table to stdout."""
 307   print('')
 308   print('=== Results Table ===')
 309   print('Format: # reps [success rate, avg total NPE]')
 310 
 311   def info_str(info_row):
 312     # num_runs (success_rate, avg_total_npe)
 313     if not info_row[0]:
 314       return '0'
 315     return '%s [%s, %s]' % (str(info_row[0]).ljust(2), info_row[1], info_row[2])
 316 
 317   nc = len(results_table[0])  # num cols
 318   out_table = [
 319       [results_table[0][0]] + [results_table[0][i] for i in range(1, nc, 3)]]
 320   for row in results_table[2:]:
 321     out_table.append([row[0]] + [info_str(row[i:i+3]) for i in range(1, nc, 3)])
 322 
 323   nc = len(out_table[0])  # num cols
 324   col_widths = [max(len(row[col]) for row in out_table) for col in range(nc)]
 325 
 326   table_string = ''
 327   for row in out_table:
 328     table_string += ''.join(
 329         [row[c].ljust(col_widths[c] + 2) for c in range(nc)]) + '\n'
 330 
 331   print(table_string)

 334 def main(argv):
 335   del argv  # Unused.
 336 
 337   name_prefix = FLAGS.exp_prefix
 338   print('Experiments prefix: %s' % name_prefix)
 339 
 340   model_types = ast.literal_eval(FLAGS.model_types)
 341 
 342   if FLAGS.data == 'success_rates':
 343     results_table = make_results_table(
 344         models=model_types, tasks=FLAGS.task_list, max_npe=FLAGS.max_npe,
 345         models_dir=FLAGS.models_dir,
 346         name_prefix=name_prefix, extra_desc='')
 347     with tf.gfile.FastGFile(FLAGS.csv_file, 'w') as f:
 348       f.write(make_csv_string(results_table))
 349 
 350     print_results_table(results_table)
 351   else:
 352     # Best code
 353     print('* = experiment is still running')
 354     print('')
 355     print('=== Best Synthesized Code ===')
 356     for model_type, desc in model_types:
 357       print('%s (%s)' % (model_type, desc))
 358       sys.stdout.flush()
 359       for tname in FLAGS.task_list:
 360         res = get_best_code_for_experiment(
 361             FLAGS.models_dir, tname, model_type, FLAGS.max_npe, desc,
 362             name_prefix=name_prefix, extra_desc='')
 363         unfinished_mark = '' if res.finished else ' *'
 364         tname += unfinished_mark
 365         if res.error == BestCodeResultError.success:
 366           print('  %s' % tname)
 367           print('    %s' % res.code)
 368           print('    R=%.6f, NPE=%s' % (res.reward, misc.int_to_si(res.npe)))
 369         elif res.error == BestCodeResultError.experiment_does_not_exist:
 370           print('  Experiment does not exist. Check arguments.')
 371           print('  Experiment folder: %s' % res.folder)
 372           break
 373         else:
 374           print('  %s' % tname)
 375           print('    (none)')
 376         sys.stdout.flush()
