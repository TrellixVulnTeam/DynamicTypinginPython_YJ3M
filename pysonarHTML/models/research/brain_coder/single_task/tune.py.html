  68 def parse_hparams_string(hparams_str):
  69   hparams = {}
  70   for term in hparams_str.split(','):
  71     if not term:
  72       continue
  73     name, value = term.split('=')
  74     hparams[name.strip()] = ast.literal_eval(value)
  75   return hparams

  78 def int_to_multibase(n, bases):
  79   digits = [0] * len(bases)
  80   for i, b in enumerate(bases):
  81     n, d = divmod(n, b)
  82     digits[i] = d
  83   return digits
  84

  86 def hparams_for_index(index, tuning_space):
  87   keys = sorted(tuning_space.keys())
  88   indices = int_to_multibase(index, [len(tuning_space[k]) for k in keys])
  89   return tf.contrib.training.HParams(
  90       **{k: tuning_space[k][i] for k, i in zip(keys, indices)})

  93 def run_tuner_loop(ns):
  94   """Run tuning loop for this worker."""
  95   is_chief = FLAGS.task_id == 0
  96   tuning_space = ns.define_tuner_hparam_space(
  97       hparam_space_type=FLAGS.hparam_space)
  98   fixed_hparams = parse_hparams_string(FLAGS.fixed_hparams)
  99   for name, value in fixed_hparams.iteritems():
 100     tuning_space[name] = [value]
 101   tuning_space_size = np.prod([len(values) for values in tuning_space.values()])
 102 
 103   num_local_trials, remainder = divmod(tuning_space_size, FLAGS.num_tuners)
 104   if FLAGS.tuner_id < remainder:
 105     num_local_trials += 1
 106   starting_trial_id = (
 107       num_local_trials * FLAGS.tuner_id + min(remainder, FLAGS.tuner_id))
 108 
 109   logging.info('tuning_space_size: %d', tuning_space_size)
 110   logging.info('num_local_trials: %d', num_local_trials)
 111   logging.info('starting_trial_id: %d', starting_trial_id)
 112 
 113   for local_trial_index in xrange(num_local_trials):
 114     trial_config = defaults.default_config_with_updates(FLAGS.config)
 115     global_trial_index = local_trial_index + starting_trial_id
 116     trial_name = 'trial_' + str(global_trial_index)
 117     trial_dir = os.path.join(FLAGS.logdir, trial_name)
 118     hparams = hparams_for_index(global_trial_index, tuning_space)
 119     ns.write_hparams_to_config(
 120         trial_config, hparams, hparam_space_type=FLAGS.hparam_space)
 121 
 122     results_list = ns.run_training(
 123         config=trial_config, tuner=None, logdir=trial_dir, is_chief=is_chief,
 124         trial_name=trial_name)
 125 
 126     if not is_chief:
 127       # Only chief worker needs to write tuning results to disk.
 128       continue
 129 
 130     objective, metrics = compute_tuning_objective(
 131         results_list, hparams, trial_name, num_trials=tuning_space_size)
 132     logging.info('metrics:\n%s', metrics)
 133     logging.info('objective: %s', objective)
 134     logging.info('programs_seen_fraction: %s',
 135                  metrics['programs_seen_fraction'])
 136     logging.info('success_rate: %s', metrics['success_rate'])
 137     logging.info('success_rate_objective_weight: %s',
 138                  FLAGS.success_rate_objective_weight)
 139 
 140     tuning_results_file = os.path.join(trial_dir, 'tuning_results.txt')
 141     with tf.gfile.FastGFile(tuning_results_file, 'a') as writer:
 142       writer.write(str(metrics) + '\n')
 143 
 144     logging.info('Trial %s complete.', trial_name)

 147 def compute_tuning_objective(results_list, hparams, trial_name, num_trials):
 148   """Compute tuning objective and metrics given results and trial information.
 149 
 150   Args:
 151     results_list: List of results dicts read from disk. These are written by
 152         workers.
 153     hparams: tf.contrib.training.HParams instance containing the hparams used
 154         in this trial (only the hparams which are being tuned).
 155     trial_name: Name of this trial. Used to create a trial directory.
 156     num_trials: Total number of trials that need to be run. This is saved in the
 157         metrics dict for future reference.
 158 
 159   Returns:
 160     objective: The objective computed for this trial. Choose the hparams for the
 161         trial with the largest objective value.
 162     metrics: Information about this trial. A dict.
 163   """
 164   found_solution = [r['found_solution'] for r in results_list]
 165   successful_program_counts = [
 166       r['npe'] for r in results_list if r['found_solution']]
 167 
 168   success_rate = sum(found_solution) / float(len(results_list))
 169 
 170   max_programs = FLAGS.max_npe  # Per run.
 171   all_program_counts = [
 172       r['npe'] if r['found_solution'] else max_programs
 173       for r in results_list]
 174   programs_seen_fraction = (
 175       float(sum(all_program_counts))
 176       / (max_programs * len(all_program_counts)))
 177 
 178   # min/max/avg stats are over successful runs.
 179   metrics = {
 180       'num_runs': len(results_list),
 181       'num_succeeded': sum(found_solution),
 182       'success_rate': success_rate,
 183       'programs_seen_fraction': programs_seen_fraction,
 184       'avg_programs': np.mean(successful_program_counts),
 185       'max_possible_programs_per_run': max_programs,
 186       'global_step': sum([r['num_batches'] for r in results_list]),
 187       'hparams': hparams.values(),
 188       'trial_name': trial_name,
 189       'num_trials': num_trials}
 190 
 191   # Report stats per tasks.
 192   tasks = [r['task'] for r in results_list]
 193   for task in set(tasks):
 194     task_list = [r for r in results_list if r['task'] == task]
 195     found_solution = [r['found_solution'] for r in task_list]
 196     successful_rewards = [
 197         r['best_reward'] for r in task_list
 198         if r['found_solution']]
 199     successful_num_batches = [
 200         r['num_batches']
 201         for r in task_list if r['found_solution']]
 202     successful_program_counts = [
 203         r['npe'] for r in task_list if r['found_solution']]
 204     metrics_append = {
 205         task + '__num_runs': len(task_list),
 206         task + '__num_succeeded': sum(found_solution),
 207         task + '__success_rate': (
 208             sum(found_solution) / float(len(task_list)))}
 209     metrics.update(metrics_append)
 210     if any(found_solution):
 211       metrics_append = {
 212           task + '__min_reward': min(successful_rewards),
 213           task + '__max_reward': max(successful_rewards),
 214           task + '__avg_reward': np.median(successful_rewards),
 215           task + '__min_programs': min(successful_program_counts),
 216           task + '__max_programs': max(successful_program_counts),
 217           task + '__avg_programs': np.mean(successful_program_counts),
 218           task + '__min_batches': min(successful_num_batches),
 219           task + '__max_batches': max(successful_num_batches),
 220           task + '__avg_batches': np.mean(successful_num_batches)}
 221       metrics.update(metrics_append)
 222 
 223   # Objective will be maximized.
 224   # Maximize success rate, minimize num programs seen.
 225   # Max objective is always 1.
 226   weight = FLAGS.success_rate_objective_weight
 227   objective = (
 228       weight * success_rate
 229       + (1 - weight) * (1 - programs_seen_fraction))
 230   metrics['objective'] = objective
 231 
 232   return objective, metrics

 235 def main(argv):

 240   if not FLAGS.logdir:

 253   if FLAGS.tuner_id >= FLAGS.num_tuners:

 255         'tuner_id flag must be strictly less than num_tuners flag.')

 257   ns, _ = run_lib.get_namespace(FLAGS.config)
