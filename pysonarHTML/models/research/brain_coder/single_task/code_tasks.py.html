 114 def concat(lists):
 115   if not lists:
 116     return []
 117   l = lists[0]
 118   for k in lists[1:]:
 119     l += k
 120   return l
 121 
 122 
 123 def concat_join(lists, sep):
 124   if not lists:
 125     return []
 126   l = lists[0]
 127   for k in lists[1:]:
 128     l += [sep] + k
 129   return l

 132 def clipped_linear(x, x0, y0, slope, y_range):
 133   min_y, max_y = y_range
 134   return min(max(slope * (x - x0) + y0, min_y), max_y)

 137 class MultiIOTaskManager(object):
 138   """Supports tasks which test the code with multiple I/O examples."""
 139 
 140   def __init__(self, task, max_code_length=32, min_code_length=0,
 141                max_execution_steps=MAX_EXECUTION_STEPS, correct_bonus=1.0,
 142                code_length_bonus=1.0, failure_reward=-2.0, reward_fn=None,
 143                require_correct_syntax=False):
 144     assert isinstance(task, BaseTask)
 145     self.task = task
 146     self.max_code_length = max_code_length
 147     self.min_code_length = min_code_length
 148     self.max_execution_steps = max_execution_steps
 149     self.require_correct_syntax = require_correct_syntax
 150     self.correct_bonus = correct_bonus
 151     self.code_length_bonus = code_length_bonus
 152     self.failure_reward = failure_reward
 153     self.time_penalty = (
 154         1.0 / (max_code_length - min_code_length)
 155         if max_code_length > min_code_length else 0.0)
 156     if reward_fn is None:
 157       self.reward_fn = r.absolute_distance_reward
 158     else:
 159       self.reward_fn = reward_fn
 160     self.input_type = (
 161         task.input_type if hasattr(task, 'input_type') else misc.IOType.integer)
 162     self.output_type = (
 163         task.output_type if hasattr(task, 'output_type')
 164         else misc.IOType.integer)
 165     self._compute_best_reward()
 166 
 167   def _compute_best_reward(self):
 168     io_seqs = self.task.make_io_set()
 169     reward = 0.0
 170     for _, output_seq in io_seqs:
 171       reward += self.reward_fn(output_seq, output_seq, self.task.base)
 172       reward += self.correct_bonus
 173       reward += self.code_length_bonus  # Bonus for shortest code.
 174     self.best_reward = reward
 175     self.good_reward = 0.75 * reward
 176     logging.info('Known best reward: %.4f', self.best_reward)
 177 
 178   def _score_batch(self, code_strings):
 179     return [self._score_code(code) for code in code_strings]
 180 
 181   def _score_code(self, code):
 182     """Run test cases on code and compute reward.
 183 
 184     Args:
 185       code: A single BF code string.
 186 
 187     Returns:
 188       misc.RewardInfo namedtuple instance containing reward and code execution
 189           information, including inputs, expected outputs, code outputs, input
 190           and output types, and reason for the reward obtained.
 191     """
 192     # Get list of 2-tuples, each containing an input sequence and an output
 193     # sequence.
 194     io_seqs = self.task.make_io_set()
 195     terminal_reward = 0.0
 196     results = []
 197     reason = 'correct'
 198     for input_seq, output_seq in io_seqs:
 199       eval_result = bf.evaluate(
 200           code, input_buffer=input_seq, timeout=0.1,
 201           max_steps=self.max_execution_steps,
 202           base=self.task.base,
 203           require_correct_syntax=self.require_correct_syntax)
 204       result, success = eval_result.output, eval_result.success
 205       if not success:
 206         # Code execution timed out.
 207         terminal_reward = self.failure_reward
 208         results = []
 209         reason = eval_result.failure_reason
 210         break
 211       else:
 212         terminal_reward += self.reward_fn(result, output_seq, self.task.base)
 213         if result == output_seq:
 214           terminal_reward += self.correct_bonus  # Bonus for correct answer.
 215 
 216           # Only add additional reward for shorter code. Subtracting reward
 217           # interferes with the main objective. Only optimize for length once
 218           # any solution is found.
 219           if self.min_code_length == self.max_code_length:
 220             terminal_reward += self.code_length_bonus
 221           else:
 222             terminal_reward += self.code_length_bonus * clipped_linear(
 223                 x=len(code), x0=self.min_code_length, y0=1.0,
 224                 slope=-self.time_penalty, y_range=(0.0, 1.0))
 225 
 226           # reason remains 'correct' if it is already
 227         elif reason == 'correct':
 228           reason = 'wrong'
 229       results.append(result)
 230 
 231     # Return list of rewards, one for each char in the code. All are 0 except
 232     # for the terminal reward.
 233     terminal_reward /= self.best_reward
 234     return misc.RewardInfo(
 235         episode_rewards=[0.0] * (len(code) - 1) + [terminal_reward],
 236         input_case=misc.IOTuple(i for i, o in io_seqs),
 237         correct_output=misc.IOTuple(o for i, o in io_seqs),
 238         code_output=misc.IOTuple(results),
 239         input_type=self.input_type,
 240         output_type=self.output_type,
 241         reason=reason)
 242 
 243   def rl_batch(self, batch_size):
 244     """Produces list of reward functions. One for each program in the batch."""
 245     return [self._score_code] * batch_size

 248 def conditional_overwrite(current_value, new_value, allowed_overwrite_values):
 249   if current_value in allowed_overwrite_values:
 250     return new_value
 251   return current_value

 254 class BaseTask(object):
 255   """A coding task.
 256 
 257   All coding tasks should inherit this class.
 258   """
 259   __metaclass__ = abc.ABCMeta
 260 
 261   def __init__(self, base=256):
 262     self.base = base  # All tasks must set the integer base that the expect.
 263 
 264   @abc.abstractmethod
 265   def make_io_set(self):
 266     """Generate a set of test cases for the task.
 267 
 268     Returns:
 269       List of tuples, where each tuple is (input_case, output_case).
 270       input_case and output_case are lists of integers.
 271     """
 272     pass

 288     super(type(self), self).__init__()

 330 class RemoveCharTask(BaseTask):
 331   """Remove character coding task.
 332 
 333   Code needs to pipe input to output, but with all the 'A' (value 1) chars
 334   removed. 'A' appears at least once in each input.
 335 
 336   Test cases are dynamically generated, allowing for the number of test cases
 337   to be a hyperparameter.
 338   """
 339 
 340   def __init__(self, base, n, min_len, max_len):
 341     super(type(self), self).__init__()
 342     self.base = base
 343     self.eos = 0
 344     self.remove_char = 1
 345     assert base >= 27
 346     self._io_pairs = self._make_io_examples(n, min_len, max_len)
 347 
 348   def _make_io_examples(self, n, min_len, max_len):
 349     """Generate test cases for the task."""
 350     rand = random.Random(6849275409234)  # Test cases are fixed, but varied.
 351     io_examples = []
 352     for _ in xrange(n):
 353       length = rand.randrange(min_len, max_len + 1)
 354       rm_char_pos = rand.randrange(0, length)
 355       input_seq = [rand.randrange(1, self.base) for _ in xrange(length)]
 356       input_seq[rm_char_pos] = self.remove_char
 357       output_seq = list(input_seq)
 358       del output_seq[rm_char_pos]
 359       output_seq.append(0)
 360       io_examples.append((input_seq, output_seq))
 361     return io_examples
 362 
 363   def make_io_set(self):
 364     return copy.deepcopy(self._io_pairs)

 367 class ReverseTaskV2(BaseTask):
 368   """Reverse string coding task (version 2).
 369 
 370   Code needs to pipe input to output, but in reverse order.
 371 
 372   Stochastic test case = new test case randomly generated for every run of
 373   `make_io_set`, i.e. different test cases every time code is scored.
 374 
 375   Task supports different types of test cases:
 376     rand-one: Code is scored on one stochastic test case.
 377     rand-many: Code is scored on 5 stochastic test cases.
 378     static-bylen: Code is scored on 5 static test cases. There is one test
 379         case for string lengths 1 through 5.
 380     rand-bylen: Code is scored on 5 stochastic test cases, where there is one
 381         test case for string lengths 1 through 5.
 382   """
 383 
 384   def __init__(self, base, reward_type):
 385     super(type(self), self).__init__()
 386     self.base = base  # base includes EOS
 387     assert base >= 27
 388     self.eos = 0
 389     self.io_pair_fn = {
 390         # One random example at a time.
 391         'rand-one': lambda: self._io_rand(1),
 392         # K randomy examples at a time (any lengths).
 393         'rand-many': lambda: self._io_rand(5),
 394         # Static examples, one for each length.
 395         'static-bylen': self._io_static_by_len,
 396         # Random examples, one for each length.
 397         'rand-bylen': self._io_rand_by_len}[reward_type]
 398 
 399   def _make_io_examples(self, sequences):
 400     outputs = [list(i) for i in sequences]
 401     for o in outputs:
 402       o.reverse()
 403       o.append(0)
 404     inputs = [i + [0] for i in sequences]
 405     return zip(inputs, outputs)
 406 
 407   def _io_rand(self, k):
 408     inputs = [(np.random.choice(26, random.randrange(1, 6)) + 1).tolist()
 409               for _ in xrange(k)]
 410     return self._make_io_examples(inputs)
 411 
 412   def _io_rand_by_len(self, k=5):
 413     inputs = [(np.random.choice(26, length) + 1).tolist()
 414               for length in xrange(1, k + 1)]
 415     return self._make_io_examples(inputs)
 416 
 417   def _io_static_by_len(self):
 418     return [
 419         ([7, 0], [7, 0]),
 420         ([6, 2, 0], [2, 6, 0]),
 421         ([5, 1, 10, 0], [10, 1, 5, 0]),
 422         ([8, 6, 5, 15, 0], [15, 5, 6, 8, 0]),
 423         ([10, 12, 5, 2, 7, 0], [7, 2, 5, 12, 10, 0])]
 424 
 425   def make_io_set(self):
 426     return self.io_pair_fn()

 530     self.base = 27

 537     self._io_pairs = self._make_io_examples(n, max_len)

 540     """Generate test cases for the task."""
 541     rand = random.Random(6849275409234)  # Test cases are fixed, but varied.
 542     io_examples = []
 543     io_examples.append(([10, 0], [0]))
 544     io_examples.append(([1, 0], [1]))
 545     io_examples.append(([1, 1, 0], [2]))
 546     io_examples.append(([9, 4, 19, 11, 5, 0], [0]))
 547     io_examples.append(([24, 11, 26, 1, 15, 0], [1]))
 548     for _ in xrange(n - 5):
 549       length = rand.randrange(2, max_len + 1)
 550       num_chars = rand.randrange(0, max_len + 1)

 642 class KnownCodeBaseTask(BaseTask):
 643   """These tasks generate their test cases from a known BF solution.
 644 
 645   This ensures that each task has a solution which is under the max character
 646   length, and that it solves the test cases under the max number of execution
 647   steps.
 648   """
 649 
 650   def __init__(self, code_solution, make_input_fn, n=100, base=256,
 651                max_steps=5000, seed=6849275409234):
 652     super(KnownCodeBaseTask, self).__init__()
 653     # Make sure known solution is less than the code length used in experiments.
 654     assert len(code_solution) < 100
 655     self.code_solution = code_solution
 656     self.make_input_fn = make_input_fn
 657     self.n = n
 658     self.base = base
 659     self.max_steps = max_steps
 660     self.seed = seed
 661     self._test_cases = list(self._test_case_generator(code_solution))
 662 
 663   def _test_case_generator(self, code_solution):
 664     rand = random.Random(self.seed)
 665     for _ in xrange(self.n):
 666       input_case = self.make_input_fn(rand)
 667       result = bf.evaluate(
 668           code_solution, input_buffer=input_case, max_steps=self.max_steps,
 669           base=self.base, require_correct_syntax=False)
 670       if not result.success:
 671         raise RuntimeError(
 672             'Program must succeed. Failed on input: %s' % input_case)
 673       yield input_case, result.output
 674 
 675   def make_io_set(self):
 676     return copy.deepcopy(self._test_cases)

 967 class JudgeRouteCircleTask(BaseTask):
 968   """Judge route circle coding task.
 969 
 970   Code needs to determine if the given route makes a closed loop.
 971   Encoding: U = 1, R = 2, D = 3, L = 4.
 972 
 973   Based on
 974   https://leetcode.com/problems/judge-route-circle/description/
 975   """
 976   base = 256
 977   input_type = misc.IOType.integer
 978   output_type = misc.IOType.integer
 979 
 980   def __init__(self, n, max_len=12):
 981     super(type(self), self).__init__()
 982     self.eos = 0
 983     self._io_pairs = self._make_io_examples(n, max_len)
 984     self.input_type = misc.IOType.integer
 985     self.output_type = misc.IOType.integer
 986 
 987   def _solve(self, input_seq):
 988     assert input_seq[-1] == 0
 989     pos = [0, 0]  # (x, y)
 990     for move in input_seq[:-1]:
 991       assert 0 < move <= 4
 992       if move & 1 == 0:  # Left or Right.
 993         pos[0] += 3 - move  # Add or subtract 1.
 994       else:
 995         pos[1] += 2 - move  # Add or subtract 1.
 996     return [int(not pos[0] and not pos[1])]
 997 
 998   def _make_io_examples(self, n, max_len):
 999     """Generate test cases for the task."""
1000     rand = random.Random(6849275409234)  # Test cases are fixed, but varied.
1001     io_examples = []
1002     io_examples.append(([0], [1]))
1003     io_examples.append(([4, 2, 0], [1]))
1004     io_examples.append(([2, 4, 0], [1]))
1005     io_examples.append(([3, 1, 0], [1]))
1006     io_examples.append(([1, 3, 0], [1]))
1007     io_examples.append(([1, 0], [0]))
1008     io_examples.append(([2, 0], [0]))
1009     io_examples.append(([3, 0], [0]))
1010     io_examples.append(([4, 0], [0]))
1011     for _ in xrange(n):
1012       is_true = rand.randrange(2)
1013       length = rand.randrange(1, max_len + 1)
1014       if is_true:
1015         # Make a true case.
1016         length = (length >> 1) << 1  # Make even.
1017         partition = (rand.randrange(length + 1) >> 1) << 1
1018         a = partition >> 1
1019         b = (length - partition) >> 1
1020         counts = {1: a, 2: b, 3: a, 4: b}
1021       else:
1022         # Make a false case.
1023         partitions = (
1024             [0]
1025             + sorted([rand.randrange(length + 1) for _ in range(3)])
1026             + [length])
1027         counts = {n: partitions[n] - partitions[n - 1] for n in range(1, 5)}
1028         if counts[1] == counts[3] and counts[2] == counts[4]:
1029           # By chance we sampled a true case. Make it false by exchanging
1030           # one count between even and odd pairs.
1031           base = 1 + 2 * rand.randrange(2)
1032           a, b = (base, base + 1) if rand.randrange(2) else (base + 1, base)
1033           if counts[a] == length or counts[b] == 0:
1034             # If counts are at their extreme values, then swap who gets
1035             # incremented and decremented.
1036             a, b = b, a
1037           counts[a] += 1
1038           counts[b] -= 1
1039           assert counts[a] <= length and counts[b] >= 0
1040       assert sum(counts.values()) == length
1041       input_seq = [n for n in xrange(1, 5) for _ in xrange(counts[n])]
1042       rand.shuffle(input_seq)
1043       input_seq += [0]
1044       output_seq = self._solve(input_seq)
1045       assert output_seq[0] == is_true
1046       io_examples.append((input_seq, output_seq))
1047     return io_examples
1048 
1049   def make_io_set(self):
1050     return copy.deepcopy(self._io_pairs)

1332 class SplitTask(BaseTask):
1333   """Split coding task.
1334 
1335   Code needs to pipe input strings to output, but insert a 0 after every 3
1336   characters. This is in essence splitting the string into intervals of length
1337   3.
1338   """
1339   base = 28
1340   input_type = misc.IOType.string
1341   output_type = misc.IOType.integer
1342 
1343   def _splicer(self, lst, insert, interval=3):
1344     for i, item in enumerate(lst):
1345       yield item
1346       if (i + 1) % interval == 0 and i < len(lst) - 1:
1347         yield insert
1348 
1349   def __init__(self):
1350     super(type(self), self).__init__()
1351     inputs = stringlist2tokens_b27(
1352         ['hello', 'orange', 'spaghetti', 'wins', 'one'])
1353     targets = [list(self._splicer(i, 27)) for i in inputs]
1354     self._test_cases = list(zip(inputs, targets))
1355 
1356   def make_io_set(self):
1357     return copy.deepcopy(self._test_cases)
