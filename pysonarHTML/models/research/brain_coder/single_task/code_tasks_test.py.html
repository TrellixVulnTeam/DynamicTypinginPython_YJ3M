  25   def testMultiIOTaskManager(self):
  26     maxlen = 100
  27     padchr = '['
  28     task = code_tasks.make_paper_task(
  29         'print', timestep_limit=maxlen, do_code_simplification=False)
  30     reward_fns = task.rl_batch(1)
  31     r = reward_fns[0]
  32     self.assertClose(
  33         r(pad('++++++++.---.+++++++...', maxlen, padchr)).episode_rewards[-1],
  34         0.2444)
  35     self.assertClose(
  36         r(pad('++++++++.---.+++++++..+++.',
  37               maxlen, padchr)).episode_rewards[-1],
  38         1.0)
  39 
  40     task = code_tasks.make_paper_task(
  41         'print', timestep_limit=maxlen, do_code_simplification=True)
  42     reward_fns = task.rl_batch(1)
  43     r = reward_fns[0]
  44     self.assertClose(
  45         r('++++++++.---.+++++++...').episode_rewards[-1],
  46         0.2444)
  47     self.assertClose(
  48         r('++++++++.---.+++++++..+++.').episode_rewards[-1],
  49         0.935)
  50     self.assertClose(
  51         r(pad('++++++++.---.+++++++..+++.',
  52               maxlen, padchr)).episode_rewards[-1],
  53         0.75)
  54 
  55     task = code_tasks.make_paper_task(
  56         'reverse', timestep_limit=maxlen, do_code_simplification=False)
  57     reward_fns = task.rl_batch(1)
  58     r = reward_fns[0]
  59     self.assertClose(
  60         r(pad('>,>,>,.<.<.<.', maxlen, padchr)).episode_rewards[-1],
  61         0.1345)
  62     self.assertClose(
  63         r(pad(',[>,]+[,<.]', maxlen, padchr)).episode_rewards[-1],
  64         1.0)
  65 
  66     task = code_tasks.make_paper_task(
  67         'reverse', timestep_limit=maxlen, do_code_simplification=True)
  68     reward_fns = task.rl_batch(1)
  69     r = reward_fns[0]
  70     self.assertClose(r('>,>,>,.<.<.<.').episode_rewards[-1], 0.1324)
  71     self.assertClose(r(',[>,]+[,<.]').episode_rewards[-1], 0.9725)
