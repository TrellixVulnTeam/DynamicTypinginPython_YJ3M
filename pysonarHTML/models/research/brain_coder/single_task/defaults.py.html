  12 def default_config():

  75 def default_config_with_updates(config_string, do_logging=True):
  76   if do_logging:
  77     logging.info('Config string: "%s"', config_string)
  78   config = default_config()
  79   config.strict_update(config_lib.Config.parse(config_string))
  80   if do_logging:
  81     logging.info('Config:\n%s', config.pretty_str())
  82   return config
