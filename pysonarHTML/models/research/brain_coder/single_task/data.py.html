  18 class DataManager(object):
  19   """Interface between environment and model."""

  21   def __init__(self, global_config, run_number=None,
  22                do_code_simplification=False):
  23     """Constructs a DataManager.
  24 
  25     Args:
  26       global_config: A config_lib.Config instance containing all config. See
  27           config in defaults.py.
  28       run_number: Which run this is (of the same experiment). This should be set
  29           when a task cycle is defined in the config. A task cycle is a list of
  30           tasks to cycle through repeatedly, and the selected task is a function
  31           of the run number, i.e. 0-th run, 1-st run, 2-nd run, etc...
  32           This can be None if only a single task is set in the config.
  33       do_code_simplification: When global_config.env.config_for_iclr is True,
  34           use this option to create code simplification (code golf) tasks, vs
  35           fixed length coding tasks. If True, a task with code simplification
  36           reward will be constructed.
  37 
  38     Raises:
  39       ValueError: If global_config.env.task and global_config.env.task_cycle
  40           are both set, or both not set. Only one should be given.
  41       ValueError: If global_config.env.task_cycle is set but run_number is None.
  42     """
  43     env_config = global_config.env
  44     self.batch_size = global_config.batch_size
  45 
  46     if env_config.task_cycle:
  47       if env_config.task:
  48         raise ValueError('Do not set both `task` and `task_cycle`.')
  49       if run_number is None:
  50         raise ValueError('Do not use task_cycle for single-run experiment.')
  51       index = run_number % len(env_config.task_cycle)
  52       self.task_name = env_config.task_cycle[index]
  53       logging.info('run_number: %d,  task_cycle index: %d', run_number, index)
  54       logging.info('task_cycle: %s', env_config.task_cycle)
  55     elif env_config.task:
  56       self.task_name = env_config.task
  57     else:
  58       raise ValueError('Either `task` or `task_cycle` must be set.')
  59     logging.info('Task for this run: "%s"', self.task_name)
  60 
  61     logging.info('config_for_iclr=True; do_code_simplification=%s',
  62                  do_code_simplification)
  63     self.rl_task = code_tasks.make_task(
  64         task_name=self.task_name,
  65         override_kwargs=ast.literal_eval(env_config.task_kwargs),
  66         max_code_length=global_config.timestep_limit,
  67         require_correct_syntax=env_config.correct_syntax,
  68         do_code_simplification=do_code_simplification,
  69         correct_bonus=env_config.task_manager_config.correct_bonus,
  70         code_length_bonus=env_config.task_manager_config.code_length_bonus)
  71 
  72   def sample_rl_batch(self):
  73     """Create reward functions from the current task.
  74 
  75     Returns:
  76       RLBatch namedtuple instance, which holds functions and information for
  77       a minibatch of episodes.
  78       * reward_fns: A reward function for each episode. Maps code string to
  79           reward.
  80       * batch_size: Number of episodes in this minibatch.
  81       * good_reward: Estimated threshold of rewards which indicate the algorithm
  82           is starting to solve the task. This is a heuristic that tries to
  83           reduce the amount of stuff written to disk.
  84     """
  85     reward_fns = self.rl_task.rl_batch(self.batch_size)
  86     return RLBatch(
  87         reward_fns=reward_fns,
  88         batch_size=self.batch_size,
  89         good_reward=self.rl_task.good_reward)
