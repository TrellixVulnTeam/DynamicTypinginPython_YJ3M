  33 def rshift_time(tensor_2d, fill=misc.BF_EOS_INT):
  34   """Right shifts a 2D tensor along the time dimension (axis-1)."""
  35   dim_0 = tf.shape(tensor_2d)[0]
  36   fill_tensor = tf.fill([dim_0, 1], fill)
  37   return tf.concat([fill_tensor, tensor_2d[:, :-1]], axis=1)

  40 def join(a, b):
  41   # Concat a and b along 0-th dim.
  42   if a is None or len(a) == 0:  # pylint: disable=g-explicit-length-test
  43     return b
  44   if b is None or len(b) == 0:  # pylint: disable=g-explicit-length-test
  45     return a
  46   return np.concatenate((a, b))

  49 def make_optimizer(kind, lr):
  50   if kind == 'sgd':
  51     return tf.train.GradientDescentOptimizer(lr)
  52   elif kind == 'adam':
  53     return tf.train.AdamOptimizer(lr)
  54   elif kind == 'rmsprop':
  55     return tf.train.RMSPropOptimizer(learning_rate=lr, decay=0.99)
  56   else:
  57     raise ValueError('Optimizer type "%s" not recognized.' % kind)

  60 class LinearWrapper(tf.contrib.rnn.RNNCell):
  61   """RNNCell wrapper that adds a linear layer to the output."""
  62 
  63   def __init__(self, cell, output_size, dtype=tf.float32, suppress_index=None):
  64     self.cell = cell
  65     self._output_size = output_size
  66     self._dtype = dtype
  67     self._suppress_index = suppress_index
  68     self.smallest_float = -2.4e38
  69 
  70   def __call__(self, inputs, state, scope=None):
  71     with tf.variable_scope(type(self).__name__):
  72       outputs, state = self.cell(inputs, state, scope=scope)
  73       logits = tf.matmul(
  74           outputs,
  75           tf.get_variable('w_output',
  76                           [self.cell.output_size, self.output_size],
  77                           dtype=self._dtype))
  78       if self._suppress_index is not None:
  79         # Replace the target index with -inf, so that it never gets selected.
  80         batch_size = tf.shape(logits)[0]
  81         logits = tf.concat(
  82             [logits[:, :self._suppress_index],
  83              tf.fill([batch_size, 1], self.smallest_float),
  84              logits[:, self._suppress_index + 1:]],
  85             axis=1)
  86 
  87     return logits, state
  88 
  89   @property
  90   def output_size(self):
  91     return self._output_size
  92 
  93   @property
  94   def state_size(self):
  95     return self.cell.state_size
  96 
  97   def zero_state(self, batch_size, dtype):
  98     return self.cell.zero_state(batch_size, dtype)

 226         tf.contrib.rnn.MultiRNNCell(
 227             [tf.contrib.rnn.BasicLSTMCell(cell_size)

 234             [tf.contrib.rnn.BasicLSTMCell(cell_size)

 236         1,
 237         dtype=dtype)

 239     obs_embedding_scope = 'obs_embed'
 240     with tf.variable_scope(
 241         obs_embedding_scope,
 242         initializer=tf.random_uniform_initializer(minval=-1.0, maxval=1.0)):
 243       obs_embeddings = tf.get_variable(
 244           'embeddings',

 246           dtype=dtype, trainable=self.embeddings_trainable)
 247       self.obs_embeddings = obs_embeddings

 253     initial_state = tf.fill([batch_size], misc.BF_EOS_INT)
 254     def loop_fn(loop_time, cell_output, cell_state, loop_state):
 255       """Function called by tf.nn.raw_rnn to instantiate body of the while_loop.
 256 
 257       See https://www.tensorflow.org/api_docs/python/tf/nn/raw_rnn for more
 258       information.
 259 
 260       When time is 0, and cell_output, cell_state, loop_state are all None,
 261       `loop_fn` will create the initial input, internal cell state, and loop
 262       state. When time > 0, `loop_fn` will operate on previous cell output,
 263       state, and loop state.
 264 
 265       Args:
 266         loop_time: A scalar tensor holding the current timestep (zero based
 267             counting).
 268         cell_output: Output of the raw_rnn cell at the current timestep.
 269         cell_state: Cell internal state at the current timestep.
 270         loop_state: Additional loop state. These tensors were returned by the
 271             previous call to `loop_fn`.
 272 
 273       Returns:
 274         elements_finished: Bool tensor of shape [batch_size] which marks each
 275             sequence in the batch as being finished or not finished.
 276         next_input: A tensor containing input to be fed into the cell at the
 277             next timestep.
 278         next_cell_state: Cell internal state to be fed into the cell at the
 279             next timestep.
 280         emit_output: Tensor to be added to the TensorArray returned by raw_rnn
 281             as output from the while_loop.
 282         next_loop_state: Additional loop state. These tensors will be fed back
 283             into the next call to `loop_fn` as `loop_state`.
 284       """
 285       if cell_output is None:  # 0th time step.
 286         next_cell_state = self.policy_cell.zero_state(batch_size, dtype)
 287         elements_finished = tf.zeros([batch_size], tf.bool)
 288         output_lengths = tf.ones([batch_size], dtype=tf.int32)
 289         next_input = tf.gather(obs_embeddings, initial_state)
 290         emit_output = None
 291         next_loop_state = (
 292             tf.TensorArray(dtype=tf.int32, size=0, dynamic_size=True),
 293             output_lengths,
 294             elements_finished
 295         )
 296       else:
 297         scaled_logits = cell_output * config.softmax_tr  # Scale temperature.
 298         prev_chosen, prev_output_lengths, prev_elements_finished = loop_state
 299         next_cell_state = cell_state
 300         chosen_outputs = tf.to_int32(tf.where(
 301             tf.logical_not(prev_elements_finished),
 302             tf.multinomial(logits=scaled_logits, num_samples=1)[:, 0],
 303             tf.zeros([batch_size], dtype=tf.int64)))
 304         elements_finished = tf.logical_or(
 305             tf.equal(chosen_outputs, misc.BF_EOS_INT),
 306             loop_time >= global_config.timestep_limit)
 307         output_lengths = tf.where(
 308             elements_finished,
 309             prev_output_lengths,
 310             # length includes EOS token. empty seq has len 1.
 311             tf.tile(tf.expand_dims(loop_time + 1, 0), [batch_size])
 312         )
 313         next_input = tf.gather(obs_embeddings, chosen_outputs)
 314         emit_output = scaled_logits
 315         next_loop_state = (prev_chosen.write(loop_time - 1, chosen_outputs),
 316                            output_lengths,
 317                            tf.logical_or(prev_elements_finished,
 318                                          elements_finished))
 319       return (elements_finished, next_input, next_cell_state, emit_output,
 320               next_loop_state)
 321 
 322     with tf.variable_scope('policy'):
 323       (decoder_outputs_ta,
 324        _,  # decoder_state
 325        (sampled_output_ta, output_lengths, _)) = tf.nn.raw_rnn(
 326            cell=self.policy_cell,
 327            loop_fn=loop_fn)
 328     policy_logits = tf.transpose(decoder_outputs_ta.stack(), (1, 0, 2),
 329                                  name='policy_logits')
 330     sampled_tokens = tf.transpose(sampled_output_ta.stack(), (1, 0),
 331                                   name='sampled_tokens')
 332     # Add SOS to beginning of the sequence.
 333     rshift_sampled_tokens = rshift_time(sampled_tokens, fill=misc.BF_EOS_INT)
 334 
 335     # Initial state is 0, 2nd state is first token.
 336     # Note: If value of last state is computed, this will be used as bootstrap.
 337     if self.a2c:
 338       with tf.variable_scope('value'):
 339         value_output, _ = tf.nn.dynamic_rnn(
 340             self.value_cell,
 341             tf.gather(obs_embeddings, rshift_sampled_tokens),
 342             sequence_length=output_lengths,
 343             dtype=dtype)
 344       value = tf.squeeze(value_output, axis=[2])
 345     else:
 346       value = tf.zeros([], dtype=dtype)
 347 
 348     # for sampling actions from the agent, and which told tensors for doing
 349     # gradient updates on the agent.
 350     self.sampled_batch = AttrDict(
 351         logits=policy_logits,
 352         value=value,
 353         tokens=sampled_tokens,
 354         episode_lengths=output_lengths,
 355         probs=tf.nn.softmax(policy_logits),
 356         log_probs=tf.nn.log_softmax(policy_logits))
 357 
 358     # adjusted_lengths can be less than the full length of each episode.
 359     # Use this to train on only part of an episode (starting from t=0).
 360     self.adjusted_lengths = tf.placeholder(
 361         tf.int32, [None], name='adjusted_lengths')
 362     self.policy_multipliers = tf.placeholder(
 363         dtype,
 364         [None, None],
 365         name='policy_multipliers')
 366     # Empirical value, i.e. discounted sum of observed future rewards from each
 367     # time step in the episode.
 368     self.empirical_values = tf.placeholder(
 369         dtype,
 370         [None, None],
 371         name='empirical_values')
 372 
 373     # Off-policy training. Just add supervised loss to the RL loss.
 374     self.off_policy_targets = tf.placeholder(
 375         tf.int32,
 376         [None, None],
 377         name='off_policy_targets')
 378     self.off_policy_target_lengths = tf.placeholder(
 379         tf.int32, [None], name='off_policy_target_lengths')
 380 
 381     self.actions = tf.placeholder(tf.int32, [None, None], name='actions')
 382     # Add SOS to beginning of the sequence.
 383     inputs = rshift_time(self.actions, fill=misc.BF_EOS_INT)
 384     with tf.variable_scope('policy', reuse=True):
 385       logits, _ = tf.nn.dynamic_rnn(
 386           self.policy_cell, tf.gather(obs_embeddings, inputs),
 387           sequence_length=self.adjusted_lengths,
 388           dtype=dtype)
 389 
 390     if self.a2c:
 391       with tf.variable_scope('value', reuse=True):
 392         value_output, _ = tf.nn.dynamic_rnn(
 393             self.value_cell,
 394             tf.gather(obs_embeddings, inputs),
 395             sequence_length=self.adjusted_lengths,
 396             dtype=dtype)
 397       value2 = tf.squeeze(value_output, axis=[2])
 398     else:
 399       value2 = tf.zeros([], dtype=dtype)
 400 
 401     self.given_batch = AttrDict(
 402         logits=logits,
 403         value=value2,
 404         tokens=sampled_tokens,
 405         episode_lengths=self.adjusted_lengths,
 406         probs=tf.nn.softmax(logits),
 407         log_probs=tf.nn.log_softmax(logits))
 408 
 409     # Episode masks.
 410     max_episode_length = tf.shape(self.actions)[1]
 411     # range_row shape: [1, max_episode_length]
 412     range_row = tf.expand_dims(tf.range(max_episode_length), 0)
 413     episode_masks = tf.cast(
 414         tf.less(range_row, tf.expand_dims(self.given_batch.episode_lengths, 1)),
 415         dtype=dtype)
 416     episode_masks_3d = tf.expand_dims(episode_masks, 2)
 417 
 418     # Length adjusted episodes.
 419     self.a_probs = a_probs = self.given_batch.probs * episode_masks_3d
 420     self.a_log_probs = a_log_probs = (
 421         self.given_batch.log_probs * episode_masks_3d)
 422     self.a_value = a_value = self.given_batch.value * episode_masks
 423     self.a_policy_multipliers = a_policy_multipliers = (
 424         self.policy_multipliers * episode_masks)
 425     if self.a2c:
 426       self.a_empirical_values = a_empirical_values = (
 427           self.empirical_values * episode_masks)
 428 
 429     # pi_loss is scalar
 430     acs_onehot = tf.one_hot(self.actions, self.action_space, dtype=dtype)
 431     self.acs_onehot = acs_onehot
 432     chosen_masked_log_probs = acs_onehot * a_log_probs
 433     pi_target = tf.expand_dims(a_policy_multipliers, -1)
 434     pi_loss_per_step = chosen_masked_log_probs * pi_target  # Maximize.
 435     self.pi_loss = pi_loss = (
 436         -tf.reduce_mean(tf.reduce_sum(pi_loss_per_step, axis=[1, 2]), axis=0)
 437         * MAGIC_LOSS_MULTIPLIER)  # Minimize.
 438     assert len(self.pi_loss.shape) == 0  # pylint: disable=g-explicit-length-test
 439 
 440     # shape: [batch_size, time]
 441     self.chosen_log_probs = tf.reduce_sum(chosen_masked_log_probs, axis=2)
 442     self.chosen_probs = tf.reduce_sum(acs_onehot * a_probs, axis=2)
 443 
 444     # loss of value function
 445     if self.a2c:
 446       vf_loss_per_step = tf.square(a_value - a_empirical_values)
 447       self.vf_loss = vf_loss = (
 448           tf.reduce_mean(tf.reduce_sum(vf_loss_per_step, axis=1), axis=0)
 449           * MAGIC_LOSS_MULTIPLIER)  # Minimize.
 450       assert len(self.vf_loss.shape) == 0  # pylint: disable=g-explicit-length-test
 451     else:
 452       self.vf_loss = vf_loss = 0.0
 453 
 454     # Maximize entropy regularizer
 455     self.entropy = entropy = (
 456         -tf.reduce_mean(
 457             tf.reduce_sum(a_probs * a_log_probs, axis=[1, 2]), axis=0)
 458         * MAGIC_LOSS_MULTIPLIER)  # Maximize
 459     self.negentropy = -entropy  # Minimize negentropy.
 460     assert len(self.negentropy.shape) == 0  # pylint: disable=g-explicit-length-test
 461 
 462     # off-policy loss
 463     self.offp_switch = tf.placeholder(dtype, [], name='offp_switch')
 464     if self.top_episodes is not None:
 465       # Add SOS to beginning of the sequence.
 466       offp_inputs = tf.gather(obs_embeddings,
 467                               rshift_time(self.off_policy_targets,
 468                                           fill=misc.BF_EOS_INT))
 469       with tf.variable_scope('policy', reuse=True):
 470         offp_logits, _ = tf.nn.dynamic_rnn(
 471             self.policy_cell, offp_inputs, self.off_policy_target_lengths,
 472             dtype=dtype)  # shape: [batch_size, time, action_space]
 473       topk_loss_per_step = tf.nn.sparse_softmax_cross_entropy_with_logits(
 474           labels=self.off_policy_targets,
 475           logits=offp_logits,
 476           name='topk_loss_per_logit')
 477       # Take mean over batch dimension so that the loss multiplier strength is
 478       # independent of batch size. Sum over time dimension.
 479       topk_loss = tf.reduce_mean(
 480           tf.reduce_sum(topk_loss_per_step, axis=1), axis=0)
 481       assert len(topk_loss.shape) == 0  # pylint: disable=g-explicit-length-test
 482       self.topk_loss = topk_loss * self.offp_switch
 483       logging.info('Including off policy loss.')
 484     else:
 485       self.topk_loss = topk_loss = 0.0
 486 
 487     self.entropy_hparam = tf.constant(
 488         config.entropy_beta, dtype=dtype, name='entropy_beta')
 489 
 490     self.pi_loss_term = pi_loss * self.pi_loss_hparam
 491     self.vf_loss_term = vf_loss * self.vf_loss_hparam
 492     self.entropy_loss_term = self.negentropy * self.entropy_hparam
 493     self.topk_loss_term = self.topk_loss_hparam * topk_loss
 494     self.loss = (
 495         self.pi_loss_term
 496         + self.vf_loss_term
 497         + self.entropy_loss_term
 498         + self.topk_loss_term)
 499 
 500     params = tf.get_collection(tf.GraphKeys.TRAINABLE_VARIABLES,
 501                                tf.get_variable_scope().name)
 502     self.trainable_variables = params
 503     self.sync_variables = self.trainable_variables
 504     non_embedding_params = [p for p in params
 505                             if obs_embedding_scope not in p.name]
 506     self.non_embedding_params = non_embedding_params
 507     self.params = params
 508 
 509     if config.regularizer:
 510       logging.info('Adding L2 regularizer with scale %.2f.',
 511                    config.regularizer)
 512       self.regularizer = config.regularizer * sum(
 513           tf.nn.l2_loss(w) for w in non_embedding_params)
 514       self.loss += self.regularizer
 515     else:
 516       logging.info('Skipping regularizer.')
 517       self.regularizer = 0.0
 518 
 519     # Only build gradients graph for local model.
 520     if self.is_local:
 521       unclipped_grads = tf.gradients(self.loss, params)
 522       self.dense_unclipped_grads = [
 523           tf.convert_to_tensor(g) for g in unclipped_grads]
 524       self.grads, self.global_grad_norm = tf.clip_by_global_norm(
 525           unclipped_grads, config.grad_clip_threshold)
 526       self.gradients_dict = dict(zip(params, self.grads))
 527       self.optimizer = make_optimizer(config.optimizer, self.learning_rate)
 528       self.all_variables = tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES,
 529                                              tf.get_variable_scope().name)
 530 
 531     self.do_iw_summaries = do_iw_summaries
 532     if self.do_iw_summaries:

 537                                              'log_iw_policy_ph')

 539                                                'log_prob_replay_ph')
 540       self.log_prob_policy_ph = tf.placeholder(tf.float32, [b],
 541                                                'log_prob_policy_ph')

 543           tf.float32, [b], 'log_norm_replay_weights_ph')
 544       self.iw_summary_op = tf.summary.merge([
 545           tf.summary.histogram('is/log_iw_replay', self.log_iw_replay_ph),

 547           tf.summary.histogram('is/log_prob_replay', self.log_prob_replay_ph),

 553   def make_summary_ops(self):
 554     """Construct summary ops for the model."""
 555     # size = number of timesteps across entire batch. Number normalized by size
 556     # will not be affected by the amount of padding at the ends of sequences
 557     # in the batch.
 558     size = tf.cast(
 559         tf.reduce_sum(self.given_batch.episode_lengths), dtype=self.dtype)
 560     offp_size = tf.cast(tf.reduce_sum(self.off_policy_target_lengths),
 561                         dtype=self.dtype)
 562     scope_prefix = self.parent_scope_name
 563 
 564     def _remove_prefix(prefix, name):
 565       assert name.startswith(prefix)
 566       return name[len(prefix):]
 567 
 568     # RL summaries.
 569     self.rl_summary_op = tf.summary.merge(
 570         [tf.summary.scalar('model/policy_loss', self.pi_loss / size),
 571          tf.summary.scalar('model/value_loss', self.vf_loss / size),
 572          tf.summary.scalar('model/topk_loss', self.topk_loss / offp_size),
 573          tf.summary.scalar('model/entropy', self.entropy / size),
 574          tf.summary.scalar('model/loss', self.loss / size),
 575          tf.summary.scalar('model/grad_norm',
 576                            tf.global_norm(self.grads)),
 577          tf.summary.scalar('model/unclipped_grad_norm', self.global_grad_norm),
 578          tf.summary.scalar('model/non_embedding_var_norm',
 579                            tf.global_norm(self.non_embedding_params)),
 580          tf.summary.scalar('hparams/entropy_beta', self.entropy_hparam),
 581          tf.summary.scalar('hparams/topk_loss_hparam', self.topk_loss_hparam),
 582          tf.summary.scalar('hparams/learning_rate', self.learning_rate),
 583          tf.summary.scalar('model/trainable_var_norm',
 584                            tf.global_norm(self.trainable_variables)),
 585          tf.summary.scalar('loss/loss', self.loss),
 586          tf.summary.scalar('loss/entropy', self.entropy_loss_term),
 587          tf.summary.scalar('loss/vf', self.vf_loss_term),
 588          tf.summary.scalar('loss/policy', self.pi_loss_term),
 589          tf.summary.scalar('loss/offp', self.topk_loss_term)] +
 590         [tf.summary.scalar(
 591             'param_norms/' + _remove_prefix(scope_prefix + '/', p.name),
 592             tf.norm(p))
 593          for p in self.params] +
 594         [tf.summary.scalar(
 595             'grad_norms/' + _remove_prefix(scope_prefix + '/', p.name),
 596             tf.norm(g))
 597          for p, g in zip(self.params, self.grads)] +
 598         [tf.summary.scalar(
 599             'unclipped_grad_norms/' + _remove_prefix(scope_prefix + '/',
 600                                                      p.name),
 601             tf.norm(g))
 602          for p, g in zip(self.params, self.dense_unclipped_grads)])

 609                        input_case, code_output, code, reason):
 610     """Logs summary about a single episode and creates a text_summary for TB.
 611 
 612     Args:
 613       session: tf.Session instance.

 645   def _rl_reward_summary(self, total_rewards):

 673   def _iw_summary(self, session, replay_iw, replay_log_probs,
 674                   norm_replay_weights, on_policy_iw,
 675                   on_policy_log_probs):
 676     """Compute summaries for importance weights at a given batch.
 677 
 678     Args:
 679       session: tf.Session instance.
 680       replay_iw: Importance weights for episodes from replay buffer.
 681       replay_log_probs: Total log probabilities of the replay episodes under the
 682           current policy.
 683       norm_replay_weights: Normalized replay weights, i.e. values in `replay_iw`
 684           divided by the total weight in the entire replay buffer. Note, this is
 685           also the probability of selecting each episode from the replay buffer
 686           (in a roulette wheel replay buffer).
 687       on_policy_iw: Importance weights for episodes sampled from the current
 688           policy.
 689       on_policy_log_probs: Total log probabilities of the on-policy episodes
 690           under the current policy.
 691 
 692     Returns:
 693       Serialized TF summaries. Use a summary writer to write these summaries to
 694       disk.
 695     """
 696     return session.run(
 697         self.iw_summary_op,
 698         {self.log_iw_replay_ph: np.log(replay_iw),
 699          self.log_iw_policy_ph: np.log(on_policy_iw),
 700          self.log_norm_replay_weights_ph: np.log(norm_replay_weights),
 701          self.log_prob_replay_ph: replay_log_probs,
 702          self.log_prob_policy_ph: on_policy_log_probs})
 703 
 704   def _compute_iw(self, policy_log_probs, replay_weights):
 705     """Compute importance weights for a batch of episodes.
 706 
 707     Arguments are iterables of length batch_size.
 708 
 709     Args:
 710       policy_log_probs: Log probability of each episode under the current
 711           policy.
 712       replay_weights: Weight of each episode in the replay buffer. 0 for
 713           episodes not sampled from the replay buffer (i.e. sampled from the
 714           policy).
 715 
 716     Returns:
 717       Numpy array of shape [batch_size] containing the importance weight for
 718       each episode in the batch.
 719     """
 720     log_total_replay_weight = log(self.experience_replay.total_weight)
 721 
 722     # importance weight
 723     # = 1 / [(1 - a) + a * exp(log(replay_weight / total_weight / p))]
 724     # = 1 / ((1-a) + a*q/p)
 725     a = float(self.replay_alpha)
 726     a_com = 1.0 - a  # compliment of a
 727     importance_weights = np.asarray(
 728         [1.0 / (a_com
 729                 + a * exp((log(replay_weight) - log_total_replay_weight)
 730                           - log_p))
 731          if replay_weight > 0 else 1.0 / a_com
 732          for log_p, replay_weight
 733          in zip(policy_log_probs, replay_weights)])
 734     return importance_weights
 735 
 736   def update_step(self, session, rl_batch, train_op, global_step_op,
 737                   return_gradients=False):
 738     """Perform gradient update on the model.
 739 
 740     Args:
 741       session: tf.Session instance.
 742       rl_batch: RLBatch instance from data.py. Use DataManager to create a
 743           RLBatch for each call to update_step. RLBatch contains a batch of
 744           tasks.
 745       train_op: A TF op which will perform the gradient update. LMAgent does not
 746           own its training op, so that trainers can do distributed training
 747           and construct a specialized training op.
 748       global_step_op: A TF op which will return the current global step when
 749           run (should not increment it).
 750       return_gradients: If True, the gradients will be saved and returned from
 751           this method call. This is useful for testing.
 752 
 753     Returns:
 754       Results from the update step in a UpdateStepResult namedtuple, including
 755       global step, global NPE, serialized summaries, and optionally gradients.
 756     """
 757     assert self.is_local
 758 
 759     # Do update for REINFORCE or REINFORCE + replay buffer.
 760     if self.experience_replay is None:
 761       # Train with on-policy REINFORCE.
 762 
 763       # Sample new programs from the policy.
 764       num_programs_from_policy = rl_batch.batch_size
 765       (batch_actions,
 766        batch_values,
 767        episode_lengths) = session.run(
 768            [self.sampled_batch.tokens, self.sampled_batch.value,
 769             self.sampled_batch.episode_lengths])
 770       if episode_lengths.size == 0:
 771         # This should not happen.
 772         logging.warn(
 773             'Shapes:\n'
 774             'batch_actions.shape: %s\n'
 775             'batch_values.shape: %s\n'
 776             'episode_lengths.shape: %s\n',
 777             batch_actions.shape, batch_values.shape, episode_lengths.shape)
 778 
 779       # Compute rewards.
 780       code_scores = compute_rewards(
 781           rl_batch, batch_actions, episode_lengths)
 782       code_strings = code_scores.code_strings
 783       batch_tot_r = code_scores.total_rewards
 784       test_cases = code_scores.test_cases
 785       code_outputs = code_scores.code_outputs
 786       reasons = code_scores.reasons
 787 
 788       # Process on-policy samples.
 789       batch_targets, batch_returns = process_episodes(
 790           code_scores.batch_rewards, episode_lengths, a2c=self.a2c,
 791           baselines=self.ema_by_len,
 792           batch_values=batch_values)
 793       batch_policy_multipliers = batch_targets
 794       batch_emp_values = batch_returns if self.a2c else [[]]
 795       adjusted_lengths = episode_lengths
 796 
 797       if self.top_episodes:
 798         assert len(self.top_episodes) > 0  # pylint: disable=g-explicit-length-test
 799         off_policy_targets = [
 800             item for item, _
 801             in self.top_episodes.random_sample(self.topk_batch_size)]
 802         off_policy_target_lengths = [len(t) for t in off_policy_targets]
 803         off_policy_targets = utils.stack_pad(off_policy_targets, pad_axes=0,
 804                                              dtype=np.int32)
 805         offp_switch = 1
 806       else:
 807         off_policy_targets = [[0]]
 808         off_policy_target_lengths = [1]
 809         offp_switch = 0
 810 
 811       fetches = {
 812           'global_step': global_step_op,
 813           'program_count': self.program_count,
 814           'summaries': self.rl_summary_op,
 815           'train_op': train_op,
 816           'gradients': self.gradients_dict if return_gradients else self.no_op}
 817       fetched = session.run(
 818           fetches,
 819           {self.actions: batch_actions,
 820            self.empirical_values: batch_emp_values,
 821            self.policy_multipliers: batch_policy_multipliers,
 822            self.adjusted_lengths: adjusted_lengths,
 823            self.off_policy_targets: off_policy_targets,
 824            self.off_policy_target_lengths: off_policy_target_lengths,
 825            self.offp_switch: offp_switch})
 826 
 827       combined_adjusted_lengths = adjusted_lengths
 828       combined_returns = batch_returns
 829     else:
 830       # Train with REINFORCE + off-policy replay buffer by using importance
 831       # sampling.
 832 
 833       # Sample new programs from the policy.
 834       # Note: batch size is constant. A full batch will be sampled, but not all
 835       # programs will be executed and added to the replay buffer. Those which
 836       # are not executed will be discarded and not counted.
 837       batch_actions, batch_values, episode_lengths, log_probs = session.run(
 838           [self.sampled_batch.tokens, self.sampled_batch.value,
 839            self.sampled_batch.episode_lengths, self.sampled_batch.log_probs])
 840       if episode_lengths.size == 0:
 841         # This should not happen.
 842         logging.warn(
 843             'Shapes:\n'
 844             'batch_actions.shape: %s\n'
 845             'batch_values.shape: %s\n'
 846             'episode_lengths.shape: %s\n',
 847             batch_actions.shape, batch_values.shape, episode_lengths.shape)
 848 
 849       # Sample from experince replay buffer
 850       empty_replay_buffer = (
 851           self.experience_replay.is_empty()
 852           if self.experience_replay is not None else True)
 853       num_programs_from_replay_buff = (
 854           self.num_replay_per_batch if not empty_replay_buffer else 0)
 855       num_programs_from_policy = (
 856           rl_batch.batch_size - num_programs_from_replay_buff)
 857       if (not empty_replay_buffer) and num_programs_from_replay_buff:
 858         result = self.experience_replay.sample_many(
 859             num_programs_from_replay_buff)
 860         experience_samples, replay_weights = zip(*result)
 861         (replay_actions,
 862          replay_rewards,
 863          _,  # log probs
 864          replay_adjusted_lengths) = zip(*experience_samples)
 865 
 866         replay_batch_actions = utils.stack_pad(replay_actions, pad_axes=0,
 867                                                dtype=np.int32)
 868 
 869         # compute log probs for replay samples under current policy
 870         all_replay_log_probs, = session.run(
 871             [self.given_batch.log_probs],
 872             {self.actions: replay_batch_actions,
 873              self.adjusted_lengths: replay_adjusted_lengths})
 874         replay_log_probs = [
 875             np.choose(replay_actions[i], all_replay_log_probs[i, :l].T).sum()
 876             for i, l in enumerate(replay_adjusted_lengths)]
 877       else:
 878         # Replay buffer is empty. Do not sample from it.
 879         replay_actions = None
 880         replay_policy_multipliers = None
 881         replay_adjusted_lengths = None
 882         replay_log_probs = None
 883         replay_weights = None
 884         replay_returns = None
 885         on_policy_weights = [0] * num_programs_from_replay_buff
 886 
 887       assert not self.a2c  # TODO(danabo): Support A2C with importance sampling.
 888 
 889       # Compute rewards.
 890       code_scores = compute_rewards(
 891           rl_batch, batch_actions, episode_lengths,
 892           batch_size=num_programs_from_policy)
 893       code_strings = code_scores.code_strings
 894       batch_tot_r = code_scores.total_rewards
 895       test_cases = code_scores.test_cases
 896       code_outputs = code_scores.code_outputs
 897       reasons = code_scores.reasons
 898 
 899       # Process on-policy samples.
 900       p = num_programs_from_policy
 901       batch_targets, batch_returns = process_episodes(
 902           code_scores.batch_rewards, episode_lengths[:p], a2c=False,
 903           baselines=self.ema_by_len)
 904       batch_policy_multipliers = batch_targets
 905       batch_emp_values = [[]]
 906       on_policy_returns = batch_returns
 907 
 908       # Process off-policy samples.
 909       if (not empty_replay_buffer) and num_programs_from_replay_buff:
 910         offp_batch_rewards = [
 911             [0.0] * (l - 1) + [r]
 912             for l, r in zip(replay_adjusted_lengths, replay_rewards)]
 913         assert len(offp_batch_rewards) == num_programs_from_replay_buff
 914         assert len(replay_adjusted_lengths) == num_programs_from_replay_buff
 915         replay_batch_targets, replay_returns = process_episodes(
 916             offp_batch_rewards, replay_adjusted_lengths, a2c=False,
 917             baselines=self.ema_by_len)
 918         # Convert 2D array back into ragged 2D list.
 919         replay_policy_multipliers = [
 920             replay_batch_targets[i, :l]
 921             for i, l
 922             in enumerate(
 923                 replay_adjusted_lengths[:num_programs_from_replay_buff])]
 924 
 925       adjusted_lengths = episode_lengths[:num_programs_from_policy]
 926 
 927       if self.top_episodes:
 928         assert len(self.top_episodes) > 0  # pylint: disable=g-explicit-length-test
 929         off_policy_targets = [
 930             item for item, _
 931             in self.top_episodes.random_sample(self.topk_batch_size)]
 932         off_policy_target_lengths = [len(t) for t in off_policy_targets]
 933         off_policy_targets = utils.stack_pad(off_policy_targets, pad_axes=0,
 934                                              dtype=np.int32)
 935         offp_switch = 1
 936       else:
 937         off_policy_targets = [[0]]
 938         off_policy_target_lengths = [1]
 939         offp_switch = 0
 940 
 941       # On-policy episodes.
 942       if num_programs_from_policy:
 943         separate_actions = [
 944             batch_actions[i, :l]
 945             for i, l in enumerate(adjusted_lengths)]
 946         chosen_log_probs = [
 947             np.choose(separate_actions[i], log_probs[i, :l].T)
 948             for i, l in enumerate(adjusted_lengths)]
 949         new_experiences = [
 950             (separate_actions[i],
 951              batch_tot_r[i],
 952              chosen_log_probs[i].sum(), l)
 953             for i, l in enumerate(adjusted_lengths)]
 954         on_policy_policy_multipliers = [
 955             batch_policy_multipliers[i, :l]
 956             for i, l in enumerate(adjusted_lengths)]
 957         (on_policy_actions,
 958          _,  # rewards
 959          on_policy_log_probs,
 960          on_policy_adjusted_lengths) = zip(*new_experiences)
 961       else:
 962         new_experiences = []
 963         on_policy_policy_multipliers = []
 964         on_policy_actions = []
 965         on_policy_log_probs = []
 966         on_policy_adjusted_lengths = []
 967 
 968       if (not empty_replay_buffer) and num_programs_from_replay_buff:
 969         # Look for new experiences in replay buffer. Assign weight if an episode
 970         # is in the buffer.
 971         on_policy_weights = [0] * num_programs_from_policy
 972         for i, cs in enumerate(code_strings):
 973           if self.experience_replay.has_key(cs):
 974             on_policy_weights[i] = self.experience_replay.get_weight(cs)
 975 
 976       # Randomly select on-policy or off policy episodes to train on.
 977       combined_actions = join(replay_actions, on_policy_actions)
 978       combined_policy_multipliers = join(
 979           replay_policy_multipliers, on_policy_policy_multipliers)
 980       combined_adjusted_lengths = join(
 981           replay_adjusted_lengths, on_policy_adjusted_lengths)
 982       combined_returns = join(replay_returns, on_policy_returns)
 983       combined_actions = utils.stack_pad(combined_actions, pad_axes=0)
 984       combined_policy_multipliers = utils.stack_pad(combined_policy_multipliers,
 985                                                     pad_axes=0)
 986       # P
 987       combined_on_policy_log_probs = join(replay_log_probs, on_policy_log_probs)
 988       # Q
 989       # Assume weight is zero for all sequences sampled from the policy.
 990       combined_q_weights = join(replay_weights, on_policy_weights)
 991 
 992       # Importance adjustment. Naive formulation:
 993       # E_{x~p}[f(x)] ~= 1/N sum_{x~p}(f(x)) ~= 1/N sum_{x~q}(f(x) * p(x)/q(x)).
 994       # p(x) is the policy, and q(x) is the off-policy distribution, i.e. replay
 995       # buffer distribution. Importance weight w(x) = p(x) / q(x).
 996 
 997       # Instead of sampling from the replay buffer only, we sample from a
 998       # mixture distribution of the policy and replay buffer.
 999       # We are sampling from the mixture a*q(x) + (1-a)*p(x), where 0 <= a <= 1.
1000       # Thus the importance weight w(x) = p(x) / (a*q(x) + (1-a)*p(x))
1001       # = 1 / ((1-a) + a*q(x)/p(x)) where q(x) is 0 for x sampled from the
1002       #                             policy.
1003       # Note: a = self.replay_alpha
1004       if empty_replay_buffer:
1005         # The replay buffer is empty.
1006         # Do no gradient update this step. The replay buffer will have stuff in
1007         # it next time.
1008         combined_policy_multipliers *= 0
1009       elif not num_programs_from_replay_buff:
1010         combined_policy_multipliers = np.ones([len(combined_actions), 1],
1011                                               dtype=np.float32)
1012       else:
1013         # If a < 1 compute importance weights
1014         # importance weight
1015         # = 1 / [(1 - a) + a * exp(log(replay_weight / total_weight / p))]
1016         # = 1 / ((1-a) + a*q/p)
1017         importance_weights = self._compute_iw(combined_on_policy_log_probs,
1018                                               combined_q_weights)
1019         if self.config.iw_normalize:
1020           importance_weights *= (
1021               float(rl_batch.batch_size) / importance_weights.sum())
1022         combined_policy_multipliers *= importance_weights.reshape(-1, 1)
1023 
1024       # Train on replay batch, top-k MLE.
1025       assert self.program_count is not None
1026       fetches = {
1027           'global_step': global_step_op,
1028           'program_count': self.program_count,
1029           'summaries': self.rl_summary_op,
1030           'train_op': train_op,
1031           'gradients': self.gradients_dict if return_gradients else self.no_op}
1032       fetched = session.run(
1033           fetches,
1034           {self.actions: combined_actions,
1035            self.empirical_values: [[]],  # replay_emp_values,
1036            self.policy_multipliers: combined_policy_multipliers,
1037            self.adjusted_lengths: combined_adjusted_lengths,
1038            self.off_policy_targets: off_policy_targets,
1039            self.off_policy_target_lengths: off_policy_target_lengths,
1040            self.offp_switch: offp_switch})
1041 
1042       # Add to experience replay buffer.
1043       self.experience_replay.add_many(
1044           objs=new_experiences,
1045           weights=[exp(r / self.replay_temperature) for r in batch_tot_r],
1046           keys=code_strings)
1047 
1048     # Update program count.
1049     session.run(
1050         [self.program_count_add_op],
1051         {self.program_count_add_ph: num_programs_from_policy})
1052 
1053     # Update EMA baselines on the mini-batch which we just did traning on.
1054     if not self.a2c:
1055       for i in xrange(rl_batch.batch_size):
1056         episode_length = combined_adjusted_lengths[i]
1057         empirical_returns = combined_returns[i, :episode_length]
1058         for j in xrange(episode_length):
1059           # Update ema_baselines in place.
1060           self.ema_by_len[j] = (
1061               self.ema_baseline_decay * self.ema_by_len[j]
1062               + (1 - self.ema_baseline_decay) * empirical_returns[j])
1063 
1064     global_step = fetched['global_step']
1065     global_npe = fetched['program_count']
1066     core_summaries = fetched['summaries']
1067     summaries_list = [core_summaries]
1068 
1069     if num_programs_from_policy:
1070       s_i = 0
1071       text_summary = self._rl_text_summary(
1072           session,
1073           global_step,
1074           global_npe,
1075           batch_tot_r[s_i],
1076           episode_lengths[s_i], test_cases[s_i],
1077           code_outputs[s_i], code_strings[s_i], reasons[s_i])
1078       reward_summary = self._rl_reward_summary(batch_tot_r)
1079 
1080       is_best = False
1081       if self.global_best_reward_fn:
1082         # Save best reward.
1083         best_reward = np.max(batch_tot_r)
1084         is_best = self.global_best_reward_fn(session, best_reward)
1085 
1086       if self.found_solution_op is not None and 'correct' in reasons:
1087         session.run(self.found_solution_op)
1088 
1089         # Save program to disk for record keeping.
1090         if self.stop_on_success:
1091           solutions = [
1092               {'code': code_strings[i], 'reward': batch_tot_r[i],
1093                'npe': global_npe}
1094               for i in xrange(len(reasons)) if reasons[i] == 'correct']
1095         elif is_best:
1096           solutions = [
1097               {'code': code_strings[np.argmax(batch_tot_r)],
1098                'reward': np.max(batch_tot_r),
1099                'npe': global_npe}]
1100         else:
1101           solutions = []
1102         if solutions:
1103           if self.assign_code_solution_fn:
1104             self.assign_code_solution_fn(session, solutions[0]['code'])
1105           with tf.gfile.FastGFile(self.logging_file, 'a') as writer:
1106             for solution_dict in solutions:
1107               writer.write(str(solution_dict) + '\n')
1108 
1109       max_i = np.argmax(batch_tot_r)
1110       max_tot_r = batch_tot_r[max_i]
1111       if max_tot_r >= self.top_reward:
1112         if max_tot_r >= self.top_reward:
1113           self.top_reward = max_tot_r
1114         logging.info('Top code: r=%.2f, \t%s', max_tot_r, code_strings[max_i])
1115       if self.top_episodes is not None:
1116         self.top_episodes.push(
1117             max_tot_r, tuple(batch_actions[max_i, :episode_lengths[max_i]]))
1118 
1119       summaries_list += [text_summary, reward_summary]
1120 
1121       if self.do_iw_summaries and not empty_replay_buffer:
1122         # prob of replay samples under replay buffer sampling.
1123         norm_replay_weights = [
1124             w / self.experience_replay.total_weight
1125             for w in replay_weights]
1126         replay_iw = self._compute_iw(replay_log_probs, replay_weights)
1127         on_policy_iw = self._compute_iw(on_policy_log_probs, on_policy_weights)
1128         summaries_list.append(
1129             self._iw_summary(
1130                 session, replay_iw, replay_log_probs, norm_replay_weights,
1131                 on_policy_iw, on_policy_log_probs))
1132 
1133     return UpdateStepResult(
1134         global_step=global_step,
1135         global_npe=global_npe,
1136         summaries_list=summaries_list,
1137         gradients_dict=fetched['gradients'])

1140 def io_to_text(io_case, io_type):
1141   if isinstance(io_case, misc.IOTuple):
1142     # If there are many strings, join them with ','.
1143     return ','.join([io_to_text(e, io_type) for e in io_case])
1144   if io_type == misc.IOType.string:
1145     # There is one string. Return it.
1146     return misc.tokens_to_text(io_case)
1147   if (io_type == misc.IOType.integer
1148       or io_type == misc.IOType.boolean):
1149     if len(io_case) == 1:
1150       return str(io_case[0])
1151     return str(io_case)

1160 def compute_rewards(rl_batch, batch_actions, episode_lengths, batch_size=None):
1161   """Compute rewards for each episode in the batch.
1162 
1163   Args:
1164     rl_batch: A data.RLBatch instance. This holds information about the task
1165         each episode is solving, and a reward function for each episode.
1166     batch_actions: Contains batch of episodes. Each sequence of actions will be
1167         converted into a BF program and then scored. A numpy array of shape
1168         [batch_size, max_sequence_length].
1169     episode_lengths: The sequence length of each episode in the batch. Iterable
1170         of length batch_size.
1171     batch_size: (optional) number of programs to score. Use this to limit the
1172         number of programs executed from this batch. For example, when doing
1173         importance sampling some of the on-policy episodes will be discarded
1174         and they should not be executed. `batch_size` can be less than or equal
1175         to the size of the input batch.
1176 
1177   Returns:
1178     CodeScoreInfo namedtuple instance. This holds not just the computed rewards,
1179     but additional information computed during code execution which can be used
1180     for debugging and monitoring. this includes: BF code strings, test cases
1181     the code was executed on, code outputs from those test cases, and reasons
1182     for success or failure.
1183   """
1184   code_strings = [
1185       ''.join([misc.bf_int2char(a) for a in action_sequence[:l]])
1186       for action_sequence, l in zip(batch_actions, episode_lengths)]
1187   if batch_size is None:
1188     batch_size = len(code_strings)
1189   else:
1190     assert batch_size <= len(code_strings)
1191     code_strings = code_strings[:batch_size]
1192 
1193   if isinstance(rl_batch.reward_fns, (list, tuple)):
1194     # reward_fns is a list of functions, same length as code_strings.
1195     assert len(rl_batch.reward_fns) >= batch_size
1196     r_fn_results = [
1197         rl_batch.reward_fns[i](code_strings[i]) for i in xrange(batch_size)]
1198   else:
1199     # reward_fns is allowed to be one function which processes a batch of code
1200     # strings. This is useful for efficiency and batch level computation.
1201     r_fn_results = rl_batch.reward_fns(code_strings)
1202 
1203   # Expecting that r_fn returns a list of rewards. Length of list equals
1204   # length of the code string (including EOS char).
1205 
1206   batch_rewards = [r.episode_rewards for r in r_fn_results]
1207   total_rewards = [sum(b) for b in batch_rewards]
1208   test_cases = [io_to_text(r.input_case, r.input_type) for r in r_fn_results]
1209   code_outputs = [io_to_text(r.code_output, r.output_type)
1210                   for r in r_fn_results]
1211   reasons = [r.reason for r in r_fn_results]
1212   return CodeScoreInfo(
1213       code_strings=code_strings,
1214       batch_rewards=batch_rewards,
1215       total_rewards=total_rewards,
1216       test_cases=test_cases,
1217       code_outputs=code_outputs,
1218       reasons=reasons)

1221 def process_episodes(
1222     batch_rewards, episode_lengths, a2c=False, baselines=None,
1223     batch_values=None):
1224   """Compute REINFORCE targets.
1225 
1226   REINFORCE here takes the form:
1227   grad_t = grad[log(pi(a_t|c_t))*target_t]
1228   where c_t is context: i.e. RNN state or environment state (or both).
1229 
1230   Two types of targets are supported:
1231   1) Advantage actor critic (a2c).
1232   2) Vanilla REINFORCE with baseline.
1233 
1234   Args:
1235     batch_rewards: Rewards received in each episode in the batch. A numpy array
1236         of shape [batch_size, max_sequence_length]. Note, these are per-timestep
1237         rewards, not total reward.
1238     episode_lengths: Length of each episode. An iterable of length batch_size.
1239     a2c: A bool. Whether to compute a2c targets (True) or vanilla targets
1240         (False).
1241     baselines: If a2c is False, provide baselines for each timestep. This is a
1242         list (or indexable container) of length max_time. Note: baselines are
1243         shared across all episodes, which is why there is no batch dimension.
1244         It is up to the caller to update baselines accordingly.
1245     batch_values: If a2c is True, provide values computed by a value estimator.
1246         A numpy array of shape [batch_size, max_sequence_length].
1247 
1248   Returns:
1249     batch_targets: REINFORCE targets for each episode and timestep. A numpy
1250         array of shape [batch_size, max_sequence_length].
1251     batch_returns: Returns computed for each episode and timestep. This is for
1252         reference, and is not used in the REINFORCE gradient update (but was
1253         used to compute the targets). A numpy array of shape
1254         [batch_size, max_sequence_length].
1255   """
1256   num_programs = len(batch_rewards)
1257   assert num_programs <= len(episode_lengths)
1258   batch_returns = [None] * num_programs
1259   batch_targets = [None] * num_programs
1260   for i in xrange(num_programs):
1261     episode_length = episode_lengths[i]
1262     assert len(batch_rewards[i]) == episode_length
1263     # Compute target for each timestep.
1264     # If we are computing A2C:
1265     #    target_t = advantage_t = R_t - V(c_t)
1266     #    where V(c_t) is a learned value function (provided as `values`).
1267     # Otherwise:
1268     #    target_t = R_t - baselines[t]
1269     #    where `baselines` are provided.
1270     # In practice we use a more generalized formulation of advantage. See docs
1271     # for `discounted_advantage_and_rewards`.
1272     if a2c:
1273       # Compute advantage.
1274       assert batch_values is not None
1275       episode_values = batch_values[i, :episode_length]
1276       episode_rewards = batch_rewards[i]
1277       emp_val, gen_adv = rollout_lib.discounted_advantage_and_rewards(
1278           episode_rewards, episode_values, gamma=1.0, lambda_=1.0)
1279       batch_returns[i] = emp_val
1280       batch_targets[i] = gen_adv
1281     else:
1282       # Compute return for each timestep. See section 3 of
1283       # https://arxiv.org/pdf/1602.01783.pdf
1284       assert baselines is not None
1285       empirical_returns = rollout_lib.discount(batch_rewards[i], gamma=1.0)
1286       targets = [None] * episode_length
1287       for j in xrange(episode_length):
1288         targets[j] = empirical_returns[j] - baselines[j]
1289       batch_returns[i] = empirical_returns
1290       batch_targets[i] = targets
1291   batch_returns = utils.stack_pad(batch_returns, 0)
1292   if num_programs:
1293     batch_targets = utils.stack_pad(batch_targets, 0)
1294   else:
1295     batch_targets = np.array([], dtype=np.float32)
1296 
1297   return (batch_targets, batch_returns)
