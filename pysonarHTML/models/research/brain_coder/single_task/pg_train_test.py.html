  23   def RunTrainingSteps(self, config_string, num_steps=10):
  24     """Run a few training steps with the given config.
  25 
  26     Just check that nothing crashes.
  27 
  28     Args:
  29       config_string: Config encoded in a string. See
  30           $REPO_PATH/common/config_lib.py
  31       num_steps: Number of training steps to run. Defaults to 10.
  32     """
  33     config = defaults.default_config_with_updates(config_string)
  34     FLAGS.master = ''
  35     FLAGS.max_npe = num_steps * config.batch_size
  36     FLAGS.summary_interval = 1
  37     FLAGS.logdir = tf.test.get_temp_dir()
  38     FLAGS.config = config_string
  39     tf.reset_default_graph()
  40     run.main(None)
