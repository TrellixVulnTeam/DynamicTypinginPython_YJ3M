  52 class CheckpointWriter(object):
  53   """Manages loading and saving GA populations to disk.
  54 
  55   This object is used by the genetic algorithm to save progress periodically
  56   so that a recent population can be loaded from disk in the event of a restart.
  57   """
  58 
  59   def __init__(self, checkpoint_dir, population_size):
  60     self.checkpoint_file = os.path.join(checkpoint_dir, 'checkpoint.pickle')
  61     self.population_size = population_size
  62 
  63   def write(self, gen, population, halloffame):
  64     """Write GA state to disk.
  65 
  66     Overwrites previous saved state.
  67 
  68     Args:
  69       gen: Generation number.
  70       population: List of Individual objects.
  71       halloffame: Hall-of-fame buffer. Typically a priority queue.
  72     """
  73     raw = cPickle.dumps((gen, population, halloffame))
  74     with tf.gfile.FastGFile(self.checkpoint_file, 'w') as f:
  75       f.write(raw)
  76 
  77   def load(self):
  78     """Loads GA state from disk.
  79 
  80     Loads whatever is on disk, which will be whatever the most recent call
  81     to `write` wrote.
  82 
  83     Returns:
  84       gen: Generation number.
  85       population: List of Individual objects.
  86       halloffame: Hall-of-fame buffer. Typically a priority queue.
  87     """
  88     with tf.gfile.FastGFile(self.checkpoint_file, 'r') as f:
  89       raw = f.read()
  90     objs = cPickle.loads(raw)
  91     # Validate data.
  92     assert isinstance(objs, tuple) and len(objs) == 3, (
  93         'Expecting a 3-tuple, but got %s instead.' % (objs,))
  94     gen, population, halloffame = objs
  95     assert isinstance(gen, int), (
  96         'Expecting `gen` to be an integer, got %s' % (gen,))
  97     assert (
  98         isinstance(population, list)
  99         and len(population) == self.population_size
 100     ), (
 101         'Expecting `population` to be a list with size %d, got %s'
 102         % (self.population_size, population))
 103     assert halloffame is None or len(halloffame) == 2, (
 104         'Expecting hall-of-fame object to have length two, got length %d'
 105         % len(halloffame))
 106     logging.info('Loaded pop from checkpoint file: "%s".',
 107                  self.checkpoint_file)
 108     return gen, population, halloffame
 109 
 110   def has_checkpoint(self):
 111     """Checks if a checkpoint exists on disk, and if so returns True."""
 112     return tf.gfile.Exists(self.checkpoint_file)

 115 def run_training(config=None, tuner=None, logdir=None, trial_name=None,  # pylint: disable=unused-argument
 116                  is_chief=True):
 117   """Do all training runs.
 118 
 119   This is the top level training function for policy gradient based models.
 120   Run this from the main function.
 121 
 122   Args:
 123     config: config_lib.Config instance containing global config (agent and
 124         environment hparams). If None, config will be parsed from FLAGS.config.
 125     tuner: (unused) A tuner instance. Leave as None if not tuning.
 126     logdir: Parent directory where all data from all runs will be written. If
 127         None, FLAGS.logdir will be used.
 128     trial_name: (unused) If tuning, set this to a unique string that identifies
 129         this trial. If `tuner` is not None, this also must be set.
 130     is_chief: True if this worker is the chief.
 131 
 132   Returns:
 133     List of results dicts which were written to disk. Each training run gets a
 134     results dict. Results dict contains metrics, i.e. (name, value) pairs which
 135     give information about the training run.
 136 
 137   Raises:
 138     ValueError: If FLAGS.num_workers does not divide FLAGS.num_repetitions.
 139     ValueError: If results dicts read from disk contain invalid data.
 140   """
 141   if not config:
 142     # If custom config is not given, get it from flags.
 143     config = defaults.default_config_with_updates(FLAGS.config)
 144   if not logdir:
 145     logdir = FLAGS.logdir
 146 
 147   if FLAGS.num_repetitions % FLAGS.num_workers != 0:
 148     raise ValueError('Number of workers must divide number of repetitions')
 149   num_local_reps = FLAGS.num_repetitions // FLAGS.num_workers
 150   logging.info('Running %d reps globally.', FLAGS.num_repetitions)
 151   logging.info('This worker will run %d local reps.', num_local_reps)
 152   if FLAGS.max_npe:
 153     max_generations = FLAGS.max_npe // config.batch_size
 154     logging.info('Max samples per rep: %d', FLAGS.max_npe)
 155     logging.info('Max generations per rep: %d', max_generations)
 156   else:
 157     max_generations = sys.maxint
 158     logging.info('Running unlimited generations.')
 159 
 160   assert FLAGS.num_workers > 0
 161   logging.info('Starting experiment. Directory: "%s"', logdir)
 162   results = results_lib.Results(logdir, FLAGS.task_id)
 163   local_results_list = results.read_this_shard()
 164   if local_results_list:
 165     if local_results_list[0]['max_npe'] != FLAGS.max_npe:
 166       raise ValueError(
 167           'Cannot resume training. Max-NPE changed. Was %s, now %s',
 168           local_results_list[0]['max_npe'], FLAGS.max_npe)
 169     if local_results_list[0]['max_global_repetitions'] != FLAGS.num_repetitions:
 170       raise ValueError(
 171           'Cannot resume training. Number of repetitions changed. Was %s, '
 172           'now %s',
 173           local_results_list[0]['max_global_repetitions'],
 174           FLAGS.num_repetitions)
 175   start_rep = len(local_results_list)
 176 
 177   for rep in xrange(start_rep, num_local_reps):
 178     global_rep = num_local_reps * FLAGS.task_id + rep
 179     logging.info(
 180         'Starting repetition: Rep = %d. (global rep = %d)',
 181         rep, global_rep)
 182 
 183     # Save data for each rep, like checkpoints, goes into separate folders.
 184     run_dir = os.path.join(logdir, 'run_%d' % global_rep)
 185 
 186     if not tf.gfile.IsDirectory(run_dir):
 187       tf.gfile.MakeDirs(run_dir)
 188     checkpoint_writer = CheckpointWriter(run_dir,
 189                                          population_size=config.batch_size)
 190 
 191     data_manager = data.DataManager(config, run_number=global_rep)
 192     task_eval_fn = ga_lib.make_task_eval_fn(data_manager.rl_task)
 193 
 194     if config.agent.algorithm == 'rand':
 195       logging.info('Running random search.')
 196       assert FLAGS.max_npe
 197       result = run_random_search(
 198           FLAGS.max_npe, run_dir, task_eval_fn, config.timestep_limit)
 199     else:
 200       assert config.agent.algorithm == 'ga'
 201       logging.info('Running genetic algorithm.')
 202       pop = ga_lib.make_population(
 203           ga_lib.random_individual(config.timestep_limit),
 204           n=config.batch_size)
 205       hof = utils.MaxUniquePriorityQueue(2)  # Hall of fame.
 206       result = ga_lib.ga_loop(
 207           pop,
 208           cxpb=config.agent.crossover_rate, mutpb=config.agent.mutation_rate,
 209           task_eval_fn=task_eval_fn,
 210           ngen=max_generations, halloffame=hof,
 211           checkpoint_writer=checkpoint_writer)
 212 
 213     logging.info('Finished rep. Num gens: %d', result.generations)
 214 
 215     results_dict = {
 216         'max_npe': FLAGS.max_npe,
 217         'batch_size': config.batch_size,
 218         'max_batches': FLAGS.max_npe // config.batch_size,
 219         'npe': result.num_programs,
 220         'max_global_repetitions': FLAGS.num_repetitions,
 221         'max_local_repetitions': num_local_reps,
 222         'code_solution': result.best_code if result.solution_found else '',
 223         'best_reward': result.reward,
 224         'num_batches': result.generations,
 225         'found_solution': result.solution_found,
 226         'task': data_manager.task_name,
 227         'global_rep': global_rep}
 228     logging.info('results_dict: %s', results_dict)
 229     results.append(results_dict)
 230 
 231   if is_chief:
 232     logging.info(
 233         'Worker is chief. Waiting for all workers to finish so that results '
 234         'can be reported to the tuner.')
 235 
 236     global_results_list, shard_stats = results.read_all(
 237         num_shards=FLAGS.num_workers)
 238     while not all(s.finished for s in shard_stats):
 239       logging.info(
 240           'Still waiting on these workers: %s',
 241           ', '.join(
 242               ['%d (%d reps left)'
 243                % (i, s.max_local_reps - s.num_local_reps_completed)
 244                for i, s in enumerate(shard_stats)
 245                if not s.finished]))
 246       sleep(60)
 247       global_results_list, shard_stats = results.read_all(
 248           num_shards=FLAGS.num_workers)
 249 
 250     logging.info(
 251         '%d results obtained. Chief worker is exiting the experiment.',
 252         len(global_results_list))
 253 
 254     return global_results_list

 257 def run_random_search(max_num_programs, checkpoint_dir, task_eval_fn,
 258                       timestep_limit):
 259   """Run uniform random search routine.
 260 
 261   Randomly samples programs from a uniform distribution until either a valid
 262   program is found, or the maximum NPE is reached. Results are written to disk
 263   and returned.
 264 
 265   Args:
 266     max_num_programs: Maximum NPE (number of programs executed). If no solution
 267         is found after this many programs are tried, the run is stopped and
 268         considered a failure.
 269     checkpoint_dir: Where to save state during the run.
 270     task_eval_fn: Function that maps code string to result containing total
 271         reward and info about success.
 272     timestep_limit: Maximum length of code strings.
 273 
 274   Returns:
 275     ga_lib.GaResult namedtuple instance. This contains the best code and highest
 276     reward found.
 277   """
 278   checkpoint_file = os.path.join(checkpoint_dir, 'random_search.txt')
 279   num_programs_seen = 0
 280   found_solution = False
 281   best_code = ''
 282   best_reward = 0.0
 283   if tf.gfile.Exists(checkpoint_file):
 284     try:
 285       with tf.gfile.FastGFile(checkpoint_file, 'r') as f:
 286         lines = list(f)
 287         num_programs_seen = int(lines[0])
 288         found_solution = bool(int(lines[1]))
 289         if found_solution:
 290           best_code = lines[2]
 291           best_reward = float(lines[3])
 292     except:  # pylint: disable=bare-except
 293       pass
 294 
 295   while not found_solution and num_programs_seen < max_num_programs:
 296     if num_programs_seen % 1000 == 0:
 297       logging.info('num_programs_seen = %d', num_programs_seen)
 298       with tf.gfile.FastGFile(checkpoint_file, 'w') as f:
 299         f.write(str(num_programs_seen) + '\n')
 300         f.write(str(int(found_solution)) + '\n')
 301 
 302     code = np.random.choice(ga_lib.GENES, timestep_limit).tolist()
 303     res = task_eval_fn(code)
 304     found_solution = res.correct
 305     num_programs_seen += 1
 306 
 307     if found_solution:
 308       best_code = ''.join(code)
 309       best_reward = res.reward
 310 
 311   logging.info('num_programs_seen = %d', num_programs_seen)
 312   logging.info('found solution: %s', found_solution)
 313   with tf.gfile.FastGFile(checkpoint_file, 'w') as f:
 314     f.write(str(num_programs_seen) + '\n')
 315     f.write(str(int(found_solution)) + '\n')
 316     if found_solution:
 317       f.write(best_code + '\n')
 318       f.write(str(best_reward) + '\n')
 319 
 320   return ga_lib.GaResult(
 321       population=[], best_code=best_code, reward=best_reward,
 322       solution_found=found_solution, generations=num_programs_seen,
 323       num_programs=num_programs_seen, max_generations=max_num_programs,
 324       max_num_programs=max_num_programs)
