  29 class Results(object):
  30   """Manages reading and writing training results to disk asynchronously.
  31 
  32   Each worker writes to its own file, so that there are no race conditions when
  33   writing happens. However any worker may read any file, as is the case for
  34   `read_all`. Writes are expected to be atomic so that workers will never
  35   read incomplete data, and this is likely to be the case on Unix systems.
  36   Reading out of date data is fine, as workers calling `read_all` will wait
  37   until data from every worker has been written before proceeding.
  38   """
  39   file_template = 'experiment_results_{0}.txt'
  40   search_regex = r'^experiment_results_([0-9])+\.txt$'
  41 
  42   def __init__(self, log_dir, shard_id=0):
  43     """Construct `Results` instance.
  44 
  45     Args:
  46       log_dir: Where to write results files.
  47       shard_id: Unique id for this file (i.e. shard). Each worker that will
  48           be writing results should use a different shard id. If there are
  49           N shards, each shard should be numbered 0 through N-1.
  50     """
  51     # Use different files for workers so that they can write to disk async.
  52     assert 0 <= shard_id
  53     self.file_name = self.file_template.format(shard_id)
  54     self.log_dir = log_dir
  55     self.results_file = os.path.join(self.log_dir, self.file_name)
  56 
  57   def append(self, metrics):
  58     """Append results to results list on disk."""
  59     with tf.gfile.FastGFile(self.results_file, 'a') as writer:
  60       writer.write(str(metrics) + '\n')
  61 
  62   def read_this_shard(self):
  63     """Read only from this shard."""
  64     return self._read_shard(self.results_file)
  65 
  66   def _read_shard(self, results_file):
  67     """Read only from the given shard file."""
  68     try:
  69       with tf.gfile.FastGFile(results_file, 'r') as reader:
  70         results = [ast.literal_eval(entry) for entry in reader]
  71     except tf.errors.NotFoundError:
  72       # No results written to disk yet. Return empty list.
  73       return []
  74     return results
  75 
  76   def _get_max_local_reps(self, shard_results):
  77     """Get maximum number of repetitions the given shard needs to complete.
  78 
  79     Worker working on each shard needs to complete a certain number of runs
  80     before it finishes. This method will return that number so that we can
  81     determine which shards are still not done.
  82 
  83     We assume that workers are including a 'max_local_repetitions' value in
  84     their results, which should be the total number of repetitions it needs to
  85     run.
  86 
  87     Args:
  88       shard_results: Dict mapping metric names to values. This should be read
  89           from a shard on disk.
  90 
  91     Returns:
  92       Maximum number of repetitions the given shard needs to complete.
  93     """
  94     mlrs = [r['max_local_repetitions'] for r in shard_results]
  95     if not mlrs:
  96       return 0
  97     for n in mlrs[1:]:
  98       assert n == mlrs[0], 'Some reps have different max rep.'
  99     return mlrs[0]

 101   def read_all(self, num_shards=None):
 102     """Read results across all shards, i.e. get global results list.
 103 
 104     Args:
 105       num_shards: (optional) specifies total number of shards. If the caller
 106           wants information about which shards are incomplete, provide this
 107           argument (so that shards which have yet to be created are still
 108           counted as incomplete shards). Otherwise, no information about
 109           incomplete shards will be returned.
 110 
 111     Returns:
 112       aggregate: Global list of results (across all shards).
 113       shard_stats: List of ShardStats instances, one for each shard. Or None if
 114           `num_shards` is None.
 115     """
 116     try:
 117       all_children = tf.gfile.ListDirectory(self.log_dir)
 118     except tf.errors.NotFoundError:
 119       if num_shards is None:
 120         return [], None
 121       return [], [[] for _ in xrange(num_shards)]
 122     shard_ids = {
 123         get_shard_id(fname): fname
 124         for fname in all_children if re.search(self.search_regex, fname)}
 125 
 126     if num_shards is None:
 127       aggregate = []
 128       shard_stats = None
 129       for results_file in shard_ids.values():
 130         aggregate.extend(self._read_shard(
 131             os.path.join(self.log_dir, results_file)))
 132     else:
 133       results_per_shard = [None] * num_shards
 134       for shard_id in xrange(num_shards):
 135         if shard_id in shard_ids:
 136           results_file = shard_ids[shard_id]
 137           results_per_shard[shard_id] = self._read_shard(
 138               os.path.join(self.log_dir, results_file))
 139         else:
 140           results_per_shard[shard_id] = []
 141 
 142       # Compute shard stats.
 143       shard_stats = []
 144       for shard_results in results_per_shard:
 145         max_local_reps = self._get_max_local_reps(shard_results)
 146         shard_stats.append(ShardStats(
 147             num_local_reps_completed=len(shard_results),
 148             max_local_reps=max_local_reps,
 149             finished=ge_non_zero(len(shard_results), max_local_reps)))
 150 
 151       # Compute aggregate.
 152       aggregate = [
 153           r for shard_results in results_per_shard for r in shard_results]
 154 
 155     return aggregate, shard_stats
