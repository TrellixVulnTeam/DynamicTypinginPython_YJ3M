  91 def convert_to_int_2d_and_pad(a, utility):
  92   ans = []
  93   #print a
  94   for b in a:
  95     temp = []
  96     if (len(b) > utility.FLAGS.max_entry_length):
  97       b = b[0:utility.FLAGS.max_entry_length]
  98     for remaining in range(len(b), utility.FLAGS.max_entry_length):
  99       b.append(utility.dummy_token)
 100     assert len(b) == utility.FLAGS.max_entry_length
 101     for word in b:
 102       temp.append(utility.word_ids[word_lookup(word, utility)])
 103     ans.append(temp)
 104   #print ans
 105   return ans

 108 def convert_to_bool_and_pad(a, utility):
 109   a = a.tolist()
 110   for i in range(len(a)):
 111     for j in range(len(a[i])):
 112       if (a[i][j] < 1):
 113         a[i][j] = False
 114       else:
 115         a[i][j] = True
 116     a[i] = a[i] + [False] * (utility.FLAGS.max_elements - len(a[i]))
 117   return a

 123 def partial_match(question, table, number):
 124   answer = []
 125   match = {}
 126   for i in range(len(table)):
 127     temp = []
 128     for j in range(len(table[i])):
 129       temp.append(0)
 130     answer.append(temp)
 131   for i in range(len(table)):
 132     for j in range(len(table[i])):
 133       for word in question:
 134         if (number):
 135           if (word == table[i][j]):
 136             answer[i][j] = 1.0
 137             match[i] = 1.0
 138         else:
 139           if (word in table[i][j]):
 140             answer[i][j] = 1.0
 141             match[i] = 1.0
 142   return answer, match

 145 def exact_match(question, table, number):
 146   #performs exact match operation
 147   answer = []
 148   match = {}
 149   matched_indices = []
 150   for i in range(len(table)):
 151     temp = []
 152     for j in range(len(table[i])):
 153       temp.append(0)
 154     answer.append(temp)
 155   for i in range(len(table)):
 156     for j in range(len(table[i])):
 157       if (number):
 158         for word in question:
 159           if (word == table[i][j]):
 160             match[i] = 1.0
 161             answer[i][j] = 1.0
 162       else:
 163         table_entry = table[i][j]
 164         for k in range(len(question)):
 165           if (k + len(table_entry) <= len(question)):
 166             if (table_entry == question[k:(k + len(table_entry))]):
 167               #if(len(table_entry) == 1):
 168               #print "match: ", table_entry, question
 169               match[i] = 1.0
 170               answer[i][j] = 1.0
 171               matched_indices.append((k, len(table_entry)))
 172   return answer, match, matched_indices

 175 def partial_column_match(question, table, number):
 176   answer = []
 177   for i in range(len(table)):
 178     answer.append(0)
 179   for i in range(len(table)):
 180     for word in question:
 181       if (word in table[i]):
 182         answer[i] = 1.0
 183   return answer

 186 def exact_column_match(question, table, number):
 187   #performs exact match on column names
 188   answer = []
 189   matched_indices = []
 190   for i in range(len(table)):
 191     answer.append(0)
 192   for i in range(len(table)):
 193     table_entry = table[i]
 194     for k in range(len(question)):
 195       if (k + len(table_entry) <= len(question)):
 196         if (table_entry == question[k:(k + len(table_entry))]):
 197           answer[i] = 1.0
 198           matched_indices.append((k, len(table_entry)))
 199   return answer, matched_indices

 202 def get_max_entry(a):
 203   e = {}
 204   for w in a:
 205     if (w != "UNK, "):
 206       if (e.has_key(w)):
 207         e[w] += 1
 208       else:
 209         e[w] = 1
 210   if (len(e) > 0):
 211     (key, val) = sorted(e.items(), key=lambda x: -1 * x[1])[0]
 212     if (val > 1):
 213       return key
 214     else:
 215       return -1.0
 216   else:
 217     return -1.0

 227 def group_by_max(table, number):
 228   #computes the most frequently occurring entry in a column
 229   answer = []
 230   for i in range(len(table)):
 231     temp = []
 232     for j in range(len(table[i])):
 233       temp.append(0)
 234     answer.append(temp)
 235   for i in range(len(table)):
 236     if (number):
 237       curr = table[i]
 238     else:
 239       curr = [list_join(w) for w in table[i]]
 240     max_entry = get_max_entry(curr)
 241     #print i, max_entry
 242     for j in range(len(curr)):
 243       if (max_entry == curr[j]):
 244         answer[i][j] = 1.0
 245       else:
 246         answer[i][j] = 0.0
 247   return answer

 265 def complete_wiki_processing(data, utility, train=True):
 266   #convert to integers and padding
 267   processed_data = []
 268   num_bad_examples = 0
 269   for example in data:
 270     number_found = 0
 271     if (example.is_bad_example):
 272       num_bad_examples += 1
 273     if (not (example.is_bad_example)):
 274       example.string_question = example.question[:]
 275       #entry match
 276       example.processed_number_columns = example.processed_number_columns[:]
 277       example.processed_word_columns = example.processed_word_columns[:]
 278       example.word_exact_match, word_match, matched_indices = exact_match(
 279           example.string_question, example.original_wc, number=False)
 280       example.number_exact_match, number_match, _ = exact_match(
 281           example.string_question, example.original_nc, number=True)
 282       if (not (pick_one(example.word_exact_match)) and not (
 283           pick_one(example.number_exact_match))):
 284         assert len(word_match) == 0
 285         assert len(number_match) == 0
 286         example.word_exact_match, word_match = partial_match(
 287             example.string_question, example.original_wc, number=False)
 288       #group by max
 289       example.word_group_by_max = group_by_max(example.original_wc, False)
 290       example.number_group_by_max = group_by_max(example.original_nc, True)
 291       #column name match
 292       example.word_column_exact_match, wcol_matched_indices = exact_column_match(
 293           example.string_question, example.original_wc_names, number=False)
 294       example.number_column_exact_match, ncol_matched_indices = exact_column_match(
 295           example.string_question, example.original_nc_names, number=False)
 296       if (not (1.0 in example.word_column_exact_match) and not (
 297           1.0 in example.number_column_exact_match)):
 298         example.word_column_exact_match = partial_column_match(
 299             example.string_question, example.original_wc_names, number=False)
 300         example.number_column_exact_match = partial_column_match(
 301             example.string_question, example.original_nc_names, number=False)
 302       if (len(word_match) > 0 or len(number_match) > 0):
 303         example.question.append(utility.entry_match_token)
 304       if (1.0 in example.word_column_exact_match or
 305           1.0 in example.number_column_exact_match):
 306         example.question.append(utility.column_match_token)
 307       example.string_question = example.question[:]
 308       example.number_lookup_matrix = np.transpose(
 309           example.number_lookup_matrix)[:]
 310       example.word_lookup_matrix = np.transpose(example.word_lookup_matrix)[:]
 311       example.columns = example.number_columns[:]
 312       example.word_columns = example.word_columns[:]
 313       example.len_total_cols = len(example.word_column_names) + len(
 314           example.number_column_names)
 315       example.column_names = example.number_column_names[:]
 316       example.word_column_names = example.word_column_names[:]
 317       example.string_column_names = example.number_column_names[:]
 318       example.string_word_column_names = example.word_column_names[:]
 319       example.sorted_number_index = []
 320       example.sorted_word_index = []
 321       example.column_mask = []
 322       example.word_column_mask = []
 323       example.processed_column_mask = []
 324       example.processed_word_column_mask = []
 325       example.word_column_entry_mask = []
 326       example.question_attention_mask = []
 327       example.question_number = example.question_number_1 = -1
 328       example.question_attention_mask = []
 329       example.ordinal_question = []
 330       example.ordinal_question_one = []
 331       new_question = []
 332       if (len(example.number_columns) > 0):
 333         example.len_col = len(example.number_columns[0])
 334       else:
 335         example.len_col = len(example.word_columns[0])
 336       for (start, length) in matched_indices:
 337         for j in range(length):
 338           example.question[start + j] = utility.unk_token
 339       #print example.question
 340       for word in example.question:
 341         if (isinstance(word, numbers.Number) or wiki_data.is_date(word)):
 342           if (not (isinstance(word, numbers.Number)) and
 343               wiki_data.is_date(word)):
 344             word = word.replace("X", "").replace("-", "")
 345           number_found += 1
 346           if (number_found == 1):
 347             example.question_number = word
 348             if (len(example.ordinal_question) > 0):
 349               example.ordinal_question[len(example.ordinal_question) - 1] = 1.0
 350             else:
 351               example.ordinal_question.append(1.0)
 352           elif (number_found == 2):
 353             example.question_number_1 = word
 354             if (len(example.ordinal_question_one) > 0):
 355               example.ordinal_question_one[len(example.ordinal_question_one) -
 356                                            1] = 1.0
 357             else:
 358               example.ordinal_question_one.append(1.0)
 359         else:
 360           new_question.append(word)
 361           example.ordinal_question.append(0.0)
 362           example.ordinal_question_one.append(0.0)
 363       example.question = [
 364           utility.word_ids[word_lookup(w, utility)] for w in new_question
 365       ]
 366       example.question_attention_mask = [0.0] * len(example.question)
 367       #when the first question number occurs before a word
 368       example.ordinal_question = example.ordinal_question[0:len(
 369           example.question)]
 370       example.ordinal_question_one = example.ordinal_question_one[0:len(
 371           example.question)]
 372       #question-padding
 373       example.question = [utility.word_ids[utility.dummy_token]] * (
 374           utility.FLAGS.question_length - len(example.question)
 375       ) + example.question
 376       example.question_attention_mask = [-10000.0] * (
 377           utility.FLAGS.question_length - len(example.question_attention_mask)
 378       ) + example.question_attention_mask
 379       example.ordinal_question = [0.0] * (utility.FLAGS.question_length -
 380                                           len(example.ordinal_question)
 381                                          ) + example.ordinal_question
 382       example.ordinal_question_one = [0.0] * (utility.FLAGS.question_length -
 383                                               len(example.ordinal_question_one)
 384                                              ) + example.ordinal_question_one
 385       if (True):
 386         #number columns and related-padding
 387         num_cols = len(example.columns)
 388         start = 0
 389         for column in example.number_columns:
 390           if (check_processed_cols(example.processed_number_columns[start],
 391                                    utility)):
 392             example.processed_column_mask.append(0.0)
 393           sorted_index = sorted(
 394               range(len(example.processed_number_columns[start])),
 395               key=lambda k: example.processed_number_columns[start][k],
 396               reverse=True)
 397           sorted_index = sorted_index + [utility.FLAGS.pad_int] * (
 398               utility.FLAGS.max_elements - len(sorted_index))
 399           example.sorted_number_index.append(sorted_index)
 400           example.columns[start] = column + [utility.FLAGS.pad_int] * (
 401               utility.FLAGS.max_elements - len(column))
 402           example.processed_number_columns[start] += [utility.FLAGS.pad_int] * (
 403               utility.FLAGS.max_elements -
 404               len(example.processed_number_columns[start]))
 405           start += 1
 406           example.column_mask.append(0.0)
 407         for remaining in range(num_cols, utility.FLAGS.max_number_cols):
 408           example.sorted_number_index.append([utility.FLAGS.pad_int] *
 409                                              (utility.FLAGS.max_elements))
 410           example.columns.append([utility.FLAGS.pad_int] *
 411                                  (utility.FLAGS.max_elements))
 412           example.processed_number_columns.append([utility.FLAGS.pad_int] *
 413                                                   (utility.FLAGS.max_elements))
 414           example.number_exact_match.append([0.0] *
 415                                             (utility.FLAGS.max_elements))
 416           example.number_group_by_max.append([0.0] *
 417                                              (utility.FLAGS.max_elements))
 418           example.column_mask.append(-100000000.0)
 419           example.processed_column_mask.append(-100000000.0)
 420           example.number_column_exact_match.append(0.0)
 421           example.column_names.append([utility.dummy_token])
 422         #word column  and related-padding
 423         start = 0
 424         word_num_cols = len(example.word_columns)
 425         for column in example.word_columns:
 426           if (check_processed_cols(example.processed_word_columns[start],
 427                                    utility)):
 428             example.processed_word_column_mask.append(0.0)
 429           sorted_index = sorted(
 430               range(len(example.processed_word_columns[start])),
 431               key=lambda k: example.processed_word_columns[start][k],
 432               reverse=True)
 433           sorted_index = sorted_index + [utility.FLAGS.pad_int] * (
 434               utility.FLAGS.max_elements - len(sorted_index))
 435           example.sorted_word_index.append(sorted_index)
 436           column = convert_to_int_2d_and_pad(column, utility)
 437           example.word_columns[start] = column + [[
 438               utility.word_ids[utility.dummy_token]
 439           ] * utility.FLAGS.max_entry_length] * (utility.FLAGS.max_elements -
 440                                                  len(column))
 441           example.processed_word_columns[start] += [utility.FLAGS.pad_int] * (
 442               utility.FLAGS.max_elements -
 443               len(example.processed_word_columns[start]))
 444           example.word_column_entry_mask.append([0] * len(column) + [
 445               utility.word_ids[utility.dummy_token]
 446           ] * (utility.FLAGS.max_elements - len(column)))
 447           start += 1
 448           example.word_column_mask.append(0.0)
 449         for remaining in range(word_num_cols, utility.FLAGS.max_word_cols):
 450           example.sorted_word_index.append([utility.FLAGS.pad_int] *
 451                                            (utility.FLAGS.max_elements))
 452           example.word_columns.append([[utility.word_ids[utility.dummy_token]] *
 453                                        utility.FLAGS.max_entry_length] *
 454                                       (utility.FLAGS.max_elements))
 455           example.word_column_entry_mask.append(
 456               [utility.word_ids[utility.dummy_token]] *
 457               (utility.FLAGS.max_elements))
 458           example.word_exact_match.append([0.0] * (utility.FLAGS.max_elements))
 459           example.word_group_by_max.append([0.0] * (utility.FLAGS.max_elements))
 460           example.processed_word_columns.append([utility.FLAGS.pad_int] *
 461                                                 (utility.FLAGS.max_elements))
 462           example.word_column_mask.append(-100000000.0)
 463           example.processed_word_column_mask.append(-100000000.0)
 464           example.word_column_exact_match.append(0.0)
 465           example.word_column_names.append([utility.dummy_token] *
 466                                            utility.FLAGS.max_entry_length)
 467         seen_tables[example.table_key] = 1
 468       #convert column and word column names to integers
 469       example.column_ids = convert_to_int_2d_and_pad(example.column_names,
 470                                                      utility)
 471       example.word_column_ids = convert_to_int_2d_and_pad(
 472           example.word_column_names, utility)
 473       for i_em in range(len(example.number_exact_match)):
 474         example.number_exact_match[i_em] = example.number_exact_match[
 475             i_em] + [0.0] * (utility.FLAGS.max_elements -
 476                              len(example.number_exact_match[i_em]))
 477         example.number_group_by_max[i_em] = example.number_group_by_max[
 478             i_em] + [0.0] * (utility.FLAGS.max_elements -
 479                              len(example.number_group_by_max[i_em]))
 480       for i_em in range(len(example.word_exact_match)):
 481         example.word_exact_match[i_em] = example.word_exact_match[
 482             i_em] + [0.0] * (utility.FLAGS.max_elements -
 483                              len(example.word_exact_match[i_em]))
 484         example.word_group_by_max[i_em] = example.word_group_by_max[
 485             i_em] + [0.0] * (utility.FLAGS.max_elements -
 486                              len(example.word_group_by_max[i_em]))
 487       example.exact_match = example.number_exact_match + example.word_exact_match
 488       example.group_by_max = example.number_group_by_max + example.word_group_by_max
 489       example.exact_column_match = example.number_column_exact_match + example.word_column_exact_match
 490       #answer and related mask, padding
 491       if (example.is_lookup):
 492         example.answer = example.calc_answer
 493         example.number_print_answer = example.number_lookup_matrix.tolist()
 494         example.word_print_answer = example.word_lookup_matrix.tolist()
 495         for i_answer in range(len(example.number_print_answer)):
 496           example.number_print_answer[i_answer] = example.number_print_answer[
 497               i_answer] + [0.0] * (utility.FLAGS.max_elements -
 498                                    len(example.number_print_answer[i_answer]))
 499         for i_answer in range(len(example.word_print_answer)):
 500           example.word_print_answer[i_answer] = example.word_print_answer[
 501               i_answer] + [0.0] * (utility.FLAGS.max_elements -
 502                                    len(example.word_print_answer[i_answer]))
 503         example.number_lookup_matrix = convert_to_bool_and_pad(
 504             example.number_lookup_matrix, utility)
 505         example.word_lookup_matrix = convert_to_bool_and_pad(
 506             example.word_lookup_matrix, utility)
 507         for remaining in range(num_cols, utility.FLAGS.max_number_cols):
 508           example.number_lookup_matrix.append([False] *
 509                                               utility.FLAGS.max_elements)
 510           example.number_print_answer.append([0.0] * utility.FLAGS.max_elements)
 511         for remaining in range(word_num_cols, utility.FLAGS.max_word_cols):
 512           example.word_lookup_matrix.append([False] *
 513                                             utility.FLAGS.max_elements)
 514           example.word_print_answer.append([0.0] * utility.FLAGS.max_elements)
 515         example.print_answer = example.number_print_answer + example.word_print_answer
 516       else:
 517         example.answer = example.calc_answer
 518         example.print_answer = [[0.0] * (utility.FLAGS.max_elements)] * (
 519             utility.FLAGS.max_number_cols + utility.FLAGS.max_word_cols)
 520       #question_number masks
 521       if (example.question_number == -1):
 522         example.question_number_mask = np.zeros([utility.FLAGS.max_elements])
 523       else:
 524         example.question_number_mask = np.ones([utility.FLAGS.max_elements])
 525       if (example.question_number_1 == -1):
 526         example.question_number_one_mask = -10000.0
 527       else:
 528         example.question_number_one_mask = np.float64(0.0)
 529       if (example.len_col > utility.FLAGS.max_elements):
 530         continue
 531       processed_data.append(example)
 532   return processed_data

 535 def add_special_words(utility):
 536   utility.words.append(utility.entry_match_token)
 537   utility.word_ids[utility.entry_match_token] = len(utility.word_ids)
 538   utility.reverse_word_ids[utility.word_ids[
 539       utility.entry_match_token]] = utility.entry_match_token

 561 def perform_word_cutoff(utility):
 562   if (utility.FLAGS.word_cutoff > 0):
 563     for word in utility.word_ids.keys():
 564       if (utility.word_count.has_key(word) and utility.word_count[word] <
 565           utility.FLAGS.word_cutoff and word != utility.unk_token and
 566           word != utility.dummy_token and word != utility.entry_match_token and
 567           word != utility.column_match_token):
 568         utility.word_ids.pop(word)
 569         utility.words.remove(word)

 572 def word_dropout(question, utility):
 573   if (utility.FLAGS.word_dropout_prob > 0.0):
 574     new_question = []
 575     for i in range(len(question)):
 576       if (question[i] != utility.dummy_token_id and
 577           utility.random.random() > utility.FLAGS.word_dropout_prob):
 578         new_question.append(utility.word_ids[utility.unk_token])
 579       else:
 580         new_question.append(question[i])
 581     return new_question
 582   else:
 583     return question
