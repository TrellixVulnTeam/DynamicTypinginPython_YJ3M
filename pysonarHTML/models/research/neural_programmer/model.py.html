  97   def LSTM_question_embedding(self, sentence, sentence_length):
  98     #LSTM processes the input question
  99     lstm_params = "question_lstm"
 100     hidden_vectors = []
 101     sentence = self.batch_question
 102     question_hidden = tf.zeros(
 103         [self.batch_size, self.utility.FLAGS.embedding_dims], self.data_type)
 104     question_c_hidden = tf.zeros(
 105         [self.batch_size, self.utility.FLAGS.embedding_dims], self.data_type)
 106     if (self.utility.FLAGS.rnn_dropout > 0.0):
 107       if (self.mode == "train"):
 108         rnn_dropout_mask = tf.cast(
 109             tf.random_uniform(
 110                 tf.shape(question_hidden), minval=0.0, maxval=1.0) <
 111             self.utility.FLAGS.rnn_dropout,
 112             self.data_type) / self.utility.FLAGS.rnn_dropout
 113       else:
 114         rnn_dropout_mask = tf.ones_like(question_hidden)
 115     for question_iterator in range(self.question_length):
 116       curr_word = sentence[:, question_iterator]
 117       question_vector = nn_utils.apply_dropout(
 118           nn_utils.get_embedding(curr_word, self.utility, self.params),
 119           self.utility.FLAGS.dropout, self.mode)
 120       question_hidden, question_c_hidden = nn_utils.LSTMCell(
 121           question_vector, question_hidden, question_c_hidden, lstm_params,
 122           self.params)
 123       if (self.utility.FLAGS.rnn_dropout > 0.0):
 124         question_hidden = question_hidden * rnn_dropout_mask
 125       hidden_vectors.append(tf.expand_dims(question_hidden, 0))
 126     hidden_vectors = tf.concat(axis=0, values=hidden_vectors)
 127     return question_hidden, hidden_vectors

 136   def question_number_softmax(self, hidden_vectors):
 137     #Attention on quetsion to decide the question number to passed to comparison ops
 138     def compute_ans(op_embedding, comparison):
 139       op_embedding = tf.expand_dims(op_embedding, 0)
 140       #dot product of operation embedding with hidden state to the left of the number occurrence
 141       first = tf.transpose(
 142           tf.matmul(op_embedding,
 143                     tf.transpose(
 144                         tf.reduce_sum(hidden_vectors * tf.tile(
 145                             tf.expand_dims(
 146                                 tf.transpose(self.batch_ordinal_question), 2),
 147                             [1, 1, self.utility.FLAGS.embedding_dims]), 0))))
 148       second = self.batch_question_number_one_mask + tf.transpose(
 149           tf.matmul(op_embedding,
 150                     tf.transpose(
 151                         tf.reduce_sum(hidden_vectors * tf.tile(
 152                             tf.expand_dims(
 153                                 tf.transpose(self.batch_ordinal_question_one), 2
 154                             ), [1, 1, self.utility.FLAGS.embedding_dims]), 0))))
 155       question_number_softmax = tf.nn.softmax(tf.concat(axis=1, values=[first, second]))
 156       if (self.mode == "test"):
 157         cond = tf.equal(question_number_softmax,
 158                         tf.reshape(
 159                             tf.reduce_max(question_number_softmax, 1),
 160                             [self.batch_size, 1]))
 161         question_number_softmax = tf.where(
 162             cond,
 163             tf.fill(tf.shape(question_number_softmax), 1.0),
 164             tf.fill(tf.shape(question_number_softmax), 0.0))
 165         question_number_softmax = tf.cast(question_number_softmax,
 166                                           self.data_type)
 167       ans = tf.reshape(
 168           tf.reduce_sum(question_number_softmax * tf.concat(
 169               axis=1, values=[self.batch_question_number, self.batch_question_number_one]),
 170                         1), [self.batch_size, 1])
 171       return ans

 189   def perform_attention(self, context_vector, hidden_vectors, length, mask):
 190     #Performs attention on hiddent_vectors using context vector
 191     context_vector = tf.tile(
 192         tf.expand_dims(context_vector, 0), [length, 1, 1])  #time * bs * d
 193     attention_softmax = tf.nn.softmax(
 194         tf.transpose(tf.reduce_sum(context_vector * hidden_vectors, 2)) +
 195         mask)  #batch_size * time
 196     attention_softmax = tf.tile(
 197         tf.expand_dims(tf.transpose(attention_softmax), 2),
 198         [1, 1, self.embedding_dims])
 199     ans_vector = tf.reduce_sum(attention_softmax * hidden_vectors, 0)
 200     return ans_vector

 222   def compute_column_softmax(self, column_controller_vector, time_step):
 223     #compute softmax over all the columns using column controller vector
 224     column_controller_vector = tf.tile(
 225         tf.expand_dims(column_controller_vector, 1),
 226         [1, self.num_cols + self.num_word_cols, 1])  #max_cols * bs * d
 227     column_controller_vector = nn_utils.apply_dropout(
 228         column_controller_vector, self.utility.FLAGS.dropout, self.mode)
 229     self.full_column_hidden_vectors = tf.concat(
 230         axis=1, values=[self.column_hidden_vectors, self.word_column_hidden_vectors])
 231     self.full_column_hidden_vectors += self.summary_text_entry_embeddings
 232     self.full_column_hidden_vectors = nn_utils.apply_dropout(
 233         self.full_column_hidden_vectors, self.utility.FLAGS.dropout, self.mode)
 234     column_logits = tf.reduce_sum(
 235         column_controller_vector * self.full_column_hidden_vectors, 2) + (
 236             self.params["word_match_feature_column_name"] *
 237             self.batch_column_exact_match) + self.full_column_mask
 238     column_softmax = tf.nn.softmax(column_logits)  #batch_size * max_cols
 239     return column_softmax

 241   def compute_first_or_last(self, select, first=True):
 242     #perform first ot last operation on row select with probabilistic row selection
 243     answer = tf.zeros_like(select)
 244     running_sum = tf.zeros([self.batch_size, 1], self.data_type)
 245     for i in range(self.max_elements):
 246       if (first):
 247         current = tf.slice(select, [0, i], [self.batch_size, 1])
 248       else:
 249         current = tf.slice(select, [0, self.max_elements - 1 - i],
 250                            [self.batch_size, 1])
 251       curr_prob = current * (1 - running_sum)
 252       curr_prob = curr_prob * tf.cast(curr_prob >= 0.0, self.data_type)
 253       running_sum += curr_prob
 254       temp_ans = []
 255       curr_prob = tf.expand_dims(tf.reshape(curr_prob, [self.batch_size]), 0)
 256       for i_ans in range(self.max_elements):
 257         if (not (first) and i_ans == self.max_elements - 1 - i):
 258           temp_ans.append(curr_prob)
 259         elif (first and i_ans == i):
 260           temp_ans.append(curr_prob)
 261         else:
 262           temp_ans.append(tf.zeros_like(curr_prob))
 263       temp_ans = tf.transpose(tf.concat(axis=0, values=temp_ans))
 264       answer += temp_ans
 265     return answer

 267   def make_hard_softmax(self, softmax):
 268     #converts soft selection to hard selection. used at test time
 269     cond = tf.equal(
 270         softmax, tf.reshape(tf.reduce_max(softmax, 1), [self.batch_size, 1]))
 271     softmax = tf.where(
 272         cond, tf.fill(tf.shape(softmax), 1.0), tf.fill(tf.shape(softmax), 0.0))
 273     softmax = tf.cast(softmax, self.data_type)
 274     return softmax

 276   def compute_max_or_min(self, select, maxi=True):
 277     #computes the argmax and argmin of a column with probabilistic row selection
 278     answer = tf.zeros([
 279         self.batch_size, self.num_cols + self.num_word_cols, self.max_elements
 280     ], self.data_type)
 281     sum_prob = tf.zeros([self.batch_size, self.num_cols + self.num_word_cols],
 282                         self.data_type)
 283     for j in range(self.max_elements):
 284       if (maxi):
 285         curr_pos = j
 286       else:
 287         curr_pos = self.max_elements - 1 - j
 288       select_index = tf.slice(self.full_processed_sorted_index_column,
 289                               [0, 0, curr_pos], [self.batch_size, -1, 1])
 290       select_mask = tf.equal(
 291           tf.tile(
 292               tf.expand_dims(
 293                   tf.tile(
 294                       tf.expand_dims(tf.range(self.max_elements), 0),
 295                       [self.batch_size, 1]), 1),
 296               [1, self.num_cols + self.num_word_cols, 1]), select_index)
 297       curr_prob = tf.expand_dims(select, 1) * tf.cast(
 298           select_mask, self.data_type) * self.select_bad_number_mask
 299       curr_prob = curr_prob * tf.expand_dims((1 - sum_prob), 2)
 300       curr_prob = curr_prob * tf.expand_dims(
 301           tf.cast((1 - sum_prob) > 0.0, self.data_type), 2)
 302       answer = tf.where(select_mask, curr_prob, answer)
 303       sum_prob += tf.reduce_sum(curr_prob, 2)
 304     return answer

 306   def perform_operations(self, softmax, full_column_softmax, select,
 307                          prev_select_1, curr_pass):
 308     #performs all the 15 operations. computes scalar output, lookup answer and row selector
 309     column_softmax = tf.slice(full_column_softmax, [0, 0],
 310                               [self.batch_size, self.num_cols])
 311     word_column_softmax = tf.slice(full_column_softmax, [0, self.num_cols],
 312                                    [self.batch_size, self.num_word_cols])
 313     init_max = self.compute_max_or_min(select, maxi=True)
 314     init_min = self.compute_max_or_min(select, maxi=False)
 315     #operations that are column  independent
 316     count = tf.reshape(tf.reduce_sum(select, 1), [self.batch_size, 1])
 317     select_full_column_softmax = tf.tile(
 318         tf.expand_dims(full_column_softmax, 2),
 319         [1, 1, self.max_elements
 320         ])  #BS * (max_cols + max_word_cols) * max_elements
 321     select_word_column_softmax = tf.tile(
 322         tf.expand_dims(word_column_softmax, 2),
 323         [1, 1, self.max_elements])  #BS * max_word_cols * max_elements
 324     select_greater = tf.reduce_sum(
 325         self.init_select_greater * select_full_column_softmax,
 326         1) * self.batch_question_number_mask  #BS * max_elements
 327     select_lesser = tf.reduce_sum(
 328         self.init_select_lesser * select_full_column_softmax,
 329         1) * self.batch_question_number_mask  #BS * max_elements
 330     select_geq = tf.reduce_sum(
 331         self.init_select_geq * select_full_column_softmax,
 332         1) * self.batch_question_number_mask  #BS * max_elements
 333     select_leq = tf.reduce_sum(
 334         self.init_select_leq * select_full_column_softmax,
 335         1) * self.batch_question_number_mask  #BS * max_elements
 336     select_max = tf.reduce_sum(init_max * select_full_column_softmax,
 337                                1)  #BS * max_elements
 338     select_min = tf.reduce_sum(init_min * select_full_column_softmax,
 339                                1)  #BS * max_elements
 340     select_prev = tf.concat(axis=1, values=[
 341         tf.slice(select, [0, 1], [self.batch_size, self.max_elements - 1]),
 342         tf.cast(tf.zeros([self.batch_size, 1]), self.data_type)
 343     ])
 344     select_next = tf.concat(axis=1, values=[
 345         tf.cast(tf.zeros([self.batch_size, 1]), self.data_type), tf.slice(
 346             select, [0, 0], [self.batch_size, self.max_elements - 1])
 347     ])
 348     select_last_rs = self.compute_first_or_last(select, False)
 349     select_first_rs = self.compute_first_or_last(select, True)
 350     select_word_match = tf.reduce_sum(self.batch_exact_match *
 351                                       select_full_column_softmax, 1)
 352     select_group_by_max = tf.reduce_sum(self.batch_group_by_max *
 353                                         select_full_column_softmax, 1)
 354     length_content = 1
 355     length_select = 13
 356     length_print = 1
 357     values = tf.concat(axis=1, values=[count])
 358     softmax_content = tf.slice(softmax, [0, 0],
 359                                [self.batch_size, length_content])
 360     #compute scalar output
 361     output = tf.reduce_sum(tf.multiply(softmax_content, values), 1)
 362     #compute lookup answer
 363     softmax_print = tf.slice(softmax, [0, length_content + length_select],
 364                              [self.batch_size, length_print])
 365     curr_print = select_full_column_softmax * tf.tile(
 366         tf.expand_dims(select, 1),
 367         [1, self.num_cols + self.num_word_cols, 1
 368         ])  #BS * max_cols * max_elements (conisders only column)
 369     self.batch_lookup_answer = curr_print * tf.tile(
 370         tf.expand_dims(softmax_print, 2),
 371         [1, self.num_cols + self.num_word_cols, self.max_elements
 372         ])  #BS * max_cols * max_elements
 373     self.batch_lookup_answer = self.batch_lookup_answer * self.select_full_mask
 374     #compute row select
 375     softmax_select = tf.slice(softmax, [0, length_content],
 376                               [self.batch_size, length_select])
 377     select_lists = [
 378         tf.expand_dims(select_prev, 1), tf.expand_dims(select_next, 1),
 379         tf.expand_dims(select_first_rs, 1), tf.expand_dims(select_last_rs, 1),
 380         tf.expand_dims(select_group_by_max, 1),
 381         tf.expand_dims(select_greater, 1), tf.expand_dims(select_lesser, 1),
 382         tf.expand_dims(select_geq, 1), tf.expand_dims(select_leq, 1),
 383         tf.expand_dims(select_max, 1), tf.expand_dims(select_min, 1),
 384         tf.expand_dims(select_word_match, 1),
 385         tf.expand_dims(self.reset_select, 1)
 386     ]
 387     select = tf.reduce_sum(
 388         tf.tile(tf.expand_dims(softmax_select, 2), [1, 1, self.max_elements]) *
 389         tf.concat(axis=1, values=select_lists), 1)
 390     select = select * self.select_whole_mask
 391     return output, select

 393   def one_pass(self, select, question_embedding, hidden_vectors, hprev,
 394                prev_select_1, curr_pass):
 395     #Performs one timestep which involves selecting an operation and a column
 396     attention_vector = self.perform_attention(
 397         hprev, hidden_vectors, self.question_length,
 398         self.batch_question_attention_mask)  #batch_size * embedding_dims
 399     controller_vector = tf.nn.relu(
 400         tf.matmul(hprev, self.params["controller_prev"]) + tf.matmul(
 401             tf.concat(axis=1, values=[question_embedding, attention_vector]), self.params[
 402                 "controller"]))
 403     column_controller_vector = tf.nn.relu(
 404         tf.matmul(hprev, self.params["column_controller_prev"]) + tf.matmul(
 405             tf.concat(axis=1, values=[question_embedding, attention_vector]), self.params[
 406                 "column_controller"]))
 407     controller_vector = nn_utils.apply_dropout(
 408         controller_vector, self.utility.FLAGS.dropout, self.mode)
 409     self.operation_logits = tf.matmul(controller_vector,
 410                                       tf.transpose(self.params_unit))
 411     softmax = tf.nn.softmax(self.operation_logits)
 412     soft_softmax = softmax
 413     #compute column softmax: bs * max_columns
 414     weighted_op_representation = tf.transpose(
 415         tf.matmul(tf.transpose(self.params_unit), tf.transpose(softmax)))
 416     column_controller_vector = tf.nn.relu(
 417         tf.matmul(
 418             tf.concat(axis=1, values=[
 419                 column_controller_vector, weighted_op_representation
 420             ]), self.params["break_conditional"]))
 421     full_column_softmax = self.compute_column_softmax(column_controller_vector,
 422                                                       curr_pass)
 423     soft_column_softmax = full_column_softmax
 424     if (self.mode == "test"):
 425       full_column_softmax = self.make_hard_softmax(full_column_softmax)
 426       softmax = self.make_hard_softmax(softmax)
 427     output, select = self.perform_operations(softmax, full_column_softmax,
 428                                              select, prev_select_1, curr_pass)
 429     return output, select, softmax, soft_softmax, full_column_softmax, soft_column_softmax

 431   def compute_lookup_error(self, val):
 432     #computes lookup error.
 433     cond = tf.equal(self.batch_print_answer, val)
 434     inter = tf.where(
 435         cond, self.init_print_error,
 436         tf.tile(
 437             tf.reshape(tf.constant(1e10, self.data_type), [1, 1, 1]), [
 438                 self.batch_size, self.utility.FLAGS.max_word_cols +
 439                 self.utility.FLAGS.max_number_cols,
 440                 self.utility.FLAGS.max_elements
 441             ]))
 442     return tf.reduce_min(tf.reduce_min(inter, 1), 1) * tf.cast(
 443         tf.greater(
 444             tf.reduce_sum(tf.reduce_sum(tf.cast(cond, self.data_type), 1), 1),
 445             0.0), self.data_type)

 447   def soft_min(self, x, y):
 448     return tf.maximum(-1.0 * (1 / (
 449         self.utility.FLAGS.soft_min_value + 0.0)) * tf.log(
 450             tf.exp(-self.utility.FLAGS.soft_min_value * x) + tf.exp(
 451                 -self.utility.FLAGS.soft_min_value * y)), tf.zeros_like(x))
 452 
 453   def error_computation(self):
 454     #computes the error of each example in a batch
 455     math_error = 0.5 * tf.square(tf.subtract(self.scalar_output, self.batch_answer))
 456     #scale math error
 457     math_error = math_error / self.rows
 458     math_error = tf.minimum(math_error, self.utility.FLAGS.max_math_error *
 459                             tf.ones(tf.shape(math_error), self.data_type))
 460     self.init_print_error = tf.where(
 461         self.batch_gold_select, -1 * tf.log(self.batch_lookup_answer + 1e-300 +
 462                                             self.invert_select_full_mask), -1 *
 463         tf.log(1 - self.batch_lookup_answer)) * self.select_full_mask
 464     print_error_1 = self.init_print_error * tf.cast(
 465         tf.equal(self.batch_print_answer, 0.0), self.data_type)
 466     print_error = tf.reduce_sum(tf.reduce_sum((print_error_1), 1), 1)
 467     for val in range(1, 58):
 468       print_error += self.compute_lookup_error(val + 0.0)
 469     print_error = print_error * self.utility.FLAGS.print_cost / self.num_entries
 470     if (self.mode == "train"):
 471       error = tf.where(
 472           tf.logical_and(
 473               tf.not_equal(self.batch_answer, 0.0),
 474               tf.not_equal(
 475                   tf.reduce_sum(tf.reduce_sum(self.batch_print_answer, 1), 1),
 476                   0.0)),
 477           self.soft_min(math_error, print_error),
 478           tf.where(
 479               tf.not_equal(self.batch_answer, 0.0), math_error, print_error))
 480     else:
 481       error = tf.where(
 482           tf.logical_and(
 483               tf.equal(self.scalar_output, 0.0),
 484               tf.equal(
 485                   tf.reduce_sum(tf.reduce_sum(self.batch_lookup_answer, 1), 1),
 486                   0.0)),
 487           tf.ones_like(math_error),
 488           tf.where(
 489               tf.equal(self.scalar_output, 0.0), print_error, math_error))
 490     return error

 492   def batch_process(self):
 493     #Computes loss and fraction of correct examples in a batch.
 494     self.params_unit = nn_utils.apply_dropout(
 495         self.params["unit"], self.utility.FLAGS.dropout, self.mode)
 496     batch_size = self.batch_size
 497     max_passes = self.max_passes
 498     num_timesteps = 1
 499     max_elements = self.max_elements
 500     select = tf.cast(
 501         tf.fill([self.batch_size, max_elements], 1.0), self.data_type)
 502     hprev = tf.cast(
 503         tf.fill([self.batch_size, self.embedding_dims], 0.0),
 504         self.data_type)  #running sum of the hidden states of the model
 505     output = tf.cast(tf.fill([self.batch_size, 1], 0.0),
 506                      self.data_type)  #output of the model
 507     correct = tf.cast(
 508         tf.fill([1], 0.0), self.data_type
 509     )  #to compute accuracy, returns number of correct examples for this batch
 510     total_error = 0.0
 511     prev_select_1 = tf.zeros_like(select)
 512     self.create_summary_embeddings()
 513     self.get_column_hidden_vectors()
 514     #get question embedding
 515     question_embedding, hidden_vectors = self.LSTM_question_embedding(
 516         self.batch_question, self.question_length)
 517     #compute arguments for comparison operation
 518     greater_question_number, lesser_question_number, geq_question_number, leq_question_number = self.question_number_softmax(
 519         hidden_vectors)
 520     self.init_select_greater = tf.cast(
 521         tf.greater(self.full_processed_column,
 522                    tf.expand_dims(greater_question_number, 2)), self.
 523         data_type) * self.select_bad_number_mask  #bs * max_cols * max_elements
 524     self.init_select_lesser = tf.cast(
 525         tf.less(self.full_processed_column,
 526                 tf.expand_dims(lesser_question_number, 2)), self.
 527         data_type) * self.select_bad_number_mask  #bs * max_cols * max_elements
 528     self.init_select_geq = tf.cast(
 529         tf.greater_equal(self.full_processed_column,
 530                          tf.expand_dims(geq_question_number, 2)), self.
 531         data_type) * self.select_bad_number_mask  #bs * max_cols * max_elements
 532     self.init_select_leq = tf.cast(
 533         tf.less_equal(self.full_processed_column,
 534                       tf.expand_dims(leq_question_number, 2)), self.
 535         data_type) * self.select_bad_number_mask  #bs * max_cols * max_elements
 536     self.init_select_word_match = 0
 537     if (self.utility.FLAGS.rnn_dropout > 0.0):
 538       if (self.mode == "train"):
 539         history_rnn_dropout_mask = tf.cast(
 540             tf.random_uniform(
 541                 tf.shape(hprev), minval=0.0, maxval=1.0) <
 542             self.utility.FLAGS.rnn_dropout,
 543             self.data_type) / self.utility.FLAGS.rnn_dropout
 544       else:
 545         history_rnn_dropout_mask = tf.ones_like(hprev)
 546     select = select * self.select_whole_mask
 547     self.batch_log_prob = tf.zeros([self.batch_size], dtype=self.data_type)
 548     #Perform max_passes and at each  pass select operation and column
 549     for curr_pass in range(max_passes):
 550       print("step: ", curr_pass)
 551       output, select, softmax, soft_softmax, column_softmax, soft_column_softmax = self.one_pass(
 552           select, question_embedding, hidden_vectors, hprev, prev_select_1,
 553           curr_pass)
 554       prev_select_1 = select
 555       #compute input to history RNN
 556       input_op = tf.transpose(
 557           tf.matmul(
 558               tf.transpose(self.params_unit), tf.transpose(
 559                   soft_softmax)))  #weighted average of emebdding of operations
 560       input_col = tf.reduce_sum(
 561           tf.expand_dims(soft_column_softmax, 2) *
 562           self.full_column_hidden_vectors, 1)
 563       history_input = tf.concat(axis=1, values=[input_op, input_col])
 564       history_input = nn_utils.apply_dropout(
 565           history_input, self.utility.FLAGS.dropout, self.mode)
 566       hprev = self.history_recurrent_step(history_input, hprev)
 567       if (self.utility.FLAGS.rnn_dropout > 0.0):
 568         hprev = hprev * history_rnn_dropout_mask
 569     self.scalar_output = output
 570     error = self.error_computation()
 571     cond = tf.less(error, 0.0001, name="cond")
 572     correct_add = tf.where(
 573         cond, tf.fill(tf.shape(cond), 1.0), tf.fill(tf.shape(cond), 0.0))
 574     correct = tf.reduce_sum(correct_add)
 575     error = error / batch_size
 576     total_error = tf.reduce_sum(error)
 577     total_correct = correct / batch_size
 578     return total_error, total_correct

 580   def compute_error(self):
 581     #Sets mask variables and performs batch processing
 582     self.batch_gold_select = self.batch_print_answer > 0.0
 583     self.full_column_mask = tf.concat(
 584         axis=1, values=[self.batch_number_column_mask, self.batch_word_column_mask])
 585     self.full_processed_column = tf.concat(
 586         axis=1,
 587         values=[self.batch_processed_number_column, self.batch_processed_word_column])
 588     self.full_processed_sorted_index_column = tf.concat(axis=1, values=[
 589         self.batch_processed_sorted_index_number_column,
 590         self.batch_processed_sorted_index_word_column
 591     ])
 592     self.select_bad_number_mask = tf.cast(
 593         tf.logical_and(
 594             tf.not_equal(self.full_processed_column,
 595                          self.utility.FLAGS.pad_int),
 596             tf.not_equal(self.full_processed_column,
 597                          self.utility.FLAGS.bad_number_pre_process)),
 598         self.data_type)
 599     self.select_mask = tf.cast(
 600         tf.logical_not(
 601             tf.equal(self.batch_number_column, self.utility.FLAGS.pad_int)),
 602         self.data_type)
 603     self.select_word_mask = tf.cast(
 604         tf.logical_not(
 605             tf.equal(self.batch_word_column_entry_mask,
 606                      self.utility.dummy_token_id)), self.data_type)
 607     self.select_full_mask = tf.concat(
 608         axis=1, values=[self.select_mask, self.select_word_mask])
 609     self.select_whole_mask = tf.maximum(
 610         tf.reshape(
 611             tf.slice(self.select_mask, [0, 0, 0],
 612                      [self.batch_size, 1, self.max_elements]),
 613             [self.batch_size, self.max_elements]),
 614         tf.reshape(
 615             tf.slice(self.select_word_mask, [0, 0, 0],
 616                      [self.batch_size, 1, self.max_elements]),
 617             [self.batch_size, self.max_elements]))
 618     self.invert_select_full_mask = tf.cast(
 619         tf.concat(axis=1, values=[
 620             tf.equal(self.batch_number_column, self.utility.FLAGS.pad_int),
 621             tf.equal(self.batch_word_column_entry_mask,
 622                      self.utility.dummy_token_id)
 623         ]), self.data_type)
 624     self.batch_lookup_answer = tf.zeros(tf.shape(self.batch_gold_select))
 625     self.reset_select = self.select_whole_mask
 626     self.rows = tf.reduce_sum(self.select_whole_mask, 1)
 627     self.num_entries = tf.reshape(
 628         tf.reduce_sum(tf.reduce_sum(self.select_full_mask, 1), 1),
 629         [self.batch_size])
 630     self.final_error, self.final_correct = self.batch_process()
 631     return self.final_error

 633   def create_graph(self, params, global_step):
 634     #Creates the graph to compute error, gradient computation and updates parameters
 635     self.params = params
 636     batch_size = self.batch_size
 637     learning_rate = tf.cast(self.utility.FLAGS.learning_rate, self.data_type)
 638     self.total_cost = self.compute_error()
 639     optimize_params = self.params.values()
 640     optimize_names = self.params.keys()
 641     print("optimize params ", optimize_names)
 642     if (self.utility.FLAGS.l2_regularizer > 0.0):
 643       reg_cost = 0.0
 644       for ind_param in self.params.keys():
 645         reg_cost += tf.nn.l2_loss(self.params[ind_param])
 646       self.total_cost += self.utility.FLAGS.l2_regularizer * reg_cost
 647     grads = tf.gradients(self.total_cost, optimize_params, name="gradients")
 648     grad_norm = 0.0
 649     for p, name in zip(grads, optimize_names):
 650       print("grads: ", p, name)
 651       if isinstance(p, tf.IndexedSlices):
 652         grad_norm += tf.reduce_sum(p.values * p.values)
 653       elif not (p == None):
 654         grad_norm += tf.reduce_sum(p * p)
 655     grad_norm = tf.sqrt(grad_norm)
 656     max_grad_norm = np.float32(self.utility.FLAGS.clip_gradients).astype(
 657         self.utility.np_data_type[self.utility.FLAGS.data_type])
 658     grad_scale = tf.minimum(
 659         tf.cast(1.0, self.data_type), max_grad_norm / grad_norm)
 660     clipped_grads = list()
 661     for p in grads:
 662       if isinstance(p, tf.IndexedSlices):
 663         tmp = p.values * grad_scale
 664         clipped_grads.append(tf.IndexedSlices(tmp, p.indices))
 665       elif not (p == None):
 666         clipped_grads.append(p * grad_scale)
 667       else:
 668         clipped_grads.append(p)
 669     grads = clipped_grads
 670     self.global_step = global_step
 671     params_list = self.params.values()
 672     params_list.append(self.global_step)
 673     adam = tf.train.AdamOptimizer(
 674         learning_rate,
 675         epsilon=tf.cast(self.utility.FLAGS.eps, self.data_type),
 676         use_locking=True)
 677     self.step = adam.apply_gradients(zip(grads, optimize_params),
 678 					global_step=self.global_step)
 679     self.init_op = tf.global_variables_initializer()
