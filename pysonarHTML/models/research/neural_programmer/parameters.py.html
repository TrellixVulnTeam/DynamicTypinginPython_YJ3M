  22 class Parameters:
  23 
  24   def __init__(self, u):
  25     self.utility = u
  26     self.init_seed_counter = 0
  27     self.word_init = {}

  29   def parameters(self, utility):
  30     params = {}
  31     inits = []
  32     embedding_dims = self.utility.FLAGS.embedding_dims
  33     params["unit"] = tf.Variable(
  34         self.RandomUniformInit([len(utility.operations_set), embedding_dims]))
  35     params["word"] = tf.Variable(
  36         self.RandomUniformInit([utility.FLAGS.vocab_size, embedding_dims]))
  37     params["word_match_feature_column_name"] = tf.Variable(
  38         self.RandomUniformInit([1]))
  39     params["controller"] = tf.Variable(
  40         self.RandomUniformInit([2 * embedding_dims, embedding_dims]))
  41     params["column_controller"] = tf.Variable(
  42         self.RandomUniformInit([2 * embedding_dims, embedding_dims]))
  43     params["column_controller_prev"] = tf.Variable(
  44         self.RandomUniformInit([embedding_dims, embedding_dims]))
  45     params["controller_prev"] = tf.Variable(
  46         self.RandomUniformInit([embedding_dims, embedding_dims]))
  47     global_step = tf.Variable(1, name="global_step")
  48     #weigths of question and history RNN (or LSTM)
  49     key_list = ["question_lstm"]
  50     for key in key_list:
  51       # Weights going from inputs to nodes.
  52       for wgts in ["ix", "fx", "cx", "ox"]:
  53         params[key + "_" + wgts] = tf.Variable(
  54             self.RandomUniformInit([embedding_dims, embedding_dims]))
  55       # Weights going from nodes to nodes.
  56       for wgts in ["im", "fm", "cm", "om"]:
  57         params[key + "_" + wgts] = tf.Variable(
  58             self.RandomUniformInit([embedding_dims, embedding_dims]))
  59       #Biases for the gates and cell
  60       for bias in ["i", "f", "c", "o"]:
  61         if (bias == "f"):
  62           print("forget gate bias")
  63           params[key + "_" + bias] = tf.Variable(
  64               tf.random_uniform([embedding_dims], 1.0, 1.1, self.utility.
  65                                 tf_data_type[self.utility.FLAGS.data_type]))
  66         else:
  67           params[key + "_" + bias] = tf.Variable(
  68               self.RandomUniformInit([embedding_dims]))
  69     params["history_recurrent"] = tf.Variable(
  70         self.RandomUniformInit([3 * embedding_dims, embedding_dims]))
  71     params["history_recurrent_bias"] = tf.Variable(
  72         self.RandomUniformInit([1, embedding_dims]))
  73     params["break_conditional"] = tf.Variable(
  74         self.RandomUniformInit([2 * embedding_dims, embedding_dims]))
  75     init = tf.global_variables_initializer()
  76     return params, global_step, init
  77 
  78   def RandomUniformInit(self, shape):
  79     """Returns a RandomUniform Tensor between -param_init and param_init."""
  80     param_seed = self.utility.FLAGS.param_seed
  81     self.init_seed_counter += 1
  82     return tf.random_uniform(
  83         shape, -1.0 *
  84         (np.float32(self.utility.FLAGS.param_init)
  85         ).astype(self.utility.np_data_type[self.utility.FLAGS.data_type]),
  86         (np.float32(self.utility.FLAGS.param_init)
  87         ).astype(self.utility.np_data_type[self.utility.FLAGS.data_type]),
  88         self.utility.tf_data_type[self.utility.FLAGS.data_type],
  89         param_seed + self.init_seed_counter)
