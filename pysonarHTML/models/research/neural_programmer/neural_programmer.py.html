 108 def evaluate(sess, data, batch_size, graph, i):
 109   #computes accuracy
 110   num_examples = 0.0
 111   gc = 0.0
 112   for j in range(0, len(data) - batch_size + 1, batch_size):
 113     [ct] = sess.run([graph.final_correct],
 114                     feed_dict=data_utils.generate_feed_dict(data, j, batch_size,
 115                                                             graph))
 116     gc += ct * batch_size
 117     num_examples += batch_size
 118   print("dev set accuracy   after ", i, " : ", gc / num_examples)
 119   print(num_examples, len(data))
 120   print("--------")

 123 def Train(graph, utility, batch_size, train_data, sess, model_dir,
 124           saver):
 125   #performs training
 126   curr = 0
 127   train_set_loss = 0.0
 128   utility.random.shuffle(train_data)
 129   start = time.time()
 130   for i in range(utility.FLAGS.train_steps):
 131     curr_step = i
 132     if (i > 0 and i % FLAGS.write_every == 0):
 133       model_file = model_dir + "/model_" + str(i)
 134       saver.save(sess, model_file)
 135     if curr + batch_size >= len(train_data):
 136       curr = 0
 137       utility.random.shuffle(train_data)
 138     step, cost_value = sess.run(
 139         [graph.step, graph.total_cost],
 140         feed_dict=data_utils.generate_feed_dict(
 141             train_data, curr, batch_size, graph, train=True, utility=utility))
 142     curr = curr + batch_size
 143     train_set_loss += cost_value
 144     if (i > 0 and i % FLAGS.eval_cycle == 0):
 145       end = time.time()
 146       time_taken = end - start
 147       print("step ", i, " ", time_taken, " seconds ")
 148       start = end
 149       print(" printing train set loss: ", train_set_loss / utility.FLAGS.eval_cycle)
 150       train_set_loss = 0.0

 153 def master(train_data, dev_data, utility):
 154   #creates TF graph and calls trainer or evaluator
 155   batch_size = utility.FLAGS.batch_size
 156   model_dir = utility.FLAGS.output_dir + "/model" + utility.FLAGS.job_id + "/"
 157   #create all paramters of the model
 158   param_class = parameters.Parameters(utility)
 159   params, global_step, init = param_class.parameters(utility)
 160   key = "test" if (FLAGS.evaluator_job) else "train"
 161   graph = model.Graph(utility, batch_size, utility.FLAGS.max_passes, mode=key)
 162   graph.create_graph(params, global_step)
 163   prev_dev_error = 0.0
 164   final_loss = 0.0
 165   final_accuracy = 0.0
 166   #start session
 167   with tf.Session() as sess:
 168     sess.run(init.name)
 169     sess.run(graph.init_op.name)
 170     to_save = params.copy()
 171     saver = tf.train.Saver(to_save, max_to_keep=500)
 172     if (FLAGS.evaluator_job):
 173       while True:
 174         selected_models = {}
 175         file_list = tf.gfile.ListDirectory(model_dir)
 176         for model_file in file_list:
 177           if ("checkpoint" in model_file or "index" in model_file or
 178               "meta" in model_file):
 179             continue
 180           if ("data" in model_file):
 181             model_file = model_file.split(".")[0]
 182           model_step = int(
 183               model_file.split("_")[len(model_file.split("_")) - 1])
 184           selected_models[model_step] = model_file
 185         file_list = sorted(selected_models.items(), key=lambda x: x[0])
 186         if (len(file_list) > 0):
 187           file_list = file_list[0:len(file_list) - 1]
 188         print("list of models: ", file_list)
 189         for model_file in file_list:
 190           model_file = model_file[1]
 191           print("restoring: ", model_file)
 192           saver.restore(sess, model_dir + "/" + model_file)
 193           model_step = int(
 194               model_file.split("_")[len(model_file.split("_")) - 1])
 195           print("evaluating on dev ", model_file, model_step)
 196           evaluate(sess, dev_data, batch_size, graph, model_step)
 197     else:
 198       ckpt = tf.train.get_checkpoint_state(model_dir)
 199       print("model dir: ", model_dir)
 200       if (not (tf.gfile.IsDirectory(utility.FLAGS.output_dir))):
 201         print("create dir: ", utility.FLAGS.output_dir)
 202         tf.gfile.MkDir(utility.FLAGS.output_dir)
 203       if (not (tf.gfile.IsDirectory(model_dir))):
 204         print("create dir: ", model_dir)
 205         tf.gfile.MkDir(model_dir)
 206       Train(graph, utility, batch_size, train_data, sess, model_dir,
 207             saver)

 209 def main(args):
 210   utility = Utility()
 211   train_name = "random-split-1-train.examples"
 212   dev_name = "random-split-1-dev.examples"
 213   test_name = "pristine-unseen-tables.examples"
 214   #load data
 215   dat = wiki_data.WikiQuestionGenerator(train_name, dev_name, test_name, FLAGS.data_dir)
 216   train_data, dev_data, test_data = dat.load()
 217   utility.words = []
 218   utility.word_ids = {}
 219   utility.reverse_word_ids = {}
 220   #construct vocabulary
 221   data_utils.construct_vocab(train_data, utility)
 222   data_utils.construct_vocab(dev_data, utility, True)
 223   data_utils.construct_vocab(test_data, utility, True)
 224   data_utils.add_special_words(utility)
 225   data_utils.perform_word_cutoff(utility)
 226   #convert data to int format and pad the inputs
 227   train_data = data_utils.complete_wiki_processing(train_data, utility, True)
 228   dev_data = data_utils.complete_wiki_processing(dev_data, utility, False)
 229   test_data = data_utils.complete_wiki_processing(test_data, utility, False)
 230   print("# train examples ", len(train_data))
 231   print("# dev examples ", len(dev_data))
 232   print("# test examples ", len(test_data))
 233   print("running open source")
 234   #construct TF graph and train or evaluate
 235   master(train_data, dev_data, utility)
