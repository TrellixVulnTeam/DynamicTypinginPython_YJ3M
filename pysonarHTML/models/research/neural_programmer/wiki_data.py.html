  36 def is_nan_or_inf(number):
  37   return math.isnan(number) or math.isinf(number)

  39 def strip_accents(s):
  40   u = unicode(s, "utf-8")
  41   u_new = ''.join(c for c in ud.normalize('NFKD', u) if ud.category(c) != 'Mn')
  42   return u_new.encode("utf-8")

  45 def correct_unicode(string):
  46   string = strip_accents(string)
  47   string = re.sub("\xc2\xa0", " ", string).strip()
  48   string = re.sub("\xe2\x80\x93", "-", string).strip()
  49   #string = re.sub(ur'[\u0300-\u036F]', "", string)
  50   string = re.sub("â€š", ",", string)
  51   string = re.sub("â€¦", "...", string)
  52   #string = re.sub("[Â·ãƒ»]", ".", string)
  53   string = re.sub("Ë†", "^", string)
  54   string = re.sub("Ëœ", "~", string)
  55   string = re.sub("â€¹", "<", string)
  56   string = re.sub("â€º", ">", string)
  57   #string = re.sub("[â€˜â€™Â´`]", "'", string)
  58   #string = re.sub("[â€œâ€Â«Â»]", "\"", string)
  59   #string = re.sub("[â€¢â€ â€¡]", "", string)
  60   #string = re.sub("[â€â€‘â€“â€”]", "-", string)
  61   string = re.sub(r'[\u2E00-\uFFFF]', "", string)
  62   string = re.sub("\\s+", " ", string).strip()
  63   return string

  66 def simple_normalize(string):
  67   string = correct_unicode(string)
  68   # Citations
  69   string = re.sub("\[(nb ?)?\d+\]", "", string)
  70   string = re.sub("\*+$", "", string)
  71   # Year in parenthesis
  72   string = re.sub("\(\d* ?-? ?\d*\)", "", string)
  73   string = re.sub("^\"(.*)\"$", "", string)
  74   return string

  77 def full_normalize(string):
  78   #print "an: ", string
  79   string = simple_normalize(string)
  80   # Remove trailing info in brackets
  81   string = re.sub("\[[^\]]*\]", "", string)
  82   # Remove most unicode characters in other languages
  83   string = re.sub(r'[\u007F-\uFFFF]', "", string.strip())
  84   # Remove trailing info in parenthesis
  85   string = re.sub("\([^)]*\)$", "", string.strip())
  86   string = final_normalize(string)
  87   # Get rid of question marks
  88   string = re.sub("\?", "", string).strip()
  89   # Get rid of trailing colons (usually occur in column titles)
  90   string = re.sub("\:$", " ", string).strip()
  91   # Get rid of slashes
  92   string = re.sub(r"/", " ", string).strip()
  93   string = re.sub(r"\\", " ", string).strip()
  94   # Replace colon, slash, and dash with space
  95   # Note: need better replacement for this when parsing time
  96   string = re.sub(r"\:", " ", string).strip()
  97   string = re.sub("/", " ", string).strip()
  98   string = re.sub("-", " ", string).strip()
  99   # Convert empty strings to UNK
 100   # Important to do this last or near last
 101   if not string:
 102     string = "UNK"
 103   return string

 105 def final_normalize(string):
 106   # Remove leading and trailing whitespace
 107   string = re.sub("\\s+", " ", string).strip()
 108   # Convert entirely to lowercase
 109   string = string.lower()
 110   # Get rid of strangely escaped newline characters
 111   string = re.sub("\\\\n", " ", string).strip()
 112   # Get rid of quotation marks
 113   string = re.sub(r"\"", "", string).strip()
 114   string = re.sub(r"\'", "", string).strip()
 115   string = re.sub(r"`", "", string).strip()
 116   # Get rid of *
 117   string = re.sub("\*", "", string).strip()
 118   return string

 120 def is_number(x):
 121   try:
 122     f = float(x)
 123     return not is_nan_or_inf(f)
 124   except ValueError:
 125     return False
 126   except TypeError:
 127     return False

 186 def is_date(word):
 187   if (not (bool(re.search("[a-z0-9]", word, re.IGNORECASE)))):
 188     return False
 189   if (len(word) != 10):
 190     return False
 191   if (word[4] != "-"):
 192     return False
 193   if (word[7] != "-"):
 194     return False
 195   for i in range(len(word)):
 196     if (not (word[i] == "X" or word[i] == "x" or word[i] == "-" or re.search(
 197         "[0-9]", word[i]))):
 198       return False
 199   return True

 221   def is_money(self, word):
 222     if (not (bool(re.search("[a-z0-9]", word, re.IGNORECASE)))):
 223       return False
 224     for i in range(len(word)):
 225       if (not (word[i] == "E" or word[i] == "." or re.search("[0-9]",
 226                                                              word[i]))):
 227         return False

 230   def remove_consecutive(self, ner_tags, ner_values):
 231     for i in range(len(ner_tags)):
 232       if ((ner_tags[i] == "NUMBER" or ner_tags[i] == "MONEY" or
 233            ner_tags[i] == "PERCENT" or ner_tags[i] == "DATE") and
 234           i + 1 < len(ner_tags) and ner_tags[i] == ner_tags[i + 1] and
 235           ner_values[i] == ner_values[i + 1] and ner_values[i] != ""):
 236         word = ner_values[i]
 237         word = word.replace(">", "").replace("<", "").replace("=", "").replace(
 238             "%", "").replace("~", "").replace("$", "").replace("£", "").replace(
 239                 "€", "")
 240         if (re.search("[A-Z]", word) and not (is_date(word)) and not (
 241             self.is_money(word))):
 242           ner_values[i] = "A"
 243         else:
 244           ner_values[i] = ","
 245     return ner_tags, ner_values

 247   def pre_process_sentence(self, tokens, ner_tags, ner_values):
 248     sentence = []
 249     tokens = tokens.split("|")
 250     ner_tags = ner_tags.split("|")
 251     ner_values = ner_values.split("|")
 252     ner_tags, ner_values = self.remove_consecutive(ner_tags, ner_values)
 253     #print "old: ", tokens
 254     for i in range(len(tokens)):
 255       word = tokens[i]
 256       if (ner_values[i] != "" and
 257           (ner_tags[i] == "NUMBER" or ner_tags[i] == "MONEY" or
 258            ner_tags[i] == "PERCENT" or ner_tags[i] == "DATE")):
 259         word = ner_values[i]
 260         word = word.replace(">", "").replace("<", "").replace("=", "").replace(
 261             "%", "").replace("~", "").replace("$", "").replace("£", "").replace(
 262                 "€", "")
 263         if (re.search("[A-Z]", word) and not (is_date(word)) and not (
 264             self.is_money(word))):
 265           word = tokens[i]
 266         if (is_number(ner_values[i])):
 267           word = float(ner_values[i])
 268         elif (is_number(word)):
 269           word = float(word)
 270         if (tokens[i] == "score"):
 271           word = "score"
 272       if (is_number(word)):
 273         word = float(word)
 274       if (not (self.annotated_word_reject.has_key(word))):
 275         if (is_number(word) or is_date(word) or self.is_money(word)):
 276           sentence.append(word)
 277         else:
 278           word = full_normalize(word)
 279           if (not (self.annotated_word_reject.has_key(word)) and
 280               bool(re.search("[a-z0-9]", word, re.IGNORECASE))):
 281             m = re.search(",", word)
 282             sentence.append(word.replace(",", ""))
 283     if (len(sentence) == 0):
 284       sentence.append("UNK")
 285     return sentence

 287   def load_annotated_data(self, in_file):
 288     self.annotated_examples = {}
 289     self.annotated_tables = {}
 290     f = tf.gfile.GFile(in_file, "r")
 291     counter = 0
 292     for line in f:
 293       if (counter > 0):
 294         line = line.strip()
 295         (question_id, utterance, context, target_value, tokens, lemma_tokens,
 296          pos_tags, ner_tags, ner_values, target_canon) = line.split("\t")
 297         question = self.pre_process_sentence(tokens, ner_tags, ner_values)
 298         target_canon = target_canon.split("|")
 299         self.annotated_examples[question_id] = WikiExample(
 300             question_id, question, target_canon, context)
 301         self.annotated_tables[context] = []
 302       counter += 1
 303     print("Annotated examples loaded ", len(self.annotated_examples))
 304     f.close()

 306   def is_number_column(self, a):
 307     for w in a:
 308       if (len(w) != 1):
 309         return False
 310       if (not (is_number(w[0]))):
 311         return False

 314   def convert_table(self, table):
 315     answer = []
 316     for i in range(len(table)):
 317       temp = []
 318       for j in range(len(table[i])):
 319         temp.append(" ".join([str(w) for w in table[i][j]]))
 320       answer.append(temp)
 321     return answer

 323   def load_annotated_tables(self):
 324     for table in self.annotated_tables.keys():
 325       annotated_table = table.replace("csv", "annotated")
 326       orig_columns = []
 327       processed_columns = []
 328       f = tf.gfile.GFile(os.path.join(self.root_folder, annotated_table), "r")
 329       counter = 0
 330       for line in f:
 331         if (counter > 0):
 332           line = line.strip()
 333           line = line + "\t" * (13 - len(line.split("\t")))
 334           (row, col, read_id, content, tokens, lemma_tokens, pos_tags, ner_tags,
 335            ner_values, number, date, num2, read_list) = line.split("\t")
 336         counter += 1
 337       f.close()
 338       max_row = int(row)
 339       max_col = int(col)
 340       for i in range(max_col + 1):
 341         orig_columns.append([])
 342         processed_columns.append([])
 343         for j in range(max_row + 1):
 344           orig_columns[i].append(bad_number)
 345           processed_columns[i].append(bad_number)
 346       #print orig_columns
 347       f = tf.gfile.GFile(os.path.join(self.root_folder, annotated_table), "r")
 348       counter = 0
 349       column_names = []
 350       for line in f:
 351         if (counter > 0):
 352           line = line.strip()
 353           line = line + "\t" * (13 - len(line.split("\t")))
 354           (row, col, read_id, content, tokens, lemma_tokens, pos_tags, ner_tags,
 355            ner_values, number, date, num2, read_list) = line.split("\t")
 356           entry = self.pre_process_sentence(tokens, ner_tags, ner_values)
 357           if (row == "-1"):
 358             column_names.append(entry)
 359           else:
 360             orig_columns[int(col)][int(row)] = entry
 361             if (len(entry) == 1 and is_number(entry[0])):
 362               processed_columns[int(col)][int(row)] = float(entry[0])
 363             else:
 364               for single_entry in entry:
 365                 if (is_number(single_entry)):
 366                   processed_columns[int(col)][int(row)] = float(single_entry)
 367                   break
 368               nt = ner_tags.split("|")
 369               nv = ner_values.split("|")
 370               for i_entry in range(len(tokens.split("|"))):
 371                 if (nt[i_entry] == "DATE" and
 372                     is_number(nv[i_entry].replace("-", "").replace("X", ""))):
 373                   processed_columns[int(col)][int(row)] = float(nv[
 374                       i_entry].replace("-", "").replace("X", ""))
 375                   #processed_columns[int(col)][int(row)] =  float(nv[i_entry])
 376             if (len(entry) == 1 and (is_number(entry[0]) or is_date(entry[0]) or
 377                                      self.is_money(entry[0]))):
 378               if (len(entry) == 1 and not (is_number(entry[0])) and
 379                   is_date(entry[0])):
 380                 entry[0] = entry[0].replace("X", "x")
 381         counter += 1
 382       word_columns = []
 383       processed_word_columns = []
 384       word_column_names = []
 385       word_column_indices = []
 386       number_columns = []
 387       processed_number_columns = []
 388       number_column_names = []
 389       number_column_indices = []
 390       for i in range(max_col + 1):
 391         if (self.is_number_column(orig_columns[i])):
 392           number_column_indices.append(i)
 393           number_column_names.append(column_names[i])
 394           temp = []
 395           for w in orig_columns[i]:
 396             if (is_number(w[0])):
 397               temp.append(w[0])
 398           number_columns.append(temp)
 399           processed_number_columns.append(processed_columns[i])
 400         else:
 401           word_column_indices.append(i)
 402           word_column_names.append(column_names[i])
 403           word_columns.append(orig_columns[i])
 404           processed_word_columns.append(processed_columns[i])
 405       table_info = TableInfo(
 406           word_columns, word_column_names, word_column_indices, number_columns,
 407           number_column_names, number_column_indices, processed_word_columns,
 408           processed_number_columns, orig_columns)
 409       self.annotated_tables[table] = table_info
 410       f.close()

 412   def answer_classification(self):
 413     lookup_questions = 0
 414     number_lookup_questions = 0
 415     word_lookup_questions = 0
 416     ambiguous_lookup_questions = 0
 417     number_questions = 0
 418     bad_questions = 0
 419     ice_bad_questions = 0
 420     tot = 0
 421     got = 0
 422     ice = {}
 423     with tf.gfile.GFile(
 424         self.root_folder + "/arvind-with-norms-2.tsv", mode="r") as f:
 425       lines = f.readlines()
 426       for line in lines:
 427         line = line.strip()
 428         if (not (self.annotated_examples.has_key(line.split("\t")[0]))):
 429           continue
 430         if (len(line.split("\t")) == 4):
 431           line = line + "\t" * (5 - len(line.split("\t")))
 432           if (not (is_number(line.split("\t")[2]))):
 433             ice_bad_questions += 1
 434         (example_id, ans_index, ans_raw, process_answer,
 435          matched_cells) = line.split("\t")
 436         if (ice.has_key(example_id)):
 437           ice[example_id].append(line.split("\t"))
 438         else:
 439           ice[example_id] = [line.split("\t")]
 440     for q_id in self.annotated_examples.keys():
 441       tot += 1
 442       example = self.annotated_examples[q_id]
 443       table_info = self.annotated_tables[example.table_key]
 444       # Figure out if the answer is numerical or lookup
 445       n_cols = len(table_info.orig_columns)
 446       n_rows = len(table_info.orig_columns[0])
 447       example.lookup_matrix = np.zeros((n_rows, n_cols))
 448       exact_matches = {}
 449       for (example_id, ans_index, ans_raw, process_answer,
 450            matched_cells) in ice[q_id]:
 451         for match_cell in matched_cells.split("|"):
 452           if (len(match_cell.split(",")) == 2):
 453             (row, col) = match_cell.split(",")
 454             row = int(row)
 455             col = int(col)
 456             if (row >= 0):
 457               exact_matches[ans_index] = 1
 458       answer_is_in_table = len(exact_matches) == len(example.answer)
 459       if (answer_is_in_table):
 460         for (example_id, ans_index, ans_raw, process_answer,
 461              matched_cells) in ice[q_id]:
 462           for match_cell in matched_cells.split("|"):
 463             if (len(match_cell.split(",")) == 2):
 464               (row, col) = match_cell.split(",")
 465               row = int(row)
 466               col = int(col)
 467               example.lookup_matrix[row, col] = float(ans_index) + 1.0
 468       example.lookup_number_answer = 0.0
 469       if (answer_is_in_table):
 470         lookup_questions += 1
 471         if len(example.answer) == 1 and is_number(example.answer[0]):
 472           example.number_answer = float(example.answer[0])
 473           number_lookup_questions += 1
 474           example.is_number_lookup = True
 475         else:
 476           #print "word lookup"
 477           example.calc_answer = example.number_answer = 0.0
 478           word_lookup_questions += 1
 479           example.is_word_lookup = True
 480       else:
 481         if (len(example.answer) == 1 and is_number(example.answer[0])):
 482           example.number_answer = example.answer[0]
 483           example.is_number_calc = True
 484         else:
 485           bad_questions += 1
 486           example.is_bad_example = True
 487           example.is_unknown_answer = True
 488       example.is_lookup = example.is_word_lookup or example.is_number_lookup
 489       if not example.is_word_lookup and not example.is_bad_example:
 490         number_questions += 1
 491         example.calc_answer = example.answer[0]
 492         example.lookup_number_answer = example.calc_answer
 493       # Split up the lookup matrix into word part and number part
 494       number_column_indices = table_info.number_column_indices
 495       word_column_indices = table_info.word_column_indices
 496       example.word_columns = table_info.word_columns
 497       example.number_columns = table_info.number_columns
 498       example.word_column_names = table_info.word_column_names
 499       example.processed_number_columns = table_info.processed_number_columns
 500       example.processed_word_columns = table_info.processed_word_columns
 501       example.number_column_names = table_info.number_column_names
 502       example.number_lookup_matrix = example.lookup_matrix[:,
 503                                                            number_column_indices]
 504       example.word_lookup_matrix = example.lookup_matrix[:, word_column_indices]

 506   def load(self):
 507     train_data = []
 508     dev_data = []
 509     test_data = []
 510     self.load_annotated_data(
 511         os.path.join(self.data_folder, "training.annotated"))
 512     self.load_annotated_tables()
 513     self.answer_classification()
 514     self.train_loader.load()
 515     self.dev_loader.load()
 516     for i in range(self.train_loader.num_questions()):
 517       example = self.train_loader.examples[i]
 518       example = self.annotated_examples[example]
 519       train_data.append(example)
 520     for i in range(self.dev_loader.num_questions()):
 521       example = self.dev_loader.examples[i]
 522       dev_data.append(self.annotated_examples[example])
 523 
 524     self.load_annotated_data(
 525         os.path.join(self.data_folder, "pristine-unseen-tables.annotated"))
 526     self.load_annotated_tables()
 527     self.answer_classification()
 528     self.test_loader.load()
 529     for i in range(self.test_loader.num_questions()):
 530       example = self.test_loader.examples[i]
 531       test_data.append(self.annotated_examples[example])
 532     return train_data, dev_data, test_data
