  31   def test_elbo(self):
  32     """A golden-value test for the ELBO (the IWAE bound with num_samples=1)."""
  33     tf.set_random_seed(1234)
  34     with self.test_session() as sess:
  35       model, inputs, targets, lengths = create_vrnn(random_seed=1234)
  36       outs = bounds.iwae(model, (inputs, targets), lengths, num_samples=1,
  37                          parallel_iterations=1)
  38       sess.run(tf.global_variables_initializer())
  39       log_p_hat, _, _ = sess.run(outs)
  40       self.assertAllClose([-21.615765, -13.614225], log_p_hat)
