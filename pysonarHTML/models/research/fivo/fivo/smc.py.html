  28 def ess_criterion(log_weights, unused_t):
  29   """A criterion that resamples based on effective sample size."""
  30   num_particles = tf.shape(log_weights)[0]
  31   # Calculate the effective sample size.
  32   ess_num = 2 * tf.reduce_logsumexp(log_weights, axis=0)
  33   ess_denom = tf.reduce_logsumexp(2 * log_weights, axis=0)
  34   log_ess = ess_num - ess_denom
  35   return log_ess <= tf.log(tf.to_float(num_particles) / 2.0)

  38 def never_resample_criterion(log_weights, unused_t):
  39   """A criterion that never resamples."""
  40   batch_size = tf.shape(log_weights)[1]
  41   return tf.cast(tf.zeros([batch_size]), tf.bool)
  42 
  43 
  44 def always_resample_criterion(log_weights, unused_t):
  45   """A criterion resamples at every timestep."""
  46   batch_size = tf.shape(log_weights)[1]
  47   return tf.cast(tf.ones([batch_size]), tf.bool)

  50 def multinomial_resampling(log_weights, states, num_particles, batch_size,
  51                            random_seed=None):
  52   """Resample states with multinomial resampling.
  53 
  54   Args:
  55     log_weights: A [num_particles, batch_size] Tensor representing a batch
  56       of batch_size logits for num_particles-ary Categorical distribution.
  57     states: A nested list of [batch_size*num_particles, data_size] Tensors that
  58       will be resampled from the groups of every num_particles-th row.
  59     num_particles: The number of particles/samples.
  60     batch_size: The batch size.
  61     random_seed: The random seed to pass to the resampling operations in
  62       the particle filter. Mainly useful for testing.
  63 
  64   Returns:
  65     resampled_states: A nested list of [batch_size*num_particles, data_size]
  66       Tensors resampled via multinomial sampling.
  67   """
  68   # Calculate the ancestor indices via resampling. Because we maintain the
  69   # log unnormalized weights, we pass the weights in as logits, allowing
  70   # the distribution object to apply a softmax and normalize them.
  71   resampling_parameters = tf.transpose(log_weights, perm=[1, 0])
  72   resampling_dist = tf.contrib.distributions.Categorical(
  73       logits=resampling_parameters)
  74   ancestors = tf.stop_gradient(
  75       resampling_dist.sample(sample_shape=num_particles, seed=random_seed))
  76 
  77   # Because the batch is flattened, we must modify ancestor_inds to index the
  78   # proper samples. The particles in the ith filter are distributed every
  79   # batch_size rows in the batch, and offset i rows from the top. So, to
  80   # correct the indices we multiply by the batch_size and add the proper offset.
  81   # Crucially, when ancestor_inds is flattened the layout of the batch is
  82   # maintained.
  83   offset = tf.expand_dims(tf.range(batch_size), 0)
  84   ancestor_inds = tf.reshape(ancestors * batch_size + offset, [-1])
  85 
  86   resampled_states = nested.gather_tensors(states, ancestor_inds)
  87   return resampled_states

  90 def _blend_tensor(blending_weights, tensor, num_particles, batch_size):
  91   """Blend tensor according to the weights.
  92 
  93   The first dimension of tensor is actually a 2d index compacted to a 1d
  94   index and similarly for blended_tensor. So if we index these Tensors
  95   by [(i, j), k], then
  96 
  97     blended_tensor[(i, j), k] =
  98       sum_l tensor[(l, j), :] * blending_weights[i, j, l].
  99 
 100   Args:
 101     blending_weights: [num_particles, batch_size, num_particles] weights where
 102       the indices represent [sample index, batch index, blending weight index].
 103     tensor: [num_particles * batch_size, state_dim] Tensor to be blended.
 104     num_particles: The number of particles/samples.
 105     batch_size: The batch size.
 106 
 107   Returns:
 108     blended_tensor: [num_particles*batch_size, state_dim] blended Tensor.
 109   """
 110   # tensor is currently [num_particles * batch_size, state_dim], so we reshape
 111   # it to [num_particles, batch_size, state_dim]. Then, transpose it to
 112   # [batch_size, state_size, num_particles].
 113   tensor = tf.transpose(
 114       tf.reshape(tensor, [num_particles, batch_size, -1]), perm=[1, 2, 0])
 115   blending_weights = tf.transpose(blending_weights, perm=[1, 2, 0])
 116   # blendeding_weights is [batch index, blending weight index, sample index].
 117   # Multiplying these gives a matrix of size [batch_size, state_size,
 118   # num_particles].
 119   tensor = tf.matmul(tensor, blending_weights)
 120   # transpose the tensor to be [num_particles, batch_size, state_size]
 121   # and then reshape it to match the original format.
 122   tensor = tf.reshape(tf.transpose(tensor, perm=[2, 0, 1]),
 123                       [num_particles*batch_size, -1])
 124   return tensor

 127 def relaxed_resampling(log_weights, states, num_particles, batch_size,
 128                        temperature=0.5, random_seed=None):
 129   """Resample states with relaxed resampling.
 130 
 131   Draw soft "ancestors" using the Gumbel-Softmax distribution.
 132 
 133   Args:
 134     log_weights: A [num_particles, batch_size] Tensor representing a batch
 135       of batch_size logits for num_particles-ary Categorical distribution.
 136     states: A nested list of [batch_size * num_particles, d] Tensors that will
 137       be resampled from the groups of every num_particles-th row.
 138     num_particles: The number of particles/samples.
 139     batch_size: The batch size.
 140     temperature: The temperature used for the relaxed one hot distribution.
 141     random_seed: The random seed to pass to the resampling operations in
 142       the particle filter. Mainly useful for testing.
 143 
 144   Returns:
 145     resampled_states: A nested list of [batch_size * num_particles, d]
 146       Tensors resampled via multinomial sampling.
 147   """
 148   # log_weights are [num_particles, batch_size], so we transpose to get a
 149   # set of batch_size distributions over [0, num_particles).
 150   resampling_parameters = tf.transpose(log_weights, perm=[1, 0])
 151   resampling_dist = tf.contrib.distributions.RelaxedOneHotCategorical(
 152       temperature,
 153       logits=resampling_parameters)
 154 
 155   # Sample num_particles samples from the distribution, resulting in a
 156   # [num_particles, batch_size, num_particles] Tensor that represents a set of
 157   # [num_particles, batch_size] blending weights. The dimensions represent
 158   # [particle index, batch index, blending weight index].
 159   ancestors = resampling_dist.sample(sample_shape=num_particles,
 160                                      seed=random_seed)
 161   def map_fn(tensor):
 162     return _blend_tensor(ancestors, tensor, num_particles, batch_size)
 163 
 164   resampled_states = nested.map_nested(map_fn, states)
 165   return resampled_states

 168 def smc(
 169     transition_fn,
 170     num_steps,
 171     num_particles=1,
 172     resampling_criterion=ess_criterion,
 173     resampling_fn=multinomial_resampling,
 174     loop_fn=None,
 175     parallel_iterations=30,
 176     swap_memory=True):
 177   """Run a sequential Monte Carlo (SMC) algorithm.
 178 
 179   This method runs an SMC algorithm that evolves systems of particles
 180   using the supplied transition function for the specified number of steps. The
 181   particles are optionally resampled using resampling_fn when indicated by
 182   resampling_criterion.
 183 
 184   Args:
 185     transition_fn: A callable that propogates a batch of particles one step.
 186       Must accept as arguments a batch of particle states and the current
 187       timestep. Must return the particle states one timestep in the future, the
 188       incremental weights of each particle as a [num_samples*batch_size] float
 189       Tensor, and optionally a set of arguments to pass to the loop_fn. If
 190       the loop args are not provided, they will be set to None. Before the
 191       first timestep transition_fn will be called with the arguments None, -1
 192       and should return the initial particle states.
 193     num_steps: A [batch_size] Tensor of ints representing the number of steps
 194       to run each filter for.
 195     num_particles: A scalar int, the number of particles to use in each filter.
 196     resampling_criterion: The resampling criterion to use for this particle
 197       filter. Must accept the current log weights and timestep and
 198       return a boolean Tensor of shape [batch_size] indicating whether each
 199       particle filter should resample. See ess_criterion and related functions
 200       for examples. When resampling_criterion is never_resample_criterion,
 201       resampling_fn is ignored and never called.
 202     resampling_fn: A callable that performs the resampling operation. Must
 203       accept as arguments the log weights, particle states, num_particles,
 204       and batch_size and return the resampled particle states. See
 205       multinomial_resampling and relaxed_resampling for examples.
 206     loop_fn: A callable that performs operations on the weights and
 207       particle states, useful for accumulating and processing state that
 208       shouldn't be resampled. At each timestep after (possibly) resampling
 209       loop_fn will be called with the previous loop_state, a set of arguments
 210       produced by transition_fn called loop_args, the resampled particle states,
 211       the current log weights as [num_particles, batch_size] float Tensor, a
 212       [batch_size] float Tensor representing whether or not each filter
 213       resampled, the current mask indicating which filters are active, and the
 214       current timestep. It must return the next loop state. Before the first
 215       timestep loop_fn will be called with the arguments None, None, None, None,
 216       -1 and must return the initial loop state. The loop state can be a
 217       possibly nested structure of Tensors and TensorArrays.
 218     parallel_iterations: The number of parallel iterations to use for the
 219       internal while loop. Note that values greater than 1 can introduce
 220       non-determinism even when resampling is deterministic.
 221     swap_memory: Whether GPU-CPU memory swapping should be enabled for the
 222       internal while loop.
 223 
 224   Returns:
 225     log_z_hat: A Tensor of shape [batch_size] containing an estimate of the log
 226       normalizing constant that converts between the unormalized target
 227       distribution (as defined by the weights) and the true target distribution.
 228     log_weights: A Tensor of shape [max_num_steps, batch_size, num_particles]
 229       containing the log weights at each timestep of the particle filter.
 230       Will not be valid for timesteps past the supplied num_steps.
 231     resampled: A float Tensor of shape [max_num_steps, batch_size] indicating
 232       when the particle filters resampled. Will be 1.0 on timesteps when
 233       resampling occurred and 0.0 on timesteps when it did not.
 234     final_loop_state: The final state returned by loop_fn. If loop_fn is None
 235       then 0 will be returned.
 236   """
 237   # batch_size represents the number of particle filters running in parallel.
 238   batch_size = tf.shape(num_steps)[0]
 239   # Create a TensorArray where element t is the [num_particles*batch_size]
 240   # sequence mask for timestep t.
 241   max_num_steps = tf.reduce_max(num_steps)
 242   seq_mask = tf.transpose(
 243       tf.sequence_mask(num_steps, maxlen=max_num_steps, dtype=tf.float32),
 244       perm=[1, 0])
 245   seq_mask = tf.tile(seq_mask, [1, num_particles])
 246   mask_ta = tf.TensorArray(seq_mask.dtype,
 247                            max_num_steps,
 248                            name='mask_ta')
 249   mask_ta = mask_ta.unstack(seq_mask)
 250   # Initialize the state.
 251   t0 = tf.constant(0, tf.int32)
 252   init_particle_state = transition_fn(None, -1)
 253 
 254   def transition(*args):
 255     transition_outs = transition_fn(*args)
 256     if len(transition_outs) == 2:
 257       return transition_outs + (None,)
 258     else:
 259       return transition_outs
 260 
 261   if loop_fn is None:
 262     loop_fn = lambda *args: 0
 263 
 264   init_loop_state = loop_fn(None, None, None, None, None, None, -1)
 265   init_states = (init_particle_state, init_loop_state)
 266   ta_names = ['log_weights', 'resampled']
 267   tas = [tf.TensorArray(tf.float32, max_num_steps, name='%s_ta' % n)
 268          for n in ta_names]
 269   log_weights_acc = tf.zeros([num_particles, batch_size], dtype=tf.float32)
 270   log_z_hat_acc = tf.zeros([batch_size], dtype=tf.float32)
 271 
 272   def while_predicate(t, *unused_args):
 273     return t < max_num_steps
 274 
 275   def while_step(t, state, tas, log_weights_acc, log_z_hat_acc):
 276     """Implements one timestep of the particle filter."""
 277     particle_state, loop_state = state
 278     cur_mask = nested.read_tas(mask_ta, t)
 279     # Propagate the particles one step.
 280     log_alpha, new_particle_state, loop_args = transition(particle_state, t)
 281     # Update the current weights with the incremental weights.
 282     log_alpha *= cur_mask
 283     log_alpha = tf.reshape(log_alpha, [num_particles, batch_size])
 284     log_weights_acc += log_alpha
 285 
 286     should_resample = resampling_criterion(log_weights_acc, t)
 287 
 288     if resampling_criterion == never_resample_criterion:
 289       resampled = tf.to_float(should_resample)
 290     else:
 291       # Compute the states as if we did resample.
 292       resampled_states = resampling_fn(
 293           log_weights_acc,
 294           new_particle_state,
 295           num_particles,
 296           batch_size)
 297       # Decide whether or not we should resample; don't resample if we are past
 298       # the end of a sequence.
 299       should_resample = tf.logical_and(should_resample,
 300                                        cur_mask[:batch_size] > 0.)
 301       float_should_resample = tf.to_float(should_resample)
 302       new_particle_state = nested.where_tensors(
 303           tf.tile(should_resample, [num_particles]),
 304           resampled_states,
 305           new_particle_state)
 306       resampled = float_should_resample
 307 
 308     new_loop_state = loop_fn(loop_state, loop_args, new_particle_state,
 309                              log_weights_acc, resampled, cur_mask, t)
 310     # Update log Z hat.
 311     log_z_hat_update = tf.reduce_logsumexp(
 312         log_weights_acc, axis=0) - tf.log(tf.to_float(num_particles))
 313     # If it is the last timestep, always add the update.
 314     log_z_hat_acc += tf.cond(t < max_num_steps - 1,
 315                              lambda: log_z_hat_update * resampled,
 316                              lambda: log_z_hat_update)
 317     # Update the TensorArrays before we reset the weights so that we capture
 318     # the incremental weights and not zeros.
 319     ta_updates = [log_weights_acc, resampled]
 320     new_tas = [ta.write(t, x) for ta, x in zip(tas, ta_updates)]
 321     # For the particle filters that resampled, reset weights to zero.
 322     log_weights_acc *= (1. - tf.tile(resampled[tf.newaxis, :],
 323                                      [num_particles, 1]))
 324     new_state = (new_particle_state, new_loop_state)
 325     return t + 1, new_state, new_tas, log_weights_acc, log_z_hat_acc
 326 
 327   _, final_state, tas, _, log_z_hat = tf.while_loop(
 328       while_predicate,
 329       while_step,
 330       loop_vars=(t0, init_states, tas, log_weights_acc, log_z_hat_acc),
 331       parallel_iterations=parallel_iterations,
 332       swap_memory=swap_memory)
 333 
 334   log_weights, resampled = [x.stack() for x in tas]
 335   log_weights = tf.transpose(log_weights, perm=[0, 2, 1])
 336   final_particle_state, final_loop_state = final_state
 337   return (log_z_hat, log_weights, resampled,
 338           final_particle_state, final_loop_state)
