  33 def run_train(config):
  34   """Runs training for a Gaussian HMM setup."""
  35 
  36   def create_logging_hook(step, bound_value, likelihood, bound_gap):
  37     """Creates a logging hook that prints the bound value periodically."""
  38     bound_label = config.bound + "/t"
  39     def summary_formatter(log_dict):
  40       string = ("Step {step}, %s: {value:.3f}, "
  41                 "likelihood: {ll:.3f}, gap: {gap:.3e}") % bound_label
  42       return string.format(**log_dict)
  43     logging_hook = tf.train.LoggingTensorHook(
  44         {"step": step, "value": bound_value,
  45          "ll": likelihood, "gap": bound_gap},
  46         every_n_iter=config.summarize_every,
  47         formatter=summary_formatter)
  48     return logging_hook

  50   def create_losses(model, observations, lengths):
  51     """Creates the loss to be optimized.
  52 
  53     Args:
  54       model: A Trainable GHMM model.
  55       observations: A set of observations.
  56       lengths: The lengths of each sequence in the observations.
  57     Returns:
  58       loss: A float Tensor that when differentiated yields the gradients
  59          to apply to the model. Should be optimized via gradient descent.
  60       bound: A float Tensor containing the value of the bound that is
  61          being optimized.
  62       true_ll: The true log-likelihood of the data under the model.
  63       bound_gap: The gap between the bound and the true log-likelihood.
  64     """
  65     # Compute lower bounds on the log likelihood.
  66     if config.bound == "elbo":
  67       ll_per_seq, _, _ = bounds.iwae(
  68           model, observations, lengths, num_samples=1,
  69           parallel_iterations=config.parallel_iterations
  70       )
  71     elif config.bound == "iwae":
  72       ll_per_seq, _, _ = bounds.iwae(
  73           model, observations, lengths, num_samples=config.num_samples,
  74           parallel_iterations=config.parallel_iterations
  75       )
  76     elif config.bound == "fivo":
  77       if config.resampling_type == "relaxed":
  78         ll_per_seq, _, _, _ = bounds.fivo(
  79             model,
  80             observations,
  81             lengths,
  82             num_samples=config.num_samples,
  83             resampling_criterion=smc.ess_criterion,
  84             resampling_type=config.resampling_type,
  85             relaxed_resampling_temperature=config.
  86             relaxed_resampling_temperature,
  87             random_seed=config.random_seed,
  88             parallel_iterations=config.parallel_iterations)
  89       else:
  90         ll_per_seq, _, _, _ = bounds.fivo(
  91             model, observations, lengths,
  92             num_samples=config.num_samples,
  93             resampling_criterion=smc.ess_criterion,
  94             resampling_type=config.resampling_type,
  95             random_seed=config.random_seed,
  96             parallel_iterations=config.parallel_iterations
  97         )
  98     ll_per_t = tf.reduce_mean(ll_per_seq / tf.to_float(lengths))
  99     # Compute the data's true likelihood under the model and the bound gap.
 100     true_ll_per_seq = model.likelihood(tf.squeeze(observations))
 101     true_ll_per_t = tf.reduce_mean(true_ll_per_seq / tf.to_float(lengths))
 102     bound_gap = true_ll_per_seq - ll_per_seq
 103     bound_gap = tf.reduce_mean(bound_gap/ tf.to_float(lengths))
 104     tf.summary.scalar("train_ll_bound", ll_per_t)
 105     tf.summary.scalar("train_true_ll", true_ll_per_t)
 106     tf.summary.scalar("bound_gap", bound_gap)
 107     return -ll_per_t, ll_per_t, true_ll_per_t, bound_gap

 109   def create_graph():
 110     """Creates the training graph."""
 111     global_step = tf.train.get_or_create_global_step()
 112     xs, lengths = datasets.create_chain_graph_dataset(
 113         config.batch_size,
 114         config.num_timesteps,
 115         steps_per_observation=1,
 116         state_size=1,
 117         transition_variance=config.variance,
 118         observation_variance=config.variance)
 119     model = ghmm.TrainableGaussianHMM(
 120         config.num_timesteps,
 121         config.proposal_type,
 122         transition_variances=config.variance,
 123         emission_variances=config.variance,
 124         random_seed=config.random_seed)
 125     loss, bound, true_ll, gap = create_losses(model, xs, lengths)
 126     opt = tf.train.AdamOptimizer(config.learning_rate)
 127     grads = opt.compute_gradients(loss, var_list=tf.trainable_variables())
 128     train_op = opt.apply_gradients(grads, global_step=global_step)
 129     return bound, true_ll, gap, train_op, global_step
 130 
 131   with tf.Graph().as_default():
 132     if config.random_seed:
 133       tf.set_random_seed(config.random_seed)
 134       np.random.seed(config.random_seed)
 135     bound, true_ll, gap, train_op, global_step = create_graph()
 136     log_hook = create_logging_hook(global_step, bound, true_ll, gap)
 137     with tf.train.MonitoredTrainingSession(
 138         master="",
 139         hooks=[log_hook],
 140         checkpoint_dir=config.logdir,
 141         save_checkpoint_secs=120,
 142         save_summaries_steps=config.summarize_every,
 143         log_step_count_steps=config.summarize_every*20) as sess:
 144       cur_step = -1
 145       while cur_step <= config.max_steps and not sess.should_stop():
 146         cur_step = sess.run(global_step)
 147         _, cur_step = sess.run([train_op, global_step])

 150 def run_eval(config):
 151   """Evaluates a Gaussian HMM using the given config."""
 152 
 153   def create_bound(model, xs, lengths):
 154     """Creates the bound to be evaluated."""
 155     if config.bound == "elbo":
 156       ll_per_seq, log_weights, _ = bounds.iwae(
 157           model, xs, lengths, num_samples=1,
 158           parallel_iterations=config.parallel_iterations
 159       )
 160     elif config.bound == "iwae":
 161       ll_per_seq, log_weights, _ = bounds.iwae(
 162           model, xs, lengths, num_samples=config.num_samples,
 163           parallel_iterations=config.parallel_iterations
 164       )
 165     elif config.bound == "fivo":
 166       ll_per_seq, log_weights, resampled, _ = bounds.fivo(
 167           model, xs, lengths,
 168           num_samples=config.num_samples,
 169           resampling_criterion=smc.ess_criterion,
 170           resampling_type=config.resampling_type,
 171           random_seed=config.random_seed,
 172           parallel_iterations=config.parallel_iterations
 173       )
 174     # Compute bound scaled by number of timesteps.
 175     bound_per_t = ll_per_seq / tf.to_float(lengths)
 176     if config.bound == "fivo":
 177       return bound_per_t, log_weights, resampled
 178     else:
 179       return bound_per_t, log_weights
 180 
 181   def create_graph():
 182     """Creates the dataset, model, and bound."""
 183     xs, lengths = datasets.create_chain_graph_dataset(
 184         config.batch_size,
 185         config.num_timesteps,
 186         steps_per_observation=1,
 187         state_size=1,
 188         transition_variance=config.variance,
 189         observation_variance=config.variance)
 190     model = ghmm.TrainableGaussianHMM(
 191         config.num_timesteps,
 192         config.proposal_type,
 193         transition_variances=config.variance,
 194         emission_variances=config.variance,
 195         random_seed=config.random_seed)
 196     true_likelihood = tf.reduce_mean(
 197         model.likelihood(tf.squeeze(xs)) / tf.to_float(lengths))
 198     outs = [true_likelihood]
 199     outs.extend(list(create_bound(model, xs, lengths)))
 200     return outs
 201 
 202   with tf.Graph().as_default():
 203     if config.random_seed:
 204       tf.set_random_seed(config.random_seed)
 205       np.random.seed(config.random_seed)
 206     graph_outs = create_graph()
 207     with tf.train.SingularMonitoredSession(
 208         checkpoint_dir=config.logdir) as sess:
 209       outs = sess.run(graph_outs)
 210       likelihood = outs[0]
 211       avg_bound = np.mean(outs[1])
 212       std = np.std(outs[1])
 213       log_weights = outs[2]
 214       log_weight_variances = np.var(log_weights, axis=2)
 215       avg_log_weight_variance = np.var(log_weight_variances, axis=1)
 216       avg_log_weight = np.mean(log_weights, axis=(1, 2))
 217       data = {"mean": avg_bound, "std": std, "log_weights": log_weights,
 218               "log_weight_means": avg_log_weight,
 219               "log_weight_variances": avg_log_weight_variance}
 220       if len(outs) == 4:
 221         data["resampled"] = outs[3]
 222         data["avg_resampled"] = np.mean(outs[3], axis=1)
 223       # Log some useful statistics.
 224       tf.logging.info("Evaled bound %s with batch_size: %d, num_samples: %d."
 225                       % (config.bound, config.batch_size, config.num_samples))
 226       tf.logging.info("mean: %f, std: %f" % (avg_bound, std))
 227       tf.logging.info("true likelihood: %s" % likelihood)
 228       tf.logging.info("avg log weight: %s" % avg_log_weight)
 229       tf.logging.info("log weight variance: %s" % avg_log_weight_variance)
 230       if len(outs) == 4:
 231         tf.logging.info("avg resamples per t: %s" % data["avg_resampled"])
 232       if not tf.gfile.Exists(config.logdir):
 233         tf.gfile.MakeDirs(config.logdir)
 234       with tf.gfile.Open(os.path.join(config.logdir, "out.npz"), "w") as fout:
 235         np.save(fout, data)
