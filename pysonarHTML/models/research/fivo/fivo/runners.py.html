  39 def create_dataset_and_model(config, split, shuffle, repeat):
  40   """Creates the dataset and model for a given config.
  41 
  42   Args:
  43     config: A configuration object with config values accessible as properties.
  44       Most likely a FLAGS object. This function expects the properties
  45       batch_size, dataset_path, dataset_type, and latent_size to be defined.
  46     split: The dataset split to load.
  47     shuffle: If true, shuffle the dataset randomly.
  48     repeat: If true, repeat the dataset endlessly.
  49   Returns:
  50     inputs: A batch of input sequences represented as a dense Tensor of shape
  51       [time, batch_size, data_dimension].
  52     targets: A batch of target sequences represented as a dense Tensor of
  53       shape [time, batch_size, data_dimension].
  54     lens: An int Tensor of shape [batch_size] representing the lengths of each
  55       sequence in the batch.
  56     model: A vrnn.VRNNCell model object.
  57   Raises:
  58     ValueError: if the config is invalid.
  59   """
  60   sigma_min = 0.0
  61   if config.dataset_type == "pianoroll":
  62     inputs, targets, lengths, mean = datasets.create_pianoroll_dataset(
  63         config.dataset_path, split, config.batch_size, shuffle=shuffle,
  64         repeat=repeat)
  65     # Convert the mean of the training set to logit space so it can be used to
  66     # initialize the bias of the generative distribution.
  67     emission_bias_init = -tf.log(
  68         1. / tf.clip_by_value(mean, 0.0001, 0.9999) - 1)
  69     emission_distribution_class = base.ConditionalBernoulliDistribution
  70   elif config.dataset_type == "speech":
  71     inputs, targets, lengths = datasets.create_speech_dataset(
  72         config.dataset_path, config.batch_size,
  73         samples_per_timestep=config.data_dimension, prefetch_buffer_size=1,
  74         shuffle=False, repeat=False)
  75     # There is no bias for the generative distribution because the test set
  76     # is assumed to be already standardized with the training set statistics.
  77     mean = None
  78     emission_bias_init = None
  79     emission_distribution_class = base.ConditionalNormalDistribution
  80   if config.model == "vrnn":
  81     model = vrnn.create_vrnn(inputs.get_shape().as_list()[2],
  82                              config.latent_size,
  83                              emission_distribution_class,
  84                              emission_bias_init=emission_bias_init,
  85                              proposal_type=config.proposal_type,
  86                              sigma_min=sigma_min,
  87                              raw_sigma_bias=0.5,
  88                              use_tilt=(config.bound == "fivo-aux"))
  89   elif config.model == "srnn":
  90     model = srnn.create_srnn(inputs.get_shape().as_list()[2],
  91                              config.latent_size,
  92                              emission_distribution_class,
  93                              emission_bias_init=emission_bias_init,
  94                              proposal_type=config.proposal_type,
  95                              sigma_min=sigma_min,
  96                              raw_sigma_bias=0.5,
  97                              use_tilt=(config.bound == "fivo-aux"))
  98   else:
  99     raise ValueError("model flag: %s is unrecognized" % config.model)
 100   return inputs, targets, lengths, model, mean

 103 def restore_checkpoint_if_exists(saver, sess, logdir):
 104   """Looks for a checkpoint and restores the session from it if found.
 105 
 106   Args:
 107     saver: A tf.train.Saver for restoring the session.
 108     sess: A TensorFlow session.
 109     logdir: The directory to look for checkpoints in.
 110   Returns:
 111     True if a checkpoint was found and restored, False otherwise.
 112   """
 113   checkpoint = tf.train.get_checkpoint_state(logdir)
 114   if checkpoint:
 115     checkpoint_name = os.path.basename(checkpoint.model_checkpoint_path)
 116     full_checkpoint_path = os.path.join(logdir, checkpoint_name)
 117     saver.restore(sess, full_checkpoint_path)
 118     return True
 119   return False

 122 def wait_for_checkpoint(saver, sess, logdir):
 123   """Loops until the session is restored from a checkpoint in logdir.
 124 
 125   Args:
 126     saver: A tf.train.Saver for restoring the session.
 127     sess: A TensorFlow session.
 128     logdir: The directory to look for checkpoints in.
 129   """
 130   while not restore_checkpoint_if_exists(saver, sess, logdir):
 131     tf.logging.info("Checkpoint not found in %s, sleeping for 60 seconds."
 132                     % logdir)
 133     time.sleep(60)

 136 def run_train(config, create_dataset_and_model_fn=create_dataset_and_model):
 137   """Runs training for a sequential latent variable model.
 138 
 139   Args:
 140     config: A configuration object with config values accessible as properties.
 141       Most likely a FLAGS object. For a list of expected properties and their
 142       meaning see the flags defined in fivo.py.
 143     create_dataset_and_model_fn: If present, calls this function to create a
 144       dataset and model instead of create_dataset_and_model() above. The
 145       signature must be the same.
 146   """
 147 
 148   def create_logging_hook(step, bound_value):
 149     """Creates a logging hook that prints the bound value periodically."""
 150     bound_label = config.bound + " bound"
 151     if config.normalize_by_seq_len:
 152       bound_label += " per timestep"
 153     else:
 154       bound_label += " per sequence"
 155     def summary_formatter(log_dict):
 156       return "Step %d, %s: %f" % (
 157           log_dict["step"], bound_label, log_dict["bound_value"])
 158     logging_hook = tf.train.LoggingTensorHook(
 159         {"step": step, "bound_value": bound_value},
 160         every_n_iter=config.summarize_every,
 161         formatter=summary_formatter)
 162     return logging_hook
 163 
 164   def create_loss():
 165     """Creates the loss to be optimized.
 166 
 167     Returns:
 168       bound: A float Tensor containing the value of the bound that is
 169         being optimized.
 170       loss: A float Tensor that when differentiated yields the gradients
 171         to apply to the model. Should be optimized via gradient descent.
 172     """
 173     inputs, targets, lengths, model, _ = create_dataset_and_model_fn(
 174         config, split="train", shuffle=True, repeat=True)
 175     # Compute lower bounds on the log likelihood.
 176     if config.bound == "elbo":
 177       ll_per_seq, _, _ = bounds.iwae(
 178           model, (inputs, targets), lengths, num_samples=1,
 179           parallel_iterations=config.parallel_iterations
 180       )
 181     elif config.bound == "iwae":
 182       ll_per_seq, _, _ = bounds.iwae(
 183           model, (inputs, targets), lengths, num_samples=config.num_samples,
 184           parallel_iterations=config.parallel_iterations
 185       )
 186     elif config.bound in ("fivo", "fivo-aux"):
 187       if config.resampling_type == "relaxed":
 188         ll_per_seq, _, _, _ = bounds.fivo(
 189             model, (inputs, targets),
 190             lengths,
 191             num_samples=config.num_samples,
 192             resampling_criterion=smc.ess_criterion,
 193             resampling_type=config.resampling_type,
 194             random_seed=config.random_seed,
 195             relaxed_resampling_temperature=config.
 196             relaxed_resampling_temperature,
 197             parallel_iterations=config.parallel_iterations
 198         )
 199       else:
 200         ll_per_seq, _, _, _ = bounds.fivo(
 201             model, (inputs, targets), lengths, num_samples=config.num_samples,
 202             resampling_criterion=smc.ess_criterion,
 203             resampling_type=config.resampling_type,
 204             random_seed=config.random_seed,
 205             parallel_iterations=config.parallel_iterations
 206         )
 207     # Compute loss scaled by number of timesteps.
 208     ll_per_t = tf.reduce_mean(ll_per_seq / tf.to_float(lengths))
 209     ll_per_seq = tf.reduce_mean(ll_per_seq)
 210 
 211     tf.summary.scalar("train_ll_per_seq", ll_per_seq)
 212     tf.summary.scalar("train_ll_per_t", ll_per_t)
 213 
 214     if config.normalize_by_seq_len:
 215       return ll_per_t, -ll_per_t
 216     else:
 217       return ll_per_seq, -ll_per_seq
 218 
 219   def create_graph():
 220     """Creates the training graph."""
 221     global_step = tf.train.get_or_create_global_step()
 222     bound, loss = create_loss()
 223     opt = tf.train.AdamOptimizer(config.learning_rate)
 224     grads = opt.compute_gradients(loss, var_list=tf.trainable_variables())
 225     train_op = opt.apply_gradients(grads, global_step=global_step)
 226     return bound, train_op, global_step
 227 
 228   device = tf.train.replica_device_setter(ps_tasks=config.ps_tasks)
 229   with tf.Graph().as_default():
 230     if config.random_seed: tf.set_random_seed(config.random_seed)
 231     with tf.device(device):
 232       bound, train_op, global_step = create_graph()
 233       log_hook = create_logging_hook(global_step, bound)
 234       start_training = not config.stagger_workers
 235       with tf.train.MonitoredTrainingSession(
 236           master=config.master,
 237           is_chief=config.task == 0,
 238           hooks=[log_hook],
 239           checkpoint_dir=config.logdir,
 240           save_checkpoint_secs=120,
 241           save_summaries_steps=config.summarize_every,
 242           log_step_count_steps=config.summarize_every) as sess:
 243         cur_step = -1
 244         while not sess.should_stop() and cur_step <= config.max_steps:
 245           if config.task > 0 and not start_training:
 246             cur_step = sess.run(global_step)
 247             tf.logging.info("task %d not active yet, sleeping at step %d" %
 248                             (config.task, cur_step))
 249             time.sleep(30)
 250             if cur_step >= config.task * 1000:
 251               start_training = True
 252           else:
 253             _, cur_step = sess.run([train_op, global_step])

 256 def run_eval(config, create_dataset_and_model_fn=create_dataset_and_model):
 257   """Runs evaluation for a sequential latent variable model.
 258 
 259   This method runs only one evaluation over the dataset, writes summaries to
 260   disk, and then terminates. It does not loop indefinitely.
 261 
 262   Args:
 263     config: A configuration object with config values accessible as properties.
 264       Most likely a FLAGS object. For a list of expected properties and their
 265       meaning see the flags defined in fivo.py.
 266     create_dataset_and_model_fn: If present, calls this function to create a
 267       dataset and model instead of create_dataset_and_model() above. The
 268       signature must be the same.
 269   """
 270 
 271   def create_graph():
 272     """Creates the evaluation graph.
 273 
 274     Returns:
 275       lower_bounds: A tuple of float Tensors containing the values of the 3
 276         evidence lower bounds, summed across the batch.
 277       total_batch_length: The total number of timesteps in the batch, summed
 278         across batch examples.
 279       batch_size: The batch size.
 280       global_step: The global step the checkpoint was loaded from.
 281     """
 282     global_step = tf.train.get_or_create_global_step()
 283     inputs, targets, lengths, model, _ = create_dataset_and_model_fn(
 284         config, split=config.split, shuffle=False, repeat=False)
 285     # Compute lower bounds on the log likelihood.
 286     elbo_ll_per_seq, _, _ = bounds.iwae(
 287         model, (inputs, targets), lengths, num_samples=1,
 288         parallel_iterations=config.parallel_iterations
 289     )
 290     iwae_ll_per_seq, _, _ = bounds.iwae(
 291         model, (inputs, targets), lengths, num_samples=config.num_samples,
 292         parallel_iterations=config.parallel_iterations
 293     )
 294     # The resampling type should only be used for training, so we ignore it.
 295     fivo_ll_per_seq, _, _, _ = bounds.fivo(
 296         model, (inputs, targets), lengths, num_samples=config.num_samples,
 297         resampling_criterion=smc.ess_criterion, random_seed=config.random_seed,
 298         parallel_iterations=config.parallel_iterations
 299     )
 300     elbo_ll = tf.reduce_sum(elbo_ll_per_seq)
 301     iwae_ll = tf.reduce_sum(iwae_ll_per_seq)
 302     fivo_ll = tf.reduce_sum(fivo_ll_per_seq)
 303     batch_size = tf.shape(lengths)[0]
 304     total_batch_length = tf.reduce_sum(lengths)
 305     return ((elbo_ll, iwae_ll, fivo_ll), total_batch_length, batch_size,
 306             global_step)

 308   def average_bounds_over_dataset(lower_bounds, total_batch_length, batch_size,
 309                                   sess):
 310     """Computes the values of the bounds, averaged over the datset.
 311 
 312     Args:
 313       lower_bounds: Tuple of float Tensors containing the values of the bounds
 314         evaluated on a single batch.
 315       total_batch_length: Integer Tensor that represents the total number of
 316         timesteps in the current batch.
 317       batch_size: Integer Tensor containing the batch size. This can vary if the
 318         requested batch_size does not evenly divide the size of the dataset.
 319       sess: A TensorFlow Session object.
 320     Returns:
 321       ll_per_t: A length 3 numpy array of floats containing each bound's average
 322         value, normalized by the total number of timesteps in the datset. Can
 323         be interpreted as a lower bound on the average log likelihood per
 324         timestep in the dataset.
 325       ll_per_seq: A length 3 numpy array of floats containing each bound's
 326         average value, normalized by the number of sequences in the dataset.
 327         Can be interpreted as a lower bound on the average log likelihood per
 328         sequence in the datset.
 329     """
 330     total_ll = np.zeros(3, dtype=np.float64)
 331     total_n_elems = 0.0
 332     total_length = 0.0
 333     while True:
 334       try:
 335         outs = sess.run([lower_bounds, batch_size, total_batch_length])
 336       except tf.errors.OutOfRangeError:
 337         break
 338       total_ll += outs[0]
 339       total_n_elems += outs[1]
 340       total_length += outs[2]
 341     ll_per_t = total_ll / total_length
 342     ll_per_seq = total_ll / total_n_elems
 343     return ll_per_t, ll_per_seq
 344 
 345   def summarize_lls(lls_per_t, lls_per_seq, summary_writer, step):
 346     """Creates log-likelihood lower bound summaries and writes them to disk.
 347 
 348     Args:
 349       lls_per_t: An array of 3 python floats, contains the values of the
 350         evaluated bounds normalized by the number of timesteps.
 351       lls_per_seq: An array of 3 python floats, contains the values of the
 352         evaluated bounds normalized by the number of sequences.
 353       summary_writer: A tf.SummaryWriter.
 354       step: The current global step.
 355     """
 356     def scalar_summary(name, value):
 357       value = tf.Summary.Value(tag=name, simple_value=value)
 358       return tf.Summary(value=[value])
 359 
 360     for i, bound in enumerate(["elbo", "iwae", "fivo"]):
 361       per_t_summary = scalar_summary("%s/%s_ll_per_t" % (config.split, bound),
 362                                      lls_per_t[i])
 363       per_seq_summary = scalar_summary("%s/%s_ll_per_seq" %
 364                                        (config.split, bound),
 365                                        lls_per_seq[i])
 366       summary_writer.add_summary(per_t_summary, global_step=step)
 367       summary_writer.add_summary(per_seq_summary, global_step=step)
 368     summary_writer.flush()
 369 
 370   with tf.Graph().as_default():
 371     if config.random_seed: tf.set_random_seed(config.random_seed)
 372     lower_bounds, total_batch_length, batch_size, global_step = create_graph()
 373     summary_dir = config.logdir + "/" + config.split
 374     summary_writer = tf.summary.FileWriter(
 375         summary_dir, flush_secs=15, max_queue=100)
 376     saver = tf.train.Saver()
 377     with tf.train.SingularMonitoredSession() as sess:
 378       wait_for_checkpoint(saver, sess, config.logdir)
 379       step = sess.run(global_step)
 380       tf.logging.info("Model restored from step %d, evaluating." % step)
 381       ll_per_t, ll_per_seq = average_bounds_over_dataset(
 382           lower_bounds, total_batch_length, batch_size, sess)
 383       summarize_lls(ll_per_t, ll_per_seq, summary_writer, step)
 384       tf.logging.info("%s elbo ll/t: %f, iwae ll/t: %f fivo ll/t: %f",
 385                       config.split, ll_per_t[0], ll_per_t[1], ll_per_t[2])
 386       tf.logging.info("%s elbo ll/seq: %f, iwae ll/seq: %f fivo ll/seq: %f",
 387                       config.split, ll_per_seq[0], ll_per_seq[1], ll_per_seq[2])

 390 def run_sample(config, create_dataset_and_model_fn=create_dataset_and_model):
 391   """Sample from the model. Only pianorolls and pose datasets are supported."""
 392 
 393   def sample_from_model(model, initial_state, initial_inputs, mean):
 394     """Samples a sequence of outputs from the model.
 395 
 396     The mean must be supplied -- if it isn't the results will be incorrect.
 397 
 398     Args:
 399       model: A model with sample_step implemented. See models/vrnn.py for an
 400         example.
 401       initial_state: The initial state of the model.
 402       initial_inputs: The initial inputs to feed into the model.
 403       mean: The mean of the training set, a Tensor of shape [data_dimension].
 404     Returns:
 405       samples: A Tensor of shape [sample_length, batch_size, num_timesteps,
 406         data_dimension] containing the samples from the model.
 407     """
 408     initial_state, initial_output = model.sample_step(initial_state,
 409                                                       initial_inputs, 0)
 410     output_ta = tf.TensorArray(size=config.sample_length,
 411                                dtype=tf.float32,
 412                                dynamic_size=False,
 413                                clear_after_read=True)
 414     output_ta = output_ta.write(0, initial_output)
 415     t0 = tf.constant(1, dtype=tf.int32)
 416 
 417     def sample_step(t, state, prev_outputs, output_ta):
 418       state, output = model.sample_step(state, prev_outputs, t)
 419       output_ta = output_ta.write(t, output)
 420       centered_output = output - mean[tf.newaxis, :]
 421       return t+1, state, centered_output, output_ta
 422 
 423     def sample_predicate(t, *unused_args):
 424       return t < config.sample_length
 425 
 426     _, _, _, output_ta = tf.while_loop(
 427         sample_predicate,
 428         sample_step,
 429         loop_vars=(t0, initial_state, initial_output, output_ta),
 430         parallel_iterations=config.parallel_iterations
 431     )
 432     samples = output_ta.stack()
 433     samples = tf.reshape(samples, [config.sample_length, config.batch_size,
 434                                    config.num_samples, config.data_dimension])
 435     return samples
 436 
 437   def create_graph():
 438     """Creates the graph to sample from the model.
 439 
 440     First, the model is conditioned on a prefix by sampling a batch of data
 441     and trimming it to prefix_length. The configured bound is used to do the
 442     conditioning. Then the final state from the conditioning is used to sample
 443     from the model.
 444 
 445     Returns:
 446       samples: A Tensor of shape [sample_length, batch_size,
 447         num_samples, data_dimension] representing samples from the model.
 448       prefixes: A Tensor of shape [prefix_length, batch_size, data_dimension]
 449         representing the prefixes the model was conditioned on.
 450     """
 451     inputs, targets, lengths, model, mean = create_dataset_and_model_fn(
 452         config, split=config.split, shuffle=True, repeat=True)
 453     input_prefixes = inputs[:config.prefix_length]
 454     target_prefixes = targets[:config.prefix_length]
 455     prefix_lengths = tf.ones_like(lengths) * config.prefix_length
 456     if config.bound == "elbo":
 457       _, _, state = bounds.iwae(
 458           model, (input_prefixes, target_prefixes),
 459           prefix_lengths, num_samples=1)
 460     elif config.bound == "iwae":
 461       _, _, state = bounds.iwae(
 462           model, (input_prefixes, target_prefixes),
 463           prefix_lengths, num_samples=config.num_samples)
 464     elif config.bound == "fivo":
 465       _, _, _, state = bounds.fivo(
 466           model, (input_prefixes, target_prefixes), prefix_lengths,
 467           num_samples=config.num_samples,
 468           resampling_criterion=smc.ess_criterion,
 469           random_seed=config.random_seed)
 470     sample_inputs = tf.tile(inputs[config.prefix_length],
 471                             [config.num_samples, 1])
 472     samples = sample_from_model(model, state, sample_inputs, mean)
 473     return samples, target_prefixes
 474 
 475   with tf.Graph().as_default():
 476     if config.random_seed:
 477       tf.set_random_seed(config.random_seed)
 478     samples, prefixes = create_graph()
 479     if config.sample_out_dir:
 480       out_dir = config.sample_our_dir
 481     else:
 482       out_dir = config.logdir
 483     if not tf.gfile.Exists(out_dir):
 484       tf.gfile.MakeDirs(out_dir)
 485     with tf.train.SingularMonitoredSession(
 486         checkpoint_dir=config.logdir) as sess:
 487       samples_out, prefixes_out = sess.run([samples, prefixes])
 488       with tf.gfile.Open(os.path.join(out_dir, "samples.npz"), "w") as fout:
 489         np.save(fout, {"prefixes": prefixes_out, "samples": samples_out})
