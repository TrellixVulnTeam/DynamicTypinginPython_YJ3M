  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function

  22 import collections
  23 import tensorflow as tf
  24 nest = tf.contrib.framework.nest
  25 
  26 from fivo import nested_utils
  27 
  28 # An example namedtuple for use in the following tests.
  29 ExampleTuple = collections.namedtuple('ExampleTuple', ['a', 'b'])
  30 
  31 
  32 class NestedUtilsTest(tf.test.TestCase):
  33 
  34   def test_map_nested_works_on_nested_structures(self):
  35     """Check that map_nested works with nested structures."""
  36     original = [1, (2, 3.2, (4., ExampleTuple(5, 6)))]
  37     expected = [2, (3, 4.2, (5., ExampleTuple(6, 7)))]
  38     out = nested_utils.map_nested(lambda x: x+1, original)
  39     self.assertEqual(expected, out)
  40 
  41   def test_map_nested_works_on_single_objects(self):
  42     """Check that map_nested works with raw objects."""
  43     original = 1
  44     expected = 2
  45     out = nested_utils.map_nested(lambda x: x+1, original)
  46     self.assertEqual(expected, out)
  47 
  48   def test_map_nested_works_on_flat_lists(self):
  49     """Check that map_nested works with a flat list."""
  50     original = [1, 2, 3]
  51     expected = [2, 3, 4]
  52     out = nested_utils.map_nested(lambda x: x+1, original)
  53     self.assertEqual(expected, out)

  55   def test_tile_tensors(self):
  56     """Checks that tile_tensors correctly tiles tensors of different ranks."""
  57     a = tf.range(20)
  58     b = tf.reshape(a, [2, 10])
  59     c = tf.reshape(a, [2, 2, 5])
  60     a_tiled = tf.tile(a, [3])
  61     b_tiled = tf.tile(b, [3, 1])
  62     c_tiled = tf.tile(c, [3, 1, 1])
  63     tensors = [a, (b, ExampleTuple(c, c))]
  64     expected_tensors = [a_tiled, (b_tiled, ExampleTuple(c_tiled, c_tiled))]
  65     tiled = nested_utils.tile_tensors(tensors, [3])
  66     nest.assert_same_structure(expected_tensors, tiled)
  67     with self.test_session() as sess:
  68       expected, out = sess.run([expected_tensors, tiled])
  69       expected = nest.flatten(expected)
  70       out = nest.flatten(out)
  71       # Check that the tiling is correct.
  72       for x, y in zip(expected, out):
  73         self.assertAllClose(x, y)
  74 
  75   def test_gather_tensors(self):
  76     a = tf.reshape(tf.range(20), [5, 4])
  77     inds = [0, 0, 1, 4]
  78     a_gathered = tf.gather(a, inds)
  79     tensors = [a, (a, ExampleTuple(a, a))]
  80     gt_gathered = [a_gathered, (a_gathered,
  81                                 ExampleTuple(a_gathered, a_gathered))]
  82     gathered = nested_utils.gather_tensors(tensors, inds)
  83     nest.assert_same_structure(gt_gathered, gathered)
  84     with self.test_session() as sess:
  85       gt, out = sess.run([gt_gathered, gathered])
  86       gt = nest.flatten(gt)
  87       out = nest.flatten(out)
  88       # Check that the gathering is correct.
  89       for x, y in zip(gt, out):
  90         self.assertAllClose(x, y)

  92   def test_tas_for_tensors(self):
  93     a = tf.reshape(tf.range(20), [5, 4])
  94     tensors = [a, (a, ExampleTuple(a, a))]
  95     tas = nested_utils.tas_for_tensors(tensors, 5)
  96     nest.assert_same_structure(tensors, tas)
  97     # We can't pass TensorArrays to sess.run so instead we turn then back into
  98     # tensors to check that they were created correctly.
  99     stacked = nested_utils.map_nested(lambda x: x.stack(), tas)
 100     with self.test_session() as sess:
 101       gt, out = sess.run([tensors, stacked])
 102       gt = nest.flatten(gt)
 103       out = nest.flatten(out)
 104       # Check that the tas were created correctly.
 105       for x, y in zip(gt, out):
 106         self.assertAllClose(x, y)

 108   def test_read_tas(self):
 109     a = tf.reshape(tf.range(20), [5, 4])
 110     a_read = a[3, :]
 111     tensors = [a, (a, ExampleTuple(a, a))]
 112     gt_read = [a_read, (a_read, ExampleTuple(a_read, a_read))]
 113     tas = nested_utils.tas_for_tensors(tensors, 5)
 114     tas_read = nested_utils.read_tas(tas, 3)
 115     nest.assert_same_structure(tas, tas_read)
 116     with self.test_session() as sess:
 117       gt, out = sess.run([gt_read, tas_read])
 118       gt = nest.flatten(gt)
 119       out = nest.flatten(out)
 120       # Check that the tas were read correctly.
 121       for x, y in zip(gt, out):
 122         self.assertAllClose(x, y)
