  63   def run_training_one_step(self, bound, dataset_type, data_dimension,
  64                             dataset_filename, dir_prefix, resampling_type,
  65                             model, batch_size=2, num_samples=3,
  66                             create_dataset_and_model_fn=(runners.create_dataset_and_model)):
  67     config = self.default_config()
  68     config.model = model
  69     config.resampling_type = resampling_type
  70     config.relaxed_resampling_temperature = 0.5
  71     config.bound = bound
  72     config.split = "train"
  73     config.dataset_type = dataset_type
  74     config.dataset_path = os.path.join(
  75         os.path.dirname(os.path.realpath(__file__)),
  76         "test_data",
  77         dataset_filename)
  78     config.max_steps = 1
  79     config.batch_size = batch_size
  80     config.num_samples = num_samples
  81     config.latent_size = 4
  82     config.data_dimension = data_dimension
  83     config.logdir = os.path.join(tf.test.get_temp_dir(), "%s-%s-%s-%s" %
  84                                  (dir_prefix, bound, dataset_type, model))
  85     runners.run_train(config,
  86                       create_dataset_and_model_fn=create_dataset_and_model_fn)
  87     return config

  89   def dummmy_dataset_and_model_fn(self, *unused_args, **unused_kwargs):
  90     # We ignore the arguments in the dummy but need to preserve prototype.
  91     batch_elements = 5
  92     sequence_length = 4
  93     data_dimensions = 3
  94     dataset = tf.data.Dataset.from_tensors(
  95         tf.zeros((sequence_length, batch_elements, data_dimensions),
  96                  dtype=tf.float32))
  97     inputs = dataset.make_one_shot_iterator().get_next()
  98     targets = tf.zeros_like(inputs)
  99     lengths = tf.constant([sequence_length] * batch_elements)
 100     mean = tf.constant((0.0, 0.0, 0.0))
 101     model = vrnn.create_vrnn(data_dimensions, 1,
 102                              base.ConditionalNormalDistribution)
 103     return inputs, targets, lengths, model, mean

 196   def run_sampling(self, model):
 197     """Test sampling from the model."""
 198     config = self.run_training_one_step(
 199         "fivo", "pianoroll", 88, "tiny_pianoroll.pkl", "test-sampling", "multinomial",
 200         model)
 201     config.prefix_length = 3
 202     config.sample_length = 6
 203     config.split = "train"
 204     config.sample_out_dir = None
 205 
 206     runners.run_sample(config)
 207     unused_samples = np.load(os.path.join(config.logdir, "samples.npz"))
