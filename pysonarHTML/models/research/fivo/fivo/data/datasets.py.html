  33 def sparse_pianoroll_to_dense(pianoroll, min_note, num_notes):
  34   """Converts a sparse pianoroll to a dense numpy array.
  35 
  36   Given a sparse pianoroll, converts it to a dense numpy array of shape
  37   [num_timesteps, num_notes] where entry i,j is 1.0 if note j is active on
  38   timestep i and 0.0 otherwise.
  39 
  40   Args:
  41     pianoroll: A sparse pianoroll object, a list of tuples where the i'th tuple
  42       contains the indices of the notes active at timestep i.
  43     min_note: The minimum note in the pianoroll, subtracted from all notes so
  44       that the minimum note becomes 0.
  45     num_notes: The number of possible different note indices, determines the
  46       second dimension of the resulting dense array.
  47   Returns:
  48     dense_pianoroll: A [num_timesteps, num_notes] numpy array of floats.
  49     num_timesteps: A python int, the number of timesteps in the pianoroll.
  50   """
  51   num_timesteps = len(pianoroll)
  52   inds = []
  53   for time, chord in enumerate(pianoroll):
  54     # Re-index the notes to start from min_note.
  55     inds.extend((time, note-min_note) for note in chord)
  56   shape = [num_timesteps, num_notes]
  57   values = [1.] * len(inds)
  58   sparse_pianoroll = coo_matrix(
  59       (values, ([x[0] for x in inds], [x[1] for x in inds])),
  60       shape=shape)
  61   return sparse_pianoroll.toarray(), num_timesteps

  64 def create_pianoroll_dataset(path,
  65                              split,
  66                              batch_size,
  67                              num_parallel_calls=DEFAULT_PARALLELISM,
  68                              shuffle=False,
  69                              repeat=False,
  70                              min_note=21,
  71                              max_note=108):
  72   """Creates a pianoroll dataset.
  73 
  74   Args:
  75     path: The path of a pickle file containing the dataset to load.
  76     split: The split to use, can be train, test, or valid.
  77     batch_size: The batch size. If repeat is False then it is not guaranteed
  78       that the true batch size will match for all batches since batch_size
  79       may not necessarily evenly divide the number of elements.
  80     num_parallel_calls: The number of threads to use for parallel processing of
  81       the data.
  82     shuffle: If true, shuffles the order of the dataset.
  83     repeat: If true, repeats the dataset endlessly.
  84     min_note: The minimum note number of the dataset. For all pianoroll datasets
  85       the minimum note is number 21, and changing this affects the dimension of
  86       the data. This is useful mostly for testing.
  87     max_note: The maximum note number of the dataset. For all pianoroll datasets
  88       the maximum note is number 108, and changing this affects the dimension of
  89       the data. This is useful mostly for testing.
  90   Returns:
  91     inputs: A batch of input sequences represented as a dense Tensor of shape
  92       [time, batch_size, data_dimension]. The sequences in inputs are the
  93       sequences in targets shifted one timestep into the future, padded with
  94       zeros. This tensor is mean-centered, with the mean taken from the pickle
  95       file key 'train_mean'.
  96     targets: A batch of target sequences represented as a dense Tensor of
  97       shape [time, batch_size, data_dimension].
  98     lens: An int Tensor of shape [batch_size] representing the lengths of each
  99       sequence in the batch.
 100     mean: A float Tensor of shape [data_dimension] containing the mean loaded
 101       from the pickle file.
 102   """
 103   # Load the data from disk.
 104   num_notes = max_note - min_note + 1
 105   with tf.gfile.Open(path, "r") as f:
 106     raw_data = pickle.load(f)
 107   pianorolls = raw_data[split]
 108   mean = raw_data["train_mean"]
 109   num_examples = len(pianorolls)
 110 
 111   def pianoroll_generator():
 112     for sparse_pianoroll in pianorolls:
 113       yield sparse_pianoroll_to_dense(sparse_pianoroll, min_note, num_notes)
 114 
 115   dataset = tf.data.Dataset.from_generator(
 116       pianoroll_generator,
 117       output_types=(tf.float64, tf.int64),
 118       output_shapes=([None, num_notes], []))
 119 
 120   if repeat: dataset = dataset.repeat()
 121   if shuffle: dataset = dataset.shuffle(num_examples)
 122 
 123   # Batch sequences togther, padding them to a common length in time.
 124   dataset = dataset.padded_batch(batch_size,
 125                                  padded_shapes=([None, num_notes], []))
 126 
 127   def process_pianoroll_batch(data, lengths):
 128     """Create mean-centered and time-major next-step prediction Tensors."""
 129     data = tf.to_float(tf.transpose(data, perm=[1, 0, 2]))
 130     lengths = tf.to_int32(lengths)
 131     targets = data
 132     # Mean center the inputs.
 133     inputs = data - tf.constant(mean, dtype=tf.float32,
 134                                 shape=[1, 1, mean.shape[0]])
 135     # Shift the inputs one step forward in time. Also remove the last timestep
 136     # so that targets and inputs are the same length.
 137     inputs = tf.pad(inputs, [[1, 0], [0, 0], [0, 0]], mode="CONSTANT")[:-1]
 138     # Mask out unused timesteps.
 139     inputs *= tf.expand_dims(tf.transpose(
 140         tf.sequence_mask(lengths, dtype=inputs.dtype)), 2)
 141     return inputs, targets, lengths
 142 
 143   dataset = dataset.map(process_pianoroll_batch,
 144                         num_parallel_calls=num_parallel_calls)
 145   dataset = dataset.prefetch(num_examples)
 146 
 147   itr = dataset.make_one_shot_iterator()
 148   inputs, targets, lengths = itr.get_next()
 149   return inputs, targets, lengths, tf.constant(mean, dtype=tf.float32)

 152 def create_human_pose_dataset(
 153     path,
 154     split,
 155     batch_size,
 156     num_parallel_calls=DEFAULT_PARALLELISM,
 157     shuffle=False,
 158     repeat=False,):
 159   """Creates a human pose dataset.
 160 
 161   Args:
 162     path: The path of a pickle file containing the dataset to load.
 163     split: The split to use, can be train, test, or valid.
 164     batch_size: The batch size. If repeat is False then it is not guaranteed
 165       that the true batch size will match for all batches since batch_size
 166       may not necessarily evenly divide the number of elements.
 167     num_parallel_calls: The number of threads to use for parallel processing of
 168       the data.
 169     shuffle: If true, shuffles the order of the dataset.
 170     repeat: If true, repeats the dataset endlessly.
 171   Returns:
 172     inputs: A batch of input sequences represented as a dense Tensor of shape
 173       [time, batch_size, data_dimension]. The sequences in inputs are the
 174       sequences in targets shifted one timestep into the future, padded with
 175       zeros. This tensor is mean-centered, with the mean taken from the pickle
 176       file key 'train_mean'.
 177     targets: A batch of target sequences represented as a dense Tensor of
 178       shape [time, batch_size, data_dimension].
 179     lens: An int Tensor of shape [batch_size] representing the lengths of each
 180       sequence in the batch.
 181     mean: A float Tensor of shape [data_dimension] containing the mean loaded
 182     from the pickle file.
 183   """
 184   # Load the data from disk.
 185   with tf.gfile.Open(path, "r") as f:
 186     raw_data = pickle.load(f)
 187 
 188   mean = raw_data["train_mean"]
 189   pose_sequences = raw_data[split]
 190   num_examples = len(pose_sequences)
 191   num_features = pose_sequences[0].shape[1]
 192 
 193   def pose_generator():
 194     """A generator that yields pose data sequences."""
 195     # Each timestep has 32 x values followed by 32 y values so is 64
 196     # dimensional.
 197     for pose_sequence in pose_sequences:
 198       yield pose_sequence, pose_sequence.shape[0]
 199 
 200   dataset = tf.data.Dataset.from_generator(
 201       pose_generator,
 202       output_types=(tf.float64, tf.int64),
 203       output_shapes=([None, num_features], []))
 204 
 205   if repeat:
 206     dataset = dataset.repeat()
 207   if shuffle:
 208     dataset = dataset.shuffle(num_examples)
 209 
 210   # Batch sequences togther, padding them to a common length in time.
 211   dataset = dataset.padded_batch(
 212       batch_size, padded_shapes=([None, num_features], []))
 213 
 214   # Post-process each batch, ensuring that it is mean-centered and time-major.
 215   def process_pose_data(data, lengths):
 216     """Creates Tensors for next step prediction and mean-centers the input."""
 217     data = tf.to_float(tf.transpose(data, perm=[1, 0, 2]))
 218     lengths = tf.to_int32(lengths)
 219     targets = data
 220     # Mean center the inputs.
 221     inputs = data - tf.constant(
 222         mean, dtype=tf.float32, shape=[1, 1, mean.shape[0]])
 223     # Shift the inputs one step forward in time. Also remove the last timestep
 224     # so that targets and inputs are the same length.
 225     inputs = tf.pad(inputs, [[1, 0], [0, 0], [0, 0]], mode="CONSTANT")[:-1]
 226     # Mask out unused timesteps.
 227     inputs *= tf.expand_dims(
 228         tf.transpose(tf.sequence_mask(lengths, dtype=inputs.dtype)), 2)
 229     return inputs, targets, lengths
 230 
 231   dataset = dataset.map(
 232       process_pose_data,
 233       num_parallel_calls=num_parallel_calls)
 234   dataset = dataset.prefetch(num_examples)
 235 
 236   itr = dataset.make_one_shot_iterator()
 237   inputs, targets, lengths = itr.get_next()
 238   return inputs, targets, lengths, tf.constant(mean, dtype=tf.float32)

 241 def create_speech_dataset(path,
 242                           batch_size,
 243                           samples_per_timestep=200,
 244                           num_parallel_calls=DEFAULT_PARALLELISM,
 245                           prefetch_buffer_size=2048,
 246                           shuffle=False,
 247                           repeat=False):
 248   """Creates a speech dataset.
 249 
 250   Args:
 251     path: The path of a possibly sharded TFRecord file containing the data.
 252     batch_size: The batch size. If repeat is False then it is not guaranteed
 253       that the true batch size will match for all batches since batch_size
 254       may not necessarily evenly divide the number of elements.
 255     samples_per_timestep: The number of audio samples per timestep. Used to
 256       reshape the data into sequences of shape [time, samples_per_timestep].
 257       Should not change except for testing -- in all speech datasets 200 is the
 258       number of samples per timestep.
 259     num_parallel_calls: The number of threads to use for parallel processing of
 260       the data.
 261     prefetch_buffer_size: The size of the prefetch queues to use after reading
 262       and processing the raw data.
 263     shuffle: If true, shuffles the order of the dataset.
 264     repeat: If true, repeats the dataset endlessly.
 265   Returns:
 266     inputs: A batch of input sequences represented as a dense Tensor of shape
 267       [time, batch_size, samples_per_timestep]. The sequences in inputs are the
 268       sequences in targets shifted one timestep into the future, padded with
 269       zeros.
 270     targets: A batch of target sequences represented as a dense Tensor of
 271       shape [time, batch_size, samples_per_timestep].
 272     lens: An int Tensor of shape [batch_size] representing the lengths of each
 273       sequence in the batch.
 274   """
 275   filenames = [path]
 276 
 277   def read_speech_example(value):
 278     """Parses a single tf.Example from the TFRecord file."""
 279     decoded = tf.decode_raw(value, out_type=tf.float32)
 280     example = tf.reshape(decoded, [-1, samples_per_timestep])
 281     length = tf.shape(example)[0]
 282     return example, length
 283 
 284   # Create the dataset from the TFRecord files
 285   dataset = tf.data.TFRecordDataset(filenames).map(
 286       read_speech_example, num_parallel_calls=num_parallel_calls)
 287   dataset = dataset.prefetch(prefetch_buffer_size)
 288 
 289   if repeat: dataset = dataset.repeat()
 290   if shuffle: dataset = dataset.shuffle(prefetch_buffer_size)
 291 
 292   dataset = dataset.padded_batch(
 293       batch_size, padded_shapes=([None, samples_per_timestep], []))
 294 
 295   def process_speech_batch(data, lengths):
 296     """Creates Tensors for next step prediction."""
 297     data = tf.transpose(data, perm=[1, 0, 2])
 298     lengths = tf.to_int32(lengths)
 299     targets = data
 300     # Shift the inputs one step forward in time. Also remove the last timestep
 301     # so that targets and inputs are the same length.
 302     inputs = tf.pad(data, [[1, 0], [0, 0], [0, 0]], mode="CONSTANT")[:-1]
 303     # Mask out unused timesteps.
 304     inputs *= tf.expand_dims(
 305         tf.transpose(tf.sequence_mask(lengths, dtype=inputs.dtype)), 2)
 306     return inputs, targets, lengths
 307 
 308   dataset = dataset.map(process_speech_batch,
 309                         num_parallel_calls=num_parallel_calls)
 310   dataset = dataset.prefetch(prefetch_buffer_size)
 311 
 312   itr = dataset.make_one_shot_iterator()
 313   inputs, targets, lengths = itr.get_next()
 314   return inputs, targets, lengths

 327 def create_chain_graph_dataset(
 328     batch_size,
 329     num_timesteps,
 330     steps_per_observation=None,
 331     state_size=1,
 332     transition_variance=1.,
 333     observation_variance=1.,
 334     transition_type=STANDARD_TRANSITION,
 335     observation_type=STANDARD_OBSERVATION,
 336     fixed_observation=None,
 337     prefetch_buffer_size=2048,
 338     dtype="float32"):
 339   """Creates a toy chain graph dataset.
 340 
 341   Creates a dataset where the data are sampled from a diffusion process. The
 342   'latent' states of the process are sampled as a chain of Normals:
 343 
 344   z0 ~ N(0, transition_variance)
 345   z1 ~ N(transition_fn(z0), transition_variance)
 346   ...
 347 
 348   where transition_fn could be round z0 or pass it through unchanged.
 349 
 350   The observations are produced every steps_per_observation timesteps as a
 351   function of the latent zs. For example if steps_per_observation is 3 then the
 352   first observation will be produced as a function of z3:
 353 
 354   x1 ~ N(observation_fn(z3), observation_variance)
 355 
 356   where observation_fn could square z3, take the absolute value, or pass
 357   it through unchanged.
 358 
 359   Only the observations are returned.
 360 
 361   Args:
 362     batch_size: The batch size. The number of trajectories to run in parallel.
 363     num_timesteps: The length of the chain of latent states (i.e. the
 364       number of z's excluding z0.
 365     steps_per_observation: The number of latent states between each observation,
 366       must evenly divide num_timesteps.
 367     state_size: The size of the latent state and observation, must be a
 368       python int.
 369     transition_variance: The variance of the transition density.
 370     observation_variance: The variance of the observation density.
 371     transition_type: Must be one of "round" or "standard". "round" means that
 372       the transition density is centered at the rounded previous latent state.
 373       "standard" centers the transition density at the previous latent state,
 374       unchanged.
 375     observation_type: Must be one of "squared", "abs" or "standard". "squared"
 376       centers the observation density at the squared latent state. "abs"
 377       centers the observaiton density at the absolute value of the current
 378       latent state. "standard" centers the observation density at the current
 379       latent state.
 380     fixed_observation: If not None, fixes all observations to be a constant.
 381       Must be a scalar.
 382     prefetch_buffer_size: The size of the prefetch queues to use after reading
 383       and processing the raw data.
 384     dtype: A string convertible to a tensorflow datatype. The datatype used
 385       to represent the states and observations.
 386   Returns:
 387     observations: A batch of observations represented as a dense Tensor of
 388       shape [num_observations, batch_size, state_size]. num_observations is
 389       num_timesteps/steps_per_observation.
 390     lens: An int Tensor of shape [batch_size] representing the lengths of each
 391       sequence in the batch. Will contain num_observations as each entry.
 392   Raises:
 393     ValueError: Raised if steps_per_observation does not evenly divide
 394       num_timesteps.
 395   """
 396   if steps_per_observation is None:
 397     steps_per_observation = num_timesteps
 398   if num_timesteps % steps_per_observation != 0:
 399     raise ValueError("steps_per_observation must evenly divide num_timesteps.")
 400   num_observations = int(num_timesteps / steps_per_observation)
 401   def data_generator():
 402     """An infinite generator of latents and observations from the model."""
 403     transition_std = np.sqrt(transition_variance)
 404     observation_std = np.sqrt(observation_variance)
 405     while True:
 406       states = []
 407       observations = []
 408       # Sample z0 ~ Normal(0, sqrt(variance)).
 409       states.append(
 410           np.random.normal(size=[state_size],
 411                            scale=observation_std).astype(dtype))
 412       # Start the range at 1 because we've already generated z0.
 413       # The range ends at num_timesteps+1 because we want to include the
 414       # num_timesteps-th step.
 415       for t in xrange(1, num_timesteps+1):
 416         if transition_type == ROUND_TRANSITION:
 417           loc = np.round(states[-1])
 418         elif transition_type == STANDARD_TRANSITION:
 419           loc = states[-1]
 420         z_t = np.random.normal(size=[state_size], loc=loc, scale=transition_std)
 421         states.append(z_t.astype(dtype))
 422         if t % steps_per_observation == 0:
 423           if fixed_observation is None:
 424             if observation_type == SQUARED_OBSERVATION:
 425               loc = np.square(states[-1])
 426             elif observation_type == ABS_OBSERVATION:
 427               loc = np.abs(states[-1])
 428             elif observation_type == STANDARD_OBSERVATION:
 429               loc = states[-1]
 430             x_t = np.random.normal(size=[state_size],
 431                                    loc=loc,
 432                                    scale=observation_std).astype(dtype)
 433           else:
 434             x_t = np.ones([state_size]) * fixed_observation
 435 
 436           observations.append(x_t)
 437       yield states, observations
 438 
 439   dataset = tf.data.Dataset.from_generator(
 440       data_generator,
 441       output_types=(tf.as_dtype(dtype), tf.as_dtype(dtype)),
 442       output_shapes=([num_timesteps+1, state_size],
 443                      [num_observations, state_size])
 444   )
 445   dataset = dataset.repeat().batch(batch_size)
 446   dataset = dataset.prefetch(prefetch_buffer_size)
 447   itr = dataset.make_one_shot_iterator()
 448   _, observations = itr.get_next()
 449   # Transpose observations from [batch, time, state_size] to
 450   # [time, batch, state_size].
 451   observations = tf.transpose(observations, perm=[1, 0, 2])
 452   lengths = tf.ones([batch_size], dtype=tf.int32) * num_observations
 453   return observations, lengths
