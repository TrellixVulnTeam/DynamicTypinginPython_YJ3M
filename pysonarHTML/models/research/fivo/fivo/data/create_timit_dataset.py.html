  43 NUM_TRAIN_FILES = 4620
  44 NUM_TEST_FILES = 1680

  47 # Regexes for reading SPHERE header files.
  48 SAMPLE_COUNT_REGEX = re.compile(r"sample_count -i (\d+)")
  49 SAMPLE_MIN_REGEX = re.compile(r"sample_min -i (-?\d+)")
  50 SAMPLE_MAX_REGEX = re.compile(r"sample_max -i (-?\d+)")
  51

  53 def get_filenames(split):
  54   """Get all wav filenames from the TIMIT archive."""
  55   path = os.path.join(FLAGS.raw_timit_dir, "TIMIT", split, "*", "*", "*.WAV")
  56   # Sort the output by name so the order is deterministic.
  57   files = sorted(glob.glob(path))
  58   return files

  61 def load_timit_wav(filename):
  62   """Loads a TIMIT wavfile into a numpy array.
  63 
  64   TIMIT wavfiles include a SPHERE header, detailed in the TIMIT docs. The first
  65   line is the header type and the second is the length of the header in bytes.
  66   After the header, the remaining bytes are actual WAV data.
  67 
  68   The header includes information about the WAV data such as the number of
  69   samples and minimum and maximum amplitude. This function asserts that the
  70   loaded wav data matches the header.
  71 
  72   Args:
  73     filename: The name of the TIMIT wavfile to load.
  74   Returns:
  75     wav: A numpy array containing the loaded wav data.
  76   """
  77   wav_file = open(filename, "rb")
  78   header_type = wav_file.readline()
  79   header_length_str = wav_file.readline()
  80   # The header length includes the length of the first two lines.
  81   header_remaining_bytes = (int(header_length_str) - len(header_type) -
  82                             len(header_length_str))
  83   header = wav_file.read(header_remaining_bytes)
  84   # Read the relevant header fields.
  85   sample_count = int(SAMPLE_COUNT_REGEX.search(header).group(1))
  86   sample_min = int(SAMPLE_MIN_REGEX.search(header).group(1))
  87   sample_max = int(SAMPLE_MAX_REGEX.search(header).group(1))
  88   wav = np.fromstring(wav_file.read(), dtype="int16").astype("float32")
  89   # Check that the loaded data conforms to the header description.
  90   assert len(wav) == sample_count
  91   assert wav.min() == sample_min
  92   assert wav.max() == sample_max
  93   return wav

  96 def preprocess(wavs, block_size, mean, std):
  97   """Normalize the wav data and reshape it into chunks."""
  98   processed_wavs = []
  99   for wav in wavs:
 100     wav = (wav - mean) / std
 101     wav_length = wav.shape[0]
 102     if wav_length % block_size != 0:
 103       pad_width = block_size - (wav_length % block_size)
 104       wav = np.pad(wav, (0, pad_width), "constant")
 105     assert wav.shape[0] % block_size == 0
 106     wav = wav.reshape((-1, block_size))
 107     processed_wavs.append(wav)
 108   return processed_wavs

 111 def create_tfrecord_from_wavs(wavs, output_file):
 112   """Writes processed wav files to disk as sharded TFRecord files."""
 113   with tf.python_io.TFRecordWriter(output_file) as builder:
 114     for wav in wavs:
 115       builder.write(wav.astype(np.float32).tobytes())

 118 def main(unused_argv):
 119   train_filenames = get_filenames("TRAIN")
 120   test_filenames = get_filenames("TEST")
 121 
 122   num_train_files = len(train_filenames)
 123   num_test_files = len(test_filenames)
 124   num_valid_files = int(num_train_files * FLAGS.valid_frac)
 125   num_train_files -= num_valid_files
 126 
 127   print("%d train / %d valid / %d test" % (
 128       num_train_files, num_valid_files, num_test_files))
 129 
 130   random.seed(1234)
 131   random.shuffle(train_filenames)
 132 
 133   valid_filenames = train_filenames[:num_valid_files]
 134   train_filenames = train_filenames[num_valid_files:]
 135 
 136   # Make sure there is no overlap in the train, test, and valid sets.
 137   train_s = set(train_filenames)
 138   test_s = set(test_filenames)
 139   valid_s = set(valid_filenames)
 140   # Disable explicit length testing to make the assertions more readable.
 141   # pylint: disable=g-explicit-length-test
 142   assert len(train_s & test_s) == 0
 143   assert len(train_s & valid_s) == 0
 144   assert len(valid_s & test_s) == 0
 145   # pylint: enable=g-explicit-length-test
 146 
 147   train_wavs = [load_timit_wav(f) for f in train_filenames]
 148   valid_wavs = [load_timit_wav(f) for f in valid_filenames]
 149   test_wavs = [load_timit_wav(f) for f in test_filenames]
 150   assert len(train_wavs) + len(valid_wavs) == NUM_TRAIN_FILES
 151   assert len(test_wavs) == NUM_TEST_FILES
 152 
 153   # Calculate the mean and standard deviation of the train set.
 154   train_stacked = np.hstack(train_wavs)
 155   train_mean = np.mean(train_stacked)
 156   train_std = np.std(train_stacked)
 157   print("train mean: %f  train std: %f" % (train_mean, train_std))
 158 
 159   # Process all data, normalizing with the train set statistics.
 160   processed_train_wavs = preprocess(train_wavs, SAMPLES_PER_TIMESTEP,
 161                                     train_mean, train_std)
 162   processed_valid_wavs = preprocess(valid_wavs, SAMPLES_PER_TIMESTEP,
 163                                     train_mean, train_std)
 164   processed_test_wavs = preprocess(test_wavs, SAMPLES_PER_TIMESTEP, train_mean,
 165                                    train_std)
 166 
 167   # Write the datasets to disk.
 168   create_tfrecord_from_wavs(
 169       processed_train_wavs,
 170       os.path.join(FLAGS.out_dir, "train"))
 171   create_tfrecord_from_wavs(
 172       processed_valid_wavs,
 173       os.path.join(FLAGS.out_dir, "valid"))
 174   create_tfrecord_from_wavs(
 175       processed_test_wavs,
 176       os.path.join(FLAGS.out_dir, "test"))
 177 
 178 
 179 if __name__ == "__main__":
 180   tf.app.run()
