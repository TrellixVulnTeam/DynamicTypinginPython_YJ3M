  29 def create_vrnn(generative_class=base.ConditionalNormalDistribution,
  30                 batch_size=2, data_size=3, rnn_hidden_size=4,
  31                 latent_size=5, fcnet_hidden_size=7, encoded_data_size=9,
  32                 encoded_latent_size=11, num_timesteps=7, data_lengths=(7, 4),
  33                 use_tilt=False, random_seed=None):
  34   """Creates a VRNN and some dummy data to feed it for testing purposes.
  35 
  36   Args:
  37     generative_class: The class of the generative distribution.
  38     batch_size: The number of elements per batch.
  39     data_size: The dimension of the vectors that make up the data sequences.
  40     rnn_hidden_size: The hidden state dimension of the RNN that forms the
  41       deterministic part of this VRNN.
  42     latent_size: The size of the stochastic latent state of the VRNN.
  43     fcnet_hidden_size: The size of the hidden layer of the fully connected
  44       networks that parameterize the conditional probability distributions
  45       of the VRNN.
  46     encoded_data_size: The size of the output of the data encoding network.
  47     encoded_latent_size: The size of the output of the latent state encoding
  48       network.
  49     num_timesteps: The maximum number of timesteps in the data.
  50     data_lengths: A tuple of size batch_size that contains the desired lengths
  51       of each sequence in the dummy data.
  52     use_tilt: Use a tilting function.
  53     random_seed: A random seed to feed the VRNN, mainly useful for testing
  54       purposes.
  55 
  56   Returns:
  57     model: A VRNN object.
  58     inputs: A Tensor of shape [num_timesteps, batch_size, data_size], the inputs
  59       to the model, also known as the observations.
  60     targets: A Tensor of shape [num_timesteps, batch_size, data_size], the
  61       desired outputs of the model.
  62     lengths: A Tensor of shape [batch_size], the lengths of the sequences in the
  63       batch.
  64   """
  65 
  66   fcnet_hidden_sizes = [fcnet_hidden_size]
  67   initializers = {"w": tf.contrib.layers.xavier_initializer(seed=random_seed),
  68                   "b": tf.zeros_initializer()}
  69   model = vrnn.create_vrnn(
  70       data_size,
  71       latent_size,
  72       generative_class,
  73       rnn_hidden_size=rnn_hidden_size,
  74       fcnet_hidden_sizes=fcnet_hidden_sizes,
  75       encoded_data_size=encoded_data_size,
  76       encoded_latent_size=encoded_latent_size,
  77       use_tilt=use_tilt,
  78       initializers=initializers,
  79       random_seed=random_seed)
  80   inputs = tf.random_uniform([num_timesteps, batch_size, data_size],
  81                              seed=random_seed, dtype=tf.float32)
  82   targets = tf.random_uniform([num_timesteps, batch_size, data_size],
  83                               seed=random_seed, dtype=tf.float32)
  84   lengths = tf.constant(data_lengths, dtype=tf.int32)
  85   return model, inputs, targets, lengths
