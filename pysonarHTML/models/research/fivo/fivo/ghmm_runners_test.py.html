  68   def eval_ghmm_notraining(self, bound, proposal_type, expected_bound_avg):
  69     config = self.default_config()
  70     config.proposal_type = proposal_type
  71     config.bound = bound
  72     config.logdir = os.path.join(
  73         tf.test.get_temp_dir(), "test-ghmm-%s-%s" % (proposal_type, bound))
  74 
  75     ghmm_runners.run_eval(config)
  76 
  77     data = np.load(os.path.join(config.logdir, "out.npz")).item()
  78     self.assertAlmostEqual(expected_bound_avg, data["mean"], places=3)
