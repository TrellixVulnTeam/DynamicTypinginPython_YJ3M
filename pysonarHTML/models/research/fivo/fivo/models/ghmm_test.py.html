  31   def test_transition_no_weights(self):
  32     with self.test_session() as sess:
  33       ghmm = GaussianHMM(3,
  34                          transition_variances=[1., 2., 3.])
  35       prev_z = tf.constant([1., 2.], dtype=tf.float32)
  36       z0 = ghmm.transition(0, prev_z)
  37       z1 = ghmm.transition(1, prev_z)
  38       z2 = ghmm.transition(2, prev_z)
  39       outs = sess.run([z0.mean(), z0.variance(),
  40                        z1.mean(), z1.variance(),
  41                        z2.mean(), z2.variance()])
  42       self.assertAllClose(outs, [[0., 0.], [1., 1.],
  43                                  [1., 2.], [2., 2.],
  44                                  [1., 2.], [3., 3.]])

 154   def test_smoothing_with_weights(self):
 155     with self.test_session() as sess:
 156       x_weight = np.array([4, 5, 6, 7], dtype=np.float32)
 157       sigma_x = np.array([5, 6, 7, 8], dtype=np.float32)
 158       z_weight = np.array([1, 2, 3], dtype=np.float32)
 159       sigma_z = np.array([1, 2, 3, 4], dtype=np.float32)
 160       z_prev = np.array([1, 2], dtype=np.float32)
 161       batch_size = 2
 162       xs = np.array([[1, 2], [3, 4], [5, 6], [7, 8]], dtype=np.float32)
 163 
 164       z_cov, x_cov, z_x_cov = self._compute_covariance_matrices(
 165           x_weight, z_weight, sigma_x, sigma_z)
 166 
 167       expected_outs = []
 168       # Compute mean and variance for z_0 when we don't condition
 169       # on previous zs.
 170       sigma_12 = z_x_cov[0, :]
 171       sigma_12_22 = np.dot(sigma_12, np.linalg.inv(x_cov))
 172       mean = np.dot(sigma_12_22, xs)
 173       variance = np.squeeze(z_cov[0, 0] - np.dot(sigma_12_22, sigma_12))
 174       expected_outs.append([mean, np.tile(variance, [batch_size])])
 175 
 176       # Compute mean and variance for remaining z_ts.
 177       for t in xrange(1, 4):
 178         sigma_12 = np.concatenate([[z_cov[t, t - 1]], z_x_cov[t, t:]])
 179         sigma_22 = np.vstack((
 180             np.hstack((z_cov[t-1, t-1], z_x_cov[t-1, t:])),
 181             np.hstack((np.transpose([z_x_cov[t-1, t:]]), x_cov[t:, t:]))
 182         ))
 183         sigma_12_22 = np.dot(sigma_12, np.linalg.inv(sigma_22))
 184         mean = np.dot(sigma_12_22, np.vstack((z_prev, xs[t:])))
 185         variance = np.squeeze(z_cov[t, t] - np.dot(sigma_12_22, sigma_12))
 186         expected_outs.append([mean, np.tile(variance, [batch_size])])
 187 
 188       ghmm = GaussianHMM(4,
 189                          transition_variances=sigma_z,
 190                          emission_variances=sigma_x,
 191                          transition_weights=z_weight,
 192                          emission_weights=x_weight)
 193       out_dists = [ghmm.smoothing(t, z_prev, xs[t:]) for t in range(0, 4)]
 194       outs = [[d.mean(), d.variance()] for d in out_dists]
 195       run_outs = sess.run(outs)
 196       self.assertAllClose(expected_outs, run_outs)

 198   def test_covariance_matrices(self):
 199     with self.test_session() as sess:
 200       x_weight = np.array([4, 5, 6, 7], dtype=np.float32)
 201       sigma_x = np.array([5, 6, 7, 8], dtype=np.float32)
 202       z_weight = np.array([1, 2, 3], dtype=np.float32)
 203       sigma_z = np.array([1, 2, 3, 4], dtype=np.float32)
 204 
 205       z_cov, x_cov, z_x_cov = self._compute_covariance_matrices(
 206           x_weight, z_weight, sigma_x, sigma_z)
 207 
 208       ghmm = GaussianHMM(4,
 209                          transition_variances=sigma_z,
 210                          emission_variances=sigma_x,
 211                          transition_weights=z_weight,
 212                          emission_weights=x_weight)
 213       self.assertAllClose(z_cov, sess.run(ghmm.sigma_z))
 214       self.assertAllClose(x_cov, sess.run(ghmm.sigma_x))
 215       self.assertAllClose(z_x_cov, sess.run(ghmm.sigma_zx))

 217   def _compute_covariance_matrices(self, x_weight, z_weight, sigma_x, sigma_z):
 218     # Create z covariance matrix from the definitions.
 219     z_cov = np.zeros([4, 4])
 220     z_cov[0, 0] = sigma_z[0]
 221     for i in range(1, 4):
 222       z_cov[i, i] = (z_cov[i - 1, i - 1] * np.square(z_weight[i - 1]) +
 223                      sigma_z[i])
 224     for i in range(4):
 225       for j in range(4):
 226         if i == j: continue
 227         min_ind = min(i, j)
 228         max_ind = max(i, j)
 229         weights = np.prod(z_weight[min_ind:max_ind])
 230         z_cov[i, j] = z_cov[min_ind, min_ind] * weights
 231     # Compute the x covariance matrix and the z-x covariance matrix.
 232     x_weights_outer = np.outer(x_weight, x_weight)
 233     x_cov = x_weights_outer * z_cov + np.diag(sigma_x)
 234     z_x_cov = x_weight * z_cov
 235     return z_cov, x_cov, z_x_cov

 237   def test_lookahead(self):
 238     x_weight = np.array([4, 5, 6, 7], dtype=np.float32)
 239     sigma_x = np.array([5, 6, 7, 8], dtype=np.float32)
 240     z_weight = np.array([1, 2, 3], dtype=np.float32)
 241     sigma_z = np.array([1, 2, 3, 4], dtype=np.float32)
 242     z_prev = np.array([1, 2], dtype=np.float32)
 243 
 244     with self.test_session() as sess:
 245       z_cov, x_cov, z_x_cov = self._compute_covariance_matrices(
 246           x_weight, z_weight, sigma_x, sigma_z)
 247 
 248       expected_outs = []
 249       for t in range(1, 4):
 250         sigma_12 = z_x_cov[t-1, t:]
 251         z_var = z_cov[t-1, t-1]
 252         mean = np.outer(z_prev, sigma_12/z_var)
 253         variance = x_cov[t:, t:] - np.outer(sigma_12, sigma_12)/ z_var
 254         expected_outs.append([mean, variance])
 255 
 256       ghmm = GaussianHMM(4,
 257                          transition_variances=sigma_z,
 258                          emission_variances=sigma_x,
 259                          transition_weights=z_weight,
 260                          emission_weights=x_weight)
 261       out_dists = [ghmm.lookahead(t, z_prev) for t in range(1, 4)]
 262       outs = [[d.mean(), d.covariance()] for d in out_dists]
 263       run_outs = sess.run(outs)
 264       self.assertAllClose(expected_outs, run_outs)

 269   def test_filtering_proposal(self):
 270     """Check that stashing the xs doesn't change the filtering distributions."""
 271     with self.test_session() as sess:
 272       ghmm = TrainableGaussianHMM(
 273           3, "filtering",
 274           transition_variances=[1., 2., 3.],
 275           emission_variances=[4., 5., 6.],
 276           transition_weights=[7., 8.],
 277           emission_weights=[9., 10., 11])
 278       observations = tf.constant([[3., 4.],
 279                                   [3., 4.],
 280                                   [3., 4.]], dtype=tf.float32)
 281       ghmm.set_observations(observations, [3, 3])
 282       z_prev = tf.constant([1., 2.], dtype=tf.float32)
 283 
 284       proposals = [ghmm._filtering_proposal(t, z_prev) for t in range(3)]
 285       dist_params = [[p.mean(), p.variance()] for p in proposals]
 286 
 287       expected_outs = [[[27./85., 36./85.], [4./85., 4./85.]],
 288                        [[95./205., 150./205.], [10./205., 10./205.]],
 289                        [[147./369., 228./369.], [18./369., 18./369.]]]
 290       self.assertAllClose(expected_outs, sess.run(dist_params))
