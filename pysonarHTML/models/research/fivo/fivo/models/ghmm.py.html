  30 class GaussianHMM(object):
  31   """A hidden markov model with 1-D Gaussian latent space and observations.
  32 
  33   This is a hidden markov model where the state and observations are
  34   one-dimensional Gaussians. The mean of each latent state is a linear
  35   function of the previous latent state, and the mean of each observation
  36   is a linear function of the current latent state.
  37 
  38   The description that follows is 0-indexed instead of 1-indexed to make
  39   it easier to reason about the parameters passed to the model.
  40 
  41   The parameters of the model are:
  42     T: The number timesteps, latent states, and observations.
  43     vz_t, t=0 to T-1: The variance of the latent state at timestep t.
  44     vx_t, t=0 to T-1: The variance of the observation at timestep t.
  45     wz_t, t=1 to T-1: The weight that defines the latent transition at t.
  46     wx_t, t=0 to T-1: The weight that defines the observation function at t.
  47 
  48   There are T vz_t, vx_t, and wx_t but only T-1 wz_t because there are only
  49   T-1 transitions in the model.
  50 
  51   Given these parameters, sampling from the model is defined as
  52 
  53     z_0 ~ N(0, vz_0)
  54     x_0 | z_0 ~ N(wx_0 * z_0, vx_0)
  55     z_1 | z_0 ~ N(wz_1 * z_0, vz_1)
  56     x_1 | z_1 ~ N(wx_1 * z_1, vx_1)
  57     ...
  58     z_{T-1} | z_{T-2} ~ N(wz_{T-1} * z_{T-2}, vz_{T-1})
  59     x_{T-1} | z_{T-1} ~ N(wx_{T-1} * z_{T-1}, vx_{T-1}).
  60   """
  61 
  62   def __init__(self,
  63                num_timesteps,
  64                transition_variances=1.,
  65                emission_variances=1.,
  66                transition_weights=1.,
  67                emission_weights=1.,
  68                dtype=tf.float32):
  69     """Creates a gaussian hidden markov model.
  70 
  71     Args:
  72       num_timesteps: A python int, the number of timesteps in the model.
  73       transition_variances: The variance of p(z_t | z_t-1). Can be a scalar,
  74         setting all variances to be the same, or a Tensor of shape
  75         [num_timesteps].
  76       emission_variances: The variance of p(x_t | z_t). Can be a scalar,
  77         setting all variances to be the same, or a Tensor of shape
  78         [num_timesteps].
  79       transition_weights: The weight that defines the linear function that
  80         produces the mean of z_t given z_{t-1}. Can be a scalar, setting
  81         all weights to be the same, or a Tensor of shape [num_timesteps-1].
  82       emission_weights: The weight that defines the linear function that
  83         produces the mean of x_t given z_t. Can be a scalar, setting
  84         all weights to be the same, or a Tensor of shape [num_timesteps].
  85       dtype: The datatype of the state.
  86     """
  87     self.num_timesteps = num_timesteps
  88     self.dtype = dtype
  89 
  90     def _expand_param(param, size):
  91       param = tf.convert_to_tensor(param, dtype=self.dtype)
  92       if not param.get_shape().as_list():
  93         param = tf.tile(param[tf.newaxis], [size])
  94 
  95       return param
  96 
  97     def _ta_for_param(param):
  98       size = tf.shape(param)[0]
  99       ta = tf.TensorArray(dtype=param.dtype,
 100                           size=size,
 101                           dynamic_size=False,
 102                           clear_after_read=False).unstack(param)
 103       return ta
 104 
 105     self.transition_variances = _ta_for_param(
 106         _expand_param(transition_variances, num_timesteps))
 107     self.transition_weights = _ta_for_param(
 108         _expand_param(transition_weights, num_timesteps-1))
 109     em_var = _expand_param(emission_variances, num_timesteps)
 110     self.emission_variances = _ta_for_param(em_var)
 111     em_w = _expand_param(emission_weights, num_timesteps)
 112     self.emission_weights = _ta_for_param(em_w)
 113     self._compute_covariances(em_w, em_var)
 114 
 115   def _compute_covariances(self, emission_weights, emission_variances):
 116     """Compute all covariance matrices.
 117 
 118     Computes the covaraince matrix for the latent variables, the observations,
 119     and the covariance between the latents and observations.
 120 
 121     Args:
 122       emission_weights: A Tensor of shape [num_timesteps] containing
 123         the emission distribution weights at each timestep.
 124       emission_variances: A Tensor of shape [num_timesteps] containing
 125         the emiision distribution variances at each timestep.
 126     """
 127     # Compute the marginal variance of each latent.
 128     z_variances = [self.transition_variances.read(0)]
 129     for i in range(1, self.num_timesteps):
 130       z_variances.append(
 131           z_variances[i-1] * tf.square(self.transition_weights.read(i-1)) +
 132           self.transition_variances.read(i))
 133     # Compute the latent covariance matrix.
 134     sigma_z = []
 135     for i in range(self.num_timesteps):
 136       sigma_z_row = []
 137       for j in range(self.num_timesteps):
 138         if i == j:
 139           sigma_z_row.append(z_variances[i])
 140           continue
 141         min_ind = min(i, j)
 142         max_ind = max(i, j)
 143         weight = tf.reduce_prod(
 144             self.transition_weights.gather(tf.range(min_ind, max_ind)))
 145         sigma_z_row.append(z_variances[min_ind] * weight)
 146       sigma_z.append(tf.stack(sigma_z_row))
 147     self.sigma_z = tf.stack(sigma_z)
 148     # Compute the observation covariance matrix.
 149     x_weights_outer = tf.einsum("i,j->ij", emission_weights, emission_weights)
 150     self.sigma_x = x_weights_outer * self.sigma_z + tf.diag(emission_variances)
 151     # Compute the latent - observation covariance matrix.
 152     # The first axis will index latents, the second axis will index observtions.
 153     self.sigma_zx = emission_weights[tf.newaxis, :] * self.sigma_z
 154     self.obs_dist = tfd.MultivariateNormalFullCovariance(
 155         loc=tf.zeros([self.num_timesteps], dtype=tf.float32),
 156         covariance_matrix=self.sigma_x)
 157 
 158   def transition(self, t, z_prev):
 159     """Compute the transition distribution p(z_t | z_t-1).
 160 
 161     Args:
 162       t: The current timestep, a scalar integer Tensor. When t=0 z_prev is
 163         mostly ignored and the distribution p(z_0) is returned. z_prev is
 164         'mostly' ignored because it is still used to derive batch_size.
 165       z_prev: A [batch_size] set of states.
 166     Returns:
 167       p(z_t | z_t-1) as a univariate normal distribution.
 168     """
 169     batch_size = tf.shape(z_prev)[0]
 170     scale = tf.sqrt(self.transition_variances.read(t))
 171     scale = tf.tile(scale[tf.newaxis], [batch_size])
 172     loc = tf.cond(tf.greater(t, 0),
 173                   lambda: self.transition_weights.read(t-1)*z_prev,
 174                   lambda: tf.zeros_like(scale))
 175     return tfd.Normal(loc=loc, scale=scale)
 176 
 177   def emission(self, t, z):
 178     """Compute the emission distribution p(x_t | z_t).
 179 
 180     Args:
 181       t: The current timestep, a scalar integer Tensor.
 182       z: A [batch_size] set of the current states.
 183     Returns:
 184       p(x_t | z_t) as a univariate normal distribution.
 185     """
 186     batch_size = tf.shape(z)[0]
 187     scale = tf.sqrt(self.emission_variances.read(t))
 188     scale = tf.tile(scale[tf.newaxis], [batch_size])
 189     loc = self.emission_weights.read(t)*z
 190     return tfd.Normal(loc=loc, scale=scale)
 191 
 192   def filtering(self, t, z_prev, x_cur):
 193     """Computes the filtering distribution p(z_t | z_{t-1}, x_t).
 194 
 195     Args:
 196       t: A python int, the index for z_t. When t is 0, z_prev is ignored,
 197         giving p(z_0 | x_0).
 198       z_prev: z_{t-1}, the previous z to condition on. A Tensor of shape
 199         [batch_size].
 200       x_cur: x_t, the current x to condition on. A Tensor of shape [batch_size].
 201     Returns:
 202       p(z_t | z_{t-1}, x_t) as a univariate normal distribution.
 203     """
 204     z_prev = tf.convert_to_tensor(z_prev)
 205     x_cur = tf.convert_to_tensor(x_cur)
 206     batch_size = tf.shape(z_prev)[0]
 207     z_var = self.transition_variances.read(t)
 208     x_var = self.emission_variances.read(t)
 209     x_weight = self.emission_weights.read(t)
 210     prev_state_weight = x_var/(tf.square(x_weight)*z_var + x_var)
 211     prev_state_weight *= tf.cond(tf.greater(t, 0),
 212                                  lambda: self.transition_weights.read(t-1),
 213                                  lambda: tf.zeros_like(prev_state_weight))
 214     cur_obs_weight = (x_weight*z_var)/(tf.square(x_weight)*z_var + x_var)
 215     loc = prev_state_weight*z_prev + cur_obs_weight*x_cur
 216     scale = tf.sqrt((z_var*x_var)/(tf.square(x_weight)*z_var + x_var))
 217     scale = tf.tile(scale[tf.newaxis], [batch_size])
 218     return tfd.Normal(loc=loc, scale=scale)
 219 
 220   def smoothing(self, t, z_prev, xs):
 221     """Computes the smoothing distribution p(z_t | z_{t-1}, x_{t:num_timesteps).
 222 
 223     Args:
 224       t: A python int, the index for z_t. When t is 0, z_prev is ignored,
 225         giving p(z_0 | x_{0:num_timesteps-1}).
 226       z_prev: z_{t-1}, the previous z to condition on. A Tensor of shape
 227         [batch_size].
 228       xs: x_{t:num_timesteps}, the future xs to condition on. A Tensor of shape
 229         [num_timesteps - t, batch_size].
 230     Returns:
 231       p(z_t | z_{t-1}, x_{t:num_timesteps}) as a univariate normal distribution.
 232     """
 233     xs = tf.convert_to_tensor(xs)
 234     z_prev = tf.convert_to_tensor(z_prev)
 235     batch_size = tf.shape(xs)[1]
 236     mess_mean, mess_prec = tf.cond(
 237         tf.less(t, self.num_timesteps-1),
 238         lambda: tf.unstack(self._compute_backwards_messages(xs[1:]).read(0)),
 239         lambda: [tf.zeros([batch_size]), tf.zeros([batch_size])])
 240     return self._smoothing_from_message(t, z_prev, xs[0], mess_mean, mess_prec)
 241 
 242   def _smoothing_from_message(self, t, z_prev, x_t, mess_mean, mess_prec):
 243     """Computes the smoothing distribution given message incoming to z_t.
 244 
 245     Computes p(z_t | z_{t-1}, x_{t:num_timesteps}) given the message incoming
 246     to the node for z_t.
 247 
 248     Args:
 249       t: A python int, the index for z_t. When t is 0, z_prev is ignored.
 250       z_prev: z_{t-1}, the previous z to condition on. A Tensor of shape
 251         [batch_size].
 252       x_t: The observation x at timestep t.
 253       mess_mean: The mean of the message incoming to z_t, in information form.
 254       mess_prec: The precision of the message incoming to z_t.
 255     Returns:
 256       p(z_t | z_{t-1}, x_{t:num_timesteps}) as a univariate normal distribution.
 257     """
 258 
 259     batch_size = tf.shape(x_t)[0]
 260     z_var = self.transition_variances.read(t)
 261     x_var = self.emission_variances.read(t)
 262     w_x = self.emission_weights.read(t)
 263 
 264     def transition_term():
 265       return (tf.square(self.transition_weights.read(t))/
 266               self.transition_variances.read(t+1))
 267 
 268     prec = 1./z_var + tf.square(w_x)/x_var + mess_prec
 269     prec += tf.cond(tf.less(t, self.num_timesteps-1),
 270                     transition_term, lambda: 0.)
 271     mean = x_t*(w_x/x_var) + mess_mean
 272     mean += tf.cond(tf.greater(t, 0),
 273                     lambda: z_prev*(self.transition_weights.read(t-1)/z_var),
 274                     lambda: 0.)
 275     mean = tf.reshape(mean / prec, [batch_size])
 276     scale = tf.reshape(tf.sqrt(1./prec), [batch_size])
 277     return tfd.Normal(loc=mean, scale=scale)
 278 
 279   def _compute_backwards_messages(self, xs):
 280     """Computes the backwards messages used in smoothing."""
 281     batch_size = tf.shape(xs)[1]
 282     num_xs = tf.shape(xs)[0]
 283     until_t = self.num_timesteps - num_xs
 284     xs = tf.TensorArray(dtype=xs.dtype,
 285                         size=num_xs,
 286                         dynamic_size=False,
 287                         clear_after_read=True).unstack(xs)
 288     messages_ta = tf.TensorArray(dtype=xs.dtype,
 289                                  size=num_xs,
 290                                  dynamic_size=False,
 291                                  clear_after_read=False)
 292 
 293     def compute_message(t, prev_mean, prev_prec, messages_ta):
 294       """Computes one step of the backwards messages."""
 295       z_var = self.transition_variances.read(t)
 296       w_z = self.transition_weights.read(t-1)
 297       x_var = self.emission_variances.read(t)
 298       w_x = self.emission_weights.read(t)
 299       cur_x = xs.read(t - until_t)
 300 
 301       # If it isn't the first message, add the terms from the transition.
 302       def transition_term():
 303         return (tf.square(self.transition_weights.read(t))/
 304                 self.transition_variances.read(t+1))
 305 
 306       unary_prec = 1/z_var + tf.square(w_x)/x_var
 307       unary_prec += tf.cond(tf.less(t, self.num_timesteps-1),
 308                             transition_term, lambda: 0.)
 309 
 310       unary_mean = (w_x / x_var) * cur_x
 311       pairwise_prec = w_z / z_var
 312 
 313       next_prec = -tf.square(pairwise_prec)/(unary_prec + prev_prec)
 314       next_mean = (pairwise_prec * (unary_mean + prev_mean) /
 315                    (unary_prec + prev_prec))
 316       next_prec = tf.reshape(next_prec, [batch_size])
 317       next_mean = tf.reshape(next_mean, [batch_size])
 318       messages_ta = messages_ta.write(t - until_t,
 319                                       tf.stack([next_mean, next_prec]))
 320       return t-1, next_mean, next_prec, messages_ta
 321 
 322     def pred(t, *unused_args):
 323       return tf.greater_equal(t, until_t)
 324 
 325     init_prec = tf.zeros([batch_size], dtype=xs.dtype)
 326     init_mean = tf.zeros([batch_size], dtype=xs.dtype)
 327     t0 = tf.constant(self.num_timesteps - 1, dtype=tf.int32)
 328 
 329     outs = tf.while_loop(pred, compute_message,
 330                          (t0, init_mean, init_prec, messages_ta))
 331     messages = outs[-1]
 332     return messages
 333 
 334   def lookahead(self, t, z_prev):
 335     """Compute the 'lookahead' distribution, p(x_{t:T} | z_{t-1}).
 336 
 337     Args:
 338       t: A scalar Tensor int, the current timestep. Must be at least 1.
 339       z_prev: The latent state at time t-1. A Tensor of shape [batch_size].
 340     Returns:
 341       p(x_{t:T} | z_{t-1}) as a multivariate normal distribution.
 342     """
 343     z_prev = tf.convert_to_tensor(z_prev)
 344     sigma_zx = self.sigma_zx[t-1, t:]
 345     z_var = self.sigma_z[t-1, t-1]
 346     mean = tf.einsum("i,j->ij", z_prev, sigma_zx) / z_var
 347     variance = (self.sigma_x[t:, t:] -
 348                 tf.einsum("i,j->ij", sigma_zx, sigma_zx) / z_var)
 349     return tfd.MultivariateNormalFullCovariance(
 350         loc=mean, covariance_matrix=variance)
 351 
 352   def likelihood(self, xs):
 353     """Compute the true marginal likelihood of the data.
 354 
 355     Args:
 356       xs: The observations, a [num_timesteps, batch_size] float Tensor.
 357     Returns:
 358       likelihoods: A [batch_size] float Tensor representing the likelihood of
 359         each sequence of observations in the batch.
 360     """
 361     return self.obs_dist.log_prob(tf.transpose(xs))

 364 class TrainableGaussianHMM(GaussianHMM, base.ELBOTrainableSequenceModel):
 365   """An interface between importance-sampling training methods and the GHMM."""
 366 
 367   def __init__(self,
 368                num_timesteps,
 369                proposal_type,
 370                transition_variances=1.,
 371                emission_variances=1.,
 372                transition_weights=1.,
 373                emission_weights=1.,
 374                random_seed=None,
 375                dtype=tf.float32):
 376     """Constructs a trainable Gaussian HMM.
 377 
 378     Args:
 379       num_timesteps: A python int, the number of timesteps in the model.
 380       proposal_type: The type of proposal to use in the importance sampling
 381         setup. Could be "filtering", "smoothing", "prior", "true-filtering",
 382         or "true-smoothing". If "true-filtering" or "true-smoothing" are
 383         selected, then the true filtering or smoothing distributions are used to
 384         propose new states. If "learned-filtering" is selected then a
 385         distribution with learnable parameters is used. Specifically at each
 386         timestep the proposal is Gaussian with mean that is a learnable linear
 387         function of the previous state and current observation. The log variance
 388         is a per-timestep learnable constant. "learned-smoothing" is similar,
 389         but the mean is a learnable linear function of the previous state and
 390         all future observations. Note that this proposal class includes the true
 391         posterior. If "prior" is selected then states are proposed from the
 392         model's prior.
 393       transition_variances: The variance of p(z_t | z_t-1). Can be a scalar,
 394         setting all variances to be the same, or a Tensor of shape
 395         [num_timesteps].
 396       emission_variances: The variance of p(x_t | z_t). Can be a scalar,
 397         setting all variances to be the same, or a Tensor of shape
 398         [num_timesteps].
 399       transition_weights: The weight that defines the linear function that
 400         produces the mean of z_t given z_{t-1}. Can be a scalar, setting
 401         all weights to be the same, or a Tensor of shape [num_timesteps-1].
 402       emission_weights: The weight that defines the linear function that
 403         produces the mean of x_t given z_t. Can be a scalar, setting
 404         all weights to be the same, or a Tensor of shape [num_timesteps].
 405       random_seed: A seed for the proposal sampling, mainly useful for testing.
 406       dtype: The datatype of the state.
 407     """
 408     super(TrainableGaussianHMM, self).__init__(
 409         num_timesteps, transition_variances, emission_variances,
 410         transition_weights, emission_weights, dtype=dtype)
 411     self.random_seed = random_seed
 412     assert proposal_type in ["filtering", "smoothing", "prior",
 413                              "true-filtering", "true-smoothing"]
 414     if proposal_type == "true-filtering":
 415       self.proposal = self._filtering_proposal
 416     elif proposal_type == "true-smoothing":
 417       self.proposal = self._smoothing_proposal
 418     elif proposal_type == "prior":
 419       self.proposal = self.transition
 420     elif proposal_type == "filtering":
 421       self._learned_proposal_fn = base.NonstationaryLinearDistribution(
 422           num_timesteps, inputs_per_timestep=[1] + [2] * (num_timesteps-1))
 423       self.proposal = self._learned_filtering_proposal
 424     elif proposal_type == "smoothing":
 425       inputs_per_timestep = [num_timesteps] + [num_timesteps - t
 426                                                for t in range(num_timesteps-1)]
 427       self._learned_proposal_fn = base.NonstationaryLinearDistribution(
 428           num_timesteps, inputs_per_timestep=inputs_per_timestep)
 429       self.proposal = self._learned_smoothing_proposal
 430 
 431   def set_observations(self, xs, seq_lengths):
 432     """Sets the observations and stores the backwards messages."""
 433     # Squeeze out data dimension since everything is 1-d.
 434     xs = tf.squeeze(xs)
 435     self.batch_size = tf.shape(xs)[1]
 436     super(TrainableGaussianHMM, self).set_observations(xs, seq_lengths)
 437     self.messages = self._compute_backwards_messages(xs[1:])
 438 
 439   def zero_state(self, batch_size, dtype):
 440     return tf.zeros([batch_size], dtype=dtype)
 441 
 442   def propose_and_weight(self, state, t):
 443     """Computes the next state and log weights for the GHMM."""
 444     state_shape = tf.shape(state)
 445     xt = self.observations[t]
 446     p_zt = self.transition(t, state)
 447     q_zt = self.proposal(t, state)
 448     zt = q_zt.sample(seed=self.random_seed)
 449     zt = tf.reshape(zt, state_shape)
 450     p_xt_given_zt = self.emission(t, zt)
 451     log_p_zt = p_zt.log_prob(zt)
 452     log_q_zt = q_zt.log_prob(zt)
 453     log_p_xt_given_zt = p_xt_given_zt.log_prob(xt)
 454     weight = log_p_zt + log_p_xt_given_zt - log_q_zt
 455     return weight, zt
 456 
 457   def _filtering_proposal(self, t, state):
 458     """Uses the stored observations to compute the filtering distribution."""
 459     cur_x = self.observations[t]
 460     return self.filtering(t, state, cur_x)
 461 
 462   def _smoothing_proposal(self, t, state):
 463     """Uses the stored messages to compute the smoothing distribution."""
 464     mess_mean, mess_prec = tf.cond(
 465         tf.less(t, self.num_timesteps-1),
 466         lambda: tf.unstack(self.messages.read(t)),
 467         lambda: [tf.zeros([self.batch_size]), tf.zeros([self.batch_size])])
 468     return self._smoothing_from_message(t, state, self.observations[t],
 469                                         mess_mean, mess_prec)
 470 
 471   def _learned_filtering_proposal(self, t, state):
 472     cur_x = self.observations[t]
 473     inputs = tf.cond(tf.greater(t, 0),
 474                      lambda: tf.stack([state, cur_x], axis=0),
 475                      lambda: cur_x[tf.newaxis, :])
 476     return self._learned_proposal_fn(t, inputs)
 477 
 478   def _learned_smoothing_proposal(self, t, state):
 479     xs = self.observations_ta.gather(tf.range(t, self.num_timesteps))
 480     inputs = tf.cond(tf.greater(t, 0),
 481                      lambda: tf.concat([state[tf.newaxis, :], xs], axis=0),
 482                      lambda: xs)
 483     return self._learned_proposal_fn(t, inputs)
