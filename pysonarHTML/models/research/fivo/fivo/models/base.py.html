  30 class ELBOTrainableSequenceModel(object):
  31   """An abstract class for ELBO-trainable sequence models to extend.
  32 
  33   Because the ELBO, IWAE, and FIVO bounds all accept the same arguments,
  34   any model that is ELBO-trainable is also IWAE- and FIVO-trainable.
  35   """

  37   def zero_state(self, batch_size, dtype):
  38     """Returns the initial state of the model as a Tensor or tuple of Tensors.
  39 
  40     Args:
  41       batch_size: The batch size.
  42       dtype: The datatype to use for the state.
  43     """
  44     raise NotImplementedError("zero_state not yet implemented.")

  46   def set_observations(self, observations, seq_lengths):
  47     """Sets the observations for the model.
  48 
  49     This method provides the model with all observed variables including both
  50     inputs and targets. It will be called before running any computations with
  51     the model that require the observations, e.g. training the model or
  52     computing bounds, and should be used to run any necessary preprocessing
  53     steps.
  54 
  55     Args:
  56       observations: A potentially nested set of Tensors containing
  57         all observations for the model, both inputs and targets. Typically
  58         a set of Tensors with shape [max_seq_len, batch_size, data_size].
  59       seq_lengths: A [batch_size] Tensor of ints encoding the length of each
  60         sequence in the batch (sequences can be padded to a common length).
  61     """
  62     self.observations = observations
  63     self.max_seq_len = tf.reduce_max(seq_lengths)
  64     self.observations_ta = nested.tas_for_tensors(
  65         observations, self.max_seq_len, clear_after_read=False)
  66     self.seq_lengths = seq_lengths

  88 class ConditionalNormalDistribution(object):
  89   """A Normal distribution conditioned on Tensor inputs via a fc network."""
  90 
  91   def __init__(self, size, hidden_layer_sizes, sigma_min=0.0,
  92                raw_sigma_bias=0.25, hidden_activation_fn=tf.nn.relu,
  93                initializers=None, name="conditional_normal_distribution"):
  94     """Creates a conditional Normal distribution.
  95 
  96     Args:
  97       size: The dimension of the random variable.
  98       hidden_layer_sizes: The sizes of the hidden layers of the fully connected
  99         network used to condition the distribution on the inputs.
 100       sigma_min: The minimum standard deviation allowed, a scalar.
 101       raw_sigma_bias: A scalar that is added to the raw standard deviation
 102         output from the fully connected network. Set to 0.25 by default to
 103         prevent standard deviations close to 0.
 104       hidden_activation_fn: The activation function to use on the hidden layers
 105         of the fully connected network.
 106       initializers: The variable intitializers to use for the fully connected
 107         network. The network is implemented using snt.nets.MLP so it must
 108         be a dictionary mapping the keys 'w' and 'b' to the initializers for
 109         the weights and biases. Defaults to xavier for the weights and zeros
 110         for the biases when initializers is None.
 111       name: The name of this distribution, used for sonnet scoping.
 112     """
 113     self.sigma_min = sigma_min
 114     self.raw_sigma_bias = raw_sigma_bias
 115     self.name = name
 116     self.size = size
 117     if initializers is None:
 118       initializers = DEFAULT_INITIALIZERS
 119     self.fcnet = snt.nets.MLP(
 120         output_sizes=hidden_layer_sizes + [2*size],
 121         activation=hidden_activation_fn,
 122         initializers=initializers,
 123         activate_final=False,
 124         use_bias=True,
 125         name=name + "_fcnet")
 126 
 127   def condition(self, tensor_list, **unused_kwargs):
 128     """Computes the parameters of a normal distribution based on the inputs."""
 129     inputs = tf.concat(tensor_list, axis=1)
 130     outs = self.fcnet(inputs)
 131     mu, sigma = tf.split(outs, 2, axis=1)
 132     sigma = tf.maximum(tf.nn.softplus(sigma + self.raw_sigma_bias),
 133                        self.sigma_min)
 134     return mu, sigma
 135 
 136   def __call__(self, *args, **kwargs):
 137     """Creates a normal distribution conditioned on the inputs."""
 138     mu, sigma = self.condition(args, **kwargs)
 139     return tf.contrib.distributions.Normal(loc=mu, scale=sigma)

 142 class ConditionalBernoulliDistribution(object):
 143   """A Bernoulli distribution conditioned on Tensor inputs via a fc net."""
 144 
 145   def __init__(self, size, hidden_layer_sizes, hidden_activation_fn=tf.nn.relu,
 146                initializers=None, bias_init=0.0,
 147                name="conditional_bernoulli_distribution"):
 148     """Creates a conditional Bernoulli distribution.
 149 
 150     Args:
 151       size: The dimension of the random variable.
 152       hidden_layer_sizes: The sizes of the hidden layers of the fully connected
 153         network used to condition the distribution on the inputs.
 154       hidden_activation_fn: The activation function to use on the hidden layers
 155         of the fully connected network.
 156       initializers: The variable intiializers to use for the fully connected
 157         network. The network is implemented using snt.nets.MLP so it must
 158         be a dictionary mapping the keys 'w' and 'b' to the initializers for
 159         the weights and biases. Defaults to xavier for the weights and zeros
 160         for the biases when initializers is None.
 161       bias_init: A scalar or vector Tensor that is added to the output of the
 162         fully-connected network that parameterizes the mean of this
 163         distribution.
 164       name: The name of this distribution, used for sonnet scoping.
 165     """
 166     self.bias_init = bias_init
 167     self.size = size
 168     if initializers is None:
 169       initializers = DEFAULT_INITIALIZERS
 170     self.fcnet = snt.nets.MLP(
 171         output_sizes=hidden_layer_sizes + [size],
 172         activation=hidden_activation_fn,
 173         initializers=initializers,
 174         activate_final=False,
 175         use_bias=True,
 176         name=name + "_fcnet")
 177 
 178   def condition(self, tensor_list):
 179     """Computes the p parameter of the Bernoulli distribution."""
 180     inputs = tf.concat(tensor_list, axis=1)
 181     return self.fcnet(inputs) + self.bias_init
 182 
 183   def __call__(self, *args):
 184     p = self.condition(args)
 185     return tf.contrib.distributions.Bernoulli(logits=p)

 188 class NormalApproximatePosterior(ConditionalNormalDistribution):
 189   """A Normally-distributed approx. posterior with res_q parameterization."""
 190 
 191   def __init__(self, size, hidden_layer_sizes, sigma_min=0.0,
 192                raw_sigma_bias=0.25, hidden_activation_fn=tf.nn.relu,
 193                initializers=None, smoothing=False,
 194                name="conditional_normal_distribution"):
 195     super(NormalApproximatePosterior, self).__init__(
 196         size, hidden_layer_sizes, sigma_min=sigma_min,
 197         raw_sigma_bias=raw_sigma_bias,
 198         hidden_activation_fn=hidden_activation_fn, initializers=initializers,
 199         name=name)
 200     self.smoothing = smoothing
 201 
 202   def condition(self, tensor_list, prior_mu, smoothing_tensors=None):
 203     """Generates the mean and variance of the normal distribution.
 204 
 205     Args:
 206       tensor_list: The list of Tensors to condition on. Will be concatenated and
 207         fed through a fully connected network.
 208       prior_mu: The mean of the prior distribution associated with this
 209         approximate posterior. Will be added to the mean produced by
 210         this approximate posterior, in res_q fashion.
 211       smoothing_tensors: A list of Tensors. If smoothing is True, these Tensors
 212         will be concatenated with the tensors in tensor_list.
 213     Returns:
 214       mu: The mean of the approximate posterior.
 215       sigma: The standard deviation of the approximate posterior.
 216     """
 217     if self.smoothing:
 218       tensor_list.extend(smoothing_tensors)
 219     mu, sigma = super(NormalApproximatePosterior, self).condition(tensor_list)
 220     return mu + prior_mu, sigma

 223 class NonstationaryLinearDistribution(object):
 224   """A set of loc-scale distributions that are linear functions of inputs.
 225 
 226   This class defines a series of location-scale distributions such that
 227   the means are learnable linear functions of the inputs and the log variances
 228   are learnable constants. The functions and log variances are different across
 229   timesteps, allowing the distributions to be nonstationary.
 230   """
 231 
 232   def __init__(self,
 233                num_timesteps,
 234                inputs_per_timestep=None,
 235                outputs_per_timestep=None,
 236                initializers=None,
 237                variance_min=0.0,
 238                output_distribution=tfd.Normal,
 239                dtype=tf.float32):
 240     """Creates a NonstationaryLinearDistribution.
 241 
 242     Args:
 243       num_timesteps: The number of timesteps, i.e. the number of distributions.
 244       inputs_per_timestep: A list of python ints, the dimension of inputs to the
 245         linear function at each timestep. If not provided, the dimension at each
 246         timestep is assumed to be 1.
 247       outputs_per_timestep: A list of python ints, the dimension of the output
 248         distribution at each timestep. If not provided, the dimension at each
 249         timestep is assumed to be 1.
 250       initializers: A dictionary containing intializers for the variables. The
 251         initializer under the key 'w' is used for the weights in the linear
 252         function and the initializer under the key 'b' is used for the biases.
 253         Defaults to xavier initialization for the weights and zeros for the
 254         biases.
 255       variance_min: Python float, the minimum variance of each distribution.
 256       output_distribution: A locatin-scale subclass of tfd.Distribution that
 257         defines the output distribution, e.g. Normal.
 258       dtype: The dtype of the weights and biases.
 259     """
 260     if not initializers:
 261       initializers = DEFAULT_INITIALIZERS
 262     if not inputs_per_timestep:
 263       inputs_per_timestep = [1] * num_timesteps
 264     if not outputs_per_timestep:
 265       outputs_per_timestep = [1] * num_timesteps
 266     self.num_timesteps = num_timesteps
 267     self.variance_min = variance_min
 268     self.initializers = initializers
 269     self.dtype = dtype
 270     self.output_distribution = output_distribution
 271 
 272     def _get_variables_ta(shapes, name, initializer, trainable=True):
 273       """Creates a sequence of variables and stores them in a TensorArray."""
 274       # Infer shape if all shapes are equal.
 275       first_shape = shapes[0]
 276       infer_shape = all(shape == first_shape for shape in shapes)
 277       ta = tf.TensorArray(
 278           dtype=dtype, size=len(shapes), dynamic_size=False,
 279           clear_after_read=False, infer_shape=infer_shape)
 280       for t, shape in enumerate(shapes):
 281         var = tf.get_variable(
 282             name % t, shape=shape, initializer=initializer, trainable=trainable)
 283         ta = ta.write(t, var)
 284       return ta
 285 
 286     bias_shapes = [[num_outputs] for num_outputs in outputs_per_timestep]
 287     self.log_variances = _get_variables_ta(
 288         bias_shapes, "proposal_log_variance_%d", initializers["b"])
 289     self.mean_biases = _get_variables_ta(
 290         bias_shapes, "proposal_b_%d", initializers["b"])
 291     weight_shapes = zip(inputs_per_timestep, outputs_per_timestep)
 292     self.mean_weights = _get_variables_ta(
 293         weight_shapes, "proposal_w_%d", initializers["w"])
 294     self.shapes = tf.TensorArray(
 295         dtype=tf.int32, size=num_timesteps,
 296         dynamic_size=False, clear_after_read=False).unstack(weight_shapes)
 297 
 298   def __call__(self, t, inputs):
 299     """Computes the distribution at timestep t.
 300 
 301     Args:
 302       t: Scalar integer Tensor, the current timestep. Must be in
 303         [0, num_timesteps).
 304       inputs: The inputs to the linear function parameterizing the mean of
 305         the current distribution. A Tensor of shape [batch_size, num_inputs_t].
 306     Returns:
 307       A tfd.Distribution subclass representing the distribution at timestep t.
 308     """
 309     b = self.mean_biases.read(t)
 310     w = self.mean_weights.read(t)
 311     shape = self.shapes.read(t)
 312     w = tf.reshape(w, shape)
 313     b = tf.reshape(b, [shape[1], 1])
 314     log_variance = self.log_variances.read(t)
 315     scale = tf.sqrt(tf.maximum(tf.exp(log_variance), self.variance_min))
 316     loc = tf.matmul(w, inputs, transpose_a=True) + b
 317     return self.output_distribution(loc=loc, scale=scale)

 320 def encode_all(inputs, encoder):
 321   """Encodes a timeseries of inputs with a time independent encoder.
 322 
 323   Args:
 324     inputs: A [time, batch, feature_dimensions] tensor.
 325     encoder: A network that takes a [batch, features_dimensions] input and
 326       encodes the input.
 327   Returns:
 328     A [time, batch, encoded_feature_dimensions] output tensor.
 329   """
 330   input_shape = tf.shape(inputs)
 331   num_timesteps, batch_size = input_shape[0], input_shape[1]
 332   reshaped_inputs = tf.reshape(inputs, [-1, inputs.shape[-1]])
 333   inputs_encoded = encoder(reshaped_inputs)
 334   inputs_encoded = tf.reshape(inputs_encoded,
 335                               [num_timesteps, batch_size, encoder.output_size])
 336   return inputs_encoded

 339 def ta_for_tensor(x, **kwargs):
 340   """Creates a TensorArray for the input tensor."""
 341   return tf.TensorArray(
 342       x.dtype, tf.shape(x)[0], dynamic_size=False, **kwargs).unstack(x)
