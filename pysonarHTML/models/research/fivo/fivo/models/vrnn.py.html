  34 class VRNN(object):
  35   """Implementation of a Variational Recurrent Neural Network (VRNN).
  36 
  37   Introduced in "A Recurrent Latent Variable Model for Sequential data"
  38   by Chung et al. https://arxiv.org/pdf/1506.02216.pdf.
  39 
  40   The VRNN is a sequence model similar to an RNN that uses stochastic latent
  41   variables to improve its representational power. It can be thought of as a
  42   sequential analogue to the variational auto-encoder (VAE).
  43 
  44   The VRNN has a deterministic RNN as its backbone, represented by the
  45   sequence of RNN hidden states h_t. At each timestep, the RNN hidden state h_t
  46   is conditioned on the previous sequence element, x_{t-1}, as well as the
  47   latent state from the previous timestep, z_{t-1}.
  48 
  49   In this implementation of the VRNN the latent state z_t is Gaussian. The
  50   model's prior over z_t (also called the transition distribution) is
  51   distributed as Normal(mu_t, diag(sigma_t^2)) where mu_t and sigma_t are the
  52   mean and standard deviation output from a fully connected network that accepts
  53   the rnn hidden state h_t as input.
  54 
  55   The emission distribution p(x_t|z_t, h_t) is conditioned on the latent state
  56   z_t as well as the current RNN hidden state h_t via a fully connected network.
  57 
  58   To increase the modeling power of the VRNN, two additional networks are
  59   used to extract features from the data and the latent state. Those networks
  60   are called data_encoder and latent_encoder respectively.
  61 
  62   For an example of how to call the VRNN's methods see sample_step.
  63 
  64   There are a few differences between this exposition and the paper.
  65   First, the indexing scheme for h_t is different than the paper's -- what the
  66   paper calls h_t we call h_{t+1}. This is the same notation used by Fraccaro
  67   et al. to describe the VRNN in the paper linked above. Also, the VRNN paper
  68   uses VAE terminology to refer to the different internal networks, so it
  69   refers to the emission distribution as the decoder. This implementation also
  70   renames the functions phi_x and phi_z in the paper to data_encoder and
  71   latent_encoder.
  72   """
  73 
  74   def __init__(self,
  75                rnn_cell,
  76                data_encoder,
  77                latent_encoder,
  78                transition,
  79                emission,
  80                random_seed=None):
  81     """Create a VRNN.
  82 
  83     Args:
  84       rnn_cell: A subclass of tf.nn.rnn_cell.RNNCell that will form the
  85         deterministic backbone of the VRNN. The inputs to the RNN will be the
  86         encoded latent state of the previous timestep with shape
  87         [batch_size, encoded_latent_size] as well as the encoded input of the
  88         current timestep, a Tensor of shape [batch_size, encoded_data_size].
  89       data_encoder: A callable that accepts a batch of data x_t and
  90         'encodes' it, e.g. runs it through a fully connected network. Must
  91         accept as argument the inputs x_t, a Tensor of the shape
  92         [batch_size, data_size] and return a Tensor of shape
  93         [batch_size, encoded_data_size]. This callable will be called multiple
  94         times in the VRNN cell so if scoping is not handled correctly then
  95         multiple copies of the variables in this network could be made. It is
  96         recommended to use a snt.nets.MLP module, which takes care of this for
  97         you.
  98       latent_encoder: A callable that accepts a latent state z_t and
  99         'encodes' it, e.g. runs it through a fully connected network. Must
 100         accept as argument a Tensor of shape [batch_size, latent_size] and
 101         return a Tensor of shape [batch_size, encoded_latent_size].
 102         This callable must also have the property 'output_size' defined,
 103         returning encoded_latent_size.
 104       transition: A callable that implements the transition distribution
 105         p(z_t|h_t). Must accept as argument the previous RNN hidden state and
 106         return a tf.distributions.Normal distribution conditioned on the input.
 107       emission: A callable that implements the emission distribution
 108         p(x_t|z_t, h_t). Must accept as arguments the encoded latent state
 109         and the RNN hidden state and return a subclass of
 110         tf.distributions.Distribution that can be used to evaluate the logprob
 111         of the targets.
 112       random_seed: The seed for the random ops. Sets the seed for sample_step.
 113     """
 114     self.random_seed = random_seed
 115     self.rnn_cell = rnn_cell
 116     self.data_encoder = data_encoder
 117     self.latent_encoder = latent_encoder
 118     self.encoded_z_size = latent_encoder.output_size
 119     self.state_size = (self.rnn_cell.state_size)
 120     self._transition = transition
 121     self._emission = emission
 122 
 123   def zero_state(self, batch_size, dtype):
 124     """The initial state of the VRNN.
 125 
 126     Contains the initial state of the RNN and the inital encoded latent.
 127 
 128     Args:
 129       batch_size: The batch size.
 130       dtype: The data type of the VRNN.
 131     Returns:
 132       zero_state: The initial state of the VRNN.
 133     """
 134     return VRNNState(
 135         rnn_state=self.rnn_cell.zero_state(batch_size, dtype),
 136         latent_encoded=tf.zeros(
 137             [batch_size, self.latent_encoder.output_size], dtype=dtype))
 138 
 139   def run_rnn(self, prev_rnn_state, prev_latent_encoded, inputs):
 140     """Runs the deterministic RNN for one step.
 141 
 142     Args:
 143       prev_rnn_state: The state of the RNN from the previous timestep.
 144       prev_latent_encoded: Float Tensor of shape
 145         [batch_size, encoded_latent_size], the previous latent state z_{t-1}
 146         run through latent_encoder.
 147       inputs: A Tensor of shape [batch_size, data_size], the current inputs to
 148         the model. Most often this is x_{t-1}, the previous token in the
 149         observation sequence.
 150     Returns:
 151       rnn_out: The output of the RNN.
 152       rnn_state: The new state of the RNN.
 153     """
 154     inputs_encoded = self.data_encoder(tf.to_float(inputs))
 155     rnn_inputs = tf.concat([inputs_encoded, prev_latent_encoded], axis=1)
 156     rnn_out, rnn_state = self.rnn_cell(rnn_inputs, prev_rnn_state)
 157     return rnn_out, rnn_state
 158 
 159   def transition(self, rnn_out):
 160     """Computes the transition distribution p(z_t|h_t).
 161 
 162     Note that p(z_t | h_t) = p(z_t| z_{1:t-1}, x_{1:t-1})
 163 
 164     Args:
 165       rnn_out: The output of the rnn for the current timestep.
 166     Returns:
 167       p(z_t | h_t): A normal distribution with event shape
 168         [batch_size, latent_size].
 169     """
 170     return self._transition(rnn_out)
 171 
 172   def emission(self, latent, rnn_out):
 173     """Computes the emission distribution p(x_t | z_t, h_t).
 174 
 175     Note that p(x_t | z_t, h_t) = p(x_t | z_{1:t}, x_{1:t-1}).
 176 
 177     Args:
 178       latent: The stochastic latent state z_t.
 179       rnn_out: The output of the rnn for the current timestep.
 180     Returns:
 181       p(x_t | z_t, h_t): A distribution with event shape
 182         [batch_size, data_size].
 183       latent_encoded: The latent state encoded with latent_encoder. Should be
 184         passed to run_rnn on the next timestep.
 185     """
 186     latent_encoded = self.latent_encoder(latent)
 187     return self._emission(latent_encoded, rnn_out), latent_encoded
 188 
 189   def sample_step(self, prev_state, inputs, unused_t):
 190     """Samples one output from the model.
 191 
 192     Args:
 193       prev_state: The previous state of the model, a VRNNState containing the
 194         previous rnn state and the previous encoded latent.
 195       inputs: A Tensor of shape [batch_size, data_size], the current inputs to
 196         the model. Most often this is x_{t-1}, the previous token in the
 197         observation sequence.
 198       unused_t: The current timestep. Not used currently.
 199     Returns:
 200       new_state: The next state of the model, a VRNNState.
 201       xt: A float Tensor of shape [batch_size, data_size], an output sampled
 202         from the emission distribution.
 203     """
 204     rnn_out, rnn_state = self.run_rnn(prev_state.rnn_state,
 205                                       prev_state.latent_encoded,
 206                                       inputs)
 207     p_zt = self.transition(rnn_out)
 208     zt = p_zt.sample(seed=self.random_seed)
 209     p_xt_given_zt, latent_encoded = self.emission(zt, rnn_out)
 210     xt = p_xt_given_zt.sample(seed=self.random_seed)
 211     new_state = VRNNState(rnn_state=rnn_state, latent_encoded=latent_encoded)
 212     return new_state, tf.to_float(xt)

 221 class TrainableVRNN(VRNN, base.ELBOTrainableSequenceModel):
 222   """A VRNN subclass with proposals and methods for training and evaluation.
 223 
 224   This class adds proposals used for training with importance-sampling based
 225   methods such as the ELBO. The model can be configured to propose from one
 226   of three proposals: a learned filtering proposal, a learned smoothing
 227   proposal, or the prior (i.e. the transition distribution).
 228 
 229   As described in the VRNN paper, the learned filtering proposal is
 230   parameterized by a fully connected neural network that accepts as input the
 231   current target x_t and the current rnn output h_t. The learned smoothing
 232   proposal is also given the hidden state of an RNN run in reverse over the
 233   inputs, so as to incorporate information about future observations. This
 234   smoothing proposal is not described in the VRNN paper.
 235 
 236   All learned proposals use the 'res_q' parameterization, meaning that instead
 237   of directly producing the mean of z_t, the proposal network predicts the
 238   'residual' from the prior's mean. This is explored more in section 3.3 of
 239   https://arxiv.org/pdf/1605.07571.pdf.
 240 
 241   During training, the latent state z_t is sampled from the proposal and the
 242   reparameterization trick is used to provide low-variance gradients.
 243 
 244   Note that the VRNN paper uses VAE terminology to refer to the different
 245   internal networks, so the proposal is referred to as the encoder.
 246   """
 247 
 248   def __init__(self,
 249                rnn_cell,
 250                data_encoder,
 251                latent_encoder,
 252                transition,
 253                emission,
 254                proposal_type,
 255                proposal=None,
 256                rev_rnn_cell=None,
 257                tilt=None,
 258                random_seed=None):
 259     """Create a trainable RNN.
 260 
 261     Args:
 262       rnn_cell: A subclass of tf.nn.rnn_cell.RNNCell that will form the
 263         deterministic backbone of the VRNN. The inputs to the RNN will be the
 264         encoded latent state of the previous timestep with shape
 265         [batch_size, encoded_latent_size] as well as the encoded input of the
 266         current timestep, a Tensor of shape [batch_size, encoded_data_size].
 267       data_encoder: A callable that accepts a batch of data x_t and
 268         'encodes' it, e.g. runs it through a fully connected network. Must
 269         accept as argument the inputs x_t, a Tensor of the shape
 270         [batch_size, data_size] and return a Tensor of shape
 271         [batch_size, encoded_data_size]. This callable will be called multiple
 272         times in the VRNN cell so if scoping is not handled correctly then
 273         multiple copies of the variables in this network could be made. It is
 274         recommended to use a snt.nets.MLP module, which takes care of this for
 275         you.
 276       latent_encoder: A callable that accepts a latent state z_t and
 277         'encodes' it, e.g. runs it through a fully connected network. Must
 278         accept as argument a Tensor of shape [batch_size, latent_size] and
 279         return a Tensor of shape [batch_size, encoded_latent_size].
 280         This callable must also have the property 'output_size' defined,
 281         returning encoded_latent_size.
 282       transition: A callable that implements the transition distribution
 283         p(z_t|h_t). Must accept as argument the previous RNN hidden state and
 284         return a tf.distributions.Normal distribution conditioned on the input.
 285       emission: A callable that implements the emission distribution
 286         p(x_t|z_t, h_t). Must accept as arguments the encoded latent state
 287         and the RNN hidden state and return a subclass of
 288         tf.distributions.Distribution that can be used to evaluate the logprob
 289         of the targets.
 290       proposal_type: A string indicating the type of proposal to use. Can
 291         be either "filtering", "smoothing", or "prior". When proposal_type is
 292         "filtering" or "smoothing", proposal must be provided. When
 293         proposal_type is "smoothing", rev_rnn_cell must also be provided.
 294       proposal: A callable that implements the proposal q(z_t| h_t, x_{1:T}).
 295         If proposal_type is "filtering" then proposal must accept as arguments
 296         the current rnn output, the encoded target of the current timestep,
 297         and the mean of the prior. If proposal_type is "smoothing" then
 298         in addition to the current rnn output and the mean of the prior
 299         proposal must accept as arguments the output of the reverse rnn.
 300         proposal should return a tf.distributions.Normal distribution
 301         conditioned on its inputs. If proposal_type is "prior" this argument is
 302         ignored.
 303       rev_rnn_cell: A subclass of tf.nn.rnn_cell.RNNCell that will aggregate
 304         observation statistics in the reverse direction. The inputs to the RNN
 305         will be the encoded reverse input of the current timestep, a Tensor of
 306         shape [batch_size, encoded_data_size].
 307       tilt: A callable that implements the log of a positive tilting function
 308         (ideally approximating log p(x_{t+1}|z_t, h_t). Must accept as arguments
 309         the encoded latent state and the RNN hidden state and return a subclass
 310         of tf.distributions.Distribution that can be used to evaluate the
 311         logprob of x_{t+1}. Optionally, None and then no tilt is used.
 312       random_seed: The seed for the random ops. Sets the seed for sample_step
 313         and __call__.
 314     """
 315     super(TrainableVRNN, self).__init__(
 316         rnn_cell, data_encoder, latent_encoder,
 317         transition, emission, random_seed=random_seed)
 318     self.rev_rnn_cell = rev_rnn_cell
 319     self._tilt = tilt
 320     assert proposal_type in ["filtering", "smoothing", "prior"]
 321     self._proposal = proposal
 322     self.proposal_type = proposal_type
 323     if proposal_type != "prior":
 324       assert proposal, "If not proposing from the prior, must provide proposal."
 325     if proposal_type == "smoothing":
 326       assert rev_rnn_cell, "Must provide rev_rnn_cell for smoothing proposal."
 327 
 328   def zero_state(self, batch_size, dtype):
 329     super_state = super(TrainableVRNN, self).zero_state(batch_size, dtype)
 330     return TrainableVRNNState(
 331         rnn_out=tf.zeros([batch_size, self.rnn_cell.output_size], dtype=dtype),
 332         **super_state._asdict())
 333 
 334   def set_observations(self, observations, seq_lengths):
 335     """Stores the model's observations.
 336 
 337     Stores the observations (inputs and targets) in TensorArrays and precomputes
 338     things for later like the reverse RNN output and encoded targets.
 339 
 340     Args:
 341       observations: The observations of the model, a tuple containing two
 342         Tensors of shape [max_seq_len, batch_size, data_size]. The Tensors
 343         should be the inputs and targets, respectively.
 344       seq_lengths: An int Tensor of shape [batch_size] containing the length
 345         of each sequence in observations.
 346     """
 347     inputs, targets = observations
 348     self.seq_lengths = seq_lengths
 349     self.max_seq_len = tf.reduce_max(seq_lengths)
 350     self.inputs_ta = base.ta_for_tensor(inputs, clear_after_read=False)
 351     self.targets_ta = base.ta_for_tensor(targets, clear_after_read=False)
 352     targets_encoded = base.encode_all(targets, self.data_encoder)
 353     self.targets_encoded_ta = base.ta_for_tensor(targets_encoded,
 354                                                  clear_after_read=False)
 355     if self.rev_rnn_cell:
 356       reverse_targets_encoded = tf.reverse_sequence(
 357           targets_encoded, seq_lengths, seq_axis=0, batch_axis=1)
 358       # Compute the reverse rnn over the targets.
 359       reverse_rnn_out, _ = tf.nn.dynamic_rnn(self.rev_rnn_cell,
 360                                              reverse_targets_encoded,
 361                                              time_major=True,
 362                                              dtype=tf.float32)
 363       reverse_rnn_out = tf.reverse_sequence(reverse_rnn_out, seq_lengths,
 364                                             seq_axis=0, batch_axis=1)
 365       self.reverse_rnn_ta = base.ta_for_tensor(reverse_rnn_out,
 366                                                clear_after_read=False)
 367 
 368   def _filtering_proposal(self, rnn_out, prior, t):
 369     """Computes the filtering proposal distribution."""
 370     return self._proposal(rnn_out,
 371                           self.targets_encoded_ta.read(t),
 372                           prior_mu=prior.mean())
 373 
 374   def _smoothing_proposal(self, rnn_out, prior, t):
 375     """Computes the smoothing proposal distribution."""
 376     return self._proposal(rnn_out,
 377                           smoothing_tensors=[self.reverse_rnn_ta.read(t)],
 378                           prior_mu=prior.mean())
 379 
 380   def proposal(self, rnn_out, prior, t):
 381     """Computes the proposal distribution specified by proposal_type.
 382 
 383     Args:
 384       rnn_out: The output of the rnn for the current timestep.
 385       prior: A tf.distributions.Normal distribution representing the prior
 386         over z_t, p(z_t | z_{1:t-1}, x_{1:t-1}). Used for 'res_q'.
 387       t: A scalar int Tensor, the current timestep.
 388     """
 389     if self.proposal_type == "filtering":
 390       return self._filtering_proposal(rnn_out, prior, t)
 391     elif self.proposal_type == "smoothing":
 392       return self._smoothing_proposal(rnn_out, prior, t)
 393     elif self.proposal_type == "prior":
 394       return self.transition(rnn_out)
 395 
 396   def tilt(self, rnn_out, latent_encoded, targets):
 397     r_func = self._tilt(rnn_out, latent_encoded)
 398     return tf.reduce_sum(r_func.log_prob(targets), axis=-1)
 399 
 400   def propose_and_weight(self, state, t):
 401     """Runs the model and computes importance weights for one timestep.
 402 
 403     Runs the model and computes importance weights, sampling from the proposal
 404     instead of the transition/prior.
 405 
 406     Args:
 407       state: The previous state of the model, a TrainableVRNNState containing
 408         the previous rnn state, the previous rnn outs, and the previous encoded
 409         latent.
 410       t: A scalar integer Tensor, the current timestep.
 411     Returns:
 412       weights: A float Tensor of shape [batch_size].
 413       new_state: The new state of the model.
 414     """
 415     inputs = self.inputs_ta.read(t)
 416     targets = self.targets_ta.read(t)
 417     rnn_out, next_rnn_state = self.run_rnn(state.rnn_state,
 418                                            state.latent_encoded,
 419                                            inputs)
 420     p_zt = self.transition(rnn_out)
 421     q_zt = self.proposal(rnn_out, p_zt, t)
 422     zt = q_zt.sample(seed=self.random_seed)
 423     p_xt_given_zt, latent_encoded = self.emission(zt, rnn_out)
 424     log_p_xt_given_zt = tf.reduce_sum(p_xt_given_zt.log_prob(targets), axis=-1)
 425     log_p_zt = tf.reduce_sum(p_zt.log_prob(zt), axis=-1)
 426     log_q_zt = tf.reduce_sum(q_zt.log_prob(zt), axis=-1)
 427     weights = log_p_zt + log_p_xt_given_zt - log_q_zt
 428     if self._tilt:
 429       prev_log_r = tf.cond(
 430           tf.greater(t, 0),
 431           lambda: self.tilt(state.rnn_out, state.latent_encoded, targets),
 432           lambda: 0.)  # On the first step, prev_log_r = 0.
 433       log_r = tf.cond(
 434           tf.less(t + 1, self.max_seq_len),
 435           lambda: self.tilt(rnn_out, latent_encoded, self.targets_ta.read(t+1)),
 436           lambda: 0.)
 437       # On the last step, log_r = 0.
 438       log_r *= tf.to_float(t < self.seq_lengths - 1)
 439       weights += log_r - prev_log_r
 440     new_state = TrainableVRNNState(rnn_state=next_rnn_state,
 441                                    rnn_out=rnn_out,
 442                                    latent_encoded=latent_encoded)
 443     return weights, new_state

 450 def create_vrnn(
 451     data_size,
 452     latent_size,
 453     emission_class,
 454     rnn_hidden_size=None,
 455     fcnet_hidden_sizes=None,
 456     encoded_data_size=None,
 457     encoded_latent_size=None,
 458     sigma_min=0.0,
 459     raw_sigma_bias=0.25,
 460     emission_bias_init=0.0,
 461     use_tilt=False,
 462     proposal_type="filtering",
 463     initializers=None,
 464     random_seed=None):
 465   """A factory method for creating VRNN cells.
 466 
 467   Args:
 468     data_size: The dimension of the vectors that make up the data sequences.
 469     latent_size: The size of the stochastic latent state of the VRNN.
 470     emission_class: The class of the emission distribution. Can be either
 471       ConditionalNormalDistribution or ConditionalBernoulliDistribution.
 472     rnn_hidden_size: The hidden state dimension of the RNN that forms the
 473       deterministic part of this VRNN. If None, then it defaults
 474       to latent_size.
 475     fcnet_hidden_sizes: A list of python integers, the size of the hidden
 476       layers of the fully connected networks that parameterize the conditional
 477       distributions of the VRNN. If None, then it defaults to one hidden
 478       layer of size latent_size.
 479     encoded_data_size: The size of the output of the data encoding network. If
 480       None, defaults to latent_size.
 481     encoded_latent_size: The size of the output of the latent state encoding
 482       network. If None, defaults to latent_size.
 483     sigma_min: The minimum value that the standard deviation of the
 484       distribution over the latent state can take.
 485     raw_sigma_bias: A scalar that is added to the raw standard deviation
 486       output from the neural networks that parameterize the prior and
 487       approximate posterior. Useful for preventing standard deviations close
 488       to zero.
 489     emission_bias_init: A bias to added to the raw output of the fully
 490       connected network that parameterizes the emission distribution. Useful
 491       for initalizing the mean of the distribution to a sensible starting point
 492       such as the mean of the training data. Only used with Bernoulli generative
 493       distributions.
 494     use_tilt: If true, create a VRNN with a tilting function.
 495     proposal_type: The type of proposal to use. Can be "filtering", "smoothing",
 496       or "prior".
 497     initializers: The variable intitializers to use for the fully connected
 498       networks and RNN cell. Must be a dictionary mapping the keys 'w' and 'b'
 499       to the initializers for the weights and biases. Defaults to xavier for
 500       the weights and zeros for the biases when initializers is None.
 501     random_seed: A random seed for the VRNN resampling operations.
 502   Returns:
 503     model: A TrainableVRNN object.
 504   """
 505   if rnn_hidden_size is None:
 506     rnn_hidden_size = latent_size
 507   if fcnet_hidden_sizes is None:
 508     fcnet_hidden_sizes = [latent_size]
 509   if encoded_data_size is None:
 510     encoded_data_size = latent_size
 511   if encoded_latent_size is None:
 512     encoded_latent_size = latent_size
 513   if initializers is None:
 514     initializers = _DEFAULT_INITIALIZERS
 515   data_encoder = snt.nets.MLP(
 516       output_sizes=fcnet_hidden_sizes + [encoded_data_size],
 517       initializers=initializers,
 518       name="data_encoder")
 519   latent_encoder = snt.nets.MLP(
 520       output_sizes=fcnet_hidden_sizes + [encoded_latent_size],
 521       initializers=initializers,
 522       name="latent_encoder")
 523   transition = base.ConditionalNormalDistribution(
 524       size=latent_size,
 525       hidden_layer_sizes=fcnet_hidden_sizes,
 526       sigma_min=sigma_min,
 527       raw_sigma_bias=raw_sigma_bias,
 528       initializers=initializers,
 529       name="prior")
 530   # Construct the emission distribution.
 531   if emission_class == base.ConditionalBernoulliDistribution:
 532     # For Bernoulli distributed outputs, we initialize the bias so that the
 533     # network generates on average the mean from the training set.
 534     emission_dist = functools.partial(base.ConditionalBernoulliDistribution,
 535                                       bias_init=emission_bias_init)
 536   else:
 537     emission_dist = base.ConditionalNormalDistribution
 538   emission = emission_dist(
 539       size=data_size,
 540       hidden_layer_sizes=fcnet_hidden_sizes,
 541       initializers=initializers,
 542       name="generative")
 543   # Construct the proposal distribution.
 544   if proposal_type in ["filtering", "smoothing"]:
 545     proposal = base.NormalApproximatePosterior(
 546         size=latent_size,
 547         hidden_layer_sizes=fcnet_hidden_sizes,
 548         sigma_min=sigma_min,
 549         raw_sigma_bias=raw_sigma_bias,
 550         initializers=initializers,
 551         smoothing=(proposal_type == "smoothing"),
 552         name="approximate_posterior")
 553   else:
 554     proposal = None
 555 
 556   if use_tilt:
 557     tilt = emission_dist(
 558         size=data_size,
 559         hidden_layer_sizes=fcnet_hidden_sizes,
 560         initializers=initializers,
 561         name="tilt")
 562   else:
 563     tilt = None
 564 
 565   rnn_cell = tf.nn.rnn_cell.LSTMCell(rnn_hidden_size,
 566                                      initializer=initializers["w"])
 567   rev_rnn_cell = tf.nn.rnn_cell.LSTMCell(rnn_hidden_size,
 568                                          initializer=initializers["w"])
 569   return TrainableVRNN(
 570       rnn_cell, data_encoder, latent_encoder, transition,
 571       emission, proposal_type, proposal=proposal, rev_rnn_cell=rev_rnn_cell,
 572       tilt=tilt, random_seed=random_seed)
