  38   def run_vrnn(self, generative_class, gt_log_p_x_given_z):
  39     """Tests the VRNN.
  40 
  41     All test values are 'golden values' derived by running the code and copying
  42     the output.
  43 
  44     Args:
  45       generative_class: The class of the generative distribution to use.
  46       gt_log_p_x_given_z: The ground-truth value of log p(x|z).
  47     """
  48     tf.set_random_seed(1234)
  49     with self.test_session() as sess:
  50       batch_size = 2
  51       model, inputs, targets, _ = create_vrnn(generative_class=generative_class,
  52                                               batch_size=batch_size,
  53                                               data_lengths=(1, 1),
  54                                               random_seed=1234)
  55       zero_state = model.zero_state(batch_size=batch_size, dtype=tf.float32)
  56       model.set_observations([inputs, targets], tf.convert_to_tensor([1, 1]))
  57       model_out = model.propose_and_weight(zero_state, 0)
  58       sess.run(tf.global_variables_initializer())
  59       log_alpha, state = sess.run(model_out)
  60       rnn_state, latent_state, rnn_out = state
  61       self.assertAllClose(
  62           rnn_state.c,
  63           [[-0.15014534, 0.0143046, 0.00160489, -0.12899463],
  64            [-0.25015137, 0.09377634, -0.05000039, -0.17123522]])
  65       self.assertAllClose(
  66           rnn_state.h,
  67           [[-0.06842659, 0.00760155, 0.00096106, -0.05434214],
  68            [-0.1109542, 0.0441804, -0.03121299, -0.07882939]]
  69       )
  70       self.assertAllClose(
  71           latent_state,
  72           [[0.025241, 0.122011, 1.066661, 0.316209, -0.25369, 0.108215,
  73             -1.501128, -0.440111, -0.40447, -0.156649, 1.206028],
  74            [0.066824, 0.519937, 0.610973, 0.977739, -0.121889, -0.223429,
  75             -0.32687, -0.578763, -0.56965, 0.751886, 0.681606]]
  76       )
  77       self.assertAllClose(rnn_out, [[-0.068427, 0.007602, 0.000961, -0.054342],
  78                                     [-0.110954, 0.04418, -0.031213, -0.078829]])
  79       gt_log_q_z = [-8.0895052, -6.75819111]
  80       gt_log_p_z = [-7.246827, -6.512877]
  81       gt_log_alpha = (np.array(gt_log_p_z) +
  82                       np.array(gt_log_p_x_given_z) -
  83                       np.array(gt_log_q_z))
  84       self.assertAllClose(log_alpha, gt_log_alpha)

  92   def run_vrnn_with_tilt(self, generative_class, gt_log_alpha):
  93     """Tests the VRNN with a tilting function.
  94 
  95     All test values are 'golden values' derived by running the code and copying
  96     the output.
  97 
  98     Args:
  99       generative_class: The class of the generative distribution to use.
 100       gt_log_alpha: The ground-truth value of log alpha.
 101     """
 102     tf.set_random_seed(1234)
 103     with self.test_session() as sess:
 104       batch_size = 2
 105       model, inputs, targets, _ = create_vrnn(generative_class=generative_class,
 106                                               batch_size=batch_size,
 107                                               data_lengths=(3, 2),
 108                                               random_seed=1234,
 109                                               use_tilt=True)
 110       zero_state = model.zero_state(batch_size=batch_size, dtype=tf.float32)
 111       model.set_observations([inputs, targets], tf.convert_to_tensor([3, 2]))
 112       model_out = model.propose_and_weight(zero_state, 0)
 113       sess.run(tf.global_variables_initializer())
 114       log_alpha, state = sess.run(model_out)
 115       rnn_state, latent_state, rnn_out = state
 116       self.assertAllClose(
 117           rnn_state.c,
 118           [[-0.15014534, 0.0143046, 0.00160489, -0.12899463],
 119            [-0.25015137, 0.09377634, -0.05000039, -0.17123522]])
 120       self.assertAllClose(
 121           rnn_state.h,
 122           [[-0.06842659, 0.00760155, 0.00096106, -0.05434214],
 123            [-0.1109542, 0.0441804, -0.03121299, -0.07882939]]
 124       )
 125       self.assertAllClose(
 126           latent_state,
 127           [[0.025241, 0.122011, 1.066661, 0.316209, -0.25369, 0.108215,
 128             -1.501128, -0.440111, -0.40447, -0.156649, 1.206028],
 129            [0.066824, 0.519937, 0.610973, 0.977739, -0.121889, -0.223429,
 130             -0.32687, -0.578763, -0.56965, 0.751886, 0.681606]]
 131       )
 132       self.assertAllClose(rnn_out, [[-0.068427, 0.007602, 0.000961, -0.054342],
 133                                     [-0.110954, 0.04418, -0.031213, -0.078829]])
 134       self.assertAllClose(log_alpha, gt_log_alpha)
