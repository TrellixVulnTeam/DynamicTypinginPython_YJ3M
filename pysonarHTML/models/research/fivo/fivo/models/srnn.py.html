  34 class SRNN(object):
  35   """Implementation of a Stochastic Recurrent Neural Network (SRNN).
  36 
  37   Introduced in "Sequential Neural Models with Stochastic Layers"
  38   by Fraccaro et al. https://arxiv.org/pdf/1605.07571.pdf.
  39 
  40   The SRNN is a sequence model similar to an RNN that uses stochastic latent
  41   variables to improve its representational power. It can be thought of as a
  42   sequential analogue to the variational auto-encoder (VAE).
  43 
  44   The SRNN has a deterministic RNN as its backbone, represented by the
  45   sequence of RNN hidden states h_t. The latent state is conditioned on
  46   the deterministic RNN states and previous latent state. Unlike the VRNN, the
  47   the RNN state is not conditioned on the previous latent state. The latent
  48   states have a Markov structure and it is assumed that
  49   p(z_t | z_{1:t-1}) = p(z_t | z_{t-1}).
  50 
  51   In this implementation of the SRNN the latent state z_t is Gaussian. The
  52   model's prior over z_t (also called the transition distribution) is
  53   distributed as Normal(mu_t, diag(sigma_t^2)) where mu_t and sigma_t are the
  54   mean and standard deviation output from a fully connected network that accepts
  55   the rnn hidden state h_t and previous latent state z_{t-1} as input.
  56 
  57   The emission distribution p(x_t|z_t, h_t) is conditioned on the latent state
  58   z_t as well as the current RNN hidden state h_t via a fully connected network.
  59 
  60   To increase the modeling power of the SRNN, two additional networks are
  61   used to extract features from the data and the latent state. Those networks
  62   are called data_encoder and latent_encoder respectively.
  63 
  64   For an example of how to call the SRNN's methods see sample_step.
  65 
  66   There are a few differences between this exposition and the paper. The main
  67   goal was to be consistent with the VRNN code. A few components are renamed.
  68   The backward RNN for approximating the posterior, g_phi_a in the paper, is the
  69   rev_rnn_cell. The forward RNN that conditions the latent distribution, d in
  70   the paper, is the rnn_cell. The paper doesn't name the NN's that serve as
  71   feature extractors, and we name them here as the data_encoder and
  72   latent_encoder.
  73   """
  74 
  75   def __init__(self,
  76                rnn_cell,
  77                data_encoder,
  78                latent_encoder,
  79                transition,
  80                emission,
  81                random_seed=None):
  82     """Create a SRNN.
  83 
  84     Args:
  85       rnn_cell: A subclass of tf.nn.rnn_cell.RNNCell that will form the
  86         deterministic backbone of the SRNN. The inputs to the RNN will be the
  87         the encoded input of the current timestep, a Tensor of shape
  88         [batch_size, encoded_data_size].
  89       data_encoder: A callable that accepts a batch of data x_t and
  90         'encodes' it, e.g. runs it through a fully connected network. Must
  91         accept as argument the inputs x_t, a Tensor of the shape
  92         [batch_size, data_size] and return a Tensor of shape
  93         [batch_size, encoded_data_size]. This callable will be called multiple
  94         times in the SRNN cell so if scoping is not handled correctly then
  95         multiple copies of the variables in this network could be made. It is
  96         recommended to use a snt.nets.MLP module, which takes care of this for
  97         you.
  98       latent_encoder: A callable that accepts a latent state z_t and
  99         'encodes' it, e.g. runs it through a fully connected network. Must
 100         accept as argument a Tensor of shape [batch_size, latent_size] and
 101         return a Tensor of shape [batch_size, encoded_latent_size].
 102         This callable must also have the property 'output_size' defined,
 103         returning encoded_latent_size.
 104       transition: A callable that implements the transition distribution
 105         p(z_t|h_t, z_t-1). Must accept as argument the previous RNN hidden state
 106         and previous encoded latent state then return a tf.distributions.Normal
 107         distribution conditioned on the input.
 108       emission: A callable that implements the emission distribution
 109         p(x_t|z_t, h_t). Must accept as arguments the encoded latent state
 110         and the RNN hidden state and return a subclass of
 111         tf.distributions.Distribution that can be used to evaluate the logprob
 112         of the targets.
 113       random_seed: The seed for the random ops. Sets the seed for sample_step.
 114     """
 115     self.random_seed = random_seed
 116     self.rnn_cell = rnn_cell
 117     self.data_encoder = data_encoder
 118     self.latent_encoder = latent_encoder
 119     self.encoded_z_size = latent_encoder.output_size
 120     self.state_size = (self.rnn_cell.state_size)
 121     self._transition = transition
 122     self._emission = emission
 123 
 124   def zero_state(self, batch_size, dtype):
 125     """The initial state of the SRNN.
 126 
 127     Contains the initial state of the RNN and the inital encoded latent.
 128 
 129     Args:
 130       batch_size: The batch size.
 131       dtype: The data type of the SRNN.
 132     Returns:
 133       zero_state: The initial state of the SRNN.
 134     """
 135     return SRNNState(
 136         rnn_state=self.rnn_cell.zero_state(batch_size, dtype),
 137         latent_encoded=tf.zeros(
 138             [batch_size, self.latent_encoder.output_size], dtype=dtype))
 139 
 140   def run_rnn(self, prev_rnn_state, inputs):
 141     """Runs the deterministic RNN for one step.
 142 
 143     Args:
 144       prev_rnn_state: The state of the RNN from the previous timestep.
 145       inputs: A Tensor of shape [batch_size, data_size], the current inputs to
 146         the model. Most often this is x_{t-1}, the previous token in the
 147         observation sequence.
 148     Returns:
 149       rnn_out: The output of the RNN.
 150       rnn_state: The new state of the RNN.
 151     """
 152     rnn_inputs = self.data_encoder(tf.to_float(inputs))
 153     rnn_out, rnn_state = self.rnn_cell(rnn_inputs, prev_rnn_state)
 154     return rnn_out, rnn_state
 155 
 156   def transition(self, rnn_out, prev_latent_encoded):
 157     """Computes the transition distribution p(z_t|h_t, z_{t-1}).
 158 
 159     Note that p(z_t | h_t, z_{t-1}) = p(z_t| z_{t-1}, x_{1:t-1})
 160 
 161     Args:
 162       rnn_out: The output of the rnn for the current timestep.
 163       prev_latent_encoded: Float Tensor of shape
 164         [batch_size, encoded_latent_size], the previous latent state z_{t-1}
 165         run through latent_encoder.
 166     Returns:
 167       p(z_t | h_t): A normal distribution with event shape
 168         [batch_size, latent_size].
 169     """
 170     return self._transition(rnn_out, prev_latent_encoded)
 171 
 172   def emission(self, latent, rnn_out):
 173     """Computes the emission distribution p(x_t | z_t, h_t).
 174 
 175     Note that p(x_t | z_t, h_t) = p(x_t | z_t, x_{1:t-1})
 176 
 177     Args:
 178       latent: The stochastic latent state z_t.
 179       rnn_out: The output of the rnn for the current timestep.
 180     Returns:
 181       p(x_t | z_t, h_t): A distribution with event shape
 182         [batch_size, data_size].
 183       latent_encoded: The latent state encoded with latent_encoder. Should be
 184         passed to transition() on the next timestep.
 185     """
 186     latent_encoded = self.latent_encoder(latent)
 187     return self._emission(latent_encoded, rnn_out), latent_encoded
 188 
 189   def sample_step(self, prev_state, inputs, unused_t):
 190     """Samples one output from the model.
 191 
 192     Args:
 193       prev_state: The previous state of the model, a SRNNState containing the
 194         previous rnn state and the previous encoded latent.
 195       inputs: A Tensor of shape [batch_size, data_size], the current inputs to
 196         the model. Most often this is x_{t-1}, the previous token in the
 197         observation sequence.
 198       unused_t: The current timestep. Not used currently.
 199     Returns:
 200       new_state: The next state of the model, a SRNNState.
 201       xt: A float Tensor of shape [batch_size, data_size], an output sampled
 202         from the emission distribution.
 203     """
 204     rnn_out, rnn_state = self.run_rnn(prev_state.rnn_state,
 205                                       inputs)
 206     p_zt = self.transition(rnn_out, prev_state.latent_encoded)
 207     zt = p_zt.sample(seed=self.random_seed)
 208     p_xt_given_zt, latent_encoded = self.emission(zt, rnn_out)
 209     xt = p_xt_given_zt.sample(seed=self.random_seed)
 210     new_state = SRNNState(rnn_state=rnn_state, latent_encoded=latent_encoded)
 211     return new_state, tf.to_float(xt)

 220 class TrainableSRNN(SRNN, base.ELBOTrainableSequenceModel):
 221   """A SRNN subclass with proposals and methods for training and evaluation.
 222 
 223   This class adds proposals used for training with importance-sampling based
 224   methods such as the ELBO. The model can be configured to propose from one
 225   of three proposals: a learned filtering proposal, a learned smoothing
 226   proposal, or the prior (i.e. the transition distribution).
 227 
 228   As described in the SRNN paper, the learned filtering proposal is
 229   parameterized by a fully connected neural network that accepts as input the
 230   current target x_t and the current rnn output h_t. The learned smoothing
 231   proposal is also given the hidden state of an RNN run in reverse over the
 232   inputs, so as to incorporate information about future observations.
 233 
 234   All learned proposals use the 'res_q' parameterization, meaning that instead
 235   of directly producing the mean of z_t, the proposal network predicts the
 236   'residual' from the prior's mean. This is explored more in section 3.3 of
 237   https://arxiv.org/pdf/1605.07571.pdf.
 238 
 239   During training, the latent state z_t is sampled from the proposal and the
 240   reparameterization trick is used to provide low-variance gradients.
 241 
 242   Note that the SRNN paper refers to the proposals as the approximate posterior,
 243   but we match the VRNN convention of referring to it as the encoder.
 244   """
 245 
 246   def __init__(self,
 247                rnn_cell,
 248                data_encoder,
 249                latent_encoder,
 250                transition,
 251                emission,
 252                proposal_type,
 253                proposal=None,
 254                rev_rnn_cell=None,
 255                tilt=None,
 256                random_seed=None):
 257     """Create a trainable RNN.
 258 
 259     Args:
 260       rnn_cell: A subclass of tf.nn.rnn_cell.RNNCell that will form the
 261         deterministic backbone of the SRNN. The inputs to the RNN will be the
 262         the encoded input of the current timestep, a Tensor of shape
 263         [batch_size, encoded_data_size].
 264       data_encoder: A callable that accepts a batch of data x_t and
 265         'encodes' it, e.g. runs it through a fully connected network. Must
 266         accept as argument the inputs x_t, a Tensor of the shape
 267         [batch_size, data_size] and return a Tensor of shape
 268         [batch_size, encoded_data_size]. This callable will be called multiple
 269         times in the SRNN cell so if scoping is not handled correctly then
 270         multiple copies of the variables in this network could be made. It is
 271         recommended to use a snt.nets.MLP module, which takes care of this for
 272         you.
 273       latent_encoder: A callable that accepts a latent state z_t and
 274         'encodes' it, e.g. runs it through a fully connected network. Must
 275         accept as argument a Tensor of shape [batch_size, latent_size] and
 276         return a Tensor of shape [batch_size, encoded_latent_size].
 277         This callable must also have the property 'output_size' defined,
 278         returning encoded_latent_size.
 279       transition: A callable that implements the transition distribution
 280         p(z_t|h_t, z_t-1). Must accept as argument the previous RNN hidden state
 281         and previous encoded latent state then return a tf.distributions.Normal
 282         distribution conditioned on the input.
 283       emission: A callable that implements the emission distribution
 284         p(x_t|z_t, h_t). Must accept as arguments the encoded latent state
 285         and the RNN hidden state and return a subclass of
 286         tf.distributions.Distribution that can be used to evaluate the logprob
 287         of the targets.
 288       proposal_type: A string indicating the type of proposal to use. Can
 289         be either "filtering", "smoothing", or "prior". When proposal_type is
 290         "filtering" or "smoothing", proposal must be provided. When
 291         proposal_type is "smoothing", rev_rnn_cell must also be provided.
 292       proposal: A callable that implements the proposal q(z_t| h_t, x_{1:T}).
 293         If proposal_type is "filtering" then proposal must accept as arguments
 294         the current rnn output, the encoded target of the current timestep,
 295         and the mean of the prior. If proposal_type is "smoothing" then
 296         in addition to the current rnn output and the mean of the prior
 297         proposal must accept as arguments the output of the reverse rnn.
 298         proposal should return a tf.distributions.Normal distribution
 299         conditioned on its inputs. If proposal_type is "prior" this argument is
 300         ignored.
 301       rev_rnn_cell: A subclass of tf.nn.rnn_cell.RNNCell that will aggregate
 302         forward rnn outputs in the reverse direction. The inputs to the RNN
 303         will be the encoded reverse input of the current timestep, a Tensor of
 304         shape [batch_size, encoded_data_size].
 305       tilt: A callable that implements the log of a positive tilting function
 306         (ideally approximating log p(x_{t+1}|z_t, h_t). Must accept as arguments
 307         the encoded latent state and the RNN hidden state and return a subclass
 308         of tf.distributions.Distribution that can be used to evaluate the
 309         logprob of x_{t+1}. Optionally, None and then no tilt is used.
 310       random_seed: The seed for the random ops. Sets the seed for sample_step
 311         and __call__.
 312     """
 313     super(TrainableSRNN, self).__init__(
 314         rnn_cell, data_encoder, latent_encoder,
 315         transition, emission, random_seed=random_seed)
 316     self.rev_rnn_cell = rev_rnn_cell
 317     self._tilt = tilt
 318     assert proposal_type in ["filtering", "smoothing", "prior"]
 319     self._proposal = proposal
 320     self.proposal_type = proposal_type
 321     if proposal_type != "prior":
 322       assert proposal, "If not proposing from the prior, must provide proposal."
 323     if proposal_type == "smoothing":
 324       assert rev_rnn_cell, "Must provide rev_rnn_cell for smoothing proposal."
 325 
 326   def zero_state(self, batch_size, dtype):
 327     super_state = super(TrainableSRNN, self).zero_state(batch_size, dtype)
 328     return TrainableSRNNState(
 329         rnn_out=tf.zeros([batch_size, self.rnn_cell.output_size], dtype=dtype),
 330         **super_state._asdict())
 331 
 332   def set_observations(self, observations, seq_lengths):
 333     """Stores the model's observations.
 334 
 335     Stores the observations (inputs and targets) in TensorArrays and precomputes
 336     things for later like the reverse RNN output and encoded targets.
 337 
 338     Args:
 339       observations: The observations of the model, a tuple containing two
 340         Tensors of shape [max_seq_len, batch_size, data_size]. The Tensors
 341         should be the inputs and targets, respectively.
 342       seq_lengths: An int Tensor of shape [batch_size] containing the length
 343         of each sequence in observations.
 344     """
 345     inputs, targets = observations
 346     self.seq_lengths = seq_lengths
 347     self.max_seq_len = tf.reduce_max(seq_lengths)
 348     self.targets_ta = base.ta_for_tensor(targets, clear_after_read=False)
 349     targets_encoded = base.encode_all(targets, self.data_encoder)
 350     self.targets_encoded_ta = base.ta_for_tensor(targets_encoded,
 351                                                  clear_after_read=False)
 352     inputs_encoded = base.encode_all(inputs, self.data_encoder)
 353     rnn_out, _ = tf.nn.dynamic_rnn(self.rnn_cell,
 354                                    inputs_encoded,
 355                                    time_major=True,
 356                                    dtype=tf.float32,
 357                                    scope="forward_rnn")
 358     self.rnn_ta = base.ta_for_tensor(rnn_out,
 359                                      clear_after_read=False)
 360     if self.rev_rnn_cell:
 361       targets_and_rnn_out = tf.concat([rnn_out, targets_encoded], 2)
 362       reversed_targets_and_rnn_out = tf.reverse_sequence(
 363           targets_and_rnn_out, seq_lengths, seq_axis=0, batch_axis=1)
 364       # Compute the reverse rnn over the targets.
 365       reverse_rnn_out, _ = tf.nn.dynamic_rnn(self.rev_rnn_cell,
 366                                              reversed_targets_and_rnn_out,
 367                                              time_major=True,
 368                                              dtype=tf.float32,
 369                                              scope="reverse_rnn")
 370       reverse_rnn_out = tf.reverse_sequence(reverse_rnn_out, seq_lengths,
 371                                             seq_axis=0, batch_axis=1)
 372       self.reverse_rnn_ta = base.ta_for_tensor(reverse_rnn_out,
 373                                                clear_after_read=False)
 374 
 375   def _filtering_proposal(self, rnn_out, prev_latent_encoded, prior, t):
 376     """Computes the filtering proposal distribution."""
 377     return self._proposal(rnn_out,
 378                           prev_latent_encoded,
 379                           self.targets_encoded_ta.read(t),
 380                           prior_mu=prior.mean())
 381 
 382   def _smoothing_proposal(self, rnn_out, prev_latent_encoded, prior, t):
 383     """Computes the smoothing proposal distribution."""
 384     return self._proposal(rnn_out,
 385                           prev_latent_encoded,
 386                           smoothing_tensors=[self.reverse_rnn_ta.read(t)],
 387                           prior_mu=prior.mean())
 388 
 389   def proposal(self, rnn_out, prev_latent_encoded, prior, t):
 390     """Computes the proposal distribution specified by proposal_type.
 391 
 392     Args:
 393       rnn_out: The output of the rnn for the current timestep.
 394       prev_latent_encoded: Float Tensor of shape
 395         [batch_size, encoded_latent_size], the previous latent state z_{t-1}
 396         run through latent_encoder.
 397       prior: A tf.distributions.Normal distribution representing the prior
 398         over z_t, p(z_t | z_{1:t-1}, x_{1:t-1}). Used for 'res_q'.
 399       t: A scalar int Tensor, the current timestep.
 400     """
 401     if self.proposal_type == "filtering":
 402       return self._filtering_proposal(rnn_out, prev_latent_encoded, prior, t)
 403     elif self.proposal_type == "smoothing":
 404       return self._smoothing_proposal(rnn_out, prev_latent_encoded, prior, t)
 405     elif self.proposal_type == "prior":
 406       return self.transition(rnn_out, prev_latent_encoded)
 407 
 408   def tilt(self, rnn_out, latent_encoded, targets):
 409     r_func = self._tilt(rnn_out, latent_encoded)
 410     return tf.reduce_sum(r_func.log_prob(targets), axis=-1)
 411 
 412   def propose_and_weight(self, state, t):
 413     """Runs the model and computes importance weights for one timestep.
 414 
 415     Runs the model and computes importance weights, sampling from the proposal
 416     instead of the transition/prior.
 417 
 418     Args:
 419       state: The previous state of the model, a TrainableSRNNState containing
 420         the previous rnn state, the previous rnn outs, and the previous encoded
 421         latent.
 422       t: A scalar integer Tensor, the current timestep.
 423     Returns:
 424       weights: A float Tensor of shape [batch_size].
 425       new_state: The new state of the model.
 426     """
 427     targets = self.targets_ta.read(t)
 428     rnn_out = self.rnn_ta.read(t)
 429     p_zt = self.transition(rnn_out, state.latent_encoded)
 430     q_zt = self.proposal(rnn_out, state.latent_encoded, p_zt, t)
 431     zt = q_zt.sample(seed=self.random_seed)
 432     p_xt_given_zt, latent_encoded = self.emission(zt, rnn_out)
 433     log_p_xt_given_zt = tf.reduce_sum(p_xt_given_zt.log_prob(targets), axis=-1)
 434     log_p_zt = tf.reduce_sum(p_zt.log_prob(zt), axis=-1)
 435     log_q_zt = tf.reduce_sum(q_zt.log_prob(zt), axis=-1)
 436     weights = log_p_zt + log_p_xt_given_zt - log_q_zt
 437     if self._tilt:
 438       prev_log_r = tf.cond(
 439           tf.greater(t, 0),
 440           lambda: self.tilt(state.rnn_out, state.latent_encoded, targets),
 441           lambda: 0.)  # On the first step, prev_log_r = 0.
 442       log_r = tf.cond(
 443           tf.less(t + 1, self.max_seq_len),
 444           lambda: self.tilt(rnn_out, latent_encoded, self.targets_ta.read(t+1)),
 445           lambda: 0.)
 446       # On the last step, log_r = 0.
 447       log_r *= tf.to_float(t < self.seq_lengths - 1)
 448       weights += log_r - prev_log_r
 449 
 450     # This reshape is required because the TensorArray reports different shapes
 451     # than the initial state provides (where the first dimension is unknown).
 452     # The difference breaks the while_loop. Reshape prevents the error.
 453     rnn_out = tf.reshape(rnn_out, tf.shape(state.rnn_out))
 454 
 455     new_state = TrainableSRNNState(rnn_out=rnn_out,
 456                                    rnn_state=state.rnn_state,  # unmodified
 457                                    latent_encoded=latent_encoded)
 458     return weights, new_state

 465 def create_srnn(
 466     data_size,
 467     latent_size,
 468     emission_class,
 469     rnn_hidden_size=None,
 470     fcnet_hidden_sizes=None,
 471     encoded_data_size=None,
 472     encoded_latent_size=None,
 473     sigma_min=0.0,
 474     raw_sigma_bias=0.25,
 475     emission_bias_init=0.0,
 476     use_tilt=False,
 477     proposal_type="filtering",
 478     initializers=None,
 479     random_seed=None):
 480   """A factory method for creating SRNN cells.
 481 
 482   Args:
 483     data_size: The dimension of the vectors that make up the data sequences.
 484     latent_size: The size of the stochastic latent state of the SRNN.
 485     emission_class: The class of the emission distribution. Can be either
 486       ConditionalNormalDistribution or ConditionalBernoulliDistribution.
 487     rnn_hidden_size: The hidden state dimension of the RNN that forms the
 488       deterministic part of this SRNN. If None, then it defaults
 489       to latent_size.
 490     fcnet_hidden_sizes: A list of python integers, the size of the hidden
 491       layers of the fully connected networks that parameterize the conditional
 492       distributions of the SRNN. If None, then it defaults to one hidden
 493       layer of size latent_size.
 494     encoded_data_size: The size of the output of the data encoding network. If
 495       None, defaults to latent_size.
 496     encoded_latent_size: The size of the output of the latent state encoding
 497       network. If None, defaults to latent_size.
 498     sigma_min: The minimum value that the standard deviation of the
 499       distribution over the latent state can take.
 500     raw_sigma_bias: A scalar that is added to the raw standard deviation
 501       output from the neural networks that parameterize the prior and
 502       approximate posterior. Useful for preventing standard deviations close
 503       to zero.
 504     emission_bias_init: A bias to added to the raw output of the fully
 505       connected network that parameterizes the emission distribution. Useful
 506       for initalizing the mean of the distribution to a sensible starting point
 507       such as the mean of the training data. Only used with Bernoulli generative
 508       distributions.
 509     use_tilt: If true, create a SRNN with a tilting function.
 510     proposal_type: The type of proposal to use. Can be "filtering", "smoothing",
 511       or "prior".
 512     initializers: The variable intitializers to use for the fully connected
 513       networks and RNN cell. Must be a dictionary mapping the keys 'w' and 'b'
 514       to the initializers for the weights and biases. Defaults to xavier for
 515       the weights and zeros for the biases when initializers is None.
 516     random_seed: A random seed for the SRNN resampling operations.
 517   Returns:
 518     model: A TrainableSRNN object.
 519   """
 520   if rnn_hidden_size is None:
 521     rnn_hidden_size = latent_size
 522   if fcnet_hidden_sizes is None:
 523     fcnet_hidden_sizes = [latent_size]
 524   if encoded_data_size is None:
 525     encoded_data_size = latent_size
 526   if encoded_latent_size is None:
 527     encoded_latent_size = latent_size
 528   if initializers is None:
 529     initializers = _DEFAULT_INITIALIZERS
 530   data_encoder = snt.nets.MLP(
 531       output_sizes=fcnet_hidden_sizes + [encoded_data_size],
 532       initializers=initializers,
 533       name="data_encoder")
 534   latent_encoder = snt.nets.MLP(
 535       output_sizes=fcnet_hidden_sizes + [encoded_latent_size],
 536       initializers=initializers,
 537       name="latent_encoder")
 538   transition = base.ConditionalNormalDistribution(
 539       size=latent_size,
 540       hidden_layer_sizes=fcnet_hidden_sizes,
 541       sigma_min=sigma_min,
 542       raw_sigma_bias=raw_sigma_bias,
 543       initializers=initializers,
 544       name="prior")
 545   # Construct the emission distribution.
 546   if emission_class == base.ConditionalBernoulliDistribution:
 547     # For Bernoulli distributed outputs, we initialize the bias so that the
 548     # network generates on average the mean from the training set.
 549     emission_dist = functools.partial(base.ConditionalBernoulliDistribution,
 550                                       bias_init=emission_bias_init)
 551   else:
 552     emission_dist = base.ConditionalNormalDistribution
 553   emission = emission_dist(
 554       size=data_size,
 555       hidden_layer_sizes=fcnet_hidden_sizes,
 556       initializers=initializers,
 557       name="generative")
 558   # Construct the proposal distribution.
 559   if proposal_type in ["filtering", "smoothing"]:
 560     proposal = base.NormalApproximatePosterior(
 561         size=latent_size,
 562         hidden_layer_sizes=fcnet_hidden_sizes,
 563         sigma_min=sigma_min,
 564         raw_sigma_bias=raw_sigma_bias,
 565         initializers=initializers,
 566         smoothing=(proposal_type == "smoothing"),
 567         name="approximate_posterior")
 568   else:
 569     proposal = None
 570 
 571   if use_tilt:
 572     tilt = emission_dist(
 573         size=data_size,
 574         hidden_layer_sizes=fcnet_hidden_sizes,
 575         initializers=initializers,
 576         name="tilt")
 577   else:
 578     tilt = None
 579 
 580   rnn_cell = tf.nn.rnn_cell.LSTMCell(rnn_hidden_size,
 581                                      initializer=initializers["w"])
 582   rev_rnn_cell = tf.nn.rnn_cell.LSTMCell(rnn_hidden_size,
 583                                          initializer=initializers["w"])
 584   return TrainableSRNN(
 585       rnn_cell, data_encoder, latent_encoder, transition,
 586       emission, proposal_type, proposal=proposal, rev_rnn_cell=rev_rnn_cell,
 587       tilt=tilt, random_seed=random_seed)
