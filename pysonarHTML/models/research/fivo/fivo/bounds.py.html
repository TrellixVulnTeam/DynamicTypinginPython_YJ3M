  33 def iwae(model,
  34          observations,
  35          seq_lengths,
  36          num_samples=1,
  37          parallel_iterations=30,
  38          swap_memory=True):
  39   """Computes the IWAE lower bound on the log marginal probability.
  40 
  41   This method accepts a stochastic latent variable model and some observations
  42   and computes a stochastic lower bound on the log marginal probability of the
  43   observations. The IWAE estimator is defined by averaging multiple importance
  44   weights. For more details see "Importance Weighted Autoencoders" by Burda
  45   et al. https://arxiv.org/abs/1509.00519.
  46 
  47   When num_samples = 1, this bound becomes the evidence lower bound (ELBO).
  48 
  49   Args:
  50     model: A subclass of ELBOTrainableSequenceModel that implements one
  51       timestep of the model. See models/vrnn.py for an example.
  52     observations: The inputs to the model. A potentially nested list or tuple of
  53       Tensors each of shape [max_seq_len, batch_size, ...]. The Tensors must
  54       have a rank at least two and have matching shapes in the first two
  55       dimensions, which represent time and the batch respectively. The model
  56       will be provided with the observations before computing the bound.
  57     seq_lengths: A [batch_size] Tensor of ints encoding the length of each
  58       sequence in the batch (sequences can be padded to a common length).
  59     num_samples: The number of samples to use.
  60     parallel_iterations: The number of parallel iterations to use for the
  61       internal while loop.
  62     swap_memory: Whether GPU-CPU memory swapping should be enabled for the
  63       internal while loop.
  64 
  65   Returns:
  66     log_p_hat: A Tensor of shape [batch_size] containing IWAE's estimate of the
  67       log marginal probability of the observations.
  68     log_weights: A Tensor of shape [max_seq_len, batch_size, num_samples]
  69       containing the log weights at each timestep. Will not be valid for
  70       timesteps past the end of a sequence.
  71   """
  72   log_p_hat, log_weights, _, final_state = fivo(
  73       model,
  74       observations,
  75       seq_lengths,
  76       num_samples=num_samples,
  77       resampling_criterion=smc.never_resample_criterion,
  78       parallel_iterations=parallel_iterations,
  79       swap_memory=swap_memory)
  80   return log_p_hat, log_weights, final_state

  83 def fivo(model,
  84          observations,
  85          seq_lengths,
  86          num_samples=1,
  87          resampling_criterion=smc.ess_criterion,
  88          resampling_type='multinomial',
  89          relaxed_resampling_temperature=0.5,
  90          parallel_iterations=30,
  91          swap_memory=True,
  92          random_seed=None):
  93   """Computes the FIVO lower bound on the log marginal probability.
  94 
  95   This method accepts a stochastic latent variable model and some observations
  96   and computes a stochastic lower bound on the log marginal probability of the
  97   observations. The lower bound is defined by a particle filter's unbiased
  98   estimate of the marginal probability of the observations. For more details see
  99   "Filtering Variational Objectives" by Maddison et al.
 100   https://arxiv.org/abs/1705.09279.
 101 
 102   When the resampling criterion is "never resample", this bound becomes IWAE.
 103 
 104   Args:
 105     model: A subclass of ELBOTrainableSequenceModel that implements one
 106       timestep of the model. See models/vrnn.py for an example.
 107     observations: The inputs to the model. A potentially nested list or tuple of
 108       Tensors each of shape [max_seq_len, batch_size, ...]. The Tensors must
 109       have a rank at least two and have matching shapes in the first two
 110       dimensions, which represent time and the batch respectively. The model
 111       will be provided with the observations before computing the bound.
 112     seq_lengths: A [batch_size] Tensor of ints encoding the length of each
 113       sequence in the batch (sequences can be padded to a common length).
 114     num_samples: The number of particles to use in each particle filter.
 115     resampling_criterion: The resampling criterion to use for this particle
 116       filter. Must accept the number of samples, the current log weights,
 117       and the current timestep and return a boolean Tensor of shape [batch_size]
 118       indicating whether each particle filter should resample. See
 119       ess_criterion and related functions for examples. When
 120       resampling_criterion is never_resample_criterion, resampling_fn is ignored
 121       and never called.
 122     resampling_type: The type of resampling, one of "multinomial" or "relaxed".
 123     relaxed_resampling_temperature: A positive temperature only used for relaxed
 124       resampling.
 125     parallel_iterations: The number of parallel iterations to use for the
 126       internal while loop. Note that values greater than 1 can introduce
 127       non-determinism even when random_seed is provided.
 128     swap_memory: Whether GPU-CPU memory swapping should be enabled for the
 129       internal while loop.
 130     random_seed: The random seed to pass to the resampling operations in
 131       the particle filter. Mainly useful for testing.
 132 
 133   Returns:
 134     log_p_hat: A Tensor of shape [batch_size] containing FIVO's estimate of the
 135       log marginal probability of the observations.
 136     log_weights: A Tensor of shape [max_seq_len, batch_size, num_samples]
 137       containing the log weights at each timestep of the particle filter. Note
 138       that on timesteps when a resampling operation is performed the log weights
 139       are reset to 0. Will not be valid for timesteps past the end of a
 140       sequence.
 141     resampled: A Tensor of shape [max_seq_len, batch_size] indicating when the
 142       particle filters resampled. Will be 1.0 on timesteps when resampling
 143       occurred and 0.0 on timesteps when it did not.
 144   """
 145   # batch_size is the number of particle filters running in parallel.
 146   batch_size = tf.shape(seq_lengths)[0]
 147 
 148   # Each sequence in the batch will be the input data for a different
 149   # particle filter. The batch will be laid out as:
 150   #   particle 1 of particle filter 1
 151   #   particle 1 of particle filter 2
 152   #   ...
 153   #   particle 1 of particle filter batch_size
 154   #   particle 2 of particle filter 1
 155   #   ...
 156   #   particle num_samples of particle filter batch_size
 157   observations = nested.tile_tensors(observations, [1, num_samples])
 158   tiled_seq_lengths = tf.tile(seq_lengths, [num_samples])
 159   model.set_observations(observations, tiled_seq_lengths)
 160 
 161   if resampling_type == 'multinomial':
 162     resampling_fn = smc.multinomial_resampling
 163   elif resampling_type == 'relaxed':
 164     resampling_fn = functools.partial(
 165         smc.relaxed_resampling, temperature=relaxed_resampling_temperature)
 166   resampling_fn = functools.partial(resampling_fn, random_seed=random_seed)
 167 
 168   def transition_fn(prev_state, t):
 169     if prev_state is None:
 170       return model.zero_state(batch_size * num_samples, tf.float32)
 171     return model.propose_and_weight(prev_state, t)
 172 
 173   log_p_hat, log_weights, resampled, final_state, _ = smc.smc(
 174       transition_fn,
 175       seq_lengths,
 176       num_particles=num_samples,
 177       resampling_criterion=resampling_criterion,
 178       resampling_fn=resampling_fn,
 179       parallel_iterations=parallel_iterations,
 180       swap_memory=swap_memory)
 181 
 182   return log_p_hat, log_weights, resampled, final_state

 184 def fivo_aux_td(
 185     model,
 186     observations,
 187     seq_lengths,
 188     num_samples=1,
 189     resampling_criterion=smc.ess_criterion,
 190     resampling_type='multinomial',
 191     relaxed_resampling_temperature=0.5,
 192     parallel_iterations=30,
 193     swap_memory=True,
 194     random_seed=None):
 195   """Experimental."""
 196   # batch_size is the number of particle filters running in parallel.
 197   batch_size = tf.shape(seq_lengths)[0]
 198   max_seq_len = tf.reduce_max(seq_lengths)
 199 
 200   # Each sequence in the batch will be the input data for a different
 201   # particle filter. The batch will be laid out as:
 202   #   particle 1 of particle filter 1
 203   #   particle 1 of particle filter 2
 204   #   ...
 205   #   particle 1 of particle filter batch_size
 206   #   particle 2 of particle filter 1
 207   #   ...
 208   #   particle num_samples of particle filter batch_size
 209   observations = nested.tile_tensors(observations, [1, num_samples])
 210   tiled_seq_lengths = tf.tile(seq_lengths, [num_samples])
 211   model.set_observations(observations, tiled_seq_lengths)
 212 
 213   if resampling_type == 'multinomial':
 214     resampling_fn = smc.multinomial_resampling
 215   elif resampling_type == 'relaxed':
 216     resampling_fn = functools.partial(
 217         smc.relaxed_resampling, temperature=relaxed_resampling_temperature)
 218   resampling_fn = functools.partial(resampling_fn, random_seed=random_seed)
 219 
 220   def transition_fn(prev_state, t):
 221     if prev_state is None:
 222       model_init_state = model.zero_state(batch_size * num_samples, tf.float32)
 223       return (tf.zeros([num_samples*batch_size], dtype=tf.float32),
 224               (tf.zeros([num_samples*batch_size, model.latent_size], dtype=tf.float32),
 225                tf.zeros([num_samples*batch_size, model.latent_size], dtype=tf.float32)),
 226               model_init_state)
 227 
 228     prev_log_r, prev_log_r_tilde, prev_model_state = prev_state
 229     (new_model_state, zt, log_q_zt, log_p_zt,
 230      log_p_x_given_z, log_r_tilde, p_ztplus1) = model(prev_model_state, t)
 231     r_tilde_mu, r_tilde_sigma_sq = log_r_tilde
 232     # Compute the weight without r.
 233     log_weight = log_p_zt + log_p_x_given_z - log_q_zt
 234     # Compute log_r and log_r_tilde.
 235     p_mu = tf.stop_gradient(p_ztplus1.mean())
 236     p_sigma_sq = tf.stop_gradient(p_ztplus1.variance())
 237     log_r = (tf.log(r_tilde_sigma_sq) -
 238              tf.log(r_tilde_sigma_sq + p_sigma_sq) -
 239              tf.square(r_tilde_mu - p_mu)/(r_tilde_sigma_sq + p_sigma_sq))
 240     # log_r is [num_samples*batch_size, latent_size]. We sum it along the last
 241     # dimension to compute log r.
 242     log_r = 0.5*tf.reduce_sum(log_r, axis=-1)
 243     # Compute prev log r tilde
 244     prev_r_tilde_mu, prev_r_tilde_sigma_sq = prev_log_r_tilde
 245     prev_log_r_tilde = -0.5*tf.reduce_sum(
 246         tf.square(tf.stop_gradient(zt) - prev_r_tilde_mu)/prev_r_tilde_sigma_sq, axis=-1)
 247     # If the sequence is on the last timestep, log_r and log_r_tilde are just zeros.
 248     last_timestep = t >= (tiled_seq_lengths - 1)
 249     log_r = tf.where(last_timestep,
 250                      tf.zeros_like(log_r),
 251                      log_r)
 252     prev_log_r_tilde = tf.where(last_timestep,
 253                                 tf.zeros_like(prev_log_r_tilde),
 254                                 prev_log_r_tilde)
 255     log_weight += tf.stop_gradient(log_r - prev_log_r)
 256     new_state = (log_r, log_r_tilde, new_model_state)
 257     loop_fn_args = (log_r, prev_log_r_tilde, log_p_x_given_z, log_r - prev_log_r)
 258     return log_weight, new_state, loop_fn_args
 259 
 260   def loop_fn(loop_state, loop_args, unused_model_state, log_weights, resampled, mask, t):
 261     if loop_state is None:
 262       return (tf.zeros([batch_size], dtype=tf.float32),
 263               tf.zeros([batch_size], dtype=tf.float32),
 264               tf.zeros([num_samples, batch_size], dtype=tf.float32))
 265     log_p_hat_acc, bellman_loss_acc, log_r_diff_acc = loop_state
 266     log_r, prev_log_r_tilde, log_p_x_given_z, log_r_diff = loop_args
 267     # Compute the log_p_hat update
 268     log_p_hat_update = tf.reduce_logsumexp(
 269         log_weights, axis=0) - tf.log(tf.to_float(num_samples))
 270     # If it is the last timestep, we always add the update.
 271     log_p_hat_acc += tf.cond(t >= max_seq_len-1,
 272                              lambda: log_p_hat_update,
 273                              lambda: log_p_hat_update * resampled)
 274     # Compute the Bellman update.
 275     log_r = tf.reshape(log_r, [num_samples, batch_size])
 276     prev_log_r_tilde = tf.reshape(prev_log_r_tilde, [num_samples, batch_size])
 277     log_p_x_given_z = tf.reshape(log_p_x_given_z, [num_samples, batch_size])
 278     mask = tf.reshape(mask, [num_samples, batch_size])
 279     # On the first timestep there is no bellman error because there is no
 280     # prev_log_r_tilde.
 281     mask = tf.cond(tf.equal(t, 0),
 282                    lambda: tf.zeros_like(mask),
 283                    lambda: mask)
 284     # On the first timestep also fix up prev_log_r_tilde, which will be -inf.
 285     prev_log_r_tilde = tf.where(
 286         tf.is_inf(prev_log_r_tilde),
 287         tf.zeros_like(prev_log_r_tilde),
 288         prev_log_r_tilde)
 289     # log_lambda is [num_samples, batch_size]
 290     log_lambda = tf.reduce_mean(prev_log_r_tilde - log_p_x_given_z - log_r,
 291                                 axis=0, keepdims=True)
 292     bellman_error = mask * tf.square(
 293         prev_log_r_tilde -
 294         tf.stop_gradient(log_lambda + log_p_x_given_z + log_r)
 295     )
 296     bellman_loss_acc += tf.reduce_mean(bellman_error, axis=0)
 297     # Compute the log_r_diff update
 298     log_r_diff_acc += mask * tf.reshape(log_r_diff, [num_samples, batch_size])
 299     return (log_p_hat_acc, bellman_loss_acc, log_r_diff_acc)
 300 
 301   log_weights, resampled, accs = smc.smc(
 302       transition_fn,
 303       seq_lengths,
 304       num_particles=num_samples,
 305       resampling_criterion=resampling_criterion,
 306       resampling_fn=resampling_fn,
 307       loop_fn=loop_fn,
 308       parallel_iterations=parallel_iterations,
 309       swap_memory=swap_memory)
 310 
 311   log_p_hat, bellman_loss, log_r_diff = accs
 312   loss_per_seq = [- log_p_hat, bellman_loss]
 313   tf.summary.scalar("bellman_loss",
 314                     tf.reduce_mean(bellman_loss / tf.to_float(seq_lengths)))
 315   tf.summary.scalar("log_r_diff",
 316                     tf.reduce_mean(tf.reduce_mean(log_r_diff, axis=0) / tf.to_float(seq_lengths)))
 317   return loss_per_seq, log_p_hat, log_weights, resampled
