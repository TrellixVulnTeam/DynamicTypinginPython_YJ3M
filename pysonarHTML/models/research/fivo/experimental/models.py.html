  38 class Q(object):
  39 
  40   def __init__(self,
  41                state_size,
  42                num_timesteps,
  43                sigma_min=1e-5,
  44                dtype=tf.float32,
  45                random_seed=None,
  46                init_mu0_to_zero=False,
  47                graph_collection_name="Q_VARS"):
  48     self.sigma_min = sigma_min
  49     self.dtype = dtype
  50     self.graph_collection_name = graph_collection_name
  51     initializers = []
  52     for t in xrange(num_timesteps):
  53       if t == 0 and init_mu0_to_zero:
  54         initializers.append(
  55             {"w": tf.zeros_initializer, "b": tf.zeros_initializer})
  56       else:
  57         initializers.append(
  58             {"w": tf.random_uniform_initializer(seed=random_seed),
  59              "b": tf.zeros_initializer})
  60 
  61     def custom_getter(getter, *args, **kwargs):
  62       out = getter(*args, **kwargs)
  63       ref = tf.get_collection_ref(self.graph_collection_name)
  64       if out not in ref:
  65         ref.append(out)
  66       return out
  67 
  68     self.mus = [
  69         snt.Linear(output_size=state_size,
  70                    initializers=initializers[t],
  71                    name="q_mu_%d" % t,
  72                    custom_getter=custom_getter
  73                   )
  74         for t in xrange(num_timesteps)
  75     ]
  76     self.sigmas = [
  77         tf.get_variable(
  78             shape=[state_size],
  79             dtype=self.dtype,
  80             name="q_sigma_%d" % (t + 1),
  81             collections=[tf.GraphKeys.GLOBAL_VARIABLES, graph_collection_name],
  82             initializer=tf.random_uniform_initializer(seed=random_seed))
  83         for t in xrange(num_timesteps)
  84     ]
  85 
  86   def q_zt(self, observation, prev_state, t):
  87     batch_size = tf.shape(prev_state)[0]
  88     q_mu = self.mus[t](tf.concat([observation, prev_state], axis=1))
  89     q_sigma = tf.maximum(tf.nn.softplus(self.sigmas[t]), self.sigma_min)
  90     q_sigma = tf.tile(q_sigma[tf.newaxis, :], [batch_size, 1])
  91     q_zt = tf.contrib.distributions.Normal(loc=q_mu, scale=tf.sqrt(q_sigma))
  92     return q_zt
  93 
  94   def summarize_weights(self):
  95     for t, sigma in enumerate(self.sigmas):
  96       tf.summary.scalar("q_sigma/%d" % t, sigma[0])
  97     for t, f in enumerate(self.mus):
  98       tf.summary.scalar("q_mu/b_%d" % t, f.b[0])
  99       tf.summary.scalar("q_mu/w_obs_%d" % t, f.w[0,0])
 100       if t != 0:
 101         tf.summary.scalar("q_mu/w_prev_state_%d" % t, f.w[1,0])

 141 
 142   def __init__(self,
 143                state_size,
 144                num_timesteps,
 145                sigma_min=1e-5,
 146                dtype=tf.float32,
 147                random_seed=None,
 148                init_mu0_to_zero=False,
 149                graph_collection_name="Q_VARS"):
 150     self.sigma_min = sigma_min
 151     self.dtype = dtype
 152     self.graph_collection_name = graph_collection_name
 153     initializers = []
 154     for t in xrange(num_timesteps):
 155       if t == 0 and init_mu0_to_zero:
 156         initializers.append(tf.zeros_initializer)
 157       else:
 158         initializers.append(tf.random_uniform_initializer(seed=random_seed))
 159 
 160     self.mus = [
 161         tf.get_variable(
 162             shape=[state_size],
 163             dtype=self.dtype,
 164             name="q_mu_%d" % (t + 1),
 165             collections=[tf.GraphKeys.GLOBAL_VARIABLES, graph_collection_name],
 166             initializer=initializers[t])
 167         for t in xrange(num_timesteps)
 168     ]
 169     self.sigmas = [
 170         tf.get_variable(
 171             shape=[state_size],
 172             dtype=self.dtype,
 173             name="q_sigma_%d" % (t + 1),
 174             collections=[tf.GraphKeys.GLOBAL_VARIABLES, graph_collection_name],
 175             initializer=tf.random_uniform_initializer(seed=random_seed))
 176         for t in xrange(num_timesteps)
 177     ]
 178 
 179   def q_zt(self, unused_observation, prev_state, t):
 180     batch_size = tf.shape(prev_state)[0]
 181     q_mu = tf.tile(self.mus[t][tf.newaxis, :], [batch_size, 1])
 182     q_sigma = tf.maximum(tf.nn.softplus(self.sigmas[t]), self.sigma_min)
 183     q_sigma = tf.tile(q_sigma[tf.newaxis, :], [batch_size, 1])
 184     q_zt = tf.contrib.distributions.Normal(loc=q_mu, scale=tf.sqrt(q_sigma))
 185     return q_zt
 186

 189       tf.summary.scalar("q_sigma/%d" % t, sigma[0])
 190     for t, f in enumerate(self.mus):
 191       tf.summary.scalar("q_mu/%d" % t, f[0])

 194 class R(object):
 195 
 196   def __init__(self,
 197                state_size,
 198                num_timesteps,
 199                sigma_min=1e-5,
 200                dtype=tf.float32,
 201                sigma_init=1.,
 202                random_seed=None,
 203                graph_collection_name="R_VARS"):
 204     self.dtype = dtype
 205     self.sigma_min = sigma_min
 206     initializers = {"w": tf.truncated_normal_initializer(seed=random_seed),
 207                     "b": tf.zeros_initializer}
 208     self.graph_collection_name=graph_collection_name
 209 
 210     def custom_getter(getter, *args, **kwargs):
 211       out = getter(*args, **kwargs)
 212       ref = tf.get_collection_ref(self.graph_collection_name)
 213       if out not in ref:
 214         ref.append(out)
 215       return out
 216 
 217     self.mus= [
 218         snt.Linear(output_size=state_size,
 219                    initializers=initializers,
 220                    name="r_mu_%d" % t,
 221                    custom_getter=custom_getter)
 222         for t in xrange(num_timesteps)
 223     ]
 224 
 225     self.sigmas = [
 226         tf.get_variable(
 227             shape=[state_size],
 228             dtype=self.dtype,
 229             name="r_sigma_%d" % (t + 1),
 230             collections=[tf.GraphKeys.GLOBAL_VARIABLES, graph_collection_name],
 231             #initializer=tf.random_uniform_initializer(seed=random_seed, maxval=100))
 232             initializer=tf.constant_initializer(sigma_init))
 233         for t in xrange(num_timesteps)
 234     ]
 235 
 236   def r_xn(self, z_t, t):
 237     batch_size = tf.shape(z_t)[0]
 238     r_mu = self.mus[t](z_t)
 239     r_sigma = tf.maximum(tf.nn.softplus(self.sigmas[t]), self.sigma_min)
 240     r_sigma = tf.tile(r_sigma[tf.newaxis, :], [batch_size, 1])
 241     return tf.contrib.distributions.Normal(
 242         loc=r_mu, scale=tf.sqrt(r_sigma))
 243 
 244   def summarize_weights(self):
 245     for t in range(len(self.mus) - 1):
 246       tf.summary.scalar("r_mu/%d" % t, self.mus[t][0])
 247       tf.summary.scalar("r_sigma/%d" % t, self.sigmas[t][0])

 250 class P(object):
 251 
 252   def __init__(self,
 253                state_size,
 254                num_timesteps,
 255                sigma_min=1e-5,
 256                variance=1.0,
 257                dtype=tf.float32,
 258                random_seed=None,
 259                trainable=True,
 260                init_bs_to_zero=False,
 261                graph_collection_name="P_VARS"):
 262     self.state_size = state_size
 263     self.num_timesteps = num_timesteps
 264     self.sigma_min = sigma_min
 265     self.dtype = dtype
 266     self.variance = variance
 267     self.graph_collection_name = graph_collection_name
 268     if init_bs_to_zero:
 269       initializers = [tf.zeros_initializer for _ in xrange(num_timesteps)]
 270     else:
 271       initializers = [tf.random_uniform_initializer(seed=random_seed) for _ in xrange(num_timesteps)]
 272 
 273     self.bs = [
 274         tf.get_variable(
 275             shape=[state_size],
 276             dtype=self.dtype,
 277             name="p_b_%d" % (t + 1),
 278             initializer=initializers[t],
 279             collections=[tf.GraphKeys.GLOBAL_VARIABLES, graph_collection_name],
 280             trainable=trainable) for t in xrange(num_timesteps)
 281     ]
 282     self.Bs = tf.cumsum(self.bs, reverse=True, axis=0)
 283 
 284   def posterior(self, observation, prev_state, t):
 285     """Computes the true posterior p(z_t|z_{t-1}, z_n)."""
 286     # bs[0] is really b_1
 287     # Bs[i] is sum from k=i+1^n b_k
 288     mu = observation - self.Bs[t]
 289     if t > 0:
 290       mu += (prev_state + self.bs[t - 1]) * float(self.num_timesteps - t)
 291     mu /= float(self.num_timesteps - t + 1)
 292     sigma = tf.ones_like(mu) * self.variance * (
 293         float(self.num_timesteps - t) / float(self.num_timesteps - t + 1))
 294     return tf.contrib.distributions.Normal(loc=mu, scale=tf.sqrt(sigma))
 295 
 296   def lookahead(self, state, t):
 297     """Computes the true lookahead distribution p(z_n|z_t)."""
 298     mu = state + self.Bs[t]
 299     sigma = tf.ones_like(state) * self.variance * float(self.num_timesteps - t)
 300     return tf.contrib.distributions.Normal(loc=mu, scale=tf.sqrt(sigma))
 301 
 302   def likelihood(self, observation):
 303     batch_size = tf.shape(observation)[0]
 304     mu = tf.tile(tf.reduce_sum(self.bs, axis=0)[tf.newaxis, :], [batch_size, 1])
 305     sigma = tf.ones_like(mu) * self.variance * (self.num_timesteps + 1)
 306     dist = tf.contrib.distributions.Normal(loc=mu, scale=tf.sqrt(sigma))
 307     # Average over the batch and take the sum over the state size
 308     return tf.reduce_mean(tf.reduce_sum(dist.log_prob(observation), axis=1))
 309 
 310   def p_zt(self, prev_state, t):
 311     """Computes the model p(z_t| z_{t-1})."""
 312     batch_size = tf.shape(prev_state)[0]
 313     if t > 0:
 314       z_mu_p = prev_state + self.bs[t - 1]
 315     else:  # p(z_0) is Normal(0,1)
 316       z_mu_p = tf.zeros([batch_size, self.state_size], dtype=self.dtype)
 317     p_zt = tf.contrib.distributions.Normal(
 318         loc=z_mu_p, scale=tf.sqrt(tf.ones_like(z_mu_p) * self.variance))
 319     return p_zt
 320 
 321   def generative(self, unused_observation, z_nm1):
 322     """Computes the model's generative distribution p(z_n| z_{n-1})."""
 323     generative_p_mu = z_nm1 + self.bs[-1]
 324     return tf.contrib.distributions.Normal(
 325         loc=generative_p_mu, scale=tf.sqrt(tf.ones_like(generative_p_mu) * self.variance))

 328 class ShortChainNonlinearP(object):
 329 
 330   def __init__(self,
 331                state_size,
 332                num_timesteps,
 333                sigma_min=1e-5,
 334                variance=1.0,
 335                observation_variance=1.0,
 336                transition_type=STANDARD_TRANSITION,
 337                transition_dist=tf.contrib.distributions.Normal,
 338                dtype=tf.float32,
 339                random_seed=None):
 340     self.state_size = state_size
 341     self.num_timesteps = num_timesteps
 342     self.sigma_min = sigma_min
 343     self.dtype = dtype
 344     self.variance = variance
 345     self.observation_variance = observation_variance
 346     self.transition_type = transition_type
 347     self.transition_dist = transition_dist
 348 
 349   def p_zt(self, prev_state, t):
 350     """Computes the model p(z_t| z_{t-1})."""
 351     batch_size = tf.shape(prev_state)[0]
 352     if t > 0:
 353       if self.transition_type == ROUND_TRANSITION:
 354         loc = tf.round(prev_state)
 355         tf.logging.info("p(z_%d | z_%d) ~ N(round(z_%d), %0.1f)" % (t, t-1, t-1, self.variance))
 356       elif self.transition_type == STANDARD_TRANSITION:
 357         loc = prev_state
 358         tf.logging.info("p(z_%d | z_%d) ~ N(z_%d, %0.1f)" % (t, t-1, t-1, self.variance))
 359     else:  # p(z_0) is Normal(0,1)
 360       loc = tf.zeros([batch_size, self.state_size], dtype=self.dtype)
 361       tf.logging.info("p(z_0) ~ N(0,%0.1f)" % self.variance)
 362 
 363     p_zt = self.transition_dist(
 364         loc=loc,
 365         scale=tf.sqrt(tf.ones_like(loc) * self.variance))
 366     return p_zt
 367 
 368   def generative(self, unused_obs, z_ni):
 369     """Computes the model's generative distribution p(x_i| z_{ni})."""
 370     if self.transition_type == ROUND_TRANSITION:
 371       loc = tf.round(z_ni)
 372     elif self.transition_type == STANDARD_TRANSITION:
 373       loc = z_ni
 374     generative_sigma_sq = tf.ones_like(loc) * self.observation_variance
 375     return self.transition_dist(
 376         loc=loc, scale=tf.sqrt(generative_sigma_sq))

 379 class BimodalPriorP(object):
 380 
 381   def __init__(self,
 382                state_size,
 383                num_timesteps,
 384                mixing_coeff=0.5,
 385                prior_mode_mean=1,
 386                sigma_min=1e-5,
 387                variance=1.0,
 388                dtype=tf.float32,
 389                random_seed=None,
 390                trainable=True,
 391                init_bs_to_zero=False,
 392                graph_collection_name="P_VARS"):
 393     self.state_size = state_size
 394     self.num_timesteps = num_timesteps
 395     self.sigma_min = sigma_min
 396     self.dtype = dtype
 397     self.variance = variance
 398     self.mixing_coeff = mixing_coeff
 399     self.prior_mode_mean = prior_mode_mean
 400 
 401     if init_bs_to_zero:
 402       initializers = [tf.zeros_initializer for _ in xrange(num_timesteps)]
 403     else:
 404       initializers = [tf.random_uniform_initializer(seed=random_seed) for _ in xrange(num_timesteps)]
 405 
 406     self.bs = [
 407         tf.get_variable(
 408             shape=[state_size],
 409             dtype=self.dtype,
 410             name="b_%d" % (t + 1),
 411             initializer=initializers[t],
 412             collections=[tf.GraphKeys.GLOBAL_VARIABLES, graph_collection_name],
 413             trainable=trainable) for t in xrange(num_timesteps)
 414     ]
 415     self.Bs = tf.cumsum(self.bs, reverse=True, axis=0)
 416 
 417   def posterior(self, observation, prev_state, t):
 418     # NOTE: This is currently wrong, but would require a refactoring of
 419     # summarize_q to fix as kl is not defined for a mixture
 420     """Computes the true posterior p(z_t|z_{t-1}, z_n)."""
 421     # bs[0] is really b_1
 422     # Bs[i] is sum from k=i+1^n b_k
 423     mu = observation - self.Bs[t]
 424     if t > 0:
 425       mu += (prev_state + self.bs[t - 1]) * float(self.num_timesteps - t)
 426     mu /= float(self.num_timesteps - t + 1)
 427     sigma = tf.ones_like(mu) * self.variance * (
 428         float(self.num_timesteps - t) / float(self.num_timesteps - t + 1))
 429     return tf.contrib.distributions.Normal(loc=mu, scale=tf.sqrt(sigma))
 430 
 431   def lookahead(self, state, t):
 432     """Computes the true lookahead distribution p(z_n|z_t)."""
 433     mu = state + self.Bs[t]
 434     sigma = tf.ones_like(state) * self.variance * float(self.num_timesteps - t)
 435     return tf.contrib.distributions.Normal(loc=mu, scale=tf.sqrt(sigma))
 436 
 437   def likelihood(self, observation):
 438     batch_size = tf.shape(observation)[0]
 439     sum_of_bs = tf.tile(tf.reduce_sum(self.bs, axis=0)[tf.newaxis, :], [batch_size, 1])
 440     sigma = tf.ones_like(sum_of_bs) * self.variance * (self.num_timesteps + 1)
 441     mu_pos = (tf.ones([batch_size, self.state_size], dtype=self.dtype) * self.prior_mode_mean) + sum_of_bs
 442     mu_neg = (tf.ones([batch_size, self.state_size], dtype=self.dtype) * -self.prior_mode_mean) + sum_of_bs
 443     zn_pos = tf.contrib.distributions.Normal(
 444         loc=mu_pos,
 445         scale=tf.sqrt(sigma))
 446     zn_neg = tf.contrib.distributions.Normal(
 447         loc=mu_neg,
 448         scale=tf.sqrt(sigma))
 449     mode_probs = tf.convert_to_tensor([self.mixing_coeff, 1-self.mixing_coeff], dtype=tf.float64)
 450     mode_probs = tf.tile(mode_probs[tf.newaxis, tf.newaxis, :], [batch_size, 1, 1])
 451     mode_selection_dist = tf.contrib.distributions.Categorical(probs=mode_probs)
 452     zn_dist = tf.contrib.distributions.Mixture(
 453         cat=mode_selection_dist,
 454         components=[zn_pos, zn_neg],
 455         validate_args=True)
 456     # Average over the batch and take the sum over the state size
 457     return tf.reduce_mean(tf.reduce_sum(zn_dist.log_prob(observation), axis=1))
 458 
 459   def p_zt(self, prev_state, t):
 460     """Computes the model p(z_t| z_{t-1})."""
 461     batch_size = tf.shape(prev_state)[0]
 462     if t > 0:
 463       z_mu_p = prev_state + self.bs[t - 1]
 464       p_zt = tf.contrib.distributions.Normal(
 465           loc=z_mu_p, scale=tf.sqrt(tf.ones_like(z_mu_p) * self.variance))
 466       return p_zt
 467     else:  # p(z_0) is mixture of two Normals
 468       mu_pos = tf.ones([batch_size, self.state_size], dtype=self.dtype) * self.prior_mode_mean
 469       mu_neg = tf.ones([batch_size, self.state_size], dtype=self.dtype) * -self.prior_mode_mean
 470       z0_pos = tf.contrib.distributions.Normal(
 471           loc=mu_pos,
 472           scale=tf.sqrt(tf.ones_like(mu_pos) * self.variance))
 473       z0_neg = tf.contrib.distributions.Normal(
 474           loc=mu_neg,
 475           scale=tf.sqrt(tf.ones_like(mu_neg) * self.variance))
 476       mode_probs = tf.convert_to_tensor([self.mixing_coeff, 1-self.mixing_coeff], dtype=tf.float64)
 477       mode_probs = tf.tile(mode_probs[tf.newaxis, tf.newaxis, :], [batch_size, 1, 1])
 478       mode_selection_dist = tf.contrib.distributions.Categorical(probs=mode_probs)
 479       z0_dist = tf.contrib.distributions.Mixture(
 480           cat=mode_selection_dist,
 481           components=[z0_pos, z0_neg],
 482           validate_args=False)
 483       return z0_dist

 491 class Model(object):
 492 
 493   def __init__(self,
 494                p,
 495                q,
 496                r,
 497                state_size,
 498                num_timesteps,
 499                dtype=tf.float32):
 500     self.p = p
 501     self.q = q
 502     self.r = r
 503     self.state_size = state_size
 504     self.num_timesteps = num_timesteps
 505     self.dtype = dtype
 506 
 507   def zero_state(self, batch_size):
 508     return tf.zeros([batch_size, self.state_size], dtype=self.dtype)
 509 
 510   def __call__(self, prev_state, observation, t):
 511     # Compute the q distribution over z, q(z_t|z_n, z_{t-1}).
 512     q_zt = self.q.q_zt(observation, prev_state, t)
 513     # Compute the p distribution over z, p(z_t|z_{t-1}).
 514     p_zt = self.p.p_zt(prev_state, t)
 515     # sample from q
 516     zt = q_zt.sample()
 517     r_xn = self.r.r_xn(zt, t)
 518     # Calculate the logprobs and sum over the state size.
 519     log_q_zt = tf.reduce_sum(q_zt.log_prob(zt), axis=1)
 520     log_p_zt = tf.reduce_sum(p_zt.log_prob(zt), axis=1)
 521     log_r_xn = tf.reduce_sum(r_xn.log_prob(observation), axis=1)
 522     # If we're at the last timestep, also calc the logprob of the observation.
 523     if t == self.num_timesteps - 1:
 524       generative_dist = self.p.generative(observation, zt)
 525       log_p_x_given_z = tf.reduce_sum(generative_dist.log_prob(observation), axis=1)
 526     else:
 527       log_p_x_given_z = tf.zeros_like(log_q_zt)
 528     return (zt, log_q_zt, log_p_zt, log_p_x_given_z, log_r_xn)

 531   def create(state_size,
 532              num_timesteps,

 541              p_type="unimodal",
 542              q_type="normal",
 543              observation_variance=1.0,
 544              transition_type=STANDARD_TRANSITION,
 545              use_bs=True):
 546     if p_type == "unimodal":
 547       p = P(state_size,
 548             num_timesteps,
 549             sigma_min=sigma_min,
 550             variance=variance,
 551             dtype=dtype,
 552             random_seed=random_seed,
 553             trainable=train_p,
 554             init_bs_to_zero=not use_bs)
 555     elif p_type == "bimodal":
 556       p = BimodalPriorP(
 557           state_size,
 558           num_timesteps,
 559           mixing_coeff=mixing_coeff,
 560           prior_mode_mean=prior_mode_mean,
 561           sigma_min=sigma_min,
 562           variance=variance,
 563           dtype=dtype,
 564           random_seed=random_seed,
 565           trainable=train_p,
 566           init_bs_to_zero=not use_bs)
 567     elif "nonlinear" in p_type:
 568       if "cauchy" in p_type:
 569         trans_dist = tf.contrib.distributions.Cauchy
 570       else:
 571         trans_dist = tf.contrib.distributions.Normal
 572       p = ShortChainNonlinearP(
 573           state_size,
 574           num_timesteps,
 575           sigma_min=sigma_min,
 576           variance=variance,
 577           observation_variance=observation_variance,
 578           transition_type=transition_type,
 579           transition_dist=trans_dist,
 580           dtype=dtype,
 581           random_seed=random_seed
 582       )
 583 
 584     if q_type == "normal":
 585       q_class = Q
 586     elif q_type == "simple_mean":
 587       q_class = SimpleMeanQ
 588     elif q_type == "prev_state":
 589       q_class = PreviousStateQ
 590     elif q_type == "observation":
 591       q_class = ObservationQ
 592 
 593     q = q_class(state_size,
 594                 num_timesteps,
 595                 sigma_min=sigma_min,
 596                 dtype=dtype,
 597                 random_seed=random_seed,
 598                 init_mu0_to_zero=not use_bs)
 599     r = R(state_size,
 600           num_timesteps,
 601           sigma_min=sigma_min,
 602           sigma_init=r_sigma_init,
 603           dtype=dtype,
 604           random_seed=random_seed)
 605     model = Model(p, q, r, state_size, num_timesteps, dtype=dtype)
 606     return model

 666   def q_zt(self, observation, next_state, t):
 667     """Computes the variational posterior q(z_{t}|z_{t+1}, z_n)."""
 668     t_backwards = self.num_timesteps - t - 1
 669     batch_size = tf.shape(next_state)[0]
 670     q_mu = self.q_mus[t_backwards](tf.concat([observation, next_state], axis=1))
 671     q_sigma = tf.maximum(
 672         tf.nn.softplus(self.q_sigmas[t_backwards]), self.sigma_min)
 673     q_sigma = tf.tile(q_sigma[tf.newaxis, :], [batch_size, 1])
 674     q_zt = tf.contrib.distributions.Normal(loc=q_mu, scale=tf.sqrt(q_sigma))
 675     return q_zt

 677   def p_zt(self, prev_state, t):
 678     """Computes the model p(z_{t+1}| z_{t})."""
 679     t_backwards = self.num_timesteps - t - 1
 680     z_mu_p = prev_state + self.bs[t_backwards]
 681     p_zt = tf.contrib.distributions.Normal(
 682         loc=z_mu_p, scale=tf.ones_like(z_mu_p))
 683     return p_zt

 685   def generative(self, unused_observation, z_nm1):
 686     """Computes the model's generative distribution p(z_n| z_{n-1})."""
 687     generative_p_mu = z_nm1 + self.bs[-1]
 688     return tf.contrib.distributions.Normal(
 689         loc=generative_p_mu, scale=tf.ones_like(generative_p_mu))

 691   def r(self, z_t, t):
 692     t_backwards = self.num_timesteps - t - 1
 693     batch_size = tf.shape(z_t)[0]
 694     r_mu = tf.tile(self.r_mus[t_backwards][tf.newaxis, :], [batch_size, 1])
 695     r_sigma = tf.maximum(
 696         tf.nn.softplus(self.r_sigmas[t_backwards]), self.sigma_min)
 697     r_sigma = tf.tile(r_sigma[tf.newaxis, :], [batch_size, 1])
 698     return tf.contrib.distributions.Normal(loc=r_mu, scale=tf.sqrt(r_sigma))

 700   def likelihood(self, observation):
 701     batch_size = tf.shape(observation)[0]
 702     mu = tf.tile(tf.reduce_sum(self.bs, axis=0)[tf.newaxis, :], [batch_size, 1])
 703     sigma = tf.ones_like(mu) * (self.num_timesteps + 1)
 704     dist = tf.contrib.distributions.Normal(loc=mu, scale=tf.sqrt(sigma))
 705     # Average over the batch and take the sum over the state size
 706     return tf.reduce_mean(tf.reduce_sum(dist.log_prob(observation), axis=1))

 708   def __call__(self, next_state, observation, t):
 709     # next state = z_{t+1}
 710     # Compute the q distribution over z, q(z_{t}|z_n, z_{t+1}).
 711     q_zt = self.q_zt(observation, next_state, t)
 712     # sample from q
 713     zt = q_zt.sample()
 714     # Compute the p distribution over z, p(z_{t+1}|z_{t}).
 715     p_zt = self.p_zt(zt, t)
 716     # Compute log p(z_{t+1} | z_t)
 717     if t == 0:
 718       log_p_zt = p_zt.log_prob(observation)
 719     else:
 720       log_p_zt = p_zt.log_prob(next_state)
 721 
 722     # Compute r prior over zt
 723     r_zt = self.r(zt, t)
 724     log_r_zt = r_zt.log_prob(zt)
 725     # Compute proposal density at zt
 726     log_q_zt = q_zt.log_prob(zt)
 727     # If we're at the last timestep, also calc the logprob of the observation.
 728 
 729     if t == self.num_timesteps - 1:
 730       p_z0_dist = tf.contrib.distributions.Normal(
 731           loc=tf.zeros_like(zt), scale=tf.ones_like(zt))
 732       z0_log_prob = p_z0_dist.log_prob(zt)
 733     else:
 734       z0_log_prob = tf.zeros_like(log_q_zt)
 735     return (zt, log_q_zt, log_p_zt, z0_log_prob, log_r_zt)

 983   def __call__(self, prev_state, observations, t):
 984     """Computes the importance weight for the model system.
 985 
 986     Args:
 987       prev_state: [batch_size, state_size] Tensor
 988       observations: [batch_size, num_observations, state_size] Tensor
 989     """
 990     # Compute the q distribution over z, q(z_t|z_n, z_{t-1}).
 991     q_zt = self.q.q_zt(observations, prev_state, t)
 992     # Compute the p distribution over z, p(z_t|z_{t-1}).

 995     zt = q_zt.sample()
 996     log_q_zt = tf.reduce_sum(q_zt.log_prob(zt), axis=1)
 997     log_p_zt = tf.reduce_sum(p_zt.log_prob(zt), axis=1)

 999       # score the remaining observations using r
1000       r_xn = self.r.r_xn(zt, t)
1001       log_r_xn = r_xn.log_prob(observations[:, self.next_obs_ind(t+1):, :])
1002       # sum over state size and observation, leaving the batch index
1003       log_r_xn = tf.reduce_sum(log_r_xn, axis=[1,2])
1004     else:
1005       log_r_xn = tf.zeros_like(log_p_zt)
1006     if t != 0 and t % self.steps_per_obs == 0:
1007       generative_dist = self.p.generative(zt, t)
1008       log_p_x_given_z = generative_dist.log_prob(observations[:,self.next_obs_ind(t),:])
1009       log_p_x_given_z = tf.reduce_sum(log_p_x_given_z, axis=1)
1010     else:
1011       log_p_x_given_z = tf.zeros_like(log_q_zt)
1012     return (zt, log_q_zt, log_p_zt, log_p_x_given_z, log_r_xn)

1061                state_size,

1063                sigma_min=1e-5,

1065                random_seed=None,

1073     def custom_getter(getter, *args, **kwargs):
1074       out = getter(*args, **kwargs)
1075       ref = tf.get_collection_ref(self.graph_collection_name)
1076       if out not in ref:
1077         ref.append(out)
1078       return out
1079 
1080     self.fns = [
1081         snt.Linear(output_size=2*state_size,
1082                    initializers=initializers,
1083                    name="r_tilde_%d" % t,
1084                    custom_getter=custom_getter)
1085         for t in xrange(num_timesteps)
1086     ]

1088   def r_zt(self, z_t, observation, t):
1089     #out = self.fns[t](tf.stop_gradient(tf.concat([z_t, observation], axis=1)))
1090     out = self.fns[t](tf.concat([z_t, observation], axis=1))
1091     mu, raw_sigma_sq = tf.split(out, 2, axis=1)
1092     sigma_sq = tf.maximum(tf.nn.softplus(raw_sigma_sq), self.sigma_min)
1093     return mu, sigma_sq

1095 class TDModel(object):
1096 
1097   def __init__(self,
1098                p,
1099                q,
1100                r_tilde,
1101                state_size,
1102                num_timesteps,
1103                dtype=tf.float32,
1104                disable_r=False):
1105     self.p = p
1106     self.q = q
1107     self.r_tilde = r_tilde
1108     self.disable_r = disable_r
1109     self.state_size = state_size
1110     self.num_timesteps = num_timesteps
1111     self.dtype = dtype
1112 
1113   def zero_state(self, batch_size):
1114     return tf.zeros([batch_size, self.state_size], dtype=self.dtype)
1115 
1116   def __call__(self, prev_state, observation, t):
1117     """Computes the importance weight for the model system.
1118 
1119     Args:
1120       prev_state: [batch_size, state_size] Tensor
1121       observations: [batch_size, num_observations, state_size] Tensor
1122     """
1123     # Compute the q distribution over z, q(z_t|z_n, z_{t-1}).
1124     q_zt = self.q.q_zt(observation, prev_state, t)
1125     # Compute the p distribution over z, p(z_t|z_{t-1}).
1126     p_zt = self.p.p_zt(prev_state, t)
1127     # sample from q and evaluate the logprobs, summing over the state size
1128     zt = q_zt.sample()
1129     # If it isn't the last timestep, compute the distribution over the next z.
1130     if t < self.num_timesteps - 1:
1131       p_ztplus1 = self.p.p_zt(zt, t+1)
1132     else:
1133       p_ztplus1 = None
1134     log_q_zt = tf.reduce_sum(q_zt.log_prob(zt), axis=1)
1135     log_p_zt = tf.reduce_sum(p_zt.log_prob(zt), axis=1)
1136 
1137     if not self.disable_r and t < self.num_timesteps-1:
1138       # score the remaining observations using r
1139       r_tilde_mu, r_tilde_sigma_sq = self.r_tilde.r_zt(zt, observation, t+1)
1140     else:
1141       r_tilde_mu = None
1142       r_tilde_sigma_sq = None
1143     if t == self.num_timesteps - 1:
1144       generative_dist = self.p.generative(observation, zt)
1145       log_p_x_given_z = tf.reduce_sum(generative_dist.log_prob(observation), axis=1)
1146     else:
1147       log_p_x_given_z = tf.zeros_like(log_q_zt)
1148     return (zt, log_q_zt, log_p_zt, log_p_x_given_z,
1149             r_tilde_mu, r_tilde_sigma_sq, p_ztplus1)
