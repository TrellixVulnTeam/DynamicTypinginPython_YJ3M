  29 def iwae(model, observation, num_timesteps, num_samples=1,
  30          summarize=False):
  31   """Compute the IWAE evidence lower bound.
  32 
  33   Args:
  34     model: A callable that computes one timestep of the model.
  35     observation: A shape [batch_size*num_samples, state_size] Tensor
  36       containing z_n, the observation for each sequence in the batch.
  37     num_timesteps: The number of timesteps in each sequence, an integer.
  38     num_samples: The number of samples to use to compute the IWAE bound.
  39   Returns:
  40     log_p_hat: The IWAE estimator of the lower bound on the log marginal.
  41     loss: A tensor that you can perform gradient descent on to optimize the
  42       bound.
  43     maintain_ema_op: A no-op included for compatibility with FIVO.
  44     states: The sequence of states sampled.
  45   """
  46   # Initialization
  47   num_instances = tf.shape(observation)[0]
  48   batch_size = tf.cast(num_instances / num_samples, tf.int32)
  49   states = [model.zero_state(num_instances)]
  50   log_weights = []
  51   log_weight_acc = tf.zeros([num_samples, batch_size], dtype=observation.dtype)
  52 
  53   for t in xrange(num_timesteps):
  54     # run the model for one timestep
  55     (zt, log_q_zt, log_p_zt, log_p_x_given_z, _) = model(
  56         states[-1], observation, t)
  57     # update accumulators
  58     states.append(zt)
  59     log_weight = log_p_zt + log_p_x_given_z - log_q_zt
  60     log_weight_acc += tf.reshape(log_weight, [num_samples, batch_size])
  61     if summarize:
  62       weight_dist = tf.contrib.distributions.Categorical(
  63           logits=tf.transpose(log_weight_acc, perm=[1, 0]),
  64           allow_nan_stats=False)
  65       weight_entropy = weight_dist.entropy()
  66       weight_entropy = tf.reduce_mean(weight_entropy)
  67       tf.summary.scalar("weight_entropy/%d" % t, weight_entropy)
  68     log_weights.append(log_weight_acc)
  69   # Compute the lower bound on the log evidence.
  70   log_p_hat = (tf.reduce_logsumexp(log_weight_acc, axis=0) -
  71                tf.log(tf.cast(num_samples, observation.dtype))) / num_timesteps
  72   loss = -tf.reduce_mean(log_p_hat)
  73   losses = [Loss("log_p_hat", loss)]
  74 
  75   # we clip off the initial state before returning.
  76   # there are no emas for iwae, so we return a noop for that
  77   return log_p_hat, losses, tf.no_op(), states[1:], log_weights

  80 def multinomial_resampling(log_weights, states, n, b):
  81   """Resample states with multinomial resampling.
  82 
  83   Args:
  84     log_weights: A (n x b) Tensor representing a batch of b logits for n-ary
  85       Categorical distribution.
  86     states: A list of (b*n x d) Tensors that will be resample in from the groups
  87      of every n-th row.
  88 
  89   Returns:
  90     resampled_states: A list of (b*n x d) Tensors resampled via stratified sampling.
  91     log_probs: A (n x b) Tensor of the log probabilities of the ancestry decisions.
  92     resampling_parameters: The Tensor of parameters of the resampling distribution.
  93     ancestors: An (n x b) Tensor of integral indices representing the ancestry decisions.
  94     resampling_dist: The distribution object for resampling.
  95   """
  96   log_weights = tf.convert_to_tensor(log_weights)
  97   states = [tf.convert_to_tensor(state) for state in states]
  98 
  99   resampling_parameters = tf.transpose(log_weights, perm=[1,0])
 100   resampling_dist = tf.contrib.distributions.Categorical(logits=resampling_parameters)
 101   ancestors = tf.stop_gradient(
 102       resampling_dist.sample(sample_shape=n))
 103   log_probs = resampling_dist.log_prob(ancestors)
 104 
 105   offset = tf.expand_dims(tf.range(b), 0)
 106   ancestor_inds = tf.reshape(ancestors * b + offset, [-1])
 107 
 108   resampled_states = []
 109   for state in states:
 110     resampled_states.append(tf.gather(state, ancestor_inds))
 111   return resampled_states, log_probs, resampling_parameters, ancestors, resampling_dist

 113 def stratified_resampling(log_weights, states, n, b):
 114   """Resample states with straitified resampling.
 115 
 116   Args:
 117     log_weights: A (n x b) Tensor representing a batch of b logits for n-ary
 118       Categorical distribution.
 119     states: A list of (b*n x d) Tensors that will be resample in from the groups
 120      of every n-th row.
 121 
 122   Returns:
 123     resampled_states: A list of (b*n x d) Tensors resampled via stratified sampling.
 124     log_probs: A (n x b) Tensor of the log probabilities of the ancestry decisions.
 125     resampling_parameters: The Tensor of parameters of the resampling distribution.
 126     ancestors: An (n x b) Tensor of integral indices representing the ancestry decisions.
 127     resampling_dist: The distribution object for resampling.
 128   """
 129   log_weights = tf.convert_to_tensor(log_weights)
 130   states = [tf.convert_to_tensor(state) for state in states]
 131 
 132   log_weights = tf.transpose(log_weights, perm=[1,0])
 133 
 134   probs = tf.nn.softmax(
 135             tf.tile(tf.expand_dims(log_weights, axis=1),
 136                     [1, n, 1])
 137   )
 138 
 139   cdfs = tf.concat([tf.zeros((b,n,1), dtype=probs.dtype), tf.cumsum(probs, axis=2)], 2)
 140 
 141   bins = tf.range(n, dtype=probs.dtype) / n
 142   bins = tf.tile(tf.reshape(bins, [1,-1,1]), [b,1,n+1])
 143 
 144   strat_cdfs = tf.minimum(tf.maximum((cdfs - bins) * n, 0.0), 1.0)
 145   resampling_parameters = strat_cdfs[:,:,1:] - strat_cdfs[:,:,:-1]
 146 
 147   resampling_dist = tf.contrib.distributions.Categorical(
 148       probs = resampling_parameters,
 149       allow_nan_stats=False)
 150 
 151   ancestors = tf.stop_gradient(
 152       resampling_dist.sample())
 153   log_probs = resampling_dist.log_prob(ancestors)
 154 
 155   ancestors = tf.transpose(ancestors, perm=[1,0])
 156   log_probs = tf.transpose(log_probs, perm=[1,0])
 157 
 158   offset = tf.expand_dims(tf.range(b), 0)
 159   ancestor_inds = tf.reshape(ancestors * b + offset, [-1])
 160 
 161   resampled_states = []
 162   for state in states:
 163     resampled_states.append(tf.gather(state, ancestor_inds))
 164 
 165   return resampled_states, log_probs, resampling_parameters, ancestors, resampling_dist

 167 def systematic_resampling(log_weights, states, n, b):
 168   """Resample states with systematic resampling.
 169 
 170   Args:
 171     log_weights: A (n x b) Tensor representing a batch of b logits for n-ary
 172       Categorical distribution.
 173     states: A list of (b*n x d) Tensors that will be resample in from the groups
 174      of every n-th row.
 175 
 176   Returns:
 177     resampled_states: A list of (b*n x d) Tensors resampled via stratified sampling.
 178     log_probs: A (n x b) Tensor of the log probabilities of the ancestry decisions.
 179     resampling_parameters: The Tensor of parameters of the resampling distribution.
 180     ancestors: An (n x b) Tensor of integral indices representing the ancestry decisions.
 181     resampling_dist: The distribution object for resampling.
 182   """
 183 
 184   log_weights = tf.convert_to_tensor(log_weights)
 185   states = [tf.convert_to_tensor(state) for state in states]
 186 
 187   log_weights = tf.transpose(log_weights, perm=[1,0])
 188 
 189   probs = tf.nn.softmax(
 190             tf.tile(tf.expand_dims(log_weights, axis=1),
 191                     [1, n, 1])
 192   )
 193 
 194   cdfs = tf.concat([tf.zeros((b,n,1), dtype=probs.dtype), tf.cumsum(probs, axis=2)], 2)
 195 
 196   bins = tf.range(n, dtype=probs.dtype) / n
 197   bins = tf.tile(tf.reshape(bins, [1,-1,1]), [b,1,n+1])
 198 
 199   strat_cdfs = tf.minimum(tf.maximum((cdfs - bins) * n, 0.0), 1.0)
 200   resampling_parameters = strat_cdfs[:,:,1:] - strat_cdfs[:,:,:-1]
 201 
 202   resampling_dist = tf.contrib.distributions.Categorical(
 203       probs=resampling_parameters,
 204       allow_nan_stats=True)
 205 
 206   U = tf.random_uniform((b, 1, 1), dtype=probs.dtype)
 207 
 208   ancestors = tf.stop_gradient(tf.reduce_sum(tf.to_float(U > strat_cdfs[:,:,1:]), axis=-1))
 209   log_probs = resampling_dist.log_prob(ancestors)
 210 
 211   ancestors = tf.transpose(ancestors, perm=[1,0])
 212   log_probs = tf.transpose(log_probs, perm=[1,0])
 213 
 214   offset = tf.expand_dims(tf.range(b, dtype=probs.dtype), 0)
 215   ancestor_inds = tf.reshape(ancestors * b + offset, [-1])
 216 
 217   resampled_states = []
 218   for state in states:
 219     resampled_states.append(tf.gather(state, ancestor_inds))
 220 
 221   return resampled_states, log_probs, resampling_parameters, ancestors, resampling_dist

 224 def log_blend(inputs, weights):
 225   """Blends state in the log space.
 226 
 227   Args:
 228     inputs: A set of scalar states, one for each particle in each particle filter.
 229       Should be [num_samples, batch_size].
 230     weights: A set of weights used to blend the state. Each set of weights
 231       should be of dimension [num_samples] (one weight for each previous particle).
 232       There should be one set of weights for each new particle in each particle filter.
 233       Thus the shape should be [num_samples, batch_size, num_samples] where
 234       the first axis indexes new particle and the last axis indexes old particles.
 235   Returns:
 236     blended: The blended states, a tensor of shape [num_samples, batch_size].
 237   """
 238   raw_max = tf.reduce_max(inputs, axis=0, keepdims=True)
 239   my_max = tf.stop_gradient(
 240       tf.where(tf.is_finite(raw_max), raw_max, tf.zeros_like(raw_max))
 241   )
 242   # Don't ask.
 243   blended = tf.log(tf.einsum("ijk,kj->ij", weights, tf.exp(inputs - raw_max))) + my_max
 244   return blended

 247 def relaxed_resampling(log_weights, states, num_samples, batch_size,
 248                        log_r_x=None, blend_type="log", temperature=0.5,
 249                        straight_through=False):
 250   """Resample states with relaxed resampling.
 251 
 252   Args:
 253     log_weights: A (n x b) Tensor representing a batch of b logits for n-ary
 254       Categorical distribution.
 255     states: A list of (b*n x d) Tensors that will be resample in from the groups
 256      of every n-th row.
 257 
 258   Returns:
 259     resampled_states: A list of (b*n x d) Tensors resampled via stratified sampling.
 260     log_probs: A (n x b) Tensor of the log probabilities of the ancestry decisions.
 261     resampling_parameters: The Tensor of parameters of the resampling distribution.
 262     ancestors: An (n x b x n) Tensor of relaxed one hot representations of the ancestry decisions.
 263     resampling_dist: The distribution object for resampling.
 264   """
 265   assert blend_type in ["log", "linear"], "Blend type must be 'log' or 'linear'."
 266   log_weights = tf.convert_to_tensor(log_weights)
 267   states = [tf.convert_to_tensor(state) for state in states]
 268   state_dim = states[0].get_shape().as_list()[-1]
 269   # weights are num_samples by batch_size, so we transpose to get a
 270   # set of batch_size distributions over [0,num_samples).
 271   resampling_parameters = tf.transpose(log_weights, perm=[1, 0])
 272   resampling_dist = tf.contrib.distributions.RelaxedOneHotCategorical(
 273       temperature,
 274       logits=resampling_parameters)
 275 
 276   # sample num_samples samples from the distribution, resulting in a
 277   # [num_samples, batch_size, num_samples] Tensor that represents a set of
 278   # [num_samples, batch_size] blending weights. The dimensions represent
 279   # [sample index, batch index, blending weight index]
 280   ancestors = resampling_dist.sample(sample_shape=num_samples)
 281   if straight_through:
 282     # Forward pass discrete choices, backwards pass soft choices
 283     hard_ancestor_indices = tf.argmax(ancestors, axis=-1)
 284     hard_ancestors = tf.one_hot(hard_ancestor_indices, num_samples,
 285                                 dtype=ancestors.dtype)
 286     ancestors = tf.stop_gradient(hard_ancestors - ancestors) + ancestors
 287   log_probs = resampling_dist.log_prob(ancestors)
 288   if log_r_x is not None and blend_type == "log":
 289     log_r_x = tf.reshape(log_r_x, [num_samples, batch_size])
 290     log_r_x = log_blend(log_r_x, ancestors)
 291     log_r_x = tf.reshape(log_r_x, [num_samples*batch_size])
 292   elif log_r_x is not None and blend_type == "linear":
 293     # If blend type is linear just add log_r to the states that will be blended
 294     # linearly.
 295     states.append(log_r_x)
 296 
 297   # transpose the 'indices' to be [batch_index, blending weight index, sample index]
 298   ancestor_inds = tf.transpose(ancestors, perm=[1, 2, 0])
 299   resampled_states = []
 300   for state in states:
 301     # state is currently [num_samples * batch_size, state_dim] so we reshape
 302     # to [num_samples, batch_size, state_dim] and then transpose to
 303     # [batch_size, state_size, num_samples]
 304     state = tf.transpose(tf.reshape(state, [num_samples, batch_size, -1]), perm=[1, 2, 0])
 305     # state is now (batch_size, state_size, num_samples)
 306     # and ancestor is (batch index, blending weight index, sample index)
 307     # multiplying these gives a matrix of size [batch_size, state_size, num_samples]
 308     next_state = tf.matmul(state, ancestor_inds)
 309     # transpose the state to be [num_samples, batch_size, state_size]
 310     # and then reshape it to match the state format.
 311     next_state = tf.reshape(tf.transpose(next_state, perm=[2,0,1]), [num_samples*batch_size, state_dim])
 312     resampled_states.append(next_state)
 313 
 314   new_dist = tf.contrib.distributions.Categorical(
 315       logits=resampling_parameters)
 316 
 317   if log_r_x is not None and blend_type == "linear":
 318     # If blend type is linear pop off log_r that we added to the states.
 319     log_r_x = tf.squeeze(resampled_states[-1])
 320     resampled_states = resampled_states[:-1]
 321   return resampled_states, log_probs, log_r_x, resampling_parameters, ancestors, new_dist

 324 def fivo(model,
 325          observation,
 326          num_timesteps,
 327          resampling_schedule,
 328          num_samples=1,
 329          use_resampling_grads=True,
 330          resampling_type="multinomial",
 331          resampling_temperature=0.5,
 332          aux=True,
 333          summarize=False):
 334   """Compute the FIVO evidence lower bound.
 335 
 336   Args:
 337     model: A callable that computes one timestep of the model.
 338     observation: A shape [batch_size*num_samples, state_size] Tensor
 339       containing z_n, the observation for each sequence in the batch.
 340     num_timesteps: The number of timesteps in each sequence, an integer.
 341     resampling_schedule: A list of booleans of length num_timesteps, contains
 342       True if a resampling should occur on a specific timestep.
 343     num_samples: The number of samples to use to compute the IWAE bound.
 344     use_resampling_grads: Whether or not to include the resampling gradients
 345       in loss.
 346     resampling type: The type of resampling, one of "multinomial", "stratified",
 347       "relaxed-logblend", "relaxed-linearblend", "relaxed-stateblend", or
 348       "systematic".
 349     resampling_temperature: A positive temperature only used for relaxed
 350       resampling.
 351     aux: If true, compute the FIVO-AUX bound.
 352   Returns:
 353     log_p_hat: The IWAE estimator of the lower bound on the log marginal.
 354     loss: A tensor that you can perform gradient descent on to optimize the
 355       bound.
 356     maintain_ema_op: An op to update the baseline ema used for the resampling
 357       gradients.
 358     states: The sequence of states sampled.
 359   """
 360   # Initialization
 361   num_instances = tf.cast(tf.shape(observation)[0], tf.int32)
 362   batch_size = tf.cast(num_instances / num_samples, tf.int32)
 363   states = [model.zero_state(num_instances)]
 364   prev_state = states[0]
 365   log_weight_acc = tf.zeros(shape=[num_samples, batch_size], dtype=observation.dtype)
 366   prev_log_r_zt = tf.zeros([num_instances], dtype=observation.dtype)
 367   log_weights = []
 368   log_weights_all = []
 369   log_p_hats = []
 370   resampling_log_probs = []
 371   for t in xrange(num_timesteps):
 372     # run the model for one timestep
 373     (zt, log_q_zt, log_p_zt, log_p_x_given_z, log_r_zt) = model(
 374         prev_state, observation, t)
 375     # update accumulators
 376     states.append(zt)
 377     log_weight = log_p_zt + log_p_x_given_z - log_q_zt
 378     if aux:
 379       if t == num_timesteps - 1:
 380         log_weight -= prev_log_r_zt
 381       else:
 382         log_weight += log_r_zt - prev_log_r_zt
 383       prev_log_r_zt = log_r_zt
 384     log_weight_acc += tf.reshape(log_weight, [num_samples, batch_size])
 385     log_weights_all.append(log_weight_acc)
 386     if resampling_schedule[t]:
 387 
 388       # These objects will be resampled
 389       to_resample = [states[-1]]
 390       if aux and "relaxed" not in resampling_type:
 391         to_resample.append(prev_log_r_zt)
 392 
 393       # do the resampling
 394       if resampling_type == "multinomial":
 395         (resampled,
 396          resampling_log_prob,
 397          _, _, _) = multinomial_resampling(log_weight_acc,
 398                                            to_resample,
 399                                            num_samples,
 400                                            batch_size)
 401       elif resampling_type == "stratified":
 402         (resampled,
 403          resampling_log_prob,
 404          _, _, _) = stratified_resampling(log_weight_acc,
 405                                           to_resample,
 406                                           num_samples,
 407                                           batch_size)
 408       elif resampling_type == "systematic":
 409         (resampled,
 410          resampling_log_prob,
 411          _, _, _) = systematic_resampling(log_weight_acc,
 412                                           to_resample,
 413                                           num_samples,
 414                                           batch_size)
 415       elif "relaxed" in resampling_type:
 416         if aux:
 417           if resampling_type == "relaxed-logblend":
 418             (resampled,
 419              resampling_log_prob,
 420              prev_log_r_zt,
 421              _, _, _) = relaxed_resampling(log_weight_acc,
 422                                            to_resample,
 423                                            num_samples,
 424                                            batch_size,
 425                                            temperature=resampling_temperature,
 426                                            log_r_x=prev_log_r_zt,
 427                                            blend_type="log")
 428           elif resampling_type == "relaxed-linearblend":
 429             (resampled,
 430              resampling_log_prob,
 431              prev_log_r_zt,
 432              _, _, _) = relaxed_resampling(log_weight_acc,
 433                                            to_resample,
 434                                            num_samples,
 435                                            batch_size,
 436                                            temperature=resampling_temperature,
 437                                            log_r_x=prev_log_r_zt,
 438                                            blend_type="linear")
 439           elif resampling_type == "relaxed-stateblend":
 440             (resampled,
 441              resampling_log_prob,
 442              _, _, _, _) = relaxed_resampling(log_weight_acc,
 443                                               to_resample,
 444                                               num_samples,
 445                                               batch_size,
 446                                               temperature=resampling_temperature)
 447             # Calculate prev_log_r_zt from the post-resampling state
 448             prev_r_zt = model.r.r_xn(resampled[0], t)
 449             prev_log_r_zt = tf.reduce_sum(
 450                 prev_r_zt.log_prob(observation), axis=[1])
 451           elif resampling_type == "relaxed-stateblend-st":
 452             (resampled,
 453              resampling_log_prob,
 454              _, _, _, _) = relaxed_resampling(log_weight_acc,
 455                                               to_resample,
 456                                               num_samples,
 457                                               batch_size,
 458                                               temperature=resampling_temperature,
 459                                               straight_through=True)
 460             # Calculate prev_log_r_zt from the post-resampling state
 461             prev_r_zt = model.r.r_xn(resampled[0], t)
 462             prev_log_r_zt = tf.reduce_sum(
 463                 prev_r_zt.log_prob(observation), axis=[1])
 464         else:
 465           (resampled,
 466            resampling_log_prob,
 467            _, _, _, _) = relaxed_resampling(log_weight_acc,
 468                                             to_resample,
 469                                             num_samples,
 470                                             batch_size,
 471                                             temperature=resampling_temperature)
 472       #if summarize:
 473       #  resampling_entropy = resampling_dist.entropy()
 474       #  resampling_entropy = tf.reduce_mean(resampling_entropy)
 475       #  tf.summary.scalar("weight_entropy/%d" % t, resampling_entropy)
 476 
 477       resampling_log_probs.append(tf.reduce_sum(resampling_log_prob, axis=0))
 478       prev_state = resampled[0]
 479       if aux and "relaxed" not in resampling_type:
 480         # Squeeze out the extra dim potentially added by resampling.
 481         # prev_log_r_zt should always be [num_instances]
 482         prev_log_r_zt = tf.squeeze(resampled[1])
 483       # Update the log p hat estimate, taking a log sum exp over the sample
 484       # dimension. The appended tensor is [batch_size].
 485       log_p_hats.append(
 486           tf.reduce_logsumexp(log_weight_acc, axis=0) - tf.log(
 487               tf.cast(num_samples, dtype=observation.dtype)))
 488       # reset the weights
 489       log_weights.append(log_weight_acc)
 490       log_weight_acc = tf.zeros_like(log_weight_acc)
 491     else:
 492       prev_state = states[-1]
 493   # Compute the final weight update. If we just resampled this will be zero.
 494   final_update = (tf.reduce_logsumexp(log_weight_acc, axis=0) -
 495                   tf.log(tf.cast(num_samples, dtype=observation.dtype)))
 496   # If we ever resampled, then sum up the previous log p hat terms
 497   if len(log_p_hats) > 0:
 498     log_p_hat = tf.reduce_sum(log_p_hats, axis=0) + final_update
 499   else:  # otherwise, log_p_hat only comes from the final update
 500     log_p_hat = final_update
 501 
 502   if use_resampling_grads and any(resampling_schedule):
 503     # compute the rewards
 504     # cumsum([a, b, c]) => [a, a+b, a+b+c]
 505     # learning signal at timestep t is
 506     #   [sum from i=t+1 to T of log_p_hat_i for t=1:T]
 507     # so we will compute (sum from i=1 to T of log_p_hat_i)
 508     # and at timestep t will subtract off (sum from i=1 to t of log_p_hat_i)
 509     # rewards is a [num_resampling_events, batch_size] Tensor
 510     rewards = tf.stop_gradient(
 511         tf.expand_dims(log_p_hat, 0) - tf.cumsum(log_p_hats, axis=0))
 512     batch_avg_rewards = tf.reduce_mean(rewards, axis=1)
 513     # compute ema baseline.
 514     # centered_rewards is [num_resampling_events, batch_size]
 515     baseline_ema = tf.train.ExponentialMovingAverage(decay=0.94)
 516     maintain_baseline_op = baseline_ema.apply([batch_avg_rewards])
 517     baseline = tf.expand_dims(baseline_ema.average(batch_avg_rewards), 1)
 518     centered_rewards = rewards - baseline
 519     if summarize:
 520       summ.summarize_learning_signal(rewards, "rewards")
 521       summ.summarize_learning_signal(centered_rewards, "centered_rewards")
 522     # compute the loss tensor.
 523     resampling_grads = tf.reduce_sum(
 524         tf.stop_gradient(centered_rewards) * resampling_log_probs, axis=0)
 525     losses = [Loss("log_p_hat", -tf.reduce_mean(log_p_hat)/num_timesteps),
 526               Loss("resampling_grads", -tf.reduce_mean(resampling_grads)/num_timesteps)]
 527   else:
 528     losses = [Loss("log_p_hat", -tf.reduce_mean(log_p_hat)/num_timesteps)]
 529     maintain_baseline_op = tf.no_op()
 530 
 531   log_p_hat /= num_timesteps
 532   # we clip off the initial state before returning.
 533   return log_p_hat, losses, maintain_baseline_op, states[1:], log_weights_all

 536 def fivo_aux_td(
 537     model,
 538     observation,
 539     num_timesteps,
 540     resampling_schedule,
 541     num_samples=1,
 542     summarize=False):
 543   """Compute the FIVO_AUX evidence lower bound."""
 544   # Initialization
 545   num_instances = tf.cast(tf.shape(observation)[0], tf.int32)
 546   batch_size = tf.cast(num_instances / num_samples, tf.int32)
 547   states = [model.zero_state(num_instances)]
 548   prev_state = states[0]
 549   log_weight_acc = tf.zeros(shape=[num_samples, batch_size], dtype=observation.dtype)
 550   prev_log_r = tf.zeros([num_instances], dtype=observation.dtype)
 551   # must be pre-resampling
 552   log_rs = []
 553   # must be post-resampling
 554   r_tilde_params = [model.r_tilde.r_zt(states[0], observation, 0)]
 555   log_r_tildes = []
 556   log_p_xs = []
 557   # contains the weight at each timestep before resampling only on resampling timesteps
 558   log_weights = []
 559   # contains weight at each timestep before resampling
 560   log_weights_all = []
 561   log_p_hats = []
 562   for t in xrange(num_timesteps):
 563     # run the model for one timestep
 564     # zt is state, [num_instances, state_dim]
 565     # log_q_zt, log_p_x_given_z is [num_instances]
 566     # r_tilde_mu, r_tilde_sigma is [num_instances, state_dim]
 567     # p_ztplus1 is a normal distribution on [num_instances, state_dim]
 568     (zt, log_q_zt, log_p_zt, log_p_x_given_z,
 569      r_tilde_mu, r_tilde_sigma_sq, p_ztplus1) = model(prev_state, observation, t)
 570 
 571     # Compute the log weight without log r.
 572     log_weight = log_p_zt + log_p_x_given_z - log_q_zt
 573 
 574     # Compute log r.
 575     if t == num_timesteps - 1:
 576       log_r = tf.zeros_like(prev_log_r)
 577     else:
 578       p_mu = p_ztplus1.mean()
 579       p_sigma_sq = p_ztplus1.variance()
 580       log_r = (tf.log(r_tilde_sigma_sq) -
 581                tf.log(r_tilde_sigma_sq + p_sigma_sq) -
 582                tf.square(r_tilde_mu - p_mu)/(r_tilde_sigma_sq + p_sigma_sq))
 583       log_r = 0.5*tf.reduce_sum(log_r, axis=-1)
 584 
 585     #log_weight += tf.stop_gradient(log_r - prev_log_r)
 586     log_weight += log_r - prev_log_r
 587     log_weight_acc += tf.reshape(log_weight, [num_samples, batch_size])
 588 
 589     # Update accumulators
 590     states.append(zt)
 591     log_weights_all.append(log_weight_acc)
 592     log_p_xs.append(log_p_x_given_z)
 593     log_rs.append(log_r)
 594 
 595     # Compute log_r_tilde as [num_instances] Tensor.
 596     prev_r_tilde_mu, prev_r_tilde_sigma_sq = r_tilde_params[-1]
 597     prev_log_r_tilde = -0.5*tf.reduce_sum(
 598         tf.square(zt - prev_r_tilde_mu)/prev_r_tilde_sigma_sq, axis=-1)
 599         #tf.square(tf.stop_gradient(zt) - r_tilde_mu)/r_tilde_sigma_sq, axis=-1)
 600         #tf.square(zt - r_tilde_mu)/r_tilde_sigma_sq, axis=-1)
 601     log_r_tildes.append(prev_log_r_tilde)
 602 
 603     # optionally resample
 604     if resampling_schedule[t]:
 605       # These objects will be resampled
 606       if t < num_timesteps - 1:
 607         to_resample = [zt, log_r, r_tilde_mu, r_tilde_sigma_sq]
 608       else:
 609         to_resample = [zt, log_r]
 610       (resampled,
 611        _, _, _, _) = multinomial_resampling(log_weight_acc,
 612                                             to_resample,
 613                                             num_samples,
 614                                             batch_size)
 615       prev_state = resampled[0]
 616       # Squeeze out the extra dim potentially added by resampling.
 617       # prev_log_r_zt and log_r_tilde should always be [num_instances]
 618       prev_log_r = tf.squeeze(resampled[1])
 619       if t < num_timesteps -1:
 620         r_tilde_params.append((resampled[2], resampled[3]))
 621       # Update the log p hat estimate, taking a log sum exp over the sample
 622       # dimension. The appended tensor is [batch_size].
 623       log_p_hats.append(
 624           tf.reduce_logsumexp(log_weight_acc, axis=0) - tf.log(
 625               tf.cast(num_samples, dtype=observation.dtype)))
 626       # reset the weights
 627       log_weights.append(log_weight_acc)
 628       log_weight_acc = tf.zeros_like(log_weight_acc)
 629     else:
 630       prev_state = zt
 631       prev_log_r = log_r
 632       if t < num_timesteps - 1:
 633         r_tilde_params.append((r_tilde_mu, r_tilde_sigma_sq))
 634 
 635   # Compute the final weight update. If we just resampled this will be zero.
 636   final_update = (tf.reduce_logsumexp(log_weight_acc, axis=0) -
 637                   tf.log(tf.cast(num_samples, dtype=observation.dtype)))
 638   # If we ever resampled, then sum up the previous log p hat terms
 639   if len(log_p_hats) > 0:
 640     log_p_hat = tf.reduce_sum(log_p_hats, axis=0) + final_update
 641   else:  # otherwise, log_p_hat only comes from the final update
 642     log_p_hat = final_update
 643 
 644   # Compute the bellman loss.
 645   # Will remove the first timestep as it is not used.
 646   # log p(x_t|z_t) is in row t-1.
 647   log_p_x = tf.reshape(tf.stack(log_p_xs),
 648                        [num_timesteps, num_samples, batch_size])
 649   # log r_t is contained in row t-1.
 650   # last column is zeros (because at timestep T (num_timesteps) r is 1.
 651   log_r = tf.reshape(tf.stack(log_rs),
 652                      [num_timesteps, num_samples, batch_size])
 653   # [num_timesteps, num_instances]. log r_tilde_t is in row t-1.
 654   log_r_tilde = tf.reshape(tf.stack(log_r_tildes),
 655                            [num_timesteps, num_samples, batch_size])
 656   log_lambda = tf.reduce_mean(log_r_tilde - log_p_x - log_r, axis=1,
 657                               keepdims=True)
 658   bellman_sos = tf.reduce_mean(tf.square(
 659       log_r_tilde - tf.stop_gradient(log_lambda + log_p_x + log_r)), axis=[0, 1])
 660   bellman_loss = tf.reduce_mean(bellman_sos)/num_timesteps
 661   tf.summary.scalar("bellman_loss", bellman_loss)
 662 
 663   if len(tf.get_collection("LOG_P_HAT_VARS")) == 0:
 664     log_p_hat_collection = list(set(tf.trainable_variables()) -
 665                                 set(tf.get_collection("R_TILDE_VARS")))
 666     for v in log_p_hat_collection:
 667       tf.add_to_collection("LOG_P_HAT_VARS", v)
 668 
 669   log_p_hat /= num_timesteps
 670   losses = [Loss("log_p_hat", -tf.reduce_mean(log_p_hat), "LOG_P_HAT_VARS"),
 671             Loss("bellman_loss", bellman_loss, "R_TILDE_VARS")]
 672 
 673   return log_p_hat, losses, tf.no_op(), states[1:], log_weights_all
