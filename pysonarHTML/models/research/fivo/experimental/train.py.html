 123 def combine_grad_lists(grad_lists):
 124   # grads is num_losses by num_variables.
 125   # each list could have different variables.
 126   # for each variable, sum the grads across all losses.
 127   grads_dict = defaultdict(list)
 128   var_dict = {}
 129   for grad_list in grad_lists:
 130     for grad, var in grad_list:
 131       if grad is not None:
 132         grads_dict[var.name].append(grad)
 133       var_dict[var.name] = var
 134 
 135   final_grads = []
 136   for var_name, var in var_dict.iteritems():
 137     grads = grads_dict[var_name]
 138     if len(grads) > 0:
 139       tf.logging.info("Var %s has combined grads from %s." %
 140                       (var_name, [g.name for g in grads]))
 141       grad = tf.reduce_sum(grads, axis=0)
 142     else:
 143       tf.logging.info("Var %s has no grads" % var_name)
 144       grad = None
 145     final_grads.append((grad, var))
 146   return final_grads

 149 def make_apply_grads_op(losses, global_step, learning_rate, lr_decay_steps):
 150   for l in losses:
 151     assert isinstance(l, bounds.Loss)
 152 
 153   lr = tf.train.exponential_decay(
 154       learning_rate, global_step, lr_decay_steps, 0.5, staircase=False)
 155   tf.summary.scalar("learning_rate", lr)
 156   opt = tf.train.AdamOptimizer(lr)
 157 
 158   ema_ops = []
 159   grads = []
 160   for loss_name, loss, loss_var_collection in losses:
 161     tf.logging.info("Computing grads of %s w.r.t. vars in collection %s" %
 162                     (loss_name, loss_var_collection))
 163     g = opt.compute_gradients(loss,
 164                               var_list=tf.get_collection(loss_var_collection))
 165     ema_ops.append(summ.summarize_grads(g, loss_name))
 166     grads.append(g)
 167 
 168   all_grads = combine_grad_lists(grads)
 169   apply_grads_op = opt.apply_gradients(all_grads, global_step=global_step)
 170 
 171   # Update the emas after applying the grads.
 172   with tf.control_dependencies([apply_grads_op]):
 173     train_op = tf.group(*ema_ops)
 174   return train_op

 177 def add_check_numerics_ops():
 178   check_op = []
 179   for op in tf.get_default_graph().get_operations():
 180     bad = ["logits/Log", "sample/Reshape", "log_prob/mul",
 181            "log_prob/SparseSoftmaxCrossEntropyWithLogits/Reshape",
 182            "entropy/Reshape", "entropy/LogSoftmax", "Categorical", "Mean"]
 183     if all([x not in op.name for x in bad]):
 184       for output in op.outputs:
 185         if output.dtype in [tf.float16, tf.float32, tf.float64]:
 186           if op._get_control_flow_context() is not None:  # pylint: disable=protected-access
 187             raise ValueError("`tf.add_check_numerics_ops() is not compatible "
 188                              "with TensorFlow control flow operations such as "
 189                              "`tf.cond()` or `tf.while_loop()`.")
 190 
 191           message = op.name + ":" + str(output.value_index)
 192           with tf.control_dependencies(check_op):
 193             check_op = [tf.check_numerics(output, message=message)]
 194   return tf.group(*check_op)

 197 def create_long_chain_graph(bound, state_size, num_obs, steps_per_obs,
 198                             batch_size, num_samples, num_eval_samples,
 199                             resampling_schedule, use_resampling_grads,
 200                             learning_rate, lr_decay_steps, dtype="float64"):
 201   num_timesteps = num_obs * steps_per_obs + 1
 202   # Make the dataset.
 203   dataset = data.make_long_chain_dataset(
 204       state_size=state_size,
 205       num_obs=num_obs,
 206       steps_per_obs=steps_per_obs,
 207       batch_size=batch_size,
 208       num_samples=num_samples,
 209       variance=FLAGS.variance,
 210       observation_variance=FLAGS.observation_variance,
 211       dtype=dtype,
 212       observation_type=FLAGS.observation_type,
 213       transition_type=FLAGS.transition_type,
 214       fixed_observation=FLAGS.fixed_observation)
 215   itr = dataset.make_one_shot_iterator()
 216   _, observations = itr.get_next()
 217   # Make the dataset for eval
 218   eval_dataset = data.make_long_chain_dataset(
 219       state_size=state_size,
 220       num_obs=num_obs,
 221       steps_per_obs=steps_per_obs,
 222       batch_size=batch_size,
 223       num_samples=num_eval_samples,
 224       variance=FLAGS.variance,
 225       observation_variance=FLAGS.observation_variance,
 226       dtype=dtype,
 227       observation_type=FLAGS.observation_type,
 228       transition_type=FLAGS.transition_type,
 229       fixed_observation=FLAGS.fixed_observation)
 230   eval_itr = eval_dataset.make_one_shot_iterator()
 231   _, eval_observations = eval_itr.get_next()
 232 
 233   # Make the model.
 234   model = models.LongChainModel.create(
 235       state_size,
 236       num_obs,
 237       steps_per_obs,
 238       observation_type=FLAGS.observation_type,
 239       transition_type=FLAGS.transition_type,
 240       variance=FLAGS.variance,
 241       observation_variance=FLAGS.observation_variance,
 242       dtype=tf.as_dtype(dtype),
 243       disable_r=FLAGS.disable_r)
 244 
 245   # Compute the bound and loss
 246   if bound == "iwae":
 247     (_, losses, ema_op, _, _) = bounds.iwae(
 248         model,
 249         observations,
 250         num_timesteps,
 251         num_samples=num_samples)
 252     (eval_log_p_hat, _, _, _, eval_log_weights) = bounds.iwae(
 253         model,
 254         eval_observations,
 255         num_timesteps,
 256         num_samples=num_eval_samples,
 257         summarize=False)
 258     eval_log_p_hat = tf.reduce_mean(eval_log_p_hat)
 259   elif bound == "fivo" or "fivo-aux":
 260     (_, losses, ema_op, _, _) = bounds.fivo(
 261         model,
 262         observations,
 263         num_timesteps,
 264         resampling_schedule=resampling_schedule,
 265         use_resampling_grads=use_resampling_grads,
 266         resampling_type=FLAGS.resampling_method,
 267         aux=("aux" in bound),
 268         num_samples=num_samples)
 269     (eval_log_p_hat, _, _, _, eval_log_weights) = bounds.fivo(
 270         model,
 271         eval_observations,
 272         num_timesteps,
 273         resampling_schedule=resampling_schedule,
 274         use_resampling_grads=False,
 275         resampling_type="multinomial",
 276         aux=("aux" in bound),
 277         num_samples=num_eval_samples,
 278         summarize=False)
 279     eval_log_p_hat = tf.reduce_mean(eval_log_p_hat)
 280 
 281   summ.summarize_ess(eval_log_weights, only_last_timestep=True)
 282 
 283   tf.summary.scalar("log_p_hat", eval_log_p_hat)
 284 
 285   # Compute and apply grads.
 286   global_step = tf.train.get_or_create_global_step()
 287 
 288   apply_grads = make_apply_grads_op(losses,
 289                                     global_step,
 290                                     learning_rate,
 291                                     lr_decay_steps)
 292 
 293   # Update the emas after applying the grads.
 294   with tf.control_dependencies([apply_grads]):
 295     train_op = tf.group(ema_op)
 296 
 297   # We can't calculate the likelihood for most of these models
 298   # so we just return zeros.
 299   eval_likelihood = tf.zeros([], dtype=dtype)
 300   return global_step, train_op, eval_log_p_hat, eval_likelihood

 491 def parse_resampling_schedule(schedule, num_timesteps):
 492   schedule = schedule.strip().lower()
 493   if schedule == "always":
 494     return [True] * (num_timesteps - 1) + [False]
 495   elif schedule == "never":
 496     return [False] * num_timesteps
 497   elif "every" in schedule:
 498     n = int(schedule.split("_")[1])
 499     return [(i+1) % n == 0 for i in xrange(num_timesteps)]
 500   else:
 501     sched = [x.strip() == "true" for x in schedule.split(",")]
 502     assert len(
 503         sched
 504     ) == num_timesteps, "Wrong number of timesteps in resampling schedule."
 505     return sched

 508 def create_log_hook(step, eval_log_p_hat, eval_likelihood):
 509   def summ_formatter(d):
 510     return ("Step {step}, log p_hat: {log_p_hat:.5f} likelihood: {likelihood:.5f}".format(**d))
 511   hook = tf.train.LoggingTensorHook(
 512       {
 513           "step": step,
 514           "log_p_hat": eval_log_p_hat,
 515           "likelihood": eval_likelihood,
 516       },
 517       every_n_iter=FLAGS.summarize_every,
 518       formatter=summ_formatter)
 519   return hook

 531 def main(unused_argv):
 532   if FLAGS.model == "long_chain":
 533     resampling_schedule = parse_resampling_schedule(FLAGS.resampling,
 534                                                     FLAGS.num_timesteps + 1)
 535   else:
 536     resampling_schedule = parse_resampling_schedule(FLAGS.resampling,
 537                                                     FLAGS.num_timesteps)
 538   if FLAGS.random_seed is None:
 539     seed = np.random.randint(0, high=10000)
 540   else:
 541     seed = FLAGS.random_seed
 542   tf.logging.info("Using random seed %d", seed)
 543 
 544   if FLAGS.model == "long_chain":
 545     assert FLAGS.q_type  == "normal", "Q type %s not supported for long chain models" % FLAGS.q_type
 546     assert FLAGS.p_type == "unimodal", "Bimodal priors are not supported for long chain models"
 547     assert not FLAGS.use_bs, "Bs are not supported with long chain models"
 548     assert FLAGS.num_timesteps == FLAGS.num_observations * FLAGS.steps_per_observation, "Num timesteps does not match."
 549     assert FLAGS.bound != "fivo-aux-td", "TD Training is not compatible with long chain models."
 550 
 551   if FLAGS.model == "forward":
 552     if "nonlinear" not in FLAGS.p_type:
 553       assert FLAGS.transition_type == models.STANDARD_TRANSITION, "Non-standard transitions not supported by the forward model."
