  28 def make_long_chain_dataset(
  29     state_size=1,
  30     num_obs=5,
  31     steps_per_obs=3,
  32     variance=1.,
  33     observation_variance=1.,
  34     batch_size=4,
  35     num_samples=1,
  36     observation_type=models.STANDARD_OBSERVATION,
  37     transition_type=models.STANDARD_TRANSITION,
  38     fixed_observation=None,
  39     dtype="float32"):
  40   """Creates a long chain data generating process.
  41 
  42   Creates a tf.data.Dataset that provides batches of data from a long
  43   chain.
  44 
  45   Args:
  46     state_size: The dimension of the state space of the process.
  47     num_obs: The number of observations in the chain.
  48     steps_per_obs: The number of steps between each observation.
  49     variance: The variance of the normal distributions used at each timestep.
  50     batch_size: The number of trajectories to include in each batch.
  51     num_samples: The number of replicas of each trajectory to include in each
  52       batch.
  53     dtype: The datatype of the states and observations.
  54   Returns:
  55     dataset: A tf.data.Dataset that can be iterated over.
  56   """
  57   num_timesteps = num_obs * steps_per_obs
  58   def data_generator():
  59     """An infinite generator of latents and observations from the model."""
  60     while True:
  61       states = []
  62       observations = []
  63       # z0 ~ Normal(0, sqrt(variance)).
  64       states.append(
  65           np.random.normal(size=[state_size],
  66                            scale=np.sqrt(variance)).astype(dtype))
  67       # start at 1 because we've already generated z0
  68       # go to num_timesteps+1 because we want to include the num_timesteps-th step
  69       for t in xrange(1, num_timesteps+1):
  70         if transition_type == models.ROUND_TRANSITION:
  71           loc = np.round(states[-1])
  72         elif transition_type == models.STANDARD_TRANSITION:
  73           loc = states[-1]
  74         new_state = np.random.normal(size=[state_size],
  75                                      loc=loc,
  76                                      scale=np.sqrt(variance))
  77         states.append(new_state.astype(dtype))
  78         if t % steps_per_obs == 0:
  79           if fixed_observation is None:
  80             if observation_type == models.SQUARED_OBSERVATION:
  81               loc = np.square(states[-1])
  82             elif observation_type == models.ABS_OBSERVATION:
  83               loc = np.abs(states[-1])
  84             elif observation_type == models.STANDARD_OBSERVATION:
  85               loc = states[-1]
  86             new_obs = np.random.normal(size=[state_size],
  87                                        loc=loc,
  88                                        scale=np.sqrt(observation_variance)).astype(dtype)
  89           else:
  90             new_obs = np.ones([state_size])* fixed_observation
  91 
  92           observations.append(new_obs)
  93       yield states, observations
  94 
  95   dataset = tf.data.Dataset.from_generator(
  96       data_generator,
  97       output_types=(tf.as_dtype(dtype), tf.as_dtype(dtype)),
  98       output_shapes=([num_timesteps+1, state_size], [num_obs, state_size]))
  99   dataset = dataset.repeat().batch(batch_size)
 100 
 101   def tile_batch(state, observation):
 102     state = tf.tile(state, [num_samples, 1, 1])
 103     observation = tf.tile(observation, [num_samples, 1, 1])
 104     return state, observation
 105 
 106   dataset = dataset.map(tile_batch, num_parallel_calls=12).prefetch(1024)
 107   return dataset

 110 def make_dataset(bs=None,
 111                  state_size=1,
 112                  num_timesteps=10,
 113                  variance=1.,
 114                  prior_type="unimodal",
 115                  bimodal_prior_weight=0.5,
 116                  bimodal_prior_mean=1,
 117                  transition_type=models.STANDARD_TRANSITION,
 118                  fixed_observation=None,
 119                  batch_size=4,
 120                  num_samples=1,
 121                  dtype='float32'):
 122   """Creates a data generating process.
 123 
 124   Creates a tf.data.Dataset that provides batches of data.
 125 
 126   Args:
 127     bs: The parameters of the data generating process. If None, new bs are
 128       randomly generated.
 129     state_size: The dimension of the state space of the process.
 130     num_timesteps: The length of the state sequences in the process.
 131     variance: The variance of the normal distributions used at each timestep.
 132     batch_size: The number of trajectories to include in each batch.
 133     num_samples: The number of replicas of each trajectory to include in each
 134       batch.
 135   Returns:
 136     bs: The true bs used to generate the data
 137     dataset: A tf.data.Dataset that can be iterated over.
 138   """
 139 
 140   if bs is None:
 141     bs = [np.random.uniform(size=[state_size]).astype(dtype) for _ in xrange(num_timesteps)]
 142     tf.logging.info("data generating processs bs: %s",
 143                     np.array(bs).reshape(num_timesteps))
 144 
 145 
 146   def data_generator():
 147     """An infinite generator of latents and observations from the model."""
 148     while True:
 149       states = []
 150       if prior_type == "unimodal" or prior_type == "nonlinear":
 151         # Prior is Normal(0, sqrt(variance)).
 152         states.append(np.random.normal(size=[state_size], scale=np.sqrt(variance)).astype(dtype))
 153       elif prior_type == "bimodal":
 154         if np.random.uniform() > bimodal_prior_weight:
 155           loc = bimodal_prior_mean
 156         else:
 157           loc = - bimodal_prior_mean
 158         states.append(np.random.normal(size=[state_size],
 159                                        loc=loc,
 160                                        scale=np.sqrt(variance)
 161                                       ).astype(dtype))
 162 
 163       for t in xrange(num_timesteps):
 164         if transition_type == models.ROUND_TRANSITION:
 165           loc = np.round(states[-1])
 166         elif transition_type == models.STANDARD_TRANSITION:
 167           loc = states[-1]
 168         loc += bs[t]
 169         new_state = np.random.normal(size=[state_size],
 170                                      loc=loc,
 171                                      scale=np.sqrt(variance)).astype(dtype)
 172         states.append(new_state)
 173 
 174       if fixed_observation is None:
 175         observation = states[-1]
 176       else:
 177         observation = np.ones_like(states[-1]) * fixed_observation
 178       yield np.array(states[:-1]), observation
 179 
 180   dataset = tf.data.Dataset.from_generator(
 181       data_generator,
 182       output_types=(tf.as_dtype(dtype), tf.as_dtype(dtype)),
 183       output_shapes=([num_timesteps, state_size], [state_size]))
 184   dataset = dataset.repeat().batch(batch_size)
 185 
 186   def tile_batch(state, observation):
 187     state = tf.tile(state, [num_samples, 1, 1])
 188     observation = tf.tile(observation, [num_samples, 1])
 189     return state, observation
 190 
 191   dataset = dataset.map(tile_batch, num_parallel_calls=12).prefetch(1024)
 192   return np.array(bs), dataset
