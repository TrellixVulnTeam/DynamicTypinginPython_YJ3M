  16 import os
  17 import sys
  18 sys.path.append(os.path.abspath(os.path.join(__file__, '../../')))
  19 import numpy as np
  20 import tensorflow as tf
  21 from config import get_config
  22 from model_n2nmn.assembler import Assembler
  23 from model_n2nmn.model import Model
  24 from util.data_reader import DataReader
  25 from util.data_reader import SampleBuilder
  26 from util.misc import prepare_dirs_and_logger
  27 from util.misc import save_config
  28 from util.misc import show_all_variables
  29 
  30 
  31 def main(_):
  32   config = prepare_dirs_and_logger(config_raw)
  33   save_config(config)
  34 
  35   rng = np.random.RandomState(config.random_seed)
  36   tf.set_random_seed(config.random_seed)
  37   config.rng = rng
  38 
  39   config.module_names = ['_key_find', '_key_filter', '_val_desc', '<eos>']
  40   config.gt_layout_tokens = ['_key_find', '_key_filter', '_val_desc', '<eos>']
  41   assembler = Assembler(config)
  42 
  43   sample_builder = SampleBuilder(config)
  44   config = sample_builder.config  # update T_encoder according to data
  45   data_train = sample_builder.data_all['train']
  46   data_reader_train = DataReader(
  47       config, data_train, assembler, shuffle=True, one_pass=False)
  48 
  49   num_vocab_txt = len(sample_builder.dict_all)
  50   num_vocab_nmn = len(assembler.module_names)
  51   num_choices = len(sample_builder.dict_all)
  52 
  53   # Network inputs
  54   text_seq_batch = tf.placeholder(tf.int32, [None, None])
  55   seq_len_batch = tf.placeholder(tf.int32, [None])
  56   ans_label_batch = tf.placeholder(tf.int32, [None])
  57   use_gt_layout = tf.constant(True, dtype=tf.bool)
  58   gt_layout_batch = tf.placeholder(tf.int32, [None, None])
  59 
  60   # The model for training
  61   model = Model(
  62       config,
  63       sample_builder.kb,
  64       text_seq_batch,
  65       seq_len_batch,
  66       num_vocab_txt=num_vocab_txt,
  67       num_vocab_nmn=num_vocab_nmn,
  68       EOS_idx=assembler.EOS_idx,
  69       num_choices=num_choices,
  70       decoder_sampling=True,
  71       use_gt_layout=use_gt_layout,
  72       gt_layout_batch=gt_layout_batch)
  73   compiler = model.compiler
  74   scores = model.scores
  75   log_seq_prob = model.log_seq_prob
  76 
  77   # Loss function
  78   softmax_loss_per_sample = tf.nn.sparse_softmax_cross_entropy_with_logits(
  79       logits=scores, labels=ans_label_batch)
  80   # The final per-sample loss, which is loss for valid expr
  81   # and invalid_expr_loss for invalid expr
  82   final_loss_per_sample = softmax_loss_per_sample  # All exprs are valid
  83 
  84   avg_sample_loss = tf.reduce_mean(final_loss_per_sample)
  85   seq_likelihood_loss = tf.reduce_mean(-log_seq_prob)
  86 
  87   total_training_loss = seq_likelihood_loss + avg_sample_loss
  88   total_loss = total_training_loss + config.weight_decay * model.l2_reg
  89 
  90   # Train with Adam optimizer
  91   solver = tf.train.AdamOptimizer()
  92   gradients = solver.compute_gradients(total_loss)
  93 
  94   # Clip gradient by L2 norm
  95   gradients = [(tf.clip_by_norm(g, config.max_grad_norm), v)
  96                for g, v in gradients]
  97   solver_op = solver.apply_gradients(gradients)
  98 
  99   # Training operation
 100   with tf.control_dependencies([solver_op]):
 101     train_step = tf.constant(0)
 102 
 103   # Write summary to TensorBoard
 104   log_writer = tf.summary.FileWriter(config.log_dir, tf.get_default_graph())
 105 
 106   loss_ph = tf.placeholder(tf.float32, [])
 107   entropy_ph = tf.placeholder(tf.float32, [])
 108   accuracy_ph = tf.placeholder(tf.float32, [])
 109   summary_train = [
 110       tf.summary.scalar('avg_sample_loss', loss_ph),
 111       tf.summary.scalar('entropy', entropy_ph),
 112       tf.summary.scalar('avg_accuracy', accuracy_ph)
 113   ]
 114   log_step_train = tf.summary.merge(summary_train)
 115 
 116   # Training
 117   sess = tf.Session()
 118   sess.run(tf.global_variables_initializer())
 119   snapshot_saver = tf.train.Saver(max_to_keep=None)  # keep all snapshots
 120   show_all_variables()
 121 
 122   avg_accuracy = 0
 123   accuracy_decay = 0.99
 124   for n_iter, batch in enumerate(data_reader_train.batches()):
 125     if n_iter >= config.max_iter:
 126       break
 127 
 128     # set up input and output tensors
 129     h = sess.partial_run_setup(
 130         fetches=[
 131             model.predicted_tokens, model.entropy_reg, scores, avg_sample_loss,
 132             train_step
 133         ],
 134         feeds=[
 135             text_seq_batch, seq_len_batch, gt_layout_batch,
 136             compiler.loom_input_tensor, ans_label_batch
 137         ])
 138 
 139     # Part 1: Generate module layout
 140     tokens, entropy_reg_val = sess.partial_run(
 141         h,
 142         fetches=(model.predicted_tokens, model.entropy_reg),
 143         feed_dict={
 144             text_seq_batch: batch['input_seq_batch'],
 145             seq_len_batch: batch['seq_len_batch'],
 146             gt_layout_batch: batch['gt_layout_batch']
 147         })
 148     # Assemble the layout tokens into network structure
 149     expr_list, expr_validity_array = assembler.assemble(tokens)
 150     # all exprs should be valid (since they are ground-truth)
 151     assert np.all(expr_validity_array)
 152     labels = batch['ans_label_batch']
 153     # Build TensorFlow Fold input for NMN
 154     expr_feed = compiler.build_feed_dict(expr_list)
 155     expr_feed[ans_label_batch] = labels
 156 
 157     # Part 2: Run NMN and learning steps
 158     scores_val, avg_sample_loss_val, _ = sess.partial_run(
 159         h, fetches=(scores, avg_sample_loss, train_step), feed_dict=expr_feed)
 160 
 161     # Compute accuracy
 162     predictions = np.argmax(scores_val, axis=1)
 163     accuracy = np.mean(
 164         np.logical_and(expr_validity_array, predictions == labels))
 165     avg_accuracy += (1 - accuracy_decay) * (accuracy - avg_accuracy)
 166 
 167     # Add to TensorBoard summary
 168     if (n_iter + 1) % config.log_interval == 0:
 169       tf.logging.info('iter = %d\n\t'
 170                       'loss = %f, accuracy (cur) = %f, '
 171                       'accuracy (avg) = %f, entropy = %f' %
 172                       (n_iter + 1, avg_sample_loss_val, accuracy, avg_accuracy,
 173                        -entropy_reg_val))
 174       summary = sess.run(
 175           fetches=log_step_train,
 176           feed_dict={
 177               loss_ph: avg_sample_loss_val,
 178               entropy_ph: -entropy_reg_val,
 179               accuracy_ph: avg_accuracy
 180           })
 181       log_writer.add_summary(summary, n_iter + 1)
 182 
 183     # Save snapshot
 184     if (n_iter + 1) % config.snapshot_interval == 0:
 185       snapshot_file = os.path.join(config.model_dir, '%08d' % (n_iter + 1))
 186       snapshot_saver.save(sess, snapshot_file, write_meta_graph=False)
 187       tf.logging.info('Snapshot saved to %s' % snapshot_file)
 188 
 189   tf.logging.info('Run finished.')
 190 
 191 
 192 if __name__ == '__main__':
 193   config_raw, unparsed = get_config()
 194   tf.app.run(main=main, argv=[sys.argv[0]] + unparsed)
