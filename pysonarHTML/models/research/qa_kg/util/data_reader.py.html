  29 class SampleBuilder:
  30 
  31   def __init__(self, config):
  32     self.config = config
  33 
  34     self.kb_raw = self.read_kb()
  35     self.data_raw = self.read_raw_data()
  36 
  37     # dictionary of entities, normal words, and relations
  38     self.dict_all = self.gen_dict()
  39     self.reverse_dict_all = dict(
  40         zip(self.dict_all.values(), self.dict_all.keys()))
  41 
  42     tf.logging.info('size of dict: %d' % len(self.dict_all))
  43 
  44     self.kb = self.build_kb()
  45     self.data_all = self.build_samples()
  46 
  47   def read_kb(self):
  48     kb_raw = []
  49     for line in open(self.config.KB_file):
  50       sub, rel, obj = line.strip().split('|')
  51       kb_raw.append((sub, rel, obj))
  52     tf.logging.info('# of KB records: %d' % len(kb_raw))
  53     return kb_raw
  54 
  55   def read_raw_data(self):
  56     data = dict()
  57     for name in self.config.data_files:
  58       raw = []
  59       tf.logging.info(
  60         'Reading data file {}'.format(self.config.data_files[name]))
  61       for line in open(self.config.data_files[name]):
  62         question, answers = line.strip().split('\t')
  63         question = question.replace('],', ']')  # ignore ',' in the template
  64         raw.append((question, answers))
  65       data[name] = raw
  66     return data
  67 
  68   def build_kb(self):
  69     tf.logging.info('Indexing KB...')
  70     kb = []
  71     for sub, rel, obj in self.kb_raw:
  72       kb.append([self.dict_all[sub], self.dict_all[rel], self.dict_all[obj]])
  73     return kb
  74 
  75   def gen_dict(self):
  76     s = set()
  77     for sub, rel, obj in self.kb_raw:
  78       s.add(sub)
  79       s.add(rel)
  80       s.add(obj)
  81     for name in self.data_raw:
  82       for question, answers in self.data_raw[name]:
  83         normal = re.split('\[[^\]]+\]', question)
  84         for phrase in normal:
  85           for word in phrase.split():
  86             s.add(word)
  87     s = list(s)
  88     d = {s[idx]: idx for idx in range(len(s))}
  89     return d
  90 
  91   def build_samples(self):
  92 
  93     def map_entity_idx(text):
  94       entities = re.findall('\[[^\]]+\]', text)
  95       for entity in entities:
  96         entity = entity[1:-1]
  97         index = self.dict_all[entity]
  98         text = text.replace('[%s]' % entity, '@%d' % index)
  99       return text
 100 
 101     data_all = dict()
 102 
 103     for name in self.data_raw:
 104       X, Y, MultiYs, qid = [], [], [], []
 105       for i, (question, answers) in enumerate(self.data_raw[name]):
 106         qdata, labels = [], []
 107         question = map_entity_idx(question)
 108         for word in question.split():
 109           if word[0] == '@':
 110             qdata.append(int(word[1:]))
 111           else:
 112             qdata.append(self.dict_all[word])
 113         for answer in answers.split('|'):
 114           labels.append(self.dict_all[answer])
 115         if len(qdata) > self.config.T_encoder:
 116           self.config.T_encoder = len(qdata)
 117         for label in labels:
 118           X.append(qdata)
 119           Y.append(label)
 120           MultiYs.append(set(labels))
 121           qid.append(i)
 122       data_all[name] = Data(X=X, Y=Y, MultiYs=MultiYs, qid=qid)
 123 
 124     return data_all

 127 def _run_prefetch(prefetch_queue, batch_loader, data, shuffle, one_pass,
 128                   config):
 129   assert len(data.X) == len(data.Y) == len(data.MultiYs) == len(data.qid)
 130   num_samples = len(data.X)
 131   batch_size = config.batch_size
 132 
 133   n_sample = 0
 134   fetch_order = config.rng.permutation(num_samples)
 135   while True:
 136     sample_ids = fetch_order[n_sample:n_sample + batch_size]
 137     batch = batch_loader.load_one_batch(sample_ids)
 138     prefetch_queue.put(batch, block=True)
 139 
 140     n_sample += len(sample_ids)
 141     if n_sample >= num_samples:
 142       if one_pass:
 143         prefetch_queue.put(None, block=True)
 144       n_sample = 0
 145       if shuffle:
 146         fetch_order = config.rng.permutation(num_samples)

 149 class DataReader:
 150   def __init__(self,
 151                config,
 152                data,
 153                assembler,
 154                shuffle=True,
 155                one_pass=False,
 156                prefetch_num=10):
 157     self.config = config
 158 
 159     self.data = data
 160     self.assembler = assembler
 161     self.batch_loader = BatchLoader(self.config,
 162                                     self.data, self.assembler)
 163 
 164     self.shuffle = shuffle
 165     self.one_pass = one_pass
 166     self.prefetch_queue = Queue(maxsize=prefetch_num)
 167     self.prefetch_thread = threading.Thread(target=_run_prefetch,
 168                                             args=(self.prefetch_queue,
 169                                                   self.batch_loader, self.data,
 170                                                   self.shuffle, self.one_pass,
 171                                                   self.config))
 172     self.prefetch_thread.daemon = True
 173     self.prefetch_thread.start()
 174 
 175   def batches(self):
 176     while True:
 177       if self.prefetch_queue.empty():
 178         tf.logging.warning('Waiting for data loading (IO is slow)...')
 179       batch = self.prefetch_queue.get(block=True)
 180       if batch is None:
 181         assert self.one_pass
 182         tf.logging.info('One pass finished!')
 183         raise StopIteration()
 184       yield batch

 187 class BatchLoader:
 188   def __init__(self, config,
 189                data, assembler):
 190     self.config = config
 191 
 192     self.data = data
 193     self.assembler = assembler
 194 
 195     self.T_encoder = config.T_encoder
 196     self.T_decoder = config.T_decoder
 197 
 198     tf.logging.info('T_encoder: %d' % self.T_encoder)
 199     tf.logging.info('T_decoder: %d' % self.T_decoder)
 200     tf.logging.info('batch size: %d' % self.config.batch_size)
 201 
 202     self.gt_layout_tokens = config.gt_layout_tokens
 203 
 204   def load_one_batch(self, sample_ids):
 205     actual_batch_size = len(sample_ids)
 206     input_seq_batch = np.zeros((self.T_encoder, actual_batch_size), np.int32)
 207     seq_len_batch = np.zeros(actual_batch_size, np.int32)
 208     ans_label_batch = np.zeros(actual_batch_size, np.int32)
 209     ans_set_labels_list = [None] * actual_batch_size
 210     question_id_list = [None] * actual_batch_size
 211     gt_layout_batch = np.zeros((self.T_decoder, actual_batch_size), np.int32)
 212 
 213     for batch_i in range(actual_batch_size):
 214       idx = sample_ids[batch_i]
 215       seq_len = len(self.data.X[idx])
 216       seq_len_batch[batch_i] = seq_len
 217       input_seq_batch[:seq_len, batch_i] = self.data.X[idx]
 218       ans_label_batch[batch_i] = self.data.Y[idx]
 219       ans_set_labels_list[batch_i] = self.data.MultiYs[idx]
 220       question_id_list[batch_i] = self.data.qid[idx]
 221 
 222       gt_layout_batch[:, batch_i] = self.assembler.module_list2tokens(
 223         self.gt_layout_tokens, self.T_decoder)
 224 
 225     batch = dict(input_seq_batch=input_seq_batch,
 226                  seq_len_batch=seq_len_batch,
 227                  ans_label_batch=ans_label_batch,
 228                  gt_layout_batch=gt_layout_batch,
 229                  ans_set_labels_list=ans_set_labels_list,
 230                  question_id_list=question_id_list)
 231     return batch
