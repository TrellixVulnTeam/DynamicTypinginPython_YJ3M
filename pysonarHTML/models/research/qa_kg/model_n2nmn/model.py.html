  16 import numpy as np
  17 import tensorflow as tf
  18 import tensorflow_fold as td
  19 from model_n2nmn import netgen_att
  20 from model_n2nmn import assembler
  21 from model_n2nmn.modules import Modules
  22 
  23 
  24 class Model:
  25 
  26   def __init__(self,
  27                config,
  28                kb,
  29                text_seq_batch,
  30                seq_length_batch,
  31                num_vocab_txt,
  32                num_vocab_nmn,
  33                EOS_idx,
  34                num_choices,
  35                decoder_sampling,
  36                use_gt_layout=None,
  37                gt_layout_batch=None,
  38                scope='neural_module_network',
  39                reuse=None):
  40     with tf.variable_scope(scope, reuse=reuse):
  41       # Part 1: Seq2seq RNN to generate module layout tokens
  42 
  43       embedding_mat = tf.get_variable(
  44         'embedding_mat', [num_vocab_txt, config.embed_dim_txt],
  45         initializer=tf.contrib.layers.xavier_initializer())
  46 
  47       with tf.variable_scope('layout_generation'):
  48         att_seq2seq = netgen_att.AttentionSeq2Seq(
  49             config, text_seq_batch, seq_length_batch, num_vocab_txt,
  50             num_vocab_nmn, EOS_idx, decoder_sampling, embedding_mat,
  51             use_gt_layout, gt_layout_batch)
  52         self.att_seq2seq = att_seq2seq
  53         predicted_tokens = att_seq2seq.predicted_tokens
  54         token_probs = att_seq2seq.token_probs
  55         word_vecs = att_seq2seq.word_vecs
  56         neg_entropy = att_seq2seq.neg_entropy
  57         self.atts = att_seq2seq.atts
  58 
  59         self.predicted_tokens = predicted_tokens
  60         self.token_probs = token_probs
  61         self.word_vecs = word_vecs
  62         self.neg_entropy = neg_entropy
  63 
  64         # log probability of each generated sequence
  65         self.log_seq_prob = tf.reduce_sum(tf.log(token_probs), axis=0)
  66 
  67       # Part 2: Neural Module Network
  68       with tf.variable_scope('layout_execution'):
  69         modules = Modules(config, kb, word_vecs, num_choices, embedding_mat)
  70         self.modules = modules
  71         # Recursion of modules
  72         att_shape = [len(kb)]
  73         # Forward declaration of module recursion
  74         att_expr_decl = td.ForwardDeclaration(td.PyObjectType(),
  75                                               td.TensorType(att_shape))
  76         # _key_find
  77         case_key_find = td.Record([('time_idx', td.Scalar(dtype='int32')),
  78                                    ('batch_idx', td.Scalar(dtype='int32'))])
  79         case_key_find = case_key_find >> td.ScopedLayer(
  80             modules.KeyFindModule, name_or_scope='KeyFindModule')
  81         # _key_filter
  82         case_key_filter = td.Record([('input_0', att_expr_decl()),
  83                                      ('time_idx', td.Scalar('int32')),
  84                                      ('batch_idx', td.Scalar('int32'))])
  85         case_key_filter = case_key_filter >> td.ScopedLayer(
  86             modules.KeyFilterModule, name_or_scope='KeyFilterModule')
  87         recursion_cases = td.OneOf(
  88             td.GetItem('module'),
  89             {'_key_find': case_key_find,
  90              '_key_filter': case_key_filter})
  91         att_expr_decl.resolve_to(recursion_cases)
  92         # _val_desc: output scores for choice (for valid expressions)
  93         predicted_scores = td.Record([('input_0', recursion_cases),
  94                                       ('time_idx', td.Scalar('int32')),
  95                                       ('batch_idx', td.Scalar('int32'))])
  96         predicted_scores = predicted_scores >> td.ScopedLayer(
  97             modules.ValDescribeModule, name_or_scope='ValDescribeModule')
  98 
  99         # For invalid expressions, define a dummy answer
 100         # so that all answers have the same form
 101         INVALID = assembler.INVALID_EXPR
 102         dummy_scores = td.Void() >> td.FromTensor(
 103             np.zeros(num_choices, np.float32))
 104         output_scores = td.OneOf(
 105             td.GetItem('module'),
 106             {'_val_desc': predicted_scores,
 107              INVALID: dummy_scores})
 108 
 109         # compile and get the output scores
 110         self.compiler = td.Compiler.create(output_scores)
 111         self.scores = self.compiler.output_tensors[0]
 112 
 113       # Regularization: Entropy + L2
 114       self.entropy_reg = tf.reduce_mean(neg_entropy)
 115       module_weights = [
 116           v for v in tf.trainable_variables()
 117           if (scope in v.op.name and v.op.name.endswith('weights'))
 118       ]
 119       self.l2_reg = tf.add_n([tf.nn.l2_loss(v) for v in module_weights])
