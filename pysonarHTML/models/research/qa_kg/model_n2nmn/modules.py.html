  21   def __init__(self, config, kb, word_vecs, num_choices, embedding_mat):
  22     self.config = config
  23 
  24     self.embedding_mat = embedding_mat
  25 
  26     # kb has shape [N_kb, 3]
  27     self.kb = kb
  28     self.embed_keys_e, self.embed_keys_r, self.embed_vals_e = self.embed_kb()
  29 
  30     # word_vecs has shape [T_decoder, N, D_txt]
  31     self.word_vecs = word_vecs
  32     self.num_choices = num_choices
  33 
  34   def embed_kb(self):
  35     keys_e, keys_r, vals_e = [], [], []
  36     for idx_sub, idx_rel, idx_obj in self.kb:
  37       keys_e.append(idx_sub)
  38       keys_r.append(idx_rel)
  39       vals_e.append(idx_obj)
  40     embed_keys_e = tf.nn.embedding_lookup(self.embedding_mat, keys_e)
  41     embed_keys_r = tf.nn.embedding_lookup(self.embedding_mat, keys_r)
  42     embed_vals_e = tf.nn.embedding_lookup(self.embedding_mat, vals_e)
  43     return embed_keys_e, embed_keys_r, embed_vals_e
  44 
  45   def _slice_word_vecs(self, time_idx, batch_idx):
  46     # this callable will be wrapped into a td.Function
  47     # In TF Fold, batch_idx and time_idx are both [N_batch, 1] tensors
  48     # time is highest dim in word_vecs
  49     joint_index = tf.stack([time_idx, batch_idx], axis=1)
  50     return tf.gather_nd(self.word_vecs, joint_index)

  52   # All the layers are wrapped with td.ScopedLayer
  53   def KeyFindModule(self,
  54                     time_idx,
  55                     batch_idx,
  56                     scope='KeyFindModule',
  57                     reuse=None):
  58     # In TF Fold, batch_idx and time_idx are both [N_batch, 1] tensors
  59     text_param = self._slice_word_vecs(time_idx, batch_idx)
  60 
  61     # Mapping: embed_keys_e x text_param -> att
  62     # Input:
  63     #   embed_keys_e: [N_kb, D_txt]
  64     #   text_param: [N, D_txt]
  65     # Output:
  66     #   att: [N, N_kb]
  67     #
  68     # Implementation:
  69     #   1. Elementwise multiplication between embed_key_e and text_param
  70     #   2. L2-normalization
  71     with tf.variable_scope(scope, reuse=reuse):
  72       m = tf.matmul(text_param, self.embed_keys_e, transpose_b=True)
  73       att = tf.nn.l2_normalize(m, dim=1)
  74     return att
  75 
  76   def KeyFilterModule(self,
  77                       input_0,
  78                       time_idx,
  79                       batch_idx,
  80                       scope='KeyFilterModule',
  81                       reuse=None):
  82     att_0 = input_0
  83     text_param = self._slice_word_vecs(time_idx, batch_idx)
  84 
  85     # Mapping: and(embed_keys_r x text_param, att) -> att
  86     # Input:
  87     #   embed_keys_r: [N_kb, D_txt]
  88     #   text_param: [N, D_txt]
  89     #   att_0: [N, N_kb]
  90     # Output:
  91     #   att: [N, N_kb]
  92     #
  93     # Implementation:
  94     #   1. Elementwise multiplication between embed_key_r and text_param
  95     #   2. L2-normalization
  96     #   3. Take the elementwise-min
  97     with tf.variable_scope(scope, reuse=reuse):
  98       m = tf.matmul(text_param, self.embed_keys_r, transpose_b=True)
  99       att_1 = tf.nn.l2_normalize(m, dim=1)
 100       att = tf.minimum(att_0, att_1)
 101     return att

 103   def ValDescribeModule(self,
 104                         input_0,
 105                         time_idx,
 106                         batch_idx,
 107                         scope='ValDescribeModule',
 108                         reuse=None):
 109     att = input_0
 110 
 111     # Mapping: att -> answer probs
 112     # Input:
 113     #   embed_vals_e: [N_kb, D_txt]
 114     #   att: [N, N_kb]
 115     #   embedding_mat: [self.num_choices, D_txt]
 116     # Output:
 117     #   answer_scores: [N, self.num_choices]
 118     #
 119     # Implementation:
 120     #   1. Attention-weighted sum over values
 121     #   2. Compute cosine similarity scores between the weighted sum and
 122     #      each candidate answer
 123     with tf.variable_scope(scope, reuse=reuse):
 124       # weighted_sum has shape [N, D_txt]
 125       weighted_sum = tf.matmul(att, self.embed_vals_e)
 126       # scores has shape [N, self.num_choices]
 127       scores = tf.matmul(
 128           weighted_sum,
 129           tf.nn.l2_normalize(self.embedding_mat, dim=1),
 130           transpose_b=True)
 131     return scores
