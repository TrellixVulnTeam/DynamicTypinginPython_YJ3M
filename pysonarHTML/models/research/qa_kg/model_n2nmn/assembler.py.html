  16 import numpy as np
  17 
  18 # the number of attention input to each module
  19 _module_input_num = {
  20     '_key_find': 0,
  21     '_key_filter': 1,
  22     '_val_desc': 1}
  23 _module_output_type = {
  24     '_key_find': 'att',
  25     '_key_filter': 'att',
  26     '_val_desc': 'ans'
  27 }
  28 
  29 INVALID_EXPR = 'INVALID_EXPR'
  30 
  31 
  32 class Assembler:
  33 
  34   def __init__(self, config):
  35     # read the module list, and record the index of each module and <eos>
  36     self.module_names = config.module_names
  37     # find the index of <eos>
  38     for n_s in range(len(self.module_names)):
  39       if self.module_names[n_s] == '<eos>':
  40         self.EOS_idx = n_s
  41         break
  42     # build a dictionary from module name to token index
  43     self.name2idx_dict = {
  44         name: n_s
  45         for n_s, name in enumerate(self.module_names)
  46     }
  47 
  48   def module_list2tokens(self, module_list, max_len=None):
  49     layout_tokens = [self.name2idx_dict[name] for name in module_list]
  50     if max_len is not None:
  51       if len(module_list) >= max_len:
  52         raise ValueError('Not enough time steps to add <eos>')
  53       layout_tokens += [self.EOS_idx] * (max_len - len(module_list))
  54     return layout_tokens
  55 
  56   def _layout_tokens2str(self, layout_tokens):
  57     return ' '.join([self.module_names[idx] for idx in layout_tokens])
  58 
  59   def _invalid_expr(self, layout_tokens, error_str):
  60     return {
  61         'module': INVALID_EXPR,
  62         'expr_str': self._layout_tokens2str(layout_tokens),
  63         'error': error_str
  64     }
  65 
  66   def _assemble_layout_tokens(self, layout_tokens, batch_idx):
  67     # Every module takes a time_idx as the index from LSTM hidden states
  68     # (even if it doesn't need it, like _and), and different arity of
  69     # attention inputs. The output type can be either attention or answer
  70     #
  71     # The final assembled expression for each instance is as follows:
  72     # expr_type :=
  73     #    {'module': '_find',        'output_type': 'att', 'time_idx': idx}
  74     #  | {'module': '_relocate',   'output_type': 'att', 'time_idx': idx,
  75     #     'inputs_0': <expr_type>}
  76     #  | {'module': '_and',         'output_type': 'att', 'time_idx': idx,
  77     #     'inputs_0': <expr_type>,  'inputs_1': <expr_type>)}
  78     #  | {'module': '_describe',      'output_type': 'ans', 'time_idx': idx,
  79     #     'inputs_0': <expr_type>}
  80     #  | {'module': INVALID_EXPR, 'expr_str': '...', 'error': '...',
  81     #     'assembly_loss': <float32>} (for invalid expressions)
  82     #
  83 
  84     # A valid layout must contain <eos>. Assembly fails if it doesn't.
  85     if not np.any(layout_tokens == self.EOS_idx):
  86       return self._invalid_expr(layout_tokens, 'cannot find <eos>')
  87 
  88     # Decoding Reverse Polish Notation with a stack
  89     decoding_stack = []
  90     for t in range(len(layout_tokens)):
  91       # decode a module/operation
  92       module_idx = layout_tokens[t]
  93       if module_idx == self.EOS_idx:
  94         break
  95       module_name = self.module_names[module_idx]
  96       expr = {
  97           'module': module_name,
  98           'output_type': _module_output_type[module_name],
  99           'time_idx': t,
 100           'batch_idx': batch_idx
 101       }
 102 
 103       input_num = _module_input_num[module_name]
 104       # Check if there are enough input in the stack
 105       if len(decoding_stack) < input_num:
 106         # Invalid expression. Not enough input.
 107         return self._invalid_expr(layout_tokens,
 108                                   'not enough input for ' + module_name)
 109 
 110       # Get the input from stack
 111       for n_input in range(input_num - 1, -1, -1):
 112         stack_top = decoding_stack.pop()
 113         if stack_top['output_type'] != 'att':
 114           # Invalid expression. Input must be attention
 115           return self._invalid_expr(layout_tokens,
 116                                     'input incompatible for ' + module_name)
 117         expr['input_%d' % n_input] = stack_top
 118 
 119       decoding_stack.append(expr)
 120 
 121     # After decoding the reverse polish expression, there should be exactly
 122     # one expression in the stack
 123     if len(decoding_stack) != 1:
 124       return self._invalid_expr(
 125           layout_tokens,
 126           'final stack size not equal to 1 (%d remains)' % len(decoding_stack))
 127 
 128     result = decoding_stack[0]
 129     # The result type should be answer, not attention
 130     if result['output_type'] != 'ans':
 131       return self._invalid_expr(layout_tokens,
 132                                 'result type must be ans, not att')
 133     return result
 134 
 135   def assemble(self, layout_tokens_batch):
 136     # layout_tokens_batch is a numpy array with shape [max_dec_len, batch_size],
 137     # containing module tokens and <eos>, in Reverse Polish Notation.
 138     _, batch_size = layout_tokens_batch.shape
 139     expr_list = [
 140         self._assemble_layout_tokens(layout_tokens_batch[:, batch_i], batch_i)
 141         for batch_i in range(batch_size)
 142     ]
 143     expr_validity = np.array(
 144         [expr['module'] != INVALID_EXPR for expr in expr_list], np.bool)
 145     return expr_list, expr_validity
