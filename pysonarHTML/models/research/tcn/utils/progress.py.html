  16 """A utility class for reporting processing progress."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import datetime
  23 
  24 
  25 class Progress(object):
  26   """A utility class for reporting processing progress."""
  27 
  28   def __init__(self, target_size):
  29     self.target_size = target_size
  30     self.current_size = 0
  31     self.start_time = datetime.datetime.now()
  32 
  33   def Update(self, current_size):
  34     """Replaces internal current_size with current_size."""
  35     self.current_size = current_size
  36 
  37   def Add(self, size):
  38     """Increments internal current_size by size."""
  39     self.current_size += size
  40 
  41   def __str__(self):
  42     processed = 1e-5 + self.current_size / float(self.target_size)
  43     current_time = datetime.datetime.now()
  44     elapsed = current_time - self.start_time
  45     eta = datetime.timedelta(
  46         seconds=elapsed.total_seconds() / processed - elapsed.total_seconds())
  47     return "%d / %d (elapsed %s eta %s)" % (
  48         self.current_size, self.target_size,
  49         str(elapsed).split(".")[0],
  50         str(eta).split(".")[0])
