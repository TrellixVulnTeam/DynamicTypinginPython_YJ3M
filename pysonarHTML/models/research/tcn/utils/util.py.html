  31 def GetFilesRecursively(topdir):
  32   """Gets all records recursively for some topdir.
  33 
  34   Args:
  35     topdir: String, path to top directory.
  36   Returns:
  37     allpaths: List of Strings, full paths to all leaf records.
  38   Raises:
  39     ValueError: If there are no files found for this directory.
  40   """
  41   assert topdir
  42   topdir = os.path.expanduser(topdir)
  43   allpaths = []
  44   for path, _, leaffiles in tf.gfile.Walk(topdir):
  45     if leaffiles:
  46       allpaths.extend([os.path.join(path, i) for i in leaffiles])
  47   if not allpaths:
  48     raise ValueError('No files found for top directory %s' % topdir)
  49   return allpaths

  52 def NoDuplicatesConstructor(loader, node, deep=False):
  53   """Check for duplicate keys."""
  54   mapping = {}
  55   for key_node, value_node in node.value:
  56     key = loader.construct_object(key_node, deep=deep)
  57     value = loader.construct_object(value_node, deep=deep)
  58     if key in mapping:
  59       raise ConstructorError('while constructing a mapping', node.start_mark,
  60                              'found duplicate key (%s)' % key,
  61                              key_node.start_mark)
  62     mapping[key] = value
  63   return loader.construct_mapping(node, deep)

  66 def WriteConfigAsYaml(config, logdir, filename):
  67   """Writes a config dict as yaml to logdir/experiment.yml."""
  68   if not tf.gfile.Exists(logdir):
  69     tf.gfile.MakeDirs(logdir)
  70   config_filename = os.path.join(logdir, filename)
  71   with tf.gfile.GFile(config_filename, 'w') as f:
  72     f.write(yaml.dump(config))
  73   tf.logging.info('wrote config to %s', config_filename)

  76 def LoadConfigDict(config_paths, model_params):
  77   """Loads config dictionary from specified yaml files or command line yaml."""
  78 
  79   # Ensure that no duplicate keys can be loaded (causing pain).
  80   yaml.add_constructor(yaml.resolver.BaseResolver.DEFAULT_MAPPING_TAG,
  81                        NoDuplicatesConstructor)
  82 
  83   # Handle either ',' or '#' separated config lists, since borg will only
  84   # accept '#'.
  85   sep = ',' if ',' in config_paths else '#'
  86 
  87   # Load flags from config file.
  88   final_config = {}
  89   if config_paths:
  90     for config_path in config_paths.split(sep):
  91       config_path = config_path.strip()
  92       if not config_path:
  93         continue
  94       config_path = os.path.abspath(config_path)
  95       tf.logging.info('Loading config from %s', config_path)
  96       with tf.gfile.GFile(config_path.strip()) as config_file:
  97         config_flags = yaml.load(config_file)
  98         final_config = DeepMergeDict(final_config, config_flags)
  99   if model_params:
 100     model_params = MaybeLoadYaml(model_params)
 101     final_config = DeepMergeDict(final_config, model_params)
 102   tf.logging.info('Final Config:\n%s', yaml.dump(final_config))
 103   return final_config

 106 def MaybeLoadYaml(item):
 107   """Parses item if it's a string. If it's a dictionary it's returned as-is."""
 108   if isinstance(item, six.string_types):
 109     return yaml.load(item)
 110   elif isinstance(item, dict):
 111     return item
 112   else:
 113     raise ValueError('Got {}, expected YAML string or dict', type(item))

 116 def DeepMergeDict(dict_x, dict_y, path=None):
 117   """Recursively merges dict_y into dict_x."""
 118   if path is None: path = []
 119   for key in dict_y:
 120     if key in dict_x:
 121       if isinstance(dict_x[key], dict) and isinstance(dict_y[key], dict):
 122         DeepMergeDict(dict_x[key], dict_y[key], path + [str(key)])
 123       elif dict_x[key] == dict_y[key]:
 124         pass  # same leaf value
 125       else:
 126         dict_x[key] = dict_y[key]
 127     else:
 128       dict_x[key] = dict_y[key]
 129   return dict_x

 132 def ParseConfigsToLuaTable(config_paths, extra_model_params=None,
 133                            save=False, save_name='final_training_config.yml',
 134                            logdir=None):
 135   """Maps config_paths and extra_model_params to a Luatable-like object."""
 136   # Parse config dict from yaml config files / command line flags.
 137   config = LoadConfigDict(config_paths, extra_model_params)
 138   if save:
 139     WriteConfigAsYaml(config, logdir, save_name)
 140   # Convert config dictionary to T object with dot notation.
 141   config = RecursivelyConvertToLuatable(config)
 142   return config

 145 def SetNestedValue(d, keys, value):
 146   """Sets a value in a nested dictionary.
 147 
 148   Example:
 149     d = {}, keys = ['data','augmentation','minscale'], value = 1.0.
 150     returns {'data': {'augmentation' : {'minscale': 1.0 }}}
 151 
 152   Args:
 153     d: A dictionary to set a nested value in.
 154     keys: list of dict keys nesting left to right.
 155     value: the nested value to set.
 156   Returns:
 157     None
 158   """
 159   for key in keys[:-1]:
 160     d = d.setdefault(key, {})
 161   d[keys[-1]] = value

 164 def RecursivelyConvertToLuatable(yaml_dict):
 165   """Converts a dictionary to a LuaTable-like T object."""
 166   if isinstance(yaml_dict, dict):
 167     yaml_dict = T(yaml_dict)
 168   for key, item in yaml_dict.iteritems():
 169     if isinstance(item, dict):
 170       yaml_dict[key] = RecursivelyConvertToLuatable(item)
 171   return yaml_dict

 174 def KNNIds(query_vec, target_seq, k=1):
 175   """Gets the knn ids to the query vec from the target sequence."""
 176   sorted_distances = KNNIdsWithDistances(query_vec, target_seq, k)
 177   return [i[0] for i in sorted_distances]
 178 
 179 
 180 def KNNIdsWithDistances(query_vec, target_seq, k=1):
 181   """Gets the knn ids to the query vec from the target sequence."""
 182   if not isinstance(np.array(target_seq), np.ndarray):
 183     target_seq = np.array(target_seq)
 184   assert np.shape(query_vec) == np.shape(target_seq[0])
 185   distances = [(i, np.linalg.norm(query_vec-target_vec)) for (
 186       i, target_vec) in enumerate(target_seq)]
 187   sorted_distances = sorted(distances, key=lambda x: x[1])
 188   return sorted_distances[:k]

 191 def CopyLocalConfigsToCNS(outdir, configs, gfs_user):
 192   """Copies experiment yaml config files to the job_logdir on /cns."""
 193   assert configs
 194   assert outdir
 195   conf_files = configs.split(',')
 196   for conf_file in conf_files:
 197     copy_command = 'fileutil --gfs_user %s cp -f %s %s' % (
 198         gfs_user, conf_file, outdir)
 199     tf.logging.info(copy_command)
 200     os.system(copy_command)

 203 def pairwise_distances(feature, squared=True):
 204   """Computes the pairwise distance matrix in numpy.
 205 
 206   Args:
 207     feature: 2-D numpy array of size [number of data, feature dimension]
 208     squared: Boolean. If true, output is the pairwise squared euclidean
 209       distance matrix; else, output is the pairwise euclidean distance matrix.
 210 
 211   Returns:
 212     pdists: 2-D numpy array of size
 213       [number of data, number of data].
 214   """
 215   triu = np.triu_indices(feature.shape[0], 1)
 216   upper_tri_pdists = np.linalg.norm(feature[triu[1]] - feature[triu[0]], axis=1)
 217   if squared:
 218     upper_tri_pdists **= 2.
 219   num_data = feature.shape[0]
 220   pdists = np.zeros((num_data, num_data))
 221   pdists[np.triu_indices(num_data, 1)] = upper_tri_pdists
 222   # Make symmetrical.
 223   pdists = pdists + pdists.T - np.diag(
 224       pdists.diagonal())
 225   return pdists

 228 def is_tfrecord_input(inp):
 229   """Checks if input is a TFRecord or list of TFRecords."""
 230   def _is_tfrecord(inp):
 231     if not isinstance(inp, str):
 232       return False
 233     _, extension = os.path.splitext(inp)
 234     return extension == '.tfrecord'
 235   if isinstance(inp, str):
 236     return _is_tfrecord(inp)
 237   if isinstance(inp, list):
 238     return all(map(_is_tfrecord, inp))
 239   return False

 242 def is_np_array(inp):
 243   if isinstance(inp, np.ndarray):
 244     return True
 245   if isinstance(inp, list):
 246     return all([isinstance(i, np.ndarray) for i in inp])
 247   return False
