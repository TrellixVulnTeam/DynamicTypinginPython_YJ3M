  17 """A convenience class replicating some lua table syntax with a python dict.
  18 
  19 In general, should behave like a dictionary except that we can use dot notation
  20  to access keys. Users should be careful to only provide keys suitable for
  21  instance variable names.
  22 
  23 Nota bene: do not use the key "keys" since it will collide with the method keys.
  24 
  25 Usage example:
  26 
  27 >>> t = T(a=5,b='kaw', c=T(v=[],x=33))
  28 >>> t.a
  29 5
  30 >>> t.z = None
  31 >>> print t
  32 T(a=5, z=None, c=T(x=33, v=[]), b='kaw')
  33 
  34 >>> t2 = T({'h':'f','x':4})
  35 >>> t2
  36 T(h='f', x=4)
  37 >>> t2['x']
  38 4
  39 """
  40 
  41 
  42 class T(object):
  43   """Class for emulating lua tables."""
  44 
  45   def __init__(self, *args, **kwargs):
  46     if len(args) > 1 or (len(args) == 1 and len(kwargs) > 0):
  47       errmsg = '''constructor only allows a single dict as a positional
  48       argument or keyword arguments'''
  49       raise ValueError(errmsg)
  50     if len(args) == 1 and isinstance(args[0], dict):
  51       self.__dict__.update(args[0])
  52     else:
  53       self.__dict__.update(kwargs)
  54 
  55   def __repr__(self):
  56     fmt = ', '.join('%s=%s' for i in range(len(self.__dict__)))
  57     kwargstr = fmt % tuple(
  58         x for tup in self.__dict__.items() for x in [str(tup[0]), repr(tup[1])])
  59     return 'T(' + kwargstr + ')'
  60 
  61   def __getitem__(self, key):
  62     return self.__dict__[key]
  63 
  64   def __setitem__(self, key, val):
  65     self.__dict__[key] = val
  66 
  67   def __delitem__(self, key):
  68     del self.__dict__[key]
  69 
  70   def __iter__(self):
  71     return iter(self.__dict__)
  72 
  73   def __len__(self):
  74     return len(self.__dict__)
  75 
  76   def keys(self):  # Needed for dict(T( ... )) to work.
  77     return self.__dict__.keys()
  78 
  79   def iteritems(self):
  80     return self.__dict__.iteritems()
