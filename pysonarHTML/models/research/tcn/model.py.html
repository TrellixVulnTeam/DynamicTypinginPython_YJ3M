  30 def get_embedder(
  31     embedder_strategy, config, images, is_training, reuse=False,
  32     l2_normalize_embedding=True):
  33   """Returns an embedder based on config.
  34 
  35   Args:
  36     embedder_strategy: String, name of embedder version to return.
  37     config: LuaTable object, training config.
  38     images: 4-D float `Tensor` containing batch images.
  39     is_training: Boolean or placeholder for boolean,
  40       indicator for whether or not we're training.
  41     reuse: Boolean: Reuse embedder variable scope.
  42     l2_normalize_embedding: Boolean, whether or not to l2 normalize the
  43       embedding.
  44   Returns:
  45     embedder: An `Embedder` object.
  46   Raises:
  47     ValueError: if unknown embedder_strategy specified.
  48   """
  49   if embedder_strategy == 'inception_baseline':
  50     pretrained_ckpt = config.inception_conv_ss_fc.pretrained_checkpoint
  51     return InceptionBaselineEmbedder(
  52         images,
  53         pretrained_ckpt,
  54         config.random_projection,
  55         config.random_projection_dim)
  56 
  57   strategy_to_embedder = {
  58       'inception_conv_ss_fc': InceptionConvSSFCEmbedder,
  59       'resnet': ResnetEmbedder,
  60   }
  61   if embedder_strategy not in strategy_to_embedder:
  62     raise ValueError('unknown embedder_strategy', embedder_strategy)
  63 
  64   embedding_size = config.embedding_size
  65   l2_reg_weight = config.learning.l2_reg_weight
  66   embedder = strategy_to_embedder[embedder_strategy](
  67       config[embedder_strategy], images, embedding_size,
  68       is_training, embedding_l2=l2_normalize_embedding,
  69       l2_reg_weight=l2_reg_weight, reuse=reuse)
  70   return embedder

  73 def build_inceptionv3_graph(images, endpoint, is_training, checkpoint,
  74                             reuse=False):
  75   """Builds an InceptionV3 model graph.
  76 
  77   Args:
  78     images: A 4-D float32 `Tensor` of batch images.
  79     endpoint: String, name of the InceptionV3 endpoint.
  80     is_training: Boolean, whether or not to build a training or inference graph.
  81     checkpoint: String, path to the pretrained model checkpoint.
  82     reuse: Boolean, whether or not we are reusing the embedder.
  83   Returns:
  84     inception_output: `Tensor` holding the InceptionV3 output.
  85     inception_variables: List of inception variables.
  86     init_fn: Function to initialize the weights (if not reusing, then None).
  87   """
  88   with slim.arg_scope(inception.inception_v3_arg_scope()):
  89     _, endpoints = inception.inception_v3(
  90         images, num_classes=1001, is_training=is_training)
  91     inception_output = endpoints[endpoint]
  92     inception_variables = slim.get_variables_to_restore()
  93     inception_variables = [
  94         i for i in inception_variables if 'global_step' not in i.name]
  95     if is_training and not reuse:
  96       init_saver = tf.train.Saver(inception_variables)
  97       def init_fn(scaffold, sess):
  98         del scaffold
  99         init_saver.restore(sess, checkpoint)
 100     else:
 101       init_fn = None
 102     return inception_output, inception_variables, init_fn

 105 class InceptionBaselineEmbedder(object):
 106   """Produces pre-trained InceptionV3 embeddings."""
 107 
 108   def __init__(self, images, pretrained_ckpt, reuse=False,
 109                random_projection=False, random_projection_dim=32):
 110     # Build InceptionV3 graph.
 111     (inception_output,
 112      self.inception_variables,
 113      self.init_fn) = build_inceptionv3_graph(
 114          images, 'Mixed_7c', False, pretrained_ckpt, reuse)
 115 
 116     # Pool 8x8x2048 -> 1x1x2048.
 117     embedding = slim.avg_pool2d(inception_output, [8, 8], stride=1)
 118     embedding = tf.squeeze(embedding, [1, 2])
 119 
 120     if random_projection:
 121       embedding = tf.matmul(
 122           embedding, tf.random_normal(
 123               shape=[2048, random_projection_dim], seed=123))
 124     self.embedding = embedding

 127 class PretrainedEmbedder(object):
 128   """Base class for embedders that take pre-trained networks as input."""
 129   __metaclass__ = ABCMeta
 130 
 131   def __init__(self, config, images, embedding_size, is_training,
 132                embedding_l2=True, l2_reg_weight=1e-6, reuse=False):
 133     """Constructor.
 134 
 135     Args:
 136       config: A T object holding training config.
 137       images: A 4-D float32 `Tensor` holding images to embed.
 138       embedding_size: Int, the size of the embedding.
 139       is_training: Boolean, whether or not this is a training or inference-time
 140         graph.
 141       embedding_l2: Boolean, whether or not to l2 normalize the embedding.
 142       l2_reg_weight: Float, weight applied to l2 weight regularization.
 143       reuse: Boolean, whether or not we're reusing this graph.
 144     """
 145     # Pull out all the embedder hyperparameters.
 146     self._config = config
 147     self._embedding_size = embedding_size
 148     self._l2_reg_weight = l2_reg_weight
 149     self._embedding_l2 = embedding_l2
 150     self._is_training = is_training
 151     self._reuse = reuse
 152 
 153     # Pull out pretrained hparams.
 154     pretrained_checkpoint = config.pretrained_checkpoint
 155     pretrained_layer = config.pretrained_layer
 156     pretrained_keep_prob = config.dropout.keep_pretrained
 157 
 158     # Build pretrained graph.
 159     (pretrained_output,
 160      self._pretrained_variables,
 161      self.init_fn) = self.build_pretrained_graph(
 162          images, pretrained_layer, pretrained_checkpoint, is_training, reuse)
 163 
 164     # Optionally drop out the activations.
 165     pretrained_output = slim.dropout(
 166         pretrained_output, keep_prob=pretrained_keep_prob,
 167         is_training=is_training)
 168     self._pretrained_output = pretrained_output
 169 
 170   @abstractmethod
 171   def build_pretrained_graph(self, images, layer, pretrained_checkpoint,
 172                              is_training, reuse):
 173     """Builds the graph for the pre-trained network.
 174 
 175     Method to be overridden by implementations.
 176 
 177     Args:
 178       images: A 4-D tf.float32 `Tensor` holding images to embed.
 179       layer: String, defining which pretrained layer to take as input
 180         to adaptation layers.
 181       pretrained_checkpoint: String, path to a checkpoint used to load
 182         pretrained weights.
 183       is_training: Boolean, whether or not we're in training mode.
 184       reuse: Boolean, whether or not to reuse embedder weights.
 185 
 186     Returns:
 187       pretrained_output: A 2 or 3-d tf.float32 `Tensor` holding pretrained
 188         activations.
 189     """
 190     pass
 191 
 192   @abstractmethod
 193   def construct_embedding(self):
 194     """Builds an embedding function on top of images.
 195 
 196     Method to be overridden by implementations.
 197 
 198     Returns:
 199       embeddings: A 2-d float32 `Tensor` of shape [batch_size, embedding_size]
 200         holding the embedded images.
 201     """
 202     pass
 203 
 204   def get_trainable_variables(self):
 205     """Gets a list of variables to optimize."""
 206     if self._config.finetune:
 207       return tf.trainable_variables()
 208     else:
 209       adaptation_only_vars = tf.get_collection(
 210           tf.GraphKeys.TRAINABLE_VARIABLES, scope=self._adaptation_scope)
 211       return adaptation_only_vars

 214 class ResnetEmbedder(PretrainedEmbedder):
 215   """Resnet TCN.
 216 
 217   ResnetV2 -> resnet adaptation layers -> optional l2 normalize -> embedding.
 218   """
 219 
 220   def __init__(self, config, images, embedding_size, is_training,
 221                embedding_l2=True, l2_reg_weight=1e-6, reuse=False):
 222     super(ResnetEmbedder, self).__init__(
 223         config, images, embedding_size, is_training, embedding_l2,
 224         l2_reg_weight, reuse)
 225 
 226   def build_pretrained_graph(
 227       self, images, resnet_layer, checkpoint, is_training, reuse=False):
 228     """See baseclass."""
 229     with slim.arg_scope(resnet_v2.resnet_arg_scope()):
 230       _, endpoints = resnet_v2.resnet_v2_50(
 231           images, is_training=is_training, reuse=reuse)
 232       resnet_layer = 'resnet_v2_50/block%d' % resnet_layer
 233       resnet_output = endpoints[resnet_layer]
 234       resnet_variables = slim.get_variables_to_restore()
 235       resnet_variables = [
 236           i for i in resnet_variables if 'global_step' not in i.name]
 237       if is_training and not reuse:
 238         init_saver = tf.train.Saver(resnet_variables)
 239         def init_fn(scaffold, sess):
 240           del scaffold
 241           init_saver.restore(sess, checkpoint)
 242       else:
 243         init_fn = None
 244 
 245       return resnet_output, resnet_variables, init_fn
 246 
 247   def construct_embedding(self):
 248     """Builds an embedding function on top of images.
 249 
 250     Method to be overridden by implementations.
 251 
 252     Returns:
 253       embeddings: A 2-d float32 `Tensor` of shape [batch_size, embedding_size]
 254         holding the embedded images.
 255     """
 256     with tf.variable_scope('tcn_net', reuse=self._reuse) as vs:
 257       self._adaptation_scope = vs.name
 258       net = self._pretrained_output
 259 
 260       # Define some adaptation blocks on top of the pre-trained resnet output.
 261       adaptation_blocks = []
 262       adaptation_block_params = [map(
 263           int, i.split('_')) for i in self._config.adaptation_blocks.split('-')]
 264       for i, (depth, num_units) in enumerate(adaptation_block_params):
 265         block = resnet_v2.resnet_v2_block(
 266             'adaptation_block_%d' % i, base_depth=depth, num_units=num_units,
 267             stride=1)
 268         adaptation_blocks.append(block)
 269 
 270       # Stack them on top of the resent output.
 271       net = resnet_utils.stack_blocks_dense(
 272           net, adaptation_blocks, output_stride=None)
 273 
 274       # Average pool the output.
 275       net = tf.reduce_mean(net, [1, 2], name='adaptation_pool', keep_dims=True)
 276 
 277       if self._config.emb_connection == 'fc':
 278         # Use fully connected layer to project to embedding layer.
 279         fc_hidden_sizes = self._config.fc_hidden_sizes
 280         if fc_hidden_sizes == 'None':
 281           fc_hidden_sizes = []
 282         else:
 283           fc_hidden_sizes = map(int, fc_hidden_sizes.split('_'))
 284         fc_hidden_keep_prob = self._config.dropout.keep_fc
 285         net = tf.squeeze(net)
 286         for fc_hidden_size in fc_hidden_sizes:
 287           net = slim.layers.fully_connected(net, fc_hidden_size)
 288           if fc_hidden_keep_prob < 1.0:
 289             net = slim.dropout(net, keep_prob=fc_hidden_keep_prob,
 290                                is_training=self._is_training)
 291 
 292         # Connect last FC layer to embedding.
 293         embedding = slim.layers.fully_connected(net, self._embedding_size,
 294                                                 activation_fn=None)
 295       else:
 296         # Use 1x1 conv layer to project to embedding layer.
 297         embedding = slim.conv2d(
 298             net, self._embedding_size, [1, 1], activation_fn=None,
 299             normalizer_fn=None, scope='embedding')
 300         embedding = tf.squeeze(embedding)
 301 
 302       # Optionally L2 normalize the embedding.
 303       if self._embedding_l2:
 304         embedding = tf.nn.l2_normalize(embedding, dim=1)
 305 
 306       return embedding
 307 
 308   def get_trainable_variables(self):
 309     """Gets a list of variables to optimize."""
 310     if self._config.finetune:
 311       return tf.trainable_variables()
 312     else:
 313       adaptation_only_vars = tf.get_collection(
 314           tf.GraphKeys.TRAINABLE_VARIABLES, scope=self._adaptation_scope)
 315       return adaptation_only_vars

 318 class InceptionEmbedderBase(PretrainedEmbedder):
 319   """Base class for embedders that take pre-trained InceptionV3 activations."""
 320 
 321   def __init__(self, config, images, embedding_size, is_training,
 322                embedding_l2=True, l2_reg_weight=1e-6, reuse=False):
 323     super(InceptionEmbedderBase, self).__init__(
 324         config, images, embedding_size, is_training, embedding_l2,
 325         l2_reg_weight, reuse)
 326 
 327   def build_pretrained_graph(
 328       self, images, inception_layer, checkpoint, is_training, reuse=False):
 329     """See baseclass."""
 330     # Build InceptionV3 graph.
 331     inception_output, inception_variables, init_fn = build_inceptionv3_graph(
 332         images, inception_layer, is_training, checkpoint, reuse)
 333     return inception_output, inception_variables, init_fn

 336 class InceptionConvSSFCEmbedder(InceptionEmbedderBase):
 337   """TCN Embedder V1.
 338 
 339   InceptionV3 (mixed_5d) -> conv layers -> spatial softmax ->
 340     fully connected -> optional l2 normalize -> embedding.
 341   """
 342 
 343   def __init__(self, config, images, embedding_size, is_training,
 344                embedding_l2=True, l2_reg_weight=1e-6, reuse=False):
 345     super(InceptionConvSSFCEmbedder, self).__init__(
 346         config, images, embedding_size, is_training, embedding_l2,
 347         l2_reg_weight, reuse)
 348 
 349     # Pull out all the hyperparameters specific to this embedder.
 350     self._additional_conv_sizes = config.additional_conv_sizes
 351     self._conv_hidden_keep_prob = config.dropout.keep_conv
 352     self._fc_hidden_sizes = config.fc_hidden_sizes
 353     self._fc_hidden_keep_prob = config.dropout.keep_fc
 354 
 355   def construct_embedding(self):
 356     """Builds a conv -> spatial softmax -> FC adaptation network."""
 357     is_training = self._is_training
 358     normalizer_params = {'is_training': is_training}
 359     with tf.variable_scope('tcn_net', reuse=self._reuse) as vs:
 360       self._adaptation_scope = vs.name
 361       with slim.arg_scope(
 362           [slim.layers.conv2d],
 363           activation_fn=tf.nn.relu,
 364           normalizer_fn=slim.batch_norm, normalizer_params=normalizer_params,
 365           weights_regularizer=slim.regularizers.l2_regularizer(
 366               self._l2_reg_weight),
 367           biases_regularizer=slim.regularizers.l2_regularizer(
 368               self._l2_reg_weight)):
 369         with slim.arg_scope(
 370             [slim.layers.fully_connected],
 371             activation_fn=tf.nn.relu,
 372             normalizer_fn=slim.batch_norm, normalizer_params=normalizer_params,
 373             weights_regularizer=slim.regularizers.l2_regularizer(
 374                 self._l2_reg_weight),
 375             biases_regularizer=slim.regularizers.l2_regularizer(
 376                 self._l2_reg_weight)):
 377 
 378           # Input to embedder is pre-trained inception output.
 379           net = self._pretrained_output
 380 
 381           # Optionally add more conv layers.
 382           for num_filters in self._additional_conv_sizes:
 383             net = slim.layers.conv2d(
 384                 net, num_filters, kernel_size=[3, 3], stride=[1, 1])
 385             net = slim.dropout(net, keep_prob=self._conv_hidden_keep_prob,
 386                                is_training=is_training)
 387 
 388           # Take the spatial soft arg-max of the last convolutional layer.
 389           # This is a form of spatial attention over the activations.
 390           # See more here: http://arxiv.org/abs/1509.06113.
 391           net = tf.contrib.layers.spatial_softmax(net)
 392           self.spatial_features = net
 393 
 394           # Add fully connected layers.
 395           net = slim.layers.flatten(net)
 396           for fc_hidden_size in self._fc_hidden_sizes:
 397             net = slim.layers.fully_connected(net, fc_hidden_size)
 398             if self._fc_hidden_keep_prob < 1.0:
 399               net = slim.dropout(net, keep_prob=self._fc_hidden_keep_prob,
 400                                  is_training=is_training)
 401 
 402           # Connect last FC layer to embedding.
 403           net = slim.layers.fully_connected(net, self._embedding_size,
 404                                             activation_fn=None)
 405 
 406           # Optionally L2 normalize the embedding.
 407           if self._embedding_l2:
 408             net = tf.nn.l2_normalize(net, dim=1)
 409 
 410           return net
