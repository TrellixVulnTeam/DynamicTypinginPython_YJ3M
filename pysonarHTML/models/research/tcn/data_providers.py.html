  33 def full_sequence_provider(file_list, num_views):
  34   """Provides full preprocessed image sequences.
  35 
  36   Args:
  37     file_list: List of strings, paths to TFRecords to preprocess.
  38     num_views: Int, the number of simultaneous viewpoints at each timestep in
  39       the dataset.
  40   Returns:
  41     preprocessed: A 4-D float32 `Tensor` holding a sequence of preprocessed
  42       images.
  43     raw_image_strings: A 2-D string `Tensor` holding a sequence of raw
  44       jpeg-encoded image strings.
  45     task: String, the name of the sequence.
  46     seq_len: Int, the number of timesteps in the sequence.
  47   """
  48   def _parse_sequence(x):
  49     context, views, seq_len = parse_sequence_example(x, num_views)
  50     task = context['task']
  51     return views, task, seq_len
  52 
  53   data_files = tf.contrib.slim.parallel_reader.get_data_files(file_list)
  54   dataset = tf.data.Dataset.from_tensor_slices(data_files)
  55   dataset = dataset.repeat(1)
  56   # Get a dataset of sequences.
  57   dataset = dataset.flat_map(record_dataset)
  58 
  59   # Build a dataset of TFRecord files.
  60   dataset = dataset.repeat(1)
  61   # Prefetch a number of opened files.
  62   dataset = dataset.prefetch(12)
  63   # Use _parse_sequence to deserialize (but not decode) image strings.
  64   dataset = dataset.map(_parse_sequence, num_parallel_calls=12)
  65   # Prefetch batches of images.
  66   dataset = dataset.prefetch(12)
  67   dataset = dataset.make_one_shot_iterator()
  68   views, task, seq_len = dataset.get_next()
  69   return views, task, seq_len

  72 def parse_labeled_example(
  73     example_proto, view_index, preprocess_fn, image_attr_keys, label_attr_keys):
  74   """Parses a labeled test example from a specified view.
  75 
  76   Args:
  77     example_proto: A scalar string Tensor.
  78     view_index: Int, index on which view to parse.
  79     preprocess_fn: A function with the signature (raw_images, is_training) ->
  80       preprocessed_images, where raw_images is a 4-D float32 image `Tensor`
  81       of raw images, is_training is a Boolean describing if we're in training,
  82       and preprocessed_images is a 4-D float32 image `Tensor` holding
  83       preprocessed images.
  84     image_attr_keys: List of Strings, names for image keys.
  85     label_attr_keys: List of Strings, names for label attributes.
  86   Returns:
  87     data: A tuple of images, attributes and tasks `Tensors`.
  88   """
  89   features = {}
  90   for attr_key in image_attr_keys:
  91     features[attr_key] = tf.FixedLenFeature((), tf.string)
  92   for attr_key in label_attr_keys:
  93     features[attr_key] = tf.FixedLenFeature((), tf.int64)
  94   parsed_features = tf.parse_single_example(example_proto, features)
  95   image_only_keys = [i for i in image_attr_keys if 'image' in i]
  96   view_image_key = image_only_keys[view_index]
  97   image = preprocessing.decode_image(parsed_features[view_image_key])
  98   preprocessed = preprocess_fn(image, is_training=False)
  99   attributes = [parsed_features[k] for k in label_attr_keys]
 100   task = parsed_features['task']
 101   return tuple([preprocessed] + attributes + [task])

 104 def labeled_data_provider(
 105     filenames, preprocess_fn, view_index, image_attr_keys, label_attr_keys,
 106     batch_size=32, num_epochs=1):
 107   """Gets a batched dataset iterator over annotated test images + labels.
 108 
 109   Provides a single view, specifed in `view_index`.
 110 
 111   Args:
 112     filenames: List of Strings, paths to tfrecords on disk.
 113     preprocess_fn: A function with the signature (raw_images, is_training) ->
 114       preprocessed_images, where raw_images is a 4-D float32 image `Tensor`
 115       of raw images, is_training is a Boolean describing if we're in training,
 116       and preprocessed_images is a 4-D float32 image `Tensor` holding
 117       preprocessed images.
 118     view_index: Int, the index of the view to embed.
 119     image_attr_keys: List of Strings, names for image keys.
 120     label_attr_keys: List of Strings, names for label attributes.
 121     batch_size: Int, size of the batch.
 122     num_epochs: Int, number of epochs over the classification dataset.
 123   Returns:
 124     batch_images: 4-d float `Tensor` holding the batch images for the view.
 125     labels: K-d int `Tensor` holding the K label attributes.
 126     tasks: 1-D String `Tensor`, holding the task names for each batch element.
 127   """
 128   dataset = tf.data.TFRecordDataset(filenames)
 129   # pylint: disable=g-long-lambda
 130   dataset = dataset.map(
 131       lambda p: parse_labeled_example(
 132           p, view_index, preprocess_fn, image_attr_keys, label_attr_keys))
 133   dataset = dataset.repeat(num_epochs)
 134   dataset = dataset.batch(batch_size)
 135   data_iterator = dataset.make_one_shot_iterator()
 136   batch_data = data_iterator.get_next()
 137   batch_images = batch_data[0]
 138 
 139   batch_labels = tf.stack(batch_data[1:-1], 1)
 140 
 141   batch_tasks = batch_data[-1]
 142 
 143   batch_images = set_image_tensor_batch_dim(batch_images, batch_size)
 144   batch_labels.set_shape([batch_size, len(label_attr_keys)])
 145   batch_tasks.set_shape([batch_size])
 146 
 147   return batch_images, batch_labels, batch_tasks

 150 def parse_sequence_example(serialized_example, num_views):
 151   """Parses a serialized sequence example into views, sequence length data."""
 152   context_features = {
 153       'task': tf.FixedLenFeature(shape=[], dtype=tf.string),
 154       'len': tf.FixedLenFeature(shape=[], dtype=tf.int64)
 155   }
 156   view_names = ['view%d' % i for i in range(num_views)]
 157   fixed_features = [
 158       tf.FixedLenSequenceFeature(
 159           shape=[], dtype=tf.string) for _ in range(len(view_names))]
 160   sequence_features = dict(zip(view_names, fixed_features))
 161   context_parse, sequence_parse = tf.parse_single_sequence_example(
 162       serialized=serialized_example,
 163       context_features=context_features,
 164       sequence_features=sequence_features)
 165   views = tf.stack([sequence_parse[v] for v in view_names])
 166   lens = [sequence_parse[v].get_shape().as_list()[0] for v in view_names]
 167   assert len(set(lens)) == 1
 168   seq_len = tf.shape(sequence_parse[view_names[-1]])[0]
 169   return context_parse, views, seq_len

 172 def get_shuffled_input_records(file_list):
 173   """Build a tf.data.Dataset of shuffled input TFRecords that repeats."""
 174   dataset = tf.data.Dataset.from_tensor_slices(file_list)
 175   dataset = dataset.shuffle(len(file_list))
 176   dataset = dataset.repeat()
 177   dataset = dataset.flat_map(record_dataset)
 178   dataset = dataset.repeat()
 179   return dataset

 182 def get_tcn_anchor_pos_indices(seq_len, num_views, num_pairs, window):
 183   """Gets batch TCN anchor positive timestep and view indices.
 184 
 185   This gets random (anchor, positive) timesteps from a sequence, and chooses
 186   2 random differing viewpoints for each anchor positive pair.
 187 
 188   Args:
 189     seq_len: Int, the size of the batch sequence in timesteps.
 190     num_views: Int, the number of simultaneous viewpoints at each timestep.
 191     num_pairs: Int, the number of pairs to build.
 192     window: Int, the window (in frames) from which to take anchor, positive
 193       and negative indices.
 194   Returns:
 195     ap_time_indices: 1-D Int `Tensor` with size [num_pairs], holding the
 196       timestep for each (anchor,pos) pair.
 197     a_view_indices: 1-D Int `Tensor` with size [num_pairs], holding the
 198       view index for each anchor.
 199     p_view_indices: 1-D Int `Tensor` with size [num_pairs], holding the
 200       view index for each positive.
 201   """
 202   # Get anchor, positive time indices.
 203   def f1():
 204     # Choose a random window-length range from the sequence.
 205     range_min = tf.random_shuffle(tf.range(seq_len-window))[0]
 206     range_max = range_min+window
 207     return tf.range(range_min, range_max)
 208   def f2():
 209     # Consider the full sequence.
 210     return tf.range(seq_len)
 211   time_indices = tf.cond(tf.greater(seq_len, window), f1, f2)
 212   shuffled_indices = tf.random_shuffle(time_indices)
 213   num_pairs = tf.minimum(seq_len, num_pairs)
 214   ap_time_indices = shuffled_indices[:num_pairs]
 215 
 216   # Get opposing anchor, positive view indices.
 217   view_indices = tf.tile(
 218       tf.expand_dims(tf.range(num_views), 0), (num_pairs, 1))
 219   shuffled_view_indices = tf.map_fn(tf.random_shuffle, view_indices)
 220   a_view_indices = shuffled_view_indices[:, 0]
 221   p_view_indices = shuffled_view_indices[:, 1]
 222   return ap_time_indices, a_view_indices, p_view_indices

 225 def set_image_tensor_batch_dim(tensor, batch_dim):
 226   """Sets the batch dimension on an image tensor."""
 227   shape = tensor.get_shape()
 228   tensor.set_shape([batch_dim, shape[1], shape[2], shape[3]])
 229   return tensor

 232 def parse_sequence_to_pairs_batch(
 233     serialized_example, preprocess_fn, is_training, num_views, batch_size,
 234     window):
 235   """Parses a serialized sequence example into a batch of preprocessed data.
 236 
 237   Args:
 238     serialized_example: A serialized SequenceExample.
 239     preprocess_fn: A function with the signature (raw_images, is_training) ->
 240       preprocessed_images.
 241     is_training: Boolean, whether or not we're in training.
 242     num_views: Int, the number of simultaneous viewpoints at each timestep in
 243       the dataset.
 244     batch_size: Int, size of the batch to get.
 245     window: Int, only take pairs from a maximium window of this size.
 246   Returns:
 247     preprocessed: A 4-D float32 `Tensor` holding preprocessed images.
 248     anchor_images: A 4-D float32 `Tensor` holding raw anchor images.
 249     pos_images: A 4-D float32 `Tensor` holding raw positive images.
 250   """
 251   _, views, seq_len = parse_sequence_example(serialized_example, num_views)
 252 
 253   # Get random (anchor, positive) timestep and viewpoint indices.
 254   num_pairs = batch_size // 2
 255   ap_time_indices, a_view_indices, p_view_indices = get_tcn_anchor_pos_indices(
 256       seq_len, num_views, num_pairs, window)
 257 
 258   # Gather the image strings.
 259   combined_anchor_indices = tf.concat(
 260       [tf.expand_dims(a_view_indices, 1),
 261        tf.expand_dims(ap_time_indices, 1)], 1)
 262   combined_pos_indices = tf.concat(
 263       [tf.expand_dims(p_view_indices, 1),
 264        tf.expand_dims(ap_time_indices, 1)], 1)
 265   anchor_images = tf.gather_nd(views, combined_anchor_indices)
 266   pos_images = tf.gather_nd(views, combined_pos_indices)
 267 
 268   # Decode images.
 269   anchor_images = tf.map_fn(
 270       preprocessing.decode_image, anchor_images, dtype=tf.float32)
 271   pos_images = tf.map_fn(
 272       preprocessing.decode_image, pos_images, dtype=tf.float32)
 273 
 274   # Concatenate [anchor, postitive] images into a batch and preprocess it.
 275   concatenated = tf.concat([anchor_images, pos_images], 0)
 276   preprocessed = preprocess_fn(concatenated, is_training)
 277   anchor_prepro, positive_prepro = tf.split(preprocessed, num_or_size_splits=2,
 278                                             axis=0)
 279 
 280   # Set static batch dimensions for all image tensors
 281   ims = [anchor_prepro, positive_prepro, anchor_images, pos_images]
 282   ims = [set_image_tensor_batch_dim(i, num_pairs) for i in ims]
 283   [anchor_prepro, positive_prepro, anchor_images, pos_images] = ims
 284 
 285   # Assign each anchor and positive the same label.
 286   anchor_labels = tf.range(1, num_pairs+1)
 287   positive_labels = tf.range(1, num_pairs+1)
 288 
 289   return (anchor_prepro, positive_prepro, anchor_images, pos_images,
 290           anchor_labels, positive_labels, seq_len)

 293 def multiview_pairs_provider(file_list,
 294                              preprocess_fn,
 295                              num_views,
 296                              window,
 297                              is_training,
 298                              batch_size,
 299                              examples_per_seq=2,
 300                              num_parallel_calls=12,
 301                              sequence_prefetch_size=12,
 302                              batch_prefetch_size=12):
 303   """Provides multi-view TCN anchor-positive image pairs.
 304 
 305   Returns batches of Multi-view TCN pairs, where each pair consists of an
 306   anchor and a positive coming from different views from the same timestep.
 307   Batches are filled one entire sequence at a time until
 308   batch_size is exhausted. Pairs are chosen randomly without replacement
 309   within a sequence.
 310 
 311   Used by:
 312     * triplet semihard loss.
 313     * clustering loss.
 314     * npairs loss.
 315     * lifted struct loss.
 316     * contrastive loss.
 317 
 318   Args:
 319     file_list: List of Strings, paths to tfrecords.
 320     preprocess_fn: A function with the signature (raw_images, is_training) ->
 321       preprocessed_images, where raw_images is a 4-D float32 image `Tensor`
 322       of raw images, is_training is a Boolean describing if we're in training,
 323       and preprocessed_images is a 4-D float32 image `Tensor` holding
 324       preprocessed images.
 325     num_views: Int, the number of simultaneous viewpoints at each timestep.
 326     window: Int, size of the window (in frames) from which to draw batch ids.
 327     is_training: Boolean, whether or not we're in training.
 328     batch_size: Int, how many examples in the batch (num pairs * 2).
 329     examples_per_seq: Int, how many examples to take per sequence.
 330     num_parallel_calls: Int, the number of elements to process in parallel by
 331       mapper.
 332     sequence_prefetch_size: Int, size of the buffer used to prefetch sequences.
 333     batch_prefetch_size: Int, size of the buffer used to prefetch batches.
 334   Returns:
 335     batch_images: A 4-D float32 `Tensor` holding preprocessed batch images.
 336     anchor_labels: A 1-D int32 `Tensor` holding anchor image labels.
 337     anchor_images: A 4-D float32 `Tensor` holding raw anchor images.
 338     positive_labels: A 1-D int32 `Tensor` holding positive image labels.
 339     pos_images: A 4-D float32 `Tensor` holding raw positive images.
 340   """
 341   def _parse_sequence(x):
 342     return parse_sequence_to_pairs_batch(
 343         x, preprocess_fn, is_training, num_views, examples_per_seq, window)
 344 
 345   # Build a buffer of shuffled input TFRecords that repeats forever.
 346   dataset = get_shuffled_input_records(file_list)
 347 
 348   # Prefetch a number of opened TFRecords.
 349   dataset = dataset.prefetch(sequence_prefetch_size)
 350 
 351   # Use _parse_sequence to map sequences to batches (one sequence per batch).
 352   dataset = dataset.map(
 353       _parse_sequence, num_parallel_calls=num_parallel_calls)
 354 
 355   # Filter out sequences that don't have at least examples_per_seq.
 356   def seq_greater_than_min(seqlen, maximum):
 357     return seqlen >= maximum
 358   filter_fn = functools.partial(seq_greater_than_min, maximum=examples_per_seq)
 359   dataset = dataset.filter(lambda a, b, c, d, e, f, seqlen: filter_fn(seqlen))
 360 
 361   # Take a number of sequences for the batch.
 362   assert batch_size % examples_per_seq == 0
 363   sequences_per_batch = batch_size // examples_per_seq
 364   dataset = dataset.batch(sequences_per_batch)
 365 
 366   # Prefetch batches of images.
 367   dataset = dataset.prefetch(batch_prefetch_size)
 368 
 369   iterator = dataset.make_one_shot_iterator()
 370   data = iterator.get_next()
 371 
 372   # Pull out images, reshape to [batch_size, ...], concatenate anchor and pos.
 373   ims = list(data[:4])
 374   anchor_labels, positive_labels = data[4:6]
 375 
 376   # Set labels shape.
 377   anchor_labels.set_shape([sequences_per_batch, None])
 378   positive_labels.set_shape([sequences_per_batch, None])
 379 
 380   def _reshape_to_batchsize(im):
 381     """[num_sequences, num_per_seq, ...] images to [batch_size, ...]."""
 382     sequence_ims = tf.split(im, num_or_size_splits=sequences_per_batch, axis=0)
 383     sequence_ims = [tf.squeeze(i) for i in sequence_ims]
 384     return tf.concat(sequence_ims, axis=0)
 385 
 386   # Reshape labels.
 387   anchor_labels = _reshape_to_batchsize(anchor_labels)
 388   positive_labels = _reshape_to_batchsize(positive_labels)
 389 
 390   def _set_shape(im):
 391     """Sets a static shape for an image tensor of [sequences_per_batch,...] ."""
 392     shape = im.get_shape()
 393     im.set_shape([sequences_per_batch, shape[1], shape[2], shape[3], shape[4]])
 394     return im
 395   ims = [_set_shape(im) for im in ims]
 396   ims = [_reshape_to_batchsize(im) for im in ims]
 397 
 398   anchor_prepro, positive_prepro, anchor_images, pos_images = ims
 399   batch_images = tf.concat([anchor_prepro, positive_prepro], axis=0)
 400 
 401   return batch_images, anchor_labels, positive_labels, anchor_images, pos_images

 404 def get_svtcn_indices(seq_len, batch_size, num_views):
 405   """Gets a random window of contiguous time indices from a sequence.
 406 
 407   Args:
 408     seq_len: Int, number of timesteps in the image sequence.
 409     batch_size: Int, size of the batch to construct.
 410     num_views: Int, the number of simultaneous viewpoints at each
 411       timestep in the dataset.
 412 
 413   Returns:
 414     time_indices: 1-D Int `Tensor` with size [batch_size], holding the
 415       timestep for each batch image.
 416     view_indices: 1-D Int `Tensor` with size [batch_size], holding the
 417       view for each batch image. This is consistent across the batch.
 418   """
 419   # Get anchor, positive time indices.
 420   def f1():
 421     # Choose a random contiguous range from within the sequence.
 422     range_min = tf.random_shuffle(tf.range(seq_len-batch_size))[0]
 423     range_max = range_min+batch_size
 424     return tf.range(range_min, range_max)
 425   def f2():
 426     # Consider the full sequence.
 427     return tf.range(seq_len)
 428   time_indices = tf.cond(tf.greater(seq_len, batch_size), f1, f2)
 429   # Get opposing anchor, positive view indices.
 430   random_view = tf.random_shuffle(tf.range(num_views))[0]
 431   view_indices = tf.tile([random_view], (batch_size,))
 432   return time_indices, view_indices

 435 def parse_sequence_to_svtcn_batch(
 436     serialized_example, preprocess_fn, is_training, num_views, batch_size):
 437   """Parses a serialized sequence example into a batch of SVTCN data."""
 438   _, views, seq_len = parse_sequence_example(serialized_example, num_views)
 439   # Get svtcn indices.
 440   time_indices, view_indices = get_svtcn_indices(seq_len, batch_size, num_views)
 441   combined_indices = tf.concat(
 442       [tf.expand_dims(view_indices, 1),
 443        tf.expand_dims(time_indices, 1)], 1)
 444 
 445   # Gather the image strings.
 446   images = tf.gather_nd(views, combined_indices)
 447 
 448   # Decode images.
 449   images = tf.map_fn(preprocessing.decode_image, images, dtype=tf.float32)
 450 
 451   # Concatenate anchor and postitive images, preprocess the batch.
 452   preprocessed = preprocess_fn(images, is_training)
 453 
 454   return preprocessed, images, time_indices

 457 def singleview_tcn_provider(file_list,
 458                             preprocess_fn,
 459                             num_views,
 460                             is_training,
 461                             batch_size,
 462                             num_parallel_calls=12,
 463                             sequence_prefetch_size=12,
 464                             batch_prefetch_size=12):
 465   """Provides data to train singleview TCNs.
 466 
 467   Args:
 468     file_list: List of Strings, paths to tfrecords.
 469     preprocess_fn: A function with the signature (raw_images, is_training) ->
 470       preprocessed_images, where raw_images is a 4-D float32 image `Tensor`
 471       of raw images, is_training is a Boolean describing if we're in training,
 472       and preprocessed_images is a 4-D float32 image `Tensor` holding
 473       preprocessed images.
 474     num_views: Int, the number of simultaneous viewpoints at each timestep.
 475     is_training: Boolean, whether or not we're in training.
 476     batch_size: Int, how many examples in the batch.
 477     num_parallel_calls: Int, the number of elements to process in parallel by
 478       mapper.
 479     sequence_prefetch_size: Int, size of the buffer used to prefetch sequences.
 480     batch_prefetch_size: Int, size of the buffer used to prefetch batches.
 481 
 482   Returns:
 483     batch_images: A 4-D float32 `Tensor` of preprocessed images.
 484     raw_images: A 4-D float32 `Tensor` of raw images.
 485     timesteps: A 1-D int32 `Tensor` of timesteps associated with each image.
 486   """
 487   def _parse_sequence(x):
 488     return parse_sequence_to_svtcn_batch(
 489         x, preprocess_fn, is_training, num_views, batch_size)
 490 
 491   # Build a buffer of shuffled input TFRecords that repeats forever.
 492   dataset = get_shuffled_input_records(file_list)
 493 
 494   # Prefetch a number of opened files.
 495   dataset = dataset.prefetch(sequence_prefetch_size)
 496 
 497   # Use _parse_sequence to map sequences to image batches.
 498   dataset = dataset.map(
 499       _parse_sequence, num_parallel_calls=num_parallel_calls)
 500 
 501   # Prefetch batches of images.
 502   dataset = dataset.prefetch(batch_prefetch_size)
 503   dataset = dataset.make_one_shot_iterator()
 504   batch_images, raw_images, timesteps = dataset.get_next()
 505   return batch_images, raw_images, timesteps
