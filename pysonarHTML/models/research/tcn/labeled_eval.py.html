  56 def nearest_cross_sequence_neighbors(data, tasks, n_neighbors=1):
  57   """Computes the n_neighbors nearest neighbors for every row in data.
  58 
  59   Args:
  60     data: A np.float32 array of shape [num_data, embedding size] holding
  61       an embedded validation / test dataset.
  62     tasks: A list of strings of size [num_data] holding the task or sequence
  63       name that each row belongs to.
  64     n_neighbors: The number of knn indices to return for each row.
  65   Returns:
  66     indices: an np.int32 array of size [num_data, n_neighbors] holding the
  67       n_neighbors nearest indices for every row in data. These are
  68       restricted to be from different named sequences (as defined in `tasks`).
  69   """
  70 
  71   # Compute the pairwise sequence adjacency matrix from `tasks`.
  72   num_data = data.shape[0]
  73   tasks = np.array(tasks)
  74   tasks = np.reshape(tasks, (num_data, 1))
  75   assert len(tasks.shape) == 2
  76   not_adjacent = (tasks != tasks.T)
  77 
  78   # Compute the symmetric pairwise distance matrix.
  79   pdist = pairwise_distances(data, metric='sqeuclidean')
  80 
  81   # For every row in the pairwise distance matrix, only consider
  82   # cross-sequence columns.
  83   indices = np.zeros((num_data, n_neighbors), dtype=np.int32)
  84   for idx in range(num_data):
  85     # Restrict to cross_sequence neighbors.
  86     distances = [(
  87         pdist[idx][i], i) for i in xrange(num_data) if not_adjacent[idx][i]]
  88     _, nearest_indices = zip(*sorted(
  89         distances, key=lambda x: x[0])[:n_neighbors])
  90     indices[idx] = nearest_indices
  91   return indices

  94 def compute_cross_sequence_recall_at_k(retrieved_labels, labels, k_list):
  95   """Compute recall@k for a given list of k values.
  96 
  97   Recall is one if an example of the same class is retrieved among the
  98     top k nearest neighbors given a query example and zero otherwise.
  99     Counting the recall for all examples and averaging the counts returns
 100     recall@k score.
 101 
 102   Args:
 103     retrieved_labels: 2-D Numpy array of KNN labels for every embedding.
 104     labels: 1-D Numpy array of shape [number of data].
 105     k_list: List of k values to evaluate recall@k.
 106 
 107   Returns:
 108     recall_list: List of recall@k values.
 109   """
 110   kvalue_to_recall = dict(zip(k_list, np.zeros(len(k_list))))
 111 
 112   # For each value of K.
 113   for k in k_list:
 114     matches = defaultdict(float)
 115     counts = defaultdict(float)
 116     # For each (row index, label value) in the query labels.
 117     for i, label_value in enumerate(labels):
 118       # Loop over the K nearest retrieved labels.
 119       if label_value in retrieved_labels[i][:k]:
 120         matches[label_value] += 1.
 121       # Increment the denominator.
 122       counts[label_value] += 1.
 123     kvalue_to_recall[k] = np.mean(
 124         [matches[l]/counts[l] for l in matches])
 125   return [kvalue_to_recall[i] for i in k_list]

 128 def compute_cross_sequence_recalls_at_k(
 129     embeddings, labels, label_attr_keys, tasks, k_list, summary_writer,
 130     training_step):
 131   """Computes and reports the recall@k for each classification problem.
 132 
 133   This takes an embedding matrix and an array of multiclass labels
 134   with size [num_data, number of classification problems], then
 135   computes the average recall@k for each classification problem
 136   as well as the average across problems.
 137 
 138   Args:
 139     embeddings: A np.float32 array of size [num_data, embedding_size]
 140       representing the embedded validation or test dataset.
 141     labels: A np.int32 array of size [num_data, num_classification_problems]
 142       holding multiclass labels for each embedding for each problem.
 143     label_attr_keys: List of strings, holds the names of the classification
 144       problems.
 145     tasks: A list of strings describing the video sequence each row
 146       belongs to. This is used to restrict the recall@k computation
 147       to cross-sequence examples.
 148     k_list: A list of ints, the k values to evaluate recall@k.
 149     summary_writer: A tf.summary.FileWriter.
 150     training_step: Int, the current training step we're evaluating.
 151   """
 152   num_data = float(embeddings.shape[0])
 153   assert labels.shape[0] == num_data
 154 
 155   # Compute knn indices.
 156   indices = nearest_cross_sequence_neighbors(
 157       embeddings, tasks, n_neighbors=max(k_list))
 158   retrieved_labels = labels[indices]
 159 
 160   # Compute the recall@k for each classification problem.
 161   recall_lists = []
 162   for idx, label_attr in enumerate(label_attr_keys):
 163     problem_labels = labels[:, idx]
 164     # Take all indices, all k labels for the problem indexed by idx.
 165     problem_retrieved = retrieved_labels[:, :, idx]
 166     recall_list = compute_cross_sequence_recall_at_k(
 167         retrieved_labels=problem_retrieved,
 168         labels=problem_labels,
 169         k_list=k_list)
 170     recall_lists.append(recall_list)
 171     for (k, recall) in zip(k_list, recall_list):
 172       recall_error = 1-recall
 173       summ = tf.Summary(value=[tf.Summary.Value(
 174           tag='validation/classification/%s error@top%d' % (
 175               label_attr, k),
 176           simple_value=recall_error)])
 177       print('%s recall@K=%d' % (label_attr, k), recall_error)
 178       summary_writer.add_summary(summ, int(training_step))
 179 
 180   # Report an average recall@k across problems.
 181   recall_lists = np.array(recall_lists)
 182   for i in range(recall_lists.shape[1]):
 183     average_recall = np.mean(recall_lists[:, i])
 184     recall_error = 1 - average_recall
 185     summ = tf.Summary(value=[tf.Summary.Value(
 186         tag='validation/classification/average error@top%d' % k_list[i],
 187         simple_value=recall_error)])
 188     print('Average recall@K=%d' % k_list[i], recall_error)
 189     summary_writer.add_summary(summ, int(training_step))

 192 def evaluate_once(
 193     estimator, input_fn_by_view, batch_size, checkpoint_path,
 194     label_attr_keys, embedding_size, num_views, k_list):
 195   """Compute the recall@k for a given checkpoint path.
 196 
 197   Args:
 198     estimator: an `Estimator` object to evaluate.
 199     input_fn_by_view: An input_fn to an `Estimator's` predict method. Takes
 200       a view index and returns a dict holding ops for getting raw images for
 201       the view.
 202     batch_size: Int, size of the labeled eval batch.
 203     checkpoint_path: String, path to the specific checkpoint being evaluated.
 204     label_attr_keys: A list of Strings, holding each attribute name.
 205     embedding_size: Int, the size of the embedding.
 206     num_views: Int, number of views in the dataset.
 207     k_list: List of ints, list of K values to compute recall at K for.
 208   """
 209   feat_matrix = np.zeros((0, embedding_size))
 210   label_vect = np.zeros((0, len(label_attr_keys)))
 211   tasks = []
 212   eval_tensor_keys = ['embeddings', 'tasks', 'classification_labels']
 213 
 214   # Iterate all views in the dataset.
 215   for view_index in range(num_views):
 216     # Set up a graph for embedding entire dataset.
 217     predictions = estimator.inference(
 218         input_fn_by_view(view_index), checkpoint_path,
 219         batch_size, predict_keys=eval_tensor_keys)
 220 
 221     # Enumerate predictions.
 222     for i, p in enumerate(predictions):
 223       if i % 100 == 0:
 224         tf.logging.info('Embedded %d images for view %d' % (i, view_index))
 225 
 226       label = p['classification_labels']
 227       task = p['tasks']
 228       embedding = p['embeddings']
 229 
 230       # Collect (embedding, label, task) data.
 231       feat_matrix = np.append(feat_matrix, [embedding], axis=0)
 232       label_vect = np.append(label_vect, [label], axis=0)
 233       tasks.append(task)
 234 
 235   # Compute recall statistics.
 236   ckpt_step = int(checkpoint_path.split('-')[-1])
 237   summary_dir = os.path.join(FLAGS.outdir, 'labeled_eval_summaries')
 238   summary_writer = tf.summary.FileWriter(summary_dir)
 239   compute_cross_sequence_recalls_at_k(
 240       feat_matrix, label_vect, label_attr_keys, tasks, k_list,
 241       summary_writer, ckpt_step)

 244 def get_labeled_tables(config):
 245   """Gets either labeled test or validation tables, based on flags."""
 246   # Get a list of filenames corresponding to labeled data.
 247   mode = FLAGS.mode
 248   if mode == 'validation':
 249     labeled_tables = util.GetFilesRecursively(config.data.labeled.validation)
 250   elif mode == 'test':
 251     labeled_tables = util.GetFilesRecursively(config.data.labeled.test)
 252   else:
 253     raise ValueError('Unknown dataset: %s' % mode)
 254   return labeled_tables

 257 def main(_):
 258   """Runs main labeled eval loop."""
 259   # Parse config dict from yaml config files / command line flags.
 260   config = util.ParseConfigsToLuaTable(FLAGS.config_paths, FLAGS.model_params)
 261 
 262   # Choose an estimator based on training strategy.
 263   checkpointdir = FLAGS.checkpointdir
 264   estimator = get_estimator(config, checkpointdir)
 265 
 266   # Get data configs.
 267   image_attr_keys = config.data.labeled.image_attr_keys
 268   label_attr_keys = config.data.labeled.label_attr_keys
 269   embedding_size = config.embedding_size
 270   num_views = config.data.num_views
 271   k_list = config.val.recall_at_k_list
 272   batch_size = config.data.batch_size
 273 
 274   # Get either labeled validation or test tables.
 275   labeled_tables = get_labeled_tables(config)
 276 
 277   def input_fn_by_view(view_index):
 278     """Returns an input_fn for use with a tf.Estimator by view."""
 279     def input_fn():
 280       # Get raw labeled images.
 281       (preprocessed_images, labels,
 282        tasks) = data_providers.labeled_data_provider(
 283            labeled_tables,
 284            estimator.preprocess_data, view_index, image_attr_keys,
 285            label_attr_keys, batch_size=batch_size)
 286       return {
 287           'batch_preprocessed': preprocessed_images,
 288           'tasks': tasks,
 289           'classification_labels': labels,
 290       }, None
 291     return input_fn
 292 
 293   # If evaluating a specific checkpoint, do that.
 294   if FLAGS.checkpoint_iter:
 295     checkpoint_path = os.path.join(
 296         '%s/model.ckpt-%s' % (checkpointdir, FLAGS.checkpoint_iter))
 297     evaluate_once(
 298         estimator, input_fn_by_view, batch_size, checkpoint_path,
 299         label_attr_keys, embedding_size, num_views, k_list)
 300   else:
 301     for checkpoint_path in tf.contrib.training.checkpoints_iterator(
 302         checkpointdir):
 303       evaluate_once(
 304           estimator, input_fn_by_view, batch_size, checkpoint_path,
 305           label_attr_keys, embedding_size, num_views, k_list)
 306 
 307 
 308 if __name__ == '__main__':
 309   tf.app.run()
