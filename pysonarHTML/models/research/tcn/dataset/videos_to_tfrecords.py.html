  82 def FindPatternFiles(path, view_pattern, errors):
  83   """Recursively find all files matching a certain pattern."""
  84   if not path:
  85     return None
  86   tf.logging.info(
  87       'Recursively searching for files matching pattern \'%s\' in %s' %
  88       (view_pattern, path))
  89   view_patt = re.compile('.*' + view_pattern)
  90   sequences = []
  91   for root, _, filenames in os.walk(path, followlinks=True):
  92     path_root = root[:len(path)]
  93     assert path_root == path
  94 
  95     for filename in filenames:
  96       if view_patt.match(filename):
  97         fullpath = os.path.join(root, re.sub(view_pattern, '', filename))
  98         shortpath = re.sub(path, '', fullpath).lstrip('/')
  99 
 100         # Determine if this sequence should be sharded or not.
 101         shard = False
 102         if FLAGS.max_per_shard > 0:
 103           shard = True
 104 
 105         # Retrieve number of frames for this sequence.
 106         num_views, length, view_paths, num_frames = GetViewInfo(
 107             fullpath + view_pattern[0] + '*')
 108         if num_views != FLAGS.expected_views:
 109           tf.logging.info('Expected %d views but found: %s' %
 110                           (FLAGS.expected_views, str(view_paths)))
 111         assert num_views == FLAGS.expected_views
 112         assert length > 0
 113         # Drop sequences if view lengths differ too much.
 114         if max(num_frames) - min(num_frames) > FLAGS.max_views_discrepancy:
 115           error_msg = (
 116               'Error: ignoring sequence with views with length difference > %d:'
 117               '%s in %s') % (FLAGS.max_views_discrepancy, str(num_frames),
 118                              fullpath)
 119           errors.append(error_msg)
 120           tf.logging.error(error_msg)
 121         else:
 122           # Append sequence info.
 123           sequences.append({'full': fullpath, 'name': shortpath, 'len': length,
 124                             'start': 0, 'end': length, 'num_views': num_views,
 125                             'shard': shard})
 126   return sorted(sequences, key=lambda k: k['name'])

 129 def ShardSequences(sequences, max_per_shard):
 130   """Find all sequences, shard and randomize them."""
 131   total_shards_len = 0
 132   total_shards = 0
 133   assert max_per_shard > 0
 134   for sequence in sequences:
 135     if sequence['shard']:
 136       sequence['shard'] = False  # Reset shard flag.
 137       length = sequence['len']
 138       start = sequence['start']
 139       end = sequence['end']
 140       name = sequence['name']
 141       assert end - start == length
 142       if length > max_per_shard:
 143         # Dividing sequence into smaller shards.
 144         num_shards = int(math.floor(length / max_per_shard)) + 1
 145         size = int(math.ceil(length / num_shards))
 146         tf.logging.info(
 147             'splitting sequence of length %d into %d shards of size %d' %
 148             (length, num_shards, size))
 149         last_end = 0
 150         for i in range(num_shards):
 151           shard_start = last_end
 152           shard_end = min(length, shard_start + size)
 153           if i == num_shards - 1:
 154             shard_end = length
 155           shard_len = shard_end - shard_start
 156           total_shards_len += shard_len
 157           shard_name = name + '_shard%02d' % i
 158           last_end = shard_end
 159 
 160           # Enqueuing shard.
 161           if i == 0:  # Replace current sequence.
 162             sequence['len'] = shard_len
 163             sequence['start'] = shard_start
 164             sequence['end'] = shard_end
 165             sequence['name'] = shard_name
 166           else:  # Enqueue new sequence.
 167             sequences.append(
 168                 {'full': sequence['full'], 'name': shard_name,
 169                  'len': shard_len, 'start': shard_start, 'end': shard_end,
 170                  'num_views': sequence['num_views'], 'shard': False})
 171 
 172         total_shards += num_shards
 173         assert last_end == length
 174 
 175   # Print resulting sharding.
 176   if total_shards > 0:
 177     tf.logging.info('%d shards of average length %d' %
 178                     (total_shards, total_shards_len / total_shards))
 179   return sorted(sequences, key=lambda k: k['name'])

 182 def RandomizeSets(sets):
 183   """Randomize each set."""
 184   for _, sequences in sorted(sets.iteritems()):
 185     if sequences:
 186       # Randomize order.
 187       shuffle(sequences)

 190 def GetSpecificFrame(vid_path, frame_index):
 191   """Gets a frame at a specified index in a video."""
 192   cap = cv2.VideoCapture(vid_path)
 193   cap.set(1, frame_index)
 194   _, bgr = cap.read()
 195   cap.release()
 196   rgb = cv2.cvtColor(bgr, cv2.COLOR_BGR2RGB)
 197   return rgb

 200 def JpegString(image, jpeg_quality=90):
 201   """Returns given PIL.Image instance as jpeg string.
 202 
 203   Args:
 204     image: A PIL image.
 205     jpeg_quality: The image quality, on a scale from 1 (worst) to 95 (best).
 206 
 207   Returns:
 208     a jpeg_string.
 209   """
 210   # This fix to PIL makes sure that we don't get an error when saving large
 211   # jpeg files. This is a workaround for a bug in PIL. The value should be
 212   # substantially larger than the size of the image being saved.
 213   ImageFile.MAXBLOCK = 640 * 512 * 64
 214 
 215   output_jpeg = StringIO()
 216   image.save(output_jpeg, 'jpeg', quality=jpeg_quality, optimize=True)
 217   return output_jpeg.getvalue()

 220 def ParallelPreprocessing(args):
 221   """Parallel preprocessing: rotation, resize and jpeg encoding to string."""
 222   (vid_path, timestep, num_timesteps, view) = args
 223   try:
 224     image = GetSpecificFrame(vid_path, timestep)
 225 
 226     # Resizing.
 227     resize_str = ''
 228     if FLAGS.resize_min_edge > 0:
 229       resize_str += ', resize ' + shapestring(image)
 230       image = cv2resizeminedge(image, FLAGS.resize_min_edge)
 231       resize_str += ' => ' + shapestring(image)
 232 
 233     # Rotating.
 234     rotate = None
 235     if FLAGS.rotate:
 236       rotate = FLAGS.rotate
 237       if FLAGS.rotate_if_matching is not None:
 238         rotate = None
 239         patt = re.compile(FLAGS.rotate_if_matching)
 240         if patt.match(vid_path) is not None:
 241           rotate = FLAGS.rotate
 242       if rotate is not None:
 243         image = cv2rotateimage(image, FLAGS.rotate)
 244 
 245     # Jpeg encoding.
 246     image = Image.fromarray(image)
 247     im_string = bytes_feature([JpegString(image)])
 248 
 249     if timestep % FLAGS.log_frequency == 0:
 250       tf.logging.info('Loaded frame %d / %d for %s (rotation %s%s) from %s' %
 251                       (timestep, num_timesteps, view, str(rotate), resize_str,
 252                        vid_path))
 253     return im_string
 254   except cv2.error as e:
 255     tf.logging.error('Error while loading frame %d of %s: %s' %
 256                      (timestep, vid_path, str(e)))
 257     return None

 260 def GetNumFrames(vid_path):
 261   """Gets the number of frames in a video."""
 262   cap = cv2.VideoCapture(vid_path)
 263   total_frames = cap.get(7)
 264   cap.release()
 265   return int(total_frames)

 268 def GetViewInfo(views_fullname):
 269   """Return information about a group of views."""
 270   view_paths = sorted(glob.glob(views_fullname))
 271   num_frames = [GetNumFrames(i) for i in view_paths]
 272   min_num_frames = min(num_frames)
 273   num_views = len(view_paths)
 274   return num_views, min_num_frames, view_paths, num_frames

 277 def AddSequence(sequence, writer, progress, errors):
 278   """Converts a sequence to a SequenceExample.
 279 
 280   Sequences have multiple viewpoint videos. Extract all frames from all
 281   viewpoint videos in parallel, build a single SequenceExample containing
 282   all viewpoint images for every timestep.
 283 
 284   Args:
 285     sequence: a dict with information on a sequence.
 286     writer: A TFRecordWriter.
 287     progress: A Progress object to report processing progress.
 288     errors: a list of string to append to in case of errors.
 289   """
 290   fullname = sequence['full']
 291   shortname = sequence['name']
 292   start = sequence['start']
 293   end = sequence['end']
 294   num_timesteps = sequence['len']
 295 
 296   # Build a list of all view paths for this fullname.
 297   path = fullname + FLAGS.view_pattern[0] + '*'
 298   tf.logging.info('Loading sequence from ' + path)
 299   view_paths = sorted(glob.glob(path))
 300   # Extract all images for all views
 301   num_frames = [GetNumFrames(i) for i in view_paths]
 302   tf.logging.info('Loading %s with [%d, %d[ (%d frames) from: %s %s' %
 303                   (shortname, start, end, num_timesteps,
 304                    str(num_frames), str(view_paths)))
 305   num_views = len(view_paths)
 306   total_timesteps = int(min(num_frames))
 307   assert num_views == FLAGS.expected_views
 308   assert num_views == sequence['num_views']
 309 
 310   # Create a worker pool to parallelize loading/rotating
 311   worker_pool = ThreadPool(multiprocessing.cpu_count())
 312 
 313   # Collect all images for each view.
 314   view_to_feature_list = {}
 315   view_images = []
 316   for view_idx, view in enumerate(
 317       ['view'+str(i) for i in range(num_views)]):
 318     # Flatten list to process in parallel
 319     work = []
 320     for i in range(start, end):
 321       work.append((view_paths[view_idx], i, total_timesteps, view))
 322     # Load and rotate images in parallel
 323     view_images.append(worker_pool.map(ParallelPreprocessing, work))
 324     # Report progress.
 325     progress.Add(len(view_images[view_idx]))
 326     tf.logging.info('%s' % str(progress))
 327 
 328   # Remove error frames from all views
 329   i = start
 330   num_errors = 0
 331   while i < len(view_images[0]):
 332     remove_frame = False
 333     # Check if one or more views have an error for this frame.
 334     for view_idx in range(num_views):
 335       if view_images[view_idx][i] is None:
 336         remove_frame = True
 337         error_msg = 'Removing frame %d for all views for %s ' % (i, fullname)
 338         errors.append(error_msg)
 339         tf.logging.error(error_msg)
 340     # Remove faulty frames.
 341     if remove_frame:
 342       num_errors += 1
 343       for view_idx in range(num_views):
 344         del view_images[view_idx][i]
 345     else:
 346       i += 1
 347 
 348   # Ignore sequences that have errors.
 349   if num_errors > 0:
 350     error_msg = 'Dropping sequence because of frame errors for %s' % fullname
 351     errors.append(error_msg)
 352     tf.logging.error(error_msg)
 353   else:
 354     # Build FeatureList objects for each view.
 355     for view_idx, view in enumerate(
 356         ['view'+str(i) for i in range(num_views)]):
 357       # Construct FeatureList from repeated feature.
 358       view_to_feature_list[view] = tf.train.FeatureList(
 359           feature=view_images[view_idx])
 360 
 361     context_features = tf.train.Features(feature={
 362         'task': bytes_feature([shortname]),
 363         'len': int64_feature([num_timesteps])
 364     })
 365     feature_lists = tf.train.FeatureLists(feature_list=view_to_feature_list)
 366     ex = tf.train.SequenceExample(
 367         context=context_features, feature_lists=feature_lists)
 368     writer.write(ex.SerializeToString())
 369     tf.logging.info('Done adding %s with %d timesteps'
 370                     % (fullname, num_timesteps))

 373 def PrintSequencesInfo(sequences, prefix):
 374   """Print information about sequences and return the total number of frames."""
 375   tf.logging.info('')
 376   tf.logging.info(prefix)
 377   num_frames = 0
 378   for sequence in sequences:
 379     shard_str = ''
 380     if sequence['shard']:
 381       shard_str = ' (sharding)'
 382     tf.logging.info('frames [%d, %d[\t(%d frames * %d views)%s\t%s' % (
 383         sequence['start'], sequence['end'], sequence['len'],
 384         sequence['num_views'], shard_str, sequence['name']))
 385     num_frames += sequence['len'] * sequence['num_views']
 386   tf.logging.info(('%d frames (all views), %d sequences, average sequence'
 387                    ' length (all views): %d') %
 388                   (num_frames, len(sequences), num_frames / len(sequences)))
 389   tf.logging.info('')
 390   return num_frames

 393 def CheckRecord(filename, sequence):
 394   """Check that an existing tfrecord corresponds to the expected sequence."""
 395   num_sequences = 0
 396   total_frames = 0
 397   for serialized_example in tf.python_io.tf_record_iterator(filename):
 398     num_sequences += 1
 399     example = tf.train.SequenceExample()
 400     example.ParseFromString(serialized_example)
 401     length = example.context.feature['len'].int64_list.value[0]
 402     name = example.context.feature['task'].bytes_list.value[0]
 403     total_frames += len(example.feature_lists.feature_list) * length
 404     if sequence['name'] != name or sequence['len'] != length:
 405       return False, total_frames
 406   if num_sequences == 0:
 407     return False, total_frames
 408   return True, total_frames

 411 def AddSequences():
 412   """Creates one training, validation."""
 413   errors = []
 414 
 415   # Generate datasets file lists.
 416   sequences = FindPatternFiles(FLAGS.input_dir, FLAGS.view_pattern, errors)
 417   num_frames = PrintSequencesInfo(sequences,
 418                                   'Found the following datasets and files:')
 419 
 420   # Sharding and randomizing sets.
 421   if FLAGS.max_per_shard > 0:
 422     sequences = ShardSequences(sequences, FLAGS.max_per_shard)
 423     num_frames = PrintSequencesInfo(sequences, 'After sharding:')
 424     tf.logging.info('')
 425 
 426   # Process sets.
 427   progress = Progress(num_frames)
 428   output_list = []
 429   for sequence in sequences:
 430     record_name = os.path.join(
 431         FLAGS.output_dir, '%s.tfrecord' % sequence['name'])
 432     if tf.gfile.Exists(record_name) and not FLAGS.overwrite:
 433       ok, num_frames = CheckRecord(record_name, sequence)
 434       if ok:
 435         progress.Add(num_frames)
 436         tf.logging.info('Skipping existing output file: %s' % record_name)
 437         continue
 438       else:
 439         tf.logging.info('File does not match sequence, reprocessing...')
 440     output_dir = os.path.dirname(record_name)
 441     if not tf.gfile.Exists(output_dir):
 442       tf.logging.info('Creating output directory: %s' % output_dir)
 443       tf.gfile.MakeDirs(output_dir)
 444     output_list.append(record_name)
 445     tf.logging.info('Writing to ' + record_name)
 446     writer = tf.python_io.TFRecordWriter(record_name)
 447     AddSequence(sequence, writer, progress, errors)
 448     writer.close()
 449   tf.logging.info('Wrote dataset files: ' + str(output_list))
 450   tf.logging.info('All errors (%d): %s' % (len(errors), str(errors)))
