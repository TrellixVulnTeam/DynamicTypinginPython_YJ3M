  94 class ImageQueue(object):
  95   """An image queue holding each stream's most recent image.
  96 
  97   Basically implements a process-safe collections.deque(maxlen=1).
  98   """
  99 
 100   def __init__(self):
 101     self.lock = multiprocessing.Lock()
 102     self._queue = multiprocessing.Queue(maxsize=1)
 103 
 104   def append(self, data):
 105     with self.lock:
 106       if self._queue.full():
 107         # Pop the first element.
 108         _ = self._queue.get()
 109       self._queue.put(data)
 110 
 111   def get(self):
 112     with self.lock:
 113       return self._queue.get()
 114 
 115   def empty(self):
 116     return self._queue.empty()
 117 
 118   def close(self):
 119     return self._queue.close()

 122 class WebcamViewer(object):
 123   """A class which displays a live stream from the webcams."""
 124 
 125   def __init__(self, display_queues):
 126     """Create a WebcamViewer instance."""
 127     self.height = FLAGS.height
 128     self.width = FLAGS.width
 129     self.queues = display_queues
 130 
 131   def _get_next_images(self):
 132     """Gets the next image to display."""
 133     # Wait for one image per view.
 134     not_found = True
 135     while not_found:
 136       if True in [q.empty() for q in self.queues]:
 137         # At least one image queue is empty; wait.
 138         continue
 139       else:
 140         # Retrieve the images.
 141         latest = [q.get() for q in self.queues]
 142         combined = np.concatenate(latest, axis=1)
 143       not_found = False
 144     return combined
 145 
 146   def run(self):
 147     """Displays the Kcam live stream in a window.
 148 
 149     This function blocks until the window is closed.
 150     """
 151     fig, rgb_axis = plt.subplots()
 152 
 153     image_rows = self.height
 154     image_cols = self.width * FLAGS.num_views
 155     initial_image = np.zeros((image_rows, image_cols, 3))
 156     rgb_image = rgb_axis.imshow(initial_image, interpolation='nearest')
 157 
 158     def update_figure(frame_index):
 159       """Animation function for matplotlib FuncAnimation. Updates the image.
 160 
 161       Args:
 162         frame_index: The frame number.
 163       Returns:
 164         An iterable of matplotlib drawables to clear.
 165       """
 166       _ = frame_index
 167       images = self._get_next_images()
 168       images = images[..., [2, 1, 0]]
 169       rgb_image.set_array(images)
 170       return rgb_image,
 171 
 172     # We must keep a reference to this animation in order for it to work.
 173     unused_animation = animation.FuncAnimation(
 174         fig, update_figure, interval=50, blit=True)
 175     mng = plt.get_current_fig_manager()
 176     mng.resize(*mng.window.maxsize())
 177     plt.show()

 180 def reconcile(queues, write_queue):
 181   """Gets a list of concurrent images from each view queue.
 182 
 183   This waits for latest images to be available in all view queues,
 184   then continuously:
 185   - Creates a list of current images for each view.
 186   - Writes the list to a queue of image lists to write to disk.
 187   Args:
 188     queues: A list of `ImageQueues`, holding the latest image from each webcam.
 189     write_queue: A multiprocessing.Queue holding lists of concurrent images.
 190   """
 191   # Loop forever.
 192   while True:
 193     # Wait till all queues have an image.
 194     if True in [q.empty() for q in queues]:
 195       continue
 196     else:
 197       # Retrieve all views' images.
 198       latest = [q.get() for q in queues]
 199       # Copy the list of all concurrent images to the write queue.
 200       write_queue.put(latest)

 203 def persist(write_queue, view_dirs):
 204   """Pulls lists of concurrent images off a write queue, writes them to disk.
 205 
 206   Args:
 207     write_queue: A multiprocessing.Queue holding lists of concurrent images;
 208       one image per view.
 209     view_dirs: A list of strings, holding the output image directories for each
 210       view.
 211   """
 212   timestep = 0
 213   while True:
 214     # Wait till there is work in the queue.
 215     if write_queue.empty():
 216       continue
 217     # Get a list of concurrent images to write to disk.
 218     view_ims = write_queue.get()
 219     for view_idx, image in enumerate(view_ims):
 220       view_base = view_dirs[view_idx]
 221       # Assign all concurrent view images the same sequence timestep.
 222       fname = os.path.join(view_base, '%s.png' % str(timestep).zfill(10))
 223       cv2.imwrite(fname, image)
 224     # Move to the next timestep.
 225     timestep += 1

 228 def get_image(camera):
 229   """Captures a single image from the camera and returns it in PIL format."""
 230   data = camera.read()
 231   _, im = data
 232   return im

 235 def capture_webcam(camera, display_queue, reconcile_queue):
 236   """Captures images from simultaneous webcams, writes them to queues.
 237 
 238   Args:
 239     camera: A cv2.VideoCapture object representing an open webcam stream.
 240     display_queue: An ImageQueue.
 241     reconcile_queue: An ImageQueue.
 242   """
 243   # Take some ramp images to allow cams to adjust for brightness etc.
 244   for i in range(60):
 245     tf.logging.info('Taking ramp image %d.' % i)
 246     get_image(camera)
 247 
 248   cnt = 0
 249   start = time.time()
 250   while True:
 251     # Get images for all cameras.
 252     im = get_image(camera)
 253     # Replace the current image in the display and reconcile queues.
 254     display_queue.append(im)
 255     reconcile_queue.append(im)
 256     cnt += 1
 257     current = time.time()
 258     if cnt % 100 == 0:
 259       tf.logging.info('Collected %s of video, %d frames at ~%.2f fps.' % (
 260           timer(start, current), cnt, cnt/(current-start)))
 261

 263 def timer(start, end):
 264   """Returns a formatted time elapsed."""
 265   hours, rem = divmod(end-start, 3600)
 266   minutes, seconds = divmod(rem, 60)
 267   return '{:0>2}:{:0>2}:{:05.2f}'.format(int(hours), int(minutes), seconds)

 276 def create_vids(view_dirs, seqname):
 277   """Creates one video per view per sequence."""
 278   vidbase = os.path.join(FLAGS.viddir, FLAGS.dataset, FLAGS.mode)
 279   if not os.path.exists(vidbase):
 280     os.makedirs(vidbase)
 281   vidpaths = []
 282   for idx, view_dir in enumerate(view_dirs):
 283     vidname = os.path.join(vidbase, '%s_view%d.mp4' % (seqname, idx))
 284     encode_vid_cmd = r'mencoder mf://%s/*.png \
 285     -mf fps=29:type=png \
 286     -ovc lavc -lavcopts vcodec=mpeg4:mbd=2:trell \
 287     -oac copy -o %s' % (view_dir, vidname)
 288     os.system(encode_vid_cmd)
 289     vidpaths.append(vidname)
 290 
 291   debugpath = None
 292   if FLAGS.debug_vids:
 293     lhs = vidpaths[FLAGS.debug_lhs_view]
 294     rhs = vidpaths[FLAGS.debug_rhs_view]
 295     debug_base = os.path.join('%s_debug' % FLAGS.viddir, FLAGS.dataset,
 296                               FLAGS.mode)
 297     if not os.path.exists(debug_base):
 298       os.makedirs(debug_base)
 299     debugpath = '%s/%s.mp4' % (debug_base, seqname)
 300     os.system(r"avconv \
 301       -i %s \
 302       -i %s \
 303       -filter_complex '[0:v]pad=iw*2:ih[int];[int][1:v]overlay=W/2:0[vid]' \
 304       -map [vid] \
 305       -c:v libx264 \
 306       -crf 23 \
 307       -preset veryfast \
 308       %s" % (lhs, rhs, debugpath))
 309 
 310   return vidpaths, debugpath

 313 def setup_paths():
 314   """Sets up the necessary paths to collect videos."""
 315   assert FLAGS.dataset
 316   assert FLAGS.mode
 317   assert FLAGS.num_views
 318 
 319   # Setup directory for final images used to create videos for this sequence.
 320   tmp_imagedir = os.path.join(FLAGS.tmp_imagedir, FLAGS.dataset, FLAGS.mode)
 321   if not os.path.exists(tmp_imagedir):
 322     os.makedirs(tmp_imagedir)
 323 
 324   # Create a base directory to hold all sequence videos if it doesn't exist.
 325   vidbase = os.path.join(FLAGS.viddir, FLAGS.dataset, FLAGS.mode)
 326   if not os.path.exists(vidbase):
 327     os.makedirs(vidbase)
 328 
 329   # Get one directory per concurrent view and a sequence name.
 330   view_dirs, seqname = get_view_dirs(vidbase, tmp_imagedir)
 331 
 332   # Get an output path to each view's video.
 333   vid_paths = []
 334   for idx, _ in enumerate(view_dirs):
 335     vid_path = os.path.join(vidbase, '%s_view%d.mp4' % (seqname, idx))
 336     vid_paths.append(vid_path)
 337 
 338   # Optionally build paths to debug_videos.
 339   debug_path = None
 340   if FLAGS.debug_vids:
 341     debug_base = os.path.join('%s_debug' % FLAGS.viddir, FLAGS.dataset,
 342                               FLAGS.mode)
 343     if not os.path.exists(debug_base):
 344       os.makedirs(debug_base)
 345     debug_path = '%s/%s.mp4' % (debug_base, seqname)
 346 
 347   return view_dirs, vid_paths, debug_path

 350 def get_view_dirs(vidbase, tmp_imagedir):
 351   """Creates and returns one view directory per webcam."""
 352   # Create and append a sequence name.
 353   if FLAGS.seqname:
 354     seqname = FLAGS.seqname
 355   else:
 356     # If there's no video directory, this is the first sequence.
 357     if not os.listdir(vidbase):
 358       seqname = '0'
 359     else:
 360       # Otherwise, get the latest sequence name and increment it.
 361       seq_names = [i.split('_')[0] for i in os.listdir(vidbase)]
 362       latest_seq = sorted(map(int, seq_names), reverse=True)[0]
 363       seqname = str(latest_seq+1)
 364     tf.logging.info('No seqname specified, using: %s' % seqname)
 365   view_dirs = [os.path.join(
 366       tmp_imagedir, '%s_view%d' % (seqname, v)) for v in range(FLAGS.num_views)]
 367   for d in view_dirs:
 368     if not os.path.exists(d):
 369       os.makedirs(d)
 370   return view_dirs, seqname

 373 def get_cameras():
 374   """Opens cameras using cv2, ensures they can take images."""
 375   # Try to get free webcam ports.
 376   if FLAGS.webcam_ports:
 377     ports = map(int, FLAGS.webcam_ports.split(','))
 378   else:
 379     ports = range(FLAGS.num_views)
 380   cameras = [cv2.VideoCapture(i) for i in ports]
 381 
 382   if not all([i.isOpened() for i in cameras]):
 383     try:
 384       # Try to find and kill hanging cv2 process_ids.
 385       output = subprocess.check_output(['lsof -t /dev/video*'], shell=True)
 386       tf.logging.info('Found hanging cv2 process_ids: \n')
 387       tf.logging.info(output)
 388       tf.logging.info('Killing hanging processes...')
 389       for process_id in output.split('\n')[:-1]:
 390         subprocess.call(['kill %s' % process_id], shell=True)
 391       time.sleep(3)
 392       # Recapture webcams.
 393       cameras = [cv2.VideoCapture(i) for i in ports]
 394     except subprocess.CalledProcessError:
 395       raise ValueError(
 396           'Cannot connect to cameras. Try running: \n'
 397           'ls -ltrh /dev/video* \n '
 398           'to see which ports your webcams are connected to. Then hand those '
 399           'ports as a comma-separated list to --webcam_ports, e.g. '
 400           '--webcam_ports 0,1')
 401 
 402   # Verify each camera is able to capture images.
 403   ims = map(get_image, cameras)
 404   assert False not in [i is not None for i in ims]
 405   return cameras

 408 def launch_images_to_videos(view_dirs, vid_paths, debug_path):
 409   """Launch job in separate process to convert images to videos."""
 410 
 411   f = 'learning/brain/research/tcn/dataset/images_to_videos.py'
 412   cmd = ['python %s ' % f]
 413   cmd += ['--view_dirs %s ' % ','.join(i for i in view_dirs)]
 414   cmd += ['--vid_paths %s ' % ','.join(i for i in vid_paths)]
 415   cmd += ['--debug_path %s ' % debug_path]
 416   cmd += ['--debug_lhs_view %s ' % FLAGS.debug_lhs_view]
 417   cmd += ['--debug_rhs_view %s ' % FLAGS.debug_rhs_view]
 418   cmd += [' & ']
 419   cmd = ''.join(i for i in cmd)
 420 
 421   # Call images_to_videos asynchronously.
 422   fnull = open(os.devnull, 'w')
 423   subprocess.Popen([cmd], stdout=fnull, stderr=subprocess.STDOUT, shell=True)
 424 
 425   for p in vid_paths:
 426     tf.logging.info('Writing final video to: %s' % p)
 427   if debug_path:
 428     tf.logging.info('Writing debug video to: %s' % debug_path)

 431 def main(_):
 432   # Initialize the camera capture objects.
 433   cameras = get_cameras()
 434   # Get one output directory per view.
 435   view_dirs, vid_paths, debug_path = setup_paths()
 436   try:
 437     # Wait for user input.
 438     try:
 439       tf.logging.info('About to write to:')
 440       for v in view_dirs:
 441         tf.logging.info(v)
 442       input('Press Enter to continue...')
 443     except SyntaxError:
 444       pass
 445 
 446     # Create a queue per view for displaying and saving images.
 447     display_queues = [ImageQueue() for _ in range(FLAGS.num_views)]
 448     reconcile_queues = [ImageQueue() for _ in range(FLAGS.num_views)]
 449 
 450     # Create a queue for collecting all tuples of multi-view images to write to
 451     # disk.
 452     write_queue = multiprocessing.Queue()
 453 
 454     processes = []
 455     # Create a process to display collected images in real time.
 456     processes.append(Process(target=display_webcams, args=(display_queues,)))
 457     # Create a process to collect the latest simultaneous images from each view.
 458     processes.append(Process(
 459         target=reconcile, args=(reconcile_queues, write_queue,)))
 460     # Create a process to collect the latest simultaneous images from each view.
 461     processes.append(Process(
 462         target=persist, args=(write_queue, view_dirs,)))
 463 
 464     for (cam, dq, rq) in zip(cameras, display_queues, reconcile_queues):
 465       processes.append(Process(
 466           target=capture_webcam, args=(cam, dq, rq,)))
 467 
 468     for p in processes:
 469       p.start()
 470     for p in processes:
 471       p.join()
 472 
 473   except KeyboardInterrupt:
 474     # Close the queues.
 475     for q in display_queues + reconcile_queues:
 476       q.close()
 477     # Release the cameras.
 478     for cam in cameras:
 479       cam.release()
 480 
 481     # Launch images_to_videos script asynchronously.
 482     launch_images_to_videos(view_dirs, vid_paths, debug_path)
 483 
 484     try:
 485       sys.exit(0)
 486     except SystemExit:
 487       os._exit(0)  # pylint: disable=protected-access
 488

 490 if __name__ == '__main__':
 491   tf.app.run()
