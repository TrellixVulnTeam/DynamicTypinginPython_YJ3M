  77 def images_to_sprite(data):
  78   """Creates the sprite image along with any necessary padding.
  79 
  80   Taken from: https://github.com/tensorflow/tensorflow/issues/6322
  81 
  82   Args:
  83     data: NxHxW[x3] tensor containing the images.
  84 
  85   Returns:
  86     data: Properly shaped HxWx3 image with any necessary padding.
  87   """
  88   if len(data.shape) == 3:
  89     data = np.tile(data[..., np.newaxis], (1, 1, 1, 3))
  90   data = data.astype(np.float32)
  91   min_v = np.min(data.reshape((data.shape[0], -1)), axis=1)
  92   data = (data.transpose(1, 2, 3, 0) - min_v).transpose(3, 0, 1, 2)
  93   max_v = np.max(data.reshape((data.shape[0], -1)), axis=1)
  94   data = (data.transpose(1, 2, 3, 0) / max_v).transpose(3, 0, 1, 2)
  95   n = int(np.ceil(np.sqrt(data.shape[0])))
  96   padding = ((0, n ** 2 - data.shape[0]), (0, 0),
  97              (0, 0)) + ((0, 0),) * (data.ndim - 3)
  98   data = np.pad(data, padding, mode='constant',
  99                 constant_values=0)
 100   # Tile the individual thumbnails into an image.
 101   data = data.reshape((n, n) + data.shape[1:]).transpose(
 102       (0, 2, 1, 3) + tuple(range(4, data.ndim + 1)))
 103   data = data.reshape((n * data.shape[1], n * data.shape[3]) + data.shape[4:])
 104   data = (data * 255).astype(np.uint8)
 105   return data

 108 def main(_):
 109   """Runs main labeled eval loop."""
 110   # Parse config dict from yaml config files / command line flags.
 111   config = util.ParseConfigsToLuaTable(FLAGS.config_paths, FLAGS.model_params)
 112 
 113   # Choose an estimator based on training strategy.
 114   checkpointdir = FLAGS.checkpointdir
 115   checkpoint_path = os.path.join(
 116       '%s/model.ckpt-%s' % (checkpointdir, FLAGS.checkpoint_iter))
 117   estimator = get_estimator(config, checkpointdir)
 118 
 119   # Get records to embed.
 120   validation_dir = FLAGS.embedding_records
 121   validation_records = util.GetFilesRecursively(validation_dir)
 122 
 123   sequences_to_data = {}
 124   for (view_embeddings, view_raw_image_strings, seqname) in estimator.inference(
 125       validation_records, checkpoint_path, config.data.embed_batch_size,
 126       num_sequences=FLAGS.num_sequences):
 127     sequences_to_data[seqname] = {
 128         'embeddings': view_embeddings,
 129         'images': view_raw_image_strings,
 130     }
 131 
 132   all_embeddings = np.zeros((0, config.embedding_size))
 133   all_ims = []
 134   all_seqnames = []
 135 
 136   num_embeddings = FLAGS.num_embed
 137   # Concatenate all views from all sequences into a big flat list.
 138   for seqname, data in sequences_to_data.iteritems():
 139     embs = data['embeddings']
 140     ims = data['images']
 141     for v in range(config.data.num_views):
 142       for (emb, im) in zip(embs[v], ims[v]):
 143         all_embeddings = np.append(all_embeddings, [emb], axis=0)
 144         all_ims.append(im)
 145         all_seqnames.append(seqname)
 146 
 147   # Choose N indices uniformly from all images.
 148   random_indices = range(all_embeddings.shape[0])
 149   random.shuffle(random_indices)
 150   viz_indices = random_indices[:num_embeddings]
 151 
 152   # Extract embs.
 153   viz_embs = np.array(all_embeddings[viz_indices])
 154 
 155   # Extract and decode ims.
 156   viz_ims = list(np.array(all_ims)[viz_indices])
 157   decoded_ims = []
 158 
 159   sprite_dim = FLAGS.sprite_dim
 160   for i, im in enumerate(viz_ims):
 161     if i % 100 == 0:
 162       print('Decoding image %d/%d.' % (i, num_embeddings))
 163     nparr_i = np.fromstring(str(im), np.uint8)
 164     img_np = cv2.imdecode(nparr_i, 1)
 165     img_np = img_np[..., [2, 1, 0]]
 166 
 167     img_np = imresize(img_np, [sprite_dim, sprite_dim, 3])
 168     decoded_ims.append(img_np)
 169   decoded_ims = np.array(decoded_ims)
 170 
 171   # Extract sequence names.
 172   outdir = FLAGS.outdir
 173 
 174   # The embedding variable, which needs to be stored
 175   # Note this must a Variable not a Tensor!
 176   embedding_var = tf.Variable(viz_embs, name='viz_embs')
 177 
 178   with tf.Session() as sess:
 179     sess.run(embedding_var.initializer)
 180     summary_writer = tf.summary.FileWriter(outdir)
 181     config = projector.ProjectorConfig()
 182     embedding = config.embeddings.add()
 183     embedding.tensor_name = embedding_var.name
 184 
 185     # Comment out if you don't want sprites
 186     embedding.sprite.image_path = os.path.join(outdir, 'sprite.png')
 187     embedding.sprite.single_image_dim.extend(
 188         [decoded_ims.shape[1], decoded_ims.shape[1]])
 189 
 190     projector.visualize_embeddings(summary_writer, config)
 191     saver = tf.train.Saver([embedding_var])
 192     saver.save(sess, os.path.join(outdir, 'model2.ckpt'), 1)
 193 
 194   sprite = images_to_sprite(decoded_ims)
 195   imsave(os.path.join(outdir, 'sprite.png'), sprite)
 196 
 197 if __name__ == '__main__':
 198   tf.app.run(main)
