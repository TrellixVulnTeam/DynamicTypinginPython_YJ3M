  25 def pairwise_squared_distance(feature):
  26   """Computes the squared pairwise distance matrix.
  27 
  28   output[i, j] = || feature[i, :] - feature[j, :] ||_2^2
  29 
  30   Args:
  31     feature: 2-D Tensor of size [number of data, feature dimension]
  32 
  33   Returns:
  34     pairwise_squared_distances: 2-D Tensor of size
  35       [number of data, number of data]
  36   """
  37   pairwise_squared_distances = tf.add(
  38       tf.reduce_sum(
  39           tf.square(feature), axis=1, keep_dims=True),
  40       tf.reduce_sum(
  41           tf.square(tf.transpose(feature)), axis=0,
  42           keep_dims=True)) - 2.0 * tf.matmul(feature, tf.transpose(feature))
  43 
  44   # Deal with numerical inaccuracies. Set small negatives to zero.
  45   pairwise_squared_distances = tf.maximum(pairwise_squared_distances, 0.0)
  46   return pairwise_squared_distances

  49 def masked_maximum(data, mask, dim=1):
  50   """Computes the axis wise maximum over chosen elements.
  51 
  52   Args:
  53     data: N-D Tensor.
  54     mask: N-D Tensor of zeros or ones.
  55     dim: The dimension over which to compute the maximum.
  56 
  57   Returns:
  58     masked_maximums: N-D Tensor.
  59       The maximized dimension is of size 1 after the operation.
  60   """
  61   axis_minimums = tf.reduce_min(data, dim, keep_dims=True)
  62   masked_maximums = tf.reduce_max(
  63       tf.multiply(
  64           data - axis_minimums, mask), dim, keep_dims=True) + axis_minimums
  65   return masked_maximums

  68 def masked_minimum(data, mask, dim=1):
  69   """Computes the axis wise minimum over chosen elements.
  70 
  71   Args:
  72     data: 2-D Tensor of size [n, m].
  73     mask: 2-D Boolean Tensor of size [n, m].
  74     dim: The dimension over which to compute the minimum.
  75 
  76   Returns:
  77     masked_minimums: N-D Tensor.
  78       The minimized dimension is of size 1 after the operation.
  79   """
  80   axis_maximums = tf.reduce_max(data, dim, keep_dims=True)
  81   masked_minimums = tf.reduce_min(
  82       tf.multiply(
  83           data - axis_maximums, mask), dim, keep_dims=True) + axis_maximums
  84   return masked_minimums

  87 def singleview_tcn_loss(
  88     embeddings, timesteps, pos_radius, neg_radius, margin=1.0,
  89     sequence_ids=None, multiseq=False):
  90   """Computes the single view triplet loss with semi-hard negative mining.
  91 
  92   The loss encourages the positive distances (between a pair of embeddings with
  93   the same labels) to be smaller than the minimum negative distance among
  94   which are at least greater than the positive distance plus the margin constant
  95   (called semi-hard negative) in the mini-batch. If no such negative exists,
  96   uses the largest negative distance instead.
  97 
  98   Anchor, positive, negative selection is as follow:
  99   Anchors: We consider every embedding timestep as an anchor.
 100   Positives: pos_radius defines a radius (in timesteps) around each anchor from
 101     which positives can be drawn. E.g. An anchor with t=10 and a pos_radius of
 102     2 produces a set of 4 (anchor,pos) pairs [(a=10, p=8), ... (a=10, p=12)].
 103   Negatives: neg_radius defines a boundary (in timesteps) around each anchor,
 104     outside of which negatives can be drawn. E.g. An anchor with t=10 and a
 105     neg_radius of 4 means negatives can be any t_neg where t_neg < 6 and
 106     t_neg > 14.
 107 
 108   Args:
 109     embeddings: 2-D Tensor of embedding vectors.
 110     timesteps: 1-D Tensor with shape [batch_size, 1] of sequence timesteps.
 111     pos_radius: int32; the size of the window (in timesteps) around each anchor
 112       timestep that a positive can be drawn from.
 113     neg_radius: int32; the size of the window (in timesteps) around each anchor
 114       timestep that defines a negative boundary. Negatives can only be chosen
 115       where negative timestep t is < negative boundary min or > negative
 116       boundary max.
 117     margin: Float; the triplet loss margin hyperparameter.
 118     sequence_ids: (Optional) 1-D Tensor with shape [batch_size, 1] of sequence
 119       ids. Together (sequence_id, sequence_timestep) give us a unique index for
 120       each image if we have multiple sequences in a batch.
 121     multiseq: Boolean, whether or not the batch is composed of multiple
 122       sequences (with possibly colliding timesteps).
 123 
 124   Returns:
 125     triplet_loss: tf.float32 scalar.
 126   """
 127   assert neg_radius > pos_radius
 128 
 129   # If timesteps shape isn't [batchsize, 1], reshape to [batch_size, 1].
 130   tshape = tf.shape(timesteps)
 131   assert tshape.shape == 2 or tshape.shape == 1
 132   if tshape.shape == 1:
 133     timesteps = tf.reshape(timesteps, [tshape[0], 1])
 134 
 135   # Build pairwise squared distance matrix.
 136   pdist_matrix = pairwise_squared_distance(embeddings)
 137 
 138   # Build pairwise binary adjacency matrix, where adjacency[i,j] is True
 139   # if timestep j is inside the positive range for timestep i and both
 140   # timesteps come from the same sequence.
 141   pos_radius = tf.cast(pos_radius, tf.int32)
 142 
 143   if multiseq:
 144     # If sequence_ids shape isn't [batchsize, 1], reshape to [batch_size, 1].
 145     tshape = tf.shape(sequence_ids)
 146     assert tshape.shape == 2 or tshape.shape == 1
 147     if tshape.shape == 1:
 148       sequence_ids = tf.reshape(sequence_ids, [tshape[0], 1])
 149 
 150     # Build pairwise binary adjacency matrix based on sequence_ids
 151     sequence_adjacency = tf.equal(sequence_ids, tf.transpose(sequence_ids))
 152 
 153     # Invert so we can select negatives only.
 154     sequence_adjacency_not = tf.logical_not(sequence_adjacency)
 155 
 156     in_pos_range = tf.logical_and(
 157         tf.less_equal(
 158             tf.abs(timesteps - tf.transpose(timesteps)), pos_radius),
 159         sequence_adjacency)
 160     # Build pairwise binary discordance matrix, where discordance[i,j] is True
 161     # if timestep j is inside the negative range for timestep i or if the
 162     # timesteps come from different sequences.
 163     in_neg_range = tf.logical_or(
 164         tf.greater(tf.abs(timesteps - tf.transpose(timesteps)), neg_radius),
 165         sequence_adjacency_not
 166     )
 167   else:
 168     in_pos_range = tf.less_equal(
 169         tf.abs(timesteps - tf.transpose(timesteps)), pos_radius)
 170     in_neg_range = tf.greater(tf.abs(timesteps - tf.transpose(timesteps)),
 171                               neg_radius)
 172 
 173   batch_size = tf.size(timesteps)
 174 
 175   # compute the mask
 176   pdist_matrix_tile = tf.tile(pdist_matrix, [batch_size, 1])
 177   mask = tf.logical_and(
 178       tf.tile(in_neg_range, [batch_size, 1]),
 179       tf.greater(pdist_matrix_tile,
 180                  tf.reshape(tf.transpose(pdist_matrix), [-1, 1])))
 181   mask_final = tf.reshape(
 182       tf.greater(
 183           tf.reduce_sum(
 184               tf.cast(
 185                   mask, dtype=tf.float32), 1, keep_dims=True),
 186           0.0), [batch_size, batch_size])
 187   mask_final = tf.transpose(mask_final)
 188 
 189   in_neg_range = tf.cast(in_neg_range, dtype=tf.float32)
 190   mask = tf.cast(mask, dtype=tf.float32)
 191 
 192   # negatives_outside: smallest D_an where D_an > D_ap
 193   negatives_outside = tf.reshape(
 194       masked_minimum(pdist_matrix_tile, mask), [batch_size, batch_size])
 195   negatives_outside = tf.transpose(negatives_outside)
 196 
 197   # negatives_inside: largest D_an
 198   negatives_inside = tf.tile(
 199       masked_maximum(pdist_matrix, in_neg_range), [1, batch_size])
 200   semi_hard_negatives = tf.where(
 201       mask_final, negatives_outside, negatives_inside)
 202 
 203   loss_mat = tf.add(margin, pdist_matrix - semi_hard_negatives)
 204 
 205   mask_positives = tf.cast(
 206       in_pos_range, dtype=tf.float32) - tf.diag(tf.ones([batch_size]))
 207 
 208   # In lifted-struct, the authors multiply 0.5 for upper triangular
 209   #   in semihard, they take all positive pairs except the diagonal.
 210   num_positives = tf.reduce_sum(mask_positives)
 211 
 212   triplet_loss = tf.truediv(
 213       tf.reduce_sum(tf.maximum(tf.multiply(loss_mat, mask_positives), 0.0)),
 214       num_positives,
 215       name='triplet_svtcn_loss')
 216 
 217   return triplet_loss
