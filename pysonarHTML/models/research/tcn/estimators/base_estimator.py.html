  42 class InitFromPretrainedCheckpointHook(session_run_hook.SessionRunHook):
  43   """Hook that can init graph from a pretrained checkpoint."""
  44 
  45   def __init__(self, pretrained_checkpoint_dir):
  46     """Initializes a `InitFromPretrainedCheckpointHook`.
  47 
  48     Args:
  49       pretrained_checkpoint_dir: The dir of pretrained checkpoint.
  50 
  51     Raises:
  52       ValueError: If pretrained_checkpoint_dir is invalid.
  53     """
  54     if pretrained_checkpoint_dir is None:
  55       raise ValueError('pretrained_checkpoint_dir must be specified.')
  56     self._pretrained_checkpoint_dir = pretrained_checkpoint_dir
  57 
  58   def begin(self):
  59     checkpoint_reader = tf.contrib.framework.load_checkpoint(
  60         self._pretrained_checkpoint_dir)
  61     variable_shape_map = checkpoint_reader.get_variable_to_shape_map()
  62 
  63     exclude_scopes = 'logits/,final_layer/,aux_'
  64     # Skip restoring global_step as to run fine tuning from step=0.
  65     exclusions = ['global_step']
  66     if exclude_scopes:
  67       exclusions.extend([scope.strip() for scope in exclude_scopes.split(',')])
  68 
  69     variable_to_restore = tf.contrib.framework.get_model_variables()
  70 
  71     # Variable filtering by given exclude_scopes.
  72     filtered_variables_to_restore = {}
  73     for v in variable_to_restore:
  74       for exclusion in exclusions:
  75         if v.name.startswith(exclusion):
  76           break
  77       else:
  78         var_name = v.name.split(':')[0]
  79         filtered_variables_to_restore[var_name] = v
  80 
  81     # Final filter by checking shape matching and skipping variables that
  82     # are not in the checkpoint.
  83     final_variables_to_restore = {}
  84     for var_name, var_tensor in filtered_variables_to_restore.iteritems():
  85       if var_name not in variable_shape_map:
  86         # Try moving average version of variable.
  87         var_name = os.path.join(var_name, 'ExponentialMovingAverage')
  88         if var_name not in variable_shape_map:
  89           tf.logging.info(
  90               'Skip init [%s] because it is not in ckpt.', var_name)
  91           # Skip variables not in the checkpoint.
  92           continue
  93 
  94       if not var_tensor.get_shape().is_compatible_with(
  95           variable_shape_map[var_name]):
  96         # Skip init variable from ckpt if shape dismatch.
  97         tf.logging.info(
  98             'Skip init [%s] from [%s] in ckpt because shape dismatch: %s vs %s',
  99             var_tensor.name, var_name,
 100             var_tensor.get_shape(), variable_shape_map[var_name])
 101         continue
 102 
 103       tf.logging.info('Init %s from %s in ckpt' % (var_tensor, var_name))
 104       final_variables_to_restore[var_name] = var_tensor
 105 
 106     self._init_fn = tf.contrib.framework.assign_from_checkpoint_fn(
 107         self._pretrained_checkpoint_dir,
 108         final_variables_to_restore)
 109 
 110   def after_create_session(self, session, coord):
 111     tf.logging.info('Restoring InceptionV3 weights.')
 112     self._init_fn(session)
 113     tf.logging.info('Done restoring InceptionV3 weights.')

 130   @abstractmethod
 131   def construct_input_fn(self, records, is_training):
 132     """Builds an estimator input_fn.
 133 
 134     The input_fn is used to pass feature and target data to the train,
 135     evaluate, and predict methods of the Estimator.
 136 
 137     Method to be overridden by implementations.
 138 
 139     Args:
 140       records: A list of Strings, paths to TFRecords with image data.
 141       is_training: Boolean, whether or not we're training.
 142 
 143     Returns:
 144       Function, that has signature of ()->(dict of features, target).
 145         features is a dict mapping feature names to `Tensors`
 146         containing the corresponding feature data (typically, just a single
 147         key/value pair 'raw_data' -> image `Tensor` for TCN.
 148         labels is a 1-D int32 `Tensor` holding labels.
 149     """
 150     pass

 152   def preprocess_data(self, images, is_training):
 153     """Preprocesses raw images for either training or inference.
 154 
 155     Args:
 156       images: A 4-D float32 `Tensor` holding images to preprocess.
 157       is_training: Boolean, whether or not we're in training.
 158 
 159     Returns:
 160       data_preprocessed: data after the preprocessor.
 161     """
 162     config = self._config
 163     height = config.data.height
 164     width = config.data.width
 165     min_scale = config.data.augmentation.minscale
 166     max_scale = config.data.augmentation.maxscale
 167     p_scale_up = config.data.augmentation.proportion_scaled_up
 168     aug_color = config.data.augmentation.color
 169     fast_mode = config.data.augmentation.fast_mode
 170     crop_strategy = config.data.preprocessing.eval_cropping
 171     preprocessed_images = preprocessing.preprocess_images(
 172         images, is_training, height, width,
 173         min_scale, max_scale, p_scale_up,
 174         aug_color=aug_color, fast_mode=fast_mode,
 175         crop_strategy=crop_strategy)
 176     return preprocessed_images

 221   def get_train_op(self, loss):
 222     """Creates a training op.
 223 
 224     Args:
 225       loss: A float32 `Tensor` representing the total training loss.
 226     Returns:
 227       train_op: A slim.learning.create_train_op train_op.
 228     Raises:
 229       ValueError: If specified optimizer isn't supported.
 230     """
 231     # Get variables to train (defined in subclass).
 232     assert self.variables_to_train
 233 
 234     # Define a learning rate schedule.
 235     decay_steps = self._config.learning.decay_steps
 236     decay_factor = self._config.learning.decay_factor
 237     learning_rate = float(self._config.learning.learning_rate)
 238 
 239     # Define a learning rate schedule.
 240     global_step = slim.get_or_create_global_step()
 241     learning_rate = tf.train.exponential_decay(
 242         learning_rate,
 243         global_step,
 244         decay_steps,
 245         decay_factor,
 246         staircase=True)
 247 
 248     # Create an optimizer.
 249     opt_type = self._config.learning.optimizer
 250     if opt_type == 'adam':
 251       opt = tf.train.AdamOptimizer(learning_rate)
 252     elif opt_type == 'momentum':
 253       opt = tf.train.MomentumOptimizer(learning_rate, 0.9)
 254     elif opt_type == 'rmsprop':
 255       opt = tf.train.RMSPropOptimizer(learning_rate, momentum=0.9,
 256                                       epsilon=1.0, decay=0.9)
 257     else:
 258       raise ValueError('Unsupported optimizer %s' % opt_type)
 259 
 260     if self._config.use_tpu:
 261       opt = tpu_optimizer.CrossShardOptimizer(opt)
 262 
 263     # Create a training op.
 264     # train_op = opt.minimize(loss, var_list=self.variables_to_train)
 265     # Create a training op.
 266     train_op = slim.learning.create_train_op(
 267         loss,
 268         optimizer=opt,
 269         variables_to_train=self.variables_to_train,
 270         update_ops=tf.get_collection(tf.GraphKeys.UPDATE_OPS))
 271 
 272     return train_op

 274   def _get_model_fn(self):
 275     """Defines behavior for training, evaluation, and inference (prediction).
 276 
 277     Returns:
 278       `model_fn` for `Estimator`.
 279     """
 280     # pylint: disable=unused-argument
 281     def model_fn(features, labels, mode, params):
 282       """Build the model based on features, labels, and mode.
 283 
 284       Args:
 285         features: Dict, strings to `Tensor` input data, returned by the
 286           input_fn.
 287         labels: The labels Tensor returned by the input_fn.
 288         mode: A string indicating the mode. This will be either
 289           tf.estimator.ModeKeys.TRAIN, tf.estimator.ModeKeys.PREDICT,
 290           or tf.estimator.ModeKeys.EVAL.
 291         params: A dict holding training parameters, passed in during TPU
 292           training.
 293 
 294       Returns:
 295         A tf.estimator.EstimatorSpec specifying train/test/inference behavior.
 296       """
 297       is_training = mode == tf.estimator.ModeKeys.TRAIN
 298 
 299       # Get preprocessed images from the features dict.
 300       batch_preprocessed = features['batch_preprocessed']
 301 
 302       # Do a forward pass to embed data.
 303       batch_encoded = self.forward(batch_preprocessed, is_training)
 304 
 305       # Optionally set the pretrained initialization function.
 306       initializer_fn = None
 307       if mode == tf.estimator.ModeKeys.TRAIN:
 308         initializer_fn = self.pretrained_init_fn
 309 
 310       # If we're training or evaluating, define total loss.
 311       total_loss = None
 312       if mode in (tf.estimator.ModeKeys.TRAIN, tf.estimator.ModeKeys.EVAL):
 313         loss = self.define_loss(batch_encoded, labels, is_training)
 314         tf.losses.add_loss(loss)
 315         total_loss = tf.losses.get_total_loss()
 316 
 317       # If we're training, define a train op.
 318       train_op = None
 319       if mode == tf.estimator.ModeKeys.TRAIN:
 320         train_op = self.get_train_op(total_loss)
 321 
 322       # If we're doing inference, set the output to be the embedded images.
 323       predictions_dict = None
 324       if mode == tf.estimator.ModeKeys.PREDICT:
 325         predictions_dict = {'embeddings': batch_encoded}
 326         # Pass through additional metadata stored in features.
 327         for k, v in features.iteritems():
 328           predictions_dict[k] = v
 329 
 330       # If we're evaluating, define some eval metrics.
 331       eval_metric_ops = None
 332       if mode == tf.estimator.ModeKeys.EVAL:
 333         eval_metric_ops = self.define_eval_metric_ops()
 334 
 335       # Define training scaffold to load pretrained weights.
 336       num_checkpoint_to_keep = self._config.logging.checkpoint.num_to_keep
 337       saver = tf.train.Saver(
 338           max_to_keep=num_checkpoint_to_keep)
 339 
 340       if is_training and self._config.use_tpu:
 341         # TPU doesn't have a scaffold option at the moment, so initialize
 342         # pretrained weights using a custom train_hook instead.
 343         return tpu_estimator.TPUEstimatorSpec(
 344             mode,
 345             loss=total_loss,
 346             eval_metrics=None,
 347             train_op=train_op,
 348             predictions=predictions_dict)
 349       else:
 350         # Build a scaffold to initialize pretrained weights.
 351         scaffold = tf.train.Scaffold(
 352             init_fn=initializer_fn,
 353             saver=saver,
 354             summary_op=None)
 355         return tf.estimator.EstimatorSpec(
 356             mode=mode,
 357             predictions=predictions_dict,
 358             loss=total_loss,
 359             train_op=train_op,
 360             eval_metric_ops=eval_metric_ops,
 361             scaffold=scaffold)
 362     return model_fn

 364   def train(self):
 365     """Runs training."""
 366     # Get a list of training tfrecords.
 367     config = self._config
 368     training_dir = config.data.training
 369     training_records = util.GetFilesRecursively(training_dir)
 370 
 371     # Define batch size.
 372     self._batch_size = config.data.batch_size
 373 
 374     # Create a subclass-defined training input function.
 375     train_input_fn = self.construct_input_fn(
 376         training_records, is_training=True)
 377 
 378     # Create the estimator.
 379     estimator = self._build_estimator(is_training=True)
 380 
 381     train_hooks = None
 382     if config.use_tpu:
 383       # TPU training initializes pretrained weights using a custom train hook.
 384       train_hooks = []
 385       if tf.train.latest_checkpoint(self._logdir) is None:
 386         train_hooks.append(
 387             InitFromPretrainedCheckpointHook(
 388                 config[config.embedder_strategy].pretrained_checkpoint))
 389 
 390     # Run training.
 391     estimator.train(input_fn=train_input_fn, hooks=train_hooks,
 392                     steps=config.learning.max_step)

 394   def _build_estimator(self, is_training):
 395     """Returns an Estimator object.
 396 
 397     Args:
 398       is_training: Boolean, whether or not we're in training mode.
 399 
 400     Returns:
 401       A tf.estimator.Estimator.
 402     """
 403     config = self._config
 404     save_checkpoints_steps = config.logging.checkpoint.save_checkpoints_steps
 405     keep_checkpoint_max = self._config.logging.checkpoint.num_to_keep
 406     if is_training and config.use_tpu:
 407       iterations = config.tpu.iterations
 408       num_shards = config.tpu.num_shards
 409       run_config = tpu_config.RunConfig(
 410           save_checkpoints_secs=None,
 411           save_checkpoints_steps=save_checkpoints_steps,
 412           keep_checkpoint_max=keep_checkpoint_max,
 413           master=FLAGS.master,
 414           evaluation_master=FLAGS.master,
 415           model_dir=self._logdir,
 416           tpu_config=tpu_config.TPUConfig(
 417               iterations_per_loop=iterations,
 418               num_shards=num_shards,
 419               per_host_input_for_training=num_shards <= 8),
 420           tf_random_seed=FLAGS.tf_random_seed)
 421 
 422       batch_size = config.data.batch_size
 423       return tpu_estimator.TPUEstimator(
 424           model_fn=self._get_model_fn(),
 425           config=run_config,
 426           use_tpu=True,
 427           train_batch_size=batch_size,
 428           eval_batch_size=batch_size)
 429     else:
 430       run_config = tf.estimator.RunConfig().replace(
 431           model_dir=self._logdir,
 432           save_checkpoints_steps=save_checkpoints_steps,
 433           keep_checkpoint_max=keep_checkpoint_max,
 434           tf_random_seed=FLAGS.tf_random_seed)
 435       return tf.estimator.Estimator(
 436           model_fn=self._get_model_fn(),
 437           config=run_config)

 439   def evaluate(self):
 440     """Runs `Estimator` validation.
 441     """
 442     config = self._config
 443 
 444     # Get a list of validation tfrecords.
 445     validation_dir = config.data.validation
 446     validation_records = util.GetFilesRecursively(validation_dir)
 447 
 448     # Define batch size.
 449     self._batch_size = config.data.batch_size
 450 
 451     # Create a subclass-defined training input function.
 452     validation_input_fn = self.construct_input_fn(
 453         validation_records, False)
 454 
 455     # Create the estimator.
 456     estimator = self._build_estimator(is_training=False)
 457 
 458     # Run validation.
 459     eval_batch_size = config.data.batch_size
 460     num_eval_samples = config.val.num_eval_samples

 462     estimator.evaluate(input_fn=validation_input_fn, steps=num_eval_batches)

 464   def inference(
 465       self, inference_input, checkpoint_path, batch_size=None, **kwargs):
 466     """Defines 3 of modes of inference.
 467 
 468     Inputs:
 469     * Mode 1: Input is an input_fn.
 470     * Mode 2: Input is a TFRecord (or list of TFRecords).
 471     * Mode 3: Input is a numpy array holding an image (or array of images).
 472 
 473     Outputs:
 474     * Mode 1: this returns an iterator over embeddings and additional
 475       metadata. See
 476       https://www.tensorflow.org/api_docs/python/tf/estimator/Estimator#predict
 477       for details.
 478     * Mode 2: Returns an iterator over tuples of
 479       (embeddings, raw_image_strings, sequence_name), where embeddings is a
 480       2-D float32 numpy array holding [sequence_size, embedding_size] image
 481       embeddings, raw_image_strings is a 1-D string numpy array holding
 482       [sequence_size] jpeg-encoded image strings, and sequence_name is a
 483       string holding the name of the embedded sequence.
 484     * Mode 3: Returns a tuple of (embeddings, raw_image_strings), where
 485       embeddings is a 2-D float32 numpy array holding
 486       [batch_size, embedding_size] image embeddings, raw_image_strings is a
 487       1-D string numpy array holding [batch_size] jpeg-encoded image strings.
 488 
 489     Args:
 490       inference_input: This can be a tf.Estimator input_fn, a TFRecord path,
 491         a list of TFRecord paths, a numpy image, or an array of numpy images.
 492       checkpoint_path: String, path to the checkpoint to restore for inference.
 493       batch_size: Int, the size of the batch to use for inference.
 494       **kwargs: Additional keyword arguments, depending on the mode.
 495         See _input_fn_inference, _tfrecord_inference, and _np_inference.
 496     Returns:
 497       inference_output: Inference output depending on mode, see above for
 498         details.
 499     Raises:
 500       ValueError: If inference_input isn't a tf.Estimator input_fn,
 501         a TFRecord path, a list of TFRecord paths, or a numpy array,
 502     """
 503     # Mode 1: input is a callable tf.Estimator input_fn.
 504     if callable(inference_input):
 505       return self._input_fn_inference(
 506           input_fn=inference_input, checkpoint_path=checkpoint_path, **kwargs)
 507     # Mode 2: Input is a TFRecord path (or list of TFRecord paths).
 508     elif util.is_tfrecord_input(inference_input):
 509       return self._tfrecord_inference(
 510           records=inference_input, checkpoint_path=checkpoint_path,
 511           batch_size=batch_size, **kwargs)
 512     # Mode 3: Input is a numpy array of raw images.
 513     elif util.is_np_array(inference_input):
 514       return self._np_inference(
 515           np_images=inference_input, checkpoint_path=checkpoint_path, **kwargs)
 516     else:
 517       raise ValueError(
 518           'inference input must be a tf.Estimator input_fn, a TFRecord path,'
 519           'a list of TFRecord paths, or a numpy array. Got: %s' % str(type(
 520               inference_input)))
 521 
 522   def _input_fn_inference(self, input_fn, checkpoint_path, predict_keys=None):
 523     """Mode 1: tf.Estimator inference.
 524 
 525     Args:
 526       input_fn: Function, that has signature of ()->(dict of features, None).
 527         This is a function called by the estimator to get input tensors (stored
 528         in the features dict) to do inference over.
 529       checkpoint_path: String, path to a specific checkpoint to restore.
 530       predict_keys: List of strings, the keys of the `Tensors` in the features
 531         dict (returned by the input_fn) to evaluate during inference.
 532     Returns:
 533       predictions: An Iterator, yielding evaluated values of `Tensors`
 534         specified in `predict_keys`.
 535     """
 536     # Create the estimator.
 537     estimator = self._build_estimator(is_training=False)
 538 
 539     # Create an iterator of predicted embeddings.
 540     predictions = estimator.predict(input_fn=input_fn,
 541                                     checkpoint_path=checkpoint_path,
 542                                     predict_keys=predict_keys)
 543     return predictions
 544 
 545   def _tfrecord_inference(self, records, checkpoint_path, batch_size,
 546                           num_sequences=-1, reuse=False):
 547     """Mode 2: TFRecord inference.
 548 
 549     Args:
 550       records: List of strings, paths to TFRecords.
 551       checkpoint_path: String, path to a specific checkpoint to restore.
 552       batch_size: Int, size of inference batch.
 553       num_sequences: Int, number of sequences to embed. If -1,
 554         embed everything.
 555       reuse: Boolean, whether or not to reuse embedder weights.
 556     Yields:
 557       (embeddings, raw_image_strings, sequence_name):
 558         embeddings is a 2-D float32 numpy array holding
 559         [sequence_size, embedding_size] image embeddings.
 560         raw_image_strings is a 1-D string numpy array holding
 561         [sequence_size] jpeg-encoded image strings.
 562         sequence_name is a string holding the name of the embedded sequence.
 563     """
 564     tf.reset_default_graph()
 565     if not isinstance(records, list):
 566       records = list(records)
 567 
 568     # Map the list of tfrecords to a dataset of preprocessed images.
 569     num_views = self._config.data.num_views
 570     (views, task, seq_len) = data_providers.full_sequence_provider(
 571         records, num_views)
 572     tensor_dict = {
 573         'raw_image_strings': views,
 574         'task': task,
 575         'seq_len': seq_len
 576     }
 577 
 578     # Create a preprocess function over raw image string placeholders.
 579     image_str_placeholder = tf.placeholder(tf.string, shape=[None])
 580     decoded = preprocessing.decode_images(image_str_placeholder)
 581     decoded.set_shape([batch_size, None, None, 3])
 582     preprocessed = self.preprocess_data(decoded, is_training=False)
 583 
 584     # Create an inference graph over preprocessed images.
 585     embeddings = self.forward(preprocessed, is_training=False, reuse=reuse)
 586 
 587     # Create a saver to restore model variables.
 588     tf.train.get_or_create_global_step()
 589     saver = tf.train.Saver(tf.all_variables())
 590 
 591     # Create a session and restore model variables.
 592     with tf.train.MonitoredSession() as sess:
 593       saver.restore(sess, checkpoint_path)
 594       cnt = 0
 595       # If num_sequences is specified, embed that many sequences, else embed
 596       # everything.
 597       try:
 598         while cnt < num_sequences if num_sequences != -1 else True:
 599           # Get a preprocessed image sequence.
 600           np_data = sess.run(tensor_dict)
 601           np_raw_images = np_data['raw_image_strings']
 602           np_seq_len = np_data['seq_len']
 603           np_task = np_data['task']
 604 
 605           # Embed each view.
 606           embedding_size = self._config.embedding_size
 607           view_embeddings = [
 608               np.zeros((0, embedding_size)) for _ in range(num_views)]
 609           for view_index in range(num_views):
 610             view_raw = np_raw_images[view_index]
 611             # Embed the full sequence.
 612             t = 0
 613             while t < np_seq_len:
 614               # Decode and preprocess the batch of image strings.
 615               embeddings_np = sess.run(
 616                   embeddings, feed_dict={
 617                       image_str_placeholder: view_raw[t:t+batch_size]})
 618               view_embeddings[view_index] = np.append(
 619                   view_embeddings[view_index], embeddings_np, axis=0)
 620               tf.logging.info('Embedded %d images for task %s' % (t, np_task))
 621               t += batch_size
 622 
 623           # Done embedding for all views.
 624           view_raw_images = np_data['raw_image_strings']
 625           yield (view_embeddings, view_raw_images, np_task)
 626           cnt += 1
 627       except tf.errors.OutOfRangeError:
 628         tf.logging.info('Done embedding entire dataset.')
 629 
 630   def _np_inference(self, np_images, checkpoint_path):
 631     """Mode 3: Call this repeatedly to do inference over numpy images.
 632 
 633     This mode is for when we we want to do real-time inference over
 634     some stream of images (represented as numpy arrays).
 635 
 636     Args:
 637       np_images: A float32 numpy array holding images to embed.
 638       checkpoint_path: String, path to a specific checkpoint to restore.
 639     Returns:
 640       (embeddings, raw_image_strings):
 641         embeddings is a 2-D float32 numpy array holding
 642         [inferred batch_size, embedding_size] image embeddings.
 643         raw_image_strings is a 1-D string numpy array holding
 644         [inferred batch_size] jpeg-encoded image strings.
 645     """
 646     if isinstance(np_images, list):
 647       np_images = np.asarray(np_images)
 648     # Add a batch dimension if only 3-dimensional.
 649     if len(np_images.shape) == 3:
 650       np_images = np.expand_dims(np_images, axis=0)
 651 
 652     # If np_images are in the range [0,255], convert to [0,1].
 653     assert np.min(np_images) >= 0.
 654     if (np.min(np_images), np.max(np_images)) == (0, 255):
 655       np_images = np_images.astype(np.float32) / 255.
 656       assert (np.min(np_images), np.max(np_images)) == (0., 1.)
 657 
 658     # If this is the first pass, set up inference graph.
 659     if not hasattr(self, '_np_inf_tensor_dict'):
 660       self._setup_np_inference(np_images, checkpoint_path)
 661 
 662     # Convert np_images to embeddings.
 663     np_tensor_dict = self._sess.run(self._np_inf_tensor_dict, feed_dict={
 664         self._image_placeholder: np_images
 665     })
 666     return np_tensor_dict['embeddings'], np_tensor_dict['raw_image_strings']
 667 
 668   def _setup_np_inference(self, np_images, checkpoint_path):
 669     """Sets up and restores inference graph, creates and caches a Session."""
 670     tf.logging.info('Restoring model weights.')
 671 
 672     # Define inference over an image placeholder.
 673     _, height, width, _ = np.shape(np_images)
 674     image_placeholder = tf.placeholder(
 675         tf.float32, shape=(None, height, width, 3))
 676 
 677     # Preprocess batch.
 678     preprocessed = self.preprocess_data(image_placeholder, is_training=False)
 679 
 680     # Unscale and jpeg encode preprocessed images for display purposes.
 681     im_strings = preprocessing.unscale_jpeg_encode(preprocessed)
 682 
 683     # Do forward pass to get embeddings.
 684     embeddings = self.forward(preprocessed, is_training=False)
 685 
 686     # Create a saver to restore model variables.
 687     tf.train.get_or_create_global_step()
 688     saver = tf.train.Saver(tf.all_variables())
 689 
 690     self._image_placeholder = image_placeholder
 691     self._batch_encoded = embeddings
 692 
 693     self._np_inf_tensor_dict = {
 694         'embeddings': embeddings,
 695         'raw_image_strings': im_strings,
 696     }
 697 
 698     # Create a session and restore model variables.
 699     self._sess = tf.Session()
 700     saver.restore(self._sess, checkpoint_path)
