  26 def get_mvtcn_estimator(loss_strategy, config, logdir):
  27   """Returns a configured MVTCN estimator."""
  28   loss_to_trainer = {
  29       'triplet_semihard': mvtcn_estimators.MVTCNTripletEstimator,
  30       'npairs': mvtcn_estimators.MVTCNNpairsEstimator,
  31   }
  32   if loss_strategy not in loss_to_trainer:
  33     raise ValueError('Unknown loss for MVTCN: %s' % loss_strategy)
  34   estimator = loss_to_trainer[loss_strategy](config, logdir)
  35   return estimator
  36

  38 def get_estimator(config, logdir):
  39   """Returns an unsupervised model trainer based on config.
  40 
  41   Args:
  42     config: A T object holding training configs.
  43     logdir: String, path to directory where model checkpoints and summaries
  44       are saved.
  45   Returns:
  46     estimator: A configured `TCNEstimator` object.
  47   Raises:
  48     ValueError: If unknown training strategy is specified.
  49   """
  50   # Get the training strategy.
  51   training_strategy = config.training_strategy
  52   if training_strategy == 'mvtcn':
  53     loss_strategy = config.loss_strategy
  54     estimator = get_mvtcn_estimator(
  55         loss_strategy, config, logdir)
  56   elif training_strategy == 'svtcn':
  57     estimator = svtcn_estimator.SVTCNTripletEstimator(config, logdir)
  58   else:
  59     raise ValueError('Unknown training strategy: %s' % training_strategy)
  60   return estimator
