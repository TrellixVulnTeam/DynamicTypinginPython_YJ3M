  29 class MVTCNEstimator(base_estimator.BaseEstimator):
  30   """Multi-view TCN base class."""
  31 
  32   def __init__(self, config, logdir):
  33     super(MVTCNEstimator, self).__init__(config, logdir)
  34 
  35   def _pairs_provider(self, records, is_training):
  36     config = self._config
  37     num_views = config.data.num_views
  38     window = config.mvtcn.window
  39     num_parallel_calls = config.data.num_parallel_calls
  40     sequence_prefetch_size = config.data.sequence_prefetch_size
  41     batch_prefetch_size = config.data.batch_prefetch_size
  42     examples_per_seq = config.data.examples_per_sequence
  43     return functools.partial(
  44         data_providers.multiview_pairs_provider,
  45         file_list=records,
  46         preprocess_fn=self.preprocess_data,
  47         num_views=num_views,
  48         window=window,
  49         is_training=is_training,
  50         examples_per_seq=examples_per_seq,
  51         num_parallel_calls=num_parallel_calls,
  52         sequence_prefetch_size=sequence_prefetch_size,
  53         batch_prefetch_size=batch_prefetch_size)
  54 
  55   def forward(self, images_concat, is_training, reuse=False):
  56     """See base class."""
  57     embedder_strategy = self._config.embedder_strategy
  58     loss_strategy = self._config.loss_strategy
  59     l2_normalize_embedding = self._config[loss_strategy].embedding_l2
  60     embedder = model_module.get_embedder(
  61         embedder_strategy,
  62         self._config,
  63         images_concat,
  64         is_training=is_training,
  65         l2_normalize_embedding=l2_normalize_embedding, reuse=reuse)
  66     embeddings_concat = embedder.construct_embedding()
  67     variables_to_train = embedder.get_trainable_variables()
  68     self.variables_to_train = variables_to_train
  69     self.pretrained_init_fn = embedder.init_fn
  70     return embeddings_concat
  71 
  72   def _collect_image_summaries(self, anchor_images, positive_images,
  73                                images_concat):
  74     image_summaries = self._config.logging.summary.image_summaries
  75     if image_summaries and not self._config.use_tpu:
  76       batch_pairs_summary = tf.concat(
  77           [anchor_images, positive_images], axis=2)
  78       tf.summary.image('training/mvtcn_pairs', batch_pairs_summary)
  79       tf.summary.image('training/images_preprocessed_concat', images_concat)

  82 class MVTCNTripletEstimator(MVTCNEstimator):
  83   """Multi-View TCN with semihard triplet loss."""
  84 
  85   def __init__(self, config, logdir):
  86     super(MVTCNTripletEstimator, self).__init__(config, logdir)
  87 
  88   def construct_input_fn(self, records, is_training):
  89     """See base class."""
  90     def input_fn(params):
  91       """Provides input to MVTCN models."""
  92       if is_training and self._config.use_tpu:
  93         batch_size = params['batch_size']
  94       else:
  95         batch_size = self._batch_size
  96       (images_concat,
  97        anchor_labels,
  98        positive_labels,
  99        anchor_images,
 100        positive_images) = self._pairs_provider(
 101            records, is_training)(batch_size=batch_size)
 102       if is_training:
 103         self._collect_image_summaries(anchor_images, positive_images,
 104                                       images_concat)
 105       labels = tf.concat([anchor_labels, positive_labels], axis=0)
 106       features = {'batch_preprocessed': images_concat}
 107       return (features, labels)
 108     return input_fn
 109 
 110   def define_loss(self, embeddings, labels, is_training):
 111     """See base class."""
 112     margin = self._config.triplet_semihard.margin
 113     loss = tf.contrib.losses.metric_learning.triplet_semihard_loss(
 114         labels=labels, embeddings=embeddings, margin=margin)
 115     self._loss = loss
 116     if is_training and not self._config.use_tpu:
 117       tf.summary.scalar('training/triplet_semihard', loss)
 118     return loss
 119 
 120   def define_eval_metric_ops(self):
 121     """See base class."""
 122     return {'validation/triplet_semihard': tf.metrics.mean(self._loss)}

 125 class MVTCNNpairsEstimator(MVTCNEstimator):
 126   """Multi-View TCN with npairs loss."""
 127 
 128   def __init__(self, config, logdir):
 129     super(MVTCNNpairsEstimator, self).__init__(config, logdir)
 130 
 131   def construct_input_fn(self, records, is_training):
 132     """See base class."""
 133     def input_fn(params):
 134       """Provides input to MVTCN models."""
 135       if is_training and self._config.use_tpu:
 136         batch_size = params['batch_size']
 137       else:
 138         batch_size = self._batch_size
 139       (images_concat,
 140        npairs_labels,
 141        _,
 142        anchor_images,
 143        positive_images) = self._pairs_provider(
 144            records, is_training)(batch_size=batch_size)
 145       if is_training:
 146         self._collect_image_summaries(anchor_images, positive_images,
 147                                       images_concat)
 148       features = {'batch_preprocessed': images_concat}
 149       return (features, npairs_labels)
 150     return input_fn
 151 
 152   def define_loss(self, embeddings, labels, is_training):
 153     """See base class."""
 154     embeddings_anchor, embeddings_positive = tf.split(embeddings, 2, axis=0)
 155     loss = tf.contrib.losses.metric_learning.npairs_loss(
 156         labels=labels, embeddings_anchor=embeddings_anchor,
 157         embeddings_positive=embeddings_positive)
 158     self._loss = loss
 159     if is_training and not self._config.use_tpu:
 160       tf.summary.scalar('training/npairs', loss)
 161     return loss
 162 
 163   def define_eval_metric_ops(self):
 164     """See base class."""
 165     return {'validation/npairs': tf.metrics.mean(self._loss)}
