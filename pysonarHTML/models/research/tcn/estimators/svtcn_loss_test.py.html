  16 """Tests for svtcn_loss.py."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import numpy as np
  22 from sklearn.metrics.pairwise import euclidean_distances
  23 from estimators import svtcn_loss
  24 import tensorflow as tf
  25 
  26 
  27 class SVTCNLoss(tf.test.TestCase):
  28 
  29   def testSVTCNLoss(self):
  30     with self.test_session():
  31       num_data = 64
  32       num_sequences = 2
  33       num_data_per_seq = num_data // num_sequences
  34       feat_dim = 6
  35       margin = 1.0
  36       times = np.tile(np.arange(num_data_per_seq, dtype=np.int32),
  37                       num_sequences)
  38       times = np.reshape(times, [times.shape[0], 1])
  39       sequence_ids = np.concatenate(
  40           [np.ones(num_data_per_seq)*i for i in range(num_sequences)])
  41       sequence_ids = np.reshape(sequence_ids, [sequence_ids.shape[0], 1])
  42 
  43       pos_radius = 6
  44       neg_radius = 12
  45 
  46       embedding = np.random.rand(num_data, feat_dim).astype(np.float32)
  47 
  48       # Compute the loss in NP
  49 
  50       # Get a positive mask, i.e. indices for each time index
  51       # that are inside the positive range.
  52       in_pos_range = np.less_equal(
  53           np.abs(times - times.transpose()), pos_radius)
  54 
  55       # Get a negative mask, i.e. indices for each time index
  56       # that are inside the negative range (> t + (neg_mult * pos_radius)
  57       # and < t - (neg_mult * pos_radius).
  58       in_neg_range = np.greater(np.abs(times - times.transpose()), neg_radius)
  59 
  60       sequence_adjacency = sequence_ids == sequence_ids.T
  61       sequence_adjacency_not = np.logical_not(sequence_adjacency)
  62 
  63       pdist_matrix = euclidean_distances(embedding, squared=True)
  64       loss_np = 0.0
  65       num_positives = 0.0
  66       for i in range(num_data):
  67         for j in range(num_data):
  68           if in_pos_range[i, j] and i != j and sequence_adjacency[i, j]:
  69             num_positives += 1.0
  70 
  71             pos_distance = pdist_matrix[i][j]
  72             neg_distances = []
  73 
  74             for k in range(num_data):
  75               if in_neg_range[i, k] or sequence_adjacency_not[i, k]:
  76                 neg_distances.append(pdist_matrix[i][k])
  77 
  78             neg_distances.sort()  # sort by distance
  79             chosen_neg_distance = neg_distances[0]
  80 
  81             for l in range(len(neg_distances)):
  82               chosen_neg_distance = neg_distances[l]
  83               if chosen_neg_distance > pos_distance:
  84                 break
  85 
  86             loss_np += np.maximum(
  87                 0.0, margin - chosen_neg_distance + pos_distance)
  88 
  89       loss_np /= num_positives
  90 
  91       # Compute the loss in TF
  92       loss_tf = svtcn_loss.singleview_tcn_loss(
  93           embeddings=tf.convert_to_tensor(embedding),
  94           timesteps=tf.convert_to_tensor(times),
  95           pos_radius=pos_radius,
  96           neg_radius=neg_radius,
  97           margin=margin,
  98           sequence_ids=tf.convert_to_tensor(sequence_ids),
  99           multiseq=True
 100       )
 101       loss_tf = loss_tf.eval()
 102       self.assertAllClose(loss_np, loss_tf)
 103
