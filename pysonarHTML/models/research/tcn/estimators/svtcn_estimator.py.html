  29 class SVTCNEstimator(base_estimator.BaseEstimator):
  30   """Single-view TCN Estimator base class."""
  31 
  32   def __init__(self, config, logdir):
  33     super(SVTCNEstimator, self).__init__(config, logdir)
  34 
  35   def construct_input_fn(self, records, is_training):
  36     """See base class."""
  37     config = self._config
  38     num_views = config.data.num_views
  39     num_parallel_calls = config.data.num_parallel_calls
  40     sequence_prefetch_size = config.data.sequence_prefetch_size
  41     batch_prefetch_size = config.data.batch_prefetch_size
  42 
  43     def input_fn():
  44       """Provides input to SVTCN models."""
  45       (images_preprocessed,
  46        images_raw,
  47        timesteps) = data_providers.singleview_tcn_provider(
  48            file_list=records,
  49            preprocess_fn=self.preprocess_data,
  50            num_views=num_views,
  51            is_training=is_training,
  52            batch_size=self._batch_size,
  53            num_parallel_calls=num_parallel_calls,
  54            sequence_prefetch_size=sequence_prefetch_size,
  55            batch_prefetch_size=batch_prefetch_size)
  56 
  57       if config.logging.summary.image_summaries and is_training:
  58         tf.summary.image('training/svtcn_images', images_raw)
  59 
  60       features = {'batch_preprocessed': images_preprocessed}
  61       return (features, timesteps)
  62     return input_fn
  63 
  64   def forward(self, images, is_training, reuse=False):
  65     """See base class."""
  66     embedder_strategy = self._config.embedder_strategy
  67     embedder = model_module.get_embedder(
  68         embedder_strategy,
  69         self._config,
  70         images,
  71         is_training=is_training, reuse=reuse)
  72     embeddings = embedder.construct_embedding()
  73 
  74     if is_training:
  75       self.variables_to_train = embedder.get_trainable_variables()
  76       self.pretrained_init_fn = embedder.init_fn
  77     return embeddings
  78

  80 class SVTCNTripletEstimator(SVTCNEstimator):
  81   """Single-View TCN with semihard triplet loss."""
  82 
  83   def __init__(self, config, logdir):
  84     super(SVTCNTripletEstimator, self).__init__(config, logdir)
  85 
  86   def define_loss(self, embeddings, timesteps, is_training):
  87     """See base class."""
  88     pos_radius = self._config.svtcn.pos_radius
  89     neg_radius = self._config.svtcn.neg_radius
  90     margin = self._config.triplet_semihard.margin
  91     loss = svtcn_loss.singleview_tcn_loss(
  92         embeddings, timesteps, pos_radius, neg_radius, margin=margin)
  93     self._loss = loss
  94     if is_training:
  95       tf.summary.scalar('training/svtcn_loss', loss)
  96     return loss
  97 
  98   def define_eval_metric_ops(self):
  99     """See base class."""
 100     return {'validation/svtcn_loss': tf.metrics.mean(self._loss)}
