 104 def SmoothEmbeddings(embs):
 105   """Temporally smoothes a sequence of embeddings."""
 106   new_embs = []
 107   window = int(FLAGS.smoothing_window)
 108   for i in range(len(embs)):
 109     min_i = max(i-window, 0)
 110     max_i = min(i+window, len(embs))
 111     new_embs.append(np.mean(embs[min_i:max_i, :], axis=0))
 112   return np.array(new_embs)

 115 def MakeImitationVideo(
 116     outdir, vidname, query_im_strs, knn_im_strs, height=640, width=360):
 117   """Creates a KNN imitation video.
 118 
 119   For each frame in vid0, pair with the frame at index in knn_indices in
 120   vids1. Write video to disk.
 121 
 122   Args:
 123     outdir: String, directory to write videos.
 124     vidname: String, name of video.
 125     query_im_strs: Numpy array holding query image strings.
 126     knn_im_strs: Numpy array holding knn image strings.
 127     height: Int, height of raw images.
 128     width: Int, width of raw images.
 129   """
 130   if not tf.gfile.Exists(outdir):
 131     tf.gfile.MakeDirs(outdir)
 132   vid_path = os.path.join(outdir, vidname)
 133   combined = zip(query_im_strs, knn_im_strs)
 134 
 135   # Create and write the video.
 136   fig = plt.figure()
 137   ax = fig.add_subplot(111)
 138   ax.set_aspect('equal')
 139   ax.get_xaxis().set_visible(False)
 140   ax.get_yaxis().set_visible(False)
 141   im = ax.imshow(
 142       np.zeros((height, width*2, 3)), cmap='gray', interpolation='nearest')
 143   im.set_clim([0, 1])
 144   plt.tight_layout(pad=0, w_pad=0, h_pad=0)
 145   # pylint: disable=invalid-name
 146   def update_img(pair):
 147     """Decode pairs of image strings, update a video."""
 148     im_i, im_j = pair
 149     nparr_i = np.fromstring(str(im_i), np.uint8)
 150     img_np_i = cv2.imdecode(nparr_i, 1)
 151     img_np_i = img_np_i[..., [2, 1, 0]]
 152     nparr_j = np.fromstring(str(im_j), np.uint8)
 153     img_np_j = cv2.imdecode(nparr_j, 1)
 154     img_np_j = img_np_j[..., [2, 1, 0]]
 155 
 156     # Optionally reshape the images to be same size.
 157     frame = np.concatenate([img_np_i, img_np_j], axis=1)
 158     im.set_data(frame)
 159     return im
 160   ani = animation.FuncAnimation(fig, update_img, combined, interval=15)
 161   writer = animation.writers['ffmpeg'](fps=15)
 162   dpi = 100
 163   tf.logging.info('Writing video to:\n %s \n' % vid_path)
 164   ani.save('%s.mp4' % vid_path, writer=writer, dpi=dpi)

 167 def GenerateImitationVideo(
 168     vid_name, query_ims, query_embs, target_ims, target_embs, height, width):
 169   """Generates a single cross-sequence imitation video.
 170 
 171   For each frame in some query sequence, find the nearest neighbor from
 172   some target sequence in embedding space.
 173 
 174   Args:
 175     vid_name: String, the name of the video.
 176     query_ims: Numpy array of shape [query sequence length, height, width, 3].
 177     query_embs: Numpy array of shape [query sequence length, embedding size].
 178     target_ims: Numpy array of shape [target sequence length, height, width,
 179       3].
 180     target_embs: Numpy array of shape [target sequence length, embedding
 181       size].
 182     height: Int, height of the raw image.
 183     width: Int, width of the raw image.
 184   """
 185   # For each query frame, find the index of the nearest neighbor in the
 186   # target video.
 187   knn_indices = [util.KNNIds(q, target_embs, k=1)[0] for q in query_embs]
 188 
 189   # Create and write out the video.
 190   assert knn_indices
 191   knn_ims = np.array([target_ims[k] for k in knn_indices])
 192   MakeImitationVideo(FLAGS.outdir, vid_name, query_ims, knn_ims, height, width)

 195 def SingleImitationVideos(
 196     query_records, target_records, config, height, width):
 197   """Generates pairwise imitation videos.
 198 
 199   This creates all pairs of target imitating query videos, where each frame
 200   on the left is matched to a nearest neighbor coming a single
 201   embedded target video.
 202 
 203   Args:
 204     query_records: List of Strings, paths to tfrecord datasets to use as
 205       queries.
 206     target_records: List of Strings, paths to tfrecord datasets to use as
 207       targets.
 208     config: A T object describing training config.
 209     height: Int, height of the raw image.
 210     width: Int, width of the raw image.
 211   """
 212   # Embed query and target data.
 213   (query_sequences_to_data,
 214    target_sequences_to_data) = EmbedQueryTargetData(
 215        query_records, target_records, config)
 216 
 217   qview = FLAGS.query_view
 218   tview = FLAGS.target_view
 219 
 220   # Loop over query videos.
 221   for task_i, data_i in query_sequences_to_data.iteritems():
 222     for task_j, data_j in target_sequences_to_data.iteritems():
 223       i_ims = data_i['images']
 224       i_embs = data_i['embeddings']
 225       query_embs = SmoothEmbeddings(i_embs[qview])
 226       query_ims = i_ims[qview]
 227 
 228       j_ims = data_j['images']
 229       j_embs = data_j['embeddings']
 230       target_embs = SmoothEmbeddings(j_embs[tview])
 231       target_ims = j_ims[tview]
 232 
 233       tf.logging.info('Generating %s imitating %s video.' % (task_j, task_i))
 234       vid_name = 'q%sv%s_im%sv%s' % (task_i, qview, task_j, tview)
 235       vid_name = vid_name.replace('/', '_')
 236       GenerateImitationVideo(vid_name, query_ims, query_embs,
 237                              target_ims, target_embs, height, width)

 240 def MultiImitationVideos(
 241     query_records, target_records, config, height, width):
 242   """Creates multi-imitation videos.
 243 
 244   This creates videos where every frame on the left is matched to a nearest
 245   neighbor coming from a set of multiple embedded target videos.
 246 
 247   Args:
 248     query_records: List of Strings, paths to tfrecord datasets to use as
 249       queries.
 250     target_records: List of Strings, paths to tfrecord datasets to use as
 251       targets.
 252     config: A T object describing training config.
 253     height: Int, height of the raw image.
 254     width: Int, width of the raw image.
 255   """
 256   # Embed query and target data.
 257   (query_sequences_to_data,
 258    target_sequences_to_data) = EmbedQueryTargetData(
 259        query_records, target_records, config)
 260 
 261   qview = FLAGS.query_view
 262   tview = FLAGS.target_view
 263 
 264   # Loop over query videos.
 265   for task_i, data_i in query_sequences_to_data.iteritems():
 266     i_ims = data_i['images']
 267     i_embs = data_i['embeddings']
 268     query_embs = SmoothEmbeddings(i_embs[qview])
 269     query_ims = i_ims[qview]
 270 
 271     all_target_embs = []
 272     all_target_ims = []
 273 
 274     # If num_imitation_vids is -1, add all seq embeddings to the target set.
 275     if FLAGS.num_multi_targets == -1:
 276       num_multi_targets = len(target_sequences_to_data)
 277     else:
 278       # Else, add some specified number of seq embeddings to the target set.
 279       num_multi_targets = FLAGS.num_multi_targets
 280     for j in range(num_multi_targets):
 281       task_j = target_sequences_to_data.keys()[j]
 282       data_j = target_sequences_to_data[task_j]
 283       print('Adding %s to target set' % task_j)
 284       j_ims = data_j['images']
 285       j_embs = data_j['embeddings']
 286 
 287       target_embs = SmoothEmbeddings(j_embs[tview])
 288       target_ims = j_ims[tview]
 289       all_target_embs.extend(target_embs)
 290       all_target_ims.extend(target_ims)
 291 
 292     # Generate a "j imitating i" video.
 293     tf.logging.info('Generating all imitating %s video.' % task_i)
 294     vid_name = 'q%sv%s_multiv%s' % (task_i, qview, tview)
 295     vid_name = vid_name.replace('/', '_')
 296     GenerateImitationVideo(vid_name, query_ims, query_embs,
 297                            all_target_ims, all_target_embs, height, width)

 300 def SameSequenceVideos(query_records, config, height, width):
 301   """Generate same sequence, cross-view imitation videos."""
 302   batch_size = config.data.embed_batch_size
 303 
 304   # Choose an estimator based on training strategy.
 305   estimator = get_estimator(config, FLAGS.checkpointdir)
 306 
 307   # Choose a checkpoint path to restore.
 308   checkpointdir = FLAGS.checkpointdir
 309   checkpoint_path = os.path.join(checkpointdir,
 310                                  'model.ckpt-%s' % FLAGS.checkpoint_iter)
 311 
 312   # Embed num_sequences query sequences, store embeddings and image strings in
 313   # query_sequences_to_data.
 314   sequences_to_data = {}
 315   for (view_embeddings, view_raw_image_strings, seqname) in estimator.inference(
 316       query_records, checkpoint_path, batch_size,
 317       num_sequences=FLAGS.num_query_sequences):
 318     sequences_to_data[seqname] = {
 319         'embeddings': view_embeddings,
 320         'images': view_raw_image_strings,
 321     }
 322 
 323   # Loop over query videos.
 324   qview = FLAGS.query_view
 325   tview = FLAGS.target_view
 326   for task_i, data_i in sequences_to_data.iteritems():
 327     ims = data_i['images']
 328     embs = data_i['embeddings']
 329     query_embs = SmoothEmbeddings(embs[qview])
 330     query_ims = ims[qview]
 331 
 332     target_embs = SmoothEmbeddings(embs[tview])
 333     target_ims = ims[tview]
 334 
 335     tf.logging.info('Generating %s imitating %s video.' % (task_i, task_i))
 336     vid_name = 'q%sv%s_im%sv%s' % (task_i, qview, task_i, tview)
 337     vid_name = vid_name.replace('/', '_')
 338     GenerateImitationVideo(vid_name, query_ims, query_embs,
 339                            target_ims, target_embs, height, width)

 342 def EmbedQueryTargetData(query_records, target_records, config):
 343   """Embeds the full set of query_records and target_records.
 344 
 345   Args:
 346     query_records: List of Strings, paths to tfrecord datasets to use as
 347       queries.
 348     target_records: List of Strings, paths to tfrecord datasets to use as
 349       targets.
 350     config: A T object describing training config.
 351 
 352   Returns:
 353     query_sequences_to_data: A dict holding 'embeddings' and 'images'
 354     target_sequences_to_data: A dict holding 'embeddings' and 'images'
 355   """
 356   batch_size = config.data.embed_batch_size
 357 
 358   # Choose an estimator based on training strategy.
 359   estimator = get_estimator(config, FLAGS.checkpointdir)
 360 
 361   # Choose a checkpoint path to restore.
 362   checkpointdir = FLAGS.checkpointdir
 363   checkpoint_path = os.path.join(checkpointdir,
 364                                  'model.ckpt-%s' % FLAGS.checkpoint_iter)
 365 
 366   # Embed num_sequences query sequences, store embeddings and image strings in
 367   # query_sequences_to_data.
 368   num_query_sequences = FLAGS.num_query_sequences
 369   num_target_sequences = FLAGS.num_target_sequences
 370   query_sequences_to_data = {}
 371   for (view_embeddings, view_raw_image_strings, seqname) in estimator.inference(
 372       query_records, checkpoint_path, batch_size,
 373       num_sequences=num_query_sequences):
 374     query_sequences_to_data[seqname] = {
 375         'embeddings': view_embeddings,
 376         'images': view_raw_image_strings,
 377     }
 378 
 379   if (query_records == target_records) and (
 380       num_query_sequences == num_target_sequences):
 381     target_sequences_to_data = query_sequences_to_data
 382   else:
 383     # Embed num_sequences target sequences, store embeddings and image strings
 384     # in sequences_to_data.
 385     target_sequences_to_data = {}
 386     for (view_embeddings, view_raw_image_strings,
 387          seqname) in estimator.inference(
 388              target_records, checkpoint_path, batch_size,
 389              num_sequences=num_target_sequences):
 390       target_sequences_to_data[seqname] = {
 391           'embeddings': view_embeddings,
 392           'images': view_raw_image_strings,
 393       }
 394   return query_sequences_to_data, target_sequences_to_data

 397 def main(_):
 398   # Parse config dict from yaml config files / command line flags.
 399   config = util.ParseConfigsToLuaTable(FLAGS.config_paths, FLAGS.model_params)
 400 
 401   # Get tables to embed.
 402   query_records_dir = FLAGS.query_records_dir
 403   query_records = util.GetFilesRecursively(query_records_dir)
 404 
 405   target_records_dir = FLAGS.target_records_dir
 406   target_records = util.GetFilesRecursively(target_records_dir)
 407 
 408   height = config.data.raw_height
 409   width = config.data.raw_width
 410   mode = FLAGS.mode
 411   if mode == 'multi':
 412     # Generate videos where target set is composed of multiple videos.
 413     MultiImitationVideos(query_records, target_records, config,
 414                          height, width)
 415   elif mode == 'single':
 416     # Generate videos where target set is a single video.
 417     SingleImitationVideos(query_records, target_records, config,
 418                           height, width)
 419   elif mode == 'same':
 420     # Generate videos where target set is the same as query, but diff view.
 421     SameSequenceVideos(query_records, config, height, width)
 422   else:
 423     raise ValueError('Unknown mode %s' % mode)
 424 
 425 if __name__ == '__main__':
 426   tf.app.run()
