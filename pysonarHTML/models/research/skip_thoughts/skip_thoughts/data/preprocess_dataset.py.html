  15 """Converts a set of text files to TFRecord format with Example protos.
  16 
  17 Each Example proto in the output contains the following fields:
  18 
  19   decode_pre: list of int64 ids corresponding to the "previous" sentence.
  20   encode: list of int64 ids corresponding to the "current" sentence.
  21   decode_post: list of int64 ids corresponding to the "post" sentence.
  22 
  23 In addition, the following files are generated:
  24 
  25   vocab.txt: List of "<word> <id>" pairs, where <id> is the integer
  26              encoding of <word> in the Example protos.
  27   word_counts.txt: List of "<word> <count>" pairs, where <count> is the number
  28                    of occurrences of <word> in the input files.
  29 
  30 The vocabulary of word ids is constructed from the top --num_words by word
  31 count. All other words get the <unk> word id.
  32 """
  33

  89 def _build_vocabulary(input_files):
  90   """Loads or builds the model vocabulary.
  91 
  92   Args:
  93     input_files: List of pre-tokenized input .txt files.
  94 
  95   Returns:
  96     vocab: A dictionary of word to id.
  97   """
  98   if FLAGS.vocab_file:
  99     tf.logging.info("Loading existing vocab file.")
 100     vocab = collections.OrderedDict()
 101     with tf.gfile.GFile(FLAGS.vocab_file, mode="r") as f:
 102       for i, line in enumerate(f):
 103         word = line.decode("utf-8").strip()
 104         assert word not in vocab, "Attempting to add word twice: %s" % word
 105         vocab[word] = i
 106     tf.logging.info("Read vocab of size %d from %s",
 107                     len(vocab), FLAGS.vocab_file)
 108     return vocab
 109 
 110   tf.logging.info("Creating vocabulary.")
 111   num = 0
 112   wordcount = collections.Counter()
 113   for input_file in input_files:
 114     tf.logging.info("Processing file: %s", input_file)
 115     for sentence in tf.gfile.FastGFile(input_file):
 116       wordcount.update(sentence.split())
 117 
 118       num += 1
 119       if num % 1000000 == 0:
 120         tf.logging.info("Processed %d sentences", num)
 121 
 122   tf.logging.info("Processed %d sentences total", num)
 123 
 124   words = list(wordcount)
 125   freqs = list(wordcount.values())
 126   sorted_indices = np.argsort(freqs)[::-1]
 127 
 128   vocab = collections.OrderedDict()
 129   vocab[special_words.EOS] = special_words.EOS_ID
 130   vocab[special_words.UNK] = special_words.UNK_ID
 131   for w_id, w_index in enumerate(sorted_indices[0:FLAGS.num_words - 2]):
 132     vocab[words[w_index]] = w_id + 2  # 0: EOS, 1: UNK.
 133 
 134   tf.logging.info("Created vocab with %d words", len(vocab))
 135 
 136   vocab_file = os.path.join(FLAGS.output_dir, "vocab.txt")
 137   with tf.gfile.FastGFile(vocab_file, "w") as f:
 138     f.write("\n".join(vocab.keys()))
 139   tf.logging.info("Wrote vocab file to %s", vocab_file)
 140 
 141   word_counts_file = os.path.join(FLAGS.output_dir, "word_counts.txt")
 142   with tf.gfile.FastGFile(word_counts_file, "w") as f:
 143     for i in sorted_indices:
 144       f.write("%s %d\n" % (words[i], freqs[i]))
 145   tf.logging.info("Wrote word counts file to %s", word_counts_file)
 146 
 147   return vocab

 150 def _int64_feature(value):
 151   """Helper for creating an Int64 Feature."""
 152   return tf.train.Feature(int64_list=tf.train.Int64List(
 153       value=[int(v) for v in value]))

 156 def _sentence_to_ids(sentence, vocab):
 157   """Helper for converting a sentence (list of words) to a list of ids."""
 158   ids = [vocab.get(w, special_words.UNK_ID) for w in sentence]
 159   if FLAGS.add_eos:
 160     ids.append(special_words.EOS_ID)
 161   return ids

 164 def _create_serialized_example(predecessor, current, successor, vocab):
 165   """Helper for creating a serialized Example proto."""
 166   example = tf.train.Example(features=tf.train.Features(feature={
 167       "decode_pre": _int64_feature(_sentence_to_ids(predecessor, vocab)),
 168       "encode": _int64_feature(_sentence_to_ids(current, vocab)),
 169       "decode_post": _int64_feature(_sentence_to_ids(successor, vocab)),
 170   }))
 171 
 172   return example.SerializeToString()

 175 def _process_input_file(filename, vocab, stats):
 176   """Processes the sentences in an input file.
 177 
 178   Args:
 179     filename: Path to a pre-tokenized input .txt file.
 180     vocab: A dictionary of word to id.
 181     stats: A Counter object for statistics.
 182 
 183   Returns:
 184     processed: A list of serialized Example protos
 185   """
 186   tf.logging.info("Processing input file: %s", filename)
 187   processed = []
 188 
 189   predecessor = None  # Predecessor sentence (list of words).
 190   current = None  # Current sentence (list of words).
 191   successor = None  # Successor sentence (list of words).
 192 
 193   for successor_str in tf.gfile.FastGFile(filename):
 194     stats.update(["sentences_seen"])
 195     successor = successor_str.split()
 196 
 197     # The first 2 sentences per file will be skipped.
 198     if predecessor and current and successor:
 199       stats.update(["sentences_considered"])
 200 
 201       # Note that we are going to insert <EOS> later, so we only allow
 202       # sentences with strictly less than max_sentence_length to pass.
 203       if FLAGS.max_sentence_length and (
 204           len(predecessor) >= FLAGS.max_sentence_length or len(current) >=
 205           FLAGS.max_sentence_length or len(successor) >=
 206           FLAGS.max_sentence_length):
 207         stats.update(["sentences_too_long"])
 208       else:
 209         serialized = _create_serialized_example(predecessor, current, successor,
 210                                                 vocab)
 211         processed.append(serialized)
 212         stats.update(["sentences_output"])
 213 
 214     predecessor = current
 215     current = successor
 216 
 217     sentences_seen = stats["sentences_seen"]
 218     sentences_output = stats["sentences_output"]
 219     if sentences_seen and sentences_seen % 100000 == 0:
 220       tf.logging.info("Processed %d sentences (%d output)", sentences_seen,
 221                       sentences_output)
 222     if FLAGS.max_sentences and sentences_output >= FLAGS.max_sentences:
 223       break
 224 
 225   tf.logging.info("Completed processing file %s", filename)
 226   return processed

 229 def _write_shard(filename, dataset, indices):
 230   """Writes a TFRecord shard."""
 231   with tf.python_io.TFRecordWriter(filename) as writer:
 232     for j in indices:
 233       writer.write(dataset[j])

 236 def _write_dataset(name, dataset, indices, num_shards):
 237   """Writes a sharded TFRecord dataset.
 238 
 239   Args:
 240     name: Name of the dataset (e.g. "train").
 241     dataset: List of serialized Example protos.
 242     indices: List of indices of 'dataset' to be written.
 243     num_shards: The number of output shards.
 244   """
 245   tf.logging.info("Writing dataset %s", name)
 246   borders = np.int32(np.linspace(0, len(indices), num_shards + 1))
 247   for i in range(num_shards):
 248     filename = os.path.join(FLAGS.output_dir, "%s-%.5d-of-%.5d" % (name, i,
 249                                                                    num_shards))
 250     shard_indices = indices[borders[i]:borders[i + 1]]
 251     _write_shard(filename, dataset, shard_indices)
 252     tf.logging.info("Wrote dataset indices [%d, %d) to output shard %s",
 253                     borders[i], borders[i + 1], filename)
 254   tf.logging.info("Finished writing %d sentences in dataset %s.",
 255                   len(indices), name)

 258 def main(unused_argv):
 259   if not FLAGS.input_files:
 260     raise ValueError("--input_files is required.")
 261   if not FLAGS.output_dir:
 262     raise ValueError("--output_dir is required.")
 263 
 264   if not tf.gfile.IsDirectory(FLAGS.output_dir):
 265     tf.gfile.MakeDirs(FLAGS.output_dir)
 266 
 267   input_files = []
 268   for pattern in FLAGS.input_files.split(","):
 269     match = tf.gfile.Glob(FLAGS.input_files)
 270     if not match:
 271       raise ValueError("Found no files matching %s" % pattern)
 272     input_files.extend(match)
 273   tf.logging.info("Found %d input files.", len(input_files))
 274 
 275   vocab = _build_vocabulary(input_files)
 276 
 277   tf.logging.info("Generating dataset.")
 278   stats = collections.Counter()
 279   dataset = []
 280   for filename in input_files:
 281     dataset.extend(_process_input_file(filename, vocab, stats))
 282     if FLAGS.max_sentences and stats["sentences_output"] >= FLAGS.max_sentences:
 283       break
 284 
 285   tf.logging.info("Generated dataset with %d sentences.", len(dataset))
 286   for k, v in stats.items():
 287     tf.logging.info("%s: %d", k, v)
 288 
 289   tf.logging.info("Shuffling dataset.")
 290   np.random.seed(123)
 291   shuffled_indices = np.random.permutation(len(dataset))
 292   val_indices = shuffled_indices[:FLAGS.num_validation_sentences]
 293   train_indices = shuffled_indices[FLAGS.num_validation_sentences:]
 294 
 295   _write_dataset("train", dataset, train_indices, FLAGS.train_output_shards)
 296   _write_dataset("validation", dataset, val_indices,
 297                  FLAGS.validation_output_shards)
 298 
 299 
 300 if __name__ == "__main__":
 301   tf.app.run()
