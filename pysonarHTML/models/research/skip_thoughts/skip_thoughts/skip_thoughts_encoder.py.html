  44 def _pad(seq, target_len):
  45   """Pads a sequence of word embeddings up to the target length.
  46 
  47   Args:
  48     seq: Sequence of word embeddings.
  49     target_len: Desired padded sequence length.
  50 
  51   Returns:
  52     embeddings: Input sequence padded with zero embeddings up to the target
  53       length.
  54     mask: A 0/1 vector with zeros corresponding to padded embeddings.
  55 
  56   Raises:
  57     ValueError: If len(seq) is not in the interval (0, target_len].
  58   """
  59   seq_len = len(seq)
  60   if seq_len <= 0 or seq_len > target_len:
  61     raise ValueError("Expected 0 < len(seq) <= %d, got %d" % (target_len,
  62                                                               seq_len))
  63 
  64   emb_dim = seq[0].shape[0]
  65   padded_seq = np.zeros(shape=(target_len, emb_dim), dtype=seq[0].dtype)
  66   mask = np.zeros(shape=(target_len,), dtype=np.int8)
  67   for i in range(seq_len):
  68     padded_seq[i] = seq[i]
  69     mask[i] = 1
  70   return padded_seq, mask

  73 def _batch_and_pad(sequences):
  74   """Batches and pads sequences of word embeddings into a 2D array.
  75 
  76   Args:
  77     sequences: A list of batch_size sequences of word embeddings.
  78 
  79   Returns:
  80     embeddings: A numpy array with shape [batch_size, padded_length, emb_dim].
  81     mask: A numpy 0/1 array with shape [batch_size, padded_length] with zeros
  82       corresponding to padded elements.
  83   """
  84   batch_embeddings = []
  85   batch_mask = []
  86   batch_len = max([len(seq) for seq in sequences])
  87   for seq in sequences:
  88     embeddings, mask = _pad(seq, batch_len)
  89     batch_embeddings.append(embeddings)
  90     batch_mask.append(mask)
  91   return np.array(batch_embeddings), np.array(batch_mask)

  94 class SkipThoughtsEncoder(object):
  95   """Skip-thoughts sentence encoder."""
  96 
  97   def __init__(self, embeddings):
  98     """Initializes the encoder.
  99 
 100     Args:
 101       embeddings: Dictionary of word to embedding vector (1D numpy array).
 102     """
 103     self._sentence_detector = nltk.data.load("tokenizers/punkt/english.pickle")
 104     self._embeddings = embeddings
 105 
 106   def _create_restore_fn(self, checkpoint_path, saver):
 107     """Creates a function that restores a model from checkpoint.
 108 
 109     Args:
 110       checkpoint_path: Checkpoint file or a directory containing a checkpoint
 111         file.
 112       saver: Saver for restoring variables from the checkpoint file.
 113 
 114     Returns:
 115       restore_fn: A function such that restore_fn(sess) loads model variables
 116         from the checkpoint file.
 117 
 118     Raises:
 119       ValueError: If checkpoint_path does not refer to a checkpoint file or a
 120         directory containing a checkpoint file.
 121     """
 122     if tf.gfile.IsDirectory(checkpoint_path):
 123       latest_checkpoint = tf.train.latest_checkpoint(checkpoint_path)
 124       if not latest_checkpoint:
 125         raise ValueError("No checkpoint file found in: %s" % checkpoint_path)
 126       checkpoint_path = latest_checkpoint
 127 
 128     def _restore_fn(sess):
 129       tf.logging.info("Loading model from checkpoint: %s", checkpoint_path)
 130       saver.restore(sess, checkpoint_path)
 131       tf.logging.info("Successfully loaded checkpoint: %s",
 132                       os.path.basename(checkpoint_path))
 133 
 134     return _restore_fn
 135 
 136   def build_graph_from_config(self, model_config, checkpoint_path):
 137     """Builds the inference graph from a configuration object.
 138 
 139     Args:
 140       model_config: Object containing configuration for building the model.
 141       checkpoint_path: Checkpoint file or a directory containing a checkpoint
 142         file.
 143 
 144     Returns:
 145       restore_fn: A function such that restore_fn(sess) loads model variables
 146         from the checkpoint file.
 147     """
 148     tf.logging.info("Building model.")
 149     model = skip_thoughts_model.SkipThoughtsModel(model_config, mode="encode")
 150     model.build()
 151     saver = tf.train.Saver()
 152 
 153     return self._create_restore_fn(checkpoint_path, saver)
 154 
 155   def build_graph_from_proto(self, graph_def_file, saver_def_file,
 156                              checkpoint_path):
 157     """Builds the inference graph from serialized GraphDef and SaverDef protos.
 158 
 159     Args:
 160       graph_def_file: File containing a serialized GraphDef proto.
 161       saver_def_file: File containing a serialized SaverDef proto.
 162       checkpoint_path: Checkpoint file or a directory containing a checkpoint
 163         file.
 164 
 165     Returns:
 166       restore_fn: A function such that restore_fn(sess) loads model variables
 167         from the checkpoint file.
 168     """
 169     # Load the Graph.
 170     tf.logging.info("Loading GraphDef from file: %s", graph_def_file)
 171     graph_def = tf.GraphDef()
 172     with tf.gfile.FastGFile(graph_def_file, "rb") as f:
 173       graph_def.ParseFromString(f.read())
 174     tf.import_graph_def(graph_def, name="")
 175 
 176     # Load the Saver.
 177     tf.logging.info("Loading SaverDef from file: %s", saver_def_file)
 178     saver_def = tf.train.SaverDef()
 179     with tf.gfile.FastGFile(saver_def_file, "rb") as f:
 180       saver_def.ParseFromString(f.read())
 181     saver = tf.train.Saver(saver_def=saver_def)
 182 
 183     return self._create_restore_fn(checkpoint_path, saver)
 184 
 185   def _tokenize(self, item):
 186     """Tokenizes an input string into a list of words."""
 187     tokenized = []
 188     for s in self._sentence_detector.tokenize(item):
 189       tokenized.extend(nltk.tokenize.word_tokenize(s))
 190 
 191     return tokenized
 192 
 193   def _word_to_embedding(self, w):
 194     """Returns the embedding of a word."""
 195     return self._embeddings.get(w, self._embeddings[special_words.UNK])
 196 
 197   def _preprocess(self, data, use_eos):
 198     """Preprocesses text for the encoder.
 199 
 200     Args:
 201       data: A list of input strings.
 202       use_eos: Whether to append the end-of-sentence word to each sentence.
 203 
 204     Returns:
 205       embeddings: A list of word embedding sequences corresponding to the input
 206         strings.
 207     """
 208     preprocessed_data = []
 209     for item in data:
 210       tokenized = self._tokenize(item)
 211       if use_eos:
 212         tokenized.append(special_words.EOS)
 213       preprocessed_data.append([self._word_to_embedding(w) for w in tokenized])
 214     return preprocessed_data
 215 
 216   def encode(self,
 217              sess,
 218              data,
 219              use_norm=True,
 220              verbose=True,
 221              batch_size=128,
 222              use_eos=False):
 223     """Encodes a sequence of sentences as skip-thought vectors.
 224 
 225     Args:
 226       sess: TensorFlow Session.
 227       data: A list of input strings.
 228       use_norm: Whether to normalize skip-thought vectors to unit L2 norm.
 229       verbose: Whether to log every batch.
 230       batch_size: Batch size for the encoder.
 231       use_eos: Whether to append the end-of-sentence word to each input
 232         sentence.
 233 
 234     Returns:
 235       thought_vectors: A list of numpy arrays corresponding to the skip-thought
 236         encodings of sentences in 'data'.
 237     """
 238     data = self._preprocess(data, use_eos)
 239     thought_vectors = []
 240 
 241     batch_indices = np.arange(0, len(data), batch_size)
 242     for batch, start_index in enumerate(batch_indices):
 243       if verbose:
 244         tf.logging.info("Batch %d / %d.", batch, len(batch_indices))
 245 
 246       embeddings, mask = _batch_and_pad(
 247           data[start_index:start_index + batch_size])
 248       feed_dict = {
 249           "encode_emb:0": embeddings,
 250           "encode_mask:0": mask,
 251       }
 252       thought_vectors.extend(
 253           sess.run("encoder/thought_vectors:0", feed_dict=feed_dict))
 254 
 255     if use_norm:
 256       thought_vectors = [v / np.linalg.norm(v) for v in thought_vectors]
 257 
 258     return thought_vectors
