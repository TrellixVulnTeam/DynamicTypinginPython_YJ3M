  29 class SkipThoughtsModel(skip_thoughts_model.SkipThoughtsModel):
  30   """Subclass of SkipThoughtsModel without the disk I/O."""
  31 
  32   def build_inputs(self):
  33     if self.mode == "encode":
  34       # Encode mode doesn't read from disk, so defer to parent.
  35       return super(SkipThoughtsModel, self).build_inputs()
  36     else:
  37       # Replace disk I/O with random Tensors.
  38       self.encode_ids = tf.random_uniform(
  39           [self.config.batch_size, 15],
  40           minval=0,
  41           maxval=self.config.vocab_size,
  42           dtype=tf.int64)
  43       self.decode_pre_ids = tf.random_uniform(
  44           [self.config.batch_size, 15],
  45           minval=0,
  46           maxval=self.config.vocab_size,
  47           dtype=tf.int64)
  48       self.decode_post_ids = tf.random_uniform(
  49           [self.config.batch_size, 15],
  50           minval=0,
  51           maxval=self.config.vocab_size,
  52           dtype=tf.int64)
  53       self.encode_mask = tf.ones_like(self.encode_ids)
  54       self.decode_pre_mask = tf.ones_like(self.decode_pre_ids)
  55       self.decode_post_mask = tf.ones_like(self.decode_post_ids)
  56

  61     super(SkipThoughtsModelTest, self).setUp()

  64   def _countModelParameters(self):
  65     """Counts the number of parameters in the model at top level scope."""
  66     counter = {}
  67     for v in tf.global_variables():
  68       name = v.op.name.split("/")[0]
  69       num_params = v.get_shape().num_elements()
  70       if not num_params:
  71         self.fail("Could not infer num_elements from Variable %s" % v.op.name)
  72       counter[name] = counter.get(name, 0) + num_params
  73     return counter
  74 
  75   def _checkModelParameters(self):
  76     """Verifies the number of parameters in the model."""
  77     param_counts = self._countModelParameters()
  78     expected_param_counts = {
  79         # vocab_size * embedding_size
  80         "word_embedding": 12400000,
  81         # GRU Cells
  82         "encoder": 21772800,
  83         "decoder_pre": 21772800,
  84         "decoder_post": 21772800,
  85         # (encoder_dim + 1) * vocab_size
  86         "logits": 48020000,
  87         "global_step": 1,
  88     }
  89     self.assertDictEqual(expected_param_counts, param_counts)
  90 
  91   def _checkOutputs(self, expected_shapes, feed_dict=None):
  92     """Verifies that the model produces expected outputs.
  93 
  94     Args:
  95       expected_shapes: A dict mapping Tensor or Tensor name to expected output
  96         shape.
  97       feed_dict: Values of Tensors to feed into Session.run().
  98     """
  99     fetches = expected_shapes.keys()
 100 
 101     with self.test_session() as sess:
 102       sess.run(tf.global_variables_initializer())
 103       outputs = sess.run(fetches, feed_dict)
 104 
 105     for index, output in enumerate(outputs):
 106       tensor = fetches[index]
 107       expected = expected_shapes[tensor]
 108       actual = output.shape
 109       if expected != actual:
 110         self.fail("Tensor %s has shape %s (expected %s)." % (tensor, actual,
 111                                                              expected))

 113   def testBuildForTraining(self):

 115     model.build()

 126         model.decode_post_mask: (128, 15),

 129         model.decode_pre_emb: (128, 15, 620),
 130         model.decode_post_emb: (128, 15, 620),
 131         # [batch_size, encoder_dim]
 132         model.thought_vectors: (128, 2400),
 133         # [batch_size * length]
 134         model.target_cross_entropy_losses[0]: (1920,),
 135         model.target_cross_entropy_losses[1]: (1920,),
 136         # [batch_size * length]
 137         model.target_cross_entropy_loss_weights[0]: (1920,),
 138         model.target_cross_entropy_loss_weights[1]: (1920,),
