  27 def model_config(input_file_pattern=None,
  28                  input_queue_capacity=640000,
  29                  num_input_reader_threads=1,
  30                  shuffle_input_data=True,
  31                  uniform_init_scale=0.1,
  32                  vocab_size=20000,
  33                  batch_size=128,
  34                  word_embedding_dim=620,
  35                  bidirectional_encoder=False,
  36                  encoder_dim=2400):
  37   """Creates a model configuration object.
  38 
  39   Args:
  40     input_file_pattern: File pattern of sharded TFRecord files containing
  41       tf.Example protobufs.
  42     input_queue_capacity: Number of examples to keep in the input queue.
  43     num_input_reader_threads: Number of threads for prefetching input
  44       tf.Examples.
  45     shuffle_input_data: Whether to shuffle the input data.
  46     uniform_init_scale: Scale of random uniform initializer.
  47     vocab_size: Number of unique words in the vocab.
  48     batch_size: Batch size (training and evaluation only).
  49     word_embedding_dim: Word embedding dimension.
  50     bidirectional_encoder: Whether to use a bidirectional or unidirectional
  51       encoder RNN.
  52     encoder_dim: Number of output dimensions of the sentence encoder.
  53 
  54   Returns:
  55     An object containing model configuration parameters.
  56   """
  57   config = _HParams()
  58   config.input_file_pattern = input_file_pattern
  59   config.input_queue_capacity = input_queue_capacity
  60   config.num_input_reader_threads = num_input_reader_threads
  61   config.shuffle_input_data = shuffle_input_data
  62   config.uniform_init_scale = uniform_init_scale
  63   config.vocab_size = vocab_size
  64   config.batch_size = batch_size
  65   config.word_embedding_dim = word_embedding_dim
  66   config.bidirectional_encoder = bidirectional_encoder
  67   config.encoder_dim = encoder_dim
  68   return config

  71 def training_config(learning_rate=0.0008,
  72                     learning_rate_decay_factor=0.5,
  73                     learning_rate_decay_steps=400000,
  74                     number_of_steps=500000,
  75                     clip_gradient_norm=5.0,
  76                     save_model_secs=600,
  77                     save_summaries_secs=600):
  78   """Creates a training configuration object.
  79 
  80   Args:
  81     learning_rate: Initial learning rate.
  82     learning_rate_decay_factor: If > 0, the learning rate decay factor.
  83     learning_rate_decay_steps: The number of steps before the learning rate
  84       decays by learning_rate_decay_factor.
  85     number_of_steps: The total number of training steps to run. Passing None
  86       will cause the training script to run indefinitely.
  87     clip_gradient_norm: If not None, then clip gradients to this value.
  88     save_model_secs: How often (in seconds) to save model checkpoints.
  89     save_summaries_secs: How often (in seconds) to save model summaries.
  90 
  91   Returns:
  92     An object containing training configuration parameters.
  93 
  94   Raises:
  95     ValueError: If learning_rate_decay_factor is set and
  96       learning_rate_decay_steps is unset.
  97   """
  98   if learning_rate_decay_factor and not learning_rate_decay_steps:
  99     raise ValueError(
 100         "learning_rate_decay_factor requires learning_rate_decay_steps.")
 101 
 102   config = _HParams()
 103   config.learning_rate = learning_rate
 104   config.learning_rate_decay_factor = learning_rate_decay_factor
 105   config.learning_rate_decay_steps = learning_rate_decay_steps
 106   config.number_of_steps = number_of_steps
 107   config.clip_gradient_norm = clip_gradient_norm
 108   config.save_model_secs = save_model_secs
 109   config.save_summaries_secs = save_summaries_secs
 110   return config
