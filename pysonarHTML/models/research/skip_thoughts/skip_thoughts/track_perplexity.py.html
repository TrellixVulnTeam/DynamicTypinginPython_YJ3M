  57 def evaluate_model(sess, losses, weights, num_batches, global_step,
  58                    summary_writer, summary_op):
  59   """Computes perplexity-per-word over the evaluation dataset.
  60 
  61   Summaries and perplexity-per-word are written out to the eval directory.
  62 
  63   Args:
  64     sess: Session object.
  65     losses: A Tensor of any shape; the target cross entropy losses for the
  66       current batch.
  67     weights: A Tensor of weights corresponding to losses.
  68     num_batches: Integer; the number of evaluation batches.
  69     global_step: Integer; global step of the model checkpoint.
  70     summary_writer: Instance of SummaryWriter.
  71     summary_op: Op for generating model summaries.
  72   """
  73   # Log model summaries on a single batch.
  74   summary_str = sess.run(summary_op)
  75   summary_writer.add_summary(summary_str, global_step)
  76 
  77   start_time = time.time()
  78   sum_losses = 0.0
  79   sum_weights = 0.0
  80   for i in range(num_batches):
  81     batch_losses, batch_weights = sess.run([losses, weights])
  82     sum_losses += np.sum(batch_losses * batch_weights)
  83     sum_weights += np.sum(batch_weights)
  84     if not i % 100:
  85       tf.logging.info("Computed losses for %d of %d batches.", i + 1,
  86                       num_batches)
  87   eval_time = time.time() - start_time
  88 
  89   perplexity = math.exp(sum_losses / sum_weights)
  90   tf.logging.info("Perplexity = %f (%.2f sec)", perplexity, eval_time)
  91 
  92   # Log perplexity to the SummaryWriter.
  93   summary = tf.Summary()
  94   value = summary.value.add()
  95   value.simple_value = perplexity
  96   value.tag = "perplexity"
  97   summary_writer.add_summary(summary, global_step)
  98 
  99   # Write the Events file to the eval directory.
 100   summary_writer.flush()
 101   tf.logging.info("Finished processing evaluation at global step %d.",
 102                   global_step)

 105 def run_once(model, losses, weights, saver, summary_writer, summary_op):
 106   """Evaluates the latest model checkpoint.
 107 
 108   Args:
 109     model: Instance of SkipThoughtsModel; the model to evaluate.
 110     losses: Tensor; the target cross entropy losses for the current batch.
 111     weights: A Tensor of weights corresponding to losses.
 112     saver: Instance of tf.train.Saver for restoring model Variables.
 113     summary_writer: Instance of FileWriter.
 114     summary_op: Op for generating model summaries.
 115   """
 116   model_path = tf.train.latest_checkpoint(FLAGS.checkpoint_dir)
 117   if not model_path:
 118     tf.logging.info("Skipping evaluation. No checkpoint found in: %s",
 119                     FLAGS.checkpoint_dir)
 120     return
 121 
 122   with tf.Session() as sess:
 123     # Load model from checkpoint.
 124     tf.logging.info("Loading model from checkpoint: %s", model_path)
 125     saver.restore(sess, model_path)
 126     global_step = tf.train.global_step(sess, model.global_step.name)
 127     tf.logging.info("Successfully loaded %s at global step = %d.",
 128                     os.path.basename(model_path), global_step)
 129     if global_step < FLAGS.min_global_step:
 130       tf.logging.info("Skipping evaluation. Global step = %d < %d", global_step,
 131                       FLAGS.min_global_step)
 132       return
 133 
 134     # Start the queue runners.
 135     coord = tf.train.Coordinator()
 136     threads = tf.train.start_queue_runners(coord=coord)
 137 
 138     num_eval_batches = int(
 139         math.ceil(FLAGS.num_eval_examples / model.config.batch_size))
 140 
 141     # Run evaluation on the latest checkpoint.
 142     try:
 143       evaluate_model(sess, losses, weights, num_eval_batches, global_step,
 144                      summary_writer, summary_op)
 145     except tf.InvalidArgumentError:
 146       tf.logging.error(
 147           "Evaluation raised InvalidArgumentError (e.g. due to Nans).")
 148     finally:
 149       coord.request_stop()
 150       coord.join(threads, stop_grace_period_secs=10)

 153 def main(unused_argv):
 154   if not FLAGS.input_file_pattern:
 155     raise ValueError("--input_file_pattern is required.")
 156   if not FLAGS.checkpoint_dir:
 157     raise ValueError("--checkpoint_dir is required.")
 158   if not FLAGS.eval_dir:
 159     raise ValueError("--eval_dir is required.")
 160 
 161   # Create the evaluation directory if it doesn't exist.
 162   eval_dir = FLAGS.eval_dir
 163   if not tf.gfile.IsDirectory(eval_dir):
 164     tf.logging.info("Creating eval directory: %s", eval_dir)
 165     tf.gfile.MakeDirs(eval_dir)
 166 
 167   g = tf.Graph()
 168   with g.as_default():
 169     # Build the model for evaluation.
 170     model_config = configuration.model_config(
 171         input_file_pattern=FLAGS.input_file_pattern,
 172         input_queue_capacity=FLAGS.num_eval_examples,
 173         shuffle_input_data=False)
 174     model = skip_thoughts_model.SkipThoughtsModel(model_config, mode="eval")
 175     model.build()
 176 
 177     losses = tf.concat(model.target_cross_entropy_losses, 0)
 178     weights = tf.concat(model.target_cross_entropy_loss_weights, 0)
 179 
 180     # Create the Saver to restore model Variables.
 181     saver = tf.train.Saver()
 182 
 183     # Create the summary operation and the summary writer.
 184     summary_op = tf.summary.merge_all()
 185     summary_writer = tf.summary.FileWriter(eval_dir)
 186 
 187     g.finalize()
 188 
 189     # Run a new evaluation run every eval_interval_secs.
 190     while True:
 191       start = time.time()
 192       tf.logging.info("Starting evaluation at " + time.strftime(
 193           "%Y-%m-%d-%H:%M:%S", time.localtime()))
 194       run_once(model, losses, weights, saver, summary_writer, summary_op)
 195       time_to_next_eval = start + FLAGS.eval_interval_secs - time.time()
 196       if time_to_next_eval > 0:
 197         time.sleep(time_to_next_eval)
 198 
 199 
 200 if __name__ == "__main__":
 201   tf.app.run()
