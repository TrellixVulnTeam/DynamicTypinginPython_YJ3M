  15 """Compute an expanded vocabulary of embeddings using a word2vec model.
  16 
  17 This script loads the word embeddings from a trained skip-thoughts model and
  18 from a trained word2vec model (typically with a larger vocabulary). It trains a
  19 linear regression model without regularization to learn a linear mapping from
  20 the word2vec embedding space to the skip-thoughts embedding space. The model is
  21 then applied to all words in the word2vec vocabulary, yielding vectors in the
  22 skip-thoughts word embedding space for the union of the two vocabularies.
  23 
  24 The linear regression task is to learn a parameter matrix W to minimize
  25   || X - Y * W ||^2,
  26 where X is a matrix of skip-thoughts embeddings of shape [num_words, dim1],
  27 Y is a matrix of word2vec embeddings of shape [num_words, dim2], and W is a
  28 matrix of shape [dim2, dim1].
  29 
  30 This is based on the "Translation Matrix" method from the paper:
  31 
  32   "Exploiting Similarities among Languages for Machine Translation"
  33   Tomas Mikolov, Quoc V. Le, Ilya Sutskever
  34   https://arxiv.org/abs/1309.4168
  35 """
  36

  69 def _load_skip_thoughts_embeddings(checkpoint_path):
  70   """Loads the embedding matrix from a skip-thoughts model checkpoint.
  71 
  72   Args:
  73     checkpoint_path: Model checkpoint file or directory containing a checkpoint
  74         file.
  75 
  76   Returns:
  77     word_embedding: A numpy array of shape [vocab_size, embedding_dim].
  78 
  79   Raises:
  80     ValueError: If no checkpoint file matches checkpoint_path.
  81   """
  82   if tf.gfile.IsDirectory(checkpoint_path):
  83     checkpoint_file = tf.train.latest_checkpoint(checkpoint_path)
  84     if not checkpoint_file:
  85       raise ValueError("No checkpoint file found in %s" % checkpoint_path)
  86   else:
  87     checkpoint_file = checkpoint_path
  88 
  89   tf.logging.info("Loading skip-thoughts embedding matrix from %s",
  90                   checkpoint_file)
  91   reader = tf.train.NewCheckpointReader(checkpoint_file)
  92   word_embedding = reader.get_tensor("word_embedding")
  93   tf.logging.info("Loaded skip-thoughts embedding matrix of shape %s",
  94                   word_embedding.shape)
  95 
  96   return word_embedding

  99 def _load_vocabulary(filename):
 100   """Loads a vocabulary file.
 101 
 102   Args:
 103     filename: Path to text file containing newline-separated words.
 104 
 105   Returns:
 106     vocab: A dictionary mapping word to word id.
 107   """
 108   tf.logging.info("Reading vocabulary from %s", filename)
 109   vocab = collections.OrderedDict()
 110   with tf.gfile.GFile(filename, mode="rb") as f:
 111     for i, line in enumerate(f):
 112       word = line.decode("utf-8").strip()
 113       assert word not in vocab, "Attempting to add word twice: %s" % word
 114       vocab[word] = i
 115   tf.logging.info("Read vocabulary of size %d", len(vocab))
 116   return vocab

 119 def _expand_vocabulary(skip_thoughts_emb, skip_thoughts_vocab, word2vec):
 120   """Runs vocabulary expansion on a skip-thoughts model using a word2vec model.
 121 
 122   Args:
 123     skip_thoughts_emb: A numpy array of shape [skip_thoughts_vocab_size,
 124         skip_thoughts_embedding_dim].
 125     skip_thoughts_vocab: A dictionary of word to id.
 126     word2vec: An instance of gensim.models.Word2Vec.
 127 
 128   Returns:
 129     combined_emb: A dictionary mapping words to embedding vectors.
 130   """
 131   # Find words shared between the two vocabularies.
 132   tf.logging.info("Finding shared words")
 133   shared_words = [w for w in word2vec.vocab if w in skip_thoughts_vocab]
 134 
 135   # Select embedding vectors for shared words.
 136   tf.logging.info("Selecting embeddings for %d shared words", len(shared_words))
 137   shared_st_emb = skip_thoughts_emb[[
 138       skip_thoughts_vocab[w] for w in shared_words
 139   ]]
 140   shared_w2v_emb = word2vec[shared_words]
 141 
 142   # Train a linear regression model on the shared embedding vectors.
 143   tf.logging.info("Training linear regression model")
 144   model = sklearn.linear_model.LinearRegression()
 145   model.fit(shared_w2v_emb, shared_st_emb)
 146 
 147   # Create the expanded vocabulary.
 148   tf.logging.info("Creating embeddings for expanded vocabuary")
 149   combined_emb = collections.OrderedDict()
 150   for w in word2vec.vocab:
 151     # Ignore words with underscores (spaces).
 152     if "_" not in w:
 153       w_emb = model.predict(word2vec[w].reshape(1, -1))
 154       combined_emb[w] = w_emb.reshape(-1)
 155 
 156   for w in skip_thoughts_vocab:
 157     combined_emb[w] = skip_thoughts_emb[skip_thoughts_vocab[w]]
 158 
 159   tf.logging.info("Created expanded vocabulary of %d words", len(combined_emb))
 160 
 161   return combined_emb

 164 def main(unused_argv):
 165   if not FLAGS.skip_thoughts_model:
 166     raise ValueError("--skip_thoughts_model is required.")
 167   if not FLAGS.skip_thoughts_vocab:
 168     raise ValueError("--skip_thoughts_vocab is required.")
 169   if not FLAGS.word2vec_model:
 170     raise ValueError("--word2vec_model is required.")
 171   if not FLAGS.output_dir:
 172     raise ValueError("--output_dir is required.")
 173 
 174   if not tf.gfile.IsDirectory(FLAGS.output_dir):
 175     tf.gfile.MakeDirs(FLAGS.output_dir)
 176 
 177   # Load the skip-thoughts embeddings and vocabulary.
 178   skip_thoughts_emb = _load_skip_thoughts_embeddings(FLAGS.skip_thoughts_model)
 179   skip_thoughts_vocab = _load_vocabulary(FLAGS.skip_thoughts_vocab)
 180 
 181   # Load the Word2Vec model.
 182   word2vec = gensim.models.KeyedVectors.load_word2vec_format(
 183       FLAGS.word2vec_model, binary=True)
 184 
 185   # Run vocabulary expansion.
 186   embedding_map = _expand_vocabulary(skip_thoughts_emb, skip_thoughts_vocab,
 187                                      word2vec)
 188 
 189   # Save the output.
 190   vocab = embedding_map.keys()
 191   vocab_file = os.path.join(FLAGS.output_dir, "vocab.txt")
 192   with tf.gfile.GFile(vocab_file, "w") as f:
 193     f.write("\n".join(vocab))
 194   tf.logging.info("Wrote vocabulary file to %s", vocab_file)
 195 
 196   embeddings = np.array(embedding_map.values())
 197   embeddings_file = os.path.join(FLAGS.output_dir, "embeddings.npy")
 198   np.save(embeddings_file, embeddings)
 199   tf.logging.info("Wrote embeddings file to %s", embeddings_file)
 200 
 201 
 202 if __name__ == "__main__":
 203   tf.app.run()
