  42 def random_orthonormal_initializer(shape, dtype=tf.float32,
  43                                    partition_info=None):  # pylint: disable=unused-argument
  44   """Variable initializer that produces a random orthonormal matrix."""
  45   if len(shape) != 2 or shape[0] != shape[1]:
  46     raise ValueError("Expecting square shape, got %s" % shape)
  47   _, u, _ = tf.svd(tf.random_normal(shape, dtype=dtype), full_matrices=True)
  48   return u

  51 class SkipThoughtsModel(object):
  52   """Skip-thoughts model."""

  54   def __init__(self, config, mode="train", input_reader=None):
  55     """Basic setup. The actual TensorFlow graph is constructed in build().
  56 
  57     Args:
  58       config: Object containing configuration parameters.
  59       mode: "train", "eval" or "encode".
  60       input_reader: Subclass of tf.ReaderBase for reading the input serialized
  61         tf.Example protocol buffers. Defaults to TFRecordReader.
  62 
  63     Raises:
  64       ValueError: If mode is invalid.
  65     """
  66     if mode not in ["train", "eval", "encode"]:
  67       raise ValueError("Unrecognized mode: %s" % mode)
  68 
  69     self.config = config
  70     self.mode = mode
  71     self.reader = input_reader if input_reader else tf.TFRecordReader()
  72 
  73     # Initializer used for non-recurrent weights.
  74     self.uniform_initializer = tf.random_uniform_initializer(
  75         minval=-self.config.uniform_init_scale,
  76         maxval=self.config.uniform_init_scale)
  77 
  78     # Input sentences represented as sequences of word ids. "encode" is the
  79     # source sentence, "decode_pre" is the previous sentence and "decode_post"
  80     # is the next sentence.
  81     # Each is an int64 Tensor with  shape [batch_size, padded_length].
  82     self.encode_ids = None
  83     self.decode_pre_ids = None
  84     self.decode_post_ids = None
  85 
  86     # Boolean masks distinguishing real words (1) from padded words (0).
  87     # Each is an int32 Tensor with shape [batch_size, padded_length].
  88     self.encode_mask = None
  89     self.decode_pre_mask = None
  90     self.decode_post_mask = None
  91 
  92     # Input sentences represented as sequences of word embeddings.
  93     # Each is a float32 Tensor with shape [batch_size, padded_length, emb_dim].
  94     self.encode_emb = None
  95     self.decode_pre_emb = None
  96     self.decode_post_emb = None
  97 
  98     # The output from the sentence encoder.
  99     # A float32 Tensor with shape [batch_size, num_gru_units].
 100     self.thought_vectors = None
 101 
 102     # The cross entropy losses and corresponding weights of the decoders. Used
 103     # for evaluation.
 104     self.target_cross_entropy_losses = []
 105     self.target_cross_entropy_loss_weights = []
 106 
 107     # The total loss to optimize.
 108     self.total_loss = None
 109 
 110   def build_inputs(self):
 111     """Builds the ops for reading input data.
 112 
 113     Outputs:
 114       self.encode_ids
 115       self.decode_pre_ids
 116       self.decode_post_ids
 117       self.encode_mask
 118       self.decode_pre_mask
 119       self.decode_post_mask
 120     """
 121     if self.mode == "encode":
 122       # Word embeddings are fed from an external vocabulary which has possibly
 123       # been expanded (see vocabulary_expansion.py).
 124       encode_ids = None
 125       decode_pre_ids = None
 126       decode_post_ids = None
 127       encode_mask = tf.placeholder(tf.int8, (None, None), name="encode_mask")
 128       decode_pre_mask = None
 129       decode_post_mask = None
 130     else:
 131       # Prefetch serialized tf.Example protos.
 132       input_queue = input_ops.prefetch_input_data(
 133           self.reader,
 134           self.config.input_file_pattern,
 135           shuffle=self.config.shuffle_input_data,
 136           capacity=self.config.input_queue_capacity,
 137           num_reader_threads=self.config.num_input_reader_threads)
 138 
 139       # Deserialize a batch.
 140       serialized = input_queue.dequeue_many(self.config.batch_size)
 141       encode, decode_pre, decode_post = input_ops.parse_example_batch(
 142           serialized)
 143 
 144       encode_ids = encode.ids
 145       decode_pre_ids = decode_pre.ids
 146       decode_post_ids = decode_post.ids
 147 
 148       encode_mask = encode.mask
 149       decode_pre_mask = decode_pre.mask
 150       decode_post_mask = decode_post.mask
 151 
 152     self.encode_ids = encode_ids
 153     self.decode_pre_ids = decode_pre_ids
 154     self.decode_post_ids = decode_post_ids
 155 
 156     self.encode_mask = encode_mask
 157     self.decode_pre_mask = decode_pre_mask
 158     self.decode_post_mask = decode_post_mask
 159 
 160   def build_word_embeddings(self):
 161     """Builds the word embeddings.
 162 
 163     Inputs:
 164       self.encode_ids
 165       self.decode_pre_ids
 166       self.decode_post_ids
 167 
 168     Outputs:
 169       self.encode_emb
 170       self.decode_pre_emb
 171       self.decode_post_emb
 172     """
 173     if self.mode == "encode":
 174       # Word embeddings are fed from an external vocabulary which has possibly
 175       # been expanded (see vocabulary_expansion.py).
 176       encode_emb = tf.placeholder(tf.float32, (
 177           None, None, self.config.word_embedding_dim), "encode_emb")
 178       # No sequences to decode.
 179       decode_pre_emb = None
 180       decode_post_emb = None
 181     else:
 182       word_emb = tf.get_variable(
 183           name="word_embedding",
 184           shape=[self.config.vocab_size, self.config.word_embedding_dim],
 185           initializer=self.uniform_initializer)
 186 
 187       encode_emb = tf.nn.embedding_lookup(word_emb, self.encode_ids)
 188       decode_pre_emb = tf.nn.embedding_lookup(word_emb, self.decode_pre_ids)
 189       decode_post_emb = tf.nn.embedding_lookup(word_emb, self.decode_post_ids)
 190 
 191     self.encode_emb = encode_emb
 192     self.decode_pre_emb = decode_pre_emb
 193     self.decode_post_emb = decode_post_emb
 194 
 195   def _initialize_gru_cell(self, num_units):
 196     """Initializes a GRU cell.
 197 
 198     The Variables of the GRU cell are initialized in a way that exactly matches
 199     the skip-thoughts paper: recurrent weights are initialized from random
 200     orthonormal matrices and non-recurrent weights are initialized from random
 201     uniform matrices.
 202 
 203     Args:
 204       num_units: Number of output units.
 205 
 206     Returns:
 207       cell: An instance of RNNCell with variable initializers that match the
 208         skip-thoughts paper.
 209     """
 210     return gru_cell.LayerNormGRUCell(
 211         num_units,
 212         w_initializer=self.uniform_initializer,
 213         u_initializer=random_orthonormal_initializer,
 214         b_initializer=tf.constant_initializer(0.0))
 215 
 216   def build_encoder(self):
 217     """Builds the sentence encoder.
 218 
 219     Inputs:
 220       self.encode_emb
 221       self.encode_mask
 222 
 223     Outputs:
 224       self.thought_vectors
 225 
 226     Raises:
 227       ValueError: if config.bidirectional_encoder is True and config.encoder_dim
 228         is odd.
 229     """
 230     with tf.variable_scope("encoder") as scope:
 231       length = tf.to_int32(tf.reduce_sum(self.encode_mask, 1), name="length")
 232 
 233       if self.config.bidirectional_encoder:
 234         if self.config.encoder_dim % 2:
 235           raise ValueError(
 236               "encoder_dim must be even when using a bidirectional encoder.")
 237         num_units = self.config.encoder_dim // 2
 238         cell_fw = self._initialize_gru_cell(num_units)  # Forward encoder
 239         cell_bw = self._initialize_gru_cell(num_units)  # Backward encoder
 240         _, states = tf.nn.bidirectional_dynamic_rnn(
 241             cell_fw=cell_fw,
 242             cell_bw=cell_bw,
 243             inputs=self.encode_emb,
 244             sequence_length=length,
 245             dtype=tf.float32,
 246             scope=scope)
 247         thought_vectors = tf.concat(states, 1, name="thought_vectors")
 248       else:
 249         cell = self._initialize_gru_cell(self.config.encoder_dim)
 250         _, state = tf.nn.dynamic_rnn(
 251             cell=cell,
 252             inputs=self.encode_emb,
 253             sequence_length=length,
 254             dtype=tf.float32,
 255             scope=scope)
 256         # Use an identity operation to name the Tensor in the Graph.
 257         thought_vectors = tf.identity(state, name="thought_vectors")
 258 
 259     self.thought_vectors = thought_vectors
 260 
 261   def _build_decoder(self, name, embeddings, targets, mask, initial_state,
 262                      reuse_logits):
 263     """Builds a sentence decoder.
 264 
 265     Args:
 266       name: Decoder name.
 267       embeddings: Batch of sentences to decode; a float32 Tensor with shape
 268         [batch_size, padded_length, emb_dim].
 269       targets: Batch of target word ids; an int64 Tensor with shape
 270         [batch_size, padded_length].
 271       mask: A 0/1 Tensor with shape [batch_size, padded_length].
 272       initial_state: Initial state of the GRU. A float32 Tensor with shape
 273         [batch_size, num_gru_cells].
 274       reuse_logits: Whether to reuse the logits weights.
 275     """
 276     # Decoder RNN.
 277     cell = self._initialize_gru_cell(self.config.encoder_dim)
 278     with tf.variable_scope(name) as scope:
 279       # Add a padding word at the start of each sentence (to correspond to the
 280       # prediction of the first word) and remove the last word.
 281       decoder_input = tf.pad(
 282           embeddings[:, :-1, :], [[0, 0], [1, 0], [0, 0]], name="input")
 283       length = tf.reduce_sum(mask, 1, name="length")
 284       decoder_output, _ = tf.nn.dynamic_rnn(
 285           cell=cell,
 286           inputs=decoder_input,
 287           sequence_length=length,
 288           initial_state=initial_state,
 289           scope=scope)
 290 
 291     # Stack batch vertically.
 292     decoder_output = tf.reshape(decoder_output, [-1, self.config.encoder_dim])
 293     targets = tf.reshape(targets, [-1])
 294     weights = tf.to_float(tf.reshape(mask, [-1]))
 295 
 296     # Logits.
 297     with tf.variable_scope("logits", reuse=reuse_logits) as scope:
 298       logits = tf.contrib.layers.fully_connected(
 299           inputs=decoder_output,
 300           num_outputs=self.config.vocab_size,
 301           activation_fn=None,
 302           weights_initializer=self.uniform_initializer,
 303           scope=scope)
 304 
 305     losses = tf.nn.sparse_softmax_cross_entropy_with_logits(
 306         labels=targets, logits=logits)
 307     batch_loss = tf.reduce_sum(losses * weights)
 308     tf.losses.add_loss(batch_loss)
 309 
 310     tf.summary.scalar("losses/" + name, batch_loss)
 311 
 312     self.target_cross_entropy_losses.append(losses)
 313     self.target_cross_entropy_loss_weights.append(weights)
 314 
 315   def build_decoders(self):
 316     """Builds the sentence decoders.
 317 
 318     Inputs:
 319       self.decode_pre_emb
 320       self.decode_post_emb
 321       self.decode_pre_ids
 322       self.decode_post_ids
 323       self.decode_pre_mask
 324       self.decode_post_mask
 325       self.thought_vectors
 326 
 327     Outputs:
 328       self.target_cross_entropy_losses
 329       self.target_cross_entropy_loss_weights
 330     """
 331     if self.mode != "encode":
 332       # Pre-sentence decoder.
 333       self._build_decoder("decoder_pre", self.decode_pre_emb,
 334                           self.decode_pre_ids, self.decode_pre_mask,
 335                           self.thought_vectors, False)
 336 
 337       # Post-sentence decoder. Logits weights are reused.
 338       self._build_decoder("decoder_post", self.decode_post_emb,
 339                           self.decode_post_ids, self.decode_post_mask,
 340                           self.thought_vectors, True)
 341 
 342   def build_loss(self):
 343     """Builds the loss Tensor.
 344 
 345     Outputs:
 346       self.total_loss
 347     """
 348     if self.mode != "encode":
 349       total_loss = tf.losses.get_total_loss()
 350       tf.summary.scalar("losses/total", total_loss)
 351 
 352       self.total_loss = total_loss
 353 
 354   def build_global_step(self):
 355     """Builds the global step Tensor.
 356 
 357     Outputs:
 358       self.global_step
 359     """
 360     self.global_step = tf.contrib.framework.create_global_step()
 361 
 362   def build(self):
 363     """Creates all ops for training, evaluation or encoding."""
 364     self.build_inputs()
 365     self.build_word_embeddings()
 366     self.build_encoder()
 367     self.build_decoders()
 368     self.build_loss()
 369     self.build_global_step()
