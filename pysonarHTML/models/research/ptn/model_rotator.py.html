  36 def _get_data_from_provider(inputs, batch_size, split_name):
  37   """Returns dictionary of batch input data processed by tf.train.batch."""
  38   images, masks = tf.train.batch(
  39       [inputs['image'], inputs['mask']],
  40       batch_size=batch_size,
  41       num_threads=8,
  42       capacity=8 * batch_size,
  43       name='batching_queues/%s' % (split_name))
  44 
  45   outputs = dict()
  46   outputs['images'] = images
  47   outputs['masks'] = masks
  48   outputs['num_samples'] = inputs['num_samples']
  49 
  50   return outputs

  53 def get_inputs(dataset_dir, dataset_name, split_name, batch_size, image_size,
  54                is_training):
  55   """Loads the given dataset and split."""
  56   del image_size  # Unused
  57   with tf.variable_scope('data_loading_%s/%s' % (dataset_name, split_name)):
  58     common_queue_min = 50
  59     common_queue_capacity = 256
  60     num_readers = 4
  61 
  62     inputs = input_generator.get(
  63         dataset_dir,
  64         dataset_name,
  65         split_name,
  66         shuffle=is_training,
  67         num_readers=num_readers,
  68         common_queue_min=common_queue_min,
  69         common_queue_capacity=common_queue_capacity)
  70 
  71     return _get_data_from_provider(inputs, batch_size, split_name)

  74 def preprocess(raw_inputs, step_size):
  75   """Selects the subset of viewpoints to train on."""
  76   shp = raw_inputs['images'].get_shape().as_list()
  77   quantity = shp[0]
  78   num_views = shp[1]
  79   image_size = shp[2]
  80   del image_size  # Unused
  81 
  82   batch_rot = np.zeros((quantity, 3), dtype=np.float32)
  83   inputs = dict()
  84   for n in xrange(step_size + 1):
  85     inputs['images_%d' % n] = []
  86     inputs['masks_%d' % n] = []
  87 
  88   for n in xrange(quantity):
  89     view_in = np.random.randint(0, num_views)
  90     rng_rot = np.random.randint(0, 2)
  91     if step_size == 1:
  92       rng_rot = np.random.randint(0, 3)
  93 
  94     delta = 0
  95     if rng_rot == 0:
  96       delta = -1
  97       batch_rot[n, 2] = 1
  98     elif rng_rot == 1:
  99       delta = 1
 100       batch_rot[n, 0] = 1
 101     else:
 102       delta = 0
 103       batch_rot[n, 1] = 1
 104 
 105     inputs['images_0'].append(raw_inputs['images'][n, view_in, :, :, :])
 106     inputs['masks_0'].append(raw_inputs['masks'][n, view_in, :, :, :])
 107 
 108     view_out = view_in
 109     for k in xrange(1, step_size + 1):
 110       view_out += delta
 111       if view_out >= num_views:
 112         view_out = 0
 113       if view_out < 0:
 114         view_out = num_views - 1
 115 
 116       inputs['images_%d' % k].append(raw_inputs['images'][n, view_out, :, :, :])
 117       inputs['masks_%d' % k].append(raw_inputs['masks'][n, view_out, :, :, :])
 118 
 119   for n in xrange(step_size + 1):
 120     inputs['images_%d' % n] = tf.stack(inputs['images_%d' % n])
 121     inputs['masks_%d' % n] = tf.stack(inputs['masks_%d' % n])
 122 
 123   inputs['actions'] = tf.constant(batch_rot, dtype=tf.float32)
 124   return inputs

 127 def get_init_fn(scopes, params):
 128   """Initialization assignment operator function used while training."""
 129   if not params.init_model:
 130     return None
 131 
 132   is_trainable = lambda x: x in tf.trainable_variables()
 133   var_list = []
 134   for scope in scopes:
 135     var_list.extend(
 136         filter(is_trainable, tf.contrib.framework.get_model_variables(scope)))
 137 
 138   init_assign_op, init_feed_dict = slim.assign_from_checkpoint(
 139       params.init_model, var_list)
 140 
 141   def init_assign_function(sess):
 142     sess.run(init_assign_op, init_feed_dict)
 143 
 144   return init_assign_function

 155 def get_loss(inputs, outputs, params):
 156   """Computes the rotator loss."""
 157   g_loss = tf.zeros(dtype=tf.float32, shape=[])
 158 
 159   if hasattr(params, 'image_weight'):
 160     g_loss += losses.add_rotator_image_loss(inputs, outputs, params.step_size,
 161                                             params.image_weight)
 162 
 163   if hasattr(params, 'mask_weight'):
 164     g_loss += losses.add_rotator_mask_loss(inputs, outputs, params.step_size,
 165                                            params.mask_weight)
 166 
 167   slim.summaries.add_scalar_summary(
 168       g_loss, 'rotator_loss', prefix='losses')
 169 
 170   return g_loss

 173 def get_train_op_for_scope(loss, optimizer, scopes, params):
 174   """Train operation function for the given scope used file training."""
 175   is_trainable = lambda x: x in tf.trainable_variables()
 176 
 177   var_list = []
 178   update_ops = []
 179 
 180   for scope in scopes:
 181     var_list.extend(
 182         filter(is_trainable, tf.contrib.framework.get_model_variables(scope)))
 183     update_ops.extend(tf.get_collection(tf.GraphKeys.UPDATE_OPS, scope))
 184 
 185   return slim.learning.create_train_op(
 186       loss,
 187       optimizer,
 188       update_ops=update_ops,
 189       variables_to_train=var_list,
 190       clip_gradient_norm=params.clip_gradient_norm)

 193 def get_metrics(inputs, outputs, params):
 194   """Aggregate the metrics for rotator model.
 195 
 196   Args:
 197     inputs: Input dictionary of the rotator model.
 198     outputs: Output dictionary returned by the rotator model.
 199     params: Hyperparameters of the rotator model.
 200 
 201   Returns:
 202     names_to_values: metrics->values (dict).
 203     names_to_updates: metrics->ops (dict).
 204   """
 205   names_to_values = dict()
 206   names_to_updates = dict()
 207 
 208   tmp_values, tmp_updates = metrics.add_image_pred_metrics(
 209       inputs, outputs, params.num_views, 3*params.image_size**2)
 210   names_to_values.update(tmp_values)
 211   names_to_updates.update(tmp_updates)
 212 
 213   tmp_values, tmp_updates = metrics.add_mask_pred_metrics(
 214       inputs, outputs, params.num_views, params.image_size**2)
 215   names_to_values.update(tmp_values)
 216   names_to_updates.update(tmp_updates)
 217 
 218   for name, value in names_to_values.iteritems():
 219     slim.summaries.add_scalar_summary(
 220         value, name, prefix='eval', print_summary=True)
 221 
 222   return names_to_values, names_to_updates

 225 def write_disk_grid(global_step, summary_freq, log_dir, input_images,
 226                     output_images, pred_images, pred_masks):
 227   """Function called by TF to save the prediction periodically."""
 228 
 229   def write_grid(grid, global_step):
 230     """Native python function to call for writing images to files."""
 231     if global_step % summary_freq == 0:
 232       img_path = os.path.join(log_dir, '%s.jpg' % str(global_step))
 233       utils.save_image(grid, img_path)
 234     return 0
 235 
 236   grid = _build_image_grid(input_images, output_images, pred_images, pred_masks)
 237   slim.summaries.add_image_summary(
 238       tf.expand_dims(grid, axis=0), name='grid_vis')
 239   save_op = tf.py_func(write_grid, [grid, global_step], [tf.int64],
 240                        'write_grid')[0]
 241   return save_op

 244 def _build_image_grid(input_images, output_images, pred_images, pred_masks):
 245   """Builds a grid image by concatenating the input images."""
 246   quantity = input_images.get_shape().as_list()[0]
 247 
 248   for row in xrange(int(quantity / 4)):
 249     for col in xrange(4):
 250       index = row * 4 + col
 251       input_img_ = input_images[index, :, :, :]
 252       output_img_ = output_images[index, :, :, :]
 253       pred_img_ = pred_images[index, :, :, :]
 254       pred_mask_ = tf.tile(pred_masks[index, :, :, :], [1, 1, 3])
 255       if col == 0:
 256         tmp_ = tf.concat([input_img_, output_img_, pred_img_, pred_mask_],
 257                          1)  ## to the right
 258       else:
 259         tmp_ = tf.concat([tmp_, input_img_, output_img_, pred_img_, pred_mask_],
 260                          1)
 261     if row == 0:
 262       out_grid = tmp_
 263     else:
 264       out_grid = tf.concat([out_grid, tmp_], 0)
 265 
 266   return out_grid
