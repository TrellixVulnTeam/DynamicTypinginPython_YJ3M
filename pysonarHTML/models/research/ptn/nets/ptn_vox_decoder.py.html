  27 @tf.contrib.framework.add_arg_scope
  28 def conv3d_transpose(inputs,
  29                      num_outputs,
  30                      kernel_size,
  31                      stride=1,
  32                      padding='SAME',
  33                      activation_fn=tf.nn.relu,
  34                      weights_initializer=tf.contrib.layers.xavier_initializer(),
  35                      biases_initializer=tf.zeros_initializer(),
  36                      reuse=None,
  37                      trainable=True,
  38                      scope=None):
  39   """Wrapper for conv3d_transpose layer.
  40 
  41   This function wraps the tf.conv3d_transpose with basic non-linearity.
  42   Tt creates a variable called `weights`, representing the kernel,
  43   that is convoled with the input. A second varibale called `biases'
  44   is added to the result of operation.
  45   """
  46   with tf.variable_scope(
  47       scope, 'Conv3d_transpose', [inputs], reuse=reuse):
  48     dtype = inputs.dtype.base_dtype
  49     kernel_d, kernel_h, kernel_w = kernel_size[0:3]
  50     num_filters_in = inputs.get_shape()[4]
  51 
  52     weights_shape = [kernel_d, kernel_h, kernel_w, num_outputs, num_filters_in]
  53     weights = tf.get_variable('weights',
  54                               shape=weights_shape,
  55                               dtype=dtype,
  56                               initializer=weights_initializer,
  57                               trainable=trainable)
  58     tf.contrib.framework.add_model_variable(weights)
  59 
  60     input_shape = inputs.get_shape().as_list()
  61     batch_size = input_shape[0]
  62     depth = input_shape[1]
  63     height = input_shape[2]
  64     width = input_shape[3]
  65 
  66     def get_deconv_dim(dim_size, stride_size):
  67       # Only support padding='SAME'.
  68       if isinstance(dim_size, tf.Tensor):
  69         dim_size = tf.multiply(dim_size, stride_size)
  70       elif dim_size is not None:
  71         dim_size *= stride_size
  72       return dim_size
  73 
  74     out_depth = get_deconv_dim(depth, stride)
  75     out_height = get_deconv_dim(height, stride)
  76     out_width = get_deconv_dim(width, stride)
  77 
  78     out_shape = [batch_size, out_depth, out_height, out_width, num_outputs]
  79     outputs = tf.nn.conv3d_transpose(inputs, weights, out_shape,
  80                                      [1, stride, stride, stride, 1],
  81                                      padding=padding)
  82 
  83     outputs.set_shape(out_shape)
  84 
  85     if biases_initializer is not None:
  86       biases = tf.get_variable('biases',
  87                                shape=[num_outputs,],
  88                                dtype=dtype,
  89                                initializer=biases_initializer,
  90                                trainable=trainable)
  91       tf.contrib.framework.add_model_variable(biases)
  92       outputs = tf.nn.bias_add(outputs, biases)
  93 
  94     if activation_fn:
  95       outputs = activation_fn(outputs)
  96     return outputs
  97

  99 def model(identities, params, is_training):
 100   """Model transforming embedding to voxels."""
 101   del is_training  # Unused
 102   f_dim = params.f_dim
 103 
 104   # Please refer to the original implementation: github.com/xcyan/nips16_PTN
 105   # In TF replication, we use a slightly different architecture.
 106   with slim.arg_scope(
 107       [slim.fully_connected, conv3d_transpose],
 108       weights_initializer=tf.truncated_normal_initializer(stddev=0.02, seed=1)):
 109     h0 = slim.fully_connected(
 110         identities, 4 * 4 * 4 * f_dim * 8, activation_fn=tf.nn.relu)
 111     h1 = tf.reshape(h0, [-1, 4, 4, 4, f_dim * 8])
 112     h1 = conv3d_transpose(
 113         h1, f_dim * 4, [4, 4, 4], stride=2, activation_fn=tf.nn.relu)
 114     h2 = conv3d_transpose(
 115         h1, int(f_dim * 3 / 2), [5, 5, 5], stride=2, activation_fn=tf.nn.relu)
 116     h3 = conv3d_transpose(
 117         h2, 1, [6, 6, 6], stride=2, activation_fn=tf.nn.sigmoid)
 118   return h3
