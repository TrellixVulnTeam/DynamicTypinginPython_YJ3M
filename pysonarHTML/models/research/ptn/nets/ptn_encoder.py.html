  16 """Training/Pretraining encoder as used in PTN (NIPS16)."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import tensorflow as tf
  22 
  23 slim = tf.contrib.slim
  24 
  25 
  26 def _preprocess(images):
  27   return images * 2 - 1
  28 
  29 
  30 def model(images, params, is_training):
  31   """Model encoding the images into view-invariant embedding."""
  32   del is_training  # Unused
  33   image_size = images.get_shape().as_list()[1]
  34   f_dim = params.f_dim
  35   fc_dim = params.fc_dim
  36   z_dim = params.z_dim
  37   outputs = dict()
  38 
  39   images = _preprocess(images)
  40   with slim.arg_scope(
  41       [slim.conv2d, slim.fully_connected],
  42       weights_initializer=tf.truncated_normal_initializer(stddev=0.02, seed=1)):
  43     h0 = slim.conv2d(images, f_dim, [5, 5], stride=2, activation_fn=tf.nn.relu)
  44     h1 = slim.conv2d(h0, f_dim * 2, [5, 5], stride=2, activation_fn=tf.nn.relu)
  45     h2 = slim.conv2d(h1, f_dim * 4, [5, 5], stride=2, activation_fn=tf.nn.relu)
  46     # Reshape layer
  47     s8 = image_size // 8
  48     h2 = tf.reshape(h2, [-1, s8 * s8 * f_dim * 4])
  49     h3 = slim.fully_connected(h2, fc_dim, activation_fn=tf.nn.relu)
  50     h4 = slim.fully_connected(h3, fc_dim, activation_fn=tf.nn.relu)
  51 
  52     outputs['ids'] = slim.fully_connected(h4, z_dim, activation_fn=tf.nn.relu)
  53     outputs['poses'] = slim.fully_connected(h4, z_dim, activation_fn=tf.nn.relu)
  54   return outputs
