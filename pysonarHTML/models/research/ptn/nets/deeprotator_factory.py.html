  16 """Factory module for different encoder/decoder network models."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import tensorflow as tf
  22 
  23 from nets import ptn_encoder
  24 from nets import ptn_im_decoder
  25 from nets import ptn_rotator
  26 
  27 _NAME_TO_NETS = {
  28     'ptn_encoder': ptn_encoder,
  29     'ptn_rotator': ptn_rotator,
  30     'ptn_im_decoder': ptn_im_decoder,
  31 }
  32 
  33 
  34 def _get_network(name):
  35   """Gets a single network component."""
  36 
  37   if name not in _NAME_TO_NETS:
  38     raise ValueError('Network name [%s] not recognized.' % name)
  39   return _NAME_TO_NETS[name].model
  40 
  41 
  42 def get(params, is_training=False, reuse=False):
  43   """Factory function to retrieve a network model.
  44 
  45   Args:
  46     params: Different parameters used througout ptn, typically FLAGS (dict)
  47     is_training: Set to True if while training (boolean)
  48     reuse: Set as True if either using a pre-trained model or
  49       in the training loop while the graph has already been built (boolean)
  50   Returns:
  51     Model function for network (inputs to outputs)
  52   """
  53 
  54   def model(inputs):
  55     """Model function corresponding to a specific network architecture."""
  56     outputs = {}
  57 
  58     # First, build the encoder.
  59     encoder_fn = _get_network(params.encoder_name)
  60     with tf.variable_scope('encoder', reuse=reuse):
  61       # Produces id/pose units
  62       features = encoder_fn(inputs['images_0'], params, is_training)
  63       outputs['ids'] = features['ids']
  64       outputs['poses_0'] = features['poses']
  65 
  66     # Second, build the rotator and decoder.
  67     rotator_fn = _get_network(params.rotator_name)
  68     with tf.variable_scope('rotator', reuse=reuse):
  69       outputs['poses_1'] = rotator_fn(outputs['poses_0'], inputs['actions'],
  70                                       params, is_training)
  71     decoder_fn = _get_network(params.decoder_name)
  72     with tf.variable_scope('decoder', reuse=reuse):
  73       dec_output = decoder_fn(outputs['ids'], outputs['poses_1'], params,
  74                               is_training)
  75       outputs['images_1'] = dec_output['images']
  76       outputs['masks_1'] = dec_output['masks']
  77 
  78     # Third, build the recurrent connection
  79     for k in range(1, params.step_size):
  80       with tf.variable_scope('rotator', reuse=True):
  81         outputs['poses_%d' % (k + 1)] = rotator_fn(
  82             outputs['poses_%d' % k], inputs['actions'], params, is_training)
  83       with tf.variable_scope('decoder', reuse=True):
  84         dec_output = decoder_fn(outputs['ids'], outputs['poses_%d' % (k + 1)],
  85                                 params, is_training)
  86         outputs['images_%d' % (k + 1)] = dec_output['images']
  87         outputs['masks_%d' % (k + 1)] = dec_output['masks']
  88 
  89     return outputs
  90 
  91   return model
