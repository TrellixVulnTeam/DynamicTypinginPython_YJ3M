  16 """Perspective Transformer Layer Implementation.
  17 
  18 Transform the volume based on 4 x 4 perspective projection matrix.
  19 
  20 Reference:
  21 (1) "Perspective Transformer Nets: Perspective Transformer Nets:
  22 Learning Single-View 3D Object Reconstruction without 3D Supervision."
  23 Xinchen Yan, Jimei Yang, Ersin Yumer, Yijie Guo, Honglak Lee. In NIPS 2016
  24 https://papers.nips.cc/paper/6206-perspective-transformer-nets-learning-single-view-3d-object-reconstruction-without-3d-supervision.pdf
  25 
  26 (2) Official implementation in Torch: https://github.com/xcyan/ptnbhwd
  27 
  28 (3) 2D Transformer implementation in TF:
  29 github.com/tensorflow/models/tree/master/research/transformer
  30 
  31 """
  32 
  33 import tensorflow as tf
  34 
  35 
  36 def transformer(voxels,
  37                 theta,
  38                 out_size,
  39                 z_near,
  40                 z_far,
  41                 name='PerspectiveTransformer'):
  42   """Perspective Transformer Layer.
  43 
  44   Args:
  45     voxels: A tensor of size [num_batch, depth, height, width, num_channels].
  46       It is the output of a deconv/upsampling conv network (tf.float32).
  47     theta: A tensor of size [num_batch, 16].
  48       It is the inverse camera transformation matrix (tf.float32).
  49     out_size: A tuple representing the size of output of
  50       transformer layer (float).
  51     z_near: A number representing the near clipping plane (float).
  52     z_far: A number representing the far clipping plane (float).
  53 
  54   Returns:
  55     A transformed tensor (tf.float32).
  56 
  57   """
  58   def _repeat(x, n_repeats):
  59     with tf.variable_scope('_repeat'):
  60       rep = tf.transpose(
  61           tf.expand_dims(tf.ones(shape=tf.stack([
  62               n_repeats,
  63           ])), 1), [1, 0])
  64       rep = tf.to_int32(rep)
  65       x = tf.matmul(tf.reshape(x, (-1, 1)), rep)
  66       return tf.reshape(x, [-1])
  67 
  68   def _interpolate(im, x, y, z, out_size):
  69     """Bilinear interploation layer.
  70 
  71     Args:
  72       im: A 5D tensor of size [num_batch, depth, height, width, num_channels].
  73         It is the input volume for the transformation layer (tf.float32).
  74       x: A tensor of size [num_batch, out_depth, out_height, out_width]
  75         representing the inverse coordinate mapping for x (tf.float32).
  76       y: A tensor of size [num_batch, out_depth, out_height, out_width]
  77         representing the inverse coordinate mapping for y (tf.float32).
  78       z: A tensor of size [num_batch, out_depth, out_height, out_width]
  79         representing the inverse coordinate mapping for z (tf.float32).
  80       out_size: A tuple representing the output size of transformation layer
  81         (float).
  82 
  83     Returns:
  84       A transformed tensor (tf.float32).
  85 
  86     """
  87     with tf.variable_scope('_interpolate'):
  88       num_batch = im.get_shape().as_list()[0]
  89       depth = im.get_shape().as_list()[1]
  90       height = im.get_shape().as_list()[2]
  91       width = im.get_shape().as_list()[3]
  92       channels = im.get_shape().as_list()[4]
  93 
  94       x = tf.to_float(x)
  95       y = tf.to_float(y)
  96       z = tf.to_float(z)
  97       depth_f = tf.to_float(depth)
  98       height_f = tf.to_float(height)
  99       width_f = tf.to_float(width)
 100       # Number of disparity interpolated.
 101       out_depth = out_size[0]
 102       out_height = out_size[1]
 103       out_width = out_size[2]
 104       zero = tf.zeros([], dtype='int32')
 105       # 0 <= z < depth, 0 <= y < height & 0 <= x < width.
 106       max_z = tf.to_int32(tf.shape(im)[1] - 1)
 107       max_y = tf.to_int32(tf.shape(im)[2] - 1)
 108       max_x = tf.to_int32(tf.shape(im)[3] - 1)
 109 
 110       # Converts scale indices from [-1, 1] to [0, width/height/depth].
 111       x = (x + 1.0) * (width_f) / 2.0
 112       y = (y + 1.0) * (height_f) / 2.0
 113       z = (z + 1.0) * (depth_f) / 2.0
 114 
 115       x0 = tf.to_int32(tf.floor(x))
 116       x1 = x0 + 1
 117       y0 = tf.to_int32(tf.floor(y))
 118       y1 = y0 + 1
 119       z0 = tf.to_int32(tf.floor(z))
 120       z1 = z0 + 1
 121 
 122       x0_clip = tf.clip_by_value(x0, zero, max_x)
 123       x1_clip = tf.clip_by_value(x1, zero, max_x)
 124       y0_clip = tf.clip_by_value(y0, zero, max_y)
 125       y1_clip = tf.clip_by_value(y1, zero, max_y)
 126       z0_clip = tf.clip_by_value(z0, zero, max_z)
 127       z1_clip = tf.clip_by_value(z1, zero, max_z)
 128       dim3 = width
 129       dim2 = width * height
 130       dim1 = width * height * depth
 131       base = _repeat(
 132           tf.range(num_batch) * dim1, out_depth * out_height * out_width)
 133       base_z0_y0 = base + z0_clip * dim2 + y0_clip * dim3
 134       base_z0_y1 = base + z0_clip * dim2 + y1_clip * dim3
 135       base_z1_y0 = base + z1_clip * dim2 + y0_clip * dim3
 136       base_z1_y1 = base + z1_clip * dim2 + y1_clip * dim3
 137 
 138       idx_z0_y0_x0 = base_z0_y0 + x0_clip
 139       idx_z0_y0_x1 = base_z0_y0 + x1_clip
 140       idx_z0_y1_x0 = base_z0_y1 + x0_clip
 141       idx_z0_y1_x1 = base_z0_y1 + x1_clip
 142       idx_z1_y0_x0 = base_z1_y0 + x0_clip
 143       idx_z1_y0_x1 = base_z1_y0 + x1_clip
 144       idx_z1_y1_x0 = base_z1_y1 + x0_clip
 145       idx_z1_y1_x1 = base_z1_y1 + x1_clip
 146 
 147       # Use indices to lookup pixels in the flat image and restore
 148       # channels dim
 149       im_flat = tf.reshape(im, tf.stack([-1, channels]))
 150       im_flat = tf.to_float(im_flat)
 151       i_z0_y0_x0 = tf.gather(im_flat, idx_z0_y0_x0)
 152       i_z0_y0_x1 = tf.gather(im_flat, idx_z0_y0_x1)
 153       i_z0_y1_x0 = tf.gather(im_flat, idx_z0_y1_x0)
 154       i_z0_y1_x1 = tf.gather(im_flat, idx_z0_y1_x1)
 155       i_z1_y0_x0 = tf.gather(im_flat, idx_z1_y0_x0)
 156       i_z1_y0_x1 = tf.gather(im_flat, idx_z1_y0_x1)
 157       i_z1_y1_x0 = tf.gather(im_flat, idx_z1_y1_x0)
 158       i_z1_y1_x1 = tf.gather(im_flat, idx_z1_y1_x1)
 159 
 160       # Finally calculate interpolated values.
 161       x0_f = tf.to_float(x0)
 162       x1_f = tf.to_float(x1)
 163       y0_f = tf.to_float(y0)
 164       y1_f = tf.to_float(y1)
 165       z0_f = tf.to_float(z0)
 166       z1_f = tf.to_float(z1)
 167       # Check the out-of-boundary case.
 168       x0_valid = tf.to_float(
 169           tf.less_equal(x0, max_x) & tf.greater_equal(x0, 0))
 170       x1_valid = tf.to_float(
 171           tf.less_equal(x1, max_x) & tf.greater_equal(x1, 0))
 172       y0_valid = tf.to_float(
 173           tf.less_equal(y0, max_y) & tf.greater_equal(y0, 0))
 174       y1_valid = tf.to_float(
 175           tf.less_equal(y1, max_y) & tf.greater_equal(y1, 0))
 176       z0_valid = tf.to_float(
 177           tf.less_equal(z0, max_z) & tf.greater_equal(z0, 0))
 178       z1_valid = tf.to_float(
 179           tf.less_equal(z1, max_z) & tf.greater_equal(z1, 0))
 180 
 181       w_z0_y0_x0 = tf.expand_dims(((x1_f - x) * (y1_f - y) *
 182                                    (z1_f - z) * x1_valid * y1_valid * z1_valid),
 183                                   1)
 184       w_z0_y0_x1 = tf.expand_dims(((x - x0_f) * (y1_f - y) *
 185                                    (z1_f - z) * x0_valid * y1_valid * z1_valid),
 186                                   1)
 187       w_z0_y1_x0 = tf.expand_dims(((x1_f - x) * (y - y0_f) *
 188                                    (z1_f - z) * x1_valid * y0_valid * z1_valid),
 189                                   1)
 190       w_z0_y1_x1 = tf.expand_dims(((x - x0_f) * (y - y0_f) *
 191                                    (z1_f - z) * x0_valid * y0_valid * z1_valid),
 192                                   1)
 193       w_z1_y0_x0 = tf.expand_dims(((x1_f - x) * (y1_f - y) *
 194                                    (z - z0_f) * x1_valid * y1_valid * z0_valid),
 195                                   1)
 196       w_z1_y0_x1 = tf.expand_dims(((x - x0_f) * (y1_f - y) *
 197                                    (z - z0_f) * x0_valid * y1_valid * z0_valid),
 198                                   1)
 199       w_z1_y1_x0 = tf.expand_dims(((x1_f - x) * (y - y0_f) *
 200                                    (z - z0_f) * x1_valid * y0_valid * z0_valid),
 201                                   1)
 202       w_z1_y1_x1 = tf.expand_dims(((x - x0_f) * (y - y0_f) *
 203                                    (z - z0_f) * x0_valid * y0_valid * z0_valid),
 204                                   1)
 205 
 206       output = tf.add_n([
 207           w_z0_y0_x0 * i_z0_y0_x0, w_z0_y0_x1 * i_z0_y0_x1,
 208           w_z0_y1_x0 * i_z0_y1_x0, w_z0_y1_x1 * i_z0_y1_x1,
 209           w_z1_y0_x0 * i_z1_y0_x0, w_z1_y0_x1 * i_z1_y0_x1,
 210           w_z1_y1_x0 * i_z1_y1_x0, w_z1_y1_x1 * i_z1_y1_x1
 211       ])
 212       return output
 213 
 214   def _meshgrid(depth, height, width, z_near, z_far):
 215     with tf.variable_scope('_meshgrid'):
 216       x_t = tf.reshape(
 217           tf.tile(tf.linspace(-1.0, 1.0, width), [height * depth]),
 218           [depth, height, width])
 219       y_t = tf.reshape(
 220           tf.tile(tf.linspace(-1.0, 1.0, height), [width * depth]),
 221           [depth, width, height])
 222       y_t = tf.transpose(y_t, [0, 2, 1])
 223       sample_grid = tf.tile(
 224           tf.linspace(float(z_near), float(z_far), depth), [width * height])
 225       z_t = tf.reshape(sample_grid, [height, width, depth])
 226       z_t = tf.transpose(z_t, [2, 0, 1])
 227 
 228       z_t = 1 / z_t
 229       d_t = 1 / z_t
 230       x_t /= z_t
 231       y_t /= z_t
 232 
 233       x_t_flat = tf.reshape(x_t, (1, -1))
 234       y_t_flat = tf.reshape(y_t, (1, -1))
 235       d_t_flat = tf.reshape(d_t, (1, -1))
 236 
 237       ones = tf.ones_like(x_t_flat)
 238       grid = tf.concat([d_t_flat, y_t_flat, x_t_flat, ones], 0)
 239       return grid
 240 
 241   def _transform(theta, input_dim, out_size, z_near, z_far):
 242     with tf.variable_scope('_transform'):
 243       num_batch = input_dim.get_shape().as_list()[0]
 244       num_channels = input_dim.get_shape().as_list()[4]
 245       theta = tf.reshape(theta, (-1, 4, 4))
 246       theta = tf.cast(theta, 'float32')
 247 
 248       out_depth = out_size[0]
 249       out_height = out_size[1]
 250       out_width = out_size[2]
 251       grid = _meshgrid(out_depth, out_height, out_width, z_near, z_far)
 252       grid = tf.expand_dims(grid, 0)
 253       grid = tf.reshape(grid, [-1])
 254       grid = tf.tile(grid, tf.stack([num_batch]))
 255       grid = tf.reshape(grid, tf.stack([num_batch, 4, -1]))
 256 
 257       # Transform A x (x_t', y_t', 1, d_t)^T -> (x_s, y_s, z_s, 1).
 258       t_g = tf.matmul(theta, grid)
 259       z_s = tf.slice(t_g, [0, 0, 0], [-1, 1, -1])
 260       y_s = tf.slice(t_g, [0, 1, 0], [-1, 1, -1])
 261       x_s = tf.slice(t_g, [0, 2, 0], [-1, 1, -1])
 262 
 263       z_s_flat = tf.reshape(z_s, [-1])
 264       y_s_flat = tf.reshape(y_s, [-1])
 265       x_s_flat = tf.reshape(x_s, [-1])
 266 
 267       input_transformed = _interpolate(input_dim, x_s_flat, y_s_flat, z_s_flat,
 268                                        out_size)
 269 
 270       output = tf.reshape(
 271           input_transformed,
 272           tf.stack([num_batch, out_depth, out_height, out_width, num_channels]))
 273 
 274       return output
 275 
 276   with tf.variable_scope(name):
 277     output = _transform(theta, voxels, out_size, z_near, z_far)
 278     return output
