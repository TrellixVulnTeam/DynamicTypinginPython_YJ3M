  27 def add_rotator_image_loss(inputs, outputs, step_size, weight_scale):
  28   """Computes the image loss of deep rotator model.
  29 
  30   Args:
  31     inputs: Input dictionary to the model containing keys
  32       such as `images_k'.
  33     outputs: Output dictionary returned by the model containing keys
  34       such as `images_k'.
  35     step_size: A scalar representing the number of recurrent
  36       steps (number of repeated out-of-plane rotations)
  37       in the deep rotator network (int).
  38     weight_scale: A reweighting factor applied over the image loss (float).
  39 
  40   Returns:
  41     A `Tensor' scalar that returns averaged L2 loss
  42       (divided by batch_size and step_size) between the
  43       ground-truth images (RGB) and predicted images (tf.float32).
  44 
  45   """
  46   batch_size = tf.shape(inputs['images_0'])[0]
  47   image_loss = 0
  48   for k in range(1, step_size + 1):
  49     image_loss += tf.nn.l2_loss(
  50         inputs['images_%d' % k] - outputs['images_%d' % k])
  51 
  52   image_loss /= tf.to_float(step_size * batch_size)
  53   slim.summaries.add_scalar_summary(
  54       image_loss, 'image_loss', prefix='losses')
  55   image_loss *= weight_scale
  56   return image_loss

  91 def add_volume_proj_loss(inputs, outputs, num_views, weight_scale):
  92   """Computes the projection loss of voxel generation model.
  93 
  94   Args:
  95     inputs: Input dictionary to the model containing keys such as
  96       `images_1'.
  97     outputs: Output dictionary returned by the model containing keys
  98       such as `masks_k' and ``projs_k'.
  99     num_views: A integer scalar represents the total number of
 100       viewpoints for each of the object (int).
 101     weight_scale: A reweighting factor applied over the projection loss (float).
 102 
 103   Returns:
 104     A `Tensor' that returns the averaged L2 loss
 105       (divided by batch_size and num_views) between the ground-truth
 106       masks (object silhouettes) and predicted masks (tf.float32).
 107 
 108   """
 109   batch_size = tf.shape(inputs['images_1'])[0]
 110   proj_loss = 0
 111   for k in range(num_views):
 112     proj_loss += tf.nn.l2_loss(
 113         outputs['masks_%d' % (k + 1)] - outputs['projs_%d' % (k + 1)])
 114   proj_loss /= tf.to_float(num_views * batch_size)
 115   slim.summaries.add_scalar_summary(
 116       proj_loss, 'proj_loss', prefix='losses')
 117   proj_loss *= weight_scale
 118   return proj_loss

 121 def add_volume_loss(inputs, outputs, num_views, weight_scale):
 122   """Computes the volume loss of voxel generation model.
 123 
 124   Args:
 125     inputs: Input dictionary to the model containing keys such as
 126       `images_1' and `voxels'.
 127     outputs: Output dictionary returned by the model containing keys
 128       such as `voxels_k'.
 129     num_views: A scalar representing the total number of
 130       viewpoints for each object (int).
 131     weight_scale: A reweighting factor applied over the volume
 132       loss (tf.float32).
 133 
 134   Returns:
 135     A `Tensor' that returns the averaged L2 loss
 136       (divided by batch_size and num_views) between the ground-truth
 137       volumes and predicted volumes (tf.float32).
 138 
 139   """
 140   batch_size = tf.shape(inputs['images_1'])[0]
 141   vol_loss = 0
 142   for k in range(num_views):
 143     vol_loss += tf.nn.l2_loss(
 144         inputs['voxels'] - outputs['voxels_%d' % (k + 1)])
 145   vol_loss /= tf.to_float(num_views * batch_size)
 146   slim.summaries.add_scalar_summary(
 147       vol_loss, 'vol_loss', prefix='losses')
 148   vol_loss *= weight_scale
 149   return vol_loss

 152 def regularization_loss(scopes, params):
 153   """Computes the weight decay as regularization during training.
 154 
 155   Args:
 156     scopes: A list of different components of the model such as
 157       ``encoder'', ``decoder'' and ``projector''.
 158     params: Parameters of the model.
 159 
 160   Returns:
 161     Regularization loss (tf.float32).
 162   """
 163 
 164   reg_loss = tf.zeros(dtype=tf.float32, shape=[])
 165   if params.weight_decay > 0:
 166     is_trainable = lambda x: x in tf.trainable_variables()
 167     is_weights = lambda x: 'weights' in x.name
 168     for scope in scopes:
 169       scope_vars = filter(is_trainable,
 170                           tf.contrib.framework.get_model_variables(scope))
 171       scope_vars = filter(is_weights, scope_vars)
 172       if scope_vars:
 173         reg_loss += tf.add_n([tf.nn.l2_loss(var) for var in scope_vars])
 174 
 175   slim.summaries.add_scalar_summary(
 176       reg_loss, 'reg_loss', prefix='losses')
 177   reg_loss *= params.weight_decay
 178   return reg_loss
