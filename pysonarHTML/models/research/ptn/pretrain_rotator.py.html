  31 flags = tf.app.flags
  32 slim = tf.contrib.slim
  33 
  34 flags.DEFINE_string('inp_dir', '',
  35                     'Directory path containing the input data (tfrecords).')
  36 flags.DEFINE_string(
  37     'dataset_name', 'shapenet_chair',
  38     'Dataset name that is to be used for training and evaluation.')
  39 flags.DEFINE_integer('z_dim', 512, '')
  40 flags.DEFINE_integer('a_dim', 3, '')
  41 flags.DEFINE_integer('f_dim', 64, '')
  42 flags.DEFINE_integer('fc_dim', 1024, '')
  43 flags.DEFINE_integer('num_views', 24, 'Num of viewpoints in the input data.')
  44 flags.DEFINE_integer('image_size', 64,
  45                      'Input images dimension (pixels) - width & height.')
  46 flags.DEFINE_integer('step_size', 1, 'Steps to take for rotation in pretraining.')
  47 flags.DEFINE_integer('batch_size', 32, 'Batch size for training.')
  48 flags.DEFINE_string('encoder_name', 'ptn_encoder',
  49                     'Name of the encoder network being used.')
  50 flags.DEFINE_string('decoder_name', 'ptn_im_decoder',
  51                     'Name of the decoder network being used.')
  52 flags.DEFINE_string('rotator_name', 'ptn_rotator',
  53                     'Name of the rotator network being used.')
  54 # Save options
  55 flags.DEFINE_string('checkpoint_dir', '/tmp/ptn_train/',
  56                     'Directory path for saving trained models and other data.')
  57 flags.DEFINE_string('model_name', 'deeprotator_pretrain',
  58                     'Name of the model used in naming the TF job. Must be different for each run.')
  59 flags.DEFINE_string('init_model', None,
  60                     'Checkpoint path of the model to initialize with.')
  61 flags.DEFINE_integer('save_every', 1000,
  62                      'Average period of steps after which we save a model.')
  63 # Optimization
  64 flags.DEFINE_float('image_weight', 10, 'Weighting factor for image loss.')
  65 flags.DEFINE_float('mask_weight', 1, 'Weighting factor for mask loss.')
  66 flags.DEFINE_float('learning_rate', 0.0001, 'Learning rate.')
  67 flags.DEFINE_float('weight_decay', 0.001, 'Weight decay parameter while training.')
  68 flags.DEFINE_float('clip_gradient_norm', 0, 'Gradient clim norm, leave 0 if no gradient clipping.')
  69 flags.DEFINE_integer('max_number_of_steps', 320000, 'Maximum number of steps for training.')
  70 # Summary
  71 flags.DEFINE_integer('save_summaries_secs', 15, 'Seconds interval for dumping TF summaries.')
  72 flags.DEFINE_integer('save_interval_secs', 60 * 5, 'Seconds interval to save models.')
  73 # Distribution
  74 flags.DEFINE_string('master', '', 'The address of the tensorflow master if running distributed.')
  75 flags.DEFINE_bool('sync_replicas', False, 'Whether to sync gradients between replicas for optimizer.')
  76 flags.DEFINE_integer('worker_replicas', 1, 'Number of worker replicas (train tasks).')
  77 flags.DEFINE_integer('backup_workers', 0, 'Number of backup workers.')
  78 flags.DEFINE_integer('ps_tasks', 0, 'Number of ps tasks.')
  79 flags.DEFINE_integer('task', 0,
  80                      'Task identifier flag to be set for each task running in distributed manner. Task number 0 '
  81                      'will be chosen as the chief.')
  82 
  83 FLAGS = flags.FLAGS

  86 def main(_):
  87   train_dir = os.path.join(FLAGS.checkpoint_dir, FLAGS.model_name, 'train')
  88   save_image_dir = os.path.join(train_dir, 'images')
  89   if not os.path.exists(train_dir):
  90     os.makedirs(train_dir)
  91   if not os.path.exists(save_image_dir):
  92     os.makedirs(save_image_dir)
  93 
  94   g = tf.Graph()
  95   with g.as_default():
  96     with tf.device(tf.train.replica_device_setter(FLAGS.ps_tasks)):
  97       global_step = slim.get_or_create_global_step()
  98       ##########
  99       ## data ##
 100       ##########
 101       train_data = model.get_inputs(
 102           FLAGS.inp_dir,
 103           FLAGS.dataset_name,
 104           'train',
 105           FLAGS.batch_size,
 106           FLAGS.image_size,
 107           is_training=True)
 108       inputs = model.preprocess(train_data, FLAGS.step_size)
 109       ###########
 110       ## model ##
 111       ###########
 112       model_fn = model.get_model_fn(FLAGS, is_training=True)
 113       outputs = model_fn(inputs)
 114       ##########
 115       ## loss ##
 116       ##########
 117       task_loss = model.get_loss(inputs, outputs, FLAGS)
 118       regularization_loss = model.get_regularization_loss(
 119           ['encoder', 'rotator', 'decoder'], FLAGS)
 120       loss = task_loss + regularization_loss
 121       ###############
 122       ## optimizer ##
 123       ###############
 124       optimizer = tf.train.AdamOptimizer(FLAGS.learning_rate)
 125       if FLAGS.sync_replicas:
 126         optimizer = tf.train.SyncReplicasOptimizer(
 127             optimizer,
 128             replicas_to_aggregate=FLAGS.workers_replicas - FLAGS.backup_workers,
 129             total_num_replicas=FLAGS.worker_replicas)
 130 
 131       ##############
 132       ## train_op ##
 133       ##############
 134       train_op = model.get_train_op_for_scope(
 135           loss, optimizer, ['encoder', 'rotator', 'decoder'], FLAGS)
 136       ###########
 137       ## saver ##
 138       ###########
 139       saver = tf.train.Saver(max_to_keep=np.minimum(5,
 140                                                     FLAGS.worker_replicas + 1))
 141 
 142       if FLAGS.task == 0:
 143         val_data = model.get_inputs(
 144             FLAGS.inp_dir,
 145             FLAGS.dataset_name,
 146             'val',
 147             FLAGS.batch_size,
 148             FLAGS.image_size,
 149             is_training=False)
 150         val_inputs = model.preprocess(val_data, FLAGS.step_size)
 151         # Note: don't compute loss here
 152         reused_model_fn = model.get_model_fn(
 153             FLAGS, is_training=False, reuse=True)
 154         val_outputs = reused_model_fn(val_inputs)
 155         with tf.device(tf.DeviceSpec(device_type='CPU')):
 156           if FLAGS.step_size == 1:
 157             vis_input_images = val_inputs['images_0'] * 255.0
 158             vis_output_images = val_inputs['images_1'] * 255.0
 159             vis_pred_images = val_outputs['images_1'] * 255.0
 160             vis_pred_masks = (val_outputs['masks_1'] * (-1) + 1) * 255.0
 161           else:
 162             rep_times = int(np.ceil(32.0 / float(FLAGS.step_size)))
 163             vis_list_1 = []
 164             vis_list_2 = []
 165             vis_list_3 = []
 166             vis_list_4 = []
 167             for j in xrange(rep_times):
 168               for k in xrange(FLAGS.step_size):
 169                 vis_input_image = val_inputs['images_0'][j],
 170                 vis_output_image = val_inputs['images_%d' % (k + 1)][j]
 171                 vis_pred_image = val_outputs['images_%d' % (k + 1)][j]
 172                 vis_pred_mask = val_outputs['masks_%d' % (k + 1)][j]
 173                 vis_list_1.append(tf.expand_dims(vis_input_image, 0))
 174                 vis_list_2.append(tf.expand_dims(vis_output_image, 0))
 175                 vis_list_3.append(tf.expand_dims(vis_pred_image, 0))
 176                 vis_list_4.append(tf.expand_dims(vis_pred_mask, 0))
 177 
 178             vis_list_1 = tf.reshape(
 179                 tf.stack(vis_list_1), [
 180                     rep_times * FLAGS.step_size, FLAGS.image_size,
 181                     FLAGS.image_size, 3
 182                 ])
 183             vis_list_2 = tf.reshape(
 184                 tf.stack(vis_list_2), [
 185                     rep_times * FLAGS.step_size, FLAGS.image_size,
 186                     FLAGS.image_size, 3
 187                 ])
 188             vis_list_3 = tf.reshape(
 189                 tf.stack(vis_list_3), [
 190                     rep_times * FLAGS.step_size, FLAGS.image_size,
 191                     FLAGS.image_size, 3
 192                 ])
 193             vis_list_4 = tf.reshape(
 194                 tf.stack(vis_list_4), [
 195                     rep_times * FLAGS.step_size, FLAGS.image_size,
 196                     FLAGS.image_size, 1
 197                 ])
 198 
 199             vis_input_images = vis_list_1 * 255.0
 200             vis_output_images = vis_list_2 * 255.0
 201             vis_pred_images = vis_list_3 * 255.0
 202             vis_pred_masks = (vis_list_4 * (-1) + 1) * 255.0
 203 
 204           write_disk_op = model.write_disk_grid(
 205               global_step=global_step,
 206               summary_freq=FLAGS.save_every,
 207               log_dir=save_image_dir,
 208               input_images=vis_input_images,
 209               output_images=vis_output_images,
 210               pred_images=vis_pred_images,
 211               pred_masks=vis_pred_masks)
 212         with tf.control_dependencies([write_disk_op]):
 213           train_op = tf.identity(train_op)
 214 
 215       #############
 216       ## init_fn ##
 217       #############
 218       init_fn = model.get_init_fn(['encoder, ' 'rotator', 'decoder'], FLAGS)
 219 
 220       ##############
 221       ## training ##
 222       ##############
 223       slim.learning.train(
 224           train_op=train_op,
 225           logdir=train_dir,
 226           init_fn=init_fn,
 227           master=FLAGS.master,
 228           is_chief=(FLAGS.task == 0),
 229           number_of_steps=FLAGS.max_number_of_steps,
 230           saver=saver,
 231           save_summaries_secs=FLAGS.save_summaries_secs,
 232           save_interval_secs=FLAGS.save_interval_secs)
 233 
 234 
 235 if __name__ == '__main__':
 236   app.run()
