  17 from __future__ import absolute_import
  18 from __future__ import division

  30 flags = tf.app.flags
  31 slim = tf.contrib.slim
  32 
  33 flags.DEFINE_string('inp_dir',
  34                     '',
  35                     'Directory path containing the input data (tfrecords).')
  36 flags.DEFINE_string(
  37     'dataset_name', 'shapenet_chair',
  38     'Dataset name that is to be used for training and evaluation.')
  39 flags.DEFINE_integer('z_dim', 512, '')
  40 flags.DEFINE_integer('f_dim', 64, '')
  41 flags.DEFINE_integer('fc_dim', 1024, '')
  42 flags.DEFINE_integer('num_views', 24, 'Num of viewpoints in the input data.')
  43 flags.DEFINE_integer('image_size', 64,
  44                      'Input images dimension (pixels) - width & height.')
  45 flags.DEFINE_integer('vox_size', 32, 'Voxel prediction dimension.')
  46 flags.DEFINE_integer('step_size', 24, 'Steps to take in rotation to fetch viewpoints.')
  47 flags.DEFINE_integer('batch_size', 6, 'Batch size while training.')
  48 flags.DEFINE_float('focal_length', 0.866, 'Focal length parameter used in perspective projection.')
  49 flags.DEFINE_float('focal_range', 1.732, 'Focal length parameter used in perspective projection.')
  50 flags.DEFINE_string('encoder_name', 'ptn_encoder',
  51                     'Name of the encoder network being used.')
  52 flags.DEFINE_string('decoder_name', 'ptn_vox_decoder',
  53                     'Name of the decoder network being used.')
  54 flags.DEFINE_string('projector_name', 'perspective_projector',
  55                     'Name of the projector network being used.')
  56 # Save options
  57 flags.DEFINE_string('checkpoint_dir', '/tmp/ptn_train/',
  58                     'Directory path for saving trained models and other data.')
  59 flags.DEFINE_string('model_name', 'ptn_finetune',
  60                     'Name of the model used in naming the TF job. Must be different for each run.')
  61 flags.DEFINE_string('init_model', None,
  62                     'Checkpoint path of the model to initialize with.')
  63 flags.DEFINE_integer('save_every', 1000,
  64                      'Average period of steps after which we save a model.')
  65 # Optimization
  66 flags.DEFINE_float('proj_weight', 10, 'Weighting factor for projection loss.')
  67 flags.DEFINE_float('volume_weight', 0, 'Weighting factor for volume loss.')
  68 flags.DEFINE_float('viewpoint_weight', 1, 'Weighting factor for viewpoint loss.')
  69 flags.DEFINE_float('learning_rate', 0.0001, 'Learning rate.')
  70 flags.DEFINE_float('weight_decay', 0.001, 'Weight decay parameter while training.')
  71 flags.DEFINE_float('clip_gradient_norm', 0, 'Gradient clim norm, leave 0 if no gradient clipping.')
  72 flags.DEFINE_integer('max_number_of_steps', 10000, 'Maximum number of steps for training.')
  73 # Summary
  74 flags.DEFINE_integer('save_summaries_secs', 15, 'Seconds interval for dumping TF summaries.')
  75 flags.DEFINE_integer('save_interval_secs', 60 * 5, 'Seconds interval to save models.')
  76 
  77 # Scheduling
  78 flags.DEFINE_string('master', '', 'The address of the tensorflow master')
  79 flags.DEFINE_bool('sync_replicas', False, 'Whether to sync gradients between replicas for optimizer.')
  80 flags.DEFINE_integer('worker_replicas', 1, 'Number of worker replicas (train tasks).')
  81 flags.DEFINE_integer('backup_workers', 0, 'Number of backup workers.')
  82 flags.DEFINE_integer('ps_tasks', 0, 'Number of ps tasks.')
  83 flags.DEFINE_integer('task', 0,
  84                      'Task identifier flag to be set for each task running in distributed manner. Task number 0 '
  85                      'will be chosen as the chief.')
  86 
  87 FLAGS = flags.FLAGS

  90 def main(_):
  91   train_dir = os.path.join(FLAGS.checkpoint_dir, FLAGS.model_name, 'train')
  92   save_image_dir = os.path.join(train_dir, 'images')
  93   if not os.path.exists(train_dir):
  94     os.makedirs(train_dir)
  95   if not os.path.exists(save_image_dir):
  96     os.makedirs(save_image_dir)
  97 
  98   g = tf.Graph()
  99   with g.as_default():
 100     with tf.device(tf.train.replica_device_setter(FLAGS.ps_tasks)):
 101       global_step = slim.get_or_create_global_step()
 102       ###########
 103       ## model ##
 104       ###########
 105       model = model_ptn.model_PTN(FLAGS)
 106       ##########
 107       ## data ##
 108       ##########
 109       train_data = model.get_inputs(
 110           FLAGS.inp_dir,
 111           FLAGS.dataset_name,
 112           'train',
 113           FLAGS.batch_size,
 114           FLAGS.image_size,
 115           FLAGS.vox_size,
 116           is_training=True)
 117       inputs = model.preprocess(train_data, FLAGS.step_size)
 118       ##############
 119       ## model_fn ##
 120       ##############
 121       model_fn = model.get_model_fn(
 122           is_training=True, reuse=False, run_projection=True)
 123       outputs = model_fn(inputs)
 124       ##################
 125       ## train_scopes ##
 126       ##################
 127       if FLAGS.init_model:
 128         train_scopes = ['decoder']
 129         init_scopes = ['encoder']
 130       else:
 131         train_scopes = ['encoder', 'decoder']
 132 
 133       ##########
 134       ## loss ##
 135       ##########
 136       task_loss = model.get_loss(inputs, outputs)
 137 
 138       regularization_loss = model.get_regularization_loss(train_scopes)
 139       loss = task_loss + regularization_loss
 140       ###############
 141       ## optimizer ##
 142       ###############
 143       optimizer = tf.train.AdamOptimizer(FLAGS.learning_rate)
 144       if FLAGS.sync_replicas:
 145         optimizer = tf.train.SyncReplicasOptimizer(
 146             optimizer,
 147             replicas_to_aggregate=FLAGS.workers_replicas - FLAGS.backup_workers,
 148             total_num_replicas=FLAGS.worker_replicas)
 149 
 150       ##############
 151       ## train_op ##
 152       ##############
 153       train_op = model.get_train_op_for_scope(loss, optimizer, train_scopes)
 154       ###########
 155       ## saver ##
 156       ###########
 157       saver = tf.train.Saver(max_to_keep=np.minimum(5,
 158                                                     FLAGS.worker_replicas + 1))
 159 
 160       if FLAGS.task == 0:
 161         params = FLAGS
 162         params.batch_size = params.num_views
 163         params.step_size = 1
 164         model.set_params(params)
 165         val_data = model.get_inputs(
 166             params.inp_dir,
 167             params.dataset_name,
 168             'val',
 169             params.batch_size,
 170             params.image_size,
 171             params.vox_size,
 172             is_training=False)
 173         val_inputs = model.preprocess(val_data, params.step_size)
 174         # Note: don't compute loss here
 175         reused_model_fn = model.get_model_fn(is_training=False, reuse=True)
 176         val_outputs = reused_model_fn(val_inputs)
 177 
 178         with tf.device(tf.DeviceSpec(device_type='CPU')):
 179           vis_input_images = val_inputs['images_1'] * 255.0
 180           vis_gt_projs = (val_outputs['masks_1'] * (-1) + 1) * 255.0
 181           vis_pred_projs = (val_outputs['projs_1'] * (-1) + 1) * 255.0
 182 
 183           vis_gt_projs = tf.concat([vis_gt_projs] * 3, axis=3)
 184           vis_pred_projs = tf.concat([vis_pred_projs] * 3, axis=3)
 185           # rescale
 186           new_size = [FLAGS.image_size] * 2
 187           vis_gt_projs = tf.image.resize_nearest_neighbor(
 188               vis_gt_projs, new_size)
 189           vis_pred_projs = tf.image.resize_nearest_neighbor(
 190               vis_pred_projs, new_size)
 191           # flip
 192           # vis_gt_projs = utils.image_flipud(vis_gt_projs)
 193           # vis_pred_projs = utils.image_flipud(vis_pred_projs)
 194           # vis_gt_projs is of shape [batch, height, width, channels]
 195           write_disk_op = model.write_disk_grid(
 196               global_step=global_step,
 197               log_dir=save_image_dir,
 198               input_images=vis_input_images,
 199               gt_projs=vis_gt_projs,
 200               pred_projs=vis_pred_projs,
 201               input_voxels=val_inputs['voxels'],
 202               output_voxels=val_outputs['voxels_1'])
 203         with tf.control_dependencies([write_disk_op]):
 204           train_op = tf.identity(train_op)
 205 
 206       #############
 207       ## init_fn ##
 208       #############
 209       if FLAGS.init_model:
 210         init_fn = model.get_init_fn(init_scopes)
 211       else:
 212         init_fn = None
 213 
 214       ##############
 215       ## training ##
 216       ##############
 217       slim.learning.train(
 218           train_op=train_op,
 219           logdir=train_dir,
 220           init_fn=init_fn,
 221           master=FLAGS.master,
 222           is_chief=(FLAGS.task == 0),
 223           number_of_steps=FLAGS.max_number_of_steps,
 224           saver=saver,
 225           save_summaries_secs=FLAGS.save_summaries_secs,
 226           save_interval_secs=FLAGS.save_interval_secs)
 227 
 228 
 229 if __name__ == '__main__':
 230   app.run()
