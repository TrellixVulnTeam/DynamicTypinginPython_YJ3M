  42   # For testing, this selects all views in input
  43   def preprocess_with_all_views(self, raw_inputs):
  44     (quantity, num_views) = raw_inputs['images'].get_shape().as_list()[:2]
  45 
  46     inputs = dict()
  47     inputs['voxels'] = []
  48     inputs['images_1'] = []
  49     for k in xrange(num_views):
  50       inputs['matrix_%d' % (k + 1)] = []
  51     inputs['matrix_1'] = []
  52     for n in xrange(quantity):
  53       for k in xrange(num_views):
  54         inputs['images_1'].append(raw_inputs['images'][n, k, :, :, :])
  55         inputs['voxels'].append(raw_inputs['voxels'][n, :, :, :, :])
  56         tf_matrix = self.get_transform_matrix(k)
  57         inputs['matrix_%d' % (k + 1)].append(tf_matrix)
  58 
  59     inputs['images_1'] = tf.stack(inputs['images_1'])
  60     inputs['voxels'] = tf.stack(inputs['voxels'])
  61     for k in xrange(num_views):
  62       inputs['matrix_%d' % (k + 1)] = tf.stack(inputs['matrix_%d' % (k + 1)])
  63 
  64     return inputs

  72   def get_loss(self, inputs, outputs):
  73     """Computes the loss used for PTN paper (projection + volume loss)."""
  74     g_loss = tf.zeros(dtype=tf.float32, shape=[])
  75 
  76     if self._params.proj_weight:
  77       g_loss += losses.add_volume_proj_loss(
  78           inputs, outputs, self._params.step_size, self._params.proj_weight)
  79 
  80     if self._params.volume_weight:
  81       g_loss += losses.add_volume_loss(inputs, outputs, 1,
  82                                        self._params.volume_weight)
  83 
  84     slim.summaries.add_scalar_summary(g_loss, 'im2vox_loss', prefix='losses')
  85 
  86     return g_loss

  88   def get_metrics(self, inputs, outputs):
  89     """Aggregate the metrics for voxel generation model.
  90 
  91     Args:
  92       inputs: Input dictionary of the voxel generation model.
  93       outputs: Output dictionary returned by the voxel generation model.
  94 
  95     Returns:
  96       names_to_values: metrics->values (dict).
  97       names_to_updates: metrics->ops (dict).
  98     """
  99     names_to_values = dict()
 100     names_to_updates = dict()
 101 
 102     tmp_values, tmp_updates = metrics.add_volume_iou_metrics(inputs, outputs)
 103 
 104     names_to_values.update(tmp_values)
 105     names_to_updates.update(tmp_updates)
 106 
 107     for name, value in names_to_values.iteritems():
 108       slim.summaries.add_scalar_summary(
 109           value, name, prefix='eval', print_summary=True)
 110 
 111     return names_to_values, names_to_updates

 113   def write_disk_grid(self,
 114                       global_step,
 115                       log_dir,
 116                       input_images,
 117                       gt_projs,
 118                       pred_projs,
 119                       input_voxels=None,
 120                       output_voxels=None):
 121     """Function called by TF to save the prediction periodically."""
 122     summary_freq = self._params.save_every
 123 
 124     def write_grid(input_images, gt_projs, pred_projs, global_step,
 125                    input_voxels, output_voxels):
 126       """Native python function to call for writing images to files."""
 127       grid = _build_image_grid(
 128           input_images,
 129           gt_projs,
 130           pred_projs,
 131           input_voxels=input_voxels,
 132           output_voxels=output_voxels)
 133 
 134       if global_step % summary_freq == 0:
 135         img_path = os.path.join(log_dir, '%s.jpg' % str(global_step))
 136         utils.save_image(grid, img_path)
 137       return grid
 138 
 139     save_op = tf.py_func(write_grid, [
 140         input_images, gt_projs, pred_projs, global_step, input_voxels,
 141         output_voxels
 142     ], [tf.uint8], 'write_grid')[0]
 143     slim.summaries.add_image_summary(
 144         tf.expand_dims(save_op, axis=0), name='grid_vis')
 145     return save_op

 147   def get_transform_matrix(self, view_out):
 148     """Get the 4x4 Perspective Transfromation matrix used for PTN."""
 149     num_views = self._params.num_views
 150     focal_length = self._params.focal_length
 151     focal_range = self._params.focal_range
 152     phi = 30
 153     theta_interval = 360.0 / num_views
 154     theta = theta_interval * view_out
 155 
 156     #  pylint: disable=invalid-name
 157     camera_matrix = np.zeros((4, 4), dtype=np.float32)
 158     intrinsic_matrix = np.eye(4, dtype=np.float32)
 159     extrinsic_matrix = np.eye(4, dtype=np.float32)
 160 
 161     sin_phi = np.sin(float(phi) / 180.0 * np.pi)
 162     cos_phi = np.cos(float(phi) / 180.0 * np.pi)
 163     sin_theta = np.sin(float(-theta) / 180.0 * np.pi)
 164     cos_theta = np.cos(float(-theta) / 180.0 * np.pi)
 165 
 166     rotation_azimuth = np.zeros((3, 3), dtype=np.float32)
 167     rotation_azimuth[0, 0] = cos_theta
 168     rotation_azimuth[2, 2] = cos_theta
 169     rotation_azimuth[0, 2] = -sin_theta
 170     rotation_azimuth[2, 0] = sin_theta
 171     rotation_azimuth[1, 1] = 1.0
 172 
 173     ## rotation axis -- x
 174     rotation_elevation = np.zeros((3, 3), dtype=np.float32)
 175     rotation_elevation[0, 0] = cos_phi
 176     rotation_elevation[0, 1] = sin_phi
 177     rotation_elevation[1, 0] = -sin_phi
 178     rotation_elevation[1, 1] = cos_phi
 179     rotation_elevation[2, 2] = 1.0
 180 
 181     rotation_matrix = np.matmul(rotation_azimuth, rotation_elevation)
 182     displacement = np.zeros((3, 1), dtype=np.float32)
 183     displacement[0, 0] = float(focal_length) + float(focal_range) / 2.0
 184     displacement = np.matmul(rotation_matrix, displacement)
 185 
 186     extrinsic_matrix[0:3, 0:3] = rotation_matrix
 187     extrinsic_matrix[0:3, 3:4] = -displacement
 188 
 189     intrinsic_matrix[2, 2] = 1.0 / float(focal_length)
 190     intrinsic_matrix[1, 1] = 1.0 / float(focal_length)
 191 
 192     camera_matrix = np.matmul(extrinsic_matrix, intrinsic_matrix)
 193     return camera_matrix

 196 def _build_image_grid(input_images,
 197                       gt_projs,
 198                       pred_projs,
 199                       input_voxels,
 200                       output_voxels,
 201                       vis_size=128):
 202   """Builds a grid image by concatenating the input images."""
 203   quantity = input_images.shape[0]
 204 
 205   for row in xrange(int(quantity / 3)):
 206     for col in xrange(3):
 207       index = row * 3 + col
 208       input_img_ = utils.resize_image(input_images[index, :, :, :], vis_size,
 209                                       vis_size)
 210       gt_proj_ = utils.resize_image(gt_projs[index, :, :, :], vis_size,
 211                                     vis_size)
 212       pred_proj_ = utils.resize_image(pred_projs[index, :, :, :], vis_size,
 213                                       vis_size)
 214       gt_voxel_vis = utils.resize_image(
 215           utils.display_voxel(input_voxels[index, :, :, :, 0]), vis_size,
 216           vis_size)
 217       pred_voxel_vis = utils.resize_image(
 218           utils.display_voxel(output_voxels[index, :, :, :, 0]), vis_size,
 219           vis_size)
 220       if col == 0:
 221         tmp_ = np.concatenate(
 222             [input_img_, gt_proj_, pred_proj_, gt_voxel_vis, pred_voxel_vis], 1)
 223       else:
 224         tmp_ = np.concatenate([
 225             tmp_, input_img_, gt_proj_, pred_proj_, gt_voxel_vis, pred_voxel_vis
 226         ], 1)
 227     if row == 0:
 228       out_grid = tmp_
 229     else:
 230       out_grid = np.concatenate([out_grid, tmp_], 0)
 231 
 232   return out_grid
