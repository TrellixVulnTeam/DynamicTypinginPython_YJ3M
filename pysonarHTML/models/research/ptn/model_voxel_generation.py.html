  58   def get_inputs(self,
  59                  dataset_dir,
  60                  dataset_name,
  61                  split_name,
  62                  batch_size,
  63                  image_size,
  64                  vox_size,
  65                  is_training=True):
  66     """Loads data for a specified dataset and split."""
  67     del image_size, vox_size
  68     with tf.variable_scope('data_loading_%s/%s' % (dataset_name, split_name)):
  69       common_queue_min = 64
  70       common_queue_capacity = 256
  71       num_readers = 4
  72 
  73       inputs = input_generator.get(
  74           dataset_dir,
  75           dataset_name,
  76           split_name,
  77           shuffle=is_training,
  78           num_readers=num_readers,
  79           common_queue_min=common_queue_min,
  80           common_queue_capacity=common_queue_capacity)
  81 
  82       images, voxels = tf.train.batch(
  83           [inputs['image'], inputs['voxel']],
  84           batch_size=batch_size,
  85           num_threads=8,
  86           capacity=8 * batch_size,
  87           name='batching_queues/%s/%s' % (dataset_name, split_name))
  88 
  89       outputs = dict()
  90       outputs['images'] = images
  91       outputs['voxels'] = voxels
  92       outputs['num_samples'] = inputs['num_samples']
  93 
  94     return outputs

  96   def preprocess(self, raw_inputs, step_size):
  97     """Selects the subset of viewpoints to train on."""
  98     (quantity, num_views) = raw_inputs['images'].get_shape().as_list()[:2]
  99 
 100     inputs = dict()
 101     inputs['voxels'] = raw_inputs['voxels']
 102 
 103     for k in xrange(step_size):
 104       inputs['images_%d' % (k + 1)] = []
 105       inputs['matrix_%d' % (k + 1)] = []
 106 
 107     for n in xrange(quantity):
 108       selected_views = np.random.choice(num_views, step_size, replace=False)
 109       for k in xrange(step_size):
 110         view_selected = selected_views[k]
 111         inputs['images_%d' %
 112                (k + 1)].append(raw_inputs['images'][n, view_selected, :, :, :])
 113         tf_matrix = self.get_transform_matrix(view_selected)
 114         inputs['matrix_%d' % (k + 1)].append(tf_matrix)
 115 
 116     for k in xrange(step_size):
 117       inputs['images_%d' % (k + 1)] = tf.stack(inputs['images_%d' % (k + 1)])
 118       inputs['matrix_%d' % (k + 1)] = tf.stack(inputs['matrix_%d' % (k + 1)])
 119 
 120     return inputs

 122   def get_init_fn(self, scopes):
 123     """Initialization assignment operator function used while training."""
 124     if not self._params.init_model:
 125       return None
 126 
 127     is_trainable = lambda x: x in tf.trainable_variables()
 128     var_list = []
 129     for scope in scopes:
 130       var_list.extend(
 131           filter(is_trainable, tf.contrib.framework.get_model_variables(scope)))
 132 
 133     init_assign_op, init_feed_dict = slim.assign_from_checkpoint(
 134         self._params.init_model, var_list)
 135 
 136     def init_assign_function(sess):
 137       sess.run(init_assign_op, init_feed_dict)
 138 
 139     return init_assign_function
 140 
 141   def get_train_op_for_scope(self, loss, optimizer, scopes):
 142     """Train operation function for the given scope used file training."""
 143     is_trainable = lambda x: x in tf.trainable_variables()
 144 
 145     var_list = []
 146     update_ops = []
 147 
 148     for scope in scopes:
 149       var_list.extend(
 150           filter(is_trainable, tf.contrib.framework.get_model_variables(scope)))
 151       update_ops.extend(tf.get_collection(tf.GraphKeys.UPDATE_OPS, scope))
 152 
 153     return slim.learning.create_train_op(
 154         loss,
 155         optimizer,
 156         update_ops=update_ops,
 157         variables_to_train=var_list,
 158         clip_gradient_norm=self._params.clip_gradient_norm)

 160   def write_disk_grid(self,
 161                       global_step,
 162                       log_dir,
 163                       input_images,
 164                       gt_projs,
 165                       pred_projs,
 166                       pred_voxels=None):
 167     """Function called by TF to save the prediction periodically."""
 168     summary_freq = self._params.save_every
 169 
 170     def write_grid(input_images, gt_projs, pred_projs, pred_voxels,
 171                    global_step):
 172       """Native python function to call for writing images to files."""
 173       grid = _build_image_grid(input_images, gt_projs, pred_projs, pred_voxels)
 174 
 175       if global_step % summary_freq == 0:
 176         img_path = os.path.join(log_dir, '%s.jpg' % str(global_step))
 177         utils.save_image(grid, img_path)
 178         with open(
 179             os.path.join(log_dir, 'pred_voxels_%s' % str(global_step)),
 180             'w') as fout:
 181           np.save(fout, pred_voxels)
 182         with open(
 183             os.path.join(log_dir, 'input_images_%s' % str(global_step)),
 184             'w') as fout:
 185           np.save(fout, input_images)
 186 
 187       return grid
 188 
 189     py_func_args = [
 190         input_images, gt_projs, pred_projs, pred_voxels, global_step
 191     ]
 192     save_grid_op = tf.py_func(write_grid, py_func_args, [tf.uint8],
 193                               'wrtie_grid')[0]
 194     slim.summaries.add_image_summary(
 195         tf.expand_dims(save_grid_op, axis=0), name='grid_vis')
 196     return save_grid_op

 199 def _build_image_grid(input_images, gt_projs, pred_projs, pred_voxels):
 200   """Build the visualization grid with py_func."""
 201   quantity, img_height, img_width = input_images.shape[:3]
 202   for row in xrange(int(quantity / 3)):
 203     for col in xrange(3):
 204       index = row * 3 + col
 205       input_img_ = input_images[index, :, :, :]
 206       gt_proj_ = gt_projs[index, :, :, :]
 207       pred_proj_ = pred_projs[index, :, :, :]
 208       pred_voxel_ = utils.display_voxel(pred_voxels[index, :, :, :, 0])
 209       pred_voxel_ = utils.resize_image(pred_voxel_, img_height, img_width)
 210       if col == 0:
 211         tmp_ = np.concatenate([input_img_, gt_proj_, pred_proj_, pred_voxel_],
 212                               1)
 213       else:
 214         tmp_ = np.concatenate(
 215             [tmp_, input_img_, gt_proj_, pred_proj_, pred_voxel_], 1)
 216     if row == 0:
 217       out_grid = tmp_
 218     else:
 219       out_grid = np.concatenate([out_grid, tmp_], 0)
 220 
 221   out_grid = out_grid.astype(np.uint8)
 222   return out_grid
