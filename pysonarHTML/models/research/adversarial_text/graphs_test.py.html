  42 def _build_random_vocabulary(vocab_size=100):
  43   """Builds and returns a dict<term, id>."""
  44   vocab = set()
  45   while len(vocab) < (vocab_size - 1):
  46     rand_word = ''.join(
  47         random.choice(string.ascii_lowercase)
  48         for _ in range(random.randint(1, 10)))
  49     vocab.add(rand_word)
  50 
  51   vocab_ids = dict([(word, i) for i, word in enumerate(vocab)])
  52   vocab_ids[data.EOS_TOKEN] = vocab_size - 1
  53   return vocab_ids

  56 def _build_random_sequence(vocab_ids):
  57   seq_len = random.randint(10, 200)
  58   ids = vocab_ids.values()
  59   seq = data.SequenceWrapper()
  60   for token_id in [random.choice(ids) for _ in range(seq_len)]:
  61     seq.add_timestep().set_token(token_id)
  62   return seq

  65 def _build_vocab_frequencies(seqs, vocab_ids):
  66   vocab_freqs = defaultdict(int)
  67   ids_to_words = dict([(i, word) for word, i in vocab_ids.iteritems()])
  68   for seq in seqs:
  69     for timestep in seq:
  70       vocab_freqs[ids_to_words[timestep.token]] += 1
  71 
  72   vocab_freqs[data.EOS_TOKEN] = 0
  73   return vocab_freqs

  76 class GraphsTest(tf.test.TestCase):
  77   """Test graph construction methods."""
  78 
  79   @classmethod
  80   def setUpClass(cls):
  81     # Make model small
  82     FLAGS.batch_size = 2
  83     FLAGS.num_timesteps = 3
  84     FLAGS.embedding_dims = 4
  85     FLAGS.rnn_num_layers = 2
  86     FLAGS.rnn_cell_size = 4
  87     FLAGS.cl_num_layers = 2
  88     FLAGS.cl_hidden_size = 4
  89     FLAGS.vocab_size = 10
  90 
  91     # Set input/output flags
  92     FLAGS.data_dir = tempfile.mkdtemp()
  93 
  94     # Build and write sequence files.
  95     vocab_ids = _build_random_vocabulary(FLAGS.vocab_size)
  96     seqs = [_build_random_sequence(vocab_ids) for _ in range(5)]
  97     seqs_label = [
  98         data.build_labeled_sequence(seq, random.choice([True, False]))
  99         for seq in seqs
 100     ]
 101     seqs_lm = [data.build_lm_sequence(seq) for seq in seqs]
 102     seqs_ae = [data.build_seq_ae_sequence(seq) for seq in seqs]
 103     seqs_rev = [data.build_reverse_sequence(seq) for seq in seqs]
 104     seqs_bidir = [
 105         data.build_bidirectional_seq(seq, rev)
 106         for seq, rev in zip(seqs, seqs_rev)
 107     ]
 108     seqs_bidir_label = [
 109         data.build_labeled_sequence(bd_seq, random.choice([True, False]))
 110         for bd_seq in seqs_bidir
 111     ]
 112 
 113     filenames = [
 114         data.TRAIN_CLASS, data.TRAIN_LM, data.TRAIN_SA, data.TEST_CLASS,
 115         data.TRAIN_REV_LM, data.TRAIN_BD_CLASS, data.TEST_BD_CLASS
 116     ]
 117     seq_lists = [
 118         seqs_label, seqs_lm, seqs_ae, seqs_label, seqs_rev, seqs_bidir,
 119         seqs_bidir_label
 120     ]
 121     for fname, seq_list in zip(filenames, seq_lists):
 122       with tf.python_io.TFRecordWriter(
 123           os.path.join(FLAGS.data_dir, fname)) as writer:
 124         for seq in seq_list:
 125           writer.write(seq.seq.SerializeToString())
 126 
 127     # Write vocab.txt and vocab_freq.txt
 128     vocab_freqs = _build_vocab_frequencies(seqs, vocab_ids)
 129     ordered_vocab_freqs = sorted(
 130         vocab_freqs.items(), key=operator.itemgetter(1), reverse=True)
 131     with open(os.path.join(FLAGS.data_dir, 'vocab.txt'), 'w') as vocab_f:
 132       with open(os.path.join(FLAGS.data_dir, 'vocab_freq.txt'), 'w') as freq_f:
 133         for word, freq in ordered_vocab_freqs:
 134           vocab_f.write('{}\n'.format(word))
 135           freq_f.write('{}\n'.format(freq))

 153   def testClassifierGraph(self):
 154     FLAGS.rnn_num_layers = 2
 155     model = graphs.VatxtModel()
 156     train_op, _, _ = model.classifier_training()
 157     # Pretrained vars: embedding + LSTM layers
 158     self.assertEqual(
 159         len(model.pretrained_variables), 1 + 2 * FLAGS.rnn_num_layers)
 160     with self.test_session() as sess:
 161       sess.run(tf.global_variables_initializer())
 162       tf.train.start_queue_runners(sess)
 163       sess.run(train_op)

 176   def testATMethods(self):
 177     at_methods = [None, 'rp', 'at', 'vat', 'atvat']
 178     for method in at_methods:
 179       FLAGS.adv_training_method = method
 180       with tf.Graph().as_default():
 181         graphs.VatxtModel().classifier_graph()
 182 
 183         # Ensure variables have been reused
 184         # Embedding + LSTM layers + hidden layers + logits layer
 185         expected_num_vars = 1 + 2 * FLAGS.rnn_num_layers + 2 * (
 186             FLAGS.cl_num_layers) + 2
 187         self.assertEqual(len(tf.trainable_variables()), expected_num_vars)
