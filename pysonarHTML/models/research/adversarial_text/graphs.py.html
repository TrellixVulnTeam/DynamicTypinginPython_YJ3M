 103 class VatxtModel(object):
 104   """Constructs training and evaluation graphs.
 105 
 106   Main methods: `classifier_training()`, `language_model_training()`,
 107   and `eval_graph()`.
 108 
 109   Variable reuse is a critical part of the model, both for sharing variables
 110   between the language model and the classifier, and for reusing variables for
 111   the adversarial loss calculation. To ensure correct variable reuse, all
 112   variables are created in Keras-style layers, wherein stateful layers (i.e.
 113   layers with variables) are represented as callable instances of the Layer
 114   class. Each time the Layer instance is called, it is using the same variables.
 115 
 116   All Layers are constructed in the __init__ method and reused in the various
 117   graph-building functions.
 118   """
 119 
 120   def __init__(self, cl_logits_input_dim=None):
 121     self.global_step = tf.train.get_or_create_global_step()
 122     self.vocab_freqs = _get_vocab_freqs()
 123 
 124     # Cache VatxtInput objects
 125     self.cl_inputs = None
 126     self.lm_inputs = None
 127 
 128     # Cache intermediate Tensors that are reused
 129     self.tensors = {}
 130 
 131     # Construct layers which are reused in constructing the LM and
 132     # Classification graphs. Instantiating them all once here ensures that
 133     # variable reuse works correctly.
 134     self.layers = {}
 135     self.layers['embedding'] = layers_lib.Embedding(
 136         FLAGS.vocab_size, FLAGS.embedding_dims, FLAGS.normalize_embeddings,
 137         self.vocab_freqs, FLAGS.keep_prob_emb)
 138     self.layers['lstm'] = layers_lib.LSTM(
 139         FLAGS.rnn_cell_size, FLAGS.rnn_num_layers, FLAGS.keep_prob_lstm_out)
 140     self.layers['lm_loss'] = layers_lib.SoftmaxLoss(
 141         FLAGS.vocab_size,
 142         FLAGS.num_candidate_samples,
 143         self.vocab_freqs,
 144         name='LM_loss')
 145 
 146     cl_logits_input_dim = cl_logits_input_dim or FLAGS.rnn_cell_size
 147     self.layers['cl_logits'] = layers_lib.cl_logits_subgraph(
 148         [FLAGS.cl_hidden_size] * FLAGS.cl_num_layers, cl_logits_input_dim,
 149         FLAGS.num_classes, FLAGS.keep_prob_cl_hidden)
 150 
 151   @property
 152   def pretrained_variables(self):
 153     return (self.layers['embedding'].trainable_weights +
 154             self.layers['lstm'].trainable_weights)
 155 
 156   def classifier_training(self):
 157     loss = self.classifier_graph()
 158     train_op = optimize(loss, self.global_step)
 159     return train_op, loss, self.global_step
 160 
 161   def language_model_training(self):
 162     loss = self.language_model_graph()
 163     train_op = optimize(loss, self.global_step)
 164     return train_op, loss, self.global_step
 165 
 166   def classifier_graph(self):
 167     """Constructs classifier graph from inputs to classifier loss.
 168 
 169     * Caches the VatxtInput object in `self.cl_inputs`
 170     * Caches tensors: `cl_embedded`, `cl_logits`, `cl_loss`
 171 
 172     Returns:
 173       loss: scalar float.
 174     """
 175     inputs = _inputs('train', pretrain=False)
 176     self.cl_inputs = inputs
 177     embedded = self.layers['embedding'](inputs.tokens)
 178     self.tensors['cl_embedded'] = embedded
 179 
 180     _, next_state, logits, loss = self.cl_loss_from_embedding(
 181         embedded, return_intermediates=True)
 182     tf.summary.scalar('classification_loss', loss)
 183     self.tensors['cl_logits'] = logits
 184     self.tensors['cl_loss'] = loss
 185 
 186     if FLAGS.single_label:
 187       indices = tf.stack([tf.range(FLAGS.batch_size), inputs.length - 1], 1)
 188       labels = tf.expand_dims(tf.gather_nd(inputs.labels, indices), 1)
 189       weights = tf.expand_dims(tf.gather_nd(inputs.weights, indices), 1)
 190     else:
 191       labels = inputs.labels
 192       weights = inputs.weights
 193     acc = layers_lib.accuracy(logits, labels, weights)
 194     tf.summary.scalar('accuracy', acc)
 195 
 196     adv_loss = (self.adversarial_loss() * tf.constant(
 197         FLAGS.adv_reg_coeff, name='adv_reg_coeff'))
 198     tf.summary.scalar('adversarial_loss', adv_loss)
 199 
 200     total_loss = loss + adv_loss
 201 
 202     with tf.control_dependencies([inputs.save_state(next_state)]):
 203       total_loss = tf.identity(total_loss)
 204       tf.summary.scalar('total_classification_loss', total_loss)
 205     return total_loss
 206 
 207   def language_model_graph(self, compute_loss=True):
 208     """Constructs LM graph from inputs to LM loss.
 209 
 210     * Caches the VatxtInput object in `self.lm_inputs`
 211     * Caches tensors: `lm_embedded`
 212 
 213     Args:
 214       compute_loss: bool, whether to compute and return the loss or stop after
 215         the LSTM computation.
 216 
 217     Returns:
 218       loss: scalar float.
 219     """
 220     inputs = _inputs('train', pretrain=True)
 221     self.lm_inputs = inputs
 222     return self._lm_loss(inputs, compute_loss=compute_loss)
 223 
 224   def _lm_loss(self,
 225                inputs,
 226                emb_key='lm_embedded',
 227                lstm_layer='lstm',
 228                lm_loss_layer='lm_loss',
 229                loss_name='lm_loss',
 230                compute_loss=True):
 231     embedded = self.layers['embedding'](inputs.tokens)
 232     self.tensors[emb_key] = embedded
 233     lstm_out, next_state = self.layers[lstm_layer](embedded, inputs.state,
 234                                                    inputs.length)
 235     if compute_loss:
 236       loss = self.layers[lm_loss_layer](
 237           [lstm_out, inputs.labels, inputs.weights])
 238       with tf.control_dependencies([inputs.save_state(next_state)]):
 239         loss = tf.identity(loss)
 240         tf.summary.scalar(loss_name, loss)
 241 
 242       return loss
 243 
 244   def eval_graph(self, dataset='test'):
 245     """Constructs classifier evaluation graph.
 246 
 247     Args:
 248       dataset: the labeled dataset to evaluate, {'train', 'test', 'valid'}.
 249 
 250     Returns:
 251       eval_ops: dict<metric name, tuple(value, update_op)>
 252       var_restore_dict: dict mapping variable restoration names to variables.
 253         Trainable variables will be mapped to their moving average names.
 254     """
 255     inputs = _inputs(dataset, pretrain=False)
 256     embedded = self.layers['embedding'](inputs.tokens)
 257     _, next_state, logits, _ = self.cl_loss_from_embedding(
 258         embedded, inputs=inputs, return_intermediates=True)
 259 
 260     if FLAGS.single_label:
 261       indices = tf.stack([tf.range(FLAGS.batch_size), inputs.length - 1], 1)
 262       labels = tf.expand_dims(tf.gather_nd(inputs.labels, indices), 1)
 263       weights = tf.expand_dims(tf.gather_nd(inputs.weights, indices), 1)
 264     else:
 265       labels = inputs.labels
 266       weights = inputs.weights
 267     eval_ops = {
 268         'accuracy':
 269             tf.contrib.metrics.streaming_accuracy(
 270                 layers_lib.predictions(logits), labels, weights)
 271     }
 272 
 273     with tf.control_dependencies([inputs.save_state(next_state)]):
 274       acc, acc_update = eval_ops['accuracy']
 275       acc_update = tf.identity(acc_update)
 276       eval_ops['accuracy'] = (acc, acc_update)
 277 
 278     var_restore_dict = make_restore_average_vars_dict()
 279     return eval_ops, var_restore_dict
 280 
 281   def cl_loss_from_embedding(self,
 282                              embedded,
 283                              inputs=None,
 284                              return_intermediates=False):
 285     """Compute classification loss from embedding.
 286 
 287     Args:
 288       embedded: 3-D float Tensor [batch_size, num_timesteps, embedding_dim]
 289       inputs: VatxtInput, defaults to self.cl_inputs.
 290       return_intermediates: bool, whether to return intermediate tensors or only
 291         the final loss.
 292 
 293     Returns:
 294       If return_intermediates is True:
 295         lstm_out, next_state, logits, loss
 296       Else:
 297         loss
 298     """
 299     if inputs is None:
 300       inputs = self.cl_inputs
 301 
 302     lstm_out, next_state = self.layers['lstm'](embedded, inputs.state,
 303                                                inputs.length)
 304     if FLAGS.single_label:
 305       indices = tf.stack([tf.range(FLAGS.batch_size), inputs.length - 1], 1)
 306       lstm_out = tf.expand_dims(tf.gather_nd(lstm_out, indices), 1)
 307       labels = tf.expand_dims(tf.gather_nd(inputs.labels, indices), 1)
 308       weights = tf.expand_dims(tf.gather_nd(inputs.weights, indices), 1)
 309     else:
 310       labels = inputs.labels
 311       weights = inputs.weights
 312     logits = self.layers['cl_logits'](lstm_out)
 313     loss = layers_lib.classification_loss(logits, labels, weights)
 314 
 315     if return_intermediates:
 316       return lstm_out, next_state, logits, loss
 317     else:
 318       return loss
 319 
 320   def adversarial_loss(self):
 321     """Compute adversarial loss based on FLAGS.adv_training_method."""
 322 
 323     def random_perturbation_loss():
 324       return adv_lib.random_perturbation_loss(self.tensors['cl_embedded'],
 325                                               self.cl_inputs.length,
 326                                               self.cl_loss_from_embedding)
 327 
 328     def adversarial_loss():
 329       return adv_lib.adversarial_loss(self.tensors['cl_embedded'],
 330                                       self.tensors['cl_loss'],
 331                                       self.cl_loss_from_embedding)
 332 
 333     def virtual_adversarial_loss():
 334       """Computes virtual adversarial loss.
 335 
 336       Uses lm_inputs and constructs the language model graph if it hasn't yet
 337       been constructed.
 338 
 339       Also ensures that the LM input states are saved for LSTM state-saving
 340       BPTT.
 341 
 342       Returns:
 343         loss: float scalar.
 344       """
 345       if self.lm_inputs is None:
 346         self.language_model_graph(compute_loss=False)
 347 
 348       def logits_from_embedding(embedded, return_next_state=False):
 349         _, next_state, logits, _ = self.cl_loss_from_embedding(
 350             embedded, inputs=self.lm_inputs, return_intermediates=True)
 351         if return_next_state:
 352           return next_state, logits
 353         else:
 354           return logits
 355 
 356       next_state, lm_cl_logits = logits_from_embedding(
 357           self.tensors['lm_embedded'], return_next_state=True)
 358 
 359       va_loss = adv_lib.virtual_adversarial_loss(
 360           lm_cl_logits, self.tensors['lm_embedded'], self.lm_inputs,
 361           logits_from_embedding)
 362 
 363       with tf.control_dependencies([self.lm_inputs.save_state(next_state)]):
 364         va_loss = tf.identity(va_loss)
 365 
 366       return va_loss
 367 
 368     def combo_loss():
 369       return adversarial_loss() + virtual_adversarial_loss()
 370 
 371     adv_training_methods = {
 372         # Random perturbation
 373         'rp': random_perturbation_loss,
 374         # Adversarial training
 375         'at': adversarial_loss,
 376         # Virtual adversarial training
 377         'vat': virtual_adversarial_loss,
 378         # Both at and vat
 379         'atvat': combo_loss,
 380         '': lambda: tf.constant(0.),
 381         None: lambda: tf.constant(0.),
 382     }
 383 
 384     with tf.name_scope('adversarial_loss'):
 385       return adv_training_methods[FLAGS.adv_training_method]()

 388 class VatxtBidirModel(VatxtModel):
 389   """Extension of VatxtModel that supports bidirectional input."""
 390 
 391   def __init__(self):
 392     super(VatxtBidirModel,
 393           self).__init__(cl_logits_input_dim=FLAGS.rnn_cell_size * 2)
 394 
 395     # Reverse LSTM and LM loss for bidirectional models
 396     self.layers['lstm_reverse'] = layers_lib.LSTM(
 397         FLAGS.rnn_cell_size,
 398         FLAGS.rnn_num_layers,
 399         FLAGS.keep_prob_lstm_out,
 400         name='LSTM_Reverse')
 401     self.layers['lm_loss_reverse'] = layers_lib.SoftmaxLoss(
 402         FLAGS.vocab_size,
 403         FLAGS.num_candidate_samples,
 404         self.vocab_freqs,
 405         name='LM_loss_reverse')
 406 
 407   @property
 408   def pretrained_variables(self):
 409     variables = super(VatxtBidirModel, self).pretrained_variables
 410     variables.extend(self.layers['lstm_reverse'].trainable_weights)
 411     return variables

 413   def classifier_graph(self):
 414     """Constructs classifier graph from inputs to classifier loss.
 415 
 416     * Caches the VatxtInput objects in `self.cl_inputs`
 417     * Caches tensors: `cl_embedded` (tuple of forward and reverse), `cl_logits`,
 418       `cl_loss`
 419 
 420     Returns:
 421       loss: scalar float.
 422     """
 423     inputs = _inputs('train', pretrain=False, bidir=True)
 424     self.cl_inputs = inputs
 425     f_inputs, _ = inputs
 426 
 427     # Embed both forward and reverse with a shared embedding
 428     embedded = [self.layers['embedding'](inp.tokens) for inp in inputs]
 429     self.tensors['cl_embedded'] = embedded
 430 
 431     _, next_states, logits, loss = self.cl_loss_from_embedding(
 432         embedded, return_intermediates=True)
 433     tf.summary.scalar('classification_loss', loss)
 434     self.tensors['cl_logits'] = logits
 435     self.tensors['cl_loss'] = loss
 436 
 437     acc = layers_lib.accuracy(logits, f_inputs.labels, f_inputs.weights)
 438     tf.summary.scalar('accuracy', acc)
 439 
 440     adv_loss = (self.adversarial_loss() * tf.constant(
 441         FLAGS.adv_reg_coeff, name='adv_reg_coeff'))
 442     tf.summary.scalar('adversarial_loss', adv_loss)
 443 
 444     total_loss = loss + adv_loss
 445 
 446 
 447     saves = [inp.save_state(state) for (inp, state) in zip(inputs, next_states)]
 448     with tf.control_dependencies(saves):
 449       total_loss = tf.identity(total_loss)
 450       tf.summary.scalar('total_classification_loss', total_loss)
 451     return total_loss
 452 
 453   def language_model_graph(self, compute_loss=True):
 454     """Constructs forward and reverse LM graphs from inputs to LM losses.
 455 
 456     * Caches the VatxtInput objects in `self.lm_inputs`
 457     * Caches tensors: `lm_embedded`, `lm_embedded_reverse`
 458 
 459     Args:
 460       compute_loss: bool, whether to compute and return the loss or stop after
 461         the LSTM computation.
 462 
 463     Returns:
 464       loss: scalar float, sum of forward and reverse losses.
 465     """
 466     inputs = _inputs('train', pretrain=True, bidir=True)
 467     self.lm_inputs = inputs
 468     f_inputs, r_inputs = inputs
 469     f_loss = self._lm_loss(f_inputs, compute_loss=compute_loss)
 470     r_loss = self._lm_loss(
 471         r_inputs,
 472         emb_key='lm_embedded_reverse',
 473         lstm_layer='lstm_reverse',
 474         lm_loss_layer='lm_loss_reverse',
 475         loss_name='lm_loss_reverse',
 476         compute_loss=compute_loss)
 477     if compute_loss:
 478       return f_loss + r_loss

 480   def eval_graph(self, dataset='test'):
 481     """Constructs classifier evaluation graph.
 482 
 483     Args:
 484       dataset: the labeled dataset to evaluate, {'train', 'test', 'valid'}.
 485 
 486     Returns:
 487       eval_ops: dict<metric name, tuple(value, update_op)>
 488       var_restore_dict: dict mapping variable restoration names to variables.
 489         Trainable variables will be mapped to their moving average names.
 490     """
 491     inputs = _inputs(dataset, pretrain=False, bidir=True)
 492     embedded = [self.layers['embedding'](inp.tokens) for inp in inputs]
 493     _, next_states, logits, _ = self.cl_loss_from_embedding(
 494         embedded, inputs=inputs, return_intermediates=True)
 495     f_inputs, _ = inputs
 496 
 497     eval_ops = {
 498         'accuracy':
 499             tf.contrib.metrics.streaming_accuracy(
 500                 layers_lib.predictions(logits), f_inputs.labels,
 501                 f_inputs.weights)
 502     }
 503 
 504     # Save states on accuracy update
 505     saves = [inp.save_state(state) for (inp, state) in zip(inputs, next_states)]
 506     with tf.control_dependencies(saves):
 507       acc, acc_update = eval_ops['accuracy']
 508       acc_update = tf.identity(acc_update)
 509       eval_ops['accuracy'] = (acc, acc_update)
 510 
 511     var_restore_dict = make_restore_average_vars_dict()
 512     return eval_ops, var_restore_dict
 513 
 514   def cl_loss_from_embedding(self,
 515                              embedded,
 516                              inputs=None,
 517                              return_intermediates=False):
 518     """Compute classification loss from embedding.
 519 
 520     Args:
 521       embedded: Length 2 tuple of 3-D float Tensor
 522         [batch_size, num_timesteps, embedding_dim].
 523       inputs: Length 2 tuple of VatxtInput, defaults to self.cl_inputs.
 524       return_intermediates: bool, whether to return intermediate tensors or only
 525         the final loss.
 526 
 527     Returns:
 528       If return_intermediates is True:
 529         lstm_out, next_states, logits, loss
 530       Else:
 531         loss
 532     """
 533     if inputs is None:
 534       inputs = self.cl_inputs
 535 
 536     out = []
 537     for (layer_name, emb, inp) in zip(['lstm', 'lstm_reverse'], embedded,
 538                                       inputs):
 539       out.append(self.layers[layer_name](emb, inp.state, inp.length))
 540     lstm_outs, next_states = zip(*out)
 541 
 542     # Concatenate output of forward and reverse LSTMs
 543     lstm_out = tf.concat(lstm_outs, 1)
 544 
 545     logits = self.layers['cl_logits'](lstm_out)
 546     f_inputs, _ = inputs  # pylint: disable=unpacking-non-sequence
 547     loss = layers_lib.classification_loss(logits, f_inputs.labels,
 548                                           f_inputs.weights)
 549 
 550     if return_intermediates:
 551       return lstm_out, next_states, logits, loss
 552     else:
 553       return loss
 554 
 555   def adversarial_loss(self):
 556     """Compute adversarial loss based on FLAGS.adv_training_method."""
 557 
 558     def random_perturbation_loss():
 559       return adv_lib.random_perturbation_loss_bidir(self.tensors['cl_embedded'],
 560                                                     self.cl_inputs[0].length,
 561                                                     self.cl_loss_from_embedding)
 562 
 563     def adversarial_loss():
 564       return adv_lib.adversarial_loss_bidir(self.tensors['cl_embedded'],
 565                                             self.tensors['cl_loss'],
 566                                             self.cl_loss_from_embedding)
 567 
 568     def virtual_adversarial_loss():
 569       """Computes virtual adversarial loss.
 570 
 571       Uses lm_inputs and constructs the language model graph if it hasn't yet
 572       been constructed.
 573 
 574       Also ensures that the LM input states are saved for LSTM state-saving
 575       BPTT.
 576 
 577       Returns:
 578         loss: float scalar.
 579       """
 580       if self.lm_inputs is None:
 581         self.language_model_graph(compute_loss=False)
 582 
 583       def logits_from_embedding(embedded, return_next_state=False):
 584         _, next_states, logits, _ = self.cl_loss_from_embedding(
 585             embedded, inputs=self.lm_inputs, return_intermediates=True)
 586         if return_next_state:
 587           return next_states, logits
 588         else:
 589           return logits
 590 
 591       lm_embedded = (self.tensors['lm_embedded'],
 592                      self.tensors['lm_embedded_reverse'])
 593       next_states, lm_cl_logits = logits_from_embedding(
 594           lm_embedded, return_next_state=True)
 595 
 596       va_loss = adv_lib.virtual_adversarial_loss_bidir(
 597           lm_cl_logits, lm_embedded, self.lm_inputs, logits_from_embedding)
 598 
 599       saves = [
 600           inp.save_state(state)
 601           for (inp, state) in zip(self.lm_inputs, next_states)
 602       ]
 603       with tf.control_dependencies(saves):
 604         va_loss = tf.identity(va_loss)
 605 
 606       return va_loss
 607 
 608     def combo_loss():
 609       return adversarial_loss() + virtual_adversarial_loss()
 610 
 611     adv_training_methods = {
 612         # Random perturbation

 642 def _get_vocab_freqs():
 643   """Returns vocab frequencies.
 644 
 645   Returns:
 646     List of integers, length=FLAGS.vocab_size.
 647 
 648   Raises:
 649     ValueError: if the length of the frequency file is not equal to the vocab
 650       size, or if the file is not found.
 651   """
 652   path = FLAGS.vocab_freq_path or os.path.join(FLAGS.data_dir, 'vocab_freq.txt')
 653 
 654   if tf.gfile.Exists(path):
 655     with tf.gfile.Open(path) as f:
 656       # Get pre-calculated frequencies of words.
 657       reader = csv.reader(f, quoting=csv.QUOTE_NONE)
 658       freqs = [int(row[-1]) for row in reader]
 659       if len(freqs) != FLAGS.vocab_size:
 660         raise ValueError('Frequency file length %d != vocab size %d' %
 661                          (len(freqs), FLAGS.vocab_size))
 662   else:
 663     if FLAGS.vocab_freq_path:
 664       raise ValueError('vocab_freq_path not found')
 665     freqs = [1] * FLAGS.vocab_size
 666 
 667   return freqs

 670 def make_restore_average_vars_dict():
 671   """Returns dict mapping moving average names to variables."""
 672   var_restore_dict = {}
 673   variable_averages = tf.train.ExponentialMovingAverage(0.999)
 674   for v in tf.global_variables():
 675     if v in tf.trainable_variables():
 676       name = variable_averages.average_name(v)
 677     else:
 678       name = v.op.name
 679     var_restore_dict[name] = v
 680   return var_restore_dict
