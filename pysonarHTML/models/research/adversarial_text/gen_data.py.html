  15 """Create TFRecord files of SequenceExample protos from dataset.
  16 
  17 Constructs 3 datasets:
  18   1. Labeled data for the LSTM classification model, optionally with label gain.
  19      "*_classification.tfrecords" (for both unidirectional and bidirectional
  20      models).
  21   2. Data for the unsupervised LM-LSTM model that predicts the next token.
  22      "*_lm.tfrecords" (generates forward and reverse data).
  23   3. Data for the unsupervised SA-LSTM model that uses Seq2Seq.
  24      "*_sa.tfrecords".
  25 """

  27 from __future__ import division
  28 from __future__ import print_function

  41 flags = tf.app.flags
  42 FLAGS = flags.FLAGS

  44 # Flags for input data are in document_generators.py
  45 flags.DEFINE_string('vocab_file', '', 'Path to the vocabulary file. Defaults '

  51                      'Enable linear label gain. If True, sentiment label will '
  52                      'be included at each timestep with linear weight '
  53                      'increase.')

  56 def build_shuffling_tf_record_writer(fname):
  57   return data.ShufflingTFRecordWriter(os.path.join(FLAGS.output_dir, fname))
  58 
  59 
  60 def build_tf_record_writer(fname):
  61   return tf.python_io.TFRecordWriter(os.path.join(FLAGS.output_dir, fname))

  64 def build_input_sequence(doc, vocab_ids):
  65   """Builds input sequence from file.
  66 
  67   Splits lines on whitespace. Treats punctuation as whitespace. For word-level
  68   sequences, only keeps terms that are in the vocab.
  69 
  70   Terms are added as token in the SequenceExample.  The EOS_TOKEN is also
  71   appended. Label and weight features are set to 0.
  72 
  73   Args:
  74     doc: Document (defined in `document_generators`) from which to build the
  75       sequence.
  76     vocab_ids: dict<term, id>.
  77 
  78   Returns:
  79     SequenceExampleWrapper.
  80   """
  81   seq = data.SequenceWrapper()
  82   for token in document_generators.tokens(doc):
  83     if token in vocab_ids:
  84       seq.add_timestep().set_token(vocab_ids[token])
  85 
  86   # Add EOS token to end
  87   seq.add_timestep().set_token(vocab_ids[data.EOS_TOKEN])
  88 
  89   return seq

  92 def make_vocab_ids(vocab_filename):
  93   if FLAGS.output_char:
  94     ret = dict([(char, i) for i, char in enumerate(string.printable)])
  95     ret[data.EOS_TOKEN] = len(string.printable)
  96     return ret
  97   else:
  98     with open(vocab_filename, encoding='utf-8') as vocab_f:
  99       return dict([(line.strip(), i) for i, line in enumerate(vocab_f)])

 102 def generate_training_data(vocab_ids, writer_lm_all, writer_seq_ae_all):
 103   """Generates training data."""
 104 
 105   # Construct training data writers
 106   writer_lm = build_shuffling_tf_record_writer(data.TRAIN_LM)
 107   writer_seq_ae = build_shuffling_tf_record_writer(data.TRAIN_SA)
 108   writer_class = build_shuffling_tf_record_writer(data.TRAIN_CLASS)
 109   writer_valid_class = build_tf_record_writer(data.VALID_CLASS)
 110   writer_rev_lm = build_shuffling_tf_record_writer(data.TRAIN_REV_LM)
 111   writer_bd_class = build_shuffling_tf_record_writer(data.TRAIN_BD_CLASS)
 112   writer_bd_valid_class = build_shuffling_tf_record_writer(data.VALID_BD_CLASS)
 113 
 114   for doc in document_generators.documents(
 115       dataset='train', include_unlabeled=True, include_validation=True):
 116     input_seq = build_input_sequence(doc, vocab_ids)
 117     if len(input_seq) < 2:
 118       continue
 119     rev_seq = data.build_reverse_sequence(input_seq)
 120     lm_seq = data.build_lm_sequence(input_seq)
 121     rev_lm_seq = data.build_lm_sequence(rev_seq)
 122     seq_ae_seq = data.build_seq_ae_sequence(input_seq)
 123     if doc.label is not None:
 124       # Used for sentiment classification.
 125       label_seq = data.build_labeled_sequence(
 126           input_seq,
 127           doc.label,
 128           label_gain=(FLAGS.label_gain and not doc.is_validation))
 129       bd_label_seq = data.build_labeled_sequence(
 130           data.build_bidirectional_seq(input_seq, rev_seq),
 131           doc.label,
 132           label_gain=(FLAGS.label_gain and not doc.is_validation))
 133       class_writer = writer_valid_class if doc.is_validation else writer_class
 134       bd_class_writer = (writer_bd_valid_class
 135                          if doc.is_validation else writer_bd_class)
 136       class_writer.write(label_seq.seq.SerializeToString())
 137       bd_class_writer.write(bd_label_seq.seq.SerializeToString())
 138 
 139     # Write
 140     lm_seq_ser = lm_seq.seq.SerializeToString()
 141     seq_ae_seq_ser = seq_ae_seq.seq.SerializeToString()
 142     writer_lm_all.write(lm_seq_ser)
 143     writer_seq_ae_all.write(seq_ae_seq_ser)
 144     if not doc.is_validation:
 145       writer_lm.write(lm_seq_ser)
 146       writer_rev_lm.write(rev_lm_seq.seq.SerializeToString())
 147       writer_seq_ae.write(seq_ae_seq_ser)
 148 
 149   # Close writers
 150   writer_lm.close()
 151   writer_seq_ae.close()
 152   writer_class.close()
 153   writer_valid_class.close()
 154   writer_rev_lm.close()
 155   writer_bd_class.close()
 156   writer_bd_valid_class.close()

 159 def generate_test_data(vocab_ids, writer_lm_all, writer_seq_ae_all):
 160   """Generates test data."""
 161   # Construct test data writers
 162   writer_lm = build_shuffling_tf_record_writer(data.TEST_LM)
 163   writer_rev_lm = build_shuffling_tf_record_writer(data.TEST_REV_LM)
 164   writer_seq_ae = build_shuffling_tf_record_writer(data.TEST_SA)
 165   writer_class = build_tf_record_writer(data.TEST_CLASS)
 166   writer_bd_class = build_shuffling_tf_record_writer(data.TEST_BD_CLASS)
 167 
 168   for doc in document_generators.documents(
 169       dataset='test', include_unlabeled=False, include_validation=True):
 170     input_seq = build_input_sequence(doc, vocab_ids)
 171     if len(input_seq) < 2:
 172       continue
 173     rev_seq = data.build_reverse_sequence(input_seq)
 174     lm_seq = data.build_lm_sequence(input_seq)
 175     rev_lm_seq = data.build_lm_sequence(rev_seq)
 176     seq_ae_seq = data.build_seq_ae_sequence(input_seq)
 177     label_seq = data.build_labeled_sequence(input_seq, doc.label)
 178     bd_label_seq = data.build_labeled_sequence(
 179         data.build_bidirectional_seq(input_seq, rev_seq), doc.label)
 180 
 181     # Write
 182     writer_class.write(label_seq.seq.SerializeToString())
 183     writer_bd_class.write(bd_label_seq.seq.SerializeToString())
 184     lm_seq_ser = lm_seq.seq.SerializeToString()
 185     seq_ae_seq_ser = seq_ae_seq.seq.SerializeToString()
 186     writer_lm.write(lm_seq_ser)
 187     writer_rev_lm.write(rev_lm_seq.seq.SerializeToString())
 188     writer_seq_ae.write(seq_ae_seq_ser)
 189     writer_lm_all.write(lm_seq_ser)
 190     writer_seq_ae_all.write(seq_ae_seq_ser)
 191 
 192   # Close test writers
 193   writer_lm.close()
 194   writer_rev_lm.close()
 195   writer_seq_ae.close()
 196   writer_class.close()
 197   writer_bd_class.close()

 200 def main(_):
 201   tf.logging.set_verbosity(tf.logging.INFO)
 202   tf.logging.info('Assigning vocabulary ids...')
 203   vocab_ids = make_vocab_ids(
 204       FLAGS.vocab_file or os.path.join(FLAGS.output_dir, 'vocab.txt'))
 205 
 206   with build_shuffling_tf_record_writer(data.ALL_LM) as writer_lm_all:
 207     with build_shuffling_tf_record_writer(data.ALL_SA) as writer_seq_ae_all:
 208 
 209       tf.logging.info('Generating training data...')
 210       generate_training_data(vocab_ids, writer_lm_all, writer_seq_ae_all)
 211 
 212       tf.logging.info('Generating test data...')
 213       generate_test_data(vocab_ids, writer_lm_all, writer_seq_ae_all)
 214 
 215 
 216 if __name__ == '__main__':
 217   tf.app.run()
