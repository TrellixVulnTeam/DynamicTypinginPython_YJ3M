  27 def cl_logits_subgraph(layer_sizes, input_size, num_classes, keep_prob=1.):
  28   """Construct multiple ReLU layers with dropout and a linear layer."""
  29   subgraph = K.models.Sequential(name='cl_logits')
  30   for i, layer_size in enumerate(layer_sizes):
  31     if i == 0:
  32       subgraph.add(
  33           K.layers.Dense(layer_size, activation='relu', input_dim=input_size))
  34     else:
  35       subgraph.add(K.layers.Dense(layer_size, activation='relu'))
  36 
  37     if keep_prob < 1.:
  38       subgraph.add(K.layers.Dropout(1. - keep_prob))
  39   subgraph.add(K.layers.Dense(1 if num_classes == 2 else num_classes))
  40   return subgraph

  43 class Embedding(K.layers.Layer):
  44   """Embedding layer with frequency-based normalization and dropout."""
  45 
  46   def __init__(self,
  47                vocab_size,
  48                embedding_dim,
  49                normalize=False,
  50                vocab_freqs=None,
  51                keep_prob=1.,
  52                **kwargs):
  53     self.vocab_size = vocab_size
  54     self.embedding_dim = embedding_dim
  55     self.normalized = normalize
  56     self.keep_prob = keep_prob
  57 
  58     if normalize:
  59       assert vocab_freqs is not None
  60       self.vocab_freqs = tf.constant(
  61           vocab_freqs, dtype=tf.float32, shape=(vocab_size, 1))
  62 
  63     super(Embedding, self).__init__(**kwargs)
  64 
  65   def build(self, input_shape):
  66     with tf.device('/cpu:0'):
  67       self.var = self.add_weight(
  68           shape=(self.vocab_size, self.embedding_dim),
  69           initializer=tf.random_uniform_initializer(-1., 1.),
  70           name='embedding',
  71           dtype=tf.float32)
  72 
  73     if self.normalized:
  74       self.var = self._normalize(self.var)
  75 
  76     super(Embedding, self).build(input_shape)
  77 
  78   def call(self, x):
  79     embedded = tf.nn.embedding_lookup(self.var, x)
  80     if self.keep_prob < 1.:
  81       shape = embedded.get_shape().as_list()
  82 
  83       # Use same dropout masks at each timestep with specifying noise_shape.
  84       # This slightly improves performance.
  85       # Please see https://arxiv.org/abs/1512.05287 for the theoretical
  86       # explanation.
  87       embedded = tf.nn.dropout(
  88           embedded, self.keep_prob, noise_shape=(shape[0], 1, shape[2]))
  89     return embedded
  90 
  91   def _normalize(self, emb):
  92     weights = self.vocab_freqs / tf.reduce_sum(self.vocab_freqs)
  93     mean = tf.reduce_sum(weights * emb, 0, keep_dims=True)
  94     var = tf.reduce_sum(weights * tf.pow(emb - mean, 2.), 0, keep_dims=True)
  95     stddev = tf.sqrt(1e-6 + var)
  96     return (emb - mean) / stddev

  99 class LSTM(object):
 100   """LSTM layer using dynamic_rnn.
 101 
 102   Exposes variables in `trainable_weights` property.
 103   """
 104 
 105   def __init__(self, cell_size, num_layers=1, keep_prob=1., name='LSTM'):
 106     self.cell_size = cell_size
 107     self.num_layers = num_layers
 108     self.keep_prob = keep_prob
 109     self.reuse = None
 110     self.trainable_weights = None
 111     self.name = name
 112 
 113   def __call__(self, x, initial_state, seq_length):
 114     with tf.variable_scope(self.name, reuse=self.reuse) as vs:
 115       cell = tf.contrib.rnn.MultiRNNCell([
 116           tf.contrib.rnn.BasicLSTMCell(
 117               self.cell_size,
 118               forget_bias=0.0,
 119               reuse=tf.get_variable_scope().reuse)
 120           for _ in xrange(self.num_layers)
 121       ])
 122 
 123       # shape(x) = (batch_size, num_timesteps, embedding_dim)
 124 
 125       lstm_out, next_state = tf.nn.dynamic_rnn(
 126           cell, x, initial_state=initial_state, sequence_length=seq_length)
 127 
 128       # shape(lstm_out) = (batch_size, timesteps, cell_size)
 129 
 130       if self.keep_prob < 1.:
 131         lstm_out = tf.nn.dropout(lstm_out, self.keep_prob)
 132 
 133       if self.reuse is None:
 134         self.trainable_weights = vs.global_variables()
 135 
 136     self.reuse = True
 137 
 138     return lstm_out, next_state

 141 class SoftmaxLoss(K.layers.Layer):
 142   """Softmax xentropy loss with candidate sampling."""
 143 
 144   def __init__(self,
 145                vocab_size,
 146                num_candidate_samples=-1,
 147                vocab_freqs=None,
 148                **kwargs):
 149     self.vocab_size = vocab_size
 150     self.num_candidate_samples = num_candidate_samples
 151     self.vocab_freqs = vocab_freqs
 152     super(SoftmaxLoss, self).__init__(**kwargs)
 153     self.multiclass_dense_layer = K.layers.Dense(self.vocab_size)
 154 
 155   def build(self, input_shape):
 156     input_shape = input_shape[0].as_list()
 157     with tf.device('/cpu:0'):
 158       self.lin_w = self.add_weight(
 159           shape=(input_shape[-1], self.vocab_size),
 160           name='lm_lin_w',
 161           initializer=K.initializers.glorot_uniform())
 162       self.lin_b = self.add_weight(
 163           shape=(self.vocab_size,),
 164           name='lm_lin_b',
 165           initializer=K.initializers.glorot_uniform())
 166       self.multiclass_dense_layer.build(input_shape)
 167 
 168     super(SoftmaxLoss, self).build(input_shape)
 169 
 170   def call(self, inputs):
 171     x, labels, weights = inputs
 172     if self.num_candidate_samples > -1:
 173       assert self.vocab_freqs is not None
 174       labels_reshaped = tf.reshape(labels, [-1])
 175       labels_reshaped = tf.expand_dims(labels_reshaped, -1)
 176       sampled = tf.nn.fixed_unigram_candidate_sampler(
 177           true_classes=labels_reshaped,
 178           num_true=1,
 179           num_sampled=self.num_candidate_samples,
 180           unique=True,
 181           range_max=self.vocab_size,
 182           unigrams=self.vocab_freqs)
 183       inputs_reshaped = tf.reshape(x, [-1, int(x.get_shape()[2])])
 184 
 185       lm_loss = tf.nn.sampled_softmax_loss(
 186           weights=tf.transpose(self.lin_w),
 187           biases=self.lin_b,
 188           labels=labels_reshaped,
 189           inputs=inputs_reshaped,
 190           num_sampled=self.num_candidate_samples,
 191           num_classes=self.vocab_size,
 192           sampled_values=sampled)
 193       lm_loss = tf.reshape(
 194           lm_loss,
 195           [int(x.get_shape()[0]), int(x.get_shape()[1])])
 196     else:
 197       logits = self.multiclass_dense_layer(x)
 198       lm_loss = tf.nn.sparse_softmax_cross_entropy_with_logits(
 199           logits=logits, labels=labels)
 200 
 201     lm_loss = tf.identity(
 202         tf.reduce_sum(lm_loss * weights) / _num_labels(weights),
 203         name='lm_xentropy_loss')
 204     return lm_loss

 207 def classification_loss(logits, labels, weights):
 208   """Computes cross entropy loss between logits and labels.
 209 
 210   Args:
 211     logits: 2-D [timesteps*batch_size, m] float tensor, where m=1 if
 212       num_classes=2, otherwise m=num_classes.
 213     labels: 1-D [timesteps*batch_size] integer tensor.
 214     weights: 1-D [timesteps*batch_size] float tensor.
 215 
 216   Returns:
 217     Loss scalar of type float.
 218   """
 219   inner_dim = logits.get_shape().as_list()[-1]
 220   with tf.name_scope('classifier_loss'):
 221     # Logistic loss
 222     if inner_dim == 1:
 223       loss = tf.nn.sigmoid_cross_entropy_with_logits(
 224           logits=tf.squeeze(logits, -1), labels=tf.cast(labels, tf.float32))
 225     # Softmax loss
 226     else:
 227       loss = tf.nn.sparse_softmax_cross_entropy_with_logits(
 228           logits=logits, labels=labels)
 229 
 230     num_lab = _num_labels(weights)
 231     tf.summary.scalar('num_labels', num_lab)
 232     return tf.identity(
 233         tf.reduce_sum(weights * loss) / num_lab, name='classification_xentropy')

 236 def accuracy(logits, targets, weights):
 237   """Computes prediction accuracy.
 238 
 239   Args:
 240     logits: 2-D classifier logits [timesteps*batch_size, num_classes]
 241     targets: 1-D [timesteps*batch_size] integer tensor.
 242     weights: 1-D [timesteps*batch_size] float tensor.
 243 
 244   Returns:
 245     Accuracy: float scalar.
 246   """
 247   with tf.name_scope('accuracy'):
 248     eq = tf.cast(tf.equal(predictions(logits), targets), tf.float32)
 249     return tf.identity(
 250         tf.reduce_sum(weights * eq) / _num_labels(weights), name='accuracy')
 251

 253 def predictions(logits):
 254   """Class prediction from logits."""
 255   inner_dim = logits.get_shape().as_list()[-1]
 256   with tf.name_scope('predictions'):
 257     # For binary classification
 258     if inner_dim == 1:
 259       pred = tf.cast(tf.greater(tf.squeeze(logits, -1), 0.), tf.int64)
 260     # For multi-class classification
 261     else:
 262       pred = tf.argmax(logits, 2)
 263     return pred

 266 def _num_labels(weights):
 267   """Number of 1's in weights. Returns 1. if 0."""
 268   num_labels = tf.reduce_sum(weights)
 269   num_labels = tf.where(tf.equal(num_labels, 0.), 1., num_labels)
 270   return num_labels

 273 def optimize(loss,
 274              global_step,
 275              max_grad_norm,
 276              lr,
 277              lr_decay,
 278              sync_replicas=False,
 279              replicas_to_aggregate=1,
 280              task_id=0):
 281   """Builds optimization graph.
 282 
 283   * Creates an optimizer, and optionally wraps with SyncReplicasOptimizer
 284   * Computes, clips, and applies gradients
 285   * Maintains moving averages for all trainable variables
 286   * Summarizes variables and gradients
 287 
 288   Args:
 289     loss: scalar loss to minimize.
 290     global_step: integer scalar Variable.
 291     max_grad_norm: float scalar. Grads will be clipped to this value.
 292     lr: float scalar, learning rate.
 293     lr_decay: float scalar, learning rate decay rate.
 294     sync_replicas: bool, whether to use SyncReplicasOptimizer.
 295     replicas_to_aggregate: int, number of replicas to aggregate when using
 296       SyncReplicasOptimizer.
 297     task_id: int, id of the current task; used to ensure proper initialization
 298       of SyncReplicasOptimizer.
 299 
 300   Returns:
 301     train_op
 302   """
 303   with tf.name_scope('optimization'):
 304     # Compute gradients.
 305     tvars = tf.trainable_variables()
 306     grads = tf.gradients(
 307         loss,
 308         tvars,
 309         aggregation_method=tf.AggregationMethod.EXPERIMENTAL_ACCUMULATE_N)
 310 
 311     # Clip non-embedding grads
 312     non_embedding_grads_and_vars = [(g, v) for (g, v) in zip(grads, tvars)
 313                                     if 'embedding' not in v.op.name]
 314     embedding_grads_and_vars = [(g, v) for (g, v) in zip(grads, tvars)
 315                                 if 'embedding' in v.op.name]
 316 
 317     ne_grads, ne_vars = zip(*non_embedding_grads_and_vars)
 318     ne_grads, _ = tf.clip_by_global_norm(ne_grads, max_grad_norm)
 319     non_embedding_grads_and_vars = zip(ne_grads, ne_vars)
 320 
 321     grads_and_vars = embedding_grads_and_vars + list(non_embedding_grads_and_vars)
 322 
 323     # Summarize
 324     _summarize_vars_and_grads(grads_and_vars)
 325 
 326     # Decaying learning rate
 327     lr = tf.train.exponential_decay(
 328         lr, global_step, 1, lr_decay, staircase=True)
 329     tf.summary.scalar('learning_rate', lr)
 330     opt = tf.train.AdamOptimizer(lr)
 331 
 332     # Track the moving averages of all trainable variables.
 333     variable_averages = tf.train.ExponentialMovingAverage(0.999, global_step)
 334 
 335     # Apply gradients
 336     if sync_replicas:
 337       opt = tf.train.SyncReplicasOptimizer(
 338           opt,
 339           replicas_to_aggregate,
 340           variable_averages=variable_averages,
 341           variables_to_average=tvars,
 342           total_num_replicas=replicas_to_aggregate)
 343       apply_gradient_op = opt.apply_gradients(
 344           grads_and_vars, global_step=global_step)
 345       with tf.control_dependencies([apply_gradient_op]):
 346         train_op = tf.no_op(name='train_op')
 347 
 348       # Initialization ops
 349       tf.add_to_collection(tf.GraphKeys.QUEUE_RUNNERS,
 350                            opt.get_chief_queue_runner())
 351       if task_id == 0:  # Chief task
 352         local_init_op = opt.chief_init_op
 353         tf.add_to_collection('chief_init_op', opt.get_init_tokens_op())
 354       else:
 355         local_init_op = opt.local_step_init_op
 356       tf.add_to_collection('local_init_op', local_init_op)
 357       tf.add_to_collection('ready_for_local_init_op',
 358                            opt.ready_for_local_init_op)
 359     else:
 360       # Non-sync optimizer
 361       apply_gradient_op = opt.apply_gradients(grads_and_vars, global_step)
 362       with tf.control_dependencies([apply_gradient_op]):
 363         train_op = variable_averages.apply(tvars)
 364 
 365     return train_op

 368 def _summarize_vars_and_grads(grads_and_vars):
 369   tf.logging.info('Trainable variables:')
 370   tf.logging.info('-' * 60)
 371   for grad, var in grads_and_vars:
 372     tf.logging.info(var)
 373 
 374     def tag(name, v=var):
 375       return v.op.name + '_' + name
 376 
 377     # Variable summary
 378     mean = tf.reduce_mean(var)
 379     tf.summary.scalar(tag('mean'), mean)
 380     with tf.name_scope(tag('stddev')):
 381       stddev = tf.sqrt(tf.reduce_mean(tf.square(var - mean)))
 382     tf.summary.scalar(tag('stddev'), stddev)
 383     tf.summary.scalar(tag('max'), tf.reduce_max(var))
 384     tf.summary.scalar(tag('min'), tf.reduce_min(var))
 385     tf.summary.histogram(tag('histogram'), var)
 386 
 387     # Gradient summary
 388     if grad is not None:
 389       if isinstance(grad, tf.IndexedSlices):
 390         grad_values = grad.values
 391       else:
 392         grad_values = grad
 393 
 394       tf.summary.histogram(tag('gradient'), grad_values)
 395       tf.summary.scalar(tag('gradient_norm'), tf.global_norm([grad_values]))
 396     else:
 397       tf.logging.info('Var %s has no gradient', var.op.name)
