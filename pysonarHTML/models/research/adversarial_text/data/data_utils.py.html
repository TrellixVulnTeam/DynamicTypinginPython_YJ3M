  50 class ShufflingTFRecordWriter(object):
  51   """Thin wrapper around TFRecordWriter that shuffles records."""
  52 
  53   def __init__(self, path):
  54     self._path = path
  55     self._records = []
  56     self._closed = False
  57 
  58   def write(self, record):
  59     assert not self._closed
  60     self._records.append(record)
  61 
  62   def close(self):
  63     assert not self._closed
  64     random.shuffle(self._records)
  65     with tf.python_io.TFRecordWriter(self._path) as f:
  66       for record in self._records:
  67         f.write(record)
  68     self._closed = True
  69 
  70   def __enter__(self):
  71     return self
  72 
  73   def __exit__(self, unused_type, unused_value, unused_traceback):
  74     self.close()

  77 class Timestep(object):
  78   """Represents a single timestep in a SequenceWrapper."""
  79 
  80   def __init__(self, token, label, weight, multivalent_tokens=False):
  81     """Constructs Timestep from empty Features."""
  82     self._token = token
  83     self._label = label
  84     self._weight = weight
  85     self._multivalent_tokens = multivalent_tokens
  86     self._fill_with_defaults()
  87 
  88   @property
  89   def token(self):
  90     if self._multivalent_tokens:
  91       raise TypeError('Timestep may contain multiple values; use `tokens`')
  92     return self._token.int64_list.value[0]
  93 
  94   @property
  95   def tokens(self):
  96     return self._token.int64_list.value
  97 
  98   @property
  99   def label(self):
 100     return self._label.int64_list.value[0]
 101 
 102   @property
 103   def weight(self):
 104     return self._weight.float_list.value[0]
 105 
 106   def set_token(self, token):
 107     if self._multivalent_tokens:
 108       raise TypeError('Timestep may contain multiple values; use `add_token`')
 109     self._token.int64_list.value[0] = token
 110     return self
 111 
 112   def add_token(self, token):
 113     self._token.int64_list.value.append(token)
 114     return self
 115 
 116   def set_label(self, label):
 117     self._label.int64_list.value[0] = label
 118     return self
 119 
 120   def set_weight(self, weight):
 121     self._weight.float_list.value[0] = weight
 122     return self
 123 
 124   def copy_from(self, timestep):
 125     self.set_token(timestep.token).set_label(timestep.label).set_weight(
 126         timestep.weight)
 127     return self
 128 
 129   def _fill_with_defaults(self):
 130     if not self._multivalent_tokens:
 131       self._token.int64_list.value.append(0)
 132     self._label.int64_list.value.append(0)
 133     self._weight.float_list.value.append(0.0)

 136 class SequenceWrapper(object):
 137   """Wrapper around tf.SequenceExample."""
 138 
 139   F_TOKEN_ID = 'token_id'
 140   F_LABEL = 'label'
 141   F_WEIGHT = 'weight'
 142 
 143   def __init__(self, multivalent_tokens=False):
 144     self._seq = tf.train.SequenceExample()
 145     self._flist = self._seq.feature_lists.feature_list
 146     self._timesteps = []
 147     self._multivalent_tokens = multivalent_tokens
 148 
 149   @property
 150   def seq(self):
 151     return self._seq
 152 
 153   @property
 154   def multivalent_tokens(self):
 155     return self._multivalent_tokens
 156 
 157   @property
 158   def _tokens(self):
 159     return self._flist[SequenceWrapper.F_TOKEN_ID].feature
 160 
 161   @property
 162   def _labels(self):
 163     return self._flist[SequenceWrapper.F_LABEL].feature
 164 
 165   @property
 166   def _weights(self):
 167     return self._flist[SequenceWrapper.F_WEIGHT].feature
 168 
 169   def add_timestep(self):
 170     timestep = Timestep(
 171         self._tokens.add(),
 172         self._labels.add(),
 173         self._weights.add(),
 174         multivalent_tokens=self._multivalent_tokens)
 175     self._timesteps.append(timestep)
 176     return timestep
 177 
 178   def __iter__(self):
 179     for timestep in self._timesteps:
 180       yield timestep
 181 
 182   def __len__(self):
 183     return len(self._timesteps)
 184 
 185   def __getitem__(self, idx):
 186     return self._timesteps[idx]

 189 def build_reverse_sequence(seq):
 190   """Builds a sequence that is the reverse of the input sequence."""
 191   reverse_seq = SequenceWrapper()
 192 
 193   # Copy all but last timestep
 194   for timestep in reversed(seq[:-1]):
 195     reverse_seq.add_timestep().copy_from(timestep)
 196 
 197   # Copy final timestep
 198   reverse_seq.add_timestep().copy_from(seq[-1])
 199 
 200   return reverse_seq
 201

 203 def build_bidirectional_seq(seq, rev_seq):
 204   bidir_seq = SequenceWrapper(multivalent_tokens=True)
 205   for forward_ts, reverse_ts in zip(seq, rev_seq):
 206     bidir_seq.add_timestep().add_token(forward_ts.token).add_token(
 207         reverse_ts.token)
 208 
 209   return bidir_seq

 212 def build_lm_sequence(seq):
 213   """Builds language model sequence from input sequence.
 214 
 215   Args:
 216     seq: SequenceWrapper.
 217 
 218   Returns:
 219     SequenceWrapper with `seq` tokens copied over to output sequence tokens and
 220     labels (offset by 1, i.e. predict next token) with weights set to 1.0,
 221     except for <eos> token.
 222   """
 223   lm_seq = SequenceWrapper()
 224   for i, timestep in enumerate(seq):
 225     if i == len(seq) - 1:
 226       lm_seq.add_timestep().set_token(timestep.token).set_label(
 227           seq[i].token).set_weight(0.0)
 228     else:
 229       lm_seq.add_timestep().set_token(timestep.token).set_label(
 230           seq[i + 1].token).set_weight(1.0)
 231   return lm_seq

 234 def build_seq_ae_sequence(seq):
 235   """Builds seq_ae sequence from input sequence.
 236 
 237   Args:
 238     seq: SequenceWrapper.
 239 
 240   Returns:
 241     SequenceWrapper with `seq` inputs copied and concatenated, and with labels
 242     copied in on the right-hand (i.e. decoder) side with weights set to 1.0.
 243     The new sequence will have length `len(seq) * 2 - 1`, as the last timestep
 244     of the encoder section and the first step of the decoder section will
 245     overlap.
 246   """
 247   seq_ae_seq = SequenceWrapper()
 248 
 249   for i in range(len(seq) * 2 - 1):
 250     ts = seq_ae_seq.add_timestep()
 251 
 252     if i < len(seq) - 1:
 253       # Encoder
 254       ts.set_token(seq[i].token)
 255     elif i == len(seq) - 1:
 256       # Transition step
 257       ts.set_token(seq[i].token)
 258       ts.set_label(seq[0].token)
 259       ts.set_weight(1.0)
 260     else:
 261       # Decoder
 262       ts.set_token(seq[i % len(seq)].token)
 263       ts.set_label(seq[(i + 1) % len(seq)].token)
 264       ts.set_weight(1.0)
 265 
 266   return seq_ae_seq

 269 def build_labeled_sequence(seq, class_label, label_gain=False):
 270   """Builds labeled sequence from input sequence.
 271 
 272   Args:
 273     seq: SequenceWrapper.
 274     class_label: integer, starting from 0.
 275     label_gain: bool. If True, class_label will be put on every timestep and
 276       weight will increase linearly from 0 to 1.
 277 
 278   Returns:
 279     SequenceWrapper with `seq` copied in and `class_label` added as label to
 280     final timestep.
 281   """
 282   label_seq = SequenceWrapper(multivalent_tokens=seq.multivalent_tokens)
 283 
 284   # Copy sequence without labels
 285   seq_len = len(seq)
 286   final_timestep = None
 287   for i, timestep in enumerate(seq):
 288     label_timestep = label_seq.add_timestep()
 289     if seq.multivalent_tokens:
 290       for token in timestep.tokens:
 291         label_timestep.add_token(token)
 292     else:
 293       label_timestep.set_token(timestep.token)
 294     if label_gain:
 295       label_timestep.set_label(int(class_label))
 296       weight = 1.0 if seq_len < 2 else float(i) / (seq_len - 1)
 297       label_timestep.set_weight(weight)
 298     if i == (seq_len - 1):
 299       final_timestep = label_timestep
 300 
 301   # Edit final timestep to have class label and weight = 1.
 302   final_timestep.set_label(int(class_label)).set_weight(1.0)
 303 
 304   return label_seq

 307 def split_by_punct(segment):
 308   """Splits str segment by punctuation, filters our empties and spaces."""
 309   return [s for s in re.split(r'\W+', segment) if s and not s.isspace()]

 312 def sort_vocab_by_frequency(vocab_freq_map):
 313   """Sorts vocab_freq_map by count.
 314 
 315   Args:
 316     vocab_freq_map: dict<str term, int count>, vocabulary terms with counts.
 317 
 318   Returns:
 319     list<tuple<str term, int count>> sorted by count, descending.
 320   """
 321   return sorted(
 322       vocab_freq_map.items(), key=operator.itemgetter(1), reverse=True)

 325 def write_vocab_and_frequency(ordered_vocab_freqs, output_dir):
 326   """Writes ordered_vocab_freqs into vocab.txt and vocab_freq.txt."""
 327   tf.gfile.MakeDirs(output_dir)
 328   with open(os.path.join(output_dir, 'vocab.txt'), 'w', encoding='utf-8') as vocab_f:
 329     with open(os.path.join(output_dir, 'vocab_freq.txt'), 'w', encoding='utf-8') as freq_f:
 330       for word, freq in ordered_vocab_freqs:
 331         vocab_f.write('{}\n'.format(word))
 332         freq_f.write('{}\n'.format(freq))
