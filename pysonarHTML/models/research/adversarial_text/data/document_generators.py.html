  73 def documents(dataset='train',
  74               include_unlabeled=False,
  75               include_validation=False):
  76   """Generates Documents based on FLAGS.dataset.
  77 
  78   Args:
  79     dataset: str, identifies folder within IMDB data directory, test or train.
  80     include_unlabeled: bool, whether to include the unsup directory. Only valid
  81       when dataset=train.
  82     include_validation: bool, whether to include validation data.
  83 
  84   Yields:
  85     Document
  86 
  87   Raises:
  88     ValueError: if include_unlabeled is true but dataset is not 'train'
  89   """
  90 
  91   if include_unlabeled and dataset != 'train':
  92     raise ValueError('If include_unlabeled=True, must use train dataset')
  93 
  94   # Set the random seed so that we have the same validation set when running
  95   # gen_data and gen_vocab.
  96   random.seed(302)
  97 
  98   ds = FLAGS.dataset
  99   if ds == 'imdb':
 100     docs_gen = imdb_documents
 101   elif ds == 'dbpedia':
 102     docs_gen = dbpedia_documents
 103   elif ds == 'rcv1':
 104     docs_gen = rcv1_documents
 105   elif ds == 'rt':
 106     docs_gen = rt_documents
 107   else:
 108     raise ValueError('Unrecognized dataset %s' % FLAGS.dataset)
 109 
 110   for doc in docs_gen(dataset, include_unlabeled, include_validation):
 111     yield doc

 114 def tokens(doc):
 115   """Given a Document, produces character or word tokens.
 116 
 117   Tokens can be either characters, or word-level tokens (unigrams and/or
 118   bigrams).
 119 
 120   Args:
 121     doc: Document to produce tokens from.
 122 
 123   Yields:
 124     token
 125 
 126   Raises:
 127     ValueError: if all FLAGS.{output_unigrams, output_bigrams, output_char}
 128       are False.
 129   """
 130   if not (FLAGS.output_unigrams or FLAGS.output_bigrams or FLAGS.output_char):
 131     raise ValueError(
 132         'At least one of {FLAGS.output_unigrams, FLAGS.output_bigrams, '
 133         'FLAGS.output_char} must be true')
 134 
 135   content = doc.content.strip()
 136   if FLAGS.lowercase:
 137     content = content.lower()
 138 
 139   if FLAGS.output_char:
 140     for char in content:
 141       yield char
 142 
 143   else:
 144     tokens_ = data_utils.split_by_punct(content)
 145     for i, token in enumerate(tokens_):
 146       if FLAGS.output_unigrams:
 147         yield token
 148 
 149       if FLAGS.output_bigrams:
 150         previous_token = (tokens_[i - 1] if i > 0 else data_utils.EOS_TOKEN)
 151         bigram = '_'.join([previous_token, token])
 152         yield bigram
 153         if (i + 1) == len(tokens_):
 154           bigram = '_'.join([token, data_utils.EOS_TOKEN])
 155           yield bigram

 158 def imdb_documents(dataset='train',
 159                    include_unlabeled=False,
 160                    include_validation=False):
 161   """Generates Documents for IMDB dataset.
 162 
 163   Data from http://ai.stanford.edu/~amaas/data/sentiment/
 164 
 165   Args:
 166     dataset: str, identifies folder within IMDB data directory, test or train.
 167     include_unlabeled: bool, whether to include the unsup directory. Only valid
 168       when dataset=train.
 169     include_validation: bool, whether to include validation data.
 170 
 171   Yields:
 172     Document
 173 
 174   Raises:
 175     ValueError: if FLAGS.imdb_input_dir is empty.
 176   """
 177   if not FLAGS.imdb_input_dir:
 178     raise ValueError('Must provide FLAGS.imdb_input_dir')
 179 
 180   tf.logging.info('Generating IMDB documents...')
 181 
 182   def check_is_validation(filename, class_label):
 183     if class_label is None:
 184       return False
 185     file_idx = int(filename.split('_')[0])
 186     is_pos_valid = (class_label and
 187                     file_idx >= FLAGS.imdb_validation_pos_start_id)
 188     is_neg_valid = (not class_label and
 189                     file_idx >= FLAGS.imdb_validation_neg_start_id)
 190     return is_pos_valid or is_neg_valid
 191 
 192   dirs = [(dataset + '/pos', True), (dataset + '/neg', False)]
 193   if include_unlabeled:
 194     dirs.append(('train/unsup', None))
 195 
 196   for d, class_label in dirs:
 197     for filename in os.listdir(os.path.join(FLAGS.imdb_input_dir, d)):
 198       is_validation = check_is_validation(filename, class_label)
 199       if is_validation and not include_validation:
 200         continue
 201 
 202       with open(os.path.join(FLAGS.imdb_input_dir, d, filename), encoding='utf-8') as imdb_f:
 203         content = imdb_f.read()
 204       yield Document(
 205           content=content,
 206           is_validation=is_validation,
 207           is_test=False,
 208           label=class_label,
 209           add_tokens=True)
 210 
 211   if FLAGS.amazon_unlabeled_input_file and include_unlabeled:
 212     with open(FLAGS.amazon_unlabeled_input_file, encoding='utf-8') as rt_f:
 213       for content in rt_f:
 214         yield Document(
 215             content=content,
 216             is_validation=False,
 217             is_test=False,
 218             label=None,
 219             add_tokens=False)

 222 def dbpedia_documents(dataset='train',
 223                       include_unlabeled=False,
 224                       include_validation=False):
 225   """Generates Documents for DBpedia dataset.
 226 
 227   Dataset linked to at https://github.com/zhangxiangxiao/Crepe.
 228 
 229   Args:
 230     dataset: str, identifies the csv file within the DBpedia data directory,
 231       test or train.
 232     include_unlabeled: bool, unused.
 233     include_validation: bool, whether to include validation data, which is a
 234       randomly selected 10% of the data.
 235 
 236   Yields:
 237     Document
 238 
 239   Raises:
 240     ValueError: if FLAGS.dbpedia_input_dir is empty.
 241   """
 242   del include_unlabeled
 243 
 244   if not FLAGS.dbpedia_input_dir:
 245     raise ValueError('Must provide FLAGS.dbpedia_input_dir')
 246 
 247   tf.logging.info('Generating DBpedia documents...')
 248 
 249   with open(os.path.join(FLAGS.dbpedia_input_dir, dataset + '.csv')) as db_f:
 250     reader = csv.reader(db_f)
 251     for row in reader:
 252       # 10% of the data is randomly held out
 253       is_validation = random.randint(1, 10) == 1
 254       if is_validation and not include_validation:
 255         continue
 256 
 257       content = row[1] + ' ' + row[2]
 258       yield Document(
 259           content=content,
 260           is_validation=is_validation,
 261           is_test=False,
 262           label=int(row[0]) - 1,  # Labels should start from 0
 263           add_tokens=True)

 266 def rcv1_documents(dataset='train',
 267                    include_unlabeled=True,
 268                    include_validation=False):
 269   # pylint:disable=line-too-long
 270   """Generates Documents for Reuters Corpus (rcv1) dataset.
 271 
 272   Dataset described at
 273   http://www.ai.mit.edu/projects/jmlr/papers/volume5/lewis04a/lyrl2004_rcv1v2_README.htm
 274 
 275   Args:
 276     dataset: str, identifies the csv file within the rcv1 data directory.
 277     include_unlabeled: bool, whether to include the unlab file. Only valid
 278       when dataset=train.
 279     include_validation: bool, whether to include validation data, which is a
 280       randomly selected 10% of the data.
 281 
 282   Yields:
 283     Document
 284 
 285   Raises:
 286     ValueError: if FLAGS.rcv1_input_dir is empty.
 287   """
 288   # pylint:enable=line-too-long
 289 
 290   if not FLAGS.rcv1_input_dir:
 291     raise ValueError('Must provide FLAGS.rcv1_input_dir')
 292 
 293   tf.logging.info('Generating rcv1 documents...')
 294 
 295   datasets = [dataset]
 296   if include_unlabeled:
 297     if dataset == 'train':
 298       datasets.append('unlab')
 299   for dset in datasets:
 300     with open(os.path.join(FLAGS.rcv1_input_dir, dset + '.csv')) as db_f:
 301       reader = csv.reader(db_f)
 302       for row in reader:
 303         # 10% of the data is randomly held out
 304         is_validation = random.randint(1, 10) == 1
 305         if is_validation and not include_validation:
 306           continue
 307 
 308         content = row[1]
 309         yield Document(
 310             content=content,
 311             is_validation=is_validation,
 312             is_test=False,
 313             label=int(row[0]),
 314             add_tokens=True)

 317 def rt_documents(dataset='train',
 318                  include_unlabeled=True,
 319                  include_validation=False):
 320   # pylint:disable=line-too-long
 321   """Generates Documents for the Rotten Tomatoes dataset.
 322 
 323   Dataset available at http://www.cs.cornell.edu/people/pabo/movie-review-data/
 324   In this dataset, amazon reviews are used for the unlabeled data.
 325 
 326   Args:
 327     dataset: str, identifies the data subdirectory.
 328     include_unlabeled: bool, whether to include the unlabeled data. Only valid
 329       when dataset=train.
 330     include_validation: bool, whether to include validation data, which is a
 331       randomly selected 10% of the data.
 332 
 333   Yields:
 334     Document
 335 
 336   Raises:
 337     ValueError: if FLAGS.rt_input_dir is empty.
 338   """
 339   # pylint:enable=line-too-long
 340 
 341   if not FLAGS.rt_input_dir:
 342     raise ValueError('Must provide FLAGS.rt_input_dir')
 343 
 344   tf.logging.info('Generating rt documents...')
 345 
 346   data_files = []
 347   input_filenames = os.listdir(FLAGS.rt_input_dir)
 348   for inp_fname in input_filenames:
 349     if inp_fname.endswith('.pos'):
 350       data_files.append((os.path.join(FLAGS.rt_input_dir, inp_fname), True))
 351     elif inp_fname.endswith('.neg'):
 352       data_files.append((os.path.join(FLAGS.rt_input_dir, inp_fname), False))
 353   if include_unlabeled and FLAGS.amazon_unlabeled_input_file:
 354     data_files.append((FLAGS.amazon_unlabeled_input_file, None))
 355 
 356   for filename, class_label in data_files:
 357     with open(filename) as rt_f:
 358       for content in rt_f:
 359         if class_label is None:
 360           # Process Amazon Review data for unlabeled dataset
 361           if content.startswith('review/text'):
 362             yield Document(
 363                 content=content,
 364                 is_validation=False,
 365                 is_test=False,
 366                 label=None,
 367                 add_tokens=False)
 368         else:
 369           # 10% of the data is randomly held out for the validation set and
 370           # another 10% of it is randomly held out for the test set
 371           random_int = random.randint(1, 10)
 372           is_validation = random_int == 1
 373           is_test = random_int == 2
 374           if (is_test and dataset != 'test') or (is_validation and
 375                                                  not include_validation):
 376             continue
 377 
 378           yield Document(
 379               content=content,
 380               is_validation=is_validation,
 381               is_test=is_test,
 382               label=class_label,
 383               add_tokens=True)
