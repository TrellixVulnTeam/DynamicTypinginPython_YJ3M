  29 class SequenceWrapperTest(tf.test.TestCase):
  30 
  31   def testDefaultTimesteps(self):
  32     seq = data.SequenceWrapper()
  33     t1 = seq.add_timestep()
  34     _ = seq.add_timestep()
  35     self.assertEqual(len(seq), 2)
  36 
  37     self.assertEqual(t1.weight, 0.0)
  38     self.assertEqual(t1.label, 0)
  39     self.assertEqual(t1.token, 0)
  40 
  41   def testSettersAndGetters(self):
  42     ts = data.SequenceWrapper().add_timestep()
  43     ts.set_token(3)
  44     ts.set_label(4)
  45     ts.set_weight(2.0)
  46     self.assertEqual(ts.token, 3)
  47     self.assertEqual(ts.label, 4)
  48     self.assertEqual(ts.weight, 2.0)
  49 
  50   def testTimestepIteration(self):
  51     seq = data.SequenceWrapper()
  52     seq.add_timestep().set_token(0)
  53     seq.add_timestep().set_token(1)
  54     seq.add_timestep().set_token(2)
  55     for i, ts in enumerate(seq):
  56       self.assertEqual(ts.token, i)
  57 
  58   def testFillsSequenceExampleCorrectly(self):
  59     seq = data.SequenceWrapper()
  60     seq.add_timestep().set_token(1).set_label(2).set_weight(3.0)
  61     seq.add_timestep().set_token(10).set_label(20).set_weight(30.0)
  62 
  63     seq_ex = seq.seq
  64     fl = seq_ex.feature_lists.feature_list
  65     fl_token = fl[data.SequenceWrapper.F_TOKEN_ID].feature
  66     fl_label = fl[data.SequenceWrapper.F_LABEL].feature
  67     fl_weight = fl[data.SequenceWrapper.F_WEIGHT].feature
  68     _ = [self.assertEqual(len(f), 2) for f in [fl_token, fl_label, fl_weight]]
  69     self.assertAllEqual([f.int64_list.value[0] for f in fl_token], [1, 10])
  70     self.assertAllEqual([f.int64_list.value[0] for f in fl_label], [2, 20])
  71     self.assertAllEqual([f.float_list.value[0] for f in fl_weight], [3.0, 30.0])

  74 class DataUtilsTest(tf.test.TestCase):
  75 
  76   def testSplitByPunct(self):
  77     output = data.split_by_punct(
  78         'hello! world, i\'ve been\nwaiting\tfor\ryou for.a long time')
  79     expected = [
  80         'hello', 'world', 'i', 've', 'been', 'waiting', 'for', 'you', 'for',
  81         'a', 'long', 'time'
  82     ]
  83     self.assertListEqual(output, expected)

  85   def _buildDummySequence(self):
  86     seq = data.SequenceWrapper()
  87     for i in range(10):
  88       seq.add_timestep().set_token(i)
  89     return seq
  90 
  91   def testBuildLMSeq(self):
  92     seq = self._buildDummySequence()
  93     lm_seq = data.build_lm_sequence(seq)
  94     for i, ts in enumerate(lm_seq):
  95       # For end of sequence, the token and label should be same, and weight
  96       # should be 0.0.
  97       if i == len(lm_seq) - 1:
  98         self.assertEqual(ts.token, i)
  99         self.assertEqual(ts.label, i)
 100         self.assertEqual(ts.weight, 0.0)
 101       else:
 102         self.assertEqual(ts.token, i)
 103         self.assertEqual(ts.label, i + 1)
 104         self.assertEqual(ts.weight, 1.0)
 105 
 106   def testBuildSAESeq(self):
 107     seq = self._buildDummySequence()
 108     sa_seq = data.build_seq_ae_sequence(seq)
 109 
 110     self.assertEqual(len(sa_seq), len(seq) * 2 - 1)
 111 
 112     # Tokens should be sequence twice, minus the EOS token at the end
 113     for i, ts in enumerate(sa_seq):
 114       self.assertEqual(ts.token, seq[i % 10].token)
 115 
 116     # Weights should be len-1 0.0's and len 1.0's.
 117     for i in range(len(seq) - 1):
 118       self.assertEqual(sa_seq[i].weight, 0.0)
 119     for i in range(len(seq) - 1, len(sa_seq)):
 120       self.assertEqual(sa_seq[i].weight, 1.0)
 121 
 122     # Labels should be len-1 0's, and then the sequence
 123     for i in range(len(seq) - 1):
 124       self.assertEqual(sa_seq[i].label, 0)
 125     for i in range(len(seq) - 1, len(sa_seq)):
 126       self.assertEqual(sa_seq[i].label, seq[i - (len(seq) - 1)].token)

 128   def testBuildLabelSeq(self):
 129     seq = self._buildDummySequence()
 130     eos_id = len(seq) - 1
 131     label_seq = data.build_labeled_sequence(seq, True)
 132     for i, ts in enumerate(label_seq[:-1]):
 133       self.assertEqual(ts.token, i)
 134       self.assertEqual(ts.label, 0)
 135       self.assertEqual(ts.weight, 0.0)
 136 
 137     final_timestep = label_seq[-1]
 138     self.assertEqual(final_timestep.token, eos_id)
 139     self.assertEqual(final_timestep.label, 1)
 140     self.assertEqual(final_timestep.weight, 1.0)
 141 
 142   def testBuildBidirLabelSeq(self):
 143     seq = self._buildDummySequence()
 144     reverse_seq = data.build_reverse_sequence(seq)
 145     bidir_seq = data.build_bidirectional_seq(seq, reverse_seq)
 146     label_seq = data.build_labeled_sequence(bidir_seq, True)
 147 
 148     for (i, ts), j in zip(
 149         enumerate(label_seq[:-1]), reversed(range(len(seq) - 1))):
 150       self.assertAllEqual(ts.tokens, [i, j])
 151       self.assertEqual(ts.label, 0)
 152       self.assertEqual(ts.weight, 0.0)
 153 
 154     final_timestep = label_seq[-1]
 155     eos_id = len(seq) - 1
 156     self.assertAllEqual(final_timestep.tokens, [eos_id, eos_id])
 157     self.assertEqual(final_timestep.label, 1)
 158     self.assertEqual(final_timestep.weight, 1.0)
 159 
 160   def testReverseSeq(self):
 161     seq = self._buildDummySequence()
 162     reverse_seq = data.build_reverse_sequence(seq)
 163     for i, ts in enumerate(reversed(reverse_seq[:-1])):
 164       self.assertEqual(ts.token, i)
 165       self.assertEqual(ts.label, 0)
 166       self.assertEqual(ts.weight, 0.0)
 167 
 168     final_timestep = reverse_seq[-1]
 169     eos_id = len(seq) - 1
 170     self.assertEqual(final_timestep.token, eos_id)
 171     self.assertEqual(final_timestep.label, 0)
 172     self.assertEqual(final_timestep.weight, 0.0)

 190   def testLabelGain(self):
 191     seq = self._buildDummySequence()
 192     label_seq = data.build_labeled_sequence(seq, True, label_gain=True)
 193     for i, ts in enumerate(label_seq):
 194       self.assertEqual(ts.token, i)
 195       self.assertEqual(ts.label, 1)
 196       self.assertNear(ts.weight, float(i) / (len(seq) - 1), 1e-3)
