  52 def random_perturbation_loss(embedded, length, loss_fn):
  53   """Adds noise to embeddings and recomputes classification loss."""
  54   noise = tf.random_normal(shape=tf.shape(embedded))
  55   perturb = _scale_l2(_mask_by_length(noise, length), FLAGS.perturb_norm_length)
  56   return loss_fn(embedded + perturb)

  59 def adversarial_loss(embedded, loss, loss_fn):
  60   """Adds gradient to embedding and recomputes classification loss."""
  61   grad, = tf.gradients(
  62       loss,
  63       embedded,
  64       aggregation_method=tf.AggregationMethod.EXPERIMENTAL_ACCUMULATE_N)
  65   grad = tf.stop_gradient(grad)
  66   perturb = _scale_l2(grad, FLAGS.perturb_norm_length)
  67   return loss_fn(embedded + perturb)

  70 def virtual_adversarial_loss(logits, embedded, inputs,
  71                              logits_from_embedding_fn):
  72   """Virtual adversarial loss.
  73 
  74   Computes virtual adversarial perturbation by finite difference method and
  75   power iteration, adds it to the embedding, and computes the KL divergence
  76   between the new logits and the original logits.
  77 
  78   Args:
  79     logits: 3-D float Tensor, [batch_size, num_timesteps, m], where m=1 if
  80       num_classes=2, otherwise m=num_classes.
  81     embedded: 3-D float Tensor, [batch_size, num_timesteps, embedding_dim].
  82     inputs: VatxtInput.
  83     logits_from_embedding_fn: callable that takes embeddings and returns
  84       classifier logits.
  85 
  86   Returns:
  87     kl: float scalar.
  88   """
  89   # Stop gradient of logits. See https://arxiv.org/abs/1507.00677 for details.
  90   logits = tf.stop_gradient(logits)
  91 
  92   # Only care about the KL divergence on the final timestep.
  93   weights = inputs.eos_weights
  94   assert weights is not None
  95   if FLAGS.single_label:
  96     indices = tf.stack([tf.range(FLAGS.batch_size), inputs.length - 1], 1)
  97     weights = tf.expand_dims(tf.gather_nd(inputs.eos_weights, indices), 1)
  98 
  99   # Initialize perturbation with random noise.
 100   # shape(embedded) = (batch_size, num_timesteps, embedding_dim)
 101   d = tf.random_normal(shape=tf.shape(embedded))
 102 
 103   # Perform finite difference method and power iteration.
 104   # See Eq.(8) in the paper http://arxiv.org/pdf/1507.00677.pdf,
 105   # Adding small noise to input and taking gradient with respect to the noise
 106   # corresponds to 1 power iteration.
 107   for _ in xrange(FLAGS.num_power_iteration):
 108     d = _scale_l2(
 109         _mask_by_length(d, inputs.length), FLAGS.small_constant_for_finite_diff)
 110 
 111     d_logits = logits_from_embedding_fn(embedded + d)
 112     kl = _kl_divergence_with_logits(logits, d_logits, weights)
 113     d, = tf.gradients(
 114         kl,
 115         d,
 116         aggregation_method=tf.AggregationMethod.EXPERIMENTAL_ACCUMULATE_N)
 117     d = tf.stop_gradient(d)
 118 
 119   perturb = _scale_l2(d, FLAGS.perturb_norm_length)
 120   vadv_logits = logits_from_embedding_fn(embedded + perturb)
 121   return _kl_divergence_with_logits(logits, vadv_logits, weights)

 124 def random_perturbation_loss_bidir(embedded, length, loss_fn):
 125   """Adds noise to embeddings and recomputes classification loss."""
 126   noise = [tf.random_normal(shape=tf.shape(emb)) for emb in embedded]
 127   masked = [_mask_by_length(n, length) for n in noise]
 128   scaled = [_scale_l2(m, FLAGS.perturb_norm_length) for m in masked]
 129   return loss_fn([e + s for (e, s) in zip(embedded, scaled)])

 132 def adversarial_loss_bidir(embedded, loss, loss_fn):
 133   """Adds gradient to embeddings and recomputes classification loss."""
 134   grads = tf.gradients(
 135       loss,
 136       embedded,
 137       aggregation_method=tf.AggregationMethod.EXPERIMENTAL_ACCUMULATE_N)
 138   adv_exs = [
 139       emb + _scale_l2(tf.stop_gradient(g), FLAGS.perturb_norm_length)
 140       for emb, g in zip(embedded, grads)
 141   ]
 142   return loss_fn(adv_exs)

 145 def virtual_adversarial_loss_bidir(logits, embedded, inputs,
 146                                    logits_from_embedding_fn):
 147   """Virtual adversarial loss for bidirectional models."""
 148   logits = tf.stop_gradient(logits)
 149   f_inputs, _ = inputs
 150   weights = f_inputs.eos_weights
 151   if FLAGS.single_label:
 152     indices = tf.stack([tf.range(FLAGS.batch_size), f_inputs.length - 1], 1)
 153     weights = tf.expand_dims(tf.gather_nd(f_inputs.eos_weights, indices), 1)
 154   assert weights is not None
 155 
 156   perturbs = [
 157       _mask_by_length(tf.random_normal(shape=tf.shape(emb)), f_inputs.length)
 158       for emb in embedded
 159   ]
 160   for _ in xrange(FLAGS.num_power_iteration):
 161     perturbs = [
 162         _scale_l2(d, FLAGS.small_constant_for_finite_diff) for d in perturbs
 163     ]
 164     d_logits = logits_from_embedding_fn(
 165         [emb + d for (emb, d) in zip(embedded, perturbs)])
 166     kl = _kl_divergence_with_logits(logits, d_logits, weights)
 167     perturbs = tf.gradients(
 168         kl,
 169         perturbs,
 170         aggregation_method=tf.AggregationMethod.EXPERIMENTAL_ACCUMULATE_N)
 171     perturbs = [tf.stop_gradient(d) for d in perturbs]
 172 
 173   perturbs = [_scale_l2(d, FLAGS.perturb_norm_length) for d in perturbs]
 174   vadv_logits = logits_from_embedding_fn(
 175       [emb + d for (emb, d) in zip(embedded, perturbs)])
 176   return _kl_divergence_with_logits(logits, vadv_logits, weights)

 179 def _mask_by_length(t, length):
 180   """Mask t, 3-D [batch, time, dim], by length, 1-D [batch,]."""
 181   maxlen = t.get_shape().as_list()[1]
 182 
 183   # Subtract 1 from length to prevent the perturbation from going on 'eos'
 184   mask = tf.sequence_mask(length - 1, maxlen=maxlen)
 185   mask = tf.expand_dims(tf.cast(mask, tf.float32), -1)
 186   # shape(mask) = (batch, num_timesteps, 1)
 187   return t * mask

 190 def _scale_l2(x, norm_length):
 191   # shape(x) = (batch, num_timesteps, d)
 192   # Divide x by max(abs(x)) for a numerically stable L2 norm.
 193   # 2norm(x) = a * 2norm(x/a)
 194   # Scale over the full sequence, dims (1, 2)
 195   alpha = tf.reduce_max(tf.abs(x), (1, 2), keep_dims=True) + 1e-12
 196   l2_norm = alpha * tf.sqrt(
 197       tf.reduce_sum(tf.pow(x / alpha, 2), (1, 2), keep_dims=True) + 1e-6)
 198   x_unit = x / l2_norm
 199   return norm_length * x_unit

 202 def _kl_divergence_with_logits(q_logits, p_logits, weights):
 203   """Returns weighted KL divergence between distributions q and p.
 204 
 205   Args:
 206     q_logits: logits for 1st argument of KL divergence shape
 207               [batch_size, num_timesteps, num_classes] if num_classes > 2, and
 208               [batch_size, num_timesteps] if num_classes == 2.
 209     p_logits: logits for 2nd argument of KL divergence with same shape q_logits.
 210     weights: 1-D float tensor with shape [batch_size, num_timesteps].
 211              Elements should be 1.0 only on end of sequences
 212 
 213   Returns:
 214     KL: float scalar.
 215   """
 216   # For logistic regression
 217   if FLAGS.num_classes == 2:
 218     q = tf.nn.sigmoid(q_logits)
 219     kl = (-tf.nn.sigmoid_cross_entropy_with_logits(logits=q_logits, labels=q) +
 220           tf.nn.sigmoid_cross_entropy_with_logits(logits=p_logits, labels=q))
 221     kl = tf.squeeze(kl, 2)
 222 
 223   # For softmax regression
 224   else:
 225     q = tf.nn.softmax(q_logits)
 226     kl = tf.reduce_sum(
 227         q * (tf.nn.log_softmax(q_logits) - tf.nn.log_softmax(p_logits)), -1)
 228 
 229   num_labels = tf.reduce_sum(weights)
 230   num_labels = tf.where(tf.equal(num_labels, 0.), 1., num_labels)
 231 
 232   kl.get_shape().assert_has_rank(2)
 233   weights.get_shape().assert_has_rank(2)
 234 
 235   loss = tf.identity(tf.reduce_sum(weights * kl) / num_labels, name='kl')
 236   return loss
