  30 class VatxtInput(object):
  31   """Wrapper around NextQueuedSequenceBatch."""
  32 
  33   def __init__(self,
  34                batch,
  35                state_name=None,
  36                tokens=None,
  37                num_states=0,
  38                eos_id=None):
  39     """Construct VatxtInput.
  40 
  41     Args:
  42       batch: NextQueuedSequenceBatch.
  43       state_name: str, name of state to fetch and save.
  44       tokens: int Tensor, tokens. Defaults to batch's F_TOKEN_ID sequence.
  45       num_states: int The number of states to store.
  46       eos_id: int Id of end of Sequence.
  47     """
  48     self._batch = batch
  49     self._state_name = state_name
  50     self._tokens = (tokens if tokens is not None else
  51                     batch.sequences[data_utils.SequenceWrapper.F_TOKEN_ID])
  52     self._num_states = num_states
  53 
  54     w = batch.sequences[data_utils.SequenceWrapper.F_WEIGHT]
  55     self._weights = w
  56 
  57     l = batch.sequences[data_utils.SequenceWrapper.F_LABEL]
  58     self._labels = l
  59 
  60     # eos weights
  61     self._eos_weights = None
  62     if eos_id:
  63       ew = tf.cast(tf.equal(self._tokens, eos_id), tf.float32)
  64       self._eos_weights = ew
  65 
  66   @property
  67   def tokens(self):
  68     return self._tokens
  69 
  70   @property
  71   def weights(self):
  72     return self._weights
  73 
  74   @property
  75   def eos_weights(self):
  76     return self._eos_weights
  77 
  78   @property
  79   def labels(self):
  80     return self._labels
  81 
  82   @property
  83   def length(self):
  84     return self._batch.length
  85 
  86   @property
  87   def state_name(self):
  88     return self._state_name
  89 
  90   @property
  91   def state(self):
  92     # LSTM tuple states
  93     state_names = _get_tuple_state_names(self._num_states, self._state_name)
  94     return tuple([
  95         tf.contrib.rnn.LSTMStateTuple(
  96             self._batch.state(c_name), self._batch.state(h_name))
  97         for c_name, h_name in state_names
  98     ])
  99 
 100   def save_state(self, value):
 101     # LSTM tuple states
 102     state_names = _get_tuple_state_names(self._num_states, self._state_name)
 103     save_ops = []
 104     for (c_state, h_state), (c_name, h_name) in zip(value, state_names):
 105       save_ops.append(self._batch.save_state(c_name, c_state))
 106       save_ops.append(self._batch.save_state(h_name, h_state))
 107     return tf.group(*save_ops)

 110 def _get_tuple_state_names(num_states, base_name):
 111   """Returns state names for use with LSTM tuple state."""
 112   state_names = [('{}_{}_c'.format(i, base_name), '{}_{}_h'.format(
 113       i, base_name)) for i in range(num_states)]
 114   return state_names

 117 def _split_bidir_tokens(batch):
 118   tokens = batch.sequences[data_utils.SequenceWrapper.F_TOKEN_ID]
 119   # Tokens have shape [batch, time, 2]
 120   # forward and reverse have shape [batch, time].
 121   forward, reverse = [
 122       tf.squeeze(t, axis=[2]) for t in tf.split(tokens, 2, axis=2)
 123   ]
 124   return forward, reverse

 127 def _filenames_for_data_spec(phase, bidir, pretrain, use_seq2seq):
 128   """Returns input filenames for configuration.
 129 
 130   Args:
 131     phase: str, 'train', 'test', or 'valid'.
 132     bidir: bool, bidirectional model.
 133     pretrain: bool, pretraining or classification.
 134     use_seq2seq: bool, seq2seq data, only valid if pretrain=True.
 135 
 136   Returns:
 137     Tuple of filenames.
 138 
 139   Raises:
 140     ValueError: if an invalid combination of arguments is provided that does not
 141       map to any data files (e.g. pretrain=False, use_seq2seq=True).
 142   """
 143   data_spec = (phase, bidir, pretrain, use_seq2seq)
 144   data_specs = {
 145       ('train', True, True, False): (data_utils.TRAIN_LM,
 146                                      data_utils.TRAIN_REV_LM),
 147       ('train', True, False, False): (data_utils.TRAIN_BD_CLASS,),
 148       ('train', False, True, False): (data_utils.TRAIN_LM,),
 149       ('train', False, True, True): (data_utils.TRAIN_SA,),
 150       ('train', False, False, False): (data_utils.TRAIN_CLASS,),
 151       ('test', True, True, False): (data_utils.TEST_LM,
 152                                     data_utils.TRAIN_REV_LM),
 153       ('test', True, False, False): (data_utils.TEST_BD_CLASS,),
 154       ('test', False, True, False): (data_utils.TEST_LM,),
 155       ('test', False, True, True): (data_utils.TEST_SA,),
 156       ('test', False, False, False): (data_utils.TEST_CLASS,),
 157       ('valid', True, False, False): (data_utils.VALID_BD_CLASS,),
 158       ('valid', False, False, False): (data_utils.VALID_CLASS,),
 159   }
 160   if data_spec not in data_specs:
 161     raise ValueError(
 162         'Data specification (phase, bidir, pretrain, use_seq2seq) %s not '
 163         'supported' % str(data_spec))
 164 
 165   return data_specs[data_spec]

 168 def _read_single_sequence_example(file_list, tokens_shape=None):
 169   """Reads and parses SequenceExamples from TFRecord-encoded file_list."""
 170   tf.logging.info('Constructing TFRecordReader from files: %s', file_list)
 171   file_queue = tf.train.string_input_producer(file_list)
 172   reader = tf.TFRecordReader()
 173   seq_key, serialized_record = reader.read(file_queue)
 174   ctx, sequence = tf.parse_single_sequence_example(
 175       serialized_record,
 176       sequence_features={
 177           data_utils.SequenceWrapper.F_TOKEN_ID:
 178               tf.FixedLenSequenceFeature(tokens_shape or [], dtype=tf.int64),
 179           data_utils.SequenceWrapper.F_LABEL:
 180               tf.FixedLenSequenceFeature([], dtype=tf.int64),
 181           data_utils.SequenceWrapper.F_WEIGHT:
 182               tf.FixedLenSequenceFeature([], dtype=tf.float32),
 183       })
 184   return seq_key, ctx, sequence

 187 def _read_and_batch(data_dir,
 188                     fname,
 189                     state_name,
 190                     state_size,
 191                     num_layers,
 192                     unroll_steps,
 193                     batch_size,
 194                     bidir_input=False):
 195   """Inputs for text model.
 196 
 197   Args:
 198     data_dir: str, directory containing TFRecord files of SequenceExample.
 199     fname: str, input file name.
 200     state_name: string, key for saved state of LSTM.
 201     state_size: int, size of LSTM state.
 202     num_layers: int, the number of layers in the LSTM.
 203     unroll_steps: int, number of timesteps to unroll for TBTT.
 204     batch_size: int, batch size.
 205     bidir_input: bool, whether the input is bidirectional. If True, creates 2
 206       states, state_name and state_name + '_reverse'.
 207 
 208   Returns:
 209     Instance of NextQueuedSequenceBatch
 210 
 211   Raises:
 212     ValueError: if file for input specification is not found.
 213   """
 214   data_path = os.path.join(data_dir, fname)
 215   if not tf.gfile.Exists(data_path):
 216     raise ValueError('Failed to find file: %s' % data_path)
 217 
 218   tokens_shape = [2] if bidir_input else []
 219   seq_key, ctx, sequence = _read_single_sequence_example(
 220       [data_path], tokens_shape=tokens_shape)
 221   # Set up stateful queue reader.
 222   state_names = _get_tuple_state_names(num_layers, state_name)
 223   initial_states = {}
 224   for c_state, h_state in state_names:
 225     initial_states[c_state] = tf.zeros(state_size)
 226     initial_states[h_state] = tf.zeros(state_size)
 227   if bidir_input:
 228     rev_state_names = _get_tuple_state_names(num_layers,
 229                                              '{}_reverse'.format(state_name))
 230     for rev_c_state, rev_h_state in rev_state_names:
 231       initial_states[rev_c_state] = tf.zeros(state_size)
 232       initial_states[rev_h_state] = tf.zeros(state_size)
 233   batch = tf.contrib.training.batch_sequences_with_states(
 234       input_key=seq_key,
 235       input_sequences=sequence,
 236       input_context=ctx,
 237       input_length=tf.shape(sequence['token_id'])[0],
 238       initial_states=initial_states,
 239       num_unroll=unroll_steps,
 240       batch_size=batch_size,
 241       allow_small_batch=False,
 242       num_threads=4,
 243       capacity=batch_size * 10,
 244       make_keys_unique=True,
 245       make_keys_unique_seed=29392)
 246   return batch

 249 def inputs(data_dir=None,
 250            phase='train',
 251            bidir=False,
 252            pretrain=False,
 253            use_seq2seq=False,
 254            state_name='lstm',
 255            state_size=None,
 256            num_layers=0,
 257            batch_size=32,
 258            unroll_steps=100,
 259            eos_id=None):
 260   """Inputs for text model.
 261 
 262   Args:
 263     data_dir: str, directory containing TFRecord files of SequenceExample.
 264     phase: str, dataset for evaluation {'train', 'valid', 'test'}.
 265     bidir: bool, bidirectional LSTM.
 266     pretrain: bool, whether to read pretraining data or classification data.
 267     use_seq2seq: bool, whether to read seq2seq data or the language model data.
 268     state_name: string, key for saved state of LSTM.
 269     state_size: int, size of LSTM state.
 270     num_layers: int, the number of LSTM layers.
 271     batch_size: int, batch size.
 272     unroll_steps: int, number of timesteps to unroll for TBTT.
 273     eos_id: int, id of end of sequence. used for the kl weights on vat
 274   Returns:
 275     Instance of VatxtInput (x2 if bidir=True and pretrain=True, i.e. forward and
 276       reverse).
 277   """
 278   with tf.name_scope('inputs'):
 279     filenames = _filenames_for_data_spec(phase, bidir, pretrain, use_seq2seq)
 280 
 281     if bidir and pretrain:
 282       # Bidirectional pretraining
 283       # Requires separate forward and reverse language model data.
 284       forward_fname, reverse_fname = filenames
 285       forward_batch = _read_and_batch(data_dir, forward_fname, state_name,
 286                                       state_size, num_layers, unroll_steps,
 287                                       batch_size)
 288       state_name_rev = state_name + '_reverse'
 289       reverse_batch = _read_and_batch(data_dir, reverse_fname, state_name_rev,
 290                                       state_size, num_layers, unroll_steps,
 291                                       batch_size)
 292       forward_input = VatxtInput(
 293           forward_batch,
 294           state_name=state_name,
 295           num_states=num_layers,
 296           eos_id=eos_id)
 297       reverse_input = VatxtInput(
 298           reverse_batch,
 299           state_name=state_name_rev,
 300           num_states=num_layers,
 301           eos_id=eos_id)
 302       return forward_input, reverse_input
 303 
 304     elif bidir:
 305       # Classifier bidirectional LSTM
 306       # Shared data source, but separate token/state streams
 307       fname, = filenames
 308       batch = _read_and_batch(
 309           data_dir,
 310           fname,
 311           state_name,
 312           state_size,
 313           num_layers,
 314           unroll_steps,
 315           batch_size,
 316           bidir_input=True)
 317       forward_tokens, reverse_tokens = _split_bidir_tokens(batch)
 318       forward_input = VatxtInput(
 319           batch,
 320           state_name=state_name,
 321           tokens=forward_tokens,
 322           num_states=num_layers)
 323       reverse_input = VatxtInput(
 324           batch,
 325           state_name=state_name + '_reverse',
 326           tokens=reverse_tokens,
 327           num_states=num_layers)
 328       return forward_input, reverse_input
 329     else:
 330       # Unidirectional LM or classifier
 331       fname, = filenames
 332       batch = _read_and_batch(
 333           data_dir,
 334           fname,
 335           state_name,
 336           state_size,
 337           num_layers,
 338           unroll_steps,
 339           batch_size,
 340           bidir_input=False)
 341       return VatxtInput(
 342           batch, state_name=state_name, num_states=num_layers, eos_id=eos_id)
