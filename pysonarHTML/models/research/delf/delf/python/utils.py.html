  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from PIL import Image
  22 from PIL import ImageFile
  23 import tensorflow as tf
  24 
  25 # To avoid PIL crashing for truncated (corrupted) images.
  26 ImageFile.LOAD_TRUNCATED_IMAGES = True
  27 
  28 
  29 def RgbLoader(path):
  30   """Helper function to read image with PIL.
  31 
  32   Args:
  33     path: Path to image to be loaded.
  34 
  35   Returns:
  36     PIL image in RGB format.
  37   """
  38   with tf.io.gfile.GFile(path, 'rb') as f:
  39     img = Image.open(f)
  40     return img.convert('RGB')
