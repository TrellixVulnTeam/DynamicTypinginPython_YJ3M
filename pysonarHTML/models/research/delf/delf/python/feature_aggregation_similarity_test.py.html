  28 class FeatureAggregationSimilarityTest(tf.test.TestCase):

  30   def testComputeVladSimilarityWorks(self):
  31     # Construct inputs.
  32     vlad_1 = np.array([0, 1, 2, 3, 4])
  33     vlad_2 = np.array([5, 6, 7, 8, 9])
  34     config = aggregation_config_pb2.AggregationConfig()
  35     config.aggregation_type = aggregation_config_pb2.AggregationConfig.VLAD
  36 
  37     # Run tested function.
  38     similarity_computer = (
  39         feature_aggregation_similarity.SimilarityAggregatedRepresentation(
  40             config))
  41     similarity = similarity_computer.ComputeSimilarity(vlad_1, vlad_2)
  42 
  43     # Define expected results.
  44     exp_similarity = 80
  45 
  46     # Compare actual and expected results.
  47     self.assertAllEqual(similarity, exp_similarity)

  49   def testComputeAsmkSimilarityWorks(self):
  50     # Construct inputs.
  51     aggregated_descriptors_1 = np.array([
  52         0.0, 0.0, -0.707107, -0.707107, 0.5, 0.866025, 0.816497, 0.577350, 1.0,
  53         0.0
  54     ])
  55     visual_words_1 = np.array([0, 1, 2, 3, 4])
  56     aggregated_descriptors_2 = np.array(
  57         [0.0, 1.0, 1.0, 0.0, 0.707107, 0.707107])
  58     visual_words_2 = np.array([1, 2, 4])
  59     config = aggregation_config_pb2.AggregationConfig()
  60     config.codebook_size = 5
  61     config.feature_dimensionality = 2
  62     config.aggregation_type = aggregation_config_pb2.AggregationConfig.ASMK
  63     config.use_l2_normalization = True
  64 
  65     # Run tested function.
  66     similarity_computer = (
  67         feature_aggregation_similarity.SimilarityAggregatedRepresentation(
  68             config))
  69     similarity = similarity_computer.ComputeSimilarity(
  70         aggregated_descriptors_1, aggregated_descriptors_2, visual_words_1,
  71         visual_words_2)
  72 
  73     # Define expected results.
  74     exp_similarity = 0.123562
  75 
  76     # Compare actual and expected results.
  77     self.assertAllClose(similarity, exp_similarity)
