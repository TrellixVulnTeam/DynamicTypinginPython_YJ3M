  15 """Python interface for DatumProto.
  16 
  17 DatumProto is protocol buffer used to serialize tensor with arbitrary shape.
  18 Please refer to datum.proto for details.
  19 
  20 Support read and write of DatumProto from/to NumPy array and file.
  21 """
  22 
  23 from __future__ import absolute_import
  24 from __future__ import division
  25 from __future__ import print_function
  26 
  27 import numpy as np
  28 import tensorflow as tf
  29 
  30 from delf import datum_pb2
  31 
  32 
  33 def ArrayToDatum(arr):
  34   """Converts NumPy array to DatumProto.
  35 
  36   Supports arrays of types:
  37     - float16 (it is converted into a float32 in DatumProto)
  38     - float32
  39     - float64 (it is converted into a float32 in DatumProto)
  40     - uint8 (it is converted into a uint32 in DatumProto)
  41     - uint16 (it is converted into a uint32 in DatumProto)
  42     - uint32
  43     - uint64 (it is converted into a uint32 in DatumProto)
  44 
  45   Args:
  46     arr: NumPy array of arbitrary shape.
  47 
  48   Returns:
  49     datum: DatumProto object.
  50 
  51   Raises:
  52     ValueError: If array type is unsupported.
  53   """
  54   datum = datum_pb2.DatumProto()
  55   if arr.dtype in ('float16', 'float32', 'float64'):
  56     datum.float_list.value.extend(arr.astype('float32').flat)
  57   elif arr.dtype in ('uint8', 'uint16', 'uint32', 'uint64'):
  58     datum.uint32_list.value.extend(arr.astype('uint32').flat)
  59   else:
  60     raise ValueError('Unsupported array type: %s' % arr.dtype)
  61 
  62   datum.shape.dim.extend(arr.shape)
  63   return datum
  64 
  65 
  66 def ArraysToDatumPair(arr_1, arr_2):
  67   """Converts numpy arrays to DatumPairProto.
  68 
  69   Supports same formats as `ArrayToDatum`, see documentation therein.
  70 
  71   Args:
  72     arr_1: NumPy array of arbitrary shape.
  73     arr_2: NumPy array of arbitrary shape.
  74 
  75   Returns:
  76     datum_pair: DatumPairProto object.
  77   """
  78   datum_pair = datum_pb2.DatumPairProto()
  79   datum_pair.first.CopyFrom(ArrayToDatum(arr_1))
  80   datum_pair.second.CopyFrom(ArrayToDatum(arr_2))
  81 
  82   return datum_pair
  83

  85 def DatumToArray(datum):
  86   """Converts data saved in DatumProto to NumPy array.
  87 
  88   Args:
  89     datum: DatumProto object.
  90 
  91   Returns:
  92     NumPy array of arbitrary shape.
  93   """
  94   if datum.HasField('float_list'):
  95     return np.array(datum.float_list.value).astype('float32').reshape(
  96         datum.shape.dim)
  97   elif datum.HasField('uint32_list'):
  98     return np.array(datum.uint32_list.value).astype('uint32').reshape(
  99         datum.shape.dim)
 100   else:
 101     raise ValueError('Input DatumProto does not have float_list or uint32_list')

 104 def DatumPairToArrays(datum_pair):
 105   """Converts data saved in DatumPairProto to NumPy arrays.
 106 
 107   Args:
 108     datum_pair: DatumPairProto object.
 109 
 110   Returns:
 111     Two NumPy arrays of arbitrary shape.
 112   """
 113   first_datum = DatumToArray(datum_pair.first)
 114   second_datum = DatumToArray(datum_pair.second)
 115   return first_datum, second_datum

 118 def SerializeToString(arr):
 119   """Converts NumPy array to serialized DatumProto.
 120 
 121   Args:
 122     arr: NumPy array of arbitrary shape.
 123 
 124   Returns:
 125     Serialized DatumProto string.
 126   """
 127   datum = ArrayToDatum(arr)
 128   return datum.SerializeToString()
 129 
 130 
 131 def SerializePairToString(arr_1, arr_2):
 132   """Converts pair of NumPy arrays to serialized DatumPairProto.
 133 
 134   Args:
 135     arr_1: NumPy array of arbitrary shape.
 136     arr_2: NumPy array of arbitrary shape.
 137 
 138   Returns:
 139     Serialized DatumPairProto string.
 140   """
 141   datum_pair = ArraysToDatumPair(arr_1, arr_2)
 142   return datum_pair.SerializeToString()

 145 def ParseFromString(string):
 146   """Converts serialized DatumProto string to NumPy array.
 147 
 148   Args:
 149     string: Serialized DatumProto string.
 150 
 151   Returns:
 152     NumPy array.
 153   """
 154   datum = datum_pb2.DatumProto()
 155   datum.ParseFromString(string)
 156   return DatumToArray(datum)

 159 def ParsePairFromString(string):
 160   """Converts serialized DatumPairProto string to NumPy arrays.
 161 
 162   Args:
 163     string: Serialized DatumProto string.
 164 
 165   Returns:
 166     Two NumPy arrays.
 167   """
 168   datum_pair = datum_pb2.DatumPairProto()
 169   datum_pair.ParseFromString(string)
 170   return DatumPairToArrays(datum_pair)

 173 def ReadFromFile(file_path):
 174   """Helper function to load data from a DatumProto format in a file.
 175 
 176   Args:
 177     file_path: Path to file containing data.
 178 
 179   Returns:
 180     data: NumPy array.
 181   """
 182   with tf.io.gfile.GFile(file_path, 'rb') as f:
 183     return ParseFromString(f.read())
 184 
 185 
 186 def ReadPairFromFile(file_path):
 187   """Helper function to load data from a DatumPairProto format in a file.
 188 
 189   Args:
 190     file_path: Path to file containing data.
 191 
 192   Returns:
 193     Two NumPy arrays.
 194   """
 195   with tf.io.gfile.GFile(file_path, 'rb') as f:
 196     return ParsePairFromString(f.read())

 198 
 199 def WriteToFile(data, file_path):
 200   """Helper function to write data to a file in DatumProto format.
 201 
 202   Args:
 203     data: NumPy array.
 204     file_path: Path to file that will be written.
 205   """
 206   serialized_data = SerializeToString(data)
 207   with tf.io.gfile.GFile(file_path, 'w') as f:
 208     f.write(serialized_data)
 209 
 210 
 211 def WritePairToFile(arr_1, arr_2, file_path):
 212   """Helper function to write pair of arrays to a file in DatumPairProto format.
 213 
 214   Args:
 215     arr_1: NumPy array of arbitrary shape.
 216     arr_2: NumPy array of arbitrary shape.
 217     file_path: Path to file that will be written.
 218   """
 219   serialized_data = SerializePairToString(arr_1, arr_2)
 220   with tf.io.gfile.GFile(file_path, 'w') as f:
 221     f.write(serialized_data)
