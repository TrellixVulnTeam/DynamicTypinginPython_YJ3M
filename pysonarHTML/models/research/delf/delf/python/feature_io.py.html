  31 def ArraysToDelfFeatures(locations,
  32                          scales,
  33                          descriptors,
  34                          attention,
  35                          orientations=None):
  36   """Converts DELF features to DelfFeatures proto.
  37 
  38   Args:
  39     locations: [N, 2] float array which denotes the selected keypoint locations.
  40       N is the number of features.
  41     scales: [N] float array with feature scales.
  42     descriptors: [N, depth] float array with DELF descriptors.
  43     attention: [N] float array with attention scores.
  44     orientations: [N] float array with orientations. If None, all orientations
  45       are set to zero.
  46 
  47   Returns:
  48     delf_features: DelfFeatures object.
  49   """
  50   num_features = len(attention)
  51   assert num_features == locations.shape[0]
  52   assert num_features == len(scales)
  53   assert num_features == descriptors.shape[0]
  54 
  55   if orientations is None:
  56     orientations = np.zeros([num_features], dtype=np.float32)
  57   else:
  58     assert num_features == len(orientations)
  59 
  60   delf_features = feature_pb2.DelfFeatures()
  61   for i in range(num_features):
  62     delf_feature = delf_features.feature.add()
  63     delf_feature.y = locations[i, 0]
  64     delf_feature.x = locations[i, 1]
  65     delf_feature.scale = scales[i]
  66     delf_feature.orientation = orientations[i]
  67     delf_feature.strength = attention[i]
  68     delf_feature.descriptor.CopyFrom(datum_io.ArrayToDatum(descriptors[i,]))
  69 
  70   return delf_features

  73 def DelfFeaturesToArrays(delf_features):
  74   """Converts data saved in DelfFeatures to numpy arrays.
  75 
  76   If there are no features, the function returns four empty arrays.
  77 
  78   Args:
  79     delf_features: DelfFeatures object.
  80 
  81   Returns:
  82     locations: [N, 2] float array which denotes the selected keypoint
  83       locations. N is the number of features.
  84     scales: [N] float array with feature scales.
  85     descriptors: [N, depth] float array with DELF descriptors.
  86     attention: [N] float array with attention scores.
  87     orientations: [N] float array with orientations.
  88   """
  89   num_features = len(delf_features.feature)
  90   if num_features == 0:
  91     return np.array([]), np.array([]), np.array([]), np.array([]), np.array([])
  92 
  93   # Figure out descriptor dimensionality by parsing first one.
  94   descriptor_dim = len(
  95       datum_io.DatumToArray(delf_features.feature[0].descriptor))
  96   locations = np.zeros([num_features, 2])
  97   scales = np.zeros([num_features])
  98   descriptors = np.zeros([num_features, descriptor_dim])
  99   attention = np.zeros([num_features])
 100   orientations = np.zeros([num_features])
 101 
 102   for i in range(num_features):
 103     delf_feature = delf_features.feature[i]
 104     locations[i, 0] = delf_feature.y
 105     locations[i, 1] = delf_feature.x
 106     scales[i] = delf_feature.scale
 107     descriptors[i,] = datum_io.DatumToArray(delf_feature.descriptor)
 108     attention[i] = delf_feature.strength
 109     orientations[i] = delf_feature.orientation
 110 
 111   return locations, scales, descriptors, attention, orientations

 114 def SerializeToString(locations,
 115                       scales,
 116                       descriptors,
 117                       attention,
 118                       orientations=None):
 119   """Converts numpy arrays to serialized DelfFeatures.
 120 
 121   Args:
 122     locations: [N, 2] float array which denotes the selected keypoint locations.
 123       N is the number of features.
 124     scales: [N] float array with feature scales.
 125     descriptors: [N, depth] float array with DELF descriptors.
 126     attention: [N] float array with attention scores.
 127     orientations: [N] float array with orientations. If None, all orientations
 128       are set to zero.
 129 
 130   Returns:
 131     Serialized DelfFeatures string.
 132   """
 133   delf_features = ArraysToDelfFeatures(locations, scales, descriptors,
 134                                        attention, orientations)
 135   return delf_features.SerializeToString()

 138 def ParseFromString(string):
 139   """Converts serialized DelfFeatures string to numpy arrays.
 140 
 141   Args:
 142     string: Serialized DelfFeatures string.
 143 
 144   Returns:
 145     locations: [N, 2] float array which denotes the selected keypoint
 146       locations. N is the number of features.
 147     scales: [N] float array with feature scales.
 148     descriptors: [N, depth] float array with DELF descriptors.
 149     attention: [N] float array with attention scores.
 150     orientations: [N] float array with orientations.
 151   """
 152   delf_features = feature_pb2.DelfFeatures()
 153   delf_features.ParseFromString(string)
 154   return DelfFeaturesToArrays(delf_features)

 157 def ReadFromFile(file_path):
 158   """Helper function to load data from a DelfFeatures format in a file.
 159 
 160   Args:
 161     file_path: Path to file containing data.
 162 
 163   Returns:
 164     locations: [N, 2] float array which denotes the selected keypoint
 165       locations. N is the number of features.
 166     scales: [N] float array with feature scales.
 167     descriptors: [N, depth] float array with DELF descriptors.
 168     attention: [N] float array with attention scores.
 169     orientations: [N] float array with orientations.
 170   """
 171   with tf.io.gfile.GFile(file_path, 'rb') as f:
 172     return ParseFromString(f.read())
 173 
 174 
 175 def WriteToFile(file_path,
 176                 locations,
 177                 scales,
 178                 descriptors,
 179                 attention,
 180                 orientations=None):
 181   """Helper function to write data to a file in DelfFeatures format.
 182 
 183   Args:
 184     file_path: Path to file that will be written.
 185     locations: [N, 2] float array which denotes the selected keypoint locations.
 186       N is the number of features.
 187     scales: [N] float array with feature scales.
 188     descriptors: [N, depth] float array with DELF descriptors.
 189     attention: [N] float array with attention scores.
 190     orientations: [N] float array with orientations. If None, all orientations
 191       are set to zero.
 192   """
 193   serialized_data = SerializeToString(locations, scales, descriptors, attention,
 194                                       orientations)
 195   with tf.io.gfile.GFile(file_path, 'w') as f:
 196     f.write(serialized_data)
