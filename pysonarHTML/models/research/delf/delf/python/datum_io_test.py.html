  32 class DatumIoTest(tf.test.TestCase):
  33 
  34   def Conversion2dTestWithType(self, dtype):
  35     original_data = np.arange(9).reshape(3, 3).astype(dtype)
  36     serialized = datum_io.SerializeToString(original_data)
  37     retrieved_data = datum_io.ParseFromString(serialized)
  38     self.assertTrue(np.array_equal(original_data, retrieved_data))
  39 
  40   def Conversion3dTestWithType(self, dtype):
  41     original_data = np.arange(24).reshape(2, 3, 4).astype(dtype)
  42     serialized = datum_io.SerializeToString(original_data)
  43     retrieved_data = datum_io.ParseFromString(serialized)
  44     self.assertTrue(np.array_equal(original_data, retrieved_data))
  45 
  46   # This test covers the following functions: ArrayToDatum, SerializeToString,
  47   # ParseFromString, DatumToArray.
  48   def testConversion2dWithType(self):
  49     self.Conversion2dTestWithType(np.uint16)
  50     self.Conversion2dTestWithType(np.uint32)
  51     self.Conversion2dTestWithType(np.uint64)
  52     self.Conversion2dTestWithType(np.float16)
  53     self.Conversion2dTestWithType(np.float32)
  54     self.Conversion2dTestWithType(np.float64)
  55 
  56   # This test covers the following functions: ArrayToDatum, SerializeToString,
  57   # ParseFromString, DatumToArray.
  58   def testConversion3dWithType(self):
  59     self.Conversion3dTestWithType(np.uint16)
  60     self.Conversion3dTestWithType(np.uint32)
  61     self.Conversion3dTestWithType(np.uint64)
  62     self.Conversion3dTestWithType(np.float16)
  63     self.Conversion3dTestWithType(np.float32)
  64     self.Conversion3dTestWithType(np.float64)
  65 
  66   def testConversionWithUnsupportedType(self):
  67     with self.assertRaisesRegex(ValueError, 'Unsupported array type'):
  68       self.Conversion3dTestWithType(int)

  70   # This test covers the following functions: ArrayToDatum, SerializeToString,
  71   # WriteToFile, ReadFromFile, ParseFromString, DatumToArray.
  72   def testWriteAndReadToFile(self):
  73     data = np.array([[[-1.0, 125.0, -2.5], [14.5, 3.5, 0.0]],
  74                      [[20.0, 0.0, 30.0], [25.5, 36.0, 42.0]]])
  75     filename = os.path.join(FLAGS.test_tmpdir, 'test.datum')
  76     datum_io.WriteToFile(data, filename)
  77     data_read = datum_io.ReadFromFile(filename)
  78     self.assertAllEqual(data_read, data)
  79 
  80   # This test covers the following functions: ArraysToDatumPair,
  81   # SerializePairToString, WritePairToFile, ReadPairFromFile,
  82   # ParsePairFromString, DatumPairToArrays.
  83   def testWriteAndReadPairToFile(self):
  84     data_1 = np.array([[[-1.0, 125.0, -2.5], [14.5, 3.5, 0.0]],
  85                        [[20.0, 0.0, 30.0], [25.5, 36.0, 42.0]]])
  86     data_2 = np.array(
  87         [[[255, 0, 5], [10, 300, 0]], [[20, 1, 100], [255, 360, 420]]],
  88         dtype='uint32')
  89     filename = os.path.join(FLAGS.test_tmpdir, 'test.datum_pair')
  90     datum_io.WritePairToFile(data_1, data_2, filename)
  91     data_read_1, data_read_2 = datum_io.ReadPairFromFile(filename)
  92     self.assertAllEqual(data_read_1, data_1)
  93     self.assertAllEqual(data_read_2, data_2)
