  23 def NormalizePixelValues(image,
  24                          pixel_value_offset=128.0,
  25                          pixel_value_scale=128.0):
  26   """Normalize image pixel values.
  27 
  28   Args:
  29     image: a uint8 tensor.
  30     pixel_value_offset: a Python float, offset for normalizing pixel values.
  31     pixel_value_scale: a Python float, scale for normalizing pixel values.
  32 
  33   Returns:
  34     image: a float32 tensor of the same shape as the input image.
  35   """
  36   image = tf.cast(image, dtype=tf.float32)
  37   image = tf.truediv(tf.subtract(image, pixel_value_offset), pixel_value_scale)
  38   return image

  41 def CalculateReceptiveBoxes(height, width, rf, stride, padding):
  42   """Calculate receptive boxes for each feature point.
  43 
  44   Args:
  45     height: The height of feature map.
  46     width: The width of feature map.
  47     rf: The receptive field size.
  48     stride: The effective stride between two adjacent feature points.
  49     padding: The effective padding size.
  50 
  51   Returns:
  52     rf_boxes: [N, 4] receptive boxes tensor. Here N equals to height x width.
  53     Each box is represented by [ymin, xmin, ymax, xmax].
  54   """
  55   x, y = tf.meshgrid(tf.range(width), tf.range(height))
  56   coordinates = tf.reshape(tf.stack([y, x], axis=2), [-1, 2])
  57   # [y,x,y,x]
  58   point_boxes = tf.cast(
  59       tf.concat([coordinates, coordinates], 1), dtype=tf.float32)
  60   bias = [-padding, -padding, -padding + rf - 1, -padding + rf - 1]
  61   rf_boxes = stride * point_boxes + bias
  62   return rf_boxes

  65 def CalculateKeypointCenters(boxes):
  66   """Helper function to compute feature centers, from RF boxes.
  67 
  68   Args:
  69     boxes: [N, 4] float tensor.
  70 
  71   Returns:
  72     centers: [N, 2] float tensor.
  73   """
  74   return tf.divide(
  75       tf.add(
  76           tf.gather(boxes, [0, 1], axis=1), tf.gather(boxes, [2, 3], axis=1)),
  77       2.0)

  80 def ApplyPcaAndWhitening(data,
  81                          pca_matrix,
  82                          pca_mean,
  83                          output_dim,
  84                          use_whitening=False,
  85                          pca_variances=None):
  86   """Applies PCA/whitening to data.
  87 
  88   Args:
  89     data: [N, dim] float tensor containing data which undergoes PCA/whitening.
  90     pca_matrix: [dim, dim] float tensor PCA matrix, row-major.
  91     pca_mean: [dim] float tensor, mean to subtract before projection.
  92     output_dim: Number of dimensions to use in output data, of type int.
  93     use_whitening: Whether whitening is to be used.
  94     pca_variances: [dim] float tensor containing PCA variances. Only used if
  95       use_whitening is True.
  96 
  97   Returns:
  98     output: [N, output_dim] float tensor with output of PCA/whitening operation.
  99   """
 100   output = tf.matmul(
 101       tf.subtract(data, pca_mean),
 102       tf.slice(pca_matrix, [0, 0], [output_dim, -1]),
 103       transpose_b=True,
 104       name='pca_matmul')
 105 
 106   # Apply whitening if desired.
 107   if use_whitening:
 108     output = tf.divide(
 109         output,
 110         tf.sqrt(tf.slice(pca_variances, [0], [output_dim])),
 111         name='whitening')
 112 
 113   return output

 116 def PostProcessDescriptors(descriptors, use_pca, pca_parameters=None):
 117   """Post-process descriptors.
 118 
 119   Args:
 120     descriptors: [N, input_dim] float tensor.
 121     use_pca: Whether to use PCA.
 122     pca_parameters: Only used if `use_pca` is True. Dict containing PCA
 123       parameter tensors, with keys 'mean', 'matrix', 'dim', 'use_whitening',
 124       'variances'.
 125 
 126   Returns:
 127     final_descriptors: [N, output_dim] float tensor with descriptors after
 128       normalization and (possibly) PCA/whitening.
 129   """
 130   # L2-normalize, and if desired apply PCA (followed by L2-normalization).
 131   final_descriptors = tf.nn.l2_normalize(
 132       descriptors, axis=1, name='l2_normalization')
 133 
 134   if use_pca:
 135     # Apply PCA, and whitening if desired.
 136     final_descriptors = ApplyPcaAndWhitening(final_descriptors,
 137                                              pca_parameters['matrix'],
 138                                              pca_parameters['mean'],
 139                                              pca_parameters['dim'],
 140                                              pca_parameters['use_whitening'],
 141                                              pca_parameters['variances'])
 142 
 143     # Re-normalize.
 144     final_descriptors = tf.nn.l2_normalize(
 145         final_descriptors, axis=1, name='pca_l2_normalization')
 146 
 147   return final_descriptors

 150 def DelfFeaturePostProcessing(boxes, descriptors, use_pca, pca_parameters=None):
 151   """Extract DELF features from input image.
 152 
 153   Args:
 154     boxes: [N, 4] float tensor which denotes the selected receptive box. N is
 155       the number of final feature points which pass through keypoint selection
 156       and NMS steps.
 157     descriptors: [N, input_dim] float tensor.
 158     use_pca: Whether to use PCA.
 159     pca_parameters: Only used if `use_pca` is True. Dict containing PCA
 160       parameter tensors, with keys 'mean', 'matrix', 'dim', 'use_whitening',
 161       'variances'.
 162 
 163   Returns:
 164     locations: [N, 2] float tensor which denotes the selected keypoint
 165       locations.
 166     final_descriptors: [N, output_dim] float tensor with DELF descriptors after
 167       normalization and (possibly) PCA/whitening.
 168   """
 169 
 170   # Get center of descriptor boxes, corresponding to feature locations.
 171   locations = CalculateKeypointCenters(boxes)
 172   final_descriptors = PostProcessDescriptors(descriptors, use_pca,
 173                                              pca_parameters)
 174 
 175   return locations, final_descriptors
