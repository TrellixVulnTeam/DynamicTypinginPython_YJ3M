  15 """Local feature aggregation extraction.
  16 
  17 For more details, please refer to the paper:
  18 "Detect-to-Retrieve: Efficient Regional Aggregation for Image Search",
  19 Proc. CVPR'19 (https://arxiv.org/abs/1812.01584).
  20 """
  21 from __future__ import absolute_import
  22 from __future__ import division
  23 from __future__ import print_function
  24 
  25 import numpy as np
  26 import tensorflow as tf
  27 
  28 from delf import aggregation_config_pb2
  29 
  30 _CLUSTER_CENTERS_VAR_NAME = "clusters"
  31 _NORM_SQUARED_TOLERANCE = 1e-12
  32 
  33 # Aliases for aggregation types.
  34 _VLAD = aggregation_config_pb2.AggregationConfig.VLAD
  35 _ASMK = aggregation_config_pb2.AggregationConfig.ASMK
  36 _ASMK_STAR = aggregation_config_pb2.AggregationConfig.ASMK_STAR
  37 
  38 
  39 class ExtractAggregatedRepresentation(object):
  40   """Class for extraction of aggregated local feature representation.
  41 
  42   Args:
  43     aggregation_config: AggregationConfig object defining type of aggregation to
  44       use.
  45 
  46   Raises:
  47     ValueError: If aggregation type is invalid.
  48   """
  49 
  50   def __init__(self, aggregation_config):
  51     self._codebook_size = aggregation_config.codebook_size
  52     self._feature_dimensionality = aggregation_config.feature_dimensionality
  53     self._aggregation_type = aggregation_config.aggregation_type
  54     self._feature_batch_size = aggregation_config.feature_batch_size
  55     self._codebook_path = aggregation_config.codebook_path
  56     self._use_regional_aggregation = aggregation_config.use_regional_aggregation
  57     self._use_l2_normalization = aggregation_config.use_l2_normalization
  58     self._num_assignments = aggregation_config.num_assignments
  59 
  60     if self._aggregation_type  not in [_VLAD, _ASMK, _ASMK_STAR]:
  61       raise ValueError("Invalid aggregation type: %d" % self._aggregation_type)
  62 
  63     # Load codebook
  64     codebook = tf.Variable(
  65         tf.zeros([self._codebook_size, self._feature_dimensionality],
  66                  dtype=tf.float32),
  67         name=_CLUSTER_CENTERS_VAR_NAME)
  68     ckpt = tf.train.Checkpoint(codebook=codebook)
  69     ckpt.restore(self._codebook_path)
  70 
  71     self._codebook = codebook
  72 
  73   def Extract(self, features, num_features_per_region=None):
  74     """Extracts aggregated representation.
  75 
  76     Args:
  77       features: [N, D] float numpy array with N local feature descriptors.
  78       num_features_per_region: Required only if computing regional aggregated
  79         representations, otherwise optional. List of number of features per
  80         region, such that sum(num_features_per_region) = N. It indicates which
  81         features correspond to each region.
  82 
  83     Returns:
  84       aggregated_descriptors: 1-D numpy array.
  85       feature_visual_words: Used only for ASMK/ASMK* aggregation type. 1-D
  86         numpy array denoting visual words corresponding to the
  87         `aggregated_descriptors`.
  88 
  89     Raises:
  90       ValueError: If inputs are misconfigured.
  91     """
  92     features = tf.cast(features, dtype=tf.float32)
  93 
  94     if num_features_per_region is None:
  95       # Use dummy value since it is unused.
  96       num_features_per_region = []
  97     else:
  98       num_features_per_region = tf.cast(num_features_per_region, dtype=tf.int32)
  99       if len(num_features_per_region
 100             ) and sum(num_features_per_region) != features.shape[0]:
 101         raise ValueError(
 102             "Incorrect arguments: sum(num_features_per_region) and "
 103             "features.shape[0] are different: %d vs %d" %
 104             (sum(num_features_per_region), features.shape[0]))
 105 
 106     # Extract features based on desired options.
 107     if self._aggregation_type == _VLAD:
 108       # Feature visual words are unused in the case of VLAD, so just return
 109       # dummy constant.
 110       feature_visual_words = tf.constant(-1, dtype=tf.int32)
 111       if self._use_regional_aggregation:
 112         aggregated_descriptors = self._ComputeRvlad(
 113             features,
 114             num_features_per_region,
 115             self._codebook,
 116             use_l2_normalization=self._use_l2_normalization,
 117             num_assignments=self._num_assignments)
 118       else:
 119         aggregated_descriptors = self._ComputeVlad(
 120             features,
 121             self._codebook,
 122             use_l2_normalization=self._use_l2_normalization,
 123             num_assignments=self._num_assignments)
 124     elif (self._aggregation_type == _ASMK or
 125           self._aggregation_type == _ASMK_STAR):
 126       if self._use_regional_aggregation:
 127         (aggregated_descriptors,
 128          feature_visual_words) = self._ComputeRasmk(
 129              features,
 130              num_features_per_region,
 131              self._codebook,
 132              num_assignments=self._num_assignments)
 133       else:
 134         (aggregated_descriptors,
 135          feature_visual_words) = self._ComputeAsmk(
 136              features,
 137              self._codebook,
 138              num_assignments=self._num_assignments)
 139 
 140     feature_visual_words_output = feature_visual_words.numpy()
 141 
 142     # If using ASMK*/RASMK*, binarize the aggregated descriptors.
 143     if self._aggregation_type == _ASMK_STAR:
 144       reshaped_aggregated_descriptors = np.reshape(
 145           aggregated_descriptors, [-1, self._feature_dimensionality])
 146       packed_descriptors = np.packbits(
 147           reshaped_aggregated_descriptors > 0, axis=1)
 148       aggregated_descriptors_output = np.reshape(packed_descriptors, [-1])
 149     else:
 150       aggregated_descriptors_output = aggregated_descriptors.numpy()
 151 
 152     return aggregated_descriptors_output, feature_visual_words_output
 153 
 154   def _ComputeVlad(self,
 155                    features,
 156                    codebook,
 157                    use_l2_normalization=True,
 158                    num_assignments=1):
 159     """Compute VLAD representation.
 160 
 161     Args:
 162       features: [N, D] float tensor.
 163       codebook: [K, D] float tensor.
 164       use_l2_normalization: If False, does not L2-normalize after aggregation.
 165       num_assignments: Number of visual words to assign a feature to.
 166 
 167     Returns:
 168       vlad: [K*D] float tensor.
 169     """
 170 
 171     def _ComputeVladEmptyFeatures():
 172       """Computes VLAD if `features` is empty.
 173 
 174       Returns:
 175         [K*D] all-zeros tensor.
 176       """
 177       return tf.zeros([self._codebook_size * self._feature_dimensionality],
 178                       dtype=tf.float32)
 179 
 180     def _ComputeVladNonEmptyFeatures():
 181       """Computes VLAD if `features` is not empty.
 182 
 183       Returns:
 184         [K*D] tensor with VLAD descriptor.
 185       """
 186       num_features = tf.shape(features)[0]
 187 
 188       # Find nearest visual words for each feature. Possibly batch the local
 189       # features to avoid OOM.
 190       if self._feature_batch_size <= 0:
 191         actual_batch_size = num_features
 192       else:
 193         actual_batch_size = self._feature_batch_size
 194 
 195       def _BatchNearestVisualWords(ind, selected_visual_words):
 196         """Compute nearest neighbor visual words for a batch of features.
 197 
 198         Args:
 199           ind: Integer index denoting feature.
 200           selected_visual_words: Partial set of visual words.
 201 
 202         Returns:
 203           output_ind: Next index.
 204           output_selected_visual_words: Updated set of visual words, including
 205             the visual words for the new batch.
 206         """
 207         # Handle case of last batch, where there may be fewer than
 208         # `actual_batch_size` features.
 209         batch_size_to_use = tf.cond(
 210             tf.greater(ind + actual_batch_size, num_features),
 211             true_fn=lambda: num_features - ind,
 212             false_fn=lambda: actual_batch_size)
 213 
 214         # Denote B = batch_size_to_use.
 215         # K*B x D.
 216         tiled_features = tf.reshape(
 217             tf.tile(
 218                 tf.slice(features, [ind, 0],
 219                          [batch_size_to_use, self._feature_dimensionality]),
 220                 [1, self._codebook_size]), [-1, self._feature_dimensionality])
 221         # K*B x D.
 222         tiled_codebook = tf.reshape(
 223             tf.tile(tf.reshape(codebook, [1, -1]), [batch_size_to_use, 1]),
 224             [-1, self._feature_dimensionality])
 225         # B x K.
 226         squared_distances = tf.reshape(
 227             tf.reduce_sum(
 228                 tf.math.squared_difference(tiled_features, tiled_codebook),
 229                 axis=1), [batch_size_to_use, self._codebook_size])
 230         # B x K.
 231         nearest_visual_words = tf.argsort(squared_distances)
 232         # B x num_assignments.
 233         batch_selected_visual_words = tf.slice(
 234             nearest_visual_words, [0, 0], [batch_size_to_use, num_assignments])
 235         selected_visual_words = tf.concat(
 236             [selected_visual_words, batch_selected_visual_words], axis=0)
 237 
 238         return ind + batch_size_to_use, selected_visual_words
 239 
 240       ind_batch = tf.constant(0, dtype=tf.int32)
 241       keep_going = lambda j, selected_visual_words: tf.less(j, num_features)
 242       selected_visual_words = tf.zeros([0, num_assignments], dtype=tf.int32)
 243       _, selected_visual_words = tf.while_loop(
 244           cond=keep_going,
 245           body=_BatchNearestVisualWords,
 246           loop_vars=[ind_batch, selected_visual_words],
 247           shape_invariants=[
 248               ind_batch.get_shape(),
 249               tf.TensorShape([None, num_assignments])
 250           ],
 251           parallel_iterations=1,
 252           back_prop=False)
 253 
 254       # Helper function to collect residuals for relevant visual words.
 255       def _ConstructVladFromAssignments(ind, vlad):
 256         """Add contributions of a feature to a VLAD descriptor.
 257 
 258         Args:
 259           ind: Integer index denoting feature.
 260           vlad: Partial VLAD descriptor.
 261 
 262         Returns:
 263           output_ind: Next index (ie, ind+1).
 264           output_vlad: VLAD descriptor updated to take into account contribution
 265             from ind-th feature.
 266         """
 267         diff = tf.tile(
 268             tf.expand_dims(features[ind],
 269                            axis=0), [num_assignments, 1]) - tf.gather(
 270                                codebook, selected_visual_words[ind])
 271         return ind + 1, tf.tensor_scatter_nd_add(
 272             vlad, tf.expand_dims(selected_visual_words[ind], axis=1),
 273             tf.cast(diff, dtype=tf.float32))
 274 
 275       ind_vlad = tf.constant(0, dtype=tf.int32)
 276       keep_going = lambda j, vlad: tf.less(j, num_features)
 277       vlad = tf.zeros([self._codebook_size, self._feature_dimensionality],
 278                       dtype=tf.float32)
 279       _, vlad = tf.while_loop(
 280           cond=keep_going,
 281           body=_ConstructVladFromAssignments,
 282           loop_vars=[ind_vlad, vlad],
 283           back_prop=False)
 284 
 285       vlad = tf.reshape(vlad,
 286                         [self._codebook_size * self._feature_dimensionality])
 287       if use_l2_normalization:
 288         vlad = tf.math.l2_normalize(vlad, epsilon=_NORM_SQUARED_TOLERANCE)
 289 
 290       return vlad
 291 
 292     return tf.cond(
 293         tf.greater(tf.size(features), 0),
 294         true_fn=_ComputeVladNonEmptyFeatures,
 295         false_fn=_ComputeVladEmptyFeatures)
 296 
 297   def _ComputeRvlad(self,
 298                     features,
 299                     num_features_per_region,
 300                     codebook,
 301                     use_l2_normalization=False,
 302                     num_assignments=1):
 303     """Compute R-VLAD representation.
 304 
 305     Args:
 306       features: [N, D] float tensor.
 307       num_features_per_region: [R] int tensor. Contains number of features per
 308         region, such that sum(num_features_per_region) = N. It indicates which
 309         features correspond to each region.
 310       codebook: [K, D] float tensor.
 311       use_l2_normalization: If True, performs L2-normalization after regional
 312         aggregation; if False (default), performs componentwise division by R
 313         after regional aggregation.
 314       num_assignments: Number of visual words to assign a feature to.
 315 
 316     Returns:
 317       rvlad: [K*D] float tensor.
 318     """
 319 
 320     def _ComputeRvladEmptyRegions():
 321       """Computes R-VLAD if `num_features_per_region` is empty.
 322 
 323       Returns:
 324         [K*D] all-zeros tensor.
 325       """
 326       return tf.zeros([self._codebook_size * self._feature_dimensionality],
 327                       dtype=tf.float32)
 328 
 329     def _ComputeRvladNonEmptyRegions():
 330       """Computes R-VLAD if `num_features_per_region` is not empty.
 331 
 332       Returns:
 333         [K*D] tensor with R-VLAD descriptor.
 334       """
 335 
 336       # Helper function to compose initial R-VLAD from image regions.
 337       def _ConstructRvladFromVlad(ind, rvlad):
 338         """Add contributions from different regions into R-VLAD.
 339 
 340         Args:
 341           ind: Integer index denoting region.
 342           rvlad: Partial R-VLAD descriptor.
 343 
 344         Returns:
 345           output_ind: Next index (ie, ind+1).
 346           output_rvlad: R-VLAD descriptor updated to take into account
 347             contribution from ind-th region.
 348         """
 349         return ind + 1, rvlad + self._ComputeVlad(
 350             tf.slice(
 351                 features, [tf.reduce_sum(num_features_per_region[:ind]), 0],
 352                 [num_features_per_region[ind], self._feature_dimensionality]),
 353             codebook,
 354             num_assignments=num_assignments)
 355 
 356       i = tf.constant(0, dtype=tf.int32)
 357       num_regions = tf.shape(num_features_per_region)[0]
 358       keep_going = lambda j, rvlad: tf.less(j, num_regions)
 359       rvlad = tf.zeros([self._codebook_size * self._feature_dimensionality],
 360                        dtype=tf.float32)
 361       _, rvlad = tf.while_loop(
 362           cond=keep_going,
 363           body=_ConstructRvladFromVlad,
 364           loop_vars=[i, rvlad],
 365           back_prop=False,
 366           parallel_iterations=1)
 367 
 368       if use_l2_normalization:
 369         rvlad = tf.math.l2_normalize(rvlad, epsilon=_NORM_SQUARED_TOLERANCE)
 370       else:
 371         rvlad /= tf.cast(num_regions, dtype=tf.float32)
 372 
 373       return rvlad
 374 
 375     return tf.cond(
 376         tf.greater(tf.size(num_features_per_region), 0),
 377         true_fn=_ComputeRvladNonEmptyRegions,
 378         false_fn=_ComputeRvladEmptyRegions)
 379 
 380   def _PerCentroidNormalization(self, unnormalized_vector):
 381     """Perform per-centroid normalization.
 382 
 383     Args:
 384       unnormalized_vector: [KxD] float tensor.
 385 
 386     Returns:
 387       per_centroid_normalized_vector: [KxD] float tensor, with normalized
 388         aggregated residuals. Some residuals may be all-zero.
 389       visual_words: Int tensor containing indices of visual words which are
 390         present for the set of features.
 391     """
 392     unnormalized_vector = tf.reshape(
 393         unnormalized_vector,
 394         [self._codebook_size, self._feature_dimensionality])
 395     per_centroid_norms = tf.norm(unnormalized_vector, axis=1)
 396 
 397     visual_words = tf.reshape(
 398         tf.where(
 399             tf.greater(
 400                 per_centroid_norms,
 401                 tf.cast(tf.sqrt(_NORM_SQUARED_TOLERANCE), dtype=tf.float32))),
 402         [-1])
 403 
 404     per_centroid_normalized_vector = tf.math.l2_normalize(
 405         unnormalized_vector, axis=1, epsilon=_NORM_SQUARED_TOLERANCE)
 406 
 407     return per_centroid_normalized_vector, visual_words
 408 
 409   def _ComputeAsmk(self, features, codebook, num_assignments=1):
 410     """Compute ASMK representation.
 411 
 412     Args:
 413       features: [N, D] float tensor.
 414       codebook: [K, D] float tensor.
 415       num_assignments: Number of visual words to assign a feature to.
 416 
 417     Returns:
 418       normalized_residuals: 1-dimensional float tensor with concatenated
 419         residuals which are non-zero. Note that the dimensionality is
 420         input-dependent.
 421       visual_words: 1-dimensional int tensor of sorted visual word ids.
 422         Dimensionality is shape(normalized_residuals)[0] / D.
 423     """
 424     unnormalized_vlad = self._ComputeVlad(
 425         features,
 426         codebook,
 427         use_l2_normalization=False,
 428         num_assignments=num_assignments)
 429 
 430     per_centroid_normalized_vlad, visual_words = self._PerCentroidNormalization(
 431         unnormalized_vlad)
 432 
 433     normalized_residuals = tf.reshape(
 434         tf.gather(per_centroid_normalized_vlad, visual_words),
 435         [tf.shape(visual_words)[0] * self._feature_dimensionality])
 436 
 437     return normalized_residuals, visual_words
 438 
 439   def _ComputeRasmk(self,
 440                     features,
 441                     num_features_per_region,
 442                     codebook,
 443                     num_assignments=1):
 444     """Compute R-ASMK representation.
 445 
 446     Args:
 447       features: [N, D] float tensor.
 448       num_features_per_region: [R] int tensor. Contains number of features per
 449         region, such that sum(num_features_per_region) = N. It indicates which
 450         features correspond to each region.
 451       codebook: [K, D] float tensor.
 452       num_assignments: Number of visual words to assign a feature to.
 453 
 454     Returns:
 455       normalized_residuals: 1-dimensional float tensor with concatenated
 456         residuals which are non-zero. Note that the dimensionality is
 457         input-dependent.
 458       visual_words: 1-dimensional int tensor of sorted visual word ids.
 459         Dimensionality is shape(normalized_residuals)[0] / D.
 460     """
 461     unnormalized_rvlad = self._ComputeRvlad(
 462         features,
 463         num_features_per_region,
 464         codebook,
 465         use_l2_normalization=False,
 466         num_assignments=num_assignments)
 467 
 468     (per_centroid_normalized_rvlad,
 469      visual_words) = self._PerCentroidNormalization(unnormalized_rvlad)
 470 
 471     normalized_residuals = tf.reshape(
 472         tf.gather(per_centroid_normalized_rvlad, visual_words),
 473         [tf.shape(visual_words)[0] * self._feature_dimensionality])
 474 
 475     return normalized_residuals, visual_words
