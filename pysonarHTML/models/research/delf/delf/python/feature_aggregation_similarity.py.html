  15 """Local feature aggregation similarity computation.
  16 
  17 For more details, please refer to the paper:
  18 "Detect-to-Retrieve: Efficient Regional Aggregation for Image Search",
  19 Proc. CVPR'19 (https://arxiv.org/abs/1812.01584).
  20 """
  21 from __future__ import absolute_import
  22 from __future__ import division
  23 from __future__ import print_function
  24 
  25 import numpy as np
  26 
  27 from delf import aggregation_config_pb2
  28 
  29 # Aliases for aggregation types.
  30 _VLAD = aggregation_config_pb2.AggregationConfig.VLAD
  31 _ASMK = aggregation_config_pb2.AggregationConfig.ASMK
  32 _ASMK_STAR = aggregation_config_pb2.AggregationConfig.ASMK_STAR
  33 
  34 
  35 class SimilarityAggregatedRepresentation(object):
  36   """Class for computing similarity of aggregated local feature representations.
  37 
  38   Args:
  39     aggregation_config: AggregationConfig object defining type of aggregation to
  40       use.
  41 
  42   Raises:
  43     ValueError: If aggregation type is invalid.
  44   """
  45 
  46   def __init__(self, aggregation_config):
  47     self._feature_dimensionality = aggregation_config.feature_dimensionality
  48     self._aggregation_type = aggregation_config.aggregation_type
  49 
  50     # Only relevant if using ASMK/ASMK*. Otherwise, ignored.
  51     self._use_l2_normalization = aggregation_config.use_l2_normalization
  52     self._alpha = aggregation_config.alpha
  53     self._tau = aggregation_config.tau
  54 
  55     # Only relevant if using ASMK*. Otherwise, ignored.
  56     self._number_bits = np.array([bin(n).count('1') for n in range(256)])
  57 
  58   def ComputeSimilarity(self,
  59                         aggregated_descriptors_1,
  60                         aggregated_descriptors_2,
  61                         feature_visual_words_1=None,
  62                         feature_visual_words_2=None):
  63     """Computes similarity between aggregated descriptors.
  64 
  65     Args:
  66       aggregated_descriptors_1: 1-D NumPy array.
  67       aggregated_descriptors_2: 1-D NumPy array.
  68       feature_visual_words_1: Used only for ASMK/ASMK* aggregation type. 1-D
  69         sorted NumPy integer array denoting visual words corresponding to
  70         `aggregated_descriptors_1`.
  71       feature_visual_words_2: Used only for ASMK/ASMK* aggregation type. 1-D
  72         sorted NumPy integer array denoting visual words corresponding to
  73         `aggregated_descriptors_2`.
  74 
  75     Returns:
  76       similarity: Float. The larger, the more similar.
  77 
  78     Raises:
  79       ValueError: If aggregation type is invalid.
  80     """
  81     if self._aggregation_type == _VLAD:
  82       similarity = np.dot(aggregated_descriptors_1, aggregated_descriptors_2)
  83     elif self._aggregation_type == _ASMK:
  84       similarity = self._AsmkSimilarity(
  85           aggregated_descriptors_1,
  86           aggregated_descriptors_2,
  87           feature_visual_words_1,
  88           feature_visual_words_2,
  89           binarized=False)
  90     elif self._aggregation_type == _ASMK_STAR:
  91       similarity = self._AsmkSimilarity(
  92           aggregated_descriptors_1,
  93           aggregated_descriptors_2,
  94           feature_visual_words_1,
  95           feature_visual_words_2,
  96           binarized=True)
  97     else:
  98       raise ValueError('Invalid aggregation type: %d' % self._aggregation_type)
  99 
 100     return similarity
 101 
 102   def _CheckAsmkDimensionality(self, aggregated_descriptors, num_visual_words,
 103                                descriptor_name):
 104     """Checks that ASMK dimensionality is as expected.
 105 
 106     Args:
 107       aggregated_descriptors: 1-D NumPy array.
 108       num_visual_words: Integer.
 109       descriptor_name: String.
 110 
 111     Raises:
 112       ValueError: If descriptor dimensionality is incorrect.
 113     """
 114     if len(aggregated_descriptors
 115           ) / num_visual_words != self._feature_dimensionality:
 116       raise ValueError(
 117           'Feature dimensionality for aggregated descriptor %s is invalid: %d;'
 118           ' expected %d.' % (descriptor_name, len(aggregated_descriptors) /
 119                              num_visual_words, self._feature_dimensionality))
 120 
 121   def _SigmaFn(self, x):
 122     """Selectivity ASMK/ASMK* similarity function.
 123 
 124     Args:
 125       x: Scalar or 1-D NumPy array.
 126 
 127     Returns:
 128       result: Same type as input, with output of selectivity function.
 129     """
 130     if np.isscalar(x):
 131       if x > self._tau:
 132         result = np.sign(x) * np.power(np.absolute(x), self._alpha)
 133       else:
 134         result = 0.0
 135     else:
 136       result = np.zeros_like(x)
 137       above_tau = np.nonzero(x > self._tau)
 138       result[above_tau] = np.sign(x[above_tau]) * np.power(
 139           np.absolute(x[above_tau]), self._alpha)
 140 
 141     return result
 142 
 143   def _BinaryNormalizedInnerProduct(self, descriptors_1, descriptors_2):
 144     """Computes normalized binary inner product.
 145 
 146     Args:
 147       descriptors_1: 1-D NumPy integer array.
 148       descriptors_2: 1-D NumPy integer array.
 149 
 150     Returns:
 151       inner_product: Float.
 152 
 153     Raises:
 154       ValueError: If the dimensionality of descriptors is different.
 155     """
 156     num_descriptors = len(descriptors_1)
 157     if num_descriptors != len(descriptors_2):
 158       raise ValueError(
 159           'Descriptors have incompatible dimensionality: %d vs %d' %
 160           (len(descriptors_1), len(descriptors_2)))
 161 
 162     h = 0
 163     for i in range(num_descriptors):
 164       h += self._number_bits[np.bitwise_xor(descriptors_1[i], descriptors_2[i])]
 165 
 166     # If local feature dimensionality is lower than 8, then use that to compute
 167     # proper binarized inner product.
 168     bits_per_descriptor = min(self._feature_dimensionality, 8)
 169 
 170     total_num_bits = bits_per_descriptor * num_descriptors
 171 
 172     return 1.0 - 2.0 * h / total_num_bits
 173 
 174   def _AsmkSimilarity(self,
 175                       aggregated_descriptors_1,
 176                       aggregated_descriptors_2,
 177                       visual_words_1,
 178                       visual_words_2,
 179                       binarized=False):
 180     """Compute ASMK-based similarity.
 181 
 182     If `aggregated_descriptors_1` or `aggregated_descriptors_2` is empty, we
 183     return a similarity of -1.0.
 184 
 185     If binarized is True, `aggregated_descriptors_1` and
 186     `aggregated_descriptors_2` must be of type uint8.
 187 
 188     Args:
 189       aggregated_descriptors_1: 1-D NumPy array.
 190       aggregated_descriptors_2: 1-D NumPy array.
 191       visual_words_1: 1-D sorted NumPy integer array denoting visual words
 192         corresponding to `aggregated_descriptors_1`.
 193       visual_words_2: 1-D sorted NumPy integer array denoting visual words
 194         corresponding to `aggregated_descriptors_2`.
 195       binarized: If True, compute ASMK* similarity.
 196 
 197     Returns:
 198       similarity: Float. The larger, the more similar.
 199 
 200     Raises:
 201       ValueError: If input descriptor dimensionality is inconsistent, or if
 202         descriptor type is unsupported.
 203     """
 204     num_visual_words_1 = len(visual_words_1)
 205     num_visual_words_2 = len(visual_words_2)
 206 
 207     if not num_visual_words_1 or not num_visual_words_2:
 208       return -1.0
 209 
 210     # Parse dimensionality used per visual word. They must be the same for both
 211     # aggregated descriptors. If using ASMK, they also must be equal to
 212     # self._feature_dimensionality.
 213     if binarized:
 214       if aggregated_descriptors_1.dtype != 'uint8':
 215         raise ValueError('Incorrect input descriptor type: %s' %
 216                          aggregated_descriptors_1.dtype)
 217       if aggregated_descriptors_2.dtype != 'uint8':
 218         raise ValueError('Incorrect input descriptor type: %s' %
 219                          aggregated_descriptors_2.dtype)
 220 
 221       per_visual_word_dimensionality = int(
 222           len(aggregated_descriptors_1) / num_visual_words_1)
 223       if len(aggregated_descriptors_2
 224             ) / num_visual_words_2 != per_visual_word_dimensionality:
 225         raise ValueError('ASMK* dimensionality is inconsistent.')
 226     else:
 227       per_visual_word_dimensionality = self._feature_dimensionality
 228       self._CheckAsmkDimensionality(aggregated_descriptors_1,
 229                                     num_visual_words_1, '1')
 230       self._CheckAsmkDimensionality(aggregated_descriptors_2,
 231                                     num_visual_words_2, '2')
 232 
 233     aggregated_descriptors_1_reshape = np.reshape(
 234         aggregated_descriptors_1,
 235         [num_visual_words_1, per_visual_word_dimensionality])
 236     aggregated_descriptors_2_reshape = np.reshape(
 237         aggregated_descriptors_2,
 238         [num_visual_words_2, per_visual_word_dimensionality])
 239 
 240     # Loop over visual words, compute similarity.
 241     unnormalized_similarity = 0.0
 242     ind_1 = 0
 243     ind_2 = 0
 244     while ind_1 < num_visual_words_1 and ind_2 < num_visual_words_2:
 245       if visual_words_1[ind_1] == visual_words_2[ind_2]:
 246         if binarized:
 247           inner_product = self._BinaryNormalizedInnerProduct(
 248               aggregated_descriptors_1_reshape[ind_1],
 249               aggregated_descriptors_2_reshape[ind_2])
 250         else:
 251           inner_product = np.dot(aggregated_descriptors_1_reshape[ind_1],
 252                                  aggregated_descriptors_2_reshape[ind_2])
 253         unnormalized_similarity += self._SigmaFn(inner_product)
 254         ind_1 += 1
 255         ind_2 += 1
 256       elif visual_words_1[ind_1] > visual_words_2[ind_2]:
 257         ind_2 += 1
 258       else:
 259         ind_1 += 1
 260 
 261     final_similarity = unnormalized_similarity
 262     if self._use_l2_normalization:
 263       final_similarity /= np.sqrt(num_visual_words_1 * num_visual_words_2)
 264 
 265     return final_similarity
