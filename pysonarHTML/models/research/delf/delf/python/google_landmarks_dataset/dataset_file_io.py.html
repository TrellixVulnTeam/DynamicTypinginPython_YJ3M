  25 RECOGNITION_TASK_ID = 'recognition'
  26 RETRIEVAL_TASK_ID = 'retrieval'

  29 def ReadSolution(file_path, task):
  30   """Reads solution from file, for a given task.
  31 
  32   Args:
  33     file_path: Path to CSV file with solution. File contains a header.
  34     task: Type of challenge task. Supported values: 'recognition', 'retrieval'.
  35 
  36   Returns:
  37     public_solution: Dict mapping test image ID to list of ground-truth IDs, for
  38       the Public subset of test images. If `task` == 'recognition', the IDs are
  39       integers corresponding to landmark IDs. If `task` == 'retrieval', the IDs
  40       are strings corresponding to index image IDs.
  41     private_solution: Same as `public_solution`, but for the private subset of
  42       test images.
  43     ignored_ids: List of test images that are ignored in scoring.
  44 
  45   Raises:
  46     ValueError: If Usage field is not Public, Private or Ignored; or if `task`
  47       is not supported.
  48   """
  49   public_solution = {}
  50   private_solution = {}
  51   ignored_ids = []
  52   with tf.io.gfile.GFile(file_path, 'r') as csv_file:
  53     reader = csv.reader(csv_file)
  54     next(reader, None)  # Skip header.
  55     for row in reader:
  56       test_id = row[0]
  57       if row[2] == 'Ignored':
  58         ignored_ids.append(test_id)
  59       else:
  60         ground_truth_ids = []
  61         if task == RECOGNITION_TASK_ID:
  62           if row[1]:
  63             for landmark_id in row[1].split(' '):
  64               ground_truth_ids.append(int(landmark_id))
  65         elif task == RETRIEVAL_TASK_ID:
  66           for image_id in row[1].split(' '):
  67             ground_truth_ids.append(image_id)
  68         else:
  69           raise ValueError('Unrecognized task: %s' % task)
  70 
  71         if row[2] == 'Public':
  72           public_solution[test_id] = ground_truth_ids
  73         elif row[2] == 'Private':
  74           private_solution[test_id] = ground_truth_ids
  75         else:
  76           raise ValueError('Test image %s has unrecognized Usage tag %s' %
  77                            (row[0], row[2]))
  78 
  79   return public_solution, private_solution, ignored_ids
  80

  82 def ReadPredictions(file_path, public_ids, private_ids, ignored_ids, task):
  83   """Reads predictions from file, for a given task.
  84 
  85   Args:
  86     file_path: Path to CSV file with predictions. File contains a header.
  87     public_ids: Set (or list) of test image IDs in Public subset of test images.
  88     private_ids: Same as `public_ids`, but for the private subset of test
  89       images.
  90     ignored_ids: Set (or list) of test image IDs that are ignored in scoring and
  91       are associated to no ground-truth.
  92     task: Type of challenge task. Supported values: 'recognition', 'retrieval'.
  93 
  94   Returns:
  95     public_predictions: Dict mapping test image ID to prediction, for the Public
  96       subset of test images. If `task` == 'recognition', the prediction is a
  97       dict with keys 'class' (integer) and 'score' (float). If `task` ==
  98       'retrieval', the prediction is a list of strings corresponding to index
  99       image IDs.
 100     private_predictions: Same as `public_predictions`, but for the private
 101       subset of test images.
 102 
 103   Raises:
 104     ValueError:
 105       - If test image ID is unrecognized/repeated;
 106       - If `task` is not supported;
 107       - If prediction is malformed.
 108   """
 109   public_predictions = {}
 110   private_predictions = {}
 111   with tf.io.gfile.GFile(file_path, 'r') as csv_file:
 112     reader = csv.reader(csv_file)
 113     next(reader, None)  # Skip header.
 114     for row in reader:
 115       # Skip row if empty.
 116       if not row:
 117         continue
 118 
 119       test_id = row[0]
 120 
 121       # Makes sure this query has not yet been seen.
 122       if test_id in public_predictions:
 123         raise ValueError('Test image %s is repeated.' % test_id)
 124       if test_id in private_predictions:
 125         raise ValueError('Test image %s is repeated' % test_id)
 126 
 127       # If ignored, skip it.
 128       if test_id in ignored_ids:
 129         continue
 130 
 131       # Only parse result if there is a prediction.
 132       if row[1]:
 133         prediction_split = row[1].split(' ')
 134         # Remove empty spaces at end (if any).
 135         if not prediction_split[-1]:
 136           prediction_split = prediction_split[:-1]
 137 
 138         if task == RECOGNITION_TASK_ID:
 139           if len(prediction_split) != 2:
 140             raise ValueError('Prediction is malformed: there should only be 2 '
 141                              'elements in second column, but found %d for test '
 142                              'image %s' % (len(prediction_split), test_id))
 143 
 144           landmark_id = int(prediction_split[0])
 145           score = float(prediction_split[1])
 146           prediction_entry = {'class': landmark_id, 'score': score}
 147         elif task == RETRIEVAL_TASK_ID:
 148           prediction_entry = prediction_split
 149         else:
 150           raise ValueError('Unrecognized task: %s' % task)
 151 
 152         if test_id in public_ids:
 153           public_predictions[test_id] = prediction_entry
 154         elif test_id in private_ids:
 155           private_predictions[test_id] = prediction_entry
 156         else:
 157           raise ValueError('test_id %s is unrecognized' % test_id)
 158 
 159   return public_predictions, private_predictions
