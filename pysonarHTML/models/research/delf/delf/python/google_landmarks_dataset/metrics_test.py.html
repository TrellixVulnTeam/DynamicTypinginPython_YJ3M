  96   def testGlobalAveragePrecisionWorks(self):
  97     # Define input.
  98     predictions = _CreateRecognitionPredictions()
  99     solution = _CreateRecognitionSolution()
 100 
 101     # Run tested function.
 102     gap = metrics.GlobalAveragePrecision(predictions, solution)
 103 
 104     # Define expected results.
 105     expected_gap = 0.166667
 106 
 107     # Compare actual and expected results.
 108     self.assertAllClose(gap, expected_gap)
