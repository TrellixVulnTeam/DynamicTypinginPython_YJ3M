  24 def _CountPositives(solution):
  25   """Counts number of test images with non-empty ground-truth in `solution`.
  26 
  27   Args:
  28     solution: Dict mapping test image ID to list of ground-truth IDs.
  29 
  30   Returns:
  31     count: Number of test images with non-empty ground-truth.
  32   """
  33   count = 0
  34   for v in solution.values():
  35     if v:
  36       count += 1
  37 
  38   return count

  41 def GlobalAveragePrecision(predictions,
  42                            recognition_solution,
  43                            ignore_non_gt_test_images=False):
  44   """Computes global average precision for recognition prediction.
  45 
  46   Args:
  47     predictions: Dict mapping test image ID to a dict with keys 'class'
  48       (integer) and 'score' (float).
  49     recognition_solution: Dict mapping test image ID to list of ground-truth
  50       landmark IDs.
  51     ignore_non_gt_test_images: If True, ignore test images which do not have
  52       associated ground-truth landmark IDs. For the Google Landmark Recognition
  53       challenge, this should be set to False.
  54 
  55   Returns:
  56     gap: Global average precision score (float).
  57   """
  58   # Compute number of expected results.
  59   num_positives = _CountPositives(recognition_solution)
  60 
  61   gap = 0.0
  62   total_predictions = 0
  63   correct_predictions = 0
  64 
  65   # Sort predictions according to Kaggle's convention:
  66   # - first by score (descending);
  67   # - then by key (ascending);
  68   # - then by class (ascending).
  69   sorted_predictions_by_key_class = sorted(
  70       predictions.items(), key=lambda item: (item[0], item[1]['class']))
  71   sorted_predictions = sorted(
  72       sorted_predictions_by_key_class,
  73       key=lambda item: item[1]['score'],
  74       reverse=True)
  75 
  76   # Loop over sorted predictions (descending order) and compute GAPs.
  77   for key, prediction in sorted_predictions:
  78     if ignore_non_gt_test_images and not recognition_solution[key]:
  79       continue
  80 
  81     total_predictions += 1
  82     if prediction['class'] in recognition_solution[key]:
  83       correct_predictions += 1
  84       gap += correct_predictions / total_predictions
  85 
  86   gap /= num_positives
  87 
  88   return gap

  91 def Top1Accuracy(predictions, recognition_solution):
  92   """Computes top-1 accuracy for recognition prediction.
  93 
  94   Note that test images without ground-truth are ignored.
  95 
  96   Args:
  97     predictions: Dict mapping test image ID to a dict with keys 'class'
  98       (integer) and 'score' (float).
  99     recognition_solution: Dict mapping test image ID to list of ground-truth
 100       landmark IDs.
 101 
 102   Returns:
 103     accuracy: Top-1 accuracy (float).
 104   """
 105   # Loop over test images in solution. If it has at least one class label, we
 106   # check if the predicion is correct.
 107   num_correct_predictions = 0
 108   num_test_images_with_ground_truth = 0
 109   for key, ground_truth in recognition_solution.items():
 110     if ground_truth:
 111       num_test_images_with_ground_truth += 1
 112       if key in predictions:
 113         if predictions[key]['class'] in ground_truth:
 114           num_correct_predictions += 1
 115 
 116   return num_correct_predictions / num_test_images_with_ground_truth

 119 def MeanAveragePrecision(predictions, retrieval_solution, max_predictions=100):
 120   """Computes mean average precision for retrieval prediction.
 121 
 122   Args:
 123     predictions: Dict mapping test image ID to a list of strings corresponding
 124       to index image IDs.
 125     retrieval_solution: Dict mapping test image ID to list of ground-truth image
 126       IDs.
 127     max_predictions: Maximum number of predictions per query to take into
 128       account. For the Google Landmark Retrieval challenge, this should be set
 129       to 100.
 130 
 131   Returns:
 132     mean_ap: Mean average precision score (float).
 133 
 134   Raises:
 135     ValueError: If a test image in `predictions` is not included in
 136       `retrieval_solutions`.
 137   """
 138   # Compute number of test images.
 139   num_test_images = len(retrieval_solution.keys())
 140 
 141   # Loop over predictions for each query and compute mAP.
 142   mean_ap = 0.0
 143   for key, prediction in predictions.items():
 144     if key not in retrieval_solution:
 145       raise ValueError('Test image %s is not part of retrieval_solution' % key)
 146 
 147     # Loop over predicted images, keeping track of those which were already
 148     # used (duplicates are skipped).
 149     ap = 0.0
 150     already_predicted = set()
 151     num_expected_retrieved = min(len(retrieval_solution[key]), max_predictions)
 152     num_correct = 0
 153     for i in range(min(len(prediction), max_predictions)):
 154       if prediction[i] not in already_predicted:
 155         if prediction[i] in retrieval_solution[key]:
 156           num_correct += 1
 157           ap += num_correct / (i + 1)
 158         already_predicted.add(prediction[i])
 159 
 160     ap /= num_expected_retrieved
 161     mean_ap += ap
 162 
 163   mean_ap /= num_test_images
 164 
 165   return mean_ap

 168 def MeanPrecisions(predictions, retrieval_solution, max_predictions=100):
 169   """Computes mean precisions for retrieval prediction.
 170 
 171   Args:
 172     predictions: Dict mapping test image ID to a list of strings corresponding
 173       to index image IDs.
 174     retrieval_solution: Dict mapping test image ID to list of ground-truth image
 175       IDs.
 176     max_predictions: Maximum number of predictions per query to take into
 177       account.
 178 
 179   Returns:
 180     mean_precisions: NumPy array with mean precisions at ranks 1 through
 181       `max_predictions`.
 182 
 183   Raises:
 184     ValueError: If a test image in `predictions` is not included in
 185       `retrieval_solutions`.
 186   """
 187   # Compute number of test images.
 188   num_test_images = len(retrieval_solution.keys())
 189 
 190   # Loop over predictions for each query and compute precisions@k.
 191   precisions = np.zeros((num_test_images, max_predictions))
 192   count_test_images = 0
 193   for key, prediction in predictions.items():
 194     if key not in retrieval_solution:
 195       raise ValueError('Test image %s is not part of retrieval_solution' % key)
 196 
 197     # Loop over predicted images, keeping track of those which were already
 198     # used (duplicates are skipped).
 199     already_predicted = set()
 200     num_correct = 0
 201     for i in range(max_predictions):
 202       if i < len(prediction):
 203         if prediction[i] not in already_predicted:
 204           if prediction[i] in retrieval_solution[key]:
 205             num_correct += 1
 206           already_predicted.add(prediction[i])
 207       precisions[count_test_images, i] = num_correct / (i + 1)
 208     count_test_images += 1
 209 
 210   mean_precisions = np.mean(precisions, axis=0)
 211 
 212   return mean_precisions

 215 def MeanMedianPosition(predictions, retrieval_solution, max_predictions=100):
 216   """Computes mean and median positions of first correct image.
 217 
 218   Args:
 219     predictions: Dict mapping test image ID to a list of strings corresponding
 220       to index image IDs.
 221     retrieval_solution: Dict mapping test image ID to list of ground-truth image
 222       IDs.
 223     max_predictions: Maximum number of predictions per query to take into
 224       account.
 225 
 226   Returns:
 227     mean_position: Float.
 228     median_position: Float.
 229 
 230   Raises:
 231     ValueError: If a test image in `predictions` is not included in
 232       `retrieval_solutions`.
 233   """
 234   # Compute number of test images.
 235   num_test_images = len(retrieval_solution.keys())
 236 
 237   # Loop over predictions for each query to find first correct ranked image.
 238   positions = (max_predictions + 1) * np.ones((num_test_images))
 239   count_test_images = 0
 240   for key, prediction in predictions.items():
 241     if key not in retrieval_solution:
 242       raise ValueError('Test image %s is not part of retrieval_solution' % key)
 243 
 244     for i in range(min(len(prediction), max_predictions)):
 245       if prediction[i] in retrieval_solution[key]:
 246         positions[count_test_images] = i + 1
 247         break
 248 
 249     count_test_images += 1
 250 
 251   mean_position = np.mean(positions)
 252   median_position = np.median(positions)
 253 
 254   return mean_position, median_position
