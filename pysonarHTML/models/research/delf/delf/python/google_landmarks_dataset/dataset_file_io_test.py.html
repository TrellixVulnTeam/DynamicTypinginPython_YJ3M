  33   def testReadRecognitionSolutionWorks(self):
  34     # Define inputs.
  35     file_path = os.path.join(FLAGS.test_tmpdir,
  36                              'recognition_solution.csv')
  37     with tf.io.gfile.GFile(file_path, 'w') as f:
  38       f.write('id,landmarks,Usage\n')
  39       f.write('0123456789abcdef,0 12,Public\n')
  40       f.write('0223456789abcdef,,Public\n')
  41       f.write('0323456789abcdef,100,Ignored\n')
  42       f.write('0423456789abcdef,1,Private\n')
  43       f.write('0523456789abcdef,,Ignored\n')
  44 
  45     # Run tested function.
  46     (public_solution, private_solution,
  47      ignored_ids) = dataset_file_io.ReadSolution(
  48          file_path, dataset_file_io.RECOGNITION_TASK_ID)
  49 
  50     # Define expected results.
  51     expected_public_solution = {
  52         '0123456789abcdef': [0, 12],
  53         '0223456789abcdef': []
  54     }
  55     expected_private_solution = {
  56         '0423456789abcdef': [1],
  57     }
  58     expected_ignored_ids = ['0323456789abcdef', '0523456789abcdef']
  59 
  60     # Compare actual and expected results.
  61     self.assertEqual(public_solution, expected_public_solution)
  62     self.assertEqual(private_solution, expected_private_solution)
  63     self.assertEqual(ignored_ids, expected_ignored_ids)

  97   def testReadRecognitionPredictionsWorks(self):
  98     # Define inputs.
  99     file_path = os.path.join(FLAGS.test_tmpdir,
 100                              'recognition_predictions.csv')
 101     with tf.io.gfile.GFile(file_path, 'w') as f:
 102       f.write('id,landmarks\n')
 103       f.write('0123456789abcdef,12 0.1 \n')
 104       f.write('0423456789abcdef,0 19.0\n')
 105       f.write('0223456789abcdef,\n')
 106       f.write('\n')
 107       f.write('0523456789abcdef,14 0.01\n')
 108     public_ids = ['0123456789abcdef', '0223456789abcdef']
 109     private_ids = ['0423456789abcdef']
 110     ignored_ids = ['0323456789abcdef', '0523456789abcdef']
 111 
 112     # Run tested function.
 113     public_predictions, private_predictions = dataset_file_io.ReadPredictions(
 114         file_path, public_ids, private_ids, ignored_ids,
 115         dataset_file_io.RECOGNITION_TASK_ID)
 116 
 117     # Define expected results.
 118     expected_public_predictions = {
 119         '0123456789abcdef': {
 120             'class': 12,
 121             'score': 0.1
 122         }
 123     }
 124     expected_private_predictions = {
 125         '0423456789abcdef': {
 126             'class': 0,
 127             'score': 19.0
 128         }
 129     }
 130 
 131     # Compare actual and expected results.
 132     self.assertEqual(public_predictions, expected_public_predictions)
 133     self.assertEqual(private_predictions, expected_private_predictions)

 135   def testReadRetrievalPredictionsWorks(self):
 136     # Define inputs.
 137     file_path = os.path.join(FLAGS.test_tmpdir,
 138                              'retrieval_predictions.csv')
 139     with tf.io.gfile.GFile(file_path, 'w') as f:

 143       f.write('0223456789abcdef,fedcba9876543210 fedcba9876543200 '
 144               'fedcba9876543220\n')
 145       f.write('\n')
 146       f.write('0523456789abcdef,\n')
 147     public_ids = ['0223456789abcdef']
 148     private_ids = ['0323456789abcdef', '0423456789abcdef']
 149     ignored_ids = ['0123456789abcdef', '0523456789abcdef']
 150 
 151     # Run tested function.
 152     public_predictions, private_predictions = dataset_file_io.ReadPredictions(
 153         file_path, public_ids, private_ids, ignored_ids,
 154         dataset_file_io.RETRIEVAL_TASK_ID)
 155 
 156     # Define expected results.
 157     expected_public_predictions = {
 158         '0223456789abcdef': [
 159             'fedcba9876543210', 'fedcba9876543200', 'fedcba9876543220'
 160         ]
 161     }
 162     expected_private_predictions = {'0423456789abcdef': ['fedcba9876543260']}
 163 
 164     # Compare actual and expected results.
 165     self.assertEqual(public_predictions, expected_public_predictions)
 166     self.assertEqual(private_predictions, expected_private_predictions)
