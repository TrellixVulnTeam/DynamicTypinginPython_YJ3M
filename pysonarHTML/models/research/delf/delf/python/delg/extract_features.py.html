  15 """Extracts DELG features for images from Revisited Oxford/Paris datasets.
  16 
  17 Note that query images are cropped before feature extraction, as required by the
  18 evaluation protocols of these datasets.
  19 
  20 The types of extracted features (local and/or global) depend on the input
  21 DelfConfig.
  22 
  23 The program checks if features already exist, and skips computation for those.
  24 """
  25 
  26 from __future__ import absolute_import
  27 from __future__ import division
  28 from __future__ import print_function
  29 
  30 import os
  31 import time
  32 
  33 from absl import app
  34 from absl import flags
  35 import numpy as np
  36 import tensorflow as tf
  37 
  38 from google.protobuf import text_format
  39 from delf import delf_config_pb2
  40 from delf import datum_io
  41 from delf import feature_io
  42 from delf import utils
  43 from delf.python.detect_to_retrieve import dataset
  44 from delf import extractor
  45 
  46 FLAGS = flags.FLAGS
  47 
  48 flags.DEFINE_string(
  49     'delf_config_path', '/tmp/delf_config_example.pbtxt',
  50     'Path to DelfConfig proto text file with configuration to be used for DELG '
  51     'extraction. Local features are extracted if use_local_features is True; '
  52     'global features are extracted if use_global_features is True.')
  53 flags.DEFINE_string(
  54     'dataset_file_path', '/tmp/gnd_roxford5k.mat',
  55     'Dataset file for Revisited Oxford or Paris dataset, in .mat format.')
  56 flags.DEFINE_string(
  57     'images_dir', '/tmp/images',
  58     'Directory where dataset images are located, all in .jpg format.')
  59 flags.DEFINE_enum('image_set', 'query', ['query', 'index'],
  60                   'Whether to extract features from query or index images.')
  61 flags.DEFINE_string(
  62     'output_features_dir', '/tmp/features',
  63     "Directory where DELG features will be written to. Each image's features "
  64     'will be written to files with same name but different extension: the '
  65     'global feature is written to a file with extension .delg_global and the '
  66     'local features are written to a file with extension .delg_local.')
  67 
  68 # Extensions.
  69 _DELG_GLOBAL_EXTENSION = '.delg_global'
  70 _DELG_LOCAL_EXTENSION = '.delg_local'
  71 _IMAGE_EXTENSION = '.jpg'
  72 
  73 # Pace to report extraction log.
  74 _STATUS_CHECK_ITERATIONS = 50
  75 
  76 
  77 def main(argv):
  78   if len(argv) > 1:
  79     raise RuntimeError('Too many command-line arguments.')
  80 
  81   # Read list of images from dataset file.
  82   print('Reading list of images from dataset file...')
  83   query_list, index_list, ground_truth = dataset.ReadDatasetFile(
  84       FLAGS.dataset_file_path)
  85   if FLAGS.image_set == 'query':
  86     image_list = query_list
  87   else:
  88     image_list = index_list
  89   num_images = len(image_list)
  90   print('done! Found %d images' % num_images)
  91 
  92   # Parse DelfConfig proto.
  93   config = delf_config_pb2.DelfConfig()
  94   with tf.io.gfile.GFile(FLAGS.delf_config_path, 'r') as f:
  95     text_format.Parse(f.read(), config)
  96 
  97   # Create output directory if necessary.
  98   if not tf.io.gfile.exists(FLAGS.output_features_dir):
  99     tf.io.gfile.makedirs(FLAGS.output_features_dir)
 100 
 101   extractor_fn = extractor.MakeExtractor(config)
 102 
 103   start = time.time()
 104   for i in range(num_images):
 105     if i == 0:
 106       print('Starting to extract features...')
 107     elif i % _STATUS_CHECK_ITERATIONS == 0:
 108       elapsed = (time.time() - start)
 109       print('Processing image %d out of %d, last %d '
 110             'images took %f seconds' %
 111             (i, num_images, _STATUS_CHECK_ITERATIONS, elapsed))
 112       start = time.time()
 113 
 114     image_name = image_list[i]
 115     input_image_filename = os.path.join(FLAGS.images_dir,
 116                                         image_name + _IMAGE_EXTENSION)
 117 
 118     # Compose output file name and decide if image should be skipped.
 119     should_skip_global = True
 120     should_skip_local = True
 121     if config.use_global_features:
 122       output_global_feature_filename = os.path.join(
 123           FLAGS.output_features_dir, image_name + _DELG_GLOBAL_EXTENSION)
 124       if not tf.io.gfile.exists(output_global_feature_filename):
 125         should_skip_global = False
 126     if config.use_local_features:
 127       output_local_feature_filename = os.path.join(
 128           FLAGS.output_features_dir, image_name + _DELG_LOCAL_EXTENSION)
 129       if not tf.io.gfile.exists(output_local_feature_filename):
 130         should_skip_local = False
 131     if should_skip_global and should_skip_local:
 132       print('Skipping %s' % image_name)
 133       continue
 134 
 135     pil_im = utils.RgbLoader(input_image_filename)
 136     resize_factor = 1.0
 137     if FLAGS.image_set == 'query':
 138       # Crop query image according to bounding box.
 139       original_image_size = max(pil_im.size)
 140       bbox = [int(round(b)) for b in ground_truth[i]['bbx']]
 141       pil_im = pil_im.crop(bbox)
 142       cropped_image_size = max(pil_im.size)
 143       resize_factor = cropped_image_size / original_image_size
 144 
 145     im = np.array(pil_im)
 146 
 147     # Extract and save features.
 148     extracted_features = extractor_fn(im, resize_factor)
 149     if config.use_global_features:
 150       global_descriptor = extracted_features['global_descriptor']
 151       datum_io.WriteToFile(global_descriptor, output_global_feature_filename)
 152     if config.use_local_features:
 153       locations = extracted_features['local_features']['locations']
 154       descriptors = extracted_features['local_features']['descriptors']
 155       feature_scales = extracted_features['local_features']['scales']
 156       attention = extracted_features['local_features']['attention']
 157       feature_io.WriteToFile(output_local_feature_filename, locations,
 158                              feature_scales, descriptors, attention)
 159 
 160 
 161 if __name__ == '__main__':
 162   app.run(main)
