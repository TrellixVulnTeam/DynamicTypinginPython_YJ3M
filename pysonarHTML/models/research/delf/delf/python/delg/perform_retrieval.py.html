  24 from absl import app
  25 from absl import flags

  29 from delf import datum_io
  30 from delf.python.detect_to_retrieve import dataset
  31 from delf.python.detect_to_retrieve import image_reranking

  33 FLAGS = flags.FLAGS
  34 
  35 flags.DEFINE_string(
  36     'dataset_file_path', '/tmp/gnd_roxford5k.mat',
  37     'Dataset file for Revisited Oxford or Paris dataset, in .mat format.')
  38 flags.DEFINE_string('query_features_dir', '/tmp/features/query',
  39                     'Directory where query DELG features are located.')
  40 flags.DEFINE_string('index_features_dir', '/tmp/features/index',
  41                     'Directory where index DELG features are located.')
  42 flags.DEFINE_boolean(
  43     'use_geometric_verification', False,
  44     'If True, performs re-ranking using local feature-based geometric '
  45     'verification.')
  46 flags.DEFINE_float(
  47     'local_feature_distance_threshold', 1.0,
  48     'Optional, only used if `use_geometric_verification` is True. '
  49     'Distance threshold below which a pair of local descriptors is considered '
  50     'a potential match, and will be fed into RANSAC.')
  51 flags.DEFINE_float(
  52     'ransac_residual_threshold', 20.0,
  53     'Optional, only used if `use_geometric_verification` is True. '
  54     'Residual error threshold for considering matches as inliers, used in '
  55     'RANSAC algorithm.')
  56 flags.DEFINE_string(
  57     'output_dir', '/tmp/retrieval',
  58     'Directory where retrieval output will be written to. A file containing '
  59     "metrics for this run is saved therein, with file name 'metrics.txt'.")
  60 
  61 # Extensions.
  62 _DELG_GLOBAL_EXTENSION = '.delg_global'
  63 _DELG_LOCAL_EXTENSION = '.delg_local'
  64 
  65 # Precision-recall ranks to use in metric computation.
  66 _PR_RANKS = (1, 5, 10)
  67 
  68 # Pace to log.
  69 _STATUS_CHECK_LOAD_ITERATIONS = 50
  70 
  71 # Output file names.
  72 _METRICS_FILENAME = 'metrics.txt'
  73 
  74 
  75 def _ReadDelgGlobalDescriptors(input_dir, image_list):
  76   """Reads DELG global features.
  77 
  78   Args:
  79     input_dir: Directory where features are located.
  80     image_list: List of image names for which to load features.
  81 
  82   Returns:
  83     global_descriptors: NumPy array of shape (len(image_list), D), where D
  84       corresponds to the global descriptor dimensionality.
  85   """
  86   num_images = len(image_list)
  87   global_descriptors = []
  88   print('Starting to collect global descriptors for %d images...' % num_images)
  89   start = time.time()
  90   for i in range(num_images):
  91     if i > 0 and i % _STATUS_CHECK_LOAD_ITERATIONS == 0:
  92       elapsed = (time.time() - start)
  93       print('Reading global descriptors for image %d out of %d, last %d '
  94             'images took %f seconds' %
  95             (i, num_images, _STATUS_CHECK_LOAD_ITERATIONS, elapsed))
  96       start = time.time()
  97 
  98     descriptor_filename = image_list[i] + _DELG_GLOBAL_EXTENSION
  99     descriptor_fullpath = os.path.join(input_dir, descriptor_filename)
 100     global_descriptors.append(datum_io.ReadFromFile(descriptor_fullpath))
 101 
 102   return np.array(global_descriptors)

 105 def main(argv):
 106   if len(argv) > 1:
 107     raise RuntimeError('Too many command-line arguments.')
 108 
 109   # Parse dataset to obtain query/index images, and ground-truth.
 110   print('Parsing dataset...')
 111   query_list, index_list, ground_truth = dataset.ReadDatasetFile(
 112       FLAGS.dataset_file_path)
 113   num_query_images = len(query_list)
 114   num_index_images = len(index_list)
 115   (_, medium_ground_truth,
 116    hard_ground_truth) = dataset.ParseEasyMediumHardGroundTruth(ground_truth)
 117   print('done! Found %d queries and %d index images' %
 118         (num_query_images, num_index_images))
 119 
 120   # Read global features.
 121   query_global_features = _ReadDelgGlobalDescriptors(FLAGS.query_features_dir,
 122                                                      query_list)
 123   index_global_features = _ReadDelgGlobalDescriptors(FLAGS.index_features_dir,
 124                                                      index_list)
 125 
 126   # Compute similarity between query and index images, potentially re-ranking
 127   # with geometric verification.
 128   ranks_before_gv = np.zeros([num_query_images, num_index_images],
 129                              dtype='int32')
 130   if FLAGS.use_geometric_verification:
 131     medium_ranks_after_gv = np.zeros([num_query_images, num_index_images],
 132                                      dtype='int32')
 133     hard_ranks_after_gv = np.zeros([num_query_images, num_index_images],
 134                                    dtype='int32')
 135   for i in range(num_query_images):
 136     print('Performing retrieval with query %d (%s)...' % (i, query_list[i]))
 137     start = time.time()
 138 
 139     # Compute similarity between global descriptors.
 140     similarities = np.dot(index_global_features, query_global_features[i])
 141     ranks_before_gv[i] = np.argsort(-similarities)
 142 
 143     # Re-rank using geometric verification.
 144     if FLAGS.use_geometric_verification:
 145       medium_ranks_after_gv[i] = image_reranking.RerankByGeometricVerification(
 146           input_ranks=ranks_before_gv[i],
 147           initial_scores=similarities,
 148           query_name=query_list[i],
 149           index_names=index_list,
 150           query_features_dir=FLAGS.query_features_dir,
 151           index_features_dir=FLAGS.index_features_dir,
 152           junk_ids=set(medium_ground_truth[i]['junk']),
 153           local_feature_extension=_DELG_LOCAL_EXTENSION,
 154           ransac_seed=0,
 155           feature_distance_threshold=FLAGS.local_feature_distance_threshold,
 156           ransac_residual_threshold=FLAGS.ransac_residual_threshold)
 157       hard_ranks_after_gv[i] = image_reranking.RerankByGeometricVerification(
 158           input_ranks=ranks_before_gv[i],
 159           initial_scores=similarities,
 160           query_name=query_list[i],
 161           index_names=index_list,
 162           query_features_dir=FLAGS.query_features_dir,
 163           index_features_dir=FLAGS.index_features_dir,
 164           junk_ids=set(hard_ground_truth[i]['junk']),
 165           local_feature_extension=_DELG_LOCAL_EXTENSION,
 166           ransac_seed=0,
 167           feature_distance_threshold=FLAGS.local_feature_distance_threshold,
 168           ransac_residual_threshold=FLAGS.ransac_residual_threshold)
 169 
 170     elapsed = (time.time() - start)
 171     print('done! Retrieval for query %d took %f seconds' % (i, elapsed))
 172 
 173   # Create output directory if necessary.
 174   if not tf.io.gfile.exists(FLAGS.output_dir):
 175     tf.io.gfile.makedirs(FLAGS.output_dir)
 176 
 177   # Compute metrics.
 178   medium_metrics = dataset.ComputeMetrics(ranks_before_gv, medium_ground_truth,
 179                                           _PR_RANKS)
 180   hard_metrics = dataset.ComputeMetrics(ranks_before_gv, hard_ground_truth,
 181                                         _PR_RANKS)
 182   if FLAGS.use_geometric_verification:
 183     medium_metrics_after_gv = dataset.ComputeMetrics(medium_ranks_after_gv,
 184                                                      medium_ground_truth,
 185                                                      _PR_RANKS)
 186     hard_metrics_after_gv = dataset.ComputeMetrics(hard_ranks_after_gv,
 187                                                    hard_ground_truth, _PR_RANKS)
 188 
 189   # Write metrics to file.
 190   mean_average_precision_dict = {
 191       'medium': medium_metrics[0],
 192       'hard': hard_metrics[0]
 193   }
 194   mean_precisions_dict = {'medium': medium_metrics[1], 'hard': hard_metrics[1]}
 195   mean_recalls_dict = {'medium': medium_metrics[2], 'hard': hard_metrics[2]}
 196   if FLAGS.use_geometric_verification:
 197     mean_average_precision_dict.update({
 198         'medium_after_gv': medium_metrics_after_gv[0],
 199         'hard_after_gv': hard_metrics_after_gv[0]
 200     })
 201     mean_precisions_dict.update({
 202         'medium_after_gv': medium_metrics_after_gv[1],
 203         'hard_after_gv': hard_metrics_after_gv[1]
 204     })
 205     mean_recalls_dict.update({
 206         'medium_after_gv': medium_metrics_after_gv[2],
 207         'hard_after_gv': hard_metrics_after_gv[2]
 208     })
 209   dataset.SaveMetricsFile(mean_average_precision_dict, mean_precisions_dict,
 210                           mean_recalls_dict, _PR_RANKS,
 211                           os.path.join(FLAGS.output_dir, _METRICS_FILENAME))
 212 
 213 
 214 if __name__ == '__main__':
 215   app.run(main)
