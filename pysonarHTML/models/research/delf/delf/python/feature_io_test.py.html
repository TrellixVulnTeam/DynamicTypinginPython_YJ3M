  15 """Tests for feature_io, the python interface of DelfFeatures."""

  17 from __future__ import absolute_import
  18 from __future__ import division

  21 import os

  23 from absl import flags
  24 import numpy as np
  25 import tensorflow as tf
  26 
  27 from delf import feature_io
  28 
  29 FLAGS = flags.FLAGS
  30 
  31 
  32 def create_data():
  33   """Creates data to be used in tests.
  34 
  35   Returns:
  36     locations: [N, 2] float array which denotes the selected keypoint
  37       locations. N is the number of features.
  38     scales: [N] float array with feature scales.
  39     descriptors: [N, depth] float array with DELF descriptors.
  40     attention: [N] float array with attention scores.
  41     orientations: [N] float array with orientations.
  42   """
  43   locations = np.arange(8, dtype=np.float32).reshape(4, 2)
  44   scales = np.arange(4, dtype=np.float32)
  45   attention = np.arange(4, dtype=np.float32)
  46   orientations = np.arange(4, dtype=np.float32)
  47   descriptors = np.zeros([4, 1024])
  48   descriptors[0,] = np.arange(1024)
  49   descriptors[1,] = np.zeros([1024])
  50   descriptors[2,] = np.ones([1024])
  51   descriptors[3,] = -np.ones([1024])
  52 
  53   return locations, scales, descriptors, attention, orientations
  54 
  55 
  56 class DelfFeaturesIoTest(tf.test.TestCase):
  57 
  58   def testConversionAndBack(self):
  59     locations, scales, descriptors, attention, orientations = create_data()
  60 
  61     serialized = feature_io.SerializeToString(locations, scales, descriptors,
  62                                               attention, orientations)
  63     parsed_data = feature_io.ParseFromString(serialized)
  64 
  65     self.assertAllEqual(locations, parsed_data[0])
  66     self.assertAllEqual(scales, parsed_data[1])
  67     self.assertAllEqual(descriptors, parsed_data[2])
  68     self.assertAllEqual(attention, parsed_data[3])
  69     self.assertAllEqual(orientations, parsed_data[4])
  70 
  71   def testConversionAndBackNoOrientations(self):
  72     locations, scales, descriptors, attention, _ = create_data()
  73 
  74     serialized = feature_io.SerializeToString(locations, scales, descriptors,
  75                                               attention)
  76     parsed_data = feature_io.ParseFromString(serialized)
  77 
  78     self.assertAllEqual(locations, parsed_data[0])
  79     self.assertAllEqual(scales, parsed_data[1])
  80     self.assertAllEqual(descriptors, parsed_data[2])
  81     self.assertAllEqual(attention, parsed_data[3])
  82     self.assertAllEqual(np.zeros([4]), parsed_data[4])
  83 
  84   def testWriteAndReadToFile(self):
  85     locations, scales, descriptors, attention, orientations = create_data()
  86 
  87     filename = os.path.join(FLAGS.test_tmpdir, 'test.delf')
  88     feature_io.WriteToFile(filename, locations, scales, descriptors, attention,
  89                            orientations)
  90     data_read = feature_io.ReadFromFile(filename)
  91 
  92     self.assertAllEqual(locations, data_read[0])
  93     self.assertAllEqual(scales, data_read[1])
  94     self.assertAllEqual(descriptors, data_read[2])
  95     self.assertAllEqual(attention, data_read[3])
  96     self.assertAllEqual(orientations, data_read[4])
  97 
  98   def testWriteAndReadToFileEmptyFile(self):
  99     filename = os.path.join(FLAGS.test_tmpdir, 'test.delf')
 100     feature_io.WriteToFile(filename, np.array([]), np.array([]), np.array([]),
 101                            np.array([]), np.array([]))
 102     data_read = feature_io.ReadFromFile(filename)
 103 
 104     self.assertAllEqual(np.array([]), data_read[0])
 105     self.assertAllEqual(np.array([]), data_read[1])
 106     self.assertAllEqual(np.array([]), data_read[2])
 107     self.assertAllEqual(np.array([]), data_read[3])
 108     self.assertAllEqual(np.array([]), data_read[4])
