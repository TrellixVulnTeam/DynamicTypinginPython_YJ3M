  34   def testParseEasyMediumHardGroundTruth(self):
  35     # Define input.
  36     ground_truth = [{
  37         'easy': np.array([10, 56, 100]),
  38         'hard': np.array([0]),
  39         'junk': np.array([6, 90])
  40     }, {
  41         'easy': np.array([], dtype='int64'),
  42         'hard': [5],
  43         'junk': [99, 100]
  44     }, {
  45         'easy': [33],
  46         'hard': [66, 99],
  47         'junk': np.array([], dtype='int64')
  48     }]
  49 
  50     # Run tested function.
  51     (easy_ground_truth, medium_ground_truth,
  52      hard_ground_truth) = dataset.ParseEasyMediumHardGroundTruth(ground_truth)
  53 
  54     # Define expected outputs.
  55     expected_easy_ground_truth = [{
  56         'ok': np.array([10, 56, 100]),
  57         'junk': np.array([6, 90, 0])
  58     }, {
  59         'ok': np.array([], dtype='int64'),
  60         'junk': np.array([99, 100, 5])
  61     }, {
  62         'ok': np.array([33]),
  63         'junk': np.array([66, 99])
  64     }]
  65     expected_medium_ground_truth = [{
  66         'ok': np.array([10, 56, 100, 0]),
  67         'junk': np.array([6, 90])
  68     }, {
  69         'ok': np.array([5]),
  70         'junk': np.array([99, 100])
  71     }, {
  72         'ok': np.array([33, 66, 99]),
  73         'junk': np.array([], dtype='int64')
  74     }]
  75     expected_hard_ground_truth = [{
  76         'ok': np.array([0]),
  77         'junk': np.array([6, 90, 10, 56, 100])
  78     }, {
  79         'ok': np.array([5]),
  80         'junk': np.array([99, 100])
  81     }, {
  82         'ok': np.array([66, 99]),
  83         'junk': np.array([33])
  84     }]
  85 
  86     # Compare actual versus expected.
  87     def _AssertListOfDictsOfArraysAreEqual(ground_truth, expected_ground_truth):
  88       """Helper function to compare ground-truth data.
  89 
  90       Args:
  91         ground_truth: List of dicts of arrays.
  92         expected_ground_truth: List of dicts of arrays.
  93       """
  94       self.assertEqual(len(ground_truth), len(expected_ground_truth))
  95 
  96       for i, ground_truth_entry in enumerate(ground_truth):
  97         self.assertEqual(sorted(ground_truth_entry.keys()), ['junk', 'ok'])
  98         self.assertAllEqual(ground_truth_entry['junk'],
  99                             expected_ground_truth[i]['junk'])
 100         self.assertAllEqual(ground_truth_entry['ok'],
 101                             expected_ground_truth[i]['ok'])
 102 
 103     _AssertListOfDictsOfArraysAreEqual(easy_ground_truth,
 104                                        expected_easy_ground_truth)
 105     _AssertListOfDictsOfArraysAreEqual(medium_ground_truth,
 106                                        expected_medium_ground_truth)
 107     _AssertListOfDictsOfArraysAreEqual(hard_ground_truth,
 108                                        expected_hard_ground_truth)

 156   def testComputeMetricsWorks(self):
 157     # Define inputs: 3 queries. For the last one, there are no expected images
 158     # to be retrieved
 159     sorted_index_ids = np.array([[4, 2, 0, 1, 3], [0, 2, 4, 1, 3],
 160                                  [0, 1, 2, 3, 4]])
 161     ground_truth = [{
 162         'ok': np.array([0, 1]),
 163         'junk': np.array([2])
 164     }, {
 165         'ok': np.array([0, 4]),
 166         'junk': np.array([], dtype='int64')
 167     }, {
 168         'ok': np.array([], dtype='int64'),
 169         'junk': np.array([], dtype='int64')
 170     }]
 171     desired_pr_ranks = [1, 2, 5]
 172 
 173     # Run tested function.
 174     (mean_average_precision, mean_precisions, mean_recalls, average_precisions,
 175      precisions, recalls) = dataset.ComputeMetrics(sorted_index_ids,
 176                                                    ground_truth,
 177                                                    desired_pr_ranks)
 178 
 179     # Define expected outputs.
 180     expected_mean_average_precision = 0.604167
 181     expected_mean_precisions = [0.5, 0.5, 0.666667]
 182     expected_mean_recalls = [0.25, 0.5, 1.0]
 183     expected_average_precisions = [0.416667, 0.791667, float('nan')]
 184     expected_precisions = [[0.0, 0.5, 0.666667], [1.0, 0.5, 0.666667],
 185                            [float('nan'),
 186                             float('nan'),
 187                             float('nan')]]
 188     expected_recalls = [[0.0, 0.5, 1.0], [0.5, 0.5, 1.0],
 189                         [float('nan'), float('nan'),
 190                          float('nan')]]
 191 
 192     # Compare actual versus expected.
 193     self.assertAllClose(mean_average_precision, expected_mean_average_precision)
 194     self.assertAllClose(mean_precisions, expected_mean_precisions)
 195     self.assertAllClose(mean_recalls, expected_mean_recalls)
 196     self.assertAllClose(average_precisions, expected_average_precisions)
 197     self.assertAllClose(precisions, expected_precisions)
 198     self.assertAllClose(recalls, expected_recalls)

 200   def testSaveMetricsFileWorks(self):
 201     # Define inputs.
 202     mean_average_precision = {'hard': 0.7, 'medium': 0.9}
 203     mean_precisions = {
 204         'hard': np.array([1.0, 0.8]),
 205         'medium': np.array([1.0, 1.0])
 206     }
 207     mean_recalls = {
 208         'hard': np.array([0.5, 0.8]),
 209         'medium': np.array([0.5, 1.0])
 210     }
 211     pr_ranks = [1, 5]
 212     output_path = os.path.join(FLAGS.test_tmpdir, 'metrics.txt')
 213 
 214     # Run tested function.
 215     dataset.SaveMetricsFile(mean_average_precision, mean_precisions,
 216                             mean_recalls, pr_ranks, output_path)
 217 
 218     # Define expected results.
 219     expected_metrics = ('hard\n'
 220                         '  mAP=70.0\n'
 221                         '  mP@k[1 5] [100.  80.]\n'
 222                         '  mR@k[1 5] [50. 80.]\n'
 223                         'medium\n'
 224                         '  mAP=90.0\n'
 225                         '  mP@k[1 5] [100. 100.]\n'
 226                         '  mR@k[1 5] [ 50. 100.]\n')
 227 
 228     # Parse actual results, and compare to expected.
 229     with tf.io.gfile.GFile(output_path) as f:
 230       metrics = f.read()
 231 
 232     self.assertEqual(metrics, expected_metrics)

 234   def testSaveAndReadMetricsWorks(self):
 235     # Define inputs.
 236     mean_average_precision = {'hard': 0.7, 'medium': 0.9}
 237     mean_precisions = {
 238         'hard': np.array([1.0, 0.8]),
 239         'medium': np.array([1.0, 1.0])
 240     }
 241     mean_recalls = {
 242         'hard': np.array([0.5, 0.8]),
 243         'medium': np.array([0.5, 1.0])
 244     }
 245     pr_ranks = [1, 5]
 246     output_path = os.path.join(FLAGS.test_tmpdir, 'metrics.txt')
 247 
 248     # Run tested functions.
 249     dataset.SaveMetricsFile(mean_average_precision, mean_precisions,
 250                             mean_recalls, pr_ranks, output_path)
 251     (read_mean_average_precision, read_pr_ranks, read_mean_precisions,
 252      read_mean_recalls) = dataset.ReadMetricsFile(output_path)
 253 
 254     # Compares actual and expected metrics.
 255     self.assertEqual(read_mean_average_precision, mean_average_precision)
 256     self.assertEqual(read_pr_ranks, pr_ranks)
 257     self.assertEqual(read_mean_precisions.keys(), mean_precisions.keys())
 258     self.assertAllEqual(read_mean_precisions['hard'], mean_precisions['hard'])
 259     self.assertAllEqual(read_mean_precisions['medium'],
 260                         mean_precisions['medium'])
 261     self.assertEqual(read_mean_recalls.keys(), mean_recalls.keys())
 262     self.assertAllEqual(read_mean_recalls['hard'], mean_recalls['hard'])
 263     self.assertAllEqual(read_mean_recalls['medium'], mean_recalls['medium'])

 272               '  mR@k[1 5] [ 50.  80.]\n'
 273               'medium\n'
 274               '  mAP=90.0\n'
 275               '  mP@k[1 5] [ 100.  100.]\n'
 276               '  mR@k[1 5] [  50.  100.]\n'
 277               'medium\n'
 278               '  mAP=90.0\n'
 279               '  mP@k[1 5] [ 100.  100.]\n'
 280               '  mR@k[1 5] [  50.  100.]\n')
