  37 # Extension of feature files.
  38 _BOX_EXTENSION = '.boxes'
  39 _DELF_EXTENSION = '.delf'

  41 # Pace to report extraction log.
  42 _STATUS_CHECK_ITERATIONS = 100

  45 def _WriteMappingBasenameToIds(index_names_ids_and_boxes, output_path):
  46   """Helper function to write CSV mapping from DELF file name to IDs.
  47 
  48   Args:
  49     index_names_ids_and_boxes: List containing 3-element lists with name, image
  50       ID and box ID.
  51     output_path: Output CSV path.
  52   """
  53   with tf.io.gfile.GFile(output_path, 'w') as f:
  54     csv_writer = csv.DictWriter(
  55         f, fieldnames=['name', 'index_image_id', 'box_id'])
  56     csv_writer.writeheader()
  57     for name_imid_boxid in index_names_ids_and_boxes:
  58       csv_writer.writerow({
  59           'name': name_imid_boxid[0],
  60           'index_image_id': name_imid_boxid[1],
  61           'box_id': name_imid_boxid[2],
  62       })

  65 def ExtractBoxesAndFeaturesToFiles(image_names, image_paths, delf_config_path,
  66                                    detector_model_dir, detector_thresh,
  67                                    output_features_dir, output_boxes_dir,
  68                                    output_mapping):
  69   """Extracts boxes and features, saving them to files.
  70 
  71   Boxes are saved to <image_name>.boxes files. DELF features are extracted for
  72   the entire image and saved into <image_name>.delf files. In addition, DELF
  73   features are extracted for each high-confidence bounding box in the image, and
  74   saved into files named <image_name>_0.delf, <image_name>_1.delf, etc.
  75 
  76   It checks if descriptors/boxes already exist, and skips computation for those.
  77 
  78   Args:
  79     image_names: List of image names. These are used to compose output file
  80       names for boxes and features.
  81     image_paths: List of image paths. image_paths[i] is the path for the image
  82       named by image_names[i]. `image_names` and `image_paths` must have the
  83       same number of elements.
  84     delf_config_path: Path to DelfConfig proto text file.
  85     detector_model_dir: Directory where detector SavedModel is located.
  86     detector_thresh: Threshold used to decide if an image's detected box
  87       undergoes feature extraction.
  88     output_features_dir: Directory where DELF features will be written to.
  89     output_boxes_dir: Directory where detected boxes will be written to.
  90     output_mapping: CSV file which maps each .delf file name to the image ID and
  91       detected box ID.
  92 
  93   Raises:
  94     ValueError: If len(image_names) and len(image_paths) are different.
  95   """
  96   num_images = len(image_names)
  97   if len(image_paths) != num_images:
  98     raise ValueError(
  99         'image_names and image_paths have different number of items')
 100 
 101   # Parse DelfConfig proto.
 102   config = delf_config_pb2.DelfConfig()
 103   with tf.io.gfile.GFile(delf_config_path, 'r') as f:
 104     text_format.Merge(f.read(), config)
 105 
 106   # Create output directories if necessary.
 107   if not tf.io.gfile.exists(output_features_dir):
 108     tf.io.gfile.makedirs(output_features_dir)
 109   if not tf.io.gfile.exists(output_boxes_dir):
 110     tf.io.gfile.makedirs(output_boxes_dir)
 111   if not tf.io.gfile.exists(os.path.dirname(output_mapping)):
 112     tf.io.gfile.makedirs(os.path.dirname(output_mapping))
 113 
 114   names_ids_and_boxes = []
 115   detector_fn = detector.MakeDetector(detector_model_dir)
 116   delf_extractor_fn = extractor.MakeExtractor(config)
 117 
 118   start = time.time()
 119   for i in range(num_images):
 120     if i == 0:
 121       print('Starting to extract features/boxes...')
 122     elif i % _STATUS_CHECK_ITERATIONS == 0:
 123       elapsed = (time.time() - start)
 124       print('Processing image %d out of %d, last %d '
 125             'images took %f seconds' %
 126             (i, num_images, _STATUS_CHECK_ITERATIONS, elapsed))
 127       start = time.time()
 128 
 129     image_name = image_names[i]
 130     output_feature_filename_whole_image = os.path.join(
 131         output_features_dir, image_name + _DELF_EXTENSION)
 132     output_box_filename = os.path.join(output_boxes_dir,
 133                                        image_name + _BOX_EXTENSION)
 134 
 135     pil_im = utils.RgbLoader(image_paths[i])
 136     width, height = pil_im.size
 137 
 138     # Extract and save boxes.
 139     if tf.io.gfile.exists(output_box_filename):
 140       print('Skipping box computation for %s' % image_name)
 141       (boxes_out, scores_out,
 142        class_indices_out) = box_io.ReadFromFile(output_box_filename)
 143     else:
 144       (boxes_out, scores_out,
 145        class_indices_out) = detector_fn(np.expand_dims(pil_im, 0))
 146       # Using only one image per batch.
 147       boxes_out = boxes_out[0]
 148       scores_out = scores_out[0]
 149       class_indices_out = class_indices_out[0]
 150       box_io.WriteToFile(output_box_filename, boxes_out, scores_out,
 151                          class_indices_out)
 152 
 153     # Select boxes with scores greater than threshold. Those will be the
 154     # ones with extracted DELF features (besides the whole image, whose DELF
 155     # features are extracted in all cases).
 156     num_delf_files = 1
 157     selected_boxes = []
 158     for box_ind, box in enumerate(boxes_out):
 159       if scores_out[box_ind] >= detector_thresh:
 160         selected_boxes.append(box)
 161     num_delf_files += len(selected_boxes)
 162 
 163     # Extract and save DELF features.
 164     for delf_file_ind in range(num_delf_files):
 165       if delf_file_ind == 0:
 166         box_name = image_name
 167         output_feature_filename = output_feature_filename_whole_image
 168       else:
 169         box_name = image_name + '_' + str(delf_file_ind - 1)
 170         output_feature_filename = os.path.join(output_features_dir,
 171                                                box_name + _DELF_EXTENSION)
 172 
 173       names_ids_and_boxes.append([box_name, i, delf_file_ind - 1])
 174 
 175       if tf.io.gfile.exists(output_feature_filename):
 176         print('Skipping DELF computation for %s' % box_name)
 177         continue
 178 
 179       if delf_file_ind >= 1:
 180         bbox_for_cropping = selected_boxes[delf_file_ind - 1]
 181         bbox_for_cropping_pil_convention = [
 182             int(math.floor(bbox_for_cropping[1] * width)),
 183             int(math.floor(bbox_for_cropping[0] * height)),
 184             int(math.ceil(bbox_for_cropping[3] * width)),
 185             int(math.ceil(bbox_for_cropping[2] * height))
 186         ]
 187         pil_cropped_im = pil_im.crop(bbox_for_cropping_pil_convention)
 188         im = np.array(pil_cropped_im)
 189       else:
 190         im = np.array(pil_im)
 191 
 192       extracted_features = delf_extractor_fn(im)
 193       locations_out = extracted_features['local_features']['locations']
 194       descriptors_out = extracted_features['local_features']['descriptors']
 195       feature_scales_out = extracted_features['local_features']['scales']
 196       attention_out = extracted_features['local_features']['attention']
 197 
 198       feature_io.WriteToFile(output_feature_filename, locations_out,
 199                              feature_scales_out, descriptors_out, attention_out)
 200 
 201   # Save mapping from output DELF name to image id and box id.
 202   _WriteMappingBasenameToIds(names_ids_and_boxes, output_mapping)
