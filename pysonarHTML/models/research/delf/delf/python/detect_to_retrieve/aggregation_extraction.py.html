  34 # Aliases for aggregation types.
  35 _VLAD = aggregation_config_pb2.AggregationConfig.VLAD
  36 _ASMK = aggregation_config_pb2.AggregationConfig.ASMK
  37 _ASMK_STAR = aggregation_config_pb2.AggregationConfig.ASMK_STAR
  38 
  39 # Extensions.
  40 _DELF_EXTENSION = '.delf'
  41 _VLAD_EXTENSION_SUFFIX = 'vlad'
  42 _ASMK_EXTENSION_SUFFIX = 'asmk'
  43 _ASMK_STAR_EXTENSION_SUFFIX = 'asmk_star'
  44 
  45 # Pace to report extraction log.
  46 _STATUS_CHECK_ITERATIONS = 50

  49 def _ReadMappingBasenameToBoxNames(input_path, index_image_names):
  50   """Reads mapping from image name to DELF file names for each box.
  51 
  52   Args:
  53     input_path: Path to CSV file containing mapping.
  54     index_image_names: List containing index image names, in order, for the
  55       dataset under consideration.
  56 
  57   Returns:
  58     images_to_box_feature_files: Dict. key=string (image name); value=list of
  59       strings (file names containing DELF features for boxes).
  60   """
  61   images_to_box_feature_files = {}
  62   with tf.io.gfile.GFile(input_path, 'r') as f:
  63     reader = csv.DictReader(f)
  64     for row in reader:
  65       index_image_name = index_image_names[int(row['index_image_id'])]
  66       if index_image_name not in images_to_box_feature_files:
  67         images_to_box_feature_files[index_image_name] = []
  68 
  69       images_to_box_feature_files[index_image_name].append(row['name'])
  70 
  71   return images_to_box_feature_files

  74 def ExtractAggregatedRepresentationsToFiles(image_names, features_dir,
  75                                             aggregation_config_path,
  76                                             mapping_path,
  77                                             output_aggregation_dir):
  78   """Extracts aggregated feature representations, saving them to files.
  79 
  80   It checks if the aggregated representation for an image already exists,
  81   and skips computation for those.
  82 
  83   Args:
  84     image_names: List of image names. These are used to compose input file names
  85       for the feature files, and the output file names for aggregated
  86       representations.
  87     features_dir: Directory where DELF features are located.
  88     aggregation_config_path: Path to AggregationConfig proto text file with
  89       configuration to be used for extraction.
  90     mapping_path: Optional CSV file which maps each .delf file name to the index
  91       image ID and detected box ID. If regional aggregation is performed, this
  92       should be set. Otherwise, this is ignored.
  93     output_aggregation_dir: Directory where aggregation output will be written
  94       to.
  95 
  96   Raises:
  97     ValueError: If AggregationConfig is malformed, or `mapping_path` is
  98       missing.
  99   """
 100   num_images = len(image_names)
 101 
 102   # Parse AggregationConfig proto, and select output extension.
 103   config = aggregation_config_pb2.AggregationConfig()
 104   with tf.io.gfile.GFile(aggregation_config_path, 'r') as f:
 105     text_format.Merge(f.read(), config)
 106   output_extension = '.'
 107   if config.use_regional_aggregation:
 108     output_extension += 'r'
 109   if config.aggregation_type == _VLAD:
 110     output_extension += _VLAD_EXTENSION_SUFFIX
 111   elif config.aggregation_type == _ASMK:
 112     output_extension += _ASMK_EXTENSION_SUFFIX
 113   elif config.aggregation_type == _ASMK_STAR:
 114     output_extension += _ASMK_STAR_EXTENSION_SUFFIX
 115   else:
 116     raise ValueError('Invalid aggregation type: %d' % config.aggregation_type)
 117 
 118   # Read index mapping path, if provided.
 119   if mapping_path:
 120     images_to_box_feature_files = _ReadMappingBasenameToBoxNames(
 121         mapping_path, image_names)
 122 
 123   # Create output directory if necessary.
 124   if not tf.io.gfile.exists(output_aggregation_dir):
 125     tf.io.gfile.makedirs(output_aggregation_dir)
 126 
 127   extractor = feature_aggregation_extractor.ExtractAggregatedRepresentation(
 128       config)
 129 
 130   start = time.time()
 131   for i in range(num_images):
 132     if i == 0:
 133       print('Starting to extract aggregation from images...')
 134     elif i % _STATUS_CHECK_ITERATIONS == 0:
 135       elapsed = (time.time() - start)
 136       print('Processing image %d out of %d, last %d '
 137             'images took %f seconds' %
 138             (i, num_images, _STATUS_CHECK_ITERATIONS, elapsed))
 139       start = time.time()
 140 
 141     image_name = image_names[i]
 142 
 143     # Compose output file name, skip extraction for this image if it already
 144     # exists.
 145     output_aggregation_filename = os.path.join(output_aggregation_dir,
 146                                                image_name + output_extension)
 147     if tf.io.gfile.exists(output_aggregation_filename):
 148       print('Skipping %s' % image_name)
 149       continue
 150 
 151     # Load DELF features.
 152     if config.use_regional_aggregation:
 153       if not mapping_path:
 154         raise ValueError(
 155             'Requested regional aggregation, but mapping_path was not '
 156             'provided')
 157       descriptors_list = []
 158       num_features_per_box = []
 159       for box_feature_file in images_to_box_feature_files[image_name]:
 160         delf_filename = os.path.join(features_dir,
 161                                      box_feature_file + _DELF_EXTENSION)
 162         _, _, box_descriptors, _, _ = feature_io.ReadFromFile(delf_filename)
 163         # If `box_descriptors` is empty, reshape it such that it can be
 164         # concatenated with other descriptors.
 165         if not box_descriptors.shape[0]:
 166           box_descriptors = np.reshape(box_descriptors,
 167                                        [0, config.feature_dimensionality])
 168         descriptors_list.append(box_descriptors)
 169         num_features_per_box.append(box_descriptors.shape[0])
 170 
 171       descriptors = np.concatenate(descriptors_list)
 172     else:
 173       input_delf_filename = os.path.join(features_dir,
 174                                          image_name + _DELF_EXTENSION)
 175       _, _, descriptors, _, _ = feature_io.ReadFromFile(input_delf_filename)
 176       # If `descriptors` is empty, reshape it to avoid extraction failure.
 177       if not descriptors.shape[0]:
 178         descriptors = np.reshape(descriptors,
 179                                  [0, config.feature_dimensionality])
 180       num_features_per_box = None
 181 
 182     # Extract and save aggregation. If using VLAD, only
 183     # `aggregated_descriptors` needs to be saved.
 184     (aggregated_descriptors,
 185      feature_visual_words) = extractor.Extract(descriptors,
 186                                                num_features_per_box)
 187     if config.aggregation_type == _VLAD:
 188       datum_io.WriteToFile(aggregated_descriptors,
 189                            output_aggregation_filename)
 190     else:
 191       datum_io.WritePairToFile(aggregated_descriptors,
 192                                feature_visual_words.astype('uint32'),
 193                                output_aggregation_filename)
