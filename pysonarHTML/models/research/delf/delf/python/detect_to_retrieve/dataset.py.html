  28 def ReadDatasetFile(dataset_file_path):
  29   """Reads dataset file in Revisited Oxford/Paris ".mat" format.
  30 
  31   Args:
  32     dataset_file_path: Path to dataset file, in .mat format.
  33 
  34   Returns:
  35     query_list: List of query image names.
  36     index_list: List of index image names.
  37     ground_truth: List containing ground-truth information for dataset. Each
  38       entry is a dict corresponding to the ground-truth information for a query.
  39       The dict may have keys 'easy', 'hard', or 'junk', mapping to a NumPy
  40       array of integers; additionally, it has a key 'bbx' mapping to a NumPy
  41       array of floats with bounding box coordinates.
  42   """
  43   with tf.io.gfile.GFile(dataset_file_path, 'rb') as f:
  44     cfg = matlab.loadmat(f)
  45 
  46   # Parse outputs according to the specificities of the dataset file.
  47   query_list = [str(im_array[0]) for im_array in np.squeeze(cfg['qimlist'])]
  48   index_list = [str(im_array[0]) for im_array in np.squeeze(cfg['imlist'])]
  49   ground_truth_raw = np.squeeze(cfg['gnd'])
  50   ground_truth = []
  51   for query_ground_truth_raw in ground_truth_raw:
  52     query_ground_truth = {}
  53     for ground_truth_key in _GROUND_TRUTH_KEYS:
  54       if ground_truth_key in query_ground_truth_raw.dtype.names:
  55         adjusted_labels = query_ground_truth_raw[ground_truth_key] - 1
  56         query_ground_truth[ground_truth_key] = adjusted_labels.flatten()
  57 
  58     query_ground_truth['bbx'] = np.squeeze(query_ground_truth_raw['bbx'])
  59     ground_truth.append(query_ground_truth)
  60 
  61   return query_list, index_list, ground_truth

  64 def _ParseGroundTruth(ok_list, junk_list):
  65   """Constructs dictionary of ok/junk indices for a data subset and query.
  66 
  67   Args:
  68     ok_list: List of NumPy arrays containing true positive indices for query.
  69     junk_list: List of NumPy arrays containing ignored indices for query.
  70 
  71   Returns:
  72     ok_junk_dict: Dict mapping 'ok' and 'junk' strings to NumPy array of
  73       indices.
  74   """
  75   ok_junk_dict = {}
  76   ok_junk_dict['ok'] = np.concatenate(ok_list)
  77   ok_junk_dict['junk'] = np.concatenate(junk_list)
  78   return ok_junk_dict

  81 def ParseEasyMediumHardGroundTruth(ground_truth):
  82   """Parses easy/medium/hard ground-truth from Revisited datasets.
  83 
  84   Args:
  85     ground_truth: Usually the output from ReadDatasetFile(). List containing
  86       ground-truth information for dataset. Each entry is a dict corresponding
  87       to the ground-truth information for a query. The dict must have keys
  88       'easy', 'hard', and 'junk', mapping to a NumPy array of integers.
  89 
  90   Returns:
  91     easy_ground_truth: List containing ground-truth information for easy subset
  92       of dataset. Each entry is a dict corresponding to the ground-truth
  93       information for a query. The dict has keys 'ok' and 'junk', mapping to a
  94       NumPy array of integers.
  95     medium_ground_truth: Same as `easy_ground_truth`, but for the medium subset.
  96     hard_ground_truth: Same as `easy_ground_truth`, but for the hard subset.
  97   """
  98   num_queries = len(ground_truth)
  99 
 100   easy_ground_truth = []
 101   medium_ground_truth = []
 102   hard_ground_truth = []
 103   for i in range(num_queries):
 104     easy_ground_truth.append(
 105         _ParseGroundTruth([ground_truth[i]['easy']],
 106                           [ground_truth[i]['junk'], ground_truth[i]['hard']]))
 107     medium_ground_truth.append(
 108         _ParseGroundTruth([ground_truth[i]['easy'], ground_truth[i]['hard']],
 109                           [ground_truth[i]['junk']]))
 110     hard_ground_truth.append(
 111         _ParseGroundTruth([ground_truth[i]['hard']],
 112                           [ground_truth[i]['junk'], ground_truth[i]['easy']]))
 113 
 114   return easy_ground_truth, medium_ground_truth, hard_ground_truth

 117 def AdjustPositiveRanks(positive_ranks, junk_ranks):
 118   """Adjusts positive ranks based on junk ranks.
 119 
 120   Args:
 121     positive_ranks: Sorted 1D NumPy integer array.
 122     junk_ranks: Sorted 1D NumPy integer array.
 123 
 124   Returns:
 125     adjusted_positive_ranks: Sorted 1D NumPy array.
 126   """
 127   if not junk_ranks.size:
 128     return positive_ranks
 129 
 130   adjusted_positive_ranks = positive_ranks
 131   j = 0
 132   for i, positive_index in enumerate(positive_ranks):
 133     while (j < len(junk_ranks) and positive_index > junk_ranks[j]):
 134       j += 1
 135 
 136     adjusted_positive_ranks[i] -= j
 137 
 138   return adjusted_positive_ranks

 141 def ComputeAveragePrecision(positive_ranks):
 142   """Computes average precision according to dataset convention.
 143 
 144   It assumes that `positive_ranks` contains the ranks for all expected positive
 145   index images to be retrieved. If `positive_ranks` is empty, returns
 146   `average_precision` = 0.
 147 
 148   Note that average precision computation here does NOT use the finite sum
 149   method (see
 150   https://en.wikipedia.org/wiki/Evaluation_measures_(information_retrieval)#Average_precision)
 151   which is common in information retrieval literature. Instead, the method
 152   implemented here integrates over the precision-recall curve by averaging two
 153   adjacent precision points, then multiplying by the recall step. This is the
 154   convention for the Revisited Oxford/Paris datasets.
 155 
 156   Args:
 157     positive_ranks: Sorted 1D NumPy integer array, zero-indexed.
 158 
 159   Returns:
 160     average_precision: Float.
 161   """
 162   average_precision = 0.0
 163 
 164   num_expected_positives = len(positive_ranks)
 165   if not num_expected_positives:
 166     return average_precision
 167 
 168   recall_step = 1.0 / num_expected_positives
 169   for i, rank in enumerate(positive_ranks):
 170     if not rank:
 171       left_precision = 1.0
 172     else:
 173       left_precision = i / rank
 174 
 175     right_precision = (i + 1) / (rank + 1)
 176     average_precision += (left_precision + right_precision) * recall_step / 2
 177 
 178   return average_precision

 181 def ComputePRAtRanks(positive_ranks, desired_pr_ranks):
 182   """Computes precision/recall at desired ranks.
 183 
 184   It assumes that `positive_ranks` contains the ranks for all expected positive
 185   index images to be retrieved. If `positive_ranks` is empty, return all-zeros
 186   `precisions`/`recalls`.
 187 
 188   If a desired rank is larger than the last positive rank, its precision is
 189   computed based on the last positive rank. For example, if `desired_pr_ranks`
 190   is [10] and `positive_ranks` = [0, 7] --> `precisions` = [0.25], `recalls` =
 191   [1.0].
 192 
 193   Args:
 194     positive_ranks: 1D NumPy integer array, zero-indexed.
 195     desired_pr_ranks: List of integers containing the desired precision/recall
 196       ranks to be reported. Eg, if precision@1/recall@1 and
 197       precision@10/recall@10 are desired, this should be set to [1, 10].
 198 
 199   Returns:
 200     precisions: Precision @ `desired_pr_ranks` (NumPy array of
 201       floats, with shape [len(desired_pr_ranks)]).
 202     recalls: Recall @ `desired_pr_ranks` (NumPy array of floats, with
 203       shape [len(desired_pr_ranks)]).
 204   """
 205   num_desired_pr_ranks = len(desired_pr_ranks)
 206   precisions = np.zeros([num_desired_pr_ranks])
 207   recalls = np.zeros([num_desired_pr_ranks])
 208 
 209   num_expected_positives = len(positive_ranks)
 210   if not num_expected_positives:
 211     return precisions, recalls
 212 
 213   positive_ranks_one_indexed = positive_ranks + 1
 214   for i, desired_pr_rank in enumerate(desired_pr_ranks):
 215     recalls[i] = np.sum(
 216         positive_ranks_one_indexed <= desired_pr_rank) / num_expected_positives
 217 
 218     # If `desired_pr_rank` is larger than last positive's rank, only compute
 219     # precision with respect to last positive's position.
 220     precision_rank = min(max(positive_ranks_one_indexed), desired_pr_rank)
 221     precisions[i] = np.sum(
 222         positive_ranks_one_indexed <= precision_rank) / precision_rank
 223 
 224   return precisions, recalls

 227 def ComputeMetrics(sorted_index_ids, ground_truth, desired_pr_ranks):
 228   """Computes metrics for retrieval results on the Revisited datasets.
 229 
 230   If there are no valid ground-truth index images for a given query, the metric
 231   results for the given query (`average_precisions`, `precisions` and `recalls`)
 232   are set to NaN, and they are not taken into account when computing the
 233   aggregated metrics (`mean_average_precision`, `mean_precisions` and
 234   `mean_recalls`) over all queries.
 235 
 236   Args:
 237     sorted_index_ids: Integer NumPy array of shape [#queries, #index_images].
 238       For each query, contains an array denoting the most relevant index images,
 239       sorted from most to least relevant.
 240     ground_truth: List containing ground-truth information for dataset. Each
 241       entry is a dict corresponding to the ground-truth information for a query.
 242       The dict has keys 'ok' and 'junk', mapping to a NumPy array of integers.
 243     desired_pr_ranks: List of integers containing the desired precision/recall
 244       ranks to be reported. Eg, if precision@1/recall@1 and
 245       precision@10/recall@10 are desired, this should be set to [1, 10]. The
 246       largest item should be <= #index_images.
 247 
 248   Returns:
 249     mean_average_precision: Mean average precision (float).
 250     mean_precisions: Mean precision @ `desired_pr_ranks` (NumPy array of
 251       floats, with shape [len(desired_pr_ranks)]).
 252     mean_recalls: Mean recall @ `desired_pr_ranks` (NumPy array of floats, with
 253       shape [len(desired_pr_ranks)]).
 254     average_precisions: Average precision for each query (NumPy array of floats,
 255       with shape [#queries]).
 256     precisions: Precision @ `desired_pr_ranks`, for each query (NumPy array of
 257       floats, with shape [#queries, len(desired_pr_ranks)]).
 258     recalls: Recall @ `desired_pr_ranks`, for each query (NumPy array of
 259       floats, with shape [#queries, len(desired_pr_ranks)]).
 260 
 261   Raises:
 262     ValueError: If largest desired PR rank in `desired_pr_ranks` >
 263       #index_images.
 264   """
 265   num_queries, num_index_images = sorted_index_ids.shape
 266   num_desired_pr_ranks = len(desired_pr_ranks)
 267 
 268   sorted_desired_pr_ranks = sorted(desired_pr_ranks)
 269 
 270   if sorted_desired_pr_ranks[-1] > num_index_images:
 271     raise ValueError(
 272         'Requested PR ranks up to %d, however there are only %d images' %
 273         (sorted_desired_pr_ranks[-1], num_index_images))
 274 
 275   # Instantiate all outputs, then loop over each query and gather metrics.
 276   mean_average_precision = 0.0
 277   mean_precisions = np.zeros([num_desired_pr_ranks])
 278   mean_recalls = np.zeros([num_desired_pr_ranks])
 279   average_precisions = np.zeros([num_queries])
 280   precisions = np.zeros([num_queries, num_desired_pr_ranks])
 281   recalls = np.zeros([num_queries, num_desired_pr_ranks])
 282   num_empty_gt_queries = 0
 283   for i in range(num_queries):
 284     ok_index_images = ground_truth[i]['ok']
 285     junk_index_images = ground_truth[i]['junk']
 286 
 287     if not ok_index_images.size:
 288       average_precisions[i] = float('nan')
 289       precisions[i, :] = float('nan')
 290       recalls[i, :] = float('nan')
 291       num_empty_gt_queries += 1
 292       continue
 293 
 294     positive_ranks = np.arange(num_index_images)[np.in1d(
 295         sorted_index_ids[i], ok_index_images)]
 296     junk_ranks = np.arange(num_index_images)[np.in1d(sorted_index_ids[i],
 297                                                      junk_index_images)]
 298 
 299     adjusted_positive_ranks = AdjustPositiveRanks(positive_ranks, junk_ranks)
 300 
 301     average_precisions[i] = ComputeAveragePrecision(adjusted_positive_ranks)
 302     precisions[i, :], recalls[i, :] = ComputePRAtRanks(adjusted_positive_ranks,
 303                                                        desired_pr_ranks)
 304 
 305     mean_average_precision += average_precisions[i]
 306     mean_precisions += precisions[i, :]
 307     mean_recalls += recalls[i, :]
 308 
 309   # Normalize aggregated metrics by number of queries.
 310   num_valid_queries = num_queries - num_empty_gt_queries
 311   mean_average_precision /= num_valid_queries
 312   mean_precisions /= num_valid_queries
 313   mean_recalls /= num_valid_queries
 314 
 315   return (mean_average_precision, mean_precisions, mean_recalls,
 316           average_precisions, precisions, recalls)

 319 def SaveMetricsFile(mean_average_precision, mean_precisions, mean_recalls,
 320                     pr_ranks, output_path):
 321   """Saves aggregated retrieval metrics to text file.
 322 
 323   Args:
 324     mean_average_precision: Dict mapping each dataset protocol to a float.
 325     mean_precisions: Dict mapping each dataset protocol to a NumPy array of
 326       floats with shape [len(pr_ranks)].
 327     mean_recalls: Dict mapping each dataset protocol to a NumPy array of floats
 328       with shape [len(pr_ranks)].
 329     pr_ranks: List of integers.
 330     output_path: Full file path.
 331   """
 332   with tf.io.gfile.GFile(output_path, 'w') as f:
 333     for k in sorted(mean_average_precision.keys()):
 334       f.write('{}\n  mAP={}\n  mP@k{} {}\n  mR@k{} {}\n'.format(
 335           k, np.around(mean_average_precision[k] * 100, decimals=2),
 336           np.array(pr_ranks), np.around(mean_precisions[k] * 100, decimals=2),
 337           np.array(pr_ranks), np.around(mean_recalls[k] * 100, decimals=2)))
 338

 340 def _ParseSpaceSeparatedStringsInBrackets(line, prefixes, ind):
 341   """Parses line containing space-separated strings in brackets.
 342 
 343   Args:
 344     line: String, containing line in metrics file with mP@k or mR@k figures.
 345     prefixes: Tuple/list of strings, containing valid prefixes.
 346     ind: Integer indicating which field within brackets is parsed.
 347 
 348   Yields:
 349     entry: String format entry.
 350 
 351   Raises:
 352     ValueError: If input line does not contain a valid prefix.
 353   """
 354   for prefix in prefixes:
 355     if line.startswith(prefix):
 356       line = line[len(prefix):]
 357       break
 358   else:
 359     raise ValueError('Line %s is malformed, cannot find valid prefixes' % line)
 360 
 361   for entry in line.split('[')[ind].split(']')[0].split():
 362     yield entry
 363

 365 def _ParsePrRanks(line):
 366   """Parses PR ranks from mP@k line in metrics file.
 367 
 368   Args:
 369     line: String, containing line in metrics file with mP@k figures.
 370 
 371   Returns:
 372     pr_ranks: List of integers, containing used ranks.
 373 
 374   Raises:
 375     ValueError: If input line is malformed.
 376   """
 377   return [
 378       int(pr_rank) for pr_rank in _ParseSpaceSeparatedStringsInBrackets(
 379           line, ['  mP@k['], 0) if pr_rank
 380   ]
 381 
 382 
 383 def _ParsePrScores(line, num_pr_ranks):
 384   """Parses PR scores from line in metrics file.
 385 
 386   Args:
 387     line: String, containing line in metrics file with mP@k or mR@k figures.
 388     num_pr_ranks: Integer, number of scores that should be in output list.
 389 
 390   Returns:
 391     pr_scores: List of floats, containing scores.
 392 
 393   Raises:
 394     ValueError: If input line is malformed.
 395   """
 396   pr_scores = [
 397       float(pr_score) for pr_score in _ParseSpaceSeparatedStringsInBrackets(
 398           line, ('  mP@k[', '  mR@k['), 1) if pr_score
 399   ]
 400 
 401   if len(pr_scores) != num_pr_ranks:
 402     raise ValueError('Line %s is malformed, expected %d scores but found %d' %
 403                      (line, num_pr_ranks, len(pr_scores)))
 404 
 405   return pr_scores

 408 def ReadMetricsFile(metrics_path):
 409   """Reads aggregated retrieval metrics from text file.
 410 
 411   Args:
 412     metrics_path: Full file path, containing aggregated retrieval metrics.
 413 
 414   Returns:
 415     mean_average_precision: Dict mapping each dataset protocol to a float.
 416     pr_ranks: List of integer ranks used in aggregated recall/precision metrics.
 417     mean_precisions: Dict mapping each dataset protocol to a NumPy array of
 418       floats with shape [len(`pr_ranks`)].
 419     mean_recalls: Dict mapping each dataset protocol to a NumPy array of floats
 420       with shape [len(`pr_ranks`)].
 421 
 422   Raises:
 423     ValueError: If input file is malformed.
 424   """
 425   with tf.io.gfile.GFile(metrics_path, 'r') as f:
 426     file_contents_stripped = [l.rstrip() for l in f]
 427 
 428   if len(file_contents_stripped) % 4:
 429     raise ValueError(
 430         'Malformed input %s: number of lines must be a multiple of 4, '
 431         'but it is %d' % (metrics_path, len(file_contents_stripped)))
 432 
 433   mean_average_precision = {}
 434   pr_ranks = []
 435   mean_precisions = {}
 436   mean_recalls = {}
 437   protocols = set()
 438   for i in range(0, len(file_contents_stripped), 4):
 439     protocol = file_contents_stripped[i]
 440     if protocol in protocols:
 441       raise ValueError(
 442           'Malformed input %s: protocol %s is found a second time' %
 443           (metrics_path, protocol))
 444     protocols.add(protocol)
 445 
 446     # Parse mAP.
 447     mean_average_precision[protocol] = float(
 448         file_contents_stripped[i + 1].split('=')[1]) / 100.0
 449 
 450     # Parse (or check consistency of) pr_ranks.
 451     parsed_pr_ranks = _ParsePrRanks(file_contents_stripped[i + 2])
 452     if not pr_ranks:
 453       pr_ranks = parsed_pr_ranks
 454     else:
 455       if parsed_pr_ranks != pr_ranks:
 456         raise ValueError('Malformed input %s: inconsistent PR ranks' %
 457                          metrics_path)
 458 
 459     # Parse mean precisions.
 460     mean_precisions[protocol] = np.array(
 461         _ParsePrScores(file_contents_stripped[i + 2], len(pr_ranks)),
 462         dtype=float) / 100.0
 463 
 464     # Parse mean recalls.
 465     mean_recalls[protocol] = np.array(
 466         _ParsePrScores(file_contents_stripped[i + 3], len(pr_ranks)),
 467         dtype=float) / 100.0
 468 
 469   return mean_average_precision, pr_ranks, mean_precisions, mean_recalls
