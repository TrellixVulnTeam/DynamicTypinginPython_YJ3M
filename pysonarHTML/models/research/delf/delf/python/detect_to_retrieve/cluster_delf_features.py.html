  15 """Clusters DELF features using the K-means algorithm.
  16 
  17 All DELF local feature descriptors for a given dataset's index images are loaded
  18 as the input.
  19 
  20 Note that:
  21 -  we only use features extracted from whole images (no features from boxes are
  22    used).
  23 -  the codebook should be trained on Paris images for Oxford retrieval
  24    experiments, and vice-versa.
  25 """
  26 
  27 from __future__ import absolute_import
  28 from __future__ import division
  29 from __future__ import print_function
  30 
  31 import argparse
  32 import os
  33 import sys
  34 import time
  35 
  36 import numpy as np
  37 import tensorflow as tf
  38 
  39 from tensorflow.python.platform import app
  40 from delf import feature_io
  41 from delf.python.detect_to_retrieve import dataset
  42 
  43 cmd_args = None
  44 
  45 # Extensions.
  46 _DELF_EXTENSION = '.delf'
  47 
  48 # Default DELF dimensionality.
  49 _DELF_DIM = 128
  50 
  51 # Pace to report log when collecting features.
  52 _STATUS_CHECK_ITERATIONS = 100
  53 
  54 
  55 class _IteratorInitHook(tf.estimator.SessionRunHook):
  56   """Hook to initialize data iterator after session is created."""
  57 
  58   def __init__(self):
  59     super(_IteratorInitHook, self).__init__()
  60     self.iterator_initializer_fn = None
  61 
  62   def after_create_session(self, session, coord):
  63     """Initialize the iterator after the session has been created."""
  64     del coord
  65     self.iterator_initializer_fn(session)
  66 
  67 
  68 def main(argv):
  69   if len(argv) > 1:
  70     raise RuntimeError('Too many command-line arguments.')
  71 
  72   # Process output directory.
  73   if tf.io.gfile.exists(cmd_args.output_cluster_dir):
  74     raise RuntimeError(
  75         'output_cluster_dir = %s already exists. This may indicate that a '
  76         'previous run already wrote checkpoints in this directory, which would '
  77         'lead to incorrect training. Please re-run this script by specifying an'
  78         ' inexisting directory.' % cmd_args.output_cluster_dir)
  79   else:
  80     tf.io.gfile.makedirs(cmd_args.output_cluster_dir)
  81 
  82   # Read list of index images from dataset file.
  83   print('Reading list of index images from dataset file...')
  84   _, index_list, _ = dataset.ReadDatasetFile(cmd_args.dataset_file_path)
  85   num_images = len(index_list)
  86   print('done! Found %d images' % num_images)
  87 
  88   # Loop over list of index images and collect DELF features.
  89   features_for_clustering = []
  90   start = time.clock()
  91   print('Starting to collect features from index images...')
  92   for i in range(num_images):
  93     if i > 0 and i % _STATUS_CHECK_ITERATIONS == 0:
  94       elapsed = (time.clock() - start)
  95       print('Processing index image %d out of %d, last %d '
  96             'images took %f seconds' %
  97             (i, num_images, _STATUS_CHECK_ITERATIONS, elapsed))
  98       start = time.clock()
  99 
 100     features_filename = index_list[i] + _DELF_EXTENSION
 101     features_fullpath = os.path.join(cmd_args.features_dir, features_filename)
 102     _, _, features, _, _ = feature_io.ReadFromFile(features_fullpath)
 103     if features.size != 0:
 104       assert features.shape[1] == _DELF_DIM
 105     for feature in features:
 106       features_for_clustering.append(feature)
 107 
 108   features_for_clustering = np.array(features_for_clustering, dtype=np.float32)
 109   print('All features were loaded! There are %d features, each with %d '
 110         'dimensions' %
 111         (features_for_clustering.shape[0], features_for_clustering.shape[1]))
 112 
 113   # Run K-means clustering.
 114   def _get_input_fn():
 115     """Helper function to create input function and hook for training.
 116 
 117     Returns:
 118       input_fn: Input function for k-means Estimator training.
 119       init_hook: Hook used to load data during training.
 120     """
 121     init_hook = _IteratorInitHook()
 122 
 123     def _input_fn():
 124       """Produces tf.data.Dataset object for k-means training.
 125 
 126       Returns:
 127         Tensor with the data for training.
 128       """
 129       features_placeholder = tf.compat.v1.placeholder(
 130           tf.float32, features_for_clustering.shape)
 131       delf_dataset = tf.data.Dataset.from_tensor_slices((features_placeholder))
 132       delf_dataset = delf_dataset.shuffle(1000).batch(
 133           features_for_clustering.shape[0])
 134       iterator = delf_dataset.make_initializable_iterator()
 135 
 136       def _initializer_fn(sess):
 137         """Initialize dataset iterator, feed in the data."""
 138         sess.run(
 139             iterator.initializer,
 140             feed_dict={features_placeholder: features_for_clustering})
 141 
 142       init_hook.iterator_initializer_fn = _initializer_fn
 143       return iterator.get_next()
 144 
 145     return _input_fn, init_hook
 146 
 147   input_fn, init_hook = _get_input_fn()
 148 
 149   kmeans = tf.compat.v1.estimator.experimental.KMeans(
 150       num_clusters=cmd_args.num_clusters,
 151       model_dir=cmd_args.output_cluster_dir,
 152       use_mini_batch=False,
 153   )
 154 
 155   print('Starting K-means clustering...')
 156   start = time.clock()
 157   for i in range(cmd_args.num_iterations):
 158     kmeans.train(input_fn, hooks=[init_hook])
 159     average_sum_squared_error = kmeans.evaluate(
 160         input_fn, hooks=[init_hook])['score'] / features_for_clustering.shape[0]
 161     elapsed = (time.clock() - start)
 162     print('K-means iteration %d (out of %d) took %f seconds, '
 163           'average-sum-of-squares: %f' %
 164           (i, cmd_args.num_iterations, elapsed, average_sum_squared_error))
 165     start = time.clock()
 166 
 167   print('K-means clustering finished!')
 168 
 169 
 170 if __name__ == '__main__':
 171   parser = argparse.ArgumentParser()
 172   parser.register('type', 'bool', lambda v: v.lower() == 'true')
 173   parser.add_argument(
 174       '--dataset_file_path',
 175       type=str,
 176       default='/tmp/gnd_roxford5k.mat',
 177       help="""
 178       Dataset file for Revisited Oxford or Paris dataset, in .mat format. The
 179       list of index images loaded from this file is used to collect local
 180       features, which are assumed to be in <image_name>.delf file format.
 181       """)
 182   parser.add_argument(
 183       '--features_dir',
 184       type=str,
 185       default='/tmp/features',
 186       help="""
 187       Directory where DELF feature files are to be found.
 188       """)
 189   parser.add_argument(
 190       '--num_clusters',
 191       type=int,
 192       default=1024,
 193       help="""
 194       Number of clusters to use.
 195       """)
 196   parser.add_argument(
 197       '--num_iterations',
 198       type=int,
 199       default=50,
 200       help="""
 201       Number of iterations to use.
 202       """)
 203   parser.add_argument(
 204       '--output_cluster_dir',
 205       type=str,
 206       default='/tmp/cluster',
 207       help="""
 208       Directory where clustering outputs are written to. This directory should
 209       not exist before running this script; it will be created during
 210       clustering.
 211       """)
 212   cmd_args, unparsed = parser.parse_known_args()
 213   app.run(main=main, argv=[sys.argv[0]] + unparsed)
