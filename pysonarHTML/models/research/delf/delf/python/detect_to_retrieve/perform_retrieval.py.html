  39 # Aliases for aggregation types.
  40 _VLAD = aggregation_config_pb2.AggregationConfig.VLAD
  41 _ASMK = aggregation_config_pb2.AggregationConfig.ASMK
  42 _ASMK_STAR = aggregation_config_pb2.AggregationConfig.ASMK_STAR
  43 
  44 # Extensions.
  45 _VLAD_EXTENSION_SUFFIX = 'vlad'
  46 _ASMK_EXTENSION_SUFFIX = 'asmk'
  47 _ASMK_STAR_EXTENSION_SUFFIX = 'asmk_star'
  48 
  49 # Precision-recall ranks to use in metric computation.
  50 _PR_RANKS = (1, 5, 10)
  51 
  52 # Pace to log.
  53 _STATUS_CHECK_LOAD_ITERATIONS = 50
  54 
  55 # Output file names.
  56 _METRICS_FILENAME = 'metrics.txt'

  59 def _ReadAggregatedDescriptors(input_dir, image_list, config):
  60   """Reads aggregated descriptors.
  61 
  62   Args:
  63     input_dir: Directory where aggregated descriptors are located.
  64     image_list: List of image names for which to load descriptors.
  65     config: AggregationConfig used for images.
  66 
  67   Returns:
  68     aggregated_descriptors: List containing #images items, each a 1D NumPy
  69       array.
  70     visual_words: If using VLAD aggregation, returns an empty list. Otherwise,
  71       returns a list containing #images items, each a 1D NumPy array.
  72   """
  73   # Compose extension of aggregated descriptors.
  74   extension = '.'
  75   if config.use_regional_aggregation:
  76     extension += 'r'
  77   if config.aggregation_type == _VLAD:
  78     extension += _VLAD_EXTENSION_SUFFIX
  79   elif config.aggregation_type == _ASMK:
  80     extension += _ASMK_EXTENSION_SUFFIX
  81   elif config.aggregation_type == _ASMK_STAR:
  82     extension += _ASMK_STAR_EXTENSION_SUFFIX
  83   else:
  84     raise ValueError('Invalid aggregation type: %d' % config.aggregation_type)
  85 
  86   num_images = len(image_list)
  87   aggregated_descriptors = []
  88   visual_words = []
  89   print('Starting to collect descriptors for %d images...' % num_images)
  90   start = time.clock()
  91   for i in range(num_images):
  92     if i > 0 and i % _STATUS_CHECK_LOAD_ITERATIONS == 0:
  93       elapsed = (time.clock() - start)
  94       print('Reading descriptors for image %d out of %d, last %d '
  95             'images took %f seconds' %
  96             (i, num_images, _STATUS_CHECK_LOAD_ITERATIONS, elapsed))
  97       start = time.clock()
  98 
  99     descriptors_filename = image_list[i] + extension
 100     descriptors_fullpath = os.path.join(input_dir, descriptors_filename)
 101     if config.aggregation_type == _VLAD:
 102       aggregated_descriptors.append(datum_io.ReadFromFile(descriptors_fullpath))
 103     else:
 104       d, v = datum_io.ReadPairFromFile(descriptors_fullpath)
 105       if config.aggregation_type == _ASMK_STAR:
 106         d = d.astype('uint8')
 107 
 108       aggregated_descriptors.append(d)
 109       visual_words.append(v)
 110 
 111   return aggregated_descriptors, visual_words

 114 def main(argv):
 115   if len(argv) > 1:
 116     raise RuntimeError('Too many command-line arguments.')
 117 
 118   # Parse dataset to obtain query/index images, and ground-truth.
 119   print('Parsing dataset...')
 120   query_list, index_list, ground_truth = dataset.ReadDatasetFile(
 121       cmd_args.dataset_file_path)
 122   num_query_images = len(query_list)
 123   num_index_images = len(index_list)
 124   (_, medium_ground_truth,
 125    hard_ground_truth) = dataset.ParseEasyMediumHardGroundTruth(ground_truth)
 126   print('done! Found %d queries and %d index images' %
 127         (num_query_images, num_index_images))
 128 
 129   # Parse AggregationConfig protos.
 130   query_config = aggregation_config_pb2.AggregationConfig()
 131   with tf.io.gfile.GFile(cmd_args.query_aggregation_config_path, 'r') as f:
 132     text_format.Merge(f.read(), query_config)
 133   index_config = aggregation_config_pb2.AggregationConfig()
 134   with tf.io.gfile.GFile(cmd_args.index_aggregation_config_path, 'r') as f:
 135     text_format.Merge(f.read(), index_config)
 136 
 137   # Read aggregated descriptors.
 138   query_aggregated_descriptors, query_visual_words = _ReadAggregatedDescriptors(
 139       cmd_args.query_aggregation_dir, query_list, query_config)
 140   index_aggregated_descriptors, index_visual_words = _ReadAggregatedDescriptors(
 141       cmd_args.index_aggregation_dir, index_list, index_config)
 142 
 143   # Create similarity computer.
 144   similarity_computer = (
 145       feature_aggregation_similarity.SimilarityAggregatedRepresentation(
 146           index_config))
 147 
 148   # Compute similarity between query and index images, potentially re-ranking
 149   # with geometric verification.
 150   ranks_before_gv = np.zeros([num_query_images, num_index_images],
 151                              dtype='int32')
 152   if cmd_args.use_geometric_verification:
 153     medium_ranks_after_gv = np.zeros([num_query_images, num_index_images],
 154                                      dtype='int32')
 155     hard_ranks_after_gv = np.zeros([num_query_images, num_index_images],
 156                                    dtype='int32')
 157   for i in range(num_query_images):
 158     print('Performing retrieval with query %d (%s)...' % (i, query_list[i]))
 159     start = time.clock()
 160 
 161     # Compute similarity between aggregated descriptors.
 162     similarities = np.zeros([num_index_images])
 163     for j in range(num_index_images):
 164       similarities[j] = similarity_computer.ComputeSimilarity(
 165           query_aggregated_descriptors[i], index_aggregated_descriptors[j],
 166           query_visual_words[i], index_visual_words[j])
 167 
 168     ranks_before_gv[i] = np.argsort(-similarities)
 169 
 170     # Re-rank using geometric verification.
 171     if cmd_args.use_geometric_verification:
 172       medium_ranks_after_gv[i] = image_reranking.RerankByGeometricVerification(
 173           ranks_before_gv[i], similarities, query_list[i], index_list,
 174           cmd_args.query_features_dir, cmd_args.index_features_dir,
 175           set(medium_ground_truth[i]['junk']))
 176       hard_ranks_after_gv[i] = image_reranking.RerankByGeometricVerification(
 177           ranks_before_gv[i], similarities, query_list[i], index_list,
 178           cmd_args.query_features_dir, cmd_args.index_features_dir,
 179           set(hard_ground_truth[i]['junk']))
 180 
 181     elapsed = (time.clock() - start)
 182     print('done! Retrieval for query %d took %f seconds' % (i, elapsed))
 183 
 184   # Create output directory if necessary.
 185   if not tf.io.gfile.exists(cmd_args.output_dir):
 186     tf.io.gfile.makedirs(cmd_args.output_dir)
 187 
 188   # Compute metrics.
 189   medium_metrics = dataset.ComputeMetrics(ranks_before_gv, medium_ground_truth,
 190                                           _PR_RANKS)
 191   hard_metrics = dataset.ComputeMetrics(ranks_before_gv, hard_ground_truth,
 192                                         _PR_RANKS)
 193   if cmd_args.use_geometric_verification:
 194     medium_metrics_after_gv = dataset.ComputeMetrics(medium_ranks_after_gv,
 195                                                      medium_ground_truth,
 196                                                      _PR_RANKS)
 197     hard_metrics_after_gv = dataset.ComputeMetrics(hard_ranks_after_gv,
 198                                                    hard_ground_truth, _PR_RANKS)
 199 
 200   # Write metrics to file.
 201   mean_average_precision_dict = {
 202       'medium': medium_metrics[0],
 203       'hard': hard_metrics[0]
 204   }
 205   mean_precisions_dict = {'medium': medium_metrics[1], 'hard': hard_metrics[1]}
 206   mean_recalls_dict = {'medium': medium_metrics[2], 'hard': hard_metrics[2]}
 207   if cmd_args.use_geometric_verification:
 208     mean_average_precision_dict.update({
 209         'medium_after_gv': medium_metrics_after_gv[0],
 210         'hard_after_gv': hard_metrics_after_gv[0]
 211     })
 212     mean_precisions_dict.update({
 213         'medium_after_gv': medium_metrics_after_gv[1],
 214         'hard_after_gv': hard_metrics_after_gv[1]
 215     })
 216     mean_recalls_dict.update({
 217         'medium_after_gv': medium_metrics_after_gv[2],
 218         'hard_after_gv': hard_metrics_after_gv[2]
 219     })
 220   dataset.SaveMetricsFile(mean_average_precision_dict, mean_precisions_dict,
 221                           mean_recalls_dict, _PR_RANKS,
 222                           os.path.join(cmd_args.output_dir, _METRICS_FILENAME))
