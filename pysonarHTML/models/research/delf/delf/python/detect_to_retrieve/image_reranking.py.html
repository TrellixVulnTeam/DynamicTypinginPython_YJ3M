  45 def MatchFeatures(query_locations,
  46                   query_descriptors,
  47                   index_image_locations,
  48                   index_image_descriptors,
  49                   ransac_seed=None,
  50                   feature_distance_threshold=0.9,
  51                   ransac_residual_threshold=10.0,
  52                   query_im_array=None,
  53                   index_im_array=None,
  54                   query_im_scale_factors=None,
  55                   index_im_scale_factors=None):
  56   """Matches local features using geometric verification.
  57 
  58   First, finds putative local feature matches by matching `query_descriptors`
  59   against a KD-tree from the `index_image_descriptors`. Then, attempts to fit an
  60   affine transformation between the putative feature corresponces using their
  61   locations.
  62 
  63   Args:
  64     query_locations: Locations of local features for query image. NumPy array of
  65       shape [#query_features, 2].
  66     query_descriptors: Descriptors of local features for query image. NumPy
  67       array of shape [#query_features, depth].
  68     index_image_locations: Locations of local features for index image. NumPy
  69       array of shape [#index_image_features, 2].
  70     index_image_descriptors: Descriptors of local features for index image.
  71       NumPy array of shape [#index_image_features, depth].
  72     ransac_seed: Seed used by RANSAC. If None (default), no seed is provided.
  73     feature_distance_threshold: Distance threshold below which a pair of
  74       features is considered a potential match, and will be fed into RANSAC.
  75     ransac_residual_threshold: Residual error threshold for considering matches
  76       as inliers, used in RANSAC algorithm.
  77     query_im_array: Optional. If not None, contains a NumPy array with the query
  78       image, used to produce match visualization, if there is a match.
  79     index_im_array: Optional. Same as `query_im_array`, but for index image.
  80     query_im_scale_factors: Optional. If not None, contains a NumPy array with
  81       the query image scales, used to produce match visualization, if there is a
  82       match. If None and a visualization will be produced, [1.0, 1.0] is used
  83       (ie, feature locations are not scaled).
  84     index_im_scale_factors: Optional. Same as `query_im_scale_factors`, but for
  85       index image.
  86 
  87   Returns:
  88     score: Number of inliers of match. If no match is found, returns 0.
  89     match_viz_bytes: Encoded image bytes with visualization of the match, if
  90       there is one, and if `query_im_array` and `index_im_array` are properly
  91       set. Otherwise, it's an empty bytes string.
  92 
  93   Raises:
  94     ValueError: If local descriptors from query and index images have different
  95       dimensionalities.
  96   """
  97   num_features_query = query_locations.shape[0]
  98   num_features_index_image = index_image_locations.shape[0]
  99   if not num_features_query or not num_features_index_image:
 100     return 0, b''
 101 
 102   local_feature_dim = query_descriptors.shape[1]
 103   if index_image_descriptors.shape[1] != local_feature_dim:
 104     raise ValueError(
 105         'Local feature dimensionality is not consistent for query and index '
 106         'images.')
 107 
 108   # Find nearest-neighbor matches using a KD tree.
 109   index_image_tree = spatial.cKDTree(index_image_descriptors)
 110   _, indices = index_image_tree.query(
 111       query_descriptors, distance_upper_bound=feature_distance_threshold)
 112 
 113   # Select feature locations for putative matches.
 114   query_locations_to_use = np.array([
 115       query_locations[i,]
 116       for i in range(num_features_query)
 117       if indices[i] != num_features_index_image
 118   ])
 119   index_image_locations_to_use = np.array([
 120       index_image_locations[indices[i],]
 121       for i in range(num_features_query)
 122       if indices[i] != num_features_index_image
 123   ])
 124 
 125   # If there are not enough putative matches, early return 0.
 126   if query_locations_to_use.shape[0] <= _MIN_RANSAC_SAMPLES:
 127     return 0, b''
 128 
 129   # Perform geometric verification using RANSAC.
 130   _, inliers = measure.ransac(
 131       (index_image_locations_to_use, query_locations_to_use),
 132       transform.AffineTransform,
 133       min_samples=_MIN_RANSAC_SAMPLES,
 134       residual_threshold=ransac_residual_threshold,
 135       max_trials=_NUM_RANSAC_TRIALS,
 136       random_state=ransac_seed)
 137   match_viz_bytes = b''
 138 
 139   if inliers is None:
 140     inliers = []
 141   elif query_im_array is not None and index_im_array is not None:
 142     if query_im_scale_factors is None:
 143       query_im_scale_factors = [1.0, 1.0]
 144     if index_im_scale_factors is None:
 145       index_im_scale_factors = [1.0, 1.0]
 146     inlier_idxs = np.nonzero(inliers)[0]
 147     _, ax = plt.subplots()
 148     ax.axis('off')
 149     ax.xaxis.set_major_locator(plt.NullLocator())
 150     ax.yaxis.set_major_locator(plt.NullLocator())
 151     plt.subplots_adjust(top=1, bottom=0, right=1, left=0, hspace=0, wspace=0)
 152     plt.margins(0, 0)
 153     feature.plot_matches(
 154         ax,
 155         query_im_array,
 156         index_im_array,
 157         query_locations_to_use * query_im_scale_factors,
 158         index_image_locations_to_use * index_im_scale_factors,
 159         np.column_stack((inlier_idxs, inlier_idxs)),
 160         only_matches=True)
 161 
 162     match_viz_io = io.BytesIO()
 163     plt.savefig(match_viz_io, format='jpeg', bbox_inches='tight', pad_inches=0)
 164     match_viz_bytes = match_viz_io.getvalue()
 165 
 166   return sum(inliers), match_viz_bytes

 169 def RerankByGeometricVerification(input_ranks,
 170                                   initial_scores,
 171                                   query_name,
 172                                   index_names,
 173                                   query_features_dir,
 174                                   index_features_dir,
 175                                   junk_ids,
 176                                   local_feature_extension=_DELF_EXTENSION,
 177                                   ransac_seed=None,
 178                                   feature_distance_threshold=0.9,
 179                                   ransac_residual_threshold=10.0):
 180   """Re-ranks retrieval results using geometric verification.
 181 
 182   Args:
 183     input_ranks: 1D NumPy array with indices of top-ranked index images, sorted
 184       from the most to the least similar.
 185     initial_scores: 1D NumPy array with initial similarity scores between query
 186       and index images. Entry i corresponds to score for image i.
 187     query_name: Name for query image (string).
 188     index_names: List of names for index images (strings).
 189     query_features_dir: Directory where query local feature file is located
 190       (string).
 191     index_features_dir: Directory where index local feature files are located
 192       (string).
 193     junk_ids: Set with indices of junk images which should not be considered
 194       during re-ranking.
 195     local_feature_extension: String, extension to use for loading local feature
 196       files.
 197     ransac_seed: Seed used by RANSAC. If None (default), no seed is provided.
 198     feature_distance_threshold: Distance threshold below which a pair of local
 199       features is considered a potential match, and will be fed into RANSAC.
 200     ransac_residual_threshold: Residual error threshold for considering matches
 201       as inliers, used in RANSAC algorithm.
 202 
 203   Returns:
 204     output_ranks: 1D NumPy array with index image indices, sorted from the most
 205       to the least similar according to the geometric verification and initial
 206       scores.
 207 
 208   Raises:
 209     ValueError: If `input_ranks`, `initial_scores` and `index_names` do not have
 210       the same number of entries.
 211   """
 212   num_index_images = len(index_names)
 213   if len(input_ranks) != num_index_images:
 214     raise ValueError('input_ranks and index_names have different number of '
 215                      'elements: %d vs %d' %
 216                      (len(input_ranks), len(index_names)))
 217   if len(initial_scores) != num_index_images:
 218     raise ValueError('initial_scores and index_names have different number of '
 219                      'elements: %d vs %d' %
 220                      (len(initial_scores), len(index_names)))
 221 
 222   # Filter out junk images from list that will be re-ranked.
 223   input_ranks_for_gv = []
 224   for ind in input_ranks:
 225     if ind not in junk_ids:
 226       input_ranks_for_gv.append(ind)
 227   num_to_rerank = min(_NUM_TO_RERANK, len(input_ranks_for_gv))
 228 
 229   # Load query image features.
 230   query_features_path = os.path.join(query_features_dir,
 231                                      query_name + local_feature_extension)
 232   query_locations, _, query_descriptors, _, _ = feature_io.ReadFromFile(
 233       query_features_path)
 234 
 235   # Initialize list containing number of inliers and initial similarity scores.
 236   inliers_and_initial_scores = []
 237   for i in range(num_index_images):
 238     inliers_and_initial_scores.append([0, initial_scores[i]])
 239 
 240   # Loop over top-ranked images and get results.
 241   print('Starting to re-rank')
 242   for i in range(num_to_rerank):
 243     if i > 0 and i % _STATUS_CHECK_GV_ITERATIONS == 0:
 244       print('Re-ranking: i = %d out of %d' % (i, num_to_rerank))
 245 
 246     index_image_id = input_ranks_for_gv[i]
 247 
 248     # Load index image features.
 249     index_image_features_path = os.path.join(
 250         index_features_dir,
 251         index_names[index_image_id] + local_feature_extension)
 252     (index_image_locations, _, index_image_descriptors, _,
 253      _) = feature_io.ReadFromFile(index_image_features_path)
 254 
 255     inliers_and_initial_scores[index_image_id][0], _ = MatchFeatures(
 256         query_locations,
 257         query_descriptors,
 258         index_image_locations,
 259         index_image_descriptors,
 260         ransac_seed=ransac_seed,
 261         feature_distance_threshold=feature_distance_threshold,
 262         ransac_residual_threshold=ransac_residual_threshold)
 263 
 264   # Sort based on (inliers_score, initial_score).
 265   def _InliersInitialScoresSorting(k):
 266     """Helper function to sort list based on two entries.
 267 
 268     Args:
 269       k: Index into `inliers_and_initial_scores`.
 270 
 271     Returns:
 272       Tuple containing inlier score and initial score.
 273     """
 274     return (inliers_and_initial_scores[k][0], inliers_and_initial_scores[k][1])
 275 
 276   output_ranks = sorted(
 277       range(num_index_images), key=_InliersInitialScoresSorting, reverse=True)
 278 
 279   return output_ranks
