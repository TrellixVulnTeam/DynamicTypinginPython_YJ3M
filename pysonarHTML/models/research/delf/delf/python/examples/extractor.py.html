  28 # Minimum dimensions below which DELF features are not extracted (empty
  29 # features are returned). This applies after any resizing is performed.
  30 _MIN_HEIGHT = 10
  31 _MIN_WIDTH = 10

  34 def ResizeImage(image, config, resize_factor=1.0):
  35   """Resizes image according to config.
  36 
  37   Args:
  38     image: Uint8 array with shape (height, width, 3).
  39     config: DelfConfig proto containing the model configuration.
  40     resize_factor: Optional float resize factor for the input image. If given,
  41       the maximum and minimum allowed image sizes in `config` are scaled by this
  42       factor. Must be non-negative.
  43 
  44   Returns:
  45     resized_image: Uint8 array with resized image.
  46     scale_factors: 2D float array, with factors used for resizing along height
  47       and width (If upscaling, larger than 1; if downscaling, smaller than 1).
  48 
  49   Raises:
  50     ValueError: If `image` has incorrect number of dimensions/channels.
  51   """
  52   if resize_factor < 0.0:
  53     raise ValueError('negative resize_factor is not allowed: %f' %
  54                      resize_factor)
  55   if image.ndim != 3:
  56     raise ValueError('image has incorrect number of dimensions: %d' %
  57                      image.ndims)
  58   height, width, channels = image.shape
  59 
  60   # Take into account resize factor.
  61   max_image_size = resize_factor * config.max_image_size
  62   min_image_size = resize_factor * config.min_image_size
  63 
  64   if channels != 3:
  65     raise ValueError('image has incorrect number of channels: %d' % channels)
  66 
  67   largest_side = max(width, height)
  68 
  69   if max_image_size >= 0 and largest_side > max_image_size:
  70     scale_factor = max_image_size / largest_side
  71   elif min_image_size >= 0 and largest_side < min_image_size:
  72     scale_factor = min_image_size / largest_side
  73   elif config.use_square_images and (height != width):
  74     scale_factor = 1.0
  75   else:
  76     # No resizing needed, early return.
  77     return image, np.ones(2, dtype=float)
  78 
  79   # Note that new_shape is in (width, height) format (PIL convention), while
  80   # scale_factors are in (height, width) convention (NumPy convention).
  81   if config.use_square_images:
  82     new_shape = (int(round(largest_side * scale_factor)),
  83                  int(round(largest_side * scale_factor)))
  84   else:
  85     new_shape = (int(round(width * scale_factor)),
  86                  int(round(height * scale_factor)))
  87 
  88   scale_factors = np.array([new_shape[1] / height, new_shape[0] / width],
  89                            dtype=float)
  90 
  91   pil_image = Image.fromarray(image)
  92   resized_image = np.array(pil_image.resize(new_shape, resample=Image.BILINEAR))
  93 
  94   return resized_image, scale_factors

  97 def MakeExtractor(config):
  98   """Creates a function to extract global and/or local features from an image.
  99 
 100   Args:
 101     config: DelfConfig proto containing the model configuration.
 102 
 103   Returns:
 104     Function that receives an image and returns features.
 105   """
 106   # Load model.
 107   model = tf.saved_model.load(config.model_path)
 108 
 109   # Input/output end-points/tensors.
 110   feeds = ['input_image:0', 'input_scales:0']
 111   fetches = []
 112   image_scales_tensor = tf.convert_to_tensor(list(config.image_scales))
 113 
 114   # Custom configuration needed when local features are used.
 115   if config.use_local_features:
 116     # Extra input/output end-points/tensors.
 117     feeds.append('input_abs_thres:0')
 118     feeds.append('input_max_feature_num:0')
 119     fetches.append('boxes:0')
 120     fetches.append('features:0')
 121     fetches.append('scales:0')
 122     fetches.append('scores:0')
 123     score_threshold_tensor = tf.constant(
 124         config.delf_local_config.score_threshold)
 125     max_feature_num_tensor = tf.constant(
 126         config.delf_local_config.max_feature_num)
 127 
 128     # If using PCA, pre-load required parameters.
 129     local_pca_parameters = {}
 130     if config.delf_local_config.use_pca:
 131       local_pca_parameters['mean'] = tf.constant(
 132           datum_io.ReadFromFile(
 133               config.delf_local_config.pca_parameters.mean_path),
 134           dtype=tf.float32)
 135       local_pca_parameters['matrix'] = tf.constant(
 136           datum_io.ReadFromFile(
 137               config.delf_local_config.pca_parameters.projection_matrix_path),
 138           dtype=tf.float32)
 139       local_pca_parameters[
 140           'dim'] = config.delf_local_config.pca_parameters.pca_dim
 141       local_pca_parameters['use_whitening'] = (
 142           config.delf_local_config.pca_parameters.use_whitening)
 143       if config.delf_local_config.pca_parameters.use_whitening:
 144         local_pca_parameters['variances'] = tf.squeeze(
 145             tf.constant(
 146                 datum_io.ReadFromFile(
 147                     config.delf_local_config.pca_parameters.pca_variances_path),
 148                 dtype=tf.float32))
 149       else:
 150         local_pca_parameters['variances'] = None
 151 
 152   # Custom configuration needed when global features are used.
 153   if config.use_global_features:
 154     # Extra output end-point.
 155     fetches.append('global_descriptors:0')
 156 
 157     # If using PCA, pre-load required parameters.
 158     global_pca_parameters = {}
 159     if config.delf_global_config.use_pca:
 160       global_pca_parameters['mean'] = tf.constant(
 161           datum_io.ReadFromFile(
 162               config.delf_global_config.pca_parameters.mean_path),
 163           dtype=tf.float32)
 164       global_pca_parameters['matrix'] = tf.constant(
 165           datum_io.ReadFromFile(
 166               config.delf_global_config.pca_parameters.projection_matrix_path),
 167           dtype=tf.float32)
 168       global_pca_parameters[
 169           'dim'] = config.delf_global_config.pca_parameters.pca_dim
 170       global_pca_parameters['use_whitening'] = (
 171           config.delf_global_config.pca_parameters.use_whitening)
 172       if config.delf_global_config.pca_parameters.use_whitening:
 173         global_pca_parameters['variances'] = tf.squeeze(
 174             tf.constant(
 175                 datum_io.ReadFromFile(config.delf_global_config.pca_parameters
 176                                       .pca_variances_path),
 177                 dtype=tf.float32))
 178       else:
 179         global_pca_parameters['variances'] = None
 180 
 181   model = model.prune(feeds=feeds, fetches=fetches)
 182 
 183   def ExtractorFn(image, resize_factor=1.0):
 184     """Receives an image and returns DELF global and/or local features.
 185 
 186     If image is too small, returns empty features.
 187 
 188     Args:
 189       image: Uint8 array with shape (height, width, 3) containing the RGB image.
 190       resize_factor: Optional float resize factor for the input image. If given,
 191         the maximum and minimum allowed image sizes in the config are scaled by
 192         this factor.
 193 
 194     Returns:
 195       extracted_features: A dict containing the extracted global descriptors
 196         (key 'global_descriptor' mapping to a [D] float array), and/or local
 197         features (key 'local_features' mapping to a dict with keys 'locations',
 198         'descriptors', 'scales', 'attention').
 199     """
 200 
 201     resized_image, scale_factors = ResizeImage(
 202         image, config, resize_factor=resize_factor)
 203 
 204     # If the image is too small, returns empty features.
 205     if resized_image.shape[0] < _MIN_HEIGHT or resized_image.shape[
 206         1] < _MIN_WIDTH:
 207       extracted_features = {'global_descriptor': np.array([])}
 208       if config.use_local_features:
 209         extracted_features.update({
 210             'local_features': {
 211                 'locations': np.array([]),
 212                 'descriptors': np.array([]),
 213                 'scales': np.array([]),
 214                 'attention': np.array([]),
 215             }
 216         })
 217       return extracted_features
 218 
 219     # Input tensors.
 220     image_tensor = tf.convert_to_tensor(resized_image)
 221 
 222     # Extracted features.
 223     extracted_features = {}
 224     output = None
 225 
 226     if config.use_local_features:
 227       output = model(image_tensor, image_scales_tensor, score_threshold_tensor,
 228                      max_feature_num_tensor)
 229     else:
 230       output = model(image_tensor, image_scales_tensor)
 231 
 232     # Post-process extracted features: normalize, PCA (optional), pooling.
 233     if config.use_global_features:
 234       raw_global_descriptors = output[-1]
 235       if config.delf_global_config.image_scales_ind:
 236         raw_global_descriptors_selected_scales = tf.gather(
 237             raw_global_descriptors,
 238             list(config.delf_global_config.image_scales_ind))
 239       else:
 240         raw_global_descriptors_selected_scales = raw_global_descriptors
 241       global_descriptors_per_scale = feature_extractor.PostProcessDescriptors(
 242           raw_global_descriptors_selected_scales,
 243           config.delf_global_config.use_pca, global_pca_parameters)
 244       unnormalized_global_descriptor = tf.reduce_sum(
 245           global_descriptors_per_scale, axis=0, name='sum_pooling')
 246       global_descriptor = tf.nn.l2_normalize(
 247           unnormalized_global_descriptor, axis=0, name='final_l2_normalization')
 248       extracted_features.update({
 249           'global_descriptor': global_descriptor.numpy(),
 250       })
 251 
 252     if config.use_local_features:
 253       boxes = output[0]
 254       raw_local_descriptors = output[1]
 255       feature_scales = output[2]
 256       attention_with_extra_dim = output[3]
 257 
 258       attention = tf.reshape(attention_with_extra_dim,
 259                              [tf.shape(attention_with_extra_dim)[0]])
 260       locations, local_descriptors = (
 261           feature_extractor.DelfFeaturePostProcessing(
 262               boxes, raw_local_descriptors, config.delf_local_config.use_pca,
 263               local_pca_parameters))
 264       locations /= scale_factors
 265 
 266       extracted_features.update({
 267           'local_features': {
 268               'locations': locations.numpy(),
 269               'descriptors': local_descriptors.numpy(),
 270               'scales': feature_scales.numpy(),
 271               'attention': attention.numpy(),
 272           }
 273       })
 274 
 275     return extracted_features
 276 
 277   return ExtractorFn
