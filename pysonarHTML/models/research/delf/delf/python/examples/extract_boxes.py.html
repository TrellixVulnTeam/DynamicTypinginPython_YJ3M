  15 """Extracts bounding boxes from a list of images, saving them to files.
  16 
  17 The images must be in JPG format. The program checks if boxes already
  18 exist, and skips computation for those.
  19 """

  42 # Extension/suffix of produced files.
  43 _BOX_EXT = '.boxes'
  44 _VIZ_SUFFIX = '_viz.jpg'
  45 
  46 # Used for plotting boxes.
  47 _BOX_EDGE_COLORS = ['r', 'y', 'b', 'm', 'k', 'g', 'c', 'w']

  49 # Pace to report extraction log.
  50 _STATUS_CHECK_ITERATIONS = 100

  53 def _ReadImageList(list_path):
  54   """Helper function to read image paths.
  55 
  56   Args:
  57     list_path: Path to list of images, one image path per line.
  58 
  59   Returns:
  60     image_paths: List of image paths.
  61   """
  62   with tf.io.gfile.GFile(list_path, 'r') as f:
  63     image_paths = f.readlines()
  64   image_paths = [entry.rstrip() for entry in image_paths]
  65   return image_paths

  68 def _FilterBoxesByScore(boxes, scores, class_indices, score_threshold):
  69   """Filter boxes based on detection scores.
  70 
  71   Boxes with detection score >= score_threshold are returned.
  72 
  73   Args:
  74     boxes: [N, 4] float array denoting bounding box coordinates, in format [top,
  75       left, bottom, right].
  76     scores: [N] float array with detection scores.
  77     class_indices: [N] int array with class indices.
  78     score_threshold: Float detection score threshold to use.
  79 
  80   Returns:
  81     selected_boxes: selected `boxes`.
  82     selected_scores: selected `scores`.
  83     selected_class_indices: selected `class_indices`.
  84   """
  85   selected_boxes = []
  86   selected_scores = []
  87   selected_class_indices = []
  88   for i, box in enumerate(boxes):
  89     if scores[i] >= score_threshold:
  90       selected_boxes.append(box)
  91       selected_scores.append(scores[i])
  92       selected_class_indices.append(class_indices[i])
  93 
  94   return np.array(selected_boxes), np.array(selected_scores), np.array(
  95       selected_class_indices)

  98 def _PlotBoxesAndSaveImage(image, boxes, output_path):
  99   """Plot boxes on image and save to output path.
 100 
 101   Args:
 102     image: Numpy array containing image.
 103     boxes: [N, 4] float array denoting bounding box coordinates, in format [top,
 104       left, bottom, right].
 105     output_path: String containing output path.
 106   """
 107   height = image.shape[0]
 108   width = image.shape[1]
 109 
 110   fig, ax = plt.subplots(1)
 111   ax.imshow(image)
 112   for i, box in enumerate(boxes):
 113     scaled_box = [
 114         box[0] * height, box[1] * width, box[2] * height, box[3] * width
 115     ]
 116     rect = patches.Rectangle([scaled_box[1], scaled_box[0]],
 117                              scaled_box[3] - scaled_box[1],
 118                              scaled_box[2] - scaled_box[0],
 119                              linewidth=3,
 120                              edgecolor=_BOX_EDGE_COLORS[i %
 121                                                         len(_BOX_EDGE_COLORS)],
 122                              facecolor='none')
 123     ax.add_patch(rect)
 124 
 125   ax.axis('off')
 126   plt.savefig(output_path, bbox_inches='tight')
 127   plt.close(fig)

 130 def main(argv):
 131   if len(argv) > 1:
 132     raise RuntimeError('Too many command-line arguments.')
 133 
 134   # Read list of images.
 135   print('Reading list of images...')
 136   image_paths = _ReadImageList(cmd_args.list_images_path)
 137   num_images = len(image_paths)
 138   print(f'done! Found {num_images} images')
 139 
 140   # Create output directories if necessary.
 141   if not tf.io.gfile.exists(cmd_args.output_dir):
 142     tf.io.gfile.makedirs(cmd_args.output_dir)
 143   if cmd_args.output_viz_dir and not tf.io.gfile.exists(
 144       cmd_args.output_viz_dir):
 145     tf.io.gfile.makedirs(cmd_args.output_viz_dir)
 146 
 147   detector_fn = detector.MakeDetector(cmd_args.detector_path)
 148 
 149   start = time.time()
 150   for i, image_path in enumerate(image_paths):
 151     # Report progress once in a while.
 152     if i == 0:
 153       print('Starting to detect objects in images...')
 154     elif i % _STATUS_CHECK_ITERATIONS == 0:
 155       elapsed = (time.time() - start)
 156       print(
 157           f'Processing image {i} out of {num_images}, last '
 158           f'{_STATUS_CHECK_ITERATIONS} images took {elapsed} seconds'
 159           )
 160       start = time.time()
 161 
 162     # If descriptor already exists, skip its computation.
 163     base_boxes_filename, _ = os.path.splitext(os.path.basename(image_path))
 164     out_boxes_filename = base_boxes_filename + _BOX_EXT
 165     out_boxes_fullpath = os.path.join(cmd_args.output_dir,
 166                                       out_boxes_filename)
 167     if tf.io.gfile.exists(out_boxes_fullpath):
 168       print(f'Skipping {image_path}')
 169       continue
 170 
 171     im = np.expand_dims(np.array(utils.RgbLoader(image_paths[i])), 0)
 172 
 173     # Extract and save boxes.
 174     (boxes_out, scores_out, class_indices_out) = detector_fn(im)
 175     (selected_boxes, selected_scores,
 176      selected_class_indices) = _FilterBoxesByScore(boxes_out[0],
 177                                                    scores_out[0],
 178                                                    class_indices_out[0],
 179                                                    cmd_args.detector_thresh)
 180 
 181     box_io.WriteToFile(out_boxes_fullpath, selected_boxes, selected_scores,
 182                        selected_class_indices)
 183     if cmd_args.output_viz_dir:
 184       out_viz_filename = base_boxes_filename + _VIZ_SUFFIX
 185       out_viz_fullpath = os.path.join(cmd_args.output_viz_dir,
 186                                       out_viz_filename)
 187       _PlotBoxesAndSaveImage(im[0], selected_boxes, out_viz_fullpath)
 188
