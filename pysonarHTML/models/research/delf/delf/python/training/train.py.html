  59 def _record_accuracy(metric, logits, labels):
  60   """Record accuracy given predicted logits and ground-truth labels."""
  61   softmax_probabilities = tf.keras.layers.Softmax()(logits)
  62   metric.update_state(labels, softmax_probabilities)

  65 def _attention_summaries(scores, global_step):
  66   """Record statistics of the attention score."""
  67   tf.summary.scalar('attention/max', tf.reduce_max(scores), step=global_step)
  68   tf.summary.scalar('attention/min', tf.reduce_min(scores), step=global_step)
  69   tf.summary.scalar('attention/mean', tf.reduce_mean(scores), step=global_step)
  70   tf.summary.scalar(
  71       'attention/percent_25',
  72       tfp.stats.percentile(scores, 25.0),
  73       step=global_step)
  74   tf.summary.scalar(
  75       'attention/percent_50',
  76       tfp.stats.percentile(scores, 50.0),
  77       step=global_step)
  78   tf.summary.scalar(
  79       'attention/percent_75',
  80       tfp.stats.percentile(scores, 75.0),
  81       step=global_step)

  84 def create_model(num_classes):
  85   """Define DELF model, and initialize classifiers."""
  86   model = delf_model.Delf(block3_strides=FLAGS.block3_strides, name='DELF')
  87   model.init_classifiers(num_classes)
  88   return model

  91 def _learning_rate_schedule(global_step_value, max_iters, initial_lr):
  92   """Calculates learning_rate with linear decay.
  93 
  94   Args:
  95     global_step_value: int, global step.
  96     max_iters: int, maximum iterations.
  97     initial_lr: float, initial learning rate.
  98 
  99   Returns:
 100     lr: float, learning rate.
 101   """
 102   lr = initial_lr * (1.0 - global_step_value / max_iters)
 103   return lr

 106 def main(argv):
 107   if len(argv) > 1:
 108     raise app.UsageError('Too many command-line arguments.')
 109 
 110   #-------------------------------------------------------------
 111   # Log flags used.
 112   logging.info('Running training script with\n')
 113   logging.info('logdir= %s', FLAGS.logdir)
 114   logging.info('initial_lr= %f', FLAGS.initial_lr)
 115   logging.info('block3_strides= %s', str(FLAGS.block3_strides))
 116 
 117   # ------------------------------------------------------------
 118   # Create the strategy.
 119   strategy = tf.distribute.MirroredStrategy()
 120   logging.info('Number of devices: %d', strategy.num_replicas_in_sync)
 121   if FLAGS.debug:
 122     print('Number of devices:', strategy.num_replicas_in_sync)
 123 
 124   max_iters = FLAGS.max_iters
 125   global_batch_size = FLAGS.batch_size
 126   image_size = 321
 127   num_eval = 1000
 128   report_interval = 100
 129   eval_interval = 1000
 130   save_interval = 20000
 131 
 132   initial_lr = FLAGS.initial_lr
 133 
 134   clip_val = tf.constant(10.0)
 135 
 136   if FLAGS.debug:
 137     global_batch_size = 4
 138     max_iters = 4
 139     num_eval = 1
 140     save_interval = 1
 141     report_interval = 1
 142 
 143   # Determine the number of classes based on the version of the dataset.
 144   gld_info = gld.GoogleLandmarksInfo()
 145   num_classes = gld_info.num_classes[FLAGS.dataset_version]
 146 
 147   # ------------------------------------------------------------
 148   # Create the distributed train/validation sets.
 149   train_dataset = gld.CreateDataset(
 150       file_pattern=FLAGS.train_file_pattern,
 151       batch_size=global_batch_size,
 152       image_size=image_size,
 153       augmentation=FLAGS.use_augmentation,
 154       seed=FLAGS.seed)
 155   validation_dataset = gld.CreateDataset(
 156       file_pattern=FLAGS.validation_file_pattern,
 157       batch_size=global_batch_size,
 158       image_size=image_size,
 159       augmentation=False,
 160       seed=FLAGS.seed)
 161 
 162   train_iterator = strategy.make_dataset_iterator(train_dataset)
 163   validation_iterator = strategy.make_dataset_iterator(validation_dataset)
 164 
 165   train_iterator.initialize()
 166   validation_iterator.initialize()
 167 
 168   # Create a checkpoint directory to store the checkpoints.
 169   checkpoint_prefix = os.path.join(FLAGS.logdir, 'delf_tf2-ckpt')
 170 
 171   # ------------------------------------------------------------
 172   # Finally, we do everything in distributed scope.
 173   with strategy.scope():
 174     # Compute loss.
 175     # Set reduction to `none` so we can do the reduction afterwards and divide
 176     # by global batch size.
 177     loss_object = tf.keras.losses.SparseCategoricalCrossentropy(
 178         from_logits=True, reduction=tf.keras.losses.Reduction.NONE)
 179 
 180     def compute_loss(labels, predictions):
 181       per_example_loss = loss_object(labels, predictions)
 182       return tf.nn.compute_average_loss(
 183           per_example_loss, global_batch_size=global_batch_size)
 184 
 185     # Set up metrics.
 186     desc_validation_loss = tf.keras.metrics.Mean(name='desc_validation_loss')
 187     attn_validation_loss = tf.keras.metrics.Mean(name='attn_validation_loss')
 188     desc_train_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(
 189         name='desc_train_accuracy')
 190     attn_train_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(
 191         name='attn_train_accuracy')
 192     desc_validation_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(
 193         name='desc_validation_accuracy')
 194     attn_validation_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(
 195         name='attn_validation_accuracy')
 196 
 197     # ------------------------------------------------------------
 198     # Setup DELF model and optimizer.
 199     model = create_model(num_classes)
 200     logging.info('Model, datasets loaded.\nnum_classes= %d', num_classes)
 201 
 202     optimizer = tf.keras.optimizers.SGD(learning_rate=initial_lr, momentum=0.9)
 203 
 204     # Setup summary writer.
 205     summary_writer = tf.summary.create_file_writer(
 206         os.path.join(FLAGS.logdir, 'train_logs'), flush_millis=10000)
 207 
 208     # Setup checkpoint directory.
 209     checkpoint = tf.train.Checkpoint(optimizer=optimizer, model=model)
 210     manager = tf.train.CheckpointManager(
 211         checkpoint, checkpoint_prefix, max_to_keep=3)
 212 
 213     # ------------------------------------------------------------
 214     # Train step to run on one GPU.
 215     def train_step(inputs):
 216       """Train one batch."""
 217       images, labels = inputs
 218       # Temporary workaround to avoid some corrupted labels.
 219       labels = tf.clip_by_value(labels, 0, model.num_classes)
 220 
 221       global_step = optimizer.iterations
 222       tf.summary.scalar(
 223           'image_range/max', tf.reduce_max(images), step=global_step)
 224       tf.summary.scalar(
 225           'image_range/min', tf.reduce_min(images), step=global_step)
 226 
 227       def _backprop_loss(tape, loss, weights):
 228         """Backpropogate losses using clipped gradients.
 229 
 230         Args:
 231           tape: gradient tape.
 232           loss: scalar Tensor, loss value.
 233           weights: keras model weights.
 234         """
 235         gradients = tape.gradient(loss, weights)
 236         clipped, _ = tf.clip_by_global_norm(gradients, clip_norm=clip_val)
 237         optimizer.apply_gradients(zip(clipped, weights))
 238 
 239       # Record gradients and loss through backbone.
 240       with tf.GradientTape() as desc_tape:
 241 
 242         blocks = {}
 243         prelogits = model.backbone(
 244             images, intermediates_dict=blocks, training=True)
 245 
 246         # Report sparsity.
 247         activations_zero_fractions = {
 248             'sparsity/%s' % k: tf.nn.zero_fraction(v)
 249             for k, v in blocks.items()
 250         }
 251         for k, v in activations_zero_fractions.items():
 252           tf.summary.scalar(k, v, step=global_step)
 253 
 254         # Apply descriptor classifier.
 255         logits = model.desc_classification(prelogits)
 256 
 257         desc_loss = compute_loss(labels, logits)
 258 
 259       # Backprop only through backbone weights.
 260       _backprop_loss(desc_tape, desc_loss, model.desc_trainable_weights)
 261 
 262       # Record descriptor train accuracy.
 263       _record_accuracy(desc_train_accuracy, logits, labels)
 264 
 265       # Record gradients and loss through attention block.
 266       with tf.GradientTape() as attn_tape:
 267         block3 = blocks['block3']  # pytype: disable=key-error
 268 
 269         # Stopping gradients according to DELG paper:
 270         # (https://arxiv.org/abs/2001.05027).
 271         block3 = tf.stop_gradient(block3)
 272 
 273         prelogits, scores, _ = model.attention(block3, training=True)
 274         _attention_summaries(scores, global_step)
 275 
 276         # Apply attention block classifier.
 277         logits = model.attn_classification(prelogits)
 278 
 279         attn_loss = compute_loss(labels, logits)
 280 
 281       # Backprop only through attention weights.
 282       _backprop_loss(attn_tape, attn_loss, model.attn_trainable_weights)
 283 
 284       # Record attention train accuracy.
 285       _record_accuracy(attn_train_accuracy, logits, labels)
 286 
 287       return desc_loss, attn_loss
 288 
 289     # ------------------------------------------------------------
 290     def validation_step(inputs):
 291       """Validate one batch."""
 292       images, labels = inputs
 293       labels = tf.clip_by_value(labels, 0, model.num_classes)
 294 
 295       # Get descriptor predictions.
 296       blocks = {}
 297       prelogits = model.backbone(
 298           images, intermediates_dict=blocks, training=False)
 299       logits = model.desc_classification(prelogits, training=False)
 300       softmax_probabilities = tf.keras.layers.Softmax()(logits)
 301 
 302       validation_loss = loss_object(labels, logits)
 303       desc_validation_loss.update_state(validation_loss)
 304       desc_validation_accuracy.update_state(labels, softmax_probabilities)
 305 
 306       # Get attention predictions.
 307       block3 = blocks['block3']  # pytype: disable=key-error
 308       prelogits, _, _ = model.attention(block3, training=False)
 309 
 310       logits = model.attn_classification(prelogits, training=False)
 311       softmax_probabilities = tf.keras.layers.Softmax()(logits)
 312 
 313       validation_loss = loss_object(labels, logits)
 314       attn_validation_loss.update_state(validation_loss)
 315       attn_validation_accuracy.update_state(labels, softmax_probabilities)
 316 
 317       return desc_validation_accuracy.result(), attn_validation_accuracy.result(
 318       )
 319 
 320     # `run` replicates the provided computation and runs it
 321     # with the distributed input.
 322     @tf.function
 323     def distributed_train_step(dataset_inputs):
 324       """Get the actual losses."""
 325       # Each (desc, attn) is a list of 3 losses - crossentropy, reg, total.
 326       desc_per_replica_loss, attn_per_replica_loss = (
 327           strategy.run(train_step, args=(dataset_inputs,)))
 328 
 329       # Reduce over the replicas.
 330       desc_global_loss = strategy.reduce(
 331           tf.distribute.ReduceOp.SUM, desc_per_replica_loss, axis=None)
 332       attn_global_loss = strategy.reduce(
 333           tf.distribute.ReduceOp.SUM, attn_per_replica_loss, axis=None)
 334 
 335       return desc_global_loss, attn_global_loss
 336 
 337     @tf.function
 338     def distributed_validation_step(dataset_inputs):
 339       return strategy.run(validation_step, args=(dataset_inputs,))
 340 
 341     # ------------------------------------------------------------
 342     # *** TRAIN LOOP ***
 343     with summary_writer.as_default():
 344       with tf.summary.record_if(
 345           tf.math.equal(0, optimizer.iterations % report_interval)):
 346 
 347         global_step_value = optimizer.iterations.numpy()
 348         while global_step_value < max_iters:
 349 
 350           # input_batch : images(b, h, w, c), labels(b,).
 351           try:
 352             input_batch = train_iterator.get_next()
 353           except tf.errors.OutOfRangeError:
 354             # Break if we run out of data in the dataset.
 355             logging.info('Stopping training at global step %d, no more data',
 356                          global_step_value)
 357             break
 358 
 359           # Set learning rate for optimizer to use.
 360           global_step = optimizer.iterations
 361           global_step_value = global_step.numpy()
 362 
 363           learning_rate = _learning_rate_schedule(global_step_value, max_iters,
 364                                                   initial_lr)
 365           optimizer.learning_rate = learning_rate
 366           tf.summary.scalar(
 367               'learning_rate', optimizer.learning_rate, step=global_step)
 368 
 369           # Run the training step over num_gpu gpus.
 370           desc_dist_loss, attn_dist_loss = distributed_train_step(input_batch)
 371 
 372           # Log losses and accuracies to tensorboard.
 373           tf.summary.scalar(
 374               'loss/desc/crossentropy', desc_dist_loss, step=global_step)
 375           tf.summary.scalar(
 376               'loss/attn/crossentropy', attn_dist_loss, step=global_step)
 377           tf.summary.scalar(
 378               'train_accuracy/desc',
 379               desc_train_accuracy.result(),
 380               step=global_step)
 381           tf.summary.scalar(
 382               'train_accuracy/attn',
 383               attn_train_accuracy.result(),
 384               step=global_step)
 385 
 386           # Print to console if running locally.
 387           if FLAGS.debug:
 388             if global_step_value % report_interval == 0:
 389               print(global_step.numpy())
 390               print('desc:', desc_dist_loss.numpy())
 391               print('attn:', attn_dist_loss.numpy())
 392 
 393           # Validate once in {eval_interval*n, n \in N} steps.
 394           if global_step_value % eval_interval == 0:
 395             for i in range(num_eval):
 396               try:
 397                 validation_batch = validation_iterator.get_next()
 398                 desc_validation_result, attn_validation_result = (
 399                     distributed_validation_step(validation_batch))
 400               except tf.errors.OutOfRangeError:
 401                 logging.info('Stopping eval at batch %d, no more data', i)
 402                 break
 403 
 404             # Log validation results to tensorboard.
 405             tf.summary.scalar(
 406                 'validation/desc', desc_validation_result, step=global_step)
 407             tf.summary.scalar(
 408                 'validation/attn', attn_validation_result, step=global_step)
 409 
 410             logging.info('\nValidation(%f)\n', global_step_value)
 411             logging.info(': desc: %f\n', desc_validation_result.numpy())
 412             logging.info(': attn: %f\n', attn_validation_result.numpy())
 413             # Print to console.
 414             if FLAGS.debug:
 415               print('Validation: desc:', desc_validation_result.numpy())
 416               print('          : attn:', attn_validation_result.numpy())
 417 
 418           # Save checkpoint once (each save_interval*n, n \in N) steps.
 419           if global_step_value % save_interval == 0:
 420             save_path = manager.save()
 421             logging.info('Saved({global_step_value}) at %s', save_path)
 422 
 423             file_path = '%s/delf_weights' % FLAGS.logdir
 424             model.save_weights(file_path, save_format='tf')
 425             logging.info('Saved weights({global_step_value}) at %s', file_path)
 426 
 427           # Reset metrics for next step.
 428           desc_train_accuracy.reset_states()
 429           attn_train_accuracy.reset_states()
 430           desc_validation_loss.reset_states()
 431           attn_validation_loss.reset_states()
 432           desc_validation_accuracy.reset_states()
 433           attn_validation_accuracy.reset_states()
 434 
 435           if global_step.numpy() > max_iters:
 436             break
 437 
 438     logging.info('Finished training for %d steps.', max_iters)
 439 
 440 
 441 if __name__ == '__main__':
 442   app.run(main)
