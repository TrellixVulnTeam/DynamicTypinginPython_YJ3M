  57 def NormalizeImages(images, pixel_value_scale=0.5, pixel_value_offset=0.5):
  58   """Normalize pixel values in image.
  59 
  60   Output is computed as
  61   normalized_images = (images - pixel_value_offset) / pixel_value_scale.
  62 
  63   Args:
  64     images: `Tensor`, images to normalize.
  65     pixel_value_scale: float, scale.
  66     pixel_value_offset: float, offset.
  67 
  68   Returns:
  69     normalized_images: `Tensor`, normalized images.
  70   """
  71   images = tf.cast(images, tf.float32)
  72   normalized_images = tf.math.divide(
  73       tf.subtract(images, pixel_value_offset), pixel_value_scale)
  74   return normalized_images

  77 def _ImageNetCrop(image):
  78   """Imagenet-style crop with random bbox and aspect ratio.
  79 
  80   Args:
  81     image: a `Tensor`, image to crop.
  82 
  83   Returns:
  84     cropped_image: `Tensor`, cropped image.
  85   """
  86 
  87   params = _DataAugmentationParams()
  88   bbox = tf.constant([0.0, 0.0, 1.0, 1.0], dtype=tf.float32, shape=[1, 1, 4])
  89   (bbox_begin, bbox_size, _) = tf.image.sample_distorted_bounding_box(
  90       tf.shape(image),
  91       bounding_boxes=bbox,
  92       min_object_covered=params.min_object_covered,
  93       aspect_ratio_range=(params.aspect_ratio_range_min,
  94                           params.aspect_ratio_range_max),
  95       area_range=(params.area_range_min, params.area_range_max),
  96       max_attempts=params.max_attempts,
  97       use_image_if_no_bounding_boxes=True)
  98   cropped_image = tf.slice(image, bbox_begin, bbox_size)
  99   cropped_image.set_shape([None, None, 3])
 100 
 101   cropped_image = tf.image.resize(
 102       cropped_image, [params.input_rows, params.input_cols], method='area')
 103   if params.random_reflection:
 104     cropped_image = tf.image.random_flip_left_right(cropped_image)
 105 
 106   return cropped_image

 109 def _ParseFunction(example, name_to_features, image_size, augmentation):
 110   """Parse a single TFExample to get the image and label and process the image.
 111 
 112   Args:
 113     example: a `TFExample`.
 114     name_to_features: a `dict`. The mapping from feature names to its type.
 115     image_size: an `int`. The image size for the decoded image, on each side.
 116     augmentation: a `boolean`. True if the image will be augmented.
 117 
 118   Returns:
 119     image: a `Tensor`. The processed image.
 120     label: a `Tensor`. The ground-truth label.
 121   """
 122   parsed_example = tf.io.parse_single_example(example, name_to_features)
 123   # Parse to get image.
 124   image = parsed_example['image/encoded']
 125   image = tf.io.decode_jpeg(image)
 126   if augmentation:
 127     image = _ImageNetCrop(image)
 128   else:
 129     image = tf.image.resize(image, [image_size, image_size])
 130     image.set_shape([image_size, image_size, 3])
 131   # Parse to get label.
 132   label = parsed_example['image/class/label']
 133   return image, label

 136 def CreateDataset(file_pattern,
 137                   image_size=321,
 138                   batch_size=32,
 139                   augmentation=False,
 140                   seed=0):
 141   """Creates a dataset.
 142 
 143   Args:
 144     file_pattern: str, file pattern of the dataset files.
 145     image_size: int, image size.
 146     batch_size: int, batch size.
 147     augmentation: bool, whether to apply augmentation.
 148     seed: int, seed for shuffling the dataset.
 149 
 150   Returns:
 151      tf.data.TFRecordDataset.
 152   """
 153 
 154   filenames = tf.io.gfile.glob(file_pattern)
 155 
 156   dataset = tf.data.TFRecordDataset(filenames)
 157   dataset = dataset.repeat().shuffle(buffer_size=100, seed=seed)
 158 
 159   # Create a description of the features.
 160   feature_description = {
 161       'image/height': tf.io.FixedLenFeature([], tf.int64, default_value=0),
 162       'image/width': tf.io.FixedLenFeature([], tf.int64, default_value=0),
 163       'image/channels': tf.io.FixedLenFeature([], tf.int64, default_value=0),
 164       'image/format': tf.io.FixedLenFeature([], tf.string, default_value=''),
 165       'image/filename': tf.io.FixedLenFeature([], tf.string, default_value=''),
 166       'image/encoded': tf.io.FixedLenFeature([], tf.string, default_value=''),
 167       'image/class/label': tf.io.FixedLenFeature([], tf.int64, default_value=0),
 168   }
 169 
 170   customized_parse_func = functools.partial(
 171       _ParseFunction,
 172       name_to_features=feature_description,
 173       image_size=image_size,
 174       augmentation=augmentation)
 175   dataset = dataset.map(customized_parse_func)
 176   dataset = dataset.batch(batch_size)
 177 
 178   return dataset
