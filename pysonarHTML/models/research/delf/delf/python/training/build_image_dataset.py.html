  16 """Converts landmark image data to TFRecords file format with Example protos.
  17 
  18 The image data set is expected to reside in JPEG files ends up with '.jpg'.
  19 
  20 This script converts the training and testing data into
  21 a sharded data set consisting of TFRecord files
  22   train_directory/train-00000-of-00128
  23   train_directory/train-00001-of-00128
  24   ...
  25   train_directory/train-00127-of-00128
  26 and
  27   test_directory/test-00000-of-00128
  28   test_directory/test-00001-of-00128
  29   ...
  30   test_directory/test-00127-of-00128
  31 where we have selected 128 shards for both data sets. Each record
  32 within the TFRecord file is a serialized Example proto. The Example proto
  33 contains the following fields:
  34   image/encoded: string containing JPEG encoded image in RGB colorspace
  35   image/height: integer, image height in pixels
  36   image/width: integer, image width in pixels
  37   image/colorspace: string, specifying the colorspace, always 'RGB'
  38   image/channels: integer, specifying the number of channels, always 3
  39   image/format: string, specifying the format, always 'JPEG'
  40   image/filename: string, the unique id of the image file
  41             e.g. '97c0a12e07ae8dd5' or '650c989dd3493748'
  42 Furthermore, if the data set type is training, it would contain one more field:
  43   image/class/label: integer, the landmark_id from the input training csv file.
  44 """

  98 def _get_all_image_files_and_labels(name, csv_path, image_dir):
  99   """Process input and get the image file paths, image ids and the labels.
 100 
 101   Args:
 102     name: 'train' or 'test'.
 103     csv_path: path to the Google-landmark Dataset csv Data Sources files.
 104     image_dir: directory that stores downloaded images.
 105   Returns:
 106     image_paths: the paths to all images in the image_dir.
 107     file_ids: the unique ids of images.
 108     labels: the landmark id of all images. When name='test', the returned labels
 109       will be an empty list.
 110   Raises:
 111     ValueError: if input name is not supported.
 112   """
 113   image_paths = tf.io.gfile.glob(os.path.join(image_dir, '*.jpg'))
 114   file_ids = [os.path.basename(os.path.normpath(f))[:-4] for f in image_paths]
 115   if name == _TRAIN_SPLIT:
 116     with tf.io.gfile.GFile(csv_path, 'rb') as csv_file:
 117       df = pd.read_csv(csv_file)
 118     df = df.set_index('id')
 119     labels = [int(df.loc[fid]['landmark_id']) for fid in file_ids]
 120   elif name == _TEST_SPLIT:
 121     labels = []
 122   else:
 123     raise ValueError('Unsupported dataset split name: %s' % name)
 124   return image_paths, file_ids, labels

 127 def _get_clean_train_image_files_and_labels(csv_path, image_dir):
 128   """Get image file paths, image ids and  labels for the clean training split.
 129 
 130   Args:
 131     csv_path: path to the Google-landmark Dataset v2 CSV Data Sources files
 132               of the clean train dataset. Assumes CSV header landmark_id;images.
 133     image_dir: directory that stores downloaded images.
 134 
 135   Returns:
 136     image_paths: the paths to all images in the image_dir.
 137     file_ids: the unique ids of images.
 138     labels: the landmark id of all images.
 139     relabeling: relabeling rules created to replace actual labels with
 140                 a continuous set of labels.
 141   """
 142   # Load the content of the CSV file (landmark_id/label -> images).
 143   with tf.io.gfile.GFile(csv_path, 'rb') as csv_file:
 144     df = pd.read_csv(csv_file)
 145 
 146   # Create the dictionary (key = image_id, value = {label, file_id}).
 147   images = {}
 148   for _, row in df.iterrows():
 149     label = row['landmark_id']
 150     for file_id in row['images'].split(' '):
 151       images[file_id] = {}
 152       images[file_id]['label'] = label
 153       images[file_id]['file_id'] = file_id
 154 
 155   # Add the full image path to the dictionary of images.
 156   image_paths = tf.io.gfile.glob(os.path.join(image_dir, '*.jpg'))
 157   for image_path in image_paths:
 158     file_id = os.path.basename(os.path.normpath(image_path))[:-4]
 159     if file_id in images:
 160       images[file_id]['image_path'] = image_path
 161 
 162   # Explode the dictionary into lists (1 per image attribute).
 163   image_paths = []
 164   file_ids = []
 165   labels = []
 166   for _, value in images.items():
 167     image_paths.append(value['image_path'])
 168     file_ids.append(value['file_id'])
 169     labels.append(value['label'])
 170 
 171   # Relabel image labels to contiguous values.
 172   unique_labels = sorted(set(labels))
 173   relabeling = {label: index for index, label in enumerate(unique_labels)}
 174   new_labels = [relabeling[label] for label in labels]
 175   return image_paths, file_ids, new_labels, relabeling

 178 def _process_image(filename):
 179   """Process a single image file.
 180 
 181   Args:
 182     filename: string, path to an image file e.g., '/path/to/example.jpg'.
 183 
 184   Returns:
 185     image_buffer: string, JPEG encoding of RGB image.
 186     height: integer, image height in pixels.
 187     width: integer, image width in pixels.
 188   Raises:
 189     ValueError: if parsed image has wrong number of dimensions or channels.
 190   """
 191   # Read the image file.
 192   with tf.io.gfile.GFile(filename, 'rb') as f:
 193     image_data = f.read()
 194 
 195   # Decode the RGB JPEG.
 196   image = tf.io.decode_jpeg(image_data, channels=3)
 197 
 198   # Check that image converted to RGB
 199   if len(image.shape) != 3:
 200     raise ValueError('The parsed image number of dimensions is not 3 but %d' %
 201                      (image.shape))
 202   height = image.shape[0]
 203   width = image.shape[1]
 204   if image.shape[2] != 3:
 205     raise ValueError('The parsed image channels is not 3 but %d' %
 206                      (image.shape[2]))
 207 
 208   return image_data, height, width

 211 def _int64_feature(value):
 212   """Returns an int64_list from a bool / enum / int / uint."""
 213   return tf.train.Feature(int64_list=tf.train.Int64List(value=[value]))
 214 
 215 
 216 def _bytes_feature(value):
 217   """Returns a bytes_list from a string / byte."""
 218   return tf.train.Feature(bytes_list=tf.train.BytesList(value=[value]))

 221 def _convert_to_example(file_id, image_buffer, height, width, label=None):
 222   """Build an Example proto for the given inputs.
 223 
 224   Args:
 225     file_id: string, unique id of an image file, e.g., '97c0a12e07ae8dd5'.
 226     image_buffer: string, JPEG encoding of RGB image.
 227     height: integer, image height in pixels.
 228     width: integer, image width in pixels.
 229     label: integer, the landmark id and prediction label.
 230 
 231   Returns:
 232     Example proto.
 233   """
 234   colorspace = 'RGB'
 235   channels = 3
 236   image_format = 'JPEG'
 237   features = {
 238       'image/height': _int64_feature(height),
 239       'image/width': _int64_feature(width),
 240       'image/colorspace': _bytes_feature(colorspace.encode('utf-8')),
 241       'image/channels': _int64_feature(channels),
 242       'image/format': _bytes_feature(image_format.encode('utf-8')),
 243       'image/id': _bytes_feature(file_id.encode('utf-8')),
 244       'image/encoded': _bytes_feature(image_buffer)
 245   }
 246   if label is not None:
 247     features['image/class/label'] = _int64_feature(label)
 248   example = tf.train.Example(features=tf.train.Features(feature=features))
 249 
 250   return example

 253 def _write_tfrecord(output_prefix, image_paths, file_ids, labels):
 254   """Read image files and write image and label data into TFRecord files.
 255 
 256   Args:
 257     output_prefix: string, the prefix of output files, e.g. 'train'.
 258     image_paths: list of strings, the paths to images to be converted.
 259     file_ids: list of strings, the image unique ids.
 260     labels: list of integers, the landmark ids of images. It is an empty list
 261       when output_prefix='test'.
 262 
 263   Raises:
 264     ValueError: if the length of input images, ids and labels don't match
 265   """
 266   if output_prefix == _TEST_SPLIT:
 267     labels = [None] * len(image_paths)
 268   if not len(image_paths) == len(file_ids) == len(labels):
 269     raise ValueError('length of image_paths, file_ids, labels shoud be the' +
 270                      ' same. But they are %d, %d, %d, respectively' %
 271                      (len(image_paths), len(file_ids), len(labels)))
 272 
 273   spacing = np.linspace(0, len(image_paths), FLAGS.num_shards + 1, dtype=np.int)
 274 
 275   for shard in range(FLAGS.num_shards):
 276     output_file = os.path.join(
 277         FLAGS.output_directory,
 278         '%s-%.5d-of-%.5d' % (output_prefix, shard, FLAGS.num_shards))
 279     writer = tf.io.TFRecordWriter(output_file)
 280     print('Processing shard ', shard, ' and writing file ', output_file)
 281     for i in range(spacing[shard], spacing[shard + 1]):
 282       image_buffer, height, width = _process_image(image_paths[i])
 283       example = _convert_to_example(file_ids[i], image_buffer, height, width,
 284                                     labels[i])
 285       writer.write(example.SerializeToString())
 286     writer.close()

 289 def _write_relabeling_rules(relabeling_rules):
 290   """Write to a file the relabeling rules when the clean train dataset is used.
 291 
 292   Args:
 293     relabeling_rules: dictionary of relabeling rules applied when the clean
 294       train dataset is used (key = old_label, value = new_label).
 295   """
 296   relabeling_file_name = os.path.join(FLAGS.output_directory,
 297                                       'relabeling.csv')
 298   with tf.io.gfile.GFile(relabeling_file_name, 'w') as relabeling_file:
 299     csv_writer = csv.writer(relabeling_file, delimiter=',')
 300     csv_writer.writerow(['new_label', 'old_label'])
 301     for old_label, new_label in relabeling_rules.items():
 302       csv_writer.writerow([new_label, old_label])

 305 def _build_train_and_validation_splits(image_paths, file_ids, labels,
 306                                        validation_split_size, seed):
 307   """Create TRAIN and VALIDATION splits containg all labels in equal proportion.
 308 
 309   Args:
 310     image_paths: list of paths to the image files in the train dataset.
 311     file_ids: list of image file ids in the train dataset.
 312     labels: list of image labels in the train dataset.
 313     validation_split_size: size of the VALIDATION split as a ratio of the train
 314       dataset.
 315     seed: seed to use for shuffling the dataset for reproducibility purposes.
 316 
 317   Returns:
 318     splits : tuple containing the TRAIN and VALIDATION splits.
 319   Raises:
 320     ValueError: if the image attributes arrays don't all have the same length,
 321                 which makes the shuffling impossible.
 322   """
 323   # Ensure all image attribute arrays have the same length.
 324   total_images = len(file_ids)
 325   if not (len(image_paths) == total_images and len(labels) == total_images):
 326     raise ValueError('Inconsistencies between number of file_ids (%d), number '
 327                      'of image_paths (%d) and number of labels (%d). Cannot'
 328                      'shuffle the train dataset.'% (total_images,
 329                                                     len(image_paths),
 330                                                     len(labels)))
 331 
 332   # Stack all image attributes arrays in a single 2D array of dimensions
 333   # (3, number of images) and group by label the indices of datapoins in the
 334   # image attributes arrays. Explicitly convert label types from 'int' to 'str'
 335   # to avoid implicit conversion during stacking with image_paths and file_ids
 336   # which are 'str'.
 337   labels_str = [str(label) for label in labels]
 338   image_attrs = np.stack((image_paths, file_ids, labels_str))
 339   image_attrs_idx_by_label = {}
 340   for index, label in enumerate(labels):
 341     if label not in image_attrs_idx_by_label:
 342       image_attrs_idx_by_label[label] = []
 343     image_attrs_idx_by_label[label].append(index)
 344 
 345   # Create subsets of image attributes by label, shuffle them separately and
 346   # split each subset into TRAIN and VALIDATION splits based on the size of the
 347   # validation split.
 348   splits = {
 349       _VALIDATION_SPLIT: [],
 350       _TRAIN_SPLIT: []
 351   }
 352   rs = np.random.RandomState(np.random.MT19937(np.random.SeedSequence(seed)))
 353   for label, indexes in image_attrs_idx_by_label.items():
 354     # Create the subset for the current label.
 355     image_attrs_label = image_attrs[:, indexes]
 356     images_per_label = image_attrs_label.shape[1]
 357     # Shuffle the current label subset.
 358     columns_indices = np.arange(images_per_label)
 359     rs.shuffle(columns_indices)
 360     image_attrs_label = image_attrs_label[:, columns_indices]
 361     # Split the current label subset into TRAIN and VALIDATION splits and add
 362     # each split to the list of all splits.
 363     cutoff_idx = max(1, int(validation_split_size * images_per_label))
 364     splits[_VALIDATION_SPLIT].append(image_attrs_label[:, 0 : cutoff_idx])
 365     splits[_TRAIN_SPLIT].append(image_attrs_label[:, cutoff_idx : ])
 366 
 367   validation_split = np.concatenate(splits[_VALIDATION_SPLIT], axis=1)
 368   train_split = np.concatenate(splits[_TRAIN_SPLIT], axis=1)
 369 
 370   # Unstack the image attribute arrays in the TRAIN and VALIDATION splits and
 371   # convert them back to lists. Convert labels back to 'int' from 'str'
 372   # following the explicit type change from 'str' to 'int' for stacking.
 373   return (
 374       {
 375           _IMAGE_PATHS_KEY: validation_split[0, :].tolist(),
 376           _FILE_IDS_KEY: validation_split[1, :].tolist(),
 377           _LABELS_KEY: [int(label) for label in validation_split[2, :].tolist()]
 378       }, {
 379           _IMAGE_PATHS_KEY: train_split[0, :].tolist(),
 380           _FILE_IDS_KEY: train_split[1, :].tolist(),
 381           _LABELS_KEY: [int(label) for label in train_split[2, :].tolist()]
 382       })
 383

 385 def _build_train_tfrecord_dataset(csv_path,
 386                                   clean_csv_path,
 387                                   image_dir,
 388                                   generate_train_validation_splits,
 389                                   validation_split_size,
 390                                   seed):
 391   """Build a TFRecord dataset for the train split.
 392 
 393   Args:
 394     csv_path: path to the train Google-landmark Dataset csv Data Sources files.
 395     clean_csv_path: path to the Google-landmark Dataset v2 CSV Data Sources
 396                     files of the clean train dataset.
 397     image_dir: directory that stores downloaded images.
 398     generate_train_validation_splits: whether to split the test dataset into
 399       TRAIN and VALIDATION splits.
 400     validation_split_size: size of the VALIDATION split as a ratio of the train
 401       dataset. Only used if 'generate_train_validation_splits' is True.
 402     seed: seed to use for shuffling the dataset for reproducibility purposes.
 403       Only used if 'generate_train_validation_splits' is True.
 404 
 405   Returns:
 406     Nothing. After the function call, sharded TFRecord files are materialized.
 407   Raises:
 408     ValueError: if the size of the VALIDATION split is outside (0,1) when TRAIN
 409                 and VALIDATION splits need to be generated.
 410   """
 411   # Make sure the size of the VALIDATION split is inside (0, 1) if we need to
 412   # generate the TRAIN and VALIDATION splits.
 413   if generate_train_validation_splits:
 414     if validation_split_size <= 0 or validation_split_size >= 1:
 415       raise ValueError('Invalid VALIDATION split size. Expected inside (0,1)'
 416                        'but received %f.' % validation_split_size)
 417 
 418   if clean_csv_path:
 419     # Load clean train images and labels and write the relabeling rules.
 420     (image_paths, file_ids, labels,
 421      relabeling_rules) = _get_clean_train_image_files_and_labels(clean_csv_path,
 422                                                                  image_dir)
 423     _write_relabeling_rules(relabeling_rules)
 424   else:
 425     # Load all train images.
 426     image_paths, file_ids, labels = _get_all_image_files_and_labels(
 427         _TRAIN_SPLIT, csv_path, image_dir)
 428 
 429   if generate_train_validation_splits:
 430     # Generate the TRAIN and VALIDATION splits and write them to TFRecord.
 431     validation_split, train_split = _build_train_and_validation_splits(
 432         image_paths, file_ids, labels, validation_split_size, seed)
 433     _write_tfrecord(_VALIDATION_SPLIT,
 434                     validation_split[_IMAGE_PATHS_KEY],
 435                     validation_split[_FILE_IDS_KEY],
 436                     validation_split[_LABELS_KEY])
 437     _write_tfrecord(_TRAIN_SPLIT,
 438                     train_split[_IMAGE_PATHS_KEY],
 439                     train_split[_FILE_IDS_KEY],
 440                     train_split[_LABELS_KEY])
 441   else:
 442     # Write to TFRecord a single split, TRAIN.
 443     _write_tfrecord(_TRAIN_SPLIT, image_paths, file_ids, labels)
 444

 446 def _build_test_tfrecord_dataset(csv_path, image_dir):
 447   """Build a TFRecord dataset for the 'test' split.
 448 
 449   Args:
 450     csv_path: path to the 'test' Google-landmark Dataset csv Data Sources files.
 451     image_dir: directory that stores downloaded images.
 452 
 453   Returns:
 454     Nothing. After the function call, sharded TFRecord files are materialized.
 455   """
 456   image_paths, file_ids, labels = _get_all_image_files_and_labels(
 457       _TEST_SPLIT, csv_path, image_dir)
 458   _write_tfrecord(_TEST_SPLIT, image_paths, file_ids, labels)
