  31 class _IdentityBlock(tf.keras.Model):
  32   """_IdentityBlock is the block that has no conv layer at shortcut.
  33 
  34   Args:
  35     kernel_size: the kernel size of middle conv layer at main path
  36     filters: list of integers, the filters of 3 conv layer at main path
  37     stage: integer, current stage label, used for generating layer names
  38     block: 'a','b'..., current block label, used for generating layer names
  39     data_format: data_format for the input ('channels_first' or
  40       'channels_last').
  41   """
  42 
  43   def __init__(self, kernel_size, filters, stage, block, data_format):
  44     super(_IdentityBlock, self).__init__(name='')
  45     filters1, filters2, filters3 = filters
  46 
  47     conv_name_base = 'res' + str(stage) + block + '_branch'
  48     bn_name_base = 'bn' + str(stage) + block + '_branch'
  49     bn_axis = 1 if data_format == 'channels_first' else 3
  50 
  51     self.conv2a = layers.Conv2D(
  52         filters1, (1, 1), name=conv_name_base + '2a', data_format=data_format)
  53     self.bn2a = layers.BatchNormalization(
  54         axis=bn_axis, name=bn_name_base + '2a')
  55 
  56     self.conv2b = layers.Conv2D(
  57         filters2,
  58         kernel_size,
  59         padding='same',
  60         data_format=data_format,
  61         name=conv_name_base + '2b')
  62     self.bn2b = layers.BatchNormalization(
  63         axis=bn_axis, name=bn_name_base + '2b')
  64 
  65     self.conv2c = layers.Conv2D(
  66         filters3, (1, 1), name=conv_name_base + '2c', data_format=data_format)
  67     self.bn2c = layers.BatchNormalization(
  68         axis=bn_axis, name=bn_name_base + '2c')
  69 
  70   def call(self, input_tensor, training=False):
  71     x = self.conv2a(input_tensor)
  72     x = self.bn2a(x, training=training)
  73     x = tf.nn.relu(x)
  74 
  75     x = self.conv2b(x)
  76     x = self.bn2b(x, training=training)
  77     x = tf.nn.relu(x)
  78 
  79     x = self.conv2c(x)
  80     x = self.bn2c(x, training=training)
  81 
  82     x += input_tensor
  83     return tf.nn.relu(x)

  86 class _ConvBlock(tf.keras.Model):
  87   """_ConvBlock is the block that has a conv layer at shortcut.
  88 
  89   Args:
  90       kernel_size: the kernel size of middle conv layer at main path
  91       filters: list of integers, the filters of 3 conv layer at main path
  92       stage: integer, current stage label, used for generating layer names
  93       block: 'a','b'..., current block label, used for generating layer names
  94       data_format: data_format for the input ('channels_first' or
  95         'channels_last').
  96       strides: strides for the convolution. Note that from stage 3, the first
  97         conv layer at main path is with strides=(2,2), and the shortcut should
  98         have strides=(2,2) as well.
  99   """
 100 
 101   def __init__(self,
 102                kernel_size,
 103                filters,
 104                stage,
 105                block,
 106                data_format,
 107                strides=(2, 2)):
 108     super(_ConvBlock, self).__init__(name='')
 109     filters1, filters2, filters3 = filters
 110 
 111     conv_name_base = 'res' + str(stage) + block + '_branch'
 112     bn_name_base = 'bn' + str(stage) + block + '_branch'
 113     bn_axis = 1 if data_format == 'channels_first' else 3
 114 
 115     self.conv2a = layers.Conv2D(
 116         filters1, (1, 1),
 117         strides=strides,
 118         name=conv_name_base + '2a',
 119         data_format=data_format)
 120     self.bn2a = layers.BatchNormalization(
 121         axis=bn_axis, name=bn_name_base + '2a')
 122 
 123     self.conv2b = layers.Conv2D(
 124         filters2,
 125         kernel_size,
 126         padding='same',
 127         name=conv_name_base + '2b',
 128         data_format=data_format)
 129     self.bn2b = layers.BatchNormalization(
 130         axis=bn_axis, name=bn_name_base + '2b')
 131 
 132     self.conv2c = layers.Conv2D(
 133         filters3, (1, 1), name=conv_name_base + '2c', data_format=data_format)
 134     self.bn2c = layers.BatchNormalization(
 135         axis=bn_axis, name=bn_name_base + '2c')
 136 
 137     self.conv_shortcut = layers.Conv2D(
 138         filters3, (1, 1),
 139         strides=strides,
 140         name=conv_name_base + '1',
 141         data_format=data_format)
 142     self.bn_shortcut = layers.BatchNormalization(
 143         axis=bn_axis, name=bn_name_base + '1')
 144 
 145   def call(self, input_tensor, training=False):
 146     x = self.conv2a(input_tensor)
 147     x = self.bn2a(x, training=training)
 148     x = tf.nn.relu(x)
 149 
 150     x = self.conv2b(x)
 151     x = self.bn2b(x, training=training)
 152     x = tf.nn.relu(x)
 153 
 154     x = self.conv2c(x)
 155     x = self.bn2c(x, training=training)
 156 
 157     shortcut = self.conv_shortcut(input_tensor)
 158     shortcut = self.bn_shortcut(shortcut, training=training)
 159 
 160     x += shortcut
 161     return tf.nn.relu(x)

 165 class ResNet50(tf.keras.Model):
 166   """Instantiates the ResNet50 architecture.
 167 
 168   Args:
 169     data_format: format for the image. Either 'channels_first' or
 170       'channels_last'.  'channels_first' is typically faster on GPUs while
 171       'channels_last' is typically faster on CPUs. See
 172       https://www.tensorflow.org/performance/performance_guide#data_formats
 173     name: Prefix applied to names of variables created in the model.
 174     include_top: whether to include the fully-connected layer at the top of the
 175       network.
 176     pooling: Optional pooling mode for feature extraction when `include_top` is
 177       False. 'None' means that the output of the model will be the 4D tensor
 178       output of the last convolutional layer. 'avg' means that global average
 179       pooling will be applied to the output of the last convolutional layer, and
 180       thus the output of the model will be a 2D tensor. 'max' means that global
 181       max pooling will be applied.
 182     block3_strides: whether to add a stride of 2 to block3 to make it compatible
 183       with tf.slim ResNet implementation.
 184     average_pooling: whether to do average pooling of block4 features before
 185       global pooling.
 186     classes: optional number of classes to classify images into, only to be
 187       specified if `include_top` is True.
 188 
 189   Raises:
 190       ValueError: in case of invalid argument for data_format.
 191   """
 192 
 193   def __init__(self,
 194                data_format,
 195                name='',
 196                include_top=True,
 197                pooling=None,
 198                block3_strides=False,
 199                average_pooling=True,
 200                classes=1000):
 201     super(ResNet50, self).__init__(name=name)
 202 
 203     valid_channel_values = ('channels_first', 'channels_last')
 204     if data_format not in valid_channel_values:
 205       raise ValueError('Unknown data_format: %s. Valid values: %s' %
 206                        (data_format, valid_channel_values))
 207     self.include_top = include_top
 208     self.block3_strides = block3_strides
 209     self.average_pooling = average_pooling
 210     self.pooling = pooling
 211 
 212     def conv_block(filters, stage, block, strides=(2, 2)):
 213       return _ConvBlock(
 214           3,
 215           filters,
 216           stage=stage,
 217           block=block,
 218           data_format=data_format,
 219           strides=strides)
 220 
 221     def id_block(filters, stage, block):
 222       return _IdentityBlock(
 223           3, filters, stage=stage, block=block, data_format=data_format)
 224 
 225     self.conv1 = layers.Conv2D(
 226         64, (7, 7),
 227         strides=(2, 2),
 228         data_format=data_format,
 229         padding='same',
 230         name='conv1')
 231     bn_axis = 1 if data_format == 'channels_first' else 3
 232     self.bn_conv1 = layers.BatchNormalization(axis=bn_axis, name='bn_conv1')
 233     self.max_pool = layers.MaxPooling2D((3, 3),
 234                                         strides=(2, 2),
 235                                         data_format=data_format)
 236 
 237     self.l2a = conv_block([64, 64, 256], stage=2, block='a', strides=(1, 1))
 238     self.l2b = id_block([64, 64, 256], stage=2, block='b')
 239     self.l2c = id_block([64, 64, 256], stage=2, block='c')
 240 
 241     self.l3a = conv_block([128, 128, 512], stage=3, block='a')
 242     self.l3b = id_block([128, 128, 512], stage=3, block='b')
 243     self.l3c = id_block([128, 128, 512], stage=3, block='c')
 244     self.l3d = id_block([128, 128, 512], stage=3, block='d')
 245 
 246     self.l4a = conv_block([256, 256, 1024], stage=4, block='a')
 247     self.l4b = id_block([256, 256, 1024], stage=4, block='b')
 248     self.l4c = id_block([256, 256, 1024], stage=4, block='c')
 249     self.l4d = id_block([256, 256, 1024], stage=4, block='d')
 250     self.l4e = id_block([256, 256, 1024], stage=4, block='e')
 251     self.l4f = id_block([256, 256, 1024], stage=4, block='f')
 252 
 253     # Striding layer that can be used on top of block3 to produce feature maps
 254     # with the same resolution as the TF-Slim implementation.
 255     if self.block3_strides:
 256       self.subsampling_layer = layers.MaxPooling2D((1, 1),
 257                                                    strides=(2, 2),
 258                                                    data_format=data_format)
 259       self.l5a = conv_block([512, 512, 2048],
 260                             stage=5,
 261                             block='a',
 262                             strides=(1, 1))
 263     else:
 264       self.l5a = conv_block([512, 512, 2048], stage=5, block='a')
 265     self.l5b = id_block([512, 512, 2048], stage=5, block='b')
 266     self.l5c = id_block([512, 512, 2048], stage=5, block='c')
 267 
 268     self.avg_pool = layers.AveragePooling2D((7, 7),
 269                                             strides=(7, 7),
 270                                             data_format=data_format)
 271 
 272     if self.include_top:
 273       self.flatten = layers.Flatten()
 274       self.fc1000 = layers.Dense(classes, name='fc1000')
 275     else:
 276       reduction_indices = [1, 2] if data_format == 'channels_last' else [2, 3]
 277       reduction_indices = tf.constant(reduction_indices)
 278       if pooling == 'avg':
 279         self.global_pooling = functools.partial(
 280             tf.reduce_mean, axis=reduction_indices, keepdims=False)
 281       elif pooling == 'max':
 282         self.global_pooling = functools.partial(
 283             tf.reduce_max, axis=reduction_indices, keepdims=False)
 284       else:
 285         self.global_pooling = None
 286 
 287   def call(self, inputs, training=True, intermediates_dict=None):
 288     """Call the ResNet50 model.
 289 
 290     Args:
 291       inputs: Images to compute features for.
 292       training: Whether model is in training phase.
 293       intermediates_dict: `None` or dictionary. If not None, accumulate feature
 294         maps from intermediate blocks into the dictionary. ""
 295 
 296     Returns:
 297       Tensor with featuremap.
 298     """
 299 
 300     x = self.conv1(inputs)
 301     x = self.bn_conv1(x, training=training)
 302     x = tf.nn.relu(x)
 303     if intermediates_dict is not None:
 304       intermediates_dict['block0'] = x
 305 
 306     x = self.max_pool(x)
 307     if intermediates_dict is not None:
 308       intermediates_dict['block0mp'] = x
 309 
 310     # Block 1 (equivalent to "conv2" in Resnet paper).
 311     x = self.l2a(x, training=training)
 312     x = self.l2b(x, training=training)
 313     x = self.l2c(x, training=training)
 314     if intermediates_dict is not None:
 315       intermediates_dict['block1'] = x
 316 
 317     # Block 2 (equivalent to "conv3" in Resnet paper).
 318     x = self.l3a(x, training=training)
 319     x = self.l3b(x, training=training)
 320     x = self.l3c(x, training=training)
 321     x = self.l3d(x, training=training)
 322     if intermediates_dict is not None:
 323       intermediates_dict['block2'] = x
 324 
 325     # Block 3 (equivalent to "conv4" in Resnet paper).
 326     x = self.l4a(x, training=training)
 327     x = self.l4b(x, training=training)
 328     x = self.l4c(x, training=training)
 329     x = self.l4d(x, training=training)
 330     x = self.l4e(x, training=training)
 331     x = self.l4f(x, training=training)
 332 
 333     if self.block3_strides:
 334       x = self.subsampling_layer(x)
 335       if intermediates_dict is not None:
 336         intermediates_dict['block3'] = x
 337     else:
 338       if intermediates_dict is not None:
 339         intermediates_dict['block3'] = x
 340 
 341     x = self.l5a(x, training=training)
 342     x = self.l5b(x, training=training)
 343     x = self.l5c(x, training=training)
 344 
 345     if self.average_pooling:
 346       x = self.avg_pool(x)
 347       if intermediates_dict is not None:
 348         intermediates_dict['block4'] = x
 349     else:
 350       if intermediates_dict is not None:
 351         intermediates_dict['block4'] = x
 352 
 353     if self.include_top:
 354       return self.fc1000(self.flatten(x))
 355     elif self.global_pooling:
 356       return self.global_pooling(x)
 357     else:
 358       return x
