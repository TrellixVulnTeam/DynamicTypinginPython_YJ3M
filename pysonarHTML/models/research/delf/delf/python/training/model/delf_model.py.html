  22 import tensorflow as tf

  24 from delf.python.training.model import resnet50 as resnet

  26 layers = tf.keras.layers
  27 reg = tf.keras.regularizers
  28 
  29 _DECAY = 0.0001
  30 
  31 
  32 class AttentionModel(tf.keras.Model):
  33   """Instantiates attention model.
  34 
  35   Uses two [kernel_size x kernel_size] convolutions and softplus as activation
  36   to compute an attention map with the same resolution as the featuremap.
  37   Features l2-normalized and aggregated using attention probabilites as weights.
  38   """
  39 
  40   def __init__(self, kernel_size=1, decay=_DECAY, name='attention'):
  41     """Initialization of attention model.
  42 
  43     Args:
  44       kernel_size: int, kernel size of convolutions.
  45       decay: float, decay for l2 regularization of kernel weights.
  46       name: str, name to identify model.
  47     """
  48     super(AttentionModel, self).__init__(name=name)
  49 
  50     # First convolutional layer (called with relu activation).
  51     self.conv1 = layers.Conv2D(
  52         512,
  53         kernel_size,
  54         kernel_regularizer=reg.l2(decay),
  55         padding='same',
  56         name='attn_conv1')
  57     self.bn_conv1 = layers.BatchNormalization(axis=3, name='bn_conv1')
  58 
  59     # Second convolutional layer, with softplus activation.
  60     self.conv2 = layers.Conv2D(
  61         1,
  62         kernel_size,
  63         kernel_regularizer=reg.l2(decay),
  64         padding='same',
  65         name='attn_conv2')
  66     self.activation_layer = layers.Activation('softplus')
  67 
  68   def call(self, inputs, training=True):
  69     x = self.conv1(inputs)
  70     x = self.bn_conv1(x, training=training)
  71     x = tf.nn.relu(x)
  72 
  73     score = self.conv2(x)
  74     prob = self.activation_layer(score)
  75 
  76     # L2-normalize the featuremap before pooling.
  77     inputs = tf.nn.l2_normalize(inputs, axis=-1)
  78     feat = tf.reduce_mean(tf.multiply(inputs, prob), [1, 2], keepdims=False)
  79 
  80     return feat, prob, score
  81

  83 class Delf(tf.keras.Model):
  84   """Instantiates Keras DELF model using ResNet50 as backbone.
  85 
  86   This class implements the [DELF](https://arxiv.org/abs/1612.06321) model for
  87   extracting local features from images. The backbone is a ResNet50 network
  88   that extracts featuremaps from both conv_4 and conv_5 layers. Activations
  89   from conv_4 are used to compute an attention map of the same resolution.
  90   """
  91 
  92   def __init__(self, block3_strides=True, name='DELF'):
  93     """Initialization of DELF model.
  94 
  95     Args:
  96       block3_strides: bool, whether to add strides to the output of block3.
  97       name: str, name to identify model.
  98     """
  99     super(Delf, self).__init__(name=name)
 100 
 101     # Backbone using Keras ResNet50.
 102     self.backbone = resnet.ResNet50(
 103         'channels_last',
 104         name='backbone',
 105         include_top=False,
 106         pooling='avg',
 107         block3_strides=block3_strides,
 108         average_pooling=False)
 109 
 110     # Attention model.
 111     self.attention = AttentionModel(name='attention')
 112 
 113   # Define classifiers for training backbone and attention models.
 114   def init_classifiers(self, num_classes):
 115     self.num_classes = num_classes
 116     self.desc_classification = layers.Dense(
 117         num_classes, activation=None, kernel_regularizer=None, name='desc_fc')
 118 
 119     self.attn_classification = layers.Dense(
 120         num_classes, activation=None, kernel_regularizer=None, name='att_fc')
 121 
 122   # Weights to optimize for descriptor fine tuning.
 123   @property
 124   def desc_trainable_weights(self):
 125     return (self.backbone.trainable_weights +
 126             self.desc_classification.trainable_weights)
 127 
 128   # Weights to optimize for attention model training.
 129   @property
 130   def attn_trainable_weights(self):
 131     return (self.attention.trainable_weights +
 132             self.attn_classification.trainable_weights)
 133 
 134   def call(self, input_image, training=True):
 135     blocks = {'block3': None}
 136     self.backbone(input_image, intermediates_dict=blocks, training=training)
 137 
 138     features = blocks['block3']
 139     _, probs, _ = self.attention(features, training=training)
 140 
 141     return probs, features
