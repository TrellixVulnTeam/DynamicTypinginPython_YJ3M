  22 import tensorflow as tf

  24 from delf import feature_extractor
  25 from delf.python.training.datasets import googlelandmarks as gld
  26 from object_detection.core import box_list
  27 from object_detection.core import box_list_ops

  30 def ExtractLocalFeatures(image, image_scales, max_feature_num, abs_thres, iou,
  31                          attention_model_fn, stride_factor):
  32   """Extract local features for input image.
  33 
  34   Args:
  35     image: image tensor of type tf.uint8 with shape [h, w, channels].
  36     image_scales: 1D float tensor which contains float scales used for image
  37       pyramid construction.
  38     max_feature_num: int tensor denotes the maximum selected feature points.
  39     abs_thres: float tensor denotes the score threshold for feature selection.
  40     iou: float scalar denotes the iou threshold for NMS.
  41     attention_model_fn: model function. Follows the signature:
  42       * Args:
  43         * `images`: Image tensor which is re-scaled.
  44       * Returns:
  45         * `attention_prob`: attention map after the non-linearity.
  46         * `feature_map`: feature map after ResNet convolution.
  47     stride_factor: integer accounting for striding after block3.
  48 
  49   Returns:
  50     boxes: [N, 4] float tensor which denotes the selected receptive box. N is
  51       the number of final feature points which pass through keypoint selection
  52       and NMS steps.
  53     features: [N, depth] float tensor.
  54     feature_scales: [N] float tensor. It is the inverse of the input image
  55       scales such that larger image scales correspond to larger image regions,
  56       which is compatible with keypoints detected with other techniques, for
  57       example Congas.
  58     scores: [N, 1] float tensor denotes the attention score.
  59 
  60   """
  61   original_image_shape_float = tf.gather(
  62       tf.dtypes.cast(tf.shape(image), tf.float32), [0, 1])
  63 
  64   image_tensor = gld.NormalizeImages(
  65       image, pixel_value_offset=128.0, pixel_value_scale=128.0)
  66   image_tensor = tf.expand_dims(image_tensor, 0, name='image/expand_dims')
  67 
  68   # Hard code the feature depth and receptive field parameters for now.
  69   rf, stride, padding = [291.0, 16.0 * stride_factor, 145.0]
  70   feature_depth = 1024
  71 
  72   def _ProcessSingleScale(scale_index, boxes, features, scales, scores):
  73     """Resizes the image and run feature extraction and keypoint selection.
  74 
  75        This function will be passed into tf.while_loop() and be called
  76        repeatedly. The input boxes are collected from the previous iteration
  77        [0: scale_index -1]. We get the current scale by
  78        image_scales[scale_index], and run resize image, feature extraction and
  79        keypoint selection. Then we will get a new set of selected_boxes for
  80        current scale. In the end, we concat the previous boxes with current
  81        selected_boxes as the output.
  82     Args:
  83       scale_index: A valid index in the image_scales.
  84       boxes: Box tensor with the shape of [N, 4].
  85       features: Feature tensor with the shape of [N, depth].
  86       scales: Scale tensor with the shape of [N].
  87       scores: Attention score tensor with the shape of [N].
  88 
  89     Returns:
  90       scale_index: The next scale index for processing.
  91       boxes: Concatenated box tensor with the shape of [K, 4]. K >= N.
  92       features: Concatenated feature tensor with the shape of [K, depth].
  93       scales: Concatenated scale tensor with the shape of [K].
  94       scores: Concatenated score tensor with the shape of [K].
  95     """
  96     scale = tf.gather(image_scales, scale_index)
  97     new_image_size = tf.dtypes.cast(
  98         tf.round(original_image_shape_float * scale), tf.int32)
  99     resized_image = tf.image.resize(image_tensor, new_image_size)
 100 
 101     attention_prob, feature_map = attention_model_fn(resized_image)
 102     attention_prob = tf.squeeze(attention_prob, axis=[0])
 103     feature_map = tf.squeeze(feature_map, axis=[0])
 104 
 105     rf_boxes = feature_extractor.CalculateReceptiveBoxes(
 106         tf.shape(feature_map)[0],
 107         tf.shape(feature_map)[1], rf, stride, padding)
 108 
 109     # Re-project back to the original image space.
 110     rf_boxes = tf.divide(rf_boxes, scale)
 111     attention_prob = tf.reshape(attention_prob, [-1])
 112     feature_map = tf.reshape(feature_map, [-1, feature_depth])
 113 
 114     # Use attention score to select feature vectors.
 115     indices = tf.reshape(tf.where(attention_prob >= abs_thres), [-1])
 116     selected_boxes = tf.gather(rf_boxes, indices)
 117     selected_features = tf.gather(feature_map, indices)
 118     selected_scores = tf.gather(attention_prob, indices)
 119     selected_scales = tf.ones_like(selected_scores, tf.float32) / scale
 120 
 121     # Concat with the previous result from different scales.
 122     boxes = tf.concat([boxes, selected_boxes], 0)
 123     features = tf.concat([features, selected_features], 0)
 124     scales = tf.concat([scales, selected_scales], 0)
 125     scores = tf.concat([scores, selected_scores], 0)
 126 
 127     return scale_index + 1, boxes, features, scales, scores
 128 
 129   output_boxes = tf.zeros([0, 4], dtype=tf.float32)
 130   output_features = tf.zeros([0, feature_depth], dtype=tf.float32)
 131   output_scales = tf.zeros([0], dtype=tf.float32)
 132   output_scores = tf.zeros([0], dtype=tf.float32)
 133 
 134   # Process the first scale separately, the following scales will reuse the
 135   # graph variables.
 136   (_, output_boxes, output_features, output_scales,
 137    output_scores) = _ProcessSingleScale(0, output_boxes, output_features,
 138                                         output_scales, output_scores)
 139 
 140   i = tf.constant(1, dtype=tf.int32)
 141   num_scales = tf.shape(image_scales)[0]
 142   keep_going = lambda j, b, f, scales, scores: tf.less(j, num_scales)
 143 
 144   (_, output_boxes, output_features, output_scales,
 145    output_scores) = tf.while_loop(
 146        cond=keep_going,
 147        body=_ProcessSingleScale,
 148        loop_vars=[
 149            i, output_boxes, output_features, output_scales, output_scores
 150        ],
 151        shape_invariants=[
 152            i.get_shape(),
 153            tf.TensorShape([None, 4]),
 154            tf.TensorShape([None, feature_depth]),
 155            tf.TensorShape([None]),
 156            tf.TensorShape([None])
 157        ],
 158        back_prop=False)
 159 
 160   feature_boxes = box_list.BoxList(output_boxes)
 161   feature_boxes.add_field('features', output_features)
 162   feature_boxes.add_field('scales', output_scales)
 163   feature_boxes.add_field('scores', output_scores)
 164 
 165   nms_max_boxes = tf.minimum(max_feature_num, feature_boxes.num_boxes())
 166   final_boxes = box_list_ops.non_max_suppression(feature_boxes, iou,
 167                                                  nms_max_boxes)
 168 
 169   return final_boxes.get(), final_boxes.get_field(
 170       'features'), final_boxes.get_field('scales'), tf.expand_dims(
 171           final_boxes.get_field('scores'), 1)
