  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function

  22 from absl.testing import parameterized
  23 import tensorflow as tf
  24 
  25 from delf.python.training.model import delf_model

  27 
  28 class DelfTest(tf.test.TestCase, parameterized.TestCase):
  29 
  30   @parameterized.named_parameters(
  31       ('block3_stridesTrue', True),
  32       ('block3_stridesFalse', False),
  33   )
  34   def test_build_model(self, block3_strides):
  35     image_size = 321
  36     num_classes = 1000
  37     batch_size = 2
  38     input_shape = (batch_size, image_size, image_size, 3)
  39 
  40     model = delf_model.Delf(block3_strides=block3_strides, name='DELF')
  41     model.init_classifiers(num_classes)
  42 
  43     images = tf.random.uniform(input_shape, minval=-1.0, maxval=1.0, seed=0)
  44     blocks = {}
  45 
  46     # Get global feature by pooling block4 features.
  47     desc_prelogits = model.backbone(
  48         images, intermediates_dict=blocks, training=False)
  49     desc_logits = model.desc_classification(desc_prelogits)
  50     self.assertAllEqual(desc_prelogits.shape, (batch_size, 2048))
  51     self.assertAllEqual(desc_logits.shape, (batch_size, num_classes))
  52 
  53     features = blocks['block3']
  54     attn_prelogits, _, _ = model.attention(features)
  55     attn_logits = model.attn_classification(attn_prelogits)
  56     self.assertAllEqual(attn_prelogits.shape, (batch_size, 1024))
  57     self.assertAllEqual(attn_logits.shape, (batch_size, num_classes))
  58 
  59   @parameterized.named_parameters(
  60       ('block3_stridesTrue', True),
  61       ('block3_stridesFalse', False),
  62   )
  63   def test_train_step(self, block3_strides):
  64 
  65     image_size = 321
  66     num_classes = 1000
  67     batch_size = 2
  68     clip_val = 10.0
  69     input_shape = (batch_size, image_size, image_size, 3)
  70 
  71     model = delf_model.Delf(block3_strides=block3_strides, name='DELF')
  72     model.init_classifiers(num_classes)
  73 
  74     optimizer = tf.keras.optimizers.SGD(learning_rate=0.001, momentum=0.9)
  75 
  76     images = tf.random.uniform(input_shape, minval=0.0, maxval=1.0, seed=0)
  77     labels = tf.random.uniform((batch_size,),
  78                                minval=0,
  79                                maxval=model.num_classes - 1,
  80                                dtype=tf.int64)
  81 
  82     loss_object = tf.keras.losses.SparseCategoricalCrossentropy(
  83         from_logits=True, reduction=tf.keras.losses.Reduction.NONE)
  84 
  85     def compute_loss(labels, predictions):
  86       per_example_loss = loss_object(labels, predictions)
  87       return tf.nn.compute_average_loss(
  88           per_example_loss, global_batch_size=batch_size)
  89 
  90     with tf.GradientTape() as desc_tape:
  91       blocks = {}
  92       desc_prelogits = model.backbone(
  93           images, intermediates_dict=blocks, training=False)
  94       desc_logits = model.desc_classification(desc_prelogits)
  95       desc_logits = model.desc_classification(desc_prelogits)
  96       desc_loss = compute_loss(labels, desc_logits)
  97 
  98     gradients = desc_tape.gradient(desc_loss, model.desc_trainable_weights)
  99     clipped, _ = tf.clip_by_global_norm(gradients, clip_norm=clip_val)
 100     optimizer.apply_gradients(zip(clipped, model.desc_trainable_weights))
 101 
 102     with tf.GradientTape() as attn_tape:
 103       block3 = blocks['block3']
 104       block3 = tf.stop_gradient(block3)
 105       attn_prelogits, _, _ = model.attention(block3, training=True)
 106       attn_logits = model.attn_classification(attn_prelogits)
 107       attn_loss = compute_loss(labels, attn_logits)
 108 
 109     gradients = attn_tape.gradient(attn_loss, model.attn_trainable_weights)
 110     clipped, _ = tf.clip_by_global_norm(gradients, clip_norm=clip_val)
 111     optimizer.apply_gradients(zip(clipped, model.attn_trainable_weights))
