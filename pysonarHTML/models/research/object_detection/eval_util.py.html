  77 def write_metrics(metrics, global_step, summary_dir):
  78   """Write metrics to a summary directory.
  79 
  80   Args:
  81     metrics: A dictionary containing metric names and values.
  82     global_step: Global step at which the metrics are computed.
  83     summary_dir: Directory to write tensorflow summaries to.
  84   """
  85   tf.logging.info('Writing metrics to tf summary.')
  86   summary_writer = tf.summary.FileWriterCache.get(summary_dir)
  87   for key in sorted(metrics):
  88     summary = tf.Summary(value=[
  89         tf.Summary.Value(tag=key, simple_value=metrics[key]),
  90     ])
  91     summary_writer.add_summary(summary, global_step)
  92     tf.logging.info('%s: %f', key, metrics[key])
  93   tf.logging.info('Metrics written to tf summary.')

  97 def visualize_detection_results(result_dict,
  98                                 tag,
  99                                 global_step,
 100                                 categories,
 101                                 summary_dir='',
 102                                 export_dir='',
 103                                 agnostic_mode=False,
 104                                 show_groundtruth=False,
 105                                 groundtruth_box_visualization_color='black',
 106                                 min_score_thresh=.5,
 107                                 max_num_predictions=20,
 108                                 skip_scores=False,
 109                                 skip_labels=False,
 110                                 keep_image_id_for_visualization_export=False):
 111   """Visualizes detection results and writes visualizations to image summaries.
 112 
 113   This function visualizes an image with its detected bounding boxes and writes
 114   to image summaries which can be viewed on tensorboard.  It optionally also
 115   writes images to a directory. In the case of missing entry in the label map,
 116   unknown class name in the visualization is shown as "N/A".
 117 
 118   Args:
 119     result_dict: a dictionary holding groundtruth and detection
 120       data corresponding to each image being evaluated.  The following keys
 121       are required:
 122         'original_image': a numpy array representing the image with shape
 123           [1, height, width, 3] or [1, height, width, 1]
 124         'detection_boxes': a numpy array of shape [N, 4]
 125         'detection_scores': a numpy array of shape [N]
 126         'detection_classes': a numpy array of shape [N]
 127       The following keys are optional:
 128         'groundtruth_boxes': a numpy array of shape [N, 4]
 129         'groundtruth_keypoints': a numpy array of shape [N, num_keypoints, 2]
 130       Detections are assumed to be provided in decreasing order of score and for
 131       display, and we assume that scores are probabilities between 0 and 1.
 132     tag: tensorboard tag (string) to associate with image.
 133     global_step: global step at which the visualization are generated.
 134     categories: a list of dictionaries representing all possible categories.
 135       Each dict in this list has the following keys:
 136           'id': (required) an integer id uniquely identifying this category
 137           'name': (required) string representing category name
 138             e.g., 'cat', 'dog', 'pizza'
 139           'supercategory': (optional) string representing the supercategory
 140             e.g., 'animal', 'vehicle', 'food', etc
 141     summary_dir: the output directory to which the image summaries are written.
 142     export_dir: the output directory to which images are written.  If this is
 143       empty (default), then images are not exported.
 144     agnostic_mode: boolean (default: False) controlling whether to evaluate in
 145       class-agnostic mode or not.
 146     show_groundtruth: boolean (default: False) controlling whether to show
 147       groundtruth boxes in addition to detected boxes
 148     groundtruth_box_visualization_color: box color for visualizing groundtruth
 149       boxes
 150     min_score_thresh: minimum score threshold for a box to be visualized
 151     max_num_predictions: maximum number of detections to visualize
 152     skip_scores: whether to skip score when drawing a single detection
 153     skip_labels: whether to skip label when drawing a single detection
 154     keep_image_id_for_visualization_export: whether to keep image identifier in
 155       filename when exported to export_dir
 156   Raises:
 157     ValueError: if result_dict does not contain the expected keys (i.e.,
 158       'original_image', 'detection_boxes', 'detection_scores',
 159       'detection_classes')
 160   """
 161   detection_fields = fields.DetectionResultFields
 162   input_fields = fields.InputDataFields
 163   if not set([
 164       input_fields.original_image,
 165       detection_fields.detection_boxes,
 166       detection_fields.detection_scores,
 167       detection_fields.detection_classes,
 168   ]).issubset(set(result_dict.keys())):
 169     raise ValueError('result_dict does not contain all expected keys.')
 170   if show_groundtruth and input_fields.groundtruth_boxes not in result_dict:
 171     raise ValueError('If show_groundtruth is enabled, result_dict must contain '
 172                      'groundtruth_boxes.')
 173   tf.logging.info('Creating detection visualizations.')
 174   category_index = label_map_util.create_category_index(categories)
 175 
 176   image = np.squeeze(result_dict[input_fields.original_image], axis=0)
 177   if image.shape[2] == 1:  # If one channel image, repeat in RGB.
 178     image = np.tile(image, [1, 1, 3])
 179   detection_boxes = result_dict[detection_fields.detection_boxes]
 180   detection_scores = result_dict[detection_fields.detection_scores]
 181   detection_classes = np.int32((result_dict[
 182       detection_fields.detection_classes]))
 183   detection_keypoints = result_dict.get(detection_fields.detection_keypoints)
 184   detection_masks = result_dict.get(detection_fields.detection_masks)
 185   detection_boundaries = result_dict.get(detection_fields.detection_boundaries)
 186 
 187   # Plot groundtruth underneath detections
 188   if show_groundtruth:
 189     groundtruth_boxes = result_dict[input_fields.groundtruth_boxes]
 190     groundtruth_keypoints = result_dict.get(input_fields.groundtruth_keypoints)
 191     vis_utils.visualize_boxes_and_labels_on_image_array(
 192         image=image,
 193         boxes=groundtruth_boxes,
 194         classes=None,
 195         scores=None,
 196         category_index=category_index,
 197         keypoints=groundtruth_keypoints,
 198         use_normalized_coordinates=False,
 199         max_boxes_to_draw=None,
 200         groundtruth_box_visualization_color=groundtruth_box_visualization_color)
 201   vis_utils.visualize_boxes_and_labels_on_image_array(
 202       image,
 203       detection_boxes,
 204       detection_classes,
 205       detection_scores,
 206       category_index,
 207       instance_masks=detection_masks,
 208       instance_boundaries=detection_boundaries,
 209       keypoints=detection_keypoints,
 210       use_normalized_coordinates=False,
 211       max_boxes_to_draw=max_num_predictions,
 212       min_score_thresh=min_score_thresh,
 213       agnostic_mode=agnostic_mode,
 214       skip_scores=skip_scores,
 215       skip_labels=skip_labels)
 216 
 217   if export_dir:
 218     if keep_image_id_for_visualization_export and result_dict[fields.
 219                                                               InputDataFields()
 220                                                               .key]:
 221       export_path = os.path.join(export_dir, 'export-{}-{}.png'.format(
 222           tag, result_dict[fields.InputDataFields().key]))
 223     else:
 224       export_path = os.path.join(export_dir, 'export-{}.png'.format(tag))
 225     vis_utils.save_image_array_as_png(image, export_path)
 226 
 227   summary = tf.Summary(value=[
 228       tf.Summary.Value(
 229           tag=tag,
 230           image=tf.Summary.Image(
 231               encoded_image_string=vis_utils.encode_image_array_as_png_str(
 232                   image)))
 233   ])
 234   summary_writer = tf.summary.FileWriterCache.get(summary_dir)
 235   summary_writer.add_summary(summary, global_step)
 236 
 237   tf.logging.info('Detection visualizations written to summary with tag %s.',
 238                   tag)

 241 def _run_checkpoint_once(tensor_dict,
 242                          evaluators=None,
 243                          batch_processor=None,
 244                          checkpoint_dirs=None,
 245                          variables_to_restore=None,
 246                          restore_fn=None,
 247                          num_batches=1,
 248                          master='',
 249                          save_graph=False,
 250                          save_graph_dir='',
 251                          losses_dict=None,
 252                          eval_export_path=None,
 253                          process_metrics_fn=None):
 254   """Evaluates metrics defined in evaluators and returns summaries.
 255 
 256   This function loads the latest checkpoint in checkpoint_dirs and evaluates
 257   all metrics defined in evaluators. The metrics are processed in batch by the
 258   batch_processor.
 259 
 260   Args:
 261     tensor_dict: a dictionary holding tensors representing a batch of detections
 262       and corresponding groundtruth annotations.
 263     evaluators: a list of object of type DetectionEvaluator to be used for
 264       evaluation. Note that the metric names produced by different evaluators
 265       must be unique.
 266     batch_processor: a function taking four arguments:
 267       1. tensor_dict: the same tensor_dict that is passed in as the first
 268         argument to this function.
 269       2. sess: a tensorflow session
 270       3. batch_index: an integer representing the index of the batch amongst
 271         all batches
 272       By default, batch_processor is None, which defaults to running:
 273         return sess.run(tensor_dict)
 274       To skip an image, it suffices to return an empty dictionary in place of
 275       result_dict.
 276     checkpoint_dirs: list of directories to load into an EnsembleModel. If it
 277       has only one directory, EnsembleModel will not be used --
 278         a DetectionModel
 279       will be instantiated directly. Not used if restore_fn is set.
 280     variables_to_restore: None, or a dictionary mapping variable names found in
 281       a checkpoint to model variables. The dictionary would normally be
 282       generated by creating a tf.train.ExponentialMovingAverage object and
 283       calling its variables_to_restore() method. Not used if restore_fn is set.
 284     restore_fn: None, or a function that takes a tf.Session object and correctly
 285       restores all necessary variables from the correct checkpoint file. If
 286       None, attempts to restore from the first directory in checkpoint_dirs.
 287     num_batches: the number of batches to use for evaluation.
 288     master: the location of the Tensorflow session.
 289     save_graph: whether or not the Tensorflow graph is stored as a pbtxt file.
 290     save_graph_dir: where to store the Tensorflow graph on disk. If save_graph
 291       is True this must be non-empty.
 292     losses_dict: optional dictionary of scalar detection losses.
 293     eval_export_path: Path for saving a json file that contains the detection
 294       results in json format.
 295     process_metrics_fn: a callback called with evaluation results after each
 296       evaluation is done.  It could be used e.g. to back up checkpoints with
 297       best evaluation scores, or to call an external system to update evaluation
 298       results in order to drive best hyper-parameter search.  Parameters are:
 299       int checkpoint_number, Dict[str, ObjectDetectionEvalMetrics] metrics,
 300       str checkpoint_file path.
 301 
 302   Returns:
 303     global_step: the count of global steps.
 304     all_evaluator_metrics: A dictionary containing metric names and values.
 305 
 306   Raises:
 307     ValueError: if restore_fn is None and checkpoint_dirs doesn't have at least
 308       one element.
 309     ValueError: if save_graph is True and save_graph_dir is not defined.
 310   """
 311   if save_graph and not save_graph_dir:
 312     raise ValueError('`save_graph_dir` must be defined.')
 313   sess = tf.Session(master, graph=tf.get_default_graph())
 314   sess.run(tf.global_variables_initializer())
 315   sess.run(tf.local_variables_initializer())
 316   sess.run(tf.tables_initializer())
 317   checkpoint_file = None
 318   if restore_fn:
 319     restore_fn(sess)
 320   else:
 321     if not checkpoint_dirs:
 322       raise ValueError('`checkpoint_dirs` must have at least one entry.')
 323     checkpoint_file = tf.train.latest_checkpoint(checkpoint_dirs[0])
 324     saver = tf.train.Saver(variables_to_restore)
 325     saver.restore(sess, checkpoint_file)
 326 
 327   if save_graph:
 328     tf.train.write_graph(sess.graph_def, save_graph_dir, 'eval.pbtxt')
 329 
 330   counters = {'skipped': 0, 'success': 0}
 331   aggregate_result_losses_dict = collections.defaultdict(list)
 332   with slim.queues.QueueRunners(sess):
 333     try:
 334       for batch in range(int(num_batches)):
 335         if (batch + 1) % 100 == 0:
 336           tf.logging.info('Running eval ops batch %d/%d', batch + 1,
 337                           num_batches)
 338         if not batch_processor:
 339           try:
 340             if not losses_dict:
 341               losses_dict = {}
 342             result_dict, result_losses_dict = sess.run([tensor_dict,
 343                                                         losses_dict])
 344             counters['success'] += 1
 345           except tf.errors.InvalidArgumentError:
 346             tf.logging.info('Skipping image')
 347             counters['skipped'] += 1
 348             result_dict = {}
 349         else:
 350           result_dict, result_losses_dict = batch_processor(
 351               tensor_dict, sess, batch, counters, losses_dict=losses_dict)
 352         if not result_dict:
 353           continue
 354         for key, value in iter(result_losses_dict.items()):
 355           aggregate_result_losses_dict[key].append(value)
 356         for evaluator in evaluators:
 357           # TODO(b/65130867): Use image_id tensor once we fix the input data
 358           # decoders to return correct image_id.
 359           # TODO(akuznetsa): result_dict contains batches of images, while
 360           # add_single_ground_truth_image_info expects a single image. Fix
 361           if (isinstance(result_dict, dict) and
 362               fields.InputDataFields.key in result_dict and
 363               result_dict[fields.InputDataFields.key]):
 364             image_id = result_dict[fields.InputDataFields.key]
 365           else:
 366             image_id = batch
 367           evaluator.add_single_ground_truth_image_info(
 368               image_id=image_id, groundtruth_dict=result_dict)
 369           evaluator.add_single_detected_image_info(
 370               image_id=image_id, detections_dict=result_dict)
 371       tf.logging.info('Running eval batches done.')
 372     except tf.errors.OutOfRangeError:
 373       tf.logging.info('Done evaluating -- epoch limit reached')
 374     finally:
 375       # When done, ask the threads to stop.
 376       tf.logging.info('# success: %d', counters['success'])
 377       tf.logging.info('# skipped: %d', counters['skipped'])
 378       all_evaluator_metrics = {}
 379       if eval_export_path and eval_export_path is not None:
 380         for evaluator in evaluators:
 381           if (isinstance(evaluator, coco_evaluation.CocoDetectionEvaluator) or
 382               isinstance(evaluator, coco_evaluation.CocoMaskEvaluator)):
 383             tf.logging.info('Started dumping to json file.')
 384             evaluator.dump_detections_to_json_file(
 385                 json_output_path=eval_export_path)
 386             tf.logging.info('Finished dumping to json file.')
 387       for evaluator in evaluators:
 388         metrics = evaluator.evaluate()
 389         evaluator.clear()
 390         if any(key in all_evaluator_metrics for key in metrics):
 391           raise ValueError('Metric names between evaluators must not collide.')
 392         all_evaluator_metrics.update(metrics)
 393       global_step = tf.train.global_step(sess, tf.train.get_global_step())
 394 
 395       for key, value in iter(aggregate_result_losses_dict.items()):
 396         all_evaluator_metrics['Losses/' + key] = np.mean(value)
 397       if process_metrics_fn and checkpoint_file:
 398         m = re.search(r'model.ckpt-(\d+)$', checkpoint_file)
 399         if not m:
 400           tf.logging.error('Failed to parse checkpoint number from: %s',
 401                            checkpoint_file)
 402         else:
 403           checkpoint_number = int(m.group(1))
 404           process_metrics_fn(checkpoint_number, all_evaluator_metrics,
 405                              checkpoint_file)
 406   sess.close()
 407   return (global_step, all_evaluator_metrics)

 411 def repeated_checkpoint_run(tensor_dict,
 412                             summary_dir,
 413                             evaluators,
 414                             batch_processor=None,
 415                             checkpoint_dirs=None,
 416                             variables_to_restore=None,
 417                             restore_fn=None,
 418                             num_batches=1,
 419                             eval_interval_secs=120,
 420                             max_number_of_evaluations=None,
 421                             max_evaluation_global_step=None,
 422                             master='',
 423                             save_graph=False,
 424                             save_graph_dir='',
 425                             losses_dict=None,
 426                             eval_export_path=None,
 427                             process_metrics_fn=None):
 428   """Periodically evaluates desired tensors using checkpoint_dirs or restore_fn.
 429 
 430   This function repeatedly loads a checkpoint and evaluates a desired
 431   set of tensors (provided by tensor_dict) and hands the resulting numpy
 432   arrays to a function result_processor which can be used to further
 433   process/save/visualize the results.
 434 
 435   Args:
 436     tensor_dict: a dictionary holding tensors representing a batch of detections
 437       and corresponding groundtruth annotations.
 438     summary_dir: a directory to write metrics summaries.
 439     evaluators: a list of object of type DetectionEvaluator to be used for
 440       evaluation. Note that the metric names produced by different evaluators
 441       must be unique.
 442     batch_processor: a function taking three arguments:
 443       1. tensor_dict: the same tensor_dict that is passed in as the first
 444         argument to this function.
 445       2. sess: a tensorflow session
 446       3. batch_index: an integer representing the index of the batch amongst
 447         all batches
 448       By default, batch_processor is None, which defaults to running:
 449         return sess.run(tensor_dict)
 450     checkpoint_dirs: list of directories to load into a DetectionModel or an
 451       EnsembleModel if restore_fn isn't set. Also used to determine when to run
 452       next evaluation. Must have at least one element.
 453     variables_to_restore: None, or a dictionary mapping variable names found in
 454       a checkpoint to model variables. The dictionary would normally be
 455       generated by creating a tf.train.ExponentialMovingAverage object and
 456       calling its variables_to_restore() method. Not used if restore_fn is set.
 457     restore_fn: a function that takes a tf.Session object and correctly restores
 458       all necessary variables from the correct checkpoint file.
 459     num_batches: the number of batches to use for evaluation.
 460     eval_interval_secs: the number of seconds between each evaluation run.
 461     max_number_of_evaluations: the max number of iterations of the evaluation.
 462       If the value is left as None the evaluation continues indefinitely.
 463     max_evaluation_global_step: global step when evaluation stops.
 464     master: the location of the Tensorflow session.
 465     save_graph: whether or not the Tensorflow graph is saved as a pbtxt file.
 466     save_graph_dir: where to save on disk the Tensorflow graph. If store_graph
 467       is True this must be non-empty.
 468     losses_dict: optional dictionary of scalar detection losses.
 469     eval_export_path: Path for saving a json file that contains the detection
 470       results in json format.
 471     process_metrics_fn: a callback called with evaluation results after each
 472       evaluation is done.  It could be used e.g. to back up checkpoints with
 473       best evaluation scores, or to call an external system to update evaluation
 474       results in order to drive best hyper-parameter search.  Parameters are:
 475       int checkpoint_number, Dict[str, ObjectDetectionEvalMetrics] metrics,
 476       str checkpoint_file path.
 477 
 478   Returns:
 479     metrics: A dictionary containing metric names and values in the latest
 480       evaluation.
 481 
 482   Raises:
 483     ValueError: if max_num_of_evaluations is not None or a positive number.
 484     ValueError: if checkpoint_dirs doesn't have at least one element.
 485   """
 486   if max_number_of_evaluations and max_number_of_evaluations <= 0:
 487     raise ValueError(
 488         '`max_number_of_evaluations` must be either None or a positive number.')
 489   if max_evaluation_global_step and max_evaluation_global_step <= 0:
 490     raise ValueError(
 491         '`max_evaluation_global_step` must be either None or positive.')
 492 
 493   if not checkpoint_dirs:
 494     raise ValueError('`checkpoint_dirs` must have at least one entry.')
 495 
 496   last_evaluated_model_path = None
 497   number_of_evaluations = 0
 498   while True:
 499     start = time.time()
 500     tf.logging.info('Starting evaluation at ' + time.strftime(
 501         '%Y-%m-%d-%H:%M:%S', time.gmtime()))
 502     model_path = tf.train.latest_checkpoint(checkpoint_dirs[0])
 503     if not model_path:
 504       tf.logging.info('No model found in %s. Will try again in %d seconds',
 505                       checkpoint_dirs[0], eval_interval_secs)
 506     elif model_path == last_evaluated_model_path:
 507       tf.logging.info('Found already evaluated checkpoint. Will try again in '
 508                       '%d seconds', eval_interval_secs)
 509     else:
 510       last_evaluated_model_path = model_path
 511       global_step, metrics = _run_checkpoint_once(
 512           tensor_dict,
 513           evaluators,
 514           batch_processor,
 515           checkpoint_dirs,
 516           variables_to_restore,
 517           restore_fn,
 518           num_batches,
 519           master,
 520           save_graph,
 521           save_graph_dir,
 522           losses_dict=losses_dict,
 523           eval_export_path=eval_export_path,
 524           process_metrics_fn=process_metrics_fn)
 525       write_metrics(metrics, global_step, summary_dir)
 526       if (max_evaluation_global_step and
 527           global_step >= max_evaluation_global_step):
 528         tf.logging.info('Finished evaluation!')
 529         break
 530     number_of_evaluations += 1
 531 
 532     if (max_number_of_evaluations and
 533         number_of_evaluations >= max_number_of_evaluations):
 534       tf.logging.info('Finished evaluation!')
 535       break
 536     time_to_next_eval = start + eval_interval_secs - time.time()
 537     if time_to_next_eval > 0:
 538       time.sleep(time_to_next_eval)
 539 
 540   return metrics

 549 def _resize_detection_masks(args):
 550   detection_boxes, detection_masks, image_shape = args
 551   detection_masks_reframed = ops.reframe_box_masks_to_image_masks(
 552       detection_masks, detection_boxes, image_shape[0], image_shape[1])
 553   return tf.cast(tf.greater(detection_masks_reframed, 0.5), tf.uint8)
 554 
 555 
 556 def _resize_groundtruth_masks(args):
 557   """Resizes groundgtruth masks to the original image size."""
 558   mask, true_image_shape, original_image_shape = args
 559   true_height = true_image_shape[0]
 560   true_width = true_image_shape[1]
 561   mask = mask[:, :true_height, :true_width]
 562   mask = tf.expand_dims(mask, 3)
 563   mask = tf.image.resize_images(
 564       mask,
 565       original_image_shape,
 566       method=tf.image.ResizeMethod.NEAREST_NEIGHBOR,
 567       align_corners=True)
 568   return tf.cast(tf.squeeze(mask, 3), tf.uint8)
 569 
 570 
 571 def _scale_keypoint_to_absolute(args):
 572   keypoints, image_shape = args
 573   return keypoint_ops.scale(keypoints, image_shape[0], image_shape[1])

 576 def result_dict_for_single_example(image,
 577                                    key,
 578                                    detections,
 579                                    groundtruth=None,
 580                                    class_agnostic=False,
 581                                    scale_to_absolute=False):
 582   """Merges all detection and groundtruth information for a single example.
 583 
 584   Note that evaluation tools require classes that are 1-indexed, and so this
 585   function performs the offset. If `class_agnostic` is True, all output classes
 586   have label 1.
 587 
 588   Args:
 589     image: A single 4D uint8 image tensor of shape [1, H, W, C].
 590     key: A single string tensor identifying the image.
 591     detections: A dictionary of detections, returned from
 592       DetectionModel.postprocess().
 593     groundtruth: (Optional) Dictionary of groundtruth items, with fields:
 594       'groundtruth_boxes': [num_boxes, 4] float32 tensor of boxes, in
 595         normalized coordinates.
 596       'groundtruth_classes': [num_boxes] int64 tensor of 1-indexed classes.
 597       'groundtruth_area': [num_boxes] float32 tensor of bbox area. (Optional)
 598       'groundtruth_is_crowd': [num_boxes] int64 tensor. (Optional)
 599       'groundtruth_difficult': [num_boxes] int64 tensor. (Optional)
 600       'groundtruth_group_of': [num_boxes] int64 tensor. (Optional)
 601       'groundtruth_instance_masks': 3D int64 tensor of instance masks
 602         (Optional).
 603       'groundtruth_keypoints': [num_boxes, num_keypoints, 2] float32 tensor with
 604         keypoints (Optional).
 605     class_agnostic: Boolean indicating whether the detections are class-agnostic
 606       (i.e. binary). Default False.
 607     scale_to_absolute: Boolean indicating whether boxes and keypoints should be
 608       scaled to absolute coordinates. Note that for IoU based evaluations, it
 609       does not matter whether boxes are expressed in absolute or relative
 610       coordinates. Default False.
 611 
 612   Returns:
 613     A dictionary with:
 614     'original_image': A [1, H, W, C] uint8 image tensor.
 615     'key': A string tensor with image identifier.
 616     'detection_boxes': [max_detections, 4] float32 tensor of boxes, in
 617       normalized or absolute coordinates, depending on the value of
 618       `scale_to_absolute`.
 619     'detection_scores': [max_detections] float32 tensor of scores.
 620     'detection_classes': [max_detections] int64 tensor of 1-indexed classes.
 621     'detection_masks': [max_detections, H, W] float32 tensor of binarized
 622       masks, reframed to full image masks.
 623     'groundtruth_boxes': [num_boxes, 4] float32 tensor of boxes, in
 624       normalized or absolute coordinates, depending on the value of
 625       `scale_to_absolute`. (Optional)
 626     'groundtruth_classes': [num_boxes] int64 tensor of 1-indexed classes.
 627       (Optional)
 628     'groundtruth_area': [num_boxes] float32 tensor of bbox area. (Optional)
 629     'groundtruth_is_crowd': [num_boxes] int64 tensor. (Optional)
 630     'groundtruth_difficult': [num_boxes] int64 tensor. (Optional)
 631     'groundtruth_group_of': [num_boxes] int64 tensor. (Optional)
 632     'groundtruth_instance_masks': 3D int64 tensor of instance masks
 633       (Optional).
 634     'groundtruth_keypoints': [num_boxes, num_keypoints, 2] float32 tensor with
 635       keypoints (Optional).
 636   """
 637 
 638   if groundtruth:
 639     max_gt_boxes = tf.shape(
 640         groundtruth[fields.InputDataFields.groundtruth_boxes])[0]
 641     for gt_key in groundtruth:
 642       # expand groundtruth dict along the batch dimension.
 643       groundtruth[gt_key] = tf.expand_dims(groundtruth[gt_key], 0)
 644 
 645   for detection_key in detections:
 646     detections[detection_key] = tf.expand_dims(
 647         detections[detection_key][0], axis=0)
 648 
 649   batched_output_dict = result_dict_for_batched_example(
 650       image,
 651       tf.expand_dims(key, 0),
 652       detections,
 653       groundtruth,
 654       class_agnostic,
 655       scale_to_absolute,
 656       max_gt_boxes=max_gt_boxes)
 657 
 658   exclude_keys = [
 659       fields.InputDataFields.original_image,
 660       fields.DetectionResultFields.num_detections,
 661       fields.InputDataFields.num_groundtruth_boxes
 662   ]
 663 
 664   output_dict = {
 665       fields.InputDataFields.original_image:
 666           batched_output_dict[fields.InputDataFields.original_image]
 667   }
 668 
 669   for key in batched_output_dict:
 670     # remove the batch dimension.
 671     if key not in exclude_keys:
 672       output_dict[key] = tf.squeeze(batched_output_dict[key], 0)
 673   return output_dict

 676 def result_dict_for_batched_example(images,
 677                                     keys,
 678                                     detections,
 679                                     groundtruth=None,
 680                                     class_agnostic=False,
 681                                     scale_to_absolute=False,
 682                                     original_image_spatial_shapes=None,
 683                                     true_image_shapes=None,
 684                                     max_gt_boxes=None):
 685   """Merges all detection and groundtruth information for a single example.
 686 
 687   Note that evaluation tools require classes that are 1-indexed, and so this
 688   function performs the offset. If `class_agnostic` is True, all output classes
 689   have label 1.
 690   The groundtruth coordinates of boxes/keypoints in 'groundtruth' dictionary are
 691   normalized relative to the (potentially padded) input image, while the
 692   coordinates in 'detection' dictionary are normalized relative to the true
 693   image shape.
 694 
 695   Args:
 696     images: A single 4D uint8 image tensor of shape [batch_size, H, W, C].
 697     keys: A [batch_size] string/int tensor with image identifier.
 698     detections: A dictionary of detections, returned from
 699       DetectionModel.postprocess().
 700     groundtruth: (Optional) Dictionary of groundtruth items, with fields:
 701       'groundtruth_boxes': [batch_size, max_number_of_boxes, 4] float32 tensor
 702         of boxes, in normalized coordinates.
 703       'groundtruth_classes':  [batch_size, max_number_of_boxes] int64 tensor of
 704         1-indexed classes.
 705       'groundtruth_area': [batch_size, max_number_of_boxes] float32 tensor of
 706         bbox area. (Optional)
 707       'groundtruth_is_crowd':[batch_size, max_number_of_boxes] int64
 708         tensor. (Optional)
 709       'groundtruth_difficult': [batch_size, max_number_of_boxes] int64
 710         tensor. (Optional)
 711       'groundtruth_group_of': [batch_size, max_number_of_boxes] int64
 712         tensor. (Optional)
 713       'groundtruth_instance_masks': 4D int64 tensor of instance
 714         masks (Optional).
 715       'groundtruth_keypoints': [batch_size, max_number_of_boxes, num_keypoints,
 716         2] float32 tensor with keypoints (Optional).
 717       'groundtruth_keypoint_visibilities': [batch_size, max_number_of_boxes,
 718         num_keypoints] bool tensor with keypoint visibilities (Optional).
 719       'groundtruth_labeled_classes': [batch_size, num_classes] int64
 720         tensor of 1-indexed classes. (Optional)
 721     class_agnostic: Boolean indicating whether the detections are class-agnostic
 722       (i.e. binary). Default False.
 723     scale_to_absolute: Boolean indicating whether boxes and keypoints should be
 724       scaled to absolute coordinates. Note that for IoU based evaluations, it
 725       does not matter whether boxes are expressed in absolute or relative
 726       coordinates. Default False.
 727     original_image_spatial_shapes: A 2D int32 tensor of shape [batch_size, 2]
 728       used to resize the image. When set to None, the image size is retained.
 729     true_image_shapes: A 2D int32 tensor of shape [batch_size, 3]
 730       containing the size of the unpadded original_image.
 731     max_gt_boxes: [batch_size] tensor representing the maximum number of
 732       groundtruth boxes to pad.
 733 
 734   Returns:
 735     A dictionary with:
 736     'original_image': A [batch_size, H, W, C] uint8 image tensor.
 737     'original_image_spatial_shape': A [batch_size, 2] tensor containing the
 738       original image sizes.
 739     'true_image_shape': A [batch_size, 3] tensor containing the size of
 740       the unpadded original_image.
 741     'key': A [batch_size] string tensor with image identifier.
 742     'detection_boxes': [batch_size, max_detections, 4] float32 tensor of boxes,
 743       in normalized or absolute coordinates, depending on the value of
 744       `scale_to_absolute`.
 745     'detection_scores': [batch_size, max_detections] float32 tensor of scores.
 746     'detection_classes': [batch_size, max_detections] int64 tensor of 1-indexed
 747       classes.
 748     'detection_masks': [batch_size, max_detections, H, W] float32 tensor of
 749       binarized masks, reframed to full image masks. (Optional)
 750     'detection_keypoints': [batch_size, max_detections, num_keypoints, 2]
 751       float32 tensor containing keypoint coordinates. (Optional)
 752     'detection_keypoint_scores': [batch_size, max_detections, num_keypoints]
 753       float32 tensor containing keypoint scores. (Optional)
 754     'num_detections': [batch_size] int64 tensor containing number of valid
 755       detections.
 756     'groundtruth_boxes': [batch_size, num_boxes, 4] float32 tensor of boxes, in
 757       normalized or absolute coordinates, depending on the value of
 758       `scale_to_absolute`. (Optional)
 759     'groundtruth_classes': [batch_size, num_boxes] int64 tensor of 1-indexed
 760       classes. (Optional)
 761     'groundtruth_area': [batch_size, num_boxes] float32 tensor of bbox
 762       area. (Optional)
 763     'groundtruth_is_crowd': [batch_size, num_boxes] int64 tensor. (Optional)
 764     'groundtruth_difficult': [batch_size, num_boxes] int64 tensor. (Optional)
 765     'groundtruth_group_of': [batch_size, num_boxes] int64 tensor. (Optional)
 766     'groundtruth_instance_masks': 4D int64 tensor of instance masks
 767       (Optional).
 768     'groundtruth_keypoints': [batch_size, num_boxes, num_keypoints, 2] float32
 769       tensor with keypoints (Optional).
 770     'groundtruth_keypoint_visibilities': [batch_size, num_boxes, num_keypoints]
 771       bool tensor with keypoint visibilities (Optional).
 772     'groundtruth_labeled_classes': [batch_size, num_classes]  int64 tensor
 773       of 1-indexed classes. (Optional)
 774     'num_groundtruth_boxes': [batch_size] tensor containing the maximum number
 775       of groundtruth boxes per image.
 776 
 777   Raises:
 778     ValueError: if original_image_spatial_shape is not 2D int32 tensor of shape
 779       [2].
 780     ValueError: if true_image_shapes is not 2D int32 tensor of shape
 781       [3].
 782   """
 783   label_id_offset = 1  # Applying label id offset (b/63711816)
 784 
 785   input_data_fields = fields.InputDataFields
 786   if original_image_spatial_shapes is None:
 787     original_image_spatial_shapes = tf.tile(
 788         tf.expand_dims(tf.shape(images)[1:3], axis=0),
 789         multiples=[tf.shape(images)[0], 1])
 790   else:
 791     if (len(original_image_spatial_shapes.shape) != 2 and
 792         original_image_spatial_shapes.shape[1] != 2):
 793       raise ValueError(
 794           '`original_image_spatial_shape` should be a 2D tensor of shape '
 795           '[batch_size, 2].')
 796 
 797   if true_image_shapes is None:
 798     true_image_shapes = tf.tile(
 799         tf.expand_dims(tf.shape(images)[1:4], axis=0),
 800         multiples=[tf.shape(images)[0], 1])
 801   else:
 802     if (len(true_image_shapes.shape) != 2
 803         and true_image_shapes.shape[1] != 3):
 804       raise ValueError('`true_image_shapes` should be a 2D tensor of '
 805                        'shape [batch_size, 3].')
 806 
 807   output_dict = {
 808       input_data_fields.original_image:
 809           images,
 810       input_data_fields.key:
 811           keys,
 812       input_data_fields.original_image_spatial_shape: (
 813           original_image_spatial_shapes),
 814       input_data_fields.true_image_shape:
 815           true_image_shapes
 816   }
 817 
 818   detection_fields = fields.DetectionResultFields
 819   detection_boxes = detections[detection_fields.detection_boxes]
 820   detection_scores = detections[detection_fields.detection_scores]
 821   num_detections = tf.cast(detections[detection_fields.num_detections],
 822                            dtype=tf.int32)
 823 
 824   if class_agnostic:
 825     detection_classes = tf.ones_like(detection_scores, dtype=tf.int64)
 826   else:
 827     detection_classes = (
 828         tf.to_int64(detections[detection_fields.detection_classes]) +
 829         label_id_offset)
 830 
 831   if scale_to_absolute:
 832     output_dict[detection_fields.detection_boxes] = (
 833         shape_utils.static_or_dynamic_map_fn(
 834             _scale_box_to_absolute,
 835             elems=[detection_boxes, original_image_spatial_shapes],
 836             dtype=tf.float32))
 837   else:
 838     output_dict[detection_fields.detection_boxes] = detection_boxes
 839   output_dict[detection_fields.detection_classes] = detection_classes
 840   output_dict[detection_fields.detection_scores] = detection_scores
 841   output_dict[detection_fields.num_detections] = num_detections
 842 
 843   if detection_fields.detection_masks in detections:
 844     detection_masks = detections[detection_fields.detection_masks]
 845     # TODO(rathodv): This should be done in model's postprocess
 846     # function ideally.
 847     output_dict[detection_fields.detection_masks] = (
 848         shape_utils.static_or_dynamic_map_fn(
 849             _resize_detection_masks,
 850             elems=[detection_boxes, detection_masks,
 851                    original_image_spatial_shapes],
 852             dtype=tf.uint8))
 853 
 854   if detection_fields.detection_keypoints in detections:
 855     detection_keypoints = detections[detection_fields.detection_keypoints]
 856     output_dict[detection_fields.detection_keypoints] = detection_keypoints
 857     if scale_to_absolute:
 858       output_dict[detection_fields.detection_keypoints] = (
 859           shape_utils.static_or_dynamic_map_fn(
 860               _scale_keypoint_to_absolute,
 861               elems=[detection_keypoints, original_image_spatial_shapes],
 862               dtype=tf.float32))
 863     if detection_fields.detection_keypoint_scores in detections:
 864       output_dict[detection_fields.detection_keypoint_scores] = detections[
 865           detection_fields.detection_keypoint_scores]
 866     else:
 867       output_dict[detection_fields.detection_keypoint_scores] = tf.ones_like(
 868           detections[detection_fields.detection_keypoints][:, :, :, 0])
 869 
 870   if groundtruth:
 871     if max_gt_boxes is None:
 872       if input_data_fields.num_groundtruth_boxes in groundtruth:
 873         max_gt_boxes = groundtruth[input_data_fields.num_groundtruth_boxes]
 874       else:
 875         raise ValueError(
 876             'max_gt_boxes must be provided when processing batched examples.')
 877 
 878     if input_data_fields.groundtruth_instance_masks in groundtruth:
 879       masks = groundtruth[input_data_fields.groundtruth_instance_masks]
 880       groundtruth[input_data_fields.groundtruth_instance_masks] = (
 881           shape_utils.static_or_dynamic_map_fn(
 882               _resize_groundtruth_masks,
 883               elems=[masks, true_image_shapes, original_image_spatial_shapes],
 884               dtype=tf.uint8))
 885 
 886     output_dict.update(groundtruth)
 887 
 888     image_shape = tf.cast(tf.shape(images), tf.float32)
 889     image_height, image_width = image_shape[1], image_shape[2]
 890 
 891     def _scale_box_to_normalized_true_image(args):
 892       """Scale the box coordinates to be relative to the true image shape."""
 893       boxes, true_image_shape = args
 894       true_image_shape = tf.cast(true_image_shape, tf.float32)
 895       true_height, true_width = true_image_shape[0], true_image_shape[1]
 896       normalized_window = tf.stack([0.0, 0.0, true_height / image_height,
 897                                     true_width / image_width])
 898       return box_list_ops.change_coordinate_frame(
 899           box_list.BoxList(boxes), normalized_window).get()
 900 
 901     groundtruth_boxes = groundtruth[input_data_fields.groundtruth_boxes]
 902     groundtruth_boxes = shape_utils.static_or_dynamic_map_fn(
 903         _scale_box_to_normalized_true_image,
 904         elems=[groundtruth_boxes, true_image_shapes], dtype=tf.float32)
 905     output_dict[input_data_fields.groundtruth_boxes] = groundtruth_boxes
 906 
 907     if input_data_fields.groundtruth_keypoints in groundtruth:
 908       # If groundtruth_keypoints is in the groundtruth dictionary. Update the
 909       # coordinates to conform with the true image shape.
 910       def _scale_keypoints_to_normalized_true_image(args):
 911         """Scale the box coordinates to be relative to the true image shape."""
 912         keypoints, true_image_shape = args
 913         true_image_shape = tf.cast(true_image_shape, tf.float32)
 914         true_height, true_width = true_image_shape[0], true_image_shape[1]
 915         normalized_window = tf.stack(
 916             [0.0, 0.0, true_height / image_height, true_width / image_width])
 917         return keypoint_ops.change_coordinate_frame(keypoints,
 918                                                     normalized_window)
 919 
 920       groundtruth_keypoints = groundtruth[
 921           input_data_fields.groundtruth_keypoints]
 922       groundtruth_keypoints = shape_utils.static_or_dynamic_map_fn(
 923           _scale_keypoints_to_normalized_true_image,
 924           elems=[groundtruth_keypoints, true_image_shapes],
 925           dtype=tf.float32)
 926       output_dict[
 927           input_data_fields.groundtruth_keypoints] = groundtruth_keypoints
 928 
 929     if scale_to_absolute:
 930       groundtruth_boxes = output_dict[input_data_fields.groundtruth_boxes]
 931       output_dict[input_data_fields.groundtruth_boxes] = (
 932           shape_utils.static_or_dynamic_map_fn(
 933               _scale_box_to_absolute,
 934               elems=[groundtruth_boxes, original_image_spatial_shapes],
 935               dtype=tf.float32))
 936       if input_data_fields.groundtruth_keypoints in groundtruth:
 937         groundtruth_keypoints = output_dict[
 938             input_data_fields.groundtruth_keypoints]
 939         output_dict[input_data_fields.groundtruth_keypoints] = (
 940             shape_utils.static_or_dynamic_map_fn(
 941                 _scale_keypoint_to_absolute,
 942                 elems=[groundtruth_keypoints, original_image_spatial_shapes],
 943                 dtype=tf.float32))
 944 
 945     # For class-agnostic models, groundtruth classes all become 1.
 946     if class_agnostic:
 947       groundtruth_classes = groundtruth[input_data_fields.groundtruth_classes]
 948       groundtruth_classes = tf.ones_like(groundtruth_classes, dtype=tf.int64)
 949       output_dict[input_data_fields.groundtruth_classes] = groundtruth_classes
 950 
 951     output_dict[input_data_fields.num_groundtruth_boxes] = max_gt_boxes
 952 
 953   return output_dict

 956 def get_evaluators(eval_config, categories, evaluator_options=None):
 957   """Returns the evaluator class according to eval_config, valid for categories.
 958 
 959   Args:
 960     eval_config: An `eval_pb2.EvalConfig`.
 961     categories: A list of dicts, each of which has the following keys -
 962         'id': (required) an integer id uniquely identifying this category.
 963         'name': (required) string representing category name e.g., 'cat', 'dog'.
 964         'keypoints': (optional) dict mapping this category's keypoints to unique
 965           ids.
 966     evaluator_options: A dictionary of metric names (see
 967       EVAL_METRICS_CLASS_DICT) to `DetectionEvaluator` initialization
 968       keyword arguments. For example:
 969       evalator_options = {
 970         'coco_detection_metrics': {'include_metrics_per_category': True}
 971       }
 972 
 973   Returns:
 974     An list of instances of DetectionEvaluator.
 975 
 976   Raises:
 977     ValueError: if metric is not in the metric class dictionary.
 978   """
 979   evaluator_options = evaluator_options or {}
 980   eval_metric_fn_keys = eval_config.metrics_set
 981   if not eval_metric_fn_keys:
 982     eval_metric_fn_keys = [EVAL_DEFAULT_METRIC]
 983   evaluators_list = []
 984   for eval_metric_fn_key in eval_metric_fn_keys:
 985     if eval_metric_fn_key not in EVAL_METRICS_CLASS_DICT:
 986       raise ValueError('Metric not found: {}'.format(eval_metric_fn_key))
 987     kwargs_dict = (evaluator_options[eval_metric_fn_key] if eval_metric_fn_key
 988                    in evaluator_options else {})
 989     evaluators_list.append(EVAL_METRICS_CLASS_DICT[eval_metric_fn_key](
 990         categories,
 991         **kwargs_dict))
 992 
 993   if isinstance(eval_config, eval_pb2.EvalConfig):
 994     parameterized_metrics = eval_config.parameterized_metric
 995     for parameterized_metric in parameterized_metrics:
 996       assert parameterized_metric.HasField('parameterized_metric')
 997       if parameterized_metric.WhichOneof(
 998           'parameterized_metric') == EVAL_KEYPOINT_METRIC:
 999         keypoint_metrics = parameterized_metric.coco_keypoint_metrics
1000         # Create category to keypoints mapping dict.
1001         category_keypoints = {}
1002         class_label = keypoint_metrics.class_label
1003         category = None
1004         for cat in categories:
1005           if cat['name'] == class_label:
1006             category = cat
1007             break
1008         if not category:
1009           continue
1010         keypoints_for_this_class = category['keypoints']
1011         category_keypoints = [{
1012             'id': keypoints_for_this_class[kp_name], 'name': kp_name
1013         } for kp_name in keypoints_for_this_class]
1014         # Create keypoint evaluator for this category.
1015         evaluators_list.append(EVAL_METRICS_CLASS_DICT[EVAL_KEYPOINT_METRIC](
1016             category['id'], category_keypoints, class_label,
1017             keypoint_metrics.keypoint_label_to_sigmas))
1018   return evaluators_list

1021 def get_eval_metric_ops_for_evaluators(eval_config,
1022                                        categories,
1023                                        eval_dict):
1024   """Returns eval metrics ops to use with `tf.estimator.EstimatorSpec`.
1025 
1026   Args:
1027     eval_config: An `eval_pb2.EvalConfig`.
1028     categories: A list of dicts, each of which has the following keys -
1029         'id': (required) an integer id uniquely identifying this category.
1030         'name': (required) string representing category name e.g., 'cat', 'dog'.
1031     eval_dict: An evaluation dictionary, returned from
1032       result_dict_for_single_example().
1033 
1034   Returns:
1035     A dictionary of metric names to tuple of value_op and update_op that can be
1036     used as eval metric ops in tf.EstimatorSpec.
1037   """
1038   eval_metric_ops = {}
1039   evaluator_options = evaluator_options_from_eval_config(eval_config)
1040   evaluators_list = get_evaluators(eval_config, categories, evaluator_options)
1041   for evaluator in evaluators_list:
1042     eval_metric_ops.update(evaluator.get_estimator_eval_metric_ops(
1043         eval_dict))
1044   return eval_metric_ops

1047 def evaluator_options_from_eval_config(eval_config):
1048   """Produces a dictionary of evaluation options for each eval metric.
1049 
1050   Args:
1051     eval_config: An `eval_pb2.EvalConfig`.
1052 
1053   Returns:
1054     evaluator_options: A dictionary of metric names (see
1055       EVAL_METRICS_CLASS_DICT) to `DetectionEvaluator` initialization
1056       keyword arguments. For example:
1057       evalator_options = {
1058         'coco_detection_metrics': {'include_metrics_per_category': True}
