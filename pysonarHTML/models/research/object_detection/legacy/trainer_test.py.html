  32 def get_input_function():
  33   """A function to get test inputs. Returns an image with one box."""
  34   image = tf.random_uniform([32, 32, 3], dtype=tf.float32)
  35   key = tf.constant('image_000000')
  36   class_label = tf.random_uniform(
  37       [1], minval=0, maxval=NUMBER_OF_CLASSES, dtype=tf.int32)
  38   box_label = tf.random_uniform(
  39       [1, 4], minval=0.4, maxval=0.6, dtype=tf.float32)
  40   multiclass_scores = tf.random_uniform(
  41       [1, NUMBER_OF_CLASSES], minval=0.4, maxval=0.6, dtype=tf.float32)
  42 
  43   return {
  44       fields.InputDataFields.image: image,
  45       fields.InputDataFields.key: key,
  46       fields.InputDataFields.groundtruth_classes: class_label,
  47       fields.InputDataFields.groundtruth_boxes: box_label,
  48       fields.InputDataFields.multiclass_scores: multiclass_scores
  49   }

  52 class FakeDetectionModel(model.DetectionModel):
  53   """A simple (and poor) DetectionModel for use in test."""
  54 
  55   def __init__(self):
  56     super(FakeDetectionModel, self).__init__(num_classes=NUMBER_OF_CLASSES)
  57     self._classification_loss = losses.WeightedSigmoidClassificationLoss()
  58     self._localization_loss = losses.WeightedSmoothL1LocalizationLoss()
  59 
  60   def preprocess(self, inputs):
  61     """Input preprocessing, resizes images to 28x28.
  62 
  63     Args:
  64       inputs: a [batch, height_in, width_in, channels] float32 tensor
  65         representing a batch of images with values between 0 and 255.0.
  66 
  67     Returns:
  68       preprocessed_inputs: a [batch, 28, 28, channels] float32 tensor.
  69       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
  70         of the form [height, width, channels] indicating the shapes
  71         of true images in the resized images, as resized images can be padded
  72         with zeros.
  73     """
  74     true_image_shapes = [inputs.shape[:-1].as_list()
  75                          for _ in range(inputs.shape[-1])]
  76     return tf.image.resize_images(inputs, [28, 28]), true_image_shapes
  77 
  78   def predict(self, preprocessed_inputs, true_image_shapes):
  79     """Prediction tensors from inputs tensor.
  80 
  81     Args:
  82       preprocessed_inputs: a [batch, 28, 28, channels] float32 tensor.
  83       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
  84         of the form [height, width, channels] indicating the shapes
  85         of true images in the resized images, as resized images can be padded
  86         with zeros.
  87 
  88     Returns:
  89       prediction_dict: a dictionary holding prediction tensors to be
  90         passed to the Loss or Postprocess functions.
  91     """
  92     flattened_inputs = slim.flatten(preprocessed_inputs)
  93     class_prediction = slim.fully_connected(flattened_inputs, self._num_classes)
  94     box_prediction = slim.fully_connected(flattened_inputs, 4)
  95 
  96     return {
  97         'class_predictions_with_background': tf.reshape(
  98             class_prediction, [-1, 1, self._num_classes]),
  99         'box_encodings': tf.reshape(box_prediction, [-1, 1, 4])
 100     }
 101 
 102   def postprocess(self, prediction_dict, true_image_shapes, **params):
 103     """Convert predicted output tensors to final detections. Unused.
 104 
 105     Args:
 106       prediction_dict: a dictionary holding prediction tensors.
 107       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 108         of the form [height, width, channels] indicating the shapes
 109         of true images in the resized images, as resized images can be padded
 110         with zeros.
 111       **params: Additional keyword arguments for specific implementations of
 112         DetectionModel.
 113 
 114     Returns:
 115       detections: a dictionary with empty fields.
 116     """
 117     return {
 118         'detection_boxes': None,
 119         'detection_scores': None,
 120         'detection_classes': None,
 121         'num_detections': None
 122     }
 123 
 124   def loss(self, prediction_dict, true_image_shapes):
 125     """Compute scalar loss tensors with respect to provided groundtruth.
 126 
 127     Calling this function requires that groundtruth tensors have been
 128     provided via the provide_groundtruth function.
 129 
 130     Args:
 131       prediction_dict: a dictionary holding predicted tensors
 132       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 133         of the form [height, width, channels] indicating the shapes
 134         of true images in the resized images, as resized images can be padded
 135         with zeros.
 136 
 137     Returns:
 138       a dictionary mapping strings (loss names) to scalar tensors representing
 139         loss values.
 140     """
 141     batch_reg_targets = tf.stack(
 142         self.groundtruth_lists(fields.BoxListFields.boxes))
 143     batch_cls_targets = tf.stack(
 144         self.groundtruth_lists(fields.BoxListFields.classes))
 145     weights = tf.constant(
 146         1.0, dtype=tf.float32,
 147         shape=[len(self.groundtruth_lists(fields.BoxListFields.boxes)), 1])
 148 
 149     location_losses = self._localization_loss(
 150         prediction_dict['box_encodings'], batch_reg_targets,
 151         weights=weights)
 152     cls_losses = self._classification_loss(
 153         prediction_dict['class_predictions_with_background'], batch_cls_targets,
 154         weights=weights)
 155 
 156     loss_dict = {
 157         'localization_loss': tf.reduce_sum(location_losses),
 158         'classification_loss': tf.reduce_sum(cls_losses),
 159     }
 160     return loss_dict

 173   def restore_map(self, fine_tune_checkpoint_type='detection'):
 174     """Returns a map of variables to load from a foreign checkpoint.
 175 
 176     Args:
 177       fine_tune_checkpoint_type: whether to restore from a full detection
 178         checkpoint (with compatible variable names) or to restore from a
 179         classification checkpoint for initialization prior to training.
 180         Valid values: `detection`, `classification`. Default 'detection'.
 181 
 182     Returns:
 183       A dict mapping variable names to variables.
 184     """
 185     return {var.op.name: var for var in tf.global_variables()}

 202   def test_configure_trainer_and_train_two_steps(self):
 203     train_config_text_proto = """
 204     optimizer {
 205       adam_optimizer {
 206         learning_rate {
 207           constant_learning_rate {
 208             learning_rate: 0.01
 209           }
 210         }
 211       }
 212     }
 213     data_augmentation_options {
 214       random_adjust_brightness {
 215         max_delta: 0.2
 216       }
 217     }
 218     data_augmentation_options {
 219       random_adjust_contrast {
 220         min_delta: 0.7
 221         max_delta: 1.1
 222       }
 223     }
 224     num_steps: 2
 225     """
 226     train_config = train_pb2.TrainConfig()
 227     text_format.Merge(train_config_text_proto, train_config)
 228 
 229     train_dir = self.get_temp_dir()
 230 
 231     trainer.train(
 232         create_tensor_dict_fn=get_input_function,
 233         create_model_fn=FakeDetectionModel,
 234         train_config=train_config,
 235         master='',
 236         task=0,
 237         num_clones=1,
 238         worker_replicas=1,
 239         clone_on_cpu=True,
 240         ps_tasks=0,
 241         worker_job_name='worker',
 242         is_chief=True,
 243         train_dir=train_dir)
