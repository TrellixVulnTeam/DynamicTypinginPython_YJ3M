  37 def create_input_queue(batch_size_per_clone, create_tensor_dict_fn,
  38                        batch_queue_capacity, num_batch_queue_threads,
  39                        prefetch_queue_capacity, data_augmentation_options):
  40   """Sets up reader, prefetcher and returns input queue.
  41 
  42   Args:
  43     batch_size_per_clone: batch size to use per clone.
  44     create_tensor_dict_fn: function to create tensor dictionary.
  45     batch_queue_capacity: maximum number of elements to store within a queue.
  46     num_batch_queue_threads: number of threads to use for batching.
  47     prefetch_queue_capacity: maximum capacity of the queue used to prefetch
  48                              assembled batches.
  49     data_augmentation_options: a list of tuples, where each tuple contains a
  50       data augmentation function and a dictionary containing arguments and their
  51       values (see preprocessor.py).
  52 
  53   Returns:
  54     input queue: a batcher.BatchQueue object holding enqueued tensor_dicts
  55       (which hold images, boxes and targets).  To get a batch of tensor_dicts,
  56       call input_queue.Dequeue().
  57   """
  58   tensor_dict = create_tensor_dict_fn()
  59 
  60   tensor_dict[fields.InputDataFields.image] = tf.expand_dims(
  61       tensor_dict[fields.InputDataFields.image], 0)
  62 
  63   images = tensor_dict[fields.InputDataFields.image]
  64   float_images = tf.cast(images, dtype=tf.float32)
  65   tensor_dict[fields.InputDataFields.image] = float_images
  66 
  67   include_instance_masks = (fields.InputDataFields.groundtruth_instance_masks
  68                             in tensor_dict)
  69   include_keypoints = (fields.InputDataFields.groundtruth_keypoints
  70                        in tensor_dict)
  71   include_multiclass_scores = (fields.InputDataFields.multiclass_scores
  72                                in tensor_dict)
  73   if data_augmentation_options:
  74     tensor_dict = preprocessor.preprocess(
  75         tensor_dict, data_augmentation_options,
  76         func_arg_map=preprocessor.get_default_func_arg_map(
  77             include_label_weights=True,
  78             include_multiclass_scores=include_multiclass_scores,
  79             include_instance_masks=include_instance_masks,
  80             include_keypoints=include_keypoints))
  81 
  82   input_queue = batcher.BatchQueue(
  83       tensor_dict,
  84       batch_size=batch_size_per_clone,
  85       batch_queue_capacity=batch_queue_capacity,
  86       num_batch_queue_threads=num_batch_queue_threads,
  87       prefetch_queue_capacity=prefetch_queue_capacity)
  88   return input_queue

  91 def get_inputs(input_queue,
  92                num_classes,
  93                merge_multiple_label_boxes=False,
  94                use_multiclass_scores=False):
  95   """Dequeues batch and constructs inputs to object detection model.
  96 
  97   Args:
  98     input_queue: BatchQueue object holding enqueued tensor_dicts.
  99     num_classes: Number of classes.
 100     merge_multiple_label_boxes: Whether to merge boxes with multiple labels
 101       or not. Defaults to false. Merged boxes are represented with a single
 102       box and a k-hot encoding of the multiple labels associated with the
 103       boxes.
 104     use_multiclass_scores: Whether to use multiclass scores instead of
 105       groundtruth_classes.
 106 
 107   Returns:
 108     images: a list of 3-D float tensor of images.
 109     image_keys: a list of string keys for the images.
 110     locations_list: a list of tensors of shape [num_boxes, 4]
 111       containing the corners of the groundtruth boxes.
 112     classes_list: a list of padded one-hot (or K-hot) float32 tensors containing
 113       target classes.
 114     masks_list: a list of 3-D float tensors of shape [num_boxes, image_height,
 115       image_width] containing instance masks for objects if present in the
 116       input_queue. Else returns None.
 117     keypoints_list: a list of 3-D float tensors of shape [num_boxes,
 118       num_keypoints, 2] containing keypoints for objects if present in the
 119       input queue. Else returns None.
 120     weights_lists: a list of 1-D float32 tensors of shape [num_boxes]
 121       containing groundtruth weight for each box.
 122   """
 123   read_data_list = input_queue.dequeue()
 124   label_id_offset = 1
 125   def extract_images_and_targets(read_data):
 126     """Extract images and targets from the input dict."""
 127     image = read_data[fields.InputDataFields.image]
 128     key = ''
 129     if fields.InputDataFields.source_id in read_data:
 130       key = read_data[fields.InputDataFields.source_id]
 131     location_gt = read_data[fields.InputDataFields.groundtruth_boxes]
 132     classes_gt = tf.cast(read_data[fields.InputDataFields.groundtruth_classes],
 133                          tf.int32)
 134     classes_gt -= label_id_offset
 135 
 136     if merge_multiple_label_boxes and use_multiclass_scores:
 137       raise ValueError(
 138           'Using both merge_multiple_label_boxes and use_multiclass_scores is'
 139           'not supported'
 140       )
 141 
 142     if merge_multiple_label_boxes:
 143       location_gt, classes_gt, _ = util_ops.merge_boxes_with_multiple_labels(
 144           location_gt, classes_gt, num_classes)
 145       classes_gt = tf.cast(classes_gt, tf.float32)
 146     elif use_multiclass_scores:
 147       classes_gt = tf.cast(read_data[fields.InputDataFields.multiclass_scores],
 148                            tf.float32)
 149     else:
 150       classes_gt = util_ops.padded_one_hot_encoding(
 151           indices=classes_gt, depth=num_classes, left_pad=0)
 152     masks_gt = read_data.get(fields.InputDataFields.groundtruth_instance_masks)
 153     keypoints_gt = read_data.get(fields.InputDataFields.groundtruth_keypoints)
 154     if (merge_multiple_label_boxes and (
 155         masks_gt is not None or keypoints_gt is not None)):
 156       raise NotImplementedError('Multi-label support is only for boxes.')
 157     weights_gt = read_data.get(
 158         fields.InputDataFields.groundtruth_weights)
 159     return (image, key, location_gt, classes_gt, masks_gt, keypoints_gt,
 160             weights_gt)
 161 
 162   return zip(*map(extract_images_and_targets, read_data_list))

 165 def _create_losses(input_queue, create_model_fn, train_config):
 166   """Creates loss function for a DetectionModel.
 167 
 168   Args:
 169     input_queue: BatchQueue object holding enqueued tensor_dicts.
 170     create_model_fn: A function to create the DetectionModel.
 171     train_config: a train_pb2.TrainConfig protobuf.
 172   """
 173   detection_model = create_model_fn()
 174   (images, _, groundtruth_boxes_list, groundtruth_classes_list,
 175    groundtruth_masks_list, groundtruth_keypoints_list,
 176    groundtruth_weights_list) = get_inputs(
 177        input_queue,
 178        detection_model.num_classes,
 179        train_config.merge_multiple_label_boxes,
 180        train_config.use_multiclass_scores)
 181 
 182   preprocessed_images = []
 183   true_image_shapes = []
 184   for image in images:
 185     resized_image, true_image_shape = detection_model.preprocess(image)
 186     preprocessed_images.append(resized_image)
 187     true_image_shapes.append(true_image_shape)
 188 
 189   images = tf.concat(preprocessed_images, 0)
 190   true_image_shapes = tf.concat(true_image_shapes, 0)
 191 
 192   if any(mask is None for mask in groundtruth_masks_list):
 193     groundtruth_masks_list = None
 194   if any(keypoints is None for keypoints in groundtruth_keypoints_list):
 195     groundtruth_keypoints_list = None
 196 
 197   detection_model.provide_groundtruth(
 198       groundtruth_boxes_list,
 199       groundtruth_classes_list,
 200       groundtruth_masks_list,
 201       groundtruth_keypoints_list,
 202       groundtruth_weights_list=groundtruth_weights_list)
 203   prediction_dict = detection_model.predict(images, true_image_shapes)
 204 
 205   losses_dict = detection_model.loss(prediction_dict, true_image_shapes)
 206   for loss_tensor in losses_dict.values():
 207     tf.losses.add_loss(loss_tensor)

 210 def train(create_tensor_dict_fn,
 211           create_model_fn,
 212           train_config,
 213           master,
 214           task,
 215           num_clones,
 216           worker_replicas,
 217           clone_on_cpu,
 218           ps_tasks,
 219           worker_job_name,
 220           is_chief,
 221           train_dir,
 222           graph_hook_fn=None):
 223   """Training function for detection models.
 224 
 225   Args:
 226     create_tensor_dict_fn: a function to create a tensor input dictionary.
 227     create_model_fn: a function that creates a DetectionModel and generates
 228                      losses.
 229     train_config: a train_pb2.TrainConfig protobuf.
 230     master: BNS name of the TensorFlow master to use.
 231     task: The task id of this training instance.
 232     num_clones: The number of clones to run per machine.
 233     worker_replicas: The number of work replicas to train with.
 234     clone_on_cpu: True if clones should be forced to run on CPU.
 235     ps_tasks: Number of parameter server tasks.
 236     worker_job_name: Name of the worker job.
 237     is_chief: Whether this replica is the chief replica.
 238     train_dir: Directory to write checkpoints and training summaries to.
 239     graph_hook_fn: Optional function that is called after the inference graph is
 240       built (before optimization). This is helpful to perform additional changes
 241       to the training graph such as adding FakeQuant ops. The function should
 242       modify the default graph.
 243 
 244   Raises:
 245     ValueError: If both num_clones > 1 and train_config.sync_replicas is true.
 246   """
 247 
 248   detection_model = create_model_fn()
 249   data_augmentation_options = [
 250       preprocessor_builder.build(step)
 251       for step in train_config.data_augmentation_options]
 252 
 253   with tf.Graph().as_default():
 254     # Build a configuration specifying multi-GPU and multi-replicas.
 255     deploy_config = model_deploy.DeploymentConfig(
 256         num_clones=num_clones,
 257         clone_on_cpu=clone_on_cpu,
 258         replica_id=task,
 259         num_replicas=worker_replicas,
 260         num_ps_tasks=ps_tasks,
 261         worker_job_name=worker_job_name)
 262 
 263     # Place the global step on the device storing the variables.
 264     with tf.device(deploy_config.variables_device()):
 265       global_step = slim.create_global_step()
 266 
 267     if num_clones != 1 and train_config.sync_replicas:
 268       raise ValueError('In Synchronous SGD mode num_clones must ',
 269                        'be 1. Found num_clones: {}'.format(num_clones))
 270     batch_size = train_config.batch_size // num_clones
 271     if train_config.sync_replicas:
 272       batch_size //= train_config.replicas_to_aggregate
 273 
 274     with tf.device(deploy_config.inputs_device()):
 275       input_queue = create_input_queue(
 276           batch_size, create_tensor_dict_fn,
 277           train_config.batch_queue_capacity,
 278           train_config.num_batch_queue_threads,
 279           train_config.prefetch_queue_capacity, data_augmentation_options)
 280 
 281     # Gather initial summaries.
 282     # TODO(rathodv): See if summaries can be added/extracted from global tf
 283     # collections so that they don't have to be passed around.
 284     summaries = set(tf.get_collection(tf.GraphKeys.SUMMARIES))
 285     global_summaries = set([])
 286 
 287     model_fn = functools.partial(_create_losses,
 288                                  create_model_fn=create_model_fn,
 289                                  train_config=train_config)
 290     clones = model_deploy.create_clones(deploy_config, model_fn, [input_queue])
 291     first_clone_scope = clones[0].scope
 292 
 293     if graph_hook_fn:
 294       with tf.device(deploy_config.variables_device()):
 295         graph_hook_fn()
 296 
 297     # Gather update_ops from the first clone. These contain, for example,
 298     # the updates for the batch_norm variables created by model_fn.
 299     update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS, first_clone_scope)
 300 
 301     with tf.device(deploy_config.optimizer_device()):
 302       training_optimizer, optimizer_summary_vars = optimizer_builder.build(
 303           train_config.optimizer)
 304       for var in optimizer_summary_vars:
 305         tf.summary.scalar(var.op.name, var, family='LearningRate')
 306 
 307     sync_optimizer = None
 308     if train_config.sync_replicas:
 309       training_optimizer = tf.train.SyncReplicasOptimizer(
 310           training_optimizer,
 311           replicas_to_aggregate=train_config.replicas_to_aggregate,
 312           total_num_replicas=worker_replicas)
 313       sync_optimizer = training_optimizer
 314 
 315     with tf.device(deploy_config.optimizer_device()):
 316       regularization_losses = (None if train_config.add_regularization_loss
 317                                else [])
 318       total_loss, grads_and_vars = model_deploy.optimize_clones(
 319           clones, training_optimizer,
 320           regularization_losses=regularization_losses)
 321       total_loss = tf.check_numerics(total_loss, 'LossTensor is inf or nan.')
 322 
 323       # Optionally multiply bias gradients by train_config.bias_grad_multiplier.
 324       if train_config.bias_grad_multiplier:
 325         biases_regex_list = ['.*/biases']
 326         grads_and_vars = variables_helper.multiply_gradients_matching_regex(
 327             grads_and_vars,
 328             biases_regex_list,
 329             multiplier=train_config.bias_grad_multiplier)
 330 
 331       # Optionally freeze some layers by setting their gradients to be zero.
 332       if train_config.freeze_variables:
 333         grads_and_vars = variables_helper.freeze_gradients_matching_regex(
 334             grads_and_vars, train_config.freeze_variables)
 335 
 336       # Optionally clip gradients
 337       if train_config.gradient_clipping_by_norm > 0:
 338         with tf.name_scope('clip_grads'):
 339           grads_and_vars = slim.learning.clip_gradient_norms(
 340               grads_and_vars, train_config.gradient_clipping_by_norm)
 341 
 342       # Create gradient updates.
 343       grad_updates = training_optimizer.apply_gradients(grads_and_vars,
 344                                                         global_step=global_step)
 345       update_ops.append(grad_updates)
 346       update_op = tf.group(*update_ops, name='update_barrier')
 347       with tf.control_dependencies([update_op]):
 348         train_tensor = tf.identity(total_loss, name='train_op')
 349 
 350     # Add summaries.
 351     for model_var in slim.get_model_variables():
 352       global_summaries.add(tf.summary.histogram('ModelVars/' +
 353                                                 model_var.op.name, model_var))
 354     for loss_tensor in tf.losses.get_losses():
 355       global_summaries.add(tf.summary.scalar('Losses/' + loss_tensor.op.name,
 356                                              loss_tensor))
 357     global_summaries.add(
 358         tf.summary.scalar('Losses/TotalLoss', tf.losses.get_total_loss()))
 359 
 360     # Add the summaries from the first clone. These contain the summaries
 361     # created by model_fn and either optimize_clones() or _gather_clone_loss().
 362     summaries |= set(tf.get_collection(tf.GraphKeys.SUMMARIES,
 363                                        first_clone_scope))
 364     summaries |= global_summaries
 365 
 366     # Merge all summaries together.
 367     summary_op = tf.summary.merge(list(summaries), name='summary_op')
 368 
 369     # Soft placement allows placing on CPU ops without GPU implementation.
 370     session_config = tf.ConfigProto(allow_soft_placement=True,
 371                                     log_device_placement=False)
 372 
 373     # Save checkpoints regularly.
 374     keep_checkpoint_every_n_hours = train_config.keep_checkpoint_every_n_hours
 375     saver = tf.train.Saver(
 376         keep_checkpoint_every_n_hours=keep_checkpoint_every_n_hours)
 377 
 378     # Create ops required to initialize the model from a given checkpoint.
 379     init_fn = None
 380     if train_config.fine_tune_checkpoint:
 381       if not train_config.fine_tune_checkpoint_type:
 382         # train_config.from_detection_checkpoint field is deprecated. For
 383         # backward compatibility, fine_tune_checkpoint_type is set based on
 384         # from_detection_checkpoint.
 385         if train_config.from_detection_checkpoint:
 386           train_config.fine_tune_checkpoint_type = 'detection'
 387         else:
 388           train_config.fine_tune_checkpoint_type = 'classification'
 389       var_map = detection_model.restore_map(
 390           fine_tune_checkpoint_type=train_config.fine_tune_checkpoint_type,
 391           load_all_detection_checkpoint_vars=(
 392               train_config.load_all_detection_checkpoint_vars))
 393       available_var_map = (variables_helper.
 394                            get_variables_available_in_checkpoint(
 395                                var_map, train_config.fine_tune_checkpoint,
 396                                include_global_step=False))
 397       init_saver = tf.train.Saver(available_var_map)
 398       def initializer_fn(sess):
 399         init_saver.restore(sess, train_config.fine_tune_checkpoint)
 400       init_fn = initializer_fn
 401 
 402     slim.learning.train(
 403         train_tensor,
 404         logdir=train_dir,
 405         master=master,
 406         is_chief=is_chief,
 407         session_config=session_config,
 408         startup_delay_steps=train_config.startup_delay_steps,
 409         init_fn=init_fn,
 410         summary_op=summary_op,
 411         number_of_steps=(
 412             train_config.num_steps if train_config.num_steps else None),
 413         save_summaries_secs=120,
 414         sync_optimizer=sync_optimizer,
 415         saver=saver)
