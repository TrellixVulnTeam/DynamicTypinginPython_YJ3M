  64 def _extract_predictions_and_losses(model,
  65                                     create_input_dict_fn,
  66                                     ignore_groundtruth=False):
  67   """Constructs tensorflow detection graph and returns output tensors.
  68 
  69   Args:
  70     model: model to perform predictions with.
  71     create_input_dict_fn: function to create input tensor dictionaries.
  72     ignore_groundtruth: whether groundtruth should be ignored.
  73 
  74   Returns:
  75     prediction_groundtruth_dict: A dictionary with postprocessed tensors (keyed
  76       by standard_fields.DetectionResultsFields) and optional groundtruth
  77       tensors (keyed by standard_fields.InputDataFields).
  78     losses_dict: A dictionary containing detection losses. This is empty when
  79       ignore_groundtruth is true.
  80   """
  81   input_dict = create_input_dict_fn()
  82   prefetch_queue = prefetcher.prefetch(input_dict, capacity=500)
  83   input_dict = prefetch_queue.dequeue()
  84   original_image = tf.expand_dims(input_dict[fields.InputDataFields.image], 0)
  85   preprocessed_image, true_image_shapes = model.preprocess(
  86       tf.cast(original_image, dtype=tf.float32))
  87   prediction_dict = model.predict(preprocessed_image, true_image_shapes)
  88   detections = model.postprocess(prediction_dict, true_image_shapes)
  89 
  90   groundtruth = None
  91   losses_dict = {}
  92   if not ignore_groundtruth:
  93     groundtruth = {
  94         fields.InputDataFields.groundtruth_boxes:
  95             input_dict[fields.InputDataFields.groundtruth_boxes],
  96         fields.InputDataFields.groundtruth_classes:
  97             input_dict[fields.InputDataFields.groundtruth_classes],
  98         fields.InputDataFields.groundtruth_area:
  99             input_dict[fields.InputDataFields.groundtruth_area],
 100         fields.InputDataFields.groundtruth_is_crowd:
 101             input_dict[fields.InputDataFields.groundtruth_is_crowd],
 102         fields.InputDataFields.groundtruth_difficult:
 103             input_dict[fields.InputDataFields.groundtruth_difficult]
 104     }
 105     if fields.InputDataFields.groundtruth_group_of in input_dict:
 106       groundtruth[fields.InputDataFields.groundtruth_group_of] = (
 107           input_dict[fields.InputDataFields.groundtruth_group_of])
 108     groundtruth_masks_list = None
 109     if fields.DetectionResultFields.detection_masks in detections:
 110       groundtruth[fields.InputDataFields.groundtruth_instance_masks] = (
 111           input_dict[fields.InputDataFields.groundtruth_instance_masks])
 112       groundtruth_masks_list = [
 113           input_dict[fields.InputDataFields.groundtruth_instance_masks]]
 114     groundtruth_keypoints_list = None
 115     if fields.DetectionResultFields.detection_keypoints in detections:
 116       groundtruth[fields.InputDataFields.groundtruth_keypoints] = (
 117           input_dict[fields.InputDataFields.groundtruth_keypoints])
 118       groundtruth_keypoints_list = [
 119           input_dict[fields.InputDataFields.groundtruth_keypoints]]
 120     label_id_offset = 1
 121     model.provide_groundtruth(
 122         [input_dict[fields.InputDataFields.groundtruth_boxes]],
 123         [tf.one_hot(input_dict[fields.InputDataFields.groundtruth_classes]
 124                     - label_id_offset, depth=model.num_classes)],
 125         groundtruth_masks_list, groundtruth_keypoints_list)
 126     losses_dict.update(model.loss(prediction_dict, true_image_shapes))
 127 
 128   result_dict = eval_util.result_dict_for_single_example(
 129       original_image,
 130       input_dict[fields.InputDataFields.source_id],
 131       detections,
 132       groundtruth,
 133       class_agnostic=(
 134           fields.DetectionResultFields.detection_classes not in detections),
 135       scale_to_absolute=True)
 136   return result_dict, losses_dict

 139 def get_evaluators(eval_config, categories):
 140   """Returns the evaluator class according to eval_config, valid for categories.
 141 
 142   Args:
 143     eval_config: evaluation configurations.
 144     categories: a list of categories to evaluate.
 145   Returns:
 146     An list of instances of DetectionEvaluator.
 147 
 148   Raises:
 149     ValueError: if metric is not in the metric class dictionary.
 150   """
 151   eval_metric_fn_keys = eval_config.metrics_set
 152   if not eval_metric_fn_keys:
 153     eval_metric_fn_keys = [EVAL_DEFAULT_METRIC]
 154   evaluators_list = []
 155   for eval_metric_fn_key in eval_metric_fn_keys:
 156     if eval_metric_fn_key not in EVAL_METRICS_CLASS_DICT:
 157       raise ValueError('Metric not found: {}'.format(eval_metric_fn_key))
 158     if eval_metric_fn_key == 'oid_challenge_object_detection_metrics':
 159       logging.warning(
 160           'oid_challenge_object_detection_metrics is deprecated; '
 161           'use oid_challenge_detection_metrics instead'
 162       )
 163     if eval_metric_fn_key == 'oid_V2_detection_metrics':
 164       logging.warning(
 165           'open_images_V2_detection_metrics is deprecated; '
 166           'use oid_V2_detection_metrics instead'
 167       )
 168     evaluators_list.append(
 169         EVAL_METRICS_CLASS_DICT[eval_metric_fn_key](categories=categories))
 170   return evaluators_list

 173 def evaluate(create_input_dict_fn, create_model_fn, eval_config, categories,
 174              checkpoint_dir, eval_dir, graph_hook_fn=None, evaluator_list=None):
 175   """Evaluation function for detection models.
 176 
 177   Args:
 178     create_input_dict_fn: a function to create a tensor input dictionary.
 179     create_model_fn: a function that creates a DetectionModel.
 180     eval_config: a eval_pb2.EvalConfig protobuf.
 181     categories: a list of category dictionaries. Each dict in the list should
 182                 have an integer 'id' field and string 'name' field.
 183     checkpoint_dir: directory to load the checkpoints to evaluate from.
 184     eval_dir: directory to write evaluation metrics summary to.
 185     graph_hook_fn: Optional function that is called after the training graph is
 186       completely built. This is helpful to perform additional changes to the
 187       training graph such as optimizing batchnorm. The function should modify
 188       the default graph.
 189     evaluator_list: Optional list of instances of DetectionEvaluator. If not
 190       given, this list of metrics is created according to the eval_config.
 191 
 192   Returns:
 193     metrics: A dictionary containing metric names and values from the latest
 194       run.
 195   """
 196 
 197   model = create_model_fn()
 198 
 199   if eval_config.ignore_groundtruth and not eval_config.export_path:
 200     logging.fatal('If ignore_groundtruth=True then an export_path is '
 201                   'required. Aborting!!!')
 202 
 203   tensor_dict, losses_dict = _extract_predictions_and_losses(
 204       model=model,
 205       create_input_dict_fn=create_input_dict_fn,
 206       ignore_groundtruth=eval_config.ignore_groundtruth)
 207 
 208   def _process_batch(tensor_dict, sess, batch_index, counters,
 209                      losses_dict=None):
 210     """Evaluates tensors in tensor_dict, losses_dict and visualizes examples.
 211 
 212     This function calls sess.run on tensor_dict, evaluating the original_image
 213     tensor only on the first K examples and visualizing detections overlaid
 214     on this original_image.
 215 
 216     Args:
 217       tensor_dict: a dictionary of tensors
 218       sess: tensorflow session
 219       batch_index: the index of the batch amongst all batches in the run.
 220       counters: a dictionary holding 'success' and 'skipped' fields which can
 221         be updated to keep track of number of successful and failed runs,
 222         respectively.  If these fields are not updated, then the success/skipped
 223         counter values shown at the end of evaluation will be incorrect.
 224       losses_dict: Optional dictonary of scalar loss tensors.
 225 
 226     Returns:
 227       result_dict: a dictionary of numpy arrays
 228       result_losses_dict: a dictionary of scalar losses. This is empty if input
 229         losses_dict is None.
 230     """
 231     try:
 232       if not losses_dict:
 233         losses_dict = {}
 234       result_dict, result_losses_dict = sess.run([tensor_dict, losses_dict])
 235       counters['success'] += 1
 236     except tf.errors.InvalidArgumentError:
 237       logging.info('Skipping image')
 238       counters['skipped'] += 1
 239       return {}, {}
 240     global_step = tf.train.global_step(sess, tf.train.get_global_step())
 241     if batch_index < eval_config.num_visualizations:
 242       tag = 'image-{}'.format(batch_index)
 243       eval_util.visualize_detection_results(
 244           result_dict,
 245           tag,
 246           global_step,
 247           categories=categories,
 248           summary_dir=eval_dir,
 249           export_dir=eval_config.visualization_export_dir,
 250           show_groundtruth=eval_config.visualize_groundtruth_boxes,
 251           groundtruth_box_visualization_color=eval_config.
 252           groundtruth_box_visualization_color,
 253           min_score_thresh=eval_config.min_score_threshold,
 254           max_num_predictions=eval_config.max_num_boxes_to_visualize,
 255           skip_scores=eval_config.skip_scores,
 256           skip_labels=eval_config.skip_labels,
 257           keep_image_id_for_visualization_export=eval_config.
 258           keep_image_id_for_visualization_export)
 259     return result_dict, result_losses_dict
 260 
 261   if graph_hook_fn: graph_hook_fn()
 262 
 263   variables_to_restore = tf.global_variables()
 264   global_step = tf.train.get_or_create_global_step()
 265   variables_to_restore.append(global_step)
 266 
 267   if eval_config.use_moving_averages:
 268     variable_averages = tf.train.ExponentialMovingAverage(0.0)
 269     variables_to_restore = variable_averages.variables_to_restore()
 270   saver = tf.train.Saver(variables_to_restore)
 271 
 272   def _restore_latest_checkpoint(sess):
 273     latest_checkpoint = tf.train.latest_checkpoint(checkpoint_dir)
 274     saver.restore(sess, latest_checkpoint)
 275 
 276   if not evaluator_list:
 277     evaluator_list = get_evaluators(eval_config, categories)
 278 
 279   metrics = eval_util.repeated_checkpoint_run(
 280       tensor_dict=tensor_dict,
 281       summary_dir=eval_dir,
 282       evaluators=evaluator_list,
 283       batch_processor=_process_batch,
 284       checkpoint_dirs=[checkpoint_dir],
 285       variables_to_restore=None,
 286       restore_fn=_restore_latest_checkpoint,
 287       num_batches=eval_config.num_examples,
 288       eval_interval_secs=eval_config.eval_interval_secs,
 289       max_number_of_evaluations=(1 if eval_config.ignore_groundtruth else
 290                                  eval_config.max_evals
 291                                  if eval_config.max_evals else None),
 292       master=eval_config.eval_master,
 293       save_graph=eval_config.save_graph,
 294       save_graph_dir=(eval_dir if eval_config.save_graph else ''),
 295       losses_dict=losses_dict,
 296       eval_export_path=eval_config.export_path)
 297 
 298   return metrics
