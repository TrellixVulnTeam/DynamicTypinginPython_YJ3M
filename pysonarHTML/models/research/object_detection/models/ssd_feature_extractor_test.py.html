  38   def _build_conv_hyperparams(self, add_batch_norm=True):
  39     conv_hyperparams = hyperparams_pb2.Hyperparams()
  40     conv_hyperparams_text_proto = """
  41       activation: RELU_6
  42       regularizer {
  43         l2_regularizer {
  44         }
  45       }
  46       initializer {
  47         truncated_normal_initializer {
  48         }
  49       }
  50     """
  51     if add_batch_norm:
  52       batch_norm_proto = """
  53         batch_norm {
  54           scale: false
  55         }
  56       """
  57       conv_hyperparams_text_proto += batch_norm_proto
  58     text_format.Merge(conv_hyperparams_text_proto, conv_hyperparams)
  59     return hyperparams_builder.KerasLayerHyperparams(conv_hyperparams)

  62     with slim.arg_scope([]) as sc:

  65   @abstractmethod
  66   def _create_feature_extractor(self,
  67                                 depth_multiplier,
  68                                 pad_to_multiple,
  69                                 use_explicit_padding=False,
  70                                 num_layers=6,
  71                                 use_keras=False,
  72                                 use_depthwise=False):
  73     """Constructs a new feature extractor.
  74 
  75     Args:
  76       depth_multiplier: float depth multiplier for feature extractor
  77       pad_to_multiple: the nearest multiple to zero pad the input height and
  78         width dimensions to.
  79       use_explicit_padding: use 'VALID' padding for convolutions, but prepad
  80         inputs so that the output dimensions are the same as if 'SAME' padding
  81         were used.
  82       num_layers: number of SSD layers.
  83       use_keras: if True builds a keras-based feature extractor, if False builds
  84         a slim-based one.
  85       use_depthwise: Whether to use depthwise convolutions.
  86     Returns:
  87       an ssd_meta_arch.SSDFeatureExtractor or an
  88       ssd_meta_arch.SSDKerasFeatureExtractor object.
  89     """
  90     pass
  91 
  92   def _extract_features(self,
  93                         image_tensor,
  94                         depth_multiplier,
  95                         pad_to_multiple,
  96                         use_explicit_padding=False,
  97                         use_depthwise=False,
  98                         num_layers=6,
  99                         use_keras=False):
 100     kwargs = {}
 101     if use_explicit_padding:
 102       kwargs.update({'use_explicit_padding': use_explicit_padding})
 103     if use_depthwise:
 104       kwargs.update({'use_depthwise': use_depthwise})
 105     if num_layers != 6:
 106       kwargs.update({'num_layers': num_layers})
 107     if use_keras:
 108       kwargs.update({'use_keras': use_keras})
 109     feature_extractor = self._create_feature_extractor(
 110         depth_multiplier,
 111         pad_to_multiple,
 112         **kwargs)
 113     if use_keras:
 114       feature_maps = feature_extractor(image_tensor)
 115     else:
 116       feature_maps = feature_extractor.extract_features(image_tensor)
 117     return feature_maps

 119   def check_extract_features_returns_correct_shape(self,
 120                                                    batch_size,
 121                                                    image_height,
 122                                                    image_width,
 123                                                    depth_multiplier,
 124                                                    pad_to_multiple,
 125                                                    expected_feature_map_shapes,
 126                                                    use_explicit_padding=False,
 127                                                    num_layers=6,
 128                                                    use_keras=False,
 129                                                    use_depthwise=False):
 130 
 131     def graph_fn(image_tensor):
 132       return self._extract_features(
 133           image_tensor,
 134           depth_multiplier,
 135           pad_to_multiple,
 136           use_explicit_padding=use_explicit_padding,
 137           num_layers=num_layers,
 138           use_keras=use_keras,
 139           use_depthwise=use_depthwise)
 140 
 141     image_tensor = np.random.rand(batch_size, image_height, image_width,
 142                                   3).astype(np.float32)
 143     feature_maps = self.execute(graph_fn, [image_tensor])
 144     for feature_map, expected_shape in zip(
 145         feature_maps, expected_feature_map_shapes):
 146       self.assertAllEqual(feature_map.shape, expected_shape)

 148   def check_extract_features_returns_correct_shapes_with_dynamic_inputs(
 149       self,
 150       batch_size,
 151       image_height,
 152       image_width,
 153       depth_multiplier,
 154       pad_to_multiple,
 155       expected_feature_map_shapes,
 156       use_explicit_padding=False,
 157       num_layers=6,
 158       use_keras=False,
 159       use_depthwise=False):
 160 
 161     def graph_fn(image_height, image_width):
 162       image_tensor = tf.random_uniform([batch_size, image_height, image_width,
 163                                         3], dtype=tf.float32)
 164       return self._extract_features(
 165           image_tensor,
 166           depth_multiplier,
 167           pad_to_multiple,
 168           use_explicit_padding=use_explicit_padding,
 169           num_layers=num_layers,
 170           use_keras=use_keras,
 171           use_depthwise=use_depthwise)
 172 
 173     feature_maps = self.execute_cpu(graph_fn, [
 174         np.array(image_height, dtype=np.int32),
 175         np.array(image_width, dtype=np.int32)
 176     ])
 177     for feature_map, expected_shape in zip(
 178         feature_maps, expected_feature_map_shapes):
 179       self.assertAllEqual(feature_map.shape, expected_shape)

 181   def check_extract_features_raises_error_with_invalid_image_size(
 182       self,
 183       image_height,
 184       image_width,
 185       depth_multiplier,
 186       pad_to_multiple,
 187       use_keras=False,
 188       use_depthwise=False):
 189     preprocessed_inputs = tf.placeholder(tf.float32, (4, None, None, 3))
 190     feature_maps = self._extract_features(
 191         preprocessed_inputs,
 192         depth_multiplier,
 193         pad_to_multiple,
 194         use_keras=use_keras,
 195         use_depthwise=use_depthwise)
 196     test_preprocessed_image = np.random.rand(4, image_height, image_width, 3)
 197     with self.test_session() as sess:
 198       sess.run(tf.global_variables_initializer())
 199       with self.assertRaises(tf.errors.InvalidArgumentError):
 200         sess.run(feature_maps,
 201                  feed_dict={preprocessed_inputs: test_preprocessed_image})

 203   def check_feature_extractor_variables_under_scope(self,
 204                                                     depth_multiplier,
 205                                                     pad_to_multiple,
 206                                                     scope_name,
 207                                                     use_keras=False,
 208                                                     use_depthwise=False):
 209     variables = self.get_feature_extractor_variables(
 210         depth_multiplier,
 211         pad_to_multiple,
 212         use_keras=use_keras,
 213         use_depthwise=use_depthwise)
 214     for variable in variables:
 215       self.assertTrue(variable.name.startswith(scope_name))

 217   def get_feature_extractor_variables(self,
 218                                       depth_multiplier,
 219                                       pad_to_multiple,
 220                                       use_keras=False,
 221                                       use_depthwise=False):
 222     g = tf.Graph()
 223     with g.as_default():
 224       preprocessed_inputs = tf.placeholder(tf.float32, (4, None, None, 3))
 225       self._extract_features(
 226           preprocessed_inputs,
 227           depth_multiplier,
 228           pad_to_multiple,
 229           use_keras=use_keras,
 230           use_depthwise=use_depthwise)
 231       return g.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)
