  37 # A modified config of mobilenet v1 that makes it more detection friendly,
  38 def _create_modified_mobilenet_config():
  39   conv_defs = copy.deepcopy(mobilenet_v1.MOBILENETV1_CONV_DEFS)
  40   conv_defs[-2] = mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=2, depth=512)
  41   conv_defs[-1] = mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=256)
  42   return conv_defs

  45 class SSDMobileNetV1FpnFeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
  46   """SSD Feature Extractor using MobilenetV1 FPN features."""
  47 
  48   def __init__(self,
  49                is_training,
  50                depth_multiplier,
  51                min_depth,
  52                pad_to_multiple,
  53                conv_hyperparams_fn,
  54                fpn_min_level=3,
  55                fpn_max_level=7,
  56                additional_layer_depth=256,
  57                reuse_weights=None,
  58                use_explicit_padding=False,
  59                use_depthwise=False,
  60                use_native_resize_op=False,
  61                override_base_feature_extractor_hyperparams=False):
  62     """SSD FPN feature extractor based on Mobilenet v1 architecture.
  63 
  64     Args:
  65       is_training: whether the network is in training mode.
  66       depth_multiplier: float depth multiplier for feature extractor.
  67       min_depth: minimum feature extractor depth.
  68       pad_to_multiple: the nearest multiple to zero pad the input height and
  69         width dimensions to.
  70       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  71         and separable_conv2d ops in the layers that are added on top of the base
  72         feature extractor.
  73       fpn_min_level: the highest resolution feature map to use in FPN. The valid
  74         values are {2, 3, 4, 5} which map to MobileNet v1 layers
  75         {Conv2d_3_pointwise, Conv2d_5_pointwise, Conv2d_11_pointwise,
  76         Conv2d_13_pointwise}, respectively.
  77       fpn_max_level: the smallest resolution feature map to construct or use in
  78         FPN. FPN constructions uses features maps starting from fpn_min_level
  79         upto the fpn_max_level. In the case that there are not enough feature
  80         maps in the backbone network, additional feature maps are created by
  81         applying stride 2 convolutions until we get the desired number of fpn
  82         levels.
  83       additional_layer_depth: additional feature map layer channel depth.
  84       reuse_weights: whether to reuse variables. Default is None.
  85       use_explicit_padding: Whether to use explicit padding when extracting
  86         features. Default is False.
  87       use_depthwise: Whether to use depthwise convolutions. Default is False.
  88       use_native_resize_op: Whether to use tf.image.nearest_neighbor_resize
  89         to do upsampling in FPN. Default is false.
  90       override_base_feature_extractor_hyperparams: Whether to override
  91         hyperparameters of the base feature extractor with the one from
  92         `conv_hyperparams_fn`.
  93     """
  94     super(SSDMobileNetV1FpnFeatureExtractor, self).__init__(
  95         is_training=is_training,
  96         depth_multiplier=depth_multiplier,
  97         min_depth=min_depth,
  98         pad_to_multiple=pad_to_multiple,
  99         conv_hyperparams_fn=conv_hyperparams_fn,
 100         reuse_weights=reuse_weights,
 101         use_explicit_padding=use_explicit_padding,
 102         use_depthwise=use_depthwise,
 103         override_base_feature_extractor_hyperparams=
 104         override_base_feature_extractor_hyperparams)
 105     self._fpn_min_level = fpn_min_level
 106     self._fpn_max_level = fpn_max_level
 107     self._additional_layer_depth = additional_layer_depth
 108     self._conv_defs = None
 109     if self._use_depthwise:
 110       self._conv_defs = _create_modified_mobilenet_config()
 111     self._use_native_resize_op = use_native_resize_op
 112 
 113   def preprocess(self, resized_inputs):
 114     """SSD preprocessing.
 115 
 116     Maps pixel values to the range [-1, 1].
 117 
 118     Args:
 119       resized_inputs: a [batch, height, width, channels] float tensor
 120         representing a batch of images.
 121 
 122     Returns:
 123       preprocessed_inputs: a [batch, height, width, channels] float tensor
 124         representing a batch of images.
 125     """
 126     return (2.0 / 255.0) * resized_inputs - 1.0
 127 
 128   def extract_features(self, preprocessed_inputs):
 129     """Extract features from preprocessed inputs.
 130 
 131     Args:
 132       preprocessed_inputs: a [batch, height, width, channels] float tensor
 133         representing a batch of images.
 134 
 135     Returns:
 136       feature_maps: a list of tensors where the ith tensor has shape
 137         [batch, height_i, width_i, depth_i]
 138     """
 139     preprocessed_inputs = shape_utils.check_min_image_dim(
 140         33, preprocessed_inputs)
 141 
 142     with tf.variable_scope('MobilenetV1',
 143                            reuse=self._reuse_weights) as scope:
 144       with slim.arg_scope(
 145           mobilenet_v1.mobilenet_v1_arg_scope(
 146               is_training=None, regularize_depthwise=True)):
 147         with (slim.arg_scope(self._conv_hyperparams_fn())
 148               if self._override_base_feature_extractor_hyperparams
 149               else context_manager.IdentityContextManager()):
 150           _, image_features = mobilenet_v1.mobilenet_v1_base(
 151               ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple),
 152               final_endpoint='Conv2d_13_pointwise',
 153               min_depth=self._min_depth,
 154               depth_multiplier=self._depth_multiplier,
 155               conv_defs=self._conv_defs,
 156               use_explicit_padding=self._use_explicit_padding,
 157               scope=scope)
 158 
 159       depth_fn = lambda d: max(int(d * self._depth_multiplier), self._min_depth)
 160       with slim.arg_scope(self._conv_hyperparams_fn()):
 161         with tf.variable_scope('fpn', reuse=self._reuse_weights):
 162           feature_blocks = [
 163               'Conv2d_3_pointwise', 'Conv2d_5_pointwise', 'Conv2d_11_pointwise',
 164               'Conv2d_13_pointwise'
 165           ]
 166           base_fpn_max_level = min(self._fpn_max_level, 5)
 167           feature_block_list = []
 168           for level in range(self._fpn_min_level, base_fpn_max_level + 1):
 169             feature_block_list.append(feature_blocks[level - 2])
 170           fpn_features = feature_map_generators.fpn_top_down_feature_maps(
 171               [(key, image_features[key]) for key in feature_block_list],
 172               depth=depth_fn(self._additional_layer_depth),
 173               use_depthwise=self._use_depthwise,
 174               use_explicit_padding=self._use_explicit_padding,
 175               use_native_resize_op=self._use_native_resize_op)
 176           feature_maps = []
 177           for level in range(self._fpn_min_level, base_fpn_max_level + 1):
 178             feature_maps.append(fpn_features['top_down_{}'.format(
 179                 feature_blocks[level - 2])])
 180           last_feature_map = fpn_features['top_down_{}'.format(
 181               feature_blocks[base_fpn_max_level - 2])]
 182           # Construct coarse features
 183           padding = 'VALID' if self._use_explicit_padding else 'SAME'
 184           kernel_size = 3
 185           for i in range(base_fpn_max_level + 1, self._fpn_max_level + 1):
 186             if self._use_depthwise:
 187               conv_op = functools.partial(
 188                   slim.separable_conv2d, depth_multiplier=1)
 189             else:
 190               conv_op = slim.conv2d
 191             if self._use_explicit_padding:
 192               last_feature_map = ops.fixed_padding(
 193                   last_feature_map, kernel_size)
 194             last_feature_map = conv_op(
 195                 last_feature_map,
 196                 num_outputs=depth_fn(self._additional_layer_depth),
 197                 kernel_size=[kernel_size, kernel_size],
 198                 stride=2,
 199                 padding=padding,
 200                 scope='bottom_up_Conv2d_{}'.format(i - base_fpn_max_level + 13))
 201             feature_maps.append(last_feature_map)
 202     return feature_maps
