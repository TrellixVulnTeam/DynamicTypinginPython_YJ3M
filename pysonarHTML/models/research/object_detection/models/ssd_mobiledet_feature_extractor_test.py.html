  27   def _create_feature_extractor(self,
  28                                 feature_extractor_cls,
  29                                 is_training=False,
  30                                 depth_multiplier=1.0,
  31                                 pad_to_multiple=1,
  32                                 use_explicit_padding=False,
  33                                 use_keras=False):
  34     """Constructs a new MobileDet feature extractor.
  35 
  36     Args:
  37       feature_extractor_cls: feature extractor class.
  38       is_training: whether the network is in training mode.
  39       depth_multiplier: float depth multiplier for feature extractor
  40       pad_to_multiple: the nearest multiple to zero pad the input height and
  41         width dimensions to.
  42       use_explicit_padding: If True, we will use 'VALID' padding for
  43         convolutions, but prepad inputs so that the output dimensions are the
  44         same as if 'SAME' padding were used.
  45       use_keras: if True builds a keras-based feature extractor, if False builds
  46         a slim-based one.
  47 
  48     Returns:
  49       an ssd_meta_arch.SSDMobileDetFeatureExtractor object.
  50     """
  51     min_depth = 32
  52     return feature_extractor_cls(
  53         is_training,
  54         depth_multiplier,
  55         min_depth,
  56         pad_to_multiple,
  57         self.conv_hyperparams_fn,
  58         use_explicit_padding=use_explicit_padding)
  59 
  60   def test_mobiledet_cpu_returns_correct_shapes(self):
  61     expected_feature_map_shapes = [(2, 40, 20, 72),
  62                                    (2, 20, 10, 144),
  63                                    (2, 10, 5, 512),
  64                                    (2, 5, 3, 256),
  65                                    (2, 3, 2, 256),
  66                                    (2, 2, 1, 128)]
  67     feature_extractor = self._create_feature_extractor(
  68         ssd_mobiledet_feature_extractor.SSDMobileDetCPUFeatureExtractor)
  69     image = tf.random.normal((2, 640, 320, 3))
  70     feature_maps = feature_extractor.extract_features(image)
  71 
  72     self.assertEqual(len(expected_feature_map_shapes), len(feature_maps))
  73     for expected_shape, x in zip(expected_feature_map_shapes, feature_maps):
  74       self.assertTrue(x.shape.is_compatible_with(expected_shape))

 108   def _check_quantization(self, model_fn):
 109     checkpoint_dir = self.get_temp_dir()
 110 
 111     with tf.Graph().as_default() as training_graph:
 112       model_fn(is_training=True)
 113       contrib_quantize.experimental_create_training_graph(training_graph)
 114       with self.session(graph=training_graph) as sess:
 115         sess.run(tf.global_variables_initializer())
 116         tf.train.Saver().save(sess, checkpoint_dir)
 117 
 118     with tf.Graph().as_default() as eval_graph:
 119       model_fn(is_training=False)
 120       contrib_quantize.experimental_create_eval_graph(eval_graph)
 121       with self.session(graph=eval_graph) as sess:
 122         tf.train.Saver().restore(sess, checkpoint_dir)
