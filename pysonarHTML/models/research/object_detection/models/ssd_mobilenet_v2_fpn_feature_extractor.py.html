  38 # A modified config of mobilenet v2 that makes it more detection friendly.
  39 def _create_modified_mobilenet_config():
  40   conv_defs = copy.deepcopy(mobilenet_v2.V2_DEF)
  41   conv_defs['spec'][-1] = mobilenet.op(
  42       slim.conv2d, stride=1, kernel_size=[1, 1], num_outputs=256)
  43   return conv_defs

  46 class SSDMobileNetV2FpnFeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
  47   """SSD Feature Extractor using MobilenetV2 FPN features."""
  48 
  49   def __init__(self,
  50                is_training,
  51                depth_multiplier,
  52                min_depth,
  53                pad_to_multiple,
  54                conv_hyperparams_fn,
  55                fpn_min_level=3,
  56                fpn_max_level=7,
  57                additional_layer_depth=256,
  58                reuse_weights=None,
  59                use_explicit_padding=False,
  60                use_depthwise=False,
  61                use_native_resize_op=False,
  62                override_base_feature_extractor_hyperparams=False):
  63     """SSD FPN feature extractor based on Mobilenet v2 architecture.
  64 
  65     Args:
  66       is_training: whether the network is in training mode.
  67       depth_multiplier: float depth multiplier for feature extractor.
  68       min_depth: minimum feature extractor depth.
  69       pad_to_multiple: the nearest multiple to zero pad the input height and
  70         width dimensions to.
  71       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  72         and separable_conv2d ops in the layers that are added on top of the base
  73         feature extractor.
  74       fpn_min_level: the highest resolution feature map to use in FPN. The valid
  75         values are {2, 3, 4, 5} which map to MobileNet v2 layers
  76         {layer_4, layer_7, layer_14, layer_19}, respectively.
  77       fpn_max_level: the smallest resolution feature map to construct or use in
  78         FPN. FPN constructions uses features maps starting from fpn_min_level
  79         upto the fpn_max_level. In the case that there are not enough feature
  80         maps in the backbone network, additional feature maps are created by
  81         applying stride 2 convolutions until we get the desired number of fpn
  82         levels.
  83       additional_layer_depth: additional feature map layer channel depth.
  84       reuse_weights: whether to reuse variables. Default is None.
  85       use_explicit_padding: Whether to use explicit padding when extracting
  86         features. Default is False.
  87       use_depthwise: Whether to use depthwise convolutions. Default is False.
  88       use_native_resize_op: Whether to use tf.image.nearest_neighbor_resize
  89         to do upsampling in FPN. Default is false.
  90       override_base_feature_extractor_hyperparams: Whether to override
  91         hyperparameters of the base feature extractor with the one from
  92         `conv_hyperparams_fn`.
  93     """
  94     super(SSDMobileNetV2FpnFeatureExtractor, self).__init__(
  95         is_training=is_training,
  96         depth_multiplier=depth_multiplier,
  97         min_depth=min_depth,
  98         pad_to_multiple=pad_to_multiple,
  99         conv_hyperparams_fn=conv_hyperparams_fn,
 100         reuse_weights=reuse_weights,
 101         use_explicit_padding=use_explicit_padding,
 102         use_depthwise=use_depthwise,
 103         override_base_feature_extractor_hyperparams=
 104         override_base_feature_extractor_hyperparams)
 105     self._fpn_min_level = fpn_min_level
 106     self._fpn_max_level = fpn_max_level
 107     self._additional_layer_depth = additional_layer_depth
 108     self._conv_defs = None
 109     if self._use_depthwise:
 110       self._conv_defs = _create_modified_mobilenet_config()
 111     self._use_native_resize_op = use_native_resize_op
 112 
 113   def preprocess(self, resized_inputs):
 114     """SSD preprocessing.
 115 
 116     Maps pixel values to the range [-1, 1].
 117 
 118     Args:
 119       resized_inputs: a [batch, height, width, channels] float tensor
 120         representing a batch of images.
 121 
 122     Returns:
 123       preprocessed_inputs: a [batch, height, width, channels] float tensor
 124         representing a batch of images.
 125     """
 126     return (2.0 / 255.0) * resized_inputs - 1.0
 127 
 128   def extract_features(self, preprocessed_inputs):
 129     """Extract features from preprocessed inputs.
 130 
 131     Args:
 132       preprocessed_inputs: a [batch, height, width, channels] float tensor
 133         representing a batch of images.
 134 
 135     Returns:
 136       feature_maps: a list of tensors where the ith tensor has shape
 137         [batch, height_i, width_i, depth_i]
 138     """
 139     preprocessed_inputs = shape_utils.check_min_image_dim(
 140         33, preprocessed_inputs)
 141 
 142     with tf.variable_scope('MobilenetV2', reuse=self._reuse_weights) as scope:
 143       with slim.arg_scope(
 144           mobilenet_v2.training_scope(is_training=None, bn_decay=0.9997)), \
 145           slim.arg_scope(
 146               [mobilenet.depth_multiplier], min_depth=self._min_depth):
 147         with (slim.arg_scope(self._conv_hyperparams_fn())
 148               if self._override_base_feature_extractor_hyperparams else
 149               context_manager.IdentityContextManager()):
 150           _, image_features = mobilenet_v2.mobilenet_base(
 151               ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple),
 152               final_endpoint='layer_19',
 153               depth_multiplier=self._depth_multiplier,
 154               conv_defs=self._conv_defs,
 155               use_explicit_padding=self._use_explicit_padding,
 156               scope=scope)
 157       depth_fn = lambda d: max(int(d * self._depth_multiplier), self._min_depth)
 158       with slim.arg_scope(self._conv_hyperparams_fn()):
 159         with tf.variable_scope('fpn', reuse=self._reuse_weights):
 160           feature_blocks = [
 161               'layer_4', 'layer_7', 'layer_14', 'layer_19'
 162           ]
 163           base_fpn_max_level = min(self._fpn_max_level, 5)
 164           feature_block_list = []
 165           for level in range(self._fpn_min_level, base_fpn_max_level + 1):
 166             feature_block_list.append(feature_blocks[level - 2])
 167           fpn_features = feature_map_generators.fpn_top_down_feature_maps(
 168               [(key, image_features[key]) for key in feature_block_list],
 169               depth=depth_fn(self._additional_layer_depth),
 170               use_depthwise=self._use_depthwise,
 171               use_explicit_padding=self._use_explicit_padding,
 172               use_native_resize_op=self._use_native_resize_op)
 173           feature_maps = []
 174           for level in range(self._fpn_min_level, base_fpn_max_level + 1):
 175             feature_maps.append(fpn_features['top_down_{}'.format(
 176                 feature_blocks[level - 2])])
 177           last_feature_map = fpn_features['top_down_{}'.format(
 178               feature_blocks[base_fpn_max_level - 2])]
 179           # Construct coarse features
 180           padding = 'VALID' if self._use_explicit_padding else 'SAME'
 181           kernel_size = 3
 182           for i in range(base_fpn_max_level + 1, self._fpn_max_level + 1):
 183             if self._use_depthwise:
 184               conv_op = functools.partial(
 185                   slim.separable_conv2d, depth_multiplier=1)
 186             else:
 187               conv_op = slim.conv2d
 188             if self._use_explicit_padding:
 189               last_feature_map = ops.fixed_padding(
 190                   last_feature_map, kernel_size)
 191             last_feature_map = conv_op(
 192                 last_feature_map,
 193                 num_outputs=depth_fn(self._additional_layer_depth),
 194                 kernel_size=[kernel_size, kernel_size],
 195                 stride=2,
 196                 padding=padding,
 197                 scope='bottom_up_Conv2d_{}'.format(i - base_fpn_max_level + 19))
 198             feature_maps.append(last_feature_map)
 199     return feature_maps
