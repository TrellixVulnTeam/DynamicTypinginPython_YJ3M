  29 class EmbeddedSSDMobileNetV1FeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
  30   """Embedded-friendly SSD Feature Extractor using MobilenetV1 features.
  31 
  32   This feature extractor is similar to SSD MobileNetV1 feature extractor, and
  33   it fixes input resolution to be 256x256, reduces the number of feature maps
  34   used for box prediction and ensures convolution kernel to be no larger
  35   than input tensor in spatial dimensions.
  36 
  37   This feature extractor requires support of the following ops if used in
  38   embedded devices:
  39   - Conv
  40   - DepthwiseConv
  41   - Relu6
  42 
  43   All conv/depthwiseconv use SAME padding, and no additional spatial padding is
  44   needed.
  45   """
  46 
  47   def __init__(self,
  48                is_training,
  49                depth_multiplier,
  50                min_depth,
  51                pad_to_multiple,
  52                conv_hyperparams_fn,
  53                reuse_weights=None,
  54                use_explicit_padding=False,
  55                use_depthwise=False,
  56                override_base_feature_extractor_hyperparams=False):
  57     """MobileNetV1 Feature Extractor for Embedded-friendly SSD Models.
  58 
  59     Args:
  60       is_training: whether the network is in training mode.
  61       depth_multiplier: float depth multiplier for feature extractor.
  62       min_depth: minimum feature extractor depth.
  63       pad_to_multiple: the nearest multiple to zero pad the input height and
  64         width dimensions to. For EmbeddedSSD it must be set to 1.
  65       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  66         and separable_conv2d ops in the layers that are added on top of the
  67         base feature extractor.
  68       reuse_weights: Whether to reuse variables. Default is None.
  69       use_explicit_padding: Whether to use explicit padding when extracting
  70         features. Default is False.
  71       use_depthwise: Whether to use depthwise convolutions. Default is False.
  72       override_base_feature_extractor_hyperparams: Whether to override
  73         hyperparameters of the base feature extractor with the one from
  74         `conv_hyperparams_fn`.
  75 
  76     Raises:
  77       ValueError: upon invalid `pad_to_multiple` values.
  78     """
  79     if pad_to_multiple != 1:
  80       raise ValueError('Embedded-specific SSD only supports `pad_to_multiple` '
  81                        'of 1.')
  82 
  83     super(EmbeddedSSDMobileNetV1FeatureExtractor, self).__init__(
  84         is_training, depth_multiplier, min_depth, pad_to_multiple,
  85         conv_hyperparams_fn, reuse_weights, use_explicit_padding, use_depthwise,
  86         override_base_feature_extractor_hyperparams)
  87 
  88   def preprocess(self, resized_inputs):
  89     """SSD preprocessing.
  90 
  91     Maps pixel values to the range [-1, 1].
  92 
  93     Args:
  94       resized_inputs: a [batch, height, width, channels] float tensor
  95         representing a batch of images.
  96 
  97     Returns:
  98       preprocessed_inputs: a [batch, height, width, channels] float tensor
  99         representing a batch of images.
 100     """
 101     return (2.0 / 255.0) * resized_inputs - 1.0
 102 
 103   def extract_features(self, preprocessed_inputs):
 104     """Extract features from preprocessed inputs.
 105 
 106     Args:
 107       preprocessed_inputs: a [batch, height, width, channels] float tensor
 108         representing a batch of images.
 109 
 110     Returns:
 111       feature_maps: a list of tensors where the ith tensor has shape
 112         [batch, height_i, width_i, depth_i]
 113 
 114     Raises:
 115       ValueError: if image height or width are not 256 pixels.
 116     """
 117     image_shape = preprocessed_inputs.get_shape()
 118     image_shape.assert_has_rank(4)
 119     image_height = image_shape[1].value
 120     image_width = image_shape[2].value
 121 
 122     if image_height is None or image_width is None:
 123       shape_assert = tf.Assert(
 124           tf.logical_and(tf.equal(tf.shape(preprocessed_inputs)[1], 256),
 125                          tf.equal(tf.shape(preprocessed_inputs)[2], 256)),
 126           ['image size must be 256 in both height and width.'])
 127       with tf.control_dependencies([shape_assert]):
 128         preprocessed_inputs = tf.identity(preprocessed_inputs)
 129     elif image_height != 256 or image_width != 256:
 130       raise ValueError('image size must be = 256 in both height and width;'
 131                        ' image dim = %d,%d' % (image_height, image_width))
 132 
 133     feature_map_layout = {
 134         'from_layer': [
 135             'Conv2d_11_pointwise', 'Conv2d_13_pointwise', '', '', ''
 136         ],
 137         'layer_depth': [-1, -1, 512, 256, 256],
 138         'conv_kernel_size': [-1, -1, 3, 3, 2],
 139         'use_explicit_padding': self._use_explicit_padding,
 140         'use_depthwise': self._use_depthwise,
 141     }
 142 
 143     with tf.variable_scope('MobilenetV1',
 144                            reuse=self._reuse_weights) as scope:
 145       with slim.arg_scope(
 146           mobilenet_v1.mobilenet_v1_arg_scope(is_training=None)):
 147         with (slim.arg_scope(self._conv_hyperparams_fn())
 148               if self._override_base_feature_extractor_hyperparams
 149               else context_manager.IdentityContextManager()):
 150           _, image_features = mobilenet_v1.mobilenet_v1_base(
 151               ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple),
 152               final_endpoint='Conv2d_13_pointwise',
 153               min_depth=self._min_depth,
 154               depth_multiplier=self._depth_multiplier,
 155               use_explicit_padding=self._use_explicit_padding,
 156               scope=scope)
 157       with slim.arg_scope(self._conv_hyperparams_fn()):
 158         feature_maps = feature_map_generators.multi_resolution_feature_maps(
 159             feature_map_layout=feature_map_layout,
 160             depth_multiplier=self._depth_multiplier,
 161             min_depth=self._min_depth,
 162             insert_1x1_conv=True,
 163             image_features=image_features)
 164 
 165     return list(feature_maps.values())
