  38 def pnasnet_large_arg_scope_for_detection(is_batch_norm_training=False):
  39   """Defines the default arg scope for the PNASNet Large for object detection.
  40 
  41   This provides a small edit to switch batch norm training on and off.
  42 
  43   Args:
  44     is_batch_norm_training: Boolean indicating whether to train with batch norm.
  45 
  46   Returns:
  47     An `arg_scope` to use for the PNASNet Large Model.
  48   """
  49   imagenet_scope = pnasnet.pnasnet_large_arg_scope()
  50   with arg_scope(imagenet_scope):
  51     with arg_scope([slim.batch_norm], is_training=is_batch_norm_training) as sc:
  52       return sc

  55 def _filter_scaling(reduction_indices, start_cell_num):
  56   """Compute the expected filter scaling at given PNASNet cell start_cell_num.
  57 
  58   In the pnasnet.py code, filter_scaling starts at 1.0. We instead
  59   adapt filter scaling to depend on the starting cell.
  60   At first cells, before any reduction, filter_scalling is 1.0. With passing
  61   any reduction cell, the filter_scaling is multiplied by 2.
  62 
  63   Args:
  64     reduction_indices: list of int indices.
  65     start_cell_num: int.
  66   Returns:
  67     filter_scaling: float.
  68   """
  69   filter_scaling = 1.0
  70   for ind in reduction_indices:
  71     if ind < start_cell_num:
  72       filter_scaling *= 2.0
  73   return filter_scaling

  76 # Note: This is largely a copy of _build_pnasnet_base inside pnasnet.py but
  77 # with special edits to remove instantiation of the stem and the special
  78 # ability to receive as input a pair of hidden states. It constructs only
  79 # a sub-network from the original PNASNet model, starting from the
  80 # start_cell_num cell and with modified final layer.
  81 def _build_pnasnet_base(
  82     hidden_previous, hidden, normal_cell, hparams, true_cell_num,
  83     start_cell_num):
  84   """Constructs a PNASNet image model for proposal classifier features."""
  85 
  86   # Find where to place the reduction cells or stride normal cells
  87   reduction_indices = nasnet_utils.calc_reduction_layers(
  88       hparams.num_cells, hparams.num_reduction_layers)
  89   filter_scaling = _filter_scaling(reduction_indices, start_cell_num)
  90 
  91   # Note: The None is prepended to match the behavior of _imagenet_stem()
  92   cell_outputs = [None, hidden_previous, hidden]
  93   net = hidden
  94 
  95   # Run the cells
  96   for cell_num in range(start_cell_num, hparams.num_cells):
  97     is_reduction = cell_num in reduction_indices
  98     stride = 2 if is_reduction else 1
  99     if is_reduction: filter_scaling *= hparams.filter_scaling_rate
 100     prev_layer = cell_outputs[-2]
 101     net = normal_cell(
 102         net,
 103         scope='cell_{}'.format(cell_num),
 104         filter_scaling=filter_scaling,
 105         stride=stride,
 106         prev_layer=prev_layer,
 107         cell_num=true_cell_num)
 108     true_cell_num += 1
 109     cell_outputs.append(net)
 110 
 111   # Final nonlinearity.
 112   # Note that we have dropped the final pooling, dropout and softmax layers
 113   # from the default pnasnet version.
 114   with tf.variable_scope('final_layer'):
 115     net = tf.nn.relu(net)
 116   return net

 151   def preprocess(self, resized_inputs):
 152     """Faster R-CNN with PNAS preprocessing.
 153 
 154     Maps pixel values to the range [-1, 1].
 155 
 156     Args:
 157       resized_inputs: A [batch, height_in, width_in, channels] float32 tensor
 158         representing a batch of images with values between 0 and 255.0.
 159 
 160     Returns:
 161       preprocessed_inputs: A [batch, height_out, width_out, channels] float32
 162         tensor representing a batch of images.
 163 
 164     """
 165     return (2.0 / 255.0) * resized_inputs - 1.0

 167   def _extract_proposal_features(self, preprocessed_inputs, scope):
 168     """Extracts first stage RPN features.
 169 
 170     Extracts features using the first half of the PNASNet network.
 171     We construct the network in `align_feature_maps=True` mode, which means
 172     that all VALID paddings in the network are changed to SAME padding so that
 173     the feature maps are aligned.
 174 
 175     Args:
 176       preprocessed_inputs: A [batch, height, width, channels] float32 tensor
 177         representing a batch of images.
 178       scope: A scope name.
 179 
 180     Returns:
 181       rpn_feature_map: A tensor with shape [batch, height, width, depth]
 182       end_points: A dictionary mapping feature extractor tensor names to tensors
 183 
 184     Raises:
 185       ValueError: If the created network is missing the required activation.
 186     """
 187     del scope
 188 
 189     if len(preprocessed_inputs.get_shape().as_list()) != 4:
 190       raise ValueError('`preprocessed_inputs` must be 4 dimensional, got a '
 191                        'tensor of shape %s' % preprocessed_inputs.get_shape())
 192 
 193     with slim.arg_scope(pnasnet_large_arg_scope_for_detection(
 194         is_batch_norm_training=self._train_batch_norm)):
 195       with arg_scope([slim.conv2d,
 196                       slim.batch_norm,
 197                       slim.separable_conv2d],
 198                      reuse=self._reuse_weights):
 199         _, end_points = pnasnet.build_pnasnet_large(
 200             preprocessed_inputs, num_classes=None,
 201             is_training=self._is_training,
 202             final_endpoint='Cell_7')
 203 
 204     # Note that both 'Cell_6' and 'Cell_7' have equal depth = 2160.
 205     # Cell_7 is the last cell before second reduction.
 206     rpn_feature_map = tf.concat([end_points['Cell_6'],
 207                                  end_points['Cell_7']], 3)
 208 
 209     # pnasnet.py does not maintain the batch size in the first dimension.
 210     # This work around permits us retaining the batch for below.
 211     batch = preprocessed_inputs.get_shape().as_list()[0]
 212     shape_without_batch = rpn_feature_map.get_shape().as_list()[1:]
 213     rpn_feature_map_shape = [batch] + shape_without_batch
 214     rpn_feature_map.set_shape(rpn_feature_map_shape)
 215 
 216     return rpn_feature_map, end_points

 218   def _extract_box_classifier_features(self, proposal_feature_maps, scope):
 219     """Extracts second stage box classifier features.
 220 
 221     This function reconstructs the "second half" of the PNASNet
 222     network after the part defined in `_extract_proposal_features`.
 223 
 224     Args:
 225       proposal_feature_maps: A 4-D float tensor with shape
 226         [batch_size * self.max_num_proposals, crop_height, crop_width, depth]
 227         representing the feature map cropped to each proposal.
 228       scope: A scope name.
 229 
 230     Returns:
 231       proposal_classifier_features: A 4-D float tensor with shape
 232         [batch_size * self.max_num_proposals, height, width, depth]
 233         representing box classifier features for each proposal.
 234     """
 235     del scope
 236 
 237     # Number of used stem cells.
 238     num_stem_cells = 2
 239 
 240     # Note that we always feed into 2 layers of equal depth
 241     # where the first N channels corresponds to previous hidden layer
 242     # and the second N channels correspond to the final hidden layer.
 243     hidden_previous, hidden = tf.split(proposal_feature_maps, 2, axis=3)
 244 
 245     # Note that what follows is largely a copy of build_pnasnet_large() within
 246     # pnasnet.py. We are copying to minimize code pollution in slim.
 247 
 248     # TODO(shlens,skornblith): Determine the appropriate drop path schedule.
 249     # For now the schedule is the default (1.0->0.7 over 250,000 train steps).
 250     hparams = pnasnet.large_imagenet_config()
 251     if not self._is_training:
 252       hparams.set_hparam('drop_path_keep_prob', 1.0)
 253 
 254     # Calculate the total number of cells in the network
 255     total_num_cells = hparams.num_cells + num_stem_cells
 256 
 257     normal_cell = pnasnet.PNasNetNormalCell(
 258         hparams.num_conv_filters, hparams.drop_path_keep_prob,
 259         total_num_cells, hparams.total_training_steps)
 260     with arg_scope([slim.dropout, nasnet_utils.drop_path],
 261                    is_training=self._is_training):
 262       with arg_scope([slim.batch_norm], is_training=self._train_batch_norm):
 263         with arg_scope([slim.avg_pool2d,
 264                         slim.max_pool2d,
 265                         slim.conv2d,
 266                         slim.batch_norm,
 267                         slim.separable_conv2d,
 268                         nasnet_utils.factorized_reduction,
 269                         nasnet_utils.global_avg_pool,
 270                         nasnet_utils.get_channel_index,
 271                         nasnet_utils.get_channel_dim],
 272                        data_format=hparams.data_format):
 273 
 274           # This corresponds to the cell number just past 'Cell_7' used by
 275           # _extract_proposal_features().
 276           start_cell_num = 8
 277           true_cell_num = start_cell_num + num_stem_cells
 278 
 279           with slim.arg_scope(pnasnet.pnasnet_large_arg_scope()):
 280             net = _build_pnasnet_base(
 281                 hidden_previous,
 282                 hidden,
 283                 normal_cell=normal_cell,
 284                 hparams=hparams,
 285                 true_cell_num=true_cell_num,
 286                 start_cell_num=start_cell_num)
 287 
 288     proposal_classifier_features = net
 289     return proposal_classifier_features

 291   def restore_from_classification_checkpoint_fn(
 292       self,
 293       first_stage_feature_extractor_scope,
 294       second_stage_feature_extractor_scope):
 295     """Returns a map of variables to load from a foreign checkpoint.
 296 
 297     Note that this overrides the default implementation in
 298     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor which does not work for
 299     PNASNet checkpoints.
 300 
 301     Args:
 302       first_stage_feature_extractor_scope: A scope name for the first stage
 303         feature extractor.
 304       second_stage_feature_extractor_scope: A scope name for the second stage
 305         feature extractor.
 306 
 307     Returns:
 308       A dict mapping variable names (to load from a checkpoint) to variables in
 309       the model graph.
 310     """
 311     variables_to_restore = {}
 312     for variable in variables_helper.get_global_variables_safely():
 313       if variable.op.name.startswith(
 314           first_stage_feature_extractor_scope):
 315         var_name = variable.op.name.replace(
 316             first_stage_feature_extractor_scope + '/', '')
 317         var_name += '/ExponentialMovingAverage'
 318         variables_to_restore[var_name] = variable
 319       if variable.op.name.startswith(
 320           second_stage_feature_extractor_scope):
 321         var_name = variable.op.name.replace(
 322             second_stage_feature_extractor_scope + '/', '')
 323         var_name += '/ExponentialMovingAverage'
 324         variables_to_restore[var_name] = variable
 325     return variables_to_restore
