  40   def __init__(self,
  41                is_training,
  42                first_stage_features_stride,
  43                batch_norm_trainable=False,
  44                weight_decay=0.0):
  45     """Constructor.
  46 
  47     Args:
  48       is_training: See base class.
  49       first_stage_features_stride: See base class.
  50       batch_norm_trainable: See base class.
  51       weight_decay: See base class.
  52 
  53     Raises:
  54       ValueError: If `first_stage_features_stride` is not 8 or 16.
  55     """
  56     if first_stage_features_stride != 8 and first_stage_features_stride != 16:
  57       raise ValueError('`first_stage_features_stride` must be 8 or 16.')
  58     super(FasterRCNNInceptionResnetV2KerasFeatureExtractor, self).__init__(
  59         is_training, first_stage_features_stride, batch_norm_trainable,
  60         weight_decay)
  61     self._variable_dict = {}

  63   def preprocess(self, resized_inputs):
  64     """Faster R-CNN with Inception Resnet v2 preprocessing.
  65 
  66     Maps pixel values to the range [-1, 1].
  67 
  68     Args:
  69       resized_inputs: A [batch, height_in, width_in, channels] float32 tensor
  70         representing a batch of images with values between 0 and 255.0.
  71 
  72     Returns:
  73       preprocessed_inputs: A [batch, height_out, width_out, channels] float32
  74         tensor representing a batch of images.
  75 
  76     """
  77     return (2.0 / 255.0) * resized_inputs - 1.0

  79   def get_proposal_feature_extractor_model(self, name=None):
  80     """Returns a model that extracts first stage RPN features.
  81 
  82     Extracts features using the first half of the Inception Resnet v2 network.
  83     We construct the network in `align_feature_maps=True` mode, which means
  84     that all VALID paddings in the network are changed to SAME padding so that
  85     the feature maps are aligned.
  86 
  87     Args:
  88       name: A scope name to construct all variables within.
  89 
  90     Returns:
  91       A Keras model that takes preprocessed_inputs:
  92         A [batch, height, width, channels] float32 tensor
  93         representing a batch of images.
  94 
  95       And returns rpn_feature_map:
  96         A tensor with shape [batch, height, width, depth]
  97     """
  98     with tf.name_scope(name):
  99       with tf.name_scope('InceptionResnetV2'):
 100         model = inception_resnet_v2.inception_resnet_v2(
 101               self._train_batch_norm,
 102               output_stride=self._first_stage_features_stride,
 103               align_feature_maps=True,
 104               weight_decay=self._weight_decay,
 105               weights=None,
 106               include_top=False)
 107         proposal_features = model.get_layer(
 108             name='block17_20_ac').output
 109         keras_model = tf.keras.Model(
 110             inputs=model.inputs,
 111             outputs=proposal_features)
 112         for variable in keras_model.variables:
 113           self._variable_dict[variable.name[:-2]] = variable
 114         return keras_model

 116   def get_box_classifier_feature_extractor_model(self, name=None):
 117     """Returns a model that extracts second stage box classifier features.
 118 
 119     This function reconstructs the "second half" of the Inception ResNet v2
 120     network after the part defined in `get_proposal_feature_extractor_model`.
 121 
 122     Args:
 123       name: A scope name to construct all variables within.
 124 
 125     Returns:
 126       A Keras model that takes proposal_feature_maps:
 127         A 4-D float tensor with shape
 128         [batch_size * self.max_num_proposals, crop_height, crop_width, depth]
 129         representing the feature map cropped to each proposal.
 130       And returns proposal_classifier_features:
 131         A 4-D float tensor with shape
 132         [batch_size * self.max_num_proposals, height, width, depth]
 133         representing box classifier features for each proposal.
 134     """
 135     with tf.name_scope(name):
 136       with tf.name_scope('InceptionResnetV2'):
 137         model = inception_resnet_v2.inception_resnet_v2(
 138             self._train_batch_norm,
 139             output_stride=16,
 140             align_feature_maps=False,
 141             weight_decay=self._weight_decay,
 142             weights=None,
 143             include_top=False)
 144 
 145         proposal_feature_maps = model.get_layer(
 146             name='block17_20_ac').output
 147         proposal_classifier_features = model.get_layer(
 148             name='conv_7b_ac').output
 149 
 150         keras_model = model_util.extract_submodel(
 151             model=model,
 152             inputs=proposal_feature_maps,
 153             outputs=proposal_classifier_features)
 154         for variable in keras_model.variables:
 155           self._variable_dict[variable.name[:-2]] = variable
 156         return keras_model
