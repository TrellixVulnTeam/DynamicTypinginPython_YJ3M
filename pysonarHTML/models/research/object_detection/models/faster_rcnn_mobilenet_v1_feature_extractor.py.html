  27 def _get_mobilenet_conv_no_last_stride_defs(conv_depth_ratio_in_percentage):
  28   if conv_depth_ratio_in_percentage not in [25, 50, 75, 100]:
  29     raise ValueError(
  30         'Only the following ratio percentages are supported: 25, 50, 75, 100')
  31   conv_depth_ratio_in_percentage = float(conv_depth_ratio_in_percentage) / 100.0
  32   channels = np.array([
  33       32, 64, 128, 128, 256, 256, 512, 512, 512, 512, 512, 512, 1024, 1024
  34   ], dtype=np.float32)
  35   channels = (channels * conv_depth_ratio_in_percentage).astype(np.int32)
  36   return [
  37       mobilenet_v1.Conv(kernel=[3, 3], stride=2, depth=channels[0]),
  38       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[1]),
  39       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=2, depth=channels[2]),
  40       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[3]),
  41       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=2, depth=channels[4]),
  42       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[5]),
  43       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=2, depth=channels[6]),
  44       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[7]),
  45       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[8]),
  46       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[9]),
  47       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[10]),
  48       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[11]),
  49       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[12]),
  50       mobilenet_v1.DepthSepConv(kernel=[3, 3], stride=1, depth=channels[13])
  51   ]

  54 class FasterRCNNMobilenetV1FeatureExtractor(
  55     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor):
  56   """Faster R-CNN Mobilenet V1 feature extractor implementation."""
  57 
  58   def __init__(self,
  59                is_training,
  60                first_stage_features_stride,
  61                batch_norm_trainable=False,
  62                reuse_weights=None,
  63                weight_decay=0.0,
  64                depth_multiplier=1.0,
  65                min_depth=16,
  66                skip_last_stride=False,
  67                conv_depth_ratio_in_percentage=100):
  68     """Constructor.
  69 
  70     Args:
  71       is_training: See base class.
  72       first_stage_features_stride: See base class.
  73       batch_norm_trainable: See base class.
  74       reuse_weights: See base class.
  75       weight_decay: See base class.
  76       depth_multiplier: float depth multiplier for feature extractor.
  77       min_depth: minimum feature extractor depth.
  78       skip_last_stride: Skip the last stride if True.
  79       conv_depth_ratio_in_percentage: Conv depth ratio in percentage. Only
  80         applied if skip_last_stride is True.
  81 
  82     Raises:
  83       ValueError: If `first_stage_features_stride` is not 8 or 16.
  84     """
  85     if first_stage_features_stride != 8 and first_stage_features_stride != 16:
  86       raise ValueError('`first_stage_features_stride` must be 8 or 16.')
  87     self._depth_multiplier = depth_multiplier
  88     self._min_depth = min_depth
  89     self._skip_last_stride = skip_last_stride
  90     self._conv_depth_ratio_in_percentage = conv_depth_ratio_in_percentage
  91     super(FasterRCNNMobilenetV1FeatureExtractor, self).__init__(
  92         is_training, first_stage_features_stride, batch_norm_trainable,
  93         reuse_weights, weight_decay)
  94 
  95   def preprocess(self, resized_inputs):
  96     """Faster R-CNN Mobilenet V1 preprocessing.
  97 
  98     Maps pixel values to the range [-1, 1].
  99 
 100     Args:
 101       resized_inputs: a [batch, height, width, channels] float tensor
 102         representing a batch of images.
 103 
 104     Returns:
 105       preprocessed_inputs: a [batch, height, width, channels] float tensor
 106         representing a batch of images.
 107     """
 108     return (2.0 / 255.0) * resized_inputs - 1.0
 109 
 110   def _extract_proposal_features(self, preprocessed_inputs, scope):
 111     """Extracts first stage RPN features.
 112 
 113     Args:
 114       preprocessed_inputs: A [batch, height, width, channels] float32 tensor
 115         representing a batch of images.
 116       scope: A scope name.
 117 
 118     Returns:
 119       rpn_feature_map: A tensor with shape [batch, height, width, depth]
 120       activations: A dictionary mapping feature extractor tensor names to
 121         tensors
 122 
 123     Raises:
 124       InvalidArgumentError: If the spatial size of `preprocessed_inputs`
 125         (height or width) is less than 33.
 126       ValueError: If the created network is missing the required activation.
 127     """
 128 
 129     preprocessed_inputs.get_shape().assert_has_rank(4)
 130     preprocessed_inputs = shape_utils.check_min_image_dim(
 131         min_dim=33, image_tensor=preprocessed_inputs)
 132 
 133     with slim.arg_scope(
 134         mobilenet_v1.mobilenet_v1_arg_scope(
 135             is_training=self._train_batch_norm,
 136             weight_decay=self._weight_decay)):
 137       with tf.variable_scope('MobilenetV1',
 138                              reuse=self._reuse_weights) as scope:
 139         params = {}
 140         if self._skip_last_stride:
 141           params['conv_defs'] = _get_mobilenet_conv_no_last_stride_defs(
 142               conv_depth_ratio_in_percentage=self.
 143               _conv_depth_ratio_in_percentage)
 144         _, activations = mobilenet_v1.mobilenet_v1_base(
 145             preprocessed_inputs,
 146             final_endpoint='Conv2d_11_pointwise',
 147             min_depth=self._min_depth,
 148             depth_multiplier=self._depth_multiplier,
 149             scope=scope,
 150             **params)
 151     return activations['Conv2d_11_pointwise'], activations
 152 
 153   def _extract_box_classifier_features(self, proposal_feature_maps, scope):
 154     """Extracts second stage box classifier features.
 155 
 156     Args:
 157       proposal_feature_maps: A 4-D float tensor with shape
 158         [batch_size * self.max_num_proposals, crop_height, crop_width, depth]
 159         representing the feature map cropped to each proposal.
 160       scope: A scope name (unused).
 161 
 162     Returns:
 163       proposal_classifier_features: A 4-D float tensor with shape
 164         [batch_size * self.max_num_proposals, height, width, depth]
 165         representing box classifier features for each proposal.
 166     """
 167     net = proposal_feature_maps
 168 
 169     conv_depth = 1024
 170     if self._skip_last_stride:
 171       conv_depth_ratio = float(self._conv_depth_ratio_in_percentage) / 100.0
 172       conv_depth = int(float(conv_depth) * conv_depth_ratio)
 173 
 174     depth = lambda d: max(int(d * 1.0), 16)
 175     with tf.variable_scope('MobilenetV1', reuse=self._reuse_weights):
 176       with slim.arg_scope(
 177           mobilenet_v1.mobilenet_v1_arg_scope(
 178               is_training=self._train_batch_norm,
 179               weight_decay=self._weight_decay)):
 180         with slim.arg_scope(
 181             [slim.conv2d, slim.separable_conv2d], padding='SAME'):
 182           net = slim.separable_conv2d(
 183               net,
 184               depth(conv_depth), [3, 3],
 185               depth_multiplier=1,
 186               stride=2,
 187               scope='Conv2d_12_pointwise')
 188           return slim.separable_conv2d(
 189               net,
 190               depth(conv_depth), [3, 3],
 191               depth_multiplier=1,
 192               stride=1,
 193               scope='Conv2d_13_pointwise')
