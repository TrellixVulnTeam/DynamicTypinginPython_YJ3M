  35 Block = collections.namedtuple(
  36     'Block', ['inputs', 'output_level', 'kernel_size', 'expansion_size'])

  38 _MNASFPN_CELL_CONFIG = [
  39     Block(inputs=(1, 2), output_level=4, kernel_size=3, expansion_size=256),

  52 def _maybe_pad(feature, use_explicit_padding, kernel_size=3):

  54                            kernel_size) if use_explicit_padding else feature

  57 # Wrapper around mobilenet.depth_multiplier
  58 def _apply_multiplier(d, multiplier, min_depth):
  59   p = {'num_outputs': d}
  60   mobilenet.depth_multiplier(
  61       p, multiplier=multiplier, divisible_by=8, min_depth=min_depth)
  62   return p['num_outputs']

  65 def _apply_size_dependent_ordering(input_feature, feature_level, block_level,
  66                                    expansion_size, use_explicit_padding,
  67                                    use_native_resize_op):
  68   """Applies Size-Dependent-Ordering when resizing feature maps.
  69 
  70      See https://arxiv.org/abs/1912.01106
  71 
  72   Args:
  73     input_feature: input feature map to be resized.
  74     feature_level: the level of the input feature.
  75     block_level: the desired output level for the block.
  76     expansion_size: the expansion size for the block.
  77     use_explicit_padding: Whether to use explicit padding.
  78     use_native_resize_op: Whether to use native resize op.
  79 
  80   Returns:
  81     A transformed feature at the desired resolution and expansion size.
  82   """
  83   padding = 'VALID' if use_explicit_padding else 'SAME'
  84   if feature_level >= block_level:  # Perform 1x1 then upsampling.
  85     node = slim.conv2d(
  86         input_feature,
  87         expansion_size, [1, 1],
  88         activation_fn=None,
  89         normalizer_fn=slim.batch_norm,
  90         padding=padding,
  91         scope='Conv1x1')
  92     if feature_level == block_level:
  93       return node
  94     scale = 2**(feature_level - block_level)
  95     if use_native_resize_op:
  96       input_shape = shape_utils.combined_static_and_dynamic_shape(node)
  97       node = tf.image.resize_nearest_neighbor(
  98           node, [input_shape[1] * scale, input_shape[2] * scale])
  99     else:
 100       node = ops.nearest_neighbor_upsampling(node, scale=scale)
 101   else:  # Perform downsampling then 1x1.
 102     stride = 2**(block_level - feature_level)
 103     node = slim.max_pool2d(
 104         _maybe_pad(input_feature, use_explicit_padding), [3, 3],
 105         stride=[stride, stride],
 106         padding=padding,
 107         scope='Downsample')
 108     node = slim.conv2d(
 109         node,
 110         expansion_size, [1, 1],
 111         activation_fn=None,
 112         normalizer_fn=slim.batch_norm,
 113         padding=padding,
 114         scope='Conv1x1')
 115   return node

 118 def _mnasfpn_cell(feature_maps,
 119                   feature_levels,
 120                   cell_spec,
 121                   output_channel=48,
 122                   use_explicit_padding=False,
 123                   use_native_resize_op=False,
 124                   multiplier_func=None):
 125   """Create a MnasFPN cell.
 126 
 127   Args:
 128     feature_maps: input feature maps.
 129     feature_levels: levels of the feature maps.
 130     cell_spec: A list of Block configs.
 131     output_channel: Number of features for the input, output and intermediate
 132       feature maps.
 133     use_explicit_padding: Whether to use explicit padding.
 134     use_native_resize_op: Whether to use native resize op.
 135     multiplier_func: Depth-multiplier function. If None, use identity function.
 136 
 137   Returns:
 138     A transformed list of feature maps at the same resolutions as the inputs.
 139   """
 140   # This is the level where multipliers are realized.
 141   if multiplier_func is None:
 142     multiplier_func = lambda x: x
 143   num_outputs = len(feature_maps)
 144   cell_features = list(feature_maps)
 145   cell_levels = list(feature_levels)
 146   padding = 'VALID' if use_explicit_padding else 'SAME'
 147   for bi, block in enumerate(cell_spec):
 148     with tf.variable_scope('block_{}'.format(bi)):
 149       block_level = block.output_level
 150       intermediate_feature = None
 151       for i, inp in enumerate(block.inputs):
 152         with tf.variable_scope('input_{}'.format(i)):
 153           input_level = cell_levels[inp]
 154           node = _apply_size_dependent_ordering(
 155               cell_features[inp], input_level, block_level,
 156               multiplier_func(block.expansion_size), use_explicit_padding,
 157               use_native_resize_op)
 158         # Add features incrementally to avoid producing AddN, which doesn't
 159         # play well with TfLite.
 160         if intermediate_feature is None:
 161           intermediate_feature = node
 162         else:
 163           intermediate_feature += node
 164       node = tf.nn.relu6(intermediate_feature)
 165       node = slim.separable_conv2d(
 166           _maybe_pad(node, use_explicit_padding, block.kernel_size),
 167           multiplier_func(output_channel),
 168           block.kernel_size,
 169           activation_fn=None,
 170           normalizer_fn=slim.batch_norm,
 171           padding=padding,
 172           scope='SepConv')
 173     cell_features.append(node)
 174     cell_levels.append(block_level)
 175 
 176   # Cell-wide residuals.
 177   out_idx = range(len(cell_features) - num_outputs, len(cell_features))
 178   for in_i, out_i in enumerate(out_idx):
 179     if cell_features[out_i].shape.as_list(
 180     ) == cell_features[in_i].shape.as_list():
 181       cell_features[out_i] += cell_features[in_i]
 182 
 183   return cell_features[-num_outputs:]

 186 def mnasfpn(feature_maps,
 187             head_def,
 188             output_channel=48,
 189             use_explicit_padding=False,
 190             use_native_resize_op=False,
 191             multiplier_func=None):
 192   """Create the MnasFPN head given head_def."""
 193   features = feature_maps
 194   for ci, cell_spec in enumerate(head_def['spec']):
 195     with tf.variable_scope('cell_{}'.format(ci)):
 196       features = _mnasfpn_cell(features, head_def['feature_levels'], cell_spec,
 197                                output_channel, use_explicit_padding,
 198                                use_native_resize_op, multiplier_func)
 199   return features

 202 def training_scope(l2_weight_decay=1e-4, is_training=None):
 203   """Arg scope for training MnasFPN."""
 204   with slim.arg_scope(
 205       [slim.conv2d],
 206       weights_initializer=tf.initializers.he_normal(),
 207       weights_regularizer=slim.l2_regularizer(l2_weight_decay)), \
 208       slim.arg_scope(
 209           [slim.separable_conv2d],
 210           weights_initializer=tf.initializers.truncated_normal(
 211               stddev=0.536),  # He_normal for 3x3 depthwise kernel.
 212           weights_regularizer=slim.l2_regularizer(l2_weight_decay)), \
 213       slim.arg_scope([slim.batch_norm],
 214                      is_training=is_training,
 215                      epsilon=0.01,
 216                      decay=0.99,
 217                      center=True,
 218                      scale=True) as s:
 219     return s

 222 class SSDMobileNetV2MnasFPNFeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
 223   """SSD Feature Extractor using MobilenetV2 MnasFPN features."""
 224 
 225   def __init__(self,
 226                is_training,
 227                depth_multiplier,
 228                min_depth,
 229                pad_to_multiple,
 230                conv_hyperparams_fn,
 231                fpn_min_level=3,
 232                fpn_max_level=6,
 233                additional_layer_depth=48,
 234                head_def=None,
 235                reuse_weights=None,
 236                use_explicit_padding=False,
 237                use_depthwise=False,
 238                use_native_resize_op=False,
 239                override_base_feature_extractor_hyperparams=False,
 240                data_format='channels_last'):
 241     """SSD MnasFPN feature extractor based on Mobilenet v2 architecture.
 242 
 243     See https://arxiv.org/abs/1912.01106
 244 
 245     Args:
 246       is_training: whether the network is in training mode.
 247       depth_multiplier: float depth multiplier for feature extractor.
 248       min_depth: minimum feature extractor depth.
 249       pad_to_multiple: the nearest multiple to zero pad the input height and
 250         width dimensions to.
 251       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
 252         and separable_conv2d ops in the layers that are added on top of the base
 253         feature extractor.
 254       fpn_min_level: the highest resolution feature map to use in MnasFPN.
 255         Currently the only valid value is 3.
 256       fpn_max_level: the smallest resolution feature map to construct or use in
 257         MnasFPN. Currentl the only valid value is 6.
 258       additional_layer_depth: additional feature map layer channel depth for
 259         NAS-FPN.
 260       head_def: A dictionary specifying the MnasFPN head architecture. Default
 261         uses MNASFPN_DEF.
 262       reuse_weights: whether to reuse variables. Default is None.
 263       use_explicit_padding: Whether to use explicit padding when extracting
 264         features. Default is False.
 265       use_depthwise: Whether to use depthwise convolutions. Default is False.
 266       use_native_resize_op: Whether to use native resize op. Default is False.
 267       override_base_feature_extractor_hyperparams: Whether to override
 268         hyperparameters of the base feature extractor with the one from
 269         `conv_hyperparams_fn`.
 270       data_format: The ordering of the dimensions in the inputs, The valid
 271         values are {'channels_first', 'channels_last').
 272     """
 273     super(SSDMobileNetV2MnasFPNFeatureExtractor, self).__init__(
 274         is_training=is_training,
 275         depth_multiplier=depth_multiplier,
 276         min_depth=min_depth,
 277         pad_to_multiple=pad_to_multiple,
 278         conv_hyperparams_fn=conv_hyperparams_fn,
 279         reuse_weights=reuse_weights,
 280         use_explicit_padding=use_explicit_padding,
 281         use_depthwise=use_depthwise,
 282         override_base_feature_extractor_hyperparams=(
 283             override_base_feature_extractor_hyperparams))
 284     if fpn_min_level != 3 or fpn_max_level != 6:
 285       raise ValueError('Min and max levels of MnasFPN must be 3 and 6 for now.')
 286     self._fpn_min_level = fpn_min_level
 287     self._fpn_max_level = fpn_max_level
 288     self._fpn_layer_depth = additional_layer_depth
 289     self._head_def = head_def if head_def else MNASFPN_DEF
 290     self._data_format = data_format
 291     self._use_native_resize_op = use_native_resize_op
 292 
 293   def preprocess(self, resized_inputs):
 294     """SSD preprocessing.
 295 
 296     Maps pixel values to the range [-1, 1].
 297 
 298     Args:
 299       resized_inputs: a [batch, height, width, channels] float tensor
 300         representing a batch of images.
 301 
 302     Returns:
 303       preprocessed_inputs: a [batch, height, width, channels] float tensor
 304         representing a batch of images.
 305     """
 306     return (2.0 / 255.0) * resized_inputs - 1.0
 307 
 308   def _verify_config(self, inputs):
 309     """Verify that MnasFPN config and its inputs."""
 310     num_inputs = len(inputs)
 311     assert len(self._head_def['feature_levels']) == num_inputs
 312 
 313     base_width = inputs[0].shape.as_list(
 314     )[1] * 2**self._head_def['feature_levels'][0]
 315     for i in range(1, num_inputs):
 316       width = inputs[i].shape.as_list()[1]
 317       level = self._head_def['feature_levels'][i]
 318       expected_width = base_width // 2**level
 319       if width != expected_width:
 320         raise ValueError(
 321             'Resolution of input {} does not match its level {}.'.format(
 322                 i, level))
 323 
 324     for cell_spec in self._head_def['spec']:
 325       # The last K nodes in a cell are the inputs to the next cell. Assert that
 326       # their feature maps are at the right level.
 327       for i in range(num_inputs):
 328         if cell_spec[-num_inputs +
 329                      i].output_level != self._head_def['feature_levels'][i]:
 330           raise ValueError(
 331               'Mismatch between node level {} and desired output level {}.'
 332               .format(cell_spec[-num_inputs + i].output_level,
 333                       self._head_def['feature_levels'][i]))
 334       # Assert that each block only uses precending blocks.
 335       for bi, block_spec in enumerate(cell_spec):
 336         for inp in block_spec.inputs:
 337           if inp >= bi + num_inputs:
 338             raise ValueError(
 339                 'Block {} is trying to access uncreated block {}.'.format(
 340                     bi, inp))
 341 
 342   def extract_features(self, preprocessed_inputs):
 343     """Extract features from preprocessed inputs.
 344 
 345     Args:
 346       preprocessed_inputs: a [batch, height, width, channels] float tensor
 347         representing a batch of images.
 348 
 349     Returns:
 350       feature_maps: a list of tensors where the ith tensor has shape
 351         [batch, height_i, width_i, depth_i]
 352     """
 353     preprocessed_inputs = shape_utils.check_min_image_dim(
 354         33, preprocessed_inputs)
 355     with tf.variable_scope('MobilenetV2', reuse=self._reuse_weights) as scope:
 356       with slim.arg_scope(
 357           mobilenet_v2.training_scope(is_training=None, bn_decay=0.99)), \
 358           slim.arg_scope(
 359               [mobilenet.depth_multiplier], min_depth=self._min_depth):
 360         with slim.arg_scope(
 361             training_scope(l2_weight_decay=4e-5,
 362                            is_training=self._is_training)):
 363 
 364           _, image_features = mobilenet_v2.mobilenet_base(
 365               ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple),
 366               final_endpoint='layer_18',
 367               depth_multiplier=self._depth_multiplier,
 368               use_explicit_padding=self._use_explicit_padding,
 369               scope=scope)
 370 
 371     multiplier_func = functools.partial(
 372         _apply_multiplier,
 373         multiplier=self._depth_multiplier,
 374         min_depth=self._min_depth)
 375     with tf.variable_scope('MnasFPN', reuse=self._reuse_weights):
 376       with slim.arg_scope(
 377           training_scope(l2_weight_decay=1e-4, is_training=self._is_training)):
 378         # Create C6 by downsampling C5.
 379         c6 = slim.max_pool2d(
 380             _maybe_pad(image_features['layer_18'], self._use_explicit_padding),
 381             [3, 3],
 382             stride=[2, 2],
 383             padding='VALID' if self._use_explicit_padding else 'SAME',
 384             scope='C6_downsample')
 385         c6 = slim.conv2d(
 386             c6,
 387             multiplier_func(self._fpn_layer_depth),
 388             [1, 1],
 389             activation_fn=tf.identity,
 390             normalizer_fn=slim.batch_norm,
 391             weights_regularizer=None,  # this 1x1 has no kernel regularizer.
 392             padding='VALID',
 393             scope='C6_Conv1x1')
 394         image_features['C6'] = tf.identity(c6)  # Needed for quantization.
 395         for k in sorted(image_features.keys()):
 396           tf.logging.error('{}: {}'.format(k, image_features[k]))
 397 
 398         mnasfpn_inputs = [
 399             image_features['layer_7'],  # C3
 400             image_features['layer_14'],  # C4
 401             image_features['layer_18'],  # C5
 402             image_features['C6']  # C6
 403         ]
 404         self._verify_config(mnasfpn_inputs)
 405         feature_maps = mnasfpn(
 406             mnasfpn_inputs,
 407             head_def=self._head_def,
 408             output_channel=self._fpn_layer_depth,
 409             use_explicit_padding=self._use_explicit_padding,
 410             use_native_resize_op=self._use_native_resize_op,
 411             multiplier_func=multiplier_func)
 412     return feature_maps
