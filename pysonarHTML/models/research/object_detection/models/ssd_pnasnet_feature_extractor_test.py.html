  16 """Tests for ssd_pnas_feature_extractor."""
  17 import numpy as np
  18 import tensorflow.compat.v1 as tf
  19 
  20 from object_detection.models import ssd_feature_extractor_test
  21 from object_detection.models import ssd_pnasnet_feature_extractor
  22 
  23 
  24 class SsdPnasNetFeatureExtractorTest(
  25     ssd_feature_extractor_test.SsdFeatureExtractorTestBase):
  26 
  27   def _create_feature_extractor(self,
  28                                 depth_multiplier,
  29                                 pad_to_multiple,
  30                                 use_explicit_padding=False,
  31                                 num_layers=6,
  32                                 is_training=True):
  33     """Constructs a new feature extractor.
  34 
  35     Args:
  36       depth_multiplier: float depth multiplier for feature extractor
  37       pad_to_multiple: the nearest multiple to zero pad the input height and
  38         width dimensions to.
  39       use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
  40         inputs so that the output dimensions are the same as if 'SAME' padding
  41         were used.
  42       num_layers: number of SSD layers.
  43       is_training: whether the network is in training mode.
  44     Returns:
  45       an ssd_meta_arch.SSDFeatureExtractor object.
  46     """
  47     min_depth = 32
  48     return ssd_pnasnet_feature_extractor.SSDPNASNetFeatureExtractor(
  49         is_training,
  50         depth_multiplier,
  51         min_depth,
  52         pad_to_multiple,
  53         self.conv_hyperparams_fn,
  54         use_explicit_padding=use_explicit_padding,
  55         num_layers=num_layers)
  56 
  57   def test_extract_features_returns_correct_shapes_128(self):
  58     image_height = 128
  59     image_width = 128
  60     depth_multiplier = 1.0
  61     pad_to_multiple = 1
  62     expected_feature_map_shape = [(2, 8, 8, 2160), (2, 4, 4, 4320),
  63                                   (2, 2, 2, 512), (2, 1, 1, 256),
  64                                   (2, 1, 1, 256), (2, 1, 1, 128)]
  65     self.check_extract_features_returns_correct_shape(
  66         2, image_height, image_width, depth_multiplier, pad_to_multiple,
  67         expected_feature_map_shape)
  68 
  69   def test_extract_features_returns_correct_shapes_299(self):
  70     image_height = 299
  71     image_width = 299
  72     depth_multiplier = 1.0
  73     pad_to_multiple = 1
  74     expected_feature_map_shape = [(2, 19, 19, 2160), (2, 10, 10, 4320),
  75                                   (2, 5, 5, 512), (2, 3, 3, 256),
  76                                   (2, 2, 2, 256), (2, 1, 1, 128)]
  77     self.check_extract_features_returns_correct_shape(
  78         2, image_height, image_width, depth_multiplier, pad_to_multiple,
  79         expected_feature_map_shape)
  80 
  81   def test_preprocess_returns_correct_value_range(self):
  82     image_height = 128
  83     image_width = 128
  84     depth_multiplier = 1
  85     pad_to_multiple = 1
  86     test_image = np.random.rand(2, image_height, image_width, 3)
  87     feature_extractor = self._create_feature_extractor(depth_multiplier,
  88                                                        pad_to_multiple)
  89     preprocessed_image = feature_extractor.preprocess(test_image)
  90     self.assertTrue(np.all(np.less_equal(np.abs(preprocessed_image), 1.0)))
  91 
  92   def test_extract_features_with_fewer_layers(self):
  93     image_height = 128
  94     image_width = 128
  95     depth_multiplier = 1.0
  96     pad_to_multiple = 1
  97     expected_feature_map_shape = [(2, 8, 8, 2160), (2, 4, 4, 4320),
  98                                   (2, 2, 2, 512), (2, 1, 1, 256)]
  99     self.check_extract_features_returns_correct_shape(
 100         2, image_height, image_width, depth_multiplier, pad_to_multiple,
 101         expected_feature_map_shape, num_layers=4)
