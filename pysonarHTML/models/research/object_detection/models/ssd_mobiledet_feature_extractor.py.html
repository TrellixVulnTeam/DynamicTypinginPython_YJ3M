  31 def _scale_filters(filters, multiplier, base=8):
  32   """Scale the filters accordingly to (multiplier, base)."""
  33   round_half_up = int(int(filters) * multiplier / base + 0.5)
  34   result = int(round_half_up * base)
  35   return max(result, base)

  38 def _swish6(h):
  39   with tf.name_scope('swish6'):
  40     return h * tf.nn.relu6(h + np.float32(3)) * np.float32(1. / 6.)

  43 def _conv(h, filters, kernel_size, strides=1,
  44           normalizer_fn=slim.batch_norm, activation_fn=tf.nn.relu6):
  45   if activation_fn is None:
  46     raise ValueError('Activation function cannot be None. Use tf.identity '
  47                      'instead to better support quantized training.')
  48   return slim.conv2d(
  49       h,
  50       filters,
  51       kernel_size,
  52       stride=strides,
  53       activation_fn=activation_fn,
  54       normalizer_fn=normalizer_fn,
  55       weights_initializer=tf.initializers.he_normal(),
  56       weights_regularizer=slim.l2_regularizer(BACKBONE_WEIGHT_DECAY),
  57       padding='SAME')
  58

  60 def _separable_conv(
  61     h, filters, kernel_size, strides=1, activation_fn=tf.nn.relu6):
  62   """Separable convolution layer."""
  63   if activation_fn is None:
  64     raise ValueError('Activation function cannot be None. Use tf.identity '
  65                      'instead to better support quantized training.')
  66   # Depthwise variant of He initialization derived under the principle proposed
  67   # in the original paper. Note the original He normalization was designed for
  68   # full convolutions and calling tf.initializers.he_normal() can over-estimate
  69   # the fan-in of a depthwise kernel by orders of magnitude.
  70   stddev = (2.0 / kernel_size**2)**0.5 / .87962566103423978
  71   depthwise_initializer = tf.initializers.truncated_normal(stddev=stddev)
  72   return slim.separable_conv2d(
  73       h,
  74       filters,
  75       kernel_size,
  76       stride=strides,
  77       activation_fn=activation_fn,
  78       normalizer_fn=slim.batch_norm,
  79       weights_initializer=depthwise_initializer,
  80       pointwise_initializer=tf.initializers.he_normal(),
  81       weights_regularizer=slim.l2_regularizer(BACKBONE_WEIGHT_DECAY),
  82       padding='SAME')

  85 def _squeeze_and_excite(h, hidden_dim, activation_fn=tf.nn.relu6):
  86   with tf.variable_scope(None, default_name='SqueezeExcite'):
  87     height, width = h.shape[1], h.shape[2]
  88     u = slim.avg_pool2d(h, [height, width], stride=1, padding='VALID')
  89     u = _conv(u, hidden_dim, 1,
  90               normalizer_fn=None, activation_fn=activation_fn)
  91     u = _conv(u, h.shape[-1], 1,
  92               normalizer_fn=None, activation_fn=tf.nn.sigmoid)
  93     return u * h

  96 def _inverted_bottleneck_no_expansion(
  97     h, filters, activation_fn=tf.nn.relu6,
  98     kernel_size=3, strides=1, use_se=False):
  99   """Inverted bottleneck layer without the first 1x1 expansion convolution."""
 100   with tf.variable_scope(None, default_name='IBNNoExpansion'):
 101     # Setting filters to None will make _separable_conv a depthwise conv.
 102     h = _separable_conv(
 103         h, None, kernel_size, strides=strides, activation_fn=activation_fn)
 104     if use_se:
 105       hidden_dim = _scale_filters(h.shape[-1], 0.25)
 106       h = _squeeze_and_excite(h, hidden_dim, activation_fn=activation_fn)
 107     h = _conv(h, filters, 1, activation_fn=tf.identity)
 108     return h

 111 def _inverted_bottleneck(
 112     h, filters, activation_fn=tf.nn.relu6,
 113     kernel_size=3, expansion=8, strides=1, use_se=False, residual=True):
 114   """Inverted bottleneck layer."""
 115   with tf.variable_scope(None, default_name='IBN'):
 116     shortcut = h
 117     expanded_filters = int(h.shape[-1]) * expansion
 118     if expansion <= 1:
 119       raise ValueError('Expansion factor must be greater than 1.')
 120     h = _conv(h, expanded_filters, 1, activation_fn=activation_fn)
 121     # Setting filters to None will make _separable_conv a depthwise conv.
 122     h = _separable_conv(h, None, kernel_size, strides=strides,
 123                         activation_fn=activation_fn)
 124     if use_se:
 125       hidden_dim = _scale_filters(expanded_filters, 0.25)
 126       h = _squeeze_and_excite(h, hidden_dim, activation_fn=activation_fn)
 127     h = _conv(h, filters, 1, activation_fn=tf.identity)
 128     if residual:
 129       h = h + shortcut
 130     return h

 133 def _fused_conv(
 134     h, filters, activation_fn=tf.nn.relu6,
 135     kernel_size=3, expansion=8, strides=1, use_se=False, residual=True):
 136   """Fused convolution layer."""
 137   with tf.variable_scope(None, default_name='FusedConv'):
 138     shortcut = h
 139     expanded_filters = int(h.shape[-1]) * expansion
 140     if expansion <= 1:
 141       raise ValueError('Expansion factor must be greater than 1.')
 142     h = _conv(h, expanded_filters, kernel_size, strides=strides,
 143               activation_fn=activation_fn)
 144     if use_se:
 145       hidden_dim = _scale_filters(expanded_filters, 0.25)
 146       h = _squeeze_and_excite(h, hidden_dim, activation_fn=activation_fn)
 147     h = _conv(h, filters, 1, activation_fn=tf.identity)
 148     if residual:
 149       h = h + shortcut
 150     return h

 153 def _tucker_conv(
 154     h, filters, activation_fn=tf.nn.relu6,
 155     kernel_size=3, input_rank_ratio=0.25, output_rank_ratio=0.25,
 156     strides=1, residual=True):
 157   """Tucker convolution layer (generalized bottleneck)."""
 158   with tf.variable_scope(None, default_name='TuckerConv'):
 159     shortcut = h
 160     input_rank = _scale_filters(h.shape[-1], input_rank_ratio)
 161     h = _conv(h, input_rank, 1, activation_fn=activation_fn)
 162     output_rank = _scale_filters(filters, output_rank_ratio)
 163     h = _conv(h, output_rank, kernel_size, strides=strides,
 164               activation_fn=activation_fn)
 165     h = _conv(h, filters, 1, activation_fn=tf.identity)
 166     if residual:
 167       h = h + shortcut
 168     return h

 171 def mobiledet_cpu_backbone(h, multiplier=1.0):
 172   """Build a MobileDet CPU backbone."""
 173   def _scale(filters):
 174     return _scale_filters(filters, multiplier)
 175   ibn = functools.partial(
 176       _inverted_bottleneck, use_se=True, activation_fn=_swish6)
 177 
 178   endpoints = {}
 179   h = _conv(h, _scale(16), 3, strides=2, activation_fn=_swish6)
 180   h = _inverted_bottleneck_no_expansion(
 181       h, _scale(8), use_se=True, activation_fn=_swish6)
 182   endpoints['C1'] = h
 183   h = ibn(h, _scale(16), expansion=4, strides=2, residual=False)
 184   endpoints['C2'] = h
 185   h = ibn(h, _scale(32), expansion=8, strides=2, residual=False)
 186   h = ibn(h, _scale(32), expansion=4)
 187   h = ibn(h, _scale(32), expansion=4)
 188   h = ibn(h, _scale(32), expansion=4)
 189   endpoints['C3'] = h
 190   h = ibn(h, _scale(72), kernel_size=5, expansion=8, strides=2, residual=False)
 191   h = ibn(h, _scale(72), expansion=8)
 192   h = ibn(h, _scale(72), kernel_size=5, expansion=4)
 193   h = ibn(h, _scale(72), expansion=4)
 194   h = ibn(h, _scale(72), expansion=8, residual=False)
 195   h = ibn(h, _scale(72), expansion=8)
 196   h = ibn(h, _scale(72), expansion=8)
 197   h = ibn(h, _scale(72), expansion=8)
 198   endpoints['C4'] = h
 199   h = ibn(h, _scale(104), kernel_size=5, expansion=8, strides=2, residual=False)
 200   h = ibn(h, _scale(104), kernel_size=5, expansion=4)
 201   h = ibn(h, _scale(104), kernel_size=5, expansion=4)
 202   h = ibn(h, _scale(104), expansion=4)
 203   h = ibn(h, _scale(144), expansion=8, residual=False)
 204   endpoints['C5'] = h
 205   return endpoints

 208 def mobiledet_dsp_backbone(h, multiplier=1.0):
 209   """Build a MobileDet DSP backbone."""
 210   def _scale(filters):
 211     return _scale_filters(filters, multiplier)
 212 
 213   ibn = functools.partial(_inverted_bottleneck, activation_fn=tf.nn.relu6)
 214   fused = functools.partial(_fused_conv, activation_fn=tf.nn.relu6)
 215   tucker = functools.partial(_tucker_conv, activation_fn=tf.nn.relu6)
 216 
 217   endpoints = {}
 218   h = _conv(h, _scale(32), 3, strides=2, activation_fn=tf.nn.relu6)
 219   h = _inverted_bottleneck_no_expansion(
 220       h, _scale(24), activation_fn=tf.nn.relu6)
 221   endpoints['C1'] = h
 222   h = fused(h, _scale(32), expansion=4, strides=2, residual=False)
 223   h = fused(h, _scale(32), expansion=4)
 224   h = ibn(h, _scale(32), expansion=4)
 225   h = tucker(h, _scale(32), input_rank_ratio=0.25, output_rank_ratio=0.75)
 226   endpoints['C2'] = h
 227   h = fused(h, _scale(64), expansion=8, strides=2, residual=False)
 228   h = ibn(h, _scale(64), expansion=4)
 229   h = fused(h, _scale(64), expansion=4)
 230   h = fused(h, _scale(64), expansion=4)
 231   endpoints['C3'] = h
 232   h = fused(h, _scale(120), expansion=8, strides=2, residual=False)
 233   h = ibn(h, _scale(120), expansion=4)
 234   h = ibn(h, _scale(120), expansion=8)
 235   h = ibn(h, _scale(120), expansion=8)
 236   h = fused(h, _scale(144), expansion=8, residual=False)
 237   h = ibn(h, _scale(144), expansion=8)
 238   h = ibn(h, _scale(144), expansion=8)
 239   h = ibn(h, _scale(144), expansion=8)
 240   endpoints['C4'] = h
 241   h = ibn(h, _scale(160), expansion=4, strides=2, residual=False)
 242   h = ibn(h, _scale(160), expansion=4)
 243   h = fused(h, _scale(160), expansion=4)
 244   h = tucker(h, _scale(160), input_rank_ratio=0.75, output_rank_ratio=0.75)
 245   h = ibn(h, _scale(240), expansion=8, residual=False)
 246   endpoints['C5'] = h
 247   return endpoints

 250 def mobiledet_edgetpu_backbone(h, multiplier=1.0):
 251   """Build a MobileDet EdgeTPU backbone."""
 252   def _scale(filters):
 253     return _scale_filters(filters, multiplier)
 254 
 255   ibn = functools.partial(_inverted_bottleneck, activation_fn=tf.nn.relu6)
 256   fused = functools.partial(_fused_conv, activation_fn=tf.nn.relu6)
 257   tucker = functools.partial(_tucker_conv, activation_fn=tf.nn.relu6)
 258 
 259   endpoints = {}
 260   h = _conv(h, _scale(32), 3, strides=2, activation_fn=tf.nn.relu6)
 261   h = tucker(h, _scale(16),
 262              input_rank_ratio=0.25, output_rank_ratio=0.75, residual=False)
 263   endpoints['C1'] = h
 264   h = fused(h, _scale(16), expansion=8, strides=2, residual=False)
 265   h = fused(h, _scale(16), expansion=4)
 266   h = fused(h, _scale(16), expansion=8)
 267   h = fused(h, _scale(16), expansion=4)
 268   endpoints['C2'] = h
 269   h = fused(h, _scale(40), expansion=8, kernel_size=5, strides=2,
 270             residual=False)
 271   h = fused(h, _scale(40), expansion=4)
 272   h = fused(h, _scale(40), expansion=4)
 273   h = fused(h, _scale(40), expansion=4)
 274   endpoints['C3'] = h
 275   h = ibn(h, _scale(72), expansion=8, strides=2, residual=False)
 276   h = ibn(h, _scale(72), expansion=8)
 277   h = fused(h, _scale(72), expansion=4)
 278   h = fused(h, _scale(72), expansion=4)
 279   h = ibn(h, _scale(96), expansion=8, kernel_size=5, residual=False)
 280   h = ibn(h, _scale(96), expansion=8, kernel_size=5)
 281   h = ibn(h, _scale(96), expansion=8)
 282   h = ibn(h, _scale(96), expansion=8)
 283   endpoints['C4'] = h
 284   h = ibn(h, _scale(120), expansion=8, kernel_size=5, strides=2, residual=False)
 285   h = ibn(h, _scale(120), expansion=8)
 286   h = ibn(h, _scale(120), expansion=4, kernel_size=5)
 287   h = ibn(h, _scale(120), expansion=8)
 288   h = ibn(h, _scale(384), expansion=8, kernel_size=5, residual=False)
 289   endpoints['C5'] = h
 290   return endpoints

 293 class SSDMobileDetFeatureExtractorBase(ssd_meta_arch.SSDFeatureExtractor):
 294   """Base class of SSD feature extractor using MobileDet features."""
 295 
 296   def __init__(self,
 297                backbone_fn,
 298                is_training,
 299                depth_multiplier,
 300                min_depth,
 301                pad_to_multiple,
 302                conv_hyperparams_fn,
 303                reuse_weights=None,
 304                use_explicit_padding=False,
 305                use_depthwise=False,
 306                override_base_feature_extractor_hyperparams=False,
 307                scope_name='MobileDet'):
 308     """MobileDet Feature Extractor for SSD Models.
 309 
 310     Reference:
 311       https://arxiv.org/abs/2004.14525
 312 
 313     Args:
 314       backbone_fn: function to construct the MobileDet backbone.
 315       is_training: whether the network is in training mode.
 316       depth_multiplier: float depth multiplier for feature extractor.
 317       min_depth: Integer, minimum feature extractor depth (number of filters).
 318       pad_to_multiple: the nearest multiple to zero pad the input height and
 319         width dimensions to.
 320       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
 321         and separable_conv2d ops in the layers that are added on top of the base
 322         feature extractor.
 323       reuse_weights: Whether to reuse variables. Default is None.
 324       use_explicit_padding: Whether to use explicit padding when extracting
 325         features.
 326       use_depthwise: Whether to use depthwise convolutions in the SSD head.
 327       override_base_feature_extractor_hyperparams: Whether to override
 328         hyperparameters of the base feature extractor with the one from
 329         `conv_hyperparams_fn`.
 330       scope_name: scope name (string) of network variables.
 331     """
 332     if use_explicit_padding:
 333       raise NotImplementedError(
 334           'Explicit padding is not yet supported in MobileDet backbones.')
 335 
 336     super(SSDMobileDetFeatureExtractorBase, self).__init__(
 337         is_training=is_training,
 338         depth_multiplier=depth_multiplier,
 339         min_depth=min_depth,
 340         pad_to_multiple=pad_to_multiple,
 341         conv_hyperparams_fn=conv_hyperparams_fn,
 342         reuse_weights=reuse_weights,
 343         use_explicit_padding=use_explicit_padding,
 344         use_depthwise=use_depthwise,
 345         override_base_feature_extractor_hyperparams=override_base_feature_extractor_hyperparams
 346     )
 347     self._backbone_fn = backbone_fn
 348     self._scope_name = scope_name
 349 
 350   def preprocess(self, resized_inputs):
 351     """SSD preprocessing.
 352 
 353     Maps pixel values to the range [-1, 1]. The preprocessing assumes an input
 354     value range of [0, 255].
 355 
 356     Args:
 357       resized_inputs: a [batch, height, width, channels] float tensor
 358         representing a batch of images.
 359 
 360     Returns:
 361       preprocessed_inputs: a [batch, height, width, channels] float tensor
 362         representing a batch of images.
 363     """
 364     return (2.0 / 255.0) * resized_inputs - 1.0
 365 
 366   def extract_features(self, preprocessed_inputs):
 367     """Extract features from preprocessed inputs.
 368 
 369     Args:
 370       preprocessed_inputs: a [batch, height, width, channels] float tensor
 371         representing a batch of images.
 372 
 373     Returns:
 374       feature_maps: a list of tensors where the ith tensor has shape
 375         [batch, height_i, width_i, depth_i]
 376     """
 377     preprocessed_inputs = shape_utils.check_min_image_dim(
 378         33, preprocessed_inputs)
 379     padded_inputs = ops.pad_to_multiple(
 380         preprocessed_inputs, self._pad_to_multiple)
 381 
 382     feature_map_layout = {
 383         'from_layer': ['C4', 'C5', '', '', '', ''],
 384         # Do not specify the layer depths (number of filters) for C4 and C5, as
 385         # their values are determined based on the backbone.
 386         'layer_depth': [-1, -1, 512, 256, 256, 128],
 387         'use_depthwise': self._use_depthwise,
 388         'use_explicit_padding': self._use_explicit_padding,
 389     }
 390 
 391     with tf.variable_scope(self._scope_name, reuse=self._reuse_weights):
 392       with slim.arg_scope([slim.batch_norm],
 393                           is_training=self._is_training,
 394                           epsilon=0.01, decay=0.99, center=True, scale=True):
 395         endpoints = self._backbone_fn(
 396             padded_inputs,
 397             multiplier=self._depth_multiplier)
 398 
 399       image_features = {'C4': endpoints['C4'], 'C5': endpoints['C5']}
 400       with slim.arg_scope(self._conv_hyperparams_fn()):
 401         feature_maps = feature_map_generators.multi_resolution_feature_maps(
 402             feature_map_layout=feature_map_layout,
 403             depth_multiplier=self._depth_multiplier,
 404             min_depth=self._min_depth,
 405             insert_1x1_conv=True,
 406             image_features=image_features)
 407 
 408     return list(feature_maps.values())
