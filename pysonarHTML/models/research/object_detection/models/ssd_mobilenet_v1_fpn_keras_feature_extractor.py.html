  34 # A modified config of mobilenet v1 that makes it more detection friendly.
  35 def _create_modified_mobilenet_config():
  36   conv_def_block_12 = model_utils.ConvDefs(conv_name='conv_pw_12', filters=512)
  37   conv_def_block_13 = model_utils.ConvDefs(conv_name='conv_pw_13', filters=256)
  38   return [conv_def_block_12, conv_def_block_13]

  41 class SSDMobileNetV1FpnKerasFeatureExtractor(
  42     ssd_meta_arch.SSDKerasFeatureExtractor):
  43   """SSD Feature Extractor using Keras-based MobilenetV1 FPN features."""
  44 
  45   def __init__(self,
  46                is_training,
  47                depth_multiplier,
  48                min_depth,
  49                pad_to_multiple,
  50                conv_hyperparams,
  51                freeze_batchnorm,
  52                inplace_batchnorm_update,
  53                fpn_min_level=3,
  54                fpn_max_level=7,
  55                additional_layer_depth=256,
  56                use_explicit_padding=False,
  57                use_depthwise=False,
  58                use_native_resize_op=False,
  59                override_base_feature_extractor_hyperparams=False,
  60                name=None):
  61     """SSD Keras based FPN feature extractor Mobilenet v1 architecture.
  62 
  63     Args:
  64       is_training: whether the network is in training mode.
  65       depth_multiplier: float depth multiplier for feature extractor.
  66       min_depth: minimum feature extractor depth.
  67       pad_to_multiple: the nearest multiple to zero pad the input height and
  68         width dimensions to.
  69       conv_hyperparams: a `hyperparams_builder.KerasLayerHyperparams` object
  70         containing convolution hyperparameters for the layers added on top of
  71         the base feature extractor.
  72       freeze_batchnorm: whether to freeze batch norm parameters during
  73         training or not. When training with a small batch size (e.g. 1), it is
  74         desirable to freeze batch norm update and use pretrained batch norm
  75         params.
  76       inplace_batchnorm_update: whether to update batch norm moving average
  77         values inplace. When this is false train op must add a control
  78         dependency on tf.graphkeys.UPDATE_OPS collection in order to update
  79         batch norm statistics.
  80       fpn_min_level: the highest resolution feature map to use in FPN. The valid
  81         values are {2, 3, 4, 5} which map to MobileNet v1 layers
  82         {Conv2d_3_pointwise, Conv2d_5_pointwise, Conv2d_11_pointwise,
  83         Conv2d_13_pointwise}, respectively.
  84       fpn_max_level: the smallest resolution feature map to construct or use in
  85         FPN. FPN constructions uses features maps starting from fpn_min_level
  86         upto the fpn_max_level. In the case that there are not enough feature
  87         maps in the backbone network, additional feature maps are created by
  88         applying stride 2 convolutions until we get the desired number of fpn
  89         levels.
  90       additional_layer_depth: additional feature map layer channel depth.
  91       use_explicit_padding: Whether to use explicit padding when extracting
  92         features. Default is False.
  93       use_depthwise: whether to use depthwise convolutions. Default is False.
  94       use_native_resize_op: Whether to use tf.image.nearest_neighbor_resize
  95         to do upsampling in FPN. Default is false.
  96       override_base_feature_extractor_hyperparams: Whether to override
  97         hyperparameters of the base feature extractor with the one from
  98         `conv_hyperparams`.
  99       name: a string name scope to assign to the model. If 'None', Keras
 100         will auto-generate one from the class name.
 101     """
 102     super(SSDMobileNetV1FpnKerasFeatureExtractor, self).__init__(
 103         is_training=is_training,
 104         depth_multiplier=depth_multiplier,
 105         min_depth=min_depth,
 106         pad_to_multiple=pad_to_multiple,
 107         conv_hyperparams=conv_hyperparams,
 108         freeze_batchnorm=freeze_batchnorm,
 109         inplace_batchnorm_update=inplace_batchnorm_update,
 110         use_explicit_padding=use_explicit_padding,
 111         use_depthwise=use_depthwise,
 112         override_base_feature_extractor_hyperparams=
 113         override_base_feature_extractor_hyperparams,
 114         name=name)
 115     self._fpn_min_level = fpn_min_level
 116     self._fpn_max_level = fpn_max_level
 117     self._additional_layer_depth = additional_layer_depth
 118     self._conv_defs = None
 119     if self._use_depthwise:
 120       self._conv_defs = _create_modified_mobilenet_config()
 121     self._use_native_resize_op = use_native_resize_op
 122     self._feature_blocks = [
 123         'Conv2d_3_pointwise', 'Conv2d_5_pointwise', 'Conv2d_11_pointwise',
 124         'Conv2d_13_pointwise'
 125     ]
 126     self._mobilenet_v1 = None
 127     self._fpn_features_generator = None
 128     self._coarse_feature_layers = []
 129 
 130   def build(self, input_shape):
 131     full_mobilenet_v1 = mobilenet_v1.mobilenet_v1(
 132         batchnorm_training=(self._is_training and not self._freeze_batchnorm),
 133         conv_hyperparams=(self._conv_hyperparams
 134                           if self._override_base_feature_extractor_hyperparams
 135                           else None),
 136         weights=None,
 137         use_explicit_padding=self._use_explicit_padding,
 138         alpha=self._depth_multiplier,
 139         min_depth=self._min_depth,
 140         conv_defs=self._conv_defs,
 141         include_top=False)
 142     conv2d_3_pointwise = full_mobilenet_v1.get_layer(
 143         name='conv_pw_3_relu').output
 144     conv2d_5_pointwise = full_mobilenet_v1.get_layer(
 145         name='conv_pw_5_relu').output
 146     conv2d_11_pointwise = full_mobilenet_v1.get_layer(
 147         name='conv_pw_11_relu').output
 148     conv2d_13_pointwise = full_mobilenet_v1.get_layer(
 149         name='conv_pw_13_relu').output
 150     self._mobilenet_v1 = tf.keras.Model(
 151         inputs=full_mobilenet_v1.inputs,
 152         outputs=[conv2d_3_pointwise, conv2d_5_pointwise,
 153                  conv2d_11_pointwise, conv2d_13_pointwise]
 154     )
 155     # pylint:disable=g-long-lambda
 156     self._depth_fn = lambda d: max(
 157         int(d * self._depth_multiplier), self._min_depth)
 158     self._base_fpn_max_level = min(self._fpn_max_level, 5)
 159     self._num_levels = self._base_fpn_max_level + 1 - self._fpn_min_level
 160     self._fpn_features_generator = (
 161         feature_map_generators.KerasFpnTopDownFeatureMaps(
 162             num_levels=self._num_levels,
 163             depth=self._depth_fn(self._additional_layer_depth),
 164             use_depthwise=self._use_depthwise,
 165             use_explicit_padding=self._use_explicit_padding,
 166             use_native_resize_op=self._use_native_resize_op,
 167             is_training=self._is_training,
 168             conv_hyperparams=self._conv_hyperparams,
 169             freeze_batchnorm=self._freeze_batchnorm,
 170             name='FeatureMaps'))
 171     # Construct coarse feature layers
 172     padding = 'VALID' if self._use_explicit_padding else 'SAME'
 173     kernel_size = 3
 174     stride = 2
 175     for i in range(self._base_fpn_max_level + 1, self._fpn_max_level + 1):
 176       coarse_feature_layers = []
 177       if self._use_explicit_padding:
 178         def fixed_padding(features, kernel_size=kernel_size):
 179           return ops.fixed_padding(features, kernel_size)
 180         coarse_feature_layers.append(tf.keras.layers.Lambda(
 181             fixed_padding, name='fixed_padding'))
 182       layer_name = 'bottom_up_Conv2d_{}'.format(
 183           i - self._base_fpn_max_level + 13)
 184       conv_block = feature_map_generators.create_conv_block(
 185           self._use_depthwise, kernel_size, padding, stride, layer_name,
 186           self._conv_hyperparams, self._is_training, self._freeze_batchnorm,
 187           self._depth_fn(self._additional_layer_depth))
 188       coarse_feature_layers.extend(conv_block)
 189       self._coarse_feature_layers.append(coarse_feature_layers)
 190     self.built = True
 191 
 192   def preprocess(self, resized_inputs):
 193     """SSD preprocessing.
 194 
 195     Maps pixel values to the range [-1, 1].
 196 
 197     Args:
 198       resized_inputs: a [batch, height, width, channels] float tensor
 199         representing a batch of images.
 200 
 201     Returns:
 202       preprocessed_inputs: a [batch, height, width, channels] float tensor
 203         representing a batch of images.
 204     """
 205     return (2.0 / 255.0) * resized_inputs - 1.0
 206 
 207   def _extract_features(self, preprocessed_inputs):
 208     """Extract features from preprocessed inputs.
 209 
 210     Args:
 211       preprocessed_inputs: a [batch, height, width, channels] float tensor
 212         representing a batch of images.
 213 
 214     Returns:
 215       feature_maps: a list of tensors where the ith tensor has shape
 216         [batch, height_i, width_i, depth_i]
 217     """
 218     preprocessed_inputs = shape_utils.check_min_image_dim(
 219         33, preprocessed_inputs)
 220 
 221     image_features = self._mobilenet_v1(
 222         ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple))
 223 
 224     feature_block_list = []
 225     for level in range(self._fpn_min_level, self._base_fpn_max_level + 1):
 226       feature_block_list.append(self._feature_blocks[level - 2])
 227 
 228     feature_start_index = len(self._feature_blocks) - self._num_levels
 229     fpn_input_image_features = [
 230         (key, image_features[feature_start_index + index])
 231         for index, key in enumerate(feature_block_list)]
 232     fpn_features = self._fpn_features_generator(fpn_input_image_features)
 233 
 234     feature_maps = []
 235     for level in range(self._fpn_min_level, self._base_fpn_max_level + 1):
 236       feature_maps.append(fpn_features['top_down_{}'.format(
 237           self._feature_blocks[level - 2])])
 238     last_feature_map = fpn_features['top_down_{}'.format(
 239         self._feature_blocks[self._base_fpn_max_level - 2])]
 240 
 241     for coarse_feature_layers in self._coarse_feature_layers:
 242       for layer in coarse_feature_layers:
 243         last_feature_map = layer(last_feature_map)
 244       feature_maps.append(last_feature_map)
 245     return feature_maps
