  63   def test_preprocess_returns_correct_value_range(self):
  64     image_height = 128
  65     image_width = 128
  66     depth_multiplier = 1
  67     pad_to_multiple = 1
  68     test_image = tf.constant(np.random.rand(4, image_height, image_width, 3))
  69     feature_extractor = self._create_feature_extractor(depth_multiplier,
  70                                                        pad_to_multiple)
  71     preprocessed_image = feature_extractor.preprocess(test_image)
  72     with self.test_session() as sess:
  73       test_image_out, preprocessed_image_out = sess.run(
  74           [test_image, preprocessed_image])
  75       self.assertAllClose(preprocessed_image_out,
  76                           test_image_out - [[123.68, 116.779, 103.939]])
