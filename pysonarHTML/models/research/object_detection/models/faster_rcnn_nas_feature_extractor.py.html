  40 def nasnet_large_arg_scope_for_detection(is_batch_norm_training=False):
  41   """Defines the default arg scope for the NASNet-A Large for object detection.
  42 
  43   This provides a small edit to switch batch norm training on and off.
  44 
  45   Args:
  46     is_batch_norm_training: Boolean indicating whether to train with batch norm.
  47 
  48   Returns:
  49     An `arg_scope` to use for the NASNet Large Model.
  50   """
  51   imagenet_scope = nasnet.nasnet_large_arg_scope()
  52   with arg_scope(imagenet_scope):
  53     with arg_scope([slim.batch_norm], is_training=is_batch_norm_training) as sc:
  54       return sc

  57 # Note: This is largely a copy of _build_nasnet_base inside nasnet.py but
  58 # with special edits to remove instantiation of the stem and the special
  59 # ability to receive as input a pair of hidden states.
  60 def _build_nasnet_base(hidden_previous,
  61                        hidden,
  62                        normal_cell,
  63                        reduction_cell,
  64                        hparams,
  65                        true_cell_num,
  66                        start_cell_num):
  67   """Constructs a NASNet image model."""
  68 
  69   # Find where to place the reduction cells or stride normal cells
  70   reduction_indices = nasnet_utils.calc_reduction_layers(
  71       hparams.num_cells, hparams.num_reduction_layers)
  72 
  73   # Note: The None is prepended to match the behavior of _imagenet_stem()
  74   cell_outputs = [None, hidden_previous, hidden]
  75   net = hidden
  76 
  77   # NOTE: In the nasnet.py code, filter_scaling starts at 1.0. We instead
  78   # start at 2.0 because 1 reduction cell has been created which would
  79   # update the filter_scaling to 2.0.
  80   filter_scaling = 2.0
  81 
  82   # Run the cells
  83   for cell_num in range(start_cell_num, hparams.num_cells):
  84     stride = 1
  85     if hparams.skip_reduction_layer_input:
  86       prev_layer = cell_outputs[-2]
  87     if cell_num in reduction_indices:
  88       filter_scaling *= hparams.filter_scaling_rate
  89       net = reduction_cell(
  90           net,
  91           scope='reduction_cell_{}'.format(reduction_indices.index(cell_num)),
  92           filter_scaling=filter_scaling,
  93           stride=2,
  94           prev_layer=cell_outputs[-2],
  95           cell_num=true_cell_num)
  96       true_cell_num += 1
  97       cell_outputs.append(net)
  98     if not hparams.skip_reduction_layer_input:
  99       prev_layer = cell_outputs[-2]
 100     net = normal_cell(
 101         net,
 102         scope='cell_{}'.format(cell_num),
 103         filter_scaling=filter_scaling,
 104         stride=stride,
 105         prev_layer=prev_layer,
 106         cell_num=true_cell_num)
 107     true_cell_num += 1
 108     cell_outputs.append(net)
 109 
 110   # Final nonlinearity.
 111   # Note that we have dropped the final pooling, dropout and softmax layers
 112   # from the default nasnet version.
 113   with tf.variable_scope('final_layer'):
 114     net = tf.nn.relu(net)
 115   return net

 126   def __init__(self,
 127                is_training,
 128                first_stage_features_stride,
 129                batch_norm_trainable=False,
 130                reuse_weights=None,
 131                weight_decay=0.0):
 132     """Constructor.
 133 
 134     Args:
 135       is_training: See base class.
 136       first_stage_features_stride: See base class.
 137       batch_norm_trainable: See base class.
 138       reuse_weights: See base class.
 139       weight_decay: See base class.

 141     Raises:
 142       ValueError: If `first_stage_features_stride` is not 16.

 147         is_training, first_stage_features_stride, batch_norm_trainable,
 148         reuse_weights, weight_decay)

 150   def preprocess(self, resized_inputs):
 151     """Faster R-CNN with NAS preprocessing.
 152 
 153     Maps pixel values to the range [-1, 1].
 154 
 155     Args:
 156       resized_inputs: A [batch, height_in, width_in, channels] float32 tensor
 157         representing a batch of images with values between 0 and 255.0.
 158 
 159     Returns:
 160       preprocessed_inputs: A [batch, height_out, width_out, channels] float32
 161         tensor representing a batch of images.
 162 
 163     """
 164     return (2.0 / 255.0) * resized_inputs - 1.0

 166   def _extract_proposal_features(self, preprocessed_inputs, scope):
 167     """Extracts first stage RPN features.
 168 
 169     Extracts features using the first half of the NASNet network.
 170     We construct the network in `align_feature_maps=True` mode, which means
 171     that all VALID paddings in the network are changed to SAME padding so that
 172     the feature maps are aligned.
 173 
 174     Args:
 175       preprocessed_inputs: A [batch, height, width, channels] float32 tensor
 176         representing a batch of images.
 177       scope: A scope name.
 178 
 179     Returns:
 180       rpn_feature_map: A tensor with shape [batch, height, width, depth]
 181       end_points: A dictionary mapping feature extractor tensor names to tensors
 182 
 183     Raises:
 184       ValueError: If the created network is missing the required activation.
 185     """
 186     del scope
 187 
 188     if len(preprocessed_inputs.get_shape().as_list()) != 4:
 189       raise ValueError('`preprocessed_inputs` must be 4 dimensional, got a '
 190                        'tensor of shape %s' % preprocessed_inputs.get_shape())
 191 
 192     with slim.arg_scope(nasnet_large_arg_scope_for_detection(
 193         is_batch_norm_training=self._train_batch_norm)):
 194       with arg_scope([slim.conv2d,
 195                       slim.batch_norm,
 196                       slim.separable_conv2d],
 197                      reuse=self._reuse_weights):
 198         _, end_points = nasnet.build_nasnet_large(
 199             preprocessed_inputs, num_classes=None,
 200             is_training=self._is_training,
 201             final_endpoint='Cell_11')
 202 
 203     # Note that both 'Cell_10' and 'Cell_11' have equal depth = 2016.
 204     rpn_feature_map = tf.concat([end_points['Cell_10'],
 205                                  end_points['Cell_11']], 3)
 206 
 207     # nasnet.py does not maintain the batch size in the first dimension.
 208     # This work around permits us retaining the batch for below.
 209     batch = preprocessed_inputs.get_shape().as_list()[0]
 210     shape_without_batch = rpn_feature_map.get_shape().as_list()[1:]
 211     rpn_feature_map_shape = [batch] + shape_without_batch
 212     rpn_feature_map.set_shape(rpn_feature_map_shape)
 213 
 214     return rpn_feature_map, end_points

 216   def _extract_box_classifier_features(self, proposal_feature_maps, scope):
 217     """Extracts second stage box classifier features.
 218 
 219     This function reconstructs the "second half" of the NASNet-A
 220     network after the part defined in `_extract_proposal_features`.
 221 
 222     Args:
 223       proposal_feature_maps: A 4-D float tensor with shape
 224         [batch_size * self.max_num_proposals, crop_height, crop_width, depth]
 225         representing the feature map cropped to each proposal.
 226       scope: A scope name.
 227 
 228     Returns:
 229       proposal_classifier_features: A 4-D float tensor with shape
 230         [batch_size * self.max_num_proposals, height, width, depth]
 231         representing box classifier features for each proposal.
 232     """
 233     del scope
 234 
 235     # Note that we always feed into 2 layers of equal depth
 236     # where the first N channels corresponds to previous hidden layer
 237     # and the second N channels correspond to the final hidden layer.
 238     hidden_previous, hidden = tf.split(proposal_feature_maps, 2, axis=3)
 239 
 240     # Note that what follows is largely a copy of build_nasnet_large() within
 241     # nasnet.py. We are copying to minimize code pollution in slim.
 242 
 243     # TODO(shlens,skornblith): Determine the appropriate drop path schedule.
 244     # For now the schedule is the default (1.0->0.7 over 250,000 train steps).
 245     hparams = nasnet.large_imagenet_config()
 246     if not self._is_training:
 247       hparams.set_hparam('drop_path_keep_prob', 1.0)
 248 
 249     # Calculate the total number of cells in the network
 250     # -- Add 2 for the reduction cells.
 251     total_num_cells = hparams.num_cells + 2
 252     # -- And add 2 for the stem cells for ImageNet training.
 253     total_num_cells += 2
 254 
 255     normal_cell = nasnet_utils.NasNetANormalCell(
 256         hparams.num_conv_filters, hparams.drop_path_keep_prob,
 257         total_num_cells, hparams.total_training_steps)
 258     reduction_cell = nasnet_utils.NasNetAReductionCell(
 259         hparams.num_conv_filters, hparams.drop_path_keep_prob,
 260         total_num_cells, hparams.total_training_steps)
 261     with arg_scope([slim.dropout, nasnet_utils.drop_path],
 262                    is_training=self._is_training):
 263       with arg_scope([slim.batch_norm], is_training=self._train_batch_norm):
 264         with arg_scope([slim.avg_pool2d,
 265                         slim.max_pool2d,
 266                         slim.conv2d,
 267                         slim.batch_norm,
 268                         slim.separable_conv2d,
 269                         nasnet_utils.factorized_reduction,
 270                         nasnet_utils.global_avg_pool,
 271                         nasnet_utils.get_channel_index,
 272                         nasnet_utils.get_channel_dim],
 273                        data_format=hparams.data_format):
 274 
 275           # This corresponds to the cell number just past 'Cell_11' used by
 276           # by _extract_proposal_features().
 277           start_cell_num = 12
 278           # Note that this number equals:
 279           #  start_cell_num + 2 stem cells + 1 reduction cell
 280           true_cell_num = 15
 281 
 282           with slim.arg_scope(nasnet.nasnet_large_arg_scope()):
 283             net = _build_nasnet_base(hidden_previous,
 284                                      hidden,
 285                                      normal_cell=normal_cell,
 286                                      reduction_cell=reduction_cell,
 287                                      hparams=hparams,
 288                                      true_cell_num=true_cell_num,
 289                                      start_cell_num=start_cell_num)
 290 
 291     proposal_classifier_features = net
 292     return proposal_classifier_features

 294   def restore_from_classification_checkpoint_fn(
 295       self,
 296       first_stage_feature_extractor_scope,
 297       second_stage_feature_extractor_scope):
 298     """Returns a map of variables to load from a foreign checkpoint.
 299 
 300     Note that this overrides the default implementation in
 301     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor which does not work for
 302     NASNet-A checkpoints.
 303 
 304     Args:
 305       first_stage_feature_extractor_scope: A scope name for the first stage
 306         feature extractor.
 307       second_stage_feature_extractor_scope: A scope name for the second stage
 308         feature extractor.
 309 
 310     Returns:
 311       A dict mapping variable names (to load from a checkpoint) to variables in
 312       the model graph.
 313     """
 314     # Note that the NAS checkpoint only contains the moving average version of
 315     # the Variables so we need to generate an appropriate dictionary mapping.
 316     variables_to_restore = {}
 317     for variable in variables_helper.get_global_variables_safely():
 318       if variable.op.name.startswith(
 319           first_stage_feature_extractor_scope):
 320         var_name = variable.op.name.replace(
 321             first_stage_feature_extractor_scope + '/', '')
 322         var_name += '/ExponentialMovingAverage'
 323         variables_to_restore[var_name] = variable
 324       if variable.op.name.startswith(
 325           second_stage_feature_extractor_scope):
 326         var_name = variable.op.name.replace(
 327             second_stage_feature_extractor_scope + '/', '')
 328         var_name += '/ExponentialMovingAverage'
 329         variables_to_restore[var_name] = variable
 330     return variables_to_restore
