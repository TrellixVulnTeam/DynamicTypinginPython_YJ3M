  16 """Tests for models.faster_rcnn_inception_resnet_v2_keras_feature_extractor."""
  17 
  18 import tensorflow.compat.v1 as tf
  19 
  20 from object_detection.models import faster_rcnn_inception_resnet_v2_keras_feature_extractor as frcnn_inc_res
  21 
  22 
  23 class FasterRcnnInceptionResnetV2KerasFeatureExtractorTest(tf.test.TestCase):
  24 
  25   def _build_feature_extractor(self, first_stage_features_stride):
  26     return frcnn_inc_res.FasterRCNNInceptionResnetV2KerasFeatureExtractor(
  27         is_training=False,
  28         first_stage_features_stride=first_stage_features_stride,
  29         batch_norm_trainable=False,
  30         weight_decay=0.0)
  31 
  32   def test_extract_proposal_features_returns_expected_size(self):
  33     feature_extractor = self._build_feature_extractor(
  34         first_stage_features_stride=16)
  35     preprocessed_inputs = tf.random_uniform(
  36         [1, 299, 299, 3], maxval=255, dtype=tf.float32)
  37     rpn_feature_map = feature_extractor.get_proposal_feature_extractor_model(
  38         name='TestScope')(preprocessed_inputs)
  39     features_shape = tf.shape(rpn_feature_map)
  40 
  41     init_op = tf.global_variables_initializer()
  42     with self.test_session() as sess:
  43       sess.run(init_op)
  44       features_shape_out = sess.run(features_shape)
  45       self.assertAllEqual(features_shape_out, [1, 19, 19, 1088])
  46

  90   def test_extract_box_classifier_features_returns_expected_size(self):
  91     feature_extractor = self._build_feature_extractor(
  92         first_stage_features_stride=16)
  93     proposal_feature_maps = tf.random_uniform(
  94         [2, 17, 17, 1088], maxval=255, dtype=tf.float32)
  95     model = feature_extractor.get_box_classifier_feature_extractor_model(
  96         name='TestScope')
  97     proposal_classifier_features = (
  98         model(proposal_feature_maps))
  99     features_shape = tf.shape(proposal_classifier_features)
 100 
 101     init_op = tf.global_variables_initializer()
 102     with self.test_session() as sess:
 103       sess.run(init_op)
 104       features_shape_out = sess.run(features_shape)
 105       self.assertAllEqual(features_shape_out, [2, 8, 8, 1536])
