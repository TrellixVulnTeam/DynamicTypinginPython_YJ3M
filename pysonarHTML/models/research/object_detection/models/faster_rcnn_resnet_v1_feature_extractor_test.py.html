  18 import numpy as np

  21 from object_detection.models import faster_rcnn_resnet_v1_feature_extractor as faster_rcnn_resnet_v1

  24 class FasterRcnnResnetV1FeatureExtractorTest(tf.test.TestCase):

  26   def _build_feature_extractor(self,
  27                                first_stage_features_stride,
  28                                activation_fn=tf.nn.relu,
  29                                architecture='resnet_v1_101'):
  30     feature_extractor_map = {
  31         'resnet_v1_50':
  32             faster_rcnn_resnet_v1.FasterRCNNResnet50FeatureExtractor,
  33         'resnet_v1_101':
  34             faster_rcnn_resnet_v1.FasterRCNNResnet101FeatureExtractor,
  35         'resnet_v1_152':
  36             faster_rcnn_resnet_v1.FasterRCNNResnet152FeatureExtractor
  37     }
  38     return feature_extractor_map[architecture](
  39         is_training=False,
  40         first_stage_features_stride=first_stage_features_stride,
  41         activation_fn=activation_fn,
  42         batch_norm_trainable=False,
  43         reuse_weights=None,
  44         weight_decay=0.0)
  45 
  46   def test_extract_proposal_features_returns_expected_size(self):
  47     for architecture in ['resnet_v1_50', 'resnet_v1_101', 'resnet_v1_152']:
  48       feature_extractor = self._build_feature_extractor(
  49           first_stage_features_stride=16, architecture=architecture)
  50       preprocessed_inputs = tf.random_uniform(
  51           [4, 224, 224, 3], maxval=255, dtype=tf.float32)
  52       rpn_feature_map, _ = feature_extractor.extract_proposal_features(
  53           preprocessed_inputs, scope='TestScope')
  54       features_shape = tf.shape(rpn_feature_map)
  55 
  56       init_op = tf.global_variables_initializer()
  57       with self.test_session() as sess:
  58         sess.run(init_op)
  59         features_shape_out = sess.run(features_shape)
  60         self.assertAllEqual(features_shape_out, [4, 14, 14, 1024])
  61

  96   def test_extract_proposal_features_dies_on_very_small_images(self):
  97     feature_extractor = self._build_feature_extractor(
  98         first_stage_features_stride=16)
  99     preprocessed_inputs = tf.placeholder(tf.float32, (4, None, None, 3))
 100     rpn_feature_map, _ = feature_extractor.extract_proposal_features(
 101         preprocessed_inputs, scope='TestScope')
 102     features_shape = tf.shape(rpn_feature_map)
 103 
 104     init_op = tf.global_variables_initializer()
 105     with self.test_session() as sess:
 106       sess.run(init_op)
 107       with self.assertRaises(tf.errors.InvalidArgumentError):
 108         sess.run(
 109             features_shape,
 110             feed_dict={preprocessed_inputs: np.random.rand(4, 32, 32, 3)})

 137   def test_overwriting_activation_fn(self):
 138     for architecture in ['resnet_v1_50', 'resnet_v1_101', 'resnet_v1_152']:
 139       feature_extractor = self._build_feature_extractor(
 140           first_stage_features_stride=16,
 141           architecture=architecture,
 142           activation_fn=tf.nn.relu6)
 143       preprocessed_inputs = tf.random_uniform([4, 224, 224, 3],
 144                                               maxval=255,
 145                                               dtype=tf.float32)
 146       rpn_feature_map, _ = feature_extractor.extract_proposal_features(
 147           preprocessed_inputs, scope='TestStage1Scope')
 148       _ = feature_extractor.extract_box_classifier_features(
 149           rpn_feature_map, scope='TestStaget2Scope')
 150       conv_ops = [
 151           op for op in tf.get_default_graph().get_operations()
 152           if op.type == 'Relu6'
 153       ]
 154       op_names = [op.name for op in conv_ops]
 155 
 156       self.assertIsNotNone(conv_ops)
 157       self.assertIn('TestStage1Scope/resnet_v1_50/resnet_v1_50/conv1/Relu6',
 158                     op_names)
 159       self.assertIn(
 160           'TestStaget2Scope/resnet_v1_50/block4/unit_1/bottleneck_v1/conv1/Relu6',
 161           op_names)
