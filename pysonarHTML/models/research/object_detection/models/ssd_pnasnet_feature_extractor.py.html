  33 def pnasnet_large_arg_scope_for_detection(is_batch_norm_training=False):
  34   """Defines the default arg scope for the PNASNet Large for object detection.
  35 
  36   This provides a small edit to switch batch norm training on and off.
  37 
  38   Args:
  39     is_batch_norm_training: Boolean indicating whether to train with batch norm.
  40     Default is False.
  41 
  42   Returns:
  43     An `arg_scope` to use for the PNASNet Large Model.
  44   """
  45   imagenet_scope = pnasnet.pnasnet_large_arg_scope()
  46   with slim.arg_scope(imagenet_scope):
  47     with slim.arg_scope([slim.batch_norm],
  48                         is_training=is_batch_norm_training) as sc:
  49       return sc

  52 class SSDPNASNetFeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
  53   """SSD Feature Extractor using PNASNet features."""
  54 
  55   def __init__(self,
  56                is_training,
  57                depth_multiplier,
  58                min_depth,
  59                pad_to_multiple,
  60                conv_hyperparams_fn,
  61                reuse_weights=None,
  62                use_explicit_padding=False,
  63                use_depthwise=False,
  64                num_layers=6,
  65                override_base_feature_extractor_hyperparams=False):
  66     """PNASNet Feature Extractor for SSD Models.
  67 
  68     Args:
  69       is_training: whether the network is in training mode.
  70       depth_multiplier: float depth multiplier for feature extractor.
  71       min_depth: minimum feature extractor depth.
  72       pad_to_multiple: the nearest multiple to zero pad the input height and
  73         width dimensions to.
  74       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  75         and separable_conv2d ops in the layers that are added on top of the
  76         base feature extractor.
  77       reuse_weights: Whether to reuse variables. Default is None.
  78       use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
  79         inputs so that the output dimensions are the same as if 'SAME' padding
  80         were used.
  81       use_depthwise: Whether to use depthwise convolutions.
  82       num_layers: Number of SSD layers.
  83       override_base_feature_extractor_hyperparams: Whether to override
  84         hyperparameters of the base feature extractor with the one from
  85         `conv_hyperparams_fn`.
  86     """
  87     super(SSDPNASNetFeatureExtractor, self).__init__(
  88         is_training=is_training,
  89         depth_multiplier=depth_multiplier,
  90         min_depth=min_depth,
  91         pad_to_multiple=pad_to_multiple,
  92         conv_hyperparams_fn=conv_hyperparams_fn,
  93         reuse_weights=reuse_weights,
  94         use_explicit_padding=use_explicit_padding,
  95         use_depthwise=use_depthwise,
  96         num_layers=num_layers,
  97         override_base_feature_extractor_hyperparams=
  98         override_base_feature_extractor_hyperparams)
  99 
 100   def preprocess(self, resized_inputs):
 101     """SSD preprocessing.
 102 
 103     Maps pixel values to the range [-1, 1].
 104 
 105     Args:
 106       resized_inputs: a [batch, height, width, channels] float tensor
 107         representing a batch of images.
 108 
 109     Returns:
 110       preprocessed_inputs: a [batch, height, width, channels] float tensor
 111         representing a batch of images.
 112     """
 113     return (2.0 / 255.0) * resized_inputs - 1.0
 114 
 115   def extract_features(self, preprocessed_inputs):
 116     """Extract features from preprocessed inputs.
 117 
 118     Args:
 119       preprocessed_inputs: a [batch, height, width, channels] float tensor
 120         representing a batch of images.
 121 
 122     Returns:
 123       feature_maps: a list of tensors where the ith tensor has shape
 124         [batch, height_i, width_i, depth_i]
 125     """
 126 
 127     feature_map_layout = {
 128         'from_layer': ['Cell_7', 'Cell_11', '', '', '', ''][:self._num_layers],
 129         'layer_depth': [-1, -1, 512, 256, 256, 128][:self._num_layers],
 130         'use_explicit_padding': self._use_explicit_padding,
 131         'use_depthwise': self._use_depthwise,
 132     }
 133 
 134     with slim.arg_scope(
 135         pnasnet_large_arg_scope_for_detection(
 136             is_batch_norm_training=self._is_training)):
 137       with slim.arg_scope([slim.conv2d, slim.batch_norm, slim.separable_conv2d],
 138                           reuse=self._reuse_weights):
 139         with (slim.arg_scope(self._conv_hyperparams_fn())
 140               if self._override_base_feature_extractor_hyperparams else
 141               context_manager.IdentityContextManager()):
 142           _, image_features = pnasnet.build_pnasnet_large(
 143               ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple),
 144               num_classes=None,
 145               is_training=self._is_training,
 146               final_endpoint='Cell_11')
 147     with tf.variable_scope('SSD_feature_maps', reuse=self._reuse_weights):
 148       with slim.arg_scope(self._conv_hyperparams_fn()):
 149         feature_maps = feature_map_generators.multi_resolution_feature_maps(
 150             feature_map_layout=feature_map_layout,
 151             depth_multiplier=self._depth_multiplier,
 152             min_depth=self._min_depth,
 153             insert_1x1_conv=True,
 154             image_features=image_features)
 155 
 156     return list(feature_maps.values())

 158   def restore_from_classification_checkpoint_fn(self, feature_extractor_scope):
 159     """Returns a map of variables to load from a foreign checkpoint.
 160 
 161     Note that this overrides the default implementation in
 162     ssd_meta_arch.SSDFeatureExtractor which does not work for PNASNet
 163     checkpoints.
 164 
 165     Args:
 166       feature_extractor_scope: A scope name for the first stage feature
 167         extractor.
 168 
 169     Returns:
 170       A dict mapping variable names (to load from a checkpoint) to variables in
 171       the model graph.
 172     """
 173     variables_to_restore = {}
 174     for variable in variables_helper.get_global_variables_safely():
 175       if variable.op.name.startswith(feature_extractor_scope):
 176         var_name = variable.op.name.replace(feature_extractor_scope + '/', '')
 177         var_name += '/ExponentialMovingAverage'
 178         variables_to_restore[var_name] = variable
 179     return variables_to_restore
