  96   def test_extract_features_returns_correct_shapes_with_min_depth(
  97       self, use_keras):
  98     image_height = 256
  99     image_width = 256
 100     depth_multiplier = 1.0
 101     pad_to_multiple = 1
 102     min_depth = 320
 103     expected_feature_map_shape = [(2, 32, 32, min_depth),
 104                                   (2, 16, 16, min_depth),
 105                                   (2, 8, 8, min_depth),
 106                                   (2, 4, 4, min_depth),
 107                                   (2, 2, 2, min_depth)]
 108 
 109     def graph_fn(image_tensor):
 110       feature_extractor = self._create_feature_extractor(
 111           depth_multiplier, pad_to_multiple, min_depth=min_depth,
 112           use_keras=use_keras)
 113       if use_keras:
 114         return feature_extractor(image_tensor)
 115       return feature_extractor.extract_features(image_tensor)
 116 
 117     image_tensor = np.random.rand(2, image_height, image_width,
 118                                   3).astype(np.float32)
 119     feature_maps = self.execute(graph_fn, [image_tensor])
 120     for feature_map, expected_shape in zip(feature_maps,
 121                                            expected_feature_map_shape):
 122       self.assertAllEqual(feature_map.shape, expected_shape)

 148   def test_preprocess_returns_correct_value_range(self, use_keras):
 149     image_height = 128
 150     image_width = 128
 151     depth_multiplier = 1
 152     pad_to_multiple = 1
 153     test_image = tf.constant(np.random.rand(4, image_height, image_width, 3))
 154     feature_extractor = self._create_feature_extractor(depth_multiplier,
 155                                                        pad_to_multiple,
 156                                                        use_keras=use_keras)
 157     preprocessed_image = feature_extractor.preprocess(test_image)
 158     with self.test_session() as sess:
 159       test_image_out, preprocessed_image_out = sess.run(
 160           [test_image, preprocessed_image])
 161       self.assertAllClose(preprocessed_image_out,
 162                           test_image_out - [[123.68, 116.779, 103.939]])

 174   def test_variable_count(self, use_keras):
 175     depth_multiplier = 1
 176     pad_to_multiple = 1
 177     variables = self.get_feature_extractor_variables(
 178         depth_multiplier,
 179         pad_to_multiple,
 180         use_keras=use_keras)
 181     # The number of expected variables in resnet_v1_50, resnet_v1_101,
 182     # and resnet_v1_152 is 279, 534, and 789 respectively.
 183     expected_variables_len = 279
 184     scope_name = self._resnet_scope_name(use_keras)
 185     if scope_name in ('ResNet101V1_FPN', 'resnet_v1_101'):
 186       expected_variables_len = 534
 187     elif scope_name in ('ResNet152V1_FPN', 'resnet_v1_152'):
 188       expected_variables_len = 789
 189     self.assertEqual(len(variables), expected_variables_len)
