  27 class SsdMobilenetV3LargeFeatureExtractorTest(
  28     ssd_mobilenet_v3_feature_extractor_testbase
  29     ._SsdMobilenetV3FeatureExtractorTestBase):
  30 
  31   def _get_input_sizes(self):
  32     """Return first two input feature map sizes."""
  33     return [672, 480]
  34 
  35   def _create_feature_extractor(self,
  36                                 depth_multiplier,
  37                                 pad_to_multiple,
  38                                 use_explicit_padding=False,
  39                                 use_keras=False):
  40     """Constructs a new Mobilenet V3-Large feature extractor.
  41 
  42     Args:
  43       depth_multiplier: float depth multiplier for feature extractor
  44       pad_to_multiple: the nearest multiple to zero pad the input height and
  45         width dimensions to.
  46       use_explicit_padding: use 'VALID' padding for convolutions, but prepad
  47         inputs so that the output dimensions are the same as if 'SAME' padding
  48         were used.
  49       use_keras: if True builds a keras-based feature extractor, if False builds
  50         a slim-based one.
  51 
  52     Returns:
  53       an ssd_meta_arch.SSDFeatureExtractor object.
  54     """
  55     min_depth = 32
  56     return (
  57         ssd_mobilenet_v3_feature_extractor.SSDMobileNetV3LargeFeatureExtractor(
  58             False,
  59             depth_multiplier,
  60             min_depth,
  61             pad_to_multiple,
  62             self.conv_hyperparams_fn,
  63             use_explicit_padding=use_explicit_padding))
  64
