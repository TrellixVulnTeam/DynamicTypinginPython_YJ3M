  29 class _SSDResnetPpnFeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
  30   """SSD feature extractor based on resnet architecture and PPN."""
  31 
  32   def __init__(self,
  33                is_training,
  34                depth_multiplier,
  35                min_depth,
  36                pad_to_multiple,
  37                conv_hyperparams_fn,
  38                resnet_base_fn,
  39                resnet_scope_name,
  40                reuse_weights=None,
  41                use_explicit_padding=False,
  42                use_depthwise=False,
  43                base_feature_map_depth=1024,
  44                num_layers=6,
  45                override_base_feature_extractor_hyperparams=False,
  46                use_bounded_activations=False):
  47     """Resnet based PPN Feature Extractor for SSD Models.
  48 
  49     See go/pooling-pyramid for more details about PPN.
  50 
  51     Args:
  52       is_training: whether the network is in training mode.
  53       depth_multiplier: float depth multiplier for feature extractor.
  54       min_depth: minimum feature extractor depth.
  55       pad_to_multiple: the nearest multiple to zero pad the input height and
  56         width dimensions to.
  57       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  58         and separable_conv2d ops in the layers that are added on top of the
  59         base feature extractor.
  60       resnet_base_fn: base resnet network to use.
  61       resnet_scope_name: scope name to construct resnet
  62       reuse_weights: Whether to reuse variables. Default is None.
  63       use_explicit_padding: Whether to use explicit padding when extracting
  64         features. Default is False.
  65       use_depthwise: Whether to use depthwise convolutions. Default is False.
  66       base_feature_map_depth: Depth of the base feature before the max pooling.
  67       num_layers: Number of layers used to make predictions. They are pooled
  68         from the base feature.
  69       override_base_feature_extractor_hyperparams: Whether to override
  70         hyperparameters of the base feature extractor with the one from
  71         `conv_hyperparams_fn`.
  72       use_bounded_activations: Whether or not to use bounded activations for
  73         resnet v1 bottleneck residual unit. Bounded activations better lend
  74         themselves to quantized inference.
  75     """
  76     super(_SSDResnetPpnFeatureExtractor, self).__init__(
  77         is_training, depth_multiplier, min_depth, pad_to_multiple,
  78         conv_hyperparams_fn, reuse_weights, use_explicit_padding, use_depthwise,
  79         override_base_feature_extractor_hyperparams)
  80     self._resnet_base_fn = resnet_base_fn
  81     self._resnet_scope_name = resnet_scope_name
  82     self._base_feature_map_depth = base_feature_map_depth
  83     self._num_layers = num_layers
  84     self._use_bounded_activations = use_bounded_activations
  85 
  86   def _filter_features(self, image_features):
  87     # TODO(rathodv): Change resnet endpoint to strip scope prefixes instead
  88     # of munging the scope here.
  89     filtered_image_features = dict({})
  90     for key, feature in image_features.items():
  91       feature_name = key.split('/')[-1]
  92       if feature_name in ['block2', 'block3', 'block4']:
  93         filtered_image_features[feature_name] = feature
  94     return filtered_image_features
  95 
  96   def preprocess(self, resized_inputs):
  97     """SSD preprocessing.
  98 
  99     VGG style channel mean subtraction as described here:
 100     https://gist.github.com/ksimonyan/211839e770f7b538e2d8#file-readme-mdnge.
 101     Note that if the number of channels is not equal to 3, the mean subtraction
 102     will be skipped and the original resized_inputs will be returned.
 103 
 104     Args:
 105       resized_inputs: a [batch, height, width, channels] float tensor
 106         representing a batch of images.
 107 
 108     Returns:
 109       preprocessed_inputs: a [batch, height, width, channels] float tensor
 110         representing a batch of images.
 111     """
 112     if resized_inputs.shape.as_list()[3] == 3:
 113       channel_means = [123.68, 116.779, 103.939]
 114       return resized_inputs - [[channel_means]]
 115     else:
 116       return resized_inputs
 117 
 118   def extract_features(self, preprocessed_inputs):
 119     """Extract features from preprocessed inputs.
 120 
 121     Args:
 122       preprocessed_inputs: a [batch, height, width, channels] float tensor
 123         representing a batch of images.
 124 
 125     Returns:
 126       feature_maps: a list of tensors where the ith tensor has shape
 127         [batch, height_i, width_i, depth_i]
 128 
 129     Raises:
 130       ValueError: depth multiplier is not supported.
 131     """
 132     if self._depth_multiplier != 1.0:
 133       raise ValueError('Depth multiplier not supported.')
 134 
 135     preprocessed_inputs = shape_utils.check_min_image_dim(
 136         129, preprocessed_inputs)
 137 
 138     with tf.variable_scope(
 139         self._resnet_scope_name, reuse=self._reuse_weights) as scope:
 140       with slim.arg_scope(resnet_v1.resnet_arg_scope()):
 141         with (slim.arg_scope(self._conv_hyperparams_fn())
 142               if self._override_base_feature_extractor_hyperparams else
 143               context_manager.IdentityContextManager()):
 144           with slim.arg_scope(
 145               [resnet_v1.bottleneck],
 146               use_bounded_activations=self._use_bounded_activations):
 147             _, activations = self._resnet_base_fn(
 148                 inputs=ops.pad_to_multiple(preprocessed_inputs,
 149                                            self._pad_to_multiple),
 150                 num_classes=None,
 151                 is_training=None,
 152                 global_pool=False,
 153                 output_stride=None,
 154                 store_non_strided_activations=True,
 155                 scope=scope)
 156 
 157       with slim.arg_scope(self._conv_hyperparams_fn()):
 158         feature_maps = feature_map_generators.pooling_pyramid_feature_maps(
 159             base_feature_map_depth=self._base_feature_map_depth,
 160             num_layers=self._num_layers,
 161             image_features={
 162                 'image_features': self._filter_features(activations)['block3']
 163             })
 164     return list(feature_maps.values())
