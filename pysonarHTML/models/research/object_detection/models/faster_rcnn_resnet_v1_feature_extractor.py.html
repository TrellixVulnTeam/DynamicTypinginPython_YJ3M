  16 """Resnet V1 Faster R-CNN implementation.
  17 
  18 See "Deep Residual Learning for Image Recognition" by He et al., 2015.
  19 https://arxiv.org/abs/1512.03385
  20 
  21 Note: this implementation assumes that the classification checkpoint used
  22 to finetune this model is trained using the same configuration as that of
  23 the MSRA provided checkpoints
  24 (see https://github.com/KaimingHe/deep-residual-networks), e.g., with
  25 same preprocessing, batch norm scaling, etc.
  26 """

  35 class FasterRCNNResnetV1FeatureExtractor(
  36     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor):
  37   """Faster R-CNN Resnet V1 feature extractor implementation."""
  38 
  39   def __init__(self,
  40                architecture,
  41                resnet_model,
  42                is_training,
  43                first_stage_features_stride,
  44                batch_norm_trainable=False,
  45                reuse_weights=None,
  46                weight_decay=0.0,
  47                activation_fn=tf.nn.relu):
  48     """Constructor.
  49 
  50     Args:
  51       architecture: Architecture name of the Resnet V1 model.
  52       resnet_model: Definition of the Resnet V1 model.
  53       is_training: See base class.
  54       first_stage_features_stride: See base class.
  55       batch_norm_trainable: See base class.
  56       reuse_weights: See base class.
  57       weight_decay: See base class.
  58       activation_fn: Activaton functon to use in Resnet V1 model.
  59 
  60     Raises:
  61       ValueError: If `first_stage_features_stride` is not 8 or 16.
  62     """
  63     if first_stage_features_stride != 8 and first_stage_features_stride != 16:
  64       raise ValueError('`first_stage_features_stride` must be 8 or 16.')
  65     self._architecture = architecture
  66     self._resnet_model = resnet_model
  67     self._activation_fn = activation_fn
  68     super(FasterRCNNResnetV1FeatureExtractor,
  69           self).__init__(is_training, first_stage_features_stride,
  70                          batch_norm_trainable, reuse_weights, weight_decay)
  71 
  72   def preprocess(self, resized_inputs):
  73     """Faster R-CNN Resnet V1 preprocessing.
  74 
  75     VGG style channel mean subtraction as described here:
  76     https://gist.github.com/ksimonyan/211839e770f7b538e2d8#file-readme-md
  77     Note that if the number of channels is not equal to 3, the mean subtraction
  78     will be skipped and the original resized_inputs will be returned.
  79 
  80     Args:
  81       resized_inputs: A [batch, height_in, width_in, channels] float32 tensor
  82         representing a batch of images with values between 0 and 255.0.
  83 
  84     Returns:
  85       preprocessed_inputs: A [batch, height_out, width_out, channels] float32
  86         tensor representing a batch of images.
  87 
  88     """
  89     if resized_inputs.shape.as_list()[3] == 3:
  90       channel_means = [123.68, 116.779, 103.939]
  91       return resized_inputs - [[channel_means]]
  92     else:
  93       return resized_inputs
  94 
  95   def _extract_proposal_features(self, preprocessed_inputs, scope):
  96     """Extracts first stage RPN features.
  97 
  98     Args:
  99       preprocessed_inputs: A [batch, height, width, channels] float32 tensor
 100         representing a batch of images.
 101       scope: A scope name.
 102 
 103     Returns:
 104       rpn_feature_map: A tensor with shape [batch, height, width, depth]
 105       activations: A dictionary mapping feature extractor tensor names to
 106         tensors
 107 
 108     Raises:
 109       InvalidArgumentError: If the spatial size of `preprocessed_inputs`
 110         (height or width) is less than 33.
 111       ValueError: If the created network is missing the required activation.
 112     """
 113     if len(preprocessed_inputs.get_shape().as_list()) != 4:
 114       raise ValueError('`preprocessed_inputs` must be 4 dimensional, got a '
 115                        'tensor of shape %s' % preprocessed_inputs.get_shape())
 116     shape_assert = tf.Assert(
 117         tf.logical_and(
 118             tf.greater_equal(tf.shape(preprocessed_inputs)[1], 33),
 119             tf.greater_equal(tf.shape(preprocessed_inputs)[2], 33)),
 120         ['image size must at least be 33 in both height and width.'])
 121 
 122     with tf.control_dependencies([shape_assert]):
 123       # Disables batchnorm for fine-tuning with smaller batch sizes.
 124       # TODO(chensun): Figure out if it is needed when image
 125       # batch size is bigger.
 126       with slim.arg_scope(
 127           resnet_utils.resnet_arg_scope(
 128               batch_norm_epsilon=1e-5,
 129               batch_norm_scale=True,
 130               activation_fn=self._activation_fn,
 131               weight_decay=self._weight_decay)):
 132         with tf.variable_scope(
 133             self._architecture, reuse=self._reuse_weights) as var_scope:
 134           _, activations = self._resnet_model(
 135               preprocessed_inputs,
 136               num_classes=None,
 137               is_training=self._train_batch_norm,
 138               global_pool=False,
 139               output_stride=self._first_stage_features_stride,
 140               spatial_squeeze=False,
 141               scope=var_scope)
 142 
 143     handle = scope + '/%s/block3' % self._architecture
 144     return activations[handle], activations
 145 
 146   def _extract_box_classifier_features(self, proposal_feature_maps, scope):
 147     """Extracts second stage box classifier features.
 148 
 149     Args:
 150       proposal_feature_maps: A 4-D float tensor with shape
 151         [batch_size * self.max_num_proposals, crop_height, crop_width, depth]
 152         representing the feature map cropped to each proposal.
 153       scope: A scope name (unused).
 154 
 155     Returns:
 156       proposal_classifier_features: A 4-D float tensor with shape
 157         [batch_size * self.max_num_proposals, height, width, depth]
 158         representing box classifier features for each proposal.
 159     """
 160     with tf.variable_scope(self._architecture, reuse=self._reuse_weights):
 161       with slim.arg_scope(
 162           resnet_utils.resnet_arg_scope(
 163               batch_norm_epsilon=1e-5,
 164               batch_norm_scale=True,
 165               activation_fn=self._activation_fn,
 166               weight_decay=self._weight_decay)):
 167         with slim.arg_scope([slim.batch_norm],
 168                             is_training=self._train_batch_norm):
 169           blocks = [
 170               resnet_utils.Block('block4', resnet_v1.bottleneck, [{
 171                   'depth': 2048,
 172                   'depth_bottleneck': 512,
 173                   'stride': 1
 174               }] * 3)
 175           ]
 176           proposal_classifier_features = resnet_utils.stack_blocks_dense(
 177               proposal_feature_maps, blocks)
 178     return proposal_classifier_features
