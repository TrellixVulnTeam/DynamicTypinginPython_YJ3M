  17 """Functions to generate a list of feature maps based on image features.
  18 
  19 Provides several feature map generators that can be used to build object
  20 detection feature extractors.
  21 
  22 Object detection feature extractors usually are built by stacking two components
  23 - A base feature extractor such as Inception V3 and a feature map generator.
  24 Feature map generators build on the base feature extractors and produce a list
  25 of final feature maps.
  26 """

  45 def get_depth_fn(depth_multiplier, min_depth):
  46   """Builds a callable to compute depth (output channels) of conv filters.
  47 
  48   Args:
  49     depth_multiplier: a multiplier for the nominal depth.
  50     min_depth: a lower bound on the depth of filters.
  51 
  52   Returns:
  53     A callable that takes in a nominal depth and returns the depth to use.
  54   """
  55   def multiply_depth(depth):
  56     new_depth = int(depth * depth_multiplier)
  57     return max(new_depth, min_depth)
  58   return multiply_depth

  61 def create_conv_block(
  62     use_depthwise, kernel_size, padding, stride, layer_name, conv_hyperparams,
  63     is_training, freeze_batchnorm, depth):
  64   """Create Keras layers for depthwise & non-depthwise convolutions.
  65 
  66   Args:
  67     use_depthwise: Whether to use depthwise separable conv instead of regular
  68       conv.
  69     kernel_size: A list of length 2: [kernel_height, kernel_width] of the
  70       filters. Can be an int if both values are the same.
  71     padding: One of 'VALID' or 'SAME'.
  72     stride: A list of length 2: [stride_height, stride_width], specifying the
  73       convolution stride. Can be an int if both strides are the same.
  74     layer_name: String. The name of the layer.
  75     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  76       containing hyperparameters for convolution ops.
  77     is_training: Indicates whether the feature generator is in training mode.
  78     freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
  79       training or not. When training with a small batch size (e.g. 1), it is
  80       desirable to freeze batch norm update and use pretrained batch norm
  81       params.
  82     depth: Depth of output feature maps.
  83 
  84   Returns:
  85     A list of conv layers.
  86   """
  87   layers = []
  88   if use_depthwise:
  89     kwargs = conv_hyperparams.params()
  90     # Both the regularizer and initializer apply to the depthwise layer,
  91     # so we remap the kernel_* to depthwise_* here.
  92     kwargs['depthwise_regularizer'] = kwargs['kernel_regularizer']
  93     kwargs['depthwise_initializer'] = kwargs['kernel_initializer']
  94     layers.append(
  95         tf.keras.layers.SeparableConv2D(
  96             depth, [kernel_size, kernel_size],
  97             depth_multiplier=1,
  98             padding=padding,
  99             strides=stride,
 100             name=layer_name + '_depthwise_conv',
 101             **kwargs))
 102   else:
 103     layers.append(tf.keras.layers.Conv2D(
 104         depth,
 105         [kernel_size, kernel_size],
 106         padding=padding,
 107         strides=stride,
 108         name=layer_name + '_conv',
 109         **conv_hyperparams.params()))
 110   layers.append(
 111       conv_hyperparams.build_batch_norm(
 112           training=(is_training and not freeze_batchnorm),
 113           name=layer_name + '_batchnorm'))
 114   layers.append(
 115       conv_hyperparams.build_activation_layer(
 116           name=layer_name))
 117   return layers

 120 class KerasMultiResolutionFeatureMaps(tf.keras.Model):
 121   """Generates multi resolution feature maps from input image features.
 122 
 123   A Keras model that generates multi-scale feature maps for detection as in the
 124   SSD papers by Liu et al: https://arxiv.org/pdf/1512.02325v2.pdf, See Sec 2.1.
 125 
 126   More specifically, when called on inputs it performs the following two tasks:
 127   1) If a layer name is provided in the configuration, returns that layer as a
 128      feature map.
 129   2) If a layer name is left as an empty string, constructs a new feature map
 130      based on the spatial shape and depth configuration. Note that the current
 131      implementation only supports generating new layers using convolution of
 132      stride 2 resulting in a spatial resolution reduction by a factor of 2.
 133      By default convolution kernel size is set to 3, and it can be customized
 134      by caller.
 135 
 136   An example of the configuration for Inception V3:
 137   {
 138     'from_layer': ['Mixed_5d', 'Mixed_6e', 'Mixed_7c', '', '', ''],
 139     'layer_depth': [-1, -1, -1, 512, 256, 128]
 140   }
 141 
 142   When this feature generator object is called on input image_features:
 143     Args:
 144       image_features: A dictionary of handles to activation tensors from the
 145         base feature extractor.
 146 
 147     Returns:
 148       feature_maps: an OrderedDict mapping keys (feature map names) to
 149         tensors where each tensor has shape [batch, height_i, width_i, depth_i].
 150   """
 151 
 152   def __init__(self,
 153                feature_map_layout,
 154                depth_multiplier,
 155                min_depth,
 156                insert_1x1_conv,
 157                is_training,
 158                conv_hyperparams,
 159                freeze_batchnorm,
 160                name=None):
 161     """Constructor.
 162 
 163     Args:
 164       feature_map_layout: Dictionary of specifications for the feature map
 165         layouts in the following format (Inception V2/V3 respectively):
 166         {
 167           'from_layer': ['Mixed_3c', 'Mixed_4c', 'Mixed_5c', '', '', ''],
 168           'layer_depth': [-1, -1, -1, 512, 256, 128]
 169         }
 170         or
 171         {
 172           'from_layer': ['Mixed_5d', 'Mixed_6e', 'Mixed_7c', '', '', ''],
 173           'layer_depth': [-1, -1, -1, 512, 256, 128]
 174         }
 175         If 'from_layer' is specified, the specified feature map is directly used
 176         as a box predictor layer, and the layer_depth is directly infered from
 177         the feature map (instead of using the provided 'layer_depth' parameter).
 178         In this case, our convention is to set 'layer_depth' to -1 for clarity.
 179         Otherwise, if 'from_layer' is an empty string, then the box predictor
 180         layer will be built from the previous layer using convolution
 181         operations. Note that the current implementation only supports
 182         generating new layers using convolutions of stride 2 (resulting in a
 183         spatial resolution reduction by a factor of 2), and will be extended to
 184         a more flexible design. Convolution kernel size is set to 3 by default,
 185         and can be customized by 'conv_kernel_size' parameter (similarily,
 186         'conv_kernel_size' should be set to -1 if 'from_layer' is specified).
 187         The created convolution operation will be a normal 2D convolution by
 188         default, and a depthwise convolution followed by 1x1 convolution if
 189         'use_depthwise' is set to True.
 190       depth_multiplier: Depth multiplier for convolutional layers.
 191       min_depth: Minimum depth for convolutional layers.
 192       insert_1x1_conv: A boolean indicating whether an additional 1x1
 193         convolution should be inserted before shrinking the feature map.
 194       is_training: Indicates whether the feature generator is in training mode.
 195       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 196         containing hyperparameters for convolution ops.
 197       freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
 198         training or not. When training with a small batch size (e.g. 1), it is
 199         desirable to freeze batch norm update and use pretrained batch norm
 200         params.
 201       name: A string name scope to assign to the model. If 'None', Keras
 202         will auto-generate one from the class name.
 203     """
 204     super(KerasMultiResolutionFeatureMaps, self).__init__(name=name)
 205 
 206     self.feature_map_layout = feature_map_layout
 207     self.convolutions = []
 208 
 209     depth_fn = get_depth_fn(depth_multiplier, min_depth)
 210 
 211     base_from_layer = ''
 212     use_explicit_padding = False
 213     if 'use_explicit_padding' in feature_map_layout:
 214       use_explicit_padding = feature_map_layout['use_explicit_padding']
 215     use_depthwise = False
 216     if 'use_depthwise' in feature_map_layout:
 217       use_depthwise = feature_map_layout['use_depthwise']
 218     for index, from_layer in enumerate(feature_map_layout['from_layer']):
 219       net = []
 220       layer_depth = feature_map_layout['layer_depth'][index]
 221       conv_kernel_size = 3
 222       if 'conv_kernel_size' in feature_map_layout:
 223         conv_kernel_size = feature_map_layout['conv_kernel_size'][index]
 224       if from_layer:
 225         base_from_layer = from_layer
 226       else:
 227         if insert_1x1_conv:
 228           layer_name = '{}_1_Conv2d_{}_1x1_{}'.format(
 229               base_from_layer, index, depth_fn(layer_depth // 2))
 230           net.append(tf.keras.layers.Conv2D(depth_fn(layer_depth // 2),
 231                                             [1, 1],
 232                                             padding='SAME',
 233                                             strides=1,
 234                                             name=layer_name + '_conv',
 235                                             **conv_hyperparams.params()))
 236           net.append(
 237               conv_hyperparams.build_batch_norm(
 238                   training=(is_training and not freeze_batchnorm),
 239                   name=layer_name + '_batchnorm'))
 240           net.append(
 241               conv_hyperparams.build_activation_layer(
 242                   name=layer_name))
 243 
 244         layer_name = '{}_2_Conv2d_{}_{}x{}_s2_{}'.format(
 245             base_from_layer, index, conv_kernel_size, conv_kernel_size,
 246             depth_fn(layer_depth))
 247         stride = 2
 248         padding = 'SAME'
 249         if use_explicit_padding:
 250           padding = 'VALID'
 251           # We define this function here while capturing the value of
 252           # conv_kernel_size, to avoid holding a reference to the loop variable
 253           # conv_kernel_size inside of a lambda function
 254           def fixed_padding(features, kernel_size=conv_kernel_size):
 255             return ops.fixed_padding(features, kernel_size)
 256           net.append(tf.keras.layers.Lambda(fixed_padding))
 257         # TODO(rathodv): Add some utilities to simplify the creation of
 258         # Depthwise & non-depthwise convolutions w/ normalization & activations
 259         if use_depthwise:
 260           net.append(tf.keras.layers.DepthwiseConv2D(
 261               [conv_kernel_size, conv_kernel_size],
 262               depth_multiplier=1,
 263               padding=padding,
 264               strides=stride,
 265               name=layer_name + '_depthwise_conv',
 266               **conv_hyperparams.params()))
 267           net.append(
 268               conv_hyperparams.build_batch_norm(
 269                   training=(is_training and not freeze_batchnorm),
 270                   name=layer_name + '_depthwise_batchnorm'))
 271           net.append(
 272               conv_hyperparams.build_activation_layer(
 273                   name=layer_name + '_depthwise'))
 274 
 275           net.append(tf.keras.layers.Conv2D(depth_fn(layer_depth), [1, 1],
 276                                             padding='SAME',
 277                                             strides=1,
 278                                             name=layer_name + '_conv',
 279                                             **conv_hyperparams.params()))
 280           net.append(
 281               conv_hyperparams.build_batch_norm(
 282                   training=(is_training and not freeze_batchnorm),
 283                   name=layer_name + '_batchnorm'))
 284           net.append(
 285               conv_hyperparams.build_activation_layer(
 286                   name=layer_name))
 287 
 288         else:
 289           net.append(tf.keras.layers.Conv2D(
 290               depth_fn(layer_depth),
 291               [conv_kernel_size, conv_kernel_size],
 292               padding=padding,
 293               strides=stride,
 294               name=layer_name + '_conv',
 295               **conv_hyperparams.params()))
 296           net.append(
 297               conv_hyperparams.build_batch_norm(
 298                   training=(is_training and not freeze_batchnorm),
 299                   name=layer_name + '_batchnorm'))
 300           net.append(
 301               conv_hyperparams.build_activation_layer(
 302                   name=layer_name))
 303 
 304       # Until certain bugs are fixed in checkpointable lists,
 305       # this net must be appended only once it's been filled with layers
 306       self.convolutions.append(net)
 307 
 308   def call(self, image_features):
 309     """Generate the multi-resolution feature maps.
 310 
 311     Executed when calling the `.__call__` method on input.
 312 
 313     Args:
 314       image_features: A dictionary of handles to activation tensors from the
 315         base feature extractor.
 316 
 317     Returns:
 318       feature_maps: an OrderedDict mapping keys (feature map names) to
 319         tensors where each tensor has shape [batch, height_i, width_i, depth_i].
 320     """
 321     feature_maps = []
 322     feature_map_keys = []
 323 
 324     for index, from_layer in enumerate(self.feature_map_layout['from_layer']):
 325       if from_layer:
 326         feature_map = image_features[from_layer]
 327         feature_map_keys.append(from_layer)
 328       else:
 329         feature_map = feature_maps[-1]
 330         for layer in self.convolutions[index]:
 331           feature_map = layer(feature_map)
 332         layer_name = self.convolutions[index][-1].name
 333         feature_map_keys.append(layer_name)
 334       feature_maps.append(feature_map)
 335     return collections.OrderedDict(
 336         [(x, y) for (x, y) in zip(feature_map_keys, feature_maps)])

 339 def multi_resolution_feature_maps(feature_map_layout, depth_multiplier,
 340                                   min_depth, insert_1x1_conv, image_features,
 341                                   pool_residual=False):
 342   """Generates multi resolution feature maps from input image features.
 343 
 344   Generates multi-scale feature maps for detection as in the SSD papers by
 345   Liu et al: https://arxiv.org/pdf/1512.02325v2.pdf, See Sec 2.1.
 346 
 347   More specifically, it performs the following two tasks:
 348   1) If a layer name is provided in the configuration, returns that layer as a
 349      feature map.
 350   2) If a layer name is left as an empty string, constructs a new feature map
 351      based on the spatial shape and depth configuration. Note that the current
 352      implementation only supports generating new layers using convolution of
 353      stride 2 resulting in a spatial resolution reduction by a factor of 2.
 354      By default convolution kernel size is set to 3, and it can be customized
 355      by caller.
 356 
 357   An example of the configuration for Inception V3:
 358   {
 359     'from_layer': ['Mixed_5d', 'Mixed_6e', 'Mixed_7c', '', '', ''],
 360     'layer_depth': [-1, -1, -1, 512, 256, 128]
 361   }
 362 
 363   Args:
 364     feature_map_layout: Dictionary of specifications for the feature map
 365       layouts in the following format (Inception V2/V3 respectively):
 366       {
 367         'from_layer': ['Mixed_3c', 'Mixed_4c', 'Mixed_5c', '', '', ''],
 368         'layer_depth': [-1, -1, -1, 512, 256, 128]
 369       }
 370       or
 371       {
 372         'from_layer': ['Mixed_5d', 'Mixed_6e', 'Mixed_7c', '', '', ''],
 373         'layer_depth': [-1, -1, -1, 512, 256, 128]
 374       }
 375       If 'from_layer' is specified, the specified feature map is directly used
 376       as a box predictor layer, and the layer_depth is directly infered from the
 377       feature map (instead of using the provided 'layer_depth' parameter). In
 378       this case, our convention is to set 'layer_depth' to -1 for clarity.
 379       Otherwise, if 'from_layer' is an empty string, then the box predictor
 380       layer will be built from the previous layer using convolution operations.
 381       Note that the current implementation only supports generating new layers
 382       using convolutions of stride 2 (resulting in a spatial resolution
 383       reduction by a factor of 2), and will be extended to a more flexible
 384       design. Convolution kernel size is set to 3 by default, and can be
 385       customized by 'conv_kernel_size' parameter (similarily, 'conv_kernel_size'
 386       should be set to -1 if 'from_layer' is specified). The created convolution
 387       operation will be a normal 2D convolution by default, and a depthwise
 388       convolution followed by 1x1 convolution if 'use_depthwise' is set to True.
 389     depth_multiplier: Depth multiplier for convolutional layers.
 390     min_depth: Minimum depth for convolutional layers.
 391     insert_1x1_conv: A boolean indicating whether an additional 1x1 convolution
 392       should be inserted before shrinking the feature map.
 393     image_features: A dictionary of handles to activation tensors from the
 394       base feature extractor.
 395     pool_residual: Whether to add an average pooling layer followed by a
 396       residual connection between subsequent feature maps when the channel
 397       depth match. For example, with option 'layer_depth': [-1, 512, 256, 256],
 398       a pooling and residual layer is added between the third and forth feature
 399       map. This option is better used with Weight Shared Convolution Box
 400       Predictor when all feature maps have the same channel depth to encourage
 401       more consistent features across multi-scale feature maps.
 402 
 403   Returns:
 404     feature_maps: an OrderedDict mapping keys (feature map names) to
 405       tensors where each tensor has shape [batch, height_i, width_i, depth_i].
 406 
 407   Raises:
 408     ValueError: if the number entries in 'from_layer' and
 409       'layer_depth' do not match.
 410     ValueError: if the generated layer does not have the same resolution
 411       as specified.
 412   """
 413   depth_fn = get_depth_fn(depth_multiplier, min_depth)
 414 
 415   feature_map_keys = []
 416   feature_maps = []
 417   base_from_layer = ''
 418   use_explicit_padding = False
 419   if 'use_explicit_padding' in feature_map_layout:
 420     use_explicit_padding = feature_map_layout['use_explicit_padding']
 421   use_depthwise = False
 422   if 'use_depthwise' in feature_map_layout:
 423     use_depthwise = feature_map_layout['use_depthwise']
 424   for index, from_layer in enumerate(feature_map_layout['from_layer']):
 425     layer_depth = feature_map_layout['layer_depth'][index]
 426     conv_kernel_size = 3
 427     if 'conv_kernel_size' in feature_map_layout:
 428       conv_kernel_size = feature_map_layout['conv_kernel_size'][index]
 429     if from_layer:
 430       feature_map = image_features[from_layer]
 431       base_from_layer = from_layer
 432       feature_map_keys.append(from_layer)
 433     else:
 434       pre_layer = feature_maps[-1]
 435       pre_layer_depth = pre_layer.get_shape().as_list()[3]
 436       intermediate_layer = pre_layer
 437       if insert_1x1_conv:
 438         layer_name = '{}_1_Conv2d_{}_1x1_{}'.format(
 439             base_from_layer, index, depth_fn(layer_depth // 2))
 440         intermediate_layer = slim.conv2d(
 441             pre_layer,
 442             depth_fn(layer_depth // 2), [1, 1],
 443             padding='SAME',
 444             stride=1,
 445             scope=layer_name)
 446       layer_name = '{}_2_Conv2d_{}_{}x{}_s2_{}'.format(
 447           base_from_layer, index, conv_kernel_size, conv_kernel_size,
 448           depth_fn(layer_depth))
 449       stride = 2
 450       padding = 'SAME'
 451       if use_explicit_padding:
 452         padding = 'VALID'
 453         intermediate_layer = ops.fixed_padding(
 454             intermediate_layer, conv_kernel_size)
 455       if use_depthwise:
 456         feature_map = slim.separable_conv2d(
 457             intermediate_layer,
 458             None, [conv_kernel_size, conv_kernel_size],
 459             depth_multiplier=1,
 460             padding=padding,
 461             stride=stride,
 462             scope=layer_name + '_depthwise')
 463         feature_map = slim.conv2d(
 464             feature_map,
 465             depth_fn(layer_depth), [1, 1],
 466             padding='SAME',
 467             stride=1,
 468             scope=layer_name)
 469         if pool_residual and pre_layer_depth == depth_fn(layer_depth):
 470           feature_map += slim.avg_pool2d(
 471               pre_layer, [3, 3],
 472               padding='SAME',
 473               stride=2,
 474               scope=layer_name + '_pool')
 475       else:
 476         feature_map = slim.conv2d(
 477             intermediate_layer,
 478             depth_fn(layer_depth), [conv_kernel_size, conv_kernel_size],
 479             padding=padding,
 480             stride=stride,
 481             scope=layer_name)
 482       feature_map_keys.append(layer_name)
 483     feature_maps.append(feature_map)
 484   return collections.OrderedDict(
 485       [(x, y) for (x, y) in zip(feature_map_keys, feature_maps)])

 488 class KerasFpnTopDownFeatureMaps(tf.keras.Model):
 489   """Generates Keras based `top-down` feature maps for Feature Pyramid Networks.
 490 
 491   See https://arxiv.org/abs/1612.03144 for details.
 492   """
 493 
 494   def __init__(self,
 495                num_levels,
 496                depth,
 497                is_training,
 498                conv_hyperparams,
 499                freeze_batchnorm,
 500                use_depthwise=False,
 501                use_explicit_padding=False,
 502                use_bounded_activations=False,
 503                use_native_resize_op=False,
 504                scope=None,
 505                name=None):
 506     """Constructor.
 507 
 508     Args:
 509       num_levels: the number of image features.
 510       depth: depth of output feature maps.
 511       is_training: Indicates whether the feature generator is in training mode.
 512       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 513         containing hyperparameters for convolution ops.
 514       freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
 515         training or not. When training with a small batch size (e.g. 1), it is
 516         desirable to freeze batch norm update and use pretrained batch norm
 517         params.
 518       use_depthwise: whether to use depthwise separable conv instead of regular
 519         conv.
 520       use_explicit_padding: whether to use explicit padding.
 521       use_bounded_activations: Whether or not to clip activations to range
 522         [-ACTIVATION_BOUND, ACTIVATION_BOUND]. Bounded activations better lend
 523         themselves to quantized inference.
 524       use_native_resize_op: If True, uses tf.image.resize_nearest_neighbor op
 525         for the upsampling process instead of reshape and broadcasting
 526         implementation.
 527       scope: A scope name to wrap this op under.
 528       name: A string name scope to assign to the model. If 'None', Keras
 529         will auto-generate one from the class name.
 530     """
 531     super(KerasFpnTopDownFeatureMaps, self).__init__(name=name)
 532 
 533     self.scope = scope if scope else 'top_down'
 534     self.top_layers = []
 535     self.residual_blocks = []
 536     self.top_down_blocks = []
 537     self.reshape_blocks = []
 538     self.conv_layers = []
 539 
 540     padding = 'VALID' if use_explicit_padding else 'SAME'
 541     stride = 1
 542     kernel_size = 3
 543     def clip_by_value(features):
 544       return tf.clip_by_value(features, -ACTIVATION_BOUND, ACTIVATION_BOUND)
 545 
 546     # top layers
 547     self.top_layers.append(tf.keras.layers.Conv2D(
 548         depth, [1, 1], strides=stride, padding=padding,
 549         name='projection_%d' % num_levels,
 550         **conv_hyperparams.params(use_bias=True)))
 551     if use_bounded_activations:
 552       self.top_layers.append(tf.keras.layers.Lambda(
 553           clip_by_value, name='clip_by_value'))
 554 
 555     for level in reversed(list(range(num_levels - 1))):
 556       # to generate residual from image features
 557       residual_net = []
 558       # to preprocess top_down (the image feature map from last layer)
 559       top_down_net = []
 560       # to reshape top_down according to residual if necessary
 561       reshaped_residual = []
 562       # to apply convolution layers to feature map
 563       conv_net = []
 564 
 565       # residual block
 566       residual_net.append(tf.keras.layers.Conv2D(
 567           depth, [1, 1], padding=padding, strides=1,
 568           name='projection_%d' % (level + 1),
 569           **conv_hyperparams.params(use_bias=True)))
 570       if use_bounded_activations:
 571         residual_net.append(tf.keras.layers.Lambda(
 572             clip_by_value, name='clip_by_value'))
 573 
 574       # top-down block
 575       # TODO (b/128922690): clean-up of ops.nearest_neighbor_upsampling
 576       if use_native_resize_op:
 577         def resize_nearest_neighbor(image):
 578           image_shape = shape_utils.combined_static_and_dynamic_shape(image)
 579           return tf.image.resize_nearest_neighbor(
 580               image, [image_shape[1] * 2, image_shape[2] * 2])
 581         top_down_net.append(tf.keras.layers.Lambda(
 582             resize_nearest_neighbor, name='nearest_neighbor_upsampling'))
 583       else:
 584         def nearest_neighbor_upsampling(image):
 585           return ops.nearest_neighbor_upsampling(image, scale=2)
 586         top_down_net.append(tf.keras.layers.Lambda(
 587             nearest_neighbor_upsampling, name='nearest_neighbor_upsampling'))
 588 
 589       # reshape block
 590       if use_explicit_padding:
 591         def reshape(inputs):
 592           residual_shape = tf.shape(inputs[0])
 593           return inputs[1][:, :residual_shape[1], :residual_shape[2], :]
 594         reshaped_residual.append(
 595             tf.keras.layers.Lambda(reshape, name='reshape'))
 596 
 597       # down layers
 598       if use_bounded_activations:
 599         conv_net.append(tf.keras.layers.Lambda(
 600             clip_by_value, name='clip_by_value'))
 601 
 602       if use_explicit_padding:
 603         def fixed_padding(features, kernel_size=kernel_size):
 604           return ops.fixed_padding(features, kernel_size)
 605         conv_net.append(tf.keras.layers.Lambda(
 606             fixed_padding, name='fixed_padding'))
 607 
 608       layer_name = 'smoothing_%d' % (level + 1)
 609       conv_block = create_conv_block(
 610           use_depthwise, kernel_size, padding, stride, layer_name,
 611           conv_hyperparams, is_training, freeze_batchnorm, depth)
 612       conv_net.extend(conv_block)
 613 
 614       self.residual_blocks.append(residual_net)
 615       self.top_down_blocks.append(top_down_net)
 616       self.reshape_blocks.append(reshaped_residual)
 617       self.conv_layers.append(conv_net)
 618 
 619   def call(self, image_features):
 620     """Generate the multi-resolution feature maps.
 621 
 622     Executed when calling the `.__call__` method on input.
 623 
 624     Args:
 625       image_features: list of tuples of (tensor_name, image_feature_tensor).
 626         Spatial resolutions of succesive tensors must reduce exactly by a factor
 627         of 2.
 628 
 629     Returns:
 630       feature_maps: an OrderedDict mapping keys (feature map names) to
 631         tensors where each tensor has shape [batch, height_i, width_i, depth_i].
 632     """
 633     output_feature_maps_list = []
 634     output_feature_map_keys = []
 635 
 636     with tf.name_scope(self.scope):
 637       top_down = image_features[-1][1]
 638       for layer in self.top_layers:
 639         top_down = layer(top_down)
 640       output_feature_maps_list.append(top_down)
 641       output_feature_map_keys.append('top_down_%s' % image_features[-1][0])
 642 
 643       num_levels = len(image_features)
 644       for index, level in enumerate(reversed(list(range(num_levels - 1)))):
 645         residual = image_features[level][1]
 646         top_down = output_feature_maps_list[-1]
 647         for layer in self.residual_blocks[index]:
 648           residual = layer(residual)
 649         for layer in self.top_down_blocks[index]:
 650           top_down = layer(top_down)
 651         for layer in self.reshape_blocks[index]:
 652           top_down = layer([residual, top_down])
 653         top_down += residual
 654         for layer in self.conv_layers[index]:
 655           top_down = layer(top_down)
 656         output_feature_maps_list.append(top_down)
 657         output_feature_map_keys.append('top_down_%s' % image_features[level][0])
 658     return collections.OrderedDict(reversed(
 659         list(zip(output_feature_map_keys, output_feature_maps_list))))

 662 def fpn_top_down_feature_maps(image_features,
 663                               depth,
 664                               use_depthwise=False,
 665                               use_explicit_padding=False,
 666                               use_bounded_activations=False,
 667                               scope=None,
 668                               use_native_resize_op=False):
 669   """Generates `top-down` feature maps for Feature Pyramid Networks.
 670 
 671   See https://arxiv.org/abs/1612.03144 for details.
 672 
 673   Args:
 674     image_features: list of tuples of (tensor_name, image_feature_tensor).
 675       Spatial resolutions of succesive tensors must reduce exactly by a factor
 676       of 2.
 677     depth: depth of output feature maps.
 678     use_depthwise: whether to use depthwise separable conv instead of regular
 679       conv.
 680     use_explicit_padding: whether to use explicit padding.
 681     use_bounded_activations: Whether or not to clip activations to range
 682       [-ACTIVATION_BOUND, ACTIVATION_BOUND]. Bounded activations better lend
 683       themselves to quantized inference.
 684     scope: A scope name to wrap this op under.
 685     use_native_resize_op: If True, uses tf.image.resize_nearest_neighbor op for
 686       the upsampling process instead of reshape and broadcasting implementation.
 687 
 688   Returns:
 689     feature_maps: an OrderedDict mapping keys (feature map names) to
 690       tensors where each tensor has shape [batch, height_i, width_i, depth_i].
 691   """
 692   with tf.name_scope(scope, 'top_down'):
 693     num_levels = len(image_features)
 694     output_feature_maps_list = []
 695     output_feature_map_keys = []
 696     padding = 'VALID' if use_explicit_padding else 'SAME'
 697     kernel_size = 3
 698     with slim.arg_scope(
 699         [slim.conv2d, slim.separable_conv2d], padding=padding, stride=1):
 700       top_down = slim.conv2d(
 701           image_features[-1][1],
 702           depth, [1, 1], activation_fn=None, normalizer_fn=None,
 703           scope='projection_%d' % num_levels)
 704       if use_bounded_activations:
 705         top_down = tf.clip_by_value(top_down, -ACTIVATION_BOUND,
 706                                     ACTIVATION_BOUND)
 707       output_feature_maps_list.append(top_down)
 708       output_feature_map_keys.append(
 709           'top_down_%s' % image_features[-1][0])
 710 
 711       for level in reversed(list(range(num_levels - 1))):
 712         if use_native_resize_op:
 713           with tf.name_scope('nearest_neighbor_upsampling'):
 714             top_down_shape = shape_utils.combined_static_and_dynamic_shape(
 715                 top_down)
 716             top_down = tf.image.resize_nearest_neighbor(
 717                 top_down, [top_down_shape[1] * 2, top_down_shape[2] * 2])
 718         else:
 719           top_down = ops.nearest_neighbor_upsampling(top_down, scale=2)
 720         residual = slim.conv2d(
 721             image_features[level][1], depth, [1, 1],
 722             activation_fn=None, normalizer_fn=None,
 723             scope='projection_%d' % (level + 1))
 724         if use_bounded_activations:
 725           residual = tf.clip_by_value(residual, -ACTIVATION_BOUND,
 726                                       ACTIVATION_BOUND)
 727         if use_explicit_padding:
 728           # slice top_down to the same shape as residual
 729           residual_shape = tf.shape(residual)
 730           top_down = top_down[:, :residual_shape[1], :residual_shape[2], :]
 731         top_down += residual
 732         if use_bounded_activations:
 733           top_down = tf.clip_by_value(top_down, -ACTIVATION_BOUND,
 734                                       ACTIVATION_BOUND)
 735         if use_depthwise:
 736           conv_op = functools.partial(slim.separable_conv2d, depth_multiplier=1)
 737         else:
 738           conv_op = slim.conv2d
 739         pre_output = top_down
 740         if use_explicit_padding:
 741           pre_output = ops.fixed_padding(pre_output, kernel_size)
 742         output_feature_maps_list.append(conv_op(
 743             pre_output,
 744             depth, [kernel_size, kernel_size],
 745             scope='smoothing_%d' % (level + 1)))
 746         output_feature_map_keys.append('top_down_%s' % image_features[level][0])
 747       return collections.OrderedDict(reversed(
 748           list(zip(output_feature_map_keys, output_feature_maps_list))))

 751 def pooling_pyramid_feature_maps(base_feature_map_depth, num_layers,
 752                                  image_features, replace_pool_with_conv=False):
 753   """Generates pooling pyramid feature maps.
 754 
 755   The pooling pyramid feature maps is motivated by
 756   multi_resolution_feature_maps. The main difference are that it is simpler and
 757   reduces the number of free parameters.
 758 
 759   More specifically:
 760    - Instead of using convolutions to shrink the feature map, it uses max
 761      pooling, therefore totally gets rid of the parameters in convolution.
 762    - By pooling feature from larger map up to a single cell, it generates
 763      features in the same feature space.
 764    - Instead of independently making box predictions from individual maps, it
 765      shares the same classifier across different feature maps, therefore reduces
 766      the "mis-calibration" across different scales.
 767 
 768   See go/ppn-detection for more details.
 769 
 770   Args:
 771     base_feature_map_depth: Depth of the base feature before the max pooling.
 772     num_layers: Number of layers used to make predictions. They are pooled
 773       from the base feature.
 774     image_features: A dictionary of handles to activation tensors from the
 775       feature extractor.
 776     replace_pool_with_conv: Whether or not to replace pooling operations with
 777       convolutions in the PPN. Default is False.
 778 
 779   Returns:
 780     feature_maps: an OrderedDict mapping keys (feature map names) to
 781       tensors where each tensor has shape [batch, height_i, width_i, depth_i].
 782   Raises:
 783     ValueError: image_features does not contain exactly one entry
 784   """
 785   if len(image_features) != 1:
 786     raise ValueError('image_features should be a dictionary of length 1.')
 787   image_features = image_features[list(image_features.keys())[0]]
 788 
 789   feature_map_keys = []
 790   feature_maps = []
 791   feature_map_key = 'Base_Conv2d_1x1_%d' % base_feature_map_depth
 792   if base_feature_map_depth > 0:
 793     image_features = slim.conv2d(
 794         image_features,
 795         base_feature_map_depth,
 796         [1, 1],  # kernel size
 797         padding='SAME', stride=1, scope=feature_map_key)
 798     # Add a 1x1 max-pooling node (a no op node) immediately after the conv2d for
 799     # TPU v1 compatibility.  Without the following dummy op, TPU runtime
 800     # compiler will combine the convolution with one max-pooling below into a
 801     # single cycle, so getting the conv2d feature becomes impossible.
 802     image_features = slim.max_pool2d(
 803         image_features, [1, 1], padding='SAME', stride=1, scope=feature_map_key)
 804   feature_map_keys.append(feature_map_key)
 805   feature_maps.append(image_features)
 806   feature_map = image_features
 807   if replace_pool_with_conv:
 808     with slim.arg_scope([slim.conv2d], padding='SAME', stride=2):
 809       for i in range(num_layers - 1):
 810         feature_map_key = 'Conv2d_{}_3x3_s2_{}'.format(i,
 811                                                        base_feature_map_depth)
 812         feature_map = slim.conv2d(
 813             feature_map, base_feature_map_depth, [3, 3], scope=feature_map_key)
 814         feature_map_keys.append(feature_map_key)
 815         feature_maps.append(feature_map)
 816   else:
 817     with slim.arg_scope([slim.max_pool2d], padding='SAME', stride=2):
 818       for i in range(num_layers - 1):
 819         feature_map_key = 'MaxPool2d_%d_2x2' % i
 820         feature_map = slim.max_pool2d(
 821             feature_map, [2, 2], padding='SAME', scope=feature_map_key)
 822         feature_map_keys.append(feature_map_key)
 823         feature_maps.append(feature_map)
 824   return collections.OrderedDict(
 825       [(x, y) for (x, y) in zip(feature_map_keys, feature_maps)])
