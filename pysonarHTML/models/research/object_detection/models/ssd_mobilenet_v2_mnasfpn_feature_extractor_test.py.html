  16 """Tests for ssd_mobilenet_v2_nas_fpn_feature_extractor."""
  17 import numpy as np
  18 import tensorflow.compat.v1 as tf
  19 
  20 from object_detection.models import ssd_feature_extractor_test
  21 from object_detection.models import ssd_mobilenet_v2_mnasfpn_feature_extractor as mnasfpn_feature_extractor
  22 
  23 
  24 class SsdMobilenetV2MnasFPNFeatureExtractorTest(
  25     ssd_feature_extractor_test.SsdFeatureExtractorTestBase):
  26 
  27   def _create_feature_extractor(self,
  28                                 depth_multiplier,
  29                                 pad_to_multiple,
  30                                 use_explicit_padding=False):
  31     min_depth = 16
  32     is_training = True
  33     fpn_num_filters = 48
  34     return mnasfpn_feature_extractor.SSDMobileNetV2MnasFPNFeatureExtractor(
  35         is_training,
  36         depth_multiplier,
  37         min_depth,
  38         pad_to_multiple,
  39         self.conv_hyperparams_fn,
  40         additional_layer_depth=fpn_num_filters,
  41         use_explicit_padding=use_explicit_padding)
  42 
  43   def test_extract_features_returns_correct_shapes_320_256(self):
  44     image_height = 320
  45     image_width = 256
  46     depth_multiplier = 1.0
  47     pad_to_multiple = 1
  48     expected_feature_map_shape = [(2, 40, 32, 48), (2, 20, 16, 48),
  49                                   (2, 10, 8, 48), (2, 5, 4, 48)]
  50     self.check_extract_features_returns_correct_shape(
  51         2, image_height, image_width, depth_multiplier, pad_to_multiple,
  52         expected_feature_map_shape, use_explicit_padding=False)
  53     self.check_extract_features_returns_correct_shape(
  54         2, image_height, image_width, depth_multiplier, pad_to_multiple,
  55         expected_feature_map_shape, use_explicit_padding=True)
  56 
  57   def test_extract_features_returns_correct_shapes_enforcing_min_depth(self):
  58     image_height = 256
  59     image_width = 256
  60     depth_multiplier = 0.5**12
  61     pad_to_multiple = 1
  62     expected_feature_map_shape = [(2, 32, 32, 16), (2, 16, 16, 16),
  63                                   (2, 8, 8, 16), (2, 4, 4, 16)]
  64     self.check_extract_features_returns_correct_shape(
  65         2, image_height, image_width, depth_multiplier, pad_to_multiple,
  66         expected_feature_map_shape, use_explicit_padding=False)
  67     self.check_extract_features_returns_correct_shape(
  68         2, image_height, image_width, depth_multiplier, pad_to_multiple,
  69         expected_feature_map_shape, use_explicit_padding=True)
  70 
  71   def test_preprocess_returns_correct_value_range(self):
  72     image_height = 320
  73     image_width = 320
  74     depth_multiplier = 1
  75     pad_to_multiple = 1
  76     test_image = np.random.rand(2, image_height, image_width, 3)
  77     feature_extractor = self._create_feature_extractor(depth_multiplier,
  78                                                        pad_to_multiple)
  79     preprocessed_image = feature_extractor.preprocess(test_image)
  80     self.assertTrue(np.all(np.less_equal(np.abs(preprocessed_image), 1.0)))
