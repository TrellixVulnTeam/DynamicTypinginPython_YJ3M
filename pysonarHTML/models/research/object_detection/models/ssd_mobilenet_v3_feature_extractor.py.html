  29 class SSDMobileNetV3FeatureExtractorBase(ssd_meta_arch.SSDFeatureExtractor):
  30   """Base class of SSD feature extractor using MobilenetV3 features."""
  31 
  32   def __init__(self,
  33                conv_defs,
  34                from_layer,
  35                is_training,
  36                depth_multiplier,
  37                min_depth,
  38                pad_to_multiple,
  39                conv_hyperparams_fn,
  40                reuse_weights=None,
  41                use_explicit_padding=False,
  42                use_depthwise=False,
  43                override_base_feature_extractor_hyperparams=False,
  44                scope_name='MobilenetV3'):
  45     """MobileNetV3 Feature Extractor for SSD Models.
  46 
  47     MobileNet v3. Details found in:
  48     https://arxiv.org/abs/1905.02244
  49 
  50     Args:
  51       conv_defs: MobileNetV3 conv defs for backbone.
  52       from_layer: A cell of two layer names (string) to connect to the 1st and
  53         2nd inputs of the SSD head.
  54       is_training: whether the network is in training mode.
  55       depth_multiplier: float depth multiplier for feature extractor.
  56       min_depth: minimum feature extractor depth.
  57       pad_to_multiple: the nearest multiple to zero pad the input height and
  58         width dimensions to.
  59       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  60         and separable_conv2d ops in the layers that are added on top of the base
  61         feature extractor.
  62       reuse_weights: Whether to reuse variables. Default is None.
  63       use_explicit_padding: Whether to use explicit padding when extracting
  64         features. Default is False.
  65       use_depthwise: Whether to use depthwise convolutions. Default is False.
  66       override_base_feature_extractor_hyperparams: Whether to override
  67         hyperparameters of the base feature extractor with the one from
  68         `conv_hyperparams_fn`.
  69       scope_name: scope name (string) of network variables.
  70     """
  71     super(SSDMobileNetV3FeatureExtractorBase, self).__init__(
  72         is_training=is_training,
  73         depth_multiplier=depth_multiplier,
  74         min_depth=min_depth,
  75         pad_to_multiple=pad_to_multiple,
  76         conv_hyperparams_fn=conv_hyperparams_fn,
  77         reuse_weights=reuse_weights,
  78         use_explicit_padding=use_explicit_padding,
  79         use_depthwise=use_depthwise,
  80         override_base_feature_extractor_hyperparams=override_base_feature_extractor_hyperparams
  81     )
  82     self._conv_defs = conv_defs
  83     self._from_layer = from_layer
  84     self._scope_name = scope_name
  85 
  86   def preprocess(self, resized_inputs):
  87     """SSD preprocessing.
  88 
  89     Maps pixel values to the range [-1, 1].
  90 
  91     Args:
  92       resized_inputs: a [batch, height, width, channels] float tensor
  93         representing a batch of images.
  94 
  95     Returns:
  96       preprocessed_inputs: a [batch, height, width, channels] float tensor
  97         representing a batch of images.
  98     """
  99     return (2.0 / 255.0) * resized_inputs - 1.0
 100 
 101   def extract_features(self, preprocessed_inputs):
 102     """Extract features from preprocessed inputs.
 103 
 104     Args:
 105       preprocessed_inputs: a [batch, height, width, channels] float tensor
 106         representing a batch of images.
 107 
 108     Returns:
 109       feature_maps: a list of tensors where the ith tensor has shape
 110         [batch, height_i, width_i, depth_i]
 111     Raises:
 112       ValueError if conv_defs is not provided or from_layer does not meet the
 113         size requirement.
 114     """
 115 
 116     if not self._conv_defs:
 117       raise ValueError('Must provide backbone conv defs.')
 118 
 119     if len(self._from_layer) != 2:
 120       raise ValueError('SSD input feature names are not provided.')
 121 
 122     preprocessed_inputs = shape_utils.check_min_image_dim(
 123         33, preprocessed_inputs)
 124 
 125     feature_map_layout = {
 126         'from_layer': [
 127             self._from_layer[0], self._from_layer[1], '', '', '', ''
 128         ],
 129         'layer_depth': [-1, -1, 512, 256, 256, 128],
 130         'use_depthwise': self._use_depthwise,
 131         'use_explicit_padding': self._use_explicit_padding,
 132     }
 133 
 134     with tf.variable_scope(
 135         self._scope_name, reuse=self._reuse_weights) as scope:
 136       with slim.arg_scope(
 137           mobilenet_v3.training_scope(is_training=None, bn_decay=0.9997)), \
 138           slim.arg_scope(
 139               [mobilenet.depth_multiplier], min_depth=self._min_depth):
 140         with (slim.arg_scope(self._conv_hyperparams_fn())
 141               if self._override_base_feature_extractor_hyperparams else
 142               context_manager.IdentityContextManager()):
 143           _, image_features = mobilenet_v3.mobilenet_base(
 144               ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple),
 145               conv_defs=self._conv_defs,
 146               final_endpoint=self._from_layer[1],
 147               depth_multiplier=self._depth_multiplier,
 148               use_explicit_padding=self._use_explicit_padding,
 149               scope=scope)
 150         with slim.arg_scope(self._conv_hyperparams_fn()):
 151           feature_maps = feature_map_generators.multi_resolution_feature_maps(
 152               feature_map_layout=feature_map_layout,
 153               depth_multiplier=self._depth_multiplier,
 154               min_depth=self._min_depth,
 155               insert_1x1_conv=True,
 156               image_features=image_features)
 157 
 158     return list(feature_maps.values())

 161 class SSDMobileNetV3LargeFeatureExtractor(SSDMobileNetV3FeatureExtractorBase):
 162   """Mobilenet V3-Large feature extractor."""
 163 
 164   def __init__(self,
 165                is_training,
 166                depth_multiplier,
 167                min_depth,
 168                pad_to_multiple,
 169                conv_hyperparams_fn,
 170                reuse_weights=None,
 171                use_explicit_padding=False,
 172                use_depthwise=False,
 173                override_base_feature_extractor_hyperparams=False,
 174                scope_name='MobilenetV3'):
 175     super(SSDMobileNetV3LargeFeatureExtractor, self).__init__(
 176         conv_defs=mobilenet_v3.V3_LARGE_DETECTION,
 177         from_layer=['layer_14/expansion_output', 'layer_17'],
 178         is_training=is_training,
 179         depth_multiplier=depth_multiplier,
 180         min_depth=min_depth,
 181         pad_to_multiple=pad_to_multiple,
 182         conv_hyperparams_fn=conv_hyperparams_fn,
 183         reuse_weights=reuse_weights,
 184         use_explicit_padding=use_explicit_padding,
 185         use_depthwise=use_depthwise,
 186         override_base_feature_extractor_hyperparams=override_base_feature_extractor_hyperparams,
 187         scope_name=scope_name
