  28 def _batch_norm_arg_scope(list_ops,
  29                           use_batch_norm=True,
  30                           batch_norm_decay=0.9997,
  31                           batch_norm_epsilon=0.001,
  32                           batch_norm_scale=False,
  33                           train_batch_norm=False):
  34   """Slim arg scope for InceptionV2 batch norm."""
  35   if use_batch_norm:
  36     batch_norm_params = {
  37         'is_training': train_batch_norm,
  38         'scale': batch_norm_scale,
  39         'decay': batch_norm_decay,
  40         'epsilon': batch_norm_epsilon
  41     }
  42     normalizer_fn = slim.batch_norm
  43   else:
  44     normalizer_fn = None
  45     batch_norm_params = None
  46 
  47   return slim.arg_scope(list_ops,
  48                         normalizer_fn=normalizer_fn,
  49                         normalizer_params=batch_norm_params)

  52 class FasterRCNNInceptionV2FeatureExtractor(
  53     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor):
  54   """Faster R-CNN Inception V2 feature extractor implementation."""
  55 
  56   def __init__(self,
  57                is_training,
  58                first_stage_features_stride,
  59                batch_norm_trainable=False,
  60                reuse_weights=None,
  61                weight_decay=0.0,
  62                depth_multiplier=1.0,
  63                min_depth=16):
  64     """Constructor.
  65 
  66     Args:
  67       is_training: See base class.
  68       first_stage_features_stride: See base class.
  69       batch_norm_trainable: See base class.
  70       reuse_weights: See base class.
  71       weight_decay: See base class.
  72       depth_multiplier: float depth multiplier for feature extractor.
  73       min_depth: minimum feature extractor depth.
  74 
  75     Raises:
  76       ValueError: If `first_stage_features_stride` is not 8 or 16.
  77     """
  78     if first_stage_features_stride != 8 and first_stage_features_stride != 16:
  79       raise ValueError('`first_stage_features_stride` must be 8 or 16.')
  80     self._depth_multiplier = depth_multiplier
  81     self._min_depth = min_depth
  82     super(FasterRCNNInceptionV2FeatureExtractor, self).__init__(
  83         is_training, first_stage_features_stride, batch_norm_trainable,
  84         reuse_weights, weight_decay)
  85 
  86   def preprocess(self, resized_inputs):
  87     """Faster R-CNN Inception V2 preprocessing.
  88 
  89     Maps pixel values to the range [-1, 1].
  90 
  91     Args:
  92       resized_inputs: a [batch, height, width, channels] float tensor
  93         representing a batch of images.
  94 
  95     Returns:
  96       preprocessed_inputs: a [batch, height, width, channels] float tensor
  97         representing a batch of images.
  98     """
  99     return (2.0 / 255.0) * resized_inputs - 1.0

 101   def _extract_proposal_features(self, preprocessed_inputs, scope):
 102     """Extracts first stage RPN features.
 103 
 104     Args:
 105       preprocessed_inputs: A [batch, height, width, channels] float32 tensor
 106         representing a batch of images.
 107       scope: A scope name.
 108 
 109     Returns:
 110       rpn_feature_map: A tensor with shape [batch, height, width, depth]
 111       activations: A dictionary mapping feature extractor tensor names to
 112         tensors
 113 
 114     Raises:
 115       InvalidArgumentError: If the spatial size of `preprocessed_inputs`
 116         (height or width) is less than 33.
 117       ValueError: If the created network is missing the required activation.
 118     """
 119 
 120     preprocessed_inputs.get_shape().assert_has_rank(4)
 121     shape_assert = tf.Assert(
 122         tf.logical_and(tf.greater_equal(tf.shape(preprocessed_inputs)[1], 33),
 123                        tf.greater_equal(tf.shape(preprocessed_inputs)[2], 33)),
 124         ['image size must at least be 33 in both height and width.'])
 125 
 126     with tf.control_dependencies([shape_assert]):
 127       with tf.variable_scope('InceptionV2',
 128                              reuse=self._reuse_weights) as scope:
 129         with _batch_norm_arg_scope([slim.conv2d, slim.separable_conv2d],
 130                                    batch_norm_scale=True,
 131                                    train_batch_norm=self._train_batch_norm):
 132           _, activations = inception_v2.inception_v2_base(
 133               preprocessed_inputs,
 134               final_endpoint='Mixed_4e',
 135               min_depth=self._min_depth,
 136               depth_multiplier=self._depth_multiplier,
 137               scope=scope)
 138 
 139     return activations['Mixed_4e'], activations

 141   def _extract_box_classifier_features(self, proposal_feature_maps, scope):
 142     """Extracts second stage box classifier features.
 143 
 144     Args:
 145       proposal_feature_maps: A 4-D float tensor with shape
 146         [batch_size * self.max_num_proposals, crop_height, crop_width, depth]
 147         representing the feature map cropped to each proposal.
 148       scope: A scope name (unused).
 149 
 150     Returns:
 151       proposal_classifier_features: A 4-D float tensor with shape
 152         [batch_size * self.max_num_proposals, height, width, depth]
 153         representing box classifier features for each proposal.
 154     """
 155     net = proposal_feature_maps
 156 
 157     depth = lambda d: max(int(d * self._depth_multiplier), self._min_depth)
 158     trunc_normal = lambda stddev: tf.truncated_normal_initializer(0.0, stddev)
 159 
 160     data_format = 'NHWC'
 161     concat_dim = 3 if data_format == 'NHWC' else 1
 162 
 163     with tf.variable_scope('InceptionV2', reuse=self._reuse_weights):
 164       with slim.arg_scope(
 165           [slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 166           stride=1,
 167           padding='SAME',
 168           data_format=data_format):
 169         with _batch_norm_arg_scope([slim.conv2d, slim.separable_conv2d],
 170                                    batch_norm_scale=True,
 171                                    train_batch_norm=self._train_batch_norm):
 172 
 173           with tf.variable_scope('Mixed_5a'):
 174             with tf.variable_scope('Branch_0'):
 175               branch_0 = slim.conv2d(
 176                   net, depth(128), [1, 1],
 177                   weights_initializer=trunc_normal(0.09),
 178                   scope='Conv2d_0a_1x1')
 179               branch_0 = slim.conv2d(branch_0, depth(192), [3, 3], stride=2,
 180                                      scope='Conv2d_1a_3x3')
 181             with tf.variable_scope('Branch_1'):
 182               branch_1 = slim.conv2d(
 183                   net, depth(192), [1, 1],
 184                   weights_initializer=trunc_normal(0.09),
 185                   scope='Conv2d_0a_1x1')
 186               branch_1 = slim.conv2d(branch_1, depth(256), [3, 3],
 187                                      scope='Conv2d_0b_3x3')
 188               branch_1 = slim.conv2d(branch_1, depth(256), [3, 3], stride=2,
 189                                      scope='Conv2d_1a_3x3')
 190             with tf.variable_scope('Branch_2'):
 191               branch_2 = slim.max_pool2d(net, [3, 3], stride=2,
 192                                          scope='MaxPool_1a_3x3')
 193             net = tf.concat([branch_0, branch_1, branch_2], concat_dim)
 194 
 195           with tf.variable_scope('Mixed_5b'):
 196             with tf.variable_scope('Branch_0'):
 197               branch_0 = slim.conv2d(net, depth(352), [1, 1],
 198                                      scope='Conv2d_0a_1x1')
 199             with tf.variable_scope('Branch_1'):
 200               branch_1 = slim.conv2d(
 201                   net, depth(192), [1, 1],
 202                   weights_initializer=trunc_normal(0.09),
 203                   scope='Conv2d_0a_1x1')
 204               branch_1 = slim.conv2d(branch_1, depth(320), [3, 3],
 205                                      scope='Conv2d_0b_3x3')
 206             with tf.variable_scope('Branch_2'):
 207               branch_2 = slim.conv2d(
 208                   net, depth(160), [1, 1],
 209                   weights_initializer=trunc_normal(0.09),
 210                   scope='Conv2d_0a_1x1')
 211               branch_2 = slim.conv2d(branch_2, depth(224), [3, 3],
 212                                      scope='Conv2d_0b_3x3')
 213               branch_2 = slim.conv2d(branch_2, depth(224), [3, 3],
 214                                      scope='Conv2d_0c_3x3')
 215             with tf.variable_scope('Branch_3'):
 216               branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 217               branch_3 = slim.conv2d(
 218                   branch_3, depth(128), [1, 1],
 219                   weights_initializer=trunc_normal(0.1),
 220                   scope='Conv2d_0b_1x1')
 221             net = tf.concat([branch_0, branch_1, branch_2, branch_3],
 222                             concat_dim)
 223 
 224           with tf.variable_scope('Mixed_5c'):
 225             with tf.variable_scope('Branch_0'):
 226               branch_0 = slim.conv2d(net, depth(352), [1, 1],
 227                                      scope='Conv2d_0a_1x1')
 228             with tf.variable_scope('Branch_1'):
 229               branch_1 = slim.conv2d(
 230                   net, depth(192), [1, 1],
 231                   weights_initializer=trunc_normal(0.09),
 232                   scope='Conv2d_0a_1x1')
 233               branch_1 = slim.conv2d(branch_1, depth(320), [3, 3],
 234                                      scope='Conv2d_0b_3x3')
 235             with tf.variable_scope('Branch_2'):
 236               branch_2 = slim.conv2d(
 237                   net, depth(192), [1, 1],
 238                   weights_initializer=trunc_normal(0.09),
 239                   scope='Conv2d_0a_1x1')
 240               branch_2 = slim.conv2d(branch_2, depth(224), [3, 3],
 241                                      scope='Conv2d_0b_3x3')
 242               branch_2 = slim.conv2d(branch_2, depth(224), [3, 3],
 243                                      scope='Conv2d_0c_3x3')
 244             with tf.variable_scope('Branch_3'):
 245               branch_3 = slim.max_pool2d(net, [3, 3], scope='MaxPool_0a_3x3')
 246               branch_3 = slim.conv2d(
 247                   branch_3, depth(128), [1, 1],
 248                   weights_initializer=trunc_normal(0.1),
 249                   scope='Conv2d_0b_1x1')
 250             proposal_classifier_features = tf.concat(
 251                 [branch_0, branch_1, branch_2, branch_3], concat_dim)
 252 
 253     return proposal_classifier_features
