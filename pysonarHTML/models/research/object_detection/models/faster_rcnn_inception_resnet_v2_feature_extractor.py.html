  33 class FasterRCNNInceptionResnetV2FeatureExtractor(
  34     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor):
  35   """Faster R-CNN with Inception Resnet v2 feature extractor implementation."""
  36 
  37   def __init__(self,
  38                is_training,
  39                first_stage_features_stride,
  40                batch_norm_trainable=False,
  41                reuse_weights=None,
  42                weight_decay=0.0):
  43     """Constructor.
  44 
  45     Args:
  46       is_training: See base class.
  47       first_stage_features_stride: See base class.
  48       batch_norm_trainable: See base class.
  49       reuse_weights: See base class.
  50       weight_decay: See base class.
  51 
  52     Raises:
  53       ValueError: If `first_stage_features_stride` is not 8 or 16.
  54     """
  55     if first_stage_features_stride != 8 and first_stage_features_stride != 16:
  56       raise ValueError('`first_stage_features_stride` must be 8 or 16.')
  57     super(FasterRCNNInceptionResnetV2FeatureExtractor, self).__init__(
  58         is_training, first_stage_features_stride, batch_norm_trainable,
  59         reuse_weights, weight_decay)
  60 
  61   def preprocess(self, resized_inputs):
  62     """Faster R-CNN with Inception Resnet v2 preprocessing.
  63 
  64     Maps pixel values to the range [-1, 1].
  65 
  66     Args:
  67       resized_inputs: A [batch, height_in, width_in, channels] float32 tensor
  68         representing a batch of images with values between 0 and 255.0.
  69 
  70     Returns:
  71       preprocessed_inputs: A [batch, height_out, width_out, channels] float32
  72         tensor representing a batch of images.
  73 
  74     """
  75     return (2.0 / 255.0) * resized_inputs - 1.0
  76 
  77   def _extract_proposal_features(self, preprocessed_inputs, scope):
  78     """Extracts first stage RPN features.
  79 
  80     Extracts features using the first half of the Inception Resnet v2 network.
  81     We construct the network in `align_feature_maps=True` mode, which means
  82     that all VALID paddings in the network are changed to SAME padding so that
  83     the feature maps are aligned.
  84 
  85     Args:
  86       preprocessed_inputs: A [batch, height, width, channels] float32 tensor
  87         representing a batch of images.
  88       scope: A scope name.
  89 
  90     Returns:
  91       rpn_feature_map: A tensor with shape [batch, height, width, depth]
  92     Raises:
  93       InvalidArgumentError: If the spatial size of `preprocessed_inputs`
  94         (height or width) is less than 33.
  95       ValueError: If the created network is missing the required activation.
  96     """
  97     if len(preprocessed_inputs.get_shape().as_list()) != 4:
  98       raise ValueError('`preprocessed_inputs` must be 4 dimensional, got a '
  99                        'tensor of shape %s' % preprocessed_inputs.get_shape())
 100 
 101     with slim.arg_scope(inception_resnet_v2.inception_resnet_v2_arg_scope(
 102         weight_decay=self._weight_decay)):
 103       # Forces is_training to False to disable batch norm update.
 104       with slim.arg_scope([slim.batch_norm],
 105                           is_training=self._train_batch_norm):
 106         with tf.variable_scope('InceptionResnetV2',
 107                                reuse=self._reuse_weights) as scope:
 108           return inception_resnet_v2.inception_resnet_v2_base(
 109               preprocessed_inputs, final_endpoint='PreAuxLogits',
 110               scope=scope, output_stride=self._first_stage_features_stride,
 111               align_feature_maps=True)
 112 
 113   def _extract_box_classifier_features(self, proposal_feature_maps, scope):
 114     """Extracts second stage box classifier features.
 115 
 116     This function reconstructs the "second half" of the Inception ResNet v2
 117     network after the part defined in `_extract_proposal_features`.
 118 
 119     Args:
 120       proposal_feature_maps: A 4-D float tensor with shape
 121         [batch_size * self.max_num_proposals, crop_height, crop_width, depth]
 122         representing the feature map cropped to each proposal.
 123       scope: A scope name.
 124 
 125     Returns:
 126       proposal_classifier_features: A 4-D float tensor with shape
 127         [batch_size * self.max_num_proposals, height, width, depth]
 128         representing box classifier features for each proposal.
 129     """
 130     with tf.variable_scope('InceptionResnetV2', reuse=self._reuse_weights):
 131       with slim.arg_scope(inception_resnet_v2.inception_resnet_v2_arg_scope(
 132           weight_decay=self._weight_decay)):
 133         # Forces is_training to False to disable batch norm update.
 134         with slim.arg_scope([slim.batch_norm],
 135                             is_training=self._train_batch_norm):
 136           with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 137                               stride=1, padding='SAME'):
 138             with tf.variable_scope('Mixed_7a'):
 139               with tf.variable_scope('Branch_0'):
 140                 tower_conv = slim.conv2d(proposal_feature_maps,
 141                                          256, 1, scope='Conv2d_0a_1x1')
 142                 tower_conv_1 = slim.conv2d(
 143                     tower_conv, 384, 3, stride=2,
 144                     padding='VALID', scope='Conv2d_1a_3x3')
 145               with tf.variable_scope('Branch_1'):
 146                 tower_conv1 = slim.conv2d(
 147                     proposal_feature_maps, 256, 1, scope='Conv2d_0a_1x1')
 148                 tower_conv1_1 = slim.conv2d(
 149                     tower_conv1, 288, 3, stride=2,
 150                     padding='VALID', scope='Conv2d_1a_3x3')
 151               with tf.variable_scope('Branch_2'):
 152                 tower_conv2 = slim.conv2d(
 153                     proposal_feature_maps, 256, 1, scope='Conv2d_0a_1x1')
 154                 tower_conv2_1 = slim.conv2d(tower_conv2, 288, 3,
 155                                             scope='Conv2d_0b_3x3')
 156                 tower_conv2_2 = slim.conv2d(
 157                     tower_conv2_1, 320, 3, stride=2,
 158                     padding='VALID', scope='Conv2d_1a_3x3')
 159               with tf.variable_scope('Branch_3'):
 160                 tower_pool = slim.max_pool2d(
 161                     proposal_feature_maps, 3, stride=2, padding='VALID',
 162                     scope='MaxPool_1a_3x3')
 163               net = tf.concat(
 164                   [tower_conv_1, tower_conv1_1, tower_conv2_2, tower_pool], 3)
 165             net = slim.repeat(net, 9, inception_resnet_v2.block8, scale=0.20)
 166             net = inception_resnet_v2.block8(net, activation_fn=None)
 167             proposal_classifier_features = slim.conv2d(
 168                 net, 1536, 1, scope='Conv2d_7b_1x1')
 169         return proposal_classifier_features

 171   def restore_from_classification_checkpoint_fn(
 172       self,
 173       first_stage_feature_extractor_scope,
 174       second_stage_feature_extractor_scope):
 175     """Returns a map of variables to load from a foreign checkpoint.
 176 
 177     Note that this overrides the default implementation in
 178     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor which does not work for
 179     InceptionResnetV2 checkpoints.
 180 
 181     TODO(jonathanhuang,rathodv): revisit whether it's possible to force the
 182     `Repeat` namescope as created in `_extract_box_classifier_features` to
 183     start counting at 2 (e.g. `Repeat_2`) so that the default restore_fn can
 184     be used.
 185 
 186     Args:
 187       first_stage_feature_extractor_scope: A scope name for the first stage
 188         feature extractor.
 189       second_stage_feature_extractor_scope: A scope name for the second stage
 190         feature extractor.
 191 
 192     Returns:
 193       A dict mapping variable names (to load from a checkpoint) to variables in
 194       the model graph.
 195     """
 196 
 197     variables_to_restore = {}
 198     for variable in variables_helper.get_global_variables_safely():
 199       if variable.op.name.startswith(
 200           first_stage_feature_extractor_scope):
 201         var_name = variable.op.name.replace(
 202             first_stage_feature_extractor_scope + '/', '')
 203         variables_to_restore[var_name] = variable
 204       if variable.op.name.startswith(
 205           second_stage_feature_extractor_scope):
 206         var_name = variable.op.name.replace(
 207             second_stage_feature_extractor_scope
 208             + '/InceptionResnetV2/Repeat', 'InceptionResnetV2/Repeat_2')
 209         var_name = var_name.replace(
 210             second_stage_feature_extractor_scope + '/', '')
 211         variables_to_restore[var_name] = variable
 212     return variables_to_restore
