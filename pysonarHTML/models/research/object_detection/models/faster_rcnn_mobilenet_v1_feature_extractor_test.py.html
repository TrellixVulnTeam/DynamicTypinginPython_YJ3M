  16 """Tests for faster_rcnn_mobilenet_v1_feature_extractor."""
  17 
  18 import numpy as np
  19 import tensorflow.compat.v1 as tf
  20 
  21 from object_detection.models import faster_rcnn_mobilenet_v1_feature_extractor as faster_rcnn_mobilenet_v1
  22 
  23 
  24 class FasterRcnnMobilenetV1FeatureExtractorTest(tf.test.TestCase):
  25 
  26   def _build_feature_extractor(self, first_stage_features_stride):
  27     return faster_rcnn_mobilenet_v1.FasterRCNNMobilenetV1FeatureExtractor(
  28         is_training=False,
  29         first_stage_features_stride=first_stage_features_stride,
  30         batch_norm_trainable=False,
  31         reuse_weights=None,
  32         weight_decay=0.0)
  33 
  34   def test_extract_proposal_features_returns_expected_size(self):
  35     feature_extractor = self._build_feature_extractor(
  36         first_stage_features_stride=16)
  37     preprocessed_inputs = tf.random_uniform(
  38         [4, 224, 224, 3], maxval=255, dtype=tf.float32)
  39     rpn_feature_map, _ = feature_extractor.extract_proposal_features(
  40         preprocessed_inputs, scope='TestScope')
  41     features_shape = tf.shape(rpn_feature_map)
  42 
  43     init_op = tf.global_variables_initializer()
  44     with self.test_session() as sess:
  45       sess.run(init_op)
  46       features_shape_out = sess.run(features_shape)
  47       self.assertAllEqual(features_shape_out, [4, 14, 14, 512])
  48

  83   def test_extract_proposal_features_dies_on_very_small_images(self):
  84     feature_extractor = self._build_feature_extractor(
  85         first_stage_features_stride=16)
  86     preprocessed_inputs = tf.placeholder(tf.float32, (4, None, None, 3))
  87     rpn_feature_map, _ = feature_extractor.extract_proposal_features(
  88         preprocessed_inputs, scope='TestScope')
  89     features_shape = tf.shape(rpn_feature_map)
  90 
  91     init_op = tf.global_variables_initializer()
  92     with self.test_session() as sess:
  93       sess.run(init_op)
  94       with self.assertRaises(tf.errors.InvalidArgumentError):
  95         sess.run(
  96             features_shape,
  97             feed_dict={preprocessed_inputs: np.random.rand(4, 32, 32, 3)})
  98 
  99   def test_extract_proposal_features_dies_with_incorrect_rank_inputs(self):

 101         first_stage_features_stride=16)

 103         [224, 224, 3], maxval=255, dtype=tf.float32)
 104     with self.assertRaises(ValueError):
 105       feature_extractor.extract_proposal_features(
 106           preprocessed_inputs, scope='TestScope')

 108   def test_extract_box_classifier_features_returns_expected_size(self):
 109     feature_extractor = self._build_feature_extractor(
 110         first_stage_features_stride=16)
 111     proposal_feature_maps = tf.random_uniform(
 112         [3, 14, 14, 576], maxval=255, dtype=tf.float32)
 113     proposal_classifier_features = (
 114         feature_extractor.extract_box_classifier_features(
 115             proposal_feature_maps, scope='TestScope'))
 116     features_shape = tf.shape(proposal_classifier_features)
 117 
 118     init_op = tf.global_variables_initializer()
 119     with self.test_session() as sess:
 120       sess.run(init_op)
 121       features_shape_out = sess.run(features_shape)
 122       self.assertAllEqual(features_shape_out, [3, 7, 7, 1024])
