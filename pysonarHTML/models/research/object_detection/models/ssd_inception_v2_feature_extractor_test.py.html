  27   def _create_feature_extractor(self,
  28                                 depth_multiplier,
  29                                 pad_to_multiple,
  30                                 use_explicit_padding=False,
  31                                 num_layers=6,
  32                                 is_training=True):
  33     """Constructs a SsdInceptionV2FeatureExtractor.
  34 
  35     Args:
  36       depth_multiplier: float depth multiplier for feature extractor
  37       pad_to_multiple: the nearest multiple to zero pad the input height and
  38         width dimensions to.
  39       use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
  40         inputs so that the output dimensions are the same as if 'SAME' padding
  41         were used.
  42       num_layers: number of SSD layers.
  43       is_training: whether the network is in training mode.
  44 
  45     Returns:
  46       an ssd_inception_v2_feature_extractor.SsdInceptionV2FeatureExtractor.
  47     """
  48     min_depth = 32
  49     return ssd_inception_v2_feature_extractor.SSDInceptionV2FeatureExtractor(
  50         is_training,
  51         depth_multiplier,
  52         min_depth,
  53         pad_to_multiple,
  54         self.conv_hyperparams_fn,
  55         num_layers=num_layers,
  56         override_base_feature_extractor_hyperparams=True)

 126   def test_preprocess_returns_correct_value_range(self):
 127     image_height = 128
 128     image_width = 128
 129     depth_multiplier = 1
 130     pad_to_multiple = 1
 131     test_image = np.random.rand(4, image_height, image_width, 3)
 132     feature_extractor = self._create_feature_extractor(depth_multiplier,
 133                                                        pad_to_multiple)
 134     preprocessed_image = feature_extractor.preprocess(test_image)
 135     self.assertTrue(np.all(np.less_equal(np.abs(preprocessed_image), 1.0)))
