  16 """Tests for ssd_mobilenet_v1_fpn_feature_extractor.
  17 
  18 By using parameterized test decorator, this test serves for both Slim-based and
  19 Keras-based Mobilenet V1 FPN feature extractors in SSD.

  37   def _create_feature_extractor(self, depth_multiplier, pad_to_multiple,
  38                                 is_training=True, use_explicit_padding=False,
  39                                 use_keras=False):
  40     """Constructs a new feature extractor.
  41 
  42     Args:
  43       depth_multiplier: float depth multiplier for feature extractor
  44       pad_to_multiple: the nearest multiple to zero pad the input height and
  45         width dimensions to.
  46       is_training: whether the network is in training mode.
  47       use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
  48         inputs so that the output dimensions are the same as if 'SAME' padding
  49         were used.
  50       use_keras: if True builds a keras-based feature extractor, if False builds
  51         a slim-based one.
  52     Returns:
  53       an ssd_meta_arch.SSDFeatureExtractor object.
  54     """
  55     min_depth = 32
  56     if use_keras:
  57       return (ssd_mobilenet_v1_fpn_keras_feature_extractor.
  58               SSDMobileNetV1FpnKerasFeatureExtractor(
  59                   is_training=is_training,
  60                   depth_multiplier=depth_multiplier,
  61                   min_depth=min_depth,
  62                   pad_to_multiple=pad_to_multiple,
  63                   conv_hyperparams=self._build_conv_hyperparams(
  64                       add_batch_norm=False),
  65                   freeze_batchnorm=False,
  66                   inplace_batchnorm_update=False,
  67                   use_explicit_padding=use_explicit_padding,
  68                   use_depthwise=True,
  69                   name='MobilenetV1_FPN'))
  70     else:
  71       return (ssd_mobilenet_v1_fpn_feature_extractor.
  72               SSDMobileNetV1FpnFeatureExtractor(
  73                   is_training,
  74                   depth_multiplier,
  75                   min_depth,
  76                   pad_to_multiple,
  77                   self.conv_hyperparams_fn,
  78                   use_depthwise=True,
  79                   use_explicit_padding=use_explicit_padding))
  80 
  81   def test_extract_features_returns_correct_shapes_256(self, use_keras):
  82     image_height = 256

  90         2, image_height, image_width, depth_multiplier, pad_to_multiple,

  92         use_keras=use_keras)

  94         2, image_height, image_width, depth_multiplier, pad_to_multiple,

  96         use_keras=use_keras)

 204   def test_fused_batchnorm(self, use_keras):
 205     image_height = 256
 206     image_width = 256
 207     depth_multiplier = 1
 208     pad_to_multiple = 1
 209     image_placeholder = tf.placeholder(tf.float32,
 210                                        [1, image_height, image_width, 3])
 211     feature_extractor = self._create_feature_extractor(depth_multiplier,
 212                                                        pad_to_multiple,
 213                                                        use_keras=use_keras)
 214     preprocessed_image = feature_extractor.preprocess(image_placeholder)
 215     if use_keras:
 216       _ = feature_extractor(preprocessed_image)
 217     else:
 218       _ = feature_extractor.extract_features(preprocessed_image)
 219 
 220     self.assertTrue(
 221         any('FusedBatchNorm' in op.type
 222             for op in tf.get_default_graph().get_operations()))
