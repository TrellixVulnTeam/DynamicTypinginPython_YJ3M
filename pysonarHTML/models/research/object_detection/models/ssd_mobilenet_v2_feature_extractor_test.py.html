  34   def _create_feature_extractor(self,
  35                                 depth_multiplier,
  36                                 pad_to_multiple,
  37                                 use_explicit_padding=False,
  38                                 num_layers=6,
  39                                 use_keras=False):
  40     """Constructs a new feature extractor.
  41 
  42     Args:
  43       depth_multiplier: float depth multiplier for feature extractor
  44       pad_to_multiple: the nearest multiple to zero pad the input height and
  45         width dimensions to.
  46       use_explicit_padding: use 'VALID' padding for convolutions, but prepad
  47         inputs so that the output dimensions are the same as if 'SAME' padding
  48         were used.
  49       num_layers: number of SSD layers.
  50       use_keras: if True builds a keras-based feature extractor, if False builds
  51         a slim-based one.
  52     Returns:
  53       an ssd_meta_arch.SSDFeatureExtractor object.
  54     """
  55     min_depth = 32
  56     if use_keras:
  57       return (ssd_mobilenet_v2_keras_feature_extractor.
  58               SSDMobileNetV2KerasFeatureExtractor(
  59                   is_training=False,
  60                   depth_multiplier=depth_multiplier,
  61                   min_depth=min_depth,
  62                   pad_to_multiple=pad_to_multiple,
  63                   conv_hyperparams=self._build_conv_hyperparams(),
  64                   freeze_batchnorm=False,
  65                   inplace_batchnorm_update=False,
  66                   use_explicit_padding=use_explicit_padding,
  67                   num_layers=num_layers,
  68                   name='MobilenetV2'))
  69     else:
  70       return ssd_mobilenet_v2_feature_extractor.SSDMobileNetV2FeatureExtractor(
  71           False,
  72           depth_multiplier,
  73           min_depth,
  74           pad_to_multiple,
  75           self.conv_hyperparams_fn,
  76           use_explicit_padding=use_explicit_padding,
  77           num_layers=num_layers)

 192   def test_has_fused_batchnorm(self, use_keras):
 193     image_height = 40
 194     image_width = 40
 195     depth_multiplier = 1
 196     pad_to_multiple = 1
 197     image_placeholder = tf.placeholder(tf.float32,
 198                                        [1, image_height, image_width, 3])
 199     feature_extractor = self._create_feature_extractor(depth_multiplier,
 200                                                        pad_to_multiple,
 201                                                        use_keras=use_keras)
 202     preprocessed_image = feature_extractor.preprocess(image_placeholder)
 203     if use_keras:
 204       _ = feature_extractor(preprocessed_image)
 205     else:
 206       _ = feature_extractor.extract_features(preprocessed_image)
 207     self.assertTrue(any('FusedBatchNorm' in op.type
 208                         for op in tf.get_default_graph().get_operations()))
