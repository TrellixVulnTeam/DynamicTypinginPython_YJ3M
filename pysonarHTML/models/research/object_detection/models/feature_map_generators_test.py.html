  61   def _build_conv_hyperparams(self):
  62     conv_hyperparams = hyperparams_pb2.Hyperparams()
  63     conv_hyperparams_text_proto = """
  64       regularizer {
  65         l2_regularizer {
  66         }
  67       }
  68       initializer {
  69         truncated_normal_initializer {
  70         }
  71       }
  72     """
  73     text_format.Merge(conv_hyperparams_text_proto, conv_hyperparams)
  74     return hyperparams_builder.KerasLayerHyperparams(conv_hyperparams)

  76   def _build_feature_map_generator(self, feature_map_layout, use_keras,
  77                                    pool_residual=False):
  78     if use_keras:
  79       return feature_map_generators.KerasMultiResolutionFeatureMaps(
  80           feature_map_layout=feature_map_layout,
  81           depth_multiplier=1,
  82           min_depth=32,
  83           insert_1x1_conv=True,
  84           freeze_batchnorm=False,
  85           is_training=True,
  86           conv_hyperparams=self._build_conv_hyperparams(),
  87           name='FeatureMaps'
  88       )
  89     else:
  90       def feature_map_generator(image_features):
  91         return feature_map_generators.multi_resolution_feature_maps(
  92             feature_map_layout=feature_map_layout,
  93             depth_multiplier=1,
  94             min_depth=32,
  95             insert_1x1_conv=True,
  96             image_features=image_features,
  97             pool_residual=pool_residual)
  98       return feature_map_generator

 100   def test_get_expected_feature_map_shapes_with_inception_v2(self, use_keras):
 101     image_features = {
 102         'Mixed_3c': tf.random_uniform([4, 28, 28, 256], dtype=tf.float32),
 103         'Mixed_4c': tf.random_uniform([4, 14, 14, 576], dtype=tf.float32),
 104         'Mixed_5c': tf.random_uniform([4, 7, 7, 1024], dtype=tf.float32)
 105     }
 106     feature_map_generator = self._build_feature_map_generator(
 107         feature_map_layout=INCEPTION_V2_LAYOUT,

 118         'Mixed_5c_2_Conv2d_5_3x3_s2_256': (4, 1, 1, 256)}
 119 
 120     init_op = tf.global_variables_initializer()
 121     with self.test_session() as sess:
 122       sess.run(init_op)
 123       out_feature_maps = sess.run(feature_maps)
 124       out_feature_map_shapes = dict(
 125           (key, value.shape) for key, value in out_feature_maps.items())

 558   def test_use_bounded_activations_clip_value(
 559       self, use_native_resize_op, use_keras):
 560     tf_graph = tf.Graph()
 561     with tf_graph.as_default():
 562       image_features = [
 563           ('block2', 255 * tf.ones([4, 8, 8, 256], dtype=tf.float32)),
 564           ('block3', 255 * tf.ones([4, 4, 4, 256], dtype=tf.float32)),
 565           ('block4', 255 * tf.ones([4, 2, 2, 256], dtype=tf.float32)),
 566           ('block5', 255 * tf.ones([4, 1, 1, 256], dtype=tf.float32))
 567       ]
 568       feature_map_generator = self._build_feature_map_generator(
 569           image_features=image_features,
 570           depth=128,
 571           use_keras=use_keras,
 572           use_bounded_activations=True,
 573           use_native_resize_op=use_native_resize_op)
 574       feature_map_generator(image_features)
 575 
 576       if use_keras:
 577         expected_clip_by_value_ops = dict.fromkeys([
 578             'FeatureMaps/top_down/clip_by_value/clip_by_value',
 579             'FeatureMaps/top_down/clip_by_value_1/clip_by_value',
 580             'FeatureMaps/top_down/clip_by_value_2/clip_by_value',
 581             'FeatureMaps/top_down/clip_by_value_3/clip_by_value',
 582             'FeatureMaps/top_down/clip_by_value_4/clip_by_value',
 583             'FeatureMaps/top_down/clip_by_value_5/clip_by_value',
 584             'FeatureMaps/top_down/clip_by_value_6/clip_by_value',
 585         ])
 586       else:
 587         expected_clip_by_value_ops = [
 588             'top_down/clip_by_value', 'top_down/clip_by_value_1',
 589             'top_down/clip_by_value_2', 'top_down/clip_by_value_3',
 590             'top_down/clip_by_value_4', 'top_down/clip_by_value_5',
 591             'top_down/clip_by_value_6'
 592         ]
 593 
 594       # Gathers activation tensors before and after clip_by_value operations.
 595       activations = {}
 596       for clip_by_value_op in expected_clip_by_value_ops:
 597         clip_input_tensor = tf_graph.get_operation_by_name(
 598             '{}/Minimum'.format(clip_by_value_op)).inputs[0]
 599         clip_output_tensor = tf_graph.get_tensor_by_name(
 600             '{}:0'.format(clip_by_value_op))
 601         activations.update({
 602             'before_{}'.format(clip_by_value_op): clip_input_tensor,
 603             'after_{}'.format(clip_by_value_op): clip_output_tensor,
 604         })
 605 
 606       expected_lower_bound = -feature_map_generators.ACTIVATION_BOUND
 607       expected_upper_bound = feature_map_generators.ACTIVATION_BOUND
 608       init_op = tf.global_variables_initializer()
 609       with self.test_session() as session:
 610         session.run(init_op)
 611         activations_output = session.run(activations)
 612         for clip_by_value_op in expected_clip_by_value_ops:
 613           # Before clipping, activations are beyound the expected bound because
 614           # of large input image_features values.
 615           activations_before_clipping = (
 616               activations_output['before_{}'.format(clip_by_value_op)])
 617           before_clipping_lower_bound = np.amin(activations_before_clipping)
 618           before_clipping_upper_bound = np.amax(activations_before_clipping)
 619           self.assertLessEqual(before_clipping_lower_bound,
 620                                expected_lower_bound)
 621           self.assertGreaterEqual(before_clipping_upper_bound,
 622                                   expected_upper_bound)
 623 
 624           # After clipping, activations are bounded as expectation.
 625           activations_after_clipping = (
 626               activations_output['after_{}'.format(clip_by_value_op)])
 627           after_clipping_lower_bound = np.amin(activations_after_clipping)
 628           after_clipping_upper_bound = np.amax(activations_after_clipping)
 629           self.assertGreaterEqual(after_clipping_lower_bound,
 630                                   expected_lower_bound)
 631           self.assertLessEqual(after_clipping_upper_bound, expected_upper_bound)
