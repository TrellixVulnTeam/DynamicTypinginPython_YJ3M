  37 class SSDResnetV1FpnFeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
  38   """SSD FPN feature extractor based on Resnet v1 architecture."""
  39 
  40   def __init__(self,
  41                is_training,
  42                depth_multiplier,
  43                min_depth,
  44                pad_to_multiple,
  45                conv_hyperparams_fn,
  46                resnet_base_fn,
  47                resnet_scope_name,
  48                fpn_scope_name,
  49                fpn_min_level=3,
  50                fpn_max_level=7,
  51                additional_layer_depth=256,
  52                reuse_weights=None,
  53                use_explicit_padding=False,
  54                use_depthwise=False,
  55                use_native_resize_op=False,
  56                override_base_feature_extractor_hyperparams=False):
  57     """SSD FPN feature extractor based on Resnet v1 architecture.
  58 
  59     Args:
  60       is_training: whether the network is in training mode.
  61       depth_multiplier: float depth multiplier for feature extractor.
  62       min_depth: minimum feature extractor depth.
  63       pad_to_multiple: the nearest multiple to zero pad the input height and
  64         width dimensions to.
  65       conv_hyperparams_fn: A function to construct tf slim arg_scope for conv2d
  66         and separable_conv2d ops in the layers that are added on top of the
  67         base feature extractor.
  68       resnet_base_fn: base resnet network to use.
  69       resnet_scope_name: scope name under which to construct resnet
  70       fpn_scope_name: scope name under which to construct the feature pyramid
  71         network.
  72       fpn_min_level: the highest resolution feature map to use in FPN. The valid
  73         values are {2, 3, 4, 5} which map to Resnet blocks {1, 2, 3, 4}
  74         respectively.
  75       fpn_max_level: the smallest resolution feature map to construct or use in
  76         FPN. FPN constructions uses features maps starting from fpn_min_level
  77         upto the fpn_max_level. In the case that there are not enough feature
  78         maps in the backbone network, additional feature maps are created by
  79         applying stride 2 convolutions until we get the desired number of fpn
  80         levels.
  81       additional_layer_depth: additional feature map layer channel depth.
  82       reuse_weights: Whether to reuse variables. Default is None.
  83       use_explicit_padding: Whether to use explicit padding when extracting
  84         features. Default is False. UNUSED currently.
  85       use_depthwise: Whether to use depthwise convolutions. UNUSED currently.
  86       use_native_resize_op: Whether to use tf.image.nearest_neighbor_resize
  87         to do upsampling in FPN. Default is false.
  88       override_base_feature_extractor_hyperparams: Whether to override
  89         hyperparameters of the base feature extractor with the one from
  90         `conv_hyperparams_fn`.
  91 
  92     Raises:
  93       ValueError: On supplying invalid arguments for unused arguments.
  94     """
  95     super(SSDResnetV1FpnFeatureExtractor, self).__init__(
  96         is_training=is_training,
  97         depth_multiplier=depth_multiplier,
  98         min_depth=min_depth,
  99         pad_to_multiple=pad_to_multiple,
 100         conv_hyperparams_fn=conv_hyperparams_fn,
 101         reuse_weights=reuse_weights,
 102         use_explicit_padding=use_explicit_padding,
 103         use_depthwise=use_depthwise,
 104         override_base_feature_extractor_hyperparams=
 105         override_base_feature_extractor_hyperparams)
 106     if self._use_explicit_padding is True:
 107       raise ValueError('Explicit padding is not a valid option.')
 108     self._resnet_base_fn = resnet_base_fn
 109     self._resnet_scope_name = resnet_scope_name
 110     self._fpn_scope_name = fpn_scope_name
 111     self._fpn_min_level = fpn_min_level
 112     self._fpn_max_level = fpn_max_level
 113     self._additional_layer_depth = additional_layer_depth
 114     self._use_native_resize_op = use_native_resize_op
 115 
 116   def preprocess(self, resized_inputs):
 117     """SSD preprocessing.
 118 
 119     VGG style channel mean subtraction as described here:
 120     https://gist.github.com/ksimonyan/211839e770f7b538e2d8#file-readme-mdnge.
 121     Note that if the number of channels is not equal to 3, the mean subtraction
 122     will be skipped and the original resized_inputs will be returned.
 123 
 124     Args:
 125       resized_inputs: a [batch, height, width, channels] float tensor
 126         representing a batch of images.
 127 
 128     Returns:
 129       preprocessed_inputs: a [batch, height, width, channels] float tensor
 130         representing a batch of images.
 131     """
 132     if resized_inputs.shape.as_list()[3] == 3:
 133       channel_means = [123.68, 116.779, 103.939]
 134       return resized_inputs - [[channel_means]]
 135     else:
 136       return resized_inputs
 137 
 138   def _filter_features(self, image_features):
 139     # TODO(rathodv): Change resnet endpoint to strip scope prefixes instead
 140     # of munging the scope here.
 141     filtered_image_features = dict({})
 142     for key, feature in image_features.items():
 143       feature_name = key.split('/')[-1]
 144       if feature_name in ['block1', 'block2', 'block3', 'block4']:
 145         filtered_image_features[feature_name] = feature
 146     return filtered_image_features
 147 
 148   def extract_features(self, preprocessed_inputs):
 149     """Extract features from preprocessed inputs.
 150 
 151     Args:
 152       preprocessed_inputs: a [batch, height, width, channels] float tensor
 153         representing a batch of images.
 154 
 155     Returns:
 156       feature_maps: a list of tensors where the ith tensor has shape
 157         [batch, height_i, width_i, depth_i]
 158     """
 159     preprocessed_inputs = shape_utils.check_min_image_dim(
 160         129, preprocessed_inputs)
 161 
 162     with tf.variable_scope(
 163         self._resnet_scope_name, reuse=self._reuse_weights) as scope:
 164       with slim.arg_scope(resnet_v1.resnet_arg_scope()):
 165         with (slim.arg_scope(self._conv_hyperparams_fn())
 166               if self._override_base_feature_extractor_hyperparams else
 167               context_manager.IdentityContextManager()):
 168           _, image_features = self._resnet_base_fn(
 169               inputs=ops.pad_to_multiple(preprocessed_inputs,
 170                                          self._pad_to_multiple),
 171               num_classes=None,
 172               is_training=None,
 173               global_pool=False,
 174               output_stride=None,
 175               store_non_strided_activations=True,
 176               min_base_depth=self._min_depth,
 177               depth_multiplier=self._depth_multiplier,
 178               scope=scope)
 179           image_features = self._filter_features(image_features)
 180       depth_fn = lambda d: max(int(d * self._depth_multiplier), self._min_depth)
 181       with slim.arg_scope(self._conv_hyperparams_fn()):
 182         with tf.variable_scope(self._fpn_scope_name,
 183                                reuse=self._reuse_weights):
 184           base_fpn_max_level = min(self._fpn_max_level, 5)
 185           feature_block_list = []
 186           for level in range(self._fpn_min_level, base_fpn_max_level + 1):
 187             feature_block_list.append('block{}'.format(level - 1))
 188           fpn_features = feature_map_generators.fpn_top_down_feature_maps(
 189               [(key, image_features[key]) for key in feature_block_list],
 190               depth=depth_fn(self._additional_layer_depth),
 191               use_native_resize_op=self._use_native_resize_op)
 192           feature_maps = []
 193           for level in range(self._fpn_min_level, base_fpn_max_level + 1):
 194             feature_maps.append(
 195                 fpn_features['top_down_block{}'.format(level - 1)])
 196           last_feature_map = fpn_features['top_down_block{}'.format(
 197               base_fpn_max_level - 1)]
 198           # Construct coarse features
 199           for i in range(base_fpn_max_level, self._fpn_max_level):
 200             last_feature_map = slim.conv2d(
 201                 last_feature_map,
 202                 num_outputs=depth_fn(self._additional_layer_depth),
 203                 kernel_size=[3, 3],
 204                 stride=2,
 205                 padding='SAME',
 206                 scope='bottom_up_block{}'.format(i))
 207             feature_maps.append(last_feature_map)
 208     return feature_maps
