  51   def _create_feature_extractor(self,
  52                                 depth_multiplier,
  53                                 pad_to_multiple,
  54                                 is_training=True,
  55                                 use_explicit_padding=False,
  56                                 use_keras=False,
  57                                 use_depthwise=False):
  58     """Constructs a new feature extractor.
  59 
  60     Args:
  61       depth_multiplier: float depth multiplier for feature extractor
  62       pad_to_multiple: the nearest multiple to zero pad the input height and
  63         width dimensions to.
  64       is_training: whether the network is in training mode.
  65       use_explicit_padding: Use 'VALID' padding for convolutions, but prepad
  66         inputs so that the output dimensions are the same as if 'SAME' padding
  67         were used.
  68       use_keras: if True builds a keras-based feature extractor, if False builds
  69         a slim-based one.
  70       use_depthwise: Whether to use depthwise convolutions.
  71     Returns:
  72       an ssd_meta_arch.SSDFeatureExtractor object.
  73     """
  74     min_depth = 32
  75     if use_keras:
  76       return (ssd_mobilenet_v2_fpn_keras_feature_extractor
  77               .SSDMobileNetV2FpnKerasFeatureExtractor(
  78                   is_training=is_training,
  79                   depth_multiplier=depth_multiplier,
  80                   min_depth=min_depth,
  81                   pad_to_multiple=pad_to_multiple,
  82                   conv_hyperparams=self._build_conv_hyperparams(
  83                       add_batch_norm=False),
  84                   freeze_batchnorm=False,
  85                   inplace_batchnorm_update=False,
  86                   use_explicit_padding=use_explicit_padding,
  87                   use_depthwise=use_depthwise,
  88                   name='MobilenetV2_FPN'))
  89     else:
  90       return (ssd_mobilenet_v2_fpn_feature_extractor
  91               .SSDMobileNetV2FpnFeatureExtractor(
  92                   is_training,
  93                   depth_multiplier,
  94                   min_depth,
  95                   pad_to_multiple,
  96                   self.conv_hyperparams_fn,
  97                   use_depthwise=use_depthwise,
  98                   use_explicit_padding=use_explicit_padding))

 264   def test_preprocess_returns_correct_value_range(self, use_keras,
 265                                                   use_depthwise):
 266     image_height = 256
 267     image_width = 256
 268     depth_multiplier = 1
 269     pad_to_multiple = 1
 270     test_image = np.random.rand(2, image_height, image_width, 3)
 271     feature_extractor = self._create_feature_extractor(

 273         pad_to_multiple,
 274         use_keras=use_keras,

 276     preprocessed_image = feature_extractor.preprocess(test_image)
 277     self.assertTrue(np.all(np.less_equal(np.abs(preprocessed_image), 1.0)))

 290   def test_fused_batchnorm(self, use_keras, use_depthwise):
 291     image_height = 256
 292     image_width = 256
 293     depth_multiplier = 1
 294     pad_to_multiple = 1
 295     image_placeholder = tf.placeholder(tf.float32,
 296                                        [1, image_height, image_width, 3])
 297     feature_extractor = self._create_feature_extractor(
 298         depth_multiplier,
 299         pad_to_multiple,
 300         use_keras=use_keras,
 301         use_depthwise=use_depthwise)
 302     preprocessed_image = feature_extractor.preprocess(image_placeholder)
 303     if use_keras:
 304       _ = feature_extractor(preprocessed_image)
 305     else:
 306       _ = feature_extractor.extract_features(preprocessed_image)
 307     self.assertTrue(
 308         any('FusedBatchNorm' in op.type
 309             for op in tf.get_default_graph().get_operations()))

 324   def test_get_expected_feature_map_variable_names(self, use_keras,
 325                                                    use_depthwise):
 326     depth_multiplier = 1.0
 327     pad_to_multiple = 1
 328 
 329     slim_expected_feature_maps_variables = set([
 330         # Slim Mobilenet V2 feature maps
 331         'MobilenetV2/expanded_conv_4/depthwise/depthwise_weights',
 332         'MobilenetV2/expanded_conv_7/depthwise/depthwise_weights',
 333         'MobilenetV2/expanded_conv_14/depthwise/depthwise_weights',
 334         'MobilenetV2/Conv_1/weights',
 335         # FPN layers
 336         'MobilenetV2/fpn/bottom_up_Conv2d_20/weights',
 337         'MobilenetV2/fpn/bottom_up_Conv2d_21/weights',
 338         'MobilenetV2/fpn/smoothing_1/weights',
 339         'MobilenetV2/fpn/smoothing_2/weights',
 340         'MobilenetV2/fpn/projection_1/weights',
 341         'MobilenetV2/fpn/projection_2/weights',
 342         'MobilenetV2/fpn/projection_3/weights',
 343     ])
 344     slim_expected_feature_maps_variables_with_depthwise = set([
 345         # Slim Mobilenet V2 feature maps
 346         'MobilenetV2/expanded_conv_4/depthwise/depthwise_weights',
 347         'MobilenetV2/expanded_conv_7/depthwise/depthwise_weights',
 348         'MobilenetV2/expanded_conv_14/depthwise/depthwise_weights',
 349         'MobilenetV2/Conv_1/weights',
 350         # FPN layers
 351         'MobilenetV2/fpn/bottom_up_Conv2d_20/pointwise_weights',
 352         'MobilenetV2/fpn/bottom_up_Conv2d_20/depthwise_weights',
 353         'MobilenetV2/fpn/bottom_up_Conv2d_21/pointwise_weights',
 354         'MobilenetV2/fpn/bottom_up_Conv2d_21/depthwise_weights',
 355         'MobilenetV2/fpn/smoothing_1/depthwise_weights',
 356         'MobilenetV2/fpn/smoothing_1/pointwise_weights',
 357         'MobilenetV2/fpn/smoothing_2/depthwise_weights',
 358         'MobilenetV2/fpn/smoothing_2/pointwise_weights',
 359         'MobilenetV2/fpn/projection_1/weights',
 360         'MobilenetV2/fpn/projection_2/weights',
 361         'MobilenetV2/fpn/projection_3/weights',
 362     ])
 363     keras_expected_feature_maps_variables = set([
 364         # Keras Mobilenet V2 feature maps
 365         'MobilenetV2_FPN/block_4_depthwise/depthwise_kernel',
 366         'MobilenetV2_FPN/block_7_depthwise/depthwise_kernel',
 367         'MobilenetV2_FPN/block_14_depthwise/depthwise_kernel',
 368         'MobilenetV2_FPN/Conv_1/kernel',
 369         # FPN layers
 370         'MobilenetV2_FPN/bottom_up_Conv2d_20_conv/kernel',
 371         'MobilenetV2_FPN/bottom_up_Conv2d_21_conv/kernel',
 372         'MobilenetV2_FPN/FeatureMaps/top_down/smoothing_1_conv/kernel',
 373         'MobilenetV2_FPN/FeatureMaps/top_down/smoothing_2_conv/kernel',
 374         'MobilenetV2_FPN/FeatureMaps/top_down/projection_1/kernel',
 375         'MobilenetV2_FPN/FeatureMaps/top_down/projection_2/kernel',
 376         'MobilenetV2_FPN/FeatureMaps/top_down/projection_3/kernel'
 377     ])
 378     keras_expected_feature_maps_variables_with_depthwise = set([
 379         # Keras Mobilenet V2 feature maps
 380         'MobilenetV2_FPN/block_4_depthwise/depthwise_kernel',
 381         'MobilenetV2_FPN/block_7_depthwise/depthwise_kernel',
 382         'MobilenetV2_FPN/block_14_depthwise/depthwise_kernel',
 383         'MobilenetV2_FPN/Conv_1/kernel',
 384         # FPN layers
 385         'MobilenetV2_FPN/bottom_up_Conv2d_20_depthwise_conv/depthwise_kernel',
 386         'MobilenetV2_FPN/bottom_up_Conv2d_20_depthwise_conv/pointwise_kernel',
 387         'MobilenetV2_FPN/bottom_up_Conv2d_21_depthwise_conv/depthwise_kernel',
 388         'MobilenetV2_FPN/bottom_up_Conv2d_21_depthwise_conv/pointwise_kernel',
 389         ('MobilenetV2_FPN/FeatureMaps/top_down/smoothing_1_depthwise_conv/'
 390          'depthwise_kernel'),
 391         ('MobilenetV2_FPN/FeatureMaps/top_down/smoothing_1_depthwise_conv/'
 392          'pointwise_kernel'),
 393         ('MobilenetV2_FPN/FeatureMaps/top_down/smoothing_2_depthwise_conv/'
 394          'depthwise_kernel'),
 395         ('MobilenetV2_FPN/FeatureMaps/top_down/smoothing_2_depthwise_conv/'
 396          'pointwise_kernel'),
 397         'MobilenetV2_FPN/FeatureMaps/top_down/projection_1/kernel',
 398         'MobilenetV2_FPN/FeatureMaps/top_down/projection_2/kernel',
 399         'MobilenetV2_FPN/FeatureMaps/top_down/projection_3/kernel'
 400     ])
 401 
 402     g = tf.Graph()
 403     with g.as_default():
 404       preprocessed_inputs = tf.placeholder(tf.float32, (4, None, None, 3))
 405       feature_extractor = self._create_feature_extractor(
 406           depth_multiplier,
 407           pad_to_multiple,
 408           use_keras=use_keras,
 409           use_depthwise=use_depthwise)
 410       if use_keras:
 411         _ = feature_extractor(preprocessed_inputs)
 412         expected_feature_maps_variables = keras_expected_feature_maps_variables
 413         if use_depthwise:
 414           expected_feature_maps_variables = (
 415               keras_expected_feature_maps_variables_with_depthwise)
 416       else:
 417         _ = feature_extractor.extract_features(preprocessed_inputs)
 418         expected_feature_maps_variables = slim_expected_feature_maps_variables
 419         if use_depthwise:
 420           expected_feature_maps_variables = (
 421               slim_expected_feature_maps_variables_with_depthwise)
 422       actual_variable_set = set([
 423           var.op.name for var in g.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)
 424       ])
 425       variable_intersection = expected_feature_maps_variables.intersection(
 426           actual_variable_set)
 427       self.assertSetEqual(expected_feature_maps_variables,
 428                           variable_intersection)
