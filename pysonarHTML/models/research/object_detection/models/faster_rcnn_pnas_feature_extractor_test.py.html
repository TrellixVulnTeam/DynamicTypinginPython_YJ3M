  16 """Tests for models.faster_rcnn_pnas_feature_extractor."""
  17 
  18 import tensorflow.compat.v1 as tf
  19 
  20 from object_detection.models import faster_rcnn_pnas_feature_extractor as frcnn_pnas
  21 
  22 
  23 class FasterRcnnPNASFeatureExtractorTest(tf.test.TestCase):
  24 
  25   def _build_feature_extractor(self, first_stage_features_stride):
  26     return frcnn_pnas.FasterRCNNPNASFeatureExtractor(
  27         is_training=False,
  28         first_stage_features_stride=first_stage_features_stride,
  29         batch_norm_trainable=False,
  30         reuse_weights=None,
  31         weight_decay=0.0)
  32 
  33   def test_extract_proposal_features_returns_expected_size(self):
  34     feature_extractor = self._build_feature_extractor(
  35         first_stage_features_stride=16)
  36     preprocessed_inputs = tf.random_uniform(
  37         [1, 299, 299, 3], maxval=255, dtype=tf.float32)
  38     rpn_feature_map, _ = feature_extractor.extract_proposal_features(
  39         preprocessed_inputs, scope='TestScope')
  40     features_shape = tf.shape(rpn_feature_map)
  41 
  42     init_op = tf.global_variables_initializer()
  43     with self.test_session() as sess:
  44       sess.run(init_op)
  45       features_shape_out = sess.run(features_shape)
  46       self.assertAllEqual(features_shape_out, [1, 19, 19, 4320])
  47

  91   def test_extract_box_classifier_features_returns_expected_size(self):
  92     feature_extractor = self._build_feature_extractor(
  93         first_stage_features_stride=16)
  94     proposal_feature_maps = tf.random_uniform(
  95         [2, 17, 17, 1088], maxval=255, dtype=tf.float32)
  96     proposal_classifier_features = (
  97         feature_extractor.extract_box_classifier_features(
  98             proposal_feature_maps, scope='TestScope'))
  99     features_shape = tf.shape(proposal_classifier_features)
 100 
 101     init_op = tf.global_variables_initializer()
 102     with self.test_session() as sess:
 103       sess.run(init_op)
 104       features_shape_out = sess.run(features_shape)
 105       self.assertAllEqual(features_shape_out, [2, 9, 9, 4320])

 107   def test_filter_scaling_computation(self):
 108     expected_filter_scaling = {
 109         ((4, 8), 2): 1.0,
 110         ((4, 8), 7): 2.0,
 111         ((4, 8), 8): 2.0,
 112         ((4, 8), 9): 4.0
 113     }
 114     for args, filter_scaling in expected_filter_scaling.items():
 115       reduction_indices, start_cell_num = args
 116       self.assertAlmostEqual(
 117           frcnn_pnas._filter_scaling(reduction_indices, start_cell_num),
 118           filter_scaling)
