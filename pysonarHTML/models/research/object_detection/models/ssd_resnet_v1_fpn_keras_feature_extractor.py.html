  43 class SSDResNetV1FpnKerasFeatureExtractor(
  44     ssd_meta_arch.SSDKerasFeatureExtractor):
  45   """SSD Feature Extractor using Keras-based ResnetV1 FPN features."""
  46 
  47   def __init__(self,
  48                is_training,
  49                depth_multiplier,
  50                min_depth,
  51                pad_to_multiple,
  52                conv_hyperparams,
  53                freeze_batchnorm,
  54                inplace_batchnorm_update,
  55                resnet_v1_base_model,
  56                resnet_v1_base_model_name,
  57                fpn_min_level=3,
  58                fpn_max_level=7,
  59                additional_layer_depth=256,
  60                reuse_weights=None,
  61                use_explicit_padding=None,
  62                use_depthwise=None,
  63                override_base_feature_extractor_hyperparams=False,
  64                name=None):
  65     """SSD Keras based FPN feature extractor Resnet v1 architecture.
  66 
  67     Args:
  68       is_training: whether the network is in training mode.
  69       depth_multiplier: float depth multiplier for feature extractor.
  70       min_depth: minimum feature extractor depth.
  71       pad_to_multiple: the nearest multiple to zero pad the input height and
  72         width dimensions to.
  73       conv_hyperparams: a `hyperparams_builder.KerasLayerHyperparams` object
  74         containing convolution hyperparameters for the layers added on top of
  75         the base feature extractor.
  76       freeze_batchnorm: whether to freeze batch norm parameters during
  77         training or not. When training with a small batch size (e.g. 1), it is
  78         desirable to freeze batch norm update and use pretrained batch norm
  79         params.
  80       inplace_batchnorm_update: whether to update batch norm moving average
  81         values inplace. When this is false train op must add a control
  82         dependency on tf.graphkeys.UPDATE_OPS collection in order to update
  83         batch norm statistics.
  84       resnet_v1_base_model: base resnet v1 network to use. One of
  85         the resnet_v1.resnet_v1_{50,101,152} models.
  86       resnet_v1_base_model_name: model name under which to construct resnet v1.
  87       fpn_min_level: the highest resolution feature map to use in FPN. The valid
  88         values are {2, 3, 4, 5} which map to Resnet blocks {1, 2, 3, 4}
  89         respectively.
  90       fpn_max_level: the smallest resolution feature map to construct or use in
  91         FPN. FPN constructions uses features maps starting from fpn_min_level
  92         upto the fpn_max_level. In the case that there are not enough feature
  93         maps in the backbone network, additional feature maps are created by
  94         applying stride 2 convolutions until we get the desired number of fpn
  95         levels.
  96       additional_layer_depth: additional feature map layer channel depth.
  97       reuse_weights: whether to reuse variables. Default is None.
  98       use_explicit_padding: whether to use explicit padding when extracting
  99         features. Default is None, as it's an invalid option and not implemented
 100         in this feature extractor.
 101       use_depthwise: Whether to use depthwise convolutions. UNUSED currently.
 102       override_base_feature_extractor_hyperparams: Whether to override
 103         hyperparameters of the base feature extractor with the one from
 104         `conv_hyperparams`.
 105       name: a string name scope to assign to the model. If 'None', Keras
 106         will auto-generate one from the class name.
 107     """
 108     super(SSDResNetV1FpnKerasFeatureExtractor, self).__init__(
 109         is_training=is_training,
 110         depth_multiplier=depth_multiplier,
 111         min_depth=min_depth,
 112         pad_to_multiple=pad_to_multiple,
 113         conv_hyperparams=conv_hyperparams,
 114         freeze_batchnorm=freeze_batchnorm,
 115         inplace_batchnorm_update=inplace_batchnorm_update,
 116         use_explicit_padding=None,
 117         use_depthwise=None,
 118         override_base_feature_extractor_hyperparams=
 119         override_base_feature_extractor_hyperparams,
 120         name=name)
 121     if self._use_explicit_padding:
 122       raise ValueError('Explicit padding is not a valid option.')
 123     if self._use_depthwise:
 124       raise ValueError('Depthwise is not a valid option.')
 125     self._fpn_min_level = fpn_min_level
 126     self._fpn_max_level = fpn_max_level
 127     self._additional_layer_depth = additional_layer_depth
 128     self._resnet_v1_base_model = resnet_v1_base_model
 129     self._resnet_v1_base_model_name = resnet_v1_base_model_name
 130     self._resnet_block_names = ['block1', 'block2', 'block3', 'block4']
 131     self.classification_backbone = None
 132     self._fpn_features_generator = None
 133     self._coarse_feature_layers = []
 134 
 135   def build(self, input_shape):
 136     full_resnet_v1_model = self._resnet_v1_base_model(
 137         batchnorm_training=(self._is_training and not self._freeze_batchnorm),
 138         conv_hyperparams=(self._conv_hyperparams
 139                           if self._override_base_feature_extractor_hyperparams
 140                           else None),
 141         depth_multiplier=self._depth_multiplier,
 142         min_depth=self._min_depth,
 143         classes=None,
 144         weights=None,
 145         include_top=False)
 146     output_layers = _RESNET_MODEL_OUTPUT_LAYERS[self._resnet_v1_base_model_name]
 147     outputs = [full_resnet_v1_model.get_layer(output_layer_name).output
 148                for output_layer_name in output_layers]
 149     self.classification_backbone = tf.keras.Model(
 150         inputs=full_resnet_v1_model.inputs,
 151         outputs=outputs)
 152     # pylint:disable=g-long-lambda
 153     self._depth_fn = lambda d: max(
 154         int(d * self._depth_multiplier), self._min_depth)
 155     self._base_fpn_max_level = min(self._fpn_max_level, 5)
 156     self._num_levels = self._base_fpn_max_level + 1 - self._fpn_min_level
 157     self._fpn_features_generator = (
 158         feature_map_generators.KerasFpnTopDownFeatureMaps(
 159             num_levels=self._num_levels,
 160             depth=self._depth_fn(self._additional_layer_depth),
 161             is_training=self._is_training,
 162             conv_hyperparams=self._conv_hyperparams,
 163             freeze_batchnorm=self._freeze_batchnorm,
 164             name='FeatureMaps'))
 165     # Construct coarse feature layers
 166     depth = self._depth_fn(self._additional_layer_depth)
 167     for i in range(self._base_fpn_max_level, self._fpn_max_level):
 168       layers = []
 169       layer_name = 'bottom_up_block{}'.format(i)
 170       layers.append(
 171           tf.keras.layers.Conv2D(
 172               depth,
 173               [3, 3],
 174               padding='SAME',
 175               strides=2,
 176               name=layer_name + '_conv',
 177               **self._conv_hyperparams.params()))
 178       layers.append(
 179           self._conv_hyperparams.build_batch_norm(
 180               training=(self._is_training and not self._freeze_batchnorm),
 181               name=layer_name + '_batchnorm'))
 182       layers.append(
 183           self._conv_hyperparams.build_activation_layer(
 184               name=layer_name))
 185       self._coarse_feature_layers.append(layers)
 186     self.built = True
 187 
 188   def preprocess(self, resized_inputs):
 189     """SSD preprocessing.
 190 
 191     VGG style channel mean subtraction as described here:
 192     https://gist.github.com/ksimonyan/211839e770f7b538e2d8#file-readme-mdnge.
 193     Note that if the number of channels is not equal to 3, the mean subtraction
 194     will be skipped and the original resized_inputs will be returned.
 195 
 196     Args:
 197       resized_inputs: a [batch, height, width, channels] float tensor
 198         representing a batch of images.
 199 
 200     Returns:
 201       preprocessed_inputs: a [batch, height, width, channels] float tensor
 202         representing a batch of images.
 203     """
 204     if resized_inputs.shape.as_list()[3] == 3:
 205       channel_means = [123.68, 116.779, 103.939]
 206       return resized_inputs - [[channel_means]]
 207     else:
 208       return resized_inputs
 209 
 210   def _extract_features(self, preprocessed_inputs):
 211     """Extract features from preprocessed inputs.
 212 
 213     Args:
 214       preprocessed_inputs: a [batch, height, width, channels] float tensor
 215         representing a batch of images.
 216 
 217     Returns:
 218       feature_maps: a list of tensors where the ith tensor has shape
 219         [batch, height_i, width_i, depth_i]
 220     """
 221     preprocessed_inputs = shape_utils.check_min_image_dim(
 222         129, preprocessed_inputs)
 223 
 224     image_features = self.classification_backbone(
 225         ops.pad_to_multiple(preprocessed_inputs, self._pad_to_multiple))
 226 
 227     feature_block_list = []
 228     for level in range(self._fpn_min_level, self._base_fpn_max_level + 1):
 229       feature_block_list.append('block{}'.format(level - 1))
 230     feature_block_map = dict(
 231         list(zip(self._resnet_block_names, image_features)))
 232     fpn_input_image_features = [
 233         (feature_block, feature_block_map[feature_block])
 234         for feature_block in feature_block_list]
 235     fpn_features = self._fpn_features_generator(fpn_input_image_features)
 236 
 237     feature_maps = []
 238     for level in range(self._fpn_min_level, self._base_fpn_max_level + 1):
 239       feature_maps.append(fpn_features['top_down_block{}'.format(level-1)])
 240     last_feature_map = fpn_features['top_down_block{}'.format(
 241         self._base_fpn_max_level - 1)]
 242 
 243     for coarse_feature_layers in self._coarse_feature_layers:
 244       for layer in coarse_feature_layers:
 245         last_feature_map = layer(last_feature_map)
 246       feature_maps.append(last_feature_map)
 247     return feature_maps
 248 
 249   def restore_from_classification_checkpoint_fn(self, feature_extractor_scope):
 250     """Returns a map for restoring from an (object-based) checkpoint.
 251 
 252     Args:
 253       feature_extractor_scope: A scope name for the feature extractor (unused).
 254 
 255     Returns:
 256       A dict mapping keys to Keras models
 257     """
 258     return {'feature_extractor': self.classification_backbone}
