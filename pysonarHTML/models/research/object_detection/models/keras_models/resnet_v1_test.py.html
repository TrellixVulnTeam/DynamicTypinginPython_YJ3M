  15 """Tests for resnet_v1.py.
  16 
  17 This test mainly focuses on comparing slim resnet v1 and Keras resnet v1 for
  18 object detection. To verify the consistency of the two models, we compare:
  19   1. Output shape of each layer given different inputs.
  20   2. Number of global variables.
  21 """

  24 from six.moves import zip
  25 import tensorflow.compat.v1 as tf

  32 from object_detection.utils import test_case

  44     'conv4_block23_out': (4, 14, 14, 1024),

  51     'conv4_block36_out': (4, 14, 14, 1024),

  55 _RESNET_NAMES = ['resnet_v1_50', 'resnet_v1_101', 'resnet_v1_152']
  56 _RESNET_MODELS = [
  57     resnet_v1.resnet_v1_50, resnet_v1.resnet_v1_101, resnet_v1.resnet_v1_152
  58 ]
  59 _RESNET_SHAPES = [
  60     _EXPECTED_SHAPES_224_RESNET50, _EXPECTED_SHAPES_224_RESNET101,
  61     _EXPECTED_SHAPES_224_RESNET152
  62 ]

  64 _NUM_CHANNELS = 3
  65 _BATCH_SIZE = 4

  68 class ResnetV1Test(test_case.TestCase):

  70   def _build_conv_hyperparams(self):
  71     conv_hyperparams = hyperparams_pb2.Hyperparams()
  72     conv_hyperparams_text_proto = """
  73       activation: RELU_6,
  74       regularizer {
  75         l2_regularizer {
  76           weight: 0.0004
  77         }
  78       }
  79       initializer {
  80         truncated_normal_initializer {
  81           stddev: 0.03
  82           mean: 0.0
  83         }
  84       }
  85       batch_norm {
  86         scale: true,
  87         decay: 0.997,
  88         epsilon: 0.001,
  89       }
  90     """
  91     text_format.Merge(conv_hyperparams_text_proto, conv_hyperparams)
  92     return hyperparams_builder.KerasLayerHyperparams(conv_hyperparams)
  93 
  94   def _create_application_with_layer_outputs(self,
  95                                              model_index,
  96                                              batchnorm_training,
  97                                              batchnorm_scale=True,
  98                                              weight_decay=0.0001,
  99                                              default_batchnorm_momentum=0.997,
 100                                              default_batchnorm_epsilon=1e-5):
 101     """Constructs Keras resnet_v1 that extracts layer outputs."""
 102     # Have to clear the Keras backend to ensure isolation in layer naming
 103     tf.keras.backend.clear_session()
 104     layer_names = _RESNET_SHAPES[model_index].keys()
 105     full_model = _RESNET_MODELS[model_index](
 106         batchnorm_training=batchnorm_training,
 107         weights=None,
 108         batchnorm_scale=batchnorm_scale,
 109         weight_decay=weight_decay,
 110         default_batchnorm_momentum=default_batchnorm_momentum,
 111         default_batchnorm_epsilon=default_batchnorm_epsilon,
 112         include_top=False)
 113 
 114     layer_outputs = [
 115         full_model.get_layer(name=layer).output for layer in layer_names
 116     ]
 117     return tf.keras.Model(inputs=full_model.inputs, outputs=layer_outputs)
 118 
 119   def _check_returns_correct_shape(self,
 120                                    image_height,
 121                                    image_width,
 122                                    model_index,
 123                                    expected_feature_map_shape,
 124                                    batchnorm_training=True,
 125                                    batchnorm_scale=True,
 126                                    weight_decay=0.0001,
 127                                    default_batchnorm_momentum=0.997,
 128                                    default_batchnorm_epsilon=1e-5):
 129     model = self._create_application_with_layer_outputs(
 130         model_index=model_index,
 131         batchnorm_training=batchnorm_training,
 132         batchnorm_scale=batchnorm_scale,
 133         weight_decay=weight_decay,
 134         default_batchnorm_momentum=default_batchnorm_momentum,
 135         default_batchnorm_epsilon=default_batchnorm_epsilon)
 136 
 137     image_tensor = np.random.rand(_BATCH_SIZE, image_height, image_width,
 138                                   _NUM_CHANNELS).astype(np.float32)
 139     feature_maps = model(image_tensor)
 140     layer_names = _RESNET_SHAPES[model_index].keys()
 141     for feature_map, layer_name in zip(feature_maps, layer_names):
 142       expected_shape = _RESNET_SHAPES[model_index][layer_name]
 143       self.assertAllEqual(feature_map.shape, expected_shape)
 144 
 145   def _get_variables(self, model_index):
 146     tf.keras.backend.clear_session()
 147     model = self._create_application_with_layer_outputs(
 148         model_index, batchnorm_training=False)
 149     preprocessed_inputs = tf.placeholder(tf.float32,
 150                                          (4, None, None, _NUM_CHANNELS))
 151     model(preprocessed_inputs)
 152     return model.variables
 153 
 154   def test_returns_correct_shapes_224(self):
 155     image_height = 224
 156     image_width = 224
 157     for model_index, _ in enumerate(_RESNET_NAMES):
 158       expected_feature_map_shape = _RESNET_SHAPES[model_index]
 159       self._check_returns_correct_shape(image_height, image_width, model_index,
 160                                         expected_feature_map_shape)
 161 
 162   def test_hyperparam_override(self):
 163     for model_name in _RESNET_MODELS:
 164       model = model_name(
 165           batchnorm_training=True,
 166           default_batchnorm_momentum=0.2,
 167           default_batchnorm_epsilon=0.1,
 168           weights=None,
 169           include_top=False)
 170       bn_layer = model.get_layer(name='conv1_bn')
 171       self.assertAllClose(bn_layer.momentum, 0.2)
 172       self.assertAllClose(bn_layer.epsilon, 0.1)
 173 
 174   def test_variable_count(self):
 175     # The number of variables from slim resnetv1-* model.
 176     variable_nums = [265, 520, 775]
 177     for model_index, var_num in enumerate(variable_nums):
 178       variables = self._get_variables(model_index)
 179       self.assertEqual(len(variables), var_num)
