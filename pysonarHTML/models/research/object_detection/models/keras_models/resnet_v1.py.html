  28 def _fixed_padding(inputs, kernel_size, rate=1):  # pylint: disable=invalid-name
  29   """Pads the input along the spatial dimensions independently of input size.
  30 
  31   Pads the input such that if it was used in a convolution with 'VALID' padding,
  32   the output would have the same dimensions as if the unpadded input was used
  33   in a convolution with 'SAME' padding.
  34 
  35   Args:
  36     inputs: A tensor of size [batch, height_in, width_in, channels].
  37     kernel_size: The kernel to be used in the conv2d or max_pool2d operation.
  38     rate: An integer, rate for atrous convolution.
  39 
  40   Returns:
  41     output: A tensor of size [batch, height_out, width_out, channels] with the
  42       input, either intact (if kernel_size == 1) or padded (if kernel_size > 1).
  43   """
  44   kernel_size_effective = kernel_size + (kernel_size - 1) * (rate - 1)
  45   pad_total = kernel_size_effective - 1
  46   pad_beg = pad_total // 2
  47   pad_end = pad_total - pad_beg
  48   padded_inputs = tf.pad(
  49       inputs, [[0, 0], [pad_beg, pad_end], [pad_beg, pad_end], [0, 0]])
  50   return padded_inputs

  53 class _LayersOverride(object):
  54   """Alternative Keras layers interface for the Keras Resnet V1."""
  55 
  56   def __init__(self,
  57                batchnorm_training,
  58                batchnorm_scale=True,
  59                default_batchnorm_momentum=0.997,
  60                default_batchnorm_epsilon=1e-5,
  61                weight_decay=0.0001,
  62                conv_hyperparams=None,
  63                min_depth=8,
  64                depth_multiplier=1):
  65     """Alternative tf.keras.layers interface, for use by the Keras Resnet V1.
  66 
  67     The class is used by the Keras applications kwargs injection API to
  68     modify the Resnet V1 Keras application with changes required by
  69     the Object Detection API.
  70 
  71     Args:
  72       batchnorm_training: Bool. Assigned to Batch norm layer `training` param
  73         when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
  74       batchnorm_scale: If True, uses an explicit `gamma` multiplier to scale
  75         the activations in the batch normalization layer.
  76       default_batchnorm_momentum: Float. When 'conv_hyperparams' is None,
  77         batch norm layers will be constructed using this value as the momentum.
  78       default_batchnorm_epsilon: Float. When 'conv_hyperparams' is None,
  79         batch norm layers will be constructed using this value as the epsilon.
  80       weight_decay: The weight decay to use for regularizing the model.
  81       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  82         containing hyperparameters for convolution ops. Optionally set to `None`
  83         to use default resnet_v1 layer builders.
  84       min_depth: Minimum number of filters in the convolutional layers.
  85       depth_multiplier: The depth multiplier to modify the number of filters
  86         in the convolutional layers.
  87     """
  88     self._batchnorm_training = batchnorm_training
  89     self._batchnorm_scale = batchnorm_scale
  90     self._default_batchnorm_momentum = default_batchnorm_momentum
  91     self._default_batchnorm_epsilon = default_batchnorm_epsilon
  92     self._conv_hyperparams = conv_hyperparams
  93     self._min_depth = min_depth
  94     self._depth_multiplier = depth_multiplier
  95     self.regularizer = tf.keras.regularizers.l2(weight_decay)
  96     self.initializer = tf.variance_scaling_initializer()
  97 
  98   def _FixedPaddingLayer(self, kernel_size, rate=1):
  99     return tf.keras.layers.Lambda(
 100         lambda x: _fixed_padding(x, kernel_size, rate))
 101 
 102   def Conv2D(self, filters, kernel_size, **kwargs):
 103     """Builds a Conv2D layer according to the current Object Detection config.
 104 
 105     Overrides the Keras Resnet application's convolutions with ones that
 106     follow the spec specified by the Object Detection hyperparameters.
 107 
 108     Args:
 109       filters: The number of filters to use for the convolution.
 110       kernel_size: The kernel size to specify the height and width of the 2D
 111         convolution window.
 112       **kwargs: Keyword args specified by the Keras application for
 113         constructing the convolution.
 114 
 115     Returns:
 116       A one-arg callable that will either directly apply a Keras Conv2D layer to
 117       the input argument, or that will first pad the input then apply a Conv2D
 118       layer.
 119     """
 120     # Apply the minimum depth to the convolution layers.
 121     filters = max(int(filters * self._depth_multiplier), self._min_depth)
 122 
 123     if self._conv_hyperparams:
 124       kwargs = self._conv_hyperparams.params(**kwargs)
 125     else:
 126       kwargs['kernel_regularizer'] = self.regularizer
 127       kwargs['kernel_initializer'] = self.initializer
 128 
 129     # Set use_bias as false to keep it consistent with Slim Resnet model.
 130     kwargs['use_bias'] = False
 131 
 132     kwargs['padding'] = 'same'
 133     stride = kwargs.get('strides')
 134     if stride and kernel_size and stride > 1 and kernel_size > 1:
 135       kwargs['padding'] = 'valid'
 136       def padded_conv(features):  # pylint: disable=invalid-name
 137         padded_features = self._FixedPaddingLayer(kernel_size)(features)
 138         return tf.keras.layers.Conv2D(
 139             filters, kernel_size, **kwargs)(padded_features)
 140       return padded_conv
 141     else:
 142       return tf.keras.layers.Conv2D(filters, kernel_size, **kwargs)
 143 
 144   def Activation(self, *args, **kwargs):  # pylint: disable=unused-argument
 145     """Builds an activation layer.
 146 
 147     Overrides the Keras application Activation layer specified by the
 148     Object Detection configuration.
 149 
 150     Args:
 151       *args: Ignored,
 152         required to match the `tf.keras.layers.Activation` interface.
 153       **kwargs: Only the name is used,
 154         required to match `tf.keras.layers.Activation` interface.
 155 
 156     Returns:
 157       An activation layer specified by the Object Detection hyperparameter
 158       configurations.
 159     """
 160     name = kwargs.get('name')
 161     if self._conv_hyperparams:
 162       return self._conv_hyperparams.build_activation_layer(name=name)
 163     else:
 164       return tf.keras.layers.Lambda(tf.nn.relu, name=name)
 165 
 166   def BatchNormalization(self, **kwargs):
 167     """Builds a normalization layer.
 168 
 169     Overrides the Keras application batch norm with the norm specified by the
 170     Object Detection configuration.
 171 
 172     Args:
 173       **kwargs: Only the name is used, all other params ignored.
 174         Required for matching `layers.BatchNormalization` calls in the Keras
 175         application.
 176 
 177     Returns:
 178       A normalization layer specified by the Object Detection hyperparameter
 179       configurations.
 180     """
 181     name = kwargs.get('name')
 182     if self._conv_hyperparams:
 183       return self._conv_hyperparams.build_batch_norm(
 184           training=self._batchnorm_training,
 185           name=name)
 186     else:
 187       kwargs['scale'] = self._batchnorm_scale
 188       kwargs['epsilon'] = self._default_batchnorm_epsilon
 189       return freezable_batch_norm.FreezableBatchNorm(
 190           training=self._batchnorm_training,
 191           momentum=self._default_batchnorm_momentum,
 192           **kwargs)
 193 
 194   def Input(self, shape):
 195     """Builds an Input layer.
 196 
 197     Overrides the Keras application Input layer with one that uses a
 198     tf.placeholder_with_default instead of a tf.placeholder. This is necessary
 199     to ensure the application works when run on a TPU.
 200 
 201     Args:
 202       shape: A tuple of integers representing the shape of the input, which
 203         includes both spatial share and channels, but not the batch size.
 204         Elements of this tuple can be None; 'None' elements represent dimensions
 205         where the shape is not known.
 206 
 207     Returns:
 208       An input layer for the specified shape that internally uses a
 209       placeholder_with_default.
 210     """
 211     default_size = 224
 212     default_batch_size = 1
 213     shape = list(shape)
 214     default_shape = [default_size if dim is None else dim for dim in shape]
 215 
 216     input_tensor = tf.constant(0.0, shape=[default_batch_size] + default_shape)
 217 
 218     placeholder_with_default = tf.placeholder_with_default(
 219         input=input_tensor, shape=[None] + shape)
 220     return model_utils.input_layer(shape, placeholder_with_default)
 221 
 222   def MaxPooling2D(self, pool_size, **kwargs):
 223     """Builds a MaxPooling2D layer with default padding as 'SAME'.
 224 
 225     This is specified by the default resnet arg_scope in slim.
 226 
 227     Args:
 228       pool_size: The pool size specified by the Keras application.
 229       **kwargs: Ignored, required to match the Keras applications usage.
 230 
 231     Returns:
 232       A MaxPooling2D layer with default padding as 'SAME'.
 233     """
 234     kwargs['padding'] = 'same'
 235     return tf.keras.layers.MaxPooling2D(pool_size, **kwargs)
 236 
 237   # Add alias as Keras also has it.
 238   MaxPool2D = MaxPooling2D  # pylint: disable=invalid-name
 239 
 240   def ZeroPadding2D(self, padding, **kwargs):  # pylint: disable=unused-argument
 241     """Replaces explicit padding in the Keras application with a no-op.
 242 
 243     Args:
 244       padding: The padding values for image height and width.
 245       **kwargs: Ignored, required to match the Keras applications usage.
 246 
 247     Returns:
 248       A no-op identity lambda.
 249     """
 250     return lambda x: x
 251 
 252   # Forward all non-overridden methods to the keras layers
 253   def __getattr__(self, item):
 254     return getattr(tf.keras.layers, item)

 258 def resnet_v1_50(batchnorm_training,
 259                  batchnorm_scale=True,
 260                  default_batchnorm_momentum=0.997,
 261                  default_batchnorm_epsilon=1e-5,
 262                  weight_decay=0.0001,
 263                  conv_hyperparams=None,
 264                  min_depth=8,
 265                  depth_multiplier=1,
 266                  **kwargs):
 267   """Instantiates the Resnet50 architecture, modified for object detection.
 268 
 269   Args:
 270     batchnorm_training: Bool. Assigned to Batch norm layer `training` param
 271       when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
 272     batchnorm_scale: If True, uses an explicit `gamma` multiplier to scale
 273       the activations in the batch normalization layer.
 274     default_batchnorm_momentum: Float. When 'conv_hyperparams' is None,
 275       batch norm layers will be constructed using this value as the momentum.
 276     default_batchnorm_epsilon: Float. When 'conv_hyperparams' is None,
 277       batch norm layers will be constructed using this value as the epsilon.
 278     weight_decay: The weight decay to use for regularizing the model.
 279     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 280       containing hyperparameters for convolution ops. Optionally set to `None`
 281       to use default resnet_v1 layer builders.
 282     min_depth: Minimum number of filters in the convolutional layers.
 283     depth_multiplier: The depth multiplier to modify the number of filters
 284       in the convolutional layers.
 285     **kwargs: Keyword arguments forwarded directly to the
 286       `tf.keras.applications.Mobilenet` method that constructs the Keras
 287       model.
 288 
 289   Returns:
 290     A Keras ResnetV1-50 model instance.
 291   """
 292   layers_override = _LayersOverride(
 293       batchnorm_training,
 294       batchnorm_scale=batchnorm_scale,
 295       default_batchnorm_momentum=default_batchnorm_momentum,
 296       default_batchnorm_epsilon=default_batchnorm_epsilon,
 297       conv_hyperparams=conv_hyperparams,
 298       weight_decay=weight_decay,
 299       min_depth=min_depth,
 300       depth_multiplier=depth_multiplier)
 301   return tf.keras.applications.resnet.ResNet50(
 302       layers=layers_override, **kwargs)
