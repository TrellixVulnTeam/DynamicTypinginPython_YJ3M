  17 """Tests for mobilenet_v1.py.
  18 
  19 This test mainly focuses on comparing slim MobilenetV1 and Keras MobilenetV1 for
  20 object detection. To verify the consistency of the two models, we compare:
  21   1. Output shape of each layer given different inputs
  22   2. Number of global variables
  23 
  24 We also visualize the model structure via Tensorboard, and compare the model
  25 layout and the parameters of each Op to make sure the two implementations are
  26 consistent.

  69   def _build_conv_hyperparams(self):
  70     conv_hyperparams = hyperparams_pb2.Hyperparams()
  71     conv_hyperparams_text_proto = """
  72       activation: RELU_6
  73       regularizer {
  74         l2_regularizer {
  75         }
  76       }
  77       initializer {
  78         truncated_normal_initializer {
  79         }
  80       }
  81       batch_norm {
  82         train: true,
  83         scale: false,
  84         center: true,
  85         decay: 0.2,
  86         epsilon: 0.1,
  87       }
  88     """
  89     text_format.Merge(conv_hyperparams_text_proto, conv_hyperparams)
  90     return hyperparams_builder.KerasLayerHyperparams(conv_hyperparams)

  92   def _create_application_with_layer_outputs(
  93       self, layer_names, batchnorm_training,
  94       conv_hyperparams=None,
  95       use_explicit_padding=False,
  96       alpha=1.0,
  97       min_depth=None,
  98       conv_defs=None):
  99     """Constructs Keras MobilenetV1 that extracts intermediate layer outputs."""
 100     if not layer_names:
 101       layer_names = _KERAS_LAYERS_TO_CHECK
 102     full_model = mobilenet_v1.mobilenet_v1(
 103         batchnorm_training=batchnorm_training,
 104         conv_hyperparams=conv_hyperparams,
 105         weights=None,
 106         use_explicit_padding=use_explicit_padding,
 107         alpha=alpha,
 108         min_depth=min_depth,
 109         conv_defs=conv_defs,
 110         include_top=False)
 111     layer_outputs = [full_model.get_layer(name=layer).output
 112                      for layer in layer_names]
 113     return tf.keras.Model(
 114         inputs=full_model.inputs,
 115         outputs=layer_outputs)
 116 
 117   def _check_returns_correct_shape(
 118       self, image_height, image_width, depth_multiplier,
 119       expected_feature_map_shape, use_explicit_padding=False, min_depth=8,
 120       layer_names=None, conv_defs=None):
 121     def graph_fn(image_tensor):
 122       model = self._create_application_with_layer_outputs(
 123           layer_names=layer_names,
 124           batchnorm_training=False,
 125           use_explicit_padding=use_explicit_padding,
 126           min_depth=min_depth,
 127           alpha=depth_multiplier,
 128           conv_defs=conv_defs)
 129       return model(image_tensor)
 130 
 131     image_tensor = np.random.rand(_BATCH_SIZE, image_height, image_width,
 132                                   _NUM_CHANNELS).astype(np.float32)
 133     feature_maps = self.execute(graph_fn, [image_tensor])
 134 
 135     for feature_map, expected_shape in zip(feature_maps,
 136                                            expected_feature_map_shape):
 137       self.assertAllEqual(feature_map.shape, expected_shape)
 138 
 139   def _check_returns_correct_shapes_with_dynamic_inputs(
 140       self, image_height, image_width, depth_multiplier,
 141       expected_feature_map_shape, use_explicit_padding=False, min_depth=8,
 142       layer_names=None):
 143     def graph_fn(image_height, image_width):
 144       image_tensor = tf.random_uniform([_BATCH_SIZE, image_height, image_width,
 145                                         _NUM_CHANNELS], dtype=tf.float32)
 146       model = self._create_application_with_layer_outputs(
 147           layer_names=layer_names,
 148           batchnorm_training=False,
 149           use_explicit_padding=use_explicit_padding,
 150           alpha=depth_multiplier)
 151       return model(image_tensor)
 152 
 153     feature_maps = self.execute_cpu(graph_fn, [
 154         np.array(image_height, dtype=np.int32),
 155         np.array(image_width, dtype=np.int32)
 156     ])
 157 
 158     for feature_map, expected_shape in zip(feature_maps,
 159                                            expected_feature_map_shape):
 160       self.assertAllEqual(feature_map.shape, expected_shape)
 161 
 162   def _get_variables(self, depth_multiplier, layer_names=None):
 163     g = tf.Graph()
 164     with g.as_default():
 165       preprocessed_inputs = tf.placeholder(
 166           tf.float32, (4, None, None, _NUM_CHANNELS))
 167       model = self._create_application_with_layer_outputs(
 168           layer_names=layer_names,
 169           batchnorm_training=False, use_explicit_padding=False,
 170           alpha=depth_multiplier)
 171       model(preprocessed_inputs)
 172       return g.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)

 240   def test_hyperparam_override(self):
 241     hyperparams = self._build_conv_hyperparams()
 242     model = mobilenet_v1.mobilenet_v1(
 243         batchnorm_training=True,
 244         conv_hyperparams=hyperparams,
 245         weights=None,
 246         use_explicit_padding=False,
 247         alpha=1.0,
 248         min_depth=32,
 249         include_top=False)
 250     hyperparams.params()
 251     bn_layer = model.get_layer(name='conv_pw_5_bn')
 252     self.assertAllClose(bn_layer.momentum, 0.2)
 253     self.assertAllClose(bn_layer.epsilon, 0.1)
 254 
 255   def test_variable_count(self):

 257     variables = self._get_variables(depth_multiplier)

 259     self.assertEqual(len(variables), 135)
