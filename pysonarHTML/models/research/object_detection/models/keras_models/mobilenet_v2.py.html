  30 def _make_divisible(v, divisor, min_value=None):
  31   if min_value is None:
  32     min_value = divisor
  33   new_v = max(min_value, int(v + divisor / 2) // divisor * divisor)
  34   # Make sure that round down does not go down by more than 10%.
  35   if new_v < 0.9 * v:
  36     new_v += divisor
  37   return new_v

  40 class _LayersOverride(object):
  41   """Alternative Keras layers interface for the Keras MobileNetV2."""
  42 
  43   def __init__(self,
  44                batchnorm_training,
  45                default_batchnorm_momentum=0.999,
  46                conv_hyperparams=None,
  47                use_explicit_padding=False,
  48                alpha=1.0,
  49                min_depth=None,
  50                conv_defs=None):
  51     """Alternative tf.keras.layers interface, for use by the Keras MobileNetV2.
  52 
  53     It is used by the Keras applications kwargs injection API to
  54     modify the Mobilenet v2 Keras application with changes required by
  55     the Object Detection API.
  56 
  57     These injected interfaces make the following changes to the network:
  58 
  59     - Applies the Object Detection hyperparameter configuration
  60     - Supports FreezableBatchNorms
  61     - Adds support for a min number of filters for each layer
  62     - Makes the `alpha` parameter affect the final convolution block even if it
  63         is less than 1.0
  64     - Adds support for explicit padding of convolutions
  65 
  66     Args:
  67       batchnorm_training: Bool. Assigned to Batch norm layer `training` param
  68         when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
  69       default_batchnorm_momentum: Float. When 'conv_hyperparams' is None,
  70         batch norm layers will be constructed using this value as the momentum.
  71       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  72         containing hyperparameters for convolution ops. Optionally set to `None`
  73         to use default mobilenet_v2 layer builders.
  74       use_explicit_padding: If True, use 'valid' padding for convolutions,
  75         but explicitly pre-pads inputs so that the output dimensions are the
  76         same as if 'same' padding were used. Off by default.
  77       alpha: The width multiplier referenced in the MobileNetV2 paper. It
  78         modifies the number of filters in each convolutional layer.
  79       min_depth: Minimum number of filters in the convolutional layers.
  80       conv_defs: Network layout to specify the mobilenet_v2 body. Default is
  81         `None` to use the default mobilenet_v2 network layout.
  82     """
  83     self._alpha = alpha
  84     self._batchnorm_training = batchnorm_training
  85     self._default_batchnorm_momentum = default_batchnorm_momentum
  86     self._conv_hyperparams = conv_hyperparams
  87     self._use_explicit_padding = use_explicit_padding
  88     self._min_depth = min_depth
  89     self._conv_defs = conv_defs
  90     self.regularizer = tf.keras.regularizers.l2(0.00004 * 0.5)
  91     self.initializer = tf.truncated_normal_initializer(stddev=0.09)
  92 
  93   def _FixedPaddingLayer(self, kernel_size):
  94     return tf.keras.layers.Lambda(lambda x: ops.fixed_padding(x, kernel_size))
  95 
  96   def Conv2D(self, filters, **kwargs):
  97     """Builds a Conv2D layer according to the current Object Detection config.
  98 
  99     Overrides the Keras MobileNetV2 application's convolutions with ones that
 100     follow the spec specified by the Object Detection hyperparameters.
 101 
 102     Args:
 103       filters: The number of filters to use for the convolution.
 104       **kwargs: Keyword args specified by the Keras application for
 105         constructing the convolution.
 106 
 107     Returns:
 108       A one-arg callable that will either directly apply a Keras Conv2D layer to
 109       the input argument, or that will first pad the input then apply a Conv2D
 110       layer.
 111     """
 112     # Make sure 'alpha' is always applied to the last convolution block's size
 113     # (This overrides the Keras application's functionality)
 114     layer_name = kwargs.get('name')
 115     if layer_name == 'Conv_1':
 116       if self._conv_defs:
 117         filters = model_utils.get_conv_def(self._conv_defs, 'Conv_1')
 118       else:
 119         filters = 1280
 120       if self._alpha < 1.0:
 121         filters = _make_divisible(filters * self._alpha, 8)
 122 
 123     # Apply the minimum depth to the convolution layers
 124     if (self._min_depth and (filters < self._min_depth)
 125         and not kwargs.get('name').endswith('expand')):
 126       filters = self._min_depth
 127 
 128     if self._conv_hyperparams:
 129       kwargs = self._conv_hyperparams.params(**kwargs)
 130     else:
 131       kwargs['kernel_regularizer'] = self.regularizer
 132       kwargs['kernel_initializer'] = self.initializer
 133 
 134     kwargs['padding'] = 'same'
 135     kernel_size = kwargs.get('kernel_size')
 136     if self._use_explicit_padding and kernel_size > 1:
 137       kwargs['padding'] = 'valid'
 138       def padded_conv(features):
 139         padded_features = self._FixedPaddingLayer(kernel_size)(features)
 140         return tf.keras.layers.Conv2D(filters, **kwargs)(padded_features)
 141 
 142       return padded_conv
 143     else:
 144       return tf.keras.layers.Conv2D(filters, **kwargs)
 145 
 146   def DepthwiseConv2D(self, **kwargs):
 147     """Builds a DepthwiseConv2D according to the Object Detection config.
 148 
 149     Overrides the Keras MobileNetV2 application's convolutions with ones that
 150     follow the spec specified by the Object Detection hyperparameters.
 151 
 152     Args:
 153       **kwargs: Keyword args specified by the Keras application for
 154         constructing the convolution.
 155 
 156     Returns:
 157       A one-arg callable that will either directly apply a Keras DepthwiseConv2D
 158       layer to the input argument, or that will first pad the input then apply
 159       the depthwise convolution.
 160     """
 161     if self._conv_hyperparams:
 162       kwargs = self._conv_hyperparams.params(**kwargs)
 163       # Both the regularizer and initializer apply to the depthwise layer in
 164       # MobilenetV1, so we remap the kernel_* to depthwise_* here.
 165       kwargs['depthwise_regularizer'] = kwargs['kernel_regularizer']
 166       kwargs['depthwise_initializer'] = kwargs['kernel_initializer']
 167     else:
 168       kwargs['depthwise_regularizer'] = self.regularizer
 169       kwargs['depthwise_initializer'] = self.initializer
 170 
 171     kwargs['padding'] = 'same'
 172     kernel_size = kwargs.get('kernel_size')
 173     if self._use_explicit_padding and kernel_size > 1:
 174       kwargs['padding'] = 'valid'
 175       def padded_depthwise_conv(features):
 176         padded_features = self._FixedPaddingLayer(kernel_size)(features)
 177         return tf.keras.layers.DepthwiseConv2D(**kwargs)(padded_features)
 178 
 179       return padded_depthwise_conv
 180     else:
 181       return tf.keras.layers.DepthwiseConv2D(**kwargs)

 183   def BatchNormalization(self, **kwargs):
 184     """Builds a normalization layer.
 185 
 186     Overrides the Keras application batch norm with the norm specified by the
 187     Object Detection configuration.
 188 
 189     Args:
 190       **kwargs: Only the name is used, all other params ignored.
 191         Required for matching `layers.BatchNormalization` calls in the Keras
 192         application.
 193 
 194     Returns:
 195       A normalization layer specified by the Object Detection hyperparameter
 196       configurations.
 197     """
 198     name = kwargs.get('name')
 199     if self._conv_hyperparams:
 200       return self._conv_hyperparams.build_batch_norm(
 201           training=self._batchnorm_training,
 202           name=name)
 203     else:
 204       return freezable_batch_norm.FreezableBatchNorm(
 205           training=self._batchnorm_training,
 206           epsilon=1e-3,
 207           momentum=self._default_batchnorm_momentum,
 208           name=name)

 210   def Input(self, shape):
 211     """Builds an Input layer.
 212 
 213     Overrides the Keras application Input layer with one that uses a
 214     tf.placeholder_with_default instead of a tf.placeholder. This is necessary
 215     to ensure the application works when run on a TPU.
 216 
 217     Args:
 218       shape: The shape for the input layer to use. (Does not include a dimension
 219         for the batch size).
 220     Returns:
 221       An input layer for the specified shape that internally uses a
 222       placeholder_with_default.
 223     """
 224     default_size = 224
 225     default_batch_size = 1
 226     shape = list(shape)
 227     default_shape = [default_size if dim is None else dim for dim in shape]
 228 
 229     input_tensor = tf.constant(0.0, shape=[default_batch_size] + default_shape)
 230 
 231     placeholder_with_default = tf.placeholder_with_default(
 232         input=input_tensor, shape=[None] + shape)
 233     return model_utils.input_layer(shape, placeholder_with_default)
 234

 236   def ReLU(self, *args, **kwargs):
 237     """Builds an activation layer.
 238 
 239     Overrides the Keras application ReLU with the activation specified by the
 240     Object Detection configuration.
 241 
 242     Args:
 243       *args: Ignored, required to match the `tf.keras.ReLU` interface
 244       **kwargs: Only the name is used,
 245         required to match `tf.keras.ReLU` interface
 246 
 247     Returns:
 248       An activation layer specified by the Object Detection hyperparameter
 249       configurations.
 250     """
 251     name = kwargs.get('name')
 252     if self._conv_hyperparams:
 253       return self._conv_hyperparams.build_activation_layer(name=name)
 254     else:
 255       return tf.keras.layers.Lambda(tf.nn.relu6, name=name)

 262     Args:
 263       **kwargs: Ignored, required to match the Keras applications usage.
 264 
 265     Returns:
 266       A no-op identity lambda.

 268     return lambda x: x

 276 def mobilenet_v2(batchnorm_training,
 277                  default_batchnorm_momentum=0.9997,
 278                  conv_hyperparams=None,
 279                  use_explicit_padding=False,
 280                  alpha=1.0,
 281                  min_depth=None,
 282                  conv_defs=None,
 283                  **kwargs):
 284   """Instantiates the MobileNetV2 architecture, modified for object detection.
 285 
 286   This wraps the MobileNetV2 tensorflow Keras application, but uses the
 287   Keras application's kwargs-based monkey-patching API to override the Keras
 288   architecture with the following changes:
 289 
 290   - Changes the default batchnorm momentum to 0.9997
 291   - Applies the Object Detection hyperparameter configuration
 292   - Supports FreezableBatchNorms
 293   - Adds support for a min number of filters for each layer
 294   - Makes the `alpha` parameter affect the final convolution block even if it
 295       is less than 1.0
 296   - Adds support for explicit padding of convolutions
 297   - Makes the Input layer use a tf.placeholder_with_default instead of a
 298       tf.placeholder, to work on TPUs.
 299 
 300   Args:
 301       batchnorm_training: Bool. Assigned to Batch norm layer `training` param
 302         when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
 303       default_batchnorm_momentum: Float. When 'conv_hyperparams' is None,
 304         batch norm layers will be constructed using this value as the momentum.
 305       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 306         containing hyperparameters for convolution ops. Optionally set to `None`
 307         to use default mobilenet_v2 layer builders.
 308       use_explicit_padding: If True, use 'valid' padding for convolutions,
 309         but explicitly pre-pads inputs so that the output dimensions are the
 310         same as if 'same' padding were used. Off by default.
 311       alpha: The width multiplier referenced in the MobileNetV2 paper. It
 312         modifies the number of filters in each convolutional layer.
 313       min_depth: Minimum number of filters in the convolutional layers.
 314       conv_defs: Network layout to specify the mobilenet_v2 body. Default is
 315         `None` to use the default mobilenet_v2 network layout.
 316       **kwargs: Keyword arguments forwarded directly to the
 317         `tf.keras.applications.MobilenetV2` method that constructs the Keras
 318         model.
 319 
 320   Returns:
 321       A Keras model instance.
 322   """
 323   layers_override = _LayersOverride(
 324       batchnorm_training,
 325       default_batchnorm_momentum=default_batchnorm_momentum,
 326       conv_hyperparams=conv_hyperparams,
 327       use_explicit_padding=use_explicit_padding,
 328       min_depth=min_depth,
 329       alpha=alpha,
 330       conv_defs=conv_defs)
 331   return tf.keras.applications.MobileNetV2(alpha=alpha,
 332                                            layers=layers_override,
 333                                            **kwargs)
