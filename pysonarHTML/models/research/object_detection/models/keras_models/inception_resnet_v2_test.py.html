  18 
  19 This test mainly focuses on comparing slim inception resnet v2 and Keras
  20 inception resnet v2 for object detection. To verify the consistency of the two
  21 models, we compare:
  22   1. Output shape of each layer given different inputs
  23   2. Number of global variables
  24 
  25 We also visualize the model structure via Tensorboard, and compare the model
  26 layout and the parameters of each Op to make sure the two implementations are
  27 consistent.

 105   def _create_application_with_layer_outputs(
 106       self, layer_names, batchnorm_training,
 107       output_stride=16,
 108       align_feature_maps=False,
 109       batchnorm_scale=False,
 110       weight_decay=0.00004,
 111       default_batchnorm_momentum=0.9997,
 112       default_batchnorm_epsilon=0.001,):
 113     """Constructs Keras inception_resnet_v2 that extracts layer outputs."""
 114     # Have to clear the Keras backend to ensure isolation in layer naming
 115     tf.keras.backend.clear_session()
 116     if not layer_names:
 117       layer_names = _KERAS_LAYERS_TO_CHECK
 118     full_model = inception_resnet_v2.inception_resnet_v2(
 119         batchnorm_training=batchnorm_training,
 120         output_stride=output_stride,
 121         align_feature_maps=align_feature_maps,
 122         weights=None,
 123         batchnorm_scale=batchnorm_scale,
 124         weight_decay=weight_decay,
 125         default_batchnorm_momentum=default_batchnorm_momentum,
 126         default_batchnorm_epsilon=default_batchnorm_epsilon,
 127         include_top=False)
 128     layer_outputs = [full_model.get_layer(name=layer).output
 129                      for layer in layer_names]
 130     return tf.keras.Model(
 131         inputs=full_model.inputs,
 132         outputs=layer_outputs)
 133 
 134   def _check_returns_correct_shape(
 135       self, image_height, image_width,
 136       expected_feature_map_shape, layer_names=None, batchnorm_training=True,
 137       output_stride=16,
 138       align_feature_maps=False,
 139       batchnorm_scale=False,
 140       weight_decay=0.00004,
 141       default_batchnorm_momentum=0.9997,
 142       default_batchnorm_epsilon=0.001,):
 143     if not layer_names:
 144       layer_names = _KERAS_LAYERS_TO_CHECK
 145     model = self._create_application_with_layer_outputs(
 146         layer_names=layer_names,
 147         batchnorm_training=batchnorm_training,
 148         output_stride=output_stride,
 149         align_feature_maps=align_feature_maps,
 150         batchnorm_scale=batchnorm_scale,
 151         weight_decay=weight_decay,
 152         default_batchnorm_momentum=default_batchnorm_momentum,
 153         default_batchnorm_epsilon=default_batchnorm_epsilon)
 154 
 155     image_tensor = np.random.rand(_BATCH_SIZE, image_height, image_width,
 156                                   _NUM_CHANNELS).astype(np.float32)
 157     feature_maps = model(image_tensor)
 158 
 159     for feature_map, layer_name in zip(feature_maps, layer_names):
 160       endpoint_name = _KERAS_TO_SLIM_ENDPOINT_NAMES[layer_name]
 161       expected_shape = expected_feature_map_shape[endpoint_name]
 162       self.assertAllEqual(feature_map.shape, expected_shape)
 163 
 164   def _get_variables(self, layer_names=None):
 165     tf.keras.backend.clear_session()
 166     model = self._create_application_with_layer_outputs(
 167         layer_names=layer_names,
 168         batchnorm_training=False)
 169     preprocessed_inputs = tf.placeholder(
 170         tf.float32, (4, None, None, _NUM_CHANNELS))
 171     model(preprocessed_inputs)
 172     return model.variables
 173 
 174   def test_returns_correct_shapes_128(self):
 175     image_height = 128
 176     image_width = 128
 177     expected_feature_map_shape = (
 178         _SLIM_ENDPOINT_SHAPES_128)
 179     self._check_returns_correct_shape(
 180         image_height, image_width, expected_feature_map_shape,
 181         align_feature_maps=True)
 182 
 183   def test_returns_correct_shapes_128_output_stride_8(self):
 184     image_height = 128
 185     image_width = 128
 186     expected_feature_map_shape = (
 187         _SLIM_ENDPOINT_SHAPES_128_STRIDE_8)
 188 
 189     # Output stride of 8 not defined beyond 'block17_20_ac', which is
 190     # PreAuxLogits in slim. So, we exclude those layers in our Keras vs Slim
 191     # comparison.
 192     excluded_layers = {'mixed_7a', 'conv_7b_ac'}
 193     layer_names = [l for l in _KERAS_LAYERS_TO_CHECK
 194                    if l not in excluded_layers]
 195     self._check_returns_correct_shape(
 196         image_height, image_width, expected_feature_map_shape,
 197         layer_names=layer_names, output_stride=8, align_feature_maps=True)
 198 
 199   def test_returns_correct_shapes_128_align_feature_maps_false(
 200       self):
 201     image_height = 128
 202     image_width = 128
 203     expected_feature_map_shape = (
 204         _SLIM_ENDPOINT_SHAPES_128_ALIGN_FEATURE_MAPS_FALSE)
 205     self._check_returns_correct_shape(
 206         image_height, image_width, expected_feature_map_shape,
 207         align_feature_maps=False)

 209   def test_hyperparam_override(self):
 210     model = inception_resnet_v2.inception_resnet_v2(
 211         batchnorm_training=True,
 212         default_batchnorm_momentum=0.2,
 213         default_batchnorm_epsilon=0.1,
 214         weights=None,
 215         include_top=False)
 216     bn_layer = model.get_layer(name='freezable_batch_norm')
 217     self.assertAllClose(bn_layer.momentum, 0.2)
 218     self.assertAllClose(bn_layer.epsilon, 0.1)
 219 
 220   def test_variable_count(self):
 221     variables = self._get_variables()
 222     # 896 is the number of variables from slim inception resnet v2 model.
 223     self.assertEqual(len(variables), 896)
