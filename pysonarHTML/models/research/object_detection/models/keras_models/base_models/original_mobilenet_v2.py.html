  16 """MobileNet v2 models for Keras.
  17 
  18 MobileNetV2 is a general architecture and can be used for multiple use cases.
  19 Depending on the use case, it can use different input layer size and
  20 different width factors. This allows different width models to reduce
  21 the number of multiply-adds and thereby
  22 reduce inference cost on mobile devices.
  23 
  24 MobileNetV2 is very similar to the original MobileNet,
  25 except that it uses inverted residual blocks with
  26 bottlenecking features. It has a drastically lower
  27 parameter count than the original MobileNet.
  28 MobileNets support any input size greater
  29 than 32 x 32, with larger image sizes
  30 offering better performance.
  31 
  32 The number of parameters and number of multiply-adds
  33 can be modified by using the `alpha` parameter,
  34 which increases/decreases the number of filters in each layer.
  35 By altering the image size and `alpha` parameter,
  36 all 22 models from the paper can be built, with ImageNet weights provided.
  37 
  38 The paper demonstrates the performance of MobileNets using `alpha` values of
  39 1.0 (also called 100 % MobileNet), 0.35, 0.5, 0.75, 1.0, 1.3, and 1.4
  40 
  41 For each of these `alpha` values, weights for 5 different input image sizes
  42 are provided (224, 192, 160, 128, and 96).

 112 def _obtain_input_shape(
 113     input_shape,
 114     default_size,
 115     min_size,
 116     data_format,
 117     require_flatten):
 118   """Internal utility to compute/validate an ImageNet model's input shape.
 119 
 120   Arguments:
 121       input_shape: either None (will return the default network input shape),
 122           or a user-provided shape to be validated.
 123       default_size: default input width/height for the model.
 124       min_size: minimum input width/height accepted by the model.
 125       data_format: image data format to use.
 126       require_flatten: whether the model is expected to
 127           be linked to a classifier via a Flatten layer.
 128 
 129   Returns:
 130       An integer shape tuple (may include None entries).
 131 
 132   Raises:
 133       ValueError: in case of invalid argument values.
 134   """
 135   if input_shape and len(input_shape) == 3:
 136     if data_format == 'channels_first':
 137       if input_shape[0] not in {1, 3}:
 138         warnings.warn(
 139             'This model usually expects 1 or 3 input channels. '
 140             'However, it was passed an input_shape with ' +
 141             str(input_shape[0]) + ' input channels.')
 142       default_shape = (input_shape[0], default_size, default_size)
 143     else:
 144       if input_shape[-1] not in {1, 3}:
 145         warnings.warn(
 146             'This model usually expects 1 or 3 input channels. '
 147             'However, it was passed an input_shape with ' +
 148             str(input_shape[-1]) + ' input channels.')
 149       default_shape = (default_size, default_size, input_shape[-1])
 150   else:
 151     if data_format == 'channels_first':
 152       default_shape = (3, default_size, default_size)
 153     else:
 154       default_shape = (default_size, default_size, 3)
 155   if input_shape:
 156     if data_format == 'channels_first':
 157       if input_shape is not None:
 158         if len(input_shape) != 3:
 159           raise ValueError(
 160               '`input_shape` must be a tuple of three integers.')
 161         if ((input_shape[1] is not None and input_shape[1] < min_size) or
 162             (input_shape[2] is not None and input_shape[2] < min_size)):
 163           raise ValueError('Input size must be at least ' +
 164                            str(min_size) + 'x' + str(min_size) +
 165                            '; got `input_shape=' +
 166                            str(input_shape) + '`')
 167     else:
 168       if input_shape is not None:
 169         if len(input_shape) != 3:
 170           raise ValueError(
 171               '`input_shape` must be a tuple of three integers.')
 172         if ((input_shape[0] is not None and input_shape[0] < min_size) or
 173             (input_shape[1] is not None and input_shape[1] < min_size)):
 174           raise ValueError('Input size must be at least ' +
 175                            str(min_size) + 'x' + str(min_size) +
 176                            '; got `input_shape=' +
 177                            str(input_shape) + '`')
 178   else:
 179     if require_flatten:
 180       input_shape = default_shape
 181     else:
 182       if data_format == 'channels_first':
 183         input_shape = (3, None, None)
 184       else:
 185         input_shape = (None, None, 3)
 186   if require_flatten:
 187     if None in input_shape:
 188       raise ValueError('If `include_top` is True, '
 189                        'you should specify a static `input_shape`. '
 190                        'Got `input_shape=' + str(input_shape) + '`')
 191   return input_shape

 194 def preprocess_input(x):
 195   """Preprocesses a numpy array encoding a batch of images.
 196 
 197   This function applies the "Inception" preprocessing which converts
 198   the RGB values from [0, 255] to [-1, 1]. Note that this preprocessing
 199   function is different from `imagenet_utils.preprocess_input()`.
 200 
 201   Arguments:
 202     x: a 4D numpy array consists of RGB values within [0, 255].
 203 
 204   Returns:
 205     Preprocessed array.
 206   """
 207   x /= 128.
 208   x -= 1.
 209   return x.astype(np.float32)

 212 # This function is taken from the original tf repo.
 213 # It ensures that all layers have a channel number that is divisible by 8
 214 # It can be seen here:
 215 # https://github.com/tensorflow/models/blob/master/research/slim/nets/mobilenet/mobilenet.py
 216 
 217 
 218 def _make_divisible(v, divisor, min_value=None):
 219   if min_value is None:
 220     min_value = divisor
 221   new_v = max(min_value, int(v + divisor / 2) // divisor * divisor)
 222   # Make sure that round down does not go down by more than 10%.
 223   if new_v < 0.9 * v:
 224     new_v += divisor
 225   return new_v

 228 def mobilenet_v2(input_shape=None,
 229                  alpha=1.0,
 230                  include_top=True,
 231                  classes=1000):
 232   """Instantiates the MobileNetV2 architecture.
 233 
 234   To load a MobileNetV2 model via `load_model`, import the custom
 235   objects `relu6` and pass them to the `custom_objects` parameter.
 236   E.g.
 237   model = load_model('mobilenet.h5', custom_objects={
 238                      'relu6': mobilenet.relu6})
 239 
 240   Arguments:
 241     input_shape: optional shape tuple, to be specified if you would
 242       like to use a model with an input img resolution that is not
 243       (224, 224, 3).
 244       It should have exactly 3 inputs channels (224, 224, 3).
 245       You can also omit this option if you would like
 246       to infer input_shape from an input_tensor.
 247       If you choose to include both input_tensor and input_shape then
 248       input_shape will be used if they match, if the shapes
 249       do not match then we will throw an error.
 250       E.g. `(160, 160, 3)` would be one valid value.
 251     alpha: controls the width of the network. This is known as the
 252     width multiplier in the MobileNetV2 paper.
 253       - If `alpha` < 1.0, proportionally decreases the number
 254           of filters in each layer.
 255       - If `alpha` > 1.0, proportionally increases the number
 256           of filters in each layer.
 257       - If `alpha` = 1, default number of filters from the paper
 258            are used at each layer.
 259     include_top: whether to include the fully-connected
 260       layer at the top of the network.
 261     classes: optional number of classes to classify images
 262       into, only to be specified if `include_top` is True, and
 263       if no `weights` argument is specified.
 264 
 265   Returns:
 266     A Keras model instance.
 267 
 268   Raises:
 269     ValueError: in case of invalid argument for `weights`,
 270         or invalid input shape or invalid depth_multiplier, alpha,
 271         rows when weights='imagenet'
 272   """
 273 
 274   # Determine proper input shape and default size.
 275   # If input_shape is None and no input_tensor
 276   if input_shape is None:
 277     default_size = 224
 278 
 279   # If input_shape is not None, assume default size
 280   else:
 281     if K.image_data_format() == 'channels_first':
 282       rows = input_shape[1]
 283       cols = input_shape[2]
 284     else:
 285       rows = input_shape[0]
 286       cols = input_shape[1]
 287 
 288     if rows == cols and rows in [96, 128, 160, 192, 224]:
 289       default_size = rows
 290     else:
 291       default_size = 224
 292 
 293   input_shape = _obtain_input_shape(input_shape,
 294                                     default_size=default_size,
 295                                     min_size=32,
 296                                     data_format=K.image_data_format(),
 297                                     require_flatten=include_top)
 298 
 299   if K.image_data_format() == 'channels_last':
 300     row_axis, col_axis = (0, 1)
 301   else:
 302     row_axis, col_axis = (1, 2)
 303   rows = input_shape[row_axis]
 304   cols = input_shape[col_axis]
 305 
 306   if K.image_data_format() != 'channels_last':
 307     warnings.warn('The MobileNet family of models is only available '
 308                   'for the input data format "channels_last" '
 309                   '(width, height, channels). '
 310                   'However your settings specify the default '
 311                   'data format "channels_first" (channels, width, height).'
 312                   ' You should set `image_data_format="channels_last"` '
 313                   'in your Keras config located at ~/.keras/keras.json. '
 314                   'The model being returned right now will expect inputs '
 315                   'to follow the "channels_last" data format.')
 316     K.set_image_data_format('channels_last')
 317     old_data_format = 'channels_first'
 318   else:
 319     old_data_format = None
 320 
 321   img_input = Input(shape=input_shape)
 322 
 323   first_block_filters = _make_divisible(32 * alpha, 8)
 324   x = Conv2D(first_block_filters,
 325              kernel_size=3,
 326              strides=(2, 2), padding='same',
 327              use_bias=False, name='Conv1')(img_input)
 328   x = BatchNormalization(epsilon=1e-3, momentum=0.999, name='bn_Conv1')(x)
 329   x = Activation(relu6, name='Conv1_relu')(x)
 330 
 331   x = _first_inverted_res_block(x,
 332                                 filters=16,
 333                                 alpha=alpha,
 334                                 stride=1,
 335                                 block_id=0)
 336 
 337   x = _inverted_res_block(x, filters=24, alpha=alpha, stride=2,
 338                           expansion=6, block_id=1)
 339   x = _inverted_res_block(x, filters=24, alpha=alpha, stride=1,
 340                           expansion=6, block_id=2)
 341 
 342   x = _inverted_res_block(x, filters=32, alpha=alpha, stride=2,
 343                           expansion=6, block_id=3)
 344   x = _inverted_res_block(x, filters=32, alpha=alpha, stride=1,
 345                           expansion=6, block_id=4)
 346   x = _inverted_res_block(x, filters=32, alpha=alpha, stride=1,
 347                           expansion=6, block_id=5)
 348 
 349   x = _inverted_res_block(x, filters=64, alpha=alpha, stride=2,
 350                           expansion=6, block_id=6)
 351   x = _inverted_res_block(x, filters=64, alpha=alpha, stride=1,
 352                           expansion=6, block_id=7)
 353   x = _inverted_res_block(x, filters=64, alpha=alpha, stride=1,
 354                           expansion=6, block_id=8)
 355   x = _inverted_res_block(x, filters=64, alpha=alpha, stride=1,
 356                           expansion=6, block_id=9)
 357 
 358   x = _inverted_res_block(x, filters=96, alpha=alpha, stride=1,
 359                           expansion=6, block_id=10)
 360   x = _inverted_res_block(x, filters=96, alpha=alpha, stride=1,
 361                           expansion=6, block_id=11)
 362   x = _inverted_res_block(x, filters=96, alpha=alpha, stride=1,
 363                           expansion=6, block_id=12)
 364 
 365   x = _inverted_res_block(x, filters=160, alpha=alpha, stride=2,
 366                           expansion=6, block_id=13)
 367   x = _inverted_res_block(x, filters=160, alpha=alpha, stride=1,
 368                           expansion=6, block_id=14)
 369   x = _inverted_res_block(x, filters=160, alpha=alpha, stride=1,
 370                           expansion=6, block_id=15)
 371 
 372   x = _inverted_res_block(x, filters=320, alpha=alpha, stride=1,
 373                           expansion=6, block_id=16)
 374 
 375   # no alpha applied to last conv as stated in the paper:
 376   # if the width multiplier is greater than 1 we
 377   # increase the number of output channels
 378   if alpha > 1.0:
 379     last_block_filters = _make_divisible(1280 * alpha, 8)
 380   else:
 381     last_block_filters = 1280
 382 
 383   x = Conv2D(last_block_filters,
 384              kernel_size=1,
 385              use_bias=False,
 386              name='Conv_1')(x)
 387   x = BatchNormalization(epsilon=1e-3, momentum=0.999, name='Conv_1_bn')(x)
 388   x = Activation(relu6, name='out_relu')(x)
 389 
 390   if include_top:
 391     x = GlobalAveragePooling2D()(x)
 392     x = Dense(classes, activation='softmax',
 393               use_bias=True, name='Logits')(x)
 394 
 395   # Ensure that the model takes into account
 396   # any potential predecessors of `input_tensor`.
 397   inputs = img_input
 398 
 399   # Create model.
 400   model = Model(inputs, x, name='mobilenetv2_%0.2f_%s' % (alpha, rows))
 401 
 402   if old_data_format:
 403     K.set_image_data_format(old_data_format)
 404   return model

 407 def _inverted_res_block(inputs, expansion, stride, alpha, filters, block_id):
 408   """Build an inverted res block."""
 409   in_channels = int(inputs.shape[-1])
 410   pointwise_conv_filters = int(filters * alpha)
 411   pointwise_filters = _make_divisible(pointwise_conv_filters, 8)
 412   # Expand
 413 
 414   x = Conv2D(expansion * in_channels, kernel_size=1, padding='same',
 415              use_bias=False, activation=None,
 416              name='mobl%d_conv_expand' % block_id)(inputs)
 417   x = BatchNormalization(epsilon=1e-3, momentum=0.999,
 418                          name='bn%d_conv_bn_expand' %
 419                          block_id)(x)
 420   x = Activation(relu6, name='conv_%d_relu' % block_id)(x)
 421 
 422   # Depthwise
 423   x = DepthwiseConv2D(kernel_size=3, strides=stride, activation=None,
 424                       use_bias=False, padding='same',
 425                       name='mobl%d_conv_depthwise' % block_id)(x)
 426   x = BatchNormalization(epsilon=1e-3, momentum=0.999,
 427                          name='bn%d_conv_depthwise' % block_id)(x)
 428 
 429   x = Activation(relu6, name='conv_dw_%d_relu' % block_id)(x)
 430 
 431   # Project
 432   x = Conv2D(pointwise_filters,
 433              kernel_size=1, padding='same', use_bias=False, activation=None,
 434              name='mobl%d_conv_project' % block_id)(x)
 435   x = BatchNormalization(epsilon=1e-3, momentum=0.999,
 436                          name='bn%d_conv_bn_project' % block_id)(x)
 437 
 438   if in_channels == pointwise_filters and stride == 1:
 439     return Add(name='res_connect_' + str(block_id))([inputs, x])
 440 
 441   return x

 444 def _first_inverted_res_block(inputs,
 445                               stride,
 446                               alpha, filters, block_id):
 447   """Build the first inverted res block."""
 448   in_channels = int(inputs.shape[-1])
 449   pointwise_conv_filters = int(filters * alpha)
 450   pointwise_filters = _make_divisible(pointwise_conv_filters, 8)
 451 
 452   # Depthwise
 453   x = DepthwiseConv2D(kernel_size=3,
 454                       strides=stride, activation=None,
 455                       use_bias=False, padding='same',
 456                       name='mobl%d_conv_depthwise' %
 457                       block_id)(inputs)
 458   x = BatchNormalization(epsilon=1e-3, momentum=0.999,
 459                          name='bn%d_conv_depthwise' %
 460                          block_id)(x)
 461   x = Activation(relu6, name='conv_dw_%d_relu' % block_id)(x)
 462 
 463   # Project
 464   x = Conv2D(pointwise_filters,
 465              kernel_size=1,
 466              padding='same',
 467              use_bias=False,
 468              activation=None,
 469              name='mobl%d_conv_project' %
 470              block_id)(x)
 471   x = BatchNormalization(epsilon=1e-3, momentum=0.999,
 472                          name='bn%d_conv_project' %
 473                          block_id)(x)
 474 
 475   if in_channels == pointwise_filters and stride == 1:
 476     return Add(name='res_connect_' + str(block_id))([inputs, x])
 477 
 478   return x
