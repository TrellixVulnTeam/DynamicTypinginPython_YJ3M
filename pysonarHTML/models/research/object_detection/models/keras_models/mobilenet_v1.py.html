  28 def _fixed_padding(inputs, kernel_size, rate=1):  # pylint: disable=invalid-name
  29   """Pads the input along the spatial dimensions independently of input size.
  30 
  31   Pads the input such that if it was used in a convolution with 'VALID' padding,
  32   the output would have the same dimensions as if the unpadded input was used
  33   in a convolution with 'SAME' padding.
  34 
  35   Args:
  36     inputs: A tensor of size [batch, height_in, width_in, channels].
  37     kernel_size: The kernel to be used in the conv2d or max_pool2d operation.
  38     rate: An integer, rate for atrous convolution.
  39 
  40   Returns:
  41     output: A tensor of size [batch, height_out, width_out, channels] with the
  42       input, either intact (if kernel_size == 1) or padded (if kernel_size > 1).
  43   """
  44   kernel_size_effective = [kernel_size[0] + (kernel_size[0] - 1) * (rate - 1),
  45                            kernel_size[0] + (kernel_size[0] - 1) * (rate - 1)]
  46   pad_total = [kernel_size_effective[0] - 1, kernel_size_effective[1] - 1]
  47   pad_beg = [pad_total[0] // 2, pad_total[1] // 2]
  48   pad_end = [pad_total[0] - pad_beg[0], pad_total[1] - pad_beg[1]]
  49   padded_inputs = tf.pad(inputs, [[0, 0], [pad_beg[0], pad_end[0]],
  50                                   [pad_beg[1], pad_end[1]], [0, 0]])
  51   return padded_inputs

  54 class _LayersOverride(object):
  55   """Alternative Keras layers interface for the Keras MobileNetV1."""
  56 
  57   def __init__(self,
  58                batchnorm_training,
  59                default_batchnorm_momentum=0.999,
  60                conv_hyperparams=None,
  61                use_explicit_padding=False,
  62                alpha=1.0,
  63                min_depth=None,
  64                conv_defs=None):
  65     """Alternative tf.keras.layers interface, for use by the Keras MobileNetV1.
  66 
  67     It is used by the Keras applications kwargs injection API to
  68     modify the MobilenetV1 Keras application with changes required by
  69     the Object Detection API.
  70 
  71     These injected interfaces make the following changes to the network:
  72 
  73     - Applies the Object Detection hyperparameter configuration
  74     - Supports FreezableBatchNorms
  75     - Adds support for a min number of filters for each layer
  76     - Makes the `alpha` parameter affect the final convolution block even if it
  77         is less than 1.0
  78     - Adds support for explicit padding of convolutions
  79 
  80     Args:
  81       batchnorm_training: Bool. Assigned to Batch norm layer `training` param
  82         when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
  83       default_batchnorm_momentum: Float. When 'conv_hyperparams' is None,
  84         batch norm layers will be constructed using this value as the momentum.
  85       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  86         containing hyperparameters for convolution ops. Optionally set to `None`
  87         to use default mobilenet_v1 layer builders.
  88       use_explicit_padding: If True, use 'valid' padding for convolutions,
  89         but explicitly pre-pads inputs so that the output dimensions are the
  90         same as if 'same' padding were used. Off by default.
  91       alpha: The width multiplier referenced in the MobileNetV1 paper. It
  92         modifies the number of filters in each convolutional layer. It's called
  93         depth multiplier in Keras application MobilenetV1.
  94       min_depth: Minimum number of filters in the convolutional layers.
  95       conv_defs: Network layout to specify the mobilenet_v1 body. Default is
  96         `None` to use the default mobilenet_v1 network layout.
  97     """
  98     self._alpha = alpha
  99     self._batchnorm_training = batchnorm_training
 100     self._default_batchnorm_momentum = default_batchnorm_momentum
 101     self._conv_hyperparams = conv_hyperparams
 102     self._use_explicit_padding = use_explicit_padding
 103     self._min_depth = min_depth
 104     self._conv_defs = conv_defs
 105     self.regularizer = tf.keras.regularizers.l2(0.00004 * 0.5)
 106     self.initializer = tf.truncated_normal_initializer(stddev=0.09)
 107 
 108   def _FixedPaddingLayer(self, kernel_size, rate=1):
 109     return tf.keras.layers.Lambda(
 110         lambda x: _fixed_padding(x, kernel_size, rate))
 111 
 112   def Conv2D(self, filters, kernel_size, **kwargs):
 113     """Builds a Conv2D layer according to the current Object Detection config.
 114 
 115     Overrides the Keras MobileNetV1 application's convolutions with ones that
 116     follow the spec specified by the Object Detection hyperparameters.
 117 
 118     Args:
 119       filters: The number of filters to use for the convolution.
 120       kernel_size: The kernel size to specify the height and width of the 2D
 121         convolution window. In this function, the kernel size is expected to
 122         be pair of numbers and the numbers must be equal for this function.
 123       **kwargs: Keyword args specified by the Keras application for
 124         constructing the convolution.
 125 
 126     Returns:
 127       A one-arg callable that will either directly apply a Keras Conv2D layer to
 128       the input argument, or that will first pad the input then apply a Conv2D
 129       layer.
 130 
 131     Raises:
 132       ValueError: if kernel size is not a pair of equal
 133         integers (representing a square kernel).
 134     """
 135     if not isinstance(kernel_size, tuple):
 136       raise ValueError('kernel is expected to be a tuple.')
 137     if len(kernel_size) != 2:
 138       raise ValueError('kernel is expected to be length two.')
 139     if kernel_size[0] != kernel_size[1]:
 140       raise ValueError('kernel is expected to be square.')
 141     layer_name = kwargs['name']
 142     if self._conv_defs:
 143       conv_filters = model_utils.get_conv_def(self._conv_defs, layer_name)
 144       if conv_filters:
 145         filters = conv_filters
 146     # Apply the width multiplier and the minimum depth to the convolution layers
 147     filters = int(filters * self._alpha)
 148     if self._min_depth and filters < self._min_depth:
 149       filters = self._min_depth
 150 
 151     if self._conv_hyperparams:
 152       kwargs = self._conv_hyperparams.params(**kwargs)
 153     else:
 154       kwargs['kernel_regularizer'] = self.regularizer
 155       kwargs['kernel_initializer'] = self.initializer
 156 
 157     kwargs['padding'] = 'same'
 158     if self._use_explicit_padding and kernel_size[0] > 1:
 159       kwargs['padding'] = 'valid'
 160       def padded_conv(features):  # pylint: disable=invalid-name
 161         padded_features = self._FixedPaddingLayer(kernel_size)(features)
 162         return tf.keras.layers.Conv2D(
 163             filters, kernel_size, **kwargs)(padded_features)
 164       return padded_conv
 165     else:
 166       return tf.keras.layers.Conv2D(filters, kernel_size, **kwargs)
 167 
 168   def DepthwiseConv2D(self, kernel_size, **kwargs):
 169     """Builds a DepthwiseConv2D according to the Object Detection config.
 170 
 171     Overrides the Keras MobileNetV2 application's convolutions with ones that
 172     follow the spec specified by the Object Detection hyperparameters.
 173 
 174     Args:
 175       kernel_size: The kernel size to specify the height and width of the 2D
 176         convolution window.
 177       **kwargs: Keyword args specified by the Keras application for
 178         constructing the convolution.
 179 
 180     Returns:
 181       A one-arg callable that will either directly apply a Keras DepthwiseConv2D
 182       layer to the input argument, or that will first pad the input then apply
 183       the depthwise convolution.
 184     """
 185     if self._conv_hyperparams:
 186       kwargs = self._conv_hyperparams.params(**kwargs)
 187       # Both regularizer and initializaer also applies to depthwise layer in
 188       # MobilenetV1, so we remap the kernel_* to depthwise_* here.
 189       kwargs['depthwise_regularizer'] = kwargs['kernel_regularizer']
 190       kwargs['depthwise_initializer'] = kwargs['kernel_initializer']
 191     else:
 192       kwargs['depthwise_regularizer'] = self.regularizer
 193       kwargs['depthwise_initializer'] = self.initializer
 194 
 195     kwargs['padding'] = 'same'
 196     if self._use_explicit_padding:
 197       kwargs['padding'] = 'valid'
 198       def padded_depthwise_conv(features):  # pylint: disable=invalid-name
 199         padded_features = self._FixedPaddingLayer(kernel_size)(features)
 200         return tf.keras.layers.DepthwiseConv2D(
 201             kernel_size, **kwargs)(padded_features)
 202       return padded_depthwise_conv
 203     else:
 204       return tf.keras.layers.DepthwiseConv2D(kernel_size, **kwargs)

 206   def BatchNormalization(self, **kwargs):
 207     """Builds a normalization layer.
 208 
 209     Overrides the Keras application batch norm with the norm specified by the
 210     Object Detection configuration.
 211 
 212     Args:
 213       **kwargs: Only the name is used, all other params ignored.
 214         Required for matching `layers.BatchNormalization` calls in the Keras
 215         application.
 216 
 217     Returns:
 218       A normalization layer specified by the Object Detection hyperparameter
 219       configurations.
 220     """
 221     name = kwargs.get('name')
 222     if self._conv_hyperparams:
 223       return self._conv_hyperparams.build_batch_norm(
 224           training=self._batchnorm_training,
 225           name=name)
 226     else:
 227       return freezable_batch_norm.FreezableBatchNorm(
 228           training=self._batchnorm_training,
 229           epsilon=1e-3,
 230           momentum=self._default_batchnorm_momentum,
 231           name=name)

 233   def Input(self, shape):
 234     """Builds an Input layer.
 235 
 236     Overrides the Keras application Input layer with one that uses a
 237     tf.placeholder_with_default instead of a tf.placeholder. This is necessary
 238     to ensure the application works when run on a TPU.
 239 
 240     Args:
 241       shape: The shape for the input layer to use. (Does not include a dimension
 242         for the batch size).
 243     Returns:
 244       An input layer for the specified shape that internally uses a
 245       placeholder_with_default.
 246     """
 247     default_size = 224
 248     default_batch_size = 1
 249     shape = list(shape)
 250     default_shape = [default_size if dim is None else dim for dim in shape]
 251 
 252     input_tensor = tf.constant(0.0, shape=[default_batch_size] + default_shape)
 253 
 254     placeholder_with_default = tf.placeholder_with_default(
 255         input=input_tensor, shape=[None] + shape)
 256     return model_utils.input_layer(shape, placeholder_with_default)
 257 
 258   # pylint: disable=unused-argument
 259   def ReLU(self, *args, **kwargs):
 260     """Builds an activation layer.
 261 
 262     Overrides the Keras application ReLU with the activation specified by the
 263     Object Detection configuration.
 264 
 265     Args:
 266       *args: Ignored, required to match the `tf.keras.ReLU` interface
 267       **kwargs: Only the name is used,
 268         required to match `tf.keras.ReLU` interface
 269 
 270     Returns:
 271       An activation layer specified by the Object Detection hyperparameter
 272       configurations.
 273     """
 274     name = kwargs.get('name')
 275     if self._conv_hyperparams:
 276       return self._conv_hyperparams.build_activation_layer(name=name)
 277     else:
 278       return tf.keras.layers.Lambda(tf.nn.relu6, name=name)

 283     """Replaces explicit padding in the Keras application with a no-op.

 285     Args:

 289     Returns:
 290       A no-op identity lambda.

 300 # pylint: disable=invalid-name
 301 def mobilenet_v1(batchnorm_training,
 302                  default_batchnorm_momentum=0.9997,
 303                  conv_hyperparams=None,
 304                  use_explicit_padding=False,
 305                  alpha=1.0,
 306                  min_depth=None,
 307                  conv_defs=None,
 308                  **kwargs):
 309   """Instantiates the MobileNetV1 architecture, modified for object detection.
 310 
 311   This wraps the MobileNetV1 tensorflow Keras application, but uses the
 312   Keras application's kwargs-based monkey-patching API to override the Keras
 313   architecture with the following changes:
 314 
 315   - Changes the default batchnorm momentum to 0.9997
 316   - Applies the Object Detection hyperparameter configuration
 317   - Supports FreezableBatchNorms
 318   - Adds support for a min number of filters for each layer
 319   - Makes the `alpha` parameter affect the final convolution block even if it
 320       is less than 1.0
 321   - Adds support for explicit padding of convolutions
 322   - Makes the Input layer use a tf.placeholder_with_default instead of a
 323       tf.placeholder, to work on TPUs.
 324 
 325   Args:
 326       batchnorm_training: Bool. Assigned to Batch norm layer `training` param
 327         when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
 328       default_batchnorm_momentum: Float. When 'conv_hyperparams' is None,
 329         batch norm layers will be constructed using this value as the momentum.
 330       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 331         containing hyperparameters for convolution ops. Optionally set to `None`
 332         to use default mobilenet_v1 layer builders.
 333       use_explicit_padding: If True, use 'valid' padding for convolutions,
 334         but explicitly pre-pads inputs so that the output dimensions are the
 335         same as if 'same' padding were used. Off by default.
 336       alpha: The width multiplier referenced in the MobileNetV1 paper. It
 337         modifies the number of filters in each convolutional layer.
 338       min_depth: Minimum number of filters in the convolutional layers.
 339       conv_defs: Network layout to specify the mobilenet_v1 body. Default is
 340         `None` to use the default mobilenet_v1 network layout.
 341       **kwargs: Keyword arguments forwarded directly to the
 342         `tf.keras.applications.Mobilenet` method that constructs the Keras
 343         model.
 344 
 345   Returns:
 346       A Keras model instance.
 347   """
 348   layers_override = _LayersOverride(
 349       batchnorm_training,
 350       default_batchnorm_momentum=default_batchnorm_momentum,
 351       conv_hyperparams=conv_hyperparams,
 352       use_explicit_padding=use_explicit_padding,
 353       min_depth=min_depth,
 354       alpha=alpha,
 355       conv_defs=conv_defs)
 356   return tf.keras.applications.MobileNet(
 357       alpha=alpha, layers=layers_override, **kwargs)
