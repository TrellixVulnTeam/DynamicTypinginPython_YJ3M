  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function

  22 import tensorflow.compat.v1 as tf
  23 
  24 from object_detection.core import freezable_batch_norm

  27 class _LayersOverride(object):
  28   """Alternative Keras layers interface for the Keras InceptionResNetV2."""
  29 
  30   def __init__(self,
  31                batchnorm_training,
  32                output_stride=16,
  33                align_feature_maps=False,
  34                batchnorm_scale=False,
  35                default_batchnorm_momentum=0.999,
  36                default_batchnorm_epsilon=1e-3,
  37                weight_decay=0.00004):
  38     """Alternative tf.keras.layers interface, for use by InceptionResNetV2.
  39 
  40     It is used by the Keras applications kwargs injection API to
  41     modify the Inception Resnet V2 Keras application with changes required by
  42     the Object Detection API.
  43 
  44     These injected interfaces make the following changes to the network:
  45 
  46     - Supports freezing batch norm layers
  47     - Adds support for feature map alignment (like in the Slim model)
  48     - Adds support for changing the output stride (like in the Slim model)
  49     - Adds support for overriding various batch norm hyperparameters
  50 
  51     Because the Keras inception resnet v2 application does not assign explicit
  52     names to most individual layers, the injection of output stride support
  53     works by identifying convolution layers according to their filter counts
  54     and pre-feature-map-alignment padding arguments.
  55 
  56     Args:
  57       batchnorm_training: Bool. Assigned to Batch norm layer `training` param
  58         when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
  59       output_stride: A scalar that specifies the requested ratio of input to
  60         output spatial resolution. Only supports 8 and 16.
  61       align_feature_maps: When true, changes all the VALID paddings in the
  62         network to SAME padding so that the feature maps are aligned.
  63       batchnorm_scale: If True, uses an explicit `gamma` multiplier to scale the
  64         activations in the batch normalization layer.
  65       default_batchnorm_momentum: Float. Batch norm layers will be constructed
  66         using this value as the momentum.
  67       default_batchnorm_epsilon: small float added to variance to avoid
  68         dividing by zero.
  69       weight_decay: the l2 regularization weight decay for weights variables.
  70         (gets multiplied by 0.5 to map from slim l2 regularization weight to
  71         Keras l2 regularization weight).
  72     """
  73     self._use_atrous = output_stride == 8
  74     self._align_feature_maps = align_feature_maps
  75     self._batchnorm_training = batchnorm_training
  76     self._batchnorm_scale = batchnorm_scale
  77     self._default_batchnorm_momentum = default_batchnorm_momentum
  78     self._default_batchnorm_epsilon = default_batchnorm_epsilon
  79     self.regularizer = tf.keras.regularizers.l2(weight_decay * 0.5)
  80 
  81   def Conv2D(self, filters, kernel_size, **kwargs):
  82     """Builds a Conv2D layer according to the current Object Detection config.
  83 
  84     Overrides the Keras InceptionResnetV2 application's convolutions with ones
  85     that follow the spec specified by the Object Detection hyperparameters.
  86 
  87     If feature map alignment is enabled, the padding will be forced to 'same'.
  88     If output_stride is 8, some conv2d layers will be matched according to
  89     their name or filter counts or pre-alignment padding parameters, and will
  90     have the correct 'dilation rate' or 'strides' set.
  91 
  92     Args:
  93       filters: The number of filters to use for the convolution.
  94       kernel_size: The kernel size to specify the height and width of the 2D
  95         convolution window.
  96       **kwargs: Keyword args specified by the Keras application for
  97         constructing the convolution.
  98 
  99     Returns:
 100       A Keras Conv2D layer specified by the Object Detection hyperparameter
 101       configurations.
 102     """
 103     kwargs['kernel_regularizer'] = self.regularizer
 104     kwargs['bias_regularizer'] = self.regularizer
 105 
 106     # Because the Keras application does not set explicit names for most layers,
 107     # (instead allowing names to auto-increment), we must match individual
 108     # layers in the model according to their filter count, name, or
 109     # pre-alignment mapping. This means we can only align the feature maps
 110     # after we have applied our updates in cases where output_stride=8.
 111     if self._use_atrous and (filters == 384):
 112       kwargs['strides'] = 1
 113 
 114     name = kwargs.get('name')
 115     if self._use_atrous and (
 116         (name and 'block17' in name) or
 117         (filters == 128 or filters == 160 or
 118          (filters == 192 and kwargs.get('padding', '').lower() != 'valid'))):
 119       kwargs['dilation_rate'] = 2
 120 
 121     if self._align_feature_maps:
 122       kwargs['padding'] = 'same'
 123 
 124     return tf.keras.layers.Conv2D(filters, kernel_size, **kwargs)
 125 
 126   def MaxPooling2D(self, pool_size, strides, **kwargs):
 127     """Builds a pooling layer according to the current Object Detection config.
 128 
 129     Overrides the Keras InceptionResnetV2 application's MaxPooling2D layers with
 130     ones that follow the spec specified by the Object Detection hyperparameters.
 131 
 132     If feature map alignment is enabled, the padding will be forced to 'same'.
 133     If output_stride is 8, some pooling layers will be matched according to
 134     their pre-alignment padding parameters, and will have their 'strides'
 135     argument overridden.
 136 
 137     Args:
 138       pool_size: The pool size specified by the Keras application.
 139       strides: The strides specified by the unwrapped Keras application.
 140       **kwargs: Keyword args specified by the Keras application for
 141         constructing the max pooling layer.
 142 
 143     Returns:
 144       A MaxPool2D layer specified by the Object Detection hyperparameter
 145       configurations.
 146     """
 147     if self._use_atrous and kwargs.get('padding', '').lower() == 'valid':
 148       strides = 1
 149 
 150     if self._align_feature_maps:
 151       kwargs['padding'] = 'same'
 152 
 153     return tf.keras.layers.MaxPool2D(pool_size, strides=strides, **kwargs)
 154 
 155   # We alias MaxPool2D because Keras has that alias
 156   MaxPool2D = MaxPooling2D  # pylint: disable=invalid-name
 157 
 158   def BatchNormalization(self, **kwargs):
 159     """Builds a normalization layer.
 160 
 161     Overrides the Keras application batch norm with the norm specified by the
 162     Object Detection configuration.
 163 
 164     Args:
 165       **kwargs: Keyword arguments from the `layers.BatchNormalization` calls in
 166         the Keras application.
 167 
 168     Returns:
 169       A normalization layer specified by the Object Detection hyperparameter
 170       configurations.
 171     """
 172     kwargs['scale'] = self._batchnorm_scale
 173     return freezable_batch_norm.FreezableBatchNorm(
 174         training=self._batchnorm_training,
 175         epsilon=self._default_batchnorm_epsilon,
 176         momentum=self._default_batchnorm_momentum,
 177         **kwargs)
 178 
 179   # Forward all non-overridden methods to the keras layers
 180   def __getattr__(self, item):
 181     return getattr(tf.keras.layers, item)

 184 # pylint: disable=invalid-name
 185 def inception_resnet_v2(
 186     batchnorm_training,
 187     output_stride=16,
 188     align_feature_maps=False,
 189     batchnorm_scale=False,
 190     weight_decay=0.00004,
 191     default_batchnorm_momentum=0.9997,
 192     default_batchnorm_epsilon=0.001,
 193     **kwargs):
 194   """Instantiates the InceptionResnetV2 architecture.
 195 
 196   (Modified for object detection)
 197 
 198   This wraps the InceptionResnetV2 tensorflow Keras application, but uses the
 199   Keras application's kwargs-based monkey-patching API to override the Keras
 200   architecture with the following changes:
 201 
 202   - Supports freezing batch norm layers with FreezableBatchNorms
 203   - Adds support for feature map alignment (like in the Slim model)
 204   - Adds support for changing the output stride (like in the Slim model)
 205   - Changes the default batchnorm momentum to 0.9997
 206   - Adds support for overriding various batchnorm hyperparameters
 207 
 208   Args:
 209       batchnorm_training: Bool. Assigned to Batch norm layer `training` param
 210         when constructing `freezable_batch_norm.FreezableBatchNorm` layers.
 211       output_stride: A scalar that specifies the requested ratio of input to
 212         output spatial resolution. Only supports 8 and 16.
 213       align_feature_maps: When true, changes all the VALID paddings in the
 214         network to SAME padding so that the feature maps are aligned.
 215       batchnorm_scale: If True, uses an explicit `gamma` multiplier to scale the
 216         activations in the batch normalization layer.
 217       weight_decay: the l2 regularization weight decay for weights variables.
 218         (gets multiplied by 0.5 to map from slim l2 regularization weight to
 219         Keras l2 regularization weight).
 220       default_batchnorm_momentum: Float. Batch norm layers will be constructed
 221         using this value as the momentum.
 222       default_batchnorm_epsilon: small float added to variance to avoid
 223         dividing by zero.
 224       **kwargs: Keyword arguments forwarded directly to the
 225         `tf.keras.applications.InceptionResNetV2` method that constructs the
 226         Keras model.
 227 
 228   Returns:
 229       A Keras model instance.
 230   """
 231   if output_stride != 8 and output_stride != 16:
 232     raise ValueError('output_stride must be 8 or 16.')
 233 
 234   layers_override = _LayersOverride(
 235       batchnorm_training,
 236       output_stride,
 237       align_feature_maps=align_feature_maps,
 238       batchnorm_scale=batchnorm_scale,
 239       default_batchnorm_momentum=default_batchnorm_momentum,
 240       default_batchnorm_epsilon=default_batchnorm_epsilon,
 241       weight_decay=weight_decay)
 242   return tf.keras.applications.InceptionResNetV2(
 243       layers=layers_override, **kwargs)
