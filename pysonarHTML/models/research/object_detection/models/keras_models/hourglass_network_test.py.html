  15 """Testing the Hourglass network."""
  16 
  17 from absl.testing import parameterized
  18 import numpy as np
  19 import tensorflow.compat.v1 as tf
  20 
  21 from object_detection.models.keras_models import hourglass_network as hourglass
  22 
  23 
  24 class HourglassFeatureExtractorTest(tf.test.TestCase, parameterized.TestCase):
  25 
  26   def test_identity_layer(self):
  27 
  28     layer = hourglass.IdentityLayer()
  29     output = layer(np.zeros((2, 32, 32, 3), dtype=np.float32))
  30     self.assertEqual(output.shape, (2, 32, 32, 3))
  31 
  32   def test_skip_conv_layer_stride_1(self):
  33 
  34     layer = hourglass.SkipConvolution(out_channels=8, stride=1)
  35     output = layer(np.zeros((2, 32, 32, 3), dtype=np.float32))
  36     self.assertEqual(output.shape, (2, 32, 32, 8))
  37 
  38   def test_skip_conv_layer_stride_2(self):
  39 
  40     layer = hourglass.SkipConvolution(out_channels=8, stride=2)
  41     output = layer(np.zeros((2, 32, 32, 3), dtype=np.float32))
  42     self.assertEqual(output.shape, (2, 16, 16, 8))
  43 
  44   @parameterized.parameters([{'kernel_size': 1},
  45                              {'kernel_size': 3},
  46                              {'kernel_size': 7}])
  47   def test_conv_block(self, kernel_size):
  48 
  49     layer = hourglass.ConvolutionalBlock(
  50         out_channels=8, kernel_size=kernel_size, stride=1)
  51     output = layer(np.zeros((2, 32, 32, 3), dtype=np.float32))
  52     self.assertEqual(output.shape, (2, 32, 32, 8))
  53 
  54     layer = hourglass.ConvolutionalBlock(
  55         out_channels=8, kernel_size=kernel_size, stride=2)
  56     output = layer(np.zeros((2, 32, 32, 3), dtype=np.float32))
  57     self.assertEqual(output.shape, (2, 16, 16, 8))
  58 
  59   def test_residual_block_stride_1(self):
  60 
  61     layer = hourglass.ResidualBlock(out_channels=8, stride=1)
  62     output = layer(np.zeros((2, 32, 32, 8), dtype=np.float32))
  63     self.assertEqual(output.shape, (2, 32, 32, 8))
  64 
  65   def test_residual_block_stride_2(self):
  66 
  67     layer = hourglass.ResidualBlock(out_channels=8, stride=2,
  68                                     skip_conv=True)
  69     output = layer(np.zeros((2, 32, 32, 8), dtype=np.float32))
  70     self.assertEqual(output.shape, (2, 16, 16, 8))
  71 
  72   def test_input_downsample_block(self):
  73 
  74     layer = hourglass.InputDownsampleBlock(
  75         out_channels_initial_conv=4, out_channels_residual_block=8)
  76     output = layer(np.zeros((2, 32, 32, 8), dtype=np.float32))
  77     self.assertEqual(output.shape, (2, 8, 8, 8))
  78 
  79   def test_encoder_decoder_block(self):
  80 
  81     layer = hourglass.EncoderDecoderBlock(
  82         num_stages=4, blocks_per_stage=[2, 3, 4, 5, 6],
  83         channel_dims=[4, 6, 8, 10, 12])
  84     output = layer(np.zeros((2, 64, 64, 4), dtype=np.float32))
  85     self.assertEqual(output.shape, (2, 64, 64, 4))
  86 
  87   def test_hourglass_feature_extractor(self):
  88 
  89     model = hourglass.HourglassNetwork(
  90         num_stages=4, blocks_per_stage=[2, 3, 4, 5, 6],
  91         channel_dims=[4, 6, 8, 10, 12, 14], num_hourglasses=2)
  92     outputs = model(np.zeros((2, 64, 64, 3), dtype=np.float32))
  93     self.assertEqual(outputs[0].shape, (2, 16, 16, 6))
  94     self.assertEqual(outputs[1].shape, (2, 16, 16, 6))
  95 
  96 
  97 if __name__ == '__main__':
  98   tf.enable_v2_behavior()
  99   tf.test.main()
