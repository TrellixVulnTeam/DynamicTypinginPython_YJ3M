  29 class IdentityLayer(tf.keras.layers.Layer):
  30   """A layer which passes through the input as it is."""
  31 
  32   def call(self, inputs):
  33     return inputs

  36 def _get_padding_for_kernel_size(kernel_size):
  37   if kernel_size == 7:
  38     return (3, 3)
  39   elif kernel_size == 3:
  40     return (1, 1)
  41   else:
  42     raise ValueError('Padding for kernel size {} not known.'.format(
  43         kernel_size))

  46 class ConvolutionalBlock(tf.keras.layers.Layer):
  47   """Block that aggregates Convolution + Norm layer + ReLU."""
  48 
  49   def __init__(self, kernel_size, out_channels, stride=1, relu=True,
  50                padding='same'):
  51     """Initializes the Convolutional block.
  52 
  53     Args:
  54       kernel_size: int, convolution kernel size.
  55       out_channels: int, the desired number of output channels.
  56       stride: Integer, stride used in the convolution.
  57       relu: bool, whether to use relu at the end of the layer.
  58       padding: str, the padding scheme to use when kernel_size <= 1
  59     """
  60     super(ConvolutionalBlock, self).__init__()
  61 
  62     if kernel_size > 1:
  63       padding = 'valid'
  64       padding_size = _get_padding_for_kernel_size(kernel_size)
  65 
  66       # TODO(vighneshb) Explore if removing and using padding option in conv
  67       # layer works.
  68       self.pad = tf.keras.layers.ZeroPadding2D(padding_size)
  69     else:
  70       self.pad = IdentityLayer()
  71 
  72     self.conv = tf.keras.layers.Conv2D(
  73         filters=out_channels, kernel_size=kernel_size, use_bias=False,
  74         strides=stride, padding=padding)
  75 
  76     self.norm = tf.keras.layers.experimental.SyncBatchNormalization(
  77         name='batchnorm', epsilon=1e-5, momentum=0.1)
  78 
  79     if relu:
  80       self.relu = tf.keras.layers.ReLU()
  81     else:
  82       self.relu = IdentityLayer()
  83 
  84   def call(self, inputs):
  85     net = self.pad(inputs)
  86     net = self.conv(net)
  87     net = self.norm(net)
  88     return self.relu(net)

  91 class SkipConvolution(ConvolutionalBlock):
  92   """The skip connection layer for a ResNet."""
  93 
  94   def __init__(self, out_channels, stride):
  95     """Initializes the skip convolution layer.
  96 
  97     Args:
  98       out_channels: int, the desired number of output channels.
  99       stride: int, the stride for the layer.
 100     """
 101     super(SkipConvolution, self).__init__(
 102         out_channels=out_channels, kernel_size=1, stride=stride, relu=False)

 105 class ResidualBlock(tf.keras.layers.Layer):
 106   """A Residual block."""
 107 
 108   def __init__(self, out_channels, skip_conv=False, kernel_size=3, stride=1,
 109                padding='same'):
 110     """Initializes the Residual block.
 111 
 112     Args:
 113       out_channels: int, the desired number of output channels.
 114       skip_conv: bool, whether to use a conv layer for skip connections.
 115       kernel_size: int, convolution kernel size.
 116       stride: Integer, stride used in the convolution.
 117       padding: str, the type of padding to use.
 118     """
 119 
 120     super(ResidualBlock, self).__init__()
 121     self.conv_block = ConvolutionalBlock(
 122         kernel_size=kernel_size, out_channels=out_channels, stride=stride)
 123 
 124     self.conv = tf.keras.layers.Conv2D(
 125         filters=out_channels, kernel_size=kernel_size, use_bias=False,
 126         strides=1, padding=padding)
 127     self.norm = tf.keras.layers.experimental.SyncBatchNormalization(
 128         name='batchnorm', epsilon=1e-5, momentum=0.1)
 129 
 130     if skip_conv:
 131       self.skip = SkipConvolution(out_channels=out_channels,
 132                                   stride=stride)
 133     else:
 134       self.skip = IdentityLayer()
 135 
 136     self.relu = tf.keras.layers.ReLU()
 137 
 138   def call(self, inputs):
 139     net = self.conv_block(inputs)
 140     net = self.conv(net)
 141     net = self.norm(net)
 142     net_skip = self.skip(inputs)
 143     return self.relu(net + net_skip)

 146 class InputDownsampleBlock(tf.keras.layers.Layer):
 147   """Block for the initial feature downsampling."""
 148 
 149   def __init__(self, out_channels_initial_conv, out_channels_residual_block):
 150     """Initializes the downsample block.
 151 
 152     Args:
 153       out_channels_initial_conv: int, the desired number of output channels
 154         in the initial conv layer.
 155       out_channels_residual_block: int, the desired number of output channels
 156         in the underlying residual block.
 157     """
 158 
 159     super(InputDownsampleBlock, self).__init__()
 160     self.conv_block = ConvolutionalBlock(
 161         kernel_size=7, out_channels=out_channels_initial_conv, stride=2,
 162         padding='valid')
 163     self.residual_block = ResidualBlock(
 164         out_channels=out_channels_residual_block, stride=2, skip_conv=True)
 165 
 166   def call(self, inputs):
 167     return self.residual_block(self.conv_block(inputs))

 170 def _make_repeated_residual_blocks(out_channels, num_blocks,
 171                                    initial_stride=1, residual_channels=None):
 172   """Stack Residual blocks one after the other.
 173 
 174   Args:
 175     out_channels: int, the desired number of output channels.
 176     num_blocks: int, the number of residual blocks to be stacked.
 177     initial_stride: int, the stride of the initial residual block.
 178     residual_channels: int, the desired number of output channels in the
 179       intermediate residual blocks. If not specifed, we use out_channels.
 180 
 181   Returns:
 182     blocks: A list of residual blocks to be applied in sequence.
 183 
 184   """
 185 
 186   blocks = []
 187 
 188   if residual_channels is None:
 189     residual_channels = out_channels
 190 
 191   for i in range(num_blocks - 1):
 192     stride = initial_stride if i == 0 else 1
 193     skip_conv = stride > 1
 194 
 195     blocks.append(
 196         ResidualBlock(out_channels=residual_channels, stride=stride,
 197                       skip_conv=skip_conv)
 198     )
 199 
 200   skip_conv = residual_channels != out_channels
 201   blocks.append(ResidualBlock(out_channels=out_channels, skip_conv=skip_conv))
 202 
 203   return blocks
 204 
 205 
 206 def _apply_blocks(inputs, blocks):
 207   net = inputs
 208 
 209   for block in blocks:
 210     net = block(net)
 211 
 212   return net

 215 class EncoderDecoderBlock(tf.keras.layers.Layer):
 216   """An encoder-decoder block which recursively defines the hourglass network."""
 217 
 218   def __init__(self, num_stages, channel_dims, blocks_per_stage):
 219     """Initializes the encoder-decoder block.
 220 
 221     Args:
 222       num_stages: int, Number of stages in the network. At each stage we have 2
 223         encoder and 1 decoder blocks. The second encoder block downsamples the
 224         input.
 225       channel_dims: int list, the output channels dimensions of stages in
 226         the network. `channel_dims[0]` is used to define the number of
 227         channels in the first encoder block and `channel_dims[1]` is used to
 228         define the number of channels in the second encoder block. The channels
 229         in the recursive inner layers are defined using `channel_dims[1:]`
 230       blocks_per_stage: int list, number of residual blocks to use at each
 231         stage. `blocks_per_stage[0]` defines the number of blocks at the
 232         current stage and `blocks_per_stage[1:]` is used at further stages.
 233     """
 234 
 235     super(EncoderDecoderBlock, self).__init__()
 236 
 237     out_channels = channel_dims[0]
 238     out_channels_downsampled = channel_dims[1]
 239 
 240     self.encoder_block1 = _make_repeated_residual_blocks(
 241         out_channels=out_channels, num_blocks=blocks_per_stage[0],
 242         initial_stride=1)
 243     self.encoder_block2 = _make_repeated_residual_blocks(
 244         out_channels=out_channels_downsampled,
 245         num_blocks=blocks_per_stage[0], initial_stride=2)
 246 
 247     if num_stages > 1:
 248       self.inner_block = [
 249           EncoderDecoderBlock(num_stages - 1, channel_dims[1:],
 250                               blocks_per_stage[1:])
 251       ]
 252     else:
 253       self.inner_block = _make_repeated_residual_blocks(
 254           out_channels=out_channels_downsampled,
 255           num_blocks=blocks_per_stage[1])
 256 
 257     self.decoder_block = _make_repeated_residual_blocks(
 258         residual_channels=out_channels_downsampled,
 259         out_channels=out_channels, num_blocks=blocks_per_stage[0])
 260     self.upsample = tf.keras.layers.UpSampling2D(2)
 261 
 262     self.merge_features = tf.keras.layers.Add()
 263 
 264   def call(self, inputs):
 265 
 266     encoded_outputs = _apply_blocks(inputs, self.encoder_block1)
 267     encoded_downsampled_outputs = _apply_blocks(inputs, self.encoder_block2)
 268     inner_block_outputs = _apply_blocks(
 269         encoded_downsampled_outputs, self.inner_block)
 270 
 271     decoded_outputs = _apply_blocks(inner_block_outputs, self.decoder_block)
 272     upsampled_outputs = self.upsample(decoded_outputs)
 273 
 274     return self.merge_features([encoded_outputs, upsampled_outputs])

 277 class HourglassNetwork(tf.keras.Model):
 278   """The hourglass network."""
 279 
 280   def __init__(self, num_stages, channel_dims, blocks_per_stage,
 281                num_hourglasses):
 282     """Intializes the feature extractor.
 283 
 284     Args:
 285       num_stages: int, Number of stages in the network. At each stage we have 2
 286         encoder and 1 decoder blocks. The second encoder block downsamples the
 287         input.
 288       channel_dims: int list, the output channel dimensions of stages in
 289         the network. `channel_dims[0]` and `channel_dims[1]` are used to define
 290         the initial downsampling block. `channel_dims[1:]` is used to define
 291         the hourglass network(s) which follow(s).
 292       blocks_per_stage: int list, number of residual blocks to use at each
 293         stage in the hourglass network
 294       num_hourglasses: int, number of hourglas networks to stack
 295         sequentially.
 296     """
 297 
 298     super(HourglassNetwork, self).__init__()
 299 
 300     self.num_hourglasses = num_hourglasses
 301     self.downsample_input = InputDownsampleBlock(
 302         out_channels_initial_conv=channel_dims[0],
 303         out_channels_residual_block=channel_dims[1]
 304     )
 305 
 306     self.hourglass_network = []
 307     self.output_conv = []
 308     for _ in range(self.num_hourglasses):
 309       self.hourglass_network.append(
 310           EncoderDecoderBlock(
 311               num_stages=num_stages, channel_dims=channel_dims[1:],
 312               blocks_per_stage=blocks_per_stage)
 313       )
 314       self.output_conv.append(
 315           ConvolutionalBlock(kernel_size=3, out_channels=channel_dims[1])
 316       )
 317 
 318     self.intermediate_conv1 = []
 319     self.intermediate_conv2 = []
 320     self.intermediate_residual = []
 321 
 322     for _ in range(self.num_hourglasses - 1):
 323       self.intermediate_conv1.append(
 324           ConvolutionalBlock(
 325               kernel_size=1, out_channels=channel_dims[1], relu=False)
 326       )
 327       self.intermediate_conv2.append(
 328           ConvolutionalBlock(
 329               kernel_size=1, out_channels=channel_dims[1], relu=False)
 330       )
 331       self.intermediate_residual.append(
 332           ResidualBlock(out_channels=channel_dims[1])
 333       )
 334 
 335     self.intermediate_relu = tf.keras.layers.ReLU()
 336 
 337   def call(self, inputs):
 338 
 339     inputs = self.downsample_input(inputs)
 340     outputs = []
 341 
 342     for i in range(self.num_hourglasses):
 343 
 344       hourglass_output = self.hourglass_network[i](inputs)
 345 
 346       output = self.output_conv[i](hourglass_output)
 347       outputs.append(output)
 348 
 349       if i < self.num_hourglasses - 1:
 350         secondary_output = (self.intermediate_conv1[i](inputs) +
 351                             self.intermediate_conv2[i](output))
 352         secondary_output = self.intermediate_relu(secondary_output)
 353         inputs = self.intermediate_residual[i](secondary_output)
 354 
 355     return outputs
 356 
 357   @property
 358   def out_stride(self):
 359     """The stride in the output image of the network."""
 360     return 4
 361 
 362   @property
 363   def num_feature_outputs(self):
 364     """Ther number of feature outputs returned by the feature extractor."""
 365     return self.num_hourglasses
