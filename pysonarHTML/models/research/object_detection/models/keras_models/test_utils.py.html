  46 # def _get_slim_endpoint_shapes(inputs, depth_multiplier=1.0, min_depth=8,
  47 #                               use_explicit_padding=False):
  48 #   with slim.arg_scope([slim.conv2d, slim.separable_conv2d],
  49 #                       normalizer_fn=slim.batch_norm):
  50 #     _, end_points = mobilenet_v1.mobilenet_v1_base(
  51 #         inputs, final_endpoint='Conv2d_13_pointwise',
  52 #         depth_multiplier=depth_multiplier, min_depth=min_depth,
  53 #         use_explicit_padding=use_explicit_padding)
  54 #     return [end_points[endpoint_name].get_shape()
