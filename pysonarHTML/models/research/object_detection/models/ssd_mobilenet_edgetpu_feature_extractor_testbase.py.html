  29   @abc.abstractmethod
  30   def _get_input_sizes(self):
  31     """Return feature map sizes for the two inputs to SSD head."""
  32     pass

  34   def test_extract_features_returns_correct_shapes_128(self):
  35     image_height = 128
  36     image_width = 128
  37     depth_multiplier = 1.0
  38     pad_to_multiple = 1
  39     input_feature_sizes = self._get_input_sizes()
  40     expected_feature_map_shape = [(2, 8, 8, input_feature_sizes[0]),
  41                                   (2, 4, 4, input_feature_sizes[1]),
  42                                   (2, 2, 2, 512), (2, 1, 1, 256), (2, 1, 1,
  43                                                                    256),
  44                                   (2, 1, 1, 128)]
  45     self.check_extract_features_returns_correct_shape(
  46         2,
  47         image_height,
  48         image_width,
  49         depth_multiplier,
  50         pad_to_multiple,
  51         expected_feature_map_shape,
  52         use_keras=False)

  89   def test_preprocess_returns_correct_value_range(self):
  90     image_height = 128
  91     image_width = 128
  92     depth_multiplier = 1
  93     pad_to_multiple = 1
  94     test_image = np.random.rand(4, image_height, image_width, 3)
  95     feature_extractor = self._create_feature_extractor(
  96         depth_multiplier, pad_to_multiple, use_keras=False)
  97     preprocessed_image = feature_extractor.preprocess(test_image)
  98     self.assertTrue(np.all(np.less_equal(np.abs(preprocessed_image), 1.0)))
  99 
 100   def test_has_fused_batchnorm(self):
 101     image_height = 128
 102     image_width = 128
 103     depth_multiplier = 1
 104     pad_to_multiple = 1
 105     image_placeholder = tf.placeholder(tf.float32,
 106                                        [1, image_height, image_width, 3])
 107     feature_extractor = self._create_feature_extractor(
 108         depth_multiplier, pad_to_multiple, use_keras=False)
 109     preprocessed_image = feature_extractor.preprocess(image_placeholder)
 110     _ = feature_extractor.extract_features(preprocessed_image)
 111     self.assertTrue(any('FusedBatchNorm' in op.type
 112                         for op in tf.get_default_graph().get_operations()))
