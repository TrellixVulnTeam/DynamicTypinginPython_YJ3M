  23 class SSDResnet50V1FeatureExtractorTest(
  24     ssd_resnet_v1_fpn_feature_extractor_testbase.
  25     SSDResnetFPNFeatureExtractorTestBase):
  26   """SSDResnet50v1Fpn feature extractor test."""
  27 
  28   def _create_feature_extractor(self, depth_multiplier, pad_to_multiple,
  29                                 use_explicit_padding=False, min_depth=32,
  30                                 use_keras=False):
  31     is_training = True
  32     if use_keras:
  33       return (ssd_resnet_v1_fpn_keras_feature_extractor.
  34               SSDResNet50V1FpnKerasFeatureExtractor(
  35                   is_training=is_training,
  36                   depth_multiplier=depth_multiplier,
  37                   min_depth=min_depth,
  38                   pad_to_multiple=pad_to_multiple,
  39                   conv_hyperparams=self._build_conv_hyperparams(
  40                       add_batch_norm=False),
  41                   freeze_batchnorm=False,
  42                   inplace_batchnorm_update=False,
  43                   name='ResNet50V1_FPN'))
  44     else:
  45       return (
  46           ssd_resnet_v1_fpn_feature_extractor.SSDResnet50V1FpnFeatureExtractor(
  47               is_training, depth_multiplier, min_depth, pad_to_multiple,
  48               self.conv_hyperparams_fn,
  49               use_explicit_padding=use_explicit_padding))
  50 
  51   def _resnet_scope_name(self, use_keras=False):
  52     if use_keras:
  53       return 'ResNet50V1_FPN'
  54     return 'resnet_v1_50'
