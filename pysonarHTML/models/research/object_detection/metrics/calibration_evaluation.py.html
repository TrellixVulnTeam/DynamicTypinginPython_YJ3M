  38   def __init__(self,
  39                categories,
  40                iou_threshold=0.5):
  41     """Constructor.
  42 
  43     Args:
  44       categories: A list of dicts, each of which has the following keys -
  45         'id': (required) an integer id uniquely identifying this category.
  46         'name': (required) string representing category name e.g., 'cat', 'dog'.
  47       iou_threshold: Threshold above which to consider a box as matched during
  48         evaluation.
  49     """
  50     super(CalibrationDetectionEvaluator, self).__init__(categories)
  51 
  52     # Constructing target_assigner to match detections to groundtruth.
  53     similarity_calc = region_similarity_calculator.IouSimilarity()
  54     matcher = argmax_matcher.ArgMaxMatcher(
  55         matched_threshold=iou_threshold, unmatched_threshold=iou_threshold)
  56     box_coder = mean_stddev_box_coder.MeanStddevBoxCoder(stddev=0.1)
  57     self._target_assigner = target_assigner.TargetAssigner(
  58         similarity_calc, matcher, box_coder)
  59 
  60   def match_single_image_info(self, image_info):
  61     """Match detections to groundtruth for a single image.
  62 
  63     Detections are matched to available groundtruth in the image based on the
  64     IOU threshold from the constructor.  The classes of the detections and
  65     groundtruth matches are then compared. Detections that do not have IOU above
  66     the required threshold or have different classes from their match are
  67     considered negative matches. All inputs in `image_info` originate or are
  68     inferred from the eval_dict passed to class method
  69     `get_estimator_eval_metric_ops`.
  70 
  71     Args:
  72       image_info: a tuple or list containing the following (in order):
  73         - gt_boxes: tf.float32 tensor of groundtruth boxes.
  74         - gt_classes: tf.int64 tensor of groundtruth classes associated with
  75             groundtruth boxes.
  76         - num_gt_box: scalar indicating the number of groundtruth boxes per
  77             image.
  78         - det_boxes: tf.float32 tensor of detection boxes.
  79         - det_classes: tf.int64 tensor of detection classes associated with
  80             detection boxes.
  81         - num_det_box: scalar indicating the number of detection boxes per
  82             image.
  83     Returns:
  84       is_class_matched: tf.int64 tensor identical in shape to det_boxes,
  85         indicating whether detection boxes matched with and had the same
  86         class as groundtruth annotations.
  87     """
  88     (gt_boxes, gt_classes, num_gt_box, det_boxes, det_classes,
  89      num_det_box) = image_info
  90     detection_boxes = det_boxes[:num_det_box]
  91     detection_classes = det_classes[:num_det_box]
  92     groundtruth_boxes = gt_boxes[:num_gt_box]
  93     groundtruth_classes = gt_classes[:num_gt_box]
  94     det_boxlist = box_list.BoxList(detection_boxes)
  95     gt_boxlist = box_list.BoxList(groundtruth_boxes)
  96 
  97     # Target assigner requires classes in one-hot format. An additional
  98     # dimension is required since gt_classes are 1-indexed; the zero index is
  99     # provided to all non-matches.
 100     one_hot_depth = tf.cast(tf.add(tf.reduce_max(groundtruth_classes), 1),
 101                             dtype=tf.int32)
 102     gt_classes_one_hot = tf.one_hot(
 103         groundtruth_classes, one_hot_depth, dtype=tf.float32)
 104     one_hot_cls_targets, _, _, _, _ = self._target_assigner.assign(
 105         det_boxlist,
 106         gt_boxlist,
 107         gt_classes_one_hot,
 108         unmatched_class_label=tf.zeros(shape=one_hot_depth, dtype=tf.float32))
 109     # Transform from one-hot back to indexes.
 110     cls_targets = tf.argmax(one_hot_cls_targets, axis=1)
 111     is_class_matched = tf.cast(
 112         tf.equal(tf.cast(cls_targets, tf.int64), detection_classes),
 113         dtype=tf.int64)
 114     return is_class_matched

 116   def get_estimator_eval_metric_ops(self, eval_dict):
 117     """Returns a dictionary of eval metric ops.
 118 
 119     Note that once value_op is called, the detections and groundtruth added via
 120     update_op are cleared.
 121 
 122     This function can take in groundtruth and detections for a batch of images,
 123     or for a single image. For the latter case, the batch dimension for input
 124     tensors need not be present.
 125 
 126     Args:
 127       eval_dict: A dictionary that holds tensors for evaluating object detection
 128         performance. For single-image evaluation, this dictionary may be
 129         produced from eval_util.result_dict_for_single_example(). If multi-image
 130         evaluation, `eval_dict` should contain the fields
 131         'num_groundtruth_boxes_per_image' and 'num_det_boxes_per_image' to
 132         properly unpad the tensors from the batch.
 133 
 134     Returns:
 135       a dictionary of metric names to tuple of value_op and update_op that can
 136       be used as eval metric ops in tf.estimator.EstimatorSpec. Note that all
 137       update ops must be run together and similarly all value ops must be run
 138       together to guarantee correct behaviour.
 139     """
 140     # Unpack items from the evaluation dictionary.
 141     input_data_fields = standard_fields.InputDataFields
 142     detection_fields = standard_fields.DetectionResultFields
 143     image_id = eval_dict[input_data_fields.key]
 144     groundtruth_boxes = eval_dict[input_data_fields.groundtruth_boxes]
 145     groundtruth_classes = eval_dict[input_data_fields.groundtruth_classes]
 146     detection_boxes = eval_dict[detection_fields.detection_boxes]
 147     detection_scores = eval_dict[detection_fields.detection_scores]
 148     detection_classes = eval_dict[detection_fields.detection_classes]
 149     num_gt_boxes_per_image = eval_dict.get(
 150         'num_groundtruth_boxes_per_image', None)
 151     num_det_boxes_per_image = eval_dict.get('num_det_boxes_per_image', None)
 152     is_annotated_batched = eval_dict.get('is_annotated', None)
 153 
 154     if not image_id.shape.as_list():
 155       # Apply a batch dimension to all tensors.
 156       image_id = tf.expand_dims(image_id, 0)
 157       groundtruth_boxes = tf.expand_dims(groundtruth_boxes, 0)
 158       groundtruth_classes = tf.expand_dims(groundtruth_classes, 0)
 159       detection_boxes = tf.expand_dims(detection_boxes, 0)
 160       detection_scores = tf.expand_dims(detection_scores, 0)
 161       detection_classes = tf.expand_dims(detection_classes, 0)
 162 
 163       if num_gt_boxes_per_image is None:
 164         num_gt_boxes_per_image = tf.shape(groundtruth_boxes)[1:2]
 165       else:
 166         num_gt_boxes_per_image = tf.expand_dims(num_gt_boxes_per_image, 0)
 167 
 168       if num_det_boxes_per_image is None:
 169         num_det_boxes_per_image = tf.shape(detection_boxes)[1:2]
 170       else:
 171         num_det_boxes_per_image = tf.expand_dims(num_det_boxes_per_image, 0)
 172 
 173       if is_annotated_batched is None:
 174         is_annotated_batched = tf.constant([True])
 175       else:
 176         is_annotated_batched = tf.expand_dims(is_annotated_batched, 0)
 177     else:
 178       if num_gt_boxes_per_image is None:
 179         num_gt_boxes_per_image = tf.tile(
 180             tf.shape(groundtruth_boxes)[1:2],
 181             multiples=tf.shape(groundtruth_boxes)[0:1])
 182       if num_det_boxes_per_image is None:
 183         num_det_boxes_per_image = tf.tile(
 184             tf.shape(detection_boxes)[1:2],
 185             multiples=tf.shape(detection_boxes)[0:1])
 186       if is_annotated_batched is None:
 187         is_annotated_batched = tf.ones_like(image_id, dtype=tf.bool)
 188 
 189     # Filter images based on is_annotated_batched and match detections.
 190     image_info = [tf.boolean_mask(tensor, is_annotated_batched) for tensor in
 191                   [groundtruth_boxes, groundtruth_classes,
 192                    num_gt_boxes_per_image, detection_boxes, detection_classes,
 193                    num_det_boxes_per_image]]
 194     is_class_matched = tf.map_fn(
 195         self.match_single_image_info, image_info, dtype=tf.int64)
 196     y_true = tf.squeeze(is_class_matched)
 197     y_pred = tf.squeeze(tf.boolean_mask(detection_scores, is_annotated_batched))
 198     ece, update_op = calibration_metrics.expected_calibration_error(
 199         y_true, y_pred)
 200     return {'CalibrationError/ExpectedCalibrationError': (ece, update_op)}
