  41   def _get_ece(self, ece_op, update_op):
  42     """Return scalar expected calibration error."""
  43     with self.test_session() as sess:
  44       metrics_vars = tf.get_collection(tf.GraphKeys.METRIC_VARIABLES)
  45       sess.run(tf.variables_initializer(var_list=metrics_vars))
  46       _ = sess.run(update_op)
  47     return sess.run(ece_op)
  48 
  49   def testGetECEWithMatchingGroundtruthAndDetections(self):
  50     """Tests that ECE is calculated correctly when box matches exist."""
  51     calibration_evaluator = calibration_evaluation.CalibrationDetectionEvaluator(
  52         _get_categories_list(), iou_threshold=0.5)
  53     input_data_fields = standard_fields.InputDataFields
  54     detection_fields = standard_fields.DetectionResultFields
  55     # All gt and detection boxes match.
  56     base_eval_dict = {
  57         input_data_fields.key:
  58             tf.constant(['image_1', 'image_2', 'image_3']),
  59         input_data_fields.groundtruth_boxes:
  60             tf.constant([[[100., 100., 200., 200.]],
  61                          [[50., 50., 100., 100.]],
  62                          [[25., 25., 50., 50.]]],
  63                         dtype=tf.float32),
  64         detection_fields.detection_boxes:
  65             tf.constant([[[100., 100., 200., 200.]],
  66                          [[50., 50., 100., 100.]],
  67                          [[25., 25., 50., 50.]]],
  68                         dtype=tf.float32),
  69         input_data_fields.groundtruth_classes:
  70             tf.constant([[1], [2], [3]], dtype=tf.int64),
  71         # Note that, in the zero ECE case, the detection class for image_2
  72         # should NOT match groundtruth, since the detection score is zero.
  73         detection_fields.detection_scores:
  74             tf.constant([[1.0], [0.0], [1.0]], dtype=tf.float32)
  75     }
  76 
  77     # Zero ECE (perfectly calibrated).
  78     zero_ece_eval_dict = base_eval_dict.copy()
  79     zero_ece_eval_dict[detection_fields.detection_classes] = tf.constant(
  80         [[1], [1], [3]], dtype=tf.int64)
  81     zero_ece_op, zero_ece_update_op = (
  82         calibration_evaluator.get_estimator_eval_metric_ops(zero_ece_eval_dict)
  83         ['CalibrationError/ExpectedCalibrationError'])
  84     zero_ece = self._get_ece(zero_ece_op, zero_ece_update_op)
  85     self.assertAlmostEqual(zero_ece, 0.0)
  86 
  87     # ECE of 1 (poorest calibration).
  88     one_ece_eval_dict = base_eval_dict.copy()
  89     one_ece_eval_dict[detection_fields.detection_classes] = tf.constant(
  90         [[3], [2], [1]], dtype=tf.int64)
  91     one_ece_op, one_ece_update_op = (
  92         calibration_evaluator.get_estimator_eval_metric_ops(one_ece_eval_dict)
  93         ['CalibrationError/ExpectedCalibrationError'])
  94     one_ece = self._get_ece(one_ece_op, one_ece_update_op)
  95     self.assertAlmostEqual(one_ece, 1.0)

  97   def testGetECEWithUnmatchedGroundtruthAndDetections(self):
  98     """Tests that ECE is correctly calculated when boxes are unmatched."""
  99     calibration_evaluator = calibration_evaluation.CalibrationDetectionEvaluator(
 100         _get_categories_list(), iou_threshold=0.5)
 101     input_data_fields = standard_fields.InputDataFields
 102     detection_fields = standard_fields.DetectionResultFields
 103     # No gt and detection boxes match.
 104     eval_dict = {
 105         input_data_fields.key:
 106             tf.constant(['image_1', 'image_2', 'image_3']),
 107         input_data_fields.groundtruth_boxes:
 108             tf.constant([[[100., 100., 200., 200.]],
 109                          [[50., 50., 100., 100.]],
 110                          [[25., 25., 50., 50.]]],
 111                         dtype=tf.float32),
 112         detection_fields.detection_boxes:
 113             tf.constant([[[50., 50., 100., 100.]],
 114                          [[25., 25., 50., 50.]],
 115                          [[100., 100., 200., 200.]]],
 116                         dtype=tf.float32),
 117         input_data_fields.groundtruth_classes:
 118             tf.constant([[1], [2], [3]], dtype=tf.int64),
 119         detection_fields.detection_classes:
 120             tf.constant([[1], [1], [3]], dtype=tf.int64),
 121         # Detection scores of zero when boxes are unmatched = ECE of zero.
 122         detection_fields.detection_scores:
 123             tf.constant([[0.0], [0.0], [0.0]], dtype=tf.float32)
 124     }
 125 
 126     ece_op, update_op = calibration_evaluator.get_estimator_eval_metric_ops(
 127         eval_dict)['CalibrationError/ExpectedCalibrationError']
 128     ece = self._get_ece(ece_op, update_op)
 129     self.assertAlmostEqual(ece, 0.0)

 131   def testGetECEWithBatchedDetections(self):
 132     """Tests that ECE is correct with multiple detections per image."""
 133     calibration_evaluator = calibration_evaluation.CalibrationDetectionEvaluator(
 134         _get_categories_list(), iou_threshold=0.5)
 135     input_data_fields = standard_fields.InputDataFields
 136     detection_fields = standard_fields.DetectionResultFields
 137     # Note that image_2 has mismatched classes and detection scores but should
 138     # still produce ECE of 0 because detection scores are also 0.
 139     eval_dict = {
 140         input_data_fields.key:
 141             tf.constant(['image_1', 'image_2', 'image_3']),
 142         input_data_fields.groundtruth_boxes:
 143             tf.constant([[[100., 100., 200., 200.], [50., 50., 100., 100.]],
 144                          [[50., 50., 100., 100.], [100., 100., 200., 200.]],
 145                          [[25., 25., 50., 50.], [100., 100., 200., 200.]]],
 146                         dtype=tf.float32),
 147         detection_fields.detection_boxes:
 148             tf.constant([[[100., 100., 200., 200.], [50., 50., 100., 100.]],
 149                          [[50., 50., 100., 100.], [25., 25., 50., 50.]],
 150                          [[25., 25., 50., 50.], [100., 100., 200., 200.]]],
 151                         dtype=tf.float32),
 152         input_data_fields.groundtruth_classes:
 153             tf.constant([[1, 2], [2, 3], [3, 1]], dtype=tf.int64),
 154         detection_fields.detection_classes:
 155             tf.constant([[1, 2], [1, 1], [3, 1]], dtype=tf.int64),
 156         detection_fields.detection_scores:
 157             tf.constant([[1.0, 1.0], [0.0, 0.0], [1.0, 1.0]], dtype=tf.float32)
 158     }
 159 
 160     ece_op, update_op = calibration_evaluator.get_estimator_eval_metric_ops(
 161         eval_dict)['CalibrationError/ExpectedCalibrationError']
 162     ece = self._get_ece(ece_op, update_op)
 163     self.assertAlmostEqual(ece, 0.0)

 182                          [[50., 50., 100., 100.]],
 183                          [[25., 25., 50., 50.]]],
 184                         dtype=tf.float32),
 185         input_data_fields.groundtruth_classes:
 186             tf.constant([[1], [2], [1]], dtype=tf.int64),
 187         detection_fields.detection_classes:
 188             tf.constant([[1], [1], [3]], dtype=tf.int64),
