  15 """Wrappers for third party pycocotools to be used within object_detection.
  16 
  17 Note that nothing in this file is tensorflow related and thus cannot
  18 be called directly as a slim metric, for example.
  19 
  20 TODO(jonathanhuang): wrap as a slim metric in metrics.py
  21 
  22 
  23 Usage example: given a set of images with ids in the list image_ids
  24 and corresponding lists of numpy arrays encoding groundtruth (boxes and classes)
  25 and detections (boxes, scores and classes), where elements of each list
  26 correspond to detections/annotations of a single image,
  27 then evaluation (in multi-class mode) can be invoked as follows:
  28 
  29   groundtruth_dict = coco_tools.ExportGroundtruthToCOCO(
  30       image_ids, groundtruth_boxes_list, groundtruth_classes_list,
  31       max_num_classes, output_path=None)
  32   detections_list = coco_tools.ExportDetectionsToCOCO(
  33       image_ids, detection_boxes_list, detection_scores_list,
  34       detection_classes_list, output_path=None)
  35   groundtruth = coco_tools.COCOWrapper(groundtruth_dict)
  36   detections = groundtruth.LoadAnnotations(detections_list)
  37   evaluator = coco_tools.COCOEvalWrapper(groundtruth, detections,
  38                                          agnostic_mode=False)
  39   metrics = evaluator.ComputeMetrics()
  40 
  41 """

  62 class COCOWrapper(coco.COCO):
  63   """Wrapper for the pycocotools COCO class."""
  64 
  65   def __init__(self, dataset, detection_type='bbox'):
  66     """COCOWrapper constructor.
  67 
  68     See http://mscoco.org/dataset/#format for a description of the format.
  69     By default, the coco.COCO class constructor reads from a JSON file.
  70     This function duplicates the same behavior but loads from a dictionary,
  71     allowing us to perform evaluation without writing to external storage.
  72 
  73     Args:
  74       dataset: a dictionary holding bounding box annotations in the COCO format.
  75       detection_type: type of detections being wrapped. Can be one of ['bbox',
  76         'segmentation']
  77 
  78     Raises:
  79       ValueError: if detection_type is unsupported.
  80     """
  81     supported_detection_types = ['bbox', 'segmentation']
  82     if detection_type not in supported_detection_types:
  83       raise ValueError('Unsupported detection type: {}. '
  84                        'Supported values are: {}'.format(
  85                            detection_type, supported_detection_types))
  86     self._detection_type = detection_type
  87     coco.COCO.__init__(self)
  88     self.dataset = dataset
  89     self.createIndex()
  90 
  91   def LoadAnnotations(self, annotations):
  92     """Load annotations dictionary into COCO datastructure.
  93 
  94     See http://mscoco.org/dataset/#format for a description of the annotations
  95     format.  As above, this function replicates the default behavior of the API
  96     but does not require writing to external storage.
  97 
  98     Args:
  99       annotations: python list holding object detection results where each
 100         detection is encoded as a dict with required keys ['image_id',
 101         'category_id', 'score'] and one of ['bbox', 'segmentation'] based on
 102         `detection_type`.
 103 
 104     Returns:
 105       a coco.COCO datastructure holding object detection annotations results
 106 
 107     Raises:
 108       ValueError: if annotations is not a list
 109       ValueError: if annotations do not correspond to the images contained
 110         in self.
 111     """
 112     results = coco.COCO()
 113     results.dataset['images'] = [img for img in self.dataset['images']]
 114 
 115     tf.logging.info('Loading and preparing annotation results...')
 116     tic = time.time()
 117 
 118     if not isinstance(annotations, list):
 119       raise ValueError('annotations is not a list of objects')
 120     annotation_img_ids = [ann['image_id'] for ann in annotations]
 121     if (set(annotation_img_ids) != (set(annotation_img_ids)
 122                                     & set(self.getImgIds()))):
 123       raise ValueError('Results do not correspond to current coco set')
 124     results.dataset['categories'] = copy.deepcopy(self.dataset['categories'])
 125     if self._detection_type == 'bbox':
 126       for idx, ann in enumerate(annotations):
 127         bb = ann['bbox']
 128         ann['area'] = bb[2] * bb[3]
 129         ann['id'] = idx + 1
 130         ann['iscrowd'] = 0
 131     elif self._detection_type == 'segmentation':
 132       for idx, ann in enumerate(annotations):
 133         ann['area'] = mask.area(ann['segmentation'])
 134         ann['bbox'] = mask.toBbox(ann['segmentation'])
 135         ann['id'] = idx + 1
 136         ann['iscrowd'] = 0
 137     tf.logging.info('DONE (t=%0.2fs)', (time.time() - tic))
 138 
 139     results.dataset['annotations'] = annotations
 140     results.createIndex()
 141     return results

 144 class COCOEvalWrapper(cocoeval.COCOeval):
 145   """Wrapper for the pycocotools COCOeval class.
 146 
 147   To evaluate, create two objects (groundtruth_dict and detections_list)
 148   using the conventions listed at http://mscoco.org/dataset/#format.
 149   Then call evaluation as follows:
 150 
 151     groundtruth = coco_tools.COCOWrapper(groundtruth_dict)
 152     detections = groundtruth.LoadAnnotations(detections_list)
 153     evaluator = coco_tools.COCOEvalWrapper(groundtruth, detections,
 154                                            agnostic_mode=False)
 155 
 156     metrics = evaluator.ComputeMetrics()
 157   """
 158 
 159   def __init__(self, groundtruth=None, detections=None, agnostic_mode=False,
 160                iou_type='bbox', oks_sigmas=None):
 161     """COCOEvalWrapper constructor.
 162 
 163     Note that for the area-based metrics to be meaningful, detection and
 164     groundtruth boxes must be in image coordinates measured in pixels.
 165 
 166     Args:
 167       groundtruth: a coco.COCO (or coco_tools.COCOWrapper) object holding
 168         groundtruth annotations
 169       detections: a coco.COCO (or coco_tools.COCOWrapper) object holding
 170         detections
 171       agnostic_mode: boolean (default: False).  If True, evaluation ignores
 172         class labels, treating all detections as proposals.
 173       iou_type: IOU type to use for evaluation. Supports `bbox', `segm`,
 174         `keypoints`.
 175       oks_sigmas: Float numpy array holding the OKS variances for keypoints.
 176     """
 177     cocoeval.COCOeval.__init__(self, groundtruth, detections, iouType=iou_type)
 178     if oks_sigmas is not None:
 179       self.params.kpt_oks_sigmas = oks_sigmas
 180     if agnostic_mode:
 181       self.params.useCats = 0
 182     self._iou_type = iou_type
 183 
 184   def GetCategory(self, category_id):
 185     """Fetches dictionary holding category information given category id.
 186 
 187     Args:
 188       category_id: integer id
 189     Returns:
 190       dictionary holding 'id', 'name'.
 191     """
 192     return self.cocoGt.cats[category_id]
 193 
 194   def GetAgnosticMode(self):
 195     """Returns true if COCO Eval is configured to evaluate in agnostic mode."""
 196     return self.params.useCats == 0
 197 
 198   def GetCategoryIdList(self):
 199     """Returns list of valid category ids."""
 200     return self.params.catIds
 201 
 202   def ComputeMetrics(self,
 203                      include_metrics_per_category=False,
 204                      all_metrics_per_category=False):
 205     """Computes detection/keypoint metrics.
 206 
 207     Args:
 208       include_metrics_per_category: If True, will include metrics per category.
 209       all_metrics_per_category: If true, include all the summery metrics for
 210         each category in per_category_ap. Be careful with setting it to true if
 211         you have more than handful of categories, because it will pollute
 212         your mldash.
 213 
 214     Returns:
 215       1. summary_metrics: a dictionary holding:
 216         'Precision/mAP': mean average precision over classes averaged over IOU
 217           thresholds ranging from .5 to .95 with .05 increments
 218         'Precision/mAP@.50IOU': mean average precision at 50% IOU
 219         'Precision/mAP@.75IOU': mean average precision at 75% IOU
 220         'Precision/mAP (small)': mean average precision for small objects
 221                         (area < 32^2 pixels). NOTE: not present for 'keypoints'
 222         'Precision/mAP (medium)': mean average precision for medium sized
 223                         objects (32^2 pixels < area < 96^2 pixels)
 224         'Precision/mAP (large)': mean average precision for large objects
 225                         (96^2 pixels < area < 10000^2 pixels)
 226         'Recall/AR@1': average recall with 1 detection
 227         'Recall/AR@10': average recall with 10 detections
 228         'Recall/AR@100': average recall with 100 detections
 229         'Recall/AR@100 (small)': average recall for small objects with 100
 230           detections. NOTE: not present for 'keypoints'
 231         'Recall/AR@100 (medium)': average recall for medium objects with 100
 232           detections
 233         'Recall/AR@100 (large)': average recall for large objects with 100
 234           detections
 235       2. per_category_ap: a dictionary holding category specific results with
 236         keys of the form: 'Precision mAP ByCategory/category'
 237         (without the supercategory part if no supercategories exist).
 238         For backward compatibility 'PerformanceByCategory' is included in the
 239         output regardless of all_metrics_per_category.
 240         If evaluating class-agnostic mode, per_category_ap is an empty
 241         dictionary.
 242 
 243     Raises:
 244       ValueError: If category_stats does not exist.
 245     """
 246     self.evaluate()
 247     self.accumulate()
 248     self.summarize()
 249 
 250     summary_metrics = {}
 251     if self._iou_type in ['bbox', 'segm']:
 252       summary_metrics = OrderedDict([('Precision/mAP', self.stats[0]),
 253                                      ('Precision/mAP@.50IOU', self.stats[1]),
 254                                      ('Precision/mAP@.75IOU', self.stats[2]),
 255                                      ('Precision/mAP (small)', self.stats[3]),
 256                                      ('Precision/mAP (medium)', self.stats[4]),
 257                                      ('Precision/mAP (large)', self.stats[5]),
 258                                      ('Recall/AR@1', self.stats[6]),
 259                                      ('Recall/AR@10', self.stats[7]),
 260                                      ('Recall/AR@100', self.stats[8]),
 261                                      ('Recall/AR@100 (small)', self.stats[9]),
 262                                      ('Recall/AR@100 (medium)', self.stats[10]),
 263                                      ('Recall/AR@100 (large)', self.stats[11])])
 264     elif self._iou_type == 'keypoints':
 265       category_id = self.GetCategoryIdList()[0]
 266       category_name = self.GetCategory(category_id)['name']
 267       summary_metrics = OrderedDict([])
 268       summary_metrics['Precision/mAP ByCategory/{}'.format(
 269           category_name)] = self.stats[0]
 270       summary_metrics['Precision/mAP@.50IOU ByCategory/{}'.format(
 271           category_name)] = self.stats[1]
 272       summary_metrics['Precision/mAP@.75IOU ByCategory/{}'.format(
 273           category_name)] = self.stats[2]
 274       summary_metrics['Precision/mAP (medium) ByCategory/{}'.format(
 275           category_name)] = self.stats[3]
 276       summary_metrics['Precision/mAP (large) ByCategory/{}'.format(
 277           category_name)] = self.stats[4]
 278       summary_metrics['Recall/AR@1 ByCategory/{}'.format(
 279           category_name)] = self.stats[5]
 280       summary_metrics['Recall/AR@10 ByCategory/{}'.format(
 281           category_name)] = self.stats[6]
 282       summary_metrics['Recall/AR@100 ByCategory/{}'.format(
 283           category_name)] = self.stats[7]
 284       summary_metrics['Recall/AR@100 (medium) ByCategory/{}'.format(
 285           category_name)] = self.stats[8]
 286       summary_metrics['Recall/AR@100 (large) ByCategory/{}'.format(
 287           category_name)] = self.stats[9]
 288     if not include_metrics_per_category:
 289       return summary_metrics, {}
 290     if not hasattr(self, 'category_stats'):
 291       raise ValueError('Category stats do not exist')
 292     per_category_ap = OrderedDict([])
 293     if self.GetAgnosticMode():
 294       return summary_metrics, per_category_ap
 295     for category_index, category_id in enumerate(self.GetCategoryIdList()):
 296       category = self.GetCategory(category_id)['name']
 297       # Kept for backward compatilbility
 298       per_category_ap['PerformanceByCategory/mAP/{}'.format(
 299           category)] = self.category_stats[0][category_index]
 300       if all_metrics_per_category:
 301         per_category_ap['Precision mAP ByCategory/{}'.format(
 302             category)] = self.category_stats[0][category_index]
 303         per_category_ap['Precision mAP@.50IOU ByCategory/{}'.format(
 304             category)] = self.category_stats[1][category_index]
 305         per_category_ap['Precision mAP@.75IOU ByCategory/{}'.format(
 306             category)] = self.category_stats[2][category_index]
 307         per_category_ap['Precision mAP (small) ByCategory/{}'.format(
 308             category)] = self.category_stats[3][category_index]
 309         per_category_ap['Precision mAP (medium) ByCategory/{}'.format(
 310             category)] = self.category_stats[4][category_index]
 311         per_category_ap['Precision mAP (large) ByCategory/{}'.format(
 312             category)] = self.category_stats[5][category_index]
 313         per_category_ap['Recall AR@1 ByCategory/{}'.format(
 314             category)] = self.category_stats[6][category_index]
 315         per_category_ap['Recall AR@10 ByCategory/{}'.format(
 316             category)] = self.category_stats[7][category_index]
 317         per_category_ap['Recall AR@100 ByCategory/{}'.format(
 318             category)] = self.category_stats[8][category_index]
 319         per_category_ap['Recall AR@100 (small) ByCategory/{}'.format(
 320             category)] = self.category_stats[9][category_index]
 321         per_category_ap['Recall AR@100 (medium) ByCategory/{}'.format(
 322             category)] = self.category_stats[10][category_index]
 323         per_category_ap['Recall AR@100 (large) ByCategory/{}'.format(
 324             category)] = self.category_stats[11][category_index]
 325 
 326     return summary_metrics, per_category_ap

 329 def _ConvertBoxToCOCOFormat(box):
 330   """Converts a box in [ymin, xmin, ymax, xmax] format to COCO format.
 331 
 332   This is a utility function for converting from our internal
 333   [ymin, xmin, ymax, xmax] convention to the convention used by the COCO API
 334   i.e., [xmin, ymin, width, height].
 335 
 336   Args:
 337     box: a [ymin, xmin, ymax, xmax] numpy array
 338 
 339   Returns:
 340     a list of floats representing [xmin, ymin, width, height]
 341   """
 342   return [float(box[1]), float(box[0]), float(box[3] - box[1]),
 343           float(box[2] - box[0])]

 346 def _RleCompress(masks):
 347   """Compresses mask using Run-length encoding provided by pycocotools.
 348 
 349   Args:
 350     masks: uint8 numpy array of shape [mask_height, mask_width] with values in
 351     {0, 1}.
 352 
 353   Returns:
 354     A pycocotools Run-length encoding of the mask.
 355   """
 356   return mask.encode(np.asfortranarray(masks))

 359 def ExportSingleImageGroundtruthToCoco(image_id,
 360                                        next_annotation_id,
 361                                        category_id_set,
 362                                        groundtruth_boxes,
 363                                        groundtruth_classes,
 364                                        groundtruth_keypoints=None,
 365                                        groundtruth_keypoint_visibilities=None,
 366                                        groundtruth_masks=None,
 367                                        groundtruth_is_crowd=None,
 368                                        groundtruth_area=None):
 369   """Export groundtruth of a single image to COCO format.
 370 
 371   This function converts groundtruth detection annotations represented as numpy
 372   arrays to dictionaries that can be ingested by the COCO evaluation API. Note
 373   that the image_ids provided here must match the ones given to
 374   ExportSingleImageDetectionsToCoco. We assume that boxes and classes are in
 375   correspondence - that is: groundtruth_boxes[i, :], and
 376   groundtruth_classes[i] are associated with the same groundtruth annotation.
 377 
 378   In the exported result, "area" fields are always set to the area of the
 379   groundtruth bounding box.
 380 
 381   Args:
 382     image_id: a unique image identifier either of type integer or string.
 383     next_annotation_id: integer specifying the first id to use for the
 384       groundtruth annotations. All annotations are assigned a continuous integer
 385       id starting from this value.
 386     category_id_set: A set of valid class ids. Groundtruth with classes not in
 387       category_id_set are dropped.
 388     groundtruth_boxes: numpy array (float32) with shape [num_gt_boxes, 4]
 389     groundtruth_classes: numpy array (int) with shape [num_gt_boxes]
 390     groundtruth_keypoints: optional float numpy array of keypoints
 391       with shape [num_gt_boxes, num_keypoints, 2].
 392     groundtruth_keypoint_visibilities: optional integer numpy array of keypoint
 393       visibilities with shape [num_gt_boxes, num_keypoints]. Integer is treated
 394       as an enum with 0=not labels, 1=labeled but not visible and 2=labeled and
 395       visible.
 396     groundtruth_masks: optional uint8 numpy array of shape [num_detections,
 397       image_height, image_width] containing detection_masks.
 398     groundtruth_is_crowd: optional numpy array (int) with shape [num_gt_boxes]
 399       indicating whether groundtruth boxes are crowd.
 400     groundtruth_area: numpy array (float32) with shape [num_gt_boxes]. If
 401       provided, then the area values (in the original absolute coordinates) will
 402       be populated instead of calculated from bounding box coordinates.
 403 
 404   Returns:
 405     a list of groundtruth annotations for a single image in the COCO format.
 406 
 407   Raises:
 408     ValueError: if (1) groundtruth_boxes and groundtruth_classes do not have the
 409       right lengths or (2) if each of the elements inside these lists do not
 410       have the correct shapes or (3) if image_ids are not integers
 411   """
 412 
 413   if len(groundtruth_classes.shape) != 1:
 414     raise ValueError('groundtruth_classes is '
 415                      'expected to be of rank 1.')
 416   if len(groundtruth_boxes.shape) != 2:
 417     raise ValueError('groundtruth_boxes is expected to be of '
 418                      'rank 2.')
 419   if groundtruth_boxes.shape[1] != 4:
 420     raise ValueError('groundtruth_boxes should have '
 421                      'shape[1] == 4.')
 422   num_boxes = groundtruth_classes.shape[0]
 423   if num_boxes != groundtruth_boxes.shape[0]:
 424     raise ValueError('Corresponding entries in groundtruth_classes, '
 425                      'and groundtruth_boxes should have '
 426                      'compatible shapes (i.e., agree on the 0th dimension).'
 427                      'Classes shape: %d. Boxes shape: %d. Image ID: %s' % (
 428                          groundtruth_classes.shape[0],
 429                          groundtruth_boxes.shape[0], image_id))
 430   has_is_crowd = groundtruth_is_crowd is not None
 431   if has_is_crowd and len(groundtruth_is_crowd.shape) != 1:
 432     raise ValueError('groundtruth_is_crowd is expected to be of rank 1.')
 433   has_keypoints = groundtruth_keypoints is not None
 434   has_keypoint_visibilities = groundtruth_keypoint_visibilities is not None
 435   if has_keypoints and not has_keypoint_visibilities:
 436     groundtruth_keypoint_visibilities = np.full(
 437         (num_boxes, groundtruth_keypoints.shape[1]), 2)
 438   groundtruth_list = []
 439   for i in range(num_boxes):
 440     if groundtruth_classes[i] in category_id_set:
 441       iscrowd = groundtruth_is_crowd[i] if has_is_crowd else 0
 442       if groundtruth_area is not None and groundtruth_area[i] > 0:
 443         area = float(groundtruth_area[i])
 444       else:
 445         area = float((groundtruth_boxes[i, 2] - groundtruth_boxes[i, 0]) *
 446                      (groundtruth_boxes[i, 3] - groundtruth_boxes[i, 1]))
 447       export_dict = {
 448           'id':
 449               next_annotation_id + i,
 450           'image_id':
 451               image_id,
 452           'category_id':
 453               int(groundtruth_classes[i]),
 454           'bbox':
 455               list(_ConvertBoxToCOCOFormat(groundtruth_boxes[i, :])),
 456           'area': area,
 457           'iscrowd':
 458               iscrowd
 459       }
 460       if groundtruth_masks is not None:
 461         export_dict['segmentation'] = _RleCompress(groundtruth_masks[i])
 462       if has_keypoints:
 463         keypoints = groundtruth_keypoints[i]
 464         visibilities = np.reshape(groundtruth_keypoint_visibilities[i], [-1])
 465         coco_keypoints = []
 466         num_valid_keypoints = 0
 467         for keypoint, visibility in zip(keypoints, visibilities):
 468           # Convert from [y, x] to [x, y] as mandated by COCO.
 469           coco_keypoints.append(float(keypoint[1]))
 470           coco_keypoints.append(float(keypoint[0]))
 471           coco_keypoints.append(int(visibility))
 472           if int(visibility) > 0:
 473             num_valid_keypoints = num_valid_keypoints + 1
 474         export_dict['keypoints'] = coco_keypoints
 475         export_dict['num_keypoints'] = num_valid_keypoints
 476 
 477       groundtruth_list.append(export_dict)
 478   return groundtruth_list

 481 def ExportGroundtruthToCOCO(image_ids,
 482                             groundtruth_boxes,
 483                             groundtruth_classes,
 484                             categories,
 485                             output_path=None):
 486   """Export groundtruth detection annotations in numpy arrays to COCO API.
 487 
 488   This function converts a set of groundtruth detection annotations represented
 489   as numpy arrays to dictionaries that can be ingested by the COCO API.
 490   Inputs to this function are three lists: image ids for each groundtruth image,
 491   groundtruth boxes for each image and groundtruth classes respectively.
 492   Note that the image_ids provided here must match the ones given to the
 493   ExportDetectionsToCOCO function in order for evaluation to work properly.
 494   We assume that for each image, boxes, scores and classes are in
 495   correspondence --- that is: image_id[i], groundtruth_boxes[i, :] and
 496   groundtruth_classes[i] are associated with the same groundtruth annotation.
 497 
 498   In the exported result, "area" fields are always set to the area of the
 499   groundtruth bounding box and "iscrowd" fields are always set to 0.
 500   TODO(jonathanhuang): pass in "iscrowd" array for evaluating on COCO dataset.
 501 
 502   Args:
 503     image_ids: a list of unique image identifier either of type integer or
 504       string.
 505     groundtruth_boxes: list of numpy arrays with shape [num_gt_boxes, 4]
 506       (note that num_gt_boxes can be different for each entry in the list)
 507     groundtruth_classes: list of numpy arrays (int) with shape [num_gt_boxes]
 508       (note that num_gt_boxes can be different for each entry in the list)
 509     categories: a list of dictionaries representing all possible categories.
 510         Each dict in this list has the following keys:
 511           'id': (required) an integer id uniquely identifying this category
 512           'name': (required) string representing category name
 513             e.g., 'cat', 'dog', 'pizza'
 514           'supercategory': (optional) string representing the supercategory
 515             e.g., 'animal', 'vehicle', 'food', etc
 516     output_path: (optional) path for exporting result to JSON
 517   Returns:
 518     dictionary that can be read by COCO API
 519   Raises:
 520     ValueError: if (1) groundtruth_boxes and groundtruth_classes do not have the
 521       right lengths or (2) if each of the elements inside these lists do not
 522       have the correct shapes or (3) if image_ids are not integers
 523   """
 524   category_id_set = set([cat['id'] for cat in categories])
 525   groundtruth_export_list = []
 526   image_export_list = []
 527   if not len(image_ids) == len(groundtruth_boxes) == len(groundtruth_classes):
 528     raise ValueError('Input lists must have the same length')
 529 
 530   # For reasons internal to the COCO API, it is important that annotation ids
 531   # are not equal to zero; we thus start counting from 1.
 532   annotation_id = 1
 533   for image_id, boxes, classes in zip(image_ids, groundtruth_boxes,
 534                                       groundtruth_classes):
 535     image_export_list.append({'id': image_id})
 536     groundtruth_export_list.extend(ExportSingleImageGroundtruthToCoco(
 537         image_id,
 538         annotation_id,
 539         category_id_set,
 540         boxes,
 541         classes))
 542     num_boxes = classes.shape[0]
 543     annotation_id += num_boxes
 544 
 545   groundtruth_dict = {
 546       'annotations': groundtruth_export_list,
 547       'images': image_export_list,
 548       'categories': categories
 549   }
 550   if output_path:
 551     with tf.gfile.GFile(output_path, 'w') as fid:

 556 def ExportSingleImageDetectionBoxesToCoco(image_id,
 557                                           category_id_set,
 558                                           detection_boxes,
 559                                           detection_scores,
 560                                           detection_classes,
 561                                           detection_keypoints=None,
 562                                           detection_keypoint_visibilities=None):
 563   """Export detections of a single image to COCO format.
 564 
 565   This function converts detections represented as numpy arrays to dictionaries
 566   that can be ingested by the COCO evaluation API. Note that the image_ids
 567   provided here must match the ones given to the
 568   ExporSingleImageDetectionBoxesToCoco. We assume that boxes, and classes are in
 569   correspondence - that is: boxes[i, :], and classes[i]
 570   are associated with the same groundtruth annotation.
 571 
 572   Args:
 573     image_id: unique image identifier either of type integer or string.
 574     category_id_set: A set of valid class ids. Detections with classes not in
 575       category_id_set are dropped.
 576     detection_boxes: float numpy array of shape [num_detections, 4] containing
 577       detection boxes.
 578     detection_scores: float numpy array of shape [num_detections] containing
 579       scored for the detection boxes.
 580     detection_classes: integer numpy array of shape [num_detections] containing
 581       the classes for detection boxes.
 582     detection_keypoints: optional float numpy array of keypoints
 583       with shape [num_detections, num_keypoints, 2].
 584     detection_keypoint_visibilities: optional integer numpy array of keypoint
 585       visibilities with shape [num_detections, num_keypoints]. Integer is
 586       treated as an enum with 0=not labels, 1=labeled but not visible and
 587       2=labeled and visible.
 588 
 589   Returns:
 590     a list of detection annotations for a single image in the COCO format.
 591 
 592   Raises:
 593     ValueError: if (1) detection_boxes, detection_scores and detection_classes
 594       do not have the right lengths or (2) if each of the elements inside these
 595       lists do not have the correct shapes or (3) if image_ids are not integers.
 596   """
 597 
 598   if len(detection_classes.shape) != 1 or len(detection_scores.shape) != 1:
 599     raise ValueError('All entries in detection_classes and detection_scores'
 600                      'expected to be of rank 1.')
 601   if len(detection_boxes.shape) != 2:
 602     raise ValueError('All entries in detection_boxes expected to be of '
 603                      'rank 2.')
 604   if detection_boxes.shape[1] != 4:
 605     raise ValueError('All entries in detection_boxes should have '
 606                      'shape[1] == 4.')
 607   num_boxes = detection_classes.shape[0]
 608   if not num_boxes == detection_boxes.shape[0] == detection_scores.shape[0]:
 609     raise ValueError('Corresponding entries in detection_classes, '
 610                      'detection_scores and detection_boxes should have '
 611                      'compatible shapes (i.e., agree on the 0th dimension). '
 612                      'Classes shape: %d. Boxes shape: %d. '
 613                      'Scores shape: %d' % (
 614                          detection_classes.shape[0], detection_boxes.shape[0],
 615                          detection_scores.shape[0]
 616                      ))
 617   detections_list = []
 618   for i in range(num_boxes):
 619     if detection_classes[i] in category_id_set:
 620       export_dict = {
 621           'image_id':
 622               image_id,
 623           'category_id':
 624               int(detection_classes[i]),
 625           'bbox':
 626               list(_ConvertBoxToCOCOFormat(detection_boxes[i, :])),
 627           'score':
 628               float(detection_scores[i]),
 629       }
 630       if detection_keypoints is not None:
 631         keypoints = detection_keypoints[i]
 632         num_keypoints = keypoints.shape[0]
 633         if detection_keypoint_visibilities is None:
 634           detection_keypoint_visibilities = np.full((num_boxes, num_keypoints),
 635                                                     2)
 636         visibilities = np.reshape(detection_keypoint_visibilities[i], [-1])
 637         coco_keypoints = []
 638         for keypoint, visibility in zip(keypoints, visibilities):
 639           # Convert from [y, x] to [x, y] as mandated by COCO.
 640           coco_keypoints.append(float(keypoint[1]))
 641           coco_keypoints.append(float(keypoint[0]))
 642           coco_keypoints.append(int(visibility))
 643         export_dict['keypoints'] = coco_keypoints
 644         export_dict['num_keypoints'] = num_keypoints
 645       detections_list.append(export_dict)
 646 
 647   return detections_list
 648

 708 def ExportDetectionsToCOCO(image_ids,
 709                            detection_boxes,
 710                            detection_scores,
 711                            detection_classes,
 712                            categories,
 713                            output_path=None):
 714   """Export detection annotations in numpy arrays to COCO API.
 715 
 716   This function converts a set of predicted detections represented
 717   as numpy arrays to dictionaries that can be ingested by the COCO API.
 718   Inputs to this function are lists, consisting of boxes, scores and
 719   classes, respectively, corresponding to each image for which detections
 720   have been produced.  Note that the image_ids provided here must
 721   match the ones given to the ExportGroundtruthToCOCO function in order
 722   for evaluation to work properly.
 723 
 724   We assume that for each image, boxes, scores and classes are in
 725   correspondence --- that is: detection_boxes[i, :], detection_scores[i] and
 726   detection_classes[i] are associated with the same detection.
 727 
 728   Args:
 729     image_ids: a list of unique image identifier either of type integer or
 730       string.
 731     detection_boxes: list of numpy arrays with shape [num_detection_boxes, 4]
 732     detection_scores: list of numpy arrays (float) with shape
 733       [num_detection_boxes]. Note that num_detection_boxes can be different
 734       for each entry in the list.
 735     detection_classes: list of numpy arrays (int) with shape
 736       [num_detection_boxes]. Note that num_detection_boxes can be different
 737       for each entry in the list.
 738     categories: a list of dictionaries representing all possible categories.
 739       Each dict in this list must have an integer 'id' key uniquely identifying
 740       this category.
 741     output_path: (optional) path for exporting result to JSON
 742 
 743   Returns:
 744     list of dictionaries that can be read by COCO API, where each entry
 745     corresponds to a single detection and has keys from:
 746     ['image_id', 'category_id', 'bbox', 'score'].
 747   Raises:
 748     ValueError: if (1) detection_boxes and detection_classes do not have the
 749       right lengths or (2) if each of the elements inside these lists do not
 750       have the correct shapes or (3) if image_ids are not integers.
 751   """
 752   category_id_set = set([cat['id'] for cat in categories])
 753   detections_export_list = []
 754   if not (len(image_ids) == len(detection_boxes) == len(detection_scores) ==
 755           len(detection_classes)):
 756     raise ValueError('Input lists must have the same length')
 757   for image_id, boxes, scores, classes in zip(image_ids, detection_boxes,
 758                                               detection_scores,
 759                                               detection_classes):
 760     detections_export_list.extend(ExportSingleImageDetectionBoxesToCoco(
 761         image_id,
 762         category_id_set,
 763         boxes,
 764         scores,
 765         classes))
 766   if output_path:
 767     with tf.gfile.GFile(output_path, 'w') as fid:
 768       json_utils.Dump(detections_export_list, fid, float_digits=4, indent=2)
 769   return detections_export_list

 773                          detection_masks,
 774                          detection_scores,
 775                          detection_classes,
 776                          categories,
 777                          output_path=None):
 778   """Export segmentation masks in numpy arrays to COCO API.
 779 
 780   This function converts a set of predicted instance masks represented
 781   as numpy arrays to dictionaries that can be ingested by the COCO API.
 782   Inputs to this function are lists, consisting of segments, scores and
 783   classes, respectively, corresponding to each image for which detections
 784   have been produced.

 788   (e.g. in map reduce), otherwise the memory consumption is large.
 789 
 790   We assume that for each image, masks, scores and classes are in
 791   correspondence --- that is: detection_masks[i, :, :, :], detection_scores[i]
 792   and detection_classes[i] are associated with the same detection.
 793 
 794   Args:
 795     image_ids: list of image ids (typically ints or strings)
 796     detection_masks: list of numpy arrays with shape [num_detection, h, w, 1]
 797       and type uint8. The height and width should match the shape of
 798       corresponding image.
 799     detection_scores: list of numpy arrays (float) with shape
 800       [num_detection]. Note that num_detection can be different
 801       for each entry in the list.
 802     detection_classes: list of numpy arrays (int) with shape
 803       [num_detection]. Note that num_detection can be different
 804       for each entry in the list.
 805     categories: a list of dictionaries representing all possible categories.
 806       Each dict in this list must have an integer 'id' key uniquely identifying
 807       this category.
 808     output_path: (optional) path for exporting result to JSON
 809 
 810   Returns:
 811     list of dictionaries that can be read by COCO API, where each entry
 812     corresponds to a single detection and has keys from:
 813     ['image_id', 'category_id', 'segmentation', 'score'].
 814 
 815   Raises:
 816     ValueError: if detection_masks and detection_classes do not have the
 817       right lengths or if each of the elements inside these lists do not
 818       have the correct shapes.
 819   """
 820   if not (len(image_ids) == len(detection_masks) == len(detection_scores) ==
 821           len(detection_classes)):
 822     raise ValueError('Input lists must have the same length')
 823 
 824   segment_export_list = []
 825   for image_id, masks, scores, classes in zip(image_ids, detection_masks,
 826                                               detection_scores,
 827                                               detection_classes):
 828 
 829     if len(classes.shape) != 1 or len(scores.shape) != 1:
 830       raise ValueError('All entries in detection_classes and detection_scores'
 831                        'expected to be of rank 1.')
 832     if len(masks.shape) != 4:
 833       raise ValueError('All entries in masks expected to be of '
 834                        'rank 4. Given {}'.format(masks.shape))
 835 
 836     num_boxes = classes.shape[0]
 837     if not num_boxes == masks.shape[0] == scores.shape[0]:
 838       raise ValueError('Corresponding entries in segment_classes, '
 839                        'detection_scores and detection_boxes should have '
 840                        'compatible shapes (i.e., agree on the 0th dimension).')
 841 
 842     category_id_set = set([cat['id'] for cat in categories])
 843     segment_export_list.extend(ExportSingleImageDetectionMasksToCoco(
 844         image_id, category_id_set, np.squeeze(masks, axis=3), scores, classes))
 845 
 846   if output_path:
 847     with tf.gfile.GFile(output_path, 'w') as fid:
 848       json_utils.Dump(segment_export_list, fid, float_digits=4, indent=2)
 849   return segment_export_list

 852 def ExportKeypointsToCOCO(image_ids,
 853                           detection_keypoints,
 854                           detection_scores,
 855                           detection_classes,
 856                           categories,
 857                           output_path=None):
 858   """Exports keypoints in numpy arrays to COCO API.
 859 
 860   This function converts a set of predicted keypoints represented
 861   as numpy arrays to dictionaries that can be ingested by the COCO API.
 862   Inputs to this function are lists, consisting of keypoints, scores and
 863   classes, respectively, corresponding to each image for which detections
 864   have been produced.
 865 
 866   We assume that for each image, keypoints, scores and classes are in
 867   correspondence --- that is: detection_keypoints[i, :, :, :],
 868   detection_scores[i] and detection_classes[i] are associated with the same
 869   detection.
 870 
 871   Args:
 872     image_ids: list of image ids (typically ints or strings)
 873     detection_keypoints: list of numpy arrays with shape
 874       [num_detection, num_keypoints, 2] and type float32 in absolute
 875       x-y coordinates.
 876     detection_scores: list of numpy arrays (float) with shape
 877       [num_detection]. Note that num_detection can be different
 878       for each entry in the list.
 879     detection_classes: list of numpy arrays (int) with shape
 880       [num_detection]. Note that num_detection can be different
 881       for each entry in the list.
 882     categories: a list of dictionaries representing all possible categories.
 883       Each dict in this list must have an integer 'id' key uniquely identifying
 884       this category and an integer 'num_keypoints' key specifying the number of
 885       keypoints the category has.
 886     output_path: (optional) path for exporting result to JSON
 887 
 888   Returns:
 889     list of dictionaries that can be read by COCO API, where each entry
 890     corresponds to a single detection and has keys from:
 891     ['image_id', 'category_id', 'keypoints', 'score'].
 892 
 893   Raises:
 894     ValueError: if detection_keypoints and detection_classes do not have the
 895       right lengths or if each of the elements inside these lists do not
 896       have the correct shapes.
 897   """
 898   if not (len(image_ids) == len(detection_keypoints) ==
 899           len(detection_scores) == len(detection_classes)):
 900     raise ValueError('Input lists must have the same length')
 901 
 902   keypoints_export_list = []
 903   for image_id, keypoints, scores, classes in zip(
 904       image_ids, detection_keypoints, detection_scores, detection_classes):
 905 
 906     if len(classes.shape) != 1 or len(scores.shape) != 1:
 907       raise ValueError('All entries in detection_classes and detection_scores'
 908                        'expected to be of rank 1.')
 909     if len(keypoints.shape) != 3:
 910       raise ValueError('All entries in keypoints expected to be of '
 911                        'rank 3. Given {}'.format(keypoints.shape))
 912 
 913     num_boxes = classes.shape[0]
 914     if not num_boxes == keypoints.shape[0] == scores.shape[0]:
 915       raise ValueError('Corresponding entries in detection_classes, '
 916                        'detection_keypoints, and detection_scores should have '
 917                        'compatible shapes (i.e., agree on the 0th dimension).')
 918 
 919     category_id_set = set([cat['id'] for cat in categories])
 920     category_id_to_num_keypoints_map = {
 921         cat['id']: cat['num_keypoints'] for cat in categories
 922         if 'num_keypoints' in cat}
 923 
 924     for i in range(num_boxes):
 925       if classes[i] not in category_id_set:
 926         raise ValueError('class id should be in category_id_set\n')
 927 
 928       if classes[i] in category_id_to_num_keypoints_map:
 929         num_keypoints = category_id_to_num_keypoints_map[classes[i]]
 930         # Adds extra ones to indicate the visibility for each keypoint as is
 931         # recommended by MSCOCO.
 932         instance_keypoints = np.concatenate(
 933             [keypoints[i, 0:num_keypoints, :],
 934              np.expand_dims(np.ones(num_keypoints), axis=1)],
 935             axis=1).astype(int)
 936 
 937         instance_keypoints = instance_keypoints.flatten().tolist()
 938         keypoints_export_list.append({
 939             'image_id': image_id,
 940             'category_id': int(classes[i]),
 941             'keypoints': instance_keypoints,
 942             'score': float(scores[i])
 943         })
 944 
 945   if output_path:
 946     with tf.gfile.GFile(output_path, 'w') as fid:
 947       json_utils.Dump(keypoints_export_list, fid, float_digits=4, indent=2)
 948   return keypoints_export_list
