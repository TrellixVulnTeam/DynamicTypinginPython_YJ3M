  15 r"""Runs evaluation using OpenImages groundtruth and predictions.
  16 
  17 Example usage:
  18 python \
  19 models/research/object_detection/metrics/oid_vrd_challenge_evaluation.py \
  20     --input_annotations_vrd=/path/to/input/annotations-human-bbox.csv \
  21     --input_annotations_labels=/path/to/input/annotations-label.csv \
  22     --input_class_labelmap=/path/to/input/class_labelmap.pbtxt \
  23     --input_relationship_labelmap=/path/to/input/relationship_labelmap.pbtxt \
  24     --input_predictions=/path/to/input/predictions.csv \
  25     --output_metrics=/path/to/output/metric.csv \
  26 
  27 CSVs with bounding box annotations and image label (including the image URLs)
  28 can be downloaded from the Open Images Challenge website:
  29 https://storage.googleapis.com/openimages/web/challenge.html
  30 The format of the input csv and the metrics itself are described on the
  31 challenge website.
  32 """
  33 
  34 from __future__ import absolute_import
  35 from __future__ import division
  36 from __future__ import print_function
  37 
  38 import argparse
  39 import pandas as pd
  40 from google.protobuf import text_format
  41 
  42 from object_detection.metrics import io_utils
  43 from object_detection.metrics import oid_vrd_challenge_evaluation_utils as utils
  44 from object_detection.protos import string_int_label_map_pb2
  45 from object_detection.utils import vrd_evaluation
  46 
  47 
  48 def _load_labelmap(labelmap_path):
  49   """Loads labelmap from the labelmap path.
  50 
  51   Args:
  52     labelmap_path: Path to the labelmap.
  53 
  54   Returns:
  55     A dictionary mapping class name to class numerical id.
  56   """
  57 
  58   label_map = string_int_label_map_pb2.StringIntLabelMap()
  59   with open(labelmap_path, 'r') as fid:
  60     label_map_string = fid.read()
  61     text_format.Merge(label_map_string, label_map)
  62   labelmap_dict = {}
  63   for item in label_map.item:
  64     labelmap_dict[item.name] = item.id
  65   return labelmap_dict
  66 
  67 
  68 def _swap_labelmap_dict(labelmap_dict):
  69   """Swaps keys and labels in labelmap.
  70 
  71   Args:
  72     labelmap_dict: Input dictionary.
  73 
  74   Returns:
  75     A dictionary mapping class name to class numerical id.
  76   """
  77   return dict((v, k) for k, v in labelmap_dict.iteritems())
  78 
  79 
  80 def main(parsed_args):
  81   all_box_annotations = pd.read_csv(parsed_args.input_annotations_boxes)
  82   all_label_annotations = pd.read_csv(parsed_args.input_annotations_labels)
  83   all_annotations = pd.concat([all_box_annotations, all_label_annotations])
  84 
  85   class_label_map = _load_labelmap(parsed_args.input_class_labelmap)
  86   relationship_label_map = _load_labelmap(
  87       parsed_args.input_relationship_labelmap)
  88 
  89   relation_evaluator = vrd_evaluation.VRDRelationDetectionEvaluator()
  90   phrase_evaluator = vrd_evaluation.VRDPhraseDetectionEvaluator()
  91 
  92   for _, groundtruth in enumerate(all_annotations.groupby('ImageID')):
  93     image_id, image_groundtruth = groundtruth
  94     groundtruth_dictionary = utils.build_groundtruth_vrd_dictionary(
  95         image_groundtruth, class_label_map, relationship_label_map)
  96 
  97     relation_evaluator.add_single_ground_truth_image_info(
  98         image_id, groundtruth_dictionary)
  99     phrase_evaluator.add_single_ground_truth_image_info(image_id,
 100                                                         groundtruth_dictionary)
 101 
 102   all_predictions = pd.read_csv(parsed_args.input_predictions)
 103   for _, prediction_data in enumerate(all_predictions.groupby('ImageID')):
 104     image_id, image_predictions = prediction_data
 105     prediction_dictionary = utils.build_predictions_vrd_dictionary(
 106         image_predictions, class_label_map, relationship_label_map)
 107 
 108     relation_evaluator.add_single_detected_image_info(image_id,
 109                                                       prediction_dictionary)
 110     phrase_evaluator.add_single_detected_image_info(image_id,
 111                                                     prediction_dictionary)
 112 
 113   relation_metrics = relation_evaluator.evaluate(
 114       relationships=_swap_labelmap_dict(relationship_label_map))
 115   phrase_metrics = phrase_evaluator.evaluate(
 116       relationships=_swap_labelmap_dict(relationship_label_map))
 117 
 118   with open(parsed_args.output_metrics, 'w') as fid:
 119     io_utils.write_csv(fid, relation_metrics)
 120     io_utils.write_csv(fid, phrase_metrics)
 121 
 122 
 123 if __name__ == '__main__':
 124 
 125   parser = argparse.ArgumentParser(
 126       description=
 127       'Evaluate Open Images Visual Relationship Detection predictions.')
 128   parser.add_argument(
 129       '--input_annotations_vrd',
 130       required=True,
 131       help='File with groundtruth vrd annotations.')
 132   parser.add_argument(
 133       '--input_annotations_labels',
 134       required=True,
 135       help='File with groundtruth labels annotations')
 136   parser.add_argument(
 137       '--input_predictions',
 138       required=True,
 139       help="""File with detection predictions; NOTE: no postprocessing is
 140       applied in the evaluation script.""")
 141   parser.add_argument(
 142       '--input_class_labelmap',
 143       required=True,
 144       help="""OpenImages Challenge labelmap; note: it is expected to include
 145       attributes.""")
 146   parser.add_argument(
 147       '--input_relationship_labelmap',
 148       required=True,
 149       help="""OpenImages Challenge relationship labelmap.""")
 150   parser.add_argument(
 151       '--output_metrics', required=True, help='Output file with csv metrics')
 152 
 153   args = parser.parse_args()
 154   main(args)
