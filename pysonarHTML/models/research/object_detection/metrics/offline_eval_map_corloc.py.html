  15 r"""Evaluation executable for detection data.
  16 
  17 This executable evaluates precomputed detections produced by a detection
  18 model and writes the evaluation results into csv file metrics.csv, stored
  19 in the directory, specified by --eval_dir.
  20 
  21 The evaluation metrics set is supplied in object_detection.protos.EvalConfig
  22 in metrics_set field.
  23 Currently two set of metrics are supported:
  24 - pascal_voc_metrics: standard PASCAL VOC 2007 metric
  25 - open_images_detection_metrics: Open Image V2 metric
  26 All other field of object_detection.protos.EvalConfig are ignored.
  27 
  28 Example usage:
  29     ./compute_metrics \
  30         --eval_dir=path/to/eval_dir \
  31         --eval_config_path=path/to/evaluation/configuration/file \
  32         --input_config_path=path/to/input/configuration/file
  33 """
  34 import csv
  35 import os
  36 import re
  37 import tensorflow.compat.v1 as tf
  38 
  39 from object_detection.core import standard_fields
  40 from object_detection.legacy import evaluator
  41 from object_detection.metrics import tf_example_parser
  42 from object_detection.utils import config_util
  43 from object_detection.utils import label_map_util
  44 
  45 flags = tf.app.flags
  46 tf.logging.set_verbosity(tf.logging.INFO)
  47 
  48 flags.DEFINE_string('eval_dir', None, 'Directory to write eval summaries to.')
  49 flags.DEFINE_string('eval_config_path', None,
  50                     'Path to an eval_pb2.EvalConfig config file.')
  51 flags.DEFINE_string('input_config_path', None,
  52                     'Path to an eval_pb2.InputConfig config file.')
  53 
  54 FLAGS = flags.FLAGS

  57 def _generate_sharded_filenames(filename):
  58   m = re.search(r'@(\d{1,})', filename)
  59   if m:
  60     num_shards = int(m.group(1))
  61     return [
  62         re.sub(r'@(\d{1,})', '-%.5d-of-%.5d' % (i, num_shards), filename)
  63         for i in range(num_shards)
  64     ]
  65   else:
  66     return [filename]
  67 
  68 
  69 def _generate_filenames(filenames):
  70   result = []
  71   for filename in filenames:
  72     result += _generate_sharded_filenames(filename)
  73   return result

  76 def read_data_and_evaluate(input_config, eval_config):
  77   """Reads pre-computed object detections and groundtruth from tf_record.
  78 
  79   Args:
  80     input_config: input config proto of type
  81       object_detection.protos.InputReader.
  82     eval_config: evaluation config proto of type
  83       object_detection.protos.EvalConfig.
  84 
  85   Returns:
  86     Evaluated detections metrics.
  87 
  88   Raises:
  89     ValueError: if input_reader type is not supported or metric type is unknown.
  90   """
  91   if input_config.WhichOneof('input_reader') == 'tf_record_input_reader':
  92     input_paths = input_config.tf_record_input_reader.input_path
  93 
  94     categories = label_map_util.create_categories_from_labelmap(
  95         input_config.label_map_path)
  96 
  97     object_detection_evaluators = evaluator.get_evaluators(
  98         eval_config, categories)
  99     # Support a single evaluator
 100     object_detection_evaluator = object_detection_evaluators[0]
 101 
 102     skipped_images = 0
 103     processed_images = 0
 104     for input_path in _generate_filenames(input_paths):
 105       tf.logging.info('Processing file: {0}'.format(input_path))
 106 
 107       record_iterator = tf.python_io.tf_record_iterator(path=input_path)
 108       data_parser = tf_example_parser.TfExampleDetectionAndGTParser()
 109 
 110       for string_record in record_iterator:
 111         tf.logging.log_every_n(tf.logging.INFO, 'Processed %d images...', 1000,
 112                                processed_images)
 113         processed_images += 1
 114 
 115         example = tf.train.Example()
 116         example.ParseFromString(string_record)
 117         decoded_dict = data_parser.parse(example)
 118 
 119         if decoded_dict:
 120           object_detection_evaluator.add_single_ground_truth_image_info(
 121               decoded_dict[standard_fields.DetectionResultFields.key],
 122               decoded_dict)
 123           object_detection_evaluator.add_single_detected_image_info(
 124               decoded_dict[standard_fields.DetectionResultFields.key],
 125               decoded_dict)
 126         else:
 127           skipped_images += 1
 128           tf.logging.info('Skipped images: {0}'.format(skipped_images))
 129 
 130     return object_detection_evaluator.evaluate()
 131 
 132   raise ValueError('Unsupported input_reader_config.')

 135 def write_metrics(metrics, output_dir):
 136   """Write metrics to the output directory.
 137 
 138   Args:
 139     metrics: A dictionary containing metric names and values.
 140     output_dir: Directory to write metrics to.
 141   """
 142   tf.logging.info('Writing metrics.')
 143 
 144   with open(os.path.join(output_dir, 'metrics.csv'), 'w') as csvfile:
 145     metrics_writer = csv.writer(csvfile, delimiter=',')
 146     for metric_name, metric_value in metrics.items():
 147       metrics_writer.writerow([metric_name, str(metric_value)])
 148

 150 def main(argv):
 151   del argv
 152   required_flags = ['input_config_path', 'eval_config_path', 'eval_dir']
 153   for flag_name in required_flags:
 154     if not getattr(FLAGS, flag_name):
 155       raise ValueError('Flag --{} is required'.format(flag_name))
 156 
 157   configs = config_util.get_configs_from_multiple_files(
 158       eval_input_config_path=FLAGS.input_config_path,
 159       eval_config_path=FLAGS.eval_config_path)
 160 
 161   eval_config = configs['eval_config']
 162   input_config = configs['eval_input_config']
 163 
 164   metrics = read_data_and_evaluate(input_config, eval_config)
 165 
 166   # Save metrics
 167   write_metrics(metrics, FLAGS.eval_dir)
 168 
 169 
 170 if __name__ == '__main__':
 171   tf.app.run(main)
