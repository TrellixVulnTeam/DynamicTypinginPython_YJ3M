  15 """Tests for calibration_metrics."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import numpy as np
  22 import tensorflow.compat.v1 as tf
  23 from object_detection.metrics import calibration_metrics
  24 
  25 
  26 class CalibrationLibTest(tf.test.TestCase):
  27 
  28   @staticmethod
  29   def _get_calibration_placeholders():
  30     """Returns TF placeholders for y_true and y_pred."""
  31     return (tf.placeholder(tf.int64, shape=(None)),
  32             tf.placeholder(tf.float32, shape=(None)))
  33 
  34   def test_expected_calibration_error_all_bins_filled(self):
  35     """Test expected calibration error when all bins contain predictions."""
  36     y_true, y_pred = self._get_calibration_placeholders()
  37     expected_ece_op, update_op = calibration_metrics.expected_calibration_error(
  38         y_true, y_pred, nbins=2)
  39     with self.test_session() as sess:
  40       metrics_vars = tf.get_collection(tf.GraphKeys.METRIC_VARIABLES)
  41       sess.run(tf.variables_initializer(var_list=metrics_vars))
  42       # Bin calibration errors (|confidence - accuracy| * bin_weight):
  43       # - [0,0.5): |0.2 - 0.333| * (3/5) = 0.08
  44       # - [0.5, 1]: |0.75 - 0.5| * (2/5) = 0.1
  45       sess.run(
  46           update_op,
  47           feed_dict={
  48               y_pred: np.array([0., 0.2, 0.4, 0.5, 1.0]),
  49               y_true: np.array([0, 0, 1, 0, 1])
  50           })
  51     actual_ece = 0.08 + 0.1
  52     expected_ece = sess.run(expected_ece_op)
  53     self.assertAlmostEqual(actual_ece, expected_ece)
  54 
  55   def test_expected_calibration_error_all_bins_not_filled(self):
  56     """Test expected calibration error when no predictions for one bin."""
  57     y_true, y_pred = self._get_calibration_placeholders()
  58     expected_ece_op, update_op = calibration_metrics.expected_calibration_error(
  59         y_true, y_pred, nbins=2)
  60     with self.test_session() as sess:
  61       metrics_vars = tf.get_collection(tf.GraphKeys.METRIC_VARIABLES)
  62       sess.run(tf.variables_initializer(var_list=metrics_vars))
  63       # Bin calibration errors (|confidence - accuracy| * bin_weight):
  64       # - [0,0.5): |0.2 - 0.333| * (3/5) = 0.08
  65       # - [0.5, 1]: |0.75 - 0.5| * (2/5) = 0.1
  66       sess.run(
  67           update_op,
  68           feed_dict={
  69               y_pred: np.array([0., 0.2, 0.4]),
  70               y_true: np.array([0, 0, 1])
  71           })
  72     actual_ece = np.abs(0.2 - (1 / 3.))
  73     expected_ece = sess.run(expected_ece_op)
  74     self.assertAlmostEqual(actual_ece, expected_ece)
  75 
  76   def test_expected_calibration_error_with_multiple_data_streams(self):
  77     """Test expected calibration error when multiple data batches provided."""
  78     y_true, y_pred = self._get_calibration_placeholders()
  79     expected_ece_op, update_op = calibration_metrics.expected_calibration_error(
  80         y_true, y_pred, nbins=2)
  81     with self.test_session() as sess:
  82       metrics_vars = tf.get_collection(tf.GraphKeys.METRIC_VARIABLES)
  83       sess.run(tf.variables_initializer(var_list=metrics_vars))
  84       # Identical data to test_expected_calibration_error_all_bins_filled,
  85       # except split over three batches.
  86       sess.run(
  87           update_op,
  88           feed_dict={
  89               y_pred: np.array([0., 0.2]),
  90               y_true: np.array([0, 0])
  91           })
  92       sess.run(
  93           update_op,
  94           feed_dict={
  95               y_pred: np.array([0.4, 0.5]),
  96               y_true: np.array([1, 0])
  97           })
  98       sess.run(
  99           update_op, feed_dict={
 100               y_pred: np.array([1.0]),
 101               y_true: np.array([1])
 102           })
 103     actual_ece = 0.08 + 0.1
 104     expected_ece = sess.run(expected_ece_op)
 105     self.assertAlmostEqual(actual_ece, expected_ece)
