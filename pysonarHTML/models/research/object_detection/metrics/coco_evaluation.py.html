  30 class CocoDetectionEvaluator(object_detection_evaluation.DetectionEvaluator):
  31   """Class to evaluate COCO detection metrics."""

  33   def __init__(self,
  34                categories,
  35                include_metrics_per_category=False,
  36                all_metrics_per_category=False):
  37     """Constructor.
  38 
  39     Args:
  40       categories: A list of dicts, each of which has the following keys -
  41         'id': (required) an integer id uniquely identifying this category.
  42         'name': (required) string representing category name e.g., 'cat', 'dog'.
  43       include_metrics_per_category: If True, include metrics for each category.
  44       all_metrics_per_category: Whether to include all the summary metrics for
  45         each category in per_category_ap. Be careful with setting it to true if
  46         you have more than handful of categories, because it will pollute
  47         your mldash.
  48     """
  49     super(CocoDetectionEvaluator, self).__init__(categories)
  50     # _image_ids is a dictionary that maps unique image ids to Booleans which
  51     # indicate whether a corresponding detection has been added.
  52     self._image_ids = {}
  53     self._groundtruth_list = []
  54     self._detection_boxes_list = []
  55     self._category_id_set = set([cat['id'] for cat in self._categories])
  56     self._annotation_id = 1
  57     self._metrics = None
  58     self._include_metrics_per_category = include_metrics_per_category
  59     self._all_metrics_per_category = all_metrics_per_category

  61   def clear(self):
  62     """Clears the state to prepare for a fresh evaluation."""
  63     self._image_ids.clear()
  64     self._groundtruth_list = []
  65     self._detection_boxes_list = []

  67   def add_single_ground_truth_image_info(self,
  68                                          image_id,
  69                                          groundtruth_dict):
  70     """Adds groundtruth for a single image to be used for evaluation.
  71 
  72     If the image has already been added, a warning is logged, and groundtruth is
  73     ignored.
  74 
  75     Args:
  76       image_id: A unique string/integer identifier for the image.
  77       groundtruth_dict: A dictionary containing -
  78         InputDataFields.groundtruth_boxes: float32 numpy array of shape
  79           [num_boxes, 4] containing `num_boxes` groundtruth boxes of the format
  80           [ymin, xmin, ymax, xmax] in absolute image coordinates.
  81         InputDataFields.groundtruth_classes: integer numpy array of shape
  82           [num_boxes] containing 1-indexed groundtruth classes for the boxes.
  83         InputDataFields.groundtruth_is_crowd (optional): integer numpy array of
  84           shape [num_boxes] containing iscrowd flag for groundtruth boxes.
  85         InputDataFields.groundtruth_area (optional): float numpy array of
  86           shape [num_boxes] containing the area (in the original absolute
  87           coordinates) of the annotated object.
  88         InputDataFields.groundtruth_keypoints (optional): float numpy array of
  89           keypoints with shape [num_boxes, num_keypoints, 2].
  90         InputDataFields.groundtruth_keypoint_visibilities (optional): integer
  91           numpy array of keypoint visibilities with shape [num_gt_boxes,
  92           num_keypoints]. Integer is treated as an enum with 0=not labeled,
  93           1=labeled but not visible and 2=labeled and visible.
  94     """
  95     if image_id in self._image_ids:
  96       tf.logging.warning('Ignoring ground truth with image id %s since it was '
  97                          'previously added', image_id)
  98       return
  99 
 100     # Drop optional fields if empty tensor.
 101     groundtruth_is_crowd = groundtruth_dict.get(
 102         standard_fields.InputDataFields.groundtruth_is_crowd)
 103     groundtruth_area = groundtruth_dict.get(
 104         standard_fields.InputDataFields.groundtruth_area)
 105     groundtruth_keypoints = groundtruth_dict.get(
 106         standard_fields.InputDataFields.groundtruth_keypoints)
 107     groundtruth_keypoint_visibilities = groundtruth_dict.get(
 108         standard_fields.InputDataFields.groundtruth_keypoint_visibilities)
 109     if groundtruth_is_crowd is not None and not groundtruth_is_crowd.shape[0]:
 110       groundtruth_is_crowd = None
 111     if groundtruth_area is not None and not groundtruth_area.shape[0]:
 112       groundtruth_area = None
 113     if groundtruth_keypoints is not None and not groundtruth_keypoints.shape[0]:
 114       groundtruth_keypoints = None
 115     if groundtruth_keypoint_visibilities is not None and not groundtruth_keypoint_visibilities.shape[
 116         0]:
 117       groundtruth_keypoint_visibilities = None
 118 
 119     self._groundtruth_list.extend(
 120         coco_tools.ExportSingleImageGroundtruthToCoco(
 121             image_id=image_id,
 122             next_annotation_id=self._annotation_id,
 123             category_id_set=self._category_id_set,
 124             groundtruth_boxes=groundtruth_dict[
 125                 standard_fields.InputDataFields.groundtruth_boxes],
 126             groundtruth_classes=groundtruth_dict[
 127                 standard_fields.InputDataFields.groundtruth_classes],
 128             groundtruth_is_crowd=groundtruth_is_crowd,
 129             groundtruth_area=groundtruth_area,
 130             groundtruth_keypoints=groundtruth_keypoints,
 131             groundtruth_keypoint_visibilities=groundtruth_keypoint_visibilities)
 132     )
 133 
 134     self._annotation_id += groundtruth_dict[standard_fields.InputDataFields.
 135                                             groundtruth_boxes].shape[0]
 136     # Boolean to indicate whether a detection has been added for this image.
 137     self._image_ids[image_id] = False
 138 
 139   def add_single_detected_image_info(self,
 140                                      image_id,
 141                                      detections_dict):
 142     """Adds detections for a single image to be used for evaluation.
 143 
 144     If a detection has already been added for this image id, a warning is
 145     logged, and the detection is skipped.
 146 
 147     Args:
 148       image_id: A unique string/integer identifier for the image.
 149       detections_dict: A dictionary containing -
 150         DetectionResultFields.detection_boxes: float32 numpy array of shape
 151           [num_boxes, 4] containing `num_boxes` detection boxes of the format
 152           [ymin, xmin, ymax, xmax] in absolute image coordinates.
 153         DetectionResultFields.detection_scores: float32 numpy array of shape
 154           [num_boxes] containing detection scores for the boxes.
 155         DetectionResultFields.detection_classes: integer numpy array of shape
 156           [num_boxes] containing 1-indexed detection classes for the boxes.
 157         DetectionResultFields.detection_keypoints (optional): float numpy array
 158           of keypoints with shape [num_boxes, num_keypoints, 2].
 159     Raises:
 160       ValueError: If groundtruth for the image_id is not available.
 161     """
 162     if image_id not in self._image_ids:
 163       raise ValueError('Missing groundtruth for image id: {}'.format(image_id))
 164 
 165     if self._image_ids[image_id]:
 166       tf.logging.warning('Ignoring detection with image id %s since it was '
 167                          'previously added', image_id)
 168       return
 169 
 170     # Drop optional fields if empty tensor.
 171     detection_keypoints = detections_dict.get(
 172         standard_fields.DetectionResultFields.detection_keypoints)
 173     if detection_keypoints is not None and not detection_keypoints.shape[0]:
 174       detection_keypoints = None
 175     self._detection_boxes_list.extend(
 176         coco_tools.ExportSingleImageDetectionBoxesToCoco(
 177             image_id=image_id,
 178             category_id_set=self._category_id_set,
 179             detection_boxes=detections_dict[
 180                 standard_fields.DetectionResultFields.detection_boxes],
 181             detection_scores=detections_dict[
 182                 standard_fields.DetectionResultFields.detection_scores],
 183             detection_classes=detections_dict[
 184                 standard_fields.DetectionResultFields.detection_classes],
 185             detection_keypoints=detection_keypoints))
 186     self._image_ids[image_id] = True

 188   def dump_detections_to_json_file(self, json_output_path):
 189     """Saves the detections into json_output_path in the format used by MS COCO.
 190 
 191     Args:
 192       json_output_path: String containing the output file's path. It can be also
 193         None. In that case nothing will be written to the output file.
 194     """

 196       with tf.gfile.GFile(json_output_path, 'w') as fid:

 198         json_utils.Dump(
 199             obj=self._detection_boxes_list, fid=fid, float_digits=4, indent=2)

 201   def evaluate(self):
 202     """Evaluates the detection boxes and returns a dictionary of coco metrics.
 203 
 204     Returns:
 205       A dictionary holding -
 206 
 207       1. summary_metrics:
 208       'DetectionBoxes_Precision/mAP': mean average precision over classes
 209         averaged over IOU thresholds ranging from .5 to .95 with .05
 210         increments.
 211       'DetectionBoxes_Precision/mAP@.50IOU': mean average precision at 50% IOU
 212       'DetectionBoxes_Precision/mAP@.75IOU': mean average precision at 75% IOU
 213       'DetectionBoxes_Precision/mAP (small)': mean average precision for small
 214         objects (area < 32^2 pixels).
 215       'DetectionBoxes_Precision/mAP (medium)': mean average precision for
 216         medium sized objects (32^2 pixels < area < 96^2 pixels).
 217       'DetectionBoxes_Precision/mAP (large)': mean average precision for large
 218         objects (96^2 pixels < area < 10000^2 pixels).
 219       'DetectionBoxes_Recall/AR@1': average recall with 1 detection.
 220       'DetectionBoxes_Recall/AR@10': average recall with 10 detections.
 221       'DetectionBoxes_Recall/AR@100': average recall with 100 detections.
 222       'DetectionBoxes_Recall/AR@100 (small)': average recall for small objects
 223         with 100.
 224       'DetectionBoxes_Recall/AR@100 (medium)': average recall for medium objects
 225         with 100.
 226       'DetectionBoxes_Recall/AR@100 (large)': average recall for large objects
 227         with 100 detections.
 228 
 229       2. per_category_ap: if include_metrics_per_category is True, category
 230       specific results with keys of the form:
 231       'Precision mAP ByCategory/category' (without the supercategory part if
 232       no supercategories exist). For backward compatibility
 233       'PerformanceByCategory' is included in the output regardless of
 234       all_metrics_per_category.
 235     """
 236     tf.logging.info('Performing evaluation on %d images.', len(self._image_ids))
 237     groundtruth_dict = {
 238         'annotations': self._groundtruth_list,
 239         'images': [{'id': image_id} for image_id in self._image_ids],
 240         'categories': self._categories
 241     }
 242     coco_wrapped_groundtruth = coco_tools.COCOWrapper(groundtruth_dict)
 243     coco_wrapped_detections = coco_wrapped_groundtruth.LoadAnnotations(
 244         self._detection_boxes_list)
 245     box_evaluator = coco_tools.COCOEvalWrapper(
 246         coco_wrapped_groundtruth, coco_wrapped_detections, agnostic_mode=False)
 247     box_metrics, box_per_category_ap = box_evaluator.ComputeMetrics(
 248         include_metrics_per_category=self._include_metrics_per_category,
 249         all_metrics_per_category=self._all_metrics_per_category)
 250     box_metrics.update(box_per_category_ap)
 251     box_metrics = {'DetectionBoxes_'+ key: value
 252                    for key, value in iter(box_metrics.items())}
 253     return box_metrics

 255   def add_eval_dict(self, eval_dict):
 256     """Observes an evaluation result dict for a single example.
 257 
 258     When executing eagerly, once all observations have been observed by this
 259     method you can use `.evaluate()` to get the final metrics.
 260 
 261     When using `tf.estimator.Estimator` for evaluation this function is used by
 262     `get_estimator_eval_metric_ops()` to construct the metric update op.
 263 
 264     Args:
 265       eval_dict: A dictionary that holds tensors for evaluating an object
 266         detection model, returned from
 267         eval_util.result_dict_for_single_example().
 268 
 269     Returns:
 270       None when executing eagerly, or an update_op that can be used to update
 271       the eval metrics in `tf.estimator.EstimatorSpec`.
 272     """
 273     def update_op(
 274         image_id_batched,
 275         groundtruth_boxes_batched,
 276         groundtruth_classes_batched,
 277         groundtruth_is_crowd_batched,
 278         num_gt_boxes_per_image,
 279         detection_boxes_batched,
 280         detection_scores_batched,
 281         detection_classes_batched,
 282         num_det_boxes_per_image,
 283         is_annotated_batched):
 284       """Update operation for adding batch of images to Coco evaluator."""
 285 
 286       for (image_id, gt_box, gt_class, gt_is_crowd, num_gt_box, det_box,
 287            det_score, det_class, num_det_box, is_annotated) in zip(
 288                image_id_batched, groundtruth_boxes_batched,
 289                groundtruth_classes_batched, groundtruth_is_crowd_batched,
 290                num_gt_boxes_per_image,
 291                detection_boxes_batched, detection_scores_batched,
 292                detection_classes_batched, num_det_boxes_per_image,
 293                is_annotated_batched):
 294         if is_annotated:
 295           self.add_single_ground_truth_image_info(
 296               image_id, {
 297                   'groundtruth_boxes': gt_box[:num_gt_box],
 298                   'groundtruth_classes': gt_class[:num_gt_box],
 299                   'groundtruth_is_crowd': gt_is_crowd[:num_gt_box]
 300               })
 301           self.add_single_detected_image_info(
 302               image_id,
 303               {'detection_boxes': det_box[:num_det_box],
 304                'detection_scores': det_score[:num_det_box],
 305                'detection_classes': det_class[:num_det_box]})
 306 
 307     # Unpack items from the evaluation dictionary.
 308     input_data_fields = standard_fields.InputDataFields
 309     detection_fields = standard_fields.DetectionResultFields
 310     image_id = eval_dict[input_data_fields.key]
 311     groundtruth_boxes = eval_dict[input_data_fields.groundtruth_boxes]
 312     groundtruth_classes = eval_dict[input_data_fields.groundtruth_classes]
 313     groundtruth_is_crowd = eval_dict.get(
 314         input_data_fields.groundtruth_is_crowd, None)
 315     detection_boxes = eval_dict[detection_fields.detection_boxes]
 316     detection_scores = eval_dict[detection_fields.detection_scores]
 317     detection_classes = eval_dict[detection_fields.detection_classes]
 318     num_gt_boxes_per_image = eval_dict.get(
 319         'num_groundtruth_boxes_per_image', None)
 320     num_det_boxes_per_image = eval_dict.get('num_det_boxes_per_image', None)
 321     is_annotated = eval_dict.get('is_annotated', None)
 322 
 323     if groundtruth_is_crowd is None:
 324       groundtruth_is_crowd = tf.zeros_like(groundtruth_classes, dtype=tf.bool)
 325     if not image_id.shape.as_list():
 326       # Apply a batch dimension to all tensors.
 327       image_id = tf.expand_dims(image_id, 0)
 328       groundtruth_boxes = tf.expand_dims(groundtruth_boxes, 0)
 329       groundtruth_classes = tf.expand_dims(groundtruth_classes, 0)
 330       groundtruth_is_crowd = tf.expand_dims(groundtruth_is_crowd, 0)
 331       detection_boxes = tf.expand_dims(detection_boxes, 0)
 332       detection_scores = tf.expand_dims(detection_scores, 0)
 333       detection_classes = tf.expand_dims(detection_classes, 0)
 334 
 335       if num_gt_boxes_per_image is None:
 336         num_gt_boxes_per_image = tf.shape(groundtruth_boxes)[1:2]
 337       else:
 338         num_gt_boxes_per_image = tf.expand_dims(num_gt_boxes_per_image, 0)
 339 
 340       if num_det_boxes_per_image is None:
 341         num_det_boxes_per_image = tf.shape(detection_boxes)[1:2]
 342       else:
 343         num_det_boxes_per_image = tf.expand_dims(num_det_boxes_per_image, 0)
 344 
 345       if is_annotated is None:
 346         is_annotated = tf.constant([True])
 347       else:
 348         is_annotated = tf.expand_dims(is_annotated, 0)
 349     else:
 350       if num_gt_boxes_per_image is None:
 351         num_gt_boxes_per_image = tf.tile(
 352             tf.shape(groundtruth_boxes)[1:2],
 353             multiples=tf.shape(groundtruth_boxes)[0:1])
 354       if num_det_boxes_per_image is None:
 355         num_det_boxes_per_image = tf.tile(
 356             tf.shape(detection_boxes)[1:2],
 357             multiples=tf.shape(detection_boxes)[0:1])
 358       if is_annotated is None:
 359         is_annotated = tf.ones_like(image_id, dtype=tf.bool)
 360 
 361     return tf.py_func(update_op, [image_id,
 362                                   groundtruth_boxes,
 363                                   groundtruth_classes,
 364                                   groundtruth_is_crowd,
 365                                   num_gt_boxes_per_image,
 366                                   detection_boxes,
 367                                   detection_scores,
 368                                   detection_classes,
 369                                   num_det_boxes_per_image,
 370                                   is_annotated], [])

 372   def get_estimator_eval_metric_ops(self, eval_dict):
 373     """Returns a dictionary of eval metric ops.
 374 
 375     Note that once value_op is called, the detections and groundtruth added via
 376     update_op are cleared.
 377 
 378     This function can take in groundtruth and detections for a batch of images,
 379     or for a single image. For the latter case, the batch dimension for input
 380     tensors need not be present.
 381 
 382     Args:
 383       eval_dict: A dictionary that holds tensors for evaluating object detection
 384         performance. For single-image evaluation, this dictionary may be
 385         produced from eval_util.result_dict_for_single_example(). If multi-image
 386         evaluation, `eval_dict` should contain the fields
 387         'num_groundtruth_boxes_per_image' and 'num_det_boxes_per_image' to
 388         properly unpad the tensors from the batch.
 389 
 390     Returns:
 391       a dictionary of metric names to tuple of value_op and update_op that can
 392       be used as eval metric ops in tf.estimator.EstimatorSpec. Note that all
 393       update ops must be run together and similarly all value ops must be run
 394       together to guarantee correct behaviour.
 395     """
 396     update_op = self.add_eval_dict(eval_dict)
 397     metric_names = ['DetectionBoxes_Precision/mAP',
 398                     'DetectionBoxes_Precision/mAP@.50IOU',
 399                     'DetectionBoxes_Precision/mAP@.75IOU',
 400                     'DetectionBoxes_Precision/mAP (large)',
 401                     'DetectionBoxes_Precision/mAP (medium)',
 402                     'DetectionBoxes_Precision/mAP (small)',
 403                     'DetectionBoxes_Recall/AR@1',
 404                     'DetectionBoxes_Recall/AR@10',
 405                     'DetectionBoxes_Recall/AR@100',
 406                     'DetectionBoxes_Recall/AR@100 (large)',
 407                     'DetectionBoxes_Recall/AR@100 (medium)',
 408                     'DetectionBoxes_Recall/AR@100 (small)']
 409     if self._include_metrics_per_category:
 410       for category_dict in self._categories:
 411         metric_names.append('DetectionBoxes_PerformanceByCategory/mAP/' +
 412                             category_dict['name'])
 413 
 414     def first_value_func():
 415       self._metrics = self.evaluate()
 416       self.clear()
 417       return np.float32(self._metrics[metric_names[0]])
 418 
 419     def value_func_factory(metric_name):
 420       def value_func():
 421         return np.float32(self._metrics[metric_name])
 422       return value_func
 423 
 424     # Ensure that the metrics are only evaluated once.
 425     first_value_op = tf.py_func(first_value_func, [], tf.float32)
 426     eval_metric_ops = {metric_names[0]: (first_value_op, update_op)}
 427     with tf.control_dependencies([first_value_op]):
 428       for metric_name in metric_names[1:]:
 429         eval_metric_ops[metric_name] = (tf.py_func(
 430             value_func_factory(metric_name), [], np.float32), update_op)
 431     return eval_metric_ops

 434 def _check_mask_type_and_value(array_name, masks):
 435   """Checks whether mask dtype is uint8 and the values are either 0 or 1."""
 436   if masks.dtype != np.uint8:
 437     raise ValueError('{} must be of type np.uint8. Found {}.'.format(
 438         array_name, masks.dtype))
 439   if np.any(np.logical_and(masks != 0, masks != 1)):
 440     raise ValueError('{} elements can only be either 0 or 1.'.format(
 441         array_name))

 444 class CocoKeypointEvaluator(CocoDetectionEvaluator):
 445   """Class to evaluate COCO keypoint metrics."""
 446 
 447   def __init__(self,
 448                category_id,
 449                category_keypoints,
 450                class_text,
 451                oks_sigmas=None):
 452     """Constructor.
 453 
 454     Args:
 455       category_id: An integer id uniquely identifying this category.
 456       category_keypoints: A list specifying keypoint mappings, with items:
 457           'id': (required) an integer id identifying the keypoint.
 458           'name': (required) a string representing the keypoint name.
 459       class_text: A string representing the category name for which keypoint
 460         metrics are to be computed.
 461       oks_sigmas: A dict of keypoint name to standard deviation values for OKS
 462         metrics. If not provided, default value of 0.05 will be used.
 463     """
 464     self._category_id = category_id
 465     self._category_name = class_text
 466     self._keypoint_ids = sorted(
 467         [keypoint['id'] for keypoint in category_keypoints])
 468     kpt_id_to_name = {kpt['id']: kpt['name'] for kpt in category_keypoints}
 469     if oks_sigmas:
 470       self._oks_sigmas = np.array([
 471           oks_sigmas[kpt_id_to_name[idx]] for idx in self._keypoint_ids
 472       ])
 473     else:
 474       # Default all per-keypoint sigmas to 0.
 475       self._oks_sigmas = np.full((len(self._keypoint_ids)), 0.05)
 476       tf.logging.warning('No default keypoint OKS sigmas provided. Will use '
 477                          '0.05')
 478     tf.logging.info('Using the following keypoint OKS sigmas: {}'.format(
 479         self._oks_sigmas))
 480     self._metrics = None
 481     super(CocoKeypointEvaluator, self).__init__([{
 482         'id': self._category_id,
 483         'name': class_text
 484     }])

 486   def add_single_ground_truth_image_info(self, image_id, groundtruth_dict):
 487     """Adds groundtruth for a single image with keypoints.
 488 
 489     If the image has already been added, a warning is logged, and groundtruth
 490     is ignored.
 491 
 492     Args:
 493       image_id: A unique string/integer identifier for the image.
 494       groundtruth_dict: A dictionary containing -
 495         InputDataFields.groundtruth_boxes: float32 numpy array of shape
 496           [num_boxes, 4] containing `num_boxes` groundtruth boxes of the format
 497           [ymin, xmin, ymax, xmax] in absolute image coordinates.
 498         InputDataFields.groundtruth_classes: integer numpy array of shape
 499           [num_boxes] containing 1-indexed groundtruth classes for the boxes.
 500         InputDataFields.groundtruth_is_crowd (optional): integer numpy array of
 501           shape [num_boxes] containing iscrowd flag for groundtruth boxes.
 502         InputDataFields.groundtruth_area (optional): float numpy array of
 503           shape [num_boxes] containing the area (in the original absolute
 504           coordinates) of the annotated object.
 505         InputDataFields.groundtruth_keypoints: float numpy array of
 506           keypoints with shape [num_boxes, num_keypoints, 2].
 507         InputDataFields.groundtruth_keypoint_visibilities (optional): integer
 508           numpy array of keypoint visibilities with shape [num_gt_boxes,
 509           num_keypoints]. Integer is treated as an enum with 0=not labels,
 510           1=labeled but not visible and 2=labeled and visible.
 511     """
 512 
 513     # Keep only the groundtruth for our category and its keypoints.
 514     groundtruth_classes = groundtruth_dict[
 515         standard_fields.InputDataFields.groundtruth_classes]
 516     groundtruth_boxes = groundtruth_dict[
 517         standard_fields.InputDataFields.groundtruth_boxes]
 518     groundtruth_keypoints = groundtruth_dict[
 519         standard_fields.InputDataFields.groundtruth_keypoints]
 520     class_indices = [
 521         idx for idx, gt_class_id in enumerate(groundtruth_classes)
 522         if gt_class_id == self._category_id
 523     ]
 524     filtered_groundtruth_classes = np.take(
 525         groundtruth_classes, class_indices, axis=0)
 526     filtered_groundtruth_boxes = np.take(
 527         groundtruth_boxes, class_indices, axis=0)
 528     filtered_groundtruth_keypoints = np.take(
 529         groundtruth_keypoints, class_indices, axis=0)
 530     filtered_groundtruth_keypoints = np.take(
 531         filtered_groundtruth_keypoints, self._keypoint_ids, axis=1)
 532 
 533     filtered_groundtruth_dict = {}
 534     filtered_groundtruth_dict[
 535         standard_fields.InputDataFields
 536         .groundtruth_classes] = filtered_groundtruth_classes
 537     filtered_groundtruth_dict[standard_fields.InputDataFields
 538                               .groundtruth_boxes] = filtered_groundtruth_boxes
 539     filtered_groundtruth_dict[
 540         standard_fields.InputDataFields
 541         .groundtruth_keypoints] = filtered_groundtruth_keypoints
 542 
 543     if (standard_fields.InputDataFields.groundtruth_is_crowd in
 544         groundtruth_dict.keys()):
 545       groundtruth_is_crowd = groundtruth_dict[
 546           standard_fields.InputDataFields.groundtruth_is_crowd]
 547       filtered_groundtruth_is_crowd = np.take(groundtruth_is_crowd,
 548                                               class_indices, 0)
 549       filtered_groundtruth_dict[
 550           standard_fields.InputDataFields
 551           .groundtruth_is_crowd] = filtered_groundtruth_is_crowd
 552     if (standard_fields.InputDataFields.groundtruth_area in
 553         groundtruth_dict.keys()):
 554       groundtruth_area = groundtruth_dict[
 555           standard_fields.InputDataFields.groundtruth_area]
 556       filtered_groundtruth_area = np.take(groundtruth_area, class_indices, 0)
 557       filtered_groundtruth_dict[
 558           standard_fields.InputDataFields
 559           .groundtruth_area] = filtered_groundtruth_area
 560     if (standard_fields.InputDataFields.groundtruth_keypoint_visibilities in
 561         groundtruth_dict.keys()):
 562       groundtruth_keypoint_visibilities = groundtruth_dict[
 563           standard_fields.InputDataFields.groundtruth_keypoint_visibilities]
 564       filtered_groundtruth_keypoint_visibilities = np.take(
 565           groundtruth_keypoint_visibilities, class_indices, axis=0)
 566       filtered_groundtruth_keypoint_visibilities = np.take(
 567           filtered_groundtruth_keypoint_visibilities,
 568           self._keypoint_ids,
 569           axis=1)
 570       filtered_groundtruth_dict[
 571           standard_fields.InputDataFields.
 572           groundtruth_keypoint_visibilities] = filtered_groundtruth_keypoint_visibilities
 573 
 574     super(CocoKeypointEvaluator,
 575           self).add_single_ground_truth_image_info(image_id,
 576                                                    filtered_groundtruth_dict)

 578   def add_single_detected_image_info(self, image_id, detections_dict):
 579     """Adds detections for a single image and the specific category for which keypoints are evaluated.
 580 
 581     If a detection has already been added for this image id, a warning is
 582     logged, and the detection is skipped.
 583 
 584     Args:
 585       image_id: A unique string/integer identifier for the image.
 586       detections_dict: A dictionary containing -
 587         DetectionResultFields.detection_boxes: float32 numpy array of shape

 593           [num_boxes] containing 1-indexed detection classes for the boxes.
 594         DetectionResultFields.detection_keypoints: float numpy array of
 595           keypoints with shape [num_boxes, num_keypoints, 2].
 596 
 597     Raises:
 598       ValueError: If groundtruth for the image_id is not available.
 599     """
 600 
 601     # Keep only the detections for our category and its keypoints.
 602     detection_classes = detections_dict[
 603         standard_fields.DetectionResultFields.detection_classes]
 604     detection_boxes = detections_dict[
 605         standard_fields.DetectionResultFields.detection_boxes]
 606     detection_scores = detections_dict[
 607         standard_fields.DetectionResultFields.detection_scores]
 608     detection_keypoints = detections_dict[
 609         standard_fields.DetectionResultFields.detection_keypoints]

 611         idx for idx, class_id in enumerate(detection_classes)
 612         if class_id == self._category_id
 613     ]

 615         detection_classes, class_indices, axis=0)
 616     filtered_detection_boxes = np.take(detection_boxes, class_indices, axis=0)
 617     filtered_detection_scores = np.take(detection_scores, class_indices, axis=0)
 618     filtered_detection_keypoints = np.take(
 619         detection_keypoints, class_indices, axis=0)
 620     filtered_detection_keypoints = np.take(
 621         filtered_detection_keypoints, self._keypoint_ids, axis=1)
 622 
 623     filtered_detections_dict = {}
 624     filtered_detections_dict[standard_fields.DetectionResultFields
 625                              .detection_classes] = filtered_detection_classes
 626     filtered_detections_dict[standard_fields.DetectionResultFields
 627                              .detection_boxes] = filtered_detection_boxes
 628     filtered_detections_dict[standard_fields.DetectionResultFields
 629                              .detection_scores] = filtered_detection_scores
 630     filtered_detections_dict[standard_fields.DetectionResultFields.
 631                              detection_keypoints] = filtered_detection_keypoints

 634           self).add_single_detected_image_info(image_id,

 637   def evaluate(self):
 638     """Evaluates the keypoints and returns a dictionary of coco metrics.
 639 
 640     Returns:
 641       A dictionary holding -
 642 
 643       1. summary_metrics:
 644       'Keypoints_Precision/mAP': mean average precision over classes
 645         averaged over OKS thresholds ranging from .5 to .95 with .05
 646         increments.
 647       'Keypoints_Precision/mAP@.50IOU': mean average precision at 50% OKS
 648       'Keypoints_Precision/mAP@.75IOU': mean average precision at 75% OKS
 649       'Keypoints_Precision/mAP (medium)': mean average precision for medium
 650         sized objects (32^2 pixels < area < 96^2 pixels).
 651       'Keypoints_Precision/mAP (large)': mean average precision for large
 652         objects (96^2 pixels < area < 10000^2 pixels).
 653       'Keypoints_Recall/AR@1': average recall with 1 detection.
 654       'Keypoints_Recall/AR@10': average recall with 10 detections.
 655       'Keypoints_Recall/AR@100': average recall with 100 detections.
 656       'Keypoints_Recall/AR@100 (medium)': average recall for medium objects with
 657         100.
 658       'Keypoints_Recall/AR@100 (large)': average recall for large objects with
 659         100 detections.
 660     """
 661     tf.logging.info('Performing evaluation on %d images.', len(self._image_ids))
 662     groundtruth_dict = {
 663         'annotations': self._groundtruth_list,
 664         'images': [{'id': image_id} for image_id in self._image_ids],
 665         'categories': self._categories
 666     }
 667     coco_wrapped_groundtruth = coco_tools.COCOWrapper(
 668         groundtruth_dict, detection_type='bbox')
 669     coco_wrapped_detections = coco_wrapped_groundtruth.LoadAnnotations(
 670         self._detection_boxes_list)
 671     keypoint_evaluator = coco_tools.COCOEvalWrapper(
 672         coco_wrapped_groundtruth,
 673         coco_wrapped_detections,
 674         agnostic_mode=False,
 675         iou_type='keypoints',
 676         oks_sigmas=self._oks_sigmas)
 677     keypoint_metrics, _ = keypoint_evaluator.ComputeMetrics(
 678         include_metrics_per_category=False, all_metrics_per_category=False)
 679     keypoint_metrics = {
 680         'Keypoints_' + key: value
 681         for key, value in iter(keypoint_metrics.items())
 682     }
 683     return keypoint_metrics
 684 
 685   def add_eval_dict(self, eval_dict):
 686     """Observes an evaluation result dict for a single example.
 687 
 688     When executing eagerly, once all observations have been observed by this
 689     method you can use `.evaluate()` to get the final metrics.
 690 
 691     When using `tf.estimator.Estimator` for evaluation this function is used by
 692     `get_estimator_eval_metric_ops()` to construct the metric update op.
 693 
 694     Args:
 695       eval_dict: A dictionary that holds tensors for evaluating an object
 696         detection model, returned from
 697         eval_util.result_dict_for_single_example().
 698 
 699     Returns:
 700       None when executing eagerly, or an update_op that can be used to update
 701       the eval metrics in `tf.estimator.EstimatorSpec`.
 702     """
 703     def update_op(
 704         image_id_batched,
 705         groundtruth_boxes_batched,
 706         groundtruth_classes_batched,
 707         groundtruth_is_crowd_batched,
 708         groundtruth_area_batched,
 709         groundtruth_keypoints_batched,
 710         groundtruth_keypoint_visibilities_batched,
 711         num_gt_boxes_per_image,
 712         detection_boxes_batched,
 713         detection_scores_batched,
 714         detection_classes_batched,
 715         detection_keypoints_batched,
 716         num_det_boxes_per_image,
 717         is_annotated_batched):
 718       """Update operation for adding batch of images to Coco evaluator."""
 719 
 720       for (image_id, gt_box, gt_class, gt_is_crowd, gt_area, gt_keyp,
 721            gt_keyp_vis, num_gt_box, det_box, det_score, det_class, det_keyp,
 722            num_det_box, is_annotated) in zip(
 723                image_id_batched, groundtruth_boxes_batched,
 724                groundtruth_classes_batched, groundtruth_is_crowd_batched,
 725                groundtruth_area_batched, groundtruth_keypoints_batched,
 726                groundtruth_keypoint_visibilities_batched,
 727                num_gt_boxes_per_image, detection_boxes_batched,
 728                detection_scores_batched, detection_classes_batched,
 729                detection_keypoints_batched, num_det_boxes_per_image,
 730                is_annotated_batched):
 731         if is_annotated:
 732           self.add_single_ground_truth_image_info(
 733               image_id, {
 734                   'groundtruth_boxes': gt_box[:num_gt_box],
 735                   'groundtruth_classes': gt_class[:num_gt_box],
 736                   'groundtruth_is_crowd': gt_is_crowd[:num_gt_box],
 737                   'groundtruth_area': gt_area[:num_gt_box],
 738                   'groundtruth_keypoints': gt_keyp[:num_gt_box],
 739                   'groundtruth_keypoint_visibilities': gt_keyp_vis[:num_gt_box]
 740               })
 741           self.add_single_detected_image_info(
 742               image_id, {
 743                   'detection_boxes': det_box[:num_det_box],
 744                   'detection_scores': det_score[:num_det_box],
 745                   'detection_classes': det_class[:num_det_box],
 746                   'detection_keypoints': det_keyp[:num_det_box],
 747               })
 748 
 749     # Unpack items from the evaluation dictionary.
 750     input_data_fields = standard_fields.InputDataFields
 751     detection_fields = standard_fields.DetectionResultFields
 752     image_id = eval_dict[input_data_fields.key]
 753     groundtruth_boxes = eval_dict[input_data_fields.groundtruth_boxes]
 754     groundtruth_classes = eval_dict[input_data_fields.groundtruth_classes]
 755     groundtruth_is_crowd = eval_dict.get(input_data_fields.groundtruth_is_crowd,
 756                                          None)
 757     groundtruth_area = eval_dict.get(input_data_fields.groundtruth_area, None)
 758     groundtruth_keypoints = eval_dict[input_data_fields.groundtruth_keypoints]
 759     groundtruth_keypoint_visibilities = eval_dict.get(
 760         input_data_fields.groundtruth_keypoint_visibilities, None)
 761     detection_boxes = eval_dict[detection_fields.detection_boxes]
 762     detection_scores = eval_dict[detection_fields.detection_scores]
 763     detection_classes = eval_dict[detection_fields.detection_classes]
 764     detection_keypoints = eval_dict[detection_fields.detection_keypoints]
 765     num_gt_boxes_per_image = eval_dict.get(
 766         'num_groundtruth_boxes_per_image', None)
 767     num_det_boxes_per_image = eval_dict.get('num_det_boxes_per_image', None)
 768     is_annotated = eval_dict.get('is_annotated', None)
 769 
 770     if groundtruth_is_crowd is None:
 771       groundtruth_is_crowd = tf.zeros_like(groundtruth_classes, dtype=tf.bool)
 772 
 773     if groundtruth_area is None:
 774       groundtruth_area = tf.zeros_like(groundtruth_classes, dtype=tf.float32)
 775 
 776     if not image_id.shape.as_list():
 777       # Apply a batch dimension to all tensors.
 778       image_id = tf.expand_dims(image_id, 0)
 779       groundtruth_boxes = tf.expand_dims(groundtruth_boxes, 0)
 780       groundtruth_classes = tf.expand_dims(groundtruth_classes, 0)
 781       groundtruth_is_crowd = tf.expand_dims(groundtruth_is_crowd, 0)
 782       groundtruth_area = tf.expand_dims(groundtruth_area, 0)
 783       groundtruth_keypoints = tf.expand_dims(groundtruth_keypoints, 0)
 784       detection_boxes = tf.expand_dims(detection_boxes, 0)
 785       detection_scores = tf.expand_dims(detection_scores, 0)
 786       detection_classes = tf.expand_dims(detection_classes, 0)
 787       detection_keypoints = tf.expand_dims(detection_keypoints, 0)
 788 
 789       if num_gt_boxes_per_image is None:
 790         num_gt_boxes_per_image = tf.shape(groundtruth_boxes)[1:2]
 791       else:
 792         num_gt_boxes_per_image = tf.expand_dims(num_gt_boxes_per_image, 0)
 793 
 794       if num_det_boxes_per_image is None:
 795         num_det_boxes_per_image = tf.shape(detection_boxes)[1:2]
 796       else:
 797         num_det_boxes_per_image = tf.expand_dims(num_det_boxes_per_image, 0)
 798 
 799       if is_annotated is None:
 800         is_annotated = tf.constant([True])
 801       else:
 802         is_annotated = tf.expand_dims(is_annotated, 0)
 803 
 804       if groundtruth_keypoint_visibilities is None:
 805         groundtruth_keypoint_visibilities = tf.fill([
 806             tf.shape(groundtruth_boxes)[1],
 807             tf.shape(groundtruth_keypoints)[2]
 808         ], tf.constant(2, dtype=tf.int32))
 809       groundtruth_keypoint_visibilities = tf.expand_dims(
 810           groundtruth_keypoint_visibilities, 0)
 811     else:
 812       if num_gt_boxes_per_image is None:
 813         num_gt_boxes_per_image = tf.tile(
 814             tf.shape(groundtruth_boxes)[1:2],
 815             multiples=tf.shape(groundtruth_boxes)[0:1])
 816       if num_det_boxes_per_image is None:
 817         num_det_boxes_per_image = tf.tile(
 818             tf.shape(detection_boxes)[1:2],
 819             multiples=tf.shape(detection_boxes)[0:1])
 820       if is_annotated is None:
 821         is_annotated = tf.ones_like(image_id, dtype=tf.bool)
 822       if groundtruth_keypoint_visibilities is None:
 823         groundtruth_keypoint_visibilities = tf.fill([
 824             tf.shape(groundtruth_keypoints)[1],
 825             tf.shape(groundtruth_keypoints)[2]
 826         ], tf.constant(2, dtype=tf.int32))
 827         groundtruth_keypoint_visibilities = tf.tile(
 828             tf.expand_dims(groundtruth_keypoint_visibilities, 0),
 829             multiples=[tf.shape(groundtruth_keypoints)[0], 1, 1])
 830 
 831     return tf.py_func(update_op, [
 832         image_id, groundtruth_boxes, groundtruth_classes, groundtruth_is_crowd,
 833         groundtruth_area, groundtruth_keypoints,
 834         groundtruth_keypoint_visibilities, num_gt_boxes_per_image,
 835         detection_boxes, detection_scores, detection_classes,
 836         detection_keypoints, num_det_boxes_per_image, is_annotated
 837     ], [])
 838 
 839   def get_estimator_eval_metric_ops(self, eval_dict):
 840     """Returns a dictionary of eval metric ops.
 841 
 842     Note that once value_op is called, the detections and groundtruth added via
 843     update_op are cleared.
 844 
 845     This function can take in groundtruth and detections for a batch of images,
 846     or for a single image. For the latter case, the batch dimension for input
 847     tensors need not be present.
 848 
 849     Args:
 850       eval_dict: A dictionary that holds tensors for evaluating object detection
 851         performance. For single-image evaluation, this dictionary may be
 852         produced from eval_util.result_dict_for_single_example(). If multi-image
 853         evaluation, `eval_dict` should contain the fields
 854         'num_groundtruth_boxes_per_image' and 'num_det_boxes_per_image' to
 855         properly unpad the tensors from the batch.
 856 
 857     Returns:
 858       a dictionary of metric names to tuple of value_op and update_op that can
 859       be used as eval metric ops in tf.estimator.EstimatorSpec. Note that all
 860       update ops must be run together and similarly all value ops must be run
 861       together to guarantee correct behaviour.
 862     """
 863     update_op = self.add_eval_dict(eval_dict)
 864     category = self._category_name
 865     metric_names = [
 866         'Keypoints_Precision/mAP ByCategory/{}'.format(category),
 867         'Keypoints_Precision/mAP@.50IOU ByCategory/{}'.format(category),
 868         'Keypoints_Precision/mAP@.75IOU ByCategory/{}'.format(category),
 869         'Keypoints_Precision/mAP (large) ByCategory/{}'.format(category),
 870         'Keypoints_Precision/mAP (medium) ByCategory/{}'.format(category),
 871         'Keypoints_Recall/AR@1 ByCategory/{}'.format(category),
 872         'Keypoints_Recall/AR@10 ByCategory/{}'.format(category),
 873         'Keypoints_Recall/AR@100 ByCategory/{}'.format(category),
 874         'Keypoints_Recall/AR@100 (large) ByCategory/{}'.format(category),
 875         'Keypoints_Recall/AR@100 (medium) ByCategory/{}'.format(category)
 876     ]
 877 
 878     def first_value_func():
 879       self._metrics = self.evaluate()
 880       self.clear()
 881       return np.float32(self._metrics[metric_names[0]])
 882 
 883     def value_func_factory(metric_name):
 884       def value_func():
 885         return np.float32(self._metrics[metric_name])
 886       return value_func
 887 
 888     # Ensure that the metrics are only evaluated once.
 889     first_value_op = tf.py_func(first_value_func, [], tf.float32)
 890     eval_metric_ops = {metric_names[0]: (first_value_op, update_op)}
 891     with tf.control_dependencies([first_value_op]):
 892       for metric_name in metric_names[1:]:
 893         eval_metric_ops[metric_name] = (tf.py_func(
 894             value_func_factory(metric_name), [], np.float32), update_op)
 895     return eval_metric_ops

 919   def clear(self):

 921     self._image_id_to_mask_shape_map.clear()
 922     self._image_ids_with_detections.clear()
 923     self._groundtruth_list = []
 924     self._detection_masks_list = []

 926   def add_single_ground_truth_image_info(self,
 927                                          image_id,
 928                                          groundtruth_dict):
 929     """Adds groundtruth for a single image to be used for evaluation.
 930 
 931     If the image has already been added, a warning is logged, and groundtruth is
 932     ignored.
 933 
 934     Args:
 935       image_id: A unique string/integer identifier for the image.
 936       groundtruth_dict: A dictionary containing -
 937         InputDataFields.groundtruth_boxes: float32 numpy array of shape
 938           [num_boxes, 4] containing `num_boxes` groundtruth boxes of the format
 939           [ymin, xmin, ymax, xmax] in absolute image coordinates.
 940         InputDataFields.groundtruth_classes: integer numpy array of shape
 941           [num_boxes] containing 1-indexed groundtruth classes for the boxes.
 942         InputDataFields.groundtruth_instance_masks: uint8 numpy array of shape
 943           [num_boxes, image_height, image_width] containing groundtruth masks
 944           corresponding to the boxes. The elements of the array must be in
 945           {0, 1}.
 946     """
 947     if image_id in self._image_id_to_mask_shape_map:
 948       tf.logging.warning('Ignoring ground truth with image id %s since it was '
 949                          'previously added', image_id)
 950       return
 951 
 952     groundtruth_instance_masks = groundtruth_dict[
 953         standard_fields.InputDataFields.groundtruth_instance_masks]
 954     _check_mask_type_and_value(standard_fields.InputDataFields.
 955                                groundtruth_instance_masks,
 956                                groundtruth_instance_masks)
 957     self._groundtruth_list.extend(
 958         coco_tools.
 959         ExportSingleImageGroundtruthToCoco(
 960             image_id=image_id,
 961             next_annotation_id=self._annotation_id,
 962             category_id_set=self._category_id_set,
 963             groundtruth_boxes=groundtruth_dict[standard_fields.InputDataFields.
 964                                                groundtruth_boxes],
 965             groundtruth_classes=groundtruth_dict[standard_fields.
 966                                                  InputDataFields.
 967                                                  groundtruth_classes],
 968             groundtruth_masks=groundtruth_instance_masks))
 969     self._annotation_id += groundtruth_dict[standard_fields.InputDataFields.
 970                                             groundtruth_boxes].shape[0]
 971     self._image_id_to_mask_shape_map[image_id] = groundtruth_dict[
 972         standard_fields.InputDataFields.groundtruth_instance_masks].shape
 973 
 974   def add_single_detected_image_info(self,
 975                                      image_id,
 976                                      detections_dict):
 977     """Adds detections for a single image to be used for evaluation.
 978 
 979     If a detection has already been added for this image id, a warning is
 980     logged, and the detection is skipped.
 981 
 982     Args:
 983       image_id: A unique string/integer identifier for the image.
 984       detections_dict: A dictionary containing -
 985         DetectionResultFields.detection_scores: float32 numpy array of shape
 986           [num_boxes] containing detection scores for the boxes.
 987         DetectionResultFields.detection_classes: integer numpy array of shape
 988           [num_boxes] containing 1-indexed detection classes for the boxes.
 989         DetectionResultFields.detection_masks: optional uint8 numpy array of
 990           shape [num_boxes, image_height, image_width] containing instance
 991           masks corresponding to the boxes. The elements of the array must be
 992           in {0, 1}.
 993 
 994     Raises:
 995       ValueError: If groundtruth for the image_id is not available or if
 996         spatial shapes of groundtruth_instance_masks and detection_masks are
 997         incompatible.
 998     """
 999     if image_id not in self._image_id_to_mask_shape_map:
1000       raise ValueError('Missing groundtruth for image id: {}'.format(image_id))
1001 
1002     if image_id in self._image_ids_with_detections:
1003       tf.logging.warning('Ignoring detection with image id %s since it was '
1004                          'previously added', image_id)
1005       return
1006 
1007     groundtruth_masks_shape = self._image_id_to_mask_shape_map[image_id]
1008     detection_masks = detections_dict[standard_fields.DetectionResultFields.
1009                                       detection_masks]
1010     if groundtruth_masks_shape[1:] != detection_masks.shape[1:]:
1011       raise ValueError('Spatial shape of groundtruth masks and detection masks '
1012                        'are incompatible: {} vs {}'.format(
1013                            groundtruth_masks_shape,
1014                            detection_masks.shape))
1015     _check_mask_type_and_value(standard_fields.DetectionResultFields.
1016                                detection_masks,
1017                                detection_masks)
1018     self._detection_masks_list.extend(
1019         coco_tools.ExportSingleImageDetectionMasksToCoco(
1020             image_id=image_id,
1021             category_id_set=self._category_id_set,
1022             detection_masks=detection_masks,
1023             detection_scores=detections_dict[standard_fields.

1044   def evaluate(self):
1045     """Evaluates the detection masks and returns a dictionary of coco metrics.
1046 
1047     Returns:
1048       A dictionary holding -
1049 
1050       1. summary_metrics:
1051       'DetectionMasks_Precision/mAP': mean average precision over classes
1052         averaged over IOU thresholds ranging from .5 to .95 with .05 increments.
1053       'DetectionMasks_Precision/mAP@.50IOU': mean average precision at 50% IOU.
1054       'DetectionMasks_Precision/mAP@.75IOU': mean average precision at 75% IOU.
1055       'DetectionMasks_Precision/mAP (small)': mean average precision for small
1056         objects (area < 32^2 pixels).
1057       'DetectionMasks_Precision/mAP (medium)': mean average precision for medium
1058         sized objects (32^2 pixels < area < 96^2 pixels).
1059       'DetectionMasks_Precision/mAP (large)': mean average precision for large
1060         objects (96^2 pixels < area < 10000^2 pixels).
1061       'DetectionMasks_Recall/AR@1': average recall with 1 detection.
1062       'DetectionMasks_Recall/AR@10': average recall with 10 detections.
1063       'DetectionMasks_Recall/AR@100': average recall with 100 detections.
1064       'DetectionMasks_Recall/AR@100 (small)': average recall for small objects
1065         with 100 detections.
1066       'DetectionMasks_Recall/AR@100 (medium)': average recall for medium objects
1067         with 100 detections.
1068       'DetectionMasks_Recall/AR@100 (large)': average recall for large objects
1069         with 100 detections.
1070 
1071       2. per_category_ap: if include_metrics_per_category is True, category
1072       specific results with keys of the form:
1073       'Precision mAP ByCategory/category' (without the supercategory part if
1074       no supercategories exist). For backward compatibility
1075       'PerformanceByCategory' is included in the output regardless of
1076       all_metrics_per_category.
1077     """
1078     groundtruth_dict = {
1079         'annotations': self._groundtruth_list,
1080         'images': [{'id': image_id, 'height': shape[1], 'width': shape[2]}
1081                    for image_id, shape in self._image_id_to_mask_shape_map.
1082                    items()],
1083         'categories': self._categories
1084     }
1085     coco_wrapped_groundtruth = coco_tools.COCOWrapper(
1086         groundtruth_dict, detection_type='segmentation')
1087     coco_wrapped_detection_masks = coco_wrapped_groundtruth.LoadAnnotations(
1088         self._detection_masks_list)
1089     mask_evaluator = coco_tools.COCOEvalWrapper(
1090         coco_wrapped_groundtruth, coco_wrapped_detection_masks,
1091         agnostic_mode=False, iou_type='segm')
1092     mask_metrics, mask_per_category_ap = mask_evaluator.ComputeMetrics(
1093         include_metrics_per_category=self._include_metrics_per_category)
1094     mask_metrics.update(mask_per_category_ap)
1095     mask_metrics = {'DetectionMasks_'+ key: value
1096                     for key, value in mask_metrics.items()}
1097     return mask_metrics
1098 
1099   def add_eval_dict(self, eval_dict):
1100     """Observes an evaluation result dict for a single example.
1101 
1102     When executing eagerly, once all observations have been observed by this
1103     method you can use `.evaluate()` to get the final metrics.
1104 
1105     When using `tf.estimator.Estimator` for evaluation this function is used by
1106     `get_estimator_eval_metric_ops()` to construct the metric update op.
1107 
1108     Args:
1109       eval_dict: A dictionary that holds tensors for evaluating an object
1110         detection model, returned from
1111         eval_util.result_dict_for_single_example().
1112 
1113     Returns:
1114       None when executing eagerly, or an update_op that can be used to update
1115       the eval metrics in `tf.estimator.EstimatorSpec`.
1116     """
1117     def update_op(image_id_batched, groundtruth_boxes_batched,
1118                   groundtruth_classes_batched,
1119                   groundtruth_instance_masks_batched,
1120                   groundtruth_is_crowd_batched, num_gt_boxes_per_image,
1121                   detection_scores_batched, detection_classes_batched,
1122                   detection_masks_batched, num_det_boxes_per_image):
1123       """Update op for metrics."""
1124 
1125       for (image_id, groundtruth_boxes, groundtruth_classes,
1126            groundtruth_instance_masks, groundtruth_is_crowd, num_gt_box,
1127            detection_scores, detection_classes,
1128            detection_masks, num_det_box) in zip(
1129                image_id_batched, groundtruth_boxes_batched,
1130                groundtruth_classes_batched, groundtruth_instance_masks_batched,
1131                groundtruth_is_crowd_batched, num_gt_boxes_per_image,
1132                detection_scores_batched, detection_classes_batched,
1133                detection_masks_batched, num_det_boxes_per_image):
1134         self.add_single_ground_truth_image_info(
1135             image_id, {
1136                 'groundtruth_boxes':
1137                     groundtruth_boxes[:num_gt_box],
1138                 'groundtruth_classes':
1139                     groundtruth_classes[:num_gt_box],
1140                 'groundtruth_instance_masks':
1141                     groundtruth_instance_masks[:num_gt_box],
1142                 'groundtruth_is_crowd':
1143                     groundtruth_is_crowd[:num_gt_box]
1144             })
1145         self.add_single_detected_image_info(
1146             image_id, {
1147                 'detection_scores': detection_scores[:num_det_box],
1148                 'detection_classes': detection_classes[:num_det_box],
1149                 'detection_masks': detection_masks[:num_det_box]
1150             })
1151 
1152     # Unpack items from the evaluation dictionary.
1153     input_data_fields = standard_fields.InputDataFields
1154     detection_fields = standard_fields.DetectionResultFields
1155     image_id = eval_dict[input_data_fields.key]
1156     groundtruth_boxes = eval_dict[input_data_fields.groundtruth_boxes]
1157     groundtruth_classes = eval_dict[input_data_fields.groundtruth_classes]
1158     groundtruth_instance_masks = eval_dict[
1159         input_data_fields.groundtruth_instance_masks]
1160     groundtruth_is_crowd = eval_dict.get(
1161         input_data_fields.groundtruth_is_crowd, None)
1162     num_gt_boxes_per_image = eval_dict.get(
1163         input_data_fields.num_groundtruth_boxes, None)
1164     detection_scores = eval_dict[detection_fields.detection_scores]
1165     detection_classes = eval_dict[detection_fields.detection_classes]
1166     detection_masks = eval_dict[detection_fields.detection_masks]
1167     num_det_boxes_per_image = eval_dict.get(detection_fields.num_detections,
1168                                             None)
1169 
1170     if groundtruth_is_crowd is None:
1171       groundtruth_is_crowd = tf.zeros_like(groundtruth_classes, dtype=tf.bool)
1172 
1173     if not image_id.shape.as_list():
1174       # Apply a batch dimension to all tensors.
1175       image_id = tf.expand_dims(image_id, 0)
1176       groundtruth_boxes = tf.expand_dims(groundtruth_boxes, 0)
1177       groundtruth_classes = tf.expand_dims(groundtruth_classes, 0)
1178       groundtruth_instance_masks = tf.expand_dims(groundtruth_instance_masks, 0)
1179       groundtruth_is_crowd = tf.expand_dims(groundtruth_is_crowd, 0)
1180       detection_scores = tf.expand_dims(detection_scores, 0)
1181       detection_classes = tf.expand_dims(detection_classes, 0)
1182       detection_masks = tf.expand_dims(detection_masks, 0)
1183 
1184       if num_gt_boxes_per_image is None:
1185         num_gt_boxes_per_image = tf.shape(groundtruth_boxes)[1:2]
1186       else:
1187         num_gt_boxes_per_image = tf.expand_dims(num_gt_boxes_per_image, 0)
1188 
1189       if num_det_boxes_per_image is None:
1190         num_det_boxes_per_image = tf.shape(detection_scores)[1:2]
1191       else:
1192         num_det_boxes_per_image = tf.expand_dims(num_det_boxes_per_image, 0)
1193     else:
1194       if num_gt_boxes_per_image is None:
1195         num_gt_boxes_per_image = tf.tile(
1196             tf.shape(groundtruth_boxes)[1:2],
1197             multiples=tf.shape(groundtruth_boxes)[0:1])
1198       if num_det_boxes_per_image is None:
1199         num_det_boxes_per_image = tf.tile(
1200             tf.shape(detection_scores)[1:2],
1201             multiples=tf.shape(detection_scores)[0:1])
1202 
1203     return tf.py_func(update_op, [
1204         image_id, groundtruth_boxes, groundtruth_classes,
1205         groundtruth_instance_masks, groundtruth_is_crowd,
1206         num_gt_boxes_per_image, detection_scores, detection_classes,
1207         detection_masks, num_det_boxes_per_image
1208     ], [])
1209 
1210   def get_estimator_eval_metric_ops(self, eval_dict):
1211     """Returns a dictionary of eval metric ops.
1212 
1213     Note that once value_op is called, the detections and groundtruth added via
1214     update_op are cleared.
1215 
1216     Args:
1217       eval_dict: A dictionary that holds tensors for evaluating object detection
1218         performance. For single-image evaluation, this dictionary may be
1219         produced from eval_util.result_dict_for_single_example(). If multi-image
1220         evaluation, `eval_dict` should contain the fields
1221         'num_groundtruth_boxes_per_image' and 'num_det_boxes_per_image' to
1222         properly unpad the tensors from the batch.
1223 
1224     Returns:
1225       a dictionary of metric names to tuple of value_op and update_op that can
1226       be used as eval metric ops in tf.estimator.EstimatorSpec. Note that all
1227       update ops  must be run together and similarly all value ops must be run
1228       together to guarantee correct behaviour.
1229     """
1230     update_op = self.add_eval_dict(eval_dict)
1231     metric_names = ['DetectionMasks_Precision/mAP',
1232                     'DetectionMasks_Precision/mAP@.50IOU',
1233                     'DetectionMasks_Precision/mAP@.75IOU',
1234                     'DetectionMasks_Precision/mAP (large)',
1235                     'DetectionMasks_Precision/mAP (medium)',
1236                     'DetectionMasks_Precision/mAP (small)',
1237                     'DetectionMasks_Recall/AR@1',
1238                     'DetectionMasks_Recall/AR@10',
1239                     'DetectionMasks_Recall/AR@100',
1240                     'DetectionMasks_Recall/AR@100 (large)',
1241                     'DetectionMasks_Recall/AR@100 (medium)',
1242                     'DetectionMasks_Recall/AR@100 (small)']
1243     if self._include_metrics_per_category:
1244       for category_dict in self._categories:
1245         metric_names.append('DetectionMasks_PerformanceByCategory/mAP/' +
1246                             category_dict['name'])
1247 
1248     def first_value_func():
1249       self._metrics = self.evaluate()
1250       self.clear()
1251       return np.float32(self._metrics[metric_names[0]])
1252 
1253     def value_func_factory(metric_name):
1254       def value_func():
1255         return np.float32(self._metrics[metric_name])
1256       return value_func
1257 
1258     # Ensure that the metrics are only evaluated once.
1259     first_value_op = tf.py_func(first_value_func, [], tf.float32)
1260     eval_metric_ops = {metric_names[0]: (first_value_op, update_op)}
1261     with tf.control_dependencies([first_value_op]):
1262       for metric_name in metric_names[1:]:
1263         eval_metric_ops[metric_name] = (tf.py_func(
1264             value_func_factory(metric_name), [], np.float32), update_op)
1265     return eval_metric_ops
