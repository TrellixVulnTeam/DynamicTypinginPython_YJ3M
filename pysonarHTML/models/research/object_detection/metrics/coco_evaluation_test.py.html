 173   def testRejectionOnDuplicateGroundtruth(self):
 174     """Tests that groundtruth cannot be added more than once for an image."""
 175     coco_evaluator = coco_evaluation.CocoDetectionEvaluator(
 176         _get_categories_list())
 177     #  Add groundtruth
 178     image_key1 = 'img1'
 179     groundtruth_boxes1 = np.array([[0, 0, 1, 1], [0, 0, 2, 2], [0, 0, 3, 3]],
 180                                   dtype=float)
 181     groundtruth_class_labels1 = np.array([1, 3, 1], dtype=int)
 182     coco_evaluator.add_single_ground_truth_image_info(image_key1, {
 183         standard_fields.InputDataFields.groundtruth_boxes:
 184             groundtruth_boxes1,
 185         standard_fields.InputDataFields.groundtruth_classes:
 186             groundtruth_class_labels1
 187     })
 188     groundtruth_lists_len = len(coco_evaluator._groundtruth_list)
 189 
 190     # Add groundtruth with the same image id.
 191     coco_evaluator.add_single_ground_truth_image_info(image_key1, {
 192         standard_fields.InputDataFields.groundtruth_boxes:
 193             groundtruth_boxes1,
 194         standard_fields.InputDataFields.groundtruth_classes:
 195             groundtruth_class_labels1
 196     })
 197     self.assertEqual(groundtruth_lists_len,
 198                      len(coco_evaluator._groundtruth_list))
