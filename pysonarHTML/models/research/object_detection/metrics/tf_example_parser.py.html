  15 """Tensorflow Example proto parser for data loading.

  17 A parser to decode data containing serialized tensorflow.Example
  18 protos into materialized tensors (numpy arrays).
  19 """

  27 class FloatParser(data_parser.DataToNumpyParser):
  28   """Tensorflow Example float parser."""
  29 
  30   def __init__(self, field_name):
  31     self.field_name = field_name
  32 
  33   def parse(self, tf_example):
  34     return np.array(
  35         tf_example.features.feature[self.field_name].float_list.value,
  36         dtype=np.float).transpose() if tf_example.features.feature[
  37             self.field_name].HasField("float_list") else None
  38

  40 class StringParser(data_parser.DataToNumpyParser):
  41   """Tensorflow Example string parser."""
  42 
  43   def __init__(self, field_name):
  44     self.field_name = field_name
  45 
  46   def parse(self, tf_example):
  47     return b"".join(tf_example.features.feature[
  48         self.field_name].bytes_list.value) if tf_example.features.feature[
  49             self.field_name].HasField("bytes_list") else None

  65 class BoundingBoxParser(data_parser.DataToNumpyParser):
  66   """Tensorflow Example bounding box parser."""
  67 
  68   def __init__(self, xmin_field_name, ymin_field_name, xmax_field_name,
  69                ymax_field_name):
  70     self.field_names = [
  71         ymin_field_name, xmin_field_name, ymax_field_name, xmax_field_name
  72     ]
  73 
  74   def parse(self, tf_example):
  75     result = []
  76     parsed = True
  77     for field_name in self.field_names:
  78       result.append(tf_example.features.feature[field_name].float_list.value)
  79       parsed &= (
  80           tf_example.features.feature[field_name].HasField("float_list"))
  81 
  82     return np.array(result).transpose() if parsed else None

  85 class TfExampleDetectionAndGTParser(data_parser.DataToNumpyParser):
  86   """Tensorflow Example proto parser."""
  87 
  88   def __init__(self):
  89     self.items_to_handlers = {
  90         fields.DetectionResultFields.key:
  91             StringParser(fields.TfExampleFields.source_id),
  92         # Object ground truth boxes and classes.
  93         fields.InputDataFields.groundtruth_boxes: (BoundingBoxParser(
  94             fields.TfExampleFields.object_bbox_xmin,
  95             fields.TfExampleFields.object_bbox_ymin,
  96             fields.TfExampleFields.object_bbox_xmax,
  97             fields.TfExampleFields.object_bbox_ymax)),
  98         fields.InputDataFields.groundtruth_classes: (
  99             Int64Parser(fields.TfExampleFields.object_class_label)),
 100         # Object detections.
 101         fields.DetectionResultFields.detection_boxes: (BoundingBoxParser(
 102             fields.TfExampleFields.detection_bbox_xmin,
 103             fields.TfExampleFields.detection_bbox_ymin,
 104             fields.TfExampleFields.detection_bbox_xmax,
 105             fields.TfExampleFields.detection_bbox_ymax)),
 106         fields.DetectionResultFields.detection_classes: (
 107             Int64Parser(fields.TfExampleFields.detection_class_label)),
 108         fields.DetectionResultFields.detection_scores: (
 109             FloatParser(fields.TfExampleFields.detection_score)),
 110     }
 111 
 112     self.optional_items_to_handlers = {
 113         fields.InputDataFields.groundtruth_difficult:
 114             Int64Parser(fields.TfExampleFields.object_difficult),
 115         fields.InputDataFields.groundtruth_group_of:
 116             Int64Parser(fields.TfExampleFields.object_group_of),
 117         fields.InputDataFields.groundtruth_image_classes:
 118             Int64Parser(fields.TfExampleFields.image_class_label),
 119     }
 120 
 121   def parse(self, tf_example):
 122     """Parses tensorflow example and returns a tensor dictionary.
 123 
 124     Args:
 125       tf_example: a tf.Example object.
 126 
 127     Returns:
 128       A dictionary of the following numpy arrays:
 129       fields.DetectionResultFields.source_id - string containing original image
 130       id.
 131       fields.InputDataFields.groundtruth_boxes - a numpy array containing
 132       groundtruth boxes.
 133       fields.InputDataFields.groundtruth_classes - a numpy array containing
 134       groundtruth classes.
 135       fields.InputDataFields.groundtruth_group_of - a numpy array containing
 136       groundtruth group of flag (optional, None if not specified).
 137       fields.InputDataFields.groundtruth_difficult - a numpy array containing
 138       groundtruth difficult flag (optional, None if not specified).
 139       fields.InputDataFields.groundtruth_image_classes - a numpy array
 140       containing groundtruth image-level labels.
 141       fields.DetectionResultFields.detection_boxes - a numpy array containing
 142       detection boxes.
 143       fields.DetectionResultFields.detection_classes - a numpy array containing
 144       detection class labels.
 145       fields.DetectionResultFields.detection_scores - a numpy array containing
 146       detection scores.
 147       Returns None if tf.Example was not parsed or non-optional fields were not
 148       found.
 149     """
 150     results_dict = {}
 151     parsed = True
 152     for key, parser in self.items_to_handlers.items():
 153       results_dict[key] = parser.parse(tf_example)
 154       parsed &= (results_dict[key] is not None)
 155 
 156     for key, parser in self.optional_items_to_handlers.items():
 157       results_dict[key] = parser.parse(tf_example)
 158 
 159     return results_dict if parsed else None
