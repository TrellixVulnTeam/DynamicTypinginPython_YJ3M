  15 """Tests for object_detection.data_decoders.tf_example_parser."""

  17 import numpy as np
  18 import numpy.testing as np_testing
  19 import tensorflow.compat.v1 as tf
  20 
  21 from object_detection.core import standard_fields as fields
  22 from object_detection.metrics import tf_example_parser
  23 
  24 
  25 class TfExampleDecoderTest(tf.test.TestCase):
  26 
  27   def _Int64Feature(self, value):
  28     return tf.train.Feature(int64_list=tf.train.Int64List(value=value))
  29 
  30   def _FloatFeature(self, value):
  31     return tf.train.Feature(float_list=tf.train.FloatList(value=value))
  32 
  33   def _BytesFeature(self, value):
  34     return tf.train.Feature(bytes_list=tf.train.BytesList(value=[value]))
  35 
  36   def testParseDetectionsAndGT(self):
  37     source_id = b'abc.jpg'
  38     # y_min, x_min, y_max, x_max
  39     object_bb = np.array([[0.0, 0.5, 0.3], [0.0, 0.1, 0.6], [1.0, 0.6, 0.8],
  40                           [1.0, 0.6, 0.7]]).transpose()
  41     detection_bb = np.array([[0.1, 0.2], [0.0, 0.8], [1.0, 0.6],
  42                              [1.0, 0.85]]).transpose()
  43 
  44     object_class_label = [1, 1, 2]
  45     object_difficult = [1, 0, 0]
  46     object_group_of = [0, 0, 1]
  47     verified_labels = [1, 2, 3, 4]
  48     detection_class_label = [2, 1]
  49     detection_score = [0.5, 0.3]
  50     features = {
  51         fields.TfExampleFields.source_id:
  52             self._BytesFeature(source_id),
  53         fields.TfExampleFields.object_bbox_ymin:
  54             self._FloatFeature(object_bb[:, 0].tolist()),
  55         fields.TfExampleFields.object_bbox_xmin:
  56             self._FloatFeature(object_bb[:, 1].tolist()),
  57         fields.TfExampleFields.object_bbox_ymax:
  58             self._FloatFeature(object_bb[:, 2].tolist()),
  59         fields.TfExampleFields.object_bbox_xmax:
  60             self._FloatFeature(object_bb[:, 3].tolist()),
  61         fields.TfExampleFields.detection_bbox_ymin:
  62             self._FloatFeature(detection_bb[:, 0].tolist()),
  63         fields.TfExampleFields.detection_bbox_xmin:
  64             self._FloatFeature(detection_bb[:, 1].tolist()),
  65         fields.TfExampleFields.detection_bbox_ymax:
  66             self._FloatFeature(detection_bb[:, 2].tolist()),
  67         fields.TfExampleFields.detection_bbox_xmax:
  68             self._FloatFeature(detection_bb[:, 3].tolist()),
  69         fields.TfExampleFields.detection_class_label:
  70             self._Int64Feature(detection_class_label),
  71         fields.TfExampleFields.detection_score:
  72             self._FloatFeature(detection_score),
  73     }
  74 
  75     example = tf.train.Example(features=tf.train.Features(feature=features))
  76     parser = tf_example_parser.TfExampleDetectionAndGTParser()
  77 
  78     results_dict = parser.parse(example)
  79     self.assertIsNone(results_dict)
  80 
  81     features[fields.TfExampleFields.object_class_label] = (
  82         self._Int64Feature(object_class_label))
  83     features[fields.TfExampleFields.object_difficult] = (
  84         self._Int64Feature(object_difficult))
  85 
  86     example = tf.train.Example(features=tf.train.Features(feature=features))
  87     results_dict = parser.parse(example)
  88 
  89     self.assertIsNotNone(results_dict)
  90     self.assertEqual(source_id, results_dict[fields.DetectionResultFields.key])
  91     np_testing.assert_almost_equal(
  92         object_bb, results_dict[fields.InputDataFields.groundtruth_boxes])
  93     np_testing.assert_almost_equal(
  94         detection_bb,
  95         results_dict[fields.DetectionResultFields.detection_boxes])
  96     np_testing.assert_almost_equal(
  97         detection_score,
  98         results_dict[fields.DetectionResultFields.detection_scores])
  99     np_testing.assert_almost_equal(
 100         detection_class_label,
 101         results_dict[fields.DetectionResultFields.detection_classes])
 102     np_testing.assert_almost_equal(
 103         object_difficult,
 104         results_dict[fields.InputDataFields.groundtruth_difficult])
 105     np_testing.assert_almost_equal(
 106         object_class_label,
 107         results_dict[fields.InputDataFields.groundtruth_classes])
 108 
 109     parser = tf_example_parser.TfExampleDetectionAndGTParser()
 110 
 111     features[fields.TfExampleFields.object_group_of] = (
 112         self._Int64Feature(object_group_of))
 113 
 114     example = tf.train.Example(features=tf.train.Features(feature=features))
 115     results_dict = parser.parse(example)
 116     self.assertIsNotNone(results_dict)
 117     np_testing.assert_equal(
 118         object_group_of,
 119         results_dict[fields.InputDataFields.groundtruth_group_of])
 120 
 121     features[fields.TfExampleFields.image_class_label] = (
 122         self._Int64Feature(verified_labels))
 123 
 124     example = tf.train.Example(features=tf.train.Features(feature=features))
 125     results_dict = parser.parse(example)
 126     self.assertIsNotNone(results_dict)
 127     np_testing.assert_equal(
 128         verified_labels,
 129         results_dict[fields.InputDataFields.groundtruth_image_classes])

 131   def testParseString(self):
 132     string_val = b'abc'
 133     features = {'string': self._BytesFeature(string_val)}
 134     example = tf.train.Example(features=tf.train.Features(feature=features))
 135 
 136     parser = tf_example_parser.StringParser('string')
 137     result = parser.parse(example)
 138     self.assertIsNotNone(result)
 139     self.assertEqual(result, string_val)
 140 
 141     parser = tf_example_parser.StringParser('another_string')
 142     result = parser.parse(example)
 143     self.assertIsNone(result)
 144 
 145   def testParseFloat(self):
 146     float_array_val = [1.5, 1.4, 2.0]
 147     features = {'floats': self._FloatFeature(float_array_val)}

 154 
 155     parser = tf_example_parser.StringParser('another_floats')
 156     result = parser.parse(example)
 157     self.assertIsNone(result)

 173   def testBoundingBoxParser(self):
 174     bounding_boxes = np.array([[0.0, 0.5, 0.3], [0.0, 0.1, 0.6],
 175                                [1.0, 0.6, 0.8], [1.0, 0.6, 0.7]]).transpose()
 176     features = {
 177         'ymin': self._FloatFeature(bounding_boxes[:, 0]),
 178         'xmin': self._FloatFeature(bounding_boxes[:, 1]),
 179         'ymax': self._FloatFeature(bounding_boxes[:, 2]),
 180         'xmax': self._FloatFeature(bounding_boxes[:, 3])
 181     }
 182 
 183     example = tf.train.Example(features=tf.train.Features(feature=features))
 184 
 185     parser = tf_example_parser.BoundingBoxParser('xmin', 'ymin', 'xmax', 'ymax')
 186     result = parser.parse(example)
 187     self.assertIsNotNone(result)
 188     np_testing.assert_almost_equal(result, bounding_boxes)
 189 
 190     parser = tf_example_parser.BoundingBoxParser('xmin', 'ymin', 'xmax',
 191                                                  'another_ymax')
 192     result = parser.parse(example)
 193     self.assertIsNone(result)
