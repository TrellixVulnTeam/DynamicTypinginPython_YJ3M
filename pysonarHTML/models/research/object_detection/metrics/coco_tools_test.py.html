  30   def setUp(self):
  31     groundtruth_annotations_list = [
  32         {
  33             'id': 1,
  34             'image_id': 'first',
  35             'category_id': 1,
  36             'bbox': [100., 100., 100., 100.],
  37             'area': 100.**2,
  38             'iscrowd': 0
  39         },
  40         {
  41             'id': 2,
  42             'image_id': 'second',
  43             'category_id': 1,
  44             'bbox': [50., 50., 50., 50.],
  45             'area': 50.**2,
  46             'iscrowd': 0
  47         },
  48     ]
  49     image_list = [{'id': 'first'}, {'id': 'second'}]
  50     category_list = [{'id': 0, 'name': 'person'},
  51                      {'id': 1, 'name': 'cat'},
  52                      {'id': 2, 'name': 'dog'}]
  53     self._groundtruth_dict = {
  54         'annotations': groundtruth_annotations_list,
  55         'images': image_list,
  56         'categories': category_list
  57     }
  58 
  59     self._detections_list = [
  60         {
  61             'image_id': 'first',
  62             'category_id': 1,
  63             'bbox': [100., 100., 100., 100.],
  64             'score': .8
  65         },
  66         {
  67             'image_id': 'second',
  68             'category_id': 1,
  69             'bbox': [50., 50., 50., 50.],
  70             'score': .7
  71         },
  72     ]

  74   def testCocoWrappers(self):
  75     groundtruth = coco_tools.COCOWrapper(self._groundtruth_dict)
  76     detections = groundtruth.LoadAnnotations(self._detections_list)
  77     evaluator = coco_tools.COCOEvalWrapper(groundtruth, detections)
  78     summary_metrics, _ = evaluator.ComputeMetrics()
  79     self.assertAlmostEqual(1.0, summary_metrics['Precision/mAP'])

  81   def testExportGroundtruthToCOCO(self):
  82     image_ids = ['first', 'second']
  83     groundtruth_boxes = [np.array([[100, 100, 200, 200]], np.float),
  84                          np.array([[50, 50, 100, 100]], np.float)]
  85     groundtruth_classes = [np.array([1], np.int32), np.array([1], np.int32)]
  86     categories = [{'id': 0, 'name': 'person'},
  87                   {'id': 1, 'name': 'cat'},
  88                   {'id': 2, 'name': 'dog'}]
  89     output_path = os.path.join(tf.test.get_temp_dir(), 'groundtruth.json')
  90     result = coco_tools.ExportGroundtruthToCOCO(
  91         image_ids,
  92         groundtruth_boxes,
  93         groundtruth_classes,
  94         categories,
  95         output_path=output_path)
  96     self.assertDictEqual(result, self._groundtruth_dict)
  97     with tf.gfile.GFile(output_path, 'r') as f:
  98       written_result = f.read()
  99       # The json output should have floats written to 4 digits of precision.
 100       matcher = re.compile(r'"bbox":\s+\[\n\s+\d+.\d\d\d\d,', re.MULTILINE)
 101       self.assertTrue(matcher.findall(written_result))
 102       written_result = json.loads(written_result)
 103       self.assertAlmostEqual(result, written_result)

 131   def testExportSegmentsToCOCO(self):
 132     image_ids = ['first', 'second']
 133     detection_masks = [np.array(
 134         [[[0, 1, 0, 1], [0, 1, 1, 0], [0, 0, 0, 1], [0, 1, 0, 1]]],
 135         dtype=np.uint8), np.array(
 136             [[[0, 1, 0, 1], [0, 1, 1, 0], [0, 0, 0, 1], [0, 1, 0, 1]]],
 137             dtype=np.uint8)]
 138 
 139     for i, detection_mask in enumerate(detection_masks):
 140       detection_masks[i] = detection_mask[:, :, :, None]
 141 
 142     detection_scores = [np.array([.8], np.float), np.array([.7], np.float)]
 143     detection_classes = [np.array([1], np.int32), np.array([1], np.int32)]
 144 
 145     categories = [{'id': 0, 'name': 'person'},
 146                   {'id': 1, 'name': 'cat'},
 147                   {'id': 2, 'name': 'dog'}]
 148     output_path = os.path.join(tf.test.get_temp_dir(), 'segments.json')
 149     result = coco_tools.ExportSegmentsToCOCO(
 150         image_ids,
 151         detection_masks,
 152         detection_scores,
 153         detection_classes,
 154         categories,
 155         output_path=output_path)
 156     with tf.gfile.GFile(output_path, 'r') as f:
 157       written_result = f.read()
 158       written_result = json.loads(written_result)
 159       mask_load = mask.decode([written_result[0]['segmentation']])
 160       self.assertTrue(np.allclose(mask_load, detection_masks[0]))
 161       self.assertAlmostEqual(result, written_result)

 195   def testSingleImageDetectionBoxesExport(self):
 196     boxes = np.array([[0, 0, 1, 1],
 197                       [0, 0, .5, .5],
 198                       [.5, .5, 1, 1]], dtype=np.float32)
 199     classes = np.array([1, 2, 3], dtype=np.int32)
 200     scores = np.array([0.8, 0.2, 0.7], dtype=np.float32)
 201     coco_boxes = np.array([[0, 0, 1, 1],
 202                            [0, 0, .5, .5],
 203                            [.5, .5, .5, .5]], dtype=np.float32)
 204     coco_annotations = coco_tools.ExportSingleImageDetectionBoxesToCoco(
 205         image_id='first_image',
 206         category_id_set=set([1, 2, 3]),
 207         detection_boxes=boxes,
 208         detection_classes=classes,
 209         detection_scores=scores)
 210     for i, annotation in enumerate(coco_annotations):
 211       self.assertEqual(annotation['image_id'], 'first_image')
 212       self.assertEqual(annotation['category_id'], classes[i])
 213       self.assertAlmostEqual(annotation['score'], scores[i])
 214       self.assertTrue(np.all(np.isclose(annotation['bbox'], coco_boxes[i])))

 239   def testSingleImageGroundtruthExport(self):
 240     masks = np.array(
 241         [[[1, 1,], [1, 1]],
 242          [[0, 0], [0, 1]],
 243          [[0, 0], [0, 0]]], dtype=np.uint8)
 244     boxes = np.array([[0, 0, 1, 1],
 245                       [0, 0, .5, .5],

 256     coco_annotations = coco_tools.ExportSingleImageGroundtruthToCoco(

 271       self.assertEqual(annotation['id'], i + next_annotation_id)

 293   def testSingleImageGroundtruthExportWithKeypoints(self):
 294     boxes = np.array([[0, 0, 1, 1],
 295                       [0, 0, .5, .5],
 296                       [.5, .5, 1, 1]], dtype=np.float32)
 297     coco_boxes = np.array([[0, 0, 1, 1],
 298                            [0, 0, .5, .5],
 299                            [.5, .5, .5, .5]], dtype=np.float32)
 300     keypoints = np.array([[[0, 0], [0.25, 0.25], [0.75, 0.75]],
 301                           [[0, 0], [0.125, 0.125], [0.375, 0.375]],
 302                           [[0.5, 0.5], [0.75, 0.75], [1.0, 1.0]]],
 303                          dtype=np.float32)
 304     visibilities = np.array([[2, 2, 2],
 305                              [2, 2, 0],
 306                              [2, 0, 0]], dtype=np.int32)
 307     areas = np.array([15., 16., 17.])
 308 
 309     classes = np.array([1, 2, 3], dtype=np.int32)
 310     is_crowd = np.array([0, 1, 0], dtype=np.int32)
 311     next_annotation_id = 1
 312 
 313     # Tests exporting without passing in is_crowd (for backward compatibility).
 314     coco_annotations = coco_tools.ExportSingleImageGroundtruthToCoco(
 315         image_id='first_image',
 316         category_id_set=set([1, 2, 3]),
 317         next_annotation_id=next_annotation_id,
 318         groundtruth_boxes=boxes,
 319         groundtruth_classes=classes,
 320         groundtruth_keypoints=keypoints,
 321         groundtruth_keypoint_visibilities=visibilities,
 322         groundtruth_area=areas)
 323     for i, annotation in enumerate(coco_annotations):
 324       self.assertTrue(np.all(np.isclose(annotation['bbox'], coco_boxes[i])))
 325       self.assertEqual(annotation['image_id'], 'first_image')
 326       self.assertEqual(annotation['category_id'], classes[i])
 327       self.assertEqual(annotation['id'], i + next_annotation_id)
 328       self.assertEqual(annotation['num_keypoints'], 3 - i)
 329       self.assertEqual(annotation['area'], 15.0 + i)
 330       self.assertTrue(
 331           np.all(np.isclose(annotation['keypoints'][0::3], keypoints[i, :, 1])))
 332       self.assertTrue(
 333           np.all(np.isclose(annotation['keypoints'][1::3], keypoints[i, :, 0])))
 334       self.assertTrue(
 335           np.all(np.equal(annotation['keypoints'][2::3], visibilities[i])))
 336 
 337     # Tests exporting with is_crowd.
 338     coco_annotations = coco_tools.ExportSingleImageGroundtruthToCoco(
 339         image_id='first_image',
 340         category_id_set=set([1, 2, 3]),
 341         next_annotation_id=next_annotation_id,
 342         groundtruth_boxes=boxes,
 343         groundtruth_classes=classes,
 344         groundtruth_keypoints=keypoints,
 345         groundtruth_keypoint_visibilities=visibilities,
 346         groundtruth_is_crowd=is_crowd)
 347     for i, annotation in enumerate(coco_annotations):
 348       self.assertTrue(np.all(np.isclose(annotation['bbox'], coco_boxes[i])))
 349       self.assertEqual(annotation['image_id'], 'first_image')
 350       self.assertEqual(annotation['category_id'], classes[i])
 351       self.assertEqual(annotation['iscrowd'], is_crowd[i])
 352       self.assertEqual(annotation['id'], i + next_annotation_id)
 353       self.assertEqual(annotation['num_keypoints'], 3 - i)
 354       self.assertTrue(
 355           np.all(np.isclose(annotation['keypoints'][0::3], keypoints[i, :, 1])))
 356       self.assertTrue(
 357           np.all(np.isclose(annotation['keypoints'][1::3], keypoints[i, :, 0])))
 358       self.assertTrue(
 359           np.all(np.equal(annotation['keypoints'][2::3], visibilities[i])))
 360       # Testing the area values are derived from the bounding boxes.
 361       if i == 0:
 362         self.assertAlmostEqual(annotation['area'], 1.0)
 363       else:
 364         self.assertAlmostEqual(annotation['area'], 0.25)
