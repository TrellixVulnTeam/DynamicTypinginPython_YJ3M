  59 class _ClassTensorHandler(slim_example_decoder.Tensor):
  60   """An ItemHandler to fetch class ids from class text."""
  61 
  62   def __init__(self,
  63                tensor_key,
  64                label_map_proto_file,
  65                shape_keys=None,
  66                shape=None,
  67                default_value=''):
  68     """Initializes the LookupTensor handler.
  69 
  70     Simply calls a vocabulary (most often, a label mapping) lookup.
  71 
  72     Args:
  73       tensor_key: the name of the `TFExample` feature to read the tensor from.
  74       label_map_proto_file: File path to a text format LabelMapProto message
  75         mapping class text to id.
  76       shape_keys: Optional name or list of names of the TF-Example feature in
  77         which the tensor shape is stored. If a list, then each corresponds to
  78         one dimension of the shape.
  79       shape: Optional output shape of the `Tensor`. If provided, the `Tensor` is
  80         reshaped accordingly.
  81       default_value: The value used when the `tensor_key` is not found in a
  82         particular `TFExample`.
  83 
  84     Raises:
  85       ValueError: if both `shape_keys` and `shape` are specified.
  86     """
  87     name_to_id = label_map_util.get_label_map_dict(
  88         label_map_proto_file, use_display_name=False)
  89     # We use a default_value of -1, but we expect all labels to be contained
  90     # in the label map.
  91     try:
  92       # Dynamically try to load the tf v2 lookup, falling back to contrib
  93       lookup = tf.compat.v2.lookup
  94       hash_table_class = tf.compat.v2.lookup.StaticHashTable
  95     except AttributeError:
  96       lookup = contrib_lookup
  97       hash_table_class = contrib_lookup.HashTable
  98     name_to_id_table = hash_table_class(
  99         initializer=lookup.KeyValueTensorInitializer(
 100             keys=tf.constant(list(name_to_id.keys())),
 101             values=tf.constant(list(name_to_id.values()), dtype=tf.int64)),
 102         default_value=-1)
 103     display_name_to_id = label_map_util.get_label_map_dict(
 104         label_map_proto_file, use_display_name=True)
 105     # We use a default_value of -1, but we expect all labels to be contained
 106     # in the label map.
 107     display_name_to_id_table = hash_table_class(
 108         initializer=lookup.KeyValueTensorInitializer(
 109             keys=tf.constant(list(display_name_to_id.keys())),
 110             values=tf.constant(
 111                 list(display_name_to_id.values()), dtype=tf.int64)),
 112         default_value=-1)
 113 
 114     self._name_to_id_table = name_to_id_table
 115     self._display_name_to_id_table = display_name_to_id_table
 116     super(_ClassTensorHandler, self).__init__(tensor_key, shape_keys, shape,
 117                                               default_value)
 118 
 119   def tensors_to_item(self, keys_to_tensors):
 120     unmapped_tensor = super(_ClassTensorHandler,
 121                             self).tensors_to_item(keys_to_tensors)
 122     return tf.maximum(self._name_to_id_table.lookup(unmapped_tensor),
 123                       self._display_name_to_id_table.lookup(unmapped_tensor))

 126 class _BackupHandler(slim_example_decoder.ItemHandler):
 127   """An ItemHandler that tries two ItemHandlers in order."""
 128 
 129   def __init__(self, handler, backup):
 130     """Initializes the BackupHandler handler.
 131 
 132     If the first Handler's tensors_to_item returns a Tensor with no elements,
 133     the second Handler is used.
 134 
 135     Args:
 136       handler: The primary ItemHandler.
 137       backup: The backup ItemHandler.
 138 
 139     Raises:
 140       ValueError: if either is not an ItemHandler.
 141     """
 142     if not isinstance(handler, slim_example_decoder.ItemHandler):
 143       raise ValueError('Primary handler is of type %s instead of ItemHandler' %
 144                        type(handler))
 145     if not isinstance(backup, slim_example_decoder.ItemHandler):
 146       raise ValueError(
 147           'Backup handler is of type %s instead of ItemHandler' % type(backup))
 148     self._handler = handler
 149     self._backup = backup
 150     super(_BackupHandler, self).__init__(handler.keys + backup.keys)
 151 
 152   def tensors_to_item(self, keys_to_tensors):
 153     item = self._handler.tensors_to_item(keys_to_tensors)
 154     return tf.cond(
 155         pred=tf.equal(tf.reduce_prod(tf.shape(item)), 0),
 156         true_fn=lambda: self._backup.tensors_to_item(keys_to_tensors),
 157         false_fn=lambda: item)

 160 class TfExampleDecoder(data_decoder.DataDecoder):
 161   """Tensorflow Example proto decoder."""
 162 
 163   def __init__(self,
 164                load_instance_masks=False,
 165                instance_mask_type=input_reader_pb2.NUMERICAL_MASKS,
 166                label_map_proto_file=None,
 167                use_display_name=False,
 168                dct_method='',
 169                num_keypoints=0,
 170                num_additional_channels=0,
 171                load_multiclass_scores=False,
 172                load_context_features=False,
 173                expand_hierarchy_labels=False):
 174     """Constructor sets keys_to_features and items_to_handlers.
 175 
 176     Args:
 177       load_instance_masks: whether or not to load and handle instance masks.
 178       instance_mask_type: type of instance masks. Options are provided in
 179         input_reader.proto. This is only used if `load_instance_masks` is True.
 180       label_map_proto_file: a file path to a
 181         object_detection.protos.StringIntLabelMap proto. If provided, then the
 182         mapped IDs of 'image/object/class/text' will take precedence over the
 183         existing 'image/object/class/label' ID.  Also, if provided, it is
 184         assumed that 'image/object/class/text' will be in the data.
 185       use_display_name: whether or not to use the `display_name` for label
 186         mapping (instead of `name`).  Only used if label_map_proto_file is
 187         provided.
 188       dct_method: An optional string. Defaults to None. It only takes
 189         effect when image format is jpeg, used to specify a hint about the
 190         algorithm used for jpeg decompression. Currently valid values
 191         are ['INTEGER_FAST', 'INTEGER_ACCURATE']. The hint may be ignored, for
 192         example, the jpeg library does not have that specific option.
 193       num_keypoints: the number of keypoints per object.
 194       num_additional_channels: how many additional channels to use.
 195       load_multiclass_scores: Whether to load multiclass scores associated with
 196         boxes.
 197       load_context_features: Whether to load information from context_features,
 198         to provide additional context to a detection model for training and/or
 199         inference.
 200       expand_hierarchy_labels: Expands the object and image labels taking into
 201         account the provided hierarchy in the label_map_proto_file. For positive
 202         classes, the labels are extended to ancestor. For negative classes,
 203         the labels are expanded to descendants.
 204 
 205     Raises:
 206       ValueError: If `instance_mask_type` option is not one of
 207         input_reader_pb2.DEFAULT, input_reader_pb2.NUMERICAL, or
 208         input_reader_pb2.PNG_MASKS.
 209       ValueError: If `expand_labels_hierarchy` is True, but the
 210         `label_map_proto_file` is not provided.
 211     """
 212     # TODO(rathodv): delete unused `use_display_name` argument once we change
 213     # other decoders to handle label maps similarly.
 214     del use_display_name
 215     self.keys_to_features = {
 216         'image/encoded':
 217             tf.FixedLenFeature((), tf.string, default_value=''),
 218         'image/format':
 219             tf.FixedLenFeature((), tf.string, default_value='jpeg'),
 220         'image/filename':
 221             tf.FixedLenFeature((), tf.string, default_value=''),
 222         'image/key/sha256':
 223             tf.FixedLenFeature((), tf.string, default_value=''),
 224         'image/source_id':
 225             tf.FixedLenFeature((), tf.string, default_value=''),
 226         'image/height':
 227             tf.FixedLenFeature((), tf.int64, default_value=1),
 228         'image/width':
 229             tf.FixedLenFeature((), tf.int64, default_value=1),
 230         # Image-level labels.
 231         'image/class/text':
 232             tf.VarLenFeature(tf.string),
 233         'image/class/label':
 234             tf.VarLenFeature(tf.int64),
 235         'image/class/confidence':
 236             tf.VarLenFeature(tf.float32),
 237         # Object boxes and classes.
 238         'image/object/bbox/xmin':
 239             tf.VarLenFeature(tf.float32),
 240         'image/object/bbox/xmax':
 241             tf.VarLenFeature(tf.float32),
 242         'image/object/bbox/ymin':
 243             tf.VarLenFeature(tf.float32),
 244         'image/object/bbox/ymax':
 245             tf.VarLenFeature(tf.float32),
 246         'image/object/class/label':
 247             tf.VarLenFeature(tf.int64),
 248         'image/object/class/text':
 249             tf.VarLenFeature(tf.string),
 250         'image/object/area':
 251             tf.VarLenFeature(tf.float32),
 252         'image/object/is_crowd':
 253             tf.VarLenFeature(tf.int64),
 254         'image/object/difficult':
 255             tf.VarLenFeature(tf.int64),
 256         'image/object/group_of':
 257             tf.VarLenFeature(tf.int64),
 258         'image/object/weight':
 259             tf.VarLenFeature(tf.float32),
 260 
 261     }
 262     # We are checking `dct_method` instead of passing it directly in order to
 263     # ensure TF version 1.6 compatibility.
 264     if dct_method:
 265       image = slim_example_decoder.Image(
 266           image_key='image/encoded',
 267           format_key='image/format',
 268           channels=3,
 269           dct_method=dct_method)
 270       additional_channel_image = slim_example_decoder.Image(
 271           image_key='image/additional_channels/encoded',
 272           format_key='image/format',
 273           channels=1,
 274           repeated=True,
 275           dct_method=dct_method)
 276     else:
 277       image = slim_example_decoder.Image(
 278           image_key='image/encoded', format_key='image/format', channels=3)
 279       additional_channel_image = slim_example_decoder.Image(
 280           image_key='image/additional_channels/encoded',
 281           format_key='image/format',
 282           channels=1,
 283           repeated=True)
 284     self.items_to_handlers = {
 285         fields.InputDataFields.image:
 286             image,
 287         fields.InputDataFields.source_id: (
 288             slim_example_decoder.Tensor('image/source_id')),
 289         fields.InputDataFields.key: (
 290             slim_example_decoder.Tensor('image/key/sha256')),
 291         fields.InputDataFields.filename: (
 292             slim_example_decoder.Tensor('image/filename')),
 293         # Image-level labels.
 294         fields.InputDataFields.groundtruth_image_confidences: (
 295             slim_example_decoder.Tensor('image/class/confidence')),
 296         # Object boxes and classes.
 297         fields.InputDataFields.groundtruth_boxes: (
 298             slim_example_decoder.BoundingBox(['ymin', 'xmin', 'ymax', 'xmax'],
 299                                              'image/object/bbox/')),
 300         fields.InputDataFields.groundtruth_area:
 301             slim_example_decoder.Tensor('image/object/area'),
 302         fields.InputDataFields.groundtruth_is_crowd: (
 303             slim_example_decoder.Tensor('image/object/is_crowd')),
 304         fields.InputDataFields.groundtruth_difficult: (
 305             slim_example_decoder.Tensor('image/object/difficult')),
 306         fields.InputDataFields.groundtruth_group_of: (
 307             slim_example_decoder.Tensor('image/object/group_of')),
 308         fields.InputDataFields.groundtruth_weights: (
 309             slim_example_decoder.Tensor('image/object/weight')),
 310 
 311     }
 312     if load_multiclass_scores:
 313       self.keys_to_features[
 314           'image/object/class/multiclass_scores'] = tf.VarLenFeature(tf.float32)
 315       self.items_to_handlers[fields.InputDataFields.multiclass_scores] = (
 316           slim_example_decoder.Tensor('image/object/class/multiclass_scores'))
 317 
 318     if load_context_features:
 319       self.keys_to_features[
 320           'image/context_features'] = tf.VarLenFeature(tf.float32)
 321       self.items_to_handlers[fields.InputDataFields.context_features] = (
 322           slim_example_decoder.ItemHandlerCallback(
 323               ['image/context_features', 'image/context_feature_length'],
 324               self._reshape_context_features))
 325 
 326       self.keys_to_features[
 327           'image/context_feature_length'] = tf.FixedLenFeature((), tf.int64)
 328       self.items_to_handlers[fields.InputDataFields.context_feature_length] = (
 329           slim_example_decoder.Tensor('image/context_feature_length'))
 330 
 331     if num_additional_channels > 0:
 332       self.keys_to_features[
 333           'image/additional_channels/encoded'] = tf.FixedLenFeature(
 334               (num_additional_channels,), tf.string)
 335       self.items_to_handlers[
 336           fields.InputDataFields.
 337           image_additional_channels] = additional_channel_image
 338     self._num_keypoints = num_keypoints
 339     if num_keypoints > 0:
 340       self.keys_to_features['image/object/keypoint/x'] = (
 341           tf.VarLenFeature(tf.float32))
 342       self.keys_to_features['image/object/keypoint/y'] = (
 343           tf.VarLenFeature(tf.float32))
 344       self.keys_to_features['image/object/keypoint/visibility'] = (
 345           tf.VarLenFeature(tf.int64))
 346       self.items_to_handlers[fields.InputDataFields.groundtruth_keypoints] = (
 347           slim_example_decoder.ItemHandlerCallback(
 348               ['image/object/keypoint/y', 'image/object/keypoint/x'],
 349               self._reshape_keypoints))
 350       kpt_vis_field = fields.InputDataFields.groundtruth_keypoint_visibilities
 351       self.items_to_handlers[kpt_vis_field] = (
 352           slim_example_decoder.ItemHandlerCallback(
 353               ['image/object/keypoint/x', 'image/object/keypoint/visibility'],
 354               self._reshape_keypoint_visibilities))
 355     if load_instance_masks:
 356       if instance_mask_type in (input_reader_pb2.DEFAULT,
 357                                 input_reader_pb2.NUMERICAL_MASKS):
 358         self.keys_to_features['image/object/mask'] = (
 359             tf.VarLenFeature(tf.float32))
 360         self.items_to_handlers[
 361             fields.InputDataFields.groundtruth_instance_masks] = (
 362                 slim_example_decoder.ItemHandlerCallback(
 363                     ['image/object/mask', 'image/height', 'image/width'],
 364                     self._reshape_instance_masks))
 365       elif instance_mask_type == input_reader_pb2.PNG_MASKS:
 366         self.keys_to_features['image/object/mask'] = tf.VarLenFeature(tf.string)
 367         self.items_to_handlers[
 368             fields.InputDataFields.groundtruth_instance_masks] = (
 369                 slim_example_decoder.ItemHandlerCallback(
 370                     ['image/object/mask', 'image/height', 'image/width'],
 371                     self._decode_png_instance_masks))
 372       else:
 373         raise ValueError('Did not recognize the `instance_mask_type` option.')
 374     if label_map_proto_file:
 375       # If the label_map_proto is provided, try to use it in conjunction with
 376       # the class text, and fall back to a materialized ID.
 377       label_handler = _BackupHandler(
 378           _ClassTensorHandler(
 379               'image/object/class/text', label_map_proto_file,
 380               default_value=''),
 381           slim_example_decoder.Tensor('image/object/class/label'))
 382       image_label_handler = _BackupHandler(
 383           _ClassTensorHandler(
 384               fields.TfExampleFields.image_class_text,
 385               label_map_proto_file,
 386               default_value=''),
 387           slim_example_decoder.Tensor(fields.TfExampleFields.image_class_label))
 388     else:
 389       label_handler = slim_example_decoder.Tensor('image/object/class/label')
 390       image_label_handler = slim_example_decoder.Tensor(
 391           fields.TfExampleFields.image_class_label)
 392     self.items_to_handlers[
 393         fields.InputDataFields.groundtruth_classes] = label_handler
 394     self.items_to_handlers[
 395         fields.InputDataFields.groundtruth_image_classes] = image_label_handler
 396 
 397     self._expand_hierarchy_labels = expand_hierarchy_labels
 398     self._ancestors_lut = None
 399     self._descendants_lut = None
 400     if expand_hierarchy_labels:
 401       if label_map_proto_file:
 402         ancestors_lut, descendants_lut = (
 403             label_map_util.get_label_map_hierarchy_lut(label_map_proto_file,
 404                                                        True))
 405         self._ancestors_lut = tf.constant(ancestors_lut, dtype=tf.int64)
 406         self._descendants_lut = tf.constant(descendants_lut, dtype=tf.int64)
 407       else:
 408         raise ValueError('In order to expand labels, the label_map_proto_file '
 409                          'has to be provided.')
 410 
 411   def decode(self, tf_example_string_tensor):
 412     """Decodes serialized tensorflow example and returns a tensor dictionary.
 413 
 414     Args:
 415       tf_example_string_tensor: a string tensor holding a serialized tensorflow
 416         example proto.
 417 
 418     Returns:
 419       A dictionary of the following tensors.
 420       fields.InputDataFields.image - 3D uint8 tensor of shape [None, None, 3]
 421         containing image.
 422       fields.InputDataFields.original_image_spatial_shape - 1D int32 tensor of
 423         shape [2] containing shape of the image.
 424       fields.InputDataFields.source_id - string tensor containing original
 425         image id.
 426       fields.InputDataFields.key - string tensor with unique sha256 hash key.
 427       fields.InputDataFields.filename - string tensor with original dataset
 428         filename.
 429       fields.InputDataFields.groundtruth_boxes - 2D float32 tensor of shape
 430         [None, 4] containing box corners.
 431       fields.InputDataFields.groundtruth_classes - 1D int64 tensor of shape
 432         [None] containing classes for the boxes.
 433       fields.InputDataFields.groundtruth_weights - 1D float32 tensor of
 434         shape [None] indicating the weights of groundtruth boxes.
 435       fields.InputDataFields.groundtruth_area - 1D float32 tensor of shape
 436         [None] containing containing object mask area in pixel squared.
 437       fields.InputDataFields.groundtruth_is_crowd - 1D bool tensor of shape
 438         [None] indicating if the boxes enclose a crowd.
 439 
 440     Optional:
 441       fields.InputDataFields.groundtruth_image_confidences - 1D float tensor of
 442         shape [None] indicating if a class is present in the image (1.0) or
 443         a class is not present in the image (0.0).
 444       fields.InputDataFields.image_additional_channels - 3D uint8 tensor of
 445         shape [None, None, num_additional_channels]. 1st dim is height; 2nd dim
 446         is width; 3rd dim is the number of additional channels.
 447       fields.InputDataFields.groundtruth_difficult - 1D bool tensor of shape
 448         [None] indicating if the boxes represent `difficult` instances.
 449       fields.InputDataFields.groundtruth_group_of - 1D bool tensor of shape
 450         [None] indicating if the boxes represent `group_of` instances.
 451       fields.InputDataFields.groundtruth_keypoints - 3D float32 tensor of
 452         shape [None, num_keypoints, 2] containing keypoints, where the
 453         coordinates of the keypoints are ordered (y, x).
 454       fields.InputDataFields.groundtruth_keypoint_visibilities - 2D bool
 455         tensor of shape [None, num_keypoints] containing keypoint visibilites.
 456       fields.InputDataFields.groundtruth_instance_masks - 3D float32 tensor of
 457         shape [None, None, None] containing instance masks.
 458       fields.InputDataFields.groundtruth_image_classes - 1D int64 of shape
 459         [None] containing classes for the boxes.
 460       fields.InputDataFields.multiclass_scores - 1D float32 tensor of shape
 461         [None * num_classes] containing flattened multiclass scores for
 462         groundtruth boxes.
 463       fields.InputDataFields.context_features - 1D float32 tensor of shape
 464         [context_feature_length * num_context_features]
 465       fields.InputDataFields.context_feature_length - int32 tensor specifying
 466         the length of each feature in context_features
 467     """
 468     serialized_example = tf.reshape(tf_example_string_tensor, shape=[])
 469     decoder = slim_example_decoder.TFExampleDecoder(self.keys_to_features,
 470                                                     self.items_to_handlers)
 471     keys = decoder.list_items()
 472     tensors = decoder.decode(serialized_example, items=keys)
 473     tensor_dict = dict(zip(keys, tensors))
 474     is_crowd = fields.InputDataFields.groundtruth_is_crowd
 475     tensor_dict[is_crowd] = tf.cast(tensor_dict[is_crowd], dtype=tf.bool)
 476     tensor_dict[fields.InputDataFields.image].set_shape([None, None, 3])
 477     tensor_dict[fields.InputDataFields.original_image_spatial_shape] = tf.shape(
 478         tensor_dict[fields.InputDataFields.image])[:2]
 479 
 480     if fields.InputDataFields.image_additional_channels in tensor_dict:
 481       channels = tensor_dict[fields.InputDataFields.image_additional_channels]
 482       channels = tf.squeeze(channels, axis=3)
 483       channels = tf.transpose(channels, perm=[1, 2, 0])
 484       tensor_dict[fields.InputDataFields.image_additional_channels] = channels
 485 
 486     def default_groundtruth_weights():
 487       return tf.ones(
 488           [tf.shape(tensor_dict[fields.InputDataFields.groundtruth_boxes])[0]],
 489           dtype=tf.float32)
 490 
 491     tensor_dict[fields.InputDataFields.groundtruth_weights] = tf.cond(
 492         tf.greater(
 493             tf.shape(
 494                 tensor_dict[fields.InputDataFields.groundtruth_weights])[0],
 495             0), lambda: tensor_dict[fields.InputDataFields.groundtruth_weights],
 496         default_groundtruth_weights)
 497 
 498     if fields.InputDataFields.groundtruth_keypoints in tensor_dict:
 499       # Set all keypoints that are not labeled to NaN.
 500       gt_kpt_fld = fields.InputDataFields.groundtruth_keypoints
 501       gt_kpt_vis_fld = fields.InputDataFields.groundtruth_keypoint_visibilities
 502       visibilities_tiled = tf.tile(
 503           tf.expand_dims(tensor_dict[gt_kpt_vis_fld], -1),
 504           [1, 1, 2])
 505       tensor_dict[gt_kpt_fld] = tf.where(
 506           visibilities_tiled,
 507           tensor_dict[gt_kpt_fld],
 508           np.nan * tf.ones_like(tensor_dict[gt_kpt_fld]))
 509 
 510     if self._expand_hierarchy_labels:
 511       input_fields = fields.InputDataFields
 512       image_classes, image_confidences = self._expand_image_label_hierarchy(
 513           tensor_dict[input_fields.groundtruth_image_classes],
 514           tensor_dict[input_fields.groundtruth_image_confidences])
 515       tensor_dict[input_fields.groundtruth_image_classes] = image_classes
 516       tensor_dict[input_fields.groundtruth_image_confidences] = (
 517           image_confidences)
 518 
 519       box_fields = [
 520           fields.InputDataFields.groundtruth_group_of,
 521           fields.InputDataFields.groundtruth_is_crowd,
 522           fields.InputDataFields.groundtruth_difficult,
 523           fields.InputDataFields.groundtruth_area,
 524           fields.InputDataFields.groundtruth_boxes,
 525           fields.InputDataFields.groundtruth_weights,
 526       ]
 527 
 528       def expand_field(field_name):
 529         return self._expansion_box_field_labels(
 530             tensor_dict[input_fields.groundtruth_classes],
 531             tensor_dict[field_name])
 532 
 533       # pylint: disable=cell-var-from-loop
 534       for field in box_fields:
 535         if field in tensor_dict:
 536           tensor_dict[field] = tf.cond(
 537               tf.size(tensor_dict[field]) > 0, lambda: expand_field(field),
 538               lambda: tensor_dict[field])
 539       # pylint: enable=cell-var-from-loop
 540 
 541       tensor_dict[input_fields.groundtruth_classes] = (
 542           self._expansion_box_field_labels(
 543               tensor_dict[input_fields.groundtruth_classes],
 544               tensor_dict[input_fields.groundtruth_classes], True))
 545 
 546     if fields.InputDataFields.groundtruth_group_of in tensor_dict:
 547       group_of = fields.InputDataFields.groundtruth_group_of
 548       tensor_dict[group_of] = tf.cast(tensor_dict[group_of], dtype=tf.bool)
 549 
 550     return tensor_dict

 552   def _reshape_keypoints(self, keys_to_tensors):
 553     """Reshape keypoints.
 554 
 555     The keypoints are reshaped to [num_instances, num_keypoints, 2].
 556 
 557     Args:
 558       keys_to_tensors: a dictionary from keys to tensors. Expected keys are:
 559         'image/object/keypoint/x'
 560         'image/object/keypoint/y'
 561 
 562     Returns:
 563       A 3-D float tensor of shape [num_instances, num_keypoints, 2] with values
 564         in [0, 1].
 565     """
 566     y = keys_to_tensors['image/object/keypoint/y']
 567     if isinstance(y, tf.SparseTensor):
 568       y = tf.sparse_tensor_to_dense(y)
 569     y = tf.expand_dims(y, 1)
 570     x = keys_to_tensors['image/object/keypoint/x']
 571     if isinstance(x, tf.SparseTensor):
 572       x = tf.sparse_tensor_to_dense(x)
 573     x = tf.expand_dims(x, 1)
 574     keypoints = tf.concat([y, x], 1)
 575     keypoints = tf.reshape(keypoints, [-1, self._num_keypoints, 2])
 576     return keypoints

 578   def _reshape_keypoint_visibilities(self, keys_to_tensors):
 579     """Reshape keypoint visibilities.
 580 
 581     The keypoint visibilities are reshaped to [num_instances,
 582     num_keypoints].
 583 
 584     The raw keypoint visibilities are expected to conform to the
 585     MSCoco definition. See Visibility enum.
 586 
 587     The returned boolean is True for the labeled case (either
 588     Visibility.NOT_VISIBLE or Visibility.VISIBLE). These are the same categories
 589     that COCO uses to evaluate keypoint detection performance:
 590     http://cocodataset.org/#keypoints-eval
 591 
 592     If image/object/keypoint/visibility is not provided, visibilities will be
 593     set to True for finite keypoint coordinate values, and 0 if the coordinates
 594     are NaN.
 595 
 596     Args:
 597       keys_to_tensors: a dictionary from keys to tensors. Expected keys are:
 598         'image/object/keypoint/x'
 599         'image/object/keypoint/visibility'
 600 
 601     Returns:
 602       A 2-D bool tensor of shape [num_instances, num_keypoints] with values
 603         in {0, 1}. 1 if the keypoint is labeled, 0 otherwise.
 604     """
 605     x = keys_to_tensors['image/object/keypoint/x']
 606     vis = keys_to_tensors['image/object/keypoint/visibility']
 607     if isinstance(vis, tf.SparseTensor):
 608       vis = tf.sparse_tensor_to_dense(vis)
 609     if isinstance(x, tf.SparseTensor):
 610       x = tf.sparse_tensor_to_dense(x)
 611 
 612     default_vis = tf.where(
 613         tf.math.is_nan(x),
 614         Visibility.UNLABELED.value * tf.ones_like(x, dtype=tf.int64),
 615         Visibility.VISIBLE.value * tf.ones_like(x, dtype=tf.int64))
 616     # Use visibility if provided, otherwise use the default visibility.
 617     vis = tf.cond(tf.equal(tf.size(x), tf.size(vis)),
 618                   true_fn=lambda: vis,
 619                   false_fn=lambda: default_vis)
 620     vis = tf.math.logical_or(
 621         tf.math.equal(vis, Visibility.NOT_VISIBLE.value),
 622         tf.math.equal(vis, Visibility.VISIBLE.value))
 623     vis = tf.reshape(vis, [-1, self._num_keypoints])
 624     return vis

 669   def _decode_png_instance_masks(self, keys_to_tensors):
 670     """Decode PNG instance segmentation masks and stack into dense tensor.
 671 
 672     The instance segmentation masks are reshaped to [num_instances, height,
 673     width].
 674 
 675     Args:
 676       keys_to_tensors: a dictionary from keys to tensors.
 677 
 678     Returns:
 679       A 3-D float tensor of shape [num_instances, height, width] with values
 680         in {0, 1}.
 681     """
 682 
 683     def decode_png_mask(image_buffer):
 684       image = tf.squeeze(
 685           tf.image.decode_image(image_buffer, channels=1), axis=2)
 686       image.set_shape([None, None])
 687       image = tf.cast(tf.greater(image, 0), dtype=tf.float32)
 688       return image
 689 
 690     png_masks = keys_to_tensors['image/object/mask']
 691     height = keys_to_tensors['image/height']

 695     return tf.cond(
 696         tf.greater(tf.size(png_masks), 0),
 697         lambda: tf.map_fn(decode_png_mask, png_masks, dtype=tf.float32),
 698         lambda: tf.zeros(tf.cast(tf.stack([0, height, width]), dtype=tf.int32)))

 700   def _expand_image_label_hierarchy(self, image_classes, image_confidences):
 701     """Expand image level labels according to the hierarchy.
 702 
 703     Args:
 704       image_classes: Int64 tensor with the image level class ids for a sample.
 705       image_confidences: Float tensor signaling whether a class id is present in
 706         the image (1.0) or not present (0.0).
 707 
 708     Returns:
 709       new_image_classes: Int64 tensor equal to expanding image_classes.
 710       new_image_confidences: Float tensor equal to expanding image_confidences.
 711     """
 712 
 713     def expand_labels(relation_tensor, confidence_value):
 714       """Expand to ancestors or descendants depending on arguments."""
 715       mask = tf.equal(image_confidences, confidence_value)
 716       target_image_classes = tf.boolean_mask(image_classes, mask)
 717       expanded_indices = tf.reduce_any((tf.gather(
 718           relation_tensor, target_image_classes - _LABEL_OFFSET, axis=0) > 0),
 719                                        axis=0)
 720       expanded_indices = tf.where(expanded_indices)[:, 0] + _LABEL_OFFSET
 721       new_groundtruth_image_classes = (
 722           tf.concat([
 723               tf.boolean_mask(image_classes, tf.logical_not(mask)),
 724               expanded_indices,
 725           ],
 726                     axis=0))
 727       new_groundtruth_image_confidences = (
 728           tf.concat([
 729               tf.boolean_mask(image_confidences, tf.logical_not(mask)),
 730               tf.ones([tf.shape(expanded_indices)[0]],
 731                       dtype=image_confidences.dtype) * confidence_value,
 732           ],
 733                     axis=0))
 734       return new_groundtruth_image_classes, new_groundtruth_image_confidences
 735 
 736     image_classes, image_confidences = expand_labels(self._ancestors_lut, 1.0)
 737     new_image_classes, new_image_confidences = expand_labels(
 738         self._descendants_lut, 0.0)
 739     return new_image_classes, new_image_confidences
 740 
 741   def _expansion_box_field_labels(self,
 742                                   object_classes,
 743                                   object_field,
 744                                   copy_class_id=False):
 745     """Expand the labels of a specific object field according to the hierarchy.
 746 
 747     Args:
 748       object_classes: Int64 tensor with the class id for each element in
 749         object_field.
 750       object_field: Tensor to be expanded.
 751       copy_class_id: Boolean to choose whether to use class id values in the
 752         output tensor instead of replicating the original values.
 753 
 754     Returns:
 755       A tensor with the result of expanding object_field.
 756     """
 757     expanded_indices = tf.gather(
 758         self._ancestors_lut, object_classes - _LABEL_OFFSET, axis=0)
 759     if copy_class_id:
 760       new_object_field = tf.where(expanded_indices > 0)[:, 1] + _LABEL_OFFSET
 761     else:
 762       new_object_field = tf.repeat(
 763           object_field, tf.reduce_sum(expanded_indices, axis=1), axis=0)
 764     return new_object_field
