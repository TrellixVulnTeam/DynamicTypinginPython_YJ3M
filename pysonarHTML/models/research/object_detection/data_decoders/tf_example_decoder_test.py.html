  31   def _create_encoded_and_decoded_data(self, data, encoding_type):
  32     if encoding_type == 'jpeg':
  33       encode_fn = tf.image.encode_jpeg
  34       decode_fn = tf.image.decode_jpeg
  35     elif encoding_type == 'png':
  36       encode_fn = tf.image.encode_png
  37       decode_fn = tf.image.decode_png
  38     else:
  39       raise ValueError('Invalid encoding type.')
  40 
  41     def prepare_data_fn():
  42       encoded_data = encode_fn(data)
  43       decoded_data = decode_fn(encoded_data)
  44       return encoded_data, decoded_data
  45 
  46     return self.execute_cpu(prepare_data_fn, [])

  48   def testDecodeAdditionalChannels(self):
  49     image = np.random.randint(256, size=(4, 5, 3)).astype(np.uint8)
  50     encoded_jpeg, _ = self._create_encoded_and_decoded_data(image, 'jpeg')
  51 
  52     additional_channel = np.random.randint(256, size=(4, 5, 1)).astype(np.uint8)
  53     (encoded_additional_channel,
  54      decoded_additional_channel) = self._create_encoded_and_decoded_data(
  55          additional_channel, 'jpeg')
  56 
  57     def graph_fn():
  58       example = tf.train.Example(
  59           features=tf.train.Features(
  60               feature={
  61                   'image/encoded':
  62                       dataset_util.bytes_feature(encoded_jpeg),
  63                   'image/additional_channels/encoded':
  64                       dataset_util.bytes_list_feature(
  65                           [encoded_additional_channel] * 2),
  66                   'image/format':
  67                       dataset_util.bytes_feature(six.b('jpeg')),
  68                   'image/source_id':
  69                       dataset_util.bytes_feature(six.b('image_id')),
  70               })).SerializeToString()
  71 
  72       example_decoder = tf_example_decoder.TfExampleDecoder(
  73           num_additional_channels=2)
  74       return example_decoder.decode(tf.convert_to_tensor(example))
  75 
  76     tensor_dict = self.execute_cpu(graph_fn, [])
  77     self.assertAllEqual(
  78         np.concatenate([decoded_additional_channel] * 2, axis=2),
  79         tensor_dict[fields.InputDataFields.image_additional_channels])

 186       example = tf.train.Example(

 188               feature={
 189                   'image/encoded':

 193                   'image/object/mask':
 194                       dataset_util.bytes_list_feature(encoded_masks)
 195               })).SerializeToString()
 196 
 197       example_decoder = tf_example_decoder.TfExampleDecoder(
 198           load_instance_masks=True,
 199           instance_mask_type=input_reader_pb2.PNG_MASKS)
 200       return example_decoder.decode(tf.convert_to_tensor(example))
 201 
 202     tensor_dict = self.execute_cpu(graph_fn, [])
