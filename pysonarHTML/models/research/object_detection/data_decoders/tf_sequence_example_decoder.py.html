  38 class _ClassTensorHandler(slim_example_decoder.Tensor):
  39   """An ItemHandler to fetch class ids from class text."""
  40 
  41   def __init__(self,
  42                tensor_key,
  43                label_map_proto_file,
  44                shape_keys=None,
  45                shape=None,
  46                default_value=''):
  47     """Initializes the LookupTensor handler.
  48 
  49     Simply calls a vocabulary (most often, a label mapping) lookup.
  50 
  51     Args:
  52       tensor_key: the name of the `TFExample` feature to read the tensor from.
  53       label_map_proto_file: File path to a text format LabelMapProto message
  54         mapping class text to id.
  55       shape_keys: Optional name or list of names of the TF-Example feature in
  56         which the tensor shape is stored. If a list, then each corresponds to
  57         one dimension of the shape.
  58       shape: Optional output shape of the `Tensor`. If provided, the `Tensor` is
  59         reshaped accordingly.
  60       default_value: The value used when the `tensor_key` is not found in a
  61         particular `TFExample`.
  62 
  63     Raises:
  64       ValueError: if both `shape_keys` and `shape` are specified.
  65     """
  66     name_to_id = label_map_util.get_label_map_dict(
  67         label_map_proto_file, use_display_name=False)
  68     # We use a default_value of -1, but we expect all labels to be contained
  69     # in the label map.
  70     try:
  71       # Dynamically try to load the tf v2 lookup, falling back to contrib
  72       lookup = tf.compat.v2.lookup
  73       hash_table_class = tf.compat.v2.lookup.StaticHashTable
  74     except AttributeError:
  75       lookup = contrib_lookup
  76       hash_table_class = contrib_lookup.HashTable
  77     name_to_id_table = hash_table_class(
  78         initializer=lookup.KeyValueTensorInitializer(
  79             keys=tf.constant(list(name_to_id.keys())),
  80             values=tf.constant(list(name_to_id.values()), dtype=tf.int64)),
  81         default_value=-1)
  82 
  83     self._name_to_id_table = name_to_id_table
  84     super(_ClassTensorHandler, self).__init__(tensor_key, shape_keys, shape,
  85                                               default_value)
  86 
  87   def tensors_to_item(self, keys_to_tensors):
  88     unmapped_tensor = super(_ClassTensorHandler,
  89                             self).tensors_to_item(keys_to_tensors)
  90     return self._name_to_id_table.lookup(unmapped_tensor)

  93 class TfSequenceExampleDecoder(data_decoder.DataDecoder):
  94   """Tensorflow Sequence Example proto decoder for Object Detection.
  95 
  96   Sequence examples contain sequences of images which share common
  97   features. The structure of TfSequenceExamples can be seen in
  98   dataset_tools/seq_example_util.py
  99 
 100   For the TFODAPI, the following fields are required:
 101     Shared features:
 102       'image/format'
 103       'image/height'
 104       'image/width'
 105 
 106     Features with an entry for each image, where bounding box features can
 107     be empty lists if the image does not contain any objects:
 108       'image/encoded'
 109       'image/source_id'
 110       'region/bbox/xmin'
 111       'region/bbox/xmax'
 112       'region/bbox/ymin'
 113       'region/bbox/ymax'
 114       'region/label/string'
 115 
 116   Optionally, the sequence example can include context_features for use in
 117   Context R-CNN (see https://arxiv.org/abs/1912.03538):
 118     'image/context_features'
 119     'image/context_feature_length'
 120   """
 121 
 122   def __init__(self,
 123                label_map_proto_file,
 124                load_context_features=False,
 125                use_display_name=False,
 126                fully_annotated=False):
 127     """Constructs `TfSequenceExampleDecoder` object.
 128 
 129     Args:
 130       label_map_proto_file: a file path to a
 131         object_detection.protos.StringIntLabelMap proto. The
 132         label map will be used to map IDs of 'region/label/string'.
 133         It is assumed that 'region/label/string' will be in the data.
 134       load_context_features: Whether to load information from context_features,
 135         to provide additional context to a detection model for training and/or
 136         inference
 137       use_display_name: whether or not to use the `display_name` for label
 138         mapping (instead of `name`).  Only used if label_map_proto_file is
 139         provided.
 140       fully_annotated: If True, will assume that every frame (whether it has
 141         boxes or not), has been fully annotated. If False, a
 142         'region/is_annotated' field must be provided in the dataset which
 143         indicates which frames have annotations. Default False.
 144     """
 145     # Specifies how the tf.SequenceExamples are decoded.
 146     self._context_keys_to_features = {
 147         'image/format': tf.FixedLenFeature((), tf.string, default_value='jpeg'),
 148         'image/height': tf.FixedLenFeature((), tf.int64),
 149         'image/width': tf.FixedLenFeature((), tf.int64),
 150     }
 151     self._sequence_keys_to_feature_lists = {
 152         'image/encoded': tf.FixedLenSequenceFeature([], dtype=tf.string),
 153         'image/source_id': tf.FixedLenSequenceFeature([], dtype=tf.string),
 154         'region/bbox/xmin': tf.VarLenFeature(dtype=tf.float32),
 155         'region/bbox/xmax': tf.VarLenFeature(dtype=tf.float32),
 156         'region/bbox/ymin': tf.VarLenFeature(dtype=tf.float32),
 157         'region/bbox/ymax': tf.VarLenFeature(dtype=tf.float32),
 158         'region/label/string': tf.VarLenFeature(dtype=tf.string),
 159         'region/label/confidence': tf.VarLenFeature(dtype=tf.float32),
 160     }
 161 
 162     self._items_to_handlers = {
 163         # Context.
 164         fields.InputDataFields.image_height:
 165             slim_example_decoder.Tensor('image/height'),
 166         fields.InputDataFields.image_width:
 167             slim_example_decoder.Tensor('image/width'),
 168 
 169         # Sequence.
 170         fields.InputDataFields.num_groundtruth_boxes:
 171             slim_example_decoder.NumBoxesSequence('region/bbox/xmin'),
 172         fields.InputDataFields.groundtruth_boxes:
 173             slim_example_decoder.BoundingBoxSequence(
 174                 prefix='region/bbox/', default_value=0.0),
 175         fields.InputDataFields.groundtruth_weights:
 176             slim_example_decoder.Tensor('region/label/confidence'),
 177     }
 178 
 179     # If the dataset is sparsely annotated, parse sequence features which
 180     # indicate which frames have been labeled.
 181     if not fully_annotated:
 182       self._sequence_keys_to_feature_lists['region/is_annotated'] = (
 183           tf.FixedLenSequenceFeature([], dtype=tf.int64))
 184       self._items_to_handlers[fields.InputDataFields.is_annotated] = (
 185           slim_example_decoder.Tensor('region/is_annotated'))
 186 
 187     self._items_to_handlers[fields.InputDataFields.image] = (
 188         slim_example_decoder.Tensor('image/encoded'))
 189     self._items_to_handlers[fields.InputDataFields.source_id] = (
 190         slim_example_decoder.Tensor('image/source_id'))
 191 
 192     label_handler = _ClassTensorHandler(
 193         'region/label/string', label_map_proto_file, default_value='')
 194 
 195     self._items_to_handlers[
 196         fields.InputDataFields.groundtruth_classes] = label_handler
 197 
 198     if load_context_features:
 199       self._context_keys_to_features['image/context_features'] = (
 200           tf.VarLenFeature(dtype=tf.float32))
 201       self._items_to_handlers[fields.InputDataFields.context_features] = (
 202           slim_example_decoder.ItemHandlerCallback(
 203               ['image/context_features', 'image/context_feature_length'],
 204               self._reshape_context_features))
 205 
 206       self._context_keys_to_features['image/context_feature_length'] = (
 207           tf.FixedLenFeature((), tf.int64))
 208       self._items_to_handlers[fields.InputDataFields.context_feature_length] = (
 209           slim_example_decoder.Tensor('image/context_feature_length'))
 210     self._fully_annotated = fully_annotated
 211 
 212   def decode(self, tf_seq_example_string_tensor):
 213     """Decodes serialized `tf.SequenceExample`s and returns a tensor dictionary.
 214 
 215     Args:
 216       tf_seq_example_string_tensor: a string tensor holding a serialized
 217         `tf.SequenceExample`.
 218 
 219     Returns:
 220       A list of dictionaries with (at least) the following tensors:
 221       fields.InputDataFields.source_id: a [num_frames] string tensor with a
 222         unique ID for each frame.
 223       fields.InputDataFields.num_groundtruth_boxes: a [num_frames] int32 tensor
 224         specifying the number of boxes in each frame.
 225       fields.InputDataFields.groundtruth_boxes: a [num_frames, num_boxes, 4]
 226         float32 tensor with bounding boxes for each frame. Note that num_boxes
 227         is the maximum boxes seen in any individual frame. Any frames with fewer
 228         boxes are padded with 0.0.
 229       fields.InputDataFields.groundtruth_classes: a [num_frames, num_boxes]
 230         int32 tensor with class indices for each box in each frame.
 231       fields.InputDataFields.groundtruth_weights: a [num_frames, num_boxes]
 232         float32 tensor with weights of the groundtruth boxes.
 233       fields.InputDataFields.is_annotated: a [num_frames] bool tensor specifying
 234         whether the image was annotated or not. If False, the corresponding
 235         entries in the groundtruth tensor will be ignored.
 236       fields.InputDataFields.context_features - 1D float32 tensor of shape
 237         [context_feature_length * num_context_features]
 238       fields.InputDataFields.context_feature_length - int32 tensor specifying
 239         the length of each feature in context_features
 240       fields.InputDataFields.image: a [num_frames] string tensor with
 241         the encoded images.
 242     """
 243     serialized_example = tf.reshape(tf_seq_example_string_tensor, shape=[])
 244     decoder = slim_example_decoder.TFSequenceExampleDecoder(
 245         self._context_keys_to_features, self._sequence_keys_to_feature_lists,
 246         self._items_to_handlers)
 247     keys = decoder.list_items()
 248     tensors = decoder.decode(serialized_example, items=keys)
 249     tensor_dict = dict(list(zip(keys, tensors)))
 250     tensor_dict[fields.InputDataFields.groundtruth_boxes].set_shape(
 251         [None, None, 4])
 252     tensor_dict[fields.InputDataFields.num_groundtruth_boxes] = tf.cast(
 253         tensor_dict[fields.InputDataFields.num_groundtruth_boxes],
 254         dtype=tf.int32)
 255     tensor_dict[fields.InputDataFields.groundtruth_classes] = tf.cast(
 256         tensor_dict[fields.InputDataFields.groundtruth_classes], dtype=tf.int32)
 257     tensor_dict[fields.InputDataFields.original_image_spatial_shape] = tf.cast(
 258         tf.stack([
 259             tensor_dict[fields.InputDataFields.image_height],
 260             tensor_dict[fields.InputDataFields.image_width]
 261         ]),
 262         dtype=tf.int32)
 263     tensor_dict.pop(fields.InputDataFields.image_height)
 264     tensor_dict.pop(fields.InputDataFields.image_width)
 265 
 266     def default_groundtruth_weights():
 267       """Produces weights of 1.0 for each valid box, and 0.0 otherwise."""
 268       num_boxes_per_frame = tensor_dict[
 269           fields.InputDataFields.num_groundtruth_boxes]
 270       max_num_boxes = tf.reduce_max(num_boxes_per_frame)
 271       num_boxes_per_frame_tiled = tf.tile(
 272           tf.expand_dims(num_boxes_per_frame, axis=-1),
 273           multiples=tf.stack([1, max_num_boxes]))
 274       range_tiled = tf.tile(
 275           tf.expand_dims(tf.range(max_num_boxes), axis=0),
 276           multiples=tf.stack([tf.shape(num_boxes_per_frame)[0], 1]))
 277       return tf.cast(
 278           tf.greater(num_boxes_per_frame_tiled, range_tiled), tf.float32)
 279 
 280     tensor_dict[fields.InputDataFields.groundtruth_weights] = tf.cond(
 281         tf.greater(
 282             tf.size(tensor_dict[fields.InputDataFields.groundtruth_weights]),
 283             0), lambda: tensor_dict[fields.InputDataFields.groundtruth_weights],
 284         default_groundtruth_weights)
 285 
 286     if self._fully_annotated:
 287       tensor_dict[fields.InputDataFields.is_annotated] = tf.ones_like(
 288           tensor_dict[fields.InputDataFields.num_groundtruth_boxes],
 289           dtype=tf.bool)
 290     else:
 291       tensor_dict[fields.InputDataFields.is_annotated] = tf.cast(
 292           tensor_dict[fields.InputDataFields.is_annotated], dtype=tf.bool)
 293 
 294     return tensor_dict
 295 
 296   def _reshape_context_features(self, keys_to_tensors):
 297     """Reshape context features.
 298 
 299     The instance context_features are reshaped to
 300       [num_context_features, context_feature_length]
 301 
 302     Args:
 303       keys_to_tensors: a dictionary from keys to tensors.
 304 
 305     Returns:
 306       A 2-D float tensor of shape [num_context_features, context_feature_length]
 307     """
 308     context_feature_length = keys_to_tensors['image/context_feature_length']
 309     to_shape = tf.cast(tf.stack([-1, context_feature_length]), tf.int32)
 310     context_features = keys_to_tensors['image/context_features']
 311     if isinstance(context_features, tf.SparseTensor):
 312       context_features = tf.sparse_tensor_to_dense(context_features)
 313     context_features = tf.reshape(context_features, to_shape)
 314     return context_features
