 119   def _import_graph_and_run_inference(self, tflite_graph_file, num_channels=3):
 120     """Imports a tflite graph, runs single inference and returns outputs."""
 121     graph = tf.Graph()
 122     with graph.as_default():
 123       graph_def = tf.GraphDef()
 124       with tf.gfile.Open(tflite_graph_file, mode='rb') as f:
 125         graph_def.ParseFromString(f.read())
 126       tf.import_graph_def(graph_def, name='')
 127       input_tensor = graph.get_tensor_by_name('normalized_input_image_tensor:0')
 128       box_encodings = graph.get_tensor_by_name('raw_outputs/box_encodings:0')
 129       class_predictions = graph.get_tensor_by_name(
 130           'raw_outputs/class_predictions:0')
 131       with self.test_session(graph) as sess:
 132         [box_encodings_np, class_predictions_np] = sess.run(
 133             [box_encodings, class_predictions],
 134             feed_dict={input_tensor: np.random.rand(1, 10, 10, num_channels)})
 135     return box_encodings_np, class_predictions_np

 137   def _export_graph(self,
 138                     pipeline_config,
 139                     num_channels=3,
 140                     additional_output_tensors=()):
 141     """Exports a tflite graph."""
 142     output_dir = self.get_temp_dir()
 143     trained_checkpoint_prefix = os.path.join(output_dir, 'model.ckpt')
 144     tflite_graph_file = os.path.join(output_dir, 'tflite_graph.pb')
 145 
 146     quantize = pipeline_config.HasField('graph_rewriter')
 147     self._save_checkpoint_from_mock_model(
 148         trained_checkpoint_prefix,
 149         use_moving_averages=pipeline_config.eval_config.use_moving_averages,
 150         quantize=quantize,
 151         num_channels=num_channels)
 152     with mock.patch.object(
 153         model_builder, 'build', autospec=True) as mock_builder:
 154       mock_builder.return_value = FakeModel()
 155 
 156       with tf.Graph().as_default():
 157         tf.identity(
 158             tf.constant([[1, 2], [3, 4]], tf.uint8), name='UnattachedTensor')
 159         export_tflite_ssd_graph_lib.export_tflite_graph(
 160             pipeline_config=pipeline_config,
 161             trained_checkpoint_prefix=trained_checkpoint_prefix,
 162             output_dir=output_dir,
 163             add_postprocessing_op=False,
 164             max_detections=10,
 165             max_classes_per_detection=1,
 166             additional_output_tensors=additional_output_tensors)
 167     return tflite_graph_file

 330     pipeline_config.model.ssd.box_coder.faster_rcnn_box_coder.x_scale = 10.0
 331     pipeline_config.model.ssd.box_coder.faster_rcnn_box_coder.height_scale = 5.0
 332     pipeline_config.model.ssd.box_coder.faster_rcnn_box_coder.width_scale = 5.0

 334         pipeline_config)

 336     graph = tf.Graph()
 337     with graph.as_default():
 338       graph_def = tf.GraphDef()

 341       all_op_names = [node.name for node in graph_def.node]
 342       self.assertIn('TFLite_Detection_PostProcess', all_op_names)
 343       self.assertNotIn('UnattachedTensor', all_op_names)
 344       for node in graph_def.node:
 345         if node.name == 'TFLite_Detection_PostProcess':
 346           self.assertTrue(node.attr['_output_quantized'].b)
 347           self.assertTrue(
 348               node.attr['_support_output_type_float_in_quantized_op'].b)
 349           self.assertEqual(node.attr['y_scale'].f, 10.0)
 350           self.assertEqual(node.attr['x_scale'].f, 10.0)
 351           self.assertEqual(node.attr['h_scale'].f, 5.0)
 352           self.assertEqual(node.attr['w_scale'].f, 5.0)
 353           self.assertEqual(node.attr['num_classes'].i, 2)
 354           self.assertTrue(
 355               all([
 356                   t == types_pb2.DT_FLOAT
 357                   for t in node.attr['_output_types'].list.type
