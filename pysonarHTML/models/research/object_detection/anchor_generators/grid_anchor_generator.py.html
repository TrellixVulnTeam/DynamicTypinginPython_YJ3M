  16 """Generates grid anchors on the fly as used in Faster RCNN.
  17 
  18 Generates grid anchors on the fly as described in:
  19 "Faster R-CNN: Towards Real-Time Object Detection with Region Proposal Networks"
  20 Shaoqing Ren, Kaiming He, Ross Girshick, and Jian Sun.
  21 """
  22 
  23 import tensorflow.compat.v1 as tf
  24 
  25 from object_detection.core import anchor_generator
  26 from object_detection.core import box_list
  27 from object_detection.utils import ops

  30 class GridAnchorGenerator(anchor_generator.AnchorGenerator):
  31   """Generates a grid of anchors at given scales and aspect ratios."""
  32 
  33   def __init__(self,
  34                scales=(0.5, 1.0, 2.0),
  35                aspect_ratios=(0.5, 1.0, 2.0),
  36                base_anchor_size=None,
  37                anchor_stride=None,
  38                anchor_offset=None):
  39     """Constructs a GridAnchorGenerator.
  40 
  41     Args:
  42       scales: a list of (float) scales, default=(0.5, 1.0, 2.0)
  43       aspect_ratios: a list of (float) aspect ratios, default=(0.5, 1.0, 2.0)
  44       base_anchor_size: base anchor size as height, width (
  45                         (length-2 float32 list or tensor, default=[256, 256])
  46       anchor_stride: difference in centers between base anchors for adjacent
  47                      grid positions (length-2 float32 list or tensor,
  48                      default=[16, 16])
  49       anchor_offset: center of the anchor with scale and aspect ratio 1 for the
  50                      upper left element of the grid, this should be zero for
  51                      feature networks with only VALID padding and even receptive
  52                      field size, but may need additional calculation if other
  53                      padding is used (length-2 float32 list or tensor,
  54                      default=[0, 0])
  55     """
  56     # Handle argument defaults
  57     if base_anchor_size is None:
  58       base_anchor_size = [256, 256]
  59     if anchor_stride is None:
  60       anchor_stride = [16, 16]
  61     if anchor_offset is None:
  62       anchor_offset = [0, 0]
  63 
  64     self._scales = scales
  65     self._aspect_ratios = aspect_ratios
  66     self._base_anchor_size = base_anchor_size
  67     self._anchor_stride = anchor_stride
  68     self._anchor_offset = anchor_offset
  69 
  70   def name_scope(self):
  71     return 'GridAnchorGenerator'
  72 
  73   def num_anchors_per_location(self):
  74     """Returns the number of anchors per spatial location.
  75 
  76     Returns:
  77       a list of integers, one for each expected feature map to be passed to
  78       the `generate` function.
  79     """
  80     return [len(self._scales) * len(self._aspect_ratios)]
  81 
  82   def _generate(self, feature_map_shape_list):
  83     """Generates a collection of bounding boxes to be used as anchors.
  84 
  85     Args:
  86       feature_map_shape_list: list of pairs of convnet layer resolutions in the
  87         format [(height_0, width_0)].  For example, setting
  88         feature_map_shape_list=[(8, 8)] asks for anchors that correspond
  89         to an 8x8 layer.  For this anchor generator, only lists of length 1 are
  90         allowed.
  91 
  92     Returns:
  93       boxes_list: a list of BoxLists each holding anchor boxes corresponding to
  94         the input feature map shapes.
  95 
  96     Raises:
  97       ValueError: if feature_map_shape_list, box_specs_list do not have the same
  98         length.
  99       ValueError: if feature_map_shape_list does not consist of pairs of
 100         integers
 101     """
 102     if not (isinstance(feature_map_shape_list, list)
 103             and len(feature_map_shape_list) == 1):
 104       raise ValueError('feature_map_shape_list must be a list of length 1.')
 105     if not all([isinstance(list_item, tuple) and len(list_item) == 2
 106                 for list_item in feature_map_shape_list]):
 107       raise ValueError('feature_map_shape_list must be a list of pairs.')
 108 
 109     # Create constants in init_scope so they can be created in tf.functions
 110     # and accessed from outside of the function.
 111     with tf.init_scope():
 112       self._base_anchor_size = tf.cast(tf.convert_to_tensor(
 113           self._base_anchor_size), dtype=tf.float32)
 114       self._anchor_stride = tf.cast(tf.convert_to_tensor(
 115           self._anchor_stride), dtype=tf.float32)
 116       self._anchor_offset = tf.cast(tf.convert_to_tensor(
 117           self._anchor_offset), dtype=tf.float32)
 118 
 119     grid_height, grid_width = feature_map_shape_list[0]
 120     scales_grid, aspect_ratios_grid = ops.meshgrid(self._scales,
 121                                                    self._aspect_ratios)
 122     scales_grid = tf.reshape(scales_grid, [-1])
 123     aspect_ratios_grid = tf.reshape(aspect_ratios_grid, [-1])
 124     anchors = tile_anchors(grid_height,
 125                            grid_width,
 126                            scales_grid,
 127                            aspect_ratios_grid,
 128                            self._base_anchor_size,
 129                            self._anchor_stride,
 130                            self._anchor_offset)
 131 
 132     num_anchors = anchors.num_boxes_static()
 133     if num_anchors is None:
 134       num_anchors = anchors.num_boxes()
 135     anchor_indices = tf.zeros([num_anchors])
 136     anchors.add_field('feature_map_index', anchor_indices)
 137     return [anchors]

 140 def tile_anchors(grid_height,
 141                  grid_width,
 142                  scales,
 143                  aspect_ratios,
 144                  base_anchor_size,
 145                  anchor_stride,
 146                  anchor_offset):
 147   """Create a tiled set of anchors strided along a grid in image space.
 148 
 149   This op creates a set of anchor boxes by placing a "basis" collection of
 150   boxes with user-specified scales and aspect ratios centered at evenly
 151   distributed points along a grid.  The basis collection is specified via the
 152   scale and aspect_ratios arguments.  For example, setting scales=[.1, .2, .2]
 153   and aspect ratios = [2,2,1/2] means that we create three boxes: one with scale
 154   .1, aspect ratio 2, one with scale .2, aspect ratio 2, and one with scale .2
 155   and aspect ratio 1/2.  Each box is multiplied by "base_anchor_size" before
 156   placing it over its respective center.
 157 
 158   Grid points are specified via grid_height, grid_width parameters as well as
 159   the anchor_stride and anchor_offset parameters.
 160 
 161   Args:
 162     grid_height: size of the grid in the y direction (int or int scalar tensor)
 163     grid_width: size of the grid in the x direction (int or int scalar tensor)
 164     scales: a 1-d  (float) tensor representing the scale of each box in the
 165       basis set.
 166     aspect_ratios: a 1-d (float) tensor representing the aspect ratio of each
 167       box in the basis set.  The length of the scales and aspect_ratios tensors
 168       must be equal.
 169     base_anchor_size: base anchor size as [height, width]
 170       (float tensor of shape [2])
 171     anchor_stride: difference in centers between base anchors for adjacent grid
 172                    positions (float tensor of shape [2])
 173     anchor_offset: center of the anchor with scale and aspect ratio 1 for the
 174                    upper left element of the grid, this should be zero for
 175                    feature networks with only VALID padding and even receptive
 176                    field size, but may need some additional calculation if other
 177                    padding is used (float tensor of shape [2])
 178   Returns:
 179     a BoxList holding a collection of N anchor boxes
 180   """
 181   ratio_sqrts = tf.sqrt(aspect_ratios)
 182   heights = scales / ratio_sqrts * base_anchor_size[0]
 183   widths = scales * ratio_sqrts * base_anchor_size[1]
 184 
 185   # Get a grid of box centers
 186   y_centers = tf.cast(tf.range(grid_height), dtype=tf.float32)
 187   y_centers = y_centers * anchor_stride[0] + anchor_offset[0]
 188   x_centers = tf.cast(tf.range(grid_width), dtype=tf.float32)
 189   x_centers = x_centers * anchor_stride[1] + anchor_offset[1]
 190   x_centers, y_centers = ops.meshgrid(x_centers, y_centers)
 191 
 192   widths_grid, x_centers_grid = ops.meshgrid(widths, x_centers)
 193   heights_grid, y_centers_grid = ops.meshgrid(heights, y_centers)
 194   bbox_centers = tf.stack([y_centers_grid, x_centers_grid], axis=3)
 195   bbox_sizes = tf.stack([heights_grid, widths_grid], axis=3)
 196   bbox_centers = tf.reshape(bbox_centers, [-1, 2])
 197   bbox_sizes = tf.reshape(bbox_sizes, [-1, 2])
 198   bbox_corners = _center_size_bbox_to_corners_bbox(bbox_centers, bbox_sizes)
 199   return box_list.BoxList(bbox_corners)

 202 def _center_size_bbox_to_corners_bbox(centers, sizes):
 203   """Converts bbox center-size representation to corners representation.
 204 
 205   Args:
 206     centers: a tensor with shape [N, 2] representing bounding box centers
 207     sizes: a tensor with shape [N, 2] representing bounding boxes
 208 
 209   Returns:
 210     corners: tensor with shape [N, 4] representing bounding boxes in corners
 211       representation
 212   """
 213   return tf.concat([centers - .5 * sizes, centers + .5 * sizes], 1)
