  26   def test_construct_single_anchor(self):
  27     def graph_fn():
  28       min_level = 5
  29       max_level = 5
  30       anchor_scale = 4.0
  31       aspect_ratios = [1.0]
  32       scales_per_octave = 1
  33       im_height = 64
  34       im_width = 64
  35       feature_map_shape_list = [(2, 2)]
  36       anchor_generator = mg.MultiscaleGridAnchorGenerator(
  37           min_level, max_level, anchor_scale, aspect_ratios, scales_per_octave,
  38           normalize_coordinates=False)
  39       anchors_list = anchor_generator.generate(
  40           feature_map_shape_list, im_height=im_height, im_width=im_width)
  41       anchor_corners = anchors_list[0].get()
  42       return anchor_corners
  43 
  44     exp_anchor_corners = [[-48, -48, 80, 80],
  45                           [-48, -16, 80, 112],
  46                           [-16, -48, 112, 80],
  47                           [-16, -16, 112, 112]]
  48     anchor_corners_out = self.execute(graph_fn, [])
  49     self.assertAllClose(anchor_corners_out, exp_anchor_corners)

 127       scales_per_octave = 1
 128       im_height = tf.constant(64)

 137       return anchor_corners
 138 
 139     exp_anchor_corners = [[-64, -64, 64, 64],
 140                           [-64, -32, 64, 96],
 141                           [-32, -64, 96, 64],
 142                           [-32, -32, 96, 96]]

 269   def test_construct_single_anchors_on_feature_maps_with_dynamic_shape(self):
 270 
 271     def graph_fn(feature_map1_height, feature_map1_width, feature_map2_height,
 272                  feature_map2_width):
 273       min_level = 5
 274       max_level = 6
 275       anchor_scale = 4.0
 276       aspect_ratios = [1.0]
 277       scales_per_octave = 1
 278       im_height = 64
 279       im_width = 64
 280       feature_map_shape_list = [(feature_map1_height, feature_map1_width),
 281                                 (feature_map2_height, feature_map2_width)]
 282       anchor_generator = mg.MultiscaleGridAnchorGenerator(
 283           min_level, max_level, anchor_scale, aspect_ratios, scales_per_octave,
 284           normalize_coordinates=False)
 285       anchors_list = anchor_generator.generate(feature_map_shape_list,
 286                                                im_height=im_height,
 287                                                im_width=im_width)
 288       anchor_corners = [anchors.get() for anchors in anchors_list]
 289       return anchor_corners
 290 
 291     anchor_corners_out = np.concatenate(
 292         self.execute_cpu(graph_fn, [
 293             np.array(2, dtype=np.int32),
 294             np.array(2, dtype=np.int32),
 295             np.array(1, dtype=np.int32),
 296             np.array(1, dtype=np.int32)
 297         ]),
 298         axis=0)
 299     exp_anchor_corners = [[-48, -48, 80, 80],
 300                           [-48, -16, 80, 112],
 301                           [-16, -48, 112, 80],
 302                           [-16, -16, 112, 112],
 303                           [-96, -96, 160, 160]]
 304     self.assertAllClose(anchor_corners_out, exp_anchor_corners)
