  26   def test_construct_single_anchor(self):
  27     def graph_fn():
  28       anchor_strides = [(32, 32),]
  29       anchor_offsets = [(16, 16),]
  30       base_sizes = [(128.0,)]
  31       aspect_ratios = [(1.0,)]
  32       im_height = 64
  33       im_width = 64
  34       feature_map_shape_list = [(2, 2)]
  35       anchor_generator = fg.FlexibleGridAnchorGenerator(
  36           base_sizes, aspect_ratios, anchor_strides, anchor_offsets,
  37           normalize_coordinates=False)
  38       anchors_list = anchor_generator.generate(
  39           feature_map_shape_list, im_height=im_height, im_width=im_width)
  40       anchor_corners = anchors_list[0].get()
  41       return anchor_corners
  42     anchor_corners_out = self.execute(graph_fn, [])
  43     exp_anchor_corners = [[-48, -48, 80, 80],
  44                           [-48, -16, 80, 112],
  45                           [-16, -48, 112, 80],
  46                           [-16, -16, 112, 112]]
  47     self.assertAllClose(anchor_corners_out, exp_anchor_corners)

  73   def test_construct_normalized_anchors_fails_with_unit_dimensions(self):
  74     anchor_generator = fg.FlexibleGridAnchorGenerator(
  75         [(32.0,)], [(1.0,)], [(32, 32),], [(16, 16),],
  76         normalize_coordinates=True)
  77     with self.assertRaisesRegexp(ValueError, 'Normalized coordinates'):
  78       anchor_generator.generate(
  79           feature_map_shape_list=[(2, 2)], im_height=1, im_width=1)

 104   def test_num_anchors_per_location(self):
 105     anchor_strides = [(32, 32), (64, 64)]
 106     anchor_offsets = [(16, 16), (32, 32)]
 107     base_sizes = [(32.0, 64.0, 96.0, 32.0, 64.0, 96.0),
 108                   (64.0, 128.0, 172.0, 64.0, 128.0, 172.0)]
 109     aspect_ratios = [(1.0, 1.0, 1.0, 2.0, 2.0, 2.0),
 110                      (1.0, 1.0, 1.0, 2.0, 2.0, 2.0)]
 111     anchor_generator = fg.FlexibleGridAnchorGenerator(
 112         base_sizes, aspect_ratios, anchor_strides, anchor_offsets,
 113         normalize_coordinates=False)
 114     self.assertEqual(anchor_generator.num_anchors_per_location(), [6, 6])

 254   def test_construct_single_anchors_on_feature_maps_with_dynamic_shape(self):
 255 
 256     def graph_fn(feature_map1_height, feature_map1_width, feature_map2_height,
 257                  feature_map2_width):
 258       anchor_strides = [(32, 32), (64, 64)]
 259       anchor_offsets = [(16, 16), (32, 32)]
 260       base_sizes = [(128.0,), (256.0,)]
 261       aspect_ratios = [(1.0,), (1.0,)]
 262       im_height = 64
 263       im_width = 64
 264       feature_map_shape_list = [(feature_map1_height, feature_map1_width),
 265                                 (feature_map2_height, feature_map2_width)]
 266       anchor_generator = fg.FlexibleGridAnchorGenerator(
 267           base_sizes, aspect_ratios, anchor_strides, anchor_offsets,
 268           normalize_coordinates=False)
 269       anchors_list = anchor_generator.generate(feature_map_shape_list,
 270                                                im_height=im_height,
 271                                                im_width=im_width)
 272       anchor_corners = [anchors.get() for anchors in anchors_list]
 273       return anchor_corners
 274 
 275     anchor_corners_out = np.concatenate(
 276         self.execute_cpu(graph_fn, [
 277             np.array(2, dtype=np.int32),
 278             np.array(2, dtype=np.int32),
 279             np.array(1, dtype=np.int32),
 280             np.array(1, dtype=np.int32)
 281         ]),
 282         axis=0)
 283     exp_anchor_corners = [[-48, -48, 80, 80],
 284                           [-48, -16, 80, 112],
 285                           [-16, -48, 112, 80],
 286                           [-16, -16, 112, 112],
 287                           [-96, -96, 160, 160]]
 288     self.assertAllClose(anchor_corners_out, exp_anchor_corners)
