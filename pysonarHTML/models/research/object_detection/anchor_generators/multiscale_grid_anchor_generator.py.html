  15 """Generates grid anchors on the fly corresponding to multiple CNN layers.
  16 
  17 Generates grid anchors on the fly corresponding to multiple CNN layers as
  18 described in:
  19 "Focal Loss for Dense Object Detection" (https://arxiv.org/abs/1708.02002)
  20 T.-Y. Lin, P. Goyal, R. Girshick, K. He, P. Dollar
  21 """
  22 
  23 import tensorflow.compat.v1 as tf
  24 
  25 from object_detection.anchor_generators import grid_anchor_generator
  26 from object_detection.core import anchor_generator
  27 from object_detection.core import box_list_ops
  28 
  29 
  30 class MultiscaleGridAnchorGenerator(anchor_generator.AnchorGenerator):
  31   """Generate a grid of anchors for multiple CNN layers of different scale."""
  32 
  33   def __init__(self, min_level, max_level, anchor_scale, aspect_ratios,
  34                scales_per_octave, normalize_coordinates=True):
  35     """Constructs a MultiscaleGridAnchorGenerator.
  36 
  37     To construct anchors, at multiple scale resolutions, one must provide a
  38     the minimum level and maximum levels on a scale pyramid. To define the size
  39     of anchor, the anchor scale is provided to decide the size relatively to the
  40     stride of the corresponding feature map. The generator allows one pixel
  41     location on feature map maps to multiple anchors, that have different aspect
  42     ratios and intermediate scales.
  43 
  44     Args:
  45       min_level: minimum level in feature pyramid.
  46       max_level: maximum level in feature pyramid.
  47       anchor_scale: anchor scale and feature stride define the size of the base
  48         anchor on an image. For example, given a feature pyramid with strides
  49         [2^3, ..., 2^7] and anchor scale 4. The base anchor size is
  50         4 * [2^3, ..., 2^7].
  51       aspect_ratios: list or tuple of (float) aspect ratios to place on each
  52         grid point.
  53       scales_per_octave: integer number of intermediate scales per scale octave.
  54       normalize_coordinates: whether to produce anchors in normalized
  55         coordinates. (defaults to True).
  56     """
  57     self._anchor_grid_info = []
  58     self._aspect_ratios = aspect_ratios
  59     self._scales_per_octave = scales_per_octave
  60     self._normalize_coordinates = normalize_coordinates
  61 
  62     scales = [2**(float(scale) / scales_per_octave)
  63               for scale in range(scales_per_octave)]
  64     aspects = list(aspect_ratios)
  65 
  66     for level in range(min_level, max_level + 1):
  67       anchor_stride = [2**level, 2**level]
  68       base_anchor_size = [2**level * anchor_scale, 2**level * anchor_scale]
  69       self._anchor_grid_info.append({
  70           'level': level,
  71           'info': [scales, aspects, base_anchor_size, anchor_stride]
  72       })
  73 
  74   def name_scope(self):
  75     return 'MultiscaleGridAnchorGenerator'
  76 
  77   def num_anchors_per_location(self):
  78     """Returns the number of anchors per spatial location.
  79 
  80     Returns:
  81       a list of integers, one for each expected feature map to be passed to
  82       the Generate function.
  83     """
  84     return len(self._anchor_grid_info) * [
  85         len(self._aspect_ratios) * self._scales_per_octave]
  86 
  87   def _generate(self, feature_map_shape_list, im_height=1, im_width=1):
  88     """Generates a collection of bounding boxes to be used as anchors.
  89 
  90     For training, we require the input image shape to be statically defined.
  91     That is, im_height and im_width should be integers rather than tensors.
  92     For inference, im_height and im_width can be either integers (for fixed
  93     image size), or tensors (for arbitrary image size).
  94 
  95     Args:
  96       feature_map_shape_list: list of pairs of convnet layer resolutions in the
  97         format [(height_0, width_0), (height_1, width_1), ...]. For example,
  98         setting feature_map_shape_list=[(8, 8), (7, 7)] asks for anchors that
  99         correspond to an 8x8 layer followed by a 7x7 layer.
 100       im_height: the height of the image to generate the grid for. If both
 101         im_height and im_width are 1, anchors can only be generated in
 102         absolute coordinates.
 103       im_width: the width of the image to generate the grid for. If both
 104         im_height and im_width are 1, anchors can only be generated in
 105         absolute coordinates.
 106 
 107     Returns:
 108       boxes_list: a list of BoxLists each holding anchor boxes corresponding to
 109         the input feature map shapes.
 110     Raises:
 111       ValueError: if im_height and im_width are not integers.
 112       ValueError: if im_height and im_width are 1, but normalized coordinates
 113         were requested.
 114     """
 115     anchor_grid_list = []
 116     for feat_shape, grid_info in zip(feature_map_shape_list,
 117                                      self._anchor_grid_info):
 118       # TODO(rathodv) check the feature_map_shape_list is consistent with
 119       # self._anchor_grid_info
 120       level = grid_info['level']
 121       stride = 2**level
 122       scales, aspect_ratios, base_anchor_size, anchor_stride = grid_info['info']
 123       feat_h = feat_shape[0]
 124       feat_w = feat_shape[1]
 125       anchor_offset = [0, 0]
 126       if isinstance(im_height, int) and isinstance(im_width, int):
 127         if im_height % 2.0**level == 0 or im_height == 1:
 128           anchor_offset[0] = stride / 2.0
 129         if im_width % 2.0**level == 0 or im_width == 1:
 130           anchor_offset[1] = stride / 2.0
 131       if tf.is_tensor(im_height) and tf.is_tensor(im_width):
 132         anchor_offset[0] = stride / 2.0
 133         anchor_offset[1] = stride / 2.0
 134       ag = grid_anchor_generator.GridAnchorGenerator(
 135           scales,
 136           aspect_ratios,
 137           base_anchor_size=base_anchor_size,
 138           anchor_stride=anchor_stride,
 139           anchor_offset=anchor_offset)
 140       (anchor_grid,) = ag.generate(feature_map_shape_list=[(feat_h, feat_w)])
 141 
 142       if self._normalize_coordinates:
 143         if im_height == 1 or im_width == 1:
 144           raise ValueError(
 145               'Normalized coordinates were requested upon construction of the '
 146               'MultiscaleGridAnchorGenerator, but a subsequent call to '
 147               'generate did not supply dimension information.')
 148         anchor_grid = box_list_ops.to_normalized_coordinates(
 149             anchor_grid, im_height, im_width, check_range=False)
 150       anchor_grid_list.append(anchor_grid)
 151 
 152     return anchor_grid_list
