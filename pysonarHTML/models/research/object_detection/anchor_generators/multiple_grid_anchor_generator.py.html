  16 """Generates grid anchors on the fly corresponding to multiple CNN layers.
  17 
  18 Generates grid anchors on the fly corresponding to multiple CNN layers as
  19 described in:
  20 "SSD: Single Shot MultiBox Detector"
  21 Wei Liu, Dragomir Anguelov, Dumitru Erhan, Christian Szegedy, Scott Reed,
  22 Cheng-Yang Fu, Alexander C. Berg
  23 (see Section 2.2: Choosing scales and aspect ratios for default boxes)
  24 """

  26 import numpy as np
  27 
  28 import tensorflow.compat.v1 as tf

  31 from object_detection.core import anchor_generator

  35 class MultipleGridAnchorGenerator(anchor_generator.AnchorGenerator):
  36   """Generate a grid of anchors for multiple CNN layers."""
  37 
  38   def __init__(self,
  39                box_specs_list,
  40                base_anchor_size=None,
  41                anchor_strides=None,
  42                anchor_offsets=None,
  43                clip_window=None):
  44     """Constructs a MultipleGridAnchorGenerator.
  45 
  46     To construct anchors, at multiple grid resolutions, one must provide a
  47     list of feature_map_shape_list (e.g., [(8, 8), (4, 4)]), and for each grid
  48     size, a corresponding list of (scale, aspect ratio) box specifications.
  49 
  50     For example:
  51     box_specs_list = [[(.1, 1.0), (.1, 2.0)],  # for 8x8 grid
  52                       [(.2, 1.0), (.3, 1.0), (.2, 2.0)]]  # for 4x4 grid
  53 
  54     To support the fully convolutional setting, we pass grid sizes in at
  55     generation time, while scale and aspect ratios are fixed at construction
  56     time.
  57 
  58     Args:
  59       box_specs_list: list of list of (scale, aspect ratio) pairs with the
  60         outside list having the same number of entries as feature_map_shape_list
  61         (which is passed in at generation time).
  62       base_anchor_size: base anchor size as [height, width]
  63                         (length-2 float numpy or Tensor, default=[1.0, 1.0]).
  64                         The height and width values are normalized to the
  65                         minimum dimension of the input height and width, so that
  66                         when the base anchor height equals the base anchor
  67                         width, the resulting anchor is square even if the input
  68                         image is not square.
  69       anchor_strides: list of pairs of strides in pixels (in y and x directions
  70         respectively). For example, setting anchor_strides=[(25, 25), (50, 50)]
  71         means that we want the anchors corresponding to the first layer to be
  72         strided by 25 pixels and those in the second layer to be strided by 50
  73         pixels in both y and x directions. If anchor_strides=None, they are set
  74         to be the reciprocal of the corresponding feature map shapes.
  75       anchor_offsets: list of pairs of offsets in pixels (in y and x directions
  76         respectively). The offset specifies where we want the center of the
  77         (0, 0)-th anchor to lie for each layer. For example, setting
  78         anchor_offsets=[(10, 10), (20, 20)]) means that we want the
  79         (0, 0)-th anchor of the first layer to lie at (10, 10) in pixel space
  80         and likewise that we want the (0, 0)-th anchor of the second layer to
  81         lie at (25, 25) in pixel space. If anchor_offsets=None, then they are
  82         set to be half of the corresponding anchor stride.
  83       clip_window: a tensor of shape [4] specifying a window to which all
  84         anchors should be clipped. If clip_window is None, then no clipping
  85         is performed.
  86 
  87     Raises:
  88       ValueError: if box_specs_list is not a list of list of pairs
  89       ValueError: if clip_window is not either None or a tensor of shape [4]
  90     """
  91     if isinstance(box_specs_list, list) and all(
  92         [isinstance(list_item, list) for list_item in box_specs_list]):
  93       self._box_specs = box_specs_list
  94     else:
  95       raise ValueError('box_specs_list is expected to be a '
  96                        'list of lists of pairs')
  97     if base_anchor_size is None:
  98       base_anchor_size = [256, 256]
  99     self._base_anchor_size = base_anchor_size
 100     self._anchor_strides = anchor_strides
 101     self._anchor_offsets = anchor_offsets
 102     if clip_window is not None and clip_window.get_shape().as_list() != [4]:
 103       raise ValueError('clip_window must either be None or a shape [4] tensor')
 104     self._clip_window = clip_window
 105     self._scales = []
 106     self._aspect_ratios = []
 107     for box_spec in self._box_specs:
 108       if not all([isinstance(entry, tuple) and len(entry) == 2
 109                   for entry in box_spec]):
 110         raise ValueError('box_specs_list is expected to be a '
 111                          'list of lists of pairs')
 112       scales, aspect_ratios = zip(*box_spec)
 113       self._scales.append(scales)
 114       self._aspect_ratios.append(aspect_ratios)
 115 
 116     for arg, arg_name in zip([self._anchor_strides, self._anchor_offsets],
 117                              ['anchor_strides', 'anchor_offsets']):
 118       if arg and not (isinstance(arg, list) and
 119                       len(arg) == len(self._box_specs)):
 120         raise ValueError('%s must be a list with the same length '
 121                          'as self._box_specs' % arg_name)
 122       if arg and not all([
 123           isinstance(list_item, tuple) and len(list_item) == 2
 124           for list_item in arg
 125       ]):
 126         raise ValueError('%s must be a list of pairs.' % arg_name)
 127 
 128   def name_scope(self):
 129     return 'MultipleGridAnchorGenerator'
 130 
 131   def num_anchors_per_location(self):
 132     """Returns the number of anchors per spatial location.
 133 
 134     Returns:
 135       a list of integers, one for each expected feature map to be passed to
 136       the Generate function.
 137     """
 138     return [len(box_specs) for box_specs in self._box_specs]
 139 
 140   def _generate(self, feature_map_shape_list, im_height=1, im_width=1):
 141     """Generates a collection of bounding boxes to be used as anchors.
 142 
 143     The number of anchors generated for a single grid with shape MxM where we
 144     place k boxes over each grid center is k*M^2 and thus the total number of
 145     anchors is the sum over all grids. In our box_specs_list example
 146     (see the constructor docstring), we would place two boxes over each grid
 147     point on an 8x8 grid and three boxes over each grid point on a 4x4 grid and
 148     thus end up with 2*8^2 + 3*4^2 = 176 anchors in total. The layout of the
 149     output anchors follows the order of how the grid sizes and box_specs are
 150     specified (with box_spec index varying the fastest, followed by width
 151     index, then height index, then grid index).
 152 
 153     Args:
 154       feature_map_shape_list: list of pairs of convnet layer resolutions in the
 155         format [(height_0, width_0), (height_1, width_1), ...]. For example,
 156         setting feature_map_shape_list=[(8, 8), (7, 7)] asks for anchors that
 157         correspond to an 8x8 layer followed by a 7x7 layer.
 158       im_height: the height of the image to generate the grid for. If both
 159         im_height and im_width are 1, the generated anchors default to
 160         absolute coordinates, otherwise normalized coordinates are produced.
 161       im_width: the width of the image to generate the grid for. If both
 162         im_height and im_width are 1, the generated anchors default to
 163         absolute coordinates, otherwise normalized coordinates are produced.
 164 
 165     Returns:
 166       boxes_list: a list of BoxLists each holding anchor boxes corresponding to
 167         the input feature map shapes.
 168 
 169     Raises:
 170       ValueError: if feature_map_shape_list, box_specs_list do not have the same
 171         length.
 172       ValueError: if feature_map_shape_list does not consist of pairs of
 173         integers
 174     """
 175     if not (isinstance(feature_map_shape_list, list)
 176             and len(feature_map_shape_list) == len(self._box_specs)):
 177       raise ValueError('feature_map_shape_list must be a list with the same '
 178                        'length as self._box_specs')
 179     if not all([isinstance(list_item, tuple) and len(list_item) == 2
 180                 for list_item in feature_map_shape_list]):
 181       raise ValueError('feature_map_shape_list must be a list of pairs.')
 182 
 183     im_height = tf.cast(im_height, dtype=tf.float32)
 184     im_width = tf.cast(im_width, dtype=tf.float32)
 185 
 186     if not self._anchor_strides:
 187       anchor_strides = [(1.0 / tf.cast(pair[0], dtype=tf.float32),
 188                          1.0 / tf.cast(pair[1], dtype=tf.float32))
 189                         for pair in feature_map_shape_list]
 190     else:
 191       anchor_strides = [(tf.cast(stride[0], dtype=tf.float32) / im_height,
 192                          tf.cast(stride[1], dtype=tf.float32) / im_width)
 193                         for stride in self._anchor_strides]
 194     if not self._anchor_offsets:
 195       anchor_offsets = [(0.5 * stride[0], 0.5 * stride[1])
 196                         for stride in anchor_strides]
 197     else:
 198       anchor_offsets = [(tf.cast(offset[0], dtype=tf.float32) / im_height,
 199                          tf.cast(offset[1], dtype=tf.float32) / im_width)
 200                         for offset in self._anchor_offsets]
 201 
 202     for arg, arg_name in zip([anchor_strides, anchor_offsets],
 203                              ['anchor_strides', 'anchor_offsets']):
 204       if not (isinstance(arg, list) and len(arg) == len(self._box_specs)):
 205         raise ValueError('%s must be a list with the same length '
 206                          'as self._box_specs' % arg_name)
 207       if not all([isinstance(list_item, tuple) and len(list_item) == 2
 208                   for list_item in arg]):
 209         raise ValueError('%s must be a list of pairs.' % arg_name)
 210 
 211     anchor_grid_list = []
 212     min_im_shape = tf.minimum(im_height, im_width)
 213     scale_height = min_im_shape / im_height
 214     scale_width = min_im_shape / im_width
 215     if not tf.is_tensor(self._base_anchor_size):
 216       base_anchor_size = [
 217           scale_height * tf.constant(self._base_anchor_size[0],
 218                                      dtype=tf.float32),
 219           scale_width * tf.constant(self._base_anchor_size[1],
 220                                     dtype=tf.float32)
 221       ]
 222     else:
 223       base_anchor_size = [
 224           scale_height * self._base_anchor_size[0],
 225           scale_width * self._base_anchor_size[1]
 226       ]
 227     for feature_map_index, (grid_size, scales, aspect_ratios, stride,
 228                             offset) in enumerate(
 229                                 zip(feature_map_shape_list, self._scales,
 230                                     self._aspect_ratios, anchor_strides,
 231                                     anchor_offsets)):
 232       tiled_anchors = grid_anchor_generator.tile_anchors(
 233           grid_height=grid_size[0],
 234           grid_width=grid_size[1],
 235           scales=scales,
 236           aspect_ratios=aspect_ratios,
 237           base_anchor_size=base_anchor_size,
 238           anchor_stride=stride,
 239           anchor_offset=offset)
 240       if self._clip_window is not None:
 241         tiled_anchors = box_list_ops.clip_to_window(
 242             tiled_anchors, self._clip_window, filter_nonoverlapping=False)
 243       num_anchors_in_layer = tiled_anchors.num_boxes_static()
 244       if num_anchors_in_layer is None:
 245         num_anchors_in_layer = tiled_anchors.num_boxes()
 246       anchor_indices = feature_map_index * tf.ones([num_anchors_in_layer])
 247       tiled_anchors.add_field('feature_map_index', anchor_indices)
 248       anchor_grid_list.append(tiled_anchors)
 249 
 250     return anchor_grid_list

 253 def create_ssd_anchors(num_layers=6,
 254                        min_scale=0.2,
 255                        max_scale=0.95,
 256                        scales=None,
 257                        aspect_ratios=(1.0, 2.0, 3.0, 1.0 / 2, 1.0 / 3),
 258                        interpolated_scale_aspect_ratio=1.0,
 259                        base_anchor_size=None,
 260                        anchor_strides=None,
 261                        anchor_offsets=None,
 262                        reduce_boxes_in_lowest_layer=True):
 263   """Creates MultipleGridAnchorGenerator for SSD anchors.
 264 
 265   This function instantiates a MultipleGridAnchorGenerator that reproduces
 266   ``default box`` construction proposed by Liu et al in the SSD paper.
 267   See Section 2.2 for details. Grid sizes are assumed to be passed in
 268   at generation time from finest resolution to coarsest resolution --- this is
 269   used to (linearly) interpolate scales of anchor boxes corresponding to the
 270   intermediate grid sizes.
 271 
 272   Anchors that are returned by calling the `generate` method on the returned
 273   MultipleGridAnchorGenerator object are always in normalized coordinates
 274   and clipped to the unit square: (i.e. all coordinates lie in [0, 1]x[0, 1]).
 275 
 276   Args:
 277     num_layers: integer number of grid layers to create anchors for (actual
 278       grid sizes passed in at generation time)
 279     min_scale: scale of anchors corresponding to finest resolution (float)
 280     max_scale: scale of anchors corresponding to coarsest resolution (float)
 281     scales: As list of anchor scales to use. When not None and not empty,
 282       min_scale and max_scale are not used.
 283     aspect_ratios: list or tuple of (float) aspect ratios to place on each
 284       grid point.
 285     interpolated_scale_aspect_ratio: An additional anchor is added with this
 286       aspect ratio and a scale interpolated between the scale for a layer
 287       and the scale for the next layer (1.0 for the last layer).
 288       This anchor is not included if this value is 0.
 289     base_anchor_size: base anchor size as [height, width].
 290       The height and width values are normalized to the minimum dimension of the
 291       input height and width, so that when the base anchor height equals the
 292       base anchor width, the resulting anchor is square even if the input image
 293       is not square.
 294     anchor_strides: list of pairs of strides in pixels (in y and x directions
 295       respectively). For example, setting anchor_strides=[(25, 25), (50, 50)]
 296       means that we want the anchors corresponding to the first layer to be
 297       strided by 25 pixels and those in the second layer to be strided by 50
 298       pixels in both y and x directions. If anchor_strides=None, they are set to
 299       be the reciprocal of the corresponding feature map shapes.
 300     anchor_offsets: list of pairs of offsets in pixels (in y and x directions
 301       respectively). The offset specifies where we want the center of the
 302       (0, 0)-th anchor to lie for each layer. For example, setting
 303       anchor_offsets=[(10, 10), (20, 20)]) means that we want the
 304       (0, 0)-th anchor of the first layer to lie at (10, 10) in pixel space
 305       and likewise that we want the (0, 0)-th anchor of the second layer to lie
 306       at (25, 25) in pixel space. If anchor_offsets=None, then they are set to
 307       be half of the corresponding anchor stride.
 308     reduce_boxes_in_lowest_layer: a boolean to indicate whether the fixed 3
 309       boxes per location is used in the lowest layer.
 310 
 311   Returns:
 312     a MultipleGridAnchorGenerator
 313   """
 314   if base_anchor_size is None:
 315     base_anchor_size = [1.0, 1.0]
 316   box_specs_list = []
 317   if scales is None or not scales:
 318     scales = [min_scale + (max_scale - min_scale) * i / (num_layers - 1)
 319               for i in range(num_layers)] + [1.0]
 320   else:
 321     # Add 1.0 to the end, which will only be used in scale_next below and used
 322     # for computing an interpolated scale for the largest scale in the list.
 323     scales += [1.0]
 324 
 325   for layer, scale, scale_next in zip(
 326       range(num_layers), scales[:-1], scales[1:]):
 327     layer_box_specs = []
 328     if layer == 0 and reduce_boxes_in_lowest_layer:
 329       layer_box_specs = [(0.1, 1.0), (scale, 2.0), (scale, 0.5)]
 330     else:
 331       for aspect_ratio in aspect_ratios:
 332         layer_box_specs.append((scale, aspect_ratio))
 333       # Add one more anchor, with a scale between the current scale, and the
 334       # scale for the next layer, with a specified aspect ratio (1.0 by
 335       # default).
 336       if interpolated_scale_aspect_ratio > 0.0:
 337         layer_box_specs.append((np.sqrt(scale*scale_next),
 338                                 interpolated_scale_aspect_ratio))
 339     box_specs_list.append(layer_box_specs)
 340 
 341   return MultipleGridAnchorGenerator(box_specs_list, base_anchor_size,
 342                                      anchor_strides, anchor_offsets)
