  62 def get_class_name_from_filename(file_name):
  63   """Gets the class name from a file.
  64 
  65   Args:
  66     file_name: The file name to get the class name from.
  67                ie. "american_pit_bull_terrier_105.jpg"
  68 
  69   Returns:
  70     A string of the class name.
  71   """
  72   match = re.match(r'([A-Za-z_]+)(_[0-9]+\.jpg)', file_name, re.I)
  73   return match.groups()[0]

  76 def dict_to_tf_example(data,
  77                        mask_path,
  78                        label_map_dict,
  79                        image_subdirectory,
  80                        ignore_difficult_instances=False,
  81                        faces_only=True,
  82                        mask_type='png'):
  83   """Convert XML derived dict to tf.Example proto.
  84 
  85   Notice that this function normalizes the bounding box coordinates provided
  86   by the raw data.
  87 
  88   Args:
  89     data: dict holding PASCAL XML fields for a single image (obtained by
  90       running dataset_util.recursive_parse_xml_to_dict)
  91     mask_path: String path to PNG encoded mask.
  92     label_map_dict: A map from string label names to integers ids.
  93     image_subdirectory: String specifying subdirectory within the
  94       Pascal dataset directory holding the actual image data.
  95     ignore_difficult_instances: Whether to skip difficult instances in the
  96       dataset  (default: False).
  97     faces_only: If True, generates bounding boxes for pet faces.  Otherwise
  98       generates bounding boxes (as well as segmentations for full pet bodies).
  99     mask_type: 'numerical' or 'png'. 'png' is recommended because it leads to
 100       smaller file sizes.
 101 
 102   Returns:
 103     example: The converted tf.Example.
 104 
 105   Raises:
 106     ValueError: if the image pointed to by data['filename'] is not a valid JPEG
 107   """
 108   img_path = os.path.join(image_subdirectory, data['filename'])
 109   with tf.gfile.GFile(img_path, 'rb') as fid:
 110     encoded_jpg = fid.read()
 111   encoded_jpg_io = io.BytesIO(encoded_jpg)
 112   image = PIL.Image.open(encoded_jpg_io)
 113   if image.format != 'JPEG':
 114     raise ValueError('Image format not JPEG')
 115   key = hashlib.sha256(encoded_jpg).hexdigest()
 116 
 117   with tf.gfile.GFile(mask_path, 'rb') as fid:
 118     encoded_mask_png = fid.read()
 119   encoded_png_io = io.BytesIO(encoded_mask_png)
 120   mask = PIL.Image.open(encoded_png_io)
 121   if mask.format != 'PNG':
 122     raise ValueError('Mask format not PNG')
 123 
 124   mask_np = np.asarray(mask)
 125   nonbackground_indices_x = np.any(mask_np != 2, axis=0)
 126   nonbackground_indices_y = np.any(mask_np != 2, axis=1)
 127   nonzero_x_indices = np.where(nonbackground_indices_x)
 128   nonzero_y_indices = np.where(nonbackground_indices_y)
 129 
 130   width = int(data['size']['width'])
 131   height = int(data['size']['height'])
 132 
 133   xmins = []
 134   ymins = []
 135   xmaxs = []
 136   ymaxs = []
 137   classes = []
 138   classes_text = []
 139   truncated = []
 140   poses = []
 141   difficult_obj = []
 142   masks = []
 143   if 'object' in data:
 144     for obj in data['object']:
 145       difficult = bool(int(obj['difficult']))
 146       if ignore_difficult_instances and difficult:
 147         continue
 148       difficult_obj.append(int(difficult))
 149 
 150       if faces_only:
 151         xmin = float(obj['bndbox']['xmin'])
 152         xmax = float(obj['bndbox']['xmax'])
 153         ymin = float(obj['bndbox']['ymin'])
 154         ymax = float(obj['bndbox']['ymax'])
 155       else:
 156         xmin = float(np.min(nonzero_x_indices))
 157         xmax = float(np.max(nonzero_x_indices))
 158         ymin = float(np.min(nonzero_y_indices))
 159         ymax = float(np.max(nonzero_y_indices))
 160 
 161       xmins.append(xmin / width)
 162       ymins.append(ymin / height)
 163       xmaxs.append(xmax / width)
 164       ymaxs.append(ymax / height)
 165       class_name = get_class_name_from_filename(data['filename'])
 166       classes_text.append(class_name.encode('utf8'))
 167       classes.append(label_map_dict[class_name])
 168       truncated.append(int(obj['truncated']))
 169       poses.append(obj['pose'].encode('utf8'))
 170       if not faces_only:
 171         mask_remapped = (mask_np != 2).astype(np.uint8)
 172         masks.append(mask_remapped)
 173 
 174   feature_dict = {
 175       'image/height': dataset_util.int64_feature(height),
 176       'image/width': dataset_util.int64_feature(width),
 177       'image/filename': dataset_util.bytes_feature(
 178           data['filename'].encode('utf8')),
 179       'image/source_id': dataset_util.bytes_feature(
 180           data['filename'].encode('utf8')),
 181       'image/key/sha256': dataset_util.bytes_feature(key.encode('utf8')),
 182       'image/encoded': dataset_util.bytes_feature(encoded_jpg),
 183       'image/format': dataset_util.bytes_feature('jpeg'.encode('utf8')),
 184       'image/object/bbox/xmin': dataset_util.float_list_feature(xmins),
 185       'image/object/bbox/xmax': dataset_util.float_list_feature(xmaxs),
 186       'image/object/bbox/ymin': dataset_util.float_list_feature(ymins),
 187       'image/object/bbox/ymax': dataset_util.float_list_feature(ymaxs),
 188       'image/object/class/text': dataset_util.bytes_list_feature(classes_text),
 189       'image/object/class/label': dataset_util.int64_list_feature(classes),
 190       'image/object/difficult': dataset_util.int64_list_feature(difficult_obj),
 191       'image/object/truncated': dataset_util.int64_list_feature(truncated),
 192       'image/object/view': dataset_util.bytes_list_feature(poses),
 193   }
 194   if not faces_only:
 195     if mask_type == 'numerical':
 196       mask_stack = np.stack(masks).astype(np.float32)
 197       masks_flattened = np.reshape(mask_stack, [-1])
 198       feature_dict['image/object/mask'] = (
 199           dataset_util.float_list_feature(masks_flattened.tolist()))
 200     elif mask_type == 'png':
 201       encoded_mask_png_list = []
 202       for mask in masks:
 203         img = PIL.Image.fromarray(mask)
 204         output = io.BytesIO()
 205         img.save(output, format='PNG')
 206         encoded_mask_png_list.append(output.getvalue())
 207       feature_dict['image/object/mask'] = (
 208           dataset_util.bytes_list_feature(encoded_mask_png_list))
 209 
 210   example = tf.train.Example(features=tf.train.Features(feature=feature_dict))
 211   return example

 214 def create_tf_record(output_filename,
 215                      num_shards,
 216                      label_map_dict,
 217                      annotations_dir,
 218                      image_dir,
 219                      examples,
 220                      faces_only=True,
 221                      mask_type='png'):
 222   """Creates a TFRecord file from examples.
 223 
 224   Args:
 225     output_filename: Path to where output file is saved.
 226     num_shards: Number of shards for output file.
 227     label_map_dict: The label map dictionary.
 228     annotations_dir: Directory where annotation files are stored.
 229     image_dir: Directory where image files are stored.
 230     examples: Examples to parse and save to tf record.
 231     faces_only: If True, generates bounding boxes for pet faces.  Otherwise
 232       generates bounding boxes (as well as segmentations for full pet bodies).
 233     mask_type: 'numerical' or 'png'. 'png' is recommended because it leads to
 234       smaller file sizes.
 235   """
 236   with contextlib2.ExitStack() as tf_record_close_stack:
 237     output_tfrecords = tf_record_creation_util.open_sharded_output_tfrecords(
 238         tf_record_close_stack, output_filename, num_shards)
 239     for idx, example in enumerate(examples):
 240       if idx % 100 == 0:
 241         logging.info('On image %d of %d', idx, len(examples))
 242       xml_path = os.path.join(annotations_dir, 'xmls', example + '.xml')
 243       mask_path = os.path.join(annotations_dir, 'trimaps', example + '.png')
 244 
 245       if not os.path.exists(xml_path):
 246         logging.warning('Could not find %s, ignoring example.', xml_path)
 247         continue
 248       with tf.gfile.GFile(xml_path, 'r') as fid:
 249         xml_str = fid.read()
 250       xml = etree.fromstring(xml_str)
 251       data = dataset_util.recursive_parse_xml_to_dict(xml)['annotation']
 252 
 253       try:
 254         tf_example = dict_to_tf_example(
 255             data,
 256             mask_path,
 257             label_map_dict,
 258             image_dir,
 259             faces_only=faces_only,
 260             mask_type=mask_type)
 261         if tf_example:
 262           shard_idx = idx % num_shards
 263           output_tfrecords[shard_idx].write(tf_example.SerializeToString())
 264       except ValueError:
 265         logging.warning('Invalid example: %s, ignoring.', xml_path)

 268 # TODO(derekjchow): Add test for pet/PASCAL main files.
 269 def main(_):
 270   data_dir = FLAGS.data_dir
 271   label_map_dict = label_map_util.get_label_map_dict(FLAGS.label_map_path)
 272 
 273   logging.info('Reading from Pet dataset.')
 274   image_dir = os.path.join(data_dir, 'images')
 275   annotations_dir = os.path.join(data_dir, 'annotations')
 276   examples_path = os.path.join(annotations_dir, 'trainval.txt')
 277   examples_list = dataset_util.read_examples_list(examples_path)
 278 
 279   # Test images are not included in the downloaded data set, so we shall perform
 280   # our own split.
 281   random.seed(42)
 282   random.shuffle(examples_list)
 283   num_examples = len(examples_list)
 284   num_train = int(0.7 * num_examples)
 285   train_examples = examples_list[:num_train]
 286   val_examples = examples_list[num_train:]
 287   logging.info('%d training and %d validation examples.',
 288                len(train_examples), len(val_examples))
 289 
 290   train_output_path = os.path.join(FLAGS.output_dir, 'pet_faces_train.record')
 291   val_output_path = os.path.join(FLAGS.output_dir, 'pet_faces_val.record')
 292   if not FLAGS.faces_only:
 293     train_output_path = os.path.join(FLAGS.output_dir,
 294                                      'pets_fullbody_with_masks_train.record')
 295     val_output_path = os.path.join(FLAGS.output_dir,
 296                                    'pets_fullbody_with_masks_val.record')
 297   create_tf_record(
 298       train_output_path,
 299       FLAGS.num_shards,
 300       label_map_dict,
 301       annotations_dir,
 302       image_dir,
 303       train_examples,
 304       faces_only=FLAGS.faces_only,
 305       mask_type=FLAGS.mask_type)
 306   create_tf_record(
 307       val_output_path,
 308       FLAGS.num_shards,
 309       label_map_dict,
 310       annotations_dir,
 311       image_dir,
 312       val_examples,
 313       faces_only=FLAGS.faces_only,
 314       mask_type=FLAGS.mask_type)
 315 
 316 
 317 if __name__ == '__main__':
 318   tf.app.run()
