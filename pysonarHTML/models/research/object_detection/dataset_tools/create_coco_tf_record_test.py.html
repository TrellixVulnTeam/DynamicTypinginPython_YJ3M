  31   def _assertProtoEqual(self, proto_field, expectation):
  32     """Helper function to assert if a proto field equals some value.
  33 
  34     Args:
  35       proto_field: The protobuf field to compare.
  36       expectation: The expected value of the protobuf field.
  37     """
  38     proto_list = [p for p in proto_field]
  39     self.assertListEqual(proto_list, expectation)
  40 
  41   def _assertProtoClose(self, proto_field, expectation):
  42     """Helper function to assert if a proto field nearly equals some value.
  43 
  44     Args:
  45       proto_field: The protobuf field to compare.
  46       expectation: The expected value of the protobuf field.
  47     """
  48     proto_list = [p for p in proto_field]
  49     self.assertAllClose(proto_list, expectation)

 206   def test_create_tf_example_with_keypoints(self):
 207     image_dir = self.get_temp_dir()
 208     image_file_name = 'tmp_image.jpg'
 209     image_data = np.random.randint(low=0, high=256, size=(256, 256, 3)).astype(
 210         np.uint8)
 211     save_path = os.path.join(image_dir, image_file_name)
 212     image = PIL.Image.fromarray(image_data, 'RGB')
 213     image.save(save_path)
 214 
 215     image = {
 216         'file_name': image_file_name,
 217         'height': 256,
 218         'width': 256,
 219         'id': 11,
 220     }
 221 
 222     min_x, min_y = 64, 64
 223     max_x, max_y = 128, 128
 224     keypoints = []
 225     num_visible_keypoints = 0
 226     xv = []
 227     yv = []
 228     vv = []
 229     for _ in range(17):
 230       xc = min_x + int(np.random.rand()*(max_x - min_x))
 231       yc = min_y + int(np.random.rand()*(max_y - min_y))
 232       vis = np.random.randint(0, 3)
 233       xv.append(xc)
 234       yv.append(yc)
 235       vv.append(vis)
 236       keypoints.extend([xc, yc, vis])
 237       num_visible_keypoints += (vis > 0)
 238 
 239     annotations_list = [{
 240         'area': 0.5,
 241         'iscrowd': False,
 242         'image_id': 11,
 243         'bbox': [64, 64, 128, 128],
 244         'category_id': 1,
 245         'id': 1000
 246     }]
 247 
 248     keypoint_annotations_dict = {
 249         1000: {
 250             'keypoints': keypoints,
 251             'num_keypoints': num_visible_keypoints
 252         }
 253     }
 254 
 255     category_index = {
 256         1: {
 257             'name': 'person',
 258             'id': 1
 259         }
 260     }
 261 
 262     (_, example, _,
 263      num_keypoint_annotation_skipped) = create_coco_tf_record.create_tf_example(
 264          image,
 265          annotations_list,
 266          image_dir,
 267          category_index,
 268          include_masks=False,
 269          keypoint_annotations_dict=keypoint_annotations_dict)
 270 
 271     self.assertEqual(num_keypoint_annotation_skipped, 0)
 272     self._assertProtoEqual(
 273         example.features.feature['image/height'].int64_list.value, [256])
 274     self._assertProtoEqual(
 275         example.features.feature['image/width'].int64_list.value, [256])
 276     self._assertProtoEqual(
 277         example.features.feature['image/filename'].bytes_list.value,
 278         [six.b(image_file_name)])
 279     self._assertProtoEqual(
 280         example.features.feature['image/source_id'].bytes_list.value,
 281         [six.b(str(image['id']))])
 282     self._assertProtoEqual(
 283         example.features.feature['image/format'].bytes_list.value,
 284         [six.b('jpeg')])
 285     self._assertProtoEqual(
 286         example.features.feature['image/object/bbox/xmin'].float_list.value,
 287         [0.25])
 288     self._assertProtoEqual(
 289         example.features.feature['image/object/bbox/ymin'].float_list.value,
 290         [0.25])
 291     self._assertProtoEqual(
 292         example.features.feature['image/object/bbox/xmax'].float_list.value,
 293         [0.75])
 294     self._assertProtoEqual(
 295         example.features.feature['image/object/bbox/ymax'].float_list.value,
 296         [0.75])
 297     self._assertProtoEqual(
 298         example.features.feature['image/object/class/text'].bytes_list.value,
 299         [six.b('person')])
 300     self._assertProtoClose(
 301         example.features.feature['image/object/keypoint/x'].float_list.value,
 302         np.array(xv, dtype=np.float32) / 256)
 303     self._assertProtoClose(
 304         example.features.feature['image/object/keypoint/y'].float_list.value,
 305         np.array(yv, dtype=np.float32) / 256)
 306     self._assertProtoEqual(
 307         example.features.feature['image/object/keypoint/text'].bytes_list.value,
 308         create_coco_tf_record._COCO_KEYPOINT_NAMES)
 309     self._assertProtoEqual(
 310         example.features.feature[
 311             'image/object/keypoint/visibility'].int64_list.value, vv)

 313   def test_create_sharded_tf_record(self):
 314     tmp_dir = self.get_temp_dir()
 315     image_paths = ['tmp1_image.jpg', 'tmp2_image.jpg']
 316     for image_path in image_paths:
 317       image_data = np.random.rand(256, 256, 3)
 318       save_path = os.path.join(tmp_dir, image_path)
 319       image = PIL.Image.fromarray(image_data, 'RGB')
 320       image.save(save_path)
 321 
 322     images = [{
 323         'file_name': image_paths[0],
 324         'height': 256,
 325         'width': 256,
 326         'id': 11,
 327     }, {
 328         'file_name': image_paths[1],
 329         'height': 256,
 330         'width': 256,
 331         'id': 12,
 332     }]
 333 
 334     annotations = [{
 335         'area': .5,
 336         'iscrowd': False,
 337         'image_id': 11,
 338         'bbox': [64, 64, 128, 128],
 339         'category_id': 2,
 340         'id': 1000,
 341     }]
 342 
 343     category_index = [{
 344         'name': 'dog',
 345         'id': 1
 346     }, {
 347         'name': 'cat',
 348         'id': 2
 349     }, {
 350         'name': 'human',
 351         'id': 3
 352     }]
 353     groundtruth_data = {'images': images, 'annotations': annotations,
 354                         'categories': category_index}
 355     annotation_file = os.path.join(tmp_dir, 'annotation.json')
 356     with open(annotation_file, 'w') as annotation_fid:
 357       json.dump(groundtruth_data, annotation_fid)
 358 
 359     output_path = os.path.join(tmp_dir, 'out.record')
 360     create_coco_tf_record._create_tf_record_from_coco_annotations(
 361         annotation_file,
 362         tmp_dir,
 363         output_path,
 364         False,
 365         2)
 366     self.assertTrue(os.path.exists(output_path + '-00000-of-00002'))
 367     self.assertTrue(os.path.exists(output_path + '-00001-of-00002'))
