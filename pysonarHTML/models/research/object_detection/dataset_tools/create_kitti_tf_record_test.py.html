  28 class CreateKittiTFRecordTest(tf.test.TestCase):
  29 
  30   def _assertProtoEqual(self, proto_field, expectation):
  31     """Helper function to assert if a proto field equals some value.
  32 
  33     Args:
  34       proto_field: The protobuf field to compare.
  35       expectation: The expected value of the protobuf field.
  36     """
  37     proto_list = [p for p in proto_field]
  38     self.assertListEqual(proto_list, expectation)
  39 
  40   def test_dict_to_tf_example(self):
  41     image_file_name = 'tmp_image.jpg'
  42     image_data = np.random.rand(256, 256, 3)
  43     save_path = os.path.join(self.get_temp_dir(), image_file_name)
  44     image = PIL.Image.fromarray(image_data, 'RGB')
  45     image.save(save_path)
  46 
  47     annotations = {}
  48     annotations['2d_bbox_left'] = np.array([64])
  49     annotations['2d_bbox_top'] = np.array([64])
  50     annotations['2d_bbox_right'] = np.array([192])
  51     annotations['2d_bbox_bottom'] = np.array([192])
  52     annotations['type'] = ['car']
  53     annotations['truncated'] = np.array([1])
  54     annotations['alpha'] = np.array([2])
  55     annotations['3d_bbox_height'] = np.array([10])
  56     annotations['3d_bbox_width'] = np.array([11])
  57     annotations['3d_bbox_length'] = np.array([12])
  58     annotations['3d_bbox_x'] = np.array([13])
  59     annotations['3d_bbox_y'] = np.array([14])
  60     annotations['3d_bbox_z'] = np.array([15])
  61     annotations['3d_bbox_rot_y'] = np.array([4])
  62 
  63     label_map_dict = {
  64         'background': 0,
  65         'car': 1,
  66     }
  67 
  68     example = create_kitti_tf_record.prepare_example(
  69         save_path,
  70         annotations,
  71         label_map_dict)
  72 
  73     self._assertProtoEqual(
  74         example.features.feature['image/height'].int64_list.value, [256])
  75     self._assertProtoEqual(
  76         example.features.feature['image/width'].int64_list.value, [256])
  77     self._assertProtoEqual(
  78         example.features.feature['image/filename'].bytes_list.value,
  79         [six.b(save_path)])
  80     self._assertProtoEqual(
  81         example.features.feature['image/source_id'].bytes_list.value,
  82         [six.b(save_path)])
  83     self._assertProtoEqual(
  84         example.features.feature['image/format'].bytes_list.value,
  85         [six.b('png')])
  86     self._assertProtoEqual(
  87         example.features.feature['image/object/bbox/xmin'].float_list.value,
  88         [0.25])
  89     self._assertProtoEqual(
  90         example.features.feature['image/object/bbox/ymin'].float_list.value,
  91         [0.25])
  92     self._assertProtoEqual(
  93         example.features.feature['image/object/bbox/xmax'].float_list.value,
  94         [0.75])
  95     self._assertProtoEqual(
  96         example.features.feature['image/object/bbox/ymax'].float_list.value,
  97         [0.75])
  98     self._assertProtoEqual(
  99         example.features.feature['image/object/class/text'].bytes_list.value,
 100         [six.b('car')])
 101     self._assertProtoEqual(
 102         example.features.feature['image/object/class/label'].int64_list.value,
 103         [1])
 104     self._assertProtoEqual(
 105         example.features.feature['image/object/truncated'].float_list.value,
 106         [1])
 107     self._assertProtoEqual(
 108         example.features.feature['image/object/alpha'].float_list.value,
 109         [2])
 110     self._assertProtoEqual(example.features.feature[
 111         'image/object/3d_bbox/height'].float_list.value, [10])
 112     self._assertProtoEqual(
 113         example.features.feature['image/object/3d_bbox/width'].float_list.value,
 114         [11])

 116         'image/object/3d_bbox/length'].float_list.value, [12])
