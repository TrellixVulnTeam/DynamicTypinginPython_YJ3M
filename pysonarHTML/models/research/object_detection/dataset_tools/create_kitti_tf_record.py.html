  16 r"""Convert raw KITTI detection dataset to TFRecord for object_detection.
  17 
  18 Converts KITTI detection dataset to TFRecords with a standard format allowing
  19   to use this dataset to train object detectors. The raw dataset can be
  20   downloaded from:
  21   http://kitti.is.tue.mpg.de/kitti/data_object_image_2.zip.
  22   http://kitti.is.tue.mpg.de/kitti/data_object_label_2.zip
  23   Permission can be requested at the main website.
  24 
  25   KITTI detection dataset contains 7481 training images. Using this code with
  26   the default settings will set aside the first 500 images as a validation set.
  27   This can be altered using the flags, see details below.
  28 
  29 Example usage:
  30     python object_detection/dataset_tools/create_kitti_tf_record.py \
  31         --data_dir=/home/user/kitti \
  32         --output_path=/home/user/kitti.record
  33 """

  36 from __future__ import print_function

  72 def convert_kitti_to_tfrecords(data_dir, output_path, classes_to_use,
  73                                label_map_path, validation_set_size):
  74   """Convert the KITTI detection dataset to TFRecords.
  75 
  76   Args:
  77     data_dir: The full path to the unzipped folder containing the unzipped data
  78       from data_object_image_2 and data_object_label_2.zip.
  79       Folder structure is assumed to be: data_dir/training/label_2 (annotations)
  80       and data_dir/data_object_image_2/training/image_2 (images).
  81     output_path: The path to which TFRecord files will be written. The TFRecord
  82       with the training set will be located at: <output_path>_train.tfrecord
  83       And the TFRecord with the validation set will be located at:
  84       <output_path>_val.tfrecord
  85     classes_to_use: List of strings naming the classes for which data should be
  86       converted. Use the same names as presented in the KIITI README file.
  87       Adding dontcare class will remove all other bounding boxes that overlap
  88       with areas marked as dontcare regions.
  89     label_map_path: Path to label map proto
  90     validation_set_size: How many images should be left as the validation set.
  91       (Ffirst `validation_set_size` examples are selected to be in the
  92       validation set).
  93   """
  94   label_map_dict = label_map_util.get_label_map_dict(label_map_path)
  95   train_count = 0
  96   val_count = 0
  97 
  98   annotation_dir = os.path.join(data_dir,
  99                                 'training',
 100                                 'label_2')
 101 
 102   image_dir = os.path.join(data_dir,
 103                            'data_object_image_2',
 104                            'training',
 105                            'image_2')
 106 
 107   train_writer = tf.python_io.TFRecordWriter('%s_train.tfrecord'%
 108                                              output_path)
 109   val_writer = tf.python_io.TFRecordWriter('%s_val.tfrecord'%
 110                                            output_path)
 111 
 112   images = sorted(tf.gfile.ListDirectory(image_dir))
 113   for img_name in images:
 114     img_num = int(img_name.split('.')[0])
 115     is_validation_img = img_num < validation_set_size
 116     img_anno = read_annotation_file(os.path.join(annotation_dir,
 117                                                  str(img_num).zfill(6)+'.txt'))
 118 
 119     image_path = os.path.join(image_dir, img_name)
 120 
 121     # Filter all bounding boxes of this frame that are of a legal class, and
 122     # don't overlap with a dontcare region.
 123     # TODO(talremez) filter out targets that are truncated or heavily occluded.
 124     annotation_for_image = filter_annotations(img_anno, classes_to_use)
 125 
 126     example = prepare_example(image_path, annotation_for_image, label_map_dict)
 127     if is_validation_img:
 128       val_writer.write(example.SerializeToString())
 129       val_count += 1
 130     else:
 131       train_writer.write(example.SerializeToString())
 132       train_count += 1
 133 
 134   train_writer.close()
 135   val_writer.close()

 138 def prepare_example(image_path, annotations, label_map_dict):
 139   """Converts a dictionary with annotations for an image to tf.Example proto.
 140 
 141   Args:
 142     image_path: The complete path to image.
 143     annotations: A dictionary representing the annotation of a single object
 144       that appears in the image.
 145     label_map_dict: A map from string label names to integer ids.
 146 
 147   Returns:
 148     example: The converted tf.Example.
 149   """
 150   with tf.gfile.GFile(image_path, 'rb') as fid:
 151     encoded_png = fid.read()
 152   encoded_png_io = io.BytesIO(encoded_png)
 153   image = pil.open(encoded_png_io)
 154   image = np.asarray(image)
 155 
 156   key = hashlib.sha256(encoded_png).hexdigest()
 157 
 158   width = int(image.shape[1])
 159   height = int(image.shape[0])
 160 
 161   xmin_norm = annotations['2d_bbox_left'] / float(width)
 162   ymin_norm = annotations['2d_bbox_top'] / float(height)
 163   xmax_norm = annotations['2d_bbox_right'] / float(width)
 164   ymax_norm = annotations['2d_bbox_bottom'] / float(height)
 165 
 166   difficult_obj = [0]*len(xmin_norm)
 167 
 168   example = tf.train.Example(features=tf.train.Features(feature={
 169       'image/height': dataset_util.int64_feature(height),
 170       'image/width': dataset_util.int64_feature(width),
 171       'image/filename': dataset_util.bytes_feature(image_path.encode('utf8')),
 172       'image/source_id': dataset_util.bytes_feature(image_path.encode('utf8')),
 173       'image/key/sha256': dataset_util.bytes_feature(key.encode('utf8')),
 174       'image/encoded': dataset_util.bytes_feature(encoded_png),
 175       'image/format': dataset_util.bytes_feature('png'.encode('utf8')),
 176       'image/object/bbox/xmin': dataset_util.float_list_feature(xmin_norm),
 177       'image/object/bbox/xmax': dataset_util.float_list_feature(xmax_norm),
 178       'image/object/bbox/ymin': dataset_util.float_list_feature(ymin_norm),
 179       'image/object/bbox/ymax': dataset_util.float_list_feature(ymax_norm),
 180       'image/object/class/text': dataset_util.bytes_list_feature(
 181           [x.encode('utf8') for x in annotations['type']]),
 182       'image/object/class/label': dataset_util.int64_list_feature(
 183           [label_map_dict[x] for x in annotations['type']]),
 184       'image/object/difficult': dataset_util.int64_list_feature(difficult_obj),
 185       'image/object/truncated': dataset_util.float_list_feature(
 186           annotations['truncated']),
 187       'image/object/alpha': dataset_util.float_list_feature(
 188           annotations['alpha']),
 189       'image/object/3d_bbox/height': dataset_util.float_list_feature(
 190           annotations['3d_bbox_height']),
 191       'image/object/3d_bbox/width': dataset_util.float_list_feature(
 192           annotations['3d_bbox_width']),
 193       'image/object/3d_bbox/length': dataset_util.float_list_feature(
 194           annotations['3d_bbox_length']),
 195       'image/object/3d_bbox/x': dataset_util.float_list_feature(
 196           annotations['3d_bbox_x']),
 197       'image/object/3d_bbox/y': dataset_util.float_list_feature(
 198           annotations['3d_bbox_y']),
 199       'image/object/3d_bbox/z': dataset_util.float_list_feature(
 200           annotations['3d_bbox_z']),
 201       'image/object/3d_bbox/rot_y': dataset_util.float_list_feature(
 202           annotations['3d_bbox_rot_y']),
 203   }))
 204 
 205   return example

 208 def filter_annotations(img_all_annotations, used_classes):
 209   """Filters out annotations from the unused classes and dontcare regions.
 210 
 211   Filters out the annotations that belong to classes we do now wish to use and
 212   (optionally) also removes all boxes that overlap with dontcare regions.
 213 
 214   Args:
 215     img_all_annotations: A list of annotation dictionaries. See documentation of
 216       read_annotation_file for more details about the format of the annotations.
 217     used_classes: A list of strings listing the classes we want to keep, if the
 218     list contains "dontcare", all bounding boxes with overlapping with dont
 219     care regions will also be filtered out.
 220 
 221   Returns:
 222     img_filtered_annotations: A list of annotation dictionaries that have passed
 223       the filtering.
 224   """
 225 
 226   img_filtered_annotations = {}
 227 
 228   # Filter the type of the objects.
 229   relevant_annotation_indices = [
 230       i for i, x in enumerate(img_all_annotations['type']) if x in used_classes
 231   ]
 232 
 233   for key in img_all_annotations.keys():
 234     img_filtered_annotations[key] = (
 235         img_all_annotations[key][relevant_annotation_indices])
 236 
 237   if 'dontcare' in used_classes:
 238     dont_care_indices = [i for i,
 239                          x in enumerate(img_filtered_annotations['type'])
 240                          if x == 'dontcare']
 241 
 242     # bounding box format [y_min, x_min, y_max, x_max]
 243     all_boxes = np.stack([img_filtered_annotations['2d_bbox_top'],
 244                           img_filtered_annotations['2d_bbox_left'],
 245                           img_filtered_annotations['2d_bbox_bottom'],
 246                           img_filtered_annotations['2d_bbox_right']],
 247                          axis=1)
 248 
 249     ious = iou(boxes1=all_boxes,
 250                boxes2=all_boxes[dont_care_indices])
 251 
 252     # Remove all bounding boxes that overlap with a dontcare region.
 253     if ious.size > 0:
 254       boxes_to_remove = np.amax(ious, axis=1) > 0.0
 255       for key in img_all_annotations.keys():
 256         img_filtered_annotations[key] = (
 257             img_filtered_annotations[key][np.logical_not(boxes_to_remove)])
 258 
 259   return img_filtered_annotations

 262 def read_annotation_file(filename):
 263   """Reads a KITTI annotation file.
 264 
 265   Converts a KITTI annotation file into a dictionary containing all the
 266   relevant information.
 267 
 268   Args:
 269     filename: the path to the annotataion text file.
 270 
 271   Returns:
 272     anno: A dictionary with the converted annotation information. See annotation
 273     README file for details on the different fields.
 274   """
 275   with open(filename) as f:
 276     content = f.readlines()
 277   content = [x.strip().split(' ') for x in content]
 278 
 279   anno = {}
 280   anno['type'] = np.array([x[0].lower() for x in content])
 281   anno['truncated'] = np.array([float(x[1]) for x in content])
 282   anno['occluded'] = np.array([int(x[2]) for x in content])
 283   anno['alpha'] = np.array([float(x[3]) for x in content])
 284 
 285   anno['2d_bbox_left'] = np.array([float(x[4]) for x in content])
 286   anno['2d_bbox_top'] = np.array([float(x[5]) for x in content])
 287   anno['2d_bbox_right'] = np.array([float(x[6]) for x in content])
 288   anno['2d_bbox_bottom'] = np.array([float(x[7]) for x in content])
 289 
 290   anno['3d_bbox_height'] = np.array([float(x[8]) for x in content])
 291   anno['3d_bbox_width'] = np.array([float(x[9]) for x in content])
 292   anno['3d_bbox_length'] = np.array([float(x[10]) for x in content])
 293   anno['3d_bbox_x'] = np.array([float(x[11]) for x in content])
 294   anno['3d_bbox_y'] = np.array([float(x[12]) for x in content])
 295   anno['3d_bbox_z'] = np.array([float(x[13]) for x in content])
 296   anno['3d_bbox_rot_y'] = np.array([float(x[14]) for x in content])
 297 
 298   return anno
