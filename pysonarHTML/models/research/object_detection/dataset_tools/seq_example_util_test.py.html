  31   def test_make_unlabeled_example(self):
  32     num_frames = 5
  33     image_height = 100
  34     image_width = 200
  35     dataset_name = b'unlabeled_dataset'
  36     video_id = b'video_000'
  37     images = tf.cast(tf.random.uniform(
  38         [num_frames, image_height, image_width, 3],
  39         maxval=256,
  40         dtype=tf.int32), dtype=tf.uint8)
  41     image_source_ids = [str(idx) for idx in range(num_frames)]
  42     images_list = tf.unstack(images, axis=0)
  43     encoded_images_list = [tf.io.encode_jpeg(image) for image in images_list]
  44     with tf.Session() as sess:
  45       encoded_images = sess.run(encoded_images_list)
  46     seq_example = seq_example_util.make_sequence_example(
  47         dataset_name=dataset_name,
  48         video_id=video_id,
  49         encoded_images=encoded_images,
  50         image_height=image_height,
  51         image_width=image_width,
  52         image_format='JPEG',
  53         image_source_ids=image_source_ids)
  54 
  55     context_feature_dict = seq_example.context.feature
  56     self.assertEqual(
  57         dataset_name,
  58         context_feature_dict['example/dataset_name'].bytes_list.value[0])
  59     self.assertEqual(
  60         0,
  61         context_feature_dict['clip/start/timestamp'].int64_list.value[0])
  62     self.assertEqual(
  63         num_frames - 1,
  64         context_feature_dict['clip/end/timestamp'].int64_list.value[0])
  65     self.assertEqual(
  66         num_frames,
  67         context_feature_dict['clip/frames'].int64_list.value[0])
  68     self.assertEqual(
  69         3,
  70         context_feature_dict['image/channels'].int64_list.value[0])
  71     self.assertEqual(
  72         b'JPEG',
  73         context_feature_dict['image/format'].bytes_list.value[0])
  74     self.assertEqual(
  75         image_height,
  76         context_feature_dict['image/height'].int64_list.value[0])
  77     self.assertEqual(
  78         image_width,
  79         context_feature_dict['image/width'].int64_list.value[0])
  80     self.assertEqual(
  81         video_id,
  82         context_feature_dict['clip/media_id'].bytes_list.value[0])
  83 
  84     seq_feature_dict = seq_example.feature_lists.feature_list
  85     self.assertLen(
  86         seq_feature_dict['image/encoded'].feature[:],
  87         num_frames)
  88     timestamps = [
  89         feature.int64_list.value[0] for feature
  90         in seq_feature_dict['image/timestamp'].feature]
  91     self.assertAllEqual(list(range(num_frames)), timestamps)
  92     source_ids = [
  93         feature.bytes_list.value[0] for feature
  94         in seq_feature_dict['image/source_id'].feature]
  95     self.assertAllEqual(
  96         [six.ensure_binary(str(idx)) for idx in range(num_frames)],
  97         source_ids)

  99   def test_make_labeled_example(self):
 100     num_frames = 2
 101     image_height = 100
 102     image_width = 200
 103     dataset_name = b'unlabeled_dataset'
 104     video_id = b'video_000'
 105     labels = [b'dog', b'cat']
 106     images = tf.cast(tf.random.uniform(
 107         [num_frames, image_height, image_width, 3],
 108         maxval=256,
 109         dtype=tf.int32), dtype=tf.uint8)
 110     images_list = tf.unstack(images, axis=0)
 111     encoded_images_list = [tf.io.encode_jpeg(image) for image in images_list]
 112     with tf.Session() as sess:
 113       encoded_images = sess.run(encoded_images_list)
 114     timestamps = [100000, 110000]
 115     is_annotated = [1, 0]
 116     bboxes = [
 117         np.array([[0., 0., 0., 0.],
 118                   [0., 0., 1., 1.]], dtype=np.float32),
 119         np.zeros([0, 4], dtype=np.float32)
 120     ]
 121     label_strings = [
 122         np.array(labels),
 123         np.array([])
 124     ]
 125 
 126     seq_example = seq_example_util.make_sequence_example(
 127         dataset_name=dataset_name,
 128         video_id=video_id,
 129         encoded_images=encoded_images,
 130         image_height=image_height,
 131         image_width=image_width,
 132         timestamps=timestamps,
 133         is_annotated=is_annotated,
 134         bboxes=bboxes,
 135         label_strings=label_strings)
 136 
 137     context_feature_dict = seq_example.context.feature
 138     self.assertEqual(
 139         dataset_name,
 140         context_feature_dict['example/dataset_name'].bytes_list.value[0])
 141     self.assertEqual(
 142         timestamps[0],
 143         context_feature_dict['clip/start/timestamp'].int64_list.value[0])
 144     self.assertEqual(
 145         timestamps[-1],
 146         context_feature_dict['clip/end/timestamp'].int64_list.value[0])
 147     self.assertEqual(
 148         num_frames,
 149         context_feature_dict['clip/frames'].int64_list.value[0])
 150 
 151     seq_feature_dict = seq_example.feature_lists.feature_list
 152     self.assertLen(
 153         seq_feature_dict['image/encoded'].feature[:],
 154         num_frames)
 155     actual_timestamps = [
 156         feature.int64_list.value[0] for feature
 157         in seq_feature_dict['image/timestamp'].feature]
 158     self.assertAllEqual(timestamps, actual_timestamps)
 159     # Frame 0.
 160     self.assertAllEqual(
 161         is_annotated[0],
 162         seq_feature_dict['region/is_annotated'].feature[0].int64_list.value[0])
 163     self.assertAllClose(
 164         [0., 0.],
 165         seq_feature_dict['region/bbox/ymin'].feature[0].float_list.value[:])
 166     self.assertAllClose(
 167         [0., 0.],
 168         seq_feature_dict['region/bbox/xmin'].feature[0].float_list.value[:])
 169     self.assertAllClose(
 170         [0., 1.],
 171         seq_feature_dict['region/bbox/ymax'].feature[0].float_list.value[:])
 172     self.assertAllClose(
 173         [0., 1.],
 174         seq_feature_dict['region/bbox/xmax'].feature[0].float_list.value[:])
 175     self.assertAllEqual(
 176         labels,
 177         seq_feature_dict['region/label/string'].feature[0].bytes_list.value[:])
 178 
 179     # Frame 1.
 180     self.assertAllEqual(
 181         is_annotated[1],
 182         seq_feature_dict['region/is_annotated'].feature[1].int64_list.value[0])
 183     self.assertAllClose(
 184         [],
 185         seq_feature_dict['region/bbox/ymin'].feature[1].float_list.value[:])
 186     self.assertAllClose(
 187         [],
 188         seq_feature_dict['region/bbox/xmin'].feature[1].float_list.value[:])
 189     self.assertAllClose(
 190         [],
 191         seq_feature_dict['region/bbox/ymax'].feature[1].float_list.value[:])
 192     self.assertAllClose(
 193         [],
 194         seq_feature_dict['region/bbox/xmax'].feature[1].float_list.value[:])
 195     self.assertAllEqual(
 196         [],
 197         seq_feature_dict['region/label/string'].feature[1].bytes_list.value[:])

 199   def test_make_labeled_example_with_predictions(self):
 200     num_frames = 2
 201     image_height = 100
 202     image_width = 200
 203     dataset_name = b'unlabeled_dataset'
 204     video_id = b'video_000'
 205     images = tf.cast(tf.random.uniform(
 206         [num_frames, image_height, image_width, 3],
 207         maxval=256,
 208         dtype=tf.int32), dtype=tf.uint8)
 209     images_list = tf.unstack(images, axis=0)
 210     encoded_images_list = [tf.io.encode_jpeg(image) for image in images_list]
 211     with tf.Session() as sess:
 212       encoded_images = sess.run(encoded_images_list)
 213     bboxes = [
 214         np.array([[0., 0., 0.75, 0.75],
 215                   [0., 0., 1., 1.]], dtype=np.float32),
 216         np.array([[0., 0.25, 0.5, 0.75]], dtype=np.float32)
 217     ]
 218     label_strings = [
 219         np.array(['cat', 'frog']),
 220         np.array(['cat'])
 221     ]
 222     detection_bboxes = [
 223         np.array([[0., 0., 0.75, 0.75]], dtype=np.float32),
 224         np.zeros([0, 4], dtype=np.float32)
 225     ]
 226     detection_classes = [
 227         np.array([5], dtype=np.int64),
 228         np.array([], dtype=np.int64)
 229     ]
 230     detection_scores = [
 231         np.array([0.9], dtype=np.float32),
 232         np.array([], dtype=np.float32)
 233     ]
 234 
 235     seq_example = seq_example_util.make_sequence_example(
 236         dataset_name=dataset_name,
 237         video_id=video_id,
 238         encoded_images=encoded_images,
 239         image_height=image_height,
 240         image_width=image_width,
 241         bboxes=bboxes,
 242         label_strings=label_strings,
 243         detection_bboxes=detection_bboxes,
 244         detection_classes=detection_classes,
 245         detection_scores=detection_scores)
 246 
 247     context_feature_dict = seq_example.context.feature
 248     self.assertEqual(
 249         dataset_name,
 250         context_feature_dict['example/dataset_name'].bytes_list.value[0])
 251     self.assertEqual(
 252         0,
 253         context_feature_dict['clip/start/timestamp'].int64_list.value[0])
 254     self.assertEqual(
 255         1,
 256         context_feature_dict['clip/end/timestamp'].int64_list.value[0])
 257     self.assertEqual(
 258         num_frames,
 259         context_feature_dict['clip/frames'].int64_list.value[0])
 260 
 261     seq_feature_dict = seq_example.feature_lists.feature_list
 262     self.assertLen(
 263         seq_feature_dict['image/encoded'].feature[:],
 264         num_frames)
 265     actual_timestamps = [
 266         feature.int64_list.value[0] for feature
 267         in seq_feature_dict['image/timestamp'].feature]
 268     self.assertAllEqual([0, 1], actual_timestamps)
 269     # Frame 0.
 270     self.assertAllEqual(
 271         1,
 272         seq_feature_dict['region/is_annotated'].feature[0].int64_list.value[0])
 273     self.assertAllClose(
 274         [0., 0.],
 275         seq_feature_dict['region/bbox/ymin'].feature[0].float_list.value[:])
 276     self.assertAllClose(
 277         [0., 0.],
 278         seq_feature_dict['region/bbox/xmin'].feature[0].float_list.value[:])
 279     self.assertAllClose(
 280         [0.75, 1.],
 281         seq_feature_dict['region/bbox/ymax'].feature[0].float_list.value[:])
 282     self.assertAllClose(
 283         [0.75, 1.],
 284         seq_feature_dict['region/bbox/xmax'].feature[0].float_list.value[:])
 285     self.assertAllEqual(
 286         ['cat', 'frog'],
 287         seq_feature_dict['region/label/string'].feature[0].bytes_list.value[:])
 288     self.assertAllClose(
 289         [0.],
 290         seq_feature_dict[
 291             'predicted/region/bbox/ymin'].feature[0].float_list.value[:])
 292     self.assertAllClose(
 293         [0.],
 294         seq_feature_dict[
 295             'predicted/region/bbox/xmin'].feature[0].float_list.value[:])
 296     self.assertAllClose(
 297         [0.75],
 298         seq_feature_dict[
 299             'predicted/region/bbox/ymax'].feature[0].float_list.value[:])
 300     self.assertAllClose(
 301         [0.75],
 302         seq_feature_dict[
 303             'predicted/region/bbox/xmax'].feature[0].float_list.value[:])
 304     self.assertAllEqual(
 305         [5],
 306         seq_feature_dict[
 307             'predicted/region/label/index'].feature[0].int64_list.value[:])
 308     self.assertAllClose(
 309         [0.9],
 310         seq_feature_dict[
 311             'predicted/region/label/confidence'].feature[0].float_list.value[:])
 312 
 313     # Frame 1.
 314     self.assertAllEqual(
 315         1,
 316         seq_feature_dict['region/is_annotated'].feature[1].int64_list.value[0])
 317     self.assertAllClose(
 318         [0.0],
 319         seq_feature_dict['region/bbox/ymin'].feature[1].float_list.value[:])
 320     self.assertAllClose(
 321         [0.25],
 322         seq_feature_dict['region/bbox/xmin'].feature[1].float_list.value[:])
 323     self.assertAllClose(
 324         [0.5],
 325         seq_feature_dict['region/bbox/ymax'].feature[1].float_list.value[:])
 326     self.assertAllClose(
 327         [0.75],
 328         seq_feature_dict['region/bbox/xmax'].feature[1].float_list.value[:])
 329     self.assertAllEqual(
 330         ['cat'],
 331         seq_feature_dict['region/label/string'].feature[1].bytes_list.value[:])

 336     self.assertAllClose(
