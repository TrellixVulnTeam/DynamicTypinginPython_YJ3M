  25 def context_float_feature(ndarray):
  26   """Converts a numpy float array to a context float feature.
  27 
  28   Args:
  29     ndarray: A numpy float array.
  30 
  31   Returns:
  32     A context float feature.
  33   """
  34   feature = tf.train.Feature()
  35   for val in ndarray:
  36     feature.float_list.value.append(val)
  37   return feature

  55 def context_bytes_feature(ndarray):
  56   """Converts a numpy bytes array to a context bytes feature.
  57 
  58   Args:
  59     ndarray: A numpy bytes array.
  60 
  61   Returns:
  62     A context bytes feature.
  63   """
  64   feature = tf.train.Feature()
  65   for val in ndarray:
  66     if isinstance(val, np.ndarray):
  67       val = val.tolist()
  68     feature.bytes_list.value.append(tf.compat.as_bytes(val))
  69   return feature

  72 def sequence_float_feature(ndarray):
  73   """Converts a numpy float array to a sequence float feature.
  74 
  75   Args:
  76     ndarray: A numpy float array.
  77 
  78   Returns:
  79     A sequence float feature.
  80   """
  81   feature_list = tf.train.FeatureList()
  82   for row in ndarray:
  83     feature = feature_list.feature.add()
  84     if row.size:
  85       feature.float_list.value[:] = row
  86   return feature_list

 106 def sequence_bytes_feature(ndarray):
 107   """Converts a bytes float array to a sequence bytes feature.
 108 
 109   Args:
 110     ndarray: A numpy bytes array.
 111 
 112   Returns:
 113     A sequence bytes feature.
 114   """
 115   feature_list = tf.train.FeatureList()
 116   for row in ndarray:
 117     if isinstance(row, np.ndarray):
 118       row = row.tolist()
 119     feature = feature_list.feature.add()
 120     if row:
 121       row = [tf.compat.as_bytes(val) for val in row]
 122       feature.bytes_list.value[:] = row
 123   return feature_list

 126 def boxes_to_box_components(bboxes):
 127   """Converts a list of numpy arrays (boxes) to box components.
 128 
 129   Args:
 130     bboxes: A numpy array of bounding boxes.
 131 
 132   Returns:
 133     Bounding box component lists.
 134   """
 135   ymin_list = []
 136   xmin_list = []
 137   ymax_list = []
 138   xmax_list = []
 139   for bbox in bboxes:
 140     bbox = np.array(bbox).astype(np.float32)
 141     ymin, xmin, ymax, xmax = np.split(bbox, 4, axis=1)
 142     ymin_list.append(np.reshape(ymin, [-1]))
 143     xmin_list.append(np.reshape(xmin, [-1]))
 144     ymax_list.append(np.reshape(ymax, [-1]))
 145     xmax_list.append(np.reshape(xmax, [-1]))
 146   return ymin_list, xmin_list, ymax_list, xmax_list

 149 def make_sequence_example(dataset_name,
 150                           video_id,
 151                           encoded_images,
 152                           image_height,
 153                           image_width,
 154                           image_format=None,
 155                           image_source_ids=None,
 156                           timestamps=None,
 157                           is_annotated=None,
 158                           bboxes=None,
 159                           label_strings=None,
 160                           detection_bboxes=None,
 161                           detection_classes=None,
 162                           detection_scores=None):
 163   """Constructs tf.SequenceExamples.
 164 
 165   Args:
 166     dataset_name: String with dataset name.
 167     video_id: String with video id.
 168     encoded_images: A [num_frames] list (or numpy array) of encoded image
 169       frames.
 170     image_height: Height of the images.
 171     image_width: Width of the images.
 172     image_format: Format of encoded images.
 173     image_source_ids: (Optional) A [num_frames] list of unique string ids for
 174       each image.
 175     timestamps: (Optional) A [num_frames] list (or numpy array) array with image
 176       timestamps.
 177     is_annotated: (Optional) A [num_frames] list (or numpy array) array
 178       in which each element indicates whether the frame has been annotated
 179       (1) or not (0).
 180     bboxes: (Optional) A list (with num_frames elements) of [num_boxes_i, 4]
 181       numpy float32 arrays holding boxes for each frame.
 182     label_strings: (Optional) A list (with num_frames_elements) of [num_boxes_i]
 183       numpy string arrays holding object string labels for each frame.
 184     detection_bboxes: (Optional) A list (with num_frames elements) of
 185       [num_boxes_i, 4] numpy float32 arrays holding prediction boxes for each
 186       frame.
 187     detection_classes: (Optional) A list (with num_frames_elements) of
 188       [num_boxes_i] numpy int64 arrays holding predicted classes for each frame.
 189     detection_scores: (Optional) A list (with num_frames_elements) of
 190       [num_boxes_i] numpy float32 arrays holding predicted object scores for
 191       each frame.
 192 
 193   Returns:
 194     A tf.train.SequenceExample.
 195   """
 196   num_frames = len(encoded_images)
 197   image_encoded = np.expand_dims(encoded_images, axis=-1)
 198   if timestamps is None:
 199     timestamps = np.arange(num_frames)
 200   image_timestamps = np.expand_dims(timestamps, axis=-1)
 201 
 202   # Context fields.
 203   context_dict = {
 204       'example/dataset_name': context_bytes_feature([dataset_name]),
 205       'clip/start/timestamp': context_int64_feature([image_timestamps[0][0]]),
 206       'clip/end/timestamp': context_int64_feature([image_timestamps[-1][0]]),
 207       'clip/frames': context_int64_feature([num_frames]),
 208       'image/channels': context_int64_feature([3]),
 209       'image/height': context_int64_feature([image_height]),
 210       'image/width': context_int64_feature([image_width]),
 211       'clip/media_id': context_bytes_feature([video_id])
 212   }
 213 
 214   # Sequence fields.
 215   feature_list = {
 216       'image/encoded': sequence_bytes_feature(image_encoded),
 217       'image/timestamp': sequence_int64_feature(image_timestamps),
 218   }
 219 
 220   # Add optional fields.
 221   if image_format is not None:
 222     context_dict['image/format'] = context_bytes_feature([image_format])
 223   if image_source_ids is not None:
 224     feature_list['image/source_id'] = sequence_bytes_feature(image_source_ids)
 225   if bboxes is not None:
 226     bbox_ymin, bbox_xmin, bbox_ymax, bbox_xmax = boxes_to_box_components(bboxes)
 227     feature_list['region/bbox/xmin'] = sequence_float_feature(bbox_xmin)
 228     feature_list['region/bbox/xmax'] = sequence_float_feature(bbox_xmax)
 229     feature_list['region/bbox/ymin'] = sequence_float_feature(bbox_ymin)
 230     feature_list['region/bbox/ymax'] = sequence_float_feature(bbox_ymax)
 231     if is_annotated is None:
 232       is_annotated = np.ones(num_frames, dtype=np.int64)
 233     is_annotated = np.expand_dims(is_annotated, axis=-1)
 234     feature_list['region/is_annotated'] = sequence_int64_feature(is_annotated)
 235 
 236   if label_strings is not None:
 237     feature_list['region/label/string'] = sequence_bytes_feature(
 238         label_strings)
 239 
 240   if detection_bboxes is not None:
 241     det_bbox_ymin, det_bbox_xmin, det_bbox_ymax, det_bbox_xmax = (
 242         boxes_to_box_components(detection_bboxes))
 243     feature_list['predicted/region/bbox/xmin'] = sequence_float_feature(
 244         det_bbox_xmin)
 245     feature_list['predicted/region/bbox/xmax'] = sequence_float_feature(
 246         det_bbox_xmax)
 247     feature_list['predicted/region/bbox/ymin'] = sequence_float_feature(
 248         det_bbox_ymin)
 249     feature_list['predicted/region/bbox/ymax'] = sequence_float_feature(
 250         det_bbox_ymax)
 251   if detection_classes is not None:
 252     feature_list['predicted/region/label/index'] = sequence_int64_feature(
 253         detection_classes)
 254   if detection_scores is not None:
 255     feature_list['predicted/region/label/confidence'] = sequence_float_feature(
 256         detection_scores)
 257 
 258   context = tf.train.Features(feature=context_dict)
 259   feature_lists = tf.train.FeatureLists(feature_list=feature_list)
 260 
 261   sequence_example = tf.train.SequenceExample(
 262       context=context,
 263       feature_lists=feature_lists)
 264   return sequence_example
