  81 def create_tf_example(image,
  82                       annotations_list,
  83                       image_dir,
  84                       category_index,
  85                       include_masks=False,
  86                       keypoint_annotations_dict=None):
  87   """Converts image and annotations to a tf.Example proto.
  88 
  89   Args:
  90     image: dict with keys: [u'license', u'file_name', u'coco_url', u'height',
  91       u'width', u'date_captured', u'flickr_url', u'id']
  92     annotations_list:
  93       list of dicts with keys: [u'segmentation', u'area', u'iscrowd',
  94         u'image_id', u'bbox', u'category_id', u'id'] Notice that bounding box
  95         coordinates in the official COCO dataset are given as [x, y, width,
  96         height] tuples using absolute coordinates where x, y represent the
  97         top-left (0-indexed) corner.  This function converts to the format
  98         expected by the Tensorflow Object Detection API (which is which is
  99         [ymin, xmin, ymax, xmax] with coordinates normalized relative to image
 100         size).
 101     image_dir: directory containing the image files.
 102     category_index: a dict containing COCO category information keyed by the
 103       'id' field of each category.  See the label_map_util.create_category_index
 104       function.
 105     include_masks: Whether to include instance segmentations masks
 106       (PNG encoded) in the result. default: False.
 107     keypoint_annotations_dict: A dictionary that maps from annotation_id to a
 108       dictionary with keys: [u'keypoints', u'num_keypoints'] represeting the
 109       keypoint information for this person object annotation. If None, then
 110       no keypoint annotations will be populated.
 111 
 112   Returns:
 113     example: The converted tf.Example
 114     num_annotations_skipped: Number of (invalid) annotations that were ignored.
 115 
 116   Raises:
 117     ValueError: if the image pointed to by data['filename'] is not a valid JPEG
 118   """
 119   image_height = image['height']
 120   image_width = image['width']
 121   filename = image['file_name']
 122   image_id = image['id']
 123 
 124   full_path = os.path.join(image_dir, filename)
 125   with tf.gfile.GFile(full_path, 'rb') as fid:
 126     encoded_jpg = fid.read()
 127   encoded_jpg_io = io.BytesIO(encoded_jpg)
 128   image = PIL.Image.open(encoded_jpg_io)
 129   key = hashlib.sha256(encoded_jpg).hexdigest()
 130 
 131   xmin = []
 132   xmax = []
 133   ymin = []
 134   ymax = []
 135   is_crowd = []
 136   category_names = []
 137   category_ids = []
 138   area = []
 139   encoded_mask_png = []
 140   keypoints_x = []
 141   keypoints_y = []
 142   keypoints_visibility = []
 143   keypoints_name = []
 144   num_keypoints = []
 145   include_keypoint = keypoint_annotations_dict is not None
 146   num_annotations_skipped = 0
 147   num_keypoint_annotation_used = 0
 148   num_keypoint_annotation_skipped = 0
 149   for object_annotations in annotations_list:
 150     (x, y, width, height) = tuple(object_annotations['bbox'])
 151     if width <= 0 or height <= 0:
 152       num_annotations_skipped += 1
 153       continue
 154     if x + width > image_width or y + height > image_height:
 155       num_annotations_skipped += 1
 156       continue
 157     xmin.append(float(x) / image_width)
 158     xmax.append(float(x + width) / image_width)
 159     ymin.append(float(y) / image_height)
 160     ymax.append(float(y + height) / image_height)
 161     is_crowd.append(object_annotations['iscrowd'])
 162     category_id = int(object_annotations['category_id'])
 163     category_ids.append(category_id)
 164     category_names.append(category_index[category_id]['name'].encode('utf8'))
 165     area.append(object_annotations['area'])
 166 
 167     if include_masks:
 168       run_len_encoding = mask.frPyObjects(object_annotations['segmentation'],
 169                                           image_height, image_width)
 170       binary_mask = mask.decode(run_len_encoding)
 171       if not object_annotations['iscrowd']:
 172         binary_mask = np.amax(binary_mask, axis=2)
 173       pil_image = PIL.Image.fromarray(binary_mask)
 174       output_io = io.BytesIO()
 175       pil_image.save(output_io, format='PNG')
 176       encoded_mask_png.append(output_io.getvalue())
 177 
 178     if include_keypoint:
 179       annotation_id = object_annotations['id']
 180       if annotation_id in keypoint_annotations_dict:
 181         num_keypoint_annotation_used += 1
 182         keypoint_annotations = keypoint_annotations_dict[annotation_id]
 183         keypoints = keypoint_annotations['keypoints']
 184         num_kpts = keypoint_annotations['num_keypoints']
 185         keypoints_x_abs = keypoints[::3]
 186         keypoints_x.extend(
 187             [float(x_abs) / image_width for x_abs in keypoints_x_abs])
 188         keypoints_y_abs = keypoints[1::3]
 189         keypoints_y.extend(
 190             [float(y_abs) / image_height for y_abs in keypoints_y_abs])
 191         keypoints_visibility.extend(keypoints[2::3])
 192         keypoints_name.extend(_COCO_KEYPOINT_NAMES)
 193         num_keypoints.append(num_kpts)
 194       else:
 195         keypoints_x.extend([0.0] * len(_COCO_KEYPOINT_NAMES))
 196         keypoints_y.extend([0.0] * len(_COCO_KEYPOINT_NAMES))
 197         keypoints_visibility.extend([0] * len(_COCO_KEYPOINT_NAMES))
 198         keypoints_name.extend(_COCO_KEYPOINT_NAMES)
 199         num_keypoints.append(0)
 200   feature_dict = {
 201       'image/height':
 202           dataset_util.int64_feature(image_height),
 203       'image/width':
 204           dataset_util.int64_feature(image_width),
 205       'image/filename':
 206           dataset_util.bytes_feature(filename.encode('utf8')),
 207       'image/source_id':
 208           dataset_util.bytes_feature(str(image_id).encode('utf8')),
 209       'image/key/sha256':
 210           dataset_util.bytes_feature(key.encode('utf8')),
 211       'image/encoded':
 212           dataset_util.bytes_feature(encoded_jpg),
 213       'image/format':
 214           dataset_util.bytes_feature('jpeg'.encode('utf8')),
 215       'image/object/bbox/xmin':
 216           dataset_util.float_list_feature(xmin),
 217       'image/object/bbox/xmax':
 218           dataset_util.float_list_feature(xmax),
 219       'image/object/bbox/ymin':
 220           dataset_util.float_list_feature(ymin),
 221       'image/object/bbox/ymax':
 222           dataset_util.float_list_feature(ymax),
 223       'image/object/class/text':
 224           dataset_util.bytes_list_feature(category_names),
 225       'image/object/is_crowd':
 226           dataset_util.int64_list_feature(is_crowd),
 227       'image/object/area':
 228           dataset_util.float_list_feature(area),
 229   }
 230   if include_masks:
 231     feature_dict['image/object/mask'] = (
 232         dataset_util.bytes_list_feature(encoded_mask_png))
 233   if include_keypoint:
 234     feature_dict['image/object/keypoint/x'] = (
 235         dataset_util.float_list_feature(keypoints_x))
 236     feature_dict['image/object/keypoint/y'] = (
 237         dataset_util.float_list_feature(keypoints_y))
 238     feature_dict['image/object/keypoint/num'] = (
 239         dataset_util.int64_list_feature(num_keypoints))
 240     feature_dict['image/object/keypoint/visibility'] = (
 241         dataset_util.int64_list_feature(keypoints_visibility))
 242     feature_dict['image/object/keypoint/text'] = (
 243         dataset_util.bytes_list_feature(keypoints_name))
 244     num_keypoint_annotation_skipped = (
 245         len(keypoint_annotations_dict) - num_keypoint_annotation_used)
 246 
 247   example = tf.train.Example(features=tf.train.Features(feature=feature_dict))
 248   return key, example, num_annotations_skipped, num_keypoint_annotation_skipped

 251 def _create_tf_record_from_coco_annotations(annotations_file, image_dir,
 252                                             output_path, include_masks,
 253                                             num_shards,
 254                                             keypoint_annotations_file=''):
 255   """Loads COCO annotation json files and converts to tf.Record format.
 256 
 257   Args:
 258     annotations_file: JSON file containing bounding box annotations.
 259     image_dir: Directory containing the image files.
 260     output_path: Path to output tf.Record file.
 261     include_masks: Whether to include instance segmentations masks
 262       (PNG encoded) in the result. default: False.
 263     num_shards: number of output file shards.
 264     keypoint_annotations_file: JSON file containing the person keypoint
 265       annotations. If empty, then no person keypoint annotations will be
 266       generated.
 267   """
 268   with contextlib2.ExitStack() as tf_record_close_stack, \
 269       tf.gfile.GFile(annotations_file, 'r') as fid:
 270     output_tfrecords = tf_record_creation_util.open_sharded_output_tfrecords(
 271         tf_record_close_stack, output_path, num_shards)
 272     groundtruth_data = json.load(fid)
 273     images = groundtruth_data['images']
 274     category_index = label_map_util.create_category_index(
 275         groundtruth_data['categories'])
 276 
 277     annotations_index = {}
 278     if 'annotations' in groundtruth_data:
 279       logging.info('Found groundtruth annotations. Building annotations index.')
 280       for annotation in groundtruth_data['annotations']:
 281         image_id = annotation['image_id']
 282         if image_id not in annotations_index:
 283           annotations_index[image_id] = []
 284         annotations_index[image_id].append(annotation)
 285     missing_annotation_count = 0
 286     for image in images:
 287       image_id = image['id']
 288       if image_id not in annotations_index:
 289         missing_annotation_count += 1
 290         annotations_index[image_id] = []
 291     logging.info('%d images are missing annotations.', missing_annotation_count)
 292 
 293     keypoint_annotations_index = {}
 294     if keypoint_annotations_file:
 295       with tf.gfile.GFile(keypoint_annotations_file, 'r') as kid:
 296         keypoint_groundtruth_data = json.load(kid)
 297       if 'annotations' in keypoint_groundtruth_data:
 298         for annotation in keypoint_groundtruth_data['annotations']:
 299           image_id = annotation['image_id']
 300           if image_id not in keypoint_annotations_index:
 301             keypoint_annotations_index[image_id] = {}
 302           keypoint_annotations_index[image_id][annotation['id']] = annotation
 303 
 304     total_num_annotations_skipped = 0
 305     total_num_keypoint_annotations_skipped = 0
 306     for idx, image in enumerate(images):
 307       if idx % 100 == 0:
 308         logging.info('On image %d of %d', idx, len(images))
 309       annotations_list = annotations_index[image['id']]
 310       keypoint_annotations_dict = None
 311       if keypoint_annotations_file:
 312         keypoint_annotations_dict = {}
 313         if image['id'] in keypoint_annotations_index:
 314           keypoint_annotations_dict = keypoint_annotations_index[image['id']]
 315       (_, tf_example, num_annotations_skipped,
 316        num_keypoint_annotations_skipped) = create_tf_example(
 317            image, annotations_list, image_dir, category_index, include_masks,
 318            keypoint_annotations_dict)
 319       total_num_annotations_skipped += num_annotations_skipped
 320       total_num_keypoint_annotations_skipped += num_keypoint_annotations_skipped
 321       shard_idx = idx % num_shards
 322       output_tfrecords[shard_idx].write(tf_example.SerializeToString())
 323     logging.info('Finished writing, skipped %d annotations.',
 324                  total_num_annotations_skipped)
 325     if keypoint_annotations_file:
 326       logging.info('Finished writing, skipped %d keypoint annotations.',
 327                    total_num_keypoint_annotations_skipped)

 330 def main(_):
 331   assert FLAGS.train_image_dir, '`train_image_dir` missing.'
 332   assert FLAGS.val_image_dir, '`val_image_dir` missing.'
 333   assert FLAGS.test_image_dir, '`test_image_dir` missing.'
 334   assert FLAGS.train_annotations_file, '`train_annotations_file` missing.'
 335   assert FLAGS.val_annotations_file, '`val_annotations_file` missing.'
 336   assert FLAGS.testdev_annotations_file, '`testdev_annotations_file` missing.'
 337 
 338   if not tf.gfile.IsDirectory(FLAGS.output_dir):
 339     tf.gfile.MakeDirs(FLAGS.output_dir)
 340   train_output_path = os.path.join(FLAGS.output_dir, 'coco_train.record')
 341   val_output_path = os.path.join(FLAGS.output_dir, 'coco_val.record')
 342   testdev_output_path = os.path.join(FLAGS.output_dir, 'coco_testdev.record')
 343 
 344   _create_tf_record_from_coco_annotations(
 345       FLAGS.train_annotations_file,
 346       FLAGS.train_image_dir,
 347       train_output_path,
 348       FLAGS.include_masks,
 349       num_shards=100,
 350       keypoint_annotations_file=FLAGS.train_keypoint_annotations_file)
 351   _create_tf_record_from_coco_annotations(
 352       FLAGS.val_annotations_file,
 353       FLAGS.val_image_dir,
 354       val_output_path,
 355       FLAGS.include_masks,
 356       num_shards=100,
 357       keypoint_annotations_file=FLAGS.val_keypoint_annotations_file)
 358   _create_tf_record_from_coco_annotations(
 359       FLAGS.testdev_annotations_file,
 360       FLAGS.test_image_dir,
 361       testdev_output_path,
 362       FLAGS.include_masks,
 363       num_shards=100)
 364 
 365 
 366 if __name__ == '__main__':
 367   tf.app.run()
