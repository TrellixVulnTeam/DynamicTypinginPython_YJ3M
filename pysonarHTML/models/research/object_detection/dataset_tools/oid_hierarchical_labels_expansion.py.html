  16 r"""An executable to expand image-level labels, boxes and segments.
  17 
  18 The expansion is performed using class hierarchy, provided in JSON file.
  19 
  20 The expected file formats are the following:
  21 - for box and segment files: CSV file is expected to have LabelName field
  22 - for image-level labels: CSV file is expected to have LabelName and Confidence
  23 fields
  24 
  25 Note, that LabelName is the only field used for expansion.
  26 
  27 Example usage:
  28 python models/research/object_detection/dataset_tools/\
  29 oid_hierarchical_labels_expansion.py \
  30 --json_hierarchy_file=<path to JSON hierarchy> \
  31 --input_annotations=<input csv file> \
  32 --output_annotations=<output csv file> \
  33 --annotation_type=<1 (for boxes and segments) or 2 (for image-level labels)>
  34 """

  62 def _update_dict(initial_dict, update):
  63   """Updates dictionary with update content.
  64 
  65   Args:
  66    initial_dict: initial dictionary.
  67    update: updated dictionary.
  68   """
  69 
  70   for key, value_list in update.items():
  71     if key in initial_dict:
  72       initial_dict[key].update(value_list)
  73     else:
  74       initial_dict[key] = set(value_list)

  77 def _build_plain_hierarchy(hierarchy, skip_root=False):
  78   """Expands tree hierarchy representation to parent-child dictionary.
  79 
  80   Args:
  81    hierarchy: labels hierarchy as JSON file.
  82    skip_root: if true skips root from the processing (done for the case when all
  83      classes under hierarchy are collected under virtual node).
  84 
  85   Returns:
  86     keyed_parent - dictionary of parent - all its children nodes.
  87     keyed_child  - dictionary of children - all its parent nodes
  88     children - all children of the current node.
  89   """
  90   all_children = set([])
  91   all_keyed_parent = {}
  92   all_keyed_child = {}
  93   if 'Subcategory' in hierarchy:
  94     for node in hierarchy['Subcategory']:
  95       keyed_parent, keyed_child, children = _build_plain_hierarchy(node)
  96       # Update is not done through dict.update() since some children have multi-
  97       # ple parents in the hiearchy.
  98       _update_dict(all_keyed_parent, keyed_parent)
  99       _update_dict(all_keyed_child, keyed_child)
 100       all_children.update(children)
 101 
 102   if not skip_root:
 103     all_keyed_parent[hierarchy['LabelName']] = copy.deepcopy(all_children)
 104     all_children.add(hierarchy['LabelName'])
 105     for child, _ in all_keyed_child.items():
 106       all_keyed_child[child].add(hierarchy['LabelName'])
 107     all_keyed_child[hierarchy['LabelName']] = set([])
 108 
 109   return all_keyed_parent, all_keyed_child, all_children

 112 class OIDHierarchicalLabelsExpansion(object):
 113   """ Main class to perform labels hierachical expansion."""
 114 
 115   def __init__(self, hierarchy):
 116     """Constructor.
 117 
 118     Args:
 119       hierarchy: labels hierarchy as JSON object.
 120     """
 121 
 122     self._hierarchy_keyed_parent, self._hierarchy_keyed_child, _ = (
 123         _build_plain_hierarchy(hierarchy, skip_root=True))
 124 
 125   def expand_boxes_or_segments_from_csv(self, csv_row,
 126                                         labelname_column_index=1):
 127     """Expands a row containing bounding boxes/segments from CSV file.
 128 
 129     Args:
 130       csv_row: a single row of Open Images released groundtruth file.
 131       labelname_column_index: 0-based index of LabelName column in CSV file.
 132 
 133     Returns:
 134       a list of strings (including the initial row) corresponding to the ground
 135       truth expanded to multiple annotation for evaluation with Open Images
 136       Challenge 2018/2019 metrics.
 137     """
 138     # Row header is expected to be the following for boxes:
 139     # ImageID,LabelName,Confidence,XMin,XMax,YMin,YMax,IsGroupOf
 140     # Row header is expected to be the following for segments:
 141     # ImageID,LabelName,ImageWidth,ImageHeight,XMin,XMax,YMin,YMax,
 142     # IsGroupOf,Mask
 143     split_csv_row = six.ensure_str(csv_row).split(',')
 144     result = [csv_row]
 145     assert split_csv_row[
 146         labelname_column_index] in self._hierarchy_keyed_child
 147     parent_nodes = self._hierarchy_keyed_child[
 148         split_csv_row[labelname_column_index]]
 149     for parent_node in parent_nodes:
 150       split_csv_row[labelname_column_index] = parent_node
 151       result.append(','.join(split_csv_row))
 152     return result
 153 
 154   def expand_labels_from_csv(self,
 155                              csv_row,
 156                              labelname_column_index=1,
 157                              confidence_column_index=2):
 158     """Expands a row containing labels from CSV file.
 159 
 160     Args:
 161       csv_row: a single row of Open Images released groundtruth file.
 162       labelname_column_index: 0-based index of LabelName column in CSV file.
 163       confidence_column_index: 0-based index of Confidence column in CSV file.
 164 
 165     Returns:
 166       a list of strings (including the initial row) corresponding to the ground
 167       truth expanded to multiple annotation for evaluation with Open Images
 168       Challenge 2018/2019 metrics.
 169     """
 170     # Row header is expected to be exactly:
 171     # ImageID,Source,LabelName,Confidence
 172     split_csv_row = six.ensure_str(csv_row).split(',')
 173     result = [csv_row]
 174     if int(split_csv_row[confidence_column_index]) == 1:
 175       assert split_csv_row[
 176           labelname_column_index] in self._hierarchy_keyed_child
 177       parent_nodes = self._hierarchy_keyed_child[
 178           split_csv_row[labelname_column_index]]
 179       for parent_node in parent_nodes:
 180         split_csv_row[labelname_column_index] = parent_node
 181         result.append(','.join(split_csv_row))
 182     else:
 183       assert split_csv_row[
 184           labelname_column_index] in self._hierarchy_keyed_parent
 185       child_nodes = self._hierarchy_keyed_parent[
 186           split_csv_row[labelname_column_index]]
 187       for child_node in child_nodes:
 188         split_csv_row[labelname_column_index] = child_node
 189         result.append(','.join(split_csv_row))
 190     return result

 193 def main(unused_args):
 194 
 195   del unused_args
 196 
 197   with open(FLAGS.json_hierarchy_file) as f:
 198     hierarchy = json.load(f)
 199   expansion_generator = OIDHierarchicalLabelsExpansion(hierarchy)
 200   labels_file = False
 201   if FLAGS.annotation_type == 2:
 202     labels_file = True
 203   elif FLAGS.annotation_type != 1:
 204     print('--annotation_type expected value is 1 or 2.')
 205     return -1
 206   confidence_column_index = -1
 207   labelname_column_index = -1
 208   with open(FLAGS.input_annotations, 'r') as source:
 209     with open(FLAGS.output_annotations, 'w') as target:
 210       header = source.readline()
 211       target.writelines([header])
 212       column_names = header.strip().split(',')
 213       labelname_column_index = column_names.index('LabelName')
 214       if labels_file:
 215         confidence_column_index = column_names.index('Confidence')
 216       for line in source:
 217         if labels_file:
 218           expanded_lines = expansion_generator.expand_labels_from_csv(
 219               line, labelname_column_index, confidence_column_index)
 220         else:
 221           expanded_lines = (
 222               expansion_generator.expand_boxes_or_segments_from_csv(
 223                   line, labelname_column_index))
 224         target.writelines(expanded_lines)
 225
