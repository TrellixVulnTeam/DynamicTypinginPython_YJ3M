  45 def get_prediction_tensor_shapes(pipeline_config):
  46   """Gets static shapes of tensors by building the graph on CPU.
  47 
  48   This function builds the graph on CPU and obtain static shapes of output
  49   tensors from TPUPartitionedCall. Shapes information are later used for setting
  50   shapes of tensors when TPU graphs are built. This is necessary because tensors
  51   coming out of TPUPartitionedCall lose their shape information, which are
  52   needed for a lot of CPU operations later.
  53   Args:
  54     pipeline_config: A TrainEvalPipelineConfig proto.
  55 
  56   Returns:
  57     A python dict of tensors' names and their shapes.
  58   """
  59   detection_model = model_builder.build(
  60       pipeline_config.model, is_training=False)
  61   _, input_tensors = exporter.input_placeholder_fn_map['image_tensor']()
  62   inputs = tf.cast(input_tensors, dtype=tf.float32)
  63   preprocessed_inputs, true_image_shapes = detection_model.preprocess(inputs)
  64   prediction_dict = detection_model.predict(preprocessed_inputs,
  65                                             true_image_shapes)
  66 
  67   return {
  68       BOX_ENCODINGS:
  69           prediction_dict[BOX_ENCODINGS].shape.as_list(),
  70       CLASS_PREDICTIONS_WITH_BACKGROUND:
  71           prediction_dict[CLASS_PREDICTIONS_WITH_BACKGROUND].shape.as_list(),
  72       ANCHORS:
  73           prediction_dict[ANCHORS].shape.as_list(),
  74   }

  77 def recover_shape(preprocessed_inputs, prediction_outputs, shapes_info):
  78   """Recovers shape from TPUPartitionedCall.
  79 
  80   Args:
  81     preprocessed_inputs: 4D tensor, shaped (batch, channels, height, width)
  82     prediction_outputs: Python list of tensors, in the following order -
  83       box_encodings - 3D tensor, shaped (code_size, batch, num_anchors);
  84       class_predictions_with_background - 3D tensor, shaped (num_classes + 1,
  85       batch, num_anchors); anchors - 2D tensor, shaped (4, num_anchors)
  86     shapes_info: Python dict of tensor shapes as lists.
  87 
  88   Returns:
  89     preprocessed_inputs: 4D tensor, shaped (batch, height, width, channels)
  90     box_encodings: 3D tensor, shaped (batch, num_anchors, code_size)
  91     class_predictions_with_background: 3D tensor,
  92         shaped (batch, num_anchors, num_classes + 1)
  93     anchors: 2D tensor, shaped (num_anchors, 4)
  94   """
  95   # Dimshuffle: (b, c, h, w) -> (b, h, w, c)
  96   preprocessed_inputs = tf.transpose(preprocessed_inputs, perm=[0, 2, 3, 1])
  97 
  98   box_encodings = tf.transpose(prediction_outputs[0], perm=[1, 2, 0])
  99   # [None, None, detection_model._box_coder.code_size]
 100   box_encodings.set_shape(shapes_info[BOX_ENCODINGS])
 101 
 102   class_predictions_with_background = tf.transpose(
 103       prediction_outputs[1], perm=[1, 2, 0])
 104   # [None, None, num_classes + 1]
 105   class_predictions_with_background.set_shape(
 106       shapes_info[CLASS_PREDICTIONS_WITH_BACKGROUND])
 107 
 108   anchors = tf.transpose(prediction_outputs[2], perm=[1, 0])
 109   # [None, 4]
 110   anchors.set_shape(shapes_info[ANCHORS])
 111 
 112   return (preprocessed_inputs, box_encodings, class_predictions_with_background,
 113           anchors)

 116 def build_graph(pipeline_config,
 117                 shapes_info,
 118                 input_type='encoded_image_string_tensor',
 119                 use_bfloat16=False):
 120   """Builds TPU serving graph of ssd to be exported.
 121 
 122   Args:
 123     pipeline_config: A TrainEvalPipelineConfig proto.
 124     shapes_info: A python dict of tensors' names and their shapes, returned by
 125       `get_prediction_tensor_shapes()`.
 126     input_type: One of
 127                 'encoded_image_string_tensor': a 1d tensor with dtype=tf.string
 128                 'image_tensor': a 4d tensor with dtype=tf.uint8
 129                 'tf_example': a 1d tensor with dtype=tf.string
 130     use_bfloat16: If true, use tf.bfloat16 on TPU.
 131 
 132   Returns:
 133     placeholder_tensor: A placeholder tensor, type determined by `input_type`.
 134     result_tensor_dict: A python dict of tensors' names and tensors.
 135   """
 136 
 137   detection_model = model_builder.build(
 138       pipeline_config.model, is_training=False)
 139 
 140   placeholder_tensor, input_tensors = \
 141       exporter.input_placeholder_fn_map[input_type]()
 142 
 143   inputs = tf.cast(input_tensors, dtype=tf.float32)
 144   preprocessed_inputs, true_image_shapes = detection_model.preprocess(inputs)
 145 
 146   # Dimshuffle: (b, h, w, c) -> (b, c, h, w)
 147   # This is to avoid extra padding due to TPU memory layout:
 148   # We swap larger dimensions in and smaller dimensions out, so that small
 149   # dimensions don't get padded tens / hundreds times of its own size.
 150   # This trick is applied to other similar tensors below.
 151   preprocessed_inputs = tf.transpose(preprocessed_inputs, perm=[0, 3, 1, 2])
 152   if use_bfloat16:
 153     preprocessed_inputs = tf.cast(preprocessed_inputs, dtype=tf.bfloat16)
 154 
 155   def predict_tpu_subgraph(preprocessed_inputs, true_image_shapes):
 156     """Wraps over the CPU version of `predict()`.
 157 
 158     This builds a same graph as the original `predict()`, manipulates
 159     result tensors' dimensions to be memory efficient on TPU, and
 160     returns them as list of tensors.
 161 
 162     Args:
 163       preprocessed_inputs: A 4D tensor of shape (batch, channels, height, width)
 164       true_image_shapes: True image shapes tensor.
 165 
 166     Returns:
 167       A Python list of tensors:
 168         box_encodings: 3D tensor of shape (code_size, batch_size, num_anchors)
 169         class_predictions_with_background: 3D tensor,
 170             shape (num_classes + 1, batch_size, num_anchors)
 171         anchors: 2D tensor of shape (4, num_anchors)
 172     """
 173     # Dimshuffle: (b, c, h, w) -> (b, h, w, c)
 174     preprocessed_inputs = tf.transpose(preprocessed_inputs, perm=[0, 2, 3, 1])
 175     if use_bfloat16:
 176       with bfloat16_scope():
 177         prediction_dict = detection_model.predict(preprocessed_inputs,
 178                                                   true_image_shapes)
 179     else:
 180       prediction_dict = detection_model.predict(preprocessed_inputs,
 181                                                 true_image_shapes)
 182 
 183     # Dimshuffle: (batch, anchors, depth) -> (depth, batch, anchors)
 184     return [
 185         tf.transpose(prediction_dict[BOX_ENCODINGS], perm=[2, 0, 1]),
 186         tf.transpose(
 187             prediction_dict[CLASS_PREDICTIONS_WITH_BACKGROUND], perm=[2, 0, 1]),
 188         tf.transpose(prediction_dict[ANCHORS], perm=[1, 0]),
 189     ]
 190 
 191   @function.Defun(capture_resource_var_by_value=False)
 192   def predict_tpu():
 193     return tpu.rewrite(predict_tpu_subgraph,
 194                        [preprocessed_inputs, true_image_shapes])
 195 
 196   prediction_outputs = tpu_functional.TPUPartitionedCall(
 197       args=predict_tpu.captured_inputs,
 198       device_ordinal=tpu_ops.tpu_ordinal_selector(),
 199       Tout=[o.type for o in predict_tpu.definition.signature.output_arg],
 200       f=predict_tpu)
 201 
 202   (preprocessed_inputs, box_encodings, class_predictions_with_background,
 203    anchors) = recover_shape(preprocessed_inputs, prediction_outputs,
 204                             shapes_info)
 205 
 206   output_tensors = {
 207       'preprocessed_inputs': preprocessed_inputs,
 208       BOX_ENCODINGS: box_encodings,
 209       CLASS_PREDICTIONS_WITH_BACKGROUND: class_predictions_with_background,
 210       ANCHORS: anchors,
 211   }
 212 
 213   if use_bfloat16:
 214     output_tensors = utils.bfloat16_to_float32_nested(output_tensors)
 215 
 216   postprocessed_tensors = detection_model.postprocess(output_tensors,
 217                                                       true_image_shapes)
 218   result_tensor_dict = exporter.add_output_tensor_nodes(postprocessed_tensors,
 219                                                         'inference_op')
 220 
 221   return placeholder_tensor, result_tensor_dict
