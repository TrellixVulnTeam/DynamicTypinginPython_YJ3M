  39 def parse_pipeline_config(pipeline_config_file):
  40   """Returns pipeline config and meta architecture name."""
  41   with tf.gfile.GFile(pipeline_config_file, 'r') as config_file:
  42     config_str = config_file.read()
  43   pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
  44   text_format.Merge(config_str, pipeline_config)
  45   meta_arch = pipeline_config.model.WhichOneof('model')
  46 
  47   return pipeline_config, meta_arch

  50 def export(pipeline_config_file,
  51            ckpt_path,
  52            export_dir,
  53            input_placeholder_name='placeholder_tensor',
  54            input_type='encoded_image_string_tensor',
  55            use_bfloat16=False):
  56   """Exports as SavedModel.
  57 
  58   Args:
  59     pipeline_config_file: Pipeline config file name.
  60     ckpt_path: Training checkpoint path.
  61     export_dir: Directory to export SavedModel.
  62     input_placeholder_name: input placeholder's name in SavedModel signature.
  63     input_type: One of
  64                 'encoded_image_string_tensor': a 1d tensor with dtype=tf.string
  65                 'image_tensor': a 4d tensor with dtype=tf.uint8
  66                 'tf_example': a 1d tensor with dtype=tf.string
  67     use_bfloat16: If true, use tf.bfloat16 on TPU.
  68   """
  69   pipeline_config, meta_arch = parse_pipeline_config(pipeline_config_file)
  70 
  71   shapes_info = model_map[meta_arch].get_prediction_tensor_shapes(
  72       pipeline_config)
  73 
  74   with tf.Graph().as_default(), tf.Session() as sess:
  75     placeholder_tensor, result_tensor_dict = model_map[meta_arch].build_graph(
  76         pipeline_config, shapes_info, input_type, use_bfloat16)
  77 
  78     saver = tf.train.Saver()
  79     init_op = tf.global_variables_initializer()
  80 
  81     sess.run(init_op)
  82     if ckpt_path is not None:
  83       saver.restore(sess, ckpt_path)
  84 
  85     # export saved model
  86     builder = tf.saved_model.builder.SavedModelBuilder(export_dir)
  87     tensor_info_inputs = {
  88         input_placeholder_name:
  89             tf.saved_model.utils.build_tensor_info(placeholder_tensor)
  90     }
  91     tensor_info_outputs = {
  92         k: tf.saved_model.utils.build_tensor_info(v)
  93         for k, v in result_tensor_dict.items()
  94     }
  95     detection_signature = (
  96         tf.saved_model.signature_def_utils.build_signature_def(
  97             inputs=tensor_info_inputs,
  98             outputs=tensor_info_outputs,
  99             method_name=tf.saved_model.signature_constants.PREDICT_METHOD_NAME))
 100 
 101     tf.logging.info('Inputs:\n{}\nOutputs:{}\nPredict method name:{}'.format(
 102         tensor_info_inputs, tensor_info_outputs,
 103         tf.saved_model.signature_constants.PREDICT_METHOD_NAME))
 104     # Graph for TPU.
 105     builder.add_meta_graph_and_variables(
 106         sess, [
 107             tf.saved_model.tag_constants.SERVING,
 108             tf.saved_model.tag_constants.TPU
 109         ],
 110         signature_def_map={
 111             tf.saved_model.signature_constants
 112             .DEFAULT_SERVING_SIGNATURE_DEF_KEY:
 113                 detection_signature,
 114         },
 115         strip_default_attrs=True)
 116     # Graph for CPU, this is for passing infra validation.
 117     builder.add_meta_graph(
 118         [tf.saved_model.tag_constants.SERVING],
 119         signature_def_map={
 120             tf.saved_model.signature_constants
 121             .DEFAULT_SERVING_SIGNATURE_DEF_KEY:
 122                 detection_signature,
 123         },
 124         strip_default_attrs=True)
 125     builder.save(as_text=False)
 126     tf.logging.info('Model saved to {}'.format(export_dir))

 129 def run_inference(inputs,
 130                   pipeline_config_file,
 131                   ckpt_path,
 132                   input_type='encoded_image_string_tensor',
 133                   use_bfloat16=False,
 134                   repeat=1):
 135   """Runs inference on TPU.
 136 
 137   Args:
 138     inputs: Input image with the same type as `input_type`
 139     pipeline_config_file: Pipeline config file name.
 140     ckpt_path: Training checkpoint path.
 141     input_type: One of
 142                 'encoded_image_string_tensor': a 1d tensor with dtype=tf.string
 143                 'image_tensor': a 4d tensor with dtype=tf.uint8
 144                 'tf_example': a 1d tensor with dtype=tf.string
 145     use_bfloat16: If true, use tf.bfloat16 on TPU.
 146     repeat: Number of times to repeat running the provided input for profiling.
 147 
 148   Returns:
 149     A dict of resulting tensors.
 150   """
 151 
 152   pipeline_config, meta_arch = parse_pipeline_config(pipeline_config_file)
 153 
 154   shapes_info = model_map[meta_arch].get_prediction_tensor_shapes(
 155       pipeline_config)
 156 
 157   with tf.Graph().as_default(), tf.Session() as sess:
 158     placeholder_tensor, result_tensor_dict = model_map[meta_arch].build_graph(
 159         pipeline_config, shapes_info, input_type, use_bfloat16)
 160 
 161     saver = tf.train.Saver()
 162     init_op = tf.global_variables_initializer()
 163 
 164     sess.run(tpu.initialize_system())
 165 
 166     sess.run(init_op)
 167     if ckpt_path is not None:
 168       saver.restore(sess, ckpt_path)
 169 
 170     for _ in range(repeat):
 171       tensor_dict_out = sess.run(
 172           result_tensor_dict, feed_dict={placeholder_tensor: [inputs]})
 173 
 174     sess.run(tpu.shutdown_system())
 175 
 176     return tensor_dict_out

 179 def run_inference_from_saved_model(inputs,
 180                                    saved_model_dir,
 181                                    input_placeholder_name='placeholder_tensor',
 182                                    repeat=1):
 183   """Loads saved model and run inference on TPU.
 184 
 185   Args:
 186     inputs: Input image with the same type as `input_type`
 187     saved_model_dir: The directory SavedModel being exported to.
 188     input_placeholder_name: input placeholder's name in SavedModel signature.
 189     repeat: Number of times to repeat running the provided input for profiling.
 190 
 191   Returns:
 192     A dict of resulting tensors.
 193   """
 194   with tf.Graph().as_default(), tf.Session() as sess:
 195     meta_graph = loader.load(sess, [tag_constants.SERVING, tag_constants.TPU],
 196                              saved_model_dir)
 197 
 198     sess.run(tpu.initialize_system())
 199 
 200     key_prediction = signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY
 201 
 202     tensor_name_input = (
 203         meta_graph.signature_def[key_prediction].inputs[input_placeholder_name]
 204         .name)
 205     tensor_name_output = {
 206         k: v.name
 207         for k, v in (meta_graph.signature_def[key_prediction].outputs.items())
 208     }
 209 
 210     for _ in range(repeat):
 211       tensor_dict_out = sess.run(
 212           tensor_name_output, feed_dict={tensor_name_input: [inputs]})
 213 
 214     sess.run(tpu.shutdown_system())
 215 
 216     return tensor_dict_out
