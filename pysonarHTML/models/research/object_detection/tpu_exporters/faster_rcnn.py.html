  61 def modify_config(pipeline_config):
  62   """Modifies pipeline config to build the correct graph for TPU."""
  63   # faster_rcnn.use_static_shapes and faster_rcnn.use_static_shapes_for_eval
  64   # are set to True in order for detection_model.use_static_shapes to be True.
  65   # We need to set this so that clip_to_window in _predict_first_stage
  66   # can work on TPU. However as a side-effect, the flag forces the use of
  67   # padded version of NMS.
  68   pipeline_config.model.faster_rcnn.use_static_shapes = True
  69   pipeline_config.model.faster_rcnn.use_static_shapes_for_eval = True
  70   pipeline_config.model.faster_rcnn.use_matmul_crop_and_resize = True
  71   pipeline_config.model.faster_rcnn.clip_anchors_to_image = True
  72   return pipeline_config

  75 def get_prediction_tensor_shapes(pipeline_config):
  76   """Gets static shapes of tensors by building the graph on CPU.
  77 
  78   This function builds the graph on CPU and obtain static shapes of output
  79   tensors from TPUPartitionedCall. Shapes information are later used for setting
  80   shapes of tensors when TPU graphs are built. This is necessary because tensors
  81   coming out of TPUPartitionedCall lose their shape information, which are
  82   needed for a lot of CPU operations later.
  83 
  84   Args:
  85     pipeline_config: A TrainEvalPipelineConfig proto.
  86 
  87   Returns:
  88     A python dict of tensors' names and their shapes.
  89   """
  90   pipeline_config = modify_config(pipeline_config)
  91   detection_model = INPUT_BUILDER_UTIL_MAP['model_build'](
  92       pipeline_config.model, is_training=False)
  93 
  94   _, input_tensors = exporter.input_placeholder_fn_map['image_tensor']()
  95 
  96   inputs = tf.cast(input_tensors, dtype=tf.float32)
  97   preprocessed_inputs, true_image_shapes = detection_model.preprocess(inputs)
  98 
  99   prediction_dict = detection_model.predict(preprocessed_inputs,
 100                                             true_image_shapes)
 101 
 102   shapes_info = {}
 103   for k, v in prediction_dict.items():
 104     if isinstance(v, list):
 105       shapes_info[k] = [item.shape.as_list() for item in v]
 106     else:
 107       shapes_info[k] = v.shape.as_list()
 108   return shapes_info

 111 def build_graph(pipeline_config,
 112                 shapes_info,
 113                 input_type='encoded_image_string_tensor',
 114                 use_bfloat16=True):
 115   """Builds serving graph of faster_rcnn to be exported.
 116 
 117   Args:
 118     pipeline_config: A TrainEvalPipelineConfig proto.
 119     shapes_info: A python dict of tensors' names and their shapes, returned by
 120       `get_prediction_tensor_shapes()`.
 121     input_type: One of
 122                 'encoded_image_string_tensor': a 1d tensor with dtype=tf.string
 123                 'image_tensor': a 4d tensor with dtype=tf.uint8
 124                 'tf_example': a 1d tensor with dtype=tf.string
 125     use_bfloat16: If true, use tf.bfloat16 on TPU.
 126 
 127   Returns:
 128     placeholder_tensor: A placeholder tensor, type determined by `input_type`.
 129     result_tensor_dict: A python dict of tensors' names and tensors.
 130   """
 131   pipeline_config = modify_config(pipeline_config)
 132   detection_model = INPUT_BUILDER_UTIL_MAP['model_build'](
 133       pipeline_config.model, is_training=False)
 134 
 135   placeholder_tensor, input_tensors = \
 136       exporter.input_placeholder_fn_map[input_type]()
 137 
 138   # CPU pre-processing
 139   inputs = tf.cast(input_tensors, dtype=tf.float32)
 140   preprocessed_inputs, true_image_shapes = detection_model.preprocess(inputs)
 141 
 142   # Dimshuffle: [b, h, w, c] -> [b, c, h, w]
 143   preprocessed_inputs = tf.transpose(preprocessed_inputs, perm=[0, 3, 1, 2])
 144   if use_bfloat16:
 145     preprocessed_inputs = tf.cast(preprocessed_inputs, dtype=tf.bfloat16)
 146 
 147   # TPU feature extraction
 148   def tpu_subgraph_predict_fn(preprocessed_inputs, true_image_shapes):
 149     """Defines the first part of graph on TPU."""
 150     # [b, c, h, w] -> [b, h, w, c]
 151     preprocessed_inputs = tf.transpose(preprocessed_inputs, perm=[0, 2, 3, 1])
 152 
 153     prediction_dict = detection_model.predict(preprocessed_inputs,
 154                                               true_image_shapes)
 155 
 156     return (
 157         # [batch, anchor, depth] -> [depth, batch, anchor]
 158         tf.transpose(prediction_dict[RPN_BOX_ENCODINGS], perm=[2, 0, 1]),
 159         # [batch, anchor, depth] -> [depth, batch, anchor]
 160         tf.transpose(
 161             prediction_dict[RPN_OBJECTNESS_PREDICTIONS_WITH_BACKGROUND],
 162             perm=[2, 0, 1]),
 163         # [anchors, depth]
 164         tf.transpose(prediction_dict[ANCHORS], perm=[1, 0]),
 165         # [num_proposals, num_classes, code_size]
 166         prediction_dict[REFINED_BOX_ENCODINGS],
 167         prediction_dict[CLASS_PREDICTIONS_WITH_BACKGROUND],
 168         prediction_dict[NUM_PROPOSALS],
 169         prediction_dict[PROPOSAL_BOXES])
 170 
 171   @function.Defun(capture_resource_var_by_value=False)
 172   def tpu_subgraph_predict():
 173     if use_bfloat16:
 174       with bfloat16_scope():
 175         return tpu.rewrite(tpu_subgraph_predict_fn,
 176                            [preprocessed_inputs, true_image_shapes])
 177     else:
 178       return tpu.rewrite(tpu_subgraph_predict_fn,
 179                          [preprocessed_inputs, true_image_shapes])
 180 
 181   (rpn_box_encodings, rpn_objectness_predictions_with_background, anchors,
 182    refined_box_encodings, class_predictions_with_background, num_proposals,
 183    proposal_boxes) = tpu_functional.TPUPartitionedCall(
 184        args=tpu_subgraph_predict.captured_inputs,
 185        device_ordinal=tpu_ops.tpu_ordinal_selector(),
 186        Tout=[
 187            o.type for o in tpu_subgraph_predict.definition.signature.output_arg
 188        ],
 189        f=tpu_subgraph_predict)
 190 
 191   prediction_dict = {
 192       RPN_BOX_ENCODINGS:
 193           tf.transpose(rpn_box_encodings, perm=[1, 2, 0]),
 194       RPN_OBJECTNESS_PREDICTIONS_WITH_BACKGROUND:
 195           tf.transpose(
 196               rpn_objectness_predictions_with_background, perm=[1, 2, 0]),
 197       ANCHORS:
 198           tf.transpose(anchors, perm=[1, 0]),
 199       REFINED_BOX_ENCODINGS:
 200           refined_box_encodings,
 201       CLASS_PREDICTIONS_WITH_BACKGROUND:
 202           class_predictions_with_background,
 203       NUM_PROPOSALS:
 204           num_proposals,
 205       PROPOSAL_BOXES:
 206           proposal_boxes
 207   }
 208 
 209   for k in prediction_dict:
 210     if isinstance(prediction_dict[k], list):
 211       prediction_dict[k] = [
 212           prediction_dict[k][idx].set_shape(shapes_info[k][idx])
 213           for idx in len(prediction_dict[k])]
 214     else:
 215       prediction_dict[k].set_shape(shapes_info[k])
 216 
 217   if use_bfloat16:
 218     prediction_dict = utils.bfloat16_to_float32_nested(prediction_dict)
 219 
 220   # CPU post-processing (NMS)
 221   postprocessed_tensors = detection_model.postprocess(prediction_dict,
 222                                                       true_image_shapes)
 223   result_tensor_dict = exporter.add_output_tensor_nodes(postprocessed_tensors,
 224                                                         'inference_op')
 225 
 226   return placeholder_tensor, result_tensor_dict
