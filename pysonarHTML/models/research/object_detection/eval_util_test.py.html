  62   def _make_evaluation_dict(self,
  63                             resized_groundtruth_masks=False,
  64                             batch_size=1,
  65                             max_gt_boxes=None,
  66                             scale_to_absolute=False):
  67     input_data_fields = fields.InputDataFields
  68     detection_fields = fields.DetectionResultFields
  69 
  70     image = tf.zeros(shape=[batch_size, 20, 20, 3], dtype=tf.uint8)
  71     if batch_size == 1:
  72       key = tf.constant('image1')
  73     else:
  74       key = tf.constant([str(i) for i in range(batch_size)])
  75     detection_boxes = tf.tile(tf.constant([[[0., 0., 1., 1.]]]),
  76                               multiples=[batch_size, 1, 1])
  77     detection_scores = tf.tile(tf.constant([[0.8]]), multiples=[batch_size, 1])
  78     detection_classes = tf.tile(tf.constant([[0]]), multiples=[batch_size, 1])
  79     detection_masks = tf.tile(tf.ones(shape=[1, 1, 20, 20], dtype=tf.float32),
  80                               multiples=[batch_size, 1, 1, 1])
  81     num_detections = tf.ones([batch_size])
  82     groundtruth_boxes = tf.constant([[0., 0., 1., 1.]])
  83     groundtruth_classes = tf.constant([1])
  84     groundtruth_instance_masks = tf.ones(shape=[1, 20, 20], dtype=tf.uint8)
  85     groundtruth_keypoints = tf.constant([[0.0, 0.0], [0.5, 0.5], [1.0, 1.0]])
  86     if resized_groundtruth_masks:
  87       groundtruth_instance_masks = tf.ones(shape=[1, 10, 10], dtype=tf.uint8)
  88 
  89     if batch_size > 1:
  90       groundtruth_boxes = tf.tile(tf.expand_dims(groundtruth_boxes, 0),
  91                                   multiples=[batch_size, 1, 1])
  92       groundtruth_classes = tf.tile(tf.expand_dims(groundtruth_classes, 0),
  93                                     multiples=[batch_size, 1])
  94       groundtruth_instance_masks = tf.tile(
  95           tf.expand_dims(groundtruth_instance_masks, 0),
  96           multiples=[batch_size, 1, 1, 1])
  97       groundtruth_keypoints = tf.tile(
  98           tf.expand_dims(groundtruth_keypoints, 0),
  99           multiples=[batch_size, 1, 1])
 100 
 101     detections = {
 102         detection_fields.detection_boxes: detection_boxes,
 103         detection_fields.detection_scores: detection_scores,
 104         detection_fields.detection_classes: detection_classes,
 105         detection_fields.detection_masks: detection_masks,
 106         detection_fields.num_detections: num_detections
 107     }
 108     groundtruth = {
 109         input_data_fields.groundtruth_boxes: groundtruth_boxes,
 110         input_data_fields.groundtruth_classes: groundtruth_classes,
 111         input_data_fields.groundtruth_keypoints: groundtruth_keypoints,
 112         input_data_fields.groundtruth_instance_masks: groundtruth_instance_masks
 113     }
 114     if batch_size > 1:
 115       return eval_util.result_dict_for_batched_example(
 116           image, key, detections, groundtruth,
 117           scale_to_absolute=scale_to_absolute,
 118           max_gt_boxes=max_gt_boxes)
 119     else:
 120       return eval_util.result_dict_for_single_example(
 121           image, key, detections, groundtruth,
 122           scale_to_absolute=scale_to_absolute)

 130   def test_get_eval_metric_ops_for_coco_detections(self, batch_size=1,
 131                                                    max_gt_boxes=None,
 132                                                    scale_to_absolute=False):
 133     eval_config = eval_pb2.EvalConfig()
 134     eval_config.metrics_set.extend(['coco_detection_metrics'])
 135     categories = self._get_categories_list()
 136     eval_dict = self._make_evaluation_dict(batch_size=batch_size,
 137                                            max_gt_boxes=max_gt_boxes,
 138                                            scale_to_absolute=scale_to_absolute)
 139     metric_ops = eval_util.get_eval_metric_ops_for_evaluators(
 140         eval_config, categories, eval_dict)
 141     _, update_op = metric_ops['DetectionBoxes_Precision/mAP']
 142 
 143     with self.test_session() as sess:
 144       metrics = {}
 145       for key, (value_op, _) in six.iteritems(metric_ops):
 146         metrics[key] = value_op
 147       sess.run(update_op)
 148       metrics = sess.run(metrics)
 149       self.assertAlmostEqual(1.0, metrics['DetectionBoxes_Precision/mAP'])
 150       self.assertNotIn('DetectionMasks_Precision/mAP', metrics)

 222   def test_get_eval_metric_ops_for_evaluators(self):
 223     eval_config = eval_pb2.EvalConfig()
 224     eval_config.metrics_set.extend([
 225         'coco_detection_metrics', 'coco_mask_metrics',
 226         'precision_at_recall_detection_metrics'
 227     ])
 228     eval_config.include_metrics_per_category = True
 229     eval_config.recall_lower_bound = 0.2
 230     eval_config.recall_upper_bound = 0.6
 231 
 232     evaluator_options = eval_util.evaluator_options_from_eval_config(
 233         eval_config)
 234     self.assertTrue(evaluator_options['coco_detection_metrics']
 235                     ['include_metrics_per_category'])
 236     self.assertTrue(
 237         evaluator_options['coco_mask_metrics']['include_metrics_per_category'])
 238     self.assertAlmostEqual(
 239         evaluator_options['precision_at_recall_detection_metrics']
 240         ['recall_lower_bound'], eval_config.recall_lower_bound)
 241     self.assertAlmostEqual(
 242         evaluator_options['precision_at_recall_detection_metrics']
 243         ['recall_upper_bound'], eval_config.recall_upper_bound)

 245   def test_get_evaluator_with_evaluator_options(self):
 246     eval_config = eval_pb2.EvalConfig()
 247     eval_config.metrics_set.extend(
 248         ['coco_detection_metrics', 'precision_at_recall_detection_metrics'])
 249     eval_config.include_metrics_per_category = True
 250     eval_config.recall_lower_bound = 0.2
 251     eval_config.recall_upper_bound = 0.6
 252     categories = self._get_categories_list()
 253 
 254     evaluator_options = eval_util.evaluator_options_from_eval_config(
 255         eval_config)
 256     evaluator = eval_util.get_evaluators(eval_config, categories,
 257                                          evaluator_options)
 258 
 259     self.assertTrue(evaluator[0]._include_metrics_per_category)
 260     self.assertAlmostEqual(evaluator[1]._recall_lower_bound,
 261                            eval_config.recall_lower_bound)
 262     self.assertAlmostEqual(evaluator[1]._recall_upper_bound,
 263                            eval_config.recall_upper_bound)

 265   def test_get_evaluator_with_no_evaluator_options(self):
 266     eval_config = eval_pb2.EvalConfig()
 267     eval_config.metrics_set.extend(
 268         ['coco_detection_metrics', 'precision_at_recall_detection_metrics'])
 269     eval_config.include_metrics_per_category = True
 270     eval_config.recall_lower_bound = 0.2
 271     eval_config.recall_upper_bound = 0.6
 272     categories = self._get_categories_list()
 273 
 274     evaluator = eval_util.get_evaluators(
 275         eval_config, categories, evaluator_options=None)
 276 
 277     # Even though we are setting eval_config.include_metrics_per_category = True
 278     # and bounds on recall, these options are never passed into the
 279     # DetectionEvaluator constructor (via `evaluator_options`).
 280     self.assertFalse(evaluator[0]._include_metrics_per_category)
 281     self.assertAlmostEqual(evaluator[1]._recall_lower_bound, 0.0)
 282     self.assertAlmostEqual(evaluator[1]._recall_upper_bound, 1.0)

 284   def test_get_evaluator_with_keypoint_metrics(self):
 285     eval_config = eval_pb2.EvalConfig()
 286     person_keypoints_metric = eval_config.parameterized_metric.add()
 287     person_keypoints_metric.coco_keypoint_metrics.class_label = 'person'
 288     person_keypoints_metric.coco_keypoint_metrics.keypoint_label_to_sigmas[
 289         'left_eye'] = 0.1
 290     person_keypoints_metric.coco_keypoint_metrics.keypoint_label_to_sigmas[
 291         'right_eye'] = 0.2
 292     dog_keypoints_metric = eval_config.parameterized_metric.add()
 293     dog_keypoints_metric.coco_keypoint_metrics.class_label = 'dog'
 294     dog_keypoints_metric.coco_keypoint_metrics.keypoint_label_to_sigmas[
 295         'tail_start'] = 0.3
 296     dog_keypoints_metric.coco_keypoint_metrics.keypoint_label_to_sigmas[
 297         'mouth'] = 0.4
 298     categories = self._get_categories_list_with_keypoints()
 299 
 300     evaluator = eval_util.get_evaluators(
 301         eval_config, categories, evaluator_options=None)
 302 
 303     # Verify keypoint evaluator class variables.
 304     self.assertLen(evaluator, 3)
 305     self.assertFalse(evaluator[0]._include_metrics_per_category)
 306     self.assertEqual(evaluator[1]._category_name, 'person')
 307     self.assertEqual(evaluator[2]._category_name, 'dog')
 308     self.assertAllEqual(evaluator[1]._keypoint_ids, [0, 3])
 309     self.assertAllEqual(evaluator[2]._keypoint_ids, [1, 2])
 310     self.assertAllClose([0.1, 0.2], evaluator[1]._oks_sigmas)
 311     self.assertAllClose([0.3, 0.4], evaluator[2]._oks_sigmas)

 327   def test_padded_image_result_dict(self):
 328 
 329     input_data_fields = fields.InputDataFields
 330     detection_fields = fields.DetectionResultFields
 331     key = tf.constant([str(i) for i in range(2)])
 332 
 333     detection_boxes = np.array([[[0., 0., 1., 1.]], [[0.0, 0.0, 0.5, 0.5]]],
 334                                dtype=np.float32)
 335     detection_keypoints = np.array([[0.0, 0.0], [0.5, 0.5], [1.0, 1.0]],
 336                                    dtype=np.float32)
 337     detections = {
 338         detection_fields.detection_boxes:
 339             tf.constant(detection_boxes),
 340         detection_fields.detection_scores:
 341             tf.constant([[1.], [1.]]),
 342         detection_fields.detection_classes:
 343             tf.constant([[1], [2]]),
 344         detection_fields.num_detections:
 345             tf.constant([1, 1]),
 346         detection_fields.detection_keypoints:
 347             tf.tile(
 348                 tf.reshape(
 349                     tf.constant(detection_keypoints), shape=[1, 1, 3, 2]),
 350                 multiples=[2, 1, 1, 1])
 351     }
 352 
 353     gt_boxes = detection_boxes
 354     groundtruth = {
 355         input_data_fields.groundtruth_boxes:
 356             tf.constant(gt_boxes),
 357         input_data_fields.groundtruth_classes:
 358             tf.constant([[1.], [1.]]),
 359         input_data_fields.groundtruth_keypoints:
 360             tf.tile(
 361                 tf.reshape(
 362                     tf.constant(detection_keypoints), shape=[1, 1, 3, 2]),
 363                 multiples=[2, 1, 1, 1])
 364     }
 365 
 366     image = tf.zeros((2, 100, 100, 3), dtype=tf.float32)
 367 
 368     true_image_shapes = tf.constant([[100, 100, 3], [50, 100, 3]])
 369     original_image_spatial_shapes = tf.constant([[200, 200], [150, 300]])
 370 
 371     result = eval_util.result_dict_for_batched_example(
 372         image, key, detections, groundtruth,
 373         scale_to_absolute=True,
 374         true_image_shapes=true_image_shapes,
 375         original_image_spatial_shapes=original_image_spatial_shapes,
 376         max_gt_boxes=tf.constant(1))
 377 
 378     with self.test_session() as sess:
 379       result = sess.run(result)
 380       self.assertAllEqual(
 381           [[[0., 0., 200., 200.]], [[0.0, 0.0, 150., 150.]]],
 382           result[input_data_fields.groundtruth_boxes])
 383       self.assertAllClose([[[[0., 0.], [100., 100.], [200., 200.]]],
 384                            [[[0., 0.], [150., 150.], [300., 300.]]]],
 385                           result[input_data_fields.groundtruth_keypoints])
 386 
 387       # Predictions from the model are not scaled.
 388       self.assertAllEqual(
 389           [[[0., 0., 200., 200.]], [[0.0, 0.0, 75., 150.]]],
 390           result[detection_fields.detection_boxes])
 391       self.assertAllClose([[[[0., 0.], [100., 100.], [200., 200.]]],
 392                            [[[0., 0.], [75., 150.], [150., 300.]]]],
 393                           result[detection_fields.detection_keypoints])
