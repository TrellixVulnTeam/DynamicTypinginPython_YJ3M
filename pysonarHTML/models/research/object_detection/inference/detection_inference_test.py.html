  30 def get_mock_tfrecord_path():
  31   return os.path.join(tf.test.get_temp_dir(), 'mock.tfrec')
  32

  34 def create_mock_tfrecord():
  35   pil_image = Image.fromarray(np.array([[[123, 0, 0]]], dtype=np.uint8), 'RGB')
  36   image_output_stream = six.BytesIO()
  37   pil_image.save(image_output_stream, format='png')
  38   encoded_image = image_output_stream.getvalue()
  39 
  40   feature_map = {
  41       'test_field':
  42           dataset_util.float_list_feature([1, 2, 3, 4]),
  43       standard_fields.TfExampleFields.image_encoded:
  44           dataset_util.bytes_feature(encoded_image),
  45   }
  46 
  47   tf_example = tf.train.Example(features=tf.train.Features(feature=feature_map))
  48   with tf.python_io.TFRecordWriter(get_mock_tfrecord_path()) as writer:
  49     writer.write(tf_example.SerializeToString())
  50   return encoded_image

  53 def get_mock_graph_path():
  54   return os.path.join(tf.test.get_temp_dir(), 'mock_graph.pb')
  55 
  56 
  57 def create_mock_graph():
  58   g = tf.Graph()
  59   with g.as_default():
  60     in_image_tensor = tf.placeholder(
  61         tf.uint8, shape=[1, None, None, 3], name='image_tensor')
  62     tf.constant([2.0], name='num_detections')
  63     tf.constant(
  64         [[[0, 0.8, 0.7, 1], [0.1, 0.2, 0.8, 0.9], [0.2, 0.3, 0.4, 0.5]]],
  65         name='detection_boxes')
  66     tf.constant([[0.1, 0.2, 0.3]], name='detection_scores')
  67     tf.identity(
  68         tf.constant([[1.0, 2.0, 3.0]]) *
  69         tf.reduce_sum(tf.cast(in_image_tensor, dtype=tf.float32)),
  70         name='detection_classes')
  71     graph_def = g.as_graph_def()
  72 
  73   with tf.gfile.Open(get_mock_graph_path(), 'w') as fl:
  74     fl.write(graph_def.SerializeToString())
  75

  77 class InferDetectionsTests(tf.test.TestCase):
  78 
  79   def test_simple(self):
  80     create_mock_graph()
  81     encoded_image = create_mock_tfrecord()
  82 
  83     serialized_example_tensor, image_tensor = detection_inference.build_input(
  84         [get_mock_tfrecord_path()])
  85     self.assertAllEqual(image_tensor.get_shape().as_list(), [1, None, None, 3])
  86 
  87     (detected_boxes_tensor, detected_scores_tensor,
  88      detected_labels_tensor) = detection_inference.build_inference_graph(
  89          image_tensor, get_mock_graph_path())
  90 
  91     with self.test_session(use_gpu=False) as sess:
  92       sess.run(tf.global_variables_initializer())
  93       sess.run(tf.local_variables_initializer())
  94       tf.train.start_queue_runners()
  95 
  96       tf_example = detection_inference.infer_detections_and_add_to_example(
  97           serialized_example_tensor, detected_boxes_tensor,
  98           detected_scores_tensor, detected_labels_tensor, False)
  99     expected_example = tf.train.Example()
 100     text_format.Merge(r"""
 101         features {
 102           feature {
 103             key: "image/detection/bbox/ymin"
 104             value { float_list { value: [0.0, 0.1] } } }
 105           feature {
 106             key: "image/detection/bbox/xmin"
 107             value { float_list { value: [0.8, 0.2] } } }
 108           feature {
 109             key: "image/detection/bbox/ymax"
 110             value { float_list { value: [0.7, 0.8] } } }
 111           feature {
 112             key: "image/detection/bbox/xmax"
 113             value { float_list { value: [1.0, 0.9] } } }
 114           feature {
 115             key: "image/detection/label"
 116             value { int64_list { value: [123, 246] } } }
 117           feature {
 118             key: "image/detection/score"
 119             value { float_list { value: [0.1, 0.2] } } }
 120           feature {
 121             key: "test_field"
 122             value { float_list { value: [1.0, 2.0, 3.0, 4.0] } } } }""",
 123                       expected_example)
 124     expected_example.features.feature[
 125         standard_fields.TfExampleFields
 126         .image_encoded].CopyFrom(dataset_util.bytes_feature(encoded_image))
 127     self.assertProtoEquals(expected_example, tf_example)
