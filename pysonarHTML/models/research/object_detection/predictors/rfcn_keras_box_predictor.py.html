  27 class RfcnKerasBoxPredictor(box_predictor.KerasBoxPredictor):
  28   """RFCN Box Predictor.
  29 
  30   Applies a position sensitive ROI pooling on position sensitive feature maps to
  31   predict classes and refined locations. See https://arxiv.org/abs/1605.06409
  32   for details.
  33 
  34   This is used for the second stage of the RFCN meta architecture. Notice that
  35   locations are *not* shared across classes, thus for each anchor, a separate
  36   prediction is made for each class.
  37   """
  38 
  39   def __init__(self,
  40                is_training,
  41                num_classes,
  42                conv_hyperparams,
  43                freeze_batchnorm,
  44                num_spatial_bins,
  45                depth,
  46                crop_size,
  47                box_code_size,
  48                name=None):
  49     """Constructor.
  50 
  51     Args:
  52       is_training: Indicates whether the BoxPredictor is in training mode.
  53       num_classes: number of classes.  Note that num_classes *does not*
  54         include the background category, so if groundtruth labels take values
  55         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  56         assigned classification targets can range from {0,... K}).
  57       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  58         containing hyperparameters for convolution ops.
  59       freeze_batchnorm: Whether to freeze batch norm parameters during
  60         training or not. When training with a small batch size (e.g. 1), it is
  61         desirable to freeze batch norm update and use pretrained batch norm
  62         params.
  63       num_spatial_bins: A list of two integers `[spatial_bins_y,
  64         spatial_bins_x]`.
  65       depth: Target depth to reduce the input feature maps to.
  66       crop_size: A list of two integers `[crop_height, crop_width]`.
  67       box_code_size: Size of encoding for each box.
  68       name: A string name scope to assign to the box predictor. If `None`, Keras
  69         will auto-generate one from the class name.
  70     """
  71     super(RfcnKerasBoxPredictor, self).__init__(
  72         is_training, num_classes, freeze_batchnorm=freeze_batchnorm,
  73         inplace_batchnorm_update=False, name=name)
  74     self._freeze_batchnorm = freeze_batchnorm
  75     self._conv_hyperparams = conv_hyperparams
  76     self._num_spatial_bins = num_spatial_bins
  77     self._depth = depth
  78     self._crop_size = crop_size
  79     self._box_code_size = box_code_size
  80 
  81     # Build the shared layers used for both heads
  82     self._shared_conv_layers = []
  83     self._shared_conv_layers.append(
  84         tf.keras.layers.Conv2D(
  85             self._depth,
  86             [1, 1],
  87             padding='SAME',
  88             name='reduce_depth_conv',
  89             **self._conv_hyperparams.params()))
  90     self._shared_conv_layers.append(
  91         self._conv_hyperparams.build_batch_norm(
  92             training=(self._is_training and not self._freeze_batchnorm),
  93             name='reduce_depth_batchnorm'))
  94     self._shared_conv_layers.append(
  95         self._conv_hyperparams.build_activation_layer(
  96             name='reduce_depth_activation'))
  97 
  98     self._box_encoder_layers = []
  99     location_feature_map_depth = (self._num_spatial_bins[0] *
 100                                   self._num_spatial_bins[1] *
 101                                   self.num_classes *
 102                                   self._box_code_size)
 103     self._box_encoder_layers.append(
 104         tf.keras.layers.Conv2D(
 105             location_feature_map_depth,
 106             [1, 1],
 107             padding='SAME',
 108             name='refined_locations_conv',
 109             **self._conv_hyperparams.params()))
 110     self._box_encoder_layers.append(
 111         self._conv_hyperparams.build_batch_norm(
 112             training=(self._is_training and not self._freeze_batchnorm),
 113             name='refined_locations_batchnorm'))
 114 
 115     self._class_predictor_layers = []
 116     self._total_classes = self.num_classes + 1  # Account for background class.
 117     class_feature_map_depth = (self._num_spatial_bins[0] *
 118                                self._num_spatial_bins[1] *
 119                                self._total_classes)
 120     self._class_predictor_layers.append(
 121         tf.keras.layers.Conv2D(
 122             class_feature_map_depth,
 123             [1, 1],
 124             padding='SAME',
 125             name='class_predictions_conv',
 126             **self._conv_hyperparams.params()))
 127     self._class_predictor_layers.append(
 128         self._conv_hyperparams.build_batch_norm(
 129             training=(self._is_training and not self._freeze_batchnorm),
 130             name='class_predictions_batchnorm'))
 131 
 132   @property
 133   def num_classes(self):
 134     return self._num_classes
 135 
 136   def _predict(self, image_features, proposal_boxes, **kwargs):
 137     """Computes encoded object locations and corresponding confidences.
 138 
 139     Args:
 140       image_features: A list of float tensors of shape [batch_size, height_i,
 141       width_i, channels_i] containing features for a batch of images.
 142       proposal_boxes: A float tensor of shape [batch_size, num_proposals,
 143         box_code_size].
 144       **kwargs: Unused Keyword args
 145 
 146     Returns:
 147       box_encodings: A list of float tensors of shape
 148         [batch_size, num_anchors_i, q, code_size] representing the location of
 149         the objects, where q is 1 or the number of classes. Each entry in the
 150         list corresponds to a feature map in the input `image_features` list.
 151       class_predictions_with_background: A list of float tensors of shape
 152         [batch_size, num_anchors_i, num_classes + 1] representing the class
 153         predictions for the proposals. Each entry in the list corresponds to a
 154         feature map in the input `image_features` list.
 155 
 156     Raises:
 157       ValueError: if num_predictions_per_location is not 1 or if
 158         len(image_features) is not 1.
 159     """
 160     if len(image_features) != 1:
 161       raise ValueError('length of `image_features` must be 1. Found {}'.
 162                        format(len(image_features)))
 163     image_feature = image_features[0]
 164     batch_size = tf.shape(proposal_boxes)[0]
 165     num_boxes = tf.shape(proposal_boxes)[1]
 166     net = image_feature
 167     for layer in self._shared_conv_layers:
 168       net = layer(net)
 169 
 170     # Location predictions.
 171     box_net = net
 172     for layer in self._box_encoder_layers:
 173       box_net = layer(box_net)
 174     box_encodings = ops.batch_position_sensitive_crop_regions(
 175         box_net,
 176         boxes=proposal_boxes,
 177         crop_size=self._crop_size,
 178         num_spatial_bins=self._num_spatial_bins,
 179         global_pool=True)
 180     box_encodings = tf.squeeze(box_encodings, axis=[2, 3])
 181     box_encodings = tf.reshape(box_encodings,
 182                                [batch_size * num_boxes, 1, self.num_classes,
 183                                 self._box_code_size])
 184 
 185     # Class predictions.
 186     class_net = net
 187     for layer in self._class_predictor_layers:
 188       class_net = layer(class_net)
 189     class_predictions_with_background = (
 190         ops.batch_position_sensitive_crop_regions(
 191             class_net,
 192             boxes=proposal_boxes,
 193             crop_size=self._crop_size,
 194             num_spatial_bins=self._num_spatial_bins,
 195             global_pool=True))
 196     class_predictions_with_background = tf.squeeze(
 197         class_predictions_with_background, axis=[2, 3])
 198     class_predictions_with_background = tf.reshape(
 199         class_predictions_with_background,
 200         [batch_size * num_boxes, 1, self._total_classes])
 201 
 202     return {BOX_ENCODINGS: [box_encodings],
 203             CLASS_PREDICTIONS_WITH_BACKGROUND:
 204             [class_predictions_with_background]}
