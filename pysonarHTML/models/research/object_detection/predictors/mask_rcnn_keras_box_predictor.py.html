  21 CLASS_PREDICTIONS_WITH_BACKGROUND = (
  22     box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND)

  26 class MaskRCNNKerasBoxPredictor(box_predictor.KerasBoxPredictor):
  27   """Mask R-CNN Box Predictor.
  28 
  29   See Mask R-CNN: He, K., Gkioxari, G., Dollar, P., & Girshick, R. (2017).
  30   Mask R-CNN. arXiv preprint arXiv:1703.06870.
  31 
  32   This is used for the second stage of the Mask R-CNN detector where proposals
  33   cropped from an image are arranged along the batch dimension of the input
  34   image_features tensor. Notice that locations are *not* shared across classes,
  35   thus for each anchor, a separate prediction is made for each class.
  36 
  37   In addition to predicting boxes and classes, optionally this class allows
  38   predicting masks and/or keypoints inside detection boxes.
  39 
  40   Currently this box predictor makes per-class predictions; that is, each
  41   anchor makes a separate box prediction for each class.
  42   """
  43 
  44   def __init__(self,
  45                is_training,
  46                num_classes,
  47                freeze_batchnorm,
  48                box_prediction_head,
  49                class_prediction_head,
  50                third_stage_heads,
  51                name=None):
  52     """Constructor.
  53 
  54     Args:
  55       is_training: Indicates whether the BoxPredictor is in training mode.
  56       num_classes: number of classes.  Note that num_classes *does not*
  57         include the background category, so if groundtruth labels take values
  58         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  59         assigned classification targets can range from {0,... K}).
  60       freeze_batchnorm: Whether to freeze batch norm parameters during
  61         training or not. When training with a small batch size (e.g. 1), it is
  62         desirable to freeze batch norm update and use pretrained batch norm
  63         params.
  64       box_prediction_head: The head that predicts the boxes in second stage.
  65       class_prediction_head: The head that predicts the classes in second stage.
  66       third_stage_heads: A dictionary mapping head names to mask rcnn head
  67         classes.
  68       name: A string name scope to assign to the model. If `None`, Keras
  69         will auto-generate one from the class name.
  70     """
  71     super(MaskRCNNKerasBoxPredictor, self).__init__(
  72         is_training, num_classes, freeze_batchnorm=freeze_batchnorm,
  73         inplace_batchnorm_update=False, name=name)
  74     self._box_prediction_head = box_prediction_head
  75     self._class_prediction_head = class_prediction_head
  76     self._third_stage_heads = third_stage_heads
  77 
  78   @property
  79   def num_classes(self):
  80     return self._num_classes
  81 
  82   def get_second_stage_prediction_heads(self):
  83     return BOX_ENCODINGS, CLASS_PREDICTIONS_WITH_BACKGROUND
  84 
  85   def get_third_stage_prediction_heads(self):
  86     return sorted(self._third_stage_heads.keys())
  87 
  88   def _predict(self,
  89                image_features,
  90                prediction_stage=2,
  91                **kwargs):
  92     """Optionally computes encoded object locations, confidences, and masks.
  93 
  94     Predicts the heads belonging to the given prediction stage.
  95 
  96     Args:
  97       image_features: A list of float tensors of shape
  98         [batch_size, height_i, width_i, channels_i] containing roi pooled
  99         features for each image. The length of the list should be 1 otherwise
 100         a ValueError will be raised.
 101       prediction_stage: Prediction stage. Acceptable values are 2 and 3.
 102       **kwargs: Unused Keyword args
 103 
 104     Returns:
 105       A dictionary containing the predicted tensors that are listed in
 106       self._prediction_heads. A subset of the following keys will exist in the
 107       dictionary:
 108         BOX_ENCODINGS: A float tensor of shape
 109           [batch_size, 1, num_classes, code_size] representing the
 110           location of the objects.
 111         CLASS_PREDICTIONS_WITH_BACKGROUND: A float tensor of shape
 112           [batch_size, 1, num_classes + 1] representing the class
 113           predictions for the proposals.
 114         MASK_PREDICTIONS: A float tensor of shape
 115           [batch_size, 1, num_classes, image_height, image_width]
 116 
 117     Raises:
 118       ValueError: If num_predictions_per_location is not 1 or if
 119         len(image_features) is not 1.
 120       ValueError: if prediction_stage is not 2 or 3.
 121     """
 122     if len(image_features) != 1:
 123       raise ValueError('length of `image_features` must be 1. Found {}'.format(
 124           len(image_features)))
 125     image_feature = image_features[0]
 126     predictions_dict = {}
 127 
 128     if prediction_stage == 2:
 129       predictions_dict[BOX_ENCODINGS] = self._box_prediction_head(image_feature)
 130       predictions_dict[CLASS_PREDICTIONS_WITH_BACKGROUND] = (
 131           self._class_prediction_head(image_feature))
 132     elif prediction_stage == 3:
 133       for prediction_head in self.get_third_stage_prediction_heads():
 134         head_object = self._third_stage_heads[prediction_head]
 135         predictions_dict[prediction_head] = head_object(image_feature)
 136     else:
 137       raise ValueError('prediction_stage should be either 2 or 3.')
 138 
 139     return predictions_dict
