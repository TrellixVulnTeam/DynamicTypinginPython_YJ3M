  16 """Tests for object_detection.predictors.mask_rcnn_box_predictor."""
  17 import numpy as np

  20 from google.protobuf import text_format
  21 from object_detection.builders import box_predictor_builder
  22 from object_detection.builders import hyperparams_builder
  23 from object_detection.predictors import mask_rcnn_box_predictor as box_predictor
  24 from object_detection.protos import hyperparams_pb2
  25 from object_detection.utils import test_case

  27 
  28 class MaskRCNNBoxPredictorTest(test_case.TestCase):

  30   def _build_arg_scope_with_hyperparams(self,
  31                                         op_type=hyperparams_pb2.Hyperparams.FC):
  32     hyperparams = hyperparams_pb2.Hyperparams()
  33     hyperparams_text_proto = """
  34       activation: NONE
  35       regularizer {
  36         l2_regularizer {
  37         }
  38       }
  39       initializer {
  40         truncated_normal_initializer {
  41         }
  42       }
  43     """
  44     text_format.Merge(hyperparams_text_proto, hyperparams)
  45     hyperparams.op = op_type
  46     return hyperparams_builder.build(hyperparams, is_training=True)
  47 
  48   def test_get_boxes_with_five_classes(self):
  49     def graph_fn(image_features):
  50       mask_box_predictor = box_predictor_builder.build_mask_rcnn_box_predictor(
  51           is_training=False,
  52           num_classes=5,
  53           fc_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
  54           use_dropout=False,
  55           dropout_keep_prob=0.5,
  56           box_code_size=4,
  57       )
  58       box_predictions = mask_box_predictor.predict(
  59           [image_features],
  60           num_predictions_per_location=[1],
  61           scope='BoxPredictor',
  62           prediction_stage=2)
  63       return (box_predictions[box_predictor.BOX_ENCODINGS],
  64               box_predictions[box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND])
  65     image_features = np.random.rand(2, 7, 7, 3).astype(np.float32)
  66     (box_encodings,
  67      class_predictions_with_background) = self.execute(graph_fn,
  68                                                        [image_features])
  69     self.assertAllEqual(box_encodings.shape, [2, 1, 5, 4])
  70     self.assertAllEqual(class_predictions_with_background.shape, [2, 1, 6])
  71 
  72   def test_get_boxes_with_five_classes_share_box_across_classes(self):
  73     def graph_fn(image_features):
  74       mask_box_predictor = box_predictor_builder.build_mask_rcnn_box_predictor(
  75           is_training=False,
  76           num_classes=5,
  77           fc_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
  78           use_dropout=False,
  79           dropout_keep_prob=0.5,
  80           box_code_size=4,
  81           share_box_across_classes=True
  82       )
  83       box_predictions = mask_box_predictor.predict(
  84           [image_features],
  85           num_predictions_per_location=[1],
  86           scope='BoxPredictor',
  87           prediction_stage=2)
  88       return (box_predictions[box_predictor.BOX_ENCODINGS],
  89               box_predictions[box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND])
  90     image_features = np.random.rand(2, 7, 7, 3).astype(np.float32)
  91     (box_encodings,
  92      class_predictions_with_background) = self.execute(graph_fn,
  93                                                        [image_features])
  94     self.assertAllEqual(box_encodings.shape, [2, 1, 1, 4])
  95     self.assertAllEqual(class_predictions_with_background.shape, [2, 1, 6])

 108   def test_get_instance_masks(self):
 109     def graph_fn(image_features):
 110       mask_box_predictor = box_predictor_builder.build_mask_rcnn_box_predictor(
 111           is_training=False,
 112           num_classes=5,
 113           fc_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
 114           use_dropout=False,
 115           dropout_keep_prob=0.5,
 116           box_code_size=4,
 117           conv_hyperparams_fn=self._build_arg_scope_with_hyperparams(
 118               op_type=hyperparams_pb2.Hyperparams.CONV),
 119           predict_instance_masks=True)
 120       box_predictions = mask_box_predictor.predict(
 121           [image_features],
 122           num_predictions_per_location=[1],
 123           scope='BoxPredictor',
 124           prediction_stage=3)
 125       return (box_predictions[box_predictor.MASK_PREDICTIONS],)
 126     image_features = np.random.rand(2, 7, 7, 3).astype(np.float32)
 127     mask_predictions = self.execute(graph_fn, [image_features])
 128     self.assertAllEqual(mask_predictions.shape, [2, 1, 5, 14, 14])

 130   def test_do_not_return_instance_masks_without_request(self):
 131     image_features = tf.random_uniform([2, 7, 7, 3], dtype=tf.float32)
 132     mask_box_predictor = box_predictor_builder.build_mask_rcnn_box_predictor(
 133         is_training=False,
 134         num_classes=5,
 135         fc_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
 136         use_dropout=False,
 137         dropout_keep_prob=0.5,
 138         box_code_size=4)
 139     box_predictions = mask_box_predictor.predict(
 140         [image_features],
 141         num_predictions_per_location=[1],
 142         scope='BoxPredictor',
 143         prediction_stage=2)
 144     self.assertEqual(len(box_predictions), 2)
 145     self.assertTrue(box_predictor.BOX_ENCODINGS in box_predictions)
 146     self.assertTrue(box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND
 147                     in box_predictions)
