 249   def test_no_dangling_outputs(self):
 250     image_features = tf.placeholder(dtype=tf.float32, shape=[4, None, None, 64])
 251     conv_box_predictor = (
 252         box_predictor_builder.build_convolutional_box_predictor(
 253             is_training=False,
 254             num_classes=0,
 255             conv_hyperparams_fn=self._build_arg_scope_with_conv_hyperparams(),
 256             min_depth=0,
 257             max_depth=32,
 258             num_layers_before_predictor=1,
 259             dropout_keep_prob=0.8,
 260             kernel_size=3,
 261             box_code_size=4,
 262             use_dropout=True,
 263             use_depthwise=True))
 264     box_predictions = conv_box_predictor.predict(
 265         [image_features], num_predictions_per_location=[5],
 266         scope='BoxPredictor')
 267     tf.concat(
 268         box_predictions[box_predictor.BOX_ENCODINGS], axis=1)
 269     tf.concat(
 270         box_predictions[box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND],
 271         axis=1)
 272 
 273     bad_dangling_ops = []
 274     types_safe_to_dangle = set(['Assign', 'Mul', 'Const'])
 275     for op in tf.get_default_graph().get_operations():
 276       if (not op.outputs) or (not op.outputs[0].consumers()):
 277         if 'BoxPredictor' in op.name:
 278           if op.type not in types_safe_to_dangle:
 279             bad_dangling_ops.append(op)
 280 
 281     self.assertEqual(bad_dangling_ops, [])

 868     box_code_size = 4
 869     num_classes_without_background = 3
 870     other_head_name = 'Mask'
 871     mask_height = 5
 872     mask_width = 5
 873     num_predictions_per_location = 5
 874 
 875     def graph_fn(image_features):
 876       box_prediction_head = box_head.WeightSharedConvolutionalBoxHead(
 877           box_code_size)
 878       class_prediction_head = class_head.WeightSharedConvolutionalClassHead(
 879           num_classes_without_background + 1)
 880       other_heads = {
 881           other_head_name:
 882               mask_head.WeightSharedConvolutionalMaskHead(
 883                   num_classes_without_background,
 884                   mask_height=mask_height,
 885                   mask_width=mask_width)
 886       }
 887       conv_box_predictor = box_predictor.WeightSharedConvolutionalBoxPredictor(
 888           is_training=False,
 889           num_classes=num_classes_without_background,
 890           box_prediction_head=box_prediction_head,
 891           class_prediction_head=class_prediction_head,
 892           other_heads=other_heads,
 893           conv_hyperparams_fn=self._build_arg_scope_with_conv_hyperparams(),
 894           depth=32,
 895           num_layers_before_predictor=2)
 896       box_predictions = conv_box_predictor.predict(
 897           [image_features],
 898           num_predictions_per_location=[num_predictions_per_location],
 899           scope='BoxPredictor')
 900       for key, value in box_predictions.items():
 901         box_predictions[key] = tf.concat(value, axis=1)
 902       assert len(box_predictions) == 3
 903       return (box_predictions[box_predictor.BOX_ENCODINGS],
 904               box_predictions[box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND],
 905               box_predictions[other_head_name])
 906 
 907     batch_size = 4
 908     feature_ht = 8
 909     feature_wt = 8
 910     image_features = np.random.rand(batch_size, feature_ht, feature_wt,
 911                                     64).astype(np.float32)
 912     (box_encodings, class_predictions, other_head_predictions) = self.execute(
 913         graph_fn, [image_features])
 914     num_anchors = feature_ht * feature_wt * num_predictions_per_location
 915     self.assertAllEqual(box_encodings.shape,
 916                         [batch_size, num_anchors, box_code_size])

 918         class_predictions.shape,
