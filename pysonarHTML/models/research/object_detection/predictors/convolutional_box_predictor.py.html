  46 class ConvolutionalBoxPredictor(box_predictor.BoxPredictor):
  47   """Convolutional Box Predictor.
  48 
  49   Optionally add an intermediate 1x1 convolutional layer after features and
  50   predict in parallel branches box_encodings and
  51   class_predictions_with_background.
  52 
  53   Currently this box predictor assumes that predictions are "shared" across
  54   classes --- that is each anchor makes box predictions which do not depend
  55   on class.
  56   """
  57 
  58   def __init__(self,
  59                is_training,
  60                num_classes,
  61                box_prediction_head,
  62                class_prediction_head,
  63                other_heads,
  64                conv_hyperparams_fn,
  65                num_layers_before_predictor,
  66                min_depth,
  67                max_depth):
  68     """Constructor.
  69 
  70     Args:
  71       is_training: Indicates whether the BoxPredictor is in training mode.
  72       num_classes: number of classes.  Note that num_classes *does not*
  73         include the background category, so if groundtruth labels take values
  74         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  75         assigned classification targets can range from {0,... K}).
  76       box_prediction_head: The head that predicts the boxes.
  77       class_prediction_head: The head that predicts the classes.
  78       other_heads: A dictionary mapping head names to convolutional
  79         head classes.
  80       conv_hyperparams_fn: A function to generate tf-slim arg_scope with
  81         hyperparameters for convolution ops.
  82       num_layers_before_predictor: Number of the additional conv layers before
  83         the predictor.
  84       min_depth: Minimum feature depth prior to predicting box encodings
  85         and class predictions.
  86       max_depth: Maximum feature depth prior to predicting box encodings
  87         and class predictions. If max_depth is set to 0, no additional
  88         feature map will be inserted before location and class predictions.
  89 
  90     Raises:
  91       ValueError: if min_depth > max_depth.
  92     """
  93     super(ConvolutionalBoxPredictor, self).__init__(is_training, num_classes)
  94     self._box_prediction_head = box_prediction_head
  95     self._class_prediction_head = class_prediction_head
  96     self._other_heads = other_heads
  97     self._conv_hyperparams_fn = conv_hyperparams_fn
  98     self._min_depth = min_depth
  99     self._max_depth = max_depth
 100     self._num_layers_before_predictor = num_layers_before_predictor
 101 
 102   @property
 103   def num_classes(self):
 104     return self._num_classes
 105 
 106   def _predict(self, image_features, num_predictions_per_location_list):
 107     """Computes encoded object locations and corresponding confidences.
 108 
 109     Args:
 110       image_features: A list of float tensors of shape [batch_size, height_i,
 111         width_i, channels_i] containing features for a batch of images.
 112       num_predictions_per_location_list: A list of integers representing the
 113         number of box predictions to be made per spatial location for each
 114         feature map.
 115 
 116     Returns:
 117       A dictionary containing:
 118         box_encodings: A list of float tensors of shape
 119           [batch_size, num_anchors_i, q, code_size] representing the location of
 120           the objects, where q is 1 or the number of classes. Each entry in the
 121           list corresponds to a feature map in the input `image_features` list.
 122         class_predictions_with_background: A list of float tensors of shape
 123           [batch_size, num_anchors_i, num_classes + 1] representing the class
 124           predictions for the proposals. Each entry in the list corresponds to a
 125           feature map in the input `image_features` list.
 126         (optional) Predictions from other heads.
 127     """
 128     predictions = {
 129         BOX_ENCODINGS: [],
 130         CLASS_PREDICTIONS_WITH_BACKGROUND: [],
 131     }
 132     for head_name in self._other_heads.keys():
 133       predictions[head_name] = []
 134     # TODO(rathodv): Come up with a better way to generate scope names
 135     # in box predictor once we have time to retrain all models in the zoo.
 136     # The following lines create scope names to be backwards compatible with the
 137     # existing checkpoints.
 138     box_predictor_scopes = [_NoopVariableScope()]
 139     if len(image_features) > 1:
 140       box_predictor_scopes = [
 141           tf.variable_scope('BoxPredictor_{}'.format(i))
 142           for i in range(len(image_features))
 143       ]
 144     for (image_feature,
 145          num_predictions_per_location, box_predictor_scope) in zip(
 146              image_features, num_predictions_per_location_list,
 147              box_predictor_scopes):
 148       net = image_feature
 149       with box_predictor_scope:
 150         with slim.arg_scope(self._conv_hyperparams_fn()):
 151           with slim.arg_scope([slim.dropout], is_training=self._is_training):
 152             # Add additional conv layers before the class predictor.
 153             features_depth = static_shape.get_depth(image_feature.get_shape())
 154             depth = max(min(features_depth, self._max_depth), self._min_depth)
 155             tf.logging.info('depth of additional conv before box predictor: {}'.
 156                             format(depth))
 157             if depth > 0 and self._num_layers_before_predictor > 0:
 158               for i in range(self._num_layers_before_predictor):
 159                 net = slim.conv2d(
 160                     net,
 161                     depth, [1, 1],
 162                     reuse=tf.AUTO_REUSE,
 163                     scope='Conv2d_%d_1x1_%d' % (i, depth))
 164             sorted_keys = sorted(self._other_heads.keys())
 165             sorted_keys.append(BOX_ENCODINGS)
 166             sorted_keys.append(CLASS_PREDICTIONS_WITH_BACKGROUND)
 167             for head_name in sorted_keys:
 168               if head_name == BOX_ENCODINGS:
 169                 head_obj = self._box_prediction_head
 170               elif head_name == CLASS_PREDICTIONS_WITH_BACKGROUND:
 171                 head_obj = self._class_prediction_head
 172               else:
 173                 head_obj = self._other_heads[head_name]
 174               prediction = head_obj.predict(
 175                   features=net,
 176                   num_predictions_per_location=num_predictions_per_location)
 177               predictions[head_name].append(prediction)
 178     return predictions

 190 class WeightSharedConvolutionalBoxPredictor(box_predictor.BoxPredictor):
 191   """Convolutional Box Predictor with weight sharing.
 192 
 193   Defines the box predictor as defined in
 194   https://arxiv.org/abs/1708.02002. This class differs from
 195   ConvolutionalBoxPredictor in that it shares weights and biases while
 196   predicting from different feature maps. However, batch_norm parameters are not
 197   shared because the statistics of the activations vary among the different
 198   feature maps.
 199 
 200   Also note that separate multi-layer towers are constructed for the box
 201   encoding and class predictors respectively.
 202   """
 203 
 204   def __init__(self,
 205                is_training,
 206                num_classes,
 207                box_prediction_head,
 208                class_prediction_head,
 209                other_heads,
 210                conv_hyperparams_fn,
 211                depth,
 212                num_layers_before_predictor,
 213                kernel_size=3,
 214                apply_batch_norm=False,
 215                share_prediction_tower=False,
 216                use_depthwise=False):
 217     """Constructor.
 218 
 219     Args:
 220       is_training: Indicates whether the BoxPredictor is in training mode.
 221       num_classes: number of classes.  Note that num_classes *does not*
 222         include the background category, so if groundtruth labels take values
 223         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 224         assigned classification targets can range from {0,... K}).
 225       box_prediction_head: The head that predicts the boxes.
 226       class_prediction_head: The head that predicts the classes.
 227       other_heads: A dictionary mapping head names to convolutional
 228         head classes.
 229       conv_hyperparams_fn: A function to generate tf-slim arg_scope with
 230         hyperparameters for convolution ops.
 231       depth: depth of conv layers.
 232       num_layers_before_predictor: Number of the additional conv layers before
 233         the predictor.
 234       kernel_size: Size of final convolution kernel.
 235       apply_batch_norm: Whether to apply batch normalization to conv layers in
 236         this predictor.
 237       share_prediction_tower: Whether to share the multi-layer tower among box
 238         prediction head, class prediction head and other heads.
 239       use_depthwise: Whether to use depthwise separable conv2d instead of
 240        regular conv2d.
 241     """
 242     super(WeightSharedConvolutionalBoxPredictor, self).__init__(is_training,
 243                                                                 num_classes)
 244     self._box_prediction_head = box_prediction_head
 245     self._class_prediction_head = class_prediction_head
 246     self._other_heads = other_heads
 247     self._conv_hyperparams_fn = conv_hyperparams_fn
 248     self._depth = depth
 249     self._num_layers_before_predictor = num_layers_before_predictor
 250     self._kernel_size = kernel_size
 251     self._apply_batch_norm = apply_batch_norm
 252     self._share_prediction_tower = share_prediction_tower
 253     self._use_depthwise = use_depthwise
 254 
 255   @property
 256   def num_classes(self):
 257     return self._num_classes
 258 
 259   def _insert_additional_projection_layer(self, image_feature,
 260                                           inserted_layer_counter,
 261                                           target_channel):
 262     if inserted_layer_counter < 0:
 263       return image_feature, inserted_layer_counter
 264     image_feature = slim.conv2d(
 265         image_feature,
 266         target_channel, [1, 1],
 267         stride=1,
 268         padding='SAME',
 269         activation_fn=None,
 270         normalizer_fn=(tf.identity if self._apply_batch_norm else None),
 271         scope='ProjectionLayer/conv2d_{}'.format(
 272             inserted_layer_counter))
 273     if self._apply_batch_norm:
 274       image_feature = slim.batch_norm(
 275           image_feature,
 276           scope='ProjectionLayer/conv2d_{}/BatchNorm'.format(
 277               inserted_layer_counter))
 278     inserted_layer_counter += 1
 279     return image_feature, inserted_layer_counter
 280 
 281   def _compute_base_tower(self, tower_name_scope, image_feature, feature_index):
 282     net = image_feature
 283     for i in range(self._num_layers_before_predictor):
 284       if self._use_depthwise:
 285         conv_op = functools.partial(slim.separable_conv2d, depth_multiplier=1)
 286       else:
 287         conv_op = slim.conv2d
 288       net = conv_op(
 289           net,
 290           self._depth, [self._kernel_size, self._kernel_size],
 291           stride=1,
 292           padding='SAME',
 293           activation_fn=None,
 294           normalizer_fn=(tf.identity if self._apply_batch_norm else None),
 295           scope='{}/conv2d_{}'.format(tower_name_scope, i))
 296       if self._apply_batch_norm:
 297         net = slim.batch_norm(
 298             net,
 299             scope='{}/conv2d_{}/BatchNorm/feature_{}'.
 300             format(tower_name_scope, i, feature_index))
 301       net = tf.nn.relu6(net)
 302     return net
 303 
 304   def _predict_head(self, head_name, head_obj, image_feature, box_tower_feature,
 305                     feature_index, num_predictions_per_location):
 306     if head_name == CLASS_PREDICTIONS_WITH_BACKGROUND:
 307       tower_name_scope = 'ClassPredictionTower'
 308     else:
 309       tower_name_scope = head_name + 'PredictionTower'
 310     if self._share_prediction_tower:
 311       head_tower_feature = box_tower_feature
 312     else:
 313       head_tower_feature = self._compute_base_tower(
 314           tower_name_scope=tower_name_scope,
 315           image_feature=image_feature,
 316           feature_index=feature_index)
 317     return head_obj.predict(
 318         features=head_tower_feature,
 319         num_predictions_per_location=num_predictions_per_location)

 321   def _predict(self, image_features, num_predictions_per_location_list):
 322     """Computes encoded object locations and corresponding confidences.
 323 
 324     Args:
 325       image_features: A list of float tensors of shape [batch_size, height_i,
 326         width_i, channels] containing features for a batch of images. Note that
 327         when not all tensors in the list have the same number of channels, an
 328         additional projection layer will be added on top the tensor to generate
 329         feature map with number of channels consitent with the majority.
 330       num_predictions_per_location_list: A list of integers representing the
 331         number of box predictions to be made per spatial location for each
 332         feature map. Note that all values must be the same since the weights are
 333         shared.
 334 
 335     Returns:
 336       A dictionary containing:
 337         box_encodings: A list of float tensors of shape
 338           [batch_size, num_anchors_i, code_size] representing the location of
 339           the objects. Each entry in the list corresponds to a feature map in
 340           the input `image_features` list.
 341         class_predictions_with_background: A list of float tensors of shape
 342           [batch_size, num_anchors_i, num_classes + 1] representing the class
 343           predictions for the proposals. Each entry in the list corresponds to a
 344           feature map in the input `image_features` list.
 345         (optional) Predictions from other heads.
 346           E.g., mask_predictions: A list of float tensors of shape
 347           [batch_size, num_anchord_i, num_classes, mask_height, mask_width].
 348 
 349 
 350     Raises:
 351       ValueError: If the num predictions per locations differs between the
 352         feature maps.
 353     """
 354     if len(set(num_predictions_per_location_list)) > 1:
 355       raise ValueError('num predictions per location must be same for all'
 356                        'feature maps, found: {}'.format(
 357                            num_predictions_per_location_list))
 358     feature_channels = [
 359         shape_utils.get_dim_as_int(image_feature.shape[3])
 360         for image_feature in image_features
 361     ]
 362     has_different_feature_channels = len(set(feature_channels)) > 1
 363     if has_different_feature_channels:
 364       inserted_layer_counter = 0
 365       target_channel = max(set(feature_channels), key=feature_channels.count)
 366       tf.logging.info('Not all feature maps have the same number of '
 367                       'channels, found: {}, appending additional projection '
 368                       'layers to bring all feature maps to uniformly have {} '
 369                       'channels.'.format(feature_channels, target_channel))
 370     else:
 371       # Place holder variables if has_different_feature_channels is False.
 372       target_channel = -1
 373       inserted_layer_counter = -1
 374     predictions = {
 375         BOX_ENCODINGS: [],
 376         CLASS_PREDICTIONS_WITH_BACKGROUND: [],
 377     }
 378     for head_name in self._other_heads.keys():
 379       predictions[head_name] = []
 380     for feature_index, (image_feature,
 381                         num_predictions_per_location) in enumerate(
 382                             zip(image_features,
 383                                 num_predictions_per_location_list)):
 384       with tf.variable_scope('WeightSharedConvolutionalBoxPredictor',
 385                              reuse=tf.AUTO_REUSE):
 386         with slim.arg_scope(self._conv_hyperparams_fn()):
 387           # TODO(wangjiang) Pass is_training to the head class directly.
 388           with slim.arg_scope([slim.dropout], is_training=self._is_training):
 389             (image_feature,
 390              inserted_layer_counter) = self._insert_additional_projection_layer(
 391                  image_feature, inserted_layer_counter, target_channel)
 392             if self._share_prediction_tower:
 393               box_tower_scope = 'PredictionTower'
 394             else:
 395               box_tower_scope = 'BoxPredictionTower'
 396             box_tower_feature = self._compute_base_tower(
 397                 tower_name_scope=box_tower_scope,
 398                 image_feature=image_feature,
 399                 feature_index=feature_index)
 400             box_encodings = self._box_prediction_head.predict(
 401                 features=box_tower_feature,
 402                 num_predictions_per_location=num_predictions_per_location)
 403             predictions[BOX_ENCODINGS].append(box_encodings)
 404             sorted_keys = sorted(self._other_heads.keys())
 405             sorted_keys.append(CLASS_PREDICTIONS_WITH_BACKGROUND)
 406             for head_name in sorted_keys:
 407               if head_name == CLASS_PREDICTIONS_WITH_BACKGROUND:
 408                 head_obj = self._class_prediction_head
 409               else:
 410                 head_obj = self._other_heads[head_name]
 411               prediction = self._predict_head(
 412                   head_name=head_name,
 413                   head_obj=head_obj,
 414                   image_feature=image_feature,
 415                   box_tower_feature=box_tower_feature,
 416                   feature_index=feature_index,
 417                   num_predictions_per_location=num_predictions_per_location)
 418               predictions[head_name].append(prediction)
 419     return predictions
