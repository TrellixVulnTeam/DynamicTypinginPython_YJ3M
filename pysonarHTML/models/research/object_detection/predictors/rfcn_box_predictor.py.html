  29 class RfcnBoxPredictor(box_predictor.BoxPredictor):
  30   """RFCN Box Predictor.
  31 
  32   Applies a position sensitive ROI pooling on position sensitive feature maps to
  33   predict classes and refined locations. See https://arxiv.org/abs/1605.06409
  34   for details.
  35 
  36   This is used for the second stage of the RFCN meta architecture. Notice that
  37   locations are *not* shared across classes, thus for each anchor, a separate
  38   prediction is made for each class.
  39   """
  40 
  41   def __init__(self,
  42                is_training,
  43                num_classes,
  44                conv_hyperparams_fn,
  45                num_spatial_bins,
  46                depth,
  47                crop_size,
  48                box_code_size):
  49     """Constructor.
  50 
  51     Args:
  52       is_training: Indicates whether the BoxPredictor is in training mode.
  53       num_classes: number of classes.  Note that num_classes *does not*
  54         include the background category, so if groundtruth labels take values
  55         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  56         assigned classification targets can range from {0,... K}).
  57       conv_hyperparams_fn: A function to construct tf-slim arg_scope with
  58         hyperparameters for convolutional layers.
  59       num_spatial_bins: A list of two integers `[spatial_bins_y,
  60         spatial_bins_x]`.
  61       depth: Target depth to reduce the input feature maps to.
  62       crop_size: A list of two integers `[crop_height, crop_width]`.
  63       box_code_size: Size of encoding for each box.
  64     """
  65     super(RfcnBoxPredictor, self).__init__(is_training, num_classes)
  66     self._conv_hyperparams_fn = conv_hyperparams_fn
  67     self._num_spatial_bins = num_spatial_bins
  68     self._depth = depth
  69     self._crop_size = crop_size
  70     self._box_code_size = box_code_size
  71 
  72   @property
  73   def num_classes(self):
  74     return self._num_classes
  75 
  76   def _predict(self, image_features, num_predictions_per_location,
  77                proposal_boxes):
  78     """Computes encoded object locations and corresponding confidences.
  79 
  80     Args:
  81       image_features: A list of float tensors of shape [batch_size, height_i,
  82       width_i, channels_i] containing features for a batch of images.
  83       num_predictions_per_location: A list of integers representing the number
  84         of box predictions to be made per spatial location for each feature map.
  85         Currently, this must be set to [1], or an error will be raised.
  86       proposal_boxes: A float tensor of shape [batch_size, num_proposals,
  87         box_code_size].
  88 
  89     Returns:
  90       box_encodings: A list of float tensors of shape
  91         [batch_size, num_anchors_i, q, code_size] representing the location of
  92         the objects, where q is 1 or the number of classes. Each entry in the
  93         list corresponds to a feature map in the input `image_features` list.
  94       class_predictions_with_background: A list of float tensors of shape
  95         [batch_size, num_anchors_i, num_classes + 1] representing the class
  96         predictions for the proposals. Each entry in the list corresponds to a
  97         feature map in the input `image_features` list.
  98 
  99     Raises:
 100       ValueError: if num_predictions_per_location is not 1 or if
 101         len(image_features) is not 1.
 102     """
 103     if (len(num_predictions_per_location) != 1 or
 104         num_predictions_per_location[0] != 1):
 105       raise ValueError('Currently RfcnBoxPredictor only supports '
 106                        'predicting a single box per class per location.')
 107     if len(image_features) != 1:
 108       raise ValueError('length of `image_features` must be 1. Found {}'.
 109                        format(len(image_features)))
 110     image_feature = image_features[0]
 111     num_predictions_per_location = num_predictions_per_location[0]
 112     batch_size = tf.shape(proposal_boxes)[0]
 113     num_boxes = tf.shape(proposal_boxes)[1]
 114     net = image_feature
 115     with slim.arg_scope(self._conv_hyperparams_fn()):
 116       net = slim.conv2d(net, self._depth, [1, 1], scope='reduce_depth')
 117       # Location predictions.
 118       location_feature_map_depth = (self._num_spatial_bins[0] *
 119                                     self._num_spatial_bins[1] *
 120                                     self.num_classes *
 121                                     self._box_code_size)
 122       location_feature_map = slim.conv2d(net, location_feature_map_depth,
 123                                          [1, 1], activation_fn=None,
 124                                          scope='refined_locations')
 125       box_encodings = ops.batch_position_sensitive_crop_regions(
 126           location_feature_map,
 127           boxes=proposal_boxes,
 128           crop_size=self._crop_size,
 129           num_spatial_bins=self._num_spatial_bins,
 130           global_pool=True)
 131       box_encodings = tf.squeeze(box_encodings, axis=[2, 3])
 132       box_encodings = tf.reshape(box_encodings,
 133                                  [batch_size * num_boxes, 1, self.num_classes,
 134                                   self._box_code_size])
 135 
 136       # Class predictions.
 137       total_classes = self.num_classes + 1  # Account for background class.
 138       class_feature_map_depth = (self._num_spatial_bins[0] *
 139                                  self._num_spatial_bins[1] *
 140                                  total_classes)
 141       class_feature_map = slim.conv2d(net, class_feature_map_depth, [1, 1],
 142                                       activation_fn=None,
 143                                       scope='class_predictions')
 144       class_predictions_with_background = (
 145           ops.batch_position_sensitive_crop_regions(
 146               class_feature_map,
 147               boxes=proposal_boxes,
 148               crop_size=self._crop_size,
 149               num_spatial_bins=self._num_spatial_bins,
 150               global_pool=True))
 151       class_predictions_with_background = tf.squeeze(
 152           class_predictions_with_background, axis=[2, 3])
 153       class_predictions_with_background = tf.reshape(
 154           class_predictions_with_background,
 155           [batch_size * num_boxes, 1, total_classes])
 156 
 157     return {BOX_ENCODINGS: [box_encodings],
 158             CLASS_PREDICTIONS_WITH_BACKGROUND:
 159             [class_predictions_with_background]}
