  36 class MaskRCNNMaskHead(head.Head):
  37   """Mask RCNN mask prediction head.
  38 
  39   Please refer to Mask RCNN paper:
  40   https://arxiv.org/abs/1703.06870
  41   """
  42 
  43   def __init__(self,
  44                num_classes,
  45                conv_hyperparams_fn=None,
  46                mask_height=14,
  47                mask_width=14,
  48                mask_prediction_num_conv_layers=2,
  49                mask_prediction_conv_depth=256,
  50                masks_are_class_agnostic=False,
  51                convolve_then_upsample=False):
  52     """Constructor.
  53 
  54     Args:
  55       num_classes: number of classes.  Note that num_classes *does not*
  56         include the background category, so if groundtruth labels take values
  57         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  58         assigned classification targets can range from {0,... K}).
  59       conv_hyperparams_fn: A function to generate tf-slim arg_scope with
  60         hyperparameters for convolution ops.
  61       mask_height: Desired output mask height. The default value is 14.
  62       mask_width: Desired output mask width. The default value is 14.
  63       mask_prediction_num_conv_layers: Number of convolution layers applied to
  64         the image_features in mask prediction branch.
  65       mask_prediction_conv_depth: The depth for the first conv2d_transpose op
  66         applied to the image_features in the mask prediction branch. If set
  67         to 0, the depth of the convolution layers will be automatically chosen
  68         based on the number of object classes and the number of channels in the
  69         image features.
  70       masks_are_class_agnostic: Boolean determining if the mask-head is
  71         class-agnostic or not.
  72       convolve_then_upsample: Whether to apply convolutions on mask features
  73         before upsampling using nearest neighbor resizing. Otherwise, mask
  74         features are resized to [`mask_height`, `mask_width`] using bilinear
  75         resizing before applying convolutions.
  76 
  77     Raises:
  78       ValueError: conv_hyperparams_fn is None.
  79     """
  80     super(MaskRCNNMaskHead, self).__init__()
  81     self._num_classes = num_classes
  82     self._conv_hyperparams_fn = conv_hyperparams_fn
  83     self._mask_height = mask_height
  84     self._mask_width = mask_width
  85     self._mask_prediction_num_conv_layers = mask_prediction_num_conv_layers
  86     self._mask_prediction_conv_depth = mask_prediction_conv_depth
  87     self._masks_are_class_agnostic = masks_are_class_agnostic
  88     self._convolve_then_upsample = convolve_then_upsample
  89     if conv_hyperparams_fn is None:
  90       raise ValueError('conv_hyperparams_fn is None.')
  91 
  92   def _get_mask_predictor_conv_depth(self,
  93                                      num_feature_channels,
  94                                      num_classes,
  95                                      class_weight=3.0,
  96                                      feature_weight=2.0):
  97     """Computes the depth of the mask predictor convolutions.
  98 
  99     Computes the depth of the mask predictor convolutions given feature channels
 100     and number of classes by performing a weighted average of the two in
 101     log space to compute the number of convolution channels. The weights that
 102     are used for computing the weighted average do not need to sum to 1.
 103 
 104     Args:
 105       num_feature_channels: An integer containing the number of feature
 106         channels.
 107       num_classes: An integer containing the number of classes.
 108       class_weight: Class weight used in computing the weighted average.
 109       feature_weight: Feature weight used in computing the weighted average.
 110 
 111     Returns:
 112       An integer containing the number of convolution channels used by mask
 113         predictor.
 114     """
 115     num_feature_channels_log = math.log(float(num_feature_channels), 2.0)
 116     num_classes_log = math.log(float(num_classes), 2.0)
 117     weighted_num_feature_channels_log = (
 118         num_feature_channels_log * feature_weight)
 119     weighted_num_classes_log = num_classes_log * class_weight
 120     total_weight = feature_weight + class_weight
 121     num_conv_channels_log = round(
 122         (weighted_num_feature_channels_log + weighted_num_classes_log) /
 123         total_weight)
 124     return int(math.pow(2.0, num_conv_channels_log))
 125 
 126   def predict(self, features, num_predictions_per_location=1):
 127     """Performs mask prediction.
 128 
 129     Args:
 130       features: A float tensor of shape [batch_size, height, width, channels]
 131         containing features for a batch of images.
 132       num_predictions_per_location: Int containing number of predictions per
 133         location.
 134 
 135     Returns:
 136       instance_masks: A float tensor of shape
 137           [batch_size, 1, num_classes, mask_height, mask_width].
 138 
 139     Raises:
 140       ValueError: If num_predictions_per_location is not 1.
 141     """
 142     if num_predictions_per_location != 1:
 143       raise ValueError('Only num_predictions_per_location=1 is supported')
 144     num_conv_channels = self._mask_prediction_conv_depth
 145     if num_conv_channels == 0:
 146       num_feature_channels = features.get_shape().as_list()[3]
 147       num_conv_channels = self._get_mask_predictor_conv_depth(
 148           num_feature_channels, self._num_classes)
 149     with slim.arg_scope(self._conv_hyperparams_fn()):
 150       if not self._convolve_then_upsample:
 151         features = tf.image.resize_bilinear(
 152             features, [self._mask_height, self._mask_width],
 153             align_corners=True)
 154       for _ in range(self._mask_prediction_num_conv_layers - 1):
 155         features = slim.conv2d(
 156             features,
 157             num_outputs=num_conv_channels,
 158             kernel_size=[3, 3])
 159       if self._convolve_then_upsample:
 160         # Replace Transposed Convolution with a Nearest Neighbor upsampling step
 161         # followed by 3x3 convolution.
 162         height_scale = self._mask_height // features.shape[1].value
 163         width_scale = self._mask_width // features.shape[2].value
 164         features = ops.nearest_neighbor_upsampling(
 165             features, height_scale=height_scale, width_scale=width_scale)
 166         features = slim.conv2d(
 167             features,
 168             num_outputs=num_conv_channels,
 169             kernel_size=[3, 3])
 170 
 171       num_masks = 1 if self._masks_are_class_agnostic else self._num_classes
 172       mask_predictions = slim.conv2d(
 173           features,
 174           num_outputs=num_masks,
 175           activation_fn=None,
 176           normalizer_fn=None,
 177           kernel_size=[3, 3])
 178       return tf.expand_dims(
 179           tf.transpose(mask_predictions, perm=[0, 3, 1, 2]),
 180           axis=1,
 181           name='MaskPredictor')

 184 class ConvolutionalMaskHead(head.Head):
 185   """Convolutional class prediction head."""
 186 
 187   def __init__(self,
 188                is_training,
 189                num_classes,
 190                use_dropout,
 191                dropout_keep_prob,
 192                kernel_size,
 193                use_depthwise=False,
 194                mask_height=7,
 195                mask_width=7,
 196                masks_are_class_agnostic=False):
 197     """Constructor.
 198 
 199     Args:
 200       is_training: Indicates whether the BoxPredictor is in training mode.
 201       num_classes: Number of classes.
 202       use_dropout: Option to use dropout or not.  Note that a single dropout
 203         op is applied here prior to both box and class predictions, which stands
 204         in contrast to the ConvolutionalBoxPredictor below.
 205       dropout_keep_prob: Keep probability for dropout.
 206         This is only used if use_dropout is True.
 207       kernel_size: Size of final convolution kernel.  If the
 208         spatial resolution of the feature map is smaller than the kernel size,
 209         then the kernel size is automatically set to be
 210         min(feature_width, feature_height).
 211       use_depthwise: Whether to use depthwise convolutions for prediction
 212         steps. Default is False.
 213       mask_height: Desired output mask height. The default value is 7.
 214       mask_width: Desired output mask width. The default value is 7.
 215       masks_are_class_agnostic: Boolean determining if the mask-head is
 216         class-agnostic or not.
 217 
 218     Raises:
 219       ValueError: if min_depth > max_depth.
 220     """
 221     super(ConvolutionalMaskHead, self).__init__()
 222     self._is_training = is_training
 223     self._num_classes = num_classes
 224     self._use_dropout = use_dropout
 225     self._dropout_keep_prob = dropout_keep_prob
 226     self._kernel_size = kernel_size
 227     self._use_depthwise = use_depthwise
 228     self._mask_height = mask_height
 229     self._mask_width = mask_width
 230     self._masks_are_class_agnostic = masks_are_class_agnostic
 231 
 232   def predict(self, features, num_predictions_per_location):
 233     """Predicts boxes.
 234 
 235     Args:
 236       features: A float tensor of shape [batch_size, height, width, channels]
 237         containing image features.
 238       num_predictions_per_location: Number of box predictions to be made per
 239         spatial location.
 240 
 241     Returns:
 242       mask_predictions: A float tensors of shape
 243         [batch_size, num_anchors, num_masks, mask_height, mask_width]
 244         representing the mask predictions for the proposals.
 245     """
 246     image_feature = features
 247     # Add a slot for the background class.
 248     if self._masks_are_class_agnostic:
 249       num_masks = 1
 250     else:
 251       num_masks = self._num_classes
 252     num_mask_channels = num_masks * self._mask_height * self._mask_width
 253     net = image_feature
 254     if self._use_dropout:
 255       net = slim.dropout(net, keep_prob=self._dropout_keep_prob)
 256     if self._use_depthwise:
 257       mask_predictions = slim.separable_conv2d(
 258           net, None, [self._kernel_size, self._kernel_size],
 259           padding='SAME', depth_multiplier=1, stride=1,
 260           rate=1, scope='MaskPredictor_depthwise')
 261       mask_predictions = slim.conv2d(
 262           mask_predictions,
 263           num_predictions_per_location * num_mask_channels,
 264           [1, 1],
 265           activation_fn=None,
 266           normalizer_fn=None,
 267           normalizer_params=None,
 268           scope='MaskPredictor')
 269     else:
 270       mask_predictions = slim.conv2d(
 271           net,
 272           num_predictions_per_location * num_mask_channels,
 273           [self._kernel_size, self._kernel_size],
 274           activation_fn=None,
 275           normalizer_fn=None,
 276           normalizer_params=None,
 277           scope='MaskPredictor')
 278     batch_size = features.get_shape().as_list()[0]
 279     if batch_size is None:
 280       batch_size = tf.shape(features)[0]
 281     mask_predictions = tf.reshape(
 282         mask_predictions,
 283         [batch_size, -1, num_masks, self._mask_height, self._mask_width])
 284     return mask_predictions

 289 class WeightSharedConvolutionalMaskHead(head.Head):
 290   """Weight shared convolutional mask prediction head."""
 291 
 292   def __init__(self,
 293                num_classes,
 294                kernel_size=3,
 295                use_dropout=False,
 296                dropout_keep_prob=0.8,
 297                mask_height=7,
 298                mask_width=7,
 299                masks_are_class_agnostic=False):
 300     """Constructor.
 301 
 302     Args:
 303       num_classes: number of classes.  Note that num_classes *does not*
 304         include the background category, so if groundtruth labels take values
 305         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 306         assigned classification targets can range from {0,... K}).
 307       kernel_size: Size of final convolution kernel.
 308       use_dropout: Whether to apply dropout to class prediction head.
 309       dropout_keep_prob: Probability of keeping activiations.
 310       mask_height: Desired output mask height. The default value is 7.
 311       mask_width: Desired output mask width. The default value is 7.
 312       masks_are_class_agnostic: Boolean determining if the mask-head is
 313         class-agnostic or not.
 314     """
 315     super(WeightSharedConvolutionalMaskHead, self).__init__()
 316     self._num_classes = num_classes
 317     self._kernel_size = kernel_size
 318     self._use_dropout = use_dropout
 319     self._dropout_keep_prob = dropout_keep_prob
 320     self._mask_height = mask_height
 321     self._mask_width = mask_width
 322     self._masks_are_class_agnostic = masks_are_class_agnostic
 323 
 324   def predict(self, features, num_predictions_per_location):
 325     """Predicts boxes.
 326 
 327     Args:
 328       features: A float tensor of shape [batch_size, height, width, channels]
 329         containing image features.
 330       num_predictions_per_location: Number of box predictions to be made per
 331         spatial location.
 332 
 333     Returns:
 334       mask_predictions: A tensor of shape
 335         [batch_size, num_anchors, num_classes, mask_height, mask_width]
 336         representing the mask predictions for the proposals.
 337     """
 338     mask_predictions_net = features
 339     if self._masks_are_class_agnostic:
 340       num_masks = 1
 341     else:
 342       num_masks = self._num_classes
 343     num_mask_channels = num_masks * self._mask_height * self._mask_width
 344     if self._use_dropout:
 345       mask_predictions_net = slim.dropout(
 346           mask_predictions_net, keep_prob=self._dropout_keep_prob)
 347     mask_predictions = slim.conv2d(
 348         mask_predictions_net,
 349         num_predictions_per_location * num_mask_channels,
 350         [self._kernel_size, self._kernel_size],
 351         activation_fn=None, stride=1, padding='SAME',
 352         normalizer_fn=None,
 353         scope='MaskPredictor')
 354     batch_size = features.get_shape().as_list()[0]
 355     if batch_size is None:
 356       batch_size = tf.shape(features)[0]
 357     mask_predictions = tf.reshape(
 358         mask_predictions,
 359         [batch_size, -1, num_masks, self._mask_height, self._mask_width])
 360     return mask_predictions
