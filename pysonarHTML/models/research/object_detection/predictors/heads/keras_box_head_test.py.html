  26 class ConvolutionalKerasBoxHeadTest(test_case.TestCase):
  27 
  28   def _build_conv_hyperparams(self):
  29     conv_hyperparams = hyperparams_pb2.Hyperparams()
  30     conv_hyperparams_text_proto = """
  31     activation: NONE
  32       regularizer {
  33         l2_regularizer {
  34         }
  35       }
  36       initializer {
  37         truncated_normal_initializer {
  38         }
  39       }
  40     """
  41     text_format.Merge(conv_hyperparams_text_proto, conv_hyperparams)
  42     return hyperparams_builder.KerasLayerHyperparams(conv_hyperparams)
  43 
  44   def test_prediction_size_depthwise_false(self):
  45     conv_hyperparams = self._build_conv_hyperparams()
  46     box_prediction_head = keras_box_head.ConvolutionalBoxHead(
  47         is_training=True,
  48         box_code_size=4,
  49         kernel_size=3,
  50         conv_hyperparams=conv_hyperparams,
  51         freeze_batchnorm=False,
  52         num_predictions_per_location=1,
  53         use_depthwise=False)
  54     image_feature = tf.random_uniform(
  55         [64, 17, 19, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  56     box_encodings = box_prediction_head(image_feature)
  57     self.assertAllEqual([64, 323, 1, 4], box_encodings.get_shape().as_list())
  58 
  59   def test_prediction_size_depthwise_true(self):
  60     conv_hyperparams = self._build_conv_hyperparams()
  61     box_prediction_head = keras_box_head.ConvolutionalBoxHead(
  62         is_training=True,
  63         box_code_size=4,
  64         kernel_size=3,
  65         conv_hyperparams=conv_hyperparams,
  66         freeze_batchnorm=False,

  68         use_depthwise=True)
  69     image_feature = tf.random_uniform(
  70         [64, 17, 19, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  71     box_encodings = box_prediction_head(image_feature)
  72     self.assertAllEqual([64, 323, 1, 4], box_encodings.get_shape().as_list())

  75 class MaskRCNNKerasBoxHeadTest(test_case.TestCase):
  76 
  77   def _build_fc_hyperparams(
  78       self, op_type=hyperparams_pb2.Hyperparams.FC):
  79     hyperparams = hyperparams_pb2.Hyperparams()
  80     hyperparams_text_proto = """
  81       activation: NONE
  82       regularizer {
  83         l2_regularizer {
  84         }
  85       }
  86       initializer {
  87         truncated_normal_initializer {
  88         }
  89       }
  90     """
  91     text_format.Merge(hyperparams_text_proto, hyperparams)
  92     hyperparams.op = op_type
  93     return hyperparams_builder.KerasLayerHyperparams(hyperparams)
  94 
  95   def test_prediction_size(self):
  96     box_prediction_head = keras_box_head.MaskRCNNBoxHead(
  97         is_training=False,
  98         num_classes=20,
  99         fc_hyperparams=self._build_fc_hyperparams(),
 100         freeze_batchnorm=False,
 101         use_dropout=True,
 102         dropout_keep_prob=0.5,
 103         box_code_size=4,
 104         share_box_across_classes=False)
 105     roi_pooled_features = tf.random_uniform(
 106         [64, 7, 7, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
 107     prediction = box_prediction_head(roi_pooled_features)
 108     self.assertAllEqual([64, 1, 20, 4], prediction.get_shape().as_list())
