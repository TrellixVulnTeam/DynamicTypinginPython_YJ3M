  36 class ConvolutionalMaskHead(head.KerasHead):
  37   """Convolutional class prediction head."""
  38 
  39   def __init__(self,
  40                is_training,
  41                num_classes,
  42                use_dropout,
  43                dropout_keep_prob,
  44                kernel_size,
  45                num_predictions_per_location,
  46                conv_hyperparams,
  47                freeze_batchnorm,
  48                use_depthwise=False,
  49                mask_height=7,
  50                mask_width=7,
  51                masks_are_class_agnostic=False,
  52                name=None):
  53     """Constructor.
  54 
  55     Args:
  56       is_training: Indicates whether the BoxPredictor is in training mode.
  57       num_classes: Number of classes.
  58       use_dropout: Option to use dropout or not.  Note that a single dropout
  59         op is applied here prior to both box and class predictions, which stands
  60         in contrast to the ConvolutionalBoxPredictor below.
  61       dropout_keep_prob: Keep probability for dropout.
  62         This is only used if use_dropout is True.
  63       kernel_size: Size of final convolution kernel.  If the
  64         spatial resolution of the feature map is smaller than the kernel size,
  65         then the kernel size is automatically set to be
  66         min(feature_width, feature_height).
  67       num_predictions_per_location: Number of box predictions to be made per
  68         spatial location. Int specifying number of boxes per location.
  69       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  70         containing hyperparameters for convolution ops.
  71       freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
  72         training or not. When training with a small batch size (e.g. 1), it is
  73         desirable to freeze batch norm update and use pretrained batch norm
  74         params.
  75       use_depthwise: Whether to use depthwise convolutions for prediction
  76         steps. Default is False.
  77       mask_height: Desired output mask height. The default value is 7.
  78       mask_width: Desired output mask width. The default value is 7.
  79       masks_are_class_agnostic: Boolean determining if the mask-head is
  80         class-agnostic or not.
  81       name: A string name scope to assign to the model. If `None`, Keras
  82         will auto-generate one from the class name.
  83 
  84     Raises:
  85       ValueError: if min_depth > max_depth.
  86     """
  87     super(ConvolutionalMaskHead, self).__init__(name=name)
  88     self._is_training = is_training
  89     self._num_classes = num_classes
  90     self._use_dropout = use_dropout
  91     self._dropout_keep_prob = dropout_keep_prob
  92     self._kernel_size = kernel_size
  93     self._num_predictions_per_location = num_predictions_per_location
  94     self._use_depthwise = use_depthwise
  95     self._mask_height = mask_height
  96     self._mask_width = mask_width
  97     self._masks_are_class_agnostic = masks_are_class_agnostic
  98 
  99     self._mask_predictor_layers = []
 100 
 101     # Add a slot for the background class.
 102     if self._masks_are_class_agnostic:
 103       self._num_masks = 1
 104     else:
 105       self._num_masks = self._num_classes
 106 
 107     num_mask_channels = self._num_masks * self._mask_height * self._mask_width
 108 
 109     if self._use_dropout:
 110       self._mask_predictor_layers.append(
 111           # The Dropout layer's `training` parameter for the call method must
 112           # be set implicitly by the Keras set_learning_phase. The object
 113           # detection training code takes care of this.
 114           tf.keras.layers.Dropout(rate=1.0 - self._dropout_keep_prob))
 115     if self._use_depthwise:
 116       self._mask_predictor_layers.append(
 117           tf.keras.layers.DepthwiseConv2D(
 118               [self._kernel_size, self._kernel_size],
 119               padding='SAME',
 120               depth_multiplier=1,
 121               strides=1,
 122               dilation_rate=1,
 123               name='MaskPredictor_depthwise',
 124               **conv_hyperparams.params()))
 125       self._mask_predictor_layers.append(
 126           conv_hyperparams.build_batch_norm(
 127               training=(is_training and not freeze_batchnorm),
 128               name='MaskPredictor_depthwise_batchnorm'))
 129       self._mask_predictor_layers.append(
 130           conv_hyperparams.build_activation_layer(
 131               name='MaskPredictor_depthwise_activation'))
 132       self._mask_predictor_layers.append(
 133           tf.keras.layers.Conv2D(
 134               num_predictions_per_location * num_mask_channels, [1, 1],
 135               name='MaskPredictor',
 136               **conv_hyperparams.params(use_bias=True)))
 137     else:
 138       self._mask_predictor_layers.append(
 139           tf.keras.layers.Conv2D(
 140               num_predictions_per_location * num_mask_channels,
 141               [self._kernel_size, self._kernel_size],
 142               padding='SAME',
 143               name='MaskPredictor',
 144               **conv_hyperparams.params(use_bias=True)))
 145 
 146   def _predict(self, features):
 147     """Predicts boxes.
 148 
 149     Args:
 150       features: A float tensor of shape [batch_size, height, width, channels]
 151         containing image features.
 152 
 153     Returns:
 154       mask_predictions: A float tensors of shape
 155         [batch_size, num_anchors, num_masks, mask_height, mask_width]
 156         representing the mask predictions for the proposals.
 157     """
 158     mask_predictions = features
 159     for layer in self._mask_predictor_layers:
 160       mask_predictions = layer(mask_predictions)
 161     batch_size = features.get_shape().as_list()[0]
 162     if batch_size is None:
 163       batch_size = tf.shape(features)[0]
 164     mask_predictions = tf.reshape(
 165         mask_predictions,
 166         [batch_size, -1, self._num_masks, self._mask_height, self._mask_width])
 167     return mask_predictions

 170 class MaskRCNNMaskHead(head.KerasHead):
 171   """Mask RCNN mask prediction head.
 172 
 173   This is a piece of Mask RCNN which is responsible for predicting
 174   just the pixelwise foreground scores for regions within the boxes.
 175 
 176   Please refer to Mask RCNN paper:
 177   https://arxiv.org/abs/1703.06870
 178   """
 179 
 180   def __init__(self,
 181                is_training,
 182                num_classes,
 183                freeze_batchnorm,
 184                conv_hyperparams,
 185                mask_height=14,
 186                mask_width=14,
 187                mask_prediction_num_conv_layers=2,
 188                mask_prediction_conv_depth=256,
 189                masks_are_class_agnostic=False,
 190                convolve_then_upsample=False,
 191                name=None):
 192     """Constructor.
 193 
 194     Args:
 195       is_training: Indicates whether the Mask head is in training mode.
 196       num_classes: number of classes.  Note that num_classes *does not*
 197         include the background category, so if groundtruth labels take values
 198         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 199         assigned classification targets can range from {0,... K}).
 200       freeze_batchnorm: Whether to freeze batch norm parameters during
 201         training or not. When training with a small batch size (e.g. 1), it is
 202         desirable to freeze batch norm update and use pretrained batch norm
 203         params.
 204       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 205         containing hyperparameters for convolution ops.
 206       mask_height: Desired output mask height. The default value is 14.
 207       mask_width: Desired output mask width. The default value is 14.
 208       mask_prediction_num_conv_layers: Number of convolution layers applied to
 209         the image_features in mask prediction branch.
 210       mask_prediction_conv_depth: The depth for the first conv2d_transpose op
 211         applied to the image_features in the mask prediction branch. If set
 212         to 0, the depth of the convolution layers will be automatically chosen
 213         based on the number of object classes and the number of channels in the
 214         image features.
 215       masks_are_class_agnostic: Boolean determining if the mask-head is
 216         class-agnostic or not.
 217       convolve_then_upsample: Whether to apply convolutions on mask features
 218         before upsampling using nearest neighbor resizing. Otherwise, mask
 219         features are resized to [`mask_height`, `mask_width`] using bilinear
 220         resizing before applying convolutions.
 221       name: A string name scope to assign to the mask head. If `None`, Keras
 222         will auto-generate one from the class name.
 223     """
 224     super(MaskRCNNMaskHead, self).__init__(name=name)
 225     self._is_training = is_training
 226     self._freeze_batchnorm = freeze_batchnorm
 227     self._num_classes = num_classes
 228     self._conv_hyperparams = conv_hyperparams
 229     self._mask_height = mask_height
 230     self._mask_width = mask_width
 231     self._mask_prediction_num_conv_layers = mask_prediction_num_conv_layers
 232     self._mask_prediction_conv_depth = mask_prediction_conv_depth
 233     self._masks_are_class_agnostic = masks_are_class_agnostic
 234     self._convolve_then_upsample = convolve_then_upsample
 235 
 236     self._mask_predictor_layers = []
 237 
 238   def build(self, input_shapes):
 239     num_conv_channels = self._mask_prediction_conv_depth
 240     if num_conv_channels == 0:
 241       num_feature_channels = input_shapes.as_list()[3]
 242       num_conv_channels = self._get_mask_predictor_conv_depth(
 243           num_feature_channels, self._num_classes)
 244 
 245     for i in range(self._mask_prediction_num_conv_layers - 1):
 246       self._mask_predictor_layers.append(
 247           tf.keras.layers.Conv2D(
 248               num_conv_channels,
 249               [3, 3],
 250               padding='SAME',
 251               name='MaskPredictor_conv2d_{}'.format(i),
 252               **self._conv_hyperparams.params()))
 253       self._mask_predictor_layers.append(
 254           self._conv_hyperparams.build_batch_norm(
 255               training=(self._is_training and not self._freeze_batchnorm),
 256               name='MaskPredictor_batchnorm_{}'.format(i)))
 257       self._mask_predictor_layers.append(
 258           self._conv_hyperparams.build_activation_layer(
 259               name='MaskPredictor_activation_{}'.format(i)))
 260 
 261     if self._convolve_then_upsample:
 262       # Replace Transposed Convolution with a Nearest Neighbor upsampling step
 263       # followed by 3x3 convolution.
 264       height_scale = self._mask_height // shape_utils.get_dim_as_int(
 265           input_shapes[1])
 266       width_scale = self._mask_width // shape_utils.get_dim_as_int(
 267           input_shapes[2])
 268       # pylint: disable=g-long-lambda
 269       self._mask_predictor_layers.append(tf.keras.layers.Lambda(
 270           lambda features: ops.nearest_neighbor_upsampling(
 271               features, height_scale=height_scale, width_scale=width_scale)
 272       ))
 273       # pylint: enable=g-long-lambda
 274       self._mask_predictor_layers.append(
 275           tf.keras.layers.Conv2D(
 276               num_conv_channels,
 277               [3, 3],
 278               padding='SAME',
 279               name='MaskPredictor_upsample_conv2d',
 280               **self._conv_hyperparams.params()))
 281       self._mask_predictor_layers.append(
 282           self._conv_hyperparams.build_batch_norm(
 283               training=(self._is_training and not self._freeze_batchnorm),
 284               name='MaskPredictor_upsample_batchnorm'))
 285       self._mask_predictor_layers.append(
 286           self._conv_hyperparams.build_activation_layer(
 287               name='MaskPredictor_upsample_activation'))
 288 
 289     num_masks = 1 if self._masks_are_class_agnostic else self._num_classes
 290     self._mask_predictor_layers.append(
 291         tf.keras.layers.Conv2D(
 292             num_masks,
 293             [3, 3],
 294             padding='SAME',
 295             name='MaskPredictor_last_conv2d',
 296             **self._conv_hyperparams.params(use_bias=True)))
 297 
 298     self.built = True
 299 
 300   def _get_mask_predictor_conv_depth(self,
 301                                      num_feature_channels,
 302                                      num_classes,
 303                                      class_weight=3.0,
 304                                      feature_weight=2.0):
 305     """Computes the depth of the mask predictor convolutions.
 306 
 307     Computes the depth of the mask predictor convolutions given feature channels
 308     and number of classes by performing a weighted average of the two in
 309     log space to compute the number of convolution channels. The weights that
 310     are used for computing the weighted average do not need to sum to 1.
 311 
 312     Args:
 313       num_feature_channels: An integer containing the number of feature
 314         channels.
 315       num_classes: An integer containing the number of classes.
 316       class_weight: Class weight used in computing the weighted average.
 317       feature_weight: Feature weight used in computing the weighted average.
 318 
 319     Returns:
 320       An integer containing the number of convolution channels used by mask
 321         predictor.
 322     """
 323     num_feature_channels_log = math.log(float(num_feature_channels), 2.0)
 324     num_classes_log = math.log(float(num_classes), 2.0)
 325     weighted_num_feature_channels_log = (
 326         num_feature_channels_log * feature_weight)
 327     weighted_num_classes_log = num_classes_log * class_weight
 328     total_weight = feature_weight + class_weight
 329     num_conv_channels_log = round(
 330         (weighted_num_feature_channels_log + weighted_num_classes_log) /
 331         total_weight)
 332     return int(math.pow(2.0, num_conv_channels_log))
 333 
 334   def _predict(self, features):
 335     """Predicts pixelwise foreground scores for regions within the boxes.
 336 
 337     Args:
 338       features: A float tensor of shape [batch_size, height, width, channels]
 339         containing features for a batch of images.
 340 
 341     Returns:
 342       instance_masks: A float tensor of shape
 343           [batch_size, 1, num_classes, mask_height, mask_width].
 344     """
 345     if not self._convolve_then_upsample:
 346       features = tf.image.resize_bilinear(
 347           features, [self._mask_height, self._mask_width],
 348           align_corners=True)
 349 
 350     mask_predictions = features
 351     for layer in self._mask_predictor_layers:
 352       mask_predictions = layer(mask_predictions)
 353     return tf.expand_dims(
 354         tf.transpose(mask_predictions, perm=[0, 3, 1, 2]),
 355         axis=1,
 356         name='MaskPredictor')

 359 class WeightSharedConvolutionalMaskHead(head.KerasHead):
 360   """Weight shared convolutional mask prediction head based on Keras."""
 361 
 362   def __init__(self,
 363                num_classes,
 364                num_predictions_per_location,
 365                conv_hyperparams,
 366                kernel_size=3,
 367                use_dropout=False,
 368                dropout_keep_prob=0.8,
 369                mask_height=7,
 370                mask_width=7,
 371                masks_are_class_agnostic=False,
 372                name=None):
 373     """Constructor.
 374 
 375     Args:
 376       num_classes: number of classes.  Note that num_classes *does not*
 377         include the background category, so if groundtruth labels take values
 378         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 379         assigned classification targets can range from {0,... K}).
 380       num_predictions_per_location: Number of box predictions to be made per
 381         spatial location. Int specifying number of boxes per location.
 382       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 383         containing hyperparameters for convolution ops.
 384       kernel_size: Size of final convolution kernel.
 385       use_dropout: Whether to apply dropout to class prediction head.
 386       dropout_keep_prob: Probability of keeping activiations.
 387       mask_height: Desired output mask height. The default value is 7.
 388       mask_width: Desired output mask width. The default value is 7.
 389       masks_are_class_agnostic: Boolean determining if the mask-head is
 390         class-agnostic or not.
 391       name: A string name scope to assign to the model. If `None`, Keras
 392         will auto-generate one from the class name.
 393 
 394     Raises:
 395       ValueError: if min_depth > max_depth.
 396     """
 397     super(WeightSharedConvolutionalMaskHead, self).__init__(name=name)
 398     self._num_classes = num_classes
 399     self._num_predictions_per_location = num_predictions_per_location
 400     self._kernel_size = kernel_size
 401     self._use_dropout = use_dropout
 402     self._dropout_keep_prob = dropout_keep_prob
 403     self._mask_height = mask_height
 404     self._mask_width = mask_width
 405     self._masks_are_class_agnostic = masks_are_class_agnostic
 406 
 407     self._mask_predictor_layers = []
 408 
 409     if self._masks_are_class_agnostic:
 410       self._num_masks = 1
 411     else:
 412       self._num_masks = self._num_classes
 413     num_mask_channels = self._num_masks * self._mask_height * self._mask_width
 414 
 415     if self._use_dropout:
 416       self._mask_predictor_layers.append(
 417           tf.keras.layers.Dropout(rate=1.0 - self._dropout_keep_prob))
 418     self._mask_predictor_layers.append(
 419         tf.keras.layers.Conv2D(
 420             num_predictions_per_location * num_mask_channels,
 421             [self._kernel_size, self._kernel_size],
 422             padding='SAME',
 423             name='MaskPredictor',
 424             **conv_hyperparams.params(use_bias=True)))
 425 
 426   def _predict(self, features):
 427     """Predicts boxes.
 428 
 429     Args:
 430       features: A float tensor of shape [batch_size, height, width, channels]
 431         containing image features.
 432 
 433     Returns:
 434       mask_predictions: A tensor of shape
 435         [batch_size, num_anchors, num_classes, mask_height, mask_width]
 436         representing the mask predictions for the proposals.
 437     """
 438     mask_predictions = features
 439     for layer in self._mask_predictor_layers:
 440       mask_predictions = layer(mask_predictions)
 441     batch_size = features.get_shape().as_list()[0]
 442     if batch_size is None:
 443       batch_size = tf.shape(features)[0]
 444     mask_predictions = tf.reshape(
 445         mask_predictions,
 446         [batch_size, -1, self._num_masks, self._mask_height, self._mask_width])
 447     return mask_predictions
