  16 """Class Head.
  17 
  18 Contains Class prediction head classes for different meta architectures.
  19 All the class prediction heads have a predict function that receives the
  20 `features` as the first argument and returns class predictions with background.
  21 """
  22 import functools
  23 import tensorflow.compat.v1 as tf
  24 import tf_slim as slim
  25 
  26 from object_detection.predictors.heads import head
  27 
  28 
  29 class MaskRCNNClassHead(head.Head):
  30   """Mask RCNN class prediction head.
  31 
  32   Please refer to Mask RCNN paper:
  33   https://arxiv.org/abs/1703.06870
  34   """
  35 
  36   def __init__(self,
  37                is_training,
  38                num_class_slots,
  39                fc_hyperparams_fn,
  40                use_dropout,
  41                dropout_keep_prob,
  42                scope='ClassPredictor'):
  43     """Constructor.
  44 
  45     Args:
  46       is_training: Indicates whether the BoxPredictor is in training mode.
  47       num_class_slots: number of class slots. Note that num_class_slots may or
  48         may not include an implicit background category.
  49       fc_hyperparams_fn: A function to generate tf-slim arg_scope with
  50         hyperparameters for fully connected ops.
  51       use_dropout: Option to use dropout or not.  Note that a single dropout
  52         op is applied here prior to both box and class predictions, which stands
  53         in contrast to the ConvolutionalBoxPredictor below.
  54       dropout_keep_prob: Keep probability for dropout.
  55         This is only used if use_dropout is True.
  56       scope: Scope name for the convolution operation.
  57     """
  58     super(MaskRCNNClassHead, self).__init__()
  59     self._is_training = is_training
  60     self._num_class_slots = num_class_slots
  61     self._fc_hyperparams_fn = fc_hyperparams_fn
  62     self._use_dropout = use_dropout
  63     self._dropout_keep_prob = dropout_keep_prob
  64     self._scope = scope
  65 
  66   def predict(self, features, num_predictions_per_location=1):
  67     """Predicts boxes and class scores.
  68 
  69     Args:
  70       features: A float tensor of shape [batch_size, height, width, channels]
  71         containing features for a batch of images.
  72       num_predictions_per_location: Int containing number of predictions per
  73         location.
  74 
  75     Returns:
  76       class_predictions_with_background: A float tensor of shape
  77         [batch_size, 1, num_class_slots] representing the class predictions for
  78         the proposals.
  79 
  80     Raises:
  81       ValueError: If num_predictions_per_location is not 1.
  82     """
  83     if num_predictions_per_location != 1:
  84       raise ValueError('Only num_predictions_per_location=1 is supported')
  85     spatial_averaged_roi_pooled_features = tf.reduce_mean(
  86         features, [1, 2], keep_dims=True, name='AvgPool')
  87     flattened_roi_pooled_features = slim.flatten(
  88         spatial_averaged_roi_pooled_features)
  89     if self._use_dropout:
  90       flattened_roi_pooled_features = slim.dropout(
  91           flattened_roi_pooled_features,
  92           keep_prob=self._dropout_keep_prob,
  93           is_training=self._is_training)
  94 
  95     with slim.arg_scope(self._fc_hyperparams_fn()):
  96       class_predictions_with_background = slim.fully_connected(
  97           flattened_roi_pooled_features,
  98           self._num_class_slots,
  99           reuse=tf.AUTO_REUSE,
 100           activation_fn=None,
 101           scope=self._scope)
 102     class_predictions_with_background = tf.reshape(
 103         class_predictions_with_background,
 104         [-1, 1, self._num_class_slots])
 105     return class_predictions_with_background

 108 class ConvolutionalClassHead(head.Head):
 109   """Convolutional class prediction head."""
 110 
 111   def __init__(self,
 112                is_training,
 113                num_class_slots,
 114                use_dropout,
 115                dropout_keep_prob,
 116                kernel_size,
 117                apply_sigmoid_to_scores=False,
 118                class_prediction_bias_init=0.0,
 119                use_depthwise=False,
 120                scope='ClassPredictor'):
 121     """Constructor.
 122 
 123     Args:
 124       is_training: Indicates whether the BoxPredictor is in training mode.
 125       num_class_slots: number of class slots. Note that num_class_slots may or
 126         may not include an implicit background category.
 127       use_dropout: Option to use dropout or not.  Note that a single dropout
 128         op is applied here prior to both box and class predictions, which stands
 129         in contrast to the ConvolutionalBoxPredictor below.
 130       dropout_keep_prob: Keep probability for dropout.
 131         This is only used if use_dropout is True.
 132       kernel_size: Size of final convolution kernel.  If the
 133         spatial resolution of the feature map is smaller than the kernel size,
 134         then the kernel size is automatically set to be
 135         min(feature_width, feature_height).
 136       apply_sigmoid_to_scores: if True, apply the sigmoid on the output
 137         class_predictions.
 138       class_prediction_bias_init: constant value to initialize bias of the last
 139         conv2d layer before class prediction.
 140       use_depthwise: Whether to use depthwise convolutions for prediction
 141         steps. Default is False.
 142       scope: Scope name for the convolution operation.
 143 
 144     Raises:
 145       ValueError: if min_depth > max_depth.
 146       ValueError: if use_depthwise is True and kernel_size is 1.
 147     """
 148     if use_depthwise and (kernel_size == 1):
 149       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
 150 
 151     super(ConvolutionalClassHead, self).__init__()
 152     self._is_training = is_training
 153     self._num_class_slots = num_class_slots
 154     self._use_dropout = use_dropout
 155     self._dropout_keep_prob = dropout_keep_prob
 156     self._kernel_size = kernel_size
 157     self._apply_sigmoid_to_scores = apply_sigmoid_to_scores
 158     self._class_prediction_bias_init = class_prediction_bias_init
 159     self._use_depthwise = use_depthwise
 160     self._scope = scope
 161 
 162   def predict(self, features, num_predictions_per_location):
 163     """Predicts boxes.
 164 
 165     Args:
 166       features: A float tensor of shape [batch_size, height, width, channels]
 167         containing image features.
 168       num_predictions_per_location: Number of box predictions to be made per
 169         spatial location.
 170 
 171     Returns:
 172       class_predictions_with_background: A float tensors of shape
 173         [batch_size, num_anchors, num_class_slots] representing the class
 174         predictions for the proposals.
 175     """
 176     net = features
 177     if self._use_dropout:
 178       net = slim.dropout(net, keep_prob=self._dropout_keep_prob)
 179     if self._use_depthwise:
 180       depthwise_scope = self._scope + '_depthwise'
 181       class_predictions_with_background = slim.separable_conv2d(
 182           net, None, [self._kernel_size, self._kernel_size],
 183           padding='SAME', depth_multiplier=1, stride=1,
 184           rate=1, scope=depthwise_scope)
 185       class_predictions_with_background = slim.conv2d(
 186           class_predictions_with_background,
 187           num_predictions_per_location * self._num_class_slots, [1, 1],
 188           activation_fn=None,
 189           normalizer_fn=None,
 190           normalizer_params=None,
 191           scope=self._scope)
 192     else:
 193       class_predictions_with_background = slim.conv2d(
 194           net,
 195           num_predictions_per_location * self._num_class_slots,
 196           [self._kernel_size, self._kernel_size],
 197           activation_fn=None,
 198           normalizer_fn=None,
 199           normalizer_params=None,
 200           scope=self._scope,
 201           biases_initializer=tf.constant_initializer(
 202               self._class_prediction_bias_init))
 203     if self._apply_sigmoid_to_scores:
 204       class_predictions_with_background = tf.sigmoid(
 205           class_predictions_with_background)
 206     batch_size = features.get_shape().as_list()[0]
 207     if batch_size is None:
 208       batch_size = tf.shape(features)[0]
 209     class_predictions_with_background = tf.reshape(
 210         class_predictions_with_background,
 211         [batch_size, -1, self._num_class_slots])
 212     return class_predictions_with_background

 217 class WeightSharedConvolutionalClassHead(head.Head):
 218   """Weight shared convolutional class prediction head.
 219 
 220   This head allows sharing the same set of parameters (weights) when called more
 221   then once on different feature maps.
 222   """
 223 
 224   def __init__(self,
 225                num_class_slots,
 226                kernel_size=3,
 227                class_prediction_bias_init=0.0,
 228                use_dropout=False,
 229                dropout_keep_prob=0.8,
 230                use_depthwise=False,
 231                score_converter_fn=tf.identity,
 232                return_flat_predictions=True,
 233                scope='ClassPredictor'):
 234     """Constructor.
 235 
 236     Args:
 237       num_class_slots: number of class slots. Note that num_class_slots may or
 238         may not include an implicit background category.
 239       kernel_size: Size of final convolution kernel.
 240       class_prediction_bias_init: constant value to initialize bias of the last
 241         conv2d layer before class prediction.
 242       use_dropout: Whether to apply dropout to class prediction head.
 243       dropout_keep_prob: Probability of keeping activiations.
 244       use_depthwise: Whether to use depthwise convolutions for prediction
 245         steps. Default is False.
 246       score_converter_fn: Callable elementwise nonlinearity (that takes tensors
 247         as inputs and returns tensors).
 248       return_flat_predictions: If true, returns flattened prediction tensor
 249         of shape [batch, height * width * num_predictions_per_location,
 250         box_coder]. Otherwise returns the prediction tensor before reshaping,
 251         whose shape is [batch, height, width, num_predictions_per_location *
 252         num_class_slots].
 253       scope: Scope name for the convolution operation.
 254 
 255     Raises:
 256       ValueError: if use_depthwise is True and kernel_size is 1.
 257     """
 258     if use_depthwise and (kernel_size == 1):
 259       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
 260 
 261     super(WeightSharedConvolutionalClassHead, self).__init__()
 262     self._num_class_slots = num_class_slots
 263     self._kernel_size = kernel_size
 264     self._class_prediction_bias_init = class_prediction_bias_init
 265     self._use_dropout = use_dropout
 266     self._dropout_keep_prob = dropout_keep_prob
 267     self._use_depthwise = use_depthwise
 268     self._score_converter_fn = score_converter_fn
 269     self._return_flat_predictions = return_flat_predictions
 270     self._scope = scope
 271 
 272   def predict(self, features, num_predictions_per_location):
 273     """Predicts boxes.
 274 
 275     Args:
 276       features: A float tensor of shape [batch_size, height, width, channels]
 277         containing image features.
 278       num_predictions_per_location: Number of box predictions to be made per
 279         spatial location.
 280 
 281     Returns:
 282       class_predictions_with_background: A tensor of shape
 283         [batch_size, num_anchors, num_class_slots] representing the class
 284         predictions for the proposals, or a tensor of shape [batch, height,
 285         width, num_predictions_per_location * num_class_slots] representing
 286         class predictions before reshaping if self._return_flat_predictions is
 287         False.
 288     """
 289     class_predictions_net = features
 290     if self._use_dropout:
 291       class_predictions_net = slim.dropout(
 292           class_predictions_net, keep_prob=self._dropout_keep_prob)
 293     if self._use_depthwise:
 294       conv_op = functools.partial(slim.separable_conv2d, depth_multiplier=1)
 295     else:
 296       conv_op = slim.conv2d
 297     class_predictions_with_background = conv_op(
 298         class_predictions_net,
 299         num_predictions_per_location * self._num_class_slots,
 300         [self._kernel_size, self._kernel_size],
 301         activation_fn=None, stride=1, padding='SAME',
 302         normalizer_fn=None,
 303         biases_initializer=tf.constant_initializer(
 304             self._class_prediction_bias_init),
 305         scope=self._scope)
 306     batch_size = features.get_shape().as_list()[0]
 307     if batch_size is None:
 308       batch_size = tf.shape(features)[0]
 309     class_predictions_with_background = self._score_converter_fn(
 310         class_predictions_with_background)
 311     if self._return_flat_predictions:
 312       class_predictions_with_background = tf.reshape(
 313           class_predictions_with_background,
 314           [batch_size, -1, self._num_class_slots])
 315     return class_predictions_with_background
