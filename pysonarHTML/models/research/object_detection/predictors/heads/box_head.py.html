  16 """Box Head.
  17 
  18 Contains Box prediction head classes for different meta architectures.
  19 All the box prediction heads have a predict function that receives the
  20 `features` as the first argument and returns `box_encodings`.
  21 """
  22 import functools
  23 import tensorflow.compat.v1 as tf
  24 import tf_slim as slim
  25 
  26 from object_detection.predictors.heads import head
  27 
  28 
  29 class MaskRCNNBoxHead(head.Head):
  30   """Box prediction head.
  31 
  32   Please refer to Mask RCNN paper:
  33   https://arxiv.org/abs/1703.06870
  34   """
  35 
  36   def __init__(self,
  37                is_training,
  38                num_classes,
  39                fc_hyperparams_fn,
  40                use_dropout,
  41                dropout_keep_prob,
  42                box_code_size,
  43                share_box_across_classes=False):
  44     """Constructor.
  45 
  46     Args:
  47       is_training: Indicates whether the BoxPredictor is in training mode.
  48       num_classes: number of classes.  Note that num_classes *does not*
  49         include the background category, so if groundtruth labels take values
  50         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  51         assigned classification targets can range from {0,... K}).
  52       fc_hyperparams_fn: A function to generate tf-slim arg_scope with
  53         hyperparameters for fully connected ops.
  54       use_dropout: Option to use dropout or not.  Note that a single dropout
  55         op is applied here prior to both box and class predictions, which stands
  56         in contrast to the ConvolutionalBoxPredictor below.
  57       dropout_keep_prob: Keep probability for dropout.
  58         This is only used if use_dropout is True.
  59       box_code_size: Size of encoding for each box.
  60       share_box_across_classes: Whether to share boxes across classes rather
  61         than use a different box for each class.
  62     """
  63     super(MaskRCNNBoxHead, self).__init__()
  64     self._is_training = is_training
  65     self._num_classes = num_classes
  66     self._fc_hyperparams_fn = fc_hyperparams_fn
  67     self._use_dropout = use_dropout
  68     self._dropout_keep_prob = dropout_keep_prob
  69     self._box_code_size = box_code_size
  70     self._share_box_across_classes = share_box_across_classes
  71 
  72   def predict(self, features, num_predictions_per_location=1):
  73     """Predicts boxes.
  74 
  75     Args:
  76       features: A float tensor of shape [batch_size, height, width,
  77         channels] containing features for a batch of images.
  78       num_predictions_per_location: Int containing number of predictions per
  79         location.
  80 
  81     Returns:
  82       box_encodings: A float tensor of shape
  83         [batch_size, 1, num_classes, code_size] representing the location of the
  84         objects.
  85 
  86     Raises:
  87       ValueError: If num_predictions_per_location is not 1.
  88     """
  89     if num_predictions_per_location != 1:
  90       raise ValueError('Only num_predictions_per_location=1 is supported')
  91     spatial_averaged_roi_pooled_features = tf.reduce_mean(
  92         features, [1, 2], keep_dims=True, name='AvgPool')
  93     flattened_roi_pooled_features = slim.flatten(
  94         spatial_averaged_roi_pooled_features)
  95     if self._use_dropout:
  96       flattened_roi_pooled_features = slim.dropout(
  97           flattened_roi_pooled_features,
  98           keep_prob=self._dropout_keep_prob,
  99           is_training=self._is_training)
 100     number_of_boxes = 1
 101     if not self._share_box_across_classes:
 102       number_of_boxes = self._num_classes
 103 
 104     with slim.arg_scope(self._fc_hyperparams_fn()):
 105       box_encodings = slim.fully_connected(
 106           flattened_roi_pooled_features,
 107           number_of_boxes * self._box_code_size,
 108           reuse=tf.AUTO_REUSE,
 109           activation_fn=None,
 110           scope='BoxEncodingPredictor')
 111     box_encodings = tf.reshape(box_encodings,
 112                                [-1, 1, number_of_boxes, self._box_code_size])
 113     return box_encodings

 116 class ConvolutionalBoxHead(head.Head):
 117   """Convolutional box prediction head."""
 118 
 119   def __init__(self,
 120                is_training,
 121                box_code_size,
 122                kernel_size,
 123                use_depthwise=False,
 124                box_encodings_clip_range=None):
 125     """Constructor.
 126 
 127     Args:
 128       is_training: Indicates whether the BoxPredictor is in training mode.
 129       box_code_size: Size of encoding for each box.
 130       kernel_size: Size of final convolution kernel.  If the
 131         spatial resolution of the feature map is smaller than the kernel size,
 132         then the kernel size is automatically set to be
 133         min(feature_width, feature_height).
 134       use_depthwise: Whether to use depthwise convolutions for prediction
 135         steps. Default is False.
 136       box_encodings_clip_range: Min and max values for clipping box_encodings.
 137 
 138     Raises:
 139       ValueError: if min_depth > max_depth.
 140       ValueError: if use_depthwise is True and kernel_size is 1.
 141     """
 142     if use_depthwise and (kernel_size == 1):
 143       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
 144 
 145     super(ConvolutionalBoxHead, self).__init__()
 146     self._is_training = is_training
 147     self._box_code_size = box_code_size
 148     self._kernel_size = kernel_size
 149     self._use_depthwise = use_depthwise
 150     self._box_encodings_clip_range = box_encodings_clip_range
 151 
 152   def predict(self, features, num_predictions_per_location):
 153     """Predicts boxes.
 154 
 155     Args:
 156       features: A float tensor of shape [batch_size, height, width, channels]
 157         containing image features.
 158       num_predictions_per_location: Number of box predictions to be made per
 159         spatial location. Int specifying number of boxes per location.
 160 
 161     Returns:
 162       box_encodings: A float tensors of shape
 163         [batch_size, num_anchors, q, code_size] representing the location of
 164         the objects, where q is 1 or the number of classes.
 165     """
 166     net = features
 167     if self._use_depthwise:
 168       box_encodings = slim.separable_conv2d(
 169           net, None, [self._kernel_size, self._kernel_size],
 170           padding='SAME', depth_multiplier=1, stride=1,
 171           rate=1, scope='BoxEncodingPredictor_depthwise')
 172       box_encodings = slim.conv2d(
 173           box_encodings,
 174           num_predictions_per_location * self._box_code_size, [1, 1],
 175           activation_fn=None,
 176           normalizer_fn=None,
 177           normalizer_params=None,
 178           scope='BoxEncodingPredictor')
 179     else:
 180       box_encodings = slim.conv2d(
 181           net, num_predictions_per_location * self._box_code_size,
 182           [self._kernel_size, self._kernel_size],
 183           activation_fn=None,
 184           normalizer_fn=None,
 185           normalizer_params=None,
 186           scope='BoxEncodingPredictor')
 187     batch_size = features.get_shape().as_list()[0]
 188     if batch_size is None:
 189       batch_size = tf.shape(features)[0]
 190     # Clipping the box encodings to make the inference graph TPU friendly.
 191     if self._box_encodings_clip_range is not None:
 192       box_encodings = tf.clip_by_value(
 193           box_encodings, self._box_encodings_clip_range.min,
 194           self._box_encodings_clip_range.max)
 195     box_encodings = tf.reshape(box_encodings,
 196                                [batch_size, -1, 1, self._box_code_size])
 197     return box_encodings

 202 class WeightSharedConvolutionalBoxHead(head.Head):
 203   """Weight shared convolutional box prediction head.
 204 
 205   This head allows sharing the same set of parameters (weights) when called more
 206   then once on different feature maps.
 207   """
 208 
 209   def __init__(self,
 210                box_code_size,
 211                kernel_size=3,
 212                use_depthwise=False,
 213                box_encodings_clip_range=None,
 214                return_flat_predictions=True):
 215     """Constructor.
 216 
 217     Args:
 218       box_code_size: Size of encoding for each box.
 219       kernel_size: Size of final convolution kernel.
 220       use_depthwise: Whether to use depthwise convolutions for prediction steps.
 221         Default is False.
 222       box_encodings_clip_range: Min and max values for clipping box_encodings.
 223       return_flat_predictions: If true, returns flattened prediction tensor
 224         of shape [batch, height * width * num_predictions_per_location,
 225         box_coder]. Otherwise returns the prediction tensor before reshaping,
 226         whose shape is [batch, height, width, num_predictions_per_location *
 227         num_class_slots].
 228 
 229     Raises:
 230       ValueError: if use_depthwise is True and kernel_size is 1.
 231     """
 232     if use_depthwise and (kernel_size == 1):
 233       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
 234 
 235     super(WeightSharedConvolutionalBoxHead, self).__init__()
 236     self._box_code_size = box_code_size
 237     self._kernel_size = kernel_size
 238     self._use_depthwise = use_depthwise
 239     self._box_encodings_clip_range = box_encodings_clip_range
 240     self._return_flat_predictions = return_flat_predictions
 241 
 242   def predict(self, features, num_predictions_per_location):
 243     """Predicts boxes.
 244 
 245     Args:
 246       features: A float tensor of shape [batch_size, height, width, channels]
 247         containing image features.
 248       num_predictions_per_location: Number of box predictions to be made per
 249         spatial location.
 250 
 251     Returns:
 252       box_encodings: A float tensor of shape
 253         [batch_size, num_anchors, code_size] representing the location of
 254         the objects, or a float tensor of shape [batch, height, width,
 255         num_predictions_per_location * box_code_size] representing grid box
 256         location predictions if self._return_flat_predictions is False.
 257     """
 258     box_encodings_net = features
 259     if self._use_depthwise:
 260       conv_op = functools.partial(slim.separable_conv2d, depth_multiplier=1)
 261     else:
 262       conv_op = slim.conv2d
 263     box_encodings = conv_op(
 264         box_encodings_net,
 265         num_predictions_per_location * self._box_code_size,
 266         [self._kernel_size, self._kernel_size],
 267         activation_fn=None, stride=1, padding='SAME',
 268         normalizer_fn=None,
 269         scope='BoxPredictor')
 270     batch_size = features.get_shape().as_list()[0]
 271     if batch_size is None:
 272       batch_size = tf.shape(features)[0]
 273     # Clipping the box encodings to make the inference graph TPU friendly.
 274     if self._box_encodings_clip_range is not None:
 275       box_encodings = tf.clip_by_value(
 276           box_encodings, self._box_encodings_clip_range.min,
 277           self._box_encodings_clip_range.max)
 278     if self._return_flat_predictions:
 279       box_encodings = tf.reshape(box_encodings,
 280                                  [batch_size, -1, self._box_code_size])
 281     return box_encodings
