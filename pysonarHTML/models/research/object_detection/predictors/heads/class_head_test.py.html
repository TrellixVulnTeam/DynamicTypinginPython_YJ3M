  16 """Tests for object_detection.predictors.heads.class_head."""
  17 import tensorflow.compat.v1 as tf
  18 
  19 from google.protobuf import text_format
  20 from object_detection.builders import hyperparams_builder
  21 from object_detection.predictors.heads import class_head
  22 from object_detection.protos import hyperparams_pb2
  23 from object_detection.utils import test_case
  24 
  25 
  26 class MaskRCNNClassHeadTest(test_case.TestCase):
  27 
  28   def _build_arg_scope_with_hyperparams(self,
  29                                         op_type=hyperparams_pb2.Hyperparams.FC):
  30     hyperparams = hyperparams_pb2.Hyperparams()
  31     hyperparams_text_proto = """
  32       activation: NONE
  33       regularizer {
  34         l2_regularizer {
  35         }
  36       }
  37       initializer {
  38         truncated_normal_initializer {
  39         }
  40       }
  41     """
  42     text_format.Merge(hyperparams_text_proto, hyperparams)
  43     hyperparams.op = op_type
  44     return hyperparams_builder.build(hyperparams, is_training=True)
  45 
  46   def test_prediction_size(self):
  47     class_prediction_head = class_head.MaskRCNNClassHead(
  48         is_training=False,
  49         num_class_slots=20,
  50         fc_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
  51         use_dropout=True,
  52         dropout_keep_prob=0.5)
  53     roi_pooled_features = tf.random_uniform(
  54         [64, 7, 7, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  55     prediction = class_prediction_head.predict(
  56         features=roi_pooled_features, num_predictions_per_location=1)
  57     self.assertAllEqual([64, 1, 20], prediction.get_shape().as_list())
  58 
  59   def test_scope_name(self):
  60     expected_var_names = set([
  61         """ClassPredictor/weights""",
  62         """ClassPredictor/biases"""
  63     ])
  64 
  65     g = tf.Graph()
  66     with g.as_default():
  67       class_prediction_head = class_head.MaskRCNNClassHead(
  68           is_training=True,
  69           num_class_slots=20,
  70           fc_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
  71           use_dropout=True,
  72           dropout_keep_prob=0.5)
  73       image_feature = tf.random_uniform(
  74           [64, 17, 19, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  75       class_prediction_head.predict(
  76           features=image_feature,
  77           num_predictions_per_location=1)
  78       actual_variable_set = set([
  79           var.op.name for var in g.get_collection(tf.GraphKeys.GLOBAL_VARIABLES)
  80       ])
  81       self.assertSetEqual(expected_var_names, actual_variable_set)
