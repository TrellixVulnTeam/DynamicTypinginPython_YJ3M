  16 """Tests for object_detection.predictors.heads.mask_head."""
  17 import tensorflow.compat.v1 as tf
  18 
  19 from google.protobuf import text_format
  20 from object_detection.builders import hyperparams_builder
  21 from object_detection.predictors.heads import mask_head
  22 from object_detection.protos import hyperparams_pb2
  23 from object_detection.utils import test_case
  24 
  25 
  26 class MaskRCNNMaskHeadTest(test_case.TestCase):
  27 
  28   def _build_arg_scope_with_hyperparams(self,
  29                                         op_type=hyperparams_pb2.Hyperparams.FC):
  30     hyperparams = hyperparams_pb2.Hyperparams()
  31     hyperparams_text_proto = """
  32       activation: NONE
  33       regularizer {
  34         l2_regularizer {
  35         }
  36       }
  37       initializer {
  38         truncated_normal_initializer {
  39         }
  40       }
  41     """
  42     text_format.Merge(hyperparams_text_proto, hyperparams)
  43     hyperparams.op = op_type
  44     return hyperparams_builder.build(hyperparams, is_training=True)
  45 
  46   def test_prediction_size(self):
  47     mask_prediction_head = mask_head.MaskRCNNMaskHead(
  48         num_classes=20,
  49         conv_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
  50         mask_height=14,
  51         mask_width=14,
  52         mask_prediction_num_conv_layers=2,
  53         mask_prediction_conv_depth=256,
  54         masks_are_class_agnostic=False)
  55     roi_pooled_features = tf.random_uniform(
  56         [64, 7, 7, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  57     prediction = mask_prediction_head.predict(
  58         features=roi_pooled_features, num_predictions_per_location=1)
  59     self.assertAllEqual([64, 1, 20, 14, 14], prediction.get_shape().as_list())
  60 
  61   def test_prediction_size_with_convolve_then_upsample(self):
  62     mask_prediction_head = mask_head.MaskRCNNMaskHead(
  63         num_classes=20,
  64         conv_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
  65         mask_height=28,
  66         mask_width=28,
  67         mask_prediction_num_conv_layers=2,
  68         mask_prediction_conv_depth=256,
  69         masks_are_class_agnostic=True,
  70         convolve_then_upsample=True)
  71     roi_pooled_features = tf.random_uniform(
  72         [64, 14, 14, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  73     prediction = mask_prediction_head.predict(
  74         features=roi_pooled_features, num_predictions_per_location=1)
  75     self.assertAllEqual([64, 1, 1, 28, 28], prediction.get_shape().as_list())

  98   def test_prediction_size(self):
  99     mask_prediction_head = mask_head.ConvolutionalMaskHead(

 101         num_classes=20,
 102         use_dropout=True,
 103         dropout_keep_prob=0.5,
 104         kernel_size=3,
 105         mask_height=7,
 106         mask_width=7)

 111         num_predictions_per_location=1)
