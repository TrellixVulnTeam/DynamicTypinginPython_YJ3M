  26 class ConvolutionalMaskPredictorTest(test_case.TestCase):
  27 
  28   def _build_conv_hyperparams(self):
  29     conv_hyperparams = hyperparams_pb2.Hyperparams()
  30     conv_hyperparams_text_proto = """
  31     activation: NONE
  32       regularizer {
  33         l2_regularizer {
  34         }
  35       }
  36       initializer {
  37         truncated_normal_initializer {
  38         }
  39       }
  40     """
  41     text_format.Merge(conv_hyperparams_text_proto, conv_hyperparams)
  42     return hyperparams_builder.KerasLayerHyperparams(conv_hyperparams)
  43 
  44   def test_prediction_size_use_depthwise_false(self):
  45     conv_hyperparams = self._build_conv_hyperparams()
  46     mask_prediction_head = keras_mask_head.ConvolutionalMaskHead(
  47         is_training=True,
  48         num_classes=20,
  49         use_dropout=True,
  50         dropout_keep_prob=0.5,
  51         kernel_size=3,
  52         conv_hyperparams=conv_hyperparams,
  53         freeze_batchnorm=False,
  54         num_predictions_per_location=1,
  55         use_depthwise=False,
  56         mask_height=7,
  57         mask_width=7)
  58     image_feature = tf.random_uniform(
  59         [64, 17, 19, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  60     mask_predictions = mask_prediction_head(image_feature)
  61     self.assertAllEqual([64, 323, 20, 7, 7],
  62                         mask_predictions.get_shape().as_list())
  63

 127 class MaskRCNNMaskHeadTest(test_case.TestCase):
 128 
 129   def _build_conv_hyperparams(self,
 130                               op_type=hyperparams_pb2.Hyperparams.CONV):
 131     hyperparams = hyperparams_pb2.Hyperparams()
 132     hyperparams_text_proto = """
 133       activation: NONE
 134       regularizer {
 135         l2_regularizer {
 136         }
 137       }
 138       initializer {
 139         truncated_normal_initializer {
 140         }
 141       }
 142     """
 143     text_format.Merge(hyperparams_text_proto, hyperparams)
 144     hyperparams.op = op_type
 145     return hyperparams_builder.KerasLayerHyperparams(hyperparams)
 146 
 147   def test_prediction_size(self):
 148     mask_prediction_head = keras_mask_head.MaskRCNNMaskHead(
 149         is_training=True,
 150         num_classes=20,
 151         conv_hyperparams=self._build_conv_hyperparams(),
 152         freeze_batchnorm=False,
 153         mask_height=14,
 154         mask_width=14,
 155         mask_prediction_num_conv_layers=2,
 156         mask_prediction_conv_depth=256,
 157         masks_are_class_agnostic=False)
 158     roi_pooled_features = tf.random_uniform(
 159         [64, 7, 7, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
 160     prediction = mask_prediction_head(roi_pooled_features)
 161     self.assertAllEqual([64, 1, 20, 14, 14], prediction.get_shape().as_list())
 162 
 163   def test_prediction_size_with_convolve_then_upsample(self):
 164     mask_prediction_head = keras_mask_head.MaskRCNNMaskHead(
 165         is_training=True,
 166         num_classes=20,
 167         conv_hyperparams=self._build_conv_hyperparams(),
 168         freeze_batchnorm=False,
 169         mask_height=28,
 170         mask_width=28,
 171         mask_prediction_num_conv_layers=2,
 172         mask_prediction_conv_depth=256,
 173         masks_are_class_agnostic=True,
 174         convolve_then_upsample=True)
 175     roi_pooled_features = tf.random_uniform(
 176         [64, 14, 14, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
 177     prediction = mask_prediction_head(roi_pooled_features)
 178     self.assertAllEqual([64, 1, 1, 28, 28], prediction.get_shape().as_list())
