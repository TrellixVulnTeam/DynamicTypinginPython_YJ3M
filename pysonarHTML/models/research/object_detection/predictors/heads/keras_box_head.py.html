  16 """Box Head.
  17 
  18 Contains Box prediction head classes for different meta architectures.
  19 All the box prediction heads have a _predict function that receives the
  20 `features` as the first argument and returns `box_encodings`.
  21 """
  22 import tensorflow.compat.v1 as tf
  23 
  24 from object_detection.predictors.heads import head
  25 
  26 
  27 class ConvolutionalBoxHead(head.KerasHead):
  28   """Convolutional box prediction head."""
  29 
  30   def __init__(self,
  31                is_training,
  32                box_code_size,
  33                kernel_size,
  34                num_predictions_per_location,
  35                conv_hyperparams,
  36                freeze_batchnorm,
  37                use_depthwise=False,
  38                box_encodings_clip_range=None,
  39                name=None):
  40     """Constructor.
  41 
  42     Args:
  43       is_training: Indicates whether the BoxPredictor is in training mode.
  44       box_code_size: Size of encoding for each box.
  45       kernel_size: Size of final convolution kernel.  If the
  46         spatial resolution of the feature map is smaller than the kernel size,
  47         then the kernel size is automatically set to be
  48         min(feature_width, feature_height).
  49       num_predictions_per_location: Number of box predictions to be made per
  50         spatial location. Int specifying number of boxes per location.
  51       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  52         containing hyperparameters for convolution ops.
  53       freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
  54         training or not. When training with a small batch size (e.g. 1), it is
  55         desirable to freeze batch norm update and use pretrained batch norm
  56         params.
  57       use_depthwise: Whether to use depthwise convolutions for prediction
  58         steps. Default is False.
  59       box_encodings_clip_range: Min and max values for clipping box_encodings.
  60       name: A string name scope to assign to the model. If `None`, Keras
  61         will auto-generate one from the class name.
  62 
  63     Raises:
  64       ValueError: if min_depth > max_depth.
  65       ValueError: if use_depthwise is True and kernel_size is 1.
  66     """
  67     if use_depthwise and (kernel_size == 1):
  68       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
  69 
  70     super(ConvolutionalBoxHead, self).__init__(name=name)
  71     self._is_training = is_training
  72     self._box_code_size = box_code_size
  73     self._kernel_size = kernel_size
  74     self._num_predictions_per_location = num_predictions_per_location
  75     self._use_depthwise = use_depthwise
  76     self._box_encodings_clip_range = box_encodings_clip_range
  77 
  78     self._box_encoder_layers = []
  79 
  80     if self._use_depthwise:
  81       self._box_encoder_layers.append(
  82           tf.keras.layers.DepthwiseConv2D(
  83               [self._kernel_size, self._kernel_size],
  84               padding='SAME',
  85               depth_multiplier=1,
  86               strides=1,
  87               dilation_rate=1,
  88               name='BoxEncodingPredictor_depthwise',
  89               **conv_hyperparams.params()))
  90       self._box_encoder_layers.append(
  91           conv_hyperparams.build_batch_norm(
  92               training=(is_training and not freeze_batchnorm),
  93               name='BoxEncodingPredictor_depthwise_batchnorm'))
  94       self._box_encoder_layers.append(
  95           conv_hyperparams.build_activation_layer(
  96               name='BoxEncodingPredictor_depthwise_activation'))
  97       self._box_encoder_layers.append(
  98           tf.keras.layers.Conv2D(
  99               num_predictions_per_location * self._box_code_size, [1, 1],
 100               name='BoxEncodingPredictor',
 101               **conv_hyperparams.params(use_bias=True)))
 102     else:
 103       self._box_encoder_layers.append(
 104           tf.keras.layers.Conv2D(
 105               num_predictions_per_location * self._box_code_size,
 106               [self._kernel_size, self._kernel_size],
 107               padding='SAME',
 108               name='BoxEncodingPredictor',
 109               **conv_hyperparams.params(use_bias=True)))
 110 
 111   def _predict(self, features):
 112     """Predicts boxes.
 113 
 114     Args:
 115       features: A float tensor of shape [batch_size, height, width, channels]
 116         containing image features.
 117 
 118     Returns:
 119       box_encodings: A float tensor of shape
 120         [batch_size, num_anchors, q, code_size] representing the location of
 121         the objects, where q is 1 or the number of classes.
 122     """
 123     box_encodings = features
 124     for layer in self._box_encoder_layers:
 125       box_encodings = layer(box_encodings)
 126     batch_size = features.get_shape().as_list()[0]
 127     if batch_size is None:
 128       batch_size = tf.shape(features)[0]
 129     # Clipping the box encodings to make the inference graph TPU friendly.
 130     if self._box_encodings_clip_range is not None:
 131       box_encodings = tf.clip_by_value(
 132           box_encodings, self._box_encodings_clip_range.min,
 133           self._box_encodings_clip_range.max)
 134     box_encodings = tf.reshape(box_encodings,
 135                                [batch_size, -1, 1, self._box_code_size])
 136     return box_encodings

 139 class MaskRCNNBoxHead(head.KerasHead):
 140   """Box prediction head.
 141 
 142   This is a piece of Mask RCNN which is responsible for predicting
 143   just the box encodings.
 144 
 145   Please refer to Mask RCNN paper:
 146   https://arxiv.org/abs/1703.06870
 147   """
 148 
 149   def __init__(self,
 150                is_training,
 151                num_classes,
 152                fc_hyperparams,
 153                freeze_batchnorm,
 154                use_dropout,
 155                dropout_keep_prob,
 156                box_code_size,
 157                share_box_across_classes=False,
 158                name=None):
 159     """Constructor.
 160 
 161     Args:
 162       is_training: Indicates whether the BoxPredictor is in training mode.
 163       num_classes: number of classes.  Note that num_classes *does not*
 164         include the background category, so if groundtruth labels take values
 165         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 166         assigned classification targets can range from {0,... K}).
 167       fc_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 168         containing hyperparameters for fully connected dense ops.
 169       freeze_batchnorm: Whether to freeze batch norm parameters during
 170         training or not. When training with a small batch size (e.g. 1), it is
 171         desirable to freeze batch norm update and use pretrained batch norm
 172         params.
 173       use_dropout: Option to use dropout or not.  Note that a single dropout
 174         op is applied here prior to both box and class predictions, which stands
 175         in contrast to the ConvolutionalBoxPredictor below.
 176       dropout_keep_prob: Keep probability for dropout.
 177         This is only used if use_dropout is True.
 178       box_code_size: Size of encoding for each box.
 179       share_box_across_classes: Whether to share boxes across classes rather
 180         than use a different box for each class.
 181       name: A string name scope to assign to the box head. If `None`, Keras
 182         will auto-generate one from the class name.
 183     """
 184     super(MaskRCNNBoxHead, self).__init__(name=name)
 185     self._is_training = is_training
 186     self._num_classes = num_classes
 187     self._fc_hyperparams = fc_hyperparams
 188     self._freeze_batchnorm = freeze_batchnorm
 189     self._use_dropout = use_dropout
 190     self._dropout_keep_prob = dropout_keep_prob
 191     self._box_code_size = box_code_size
 192     self._share_box_across_classes = share_box_across_classes
 193 
 194     self._box_encoder_layers = [tf.keras.layers.Flatten()]
 195 
 196     if self._use_dropout:
 197       self._box_encoder_layers.append(
 198           tf.keras.layers.Dropout(rate=1.0 - self._dropout_keep_prob))
 199 
 200     self._number_of_boxes = 1
 201     if not self._share_box_across_classes:
 202       self._number_of_boxes = self._num_classes
 203 
 204     self._box_encoder_layers.append(
 205         tf.keras.layers.Dense(self._number_of_boxes * self._box_code_size,
 206                               name='BoxEncodingPredictor_dense'))
 207     self._box_encoder_layers.append(
 208         fc_hyperparams.build_batch_norm(training=(is_training and
 209                                                   not freeze_batchnorm),
 210                                         name='BoxEncodingPredictor_batchnorm'))
 211 
 212   def _predict(self, features):
 213     """Predicts box encodings.
 214 
 215     Args:
 216       features: A float tensor of shape [batch_size, height, width,
 217         channels] containing features for a batch of images.
 218 
 219     Returns:
 220       box_encodings: A float tensor of shape
 221         [batch_size, 1, num_classes, code_size] representing the location of the
 222         objects.
 223     """
 224     spatial_averaged_roi_pooled_features = tf.reduce_mean(
 225         features, [1, 2], keep_dims=True, name='AvgPool')
 226     net = spatial_averaged_roi_pooled_features
 227     for layer in self._box_encoder_layers:
 228       net = layer(net)
 229     box_encodings = tf.reshape(net,
 230                                [-1, 1,
 231                                 self._number_of_boxes,
 232                                 self._box_code_size])
 233     return box_encodings

 238 class WeightSharedConvolutionalBoxHead(head.KerasHead):
 239   """Weight shared convolutional box prediction head based on Keras.
 240 
 241   This head allows sharing the same set of parameters (weights) when called more
 242   then once on different feature maps.
 243   """
 244 
 245   def __init__(self,
 246                box_code_size,
 247                num_predictions_per_location,
 248                conv_hyperparams,
 249                kernel_size=3,
 250                use_depthwise=False,
 251                box_encodings_clip_range=None,
 252                return_flat_predictions=True,
 253                name=None):
 254     """Constructor.
 255 
 256     Args:
 257       box_code_size: Size of encoding for each box.
 258       num_predictions_per_location: Number of box predictions to be made per
 259         spatial location. Int specifying number of boxes per location.
 260       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 261         containing hyperparameters for convolution ops.
 262       kernel_size: Size of final convolution kernel.
 263       use_depthwise: Whether to use depthwise convolutions for prediction steps.
 264         Default is False.
 265       box_encodings_clip_range: Min and max values for clipping box_encodings.
 266       return_flat_predictions: If true, returns flattened prediction tensor
 267         of shape [batch, height * width * num_predictions_per_location,
 268         box_coder]. Otherwise returns the prediction tensor before reshaping,
 269         whose shape is [batch, height, width, num_predictions_per_location *
 270         num_class_slots].
 271       name: A string name scope to assign to the model. If `None`, Keras
 272         will auto-generate one from the class name.
 273 
 274     Raises:
 275       ValueError: if use_depthwise is True and kernel_size is 1.
 276     """
 277     if use_depthwise and (kernel_size == 1):
 278       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
 279 
 280     super(WeightSharedConvolutionalBoxHead, self).__init__(name=name)
 281     self._box_code_size = box_code_size
 282     self._kernel_size = kernel_size
 283     self._num_predictions_per_location = num_predictions_per_location
 284     self._use_depthwise = use_depthwise
 285     self._box_encodings_clip_range = box_encodings_clip_range
 286     self._return_flat_predictions = return_flat_predictions
 287 
 288     self._box_encoder_layers = []
 289 
 290     if self._use_depthwise:
 291       self._box_encoder_layers.append(
 292           tf.keras.layers.SeparableConv2D(
 293               num_predictions_per_location * self._box_code_size,
 294               [self._kernel_size, self._kernel_size],
 295               padding='SAME',
 296               name='BoxPredictor',
 297               **conv_hyperparams.params(use_bias=True)))
 298     else:
 299       self._box_encoder_layers.append(
 300           tf.keras.layers.Conv2D(
 301               num_predictions_per_location * self._box_code_size,
 302               [self._kernel_size, self._kernel_size],
 303               padding='SAME',
 304               name='BoxPredictor',
 305               **conv_hyperparams.params(use_bias=True)))
 306 
 307   def _predict(self, features):
 308     """Predicts boxes.
 309 
 310     Args:
 311       features: A float tensor of shape [batch_size, height, width, channels]
 312         containing image features.
 313 
 314     Returns:
 315       box_encodings: A float tensor of shape
 316         [batch_size, num_anchors, q, code_size] representing the location of
 317         the objects, where q is 1 or the number of classes.
 318     """
 319     box_encodings = features
 320     for layer in self._box_encoder_layers:
 321       box_encodings = layer(box_encodings)
 322     batch_size = features.get_shape().as_list()[0]
 323     if batch_size is None:
 324       batch_size = tf.shape(features)[0]
 325     # Clipping the box encodings to make the inference graph TPU friendly.
 326     if self._box_encodings_clip_range is not None:
 327       box_encodings = tf.clip_by_value(
 328           box_encodings, self._box_encodings_clip_range.min,
 329           self._box_encodings_clip_range.max)
 330     if self._return_flat_predictions:
 331       box_encodings = tf.reshape(box_encodings,
 332                                  [batch_size, -1, self._box_code_size])
 333     return box_encodings
