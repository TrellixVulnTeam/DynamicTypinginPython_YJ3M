  16 """Tests for object_detection.predictors.heads.box_head."""
  17 import tensorflow.compat.v1 as tf
  18 
  19 from google.protobuf import text_format
  20 from object_detection.builders import hyperparams_builder
  21 from object_detection.predictors.heads import box_head
  22 from object_detection.protos import hyperparams_pb2
  23 from object_detection.utils import test_case
  24 
  25 
  26 class MaskRCNNBoxHeadTest(test_case.TestCase):
  27 
  28   def _build_arg_scope_with_hyperparams(self,
  29                                         op_type=hyperparams_pb2.Hyperparams.FC):
  30     hyperparams = hyperparams_pb2.Hyperparams()
  31     hyperparams_text_proto = """
  32       activation: NONE
  33       regularizer {
  34         l2_regularizer {
  35         }
  36       }
  37       initializer {
  38         truncated_normal_initializer {
  39         }
  40       }
  41     """
  42     text_format.Merge(hyperparams_text_proto, hyperparams)
  43     hyperparams.op = op_type
  44     return hyperparams_builder.build(hyperparams, is_training=True)
  45 
  46   def test_prediction_size(self):
  47     box_prediction_head = box_head.MaskRCNNBoxHead(
  48         is_training=False,
  49         num_classes=20,
  50         fc_hyperparams_fn=self._build_arg_scope_with_hyperparams(),
  51         use_dropout=True,
  52         dropout_keep_prob=0.5,
  53         box_code_size=4,
  54         share_box_across_classes=False)
  55     roi_pooled_features = tf.random_uniform(
  56         [64, 7, 7, 1024], minval=-10.0, maxval=10.0, dtype=tf.float32)
  57     prediction = box_prediction_head.predict(
  58         features=roi_pooled_features, num_predictions_per_location=1)
  59     self.assertAllEqual([64, 1, 20, 4], prediction.get_shape().as_list())
