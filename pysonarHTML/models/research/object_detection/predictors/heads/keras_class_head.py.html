  16 """Class Head.
  17 
  18 Contains Class prediction head classes for different meta architectures.
  19 All the class prediction heads have a predict function that receives the
  20 `features` as the first argument and returns class predictions with background.
  21 """
  22 import tensorflow.compat.v1 as tf
  23 
  24 from object_detection.predictors.heads import head
  25 
  26 
  27 class ConvolutionalClassHead(head.KerasHead):
  28   """Convolutional class prediction head."""
  29 
  30   def __init__(self,
  31                is_training,
  32                num_class_slots,
  33                use_dropout,
  34                dropout_keep_prob,
  35                kernel_size,
  36                num_predictions_per_location,
  37                conv_hyperparams,
  38                freeze_batchnorm,
  39                class_prediction_bias_init=0.0,
  40                use_depthwise=False,
  41                name=None):
  42     """Constructor.
  43 
  44     Args:
  45       is_training: Indicates whether the BoxPredictor is in training mode.
  46       num_class_slots: number of class slots. Note that num_class_slots may or
  47         may not include an implicit background category.
  48       use_dropout: Option to use dropout or not.  Note that a single dropout
  49         op is applied here prior to both box and class predictions, which stands
  50         in contrast to the ConvolutionalBoxPredictor below.
  51       dropout_keep_prob: Keep probability for dropout.
  52         This is only used if use_dropout is True.
  53       kernel_size: Size of final convolution kernel.  If the
  54         spatial resolution of the feature map is smaller than the kernel size,
  55         then the kernel size is automatically set to be
  56         min(feature_width, feature_height).
  57       num_predictions_per_location: Number of box predictions to be made per
  58         spatial location. Int specifying number of boxes per location.
  59       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  60         containing hyperparameters for convolution ops.
  61       freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
  62         training or not. When training with a small batch size (e.g. 1), it is
  63         desirable to freeze batch norm update and use pretrained batch norm
  64         params.
  65       class_prediction_bias_init: constant value to initialize bias of the last
  66         conv2d layer before class prediction.
  67       use_depthwise: Whether to use depthwise convolutions for prediction
  68         steps. Default is False.
  69       name: A string name scope to assign to the model. If `None`, Keras
  70         will auto-generate one from the class name.
  71 
  72     Raises:
  73       ValueError: if min_depth > max_depth.
  74       ValueError: if use_depthwise is True and kernel_size is 1.
  75     """
  76     if use_depthwise and (kernel_size == 1):
  77       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
  78 
  79     super(ConvolutionalClassHead, self).__init__(name=name)
  80     self._is_training = is_training
  81     self._use_dropout = use_dropout
  82     self._dropout_keep_prob = dropout_keep_prob
  83     self._kernel_size = kernel_size
  84     self._class_prediction_bias_init = class_prediction_bias_init
  85     self._use_depthwise = use_depthwise
  86     self._num_class_slots = num_class_slots
  87 
  88     self._class_predictor_layers = []
  89 
  90     if self._use_dropout:
  91       self._class_predictor_layers.append(
  92           # The Dropout layer's `training` parameter for the call method must
  93           # be set implicitly by the Keras set_learning_phase. The object
  94           # detection training code takes care of this.
  95           tf.keras.layers.Dropout(rate=1.0 - self._dropout_keep_prob))
  96     if self._use_depthwise:
  97       self._class_predictor_layers.append(
  98           tf.keras.layers.DepthwiseConv2D(
  99               [self._kernel_size, self._kernel_size],
 100               padding='SAME',
 101               depth_multiplier=1,
 102               strides=1,
 103               dilation_rate=1,
 104               name='ClassPredictor_depthwise',
 105               **conv_hyperparams.params()))
 106       self._class_predictor_layers.append(
 107           conv_hyperparams.build_batch_norm(
 108               training=(is_training and not freeze_batchnorm),
 109               name='ClassPredictor_depthwise_batchnorm'))
 110       self._class_predictor_layers.append(
 111           conv_hyperparams.build_activation_layer(
 112               name='ClassPredictor_depthwise_activation'))
 113       self._class_predictor_layers.append(
 114           tf.keras.layers.Conv2D(
 115               num_predictions_per_location * self._num_class_slots, [1, 1],
 116               name='ClassPredictor',
 117               **conv_hyperparams.params(use_bias=True)))
 118     else:
 119       self._class_predictor_layers.append(
 120           tf.keras.layers.Conv2D(
 121               num_predictions_per_location * self._num_class_slots,
 122               [self._kernel_size, self._kernel_size],
 123               padding='SAME',
 124               name='ClassPredictor',
 125               bias_initializer=tf.constant_initializer(
 126                   self._class_prediction_bias_init),
 127               **conv_hyperparams.params(use_bias=True)))
 128 
 129   def _predict(self, features):
 130     """Predicts boxes.
 131 
 132     Args:
 133       features: A float tensor of shape [batch_size, height, width, channels]
 134         containing image features.
 135 
 136     Returns:
 137       class_predictions_with_background: A float tensor of shape
 138         [batch_size, num_anchors, num_class_slots] representing the class
 139         predictions for the proposals.
 140     """
 141     class_predictions_with_background = features
 142     for layer in self._class_predictor_layers:
 143       class_predictions_with_background = layer(
 144           class_predictions_with_background)
 145     batch_size = features.get_shape().as_list()[0]
 146     if batch_size is None:
 147       batch_size = tf.shape(features)[0]
 148     class_predictions_with_background = tf.reshape(
 149         class_predictions_with_background,
 150         [batch_size, -1, self._num_class_slots])
 151     return class_predictions_with_background

 154 class MaskRCNNClassHead(head.KerasHead):
 155   """Mask RCNN class prediction head.
 156 
 157   This is a piece of Mask RCNN which is responsible for predicting
 158   just the class scores of boxes.
 159 
 160   Please refer to Mask RCNN paper:
 161   https://arxiv.org/abs/1703.06870
 162   """
 163 
 164   def __init__(self,
 165                is_training,
 166                num_class_slots,
 167                fc_hyperparams,
 168                freeze_batchnorm,
 169                use_dropout,
 170                dropout_keep_prob,
 171                name=None):
 172     """Constructor.
 173 
 174     Args:
 175       is_training: Indicates whether the BoxPredictor is in training mode.
 176       num_class_slots: number of class slots. Note that num_class_slots may or
 177         may not include an implicit background category.
 178       fc_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 179         containing hyperparameters for fully connected dense ops.
 180       freeze_batchnorm: Whether to freeze batch norm parameters during
 181         training or not. When training with a small batch size (e.g. 1), it is
 182         desirable to freeze batch norm update and use pretrained batch norm
 183         params.
 184       use_dropout: Option to use dropout or not.  Note that a single dropout
 185         op is applied here prior to both box and class predictions, which stands
 186         in contrast to the ConvolutionalBoxPredictor below.
 187       dropout_keep_prob: Keep probability for dropout.
 188         This is only used if use_dropout is True.
 189       name: A string name scope to assign to the class head. If `None`, Keras
 190         will auto-generate one from the class name.
 191     """
 192     super(MaskRCNNClassHead, self).__init__(name=name)
 193     self._is_training = is_training
 194     self._freeze_batchnorm = freeze_batchnorm
 195     self._num_class_slots = num_class_slots
 196     self._fc_hyperparams = fc_hyperparams
 197     self._use_dropout = use_dropout
 198     self._dropout_keep_prob = dropout_keep_prob
 199 
 200     self._class_predictor_layers = [tf.keras.layers.Flatten()]
 201 
 202     if self._use_dropout:
 203       self._class_predictor_layers.append(
 204           tf.keras.layers.Dropout(rate=1.0 - self._dropout_keep_prob))
 205 
 206     self._class_predictor_layers.append(
 207         tf.keras.layers.Dense(self._num_class_slots,
 208                               name='ClassPredictor_dense'))
 209     self._class_predictor_layers.append(
 210         fc_hyperparams.build_batch_norm(training=(is_training and
 211                                                   not freeze_batchnorm),
 212                                         name='ClassPredictor_batchnorm'))
 213 
 214   def _predict(self, features):
 215     """Predicts the class scores for boxes.
 216 
 217     Args:
 218       features: A float tensor of shape [batch_size, height, width, channels]
 219         containing features for a batch of images.
 220 
 221     Returns:
 222       class_predictions_with_background: A float tensor of shape
 223         [batch_size, 1, num_class_slots] representing the class predictions for
 224         the proposals.
 225     """
 226     spatial_averaged_roi_pooled_features = tf.reduce_mean(
 227         features, [1, 2], keep_dims=True, name='AvgPool')
 228     net = spatial_averaged_roi_pooled_features
 229     for layer in self._class_predictor_layers:
 230       net = layer(net)
 231     class_predictions_with_background = tf.reshape(
 232         net,
 233         [-1, 1, self._num_class_slots])
 234     return class_predictions_with_background

 237 class WeightSharedConvolutionalClassHead(head.KerasHead):
 238   """Weight shared convolutional class prediction head.
 239 
 240   This head allows sharing the same set of parameters (weights) when called more
 241   then once on different feature maps.
 242   """
 243 
 244   def __init__(self,
 245                num_class_slots,
 246                num_predictions_per_location,
 247                conv_hyperparams,
 248                kernel_size=3,
 249                class_prediction_bias_init=0.0,
 250                use_dropout=False,
 251                dropout_keep_prob=0.8,
 252                use_depthwise=False,
 253                score_converter_fn=tf.identity,
 254                return_flat_predictions=True,
 255                name=None):
 256     """Constructor.
 257 
 258     Args:
 259       num_class_slots: number of class slots. Note that num_class_slots may or
 260         may not include an implicit background category.
 261       num_predictions_per_location: Number of box predictions to be made per
 262         spatial location. Int specifying number of boxes per location.
 263       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 264         containing hyperparameters for convolution ops.
 265       kernel_size: Size of final convolution kernel.
 266       class_prediction_bias_init: constant value to initialize bias of the last
 267         conv2d layer before class prediction.
 268       use_dropout: Whether to apply dropout to class prediction head.
 269       dropout_keep_prob: Probability of keeping activiations.
 270       use_depthwise: Whether to use depthwise convolutions for prediction
 271         steps. Default is False.
 272       score_converter_fn: Callable elementwise nonlinearity (that takes tensors
 273         as inputs and returns tensors).
 274       return_flat_predictions: If true, returns flattened prediction tensor
 275         of shape [batch, height * width * num_predictions_per_location,
 276         box_coder]. Otherwise returns the prediction tensor before reshaping,
 277         whose shape is [batch, height, width, num_predictions_per_location *
 278         num_class_slots].
 279       name: A string name scope to assign to the model. If `None`, Keras
 280         will auto-generate one from the class name.
 281 
 282     Raises:
 283       ValueError: if use_depthwise is True and kernel_size is 1.
 284     """
 285     if use_depthwise and (kernel_size == 1):
 286       raise ValueError('Should not use 1x1 kernel when using depthwise conv')
 287 
 288     super(WeightSharedConvolutionalClassHead, self).__init__(name=name)
 289     self._num_class_slots = num_class_slots
 290     self._kernel_size = kernel_size
 291     self._class_prediction_bias_init = class_prediction_bias_init
 292     self._use_dropout = use_dropout
 293     self._dropout_keep_prob = dropout_keep_prob
 294     self._use_depthwise = use_depthwise
 295     self._score_converter_fn = score_converter_fn
 296     self._return_flat_predictions = return_flat_predictions
 297 
 298     self._class_predictor_layers = []
 299 
 300     if self._use_dropout:
 301       self._class_predictor_layers.append(
 302           tf.keras.layers.Dropout(rate=1.0 - self._dropout_keep_prob))
 303     if self._use_depthwise:
 304       self._class_predictor_layers.append(
 305           tf.keras.layers.SeparableConv2D(
 306               num_predictions_per_location * self._num_class_slots,
 307               [self._kernel_size, self._kernel_size],
 308               padding='SAME',
 309               depth_multiplier=1,
 310               strides=1,
 311               name='ClassPredictor',
 312               bias_initializer=tf.constant_initializer(
 313                   self._class_prediction_bias_init),
 314               **conv_hyperparams.params(use_bias=True)))
 315     else:
 316       self._class_predictor_layers.append(
 317           tf.keras.layers.Conv2D(
 318               num_predictions_per_location * self._num_class_slots,
 319               [self._kernel_size, self._kernel_size],
 320               padding='SAME',
 321               name='ClassPredictor',
 322               bias_initializer=tf.constant_initializer(
 323                   self._class_prediction_bias_init),
 324               **conv_hyperparams.params(use_bias=True)))
 325 
 326   def _predict(self, features):
 327     """Predicts boxes.
 328 
 329     Args:
 330       features: A float tensor of shape [batch_size, height, width, channels]
 331         containing image features.
 332 
 333     Returns:
 334       class_predictions_with_background: A float tensor of shape
 335         [batch_size, num_anchors, num_class_slots] representing the class
 336         predictions for the proposals.
 337     """
 338     class_predictions_with_background = features
 339     for layer in self._class_predictor_layers:
 340       class_predictions_with_background = layer(
 341           class_predictions_with_background)
 342     batch_size = features.get_shape().as_list()[0]
 343     if batch_size is None:
 344       batch_size = tf.shape(features)[0]
 345     class_predictions_with_background = self._score_converter_fn(
 346         class_predictions_with_background)
 347     if self._return_flat_predictions:
 348       class_predictions_with_background = tf.reshape(
 349           class_predictions_with_background,
 350           [batch_size, -1, self._num_class_slots])
 351     return class_predictions_with_background
