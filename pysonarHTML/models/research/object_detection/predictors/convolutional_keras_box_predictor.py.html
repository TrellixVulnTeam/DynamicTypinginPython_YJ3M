  49 class ConvolutionalBoxPredictor(box_predictor.KerasBoxPredictor):
  50   """Convolutional Keras Box Predictor.
  51 
  52   Optionally add an intermediate 1x1 convolutional layer after features and
  53   predict in parallel branches box_encodings and
  54   class_predictions_with_background.
  55 
  56   Currently this box predictor assumes that predictions are "shared" across
  57   classes --- that is each anchor makes box predictions which do not depend
  58   on class.
  59   """
  60 
  61   def __init__(self,
  62                is_training,
  63                num_classes,
  64                box_prediction_heads,
  65                class_prediction_heads,
  66                other_heads,
  67                conv_hyperparams,
  68                num_layers_before_predictor,
  69                min_depth,
  70                max_depth,
  71                freeze_batchnorm,
  72                inplace_batchnorm_update,
  73                name=None):
  74     """Constructor.
  75 
  76     Args:
  77       is_training: Indicates whether the BoxPredictor is in training mode.
  78       num_classes: number of classes.  Note that num_classes *does not*
  79         include the background category, so if groundtruth labels take values
  80         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  81         assigned classification targets can range from {0,... K}).
  82       box_prediction_heads: A list of heads that predict the boxes.
  83       class_prediction_heads: A list of heads that predict the classes.
  84       other_heads: A dictionary mapping head names to lists of convolutional
  85         heads.
  86       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  87         containing hyperparameters for convolution ops.
  88       num_layers_before_predictor: Number of the additional conv layers before
  89         the predictor.
  90       min_depth: Minimum feature depth prior to predicting box encodings
  91         and class predictions.
  92       max_depth: Maximum feature depth prior to predicting box encodings
  93         and class predictions. If max_depth is set to 0, no additional
  94         feature map will be inserted before location and class predictions.
  95       freeze_batchnorm: Whether to freeze batch norm parameters during
  96         training or not. When training with a small batch size (e.g. 1), it is
  97         desirable to freeze batch norm update and use pretrained batch norm
  98         params.
  99       inplace_batchnorm_update: Whether to update batch norm moving average
 100         values inplace. When this is false train op must add a control
 101         dependency on tf.graphkeys.UPDATE_OPS collection in order to update
 102         batch norm statistics.
 103       name: A string name scope to assign to the model. If `None`, Keras
 104         will auto-generate one from the class name.
 105 
 106     Raises:
 107       ValueError: if min_depth > max_depth.
 108     """
 109     super(ConvolutionalBoxPredictor, self).__init__(
 110         is_training, num_classes, freeze_batchnorm=freeze_batchnorm,
 111         inplace_batchnorm_update=inplace_batchnorm_update,
 112         name=name)
 113     if min_depth > max_depth:
 114       raise ValueError('min_depth should be less than or equal to max_depth')
 115     if len(box_prediction_heads) != len(class_prediction_heads):
 116       raise ValueError('All lists of heads must be the same length.')
 117     for other_head_list in other_heads.values():
 118       if len(box_prediction_heads) != len(other_head_list):
 119         raise ValueError('All lists of heads must be the same length.')
 120 
 121     self._prediction_heads = {
 122         BOX_ENCODINGS: box_prediction_heads,
 123         CLASS_PREDICTIONS_WITH_BACKGROUND: class_prediction_heads,
 124     }
 125 
 126     if other_heads:
 127       self._prediction_heads.update(other_heads)
 128 
 129     # We generate a consistent ordering for the prediction head names,
 130     # So that all workers build the model in the exact same order
 131     self._sorted_head_names = sorted(self._prediction_heads.keys())
 132 
 133     self._conv_hyperparams = conv_hyperparams
 134     self._min_depth = min_depth
 135     self._max_depth = max_depth
 136     self._num_layers_before_predictor = num_layers_before_predictor
 137 
 138     self._shared_nets = []
 139 
 140   def build(self, input_shapes):
 141     """Creates the variables of the layer."""
 142     if len(input_shapes) != len(self._prediction_heads[BOX_ENCODINGS]):
 143       raise ValueError('This box predictor was constructed with %d heads,'
 144                        'but there are %d inputs.' %
 145                        (len(self._prediction_heads[BOX_ENCODINGS]),
 146                         len(input_shapes)))
 147     for stack_index, input_shape in enumerate(input_shapes):
 148       net = []
 149 
 150       # Add additional conv layers before the class predictor.
 151       features_depth = static_shape.get_depth(input_shape)
 152       depth = max(min(features_depth, self._max_depth), self._min_depth)
 153       tf.logging.info(
 154           'depth of additional conv before box predictor: {}'.format(depth))
 155 
 156       if depth > 0 and self._num_layers_before_predictor > 0:
 157         for i in range(self._num_layers_before_predictor):
 158           net.append(keras.Conv2D(depth, [1, 1],
 159                                   name='SharedConvolutions_%d/Conv2d_%d_1x1_%d'
 160                                   % (stack_index, i, depth),
 161                                   padding='SAME',
 162                                   **self._conv_hyperparams.params()))
 163           net.append(self._conv_hyperparams.build_batch_norm(
 164               training=(self._is_training and not self._freeze_batchnorm),
 165               name='SharedConvolutions_%d/Conv2d_%d_1x1_%d_norm'
 166               % (stack_index, i, depth)))
 167           net.append(self._conv_hyperparams.build_activation_layer(
 168               name='SharedConvolutions_%d/Conv2d_%d_1x1_%d_activation'
 169               % (stack_index, i, depth),
 170           ))
 171       # Until certain bugs are fixed in checkpointable lists,
 172       # this net must be appended only once it's been filled with layers
 173       self._shared_nets.append(net)
 174     self.built = True
 175 
 176   def _predict(self, image_features, **kwargs):
 177     """Computes encoded object locations and corresponding confidences.
 178 
 179     Args:
 180       image_features: A list of float tensors of shape [batch_size, height_i,
 181         width_i, channels_i] containing features for a batch of images.
 182       **kwargs: Unused Keyword args
 183 
 184     Returns:
 185       box_encodings: A list of float tensors of shape
 186         [batch_size, num_anchors_i, q, code_size] representing the location of
 187         the objects, where q is 1 or the number of classes. Each entry in the
 188         list corresponds to a feature map in the input `image_features` list.
 189       class_predictions_with_background: A list of float tensors of shape
 190         [batch_size, num_anchors_i, num_classes + 1] representing the class
 191         predictions for the proposals. Each entry in the list corresponds to a
 192         feature map in the input `image_features` list.
 193     """
 194     predictions = collections.defaultdict(list)
 195 
 196     for (index, net) in enumerate(image_features):
 197 
 198       # Apply shared conv layers before the head predictors.
 199       for layer in self._shared_nets[index]:
 200         net = layer(net)
 201 
 202       for head_name in self._sorted_head_names:
 203         head_obj = self._prediction_heads[head_name][index]
 204         prediction = head_obj(net)
 205         predictions[head_name].append(prediction)
 206 
 207     return predictions

 210 class WeightSharedConvolutionalBoxPredictor(box_predictor.KerasBoxPredictor):
 211   """Convolutional Box Predictor with weight sharing based on Keras.
 212 
 213   Defines the box predictor as defined in
 214   https://arxiv.org/abs/1708.02002. This class differs from
 215   ConvolutionalBoxPredictor in that it shares weights and biases while
 216   predicting from different feature maps. However, batch_norm parameters are not
 217   shared because the statistics of the activations vary among the different
 218   feature maps.
 219 
 220   Also note that separate multi-layer towers are constructed for the box
 221   encoding and class predictors respectively.
 222   """
 223 
 224   def __init__(self,
 225                is_training,
 226                num_classes,
 227                box_prediction_head,
 228                class_prediction_head,
 229                other_heads,
 230                conv_hyperparams,
 231                depth,
 232                num_layers_before_predictor,
 233                freeze_batchnorm,
 234                inplace_batchnorm_update,
 235                kernel_size=3,
 236                apply_batch_norm=False,
 237                share_prediction_tower=False,
 238                use_depthwise=False,
 239                name=None):
 240     """Constructor.
 241 
 242     Args:
 243       is_training: Indicates whether the BoxPredictor is in training mode.
 244       num_classes: number of classes.  Note that num_classes *does not*
 245         include the background category, so if groundtruth labels take values
 246         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 247         assigned classification targets can range from {0,... K}).
 248       box_prediction_head: The head that predicts the boxes.
 249       class_prediction_head: The head that predicts the classes.
 250       other_heads: A dictionary mapping head names to convolutional
 251         head classes.
 252       conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 253         containing hyperparameters for convolution ops.
 254       depth: depth of conv layers.
 255       num_layers_before_predictor: Number of the additional conv layers before
 256         the predictor.
 257       freeze_batchnorm: Whether to freeze batch norm parameters during
 258         training or not. When training with a small batch size (e.g. 1), it is
 259         desirable to freeze batch norm update and use pretrained batch norm
 260         params.
 261       inplace_batchnorm_update: Whether to update batch norm moving average
 262         values inplace. When this is false train op must add a control
 263         dependency on tf.graphkeys.UPDATE_OPS collection in order to update
 264         batch norm statistics.
 265       kernel_size: Size of final convolution kernel.
 266       apply_batch_norm: Whether to apply batch normalization to conv layers in
 267         this predictor.
 268       share_prediction_tower: Whether to share the multi-layer tower among box
 269         prediction head, class prediction head and other heads.
 270       use_depthwise: Whether to use depthwise separable conv2d instead of
 271        regular conv2d.
 272       name: A string name scope to assign to the model. If `None`, Keras
 273         will auto-generate one from the class name.
 274     """
 275     super(WeightSharedConvolutionalBoxPredictor, self).__init__(
 276         is_training, num_classes, freeze_batchnorm=freeze_batchnorm,
 277         inplace_batchnorm_update=inplace_batchnorm_update,
 278         name=name)
 279 
 280     self._box_prediction_head = box_prediction_head
 281     self._prediction_heads = {
 282         CLASS_PREDICTIONS_WITH_BACKGROUND: class_prediction_head,
 283     }
 284     if other_heads:
 285       self._prediction_heads.update(other_heads)
 286     # We generate a consistent ordering for the prediction head names,
 287     # so that all workers build the model in the exact same order.
 288     self._sorted_head_names = sorted(self._prediction_heads.keys())
 289 
 290     self._conv_hyperparams = conv_hyperparams
 291     self._depth = depth
 292     self._num_layers_before_predictor = num_layers_before_predictor
 293     self._kernel_size = kernel_size
 294     self._apply_batch_norm = apply_batch_norm
 295     self._share_prediction_tower = share_prediction_tower
 296     self._use_depthwise = use_depthwise
 297 
 298     # Additional projection layers to bring all feature maps to uniform
 299     # channels.
 300     self._additional_projection_layers = []
 301     # The base tower layers for each head.
 302     self._base_tower_layers_for_heads = {
 303         BOX_ENCODINGS: [],
 304         CLASS_PREDICTIONS_WITH_BACKGROUND: [],
 305     }
 306     for head_name in other_heads.keys():
 307       self._base_tower_layers_for_heads[head_name] = []
 308 
 309     # A dict maps the tower_name_scope of each head to the shared conv layers in
 310     # the base tower for different feature map levels.
 311     self._head_scope_conv_layers = {}
 312 
 313   def _insert_additional_projection_layer(
 314       self, inserted_layer_counter, target_channel):
 315     projection_layers = []
 316     if inserted_layer_counter >= 0:
 317       use_bias = False if self._apply_batch_norm else True
 318       projection_layers.append(keras.Conv2D(
 319           target_channel, [1, 1], strides=1, padding='SAME',
 320           name='ProjectionLayer/conv2d_{}'.format(inserted_layer_counter),
 321           **self._conv_hyperparams.params(use_bias=use_bias)))
 322       if self._apply_batch_norm:
 323         projection_layers.append(self._conv_hyperparams.build_batch_norm(
 324             training=(self._is_training and not self._freeze_batchnorm),
 325             name='ProjectionLayer/conv2d_{}/BatchNorm'.format(
 326                 inserted_layer_counter)))
 327       inserted_layer_counter += 1
 328     return inserted_layer_counter, projection_layers
 329 
 330   def _compute_base_tower(self, tower_name_scope, feature_index):
 331     conv_layers = []
 332     batch_norm_layers = []
 333     activation_layers = []
 334     use_bias = False if self._apply_batch_norm else True
 335     for additional_conv_layer_idx in range(self._num_layers_before_predictor):
 336       layer_name = '{}/conv2d_{}'.format(
 337           tower_name_scope, additional_conv_layer_idx)
 338       if tower_name_scope not in self._head_scope_conv_layers:
 339         if self._use_depthwise:
 340           kwargs = self._conv_hyperparams.params(use_bias=use_bias)
 341           # Both the regularizer and initializer apply to the depthwise layer,
 342           # so we remap the kernel_* to depthwise_* here.
 343           kwargs['depthwise_regularizer'] = kwargs['kernel_regularizer']
 344           kwargs['depthwise_initializer'] = kwargs['kernel_initializer']
 345           conv_layers.append(
 346               tf.keras.layers.SeparableConv2D(
 347                   self._depth, [self._kernel_size, self._kernel_size],
 348                   padding='SAME',
 349                   name=layer_name,
 350                   **kwargs))
 351         else:
 352           conv_layers.append(
 353               tf.keras.layers.Conv2D(
 354                   self._depth,
 355                   [self._kernel_size, self._kernel_size],
 356                   padding='SAME',
 357                   name=layer_name,
 358                   **self._conv_hyperparams.params(use_bias=use_bias)))
 359       # Each feature gets a separate batchnorm parameter even though they share
 360       # the same convolution weights.
 361       if self._apply_batch_norm:
 362         batch_norm_layers.append(self._conv_hyperparams.build_batch_norm(
 363             training=(self._is_training and not self._freeze_batchnorm),
 364             name='{}/conv2d_{}/BatchNorm/feature_{}'.format(
 365                 tower_name_scope, additional_conv_layer_idx, feature_index)))
 366       activation_layers.append(tf.keras.layers.Lambda(tf.nn.relu6))
 367 
 368     # Set conv layers as the shared conv layers for different feature maps with
 369     # the same tower_name_scope.
 370     if tower_name_scope in self._head_scope_conv_layers:
 371       conv_layers = self._head_scope_conv_layers[tower_name_scope]
 372 
 373     # Stack the base_tower_layers in the order of conv_layer, batch_norm_layer
 374     # and activation_layer
 375     base_tower_layers = []
 376     for i in range(self._num_layers_before_predictor):
 377       base_tower_layers.extend([conv_layers[i]])
 378       if self._apply_batch_norm:
 379         base_tower_layers.extend([batch_norm_layers[i]])
 380       base_tower_layers.extend([activation_layers[i]])
 381     return conv_layers, base_tower_layers
 382 
 383   def build(self, input_shapes):
 384     """Creates the variables of the layer."""
 385     feature_channels = [
 386         shape_utils.get_dim_as_int(input_shape[3])
 387         for input_shape in input_shapes
 388     ]
 389     has_different_feature_channels = len(set(feature_channels)) > 1
 390     if has_different_feature_channels:
 391       inserted_layer_counter = 0
 392       target_channel = max(set(feature_channels), key=feature_channels.count)
 393       tf.logging.info('Not all feature maps have the same number of '
 394                       'channels, found: {}, appending additional projection '
 395                       'layers to bring all feature maps to uniformly have {} '
 396                       'channels.'.format(feature_channels, target_channel))
 397     else:
 398       # Place holder variables if has_different_feature_channels is False.
 399       target_channel = -1
 400       inserted_layer_counter = -1
 401 
 402     def _build_layers(tower_name_scope, feature_index):
 403       conv_layers, base_tower_layers = self._compute_base_tower(
 404           tower_name_scope=tower_name_scope, feature_index=feature_index)
 405       if tower_name_scope not in self._head_scope_conv_layers:
 406         self._head_scope_conv_layers[tower_name_scope] = conv_layers
 407       return base_tower_layers
 408 
 409     for feature_index in range(len(input_shapes)):
 410       # Additional projection layers should not be shared as input channels
 411       # (and thus weight shapes) are different
 412       inserted_layer_counter, projection_layers = (
 413           self._insert_additional_projection_layer(
 414               inserted_layer_counter, target_channel))
 415       self._additional_projection_layers.append(projection_layers)
 416 
 417       if self._share_prediction_tower:
 418         box_tower_scope = 'PredictionTower'
 419       else:
 420         box_tower_scope = 'BoxPredictionTower'
 421       # For box tower base
 422       box_tower_layers = _build_layers(box_tower_scope, feature_index)
 423       self._base_tower_layers_for_heads[BOX_ENCODINGS].append(box_tower_layers)
 424 
 425       for head_name in self._sorted_head_names:
 426         if head_name == CLASS_PREDICTIONS_WITH_BACKGROUND:
 427           tower_name_scope = 'ClassPredictionTower'
 428         else:
 429           tower_name_scope = '{}PredictionTower'.format(head_name)
 430         box_tower_layers = _build_layers(tower_name_scope, feature_index)
 431         self._base_tower_layers_for_heads[head_name].append(box_tower_layers)
 432 
 433     self.built = True
 434 
 435   def _predict(self, image_features, **kwargs):
 436     """Computes encoded object locations and corresponding confidences.
 437 
 438     Args:
 439       image_features: A list of float tensors of shape [batch_size, height_i,
 440         width_i, channels_i] containing features for a batch of images.
 441       **kwargs: Unused Keyword args
 442 
 443     Returns:
 444       box_encodings: A list of float tensors of shape
 445         [batch_size, num_anchors_i, q, code_size] representing the location of
 446         the objects, where q is 1 or the number of classes. Each entry in the
 447         list corresponds to a feature map in the input `image_features` list.
 448       class_predictions_with_background: A list of float tensors of shape
 449         [batch_size, num_anchors_i, num_classes + 1] representing the class
 450         predictions for the proposals. Each entry in the list corresponds to a
 451         feature map in the input `image_features` list.
 452     """
 453     predictions = collections.defaultdict(list)
 454 
 455     def _apply_layers(base_tower_layers, image_feature):
 456       for layer in base_tower_layers:
 457         image_feature = layer(image_feature)
 458       return image_feature
 459 
 460     for (index, image_feature) in enumerate(image_features):
 461       # Apply additional projection layers to image features
 462       for layer in self._additional_projection_layers[index]:
 463         image_feature = layer(image_feature)
 464 
 465       # Apply box tower layers.
 466       box_tower_feature = _apply_layers(
 467           self._base_tower_layers_for_heads[BOX_ENCODINGS][index],
 468           image_feature)
 469       box_encodings = self._box_prediction_head(box_tower_feature)
 470       predictions[BOX_ENCODINGS].append(box_encodings)
 471 
 472       for head_name in self._sorted_head_names:
 473         head_obj = self._prediction_heads[head_name]
 474         if self._share_prediction_tower:
 475           head_tower_feature = box_tower_feature
 476         else:
 477           head_tower_feature = _apply_layers(
 478               self._base_tower_layers_for_heads[head_name][index],
 479               image_feature)
 480         prediction = head_obj(head_tower_feature)
 481         predictions[head_name].append(prediction)
 482     return predictions
