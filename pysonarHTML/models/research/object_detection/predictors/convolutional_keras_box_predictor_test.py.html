  49   def test_get_boxes_for_five_aspect_ratios_per_location(self):
  50     def graph_fn(image_features):
  51       conv_box_predictor = (
  52           box_predictor_builder.build_convolutional_keras_box_predictor(
  53               is_training=False,
  54               num_classes=0,
  55               conv_hyperparams=self._build_conv_hyperparams(),
  56               freeze_batchnorm=False,
  57               inplace_batchnorm_update=False,
  58               num_predictions_per_location_list=[5],
  59               min_depth=0,
  60               max_depth=32,
  61               num_layers_before_predictor=1,
  62               use_dropout=True,
  63               dropout_keep_prob=0.8,
  64               kernel_size=1,
  65               box_code_size=4
  66           ))
  67       box_predictions = conv_box_predictor([image_features])
  68       box_encodings = tf.concat(
  69           box_predictions[box_predictor.BOX_ENCODINGS], axis=1)
  70       objectness_predictions = tf.concat(
  71           box_predictions[box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND],
  72           axis=1)
  73       return (box_encodings, objectness_predictions)
  74     image_features = np.random.rand(4, 8, 8, 64).astype(np.float32)

 843   def test_other_heads_predictions(self):
 844     box_code_size = 4
 845     num_classes_without_background = 3
 846     other_head_name = 'Mask'
 847     mask_height = 5
 848     mask_width = 5
 849     num_predictions_per_location = 5
 850 
 851     def graph_fn(image_features):
 852       box_prediction_head = keras_box_head.WeightSharedConvolutionalBoxHead(
 853           box_code_size=box_code_size,
 854           conv_hyperparams=self._build_conv_hyperparams(),
 855           num_predictions_per_location=num_predictions_per_location)
 856       class_prediction_head = keras_class_head.WeightSharedConvolutionalClassHead(
 857           num_class_slots=num_classes_without_background + 1,
 858           conv_hyperparams=self._build_conv_hyperparams(),
 859           num_predictions_per_location=num_predictions_per_location)
 860       other_heads = {
 861           other_head_name:
 862               keras_mask_head.WeightSharedConvolutionalMaskHead(
 863                   num_classes=num_classes_without_background,
 864                   conv_hyperparams=self._build_conv_hyperparams(),
 865                   num_predictions_per_location=num_predictions_per_location,
 866                   mask_height=mask_height,
 867                   mask_width=mask_width)
 868       }
 869 
 870       conv_box_predictor = box_predictor.WeightSharedConvolutionalBoxPredictor(
 871           is_training=False,
 872           num_classes=num_classes_without_background,
 873           box_prediction_head=box_prediction_head,
 874           class_prediction_head=class_prediction_head,
 875           other_heads=other_heads,
 876           conv_hyperparams=self._build_conv_hyperparams(),
 877           freeze_batchnorm=False,
 878           inplace_batchnorm_update=False,
 879           depth=32,
 880           num_layers_before_predictor=2)
 881       box_predictions = conv_box_predictor([image_features])
 882       for key, value in box_predictions.items():
 883         box_predictions[key] = tf.concat(value, axis=1)
 884       assert len(box_predictions) == 3
 885       return (box_predictions[box_predictor.BOX_ENCODINGS],
 886               box_predictions[box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND],
 887               box_predictions[other_head_name])
 888 
 889     batch_size = 4
 890     feature_ht = 8
 891     feature_wt = 8
 892     image_features = np.random.rand(batch_size, feature_ht, feature_wt,
 893                                     64).astype(np.float32)
 894     (box_encodings, class_predictions, other_head_predictions) = self.execute(
 895         graph_fn, [image_features])
 896     num_anchors = feature_ht * feature_wt * num_predictions_per_location
 897     self.assertAllEqual(box_encodings.shape,
 898                         [batch_size, num_anchors, box_code_size])
 899     self.assertAllEqual(
 900         class_predictions.shape,
 901         [batch_size, num_anchors, num_classes_without_background + 1])
 902     self.assertAllEqual(other_head_predictions.shape, [
 903         batch_size, num_anchors, num_classes_without_background, mask_height,
 904         mask_width
 905     ])
