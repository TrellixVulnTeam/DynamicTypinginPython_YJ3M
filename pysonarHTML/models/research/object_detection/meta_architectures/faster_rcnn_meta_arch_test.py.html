 226   def test_predict_correct_shapes_in_inference_mode_three_stages_with_masks(
 227       self, masks_are_class_agnostic):
 228     batch_size = 2
 229     image_size = 10
 230     with test_utils.GraphContextOrNone() as g:
 231       model = self._build_model(
 232           is_training=False,
 233           number_of_stages=3,
 234           second_stage_batch_size=2,
 235           predict_masks=True,
 236           masks_are_class_agnostic=masks_are_class_agnostic)
 237     def graph_fn():
 238       shape = [tf.random_uniform([], minval=batch_size, maxval=batch_size + 1,
 239                                  dtype=tf.int32),
 240                tf.random_uniform([], minval=image_size, maxval=image_size + 1,
 241                                  dtype=tf.int32),
 242                tf.random_uniform([], minval=image_size, maxval=image_size + 1,
 243                                  dtype=tf.int32),
 244                3]
 245       image = tf.zeros(shape)
 246       _, true_image_shapes = model.preprocess(image)
 247       detections = model.predict(image, true_image_shapes)
 248       return (detections['detection_boxes'], detections['detection_classes'],
 249               detections['detection_scores'], detections['num_detections'],
 250               detections['detection_masks'], detections['mask_predictions'])
 251     (detection_boxes, detection_scores, detection_classes,
 252      num_detections, detection_masks,
 253      mask_predictions) = self.execute_cpu(graph_fn, [], graph=g)
 254     self.assertAllEqual(detection_boxes.shape, [2, 5, 4])
 255     self.assertAllEqual(detection_masks.shape,
 256                         [2, 5, 14, 14])
 257     self.assertAllEqual(detection_classes.shape, [2, 5])
 258     self.assertAllEqual(detection_scores.shape, [2, 5])
 259     self.assertAllEqual(num_detections.shape, [2])
 260     num_classes = 1 if masks_are_class_agnostic else 2
 261     self.assertAllEqual(mask_predictions.shape,
 262                         [10, num_classes, 14, 14])

 264   def test_raw_detection_boxes_and_anchor_indices_correct(self):
 265     batch_size = 2
 266     image_size = 10
 267 
 268     with test_utils.GraphContextOrNone() as g:
 269       model = self._build_model(
 270           is_training=False,
 271           number_of_stages=2,
 272           second_stage_batch_size=2,
 273           share_box_across_classes=True,
 274           return_raw_detections_during_predict=True)
 275     def graph_fn():
 276       shape = [tf.random_uniform([], minval=batch_size, maxval=batch_size + 1,
 277                                  dtype=tf.int32),
 278                tf.random_uniform([], minval=image_size, maxval=image_size + 1,
 279                                  dtype=tf.int32),
 280                tf.random_uniform([], minval=image_size, maxval=image_size + 1,
 281                                  dtype=tf.int32),
 282                3]
 283       image = tf.zeros(shape)
 284       _, true_image_shapes = model.preprocess(image)
 285       predict_tensor_dict = model.predict(image, true_image_shapes)
 286       detections = model.postprocess(predict_tensor_dict, true_image_shapes)
 287       return (detections['detection_boxes'],
 288               detections['num_detections'],
 289               detections['detection_anchor_indices'],
 290               detections['raw_detection_boxes'],
 291               predict_tensor_dict['raw_detection_boxes'])
 292     (detection_boxes, num_detections, detection_anchor_indices,
 293      raw_detection_boxes,
 294      predict_raw_detection_boxes) = self.execute_cpu(graph_fn, [], graph=g)
 295 
 296     # Verify that the raw detections from predict and postprocess are the
 297     # same.
 298     self.assertAllClose(
 299         np.squeeze(predict_raw_detection_boxes), raw_detection_boxes)
 300     # Verify that the raw detection boxes at detection anchor indices are the
 301     # same as the postprocessed detections.
 302     for i in range(batch_size):
 303       num_detections_per_image = int(num_detections[i])
 304       detection_boxes_per_image = detection_boxes[i][
 305           :num_detections_per_image]
 306       detection_anchor_indices_per_image = detection_anchor_indices[i][
 307           :num_detections_per_image]
 308       raw_detections_per_image = np.squeeze(raw_detection_boxes[i])
 309       raw_detections_at_anchor_indices = raw_detections_per_image[
 310           detection_anchor_indices_per_image]
 311       self.assertAllClose(detection_boxes_per_image,
 312                           raw_detections_at_anchor_indices)

 318   def test_predict_gives_correct_shapes_in_train_mode_both_stages_with_masks(
 319       self, masks_are_class_agnostic):
 320     with test_utils.GraphContextOrNone() as g:
 321       model = self._build_model(
 322           is_training=True,
 323           number_of_stages=3,
 324           second_stage_batch_size=7,
 325           predict_masks=True,
 326           masks_are_class_agnostic=masks_are_class_agnostic)
 327     batch_size = 2
 328     image_size = 10
 329     max_num_proposals = 7
 330     def graph_fn():
 331       image_shape = (batch_size, image_size, image_size, 3)
 332       preprocessed_inputs = tf.zeros(image_shape, dtype=tf.float32)
 333       groundtruth_boxes_list = [
 334           tf.constant([[0, 0, .5, .5], [.5, .5, 1, 1]], dtype=tf.float32),
 335           tf.constant([[0, .5, .5, 1], [.5, 0, 1, .5]], dtype=tf.float32)
 336       ]
 337       groundtruth_classes_list = [
 338           tf.constant([[1, 0], [0, 1]], dtype=tf.float32),
 339           tf.constant([[1, 0], [1, 0]], dtype=tf.float32)
 340       ]
 341       groundtruth_weights_list = [
 342           tf.constant([1, 1], dtype=tf.float32),
 343           tf.constant([1, 1], dtype=tf.float32)]
 344       _, true_image_shapes = model.preprocess(tf.zeros(image_shape))
 345       model.provide_groundtruth(
 346           groundtruth_boxes_list,
 347           groundtruth_classes_list,
 348           groundtruth_weights_list=groundtruth_weights_list)
 349 
 350       result_tensor_dict = model.predict(preprocessed_inputs, true_image_shapes)
 351       return result_tensor_dict['mask_predictions']
 352     mask_shape_1 = 1 if masks_are_class_agnostic else model._num_classes
 353     mask_out = self.execute_cpu(graph_fn, [], graph=g)
 354     self.assertAllEqual(mask_out.shape,
 355                         (2 * max_num_proposals, mask_shape_1, 14, 14))

 357   def test_postprocess_third_stage_only_inference_mode(self):
 358     batch_size = 2
 359     initial_crop_size = 3
 360     maxpool_stride = 1
 361     height = initial_crop_size // maxpool_stride
 362     width = initial_crop_size // maxpool_stride
 363     depth = 3
 364 
 365     with test_utils.GraphContextOrNone() as g:
 366       model = self._build_model(
 367           is_training=False, number_of_stages=3,
 368           second_stage_batch_size=6, predict_masks=True)
 369     total_num_padded_proposals = batch_size * model.max_num_proposals
 370     def graph_fn(images_shape, num_proposals, proposal_boxes,
 371                  refined_box_encodings, class_predictions_with_background):
 372       _, true_image_shapes = model.preprocess(
 373           tf.zeros(images_shape))
 374       detections = model.postprocess({
 375           'refined_box_encodings': refined_box_encodings,
 376           'class_predictions_with_background':
 377           class_predictions_with_background,
 378           'num_proposals': num_proposals,
 379           'proposal_boxes': proposal_boxes,
 380           'image_shape': images_shape,
 381           'detection_boxes': tf.zeros([2, 5, 4]),
 382           'detection_masks': tf.zeros([2, 5, 14, 14]),
 383           'detection_scores': tf.zeros([2, 5]),
 384           'detection_classes': tf.zeros([2, 5]),
 385           'num_detections': tf.zeros([2]),
 386           'detection_features': tf.zeros([2, 5, width, height, depth])
 387       }, true_image_shapes)
 388       return (detections['detection_boxes'], detections['detection_masks'],
 389               detections['detection_scores'], detections['detection_classes'],
 390               detections['num_detections'],
 391               detections['detection_features'])
 392     images_shape = np.array((2, 36, 48, 3), dtype=np.int32)
 393     proposal_boxes = np.array(
 394         [[[1, 1, 2, 3],
 395           [0, 0, 1, 1],
 396           [.5, .5, .6, .6],
 397           4*[0], 4*[0], 4*[0], 4*[0], 4*[0]],
 398          [[2, 3, 6, 8],
 399           [1, 2, 5, 3],
 400           4*[0], 4*[0], 4*[0], 4*[0], 4*[0], 4*[0]]])
 401     num_proposals = np.array([3, 2], dtype=np.int32)
 402     refined_box_encodings = np.zeros(
 403         [total_num_padded_proposals, model.num_classes, 4])
 404     class_predictions_with_background = np.ones(
 405         [total_num_padded_proposals, model.num_classes+1])
 406 
 407     (detection_boxes, detection_masks, detection_scores, detection_classes,
 408      num_detections,
 409      detection_features) = self.execute_cpu(graph_fn,
 410                                             [images_shape, num_proposals,
 411                                              proposal_boxes,
 412                                              refined_box_encodings,
 413                                              class_predictions_with_background],
 414                                             graph=g)
 415     self.assertAllEqual(detection_boxes.shape, [2, 5, 4])
 416     self.assertAllEqual(detection_masks.shape, [2, 5, 14, 14])
 417     self.assertAllClose(detection_scores.shape, [2, 5])
 418     self.assertAllClose(detection_classes.shape, [2, 5])
 419     self.assertAllClose(num_detections.shape, [2])
 420     self.assertTrue(np.amax(detection_masks <= 1.0))
 421     self.assertTrue(np.amin(detection_masks >= 0.0))
 422     self.assertAllEqual(detection_features.shape,
 423                         [2, 5, width, height, depth])
 424     self.assertGreaterEqual(np.amax(detection_features), 0)
