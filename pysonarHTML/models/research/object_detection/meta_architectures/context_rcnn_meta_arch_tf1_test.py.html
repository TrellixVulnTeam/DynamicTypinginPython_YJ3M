  88 
  89     class ProposalFeatureExtractor(tf.keras.Model):
  90       """Dummy proposal feature extraction."""
  91 
  92       def __init__(self, name):
  93         super(ProposalFeatureExtractor, self).__init__(name=name)
  94         self.conv = None

  97         self.conv = tf.keras.layers.Conv2D(
  98             3, kernel_size=1, padding='SAME', name='layer1')

 482   @parameterized.named_parameters(
 483       {'testcase_name': 'static_shapes', 'static_shapes': True},

 485       )
 486   def test_prediction_end_to_end(self, static_shapes):
 487     """Runs prediction end to end and test the shape of the results."""
 488     with test_utils.GraphContextOrNone() as g:
 489       model = self._build_model(
 490           is_training=False,
 491           number_of_stages=2,
 492           second_stage_batch_size=6,
 493           use_matmul_crop_and_resize=static_shapes,
 494           clip_anchors_to_image=static_shapes,
 495           use_matmul_gather_in_matcher=static_shapes,
 496           use_static_shapes=static_shapes,
 497           num_classes=42)
 498 
 499     def graph_fn():
 500       inputs_shape = (2, 20, 20, 3)
 501       inputs = tf.cast(
 502           tf.random_uniform(inputs_shape, minval=0, maxval=255, dtype=tf.int32),
 503           dtype=tf.float32)
 504       preprocessed_inputs, true_image_shapes = model.preprocess(inputs)
 505       context_features = tf.random_uniform((2, 20, 10),
 506                                            minval=0,
 507                                            maxval=255,
 508                                            dtype=tf.float32)
 509       valid_context_size = tf.random_uniform((2,),
 510                                              minval=0,
 511                                              maxval=10,
 512                                              dtype=tf.int32)
 513       features = {
 514           fields.InputDataFields.context_features: context_features,
 515           fields.InputDataFields.valid_context_size: valid_context_size
 516       }
 517 
 518       side_inputs = model.get_side_inputs(features)
 519 
 520       prediction_dict = model.predict(preprocessed_inputs, true_image_shapes,
 521                                       **side_inputs)
 522       return (prediction_dict['rpn_box_predictor_features'],
 523               prediction_dict['rpn_box_encodings'],
 524               prediction_dict['refined_box_encodings'],
 525               prediction_dict['proposal_boxes_normalized'],
 526               prediction_dict['proposal_boxes'])
 527     execute_fn = self.execute if static_shapes else self.execute_cpu
 528     (rpn_box_predictor_features, rpn_box_encodings, refined_box_encodings,
 529      proposal_boxes_normalized, proposal_boxes) = execute_fn(graph_fn, [],
 530                                                              graph=g)
 531     self.assertAllEqual(rpn_box_predictor_features.shape, [2, 20, 20, 512])
 532     self.assertAllEqual(rpn_box_encodings.shape, [2, 3600, 4])
 533     self.assertAllEqual(refined_box_encodings.shape, [16, 42, 4])
 534     self.assertAllEqual(proposal_boxes_normalized.shape, [2, 8, 4])
 535     self.assertAllEqual(proposal_boxes.shape, [2, 8, 4])
