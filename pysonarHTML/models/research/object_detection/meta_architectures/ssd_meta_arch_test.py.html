 369       model, num_classes, num_anchors, _ = self._create_model(
 370           apply_hard_mining=False)
 371     def graph_fn(preprocessed_tensor, groundtruth_boxes1, groundtruth_boxes2,
 372                  groundtruth_classes1, groundtruth_classes2):
 373       groundtruth_boxes_list = [groundtruth_boxes1, groundtruth_boxes2]
 374       groundtruth_classes_list = [groundtruth_classes1, groundtruth_classes2]

 395         graph=g)
 396
