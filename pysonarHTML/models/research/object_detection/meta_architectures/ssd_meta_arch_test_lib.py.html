  48 class FakeSSDFeatureExtractor(ssd_meta_arch.SSDFeatureExtractor):
  49   """Fake ssd feature extracture for ssd meta arch tests."""
  50 
  51   def __init__(self):
  52     super(FakeSSDFeatureExtractor, self).__init__(
  53         is_training=True,
  54         depth_multiplier=0,
  55         min_depth=0,
  56         pad_to_multiple=1,
  57         conv_hyperparams_fn=None)
  58 
  59   def preprocess(self, resized_inputs):
  60     return tf.identity(resized_inputs)
  61 
  62   def extract_features(self, preprocessed_inputs):
  63     with tf.variable_scope('mock_model'):
  64       features = slim.conv2d(
  65           inputs=preprocessed_inputs,
  66           num_outputs=32,
  67           kernel_size=1,
  68           scope='layer1')
  69       return [features]

  72 class FakeSSDKerasFeatureExtractor(ssd_meta_arch.SSDKerasFeatureExtractor):
  73   """Fake keras based ssd feature extracture for ssd meta arch tests."""
  74 
  75   def __init__(self):
  76     with tf.name_scope('mock_model'):
  77       super(FakeSSDKerasFeatureExtractor, self).__init__(
  78           is_training=True,
  79           depth_multiplier=0,
  80           min_depth=0,
  81           pad_to_multiple=1,
  82           conv_hyperparams=None,
  83           freeze_batchnorm=False,
  84           inplace_batchnorm_update=False,
  85       )
  86 
  87       self._conv = keras.Conv2D(filters=32, kernel_size=1, name='layer1')
  88 
  89   def preprocess(self, resized_inputs):
  90     return tf.identity(resized_inputs)
  91 
  92   def _extract_features(self, preprocessed_inputs, **kwargs):
  93     with tf.name_scope('mock_model'):
  94       return [self._conv(preprocessed_inputs)]

  97 class MockAnchorGenerator2x2(anchor_generator.AnchorGenerator):
  98   """A simple 2x2 anchor grid on the unit square used for test only."""
  99 
 100   def name_scope(self):
 101     return 'MockAnchorGenerator'
 102 
 103   def num_anchors_per_location(self):
 104     return [1]
 105 
 106   def _generate(self, feature_map_shape_list, im_height, im_width):
 107     return [
 108         box_list.BoxList(
 109             tf.constant(
 110                 [
 111                     [0, 0, .5, .5],
 112                     [0, .5, .5, 1],
 113                     [.5, 0, 1, .5],
 114                     [1., 1., 1.5, 1.5]  # Anchor that is outside clip_window.
 115                 ],
 116                 tf.float32))
 117     ]
 118 
 119   def num_anchors(self):
 120     return 4

 123 class SSDMetaArchTestBase(test_case.TestCase):
 124   """Base class to test SSD based meta architectures."""
 125 
 126   def _create_model(
 127       self,
 128       model_fn=ssd_meta_arch.SSDMetaArch,
 129       apply_hard_mining=True,
 130       normalize_loc_loss_by_codesize=False,
 131       add_background_class=True,
 132       random_example_sampling=False,
 133       expected_loss_weights=model_pb2.DetectionModel().ssd.loss.NONE,
 134       min_num_negative_samples=1,
 135       desired_negative_sampling_ratio=3,
 136       predict_mask=False,
 137       use_static_shapes=False,
 138       nms_max_size_per_class=5,
 139       calibration_mapping_value=None,
 140       return_raw_detections_during_predict=False):
 141     is_training = False
 142     num_classes = 1
 143     mock_anchor_generator = MockAnchorGenerator2x2()
 144     use_keras = tf_version.is_tf2()
 145     if use_keras:
 146       mock_box_predictor = test_utils.MockKerasBoxPredictor(
 147           is_training, num_classes, add_background_class=add_background_class)
 148     else:
 149       mock_box_predictor = test_utils.MockBoxPredictor(
 150           is_training, num_classes, add_background_class=add_background_class)
 151     mock_box_coder = test_utils.MockBoxCoder()
 152     if use_keras:
 153       fake_feature_extractor = FakeSSDKerasFeatureExtractor()
 154     else:
 155       fake_feature_extractor = FakeSSDFeatureExtractor()
 156     mock_matcher = test_utils.MockMatcher()
 157     region_similarity_calculator = sim_calc.IouSimilarity()
 158     encode_background_as_zeros = False
 159 
 160     def image_resizer_fn(image):
 161       return [tf.identity(image), tf.shape(image)]
 162 
 163     classification_loss = losses.WeightedSigmoidClassificationLoss()
 164     localization_loss = losses.WeightedSmoothL1LocalizationLoss()
 165     non_max_suppression_fn = functools.partial(
 166         post_processing.batch_multiclass_non_max_suppression,
 167         score_thresh=-20.0,
 168         iou_thresh=1.0,
 169         max_size_per_class=nms_max_size_per_class,
 170         max_total_size=nms_max_size_per_class,
 171         use_static_shapes=use_static_shapes)
 172     score_conversion_fn = tf.identity
 173     calibration_config = calibration_pb2.CalibrationConfig()
 174     if calibration_mapping_value:
 175       calibration_text_proto = """
 176       function_approximation {
 177         x_y_pairs {
 178             x_y_pair {
 179               x: 0.0
 180               y: %f
 181             }
 182             x_y_pair {
 183               x: 1.0
 184               y: %f
 185             }}}""" % (calibration_mapping_value, calibration_mapping_value)
 186       text_format.Merge(calibration_text_proto, calibration_config)
 187       score_conversion_fn = (
 188           post_processing_builder._build_calibrated_score_converter(  # pylint: disable=protected-access
 189               tf.identity, calibration_config))
 190     classification_loss_weight = 1.0
 191     localization_loss_weight = 1.0
 192     negative_class_weight = 1.0
 193     normalize_loss_by_num_matches = False
 194 
 195     hard_example_miner = None
 196     if apply_hard_mining:
 197       # This hard example miner is expected to be a no-op.
 198       hard_example_miner = losses.HardExampleMiner(
 199           num_hard_examples=None, iou_threshold=1.0)
 200 
 201     random_example_sampler = None
 202     if random_example_sampling:
 203       random_example_sampler = sampler.BalancedPositiveNegativeSampler(
 204           positive_fraction=0.5)
 205 
 206     target_assigner_instance = target_assigner.TargetAssigner(
 207         region_similarity_calculator,
 208         mock_matcher,
 209         mock_box_coder,
 210         negative_class_weight=negative_class_weight)
 211 
 212     model_config = model_pb2.DetectionModel()
 213     if expected_loss_weights == model_config.ssd.loss.NONE:
 214       expected_loss_weights_fn = None
 215     else:
 216       raise ValueError('Not a valid value for expected_loss_weights.')
 217 
 218     code_size = 4
 219 
 220     kwargs = {}
 221     if predict_mask:
 222       kwargs.update({
 223           'mask_prediction_fn': test_utils.MockMaskHead(num_classes=1).predict,
 224       })
 225 
 226     model = model_fn(
 227         is_training=is_training,
 228         anchor_generator=mock_anchor_generator,
 229         box_predictor=mock_box_predictor,
 230         box_coder=mock_box_coder,
 231         feature_extractor=fake_feature_extractor,
 232         encode_background_as_zeros=encode_background_as_zeros,
 233         image_resizer_fn=image_resizer_fn,
 234         non_max_suppression_fn=non_max_suppression_fn,
 235         score_conversion_fn=score_conversion_fn,
 236         classification_loss=classification_loss,
 237         localization_loss=localization_loss,
 238         classification_loss_weight=classification_loss_weight,
 239         localization_loss_weight=localization_loss_weight,
 240         normalize_loss_by_num_matches=normalize_loss_by_num_matches,
 241         hard_example_miner=hard_example_miner,
 242         target_assigner_instance=target_assigner_instance,
 243         add_summaries=False,
 244         normalize_loc_loss_by_codesize=normalize_loc_loss_by_codesize,
 245         freeze_batchnorm=False,
 246         inplace_batchnorm_update=False,
 247         add_background_class=add_background_class,
 248         random_example_sampler=random_example_sampler,
 249         expected_loss_weights_fn=expected_loss_weights_fn,
 250         return_raw_detections_during_predict=(
 251             return_raw_detections_during_predict),
 252         **kwargs)
 253     return model, num_classes, mock_anchor_generator.num_anchors(), code_size
 254 
 255   def _get_value_for_matching_key(self, dictionary, suffix):
 256     for key in dictionary.keys():
 257       if key.endswith(suffix):
 258         return dictionary[key]
 259     raise ValueError('key not found {}'.format(suffix))
