  81     self._pad_to_multiple = pad_to_multiple

  83     self._reuse_weights = reuse_weights

  85     self._use_depthwise = use_depthwise

 129   def restore_from_classification_checkpoint_fn(self, feature_extractor_scope):
 130     """Returns a map of variables to load from a foreign checkpoint.
 131 
 132     Args:
 133       feature_extractor_scope: A scope name for the feature extractor.
 134 
 135     Returns:
 136       A dict mapping variable names (to load from a checkpoint) to variables in
 137       the model graph.
 138     """
 139     variables_to_restore = {}
 140     for variable in variables_helper.get_global_variables_safely():
 141       var_name = variable.op.name

 143         var_name = var_name.replace(feature_extractor_scope + '/', '')
 144         variables_to_restore[var_name] = variable
 145 
 146     return variables_to_restore

 253   def restore_from_classification_checkpoint_fn(self, feature_extractor_scope):
 254     """Returns a map of variables to load from a foreign checkpoint.
 255 
 256     Args:
 257       feature_extractor_scope: A scope name for the feature extractor.
 258 
 259     Returns:
 260       A dict mapping variable names (to load from a checkpoint) to variables in
 261       the model graph.
 262     """
 263     variables_to_restore = {}
 264     if tf.executing_eagerly():
 265       for variable in self.variables:
 266         # variable.name includes ":0" at the end, but the names in the
 267         # checkpoint do not have the suffix ":0". So, we strip it here.
 268         var_name = variable.name[:-2]
 269         if var_name.startswith(feature_extractor_scope + '/'):
 270           var_name = var_name.replace(feature_extractor_scope + '/', '')
 271         variables_to_restore[var_name] = variable
 272     else:
 273       # b/137854499: use global_variables.
 274       for variable in variables_helper.get_global_variables_safely():
 275         var_name = variable.op.name
 276         if var_name.startswith(feature_extractor_scope + '/'):
 277           var_name = var_name.replace(feature_extractor_scope + '/', '')
 278           variables_to_restore[var_name] = variable
 279 
 280     return variables_to_restore

 291                anchor_generator,
 292                box_predictor,
 293                box_coder,
 294                feature_extractor,
 295                encode_background_as_zeros,

 297                non_max_suppression_fn,

 302                localization_loss_weight,

 306                add_summaries=True,
 307                normalize_loc_loss_by_codesize=False,
 308                freeze_batchnorm=False,

 311                explicit_background_class=False,
 312                random_example_sampler=None,
 313                expected_loss_weights_fn=None,
 314                use_confidences_as_targets=False,
 315                implicit_example_weight=0.5,
 316                equalization_loss_config=None,
 317                return_raw_detections_during_predict=False,
 318                nms_on_host=True):
 319     """SSDMetaArch Constructor.
 320 
 321     TODO(rathodv,jonathanhuang): group NMS parameters + score converter into
 322     a class and loss parameters into a class and write config protos for
 323     postprocessing and losses.
 324 
 325     Args:
 326       is_training: A boolean indicating whether the training version of the
 327         computation graph should be constructed.
 328       anchor_generator: an anchor_generator.AnchorGenerator object.
 329       box_predictor: a box_predictor.BoxPredictor object.
 330       box_coder: a box_coder.BoxCoder object.
 331       feature_extractor: a SSDFeatureExtractor object.
 332       encode_background_as_zeros: boolean determining whether background
 333         targets are to be encoded as an all zeros vector or a one-hot
 334         vector (where background is the 0th class).
 335       image_resizer_fn: a callable for image resizing.  This callable always
 336         takes a rank-3 image tensor (corresponding to a single image) and
 337         returns a rank-3 image tensor, possibly with new spatial dimensions and
 338         a 1-D tensor of shape [3] indicating shape of true image within
 339         the resized image tensor as the resized image tensor could be padded.
 340         See builders/image_resizer_builder.py.
 341       non_max_suppression_fn: batch_multiclass_non_max_suppression
 342         callable that takes `boxes`, `scores` and optional `clip_window`
 343         inputs (with all other inputs already set) and returns a dictionary
 344         hold tensors with keys: `detection_boxes`, `detection_scores`,
 345         `detection_classes` and `num_detections`. See `post_processing.
 346         batch_multiclass_non_max_suppression` for the type and shape of these
 347         tensors.
 348       score_conversion_fn: callable elementwise nonlinearity (that takes tensors
 349         as inputs and returns tensors).  This is usually used to convert logits
 350         to probabilities.
 351       classification_loss: an object_detection.core.losses.Loss object.
 352       localization_loss: a object_detection.core.losses.Loss object.
 353       classification_loss_weight: float
 354       localization_loss_weight: float
 355       normalize_loss_by_num_matches: boolean
 356       hard_example_miner: a losses.HardExampleMiner object (can be None)
 357       target_assigner_instance: target_assigner.TargetAssigner instance to use.
 358       add_summaries: boolean (default: True) controlling whether summary ops
 359         should be added to tensorflow graph.
 360       normalize_loc_loss_by_codesize: whether to normalize localization loss
 361         by code size of the box encoder.
 362       freeze_batchnorm: Whether to freeze batch norm parameters during
 363         training or not. When training with a small batch size (e.g. 1), it is
 364         desirable to freeze batch norm update and use pretrained batch norm
 365         params.
 366       inplace_batchnorm_update: Whether to update batch norm moving average
 367         values inplace. When this is false train op must add a control
 368         dependency on tf.graphkeys.UPDATE_OPS collection in order to update
 369         batch norm statistics.
 370       add_background_class: Whether to add an implicit background class to
 371         one-hot encodings of groundtruth labels. Set to false if training a
 372         single class model or using groundtruth labels with an explicit
 373         background class.
 374       explicit_background_class: Set to true if using groundtruth labels with an

 488   def preprocess(self, inputs):
 489     """Feature-extractor specific preprocessing.
 490 
 491     SSD meta architecture uses a default clip_window of [0, 0, 1, 1] during
 492     post-processing. On calling `preprocess` method, clip_window gets updated
 493     based on `true_image_shapes` returned by `image_resizer_fn`.
 494 
 495     Args:
 496       inputs: a [batch, height_in, width_in, channels] float tensor representing
 497         a batch of images with values between 0 and 255.0.
 498 
 499     Returns:
 500       preprocessed_inputs: a [batch, height_out, width_out, channels] float
 501         tensor representing a batch of images.
 502       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 503         of the form [height, width, channels] indicating the shapes
 504         of true images in the resized images, as resized images can be padded
 505         with zeros.
 506 
 507     Raises:
 508       ValueError: if inputs tensor does not have type tf.float32
 509     """
 510     with tf.name_scope('Preprocessor'):
 511       (resized_inputs,
 512        true_image_shapes) = shape_utils.resize_images_and_return_shapes(
 513            inputs, self._image_resizer_fn)
 514 
 515       return (self._feature_extractor.preprocess(resized_inputs),
 516               true_image_shapes)
 517 
 518   def _compute_clip_window(self, preprocessed_images, true_image_shapes):
 519     """Computes clip window to use during post_processing.
 520 
 521     Computes a new clip window to use during post-processing based on
 522     `resized_image_shapes` and `true_image_shapes` only if `preprocess` method
 523     has been called. Otherwise returns a default clip window of [0, 0, 1, 1].
 524 
 525     Args:
 526       preprocessed_images: the [batch, height, width, channels] image
 527           tensor.
 528       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 529         of the form [height, width, channels] indicating the shapes
 530         of true images in the resized images, as resized images can be padded
 531         with zeros. Or None if the clip window should cover the full image.
 532 
 533     Returns:
 534       a 2-D float32 tensor of the form [batch_size, 4] containing the clip
 535       window for each image in the batch in normalized coordinates (relative to
 536       the resized dimensions) where each clip window is of the form [ymin, xmin,
 537       ymax, xmax] or a default clip window of [0, 0, 1, 1].
 538 
 539     """
 540     if true_image_shapes is None:
 541       return tf.constant([0, 0, 1, 1], dtype=tf.float32)
 542 
 543     resized_inputs_shape = shape_utils.combined_static_and_dynamic_shape(
 544         preprocessed_images)
 545     true_heights, true_widths, _ = tf.unstack(
 546         tf.cast(true_image_shapes, dtype=tf.float32), axis=1)
 547     padded_height = tf.cast(resized_inputs_shape[1], dtype=tf.float32)
 548     padded_width = tf.cast(resized_inputs_shape[2], dtype=tf.float32)
 549     return tf.stack(
 550         [
 551             tf.zeros_like(true_heights),
 552             tf.zeros_like(true_widths), true_heights / padded_height,
 553             true_widths / padded_width
 554         ],
 555         axis=1)

 557   def predict(self, preprocessed_inputs, true_image_shapes):
 558     """Predicts unpostprocessed tensors from input tensor.
 559 
 560     This function takes an input batch of images and runs it through the forward
 561     pass of the network to yield unpostprocessesed predictions.
 562 
 563     A side effect of calling the predict method is that self._anchors is
 564     populated with a box_list.BoxList of anchors.  These anchors must be
 565     constructed before the postprocess or loss functions can be called.
 566 
 567     Args:
 568       preprocessed_inputs: a [batch, height, width, channels] image tensor.
 569       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 570         of the form [height, width, channels] indicating the shapes
 571         of true images in the resized images, as resized images can be padded
 572         with zeros.
 573 
 574     Returns:
 575       prediction_dict: a dictionary holding "raw" prediction tensors:
 576         1) preprocessed_inputs: the [batch, height, width, channels] image
 577           tensor.
 578         2) box_encodings: 4-D float tensor of shape [batch_size, num_anchors,
 579           box_code_dimension] containing predicted boxes.
 580         3) class_predictions_with_background: 3-D float tensor of shape
 581           [batch_size, num_anchors, num_classes+1] containing class predictions
 582           (logits) for each of the anchors.  Note that this tensor *includes*
 583           background class predictions (at class index 0).
 584         4) feature_maps: a list of tensors where the ith tensor has shape
 585           [batch, height_i, width_i, depth_i].
 586         5) anchors: 2-D float tensor of shape [num_anchors, 4] containing
 587           the generated anchors in normalized coordinates.
 588         6) final_anchors: 3-D float tensor of shape [batch_size, num_anchors, 4]
 589           containing the generated anchors in normalized coordinates.
 590         If self._return_raw_detections_during_predict is True, the dictionary
 591         will also contain:
 592         7) raw_detection_boxes: a 4-D float32 tensor with shape
 593           [batch_size, self.max_num_proposals, 4] in normalized coordinates.
 594         8) raw_detection_feature_map_indices: a 3-D int32 tensor with shape
 595           [batch_size, self.max_num_proposals].
 596     """
 597     if self._inplace_batchnorm_update:
 598       batchnorm_updates_collections = None
 599     else:
 600       batchnorm_updates_collections = tf.GraphKeys.UPDATE_OPS
 601     if self._feature_extractor.is_keras_model:
 602       feature_maps = self._feature_extractor(preprocessed_inputs)
 603     else:
 604       with slim.arg_scope([slim.batch_norm],
 605                           is_training=(self._is_training and
 606                                        not self._freeze_batchnorm),
 607                           updates_collections=batchnorm_updates_collections):
 608         with tf.variable_scope(None, self._extract_features_scope,
 609                                [preprocessed_inputs]):
 610           feature_maps = self._feature_extractor.extract_features(
 611               preprocessed_inputs)
 612 
 613     feature_map_spatial_dims = self._get_feature_map_spatial_dims(
 614         feature_maps)
 615     image_shape = shape_utils.combined_static_and_dynamic_shape(
 616         preprocessed_inputs)
 617     boxlist_list = self._anchor_generator.generate(
 618         feature_map_spatial_dims,
 619         im_height=image_shape[1],
 620         im_width=image_shape[2])
 621     self._anchors = box_list_ops.concatenate(boxlist_list)
 622     if self._box_predictor.is_keras_model:
 623       predictor_results_dict = self._box_predictor(feature_maps)
 624     else:
 625       with slim.arg_scope([slim.batch_norm],
 626                           is_training=(self._is_training and
 627                                        not self._freeze_batchnorm),
 628                           updates_collections=batchnorm_updates_collections):
 629         predictor_results_dict = self._box_predictor.predict(
 630             feature_maps, self._anchor_generator.num_anchors_per_location())
 631     predictions_dict = {
 632         'preprocessed_inputs':
 633             preprocessed_inputs,
 634         'feature_maps':
 635             feature_maps,
 636         'anchors':
 637             self._anchors.get(),
 638         'final_anchors':
 639             tf.tile(
 640                 tf.expand_dims(self._anchors.get(), 0), [image_shape[0], 1, 1])
 641     }
 642     for prediction_key, prediction_list in iter(predictor_results_dict.items()):
 643       prediction = tf.concat(prediction_list, axis=1)
 644       if (prediction_key == 'box_encodings' and prediction.shape.ndims == 4 and
 645           prediction.shape[2] == 1):
 646         prediction = tf.squeeze(prediction, axis=2)
 647       predictions_dict[prediction_key] = prediction
 648     if self._return_raw_detections_during_predict:
 649       predictions_dict.update(self._raw_detections_and_feature_map_inds(
 650           predictions_dict['box_encodings'], boxlist_list))
 651     self._batched_prediction_tensor_names = [x for x in predictions_dict
 652                                              if x != 'anchors']
 653     return predictions_dict

 655   def _raw_detections_and_feature_map_inds(self, box_encodings, boxlist_list):
 656     anchors = self._anchors.get()
 657     raw_detection_boxes, _ = self._batch_decode(box_encodings, anchors)
 658     batch_size, _, _ = shape_utils.combined_static_and_dynamic_shape(
 659         raw_detection_boxes)
 660     feature_map_indices = (
 661         self._anchor_generator.anchor_index_to_feature_map_index(boxlist_list))
 662     feature_map_indices_batched = tf.tile(
 663         tf.expand_dims(feature_map_indices, 0),
 664         multiples=[batch_size, 1])
 665     return {
 666         fields.PredictionFields.raw_detection_boxes: raw_detection_boxes,
 667         fields.PredictionFields.raw_detection_feature_map_indices:
 668             feature_map_indices_batched
 669     }

 671   def _get_feature_map_spatial_dims(self, feature_maps):
 672     """Return list of spatial dimensions for each feature map in a list.
 673 
 674     Args:
 675       feature_maps: a list of tensors where the ith tensor has shape
 676           [batch, height_i, width_i, depth_i].

 678     Returns:

 687   def postprocess(self, prediction_dict, true_image_shapes):
 688     """Converts prediction tensors to final detections.
 689 
 690     This function converts raw predictions tensors to final detection results by
 691     slicing off the background class, decoding box predictions and applying
 692     non max suppression and clipping to the image window.
 693 
 694     See base class for output format conventions.  Note also that by default,
 695     scores are to be interpreted as logits, but if a score_conversion_fn is
 696     used, then scores are remapped (and may thus have a different
 697     interpretation).
 698 
 699     Args:
 700       prediction_dict: a dictionary holding prediction tensors with
 701         1) preprocessed_inputs: a [batch, height, width, channels] image
 702           tensor.
 703         2) box_encodings: 3-D float tensor of shape [batch_size, num_anchors,
 704           box_code_dimension] containing predicted boxes.
 705         3) class_predictions_with_background: 3-D float tensor of shape
 706           [batch_size, num_anchors, num_classes+1] containing class predictions
 707           (logits) for each of the anchors.  Note that this tensor *includes*
 708           background class predictions.
 709         4) mask_predictions: (optional) a 5-D float tensor of shape
 710           [batch_size, num_anchors, q, mask_height, mask_width]. `q` can be
 711           either number of classes or 1 depending on whether a separate mask is
 712           predicted per class.
 713       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 714         of the form [height, width, channels] indicating the shapes
 715         of true images in the resized images, as resized images can be padded
 716         with zeros. Or None, if the clip window should cover the full image.
 717 
 718     Returns:
 719       detections: a dictionary containing the following fields
 720         detection_boxes: [batch, max_detections, 4] tensor with post-processed
 721           detection boxes.
 722         detection_scores: [batch, max_detections] tensor with scalar scores for
 723           post-processed detection boxes.
 724         detection_multiclass_scores: [batch, max_detections,
 725           num_classes_with_background] tensor with class score distribution for
 726           post-processed detection boxes including background class if any.
 727         detection_classes: [batch, max_detections] tensor with classes for
 728           post-processed detection classes.
 729         detection_keypoints: [batch, max_detections, num_keypoints, 2] (if
 730           encoded in the prediction_dict 'box_encodings')
 731         detection_masks: [batch_size, max_detections, mask_height, mask_width]
 732           (optional)
 733         num_detections: [batch]
 734         raw_detection_boxes: [batch, total_detections, 4] tensor with decoded
 735           detection boxes before Non-Max Suppression.
 736         raw_detection_score: [batch, total_detections,
 737           num_classes_with_background] tensor of multi-class scores for raw
 738           detection boxes.
 739     Raises:
 740       ValueError: if prediction_dict does not contain `box_encodings` or
 741         `class_predictions_with_background` fields.
 742     """
 743     if ('box_encodings' not in prediction_dict or
 744         'class_predictions_with_background' not in prediction_dict):
 745       raise ValueError('prediction_dict does not contain expected entries.')
 746     if 'anchors' not in prediction_dict:
 747       prediction_dict['anchors'] = self.anchors.get()
 748     with tf.name_scope('Postprocessor'):
 749       preprocessed_images = prediction_dict['preprocessed_inputs']
 750       box_encodings = prediction_dict['box_encodings']
 751       box_encodings = tf.identity(box_encodings, 'raw_box_encodings')
 752       class_predictions_with_background = (
 753           prediction_dict['class_predictions_with_background'])
 754       detection_boxes, detection_keypoints = self._batch_decode(
 755           box_encodings, prediction_dict['anchors'])
 756       detection_boxes = tf.identity(detection_boxes, 'raw_box_locations')
 757       detection_boxes = tf.expand_dims(detection_boxes, axis=2)
 758 
 759       detection_scores_with_background = self._score_conversion_fn(
 760           class_predictions_with_background)
 761       detection_scores = tf.identity(detection_scores_with_background,
 762                                      'raw_box_scores')
 763       if self._add_background_class or self._explicit_background_class:
 764         detection_scores = tf.slice(detection_scores, [0, 0, 1], [-1, -1, -1])
 765       additional_fields = None
 766 
 767       batch_size = (
 768           shape_utils.combined_static_and_dynamic_shape(preprocessed_images)[0])
 769 
 770       if 'feature_maps' in prediction_dict:
 771         feature_map_list = []
 772         for feature_map in prediction_dict['feature_maps']:
 773           feature_map_list.append(tf.reshape(feature_map, [batch_size, -1]))
 774         box_features = tf.concat(feature_map_list, 1)
 775         box_features = tf.identity(box_features, 'raw_box_features')
 776       additional_fields = {
 777           'multiclass_scores': detection_scores_with_background
 778       }
 779       if self._anchors is not None:
 780         num_boxes = (self._anchors.num_boxes_static() or
 781                      self._anchors.num_boxes())
 782         anchor_indices = tf.range(num_boxes)
 783         batch_anchor_indices = tf.tile(
 784             tf.expand_dims(anchor_indices, 0), [batch_size, 1])
 785         # All additional fields need to be float.
 786         additional_fields.update({
 787             'anchor_indices': tf.cast(batch_anchor_indices, tf.float32),
 788         })
 789       if detection_keypoints is not None:
 790         detection_keypoints = tf.identity(
 791             detection_keypoints, 'raw_keypoint_locations')
 792         additional_fields[fields.BoxListFields.keypoints] = detection_keypoints
 793 
 794       (nmsed_boxes, nmsed_scores, nmsed_classes, nmsed_masks,
 795        nmsed_additional_fields,
 796        num_detections) = self._non_max_suppression_fn(
 797            detection_boxes,
 798            detection_scores,
 799            clip_window=self._compute_clip_window(
 800                preprocessed_images, true_image_shapes),
 801            additional_fields=additional_fields,
 802            masks=prediction_dict.get('mask_predictions'))
 803 
 804       detection_dict = {
 805           fields.DetectionResultFields.detection_boxes:
 806               nmsed_boxes,
 807           fields.DetectionResultFields.detection_scores:
 808               nmsed_scores,
 809           fields.DetectionResultFields.detection_classes:
 810               nmsed_classes,
 811           fields.DetectionResultFields.num_detections:
 812               tf.cast(num_detections, dtype=tf.float32),
 813           fields.DetectionResultFields.raw_detection_boxes:
 814               tf.squeeze(detection_boxes, axis=2),
 815           fields.DetectionResultFields.raw_detection_scores:
 816               detection_scores_with_background
 817       }
 818       if (nmsed_additional_fields is not None and
 819           fields.InputDataFields.multiclass_scores in nmsed_additional_fields):
 820         detection_dict[
 821             fields.DetectionResultFields.detection_multiclass_scores] = (
 822                 nmsed_additional_fields[
 823                     fields.InputDataFields.multiclass_scores])
 824       if (nmsed_additional_fields is not None and
 825           'anchor_indices' in nmsed_additional_fields):
 826         detection_dict.update({
 827             fields.DetectionResultFields.detection_anchor_indices:
 828                 tf.cast(nmsed_additional_fields['anchor_indices'], tf.int32),
 829         })
 830       if (nmsed_additional_fields is not None and
 831           fields.BoxListFields.keypoints in nmsed_additional_fields):
 832         detection_dict[fields.DetectionResultFields.detection_keypoints] = (
 833             nmsed_additional_fields[fields.BoxListFields.keypoints])
 834       if nmsed_masks is not None:
 835         detection_dict[
 836             fields.DetectionResultFields.detection_masks] = nmsed_masks
 837       return detection_dict

 839   def loss(self, prediction_dict, true_image_shapes, scope=None):
 840     """Compute scalar loss tensors with respect to provided groundtruth.
 841 
 842     Calling this function requires that groundtruth tensors have been
 843     provided via the provide_groundtruth function.
 844 
 845     Args:
 846       prediction_dict: a dictionary holding prediction tensors with
 847         1) box_encodings: 3-D float tensor of shape [batch_size, num_anchors,
 848           box_code_dimension] containing predicted boxes.
 849         2) class_predictions_with_background: 3-D float tensor of shape
 850           [batch_size, num_anchors, num_classes+1] containing class predictions
 851           (logits) for each of the anchors. Note that this tensor *includes*
 852           background class predictions.
 853       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 854         of the form [height, width, channels] indicating the shapes
 855         of true images in the resized images, as resized images can be padded
 856         with zeros.
 857       scope: Optional scope name.
 858 
 859     Returns:
 860       a dictionary mapping loss keys (`localization_loss` and
 861         `classification_loss`) to scalar tensors representing corresponding loss
 862         values.
 863     """
 864     with tf.name_scope(scope, 'Loss', prediction_dict.values()):
 865       keypoints = None
 866       if self.groundtruth_has_field(fields.BoxListFields.keypoints):
 867         keypoints = self.groundtruth_lists(fields.BoxListFields.keypoints)
 868       weights = None
 869       if self.groundtruth_has_field(fields.BoxListFields.weights):
 870         weights = self.groundtruth_lists(fields.BoxListFields.weights)
 871       confidences = None
 872       if self.groundtruth_has_field(fields.BoxListFields.confidences):
 873         confidences = self.groundtruth_lists(fields.BoxListFields.confidences)
 874       (batch_cls_targets, batch_cls_weights, batch_reg_targets,
 875        batch_reg_weights, batch_match) = self._assign_targets(
 876            self.groundtruth_lists(fields.BoxListFields.boxes),
 877            self.groundtruth_lists(fields.BoxListFields.classes),
 878            keypoints, weights, confidences)
 879       match_list = [matcher.Match(match) for match in tf.unstack(batch_match)]
 880       if self._add_summaries:
 881         self._summarize_target_assignment(
 882             self.groundtruth_lists(fields.BoxListFields.boxes), match_list)
 883 
 884       if self._random_example_sampler:
 885         batch_cls_per_anchor_weights = tf.reduce_mean(
 886             batch_cls_weights, axis=-1)
 887         batch_sampled_indicator = tf.cast(
 888             shape_utils.static_or_dynamic_map_fn(
 889                 self._minibatch_subsample_fn,
 890                 [batch_cls_targets, batch_cls_per_anchor_weights],
 891                 dtype=tf.bool,
 892                 parallel_iterations=self._parallel_iterations,
 893                 back_prop=True), dtype=tf.float32)
 894         batch_reg_weights = tf.multiply(batch_sampled_indicator,
 895                                         batch_reg_weights)
 896         batch_cls_weights = tf.multiply(
 897             tf.expand_dims(batch_sampled_indicator, -1),
 898             batch_cls_weights)
 899 
 900       losses_mask = None
 901       if self.groundtruth_has_field(fields.InputDataFields.is_annotated):
 902         losses_mask = tf.stack(self.groundtruth_lists(
 903             fields.InputDataFields.is_annotated))
 904 
 905 
 906       location_losses = self._localization_loss(
 907           prediction_dict['box_encodings'],
 908           batch_reg_targets,
 909           ignore_nan_targets=True,
 910           weights=batch_reg_weights,
 911           losses_mask=losses_mask)
 912 
 913       cls_losses = self._classification_loss(
 914           prediction_dict['class_predictions_with_background'],
 915           batch_cls_targets,
 916           weights=batch_cls_weights,
 917           losses_mask=losses_mask)
 918 
 919       if self._expected_loss_weights_fn:
 920         # Need to compute losses for assigned targets against the
 921         # unmatched_class_label as well as their assigned targets.
 922         # simplest thing (but wasteful) is just to calculate all losses
 923         # twice
 924         batch_size, num_anchors, num_classes = batch_cls_targets.get_shape()
 925         unmatched_targets = tf.ones([batch_size, num_anchors, 1
 926                                     ]) * self._unmatched_class_label
 927 
 928         unmatched_cls_losses = self._classification_loss(
 929             prediction_dict['class_predictions_with_background'],
 930             unmatched_targets,
 931             weights=batch_cls_weights,
 932             losses_mask=losses_mask)
 933 
 934         if cls_losses.get_shape().ndims == 3:
 935           batch_size, num_anchors, num_classes = cls_losses.get_shape()
 936           cls_losses = tf.reshape(cls_losses, [batch_size, -1])
 937           unmatched_cls_losses = tf.reshape(unmatched_cls_losses,
 938                                             [batch_size, -1])
 939           batch_cls_targets = tf.reshape(
 940               batch_cls_targets, [batch_size, num_anchors * num_classes, -1])
 941           batch_cls_targets = tf.concat(
 942               [1 - batch_cls_targets, batch_cls_targets], axis=-1)
 943 
 944           location_losses = tf.tile(location_losses, [1, num_classes])
 945 
 946         foreground_weights, background_weights = (
 947             self._expected_loss_weights_fn(batch_cls_targets))
 948 
 949         cls_losses = (
 950             foreground_weights * cls_losses +
 951             background_weights * unmatched_cls_losses)
 952 
 953         location_losses *= foreground_weights
 954 
 955         classification_loss = tf.reduce_sum(cls_losses)
 956         localization_loss = tf.reduce_sum(location_losses)
 957       elif self._hard_example_miner:
 958         cls_losses = ops.reduce_sum_trailing_dimensions(cls_losses, ndims=2)
 959         (localization_loss, classification_loss) = self._apply_hard_mining(
 960             location_losses, cls_losses, prediction_dict, match_list)
 961         if self._add_summaries:
 962           self._hard_example_miner.summarize()
 963       else:
 964         cls_losses = ops.reduce_sum_trailing_dimensions(cls_losses, ndims=2)
 965         localization_loss = tf.reduce_sum(location_losses)
 966         classification_loss = tf.reduce_sum(cls_losses)
 967 
 968       # Optionally normalize by number of positive matches
 969       normalizer = tf.constant(1.0, dtype=tf.float32)
 970       if self._normalize_loss_by_num_matches:
 971         normalizer = tf.maximum(tf.cast(tf.reduce_sum(batch_reg_weights),
 972                                         dtype=tf.float32),
 973                                 1.0)
 974 
 975       localization_loss_normalizer = normalizer
 976       if self._normalize_loc_loss_by_codesize:
 977         localization_loss_normalizer *= self._box_coder.code_size
 978       localization_loss = tf.multiply((self._localization_loss_weight /
 979                                        localization_loss_normalizer),
 980                                       localization_loss,
 981                                       name='localization_loss')
 982       classification_loss = tf.multiply((self._classification_loss_weight /
 983                                          normalizer), classification_loss,
 984                                         name='classification_loss')
 985 
 986       loss_dict = {
 987           'Loss/localization_loss': localization_loss,
 988           'Loss/classification_loss': classification_loss
 989       }
 990 
 991 
 992     return loss_dict

 994   def _minibatch_subsample_fn(self, inputs):
 995     """Randomly samples anchors for one image.
 996 
 997     Args:
 998       inputs: a list of 2 inputs. First one is a tensor of shape [num_anchors,
 999         num_classes] indicating targets assigned to each anchor. Second one
1000         is a tensor of shape [num_anchors] indicating the class weight of each
1001         anchor.
1002 
1003     Returns:
1004       batch_sampled_indicator: bool tensor of shape [num_anchors] indicating
1005         whether the anchor should be selected for loss computation.
1006     """
1007     cls_targets, cls_weights = inputs
1008     if self._add_background_class:
1009       # Set background_class bits to 0 so that the positives_indicator
1010       # computation would not consider background class.
1011       background_class = tf.zeros_like(tf.slice(cls_targets, [0, 0], [-1, 1]))
1012       regular_class = tf.slice(cls_targets, [0, 1], [-1, -1])
1013       cls_targets = tf.concat([background_class, regular_class], 1)
1014     positives_indicator = tf.reduce_sum(cls_targets, axis=1)
1015     return self._random_example_sampler.subsample(
1016         tf.cast(cls_weights, tf.bool),
1017         batch_size=None,
1018         labels=tf.cast(positives_indicator, tf.bool))

1020   def _summarize_anchor_classification_loss(self, class_ids, cls_losses):
1021     positive_indices = tf.where(tf.greater(class_ids, 0))
1022     positive_anchor_cls_loss = tf.squeeze(
1023         tf.gather(cls_losses, positive_indices), axis=1)
1024     visualization_utils.add_cdf_image_summary(positive_anchor_cls_loss,
1025                                               'PositiveAnchorLossCDF')
1026     negative_indices = tf.where(tf.equal(class_ids, 0))
1027     negative_anchor_cls_loss = tf.squeeze(
1028         tf.gather(cls_losses, negative_indices), axis=1)
1029     visualization_utils.add_cdf_image_summary(negative_anchor_cls_loss,
1030                                               'NegativeAnchorLossCDF')

1032   def _assign_targets(self,
1033                       groundtruth_boxes_list,
1034                       groundtruth_classes_list,
1035                       groundtruth_keypoints_list=None,
1036                       groundtruth_weights_list=None,
1037                       groundtruth_confidences_list=None):
1038     """Assign groundtruth targets.
1039 
1040     Adds a background class to each one-hot encoding of groundtruth classes
1041     and uses target assigner to obtain regression and classification targets.
1042 
1043     Args:
1044       groundtruth_boxes_list: a list of 2-D tensors of shape [num_boxes, 4]
1045         containing coordinates of the groundtruth boxes.
1046           Groundtruth boxes are provided in [y_min, x_min, y_max, x_max]
1047           format and assumed to be normalized and clipped
1048           relative to the image window with y_min <= y_max and x_min <= x_max.
1049       groundtruth_classes_list: a list of 2-D one-hot (or k-hot) tensors of
1050         shape [num_boxes, num_classes] containing the class targets with the 0th
1051         index assumed to map to the first non-background class.
1052       groundtruth_keypoints_list: (optional) a list of 3-D tensors of shape
1053         [num_boxes, num_keypoints, 2]
1054       groundtruth_weights_list: A list of 1-D tf.float32 tensors of shape
1055         [num_boxes] containing weights for groundtruth boxes.
1056       groundtruth_confidences_list: A list of 2-D tf.float32 tensors of shape
1057         [num_boxes, num_classes] containing class confidences for
1058         groundtruth boxes.
1059 
1060     Returns:
1061       batch_cls_targets: a tensor with shape [batch_size, num_anchors,
1062         num_classes],
1063       batch_cls_weights: a tensor with shape [batch_size, num_anchors],
1064       batch_reg_targets: a tensor with shape [batch_size, num_anchors,
1065         box_code_dimension]
1066       batch_reg_weights: a tensor with shape [batch_size, num_anchors],
1067       match: an int32 tensor of shape [batch_size, num_anchors], containing
1068         result of anchor groundtruth matching. Each position in the tensor
1069         indicates an anchor and holds the following meaning:
1070         (1) if match[x, i] >= 0, anchor i is matched with groundtruth
1071             match[x, i].
1072         (2) if match[x, i]=-1, anchor i is marked to be background .
1073         (3) if match[x, i]=-2, anchor i is ignored since it is not background
1074             and does not have sufficient overlap to call it a foreground.
1075     """
1076     groundtruth_boxlists = [
1077         box_list.BoxList(boxes) for boxes in groundtruth_boxes_list
1078     ]
1079     train_using_confidences = (self._is_training and
1080                                self._use_confidences_as_targets)
1081     if self._add_background_class:

1179   def _apply_hard_mining(self, location_losses, cls_losses, prediction_dict,
1180                          match_list):
1181     """Applies hard mining to anchorwise losses.
1182 
1183     Args:
1184       location_losses: Float tensor of shape [batch_size, num_anchors]
1185         representing anchorwise location losses.
1186       cls_losses: Float tensor of shape [batch_size, num_anchors]
1187         representing anchorwise classification losses.
1188       prediction_dict: p a dictionary holding prediction tensors with
1189         1) box_encodings: 3-D float tensor of shape [batch_size, num_anchors,
1190           box_code_dimension] containing predicted boxes.
1191         2) class_predictions_with_background: 3-D float tensor of shape
1192           [batch_size, num_anchors, num_classes+1] containing class predictions
1193           (logits) for each of the anchors.  Note that this tensor *includes*
1194           background class predictions.
1195         3) anchors: (optional) 2-D float tensor of shape [num_anchors, 4].
1196       match_list: a list of matcher.Match objects encoding the match between
1197         anchors and groundtruth boxes for each image of the batch,
1198         with rows of the Match objects corresponding to groundtruth boxes
1199         and columns corresponding to anchors.
1200 
1201     Returns:
1202       mined_location_loss: a float scalar with sum of localization losses from
1203         selected hard examples.
1204       mined_cls_loss: a float scalar with sum of classification losses from
1205         selected hard examples.
1206     """
1207     class_predictions = prediction_dict['class_predictions_with_background']
1208     if self._add_background_class:
1209       class_predictions = tf.slice(class_predictions, [0, 0, 1], [-1, -1, -1])
1210 
1211     if 'anchors' not in prediction_dict:
1212       prediction_dict['anchors'] = self.anchors.get()
1213     decoded_boxes, _ = self._batch_decode(prediction_dict['box_encodings'],
1214                                           prediction_dict['anchors'])
1215     decoded_box_tensors_list = tf.unstack(decoded_boxes)
1216     class_prediction_list = tf.unstack(class_predictions)
1217     decoded_boxlist_list = []
1218     for box_location, box_score in zip(decoded_box_tensors_list,
1219                                        class_prediction_list):
1220       decoded_boxlist = box_list.BoxList(box_location)
1221       decoded_boxlist.add_field('scores', box_score)
1222       decoded_boxlist_list.append(decoded_boxlist)
1223     return self._hard_example_miner(
1224         location_losses=location_losses,
1225         cls_losses=cls_losses,
1226         decoded_boxlist_list=decoded_boxlist_list,
1227         match_list=match_list)

1229   def _batch_decode(self, box_encodings, anchors):
1230     """Decodes a batch of box encodings with respect to the anchors.
1231 
1232     Args:
1233       box_encodings: A float32 tensor of shape
1234         [batch_size, num_anchors, box_code_size] containing box encodings.
1235       anchors: A tensor of shape [num_anchors, 4].
1236 
1237     Returns:
1238       decoded_boxes: A float32 tensor of shape
1239         [batch_size, num_anchors, 4] containing the decoded boxes.
1240       decoded_keypoints: A float32 tensor of shape
1241         [batch_size, num_anchors, num_keypoints, 2] containing the decoded
1242         keypoints if present in the input `box_encodings`, None otherwise.
1243     """
1244     combined_shape = shape_utils.combined_static_and_dynamic_shape(
1245         box_encodings)
1246     batch_size = combined_shape[0]
1247     tiled_anchor_boxes = tf.tile(tf.expand_dims(anchors, 0), [batch_size, 1, 1])
1248     tiled_anchors_boxlist = box_list.BoxList(
1249         tf.reshape(tiled_anchor_boxes, [-1, 4]))
1250     decoded_boxes = self._box_coder.decode(
1251         tf.reshape(box_encodings, [-1, self._box_coder.code_size]),
1252         tiled_anchors_boxlist)
1253     decoded_keypoints = None
1254     if decoded_boxes.has_field(fields.BoxListFields.keypoints):
1255       decoded_keypoints = decoded_boxes.get_field(
1256           fields.BoxListFields.keypoints)
1257       num_keypoints = decoded_keypoints.get_shape()[1]
1258       decoded_keypoints = tf.reshape(
1259           decoded_keypoints,
1260           tf.stack([combined_shape[0], combined_shape[1], num_keypoints, 2]))
1261     decoded_boxes = tf.reshape(decoded_boxes.get(), tf.stack(
1262         [combined_shape[0], combined_shape[1], 4]))
1263     return decoded_boxes, decoded_keypoints

1285   def restore_map(self,
1286                   fine_tune_checkpoint_type='detection',
1287                   load_all_detection_checkpoint_vars=False):
1288     """Returns a map of variables to load from a foreign checkpoint.
1289 
1290     See parent class for details.
1291 
1292     Args:
1293       fine_tune_checkpoint_type: whether to restore from a full detection
1294         checkpoint (with compatible variable names) or to restore from a
1295         classification checkpoint for initialization prior to training.
1296         Valid values: `detection`, `classification`. Default 'detection'.
1297       load_all_detection_checkpoint_vars: whether to load all variables (when
1298          `fine_tune_checkpoint_type='detection'`). If False, only variables
1299          within the appropriate scopes are included. Default False.
1300 
1301     Returns:
1302       A dict mapping variable names (to load from a checkpoint) to variables in
1303       the model graph.
1304     Raises:
1305       ValueError: if fine_tune_checkpoint_type is neither `classification`
1306         nor `detection`.
1307     """
1308     if fine_tune_checkpoint_type == 'classification':
1309       return self._feature_extractor.restore_from_classification_checkpoint_fn(
1310           self._extract_features_scope)
1311 
1312     elif fine_tune_checkpoint_type == 'detection':
1313       variables_to_restore = {}
1314       if tf.executing_eagerly():
1315         if load_all_detection_checkpoint_vars:
1316           # Grab all detection vars by name
1317           for variable in self.variables:
1318             # variable.name includes ":0" at the end, but the names in the
1319             # checkpoint do not have the suffix ":0". So, we strip it here.
1320             var_name = variable.name[:-2]
1321             variables_to_restore[var_name] = variable
1322         else:
1323           # Grab just the feature extractor vars by name
1324           for variable in self._feature_extractor.variables:
1325             # variable.name includes ":0" at the end, but the names in the
1326             # checkpoint do not have the suffix ":0". So, we strip it here.
1327             var_name = variable.name[:-2]
1328             variables_to_restore[var_name] = variable
1329       else:
1330         for variable in variables_helper.get_global_variables_safely():
1331           var_name = variable.op.name
1332           if load_all_detection_checkpoint_vars:
1333             variables_to_restore[var_name] = variable
1334           else:
1335             if var_name.startswith(self._extract_features_scope):
1336               variables_to_restore[var_name] = variable
1337 
1338       return variables_to_restore
1339 
1340     else:
1341       raise ValueError('Not supported fine_tune_checkpoint_type: {}'.format(
1342           fine_tune_checkpoint_type))

1344   def updates(self):
1345     """Returns a list of update operators for this model.
1346 
1347     Returns a list of update operators for this model that must be executed at
1348     each training step. The estimator's train op needs to have a control
1349     dependency on these updates.
1350 
1351     Returns:
1352       A list of update operators.
1353     """
1354     update_ops = []
1355     slim_update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS)
1356     # Copy the slim ops to avoid modifying the collection
1357     if slim_update_ops:
1358       update_ops.extend(slim_update_ops)
1359     if self._box_predictor.is_keras_model:
1360       update_ops.extend(self._box_predictor.get_updates_for(None))
1361       update_ops.extend(self._box_predictor.get_updates_for(
1362           self._box_predictor.inputs))
1363     if self._feature_extractor.is_keras_model:
1364       update_ops.extend(self._feature_extractor.get_updates_for(None))
1365       update_ops.extend(self._feature_extractor.get_updates_for(
1366           self._feature_extractor.inputs))
1367     return update_ops
