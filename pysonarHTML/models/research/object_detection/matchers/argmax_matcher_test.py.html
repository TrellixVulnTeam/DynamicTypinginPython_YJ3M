  27   def test_return_correct_matches_with_default_thresholds(self):
  28 
  29     def graph_fn(similarity_matrix):
  30       matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=None)
  31       match = matcher.match(similarity_matrix)
  32       matched_cols = match.matched_column_indicator()
  33       unmatched_cols = match.unmatched_column_indicator()
  34       match_results = match.match_results
  35       return (matched_cols, unmatched_cols, match_results)
  36 
  37     similarity = np.array([[1., 1, 1, 3, 1],
  38                            [2, -1, 2, 0, 4],
  39                            [3, 0, -1, 0, 0]], dtype=np.float32)
  40     expected_matched_rows = np.array([2, 0, 1, 0, 1])
  41     (res_matched_cols, res_unmatched_cols,
  42      res_match_results) = self.execute(graph_fn, [similarity])
  43 
  44     self.assertAllEqual(res_match_results[res_matched_cols],
  45                         expected_matched_rows)
  46     self.assertAllEqual(np.nonzero(res_matched_cols)[0], [0, 1, 2, 3, 4])
  47     self.assertFalse(np.all(res_unmatched_cols))

  49   def test_return_correct_matches_with_empty_rows(self):
  50 
  51     def graph_fn(similarity_matrix):
  52       matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=None)
  53       match = matcher.match(similarity_matrix)
  54       return match.unmatched_column_indicator()
  55     similarity = 0.2 * np.ones([0, 5], dtype=np.float32)
  56     res_unmatched_cols = self.execute(graph_fn, [similarity])
  57     self.assertAllEqual(np.nonzero(res_unmatched_cols)[0], np.arange(5))

 185   def test_return_correct_matches_using_force_match_padded_groundtruth(self):
 186     def graph_fn(similarity, valid_rows):
 187       matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=3.,
 188                                              unmatched_threshold=2.,
 189                                              force_match_for_each_row=True)
 190       match = matcher.match(similarity, valid_rows)
 191       matched_cols = match.matched_column_indicator()
 192       unmatched_cols = match.unmatched_column_indicator()
 193       match_results = match.match_results
 194       return (matched_cols, unmatched_cols, match_results)
 195 
 196     similarity = np.array([[1, 1, 1, 3, 1],
 197                            [-1, 0, -2, -2, -1],
 198                            [0, 0, 0, 0, 0],
 199                            [3, 0, -1, 2, 0],
 200                            [0, 0, 0, 0, 0]], dtype=np.float32)
 201     valid_rows = np.array([True, True, False, True, False])
 202     expected_matched_cols = np.array([0, 1, 3])
 203     expected_matched_rows = np.array([3, 1, 0])
 204     expected_unmatched_cols = np.array([2, 4])  # col 2 has too high max val
 205 
 206     (res_matched_cols, res_unmatched_cols,
 207      match_results) = self.execute(graph_fn, [similarity, valid_rows])
 208     self.assertAllEqual(match_results[res_matched_cols], expected_matched_rows)
 209     self.assertAllEqual(np.nonzero(res_matched_cols)[0], expected_matched_cols)
 210     self.assertAllEqual(np.nonzero(res_unmatched_cols)[0],
 211                         expected_unmatched_cols)
