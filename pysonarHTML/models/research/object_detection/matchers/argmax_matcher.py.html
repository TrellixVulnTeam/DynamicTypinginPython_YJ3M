  16 """Argmax matcher implementation.
  17 
  18 This class takes a similarity matrix and matches columns to rows based on the
  19 maximum value per column. One can specify matched_thresholds and
  20 to prevent columns from matching to rows (generally resulting in a negative
  21 training example) and unmatched_theshold to ignore the match (generally
  22 resulting in neither a positive or negative training example).
  23 
  24 This matcher is used in Fast(er)-RCNN.
  25 
  26 Note: matchers are used in TargetAssigners. There is a create_target_assigner
  27 factory function for popular implementations.
  28 """
  29 import tensorflow.compat.v1 as tf
  30 
  31 from object_detection.core import matcher
  32 from object_detection.utils import shape_utils
  33 
  34 
  35 class ArgMaxMatcher(matcher.Matcher):
  36   """Matcher based on highest value.
  37 
  38   This class computes matches from a similarity matrix. Each column is matched
  39   to a single row.
  40 
  41   To support object detection target assignment this class enables setting both
  42   matched_threshold (upper threshold) and unmatched_threshold (lower thresholds)
  43   defining three categories of similarity which define whether examples are
  44   positive, negative, or ignored:
  45   (1) similarity >= matched_threshold: Highest similarity. Matched/Positive!
  46   (2) matched_threshold > similarity >= unmatched_threshold: Medium similarity.
  47           Depending on negatives_lower_than_unmatched, this is either
  48           Unmatched/Negative OR Ignore.
  49   (3) unmatched_threshold > similarity: Lowest similarity. Depending on flag
  50           negatives_lower_than_unmatched, either Unmatched/Negative OR Ignore.
  51   For ignored matches this class sets the values in the Match object to -2.
  52   """
  53 
  54   def __init__(self,
  55                matched_threshold,
  56                unmatched_threshold=None,
  57                negatives_lower_than_unmatched=True,
  58                force_match_for_each_row=False,
  59                use_matmul_gather=False):
  60     """Construct ArgMaxMatcher.
  61 
  62     Args:
  63       matched_threshold: Threshold for positive matches. Positive if
  64         sim >= matched_threshold, where sim is the maximum value of the
  65         similarity matrix for a given column. Set to None for no threshold.
  66       unmatched_threshold: Threshold for negative matches. Negative if
  67         sim < unmatched_threshold. Defaults to matched_threshold
  68         when set to None.
  69       negatives_lower_than_unmatched: Boolean which defaults to True. If True
  70         then negative matches are the ones below the unmatched_threshold,
  71         whereas ignored matches are in between the matched and umatched
  72         threshold. If False, then negative matches are in between the matched
  73         and unmatched threshold, and everything lower than unmatched is ignored.
  74       force_match_for_each_row: If True, ensures that each row is matched to
  75         at least one column (which is not guaranteed otherwise if the
  76         matched_threshold is high). Defaults to False. See
  77         argmax_matcher_test.testMatcherForceMatch() for an example.
  78       use_matmul_gather: Force constructed match objects to use matrix
  79         multiplication based gather instead of standard tf.gather.
  80         (Default: False).
  81 
  82     Raises:
  83       ValueError: if unmatched_threshold is set but matched_threshold is not set
  84         or if unmatched_threshold > matched_threshold.
  85     """
  86     super(ArgMaxMatcher, self).__init__(use_matmul_gather=use_matmul_gather)
  87     if (matched_threshold is None) and (unmatched_threshold is not None):
  88       raise ValueError('Need to also define matched_threshold when'
  89                        'unmatched_threshold is defined')
  90     self._matched_threshold = matched_threshold
  91     if unmatched_threshold is None:
  92       self._unmatched_threshold = matched_threshold
  93     else:
  94       if unmatched_threshold > matched_threshold:
  95         raise ValueError('unmatched_threshold needs to be smaller or equal'
  96                          'to matched_threshold')
  97       self._unmatched_threshold = unmatched_threshold
  98     if not negatives_lower_than_unmatched:
  99       if self._unmatched_threshold == self._matched_threshold:
 100         raise ValueError('When negatives are in between matched and '
 101                          'unmatched thresholds, these cannot be of equal '
 102                          'value. matched: {}, unmatched: {}'.format(
 103                              self._matched_threshold,
 104                              self._unmatched_threshold))
 105     self._force_match_for_each_row = force_match_for_each_row
 106     self._negatives_lower_than_unmatched = negatives_lower_than_unmatched
 107 
 108   def _match(self, similarity_matrix, valid_rows):
 109     """Tries to match each column of the similarity matrix to a row.
 110 
 111     Args:
 112       similarity_matrix: tensor of shape [N, M] representing any similarity
 113         metric.
 114       valid_rows: a boolean tensor of shape [N] indicating valid rows.
 115 
 116     Returns:
 117       Match object with corresponding matches for each of M columns.
 118     """
 119 
 120     def _match_when_rows_are_empty():
 121       """Performs matching when the rows of similarity matrix are empty.
 122 
 123       When the rows are empty, all detections are false positives. So we return
 124       a tensor of -1's to indicate that the columns do not match to any rows.
 125 
 126       Returns:
 127         matches:  int32 tensor indicating the row each column matches to.
 128       """
 129       similarity_matrix_shape = shape_utils.combined_static_and_dynamic_shape(
 130           similarity_matrix)
 131       return -1 * tf.ones([similarity_matrix_shape[1]], dtype=tf.int32)
 132 
 133     def _match_when_rows_are_non_empty():
 134       """Performs matching when the rows of similarity matrix are non empty.
 135 
 136       Returns:
 137         matches:  int32 tensor indicating the row each column matches to.
 138       """
 139       # Matches for each column
 140       matches = tf.argmax(similarity_matrix, 0, output_type=tf.int32)
 141 
 142       # Deal with matched and unmatched threshold
 143       if self._matched_threshold is not None:
 144         # Get logical indices of ignored and unmatched columns as tf.int64
 145         matched_vals = tf.reduce_max(similarity_matrix, 0)
 146         below_unmatched_threshold = tf.greater(self._unmatched_threshold,
 147                                                matched_vals)
 148         between_thresholds = tf.logical_and(
 149             tf.greater_equal(matched_vals, self._unmatched_threshold),
 150             tf.greater(self._matched_threshold, matched_vals))
 151 
 152         if self._negatives_lower_than_unmatched:
 153           matches = self._set_values_using_indicator(matches,
 154                                                      below_unmatched_threshold,
 155                                                      -1)
 156           matches = self._set_values_using_indicator(matches,
 157                                                      between_thresholds,
 158                                                      -2)
 159         else:
 160           matches = self._set_values_using_indicator(matches,
 161                                                      below_unmatched_threshold,
 162                                                      -2)
 163           matches = self._set_values_using_indicator(matches,
 164                                                      between_thresholds,
 165                                                      -1)
 166 
 167       if self._force_match_for_each_row:
 168         similarity_matrix_shape = shape_utils.combined_static_and_dynamic_shape(
 169             similarity_matrix)
 170         force_match_column_ids = tf.argmax(similarity_matrix, 1,
 171                                            output_type=tf.int32)
 172         force_match_column_indicators = (
 173             tf.one_hot(
 174                 force_match_column_ids, depth=similarity_matrix_shape[1]) *
 175             tf.cast(tf.expand_dims(valid_rows, axis=-1), dtype=tf.float32))
 176         force_match_row_ids = tf.argmax(force_match_column_indicators, 0,
 177                                         output_type=tf.int32)
 178         force_match_column_mask = tf.cast(
 179             tf.reduce_max(force_match_column_indicators, 0), tf.bool)
 180         final_matches = tf.where(force_match_column_mask,
 181                                  force_match_row_ids, matches)
 182         return final_matches
 183       else:
 184         return matches
 185 
 186     if similarity_matrix.shape.is_fully_defined():
 187       if shape_utils.get_dim_as_int(similarity_matrix.shape[0]) == 0:
 188         return _match_when_rows_are_empty()
 189       else:
 190         return _match_when_rows_are_non_empty()
 191     else:
 192       return tf.cond(
 193           tf.greater(tf.shape(similarity_matrix)[0], 0),
 194           _match_when_rows_are_non_empty, _match_when_rows_are_empty)
 195 
 196   def _set_values_using_indicator(self, x, indicator, val):
 197     """Set the indicated fields of x to val.
 198 
 199     Args:
 200       x: tensor.
 201       indicator: boolean with same shape as x.
 202       val: scalar with value to set.
 203 
 204     Returns:
 205       modified tensor.
 206     """
 207     indicator = tf.cast(indicator, x.dtype)
 208     return tf.add(tf.multiply(x, 1 - indicator), val * indicator)
