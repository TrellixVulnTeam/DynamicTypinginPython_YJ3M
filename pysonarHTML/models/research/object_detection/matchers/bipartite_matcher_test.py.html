  16 """Tests for object_detection.core.bipartite_matcher."""

  18 import numpy as np
  19 import tensorflow.compat.v1 as tf
  20 
  21 from object_detection.matchers import bipartite_matcher
  22 from object_detection.utils import test_case
  23 
  24 
  25 class GreedyBipartiteMatcherTest(test_case.TestCase):
  26 
  27   def test_get_expected_matches_when_all_rows_are_valid(self):
  28     similarity_matrix = np.array([[0.50, 0.1, 0.8], [0.15, 0.2, 0.3]],
  29                                  dtype=np.float32)
  30     valid_rows = np.ones([2], dtype=np.bool)
  31     expected_match_results = [-1, 1, 0]
  32     def graph_fn(similarity_matrix, valid_rows):
  33       matcher = bipartite_matcher.GreedyBipartiteMatcher()
  34       match = matcher.match(similarity_matrix, valid_rows=valid_rows)
  35       return match._match_results
  36     match_results_out = self.execute(graph_fn, [similarity_matrix, valid_rows])
  37     self.assertAllEqual(match_results_out, expected_match_results)
  38
