  70 def _prepare_groundtruth_for_eval(detection_model, class_agnostic,
  71                                   max_number_of_boxes):
  72   """Extracts groundtruth data from detection_model and prepares it for eval.
  73 
  74   Args:
  75     detection_model: A `DetectionModel` object.
  76     class_agnostic: Whether the detections are class_agnostic.
  77     max_number_of_boxes: Max number of groundtruth boxes.
  78 
  79   Returns:
  80     A tuple of:
  81     groundtruth: Dictionary with the following fields:
  82       'groundtruth_boxes': [batch_size, num_boxes, 4] float32 tensor of boxes,
  83         in normalized coordinates.
  84       'groundtruth_classes': [batch_size, num_boxes] int64 tensor of 1-indexed
  85         classes.
  86       'groundtruth_masks': 4D float32 tensor of instance masks (if provided in
  87         groundtruth)
  88       'groundtruth_is_crowd': [batch_size, num_boxes] bool tensor indicating
  89         is_crowd annotations (if provided in groundtruth).
  90       'groundtruth_area': [batch_size, num_boxes] float32 tensor indicating
  91         the area (in the original absolute coordinates) of annotations (if
  92         provided in groundtruth).
  93       'num_groundtruth_boxes': [batch_size] tensor containing the maximum number
  94         of groundtruth boxes per image..
  95       'groundtruth_keypoints': [batch_size, num_boxes, num_keypoints, 2] float32
  96         tensor of keypoints (if provided in groundtruth).
  97       'groundtruth_group_of': [batch_size, num_boxes] bool tensor indicating
  98         group_of annotations (if provided in groundtruth).
  99       'groundtruth_labeled_classes': [batch_size, num_classes] int64
 100         tensor of 1-indexed classes.
 101     class_agnostic: Boolean indicating whether detections are class agnostic.
 102   """
 103   input_data_fields = fields.InputDataFields()
 104   groundtruth_boxes = tf.stack(
 105       detection_model.groundtruth_lists(fields.BoxListFields.boxes))
 106   groundtruth_boxes_shape = tf.shape(groundtruth_boxes)
 107   # For class-agnostic models, groundtruth one-hot encodings collapse to all
 108   # ones.
 109   if class_agnostic:
 110     groundtruth_classes_one_hot = tf.ones(
 111         [groundtruth_boxes_shape[0], groundtruth_boxes_shape[1], 1])
 112   else:
 113     groundtruth_classes_one_hot = tf.stack(
 114         detection_model.groundtruth_lists(fields.BoxListFields.classes))
 115   label_id_offset = 1  # Applying label id offset (b/63711816)
 116   groundtruth_classes = (
 117       tf.argmax(groundtruth_classes_one_hot, axis=2) + label_id_offset)
 118   groundtruth = {
 119       input_data_fields.groundtruth_boxes: groundtruth_boxes,
 120       input_data_fields.groundtruth_classes: groundtruth_classes
 121   }
 122   if detection_model.groundtruth_has_field(fields.BoxListFields.masks):
 123     groundtruth[input_data_fields.groundtruth_instance_masks] = tf.stack(
 124         detection_model.groundtruth_lists(fields.BoxListFields.masks))
 125 
 126   if detection_model.groundtruth_has_field(fields.BoxListFields.is_crowd):
 127     groundtruth[input_data_fields.groundtruth_is_crowd] = tf.stack(
 128         detection_model.groundtruth_lists(fields.BoxListFields.is_crowd))
 129 
 130   if detection_model.groundtruth_has_field(input_data_fields.groundtruth_area):
 131     groundtruth[input_data_fields.groundtruth_area] = tf.stack(
 132         detection_model.groundtruth_lists(input_data_fields.groundtruth_area))
 133 
 134   if detection_model.groundtruth_has_field(fields.BoxListFields.keypoints):
 135     groundtruth[input_data_fields.groundtruth_keypoints] = tf.stack(
 136         detection_model.groundtruth_lists(fields.BoxListFields.keypoints))
 137 
 138   if detection_model.groundtruth_has_field(
 139       fields.BoxListFields.keypoint_visibilities):
 140     groundtruth[input_data_fields.groundtruth_keypoint_visibilities] = tf.stack(
 141         detection_model.groundtruth_lists(
 142             fields.BoxListFields.keypoint_visibilities))
 143 
 144   if detection_model.groundtruth_has_field(fields.BoxListFields.group_of):
 145     groundtruth[input_data_fields.groundtruth_group_of] = tf.stack(
 146         detection_model.groundtruth_lists(fields.BoxListFields.group_of))
 147 
 148   if detection_model.groundtruth_has_field(
 149       fields.InputDataFields.groundtruth_labeled_classes):
 150     labeled_classes_list = detection_model.groundtruth_lists(
 151         fields.InputDataFields.groundtruth_labeled_classes)
 152     labeled_classes = [
 153         tf.where(x)[:, 0] + label_id_offset for x in labeled_classes_list
 154     ]
 155     if len(labeled_classes) > 1:
 156       num_classes = labeled_classes_list[0].shape[0]
 157       padded_labeled_classes = []
 158       for x in labeled_classes:
 159         padding = num_classes - tf.shape(x)[0]
 160         padded_labeled_classes.append(tf.pad(x, [[0, padding]]))
 161       groundtruth[input_data_fields.groundtruth_labeled_classes] = tf.stack(
 162           padded_labeled_classes)
 163     else:
 164       groundtruth[input_data_fields.groundtruth_labeled_classes] = tf.stack(
 165           labeled_classes)
 166 
 167   groundtruth[input_data_fields.num_groundtruth_boxes] = (
 168       tf.tile([max_number_of_boxes], multiples=[groundtruth_boxes_shape[0]]))
 169   return groundtruth

 172 def unstack_batch(tensor_dict, unpad_groundtruth_tensors=True):
 173   """Unstacks all tensors in `tensor_dict` along 0th dimension.
 174 
 175   Unstacks tensor from the tensor dict along 0th dimension and returns a
 176   tensor_dict containing values that are lists of unstacked, unpadded tensors.
 177 
 178   Tensors in the `tensor_dict` are expected to be of one of the three shapes:
 179   1. [batch_size]
 180   2. [batch_size, height, width, channels]
 181   3. [batch_size, num_boxes, d1, d2, ... dn]
 182 
 183   When unpad_groundtruth_tensors is set to true, unstacked tensors of form 3
 184   above are sliced along the `num_boxes` dimension using the value in tensor
 185   field.InputDataFields.num_groundtruth_boxes.
 186 
 187   Note that this function has a static list of input data fields and has to be
 188   kept in sync with the InputDataFields defined in core/standard_fields.py
 189 
 190   Args:
 191     tensor_dict: A dictionary of batched groundtruth tensors.
 192     unpad_groundtruth_tensors: Whether to remove padding along `num_boxes`
 193       dimension of the groundtruth tensors.
 194 
 195   Returns:
 196     A dictionary where the keys are from fields.InputDataFields and values are
 197     a list of unstacked (optionally unpadded) tensors.
 198 
 199   Raises:
 200     ValueError: If unpad_tensors is True and `tensor_dict` does not contain
 201       `num_groundtruth_boxes` tensor.
 202   """
 203   unbatched_tensor_dict = {
 204       key: tf.unstack(tensor) for key, tensor in tensor_dict.items()
 205   }
 206   if unpad_groundtruth_tensors:
 207     if (fields.InputDataFields.num_groundtruth_boxes not in
 208         unbatched_tensor_dict):
 209       raise ValueError('`num_groundtruth_boxes` not found in tensor_dict. '
 210                        'Keys available: {}'.format(
 211                            unbatched_tensor_dict.keys()))
 212     unbatched_unpadded_tensor_dict = {}
 213     unpad_keys = set([
 214         # List of input data fields that are padded along the num_boxes
 215         # dimension. This list has to be kept in sync with InputDataFields in
 216         # standard_fields.py.
 217         fields.InputDataFields.groundtruth_instance_masks,
 218         fields.InputDataFields.groundtruth_classes,
 219         fields.InputDataFields.groundtruth_boxes,
 220         fields.InputDataFields.groundtruth_keypoints,
 221         fields.InputDataFields.groundtruth_keypoint_visibilities,
 222         fields.InputDataFields.groundtruth_group_of,
 223         fields.InputDataFields.groundtruth_difficult,
 224         fields.InputDataFields.groundtruth_is_crowd,
 225         fields.InputDataFields.groundtruth_area,
 226         fields.InputDataFields.groundtruth_weights
 227     ]).intersection(set(unbatched_tensor_dict.keys()))
 228 
 229     for key in unpad_keys:
 230       unpadded_tensor_list = []
 231       for num_gt, padded_tensor in zip(
 232           unbatched_tensor_dict[fields.InputDataFields.num_groundtruth_boxes],
 233           unbatched_tensor_dict[key]):
 234         tensor_shape = shape_utils.combined_static_and_dynamic_shape(
 235             padded_tensor)
 236         slice_begin = tf.zeros([len(tensor_shape)], dtype=tf.int32)
 237         slice_size = tf.stack(
 238             [num_gt] + [-1 if dim is None else dim for dim in tensor_shape[1:]])
 239         unpadded_tensor = tf.slice(padded_tensor, slice_begin, slice_size)
 240         unpadded_tensor_list.append(unpadded_tensor)
 241       unbatched_unpadded_tensor_dict[key] = unpadded_tensor_list
 242 
 243     unbatched_tensor_dict.update(unbatched_unpadded_tensor_dict)
 244 
 245   return unbatched_tensor_dict

 248 def provide_groundtruth(model, labels):
 249   """Provides the labels to a model as groundtruth.
 250 
 251   This helper function extracts the corresponding boxes, classes,
 252   keypoints, weights, masks, etc. from the labels, and provides it
 253   as groundtruth to the models.
 254 
 255   Args:
 256     model: The detection model to provide groundtruth to.
 257     labels: The labels for the training or evaluation inputs.
 258   """
 259   gt_boxes_list = labels[fields.InputDataFields.groundtruth_boxes]
 260   gt_classes_list = labels[fields.InputDataFields.groundtruth_classes]
 261   gt_masks_list = None
 262   if fields.InputDataFields.groundtruth_instance_masks in labels:
 263     gt_masks_list = labels[
 264         fields.InputDataFields.groundtruth_instance_masks]
 265   gt_keypoints_list = None
 266   if fields.InputDataFields.groundtruth_keypoints in labels:
 267     gt_keypoints_list = labels[fields.InputDataFields.groundtruth_keypoints]
 268   gt_keypoint_visibilities_list = None
 269   if fields.InputDataFields.groundtruth_keypoint_visibilities in labels:
 270     gt_keypoint_visibilities_list = labels[
 271         fields.InputDataFields.groundtruth_keypoint_visibilities]
 272   gt_weights_list = None
 273   if fields.InputDataFields.groundtruth_weights in labels:
 274     gt_weights_list = labels[fields.InputDataFields.groundtruth_weights]
 275   gt_confidences_list = None
 276   if fields.InputDataFields.groundtruth_confidences in labels:
 277     gt_confidences_list = labels[
 278         fields.InputDataFields.groundtruth_confidences]
 279   gt_is_crowd_list = None
 280   if fields.InputDataFields.groundtruth_is_crowd in labels:
 281     gt_is_crowd_list = labels[fields.InputDataFields.groundtruth_is_crowd]
 282   gt_group_of_list = None
 283   if fields.InputDataFields.groundtruth_group_of in labels:
 284     gt_group_of_list = labels[fields.InputDataFields.groundtruth_group_of]
 285   gt_area_list = None
 286   if fields.InputDataFields.groundtruth_area in labels:
 287     gt_area_list = labels[fields.InputDataFields.groundtruth_area]
 288   gt_labeled_classes = None
 289   if fields.InputDataFields.groundtruth_labeled_classes in labels:
 290     gt_labeled_classes = labels[
 291         fields.InputDataFields.groundtruth_labeled_classes]
 292   model.provide_groundtruth(
 293       groundtruth_boxes_list=gt_boxes_list,
 294       groundtruth_classes_list=gt_classes_list,
 295       groundtruth_confidences_list=gt_confidences_list,
 296       groundtruth_labeled_classes=gt_labeled_classes,
 297       groundtruth_masks_list=gt_masks_list,
 298       groundtruth_keypoints_list=gt_keypoints_list,
 299       groundtruth_keypoint_visibilities_list=gt_keypoint_visibilities_list,
 300       groundtruth_weights_list=gt_weights_list,
 301       groundtruth_is_crowd_list=gt_is_crowd_list,
 302       groundtruth_group_of_list=gt_group_of_list,
 303       groundtruth_area_list=gt_area_list)

 306 def create_model_fn(detection_model_fn, configs, hparams, use_tpu=False,
 307                     postprocess_on_cpu=False):
 308   """Creates a model function for `Estimator`.
 309 
 310   Args:
 311     detection_model_fn: Function that returns a `DetectionModel` instance.
 312     configs: Dictionary of pipeline config objects.
 313     hparams: `HParams` object.
 314     use_tpu: Boolean indicating whether model should be constructed for
 315         use on TPU.
 316     postprocess_on_cpu: When use_tpu and postprocess_on_cpu is true, postprocess
 317         is scheduled on the host cpu.
 318 
 319   Returns:
 320     `model_fn` for `Estimator`.
 321   """
 322   train_config = configs['train_config']
 323   eval_input_config = configs['eval_input_config']
 324   eval_config = configs['eval_config']
 325 
 326   def model_fn(features, labels, mode, params=None):
 327     """Constructs the object detection model.
 328 
 329     Args:
 330       features: Dictionary of feature tensors, returned from `input_fn`.
 331       labels: Dictionary of groundtruth tensors if mode is TRAIN or EVAL,
 332         otherwise None.
 333       mode: Mode key from tf.estimator.ModeKeys.
 334       params: Parameter dictionary passed from the estimator.
 335 
 336     Returns:
 337       An `EstimatorSpec` that encapsulates the model and its serving
 338         configurations.
 339     """
 340     params = params or {}
 341     total_loss, train_op, detections, export_outputs = None, None, None, None
 342     is_training = mode == tf.estimator.ModeKeys.TRAIN
 343 
 344     # Make sure to set the Keras learning phase. True during training,
 345     # False for inference.
 346     tf.keras.backend.set_learning_phase(is_training)
 347     # Set policy for mixed-precision training with Keras-based models.
 348     if use_tpu and train_config.use_bfloat16:
 349       from tensorflow.python.keras.engine import base_layer_utils  # pylint: disable=g-import-not-at-top
 350       # Enable v2 behavior, as `mixed_bfloat16` is only supported in TF 2.0.
 351       base_layer_utils.enable_v2_dtype_behavior()
 352       tf.compat.v2.keras.mixed_precision.experimental.set_policy(
 353           'mixed_bfloat16')
 354     detection_model = detection_model_fn(
 355         is_training=is_training, add_summaries=(not use_tpu))
 356     scaffold_fn = None
 357 
 358     if mode == tf.estimator.ModeKeys.TRAIN:
 359       labels = unstack_batch(
 360           labels,
 361           unpad_groundtruth_tensors=train_config.unpad_groundtruth_tensors)
 362     elif mode == tf.estimator.ModeKeys.EVAL:
 363       # For evaling on train data, it is necessary to check whether groundtruth
 364       # must be unpadded.
 365       boxes_shape = (
 366           labels[fields.InputDataFields.groundtruth_boxes].get_shape()
 367           .as_list())
 368       unpad_groundtruth_tensors = boxes_shape[1] is not None and not use_tpu
 369       labels = unstack_batch(
 370           labels, unpad_groundtruth_tensors=unpad_groundtruth_tensors)
 371 
 372     if mode in (tf.estimator.ModeKeys.TRAIN, tf.estimator.ModeKeys.EVAL):
 373       provide_groundtruth(detection_model, labels)
 374 
 375     preprocessed_images = features[fields.InputDataFields.image]
 376 
 377     side_inputs = detection_model.get_side_inputs(features)
 378 
 379     if use_tpu and train_config.use_bfloat16:
 380       with contrib_tpu.bfloat16_scope():
 381         prediction_dict = detection_model.predict(
 382             preprocessed_images,
 383             features[fields.InputDataFields.true_image_shape], **side_inputs)
 384         prediction_dict = ops.bfloat16_to_float32_nested(prediction_dict)
 385     else:
 386       prediction_dict = detection_model.predict(
 387           preprocessed_images,
 388           features[fields.InputDataFields.true_image_shape], **side_inputs)
 389 
 390     def postprocess_wrapper(args):
 391       return detection_model.postprocess(args[0], args[1])
 392 
 393     if mode in (tf.estimator.ModeKeys.EVAL, tf.estimator.ModeKeys.PREDICT):
 394       if use_tpu and postprocess_on_cpu:
 395         detections = contrib_tpu.outside_compilation(
 396             postprocess_wrapper,
 397             (prediction_dict,
 398              features[fields.InputDataFields.true_image_shape]))
 399       else:
 400         detections = postprocess_wrapper((
 401             prediction_dict,
 402             features[fields.InputDataFields.true_image_shape]))
 403 
 404     if mode == tf.estimator.ModeKeys.TRAIN:
 405       load_pretrained = hparams.load_pretrained if hparams else False
 406       if train_config.fine_tune_checkpoint and load_pretrained:
 407         if not train_config.fine_tune_checkpoint_type:
 408           # train_config.from_detection_checkpoint field is deprecated. For
 409           # backward compatibility, set train_config.fine_tune_checkpoint_type
 410           # based on train_config.from_detection_checkpoint.
 411           if train_config.from_detection_checkpoint:
 412             train_config.fine_tune_checkpoint_type = 'detection'
 413           else:
 414             train_config.fine_tune_checkpoint_type = 'classification'
 415         asg_map = detection_model.restore_map(
 416             fine_tune_checkpoint_type=train_config.fine_tune_checkpoint_type,
 417             load_all_detection_checkpoint_vars=(
 418                 train_config.load_all_detection_checkpoint_vars))
 419         available_var_map = (
 420             variables_helper.get_variables_available_in_checkpoint(
 421                 asg_map,
 422                 train_config.fine_tune_checkpoint,
 423                 include_global_step=False))
 424         if use_tpu:
 425 
 426           def tpu_scaffold():
 427             tf.train.init_from_checkpoint(train_config.fine_tune_checkpoint,
 428                                           available_var_map)
 429             return tf.train.Scaffold()
 430 
 431           scaffold_fn = tpu_scaffold
 432         else:
 433           tf.train.init_from_checkpoint(train_config.fine_tune_checkpoint,
 434                                         available_var_map)
 435 
 436     if mode in (tf.estimator.ModeKeys.TRAIN, tf.estimator.ModeKeys.EVAL):
 437       if (mode == tf.estimator.ModeKeys.EVAL and
 438           eval_config.use_dummy_loss_in_eval):
 439         total_loss = tf.constant(1.0)
 440         losses_dict = {'Loss/total_loss': total_loss}
 441       else:
 442         losses_dict = detection_model.loss(
 443             prediction_dict, features[fields.InputDataFields.true_image_shape])
 444         losses = [loss_tensor for loss_tensor in losses_dict.values()]
 445         if train_config.add_regularization_loss:
 446           regularization_losses = detection_model.regularization_losses()
 447           if use_tpu and train_config.use_bfloat16:
 448             regularization_losses = ops.bfloat16_to_float32_nested(
 449                 regularization_losses)
 450           if regularization_losses:
 451             regularization_loss = tf.add_n(
 452                 regularization_losses, name='regularization_loss')
 453             losses.append(regularization_loss)
 454             losses_dict['Loss/regularization_loss'] = regularization_loss
 455         total_loss = tf.add_n(losses, name='total_loss')
 456         losses_dict['Loss/total_loss'] = total_loss
 457 
 458       if 'graph_rewriter_config' in configs:
 459         graph_rewriter_fn = graph_rewriter_builder.build(
 460             configs['graph_rewriter_config'], is_training=is_training)
 461         graph_rewriter_fn()
 462 
 463       # TODO(rathodv): Stop creating optimizer summary vars in EVAL mode once we
 464       # can write learning rate summaries on TPU without host calls.
 465       global_step = tf.train.get_or_create_global_step()
 466       training_optimizer, optimizer_summary_vars = optimizer_builder.build(
 467           train_config.optimizer)
 468 
 469     if mode == tf.estimator.ModeKeys.TRAIN:
 470       if use_tpu:
 471         training_optimizer = contrib_tpu.CrossShardOptimizer(training_optimizer)
 472 
 473       # Optionally freeze some layers by setting their gradients to be zero.
 474       trainable_variables = None
 475       include_variables = (
 476           train_config.update_trainable_variables
 477           if train_config.update_trainable_variables else None)
 478       exclude_variables = (
 479           train_config.freeze_variables
 480           if train_config.freeze_variables else None)
 481       trainable_variables = slim.filter_variables(
 482           tf.trainable_variables(),
 483           include_patterns=include_variables,
 484           exclude_patterns=exclude_variables)
 485 
 486       clip_gradients_value = None
 487       if train_config.gradient_clipping_by_norm > 0:
 488         clip_gradients_value = train_config.gradient_clipping_by_norm
 489 
 490       if not use_tpu:
 491         for var in optimizer_summary_vars:
 492           tf.summary.scalar(var.op.name, var)
 493       summaries = [] if use_tpu else None
 494       if train_config.summarize_gradients:
 495         summaries = ['gradients', 'gradient_norm', 'global_gradient_norm']
 496       train_op = slim.optimizers.optimize_loss(
 497           loss=total_loss,
 498           global_step=global_step,
 499           learning_rate=None,
 500           clip_gradients=clip_gradients_value,
 501           optimizer=training_optimizer,
 502           update_ops=detection_model.updates(),
 503           variables=trainable_variables,
 504           summaries=summaries,
 505           name='')  # Preventing scope prefix on all variables.
 506 
 507     if mode == tf.estimator.ModeKeys.PREDICT:
 508       exported_output = exporter_lib.add_output_tensor_nodes(detections)
 509       export_outputs = {
 510           tf.saved_model.signature_constants.PREDICT_METHOD_NAME:
 511               tf.estimator.export.PredictOutput(exported_output)
 512       }
 513 
 514     eval_metric_ops = None
 515     scaffold = None
 516     if mode == tf.estimator.ModeKeys.EVAL:
 517       class_agnostic = (
 518           fields.DetectionResultFields.detection_classes not in detections)
 519       groundtruth = _prepare_groundtruth_for_eval(
 520           detection_model, class_agnostic,
 521           eval_input_config.max_number_of_boxes)
 522       use_original_images = fields.InputDataFields.original_image in features
 523       if use_original_images:
 524         eval_images = features[fields.InputDataFields.original_image]
 525         true_image_shapes = tf.slice(
 526             features[fields.InputDataFields.true_image_shape], [0, 0], [-1, 3])
 527         original_image_spatial_shapes = features[fields.InputDataFields
 528                                                  .original_image_spatial_shape]
 529       else:
 530         eval_images = features[fields.InputDataFields.image]
 531         true_image_shapes = None
 532         original_image_spatial_shapes = None
 533 
 534       eval_dict = eval_util.result_dict_for_batched_example(
 535           eval_images,
 536           features[inputs.HASH_KEY],
 537           detections,
 538           groundtruth,
 539           class_agnostic=class_agnostic,
 540           scale_to_absolute=True,
 541           original_image_spatial_shapes=original_image_spatial_shapes,
 542           true_image_shapes=true_image_shapes)
 543 
 544       if fields.InputDataFields.image_additional_channels in features:
 545         eval_dict[fields.InputDataFields.image_additional_channels] = features[
 546             fields.InputDataFields.image_additional_channels]
 547 
 548       if class_agnostic:
 549         category_index = label_map_util.create_class_agnostic_category_index()
 550       else:
 551         category_index = label_map_util.create_category_index_from_labelmap(
 552             eval_input_config.label_map_path)
 553       vis_metric_ops = None
 554       if not use_tpu and use_original_images:
 555         keypoint_edges = [
 556             (kp.start, kp.end) for kp in eval_config.keypoint_edge]
 557 
 558         eval_metric_op_vis = vis_utils.VisualizeSingleFrameDetections(
 559             category_index,
 560             max_examples_to_draw=eval_config.num_visualizations,
 561             max_boxes_to_draw=eval_config.max_num_boxes_to_visualize,
 562             min_score_thresh=eval_config.min_score_threshold,
 563             use_normalized_coordinates=False,
 564             keypoint_edges=keypoint_edges or None)
 565         vis_metric_ops = eval_metric_op_vis.get_estimator_eval_metric_ops(
 566             eval_dict)
 567 
 568       # Eval metrics on a single example.
 569       eval_metric_ops = eval_util.get_eval_metric_ops_for_evaluators(
 570           eval_config, list(category_index.values()), eval_dict)
 571       for loss_key, loss_tensor in iter(losses_dict.items()):
 572         eval_metric_ops[loss_key] = tf.metrics.mean(loss_tensor)
 573       for var in optimizer_summary_vars:
 574         eval_metric_ops[var.op.name] = (var, tf.no_op())
 575       if vis_metric_ops is not None:
 576         eval_metric_ops.update(vis_metric_ops)
 577       eval_metric_ops = {str(k): v for k, v in eval_metric_ops.items()}
 578 
 579       if eval_config.use_moving_averages:
 580         variable_averages = tf.train.ExponentialMovingAverage(0.0)
 581         variables_to_restore = variable_averages.variables_to_restore()
 582         keep_checkpoint_every_n_hours = (
 583             train_config.keep_checkpoint_every_n_hours)
 584         saver = tf.train.Saver(
 585             variables_to_restore,
 586             keep_checkpoint_every_n_hours=keep_checkpoint_every_n_hours)
 587         scaffold = tf.train.Scaffold(saver=saver)
 588 
 589     # EVAL executes on CPU, so use regular non-TPU EstimatorSpec.
 590     if use_tpu and mode != tf.estimator.ModeKeys.EVAL:
 591       return contrib_tpu.TPUEstimatorSpec(
 592           mode=mode,
 593           scaffold_fn=scaffold_fn,
 594           predictions=detections,
 595           loss=total_loss,
 596           train_op=train_op,
 597           eval_metrics=eval_metric_ops,
 598           export_outputs=export_outputs)
 599     else:
 600       if scaffold is None:
 601         keep_checkpoint_every_n_hours = (
 602             train_config.keep_checkpoint_every_n_hours)
 603         saver = tf.train.Saver(
 604             sharded=True,
 605             keep_checkpoint_every_n_hours=keep_checkpoint_every_n_hours,
 606             save_relative_paths=True)
 607         tf.add_to_collection(tf.GraphKeys.SAVERS, saver)
 608         scaffold = tf.train.Scaffold(saver=saver)
 609       return tf.estimator.EstimatorSpec(
 610           mode=mode,
 611           predictions=detections,
 612           loss=total_loss,
 613           train_op=train_op,
 614           eval_metric_ops=eval_metric_ops,
 615           export_outputs=export_outputs,
 616           scaffold=scaffold)
 617 
 618   return model_fn

 621 def create_estimator_and_inputs(run_config,
 622                                 hparams,
 623                                 pipeline_config_path,
 624                                 config_override=None,
 625                                 train_steps=None,
 626                                 sample_1_of_n_eval_examples=1,
 627                                 sample_1_of_n_eval_on_train_examples=1,
 628                                 model_fn_creator=create_model_fn,
 629                                 use_tpu_estimator=False,
 630                                 use_tpu=False,
 631                                 num_shards=1,
 632                                 params=None,
 633                                 override_eval_num_epochs=True,
 634                                 save_final_config=False,
 635                                 postprocess_on_cpu=False,
 636                                 export_to_tpu=None,
 637                                 **kwargs):
 638   """Creates `Estimator`, input functions, and steps.
 639 
 640   Args:
 641     run_config: A `RunConfig`.
 642     hparams: A `HParams`.
 643     pipeline_config_path: A path to a pipeline config file.
 644     config_override: A pipeline_pb2.TrainEvalPipelineConfig text proto to
 645       override the config from `pipeline_config_path`.
 646     train_steps: Number of training steps. If None, the number of training steps
 647       is set from the `TrainConfig` proto.
 648     sample_1_of_n_eval_examples: Integer representing how often an eval example
 649       should be sampled. If 1, will sample all examples.
 650     sample_1_of_n_eval_on_train_examples: Similar to
 651       `sample_1_of_n_eval_examples`, except controls the sampling of training
 652       data for evaluation.
 653     model_fn_creator: A function that creates a `model_fn` for `Estimator`.
 654       Follows the signature:
 655 
 656       * Args:
 657         * `detection_model_fn`: Function that returns `DetectionModel` instance.
 658         * `configs`: Dictionary of pipeline config objects.
 659         * `hparams`: `HParams` object.
 660       * Returns:
 661         `model_fn` for `Estimator`.
 662 
 663     use_tpu_estimator: Whether a `TPUEstimator` should be returned. If False,
 664       an `Estimator` will be returned.
 665     use_tpu: Boolean, whether training and evaluation should run on TPU. Only
 666       used if `use_tpu_estimator` is True.
 667     num_shards: Number of shards (TPU cores). Only used if `use_tpu_estimator`
 668       is True.
 669     params: Parameter dictionary passed from the estimator. Only used if
 670       `use_tpu_estimator` is True.
 671     override_eval_num_epochs: Whether to overwrite the number of epochs to 1 for
 672       eval_input.
 673     save_final_config: Whether to save final config (obtained after applying
 674       overrides) to `estimator.model_dir`.
 675     postprocess_on_cpu: When use_tpu and postprocess_on_cpu are true,
 676       postprocess is scheduled on the host cpu.
 677     export_to_tpu: When use_tpu and export_to_tpu are true,
 678       `export_savedmodel()` exports a metagraph for serving on TPU besides the
 679       one on CPU.
 680     **kwargs: Additional keyword arguments for configuration override.
 681 
 682   Returns:
 683     A dictionary with the following fields:
 684     'estimator': An `Estimator` or `TPUEstimator`.
 685     'train_input_fn': A training input function.
 686     'eval_input_fns': A list of all evaluation input functions.
 687     'eval_input_names': A list of names for each evaluation input.
 688     'eval_on_train_input_fn': An evaluation-on-train input function.
 689     'predict_input_fn': A prediction input function.
 690     'train_steps': Number of training steps. Either directly from input or from
 691       configuration.
 692   """
 693   get_configs_from_pipeline_file = MODEL_BUILD_UTIL_MAP[
 694       'get_configs_from_pipeline_file']
 695   merge_external_params_with_configs = MODEL_BUILD_UTIL_MAP[
 696       'merge_external_params_with_configs']
 697   create_pipeline_proto_from_configs = MODEL_BUILD_UTIL_MAP[
 698       'create_pipeline_proto_from_configs']
 699   create_train_input_fn = MODEL_BUILD_UTIL_MAP['create_train_input_fn']
 700   create_eval_input_fn = MODEL_BUILD_UTIL_MAP['create_eval_input_fn']
 701   create_predict_input_fn = MODEL_BUILD_UTIL_MAP['create_predict_input_fn']
 702   detection_model_fn_base = MODEL_BUILD_UTIL_MAP['detection_model_fn_base']
 703 
 704   configs = get_configs_from_pipeline_file(
 705       pipeline_config_path, config_override=config_override)
 706   kwargs.update({
 707       'train_steps': train_steps,
 708       'use_bfloat16': configs['train_config'].use_bfloat16 and use_tpu
 709   })
 710   if sample_1_of_n_eval_examples >= 1:
 711     kwargs.update({
 712         'sample_1_of_n_eval_examples': sample_1_of_n_eval_examples
 713     })
 714   if override_eval_num_epochs:
 715     kwargs.update({'eval_num_epochs': 1})
 716     tf.logging.warning(
 717         'Forced number of epochs for all eval validations to be 1.')
 718   configs = merge_external_params_with_configs(
 719       configs, hparams, kwargs_dict=kwargs)
 720   model_config = configs['model']
 721   train_config = configs['train_config']
 722   train_input_config = configs['train_input_config']
 723   eval_config = configs['eval_config']
 724   eval_input_configs = configs['eval_input_configs']
 725   eval_on_train_input_config = copy.deepcopy(train_input_config)
 726   eval_on_train_input_config.sample_1_of_n_examples = (
 727       sample_1_of_n_eval_on_train_examples)
 728   if override_eval_num_epochs and eval_on_train_input_config.num_epochs != 1:
 729     tf.logging.warning('Expected number of evaluation epochs is 1, but '
 730                        'instead encountered `eval_on_train_input_config'
 731                        '.num_epochs` = '
 732                        '{}. Overwriting `num_epochs` to 1.'.format(
 733                            eval_on_train_input_config.num_epochs))
 734     eval_on_train_input_config.num_epochs = 1
 735 
 736   # update train_steps from config but only when non-zero value is provided
 737   if train_steps is None and train_config.num_steps != 0:
 738     train_steps = train_config.num_steps
 739 
 740   detection_model_fn = functools.partial(
 741       detection_model_fn_base, model_config=model_config)
 742 
 743   # Create the input functions for TRAIN/EVAL/PREDICT.
 744   train_input_fn = create_train_input_fn(
 745       train_config=train_config,
 746       train_input_config=train_input_config,
 747       model_config=model_config)
 748   eval_input_fns = [
 749       create_eval_input_fn(
 750           eval_config=eval_config,
 751           eval_input_config=eval_input_config,
 752           model_config=model_config) for eval_input_config in eval_input_configs
 753   ]
 754   eval_input_names = [
 755       eval_input_config.name for eval_input_config in eval_input_configs
 756   ]
 757   eval_on_train_input_fn = create_eval_input_fn(
 758       eval_config=eval_config,
 759       eval_input_config=eval_on_train_input_config,
 760       model_config=model_config)
 761   predict_input_fn = create_predict_input_fn(
 762       model_config=model_config, predict_input_config=eval_input_configs[0])
 763 
 764   # Read export_to_tpu from hparams if not passed.
 765   if export_to_tpu is None:
 766     export_to_tpu = hparams.get('export_to_tpu', False)
 767   tf.logging.info('create_estimator_and_inputs: use_tpu %s, export_to_tpu %s',
 768                   use_tpu, export_to_tpu)
 769   model_fn = model_fn_creator(detection_model_fn, configs, hparams, use_tpu,
 770                               postprocess_on_cpu)
 771   if use_tpu_estimator:
 772     estimator = contrib_tpu.TPUEstimator(
 773         model_fn=model_fn,
 774         train_batch_size=train_config.batch_size,
 775         # For each core, only batch size 1 is supported for eval.
 776         eval_batch_size=num_shards * 1 if use_tpu else 1,
 777         use_tpu=use_tpu,
 778         config=run_config,
 779         export_to_tpu=export_to_tpu,
 780         eval_on_tpu=False,  # Eval runs on CPU, so disable eval on TPU
 781         params=params if params else {})
 782   else:
 783     estimator = tf.estimator.Estimator(model_fn=model_fn, config=run_config)
 784 
 785   # Write the as-run pipeline config to disk.
 786   if run_config.is_chief and save_final_config:
 787     pipeline_config_final = create_pipeline_proto_from_configs(configs)
 788     config_util.save_pipeline_config(pipeline_config_final, estimator.model_dir)
 789 
 790   return dict(
 791       estimator=estimator,
 792       train_input_fn=train_input_fn,
 793       eval_input_fns=eval_input_fns,
 794       eval_input_names=eval_input_names,
 795       eval_on_train_input_fn=eval_on_train_input_fn,
 796       predict_input_fn=predict_input_fn,
 797       train_steps=train_steps)

 800 def create_train_and_eval_specs(train_input_fn,
 801                                 eval_input_fns,
 802                                 eval_on_train_input_fn,
 803                                 predict_input_fn,
 804                                 train_steps,
 805                                 eval_on_train_data=False,
 806                                 final_exporter_name='Servo',
 807                                 eval_spec_names=None):
 808   """Creates a `TrainSpec` and `EvalSpec`s.
 809 
 810   Args:
 811     train_input_fn: Function that produces features and labels on train data.
 812     eval_input_fns: A list of functions that produce features and labels on eval
 813       data.
 814     eval_on_train_input_fn: Function that produces features and labels for
 815       evaluation on train data.
 816     predict_input_fn: Function that produces features for inference.
 817     train_steps: Number of training steps.
 818     eval_on_train_data: Whether to evaluate model on training data. Default is
 819       False.
 820     final_exporter_name: String name given to `FinalExporter`.
 821     eval_spec_names: A list of string names for each `EvalSpec`.
 822 
 823   Returns:
 824     Tuple of `TrainSpec` and list of `EvalSpecs`. If `eval_on_train_data` is
 825     True, the last `EvalSpec` in the list will correspond to training data. The
 826     rest EvalSpecs in the list are evaluation datas.
 827   """
 828   train_spec = tf.estimator.TrainSpec(
 829       input_fn=train_input_fn, max_steps=train_steps)
 830 
 831   if eval_spec_names is None:
 832     eval_spec_names = [str(i) for i in range(len(eval_input_fns))]
 833 
 834   eval_specs = []
 835   for index, (eval_spec_name, eval_input_fn) in enumerate(
 836       zip(eval_spec_names, eval_input_fns)):
 837     # Uses final_exporter_name as exporter_name for the first eval spec for
 838     # backward compatibility.
 839     if index == 0:
 840       exporter_name = final_exporter_name
 841     else:
 842       exporter_name = '{}_{}'.format(final_exporter_name, eval_spec_name)
 843     exporter = tf.estimator.FinalExporter(
 844         name=exporter_name, serving_input_receiver_fn=predict_input_fn)
 845     eval_specs.append(
 846         tf.estimator.EvalSpec(
 847             name=eval_spec_name,
 848             input_fn=eval_input_fn,
 849             steps=None,
 850             exporters=exporter))
 851 
 852   if eval_on_train_data:
 853     eval_specs.append(
 854         tf.estimator.EvalSpec(
 855             name='eval_on_train', input_fn=eval_on_train_input_fn, steps=None))
 856 
 857   return train_spec, eval_specs

 860 def _evaluate_checkpoint(estimator,
 861                          input_fn,
 862                          checkpoint_path,
 863                          name,
 864                          max_retries=0):
 865   """Evaluates a checkpoint.
 866 
 867   Args:
 868     estimator: Estimator object to use for evaluation.
 869     input_fn: Input function to use for evaluation.
 870     checkpoint_path: Path of the checkpoint to evaluate.
 871     name: Namescope for eval summary.
 872     max_retries: Maximum number of times to retry the evaluation on encountering
 873       a tf.errors.InvalidArgumentError. If negative, will always retry the
 874       evaluation.
 875 
 876   Returns:
 877     Estimator evaluation results.
 878   """
 879   always_retry = True if max_retries < 0 else False
 880   retries = 0
 881   while always_retry or retries <= max_retries:
 882     try:
 883       return estimator.evaluate(
 884           input_fn=input_fn,
 885           steps=None,
 886           checkpoint_path=checkpoint_path,
 887           name=name)
 888     except tf.errors.InvalidArgumentError as e:
 889       if always_retry or retries < max_retries:
 890         tf.logging.info('Retrying checkpoint evaluation after exception: %s', e)
 891         retries += 1
 892       else:
 893         raise e

 896 def continuous_eval(estimator,
 897                     model_dir,
 898                     input_fn,
 899                     train_steps,
 900                     name,
 901                     max_retries=0):
 902   """Perform continuous evaluation on checkpoints written to a model directory.
 903 
 904   Args:
 905     estimator: Estimator object to use for evaluation.
 906     model_dir: Model directory to read checkpoints for continuous evaluation.
 907     input_fn: Input function to use for evaluation.
 908     train_steps: Number of training steps. This is used to infer the last
 909       checkpoint and stop evaluation loop.
 910     name: Namescope for eval summary.
 911     max_retries: Maximum number of times to retry the evaluation on encountering
 912       a tf.errors.InvalidArgumentError. If negative, will always retry the
 913       evaluation.
 914   """
 915 
 916   def terminate_eval():
 917     tf.logging.info('Terminating eval after 180 seconds of no checkpoints')
 918     return True
 919 
 920   for ckpt in tf.train.checkpoints_iterator(
 921       model_dir, min_interval_secs=180, timeout=None,
 922       timeout_fn=terminate_eval):
 923 
 924     tf.logging.info('Starting Evaluation.')
 925     try:
 926       eval_results = _evaluate_checkpoint(
 927           estimator=estimator,
 928           input_fn=input_fn,
 929           checkpoint_path=ckpt,
 930           name=name,
 931           max_retries=max_retries)
 932       tf.logging.info('Eval results: %s' % eval_results)
 933 
 934       # Terminate eval job when final checkpoint is reached
 935       current_step = int(os.path.basename(ckpt).split('-')[1])
 936       if current_step >= train_steps:
 937         tf.logging.info(
 938             'Evaluation finished after training step %d' % current_step)
 939         break
 940 
 941     except tf.errors.NotFoundError:
 942       tf.logging.info(
 943           'Checkpoint %s no longer exists, skipping checkpoint' % ckpt)
