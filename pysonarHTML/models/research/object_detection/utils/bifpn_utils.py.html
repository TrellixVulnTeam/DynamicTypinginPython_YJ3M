  15 """Functions to manipulate feature map pyramids, such as for FPNs and BiFPNs.
  16

  18 such as combining multiple feature maps, upsampling or downsampling feature
  19 maps, and applying blocks of convolution, batchnorm, and activation layers.
  20 """

  27 def create_conv_block(name, num_filters, kernel_size, strides, padding,
  28                       use_separable, apply_batchnorm, apply_activation,
  29                       conv_hyperparams, is_training, freeze_batchnorm):
  30   """Create Keras layers for regular or separable convolutions.
  31 
  32   Args:
  33     name: String. The name of the layer.
  34     num_filters: Number of filters (channels) for the output feature maps.
  35     kernel_size: A list of length 2: [kernel_height, kernel_width] of the
  36       filters, or a single int if both values are the same.
  37     strides: A list of length 2: [stride_height, stride_width], specifying the
  38       convolution stride, or a single int if both strides are the same.
  39     padding: One of 'VALID' or 'SAME'.
  40     use_separable: Bool. Whether to use depthwise separable convolution instead
  41       of regular convolution.
  42     apply_batchnorm: Bool. Whether to apply a batch normalization layer after
  43       convolution, constructed according to the conv_hyperparams.
  44     apply_activation: Bool. Whether to apply an activation layer after
  45       convolution, constructed according to the conv_hyperparams.
  46     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
  47       containing hyperparameters for convolution ops.
  48     is_training: Bool. Whether the feature generator is in training mode.
  49     freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
  50       training or not. When training with a small batch size (e.g. 1), it is
  51       desirable to freeze batch norm update and use pretrained batch norm
  52       params.
  53 
  54   Returns:
  55     A list of keras layers, including (regular or seperable) convolution, and
  56     optionally batch normalization and activation layers.
  57   """
  58   layers = []
  59   if use_separable:
  60     kwargs = conv_hyperparams.params()
  61     # Both the regularizer and initializer apply to the depthwise layer,
  62     # so we remap the kernel_* to depthwise_* here.
  63     kwargs['depthwise_regularizer'] = kwargs['kernel_regularizer']
  64     kwargs['depthwise_initializer'] = kwargs['kernel_initializer']
  65     # TODO(aom): Verify that the pointwise regularizer/initializer should be set
  66     # here, since this is not the case in feature_map_generators.py
  67     kwargs['pointwise_regularizer'] = kwargs['kernel_regularizer']
  68     kwargs['pointwise_initializer'] = kwargs['kernel_initializer']
  69     layers.append(
  70         tf.keras.layers.SeparableConv2D(
  71             filters=num_filters,
  72             kernel_size=kernel_size,
  73             depth_multiplier=1,
  74             padding=padding,
  75             strides=strides,
  76             name=name + '_separable_conv',
  77             **kwargs))
  78   else:
  79     layers.append(
  80         tf.keras.layers.Conv2D(
  81             filters=num_filters,
  82             kernel_size=kernel_size,
  83             padding=padding,
  84             strides=strides,
  85             name=name + '_conv',
  86             **conv_hyperparams.params()))
  87 
  88   if apply_batchnorm:
  89     layers.append(
  90         conv_hyperparams.build_batch_norm(
  91             training=(is_training and not freeze_batchnorm),
  92             name=name + '_batchnorm'))
  93 
  94   if apply_activation:
  95     layers.append(
  96         conv_hyperparams.build_activation_layer(name=name + '_activation'))
  97 
  98   return layers
  99

 101 def create_downsample_feature_map_ops(scale, downsample_method,
 102                                       conv_hyperparams, is_training,
 103                                       freeze_batchnorm, name):
 104   """Creates Keras layers for downsampling feature maps.
 105 
 106   Args:
 107     scale: Int. The scale factor by which to downsample input feature maps. For
 108       example, in the case of a typical feature map pyramid, the scale factor
 109       between level_i and level_i+1 is 2.
 110     downsample_method: String. The method used for downsampling. Currently
 111       supported methods include 'max_pooling', 'avg_pooling', and
 112       'depthwise_conv'.
 113     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 114       containing hyperparameters for convolution ops.
 115     is_training: Bool. Whether the feature generator is in training mode.
 116     freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
 117       training or not. When training with a small batch size (e.g. 1), it is
 118       desirable to freeze batch norm update and use pretrained batch norm
 119       params.
 120     name: String. The name used to prefix the constructed layers.
 121 
 122   Returns:
 123     A list of Keras layers which will downsample input feature maps by the
 124     desired scale factor.
 125   """
 126   layers = []
 127   padding = 'SAME'
 128   stride = int(scale)
 129   kernel_size = stride + 1
 130   if downsample_method == 'max_pooling':
 131     layers.append(
 132         tf.keras.layers.MaxPooling2D(
 133             pool_size=kernel_size,
 134             strides=stride,
 135             padding=padding,
 136             name=name + '_downsample_max_x{}'.format(stride)))
 137   elif downsample_method == 'avg_pooling':
 138     layers.append(
 139         tf.keras.layers.AveragePooling2D(
 140             pool_size=kernel_size,
 141             strides=stride,
 142             padding=padding,
 143             name=name + '_downsample_avg_x{}'.format(stride)))
 144   elif downsample_method == 'depthwise_conv':
 145     layers.append(
 146         tf.keras.layers.DepthwiseConv2D(
 147             kernel_size=kernel_size,
 148             strides=stride,
 149             padding=padding,
 150             name=name + '_downsample_depthwise_x{}'.format(stride)))
 151     layers.append(
 152         conv_hyperparams.build_batch_norm(
 153             training=(is_training and not freeze_batchnorm),
 154             name=name + '_downsample_batchnorm'))
 155     layers.append(
 156         conv_hyperparams.build_activation_layer(name=name +
 157                                                 '_downsample_activation'))
 158   else:
 159     raise ValueError('Unknown downsample method: {}'.format(downsample_method))
 160 
 161   return layers

 164 def create_upsample_feature_map_ops(scale, use_native_resize_op, name):
 165   """Creates Keras layers for upsampling feature maps.
 166 
 167   Args:
 168     scale: Int. The scale factor by which to upsample input feature maps. For
 169       example, in the case of a typical feature map pyramid, the scale factor
 170       between level_i and level_i-1 is 2.
 171     use_native_resize_op: If True, uses tf.image.resize_nearest_neighbor op for
 172       the upsampling process instead of reshape and broadcasting implementation.
 173     name: String. The name used to prefix the constructed layers.
 174 
 175   Returns:
 176     A list of Keras layers which will upsample input feature maps by the
 177     desired scale factor.
 178   """
 179   layers = []
 180   if use_native_resize_op:
 181 
 182     def resize_nearest_neighbor(image):
 183       image_shape = shape_utils.combined_static_and_dynamic_shape(image)
 184       return tf.image.resize_nearest_neighbor(
 185           image, [image_shape[1] * scale, image_shape[2] * scale])
 186 
 187     layers.append(
 188         tf.keras.layers.Lambda(
 189             resize_nearest_neighbor,
 190             name=name + 'nearest_neighbor_upsampling_x{}'.format(scale)))
 191   else:
 192 
 193     def nearest_neighbor_upsampling(image):
 194       return ops.nearest_neighbor_upsampling(image, scale=scale)
 195 
 196     layers.append(
 197         tf.keras.layers.Lambda(
 198             nearest_neighbor_upsampling,
 199             name=name + 'nearest_neighbor_upsampling_x{}'.format(scale)))
 200 
 201   return layers

 204 def create_resample_feature_map_ops(input_scale_factor, output_scale_factor,
 205                                     downsample_method, use_native_resize_op,
 206                                     conv_hyperparams, is_training,
 207                                     freeze_batchnorm, name):
 208   """Creates Keras layers for downsampling or upsampling feature maps.
 209 
 210   Args:
 211     input_scale_factor: Int. Scale factor of the input feature map. For example,
 212       for a feature pyramid where each successive level halves its spatial
 213       resolution, the scale factor of a level is 2^level. The input and output
 214       scale factors are used to compute the scale for upsampling or downsamling,
 215       so they should be evenly divisible.
 216     output_scale_factor: Int. Scale factor of the output feature map. See
 217       input_scale_factor for additional details.
 218     downsample_method: String. The method used for downsampling. See
 219       create_downsample_feature_map_ops for details on supported methods.
 220     use_native_resize_op: If True, uses tf.image.resize_nearest_neighbor op for
 221       the upsampling process instead of reshape and broadcasting implementation.
 222       See create_upsample_feature_map_ops for details.
 223     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 224       containing hyperparameters for convolution ops.
 225     is_training: Bool. Whether the feature generator is in training mode.
 226     freeze_batchnorm: Bool. Whether to freeze batch norm parameters during
 227       training or not. When training with a small batch size (e.g. 1), it is
 228       desirable to freeze batch norm update and use pretrained batch norm
 229       params.
 230     name: String. The name used to prefix the constructed layers.
 231 
 232   Returns:
 233     A list of Keras layers which will downsample or upsample input feature maps
 234     to match the desired output feature map scale.
 235   """
 236   if input_scale_factor < output_scale_factor:
 237     if output_scale_factor % input_scale_factor != 0:
 238       raise ValueError('Invalid scale factor: input scale 1/{} not divisible by'
 239                        'output scale 1/{}'.format(input_scale_factor,
 240                                                   output_scale_factor))
 241     scale = output_scale_factor // input_scale_factor
 242     return create_downsample_feature_map_ops(scale, downsample_method,
 243                                              conv_hyperparams, is_training,
 244                                              freeze_batchnorm, name)
 245   elif input_scale_factor > output_scale_factor:
 246     if input_scale_factor % output_scale_factor != 0:
 247       raise ValueError('Invalid scale factor: input scale 1/{} not a divisor of'
 248                        'output scale 1/{}'.format(input_scale_factor,
 249                                                   output_scale_factor))
 250     scale = input_scale_factor // output_scale_factor
 251     return create_upsample_feature_map_ops(scale, use_native_resize_op, name)
 252   else:
 253     return []

 257 class BiFPNCombineLayer(tf.keras.layers.Layer):
 258   """Combines multiple input feature maps into a single output feature map.
 259 
 260   A Keras layer which combines multiple input feature maps into a single output
 261   feature map, according to the desired combination method. Options for
 262   combining feature maps include simple summation, or several types of weighted
 263   sums using learned weights for each input feature map. These include
 264   'weighted_sum', 'attention', and 'fast_attention'. For more details, see the
 265   EfficientDet paper by Tan et al, see arxiv.org/abs/1911.09070.
 266 
 267   Specifically, this layer takes a list of tensors as input, all of the same
 268   shape, and returns a single tensor, also of the same shape.
 269   """
 270 
 271   def __init__(self, combine_method, **kwargs):
 272     """Constructor.
 273 
 274     Args:
 275       combine_method: String. The method used to combine the input feature maps
 276         into a single output feature map. One of 'sum', 'weighted_sum',
 277         'attention', or 'fast_attention'.
 278       **kwargs: Additional Keras layer arguments.
 279     """
 280     super(BiFPNCombineLayer, self).__init__(**kwargs)
 281     self.combine_method = combine_method
 282 
 283   def _combine_weighted_sum(self, inputs):
 284     return tf.squeeze(
 285         tf.linalg.matmul(tf.stack(inputs, axis=-1), self.per_input_weights),
 286         axis=[-1])
 287 
 288   def _combine_attention(self, inputs):
 289     normalized_weights = tf.nn.softmax(self.per_input_weights)
 290     return tf.squeeze(
 291         tf.linalg.matmul(tf.stack(inputs, axis=-1), normalized_weights),
 292         axis=[-1])
 293 
 294   def _combine_fast_attention(self, inputs):
 295     weights_non_neg = tf.nn.relu(self.per_input_weights)
 296     normalizer = tf.reduce_sum(weights_non_neg) + 0.0001
 297     normalized_weights = weights_non_neg / normalizer
 298     return tf.squeeze(
 299         tf.linalg.matmul(tf.stack(inputs, axis=-1), normalized_weights),
 300         axis=[-1])
 301 
 302   def build(self, input_shape):
 303     if not isinstance(input_shape, list):
 304       raise ValueError('A BiFPN combine layer should be called '
 305                        'on a list of inputs.')
 306     if len(input_shape) < 2:
 307       raise ValueError('A BiFPN combine layer should be called '
 308                        'on a list of at least 2 inputs. '
 309                        'Got ' + str(len(input_shape)) + ' inputs.')
 310     if self.combine_method == 'sum':
 311       self._combine_op = tf.keras.layers.Add()
 312     elif self.combine_method == 'weighted_sum':
 313       self._combine_op = self._combine_weighted_sum
 314     elif self.combine_method == 'attention':
 315       self._combine_op = self._combine_attention
 316     elif self.combine_method == 'fast_attention':
 317       self._combine_op = self._combine_fast_attention
 318     else:
 319       raise ValueError('Unknown combine type: {}'.format(self.combine_method))
 320     if self.combine_method in {'weighted_sum', 'attention', 'fast_attention'}:
 321       self.per_input_weights = self.add_weight(
 322           name='bifpn_combine_weights',
 323           shape=(len(input_shape), 1),
 324           initializer='ones',
 325           trainable=True)
 326     super(BiFPNCombineLayer, self).build(input_shape)
 327 
 328   def call(self, inputs):
 329     """Combines multiple input feature maps into a single output feature map.
 330 
 331     Executed when calling the `.__call__` method on input.
 332 
 333     Args:
 334       inputs: A list of tensors where all tensors have the same shape, [batch,
 335         height_i, width_i, depth_i].
 336 
 337     Returns:
 338       A single tensor, with the same shape as the input tensors,
 339         [batch, height_i, width_i, depth_i].
 340     """
 341     return self._combine_op(inputs)
 342 
 343   def compute_output_shape(self, input_shape):
 344     output_shape = input_shape[0]
 345     for i in range(1, len(input_shape)):
 346       if input_shape[i] != output_shape:
 347         raise ValueError(
 348             'Inputs could not be combined. Shapes should match, '
 349             'but input_shape[0] is {} while input_shape[{}] is {}'.format(
 350                 output_shape, i, input_shape[i]))
