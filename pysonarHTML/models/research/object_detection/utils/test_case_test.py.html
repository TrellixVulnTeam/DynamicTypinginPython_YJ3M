  15 """Tests for google3.third_party.tensorflow_models.object_detection.utils.test_case."""
  16 
  17 import numpy as np
  18 import tensorflow.compat.v1 as tf
  19 from object_detection.utils import test_case
  20 
  21 
  22 class TestCaseTest(test_case.TestCase):
  23 
  24   def test_simple(self):
  25     def graph_fn(tensora, tensorb):
  26       return tf.tensordot(tensora, tensorb, axes=1)
  27 
  28     tensora_np = np.ones(20)
  29     tensorb_np = tensora_np * 2
  30     output = self.execute(graph_fn, [tensora_np, tensorb_np])
  31     self.assertAllClose(output, 40.0)
  32 
  33   def test_two_outputs(self):
  34     def graph_fn(tensora, tensorb):
  35       return tensora + tensorb, tensora - tensorb
  36     tensora_np = np.ones(20)
  37     tensorb_np = tensora_np * 2
  38     output = self.execute(graph_fn, [tensora_np, tensorb_np])
  39     self.assertAllClose(output[0], tensora_np + tensorb_np)
  40     self.assertAllClose(output[1], tensora_np - tensorb_np)
  41 
  42   def test_function_with_tf_assert(self):
  43     def compute_fn(image):
  44       return tf.image.pad_to_bounding_box(image, 0, 0, 40, 40)
  45 
  46     image_np = np.random.rand(2, 20, 30, 3)
  47     output = self.execute(compute_fn, [image_np])
  48     self.assertAllEqual(output.shape, [2, 40, 40, 3])
  49 
  50   def test_tf2_only_test(self):
  51     """Set up tests only to run with TF2."""
  52     if self.is_tf2():
  53       def graph_fn(tensora, tensorb):
  54         return tensora + tensorb, tensora - tensorb
  55       tensora_np = np.ones(20)
  56       tensorb_np = tensora_np * 2
  57       output = self.execute_tf2(graph_fn, [tensora_np, tensorb_np])
  58       self.assertAllClose(output[0], tensora_np + tensorb_np)
  59       self.assertAllClose(output[1], tensora_np - tensorb_np)
  60 
  61   def test_tpu_only_test(self):
  62     """Set up tests only to run with TPU."""
  63     if self.has_tpu():
  64       def graph_fn(tensora, tensorb):
  65         return tensora + tensorb, tensora - tensorb
  66       tensora_np = np.ones(20)
  67       tensorb_np = tensora_np * 2
  68       output = self.execute_tpu(graph_fn, [tensora_np, tensorb_np])
  69       self.assertAllClose(output[0], tensora_np + tensorb_np)
  70       self.assertAllClose(output[1], tensora_np - tensorb_np)

  72 if __name__ == '__main__':
