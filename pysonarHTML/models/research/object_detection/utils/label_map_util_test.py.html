  33   def _generate_label_map(self, num_classes):
  34     label_map_proto = string_int_label_map_pb2.StringIntLabelMap()
  35     for i in range(1, num_classes + 1):
  36       item = label_map_proto.item.add()
  37       item.id = i
  38       item.name = 'label_' + str(i)
  39       item.display_name = str(i)
  40     return label_map_proto
  41 
  42   def _generate_label_map_with_hierarchy(self, num_classes, ancestors_dict,
  43                                          descendants_dict):
  44     label_map_proto = string_int_label_map_pb2.StringIntLabelMap()
  45     for i in range(1, num_classes + 1):
  46       item = label_map_proto.item.add()
  47       item.id = i
  48       item.name = 'label_' + str(i)
  49       item.display_name = str(i)
  50       if i in ancestors_dict:
  51         for anc_i in ancestors_dict[i]:
  52           item.ancestor_ids.append(anc_i)
  53       if i in descendants_dict:
  54         for desc_i in descendants_dict[i]:
  55           item.descendant_ids.append(desc_i)
  56     return label_map_proto

  58   def test_get_label_map_dict(self):
  59     label_map_string = """
  60       item {
  61         id:2
  62         name:'cat'
  63       }
  64       item {
  65         id:1
  66         name:'dog'
  67       }
  68     """
  69     label_map_path = os.path.join(self.get_temp_dir(), 'label_map.pbtxt')
  70     with tf.gfile.Open(label_map_path, 'wb') as f:
  71       f.write(label_map_string)
  72 
  73     label_map_dict = label_map_util.get_label_map_dict(label_map_path)
  74     self.assertEqual(label_map_dict['dog'], 1)
  75     self.assertEqual(label_map_dict['cat'], 2)

 184   def test_keep_categories_with_unique_id(self):
 185     label_map_proto = string_int_label_map_pb2.StringIntLabelMap()
 186     label_map_string = """
 187       item {
 188         id:2
 189         name:'cat'
 190       }
 191       item {
 192         id:1
 193         name:'child'
 194       }
 195       item {
 196         id:1
 197         name:'person'
 198       }
 199       item {
 200         id:1
 201         name:'n00007846'
 202       }
 203     """
 204     text_format.Merge(label_map_string, label_map_proto)
 205     categories = label_map_util.convert_label_map_to_categories(
 206         label_map_proto, max_num_classes=3)
 207     self.assertListEqual([{
 208         'id': 2,
 209         'name': u'cat'
 210     }, {
 211         'id': 1,
 212         'name': u'child'
 213     }], categories)

 424   def test_get_label_map_hierarchy_lut(self):
 425     num_classes = 5
 426     ancestors = {2: [1, 3], 5: [1]}
 427     descendants = {1: [2], 5: [1, 2]}
 428     label_map = self._generate_label_map_with_hierarchy(num_classes, ancestors,
 429                                                         descendants)
 430     gt_hierarchy_dict_lut = {
 431         'ancestors':
 432             np.array([
 433                 [1, 0, 0, 0, 0],
 434                 [1, 1, 1, 0, 0],
 435                 [0, 0, 1, 0, 0],
 436                 [0, 0, 0, 1, 0],
 437                 [1, 0, 0, 0, 1],
 438             ]),
 439         'descendants':
 440             np.array([
 441                 [1, 1, 0, 0, 0],
 442                 [0, 1, 0, 0, 0],
 443                 [0, 0, 1, 0, 0],
 444                 [0, 0, 0, 1, 0],
 445                 [1, 1, 0, 0, 1],
 446             ]),
 447     }
 448     ancestors_lut, descendants_lut = (
 449         label_map_util.get_label_map_hierarchy_lut(label_map, True))
 450     np.testing.assert_array_equal(gt_hierarchy_dict_lut['ancestors'],
 451                                   ancestors_lut)
 452     np.testing.assert_array_equal(gt_hierarchy_dict_lut['descendants'],
 453                                   descendants_lut)
