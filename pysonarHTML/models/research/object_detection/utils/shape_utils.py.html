  31 def _is_tensor(t):
  32   """Returns a boolean indicating whether the input is a tensor.
  33 
  34   Args:
  35     t: the input to be tested.
  36 
  37   Returns:
  38     a boolean that indicates whether t is a tensor.
  39   """
  40   return isinstance(t, (tf.Tensor, tf.SparseTensor, tf.Variable))

  43 def _set_dim_0(t, d0):
  44   """Sets the 0-th dimension of the input tensor.
  45 
  46   Args:
  47     t: the input tensor, assuming the rank is at least 1.
  48     d0: an integer indicating the 0-th dimension of the input tensor.
  49 
  50   Returns:
  51     the tensor t with the 0-th dimension set.
  52   """
  53   t_shape = t.get_shape().as_list()
  54   t_shape[0] = d0
  55   t.set_shape(t_shape)
  56   return t

  59 def pad_tensor(t, length):
  60   """Pads the input tensor with 0s along the first dimension up to the length.
  61 
  62   Args:
  63     t: the input tensor, assuming the rank is at least 1.
  64     length: a tensor of shape [1]  or an integer, indicating the first dimension
  65       of the input tensor t after padding, assuming length <= t.shape[0].
  66 
  67   Returns:
  68     padded_t: the padded tensor, whose first dimension is length. If the length
  69       is an integer, the first dimension of padded_t is set to length
  70       statically.
  71   """
  72 
  73   # Computing the padding statically makes the operation work with XLA.
  74   rank = len(t.get_shape())
  75   paddings = [[0 for _ in range(2)] for _ in range(rank)]
  76   t_d0 = tf.shape(t)[0]
  77 
  78   if isinstance(length, int) or len(length.get_shape()) == 0:  # pylint:disable=g-explicit-length-test
  79     paddings[0][1] = length - t_d0
  80   else:
  81     paddings[0][1] = length[0] - t_d0
  82 
  83   return tf.pad(t, paddings)

  86 def clip_tensor(t, length):
  87   """Clips the input tensor along the first dimension up to the length.
  88 
  89   Args:
  90     t: the input tensor, assuming the rank is at least 1.
  91     length: a tensor of shape [1]  or an integer, indicating the first dimension
  92       of the input tensor t after clipping, assuming length <= t.shape[0].
  93 
  94   Returns:
  95     clipped_t: the clipped tensor, whose first dimension is length. If the
  96       length is an integer, the first dimension of clipped_t is set to length
  97       statically.
  98   """
  99   clipped_t = tf.gather(t, tf.range(length))
 100   if not _is_tensor(length):
 101     clipped_t = _set_dim_0(clipped_t, length)
 102   return clipped_t

 105 def pad_or_clip_tensor(t, length):
 106   """Pad or clip the input tensor along the first dimension.
 107 
 108   Args:
 109     t: the input tensor, assuming the rank is at least 1.
 110     length: a tensor of shape [1]  or an integer, indicating the first dimension
 111       of the input tensor t after processing.
 112 
 113   Returns:
 114     processed_t: the processed tensor, whose first dimension is length. If the
 115       length is an integer, the first dimension of the processed tensor is set
 116       to length statically.
 117   """
 118   return pad_or_clip_nd(t, [length] + t.shape.as_list()[1:])

 121 def pad_or_clip_nd(tensor, output_shape):
 122   """Pad or Clip given tensor to the output shape.
 123 
 124   Args:
 125     tensor: Input tensor to pad or clip.
 126     output_shape: A list of integers / scalar tensors (or None for dynamic dim)
 127       representing the size to pad or clip each dimension of the input tensor.
 128 
 129   Returns:
 130     Input tensor padded and clipped to the output shape.
 131   """
 132   tensor_shape = tf.shape(tensor)
 133   clip_size = [
 134       tf.where(tensor_shape[i] - shape > 0, shape, -1)
 135       if shape is not None else -1 for i, shape in enumerate(output_shape)
 136   ]
 137   clipped_tensor = tf.slice(
 138       tensor,
 139       begin=tf.zeros(len(clip_size), dtype=tf.int32),
 140       size=clip_size)
 141 
 142   # Pad tensor if the shape of clipped tensor is smaller than the expected
 143   # shape.
 144   clipped_tensor_shape = tf.shape(clipped_tensor)
 145   trailing_paddings = [
 146       shape - clipped_tensor_shape[i] if shape is not None else 0
 147       for i, shape in enumerate(output_shape)
 148   ]
 149   paddings = tf.stack(
 150       [
 151           tf.zeros(len(trailing_paddings), dtype=tf.int32),
 152           trailing_paddings
 153       ],
 154       axis=1)
 155   padded_tensor = tf.pad(clipped_tensor, paddings=paddings)
 156   output_static_shape = [
 157       dim if not isinstance(dim, tf.Tensor) else None for dim in output_shape
 158   ]
 159   padded_tensor.set_shape(output_static_shape)
 160   return padded_tensor

 163 def combined_static_and_dynamic_shape(tensor):
 164   """Returns a list containing static and dynamic values for the dimensions.
 165 
 166   Returns a list of static and dynamic values for shape dimensions. This is
 167   useful to preserve static shapes when available in reshape operation.
 168 
 169   Args:
 170     tensor: A tensor of any type.
 171 
 172   Returns:
 173     A list of size tensor.shape.ndims containing integers or a scalar tensor.
 174   """
 175   static_tensor_shape = tensor.shape.as_list()
 176   dynamic_tensor_shape = tf.shape(tensor)
 177   combined_shape = []
 178   for index, dim in enumerate(static_tensor_shape):
 179     if dim is not None:
 180       combined_shape.append(dim)
 181     else:
 182       combined_shape.append(dynamic_tensor_shape[index])
 183   return combined_shape

 186 def static_or_dynamic_map_fn(fn, elems, dtype=None,
 187                              parallel_iterations=32, back_prop=True):
 188   """Runs map_fn as a (static) for loop when possible.
 189 
 190   This function rewrites the map_fn as an explicit unstack input -> for loop
 191   over function calls -> stack result combination.  This allows our graphs to
 192   be acyclic when the batch size is static.
 193   For comparison, see https://www.tensorflow.org/api_docs/python/tf/map_fn.
 194 
 195   Note that `static_or_dynamic_map_fn` currently is not *fully* interchangeable
 196   with the default tf.map_fn function as it does not accept nested inputs (only
 197   Tensors or lists of Tensors).  Likewise, the output of `fn` can only be a
 198   Tensor or list of Tensors.
 199 
 200   TODO(jonathanhuang): make this function fully interchangeable with tf.map_fn.
 201 
 202   Args:
 203     fn: The callable to be performed. It accepts one argument, which will have
 204       the same structure as elems. Its output must have the
 205       same structure as elems.
 206     elems: A tensor or list of tensors, each of which will
 207       be unpacked along their first dimension. The sequence of the
 208       resulting slices will be applied to fn.
 209     dtype:  (optional) The output type(s) of fn. If fn returns a structure of
 210       Tensors differing from the structure of elems, then dtype is not optional
 211       and must have the same structure as the output of fn.
 212     parallel_iterations: (optional) number of batch items to process in
 213       parallel.  This flag is only used if the native tf.map_fn is used
 214       and defaults to 32 instead of 10 (unlike the standard tf.map_fn default).
 215     back_prop: (optional) True enables support for back propagation.
 216       This flag is only used if the native tf.map_fn is used.
 217 
 218   Returns:
 219     A tensor or sequence of tensors. Each tensor packs the
 220     results of applying fn to tensors unpacked from elems along the first
 221     dimension, from first to last.
 222   Raises:
 223     ValueError: if `elems` a Tensor or a list of Tensors.
 224     ValueError: if `fn` does not return a Tensor or list of Tensors
 225   """
 226   if isinstance(elems, list):
 227     for elem in elems:
 228       if not isinstance(elem, tf.Tensor):
 229         raise ValueError('`elems` must be a Tensor or list of Tensors.')
 230 
 231     elem_shapes = [elem.shape.as_list() for elem in elems]
 232     # Fall back on tf.map_fn if shapes of each entry of `elems` are None or fail
 233     # to all be the same size along the batch dimension.
 234     for elem_shape in elem_shapes:
 235       if (not elem_shape or not elem_shape[0]
 236           or elem_shape[0] != elem_shapes[0][0]):
 237         return tf.map_fn(fn, elems, dtype, parallel_iterations, back_prop)
 238     arg_tuples = zip(*[tf.unstack(elem) for elem in elems])
 239     outputs = [fn(arg_tuple) for arg_tuple in arg_tuples]
 240   else:
 241     if not isinstance(elems, tf.Tensor):
 242       raise ValueError('`elems` must be a Tensor or list of Tensors.')
 243     elems_shape = elems.shape.as_list()
 244     if not elems_shape or not elems_shape[0]:
 245       return tf.map_fn(fn, elems, dtype, parallel_iterations, back_prop)
 246     outputs = [fn(arg) for arg in tf.unstack(elems)]
 247   # Stack `outputs`, which is a list of Tensors or list of lists of Tensors
 248   if all([isinstance(output, tf.Tensor) for output in outputs]):
 249     return tf.stack(outputs)
 250   else:
 251     if all([isinstance(output, list) for output in outputs]):
 252       if all([all(
 253           [isinstance(entry, tf.Tensor) for entry in output_list])
 254               for output_list in outputs]):
 255         return [tf.stack(output_tuple) for output_tuple in zip(*outputs)]
 256   raise ValueError('`fn` should return a Tensor or a list of Tensors.')

 259 def check_min_image_dim(min_dim, image_tensor):
 260   """Checks that the image width/height are greater than some number.
 261 
 262   This function is used to check that the width and height of an image are above
 263   a certain value. If the image shape is static, this function will perform the
 264   check at graph construction time. Otherwise, if the image shape varies, an
 265   Assertion control dependency will be added to the graph.
 266 
 267   Args:
 268     min_dim: The minimum number of pixels along the width and height of the
 269              image.
 270     image_tensor: The image tensor to check size for.
 271 
 272   Returns:
 273     If `image_tensor` has dynamic size, return `image_tensor` with a Assert
 274     control dependency. Otherwise returns image_tensor.
 275 
 276   Raises:
 277     ValueError: if `image_tensor`'s' width or height is smaller than `min_dim`.
 278   """
 279   image_shape = image_tensor.get_shape()
 280   image_height = static_shape.get_height(image_shape)
 281   image_width = static_shape.get_width(image_shape)
 282   if image_height is None or image_width is None:
 283     shape_assert = tf.Assert(
 284         tf.logical_and(tf.greater_equal(tf.shape(image_tensor)[1], min_dim),
 285                        tf.greater_equal(tf.shape(image_tensor)[2], min_dim)),
 286         ['image size must be >= {} in both height and width.'.format(min_dim)])
 287     with tf.control_dependencies([shape_assert]):
 288       return tf.identity(image_tensor)
 289 
 290   if image_height < min_dim or image_width < min_dim:
 291     raise ValueError(
 292         'image size must be >= %d in both height and width; image dim = %d,%d' %
 293         (min_dim, image_height, image_width))
 294 
 295   return image_tensor

 298 def assert_shape_equal(shape_a, shape_b):
 299   """Asserts that shape_a and shape_b are equal.
 300 
 301   If the shapes are static, raises a ValueError when the shapes
 302   mismatch.
 303 
 304   If the shapes are dynamic, raises a tf InvalidArgumentError when the shapes
 305   mismatch.
 306 
 307   Args:
 308     shape_a: a list containing shape of the first tensor.
 309     shape_b: a list containing shape of the second tensor.
 310 
 311   Returns:
 312     Either a tf.no_op() when shapes are all static and a tf.assert_equal() op
 313     when the shapes are dynamic.
 314 
 315   Raises:
 316     ValueError: When shapes are both static and unequal.
 317   """
 318   if (all(isinstance(dim, int) for dim in shape_a) and
 319       all(isinstance(dim, int) for dim in shape_b)):
 320     if shape_a != shape_b:
 321       raise ValueError('Unequal shapes {}, {}'.format(shape_a, shape_b))
 322     else: return tf.no_op()
 323   else:
 324     return tf.assert_equal(shape_a, shape_b)
 325

 327 def assert_shape_equal_along_first_dimension(shape_a, shape_b):
 328   """Asserts that shape_a and shape_b are the same along the 0th-dimension.
 329 
 330   If the shapes are static, raises a ValueError when the shapes
 331   mismatch.
 332 
 333   If the shapes are dynamic, raises a tf InvalidArgumentError when the shapes
 334   mismatch.
 335 
 336   Args:
 337     shape_a: a list containing shape of the first tensor.
 338     shape_b: a list containing shape of the second tensor.
 339 
 340   Returns:
 341     Either a tf.no_op() when shapes are all static and a tf.assert_equal() op
 342     when the shapes are dynamic.
 343 
 344   Raises:
 345     ValueError: When shapes are both static and unequal.
 346   """
 347   if isinstance(shape_a[0], int) and isinstance(shape_b[0], int):
 348     if shape_a[0] != shape_b[0]:
 349       raise ValueError('Unequal first dimension {}, {}'.format(
 350           shape_a[0], shape_b[0]))
 351     else: return tf.no_op()
 352   else:
 353     return tf.assert_equal(shape_a[0], shape_b[0])

 356 def assert_box_normalized(boxes, maximum_normalized_coordinate=1.1):
 357   """Asserts the input box tensor is normalized.
 358 
 359   Args:
 360     boxes: a tensor of shape [N, 4] where N is the number of boxes.
 361     maximum_normalized_coordinate: Maximum coordinate value to be considered
 362       as normalized, default to 1.1.
 363 
 364   Returns:
 365     a tf.Assert op which fails when the input box tensor is not normalized.
 366 
 367   Raises:
 368     ValueError: When the input box tensor is not normalized.
 369   """
 370   box_minimum = tf.reduce_min(boxes)
 371   box_maximum = tf.reduce_max(boxes)
 372   return tf.Assert(
 373       tf.logical_and(
 374           tf.less_equal(box_maximum, maximum_normalized_coordinate),
 375           tf.greater_equal(box_minimum, 0)),
 376       [boxes])

 379 def flatten_dimensions(inputs, first, last):
 380   """Flattens `K-d` tensor along [first, last) dimensions.
 381 
 382   Converts `inputs` with shape [D0, D1, ..., D(K-1)] into a tensor of shape
 383   [D0, D1, ..., D(first) * D(first+1) * ... * D(last-1), D(last), ..., D(K-1)].
 384 
 385   Example:
 386   `inputs` is a tensor with initial shape [10, 5, 20, 20, 3].
 387   new_tensor = flatten_dimensions(inputs, first=1, last=3)
 388   new_tensor.shape -> [10, 100, 20, 3].
 389 
 390   Args:
 391     inputs: a tensor with shape [D0, D1, ..., D(K-1)].
 392     first: first value for the range of dimensions to flatten.
 393     last: last value for the range of dimensions to flatten. Note that the last
 394       dimension itself is excluded.
 395 
 396   Returns:
 397     a tensor with shape
 398     [D0, D1, ..., D(first) * D(first + 1) * ... * D(last - 1), D(last), ...,
 399      D(K-1)].
 400 
 401   Raises:
 402     ValueError: if first and last arguments are incorrect.
 403   """
 404   if first >= inputs.shape.ndims or last > inputs.shape.ndims:
 405     raise ValueError('`first` and `last` must be less than inputs.shape.ndims. '
 406                      'found {} and {} respectively while ndims is {}'.format(
 407                          first, last, inputs.shape.ndims))
 408   shape = combined_static_and_dynamic_shape(inputs)
 409   flattened_dim_prod = tf.reduce_prod(shape[first:last],
 410                                       keepdims=True)
 411   new_shape = tf.concat([shape[:first], flattened_dim_prod,
 412                          shape[last:]], axis=0)
 413   return tf.reshape(inputs, new_shape)
 414

 416 def flatten_first_n_dimensions(inputs, n):
 417   """Flattens `K-d` tensor along first n dimension to be a `(K-n+1)-d` tensor.
 418 
 419   Converts `inputs` with shape [D0, D1, ..., D(K-1)] into a tensor of shape
 420   [D0 * D1 * ... * D(n-1), D(n), ... D(K-1)].
 421 
 422   Example:
 423   `inputs` is a tensor with initial shape [10, 5, 20, 20, 3].
 424   new_tensor = flatten_first_n_dimensions(inputs, 2)
 425   new_tensor.shape -> [50, 20, 20, 3].
 426 
 427   Args:
 428     inputs: a tensor with shape [D0, D1, ..., D(K-1)].
 429     n: The number of dimensions to flatten.
 430 
 431   Returns:
 432     a tensor with shape [D0 * D1 * ... * D(n-1), D(n), ... D(K-1)].
 433   """
 434   return flatten_dimensions(inputs, first=0, last=n)

 437 def expand_first_dimension(inputs, dims):
 438   """Expands `K-d` tensor along first dimension to be a `(K+n-1)-d` tensor.
 439 
 440   Converts `inputs` with shape [D0, D1, ..., D(K-1)] into a tensor of shape
 441   [dims[0], dims[1], ..., dims[-1], D1, ..., D(k-1)].
 442 
 443   Example:
 444   `inputs` is a tensor with shape [50, 20, 20, 3].
 445   new_tensor = expand_first_dimension(inputs, [10, 5]).
 446   new_tensor.shape -> [10, 5, 20, 20, 3].
 447 
 448   Args:
 449     inputs: a tensor with shape [D0, D1, ..., D(K-1)].
 450     dims: List with new dimensions to expand first axis into. The length of
 451       `dims` is typically 2 or larger.
 452 
 453   Returns:
 454     a tensor with shape [dims[0], dims[1], ..., dims[-1], D1, ..., D(k-1)].
 455   """
 456   inputs_shape = combined_static_and_dynamic_shape(inputs)
 457   expanded_shape = tf.stack(dims + inputs_shape[1:])
 458 
 459   # Verify that it is possible to expand the first axis of inputs.
 460   assert_op = tf.assert_equal(
 461       inputs_shape[0], tf.reduce_prod(tf.stack(dims)),
 462       message=('First dimension of `inputs` cannot be expanded into provided '
 463                '`dims`'))
 464 
 465   with tf.control_dependencies([assert_op]):
 466     inputs_reshaped = tf.reshape(inputs, expanded_shape)
 467 
 468   return inputs_reshaped

 471 def resize_images_and_return_shapes(inputs, image_resizer_fn):
 472   """Resizes images using the given function and returns their true shapes.
 473 
 474   Args:
 475     inputs: a float32 Tensor representing a batch of inputs of shape
 476       [batch_size, height, width, channels].
 477     image_resizer_fn: a function which takes in a single image and outputs
 478       a resized image and its original shape.
 479 
 480   Returns:
 481     resized_inputs: The inputs resized according to image_resizer_fn.
 482     true_image_shapes: A integer tensor of shape [batch_size, 3]
 483       representing the height, width and number of channels in inputs.
 484   """
 485 
 486   if inputs.dtype is not tf.float32:
 487     raise ValueError('`resize_images_and_return_shapes` expects a'
 488                      ' tf.float32 tensor')
 489 
 490   # TODO(jonathanhuang): revisit whether to always use batch size as
 491   # the number of parallel iterations vs allow for dynamic batching.
 492   outputs = static_or_dynamic_map_fn(
 493       image_resizer_fn,
 494       elems=inputs,
 495       dtype=[tf.float32, tf.int32])
 496   resized_inputs = outputs[0]
 497   true_image_shapes = outputs[1]
 498 
 499   return resized_inputs, true_image_shapes
