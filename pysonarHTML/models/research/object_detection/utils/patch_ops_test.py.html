  18 from __future__ import absolute_import

  22 from absl.testing import parameterized
  23 import numpy as np
  24 import tensorflow.compat.v1 as tf

  26 from object_detection.utils import patch_ops
  27 from object_detection.utils import test_case

  30 class GetPatchMaskTest(test_case.TestCase, parameterized.TestCase):
  31 
  32   def testMaskShape(self):
  33     image_shape = [15, 10]
  34     mask = patch_ops.get_patch_mask(
  35         10, 5, patch_size=3, image_shape=image_shape)
  36     self.assertListEqual(mask.shape.as_list(), image_shape)
  37 
  38   def testHandleImageShapeWithChannels(self):
  39     image_shape = [15, 10, 3]
  40     mask = patch_ops.get_patch_mask(
  41         10, 5, patch_size=3, image_shape=image_shape)
  42     self.assertListEqual(mask.shape.as_list(), image_shape[:2])
  43 
  44   def testMaskDType(self):
  45     mask = patch_ops.get_patch_mask(2, 3, patch_size=2, image_shape=[6, 7])
  46     self.assertDTypeEqual(mask, bool)
  47 
  48   def testMaskAreaWithEvenPatchSize(self):
  49     image_shape = [6, 7]
  50     mask = patch_ops.get_patch_mask(2, 3, patch_size=2, image_shape=image_shape)
  51     expected_mask = np.array([
  52         [0, 0, 0, 0, 0, 0, 0],
  53         [0, 0, 1, 1, 0, 0, 0],
  54         [0, 0, 1, 1, 0, 0, 0],
  55         [0, 0, 0, 0, 0, 0, 0],
  56         [0, 0, 0, 0, 0, 0, 0],
  57         [0, 0, 0, 0, 0, 0, 0],
  58     ]).reshape(image_shape).astype(bool)
  59     self.assertAllEqual(mask, expected_mask)
  60 
  61   def testMaskAreaWithEvenPatchSize4(self):
  62     image_shape = [6, 7]
  63     mask = patch_ops.get_patch_mask(2, 3, patch_size=4, image_shape=image_shape)
  64     expected_mask = np.array([
  65         [0, 1, 1, 1, 1, 0, 0],
  66         [0, 1, 1, 1, 1, 0, 0],
  67         [0, 1, 1, 1, 1, 0, 0],
  68         [0, 1, 1, 1, 1, 0, 0],
  69         [0, 0, 0, 0, 0, 0, 0],
  70         [0, 0, 0, 0, 0, 0, 0],
  71     ]).reshape(image_shape).astype(bool)
  72     self.assertAllEqual(mask, expected_mask)
  73 
  74   def testMaskAreaWithOddPatchSize(self):
  75     image_shape = [6, 7]
  76     mask = patch_ops.get_patch_mask(2, 3, patch_size=3, image_shape=image_shape)
  77     expected_mask = np.array([
  78         [0, 0, 0, 0, 0, 0, 0],
  79         [0, 0, 1, 1, 1, 0, 0],
  80         [0, 0, 1, 1, 1, 0, 0],
  81         [0, 0, 1, 1, 1, 0, 0],
  82         [0, 0, 0, 0, 0, 0, 0],
  83         [0, 0, 0, 0, 0, 0, 0],
  84     ]).reshape(image_shape).astype(bool)
  85     self.assertAllEqual(mask, expected_mask)
  86 
  87   def testMaskAreaPartiallyOutsideImage(self):
  88     image_shape = [6, 7]
  89     mask = patch_ops.get_patch_mask(5, 6, patch_size=5, image_shape=image_shape)
  90     expected_mask = np.array([
  91         [0, 0, 0, 0, 0, 0, 0],
  92         [0, 0, 0, 0, 0, 0, 0],
  93         [0, 0, 0, 0, 0, 0, 0],
  94         [0, 0, 0, 0, 1, 1, 1],
  95         [0, 0, 0, 0, 1, 1, 1],
  96         [0, 0, 0, 0, 1, 1, 1],
  97     ]).reshape(image_shape).astype(bool)
  98     self.assertAllEqual(mask, expected_mask)

 100   @parameterized.parameters(
 101       {'y': 0, 'x': -1},
 102       {'y': -1, 'x': 0},
 103       {'y': 0, 'x': 11},
 104       {'y': 16, 'x': 0},
 105   )
 106   def testStaticCoordinatesOutsideImageRaisesError(self, y, x):
 107     image_shape = [15, 10]
 108     with self.assertRaises(tf.errors.InvalidArgumentError):
 109       patch_ops.get_patch_mask(y, x, patch_size=3, image_shape=image_shape)
 110 
 111   def testDynamicCoordinatesOutsideImageRaisesError(self):
 112 
 113     def graph_fn():
 114       image_shape = [15, 10]
 115       x = tf.random_uniform([], minval=-2, maxval=-1, dtype=tf.int32)
 116       y = tf.random_uniform([], minval=0, maxval=1, dtype=tf.int32)
 117       mask = patch_ops.get_patch_mask(
 118           y, x, patch_size=3, image_shape=image_shape)
 119       return mask
 120 
 121     with self.assertRaises(tf.errors.InvalidArgumentError):
 122       self.execute(graph_fn, [])
 123 
 124   @parameterized.parameters(
 125       {'patch_size': 0},
 126       {'patch_size': -1},
 127   )
 128   def testStaticNonPositivePatchSizeRaisesError(self, patch_size):
 129     image_shape = [6, 7]
 130     with self.assertRaises(tf.errors.InvalidArgumentError):
 131       patch_ops.get_patch_mask(
 132           0, 0, patch_size=patch_size, image_shape=image_shape)
 133 
 134   def testDynamicNonPositivePatchSizeRaisesError(self):
 135 
 136     def graph_fn():
 137       image_shape = [6, 7]
 138       patch_size = -1 * tf.random_uniform([], minval=0, maxval=3,
 139                                           dtype=tf.int32)
 140       mask = patch_ops.get_patch_mask(
 141           0, 0, patch_size=patch_size, image_shape=image_shape)
 142       return mask
 143 
 144     with self.assertRaises(tf.errors.InvalidArgumentError):
 145       self.execute(graph_fn, [])
