  37 def get_iterator_next_for_testing(dataset, is_tf2):
  38 
  39   # In TF2, lookup tables are not supported in one shot iterators, but
  40   # initialization is implicit.
  41   if is_tf2:
  42     return dataset.make_initializable_iterator().get_next()
  43   # In TF1, we use one shot iterator because it does not require running
  44   # a separate init op.
  45   else:
  46     return dataset.make_one_shot_iterator().get_next()

  51   def test_get_prime_multiplier_for_color_randomness(self):
  52     # Show that default multipler is not 1 and does not divide the total number
  53     # of standard colors.
  54     multiplier = visualization_utils._get_multiplier_for_color_randomness()
  55     self.assertNotEqual(
  56         0, multiplier % len(visualization_utils.STANDARD_COLORS))
  57     self.assertNotEqual(1, multiplier)
  58 
  59     # Show that with 34 colors, the closest prime number to 34/10 that
  60     # satisfies the constraints is 5.
  61     default_standard_colors = visualization_utils.STANDARD_COLORS
  62     visualization_utils.STANDARD_COLORS = [
  63         'color_{}'.format(str(i)) for i in range(34)
  64     ]
  65     multiplier = visualization_utils._get_multiplier_for_color_randomness()
  66     self.assertEqual(5, multiplier)
  67 
  68     # Show that with 110 colors, the closest prime number to 110/10 that
  69     # satisfies the constraints is 13 (since 11 equally divides 110).
  70     visualization_utils.STANDARD_COLORS = [
  71         'color_{}'.format(str(i)) for i in range(110)
  72     ]
  73     multiplier = visualization_utils._get_multiplier_for_color_randomness()
  74     self.assertEqual(13, multiplier)
  75 
  76     visualization_utils.STANDARD_COLORS = default_standard_colors

  78   def create_colorful_test_image(self):
  79     """This function creates an image that can be used to test vis functions.
  80 
  81     It makes an image composed of four colored rectangles.
  82 
  83     Returns:
  84       colorful test numpy array image.
  85     """
  86     ch255 = np.full([100, 200, 1], 255, dtype=np.uint8)
  87     ch128 = np.full([100, 200, 1], 128, dtype=np.uint8)
  88     ch0 = np.full([100, 200, 1], 0, dtype=np.uint8)
  89     imr = np.concatenate((ch255, ch128, ch128), axis=2)
  90     img = np.concatenate((ch255, ch255, ch0), axis=2)
  91     imb = np.concatenate((ch255, ch0, ch255), axis=2)
  92     imw = np.concatenate((ch128, ch128, ch128), axis=2)
  93     imu = np.concatenate((imr, img), axis=1)
  94     imd = np.concatenate((imb, imw), axis=1)
  95     image = np.concatenate((imu, imd), axis=0)
  96     return image

 164   def test_draw_bounding_boxes_on_image_tensors(self):
 165     """Tests that bounding box utility produces reasonable results."""
 166     category_index = {1: {'id': 1, 'name': 'dog'}, 2: {'id': 2, 'name': 'cat'}}
 167     fname = os.path.join(_TESTDATA_PATH, 'image1.jpg')
 168     image_np = np.array(Image.open(fname))
 169     images_np = np.stack((image_np, image_np), axis=0)
 170     original_image_shape = [[636, 512], [636, 512]]
 171 
 172     def graph_fn():
 173       images_tensor = tf.constant(value=images_np, dtype=tf.uint8)
 174       image_shape = tf.constant(original_image_shape, dtype=tf.int32)
 175       boxes = tf.constant([[[0.4, 0.25, 0.75, 0.75], [0.5, 0.3, 0.6, 0.9]],
 176                            [[0.25, 0.25, 0.75, 0.75], [0.1, 0.3, 0.6, 1.0]]])
 177       classes = tf.constant([[1, 1], [1, 2]], dtype=tf.int64)
 178       scores = tf.constant([[0.8, 0.1], [0.6, 0.5]])
 179       keypoints = tf.random.uniform((2, 2, 4, 2), maxval=1.0, dtype=tf.float32)
 180       keypoint_edges = [(0, 1), (1, 2), (2, 3), (3, 0)]
 181       images_with_boxes = (
 182           visualization_utils.draw_bounding_boxes_on_image_tensors(
 183               images_tensor,
 184               boxes,
 185               classes,
 186               scores,
 187               category_index,
 188               original_image_spatial_shape=image_shape,
 189               true_image_shape=image_shape,
 190               keypoints=keypoints,
 191               min_score_thresh=0.2,
 192               keypoint_edges=keypoint_edges))
 193       return images_with_boxes
 194 
 195     # Write output images for visualization.
 196     images_with_boxes_np = self.execute(graph_fn, [])
 197     self.assertEqual(images_np.shape[0], images_with_boxes_np.shape[0])
 198     self.assertEqual(images_np.shape[3], images_with_boxes_np.shape[3])
 199     self.assertEqual(
 200         tuple(original_image_shape[0]), images_with_boxes_np.shape[1:3])
 201     for i in range(images_with_boxes_np.shape[0]):
 202       img_name = 'image_' + str(i) + '.png'
 203       output_file = os.path.join(self.get_temp_dir(), img_name)
 204       logging.info('Writing output image %d to %s', i, output_file)
 205       image_pil = Image.fromarray(images_with_boxes_np[i, ...])
 206       image_pil.save(output_file)

 408     test_image = np.asarray([[[0, 0, 0], [0, 0, 0]],
 409                              [[0, 0, 0], [0, 0, 0]]], dtype=np.uint8)

 454   def test_eval_metric_ops(self):
 455     if self.is_tf2():
 456       self.skipTest('This test is only compatible with Tensorflow 1.X, '
 457                     'estimator eval ops are not supported in Tensorflow 2.')
 458     category_index = {1: {'id': 1, 'name': 'dog'}, 2: {'id': 2, 'name': 'cat'}}
 459     max_examples_to_draw = 4
 460     metric_op_base = 'Detections_Left_Groundtruth_Right'
 461     eval_metric_ops = visualization_utils.VisualizeSingleFrameDetections(
 462         category_index,
 463         max_examples_to_draw=max_examples_to_draw,
 464         summary_name_prefix=metric_op_base)
 465     original_image = tf.placeholder(tf.uint8, [4, None, None, 3])
 466     original_image_spatial_shape = tf.placeholder(tf.int32, [4, 2])
 467     true_image_shape = tf.placeholder(tf.int32, [4, 3])
 468     detection_boxes = tf.random_uniform([4, 20, 4],
 469                                         minval=0.0,
 470                                         maxval=1.0,
 471                                         dtype=tf.float32)
 472     detection_classes = tf.random_uniform([4, 20],
 473                                           minval=1,
 474                                           maxval=3,
 475                                           dtype=tf.int64)
 476     detection_scores = tf.random_uniform([4, 20],
 477                                          minval=0.,
 478                                          maxval=1.,
 479                                          dtype=tf.float32)
 480     groundtruth_boxes = tf.random_uniform([4, 8, 4],
 481                                           minval=0.0,
