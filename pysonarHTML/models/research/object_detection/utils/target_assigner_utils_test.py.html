  39   def test_coordinates_to_heatmap(self):
  40     def graph_fn():
  41       (y_grid, x_grid) = ta_utils.image_shape_to_grids(height=3, width=5)
  42       y_coordinates = tf.constant([1.5, 0.5], dtype=tf.float32)
  43       x_coordinates = tf.constant([2.5, 4.5], dtype=tf.float32)
  44       sigma = tf.constant([0.1, 0.5], dtype=tf.float32)
  45       channel_onehot = tf.constant([[1, 0, 0], [0, 1, 0]], dtype=tf.float32)
  46       channel_weights = tf.constant([1, 1], dtype=tf.float32)
  47       heatmap = ta_utils.coordinates_to_heatmap(y_grid, x_grid, y_coordinates,
  48                                                 x_coordinates, sigma,
  49                                                 channel_onehot, channel_weights)
  50       return heatmap
  51 
  52     heatmap = self.execute(graph_fn, [])
  53     # Peak at (1, 2) for the first class.
  54     self.assertAlmostEqual(1.0, heatmap[1, 2, 0])
  55     # Peak at (0, 4) for the second class.
  56     self.assertAlmostEqual(1.0, heatmap[0, 4, 1])

  93   def test_get_valid_keypoints_mask(self):
  94 
  95     def graph_fn():
  96       class_onehot = tf.constant(
  97           [[0, 0, 1, 0, 0],
  98            [0, 1, 0, 0, 0],
  99            [0, 0, 1, 0, 1]], dtype=tf.float32)
 100       keypoints = tf.constant(
 101           [[0.1, float('nan'), 0.2, 0.0],
 102            [0.0, 0.0, 0.1, 0.9],
 103            [3.2, 4.3, float('nan'), 0.2]],
 104           dtype=tf.float32)
 105       keypoint_coordinates = tf.stack([keypoints, keypoints], axis=2)
 106       mask, keypoints_nan_to_zeros = ta_utils.get_valid_keypoint_mask_for_class(
 107           keypoint_coordinates=keypoint_coordinates,
 108           class_id=2,
 109           class_onehot=class_onehot,
 110           keypoint_indices=[1, 2])
 111 
 112       return mask, keypoints_nan_to_zeros
 113 
 114     keypoints = np.array([[0.0, 0.2],
 115                           [0.0, 0.1],
 116                           [4.3, 0.0]])
 117     expected_mask = np.array([[0, 1], [0, 0], [1, 0]])
 118     expected_keypoints = np.stack([keypoints, keypoints], axis=2)
 119     mask, keypoints_nan_to_zeros = self.execute(graph_fn, [])
 120

 157   def test_blackout_pixel_weights_by_box_regions(self):
 158     def graph_fn():
 159       boxes = tf.constant(
 160           [[0.0, 0.0, 5, 5], [0.0, 0.0, 10.0, 20.0], [6.0, 12.0, 8.0, 18.0]],
 161           dtype=tf.float32)
 162       blackout = tf.constant([True, False, True], dtype=tf.bool)
 163       blackout_pixel_weights_by_box_regions = tf.function(
 164           ta_utils.blackout_pixel_weights_by_box_regions)
 165       output = blackout_pixel_weights_by_box_regions(10, 20, boxes, blackout)
 166       return output
 167 
 168     output = self.execute(graph_fn, [])
 169     # All zeros in region [0:6, 0:6].
 170     self.assertAlmostEqual(np.sum(output[0:6, 0:6]), 0.0)
 171     # All zeros in region [12:19, 6:9].
 172     self.assertAlmostEqual(np.sum(output[6:9, 12:19]), 0.0)
 173     # All other pixel weights should be 1.0.
 174     # 20 * 10 - 6 * 6 - 3 * 7 = 143.0
 175     self.assertAlmostEqual(np.sum(output), 143.0)
