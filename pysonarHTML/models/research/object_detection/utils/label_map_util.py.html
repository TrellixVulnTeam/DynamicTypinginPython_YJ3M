  34 def _validate_label_map(label_map):
  35   """Checks if a label map is valid.
  36 
  37   Args:
  38     label_map: StringIntLabelMap to validate.
  39 
  40   Raises:
  41     ValueError: if label map is invalid.
  42   """
  43   for item in label_map.item:
  44     if item.id < 0:
  45       raise ValueError('Label map ids should be >= 0.')
  46     if (item.id == 0 and item.name != 'background' and
  47         item.display_name != 'background'):
  48       raise ValueError('Label map id 0 is reserved for the background label')

  51 def create_category_index(categories):
  52   """Creates dictionary of COCO compatible categories keyed by category id.
  53 
  54   Args:
  55     categories: a list of dicts, each of which has the following keys:
  56       'id': (required) an integer id uniquely identifying this category.
  57       'name': (required) string representing category name
  58         e.g., 'cat', 'dog', 'pizza'.
  59 
  60   Returns:
  61     category_index: a dict containing the same entries as categories, but keyed
  62       by the 'id' field of each category.
  63   """
  64   category_index = {}
  65   for cat in categories:
  66     category_index[cat['id']] = cat
  67   return category_index

  73   Args:
  74     label_map: a StringIntLabelMapProto

  79   return max([item.id for item in label_map.item])

  82 def convert_label_map_to_categories(label_map,
  83                                     max_num_classes,
  84                                     use_display_name=True):
  85   """Given label map proto returns categories list compatible with eval.
  86 
  87   This function converts label map proto and returns a list of dicts, each of
  88   which  has the following keys:
  89     'id': (required) an integer id uniquely identifying this category.
  90     'name': (required) string representing category name
  91       e.g., 'cat', 'dog', 'pizza'.
  92     'keypoints': (optional) a dictionary of keypoint string 'label' to integer
  93       'id'.
  94   We only allow class into the list if its id-label_id_offset is
  95   between 0 (inclusive) and max_num_classes (exclusive).
  96   If there are several items mapping to the same id in the label map,
  97   we will only keep the first one in the categories list.
  98 
  99   Args:
 100     label_map: a StringIntLabelMapProto or None.  If None, a default categories
 101       list is created with max_num_classes categories.
 102     max_num_classes: maximum number of (consecutive) label indices to include.
 103     use_display_name: (boolean) choose whether to load 'display_name' field as
 104       category name.  If False or if the display_name field does not exist, uses
 105       'name' field as category names instead.
 106 
 107   Returns:
 108     categories: a list of dictionaries representing all possible categories.
 109   """
 110   categories = []
 111   list_of_ids_already_added = []
 112   if not label_map:
 113     label_id_offset = 1
 114     for class_id in range(max_num_classes):
 115       categories.append({
 116           'id': class_id + label_id_offset,
 117           'name': 'category_{}'.format(class_id + label_id_offset)
 118       })
 119     return categories
 120   for item in label_map.item:
 121     if not 0 < item.id <= max_num_classes:
 122       logging.info(
 123           'Ignore item %d since it falls outside of requested '
 124           'label range.', item.id)
 125       continue
 126     if use_display_name and item.HasField('display_name'):
 127       name = item.display_name
 128     else:
 129       name = item.name
 130     if item.id not in list_of_ids_already_added:
 131       list_of_ids_already_added.append(item.id)
 132       category = {'id': item.id, 'name': name}
 133       if item.keypoints:
 134         keypoints = {}
 135         list_of_keypoint_ids = []
 136         for kv in item.keypoints:
 137           if kv.id in list_of_keypoint_ids:
 138             raise ValueError('Duplicate keypoint ids are not allowed. '
 139                              'Found {} more than once'.format(kv.id))
 140           keypoints[kv.label] = kv.id
 141           list_of_keypoint_ids.append(kv.id)
 142         category['keypoints'] = keypoints
 143       categories.append(category)
 144   return categories

 147 def load_labelmap(path):
 148   """Loads label map proto.
 149 
 150   Args:
 151     path: path to StringIntLabelMap proto text file.
 152   Returns:
 153     a StringIntLabelMapProto
 154   """
 155   with tf.io.gfile.GFile(path, 'r') as fid:
 156     label_map_string = fid.read()
 157     label_map = string_int_label_map_pb2.StringIntLabelMap()
 158     try:
 159       text_format.Merge(label_map_string, label_map)
 160     except text_format.ParseError:
 161       label_map.ParseFromString(label_map_string)
 162   _validate_label_map(label_map)
 163   return label_map

 166 def get_label_map_dict(label_map_path_or_proto,
 167                        use_display_name=False,
 168                        fill_in_gaps_and_background=False):
 169   """Reads a label map and returns a dictionary of label names to id.
 170 
 171   Args:
 172     label_map_path_or_proto: path to StringIntLabelMap proto text file or the
 173       proto itself.
 174     use_display_name: whether to use the label map items' display names as keys.
 175     fill_in_gaps_and_background: whether to fill in gaps and background with
 176     respect to the id field in the proto. The id: 0 is reserved for the
 177     'background' class and will be added if it is missing. All other missing
 178     ids in range(1, max(id)) will be added with a dummy class name
 179     ("class_<id>") if they are missing.
 180 
 181   Returns:
 182     A dictionary mapping label names to id.
 183 
 184   Raises:
 185     ValueError: if fill_in_gaps_and_background and label_map has non-integer or
 186     negative values.
 187   """
 188   if isinstance(label_map_path_or_proto, string_types):
 189     label_map = load_labelmap(label_map_path_or_proto)
 190   else:
 191     _validate_label_map(label_map_path_or_proto)
 192     label_map = label_map_path_or_proto
 193 
 194   label_map_dict = {}
 195   for item in label_map.item:
 196     if use_display_name:
 197       label_map_dict[item.display_name] = item.id
 198     else:
 199       label_map_dict[item.name] = item.id
 200 
 201   if fill_in_gaps_and_background:
 202     values = set(label_map_dict.values())
 203 
 204     if 0 not in values:
 205       label_map_dict['background'] = 0
 206     if not all(isinstance(value, int) for value in values):
 207       raise ValueError('The values in label map must be integers in order to'
 208                        'fill_in_gaps_and_background.')
 209     if not all(value >= 0 for value in values):
 210       raise ValueError('The values in the label map must be positive.')
 211 
 212     if len(values) != max(values) + 1:
 213       # there are gaps in the labels, fill in gaps.
 214       for value in range(1, max(values)):
 215         if value not in values:
 216           # TODO(rathodv): Add a prefix 'class_' here once the tool to generate
 217           # teacher annotation adds this prefix in the data.
 218           label_map_dict[str(value)] = value
 219 
 220   return label_map_dict

 223 def get_label_map_hierarchy_lut(label_map_path_or_proto,
 224                                 include_identity=False):
 225   """Reads a label map and returns ancestors and descendants in the hierarchy.
 226 
 227   The function returns the ancestors and descendants as separate look up tables
 228    (LUT) numpy arrays of shape [max_id, max_id] where lut[i,j] = 1 when there is
 229    a hierarchical relationship between class i and j.
 230 
 231   Args:
 232     label_map_path_or_proto: path to StringIntLabelMap proto text file or the
 233       proto itself.
 234     include_identity: Boolean to indicate whether to include a class element
 235       among its ancestors and descendants. Setting this will result in the lut
 236       diagonal being set to 1.
 237 
 238   Returns:
 239     ancestors_lut: Look up table with the ancestors.
 240     descendants_lut: Look up table with the descendants.
 241   """
 242   if isinstance(label_map_path_or_proto, string_types):
 243     label_map = load_labelmap(label_map_path_or_proto)
 244   else:
 245     _validate_label_map(label_map_path_or_proto)
 246     label_map = label_map_path_or_proto
 247 
 248   hierarchy_dict = {
 249       'ancestors': collections.defaultdict(list),
 250       'descendants': collections.defaultdict(list)
 251   }
 252   max_id = -1
 253   for item in label_map.item:
 254     max_id = max(max_id, item.id)
 255     for ancestor in item.ancestor_ids:
 256       hierarchy_dict['ancestors'][item.id].append(ancestor)
 257     for descendant in item.descendant_ids:
 258       hierarchy_dict['descendants'][item.id].append(descendant)
 259 
 260   def get_graph_relations_tensor(graph_relations):
 261     graph_relations_tensor = np.zeros([max_id, max_id])
 262     for id_val, ids_related in graph_relations.items():
 263       id_val = int(id_val) - _LABEL_OFFSET
 264       for id_related in ids_related:
 265         id_related -= _LABEL_OFFSET
 266         graph_relations_tensor[id_val, id_related] = 1
 267     if include_identity:
 268       graph_relations_tensor += np.eye(max_id)
 269     return graph_relations_tensor
 270 
 271   ancestors_lut = get_graph_relations_tensor(hierarchy_dict['ancestors'])
 272   descendants_lut = get_graph_relations_tensor(hierarchy_dict['descendants'])
 273   return ancestors_lut, descendants_lut

 276 def create_categories_from_labelmap(label_map_path, use_display_name=True):
 277   """Reads a label map and returns categories list compatible with eval.
 278 
 279   This function converts label map proto and returns a list of dicts, each of
 280   which  has the following keys:
 281     'id': an integer id uniquely identifying this category.
 282     'name': string representing category name e.g., 'cat', 'dog'.
 283     'keypoints': a dictionary of keypoint string label to integer id. It is only
 284       returned when available in label map proto.
 285 
 286   Args:
 287     label_map_path: Path to `StringIntLabelMap` proto text file.
 288     use_display_name: (boolean) choose whether to load 'display_name' field
 289       as category name.  If False or if the display_name field does not exist,
 290       uses 'name' field as category names instead.
 291 
 292   Returns:
 293     categories: a list of dictionaries representing all possible categories.
 294   """
 295   label_map = load_labelmap(label_map_path)
 296   max_num_classes = max(item.id for item in label_map.item)
 297   return convert_label_map_to_categories(label_map, max_num_classes,
 298                                          use_display_name)

 301 def create_category_index_from_labelmap(label_map_path, use_display_name=True):
 302   """Reads a label map and returns a category index.
 303 
 304   Args:
 305     label_map_path: Path to `StringIntLabelMap` proto text file.
 306     use_display_name: (boolean) choose whether to load 'display_name' field
 307       as category name.  If False or if the display_name field does not exist,
 308       uses 'name' field as category names instead.
 309 
 310   Returns:
 311     A category index, which is a dictionary that maps integer ids to dicts
 312     containing categories, e.g.
 313     {1: {'id': 1, 'name': 'dog'}, 2: {'id': 2, 'name': 'cat'}, ...}
 314   """
 315   categories = create_categories_from_labelmap(label_map_path, use_display_name)
 316   return create_category_index(categories)

 319 def create_class_agnostic_category_index():
 320   """Creates a category index with a single `object` class."""
 321   return {1: {'id': 1, 'name': 'object'}}
