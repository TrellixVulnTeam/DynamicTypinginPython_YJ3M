  93 
  94     def graph_fn():
  95       t1 = tf.constant([1], dtype=tf.int32)
  96       tt1 = shape_utils.pad_or_clip_tensor(t1, 2)
  97       t2 = tf.constant([[0.1, 0.2]], dtype=tf.float32)
  98       tt2 = shape_utils.pad_or_clip_tensor(t2, 2)
  99 
 100       t3 = tf.constant([1, 2, 3], dtype=tf.int32)
 101       tt3 = shape_utils.clip_tensor(t3, 2)
 102       t4 = tf.constant([[0.1, 0.2], [0.2, 0.4], [0.5, 0.8]], dtype=tf.float32)
 103       tt4 = shape_utils.clip_tensor(t4, 2)
 104

 106       self.assertEqual(2, tt2.get_shape()[0])
 107       self.assertEqual(2, tt3.get_shape()[0])
 108       self.assertEqual(2, tt4.get_shape()[0])
 109 
 110       return tt1, tt2, tt3, tt4
 111 
 112     tt1_result, tt2_result, tt3_result, tt4_result = self.execute(graph_fn, [])
 113     self.assertAllEqual([1, 0], tt1_result)
 114     self.assertAllClose([[0.1, 0.2], [0, 0]], tt2_result)
 115     self.assertAllEqual([1, 2], tt3_result)

 161 class StaticOrDynamicMapFnTest(test_case.TestCase):
 162 
 163   def test_with_dynamic_shape(self):
 164 
 165     def fn(input_tensor):
 166       return tf.reduce_sum(input_tensor)
 167 
 168     def graph_fn(input_tensor):
 169       return shape_utils.static_or_dynamic_map_fn(fn, input_tensor)
 170 
 171     # The input has different shapes, but due to how self.execute()
 172     # works, the shape is known at graph compile time.
 173     result1 = self.execute(
 174         graph_fn, [np.array([[1, 2], [3, 1], [0, 4]]),])
 175     result2 = self.execute(
 176         graph_fn, [np.array([[-1, 1], [0, 9]]),])
 177     self.assertAllEqual(result1, [3, 4, 4])
 178     self.assertAllEqual(result2, [0, 9])
 179 
 180   def test_with_static_shape(self):
 181     def fn(input_tensor):
 182       return tf.reduce_sum(input_tensor)
 183 
 184     def graph_fn():
 185       input_tensor = tf.constant([[1, 2], [3, 1], [0, 4]], dtype=tf.float32)
 186       return shape_utils.static_or_dynamic_map_fn(fn, input_tensor)
 187 
 188     result = self.execute(graph_fn, [])
 189     self.assertAllEqual(result, [3, 4, 4])
 190 
 191   def test_with_multiple_dynamic_shapes(self):
 192     def fn(elems):
 193       input_tensor, scalar_index_tensor = elems
 194       return tf.reshape(tf.slice(input_tensor, scalar_index_tensor, [1]), [])
 195 
 196     def graph_fn(input_tensor, scalar_index_tensor):
 197       map_fn_output = shape_utils.static_or_dynamic_map_fn(
 198           fn, [input_tensor, scalar_index_tensor], dtype=tf.float32)
 199       return map_fn_output
 200 
 201     # The input has different shapes, but due to how self.execute()
 202     # works, the shape is known at graph compile time.
 203 
 204     result1 = self.execute(
 205         graph_fn, [
 206             np.array([[1, 2, 3], [4, 5, -1], [0, 6, 9]]),
 207             np.array([[0], [2], [1]]),
 208         ])
 209     result2 = self.execute(
 210         graph_fn, [
 211             np.array([[-1, 1, 0], [3, 9, 30]]),
 212             np.array([[1], [0]])
 213         ])
 214     self.assertAllEqual(result1, [1, -1, 6])
 215     self.assertAllEqual(result2, [1, 3])

 227           fn, [input_tensor, scalar_index_tensor], dtype=tf.float32)
 228       return map_fn_output
 229

 367   def test_flatten_given_dims(self):
 368 
 369     def graph_fn():
 370       inputs = tf.random_uniform([5, 2, 10, 10, 3])
 371       actual_flattened = shape_utils.flatten_dimensions(inputs, first=1, last=3)
 372       expected_flattened = tf.reshape(inputs, [5, 20, 10, 3])
 373 
 374       return actual_flattened, expected_flattened
 375 
 376     (actual_flattened_np,
 377      expected_flattened_np) = self.execute(graph_fn, [])
 378     self.assertAllClose(expected_flattened_np, actual_flattened_np)
