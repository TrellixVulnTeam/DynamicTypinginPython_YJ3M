  40   def maybe_extract_single_output(self, outputs):
  41     if isinstance(outputs, list) or isinstance(outputs, tuple):
  42       if isinstance(outputs[0], tf.Tensor):
  43         outputs_np = [output.numpy() for output in outputs]
  44       else:
  45         outputs_np = outputs
  46       if len(outputs_np) == 1:
  47         return outputs_np[0]
  48       else:
  49         return outputs_np
  50     else:
  51       if isinstance(outputs, tf.Tensor):
  52         return outputs.numpy()
  53       else:
  54         return outputs

  56   def has_tpu(self):
  57     """Returns whether there are any logical TPU devices."""
  58     return bool(tf.config.experimental.list_logical_devices(device_type='TPU'))

  60   def is_tf2(self):
  61     """Returns whether TF2 is enabled."""
  62     return tf_version.is_tf2()

  64   def execute_tpu_tf1(self, compute_fn, inputs, graph=None):
  65     """Executes compute_fn on TPU with Tensorflow 1.X.
  66 
  67     Args:
  68       compute_fn: a function containing Tensorflow computation that takes a list
  69         of input numpy tensors, performs computation and returns output numpy
  70         tensors.
  71       inputs: a list of numpy arrays to feed input to the `compute_fn`.
  72       graph: (optional) If not None, provided `graph` is used for computation
  73         instead of a brand new tf.Graph().
  74 
  75     Returns:
  76       A list of numpy arrays or a single numpy array.
  77     """
  78     with self.session(graph=(graph or tf.Graph())) as sess:
  79       placeholders = [tf.placeholder_with_default(v, v.shape) for v in inputs]
  80       def wrap_graph_fn(*args, **kwargs):
  81         results = compute_fn(*args, **kwargs)
  82         if (not (isinstance(results, dict) or isinstance(results, tf.Tensor))
  83             and hasattr(results, '__iter__')):
  84           results = list(results)
  85         return results
  86       tpu_computation = contrib_tpu.rewrite(wrap_graph_fn, placeholders)
  87       sess.run(contrib_tpu.initialize_system())
  88       sess.run([tf.global_variables_initializer(), tf.tables_initializer(),
  89                 tf.local_variables_initializer()])
  90       materialized_results = sess.run(tpu_computation,
  91                                       feed_dict=dict(zip(placeholders, inputs)))
  92       sess.run(contrib_tpu.shutdown_system())
  93     return self.maybe_extract_single_output(materialized_results)

  95   def execute_tpu_tf2(self, compute_fn, inputs):
  96     """Executes compute_fn on TPU with Tensorflow 2.X.
  97 
  98     Args:
  99       compute_fn: a function containing Tensorflow computation that takes a list
 100         of input numpy tensors, performs computation and returns output numpy
 101         tensors.
 102       inputs: a list of numpy arrays to feed input to the `compute_fn`.
 103 
 104     Returns:
 105       A list of numpy arrays or a single numpy array.
 106     """
 107     resolver = tf.distribute.cluster_resolver.TPUClusterResolver(tpu='')
 108     tf.config.experimental_connect_to_cluster(resolver)
 109     topology = tf.tpu.experimental.initialize_tpu_system(resolver)
 110     device_assignment = tf.tpu.experimental.DeviceAssignment.build(
 111         topology, num_replicas=1)
 112     strategy = tf.distribute.experimental.TPUStrategy(
 113         resolver, device_assignment=device_assignment)
 114 
 115     @tf.function
 116     def run():
 117       tf_inputs = [tf.constant(input_t) for input_t in inputs]
 118       return strategy.run(compute_fn, args=tf_inputs)
 119     outputs = run()
 120     tf.tpu.experimental.shutdown_tpu_system()
 121     return self.maybe_extract_single_output(outputs)
 122 
 123   def execute_cpu_tf1(self, compute_fn, inputs, graph=None):
 124     """Executes compute_fn on CPU with Tensorflow 1.X.
 125 
 126     Args:
 127       compute_fn: a function containing Tensorflow computation that takes a list
 128         of input numpy tensors, performs computation and returns output numpy
 129         tensors.
 130       inputs: a list of numpy arrays to feed input to the `compute_fn`.
 131       graph: (optional) If not None, provided `graph` is used for computation
 132         instead of a brand new tf.Graph().
 133 
 134     Returns:
 135       A list of numpy arrays or a single numpy array.
 136     """
 137     if self.is_tf2():
 138       raise ValueError('Required version Tenforflow 1.X is not available.')
 139     with self.session(graph=(graph or tf.Graph())) as sess:
 140       placeholders = [tf.placeholder_with_default(v, v.shape) for v in inputs]
 141       results = compute_fn(*placeholders)
 142       if (not (isinstance(results, dict) or isinstance(results, tf.Tensor)) and
 143           hasattr(results, '__iter__')):
 144         results = list(results)
 145       sess.run([tf.global_variables_initializer(), tf.tables_initializer(),
 146                 tf.local_variables_initializer()])
 147       materialized_results = sess.run(results, feed_dict=dict(zip(placeholders,
 148                                                                   inputs)))
 149     return self.maybe_extract_single_output(materialized_results)

 151   def execute_cpu_tf2(self, compute_fn, inputs):
 152     """Executes compute_fn on CPU with Tensorflow 2.X.
 153 
 154     Args:
 155       compute_fn: a function containing Tensorflow computation that takes a list
 156         of input numpy tensors, performs computation and returns output numpy
 157         tensors.
 158       inputs: a list of numpy arrays to feed input to the `compute_fn`.
 159 
 160     Returns:
 161       A list of numpy arrays or a single numpy array.
 162     """
 163     if not self.is_tf2():
 164       raise ValueError('Required version TensorFlow 2.0 is not available.')
 165     @tf.function
 166     def run():
 167       tf_inputs = [tf.constant(input_t) for input_t in inputs]
 168       return compute_fn(*tf_inputs)
 169     return self.maybe_extract_single_output(run())

 171   def execute_cpu(self, compute_fn, inputs, graph=None):
 172     """Executes compute_fn on CPU.
 173 
 174     Depending on the underlying TensorFlow installation (build deps) runs in
 175     either TF 1.X or TF 2.X style.
 176 
 177     Args:
 178       compute_fn: a function containing Tensorflow computation that takes a list
 179         of input numpy tensors, performs computation and returns output numpy
 180         tensors.
 181       inputs: a list of numpy arrays to feed input to the `compute_fn`.
 182       graph: (optional) If not None, provided `graph` is used for computation
 183         instead of a brand new tf.Graph().
 184 
 185     Returns:
 186       A list of numpy arrays or a single tensor.
 187     """
 188     if self.is_tf2():
 189       return self.execute_cpu_tf2(compute_fn, inputs)
 190     else:
 191       return self.execute_cpu_tf1(compute_fn, inputs, graph)
 192 
 193   def execute_tpu(self, compute_fn, inputs, graph=None):
 194     """Executes compute_fn on TPU.
 195 
 196     Depending on the underlying TensorFlow installation (build deps) runs in
 197     either TF 1.X or TF 2.X style.
 198 
 199     Args:
 200       compute_fn: a function containing Tensorflow computation that takes a list
 201         of input numpy tensors, performs computation and returns output numpy
 202         tensors.
 203       inputs: a list of numpy arrays to feed input to the `compute_fn`.
 204       graph: (optional) If not None, provided `graph` is used for computation
 205         instead of a brand new tf.Graph().
 206 
 207     Returns:
 208       A list of numpy arrays or a single tensor.
 209     """
 210     if not self.has_tpu():
 211       raise ValueError('No TPU Device found.')
 212     if self.is_tf2():
 213       return self.execute_tpu_tf2(compute_fn, inputs)
 214     else:
 215       return self.execute_tpu_tf1(compute_fn, inputs, graph)

 217   def execute_tf2(self, compute_fn, inputs):
 218     """Runs compute_fn with TensorFlow 2.0.
 219 
 220     Executes on TPU if available, otherwise executes on CPU.
 221 
 222     Args:
 223       compute_fn: a function containing Tensorflow computation that takes a list
 224         of input numpy tensors, performs computation and returns output numpy
 225         tensors.
 226       inputs: a list of numpy arrays to feed input to the `compute_fn`.
 227 
 228     Returns:
 229       A list of numpy arrays or a single tensor.
 230     """
 231     if not self.is_tf2():
 232       raise ValueError('Required version TensorFlow 2.0 is not available.')
 233     if self.has_tpu():
 234       return self.execute_tpu_tf2(compute_fn, inputs)
 235     else:
 236       return self.execute_cpu_tf2(compute_fn, inputs)

 261   def execute(self, compute_fn, inputs, graph=None):
 262     """Runs compute_fn with inputs and returns results.
 263 
 264     * Executes in either TF1.X or TF2.X style based on the TensorFlow version.
 265     * Executes on TPU if available, otherwise executes on CPU.
 266 
 267     Args:
 268       compute_fn: a function containing Tensorflow computation that takes a list
 269         of input numpy tensors, performs computation and returns output numpy
 270         tensors.
 271       inputs: a list of numpy arrays to feed input to the `compute_fn`.
 272       graph: (optional) If not None, provided `graph` is used for computation
 273         instead of a brand new tf.Graph().
 274 
 275     Returns:
 276       A list of numpy arrays or a single tensor.
 277     """
 278     if self.has_tpu() and tf2.enabled():
 279       return self.execute_tpu_tf2(compute_fn, inputs)
 280     elif not self.has_tpu() and tf2.enabled():
 281       return self.execute_cpu_tf2(compute_fn, inputs)
 282     elif self.has_tpu() and not tf2.enabled():
 283       return self.execute_tpu_tf1(compute_fn, inputs, graph)
 284     else:
 285       return self.execute_cpu_tf1(compute_fn, inputs, graph)
