  79   def test_meshgrid_multidimensional(self):
  80     np.random.seed(18)
  81     x = np.random.rand(4, 1, 2).astype(np.float32)
  82     y = np.random.rand(2, 3).astype(np.float32)
  83 
  84     grid_shape = list(y.shape) + list(x.shape)
  85 
  86     def graph_fn():
  87       xgrid, ygrid = ops.meshgrid(x, y)
  88       self.assertEqual(xgrid.get_shape().as_list(), grid_shape)
  89       self.assertEqual(ygrid.get_shape().as_list(), grid_shape)
  90       return xgrid, ygrid
  91 
  92     xgrid_output, ygrid_output = self.execute(graph_fn, [])
  93 
  94     # Check the shape of the output grids
  95     self.assertEqual(xgrid_output.shape, tuple(grid_shape))
  96     self.assertEqual(ygrid_output.shape, tuple(grid_shape))
  97 
  98     # Check a few elements
  99     test_elements = [((3, 0, 0), (1, 2)),
 100                      ((2, 0, 1), (0, 0)),
 101                      ((0, 0, 0), (1, 1))]
 102     for xind, yind in test_elements:
 103       # These are float equality tests, but the meshgrid op should not introduce
 104       # rounding.
 105       self.assertEqual(xgrid_output[yind + xind], x[xind])
 106       self.assertEqual(ygrid_output[yind + xind], y[yind])
