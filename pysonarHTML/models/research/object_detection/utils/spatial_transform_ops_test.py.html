 320   def test_multilevel_roi_align(self):
 321     image_size = 640
 322     fpn_min_level = 2
 323     fpn_max_level = 5
 324     batch_size = 1
 325     output_size = [2, 2]
 326     num_filters = 1
 327     features = []
 328     for level in range(fpn_min_level, fpn_max_level + 1):
 329       feat_size = int(image_size / 2**level)
 330       features.append(
 331           float(level) *
 332           np.ones([batch_size, feat_size, feat_size, num_filters],
 333                   dtype=np.float32))
 334     boxes = np.array(
 335         [
 336             [
 337                 [0, 0, 111, 111],  # Level 2.
 338                 [0, 0, 113, 113],  # Level 3.
 339                 [0, 0, 223, 223],  # Level 3.
 340                 [0, 0, 225, 225],  # Level 4.
 341                 [0, 0, 449, 449]   # Level 5.
 342             ],
 343         ],
 344         dtype=np.float32) / image_size
 345     levels = np.array([[0, 1, 1, 2, 3]], dtype=np.int32)
 346 
 347     def graph_fn(feature1, feature2, feature3, feature4, boxes, levels):
 348       roi_features = spatial_ops.multilevel_roi_align(
 349           [feature1, feature2, feature3, feature4],
 350           boxes,
 351           levels,
 352           output_size)
 353       return roi_features
 354 
 355     roi_features = self.execute(graph_fn, features + [boxes, levels])
 356     self.assertAllClose(roi_features[0][0], 2 * np.ones((2, 2, 1)))
 357     self.assertAllClose(roi_features[0][1], 3 * np.ones((2, 2, 1)))
 358     self.assertAllClose(roi_features[0][2], 3 * np.ones((2, 2, 1)))
 359     self.assertAllClose(roi_features[0][3], 4 * np.ones((2, 2, 1)))
 360     self.assertAllClose(roi_features[0][4], 5 * np.ones((2, 2, 1)))

 362   def test_large_input(self):
 363     if self.has_tpu():
 364       input_size = 1408
 365       min_level = 2
 366       max_level = 6
 367       batch_size = 2
 368       num_boxes = 512
 369       num_filters = 256
 370       output_size = [7, 7]
 371       features = []
 372       for level in range(min_level, max_level + 1):
 373         feat_size = int(input_size / 2**level)
 374         features.append(
 375             np.reshape(
 376                 np.arange(
 377                     batch_size * feat_size * feat_size * num_filters,
 378                     dtype=np.float32),
 379                 [batch_size, feat_size, feat_size, num_filters]))
 380       boxes = np.array([
 381           [[0, 0, 256, 256]]*num_boxes,
 382       ], dtype=np.float32) / input_size
 383       boxes = np.tile(boxes, [batch_size, 1, 1])
 384       levels = np.random.randint(5, size=[batch_size, num_boxes],
 385                                  dtype=np.int32)
 386       def crop_and_resize_fn():
 387         tf_features = [
 388             tf.constant(feature, dtype=tf.bfloat16) for feature in features
 389         ]
 390         return spatial_ops.multilevel_roi_align(
 391             tf_features, tf.constant(boxes), tf.constant(levels), output_size)
 392       roi_features = self.execute_tpu(crop_and_resize_fn, [])
 393       self.assertEqual(roi_features.shape,
 394                        (batch_size, num_boxes, output_size[0],
 395                         output_size[1], num_filters))
