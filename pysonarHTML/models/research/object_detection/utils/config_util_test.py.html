  45 def _write_config(config, config_path):
  46   """Writes a config object to disk."""
  47   config_text = text_format.MessageToString(config)
  48   with tf.gfile.Open(config_path, "wb") as f:
  49     f.write(config_text)

  53   """Adds a new constant learning rate."""
  54   constant_lr = optimizer.learning_rate.constant_learning_rate
  55   constant_lr.learning_rate = learning_rate

  58 def _update_optimizer_with_exponential_decay_learning_rate(
  59     optimizer, learning_rate):
  60   """Adds a new exponential decay learning rate."""
  61   exponential_lr = optimizer.learning_rate.exponential_decay_learning_rate
  62   exponential_lr.initial_learning_rate = learning_rate

  65 def _update_optimizer_with_manual_step_learning_rate(
  66     optimizer, initial_learning_rate, learning_rate_scaling):
  67   """Adds a learning rate schedule."""
  68   manual_lr = optimizer.learning_rate.manual_step_learning_rate
  69   manual_lr.initial_learning_rate = initial_learning_rate
  70   for i in range(3):
  71     schedule = manual_lr.schedule.add()
  72     schedule.learning_rate = initial_learning_rate * learning_rate_scaling**i
  73

  75 def _update_optimizer_with_cosine_decay_learning_rate(
  76     optimizer, learning_rate, warmup_learning_rate):
  77   """Adds a new cosine decay learning rate."""
  78   cosine_lr = optimizer.learning_rate.cosine_decay_learning_rate
  79   cosine_lr.learning_rate_base = learning_rate
  80   cosine_lr.warmup_learning_rate = warmup_learning_rate

 168   def test_get_configs_from_multiple_files(self):
 169     """Tests that proto configs can be read from multiple files."""
 170     temp_dir = self.get_temp_dir()
 171 
 172     # Write model config file.
 173     model_config_path = os.path.join(temp_dir, "model.config")
 174     model = model_pb2.DetectionModel()
 175     model.faster_rcnn.num_classes = 10
 176     _write_config(model, model_config_path)
 177 
 178     # Write train config file.
 179     train_config_path = os.path.join(temp_dir, "train.config")
 180     train_config = train_config = train_pb2.TrainConfig()
 181     train_config.batch_size = 32
 182     _write_config(train_config, train_config_path)
 183 
 184     # Write train input config file.
 185     train_input_config_path = os.path.join(temp_dir, "train_input.config")
 186     train_input_config = input_reader_pb2.InputReader()
 187     train_input_config.label_map_path = "path/to/label_map"
 188     _write_config(train_input_config, train_input_config_path)
 189 
 190     # Write eval config file.
 191     eval_config_path = os.path.join(temp_dir, "eval.config")
 192     eval_config = eval_pb2.EvalConfig()
 193     eval_config.num_examples = 20
 194     _write_config(eval_config, eval_config_path)
 195 
 196     # Write eval input config file.
 197     eval_input_config_path = os.path.join(temp_dir, "eval_input.config")
 198     eval_input_config = input_reader_pb2.InputReader()
 199     eval_input_config.label_map_path = "path/to/another/label_map"
 200     _write_config(eval_input_config, eval_input_config_path)
 201 
 202     configs = config_util.get_configs_from_multiple_files(
 203         model_config_path=model_config_path,
 204         train_config_path=train_config_path,
 205         train_input_config_path=train_input_config_path,
 206         eval_config_path=eval_config_path,
 207         eval_input_config_path=eval_input_config_path)
 208     self.assertProtoEquals(model, configs["model"])
 209     self.assertProtoEquals(train_config, configs["train_config"])
 210     self.assertProtoEquals(train_input_config,
 211                            configs["train_input_config"])
 212     self.assertProtoEquals(eval_config, configs["eval_config"])
 213     self.assertProtoEquals(eval_input_config, configs["eval_input_configs"][0])

 215   def _assertOptimizerWithNewLearningRate(self, optimizer_name):
 216     """Asserts successful updating of all learning rate schemes."""
 217     original_learning_rate = 0.7
 218     learning_rate_scaling = 0.1
 219     warmup_learning_rate = 0.07
 220     hparams = contrib_training.HParams(learning_rate=0.15)
 221     pipeline_config_path = os.path.join(self.get_temp_dir(), "pipeline.config")
 222 
 223     # Constant learning rate.
 224     pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
 225     optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
 226     _update_optimizer_with_constant_learning_rate(optimizer,
 227                                                   original_learning_rate)
 228     _write_config(pipeline_config, pipeline_config_path)
 229 
 230     configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
 231     configs = config_util.merge_external_params_with_configs(configs, hparams)
 232     optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
 233     constant_lr = optimizer.learning_rate.constant_learning_rate
 234     self.assertAlmostEqual(hparams.learning_rate, constant_lr.learning_rate)
 235 
 236     # Exponential decay learning rate.
 237     pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
 238     optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
 239     _update_optimizer_with_exponential_decay_learning_rate(
 240         optimizer, original_learning_rate)
 241     _write_config(pipeline_config, pipeline_config_path)
 242 
 243     configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
 244     configs = config_util.merge_external_params_with_configs(configs, hparams)
 245     optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
 246     exponential_lr = optimizer.learning_rate.exponential_decay_learning_rate
 247     self.assertAlmostEqual(hparams.learning_rate,
 248                            exponential_lr.initial_learning_rate)
 249 
 250     # Manual step learning rate.
 251     pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
 252     optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
 253     _update_optimizer_with_manual_step_learning_rate(
 254         optimizer, original_learning_rate, learning_rate_scaling)
 255     _write_config(pipeline_config, pipeline_config_path)
 256 
 257     configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
 258     configs = config_util.merge_external_params_with_configs(configs, hparams)
 259     optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
 260     manual_lr = optimizer.learning_rate.manual_step_learning_rate
 261     self.assertAlmostEqual(hparams.learning_rate,
 262                            manual_lr.initial_learning_rate)
 263     for i, schedule in enumerate(manual_lr.schedule):
 264       self.assertAlmostEqual(hparams.learning_rate * learning_rate_scaling**i,
 265                              schedule.learning_rate)
 266 
 267     # Cosine decay learning rate.
 268     pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
 269     optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
 270     _update_optimizer_with_cosine_decay_learning_rate(optimizer,
 271                                                       original_learning_rate,
 272                                                       warmup_learning_rate)
 273     _write_config(pipeline_config, pipeline_config_path)
 274 
 275     configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
 276     configs = config_util.merge_external_params_with_configs(configs, hparams)
 277     optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
 278     cosine_lr = optimizer.learning_rate.cosine_decay_learning_rate
 279 
 280     self.assertAlmostEqual(hparams.learning_rate, cosine_lr.learning_rate_base)
 281     warmup_scale_factor = warmup_learning_rate / original_learning_rate
 282     self.assertAlmostEqual(hparams.learning_rate * warmup_scale_factor,
 283                            cosine_lr.warmup_learning_rate)

 300     pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
