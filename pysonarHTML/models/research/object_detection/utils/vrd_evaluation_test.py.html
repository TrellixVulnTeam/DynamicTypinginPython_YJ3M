 193 class VRDDetectionEvaluationTest(tf.test.TestCase):
 194 
 195   def setUp(self):
 196 
 197     self.vrd_eval = vrd_evaluation._VRDDetectionEvaluation(
 198         matching_iou_threshold=0.5)
 199 
 200     image_key1 = 'img1'
 201     groundtruth_box_tuples1 = np.array(
 202         [([0, 0, 1, 1], [1, 1, 2, 2]), ([0, 0, 1, 1], [1, 2, 2, 3])],
 203         dtype=vrd_evaluation.vrd_box_data_type)
 204     groundtruth_class_tuples1 = np.array(
 205         [(1, 2, 3), (1, 4, 3)], dtype=vrd_evaluation.label_data_type)
 206     self.vrd_eval.add_single_ground_truth_image_info(
 207         image_key1, groundtruth_box_tuples1, groundtruth_class_tuples1)
 208 
 209     image_key2 = 'img2'
 210     groundtruth_box_tuples2 = np.array(
 211         [([0, 0, 1, 1], [1, 1, 2, 2])], dtype=vrd_evaluation.vrd_box_data_type)
 212     groundtruth_class_tuples2 = np.array(
 213         [(1, 4, 3)], dtype=vrd_evaluation.label_data_type)
 214     self.vrd_eval.add_single_ground_truth_image_info(
 215         image_key2, groundtruth_box_tuples2, groundtruth_class_tuples2)
 216 
 217     image_key3 = 'img3'
 218     groundtruth_box_tuples3 = np.array(
 219         [([0, 0, 1, 1], [1, 1, 2, 2])], dtype=vrd_evaluation.vrd_box_data_type)
 220     groundtruth_class_tuples3 = np.array(
 221         [(1, 2, 4)], dtype=vrd_evaluation.label_data_type)
 222     self.vrd_eval.add_single_ground_truth_image_info(
 223         image_key3, groundtruth_box_tuples3, groundtruth_class_tuples3)
 224 
 225     image_key = 'img1'
 226     detected_box_tuples = np.array(
 227         [([0, 0.3, 1, 1], [1.1, 1, 2, 2]), ([0, 0, 1, 1], [1, 1, 2, 2])],
 228         dtype=vrd_evaluation.vrd_box_data_type)
 229     detected_class_tuples = np.array(
 230         [(1, 2, 3), (1, 2, 3)], dtype=vrd_evaluation.label_data_type)
 231     detected_scores = np.array([0.7, 0.8], dtype=float)
 232     self.vrd_eval.add_single_detected_image_info(
 233         image_key, detected_box_tuples, detected_scores, detected_class_tuples)
 234 
 235     metrics = self.vrd_eval.evaluate()
 236     expected_weighted_average_precision = 0.25
 237     expected_mean_average_precision = 0.16666666666666
 238     expected_precision = np.array([1., 0.5], dtype=float)
 239     expected_recall = np.array([0.25, 0.25], dtype=float)
 240     expected_recall_50 = 0.25
 241     expected_recall_100 = 0.25
 242     expected_median_rank_50 = 0
 243     expected_median_rank_100 = 0
 244 
 245     self.assertAlmostEqual(expected_weighted_average_precision,
 246                            metrics.weighted_average_precision)
 247     self.assertAlmostEqual(expected_mean_average_precision,
 248                            metrics.mean_average_precision)
 249     self.assertAlmostEqual(expected_mean_average_precision,
 250                            metrics.mean_average_precision)
 251 
 252     self.assertAllClose(expected_precision, metrics.precisions)
 253     self.assertAllClose(expected_recall, metrics.recalls)
 254     self.assertAlmostEqual(expected_recall_50, metrics.recall_50)
 255     self.assertAlmostEqual(expected_recall_100, metrics.recall_100)
 256     self.assertAlmostEqual(expected_median_rank_50, metrics.median_rank_50)
 257     self.assertAlmostEqual(expected_median_rank_100, metrics.median_rank_100)
