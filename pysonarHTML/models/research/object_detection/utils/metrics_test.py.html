  47   def test_compute_precision_recall(self):
  48     num_gt = 10
  49     scores = np.array([0.4, 0.3, 0.6, 0.2, 0.7, 0.1], dtype=float)
  50     labels = np.array([0, 1, 1, 0, 0, 1], dtype=bool)
  51     labels_float_type = np.array([0, 1, 1, 0, 0, 1], dtype=float)
  52     accumulated_tp_count = np.array([0, 1, 1, 2, 2, 3], dtype=float)
  53     expected_precision = accumulated_tp_count / np.array([1, 2, 3, 4, 5, 6])
  54     expected_recall = accumulated_tp_count / num_gt
  55 
  56     precision, recall = metrics.compute_precision_recall(scores, labels, num_gt)
  57     precision_float_type, recall_float_type = metrics.compute_precision_recall(
  58         scores, labels_float_type, num_gt)
  59 
  60     self.assertAllClose(precision, expected_precision)
  61     self.assertAllClose(recall, expected_recall)
  62     self.assertAllClose(precision_float_type, expected_precision)
  63     self.assertAllClose(recall_float_type, expected_recall)

  77   def test_compute_average_precision(self):
  78     precision = np.array([0.8, 0.76, 0.9, 0.65, 0.7, 0.5, 0.55, 0], dtype=float)
  79     recall = np.array([0.3, 0.3, 0.4, 0.4, 0.45, 0.45, 0.5, 0.5], dtype=float)
  80     processed_precision = np.array(
  81         [0.9, 0.9, 0.9, 0.7, 0.7, 0.55, 0.55, 0], dtype=float)
  82     recall_interval = np.array([0.3, 0, 0.1, 0, 0.05, 0, 0.05, 0], dtype=float)
  83     expected_mean_ap = np.sum(recall_interval * processed_precision)
  84     mean_ap = metrics.compute_average_precision(precision, recall)
  85     self.assertAlmostEqual(expected_mean_ap, mean_ap)

  87   def test_compute_precision_recall_and_ap_no_groundtruth(self):

  90     labels = np.array([0, 0, 0, 0, 0, 0], dtype=bool)

  94     self.assertEquals(precision, expected_precision)
  95     self.assertEquals(recall, expected_recall)
  96     ap = metrics.compute_average_precision(precision, recall)
  97     self.assertTrue(np.isnan(ap))

  99   def test_compute_recall_at_k(self):
 100     num_gt = 4
 101     tp_fp = [
 102         np.array([1, 0, 0], dtype=float),
 103         np.array([0, 1], dtype=float),
 104         np.array([0, 0, 0, 0, 0], dtype=float)
 105     ]
 106     tp_fp_bool = [
 107         np.array([True, False, False], dtype=bool),
 108         np.array([False, True], dtype=float),
 109         np.array([False, False, False, False, False], dtype=float)
 110     ]
 111 
 112     recall_1 = metrics.compute_recall_at_k(tp_fp, num_gt, 1)
 113     recall_3 = metrics.compute_recall_at_k(tp_fp, num_gt, 3)
 114     recall_5 = metrics.compute_recall_at_k(tp_fp, num_gt, 5)
 115 
 116     recall_3_bool = metrics.compute_recall_at_k(tp_fp_bool, num_gt, 3)
 117 
 118     self.assertAlmostEqual(recall_1, 0.25)
 119     self.assertAlmostEqual(recall_3, 0.5)
 120     self.assertAlmostEqual(recall_3_bool, 0.5)
 121     self.assertAlmostEqual(recall_5, 0.5)
