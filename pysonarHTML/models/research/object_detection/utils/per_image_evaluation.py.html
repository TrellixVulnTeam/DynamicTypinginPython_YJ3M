  15 """Evaluate Object Detection result on a single image.
  16 
  17 Annotate each detected result as true positives or false positive according to
  18 a predefined IOU ratio. Non Maximum Supression is used by default. Multi class
  19 detection is supported by default.
  20 Based on the settings, per image evaluation is either performed on boxes or
  21 on object masks.
  22 """

  39   def __init__(self,
  40                num_groundtruth_classes,
  41                matching_iou_threshold=0.5,
  42                nms_iou_threshold=0.3,
  43                nms_max_output_boxes=50,
  44                group_of_weight=0.0):
  45     """Initialized PerImageEvaluation by evaluation parameters.
  46 
  47     Args:
  48       num_groundtruth_classes: Number of ground truth object classes
  49       matching_iou_threshold: A ratio of area intersection to union, which is
  50         the threshold to consider whether a detection is true positive or not
  51       nms_iou_threshold: IOU threshold used in Non Maximum Suppression.
  52       nms_max_output_boxes: Number of maximum output boxes in NMS.
  53       group_of_weight: Weight of the group-of boxes.
  54     """
  55     self.matching_iou_threshold = matching_iou_threshold
  56     self.nms_iou_threshold = nms_iou_threshold
  57     self.nms_max_output_boxes = nms_max_output_boxes
  58     self.num_groundtruth_classes = num_groundtruth_classes
  59     self.group_of_weight = group_of_weight

  61   def compute_object_detection_metrics(self,
  62                                        detected_boxes,
  63                                        detected_scores,
  64                                        detected_class_labels,
  65                                        groundtruth_boxes,
  66                                        groundtruth_class_labels,
  67                                        groundtruth_is_difficult_list,
  68                                        groundtruth_is_group_of_list,
  69                                        detected_masks=None,
  70                                        groundtruth_masks=None):
  71     """Evaluates detections as being tp, fp or weighted from a single image.
  72 
  73     The evaluation is done in two stages:
  74      1. All detections are matched to non group-of boxes; true positives are
  75         determined and detections matched to difficult boxes are ignored.
  76      2. Detections that are determined as false positives are matched against
  77         group-of boxes and weighted if matched.
  78 
  79     Args:
  80       detected_boxes: A float numpy array of shape [N, 4], representing N
  81         regions of detected object regions. Each row is of the format [y_min,
  82         x_min, y_max, x_max]
  83       detected_scores: A float numpy array of shape [N, 1], representing the
  84         confidence scores of the detected N object instances.
  85       detected_class_labels: A integer numpy array of shape [N, 1], repreneting
  86         the class labels of the detected N object instances.
  87       groundtruth_boxes: A float numpy array of shape [M, 4], representing M
  88         regions of object instances in ground truth
  89       groundtruth_class_labels: An integer numpy array of shape [M, 1],
  90         representing M class labels of object instances in ground truth
  91       groundtruth_is_difficult_list: A boolean numpy array of length M denoting
  92         whether a ground truth box is a difficult instance or not
  93       groundtruth_is_group_of_list: A boolean numpy array of length M denoting
  94         whether a ground truth box has group-of tag
  95       detected_masks: (optional) A uint8 numpy array of shape [N, height,
  96         width]. If not None, the metrics will be computed based on masks.
  97       groundtruth_masks: (optional) A uint8 numpy array of shape [M, height,
  98         width]. Can have empty masks, i.e. where all values are 0.
  99 
 100     Returns:
 101       scores: A list of C float numpy arrays. Each numpy array is of
 102           shape [K, 1], representing K scores detected with object class
 103           label c
 104       tp_fp_labels: A list of C boolean numpy arrays. Each numpy array
 105           is of shape [K, 1], representing K True/False positive label of
 106           object instances detected with class label c
 107       is_class_correctly_detected_in_image: a numpy integer array of
 108           shape [C, 1], indicating whether the correponding class has a least
 109           one instance being correctly detected in the image
 110     """
 111     detected_boxes, detected_scores, detected_class_labels, detected_masks = (
 112         self._remove_invalid_boxes(detected_boxes, detected_scores,
 113                                    detected_class_labels, detected_masks))
 114     scores, tp_fp_labels = self._compute_tp_fp(
 115         detected_boxes=detected_boxes,
 116         detected_scores=detected_scores,
 117         detected_class_labels=detected_class_labels,
 118         groundtruth_boxes=groundtruth_boxes,
 119         groundtruth_class_labels=groundtruth_class_labels,
 120         groundtruth_is_difficult_list=groundtruth_is_difficult_list,
 121         groundtruth_is_group_of_list=groundtruth_is_group_of_list,
 122         detected_masks=detected_masks,
 123         groundtruth_masks=groundtruth_masks)
 124 
 125     is_class_correctly_detected_in_image = self._compute_cor_loc(
 126         detected_boxes=detected_boxes,
 127         detected_scores=detected_scores,
 128         detected_class_labels=detected_class_labels,
 129         groundtruth_boxes=groundtruth_boxes,
 130         groundtruth_class_labels=groundtruth_class_labels,
 131         detected_masks=detected_masks,
 132         groundtruth_masks=groundtruth_masks)
 133 
 134     return scores, tp_fp_labels, is_class_correctly_detected_in_image
 135 
 136   def _compute_cor_loc(self,
 137                        detected_boxes,
 138                        detected_scores,
 139                        detected_class_labels,
 140                        groundtruth_boxes,
 141                        groundtruth_class_labels,
 142                        detected_masks=None,
 143                        groundtruth_masks=None):
 144     """Compute CorLoc score for object detection result.
 145 
 146     Args:
 147       detected_boxes: A float numpy array of shape [N, 4], representing N
 148         regions of detected object regions. Each row is of the format [y_min,
 149         x_min, y_max, x_max]
 150       detected_scores: A float numpy array of shape [N, 1], representing the
 151         confidence scores of the detected N object instances.
 152       detected_class_labels: A integer numpy array of shape [N, 1], repreneting
 153         the class labels of the detected N object instances.
 154       groundtruth_boxes: A float numpy array of shape [M, 4], representing M
 155         regions of object instances in ground truth
 156       groundtruth_class_labels: An integer numpy array of shape [M, 1],
 157         representing M class labels of object instances in ground truth
 158       detected_masks: (optional) A uint8 numpy array of shape [N, height,
 159         width]. If not None, the scores will be computed based on masks.
 160       groundtruth_masks: (optional) A uint8 numpy array of shape [M, height,
 161         width].
 162 
 163     Returns:
 164       is_class_correctly_detected_in_image: a numpy integer array of
 165           shape [C, 1], indicating whether the correponding class has a least
 166           one instance being correctly detected in the image
 167 
 168     Raises:
 169       ValueError: If detected masks is not None but groundtruth masks are None,
 170         or the other way around.
 171     """
 172     if (detected_masks is not None and
 173         groundtruth_masks is None) or (detected_masks is None and
 174                                        groundtruth_masks is not None):
 175       raise ValueError(
 176           'If `detected_masks` is provided, then `groundtruth_masks` should '
 177           'also be provided.')
 178 
 179     is_class_correctly_detected_in_image = np.zeros(
 180         self.num_groundtruth_classes, dtype=int)
 181     for i in range(self.num_groundtruth_classes):
 182       (gt_boxes_at_ith_class, gt_masks_at_ith_class,
 183        detected_boxes_at_ith_class, detected_scores_at_ith_class,
 184        detected_masks_at_ith_class) = self._get_ith_class_arrays(
 185            detected_boxes, detected_scores, detected_masks,
 186            detected_class_labels, groundtruth_boxes, groundtruth_masks,
 187            groundtruth_class_labels, i)
 188       is_class_correctly_detected_in_image[i] = (
 189           self._compute_is_class_correctly_detected_in_image(
 190               detected_boxes=detected_boxes_at_ith_class,
 191               detected_scores=detected_scores_at_ith_class,
 192               groundtruth_boxes=gt_boxes_at_ith_class,
 193               detected_masks=detected_masks_at_ith_class,
 194               groundtruth_masks=gt_masks_at_ith_class))
 195 
 196     return is_class_correctly_detected_in_image
 197 
 198   def _compute_is_class_correctly_detected_in_image(self,
 199                                                     detected_boxes,
 200                                                     detected_scores,
 201                                                     groundtruth_boxes,
 202                                                     detected_masks=None,
 203                                                     groundtruth_masks=None):
 204     """Compute CorLoc score for a single class.
 205 
 206     Args:
 207       detected_boxes: A numpy array of shape [N, 4] representing detected box
 208         coordinates
 209       detected_scores: A 1-d numpy array of length N representing classification
 210         score
 211       groundtruth_boxes: A numpy array of shape [M, 4] representing ground truth
 212         box coordinates
 213       detected_masks: (optional) A np.uint8 numpy array of shape [N, height,
 214         width]. If not None, the scores will be computed based on masks.
 215       groundtruth_masks: (optional) A np.uint8 numpy array of shape [M, height,
 216         width].
 217 
 218     Returns:
 219       is_class_correctly_detected_in_image: An integer 1 or 0 denoting whether a
 220           class is correctly detected in the image or not
 221     """
 222     if detected_boxes.size > 0:
 223       if groundtruth_boxes.size > 0:
 224         max_score_id = np.argmax(detected_scores)
 225         mask_mode = False
 226         if detected_masks is not None and groundtruth_masks is not None:
 227           mask_mode = True
 228         if mask_mode:
 229           detected_boxlist = np_box_mask_list.BoxMaskList(
 230               box_data=np.expand_dims(detected_boxes[max_score_id], axis=0),
 231               mask_data=np.expand_dims(detected_masks[max_score_id], axis=0))
 232           gt_boxlist = np_box_mask_list.BoxMaskList(
 233               box_data=groundtruth_boxes, mask_data=groundtruth_masks)
 234           iou = np_box_mask_list_ops.iou(detected_boxlist, gt_boxlist)
 235         else:
 236           detected_boxlist = np_box_list.BoxList(
 237               np.expand_dims(detected_boxes[max_score_id, :], axis=0))
 238           gt_boxlist = np_box_list.BoxList(groundtruth_boxes)
 239           iou = np_box_list_ops.iou(detected_boxlist, gt_boxlist)
 240         if np.max(iou) >= self.matching_iou_threshold:
 241           return 1
 242     return 0

 244   def _compute_tp_fp(self,
 245                      detected_boxes,
 246                      detected_scores,
 247                      detected_class_labels,
 248                      groundtruth_boxes,
 249                      groundtruth_class_labels,
 250                      groundtruth_is_difficult_list,
 251                      groundtruth_is_group_of_list,
 252                      detected_masks=None,
 253                      groundtruth_masks=None):
 254     """Labels true/false positives of detections of an image across all classes.
 255 
 256     Args:
 257       detected_boxes: A float numpy array of shape [N, 4], representing N
 258         regions of detected object regions. Each row is of the format [y_min,
 259         x_min, y_max, x_max]
 260       detected_scores: A float numpy array of shape [N, 1], representing the
 261         confidence scores of the detected N object instances.
 262       detected_class_labels: A integer numpy array of shape [N, 1], repreneting
 263         the class labels of the detected N object instances.
 264       groundtruth_boxes: A float numpy array of shape [M, 4], representing M
 265         regions of object instances in ground truth
 266       groundtruth_class_labels: An integer numpy array of shape [M, 1],
 267         representing M class labels of object instances in ground truth
 268       groundtruth_is_difficult_list: A boolean numpy array of length M denoting
 269         whether a ground truth box is a difficult instance or not
 270       groundtruth_is_group_of_list: A boolean numpy array of length M denoting
 271         whether a ground truth box has group-of tag
 272       detected_masks: (optional) A np.uint8 numpy array of shape [N, height,
 273         width]. If not None, the scores will be computed based on masks.
 274       groundtruth_masks: (optional) A np.uint8 numpy array of shape [M, height,
 275         width].
 276 
 277     Returns:
 278       result_scores: A list of float numpy arrays. Each numpy array is of
 279           shape [K, 1], representing K scores detected with object class
 280           label c
 281       result_tp_fp_labels: A list of boolean numpy array. Each numpy array is of
 282           shape [K, 1], representing K True/False positive label of object
 283           instances detected with class label c
 284 
 285     Raises:
 286       ValueError: If detected masks is not None but groundtruth masks are None,
 287         or the other way around.
 288     """
 289     if detected_masks is not None and groundtruth_masks is None:
 290       raise ValueError(
 291           'Detected masks is available but groundtruth masks is not.')
 292     if detected_masks is None and groundtruth_masks is not None:
 293       raise ValueError(
 294           'Groundtruth masks is available but detected masks is not.')
 295 
 296     result_scores = []
 297     result_tp_fp_labels = []
 298     for i in range(self.num_groundtruth_classes):
 299       groundtruth_is_difficult_list_at_ith_class = (
 300           groundtruth_is_difficult_list[groundtruth_class_labels == i])
 301       groundtruth_is_group_of_list_at_ith_class = (
 302           groundtruth_is_group_of_list[groundtruth_class_labels == i])
 303       (gt_boxes_at_ith_class, gt_masks_at_ith_class,
 304        detected_boxes_at_ith_class, detected_scores_at_ith_class,
 305        detected_masks_at_ith_class) = self._get_ith_class_arrays(
 306            detected_boxes, detected_scores, detected_masks,
 307            detected_class_labels, groundtruth_boxes, groundtruth_masks,
 308            groundtruth_class_labels, i)
 309       scores, tp_fp_labels = self._compute_tp_fp_for_single_class(
 310           detected_boxes=detected_boxes_at_ith_class,
 311           detected_scores=detected_scores_at_ith_class,
 312           groundtruth_boxes=gt_boxes_at_ith_class,
 313           groundtruth_is_difficult_list=groundtruth_is_difficult_list_at_ith_class,
 314           groundtruth_is_group_of_list=groundtruth_is_group_of_list_at_ith_class,
 315           detected_masks=detected_masks_at_ith_class,
 316           groundtruth_masks=gt_masks_at_ith_class)
 317       result_scores.append(scores)
 318       result_tp_fp_labels.append(tp_fp_labels)
 319     return result_scores, result_tp_fp_labels
 320 
 321   def _get_overlaps_and_scores_mask_mode(self, detected_boxes, detected_scores,
 322                                          detected_masks, groundtruth_boxes,
 323                                          groundtruth_masks,
 324                                          groundtruth_is_group_of_list):
 325     """Computes overlaps and scores between detected and groudntruth masks.
 326 
 327     Args:
 328       detected_boxes: A numpy array of shape [N, 4] representing detected box
 329         coordinates
 330       detected_scores: A 1-d numpy array of length N representing classification
 331         score
 332       detected_masks: A uint8 numpy array of shape [N, height, width]. If not
 333         None, the scores will be computed based on masks.
 334       groundtruth_boxes: A numpy array of shape [M, 4] representing ground truth
 335         box coordinates
 336       groundtruth_masks: A uint8 numpy array of shape [M, height, width].
 337       groundtruth_is_group_of_list: A boolean numpy array of length M denoting
 338         whether a ground truth box has group-of tag. If a groundtruth box is
 339         group-of box, every detection matching this box is ignored.
 340 
 341     Returns:
 342       iou: A float numpy array of size [num_detected_boxes, num_gt_boxes]. If
 343           gt_non_group_of_boxlist.num_boxes() == 0 it will be None.
 344       ioa: A float numpy array of size [num_detected_boxes, num_gt_boxes]. If
 345           gt_group_of_boxlist.num_boxes() == 0 it will be None.
 346       scores: The score of the detected boxlist.
 347       num_boxes: Number of non-maximum suppressed detected boxes.
 348     """
 349     detected_boxlist = np_box_mask_list.BoxMaskList(
 350         box_data=detected_boxes, mask_data=detected_masks)
 351     detected_boxlist.add_field('scores', detected_scores)
 352     detected_boxlist = np_box_mask_list_ops.non_max_suppression(
 353         detected_boxlist, self.nms_max_output_boxes, self.nms_iou_threshold)
 354     gt_non_group_of_boxlist = np_box_mask_list.BoxMaskList(
 355         box_data=groundtruth_boxes[~groundtruth_is_group_of_list],
 356         mask_data=groundtruth_masks[~groundtruth_is_group_of_list])
 357     gt_group_of_boxlist = np_box_mask_list.BoxMaskList(
 358         box_data=groundtruth_boxes[groundtruth_is_group_of_list],
 359         mask_data=groundtruth_masks[groundtruth_is_group_of_list])
 360     iou = np_box_mask_list_ops.iou(detected_boxlist, gt_non_group_of_boxlist)
 361     ioa = np.transpose(
 362         np_box_mask_list_ops.ioa(gt_group_of_boxlist, detected_boxlist))
 363     scores = detected_boxlist.get_field('scores')
 364     num_boxes = detected_boxlist.num_boxes()
 365     return iou, ioa, scores, num_boxes
 366 
 367   def _get_overlaps_and_scores_box_mode(self, detected_boxes, detected_scores,
 368                                         groundtruth_boxes,
 369                                         groundtruth_is_group_of_list):
 370     """Computes overlaps and scores between detected and groudntruth boxes.
 371 
 372     Args:
 373       detected_boxes: A numpy array of shape [N, 4] representing detected box
 374         coordinates
 375       detected_scores: A 1-d numpy array of length N representing classification
 376         score
 377       groundtruth_boxes: A numpy array of shape [M, 4] representing ground truth
 378         box coordinates
 379       groundtruth_is_group_of_list: A boolean numpy array of length M denoting
 380         whether a ground truth box has group-of tag. If a groundtruth box is
 381         group-of box, every detection matching this box is ignored.
 382 
 383     Returns:
 384       iou: A float numpy array of size [num_detected_boxes, num_gt_boxes]. If
 385           gt_non_group_of_boxlist.num_boxes() == 0 it will be None.
 386       ioa: A float numpy array of size [num_detected_boxes, num_gt_boxes]. If
 387           gt_group_of_boxlist.num_boxes() == 0 it will be None.
 388       scores: The score of the detected boxlist.
 389       num_boxes: Number of non-maximum suppressed detected boxes.
 390     """
 391     detected_boxlist = np_box_list.BoxList(detected_boxes)
 392     detected_boxlist.add_field('scores', detected_scores)
 393     detected_boxlist = np_box_list_ops.non_max_suppression(
 394         detected_boxlist, self.nms_max_output_boxes, self.nms_iou_threshold)
 395     gt_non_group_of_boxlist = np_box_list.BoxList(
 396         groundtruth_boxes[~groundtruth_is_group_of_list])
 397     gt_group_of_boxlist = np_box_list.BoxList(
 398         groundtruth_boxes[groundtruth_is_group_of_list])
 399     iou = np_box_list_ops.iou(detected_boxlist, gt_non_group_of_boxlist)
 400     ioa = np.transpose(
 401         np_box_list_ops.ioa(gt_group_of_boxlist, detected_boxlist))
 402     scores = detected_boxlist.get_field('scores')
 403     num_boxes = detected_boxlist.num_boxes()
 404     return iou, ioa, scores, num_boxes

 406   def _compute_tp_fp_for_single_class(self,
 407                                       detected_boxes,
 408                                       detected_scores,
 409                                       groundtruth_boxes,
 410                                       groundtruth_is_difficult_list,
 411                                       groundtruth_is_group_of_list,
 412                                       detected_masks=None,
 413                                       groundtruth_masks=None):
 414     """Labels boxes detected with the same class from the same image as tp/fp.
 415 
 416     Args:
 417       detected_boxes: A numpy array of shape [N, 4] representing detected box
 418         coordinates
 419       detected_scores: A 1-d numpy array of length N representing classification
 420         score
 421       groundtruth_boxes: A numpy array of shape [M, 4] representing ground truth
 422         box coordinates
 423       groundtruth_is_difficult_list: A boolean numpy array of length M denoting
 424         whether a ground truth box is a difficult instance or not. If a
 425         groundtruth box is difficult, every detection matching this box is
 426         ignored.
 427       groundtruth_is_group_of_list: A boolean numpy array of length M denoting
 428         whether a ground truth box has group-of tag. If a groundtruth box is
 429         group-of box, every detection matching this box is ignored.
 430       detected_masks: (optional) A uint8 numpy array of shape [N, height,
 431         width]. If not None, the scores will be computed based on masks.
 432       groundtruth_masks: (optional) A uint8 numpy array of shape [M, height,
 433         width].
 434 
 435     Returns:
 436       Two arrays of the same size, containing all boxes that were evaluated as
 437       being true positives or false positives; if a box matched to a difficult
 438       box or to a group-of box, it is ignored.
 439 
 440       scores: A numpy array representing the detection scores.
 441       tp_fp_labels: a boolean numpy array indicating whether a detection is a
 442           true positive.
 443     """
 444     if detected_boxes.size == 0:
 445       return np.array([], dtype=float), np.array([], dtype=bool)
 446 
 447     mask_mode = False
 448     if detected_masks is not None and groundtruth_masks is not None:
 449       mask_mode = True
 450 
 451     iou = np.ndarray([0, 0])
 452     ioa = np.ndarray([0, 0])
 453     iou_mask = np.ndarray([0, 0])
 454     ioa_mask = np.ndarray([0, 0])
 455     if mask_mode:
 456       # For Instance Segmentation Evaluation on Open Images V5, not all boxed
 457       # instances have corresponding segmentation annotations. Those boxes that
 458       # dont have segmentation annotations are represented as empty masks in
 459       # groundtruth_masks nd array.
 460       mask_presence_indicator = (np.sum(groundtruth_masks, axis=(1, 2)) > 0)
 461 
 462       (iou_mask, ioa_mask, scores,
 463        num_detected_boxes) = self._get_overlaps_and_scores_mask_mode(
 464            detected_boxes=detected_boxes,
 465            detected_scores=detected_scores,
 466            detected_masks=detected_masks,
 467            groundtruth_boxes=groundtruth_boxes[mask_presence_indicator, :],
 468            groundtruth_masks=groundtruth_masks[mask_presence_indicator, :],
 469            groundtruth_is_group_of_list=groundtruth_is_group_of_list[
 470                mask_presence_indicator])
 471       if sum(mask_presence_indicator) < len(mask_presence_indicator):
 472         # Not all masks are present - some masks are empty
 473         (iou, ioa, _,
 474          num_detected_boxes) = self._get_overlaps_and_scores_box_mode(
 475              detected_boxes=detected_boxes,
 476              detected_scores=detected_scores,
 477              groundtruth_boxes=groundtruth_boxes[~mask_presence_indicator, :],
 478              groundtruth_is_group_of_list=groundtruth_is_group_of_list[
 479                  ~mask_presence_indicator])
 480       num_detected_boxes = detected_boxes.shape[0]
 481     else:
 482       mask_presence_indicator = np.zeros(
 483           groundtruth_is_group_of_list.shape, dtype=bool)
 484       (iou, ioa, scores,
 485        num_detected_boxes) = self._get_overlaps_and_scores_box_mode(
 486            detected_boxes=detected_boxes,
 487            detected_scores=detected_scores,
 488            groundtruth_boxes=groundtruth_boxes,
 489            groundtruth_is_group_of_list=groundtruth_is_group_of_list)
 490 
 491     if groundtruth_boxes.size == 0:
 492       return scores, np.zeros(num_detected_boxes, dtype=bool)
 493 
 494     tp_fp_labels = np.zeros(num_detected_boxes, dtype=bool)
 495     is_matched_to_box = np.zeros(num_detected_boxes, dtype=bool)
 496     is_matched_to_difficult = np.zeros(num_detected_boxes, dtype=bool)
 497     is_matched_to_group_of = np.zeros(num_detected_boxes, dtype=bool)
 498 
 499     def compute_match_iou(iou, groundtruth_nongroup_of_is_difficult_list,
 500                           is_box):
 501       """Computes TP/FP for non group-of box matching.
 502 
 503       The function updates the following local variables:
 504         tp_fp_labels - if a box is matched to group-of
 505         is_matched_to_difficult - the detections that were processed at this are
 506           matched to difficult box.
 507         is_matched_to_box - the detections that were processed at this stage are
 508           marked as is_box.
 509 
 510       Args:
 511         iou: intersection-over-union matrix [num_gt_boxes]x[num_det_boxes].
 512         groundtruth_nongroup_of_is_difficult_list: boolean that specifies if gt
 513           box is difficult.
 514         is_box: boolean that specifies if currently boxes or masks are
 515           processed.
 516       """
 517       max_overlap_gt_ids = np.argmax(iou, axis=1)
 518       is_gt_detected = np.zeros(iou.shape[1], dtype=bool)
 519       for i in range(num_detected_boxes):
 520         gt_id = max_overlap_gt_ids[i]
 521         is_evaluatable = (not tp_fp_labels[i] and
 522                           not is_matched_to_difficult[i] and
 523                           iou[i, gt_id] >= self.matching_iou_threshold and
 524                           not is_matched_to_group_of[i])
 525         if is_evaluatable:
 526           if not groundtruth_nongroup_of_is_difficult_list[gt_id]:
 527             if not is_gt_detected[gt_id]:
 528               tp_fp_labels[i] = True
 529               is_gt_detected[gt_id] = True
 530               is_matched_to_box[i] = is_box
 531           else:
 532             is_matched_to_difficult[i] = True
 533 
 534     def compute_match_ioa(ioa, is_box):
 535       """Computes TP/FP for group-of box matching.
 536 
 537       The function updates the following local variables:
 538         is_matched_to_group_of - if a box is matched to group-of
 539         is_matched_to_box - the detections that were processed at this stage are
 540           marked as is_box.
 541 
 542       Args:
 543         ioa: intersection-over-area matrix [num_gt_boxes]x[num_det_boxes].
 544         is_box: boolean that specifies if currently boxes or masks are
 545           processed.
 546 
 547       Returns:
 548         scores_group_of: of detections matched to group-of boxes
 549         [num_groupof_matched].
 550         tp_fp_labels_group_of: boolean array of size [num_groupof_matched], all
 551           values are True.
 552       """
 553       scores_group_of = np.zeros(ioa.shape[1], dtype=float)
 554       tp_fp_labels_group_of = self.group_of_weight * np.ones(
 555           ioa.shape[1], dtype=float)
 556       max_overlap_group_of_gt_ids = np.argmax(ioa, axis=1)
 557       for i in range(num_detected_boxes):
 558         gt_id = max_overlap_group_of_gt_ids[i]
 559         is_evaluatable = (not tp_fp_labels[i] and
 560                           not is_matched_to_difficult[i] and
 561                           ioa[i, gt_id] >= self.matching_iou_threshold and
 562                           not is_matched_to_group_of[i])
 563         if is_evaluatable:
 564           is_matched_to_group_of[i] = True
 565           is_matched_to_box[i] = is_box
 566           scores_group_of[gt_id] = max(scores_group_of[gt_id], scores[i])
 567       selector = np.where((scores_group_of > 0) & (tp_fp_labels_group_of > 0))
 568       scores_group_of = scores_group_of[selector]
 569       tp_fp_labels_group_of = tp_fp_labels_group_of[selector]
 570 
 571       return scores_group_of, tp_fp_labels_group_of
 572 
 573     # The evaluation is done in two stages:
 574     # 1. Evaluate all objects that actually have instance level masks.
 575     # 2. Evaluate all objects that are not already evaluated as boxes.
 576     if iou_mask.shape[1] > 0:
 577       groundtruth_is_difficult_mask_list = groundtruth_is_difficult_list[
 578           mask_presence_indicator]
 579       groundtruth_is_group_of_mask_list = groundtruth_is_group_of_list[
 580           mask_presence_indicator]
 581       compute_match_iou(
 582           iou_mask,
 583           groundtruth_is_difficult_mask_list[
 584               ~groundtruth_is_group_of_mask_list],
 585           is_box=False)
 586 
 587     scores_mask_group_of = np.ndarray([0], dtype=float)
 588     tp_fp_labels_mask_group_of = np.ndarray([0], dtype=float)
 589     if ioa_mask.shape[1] > 0:
 590       scores_mask_group_of, tp_fp_labels_mask_group_of = compute_match_ioa(
 591           ioa_mask, is_box=False)
 592 
 593     # Tp-fp evaluation for non-group of boxes (if any).
 594     if iou.shape[1] > 0:
 595       groundtruth_is_difficult_box_list = groundtruth_is_difficult_list[
 596           ~mask_presence_indicator]
 597       groundtruth_is_group_of_box_list = groundtruth_is_group_of_list[
 598           ~mask_presence_indicator]
 599       compute_match_iou(
 600           iou,
 601           groundtruth_is_difficult_box_list[~groundtruth_is_group_of_box_list],
 602           is_box=True)
 603 
 604     scores_box_group_of = np.ndarray([0], dtype=float)
 605     tp_fp_labels_box_group_of = np.ndarray([0], dtype=float)
 606     if ioa.shape[1] > 0:
 607       scores_box_group_of, tp_fp_labels_box_group_of = compute_match_ioa(
 608           ioa, is_box=True)
 609 
 610     if mask_mode:
 611       # Note: here crowds are treated as ignore regions.
 612       valid_entries = (~is_matched_to_difficult & ~is_matched_to_group_of
 613                        & ~is_matched_to_box)
 614       return np.concatenate(
 615           (scores[valid_entries], scores_mask_group_of)), np.concatenate(
 616               (tp_fp_labels[valid_entries].astype(float),
 617                tp_fp_labels_mask_group_of))
 618     else:
 619       valid_entries = (~is_matched_to_difficult & ~is_matched_to_group_of)
 620       return np.concatenate(
 621           (scores[valid_entries], scores_box_group_of)), np.concatenate(
 622               (tp_fp_labels[valid_entries].astype(float),
 623                tp_fp_labels_box_group_of))
 624 
 625   def _get_ith_class_arrays(self, detected_boxes, detected_scores,
 626                             detected_masks, detected_class_labels,
 627                             groundtruth_boxes, groundtruth_masks,
 628                             groundtruth_class_labels, class_index):
 629     """Returns numpy arrays belonging to class with index `class_index`.
 630 
 631     Args:
 632       detected_boxes: A numpy array containing detected boxes.
 633       detected_scores: A numpy array containing detected scores.
 634       detected_masks: A numpy array containing detected masks.
 635       detected_class_labels: A numpy array containing detected class labels.
 636       groundtruth_boxes: A numpy array containing groundtruth boxes.
 637       groundtruth_masks: A numpy array containing groundtruth masks.
 638       groundtruth_class_labels: A numpy array containing groundtruth class
 639         labels.
 640       class_index: An integer index.
 641 
 642     Returns:
 643       gt_boxes_at_ith_class: A numpy array containing groundtruth boxes labeled
 644         as ith class.
 645       gt_masks_at_ith_class: A numpy array containing groundtruth masks labeled
 646         as ith class.
 647       detected_boxes_at_ith_class: A numpy array containing detected boxes
 648         corresponding to the ith class.
 649       detected_scores_at_ith_class: A numpy array containing detected scores
 650         corresponding to the ith class.
 651       detected_masks_at_ith_class: A numpy array containing detected masks
 652         corresponding to the ith class.
 653     """
 654     selected_groundtruth = (groundtruth_class_labels == class_index)
 655     gt_boxes_at_ith_class = groundtruth_boxes[selected_groundtruth]
 656     if groundtruth_masks is not None:
 657       gt_masks_at_ith_class = groundtruth_masks[selected_groundtruth]
 658     else:
 659       gt_masks_at_ith_class = None
 660     selected_detections = (detected_class_labels == class_index)
 661     detected_boxes_at_ith_class = detected_boxes[selected_detections]
 662     detected_scores_at_ith_class = detected_scores[selected_detections]
 663     if detected_masks is not None:
 664       detected_masks_at_ith_class = detected_masks[selected_detections]
 665     else:
 666       detected_masks_at_ith_class = None
 667     return (gt_boxes_at_ith_class, gt_masks_at_ith_class,
 668             detected_boxes_at_ith_class, detected_scores_at_ith_class,
 669             detected_masks_at_ith_class)

 671   def _remove_invalid_boxes(self,
 672                             detected_boxes,
 673                             detected_scores,
 674                             detected_class_labels,
 675                             detected_masks=None):
 676     """Removes entries with invalid boxes.
 677 
 678     A box is invalid if either its xmax is smaller than its xmin, or its ymax
 679     is smaller than its ymin.
 680 
 681     Args:
 682       detected_boxes: A float numpy array of size [num_boxes, 4] containing box
 683         coordinates in [ymin, xmin, ymax, xmax] format.
 684       detected_scores: A float numpy array of size [num_boxes].
 685       detected_class_labels: A int32 numpy array of size [num_boxes].
 686       detected_masks: A uint8 numpy array of size [num_boxes, height, width].
 687 
 688     Returns:
 689       valid_detected_boxes: A float numpy array of size [num_valid_boxes, 4]
 690         containing box coordinates in [ymin, xmin, ymax, xmax] format.
 691       valid_detected_scores: A float numpy array of size [num_valid_boxes].
 692       valid_detected_class_labels: A int32 numpy array of size
 693         [num_valid_boxes].
 694       valid_detected_masks: A uint8 numpy array of size
 695         [num_valid_boxes, height, width].
 696     """
 697     valid_indices = np.logical_and(detected_boxes[:, 0] < detected_boxes[:, 2],
 698                                    detected_boxes[:, 1] < detected_boxes[:, 3])
 699     detected_boxes = detected_boxes[valid_indices]
 700     detected_scores = detected_scores[valid_indices]
 701     detected_class_labels = detected_class_labels[valid_indices]
 702     if detected_masks is not None:
 703       detected_masks = detected_masks[valid_indices]
 704     return [
 705         detected_boxes, detected_scores, detected_class_labels, detected_masks
 706     ]
