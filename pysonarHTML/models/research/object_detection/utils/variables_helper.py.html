  32 # TODO(derekjchow): Consider replacing with tf.contrib.filter_variables in
  33 # tensorflow/contrib/framework/python/ops/variables.py
  34 def filter_variables(variables, filter_regex_list, invert=False):
  35   """Filters out the variables matching the filter_regex.
  36 
  37   Filter out the variables whose name matches the any of the regular
  38   expressions in filter_regex_list and returns the remaining variables.
  39   Optionally, if invert=True, the complement set is returned.
  40 
  41   Args:
  42     variables: a list of tensorflow variables.
  43     filter_regex_list: a list of string regular expressions.
  44     invert: (boolean).  If True, returns the complement of the filter set; that
  45       is, all variables matching filter_regex are kept and all others discarded.
  46 
  47   Returns:
  48     a list of filtered variables.
  49   """
  50   if tf.executing_eagerly():
  51     raise ValueError('Accessing variables is not supported in eager mode.')
  52   kept_vars = []
  53   variables_to_ignore_patterns = list([fre for fre in filter_regex_list if fre])
  54   for var in variables:
  55     add = True
  56     for pattern in variables_to_ignore_patterns:
  57       if re.match(pattern, var.op.name):
  58         add = False
  59         break
  60     if add != invert:
  61       kept_vars.append(var)
  62   return kept_vars

  65 def multiply_gradients_matching_regex(grads_and_vars, regex_list, multiplier):
  66   """Multiply gradients whose variable names match a regular expression.
  67 
  68   Args:
  69     grads_and_vars: A list of gradient to variable pairs (tuples).
  70     regex_list: A list of string regular expressions.
  71     multiplier: A (float) multiplier to apply to each gradient matching the
  72       regular expression.
  73 
  74   Returns:
  75     grads_and_vars: A list of gradient to variable pairs (tuples).
  76   """
  77   if tf.executing_eagerly():
  78     raise ValueError('Accessing variables is not supported in eager mode.')
  79   variables = [pair[1] for pair in grads_and_vars]
  80   matching_vars = filter_variables(variables, regex_list, invert=True)
  81   for var in matching_vars:
  82     logging.info('Applying multiplier %f to variable [%s]',
  83                  multiplier, var.op.name)
  84   grad_multipliers = {var: float(multiplier) for var in matching_vars}
  85   return slim.learning.multiply_gradients(grads_and_vars,
  86                                           grad_multipliers)

  89 def freeze_gradients_matching_regex(grads_and_vars, regex_list):
  90   """Freeze gradients whose variable names match a regular expression.
  91 
  92   Args:
  93     grads_and_vars: A list of gradient to variable pairs (tuples).
  94     regex_list: A list of string regular expressions.
  95 
  96   Returns:
  97     grads_and_vars: A list of gradient to variable pairs (tuples) that do not
  98       contain the variables and gradients matching the regex.
  99   """
 100   if tf.executing_eagerly():
 101     raise ValueError('Accessing variables is not supported in eager mode.')
 102   variables = [pair[1] for pair in grads_and_vars]
 103   matching_vars = filter_variables(variables, regex_list, invert=True)
 104   kept_grads_and_vars = [pair for pair in grads_and_vars
 105                          if pair[1] not in matching_vars]
 106   for var in matching_vars:
 107     logging.info('Freezing variable [%s]', var.op.name)
 108   return kept_grads_and_vars

 111 def get_variables_available_in_checkpoint(variables,
 112                                           checkpoint_path,
 113                                           include_global_step=True):
 114   """Returns the subset of variables available in the checkpoint.
 115 
 116   Inspects given checkpoint and returns the subset of variables that are
 117   available in it.
 118 
 119   TODO(rathodv): force input and output to be a dictionary.
 120 
 121   Args:
 122     variables: a list or dictionary of variables to find in checkpoint.
 123     checkpoint_path: path to the checkpoint to restore variables from.
 124     include_global_step: whether to include `global_step` variable, if it
 125       exists. Default True.
 126 
 127   Returns:
 128     A list or dictionary of variables.
 129   Raises:
 130     ValueError: if `variables` is not a list or dict.
 131   """
 132   if tf.executing_eagerly():
 133     raise ValueError('Accessing variables is not supported in eager mode.')
 134   if isinstance(variables, list):
 135     variable_names_map = {}
 136     for variable in variables:
 137       if isinstance(variable, tf_variables.PartitionedVariable):
 138         name = variable.name
 139       else:
 140         name = variable.op.name
 141       variable_names_map[name] = variable
 142   elif isinstance(variables, dict):
 143     variable_names_map = variables
 144   else:
 145     raise ValueError('`variables` is expected to be a list or dict.')
 146   ckpt_reader = tf.train.NewCheckpointReader(checkpoint_path)
 147   ckpt_vars_to_shape_map = ckpt_reader.get_variable_to_shape_map()
 148   if not include_global_step:
 149     ckpt_vars_to_shape_map.pop(tf.GraphKeys.GLOBAL_STEP, None)
 150   vars_in_ckpt = {}
 151   for variable_name, variable in sorted(variable_names_map.items()):
 152     if variable_name in ckpt_vars_to_shape_map:
 153       if ckpt_vars_to_shape_map[variable_name] == variable.shape.as_list():
 154         vars_in_ckpt[variable_name] = variable
 155       else:
 156         logging.warning('Variable [%s] is available in checkpoint, but has an '
 157                         'incompatible shape with model variable. Checkpoint '
 158                         'shape: [%s], model variable shape: [%s]. This '
 159                         'variable will not be initialized from the checkpoint.',
 160                         variable_name, ckpt_vars_to_shape_map[variable_name],
 161                         variable.shape.as_list())
 162     else:
 163       logging.warning('Variable [%s] is not available in checkpoint',
 164                       variable_name)
 165   if isinstance(variables, list):
 166     return list(vars_in_ckpt.values())
 167   return vars_in_ckpt

 170 def get_global_variables_safely():
 171   """If not executing eagerly, returns tf.global_variables().
 172 
 173   Raises a ValueError if eager execution is enabled,
 174   because the variables are not tracked when executing eagerly.
 175 
 176   If executing eagerly, use a Keras model's .variables property instead.
 177 
 178   Returns:
 179     The result of tf.global_variables()
 180   """
 181   if tf.executing_eagerly():
 182     raise ValueError('Accessing variables is not supported in eager mode.')
 183   with tf.init_scope():
 184     if tf.executing_eagerly():
 185       raise ValueError("Global variables collection is not tracked when "
 186                        "executing eagerly. Use a Keras model's `.variables` "
 187                        "attribute instead.")
 188   return tf.global_variables()
