  33   def test_returns_correct_metric_values(self):
  34     categories = [{
  35         'id': 1,
  36         'name': 'cat'
  37     }, {
  38         'id': 2,
  39         'name': 'dog'
  40     }, {
  41         'id': 3,
  42         'name': 'elephant'
  43     }]
  44 
  45     oiv2_evaluator = object_detection_evaluation.OpenImagesDetectionEvaluator(
  46         categories)
  47     image_key1 = 'img1'
  48     groundtruth_boxes1 = np.array(
  49         [[0, 0, 1, 1], [0, 0, 2, 2], [0, 0, 3, 3]], dtype=float)
  50     groundtruth_class_labels1 = np.array([1, 3, 1], dtype=int)
  51     oiv2_evaluator.add_single_ground_truth_image_info(image_key1, {
  52         standard_fields.InputDataFields.groundtruth_boxes:
  53             groundtruth_boxes1,
  54         standard_fields.InputDataFields.groundtruth_classes:
  55             groundtruth_class_labels1,
  56         standard_fields.InputDataFields.groundtruth_group_of:
  57             np.array([], dtype=bool)
  58     })
  59     image_key2 = 'img2'
  60     groundtruth_boxes2 = np.array(
  61         [[10, 10, 11, 11], [500, 500, 510, 510], [10, 10, 12, 12]], dtype=float)
  62     groundtruth_class_labels2 = np.array([1, 1, 3], dtype=int)
  63     groundtruth_is_group_of_list2 = np.array([False, True, False], dtype=bool)
  64     oiv2_evaluator.add_single_ground_truth_image_info(image_key2, {
  65         standard_fields.InputDataFields.groundtruth_boxes:
  66             groundtruth_boxes2,
  67         standard_fields.InputDataFields.groundtruth_classes:
  68             groundtruth_class_labels2,
  69         standard_fields.InputDataFields.groundtruth_group_of:
  70             groundtruth_is_group_of_list2
  71     })
  72     image_key3 = 'img3'
  73     groundtruth_boxes3 = np.array([[0, 0, 1, 1]], dtype=float)
  74     groundtruth_class_labels3 = np.array([2], dtype=int)
  75     oiv2_evaluator.add_single_ground_truth_image_info(image_key3, {
  76         standard_fields.InputDataFields.groundtruth_boxes:
  77             groundtruth_boxes3,
  78         standard_fields.InputDataFields.groundtruth_classes:
  79             groundtruth_class_labels3
  80     })
  81     # Add detections
  82     image_key = 'img2'
  83     detected_boxes = np.array(
  84         [[10, 10, 11, 11], [100, 100, 120, 120], [100, 100, 220, 220]],
  85         dtype=float)
  86     detected_class_labels = np.array([1, 1, 3], dtype=int)
  87     detected_scores = np.array([0.7, 0.8, 0.9], dtype=float)
  88     oiv2_evaluator.add_single_detected_image_info(image_key, {
  89         standard_fields.DetectionResultFields.detection_boxes:
  90             detected_boxes,
  91         standard_fields.DetectionResultFields.detection_scores:
  92             detected_scores,
  93         standard_fields.DetectionResultFields.detection_classes:
  94             detected_class_labels
  95     })
  96     metrics = oiv2_evaluator.evaluate()
  97     self.assertAlmostEqual(
  98         metrics['OpenImagesV2_PerformanceByCategory/AP@0.5IOU/dog'], 0.0)
  99     self.assertAlmostEqual(
 100         metrics['OpenImagesV2_PerformanceByCategory/AP@0.5IOU/elephant'], 0.0)
 101     self.assertAlmostEqual(
 102         metrics['OpenImagesV2_PerformanceByCategory/AP@0.5IOU/cat'], 0.16666666)
 103     self.assertAlmostEqual(metrics['OpenImagesV2_Precision/mAP@0.5IOU'],
 104                            0.05555555)
 105     oiv2_evaluator.clear()
 106     self.assertFalse(oiv2_evaluator._image_ids)
