  27 class SingleClassPerImageVrdEvaluationTest(tf.test.TestCase):
  28 
  29   def setUp(self):
  30     matching_iou_threshold = 0.5
  31     self.eval = per_image_vrd_evaluation.PerImageVRDEvaluation(
  32         matching_iou_threshold)
  33     box_data_type = np.dtype([('subject', 'f4', (4,)), ('object', 'f4', (4,))])
  34 
  35     self.detected_box_tuples = np.array(
  36         [([0, 0, 1.1, 1], [1, 1, 2, 2]), ([0, 0, 1, 1], [1, 1, 2, 2]),
  37          ([1, 1, 2, 2], [0, 0, 1.1, 1])],
  38         dtype=box_data_type)
  39     self.detected_scores = np.array([0.8, 0.2, 0.1], dtype=float)
  40     self.groundtruth_box_tuples = np.array(
  41         [([0, 0, 1, 1], [1, 1, 2, 2])], dtype=box_data_type)
  42 
  43   def test_tp_fp_eval(self):
  44     tp_fp_labels = self.eval._compute_tp_fp_for_single_class(
  45         self.detected_box_tuples, self.groundtruth_box_tuples)
  46     expected_tp_fp_labels = np.array([True, False, False], dtype=bool)
  47     self.assertTrue(np.allclose(expected_tp_fp_labels, tp_fp_labels))
  48 
  49   def test_tp_fp_eval_empty_gt(self):
  50     box_data_type = np.dtype([('subject', 'f4', (4,)), ('object', 'f4', (4,))])
  51 
  52     tp_fp_labels = self.eval._compute_tp_fp_for_single_class(
  53         self.detected_box_tuples, np.array([], dtype=box_data_type))
  54     expected_tp_fp_labels = np.array([False, False, False], dtype=bool)
  55     self.assertTrue(np.allclose(expected_tp_fp_labels, tp_fp_labels))

  58 class MultiClassPerImageVrdEvaluationTest(tf.test.TestCase):
  59 
  60   def setUp(self):
  61     matching_iou_threshold = 0.5
  62     self.eval = per_image_vrd_evaluation.PerImageVRDEvaluation(
  63         matching_iou_threshold)
  64     box_data_type = np.dtype([('subject', 'f4', (4,)), ('object', 'f4', (4,))])
  65     label_data_type = np.dtype([('subject', 'i4'), ('object', 'i4'),
  66                                 ('relation', 'i4')])
  67 
  68     self.detected_box_tuples = np.array(
  69         [([0, 0, 1, 1], [1, 1, 2, 2]), ([0, 0, 1.1, 1], [1, 1, 2, 2]),
  70          ([1, 1, 2, 2], [0, 0, 1.1, 1]), ([0, 0, 1, 1], [3, 4, 5, 6])],
  71         dtype=box_data_type)
  72     self.detected_class_tuples = np.array(
  73         [(1, 2, 3), (1, 2, 3), (1, 2, 3), (1, 4, 5)], dtype=label_data_type)
  74     self.detected_scores = np.array([0.2, 0.8, 0.1, 0.5], dtype=float)
  75 
  76     self.groundtruth_box_tuples = np.array(
  77         [([0, 0, 1, 1], [1, 1, 2, 2]), ([1, 1, 2, 2], [0, 0, 1.1, 1]),
  78          ([0, 0, 1, 1], [3, 4, 5, 5.5])],
  79         dtype=box_data_type)
  80     self.groundtruth_class_tuples = np.array(
  81         [(1, 2, 3), (1, 7, 3), (1, 4, 5)], dtype=label_data_type)
  82 
  83   def test_tp_fp_eval(self):
  84     scores, tp_fp_labels, mapping = self.eval.compute_detection_tp_fp(
  85         self.detected_box_tuples, self.detected_scores,
  86         self.detected_class_tuples, self.groundtruth_box_tuples,
  87         self.groundtruth_class_tuples)
  88 
  89     expected_scores = np.array([0.8, 0.5, 0.2, 0.1], dtype=float)
  90     expected_tp_fp_labels = np.array([True, True, False, False], dtype=bool)
  91     expected_mapping = np.array([1, 3, 0, 2])
  92 
  93     self.assertTrue(np.allclose(expected_scores, scores))
  94     self.assertTrue(np.allclose(expected_tp_fp_labels, tp_fp_labels))
  95     self.assertTrue(np.allclose(expected_mapping, mapping))
  96
