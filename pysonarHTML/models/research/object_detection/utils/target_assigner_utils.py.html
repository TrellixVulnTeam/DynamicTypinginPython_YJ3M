  22 def image_shape_to_grids(height, width):
  23   """Computes xy-grids given the shape of the image.
  24 
  25   Args:
  26     height: The height of the image.
  27     width: The width of the image.
  28 
  29   Returns:
  30     A tuple of two tensors:
  31       y_grid: A float tensor with shape [height, width] representing the
  32         y-coordinate of each pixel grid.
  33       x_grid: A float tensor with shape [height, width] representing the
  34         x-coordinate of each pixel grid.
  35   """
  36   out_height = tf.cast(height, tf.float32)
  37   out_width = tf.cast(width, tf.float32)
  38   x_range = tf.range(out_width, dtype=tf.float32)
  39   y_range = tf.range(out_height, dtype=tf.float32)
  40   x_grid, y_grid = tf.meshgrid(x_range, y_range, indexing='xy')
  41   return (y_grid, x_grid)

  44 def coordinates_to_heatmap(y_grid,
  45                            x_grid,
  46                            y_coordinates,
  47                            x_coordinates,
  48                            sigma,
  49                            channel_onehot,
  50                            channel_weights=None):
  51   """Returns the heatmap targets from a set of point coordinates.
  52 
  53   This function maps a set of point coordinates to the output heatmap image
  54   applied using a Gaussian kernel. Note that this function be can used by both
  55   object detection and keypoint estimation tasks. For object detection, the
  56   "channel" refers to the object class. For keypoint estimation, the "channel"
  57   refers to the number of keypoint types.
  58 
  59   Args:
  60     y_grid: A 2D tensor with shape [height, width] which contains the grid
  61       y-coordinates given in the (output) image dimensions.
  62     x_grid: A 2D tensor with shape [height, width] which contains the grid
  63       x-coordinates given in the (output) image dimensions.
  64     y_coordinates: A 1D tensor with shape [num_instances] representing the
  65       y-coordinates of the instances in the output space coordinates.
  66     x_coordinates: A 1D tensor with shape [num_instances] representing the
  67       x-coordinates of the instances in the output space coordinates.
  68     sigma: A 1D tensor with shape [num_instances] representing the standard
  69       deviation of the Gaussian kernel to be applied to the point.
  70     channel_onehot: A 2D tensor with shape [num_instances, num_channels]
  71       representing the one-hot encoded channel labels for each point.
  72     channel_weights: A 1D tensor with shape [num_instances] corresponding to the
  73       weight of each instance.
  74 
  75   Returns:
  76     heatmap: A tensor of size [height, width, num_channels] representing the
  77       heatmap. Output (height, width) match the dimensions of the input grids.
  78   """
  79   num_instances, num_channels = (
  80       shape_utils.combined_static_and_dynamic_shape(channel_onehot))
  81 
  82   x_grid = tf.expand_dims(x_grid, 2)
  83   y_grid = tf.expand_dims(y_grid, 2)
  84   # The raw center coordinates in the output space.
  85   x_diff = x_grid - tf.math.floor(x_coordinates)
  86   y_diff = y_grid - tf.math.floor(y_coordinates)
  87   squared_distance = x_diff**2 + y_diff**2
  88 
  89   gaussian_map = tf.exp(-squared_distance / (2 * sigma * sigma))
  90 
  91   reshaped_gaussian_map = tf.expand_dims(gaussian_map, axis=-1)
  92   reshaped_channel_onehot = tf.reshape(channel_onehot,
  93                                        (1, 1, num_instances, num_channels))
  94   gaussian_per_box_per_class_map = (
  95       reshaped_gaussian_map * reshaped_channel_onehot)
  96 
  97   if channel_weights is not None:
  98     reshaped_weights = tf.reshape(channel_weights, (1, 1, num_instances, 1))
  99     gaussian_per_box_per_class_map *= reshaped_weights
 100 
 101   # Take maximum along the "instance" dimension so that all per-instance
 102   # heatmaps of the same class are merged together.
 103   heatmap = tf.reduce_max(gaussian_per_box_per_class_map, axis=2)
 104 
 105   # Maximum of an empty tensor is -inf, the following is to avoid that.
 106   heatmap = tf.maximum(heatmap, 0)
 107 
 108   return heatmap

 111 def compute_floor_offsets_with_indices(y_source,
 112                                        x_source,
 113                                        y_target=None,
 114                                        x_target=None):
 115   """Computes offsets from floored source(floored) to target coordinates.
 116 
 117   This function computes the offsets from source coordinates ("floored" as if
 118   they were put on the grids) to target coordinates. Note that the input
 119   coordinates should be the "absolute" coordinates in terms of the output image
 120   dimensions as opposed to the normalized coordinates (i.e. values in [0, 1]).
 121 
 122   Args:
 123     y_source: A tensor with shape [num_points] representing the absolute
 124       y-coordinates (in the output image space) of the source points.
 125     x_source: A tensor with shape [num_points] representing the absolute
 126       x-coordinates (in the output image space) of the source points.
 127     y_target: A tensor with shape [num_points] representing the absolute
 128       y-coordinates (in the output image space) of the target points. If not
 129       provided, then y_source is used as the targets.
 130     x_target: A tensor with shape [num_points] representing the absolute
 131       x-coordinates (in the output image space) of the target points. If not
 132       provided, then x_source is used as the targets.
 133 
 134   Returns:
 135     A tuple of two tensors:
 136       offsets: A tensor with shape [num_points, 2] representing the offsets of
 137         each input point.
 138       indices: A tensor with shape [num_points, 2] representing the indices of
 139         where the offsets should be retrieved in the output image dimension
 140         space.
 141   """
 142   y_source_floored = tf.floor(y_source)
 143   x_source_floored = tf.floor(x_source)
 144   if y_target is None:
 145     y_target = y_source
 146   if x_target is None:
 147     x_target = x_source
 148 
 149   y_offset = y_target - y_source_floored
 150   x_offset = x_target - x_source_floored
 151 
 152   y_source_indices = tf.cast(y_source_floored, tf.int32)
 153   x_source_indices = tf.cast(x_source_floored, tf.int32)
 154 
 155   indices = tf.stack([y_source_indices, x_source_indices], axis=1)
 156   offsets = tf.stack([y_offset, x_offset], axis=1)
 157 
 158   return offsets, indices

 161 def get_valid_keypoint_mask_for_class(keypoint_coordinates,
 162                                       class_id,
 163                                       class_onehot,
 164                                       class_weights=None,
 165                                       keypoint_indices=None):
 166   """Mask keypoints by their class ids and indices.
 167 
 168   For a given task, we may want to only consider a subset of instances or
 169   keypoints. This function is used to provide the mask (in terms of weights) to
 170   mark those elements which should be considered based on the classes of the
 171   instances and optionally, their keypoint indices. Note that the NaN values
 172   in the keypoints will also be masked out.
 173 
 174   Args:
 175     keypoint_coordinates: A float tensor with shape [num_instances,
 176       num_keypoints, 2] which contains the coordinates of each keypoint.
 177     class_id: An integer representing the target class id to be selected.
 178     class_onehot: A 2D tensor of shape [num_instances, num_classes] repesents
 179       the onehot (or k-hot) encoding of the class for each instance.
 180     class_weights: A 1D tensor of shape [num_instances] repesents the weight of
 181       each instance. If not provided, all instances are weighted equally.
 182     keypoint_indices: A list of integers representing the keypoint indices used
 183       to select the values on the keypoint dimension. If provided, the output
 184       dimension will be [num_instances, len(keypoint_indices)]
 185 
 186   Returns:
 187     A tuple of tensors:
 188       mask: A float tensor of shape [num_instances, K], where K is num_keypoints
 189         or len(keypoint_indices) if provided. The tensor has values either 0 or
 190         1 indicating whether an element in the input keypoints should be used.
 191       keypoints_nan_to_zeros: Same as input keypoints with the NaN values
 192         replaced by zeros and selected columns corresponding to the
 193         keypoint_indices (if provided). The shape of this tensor will always be
 194         the same as the output mask.
 195   """
 196   num_keypoints = tf.shape(keypoint_coordinates)[1]
 197   class_mask = class_onehot[:, class_id]
 198   reshaped_class_mask = tf.tile(
 199       tf.expand_dims(class_mask, axis=-1), multiples=[1, num_keypoints])
 200   not_nan = tf.math.logical_not(tf.math.is_nan(keypoint_coordinates))
 201   mask = reshaped_class_mask * tf.cast(not_nan[:, :, 0], dtype=tf.float32)
 202   keypoints_nan_to_zeros = tf.where(not_nan, keypoint_coordinates,
 203                                     tf.zeros_like(keypoint_coordinates))
 204   if class_weights is not None:
 205     reshaped_class_weight = tf.tile(
 206         tf.expand_dims(class_weights, axis=-1), multiples=[1, num_keypoints])
 207     mask = mask * reshaped_class_weight
 208 
 209   if keypoint_indices is not None:
 210     mask = tf.gather(mask, indices=keypoint_indices, axis=1)
 211     keypoints_nan_to_zeros = tf.gather(
 212         keypoints_nan_to_zeros, indices=keypoint_indices, axis=1)
 213   return mask, keypoints_nan_to_zeros

 216 def blackout_pixel_weights_by_box_regions(height, width, boxes, blackout):
 217   """Blackout the pixel weights in the target box regions.
 218 
 219   This function is used to generate the pixel weight mask (usually in the output
 220   image dimension). The mask is to ignore some regions when computing loss.
 221 
 222   Args:
 223     height: int, height of the (output) image.
 224     width: int, width of the (output) image.
 225     boxes: A float tensor with shape [num_instances, 4] indicating the
 226       coordinates of the four corners of the boxes.
 227     blackout: A boolean tensor with shape [num_instances] indicating whether to
 228       blackout (zero-out) the weights within the box regions.
 229 
 230   Returns:
 231     A float tensor with shape [height, width] where all values within the
 232     regions of the blackout boxes are 0.0 and 1.0 else where.
 233   """
 234   (y_grid, x_grid) = image_shape_to_grids(height, width)
 235   y_grid = tf.expand_dims(y_grid, axis=0)
 236   x_grid = tf.expand_dims(x_grid, axis=0)
 237   y_min = tf.expand_dims(boxes[:, 0:1], axis=-1)
 238   x_min = tf.expand_dims(boxes[:, 1:2], axis=-1)
 239   y_max = tf.expand_dims(boxes[:, 2:3], axis=-1)
 240   x_max = tf.expand_dims(boxes[:, 3:], axis=-1)
 241 
 242   # Make the mask with all 1.0 in the box regions.
 243   # Shape: [num_instances, height, width]
 244   in_boxes = tf.cast(
 245       tf.logical_and(
 246           tf.logical_and(y_grid >= y_min, y_grid <= y_max),
 247           tf.logical_and(x_grid >= x_min, x_grid <= x_max)),
 248       dtype=tf.float32)
 249 
 250   # Shape: [num_instances, height, width]
 251   blackout = tf.tile(
 252       tf.expand_dims(tf.expand_dims(blackout, axis=-1), axis=-1),
 253       [1, height, width])
 254 
 255   # Select only the boxes specified by blackout.
 256   selected_in_boxes = tf.where(blackout, in_boxes, tf.zeros_like(in_boxes))
 257   out_boxes = tf.reduce_max(selected_in_boxes, axis=0)
 258   out_boxes = tf.ones_like(out_boxes) - out_boxes
 259   return out_boxes
