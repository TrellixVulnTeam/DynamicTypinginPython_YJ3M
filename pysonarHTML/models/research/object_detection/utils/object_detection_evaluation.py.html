  15 """object_detection_evaluation module.
  16 
  17 ObjectDetectionEvaluation is a class which manages ground truth information of a
  18 object detection dataset, and computes frequently used detection metrics such as
  19 Precision, Recall, CorLoc of the provided detection results.
  20 It supports the following operations:
  21 1) Add ground truth information of images sequentially.
  22 2) Add detection result of images sequentially.
  23 3) Evaluate detection metrics on already inserted detection results.
  24 4) Write evaluation result into a pickle file for future processing or
  25    visualization.
  26 
  27 Note: This module operates on numpy boxes and box lists.
  28 """

  51 class DetectionEvaluator(six.with_metaclass(ABCMeta, object)):
  52   """Interface for object detection evalution classes.
  53 
  54   Example usage of the Evaluator:
  55   ------------------------------
  56   evaluator = DetectionEvaluator(categories)
  57 
  58   # Detections and groundtruth for image 1.
  59   evaluator.add_single_groundtruth_image_info(...)
  60   evaluator.add_single_detected_image_info(...)
  61 
  62   # Detections and groundtruth for image 2.
  63   evaluator.add_single_groundtruth_image_info(...)
  64   evaluator.add_single_detected_image_info(...)
  65 
  66   metrics_dict = evaluator.evaluate()
  67   """
  68

  71

  74         'id': (required) an integer id uniquely identifying this category.
  75         'name': (required) string representing category name e.g., 'cat', 'dog'.
  76     """

 149 class ObjectDetectionEvaluator(DetectionEvaluator):
 150   """A class to evaluate detections."""
 151 
 152   def __init__(self,
 153                categories,
 154                matching_iou_threshold=0.5,
 155                recall_lower_bound=0.0,
 156                recall_upper_bound=1.0,
 157                evaluate_corlocs=False,
 158                evaluate_precision_recall=False,
 159                metric_prefix=None,
 160                use_weighted_mean_ap=False,
 161                evaluate_masks=False,
 162                group_of_weight=0.0):
 163     """Constructor.
 164 
 165     Args:
 166       categories: A list of dicts, each of which has the following keys -
 167         'id': (required) an integer id uniquely identifying this category.
 168         'name': (required) string representing category name e.g., 'cat', 'dog'.
 169       matching_iou_threshold: IOU threshold to use for matching groundtruth
 170         boxes to detection boxes.
 171       recall_lower_bound: lower bound of recall operating area.
 172       recall_upper_bound: upper bound of recall operating area.
 173       evaluate_corlocs: (optional) boolean which determines if corloc scores are
 174         to be returned or not.
 175       evaluate_precision_recall: (optional) boolean which determines if
 176         precision and recall values are to be returned or not.
 177       metric_prefix: (optional) string prefix for metric name; if None, no
 178         prefix is used.
 179       use_weighted_mean_ap: (optional) boolean which determines if the mean
 180         average precision is computed directly from the scores and tp_fp_labels
 181         of all classes.
 182       evaluate_masks: If False, evaluation will be performed based on boxes. If
 183         True, mask evaluation will be performed instead.
 184       group_of_weight: Weight of group-of boxes.If set to 0, detections of the
 185         correct class within a group-of box are ignored. If weight is > 0, then
 186         if at least one detection falls within a group-of box with
 187         matching_iou_threshold, weight group_of_weight is added to true
 188         positives. Consequently, if no detection falls within a group-of box,
 189         weight group_of_weight is added to false negatives.
 190 
 191     Raises:
 192       ValueError: If the category ids are not 1-indexed.
 193     """
 194     super(ObjectDetectionEvaluator, self).__init__(categories)
 195     self._num_classes = max([cat['id'] for cat in categories])
 196     if min(cat['id'] for cat in categories) < 1:
 197       raise ValueError('Classes should be 1-indexed.')
 198     self._matching_iou_threshold = matching_iou_threshold
 199     self._recall_lower_bound = recall_lower_bound
 200     self._recall_upper_bound = recall_upper_bound
 201     self._use_weighted_mean_ap = use_weighted_mean_ap
 202     self._label_id_offset = 1
 203     self._evaluate_masks = evaluate_masks
 204     self._group_of_weight = group_of_weight
 205     self._evaluation = ObjectDetectionEvaluation(
 206         num_groundtruth_classes=self._num_classes,
 207         matching_iou_threshold=self._matching_iou_threshold,
 208         recall_lower_bound=self._recall_lower_bound,
 209         recall_upper_bound=self._recall_upper_bound,
 210         use_weighted_mean_ap=self._use_weighted_mean_ap,
 211         label_id_offset=self._label_id_offset,
 212         group_of_weight=self._group_of_weight)
 213     self._image_ids = set([])
 214     self._evaluate_corlocs = evaluate_corlocs
 215     self._evaluate_precision_recall = evaluate_precision_recall
 216     self._metric_prefix = (metric_prefix + '_') if metric_prefix else ''
 217     self._expected_keys = set([
 218         standard_fields.InputDataFields.key,
 219         standard_fields.InputDataFields.groundtruth_boxes,
 220         standard_fields.InputDataFields.groundtruth_classes,
 221         standard_fields.InputDataFields.groundtruth_difficult,
 222         standard_fields.InputDataFields.groundtruth_instance_masks,
 223         standard_fields.DetectionResultFields.detection_boxes,
 224         standard_fields.DetectionResultFields.detection_scores,
 225         standard_fields.DetectionResultFields.detection_classes,
 226         standard_fields.DetectionResultFields.detection_masks
 227     ])
 228     self._build_metric_names()
 229 
 230   def get_internal_state(self):
 231     """Returns internal state and image ids that lead to the state.
 232 
 233     Note that only evaluation results will be returned (e.g. not raw predictions
 234     or groundtruth.
 235     """
 236     return self._evaluation.get_internal_state(), self._image_ids

 238   def merge_internal_state(self, image_ids, state_tuple):
 239     """Merges internal state with the existing state of evaluation.
 240 
 241     If image_id is already seen by evaluator, an error will be thrown.
 242 
 243     Args:
 244       image_ids: list of images whose state is stored in the tuple.
 245       state_tuple: state.
 246     """
 247     for image_id in image_ids:
 248       if image_id in self._image_ids:
 249         raise ValueError('Image with id {} already added.'.format(image_id))
 250 
 251     self._evaluation.merge_internal_state(state_tuple)

 253   def _build_metric_names(self):
 254     """Builds a list with metric names."""
 255     if self._recall_lower_bound > 0.0 or self._recall_upper_bound < 1.0:
 256       self._metric_names = [
 257           self._metric_prefix +
 258           'Precision/mAP@{}IOU@[{:.1f},{:.1f}]Recall'.format(
 259               self._matching_iou_threshold, self._recall_lower_bound,
 260               self._recall_upper_bound)
 261       ]
 262     else:
 263       self._metric_names = [
 264           self._metric_prefix +
 265           'Precision/mAP@{}IOU'.format(self._matching_iou_threshold)
 266       ]
 267     if self._evaluate_corlocs:
 268       self._metric_names.append(
 269           self._metric_prefix +
 270           'Precision/meanCorLoc@{}IOU'.format(self._matching_iou_threshold))
 271 
 272     category_index = label_map_util.create_category_index(self._categories)
 273     for idx in range(self._num_classes):
 274       if idx + self._label_id_offset in category_index:
 275         category_name = category_index[idx + self._label_id_offset]['name']
 276         try:
 277           category_name = six.text_type(category_name, 'utf-8')
 278         except TypeError:
 279           pass
 280         category_name = unicodedata.normalize('NFKD', category_name)
 281         if six.PY2:
 282           category_name = category_name.encode('ascii', 'ignore')
 283         self._metric_names.append(
 284             self._metric_prefix + 'PerformanceByCategory/AP@{}IOU/{}'.format(
 285                 self._matching_iou_threshold, category_name))
 286         if self._evaluate_corlocs:
 287           self._metric_names.append(
 288               self._metric_prefix +
 289               'PerformanceByCategory/CorLoc@{}IOU/{}'.format(
 290                   self._matching_iou_threshold, category_name))

 292   def add_single_ground_truth_image_info(self, image_id, groundtruth_dict):
 293     """Adds groundtruth for a single image to be used for evaluation.
 294 
 295     Args:
 296       image_id: A unique string/integer identifier for the image.
 297       groundtruth_dict: A dictionary containing -
 298         standard_fields.InputDataFields.groundtruth_boxes: float32 numpy array
 299           of shape [num_boxes, 4] containing `num_boxes` groundtruth boxes of
 300           the format [ymin, xmin, ymax, xmax] in absolute image coordinates.
 301         standard_fields.InputDataFields.groundtruth_classes: integer numpy array
 302           of shape [num_boxes] containing 1-indexed groundtruth classes for the
 303           boxes.
 304         standard_fields.InputDataFields.groundtruth_difficult: Optional length M
 305           numpy boolean array denoting whether a ground truth box is a difficult
 306           instance or not. This field is optional to support the case that no
 307           boxes are difficult.
 308         standard_fields.InputDataFields.groundtruth_instance_masks: Optional
 309           numpy array of shape [num_boxes, height, width] with values in {0, 1}.
 310 
 311     Raises:
 312       ValueError: On adding groundtruth for an image more than once. Will also
 313         raise error if instance masks are not in groundtruth dictionary.
 314     """
 315     if image_id in self._image_ids:
 316       raise ValueError('Image with id {} already added.'.format(image_id))
 317 
 318     groundtruth_classes = (
 319         groundtruth_dict[standard_fields.InputDataFields.groundtruth_classes] -
 320         self._label_id_offset)
 321     # If the key is not present in the groundtruth_dict or the array is empty
 322     # (unless there are no annotations for the groundtruth on this image)
 323     # use values from the dictionary or insert None otherwise.
 324     if (standard_fields.InputDataFields.groundtruth_difficult in six.viewkeys(
 325         groundtruth_dict) and
 326         (groundtruth_dict[standard_fields.InputDataFields.groundtruth_difficult]
 327          .size or not groundtruth_classes.size)):
 328       groundtruth_difficult = groundtruth_dict[
 329           standard_fields.InputDataFields.groundtruth_difficult]
 330     else:
 331       groundtruth_difficult = None
 332       if not len(self._image_ids) % 1000:
 333         logging.warning(
 334             'image %s does not have groundtruth difficult flag specified',
 335             image_id)
 336     groundtruth_masks = None
 337     if self._evaluate_masks:
 338       if (standard_fields.InputDataFields.groundtruth_instance_masks not in
 339           groundtruth_dict):
 340         raise ValueError('Instance masks not in groundtruth dictionary.')
 341       groundtruth_masks = groundtruth_dict[
 342           standard_fields.InputDataFields.groundtruth_instance_masks]
 343     self._evaluation.add_single_ground_truth_image_info(
 344         image_key=image_id,
 345         groundtruth_boxes=groundtruth_dict[
 346             standard_fields.InputDataFields.groundtruth_boxes],
 347         groundtruth_class_labels=groundtruth_classes,
 348         groundtruth_is_difficult_list=groundtruth_difficult,
 349         groundtruth_masks=groundtruth_masks)
 350     self._image_ids.update([image_id])
 351 
 352   def add_single_detected_image_info(self, image_id, detections_dict):
 353     """Adds detections for a single image to be used for evaluation.
 354 
 355     Args:
 356       image_id: A unique string/integer identifier for the image.
 357       detections_dict: A dictionary containing -
 358         standard_fields.DetectionResultFields.detection_boxes: float32 numpy
 359           array of shape [num_boxes, 4] containing `num_boxes` detection boxes
 360           of the format [ymin, xmin, ymax, xmax] in absolute image coordinates.
 361         standard_fields.DetectionResultFields.detection_scores: float32 numpy
 362           array of shape [num_boxes] containing detection scores for the boxes.
 363         standard_fields.DetectionResultFields.detection_classes: integer numpy
 364           array of shape [num_boxes] containing 1-indexed detection classes for
 365           the boxes.
 366         standard_fields.DetectionResultFields.detection_masks: uint8 numpy array
 367           of shape [num_boxes, height, width] containing `num_boxes` masks of
 368           values ranging between 0 and 1.
 369 
 370     Raises:
 371       ValueError: If detection masks are not in detections dictionary.
 372     """
 373     detection_classes = (
 374         detections_dict[standard_fields.DetectionResultFields.detection_classes]
 375         - self._label_id_offset)
 376     detection_masks = None
 377     if self._evaluate_masks:
 378       if (standard_fields.DetectionResultFields.detection_masks not in
 379           detections_dict):
 380         raise ValueError('Detection masks not in detections dictionary.')
 381       detection_masks = detections_dict[
 382           standard_fields.DetectionResultFields.detection_masks]
 383     self._evaluation.add_single_detected_image_info(
 384         image_key=image_id,
 385         detected_boxes=detections_dict[
 386             standard_fields.DetectionResultFields.detection_boxes],
 387         detected_scores=detections_dict[
 388             standard_fields.DetectionResultFields.detection_scores],
 389         detected_class_labels=detection_classes,
 390         detected_masks=detection_masks)

 392   def evaluate(self):
 393     """Compute evaluation result.
 394 
 395     Returns:
 396       A dictionary of metrics with the following fields -
 397 
 398       1. summary_metrics:
 399         '<prefix if not empty>_Precision/mAP@<matching_iou_threshold>IOU': mean
 400         average precision at the specified IOU threshold.
 401 
 402       2. per_category_ap: category specific results with keys of the form
 403         '<prefix if not empty>_PerformanceByCategory/
 404         mAP@<matching_iou_threshold>IOU/category'.
 405     """
 406     (per_class_ap, mean_ap, per_class_precision, per_class_recall,
 407      per_class_corloc, mean_corloc) = (
 408          self._evaluation.evaluate())
 409     pascal_metrics = {self._metric_names[0]: mean_ap}
 410     if self._evaluate_corlocs:
 411       pascal_metrics[self._metric_names[1]] = mean_corloc
 412     category_index = label_map_util.create_category_index(self._categories)
 413     for idx in range(per_class_ap.size):
 414       if idx + self._label_id_offset in category_index:
 415         category_name = category_index[idx + self._label_id_offset]['name']
 416         try:
 417           category_name = six.text_type(category_name, 'utf-8')
 418         except TypeError:
 419           pass
 420         category_name = unicodedata.normalize('NFKD', category_name)
 421         if six.PY2:
 422           category_name = category_name.encode('ascii', 'ignore')
 423         display_name = (
 424             self._metric_prefix + 'PerformanceByCategory/AP@{}IOU/{}'.format(
 425                 self._matching_iou_threshold, category_name))
 426         pascal_metrics[display_name] = per_class_ap[idx]
 427 
 428         # Optionally add precision and recall values
 429         if self._evaluate_precision_recall:
 430           display_name = (
 431               self._metric_prefix +
 432               'PerformanceByCategory/Precision@{}IOU/{}'.format(
 433                   self._matching_iou_threshold, category_name))
 434           pascal_metrics[display_name] = per_class_precision[idx]
 435           display_name = (
 436               self._metric_prefix +
 437               'PerformanceByCategory/Recall@{}IOU/{}'.format(
 438                   self._matching_iou_threshold, category_name))
 439           pascal_metrics[display_name] = per_class_recall[idx]
 440 
 441         # Optionally add CorLoc metrics.classes
 442         if self._evaluate_corlocs:
 443           display_name = (
 444               self._metric_prefix +
 445               'PerformanceByCategory/CorLoc@{}IOU/{}'.format(
 446                   self._matching_iou_threshold, category_name))
 447           pascal_metrics[display_name] = per_class_corloc[idx]
 448 
 449     return pascal_metrics

 456         use_weighted_mean_ap=self._use_weighted_mean_ap,

 460   def add_eval_dict(self, eval_dict):
 461     """Observes an evaluation result dict for a single example.
 462 
 463     When executing eagerly, once all observations have been observed by this
 464     method you can use `.evaluate()` to get the final metrics.
 465 
 466     When using `tf.estimator.Estimator` for evaluation this function is used by
 467     `get_estimator_eval_metric_ops()` to construct the metric update op.
 468 
 469     Args:
 470       eval_dict: A dictionary that holds tensors for evaluating an object
 471         detection model, returned from
 472         eval_util.result_dict_for_single_example().
 473 
 474     Returns:
 475       None when executing eagerly, or an update_op that can be used to update
 476       the eval metrics in `tf.estimator.EstimatorSpec`.
 477     """
 478     # remove unexpected fields
 479     eval_dict_filtered = dict()
 480     for key, value in eval_dict.items():
 481       if key in self._expected_keys:
 482         eval_dict_filtered[key] = value
 483 
 484     eval_dict_keys = list(eval_dict_filtered.keys())
 485 
 486     def update_op(image_id, *eval_dict_batched_as_list):
 487       """Update operation that adds batch of images to ObjectDetectionEvaluator.
 488 
 489       Args:
 490         image_id: image id (single id or an array)
 491         *eval_dict_batched_as_list: the values of the dictionary of tensors.
 492       """
 493       if np.isscalar(image_id):
 494         single_example_dict = dict(
 495             zip(eval_dict_keys, eval_dict_batched_as_list))
 496         self.add_single_ground_truth_image_info(image_id, single_example_dict)
 497         self.add_single_detected_image_info(image_id, single_example_dict)
 498       else:
 499         for unzipped_tuple in zip(*eval_dict_batched_as_list):
 500           single_example_dict = dict(zip(eval_dict_keys, unzipped_tuple))
 501           image_id = single_example_dict[standard_fields.InputDataFields.key]
 502           self.add_single_ground_truth_image_info(image_id, single_example_dict)
 503           self.add_single_detected_image_info(image_id, single_example_dict)
 504 
 505     args = [eval_dict_filtered[standard_fields.InputDataFields.key]]
 506     args.extend(six.itervalues(eval_dict_filtered))
 507     return tf.py_func(update_op, args, [])
 508 
 509   def get_estimator_eval_metric_ops(self, eval_dict):
 510     """Returns dict of metrics to use with `tf.estimator.EstimatorSpec`.
 511 
 512     Note that this must only be implemented if performing evaluation with a
 513     `tf.estimator.Estimator`.
 514 
 515     Args:
 516       eval_dict: A dictionary that holds tensors for evaluating an object
 517         detection model, returned from
 518         eval_util.result_dict_for_single_example(). It must contain
 519         standard_fields.InputDataFields.key.
 520 
 521     Returns:
 522       A dictionary of metric names to tuple of value_op and update_op that can
 523       be used as eval metric ops in `tf.estimator.EstimatorSpec`.
 524     """
 525     update_op = self.add_eval_dict(eval_dict)
 526 
 527     def first_value_func():
 528       self._metrics = self.evaluate()
 529       self.clear()
 530       return np.float32(self._metrics[self._metric_names[0]])
 531 
 532     def value_func_factory(metric_name):
 533 
 534       def value_func():
 535         return np.float32(self._metrics[metric_name])
 536 
 537       return value_func
 538 
 539     # Ensure that the metrics are only evaluated once.
 540     first_value_op = tf.py_func(first_value_func, [], tf.float32)
 541     eval_metric_ops = {self._metric_names[0]: (first_value_op, update_op)}
 542     with tf.control_dependencies([first_value_op]):
 543       for metric_name in self._metric_names[1:]:
 544         eval_metric_ops[metric_name] = (tf.py_func(
 545             value_func_factory(metric_name), [], np.float32), update_op)
 546     return eval_metric_ops

 639 class OpenImagesDetectionEvaluator(ObjectDetectionEvaluator):
 640   """A class to evaluate detections using Open Images V2 metrics.
 641 
 642     Open Images V2 introduce group_of type of bounding boxes and this metric
 643     handles those boxes appropriately.
 644   """
 645 
 646   def __init__(self,
 647                categories,
 648                matching_iou_threshold=0.5,
 649                evaluate_masks=False,
 650                evaluate_corlocs=False,
 651                metric_prefix='OpenImagesV2',
 652                group_of_weight=0.0):
 653     """Constructor.
 654 
 655     Args:
 656       categories: A list of dicts, each of which has the following keys -
 657         'id': (required) an integer id uniquely identifying this category.
 658         'name': (required) string representing category name e.g., 'cat', 'dog'.
 659       matching_iou_threshold: IOU threshold to use for matching groundtruth
 660         boxes to detection boxes.
 661       evaluate_masks: if True, evaluator evaluates masks.
 662       evaluate_corlocs: if True, additionally evaluates and returns CorLoc.
 663       metric_prefix: Prefix name of the metric.
 664       group_of_weight: Weight of the group-of bounding box. If set to 0 (default
 665         for Open Images V2 detection protocol), detections of the correct class
 666         within a group-of box are ignored. If weight is > 0, then if at least
 667         one detection falls within a group-of box with matching_iou_threshold,
 668         weight group_of_weight is added to true positives. Consequently, if no
 669         detection falls within a group-of box, weight group_of_weight is added
 670         to false negatives.
 671     """
 672 
 673     super(OpenImagesDetectionEvaluator, self).__init__(
 674         categories,
 675         matching_iou_threshold,
 676         evaluate_corlocs,
 677         metric_prefix=metric_prefix,
 678         group_of_weight=group_of_weight,
 679         evaluate_masks=evaluate_masks)
 680     self._expected_keys = set([
 681         standard_fields.InputDataFields.key,
 682         standard_fields.InputDataFields.groundtruth_boxes,
 683         standard_fields.InputDataFields.groundtruth_classes,
 684         standard_fields.InputDataFields.groundtruth_group_of,
 685         standard_fields.DetectionResultFields.detection_boxes,
 686         standard_fields.DetectionResultFields.detection_scores,
 687         standard_fields.DetectionResultFields.detection_classes,
 688     ])
 689     if evaluate_masks:
 690       self._expected_keys.add(
 691           standard_fields.InputDataFields.groundtruth_instance_masks)
 692       self._expected_keys.add(
 693           standard_fields.DetectionResultFields.detection_masks)
 694 
 695   def add_single_ground_truth_image_info(self, image_id, groundtruth_dict):
 696     """Adds groundtruth for a single image to be used for evaluation.
 697 
 698     Args:
 699       image_id: A unique string/integer identifier for the image.
 700       groundtruth_dict: A dictionary containing -
 701         standard_fields.InputDataFields.groundtruth_boxes: float32 numpy array
 702           of shape [num_boxes, 4] containing `num_boxes` groundtruth boxes of
 703           the format [ymin, xmin, ymax, xmax] in absolute image coordinates.
 704         standard_fields.InputDataFields.groundtruth_classes: integer numpy array
 705           of shape [num_boxes] containing 1-indexed groundtruth classes for the
 706           boxes.
 707         standard_fields.InputDataFields.groundtruth_group_of: Optional length M
 708           numpy boolean array denoting whether a groundtruth box contains a
 709           group of instances.
 710 
 711     Raises:
 712       ValueError: On adding groundtruth for an image more than once.
 713     """
 714     if image_id in self._image_ids:
 715       raise ValueError('Image with id {} already added.'.format(image_id))
 716 
 717     groundtruth_classes = (
 718         groundtruth_dict[standard_fields.InputDataFields.groundtruth_classes] -
 719         self._label_id_offset)
 720     # If the key is not present in the groundtruth_dict or the array is empty
 721     # (unless there are no annotations for the groundtruth on this image)
 722     # use values from the dictionary or insert None otherwise.
 723     if (standard_fields.InputDataFields.groundtruth_group_of in six.viewkeys(
 724         groundtruth_dict) and
 725         (groundtruth_dict[standard_fields.InputDataFields.groundtruth_group_of]
 726          .size or not groundtruth_classes.size)):
 727       groundtruth_group_of = groundtruth_dict[
 728           standard_fields.InputDataFields.groundtruth_group_of]
 729     else:
 730       groundtruth_group_of = None
 731       if not len(self._image_ids) % 1000:
 732         logging.warning(
 733             'image %s does not have groundtruth group_of flag specified',
 734             image_id)
 735     if self._evaluate_masks:
 736       groundtruth_masks = groundtruth_dict[
 737           standard_fields.InputDataFields.groundtruth_instance_masks]
 738     else:
 739       groundtruth_masks = None
 740 
 741     self._evaluation.add_single_ground_truth_image_info(
 742         image_id,
 743         groundtruth_dict[standard_fields.InputDataFields.groundtruth_boxes],
 744         groundtruth_classes,
 745         groundtruth_is_difficult_list=None,
 746         groundtruth_is_group_of_list=groundtruth_group_of,
 747         groundtruth_masks=groundtruth_masks)
 748     self._image_ids.update([image_id])

 751 class OpenImagesChallengeEvaluator(OpenImagesDetectionEvaluator):
 752   """A class implements Open Images Challenge metrics.
 753 
 754     Both Detection and Instance Segmentation evaluation metrics are implemented.
 755 
 756     Open Images Challenge Detection metric has two major changes in comparison
 757     with Open Images V2 detection metric:
 758     - a custom weight might be specified for detecting an object contained in
 759     a group-of box.
 760     - verified image-level labels should be explicitelly provided for
 761     evaluation: in case in image has neither positive nor negative image level
 762     label of class c, all detections of this class on this image will be
 763     ignored.
 764 
 765     Open Images Challenge Instance Segmentation metric allows to measure per
 766     formance of models in case of incomplete annotations: some instances are
 767     annotations only on box level and some - on image-level. In addition,
 768     image-level labels are taken into account as in detection metric.
 769 
 770     Open Images Challenge Detection metric default parameters:
 771     evaluate_masks = False
 772     group_of_weight = 1.0
 773 
 774 
 775     Open Images Challenge Instance Segmentation metric default parameters:
 776     evaluate_masks = True
 777     (group_of_weight will not matter)
 778   """
 779 
 780   def __init__(self,
 781                categories,
 782                evaluate_masks=False,
 783                matching_iou_threshold=0.5,
 784                evaluate_corlocs=False,
 785                group_of_weight=1.0):
 786     """Constructor.
 787 
 788     Args:
 789       categories: A list of dicts, each of which has the following keys -
 790         'id': (required) an integer id uniquely identifying this category.
 791         'name': (required) string representing category name e.g., 'cat', 'dog'.
 792       evaluate_masks: set to true for instance segmentation metric and to false
 793         for detection metric.
 794       matching_iou_threshold: IOU threshold to use for matching groundtruth
 795         boxes to detection boxes.
 796       evaluate_corlocs: if True, additionally evaluates and returns CorLoc.
 797       group_of_weight: Weight of group-of boxes. If set to 0, detections of the
 798         correct class within a group-of box are ignored. If weight is > 0, then
 799         if at least one detection falls within a group-of box with
 800         matching_iou_threshold, weight group_of_weight is added to true
 801         positives. Consequently, if no detection falls within a group-of box,
 802         weight group_of_weight is added to false negatives.
 803     """
 804     if not evaluate_masks:
 805       metrics_prefix = 'OpenImagesDetectionChallenge'
 806     else:
 807       metrics_prefix = 'OpenImagesInstanceSegmentationChallenge'
 808 
 809     super(OpenImagesChallengeEvaluator, self).__init__(
 810         categories,
 811         matching_iou_threshold,
 812         evaluate_masks=evaluate_masks,
 813         evaluate_corlocs=evaluate_corlocs,
 814         group_of_weight=group_of_weight,
 815         metric_prefix=metrics_prefix)
 816 
 817     self._evaluatable_labels = {}
 818     # Only one of the two has to be provided, but both options are given
 819     # for compatibility with previous codebase.
 820     self._expected_keys.update([
 821         standard_fields.InputDataFields.groundtruth_image_classes,
 822         standard_fields.InputDataFields.groundtruth_labeled_classes])
 823 
 824   def add_single_ground_truth_image_info(self, image_id, groundtruth_dict):
 825     """Adds groundtruth for a single image to be used for evaluation.
 826 
 827     Args:
 828       image_id: A unique string/integer identifier for the image.
 829       groundtruth_dict: A dictionary containing -
 830         standard_fields.InputDataFields.groundtruth_boxes: float32 numpy array
 831           of shape [num_boxes, 4] containing `num_boxes` groundtruth boxes of
 832           the format [ymin, xmin, ymax, xmax] in absolute image coordinates.
 833         standard_fields.InputDataFields.groundtruth_classes: integer numpy array
 834           of shape [num_boxes] containing 1-indexed groundtruth classes for the
 835           boxes.
 836         standard_fields.InputDataFields.groundtruth_image_classes: integer 1D
 837           numpy array containing all classes for which labels are verified.
 838         standard_fields.InputDataFields.groundtruth_group_of: Optional length M
 839           numpy boolean array denoting whether a groundtruth box contains a
 840           group of instances.
 841 
 842     Raises:
 843       ValueError: On adding groundtruth for an image more than once.
 844     """
 845     super(OpenImagesChallengeEvaluator,
 846           self).add_single_ground_truth_image_info(image_id, groundtruth_dict)
 847     input_fields = standard_fields.InputDataFields
 848     groundtruth_classes = (
 849         groundtruth_dict[input_fields.groundtruth_classes] -
 850         self._label_id_offset)
 851     image_classes = np.array([], dtype=int)
 852     if input_fields.groundtruth_image_classes in groundtruth_dict:
 853       image_classes = groundtruth_dict[input_fields.groundtruth_image_classes]
 854     elif input_fields.groundtruth_labeled_classes in groundtruth_dict:
 855       image_classes = groundtruth_dict[input_fields.groundtruth_labeled_classes]
 856     image_classes -= self._label_id_offset
 857     self._evaluatable_labels[image_id] = np.unique(
 858         np.concatenate((image_classes, groundtruth_classes)))
 859 
 860   def add_single_detected_image_info(self, image_id, detections_dict):
 861     """Adds detections for a single image to be used for evaluation.
 862 
 863     Args:
 864       image_id: A unique string/integer identifier for the image.
 865       detections_dict: A dictionary containing -
 866         standard_fields.DetectionResultFields.detection_boxes: float32 numpy
 867           array of shape [num_boxes, 4] containing `num_boxes` detection boxes
 868           of the format [ymin, xmin, ymax, xmax] in absolute image coordinates.
 869         standard_fields.DetectionResultFields.detection_scores: float32 numpy
 870           array of shape [num_boxes] containing detection scores for the boxes.
 871         standard_fields.DetectionResultFields.detection_classes: integer numpy
 872           array of shape [num_boxes] containing 1-indexed detection classes for
 873           the boxes.
 874 
 875     Raises:
 876       ValueError: If detection masks are not in detections dictionary.
 877     """
 878     if image_id not in self._image_ids:
 879       # Since for the correct work of evaluator it is assumed that groundtruth
 880       # is inserted first we make sure to break the code if is it not the case.
 881       self._image_ids.update([image_id])
 882       self._evaluatable_labels[image_id] = np.array([])
 883 
 884     detection_classes = (
 885         detections_dict[standard_fields.DetectionResultFields.detection_classes]
 886         - self._label_id_offset)
 887     allowed_classes = np.where(
 888         np.isin(detection_classes, self._evaluatable_labels[image_id]))
 889     detection_classes = detection_classes[allowed_classes]
 890     detected_boxes = detections_dict[
 891         standard_fields.DetectionResultFields.detection_boxes][allowed_classes]
 892     detected_scores = detections_dict[
 893         standard_fields.DetectionResultFields.detection_scores][allowed_classes]
 894 
 895     if self._evaluate_masks:
 896       detection_masks = detections_dict[standard_fields.DetectionResultFields
 897                                         .detection_masks][allowed_classes]
 898     else:
 899       detection_masks = None
 900     self._evaluation.add_single_detected_image_info(
 901         image_key=image_id,
 902         detected_boxes=detected_boxes,
 903         detected_scores=detected_scores,
 904         detected_class_labels=detection_classes,
 905         detected_masks=detection_masks)
 906 
 907   def clear(self):
 908     """Clears stored data."""
 909 
 910     super(OpenImagesChallengeEvaluator, self).clear()
 911     self._evaluatable_labels.clear()

 982 class ObjectDetectionEvaluation(object):
 983   """Internal implementation of Pascal object detection metrics."""
 984 
 985   def __init__(self,
 986                num_groundtruth_classes,
 987                matching_iou_threshold=0.5,
 988                nms_iou_threshold=1.0,
 989                nms_max_output_boxes=10000,
 990                recall_lower_bound=0.0,
 991                recall_upper_bound=1.0,
 992                use_weighted_mean_ap=False,
 993                label_id_offset=0,
 994                group_of_weight=0.0,
 995                per_image_eval_class=per_image_evaluation.PerImageEvaluation):
 996     """Constructor.
 997 
 998     Args:
 999       num_groundtruth_classes: Number of ground-truth classes.
1000       matching_iou_threshold: IOU threshold used for matching detected boxes to
1001         ground-truth boxes.
1002       nms_iou_threshold: IOU threshold used for non-maximum suppression.
1003       nms_max_output_boxes: Maximum number of boxes returned by non-maximum
1004         suppression.
1005       recall_lower_bound: lower bound of recall operating area
1006       recall_upper_bound: upper bound of recall operating area
1007       use_weighted_mean_ap: (optional) boolean which determines if the mean
1008         average precision is computed directly from the scores and tp_fp_labels
1009         of all classes.
1010       label_id_offset: The label id offset.
1011       group_of_weight: Weight of group-of boxes.If set to 0, detections of the
1012         correct class within a group-of box are ignored. If weight is > 0, then
1013         if at least one detection falls within a group-of box with
1014         matching_iou_threshold, weight group_of_weight is added to true
1015         positives. Consequently, if no detection falls within a group-of box,
1016         weight group_of_weight is added to false negatives.
1017       per_image_eval_class: The class that contains functions for computing per
1018         image metrics.
1019 
1020     Raises:
1021       ValueError: if num_groundtruth_classes is smaller than 1.
1022     """
1023     if num_groundtruth_classes < 1:
1024       raise ValueError('Need at least 1 groundtruth class for evaluation.')
1025 
1026     self.per_image_eval = per_image_eval_class(
1027         num_groundtruth_classes=num_groundtruth_classes,
1028         matching_iou_threshold=matching_iou_threshold,
1029         nms_iou_threshold=nms_iou_threshold,
1030         nms_max_output_boxes=nms_max_output_boxes,
1031         group_of_weight=group_of_weight)
1032     self.recall_lower_bound = recall_lower_bound
1033     self.recall_upper_bound = recall_upper_bound
1034     self.group_of_weight = group_of_weight
1035     self.num_class = num_groundtruth_classes
1036     self.use_weighted_mean_ap = use_weighted_mean_ap
1037     self.label_id_offset = label_id_offset
1038 
1039     self.groundtruth_boxes = {}
1040     self.groundtruth_class_labels = {}
1041     self.groundtruth_masks = {}
1042     self.groundtruth_is_difficult_list = {}
1043     self.groundtruth_is_group_of_list = {}
1044     self.num_gt_instances_per_class = np.zeros(self.num_class, dtype=float)
1045     self.num_gt_imgs_per_class = np.zeros(self.num_class, dtype=int)
1046 
1047     self._initialize_detections()
1048 
1049   def _initialize_detections(self):
1050     """Initializes internal data structures."""
1051     self.detection_keys = set()
1052     self.scores_per_class = [[] for _ in range(self.num_class)]
1053     self.tp_fp_labels_per_class = [[] for _ in range(self.num_class)]
1054     self.num_images_correctly_detected_per_class = np.zeros(self.num_class)
1055     self.average_precision_per_class = np.empty(self.num_class, dtype=float)
1056     self.average_precision_per_class.fill(np.nan)
1057     self.precisions_per_class = [np.nan] * self.num_class
1058     self.recalls_per_class = [np.nan] * self.num_class
1059     self.sum_tp_class = [np.nan] * self.num_class
1060 
1061     self.corloc_per_class = np.ones(self.num_class, dtype=float)
1062 
1063   def clear_detections(self):
1064     self._initialize_detections()
1065 
1066   def get_internal_state(self):
1067     """Returns internal state of the evaluation.
1068 
1069     NOTE: that only evaluation results will be returned
1070     (e.g. no raw predictions or groundtruth).
1071     Returns:
1072       internal state of the evaluation.
1073     """
1074     return ObjectDetectionEvaluationState(
1075         self.num_gt_instances_per_class, self.scores_per_class,
1076         self.tp_fp_labels_per_class, self.num_gt_imgs_per_class,
1077         self.num_images_correctly_detected_per_class)
1078 
1079   def merge_internal_state(self, state_tuple):
1080     """Merges internal state of the evaluation with the current state.
1081 
1082     Args:
1083       state_tuple: state tuple representing evaluation state: should be of type
1084         ObjectDetectionEvaluationState.
1085     """
1086     (num_gt_instances_per_class, scores_per_class, tp_fp_labels_per_class,
1087      num_gt_imgs_per_class, num_images_correctly_detected_per_class) = (
1088          state_tuple)
1089     assert self.num_class == len(num_gt_instances_per_class)
1090     assert self.num_class == len(scores_per_class)
1091     assert self.num_class == len(tp_fp_labels_per_class)
1092     for i in range(self.num_class):
1093       self.scores_per_class[i].extend(scores_per_class[i])
1094       self.tp_fp_labels_per_class[i].extend(tp_fp_labels_per_class[i])
1095       self.num_gt_instances_per_class[i] += num_gt_instances_per_class[i]
1096       self.num_gt_imgs_per_class[i] += num_gt_imgs_per_class[i]
1097       self.num_images_correctly_detected_per_class[
1098           i] += num_images_correctly_detected_per_class[i]

1105                                          groundtruth_is_group_of_list=None,
1106                                          groundtruth_masks=None):
1107     """Adds groundtruth for a single image to be used for evaluation.
1108 
1109     Args:
1110       image_key: A unique string/integer identifier for the image.
1111       groundtruth_boxes: float32 numpy array of shape [num_boxes, 4] containing
1112         `num_boxes` groundtruth boxes of the format [ymin, xmin, ymax, xmax] in
1113         absolute image coordinates.
1114       groundtruth_class_labels: integer numpy array of shape [num_boxes]
1115         containing 0-indexed groundtruth classes for the boxes.
1116       groundtruth_is_difficult_list: A length M numpy boolean array denoting
1117         whether a ground truth box is a difficult instance or not. To support
1118         the case that no boxes are difficult, it is by default set as None.
1119       groundtruth_is_group_of_list: A length M numpy boolean array denoting
1120         whether a ground truth box is a group-of box or not. To support the case
1121         that no boxes are groups-of, it is by default set as None.
1122       groundtruth_masks: uint8 numpy array of shape [num_boxes, height, width]
1123         containing `num_boxes` groundtruth masks. The mask values range from 0
1124         to 1.
1125     """
1126     if image_key in self.groundtruth_boxes:
1127       logging.warning(
1128           'image %s has already been added to the ground truth database.',
1129           image_key)
1130       return
1131 
1132     self.groundtruth_boxes[image_key] = groundtruth_boxes
1133     self.groundtruth_class_labels[image_key] = groundtruth_class_labels
1134     self.groundtruth_masks[image_key] = groundtruth_masks
1135     if groundtruth_is_difficult_list is None:
1136       num_boxes = groundtruth_boxes.shape[0]
1137       groundtruth_is_difficult_list = np.zeros(num_boxes, dtype=bool)
1138     self.groundtruth_is_difficult_list[
1139         image_key] = groundtruth_is_difficult_list.astype(dtype=bool)
1140     if groundtruth_is_group_of_list is None:
1141       num_boxes = groundtruth_boxes.shape[0]
1142       groundtruth_is_group_of_list = np.zeros(num_boxes, dtype=bool)
1143     if groundtruth_masks is None:
1144       num_boxes = groundtruth_boxes.shape[0]
1145       mask_presence_indicator = np.zeros(num_boxes, dtype=bool)
1146     else:
1147       mask_presence_indicator = (np.sum(groundtruth_masks,
1148                                         axis=(1, 2)) == 0).astype(dtype=bool)
1149 
1150     self.groundtruth_is_group_of_list[
1151         image_key] = groundtruth_is_group_of_list.astype(dtype=bool)
1152 
1153     self._update_ground_truth_statistics(
1154         groundtruth_class_labels,
1155         groundtruth_is_difficult_list.astype(dtype=bool)
1156         | mask_presence_indicator,  # ignore boxes without masks
1157         groundtruth_is_group_of_list.astype(dtype=bool))

1159   def add_single_detected_image_info(self,
1160                                      image_key,
1161                                      detected_boxes,
1162                                      detected_scores,
1163                                      detected_class_labels,
1164                                      detected_masks=None):
1165     """Adds detections for a single image to be used for evaluation.
1166 
1167     Args:
1168       image_key: A unique string/integer identifier for the image.
1169       detected_boxes: float32 numpy array of shape [num_boxes, 4] containing
1170         `num_boxes` detection boxes of the format [ymin, xmin, ymax, xmax] in
1171         absolute image coordinates.
1172       detected_scores: float32 numpy array of shape [num_boxes] containing
1173         detection scores for the boxes.
1174       detected_class_labels: integer numpy array of shape [num_boxes] containing
1175         0-indexed detection classes for the boxes.
1176       detected_masks: np.uint8 numpy array of shape [num_boxes, height, width]
1177         containing `num_boxes` detection masks with values ranging between 0 and
1178         1.
1179 
1180     Raises:
1181       ValueError: if the number of boxes, scores and class labels differ in
1182         length.
1183     """
1184     if (len(detected_boxes) != len(detected_scores) or
1185         len(detected_boxes) != len(detected_class_labels)):
1186       raise ValueError(
1187           'detected_boxes, detected_scores and '
1188           'detected_class_labels should all have same lengths. Got'
1189           '[%d, %d, %d]' % len(detected_boxes), len(detected_scores),
1190           len(detected_class_labels))
1191 
1192     if image_key in self.detection_keys:
1193       logging.warning(
1194           'image %s has already been added to the detection result database',
1195           image_key)
1196       return
1197 
1198     self.detection_keys.add(image_key)
1199     if image_key in self.groundtruth_boxes:
1200       groundtruth_boxes = self.groundtruth_boxes[image_key]
1201       groundtruth_class_labels = self.groundtruth_class_labels[image_key]
1202       # Masks are popped instead of look up. The reason is that we do not want
1203       # to keep all masks in memory which can cause memory overflow.
1204       groundtruth_masks = self.groundtruth_masks.pop(image_key)
1205       groundtruth_is_difficult_list = self.groundtruth_is_difficult_list[
1206           image_key]
1207       groundtruth_is_group_of_list = self.groundtruth_is_group_of_list[
1208           image_key]
1209     else:
1210       groundtruth_boxes = np.empty(shape=[0, 4], dtype=float)
1211       groundtruth_class_labels = np.array([], dtype=int)
1212       if detected_masks is None:
1213         groundtruth_masks = None
1214       else:
1215         groundtruth_masks = np.empty(shape=[0, 1, 1], dtype=float)
1216       groundtruth_is_difficult_list = np.array([], dtype=bool)
1217       groundtruth_is_group_of_list = np.array([], dtype=bool)
1218     scores, tp_fp_labels, is_class_correctly_detected_in_image = (
1219         self.per_image_eval.compute_object_detection_metrics(
1220             detected_boxes=detected_boxes,
1221             detected_scores=detected_scores,
1222             detected_class_labels=detected_class_labels,
1223             groundtruth_boxes=groundtruth_boxes,
1224             groundtruth_class_labels=groundtruth_class_labels,
1225             groundtruth_is_difficult_list=groundtruth_is_difficult_list,
1226             groundtruth_is_group_of_list=groundtruth_is_group_of_list,
1227             detected_masks=detected_masks,
1228             groundtruth_masks=groundtruth_masks))
1229 
1230     for i in range(self.num_class):
1231       if scores[i].shape[0] > 0:
1232         self.scores_per_class[i].append(scores[i])
1233         self.tp_fp_labels_per_class[i].append(tp_fp_labels[i])
1234     (self.num_images_correctly_detected_per_class
1235     ) += is_class_correctly_detected_in_image

1237   def _update_ground_truth_statistics(self, groundtruth_class_labels,
1238                                       groundtruth_is_difficult_list,
1239                                       groundtruth_is_group_of_list):
1240     """Update grouth truth statitistics.
1241 
1242     1. Difficult boxes are ignored when counting the number of ground truth
1243     instances as done in Pascal VOC devkit.
1244     2. Difficult boxes are treated as normal boxes when computing CorLoc related
1245     statitistics.
1246 
1247     Args:
1248       groundtruth_class_labels: An integer numpy array of length M, representing
1249         M class labels of object instances in ground truth
1250       groundtruth_is_difficult_list: A boolean numpy array of length M denoting
1251         whether a ground truth box is a difficult instance or not
1252       groundtruth_is_group_of_list: A boolean numpy array of length M denoting
1253         whether a ground truth box is a group-of box or not
1254     """
1255     for class_index in range(self.num_class):
1256       num_gt_instances = np.sum(groundtruth_class_labels[
1257           ~groundtruth_is_difficult_list
1258           & ~groundtruth_is_group_of_list] == class_index)
1259       num_groupof_gt_instances = self.group_of_weight * np.sum(
1260           groundtruth_class_labels[
1261               groundtruth_is_group_of_list
1262               & ~groundtruth_is_difficult_list] == class_index)
1263       self.num_gt_instances_per_class[
1264           class_index] += num_gt_instances + num_groupof_gt_instances
1265       if np.any(groundtruth_class_labels == class_index):
1266         self.num_gt_imgs_per_class[class_index] += 1
1267 
1268   def evaluate(self):
1269     """Compute evaluation result.
1270 
1271     Returns:
1272       A named tuple with the following fields -
1273         average_precision: float numpy array of average precision for
1274             each class.
1275         mean_ap: mean average precision of all classes, float scalar
1276         precisions: List of precisions, each precision is a float numpy
1277             array
1278         recalls: List of recalls, each recall is a float numpy array
1279         corloc: numpy float array
1280         mean_corloc: Mean CorLoc score for each class, float scalar
1281     """
1282     if (self.num_gt_instances_per_class == 0).any():
1283       logging.warning(
1284           'The following classes have no ground truth examples: %s',
1285           np.squeeze(np.argwhere(self.num_gt_instances_per_class == 0)) +
1286           self.label_id_offset)
1287 
1288     if self.use_weighted_mean_ap:
1289       all_scores = np.array([], dtype=float)
1290       all_tp_fp_labels = np.array([], dtype=bool)
1291     for class_index in range(self.num_class):
1292       if self.num_gt_instances_per_class[class_index] == 0:
1293         continue
1294       if not self.scores_per_class[class_index]:
1295         scores = np.array([], dtype=float)
1296         tp_fp_labels = np.array([], dtype=float)
1297       else:
1298         scores = np.concatenate(self.scores_per_class[class_index])
1299         tp_fp_labels = np.concatenate(self.tp_fp_labels_per_class[class_index])
1300       if self.use_weighted_mean_ap:
1301         all_scores = np.append(all_scores, scores)
1302         all_tp_fp_labels = np.append(all_tp_fp_labels, tp_fp_labels)
1303       precision, recall = metrics.compute_precision_recall(
1304           scores, tp_fp_labels, self.num_gt_instances_per_class[class_index])
1305       recall_within_bound_indices = [
1306           index for index, value in enumerate(recall) if
1307           value >= self.recall_lower_bound and value <= self.recall_upper_bound
1308       ]
1309       recall_within_bound = recall[recall_within_bound_indices]
1310       precision_within_bound = precision[recall_within_bound_indices]
1311 
1312       self.precisions_per_class[class_index] = precision_within_bound
1313       self.recalls_per_class[class_index] = recall_within_bound
1314       self.sum_tp_class[class_index] = tp_fp_labels.sum()
1315       average_precision = metrics.compute_average_precision(
1316           precision_within_bound, recall_within_bound)
1317       self.average_precision_per_class[class_index] = average_precision
1318       logging.info('average_precision: %f', average_precision)
1319 
1320     self.corloc_per_class = metrics.compute_cor_loc(
1321         self.num_gt_imgs_per_class,
1322         self.num_images_correctly_detected_per_class)
1323 
1324     if self.use_weighted_mean_ap:
1325       num_gt_instances = np.sum(self.num_gt_instances_per_class)
1326       precision, recall = metrics.compute_precision_recall(
1327           all_scores, all_tp_fp_labels, num_gt_instances)
1328       recall_within_bound_indices = [
1329           index for index, value in enumerate(recall) if
1330           value >= self.recall_lower_bound and value <= self.recall_upper_bound
1331       ]
1332       recall_within_bound = recall[recall_within_bound_indices]
1333       precision_within_bound = precision[recall_within_bound_indices]
1334       mean_ap = metrics.compute_average_precision(precision_within_bound,
1335                                                   recall_within_bound)
1336     else:
1337       mean_ap = np.nanmean(self.average_precision_per_class)
1338     mean_corloc = np.nanmean(self.corloc_per_class)
1339     return ObjectDetectionEvalMetrics(self.average_precision_per_class, mean_ap,
1340                                       self.precisions_per_class,
1341                                       self.recalls_per_class,
1342                                       self.corloc_per_class, mean_corloc)
