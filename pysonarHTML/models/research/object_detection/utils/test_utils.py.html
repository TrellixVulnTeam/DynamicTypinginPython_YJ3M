  38 class MockBoxCoder(box_coder.BoxCoder):
  39   """Simple `difference` BoxCoder."""
  40 
  41   @property
  42   def code_size(self):

  46     return boxes.get() - anchors.get()

  49     return box_list.BoxList(rel_codes + anchors.get())

  65 class MockBoxPredictor(box_predictor.BoxPredictor):
  66   """Simple box predictor that ignores inputs and outputs all zeros."""
  67 
  68   def __init__(self, is_training, num_classes, add_background_class=True):
  69     super(MockBoxPredictor, self).__init__(is_training, num_classes)
  70     self._add_background_class = add_background_class
  71 
  72   def _predict(self, image_features, num_predictions_per_location):
  73     image_feature = image_features[0]
  74     combined_feature_shape = shape_utils.combined_static_and_dynamic_shape(
  75         image_feature)
  76     batch_size = combined_feature_shape[0]
  77     num_anchors = (combined_feature_shape[1] * combined_feature_shape[2])
  78     code_size = 4
  79     zero = tf.reduce_sum(0 * image_feature)
  80     num_class_slots = self.num_classes
  81     if self._add_background_class:
  82       num_class_slots = num_class_slots + 1
  83     box_encodings = zero + tf.zeros(
  84         (batch_size, num_anchors, 1, code_size), dtype=tf.float32)
  85     class_predictions_with_background = zero + tf.zeros(
  86         (batch_size, num_anchors, num_class_slots), dtype=tf.float32)
  87     predictions_dict = {
  88         box_predictor.BOX_ENCODINGS:
  89             box_encodings,
  90         box_predictor.CLASS_PREDICTIONS_WITH_BACKGROUND:
  91             class_predictions_with_background
  92     }
  93     return predictions_dict

 149 def create_diagonal_gradient_image(height, width, depth):
 150   """Creates pyramid image. Useful for testing.
 151 
 152   For example, pyramid_image(5, 6, 1) looks like:
 153   # [[[ 5.  4.  3.  2.  1.  0.]
 154   #   [ 6.  5.  4.  3.  2.  1.]
 155   #   [ 7.  6.  5.  4.  3.  2.]
 156   #   [ 8.  7.  6.  5.  4.  3.]
 157   #   [ 9.  8.  7.  6.  5.  4.]]]
 158 
 159   Args:
 160     height: height of image
 161     width: width of image
 162     depth: depth of image
 163 
 164   Returns:
 165     pyramid image
 166   """
 167   row = np.arange(height)
 168   col = np.arange(width)[::-1]
 169   image_layer = np.expand_dims(row, 1) + col
 170   image_layer = np.expand_dims(image_layer, 2)
 171 
 172   image = image_layer
 173   for i in range(1, depth):
 174     image = np.concatenate((image, image_layer * pow(10, i)), 2)
 175 
 176   return image.astype(np.float32)

 179 def create_random_boxes(num_boxes, max_height, max_width):
 180   """Creates random bounding boxes of specific maximum height and width.
 181 
 182   Args:
 183     num_boxes: number of boxes.
 184     max_height: maximum height of boxes.
 185     max_width: maximum width of boxes.
 186 
 187   Returns:
 188     boxes: numpy array of shape [num_boxes, 4]. Each row is in form
 189         [y_min, x_min, y_max, x_max].
 190   """
 191 
 192   y_1 = np.random.uniform(size=(1, num_boxes)) * max_height
 193   y_2 = np.random.uniform(size=(1, num_boxes)) * max_height
 194   x_1 = np.random.uniform(size=(1, num_boxes)) * max_width
 195   x_2 = np.random.uniform(size=(1, num_boxes)) * max_width
 196 
 197   boxes = np.zeros(shape=(num_boxes, 4))
 198   boxes[:, 0] = np.minimum(y_1, y_2)
 199   boxes[:, 1] = np.minimum(x_1, x_2)
 200   boxes[:, 2] = np.maximum(y_1, y_2)
 201   boxes[:, 3] = np.maximum(x_1, x_2)
 202 
 203   return boxes.astype(np.float32)

 206 def first_rows_close_as_set(a, b, k=None, rtol=1e-6, atol=1e-6):
 207   """Checks if first K entries of two lists are close, up to permutation.
 208 
 209   Inputs to this assert are lists of items which can be compared via
 210   numpy.allclose(...) and can be sorted.
 211 
 212   Args:
 213     a: list of items which can be compared via numpy.allclose(...) and are
 214       sortable.
 215     b: list of items which can be compared via numpy.allclose(...) and are
 216       sortable.
 217     k: a non-negative integer.  If not provided, k is set to be len(a).
 218     rtol: relative tolerance.
 219     atol: absolute tolerance.
 220 
 221   Returns:
 222     boolean, True if input lists a and b have the same length and
 223     the first k entries of the inputs satisfy numpy.allclose() after
 224     sorting entries.
 225   """
 226   if not isinstance(a, list) or not isinstance(b, list) or len(a) != len(b):
 227     return False
 228   if not k:
 229     k = len(a)
 230   k = min(k, len(a))
 231   a_sorted = sorted(a[:k])
 232   b_sorted = sorted(b[:k])
 233   return all([
 234       np.allclose(entry_a, entry_b, rtol, atol)
 235       for (entry_a, entry_b) in zip(a_sorted, b_sorted)
 236   ])

 239 class GraphContextOrNone(object):
 240   """A new Graph context for TF1.X and None for TF2.X.
 241 
 242   This is useful to write model tests that work with both TF1.X and TF2.X.
 243 
 244   Example test using this pattern:
 245 
 246   class ModelTest(test_case.TestCase):
 247     def test_model(self):
 248       with test_utils.GraphContextOrNone() as g:
 249         model = Model()
 250       def compute_fn():
 251         out = model.predict()
 252         return out['detection_boxes']
 253       boxes = self.execute(compute_fn, [], graph=g)
 254       self.assertAllClose(boxes, expected_boxes)
 255   """
 256 
 257   def __init__(self):
 258     if tf_version.is_tf2():
 259       self.graph = None
 260     else:
 261       self.graph = tf.Graph().as_default()
 262 
 263   def __enter__(self):
 264     if tf_version.is_tf2():
 265       return None
 266     else:
 267       return self.graph.__enter__()
 268 
 269   def __exit__(self, ttype, value, traceback):
 270     if tf_version.is_tf2():
 271       return False
 272     else:
 273       return self.graph.__exit__(ttype, value, traceback)
