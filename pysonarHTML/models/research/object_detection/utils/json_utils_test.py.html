  15 """Tests for google3.image.understanding.object_detection.utils.json_utils."""
  16 import os
  17 
  18 import tensorflow.compat.v1 as tf
  19 
  20 from object_detection.utils import json_utils
  21 
  22 
  23 class JsonUtilsTest(tf.test.TestCase):
  24 
  25   def testDumpReasonablePrecision(self):
  26     output_path = os.path.join(tf.test.get_temp_dir(), 'test.json')
  27     with tf.gfile.GFile(output_path, 'w') as f:
  28       json_utils.Dump(1.0, f, float_digits=2)
  29     with tf.gfile.GFile(output_path, 'r') as f:
  30       self.assertEqual(f.read(), '1.00')
  31 
  32   def testDumpPassExtraParams(self):
  33     output_path = os.path.join(tf.test.get_temp_dir(), 'test.json')
  34     with tf.gfile.GFile(output_path, 'w') as f:
  35       json_utils.Dump([1.12345], f, float_digits=2, indent=3)
  36     with tf.gfile.GFile(output_path, 'r') as f:
  37       self.assertEqual(f.read(), '[\n   1.12\n]')
  38 
  39   def testDumpZeroPrecision(self):
  40     output_path = os.path.join(tf.test.get_temp_dir(), 'test.json')
  41     with tf.gfile.GFile(output_path, 'w') as f:
  42       json_utils.Dump(1.0, f, float_digits=0, indent=3)
  43     with tf.gfile.GFile(output_path, 'r') as f:
  44       self.assertEqual(f.read(), '1')
  45 
  46   def testDumpUnspecifiedPrecision(self):
  47     output_path = os.path.join(tf.test.get_temp_dir(), 'test.json')
  48     with tf.gfile.GFile(output_path, 'w') as f:
  49       json_utils.Dump(1.012345, f)
  50     with tf.gfile.GFile(output_path, 'r') as f:
  51       self.assertEqual(f.read(), '1.012345')
  52 
  53   def testDumpsReasonablePrecision(self):
  54     s = json_utils.Dumps(1.12545, float_digits=2)
  55     self.assertEqual(s, '1.13')
  56 
  57   def testDumpsPassExtraParams(self):
  58     s = json_utils.Dumps([1.0], float_digits=2, indent=3)
  59     self.assertEqual(s, '[\n   1.00\n]')
  60 
  61   def testDumpsZeroPrecision(self):
  62     s = json_utils.Dumps(1.0, float_digits=0)
  63     self.assertEqual(s, '1')
  64 
  65   def testDumpsUnspecifiedPrecision(self):
  66     s = json_utils.Dumps(1.012345)
  67     self.assertEqual(s, '1.012345')
  68 
  69   def testPrettyParams(self):
  70     s = json_utils.Dumps({'v': 1.012345, 'n': 2}, **json_utils.PrettyParams())
  71     self.assertEqual(s, '{\n  "n": 2,\n  "v": 1.0123\n}')
  72 
  73   def testPrettyParamsExtraParamsInside(self):
  74     s = json_utils.Dumps(
  75         {'v': 1.012345,
  76          'n': float('nan')}, **json_utils.PrettyParams(allow_nan=True))
  77     self.assertEqual(s, '{\n  "n": NaN,\n  "v": 1.0123\n}')
  78 
  79     with self.assertRaises(ValueError):
  80       s = json_utils.Dumps(
  81           {'v': 1.012345,
  82            'n': float('nan')}, **json_utils.PrettyParams(allow_nan=False))
  83 
  84   def testPrettyParamsExtraParamsOutside(self):
  85     s = json_utils.Dumps(
  86         {'v': 1.012345,
  87          'n': float('nan')}, allow_nan=True, **json_utils.PrettyParams())
  88     self.assertEqual(s, '{\n  "n": NaN,\n  "v": 1.0123\n}')
  89 
  90     with self.assertRaises(ValueError):
  91       s = json_utils.Dumps(
  92           {'v': 1.012345,
  93            'n': float('nan')}, allow_nan=False, **json_utils.PrettyParams())
  94 
  95 
  96 if __name__ == '__main__':
  97   tf.test.main()
