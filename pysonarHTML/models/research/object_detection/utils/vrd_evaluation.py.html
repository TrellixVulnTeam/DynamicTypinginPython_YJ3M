  15 """Evaluator class for Visual Relations Detection.
  16 
  17 VRDDetectionEvaluator is a class which manages ground truth information of a
  18 visual relations detection (vrd) dataset, and computes frequently used detection
  19 metrics such as Precision, Recall, Recall@k, of the provided vrd detection
  20 results.
  21 It supports the following operations:
  22 1) Adding ground truth information of images sequentially.
  23 2) Adding detection results of images sequentially.
  24 3) Evaluating detection metrics on already inserted detection results.
  25 
  26 Note1: groundtruth should be inserted before evaluation.
  27 Note2: This module operates on numpy boxes and box lists.
  28 """

  46 # Below standard input numpy datatypes are defined:
  47 # box_data_type - datatype of the groundtruth visual relations box annotations;
  48 # this datatype consists of two named boxes: subject bounding box and object
  49 # bounding box. Each box is of the format [y_min, x_min, y_max, x_max], each
  50 # coordinate being of type float32.
  51 # label_data_type - corresponding datatype of the visual relations label
  52 # annotaions; it consists of three numerical class labels: subject class label,
  53 # object class label and relation class label, each class label being of type
  54 # int32.
  55 vrd_box_data_type = np.dtype([('subject', 'f4', (4,)), ('object', 'f4', (4,))])
  56 single_box_data_type = np.dtype([('box', 'f4', (4,))])
  57 label_data_type = np.dtype([('subject', 'i4'), ('object', 'i4'), ('relation',
  58                                                                   'i4')])

 122   def add_single_ground_truth_image_info(self, image_id, groundtruth_dict):
 123     """Adds groundtruth for a single image to be used for evaluation.
 124 
 125     Args:
 126       image_id: A unique string/integer identifier for the image.
 127       groundtruth_dict: A dictionary containing -
 128         standard_fields.InputDataFields.groundtruth_boxes: A numpy array
 129           of structures with the shape [M, 1], representing M tuples, each tuple
 130           containing the same number of named bounding boxes.
 131           Each box is of the format [y_min, x_min, y_max, x_max] (see
 132           datatype vrd_box_data_type, single_box_data_type above).
 133         standard_fields.InputDataFields.groundtruth_classes: A numpy array of
 134           structures shape [M, 1], representing  the class labels of the
 135           corresponding bounding boxes and possibly additional classes (see
 136           datatype label_data_type above).
 137         standard_fields.InputDataFields.groundtruth_image_classes: numpy array
 138           of shape [K] containing verified labels.
 139     Raises:
 140       ValueError: On adding groundtruth for an image more than once.
 141     """
 142     if image_id in self._image_ids:
 143       raise ValueError('Image with id {} already added.'.format(image_id))
 144 
 145     groundtruth_class_tuples = (
 146         groundtruth_dict[standard_fields.InputDataFields.groundtruth_classes])
 147     groundtruth_box_tuples = (
 148         groundtruth_dict[standard_fields.InputDataFields.groundtruth_boxes])
 149 
 150     self._evaluation.add_single_ground_truth_image_info(
 151         image_key=image_id,
 152         groundtruth_box_tuples=self._process_groundtruth_boxes(
 153             groundtruth_box_tuples),
 154         groundtruth_class_tuples=groundtruth_class_tuples)
 155     self._image_ids.update([image_id])
 156     all_classes = []
 157     for field in groundtruth_box_tuples.dtype.fields:
 158       all_classes.append(groundtruth_class_tuples[field])
 159     groudtruth_positive_classes = np.unique(np.concatenate(all_classes))
 160     verified_labels = groundtruth_dict.get(
 161         standard_fields.InputDataFields.groundtruth_image_classes,
 162         np.array([], dtype=int))
 163     self._evaluatable_labels[image_id] = np.unique(
 164         np.concatenate((verified_labels, groudtruth_positive_classes)))
 165 
 166     self._negative_labels[image_id] = np.setdiff1d(verified_labels,
 167                                                    groudtruth_positive_classes)
 168 
 169   def add_single_detected_image_info(self, image_id, detections_dict):
 170     """Adds detections for a single image to be used for evaluation.
 171 
 172     Args:
 173       image_id: A unique string/integer identifier for the image.
 174       detections_dict: A dictionary containing -
 175         standard_fields.DetectionResultFields.detection_boxes: A numpy array of
 176           structures with shape [N, 1], representing N tuples, each tuple
 177           containing the same number of named bounding boxes.
 178           Each box is of the format [y_min, x_min, y_max, x_max] (as an example
 179           see datatype vrd_box_data_type, single_box_data_type above).
 180         standard_fields.DetectionResultFields.detection_scores: float32 numpy
 181           array of shape [N] containing detection scores for the boxes.
 182         standard_fields.DetectionResultFields.detection_classes: A numpy array
 183           of structures shape [N, 1], representing the class labels of the
 184           corresponding bounding boxes and possibly additional classes (see
 185           datatype label_data_type above).
 186     """
 187     if image_id not in self._image_ids:
 188       logging.warning('No groundtruth for the image with id %s.', image_id)
 189       # Since for the correct work of evaluator it is assumed that groundtruth
 190       # is inserted first we make sure to break the code if is it not the case.
 191       self._image_ids.update([image_id])
 192       self._negative_labels[image_id] = np.array([])
 193       self._evaluatable_labels[image_id] = np.array([])
 194 
 195     num_detections = detections_dict[
 196         standard_fields.DetectionResultFields.detection_boxes].shape[0]
 197     detection_class_tuples = detections_dict[
 198         standard_fields.DetectionResultFields.detection_classes]
 199     detection_box_tuples = detections_dict[
 200         standard_fields.DetectionResultFields.detection_boxes]
 201     negative_selector = np.zeros(num_detections, dtype=bool)
 202     selector = np.ones(num_detections, dtype=bool)
 203     # Only check boxable labels
 204     for field in detection_box_tuples.dtype.fields:
 205       # Verify if one of the labels is negative (this is sure FP)
 206       negative_selector |= np.isin(detection_class_tuples[field],
 207                                    self._negative_labels[image_id])
 208       # Verify if all labels are verified
 209       selector &= np.isin(detection_class_tuples[field],
 210                           self._evaluatable_labels[image_id])
 211     selector |= negative_selector
 212     self._evaluation.add_single_detected_image_info(
 213         image_key=image_id,
 214         detected_box_tuples=self._process_detection_boxes(
 215             detection_box_tuples[selector]),
 216         detected_scores=detections_dict[
 217             standard_fields.DetectionResultFields.detection_scores][selector],
 218         detected_class_tuples=detection_class_tuples[selector])
 219 
 220   def evaluate(self, relationships=None):
 221     """Compute evaluation result.
 222 
 223     Args:
 224       relationships: A dictionary of numerical label-text label mapping; if
 225         specified, returns per-relationship AP.
 226 
 227     Returns:
 228       A dictionary of metrics with the following fields -
 229 
 230       summary_metrics:
 231         'weightedAP@<matching_iou_threshold>IOU' : weighted average precision
 232         at the specified IOU threshold.
 233         'AP@<matching_iou_threshold>IOU/<relationship>' : AP per relationship.
 234         'mAP@<matching_iou_threshold>IOU': mean average precision at the
 235         specified IOU threshold.
 236         'Recall@50@<matching_iou_threshold>IOU': recall@50 at the specified IOU
 237         threshold.
 238         'Recall@100@<matching_iou_threshold>IOU': recall@100 at the specified
 239         IOU threshold.
 240       if relationships is specified, returns <relationship> in AP metrics as
 241       readable names, otherwise the names correspond to class numbers.
 242     """
 243     (weighted_average_precision, mean_average_precision, average_precisions, _,
 244      _, recall_50, recall_100, _, _) = (
 245          self._evaluation.evaluate())
 246 
 247     vrd_metrics = {
 248         (self._metric_prefix + 'weightedAP@{}IOU'.format(
 249             self._matching_iou_threshold)):
 250             weighted_average_precision,
 251         self._metric_prefix + 'mAP@{}IOU'.format(self._matching_iou_threshold):
 252             mean_average_precision,
 253         self._metric_prefix + 'Recall@50@{}IOU'.format(
 254             self._matching_iou_threshold):
 255             recall_50,
 256         self._metric_prefix + 'Recall@100@{}IOU'.format(
 257             self._matching_iou_threshold):
 258             recall_100,
 259     }
 260     if relationships:
 261       for key, average_precision in six.iteritems(average_precisions):
 262         vrd_metrics[self._metric_prefix + 'AP@{}IOU/{}'.format(
 263             self._matching_iou_threshold,
 264             relationships[key])] = average_precision
 265     else:
 266       for key, average_precision in six.iteritems(average_precisions):
 267         vrd_metrics[self._metric_prefix + 'AP@{}IOU/{}'.format(
 268             self._matching_iou_threshold, key)] = average_precision
 269 
 270     return vrd_metrics
 271 
 272   def clear(self):
 273     """Clears the state to prepare for a fresh evaluation."""
 274     self._evaluation = _VRDDetectionEvaluation(
 275         matching_iou_threshold=self._matching_iou_threshold)
 276     self._image_ids.clear()
 277     self._negative_labels.clear()
 278     self._evaluatable_labels.clear()

 327 class VRDPhraseDetectionEvaluator(VRDDetectionEvaluator):

 329 
 330   Expected groundtruth box datatype is vrd_box_data_type, expected groudtruth
 331   labels datatype is label_data_type.
 332   Expected detection box datatype is single_box_data_type, expected detection
 333   labels datatype is label_data_type.
 334   """
 335 
 336   def __init__(self, matching_iou_threshold=0.5):
 337     super(VRDPhraseDetectionEvaluator, self).__init__(
 338         matching_iou_threshold=matching_iou_threshold,
 339         metric_prefix='VRDMetric_Phrases')
 340 
 341   def _process_groundtruth_boxes(self, groundtruth_box_tuples):
 342     """Pre-processes boxes before adding them to the VRDDetectionEvaluation.
 343 
 344     In case of phrase evaluation task, evaluation expects exactly one bounding
 345     box containing all objects in the phrase. This bounding box is computed
 346     as an enclosing box of all groundtruth boxes of a phrase.
 347 
 348     Args:
 349       groundtruth_box_tuples: A numpy array of structures with the shape
 350         [M, 1], each structure containing the same number of named bounding
 351         boxes. Each box is of the format [y_min, x_min, y_max, x_max]. See
 352         vrd_box_data_type for an example of structure.
 353 
 354     Returns:
 355       result: A numpy array of structures with the shape [M, 1], each
 356         structure containing exactly one named bounding box. i-th output
 357         structure corresponds to the result of processing i-th input structure,
 358         where the named bounding box is computed as an enclosing bounding box
 359         of all bounding boxes of the i-th input structure.
 360     """
 361     first_box_key = next(six.iterkeys(groundtruth_box_tuples.dtype.fields))
 362     miny = groundtruth_box_tuples[first_box_key][:, 0]
 363     minx = groundtruth_box_tuples[first_box_key][:, 1]
 364     maxy = groundtruth_box_tuples[first_box_key][:, 2]
 365     maxx = groundtruth_box_tuples[first_box_key][:, 3]
 366     for fields in groundtruth_box_tuples.dtype.fields:
 367       miny = np.minimum(groundtruth_box_tuples[fields][:, 0], miny)
 368       minx = np.minimum(groundtruth_box_tuples[fields][:, 1], minx)
 369       maxy = np.maximum(groundtruth_box_tuples[fields][:, 2], maxy)
 370       maxx = np.maximum(groundtruth_box_tuples[fields][:, 3], maxx)
 371     data_result = []
 372     for i in range(groundtruth_box_tuples.shape[0]):
 373       data_result.append(([miny[i], minx[i], maxy[i], maxx[i]],))
 374     result = np.array(data_result, dtype=[('box', 'f4', (4,))])
 375     return result
 376

 421 class _VRDDetectionEvaluation(object):
 422   """Performs metric computation for the VRD task. This class is internal.
 423   """
 424 
 425   def __init__(self, matching_iou_threshold=0.5):
 426     """Constructor.
 427 
 428     Args:
 429       matching_iou_threshold: IOU threshold to use for matching groundtruth
 430         boxes to detection boxes.
 431     """
 432     self._per_image_eval = per_image_vrd_evaluation.PerImageVRDEvaluation(
 433         matching_iou_threshold=matching_iou_threshold)
 434 
 435     self._groundtruth_box_tuples = {}
 436     self._groundtruth_class_tuples = {}
 437     self._num_gt_instances = 0
 438     self._num_gt_imgs = 0
 439     self._num_gt_instances_per_relationship = {}
 440 
 441     self.clear_detections()
 442 
 443   def clear_detections(self):
 444     """Clears detections."""
 445     self._detection_keys = set()
 446     self._scores = []
 447     self._relation_field_values = []
 448     self._tp_fp_labels = []
 449     self._average_precisions = {}
 450     self._precisions = []
 451     self._recalls = []
 452 
 453   def add_single_ground_truth_image_info(
 454       self, image_key, groundtruth_box_tuples, groundtruth_class_tuples):
 455     """Adds groundtruth for a single image to be used for evaluation.
 456 
 457     Args:
 458       image_key: A unique string/integer identifier for the image.
 459       groundtruth_box_tuples: A numpy array of structures with the shape
 460           [M, 1], representing M tuples, each tuple containing the same number
 461           of named bounding boxes.
 462           Each box is of the format [y_min, x_min, y_max, x_max].
 463       groundtruth_class_tuples: A numpy array of structures shape [M, 1],
 464           representing  the class labels of the corresponding bounding boxes and
 465           possibly additional classes.
 466     """
 467     if image_key in self._groundtruth_box_tuples:
 468       logging.warning(
 469           'image %s has already been added to the ground truth database.',
 470           image_key)
 471       return
 472 
 473     self._groundtruth_box_tuples[image_key] = groundtruth_box_tuples
 474     self._groundtruth_class_tuples[image_key] = groundtruth_class_tuples
 475 
 476     self._update_groundtruth_statistics(groundtruth_class_tuples)
 477 
 478   def add_single_detected_image_info(self, image_key, detected_box_tuples,
 479                                      detected_scores, detected_class_tuples):
 480     """Adds detections for a single image to be used for evaluation.
 481 
 482     Args:
 483       image_key: A unique string/integer identifier for the image.
 484       detected_box_tuples: A numpy array of structures with shape [N, 1],
 485           representing N tuples, each tuple containing the same number of named
 486           bounding boxes.
 487           Each box is of the format [y_min, x_min, y_max, x_max].
 488       detected_scores: A float numpy array of shape [N, 1], representing
 489           the confidence scores of the detected N object instances.
 490       detected_class_tuples: A numpy array of structures shape [N, 1],
 491           representing the class labels of the corresponding bounding boxes and
 492           possibly additional classes.
 493     """
 494     self._detection_keys.add(image_key)
 495     if image_key in self._groundtruth_box_tuples:
 496       groundtruth_box_tuples = self._groundtruth_box_tuples[image_key]
 497       groundtruth_class_tuples = self._groundtruth_class_tuples[image_key]
 498     else:
 499       groundtruth_box_tuples = np.empty(
 500           shape=[0, 4], dtype=detected_box_tuples.dtype)
 501       groundtruth_class_tuples = np.array([], dtype=detected_class_tuples.dtype)
 502 
 503     scores, tp_fp_labels, mapping = (
 504         self._per_image_eval.compute_detection_tp_fp(
 505             detected_box_tuples=detected_box_tuples,
 506             detected_scores=detected_scores,
 507             detected_class_tuples=detected_class_tuples,
 508             groundtruth_box_tuples=groundtruth_box_tuples,
 509             groundtruth_class_tuples=groundtruth_class_tuples))
 510 
 511     self._scores += [scores]
 512     self._tp_fp_labels += [tp_fp_labels]
 513     self._relation_field_values += [detected_class_tuples[mapping]['relation']]
 514 
 515   def _update_groundtruth_statistics(self, groundtruth_class_tuples):
 516     """Updates grouth truth statistics.
 517 
 518     Args:
 519       groundtruth_class_tuples: A numpy array of structures shape [M, 1],
 520           representing  the class labels of the corresponding bounding boxes and
 521           possibly additional classes.
 522     """
 523     self._num_gt_instances += groundtruth_class_tuples.shape[0]
 524     self._num_gt_imgs += 1
 525     for relation_field_value in np.unique(groundtruth_class_tuples['relation']):
 526       if relation_field_value not in self._num_gt_instances_per_relationship:
 527         self._num_gt_instances_per_relationship[relation_field_value] = 0
 528       self._num_gt_instances_per_relationship[relation_field_value] += np.sum(
 529           groundtruth_class_tuples['relation'] == relation_field_value)
 530 
 531   def evaluate(self):
 532     """Computes evaluation result.
 533 
 534     Returns:
 535       A named tuple with the following fields -
 536         average_precision: a float number corresponding to average precision.
 537         precisions: an array of precisions.
 538         recalls: an array of recalls.
 539         recall@50: recall computed on 50 top-scoring samples.
 540         recall@100: recall computed on 100 top-scoring samples.
 541         median_rank@50: median rank computed on 50 top-scoring samples.
 542         median_rank@100: median rank computed on 100 top-scoring samples.
 543     """
 544     if self._num_gt_instances == 0:
 545       logging.warning('No ground truth instances')
 546 
 547     if not self._scores:
 548       scores = np.array([], dtype=float)
 549       tp_fp_labels = np.array([], dtype=bool)
 550     else:
 551       scores = np.concatenate(self._scores)
 552       tp_fp_labels = np.concatenate(self._tp_fp_labels)
 553       relation_field_values = np.concatenate(self._relation_field_values)
 554 
 555     for relation_field_value, _ in (six.iteritems(
 556         self._num_gt_instances_per_relationship)):
 557       precisions, recalls = metrics.compute_precision_recall(
 558           scores[relation_field_values == relation_field_value],
 559           tp_fp_labels[relation_field_values == relation_field_value],
 560           self._num_gt_instances_per_relationship[relation_field_value])
 561       self._average_precisions[
 562           relation_field_value] = metrics.compute_average_precision(
 563               precisions, recalls)
 564 
 565     self._mean_average_precision = np.mean(
 566         list(self._average_precisions.values()))
 567 
 568     self._precisions, self._recalls = metrics.compute_precision_recall(
 569         scores, tp_fp_labels, self._num_gt_instances)
 570     self._weighted_average_precision = metrics.compute_average_precision(
 571         self._precisions, self._recalls)
 572 
 573     self._recall_50 = (
 574         metrics.compute_recall_at_k(self._tp_fp_labels, self._num_gt_instances,
 575                                     50))
 576     self._median_rank_50 = (
 577         metrics.compute_median_rank_at_k(self._tp_fp_labels, 50))
 578     self._recall_100 = (
 579         metrics.compute_recall_at_k(self._tp_fp_labels, self._num_gt_instances,
 580                                     100))
 581     self._median_rank_100 = (
 582         metrics.compute_median_rank_at_k(self._tp_fp_labels, 100))
 583 
 584     return VRDDetectionEvalMetrics(
 585         self._weighted_average_precision, self._mean_average_precision,
 586         self._average_precisions, self._precisions, self._recalls,
 587         self._recall_50, self._recall_100, self._median_rank_50,
 588         self._median_rank_100)
