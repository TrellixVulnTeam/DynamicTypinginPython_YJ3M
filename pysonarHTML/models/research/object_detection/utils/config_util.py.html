  58 def get_spatial_image_size(image_resizer_config):
  59   """Returns expected spatial size of the output image from a given config.
  60 
  61   Args:
  62     image_resizer_config: An image_resizer_pb2.ImageResizer.
  63 
  64   Returns:
  65     A list of two integers of the form [height, width]. `height` and `width` are
  66     set  -1 if they cannot be determined during graph construction.
  67 
  68   Raises:
  69     ValueError: If the model type is not recognized.
  70   """
  71   if image_resizer_config.HasField("fixed_shape_resizer"):
  72     return [
  73         image_resizer_config.fixed_shape_resizer.height,
  74         image_resizer_config.fixed_shape_resizer.width
  75     ]
  76   if image_resizer_config.HasField("keep_aspect_ratio_resizer"):
  77     if image_resizer_config.keep_aspect_ratio_resizer.pad_to_max_dimension:
  78       return [image_resizer_config.keep_aspect_ratio_resizer.max_dimension] * 2
  79     else:
  80       return [-1, -1]
  81   if image_resizer_config.HasField(
  82       "identity_resizer") or image_resizer_config.HasField(
  83           "conditional_shape_resizer"):
  84     return [-1, -1]
  85   raise ValueError("Unknown image resizer type.")

 122 def get_configs_from_pipeline_file(pipeline_config_path, config_override=None):
 123   """Reads config from a file containing pipeline_pb2.TrainEvalPipelineConfig.
 124 
 125   Args:
 126     pipeline_config_path: Path to pipeline_pb2.TrainEvalPipelineConfig text
 127       proto.
 128     config_override: A pipeline_pb2.TrainEvalPipelineConfig text proto to
 129       override pipeline_config_path.
 130 
 131   Returns:
 132     Dictionary of configuration objects. Keys are `model`, `train_config`,
 133       `train_input_config`, `eval_config`, `eval_input_config`. Value are the
 134       corresponding config objects.
 135   """
 136   pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
 137   with tf.gfile.GFile(pipeline_config_path, "r") as f:
 138     proto_str = f.read()
 139     text_format.Merge(proto_str, pipeline_config)
 140   if config_override:
 141     text_format.Merge(config_override, pipeline_config)
 142   return create_configs_from_pipeline_proto(pipeline_config)
 143

 145 def create_configs_from_pipeline_proto(pipeline_config):
 146   """Creates a configs dictionary from pipeline_pb2.TrainEvalPipelineConfig.
 147 
 148   Args:
 149     pipeline_config: pipeline_pb2.TrainEvalPipelineConfig proto object.
 150 
 151   Returns:
 152     Dictionary of configuration objects. Keys are `model`, `train_config`,
 153       `train_input_config`, `eval_config`, `eval_input_configs`. Value are
 154       the corresponding config objects or list of config objects (only for
 155       eval_input_configs).
 156   """
 157   configs = {}
 158   configs["model"] = pipeline_config.model
 159   configs["train_config"] = pipeline_config.train_config
 160   configs["train_input_config"] = pipeline_config.train_input_reader
 161   configs["eval_config"] = pipeline_config.eval_config
 162   configs["eval_input_configs"] = pipeline_config.eval_input_reader
 163   # Keeps eval_input_config only for backwards compatibility. All clients should
 164   # read eval_input_configs instead.
 165   if configs["eval_input_configs"]:
 166     configs["eval_input_config"] = configs["eval_input_configs"][0]
 167   if pipeline_config.HasField("graph_rewriter"):
 168     configs["graph_rewriter_config"] = pipeline_config.graph_rewriter
 169 
 170   return configs

 173 def get_graph_rewriter_config_from_file(graph_rewriter_config_file):

 188 def create_pipeline_proto_from_configs(configs):
 189   """Creates a pipeline_pb2.TrainEvalPipelineConfig from configs dictionary.
 190 
 191   This function performs the inverse operation of
 192   create_configs_from_pipeline_proto().
 193 
 194   Args:
 195     configs: Dictionary of configs. See get_configs_from_pipeline_file().
 196 
 197   Returns:
 198     A fully populated pipeline_pb2.TrainEvalPipelineConfig.
 199   """
 200   pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
 201   pipeline_config.model.CopyFrom(configs["model"])
 202   pipeline_config.train_config.CopyFrom(configs["train_config"])
 203   pipeline_config.train_input_reader.CopyFrom(configs["train_input_config"])
 204   pipeline_config.eval_config.CopyFrom(configs["eval_config"])
 205   pipeline_config.eval_input_reader.extend(configs["eval_input_configs"])
 206   if "graph_rewriter_config" in configs:
 207     pipeline_config.graph_rewriter.CopyFrom(configs["graph_rewriter_config"])
 208   return pipeline_config

 211 def save_pipeline_config(pipeline_config, directory):
 212   """Saves a pipeline config text file to disk.
 213 
 214   Args:
 215     pipeline_config: A pipeline_pb2.TrainEvalPipelineConfig.
 216     directory: The model directory into which the pipeline config file will be
 217       saved.
 218   """
 219   if not file_io.file_exists(directory):
 220     file_io.recursive_create_dir(directory)
 221   pipeline_config_path = os.path.join(directory, "pipeline.config")
 222   config_text = text_format.MessageToString(pipeline_config)
 223   with tf.gfile.Open(pipeline_config_path, "wb") as f:
 224     tf.logging.info("Writing pipeline config file to %s",
 225                     pipeline_config_path)
 226     f.write(config_text)

 229 def get_configs_from_multiple_files(model_config_path="",
 230                                     train_config_path="",
 231                                     train_input_config_path="",
 232                                     eval_config_path="",
 233                                     eval_input_config_path="",
 234                                     graph_rewriter_config_path=""):
 235   """Reads training configuration from multiple config files.
 236 
 237   Args:
 238     model_config_path: Path to model_pb2.DetectionModel.
 239     train_config_path: Path to train_pb2.TrainConfig.
 240     train_input_config_path: Path to input_reader_pb2.InputReader.
 241     eval_config_path: Path to eval_pb2.EvalConfig.
 242     eval_input_config_path: Path to input_reader_pb2.InputReader.
 243     graph_rewriter_config_path: Path to graph_rewriter_pb2.GraphRewriter.
 244 
 245   Returns:
 246     Dictionary of configuration objects. Keys are `model`, `train_config`,
 247       `train_input_config`, `eval_config`, `eval_input_config`. Key/Values are
 248         returned only for valid (non-empty) strings.
 249   """
 250   configs = {}
 251   if model_config_path:
 252     model_config = model_pb2.DetectionModel()
 253     with tf.gfile.GFile(model_config_path, "r") as f:
 254       text_format.Merge(f.read(), model_config)
 255       configs["model"] = model_config
 256 
 257   if train_config_path:
 258     train_config = train_pb2.TrainConfig()
 259     with tf.gfile.GFile(train_config_path, "r") as f:
 260       text_format.Merge(f.read(), train_config)
 261       configs["train_config"] = train_config
 262 
 263   if train_input_config_path:
 264     train_input_config = input_reader_pb2.InputReader()
 265     with tf.gfile.GFile(train_input_config_path, "r") as f:
 266       text_format.Merge(f.read(), train_input_config)
 267       configs["train_input_config"] = train_input_config
 268 
 269   if eval_config_path:
 270     eval_config = eval_pb2.EvalConfig()
 271     with tf.gfile.GFile(eval_config_path, "r") as f:
 272       text_format.Merge(f.read(), eval_config)
 273       configs["eval_config"] = eval_config
 274 
 275   if eval_input_config_path:
 276     eval_input_config = input_reader_pb2.InputReader()
 277     with tf.gfile.GFile(eval_input_config_path, "r") as f:
 278       text_format.Merge(f.read(), eval_input_config)
 279       configs["eval_input_configs"] = [eval_input_config]
 280 
 281   if graph_rewriter_config_path:
 282     configs["graph_rewriter_config"] = get_graph_rewriter_config_from_file(
 283         graph_rewriter_config_path)
 284 
 285   return configs

 333 def _is_generic_key(key):
 334   """Determines whether the key starts with a generic config dictionary key."""
 335   for prefix in [
 336       "graph_rewriter_config",
 337       "model",
 338       "train_input_config",
 339       "train_config",
 340       "eval_config"]:
 341     if key.startswith(prefix + "."):
 342       return True
 343   return False

 346 def _check_and_convert_legacy_input_config_key(key):
 347   """Checks key and converts legacy input config update to specific update.
 348 
 349   Args:
 350     key: string indicates the target of update operation.
 351 
 352   Returns:
 353     is_valid_input_config_key: A boolean indicating whether the input key is to
 354       update input config(s).
 355     key_name: 'eval_input_configs' or 'train_input_config' string if
 356       is_valid_input_config_key is true. None if is_valid_input_config_key is
 357       false.
 358     input_name: always returns None since legacy input config key never
 359       specifies the target input config. Keeping this output only to match the
 360       output form defined for input config update.
 361     field_name: the field name in input config. `key` itself if
 362       is_valid_input_config_key is false.
 363   """
 364   key_name = None
 365   input_name = None
 366   field_name = key
 367   is_valid_input_config_key = True
 368   if field_name == "train_shuffle":
 369     key_name = "train_input_config"
 370     field_name = "shuffle"
 371   elif field_name == "eval_shuffle":
 372     key_name = "eval_input_configs"
 373     field_name = "shuffle"
 374   elif field_name == "train_input_path":
 375     key_name = "train_input_config"
 376     field_name = "input_path"
 377   elif field_name == "eval_input_path":
 378     key_name = "eval_input_configs"
 379     field_name = "input_path"
 380   elif field_name == "append_train_input_path":
 381     key_name = "train_input_config"
 382     field_name = "input_path"
 383   elif field_name == "append_eval_input_path":
 384     key_name = "eval_input_configs"
 385     field_name = "input_path"
 386   else:
 387     is_valid_input_config_key = False
 388 
 389   return is_valid_input_config_key, key_name, input_name, field_name

 392 def check_and_parse_input_config_key(configs, key):
 393   """Checks key and returns specific fields if key is valid input config update.
 394 
 395   Args:
 396     configs: Dictionary of configuration objects. See outputs from
 397       get_configs_from_pipeline_file() or get_configs_from_multiple_files().
 398     key: string indicates the target of update operation.
 399 
 400   Returns:
 401     is_valid_input_config_key: A boolean indicate whether the input key is to
 402       update input config(s).
 403     key_name: 'eval_input_configs' or 'train_input_config' string if
 404       is_valid_input_config_key is true. None if is_valid_input_config_key is
 405       false.
 406     input_name: the name of the input config to be updated. None if
 407       is_valid_input_config_key is false.
 408     field_name: the field name in input config. `key` itself if
 409       is_valid_input_config_key is false.
 410 
 411   Raises:
 412     ValueError: when the input key format doesn't match any known formats.
 413     ValueError: if key_name doesn't match 'eval_input_configs' or
 414       'train_input_config'.
 415     ValueError: if input_name doesn't match any name in train or eval input
 416       configs.
 417     ValueError: if field_name doesn't match any supported fields.
 418   """
 419   key_name = None
 420   input_name = None
 421   field_name = None
 422   fields = key.split(":")
 423   if len(fields) == 1:
 424     field_name = key
 425     return _check_and_convert_legacy_input_config_key(key)
 426   elif len(fields) == 3:
 427     key_name = fields[0]
 428     input_name = fields[1]
 429     field_name = fields[2]
 430   else:
 431     raise ValueError("Invalid key format when overriding configs.")
 432 
 433   # Checks if key_name is valid for specific update.
 434   if key_name not in ["eval_input_configs", "train_input_config"]:
 435     raise ValueError("Invalid key_name when overriding input config.")
 436 
 437   # Checks if input_name is valid for specific update. For train input config it
 438   # should match configs[key_name].name, for eval input configs it should match
 439   # the name field of one of the eval_input_configs.
 440   if isinstance(configs[key_name], input_reader_pb2.InputReader):
 441     is_valid_input_name = configs[key_name].name == input_name
 442   else:
 443     is_valid_input_name = input_name in [
 444         eval_input_config.name for eval_input_config in configs[key_name]
 445     ]
 446   if not is_valid_input_name:
 447     raise ValueError("Invalid input_name when overriding input config.")
 448 
 449   # Checks if field_name is valid for specific update.
 450   if field_name not in [
 451       "input_path", "label_map_path", "shuffle", "mask_type",
 452       "sample_1_of_n_examples"
 453   ]:
 454     raise ValueError("Invalid field_name when overriding input config.")
 455 
 456   return True, key_name, input_name, field_name

 459 def merge_external_params_with_configs(configs, hparams=None, kwargs_dict=None):
 460   """Updates `configs` dictionary based on supplied parameters.
 461 
 462   This utility is for modifying specific fields in the object detection configs.
 463   Say that one would like to experiment with different learning rates, momentum
 464   values, or batch sizes. Rather than creating a new config text file for each
 465   experiment, one can use a single base config file, and update particular
 466   values.
 467 
 468   There are two types of field overrides:
 469   1. Strategy-based overrides, which update multiple relevant configuration
 470   options. For example, updating `learning_rate` will update both the warmup and
 471   final learning rates.
 472   In this case key can be one of the following formats:
 473       1. legacy update: single string that indicates the attribute to be
 474         updated. E.g. 'label_map_path', 'eval_input_path', 'shuffle'.
 475         Note that when updating fields (e.g. eval_input_path, eval_shuffle) in
 476         eval_input_configs, the override will only be applied when
 477         eval_input_configs has exactly 1 element.
 478       2. specific update: colon separated string that indicates which field in
 479         which input_config to update. It should have 3 fields:
 480         - key_name: Name of the input config we should update, either
 481           'train_input_config' or 'eval_input_configs'
 482         - input_name: a 'name' that can be used to identify elements, especially
 483           when configs[key_name] is a repeated field.
 484         - field_name: name of the field that you want to override.
 485         For example, given configs dict as below:
 486           configs = {
 487             'model': {...}
 488             'train_config': {...}
 489             'train_input_config': {...}
 490             'eval_config': {...}
 491             'eval_input_configs': [{ name:"eval_coco", ...},
 492                                    { name:"eval_voc", ... }]
 493           }
 494         Assume we want to update the input_path of the eval_input_config
 495         whose name is 'eval_coco'. The `key` would then be:
 496         'eval_input_configs:eval_coco:input_path'
 497   2. Generic key/value, which update a specific parameter based on namespaced
 498   configuration keys. For example,
 499   `model.ssd.loss.hard_example_miner.max_negatives_per_positive` will update the
 500   hard example miner configuration for an SSD model config. Generic overrides
 501   are automatically detected based on the namespaced keys.
 502 
 503   Args:
 504     configs: Dictionary of configuration objects. See outputs from
 505       get_configs_from_pipeline_file() or get_configs_from_multiple_files().
 506     hparams: A `HParams`.
 507     kwargs_dict: Extra keyword arguments that are treated the same way as
 508       attribute/value pairs in `hparams`. Note that hyperparameters with the
 509       same names will override keyword arguments.
 510 
 511   Returns:
 512     `configs` dictionary.
 513 
 514   Raises:
 515     ValueError: when the key string doesn't match any of its allowed formats.
 516   """
 517 
 518   if kwargs_dict is None:
 519     kwargs_dict = {}
 520   if hparams:
 521     kwargs_dict.update(hparams.values())
 522   for key, value in kwargs_dict.items():
 523     tf.logging.info("Maybe overwriting %s: %s", key, value)
 524     # pylint: disable=g-explicit-bool-comparison
 525     if value == "" or value is None:
 526       continue
 527     # pylint: enable=g-explicit-bool-comparison
 528     elif _maybe_update_config_with_key_value(configs, key, value):
 529       continue
 530     elif _is_generic_key(key):
 531       _update_generic(configs, key, value)
 532     else:
 533       tf.logging.info("Ignoring config override key: %s", key)
 534   return configs

 537 def _maybe_update_config_with_key_value(configs, key, value):
 538   """Checks key type and updates `configs` with the key value pair accordingly.
 539 
 540   Args:
 541     configs: Dictionary of configuration objects. See outputs from
 542       get_configs_from_pipeline_file() or get_configs_from_multiple_files().
 543     key: String indicates the field(s) to be updated.
 544     value: Value used to override existing field value.
 545 
 546   Returns:
 547     A boolean value that indicates whether the override succeeds.
 548 
 549   Raises:
 550     ValueError: when the key string doesn't match any of the formats above.
 551   """
 552   is_valid_input_config_key, key_name, input_name, field_name = (
 553       check_and_parse_input_config_key(configs, key))
 554   if is_valid_input_config_key:
 555     update_input_reader_config(
 556         configs,
 557         key_name=key_name,
 558         input_name=input_name,
 559         field_name=field_name,
 560         value=value)
 561   elif field_name == "learning_rate":
 562     _update_initial_learning_rate(configs, value)
 563   elif field_name == "batch_size":
 564     _update_batch_size(configs, value)
 565   elif field_name == "momentum_optimizer_value":
 566     _update_momentum_optimizer_value(configs, value)
 567   elif field_name == "classification_localization_weight_ratio":
 568     # Localization weight is fixed to 1.0.
 569     _update_classification_localization_weight_ratio(configs, value)
 570   elif field_name == "focal_loss_gamma":
 571     _update_focal_loss_gamma(configs, value)
 572   elif field_name == "focal_loss_alpha":
 573     _update_focal_loss_alpha(configs, value)
 574   elif field_name == "train_steps":
 575     _update_train_steps(configs, value)
 576   elif field_name == "label_map_path":
 577     _update_label_map_path(configs, value)
 578   elif field_name == "mask_type":
 579     _update_mask_type(configs, value)
 580   elif field_name == "sample_1_of_n_eval_examples":
 581     _update_all_eval_input_configs(configs, "sample_1_of_n_examples", value)
 582   elif field_name == "eval_num_epochs":
 583     _update_all_eval_input_configs(configs, "num_epochs", value)
 584   elif field_name == "eval_with_moving_averages":
 585     _update_use_moving_averages(configs, value)
 586   elif field_name == "retain_original_images_in_eval":
 587     _update_retain_original_images(configs["eval_config"], value)
 588   elif field_name == "use_bfloat16":
 589     _update_use_bfloat16(configs, value)
 590   elif field_name == "retain_original_image_additional_channels_in_eval":
 591     _update_retain_original_image_additional_channels(configs["eval_config"],
 592                                                       value)
 593   elif field_name == "num_classes":
 594     _update_num_classes(configs["model"], value)
 595   else:
 596     return False
 597   return True

 623 def update_input_reader_config(configs,
 624                                key_name=None,
 625                                input_name=None,
 626                                field_name=None,
 627                                value=None,
 628                                path_updater=_update_tf_record_input_path):
 629   """Updates specified input reader config field.
 630 
 631   Args:
 632     configs: Dictionary of configuration objects. See outputs from
 633       get_configs_from_pipeline_file() or get_configs_from_multiple_files().
 634     key_name: Name of the input config we should update, either
 635       'train_input_config' or 'eval_input_configs'
 636     input_name: String name used to identify input config to update with. Should
 637       be either None or value of the 'name' field in one of the input reader
 638       configs.
 639     field_name: Field name in input_reader_pb2.InputReader.
 640     value: Value used to override existing field value.
 641     path_updater: helper function used to update the input path. Only used when
 642       field_name is "input_path".
 643 
 644   Raises:
 645     ValueError: when input field_name is None.
 646     ValueError: when input_name is None and number of eval_input_readers does
 647       not equal to 1.
 648   """
 649   if isinstance(configs[key_name], input_reader_pb2.InputReader):
 650     # Updates singular input_config object.
 651     target_input_config = configs[key_name]
 652     if field_name == "input_path":
 653       path_updater(input_config=target_input_config, input_path=value)
 654     else:
 655       setattr(target_input_config, field_name, value)
 656   elif input_name is None and len(configs[key_name]) == 1:
 657     # Updates first (and the only) object of input_config list.
 658     target_input_config = configs[key_name][0]
 659     if field_name == "input_path":
 660       path_updater(input_config=target_input_config, input_path=value)
 661     else:
 662       setattr(target_input_config, field_name, value)
 663   elif input_name is not None and len(configs[key_name]):
 664     # Updates input_config whose name matches input_name.
 665     update_count = 0
 666     for input_config in configs[key_name]:
 667       if input_config.name == input_name:
 668         setattr(input_config, field_name, value)
 669         update_count = update_count + 1
 670     if not update_count:
 671       raise ValueError(
 672           "Input name {} not found when overriding.".format(input_name))
 673     elif update_count > 1:
 674       raise ValueError("Duplicate input name found when overriding.")
 675   else:
 676     key_name = "None" if key_name is None else key_name
 677     input_name = "None" if input_name is None else input_name
 678     field_name = "None" if field_name is None else field_name
 679     raise ValueError("Unknown input config overriding: "
 680                      "key_name:{}, input_name:{}, field_name:{}.".format(
 681                          key_name, input_name, field_name))

 684 def _update_initial_learning_rate(configs, learning_rate):
 685   """Updates `configs` to reflect the new initial learning rate.
 686 
 687   This function updates the initial learning rate. For learning rate schedules,
 688   all other defined learning rates in the pipeline config are scaled to maintain
 689   their same ratio with the initial learning rate.
 690   The configs dictionary is updated in place, and hence not returned.
 691 
 692   Args:
 693     configs: Dictionary of configuration objects. See outputs from
 694       get_configs_from_pipeline_file() or get_configs_from_multiple_files().
 695     learning_rate: Initial learning rate for optimizer.
 696 
 697   Raises:
 698     TypeError: if optimizer type is not supported, or if learning rate type is
 699       not supported.
 700   """
 701 
 702   optimizer_type = get_optimizer_type(configs["train_config"])
 703   if optimizer_type == "rms_prop_optimizer":
 704     optimizer_config = configs["train_config"].optimizer.rms_prop_optimizer
 705   elif optimizer_type == "momentum_optimizer":
 706     optimizer_config = configs["train_config"].optimizer.momentum_optimizer
 707   elif optimizer_type == "adam_optimizer":
 708     optimizer_config = configs["train_config"].optimizer.adam_optimizer
 709   else:
 710     raise TypeError("Optimizer %s is not supported." % optimizer_type)
 711 
 712   learning_rate_type = get_learning_rate_type(optimizer_config)
 713   if learning_rate_type == "constant_learning_rate":
 714     constant_lr = optimizer_config.learning_rate.constant_learning_rate
 715     constant_lr.learning_rate = learning_rate
 716   elif learning_rate_type == "exponential_decay_learning_rate":
 717     exponential_lr = (
 718         optimizer_config.learning_rate.exponential_decay_learning_rate)
 719     exponential_lr.initial_learning_rate = learning_rate
 720   elif learning_rate_type == "manual_step_learning_rate":
 721     manual_lr = optimizer_config.learning_rate.manual_step_learning_rate
 722     original_learning_rate = manual_lr.initial_learning_rate
 723     learning_rate_scaling = float(learning_rate) / original_learning_rate
 724     manual_lr.initial_learning_rate = learning_rate
 725     for schedule in manual_lr.schedule:
 726       schedule.learning_rate *= learning_rate_scaling
 727   elif learning_rate_type == "cosine_decay_learning_rate":
 728     cosine_lr = optimizer_config.learning_rate.cosine_decay_learning_rate
 729     learning_rate_base = cosine_lr.learning_rate_base
 730     warmup_learning_rate = cosine_lr.warmup_learning_rate
 731     warmup_scale_factor = warmup_learning_rate / learning_rate_base
 732     cosine_lr.learning_rate_base = learning_rate
 733     cosine_lr.warmup_learning_rate = warmup_scale_factor * learning_rate
 734   else:
 735     raise TypeError("Learning rate %s is not supported." % learning_rate_type)

 757 def _update_generic(configs, key, value):
 758   """Update a pipeline configuration parameter based on a generic key/value.
 759 
 760   Args:
 761     configs: Dictionary of pipeline configuration protos.
 762     key: A string key, dot-delimited to represent the argument key.
 763       e.g. "model.ssd.train_config.batch_size"
 764     value: A value to set the argument to. The type of the value must match the
 765       type for the protocol buffer. Note that setting the wrong type will
 766       result in a TypeError.
 767       e.g. 42
 768 
 769   Raises:
 770     ValueError if the message key does not match the existing proto fields.
 771     TypeError the value type doesn't match the protobuf field type.
 772   """
 773   fields = key.split(".")
 774   first_field = fields.pop(0)
 775   last_field = fields.pop()
 776   message = configs[first_field]
 777   for field in fields:
 778     _validate_message_has_field(message, field)
 779     message = getattr(message, field)
 780   _validate_message_has_field(message, last_field)
 781   setattr(message, last_field, value)

 812 def _update_classification_localization_weight_ratio(configs, ratio):
 813   """Updates the classification/localization weight loss ratio.
 814 
 815   Detection models usually define a loss weight for both classification and
 816   objectness. This function updates the weights such that the ratio between
 817   classification weight to localization weight is the ratio provided.
 818   Arbitrarily, localization weight is set to 1.0.
 819 
 820   Note that in the case of Faster R-CNN, this same ratio is applied to the first
 821   stage objectness loss weight relative to localization loss weight.
 822 
 823   The configs dictionary is updated in place, and hence not returned.
 824 
 825   Args:
 826     configs: Dictionary of configuration objects. See outputs from
 827       get_configs_from_pipeline_file() or get_configs_from_multiple_files().
 828     ratio: Desired ratio of classification (and/or objectness) loss weight to
 829       localization loss weight.
 830   """
 831   meta_architecture = configs["model"].WhichOneof("model")
 832   if meta_architecture == "faster_rcnn":
 833     model = configs["model"].faster_rcnn
 834     model.first_stage_localization_loss_weight = 1.0
 835     model.first_stage_objectness_loss_weight = ratio
 836     model.second_stage_localization_loss_weight = 1.0
 837     model.second_stage_classification_loss_weight = ratio
 838   if meta_architecture == "ssd":
 839     model = configs["model"].ssd
 840     model.loss.localization_weight = 1.0
 841     model.loss.classification_weight = ratio

 998 def remove_unecessary_ema(variables_to_restore, no_ema_collection=None):
 999   """Remap and Remove EMA variable that are not created during training.
1000 
1001   ExponentialMovingAverage.variables_to_restore() returns a map of EMA names
1002   to tf variables to restore. E.g.:
1003   {
1004       conv/batchnorm/gamma/ExponentialMovingAverage: conv/batchnorm/gamma,
1005       conv_4/conv2d_params/ExponentialMovingAverage: conv_4/conv2d_params,
1006       global_step: global_step
1007   }
1008   This function takes care of the extra ExponentialMovingAverage variables
1009   that get created during eval but aren't available in the checkpoint, by
1010   remapping the key to the shallow copy of the variable itself, and remove
1011   the entry of its EMA from the variables to restore. An example resulting
1012   dictionary would look like:
1013   {
1014       conv/batchnorm/gamma: conv/batchnorm/gamma,
1015       conv_4/conv2d_params: conv_4/conv2d_params,
1016       global_step: global_step
1017   }
1018   Args:
1019     variables_to_restore: A dictionary created by ExponentialMovingAverage.
1020       variables_to_restore().
1021     no_ema_collection: A list of namescope substrings to match the variables
1022       to eliminate EMA.
1023 
1024   Returns:
1025     A variables_to_restore dictionary excluding the collection of unwanted
1026     EMA mapping.
1027   """
1028   if no_ema_collection is None:
1029     return variables_to_restore
1030 
1031   for key in variables_to_restore:
1032     if "ExponentialMovingAverage" in key:
1033       for name in no_ema_collection:
1034         if name in key:
1035           variables_to_restore[key.replace("/ExponentialMovingAverage",
1036                                            "")] = variables_to_restore[key]
1037           del variables_to_restore[key]
1038   return variables_to_restore
