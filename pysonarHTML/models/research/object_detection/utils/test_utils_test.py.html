  31   def test_diagonal_gradient_image(self):
  32     """Tests if a good pyramid image is created."""
  33     pyramid_image = test_utils.create_diagonal_gradient_image(3, 4, 2)
  34 
  35     # Test which is easy to understand.
  36     expected_first_channel = np.array([[3, 2, 1, 0],
  37                                        [4, 3, 2, 1],
  38                                        [5, 4, 3, 2]], dtype=np.float32)
  39     self.assertAllEqual(np.squeeze(pyramid_image[:, :, 0]),
  40                         expected_first_channel)
  41 
  42     # Actual test.
  43     expected_image = np.array([[[3, 30],
  44                                 [2, 20],
  45                                 [1, 10],
  46                                 [0, 0]],
  47                                [[4, 40],
  48                                 [3, 30],
  49                                 [2, 20],
  50                                 [1, 10]],
  51                                [[5, 50],
  52                                 [4, 40],
  53                                 [3, 30],
  54                                 [2, 20]]], dtype=np.float32)
  55 
  56     self.assertAllEqual(pyramid_image, expected_image)

  58   def test_random_boxes(self):
  59     """Tests if valid random boxes are created."""
  60     num_boxes = 1000
  61     max_height = 3
  62     max_width = 5
  63     boxes = test_utils.create_random_boxes(num_boxes,
  64                                            max_height,
  65                                            max_width)
  66 
  67     true_column = np.ones(shape=(num_boxes)) == 1
  68     self.assertAllEqual(boxes[:, 0] < boxes[:, 2], true_column)
  69     self.assertAllEqual(boxes[:, 1] < boxes[:, 3], true_column)
  70 
  71     self.assertGreaterEqual(boxes[:, 0].min(), 0)
  72     self.assertGreaterEqual(boxes[:, 1].min(), 0)
  73     self.assertLessEqual(boxes[:, 2].max(), max_height)
  74     self.assertLessEqual(boxes[:, 3].max(), max_width)
  75 
  76   def test_first_rows_close_as_set(self):
  77     a = [1, 2, 3, 0, 0]
  78     b = [3, 2, 1, 0, 0]
  79     k = 3
  80     self.assertTrue(test_utils.first_rows_close_as_set(a, b, k))
  81 
  82     a = [[1, 2], [1, 4], [0, 0]]
  83     b = [[1, 4 + 1e-9], [1, 2], [0, 0]]
  84     k = 2
  85     self.assertTrue(test_utils.first_rows_close_as_set(a, b, k))
  86 
  87     a = [[1, 2], [1, 4], [0, 0]]
  88     b = [[1, 4 + 1e-9], [2, 2], [0, 0]]
  89     k = 2
  90     self.assertFalse(test_utils.first_rows_close_as_set(a, b, k))
