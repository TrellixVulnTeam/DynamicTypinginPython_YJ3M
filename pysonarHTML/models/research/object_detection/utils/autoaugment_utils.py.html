 159 def blend(image1, image2, factor):
 160   """Blend image1 and image2 using 'factor'.
 161 
 162   Factor can be above 0.0.  A value of 0.0 means only image1 is used.
 163   A value of 1.0 means only image2 is used.  A value between 0.0 and
 164   1.0 means we linearly interpolate the pixel values between the two
 165   images.  A value greater than 1.0 "extrapolates" the difference
 166   between the two pixel values, and we clip the results to values
 167   between 0 and 255.
 168 
 169   Args:
 170     image1: An image Tensor of type uint8.
 171     image2: An image Tensor of type uint8.
 172     factor: A floating point value above 0.0.
 173 
 174   Returns:
 175     A blended image Tensor of type uint8.
 176   """
 177   if factor == 0.0:
 178     return tf.convert_to_tensor(image1)
 179   if factor == 1.0:
 180     return tf.convert_to_tensor(image2)
 181 
 182   image1 = tf.to_float(image1)
 183   image2 = tf.to_float(image2)
 184 
 185   difference = image2 - image1
 186   scaled = factor * difference
 187 
 188   # Do addition in float.
 189   temp = tf.to_float(image1) + scaled
 190 
 191   # Interpolate
 192   if factor > 0.0 and factor < 1.0:
 193     # Interpolation means we always stay within 0 and 255.
 194     return tf.cast(temp, tf.uint8)
 195 
 196   # Extrapolate:
 197   #
 198   # We need to clip and then cast.
 199   return tf.cast(tf.clip_by_value(temp, 0.0, 255.0), tf.uint8)

 202 def cutout(image, pad_size, replace=0):
 203   """Apply cutout (https://arxiv.org/abs/1708.04552) to image.
 204 
 205   This operation applies a (2*pad_size x 2*pad_size) mask of zeros to
 206   a random location within `img`. The pixel values filled in will be of the
 207   value `replace`. The located where the mask will be applied is randomly
 208   chosen uniformly over the whole image.
 209 
 210   Args:
 211     image: An image Tensor of type uint8.
 212     pad_size: Specifies how big the zero mask that will be generated is that
 213       is applied to the image. The mask will be of size
 214       (2*pad_size x 2*pad_size).
 215     replace: What pixel value to fill in the image in the area that has
 216       the cutout mask applied to it.
 217 
 218   Returns:
 219     An image Tensor that is of type uint8.
 220   """
 221   image_height = tf.shape(image)[0]
 222   image_width = tf.shape(image)[1]
 223 
 224   # Sample the center location in the image where the zero mask will be applied.
 225   cutout_center_height = tf.random_uniform(
 226       shape=[], minval=0, maxval=image_height,
 227       dtype=tf.int32)
 228 
 229   cutout_center_width = tf.random_uniform(
 230       shape=[], minval=0, maxval=image_width,
 231       dtype=tf.int32)
 232 
 233   lower_pad = tf.maximum(0, cutout_center_height - pad_size)
 234   upper_pad = tf.maximum(0, image_height - cutout_center_height - pad_size)
 235   left_pad = tf.maximum(0, cutout_center_width - pad_size)
 236   right_pad = tf.maximum(0, image_width - cutout_center_width - pad_size)
 237 
 238   cutout_shape = [image_height - (lower_pad + upper_pad),
 239                   image_width - (left_pad + right_pad)]
 240   padding_dims = [[lower_pad, upper_pad], [left_pad, right_pad]]
 241   mask = tf.pad(
 242       tf.zeros(cutout_shape, dtype=image.dtype),
 243       padding_dims, constant_values=1)
 244   mask = tf.expand_dims(mask, -1)
 245   mask = tf.tile(mask, [1, 1, 3])
 246   image = tf.where(
 247       tf.equal(mask, 0),
 248       tf.ones_like(image, dtype=image.dtype) * replace,
 249       image)
 250   return image

 253 def solarize(image, threshold=128):
 254   # For each pixel in the image, select the pixel
 255   # if the value is less than the threshold.
 256   # Otherwise, subtract 255 from the pixel.
 257   return tf.where(image < threshold, image, 255 - image)

 260 def solarize_add(image, addition=0, threshold=128):
 261   # For each pixel in the image less than threshold
 262   # we add 'addition' amount to it and then clip the
 263   # pixel value to be between 0 and 255. The value
 264   # of 'addition' is between -128 and 128.
 265   added_image = tf.cast(image, tf.int64) + addition
 266   added_image = tf.cast(tf.clip_by_value(added_image, 0, 255), tf.uint8)
 267   return tf.where(image < threshold, added_image, image)

 276 def contrast(image, factor):
 277   """Equivalent of PIL Contrast."""
 278   degenerate = tf.image.rgb_to_grayscale(image)
 279   # Cast before calling tf.histogram.
 280   degenerate = tf.cast(degenerate, tf.int32)
 281 
 282   # Compute the grayscale histogram, then compute the mean pixel value,
 283   # and create a constant image size of that value.  Use that as the
 284   # blending degenerate target of the original image.
 285   hist = tf.histogram_fixed_width(degenerate, [0, 255], nbins=256)
 286   mean = tf.reduce_sum(tf.cast(hist, tf.float32)) / 256.0
 287   degenerate = tf.ones_like(degenerate, dtype=tf.float32) * mean
 288   degenerate = tf.clip_by_value(degenerate, 0.0, 255.0)
 289   degenerate = tf.image.grayscale_to_rgb(tf.cast(degenerate, tf.uint8))
 290   return blend(degenerate, image, factor)

 305 def rotate(image, degrees, replace):
 306   """Rotates the image by degrees either clockwise or counterclockwise.
 307 
 308   Args:
 309     image: An image Tensor of type uint8.
 310     degrees: Float, a scalar angle in degrees to rotate all images by. If
 311       degrees is positive the image will be rotated clockwise otherwise it will
 312       be rotated counterclockwise.
 313     replace: A one or three value 1D tensor to fill empty pixels caused by
 314       the rotate operation.
 315 
 316   Returns:
 317     The rotated version of image.
 318   """
 319   # Convert from degrees to radians.
 320   degrees_to_radians = math.pi / 180.0
 321   radians = degrees * degrees_to_radians
 322 
 323   # In practice, we should randomize the rotation degrees by flipping
 324   # it negatively half the time, but that's done on 'degrees' outside
 325   # of the function.
 326   image = contrib_image.rotate(wrap(image), radians)
 327   return unwrap(image, replace)

 330 def random_shift_bbox(image, bbox, pixel_scaling, replace,
 331                       new_min_bbox_coords=None):
 332   """Move the bbox and the image content to a slightly new random location.
 333 
 334   Args:
 335     image: 3D uint8 Tensor.
 336     bbox: 1D Tensor that has 4 elements (min_y, min_x, max_y, max_x)
 337       of type float that represents the normalized coordinates between 0 and 1.
 338       The potential values for the new min corner of the bbox will be between
 339       [old_min - pixel_scaling * bbox_height/2,
 340        old_min - pixel_scaling * bbox_height/2].
 341     pixel_scaling: A float between 0 and 1 that specifies the pixel range
 342       that the new bbox location will be sampled from.
 343     replace: A one or three value 1D tensor to fill empty pixels.
 344     new_min_bbox_coords: If not None, then this is a tuple that specifies the
 345       (min_y, min_x) coordinates of the new bbox. Normally this is randomly
 346       specified, but this allows it to be manually set. The coordinates are
 347       the absolute coordinates between 0 and image height/width and are int32.
 348 
 349   Returns:
 350     The new image that will have the shifted bbox location in it along with
 351     the new bbox that contains the new coordinates.
 352   """
 353   # Obtains image height and width and create helper clip functions.
 354   image_height = tf.to_float(tf.shape(image)[0])
 355   image_width = tf.to_float(tf.shape(image)[1])
 356   def clip_y(val):
 357     return tf.clip_by_value(val, 0, tf.to_int32(image_height) - 1)
 358   def clip_x(val):
 359     return tf.clip_by_value(val, 0, tf.to_int32(image_width) - 1)
 360 
 361   # Convert bbox to pixel coordinates.
 362   min_y = tf.to_int32(image_height * bbox[0])
 363   min_x = tf.to_int32(image_width * bbox[1])
 364   max_y = clip_y(tf.to_int32(image_height * bbox[2]))
 365   max_x = clip_x(tf.to_int32(image_width * bbox[3]))
 366   bbox_height, bbox_width = (max_y - min_y + 1, max_x - min_x + 1)
 367   image_height = tf.to_int32(image_height)
 368   image_width = tf.to_int32(image_width)
 369 
 370   # Select the new min/max bbox ranges that are used for sampling the
 371   # new min x/y coordinates of the shifted bbox.
 372   minval_y = clip_y(
 373       min_y - tf.to_int32(pixel_scaling * tf.to_float(bbox_height) / 2.0))
 374   maxval_y = clip_y(
 375       min_y + tf.to_int32(pixel_scaling * tf.to_float(bbox_height) / 2.0))
 376   minval_x = clip_x(
 377       min_x - tf.to_int32(pixel_scaling * tf.to_float(bbox_width) / 2.0))
 378   maxval_x = clip_x(
 379       min_x + tf.to_int32(pixel_scaling * tf.to_float(bbox_width) / 2.0))
 380 
 381   # Sample and calculate the new unclipped min/max coordinates of the new bbox.
 382   if new_min_bbox_coords is None:
 383     unclipped_new_min_y = tf.random_uniform(
 384         shape=[], minval=minval_y, maxval=maxval_y,
 385         dtype=tf.int32)
 386     unclipped_new_min_x = tf.random_uniform(
 387         shape=[], minval=minval_x, maxval=maxval_x,
 388         dtype=tf.int32)
 389   else:
 390     unclipped_new_min_y, unclipped_new_min_x = (
 391         clip_y(new_min_bbox_coords[0]), clip_x(new_min_bbox_coords[1]))
 392   unclipped_new_max_y = unclipped_new_min_y + bbox_height - 1
 393   unclipped_new_max_x = unclipped_new_min_x + bbox_width - 1
 394 
 395   # Determine if any of the new bbox was shifted outside the current image.
 396   # This is used for determining if any of the original bbox content should be
 397   # discarded.
 398   new_min_y, new_min_x, new_max_y, new_max_x = (
 399       clip_y(unclipped_new_min_y), clip_x(unclipped_new_min_x),
 400       clip_y(unclipped_new_max_y), clip_x(unclipped_new_max_x))
 401   shifted_min_y = (new_min_y - unclipped_new_min_y) + min_y
 402   shifted_max_y = max_y - (unclipped_new_max_y - new_max_y)
 403   shifted_min_x = (new_min_x - unclipped_new_min_x) + min_x
 404   shifted_max_x = max_x - (unclipped_new_max_x - new_max_x)
 405 
 406   # Create the new bbox tensor by converting pixel integer values to floats.
 407   new_bbox = tf.stack([
 408       tf.to_float(new_min_y) / tf.to_float(image_height),
 409       tf.to_float(new_min_x) / tf.to_float(image_width),
 410       tf.to_float(new_max_y) / tf.to_float(image_height),
 411       tf.to_float(new_max_x) / tf.to_float(image_width)])
 412 
 413   # Copy the contents in the bbox and fill the old bbox location
 414   # with gray (128).
 415   bbox_content = image[shifted_min_y:shifted_max_y + 1,
 416                        shifted_min_x:shifted_max_x + 1, :]
 417 
 418   def mask_and_add_image(
 419       min_y_, min_x_, max_y_, max_x_, mask, content_tensor, image_):
 420     """Applies mask to bbox region in image then adds content_tensor to it."""
 421     mask = tf.pad(mask,
 422                   [[min_y_, (image_height - 1) - max_y_],
 423                    [min_x_, (image_width - 1) - max_x_],
 424                    [0, 0]], constant_values=1)
 425     content_tensor = tf.pad(content_tensor,
 426                             [[min_y_, (image_height - 1) - max_y_],
 427                              [min_x_, (image_width - 1) - max_x_],
 428                              [0, 0]], constant_values=0)
 429     return image_ * mask + content_tensor
 430 
 431   # Zero out original bbox location.
 432   mask = tf.zeros_like(image)[min_y:max_y+1, min_x:max_x+1, :]
 433   grey_tensor = tf.zeros_like(mask) + replace[0]
 434   image = mask_and_add_image(min_y, min_x, max_y, max_x, mask,
 435                              grey_tensor, image)
 436 
 437   # Fill in bbox content to new bbox location.
 438   mask = tf.zeros_like(bbox_content)
 439   image = mask_and_add_image(new_min_y, new_min_x, new_max_y, new_max_x, mask,
 440                              bbox_content, image)
 441 
 442   return image, new_bbox

 445 def _clip_bbox(min_y, min_x, max_y, max_x):
 446   """Clip bounding box coordinates between 0 and 1.
 447 
 448   Args:
 449     min_y: Normalized bbox coordinate of type float between 0 and 1.
 450     min_x: Normalized bbox coordinate of type float between 0 and 1.
 451     max_y: Normalized bbox coordinate of type float between 0 and 1.
 452     max_x: Normalized bbox coordinate of type float between 0 and 1.
 453 
 454   Returns:
 455     Clipped coordinate values between 0 and 1.
 456   """
 457   min_y = tf.clip_by_value(min_y, 0.0, 1.0)
 458   min_x = tf.clip_by_value(min_x, 0.0, 1.0)
 459   max_y = tf.clip_by_value(max_y, 0.0, 1.0)

 464 def _check_bbox_area(min_y, min_x, max_y, max_x, delta=0.05):
 465   """Adjusts bbox coordinates to make sure the area is > 0.
 466 
 467   Args:
 468     min_y: Normalized bbox coordinate of type float between 0 and 1.
 469     min_x: Normalized bbox coordinate of type float between 0 and 1.
 470     max_y: Normalized bbox coordinate of type float between 0 and 1.
 471     max_x: Normalized bbox coordinate of type float between 0 and 1.
 472     delta: Float, this is used to create a gap of size 2 * delta between
 473       bbox min/max coordinates that are the same on the boundary.
 474       This prevents the bbox from having an area of zero.
 475 
 476   Returns:
 477     Tuple of new bbox coordinates between 0 and 1 that will now have a
 478     guaranteed area > 0.
 479   """
 480   height = max_y - min_y
 481   width = max_x - min_x
 482   def _adjust_bbox_boundaries(min_coord, max_coord):
 483     # Make sure max is never 0 and min is never 1.
 484     max_coord = tf.maximum(max_coord, 0.0 + delta)
 485     min_coord = tf.minimum(min_coord, 1.0 - delta)
 486     return min_coord, max_coord
 487   min_y, max_y = tf.cond(tf.equal(height, 0.0),
 488                          lambda: _adjust_bbox_boundaries(min_y, max_y),
 489                          lambda: (min_y, max_y))
 490   min_x, max_x = tf.cond(tf.equal(width, 0.0),
 491                          lambda: _adjust_bbox_boundaries(min_x, max_x),
 492                          lambda: (min_x, max_x))
 493   return min_y, min_x, max_y, max_x

 496 def _scale_bbox_only_op_probability(prob):
 497   """Reduce the probability of the bbox-only operation.
 498 
 499   Probability is reduced so that we do not distort the content of too many
 500   bounding boxes that are close to each other. The value of 3.0 was a chosen
 501   hyper parameter when designing the autoaugment algorithm that we found
 502   empirically to work well.
 503 
 504   Args:
 505     prob: Float that is the probability of applying the bbox-only operation.
 506 
 507   Returns:
 508     Reduced probability.
 509   """
 510   return prob / 3.0

 513 def _apply_bbox_augmentation(image, bbox, augmentation_func, *args):
 514   """Applies augmentation_func to the subsection of image indicated by bbox.
 515 
 516   Args:
 517     image: 3D uint8 Tensor.
 518     bbox: 1D Tensor that has 4 elements (min_y, min_x, max_y, max_x)
 519       of type float that represents the normalized coordinates between 0 and 1.
 520     augmentation_func: Augmentation function that will be applied to the
 521       subsection of image.
 522     *args: Additional parameters that will be passed into augmentation_func
 523       when it is called.
 524 
 525   Returns:
 526     A modified version of image, where the bbox location in the image will
 527     have `ugmentation_func applied to it.
 528   """
 529   image_height = tf.to_float(tf.shape(image)[0])
 530   image_width = tf.to_float(tf.shape(image)[1])
 531   min_y = tf.to_int32(image_height * bbox[0])
 532   min_x = tf.to_int32(image_width * bbox[1])
 533   max_y = tf.to_int32(image_height * bbox[2])
 534   max_x = tf.to_int32(image_width * bbox[3])
 535   image_height = tf.to_int32(image_height)
 536   image_width = tf.to_int32(image_width)
 537 
 538   # Clip to be sure the max values do not fall out of range.
 539   max_y = tf.minimum(max_y, image_height - 1)
 540   max_x = tf.minimum(max_x, image_width - 1)
 541 
 542   # Get the sub-tensor that is the image within the bounding box region.
 543   bbox_content = image[min_y:max_y + 1, min_x:max_x + 1, :]
 544 
 545   # Apply the augmentation function to the bbox portion of the image.
 546   augmented_bbox_content = augmentation_func(bbox_content, *args)
 547 
 548   # Pad the augmented_bbox_content and the mask to match the shape of original
 549   # image.
 550   augmented_bbox_content = tf.pad(augmented_bbox_content,
 551                                   [[min_y, (image_height - 1) - max_y],
 552                                    [min_x, (image_width - 1) - max_x],
 553                                    [0, 0]])
 554 
 555   # Create a mask that will be used to zero out a part of the original image.
 556   mask_tensor = tf.zeros_like(bbox_content)
 557 
 558   mask_tensor = tf.pad(mask_tensor,
 559                        [[min_y, (image_height - 1) - max_y],
 560                         [min_x, (image_width - 1) - max_x],
 561                         [0, 0]],
 562                        constant_values=1)
 563   # Replace the old bbox content with the new augmented content.
 564   image = image * mask_tensor + augmented_bbox_content
 565   return image

 568 def _concat_bbox(bbox, bboxes):
 569   """Helper function that concates bbox to bboxes along the first dimension."""
 570 
 571   # Note if all elements in bboxes are -1 (_INVALID_BOX), then this means
 572   # we discard bboxes and start the bboxes Tensor with the current bbox.
 573   bboxes_sum_check = tf.reduce_sum(bboxes)
 574   bbox = tf.expand_dims(bbox, 0)
 575   # This check will be true when it is an _INVALID_BOX
 576   bboxes = tf.cond(tf.equal(bboxes_sum_check, -4.0),
 577                    lambda: bbox,
 578                    lambda: tf.concat([bboxes, bbox], 0))
 579   return bboxes

 582 def _apply_bbox_augmentation_wrapper(image, bbox, new_bboxes, prob,
 583                                      augmentation_func, func_changes_bbox,
 584                                      *args):
 585   """Applies _apply_bbox_augmentation with probability prob.
 586 
 587   Args:
 588     image: 3D uint8 Tensor.
 589     bbox: 1D Tensor that has 4 elements (min_y, min_x, max_y, max_x)
 590       of type float that represents the normalized coordinates between 0 and 1.
 591     new_bboxes: 2D Tensor that is a list of the bboxes in the image after they
 592       have been altered by aug_func. These will only be changed when
 593       func_changes_bbox is set to true. Each bbox has 4 elements
 594       (min_y, min_x, max_y, max_x) of type float that are the normalized
 595       bbox coordinates between 0 and 1.
 596     prob: Float that is the probability of applying _apply_bbox_augmentation.
 597     augmentation_func: Augmentation function that will be applied to the
 598       subsection of image.
 599     func_changes_bbox: Boolean. Does augmentation_func return bbox in addition
 600       to image.
 601     *args: Additional parameters that will be passed into augmentation_func
 602       when it is called.
 603 
 604   Returns:
 605     A tuple. Fist element is a modified version of image, where the bbox
 606     location in the image will have augmentation_func applied to it if it is
 607     chosen to be called with probability `prob`. The second element is a
 608     Tensor of Tensors of length 4 that will contain the altered bbox after
 609     applying augmentation_func.
 610   """
 611   should_apply_op = tf.cast(
 612       tf.floor(tf.random_uniform([], dtype=tf.float32) + prob), tf.bool)
 613   if func_changes_bbox:
 614     augmented_image, bbox = tf.cond(
 615         should_apply_op,
 616         lambda: augmentation_func(image, bbox, *args),
 617         lambda: (image, bbox))
 618   else:
 619     augmented_image = tf.cond(
 620         should_apply_op,
 621         lambda: _apply_bbox_augmentation(image, bbox, augmentation_func, *args),
 622         lambda: image)
 623   new_bboxes = _concat_bbox(bbox, new_bboxes)
 624   return augmented_image, new_bboxes

 627 def _apply_multi_bbox_augmentation(image, bboxes, prob, aug_func,
 628                                    func_changes_bbox, *args):
 629   """Applies aug_func to the image for each bbox in bboxes.
 630 
 631   Args:
 632     image: 3D uint8 Tensor.
 633     bboxes: 2D Tensor that is a list of the bboxes in the image. Each bbox
 634       has 4 elements (min_y, min_x, max_y, max_x) of type float.
 635     prob: Float that is the probability of applying aug_func to a specific
 636       bounding box within the image.
 637     aug_func: Augmentation function that will be applied to the
 638       subsections of image indicated by the bbox values in bboxes.
 639     func_changes_bbox: Boolean. Does augmentation_func return bbox in addition
 640       to image.
 641     *args: Additional parameters that will be passed into augmentation_func
 642       when it is called.
 643 
 644   Returns:
 645     A modified version of image, where each bbox location in the image will
 646     have augmentation_func applied to it if it is chosen to be called with
 647     probability prob independently across all bboxes. Also the final
 648     bboxes are returned that will be unchanged if func_changes_bbox is set to
 649     false and if true, the new altered ones will be returned.
 650   """
 651   # Will keep track of the new altered bboxes after aug_func is repeatedly
 652   # applied. The -1 values are a dummy value and this first Tensor will be
 653   # removed upon appending the first real bbox.
 654   new_bboxes = tf.constant(_INVALID_BOX)
 655 
 656   # If the bboxes are empty, then just give it _INVALID_BOX. The result
 657   # will be thrown away.
 658   bboxes = tf.cond(tf.equal(tf.size(bboxes), 0),
 659                    lambda: tf.constant(_INVALID_BOX),
 660                    lambda: bboxes)
 661 
 662   bboxes = tf.ensure_shape(bboxes, (None, 4))
 663 
 664   # pylint:disable=g-long-lambda
 665   # pylint:disable=line-too-long
 666   wrapped_aug_func = lambda _image, bbox, _new_bboxes: _apply_bbox_augmentation_wrapper(
 667       _image, bbox, _new_bboxes, prob, aug_func, func_changes_bbox, *args)
 668   # pylint:enable=g-long-lambda
 669   # pylint:enable=line-too-long
 670 
 671   # Setup the while_loop.
 672   num_bboxes = tf.shape(bboxes)[0]  # We loop until we go over all bboxes.
 673   idx = tf.constant(0)  # Counter for the while loop.
 674 
 675   # Conditional function when to end the loop once we go over all bboxes
 676   # images_and_bboxes contain (_image, _new_bboxes)
 677   cond = lambda _idx, _images_and_bboxes: tf.less(_idx, num_bboxes)
 678 
 679   # Shuffle the bboxes so that the augmentation order is not deterministic if
 680   # we are not changing the bboxes with aug_func.
 681   if not func_changes_bbox:
 682     loop_bboxes = tf.random.shuffle(bboxes)
 683   else:
 684     loop_bboxes = bboxes
 685 
 686   # Main function of while_loop where we repeatedly apply augmentation on the
 687   # bboxes in the image.
 688   # pylint:disable=g-long-lambda
 689   body = lambda _idx, _images_and_bboxes: [
 690       _idx + 1, wrapped_aug_func(_images_and_bboxes[0],
 691                                  loop_bboxes[_idx],
 692                                  _images_and_bboxes[1])]
 693   # pylint:enable=g-long-lambda
 694 
 695   _, (image, new_bboxes) = tf.while_loop(
 696       cond, body, [idx, (image, new_bboxes)],
 697       shape_invariants=[idx.get_shape(),
 698                         (image.get_shape(), tf.TensorShape([None, 4]))])
 699 
 700   # Either return the altered bboxes or the original ones depending on if
 701   # we altered them in anyway.
 702   if func_changes_bbox:
 703     final_bboxes = new_bboxes
 704   else:
 705     final_bboxes = bboxes
 706   return image, final_bboxes

 795 def _rotate_bbox(bbox, image_height, image_width, degrees):
 796   """Rotates the bbox coordinated by degrees.
 797 
 798   Args:
 799     bbox: 1D Tensor that has 4 elements (min_y, min_x, max_y, max_x)
 800       of type float that represents the normalized coordinates between 0 and 1.
 801     image_height: Int, height of the image.
 802     image_width: Int, height of the image.
 803     degrees: Float, a scalar angle in degrees to rotate all images by. If
 804       degrees is positive the image will be rotated clockwise otherwise it will
 805       be rotated counterclockwise.
 806 
 807   Returns:
 808     A tensor of the same shape as bbox, but now with the rotated coordinates.
 809   """
 810   image_height, image_width = (
 811       tf.to_float(image_height), tf.to_float(image_width))
 812 
 813   # Convert from degrees to radians.
 814   degrees_to_radians = math.pi / 180.0
 815   radians = degrees * degrees_to_radians
 816 
 817   # Translate the bbox to the center of the image and turn the normalized 0-1
 818   # coordinates to absolute pixel locations.
 819   # Y coordinates are made negative as the y axis of images goes down with
 820   # increasing pixel values, so we negate to make sure x axis and y axis points
 821   # are in the traditionally positive direction.
 822   min_y = -tf.to_int32(image_height * (bbox[0] - 0.5))
 823   min_x = tf.to_int32(image_width * (bbox[1] - 0.5))
 824   max_y = -tf.to_int32(image_height * (bbox[2] - 0.5))
 825   max_x = tf.to_int32(image_width * (bbox[3] - 0.5))
 826   coordinates = tf.stack(
 827       [[min_y, min_x], [min_y, max_x], [max_y, min_x], [max_y, max_x]])
 828   coordinates = tf.cast(coordinates, tf.float32)
 829   # Rotate the coordinates according to the rotation matrix clockwise if
 830   # radians is positive, else negative
 831   rotation_matrix = tf.stack(
 832       [[tf.cos(radians), tf.sin(radians)],
 833        [-tf.sin(radians), tf.cos(radians)]])
 834   new_coords = tf.cast(
 835       tf.matmul(rotation_matrix, tf.transpose(coordinates)), tf.int32)
 836   # Find min/max values and convert them back to normalized 0-1 floats.
 837   min_y = -(tf.to_float(tf.reduce_max(new_coords[0, :])) / image_height - 0.5)
 838   min_x = tf.to_float(tf.reduce_min(new_coords[1, :])) / image_width + 0.5
 839   max_y = -(tf.to_float(tf.reduce_min(new_coords[0, :])) / image_height - 0.5)
 840   max_x = tf.to_float(tf.reduce_max(new_coords[1, :])) / image_width + 0.5
 841 
 842   # Clip the bboxes to be sure the fall between [0, 1].
 843   min_y, min_x, max_y, max_x = _clip_bbox(min_y, min_x, max_y, max_x)
 844   min_y, min_x, max_y, max_x = _check_bbox_area(min_y, min_x, max_y, max_x)
 845   return tf.stack([min_y, min_x, max_y, max_x])

 891 def _shift_bbox(bbox, image_height, image_width, pixels, shift_horizontal):
 892   """Shifts the bbox coordinates by pixels.
 893 
 894   Args:
 895     bbox: 1D Tensor that has 4 elements (min_y, min_x, max_y, max_x)
 896       of type float that represents the normalized coordinates between 0 and 1.
 897     image_height: Int, height of the image.
 898     image_width: Int, width of the image.
 899     pixels: An int. How many pixels to shift the bbox.
 900     shift_horizontal: Boolean. If true then shift in X dimension else shift in
 901       Y dimension.
 902 
 903   Returns:
 904     A tensor of the same shape as bbox, but now with the shifted coordinates.
 905   """
 906   pixels = tf.to_int32(pixels)
 907   # Convert bbox to integer pixel locations.
 908   min_y = tf.to_int32(tf.to_float(image_height) * bbox[0])
 909   min_x = tf.to_int32(tf.to_float(image_width) * bbox[1])
 910   max_y = tf.to_int32(tf.to_float(image_height) * bbox[2])
 911   max_x = tf.to_int32(tf.to_float(image_width) * bbox[3])
 912 
 913   if shift_horizontal:
 914     min_x = tf.maximum(0, min_x - pixels)
 915     max_x = tf.minimum(image_width, max_x - pixels)
 916   else:
 917     min_y = tf.maximum(0, min_y - pixels)
 918     max_y = tf.minimum(image_height, max_y - pixels)
 919 
 920   # Convert bbox back to floats.
 921   min_y = tf.to_float(min_y) / tf.to_float(image_height)
 922   min_x = tf.to_float(min_x) / tf.to_float(image_width)
 923   max_y = tf.to_float(max_y) / tf.to_float(image_height)
 924   max_x = tf.to_float(max_x) / tf.to_float(image_width)
 925 
 926   # Clip the bboxes to be sure the fall between [0, 1].
 927   min_y, min_x, max_y, max_x = _clip_bbox(min_y, min_x, max_y, max_x)
 928   min_y, min_x, max_y, max_x = _check_bbox_area(min_y, min_x, max_y, max_x)
 929   return tf.stack([min_y, min_x, max_y, max_x])

 988 def _shear_bbox(bbox, image_height, image_width, level, shear_horizontal):
 989   """Shifts the bbox according to how the image was sheared.
 990 
 991   Args:
 992     bbox: 1D Tensor that has 4 elements (min_y, min_x, max_y, max_x)
 993       of type float that represents the normalized coordinates between 0 and 1.
 994     image_height: Int, height of the image.
 995     image_width: Int, height of the image.
 996     level: Float. How much to shear the image.
 997     shear_horizontal: If true then shear in X dimension else shear in
 998       the Y dimension.
 999 
1000   Returns:
1001     A tensor of the same shape as bbox, but now with the shifted coordinates.
1002   """
1003   image_height, image_width = (
1004       tf.to_float(image_height), tf.to_float(image_width))
1005 
1006   # Change bbox coordinates to be pixels.
1007   min_y = tf.to_int32(image_height * bbox[0])
1008   min_x = tf.to_int32(image_width * bbox[1])
1009   max_y = tf.to_int32(image_height * bbox[2])
1010   max_x = tf.to_int32(image_width * bbox[3])
1011   coordinates = tf.stack(
1012       [[min_y, min_x], [min_y, max_x], [max_y, min_x], [max_y, max_x]])
1013   coordinates = tf.cast(coordinates, tf.float32)
1014 
1015   # Shear the coordinates according to the translation matrix.
1016   if shear_horizontal:
1017     translation_matrix = tf.stack(
1018         [[1, 0], [-level, 1]])
1019   else:
1020     translation_matrix = tf.stack(
1021         [[1, -level], [0, 1]])
1022   translation_matrix = tf.cast(translation_matrix, tf.float32)
1023   new_coords = tf.cast(
1024       tf.matmul(translation_matrix, tf.transpose(coordinates)), tf.int32)
1025 
1026   # Find min/max values and convert them back to floats.
1027   min_y = tf.to_float(tf.reduce_min(new_coords[0, :])) / image_height
1028   min_x = tf.to_float(tf.reduce_min(new_coords[1, :])) / image_width
1029   max_y = tf.to_float(tf.reduce_max(new_coords[0, :])) / image_height
1030   max_x = tf.to_float(tf.reduce_max(new_coords[1, :])) / image_width
1031 
1032   # Clip the bboxes to be sure the fall between [0, 1].
1033   min_y, min_x, max_y, max_x = _clip_bbox(min_y, min_x, max_y, max_x)
1034   min_y, min_x, max_y, max_x = _check_bbox_area(min_y, min_x, max_y, max_x)
1035   return tf.stack([min_y, min_x, max_y, max_x])

1038 def shear_with_bboxes(image, bboxes, level, replace, shear_horizontal):
1039   """Applies Shear Transformation to the image and shifts the bboxes.
1040 
1041   Args:

1046     level: Float. How much to shear the image. This value will be between

1073 def autocontrast(image):
1074   """Implements Autocontrast function from PIL using TF ops.
1075 
1076   Args:
1077     image: A 3D uint8 tensor.
1078 
1079   Returns:
1080     The image after it has had autocontrast applied to it and will be of type
1081     uint8.
1082   """
1083 
1084   def scale_channel(image):
1085     """Scale the 2D image using the autocontrast rule."""
1086     # A possibly cheaper version can be done using cumsum/unique_with_counts
1087     # over the histogram values, rather than iterating over the entire image.
1088     # to compute mins and maxes.
1089     lo = tf.to_float(tf.reduce_min(image))
1090     hi = tf.to_float(tf.reduce_max(image))
1091 
1092     # Scale the image, making the lowest value 0 and the highest value 255.
1093     def scale_values(im):
1094       scale = 255.0 / (hi - lo)
1095       offset = -lo * scale
1096       im = tf.to_float(im) * scale + offset
1097       im = tf.clip_by_value(im, 0.0, 255.0)
1098       return tf.cast(im, tf.uint8)
1099 
1100     result = tf.cond(hi > lo, lambda: scale_values(image), lambda: image)
1101     return result
1102 
1103   # Assumes RGB for now.  Scales each channel independently
1104   # and then stacks the result.
1105   s1 = scale_channel(image[:, :, 0])
1106   s2 = scale_channel(image[:, :, 1])
1107   s3 = scale_channel(image[:, :, 2])
1108   image = tf.stack([s1, s2, s3], 2)
1109   return image

1112 def sharpness(image, factor):
1113   """Implements Sharpness function from PIL using TF ops."""
1114   orig_image = image
1115   image = tf.cast(image, tf.float32)
1116   # Make image 4D for conv operation.
1117   image = tf.expand_dims(image, 0)
1118   # SMOOTH PIL Kernel.
1119   kernel = tf.constant(
1120       [[1, 1, 1], [1, 5, 1], [1, 1, 1]], dtype=tf.float32,
1121       shape=[3, 3, 1, 1]) / 13.
1122   # Tile across channel dimension.
1123   kernel = tf.tile(kernel, [1, 1, 3, 1])
1124   strides = [1, 1, 1, 1]
1125   degenerate = tf.nn.depthwise_conv2d(
1126       image, kernel, strides, padding='VALID', rate=[1, 1])
1127   degenerate = tf.clip_by_value(degenerate, 0.0, 255.0)
1128   degenerate = tf.squeeze(tf.cast(degenerate, tf.uint8), [0])
1129 
1130   # For the borders of the resulting image, fill in the values of the
1131   # original image.
1132   mask = tf.ones_like(degenerate)
1133   padded_mask = tf.pad(mask, [[1, 1], [1, 1], [0, 0]])
1134   padded_degenerate = tf.pad(degenerate, [[1, 1], [1, 1], [0, 0]])
1135   result = tf.where(tf.equal(padded_mask, 1), padded_degenerate, orig_image)
1136 
1137   # Blend the final result.
1138   return blend(result, orig_image, factor)

1141 def equalize(image):
1142   """Implements Equalize function from PIL using TF ops."""
1143   def scale_channel(im, c):
1144     """Scale the data in the channel to implement equalize."""
1145     im = tf.cast(im[:, :, c], tf.int32)
1146     # Compute the histogram of the image channel.
1147     histo = tf.histogram_fixed_width(im, [0, 255], nbins=256)
1148 
1149     # For the purposes of computing the step, filter out the nonzeros.
1150     nonzero = tf.where(tf.not_equal(histo, 0))
1151     nonzero_histo = tf.reshape(tf.gather(histo, nonzero), [-1])
1152     step = (tf.reduce_sum(nonzero_histo) - nonzero_histo[-1]) // 255
1153 
1154     def build_lut(histo, step):
1155       # Compute the cumulative sum, shifting by step // 2
1156       # and then normalization by step.
1157       lut = (tf.cumsum(histo) + (step // 2)) // step
1158       # Shift lut, prepending with 0.
1159       lut = tf.concat([[0], lut[:-1]], 0)
1160       # Clip the counts to be in range.  This is done
1161       # in the C code for image.point.
1162       return tf.clip_by_value(lut, 0, 255)
1163 
1164     # If step is zero, return the original image.  Otherwise, build
1165     # lut from the full histogram and step and then index from it.
1166     result = tf.cond(tf.equal(step, 0),
1167                      lambda: im,
1168                      lambda: tf.gather(build_lut(histo, step), im))
1169 
1170     return tf.cast(result, tf.uint8)
1171 
1172   # Assumes RGB for now.  Scales each channel independently
1173   # and then stacks the result.
1174   s1 = scale_channel(image, 0)
1175   s2 = scale_channel(image, 1)
1176   s3 = scale_channel(image, 2)
1177   image = tf.stack([s1, s2, s3], 2)
1178   return image

1189 def unwrap(image, replace):
1190   """Unwraps an image produced by wrap.
1191 
1192   Where there is a 0 in the last channel for every spatial position,
1193   the rest of the three channels in that spatial dimension are grayed
1194   (set to 128).  Operations like translate and shear on a wrapped
1195   Tensor will leave 0s in empty locations.  Some transformations look
1196   at the intensity of values to do preprocessing, and we want these
1197   empty pixels to assume the 'average' value, rather than pure black.
1198 
1199 
1200   Args:
1201     image: A 3D Image Tensor with 4 channels.
1202     replace: A one or three value 1D tensor to fill empty pixels.
1203 
1204   Returns:
1205     image: A 3D image Tensor with 3 channels.
1206   """
1207   image_shape = tf.shape(image)
1208   # Flatten the spatial dimensions.
1209   flattened_image = tf.reshape(image, [-1, image_shape[2]])
1210 
1211   # Find all pixels where the last channel is zero.
1212   alpha_channel = flattened_image[:, 3]
1213 
1214   replace = tf.concat([replace, tf.ones([1], image.dtype)], 0)
1215 
1216   # Where they are zero, fill them in with 'replace'.
1217   flattened_image = tf.where(
1218       tf.equal(alpha_channel, 0),
1219       tf.ones_like(flattened_image, dtype=image.dtype) * replace,
1220       flattened_image)
1221 
1222   image = tf.reshape(flattened_image, image_shape)
1223   image = tf.slice(image, [0, 0, 0], [image_shape[0], image_shape[1], 3])
1224   return image

1227 def _cutout_inside_bbox(image, bbox, pad_fraction):
1228   """Generates cutout mask and the mean pixel value of the bbox.
1229 
1230   First a location is randomly chosen within the image as the center where the
1231   cutout mask will be applied. Note this can be towards the boundaries of the
1232   image, so the full cutout mask may not be applied.
1233 
1234   Args:
1235     image: 3D uint8 Tensor.
1236     bbox: 1D Tensor that has 4 elements (min_y, min_x, max_y, max_x)
1237       of type float that represents the normalized coordinates between 0 and 1.
1238     pad_fraction: Float that specifies how large the cutout mask should be in
1239       in reference to the size of the original bbox. If pad_fraction is 0.25,
1240       then the cutout mask will be of shape
1241       (0.25 * bbox height, 0.25 * bbox width).
1242 
1243   Returns:
1244     A tuple. Fist element is a tensor of the same shape as image where each
1245     element is either a 1 or 0 that is used to determine where the image
1246     will have cutout applied. The second element is the mean of the pixels
1247     in the image where the bbox is located.
1248   """
1249   image_height = tf.shape(image)[0]
1250   image_width = tf.shape(image)[1]
1251   # Transform from shape [1, 4] to [4].
1252   bbox = tf.squeeze(bbox)
1253 
1254   min_y = tf.to_int32(tf.to_float(image_height) * bbox[0])
1255   min_x = tf.to_int32(tf.to_float(image_width) * bbox[1])
1256   max_y = tf.to_int32(tf.to_float(image_height) * bbox[2])
1257   max_x = tf.to_int32(tf.to_float(image_width) * bbox[3])
1258 
1259   # Calculate the mean pixel values in the bounding box, which will be used
1260   # to fill the cutout region.
1261   mean = tf.reduce_mean(image[min_y:max_y + 1, min_x:max_x + 1],
1262                         reduction_indices=[0, 1])
1263 
1264   # Cutout mask will be size pad_size_heigh * 2 by pad_size_width * 2 if the
1265   # region lies entirely within the bbox.
1266   box_height = max_y - min_y + 1
1267   box_width = max_x - min_x + 1
1268   pad_size_height = tf.to_int32(pad_fraction * (box_height / 2))
1269   pad_size_width = tf.to_int32(pad_fraction * (box_width / 2))
1270 
1271   # Sample the center location in the image where the zero mask will be applied.
1272   cutout_center_height = tf.random_uniform(
1273       shape=[], minval=min_y, maxval=max_y+1,
1274       dtype=tf.int32)
1275 
1276   cutout_center_width = tf.random_uniform(
1277       shape=[], minval=min_x, maxval=max_x+1,
1278       dtype=tf.int32)
1279 
1280   lower_pad = tf.maximum(
1281       0, cutout_center_height - pad_size_height)
1282   upper_pad = tf.maximum(
1283       0, image_height - cutout_center_height - pad_size_height)
1284   left_pad = tf.maximum(
1285       0, cutout_center_width - pad_size_width)
1286   right_pad = tf.maximum(
1287       0, image_width - cutout_center_width - pad_size_width)
1288 
1289   cutout_shape = [image_height - (lower_pad + upper_pad),
1290                   image_width - (left_pad + right_pad)]
1291   padding_dims = [[lower_pad, upper_pad], [left_pad, right_pad]]
1292 
1293   mask = tf.pad(
1294       tf.zeros(cutout_shape, dtype=image.dtype),
1295       padding_dims, constant_values=1)
1296 
1297   mask = tf.expand_dims(mask, 2)
1298   mask = tf.tile(mask, [1, 1, 3])
1299 
1300   return mask, mean

1303 def bbox_cutout(image, bboxes, pad_fraction, replace_with_mean):
1304   """Applies cutout to the image according to bbox information.
1305 
1306   This is a cutout variant that using bbox information to make more informed
1307   decisions on where to place the cutout mask.
1308 
1309   Args:
1310     image: 3D uint8 Tensor.
1311     bboxes: 2D Tensor that is a list of the bboxes in the image. Each bbox
1312       has 4 elements (min_y, min_x, max_y, max_x) of type float with values
1313       between [0, 1].
1314     pad_fraction: Float that specifies how large the cutout mask should be in
1315       in reference to the size of the original bbox. If pad_fraction is 0.25,
1316       then the cutout mask will be of shape
1317       (0.25 * bbox height, 0.25 * bbox width).
1318     replace_with_mean: Boolean that specified what value should be filled in
1319       where the cutout mask is applied. Since the incoming image will be of
1320       uint8 and will not have had any mean normalization applied, by default
1321       we set the value to be 128. If replace_with_mean is True then we find
1322       the mean pixel values across the channel dimension and use those to fill
1323       in where the cutout mask is applied.
1324 
1325   Returns:
1326     A tuple. First element is a tensor of the same shape as image that has
1327     cutout applied to it. Second element is the bboxes that were passed in
1328     that will be unchanged.
1329   """
1330   def apply_bbox_cutout(image, bboxes, pad_fraction):
1331     """Applies cutout to a single bounding box within image."""
1332     # Choose a single bounding box to apply cutout to.
1333     random_index = tf.random_uniform(
1334         shape=[], maxval=tf.shape(bboxes)[0], dtype=tf.int32)
1335     # Select the corresponding bbox and apply cutout.
1336     chosen_bbox = tf.gather(bboxes, random_index)
1337     mask, mean = _cutout_inside_bbox(image, chosen_bbox, pad_fraction)
1338 
1339     # When applying cutout we either set the pixel value to 128 or to the mean
1340     # value inside the bbox.
1341     replace = mean if replace_with_mean else 128
1342 
1343     # Apply the cutout mask to the image. Where the mask is 0 we fill it with
1344     # `replace`.
1345     image = tf.where(
1346         tf.equal(mask, 0),
1347         tf.cast(tf.ones_like(image, dtype=image.dtype) * replace,
1348                 dtype=image.dtype),
1349         image)
1350     return image
1351 
1352   # Check to see if there are boxes, if so then apply boxcutout.
1353   image = tf.cond(tf.equal(tf.size(bboxes), 0), lambda: image,
1354                   lambda: apply_bbox_cutout(image, bboxes, pad_fraction))
1355 
1356   return image, bboxes

1489 def _parse_policy_info(name, prob, level, replace_value, augmentation_hparams):
1490   """Return the function that corresponds to `name` and update `level` param."""
1491   func = NAME_TO_FUNC[name]
1492   args = level_to_arg(augmentation_hparams)[name](level)
1493 
1494   if six.PY2:
1495     # pylint: disable=deprecated-method
1496     arg_spec = inspect.getargspec(func)
1497     # pylint: enable=deprecated-method
1498   else:
1499     arg_spec = inspect.getfullargspec(func)
1500 
1501   # Check to see if prob is passed into function. This is used for operations
1502   # where we alter bboxes independently.
1503   # pytype:disable=wrong-arg-types
1504   if 'prob' in arg_spec[0]:
1505     args = tuple([prob] + list(args))
1506   # pytype:enable=wrong-arg-types
1507 
1508   # Add in replace arg if it is required for the function that is being called.
1509   if 'replace' in arg_spec[0]:
1510     # Make sure replace is the final argument
1511     assert 'replace' == arg_spec[0][-1]
1512     args = tuple(list(args) + [replace_value])
1513 
1514   # Add bboxes as the second positional argument for the function if it does
1515   # not already exist.
1516   if 'bboxes' not in arg_spec[0]:
1517     func = bbox_wrapper(func)
1518   return (func, prob, args)

1521 def _apply_func_with_prob(func, image, args, prob, bboxes):
1522   """Apply `func` to image w/ `args` as input with probability `prob`."""
1523   assert isinstance(args, tuple)
1524   if six.PY2:
1525     # pylint: disable=deprecated-method
1526     arg_spec = inspect.getargspec(func)
1527     # pylint: enable=deprecated-method
1528   else:
1529     arg_spec = inspect.getfullargspec(func)
1530   assert 'bboxes' == arg_spec[0][1]
1531 
1532   # If prob is a function argument, then this randomness is being handled
1533   # inside the function, so make sure it is always called.
1534   if 'prob' in arg_spec[0]:
1535     prob = 1.0
1536 
1537   # Apply the function with probability `prob`.
1538   should_apply_op = tf.cast(
1539       tf.floor(tf.random_uniform([], dtype=tf.float32) + prob), tf.bool)
1540   augmented_image, augmented_bboxes = tf.cond(
1541       should_apply_op,
1542       lambda: func(image, bboxes, *args),
1543       lambda: (image, bboxes))
1544   return augmented_image, augmented_bboxes

1547 def select_and_apply_random_policy(policies, image, bboxes):
1548   """Select a random policy from `policies` and apply it to `image`."""
1549   policy_to_select = tf.random_uniform([], maxval=len(policies), dtype=tf.int32)
1550   # Note that using tf.case instead of tf.conds would result in significantly
1551   # larger graphs and would even break export for some larger policies.
1552   for (i, policy) in enumerate(policies):
1553     image, bboxes = tf.cond(
1554         tf.equal(i, policy_to_select),
1555         lambda selected_policy=policy: selected_policy(image, bboxes),
1556         lambda: (image, bboxes))
1557   return (image, bboxes)

1560 def build_and_apply_nas_policy(policies, image, bboxes,
1561                                augmentation_hparams):
1562   """Build a policy from the given policies passed in and apply to image.
1563 
1564   Args:
1565     policies: list of lists of tuples in the form `(func, prob, level)`, `func`
1566       is a string name of the augmentation function, `prob` is the probability
1567       of applying the `func` operation, `level` is the input argument for
1568       `func`.
1569     image: tf.Tensor that the resulting policy will be applied to.
1570     bboxes:
1571     augmentation_hparams: Hparams associated with the NAS learned policy.
1572 
1573   Returns:
1574     A version of image that now has data augmentation applied to it based on
1575     the `policies` pass into the function. Additionally, returns bboxes if
1576     a value for them is passed in that is not None
1577   """
1578   replace_value = [128, 128, 128]
1579 
1580   # func is the string name of the augmentation function, prob is the
1581   # probability of applying the operation and level is the parameter associated
1582   # with the tf op.
1583 
1584   # tf_policies are functions that take in an image and return an augmented
1585   # image.
1586   tf_policies = []
1587   for policy in policies:
1588     tf_policy = []
1589     # Link string name to the correct python function and make sure the correct
1590     # argument is passed into that function.
1591     for policy_info in policy:
1592       policy_info = list(policy_info) + [replace_value, augmentation_hparams]
1593 
1594       tf_policy.append(_parse_policy_info(*policy_info))
1595     # Now build the tf policy that will apply the augmentation procedue
1596     # on image.
1597     def make_final_policy(tf_policy_):
1598       def final_policy(image_, bboxes_):
1599         for func, prob, args in tf_policy_:
1600           image_, bboxes_ = _apply_func_with_prob(
1601               func, image_, args, prob, bboxes_)
1602         return image_, bboxes_
1603       return final_policy
1604     tf_policies.append(make_final_policy(tf_policy))
1605 
1606   augmented_image, augmented_bbox = select_and_apply_random_policy(
1607       tf_policies, image, bboxes)
1608   # If no bounding boxes were specified, then just return the images.
1609   return (augmented_image, augmented_bbox)

1613 def distort_image_with_autoaugment(image, bboxes, augmentation_name):
1614   """Applies the AutoAugment policy to `image` and `bboxes`.
1615 
1616   Args:
1617     image: `Tensor` of shape [height, width, 3] representing an image.
1618     bboxes: `Tensor` of shape [N, 4] representing ground truth boxes that are
1619       normalized between [0, 1].
1620     augmentation_name: The name of the AutoAugment policy to use. The available
1621       options are `v0`, `v1`, `v2`, `v3` and `test`. `v0` is the policy used for
1622       all of the results in the paper and was found to achieve the best results
1623       on the COCO dataset. `v1`, `v2` and `v3` are additional good policies
1624       found on the COCO dataset that have slight variation in what operations
1625       were used during the search procedure along with how many operations are
1626       applied in parallel to a single image (2 vs 3).
1627 
1628   Returns:
1629     A tuple containing the augmented versions of `image` and `bboxes`.
1630   """
1631   image = tf.cast(image, tf.uint8)
1632   available_policies = {'v0': policy_v0, 'v1': policy_v1, 'v2': policy_v2,
1633                         'v3': policy_v3, 'test': policy_vtest}
1634   if augmentation_name not in available_policies:
1635     raise ValueError('Invalid augmentation_name: {}'.format(augmentation_name))
1636 
1637   policy = available_policies[augmentation_name]()
1638   # Hparams that will be used for AutoAugment.
1639   augmentation_hparams = contrib_training.HParams(
1640       cutout_max_pad_fraction=0.75,
1641       cutout_bbox_replace_with_mean=False,
1642       cutout_const=100,
1643       translate_const=250,
1644       cutout_bbox_const=50,
1645       translate_bbox_const=120)
1646 
1647   augmented_image, augmented_bbox = (
1648       build_and_apply_nas_policy(policy, image, bboxes, augmentation_hparams))
1649   augmented_image = tf.cast(augmented_image, tf.float32)
1650   return augmented_image, augmented_bbox
