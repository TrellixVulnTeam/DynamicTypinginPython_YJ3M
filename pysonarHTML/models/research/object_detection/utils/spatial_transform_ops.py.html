  24 def _coordinate_vector_1d(start, end, size, align_endpoints):
  25   """Generates uniformly spaced coordinate vector.
  26 
  27   Args:
  28     start: A float tensor of shape [batch, num_boxes] indicating start values.
  29     end: A float tensor of shape [batch, num_boxes] indicating end values.
  30     size: Number of points in coordinate vector.
  31     align_endpoints: Whether to align first and last points exactly to
  32       endpoints.
  33 
  34   Returns:
  35     A 3D float tensor of shape [batch, num_boxes, size] containing grid
  36     coordinates.
  37   """
  38   start = tf.expand_dims(start, -1)
  39   end = tf.expand_dims(end, -1)
  40   length = end - start
  41   if align_endpoints:
  42     relative_grid_spacing = tf.linspace(0.0, 1.0, size)
  43     offset = 0 if size > 1 else length / 2
  44   else:
  45     relative_grid_spacing = tf.linspace(0.0, 1.0, size + 1)[:-1]
  46     offset = length / (2 * size)
  47   relative_grid_spacing = tf.reshape(relative_grid_spacing, [1, 1, size])
  48   relative_grid_spacing = tf.cast(relative_grid_spacing, dtype=start.dtype)
  49   absolute_grid = start + offset + relative_grid_spacing * length
  50   return absolute_grid

  53 def box_grid_coordinate_vectors(boxes, size_y, size_x, align_corners=False):
  54   """Generates coordinate vectors for a `size x size` grid in boxes.
  55 
  56   Each box is subdivided uniformly into a grid consisting of size x size
  57   rectangular cells. This function returns coordinate vectors describing
  58   the center of each cell.
  59 
  60   If `align_corners` is true, grid points are uniformly spread such that the
  61   corner points on the grid exactly overlap corners of the boxes.
  62 
  63   Note that output coordinates are expressed in the same coordinate frame as
  64   input boxes.
  65 
  66   Args:
  67     boxes: A float tensor of shape [batch, num_boxes, 4] containing boxes of the
  68       form [ymin, xmin, ymax, xmax].
  69     size_y: Size of the grid in y axis.
  70     size_x: Size of the grid in x axis.
  71     align_corners: Whether to align the corner grid points exactly with box
  72       corners.
  73 
  74   Returns:
  75     box_grid_y: A float tensor of shape [batch, num_boxes, size_y] containing y
  76       coordinates for grid points.
  77     box_grid_x: A float tensor of shape [batch, num_boxes, size_x] containing x
  78       coordinates for grid points.
  79   """
  80   ymin, xmin, ymax, xmax = tf.unstack(boxes, axis=-1)
  81   box_grid_y = _coordinate_vector_1d(ymin, ymax, size_y, align_corners)
  82   box_grid_x = _coordinate_vector_1d(xmin, xmax, size_x, align_corners)
  83   return box_grid_y, box_grid_x

  86 def feature_grid_coordinate_vectors(box_grid_y, box_grid_x):
  87   """Returns feature grid point coordinate vectors for bilinear interpolation.
  88 
  89   Box grid is specified in absolute coordinate system with origin at left top
  90   (0, 0). The returned coordinate vectors contain 0-based feature point indices.
  91 
  92   This function snaps each point in the box grid to nearest 4 points on the
  93   feature map.
  94 
  95   In this function we also follow the convention of treating feature pixels as
  96   point objects with no spatial extent.
  97 
  98   Args:
  99     box_grid_y: A float tensor of shape [batch, num_boxes, size] containing y
 100       coordinate vector of the box grid.
 101     box_grid_x: A float tensor of shape [batch, num_boxes, size] containing x
 102       coordinate vector of the box grid.
 103 
 104   Returns:
 105     feature_grid_y0: An int32 tensor of shape [batch, num_boxes, size]
 106       containing y coordinate vector for the top neighbors.
 107     feature_grid_x0: A int32 tensor of shape [batch, num_boxes, size]
 108       containing x coordinate vector for the left neighbors.
 109     feature_grid_y1: A int32 tensor of shape [batch, num_boxes, size]
 110       containing y coordinate vector for the bottom neighbors.
 111     feature_grid_x1: A int32 tensor of shape [batch, num_boxes, size]
 112       containing x coordinate vector for the right neighbors.
 113   """
 114   feature_grid_y0 = tf.floor(box_grid_y)
 115   feature_grid_x0 = tf.floor(box_grid_x)
 116   feature_grid_y1 = tf.floor(box_grid_y + 1)
 117   feature_grid_x1 = tf.floor(box_grid_x + 1)
 118   feature_grid_y0 = tf.cast(feature_grid_y0, dtype=tf.int32)
 119   feature_grid_y1 = tf.cast(feature_grid_y1, dtype=tf.int32)
 120   feature_grid_x0 = tf.cast(feature_grid_x0, dtype=tf.int32)
 121   feature_grid_x1 = tf.cast(feature_grid_x1, dtype=tf.int32)
 122   return (feature_grid_y0, feature_grid_x0, feature_grid_y1, feature_grid_x1)

 125 def _valid_indicator(feature_grid_y, feature_grid_x, true_feature_shapes):
 126   """Computes a indicator vector for valid indices.
 127 
 128   Computes an indicator vector which is true for points on feature map and
 129   false for points off feature map.
 130 
 131   Args:
 132     feature_grid_y: An int32 tensor of shape [batch, num_boxes, size_y]
 133       containing y coordinate vector.
 134     feature_grid_x: An int32 tensor of shape [batch, num_boxes, size_x]
 135       containing x coordinate vector.
 136     true_feature_shapes: A int32 tensor of shape [batch, num_boxes, 2]
 137       containing valid height and width of feature maps. Feature maps are
 138       assumed to be aligned to the left top corner.
 139 
 140   Returns:
 141     indices: A 1D bool tensor indicating valid feature indices.
 142   """
 143   height = tf.cast(true_feature_shapes[:, :, 0:1], dtype=feature_grid_y.dtype)
 144   width = tf.cast(true_feature_shapes[:, :, 1:2], dtype=feature_grid_x.dtype)
 145   valid_indicator = tf.logical_and(
 146       tf.expand_dims(
 147           tf.logical_and(feature_grid_y >= 0, tf.less(feature_grid_y, height)),
 148           3),
 149       tf.expand_dims(
 150           tf.logical_and(feature_grid_x >= 0, tf.less(feature_grid_x, width)),
 151           2))
 152   return tf.reshape(valid_indicator, [-1])

 155 def ravel_indices(feature_grid_y, feature_grid_x, num_levels, height, width,
 156                   box_levels):
 157   """Returns grid indices in a flattened feature map of shape [-1, channels].
 158 
 159   The returned 1-D array can be used to gather feature grid points from a
 160   feature map that has been flattened from [batch, num_levels, max_height,
 161   max_width, channels] to [batch * num_levels * max_height * max_width,
 162   channels].
 163 
 164   Args:
 165     feature_grid_y: An int32 tensor of shape [batch, num_boxes, size_y]
 166       containing y coordinate vector.
 167     feature_grid_x: An int32 tensor of shape [batch, num_boxes, size_x]
 168       containing x coordinate vector.
 169     num_levels: Number of feature levels.
 170     height: An integer indicating the padded height of feature maps.
 171     width: An integer indicating the padded width of feature maps.
 172     box_levels: An int32 tensor of shape [batch, num_boxes] indicating
 173       feature level assigned to each box.
 174 
 175   Returns:
 176     indices: A 1D int32 tensor containing feature point indices in a flattened
 177       feature grid.
 178   """
 179   num_boxes = tf.shape(feature_grid_y)[1]
 180   batch_size = tf.shape(feature_grid_y)[0]
 181   size_y = tf.shape(feature_grid_y)[2]
 182   size_x = tf.shape(feature_grid_x)[2]
 183   height_dim_offset = width
 184   level_dim_offset = height * height_dim_offset
 185   batch_dim_offset = num_levels * level_dim_offset
 186 
 187   batch_dim_indices = (
 188       tf.reshape(
 189           tf.range(batch_size) * batch_dim_offset, [batch_size, 1, 1, 1]) *
 190       tf.ones([1, num_boxes, size_y, size_x], dtype=tf.int32))
 191   box_level_indices = (
 192       tf.reshape(box_levels * level_dim_offset, [batch_size, num_boxes, 1, 1]) *
 193       tf.ones([1, 1, size_y, size_x], dtype=tf.int32))
 194   height_indices = (
 195       tf.reshape(feature_grid_y * height_dim_offset,
 196                  [batch_size, num_boxes, size_y, 1]) *
 197       tf.ones([1, 1, 1, size_x], dtype=tf.int32))
 198   width_indices = (
 199       tf.reshape(feature_grid_x, [batch_size, num_boxes, 1, size_x])
 200       * tf.ones([1, 1, size_y, 1], dtype=tf.int32))
 201   indices = (
 202       batch_dim_indices + box_level_indices + height_indices + width_indices)
 203   flattened_indices = tf.reshape(indices, [-1])
 204   return flattened_indices

 207 def pad_to_max_size(features):
 208   """Pads features to max height and max width and stacks them up.
 209 
 210   Args:
 211     features: A list of num_levels 4D float tensors of shape [batch, height_i,
 212       width_i, channels] containing feature maps.
 213 
 214   Returns:
 215     stacked_features: A 5D float tensor of shape [batch, num_levels, max_height,
 216       max_width, channels] containing stacked features.
 217     true_feature_shapes: A 2D int32 tensor of shape [num_levels, 2] containing
 218       height and width of the feature maps before padding.
 219   """
 220   if len(features) == 1:
 221     return tf.expand_dims(features[0],
 222                           1), tf.expand_dims(tf.shape(features[0])[1:3], 0)
 223 
 224   if all([feature.shape.is_fully_defined() for feature in features]):
 225     heights = [feature.shape[1] for feature in features]
 226     widths = [feature.shape[2] for feature in features]
 227     max_height = max(heights)
 228     max_width = max(widths)
 229   else:
 230     heights = [tf.shape(feature)[1] for feature in features]
 231     widths = [tf.shape(feature)[2] for feature in features]
 232     max_height = tf.reduce_max(heights)
 233     max_width = tf.reduce_max(widths)
 234   features_all = [
 235       tf.image.pad_to_bounding_box(feature, 0, 0, max_height,
 236                                    max_width) for feature in features
 237   ]
 238   features_all = tf.stack(features_all, axis=1)
 239   true_feature_shapes = tf.stack([tf.shape(feature)[1:3]
 240                                   for feature in features])
 241   return features_all, true_feature_shapes

 244 def _gather_valid_indices(tensor, indices, padding_value=0.0):
 245   """Gather values for valid indices.
 246 
 247   TODO(rathodv): We can't use ops.gather_with_padding_values due to cyclic
 248   dependency. Start using it after migrating all users of spatial ops to import
 249   this module directly rather than util/ops.py
 250 
 251   Args:
 252     tensor: A tensor to gather valid values from.
 253     indices: A 1-D int32 tensor containing indices along axis 0 of `tensor`.
 254       Invalid indices must be marked with -1.
 255     padding_value: Value to return for invalid indices.
 256 
 257   Returns:
 258     A tensor sliced based on indices. For indices that are equal to -1, returns
 259     rows of padding value.
 260   """
 261   padded_tensor = tf.concat(
 262       [
 263           padding_value *
 264           tf.ones([1, tf.shape(tensor)[-1]], dtype=tensor.dtype), tensor
 265       ],
 266       axis=0,
 267   )
 268   # tf.concat gradient op uses tf.where(condition) (which is not
 269   # supported on TPU) when the inputs to it are tf.IndexedSlices instead of
 270   # tf.Tensor. Since gradient op for tf.gather returns tf.IndexedSlices,
 271   # we add a dummy op inbetween tf.concat and tf.gather to ensure tf.concat
 272   # gradient function gets tf.Tensor inputs and not tf.IndexedSlices.
 273   padded_tensor *= 1.0
 274   return tf.gather(padded_tensor, indices + 1)

 277 def multilevel_roi_align(features, boxes, box_levels, output_size,
 278                          num_samples_per_cell_y=1, num_samples_per_cell_x=1,
 279                          align_corners=False, extrapolation_value=0.0,
 280                          scope=None):
 281   """Applies RoI Align op and returns feature for boxes.
 282 
 283   Given multiple features maps indexed by different levels, and a set of boxes
 284   where each box is mapped to a certain level, this function selectively crops
 285   and resizes boxes from the corresponding feature maps.
 286 
 287   We follow the RoI Align technique in https://arxiv.org/pdf/1703.06870.pdf
 288   figure 3. Specifically, each box is subdivided uniformly into a grid
 289   consisting of output_size[0] x output_size[1] rectangular cells. Within each
 290   cell we select `num_points` points uniformly and compute feature values using
 291   bilinear interpolation. Finally, we average pool the interpolated values in
 292   each cell to obtain a [output_size[0], output_size[1], channels] feature.
 293 
 294   If `align_corners` is true, sampling points are uniformly spread such that
 295   corner points exactly overlap corners of the boxes.
 296 
 297   In this function we also follow the convention of treating feature pixels as
 298   point objects with no spatial extent.
 299 
 300   Args:
 301     features: A list of 4D float tensors of shape [batch_size, max_height,
 302       max_width, channels] containing features. Note that each feature map must
 303       have the same number of channels.
 304     boxes: A 3D float tensor of shape [batch_size, num_boxes, 4] containing
 305       boxes of the form [ymin, xmin, ymax, xmax] in normalized coordinates.
 306     box_levels: A 3D int32 tensor of shape [batch_size, num_boxes]
 307       representing the feature level index for each box.
 308     output_size: An list of two integers [size_y, size_x] indicating the output
 309       feature size for each box.
 310     num_samples_per_cell_y: Number of grid points to sample along y axis in each
 311       cell.
 312     num_samples_per_cell_x: Number of grid points to sample along x axis in each
 313       cell.
 314     align_corners: Whether to align the corner grid points exactly with box
 315       corners.
 316     extrapolation_value: a float value to use for extrapolation.
 317     scope: Scope name to use for this op.
 318 
 319   Returns:
 320     A 5D float tensor of shape [batch_size, num_boxes, output_size[0],
 321     output_size[1], channels] representing the cropped features.
 322   """
 323   with tf.name_scope(scope, 'MultiLevelRoIAlign'):
 324     features, true_feature_shapes = pad_to_max_size(features)
 325     batch_size = tf.shape(features)[0]
 326     num_levels = features.get_shape().as_list()[1]
 327     max_feature_height = tf.shape(features)[2]
 328     max_feature_width = tf.shape(features)[3]
 329     num_filters = features.get_shape().as_list()[4]
 330     num_boxes = tf.shape(boxes)[1]
 331 
 332     # Convert boxes to absolute co-ordinates.
 333     true_feature_shapes = tf.cast(true_feature_shapes, dtype=boxes.dtype)
 334     true_feature_shapes = tf.gather(true_feature_shapes, box_levels)
 335     boxes *= tf.concat([true_feature_shapes - 1] * 2, axis=-1)
 336 
 337     size_y = output_size[0] * num_samples_per_cell_y
 338     size_x = output_size[1] * num_samples_per_cell_x
 339     box_grid_y, box_grid_x = box_grid_coordinate_vectors(
 340         boxes, size_y=size_y, size_x=size_x, align_corners=align_corners)
 341     (feature_grid_y0, feature_grid_x0, feature_grid_y1,
 342      feature_grid_x1) = feature_grid_coordinate_vectors(box_grid_y, box_grid_x)
 343     feature_grid_y = tf.reshape(
 344         tf.stack([feature_grid_y0, feature_grid_y1], axis=3),
 345         [batch_size, num_boxes, -1])
 346     feature_grid_x = tf.reshape(
 347         tf.stack([feature_grid_x0, feature_grid_x1], axis=3),
 348         [batch_size, num_boxes, -1])
 349     feature_coordinates = ravel_indices(feature_grid_y, feature_grid_x,
 350                                         num_levels, max_feature_height,
 351                                         max_feature_width, box_levels)
 352     valid_indices = _valid_indicator(feature_grid_y, feature_grid_x,
 353                                      true_feature_shapes)
 354     feature_coordinates = tf.where(valid_indices, feature_coordinates,
 355                                    -1 * tf.ones_like(feature_coordinates))
 356     flattened_features = tf.reshape(features, [-1, num_filters])
 357     flattened_feature_values = _gather_valid_indices(flattened_features,
 358                                                      feature_coordinates,
 359                                                      extrapolation_value)
 360     features_per_box = tf.reshape(
 361         flattened_feature_values,
 362         [batch_size, num_boxes, size_y * 2, size_x * 2, num_filters])
 363 
 364     # Cast tensors into dtype of features.
 365     box_grid_y = tf.cast(box_grid_y, dtype=features_per_box.dtype)
 366     box_grid_x = tf.cast(box_grid_x, dtype=features_per_box.dtype)
 367     feature_grid_y0 = tf.cast(feature_grid_y0, dtype=features_per_box.dtype)
 368     feature_grid_x0 = tf.cast(feature_grid_x0, dtype=features_per_box.dtype)
 369 
 370     # RoI Align operation is a bilinear interpolation of four
 371     # neighboring feature points f0, f1, f2, and f3 onto point y, x given by
 372     # f(y, x) = [hy, ly] * [[f00, f01], * [hx, lx]^T
 373     #                       [f10, f11]]
 374     #
 375     # Unrolling the matrix multiplies gives us:
 376     # f(y, x) = (hy * hx) f00 + (hy * lx) f01 + (ly * hx) f10 + (lx * ly) f11
 377     # f(y, x) = w00 * f00 + w01 * f01 + w10 * f10 + w11 * f11
 378     #
 379     # This can be computed by applying pointwise multiplication and sum_pool in
 380     # a 2x2 window.
 381     ly = box_grid_y - feature_grid_y0
 382     lx = box_grid_x - feature_grid_x0
 383     hy = 1.0 - ly
 384     hx = 1.0 - lx
 385 
 386     kernel_y = tf.reshape(
 387         tf.stack([hy, ly], axis=3), [batch_size, num_boxes, size_y * 2, 1])
 388 
 389     kernel_x = tf.reshape(
 390         tf.stack([hx, lx], axis=3), [batch_size, num_boxes, 1, size_x * 2])
 391 
 392     # Multiplier 4 is to make tf.nn.avg_pool behave like sum_pool.
 393     interpolation_kernel = kernel_y * kernel_x * 4
 394 
 395     # Interpolate the gathered features with computed interpolation kernels.
 396     features_per_box *= tf.expand_dims(interpolation_kernel, axis=4),
 397     features_per_box = tf.reshape(
 398         features_per_box,
 399         [batch_size * num_boxes, size_y * 2, size_x * 2, num_filters])
 400 
 401     # This combines the two pooling operations - sum_pool to perform bilinear
 402     # interpolation and avg_pool to pool the values in each bin.
 403     features_per_box = tf.nn.avg_pool(
 404         features_per_box,
 405         [1, num_samples_per_cell_y * 2, num_samples_per_cell_x * 2, 1],
 406         [1, num_samples_per_cell_y * 2, num_samples_per_cell_x * 2, 1], 'VALID')
 407     features_per_box = tf.reshape(
 408         features_per_box,
 409         [batch_size, num_boxes, output_size[0], output_size[1], num_filters])
 410 
 411     return features_per_box

 414 def native_crop_and_resize(image, boxes, crop_size, scope=None):
 415   """Same as `matmul_crop_and_resize` but uses tf.image.crop_and_resize."""
 416   def get_box_inds(proposals):
 417     proposals_shape = proposals.shape.as_list()
 418     if any(dim is None for dim in proposals_shape):
 419       proposals_shape = tf.shape(proposals)
 420     ones_mat = tf.ones(proposals_shape[:2], dtype=tf.int32)
 421     multiplier = tf.expand_dims(
 422         tf.range(start=0, limit=proposals_shape[0]), 1)
 423     return tf.reshape(ones_mat * multiplier, [-1])
 424 
 425   with tf.name_scope(scope, 'CropAndResize'):
 426     cropped_regions = tf.image.crop_and_resize(
 427         image, tf.reshape(boxes, [-1] + boxes.shape.as_list()[2:]),
 428         get_box_inds(boxes), crop_size)
 429     final_shape = tf.concat([tf.shape(boxes)[:2],
 430                              tf.shape(cropped_regions)[1:]], axis=0)
 431     return tf.reshape(cropped_regions, final_shape)

 434 def matmul_crop_and_resize(image, boxes, crop_size, extrapolation_value=0.0,
 435                            scope=None):
 436   """Matrix multiplication based implementation of the crop and resize op.
 437 
 438   Extracts crops from the input image tensor and bilinearly resizes them
 439   (possibly with aspect ratio change) to a common output size specified by
 440   crop_size. This is more general than the crop_to_bounding_box op which
 441   extracts a fixed size slice from the input image and does not allow
 442   resizing or aspect ratio change.
 443 
 444   Returns a tensor with crops from the input image at positions defined at
 445   the bounding box locations in boxes. The cropped boxes are all resized
 446   (with bilinear interpolation) to a fixed size = `[crop_height, crop_width]`.
 447   The result is a 5-D tensor `[batch, num_boxes, crop_height, crop_width,
 448   depth]`.
 449 
 450   Note that this operation is meant to replicate the behavior of the standard
 451   tf.image.crop_and_resize operation but there are a few differences.
 452   Specifically:
 453     1) There is no `box_indices` argument --- to run this op on multiple images,
 454       one must currently call this op independently on each image.
 455     2) The `crop_size` parameter is assumed to be statically defined.
 456       Moreover, the number of boxes must be strictly nonzero.
 457 
 458   Args:
 459     image: A `Tensor`. Must be one of the following types: `uint8`, `int8`,
 460       `int16`, `int32`, `int64`, `half`, 'bfloat16', `float32`, `float64`.
 461       A 4-D tensor of shape `[batch, image_height, image_width, depth]`.
 462       Both `image_height` and `image_width` need to be positive.
 463     boxes: A `Tensor` of type `float32` or 'bfloat16'.
 464       A 3-D tensor of shape `[batch, num_boxes, 4]`. The boxes are specified in
 465       normalized coordinates and are of the form `[y1, x1, y2, x2]`. A
 466       normalized coordinate value of `y` is mapped to the image coordinate at
 467       `y * (image_height - 1)`, so as the `[0, 1]` interval of normalized image
 468       height is mapped to `[0, image_height - 1] in image height coordinates.
 469       We do allow y1 > y2, in which case the sampled crop is an up-down flipped
 470       version of the original image. The width dimension is treated similarly.
 471       Normalized coordinates outside the `[0, 1]` range are allowed, in which
 472       case we use `extrapolation_value` to extrapolate the input image values.
 473     crop_size: A list of two integers `[crop_height, crop_width]`. All
 474       cropped image patches are resized to this size. The aspect ratio of the
 475       image content is not preserved. Both `crop_height` and `crop_width` need
 476       to be positive.
 477     extrapolation_value: a float value to use for extrapolation.
 478     scope: A name for the operation (optional).
 479 
 480   Returns:
 481     A 5-D tensor of shape `[batch, num_boxes, crop_height, crop_width, depth]`
 482   """
 483   with tf.name_scope(scope, 'MatMulCropAndResize'):
 484     box_levels = tf.zeros(tf.shape(boxes)[:2], dtype=tf.int32)
 485     return multilevel_roi_align([image],
 486                                 boxes,
 487                                 box_levels,
 488                                 crop_size,
 489                                 align_corners=True,
 490                                 extrapolation_value=extrapolation_value)
