  74 def _get_multiplier_for_color_randomness():
  75   """Returns a multiplier to get semi-random colors from successive indices.
  76 
  77   This function computes a prime number, p, in the range [2, 17] that:
  78   - is closest to len(STANDARD_COLORS) / 10
  79   - does not divide len(STANDARD_COLORS)
  80 
  81   If no prime numbers in that range satisfy the constraints, p is returned as 1.
  82 
  83   Once p is established, it can be used as a multiplier to select
  84   non-consecutive colors from STANDARD_COLORS:
  85   colors = [(p * i) % len(STANDARD_COLORS) for i in range(20)]
  86   """
  87   num_colors = len(STANDARD_COLORS)
  88   prime_candidates = [5, 7, 11, 13, 17]
  89 
  90   # Remove all prime candidates that divide the number of colors.
  91   prime_candidates = [p for p in prime_candidates if num_colors % p]
  92   if not prime_candidates:
  93     return 1
  94 
  95   # Return the closest prime number to num_colors / 10.
  96   abs_distance = [np.abs(num_colors / 10. - p) for p in prime_candidates]
  97   num_candidates = len(abs_distance)
  98   inds = [i for _, i in sorted(zip(abs_distance, range(num_candidates)))]
  99   return prime_candidates[inds[0]]

 166 def draw_bounding_box_on_image(image,
 167                                ymin,
 168                                xmin,
 169                                ymax,
 170                                xmax,
 171                                color='red',
 172                                thickness=4,
 173                                display_str_list=(),
 174                                use_normalized_coordinates=True):
 175   """Adds a bounding box to an image.
 176 
 177   Bounding box coordinates can be specified in either absolute (pixel) or
 178   normalized coordinates by setting the use_normalized_coordinates argument.
 179 
 180   Each string in display_str_list is displayed on a separate line above the
 181   bounding box in black text on a rectangle filled with the input 'color'.
 182   If the top of the bounding box extends to the edge of the image, the strings
 183   are displayed below the bounding box.
 184 
 185   Args:
 186     image: a PIL.Image object.
 187     ymin: ymin of bounding box.
 188     xmin: xmin of bounding box.
 189     ymax: ymax of bounding box.
 190     xmax: xmax of bounding box.
 191     color: color to draw bounding box. Default is red.
 192     thickness: line thickness. Default value is 4.
 193     display_str_list: list of strings to display in box
 194                       (each to be shown on its own line).
 195     use_normalized_coordinates: If True (default), treat coordinates
 196       ymin, xmin, ymax, xmax as relative to the image.  Otherwise treat
 197       coordinates as absolute.
 198   """
 199   draw = ImageDraw.Draw(image)
 200   im_width, im_height = image.size
 201   if use_normalized_coordinates:
 202     (left, right, top, bottom) = (xmin * im_width, xmax * im_width,
 203                                   ymin * im_height, ymax * im_height)
 204   else:
 205     (left, right, top, bottom) = (xmin, xmax, ymin, ymax)
 206   if thickness > 0:
 207     draw.line([(left, top), (left, bottom), (right, bottom), (right, top),
 208                (left, top)],
 209               width=thickness,
 210               fill=color)
 211   try:
 212     font = ImageFont.truetype('arial.ttf', 24)
 213   except IOError:
 214     font = ImageFont.load_default()
 215 
 216   # If the total height of the display strings added to the top of the bounding
 217   # box exceeds the top of the image, stack the strings below the bounding box
 218   # instead of above.
 219   display_str_heights = [font.getsize(ds)[1] for ds in display_str_list]
 220   # Each display_str has a top and bottom margin of 0.05x.
 221   total_display_str_height = (1 + 2 * 0.05) * sum(display_str_heights)
 222 
 223   if top > total_display_str_height:
 224     text_bottom = top
 225   else:
 226     text_bottom = bottom + total_display_str_height
 227   # Reverse list and print from bottom to top.
 228   for display_str in display_str_list[::-1]:
 229     text_width, text_height = font.getsize(display_str)
 230     margin = np.ceil(0.05 * text_height)
 231     draw.rectangle(
 232         [(left, text_bottom - text_height - 2 * margin), (left + text_width,
 233                                                           text_bottom)],
 234         fill=color)
 235     draw.text(
 236         (left + margin, text_bottom - text_height - margin),
 237         display_str,
 238         fill='black',
 239         font=font)
 240     text_bottom -= text_height - 2 * margin

 306 def create_visualization_fn(category_index,
 307                             include_masks=False,
 308                             include_keypoints=False,
 309                             include_keypoint_scores=False,
 310                             include_track_ids=False,
 311                             **kwargs):
 312   """Constructs a visualization function that can be wrapped in a py_func.
 313 
 314   py_funcs only accept positional arguments. This function returns a suitable
 315   function with the correct positional argument mapping. The positional
 316   arguments in order are:
 317   0: image
 318   1: boxes
 319   2: classes
 320   3: scores
 321   [4]: masks (optional)
 322   [4-5]: keypoints (optional)
 323   [4-6]: keypoint_scores (optional)
 324   [4-7]: track_ids (optional)
 325 
 326   -- Example 1 --
 327   vis_only_masks_fn = create_visualization_fn(category_index,
 328     include_masks=True, include_keypoints=False, include_track_ids=False,
 329     **kwargs)
 330   image = tf.py_func(vis_only_masks_fn,
 331                      inp=[image, boxes, classes, scores, masks],
 332                      Tout=tf.uint8)
 333 
 334   -- Example 2 --
 335   vis_masks_and_track_ids_fn = create_visualization_fn(category_index,
 336     include_masks=True, include_keypoints=False, include_track_ids=True,
 337     **kwargs)
 338   image = tf.py_func(vis_masks_and_track_ids_fn,
 339                      inp=[image, boxes, classes, scores, masks, track_ids],
 340                      Tout=tf.uint8)
 341 
 342   Args:
 343     category_index: a dict that maps integer ids to category dicts. e.g.
 344       {1: {1: 'dog'}, 2: {2: 'cat'}, ...}
 345     include_masks: Whether masks should be expected as a positional argument in
 346       the returned function.
 347     include_keypoints: Whether keypoints should be expected as a positional
 348       argument in the returned function.
 349     include_keypoint_scores: Whether keypoint scores should be expected as a
 350       positional argument in the returned function.
 351     include_track_ids: Whether track ids should be expected as a positional
 352       argument in the returned function.
 353     **kwargs: Additional kwargs that will be passed to
 354       visualize_boxes_and_labels_on_image_array.
 355 
 356   Returns:
 357     Returns a function that only takes tensors as positional arguments.
 358   """
 359 
 360   def visualization_py_func_fn(*args):
 361     """Visualization function that can be wrapped in a tf.py_func.
 362 
 363     Args:
 364       *args: First 4 positional arguments must be:
 365         image - uint8 numpy array with shape (img_height, img_width, 3).
 366         boxes - a numpy array of shape [N, 4].
 367         classes - a numpy array of shape [N].
 368         scores - a numpy array of shape [N] or None.
 369         -- Optional positional arguments --
 370         instance_masks - a numpy array of shape [N, image_height, image_width].
 371         keypoints - a numpy array of shape [N, num_keypoints, 2].
 372         keypoint_scores - a numpy array of shape [N, num_keypoints].
 373         track_ids - a numpy array of shape [N] with unique track ids.
 374 
 375     Returns:
 376       uint8 numpy array with shape (img_height, img_width, 3) with overlaid
 377       boxes.
 378     """
 379     image = args[0]
 380     boxes = args[1]
 381     classes = args[2]
 382     scores = args[3]
 383     masks = keypoints = keypoint_scores = track_ids = None
 384     pos_arg_ptr = 4  # Positional argument for first optional tensor (masks).
 385     if include_masks:
 386       masks = args[pos_arg_ptr]
 387       pos_arg_ptr += 1
 388     if include_keypoints:
 389       keypoints = args[pos_arg_ptr]
 390       pos_arg_ptr += 1
 391     if include_keypoint_scores:
 392       keypoint_scores = args[pos_arg_ptr]
 393       pos_arg_ptr += 1
 394     if include_track_ids:
 395       track_ids = args[pos_arg_ptr]
 396 
 397     return visualize_boxes_and_labels_on_image_array(
 398         image,
 399         boxes,
 400         classes,
 401         scores,
 402         category_index=category_index,
 403         instance_masks=masks,
 404         keypoints=keypoints,
 405         keypoint_scores=keypoint_scores,
 406         track_ids=track_ids,
 407         **kwargs)
 408   return visualization_py_func_fn

 411 def draw_heatmaps_on_image(image, heatmaps):
 412   """Draws heatmaps on an image.
 413 
 414   The heatmaps are handled channel by channel and different colors are used to
 415   paint different heatmap channels.
 416 
 417   Args:
 418     image: a PIL.Image object.
 419     heatmaps: a numpy array with shape [image_height, image_width, channel].
 420       Note that the image_height and image_width should match the size of input
 421       image.
 422   """
 423   draw = ImageDraw.Draw(image)
 424   channel = heatmaps.shape[2]
 425   for c in range(channel):
 426     heatmap = heatmaps[:, :, c] * 255
 427     heatmap = heatmap.astype('uint8')
 428     bitmap = Image.fromarray(heatmap, 'L')
 429     bitmap.convert('1')
 430     draw.bitmap(
 431         xy=[(0, 0)],
 432         bitmap=bitmap,
 433         fill=STANDARD_COLORS[c])

 460 def draw_heatmaps_on_image_tensors(images,
 461                                    heatmaps,
 462                                    apply_sigmoid=False):
 463   """Draws heatmaps on batch of image tensors.
 464 
 465   Args:
 466     images: A 4D uint8 image tensor of shape [N, H, W, C]. If C > 3, additional
 467       channels will be ignored. If C = 1, then we convert the images to RGB
 468       images.
 469     heatmaps: [N, h, w, channel] float32 tensor of heatmaps. Note that the
 470       heatmaps will be resized to match the input image size before overlaying
 471       the heatmaps with input images. Theoretically the heatmap height width
 472       should have the same aspect ratio as the input image to avoid potential
 473       misalignment introduced by the image resize.
 474     apply_sigmoid: Whether to apply a sigmoid layer on top of the heatmaps. If
 475       the heatmaps come directly from the prediction logits, then we should
 476       apply the sigmoid layer to make sure the values are in between [0.0, 1.0].
 477 
 478   Returns:
 479     4D image tensor of type uint8, with heatmaps overlaid on top.
 480   """
 481   # Additional channels are being ignored.
 482   if images.shape[3] > 3:
 483     images = images[:, :, :, 0:3]
 484   elif images.shape[3] == 1:
 485     images = tf.image.grayscale_to_rgb(images)
 486 
 487   _, height, width, _ = shape_utils.combined_static_and_dynamic_shape(images)
 488   if apply_sigmoid:
 489     heatmaps = tf.math.sigmoid(heatmaps)
 490   resized_heatmaps = tf.image.resize(heatmaps, size=[height, width])
 491 
 492   elems = [images, resized_heatmaps]
 493 
 494   def draw_heatmaps(image_and_heatmaps):
 495     """Draws heatmaps on image."""
 496     image_with_heatmaps = tf.py_function(
 497         draw_heatmaps_on_image_array,
 498         image_and_heatmaps,
 499         tf.uint8)
 500     return image_with_heatmaps
 501   images = tf.map_fn(draw_heatmaps, elems, dtype=tf.uint8, back_prop=False)
 502   return images

 838 def draw_keypoints_on_image(image,
 839                             keypoints,

 841                             min_score_thresh=0.5,
 842                             color='red',
 843                             radius=2,
 844                             use_normalized_coordinates=True,
 845                             keypoint_edges=None,
 846                             keypoint_edge_color='green',
 847                             keypoint_edge_width=2):
 848   """Draws keypoints on an image.
 849 
 850   Args:
 851     image: a PIL.Image object.
 852     keypoints: a numpy array with shape [num_keypoints, 2].
 853     keypoint_scores: a numpy array with shape [num_keypoints].
 854     min_score_thresh: a score threshold for visualizing keypoints. Only used if
 855       keypoint_scores is provided.
 856     color: color to draw the keypoints with. Default is red.
 857     radius: keypoint radius. Default value is 2.
 858     use_normalized_coordinates: if True (default), treat keypoint values as
 859       relative to the image.  Otherwise treat them as absolute.
 860     keypoint_edges: A list of tuples with keypoint indices that specify which
 861       keypoints should be connected by an edge, e.g. [(0, 1), (2, 4)] draws
 862       edges from keypoint 0 to 1 and from keypoint 2 to 4.

 865       value is 2.
 866   """
 867   draw = ImageDraw.Draw(image)
 868   im_width, im_height = image.size
 869   keypoints = np.array(keypoints)
 870   keypoints_x = [k[1] for k in keypoints]
 871   keypoints_y = [k[0] for k in keypoints]
 872   if use_normalized_coordinates:
 873     keypoints_x = tuple([im_width * x for x in keypoints_x])
 874     keypoints_y = tuple([im_height * y for y in keypoints_y])
 875   if keypoint_scores is not None:
 876     keypoint_scores = np.array(keypoint_scores)
 877     valid_kpt = np.greater(keypoint_scores, min_score_thresh)
 878   else:
 879     valid_kpt = np.where(np.any(np.isnan(keypoints), axis=1),
 880                          np.zeros_like(keypoints[:, 0]),
 881                          np.ones_like(keypoints[:, 0]))
 882   valid_kpt = [v for v in valid_kpt]
 883 
 884   for keypoint_x, keypoint_y, valid in zip(keypoints_x, keypoints_y, valid_kpt):
 885     if valid:
 886       draw.ellipse([(keypoint_x - radius, keypoint_y - radius),

 889   if keypoint_edges is not None:
 890     for keypoint_start, keypoint_end in keypoint_edges:
 891       if (keypoint_start < 0 or keypoint_start >= len(keypoints) or
 892           keypoint_end < 0 or keypoint_end >= len(keypoints)):
 893         continue
 894       if not (valid_kpt[keypoint_start] and valid_kpt[keypoint_end]):
 895         continue
 896       edge_coordinates = [
 897           keypoints_x[keypoint_start], keypoints_y[keypoint_start],
 898           keypoints_x[keypoint_end], keypoints_y[keypoint_end]
 899       ]
 900       draw.line(
 901           edge_coordinates, fill=keypoint_edge_color, width=keypoint_edge_width)

 904 def draw_mask_on_image_array(image, mask, color='red', alpha=0.4):
 905   """Draws mask on an image.
 906 
 907   Args:
 908     image: uint8 numpy array with shape (img_height, img_height, 3)
 909     mask: a uint8 numpy array of shape (img_height, img_height) with
 910       values between either 0 or 1.

 912     alpha: transparency value between 0 and 1. (default: 0.4)
 913 
 914   Raises:
 915     ValueError: On incorrect data type for image or masks.
 916   """
 917   if image.dtype != np.uint8:
 918     raise ValueError('`image` not of type np.uint8')
 919   if mask.dtype != np.uint8:
 920     raise ValueError('`mask` not of type np.uint8')
 921   if np.any(np.logical_and(mask != 1, mask != 0)):

 923   if image.shape[:2] != mask.shape:
 924     raise ValueError('The image has spatial dimensions %s but the mask has '
 925                      'dimensions %s' % (image.shape[:2], mask.shape))

 930       np.ones_like(mask), axis=2) * np.reshape(list(rgb), [1, 1, 3])
 931   pil_solid_color = Image.fromarray(np.uint8(solid_color)).convert('RGBA')
 932   pil_mask = Image.fromarray(np.uint8(255.0*alpha*mask)).convert('L')
 933   pil_image = Image.composite(pil_solid_color, pil_image, pil_mask)
 934   np.copyto(image, np.array(pil_image.convert('RGB')))

 937 def visualize_boxes_and_labels_on_image_array(
 938     image,
 939     boxes,
 940     classes,
 941     scores,
 942     category_index,
 943     instance_masks=None,
 944     instance_boundaries=None,
 945     keypoints=None,
 946     keypoint_scores=None,
 947     keypoint_edges=None,
 948     track_ids=None,
 949     use_normalized_coordinates=False,
 950     max_boxes_to_draw=20,
 951     min_score_thresh=.5,
 952     agnostic_mode=False,
 953     line_thickness=4,
 954     groundtruth_box_visualization_color='black',
 955     skip_boxes=False,
 956     skip_scores=False,
 957     skip_labels=False,
 958     skip_track_ids=False):
 959   """Overlay labeled boxes on an image with formatted scores and label names.
 960 
 961   This function groups boxes that correspond to the same location
 962   and creates a display string for each detection and overlays these
 963   on the image. Note that this function modifies the image in place, and returns
 964   that same image.
 965 
 966   Args:
 967     image: uint8 numpy array with shape (img_height, img_width, 3)
 968     boxes: a numpy array of shape [N, 4]
 969     classes: a numpy array of shape [N]. Note that class indices are 1-based,
 970       and match the keys in the label map.
 971     scores: a numpy array of shape [N] or None.  If scores=None, then
 972       this function assumes that the boxes to be plotted are groundtruth
 973       boxes and plot all boxes as black with no classes or scores.
 974     category_index: a dict containing category dictionaries (each holding
 975       category index `id` and category name `name`) keyed by category indices.
 976     instance_masks: a numpy array of shape [N, image_height, image_width] with
 977       values ranging between 0 and 1, can be None.
 978     instance_boundaries: a numpy array of shape [N, image_height, image_width]
 979       with values ranging between 0 and 1, can be None.
 980     keypoints: a numpy array of shape [N, num_keypoints, 2], can
 981       be None.
 982     keypoint_scores: a numpy array of shape [N, num_keypoints], can be None.
 983     keypoint_edges: A list of tuples with keypoint indices that specify which
 984       keypoints should be connected by an edge, e.g. [(0, 1), (2, 4)] draws
 985       edges from keypoint 0 to 1 and from keypoint 2 to 4.
 986     track_ids: a numpy array of shape [N] with unique track ids. If provided,
 987       color-coding of boxes will be determined by these ids, and not the class
 988       indices.
 989     use_normalized_coordinates: whether boxes is to be interpreted as
 990       normalized coordinates or not.
 991     max_boxes_to_draw: maximum number of boxes to visualize.  If None, draw
 992       all boxes.
 993     min_score_thresh: minimum score threshold for a box or keypoint to be
 994       visualized.
 995     agnostic_mode: boolean (default: False) controlling whether to evaluate in
 996       class-agnostic mode or not.  This mode will display scores but ignore
 997       classes.
 998     line_thickness: integer (default: 4) controlling line width of the boxes.
 999     groundtruth_box_visualization_color: box color for visualizing groundtruth
1000       boxes
1001     skip_boxes: whether to skip the drawing of bounding boxes.
1002     skip_scores: whether to skip score when drawing a single detection
1003     skip_labels: whether to skip label when drawing a single detection
1004     skip_track_ids: whether to skip track id when drawing a single detection
1005 
1006   Returns:
1007     uint8 numpy array with shape (img_height, img_width, 3) with overlaid boxes.
1008   """
1009   # Create a display string (and color) for every box location, group any boxes
1010   # that correspond to the same location.
1011   box_to_display_str_map = collections.defaultdict(list)
1012   box_to_color_map = collections.defaultdict(str)
1013   box_to_instance_masks_map = {}
1014   box_to_instance_boundaries_map = {}
1015   box_to_keypoints_map = collections.defaultdict(list)
1016   box_to_keypoint_scores_map = collections.defaultdict(list)
1017   box_to_track_ids_map = {}
1018   if not max_boxes_to_draw:
1019     max_boxes_to_draw = boxes.shape[0]
1020   for i in range(boxes.shape[0]):
1021     if max_boxes_to_draw == len(box_to_color_map):
1022       break
1023     if scores is None or scores[i] > min_score_thresh:
1024       box = tuple(boxes[i].tolist())
1025       if instance_masks is not None:
1026         box_to_instance_masks_map[box] = instance_masks[i]
1027       if instance_boundaries is not None:
1028         box_to_instance_boundaries_map[box] = instance_boundaries[i]
1029       if keypoints is not None:
1030         box_to_keypoints_map[box].extend(keypoints[i])
1031       if keypoint_scores is not None:
1032         box_to_keypoint_scores_map[box].extend(keypoint_scores[i])
1033       if track_ids is not None:
1034         box_to_track_ids_map[box] = track_ids[i]
1035       if scores is None:
1036         box_to_color_map[box] = groundtruth_box_visualization_color
1037       else:
1038         display_str = ''
1039         if not skip_labels:
1040           if not agnostic_mode:
1041             if classes[i] in six.viewkeys(category_index):
1042               class_name = category_index[classes[i]]['name']
1043             else:
1044               class_name = 'N/A'
1045             display_str = str(class_name)
1046         if not skip_scores:
1047           if not display_str:
1048             display_str = '{}%'.format(round(100*scores[i]))
1049           else:
1050             display_str = '{}: {}%'.format(display_str, round(100*scores[i]))
1051         if not skip_track_ids and track_ids is not None:
1052           if not display_str:
1053             display_str = 'ID {}'.format(track_ids[i])
1054           else:
1055             display_str = '{}: ID {}'.format(display_str, track_ids[i])
1056         box_to_display_str_map[box].append(display_str)
1057         if agnostic_mode:
1058           box_to_color_map[box] = 'DarkOrange'

1172 class EvalMetricOpsVisualization(six.with_metaclass(abc.ABCMeta, object)):
1173   """Abstract base class responsible for visualizations during evaluation.
1174 
1175   Currently, summary images are not run during evaluation. One way to produce
1176   evaluation images in Tensorboard is to provide tf.summary.image strings as
1177   `value_ops` in tf.estimator.EstimatorSpec's `eval_metric_ops`. This class is
1178   responsible for accruing images (with overlaid detections and groundtruth)
1179   and returning a dictionary that can be passed to `eval_metric_ops`.
1180   """
1181 
1182   def __init__(self,
1183                category_index,
1184                max_examples_to_draw=5,
1185                max_boxes_to_draw=20,
1186                min_score_thresh=0.2,
1187                use_normalized_coordinates=True,
1188                summary_name_prefix='evaluation_image',
1189                keypoint_edges=None):
1190     """Creates an EvalMetricOpsVisualization.
1191 
1192     Args:
1193       category_index: A category index (dictionary) produced from a labelmap.
1194       max_examples_to_draw: The maximum number of example summaries to produce.
1195       max_boxes_to_draw: The maximum number of boxes to draw for detections.
1196       min_score_thresh: The minimum score threshold for showing detections.
1197       use_normalized_coordinates: Whether to assume boxes and keypoints are in
1198         normalized coordinates (as opposed to absolute coordinates).
1199         Default is True.
1200       summary_name_prefix: A string prefix for each image summary.
1201       keypoint_edges: A list of tuples with keypoint indices that specify which
1202         keypoints should be connected by an edge, e.g. [(0, 1), (2, 4)] draws
1203         edges from keypoint 0 to 1 and from keypoint 2 to 4.
1204     """
1205 
1206     self._category_index = category_index
1207     self._max_examples_to_draw = max_examples_to_draw
1208     self._max_boxes_to_draw = max_boxes_to_draw
1209     self._min_score_thresh = min_score_thresh
1210     self._use_normalized_coordinates = use_normalized_coordinates
1211     self._summary_name_prefix = summary_name_prefix
1212     self._keypoint_edges = keypoint_edges
1213     self._images = []
1214 
1215   def clear(self):
1216     self._images = []
1217 
1218   def add_images(self, images):
1219     """Store a list of images, each with shape [1, H, W, C]."""
1220     if len(self._images) >= self._max_examples_to_draw:
1221       return
1222 
1223     # Store images and clip list if necessary.
1224     self._images.extend(images)
1225     if len(self._images) > self._max_examples_to_draw:
1226       self._images[self._max_examples_to_draw:] = []

1228   def get_estimator_eval_metric_ops(self, eval_dict):
1229     """Returns metric ops for use in tf.estimator.EstimatorSpec.
1230 
1231     Args:
1232       eval_dict: A dictionary that holds an image, groundtruth, and detections
1233         for a batched example. Note that, we use only the first example for
1234         visualization. See eval_util.result_dict_for_batched_example() for a
1235         convenient method for constructing such a dictionary. The dictionary
1236         contains
1237         fields.InputDataFields.original_image: [batch_size, H, W, 3] image.
1238         fields.InputDataFields.original_image_spatial_shape: [batch_size, 2]
1239           tensor containing the size of the original image.
1240         fields.InputDataFields.true_image_shape: [batch_size, 3]
1241           tensor containing the spatial size of the upadded original image.
1242         fields.InputDataFields.groundtruth_boxes - [batch_size, num_boxes, 4]
1243           float32 tensor with groundtruth boxes in range [0.0, 1.0].
1244         fields.InputDataFields.groundtruth_classes - [batch_size, num_boxes]
1245           int64 tensor with 1-indexed groundtruth classes.
1246         fields.InputDataFields.groundtruth_instance_masks - (optional)
1247           [batch_size, num_boxes, H, W] int64 tensor with instance masks.
1248         fields.InputDataFields.groundtruth_keypoints - (optional)
1249           [batch_size, num_boxes, num_keypoints, 2] float32 tensor with
1250           keypoint coordinates in format [y, x].
1251         fields.InputDataFields.groundtruth_keypoint_visibilities - (optional)
1252           [batch_size, num_boxes, num_keypoints] bool tensor with
1253           keypoint visibilities.
1254         fields.DetectionResultFields.detection_boxes - [batch_size,
1255           max_num_boxes, 4] float32 tensor with detection boxes in range [0.0,
1256           1.0].
1257         fields.DetectionResultFields.detection_classes - [batch_size,
1258           max_num_boxes] int64 tensor with 1-indexed detection classes.
1259         fields.DetectionResultFields.detection_scores - [batch_size,
1260           max_num_boxes] float32 tensor with detection scores.
1261         fields.DetectionResultFields.detection_masks - (optional) [batch_size,
1262           max_num_boxes, H, W] float32 tensor of binarized masks.
1263         fields.DetectionResultFields.detection_keypoints - (optional)
1264           [batch_size, max_num_boxes, num_keypoints, 2] float32 tensor with
1265           keypoints.
1266         fields.DetectionResultFields.detection_keypoint_scores - (optional)
1267           [batch_size, max_num_boxes, num_keypoints] float32 tensor with
1268           keypoints scores.
1269 
1270     Returns:
1271       A dictionary of image summary names to tuple of (value_op, update_op). The
1272       `update_op` is the same for all items in the dictionary, and is
1273       responsible for saving a single side-by-side image with detections and
1274       groundtruth. Each `value_op` holds the tf.summary.image string for a given
1275       image.
1276     """
1277     if self._max_examples_to_draw == 0:
1278       return {}
1279     images = self.images_from_evaluation_dict(eval_dict)
1280 
1281     def get_images():
1282       """Returns a list of images, padded to self._max_images_to_draw."""
1283       images = self._images
1284       while len(images) < self._max_examples_to_draw:
1285         images.append(np.array(0, dtype=np.uint8))
1286       self.clear()
1287       return images
1288 
1289     def image_summary_or_default_string(summary_name, image):
1290       """Returns image summaries for non-padded elements."""
1291       return tf.cond(
1292           tf.equal(tf.size(tf.shape(image)), 4),
1293           lambda: tf.summary.image(summary_name, image),
1294           lambda: tf.constant(''))
1295 
1296     if tf.executing_eagerly():
1297       update_op = self.add_images([[images[0]]])
1298       image_tensors = get_images()
1299     else:
1300       update_op = tf.py_func(self.add_images, [[images[0]]], [])
1301       image_tensors = tf.py_func(
1302           get_images, [], [tf.uint8] * self._max_examples_to_draw)
1303     eval_metric_ops = {}
1304     for i, image in enumerate(image_tensors):
1305       summary_name = self._summary_name_prefix + '/' + str(i)
1306       value_op = image_summary_or_default_string(summary_name, image)
1307       eval_metric_ops[summary_name] = (value_op, update_op)
1308     return eval_metric_ops
