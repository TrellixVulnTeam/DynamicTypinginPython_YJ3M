  31   def testExponentialDecayWithBurnin(self):
  32     def graph_fn(global_step):
  33       learning_rate_base = 1.0
  34       learning_rate_decay_steps = 3
  35       learning_rate_decay_factor = .1
  36       burnin_learning_rate = .5
  37       burnin_steps = 2
  38       min_learning_rate = .05
  39       learning_rate = learning_schedules.exponential_decay_with_burnin(
  40           global_step, learning_rate_base, learning_rate_decay_steps,
  41           learning_rate_decay_factor, burnin_learning_rate, burnin_steps,
  42           min_learning_rate)
  43       assert learning_rate.op.name.endswith('learning_rate')
  44       return (learning_rate,)
  45 
  46     output_rates = [
  47         self.execute(graph_fn, [np.array(i).astype(np.int64)]) for i in range(9)
  48     ]
  49 
  50     exp_rates = [.5, .5, 1, 1, 1, .1, .1, .1, .05]
  51     self.assertAllClose(output_rates, exp_rates, rtol=1e-4)
