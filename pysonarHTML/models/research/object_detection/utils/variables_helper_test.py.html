  30 class FilterVariablesTest(test_case.TestCase):
  31 
  32   def _create_variables(self):
  33     return [tf.Variable(1.0, name='FeatureExtractor/InceptionV3/weights'),
  34             tf.Variable(1.0, name='FeatureExtractor/InceptionV3/biases'),
  35             tf.Variable(1.0, name='StackProposalGenerator/weights'),
  36             tf.Variable(1.0, name='StackProposalGenerator/biases')]
  37 
  38   def test_return_all_variables_when_empty_regex(self):
  39     variables = self._create_variables()
  40     out_variables = variables_helper.filter_variables(variables, [''])
  41     self.assertCountEqual(out_variables, variables)
  42 
  43   def test_return_variables_which_do_not_match_single_regex(self):
  44     variables = self._create_variables()
  45     out_variables = variables_helper.filter_variables(variables,
  46                                                       ['FeatureExtractor/.*'])
  47     self.assertCountEqual(out_variables, variables[2:])
  48 
  49   def test_return_variables_which_do_not_match_any_regex_in_list(self):
  50     variables = self._create_variables()
  51     out_variables = variables_helper.filter_variables(variables, [
  52         'FeatureExtractor.*biases', 'StackProposalGenerator.*biases'
  53     ])
  54     self.assertCountEqual(out_variables, [variables[0], variables[2]])
  55 
  56   def test_return_variables_matching_empty_regex_list(self):
  57     variables = self._create_variables()
  58     out_variables = variables_helper.filter_variables(
  59         variables, [''], invert=True)
  60     self.assertCountEqual(out_variables, [])
  61 
  62   def test_return_variables_matching_some_regex_in_list(self):
  63     variables = self._create_variables()
  64     out_variables = variables_helper.filter_variables(
  65         variables,
  66         ['FeatureExtractor.*biases', 'StackProposalGenerator.*biases'],
  67         invert=True)
  68     self.assertCountEqual(out_variables, [variables[1], variables[3]])

  71 class MultiplyGradientsMatchingRegexTest(tf.test.TestCase):
  72 
  73   def _create_grads_and_vars(self):
  74     return [(tf.constant(1.0),
  75              tf.Variable(1.0, name='FeatureExtractor/InceptionV3/weights')),
  76             (tf.constant(2.0),
  77              tf.Variable(2.0, name='FeatureExtractor/InceptionV3/biases')),
  78             (tf.constant(3.0),
  79              tf.Variable(3.0, name='StackProposalGenerator/weights')),
  80             (tf.constant(4.0),
  81              tf.Variable(4.0, name='StackProposalGenerator/biases'))]
  82 
  83   def test_multiply_all_feature_extractor_variables(self):
  84     grads_and_vars = self._create_grads_and_vars()
  85     regex_list = ['FeatureExtractor/.*']
  86     multiplier = 0.0
  87     grads_and_vars = variables_helper.multiply_gradients_matching_regex(
  88         grads_and_vars, regex_list, multiplier)
  89     exp_output = [(0.0, 1.0), (0.0, 2.0), (3.0, 3.0), (4.0, 4.0)]
  90     init_op = tf.global_variables_initializer()
  91     with self.test_session() as sess:
  92       sess.run(init_op)
  93       output = sess.run(grads_and_vars)
  94       self.assertCountEqual(output, exp_output)
  95 
  96   def test_multiply_all_bias_variables(self):
  97     grads_and_vars = self._create_grads_and_vars()
  98     regex_list = ['.*/biases']
  99     multiplier = 0.0
 100     grads_and_vars = variables_helper.multiply_gradients_matching_regex(
 101         grads_and_vars, regex_list, multiplier)
 102     exp_output = [(1.0, 1.0), (0.0, 2.0), (3.0, 3.0), (0.0, 4.0)]
 103     init_op = tf.global_variables_initializer()
 104     with self.test_session() as sess:
 105       sess.run(init_op)
 106       output = sess.run(grads_and_vars)
 107       self.assertCountEqual(output, exp_output)

 135 class GetVariablesAvailableInCheckpointTest(test_case.TestCase):
 136 
 137   def test_return_all_variables_from_checkpoint(self):
 138     with tf.Graph().as_default():
 139       variables = [
 140           tf.Variable(1.0, name='weights'),
 141           tf.Variable(1.0, name='biases')
 142       ]
 143       checkpoint_path = os.path.join(self.get_temp_dir(), 'model.ckpt')
 144       init_op = tf.global_variables_initializer()
 145       saver = tf.train.Saver(variables)
 146       with self.test_session() as sess:
 147         sess.run(init_op)
 148         saver.save(sess, checkpoint_path)
 149       out_variables = variables_helper.get_variables_available_in_checkpoint(
 150           variables, checkpoint_path)
 151     self.assertCountEqual(out_variables, variables)
 152 
 153   def test_return_all_variables_from_checkpoint_with_partition(self):
 154     with tf.Graph().as_default():
 155       partitioner = tf.fixed_size_partitioner(2)
 156       variables = [
 157           tf.get_variable(
 158               name='weights', shape=(2, 2), partitioner=partitioner),
 159           tf.Variable([1.0, 2.0], name='biases')
 160       ]
 161       checkpoint_path = os.path.join(self.get_temp_dir(), 'model.ckpt')
 162       init_op = tf.global_variables_initializer()
 163       saver = tf.train.Saver(variables)
 164       with self.test_session() as sess:
 165         sess.run(init_op)
 166         saver.save(sess, checkpoint_path)
 167       out_variables = variables_helper.get_variables_available_in_checkpoint(
 168           variables, checkpoint_path)
 169     self.assertCountEqual(out_variables, variables)
 170 
 171   def test_return_variables_available_in_checkpoint(self):
 172     checkpoint_path = os.path.join(self.get_temp_dir(), 'model.ckpt')
 173     with tf.Graph().as_default():
 174       weight_variable = tf.Variable(1.0, name='weights')
 175       global_step = tf.train.get_or_create_global_step()
 176       graph1_variables = [
 177           weight_variable,
 178           global_step
 179       ]
 180       init_op = tf.global_variables_initializer()
 181       saver = tf.train.Saver(graph1_variables)
 182       with self.test_session() as sess:
 183         sess.run(init_op)
 184         saver.save(sess, checkpoint_path)
 185 
 186     with tf.Graph().as_default():
 187       graph2_variables = graph1_variables + [tf.Variable(1.0, name='biases')]
 188       out_variables = variables_helper.get_variables_available_in_checkpoint(
 189           graph2_variables, checkpoint_path, include_global_step=False)
 190     self.assertCountEqual(out_variables, [weight_variable])
 191 
 192   def test_return_variables_available_an_checkpoint_with_dict_inputs(self):
 193     checkpoint_path = os.path.join(self.get_temp_dir(), 'model.ckpt')
 194     with tf.Graph().as_default():
 195       graph1_variables = [
 196           tf.Variable(1.0, name='ckpt_weights'),
 197       ]
 198       init_op = tf.global_variables_initializer()
 199       saver = tf.train.Saver(graph1_variables)
 200       with self.test_session() as sess:
 201         sess.run(init_op)
 202         saver.save(sess, checkpoint_path)
 203 
 204     with tf.Graph().as_default():
 205       graph2_variables_dict = {
 206           'ckpt_weights': tf.Variable(1.0, name='weights'),
 207           'ckpt_biases': tf.Variable(1.0, name='biases')
 208       }
 209       out_variables = variables_helper.get_variables_available_in_checkpoint(
 210           graph2_variables_dict, checkpoint_path)
 211 
 212     self.assertIsInstance(out_variables, dict)
 213     self.assertCountEqual(list(out_variables.keys()), ['ckpt_weights'])
 214     self.assertEqual(out_variables['ckpt_weights'].op.name, 'weights')
 215 
 216   def test_return_variables_with_correct_sizes(self):
 217     checkpoint_path = os.path.join(self.get_temp_dir(), 'model.ckpt')
 218     with tf.Graph().as_default():
 219       bias_variable = tf.Variable(3.0, name='biases')
 220       global_step = tf.train.get_or_create_global_step()
 221       graph1_variables = [
 222           tf.Variable([[1.0, 2.0], [3.0, 4.0]], name='weights'),
 223           bias_variable,
 224           global_step
 225       ]
 226       init_op = tf.global_variables_initializer()
 227       saver = tf.train.Saver(graph1_variables)
 228       with self.test_session() as sess:
 229         sess.run(init_op)
 230         saver.save(sess, checkpoint_path)
 231 
 232     with tf.Graph().as_default():
 233       graph2_variables = [
 234           tf.Variable([1.0, 2.0], name='weights'),  # New variable shape.
 235           bias_variable,
 236           global_step
 237       ]
 238 
 239     out_variables = variables_helper.get_variables_available_in_checkpoint(
 240         graph2_variables, checkpoint_path, include_global_step=True)
 241     self.assertCountEqual(out_variables, [bias_variable, global_step])
