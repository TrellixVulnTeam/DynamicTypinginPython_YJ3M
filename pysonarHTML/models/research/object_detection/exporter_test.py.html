  68     with tf.control_dependencies(list(prediction_dict.values())):
  69       postprocessed_tensors = {
  70           'detection_boxes': tf.constant([[[0.0, 0.0, 0.5, 0.5],
  71                                            [0.5, 0.5, 0.8, 0.8]],
  72                                           [[0.5, 0.5, 1.0, 1.0],
  73                                            [0.0, 0.0, 0.0, 0.0]]], tf.float32),
  74           'detection_scores': tf.constant([[0.7, 0.6],
  75                                            [0.9, 0.0]], tf.float32),
  76           'detection_multiclass_scores': tf.constant([[[0.3, 0.7], [0.4, 0.6]],
  77                                                       [[0.1, 0.9], [0.0, 0.0]]],
  78                                                      tf.float32),
  79           'detection_classes': tf.constant([[0, 1],
  80                                             [1, 0]], tf.float32),
  81           'num_detections': tf.constant([2, 1], tf.float32),
  82           'raw_detection_boxes': tf.constant([[[0.0, 0.0, 0.5, 0.5],
  83                                                [0.5, 0.5, 0.8, 0.8]],
  84                                               [[0.5, 0.5, 1.0, 1.0],
  85                                                [0.0, 0.5, 0.0, 0.5]]],
  86                                              tf.float32),
  87           'raw_detection_scores': tf.constant([[0.7, 0.6],
  88                                                [0.9, 0.5]], tf.float32),
  89       }
  90       if self._add_detection_keypoints:
  91         postprocessed_tensors['detection_keypoints'] = tf.constant(
  92             np.arange(48).reshape([2, 2, 6, 2]), tf.float32)

  97         # let fake detection features have shape [4, 4, 10]
  98         postprocessed_tensors['detection_features'] = tf.constant(
  99             np.ones((2, 2, 4, 4, 10)), tf.float32)

 101     return postprocessed_tensors

 144   def _load_inference_graph(self, inference_graph_path, is_binary=True):
 145     od_graph = tf.Graph()
 146     with od_graph.as_default():
 147       od_graph_def = tf.GraphDef()
 148       with tf.gfile.GFile(inference_graph_path, mode='rb') as fid:
 149         if is_binary:
 150           od_graph_def.ParseFromString(fid.read())
 151         else:
 152           text_format.Parse(fid.read(), od_graph_def)
 153         tf.import_graph_def(od_graph_def, name='')
 154     return od_graph
