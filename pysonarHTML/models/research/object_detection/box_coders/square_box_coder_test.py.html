  16 """Tests for object_detection.box_coder.square_box_coder."""
  17 import numpy as np
  18 import tensorflow.compat.v1 as tf
  19 
  20 from object_detection.box_coders import square_box_coder
  21 from object_detection.core import box_list
  22 from object_detection.utils import test_case
  23 
  24 
  25 class SquareBoxCoderTest(test_case.TestCase):
  26 
  27   def test_correct_relative_codes_with_default_scale(self):
  28     boxes = np.array([[10.0, 10.0, 20.0, 15.0], [0.2, 0.1, 0.5, 0.4]],
  29                      np.float32)
  30     anchors = np.array([[15.0, 12.0, 30.0, 18.0], [0.1, 0.0, 0.7, 0.9]],
  31                        np.float32)
  32     expected_rel_codes = [[-0.790569, -0.263523, -0.293893],
  33                           [-0.068041, -0.272166, -0.89588]]
  34     def graph_fn(boxes, anchors):
  35       scale_factors = None
  36       boxes = box_list.BoxList(boxes)
  37       anchors = box_list.BoxList(anchors)
  38       coder = square_box_coder.SquareBoxCoder(scale_factors=scale_factors)
  39       rel_codes = coder.encode(boxes, anchors)
  40       return rel_codes
  41     rel_codes_out = self.execute(graph_fn, [boxes, anchors])
  42     self.assertAllClose(rel_codes_out, expected_rel_codes, rtol=1e-04,
  43                         atol=1e-04)

  78   def test_correct_boxes_with_default_scale(self):
  79     anchors = np.array([[15.0, 12.0, 30.0, 18.0], [0.1, 0.0, 0.7, 0.9]],
  80                        np.float32)
  81     rel_codes = np.array([[-0.5, -0.416666, -0.405465],
  82                           [-0.083333, -0.222222, -0.693147]], np.float32)
  83     expected_boxes = [[14.594306, 7.884875, 20.918861, 14.209432],
  84                       [0.155051, 0.102989, 0.522474, 0.470412]]
  85     def graph_fn(rel_codes, anchors):
  86       scale_factors = None
  87       anchors = box_list.BoxList(anchors)
  88       coder = square_box_coder.SquareBoxCoder(scale_factors=scale_factors)
  89       boxes = coder.decode(rel_codes, anchors).get()
  90       return boxes
  91     boxes_out = self.execute(graph_fn, [rel_codes, anchors])
  92     self.assertAllClose(boxes_out, expected_boxes, rtol=1e-04,
  93                         atol=1e-04)
