  28   def test_get_correct_relative_codes_after_encoding(self):
  29     boxes = np.array([[10., 10., 20., 15.],
  30                       [0.2, 0.1, 0.5, 0.4]], np.float32)
  31     keypoints = np.array([[[15., 12.], [10., 15.]],
  32                           [[0.5, 0.3], [0.2, 0.4]]], np.float32)
  33     num_keypoints = len(keypoints[0])
  34     anchors = np.array([[15., 12., 30., 18.],
  35                         [0.1, 0.0, 0.7, 0.9]], np.float32)
  36     expected_rel_codes = [
  37         [-0.5, -0.416666, -0.405465, -0.182321,
  38          -0.5, -0.5, -0.833333, 0.],
  39         [-0.083333, -0.222222, -0.693147, -1.098612,
  40          0.166667, -0.166667, -0.333333, -0.055556]
  41     ]
  42     def graph_fn(boxes, keypoints, anchors):
  43       boxes = box_list.BoxList(boxes)
  44       boxes.add_field(fields.BoxListFields.keypoints, keypoints)
  45       anchors = box_list.BoxList(anchors)
  46       coder = keypoint_box_coder.KeypointBoxCoder(num_keypoints)
  47       rel_codes = coder.encode(boxes, anchors)
  48       return rel_codes
  49     rel_codes_out = self.execute(graph_fn, [boxes, keypoints, anchors])
  50     self.assertAllClose(rel_codes_out, expected_rel_codes, rtol=1e-04,
  51                         atol=1e-04)

  53   def test_get_correct_relative_codes_after_encoding_with_scaling(self):

  59     anchors = np.array([[15., 12., 30., 18.],
  60                         [0.1, 0.0, 0.7, 0.9]], np.float32)
  61     expected_rel_codes = [
  62         [-1., -1.25, -1.62186, -0.911608,
  63          -1.0, -1.5, -1.666667, 0.],

  80   def test_get_correct_boxes_after_decoding(self):
  81     anchors = np.array([[15., 12., 30., 18.],
  82                         [0.1, 0.0, 0.7, 0.9]], np.float32)
  83     rel_codes = np.array([
  84         [-0.5, -0.416666, -0.405465, -0.182321,
  85          -0.5, -0.5, -0.833333, 0.],
  86         [-0.083333, -0.222222, -0.693147, -1.098612,
  87          0.166667, -0.166667, -0.333333, -0.055556]
  88     ], np.float32)
  89     expected_boxes = [[10., 10., 20., 15.],
  90                       [0.2, 0.1, 0.5, 0.4]]
  91     expected_keypoints = [[[15., 12.], [10., 15.]],
  92                           [[0.5, 0.3], [0.2, 0.4]]]
  93     num_keypoints = len(expected_keypoints[0])
  94     def graph_fn(rel_codes, anchors):
  95       anchors = box_list.BoxList(anchors)
  96       coder = keypoint_box_coder.KeypointBoxCoder(num_keypoints)
  97       boxes = coder.decode(rel_codes, anchors)
  98       return boxes.get(), boxes.get_field(fields.BoxListFields.keypoints)
  99     boxes_out, keypoints_out = self.execute(graph_fn, [rel_codes, anchors])
 100     self.assertAllClose(keypoints_out, expected_keypoints, rtol=1e-04,
 101                         atol=1e-04)
 102     self.assertAllClose(boxes_out, expected_boxes, rtol=1e-04,
 103                         atol=1e-04)

 132   def test_very_small_width_nan_after_encoding(self):
