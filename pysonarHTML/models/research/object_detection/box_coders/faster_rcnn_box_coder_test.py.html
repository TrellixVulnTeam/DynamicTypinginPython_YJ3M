  16 """Tests for object_detection.box_coder.faster_rcnn_box_coder."""
  17 import numpy as np

  27   def test_get_correct_relative_codes_after_encoding(self):
  28     boxes = np.array([[10.0, 10.0, 20.0, 15.0], [0.2, 0.1, 0.5, 0.4]],
  29                      np.float32)
  30     anchors = np.array([[15.0, 12.0, 30.0, 18.0], [0.1, 0.0, 0.7, 0.9]],
  31                        np.float32)
  32     expected_rel_codes = [[-0.5, -0.416666, -0.405465, -0.182321],
  33                           [-0.083333, -0.222222, -0.693147, -1.098612]]
  34     def graph_fn(boxes, anchors):
  35       boxes = box_list.BoxList(boxes)
  36       anchors = box_list.BoxList(anchors)
  37       coder = faster_rcnn_box_coder.FasterRcnnBoxCoder()
  38       rel_codes = coder.encode(boxes, anchors)
  39       return rel_codes
  40     rel_codes_out = self.execute(graph_fn, [boxes, anchors])
  41     self.assertAllClose(rel_codes_out, expected_rel_codes, rtol=1e-04,
  42                         atol=1e-04)

  44   def test_get_correct_relative_codes_after_encoding_with_scaling(self):
  45     boxes = np.array([[10.0, 10.0, 20.0, 15.0], [0.2, 0.1, 0.5, 0.4]],
  46                      np.float32)
  47     anchors = np.array([[15.0, 12.0, 30.0, 18.0], [0.1, 0.0, 0.7, 0.9]],
  48                        np.float32)
  49     expected_rel_codes = [[-1., -1.25, -1.62186, -0.911608],
  50                           [-0.166667, -0.666667, -2.772588, -5.493062]]
  51     def graph_fn(boxes, anchors):
  52       scale_factors = [2, 3, 4, 5]
  53       boxes = box_list.BoxList(boxes)
  54       anchors = box_list.BoxList(anchors)
  55       coder = faster_rcnn_box_coder.FasterRcnnBoxCoder(
  56           scale_factors=scale_factors)
  57       rel_codes = coder.encode(boxes, anchors)
  58       return rel_codes
  59     rel_codes_out = self.execute(graph_fn, [boxes, anchors])
  60     self.assertAllClose(rel_codes_out, expected_rel_codes, rtol=1e-04,
  61                         atol=1e-04)

  63   def test_get_correct_boxes_after_decoding(self):
  64     anchors = np.array([[15.0, 12.0, 30.0, 18.0], [0.1, 0.0, 0.7, 0.9]],
  65                        np.float32)
  66     rel_codes = np.array([[-0.5, -0.416666, -0.405465, -0.182321],
  67                           [-0.083333, -0.222222, -0.693147, -1.098612]],
  68                          np.float32)
  69     expected_boxes = [[10.0, 10.0, 20.0, 15.0], [0.2, 0.1, 0.5, 0.4]]
  70     def graph_fn(rel_codes, anchors):
  71       anchors = box_list.BoxList(anchors)
  72       coder = faster_rcnn_box_coder.FasterRcnnBoxCoder()
  73       boxes = coder.decode(rel_codes, anchors)
  74       return boxes.get()
  75     boxes_out = self.execute(graph_fn, [rel_codes, anchors])
  76     self.assertAllClose(boxes_out, expected_boxes, rtol=1e-04,
  77                         atol=1e-04)
