  16 """Keypoint box coder.
  17 
  18 The keypoint box coder follows the coding schema described below (this is
  19 similar to the FasterRcnnBoxCoder, except that it encodes keypoints in addition
  20 to box coordinates):
  21   ty = (y - ya) / ha
  22   tx = (x - xa) / wa
  23   th = log(h / ha)
  24   tw = log(w / wa)
  25   tky0 = (ky0 - ya) / ha
  26   tkx0 = (kx0 - xa) / wa
  27   tky1 = (ky1 - ya) / ha
  28   tkx1 = (kx1 - xa) / wa
  29   ...
  30   where x, y, w, h denote the box's center coordinates, width and height
  31   respectively. Similarly, xa, ya, wa, ha denote the anchor's center
  32   coordinates, width and height. tx, ty, tw and th denote the anchor-encoded
  33   center, width and height respectively. ky0, kx0, ky1, kx1, ... denote the
  34   keypoints' coordinates, and tky0, tkx0, tky1, tkx1, ... denote the
  35   anchor-encoded keypoint coordinates.
  36 """
  37 
  38 import tensorflow.compat.v1 as tf
  39 
  40 from object_detection.core import box_coder
  41 from object_detection.core import box_list
  42 from object_detection.core import standard_fields as fields
  43 
  44 EPSILON = 1e-8
  45 
  46 
  47 class KeypointBoxCoder(box_coder.BoxCoder):
  48   """Keypoint box coder."""
  49 
  50   def __init__(self, num_keypoints, scale_factors=None):
  51     """Constructor for KeypointBoxCoder.
  52 
  53     Args:
  54       num_keypoints: Number of keypoints to encode/decode.
  55       scale_factors: List of 4 positive scalars to scale ty, tx, th and tw.
  56         In addition to scaling ty and tx, the first 2 scalars are used to scale
  57         the y and x coordinates of the keypoints as well. If set to None, does
  58         not perform scaling.
  59     """
  60     self._num_keypoints = num_keypoints
  61 
  62     if scale_factors:
  63       assert len(scale_factors) == 4
  64       for scalar in scale_factors:
  65         assert scalar > 0
  66     self._scale_factors = scale_factors
  67     self._keypoint_scale_factors = None
  68     if scale_factors is not None:
  69       self._keypoint_scale_factors = tf.expand_dims(
  70           tf.tile([
  71               tf.cast(scale_factors[0], dtype=tf.float32),
  72               tf.cast(scale_factors[1], dtype=tf.float32)
  73           ], [num_keypoints]), 1)
  74 
  75   @property
  76   def code_size(self):
  77     return 4 + self._num_keypoints * 2
  78 
  79   def _encode(self, boxes, anchors):
  80     """Encode a box and keypoint collection with respect to anchor collection.
  81 
  82     Args:
  83       boxes: BoxList holding N boxes and keypoints to be encoded. Boxes are
  84         tensors with the shape [N, 4], and keypoints are tensors with the shape
  85         [N, num_keypoints, 2].
  86       anchors: BoxList of anchors.
  87 
  88     Returns:
  89       a tensor representing N anchor-encoded boxes of the format
  90       [ty, tx, th, tw, tky0, tkx0, tky1, tkx1, ...] where tky0 and tkx0
  91       represent the y and x coordinates of the first keypoint, tky1 and tkx1
  92       represent the y and x coordinates of the second keypoint, and so on.
  93     """
  94     # Convert anchors to the center coordinate representation.
  95     ycenter_a, xcenter_a, ha, wa = anchors.get_center_coordinates_and_sizes()
  96     ycenter, xcenter, h, w = boxes.get_center_coordinates_and_sizes()
  97     keypoints = boxes.get_field(fields.BoxListFields.keypoints)
  98     keypoints = tf.transpose(tf.reshape(keypoints,
  99                                         [-1, self._num_keypoints * 2]))
 100     num_boxes = boxes.num_boxes()
 101 
 102     # Avoid NaN in division and log below.
 103     ha += EPSILON
 104     wa += EPSILON
 105     h += EPSILON
 106     w += EPSILON
 107 
 108     tx = (xcenter - xcenter_a) / wa
 109     ty = (ycenter - ycenter_a) / ha
 110     tw = tf.log(w / wa)
 111     th = tf.log(h / ha)
 112 
 113     tiled_anchor_centers = tf.tile(
 114         tf.stack([ycenter_a, xcenter_a]), [self._num_keypoints, 1])
 115     tiled_anchor_sizes = tf.tile(
 116         tf.stack([ha, wa]), [self._num_keypoints, 1])
 117     tkeypoints = (keypoints - tiled_anchor_centers) / tiled_anchor_sizes
 118 
 119     # Scales location targets as used in paper for joint training.
 120     if self._scale_factors:
 121       ty *= self._scale_factors[0]
 122       tx *= self._scale_factors[1]
 123       th *= self._scale_factors[2]
 124       tw *= self._scale_factors[3]
 125       tkeypoints *= tf.tile(self._keypoint_scale_factors, [1, num_boxes])
 126 
 127     tboxes = tf.stack([ty, tx, th, tw])
 128     return tf.transpose(tf.concat([tboxes, tkeypoints], 0))
 129 
 130   def _decode(self, rel_codes, anchors):
 131     """Decode relative codes to boxes and keypoints.
 132 
 133     Args:
 134       rel_codes: a tensor with shape [N, 4 + 2 * num_keypoints] representing N
 135         anchor-encoded boxes and keypoints
 136       anchors: BoxList of anchors.
 137 
 138     Returns:
 139       boxes: BoxList holding N bounding boxes and keypoints.
 140     """
 141     ycenter_a, xcenter_a, ha, wa = anchors.get_center_coordinates_and_sizes()
 142 
 143     num_codes = tf.shape(rel_codes)[0]
 144     result = tf.unstack(tf.transpose(rel_codes))
 145     ty, tx, th, tw = result[:4]
 146     tkeypoints = result[4:]
 147     if self._scale_factors:
 148       ty /= self._scale_factors[0]
 149       tx /= self._scale_factors[1]
 150       th /= self._scale_factors[2]
 151       tw /= self._scale_factors[3]
 152       tkeypoints /= tf.tile(self._keypoint_scale_factors, [1, num_codes])
 153 
 154     w = tf.exp(tw) * wa
 155     h = tf.exp(th) * ha
 156     ycenter = ty * ha + ycenter_a
 157     xcenter = tx * wa + xcenter_a
 158     ymin = ycenter - h / 2.
 159     xmin = xcenter - w / 2.
 160     ymax = ycenter + h / 2.
 161     xmax = xcenter + w / 2.
 162     decoded_boxes_keypoints = box_list.BoxList(
 163         tf.transpose(tf.stack([ymin, xmin, ymax, xmax])))
 164 
 165     tiled_anchor_centers = tf.tile(
 166         tf.stack([ycenter_a, xcenter_a]), [self._num_keypoints, 1])
 167     tiled_anchor_sizes = tf.tile(
 168         tf.stack([ha, wa]), [self._num_keypoints, 1])
 169     keypoints = tkeypoints * tiled_anchor_sizes + tiled_anchor_centers
 170     keypoints = tf.reshape(tf.transpose(keypoints),
 171                            [-1, self._num_keypoints, 2])
 172     decoded_boxes_keypoints.add_field(fields.BoxListFields.keypoints, keypoints)
 173     return decoded_boxes_keypoints
