  27   def test_subsample_all_examples(self):
  28     if self.has_tpu(): return
  29     numpy_labels = np.random.permutation(300)
  30     indicator = np.array(np.ones(300) == 1, np.bool)
  31     numpy_labels = (numpy_labels - 200) > 0
  32 
  33     labels = np.array(numpy_labels, np.bool)
  34 
  35     def graph_fn(indicator, labels):
  36       sampler = (
  37           balanced_positive_negative_sampler.BalancedPositiveNegativeSampler())
  38       return sampler.subsample(indicator, 64, labels)
  39 
  40     is_sampled = self.execute_cpu(graph_fn, [indicator, labels])
  41     self.assertEqual(sum(is_sampled), 64)
  42     self.assertEqual(sum(np.logical_and(numpy_labels, is_sampled)), 32)
  43     self.assertEqual(sum(np.logical_and(
  44         np.logical_not(numpy_labels), is_sampled)), 32)

  66   def test_subsample_selection(self):
  67     if self.has_tpu(): return
  68     # Test random sampling when only some examples can be sampled:
  69     # 100 samples, 20 positives, 10 positives cannot be sampled.
  70     numpy_labels = np.arange(100)
  71     numpy_indicator = numpy_labels < 90
  72     indicator = np.array(numpy_indicator, np.bool)
  73     numpy_labels = (numpy_labels - 80) >= 0
  74 
  75     labels = np.array(numpy_labels, np.bool)
  76 
  77     def graph_fn(indicator, labels):
  78       sampler = (
  79           balanced_positive_negative_sampler.BalancedPositiveNegativeSampler())
  80       return sampler.subsample(indicator, 64, labels)
  81 
  82     is_sampled = self.execute_cpu(graph_fn, [indicator, labels])
  83     self.assertEqual(sum(is_sampled), 64)

 113   def test_subsample_selection_larger_batch_size(self):
 114     if self.has_tpu(): return
 115     # Test random sampling when total number of examples that can be sampled are
 116     # less than batch size:
 117     # 100 samples, 50 positives, 40 positives cannot be sampled, batch size 64.
 118     # It should still return 64 samples, with 4 of them that couldn't have been
 119     # sampled.
 120     numpy_labels = np.arange(100)
 121     numpy_indicator = numpy_labels < 60
 122     indicator = np.array(numpy_indicator, np.bool)
 123     numpy_labels = (numpy_labels - 50) >= 0
 124 
 125     labels = np.array(numpy_labels, np.bool)
 126 
 127     def graph_fn(indicator, labels):
 128       sampler = (
 129           balanced_positive_negative_sampler.BalancedPositiveNegativeSampler())
 130       return sampler.subsample(indicator, 64, labels)
 131 
 132     is_sampled = self.execute_cpu(graph_fn, [indicator, labels])
 133     self.assertEqual(sum(is_sampled), 60)
 134     self.assertGreaterEqual(sum(np.logical_and(numpy_labels, is_sampled)), 10)
 135     self.assertGreaterEqual(
 136         sum(np.logical_and(np.logical_not(numpy_labels), is_sampled)), 50)
 137     self.assertEqual(sum(np.logical_and(is_sampled, numpy_indicator)), 60)

 166   def test_subsample_selection_no_batch_size(self):
 167     if self.has_tpu(): return
 168     # Test random sampling when only some examples can be sampled:
 169     # 1000 samples, 6 positives (5 can be sampled).
 170     numpy_labels = np.arange(1000)
 171     numpy_indicator = numpy_labels < 999
 172     numpy_labels = (numpy_labels - 994) >= 0
 173 
 174     def graph_fn(indicator, labels):
 175       sampler = (balanced_positive_negative_sampler.
 176                  BalancedPositiveNegativeSampler(0.01))
 177       is_sampled = sampler.subsample(indicator, None, labels)
 178       return is_sampled
 179     is_sampled_out = self.execute_cpu(graph_fn, [numpy_indicator, numpy_labels])
 180     self.assertEqual(sum(is_sampled_out), 500)
 181     self.assertEqual(sum(np.logical_and(numpy_labels, is_sampled_out)), 5)
 182     self.assertEqual(sum(np.logical_and(
 183         np.logical_not(numpy_labels), is_sampled_out)), 495)
 184     self.assertAllEqual(is_sampled_out, np.logical_and(is_sampled_out,
 185                                                        numpy_indicator))
