  16 """Tests for google3.research.vale.object_detection.minibatch_sampler."""
  17 
  18 import numpy as np
  19 import tensorflow.compat.v1 as tf
  20 
  21 from object_detection.core import minibatch_sampler
  22 from object_detection.utils import test_case
  23 
  24 
  25 class MinibatchSamplerTest(test_case.TestCase):
  26 
  27   def test_subsample_indicator_when_more_true_elements_than_num_samples(self):
  28     np_indicator = np.array([True, False, True, False, True, True, False])
  29     def graph_fn(indicator):
  30       samples = minibatch_sampler.MinibatchSampler.subsample_indicator(
  31           indicator, 3)
  32       return samples
  33     samples_out = self.execute(graph_fn, [np_indicator])
  34     self.assertTrue(np.sum(samples_out), 3)
  35     self.assertAllEqual(samples_out,
  36                         np.logical_and(samples_out, np_indicator))
  37 
  38   def test_subsample_indicator_when_less_true_elements_than_num_samples(self):
  39     np_indicator = np.array([True, False, True, False, True, True, False])
  40     def graph_fn(indicator):
  41       samples = minibatch_sampler.MinibatchSampler.subsample_indicator(
  42           indicator, 5)
  43       return samples
  44     samples_out = self.execute(graph_fn, [np_indicator])
  45     self.assertTrue(np.sum(samples_out), 4)
  46     self.assertAllEqual(samples_out,
  47                         np.logical_and(samples_out, np_indicator))
  48 
  49   def test_subsample_indicator_when_num_samples_is_zero(self):
  50     np_indicator = np.array([True, False, True, False, True, True, False])
  51     def graph_fn(indicator):
  52       samples_none = minibatch_sampler.MinibatchSampler.subsample_indicator(
  53           indicator, 0)
  54       return samples_none
  55     samples_out = self.execute(graph_fn, [np_indicator])
  56     self.assertAllEqual(
  57         np.zeros_like(samples_out, dtype=bool),
  58         samples_out)
  59 
  60   def test_subsample_indicator_when_indicator_all_false(self):
  61     indicator_empty = np.zeros([0], dtype=np.bool)
  62     def graph_fn(indicator):
  63       samples_empty = minibatch_sampler.MinibatchSampler.subsample_indicator(
  64           indicator, 4)
  65       return samples_empty
  66     samples_out = self.execute(graph_fn, [indicator_empty])
  67     self.assertEqual(0, samples_out.size)
  68

  70 if __name__ == '__main__':
  71   tf.test.main()
