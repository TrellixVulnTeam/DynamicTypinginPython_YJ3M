  16 """Class to subsample minibatches by balancing positives and negatives.
  17 
  18 Subsamples minibatches based on a pre-specified positive fraction in range
  19 [0,1]. The class presumes there are many more negatives than positive examples:
  20 if the desired batch_size cannot be achieved with the pre-specified positive
  21 fraction, it fills the rest with negative examples. If this is not sufficient
  22 for obtaining the desired batch_size, it returns fewer examples.
  23 
  24 The main function to call is Subsample(self, indicator, labels). For convenience
  25 one can also call SubsampleWeights(self, weights, labels) which is defined in
  26 the minibatch_sampler base class.
  27 
  28 When is_static is True, it implements a method that guarantees static shapes.
  29 It also ensures the length of output of the subsample is always batch_size, even
  30 when number of examples set to True in indicator is less than batch_size.
  31 """
  32 
  33 import tensorflow.compat.v1 as tf
  34 
  35 from object_detection.core import minibatch_sampler
  36 
  37 
  38 class BalancedPositiveNegativeSampler(minibatch_sampler.MinibatchSampler):
  39   """Subsamples minibatches to a desired balance of positives and negatives."""
  40 
  41   def __init__(self, positive_fraction=0.5, is_static=False):
  42     """Constructs a minibatch sampler.
  43 
  44     Args:
  45       positive_fraction: desired fraction of positive examples (scalar in [0,1])
  46         in the batch.
  47       is_static: If True, uses an implementation with static shape guarantees.
  48 
  49     Raises:
  50       ValueError: if positive_fraction < 0, or positive_fraction > 1
  51     """
  52     if positive_fraction < 0 or positive_fraction > 1:
  53       raise ValueError('positive_fraction should be in range [0,1]. '
  54                        'Received: %s.' % positive_fraction)
  55     self._positive_fraction = positive_fraction
  56     self._is_static = is_static
  57 
  58   def _get_num_pos_neg_samples(self, sorted_indices_tensor, sample_size):
  59     """Counts the number of positives and negatives numbers to be sampled.
  60 
  61     Args:
  62       sorted_indices_tensor: A sorted int32 tensor of shape [N] which contains
  63         the signed indices of the examples where the sign is based on the label
  64         value. The examples that cannot be sampled are set to 0. It samples
  65         atmost sample_size*positive_fraction positive examples and remaining
  66         from negative examples.
  67       sample_size: Size of subsamples.
  68 
  69     Returns:
  70       A tuple containing the number of positive and negative labels in the
  71       subsample.
  72     """
  73     input_length = tf.shape(sorted_indices_tensor)[0]
  74     valid_positive_index = tf.greater(sorted_indices_tensor,
  75                                       tf.zeros(input_length, tf.int32))
  76     num_sampled_pos = tf.reduce_sum(tf.cast(valid_positive_index, tf.int32))
  77     max_num_positive_samples = tf.constant(
  78         int(sample_size * self._positive_fraction), tf.int32)
  79     num_positive_samples = tf.minimum(max_num_positive_samples, num_sampled_pos)
  80     num_negative_samples = tf.constant(sample_size,
  81                                        tf.int32) - num_positive_samples
  82 
  83     return num_positive_samples, num_negative_samples
  84 
  85   def _get_values_from_start_and_end(self, input_tensor, num_start_samples,
  86                                      num_end_samples, total_num_samples):
  87     """slices num_start_samples and last num_end_samples from input_tensor.
  88 
  89     Args:
  90       input_tensor: An int32 tensor of shape [N] to be sliced.
  91       num_start_samples: Number of examples to be sliced from the beginning
  92         of the input tensor.
  93       num_end_samples: Number of examples to be sliced from the end of the
  94         input tensor.
  95       total_num_samples: Sum of is num_start_samples and num_end_samples. This
  96         should be a scalar.
  97 
  98     Returns:
  99       A tensor containing the first num_start_samples and last num_end_samples
 100       from input_tensor.
 101 
 102     """
 103     input_length = tf.shape(input_tensor)[0]
 104     start_positions = tf.less(tf.range(input_length), num_start_samples)
 105     end_positions = tf.greater_equal(
 106         tf.range(input_length), input_length - num_end_samples)
 107     selected_positions = tf.logical_or(start_positions, end_positions)
 108     selected_positions = tf.cast(selected_positions, tf.float32)
 109     indexed_positions = tf.multiply(tf.cumsum(selected_positions),
 110                                     selected_positions)
 111     one_hot_selector = tf.one_hot(tf.cast(indexed_positions, tf.int32) - 1,
 112                                   total_num_samples,
 113                                   dtype=tf.float32)
 114     return tf.cast(tf.tensordot(tf.cast(input_tensor, tf.float32),
 115                                 one_hot_selector, axes=[0, 0]), tf.int32)
 116 
 117   def _static_subsample(self, indicator, batch_size, labels):
 118     """Returns subsampled minibatch.
 119 
 120     Args:
 121       indicator: boolean tensor of shape [N] whose True entries can be sampled.
 122         N should be a complie time constant.
 123       batch_size: desired batch size. This scalar cannot be None.
 124       labels: boolean tensor of shape [N] denoting positive(=True) and negative
 125         (=False) examples. N should be a complie time constant.
 126 
 127     Returns:
 128       sampled_idx_indicator: boolean tensor of shape [N], True for entries which
 129         are sampled. It ensures the length of output of the subsample is always
 130         batch_size, even when number of examples set to True in indicator is
 131         less than batch_size.
 132 
 133     Raises:
 134       ValueError: if labels and indicator are not 1D boolean tensors.
 135     """
 136     # Check if indicator and labels have a static size.
 137     if not indicator.shape.is_fully_defined():
 138       raise ValueError('indicator must be static in shape when is_static is'
 139                        'True')
 140     if not labels.shape.is_fully_defined():
 141       raise ValueError('labels must be static in shape when is_static is'
 142                        'True')
 143     if not isinstance(batch_size, int):
 144       raise ValueError('batch_size has to be an integer when is_static is'
 145                        'True.')
 146 
 147     input_length = tf.shape(indicator)[0]
 148 
 149     # Set the number of examples set True in indicator to be at least
 150     # batch_size.
 151     num_true_sampled = tf.reduce_sum(tf.cast(indicator, tf.float32))
 152     additional_false_sample = tf.less_equal(
 153         tf.cumsum(tf.cast(tf.logical_not(indicator), tf.float32)),
 154         batch_size - num_true_sampled)
 155     indicator = tf.logical_or(indicator, additional_false_sample)
 156 
 157     # Shuffle indicator and label. Need to store the permutation to restore the
 158     # order post sampling.
 159     permutation = tf.random_shuffle(tf.range(input_length))
 160     indicator = tf.gather(indicator, permutation, axis=0)
 161     labels = tf.gather(labels, permutation, axis=0)
 162 
 163     # index (starting from 1) when indicator is True, 0 when False
 164     indicator_idx = tf.where(
 165         indicator, tf.range(1, input_length + 1),
 166         tf.zeros(input_length, tf.int32))
 167 
 168     # Replace -1 for negative, +1 for positive labels
 169     signed_label = tf.where(
 170         labels, tf.ones(input_length, tf.int32),
 171         tf.scalar_mul(-1, tf.ones(input_length, tf.int32)))
 172     # negative of index for negative label, positive index for positive label,
 173     # 0 when indicator is False.
 174     signed_indicator_idx = tf.multiply(indicator_idx, signed_label)
 175     sorted_signed_indicator_idx = tf.nn.top_k(
 176         signed_indicator_idx, input_length, sorted=True).values
 177 
 178     [num_positive_samples,
 179      num_negative_samples] = self._get_num_pos_neg_samples(
 180          sorted_signed_indicator_idx, batch_size)
 181 
 182     sampled_idx = self._get_values_from_start_and_end(
 183         sorted_signed_indicator_idx, num_positive_samples,
 184         num_negative_samples, batch_size)
 185 
 186     # Shift the indices to start from 0 and remove any samples that are set as
 187     # False.
 188     sampled_idx = tf.abs(sampled_idx) - tf.ones(batch_size, tf.int32)
 189     sampled_idx = tf.multiply(
 190         tf.cast(tf.greater_equal(sampled_idx, tf.constant(0)), tf.int32),
 191         sampled_idx)
 192 
 193     sampled_idx_indicator = tf.cast(tf.reduce_sum(
 194         tf.one_hot(sampled_idx, depth=input_length),
 195         axis=0), tf.bool)
 196 
 197     # project back the order based on stored permutations
 198     idx_indicator = tf.scatter_nd(
 199         tf.expand_dims(permutation, -1), sampled_idx_indicator,
 200         shape=(input_length,))
 201     return idx_indicator
 202 
 203   def subsample(self, indicator, batch_size, labels, scope=None):
 204     """Returns subsampled minibatch.
 205 
 206     Args:
 207       indicator: boolean tensor of shape [N] whose True entries can be sampled.
 208       batch_size: desired batch size. If None, keeps all positive samples and
 209         randomly selects negative samples so that the positive sample fraction
 210         matches self._positive_fraction. It cannot be None is is_static is True.
 211       labels: boolean tensor of shape [N] denoting positive(=True) and negative
 212           (=False) examples.
 213       scope: name scope.
 214 
 215     Returns:
 216       sampled_idx_indicator: boolean tensor of shape [N], True for entries which
 217         are sampled.
 218 
 219     Raises:
 220       ValueError: if labels and indicator are not 1D boolean tensors.
 221     """
 222     if len(indicator.get_shape().as_list()) != 1:
 223       raise ValueError('indicator must be 1 dimensional, got a tensor of '
 224                        'shape %s' % indicator.get_shape())
 225     if len(labels.get_shape().as_list()) != 1:
 226       raise ValueError('labels must be 1 dimensional, got a tensor of '
 227                        'shape %s' % labels.get_shape())
 228     if labels.dtype != tf.bool:
 229       raise ValueError('labels should be of type bool. Received: %s' %
 230                        labels.dtype)
 231     if indicator.dtype != tf.bool:
 232       raise ValueError('indicator should be of type bool. Received: %s' %
 233                        indicator.dtype)
 234     with tf.name_scope(scope, 'BalancedPositiveNegativeSampler'):
 235       if self._is_static:
 236         return self._static_subsample(indicator, batch_size, labels)
 237 
 238       else:
 239         # Only sample from indicated samples
 240         negative_idx = tf.logical_not(labels)
 241         positive_idx = tf.logical_and(labels, indicator)
 242         negative_idx = tf.logical_and(negative_idx, indicator)
 243 
 244         # Sample positive and negative samples separately
 245         if batch_size is None:
 246           max_num_pos = tf.reduce_sum(tf.cast(positive_idx, dtype=tf.int32))
 247         else:
 248           max_num_pos = int(self._positive_fraction * batch_size)
 249         sampled_pos_idx = self.subsample_indicator(positive_idx, max_num_pos)
 250         num_sampled_pos = tf.reduce_sum(tf.cast(sampled_pos_idx, tf.int32))
 251         if batch_size is None:
 252           negative_positive_ratio = (
 253               1 - self._positive_fraction) / self._positive_fraction
 254           max_num_neg = tf.cast(
 255               negative_positive_ratio *
 256               tf.cast(num_sampled_pos, dtype=tf.float32),
 257               dtype=tf.int32)
 258         else:
 259           max_num_neg = batch_size - num_sampled_pos
 260         sampled_neg_idx = self.subsample_indicator(negative_idx, max_num_neg)
 261 
 262         return tf.logical_or(sampled_pos_idx, sampled_neg_idx)
