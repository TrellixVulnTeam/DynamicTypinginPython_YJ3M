 282   def test_set_keypoint_visibilities(self):
 283     keypoints_np = np.array(
 284         [
 285             [[np.nan, 0.2],
 286              [np.nan, np.nan],
 287              [-3., 7.]],
 288             [[0.5, 0.2],
 289              [4., 1.0],
 290              [-3., np.nan]],
 291         ], dtype=np.float32)
 292     initial_keypoint_visibilities_np = np.array(
 293         [
 294             [False,
 295              True,  # Will be overriden by NaN coords.
 296              False],  # Will be maintained, even though non-NaN coords.
 297             [True,
 298              False,  # Will be maintained, even though non-NaN coords.
 299              False]
 300         ])
 301     def graph_fn():
 302       keypoints = tf.constant(keypoints_np, dtype=tf.float32)
 303       initial_keypoint_visibilities = tf.constant(
 304           initial_keypoint_visibilities_np, dtype=tf.bool)
 305       keypoint_visibilities = keypoint_ops.set_keypoint_visibilities(
 306           keypoints, initial_keypoint_visibilities)
 307       return keypoint_visibilities
 308 
 309     expected_kpt_vis = [
 310         [False, False, False],
 311         [True, False, False]
 312     ]
 313     output = self.execute(graph_fn, [])
 314     self.assertAllEqual(expected_kpt_vis, output)
