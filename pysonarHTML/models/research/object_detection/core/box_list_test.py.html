  16 """Tests for object_detection.core.box_list."""
  17 import numpy as np
  18 import tensorflow.compat.v1 as tf
  19 
  20 from object_detection.core import box_list
  21 from object_detection.utils import test_case
  22 
  23 
  24 class BoxListTest(test_case.TestCase):
  25   """Tests for BoxList class."""
  26 
  27   def test_num_boxes(self):
  28     def graph_fn():
  29       data = tf.constant([[0, 0, 1, 1], [1, 1, 2, 3], [3, 4, 5, 5]], tf.float32)
  30       boxes = box_list.BoxList(data)
  31       return boxes.num_boxes()
  32     num_boxes_out = self.execute(graph_fn, [])
  33     self.assertEqual(num_boxes_out, 3)
  34 
  35   def test_get_correct_center_coordinates_and_sizes(self):
  36     boxes = np.array([[10.0, 10.0, 20.0, 15.0], [0.2, 0.1, 0.5, 0.4]],
  37                      np.float32)
  38     def graph_fn(boxes):
  39       boxes = box_list.BoxList(boxes)
  40       centers_sizes = boxes.get_center_coordinates_and_sizes()
  41       return centers_sizes
  42     centers_sizes_out = self.execute(graph_fn, [boxes])
  43     expected_centers_sizes = [[15, 0.35], [12.5, 0.25], [10, 0.3], [5, 0.3]]
  44     self.assertAllClose(centers_sizes_out, expected_centers_sizes)
  45 
  46   def test_create_box_list_with_dynamic_shape(self):
  47     def graph_fn():
  48       data = tf.constant([[0, 0, 1, 1], [1, 1, 2, 3], [3, 4, 5, 5]], tf.float32)
  49       indices = tf.reshape(tf.where(tf.greater([1, 0, 1], 0)), [-1])
  50       data = tf.gather(data, indices)
  51       assert data.get_shape().as_list() == [None, 4]
  52       boxes = box_list.BoxList(data)
  53       return boxes.num_boxes()
  54     num_boxes = self.execute(graph_fn, [])
  55     self.assertEqual(num_boxes, 2)
  56 
  57   def test_transpose_coordinates(self):
  58     boxes = np.array([[10.0, 10.0, 20.0, 15.0], [0.2, 0.1, 0.5, 0.4]],
  59                      np.float32)
  60     def graph_fn(boxes):
  61       boxes = box_list.BoxList(boxes)
  62       boxes.transpose_coordinates()
  63       return boxes.get()
  64     transpoded_boxes = self.execute(graph_fn, [boxes])
  65     expected_corners = [[10.0, 10.0, 15.0, 20.0], [0.1, 0.2, 0.4, 0.5]]
  66     self.assertAllClose(transpoded_boxes, expected_corners)
  67 
  68   def test_box_list_invalid_inputs(self):
  69     data0 = tf.constant([[[0, 0, 1, 1], [3, 4, 5, 5]]], tf.float32)
  70     data1 = tf.constant([[0, 0, 1], [1, 1, 2], [3, 4, 5]], tf.float32)
  71     data2 = tf.constant([[0, 0, 1], [1, 1, 2], [3, 4, 5]], tf.int32)
  72 
  73     with self.assertRaises(ValueError):
  74       _ = box_list.BoxList(data0)
  75     with self.assertRaises(ValueError):
  76       _ = box_list.BoxList(data1)
  77     with self.assertRaises(ValueError):
  78       _ = box_list.BoxList(data2)
  79 
  80   def test_num_boxes_static(self):
  81     box_corners = [[10.0, 10.0, 20.0, 15.0], [0.2, 0.1, 0.5, 0.4]]
  82     boxes = box_list.BoxList(tf.constant(box_corners))
  83     self.assertEqual(boxes.num_boxes_static(), 2)
  84     self.assertEqual(type(boxes.num_boxes_static()), int)
  85 
  86   def test_as_tensor_dict(self):
  87     boxes = tf.constant([[0.1, 0.1, 0.4, 0.4], [0.1, 0.1, 0.5, 0.5]],
  88                         tf.float32)
  89     boxlist = box_list.BoxList(boxes)
  90     classes = tf.constant([0, 1])
  91     boxlist.add_field('classes', classes)
  92     scores = tf.constant([0.75, 0.2])
  93     boxlist.add_field('scores', scores)
  94     tensor_dict = boxlist.as_tensor_dict()
  95 
  96     self.assertDictEqual(tensor_dict, {'scores': scores, 'classes': classes,
  97                                        'boxes': boxes})
  98 
  99   def test_as_tensor_dict_with_features(self):
 100     boxes = tf.constant([[0.1, 0.1, 0.4, 0.4], [0.1, 0.1, 0.5, 0.5]],
 101                         tf.float32)
 102     boxlist = box_list.BoxList(boxes)
 103     classes = tf.constant([0, 1])
 104     boxlist.add_field('classes', classes)
 105     scores = tf.constant([0.75, 0.2])
 106     boxlist.add_field('scores', scores)
 107     tensor_dict = boxlist.as_tensor_dict(['scores', 'classes'])
 108 
 109     self.assertDictEqual(tensor_dict, {'scores': scores, 'classes': classes})
 110 
 111   def test_as_tensor_dict_missing_field(self):
 112     boxlist = box_list.BoxList(
 113         tf.constant([[0.1, 0.1, 0.4, 0.4], [0.1, 0.1, 0.5, 0.5]], tf.float32))
 114     boxlist.add_field('classes', tf.constant([0, 1]))
 115     boxlist.add_field('scores', tf.constant([0.75, 0.2]))
 116     with self.assertRaises(ValueError):
 117       boxlist.as_tensor_dict(['foo', 'bar'])
