  27 class InputDataFields(object):
  28   """Names for the input tensors.
  29 
  30   Holds the standard data field names to use for identifying input tensors. This
  31   should be used by the decoder to identify keys for the returned tensor_dict
  32   containing input tensors. And it should be used by the model to identify the
  33   tensors it needs.
  34 
  35   Attributes:
  36     image: image.
  37     image_additional_channels: additional channels.
  38     original_image: image in the original input size.
  39     original_image_spatial_shape: image in the original input size.
  40     key: unique key corresponding to image.
  41     source_id: source of the original image.
  42     filename: original filename of the dataset (without common path).
  43     groundtruth_image_classes: image-level class labels.
  44     groundtruth_image_confidences: image-level class confidences.
  45     groundtruth_labeled_classes: image-level annotation that indicates the
  46       classes for which an image has been labeled.
  47     groundtruth_boxes: coordinates of the ground truth boxes in the image.
  48     groundtruth_classes: box-level class labels.
  49     groundtruth_confidences: box-level class confidences. The shape should be
  50       the same as the shape of groundtruth_classes.
  51     groundtruth_label_types: box-level label types (e.g. explicit negative).
  52     groundtruth_is_crowd: [DEPRECATED, use groundtruth_group_of instead]
  53       is the groundtruth a single object or a crowd.
  54     groundtruth_area: area of a groundtruth segment.
  55     groundtruth_difficult: is a `difficult` object
  56     groundtruth_group_of: is a `group_of` objects, e.g. multiple objects of the
  57       same class, forming a connected group, where instances are heavily
  58       occluding each other.
  59     proposal_boxes: coordinates of object proposal boxes.
  60     proposal_objectness: objectness score of each proposal.
  61     groundtruth_instance_masks: ground truth instance masks.
  62     groundtruth_instance_boundaries: ground truth instance boundaries.
  63     groundtruth_instance_classes: instance mask-level class labels.
  64     groundtruth_keypoints: ground truth keypoints.
  65     groundtruth_keypoint_visibilities: ground truth keypoint visibilities.
  66     groundtruth_keypoint_weights: groundtruth weight factor for keypoints.
  67     groundtruth_label_weights: groundtruth label weights.
  68     groundtruth_weights: groundtruth weight factor for bounding boxes.
  69     num_groundtruth_boxes: number of groundtruth boxes.
  70     is_annotated: whether an image has been labeled or not.
  71     true_image_shapes: true shapes of images in the resized images, as resized
  72       images can be padded with zeros.
  73     multiclass_scores: the label score per class for each box.
  74     context_features: a flattened list of contextual features.
  75     context_feature_length: the fixed length of each feature in
  76       context_features, used for reshaping.
  77     valid_context_size: the valid context size, used in filtering the padded
  78       context features.
  79     image_format: format for the images, used to decode
  80     image_height: height of images, used to decode
  81     image_width: width of images, used to decode
  82   """
  83   image = 'image'
  84   image_additional_channels = 'image_additional_channels'
  85   original_image = 'original_image'
  86   original_image_spatial_shape = 'original_image_spatial_shape'
  87   key = 'key'
  88   source_id = 'source_id'
  89   filename = 'filename'
  90   groundtruth_image_classes = 'groundtruth_image_classes'
  91   groundtruth_image_confidences = 'groundtruth_image_confidences'
  92   groundtruth_labeled_classes = 'groundtruth_labeled_classes'
  93   groundtruth_boxes = 'groundtruth_boxes'
  94   groundtruth_classes = 'groundtruth_classes'
  95   groundtruth_confidences = 'groundtruth_confidences'
  96   groundtruth_label_types = 'groundtruth_label_types'
  97   groundtruth_is_crowd = 'groundtruth_is_crowd'
  98   groundtruth_area = 'groundtruth_area'
  99   groundtruth_difficult = 'groundtruth_difficult'
 100   groundtruth_group_of = 'groundtruth_group_of'
 101   proposal_boxes = 'proposal_boxes'
 102   proposal_objectness = 'proposal_objectness'
 103   groundtruth_instance_masks = 'groundtruth_instance_masks'
 104   groundtruth_instance_boundaries = 'groundtruth_instance_boundaries'
 105   groundtruth_instance_classes = 'groundtruth_instance_classes'
 106   groundtruth_keypoints = 'groundtruth_keypoints'
 107   groundtruth_keypoint_visibilities = 'groundtruth_keypoint_visibilities'
 108   groundtruth_keypoint_weights = 'groundtruth_keypoint_weights'
 109   groundtruth_label_weights = 'groundtruth_label_weights'
 110   groundtruth_weights = 'groundtruth_weights'
 111   num_groundtruth_boxes = 'num_groundtruth_boxes'
 112   is_annotated = 'is_annotated'
 113   true_image_shape = 'true_image_shape'
 114   multiclass_scores = 'multiclass_scores'
 115   context_features = 'context_features'
 116   context_feature_length = 'context_feature_length'
 117   valid_context_size = 'valid_context_size'
 118   image_timestamps = 'image_timestamps'
 119   image_format = 'image_format'
 120   image_height = 'image_height'
 121   image_width = 'image_width'

 124 class DetectionResultFields(object):
 125   """Naming conventions for storing the output of the detector.
 126 
 127   Attributes:
 128     source_id: source of the original image.
 129     key: unique key corresponding to image.
 130     detection_boxes: coordinates of the detection boxes in the image.
 131     detection_scores: detection scores for the detection boxes in the image.
 132     detection_multiclass_scores: class score distribution (including background)
 133       for detection boxes in the image including background class.
 134     detection_classes: detection-level class labels.
 135     detection_masks: contains a segmentation mask for each detection box.
 136     detection_boundaries: contains an object boundary for each detection box.
 137     detection_keypoints: contains detection keypoints for each detection box.
 138     detection_keypoint_scores: contains detection keypoint scores.
 139     num_detections: number of detections in the batch.
 140     raw_detection_boxes: contains decoded detection boxes without Non-Max
 141       suppression.
 142     raw_detection_scores: contains class score logits for raw detection boxes.
 143     detection_anchor_indices: The anchor indices of the detections after NMS.
 144     detection_features: contains extracted features for each detected box
 145       after NMS.
 146   """
 147 
 148   source_id = 'source_id'
 149   key = 'key'
 150   detection_boxes = 'detection_boxes'
 151   detection_scores = 'detection_scores'
 152   detection_multiclass_scores = 'detection_multiclass_scores'
 153   detection_features = 'detection_features'
 154   detection_classes = 'detection_classes'
 155   detection_masks = 'detection_masks'
 156   detection_boundaries = 'detection_boundaries'
 157   detection_keypoints = 'detection_keypoints'
 158   detection_keypoint_scores = 'detection_keypoint_scores'
 159   num_detections = 'num_detections'
 160   raw_detection_boxes = 'raw_detection_boxes'
 161   raw_detection_scores = 'raw_detection_scores'
 162   detection_anchor_indices = 'detection_anchor_indices'

 211 class TfExampleFields(object):
 212   """TF-example proto feature names for object detection.
 213 
 214   Holds the standard feature names to load from an Example proto for object
 215   detection.
 216 
 217   Attributes:
 218     image_encoded: JPEG encoded string
 219     image_format: image format, e.g. "JPEG"
 220     filename: filename
 221     channels: number of channels of image
 222     colorspace: colorspace, e.g. "RGB"
 223     height: height of image in pixels, e.g. 462
 224     width: width of image in pixels, e.g. 581
 225     source_id: original source of the image
 226     image_class_text: image-level label in text format
 227     image_class_label: image-level label in numerical format
 228     image_class_confidence: image-level confidence of the label
 229     object_class_text: labels in text format, e.g. ["person", "cat"]
 230     object_class_label: labels in numbers, e.g. [16, 8]
 231     object_bbox_xmin: xmin coordinates of groundtruth box, e.g. 10, 30
 232     object_bbox_xmax: xmax coordinates of groundtruth box, e.g. 50, 40
 233     object_bbox_ymin: ymin coordinates of groundtruth box, e.g. 40, 50
 234     object_bbox_ymax: ymax coordinates of groundtruth box, e.g. 80, 70
 235     object_view: viewpoint of object, e.g. ["frontal", "left"]
 236     object_truncated: is object truncated, e.g. [true, false]
 237     object_occluded: is object occluded, e.g. [true, false]
 238     object_difficult: is object difficult, e.g. [true, false]
 239     object_group_of: is object a single object or a group of objects
 240     object_depiction: is object a depiction
 241     object_is_crowd: [DEPRECATED, use object_group_of instead]
 242       is the object a single object or a crowd
 243     object_segment_area: the area of the segment.
 244     object_weight: a weight factor for the object's bounding box.
 245     instance_masks: instance segmentation masks.
 246     instance_boundaries: instance boundaries.
 247     instance_classes: Classes for each instance segmentation mask.
 248     detection_class_label: class label in numbers.
 249     detection_bbox_ymin: ymin coordinates of a detection box.
 250     detection_bbox_xmin: xmin coordinates of a detection box.
 251     detection_bbox_ymax: ymax coordinates of a detection box.
 252     detection_bbox_xmax: xmax coordinates of a detection box.
 253     detection_score: detection score for the class label and box.
 254   """
 255   image_encoded = 'image/encoded'
 256   image_format = 'image/format'  # format is reserved keyword
 257   filename = 'image/filename'
 258   channels = 'image/channels'
 259   colorspace = 'image/colorspace'
 260   height = 'image/height'
 261   width = 'image/width'
 262   source_id = 'image/source_id'
 263   image_class_text = 'image/class/text'
 264   image_class_label = 'image/class/label'
 265   image_class_confidence = 'image/class/confidence'
 266   object_class_text = 'image/object/class/text'
 267   object_class_label = 'image/object/class/label'
 268   object_bbox_ymin = 'image/object/bbox/ymin'
 269   object_bbox_xmin = 'image/object/bbox/xmin'
 270   object_bbox_ymax = 'image/object/bbox/ymax'
 271   object_bbox_xmax = 'image/object/bbox/xmax'
 272   object_view = 'image/object/view'
 273   object_truncated = 'image/object/truncated'
 274   object_occluded = 'image/object/occluded'
 275   object_difficult = 'image/object/difficult'
 276   object_group_of = 'image/object/group_of'
 277   object_depiction = 'image/object/depiction'
 278   object_is_crowd = 'image/object/is_crowd'
 279   object_segment_area = 'image/object/segment/area'
 280   object_weight = 'image/object/weight'
 281   instance_masks = 'image/segmentation/object'
 282   instance_boundaries = 'image/boundaries/object'
 283   instance_classes = 'image/segmentation/object/class'
 284   detection_class_label = 'image/detection/label'
 285   detection_bbox_ymin = 'image/detection/bbox/ymin'
 286   detection_bbox_xmin = 'image/detection/bbox/xmin'
 287   detection_bbox_ymax = 'image/detection/bbox/ymax'
 288   detection_bbox_xmax = 'image/detection/bbox/xmax'
 289   detection_score = 'image/detection/score'
