  26   def test_class_agnostic_nms_select_with_shared_boxes(self):
  27     def graph_fn():
  28       boxes = tf.constant(
  29           [[[0, 0, 1, 1]], [[0, 0.1, 1, 1.1]], [[0, -0.1, 1, 0.9]],
  30            [[0, 10, 1, 11]], [[0, 10.1, 1, 11.1]], [[0, 100, 1, 101]],
  31            [[0, 1000, 1, 1002]], [[0, 1000, 1, 1002.1]]], tf.float32)
  32       scores = tf.constant([[.9, 0.01], [.75, 0.05], [.6, 0.01], [.95, 0],
  33                             [.5, 0.01], [.3, 0.01], [.01, .85], [.01, .5]])
  34       score_thresh = 0.1
  35       iou_thresh = .5
  36       max_classes_per_detection = 1
  37       max_output_size = 4
  38       nms, _ = post_processing.class_agnostic_non_max_suppression(
  39           boxes, scores, score_thresh, iou_thresh, max_classes_per_detection,
  40           max_output_size)
  41       return (nms.get(), nms.get_field(fields.BoxListFields.scores),
  42               nms.get_field(fields.BoxListFields.classes))
  43 
  44     exp_nms_corners = [[0, 10, 1, 11], [0, 0, 1, 1], [0, 1000, 1, 1002],
  45                        [0, 100, 1, 101]]
  46     exp_nms_scores = [.95, .9, .85, .3]
  47     exp_nms_classes = [0, 0, 1, 0]
  48 
  49     (nms_corners_output, nms_scores_output,
  50      nms_classes_output) = self.execute_cpu(graph_fn, [])
  51     self.assertAllClose(nms_corners_output, exp_nms_corners)
  52     self.assertAllClose(nms_scores_output, exp_nms_scores)
  53     self.assertAllClose(nms_classes_output, exp_nms_classes)

  96   # Two cases will be tested here: using / not using static shapes.
  97   # Named the two test cases for easier control during testing, with a flag of
  98   # '--test_filter=ClassAgnosticNonMaxSuppressionTest.test_batch_classagnostic_nms_with_batch_size_1'
  99   # or
 100   # '--test_filter=ClassAgnosticNonMaxSuppressionTest.test_batch_classagnostic_nms_with_batch_size_1_use_static_shapes'.
 101   @parameterized.named_parameters(('', False), ('_use_static_shapes', True))
 102   def test_batch_classagnostic_nms_with_batch_size_1(self,
 103                                                      use_static_shapes=False):
 104     def graph_fn():
 105       boxes = tf.constant(
 106           [[[[0, 0, 1, 1]], [[0, 0.1, 1, 1.1]], [[0, -0.1, 1, 0.9]],
 107             [[0, 10, 1, 11]], [[0, 10.1, 1, 11.1]], [[0, 100, 1, 101]],
 108             [[0, 1000, 1, 1002]], [[0, 1000, 1, 1002.1]]]], tf.float32)
 109       scores = tf.constant([[[.9, 0.01], [.75, 0.05], [.6, 0.01], [.95, 0],
 110                              [.5, 0.01], [.3, 0.01], [.01, .85], [.01, .5]]])
 111       score_thresh = 0.1
 112       iou_thresh = .5
 113       max_output_size = 4
 114       max_classes_per_detection = 1
 115       use_class_agnostic_nms = True
 116       (nmsed_boxes, nmsed_scores, nmsed_classes, nmsed_masks,
 117        nmsed_additional_fields,
 118        num_detections) = post_processing.batch_multiclass_non_max_suppression(
 119            boxes,
 120            scores,
 121            score_thresh,
 122            iou_thresh,
 123            max_size_per_class=max_output_size,
 124            max_total_size=max_output_size,
 125            use_class_agnostic_nms=use_class_agnostic_nms,
 126            use_static_shapes=use_static_shapes,
 127            max_classes_per_detection=max_classes_per_detection)
 128       self.assertIsNone(nmsed_masks)
 129       self.assertIsNone(nmsed_additional_fields)
 130       return (nmsed_boxes, nmsed_scores, nmsed_classes, num_detections)
 131     exp_nms_corners = [[[0, 10, 1, 11], [0, 0, 1, 1], [0, 1000, 1, 1002],
 132                         [0, 100, 1, 101]]]
 133     exp_nms_scores = [[.95, .9, .85, .3]]
 134     exp_nms_classes = [[0, 0, 1, 0]]
 135     (nmsed_boxes, nmsed_scores, nmsed_classes,
 136      num_detections) = self.execute_cpu(graph_fn, [])
 137     self.assertAllClose(nmsed_boxes, exp_nms_corners)
 138     self.assertAllClose(nmsed_scores, exp_nms_scores)
 139     self.assertAllClose(nmsed_classes, exp_nms_classes)
 140     self.assertEqual(num_detections, [4])
