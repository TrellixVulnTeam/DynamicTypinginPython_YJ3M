  35 def batch_iou(boxes1, boxes2):
  36   """Calculates the overlap between proposal and ground truth boxes.
  37 
  38   Some `boxes2` may have been padded.  The returned `iou` tensor for these
  39   boxes will be -1.
  40 
  41   Args:
  42     boxes1: a tensor with a shape of [batch_size, N, 4]. N is the number of
  43       proposals before groundtruth assignment. The last dimension is the pixel
  44       coordinates in [ymin, xmin, ymax, xmax] form.
  45     boxes2: a tensor with a shape of [batch_size, MAX_NUM_INSTANCES, 4]. This
  46       tensor might have paddings with a negative value.
  47 
  48   Returns:
  49     iou: a tensor with as a shape of [batch_size, N, MAX_NUM_INSTANCES].
  50   """
  51   with tf.name_scope('BatchIOU'):
  52     y1_min, x1_min, y1_max, x1_max = tf.split(
  53         value=boxes1, num_or_size_splits=4, axis=2)
  54     y2_min, x2_min, y2_max, x2_max = tf.split(
  55         value=boxes2, num_or_size_splits=4, axis=2)
  56 
  57     # Calculates the intersection area.
  58     intersection_xmin = tf.maximum(x1_min, tf.transpose(x2_min, [0, 2, 1]))
  59     intersection_xmax = tf.minimum(x1_max, tf.transpose(x2_max, [0, 2, 1]))
  60     intersection_ymin = tf.maximum(y1_min, tf.transpose(y2_min, [0, 2, 1]))
  61     intersection_ymax = tf.minimum(y1_max, tf.transpose(y2_max, [0, 2, 1]))
  62     intersection_area = tf.maximum(
  63         (intersection_xmax - intersection_xmin), 0) * tf.maximum(
  64             (intersection_ymax - intersection_ymin), 0)
  65 
  66     # Calculates the union area.
  67     area1 = (y1_max - y1_min) * (x1_max - x1_min)
  68     area2 = (y2_max - y2_min) * (x2_max - x2_min)
  69     # Adds a small epsilon to avoid divide-by-zero.
  70     union_area = area1 + tf.transpose(area2,
  71                                       [0, 2, 1]) - intersection_area + 1e-8
  72 
  73     # Calculates IoU.
  74     iou = intersection_area / union_area
  75 
  76     # Fills -1 for padded ground truth boxes.
  77     padding_mask = tf.logical_and(
  78         tf.less(intersection_xmax, 0), tf.less(intersection_ymax, 0))
  79     iou = tf.where(padding_mask, -tf.ones_like(iou), iou)
  80 
  81     return iou

  84 def _self_suppression(iou, iou_threshold, loop_condition, iou_sum):
  85   """Bounding-boxes self-suppression loop body.
  86 
  87   Args:
  88     iou: A float Tensor with shape [1, num_boxes, max_num_instance]: IOUs.
  89     iou_threshold: A scalar, representing IOU threshold.
  90     loop_condition: The loop condition returned from last iteration.
  91     iou_sum: iou_sum_new returned from last iteration.
  92 
  93   Returns:
  94     iou_suppressed: A float Tensor with shape [1, num_boxes, max_num_instance],
  95                     IOU after suppression.
  96     iou_threshold: A scalar, representing IOU threshold.
  97     loop_condition: Bool Tensor of shape [], the loop condition.
  98     iou_sum_new: The new IOU sum.
  99   """
 100   del loop_condition
 101   can_suppress_others = tf.cast(
 102       tf.reshape(tf.reduce_max(iou, 1) <= iou_threshold, [1, -1, 1]), iou.dtype)
 103   iou_suppressed = tf.reshape(
 104       tf.cast(
 105           tf.reduce_max(can_suppress_others * iou, 1) <= iou_threshold,
 106           iou.dtype), [1, -1, 1]) * iou
 107   iou_sum_new = tf.reduce_sum(iou_suppressed, [1, 2])
 108   return [
 109       iou_suppressed, iou_threshold,
 110       tf.reduce_any(iou_sum - iou_sum_new > iou_threshold), iou_sum_new
 111   ]

 114 def _cross_suppression(boxes, box_slice, iou_threshold, inner_idx):
 115   """Bounding-boxes cross-suppression loop body.
 116 
 117   Args:
 118     boxes: A float Tensor of shape [1, anchors, 4], representing boxes.
 119     box_slice: A float Tensor of shape [1, _NMS_TILE_SIZE, 4], the box tile
 120       returned from last iteration
 121     iou_threshold: A scalar, representing IOU threshold.
 122     inner_idx: A scalar, representing inner index.
 123 
 124   Returns:
 125     boxes: A float Tensor of shape [1, anchors, 4], representing boxes.
 126     ret_slice: A float Tensor of shape [1, _NMS_TILE_SIZE, 4], the box tile
 127                after suppression
 128     iou_threshold: A scalar, representing IOU threshold.
 129     inner_idx: A scalar, inner index incremented.
 130   """
 131   new_slice = tf.slice(boxes, [0, inner_idx * _NMS_TILE_SIZE, 0],
 132                        [1, _NMS_TILE_SIZE, 4])
 133   iou = batch_iou(new_slice, box_slice)
 134   ret_slice = tf.expand_dims(
 135       tf.cast(tf.reduce_all(iou < iou_threshold, [1]), box_slice.dtype),
 136       2) * box_slice
 137   return boxes, ret_slice, iou_threshold, inner_idx + 1

 140 def _suppression_loop_body(boxes, iou_threshold, output_size, idx):
 141   """Process boxes in the range [idx*_NMS_TILE_SIZE, (idx+1)*_NMS_TILE_SIZE).
 142 
 143   Args:
 144     boxes: a tensor with a shape of [1, anchors, 4].
 145     iou_threshold: a float representing the threshold for deciding whether boxes
 146       overlap too much with respect to IOU.
 147     output_size: an int32 tensor of size [1]. Representing the number of
 148       selected boxes.
 149     idx: an integer scalar representing induction variable.
 150 
 151   Returns:
 152     boxes: updated boxes.
 153     iou_threshold: pass down iou_threshold to the next iteration.
 154     output_size: the updated output_size.
 155     idx: the updated induction variable.
 156   """
 157   num_tiles = tf.shape(boxes)[1] // _NMS_TILE_SIZE
 158 
 159   # Iterates over tiles that can possibly suppress the current tile.
 160   box_slice = tf.slice(boxes, [0, idx * _NMS_TILE_SIZE, 0],
 161                        [1, _NMS_TILE_SIZE, 4])
 162   _, box_slice, _, _ = tf.while_loop(
 163       lambda _boxes, _box_slice, _threshold, inner_idx: inner_idx < idx,
 164       _cross_suppression, [boxes, box_slice, iou_threshold,
 165                            tf.constant(0)])
 166 
 167   # Iterates over the current tile to compute self-suppression.
 168   iou = batch_iou(box_slice, box_slice)
 169   mask = tf.expand_dims(
 170       tf.reshape(tf.range(_NMS_TILE_SIZE), [1, -1]) > tf.reshape(
 171           tf.range(_NMS_TILE_SIZE), [-1, 1]), 0)
 172   iou *= tf.cast(tf.logical_and(mask, iou >= iou_threshold), iou.dtype)
 173   suppressed_iou, _, _, _ = tf.while_loop(
 174       lambda _iou, _threshold, loop_condition, _iou_sum: loop_condition,
 175       _self_suppression,
 176       [iou, iou_threshold,
 177        tf.constant(True),
 178        tf.reduce_sum(iou, [1, 2])])
 179   suppressed_box = tf.reduce_sum(suppressed_iou, 1) > 0
 180   box_slice *= tf.expand_dims(1.0 - tf.cast(suppressed_box, box_slice.dtype), 2)
 181 
 182   # Uses box_slice to update the input boxes.
 183   mask = tf.reshape(
 184       tf.cast(tf.equal(tf.range(num_tiles), idx), boxes.dtype), [1, -1, 1, 1])
 185   boxes = tf.tile(tf.expand_dims(box_slice, [1]),
 186                   [1, num_tiles, 1, 1]) * mask + tf.reshape(
 187                       boxes, [1, num_tiles, _NMS_TILE_SIZE, 4]) * (1 - mask)
 188   boxes = tf.reshape(boxes, [1, -1, 4])
 189 
 190   # Updates output_size.
 191   output_size += tf.reduce_sum(
 192       tf.cast(tf.reduce_any(box_slice > 0, [2]), tf.int32), [1])
 193   return boxes, iou_threshold, output_size, idx + 1

 196 def partitioned_non_max_suppression_padded(boxes,
 197                                            scores,
 198                                            max_output_size,
 199                                            iou_threshold=0.5,
 200                                            score_threshold=float('-inf')):
 201   """A tiled version of [`tf.image.non_max_suppression_padded`](https://www.tensorflow.org/api_docs/python/tf/image/non_max_suppression_padded).
 202 
 203   The overall design of the algorithm is to handle boxes tile-by-tile:
 204 
 205   boxes = boxes.pad_to_multiple_of(tile_size)
 206   num_tiles = len(boxes) // tile_size
 207   output_boxes = []
 208   for i in range(num_tiles):
 209     box_tile = boxes[i*tile_size : (i+1)*tile_size]
 210     for j in range(i - 1):
 211       suppressing_tile = boxes[j*tile_size : (j+1)*tile_size]
 212       iou = batch_iou(box_tile, suppressing_tile)
 213       # if the box is suppressed in iou, clear it to a dot
 214       box_tile *= _update_boxes(iou)
 215     # Iteratively handle the diagonal tile.
 216     iou = _box_overlap(box_tile, box_tile)
 217     iou_changed = True
 218     while iou_changed:
 219       # boxes that are not suppressed by anything else
 220       suppressing_boxes = _get_suppressing_boxes(iou)
 221       # boxes that are suppressed by suppressing_boxes
 222       suppressed_boxes = _get_suppressed_boxes(iou, suppressing_boxes)
 223       # clear iou to 0 for boxes that are suppressed, as they cannot be used
 224       # to suppress other boxes any more
 225       new_iou = _clear_iou(iou, suppressed_boxes)
 226       iou_changed = (new_iou != iou)
 227       iou = new_iou
 228     # remaining boxes that can still suppress others, are selected boxes.
 229     output_boxes.append(_get_suppressing_boxes(iou))
 230     if len(output_boxes) >= max_output_size:
 231       break
 232 
 233   Args:
 234     boxes: A 2-D float `Tensor` of shape `[num_boxes, 4]`.
 235     scores: A 1-D float `Tensor` of shape `[num_boxes]` representing a single
 236       score corresponding to each box (each row of boxes).
 237     max_output_size: a scalar integer `Tensor` representing the maximum number
 238       of boxes to be selected by non max suppression.
 239     iou_threshold: a float representing the threshold for deciding whether boxes
 240       overlap too much with respect to IOU.
 241     score_threshold: A float representing the threshold for deciding when to
 242       remove boxes based on score.
 243 
 244   Returns:
 245     selected_indices: a tensor of shape [anchors].
 246     num_valid_boxes: a scalar int tensor.
 247     nms_proposals: a tensor with a shape of [anchors, 4]. It has
 248       same dtype as input boxes.
 249     nms_scores: a tensor with a shape of [anchors]. It has same
 250       dtype as input scores.
 251     argsort_ids: a tensor of shape [anchors], mapping from input order of boxes
 252       to output order of boxes.
 253   """
 254   num_boxes = tf.shape(boxes)[0]
 255   pad = tf.cast(
 256       tf.ceil(tf.cast(num_boxes, tf.float32) / _NMS_TILE_SIZE),
 257       tf.int32) * _NMS_TILE_SIZE - num_boxes
 258 
 259   scores, argsort_ids = tf.nn.top_k(scores, k=num_boxes, sorted=True)
 260   boxes = tf.gather(boxes, argsort_ids)
 261   num_boxes = tf.shape(boxes)[0]
 262   num_boxes += pad
 263   boxes = tf.pad(
 264       tf.cast(boxes, tf.float32), [[0, pad], [0, 0]], constant_values=-1)
 265   scores = tf.pad(tf.cast(scores, tf.float32), [[0, pad]])
 266 
 267   # mask boxes to -1 by score threshold
 268   scores_mask = tf.expand_dims(
 269       tf.cast(scores > score_threshold, boxes.dtype), axis=1)
 270   boxes = ((boxes + 1.) * scores_mask) - 1.
 271 
 272   boxes = tf.expand_dims(boxes, axis=0)
 273   scores = tf.expand_dims(scores, axis=0)
 274 
 275   def _loop_cond(unused_boxes, unused_threshold, output_size, idx):
 276     return tf.logical_and(
 277         tf.reduce_min(output_size) < max_output_size,
 278         idx < num_boxes // _NMS_TILE_SIZE)
 279 
 280   selected_boxes, _, output_size, _ = tf.while_loop(
 281       _loop_cond, _suppression_loop_body,
 282       [boxes, iou_threshold,
 283        tf.zeros([1], tf.int32),
 284        tf.constant(0)])
 285   idx = num_boxes - tf.cast(
 286       tf.nn.top_k(
 287           tf.cast(tf.reduce_any(selected_boxes > 0, [2]), tf.int32) *
 288           tf.expand_dims(tf.range(num_boxes, 0, -1), 0), max_output_size)[0],
 289       tf.int32)
 290   idx = tf.minimum(idx, num_boxes - 1 - pad)
 291   idx = tf.reshape(idx + tf.reshape(tf.range(1) * num_boxes, [-1, 1]), [-1])
 292   num_valid_boxes = tf.reduce_sum(output_size)
 293   return (idx, num_valid_boxes, tf.reshape(boxes, [-1, 4]),
 294           tf.reshape(scores, [-1]), argsort_ids)

 297 def _validate_boxes_scores_iou_thresh(boxes, scores, iou_thresh,
 298                                       change_coordinate_frame, clip_window):
 299   """Validates boxes, scores and iou_thresh.
 300 
 301   This function validates the boxes, scores, iou_thresh
 302      and if change_coordinate_frame is True, clip_window must be specified.
 303 
 304   Args:
 305     boxes: A [k, q, 4] float32 tensor containing k detections. `q` can be either
 306       number of classes or 1 depending on whether a separate box is predicted
 307       per class.
 308     scores: A [k, num_classes] float32 tensor containing the scores for each of
 309       the k detections. The scores have to be non-negative when
 310       pad_to_max_output_size is True.
 311     iou_thresh: scalar threshold for IOU (new boxes that have high IOU overlap
 312       with previously selected boxes are removed).
 313     change_coordinate_frame: Whether to normalize coordinates after clipping
 314       relative to clip_window (this can only be set to True if a clip_window is
 315       provided)
 316     clip_window: A float32 tensor of the form [y_min, x_min, y_max, x_max]
 317       representing the window to clip and normalize boxes to before performing
 318       non-max suppression.
 319 
 320   Raises:
 321     ValueError: if iou_thresh is not in [0, 1] or if input boxlist does not
 322     have a valid scores field.
 323   """
 324   if not 0 <= iou_thresh <= 1.0:
 325     raise ValueError('iou_thresh must be between 0 and 1')
 326   if scores.shape.ndims != 2:
 327     raise ValueError('scores field must be of rank 2')
 328   if shape_utils.get_dim_as_int(scores.shape[1]) is None:
 329     raise ValueError('scores must have statically defined second ' 'dimension')
 330   if boxes.shape.ndims != 3:
 331     raise ValueError('boxes must be of rank 3.')
 332   if not (shape_utils.get_dim_as_int(
 333       boxes.shape[1]) == shape_utils.get_dim_as_int(scores.shape[1]) or
 334           shape_utils.get_dim_as_int(boxes.shape[1]) == 1):
 335     raise ValueError('second dimension of boxes must be either 1 or equal '
 336                      'to the second dimension of scores')
 337   if shape_utils.get_dim_as_int(boxes.shape[2]) != 4:
 338     raise ValueError('last dimension of boxes must be of size 4.')
 339   if change_coordinate_frame and clip_window is None:
 340     raise ValueError('if change_coordinate_frame is True, then a clip_window'
 341                      'must be specified.')

 344 def _clip_window_prune_boxes(sorted_boxes, clip_window, pad_to_max_output_size,
 345                              change_coordinate_frame):
 346   """Prune boxes with zero area.
 347 
 348   Args:
 349     sorted_boxes: A BoxList containing k detections.
 350     clip_window: A float32 tensor of the form [y_min, x_min, y_max, x_max]
 351       representing the window to clip and normalize boxes to before performing
 352       non-max suppression.
 353     pad_to_max_output_size: flag indicating whether to pad to max output size or
 354       not.
 355     change_coordinate_frame: Whether to normalize coordinates after clipping
 356       relative to clip_window (this can only be set to True if a clip_window is
 357       provided).
 358 
 359   Returns:
 360     sorted_boxes: A BoxList containing k detections after pruning.
 361     num_valid_nms_boxes_cumulative: Number of valid NMS boxes
 362   """
 363   sorted_boxes = box_list_ops.clip_to_window(
 364       sorted_boxes,
 365       clip_window,
 366       filter_nonoverlapping=not pad_to_max_output_size)
 367   # Set the scores of boxes with zero area to -1 to keep the default
 368   # behaviour of pruning out zero area boxes.
 369   sorted_boxes_size = tf.shape(sorted_boxes.get())[0]
 370   non_zero_box_area = tf.cast(box_list_ops.area(sorted_boxes), tf.bool)
 371   sorted_boxes_scores = tf.where(
 372       non_zero_box_area, sorted_boxes.get_field(fields.BoxListFields.scores),
 373       -1 * tf.ones(sorted_boxes_size))
 374   sorted_boxes.add_field(fields.BoxListFields.scores, sorted_boxes_scores)
 375   num_valid_nms_boxes_cumulative = tf.reduce_sum(
 376       tf.cast(tf.greater_equal(sorted_boxes_scores, 0), tf.int32))
 377   sorted_boxes = box_list_ops.sort_by_field(sorted_boxes,
 378                                             fields.BoxListFields.scores)
 379   if change_coordinate_frame:
 380     sorted_boxes = box_list_ops.change_coordinate_frame(sorted_boxes,
 381                                                         clip_window)
 382   return sorted_boxes, num_valid_nms_boxes_cumulative

 385 def multiclass_non_max_suppression(boxes,
 386                                    scores,
 387                                    score_thresh,
 388                                    iou_thresh,
 389                                    max_size_per_class,
 390                                    max_total_size=0,
 391                                    clip_window=None,
 392                                    change_coordinate_frame=False,
 393                                    masks=None,
 394                                    boundaries=None,
 395                                    pad_to_max_output_size=False,
 396                                    use_partitioned_nms=False,
 397                                    additional_fields=None,
 398                                    soft_nms_sigma=0.0,
 399                                    use_hard_nms=False,
 400                                    scope=None):
 401   """Multi-class version of non maximum suppression.
 402 
 403   This op greedily selects a subset of detection bounding boxes, pruning
 404   away boxes that have high IOU (intersection over union) overlap (> thresh)
 405   with already selected boxes.  It operates independently for each class for
 406   which scores are provided (via the scores field of the input box_list),
 407   pruning boxes with score less than a provided threshold prior to
 408   applying NMS.
 409 
 410   Please note that this operation is performed on *all* classes, therefore any
 411   background classes should be removed prior to calling this function.
 412 
 413   Selected boxes are guaranteed to be sorted in decreasing order by score (but
 414   the sort is not guaranteed to be stable).
 415 
 416   Args:
 417     boxes: A [k, q, 4] float32 tensor containing k detections. `q` can be either
 418       number of classes or 1 depending on whether a separate box is predicted
 419       per class.
 420     scores: A [k, num_classes] float32 tensor containing the scores for each of
 421       the k detections. The scores have to be non-negative when
 422       pad_to_max_output_size is True.
 423     score_thresh: scalar threshold for score (low scoring boxes are removed).
 424     iou_thresh: scalar threshold for IOU (new boxes that have high IOU overlap
 425       with previously selected boxes are removed).
 426     max_size_per_class: maximum number of retained boxes per class.
 427     max_total_size: maximum number of boxes retained over all classes. By
 428       default returns all boxes retained after capping boxes per class.
 429     clip_window: A float32 tensor of the form [y_min, x_min, y_max, x_max]
 430       representing the window to clip and normalize boxes to before performing
 431       non-max suppression.
 432     change_coordinate_frame: Whether to normalize coordinates after clipping
 433       relative to clip_window (this can only be set to True if a clip_window
 434       is provided)
 435     masks: (optional) a [k, q, mask_height, mask_width] float32 tensor
 436       containing box masks. `q` can be either number of classes or 1 depending
 437       on whether a separate mask is predicted per class.
 438     boundaries: (optional) a [k, q, boundary_height, boundary_width] float32
 439       tensor containing box boundaries. `q` can be either number of classes or 1
 440       depending on whether a separate boundary is predicted per class.
 441     pad_to_max_output_size: If true, the output nmsed boxes are padded to be of
 442       length `max_size_per_class`. Defaults to false.
 443     use_partitioned_nms: If true, use partitioned version of
 444       non_max_suppression.
 445     additional_fields: (optional) If not None, a dictionary that maps keys to
 446       tensors whose first dimensions are all of size `k`. After non-maximum
 447       suppression, all tensors corresponding to the selected boxes will be
 448       added to resulting BoxList.
 449     soft_nms_sigma: A scalar float representing the Soft NMS sigma parameter;
 450       See Bodla et al, https://arxiv.org/abs/1704.04503).  When
 451       `soft_nms_sigma=0.0` (which is default), we fall back to standard (hard)
 452       NMS.  Soft NMS is currently only supported when pad_to_max_output_size is
 453       False.
 454     use_hard_nms: Enforce the usage of hard NMS.
 455     scope: name scope.
 456 
 457   Returns:
 458     A tuple of sorted_boxes and num_valid_nms_boxes. The sorted_boxes is a
 459       BoxList holds M boxes with a rank-1 scores field representing
 460       corresponding scores for each box with scores sorted in decreasing order
 461       and a rank-1 classes field representing a class label for each box. The
 462       num_valid_nms_boxes is a 0-D integer tensor representing the number of
 463       valid elements in `BoxList`, with the valid elements appearing first.
 464 
 465   Raises:
 466     ValueError: if iou_thresh is not in [0, 1] or if input boxlist does not have
 467       a valid scores field.
 468     ValueError: if Soft NMS (tf.image.non_max_suppression_with_scores) is not
 469       supported in the current TF version and `soft_nms_sigma` is nonzero.
 470   """
 471   _validate_boxes_scores_iou_thresh(boxes, scores, iou_thresh,
 472                                     change_coordinate_frame, clip_window)
 473   if pad_to_max_output_size and soft_nms_sigma != 0.0:
 474     raise ValueError('Soft NMS (soft_nms_sigma != 0.0) is currently not '
 475                      'supported when pad_to_max_output_size is True.')
 476 
 477   with tf.name_scope(scope, 'MultiClassNonMaxSuppression'):
 478     num_scores = tf.shape(scores)[0]
 479     num_classes = shape_utils.get_dim_as_int(scores.get_shape()[1])
 480 
 481     selected_boxes_list = []
 482     num_valid_nms_boxes_cumulative = tf.constant(0)
 483     per_class_boxes_list = tf.unstack(boxes, axis=1)
 484     if masks is not None:
 485       per_class_masks_list = tf.unstack(masks, axis=1)
 486     if boundaries is not None:
 487       per_class_boundaries_list = tf.unstack(boundaries, axis=1)
 488     boxes_ids = (range(num_classes) if len(per_class_boxes_list) > 1
 489                  else [0] * num_classes)
 490     for class_idx, boxes_idx in zip(range(num_classes), boxes_ids):
 491       per_class_boxes = per_class_boxes_list[boxes_idx]
 492       boxlist_and_class_scores = box_list.BoxList(per_class_boxes)
 493       class_scores = tf.reshape(
 494           tf.slice(scores, [0, class_idx], tf.stack([num_scores, 1])), [-1])
 495 
 496       boxlist_and_class_scores.add_field(fields.BoxListFields.scores,
 497                                          class_scores)
 498       if masks is not None:
 499         per_class_masks = per_class_masks_list[boxes_idx]
 500         boxlist_and_class_scores.add_field(fields.BoxListFields.masks,
 501                                            per_class_masks)
 502       if boundaries is not None:
 503         per_class_boundaries = per_class_boundaries_list[boxes_idx]
 504         boxlist_and_class_scores.add_field(fields.BoxListFields.boundaries,
 505                                            per_class_boundaries)
 506       if additional_fields is not None:
 507         for key, tensor in additional_fields.items():
 508           boxlist_and_class_scores.add_field(key, tensor)
 509 
 510       nms_result = None
 511       selected_scores = None
 512       if pad_to_max_output_size:
 513         max_selection_size = max_size_per_class
 514         if use_partitioned_nms:
 515           (selected_indices, num_valid_nms_boxes,
 516            boxlist_and_class_scores.data['boxes'],
 517            boxlist_and_class_scores.data['scores'],
 518            _) = partitioned_non_max_suppression_padded(
 519                boxlist_and_class_scores.get(),
 520                boxlist_and_class_scores.get_field(fields.BoxListFields.scores),
 521                max_selection_size,
 522                iou_threshold=iou_thresh,
 523                score_threshold=score_thresh)
 524         else:
 525           selected_indices, num_valid_nms_boxes = (
 526               tf.image.non_max_suppression_padded(
 527                   boxlist_and_class_scores.get(),
 528                   boxlist_and_class_scores.get_field(
 529                       fields.BoxListFields.scores),
 530                   max_selection_size,
 531                   iou_threshold=iou_thresh,
 532                   score_threshold=score_thresh,
 533                   pad_to_max_output_size=True))
 534         nms_result = box_list_ops.gather(boxlist_and_class_scores,
 535                                          selected_indices)
 536         selected_scores = nms_result.get_field(fields.BoxListFields.scores)
 537       else:
 538         max_selection_size = tf.minimum(max_size_per_class,
 539                                         boxlist_and_class_scores.num_boxes())
 540         if (hasattr(tf.image, 'non_max_suppression_with_scores') and
 541             tf.compat.forward_compatible(2019, 6, 6) and not use_hard_nms):
 542           (selected_indices, selected_scores
 543           ) = tf.image.non_max_suppression_with_scores(
 544               boxlist_and_class_scores.get(),
 545               boxlist_and_class_scores.get_field(fields.BoxListFields.scores),
 546               max_selection_size,
 547               iou_threshold=iou_thresh,
 548               score_threshold=score_thresh,
 549               soft_nms_sigma=soft_nms_sigma)
 550           num_valid_nms_boxes = tf.shape(selected_indices)[0]
 551           selected_indices = tf.concat(
 552               [selected_indices,
 553                tf.zeros(max_selection_size-num_valid_nms_boxes, tf.int32)], 0)
 554           selected_scores = tf.concat(
 555               [selected_scores,
 556                tf.zeros(max_selection_size-num_valid_nms_boxes,
 557                         tf.float32)], -1)
 558           nms_result = box_list_ops.gather(boxlist_and_class_scores,
 559                                            selected_indices)
 560         else:
 561           if soft_nms_sigma != 0:
 562             raise ValueError('Soft NMS not supported in current TF version!')
 563           selected_indices = tf.image.non_max_suppression(
 564               boxlist_and_class_scores.get(),
 565               boxlist_and_class_scores.get_field(fields.BoxListFields.scores),
 566               max_selection_size,
 567               iou_threshold=iou_thresh,
 568               score_threshold=score_thresh)
 569           num_valid_nms_boxes = tf.shape(selected_indices)[0]
 570           selected_indices = tf.concat(
 571               [selected_indices,
 572                tf.zeros(max_selection_size-num_valid_nms_boxes, tf.int32)], 0)
 573           nms_result = box_list_ops.gather(boxlist_and_class_scores,
 574                                            selected_indices)
 575           selected_scores = nms_result.get_field(fields.BoxListFields.scores)
 576       # Make the scores -1 for invalid boxes.
 577       valid_nms_boxes_indices = tf.less(
 578           tf.range(max_selection_size), num_valid_nms_boxes)
 579 
 580       nms_result.add_field(
 581           fields.BoxListFields.scores,
 582           tf.where(valid_nms_boxes_indices,
 583                    selected_scores, -1*tf.ones(max_selection_size)))
 584       num_valid_nms_boxes_cumulative += num_valid_nms_boxes
 585 
 586       nms_result.add_field(
 587           fields.BoxListFields.classes, (tf.zeros_like(
 588               nms_result.get_field(fields.BoxListFields.scores)) + class_idx))
 589       selected_boxes_list.append(nms_result)
 590     selected_boxes = box_list_ops.concatenate(selected_boxes_list)
 591     sorted_boxes = box_list_ops.sort_by_field(selected_boxes,
 592                                               fields.BoxListFields.scores)
 593     if clip_window is not None:
 594       # When pad_to_max_output_size is False, it prunes the boxes with zero
 595       # area.
 596       sorted_boxes, num_valid_nms_boxes_cumulative = _clip_window_prune_boxes(
 597           sorted_boxes, clip_window, pad_to_max_output_size,
 598           change_coordinate_frame)
 599 
 600     if max_total_size:
 601       max_total_size = tf.minimum(max_total_size, sorted_boxes.num_boxes())
 602       sorted_boxes = box_list_ops.gather(sorted_boxes, tf.range(max_total_size))
 603       num_valid_nms_boxes_cumulative = tf.where(
 604           max_total_size > num_valid_nms_boxes_cumulative,
 605           num_valid_nms_boxes_cumulative, max_total_size)
 606     # Select only the valid boxes if pad_to_max_output_size is False.
 607     if not pad_to_max_output_size:
 608       sorted_boxes = box_list_ops.gather(
 609           sorted_boxes, tf.range(num_valid_nms_boxes_cumulative))
 610 
 611     return sorted_boxes, num_valid_nms_boxes_cumulative

 614 def class_agnostic_non_max_suppression(boxes,
 615                                        scores,
 616                                        score_thresh,
 617                                        iou_thresh,
 618                                        max_classes_per_detection=1,
 619                                        max_total_size=0,
 620                                        clip_window=None,
 621                                        change_coordinate_frame=False,
 622                                        masks=None,
 623                                        boundaries=None,
 624                                        pad_to_max_output_size=False,
 625                                        use_partitioned_nms=False,
 626                                        additional_fields=None,
 627                                        soft_nms_sigma=0.0,
 628                                        scope=None):
 629   """Class-agnostic version of non maximum suppression.
 630 
 631   This op greedily selects a subset of detection bounding boxes, pruning
 632   away boxes that have high IOU (intersection over union) overlap (> thresh)
 633   with already selected boxes.  It operates on all the boxes using
 634   max scores across all classes for which scores are provided (via the scores
 635   field of the input box_list), pruning boxes with score less than a provided
 636   threshold prior to applying NMS.
 637 
 638   Please note that this operation is performed in a class-agnostic way,
 639   therefore any background classes should be removed prior to calling this
 640   function.
 641 
 642   Selected boxes are guaranteed to be sorted in decreasing order by score (but
 643   the sort is not guaranteed to be stable).
 644 
 645   Args:
 646     boxes: A [k, q, 4] float32 tensor containing k detections. `q` can be either
 647       number of classes or 1 depending on whether a separate box is predicted
 648       per class.
 649     scores: A [k, num_classes] float32 tensor containing the scores for each of
 650       the k detections. The scores have to be non-negative when
 651       pad_to_max_output_size is True.
 652     score_thresh: scalar threshold for score (low scoring boxes are removed).
 653     iou_thresh: scalar threshold for IOU (new boxes that have high IOU overlap
 654       with previously selected boxes are removed).
 655     max_classes_per_detection: maximum number of retained classes per detection
 656       box in class-agnostic NMS.
 657     max_total_size: maximum number of boxes retained over all classes. By
 658       default returns all boxes retained after capping boxes per class.
 659     clip_window: A float32 tensor of the form [y_min, x_min, y_max, x_max]
 660       representing the window to clip and normalize boxes to before performing
 661       non-max suppression.
 662     change_coordinate_frame: Whether to normalize coordinates after clipping
 663       relative to clip_window (this can only be set to True if a clip_window is
 664       provided)
 665     masks: (optional) a [k, q, mask_height, mask_width] float32 tensor
 666       containing box masks. `q` can be either number of classes or 1 depending
 667       on whether a separate mask is predicted per class.
 668     boundaries: (optional) a [k, q, boundary_height, boundary_width] float32
 669       tensor containing box boundaries. `q` can be either number of classes or 1
 670       depending on whether a separate boundary is predicted per class.
 671     pad_to_max_output_size: If true, the output nmsed boxes are padded to be of
 672       length `max_size_per_class`. Defaults to false.
 673     use_partitioned_nms: If true, use partitioned version of
 674       non_max_suppression.
 675     additional_fields: (optional) If not None, a dictionary that maps keys to
 676       tensors whose first dimensions are all of size `k`. After non-maximum
 677       suppression, all tensors corresponding to the selected boxes will be added
 678       to resulting BoxList.
 679     soft_nms_sigma: A scalar float representing the Soft NMS sigma parameter;
 680       See Bodla et al, https://arxiv.org/abs/1704.04503).  When
 681       `soft_nms_sigma=0.0` (which is default), we fall back to standard (hard)
 682       NMS.  Soft NMS is currently only supported when pad_to_max_output_size is
 683       False.
 684     scope: name scope.
 685 
 686   Returns:
 687     A tuple of sorted_boxes and num_valid_nms_boxes. The sorted_boxes is a
 688       BoxList holds M boxes with a rank-1 scores field representing
 689       corresponding scores for each box with scores sorted in decreasing order
 690       and a rank-1 classes field representing a class label for each box. The
 691       num_valid_nms_boxes is a 0-D integer tensor representing the number of
 692       valid elements in `BoxList`, with the valid elements appearing first.
 693 
 694   Raises:
 695     ValueError: if iou_thresh is not in [0, 1] or if input boxlist does not have
 696       a valid scores field or if non-zero soft_nms_sigma is provided when
 697       pad_to_max_output_size is True.
 698   """
 699   _validate_boxes_scores_iou_thresh(boxes, scores, iou_thresh,
 700                                     change_coordinate_frame, clip_window)
 701   if pad_to_max_output_size and soft_nms_sigma != 0.0:
 702     raise ValueError('Soft NMS (soft_nms_sigma != 0.0) is currently not '
 703                      'supported when pad_to_max_output_size is True.')
 704 
 705   if max_classes_per_detection > 1:
 706     raise ValueError('Max classes per detection box >1 not supported.')
 707   q = shape_utils.get_dim_as_int(boxes.shape[1])
 708   if q > 1:
 709     class_ids = tf.expand_dims(
 710         tf.argmax(scores, axis=1, output_type=tf.int32), axis=1)
 711     boxes = tf.batch_gather(boxes, class_ids)
 712     if masks is not None:
 713       masks = tf.batch_gather(masks, class_ids)
 714     if boundaries is not None:
 715       boundaries = tf.batch_gather(boundaries, class_ids)
 716   boxes = tf.squeeze(boxes, axis=[1])
 717   if masks is not None:
 718     masks = tf.squeeze(masks, axis=[1])
 719   if boundaries is not None:
 720     boundaries = tf.squeeze(boundaries, axis=[1])
 721 
 722   with tf.name_scope(scope, 'ClassAgnosticNonMaxSuppression'):
 723     boxlist_and_class_scores = box_list.BoxList(boxes)
 724     max_scores = tf.reduce_max(scores, axis=-1)
 725     classes_with_max_scores = tf.argmax(scores, axis=-1)
 726     boxlist_and_class_scores.add_field(fields.BoxListFields.scores, max_scores)
 727     if masks is not None:
 728       boxlist_and_class_scores.add_field(fields.BoxListFields.masks, masks)
 729     if boundaries is not None:
 730       boxlist_and_class_scores.add_field(fields.BoxListFields.boundaries,
 731                                          boundaries)
 732 
 733     if additional_fields is not None:
 734       for key, tensor in additional_fields.items():
 735         boxlist_and_class_scores.add_field(key, tensor)
 736 
 737     nms_result = None
 738     selected_scores = None
 739     if pad_to_max_output_size:
 740       max_selection_size = max_total_size
 741       if use_partitioned_nms:
 742         (selected_indices, num_valid_nms_boxes,
 743          boxlist_and_class_scores.data['boxes'],
 744          boxlist_and_class_scores.data['scores'],
 745          argsort_ids) = partitioned_non_max_suppression_padded(
 746              boxlist_and_class_scores.get(),
 747              boxlist_and_class_scores.get_field(fields.BoxListFields.scores),
 748              max_selection_size,
 749              iou_threshold=iou_thresh,
 750              score_threshold=score_thresh)
 751         classes_with_max_scores = tf.gather(classes_with_max_scores,
 752                                             argsort_ids)
 753       else:
 754         selected_indices, num_valid_nms_boxes = (
 755             tf.image.non_max_suppression_padded(
 756                 boxlist_and_class_scores.get(),
 757                 boxlist_and_class_scores.get_field(fields.BoxListFields.scores),
 758                 max_selection_size,
 759                 iou_threshold=iou_thresh,
 760                 score_threshold=score_thresh,
 761                 pad_to_max_output_size=True))
 762       nms_result = box_list_ops.gather(boxlist_and_class_scores,
 763                                        selected_indices)
 764       selected_scores = nms_result.get_field(fields.BoxListFields.scores)
 765     else:
 766       max_selection_size = tf.minimum(max_total_size,
 767                                       boxlist_and_class_scores.num_boxes())
 768       if (hasattr(tf.image, 'non_max_suppression_with_scores') and
 769           tf.compat.forward_compatible(2019, 6, 6)):
 770         (selected_indices, selected_scores
 771         ) = tf.image.non_max_suppression_with_scores(
 772             boxlist_and_class_scores.get(),
 773             boxlist_and_class_scores.get_field(fields.BoxListFields.scores),
 774             max_selection_size,
 775             iou_threshold=iou_thresh,
 776             score_threshold=score_thresh,
 777             soft_nms_sigma=soft_nms_sigma)
 778         num_valid_nms_boxes = tf.shape(selected_indices)[0]
 779         selected_indices = tf.concat([
 780             selected_indices,
 781             tf.zeros(max_selection_size - num_valid_nms_boxes, tf.int32)
 782         ], 0)
 783         selected_scores = tf.concat(
 784             [selected_scores,
 785              tf.zeros(max_selection_size-num_valid_nms_boxes, tf.float32)], -1)
 786         nms_result = box_list_ops.gather(boxlist_and_class_scores,
 787                                          selected_indices)
 788       else:
 789         if soft_nms_sigma != 0:
 790           raise ValueError('Soft NMS not supported in current TF version!')
 791         selected_indices = tf.image.non_max_suppression(
 792             boxlist_and_class_scores.get(),
 793             boxlist_and_class_scores.get_field(fields.BoxListFields.scores),
 794             max_selection_size,
 795             iou_threshold=iou_thresh,
 796             score_threshold=score_thresh)
 797         num_valid_nms_boxes = tf.shape(selected_indices)[0]
 798         selected_indices = tf.concat(
 799             [selected_indices,
 800              tf.zeros(max_selection_size-num_valid_nms_boxes, tf.int32)], 0)
 801         nms_result = box_list_ops.gather(boxlist_and_class_scores,
 802                                          selected_indices)
 803         selected_scores = nms_result.get_field(fields.BoxListFields.scores)
 804     valid_nms_boxes_indices = tf.less(
 805         tf.range(max_selection_size), num_valid_nms_boxes)
 806     nms_result.add_field(
 807         fields.BoxListFields.scores,
 808         tf.where(valid_nms_boxes_indices,
 809                  selected_scores, -1*tf.ones(max_selection_size)))
 810 
 811     selected_classes = tf.gather(classes_with_max_scores, selected_indices)
 812     selected_classes = tf.cast(selected_classes, tf.float32)
 813     nms_result.add_field(fields.BoxListFields.classes, selected_classes)
 814     selected_boxes = nms_result
 815     sorted_boxes = box_list_ops.sort_by_field(selected_boxes,
 816                                               fields.BoxListFields.scores)
 817 
 818     if clip_window is not None:
 819       # When pad_to_max_output_size is False, it prunes the boxes with zero
 820       # area.
 821       sorted_boxes, num_valid_nms_boxes = _clip_window_prune_boxes(
 822           sorted_boxes, clip_window, pad_to_max_output_size,
 823           change_coordinate_frame)
 824 
 825     if max_total_size:
 826       max_total_size = tf.minimum(max_total_size, sorted_boxes.num_boxes())
 827       sorted_boxes = box_list_ops.gather(sorted_boxes, tf.range(max_total_size))
 828       num_valid_nms_boxes = tf.where(max_total_size > num_valid_nms_boxes,
 829                                      num_valid_nms_boxes, max_total_size)
 830     # Select only the valid boxes if pad_to_max_output_size is False.
 831     if not pad_to_max_output_size:
 832       sorted_boxes = box_list_ops.gather(sorted_boxes,
 833                                          tf.range(num_valid_nms_boxes))
 834 
 835     return sorted_boxes, num_valid_nms_boxes

 838 def batch_multiclass_non_max_suppression(boxes,
 839                                          scores,
 840                                          score_thresh,
 841                                          iou_thresh,
 842                                          max_size_per_class,
 843                                          max_total_size=0,
 844                                          clip_window=None,
 845                                          change_coordinate_frame=False,
 846                                          num_valid_boxes=None,
 847                                          masks=None,
 848                                          additional_fields=None,
 849                                          soft_nms_sigma=0.0,
 850                                          scope=None,
 851                                          use_static_shapes=False,
 852                                          use_partitioned_nms=False,
 853                                          parallel_iterations=32,
 854                                          use_class_agnostic_nms=False,
 855                                          max_classes_per_detection=1,
 856                                          use_dynamic_map_fn=False,
 857                                          use_combined_nms=False,
 858                                          use_hard_nms=False):
 859   """Multi-class version of non maximum suppression that operates on a batch.
 860 
 861   This op is similar to `multiclass_non_max_suppression` but operates on a batch
 862   of boxes and scores. See documentation for `multiclass_non_max_suppression`
 863   for details.
 864 
 865   Args:
 866     boxes: A [batch_size, num_anchors, q, 4] float32 tensor containing
 867       detections. If `q` is 1 then same boxes are used for all classes
 868       otherwise, if `q` is equal to number of classes, class-specific boxes are
 869       used.
 870     scores: A [batch_size, num_anchors, num_classes] float32 tensor containing
 871       the scores for each of the `num_anchors` detections. The scores have to be
 872       non-negative when use_static_shapes is set True.
 873     score_thresh: scalar threshold for score (low scoring boxes are removed).
 874     iou_thresh: scalar threshold for IOU (new boxes that have high IOU overlap
 875       with previously selected boxes are removed).
 876     max_size_per_class: maximum number of retained boxes per class.
 877     max_total_size: maximum number of boxes retained over all classes. By
 878       default returns all boxes retained after capping boxes per class.
 879     clip_window: A float32 tensor of shape [batch_size, 4]  where each entry is
 880       of the form [y_min, x_min, y_max, x_max] representing the window to clip
 881       boxes to before performing non-max suppression. This argument can also be
 882       a tensor of shape [4] in which case, the same clip window is applied to
 883       all images in the batch. If clip_widow is None, all boxes are used to
 884       perform non-max suppression.
 885     change_coordinate_frame: Whether to normalize coordinates after clipping
 886       relative to clip_window (this can only be set to True if a clip_window is
 887       provided)
 888     num_valid_boxes: (optional) a Tensor of type `int32`. A 1-D tensor of shape
 889       [batch_size] representing the number of valid boxes to be considered for
 890       each image in the batch.  This parameter allows for ignoring zero
 891       paddings.
 892     masks: (optional) a [batch_size, num_anchors, q, mask_height, mask_width]
 893       float32 tensor containing box masks. `q` can be either number of classes
 894       or 1 depending on whether a separate mask is predicted per class.
 895     additional_fields: (optional) If not None, a dictionary that maps keys to
 896       tensors whose dimensions are [batch_size, num_anchors, ...].
 897     soft_nms_sigma: A scalar float representing the Soft NMS sigma parameter;
 898       See Bodla et al, https://arxiv.org/abs/1704.04503).  When
 899       `soft_nms_sigma=0.0` (which is default), we fall back to standard (hard)
 900       NMS.  Soft NMS is currently only supported when pad_to_max_output_size is
 901       False.
 902     scope: tf scope name.
 903     use_static_shapes: If true, the output nmsed boxes are padded to be of
 904       length `max_size_per_class` and it doesn't clip boxes to max_total_size.
 905       Defaults to false.
 906     use_partitioned_nms: If true, use partitioned version of
 907       non_max_suppression.
 908     parallel_iterations: (optional) number of batch items to process in
 909       parallel.
 910     use_class_agnostic_nms: If true, this uses class-agnostic non max
 911       suppression
 912     max_classes_per_detection: Maximum number of retained classes per detection
 913       box in class-agnostic NMS.
 914     use_dynamic_map_fn: If true, images in the batch will be processed within a
 915       dynamic loop. Otherwise, a static loop will be used if possible.
 916     use_combined_nms: If true, it uses tf.image.combined_non_max_suppression (
 917       multi-class version of NMS that operates on a batch).
 918       It greedily selects a subset of detection bounding boxes, pruning away
 919       boxes that have high IOU (intersection over union) overlap (> thresh) with
 920       already selected boxes. It operates independently for each batch.
 921       Within each batch, it operates independently for each class for which
 922       scores are provided (via the scores field of the input box_list),
 923       pruning boxes with score less than a provided threshold prior to applying
 924       NMS. This operation is performed on *all* batches and *all* classes
 925       in the batch, therefore any background classes should be removed prior to
 926       calling this function.
 927       Masks and additional fields are not supported.
 928       See argument checks in the code below for unsupported arguments.
 929     use_hard_nms: Enforce the usage of hard NMS.
 930 
 931   Returns:
 932     'nmsed_boxes': A [batch_size, max_detections, 4] float32 tensor
 933       containing the non-max suppressed boxes.
 934     'nmsed_scores': A [batch_size, max_detections] float32 tensor containing
 935       the scores for the boxes.
 936     'nmsed_classes': A [batch_size, max_detections] float32 tensor
 937       containing the class for boxes.
 938     'nmsed_masks': (optional) a
 939       [batch_size, max_detections, mask_height, mask_width] float32 tensor
 940       containing masks for each selected box. This is set to None if input
 941       `masks` is None.
 942     'nmsed_additional_fields': (optional) a dictionary of
 943       [batch_size, max_detections, ...] float32 tensors corresponding to the
 944       tensors specified in the input `additional_fields`. This is not returned
 945       if input `additional_fields` is None.
 946     'num_detections': A [batch_size] int32 tensor indicating the number of
 947       valid detections per batch item. Only the top num_detections[i] entries in
 948       nms_boxes[i], nms_scores[i] and nms_class[i] are valid. The rest of the
 949       entries are zero paddings.
 950 
 951   Raises:
 952     ValueError: if `q` in boxes.shape is not 1 or not equal to number of
 953       classes as inferred from scores.shape.
 954   """
 955   if use_combined_nms:
 956     if change_coordinate_frame:
 957       raise ValueError(
 958           'change_coordinate_frame (normalizing coordinates'
 959           ' relative to clip_window) is not supported by combined_nms.')
 960     if num_valid_boxes is not None:
 961       raise ValueError('num_valid_boxes is not supported by combined_nms.')
 962     if masks is not None:
 963       raise ValueError('masks is not supported by combined_nms.')
 964     if soft_nms_sigma != 0.0:
 965       raise ValueError('Soft NMS is not supported by combined_nms.')
 966     if use_class_agnostic_nms:
 967       raise ValueError('class-agnostic NMS is not supported by combined_nms.')
 968     if clip_window is not None:
 969       tf.logging.warning(
 970           'clip_window is not supported by combined_nms unless it is'
 971           ' [0. 0. 1. 1.] for each image.')
 972     if additional_fields is not None:

1068             tensors each with size [num_anchors, ...].
1069           per_image_num_valid_boxes - A tensor of type `int32`. A 1-D tensor of
