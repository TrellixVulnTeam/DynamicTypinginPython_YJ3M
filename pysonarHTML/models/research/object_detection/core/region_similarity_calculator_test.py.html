  16 """Tests for region_similarity_calculator."""
  17 import tensorflow.compat.v1 as tf
  18 
  19 from object_detection.core import box_list
  20 from object_detection.core import region_similarity_calculator
  21 from object_detection.core import standard_fields as fields
  22 from object_detection.utils import test_case
  23 
  24 
  25 class RegionSimilarityCalculatorTest(test_case.TestCase):

  27   def test_get_correct_pairwise_similarity_based_on_iou(self):
  28     def graph_fn():
  29       corners1 = tf.constant([[4.0, 3.0, 7.0, 5.0], [5.0, 6.0, 10.0, 7.0]])
  30       corners2 = tf.constant([[3.0, 4.0, 6.0, 8.0], [14.0, 14.0, 15.0, 15.0],
  31                               [0.0, 0.0, 20.0, 20.0]])
  32       boxes1 = box_list.BoxList(corners1)
  33       boxes2 = box_list.BoxList(corners2)
  34       iou_similarity_calculator = region_similarity_calculator.IouSimilarity()
  35       iou_similarity = iou_similarity_calculator.compare(boxes1, boxes2)
  36       return iou_similarity
  37     exp_output = [[2.0 / 16.0, 0, 6.0 / 400.0], [1.0 / 16.0, 0.0, 5.0 / 400.0]]
  38     iou_output = self.execute(graph_fn, [])
  39     self.assertAllClose(iou_output, exp_output)

  41   def test_get_correct_pairwise_similarity_based_on_squared_distances(self):
  42     def graph_fn():
  43       corners1 = tf.constant([[0.0, 0.0, 0.0, 0.0],
  44                               [1.0, 1.0, 0.0, 2.0]])
  45       corners2 = tf.constant([[3.0, 4.0, 1.0, 0.0],
  46                               [-4.0, 0.0, 0.0, 3.0],
  47                               [0.0, 0.0, 0.0, 0.0]])
  48       boxes1 = box_list.BoxList(corners1)
  49       boxes2 = box_list.BoxList(corners2)
  50       dist_similarity_calc = region_similarity_calculator.NegSqDistSimilarity()
  51       dist_similarity = dist_similarity_calc.compare(boxes1, boxes2)
  52       return dist_similarity
  53     exp_output = [[-26, -25, 0], [-18, -27, -6]]
  54     iou_output = self.execute(graph_fn, [])
  55     self.assertAllClose(iou_output, exp_output)

  57   def test_get_correct_pairwise_similarity_based_on_ioa(self):
  58     def graph_fn():
  59       corners1 = tf.constant([[4.0, 3.0, 7.0, 5.0], [5.0, 6.0, 10.0, 7.0]])
  60       corners2 = tf.constant([[3.0, 4.0, 6.0, 8.0], [14.0, 14.0, 15.0, 15.0],
  61                               [0.0, 0.0, 20.0, 20.0]])
  62       boxes1 = box_list.BoxList(corners1)
  63       boxes2 = box_list.BoxList(corners2)
  64       ioa_similarity_calculator = region_similarity_calculator.IoaSimilarity()
  65       ioa_similarity_1 = ioa_similarity_calculator.compare(boxes1, boxes2)
  66       ioa_similarity_2 = ioa_similarity_calculator.compare(boxes2, boxes1)
  67       return ioa_similarity_1, ioa_similarity_2
  68     exp_output_1 = [[2.0 / 12.0, 0, 6.0 / 400.0],
  69                     [1.0 / 12.0, 0.0, 5.0 / 400.0]]
  70     exp_output_2 = [[2.0 / 6.0, 1.0 / 5.0],
  71                     [0, 0],
  72                     [6.0 / 6.0, 5.0 / 5.0]]
  73     iou_output_1, iou_output_2 = self.execute(graph_fn, [])
  74     self.assertAllClose(iou_output_1, exp_output_1)
  75     self.assertAllClose(iou_output_2, exp_output_2)

  77   def test_get_correct_pairwise_similarity_based_on_thresholded_iou(self):
  78     def graph_fn():
  79       corners1 = tf.constant([[4.0, 3.0, 7.0, 5.0], [5.0, 6.0, 10.0, 7.0]])
  80       corners2 = tf.constant([[3.0, 4.0, 6.0, 8.0], [14.0, 14.0, 15.0, 15.0],
  81                               [0.0, 0.0, 20.0, 20.0]])
  82       scores = tf.constant([.3, .6])
  83       iou_threshold = .013
  84       boxes1 = box_list.BoxList(corners1)
  85       boxes1.add_field(fields.BoxListFields.scores, scores)
  86       boxes2 = box_list.BoxList(corners2)
  87       iou_similarity_calculator = (
  88           region_similarity_calculator.ThresholdedIouSimilarity(
  89               iou_threshold=iou_threshold))
  90       iou_similarity = iou_similarity_calculator.compare(boxes1, boxes2)
  91       return iou_similarity
  92     exp_output = tf.constant([[0.3, 0., 0.3], [0.6, 0., 0.]])
  93     iou_output = self.execute(graph_fn, [])
  94     self.assertAllClose(iou_output, exp_output)
