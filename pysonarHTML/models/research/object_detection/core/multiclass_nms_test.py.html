  26   def test_multiclass_nms_select_with_shared_boxes_cpu_only(self):
  27     boxes = np.array(
  28         [[[0, 0, 1, 1]], [[0, 0.1, 1, 1.1]], [[0, -0.1, 1, 0.9]],
  29          [[0, 10, 1, 11]], [[0, 10.1, 1, 11.1]], [[0, 100, 1, 101]],
  30          [[0, 1000, 1, 1002]], [[0, 1000, 1, 1002.1]]], np.float32)
  31     scores = np.array([[.9, 0.01], [.75, 0.05], [.6, 0.01], [.95, 0],
  32                        [.5, 0.01], [.3, 0.01], [.01, .85], [.01, .5]],
  33                       np.float32)
  34 
  35     def graph_fn(boxes, scores):
  36       score_thresh = 0.1
  37       iou_thresh = .5
  38       max_output_size = 4
  39       nms, _ = post_processing.multiclass_non_max_suppression(
  40           boxes, scores, score_thresh, iou_thresh, max_output_size)
  41       return (nms.get(), nms.get_field(fields.BoxListFields.scores),
  42               nms.get_field(fields.BoxListFields.classes))
  43 
  44     exp_nms_corners = [[0, 10, 1, 11],
  45                        [0, 0, 1, 1],
  46                        [0, 1000, 1, 1002],
  47                        [0, 100, 1, 101]]
  48     exp_nms_scores = [.95, .9, .85, .3]
  49     exp_nms_classes = [0, 0, 1, 0]
  50     (nms_corners_output, nms_scores_output,
  51      nms_classes_output) = self.execute_cpu(graph_fn, [boxes, scores])
  52     self.assertAllClose(nms_corners_output, exp_nms_corners)
  53     self.assertAllClose(nms_scores_output, exp_nms_scores)
  54     self.assertAllClose(nms_classes_output, exp_nms_classes)
