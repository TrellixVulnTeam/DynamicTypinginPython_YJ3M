  16 """Base target assigner module.
  17 
  18 The job of a TargetAssigner is, for a given set of anchors (bounding boxes) and
  19 groundtruth detections (bounding boxes), to assign classification and regression
  20 targets to each anchor as well as weights to each anchor (specifying, e.g.,
  21 which anchors should not contribute to training loss).
  22 
  23 It assigns classification/regression targets by performing the following steps:
  24 1) Computing pairwise similarity between anchors and groundtruth boxes using a
  25   provided RegionSimilarity Calculator
  26 2) Computing a matching based on the similarity matrix using a provided Matcher
  27 3) Assigning regression targets based on the matching and a provided BoxCoder
  28 4) Assigning classification targets based on the matching and groundtruth labels
  29 
  30 Note that TargetAssigners only operate on detections from a single
  31 image at a time, so any logic for applying a TargetAssigner to multiple
  32 images must be handled externally.
  33 """

  63 class TargetAssigner(object):
  64   """Target assigner to compute classification and regression targets."""

  66   def __init__(self,
  67                similarity_calc,
  68                matcher,
  69                box_coder_instance,
  70                negative_class_weight=1.0):
  71     """Construct Object Detection Target Assigner.
  72 
  73     Args:
  74       similarity_calc: a RegionSimilarityCalculator
  75       matcher: an object_detection.core.Matcher used to match groundtruth to
  76         anchors.
  77       box_coder_instance: an object_detection.core.BoxCoder used to encode
  78         matching groundtruth boxes with respect to anchors.
  79       negative_class_weight: classification weight to be associated to negative
  80         anchors (default: 1.0). The weight must be in [0., 1.].
  81 
  82     Raises:
  83       ValueError: if similarity_calc is not a RegionSimilarityCalculator or
  84         if matcher is not a Matcher or if box_coder is not a BoxCoder
  85     """
  86     if not isinstance(similarity_calc, sim_calc.RegionSimilarityCalculator):
  87       raise ValueError('similarity_calc must be a RegionSimilarityCalculator')
  88     if not isinstance(matcher, mat.Matcher):
  89       raise ValueError('matcher must be a Matcher')
  90     if not isinstance(box_coder_instance, box_coder.BoxCoder):
  91       raise ValueError('box_coder must be a BoxCoder')
  92     self._similarity_calc = similarity_calc
  93     self._matcher = matcher
  94     self._box_coder = box_coder_instance
  95     self._negative_class_weight = negative_class_weight
  96 
  97   @property
  98   def box_coder(self):
  99     return self._box_coder

 101   # TODO(rathodv): move labels, scores, and weights to groundtruth_boxes fields.
 102   def assign(self,
 103              anchors,
 104              groundtruth_boxes,
 105              groundtruth_labels=None,
 106              unmatched_class_label=None,
 107              groundtruth_weights=None):
 108     """Assign classification and regression targets to each anchor.
 109 
 110     For a given set of anchors and groundtruth detections, match anchors
 111     to groundtruth_boxes and assign classification and regression targets to
 112     each anchor as well as weights based on the resulting match (specifying,
 113     e.g., which anchors should not contribute to training loss).
 114 
 115     Anchors that are not matched to anything are given a classification target
 116     of self._unmatched_cls_target which can be specified via the constructor.
 117 
 118     Args:
 119       anchors: a BoxList representing N anchors
 120       groundtruth_boxes: a BoxList representing M groundtruth boxes
 121       groundtruth_labels:  a tensor of shape [M, d_1, ... d_k]
 122         with labels for each of the ground_truth boxes. The subshape
 123         [d_1, ... d_k] can be empty (corresponding to scalar inputs).  When set
 124         to None, groundtruth_labels assumes a binary problem where all
 125         ground_truth boxes get a positive label (of 1).
 126       unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
 127         which is consistent with the classification target for each
 128         anchor (and can be empty for scalar targets).  This shape must thus be
 129         compatible with the groundtruth labels that are passed to the "assign"
 130         function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
 131         If set to None, unmatched_cls_target is set to be [0] for each anchor.
 132       groundtruth_weights: a float tensor of shape [M] indicating the weight to
 133         assign to all anchors match to a particular groundtruth box. The weights
 134         must be in [0., 1.]. If None, all weights are set to 1. Generally no
 135         groundtruth boxes with zero weight match to any anchors as matchers are
 136         aware of groundtruth weights. Additionally, `cls_weights` and
 137         `reg_weights` are calculated using groundtruth weights as an added
 138         safety.
 139 
 140     Returns:
 141       cls_targets: a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k],
 142         where the subshape [d_1, ..., d_k] is compatible with groundtruth_labels
 143         which has shape [num_gt_boxes, d_1, d_2, ... d_k].
 144       cls_weights: a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k],
 145         representing weights for each element in cls_targets.
 146       reg_targets: a float32 tensor with shape [num_anchors, box_code_dimension]
 147       reg_weights: a float32 tensor with shape [num_anchors]
 148       match: an int32 tensor of shape [num_anchors] containing result of anchor
 149         groundtruth matching. Each position in the tensor indicates an anchor
 150         and holds the following meaning:
 151         (1) if match[i] >= 0, anchor i is matched with groundtruth match[i].
 152         (2) if match[i]=-1, anchor i is marked to be background .
 153         (3) if match[i]=-2, anchor i is ignored since it is not background and
 154             does not have sufficient overlap to call it a foreground.
 155 
 156     Raises:
 157       ValueError: if anchors or groundtruth_boxes are not of type
 158         box_list.BoxList
 159     """
 160     if not isinstance(anchors, box_list.BoxList):
 161       raise ValueError('anchors must be an BoxList')
 162     if not isinstance(groundtruth_boxes, box_list.BoxList):
 163       raise ValueError('groundtruth_boxes must be an BoxList')
 164 
 165     if unmatched_class_label is None:
 166       unmatched_class_label = tf.constant([0], tf.float32)
 167 
 168     if groundtruth_labels is None:
 169       groundtruth_labels = tf.ones(tf.expand_dims(groundtruth_boxes.num_boxes(),
 170                                                   0))
 171       groundtruth_labels = tf.expand_dims(groundtruth_labels, -1)
 172 
 173     unmatched_shape_assert = shape_utils.assert_shape_equal(
 174         shape_utils.combined_static_and_dynamic_shape(groundtruth_labels)[1:],
 175         shape_utils.combined_static_and_dynamic_shape(unmatched_class_label))
 176     labels_and_box_shapes_assert = shape_utils.assert_shape_equal(
 177         shape_utils.combined_static_and_dynamic_shape(
 178             groundtruth_labels)[:1],
 179         shape_utils.combined_static_and_dynamic_shape(
 180             groundtruth_boxes.get())[:1])
 181 
 182     if groundtruth_weights is None:
 183       num_gt_boxes = groundtruth_boxes.num_boxes_static()
 184       if not num_gt_boxes:
 185         num_gt_boxes = groundtruth_boxes.num_boxes()
 186       groundtruth_weights = tf.ones([num_gt_boxes], dtype=tf.float32)
 187 
 188     # set scores on the gt boxes
 189     scores = 1 - groundtruth_labels[:, 0]
 190     groundtruth_boxes.add_field(fields.BoxListFields.scores, scores)
 191 
 192     with tf.control_dependencies(
 193         [unmatched_shape_assert, labels_and_box_shapes_assert]):
 194       match_quality_matrix = self._similarity_calc.compare(groundtruth_boxes,
 195                                                            anchors)
 196       match = self._matcher.match(match_quality_matrix,
 197                                   valid_rows=tf.greater(groundtruth_weights, 0))
 198       reg_targets = self._create_regression_targets(anchors,
 199                                                     groundtruth_boxes,
 200                                                     match)
 201       cls_targets = self._create_classification_targets(groundtruth_labels,
 202                                                         unmatched_class_label,
 203                                                         match)
 204       reg_weights = self._create_regression_weights(match, groundtruth_weights)
 205 
 206       cls_weights = self._create_classification_weights(match,
 207                                                         groundtruth_weights)
 208       # convert cls_weights from per-anchor to per-class.
 209       class_label_shape = tf.shape(cls_targets)[1:]
 210       weights_shape = tf.shape(cls_weights)
 211       weights_multiple = tf.concat(
 212           [tf.ones_like(weights_shape), class_label_shape],
 213           axis=0)
 214       for _ in range(len(cls_targets.get_shape()[1:])):
 215         cls_weights = tf.expand_dims(cls_weights, -1)
 216       cls_weights = tf.tile(cls_weights, weights_multiple)
 217 
 218     num_anchors = anchors.num_boxes_static()
 219     if num_anchors is not None:
 220       reg_targets = self._reset_target_shape(reg_targets, num_anchors)
 221       cls_targets = self._reset_target_shape(cls_targets, num_anchors)
 222       reg_weights = self._reset_target_shape(reg_weights, num_anchors)
 223       cls_weights = self._reset_target_shape(cls_weights, num_anchors)
 224 
 225     return (cls_targets, cls_weights, reg_targets, reg_weights,
 226             match.match_results)
 227 
 228   def _reset_target_shape(self, target, num_anchors):
 229     """Sets the static shape of the target.
 230 
 231     Args:
 232       target: the target tensor. Its first dimension will be overwritten.
 233       num_anchors: the number of anchors, which is used to override the target's
 234         first dimension.
 235 
 236     Returns:
 237       A tensor with the shape info filled in.
 238     """
 239     target_shape = target.get_shape().as_list()
 240     target_shape[0] = num_anchors
 241     target.set_shape(target_shape)
 242     return target

 244   def _create_regression_targets(self, anchors, groundtruth_boxes, match):
 245     """Returns a regression target for each anchor.
 246 
 247     Args:
 248       anchors: a BoxList representing N anchors
 249       groundtruth_boxes: a BoxList representing M groundtruth_boxes
 250       match: a matcher.Match object
 251 
 252     Returns:
 253       reg_targets: a float32 tensor with shape [N, box_code_dimension]
 254     """
 255     matched_gt_boxes = match.gather_based_on_match(
 256         groundtruth_boxes.get(),
 257         unmatched_value=tf.zeros(4),
 258         ignored_value=tf.zeros(4))
 259     matched_gt_boxlist = box_list.BoxList(matched_gt_boxes)
 260     if groundtruth_boxes.has_field(fields.BoxListFields.keypoints):
 261       groundtruth_keypoints = groundtruth_boxes.get_field(
 262           fields.BoxListFields.keypoints)
 263       matched_keypoints = match.gather_based_on_match(
 264           groundtruth_keypoints,
 265           unmatched_value=tf.zeros(groundtruth_keypoints.get_shape()[1:]),
 266           ignored_value=tf.zeros(groundtruth_keypoints.get_shape()[1:]))
 267       matched_gt_boxlist.add_field(fields.BoxListFields.keypoints,
 268                                    matched_keypoints)
 269     matched_reg_targets = self._box_coder.encode(matched_gt_boxlist, anchors)
 270     match_results_shape = shape_utils.combined_static_and_dynamic_shape(
 271         match.match_results)
 272 
 273     # Zero out the unmatched and ignored regression targets.
 274     unmatched_ignored_reg_targets = tf.tile(
 275         self._default_regression_target(), [match_results_shape[0], 1])
 276     matched_anchors_mask = match.matched_column_indicator()
 277     reg_targets = tf.where(matched_anchors_mask,
 278                            matched_reg_targets,
 279                            unmatched_ignored_reg_targets)
 280     return reg_targets
 281 
 282   def _default_regression_target(self):
 283     """Returns the default target for anchors to regress to.
 284 
 285     Default regression targets are set to zero (though in
 286     this implementation what these targets are set to should
 287     not matter as the regression weight of any box set to
 288     regress to the default target is zero).
 289 
 290     Returns:
 291       default_target: a float32 tensor with shape [1, box_code_dimension]
 292     """
 293     return tf.constant([self._box_coder.code_size*[0]], tf.float32)

 295   def _create_classification_targets(self, groundtruth_labels,
 296                                      unmatched_class_label, match):
 297     """Create classification targets for each anchor.
 298 
 299     Assign a classification target of for each anchor to the matching
 300     groundtruth label that is provided by match.  Anchors that are not matched
 301     to anything are given the target self._unmatched_cls_target
 302 
 303     Args:
 304       groundtruth_labels:  a tensor of shape [num_gt_boxes, d_1, ... d_k]
 305         with labels for each of the ground_truth boxes. The subshape
 306         [d_1, ... d_k] can be empty (corresponding to scalar labels).
 307       unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
 308         which is consistent with the classification target for each
 309         anchor (and can be empty for scalar targets).  This shape must thus be
 310         compatible with the groundtruth labels that are passed to the "assign"
 311         function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
 312       match: a matcher.Match object that provides a matching between anchors
 313         and groundtruth boxes.
 314 
 315     Returns:
 316       a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k], where the
 317       subshape [d_1, ..., d_k] is compatible with groundtruth_labels which has
 318       shape [num_gt_boxes, d_1, d_2, ... d_k].
 319     """
 320     return match.gather_based_on_match(
 321         groundtruth_labels,
 322         unmatched_value=unmatched_class_label,
 323         ignored_value=unmatched_class_label)
 324 
 325   def _create_regression_weights(self, match, groundtruth_weights):
 326     """Set regression weight for each anchor.
 327 
 328     Only positive anchors are set to contribute to the regression loss, so this
 329     method returns a weight of 1 for every positive anchor and 0 for every
 330     negative anchor.
 331 
 332     Args:
 333       match: a matcher.Match object that provides a matching between anchors
 334         and groundtruth boxes.
 335       groundtruth_weights: a float tensor of shape [M] indicating the weight to
 336         assign to all anchors match to a particular groundtruth box.
 337 
 338     Returns:
 339       a float32 tensor with shape [num_anchors] representing regression weights.
 340     """
 341     return match.gather_based_on_match(
 342         groundtruth_weights, ignored_value=0., unmatched_value=0.)
 343 
 344   def _create_classification_weights(self,
 345                                      match,
 346                                      groundtruth_weights):
 347     """Create classification weights for each anchor.
 348 
 349     Positive (matched) anchors are associated with a weight of
 350     positive_class_weight and negative (unmatched) anchors are associated with
 351     a weight of negative_class_weight. When anchors are ignored, weights are set
 352     to zero. By default, both positive/negative weights are set to 1.0,
 353     but they can be adjusted to handle class imbalance (which is almost always
 354     the case in object detection).
 355 
 356     Args:
 357       match: a matcher.Match object that provides a matching between anchors
 358         and groundtruth boxes.
 359       groundtruth_weights: a float tensor of shape [M] indicating the weight to
 360         assign to all anchors match to a particular groundtruth box.
 361 
 362     Returns:
 363       a float32 tensor with shape [num_anchors] representing classification
 364       weights.
 365     """
 366     return match.gather_based_on_match(
 367         groundtruth_weights,
 368         ignored_value=0.,
 369         unmatched_value=self._negative_class_weight)
 370

 380 # TODO(rathodv): This method pulls in all the implementation dependencies into
 381 # core. Therefore its best to have this factory method outside of core.
 382 def create_target_assigner(reference, stage=None,
 383                            negative_class_weight=1.0, use_matmul_gather=False):
 384   """Factory function for creating standard target assigners.
 385 
 386   Args:
 387     reference: string referencing the type of TargetAssigner.
 388     stage: string denoting stage: {proposal, detection}.
 389     negative_class_weight: classification weight to be associated to negative
 390       anchors (default: 1.0)
 391     use_matmul_gather: whether to use matrix multiplication based gather which
 392       are better suited for TPUs.
 393 
 394   Returns:
 395     TargetAssigner: desired target assigner.
 396 
 397   Raises:
 398     ValueError: if combination reference+stage is invalid.
 399   """
 400   if reference == 'Multibox' and stage == 'proposal':
 401     similarity_calc = sim_calc.NegSqDistSimilarity()
 402     matcher = bipartite_matcher.GreedyBipartiteMatcher()
 403     box_coder_instance = mean_stddev_box_coder.MeanStddevBoxCoder()
 404 
 405   elif reference == 'FasterRCNN' and stage == 'proposal':
 406     similarity_calc = sim_calc.IouSimilarity()
 407     matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=0.7,
 408                                            unmatched_threshold=0.3,
 409                                            force_match_for_each_row=True,
 410                                            use_matmul_gather=use_matmul_gather)
 411     box_coder_instance = faster_rcnn_box_coder.FasterRcnnBoxCoder(
 412         scale_factors=[10.0, 10.0, 5.0, 5.0])
 413 
 414   elif reference == 'FasterRCNN' and stage == 'detection':
 415     similarity_calc = sim_calc.IouSimilarity()
 416     # Uses all proposals with IOU < 0.5 as candidate negatives.
 417     matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=0.5,
 418                                            negatives_lower_than_unmatched=True,
 419                                            use_matmul_gather=use_matmul_gather)
 420     box_coder_instance = faster_rcnn_box_coder.FasterRcnnBoxCoder(
 421         scale_factors=[10.0, 10.0, 5.0, 5.0])
 422 
 423   elif reference == 'FastRCNN':
 424     similarity_calc = sim_calc.IouSimilarity()
 425     matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=0.5,
 426                                            unmatched_threshold=0.1,
 427                                            force_match_for_each_row=False,
 428                                            negatives_lower_than_unmatched=False,
 429                                            use_matmul_gather=use_matmul_gather)
 430     box_coder_instance = faster_rcnn_box_coder.FasterRcnnBoxCoder()
 431 
 432   else:
 433     raise ValueError('No valid combination of reference and stage.')
 434 
 435   return TargetAssigner(similarity_calc, matcher, box_coder_instance,
 436                         negative_class_weight=negative_class_weight)

 439 def batch_assign(target_assigner,
 440                  anchors_batch,
 441                  gt_box_batch,
 442                  gt_class_targets_batch,
 443                  unmatched_class_label=None,
 444                  gt_weights_batch=None):
 445   """Batched assignment of classification and regression targets.
 446 
 447   Args:
 448     target_assigner: a target assigner.
 449     anchors_batch: BoxList representing N box anchors or list of BoxList objects
 450       with length batch_size representing anchor sets.
 451     gt_box_batch: a list of BoxList objects with length batch_size
 452       representing groundtruth boxes for each image in the batch
 453     gt_class_targets_batch: a list of tensors with length batch_size, where
 454       each tensor has shape [num_gt_boxes_i, classification_target_size] and
 455       num_gt_boxes_i is the number of boxes in the ith boxlist of
 456       gt_box_batch.
 457     unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
 458       which is consistent with the classification target for each
 459       anchor (and can be empty for scalar targets).  This shape must thus be
 460       compatible with the groundtruth labels that are passed to the "assign"
 461       function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
 462     gt_weights_batch: A list of 1-D tf.float32 tensors of shape
 463       [num_boxes] containing weights for groundtruth boxes.
 464 
 465   Returns:
 466     batch_cls_targets: a tensor with shape [batch_size, num_anchors,
 467       num_classes],
 468     batch_cls_weights: a tensor with shape [batch_size, num_anchors,
 469       num_classes],
 470     batch_reg_targets: a tensor with shape [batch_size, num_anchors,
 471       box_code_dimension]
 472     batch_reg_weights: a tensor with shape [batch_size, num_anchors],
 473     match: an int32 tensor of shape [batch_size, num_anchors] containing result
 474       of anchor groundtruth matching. Each position in the tensor indicates an
 475       anchor and holds the following meaning:
 476       (1) if match[x, i] >= 0, anchor i is matched with groundtruth match[x, i].
 477       (2) if match[x, i]=-1, anchor i is marked to be background .
 478       (3) if match[x, i]=-2, anchor i is ignored since it is not background and
 479           does not have sufficient overlap to call it a foreground.
 480 
 481   Raises:
 482     ValueError: if input list lengths are inconsistent, i.e.,
 483       batch_size == len(gt_box_batch) == len(gt_class_targets_batch)
 484         and batch_size == len(anchors_batch) unless anchors_batch is a single
 485         BoxList.
 486   """
 487   if not isinstance(anchors_batch, list):
 488     anchors_batch = len(gt_box_batch) * [anchors_batch]
 489   if not all(
 490       isinstance(anchors, box_list.BoxList) for anchors in anchors_batch):
 491     raise ValueError('anchors_batch must be a BoxList or list of BoxLists.')
 492   if not (len(anchors_batch)
 493           == len(gt_box_batch)
 494           == len(gt_class_targets_batch)):
 495     raise ValueError('batch size incompatible with lengths of anchors_batch, '
 496                      'gt_box_batch and gt_class_targets_batch.')
 497   cls_targets_list = []
 498   cls_weights_list = []
 499   reg_targets_list = []
 500   reg_weights_list = []
 501   match_list = []
 502   if gt_weights_batch is None:
 503     gt_weights_batch = [None] * len(gt_class_targets_batch)
 504   for anchors, gt_boxes, gt_class_targets, gt_weights in zip(
 505       anchors_batch, gt_box_batch, gt_class_targets_batch, gt_weights_batch):
 506     (cls_targets, cls_weights,
 507      reg_targets, reg_weights, match) = target_assigner.assign(
 508          anchors, gt_boxes, gt_class_targets, unmatched_class_label, gt_weights)
 509     cls_targets_list.append(cls_targets)
 510     cls_weights_list.append(cls_weights)
 511     reg_targets_list.append(reg_targets)
 512     reg_weights_list.append(reg_weights)
 513     match_list.append(match)
 514   batch_cls_targets = tf.stack(cls_targets_list)
 515   batch_cls_weights = tf.stack(cls_weights_list)
 516   batch_reg_targets = tf.stack(reg_targets_list)
 517   batch_reg_weights = tf.stack(reg_weights_list)
 518   batch_match = tf.stack(match_list)
 519   return (batch_cls_targets, batch_cls_weights, batch_reg_targets,
 520           batch_reg_weights, batch_match)

 527 def batch_get_targets(batch_match, groundtruth_tensor_list,
 528                       groundtruth_weights_list, unmatched_value,
 529                       unmatched_weight):
 530   """Returns targets based on anchor-groundtruth box matching results.
 531 
 532   Args:
 533     batch_match: An int32 tensor of shape [batch, num_anchors] containing the
 534       result of target assignment returned by TargetAssigner.assign(..).
 535     groundtruth_tensor_list: A list of groundtruth tensors of shape
 536       [num_groundtruth, d_1, d_2, ..., d_k]. The tensors can be of any type.
 537     groundtruth_weights_list: A list of weights, one per groundtruth tensor, of
 538       shape [num_groundtruth].
 539     unmatched_value: A tensor of shape [d_1, d_2, ..., d_k] of the same type as
 540       groundtruth tensor containing target value for anchors that remain
 541       unmatched.
 542     unmatched_weight: Scalar weight to assign to anchors that remain unmatched.
 543 
 544   Returns:
 545     targets: A tensor of shape [batch, num_anchors, d_1, d_2, ..., d_k]
 546       containing targets for anchors.
 547     weights: A float tensor of shape [batch, num_anchors] containing the weights
 548       to assign to each target.
 549   """
 550   match_list = tf.unstack(batch_match)
 551   targets_list = []
 552   weights_list = []
 553   for match_tensor, groundtruth_tensor, groundtruth_weight in zip(
 554       match_list, groundtruth_tensor_list, groundtruth_weights_list):
 555     match_object = mat.Match(match_tensor)
 556     targets = match_object.gather_based_on_match(
 557         groundtruth_tensor,
 558         unmatched_value=unmatched_value,
 559         ignored_value=unmatched_value)
 560     targets_list.append(targets)
 561     weights = match_object.gather_based_on_match(
 562         groundtruth_weight,
 563         unmatched_value=unmatched_weight,
 564         ignored_value=tf.zeros_like(unmatched_weight))
 565     weights_list.append(weights)
 566   return tf.stack(targets_list), tf.stack(weights_list)

 569 def batch_assign_confidences(target_assigner,
 570                              anchors_batch,
 571                              gt_box_batch,
 572                              gt_class_confidences_batch,
 573                              gt_weights_batch=None,
 574                              unmatched_class_label=None,
 575                              include_background_class=True,
 576                              implicit_class_weight=1.0):
 577   """Batched assignment of classification and regression targets.
 578 
 579   This differences between batch_assign_confidences and batch_assign_targets:
 580    - 'batch_assign_targets' supports scalar (agnostic), vector (multiclass) and
 581      tensor (high-dimensional) targets. 'batch_assign_confidences' only support
 582      scalar (agnostic) and vector (multiclass) targets.
 583    - 'batch_assign_targets' assumes the input class tensor using the binary
 584      one/K-hot encoding. 'batch_assign_confidences' takes the class confidence
 585      scores as the input, where 1 means positive classes, 0 means implicit
 586      negative classes, and -1 means explicit negative classes.
 587    - 'batch_assign_confidences' assigns the targets in the similar way as
 588      'batch_assign_targets' except that it gives different weights for implicit
 589      and explicit classes. This allows user to control the negative gradients
 590      pushed differently for implicit and explicit examples during the training.
 591 
 592   Args:
 593     target_assigner: a target assigner.
 594     anchors_batch: BoxList representing N box anchors or list of BoxList objects
 595       with length batch_size representing anchor sets.
 596     gt_box_batch: a list of BoxList objects with length batch_size
 597       representing groundtruth boxes for each image in the batch
 598     gt_class_confidences_batch: a list of tensors with length batch_size, where
 599       each tensor has shape [num_gt_boxes_i, classification_target_size] and
 600       num_gt_boxes_i is the number of boxes in the ith boxlist of
 601       gt_box_batch. Note that in this tensor, 1 means explicit positive class,
 602       -1 means explicit negative class, and 0 means implicit negative class.
 603     gt_weights_batch: A list of 1-D tf.float32 tensors of shape
 604       [num_gt_boxes_i] containing weights for groundtruth boxes.
 605     unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
 606       which is consistent with the classification target for each
 607       anchor (and can be empty for scalar targets).  This shape must thus be
 608       compatible with the groundtruth labels that are passed to the "assign"
 609       function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
 610     include_background_class: whether or not gt_class_confidences_batch includes
 611       the background class.
 612     implicit_class_weight: the weight assigned to implicit examples.
 613 
 614   Returns:
 615     batch_cls_targets: a tensor with shape [batch_size, num_anchors,
 616       num_classes],
 617     batch_cls_weights: a tensor with shape [batch_size, num_anchors,
 618       num_classes],
 619     batch_reg_targets: a tensor with shape [batch_size, num_anchors,
 620       box_code_dimension]
 621     batch_reg_weights: a tensor with shape [batch_size, num_anchors],
 622     match: an int32 tensor of shape [batch_size, num_anchors] containing result
 623       of anchor groundtruth matching. Each position in the tensor indicates an
 624       anchor and holds the following meaning:
 625       (1) if match[x, i] >= 0, anchor i is matched with groundtruth match[x, i].
 626       (2) if match[x, i]=-1, anchor i is marked to be background .
 627       (3) if match[x, i]=-2, anchor i is ignored since it is not background and
 628           does not have sufficient overlap to call it a foreground.
 629 
 630   Raises:
 631     ValueError: if input list lengths are inconsistent, i.e.,
 632       batch_size == len(gt_box_batch) == len(gt_class_targets_batch)
 633       and batch_size == len(anchors_batch) unless anchors_batch is a single
 634       BoxList, or if any element in gt_class_confidences_batch has rank > 2.
 635   """
 636   if not isinstance(anchors_batch, list):
 637     anchors_batch = len(gt_box_batch) * [anchors_batch]
 638   if not all(
 639       isinstance(anchors, box_list.BoxList) for anchors in anchors_batch):
 640     raise ValueError('anchors_batch must be a BoxList or list of BoxLists.')
 641   if not (len(anchors_batch)
 642           == len(gt_box_batch)
 643           == len(gt_class_confidences_batch)):
 644     raise ValueError('batch size incompatible with lengths of anchors_batch, '
 645                      'gt_box_batch and gt_class_confidences_batch.')
 646 
 647   cls_targets_list = []
 648   cls_weights_list = []
 649   reg_targets_list = []
 650   reg_weights_list = []
 651   match_list = []
 652   if gt_weights_batch is None:
 653     gt_weights_batch = [None] * len(gt_class_confidences_batch)
 654   for anchors, gt_boxes, gt_class_confidences, gt_weights in zip(
 655       anchors_batch, gt_box_batch, gt_class_confidences_batch,
 656       gt_weights_batch):
 657 
 658     if (gt_class_confidences is not None and
 659         len(gt_class_confidences.get_shape().as_list()) > 2):
 660       raise ValueError('The shape of the class target is not supported. ',
 661                        gt_class_confidences.get_shape())
 662 
 663     cls_targets, _, reg_targets, _, match = target_assigner.assign(
 664         anchors, gt_boxes, gt_class_confidences, unmatched_class_label,
 665         groundtruth_weights=gt_weights)
 666 
 667     if include_background_class:
 668       cls_targets_without_background = tf.slice(
 669           cls_targets, [0, 1], [-1, -1])
 670     else:
 671       cls_targets_without_background = cls_targets
 672 
 673     positive_mask = tf.greater(cls_targets_without_background, 0.0)
 674     negative_mask = tf.less(cls_targets_without_background, 0.0)
 675     explicit_example_mask = tf.logical_or(positive_mask, negative_mask)
 676     positive_anchors = tf.reduce_any(positive_mask, axis=-1)
 677 
 678     regression_weights = tf.cast(positive_anchors, dtype=tf.float32)
 679     regression_targets = (
 680         reg_targets * tf.expand_dims(regression_weights, axis=-1))
 681     regression_weights_expanded = tf.expand_dims(regression_weights, axis=-1)
 682 
 683     cls_targets_without_background = (
 684         cls_targets_without_background *
 685         (1 - tf.cast(negative_mask, dtype=tf.float32)))
 686     cls_weights_without_background = ((1 - implicit_class_weight) * tf.cast(
 687         explicit_example_mask, dtype=tf.float32) + implicit_class_weight)
 688 
 689     if include_background_class:
 690       cls_weights_background = (
 691           (1 - implicit_class_weight) * regression_weights_expanded
 692           + implicit_class_weight)
 693       classification_weights = tf.concat(
 694           [cls_weights_background, cls_weights_without_background], axis=-1)
 695       cls_targets_background = 1 - regression_weights_expanded
 696       classification_targets = tf.concat(
 697           [cls_targets_background, cls_targets_without_background], axis=-1)
 698     else:
 699       classification_targets = cls_targets_without_background
 700       classification_weights = cls_weights_without_background
 701 
 702     cls_targets_list.append(classification_targets)
 703     cls_weights_list.append(classification_weights)
 704     reg_targets_list.append(regression_targets)
 705     reg_weights_list.append(regression_weights)
 706     match_list.append(match)
 707   batch_cls_targets = tf.stack(cls_targets_list)
 708   batch_cls_weights = tf.stack(cls_weights_list)
 709   batch_reg_targets = tf.stack(reg_targets_list)
 710   batch_reg_weights = tf.stack(reg_weights_list)
 711   batch_match = tf.stack(match_list)
 712   return (batch_cls_targets, batch_cls_weights, batch_reg_targets,
 713           batch_reg_weights, batch_match)
