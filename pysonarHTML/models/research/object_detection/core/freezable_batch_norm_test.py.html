  29 class FreezableBatchNormTest(tf.test.TestCase):
  30   """Tests for FreezableBatchNorm operations."""
  31 
  32   def _build_model(self, training=None):
  33     model = tf.keras.models.Sequential()
  34     norm = freezable_batch_norm.FreezableBatchNorm(training=training,
  35                                                    input_shape=(10,),
  36                                                    momentum=0.8)
  37     model.add(norm)
  38     return model, norm
  39 
  40   def _copy_weights(self, source_weights, target_weights):
  41     for source, target in zip(source_weights, target_weights):
  42       target.assign(source)
  43 
  44   def _train_freezable_batch_norm(self, training_mean, training_var):
  45     model, _ = self._build_model()
  46     model.compile(loss='mse', optimizer='sgd')
  47 
  48     # centered on training_mean, variance training_var
  49     train_data = np.random.normal(
  50         loc=training_mean,
  51         scale=training_var,
  52         size=(1000, 10))
  53     model.fit(train_data, train_data, epochs=4, verbose=0)
  54     return model.weights
  55 
  56   def _test_batchnorm_layer(
  57       self, norm, should_be_training, test_data,
  58       testing_mean, testing_var, training_arg, training_mean, training_var):
  59     out_tensor = norm(tf.convert_to_tensor(test_data, dtype=tf.float32),
  60                       training=training_arg)
  61     out = out_tensor
  62     out -= norm.beta
  63     out /= norm.gamma
  64 
  65     if not should_be_training:
  66       out *= training_var
  67       out += (training_mean - testing_mean)
  68       out /= testing_var
  69 
  70     np.testing.assert_allclose(out.numpy().mean(), 0.0, atol=1.5e-1)
  71     np.testing.assert_allclose(out.numpy().std(), 1.0, atol=1.5e-1)
  72 
  73   def test_batchnorm_freezing_training_none(self):
  74     training_mean = 5.0
  75     training_var = 10.0
  76 
  77     testing_mean = -10.0
  78     testing_var = 5.0
  79 
  80     # Initially train the batch norm, and save the weights
  81     trained_weights = self._train_freezable_batch_norm(training_mean,
  82                                                        training_var)
  83 
  84     # Load the batch norm weights, freezing training to True.
  85     # Apply the batch norm layer to testing data and ensure it is normalized
  86     # according to the batch statistics.
  87     model, norm = self._build_model(training=True)
  88     self._copy_weights(trained_weights, model.weights)
  89 
  90     # centered on testing_mean, variance testing_var
  91     test_data = np.random.normal(
  92         loc=testing_mean,
  93         scale=testing_var,
  94         size=(1000, 10))
  95 
  96     # Test with training=True passed to the call method:
  97     training_arg = True
  98     should_be_training = True
  99     self._test_batchnorm_layer(norm, should_be_training, test_data,
 100                                testing_mean, testing_var, training_arg,
 101                                training_mean, training_var)
 102 
 103     # Reset the weights, because they may have been updating by
 104     # running with training=True
 105     self._copy_weights(trained_weights, model.weights)
 106 
 107     # Test with training=False passed to the call method:
 108     training_arg = False
 109     should_be_training = False
 110     self._test_batchnorm_layer(norm, should_be_training, test_data,
 111                                testing_mean, testing_var, training_arg,
 112                                training_mean, training_var)
 113 
 114     # Test the layer in various Keras learning phase scopes:
 115     training_arg = None
 116     should_be_training = False
 117     self._test_batchnorm_layer(norm, should_be_training, test_data,
 118                                testing_mean, testing_var, training_arg,
 119                                training_mean, training_var)
 120 
 121     tf.keras.backend.set_learning_phase(True)
 122     should_be_training = True
 123     self._test_batchnorm_layer(norm, should_be_training, test_data,
 124                                testing_mean, testing_var, training_arg,
 125                                training_mean, training_var)
 126 
 127     # Reset the weights, because they may have been updating by
 128     # running with training=True
 129     self._copy_weights(trained_weights, model.weights)
 130 
 131     tf.keras.backend.set_learning_phase(False)
 132     should_be_training = False
 133     self._test_batchnorm_layer(norm, should_be_training, test_data,
 134                                testing_mean, testing_var, training_arg,
 135                                training_mean, training_var)
