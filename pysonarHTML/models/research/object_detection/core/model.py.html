  15 """Abstract detection model.
  16 
  17 This file defines a generic base class for detection models.  Programs that are
  18 designed to work with arbitrary detection models should only depend on this
  19 class.  We intend for the functions in this class to follow tensor-in/tensor-out
  20 design, thus all functions have tensors or lists/dictionaries holding tensors as
  21 inputs and outputs.
  22 
  23 Abstractly, detection models predict output tensors given input images
  24 which can be passed to a loss function at training time or passed to a
  25 postprocessing function at eval time.  The computation graphs at a high level
  26 consequently look as follows:
  27 
  28 Training time:
  29 inputs (images tensor) -> preprocess -> predict -> loss -> outputs (loss tensor)
  30 
  31 Evaluation time:
  32 inputs (images tensor) -> preprocess -> predict -> postprocess
  33  -> outputs (boxes tensor, scores tensor, classes tensor, num_detections tensor)
  34 
  35 DetectionModels must thus implement four functions (1) preprocess, (2) predict,
  36 (3) postprocess and (4) loss.  DetectionModels should make no assumptions about
  37 the input size or aspect ratio --- they are responsible for doing any
  38 resize/reshaping necessary (see docstring for the preprocess function).
  39 Output classes are always integers in the range [0, num_classes).  Any mapping
  40 of these integers to semantic labels is to be handled outside of this class.
  41 
  42 Images are resized in the `preprocess` method. All of `preprocess`, `predict`,
  43 and `postprocess` should be reentrant.
  44 
  45 The `preprocess` method runs `image_resizer_fn` that returns resized_images and
  46 `true_image_shapes`. Since `image_resizer_fn` can pad the images with zeros,
  47 true_image_shapes indicate the slices that contain the image without padding.
  48 This is useful for padding images to be a fixed size for batching.
  49 
  50 The `postprocess` method uses the true image shapes to clip predictions that lie
  51 outside of images.
  52 
  53 By default, DetectionModels produce bounding box detections; However, we support
  54 a handful of auxiliary annotations associated with each bounding box, namely,
  55 instance masks and keypoints.
  56 """

  82   def __init__(self, num_classes):
  83     """Constructor.
  84 
  85     Args:
  86       num_classes: number of classes.  Note that num_classes *does not* include
  87       background categories that might be implicitly predicted in various
  88       implementations.
  89     """
  90     self._num_classes = num_classes
  91     self._groundtruth_lists = {}
  92 
  93     super(DetectionModel, self).__init__()

  96   def num_classes(self):
  97     return self._num_classes

  99   def groundtruth_lists(self, field):
 100     """Access list of groundtruth tensors.
 101 
 102     Args:
 103       field: a string key, options are
 104         fields.BoxListFields.{boxes,classes,masks,keypoints,
 105         keypoint_visibilities} or
 106         fields.InputDataFields.is_annotated.
 107 
 108     Returns:
 109       a list of tensors holding groundtruth information (see also
 110       provide_groundtruth function below), with one entry for each image in the
 111       batch.
 112     Raises:
 113       RuntimeError: if the field has not been provided via provide_groundtruth.
 114     """
 115     if field not in self._groundtruth_lists:
 116       raise RuntimeError('Groundtruth tensor {} has not been provided'.format(
 117           field))
 118     return self._groundtruth_lists[field]
 119 
 120   def groundtruth_has_field(self, field):
 121     """Determines whether the groundtruth includes the given field.
 122 
 123     Args:
 124       field: a string key, options are
 125         fields.BoxListFields.{boxes,classes,masks,keypoints,
 126         keypoint_visibilities} or
 127         fields.InputDataFields.is_annotated.
 128 
 129     Returns:
 130       True if the groundtruth includes the given field, False otherwise.
 131     """
 132     return field in self._groundtruth_lists

 153   @abc.abstractmethod
 154   def preprocess(self, inputs):
 155     """Input preprocessing.
 156 
 157     To be overridden by implementations.
 158 
 159     This function is responsible for any scaling/shifting of input values that
 160     is necessary prior to running the detector on an input image.
 161     It is also responsible for any resizing, padding that might be necessary
 162     as images are assumed to arrive in arbitrary sizes.  While this function
 163     could conceivably be part of the predict method (below), it is often
 164     convenient to keep these separate --- for example, we may want to preprocess
 165     on one device, place onto a queue, and let another device (e.g., the GPU)
 166     handle prediction.
 167 
 168     A few important notes about the preprocess function:
 169     + We assume that this operation does not have any trainable variables nor
 170     does it affect the groundtruth annotations in any way (thus data
 171     augmentation operations such as random cropping should be performed
 172     externally).
 173     + There is no assumption that the batchsize in this function is the same as
 174     the batch size in the predict function.  In fact, we recommend calling the
 175     preprocess function prior to calling any batching operations (which should
 176     happen outside of the model) and thus assuming that batch sizes are equal
 177     to 1 in the preprocess function.
 178     + There is also no explicit assumption that the output resolutions
 179     must be fixed across inputs --- this is to support "fully convolutional"
 180     settings in which input images can have different shapes/resolutions.
 181 
 182     Args:
 183       inputs: a [batch, height_in, width_in, channels] float32 tensor
 184         representing a batch of images with values between 0 and 255.0.
 185 
 186     Returns:
 187       preprocessed_inputs: a [batch, height_out, width_out, channels] float32
 188         tensor representing a batch of images.
 189       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 190         of the form [height, width, channels] indicating the shapes
 191         of true images in the resized images, as resized images can be padded
 192         with zeros.
 193     """
 194     pass

 196   @abc.abstractmethod
 197   def predict(self, preprocessed_inputs, true_image_shapes, **side_inputs):
 198     """Predict prediction tensors from inputs tensor.
 199 
 200     Outputs of this function can be passed to loss or postprocess functions.
 201 
 202     Args:
 203       preprocessed_inputs: a [batch, height, width, channels] float32 tensor
 204         representing a batch of images.
 205       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 206         of the form [height, width, channels] indicating the shapes
 207         of true images in the resized images, as resized images can be padded
 208         with zeros.
 209       **side_inputs: additional tensors that are required by the network.
 210 
 211     Returns:
 212       prediction_dict: a dictionary holding prediction tensors to be
 213         passed to the Loss or Postprocess functions.
 214     """
 215     pass

 217   @abc.abstractmethod
 218   def postprocess(self, prediction_dict, true_image_shapes, **params):
 219     """Convert predicted output tensors to final detections.
 220 
 221     This stage typically performs a few things such as
 222     * Non-Max Suppression to remove overlapping detection boxes.
 223     * Score conversion and background class removal.
 224 
 225     Outputs adhere to the following conventions:
 226     * Classes are integers in [0, num_classes); background classes are removed
 227       and the first non-background class is mapped to 0. If the model produces
 228       class-agnostic detections, then no output is produced for classes.
 229     * Boxes are to be interpreted as being in [y_min, x_min, y_max, x_max]
 230       format and normalized relative to the image window.
 231     * `num_detections` is provided for settings where detections are padded to a
 232       fixed number of boxes.
 233     * We do not specifically assume any kind of probabilistic interpretation
 234       of the scores --- the only important thing is their relative ordering.
 235       Thus implementations of the postprocess function are free to output
 236       logits, probabilities, calibrated probabilities, or anything else.
 237 
 238     Args:
 239       prediction_dict: a dictionary holding prediction tensors.
 240       true_image_shapes: int32 tensor of shape [batch, 3] where each row is
 241         of the form [height, width, channels] indicating the shapes
 242         of true images in the resized images, as resized images can be padded
 243         with zeros.
 244       **params: Additional keyword arguments for specific implementations of
 245         DetectionModel.
 246 
 247     Returns:
 248       detections: a dictionary containing the following fields
 249         detection_boxes: [batch, max_detections, 4]
 250         detection_scores: [batch, max_detections]
 251         detection_classes: [batch, max_detections]
 252           (If a model is producing class-agnostic detections, this field may be
 253           missing)
 254         instance_masks: [batch, max_detections, image_height, image_width]
 255           (optional)
 256         keypoints: [batch, max_detections, num_keypoints, 2] (optional)
 257         num_detections: [batch]
 258 
 259         In addition to the above fields this stage also outputs the following
 260         raw tensors:
 261 
 262         raw_detection_boxes: [batch, total_detections, 4] tensor containing
 263           all detection boxes from `prediction_dict` in the format
 264           [ymin, xmin, ymax, xmax] and normalized co-ordinates.
 265         raw_detection_scores: [batch, total_detections,
 266           num_classes_with_background] tensor of class score logits for
 267           raw detection boxes.
 268     """
 269     pass

 291   def provide_groundtruth(self,
 292                           groundtruth_boxes_list,
 293                           groundtruth_classes_list,
 294                           groundtruth_masks_list=None,
 295                           groundtruth_keypoints_list=None,
 296                           groundtruth_keypoint_visibilities_list=None,
 297                           groundtruth_weights_list=None,
 298                           groundtruth_confidences_list=None,
 299                           groundtruth_is_crowd_list=None,
 300                           groundtruth_group_of_list=None,
 301                           groundtruth_area_list=None,
 302                           is_annotated_list=None,
 303                           groundtruth_labeled_classes=None):
 304     """Provide groundtruth tensors.
 305 
 306     Args:
 307       groundtruth_boxes_list: a list of 2-D tf.float32 tensors of shape
 308         [num_boxes, 4] containing coordinates of the groundtruth boxes.
 309           Groundtruth boxes are provided in [y_min, x_min, y_max, x_max]
 310           format and assumed to be normalized and clipped
 311           relative to the image window with y_min <= y_max and x_min <= x_max.
 312       groundtruth_classes_list: a list of 2-D tf.float32 one-hot (or k-hot)
 313         tensors of shape [num_boxes, num_classes] containing the class targets
 314         with the 0th index assumed to map to the first non-background class.
 315       groundtruth_masks_list: a list of 3-D tf.float32 tensors of
 316         shape [num_boxes, height_in, width_in] containing instance
 317         masks with values in {0, 1}.  If None, no masks are provided.
 318         Mask resolution `height_in`x`width_in` must agree with the resolution
 319         of the input image tensor provided to the `preprocess` function.
 320       groundtruth_keypoints_list: a list of 3-D tf.float32 tensors of
 321         shape [num_boxes, num_keypoints, 2] containing keypoints.
 322         Keypoints are assumed to be provided in normalized coordinates and
 323         missing keypoints should be encoded as NaN (but it is recommended to use
 324         `groundtruth_keypoint_visibilities_list`).
 325       groundtruth_keypoint_visibilities_list: a list of 3-D tf.bool tensors
 326         of shape [num_boxes, num_keypoints] containing keypoint visibilities.
 327       groundtruth_weights_list: A list of 1-D tf.float32 tensors of shape
 328         [num_boxes] containing weights for groundtruth boxes.
 329       groundtruth_confidences_list: A list of 2-D tf.float32 tensors of shape
 330         [num_boxes, num_classes] containing class confidences for groundtruth
 331         boxes.
 332       groundtruth_is_crowd_list: A list of 1-D tf.bool tensors of shape
 333         [num_boxes] containing is_crowd annotations.
 334       groundtruth_group_of_list: A list of 1-D tf.bool tensors of shape
 335         [num_boxes] containing group_of annotations.
 336       groundtruth_area_list: A list of 1-D tf.float32 tensors of shape
 337         [num_boxes] containing the area (in the original absolute coordinates)
 338         of the annotations.
 339       is_annotated_list: A list of scalar tf.bool tensors indicating whether
 340         images have been labeled or not.
 341       groundtruth_labeled_classes: A list of 1-D tf.float32 tensors of shape
 342         [num_classes], containing label indices encoded as k-hot of the classes
 343         that are exhaustively annotated.
 344     """
 345     self._groundtruth_lists[fields.BoxListFields.boxes] = groundtruth_boxes_list
 346     self._groundtruth_lists[
 347         fields.BoxListFields.classes] = groundtruth_classes_list
 348     if groundtruth_weights_list:
 349       self._groundtruth_lists[fields.BoxListFields.
 350                               weights] = groundtruth_weights_list
 351     if groundtruth_confidences_list:
 352       self._groundtruth_lists[fields.BoxListFields.
 353                               confidences] = groundtruth_confidences_list
 354     if groundtruth_masks_list:
 355       self._groundtruth_lists[
 356           fields.BoxListFields.masks] = groundtruth_masks_list
 357     if groundtruth_keypoints_list:
 358       self._groundtruth_lists[
 359           fields.BoxListFields.keypoints] = groundtruth_keypoints_list
 360     if groundtruth_keypoint_visibilities_list:
 361       self._groundtruth_lists[
 362           fields.BoxListFields.keypoint_visibilities] = (
 363               groundtruth_keypoint_visibilities_list)
 364     if groundtruth_is_crowd_list:
 365       self._groundtruth_lists[
 366           fields.BoxListFields.is_crowd] = groundtruth_is_crowd_list
 367     if groundtruth_group_of_list:
 368       self._groundtruth_lists[
 369           fields.BoxListFields.group_of] = groundtruth_group_of_list
 370     if groundtruth_area_list:
 371       self._groundtruth_lists[
 372           fields.InputDataFields.groundtruth_area] = groundtruth_area_list
 373     if is_annotated_list:
 374       self._groundtruth_lists[
 375           fields.InputDataFields.is_annotated] = is_annotated_list
 376     if groundtruth_labeled_classes:
 377       self._groundtruth_lists[
 378           fields.InputDataFields
 379           .groundtruth_labeled_classes] = groundtruth_labeled_classes

 393   @abc.abstractmethod
 394   def restore_map(self, fine_tune_checkpoint_type='detection'):
 395     """Returns a map of variables to load from a foreign checkpoint.
 396 
 397     Returns a map of variable names to load from a checkpoint to variables in
 398     the model graph. This enables the model to initialize based on weights from

 400     classification model can be used to bootstrap training of an object

 409         Valid values: `detection`, `classification`. Default 'detection'.
 410 
 411     Returns:
 412       A dict mapping variable names (to load from a checkpoint) to variables in
 413       the model graph.
 414     """
 415     pass

 430   def call(self, images):
 431     """Returns detections from a batch of images.
 432 
 433     This method calls the preprocess, predict and postprocess function
 434     sequentially and returns the output.
 435 
 436     Args:
 437       images: a [batch_size, height, width, channels] float tensor.
 438 
 439     Returns:
 440        detetcions: The dict of tensors returned by the postprocess function.
 441     """
 442 
 443     preprocessed_images, shapes = self.preprocess(images)
 444     prediction_dict = self.predict(preprocessed_images, shapes)
 445     return self.postprocess(prediction_dict, shapes)
