  49 def area(boxlist, scope=None):
  50   """Computes area of boxes.
  51 
  52   Args:
  53     boxlist: BoxList holding N boxes
  54     scope: name scope.
  55 
  56   Returns:
  57     a tensor with shape [N] representing box areas.
  58   """
  59   with tf.name_scope(scope, 'Area'):
  60     y_min, x_min, y_max, x_max = tf.split(
  61         value=boxlist.get(), num_or_size_splits=4, axis=1)
  62     return tf.squeeze((y_max - y_min) * (x_max - x_min), [1])

  65 def height_width(boxlist, scope=None):
  66   """Computes height and width of boxes in boxlist.
  67 
  68   Args:
  69     boxlist: BoxList holding N boxes
  70     scope: name scope.
  71 
  72   Returns:
  73     Height: A tensor with shape [N] representing box heights.
  74     Width: A tensor with shape [N] representing box widths.
  75   """
  76   with tf.name_scope(scope, 'HeightWidth'):
  77     y_min, x_min, y_max, x_max = tf.split(
  78         value=boxlist.get(), num_or_size_splits=4, axis=1)
  79     return tf.squeeze(y_max - y_min, [1]), tf.squeeze(x_max - x_min, [1])

  82 def scale(boxlist, y_scale, x_scale, scope=None):
  83   """scale box coordinates in x and y dimensions.
  84 
  85   Args:
  86     boxlist: BoxList holding N boxes
  87     y_scale: (float) scalar tensor
  88     x_scale: (float) scalar tensor
  89     scope: name scope.
  90 
  91   Returns:
  92     boxlist: BoxList holding N boxes
  93   """
  94   with tf.name_scope(scope, 'Scale'):
  95     y_scale = tf.cast(y_scale, tf.float32)
  96     x_scale = tf.cast(x_scale, tf.float32)
  97     y_min, x_min, y_max, x_max = tf.split(
  98         value=boxlist.get(), num_or_size_splits=4, axis=1)
  99     y_min = y_scale * y_min
 100     y_max = y_scale * y_max
 101     x_min = x_scale * x_min
 102     x_max = x_scale * x_max
 103     scaled_boxlist = box_list.BoxList(
 104         tf.concat([y_min, x_min, y_max, x_max], 1))
 105     return _copy_extra_fields(scaled_boxlist, boxlist)

 108 def scale_height_width(boxlist, y_scale, x_scale, scope=None):
 109   """Scale the height and width of boxes, leaving centers unchanged.
 110 
 111   Args:
 112     boxlist: BoxList holding N boxes
 113     y_scale: (float) scalar tensor
 114     x_scale: (float) scalar tensor
 115     scope: name scope.
 116 
 117   Returns:
 118     boxlist: BoxList holding N boxes
 119   """
 120   with tf.name_scope(scope, 'ScaleHeightWidth'):
 121     y_scale = tf.cast(y_scale, tf.float32)
 122     x_scale = tf.cast(x_scale, tf.float32)
 123     yc, xc, height_orig, width_orig = boxlist.get_center_coordinates_and_sizes()
 124     y_min = yc - 0.5 * y_scale * height_orig
 125     y_max = yc + 0.5 * y_scale * height_orig
 126     x_min = xc - 0.5 * x_scale * width_orig
 127     x_max = xc + 0.5 * x_scale * width_orig
 128     scaled_boxlist = box_list.BoxList(
 129         tf.stack([y_min, x_min, y_max, x_max], 1))
 130     return _copy_extra_fields(scaled_boxlist, boxlist)

 133 def clip_to_window(boxlist, window, filter_nonoverlapping=True, scope=None):
 134   """Clip bounding boxes to a window.
 135 
 136   This op clips any input bounding boxes (represented by bounding box
 137   corners) to a window, optionally filtering out boxes that do not
 138   overlap at all with the window.
 139 
 140   Args:
 141     boxlist: BoxList holding M_in boxes
 142     window: a tensor of shape [4] representing the [y_min, x_min, y_max, x_max]
 143       window to which the op should clip boxes.
 144     filter_nonoverlapping: whether to filter out boxes that do not overlap at
 145       all with the window.
 146     scope: name scope.
 147 
 148   Returns:
 149     a BoxList holding M_out boxes where M_out <= M_in
 150   """
 151   with tf.name_scope(scope, 'ClipToWindow'):
 152     y_min, x_min, y_max, x_max = tf.split(
 153         value=boxlist.get(), num_or_size_splits=4, axis=1)
 154     win_y_min, win_x_min, win_y_max, win_x_max = tf.unstack(window)
 155     y_min_clipped = tf.maximum(tf.minimum(y_min, win_y_max), win_y_min)
 156     y_max_clipped = tf.maximum(tf.minimum(y_max, win_y_max), win_y_min)
 157     x_min_clipped = tf.maximum(tf.minimum(x_min, win_x_max), win_x_min)
 158     x_max_clipped = tf.maximum(tf.minimum(x_max, win_x_max), win_x_min)
 159     clipped = box_list.BoxList(
 160         tf.concat([y_min_clipped, x_min_clipped, y_max_clipped, x_max_clipped],
 161                   1))
 162     clipped = _copy_extra_fields(clipped, boxlist)
 163     if filter_nonoverlapping:
 164       areas = area(clipped)
 165       nonzero_area_indices = tf.cast(
 166           tf.reshape(tf.where(tf.greater(areas, 0.0)), [-1]), tf.int32)
 167       clipped = gather(clipped, nonzero_area_indices)
 168     return clipped

 171 def prune_outside_window(boxlist, window, scope=None):
 172   """Prunes bounding boxes that fall outside a given window.
 173 
 174   This function prunes bounding boxes that even partially fall outside the given
 175   window. See also clip_to_window which only prunes bounding boxes that fall
 176   completely outside the window, and clips any bounding boxes that partially
 177   overflow.
 178 
 179   Args:
 180     boxlist: a BoxList holding M_in boxes.
 181     window: a float tensor of shape [4] representing [ymin, xmin, ymax, xmax]
 182       of the window
 183     scope: name scope.
 184 
 185   Returns:
 186     pruned_corners: a tensor with shape [M_out, 4] where M_out <= M_in
 187     valid_indices: a tensor with shape [M_out] indexing the valid bounding boxes
 188      in the input tensor.
 189   """
 190   with tf.name_scope(scope, 'PruneOutsideWindow'):
 191     y_min, x_min, y_max, x_max = tf.split(
 192         value=boxlist.get(), num_or_size_splits=4, axis=1)
 193     win_y_min, win_x_min, win_y_max, win_x_max = tf.unstack(window)
 194     coordinate_violations = tf.concat([
 195         tf.less(y_min, win_y_min), tf.less(x_min, win_x_min),
 196         tf.greater(y_max, win_y_max), tf.greater(x_max, win_x_max)
 197     ], 1)
 198     valid_indices = tf.reshape(
 199         tf.where(tf.logical_not(tf.reduce_any(coordinate_violations, 1))), [-1])
 200     return gather(boxlist, valid_indices), valid_indices

 203 def prune_completely_outside_window(boxlist, window, scope=None):
 204   """Prunes bounding boxes that fall completely outside of the given window.
 205 
 206   The function clip_to_window prunes bounding boxes that fall
 207   completely outside the window, but also clips any bounding boxes that
 208   partially overflow. This function does not clip partially overflowing boxes.
 209 
 210   Args:
 211     boxlist: a BoxList holding M_in boxes.
 212     window: a float tensor of shape [4] representing [ymin, xmin, ymax, xmax]
 213       of the window
 214     scope: name scope.
 215 
 216   Returns:
 217     pruned_boxlist: a new BoxList with all bounding boxes partially or fully in
 218       the window.
 219     valid_indices: a tensor with shape [M_out] indexing the valid bounding boxes
 220      in the input tensor.
 221   """
 222   with tf.name_scope(scope, 'PruneCompleteleyOutsideWindow'):
 223     y_min, x_min, y_max, x_max = tf.split(
 224         value=boxlist.get(), num_or_size_splits=4, axis=1)
 225     win_y_min, win_x_min, win_y_max, win_x_max = tf.unstack(window)
 226     coordinate_violations = tf.concat([
 227         tf.greater_equal(y_min, win_y_max), tf.greater_equal(x_min, win_x_max),
 228         tf.less_equal(y_max, win_y_min), tf.less_equal(x_max, win_x_min)
 229     ], 1)
 230     valid_indices = tf.reshape(
 231         tf.where(tf.logical_not(tf.reduce_any(coordinate_violations, 1))), [-1])
 232     return gather(boxlist, valid_indices), valid_indices

 235 def intersection(boxlist1, boxlist2, scope=None):
 236   """Compute pairwise intersection areas between boxes.
 237 
 238   Args:
 239     boxlist1: BoxList holding N boxes
 240     boxlist2: BoxList holding M boxes
 241     scope: name scope.
 242 
 243   Returns:
 244     a tensor with shape [N, M] representing pairwise intersections
 245   """
 246   with tf.name_scope(scope, 'Intersection'):
 247     y_min1, x_min1, y_max1, x_max1 = tf.split(
 248         value=boxlist1.get(), num_or_size_splits=4, axis=1)
 249     y_min2, x_min2, y_max2, x_max2 = tf.split(
 250         value=boxlist2.get(), num_or_size_splits=4, axis=1)
 251     all_pairs_min_ymax = tf.minimum(y_max1, tf.transpose(y_max2))
 252     all_pairs_max_ymin = tf.maximum(y_min1, tf.transpose(y_min2))
 253     intersect_heights = tf.maximum(0.0, all_pairs_min_ymax - all_pairs_max_ymin)
 254     all_pairs_min_xmax = tf.minimum(x_max1, tf.transpose(x_max2))
 255     all_pairs_max_xmin = tf.maximum(x_min1, tf.transpose(x_min2))
 256     intersect_widths = tf.maximum(0.0, all_pairs_min_xmax - all_pairs_max_xmin)
 257     return intersect_heights * intersect_widths

 260 def matched_intersection(boxlist1, boxlist2, scope=None):
 261   """Compute intersection areas between corresponding boxes in two boxlists.
 262 
 263   Args:
 264     boxlist1: BoxList holding N boxes
 265     boxlist2: BoxList holding N boxes
 266     scope: name scope.
 267 
 268   Returns:
 269     a tensor with shape [N] representing pairwise intersections
 270   """
 271   with tf.name_scope(scope, 'MatchedIntersection'):
 272     y_min1, x_min1, y_max1, x_max1 = tf.split(
 273         value=boxlist1.get(), num_or_size_splits=4, axis=1)
 274     y_min2, x_min2, y_max2, x_max2 = tf.split(
 275         value=boxlist2.get(), num_or_size_splits=4, axis=1)
 276     min_ymax = tf.minimum(y_max1, y_max2)
 277     max_ymin = tf.maximum(y_min1, y_min2)
 278     intersect_heights = tf.maximum(0.0, min_ymax - max_ymin)
 279     min_xmax = tf.minimum(x_max1, x_max2)
 280     max_xmin = tf.maximum(x_min1, x_min2)
 281     intersect_widths = tf.maximum(0.0, min_xmax - max_xmin)
 282     return tf.reshape(intersect_heights * intersect_widths, [-1])

 285 def iou(boxlist1, boxlist2, scope=None):
 286   """Computes pairwise intersection-over-union between box collections.
 287 
 288   Args:
 289     boxlist1: BoxList holding N boxes
 290     boxlist2: BoxList holding M boxes
 291     scope: name scope.
 292 
 293   Returns:
 294     a tensor with shape [N, M] representing pairwise iou scores.
 295   """
 296   with tf.name_scope(scope, 'IOU'):
 297     intersections = intersection(boxlist1, boxlist2)
 298     areas1 = area(boxlist1)
 299     areas2 = area(boxlist2)
 300     unions = (
 301         tf.expand_dims(areas1, 1) + tf.expand_dims(areas2, 0) - intersections)
 302     return tf.where(
 303         tf.equal(intersections, 0.0),
 304         tf.zeros_like(intersections), tf.truediv(intersections, unions))
 305

 307 def matched_iou(boxlist1, boxlist2, scope=None):
 308   """Compute intersection-over-union between corresponding boxes in boxlists.
 309 
 310   Args:
 311     boxlist1: BoxList holding N boxes
 312     boxlist2: BoxList holding N boxes
 313     scope: name scope.
 314 
 315   Returns:
 316     a tensor with shape [N] representing pairwise iou scores.
 317   """
 318   with tf.name_scope(scope, 'MatchedIOU'):
 319     intersections = matched_intersection(boxlist1, boxlist2)
 320     areas1 = area(boxlist1)
 321     areas2 = area(boxlist2)
 322     unions = areas1 + areas2 - intersections
 323     return tf.where(
 324         tf.equal(intersections, 0.0),
 325         tf.zeros_like(intersections), tf.truediv(intersections, unions))

 328 def ioa(boxlist1, boxlist2, scope=None):
 329   """Computes pairwise intersection-over-area between box collections.
 330 
 331   intersection-over-area (IOA) between two boxes box1 and box2 is defined as
 332   their intersection area over box2's area. Note that ioa is not symmetric,
 333   that is, ioa(box1, box2) != ioa(box2, box1).
 334 
 335   Args:
 336     boxlist1: BoxList holding N boxes
 337     boxlist2: BoxList holding M boxes
 338     scope: name scope.
 339 
 340   Returns:
 341     a tensor with shape [N, M] representing pairwise ioa scores.
 342   """
 343   with tf.name_scope(scope, 'IOA'):
 344     intersections = intersection(boxlist1, boxlist2)
 345     areas = tf.expand_dims(area(boxlist2), 0)
 346     return tf.truediv(intersections, areas)

 349 def prune_non_overlapping_boxes(
 350     boxlist1, boxlist2, min_overlap=0.0, scope=None):
 351   """Prunes the boxes in boxlist1 that overlap less than thresh with boxlist2.
 352 
 353   For each box in boxlist1, we want its IOA to be more than minoverlap with
 354   at least one of the boxes in boxlist2. If it does not, we remove it.
 355 
 356   Args:
 357     boxlist1: BoxList holding N boxes.
 358     boxlist2: BoxList holding M boxes.
 359     min_overlap: Minimum required overlap between boxes, to count them as
 360                 overlapping.
 361     scope: name scope.
 362 
 363   Returns:
 364     new_boxlist1: A pruned boxlist with size [N', 4].
 365     keep_inds: A tensor with shape [N'] indexing kept bounding boxes in the
 366       first input BoxList `boxlist1`.
 367   """
 368   with tf.name_scope(scope, 'PruneNonOverlappingBoxes'):
 369     ioa_ = ioa(boxlist2, boxlist1)  # [M, N] tensor
 370     ioa_ = tf.reduce_max(ioa_, reduction_indices=[0])  # [N] tensor
 371     keep_bool = tf.greater_equal(ioa_, tf.constant(min_overlap))
 372     keep_inds = tf.squeeze(tf.where(keep_bool), axis=[1])
 373     new_boxlist1 = gather(boxlist1, keep_inds)
 374     return new_boxlist1, keep_inds

 377 def prune_small_boxes(boxlist, min_side, scope=None):
 378   """Prunes small boxes in the boxlist which have a side smaller than min_side.
 379 
 380   Args:
 381     boxlist: BoxList holding N boxes.
 382     min_side: Minimum width AND height of box to survive pruning.
 383     scope: name scope.
 384 
 385   Returns:
 386     A pruned boxlist.
 387   """
 388   with tf.name_scope(scope, 'PruneSmallBoxes'):
 389     height, width = height_width(boxlist)
 390     is_valid = tf.logical_and(tf.greater_equal(width, min_side),
 391                               tf.greater_equal(height, min_side))
 392     return gather(boxlist, tf.reshape(tf.where(is_valid), [-1]))

 395 def change_coordinate_frame(boxlist, window, scope=None):
 396   """Change coordinate frame of the boxlist to be relative to window's frame.
 397 
 398   Given a window of the form [ymin, xmin, ymax, xmax],
 399   changes bounding box coordinates from boxlist to be relative to this window
 400   (e.g., the min corner maps to (0,0) and the max corner maps to (1,1)).
 401 
 402   An example use case is data augmentation: where we are given groundtruth
 403   boxes (boxlist) and would like to randomly crop the image to some
 404   window (window). In this case we need to change the coordinate frame of
 405   each groundtruth box to be relative to this new window.
 406 
 407   Args:
 408     boxlist: A BoxList object holding N boxes.
 409     window: A rank 1 tensor [4].
 410     scope: name scope.
 411 
 412   Returns:
 413     Returns a BoxList object with N boxes.
 414   """
 415   with tf.name_scope(scope, 'ChangeCoordinateFrame'):
 416     win_height = window[2] - window[0]
 417     win_width = window[3] - window[1]
 418     boxlist_new = scale(box_list.BoxList(
 419         boxlist.get() - [window[0], window[1], window[0], window[1]]),
 420                         1.0 / win_height, 1.0 / win_width)
 421     boxlist_new = _copy_extra_fields(boxlist_new, boxlist)
 422     return boxlist_new

 425 def sq_dist(boxlist1, boxlist2, scope=None):
 426   """Computes the pairwise squared distances between box corners.
 427 
 428   This op treats each box as if it were a point in a 4d Euclidean space and
 429   computes pairwise squared distances.
 430 
 431   Mathematically, we are given two matrices of box coordinates X and Y,
 432   where X(i,:) is the i'th row of X, containing the 4 numbers defining the
 433   corners of the i'th box in boxlist1. Similarly Y(j,:) corresponds to
 434   boxlist2.  We compute
 435   Z(i,j) = ||X(i,:) - Y(j,:)||^2
 436          = ||X(i,:)||^2 + ||Y(j,:)||^2 - 2 X(i,:)' * Y(j,:),
 437 
 438   Args:
 439     boxlist1: BoxList holding N boxes
 440     boxlist2: BoxList holding M boxes
 441     scope: name scope.
 442 
 443   Returns:
 444     a tensor with shape [N, M] representing pairwise distances
 445   """
 446   with tf.name_scope(scope, 'SqDist'):
 447     sqnorm1 = tf.reduce_sum(tf.square(boxlist1.get()), 1, keep_dims=True)
 448     sqnorm2 = tf.reduce_sum(tf.square(boxlist2.get()), 1, keep_dims=True)
 449     innerprod = tf.matmul(boxlist1.get(), boxlist2.get(),
 450                           transpose_a=False, transpose_b=True)
 451     return sqnorm1 + tf.transpose(sqnorm2) - 2.0 * innerprod

 454 def boolean_mask(boxlist, indicator, fields=None, scope=None,
 455                  use_static_shapes=False, indicator_sum=None):
 456   """Select boxes from BoxList according to indicator and return new BoxList.
 457 
 458   `boolean_mask` returns the subset of boxes that are marked as "True" by the
 459   indicator tensor. By default, `boolean_mask` returns boxes corresponding to
 460   the input index list, as well as all additional fields stored in the boxlist
 461   (indexing into the first dimension).  However one can optionally only draw
 462   from a subset of fields.
 463 
 464   Args:
 465     boxlist: BoxList holding N boxes
 466     indicator: a rank-1 boolean tensor
 467     fields: (optional) list of fields to also gather from.  If None (default),
 468       all fields are gathered from.  Pass an empty fields list to only gather
 469       the box coordinates.
 470     scope: name scope.
 471     use_static_shapes: Whether to use an implementation with static shape
 472       gurantees.
 473     indicator_sum: An integer containing the sum of `indicator` vector. Only
 474       required if `use_static_shape` is True.
 475 
 476   Returns:
 477     subboxlist: a BoxList corresponding to the subset of the input BoxList
 478       specified by indicator
 479   Raises:
 480     ValueError: if `indicator` is not a rank-1 boolean tensor.
 481   """
 482   with tf.name_scope(scope, 'BooleanMask'):
 483     if indicator.shape.ndims != 1:
 484       raise ValueError('indicator should have rank 1')
 485     if indicator.dtype != tf.bool:
 486       raise ValueError('indicator should be a boolean tensor')
 487     if use_static_shapes:
 488       if not (indicator_sum and isinstance(indicator_sum, int)):
 489         raise ValueError('`indicator_sum` must be a of type int')
 490       selected_positions = tf.cast(indicator, dtype=tf.float32)
 491       indexed_positions = tf.cast(
 492           tf.multiply(
 493               tf.cumsum(selected_positions), selected_positions),
 494           dtype=tf.int32)
 495       one_hot_selector = tf.one_hot(
 496           indexed_positions - 1, indicator_sum, dtype=tf.float32)
 497       sampled_indices = tf.cast(
 498           tf.tensordot(
 499               tf.cast(tf.range(tf.shape(indicator)[0]), dtype=tf.float32),
 500               one_hot_selector,
 501               axes=[0, 0]),
 502           dtype=tf.int32)
 503       return gather(boxlist, sampled_indices, use_static_shapes=True)
 504     else:
 505       subboxlist = box_list.BoxList(tf.boolean_mask(boxlist.get(), indicator))
 506       if fields is None:
 507         fields = boxlist.get_extra_fields()
 508       for field in fields:
 509         if not boxlist.has_field(field):
 510           raise ValueError('boxlist must contain all specified fields')
 511         subfieldlist = tf.boolean_mask(boxlist.get_field(field), indicator)
 512         subboxlist.add_field(field, subfieldlist)
 513       return subboxlist

 516 def gather(boxlist, indices, fields=None, scope=None, use_static_shapes=False):
 517   """Gather boxes from BoxList according to indices and return new BoxList.
 518 
 519   By default, `gather` returns boxes corresponding to the input index list, as
 520   well as all additional fields stored in the boxlist (indexing into the
 521   first dimension).  However one can optionally only gather from a
 522   subset of fields.
 523 
 524   Args:
 525     boxlist: BoxList holding N boxes
 526     indices: a rank-1 tensor of type int32 / int64
 527     fields: (optional) list of fields to also gather from.  If None (default),
 528       all fields are gathered from.  Pass an empty fields list to only gather
 529       the box coordinates.
 530     scope: name scope.
 531     use_static_shapes: Whether to use an implementation with static shape
 532       gurantees.
 533 
 534   Returns:
 535     subboxlist: a BoxList corresponding to the subset of the input BoxList
 536     specified by indices
 537   Raises:
 538     ValueError: if specified field is not contained in boxlist or if the
 539       indices are not of type int32
 540   """
 541   with tf.name_scope(scope, 'Gather'):
 542     if len(indices.shape.as_list()) != 1:
 543       raise ValueError('indices should have rank 1')
 544     if indices.dtype != tf.int32 and indices.dtype != tf.int64:
 545       raise ValueError('indices should be an int32 / int64 tensor')
 546     gather_op = tf.gather
 547     if use_static_shapes:
 548       gather_op = ops.matmul_gather_on_zeroth_axis
 549     subboxlist = box_list.BoxList(gather_op(boxlist.get(), indices))
 550     if fields is None:
 551       fields = boxlist.get_extra_fields()
 552     fields += ['boxes']
 553     for field in fields:
 554       if not boxlist.has_field(field):
 555         raise ValueError('boxlist must contain all specified fields')
 556       subfieldlist = gather_op(boxlist.get_field(field), indices)
 557       subboxlist.add_field(field, subfieldlist)
 558     return subboxlist

 561 def concatenate(boxlists, fields=None, scope=None):
 562   """Concatenate list of BoxLists.
 563 
 564   This op concatenates a list of input BoxLists into a larger BoxList.  It also
 565   handles concatenation of BoxList fields as long as the field tensor shapes
 566   are equal except for the first dimension.
 567 
 568   Args:
 569     boxlists: list of BoxList objects
 570     fields: optional list of fields to also concatenate.  By default, all
 571       fields from the first BoxList in the list are included in the
 572       concatenation.
 573     scope: name scope.
 574 
 575   Returns:
 576     a BoxList with number of boxes equal to
 577       sum([boxlist.num_boxes() for boxlist in BoxList])
 578   Raises:
 579     ValueError: if boxlists is invalid (i.e., is not a list, is empty, or
 580       contains non BoxList objects), or if requested fields are not contained in
 581       all boxlists
 582   """
 583   with tf.name_scope(scope, 'Concatenate'):
 584     if not isinstance(boxlists, list):
 585       raise ValueError('boxlists should be a list')
 586     if not boxlists:
 587       raise ValueError('boxlists should have nonzero length')
 588     for boxlist in boxlists:
 589       if not isinstance(boxlist, box_list.BoxList):
 590         raise ValueError('all elements of boxlists should be BoxList objects')
 591     concatenated = box_list.BoxList(
 592         tf.concat([boxlist.get() for boxlist in boxlists], 0))
 593     if fields is None:
 594       fields = boxlists[0].get_extra_fields()
 595     for field in fields:
 596       first_field_shape = boxlists[0].get_field(field).get_shape().as_list()
 597       first_field_shape[0] = -1
 598       if None in first_field_shape:
 599         raise ValueError('field %s must have fully defined shape except for the'
 600                          ' 0th dimension.' % field)
 601       for boxlist in boxlists:
 602         if not boxlist.has_field(field):
 603           raise ValueError('boxlist must contain all requested fields')
 604         field_shape = boxlist.get_field(field).get_shape().as_list()
 605         field_shape[0] = -1
 606         if field_shape != first_field_shape:
 607           raise ValueError('field %s must have same shape for all boxlists '
 608                            'except for the 0th dimension.' % field)
 609       concatenated_field = tf.concat(
 610           [boxlist.get_field(field) for boxlist in boxlists], 0)
 611       concatenated.add_field(field, concatenated_field)
 612     return concatenated

 615 def sort_by_field(boxlist, field, order=SortOrder.descend, scope=None):
 616   """Sort boxes and associated fields according to a scalar field.
 617 
 618   A common use case is reordering the boxes according to descending scores.
 619 
 620   Args:
 621     boxlist: BoxList holding N boxes.
 622     field: A BoxList field for sorting and reordering the BoxList.
 623     order: (Optional) descend or ascend. Default is descend.
 624     scope: name scope.
 625 
 626   Returns:
 627     sorted_boxlist: A sorted BoxList with the field in the specified order.
 628 
 629   Raises:
 630     ValueError: if specified field does not exist
 631     ValueError: if the order is not either descend or ascend
 632   """
 633   with tf.name_scope(scope, 'SortByField'):
 634     if order != SortOrder.descend and order != SortOrder.ascend:
 635       raise ValueError('Invalid sort order')
 636 
 637     field_to_sort = boxlist.get_field(field)
 638     if len(field_to_sort.shape.as_list()) != 1:
 639       raise ValueError('Field should have rank 1')
 640 
 641     num_boxes = boxlist.num_boxes()
 642     num_entries = tf.size(field_to_sort)
 643     length_assert = tf.Assert(
 644         tf.equal(num_boxes, num_entries),
 645         ['Incorrect field size: actual vs expected.', num_entries, num_boxes])
 646 
 647     with tf.control_dependencies([length_assert]):
 648       _, sorted_indices = tf.nn.top_k(field_to_sort, num_boxes, sorted=True)
 649 
 650     if order == SortOrder.ascend:
 651       sorted_indices = tf.reverse_v2(sorted_indices, [0])
 652 
 653     return gather(boxlist, sorted_indices)

 749 def non_max_suppression(boxlist, thresh, max_output_size, scope=None):
 750   """Non maximum suppression.
 751 
 752   This op greedily selects a subset of detection bounding boxes, pruning
 753   away boxes that have high IOU (intersection over union) overlap (> thresh)
 754   with already selected boxes.  Note that this only works for a single class ---
 755   to apply NMS to multi-class predictions, use MultiClassNonMaxSuppression.
 756 
 757   Args:
 758     boxlist: BoxList holding N boxes.  Must contain a 'scores' field
 759       representing detection scores.
 760     thresh: scalar threshold
 761     max_output_size: maximum number of retained boxes
 762     scope: name scope.
 763 
 764   Returns:
 765     a BoxList holding M boxes where M <= max_output_size
 766   Raises:
 767     ValueError: if thresh is not in [0, 1]
 768   """
 769   with tf.name_scope(scope, 'NonMaxSuppression'):
 770     if not 0 <= thresh <= 1.0:
 771       raise ValueError('thresh must be between 0 and 1')
 772     if not isinstance(boxlist, box_list.BoxList):
 773       raise ValueError('boxlist must be a BoxList')
 774     if not boxlist.has_field('scores'):
 775       raise ValueError('input boxlist must have \'scores\' field')
 776     selected_indices = tf.image.non_max_suppression(
 777         boxlist.get(), boxlist.get_field('scores'),
 778         max_output_size, iou_threshold=thresh)
 779     return gather(boxlist, selected_indices)

 797 def to_normalized_coordinates(boxlist, height, width,
 798                               check_range=True, scope=None):
 799   """Converts absolute box coordinates to normalized coordinates in [0, 1].
 800 
 801   Usually one uses the dynamic shape of the image or conv-layer tensor:
 802     boxlist = box_list_ops.to_normalized_coordinates(boxlist,
 803                                                      tf.shape(images)[1],
 804                                                      tf.shape(images)[2]),
 805 
 806   This function raises an assertion failed error at graph execution time when
 807   the maximum coordinate is smaller than 1.01 (which means that coordinates are
 808   already normalized). The value 1.01 is to deal with small rounding errors.
 809 
 810   Args:
 811     boxlist: BoxList with coordinates in terms of pixel-locations.
 812     height: Maximum value for height of absolute box coordinates.
 813     width: Maximum value for width of absolute box coordinates.
 814     check_range: If True, checks if the coordinates are normalized or not.
 815     scope: name scope.
 816 
 817   Returns:
 818     boxlist with normalized coordinates in [0, 1].
 819   """
 820   with tf.name_scope(scope, 'ToNormalizedCoordinates'):
 821     height = tf.cast(height, tf.float32)
 822     width = tf.cast(width, tf.float32)
 823 
 824     if check_range:
 825       max_val = tf.reduce_max(boxlist.get())
 826       max_assert = tf.Assert(tf.greater(max_val, 1.01),
 827                              ['max value is lower than 1.01: ', max_val])
 828       with tf.control_dependencies([max_assert]):
 829         width = tf.identity(width)
 830 
 831     return scale(boxlist, 1 / height, 1 / width)

 876 def refine_boxes_multi_class(pool_boxes,

 878                              nms_iou_thresh,

 903     raise ValueError('nms_iou_thresh must be between 0 and 1')

 958 def box_voting(selected_boxes, pool_boxes, iou_thresh=0.5):
 959   """Performs box voting as described in S. Gidaris and N. Komodakis, ICCV 2015.
 960 
 961   Performs box voting as described in 'Object detection via a multi-region &
 962   semantic segmentation-aware CNN model', Gidaris and Komodakis, ICCV 2015. For
 963   each box 'B' in selected_boxes, we find the set 'S' of boxes in pool_boxes
 964   with iou overlap >= iou_thresh. The location of B is set to the weighted
 965   average location of boxes in S (scores are used for weighting). And the score
 966   of B is set to the average score of boxes in S.
 967 
 968   Args:
 969     selected_boxes: BoxList containing a subset of boxes in pool_boxes. These
 970       boxes are usually selected from pool_boxes using non max suppression.
 971     pool_boxes: BoxList containing a set of (possibly redundant) boxes.
 972     iou_thresh: (float scalar) iou threshold for matching boxes in
 973       selected_boxes and pool_boxes.
 974 
 975   Returns:
 976     BoxList containing averaged locations and scores for each box in
 977     selected_boxes.
 978 
 979   Raises:
 980     ValueError: if
 981       a) selected_boxes or pool_boxes is not a BoxList.
 982       b) if iou_thresh is not in [0, 1].
 983       c) pool_boxes does not have a scores field.
 984   """
 985   if not 0.0 <= iou_thresh <= 1.0:
 986     raise ValueError('iou_thresh must be between 0 and 1')
 987   if not isinstance(selected_boxes, box_list.BoxList):
 988     raise ValueError('selected_boxes must be a BoxList')
 989   if not isinstance(pool_boxes, box_list.BoxList):
 990     raise ValueError('pool_boxes must be a BoxList')
 991   if not pool_boxes.has_field('scores'):
 992     raise ValueError('pool_boxes must have a \'scores\' field')
 993 
 994   iou_ = iou(selected_boxes, pool_boxes)
 995   match_indicator = tf.cast(tf.greater(iou_, iou_thresh), dtype=tf.float32)
 996   num_matches = tf.reduce_sum(match_indicator, 1)
 997   # TODO(kbanoop): Handle the case where some boxes in selected_boxes do not
 998   # match to any boxes in pool_boxes. For such boxes without any matches, we
 999   # should return the original boxes without voting.
1000   match_assert = tf.Assert(
1001       tf.reduce_all(tf.greater(num_matches, 0)),
1002       ['Each box in selected_boxes must match with at least one box '
1003        'in pool_boxes.'])
1004 
1005   scores = tf.expand_dims(pool_boxes.get_field('scores'), 1)
1006   scores_assert = tf.Assert(
1007       tf.reduce_all(tf.greater_equal(scores, 0)),
1008       ['Scores must be non negative.'])
1009 
1010   with tf.control_dependencies([scores_assert, match_assert]):
1011     sum_scores = tf.matmul(match_indicator, scores)
1012   averaged_scores = tf.reshape(sum_scores, [-1]) / num_matches
1013 
1014   box_locations = tf.matmul(match_indicator,
1015                             pool_boxes.get() * scores) / sum_scores
1016   averaged_boxes = box_list.BoxList(box_locations)
1017   _copy_extra_fields(averaged_boxes, selected_boxes)
1018   averaged_boxes.add_field('scores', averaged_scores)
1019   return averaged_boxes

1045 def select_random_box(boxlist,
1046                       default_box=None,
1047                       seed=None,
1048                       scope=None):
1049   """Selects a random bounding box from a `BoxList`.
1050 
1051   Args:
1052     boxlist: A BoxList.

1054       this default box will be returned. If None, will use a default box of

1056     seed: Random seed.
1057     scope: Name scope.
1058 
1059   Returns:

1083 def get_minimal_coverage_box(boxlist,
1084                              default_box=None,
1085                              scope=None):
1086   """Creates a single bounding box which covers all boxes in the boxlist.
1087 
1088   Args:
1089     boxlist: A Boxlist.
1090     default_box: A [1, 4] float32 tensor. If no boxes are present in `boxlist`,
1091       this default box will be returned. If None, will use a default box of
1092       [[0., 0., 1., 1.]].
1093     scope: Name scope.
1094 
1095   Returns:
1096     A [1, 4] float32 tensor with a bounding box that tightly covers all the
1097     boxes in the box list. If the boxlist does not contain any boxes, the
1098     default box is returned.
1099   """
1100   with tf.name_scope(scope, 'CreateCoverageBox'):
1101     num_boxes = boxlist.num_boxes()
1102 
1103     def coverage_box(bboxes):
1104       y_min, x_min, y_max, x_max = tf.split(
1105           value=bboxes, num_or_size_splits=4, axis=1)
1106       y_min_coverage = tf.reduce_min(y_min, axis=0)
1107       x_min_coverage = tf.reduce_min(x_min, axis=0)
1108       y_max_coverage = tf.reduce_max(y_max, axis=0)
1109       x_max_coverage = tf.reduce_max(x_max, axis=0)
1110       return tf.stack(
1111           [y_min_coverage, x_min_coverage, y_max_coverage, x_max_coverage],
1112           axis=1)
1113 
1114     default_box = default_box or tf.constant([[0., 0., 1., 1.]])
1115     return tf.cond(
1116         tf.greater_equal(num_boxes, 1),
1117         true_fn=lambda: coverage_box(boxlist.get()),
1118         false_fn=lambda: default_box)

1121 def sample_boxes_by_jittering(boxlist,
1122                               num_boxes_to_sample,
1123                               stddev=0.1,
1124                               scope=None):
1125   """Samples num_boxes_to_sample boxes by jittering around boxlist boxes.
1126 
1127   It is possible that this function might generate boxes with size 0. The larger
1128   the stddev, this is more probable. For a small stddev of 0.1 this probability
1129   is very small.
1130 
1131   Args:
1132     boxlist: A boxlist containing N boxes in normalized coordinates.
1133     num_boxes_to_sample: A positive integer containing the number of boxes to
1134       sample.
1135     stddev: Standard deviation. This is used to draw random offsets for the
1136       box corners from a normal distribution. The offset is multiplied by the
1137       box size so will be larger in terms of pixels for larger boxes.
1138     scope: Name scope.
1139 
1140   Returns:
1141     sampled_boxlist: A boxlist containing num_boxes_to_sample boxes in
1142       normalized coordinates.
1143   """
1144   with tf.name_scope(scope, 'SampleBoxesByJittering'):
1145     num_boxes = boxlist.num_boxes()
1146     box_indices = tf.random_uniform(
1147         [num_boxes_to_sample],
1148         minval=0,
1149         maxval=num_boxes,
1150         dtype=tf.int32)
1151     sampled_boxes = tf.gather(boxlist.get(), box_indices)
1152     sampled_boxes_height = sampled_boxes[:, 2] - sampled_boxes[:, 0]
1153     sampled_boxes_width = sampled_boxes[:, 3] - sampled_boxes[:, 1]
1154     rand_miny_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1155     rand_minx_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1156     rand_maxy_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1157     rand_maxx_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1158     miny = rand_miny_gaussian * sampled_boxes_height + sampled_boxes[:, 0]
1159     minx = rand_minx_gaussian * sampled_boxes_width + sampled_boxes[:, 1]
1160     maxy = rand_maxy_gaussian * sampled_boxes_height + sampled_boxes[:, 2]
1161     maxx = rand_maxx_gaussian * sampled_boxes_width + sampled_boxes[:, 3]
1162     maxy = tf.maximum(miny, maxy)
1163     maxx = tf.maximum(minx, maxx)
1164     sampled_boxes = tf.stack([miny, minx, maxy, maxx], axis=1)
1165     sampled_boxes = tf.maximum(tf.minimum(sampled_boxes, 1.0), 0.0)
1166     return box_list.BoxList(sampled_boxes)
