  15 """Records previous preprocessing operations and allows them to be repeated.
  16 
  17 Used with object_detection.core.preprocessor. Passing a PreprocessorCache
  18 into individual data augmentation functions or the general preprocess() function
  19 will store all randomly generated variables in the PreprocessorCache. When
  20 a preprocessor function is called multiple times with the same
  21 PreprocessorCache object, that function will perform the same augmentation
  22 on all calls.
  23 """
  24 
  25 import collections
  26 
  27 
  28 class PreprocessorCache(object):
  29   """Dictionary wrapper storing random variables generated during preprocessing.
  30   """
  31 
  32   # Constant keys representing different preprocessing functions
  33   ROTATION90 = 'rotation90'
  34   HORIZONTAL_FLIP = 'horizontal_flip'
  35   VERTICAL_FLIP = 'vertical_flip'
  36   PIXEL_VALUE_SCALE = 'pixel_value_scale'
  37   IMAGE_SCALE = 'image_scale'
  38   RGB_TO_GRAY = 'rgb_to_gray'
  39   ADJUST_BRIGHTNESS = 'adjust_brightness'
  40   ADJUST_CONTRAST = 'adjust_contrast'
  41   ADJUST_HUE = 'adjust_hue'
  42   ADJUST_SATURATION = 'adjust_saturation'
  43   DISTORT_COLOR = 'distort_color'
  44   STRICT_CROP_IMAGE = 'strict_crop_image'
  45   CROP_IMAGE = 'crop_image'
  46   PAD_IMAGE = 'pad_image'
  47   CROP_TO_ASPECT_RATIO = 'crop_to_aspect_ratio'
  48   RESIZE_METHOD = 'resize_method'
  49   PAD_TO_ASPECT_RATIO = 'pad_to_aspect_ratio'
  50   BLACK_PATCHES = 'black_patches'
  51   ADD_BLACK_PATCH = 'add_black_patch'
  52   SELECTOR = 'selector'
  53   SELECTOR_TUPLES = 'selector_tuples'
  54   SELF_CONCAT_IMAGE = 'self_concat_image'
  55   SSD_CROP_SELECTOR_ID = 'ssd_crop_selector_id'
  56   SSD_CROP_PAD_SELECTOR_ID = 'ssd_crop_pad_selector_id'
  57   JPEG_QUALITY = 'jpeg_quality'
  58   DOWNSCALE_TO_TARGET_PIXELS = 'downscale_to_target_pixels'
  59   PATCH_GAUSSIAN = 'patch_gaussian'
  60   SQUARE_CROP_BY_SCALE = 'square_crop_scale'
  61 
  62   # 27 permitted function ids
  63   _VALID_FNS = [ROTATION90, HORIZONTAL_FLIP, VERTICAL_FLIP, PIXEL_VALUE_SCALE,
  64                 IMAGE_SCALE, RGB_TO_GRAY, ADJUST_BRIGHTNESS, ADJUST_CONTRAST,
  65                 ADJUST_HUE, ADJUST_SATURATION, DISTORT_COLOR, STRICT_CROP_IMAGE,
  66                 CROP_IMAGE, PAD_IMAGE, CROP_TO_ASPECT_RATIO, RESIZE_METHOD,
  67                 PAD_TO_ASPECT_RATIO, BLACK_PATCHES, ADD_BLACK_PATCH, SELECTOR,
  68                 SELECTOR_TUPLES, SELF_CONCAT_IMAGE, SSD_CROP_SELECTOR_ID,
  69                 SSD_CROP_PAD_SELECTOR_ID, JPEG_QUALITY,
  70                 DOWNSCALE_TO_TARGET_PIXELS, PATCH_GAUSSIAN,
  71                 SQUARE_CROP_BY_SCALE]
  72 
  73   def __init__(self):
  74     self._history = collections.defaultdict(dict)
  75 
  76   def clear(self):
  77     """Resets cache."""
  78     self._history = collections.defaultdict(dict)
  79 
  80   def get(self, function_id, key):
  81     """Gets stored value given a function id and key.
  82 
  83     Args:
  84       function_id: identifier for the preprocessing function used.
  85       key: identifier for the variable stored.
  86     Returns:
  87       value: the corresponding value, expected to be a tensor or
  88              nested structure of tensors.
  89     Raises:
  90       ValueError: if function_id is not one of the 23 valid function ids.
  91     """
  92     if function_id not in self._VALID_FNS:
  93       raise ValueError('Function id not recognized: %s.' % str(function_id))
  94     return self._history[function_id].get(key)
  95 
  96   def update(self, function_id, key, value):
  97     """Adds a value to the dictionary.
  98 
  99     Args:
 100       function_id: identifier for the preprocessing function used.
 101       key: identifier for the variable stored.
 102       value: the value to store, expected to be a tensor or nested structure
 103              of tensors.
 104     Raises:
 105       ValueError: if function_id is not one of the 23 valid function ids.
 106     """
 107     if function_id not in self._VALID_FNS:
 108       raise ValueError('Function id not recognized: %s.' % str(function_id))
 109     self._history[function_id][key] = value
