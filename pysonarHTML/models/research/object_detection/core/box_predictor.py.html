  16 """Box predictor for object detectors.
  17 
  18 Box predictors are classes that take a high level
  19 image feature map as input and produce two predictions,
  20 (1) a tensor encoding box locations, and
  21 (2) a tensor encoding classes for each box.
  22 
  23 These components are passed directly to loss functions
  24 in our detection models.
  25 
  26 These modules are separated from the main model since the same
  27 few box predictor architectures are shared across many models.
  28 """
  29 from abc import abstractmethod
  30 import tensorflow.compat.v1 as tf
  31 
  32 BOX_ENCODINGS = 'box_encodings'
  33 CLASS_PREDICTIONS_WITH_BACKGROUND = 'class_predictions_with_background'
  34 MASK_PREDICTIONS = 'mask_predictions'

  37 class BoxPredictor(object):
  38   """BoxPredictor."""
  39 
  40   def __init__(self, is_training, num_classes):
  41     """Constructor.
  42 
  43     Args:
  44       is_training: Indicates whether the BoxPredictor is in training mode.
  45       num_classes: number of classes.  Note that num_classes *does not*
  46         include the background category, so if groundtruth labels take values
  47         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  48         assigned classification targets can range from {0,... K}).
  49     """
  50     self._is_training = is_training
  51     self._num_classes = num_classes
  52 
  53   @property
  54   def is_keras_model(self):
  55     return False
  56 
  57   @property
  58   def num_classes(self):
  59     return self._num_classes
  60 
  61   def predict(self, image_features, num_predictions_per_location,
  62               scope=None, **params):
  63     """Computes encoded object locations and corresponding confidences.
  64 
  65     Takes a list of high level image feature maps as input and produces a list
  66     of box encodings and a list of class scores where each element in the output
  67     lists correspond to the feature maps in the input list.
  68 
  69     Args:
  70       image_features: A list of float tensors of shape [batch_size, height_i,
  71       width_i, channels_i] containing features for a batch of images.
  72       num_predictions_per_location: A list of integers representing the number
  73         of box predictions to be made per spatial location for each feature map.
  74       scope: Variable and Op scope name.
  75       **params: Additional keyword arguments for specific implementations of
  76               BoxPredictor.
  77 
  78     Returns:
  79       A dictionary containing at least the following tensors.
  80         box_encodings: A list of float tensors. Each entry in the list
  81           corresponds to a feature map in the input `image_features` list. All
  82           tensors in the list have one of the two following shapes:
  83           a. [batch_size, num_anchors_i, q, code_size] representing the location
  84             of the objects, where q is 1 or the number of classes.
  85           b. [batch_size, num_anchors_i, code_size].
  86         class_predictions_with_background: A list of float tensors of shape
  87           [batch_size, num_anchors_i, num_classes + 1] representing the class
  88           predictions for the proposals. Each entry in the list corresponds to a
  89           feature map in the input `image_features` list.
  90 
  91     Raises:
  92       ValueError: If length of `image_features` is not equal to length of
  93         `num_predictions_per_location`.
  94     """
  95     if len(image_features) != len(num_predictions_per_location):
  96       raise ValueError('image_feature and num_predictions_per_location must '
  97                        'be of same length, found: {} vs {}'.
  98                        format(len(image_features),
  99                               len(num_predictions_per_location)))
 100     if scope is not None:
 101       with tf.variable_scope(scope):
 102         return self._predict(image_features, num_predictions_per_location,
 103                              **params)
 104     return self._predict(image_features, num_predictions_per_location,
 105                          **params)
 106 
 107   # TODO(rathodv): num_predictions_per_location could be moved to constructor.
 108   # This is currently only used by ConvolutionalBoxPredictor.
 109   @abstractmethod
 110   def _predict(self, image_features, num_predictions_per_location, **params):
 111     """Implementations must override this method.
 112 
 113     Args:
 114       image_features: A list of float tensors of shape [batch_size, height_i,
 115         width_i, channels_i] containing features for a batch of images.
 116       num_predictions_per_location: A list of integers representing the number
 117         of box predictions to be made per spatial location for each feature map.
 118       **params: Additional keyword arguments for specific implementations of
 119               BoxPredictor.
 120 
 121     Returns:
 122       A dictionary containing at least the following tensors.
 123         box_encodings: A list of float tensors. Each entry in the list
 124           corresponds to a feature map in the input `image_features` list. All
 125           tensors in the list have one of the two following shapes:
 126           a. [batch_size, num_anchors_i, q, code_size] representing the location
 127             of the objects, where q is 1 or the number of classes.
 128           b. [batch_size, num_anchors_i, code_size].
 129         class_predictions_with_background: A list of float tensors of shape
 130           [batch_size, num_anchors_i, num_classes + 1] representing the class
 131           predictions for the proposals. Each entry in the list corresponds to a
 132           feature map in the input `image_features` list.
 133     """
 134     pass

 137 class KerasBoxPredictor(tf.keras.Model):
 138   """Keras-based BoxPredictor."""
 139 
 140   def __init__(self, is_training, num_classes, freeze_batchnorm,
 141                inplace_batchnorm_update, name=None):
 142     """Constructor.
 143 
 144     Args:
 145       is_training: Indicates whether the BoxPredictor is in training mode.
 146       num_classes: number of classes.  Note that num_classes *does not*
 147         include the background category, so if groundtruth labels take values
 148         in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 149         assigned classification targets can range from {0,... K}).
 150       freeze_batchnorm: Whether to freeze batch norm parameters during
 151         training or not. When training with a small batch size (e.g. 1), it is
 152         desirable to freeze batch norm update and use pretrained batch norm
 153         params.
 154       inplace_batchnorm_update: Whether to update batch norm moving average
 155         values inplace. When this is false train op must add a control
 156         dependency on tf.graphkeys.UPDATE_OPS collection in order to update
 157         batch norm statistics.
 158       name: A string name scope to assign to the model. If `None`, Keras
 159         will auto-generate one from the class name.
 160     """
 161     super(KerasBoxPredictor, self).__init__(name=name)
 162 
 163     self._is_training = is_training
 164     self._num_classes = num_classes
 165     self._freeze_batchnorm = freeze_batchnorm
 166     self._inplace_batchnorm_update = inplace_batchnorm_update
 167 
 168   @property
 169   def is_keras_model(self):
 170     return True
 171 
 172   @property
 173   def num_classes(self):
 174     return self._num_classes

 186       **kwargs: Additional keyword arguments for specific implementations of

 193           tensors in the list have one of the two following shapes:
