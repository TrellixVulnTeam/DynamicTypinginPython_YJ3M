  31 class PrefetcherTest(tf.test.TestCase):
  32   """Test class for prefetcher."""
  33 
  34   def test_prefetch_tensors_with_fully_defined_shapes(self):
  35     with self.test_session() as sess:
  36       batch_size = 10
  37       image_size = 32
  38       num_batches = 5
  39       examples = tf.Variable(tf.constant(0, dtype=tf.int64))
  40       counter = examples.count_up_to(num_batches)
  41       image = tf.random_normal([batch_size, image_size,
  42                                 image_size, 3],
  43                                dtype=tf.float32,
  44                                name='images')
  45       label = tf.random_uniform([batch_size, 1], 0, 10,
  46                                 dtype=tf.int32, name='labels')
  47 
  48       prefetch_queue = prefetcher.prefetch(tensor_dict={'counter': counter,
  49                                                         'image': image,
  50                                                         'label': label},
  51                                            capacity=100)
  52       tensor_dict = prefetch_queue.dequeue()
  53 
  54       self.assertAllEqual(tensor_dict['image'].get_shape().as_list(),
  55                           [batch_size, image_size, image_size, 3])
  56       self.assertAllEqual(tensor_dict['label'].get_shape().as_list(),
  57                           [batch_size, 1])
  58 
  59       tf.initialize_all_variables().run()
  60       with slim.queues.QueueRunners(sess):
  61         for _ in range(num_batches):
  62           results = sess.run(tensor_dict)
  63           self.assertEquals(results['image'].shape,
  64                             (batch_size, image_size, image_size, 3))
  65           self.assertEquals(results['label'].shape, (batch_size, 1))
  66         with self.assertRaises(tf.errors.OutOfRangeError):
  67           sess.run(tensor_dict)
  68 
  69   def test_prefetch_tensors_with_partially_defined_shapes(self):
  70     with self.test_session() as sess:
  71       batch_size = 10
  72       image_size = 32
  73       num_batches = 5
  74       examples = tf.Variable(tf.constant(0, dtype=tf.int64))
  75       counter = examples.count_up_to(num_batches)
  76       image = tf.random_normal([batch_size,
  77                                 tf.Variable(image_size),
  78                                 tf.Variable(image_size), 3],
  79                                dtype=tf.float32,
  80                                name='image')
  81       image.set_shape([batch_size, None, None, 3])
  82       label = tf.random_uniform([batch_size, tf.Variable(1)], 0,
  83                                 10, dtype=tf.int32, name='label')
  84       label.set_shape([batch_size, None])
  85 
  86       prefetch_queue = prefetcher.prefetch(tensor_dict={'counter': counter,
  87                                                         'image': image,
  88                                                         'label': label},
  89                                            capacity=100)
  90       tensor_dict = prefetch_queue.dequeue()
  91 
  92       self.assertAllEqual(tensor_dict['image'].get_shape().as_list(),
  93                           [batch_size, None, None, 3])
  94       self.assertAllEqual(tensor_dict['label'].get_shape().as_list(),
  95                           [batch_size, None])
  96 
  97       tf.initialize_all_variables().run()
  98       with slim.queues.QueueRunners(sess):
  99         for _ in range(num_batches):
 100           results = sess.run(tensor_dict)
 101           self.assertEquals(results['image'].shape,
 102                             (batch_size, image_size, image_size, 3))
 103           self.assertEquals(results['label'].shape, (batch_size, 1))
 104         with self.assertRaises(tf.errors.OutOfRangeError):
 105           sess.run(tensor_dict)
