  36   def test_get_correct_counts(self):
  37     def graph_fn():
  38       match_results = tf.constant([3, 1, -1, 0, -1, 1, -2])
  39       match = matcher.Match(match_results)
  40       num_matched_columns = match.num_matched_columns()
  41       num_unmatched_columns = match.num_unmatched_columns()
  42       num_ignored_columns = match.num_ignored_columns()
  43       num_matched_rows = match.num_matched_rows()
  44       return [num_matched_columns, num_unmatched_columns, num_ignored_columns,
  45               num_matched_rows]
  46     (num_matched_columns_out, num_unmatched_columns_out,
  47      num_ignored_columns_out,
  48      num_matched_rows_out) = self.execute_cpu(graph_fn, [])
  49     exp_num_matched_columns = 4
  50     exp_num_unmatched_columns = 2
  51     exp_num_ignored_columns = 1
  52     exp_num_matched_rows = 3
  53     self.assertAllEqual(num_matched_columns_out, exp_num_matched_columns)
  54     self.assertAllEqual(num_unmatched_columns_out, exp_num_unmatched_columns)
  55     self.assertAllEqual(num_ignored_columns_out, exp_num_ignored_columns)
  56     self.assertAllEqual(num_matched_rows_out, exp_num_matched_rows)

 130   def test_all_columns_accounted_for(self):
 131     # Note: deliberately setting to small number so not always
 132     # all possibilities appear (matched, unmatched, ignored)
 133     def graph_fn():
 134       match_results = tf.random_uniform(
 135           [num_matches], minval=-2, maxval=5, dtype=tf.int32)
 136       match = matcher.Match(match_results)
 137       matched_column_indices = match.matched_column_indices()
 138       unmatched_column_indices = match.unmatched_column_indices()
 139       ignored_column_indices = match.ignored_column_indices()
 140       return (matched_column_indices, unmatched_column_indices,
 141               ignored_column_indices)
 142     num_matches = 10
 143     matched, unmatched, ignored = self.execute(graph_fn, [])
 144     all_indices = np.hstack((matched, unmatched, ignored))
 145     all_indices_sorted = np.sort(all_indices)
 146     self.assertAllEqual(all_indices_sorted,
 147                         np.arange(num_matches, dtype=np.int32))

 149   def test_scalar_gather_based_on_match(self):
 150     def graph_fn():
 151       match_results = tf.constant([3, 1, -1, 0, -1, 5, -2])
 152       input_tensor = tf.constant([0, 1, 2, 3, 4, 5, 6, 7], dtype=tf.float32)
 153       match = matcher.Match(match_results)
 154       gathered_tensor = match.gather_based_on_match(input_tensor,
 155                                                     unmatched_value=100.,
 156                                                     ignored_value=200.)
 157       return gathered_tensor
 158     expected_gathered_tensor = [3, 1, 100, 0, 100, 5, 200]
 159     gathered_tensor_out = self.execute(graph_fn, [])
 160     self.assertAllEqual(expected_gathered_tensor, gathered_tensor_out)
