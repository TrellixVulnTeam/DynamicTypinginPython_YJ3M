  27 class FakeModel(model.DetectionModel):
  28 
  29   def __init__(self):
  30 
  31     # sub-networks containing weights of different shapes.
  32     self._network1 = tf.keras.Sequential([
  33         tf.keras.layers.Conv2D(8, 1)
  34     ])
  35 
  36     self._network2 = tf.keras.Sequential([
  37         tf.keras.layers.Conv2D(16, 1)
  38     ])
  39 
  40     super(FakeModel, self).__init__(num_classes=0)
  41 
  42   def preprocess(self, images):
  43     return images, tf.shape(images)
  44 
  45   def predict(self, images, shapes):
  46     return {'prediction': self._network2(self._network1(images))}
  47 
  48   def postprocess(self, prediction_dict, shapes):
  49     return prediction_dict
  50 
  51   def loss(self):
  52     return tf.constant(0.0)
  53 
  54   def updates(self):
  55     return []
  56 
  57   def restore_map(self):
  58     return {}
  59 
  60   def regularization_losses(self):
  61     return []
  62 
  63 
  64 class ModelTest(test_case.TestCase):
  65 
  66   def test_model_call(self):
  67 
  68     detection_model = FakeModel()
  69 
  70     def graph_fn():
  71       return detection_model(tf.zeros((1, 128, 128, 3)))
  72 
  73     result = self.execute(graph_fn, [])
  74     self.assertEqual(result['prediction'].shape,
  75                      (1, 128, 128, 16))
  76 
  77   def test_freeze(self):
  78 
  79     detection_model = FakeModel()
  80     detection_model(tf.zeros((1, 128, 128, 3)))
  81 
  82     net1_var_shapes = [tuple(var.get_shape().as_list()) for var in
  83                        detection_model._network1.trainable_variables]
  84 
  85     del detection_model
  86 
  87     detection_model = FakeModel()
  88     detection_model._network2.trainable = False
  89     detection_model(tf.zeros((1, 128, 128, 3)))
  90 
  91     var_shapes = [tuple(var.get_shape().as_list()) for var in
  92                   detection_model._network1.trainable_variables]
  93 
  94     self.assertEqual(set(net1_var_shapes), set(var_shapes))
