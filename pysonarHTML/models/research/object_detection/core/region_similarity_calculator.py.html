  16 """Region Similarity Calculators for BoxLists.
  17 
  18 Region Similarity Calculators compare a pairwise measure of similarity
  19 between the boxes in two BoxLists.
  20 """
  21 from __future__ import absolute_import
  22 from __future__ import division
  23 from __future__ import print_function
  24 
  25 from abc import ABCMeta
  26 from abc import abstractmethod
  27 
  28 import six
  29 import tensorflow.compat.v1 as tf
  30 
  31 from object_detection.core import box_list_ops
  32 from object_detection.core import standard_fields as fields
  33 
  34 
  35 class RegionSimilarityCalculator(six.with_metaclass(ABCMeta, object)):
  36   """Abstract base class for region similarity calculator."""
  37 
  38   def compare(self, boxlist1, boxlist2, scope=None):
  39     """Computes matrix of pairwise similarity between BoxLists.
  40 
  41     This op (to be overridden) computes a measure of pairwise similarity between
  42     the boxes in the given BoxLists. Higher values indicate more similarity.
  43 
  44     Note that this method simply measures similarity and does not explicitly
  45     perform a matching.
  46 
  47     Args:
  48       boxlist1: BoxList holding N boxes.
  49       boxlist2: BoxList holding M boxes.
  50       scope: Op scope name. Defaults to 'Compare' if None.
  51 
  52     Returns:
  53       a (float32) tensor of shape [N, M] with pairwise similarity score.
  54     """
  55     with tf.name_scope(scope, 'Compare', [boxlist1, boxlist2]) as scope:
  56       return self._compare(boxlist1, boxlist2)
  57 
  58   @abstractmethod
  59   def _compare(self, boxlist1, boxlist2):
  60     pass
  61

  63 class IouSimilarity(RegionSimilarityCalculator):
  64   """Class to compute similarity based on Intersection over Union (IOU) metric.
  65 
  66   This class computes pairwise similarity between two BoxLists based on IOU.
  67   """
  68 
  69   def _compare(self, boxlist1, boxlist2):
  70     """Compute pairwise IOU similarity between the two BoxLists.
  71 
  72     Args:
  73       boxlist1: BoxList holding N boxes.
  74       boxlist2: BoxList holding M boxes.
  75 
  76     Returns:
  77       A tensor with shape [N, M] representing pairwise iou scores.
  78     """
  79     return box_list_ops.iou(boxlist1, boxlist2)

  82 class NegSqDistSimilarity(RegionSimilarityCalculator):
  83   """Class to compute similarity based on the squared distance metric.
  84 
  85   This class computes pairwise similarity between two BoxLists based on the
  86   negative squared distance metric.
  87   """
  88 
  89   def _compare(self, boxlist1, boxlist2):
  90     """Compute matrix of (negated) sq distances.
  91 
  92     Args:
  93       boxlist1: BoxList holding N boxes.
  94       boxlist2: BoxList holding M boxes.
  95 
  96     Returns:
  97       A tensor with shape [N, M] representing negated pairwise squared distance.
  98     """
  99     return -1 * box_list_ops.sq_dist(boxlist1, boxlist2)

 102 class IoaSimilarity(RegionSimilarityCalculator):
 103   """Class to compute similarity based on Intersection over Area (IOA) metric.
 104 
 105   This class computes pairwise similarity between two BoxLists based on their
 106   pairwise intersections divided by the areas of second BoxLists.
 107   """
 108 
 109   def _compare(self, boxlist1, boxlist2):
 110     """Compute pairwise IOA similarity between the two BoxLists.
 111 
 112     Args:
 113       boxlist1: BoxList holding N boxes.
 114       boxlist2: BoxList holding M boxes.
 115 
 116     Returns:
 117       A tensor with shape [N, M] representing pairwise IOA scores.
 118     """
 119     return box_list_ops.ioa(boxlist1, boxlist2)

 122 class ThresholdedIouSimilarity(RegionSimilarityCalculator):
 123   """Class to compute similarity based on thresholded IOU and score.
 124 
 125   This class computes pairwise similarity between two BoxLists based on IOU and
 126   a 'score' present in boxlist1. If IOU > threshold, then the entry in the
 127   output pairwise tensor will contain `score`, otherwise 0.
 128   """
 129 
 130   def __init__(self, iou_threshold=0):
 131     """Initialize the ThresholdedIouSimilarity.
 132 
 133     Args:
 134       iou_threshold: For a given pair of boxes, if the IOU is > iou_threshold,
 135         then the comparison result will be the foreground probability of
 136         the first box, otherwise it will be zero.
 137     """
 138     super(ThresholdedIouSimilarity, self).__init__()
 139     self._iou_threshold = iou_threshold
 140 
 141   def _compare(self, boxlist1, boxlist2):
 142     """Compute pairwise IOU similarity between the two BoxLists and score.
 143 
 144     Args:
 145       boxlist1: BoxList holding N boxes. Must have a score field.
 146       boxlist2: BoxList holding M boxes.
 147 
 148     Returns:
 149       A tensor with shape [N, M] representing scores threholded by pairwise
 150       iou scores.
 151     """
 152     ious = box_list_ops.iou(boxlist1, boxlist2)
 153     scores = boxlist1.get_field(fields.BoxListFields.scores)
 154     scores = tf.expand_dims(scores, axis=1)
 155     row_replicated_scores = tf.tile(scores, [1, tf.shape(ious)[-1]])
 156     thresholded_ious = tf.where(ious > self._iou_threshold,
 157                                 row_replicated_scores, tf.zeros_like(ious))
 158 
 159     return thresholded_ious
