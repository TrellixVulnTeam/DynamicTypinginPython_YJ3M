  32   def test_batch_and_unpad_2d_tensors_of_different_sizes_in_1st_dimension(self):
  33     with self.test_session() as sess:
  34       batch_size = 3
  35       num_batches = 2
  36       examples = tf.Variable(tf.constant(2, dtype=tf.int32))
  37       counter = examples.count_up_to(num_batches * batch_size + 2)
  38       boxes = tf.tile(
  39           tf.reshape(tf.range(4), [1, 4]), tf.stack([counter, tf.constant(1)]))
  40       batch_queue = batcher.BatchQueue(
  41           tensor_dict={'boxes': boxes},
  42           batch_size=batch_size,
  43           batch_queue_capacity=100,
  44           num_batch_queue_threads=1,
  45           prefetch_queue_capacity=100)
  46       batch = batch_queue.dequeue()
  47 
  48       for tensor_dict in batch:
  49         for tensor in tensor_dict.values():
  50           self.assertAllEqual([None, 4], tensor.get_shape().as_list())
  51 
  52       tf.initialize_all_variables().run()
  53       with slim.queues.QueueRunners(sess):
  54         i = 2
  55         for _ in range(num_batches):
  56           batch_np = sess.run(batch)
  57           for tensor_dict in batch_np:
  58             for tensor in tensor_dict.values():
  59               self.assertAllEqual(tensor, np.tile(np.arange(4), (i, 1)))
  60               i += 1
  61         with self.assertRaises(tf.errors.OutOfRangeError):
  62           sess.run(batch)
