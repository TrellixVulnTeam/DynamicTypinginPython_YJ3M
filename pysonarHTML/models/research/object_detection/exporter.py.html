  42 def rewrite_nn_resize_op(is_quantized=False):
  43   """Replaces a custom nearest-neighbor resize op with the Tensorflow version.
  44 
  45   Some graphs use this custom version for TPU-compatibility.
  46 
  47   Args:
  48     is_quantized: True if the default graph is quantized.
  49   """
  50   def remove_nn():
  51     """Remove nearest neighbor upsampling structures and replace with TF op."""
  52     input_pattern = graph_matcher.OpTypePattern(
  53         'FakeQuantWithMinMaxVars' if is_quantized else '*')
  54     stack_1_pattern = graph_matcher.OpTypePattern(
  55         'Pack', inputs=[input_pattern, input_pattern], ordered_inputs=False)
  56     stack_2_pattern = graph_matcher.OpTypePattern(
  57         'Pack', inputs=[stack_1_pattern, stack_1_pattern], ordered_inputs=False)
  58     reshape_pattern = graph_matcher.OpTypePattern(
  59         'Reshape', inputs=[stack_2_pattern, 'Const'], ordered_inputs=False)
  60     consumer_pattern1 = graph_matcher.OpTypePattern(
  61         'Add|AddV2|Max|Mul', inputs=[reshape_pattern, '*'],
  62         ordered_inputs=False)
  63     consumer_pattern2 = graph_matcher.OpTypePattern(
  64         'StridedSlice', inputs=[reshape_pattern, '*', '*', '*'],
  65         ordered_inputs=False)
  66 
  67     def replace_matches(consumer_pattern):
  68       """Search for nearest neighbor pattern and replace with TF op."""
  69       match_counter = 0
  70       matcher = graph_matcher.GraphMatcher(consumer_pattern)
  71       for match in matcher.match_graph(tf.get_default_graph()):
  72         match_counter += 1
  73         projection_op = match.get_op(input_pattern)
  74         reshape_op = match.get_op(reshape_pattern)
  75         consumer_op = match.get_op(consumer_pattern)
  76         nn_resize = tf.image.resize_nearest_neighbor(
  77             projection_op.outputs[0],
  78             reshape_op.outputs[0].shape.dims[1:3],
  79             align_corners=False,
  80             name=os.path.split(reshape_op.name)[0] + '/resize_nearest_neighbor')
  81 
  82         for index, op_input in enumerate(consumer_op.inputs):
  83           if op_input == reshape_op.outputs[0]:
  84             consumer_op._update_input(index, nn_resize)  # pylint: disable=protected-access
  85             break
  86 
  87       return match_counter
  88 
  89     match_counter = replace_matches(consumer_pattern1)
  90     match_counter += replace_matches(consumer_pattern2)
  91 
  92     tf.logging.info('Found and fixed {} matches'.format(match_counter))
  93     return match_counter
  94 
  95   # Applying twice because both inputs to Add could be NN pattern
  96   total_removals = 0
  97   while remove_nn():
  98     total_removals += 1
  99     # This number is chosen based on the nas-fpn architecture.
 100     if total_removals > 4:
 101       raise ValueError('Graph removal encountered a infinite loop.')

 104 def replace_variable_values_with_moving_averages(graph,
 105                                                  current_checkpoint_file,
 106                                                  new_checkpoint_file,
 107                                                  no_ema_collection=None):
 108   """Replaces variable values in the checkpoint with their moving averages.
 109 
 110   If the current checkpoint has shadow variables maintaining moving averages of
 111   the variables defined in the graph, this function generates a new checkpoint
 112   where the variables contain the values of their moving averages.
 113 
 114   Args:
 115     graph: a tf.Graph object.
 116     current_checkpoint_file: a checkpoint containing both original variables and
 117       their moving averages.
 118     new_checkpoint_file: file path to write a new checkpoint.
 119     no_ema_collection: A list of namescope substrings to match the variables
 120       to eliminate EMA.
 121   """
 122   with graph.as_default():
 123     variable_averages = tf.train.ExponentialMovingAverage(0.0)
 124     ema_variables_to_restore = variable_averages.variables_to_restore()
 125     ema_variables_to_restore = config_util.remove_unecessary_ema(
 126         ema_variables_to_restore, no_ema_collection)
 127     with tf.Session() as sess:
 128       read_saver = tf.train.Saver(ema_variables_to_restore)
 129       read_saver.restore(sess, current_checkpoint_file)
 130       write_saver = tf.train.Saver()
 131       write_saver.save(sess, new_checkpoint_file)

 134 def _image_tensor_input_placeholder(input_shape=None):
 135   """Returns input placeholder and a 4-D uint8 image tensor."""
 136   if input_shape is None:
 137     input_shape = (None, None, None, 3)
 138   input_tensor = tf.placeholder(
 139       dtype=tf.uint8, shape=input_shape, name='image_tensor')
 140   return input_tensor, input_tensor

 143 def _tf_example_input_placeholder(input_shape=None):
 144   """Returns input that accepts a batch of strings with tf examples.
 145 
 146   Args:
 147     input_shape: the shape to resize the output decoded images to (optional).
 148 
 149   Returns:
 150     a tuple of input placeholder and the output decoded images.
 151   """
 152   batch_tf_example_placeholder = tf.placeholder(
 153       tf.string, shape=[None], name='tf_example')
 154   def decode(tf_example_string_tensor):
 155     tensor_dict = tf_example_decoder.TfExampleDecoder().decode(
 156         tf_example_string_tensor)
 157     image_tensor = tensor_dict[fields.InputDataFields.image]
 158     if input_shape is not None:
 159       image_tensor = tf.image.resize(image_tensor, input_shape[1:3])
 160     return image_tensor
 161   return (batch_tf_example_placeholder,
 162           shape_utils.static_or_dynamic_map_fn(
 163               decode,
 164               elems=batch_tf_example_placeholder,
 165               dtype=tf.uint8,
 166               parallel_iterations=32,
 167               back_prop=False))

 170 def _encoded_image_string_tensor_input_placeholder(input_shape=None):
 171   """Returns input that accepts a batch of PNG or JPEG strings.
 172 
 173   Args:
 174     input_shape: the shape to resize the output decoded images to (optional).
 175 
 176   Returns:
 177     a tuple of input placeholder and the output decoded images.
 178   """
 179   batch_image_str_placeholder = tf.placeholder(
 180       dtype=tf.string,
 181       shape=[None],
 182       name='encoded_image_string_tensor')
 183   def decode(encoded_image_string_tensor):
 184     image_tensor = tf.image.decode_image(encoded_image_string_tensor,
 185                                          channels=3)
 186     image_tensor.set_shape((None, None, 3))
 187     if input_shape is not None:
 188       image_tensor = tf.image.resize(image_tensor, input_shape[1:3])
 189     return image_tensor
 190   return (batch_image_str_placeholder,
 191           tf.map_fn(
 192               decode,
 193               elems=batch_image_str_placeholder,
 194               dtype=tf.uint8,
 195               parallel_iterations=32,
 196               back_prop=False))
 197

 199 input_placeholder_fn_map = {
 200     'image_tensor': _image_tensor_input_placeholder,
 201     'encoded_image_string_tensor':
 202     _encoded_image_string_tensor_input_placeholder,
 203     'tf_example': _tf_example_input_placeholder,
 204 }

 207 def add_output_tensor_nodes(postprocessed_tensors,
 208                             output_collection_name='inference_op'):
 209   """Adds output nodes for detection boxes and scores.
 210 
 211   Adds the following nodes for output tensors -
 212     * num_detections: float32 tensor of shape [batch_size].
 213     * detection_boxes: float32 tensor of shape [batch_size, num_boxes, 4]
 214       containing detected boxes.
 215     * detection_scores: float32 tensor of shape [batch_size, num_boxes]
 216       containing scores for the detected boxes.
 217     * detection_multiclass_scores: (Optional) float32 tensor of shape
 218       [batch_size, num_boxes, num_classes_with_background] for containing class
 219       score distribution for detected boxes including background if any.
 220     * detection_features: (Optional) float32 tensor of shape
 221       [batch, num_boxes, roi_height, roi_width, depth]
 222       containing classifier features
 223       for each detected box
 224     * detection_classes: float32 tensor of shape [batch_size, num_boxes]
 225       containing class predictions for the detected boxes.
 226     * detection_keypoints: (Optional) float32 tensor of shape
 227       [batch_size, num_boxes, num_keypoints, 2] containing keypoints for each
 228       detection box.
 229     * detection_masks: (Optional) float32 tensor of shape
 230       [batch_size, num_boxes, mask_height, mask_width] containing masks for each
 231       detection box.
 232 
 233   Args:
 234     postprocessed_tensors: a dictionary containing the following fields
 235       'detection_boxes': [batch, max_detections, 4]
 236       'detection_scores': [batch, max_detections]
 237       'detection_multiclass_scores': [batch, max_detections,
 238         num_classes_with_background]
 239       'detection_features': [batch, num_boxes, roi_height, roi_width, depth]
 240       'detection_classes': [batch, max_detections]
 241       'detection_masks': [batch, max_detections, mask_height, mask_width]
 242         (optional).
 243       'detection_keypoints': [batch, max_detections, num_keypoints, 2]
 244         (optional).
 245       'num_detections': [batch]
 246     output_collection_name: Name of collection to add output tensors to.
 247 
 248   Returns:
 249     A tensor dict containing the added output tensor nodes.
 250   """
 251   detection_fields = fields.DetectionResultFields
 252   label_id_offset = 1
 253   boxes = postprocessed_tensors.get(detection_fields.detection_boxes)
 254   scores = postprocessed_tensors.get(detection_fields.detection_scores)
 255   multiclass_scores = postprocessed_tensors.get(
 256       detection_fields.detection_multiclass_scores)
 257   box_classifier_features = postprocessed_tensors.get(
 258       detection_fields.detection_features)
 259   raw_boxes = postprocessed_tensors.get(detection_fields.raw_detection_boxes)
 260   raw_scores = postprocessed_tensors.get(detection_fields.raw_detection_scores)
 261   classes = postprocessed_tensors.get(
 262       detection_fields.detection_classes) + label_id_offset
 263   keypoints = postprocessed_tensors.get(detection_fields.detection_keypoints)
 264   masks = postprocessed_tensors.get(detection_fields.detection_masks)
 265   num_detections = postprocessed_tensors.get(detection_fields.num_detections)
 266   outputs = {}
 267   outputs[detection_fields.detection_boxes] = tf.identity(
 268       boxes, name=detection_fields.detection_boxes)
 269   outputs[detection_fields.detection_scores] = tf.identity(
 270       scores, name=detection_fields.detection_scores)
 271   if multiclass_scores is not None:
 272     outputs[detection_fields.detection_multiclass_scores] = tf.identity(
 273         multiclass_scores, name=detection_fields.detection_multiclass_scores)
 274   if box_classifier_features is not None:
 275     outputs[detection_fields.detection_features] = tf.identity(
 276         box_classifier_features,
 277         name=detection_fields.detection_features)
 278   outputs[detection_fields.detection_classes] = tf.identity(
 279       classes, name=detection_fields.detection_classes)
 280   outputs[detection_fields.num_detections] = tf.identity(
 281       num_detections, name=detection_fields.num_detections)
 282   if raw_boxes is not None:
 283     outputs[detection_fields.raw_detection_boxes] = tf.identity(
 284         raw_boxes, name=detection_fields.raw_detection_boxes)
 285   if raw_scores is not None:
 286     outputs[detection_fields.raw_detection_scores] = tf.identity(
 287         raw_scores, name=detection_fields.raw_detection_scores)
 288   if keypoints is not None:
 289     outputs[detection_fields.detection_keypoints] = tf.identity(
 290         keypoints, name=detection_fields.detection_keypoints)
 291   if masks is not None:
 292     outputs[detection_fields.detection_masks] = tf.identity(
 293         masks, name=detection_fields.detection_masks)
 294   for output_key in outputs:
 295     tf.add_to_collection(output_collection_name, outputs[output_key])
 296 
 297   return outputs

 300 def write_saved_model(saved_model_path,
 301                       frozen_graph_def,
 302                       inputs,
 303                       outputs):
 304   """Writes SavedModel to disk.
 305 
 306   If checkpoint_path is not None bakes the weights into the graph thereby
 307   eliminating the need of checkpoint files during inference. If the model
 308   was trained with moving averages, setting use_moving_averages to true
 309   restores the moving averages, otherwise the original set of variables
 310   is restored.
 311 
 312   Args:
 313     saved_model_path: Path to write SavedModel.
 314     frozen_graph_def: tf.GraphDef holding frozen graph.
 315     inputs: The input placeholder tensor.
 316     outputs: A tensor dictionary containing the outputs of a DetectionModel.
 317   """
 318   with tf.Graph().as_default():
 319     with tf.Session() as sess:
 320 
 321       tf.import_graph_def(frozen_graph_def, name='')
 322 
 323       builder = tf.saved_model.builder.SavedModelBuilder(saved_model_path)
 324 
 325       tensor_info_inputs = {
 326           'inputs': tf.saved_model.utils.build_tensor_info(inputs)}
 327       tensor_info_outputs = {}
 328       for k, v in outputs.items():
 329         tensor_info_outputs[k] = tf.saved_model.utils.build_tensor_info(v)
 330 
 331       detection_signature = (
 332           tf.saved_model.signature_def_utils.build_signature_def(
 333               inputs=tensor_info_inputs,
 334               outputs=tensor_info_outputs,
 335               method_name=tf.saved_model.signature_constants.PREDICT_METHOD_NAME
 336           ))
 337 
 338       builder.add_meta_graph_and_variables(
 339           sess,
 340           [tf.saved_model.tag_constants.SERVING],
 341           signature_def_map={
 342               tf.saved_model.signature_constants
 343               .DEFAULT_SERVING_SIGNATURE_DEF_KEY:
 344                   detection_signature,
 345           },
 346       )
 347       builder.save()

 350 def write_graph_and_checkpoint(inference_graph_def,
 351                                model_path,
 352                                input_saver_def,
 353                                trained_checkpoint_prefix):
 354   """Writes the graph and the checkpoint into disk."""
 355   for node in inference_graph_def.node:
 356     node.device = ''
 357   with tf.Graph().as_default():
 358     tf.import_graph_def(inference_graph_def, name='')
 359     with tf.Session() as sess:
 360       saver = tf.train.Saver(
 361           saver_def=input_saver_def, save_relative_paths=True)
 362       saver.restore(sess, trained_checkpoint_prefix)
 363       saver.save(sess, model_path)

 366 def _get_outputs_from_inputs(input_tensors, detection_model,
 367                              output_collection_name):
 368   inputs = tf.cast(input_tensors, dtype=tf.float32)
 369   preprocessed_inputs, true_image_shapes = detection_model.preprocess(inputs)
 370   output_tensors = detection_model.predict(
 371       preprocessed_inputs, true_image_shapes)
 372   postprocessed_tensors = detection_model.postprocess(
 373       output_tensors, true_image_shapes)
 374   return add_output_tensor_nodes(postprocessed_tensors,
 375                                  output_collection_name)

 378 def build_detection_graph(input_type, detection_model, input_shape,
 379                           output_collection_name, graph_hook_fn):
 380   """Build the detection graph."""
 381   if input_type not in input_placeholder_fn_map:
 382     raise ValueError('Unknown input type: {}'.format(input_type))
 383   placeholder_args = {}
 384   if input_shape is not None:
 385     if (input_type != 'image_tensor' and
 386         input_type != 'encoded_image_string_tensor' and
 387         input_type != 'tf_example'):
 388       raise ValueError('Can only specify input shape for `image_tensor`, '
 389                        '`encoded_image_string_tensor`, or `tf_example` '
 390                        'inputs.')
 391     placeholder_args['input_shape'] = input_shape
 392   placeholder_tensor, input_tensors = input_placeholder_fn_map[input_type](
 393       **placeholder_args)
 394   outputs = _get_outputs_from_inputs(
 395       input_tensors=input_tensors,
 396       detection_model=detection_model,
 397       output_collection_name=output_collection_name)
 398 
 399   # Add global step to the graph.
 400   slim.get_or_create_global_step()
 401 
 402   if graph_hook_fn: graph_hook_fn()
 403 
 404   return outputs, placeholder_tensor

 407 def _export_inference_graph(input_type,
 408                             detection_model,
 409                             use_moving_averages,
 410                             trained_checkpoint_prefix,
 411                             output_directory,
 412                             additional_output_tensor_names=None,
 413                             input_shape=None,
 414                             output_collection_name='inference_op',
 415                             graph_hook_fn=None,
 416                             write_inference_graph=False,
 417                             temp_checkpoint_prefix=''):
 418   """Export helper."""
 419   tf.gfile.MakeDirs(output_directory)
 420   frozen_graph_path = os.path.join(output_directory,
 421                                    'frozen_inference_graph.pb')
 422   saved_model_path = os.path.join(output_directory, 'saved_model')
 423   model_path = os.path.join(output_directory, 'model.ckpt')
 424 
 425   outputs, placeholder_tensor = build_detection_graph(
 426       input_type=input_type,
 427       detection_model=detection_model,
 428       input_shape=input_shape,
 429       output_collection_name=output_collection_name,
 430       graph_hook_fn=graph_hook_fn)
 431 
 432   profile_inference_graph(tf.get_default_graph())
 433   saver_kwargs = {}
 434   if use_moving_averages:
 435     if not temp_checkpoint_prefix:
 436       # This check is to be compatible with both version of SaverDef.
 437       if os.path.isfile(trained_checkpoint_prefix):
 438         saver_kwargs['write_version'] = saver_pb2.SaverDef.V1
 439         temp_checkpoint_prefix = tempfile.NamedTemporaryFile().name
 440       else:
 441         temp_checkpoint_prefix = tempfile.mkdtemp()
 442     replace_variable_values_with_moving_averages(
 443         tf.get_default_graph(), trained_checkpoint_prefix,
 444         temp_checkpoint_prefix)
 445     checkpoint_to_use = temp_checkpoint_prefix
 446   else:
 447     checkpoint_to_use = trained_checkpoint_prefix
 448 
 449   saver = tf.train.Saver(**saver_kwargs)
 450   input_saver_def = saver.as_saver_def()
 451 
 452   write_graph_and_checkpoint(
 453       inference_graph_def=tf.get_default_graph().as_graph_def(),
 454       model_path=model_path,
 455       input_saver_def=input_saver_def,
 456       trained_checkpoint_prefix=checkpoint_to_use)
 457   if write_inference_graph:
 458     inference_graph_def = tf.get_default_graph().as_graph_def()
 459     inference_graph_path = os.path.join(output_directory,
 460                                         'inference_graph.pbtxt')
 461     for node in inference_graph_def.node:
 462       node.device = ''
 463     with tf.gfile.GFile(inference_graph_path, 'wb') as f:
 464       f.write(str(inference_graph_def))
 465 
 466   if additional_output_tensor_names is not None:
 467     output_node_names = ','.join(outputs.keys()+additional_output_tensor_names)
 468   else:
 469     output_node_names = ','.join(outputs.keys())
 470 
 471   frozen_graph_def = freeze_graph.freeze_graph_with_def_protos(
 472       input_graph_def=tf.get_default_graph().as_graph_def(),
 473       input_saver_def=input_saver_def,
 474       input_checkpoint=checkpoint_to_use,
 475       output_node_names=output_node_names,
 476       restore_op_name='save/restore_all',
 477       filename_tensor_name='save/Const:0',
 478       output_graph=frozen_graph_path,
 479       clear_devices=True,
 480       initializer_nodes='')
 481 
 482   write_saved_model(saved_model_path, frozen_graph_def,
 483                     placeholder_tensor, outputs)
 484

 486 def export_inference_graph(input_type,
 487                            pipeline_config,
 488                            trained_checkpoint_prefix,
 489                            output_directory,
 490                            input_shape=None,
 491                            output_collection_name='inference_op',
 492                            additional_output_tensor_names=None,
 493                            write_inference_graph=False):
 494   """Exports inference graph for the model specified in the pipeline config.
 495 
 496   Args:
 497     input_type: Type of input for the graph. Can be one of ['image_tensor',
 498       'encoded_image_string_tensor', 'tf_example'].
 499     pipeline_config: pipeline_pb2.TrainAndEvalPipelineConfig proto.
 500     trained_checkpoint_prefix: Path to the trained checkpoint file.
 501     output_directory: Path to write outputs.
 502     input_shape: Sets a fixed shape for an `image_tensor` input. If not
 503       specified, will default to [None, None, None, 3].
 504     output_collection_name: Name of collection to add output tensors to.
 505       If None, does not add output tensors to a collection.
 506     additional_output_tensor_names: list of additional output
 507       tensors to include in the frozen graph.
 508     write_inference_graph: If true, writes inference graph to disk.
 509   """
 510   detection_model = model_builder.build(pipeline_config.model,
 511                                         is_training=False)
 512   graph_rewriter_fn = None
 513   if pipeline_config.HasField('graph_rewriter'):
 514     graph_rewriter_config = pipeline_config.graph_rewriter
 515     graph_rewriter_fn = graph_rewriter_builder.build(graph_rewriter_config,
 516                                                      is_training=False)
 517   _export_inference_graph(
 518       input_type,
 519       detection_model,
 520       pipeline_config.eval_config.use_moving_averages,
 521       trained_checkpoint_prefix,
 522       output_directory,
 523       additional_output_tensor_names,
 524       input_shape,
 525       output_collection_name,
 526       graph_hook_fn=graph_rewriter_fn,
 527       write_inference_graph=write_inference_graph)
 528   pipeline_config.eval_config.use_moving_averages = False
 529   config_util.save_pipeline_config(pipeline_config, output_directory)

 532 def profile_inference_graph(graph):
 533   """Profiles the inference graph.
 534 
 535   Prints model parameters and computation FLOPs given an inference graph.
 536   BatchNorms are excluded from the parameter count due to the fact that
 537   BatchNorms are usually folded. BatchNorm, Initializer, Regularizer
 538   and BiasAdd are not considered in FLOP count.
 539 
 540   Args:
 541     graph: the inference graph.
 542   """
 543   tfprof_vars_option = (
 544       contrib_tfprof.model_analyzer.TRAINABLE_VARS_PARAMS_STAT_OPTIONS)
 545   tfprof_flops_option = contrib_tfprof.model_analyzer.FLOAT_OPS_OPTIONS
 546 
 547   # Batchnorm is usually folded during inference.
 548   tfprof_vars_option['trim_name_regexes'] = ['.*BatchNorm.*']
 549   # Initializer and Regularizer are only used in training.
 550   tfprof_flops_option['trim_name_regexes'] = [
 551       '.*BatchNorm.*', '.*Initializer.*', '.*Regularizer.*', '.*BiasAdd.*'
 552   ]
 553 
 554   contrib_tfprof.model_analyzer.print_model_analysis(
 555       graph, tfprof_options=tfprof_vars_option)
 556 
 557   contrib_tfprof.model_analyzer.print_model_analysis(
 558       graph, tfprof_options=tfprof_flops_option)
