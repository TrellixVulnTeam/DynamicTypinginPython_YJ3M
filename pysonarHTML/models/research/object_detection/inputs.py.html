  55 def _multiclass_scores_or_one_hot_labels(multiclass_scores,
  56                                          groundtruth_boxes,
  57                                          groundtruth_classes, num_classes):
  58   """Returns one-hot encoding of classes when multiclass_scores is empty."""
  59   # Replace groundtruth_classes tensor with multiclass_scores tensor when its
  60   # non-empty. If multiclass_scores is empty fall back on groundtruth_classes
  61   # tensor.
  62   def true_fn():
  63     return tf.reshape(multiclass_scores,
  64                       [tf.shape(groundtruth_boxes)[0], num_classes])
  65   def false_fn():
  66     return tf.one_hot(groundtruth_classes, num_classes)
  67 
  68   return tf.cond(tf.size(multiclass_scores) > 0, true_fn, false_fn)

  71 def _convert_labeled_classes_to_k_hot(groundtruth_labeled_classes, num_classes):
  72   """Returns k-hot encoding of the labeled classes."""
  73 
  74   # If the input labeled_classes is empty, it assumes all classes are
  75   # exhaustively labeled, thus returning an all-one encoding.
  76   def true_fn():
  77     return tf.sparse_to_dense(
  78         groundtruth_labeled_classes - _LABEL_OFFSET, [num_classes],
  79         tf.constant(1, dtype=tf.float32),
  80         validate_indices=False)
  81 
  82   def false_fn():
  83     return tf.ones(num_classes, dtype=tf.float32)
  84 
  85   return tf.cond(tf.size(groundtruth_labeled_classes) > 0, true_fn, false_fn)

  88 def _remove_unrecognized_classes(class_ids, unrecognized_label):
  89   """Returns class ids with unrecognized classes filtered out."""
  90 
  91   recognized_indices = tf.where(tf.greater(class_ids, unrecognized_label))
  92   return tf.gather(class_ids, recognized_indices)

  95 def assert_or_prune_invalid_boxes(boxes):
  96   """Makes sure boxes have valid sizes (ymax >= ymin, xmax >= xmin).
  97 
  98   When the hardware supports assertions, the function raises an error when
  99   boxes have an invalid size. If assertions are not supported (e.g. on TPU),
 100   boxes with invalid sizes are filtered out.
 101 
 102   Args:
 103     boxes: float tensor of shape [num_boxes, 4]
 104 
 105   Returns:
 106     boxes: float tensor of shape [num_valid_boxes, 4] with invalid boxes
 107       filtered out.
 108 
 109   Raises:
 110     tf.errors.InvalidArgumentError: When we detect boxes with invalid size.
 111       This is not supported on TPUs.
 112   """
 113 
 114   ymin, xmin, ymax, xmax = tf.split(
 115       boxes, num_or_size_splits=4, axis=1)
 116 
 117   height_check = tf.Assert(tf.reduce_all(ymax >= ymin), [ymin, ymax])
 118   width_check = tf.Assert(tf.reduce_all(xmax >= xmin), [xmin, xmax])
 119 
 120   with tf.control_dependencies([height_check, width_check]):
 121     boxes_tensor = tf.concat([ymin, xmin, ymax, xmax], axis=1)
 122     boxlist = box_list.BoxList(boxes_tensor)
 123     # TODO(b/149221748) Remove pruning when XLA supports assertions.
 124     boxlist = box_list_ops.prune_small_boxes(boxlist, 0)
 125 
 126   return boxlist.get()

 129 def transform_input_data(tensor_dict,
 130                          model_preprocess_fn,
 131                          image_resizer_fn,
 132                          num_classes,
 133                          data_augmentation_fn=None,
 134                          merge_multiple_boxes=False,
 135                          retain_original_image=False,
 136                          use_multiclass_scores=False,
 137                          use_bfloat16=False,
 138                          retain_original_image_additional_channels=False,
 139                          keypoint_type_weight=None):
 140   """A single function that is responsible for all input data transformations.
 141 
 142   Data transformation functions are applied in the following order.
 143   1. If key fields.InputDataFields.image_additional_channels is present in
 144      tensor_dict, the additional channels will be merged into
 145      fields.InputDataFields.image.
 146   2. data_augmentation_fn (optional): applied on tensor_dict.
 147   3. model_preprocess_fn: applied only on image tensor in tensor_dict.
 148   4. keypoint_type_weight (optional): If groundtruth keypoints are in
 149      the tensor dictionary, per-keypoint weights are produced. These weights are
 150      initialized by `keypoint_type_weight` (or ones if left None).
 151      Then, for all keypoints that are not visible, the weights are set to 0 (to
 152      avoid penalizing the model in a loss function).
 153   5. image_resizer_fn: applied on original image and instance mask tensor in
 154      tensor_dict.
 155   6. one_hot_encoding: applied to classes tensor in tensor_dict.
 156   7. merge_multiple_boxes (optional): when groundtruth boxes are exactly the
 157      same they can be merged into a single box with an associated k-hot class
 158      label.
 159 
 160   Args:
 161     tensor_dict: dictionary containing input tensors keyed by
 162       fields.InputDataFields.
 163     model_preprocess_fn: model's preprocess function to apply on image tensor.
 164       This function must take in a 4-D float tensor and return a 4-D preprocess
 165       float tensor and a tensor containing the true image shape.
 166     image_resizer_fn: image resizer function to apply on groundtruth instance
 167       `masks. This function must take a 3-D float tensor of an image and a 3-D
 168       tensor of instance masks and return a resized version of these along with
 169       the true shapes.
 170     num_classes: number of max classes to one-hot (or k-hot) encode the class
 171       labels.
 172     data_augmentation_fn: (optional) data augmentation function to apply on
 173       input `tensor_dict`.
 174     merge_multiple_boxes: (optional) whether to merge multiple groundtruth boxes
 175       and classes for a given image if the boxes are exactly the same.
 176     retain_original_image: (optional) whether to retain original image in the
 177       output dictionary.
 178     use_multiclass_scores: whether to use multiclass scores as class targets
 179       instead of one-hot encoding of `groundtruth_classes`. When
 180       this is True and multiclass_scores is empty, one-hot encoding of
 181       `groundtruth_classes` is used as a fallback.
 182     use_bfloat16: (optional) a bool, whether to use bfloat16 in training.
 183     retain_original_image_additional_channels: (optional) Whether to retain
 184       original image additional channels in the output dictionary.
 185     keypoint_type_weight: A list (of length num_keypoints) containing
 186       groundtruth loss weights to use for each keypoint. If None, will use a
 187       weight of 1.
 188 
 189   Returns:
 190     A dictionary keyed by fields.InputDataFields containing the tensors obtained
 191     after applying all the transformations.
 192 
 193   Raises:
 194     KeyError: If both groundtruth_labeled_classes and groundtruth_image_classes
 195       are provided by the decoder in tensor_dict since both fields are
 196       considered to contain the same information.
 197   """
 198   out_tensor_dict = tensor_dict.copy()
 199 
 200   labeled_classes_field = fields.InputDataFields.groundtruth_labeled_classes
 201   image_classes_field = fields.InputDataFields.groundtruth_image_classes
 202   if (labeled_classes_field in out_tensor_dict and
 203       image_classes_field in out_tensor_dict):
 204     raise KeyError('groundtruth_labeled_classes and groundtruth_image_classes'
 205                    'are provided by the decoder, but only one should be set.')
 206 
 207   if labeled_classes_field in out_tensor_dict:
 208     # tf_example_decoder casts unrecognized labels to -1. Remove these
 209     # unrecognized labels before converting labeled_classes to k-hot vector.
 210     out_tensor_dict[labeled_classes_field] = _remove_unrecognized_classes(
 211         out_tensor_dict[labeled_classes_field], unrecognized_label=-1)
 212     out_tensor_dict[labeled_classes_field] = _convert_labeled_classes_to_k_hot(
 213         out_tensor_dict[labeled_classes_field], num_classes)
 214 
 215   if image_classes_field in out_tensor_dict:
 216     out_tensor_dict[labeled_classes_field] = _convert_labeled_classes_to_k_hot(
 217         out_tensor_dict[image_classes_field], num_classes)
 218 
 219   if fields.InputDataFields.multiclass_scores in out_tensor_dict:
 220     out_tensor_dict[
 221         fields.InputDataFields
 222         .multiclass_scores] = _multiclass_scores_or_one_hot_labels(
 223             out_tensor_dict[fields.InputDataFields.multiclass_scores],
 224             out_tensor_dict[fields.InputDataFields.groundtruth_boxes],
 225             out_tensor_dict[fields.InputDataFields.groundtruth_classes],
 226             num_classes)
 227 
 228   if fields.InputDataFields.groundtruth_boxes in out_tensor_dict:
 229     out_tensor_dict = util_ops.filter_groundtruth_with_nan_box_coordinates(
 230         out_tensor_dict)
 231     out_tensor_dict = util_ops.filter_unrecognized_classes(out_tensor_dict)
 232 
 233   if retain_original_image:
 234     out_tensor_dict[fields.InputDataFields.original_image] = tf.cast(
 235         image_resizer_fn(out_tensor_dict[fields.InputDataFields.image],
 236                          None)[0], tf.uint8)
 237 
 238   if fields.InputDataFields.image_additional_channels in out_tensor_dict:
 239     channels = out_tensor_dict[fields.InputDataFields.image_additional_channels]
 240     out_tensor_dict[fields.InputDataFields.image] = tf.concat(
 241         [out_tensor_dict[fields.InputDataFields.image], channels], axis=2)
 242     if retain_original_image_additional_channels:
 243       out_tensor_dict[
 244           fields.InputDataFields.image_additional_channels] = tf.cast(
 245               image_resizer_fn(channels, None)[0], tf.uint8)
 246 
 247   # Apply data augmentation ops.
 248   if data_augmentation_fn is not None:
 249     out_tensor_dict = data_augmentation_fn(out_tensor_dict)
 250 
 251   # Apply model preprocessing ops and resize instance masks.
 252   image = out_tensor_dict[fields.InputDataFields.image]
 253   preprocessed_resized_image, true_image_shape = model_preprocess_fn(
 254       tf.expand_dims(tf.cast(image, dtype=tf.float32), axis=0))
 255 
 256   preprocessed_shape = tf.shape(preprocessed_resized_image)
 257   new_height, new_width = preprocessed_shape[1], preprocessed_shape[2]
 258 
 259   im_box = tf.stack([
 260       0.0, 0.0,
 261       tf.to_float(new_height) / tf.to_float(true_image_shape[0, 0]),
 262       tf.to_float(new_width) / tf.to_float(true_image_shape[0, 1])
 263   ])
 264 
 265   if fields.InputDataFields.groundtruth_boxes in tensor_dict:
 266     bboxes = out_tensor_dict[fields.InputDataFields.groundtruth_boxes]
 267     boxlist = box_list.BoxList(bboxes)
 268     realigned_bboxes = box_list_ops.change_coordinate_frame(boxlist, im_box)
 269 
 270     realigned_boxes_tensor = realigned_bboxes.get()
 271     valid_boxes_tensor = assert_or_prune_invalid_boxes(realigned_boxes_tensor)
 272     out_tensor_dict[
 273         fields.InputDataFields.groundtruth_boxes] = valid_boxes_tensor
 274 
 275   if fields.InputDataFields.groundtruth_keypoints in tensor_dict:
 276     keypoints = out_tensor_dict[fields.InputDataFields.groundtruth_keypoints]
 277     realigned_keypoints = keypoint_ops.change_coordinate_frame(keypoints,
 278                                                                im_box)
 279     out_tensor_dict[
 280         fields.InputDataFields.groundtruth_keypoints] = realigned_keypoints
 281     flds_gt_kpt = fields.InputDataFields.groundtruth_keypoints
 282     flds_gt_kpt_vis = fields.InputDataFields.groundtruth_keypoint_visibilities
 283     flds_gt_kpt_weights = fields.InputDataFields.groundtruth_keypoint_weights
 284     if flds_gt_kpt_vis not in out_tensor_dict:
 285       out_tensor_dict[flds_gt_kpt_vis] = tf.ones_like(
 286           out_tensor_dict[flds_gt_kpt][:, :, 0],
 287           dtype=tf.bool)
 288     out_tensor_dict[flds_gt_kpt_weights] = (
 289         keypoint_ops.keypoint_weights_from_visibilities(
 290             out_tensor_dict[flds_gt_kpt_vis],
 291             keypoint_type_weight))
 292 
 293   if use_bfloat16:
 294     preprocessed_resized_image = tf.cast(
 295         preprocessed_resized_image, tf.bfloat16)
 296     if fields.InputDataFields.context_features in out_tensor_dict:
 297       out_tensor_dict[fields.InputDataFields.context_features] = tf.cast(
 298           out_tensor_dict[fields.InputDataFields.context_features], tf.bfloat16)
 299   out_tensor_dict[fields.InputDataFields.image] = tf.squeeze(
 300       preprocessed_resized_image, axis=0)
 301   out_tensor_dict[fields.InputDataFields.true_image_shape] = tf.squeeze(
 302       true_image_shape, axis=0)
 303   if fields.InputDataFields.groundtruth_instance_masks in out_tensor_dict:
 304     masks = out_tensor_dict[fields.InputDataFields.groundtruth_instance_masks]
 305     _, resized_masks, _ = image_resizer_fn(image, masks)
 306     if use_bfloat16:
 307       resized_masks = tf.cast(resized_masks, tf.bfloat16)
 308     out_tensor_dict[
 309         fields.InputDataFields.groundtruth_instance_masks] = resized_masks
 310 
 311   zero_indexed_groundtruth_classes = out_tensor_dict[
 312       fields.InputDataFields.groundtruth_classes] - _LABEL_OFFSET
 313   if use_multiclass_scores:
 314     out_tensor_dict[
 315         fields.InputDataFields.groundtruth_classes] = out_tensor_dict[
 316             fields.InputDataFields.multiclass_scores]
 317   else:
 318     out_tensor_dict[fields.InputDataFields.groundtruth_classes] = tf.one_hot(
 319         zero_indexed_groundtruth_classes, num_classes)
 320   out_tensor_dict.pop(fields.InputDataFields.multiclass_scores, None)
 321 
 322   if fields.InputDataFields.groundtruth_confidences in out_tensor_dict:
 323     groundtruth_confidences = out_tensor_dict[
 324         fields.InputDataFields.groundtruth_confidences]
 325     # Map the confidences to the one-hot encoding of classes
 326     out_tensor_dict[fields.InputDataFields.groundtruth_confidences] = (
 327         tf.reshape(groundtruth_confidences, [-1, 1]) *
 328         out_tensor_dict[fields.InputDataFields.groundtruth_classes])
 329   else:
 330     groundtruth_confidences = tf.ones_like(
 331         zero_indexed_groundtruth_classes, dtype=tf.float32)
 332     out_tensor_dict[fields.InputDataFields.groundtruth_confidences] = (
 333         out_tensor_dict[fields.InputDataFields.groundtruth_classes])
 334 
 335   if merge_multiple_boxes:
 336     merged_boxes, merged_classes, merged_confidences, _ = (
 337         util_ops.merge_boxes_with_multiple_labels(
 338             out_tensor_dict[fields.InputDataFields.groundtruth_boxes],
 339             zero_indexed_groundtruth_classes,
 340             groundtruth_confidences,
 341             num_classes))
 342     merged_classes = tf.cast(merged_classes, tf.float32)
 343     out_tensor_dict[fields.InputDataFields.groundtruth_boxes] = merged_boxes
 344     out_tensor_dict[fields.InputDataFields.groundtruth_classes] = merged_classes
 345     out_tensor_dict[fields.InputDataFields.groundtruth_confidences] = (
 346         merged_confidences)
 347   if fields.InputDataFields.groundtruth_boxes in out_tensor_dict:
 348     out_tensor_dict[fields.InputDataFields.num_groundtruth_boxes] = tf.shape(
 349         out_tensor_dict[fields.InputDataFields.groundtruth_boxes])[0]
 350 
 351   return out_tensor_dict

 354 def pad_input_data_to_static_shapes(tensor_dict,
 355                                     max_num_boxes,
 356                                     num_classes,
 357                                     spatial_image_shape=None,
 358                                     max_num_context_features=None,
 359                                     context_feature_length=None):
 360   """Pads input tensors to static shapes.
 361 
 362   In case num_additional_channels > 0, we assume that the additional channels
 363   have already been concatenated to the base image.
 364 
 365   Args:
 366     tensor_dict: Tensor dictionary of input data
 367     max_num_boxes: Max number of groundtruth boxes needed to compute shapes for
 368       padding.
 369     num_classes: Number of classes in the dataset needed to compute shapes for
 370       padding.
 371     spatial_image_shape: A list of two integers of the form [height, width]
 372       containing expected spatial shape of the image.
 373     max_num_context_features (optional): The maximum number of context
 374       features needed to compute shapes padding.
 375     context_feature_length (optional): The length of the context feature.
 376 
 377   Returns:
 378     A dictionary keyed by fields.InputDataFields containing padding shapes for
 379     tensors in the dataset.
 380 
 381   Raises:
 382     ValueError: If groundtruth classes is neither rank 1 nor rank 2, or if we
 383       detect that additional channels have not been concatenated yet, or if
 384       max_num_context_features is not specified and context_features is in the
 385       tensor dict.
 386   """
 387 
 388   if not spatial_image_shape or spatial_image_shape == [-1, -1]:
 389     height, width = None, None
 390   else:
 391     height, width = spatial_image_shape  # pylint: disable=unpacking-non-sequence
 392 
 393   num_additional_channels = 0
 394   if fields.InputDataFields.image_additional_channels in tensor_dict:
 395     num_additional_channels = shape_utils.get_dim_as_int(tensor_dict[
 396         fields.InputDataFields.image_additional_channels].shape[2])
 397 
 398   # We assume that if num_additional_channels > 0, then it has already been
 399   # concatenated to the base image (but not the ground truth).
 400   num_channels = 3
 401   if fields.InputDataFields.image in tensor_dict:
 402     num_channels = shape_utils.get_dim_as_int(
 403         tensor_dict[fields.InputDataFields.image].shape[2])
 404 
 405   if num_additional_channels:
 406     if num_additional_channels >= num_channels:
 407       raise ValueError(
 408           'Image must be already concatenated with additional channels.')
 409 
 410     if (fields.InputDataFields.original_image in tensor_dict and
 411         shape_utils.get_dim_as_int(
 412             tensor_dict[fields.InputDataFields.original_image].shape[2]) ==
 413         num_channels):
 414       raise ValueError(
 415           'Image must be already concatenated with additional channels.')
 416 
 417   if fields.InputDataFields.context_features in tensor_dict and (
 418       max_num_context_features is None):
 419     raise ValueError('max_num_context_features must be specified in the model '
 420                      'config if include_context is specified in the input '
 421                      'config')
 422 
 423   padding_shapes = {
 424       fields.InputDataFields.image: [height, width, num_channels],
 425       fields.InputDataFields.original_image_spatial_shape: [2],
 426       fields.InputDataFields.image_additional_channels: [
 427           height, width, num_additional_channels
 428       ],
 429       fields.InputDataFields.source_id: [],
 430       fields.InputDataFields.filename: [],
 431       fields.InputDataFields.key: [],
 432       fields.InputDataFields.groundtruth_difficult: [max_num_boxes],
 433       fields.InputDataFields.groundtruth_boxes: [max_num_boxes, 4],
 434       fields.InputDataFields.groundtruth_classes: [max_num_boxes, num_classes],
 435       fields.InputDataFields.groundtruth_instance_masks: [
 436           max_num_boxes, height, width
 437       ],
 438       fields.InputDataFields.groundtruth_is_crowd: [max_num_boxes],
 439       fields.InputDataFields.groundtruth_group_of: [max_num_boxes],
 440       fields.InputDataFields.groundtruth_area: [max_num_boxes],
 441       fields.InputDataFields.groundtruth_weights: [max_num_boxes],
 442       fields.InputDataFields.groundtruth_confidences: [
 443           max_num_boxes, num_classes
 444       ],
 445       fields.InputDataFields.num_groundtruth_boxes: [],
 446       fields.InputDataFields.groundtruth_label_types: [max_num_boxes],
 447       fields.InputDataFields.groundtruth_label_weights: [max_num_boxes],
 448       fields.InputDataFields.true_image_shape: [3],
 449       fields.InputDataFields.groundtruth_image_classes: [num_classes],
 450       fields.InputDataFields.groundtruth_image_confidences: [num_classes],
 451       fields.InputDataFields.groundtruth_labeled_classes: [num_classes],
 452   }
 453 
 454   if fields.InputDataFields.original_image in tensor_dict:
 455     padding_shapes[fields.InputDataFields.original_image] = [
 456         height, width,
 457         shape_utils.get_dim_as_int(tensor_dict[fields.InputDataFields.
 458                                                original_image].shape[2])
 459     ]
 460   if fields.InputDataFields.groundtruth_keypoints in tensor_dict:
 461     tensor_shape = (
 462         tensor_dict[fields.InputDataFields.groundtruth_keypoints].shape)
 463     padding_shape = [max_num_boxes,
 464                      shape_utils.get_dim_as_int(tensor_shape[1]),
 465                      shape_utils.get_dim_as_int(tensor_shape[2])]
 466     padding_shapes[fields.InputDataFields.groundtruth_keypoints] = padding_shape
 467   if fields.InputDataFields.groundtruth_keypoint_visibilities in tensor_dict:
 468     tensor_shape = tensor_dict[fields.InputDataFields.
 469                                groundtruth_keypoint_visibilities].shape
 470     padding_shape = [max_num_boxes, shape_utils.get_dim_as_int(tensor_shape[1])]
 471     padding_shapes[fields.InputDataFields.
 472                    groundtruth_keypoint_visibilities] = padding_shape
 473 
 474   if fields.InputDataFields.groundtruth_keypoint_weights in tensor_dict:
 475     tensor_shape = (
 476         tensor_dict[fields.InputDataFields.groundtruth_keypoint_weights].shape)
 477     padding_shape = [max_num_boxes, shape_utils.get_dim_as_int(tensor_shape[1])]
 478     padding_shapes[fields.InputDataFields.
 479                    groundtruth_keypoint_weights] = padding_shape
 480 
 481   # Prepare for ContextRCNN related fields.
 482   if fields.InputDataFields.context_features in tensor_dict:
 483     padding_shape = [max_num_context_features, context_feature_length]
 484     padding_shapes[fields.InputDataFields.context_features] = padding_shape
 485 
 486     tensor_shape = tf.shape(
 487         tensor_dict[fields.InputDataFields.context_features])
 488     tensor_dict[fields.InputDataFields.valid_context_size] = tensor_shape[0]
 489     padding_shapes[fields.InputDataFields.valid_context_size] = []
 490   if fields.InputDataFields.context_feature_length in tensor_dict:
 491     padding_shapes[fields.InputDataFields.context_feature_length] = []
 492 
 493   if fields.InputDataFields.is_annotated in tensor_dict:
 494     padding_shapes[fields.InputDataFields.is_annotated] = []
 495 
 496   padded_tensor_dict = {}
 497   for tensor_name in tensor_dict:
 498     padded_tensor_dict[tensor_name] = shape_utils.pad_or_clip_nd(
 499         tensor_dict[tensor_name], padding_shapes[tensor_name])
 500 
 501   # Make sure that the number of groundtruth boxes now reflects the
 502   # padded/clipped tensors.
 503   if fields.InputDataFields.num_groundtruth_boxes in padded_tensor_dict:
 504     padded_tensor_dict[fields.InputDataFields.num_groundtruth_boxes] = (
 505         tf.minimum(
 506             padded_tensor_dict[fields.InputDataFields.num_groundtruth_boxes],
 507             max_num_boxes))
 508   return padded_tensor_dict

 511 def augment_input_data(tensor_dict, data_augmentation_options):
 512   """Applies data augmentation ops to input tensors.
 513 
 514   Args:
 515     tensor_dict: A dictionary of input tensors keyed by fields.InputDataFields.
 516     data_augmentation_options: A list of tuples, where each tuple contains a
 517       function and a dictionary that contains arguments and their values.
 518       Usually, this is the output of core/preprocessor.build.
 519 
 520   Returns:
 521     A dictionary of tensors obtained by applying data augmentation ops to the
 522     input tensor dictionary.
 523   """
 524   tensor_dict[fields.InputDataFields.image] = tf.expand_dims(
 525       tf.cast(tensor_dict[fields.InputDataFields.image], dtype=tf.float32), 0)
 526 
 527   include_instance_masks = (fields.InputDataFields.groundtruth_instance_masks
 528                             in tensor_dict)
 529   include_keypoints = (fields.InputDataFields.groundtruth_keypoints
 530                        in tensor_dict)
 531   include_keypoint_visibilities = (
 532       fields.InputDataFields.groundtruth_keypoint_visibilities in tensor_dict)
 533   include_label_weights = (fields.InputDataFields.groundtruth_weights
 534                            in tensor_dict)
 535   include_label_confidences = (fields.InputDataFields.groundtruth_confidences
 536                                in tensor_dict)
 537   include_multiclass_scores = (fields.InputDataFields.multiclass_scores in
 538                                tensor_dict)
 539   tensor_dict = preprocessor.preprocess(
 540       tensor_dict, data_augmentation_options,
 541       func_arg_map=preprocessor.get_default_func_arg_map(
 542           include_label_weights=include_label_weights,
 543           include_label_confidences=include_label_confidences,
 544           include_multiclass_scores=include_multiclass_scores,
 545           include_instance_masks=include_instance_masks,
 546           include_keypoints=include_keypoints,
 547           include_keypoint_visibilities=include_keypoint_visibilities))
 548   tensor_dict[fields.InputDataFields.image] = tf.squeeze(
 549       tensor_dict[fields.InputDataFields.image], axis=0)
 550   return tensor_dict

 553 def _get_labels_dict(input_dict):
 554   """Extracts labels dict from input dict."""
 555   required_label_keys = [
 556       fields.InputDataFields.num_groundtruth_boxes,
 557       fields.InputDataFields.groundtruth_boxes,
 558       fields.InputDataFields.groundtruth_classes,
 559       fields.InputDataFields.groundtruth_weights,
 560   ]
 561   labels_dict = {}
 562   for key in required_label_keys:
 563     labels_dict[key] = input_dict[key]
 564 
 565   optional_label_keys = [
 566       fields.InputDataFields.groundtruth_confidences,
 567       fields.InputDataFields.groundtruth_labeled_classes,
 568       fields.InputDataFields.groundtruth_keypoints,
 569       fields.InputDataFields.groundtruth_instance_masks,
 570       fields.InputDataFields.groundtruth_area,
 571       fields.InputDataFields.groundtruth_is_crowd,
 572       fields.InputDataFields.groundtruth_group_of,
 573       fields.InputDataFields.groundtruth_difficult,
 574       fields.InputDataFields.groundtruth_keypoint_visibilities,
 575       fields.InputDataFields.groundtruth_keypoint_weights,
 576   ]
 577 
 578   for key in optional_label_keys:
 579     if key in input_dict:
 580       labels_dict[key] = input_dict[key]
 581   if fields.InputDataFields.groundtruth_difficult in labels_dict:
 582     labels_dict[fields.InputDataFields.groundtruth_difficult] = tf.cast(
 583         labels_dict[fields.InputDataFields.groundtruth_difficult], tf.int32)
 584   return labels_dict

 587 def _replace_empty_string_with_random_number(string_tensor):
 588   """Returns string unchanged if non-empty, and random string tensor otherwise.
 589 
 590   The random string is an integer 0 and 2**63 - 1, casted as string.
 591 
 592 
 593   Args:
 594     string_tensor: A tf.tensor of dtype string.
 595 
 596   Returns:
 597     out_string: A tf.tensor of dtype string. If string_tensor contains the empty
 598       string, out_string will contain a random integer casted to a string.
 599       Otherwise string_tensor is returned unchanged.
 600 
 601   """
 602 
 603   empty_string = tf.constant('', dtype=tf.string, name='EmptyString')
 604 
 605   random_source_id = tf.as_string(
 606       tf.random_uniform(shape=[], maxval=2**63 - 1, dtype=tf.int64))
 607 
 608   out_string = tf.cond(
 609       tf.equal(string_tensor, empty_string),
 610       true_fn=lambda: random_source_id,
 611       false_fn=lambda: string_tensor)
 612 
 613   return out_string

 616 def _get_features_dict(input_dict, include_source_id=False):
 617   """Extracts features dict from input dict."""
 618 
 619   source_id = _replace_empty_string_with_random_number(
 620       input_dict[fields.InputDataFields.source_id])
 621 
 622   hash_from_source_id = tf.string_to_hash_bucket_fast(source_id, HASH_BINS)
 623   features = {
 624       fields.InputDataFields.image:
 625           input_dict[fields.InputDataFields.image],
 626       HASH_KEY: tf.cast(hash_from_source_id, tf.int32),
 627       fields.InputDataFields.true_image_shape:
 628           input_dict[fields.InputDataFields.true_image_shape],
 629       fields.InputDataFields.original_image_spatial_shape:
 630           input_dict[fields.InputDataFields.original_image_spatial_shape]
 631   }
 632   if include_source_id:
 633     features[fields.InputDataFields.source_id] = source_id
 634   if fields.InputDataFields.original_image in input_dict:
 635     features[fields.InputDataFields.original_image] = input_dict[
 636         fields.InputDataFields.original_image]
 637   if fields.InputDataFields.image_additional_channels in input_dict:
 638     features[fields.InputDataFields.image_additional_channels] = input_dict[
 639         fields.InputDataFields.image_additional_channels]
 640   if fields.InputDataFields.context_features in input_dict:
 641     features[fields.InputDataFields.context_features] = input_dict[
 642         fields.InputDataFields.context_features]
 643   if fields.InputDataFields.valid_context_size in input_dict:
 644     features[fields.InputDataFields.valid_context_size] = input_dict[
 645         fields.InputDataFields.valid_context_size]
 646   return features

 669 def train_input(train_config, train_input_config,
 670                 model_config, model=None, params=None, input_context=None):
 671   """Returns `features` and `labels` tensor dictionaries for training.
 672 
 673   Args:
 674     train_config: A train_pb2.TrainConfig.
 675     train_input_config: An input_reader_pb2.InputReader.
 676     model_config: A model_pb2.DetectionModel.
 677     model: A pre-constructed Detection Model.
 678       If None, one will be created from the config.
 679     params: Parameter dictionary passed from the estimator.
 680     input_context: optional, A tf.distribute.InputContext object used to
 681       shard filenames and compute per-replica batch_size when this function
 682       is being called per-replica.
 683 
 684   Returns:
 685     A tf.data.Dataset that holds (features, labels) tuple.
 686 
 687     features: Dictionary of feature tensors.
 688       features[fields.InputDataFields.image] is a [batch_size, H, W, C]
 689         float32 tensor with preprocessed images.
 690       features[HASH_KEY] is a [batch_size] int32 tensor representing unique
 691         identifiers for the images.
 692       features[fields.InputDataFields.true_image_shape] is a [batch_size, 3]
 693         int32 tensor representing the true image shapes, as preprocessed
 694         images could be padded.
 695       features[fields.InputDataFields.original_image] (optional) is a
 696         [batch_size, H, W, C] float32 tensor with original images.
 697     labels: Dictionary of groundtruth tensors.
 698       labels[fields.InputDataFields.num_groundtruth_boxes] is a [batch_size]
 699         int32 tensor indicating the number of groundtruth boxes.
 700       labels[fields.InputDataFields.groundtruth_boxes] is a
 701         [batch_size, num_boxes, 4] float32 tensor containing the corners of
 702         the groundtruth boxes.
 703       labels[fields.InputDataFields.groundtruth_classes] is a
 704         [batch_size, num_boxes, num_classes] float32 one-hot tensor of
 705         classes.
 706       labels[fields.InputDataFields.groundtruth_weights] is a
 707         [batch_size, num_boxes] float32 tensor containing groundtruth weights
 708         for the boxes.
 709       -- Optional --
 710       labels[fields.InputDataFields.groundtruth_instance_masks] is a
 711         [batch_size, num_boxes, H, W] float32 tensor containing only binary
 712         values, which represent instance masks for objects.
 713       labels[fields.InputDataFields.groundtruth_keypoints] is a
 714         [batch_size, num_boxes, num_keypoints, 2] float32 tensor containing
 715         keypoints for each box.
 716       labels[fields.InputDataFields.groundtruth_weights] is a
 717         [batch_size, num_boxes, num_keypoints] float32 tensor containing
 718         groundtruth weights for the keypoints.
 719       labels[fields.InputDataFields.groundtruth_visibilities] is a
 720         [batch_size, num_boxes, num_keypoints] bool tensor containing
 721         groundtruth visibilities for each keypoint.
 722       labels[fields.InputDataFields.groundtruth_labeled_classes] is a
 723         [batch_size, num_classes] float32 k-hot tensor of classes.
 724 
 725   Raises:
 726     TypeError: if the `train_config`, `train_input_config` or `model_config`
 727       are not of the correct type.
 728   """
 729   if not isinstance(train_config, train_pb2.TrainConfig):
 730     raise TypeError('For training mode, the `train_config` must be a '
 731                     'train_pb2.TrainConfig.')
 732   if not isinstance(train_input_config, input_reader_pb2.InputReader):
 733     raise TypeError('The `train_input_config` must be a '
 734                     'input_reader_pb2.InputReader.')
 735   if not isinstance(model_config, model_pb2.DetectionModel):
 736     raise TypeError('The `model_config` must be a '
 737                     'model_pb2.DetectionModel.')
 738 
 739   if model is None:
 740     model_preprocess_fn = INPUT_BUILDER_UTIL_MAP['model_build'](
 741         model_config, is_training=True).preprocess
 742   else:
 743     model_preprocess_fn = model.preprocess
 744 
 745   num_classes = config_util.get_number_of_classes(model_config)
 746 
 747   def transform_and_pad_input_data_fn(tensor_dict):
 748     """Combines transform and pad operation."""
 749     data_augmentation_options = [
 750         preprocessor_builder.build(step)
 751         for step in train_config.data_augmentation_options
 752     ]
 753     data_augmentation_fn = functools.partial(
 754         augment_input_data,
 755         data_augmentation_options=data_augmentation_options)
 756 
 757     image_resizer_config = config_util.get_image_resizer_config(model_config)
 758     image_resizer_fn = image_resizer_builder.build(image_resizer_config)
 759     keypoint_type_weight = train_input_config.keypoint_type_weight or None
 760     transform_data_fn = functools.partial(
 761         transform_input_data, model_preprocess_fn=model_preprocess_fn,
 762         image_resizer_fn=image_resizer_fn,
 763         num_classes=num_classes,
 764         data_augmentation_fn=data_augmentation_fn,
 765         merge_multiple_boxes=train_config.merge_multiple_label_boxes,
 766         retain_original_image=train_config.retain_original_images,
 767         use_multiclass_scores=train_config.use_multiclass_scores,
 768         use_bfloat16=train_config.use_bfloat16,
 769         keypoint_type_weight=keypoint_type_weight)
 770 
 771     tensor_dict = pad_input_data_to_static_shapes(
 772         tensor_dict=transform_data_fn(tensor_dict),
 773         max_num_boxes=train_input_config.max_number_of_boxes,
 774         num_classes=num_classes,
 775         spatial_image_shape=config_util.get_spatial_image_size(
 776             image_resizer_config),
 777         max_num_context_features=config_util.get_max_num_context_features(
 778             model_config),
 779         context_feature_length=config_util.get_context_feature_length(
 780             model_config))
 781     include_source_id = train_input_config.include_source_id
 782     return (_get_features_dict(tensor_dict, include_source_id),
 783             _get_labels_dict(tensor_dict))
 784   reduce_to_frame_fn = get_reduce_to_frame_fn(train_input_config, True)
 785 
 786   dataset = INPUT_BUILDER_UTIL_MAP['dataset_build'](
 787       train_input_config,
 788       transform_input_data_fn=transform_and_pad_input_data_fn,
 789       batch_size=params['batch_size'] if params else train_config.batch_size,
 790       input_context=input_context,
 791       reduce_to_frame_fn=reduce_to_frame_fn)
 792   return dataset

 936 def create_predict_input_fn(model_config, predict_input_config):
 937   """Creates a predict `input` function for `Estimator`.
 938

 940     model_config: A model_pb2.DetectionModel.

 944     `input_fn` for `Estimator` in PREDICT mode.

 949 
 950     Args:
 951       params: Parameter dictionary passed from the estimator.
 952 
 953     Returns:
 954       `ServingInputReceiver`.
 955     """
 956     del params

 990 def get_reduce_to_frame_fn(input_reader_config, is_training):
 991   """Returns a function reducing sequence tensors to single frame tensors.
 992 
 993   If the input type is not TF_SEQUENCE_EXAMPLE, the tensors are passed through
 994   this function unchanged. Otherwise, when in training mode, a single frame is
 995   selected at random from the sequence example, and the tensors for that frame
 996   are converted to single frame tensors, with all associated context features.
 997   In evaluation mode all frames are converted to single frame tensors with
 998   copied context tensors. After the sequence example tensors are converted into
 999   one or many single frame tensors, the images from each frame are decoded.
1000 
1001   Args:
1002     input_reader_config: An input_reader_pb2.InputReader.
1003     is_training: Whether we are in training mode.
1004 
1005   Returns:
1006     `reduce_to_frame_fn` for the dataset builder
1007   """
1008   if input_reader_config.input_type != (
1009       input_reader_pb2.InputType.TF_SEQUENCE_EXAMPLE):
1010     return lambda d: d
1011   else:
1012     def reduce_to_frame(dataset):
1013       """Returns a function reducing sequence tensors to single frame tensors.
1014 
1015       Args:
1016         dataset: A tf dataset containing sequence tensors.
1017 
1018       Returns:
1019         A tf dataset containing single frame tensors.
1020       """
1021       if is_training:
1022         def get_single_frame(tensor_dict):
1023           """Returns a random frame from a sequence.
1024 
1025           Picks a random frame and returns slices of sequence tensors
1026           corresponding to the random frame. Returns non-sequence tensors
1027           unchanged.
1028 
1029           Args:
1030             tensor_dict: A dictionary containing sequence tensors.
1031 
1032           Returns:
1033             Tensors for a single random frame within the sequence.
1034           """
1035           num_frames = tf.cast(
1036               tf.shape(tensor_dict[fields.InputDataFields.source_id])[0],
1037               dtype=tf.int32)
1038           frame_index = tf.random.uniform((), minval=0, maxval=num_frames,
1039                                           dtype=tf.int32)
1040           out_tensor_dict = {}
1041           for key in tensor_dict:
1042             if key in fields.SEQUENCE_FIELDS:
1043               # Slice random frame from sequence tensors
1044               out_tensor_dict[key] = tensor_dict[key][frame_index]
1045             else:
1046               # Copy all context tensors.
1047               out_tensor_dict[key] = tensor_dict[key]
1048           return out_tensor_dict
