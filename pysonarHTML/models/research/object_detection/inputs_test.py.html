  38 def _get_configs_for_model(model_name):
  39   """Returns configurations for model."""
  40   fname = os.path.join(tf.resource_loader.get_data_files_path(),
  41                        'samples/configs/' + model_name + '.config')
  42   label_map_path = os.path.join(tf.resource_loader.get_data_files_path(),
  43                                 'data/pet_label_map.pbtxt')
  44   data_path = os.path.join(tf.resource_loader.get_data_files_path(),
  45                            'test_data/pets_examples.record')
  46   configs = config_util.get_configs_from_pipeline_file(fname)
  47   override_dict = {
  48       'train_input_path': data_path,
  49       'eval_input_path': data_path,
  50       'label_map_path': label_map_path
  51   }
  52   return config_util.merge_external_params_with_configs(
  53       configs, kwargs_dict=override_dict)

  75 def _make_initializable_iterator(dataset):
  76   """Creates an iterator, and initializes tables.
  77 
  78   Args:
  79     dataset: A `tf.data.Dataset` object.
  80 
  81   Returns:
  82     A `tf.data.Iterator`.
  83   """
  84   iterator = tf.data.make_initializable_iterator(dataset)
  85   tf.add_to_collection(tf.GraphKeys.TABLE_INITIALIZERS, iterator.initializer)
  86   return iterator
