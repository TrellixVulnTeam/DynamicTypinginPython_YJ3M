  90 def _get_configs_for_model(model_name):
  91   """Returns configurations for model."""
  92   filename = get_pipeline_config_path(model_name)
  93   data_path = _get_data_path(model_name)
  94   if model_name == MODEL_NAME_FOR_KEYPOINTS_TEST:
  95     label_map_path = _get_keypoints_labelmap_path()
  96   elif model_name == MODEL_NAME_FOR_SEQUENCE_EXAMPLE_TEST:
  97     label_map_path = _get_sequence_example_labelmap_path()
  98   else:
  99     label_map_path = _get_labelmap_path()
 100   configs = config_util.get_configs_from_pipeline_file(filename)
 101   override_dict = {
 102       'train_input_path': data_path,
 103       'eval_input_path': data_path,
 104       'label_map_path': label_map_path
 105   }
 106   configs = config_util.merge_external_params_with_configs(
 107       configs, kwargs_dict=override_dict)
 108   return configs

 111 def _make_initializable_iterator(dataset):
 112   """Creates an iterator, and initializes tables.
 113 
 114   Args:
 115     dataset: A `tf.data.Dataset` object.
 116 
 117   Returns:
 118     A `tf.data.Iterator`.
 119   """
 120   iterator = tf.data.make_initializable_iterator(dataset)
 121   tf.add_to_collection(tf.GraphKeys.TABLE_INITIALIZERS, iterator.initializer)
 122   return iterator

 131   def _assert_model_fn_for_train_eval(self, configs, mode,
 132                                       class_agnostic=False):
 133     model_config = configs['model']
 134     train_config = configs['train_config']
 135     with tf.Graph().as_default():
 136       if mode == 'train':
 137         features, labels = _make_initializable_iterator(
 138             inputs.create_train_input_fn(configs['train_config'],
 139                                          configs['train_input_config'],
 140                                          configs['model'])()).get_next()
 141         model_mode = tf.estimator.ModeKeys.TRAIN
 142         batch_size = train_config.batch_size
 143       elif mode == 'eval':
 144         features, labels = _make_initializable_iterator(
 145             inputs.create_eval_input_fn(configs['eval_config'],
 146                                         configs['eval_input_config'],
 147                                         configs['model'])()).get_next()
 148         model_mode = tf.estimator.ModeKeys.EVAL
 149         batch_size = 1
 150       elif mode == 'eval_on_train':
 151         features, labels = _make_initializable_iterator(
 152             inputs.create_eval_input_fn(configs['eval_config'],
 153                                         configs['train_input_config'],
 154                                         configs['model'])()).get_next()
 155         model_mode = tf.estimator.ModeKeys.EVAL
 156         batch_size = 1
 157 
 158       detection_model_fn = functools.partial(
 159           model_builder.build, model_config=model_config, is_training=True)
 160 
 161       hparams = model_hparams.create_hparams(
 162           hparams_overrides='load_pretrained=false')
 163 
 164       model_fn = model_lib.create_model_fn(detection_model_fn, configs, hparams)
 165       estimator_spec = model_fn(features, labels, model_mode)
 166 
 167       self.assertIsNotNone(estimator_spec.loss)
 168       self.assertIsNotNone(estimator_spec.predictions)
 169       if mode == 'eval' or mode == 'eval_on_train':
 170         if class_agnostic:
 171           self.assertNotIn('detection_classes', estimator_spec.predictions)
 172         else:
 173           detection_classes = estimator_spec.predictions['detection_classes']
 174           self.assertEqual(batch_size, detection_classes.shape.as_list()[0])
 175           self.assertEqual(tf.float32, detection_classes.dtype)
 176         detection_boxes = estimator_spec.predictions['detection_boxes']
 177         detection_scores = estimator_spec.predictions['detection_scores']
 178         num_detections = estimator_spec.predictions['num_detections']
 179         self.assertEqual(batch_size, detection_boxes.shape.as_list()[0])
 180         self.assertEqual(tf.float32, detection_boxes.dtype)
 181         self.assertEqual(batch_size, detection_scores.shape.as_list()[0])
 182         self.assertEqual(tf.float32, detection_scores.dtype)
 183         self.assertEqual(tf.float32, num_detections.dtype)
 184         if mode == 'eval':
 185           self.assertIn('Detections_Left_Groundtruth_Right/0',
 186                         estimator_spec.eval_metric_ops)
 187       if model_mode == tf.estimator.ModeKeys.TRAIN:
 188         self.assertIsNotNone(estimator_spec.train_op)
 189       return estimator_spec

 411   def test_unbatch_without_unpadding(self):
 412     image_placeholder = tf.placeholder(tf.float32, [2, None, None, None])
 413     groundtruth_boxes_placeholder = tf.placeholder(tf.float32, [2, None, None])
 414     groundtruth_classes_placeholder = tf.placeholder(tf.float32,

 419         fields.InputDataFields.image:
 420             image_placeholder,
 421         fields.InputDataFields.groundtruth_boxes:

 423         fields.InputDataFields.groundtruth_classes:

 425         fields.InputDataFields.groundtruth_weights:

 443           })

 445       self.assertAllEqual(image_out.shape, [4, 4, 3])
 446     for groundtruth_boxes_out in unbatched_tensor_dict_out[
 447         fields.InputDataFields.groundtruth_boxes]:
 448       self.assertAllEqual(groundtruth_boxes_out.shape, [5, 4])
 449     for groundtruth_classes_out in unbatched_tensor_dict_out[
 450         fields.InputDataFields.groundtruth_classes]:

 452     for groundtruth_weights_out in unbatched_tensor_dict_out[
 453         fields.InputDataFields.groundtruth_weights]:
 454       self.assertAllEqual(groundtruth_weights_out.shape, [5])
