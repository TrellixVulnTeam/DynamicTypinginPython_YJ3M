  64 def _get_config_kwarg_overrides():
  65   """Returns overrides to the configs that insert the correct local paths."""
  66   data_path = _get_data_path()
  67   label_map_path = _get_labelmap_path()

  69       'train_input_path': data_path,
  70       'eval_input_path': data_path,
  71       'label_map_path': label_map_path

 109 class SimpleModel(model.DetectionModel):
 110   """A model with a single weight vector."""
 111 
 112   def __init__(self, num_classes=1):
 113     super(SimpleModel, self).__init__(num_classes)
 114     self.weight = tf.keras.backend.variable(np.ones(10), name='weight')
 115 
 116   def postprocess(self, prediction_dict, true_image_shapes):
 117     return {}
 118 
 119   def updates(self):
 120     return []
 121 
 122   def restore_map(self, *args, **kwargs):
 123     return {'model': self}
 124 
 125   def preprocess(self, _):
 126     return tf.zeros((1, 128, 128, 3)), tf.constant([[128, 128, 3]])
 127 
 128   def provide_groundtruth(self, *args, **kwargs):
 129     pass
 130 
 131   def predict(self, pred_inputs, true_image_shapes):
 132     return {'prediction':
 133             tf.abs(tf.reduce_sum(self.weight) * tf.reduce_sum(pred_inputs))}
 134 
 135   def loss(self, prediction_dict, _):
 136     return {'loss': tf.reduce_sum(prediction_dict['prediction'])}
 137 
 138   def regularization_losses(self):
 139     return []

 142 class ModelCheckpointTest(tf.test.TestCase):
 143   """Test for model checkpoint related functionality."""
 144 
 145   def test_checkpoint_max_to_keep(self):
 146     """Test that only the most recent checkpoints are kept."""
 147 
 148     with mock.patch.object(
 149         model_builder, 'build', autospec=True) as mock_builder:
 150       mock_builder.return_value = SimpleModel()
 151 
 152       hparams = model_hparams.create_hparams(
 153           hparams_overrides='load_pretrained=false')
 154       pipeline_config_path = get_pipeline_config_path(MODEL_NAME_FOR_TEST)
 155       config_kwarg_overrides = _get_config_kwarg_overrides()
 156       model_dir = tempfile.mkdtemp(dir=self.get_temp_dir())
 157 
 158       model_lib_v2.train_loop(
 159           hparams, pipeline_config_path, model_dir=model_dir,
 160           train_steps=20, checkpoint_every_n=2, checkpoint_max_to_keep=3,
 161           **config_kwarg_overrides
 162       )
 163       ckpt_files = tf.io.gfile.glob(os.path.join(model_dir, 'ckpt-*.index'))
 164       self.assertEqual(len(ckpt_files), 3,
 165                        '{} not of length 3.'.format(ckpt_files))

 174 class CheckpointV2Test(tf.test.TestCase):
 175 
 176   def setUp(self):
 177     super(CheckpointV2Test, self).setUp()
 178 
 179     self._model = SimpleModel()
 180     tf.keras.backend.set_value(self._model.weight, np.ones(10) * 42)
 181     ckpt = tf.train.Checkpoint(model=self._model)
 182 
 183     self._test_dir = tf.test.get_temp_dir()
 184     self._ckpt_path = ckpt.save(os.path.join(self._test_dir, 'ckpt'))
 185     tf.keras.backend.set_value(self._model.weight, np.ones(10))
 186 
 187     pipeline_config_path = get_pipeline_config_path(MODEL_NAME_FOR_TEST)
 188     configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
 189     configs = config_util.merge_external_params_with_configs(
 190         configs, kwargs_dict=_get_config_kwarg_overrides())
 191     self._train_input_fn = inputs.create_train_input_fn(
 192         configs['train_config'],
 193         configs['train_input_config'],
 194         configs['model'])
 195 
 196   def test_restore_v2(self):
 197     """Test that restoring a v2 style checkpoint works."""
 198 
 199     model_lib_v2.load_fine_tune_checkpoint(
 200         self._model, self._ckpt_path, checkpoint_type='',
 201         checkpoint_version=train_pb2.CheckpointVersion.V2,
 202         load_all_detection_checkpoint_vars=True,
 203         input_dataset=self._train_input_fn(),
 204         unpad_groundtruth_tensors=True)
 205     np.testing.assert_allclose(self._model.weight.numpy(), 42)
 206 
 207   def test_restore_map_incompatible_error(self):
 208     """Test that restoring an incompatible restore map causes an error."""
 209 
 210     with self.assertRaisesRegex(TypeError,
 211                                 r'.*received a \(str -> ResourceVariable\).*'):
 212       model_lib_v2.load_fine_tune_checkpoint(
 213           IncompatibleModel(), self._ckpt_path, checkpoint_type='',
 214           checkpoint_version=train_pb2.CheckpointVersion.V2,
 215           load_all_detection_checkpoint_vars=True,
 216           input_dataset=self._train_input_fn(),
 217           unpad_groundtruth_tensors=True)
