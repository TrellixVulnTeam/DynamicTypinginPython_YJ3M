  25 def build(post_processing_config):
  26   """Builds callables for post-processing operations.
  27 
  28   Builds callables for non-max suppression, score conversion, and (optionally)
  29   calibration based on the configuration.
  30 
  31   Non-max suppression callable takes `boxes`, `scores`, and optionally
  32   `clip_window`, `parallel_iterations` `masks, and `scope` as inputs. It returns
  33   `nms_boxes`, `nms_scores`, `nms_classes` `nms_masks` and `num_detections`. See
  34   post_processing.batch_multiclass_non_max_suppression for the type and shape
  35   of these tensors.
  36 
  37   Score converter callable should be called with `input` tensor. The callable
  38   returns the output from one of 3 tf operations based on the configuration -
  39   tf.identity, tf.sigmoid or tf.nn.softmax. If a calibration config is provided,
  40   score_converter also applies calibration transformations, as defined in
  41   calibration_builder.py. See tensorflow documentation for argument and return
  42   value descriptions.
  43 
  44   Args:
  45     post_processing_config: post_processing.proto object containing the
  46       parameters for the post-processing operations.
  47 
  48   Returns:
  49     non_max_suppressor_fn: Callable for non-max suppression.
  50     score_converter_fn: Callable for score conversion.
  51 
  52   Raises:
  53     ValueError: if the post_processing_config is of incorrect type.
  54   """
  55   if not isinstance(post_processing_config, post_processing_pb2.PostProcessing):
  56     raise ValueError('post_processing_config not of type '
  57                      'post_processing_pb2.Postprocessing.')
  58   non_max_suppressor_fn = _build_non_max_suppressor(
  59       post_processing_config.batch_non_max_suppression)
  60   score_converter_fn = _build_score_converter(
  61       post_processing_config.score_converter,
  62       post_processing_config.logit_scale)
  63   if post_processing_config.HasField('calibration_config'):
  64     score_converter_fn = _build_calibrated_score_converter(
  65         score_converter_fn,
  66         post_processing_config.calibration_config)
  67   return non_max_suppressor_fn, score_converter_fn
  68

  70 def _build_non_max_suppressor(nms_config):
  71   """Builds non-max suppresson based on the nms config.
  72 
  73   Args:
  74     nms_config: post_processing_pb2.PostProcessing.BatchNonMaxSuppression proto.
  75 
  76   Returns:
  77     non_max_suppressor_fn: Callable non-max suppressor.
  78 
  79   Raises:
  80     ValueError: On incorrect iou_threshold or on incompatible values of
  81       max_total_detections and max_detections_per_class or on negative
  82       soft_nms_sigma.
  83   """
  84   if nms_config.iou_threshold < 0 or nms_config.iou_threshold > 1.0:
  85     raise ValueError('iou_threshold not in [0, 1.0].')
  86   if nms_config.max_detections_per_class > nms_config.max_total_detections:
  87     raise ValueError('max_detections_per_class should be no greater than '
  88                      'max_total_detections.')
  89   if nms_config.soft_nms_sigma < 0.0:
  90     raise ValueError('soft_nms_sigma should be non-negative.')
  91   if nms_config.use_combined_nms and nms_config.use_class_agnostic_nms:
  92     raise ValueError('combined_nms does not support class_agnostic_nms.')
  93   non_max_suppressor_fn = functools.partial(
  94       post_processing.batch_multiclass_non_max_suppression,
  95       score_thresh=nms_config.score_threshold,
  96       iou_thresh=nms_config.iou_threshold,
  97       max_size_per_class=nms_config.max_detections_per_class,
  98       max_total_size=nms_config.max_total_detections,
  99       use_static_shapes=nms_config.use_static_shapes,
 100       use_class_agnostic_nms=nms_config.use_class_agnostic_nms,
 101       max_classes_per_detection=nms_config.max_classes_per_detection,
 102       soft_nms_sigma=nms_config.soft_nms_sigma,
 103       use_partitioned_nms=nms_config.use_partitioned_nms,
 104       use_combined_nms=nms_config.use_combined_nms,
 105       change_coordinate_frame=nms_config.change_coordinate_frame,
 106       use_hard_nms=nms_config.use_hard_nms)
 107 
 108   return non_max_suppressor_fn

 111 def _score_converter_fn_with_logit_scale(tf_score_converter_fn, logit_scale):
 112   """Create a function to scale logits then apply a Tensorflow function."""
 113   def score_converter_fn(logits):
 114     scaled_logits = tf.multiply(logits, 1.0 / logit_scale, name='scale_logits')
 115     return tf_score_converter_fn(scaled_logits, name='convert_scores')
 116   score_converter_fn.__name__ = '%s_with_logit_scale' % (
 117       tf_score_converter_fn.__name__)
 118   return score_converter_fn

 121 def _build_score_converter(score_converter_config, logit_scale):
 122   """Builds score converter based on the config.
 123 
 124   Builds one of [tf.identity, tf.sigmoid, tf.softmax] score converters based on
 125   the config.
 126 
 127   Args:
 128     score_converter_config: post_processing_pb2.PostProcessing.score_converter.
 129     logit_scale: temperature to use for SOFTMAX score_converter.
 130 
 131   Returns:
 132     Callable score converter op.
 133 
 134   Raises:
 135     ValueError: On unknown score converter.
 136   """
 137   if score_converter_config == post_processing_pb2.PostProcessing.IDENTITY:
 138     return _score_converter_fn_with_logit_scale(tf.identity, logit_scale)
 139   if score_converter_config == post_processing_pb2.PostProcessing.SIGMOID:
 140     return _score_converter_fn_with_logit_scale(tf.sigmoid, logit_scale)
 141   if score_converter_config == post_processing_pb2.PostProcessing.SOFTMAX:
 142     return _score_converter_fn_with_logit_scale(tf.nn.softmax, logit_scale)
 143   raise ValueError('Unknown score converter.')

 146 def _build_calibrated_score_converter(score_converter_fn, calibration_config):
 147   """Wraps a score_converter_fn, adding a calibration step.
 148 
 149   Builds a score converter function with a calibration transformation according
 150   to calibration_builder.py. The score conversion function may be applied before
 151   or after the calibration transformation, depending on the calibration method.
 152   If the method is temperature scaling, the score conversion is
 153   after the calibration transformation. Otherwise, the score conversion is
 154   before the calibration transformation. Calibration applies positive monotonic
 155   transformations to inputs (i.e. score ordering is strictly preserved or
 156   adjacent scores are mapped to the same score). When calibration is
 157   class-agnostic, the highest-scoring class remains unchanged, unless two
 158   adjacent scores are mapped to the same value and one class arbitrarily
 159   selected to break the tie. In per-class calibration, it's possible (though
 160   rare in practice) that the highest-scoring class will change, since positive
 161   monotonicity is only required to hold within each class.
 162 
 163   Args:
 164     score_converter_fn: callable that takes logit scores as input.
 165     calibration_config: post_processing_pb2.PostProcessing.calibration_config.
 166 
 167   Returns:
 168     Callable calibrated score coverter op.
 169   """
 170   calibration_fn = calibration_builder.build(calibration_config)
 171   def calibrated_score_converter_fn(logits):
 172     if (calibration_config.WhichOneof('calibrator') ==
 173         'temperature_scaling_calibration'):
 174       calibrated_logits = calibration_fn(logits)
 175       return score_converter_fn(calibrated_logits)
 176     else:
 177       converted_logits = score_converter_fn(logits)
 178       return calibration_fn(converted_logits)
 179 
 180   calibrated_score_converter_fn.__name__ = (
 181       'calibrate_with_%s' % calibration_config.WhichOneof('calibrator'))
 182   return calibrated_score_converter_fn
