  24 def _get_step_config_from_proto(preprocessor_step_config, step_name):
  25   """Returns the value of a field named step_name from proto.
  26 
  27   Args:
  28     preprocessor_step_config: A preprocessor_pb2.PreprocessingStep object.
  29     step_name: Name of the field to get value from.
  30 
  31   Returns:
  32     result_dict: a sub proto message from preprocessor_step_config which will be
  33                  later converted to a dictionary.
  34 
  35   Raises:
  36     ValueError: If field does not exist in proto.
  37   """
  38   for field, value in preprocessor_step_config.ListFields():
  39     if field.name == step_name:
  40       return value
  41 
  42   raise ValueError('Could not get field %s from proto!' % step_name)

  45 def _get_dict_from_proto(config):
  46   """Helper function to put all proto fields into a dictionary.
  47 
  48   For many preprocessing steps, there's an trivial 1-1 mapping from proto fields
  49   to function arguments. This function automatically populates a dictionary with
  50   the arguments from the proto.
  51 
  52   Protos that CANNOT be trivially populated include:
  53   * nested messages.
  54   * steps that check if an optional field is set (ie. where None != 0).
  55   * protos that don't map 1-1 to arguments (ie. list should be reshaped).
  56   * fields requiring additional validation (ie. repeated field has n elements).
  57 
  58   Args:
  59     config: A protobuf object that does not violate the conditions above.
  60 
  61   Returns:
  62     result_dict: |config| converted into a python dictionary.
  63   """
  64   result_dict = {}
  65   for field, value in config.ListFields():
  66     result_dict[field.name] = value
  67   return result_dict

 126 def build(preprocessor_step_config):
 127   """Builds preprocessing step based on the configuration.
 128 
 129   Args:
 130     preprocessor_step_config: PreprocessingStep configuration proto.
 131 
 132   Returns:
 133     function, argmap: A callable function and an argument map to call function
 134                       with.
 135 
 136   Raises:
 137     ValueError: On invalid configuration.
 138   """
 139   step_type = preprocessor_step_config.WhichOneof('preprocessing_step')
 140 
 141   if step_type in PREPROCESSING_FUNCTION_MAP:
 142     preprocessing_function = PREPROCESSING_FUNCTION_MAP[step_type]
 143     step_config = _get_step_config_from_proto(preprocessor_step_config,
 144                                               step_type)
 145     function_args = _get_dict_from_proto(step_config)
 146     return (preprocessing_function, function_args)
 147 
 148   if step_type == 'random_horizontal_flip':
 149     config = preprocessor_step_config.random_horizontal_flip
 150     return (preprocessor.random_horizontal_flip,
 151             {
 152                 'keypoint_flip_permutation': tuple(
 153                     config.keypoint_flip_permutation) or None,
 154             })
 155 
 156   if step_type == 'random_vertical_flip':
 157     config = preprocessor_step_config.random_vertical_flip
 158     return (preprocessor.random_vertical_flip,
 159             {
 160                 'keypoint_flip_permutation': tuple(
 161                     config.keypoint_flip_permutation) or None,
 162             })
 163 
 164   if step_type == 'random_rotation90':
 165     return (preprocessor.random_rotation90, {})
 166 
 167   if step_type == 'random_crop_image':
 168     config = preprocessor_step_config.random_crop_image
 169     return (preprocessor.random_crop_image,
 170             {
 171                 'min_object_covered': config.min_object_covered,
 172                 'aspect_ratio_range': (config.min_aspect_ratio,
 173                                        config.max_aspect_ratio),
 174                 'area_range': (config.min_area, config.max_area),
 175                 'overlap_thresh': config.overlap_thresh,
 176                 'clip_boxes': config.clip_boxes,
 177                 'random_coef': config.random_coef,
 178             })
 179 
 180   if step_type == 'random_pad_image':
 181     config = preprocessor_step_config.random_pad_image
 182     min_image_size = None
 183     if (config.HasField('min_image_height') !=
 184         config.HasField('min_image_width')):
 185       raise ValueError('min_image_height and min_image_width should be either '
 186                        'both set or both unset.')
 187     if config.HasField('min_image_height'):
 188       min_image_size = (config.min_image_height, config.min_image_width)
 189 
 190     max_image_size = None
 191     if (config.HasField('max_image_height') !=
 192         config.HasField('max_image_width')):
 193       raise ValueError('max_image_height and max_image_width should be either '
 194                        'both set or both unset.')
 195     if config.HasField('max_image_height'):
 196       max_image_size = (config.max_image_height, config.max_image_width)
 197 
 198     pad_color = config.pad_color or None
 199     if pad_color:
 200       if len(pad_color) != 3:
 201         tf.logging.warn('pad_color should have 3 elements (RGB) if set!')
 202 
 203       pad_color = tf.cast([x for x in config.pad_color], dtype=tf.float32)
 204     return (preprocessor.random_pad_image,
 205             {
 206                 'min_image_size': min_image_size,
 207                 'max_image_size': max_image_size,
 208                 'pad_color': pad_color,
 209             })
 210 
 211   if step_type == 'random_absolute_pad_image':
 212     config = preprocessor_step_config.random_absolute_pad_image
 213 
 214     max_height_padding = config.max_height_padding or 1
 215     max_width_padding = config.max_width_padding or 1
 216 
 217     pad_color = config.pad_color or None
 218     if pad_color:
 219       if len(pad_color) != 3:
 220         tf.logging.warn('pad_color should have 3 elements (RGB) if set!')
 221 
 222       pad_color = tf.cast([x for x in config.pad_color], dtype=tf.float32)
 223 
 224     return (preprocessor.random_absolute_pad_image,
 225             {
 226                 'max_height_padding': max_height_padding,
 227                 'max_width_padding': max_width_padding,
 228                 'pad_color': pad_color,
 229             })
 230   if step_type == 'random_crop_pad_image':
 231     config = preprocessor_step_config.random_crop_pad_image
 232     min_padded_size_ratio = config.min_padded_size_ratio
 233     if min_padded_size_ratio and len(min_padded_size_ratio) != 2:
 234       raise ValueError('min_padded_size_ratio should have 2 elements if set!')
 235     max_padded_size_ratio = config.max_padded_size_ratio
 236     if max_padded_size_ratio and len(max_padded_size_ratio) != 2:
 237       raise ValueError('max_padded_size_ratio should have 2 elements if set!')
 238     pad_color = config.pad_color or None
 239     if pad_color:
 240       if len(pad_color) != 3:
 241         tf.logging.warn('pad_color should have 3 elements (RGB) if set!')
 242 
 243       pad_color = tf.cast([x for x in config.pad_color], dtype=tf.float32)
 244 
 245     kwargs = {
 246         'min_object_covered': config.min_object_covered,
 247         'aspect_ratio_range': (config.min_aspect_ratio,
 248                                config.max_aspect_ratio),
 249         'area_range': (config.min_area, config.max_area),
 250         'overlap_thresh': config.overlap_thresh,
 251         'clip_boxes': config.clip_boxes,
 252         'random_coef': config.random_coef,
 253         'pad_color': pad_color,
 254     }
 255     if min_padded_size_ratio:
 256       kwargs['min_padded_size_ratio'] = tuple(min_padded_size_ratio)
 257     if max_padded_size_ratio:
 258       kwargs['max_padded_size_ratio'] = tuple(max_padded_size_ratio)
 259     return (preprocessor.random_crop_pad_image, kwargs)
 260 
 261   if step_type == 'random_resize_method':
 262     config = preprocessor_step_config.random_resize_method
 263     return (preprocessor.random_resize_method,
 264             {
 265                 'target_size': [config.target_height, config.target_width],
 266             })
 267 
 268   if step_type == 'resize_image':
 269     config = preprocessor_step_config.resize_image
 270     method = RESIZE_METHOD_MAP[config.method]
 271     return (preprocessor.resize_image,
 272             {
 273                 'new_height': config.new_height,
 274                 'new_width': config.new_width,
 275                 'method': method
 276             })
 277 
 278   if step_type == 'random_self_concat_image':
 279     config = preprocessor_step_config.random_self_concat_image
 280     return (preprocessor.random_self_concat_image, {
 281         'concat_vertical_probability': config.concat_vertical_probability,
 282         'concat_horizontal_probability': config.concat_horizontal_probability
 283     })
 284 
 285   if step_type == 'ssd_random_crop':
 286     config = preprocessor_step_config.ssd_random_crop
 287     if config.operations:
 288       min_object_covered = [op.min_object_covered for op in config.operations]
 289       aspect_ratio_range = [(op.min_aspect_ratio, op.max_aspect_ratio)
 290                             for op in config.operations]
 291       area_range = [(op.min_area, op.max_area) for op in config.operations]
 292       overlap_thresh = [op.overlap_thresh for op in config.operations]
 293       clip_boxes = [op.clip_boxes for op in config.operations]
 294       random_coef = [op.random_coef for op in config.operations]
 295       return (preprocessor.ssd_random_crop,
 296               {
 297                   'min_object_covered': min_object_covered,
 298                   'aspect_ratio_range': aspect_ratio_range,
 299                   'area_range': area_range,
 300                   'overlap_thresh': overlap_thresh,
 301                   'clip_boxes': clip_boxes,
 302                   'random_coef': random_coef,
 303               })
 304     return (preprocessor.ssd_random_crop, {})
 305 
 306   if step_type == 'autoaugment_image':
 307     config = preprocessor_step_config.autoaugment_image
 308     return (preprocessor.autoaugment_image, {
 309         'policy_name': config.policy_name,
 310     })
 311 
 312   if step_type == 'drop_label_probabilistically':
 313     config = preprocessor_step_config.drop_label_probabilistically
 314     return (preprocessor.drop_label_probabilistically, {
 315         'dropped_label': config.label,
 316         'drop_probability': config.drop_probability,
 317     })
 318 
 319   if step_type == 'remap_labels':
 320     config = preprocessor_step_config.remap_labels
 321     return (preprocessor.remap_labels, {
 322         'original_labels': config.original_labels,
 323         'new_label': config.new_label
 324     })
 325 
 326   if step_type == 'ssd_random_crop_pad':
 327     config = preprocessor_step_config.ssd_random_crop_pad
 328     if config.operations:
 329       min_object_covered = [op.min_object_covered for op in config.operations]
 330       aspect_ratio_range = [(op.min_aspect_ratio, op.max_aspect_ratio)
 331                             for op in config.operations]
 332       area_range = [(op.min_area, op.max_area) for op in config.operations]
 333       overlap_thresh = [op.overlap_thresh for op in config.operations]
 334       clip_boxes = [op.clip_boxes for op in config.operations]
 335       random_coef = [op.random_coef for op in config.operations]
 336       min_padded_size_ratio = [tuple(op.min_padded_size_ratio)
 337                                for op in config.operations]
 338       max_padded_size_ratio = [tuple(op.max_padded_size_ratio)
 339                                for op in config.operations]
 340       pad_color = [(op.pad_color_r, op.pad_color_g, op.pad_color_b)
 341                    for op in config.operations]
 342       return (preprocessor.ssd_random_crop_pad,
 343               {
 344                   'min_object_covered': min_object_covered,
 345                   'aspect_ratio_range': aspect_ratio_range,
 346                   'area_range': area_range,
 347                   'overlap_thresh': overlap_thresh,
 348                   'clip_boxes': clip_boxes,
 349                   'random_coef': random_coef,
 350                   'min_padded_size_ratio': min_padded_size_ratio,
 351                   'max_padded_size_ratio': max_padded_size_ratio,
 352                   'pad_color': pad_color,
 353               })
 354     return (preprocessor.ssd_random_crop_pad, {})
 355 
 356   if step_type == 'ssd_random_crop_fixed_aspect_ratio':
 357     config = preprocessor_step_config.ssd_random_crop_fixed_aspect_ratio
 358     if config.operations:
 359       min_object_covered = [op.min_object_covered for op in config.operations]
 360       area_range = [(op.min_area, op.max_area) for op in config.operations]
 361       overlap_thresh = [op.overlap_thresh for op in config.operations]
 362       clip_boxes = [op.clip_boxes for op in config.operations]
 363       random_coef = [op.random_coef for op in config.operations]
 364       return (preprocessor.ssd_random_crop_fixed_aspect_ratio,
 365               {
 366                   'min_object_covered': min_object_covered,
 367                   'aspect_ratio': config.aspect_ratio,
 368                   'area_range': area_range,
 369                   'overlap_thresh': overlap_thresh,
 370                   'clip_boxes': clip_boxes,
 371                   'random_coef': random_coef,
 372               })
 373     return (preprocessor.ssd_random_crop_fixed_aspect_ratio, {})
 374 
 375   if step_type == 'ssd_random_crop_pad_fixed_aspect_ratio':
 376     config = preprocessor_step_config.ssd_random_crop_pad_fixed_aspect_ratio
 377     kwargs = {}
 378     aspect_ratio = config.aspect_ratio
 379     if aspect_ratio:
 380       kwargs['aspect_ratio'] = aspect_ratio
 381     min_padded_size_ratio = config.min_padded_size_ratio
 382     if min_padded_size_ratio:
 383       if len(min_padded_size_ratio) != 2:
 384         raise ValueError('min_padded_size_ratio should have 2 elements if set!')
 385       kwargs['min_padded_size_ratio'] = tuple(min_padded_size_ratio)
 386     max_padded_size_ratio = config.max_padded_size_ratio
 387     if max_padded_size_ratio:
 388       if len(max_padded_size_ratio) != 2:
 389         raise ValueError('max_padded_size_ratio should have 2 elements if set!')
 390       kwargs['max_padded_size_ratio'] = tuple(max_padded_size_ratio)
 391     if config.operations:
 392       kwargs['min_object_covered'] = [op.min_object_covered
 393                                       for op in config.operations]
 394       kwargs['aspect_ratio_range'] = [(op.min_aspect_ratio, op.max_aspect_ratio)
 395                                       for op in config.operations]
 396       kwargs['area_range'] = [(op.min_area, op.max_area)
 397                               for op in config.operations]
 398       kwargs['overlap_thresh'] = [op.overlap_thresh for op in config.operations]
 399       kwargs['clip_boxes'] = [op.clip_boxes for op in config.operations]
 400       kwargs['random_coef'] = [op.random_coef for op in config.operations]
 401     return (preprocessor.ssd_random_crop_pad_fixed_aspect_ratio, kwargs)
 402 
 403   if step_type == 'random_square_crop_by_scale':
 404     config = preprocessor_step_config.random_square_crop_by_scale
 405     return preprocessor.random_square_crop_by_scale, {
 406         'scale_min': config.scale_min,
 407         'scale_max': config.scale_max,
 408         'max_border': config.max_border,
 409         'num_scales': config.num_scales
 410     }
 411 
 412   raise ValueError('Unknown preprocessing step.')
