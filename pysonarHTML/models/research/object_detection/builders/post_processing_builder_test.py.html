  26   def test_build_non_max_suppressor_with_correct_parameters(self):
  27     post_processing_text_proto = """
  28       batch_non_max_suppression {
  29         score_threshold: 0.7
  30         iou_threshold: 0.6
  31         max_detections_per_class: 100
  32         max_total_detections: 300
  33         soft_nms_sigma: 0.4
  34       }
  35     """
  36     post_processing_config = post_processing_pb2.PostProcessing()
  37     text_format.Merge(post_processing_text_proto, post_processing_config)
  38     non_max_suppressor, _ = post_processing_builder.build(
  39         post_processing_config)
  40     self.assertEqual(non_max_suppressor.keywords['max_size_per_class'], 100)
  41     self.assertEqual(non_max_suppressor.keywords['max_total_size'], 300)
  42     self.assertAlmostEqual(non_max_suppressor.keywords['score_thresh'], 0.7)
  43     self.assertAlmostEqual(non_max_suppressor.keywords['iou_thresh'], 0.6)
  44     self.assertAlmostEqual(non_max_suppressor.keywords['soft_nms_sigma'], 0.4)

  71   def test_build_identity_score_converter(self):
  72     post_processing_text_proto = """
  73       score_converter: IDENTITY
  74     """
  75     post_processing_config = post_processing_pb2.PostProcessing()
  76     text_format.Merge(post_processing_text_proto, post_processing_config)
  77     _, score_converter = post_processing_builder.build(
  78         post_processing_config)
  79     self.assertEqual(score_converter.__name__, 'identity_with_logit_scale')
  80 
  81     inputs = tf.constant([1, 1], tf.float32)
  82     outputs = score_converter(inputs)
  83     with self.test_session() as sess:
  84       converted_scores = sess.run(outputs)
  85       expected_converted_scores = sess.run(inputs)
  86       self.assertAllClose(converted_scores, expected_converted_scores)

 133   def test_build_calibrator_with_nonempty_config(self):
 134     """Test that identity function used when no calibration_config specified."""
 135     # Calibration config maps all scores to 0.5.
 136     post_processing_text_proto = """
 137       score_converter: SOFTMAX
 138       calibration_config {
 139         function_approximation {
 140           x_y_pairs {
 141               x_y_pair {
 142                 x: 0.0
 143                 y: 0.5
 144               }
 145               x_y_pair {
 146                 x: 1.0
 147                 y: 0.5
 148               }}}}"""
 149     post_processing_config = post_processing_pb2.PostProcessing()
 150     text_format.Merge(post_processing_text_proto, post_processing_config)
 151     _, calibrated_score_conversion_fn = post_processing_builder.build(
 152         post_processing_config)
 153     self.assertEqual(calibrated_score_conversion_fn.__name__,
 154                      'calibrate_with_function_approximation')
 155 
 156     input_scores = tf.constant([1, 1], tf.float32)
 157     outputs = calibrated_score_conversion_fn(input_scores)
 158     with self.test_session() as sess:
 159       calibrated_scores = sess.run(outputs)
 160       expected_calibrated_scores = sess.run(tf.constant([0.5, 0.5], tf.float32))
 161       self.assertAllClose(calibrated_scores, expected_calibrated_scores)
