  25 def build_optimizers_tf_v1(optimizer_config, global_step=None):
  26   """Create a TF v1 compatible optimizer based on config.
  27 
  28   Args:
  29     optimizer_config: A Optimizer proto message.
  30     global_step: A variable representing the current step.
  31       If None, defaults to tf.train.get_or_create_global_step()
  32 
  33   Returns:
  34     An optimizer and a list of variables for summary.
  35 
  36   Raises:
  37     ValueError: when using an unsupported input data type.
  38   """
  39   optimizer_type = optimizer_config.WhichOneof('optimizer')
  40   optimizer = None
  41 
  42   summary_vars = []
  43   if optimizer_type == 'rms_prop_optimizer':
  44     config = optimizer_config.rms_prop_optimizer
  45     learning_rate = _create_learning_rate(config.learning_rate,
  46                                           global_step=global_step)
  47     summary_vars.append(learning_rate)
  48     optimizer = tf.train.RMSPropOptimizer(
  49         learning_rate,
  50         decay=config.decay,
  51         momentum=config.momentum_optimizer_value,
  52         epsilon=config.epsilon)
  53 
  54   if optimizer_type == 'momentum_optimizer':
  55     config = optimizer_config.momentum_optimizer
  56     learning_rate = _create_learning_rate(config.learning_rate,
  57                                           global_step=global_step)
  58     summary_vars.append(learning_rate)
  59     optimizer = tf.train.MomentumOptimizer(
  60         learning_rate,
  61         momentum=config.momentum_optimizer_value)
  62 
  63   if optimizer_type == 'adam_optimizer':
  64     config = optimizer_config.adam_optimizer
  65     learning_rate = _create_learning_rate(config.learning_rate,
  66                                           global_step=global_step)
  67     summary_vars.append(learning_rate)
  68     optimizer = tf.train.AdamOptimizer(learning_rate, epsilon=config.epsilon)
  69 
  70 
  71   if optimizer is None:
  72     raise ValueError('Optimizer %s not supported.' % optimizer_type)
  73 
  74   if optimizer_config.use_moving_average:
  75     optimizer = tf_opt.MovingAverageOptimizer(
  76         optimizer, average_decay=optimizer_config.moving_average_decay)
  77 
  78   return optimizer, summary_vars

  81 def build_optimizers_tf_v2(optimizer_config, global_step=None):
  82   """Create a TF v2 compatible optimizer based on config.
  83 
  84   Args:
  85     optimizer_config: A Optimizer proto message.
  86     global_step: A variable representing the current step.
  87       If None, defaults to tf.train.get_or_create_global_step()
  88 
  89   Returns:
  90     An optimizer and a list of variables for summary.
  91 
  92   Raises:
  93     ValueError: when using an unsupported input data type.
  94   """
  95   optimizer_type = optimizer_config.WhichOneof('optimizer')
  96   optimizer = None
  97 
  98   summary_vars = []
  99   if optimizer_type == 'rms_prop_optimizer':
 100     config = optimizer_config.rms_prop_optimizer
 101     learning_rate = _create_learning_rate(config.learning_rate,
 102                                           global_step=global_step)
 103     summary_vars.append(learning_rate)
 104     optimizer = tf.keras.optimizers.RMSprop(
 105         learning_rate,
 106         decay=config.decay,
 107         momentum=config.momentum_optimizer_value,
 108         epsilon=config.epsilon)
 109 
 110   if optimizer_type == 'momentum_optimizer':
 111     config = optimizer_config.momentum_optimizer
 112     learning_rate = _create_learning_rate(config.learning_rate,
 113                                           global_step=global_step)
 114     summary_vars.append(learning_rate)
 115     optimizer = tf.keras.optimizers.SGD(
 116         learning_rate,
 117         momentum=config.momentum_optimizer_value)
 118 
 119   if optimizer_type == 'adam_optimizer':
 120     config = optimizer_config.adam_optimizer
 121     learning_rate = _create_learning_rate(config.learning_rate,
 122                                           global_step=global_step)
 123     summary_vars.append(learning_rate)
 124     optimizer = tf.keras.optimizers.Adam(learning_rate, epsilon=config.epsilon)
 125 
 126   if optimizer is None:
 127     raise ValueError('Optimizer %s not supported.' % optimizer_type)
 128 
 129   if optimizer_config.use_moving_average:
 130     raise ValueError('Moving average not supported in eager mode.')
 131 
 132   return optimizer, summary_vars

 135 def build(config, global_step=None):
 136 
 137   if tf.executing_eagerly():
 138     return build_optimizers_tf_v2(config, global_step)
 139   else:
 140     return build_optimizers_tf_v1(config, global_step)

 143 def _create_learning_rate(learning_rate_config, global_step=None):
 144   """Create optimizer learning rate based on config.
 145 
 146   Args:
 147     learning_rate_config: A LearningRate proto message.
 148     global_step: A variable representing the current step.
 149       If None, defaults to tf.train.get_or_create_global_step()
 150 
 151   Returns:
 152     A learning rate.
 153 
 154   Raises:
 155     ValueError: when using an unsupported input data type.
 156   """
 157   if global_step is None:
 158     global_step = tf.train.get_or_create_global_step()
 159   learning_rate = None
 160   learning_rate_type = learning_rate_config.WhichOneof('learning_rate')
 161   if learning_rate_type == 'constant_learning_rate':
 162     config = learning_rate_config.constant_learning_rate
 163     learning_rate = tf.constant(config.learning_rate, dtype=tf.float32,
 164                                 name='learning_rate')
 165 
 166   if learning_rate_type == 'exponential_decay_learning_rate':
 167     config = learning_rate_config.exponential_decay_learning_rate
 168     learning_rate = learning_schedules.exponential_decay_with_burnin(
 169         global_step,
 170         config.initial_learning_rate,
 171         config.decay_steps,
 172         config.decay_factor,
 173         burnin_learning_rate=config.burnin_learning_rate,
 174         burnin_steps=config.burnin_steps,
 175         min_learning_rate=config.min_learning_rate,
 176         staircase=config.staircase)
 177 
 178   if learning_rate_type == 'manual_step_learning_rate':
 179     config = learning_rate_config.manual_step_learning_rate
 180     if not config.schedule:
 181       raise ValueError('Empty learning rate schedule.')
 182     learning_rate_step_boundaries = [x.step for x in config.schedule]
 183     learning_rate_sequence = [config.initial_learning_rate]
 184     learning_rate_sequence += [x.learning_rate for x in config.schedule]
 185     learning_rate = learning_schedules.manual_stepping(
 186         global_step, learning_rate_step_boundaries,
 187         learning_rate_sequence, config.warmup)
 188 
 189   if learning_rate_type == 'cosine_decay_learning_rate':
 190     config = learning_rate_config.cosine_decay_learning_rate
 191     learning_rate = learning_schedules.cosine_decay_with_warmup(
 192         global_step,
 193         config.learning_rate_base,
 194         config.total_steps,
 195         config.warmup_learning_rate,
 196         config.warmup_steps,
 197         config.hold_base_rate_steps)
 198 
 199   if learning_rate is None:
 200     raise ValueError('Learning_rate %s not supported.' % learning_rate_type)
 201 
 202   return learning_rate
