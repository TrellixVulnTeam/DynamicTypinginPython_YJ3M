  35 def build_convolutional_box_predictor(is_training,
  36                                       num_classes,
  37                                       conv_hyperparams_fn,
  38                                       min_depth,
  39                                       max_depth,
  40                                       num_layers_before_predictor,
  41                                       use_dropout,
  42                                       dropout_keep_prob,
  43                                       kernel_size,
  44                                       box_code_size,
  45                                       apply_sigmoid_to_scores=False,
  46                                       add_background_class=True,
  47                                       class_prediction_bias_init=0.0,
  48                                       use_depthwise=False,
  49                                       box_encodings_clip_range=None):
  50   """Builds the ConvolutionalBoxPredictor from the arguments.
  51 
  52   Args:
  53     is_training: Indicates whether the BoxPredictor is in training mode.
  54     num_classes: number of classes.  Note that num_classes *does not*
  55       include the background category, so if groundtruth labels take values
  56       in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
  57       assigned classification targets can range from {0,... K}).
  58     conv_hyperparams_fn: A function to generate tf-slim arg_scope with
  59       hyperparameters for convolution ops.
  60     min_depth: Minimum feature depth prior to predicting box encodings
  61       and class predictions.
  62     max_depth: Maximum feature depth prior to predicting box encodings
  63       and class predictions. If max_depth is set to 0, no additional
  64       feature map will be inserted before location and class predictions.
  65     num_layers_before_predictor: Number of the additional conv layers before
  66       the predictor.
  67     use_dropout: Option to use dropout or not.  Note that a single dropout
  68       op is applied here prior to both box and class predictions, which stands
  69       in contrast to the ConvolutionalBoxPredictor below.
  70     dropout_keep_prob: Keep probability for dropout.
  71       This is only used if use_dropout is True.
  72     kernel_size: Size of final convolution kernel.  If the
  73       spatial resolution of the feature map is smaller than the kernel size,
  74       then the kernel size is automatically set to be
  75       min(feature_width, feature_height).
  76     box_code_size: Size of encoding for each box.
  77     apply_sigmoid_to_scores: If True, apply the sigmoid on the output
  78       class_predictions.
  79     add_background_class: Whether to add an implicit background class.
  80     class_prediction_bias_init: Constant value to initialize bias of the last
  81       conv2d layer before class prediction.
  82     use_depthwise: Whether to use depthwise convolutions for prediction
  83       steps. Default is False.
  84     box_encodings_clip_range: Min and max values for clipping the box_encodings.
  85 
  86   Returns:
  87     A ConvolutionalBoxPredictor class.
  88   """
  89   box_prediction_head = box_head.ConvolutionalBoxHead(
  90       is_training=is_training,
  91       box_code_size=box_code_size,
  92       kernel_size=kernel_size,
  93       use_depthwise=use_depthwise,
  94       box_encodings_clip_range=box_encodings_clip_range)
  95   class_prediction_head = class_head.ConvolutionalClassHead(
  96       is_training=is_training,
  97       num_class_slots=num_classes + 1 if add_background_class else num_classes,
  98       use_dropout=use_dropout,
  99       dropout_keep_prob=dropout_keep_prob,
 100       kernel_size=kernel_size,
 101       apply_sigmoid_to_scores=apply_sigmoid_to_scores,
 102       class_prediction_bias_init=class_prediction_bias_init,
 103       use_depthwise=use_depthwise)
 104   other_heads = {}
 105   return convolutional_box_predictor.ConvolutionalBoxPredictor(
 106       is_training=is_training,
 107       num_classes=num_classes,
 108       box_prediction_head=box_prediction_head,
 109       class_prediction_head=class_prediction_head,
 110       other_heads=other_heads,
 111       conv_hyperparams_fn=conv_hyperparams_fn,
 112       num_layers_before_predictor=num_layers_before_predictor,
 113       min_depth=min_depth,
 114       max_depth=max_depth)

 117 def build_convolutional_keras_box_predictor(is_training,
 118                                             num_classes,
 119                                             conv_hyperparams,
 120                                             freeze_batchnorm,
 121                                             inplace_batchnorm_update,
 122                                             num_predictions_per_location_list,
 123                                             min_depth,
 124                                             max_depth,
 125                                             num_layers_before_predictor,
 126                                             use_dropout,
 127                                             dropout_keep_prob,
 128                                             kernel_size,
 129                                             box_code_size,
 130                                             add_background_class=True,
 131                                             class_prediction_bias_init=0.0,
 132                                             use_depthwise=False,
 133                                             box_encodings_clip_range=None,
 134                                             name='BoxPredictor'):
 135   """Builds the Keras ConvolutionalBoxPredictor from the arguments.
 136 
 137   Args:
 138     is_training: Indicates whether the BoxPredictor is in training mode.
 139     num_classes: number of classes.  Note that num_classes *does not*
 140       include the background category, so if groundtruth labels take values
 141       in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 142       assigned classification targets can range from {0,... K}).
 143     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 144       containing hyperparameters for convolution ops.
 145     freeze_batchnorm: Whether to freeze batch norm parameters during
 146       training or not. When training with a small batch size (e.g. 1), it is
 147       desirable to freeze batch norm update and use pretrained batch norm
 148       params.
 149     inplace_batchnorm_update: Whether to update batch norm moving average
 150       values inplace. When this is false train op must add a control
 151       dependency on tf.graphkeys.UPDATE_OPS collection in order to update
 152       batch norm statistics.
 153     num_predictions_per_location_list: A list of integers representing the
 154       number of box predictions to be made per spatial location for each
 155       feature map.
 156     min_depth: Minimum feature depth prior to predicting box encodings
 157       and class predictions.
 158     max_depth: Maximum feature depth prior to predicting box encodings
 159       and class predictions. If max_depth is set to 0, no additional
 160       feature map will be inserted before location and class predictions.
 161     num_layers_before_predictor: Number of the additional conv layers before
 162       the predictor.
 163     use_dropout: Option to use dropout or not.  Note that a single dropout
 164       op is applied here prior to both box and class predictions, which stands
 165       in contrast to the ConvolutionalBoxPredictor below.
 166     dropout_keep_prob: Keep probability for dropout.
 167       This is only used if use_dropout is True.
 168     kernel_size: Size of final convolution kernel.  If the
 169       spatial resolution of the feature map is smaller than the kernel size,
 170       then the kernel size is automatically set to be
 171       min(feature_width, feature_height).
 172     box_code_size: Size of encoding for each box.
 173     add_background_class: Whether to add an implicit background class.
 174     class_prediction_bias_init: constant value to initialize bias of the last
 175       conv2d layer before class prediction.
 176     use_depthwise: Whether to use depthwise convolutions for prediction
 177       steps. Default is False.
 178     box_encodings_clip_range: Min and max values for clipping the box_encodings.
 179     name: A string name scope to assign to the box predictor. If `None`, Keras
 180       will auto-generate one from the class name.
 181 
 182   Returns:
 183     A Keras ConvolutionalBoxPredictor class.
 184   """
 185   box_prediction_heads = []
 186   class_prediction_heads = []
 187   other_heads = {}
 188 
 189   for stack_index, num_predictions_per_location in enumerate(
 190       num_predictions_per_location_list):
 191     box_prediction_heads.append(
 192         keras_box_head.ConvolutionalBoxHead(
 193             is_training=is_training,
 194             box_code_size=box_code_size,
 195             kernel_size=kernel_size,
 196             conv_hyperparams=conv_hyperparams,
 197             freeze_batchnorm=freeze_batchnorm,
 198             num_predictions_per_location=num_predictions_per_location,
 199             use_depthwise=use_depthwise,
 200             box_encodings_clip_range=box_encodings_clip_range,
 201             name='ConvolutionalBoxHead_%d' % stack_index))
 202     class_prediction_heads.append(
 203         keras_class_head.ConvolutionalClassHead(
 204             is_training=is_training,
 205             num_class_slots=(
 206                 num_classes + 1 if add_background_class else num_classes),
 207             use_dropout=use_dropout,
 208             dropout_keep_prob=dropout_keep_prob,
 209             kernel_size=kernel_size,
 210             conv_hyperparams=conv_hyperparams,
 211             freeze_batchnorm=freeze_batchnorm,
 212             num_predictions_per_location=num_predictions_per_location,
 213             class_prediction_bias_init=class_prediction_bias_init,
 214             use_depthwise=use_depthwise,
 215             name='ConvolutionalClassHead_%d' % stack_index))
 216 
 217   return convolutional_keras_box_predictor.ConvolutionalBoxPredictor(
 218       is_training=is_training,
 219       num_classes=num_classes,
 220       box_prediction_heads=box_prediction_heads,
 221       class_prediction_heads=class_prediction_heads,
 222       other_heads=other_heads,
 223       conv_hyperparams=conv_hyperparams,
 224       num_layers_before_predictor=num_layers_before_predictor,
 225       min_depth=min_depth,
 226       max_depth=max_depth,
 227       freeze_batchnorm=freeze_batchnorm,
 228       inplace_batchnorm_update=inplace_batchnorm_update,
 229       name=name)

 232 def build_weight_shared_convolutional_box_predictor(
 233     is_training,
 234     num_classes,
 235     conv_hyperparams_fn,
 236     depth,
 237     num_layers_before_predictor,
 238     box_code_size,
 239     kernel_size=3,
 240     add_background_class=True,
 241     class_prediction_bias_init=0.0,
 242     use_dropout=False,
 243     dropout_keep_prob=0.8,
 244     share_prediction_tower=False,
 245     apply_batch_norm=True,
 246     use_depthwise=False,
 247     score_converter_fn=tf.identity,
 248     box_encodings_clip_range=None,
 249     keyword_args=None):
 250   """Builds and returns a WeightSharedConvolutionalBoxPredictor class.
 251 
 252   Args:
 253     is_training: Indicates whether the BoxPredictor is in training mode.
 254     num_classes: number of classes.  Note that num_classes *does not*
 255       include the background category, so if groundtruth labels take values
 256       in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 257       assigned classification targets can range from {0,... K}).
 258     conv_hyperparams_fn: A function to generate tf-slim arg_scope with
 259       hyperparameters for convolution ops.
 260     depth: depth of conv layers.
 261     num_layers_before_predictor: Number of the additional conv layers before
 262       the predictor.
 263     box_code_size: Size of encoding for each box.
 264     kernel_size: Size of final convolution kernel.
 265     add_background_class: Whether to add an implicit background class.
 266     class_prediction_bias_init: constant value to initialize bias of the last
 267       conv2d layer before class prediction.
 268     use_dropout: Whether to apply dropout to class prediction head.
 269     dropout_keep_prob: Probability of keeping activiations.
 270     share_prediction_tower: Whether to share the multi-layer tower between box
 271       prediction and class prediction heads.
 272     apply_batch_norm: Whether to apply batch normalization to conv layers in
 273       this predictor.
 274     use_depthwise: Whether to use depthwise separable conv2d instead of conv2d.
 275     score_converter_fn: Callable score converter to perform elementwise op on
 276       class scores.
 277     box_encodings_clip_range: Min and max values for clipping the box_encodings.
 278     keyword_args: A dictionary with additional args.
 279 
 280   Returns:
 281     A WeightSharedConvolutionalBoxPredictor class.
 282   """
 283   box_prediction_head = box_head.WeightSharedConvolutionalBoxHead(
 284       box_code_size=box_code_size,
 285       kernel_size=kernel_size,
 286       use_depthwise=use_depthwise,
 287       box_encodings_clip_range=box_encodings_clip_range)
 288   class_prediction_head = (
 289       class_head.WeightSharedConvolutionalClassHead(
 290           num_class_slots=(
 291               num_classes + 1 if add_background_class else num_classes),
 292           kernel_size=kernel_size,
 293           class_prediction_bias_init=class_prediction_bias_init,
 294           use_dropout=use_dropout,
 295           dropout_keep_prob=dropout_keep_prob,
 296           use_depthwise=use_depthwise,
 297           score_converter_fn=score_converter_fn))
 298   other_heads = {}
 299   return convolutional_box_predictor.WeightSharedConvolutionalBoxPredictor(
 300       is_training=is_training,
 301       num_classes=num_classes,
 302       box_prediction_head=box_prediction_head,
 303       class_prediction_head=class_prediction_head,
 304       other_heads=other_heads,
 305       conv_hyperparams_fn=conv_hyperparams_fn,
 306       depth=depth,
 307       num_layers_before_predictor=num_layers_before_predictor,
 308       kernel_size=kernel_size,
 309       apply_batch_norm=apply_batch_norm,
 310       share_prediction_tower=share_prediction_tower,
 311       use_depthwise=use_depthwise)

 314 def build_weight_shared_convolutional_keras_box_predictor(
 315     is_training,
 316     num_classes,
 317     conv_hyperparams,
 318     freeze_batchnorm,
 319     inplace_batchnorm_update,
 320     num_predictions_per_location_list,
 321     depth,
 322     num_layers_before_predictor,
 323     box_code_size,
 324     kernel_size=3,
 325     add_background_class=True,
 326     class_prediction_bias_init=0.0,
 327     use_dropout=False,
 328     dropout_keep_prob=0.8,
 329     share_prediction_tower=False,
 330     apply_batch_norm=True,
 331     use_depthwise=False,
 332     score_converter_fn=tf.identity,
 333     box_encodings_clip_range=None,
 334     name='WeightSharedConvolutionalBoxPredictor',
 335     keyword_args=None):
 336   """Builds the Keras WeightSharedConvolutionalBoxPredictor from the arguments.
 337 
 338   Args:
 339     is_training: Indicates whether the BoxPredictor is in training mode.
 340     num_classes: number of classes.  Note that num_classes *does not*
 341       include the background category, so if groundtruth labels take values
 342       in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 343       assigned classification targets can range from {0,... K}).
 344     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 345       containing hyperparameters for convolution ops.
 346     freeze_batchnorm: Whether to freeze batch norm parameters during
 347       training or not. When training with a small batch size (e.g. 1), it is
 348       desirable to freeze batch norm update and use pretrained batch norm
 349       params.
 350     inplace_batchnorm_update: Whether to update batch norm moving average
 351       values inplace. When this is false train op must add a control
 352       dependency on tf.graphkeys.UPDATE_OPS collection in order to update
 353       batch norm statistics.
 354     num_predictions_per_location_list: A list of integers representing the
 355       number of box predictions to be made per spatial location for each
 356       feature map.
 357     depth: depth of conv layers.
 358     num_layers_before_predictor: Number of the additional conv layers before
 359       the predictor.
 360     box_code_size: Size of encoding for each box.
 361     kernel_size: Size of final convolution kernel.
 362     add_background_class: Whether to add an implicit background class.
 363     class_prediction_bias_init: constant value to initialize bias of the last
 364       conv2d layer before class prediction.
 365     use_dropout: Whether to apply dropout to class prediction head.
 366         dropout_keep_prob: Probability of keeping activiations.
 367     share_prediction_tower: Whether to share the multi-layer tower between box
 368       prediction and class prediction heads.
 369     apply_batch_norm: Whether to apply batch normalization to conv layers in
 370       this predictor.
 371     use_depthwise: Whether to use depthwise separable conv2d instead of conv2d.
 372     score_converter_fn: Callable score converter to perform elementwise op on
 373       class scores.
 374     box_encodings_clip_range: Min and max values for clipping the box_encodings.
 375     name: A string name scope to assign to the box predictor. If `None`, Keras
 376       will auto-generate one from the class name.
 377     keyword_args: A dictionary with additional args.
 378 
 379   Returns:
 380     A Keras WeightSharedConvolutionalBoxPredictor class.
 381   """
 382   if len(set(num_predictions_per_location_list)) > 1:
 383     raise ValueError('num predictions per location must be same for all'
 384                      'feature maps, found: {}'.format(
 385                          num_predictions_per_location_list))
 386   num_predictions_per_location = num_predictions_per_location_list[0]
 387 
 388   box_prediction_head = keras_box_head.WeightSharedConvolutionalBoxHead(
 389       box_code_size=box_code_size,
 390       kernel_size=kernel_size,
 391       conv_hyperparams=conv_hyperparams,
 392       num_predictions_per_location=num_predictions_per_location,

 394       box_encodings_clip_range=box_encodings_clip_range,
 395       name='WeightSharedConvolutionalBoxHead')

 397       num_class_slots=(
 398           num_classes + 1 if add_background_class else num_classes),
 399       use_dropout=use_dropout,
 400       dropout_keep_prob=dropout_keep_prob,
 401       kernel_size=kernel_size,
 402       conv_hyperparams=conv_hyperparams,
 403       num_predictions_per_location=num_predictions_per_location,
 404       class_prediction_bias_init=class_prediction_bias_init,
 405       use_depthwise=use_depthwise,
 406       score_converter_fn=score_converter_fn,
 407       name='WeightSharedConvolutionalClassHead')
 408   other_heads = {}
 409 
 410   return (
 411       convolutional_keras_box_predictor.WeightSharedConvolutionalBoxPredictor(
 412           is_training=is_training,
 413           num_classes=num_classes,
 414           box_prediction_head=box_prediction_head,
 415           class_prediction_head=class_prediction_head,
 416           other_heads=other_heads,
 417           conv_hyperparams=conv_hyperparams,
 418           depth=depth,
 419           num_layers_before_predictor=num_layers_before_predictor,
 420           freeze_batchnorm=freeze_batchnorm,
 421           inplace_batchnorm_update=inplace_batchnorm_update,
 422           kernel_size=kernel_size,
 423           apply_batch_norm=apply_batch_norm,
 424           share_prediction_tower=share_prediction_tower,
 425           use_depthwise=use_depthwise,
 426           name=name))

 431 def build_mask_rcnn_keras_box_predictor(is_training,
 432                                         num_classes,
 433                                         fc_hyperparams,
 434                                         freeze_batchnorm,
 435                                         use_dropout,
 436                                         dropout_keep_prob,
 437                                         box_code_size,
 438                                         add_background_class=True,
 439                                         share_box_across_classes=False,
 440                                         predict_instance_masks=False,
 441                                         conv_hyperparams=None,
 442                                         mask_height=14,
 443                                         mask_width=14,
 444                                         mask_prediction_num_conv_layers=2,
 445                                         mask_prediction_conv_depth=256,
 446                                         masks_are_class_agnostic=False,
 447                                         convolve_then_upsample_masks=False):
 448   """Builds and returns a MaskRCNNKerasBoxPredictor class.
 449 
 450   Args:
 451     is_training: Indicates whether the BoxPredictor is in training mode.
 452     num_classes: number of classes.  Note that num_classes *does not*
 453       include the background category, so if groundtruth labels take values
 454       in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 455       assigned classification targets can range from {0,... K}).
 456     fc_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 457       containing hyperparameters for fully connected dense ops.
 458     freeze_batchnorm: Whether to freeze batch norm parameters during
 459       training or not. When training with a small batch size (e.g. 1), it is
 460       desirable to freeze batch norm update and use pretrained batch norm
 461       params.
 462     use_dropout: Option to use dropout or not.  Note that a single dropout
 463       op is applied here prior to both box and class predictions, which stands
 464       in contrast to the ConvolutionalBoxPredictor below.
 465     dropout_keep_prob: Keep probability for dropout.
 466       This is only used if use_dropout is True.
 467     box_code_size: Size of encoding for each box.
 468     add_background_class: Whether to add an implicit background class.
 469     share_box_across_classes: Whether to share boxes across classes rather
 470       than use a different box for each class.
 471     predict_instance_masks: If True, will add a third stage mask prediction
 472       to the returned class.
 473     conv_hyperparams: A `hyperparams_builder.KerasLayerHyperparams` object
 474       containing hyperparameters for convolution ops.
 475     mask_height: Desired output mask height. The default value is 14.
 476     mask_width: Desired output mask width. The default value is 14.
 477     mask_prediction_num_conv_layers: Number of convolution layers applied to
 478       the image_features in mask prediction branch.
 479     mask_prediction_conv_depth: The depth for the first conv2d_transpose op
 480       applied to the image_features in the mask prediction branch. If set
 481       to 0, the depth of the convolution layers will be automatically chosen
 482       based on the number of object classes and the number of channels in the
 483       image features.
 484     masks_are_class_agnostic: Boolean determining if the mask-head is
 485       class-agnostic or not.
 486     convolve_then_upsample_masks: Whether to apply convolutions on mask
 487       features before upsampling using nearest neighbor resizing. Otherwise,
 488       mask features are resized to [`mask_height`, `mask_width`] using
 489       bilinear resizing before applying convolutions.
 490 
 491   Returns:
 492     A MaskRCNNKerasBoxPredictor class.
 493   """
 494   box_prediction_head = keras_box_head.MaskRCNNBoxHead(
 495       is_training=is_training,
 496       num_classes=num_classes,
 497       fc_hyperparams=fc_hyperparams,
 498       freeze_batchnorm=freeze_batchnorm,
 499       use_dropout=use_dropout,
 500       dropout_keep_prob=dropout_keep_prob,
 501       box_code_size=box_code_size,
 502       share_box_across_classes=share_box_across_classes)
 503   class_prediction_head = keras_class_head.MaskRCNNClassHead(
 504       is_training=is_training,
 505       num_class_slots=num_classes + 1 if add_background_class else num_classes,
 506       fc_hyperparams=fc_hyperparams,
 507       freeze_batchnorm=freeze_batchnorm,
 508       use_dropout=use_dropout,
 509       dropout_keep_prob=dropout_keep_prob)
 510   third_stage_heads = {}
 511   if predict_instance_masks:
 512     third_stage_heads[
 513         mask_rcnn_box_predictor.
 514         MASK_PREDICTIONS] = keras_mask_head.MaskRCNNMaskHead(
 515             is_training=is_training,
 516             num_classes=num_classes,
 517             conv_hyperparams=conv_hyperparams,
 518             freeze_batchnorm=freeze_batchnorm,
 519             mask_height=mask_height,
 520             mask_width=mask_width,
 521             mask_prediction_num_conv_layers=mask_prediction_num_conv_layers,
 522             mask_prediction_conv_depth=mask_prediction_conv_depth,
 523             masks_are_class_agnostic=masks_are_class_agnostic,
 524             convolve_then_upsample=convolve_then_upsample_masks)
 525   return mask_rcnn_keras_box_predictor.MaskRCNNKerasBoxPredictor(
 526       is_training=is_training,
 527       num_classes=num_classes,
 528       freeze_batchnorm=freeze_batchnorm,
 529       box_prediction_head=box_prediction_head,
 530       class_prediction_head=class_prediction_head,
 531       third_stage_heads=third_stage_heads)

 534 def build_mask_rcnn_box_predictor(is_training,
 535                                   num_classes,
 536                                   fc_hyperparams_fn,
 537                                   use_dropout,
 538                                   dropout_keep_prob,
 539                                   box_code_size,
 540                                   add_background_class=True,
 541                                   share_box_across_classes=False,
 542                                   predict_instance_masks=False,
 543                                   conv_hyperparams_fn=None,
 544                                   mask_height=14,
 545                                   mask_width=14,
 546                                   mask_prediction_num_conv_layers=2,
 547                                   mask_prediction_conv_depth=256,
 548                                   masks_are_class_agnostic=False,
 549                                   convolve_then_upsample_masks=False):
 550   """Builds and returns a MaskRCNNBoxPredictor class.
 551 
 552   Args:
 553     is_training: Indicates whether the BoxPredictor is in training mode.
 554     num_classes: number of classes.  Note that num_classes *does not*
 555       include the background category, so if groundtruth labels take values
 556       in {0, 1, .., K-1}, num_classes=K (and not K+1, even though the
 557       assigned classification targets can range from {0,... K}).
 558     fc_hyperparams_fn: A function to generate tf-slim arg_scope with
 559       hyperparameters for fully connected ops.
 560     use_dropout: Option to use dropout or not.  Note that a single dropout
 561       op is applied here prior to both box and class predictions, which stands
 562       in contrast to the ConvolutionalBoxPredictor below.
 563     dropout_keep_prob: Keep probability for dropout.
 564       This is only used if use_dropout is True.
 565     box_code_size: Size of encoding for each box.
 566     add_background_class: Whether to add an implicit background class.
 567     share_box_across_classes: Whether to share boxes across classes rather
 568       than use a different box for each class.
 569     predict_instance_masks: If True, will add a third stage mask prediction
 570       to the returned class.
 571     conv_hyperparams_fn: A function to generate tf-slim arg_scope with
 572       hyperparameters for convolution ops.
 573     mask_height: Desired output mask height. The default value is 14.
 574     mask_width: Desired output mask width. The default value is 14.
 575     mask_prediction_num_conv_layers: Number of convolution layers applied to
 576       the image_features in mask prediction branch.
 577     mask_prediction_conv_depth: The depth for the first conv2d_transpose op
 578       applied to the image_features in the mask prediction branch. If set
 579       to 0, the depth of the convolution layers will be automatically chosen
 580       based on the number of object classes and the number of channels in the
 581       image features.
 582     masks_are_class_agnostic: Boolean determining if the mask-head is
 583       class-agnostic or not.
 584     convolve_then_upsample_masks: Whether to apply convolutions on mask
 585       features before upsampling using nearest neighbor resizing. Otherwise,
 586       mask features are resized to [`mask_height`, `mask_width`] using
 587       bilinear resizing before applying convolutions.
 588 
 589   Returns:
 590     A MaskRCNNBoxPredictor class.
 591   """
 592   box_prediction_head = box_head.MaskRCNNBoxHead(
 593       is_training=is_training,
 594       num_classes=num_classes,
 595       fc_hyperparams_fn=fc_hyperparams_fn,
 596       use_dropout=use_dropout,
 597       dropout_keep_prob=dropout_keep_prob,
 598       box_code_size=box_code_size,
 599       share_box_across_classes=share_box_across_classes)
 600   class_prediction_head = class_head.MaskRCNNClassHead(
 601       is_training=is_training,
 602       num_class_slots=num_classes + 1 if add_background_class else num_classes,
 603       fc_hyperparams_fn=fc_hyperparams_fn,
 604       use_dropout=use_dropout,
 605       dropout_keep_prob=dropout_keep_prob)
 606   third_stage_heads = {}
 607   if predict_instance_masks:
 608     third_stage_heads[
 609         mask_rcnn_box_predictor.
 610         MASK_PREDICTIONS] = mask_head.MaskRCNNMaskHead(
 611             num_classes=num_classes,
 612             conv_hyperparams_fn=conv_hyperparams_fn,
 613             mask_height=mask_height,
 614             mask_width=mask_width,
 615             mask_prediction_num_conv_layers=mask_prediction_num_conv_layers,
 616             mask_prediction_conv_depth=mask_prediction_conv_depth,
 617             masks_are_class_agnostic=masks_are_class_agnostic,
 618             convolve_then_upsample=convolve_then_upsample_masks)
 619   return mask_rcnn_box_predictor.MaskRCNNBoxPredictor(
 620       is_training=is_training,
 621       num_classes=num_classes,
 622       box_prediction_head=box_prediction_head,
 623       class_prediction_head=class_prediction_head,
 624       third_stage_heads=third_stage_heads)

 627 def build_score_converter(score_converter_config, is_training):
 628   """Builds score converter based on the config.
 629 
 630   Builds one of [tf.identity, tf.sigmoid] score converters based on the config
 631   and whether the BoxPredictor is for training or inference.
 632 
 633   Args:
 634     score_converter_config:
 635       box_predictor_pb2.WeightSharedConvolutionalBoxPredictor.score_converter.
 636     is_training: Indicates whether the BoxPredictor is in training mode.
 637 
 638   Returns:
 639     Callable score converter op.
 640 
 641   Raises:
 642     ValueError: On unknown score converter.
 643   """
 644   if score_converter_config == (
 645       box_predictor_pb2.WeightSharedConvolutionalBoxPredictor.IDENTITY):
 646     return tf.identity
 647   if score_converter_config == (
 648       box_predictor_pb2.WeightSharedConvolutionalBoxPredictor.SIGMOID):
 649     return tf.identity if is_training else tf.sigmoid
 650   raise ValueError('Unknown score converter.')

 657 def build(argscope_fn, box_predictor_config, is_training, num_classes,
 658           add_background_class=True):
 659   """Builds box predictor based on the configuration.
 660 
 661   Builds box predictor based on the configuration. See box_predictor.proto for
 662   configurable options. Also, see box_predictor.py for more details.
 663 
 664   Args:
 665     argscope_fn: A function that takes the following inputs:
 666         * hyperparams_pb2.Hyperparams proto
 667         * a boolean indicating if the model is in training mode.
 668       and returns a tf slim argscope for Conv and FC hyperparameters.
 669     box_predictor_config: box_predictor_pb2.BoxPredictor proto containing
 670       configuration.
 671     is_training: Whether the models is in training mode.
 672     num_classes: Number of classes to predict.
 673     add_background_class: Whether to add an implicit background class.
 674 
 675   Returns:
 676     box_predictor: box_predictor.BoxPredictor object.
 677 
 678   Raises:
 679     ValueError: On unknown box predictor.
 680   """
 681   if not isinstance(box_predictor_config, box_predictor_pb2.BoxPredictor):
 682     raise ValueError('box_predictor_config not of type '
 683                      'box_predictor_pb2.BoxPredictor.')
 684 
 685   box_predictor_oneof = box_predictor_config.WhichOneof('box_predictor_oneof')
 686 
 687   if  box_predictor_oneof == 'convolutional_box_predictor':
 688     config_box_predictor = box_predictor_config.convolutional_box_predictor
 689     conv_hyperparams_fn = argscope_fn(config_box_predictor.conv_hyperparams,
 690                                       is_training)
 691     # Optionally apply clipping to box encodings, when box_encodings_clip_range
 692     # is set.
 693     box_encodings_clip_range = None
 694     if config_box_predictor.HasField('box_encodings_clip_range'):
 695       box_encodings_clip_range = BoxEncodingsClipRange(
 696           min=config_box_predictor.box_encodings_clip_range.min,
 697           max=config_box_predictor.box_encodings_clip_range.max)
 698     return build_convolutional_box_predictor(
 699         is_training=is_training,
 700         num_classes=num_classes,
 701         add_background_class=add_background_class,
 702         conv_hyperparams_fn=conv_hyperparams_fn,
 703         use_dropout=config_box_predictor.use_dropout,
 704         dropout_keep_prob=config_box_predictor.dropout_keep_probability,
 705         box_code_size=config_box_predictor.box_code_size,
 706         kernel_size=config_box_predictor.kernel_size,
 707         num_layers_before_predictor=(
 708             config_box_predictor.num_layers_before_predictor),
 709         min_depth=config_box_predictor.min_depth,
 710         max_depth=config_box_predictor.max_depth,
 711         apply_sigmoid_to_scores=config_box_predictor.apply_sigmoid_to_scores,
 712         class_prediction_bias_init=(
 713             config_box_predictor.class_prediction_bias_init),
 714         use_depthwise=config_box_predictor.use_depthwise,
 715         box_encodings_clip_range=box_encodings_clip_range)
 716 
 717   if  box_predictor_oneof == 'weight_shared_convolutional_box_predictor':
 718     config_box_predictor = (
 719         box_predictor_config.weight_shared_convolutional_box_predictor)
 720     conv_hyperparams_fn = argscope_fn(config_box_predictor.conv_hyperparams,
 721                                       is_training)
 722     apply_batch_norm = config_box_predictor.conv_hyperparams.HasField(
 723         'batch_norm')
 724     # During training phase, logits are used to compute the loss. Only apply
 725     # sigmoid at inference to make the inference graph TPU friendly.
 726     score_converter_fn = build_score_converter(
 727         config_box_predictor.score_converter, is_training)
 728     # Optionally apply clipping to box encodings, when box_encodings_clip_range
 729     # is set.
 730     box_encodings_clip_range = None
 731     if config_box_predictor.HasField('box_encodings_clip_range'):
 732       box_encodings_clip_range = BoxEncodingsClipRange(
 733           min=config_box_predictor.box_encodings_clip_range.min,
 734           max=config_box_predictor.box_encodings_clip_range.max)
 735     keyword_args = None
 736 
 737     return build_weight_shared_convolutional_box_predictor(
 738         is_training=is_training,
 739         num_classes=num_classes,
 740         add_background_class=add_background_class,
 741         conv_hyperparams_fn=conv_hyperparams_fn,
 742         depth=config_box_predictor.depth,
 743         num_layers_before_predictor=(
 744             config_box_predictor.num_layers_before_predictor),
 745         box_code_size=config_box_predictor.box_code_size,
 746         kernel_size=config_box_predictor.kernel_size,
 747         class_prediction_bias_init=(
 748             config_box_predictor.class_prediction_bias_init),
 749         use_dropout=config_box_predictor.use_dropout,
 750         dropout_keep_prob=config_box_predictor.dropout_keep_probability,
 751         share_prediction_tower=config_box_predictor.share_prediction_tower,
 752         apply_batch_norm=apply_batch_norm,
 753         use_depthwise=config_box_predictor.use_depthwise,
 754         score_converter_fn=score_converter_fn,
 755         box_encodings_clip_range=box_encodings_clip_range,
 756         keyword_args=keyword_args)
 757 
 758 
 759   if box_predictor_oneof == 'mask_rcnn_box_predictor':
 760     config_box_predictor = box_predictor_config.mask_rcnn_box_predictor
 761     fc_hyperparams_fn = argscope_fn(config_box_predictor.fc_hyperparams,
 762                                     is_training)
 763     conv_hyperparams_fn = None
 764     if config_box_predictor.HasField('conv_hyperparams'):
 765       conv_hyperparams_fn = argscope_fn(
 766           config_box_predictor.conv_hyperparams, is_training)
 767     return build_mask_rcnn_box_predictor(
 768         is_training=is_training,
 769         num_classes=num_classes,
 770         add_background_class=add_background_class,
 771         fc_hyperparams_fn=fc_hyperparams_fn,
 772         use_dropout=config_box_predictor.use_dropout,
 773         dropout_keep_prob=config_box_predictor.dropout_keep_probability,
 774         box_code_size=config_box_predictor.box_code_size,
 775         share_box_across_classes=(
 776             config_box_predictor.share_box_across_classes),
 777         predict_instance_masks=config_box_predictor.predict_instance_masks,
 778         conv_hyperparams_fn=conv_hyperparams_fn,
 779         mask_height=config_box_predictor.mask_height,
 780         mask_width=config_box_predictor.mask_width,
 781         mask_prediction_num_conv_layers=(
 782             config_box_predictor.mask_prediction_num_conv_layers),
 783         mask_prediction_conv_depth=(
 784             config_box_predictor.mask_prediction_conv_depth),
 785         masks_are_class_agnostic=(
 786             config_box_predictor.masks_are_class_agnostic),
 787         convolve_then_upsample_masks=(
 788             config_box_predictor.convolve_then_upsample_masks))
 789 
 790   if box_predictor_oneof == 'rfcn_box_predictor':
 791     config_box_predictor = box_predictor_config.rfcn_box_predictor
 792     conv_hyperparams_fn = argscope_fn(config_box_predictor.conv_hyperparams,
 793                                       is_training)
 794     box_predictor_object = rfcn_box_predictor.RfcnBoxPredictor(
 795         is_training=is_training,
 796         num_classes=num_classes,
 797         conv_hyperparams_fn=conv_hyperparams_fn,
 798         crop_size=[config_box_predictor.crop_height,
 799                    config_box_predictor.crop_width],
 800         num_spatial_bins=[config_box_predictor.num_spatial_bins_height,
 801                           config_box_predictor.num_spatial_bins_width],
 802         depth=config_box_predictor.depth,
 803         box_code_size=config_box_predictor.box_code_size)
 804     return box_predictor_object
 805   raise ValueError('Unknown box predictor: {}'.format(box_predictor_oneof))

 808 def build_keras(hyperparams_fn, freeze_batchnorm, inplace_batchnorm_update,
 809                 num_predictions_per_location_list, box_predictor_config,
 810                 is_training, num_classes, add_background_class=True):
 811   """Builds a Keras-based box predictor based on the configuration.
 812 
 813   Builds Keras-based box predictor based on the configuration.
 814   See box_predictor.proto for configurable options. Also, see box_predictor.py
 815   for more details.
 816 
 817   Args:
 818     hyperparams_fn: A function that takes a hyperparams_pb2.Hyperparams
 819       proto and returns a `hyperparams_builder.KerasLayerHyperparams`
 820       for Conv or FC hyperparameters.
 821     freeze_batchnorm: Whether to freeze batch norm parameters during
 822       training or not. When training with a small batch size (e.g. 1), it is
 823       desirable to freeze batch norm update and use pretrained batch norm
 824       params.
 825     inplace_batchnorm_update: Whether to update batch norm moving average
 826       values inplace. When this is false train op must add a control
 827       dependency on tf.graphkeys.UPDATE_OPS collection in order to update
 828       batch norm statistics.
 829     num_predictions_per_location_list: A list of integers representing the
 830       number of box predictions to be made per spatial location for each
 831       feature map.
 832     box_predictor_config: box_predictor_pb2.BoxPredictor proto containing
 833       configuration.
 834     is_training: Whether the models is in training mode.
 835     num_classes: Number of classes to predict.
 836     add_background_class: Whether to add an implicit background class.
 837 
 838   Returns:
 839     box_predictor: box_predictor.KerasBoxPredictor object.
 840 
 841   Raises:
 842     ValueError: On unknown box predictor, or one with no Keras box predictor.
 843   """
 844   if not isinstance(box_predictor_config, box_predictor_pb2.BoxPredictor):
 845     raise ValueError('box_predictor_config not of type '
 846                      'box_predictor_pb2.BoxPredictor.')

 855     # is set.
 856     box_encodings_clip_range = None
 857     if config_box_predictor.HasField('box_encodings_clip_range'):
 858       box_encodings_clip_range = BoxEncodingsClipRange(
 859           min=config_box_predictor.box_encodings_clip_range.min,
 860           max=config_box_predictor.box_encodings_clip_range.max)
 861 
 862     return build_convolutional_keras_box_predictor(
 863         is_training=is_training,
 864         num_classes=num_classes,
 865         add_background_class=add_background_class,
 866         conv_hyperparams=conv_hyperparams,
 867         freeze_batchnorm=freeze_batchnorm,
 868         inplace_batchnorm_update=inplace_batchnorm_update,
 869         num_predictions_per_location_list=num_predictions_per_location_list,
 870         use_dropout=config_box_predictor.use_dropout,
 871         dropout_keep_prob=config_box_predictor.dropout_keep_probability,
 872         box_code_size=config_box_predictor.box_code_size,
 873         kernel_size=config_box_predictor.kernel_size,
 874         num_layers_before_predictor=(
 875             config_box_predictor.num_layers_before_predictor),
 876         min_depth=config_box_predictor.min_depth,
 877         max_depth=config_box_predictor.max_depth,
 878         class_prediction_bias_init=(
 879             config_box_predictor.class_prediction_bias_init),
 880         use_depthwise=config_box_predictor.use_depthwise,
 881         box_encodings_clip_range=box_encodings_clip_range)
 882 
 883   if box_predictor_oneof == 'weight_shared_convolutional_box_predictor':
 884     config_box_predictor = (
 885         box_predictor_config.weight_shared_convolutional_box_predictor)

 896     box_encodings_clip_range = None
 897     if config_box_predictor.HasField('box_encodings_clip_range'):
 898       box_encodings_clip_range = BoxEncodingsClipRange(
 899           min=config_box_predictor.box_encodings_clip_range.min,
 900           max=config_box_predictor.box_encodings_clip_range.max)
 901     keyword_args = None
 902 
 903     return build_weight_shared_convolutional_keras_box_predictor(
 904         is_training=is_training,
 905         num_classes=num_classes,
 906         conv_hyperparams=conv_hyperparams,
 907         freeze_batchnorm=freeze_batchnorm,
 908         inplace_batchnorm_update=inplace_batchnorm_update,
 909         num_predictions_per_location_list=num_predictions_per_location_list,
 910         depth=config_box_predictor.depth,
 911         num_layers_before_predictor=(
 912             config_box_predictor.num_layers_before_predictor),
 913         box_code_size=config_box_predictor.box_code_size,
 914         kernel_size=config_box_predictor.kernel_size,
 915         add_background_class=add_background_class,
 916         class_prediction_bias_init=(
 917             config_box_predictor.class_prediction_bias_init),

 932       conv_hyperparams = hyperparams_fn(
 933           config_box_predictor.conv_hyperparams)
 934     return build_mask_rcnn_keras_box_predictor(
 935         is_training=is_training,
 936         num_classes=num_classes,
 937         add_background_class=add_background_class,
 938         fc_hyperparams=fc_hyperparams,

 942         box_code_size=config_box_predictor.box_code_size,

 946         conv_hyperparams=conv_hyperparams,
 947         mask_height=config_box_predictor.mask_height,
 948         mask_width=config_box_predictor.mask_width,
 949         mask_prediction_num_conv_layers=(
 950             config_box_predictor.mask_prediction_num_conv_layers),
 951         mask_prediction_conv_depth=(
 952             config_box_predictor.mask_prediction_conv_depth),
 953         masks_are_class_agnostic=(

 955         convolve_then_upsample_masks=(

 963         num_classes=num_classes,
 964         conv_hyperparams=conv_hyperparams,
 965         freeze_batchnorm=freeze_batchnorm,
 966         crop_size=[config_box_predictor.crop_height,
 967                    config_box_predictor.crop_width],
 968         num_spatial_bins=[config_box_predictor.num_spatial_bins_height,

 975       'Unknown box predictor for Keras: {}'.format(box_predictor_oneof))
