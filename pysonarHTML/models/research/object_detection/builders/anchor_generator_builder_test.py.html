  40   def assert_almost_list_equal(self, expected_list, actual_list, delta=None):
  41     self.assertEqual(len(expected_list), len(actual_list))
  42     for expected_item, actual_item in zip(expected_list, actual_list):
  43       self.assertAlmostEqual(expected_item, actual_item, delta=delta)

  45   def test_build_grid_anchor_generator_with_defaults(self):
  46     anchor_generator_text_proto = """
  47       grid_anchor_generator {
  48       }
  49      """
  50     anchor_generator_proto = anchor_generator_pb2.AnchorGenerator()
  51     text_format.Merge(anchor_generator_text_proto, anchor_generator_proto)
  52     anchor_generator_object = anchor_generator_builder.build(
  53         anchor_generator_proto)
  54     self.assertIsInstance(anchor_generator_object,
  55                           grid_anchor_generator.GridAnchorGenerator)
  56     self.assertListEqual(anchor_generator_object._scales, [])
  57     self.assertListEqual(anchor_generator_object._aspect_ratios, [])
  58     self.assertAllEqual(anchor_generator_object._anchor_offset, [0, 0])
  59     self.assertAllEqual(anchor_generator_object._anchor_stride, [16, 16])
  60     self.assertAllEqual(anchor_generator_object._base_anchor_size, [256, 256])

  89   def test_build_ssd_anchor_generator_with_defaults(self):
  90     anchor_generator_text_proto = """
  91       ssd_anchor_generator {
  92         aspect_ratios: [1.0]
  93       }
  94     """
  95     anchor_generator_proto = anchor_generator_pb2.AnchorGenerator()
  96     text_format.Merge(anchor_generator_text_proto, anchor_generator_proto)
  97     anchor_generator_object = anchor_generator_builder.build(
  98         anchor_generator_proto)
  99     self.assertIsInstance(anchor_generator_object,
 100                           multiple_grid_anchor_generator.
 101                           MultipleGridAnchorGenerator)
 102     for actual_scales, expected_scales in zip(
 103         list(anchor_generator_object._scales),
 104         [(0.1, 0.2, 0.2),
 105          (0.35, 0.418),
 106          (0.499, 0.570),
 107          (0.649, 0.721),
 108          (0.799, 0.871),
 109          (0.949, 0.974)]):
 110       self.assert_almost_list_equal(expected_scales, actual_scales, delta=1e-2)
 111     for actual_aspect_ratio, expected_aspect_ratio in zip(
 112         list(anchor_generator_object._aspect_ratios),
 113         [(1.0, 2.0, 0.5)] + 5 * [(1.0, 1.0)]):
 114       self.assert_almost_list_equal(expected_aspect_ratio, actual_aspect_ratio)
 115     self.assertAllClose(anchor_generator_object._base_anchor_size, [1.0, 1.0])

 247   def test_build_multiscale_anchor_generator_custom_aspect_ratios(self):
 248     anchor_generator_text_proto = """
 249       multiscale_anchor_generator {
 250         aspect_ratios: [1.0]
 251       }
 252     """
 253     anchor_generator_proto = anchor_generator_pb2.AnchorGenerator()
 254     text_format.Merge(anchor_generator_text_proto, anchor_generator_proto)
 255     anchor_generator_object = anchor_generator_builder.build(
 256         anchor_generator_proto)
 257     self.assertIsInstance(anchor_generator_object,
 258                           multiscale_grid_anchor_generator.
 259                           MultiscaleGridAnchorGenerator)
 260     for level, anchor_grid_info in zip(
 261         range(3, 8), anchor_generator_object._anchor_grid_info):
 262       self.assertEqual(set(anchor_grid_info.keys()), set(['level', 'info']))
 263       self.assertTrue(level, anchor_grid_info['level'])
 264       self.assertEqual(len(anchor_grid_info['info']), 4)
 265       self.assertAllClose(anchor_grid_info['info'][0], [2**0, 2**0.5])
 266       self.assertTrue(anchor_grid_info['info'][1], 1.0)
 267       self.assertAllClose(anchor_grid_info['info'][2],
 268                           [4.0 * 2**level, 4.0 * 2**level])
 269       self.assertAllClose(anchor_grid_info['info'][3], [2**level, 2**level])
 270       self.assertTrue(anchor_generator_object._normalize_coordinates)
