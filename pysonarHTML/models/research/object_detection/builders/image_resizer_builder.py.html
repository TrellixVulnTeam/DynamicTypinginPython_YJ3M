  23 def _tf_resize_method(resize_method):
  24   """Maps image resize method from enumeration type to TensorFlow.
  25 
  26   Args:
  27     resize_method: The resize_method attribute of keep_aspect_ratio_resizer or
  28       fixed_shape_resizer.
  29 
  30   Returns:
  31     method: The corresponding TensorFlow ResizeMethod.
  32 
  33   Raises:
  34     ValueError: if `resize_method` is of unknown type.
  35   """
  36   dict_method = {
  37       image_resizer_pb2.BILINEAR:
  38           tf.image.ResizeMethod.BILINEAR,
  39       image_resizer_pb2.NEAREST_NEIGHBOR:
  40           tf.image.ResizeMethod.NEAREST_NEIGHBOR,
  41       image_resizer_pb2.BICUBIC:
  42           tf.image.ResizeMethod.BICUBIC,
  43       image_resizer_pb2.AREA:
  44           tf.image.ResizeMethod.AREA
  45   }
  46   if resize_method in dict_method:
  47     return dict_method[resize_method]
  48   else:
  49     raise ValueError('Unknown resize_method')

  52 def build(image_resizer_config):
  53   """Builds callable for image resizing operations.
  54 
  55   Args:
  56     image_resizer_config: image_resizer.proto object containing parameters for
  57       an image resizing operation.
  58 
  59   Returns:
  60     image_resizer_fn: Callable for image resizing.  This callable always takes
  61       a rank-3 image tensor (corresponding to a single image) and returns a
  62       rank-3 image tensor, possibly with new spatial dimensions.
  63 
  64   Raises:
  65     ValueError: if `image_resizer_config` is of incorrect type.
  66     ValueError: if `image_resizer_config.image_resizer_oneof` is of expected
  67       type.
  68     ValueError: if min_dimension > max_dimension when keep_aspect_ratio_resizer
  69       is used.
  70   """
  71   if not isinstance(image_resizer_config, image_resizer_pb2.ImageResizer):
  72     raise ValueError('image_resizer_config not of type '
  73                      'image_resizer_pb2.ImageResizer.')
  74 
  75   image_resizer_oneof = image_resizer_config.WhichOneof('image_resizer_oneof')
  76   if image_resizer_oneof == 'keep_aspect_ratio_resizer':
  77     keep_aspect_ratio_config = image_resizer_config.keep_aspect_ratio_resizer
  78     if not (keep_aspect_ratio_config.min_dimension <=
  79             keep_aspect_ratio_config.max_dimension):
  80       raise ValueError('min_dimension > max_dimension')
  81     method = _tf_resize_method(keep_aspect_ratio_config.resize_method)
  82     per_channel_pad_value = (0, 0, 0)
  83     if keep_aspect_ratio_config.per_channel_pad_value:
  84       per_channel_pad_value = tuple(keep_aspect_ratio_config.
  85                                     per_channel_pad_value)
  86     image_resizer_fn = functools.partial(
  87         preprocessor.resize_to_range,
  88         min_dimension=keep_aspect_ratio_config.min_dimension,
  89         max_dimension=keep_aspect_ratio_config.max_dimension,
  90         method=method,
  91         pad_to_max_dimension=keep_aspect_ratio_config.pad_to_max_dimension,
  92         per_channel_pad_value=per_channel_pad_value)
  93     if not keep_aspect_ratio_config.convert_to_grayscale:
  94       return image_resizer_fn
  95   elif image_resizer_oneof == 'fixed_shape_resizer':
  96     fixed_shape_resizer_config = image_resizer_config.fixed_shape_resizer
  97     method = _tf_resize_method(fixed_shape_resizer_config.resize_method)
  98     image_resizer_fn = functools.partial(
  99         preprocessor.resize_image,
 100         new_height=fixed_shape_resizer_config.height,
 101         new_width=fixed_shape_resizer_config.width,
 102         method=method)
 103     if not fixed_shape_resizer_config.convert_to_grayscale:
 104       return image_resizer_fn
 105   elif image_resizer_oneof == 'identity_resizer':
 106     def image_resizer_fn(image, masks=None, **kwargs):
 107       del kwargs
 108       if masks is None:
 109         return [image, tf.shape(image)]
 110       else:
 111         return [image, masks, tf.shape(image)]
 112     return image_resizer_fn
 113   elif image_resizer_oneof == 'conditional_shape_resizer':
 114     conditional_shape_resize_config = (
 115         image_resizer_config.conditional_shape_resizer)
 116     method = _tf_resize_method(conditional_shape_resize_config.resize_method)
 117 
 118     if conditional_shape_resize_config.condition == (
 119         image_resizer_pb2.ConditionalShapeResizer.GREATER):
 120       image_resizer_fn = functools.partial(
 121           preprocessor.resize_to_max_dimension,
 122           max_dimension=conditional_shape_resize_config.size_threshold,
 123           method=method)
 124 
 125     elif conditional_shape_resize_config.condition == (
 126         image_resizer_pb2.ConditionalShapeResizer.SMALLER):
 127       image_resizer_fn = functools.partial(
 128           preprocessor.resize_to_min_dimension,
 129           min_dimension=conditional_shape_resize_config.size_threshold,
 130           method=method)
 131     else:
 132       raise ValueError(
 133           'Invalid image resizer condition option for '
 134           'ConditionalShapeResizer: \'%s\'.'
 135           % conditional_shape_resize_config.condition)
 136     if not conditional_shape_resize_config.convert_to_grayscale:
 137       return image_resizer_fn
 138   elif image_resizer_oneof == 'pad_to_multiple_resizer':
 139     pad_to_multiple_resizer_config = (
 140         image_resizer_config.pad_to_multiple_resizer)
 141 
 142     if pad_to_multiple_resizer_config.multiple < 0:
 143       raise ValueError('`multiple` for pad_to_multiple_resizer should be > 0.')
 144 
 145     else:
 146       image_resizer_fn = functools.partial(
 147           preprocessor.resize_pad_to_multiple,
 148           multiple=pad_to_multiple_resizer_config.multiple)
 149 
 150     if not pad_to_multiple_resizer_config.convert_to_grayscale:
 151       return image_resizer_fn
 152   else:
 153     raise ValueError(
 154         'Invalid image resizer option: \'%s\'.' % image_resizer_oneof)

 156   def grayscale_image_resizer(image, masks=None):
 157     """Convert to grayscale before applying image_resizer_fn.
 158 
 159     Args:
 160       image: A 3D tensor of shape [height, width, 3]
 161       masks: (optional) rank 3 float32 tensor with shape [num_instances, height,
 162         width] containing instance masks.
 163 
 164     Returns:
 165       Note that the position of the resized_image_shape changes based on whether
 166       masks are present.
 167       resized_image: A 3D tensor of shape [new_height, new_width, 1],
 168         where the image has been resized (with bilinear interpolation) so that
 169         min(new_height, new_width) == min_dimension or
 170         max(new_height, new_width) == max_dimension.
 171       resized_masks: If masks is not None, also outputs masks. A 3D tensor of
 172         shape [num_instances, new_height, new_width].
 173       resized_image_shape: A 1D tensor of shape [3] containing shape of the
 174         resized image.
 175     """
 176     # image_resizer_fn returns [resized_image, resized_image_shape] if
 177     # mask==None, otherwise it returns
 178     # [resized_image, resized_mask, resized_image_shape]. In either case, we
 179     # only deal with first and last element of the returned list.
 180     retval = image_resizer_fn(image, masks)
 181     resized_image = retval[0]
 182     resized_image_shape = retval[-1]
 183     retval[0] = preprocessor.rgb_to_gray(resized_image)
 184     retval[-1] = tf.concat([resized_image_shape[:-1], [1]], 0)
 185     return retval
 186 
 187   return functools.partial(grayscale_image_resizer)
