  29   def assert_dictionary_close(self, dict1, dict2):
  30     """Helper to check if two dicts with floatst or integers are close."""
  31     self.assertEqual(sorted(dict1.keys()), sorted(dict2.keys()))
  32     for key in dict1:
  33       value = dict1[key]
  34       if isinstance(value, float):
  35         self.assertAlmostEqual(value, dict2[key])
  36       else:
  37         self.assertEqual(value, dict2[key])

  39   def test_build_normalize_image(self):
  40     preprocessor_text_proto = """
  41     normalize_image {
  42       original_minval: 0.0
  43       original_maxval: 255.0
  44       target_minval: -1.0
  45       target_maxval: 1.0
  46     }
  47     """
  48     preprocessor_proto = preprocessor_pb2.PreprocessingStep()
  49     text_format.Merge(preprocessor_text_proto, preprocessor_proto)
  50     function, args = preprocessor_builder.build(preprocessor_proto)
  51     self.assertEqual(function, preprocessor.normalize_image)
  52     self.assertEqual(args, {
  53         'original_minval': 0.0,
  54         'original_maxval': 255.0,
  55         'target_minval': -1.0,
  56         'target_maxval': 1.0,
  57     })
