  45   def create_model(self, model_config, is_training=True):
  46     """Builds a DetectionModel based on the model config.
  47 
  48     Args:
  49       model_config: A model.proto object containing the config for the desired
  50         DetectionModel.
  51       is_training: True if this model is being built for training purposes.
  52 
  53     Returns:
  54       DetectionModel based on the config.
  55     """
  56     return model_builder.build(model_config, is_training=is_training)

 203   def test_create_ssd_models_from_config(self):

 222       {
 223           'testcase_name': 'mask_rcnn_with_matmul',

 232       {
 233           'testcase_name': 'faster_rcnn_with_matmul',

 236       },
 237       {

 240           'enable_mask_prediction': False
 241       },
 242   )
 243   def test_create_faster_rcnn_models_from_config(self,
 244                                                  use_matmul_crop_and_resize,
 245                                                  enable_mask_prediction):
 246     model_proto = self.create_default_faster_rcnn_model_proto()
 247     faster_rcnn_config = model_proto.faster_rcnn
 248     faster_rcnn_config.use_matmul_crop_and_resize = use_matmul_crop_and_resize
 249     if enable_mask_prediction:
 250       faster_rcnn_config.second_stage_mask_prediction_loss_weight = 3.0
 251       mask_predictor_config = (
 252           faster_rcnn_config.second_stage_box_predictor.mask_rcnn_box_predictor)
 253       mask_predictor_config.predict_instance_masks = True
 254 
 255     for extractor_type, extractor_class in (
 256         self.faster_rcnn_feature_extractors().items()):
 257       faster_rcnn_config.feature_extractor.type = extractor_type
 258       model = model_builder.build(model_proto, is_training=True)
 259       self.assertIsInstance(model, faster_rcnn_meta_arch.FasterRCNNMetaArch)
 260       self.assertIsInstance(model._feature_extractor, extractor_class)
 261       if enable_mask_prediction:
 262         self.assertAlmostEqual(model._second_stage_mask_loss_weight, 3.0)
