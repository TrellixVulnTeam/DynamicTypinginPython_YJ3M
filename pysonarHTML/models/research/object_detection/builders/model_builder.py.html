 190 def _check_feature_extractor_exists(feature_extractor_type):
 191   feature_extractors = set().union(*FEATURE_EXTRACTOR_MAPS)
 192   if feature_extractor_type not in feature_extractors:
 193     raise ValueError('{} is not supported. See `model_builder.py` for features '
 194                      'extractors compatible with different versions of '
 195                      'Tensorflow'.format(feature_extractor_type))

 198 def _build_ssd_feature_extractor(feature_extractor_config,
 199                                  is_training,
 200                                  freeze_batchnorm,
 201                                  reuse_weights=None):
 202   """Builds a ssd_meta_arch.SSDFeatureExtractor based on config.
 203 
 204   Args:
 205     feature_extractor_config: A SSDFeatureExtractor proto config from ssd.proto.
 206     is_training: True if this feature extractor is being built for training.
 207     freeze_batchnorm: Whether to freeze batch norm parameters during
 208       training or not. When training with a small batch size (e.g. 1), it is
 209       desirable to freeze batch norm update and use pretrained batch norm
 210       params.
 211     reuse_weights: if the feature extractor should reuse weights.
 212 
 213   Returns:
 214     ssd_meta_arch.SSDFeatureExtractor based on config.
 215 
 216   Raises:
 217     ValueError: On invalid feature extractor type.
 218   """
 219   feature_type = feature_extractor_config.type
 220   depth_multiplier = feature_extractor_config.depth_multiplier
 221   min_depth = feature_extractor_config.min_depth
 222   pad_to_multiple = feature_extractor_config.pad_to_multiple
 223   use_explicit_padding = feature_extractor_config.use_explicit_padding
 224   use_depthwise = feature_extractor_config.use_depthwise
 225 
 226   is_keras = tf_version.is_tf2()
 227   if is_keras:
 228     conv_hyperparams = hyperparams_builder.KerasLayerHyperparams(
 229         feature_extractor_config.conv_hyperparams)
 230   else:
 231     conv_hyperparams = hyperparams_builder.build(
 232         feature_extractor_config.conv_hyperparams, is_training)
 233   override_base_feature_extractor_hyperparams = (
 234       feature_extractor_config.override_base_feature_extractor_hyperparams)
 235 
 236   if not is_keras and feature_type not in SSD_FEATURE_EXTRACTOR_CLASS_MAP:
 237     raise ValueError('Unknown ssd feature_extractor: {}'.format(feature_type))
 238 
 239   if is_keras:
 240     feature_extractor_class = SSD_KERAS_FEATURE_EXTRACTOR_CLASS_MAP[
 241         feature_type]
 242   else:
 243     feature_extractor_class = SSD_FEATURE_EXTRACTOR_CLASS_MAP[feature_type]
 244   kwargs = {
 245       'is_training':
 246           is_training,
 247       'depth_multiplier':
 248           depth_multiplier,
 249       'min_depth':
 250           min_depth,
 251       'pad_to_multiple':
 252           pad_to_multiple,
 253       'use_explicit_padding':
 254           use_explicit_padding,
 255       'use_depthwise':
 256           use_depthwise,
 257       'override_base_feature_extractor_hyperparams':
 258           override_base_feature_extractor_hyperparams
 259   }
 260 
 261   if feature_extractor_config.HasField('replace_preprocessor_with_placeholder'):
 262     kwargs.update({
 263         'replace_preprocessor_with_placeholder':
 264             feature_extractor_config.replace_preprocessor_with_placeholder
 265     })
 266 
 267   if feature_extractor_config.HasField('num_layers'):
 268     kwargs.update({'num_layers': feature_extractor_config.num_layers})
 269 
 270   if is_keras:
 271     kwargs.update({
 272         'conv_hyperparams': conv_hyperparams,
 273         'inplace_batchnorm_update': False,
 274         'freeze_batchnorm': freeze_batchnorm
 275     })
 276   else:
 277     kwargs.update({
 278         'conv_hyperparams_fn': conv_hyperparams,
 279         'reuse_weights': reuse_weights,
 280     })
 281 
 282 
 283   if feature_extractor_config.HasField('fpn'):
 284     kwargs.update({
 285         'fpn_min_level':
 286             feature_extractor_config.fpn.min_level,
 287         'fpn_max_level':
 288             feature_extractor_config.fpn.max_level,
 289         'additional_layer_depth':
 290             feature_extractor_config.fpn.additional_layer_depth,
 291     })
 292 
 293 
 294   return feature_extractor_class(**kwargs)

 297 def _build_ssd_model(ssd_config, is_training, add_summaries):
 298   """Builds an SSD detection model based on the model config.
 299 
 300   Args:
 301     ssd_config: A ssd.proto object containing the config for the desired
 302       SSDMetaArch.
 303     is_training: True if this model is being built for training purposes.
 304     add_summaries: Whether to add tf summaries in the model.
 305   Returns:
 306     SSDMetaArch based on the config.
 307 
 308   Raises:
 309     ValueError: If ssd_config.type is not recognized (i.e. not registered in
 310       model_class_map).
 311   """
 312   num_classes = ssd_config.num_classes
 313   _check_feature_extractor_exists(ssd_config.feature_extractor.type)
 314 
 315   # Feature extractor
 316   feature_extractor = _build_ssd_feature_extractor(
 317       feature_extractor_config=ssd_config.feature_extractor,
 318       freeze_batchnorm=ssd_config.freeze_batchnorm,
 319       is_training=is_training)
 320 
 321   box_coder = box_coder_builder.build(ssd_config.box_coder)
 322   matcher = matcher_builder.build(ssd_config.matcher)
 323   region_similarity_calculator = sim_calc.build(
 324       ssd_config.similarity_calculator)
 325   encode_background_as_zeros = ssd_config.encode_background_as_zeros
 326   negative_class_weight = ssd_config.negative_class_weight
 327   anchor_generator = anchor_generator_builder.build(
 328       ssd_config.anchor_generator)
 329   if feature_extractor.is_keras_model:
 330     ssd_box_predictor = box_predictor_builder.build_keras(
 331         hyperparams_fn=hyperparams_builder.KerasLayerHyperparams,
 332         freeze_batchnorm=ssd_config.freeze_batchnorm,
 333         inplace_batchnorm_update=False,
 334         num_predictions_per_location_list=anchor_generator
 335         .num_anchors_per_location(),
 336         box_predictor_config=ssd_config.box_predictor,
 337         is_training=is_training,
 338         num_classes=num_classes,
 339         add_background_class=ssd_config.add_background_class)
 340   else:
 341     ssd_box_predictor = box_predictor_builder.build(
 342         hyperparams_builder.build, ssd_config.box_predictor, is_training,
 343         num_classes, ssd_config.add_background_class)
 344   image_resizer_fn = image_resizer_builder.build(ssd_config.image_resizer)
 345   non_max_suppression_fn, score_conversion_fn = post_processing_builder.build(
 346       ssd_config.post_processing)
 347   (classification_loss, localization_loss, classification_weight,
 348    localization_weight, hard_example_miner, random_example_sampler,
 349    expected_loss_weights_fn) = losses_builder.build(ssd_config.loss)
 350   normalize_loss_by_num_matches = ssd_config.normalize_loss_by_num_matches
 351   normalize_loc_loss_by_codesize = ssd_config.normalize_loc_loss_by_codesize
 352 
 353   equalization_loss_config = ops.EqualizationLossConfig(
 354       weight=ssd_config.loss.equalization_loss.weight,
 355       exclude_prefixes=ssd_config.loss.equalization_loss.exclude_prefixes)
 356 
 357   target_assigner_instance = target_assigner.TargetAssigner(
 358       region_similarity_calculator,
 359       matcher,
 360       box_coder,
 361       negative_class_weight=negative_class_weight)
 362 
 363   ssd_meta_arch_fn = ssd_meta_arch.SSDMetaArch
 364   kwargs = {}
 365 
 366   return ssd_meta_arch_fn(
 367       is_training=is_training,
 368       anchor_generator=anchor_generator,
 369       box_predictor=ssd_box_predictor,
 370       box_coder=box_coder,
 371       feature_extractor=feature_extractor,
 372       encode_background_as_zeros=encode_background_as_zeros,
 373       image_resizer_fn=image_resizer_fn,
 374       non_max_suppression_fn=non_max_suppression_fn,
 375       score_conversion_fn=score_conversion_fn,
 376       classification_loss=classification_loss,
 377       localization_loss=localization_loss,
 378       classification_loss_weight=classification_weight,
 379       localization_loss_weight=localization_weight,
 380       normalize_loss_by_num_matches=normalize_loss_by_num_matches,
 381       hard_example_miner=hard_example_miner,
 382       target_assigner_instance=target_assigner_instance,
 383       add_summaries=add_summaries,
 384       normalize_loc_loss_by_codesize=normalize_loc_loss_by_codesize,
 385       freeze_batchnorm=ssd_config.freeze_batchnorm,
 386       inplace_batchnorm_update=ssd_config.inplace_batchnorm_update,
 387       add_background_class=ssd_config.add_background_class,
 388       explicit_background_class=ssd_config.explicit_background_class,
 389       random_example_sampler=random_example_sampler,
 390       expected_loss_weights_fn=expected_loss_weights_fn,
 391       use_confidences_as_targets=ssd_config.use_confidences_as_targets,
 392       implicit_example_weight=ssd_config.implicit_example_weight,
 393       equalization_loss_config=equalization_loss_config,
 394       return_raw_detections_during_predict=(
 395           ssd_config.return_raw_detections_during_predict),
 396       **kwargs)

 399 def _build_faster_rcnn_feature_extractor(
 400     feature_extractor_config, is_training, reuse_weights=True,
 401     inplace_batchnorm_update=False):
 402   """Builds a faster_rcnn_meta_arch.FasterRCNNFeatureExtractor based on config.
 403 
 404   Args:
 405     feature_extractor_config: A FasterRcnnFeatureExtractor proto config from
 406       faster_rcnn.proto.
 407     is_training: True if this feature extractor is being built for training.
 408     reuse_weights: if the feature extractor should reuse weights.
 409     inplace_batchnorm_update: Whether to update batch_norm inplace during
 410       training. This is required for batch norm to work correctly on TPUs. When
 411       this is false, user must add a control dependency on
 412       tf.GraphKeys.UPDATE_OPS for train/loss op in order to update the batch
 413       norm moving average parameters.
 414 
 415   Returns:
 416     faster_rcnn_meta_arch.FasterRCNNFeatureExtractor based on config.
 417 
 418   Raises:
 419     ValueError: On invalid feature extractor type.
 420   """
 421   if inplace_batchnorm_update:
 422     raise ValueError('inplace batchnorm updates not supported.')
 423   feature_type = feature_extractor_config.type
 424   first_stage_features_stride = (
 425       feature_extractor_config.first_stage_features_stride)
 426   batch_norm_trainable = feature_extractor_config.batch_norm_trainable
 427 
 428   if feature_type not in FASTER_RCNN_FEATURE_EXTRACTOR_CLASS_MAP:
 429     raise ValueError('Unknown Faster R-CNN feature_extractor: {}'.format(
 430         feature_type))
 431   feature_extractor_class = FASTER_RCNN_FEATURE_EXTRACTOR_CLASS_MAP[
 432       feature_type]
 433   return feature_extractor_class(
 434       is_training, first_stage_features_stride,
 435       batch_norm_trainable, reuse_weights=reuse_weights)

 442 
 443   Args:
 444     feature_extractor_config: A FasterRcnnFeatureExtractor proto config from
 445       faster_rcnn.proto.

 447     inplace_batchnorm_update: Whether to update batch_norm inplace during

 449       this is false, user must add a control dependency on
 450       tf.GraphKeys.UPDATE_OPS for train/loss op in order to update the batch
 451       norm moving average parameters.
 452 
 453   Returns:
 454     faster_rcnn_meta_arch.FasterRCNNKerasFeatureExtractor based on config.
 455

 458   """

 460     raise ValueError('inplace batchnorm updates not supported.')
 461   feature_type = feature_extractor_config.type
 462   first_stage_features_stride = (
 463       feature_extractor_config.first_stage_features_stride)
 464   batch_norm_trainable = feature_extractor_config.batch_norm_trainable
 465 
 466   if feature_type not in FASTER_RCNN_KERAS_FEATURE_EXTRACTOR_CLASS_MAP:
 467     raise ValueError('Unknown Faster R-CNN feature_extractor: {}'.format(
 468         feature_type))
 469   feature_extractor_class = FASTER_RCNN_KERAS_FEATURE_EXTRACTOR_CLASS_MAP[
 470       feature_type]
 471   return feature_extractor_class(
 472       is_training, first_stage_features_stride,
 473       batch_norm_trainable)

 476 def _build_faster_rcnn_model(frcnn_config, is_training, add_summaries):
 477   """Builds a Faster R-CNN or R-FCN detection model based on the model config.
 478 
 479   Builds R-FCN model if the second_stage_box_predictor in the config is of type
 480   `rfcn_box_predictor` else builds a Faster R-CNN model.
 481 
 482   Args:
 483     frcnn_config: A faster_rcnn.proto object containing the config for the
 484       desired FasterRCNNMetaArch or RFCNMetaArch.
 485     is_training: True if this model is being built for training purposes.
 486     add_summaries: Whether to add tf summaries in the model.
 487 
 488   Returns:
 489     FasterRCNNMetaArch based on the config.
 490 
 491   Raises:
 492     ValueError: If frcnn_config.type is not recognized (i.e. not registered in
 493       model_class_map).
 494   """
 495   num_classes = frcnn_config.num_classes
 496   image_resizer_fn = image_resizer_builder.build(frcnn_config.image_resizer)
 497   _check_feature_extractor_exists(frcnn_config.feature_extractor.type)
 498   is_keras = tf_version.is_tf2()
 499 
 500   if is_keras:
 501     feature_extractor = _build_faster_rcnn_keras_feature_extractor(
 502         frcnn_config.feature_extractor, is_training,
 503         inplace_batchnorm_update=frcnn_config.inplace_batchnorm_update)
 504   else:
 505     feature_extractor = _build_faster_rcnn_feature_extractor(
 506         frcnn_config.feature_extractor, is_training,
 507         inplace_batchnorm_update=frcnn_config.inplace_batchnorm_update)
 508 
 509   number_of_stages = frcnn_config.number_of_stages
 510   first_stage_anchor_generator = anchor_generator_builder.build(
 511       frcnn_config.first_stage_anchor_generator)
 512 
 513   first_stage_target_assigner = target_assigner.create_target_assigner(
 514       'FasterRCNN',
 515       'proposal',
 516       use_matmul_gather=frcnn_config.use_matmul_gather_in_matcher)
 517   first_stage_atrous_rate = frcnn_config.first_stage_atrous_rate
 518   if is_keras:
 519     first_stage_box_predictor_arg_scope_fn = (
 520         hyperparams_builder.KerasLayerHyperparams(
 521             frcnn_config.first_stage_box_predictor_conv_hyperparams))
 522   else:
 523     first_stage_box_predictor_arg_scope_fn = hyperparams_builder.build(
 524         frcnn_config.first_stage_box_predictor_conv_hyperparams, is_training)
 525   first_stage_box_predictor_kernel_size = (
 526       frcnn_config.first_stage_box_predictor_kernel_size)
 527   first_stage_box_predictor_depth = frcnn_config.first_stage_box_predictor_depth
 528   first_stage_minibatch_size = frcnn_config.first_stage_minibatch_size
 529   use_static_shapes = frcnn_config.use_static_shapes and (
 530       frcnn_config.use_static_shapes_for_eval or is_training)
 531   first_stage_sampler = sampler.BalancedPositiveNegativeSampler(
 532       positive_fraction=frcnn_config.first_stage_positive_balance_fraction,
 533       is_static=(frcnn_config.use_static_balanced_label_sampler and
 534                  use_static_shapes))
 535   first_stage_max_proposals = frcnn_config.first_stage_max_proposals
 536   if (frcnn_config.first_stage_nms_iou_threshold < 0 or
 537       frcnn_config.first_stage_nms_iou_threshold > 1.0):
 538     raise ValueError('iou_threshold not in [0, 1.0].')
 539   if (is_training and frcnn_config.second_stage_batch_size >
 540       first_stage_max_proposals):
 541     raise ValueError('second_stage_batch_size should be no greater than '
 542                      'first_stage_max_proposals.')
 543   first_stage_non_max_suppression_fn = functools.partial(
 544       post_processing.batch_multiclass_non_max_suppression,
 545       score_thresh=frcnn_config.first_stage_nms_score_threshold,
 546       iou_thresh=frcnn_config.first_stage_nms_iou_threshold,
 547       max_size_per_class=frcnn_config.first_stage_max_proposals,
 548       max_total_size=frcnn_config.first_stage_max_proposals,
 549       use_static_shapes=use_static_shapes,
 550       use_partitioned_nms=frcnn_config.use_partitioned_nms_in_first_stage,
 551       use_combined_nms=frcnn_config.use_combined_nms_in_first_stage)
 552   first_stage_loc_loss_weight = (
 553       frcnn_config.first_stage_localization_loss_weight)
 554   first_stage_obj_loss_weight = frcnn_config.first_stage_objectness_loss_weight
 555 
 556   initial_crop_size = frcnn_config.initial_crop_size
 557   maxpool_kernel_size = frcnn_config.maxpool_kernel_size
 558   maxpool_stride = frcnn_config.maxpool_stride
 559 
 560   second_stage_target_assigner = target_assigner.create_target_assigner(
 561       'FasterRCNN',
 562       'detection',
 563       use_matmul_gather=frcnn_config.use_matmul_gather_in_matcher)
 564   if is_keras:
 565     second_stage_box_predictor = box_predictor_builder.build_keras(
 566         hyperparams_builder.KerasLayerHyperparams,
 567         freeze_batchnorm=False,
 568         inplace_batchnorm_update=False,
 569         num_predictions_per_location_list=[1],
 570         box_predictor_config=frcnn_config.second_stage_box_predictor,
 571         is_training=is_training,
 572         num_classes=num_classes)
 573   else:
 574     second_stage_box_predictor = box_predictor_builder.build(
 575         hyperparams_builder.build,
 576         frcnn_config.second_stage_box_predictor,
 577         is_training=is_training,
 578         num_classes=num_classes)
 579   second_stage_batch_size = frcnn_config.second_stage_batch_size
 580   second_stage_sampler = sampler.BalancedPositiveNegativeSampler(
 581       positive_fraction=frcnn_config.second_stage_balance_fraction,
 582       is_static=(frcnn_config.use_static_balanced_label_sampler and
 583                  use_static_shapes))
 584   (second_stage_non_max_suppression_fn, second_stage_score_conversion_fn
 585   ) = post_processing_builder.build(frcnn_config.second_stage_post_processing)
 586   second_stage_localization_loss_weight = (
 587       frcnn_config.second_stage_localization_loss_weight)
 588   second_stage_classification_loss = (
 589       losses_builder.build_faster_rcnn_classification_loss(
 590           frcnn_config.second_stage_classification_loss))
 591   second_stage_classification_loss_weight = (
 592       frcnn_config.second_stage_classification_loss_weight)
 593   second_stage_mask_prediction_loss_weight = (
 594       frcnn_config.second_stage_mask_prediction_loss_weight)
 595 
 596   hard_example_miner = None
 597   if frcnn_config.HasField('hard_example_miner'):
 598     hard_example_miner = losses_builder.build_hard_example_miner(
 599         frcnn_config.hard_example_miner,
 600         second_stage_classification_loss_weight,
 601         second_stage_localization_loss_weight)
 602 
 603   crop_and_resize_fn = (
 604       ops.matmul_crop_and_resize if frcnn_config.use_matmul_crop_and_resize
 605       else ops.native_crop_and_resize)
 606   clip_anchors_to_image = (
 607       frcnn_config.clip_anchors_to_image)
 608 
 609   common_kwargs = {
 610       'is_training':
 611           is_training,
 612       'num_classes':
 613           num_classes,
 614       'image_resizer_fn':
 615           image_resizer_fn,
 616       'feature_extractor':
 617           feature_extractor,
 618       'number_of_stages':
 619           number_of_stages,
 620       'first_stage_anchor_generator':
 621           first_stage_anchor_generator,
 622       'first_stage_target_assigner':
 623           first_stage_target_assigner,
 624       'first_stage_atrous_rate':
 625           first_stage_atrous_rate,
 626       'first_stage_box_predictor_arg_scope_fn':
 627           first_stage_box_predictor_arg_scope_fn,
 628       'first_stage_box_predictor_kernel_size':
 629           first_stage_box_predictor_kernel_size,
 630       'first_stage_box_predictor_depth':
 631           first_stage_box_predictor_depth,
 632       'first_stage_minibatch_size':
 633           first_stage_minibatch_size,
 634       'first_stage_sampler':
 635           first_stage_sampler,
 636       'first_stage_non_max_suppression_fn':
 637           first_stage_non_max_suppression_fn,
 638       'first_stage_max_proposals':
 639           first_stage_max_proposals,
 640       'first_stage_localization_loss_weight':
 641           first_stage_loc_loss_weight,
 642       'first_stage_objectness_loss_weight':
 643           first_stage_obj_loss_weight,
 644       'second_stage_target_assigner':
 645           second_stage_target_assigner,
 646       'second_stage_batch_size':
 647           second_stage_batch_size,
 648       'second_stage_sampler':
 649           second_stage_sampler,
 650       'second_stage_non_max_suppression_fn':
 651           second_stage_non_max_suppression_fn,
 652       'second_stage_score_conversion_fn':
 653           second_stage_score_conversion_fn,
 654       'second_stage_localization_loss_weight':
 655           second_stage_localization_loss_weight,
 656       'second_stage_classification_loss':
 657           second_stage_classification_loss,
 658       'second_stage_classification_loss_weight':
 659           second_stage_classification_loss_weight,
 660       'hard_example_miner':
 661           hard_example_miner,
 662       'add_summaries':
 663           add_summaries,
 664       'crop_and_resize_fn':
 665           crop_and_resize_fn,
 666       'clip_anchors_to_image':
 667           clip_anchors_to_image,

 669           use_static_shapes,

 671           frcnn_config.resize_masks,
 672       'return_raw_detections_during_predict':
 673           frcnn_config.return_raw_detections_during_predict,
 674       'output_final_box_features':
 675           frcnn_config.output_final_box_features
 676   }
 677 
 678   if ((not is_keras and isinstance(second_stage_box_predictor,
 679                                    rfcn_box_predictor.RfcnBoxPredictor)) or
 680       (is_keras and
 681        isinstance(second_stage_box_predictor,
 682                   rfcn_keras_box_predictor.RfcnKerasBoxPredictor))):
 683     return rfcn_meta_arch.RFCNMetaArch(
 684         second_stage_rfcn_box_predictor=second_stage_box_predictor,
 685         **common_kwargs)
 686   elif frcnn_config.HasField('context_config'):

 689         'attention_bottleneck_dimension':
 690             context_config.attention_bottleneck_dimension,

 694     return context_rcnn_meta_arch.ContextRCNNMetaArch(
 695         initial_crop_size=initial_crop_size,

 697         maxpool_stride=maxpool_stride,
 698         second_stage_mask_rcnn_box_predictor=second_stage_box_predictor,
 699         second_stage_mask_prediction_loss_weight=(
 700             second_stage_mask_prediction_loss_weight),

 728 def keypoint_proto_to_params(kp_config, keypoint_map_dict):
 729   """Converts CenterNet.KeypointEstimation proto to parameter namedtuple."""
 730   label_map_item = keypoint_map_dict[kp_config.keypoint_class_name]
 731 
 732   classification_loss, localization_loss, _, _, _, _, _ = (
 733       losses_builder.build(kp_config.loss))
 734 
 735   keypoint_indices = [
 736       keypoint.id for keypoint in label_map_item.keypoints
 737   ]
 738   keypoint_labels = [
 739       keypoint.label for keypoint in label_map_item.keypoints
 740   ]
 741   keypoint_std_dev_dict = {
 742       label: KEYPOINT_STD_DEV_DEFAULT for label in keypoint_labels

 744   if kp_config.keypoint_label_to_std:
 745     for label, value in kp_config.keypoint_label_to_std.items():
 746       keypoint_std_dev_dict[label] = value
 747   keypoint_std_dev = [keypoint_std_dev_dict[label] for label in keypoint_labels]
 748   return center_net_meta_arch.KeypointEstimationParams(
 749       task_name=kp_config.task_name,
 750       class_id=label_map_item.id - CLASS_ID_OFFSET,
 751       keypoint_indices=keypoint_indices,
 752       classification_loss=classification_loss,
 753       localization_loss=localization_loss,
 754       keypoint_labels=keypoint_labels,
 755       keypoint_std_dev=keypoint_std_dev,
 756       task_loss_weight=kp_config.task_loss_weight,
 757       keypoint_regression_loss_weight=kp_config.keypoint_regression_loss_weight,
 758       keypoint_heatmap_loss_weight=kp_config.keypoint_heatmap_loss_weight,
 759       keypoint_offset_loss_weight=kp_config.keypoint_offset_loss_weight,
 760       heatmap_bias_init=kp_config.heatmap_bias_init,

 762           kp_config.keypoint_candidate_score_threshold),

 824 def _build_center_net_model(center_net_config, is_training, add_summaries):
 825   """Build a CenterNet detection model.
 826 
 827   Args:
 828     center_net_config: A CenterNet proto object with model configuration.
 829     is_training: True if this model is being built for training purposes.
 830     add_summaries: Whether to add tf summaries in the model.
 831 
 832   Returns:
 833     CenterNetMetaArch based on the config.
 834 
 835   """
 836 
 837   image_resizer_fn = image_resizer_builder.build(
 838       center_net_config.image_resizer)
 839   _check_feature_extractor_exists(center_net_config.feature_extractor.type)
 840   feature_extractor = _build_center_net_feature_extractor(
 841       center_net_config.feature_extractor)
 842   object_center_params = object_center_proto_to_params(
 843       center_net_config.object_center_params)
 844 
 845   object_detection_params = None
 846   if center_net_config.HasField('object_detection_task'):
 847     object_detection_params = object_detection_proto_to_params(
 848         center_net_config.object_detection_task)
 849 
 850   keypoint_params_dict = None
 851   if center_net_config.keypoint_estimation_task:
 852     label_map_proto = label_map_util.load_labelmap(
 853         center_net_config.keypoint_label_map_path)
 854     keypoint_map_dict = {
 855         item.name: item for item in label_map_proto.item if item.keypoints
 856     }
 857     keypoint_params_dict = {}
 858     keypoint_class_id_set = set()
 859     all_keypoint_indices = []
 860     for task in center_net_config.keypoint_estimation_task:
 861       kp_params = keypoint_proto_to_params(task, keypoint_map_dict)
 862       keypoint_params_dict[task.task_name] = kp_params
 863       all_keypoint_indices.extend(kp_params.keypoint_indices)
 864       if kp_params.class_id in keypoint_class_id_set:
 865         raise ValueError(('Multiple keypoint tasks map to the same class id is '
 866                           'not allowed: %d' % kp_params.class_id))
 867       else:
 868         keypoint_class_id_set.add(kp_params.class_id)
 869     if len(all_keypoint_indices) > len(set(all_keypoint_indices)):
 870       raise ValueError('Some keypoint indices are used more than once.')
 871 
 872   mask_params = None
 873   if center_net_config.HasField('mask_estimation_task'):
 874     mask_params = mask_proto_to_params(center_net_config.mask_estimation_task)
 875 
 876   return center_net_meta_arch.CenterNetMetaArch(
 877       is_training=is_training,
 878       add_summaries=add_summaries,
 879       num_classes=center_net_config.num_classes,
 880       feature_extractor=feature_extractor,
 881       image_resizer_fn=image_resizer_fn,
 882       object_center_params=object_center_params,
 883       object_detection_params=object_detection_params,
 884       keypoint_params_dict=keypoint_params_dict,
 885       mask_params=mask_params)

 911 def build(model_config, is_training, add_summaries=True):
 912   """Builds a DetectionModel based on the model config.
 913 
 914   Args:
 915     model_config: A model.proto object containing the config for the desired
 916       DetectionModel.
 917     is_training: True if this model is being built for training purposes.
 918     add_summaries: Whether to add tensorflow summaries in the model graph.
 919   Returns:
 920     DetectionModel based on the config.
 921 
 922   Raises:
 923     ValueError: On invalid meta architecture or model.
 924   """
 925   if not isinstance(model_config, model_pb2.DetectionModel):
 926     raise ValueError('model_config not of type model_pb2.DetectionModel.')
 927 
 928   meta_architecture = model_config.WhichOneof('model')
 929 
 930   if meta_architecture not in META_ARCH_BUILDER_MAP:
 931     raise ValueError('Unknown meta architecture: {}'.format(meta_architecture))
 932   else:
 933     build_func = META_ARCH_BUILDER_MAP[meta_architecture]
 934     return build_func(getattr(model_config, meta_architecture), is_training,
 935                       add_summaries)
