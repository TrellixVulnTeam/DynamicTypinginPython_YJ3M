 128   def test_return_l1_regularized_weights(self):
 129     conv_hyperparams_text_proto = """
 130       regularizer {
 131         l1_regularizer {
 132           weight: 0.5
 133         }
 134       }
 135       initializer {
 136         truncated_normal_initializer {
 137         }
 138       }
 139     """
 140     conv_hyperparams_proto = hyperparams_pb2.Hyperparams()
 141     text_format.Merge(conv_hyperparams_text_proto, conv_hyperparams_proto)
 142     scope_fn = hyperparams_builder.build(conv_hyperparams_proto,
 143                                          is_training=True)
 144     scope = scope_fn()
 145     conv_scope_arguments = list(scope.values())[0]
 146     regularizer = conv_scope_arguments['weights_regularizer']
 147     weights = np.array([1., -1, 4., 2.])
 148     with self.test_session() as sess:
 149       result = sess.run(regularizer(tf.constant(weights)))
 150     self.assertAllClose(np.abs(weights).sum() * 0.5, result)

 616   def _assert_variance_in_range(self, initializer, shape, variance,
 617                                 tol=1e-2):
 618     with tf.Graph().as_default() as g:
 619       with self.test_session(graph=g) as sess:
 620         var = tf.get_variable(
 621             name='test',
 622             shape=shape,
 623             dtype=tf.float32,
 624             initializer=initializer)
