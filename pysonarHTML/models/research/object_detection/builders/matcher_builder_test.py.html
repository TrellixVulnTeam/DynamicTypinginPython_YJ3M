  16 """Tests for matcher_builder."""
  17 
  18 import tensorflow.compat.v1 as tf
  19 
  20 from google.protobuf import text_format
  21 from object_detection.builders import matcher_builder
  22 from object_detection.matchers import argmax_matcher
  23 from object_detection.matchers import bipartite_matcher
  24 from object_detection.protos import matcher_pb2
  25 
  26 
  27 class MatcherBuilderTest(tf.test.TestCase):
  28 
  29   def test_build_arg_max_matcher_with_defaults(self):
  30     matcher_text_proto = """
  31       argmax_matcher {
  32       }
  33     """
  34     matcher_proto = matcher_pb2.Matcher()
  35     text_format.Merge(matcher_text_proto, matcher_proto)
  36     matcher_object = matcher_builder.build(matcher_proto)
  37     self.assertTrue(isinstance(matcher_object, argmax_matcher.ArgMaxMatcher))
  38     self.assertAlmostEqual(matcher_object._matched_threshold, 0.5)
  39     self.assertAlmostEqual(matcher_object._unmatched_threshold, 0.5)
  40     self.assertTrue(matcher_object._negatives_lower_than_unmatched)
  41     self.assertFalse(matcher_object._force_match_for_each_row)
  42 
  43   def test_build_arg_max_matcher_without_thresholds(self):
  44     matcher_text_proto = """
  45       argmax_matcher {
  46         ignore_thresholds: true
  47       }
  48     """
  49     matcher_proto = matcher_pb2.Matcher()
  50     text_format.Merge(matcher_text_proto, matcher_proto)
  51     matcher_object = matcher_builder.build(matcher_proto)
  52     self.assertTrue(isinstance(matcher_object, argmax_matcher.ArgMaxMatcher))
  53     self.assertEqual(matcher_object._matched_threshold, None)
  54     self.assertEqual(matcher_object._unmatched_threshold, None)
  55     self.assertTrue(matcher_object._negatives_lower_than_unmatched)
  56     self.assertFalse(matcher_object._force_match_for_each_row)
  57 
  58   def test_build_arg_max_matcher_with_non_default_parameters(self):
  59     matcher_text_proto = """
  60       argmax_matcher {
  61         matched_threshold: 0.7
  62         unmatched_threshold: 0.3
  63         negatives_lower_than_unmatched: false
  64         force_match_for_each_row: true
  65         use_matmul_gather: true
  66       }
  67     """
  68     matcher_proto = matcher_pb2.Matcher()
  69     text_format.Merge(matcher_text_proto, matcher_proto)
  70     matcher_object = matcher_builder.build(matcher_proto)
  71     self.assertTrue(isinstance(matcher_object, argmax_matcher.ArgMaxMatcher))
  72     self.assertAlmostEqual(matcher_object._matched_threshold, 0.7)
  73     self.assertAlmostEqual(matcher_object._unmatched_threshold, 0.3)
  74     self.assertFalse(matcher_object._negatives_lower_than_unmatched)
  75     self.assertTrue(matcher_object._force_match_for_each_row)
  76     self.assertTrue(matcher_object._use_matmul_gather)
  77 
  78   def test_build_bipartite_matcher(self):
  79     matcher_text_proto = """
  80       bipartite_matcher {
  81       }
  82     """
  83     matcher_proto = matcher_pb2.Matcher()
  84     text_format.Merge(matcher_text_proto, matcher_proto)
  85     matcher_object = matcher_builder.build(matcher_proto)
  86     self.assertTrue(
  87         isinstance(matcher_object, bipartite_matcher.GreedyBipartiteMatcher))
  88 
  89   def test_raise_error_on_empty_matcher(self):
  90     matcher_text_proto = """
  91     """
  92     matcher_proto = matcher_pb2.Matcher()
  93     text_format.Merge(matcher_text_proto, matcher_proto)
  94     with self.assertRaises(ValueError):
  95       matcher_builder.build(matcher_proto)
