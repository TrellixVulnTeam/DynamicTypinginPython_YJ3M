 391   def test_build_all_loss_parameters(self):
 392     losses_text_proto = """
 393       localization_loss {
 394         weighted_l2 {
 395         }
 396       }
 397       classification_loss {
 398         weighted_softmax {
 399         }
 400       }
 401       hard_example_miner {
 402       }
 403       classification_weight: 0.8
 404       localization_weight: 0.2
 405     """
 406     losses_proto = losses_pb2.Loss()
 407     text_format.Merge(losses_text_proto, losses_proto)
 408     (classification_loss, localization_loss, classification_weight,
 409      localization_weight, hard_example_miner, _,
 410      _) = losses_builder.build(losses_proto)
 411     self.assertIsInstance(hard_example_miner, losses.HardExampleMiner)
 412     self.assertIsInstance(classification_loss,
 413                           losses.WeightedSoftmaxClassificationLoss)
 414     self.assertIsInstance(localization_loss,
 415                           losses.WeightedL2LocalizationLoss)
 416     self.assertAlmostEqual(classification_weight, 0.8)
 417     self.assertAlmostEqual(localization_weight, 0.2)
