  37 def make_initializable_iterator(dataset):
  38   """Creates an iterator, and initializes tables.
  39 
  40   This is useful in cases where make_one_shot_iterator wouldn't work because
  41   the graph contains a hash table that needs to be initialized.
  42 
  43   Args:
  44     dataset: A `tf.data.Dataset` object.
  45 
  46   Returns:
  47     A `tf.data.Iterator`.
  48   """
  49   iterator = dataset.make_initializable_iterator()
  50   tf.add_to_collection(tf.GraphKeys.TABLE_INITIALIZERS, iterator.initializer)
  51   return iterator

  54 def read_dataset(file_read_func, input_files, config,
  55                  filename_shard_fn=None):
  56   """Reads a dataset, and handles repetition and shuffling.
  57 
  58   Args:
  59     file_read_func: Function to use in tf_data.parallel_interleave, to
  60       read every individual file into a tf.data.Dataset.
  61     input_files: A list of file paths to read.
  62     config: A input_reader_builder.InputReader object.
  63     filename_shard_fn: optional, A funciton used to shard filenames across
  64       replicas. This function takes as input a TF dataset of filenames and
  65       is expected to return its sharded version. It is useful when the
  66       dataset is being loaded on one of possibly many replicas and we want
  67       to evenly shard the files between the replicas.
  68 
  69   Returns:
  70     A tf.data.Dataset of (undecoded) tf-records based on config.
  71 
  72   Raises:
  73     RuntimeError: If no files are found at the supplied path(s).
  74   """
  75   # Shard, shuffle, and read files.
  76   filenames = tf.gfile.Glob(input_files)
  77   if not filenames:
  78     raise RuntimeError('Did not find any input files matching the glob pattern '
  79                        '{}'.format(input_files))
  80   num_readers = config.num_readers
  81   if num_readers > len(filenames):
  82     num_readers = len(filenames)
  83     tf.logging.warning('num_readers has been reduced to %d to match input file '
  84                        'shards.' % num_readers)
  85   filename_dataset = tf.data.Dataset.from_tensor_slices(filenames)
  86   if config.shuffle:
  87     filename_dataset = filename_dataset.shuffle(
  88         config.filenames_shuffle_buffer_size)
  89   elif num_readers > 1:
  90     tf.logging.warning('`shuffle` is false, but the input data stream is '
  91                        'still slightly shuffled since `num_readers` > 1.')
  92   if filename_shard_fn:
  93     filename_dataset = filename_shard_fn(filename_dataset)
  94 
  95   filename_dataset = filename_dataset.repeat(config.num_epochs or None)
  96   records_dataset = filename_dataset.apply(
  97       tf_data.parallel_interleave(
  98           file_read_func,
  99           cycle_length=num_readers,
 100           block_length=config.read_block_length,
 101           sloppy=config.shuffle))
 102   if config.shuffle:
 103     records_dataset = records_dataset.shuffle(config.shuffle_buffer_size)
 104   return records_dataset

 107 def shard_function_for_context(input_context):
 108   """Returns a function that shards filenames based on the input context."""
 109 
 110   if input_context is None:
 111     return None
 112 
 113   def shard_fn(dataset):
 114     return dataset.shard(
 115         input_context.num_input_pipelines, input_context.input_pipeline_id)
 116 
 117   return shard_fn

 120 def build(input_reader_config, batch_size=None, transform_input_data_fn=None,
 121           input_context=None, reduce_to_frame_fn=None):
 122   """Builds a tf.data.Dataset.
 123 
 124   Builds a tf.data.Dataset by applying the `transform_input_data_fn` on all
 125   records. Applies a padded batch to the resulting dataset.
 126 
 127   Args:
 128     input_reader_config: A input_reader_pb2.InputReader object.
 129     batch_size: Batch size. If batch size is None, no batching is performed.
 130     transform_input_data_fn: Function to apply transformation to all records,
 131       or None if no extra decoding is required.
 132     input_context: optional, A tf.distribute.InputContext object used to
 133       shard filenames and compute per-replica batch_size when this function
 134       is being called per-replica.
 135     reduce_to_frame_fn: Function that extracts frames from tf.SequenceExample
 136       type input data.
 137 
 138   Returns:
 139     A tf.data.Dataset based on the input_reader_config.
 140 
 141   Raises:
 142     ValueError: On invalid input reader proto.
 143     ValueError: If no input paths are specified.
 144   """
 145   if not isinstance(input_reader_config, input_reader_pb2.InputReader):
 146     raise ValueError('input_reader_config not of type '
 147                      'input_reader_pb2.InputReader.')
 148 
 149   decoder = decoder_builder.build(input_reader_config)
 150 
 151   if input_reader_config.WhichOneof('input_reader') == 'tf_record_input_reader':
 152     config = input_reader_config.tf_record_input_reader
 153     if not config.input_path:
 154       raise ValueError('At least one input path must be specified in '
 155                        '`input_reader_config`.')
 156     shard_fn = shard_function_for_context(input_context)
 157     if input_context is not None:
 158       batch_size = input_context.get_per_replica_batch_size(batch_size)
 159     dataset = read_dataset(
 160         functools.partial(tf.data.TFRecordDataset, buffer_size=8 * 1000 * 1000),
 161         config.input_path[:], input_reader_config, filename_shard_fn=shard_fn)
 162     if input_reader_config.sample_1_of_n_examples > 1:
 163       dataset = dataset.shard(input_reader_config.sample_1_of_n_examples, 0)
 164     # TODO(rathodv): make batch size a required argument once the old binaries
 165     # are deleted.
 166     dataset = dataset.map(decoder.decode, tf.data.experimental.AUTOTUNE)
 167     if reduce_to_frame_fn:
 168       dataset = reduce_to_frame_fn(dataset)
 169     if transform_input_data_fn is not None:
 170       dataset = dataset.map(transform_input_data_fn,
 171                             tf.data.experimental.AUTOTUNE)
 172     if batch_size:
 173       dataset = dataset.apply(
 174           tf_data.batch_and_drop_remainder(batch_size))
 175     dataset = dataset.prefetch(input_reader_config.num_prefetch_batches)
 176     return dataset
 177 
 178   raise ValueError('Unsupported input_reader_config.')
