  27 class KerasLayerHyperparams(object):
  28   """
  29   A hyperparameter configuration object for Keras layers used in
  30   Object Detection models.
  31   """
  32 
  33   def __init__(self, hyperparams_config):
  34     """Builds keras hyperparameter config for layers based on the proto config.
  35 
  36     It automatically converts from Slim layer hyperparameter configs to
  37     Keras layer hyperparameters. Namely, it:
  38     - Builds Keras initializers/regularizers instead of Slim ones
  39     - sets weights_regularizer/initializer to kernel_regularizer/initializer
  40     - converts batchnorm decay to momentum
  41     - converts Slim l2 regularizer weights to the equivalent Keras l2 weights
  42 
  43     Contains a hyperparameter configuration for ops that specifies kernel
  44     initializer, kernel regularizer, activation. Also contains parameters for
  45     batch norm operators based on the configuration.
  46 
  47     Note that if the batch_norm parameters are not specified in the config
  48     (i.e. left to default) then batch norm is excluded from the config.
  49 
  50     Args:
  51       hyperparams_config: hyperparams.proto object containing
  52         hyperparameters.
  53 
  54     Raises:
  55       ValueError: if hyperparams_config is not of type hyperparams.Hyperparams.
  56     """
  57     if not isinstance(hyperparams_config,
  58                       hyperparams_pb2.Hyperparams):
  59       raise ValueError('hyperparams_config not of type '
  60                        'hyperparams_pb.Hyperparams.')
  61 
  62     self._batch_norm_params = None
  63     if hyperparams_config.HasField('batch_norm'):
  64       self._batch_norm_params = _build_keras_batch_norm_params(
  65           hyperparams_config.batch_norm)
  66 
  67     self._activation_fn = _build_activation_fn(hyperparams_config.activation)
  68     # TODO(kaftan): Unclear if these kwargs apply to separable & depthwise conv
  69     # (Those might use depthwise_* instead of kernel_*)
  70     # We should probably switch to using build_conv2d_layer and
  71     # build_depthwise_conv2d_layer methods instead.
  72     self._op_params = {
  73         'kernel_regularizer': _build_keras_regularizer(
  74             hyperparams_config.regularizer),
  75         'kernel_initializer': _build_initializer(
  76             hyperparams_config.initializer, build_for_keras=True),
  77         'activation': _build_activation_fn(hyperparams_config.activation)
  78     }
  79 
  80   def use_batch_norm(self):
  81     return self._batch_norm_params is not None
  82 
  83   def batch_norm_params(self, **overrides):
  84     """Returns a dict containing batchnorm layer construction hyperparameters.
  85 
  86     Optionally overrides values in the batchnorm hyperparam dict. Overrides
  87     only apply to individual calls of this method, and do not affect
  88     future calls.
  89 
  90     Args:
  91       **overrides: keyword arguments to override in the hyperparams dictionary
  92 
  93     Returns: dict containing the layer construction keyword arguments, with
  94       values overridden by the `overrides` keyword arguments.
  95     """
  96     if self._batch_norm_params is None:
  97       new_batch_norm_params = dict()
  98     else:
  99       new_batch_norm_params = self._batch_norm_params.copy()
 100     new_batch_norm_params.update(overrides)
 101     return new_batch_norm_params
 102 
 103   def build_batch_norm(self, training=None, **overrides):
 104     """Returns a Batch Normalization layer with the appropriate hyperparams.
 105 
 106     If the hyperparams are configured to not use batch normalization,
 107     this will return a Keras Lambda layer that only applies tf.Identity,
 108     without doing any normalization.
 109 
 110     Optionally overrides values in the batch_norm hyperparam dict. Overrides
 111     only apply to individual calls of this method, and do not affect
 112     future calls.
 113 
 114     Args:
 115       training: if True, the normalization layer will normalize using the batch
 116        statistics. If False, the normalization layer will be frozen and will
 117        act as if it is being used for inference. If None, the layer
 118        will look up the Keras learning phase at `call` time to decide what to
 119        do.
 120       **overrides: batch normalization construction args to override from the
 121         batch_norm hyperparams dictionary.
 122 
 123     Returns: Either a FreezableBatchNorm layer (if use_batch_norm() is True),
 124       or a Keras Lambda layer that applies the identity (if use_batch_norm()
 125       is False)
 126     """
 127     if self.use_batch_norm():
 128       return freezable_batch_norm.FreezableBatchNorm(
 129           training=training,
 130           **self.batch_norm_params(**overrides)
 131       )
 132     else:
 133       return tf.keras.layers.Lambda(tf.identity)
 134 
 135   def build_activation_layer(self, name='activation'):
 136     """Returns a Keras layer that applies the desired activation function.
 137 
 138     Args:
 139       name: The name to assign the Keras layer.
 140     Returns: A Keras lambda layer that applies the activation function
 141       specified in the hyperparam config, or applies the identity if the
 142       activation function is None.
 143     """
 144     if self._activation_fn:
 145       return tf.keras.layers.Lambda(self._activation_fn, name=name)
 146     else:
 147       return tf.keras.layers.Lambda(tf.identity, name=name)
 148 
 149   def params(self, include_activation=False, **overrides):
 150     """Returns a dict containing the layer construction hyperparameters to use.
 151 
 152     Optionally overrides values in the returned dict. Overrides
 153     only apply to individual calls of this method, and do not affect
 154     future calls.
 155 
 156     Args:
 157       include_activation: If False, activation in the returned dictionary will
 158         be set to `None`, and the activation must be applied via a separate
 159         layer created by `build_activation_layer`. If True, `activation` in the
 160         output param dictionary will be set to the activation function
 161         specified in the hyperparams config.
 162       **overrides: keyword arguments to override in the hyperparams dictionary.
 163 
 164     Returns: dict containing the layer construction keyword arguments, with
 165       values overridden by the `overrides` keyword arguments.
 166     """
 167     new_params = self._op_params.copy()
 168     new_params['activation'] = None
 169     if include_activation:
 170       new_params['activation'] = self._activation_fn
 171     if self.use_batch_norm() and self.batch_norm_params()['center']:
 172       new_params['use_bias'] = False
 173     else:
 174       new_params['use_bias'] = True
 175     new_params.update(**overrides)
 176     return new_params

 179 def build(hyperparams_config, is_training):
 180   """Builds tf-slim arg_scope for convolution ops based on the config.
 181 
 182   Returns an arg_scope to use for convolution ops containing weights
 183   initializer, weights regularizer, activation function, batch norm function
 184   and batch norm parameters based on the configuration.
 185 
 186   Note that if no normalization parameters are specified in the config,
 187   (i.e. left to default) then both batch norm and group norm are excluded
 188   from the arg_scope.
 189 
 190   The batch norm parameters are set for updates based on `is_training` argument
 191   and conv_hyperparams_config.batch_norm.train parameter. During training, they
 192   are updated only if batch_norm.train parameter is true. However, during eval,
 193   no updates are made to the batch norm variables. In both cases, their current
 194   values are used during forward pass.
 195 
 196   Args:
 197     hyperparams_config: hyperparams.proto object containing
 198       hyperparameters.
 199     is_training: Whether the network is in training mode.
 200 
 201   Returns:
 202     arg_scope_fn: A function to construct tf-slim arg_scope containing
 203       hyperparameters for ops.
 204 
 205   Raises:
 206     ValueError: if hyperparams_config is not of type hyperparams.Hyperparams.
 207   """
 208   if not isinstance(hyperparams_config,
 209                     hyperparams_pb2.Hyperparams):
 210     raise ValueError('hyperparams_config not of type '
 211                      'hyperparams_pb.Hyperparams.')
 212 
 213   normalizer_fn = None
 214   batch_norm_params = None
 215   if hyperparams_config.HasField('batch_norm'):
 216     normalizer_fn = slim.batch_norm
 217     batch_norm_params = _build_batch_norm_params(
 218         hyperparams_config.batch_norm, is_training)
 219   if hyperparams_config.HasField('group_norm'):
 220     normalizer_fn = slim.group_norm
 221   affected_ops = [slim.conv2d, slim.separable_conv2d, slim.conv2d_transpose]
 222   if hyperparams_config.HasField('op') and (
 223       hyperparams_config.op == hyperparams_pb2.Hyperparams.FC):
 224     affected_ops = [slim.fully_connected]
 225   def scope_fn():
 226     with (slim.arg_scope([slim.batch_norm], **batch_norm_params)
 227           if batch_norm_params is not None else
 228           context_manager.IdentityContextManager()):
 229       with slim.arg_scope(
 230           affected_ops,
 231           weights_regularizer=_build_slim_regularizer(
 232               hyperparams_config.regularizer),
 233           weights_initializer=_build_initializer(
 234               hyperparams_config.initializer),
 235           activation_fn=_build_activation_fn(hyperparams_config.activation),
 236           normalizer_fn=normalizer_fn) as sc:
 237         return sc
 238 
 239   return scope_fn

 242 def _build_activation_fn(activation_fn):
 243   """Builds a callable activation from config.
 244 
 245   Args:
 246     activation_fn: hyperparams_pb2.Hyperparams.activation
 247 
 248   Returns:
 249     Callable activation function.
 250 
 251   Raises:
 252     ValueError: On unknown activation function.
 253   """
 254   if activation_fn == hyperparams_pb2.Hyperparams.NONE:
 255     return None
 256   if activation_fn == hyperparams_pb2.Hyperparams.RELU:
 257     return tf.nn.relu
 258   if activation_fn == hyperparams_pb2.Hyperparams.RELU_6:
 259     return tf.nn.relu6
 260   if activation_fn == hyperparams_pb2.Hyperparams.SWISH:
 261     return tf.nn.swish
 262   raise ValueError('Unknown activation function: {}'.format(activation_fn))

 265 def _build_slim_regularizer(regularizer):
 266   """Builds a tf-slim regularizer from config.
 267 
 268   Args:
 269     regularizer: hyperparams_pb2.Hyperparams.regularizer proto.
 270 
 271   Returns:
 272     tf-slim regularizer.
 273 
 274   Raises:
 275     ValueError: On unknown regularizer.
 276   """
 277   regularizer_oneof = regularizer.WhichOneof('regularizer_oneof')
 278   if  regularizer_oneof == 'l1_regularizer':
 279     return slim.l1_regularizer(scale=float(regularizer.l1_regularizer.weight))
 280   if regularizer_oneof == 'l2_regularizer':
 281     return slim.l2_regularizer(scale=float(regularizer.l2_regularizer.weight))
 282   if regularizer_oneof is None:
 283     return None
 284   raise ValueError('Unknown regularizer function: {}'.format(regularizer_oneof))

 287 def _build_keras_regularizer(regularizer):
 288   """Builds a keras regularizer from config.
 289 
 290   Args:
 291     regularizer: hyperparams_pb2.Hyperparams.regularizer proto.
 292 
 293   Returns:
 294     Keras regularizer.
 295 
 296   Raises:
 297     ValueError: On unknown regularizer.
 298   """
 299   regularizer_oneof = regularizer.WhichOneof('regularizer_oneof')
 300   if  regularizer_oneof == 'l1_regularizer':
 301     return tf.keras.regularizers.l1(float(regularizer.l1_regularizer.weight))
 302   if regularizer_oneof == 'l2_regularizer':
 303     # The Keras L2 regularizer weight differs from the Slim L2 regularizer
 304     # weight by a factor of 2
 305     return tf.keras.regularizers.l2(
 306         float(regularizer.l2_regularizer.weight * 0.5))
 307   if regularizer_oneof is None:
 308     return None
 309   raise ValueError('Unknown regularizer function: {}'.format(regularizer_oneof))

 312 def _build_initializer(initializer, build_for_keras=False):
 313   """Build a tf initializer from config.
 314 
 315   Args:
 316     initializer: hyperparams_pb2.Hyperparams.regularizer proto.
 317     build_for_keras: Whether the initializers should be built for Keras
 318       operators. If false builds for Slim.
 319 
 320   Returns:
 321     tf initializer.
 322 
 323   Raises:
 324     ValueError: On unknown initializer.
 325   """
 326   initializer_oneof = initializer.WhichOneof('initializer_oneof')
 327   if initializer_oneof == 'truncated_normal_initializer':
 328     return tf.truncated_normal_initializer(
 329         mean=initializer.truncated_normal_initializer.mean,
 330         stddev=initializer.truncated_normal_initializer.stddev)
 331   if initializer_oneof == 'random_normal_initializer':
 332     return tf.random_normal_initializer(
 333         mean=initializer.random_normal_initializer.mean,
 334         stddev=initializer.random_normal_initializer.stddev)
 335   if initializer_oneof == 'variance_scaling_initializer':
 336     enum_descriptor = (hyperparams_pb2.VarianceScalingInitializer.
 337                        DESCRIPTOR.enum_types_by_name['Mode'])
 338     mode = enum_descriptor.values_by_number[initializer.
 339                                             variance_scaling_initializer.
 340                                             mode].name
 341     if build_for_keras:
 342       if initializer.variance_scaling_initializer.uniform:
 343         return tf.variance_scaling_initializer(
 344             scale=initializer.variance_scaling_initializer.factor,
 345             mode=mode.lower(),
 346             distribution='uniform')
 347       else:
 348         # In TF 1.9 release and earlier, the truncated_normal distribution was
 349         # not supported correctly. So, in these earlier versions of tensorflow,
 350         # the ValueError will be raised, and we manually truncate the
 351         # distribution scale.
 352         #
 353         # It is insufficient to just set distribution to `normal` from the
 354         # start, because the `normal` distribution in newer Tensorflow versions
 355         # creates a truncated distribution, whereas it created untruncated
 356         # distributions in older versions.
 357         try:
 358           return tf.variance_scaling_initializer(
 359               scale=initializer.variance_scaling_initializer.factor,
 360               mode=mode.lower(),
 361               distribution='truncated_normal')
 362         except ValueError:
 363           truncate_constant = 0.87962566103423978
 364           truncated_scale = initializer.variance_scaling_initializer.factor / (
 365               truncate_constant * truncate_constant
 366           )
 367           return tf.variance_scaling_initializer(
 368               scale=truncated_scale,
 369               mode=mode.lower(),
 370               distribution='normal')
 371 
 372     else:
 373       return slim.variance_scaling_initializer(
 374           factor=initializer.variance_scaling_initializer.factor,
 375           mode=mode,
 376           uniform=initializer.variance_scaling_initializer.uniform)
 377   if initializer_oneof is None:
 378     return None
 379   raise ValueError('Unknown initializer function: {}'.format(
 380       initializer_oneof))

 383 def _build_batch_norm_params(batch_norm, is_training):
 384   """Build a dictionary of batch_norm params from config.
 385 
 386   Args:
 387     batch_norm: hyperparams_pb2.ConvHyperparams.batch_norm proto.
 388     is_training: Whether the models is in training mode.
 389 
 390   Returns:
 391     A dictionary containing batch_norm parameters.
 392   """
 393   batch_norm_params = {
 394       'decay': batch_norm.decay,
 395       'center': batch_norm.center,
 396       'scale': batch_norm.scale,
 397       'epsilon': batch_norm.epsilon,
 398       # Remove is_training parameter from here and deprecate it in the proto
 399       # once we refactor Faster RCNN models to set is_training through an outer
 400       # arg_scope in the meta architecture.
 401       'is_training': is_training and batch_norm.train,
 402   }
 403   return batch_norm_params

 406 def _build_keras_batch_norm_params(batch_norm):
 407   """Build a dictionary of Keras BatchNormalization params from config.
 408 
 409   Args:
 410     batch_norm: hyperparams_pb2.ConvHyperparams.batch_norm proto.
 411 
 412   Returns:
 413     A dictionary containing Keras BatchNormalization parameters.
 414   """
 415   # Note: Although decay is defined to be 1 - momentum in batch_norm,
 416   # decay in the slim batch_norm layers was erroneously defined and is
 417   # actually the same as momentum in the Keras batch_norm layers.
 418   # For context, see: github.com/keras-team/keras/issues/6839
 419   batch_norm_params = {
 420       'momentum': batch_norm.decay,
 421       'center': batch_norm.center,
 422       'scale': batch_norm.scale,
 423       'epsilon': batch_norm.epsilon,
 424   }
 425   return batch_norm_params
