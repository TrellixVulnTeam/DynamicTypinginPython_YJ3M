  40 class LearningRateBuilderTest(tf.test.TestCase):
  41 
  42   def testBuildConstantLearningRate(self):
  43     learning_rate_text_proto = """
  44       constant_learning_rate {
  45         learning_rate: 0.004
  46       }
  47     """
  48     learning_rate_proto = optimizer_pb2.LearningRate()
  49     text_format.Merge(learning_rate_text_proto, learning_rate_proto)
  50     learning_rate = optimizer_builder._create_learning_rate(
  51         learning_rate_proto)
  52     self.assertTrue(
  53         six.ensure_str(learning_rate.op.name).endswith('learning_rate'))
  54     with self.test_session():
  55       learning_rate_out = learning_rate.eval()
  56     self.assertAlmostEqual(learning_rate_out, 0.004)
  57

 121 class OptimizerBuilderTest(tf.test.TestCase):

 123   def testBuildRMSPropOptimizer(self):
 124     optimizer_text_proto = """
 125       rms_prop_optimizer: {
 126         learning_rate: {
 127           exponential_decay_learning_rate {
 128             initial_learning_rate: 0.004
 129             decay_steps: 800720
 130             decay_factor: 0.95
 131           }
 132         }
 133         momentum_optimizer_value: 0.9
 134         decay: 0.9
 135         epsilon: 1.0
 136       }
 137       use_moving_average: false
 138     """
 139     optimizer_proto = optimizer_pb2.Optimizer()
 140     text_format.Merge(optimizer_text_proto, optimizer_proto)
 141     optimizer, _ = optimizer_builder.build(optimizer_proto)
 142     self.assertIsInstance(optimizer, tf.train.RMSPropOptimizer)
