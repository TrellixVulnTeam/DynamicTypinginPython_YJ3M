  45 def get_iterator_next_for_testing(dataset, is_tf2):
  46   iterator = dataset.make_initializable_iterator()
  47   if not is_tf2:
  48     tf.add_to_collection(tf.GraphKeys.TABLE_INITIALIZERS, iterator.initializer)
  49   return iterator.get_next()

  61   def create_tf_record(self, has_additional_channels=False, num_shards=1,
  62                        num_examples_per_shard=1):
  63 
  64     def dummy_jpeg_fn():
  65       image_tensor = np.random.randint(255, size=(4, 5, 3)).astype(np.uint8)
  66       additional_channels_tensor = np.random.randint(
  67           255, size=(4, 5, 1)).astype(np.uint8)
  68       encoded_jpeg = tf.image.encode_jpeg(image_tensor)
  69       encoded_additional_channels_jpeg = tf.image.encode_jpeg(
  70           additional_channels_tensor)
  71 
  72       return encoded_jpeg, encoded_additional_channels_jpeg
  73 
  74     encoded_jpeg, encoded_additional_channels_jpeg = self.execute(
  75         dummy_jpeg_fn, [])
  76 
  77     tmp_dir = self.get_temp_dir()
  78     flat_mask = (4 * 5) * [1.0]
  79 
  80     for i in range(num_shards):
  81       path = os.path.join(tmp_dir, '%05d.tfrecord' % i)
  82       writer = tf.python_io.TFRecordWriter(path)
  83 
  84       for j in range(num_examples_per_shard):
  85         if num_shards > 1:
  86           source_id = (str(i) + '_' + str(j)).encode()
  87         else:
  88           source_id = str(j).encode()
  89 
  90         features = {
  91             'image/source_id': dataset_util.bytes_feature(source_id),
  92             'image/encoded': dataset_util.bytes_feature(encoded_jpeg),
  93             'image/format': dataset_util.bytes_feature('jpeg'.encode('utf8')),
  94             'image/height': dataset_util.int64_feature(4),
  95             'image/width': dataset_util.int64_feature(5),
  96             'image/object/bbox/xmin': dataset_util.float_list_feature([0.0]),
  97             'image/object/bbox/xmax': dataset_util.float_list_feature([1.0]),
  98             'image/object/bbox/ymin': dataset_util.float_list_feature([0.0]),
  99             'image/object/bbox/ymax': dataset_util.float_list_feature([1.0]),
 100             'image/object/class/label': dataset_util.int64_list_feature([2]),
 101             'image/object/mask': dataset_util.float_list_feature(flat_mask),
 102         }
 103 
 104         if has_additional_channels:
 105           additional_channels_key = 'image/additional_channels/encoded'
 106           features[additional_channels_key] = dataset_util.bytes_list_feature(
 107               [encoded_additional_channels_jpeg] * 2)
 108 
 109         example = tf.train.Example(features=tf.train.Features(feature=features))
 110         writer.write(example.SerializeToString())
 111 
 112       writer.close()
 113 
 114     return os.path.join(self.get_temp_dir(), '?????.tfrecord')

 116   def _make_random_serialized_jpeg_images(self, num_frames, image_height,
 117                                           image_width):
 118     def graph_fn():
 119       images = tf.cast(tf.random.uniform(
 120           [num_frames, image_height, image_width, 3],
 121           maxval=256,
 122           dtype=tf.int32), dtype=tf.uint8)
 123       images_list = tf.unstack(images, axis=0)
 124       encoded_images_list = [tf.io.encode_jpeg(image) for image in images_list]
 125       return encoded_images_list
 126 
 127     encoded_images = self.execute(graph_fn, [])
 128     return encoded_images

 520   def _get_dataset_next(self, files, config, batch_size, num_batches_skip=0):
 521 
 522     def decode_func(value):
 523       return [tf.string_to_number(value, out_type=tf.int32)]
 524 
 525     dataset = dataset_builder.read_dataset(tf.data.TextLineDataset, files,
 526                                            config)
 527     dataset = dataset.map(decode_func)
 528     dataset = dataset.batch(batch_size)
 529 
 530     if num_batches_skip > 0:
 531       dataset = dataset.skip(num_batches_skip)
 532 
 533     return get_iterator_next_for_testing(dataset, self.is_tf2())

 535   def test_make_initializable_iterator_with_hashTable(self):
 536 
 537     def graph_fn():
 538       keys = [1, 0, -1]
 539       dataset = tf.data.Dataset.from_tensor_slices([[1, 2, -1, 5]])
 540       table = contrib_lookup.HashTable(
 541           initializer=contrib_lookup.KeyValueTensorInitializer(
 542               keys=keys, values=list(reversed(keys))),
 543           default_value=100)
 544       dataset = dataset.map(table.lookup)
 545       return dataset_builder.make_initializable_iterator(dataset).get_next()
 546 
 547     result = self.execute(graph_fn, [])
 548     self.assertAllEqual(result, [-1, 100, 1, 100])
