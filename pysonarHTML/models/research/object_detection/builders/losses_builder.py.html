  25 def build(loss_config):
  26   """Build losses based on the config.
  27 
  28   Builds classification, localization losses and optionally a hard example miner
  29   based on the config.
  30 
  31   Args:
  32     loss_config: A losses_pb2.Loss object.
  33 
  34   Returns:
  35     classification_loss: Classification loss object.
  36     localization_loss: Localization loss object.
  37     classification_weight: Classification loss weight.
  38     localization_weight: Localization loss weight.
  39     hard_example_miner: Hard example miner object.
  40     random_example_sampler: BalancedPositiveNegativeSampler object.
  41 
  42   Raises:
  43     ValueError: If hard_example_miner is used with sigmoid_focal_loss.
  44     ValueError: If random_example_sampler is getting non-positive value as
  45       desired positive example fraction.
  46   """
  47   classification_loss = _build_classification_loss(
  48       loss_config.classification_loss)
  49   localization_loss = _build_localization_loss(
  50       loss_config.localization_loss)
  51   classification_weight = loss_config.classification_weight
  52   localization_weight = loss_config.localization_weight
  53   hard_example_miner = None
  54   if loss_config.HasField('hard_example_miner'):
  55     if (loss_config.classification_loss.WhichOneof('classification_loss') ==
  56         'weighted_sigmoid_focal'):
  57       raise ValueError('HardExampleMiner should not be used with sigmoid focal '
  58                        'loss')
  59     hard_example_miner = build_hard_example_miner(
  60         loss_config.hard_example_miner,
  61         classification_weight,
  62         localization_weight)
  63   random_example_sampler = None
  64   if loss_config.HasField('random_example_sampler'):
  65     if loss_config.random_example_sampler.positive_sample_fraction <= 0:
  66       raise ValueError('RandomExampleSampler should not use non-positive'
  67                        'value as positive sample fraction.')
  68     random_example_sampler = sampler.BalancedPositiveNegativeSampler(
  69         positive_fraction=loss_config.random_example_sampler.
  70         positive_sample_fraction)
  71 
  72   if loss_config.expected_loss_weights == loss_config.NONE:
  73     expected_loss_weights_fn = None
  74   elif loss_config.expected_loss_weights == loss_config.EXPECTED_SAMPLING:
  75     expected_loss_weights_fn = functools.partial(
  76         ops.expected_classification_loss_by_expected_sampling,
  77         min_num_negative_samples=loss_config.min_num_negative_samples,
  78         desired_negative_sampling_ratio=loss_config
  79         .desired_negative_sampling_ratio)
  80   elif (loss_config.expected_loss_weights == loss_config
  81         .REWEIGHTING_UNMATCHED_ANCHORS):
  82     expected_loss_weights_fn = functools.partial(
  83         ops.expected_classification_loss_by_reweighting_unmatched_anchors,
  84         min_num_negative_samples=loss_config.min_num_negative_samples,
  85         desired_negative_sampling_ratio=loss_config
  86         .desired_negative_sampling_ratio)
  87   else:
  88     raise ValueError('Not a valid value for expected_classification_loss.')
  89 
  90   return (classification_loss, localization_loss, classification_weight,
  91           localization_weight, hard_example_miner, random_example_sampler,
  92           expected_loss_weights_fn)

  95 def build_hard_example_miner(config,
  96                              classification_weight,
  97                              localization_weight):
  98   """Builds hard example miner based on the config.
  99 
 100   Args:
 101     config: A losses_pb2.HardExampleMiner object.
 102     classification_weight: Classification loss weight.
 103     localization_weight: Localization loss weight.
 104 
 105   Returns:
 106     Hard example miner.
 107 
 108   """
 109   loss_type = None
 110   if config.loss_type == losses_pb2.HardExampleMiner.BOTH:
 111     loss_type = 'both'
 112   if config.loss_type == losses_pb2.HardExampleMiner.CLASSIFICATION:
 113     loss_type = 'cls'
 114   if config.loss_type == losses_pb2.HardExampleMiner.LOCALIZATION:
 115     loss_type = 'loc'
 116 
 117   max_negatives_per_positive = None
 118   num_hard_examples = None
 119   if config.max_negatives_per_positive > 0:
 120     max_negatives_per_positive = config.max_negatives_per_positive
 121   if config.num_hard_examples > 0:
 122     num_hard_examples = config.num_hard_examples
 123   hard_example_miner = losses.HardExampleMiner(
 124       num_hard_examples=num_hard_examples,
 125       iou_threshold=config.iou_threshold,
 126       loss_type=loss_type,
 127       cls_loss_weight=classification_weight,
 128       loc_loss_weight=localization_weight,
 129       max_negatives_per_positive=max_negatives_per_positive,
 130       min_negatives_per_image=config.min_negatives_per_image)
 131   return hard_example_miner

 134 def build_faster_rcnn_classification_loss(loss_config):
 135   """Builds a classification loss for Faster RCNN based on the loss config.
 136 
 137   Args:
 138     loss_config: A losses_pb2.ClassificationLoss object.
 139 
 140   Returns:
 141     Loss based on the config.
 142 
 143   Raises:
 144     ValueError: On invalid loss_config.
 145   """
 146   if not isinstance(loss_config, losses_pb2.ClassificationLoss):
 147     raise ValueError('loss_config not of type losses_pb2.ClassificationLoss.')
 148 
 149   loss_type = loss_config.WhichOneof('classification_loss')
 150 
 151   if loss_type == 'weighted_sigmoid':
 152     return losses.WeightedSigmoidClassificationLoss()
 153   if loss_type == 'weighted_softmax':
 154     config = loss_config.weighted_softmax
 155     return losses.WeightedSoftmaxClassificationLoss(
 156         logit_scale=config.logit_scale)
 157   if loss_type == 'weighted_logits_softmax':
 158     config = loss_config.weighted_logits_softmax
 159     return losses.WeightedSoftmaxClassificationAgainstLogitsLoss(
 160         logit_scale=config.logit_scale)
 161   if loss_type == 'weighted_sigmoid_focal':
 162     config = loss_config.weighted_sigmoid_focal
 163     alpha = None
 164     if config.HasField('alpha'):
 165       alpha = config.alpha
 166     return losses.SigmoidFocalClassificationLoss(
 167         gamma=config.gamma,
 168         alpha=alpha)
 169 
 170   # By default, Faster RCNN second stage classifier uses Softmax loss
 171   # with anchor-wise outputs.
 172   config = loss_config.weighted_softmax
 173   return losses.WeightedSoftmaxClassificationLoss(
 174       logit_scale=config.logit_scale)

 177 def _build_localization_loss(loss_config):
 178   """Builds a localization loss based on the loss config.
 179 
 180   Args:
 181     loss_config: A losses_pb2.LocalizationLoss object.
 182 
 183   Returns:
 184     Loss based on the config.
 185 
 186   Raises:
 187     ValueError: On invalid loss_config.
 188   """
 189   if not isinstance(loss_config, losses_pb2.LocalizationLoss):
 190     raise ValueError('loss_config not of type losses_pb2.LocalizationLoss.')
 191 
 192   loss_type = loss_config.WhichOneof('localization_loss')
 193 
 194   if loss_type == 'weighted_l2':
 195     return losses.WeightedL2LocalizationLoss()
 196 
 197   if loss_type == 'weighted_smooth_l1':
 198     return losses.WeightedSmoothL1LocalizationLoss(
 199         loss_config.weighted_smooth_l1.delta)
 200 
 201   if loss_type == 'weighted_iou':
 202     return losses.WeightedIOULocalizationLoss()
 203 
 204   if loss_type == 'l1_localization_loss':
 205     return losses.L1LocalizationLoss()
 206 
 207   raise ValueError('Empty loss config.')

 210 def _build_classification_loss(loss_config):
 211   """Builds a classification loss based on the loss config.
 212 
 213   Args:
 214     loss_config: A losses_pb2.ClassificationLoss object.
 215 
 216   Returns:
 217     Loss based on the config.
 218 
 219   Raises:
 220     ValueError: On invalid loss_config.
 221   """
 222   if not isinstance(loss_config, losses_pb2.ClassificationLoss):
 223     raise ValueError('loss_config not of type losses_pb2.ClassificationLoss.')
 224 
 225   loss_type = loss_config.WhichOneof('classification_loss')
 226 
 227   if loss_type == 'weighted_sigmoid':
 228     return losses.WeightedSigmoidClassificationLoss()
 229 
 230   if loss_type == 'weighted_sigmoid_focal':
 231     config = loss_config.weighted_sigmoid_focal
 232     alpha = None
 233     if config.HasField('alpha'):
 234       alpha = config.alpha
 235     return losses.SigmoidFocalClassificationLoss(
 236         gamma=config.gamma,
 237         alpha=alpha)
 238 
 239   if loss_type == 'weighted_softmax':
 240     config = loss_config.weighted_softmax
 241     return losses.WeightedSoftmaxClassificationLoss(
 242         logit_scale=config.logit_scale)
 243 
 244   if loss_type == 'weighted_logits_softmax':
 245     config = loss_config.weighted_logits_softmax
 246     return losses.WeightedSoftmaxClassificationAgainstLogitsLoss(
 247         logit_scale=config.logit_scale)
 248 
 249   if loss_type == 'bootstrapped_sigmoid':
 250     config = loss_config.bootstrapped_sigmoid
 251     return losses.BootstrappedSigmoidClassificationLoss(
 252         alpha=config.alpha,
 253         bootstrap_type=('hard' if config.hard_bootstrap else 'soft'))
 254 
 255   if loss_type == 'penalty_reduced_logistic_focal_loss':
 256     config = loss_config.penalty_reduced_logistic_focal_loss
 257     return losses.PenaltyReducedLogisticFocalLoss(
 258         alpha=config.alpha, beta=config.beta)
 259 
 260   raise ValueError('Empty loss config.')
