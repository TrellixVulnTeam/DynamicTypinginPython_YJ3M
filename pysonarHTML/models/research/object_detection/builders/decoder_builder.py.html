  16 """DataDecoder builder.
  17 
  18 Creates DataDecoders from InputReader configs.
  19 """
  20 
  21 from __future__ import absolute_import
  22 from __future__ import division
  23 from __future__ import print_function
  24 
  25 from object_detection.data_decoders import tf_example_decoder
  26 from object_detection.data_decoders import tf_sequence_example_decoder
  27 from object_detection.protos import input_reader_pb2
  28 
  29 
  30 def build(input_reader_config):
  31   """Builds a DataDecoder based only on the open source config proto.
  32 
  33   Args:
  34     input_reader_config: An input_reader_pb2.InputReader object.
  35 
  36   Returns:
  37     A DataDecoder based on the input_reader_config.
  38 
  39   Raises:
  40     ValueError: On invalid input reader proto.
  41   """
  42   if not isinstance(input_reader_config, input_reader_pb2.InputReader):
  43     raise ValueError('input_reader_config not of type '
  44                      'input_reader_pb2.InputReader.')
  45 
  46   if input_reader_config.WhichOneof('input_reader') == 'tf_record_input_reader':
  47     label_map_proto_file = None
  48     if input_reader_config.HasField('label_map_path'):
  49       label_map_proto_file = input_reader_config.label_map_path
  50     input_type = input_reader_config.input_type
  51     if input_type == input_reader_pb2.InputType.TF_EXAMPLE:
  52       decoder = tf_example_decoder.TfExampleDecoder(
  53           load_instance_masks=input_reader_config.load_instance_masks,
  54           load_multiclass_scores=input_reader_config.load_multiclass_scores,
  55           load_context_features=input_reader_config.load_context_features,
  56           instance_mask_type=input_reader_config.mask_type,
  57           label_map_proto_file=label_map_proto_file,
  58           use_display_name=input_reader_config.use_display_name,
  59           num_additional_channels=input_reader_config.num_additional_channels,
  60           num_keypoints=input_reader_config.num_keypoints,
  61           expand_hierarchy_labels=input_reader_config.expand_labels_hierarchy)
  62       return decoder
  63     elif input_type == input_reader_pb2.InputType.TF_SEQUENCE_EXAMPLE:
  64       decoder = tf_sequence_example_decoder.TfSequenceExampleDecoder(
  65           label_map_proto_file=label_map_proto_file,
  66           load_context_features=input_reader_config.load_context_features)
  67       return decoder
  68     raise ValueError('Unsupported input_type in config.')
  69 
  70   raise ValueError('Unsupported input_reader_config.')
