  20 from google.protobuf import text_format
  21 from object_detection.box_coders import faster_rcnn_box_coder

  29 class BoxCoderBuilderTest(tf.test.TestCase):
  30 
  31   def test_build_faster_rcnn_box_coder_with_defaults(self):
  32     box_coder_text_proto = """
  33       faster_rcnn_box_coder {
  34       }
  35     """
  36     box_coder_proto = box_coder_pb2.BoxCoder()
  37     text_format.Merge(box_coder_text_proto, box_coder_proto)
  38     box_coder_object = box_coder_builder.build(box_coder_proto)
  39     self.assertIsInstance(box_coder_object,
  40                           faster_rcnn_box_coder.FasterRcnnBoxCoder)
  41     self.assertEqual(box_coder_object._scale_factors, [10.0, 10.0, 5.0, 5.0])
