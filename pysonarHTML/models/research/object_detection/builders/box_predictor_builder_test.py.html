  43         }
  44       }

  54         hyperparams_proto)

 460   def test_build_box_predictor_with_mask_branch(self):
 461     box_predictor_proto = box_predictor_pb2.BoxPredictor()
 462     box_predictor_proto.mask_rcnn_box_predictor.fc_hyperparams.op = (
 463         hyperparams_pb2.Hyperparams.FC)
 464     box_predictor_proto.mask_rcnn_box_predictor.conv_hyperparams.op = (
 465         hyperparams_pb2.Hyperparams.CONV)
 466     box_predictor_proto.mask_rcnn_box_predictor.predict_instance_masks = True
 467     box_predictor_proto.mask_rcnn_box_predictor.mask_prediction_conv_depth = 512
 468     box_predictor_proto.mask_rcnn_box_predictor.mask_height = 16
 469     box_predictor_proto.mask_rcnn_box_predictor.mask_width = 16
 470     mock_argscope_fn = mock.Mock(return_value='arg_scope')
 471     box_predictor = box_predictor_builder.build(

 473         box_predictor_config=box_predictor_proto,
 474         is_training=True,
 475         num_classes=90)
 476     mock_argscope_fn.assert_has_calls(
 477         [mock.call(box_predictor_proto.mask_rcnn_box_predictor.fc_hyperparams,
 478                    True),
 479          mock.call(box_predictor_proto.mask_rcnn_box_predictor.conv_hyperparams,
 480                    True)], any_order=True)
 481     box_head = box_predictor._box_prediction_head
 482     class_head = box_predictor._class_prediction_head
 483     third_stage_heads = box_predictor._third_stage_heads
 484     self.assertFalse(box_head._use_dropout)
 485     self.assertFalse(class_head._use_dropout)

 487     self.assertAlmostEqual(class_head._dropout_keep_prob, 0.5)

 489     self.assertTrue(box_predictor._is_training)
 490     self.assertEqual(box_head._box_code_size, 4)
 491     self.assertIn(
 492         mask_rcnn_box_predictor.MASK_PREDICTIONS, third_stage_heads)
 493     self.assertEqual(
 494         third_stage_heads[mask_rcnn_box_predictor.MASK_PREDICTIONS]
 495         ._mask_prediction_conv_depth, 512)
