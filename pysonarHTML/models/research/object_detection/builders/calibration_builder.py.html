  22 def _find_interval_containing_new_value(x, new_value):
  23   """Find the index of x (ascending-ordered) after which new_value occurs."""
  24   new_value_shape = shape_utils.combined_static_and_dynamic_shape(new_value)[0]
  25   x_shape = shape_utils.combined_static_and_dynamic_shape(x)[0]
  26   compare = tf.cast(tf.reshape(new_value, shape=(new_value_shape, 1)) >=
  27                     tf.reshape(x, shape=(1, x_shape)),
  28                     dtype=tf.int32)
  29   diff = compare[:, 1:] - compare[:, :-1]
  30   interval_idx = tf.argmin(diff, axis=1)
  31   return interval_idx

  34 def _tf_linear_interp1d(x_to_interpolate, fn_x, fn_y):
  35   """Tensorflow implementation of 1d linear interpolation.
  36 
  37   Args:
  38     x_to_interpolate: tf.float32 Tensor of shape (num_examples,) over which 1d
  39       linear interpolation is performed.
  40     fn_x: Monotonically-increasing, non-repeating tf.float32 Tensor of shape
  41       (length,) used as the domain to approximate a function.
  42     fn_y: tf.float32 Tensor of shape (length,) used as the range to approximate
  43       a function.
  44 
  45   Returns:
  46     tf.float32 Tensor of shape (num_examples,)
  47   """
  48   x_pad = tf.concat([fn_x[:1] - 1, fn_x, fn_x[-1:] + 1], axis=0)
  49   y_pad = tf.concat([fn_y[:1], fn_y, fn_y[-1:]], axis=0)
  50   interval_idx = _find_interval_containing_new_value(x_pad, x_to_interpolate)
  51 
  52   # Interpolate
  53   alpha = (
  54       (x_to_interpolate - tf.gather(x_pad, interval_idx)) /
  55       (tf.gather(x_pad, interval_idx + 1) - tf.gather(x_pad, interval_idx)))
  56   interpolation = ((1 - alpha) * tf.gather(y_pad, interval_idx) +
  57                    alpha * tf.gather(y_pad, interval_idx + 1))
  58 
  59   return interpolation

  62 def _function_approximation_proto_to_tf_tensors(x_y_pairs_message):
  63   """Extracts (x,y) pairs from a XYPairs message.
  64 
  65   Args:
  66     x_y_pairs_message: calibration_pb2..XYPairs proto
  67   Returns:
  68     tf_x: tf.float32 tensor of shape (number_xy_pairs,) for function domain.
  69     tf_y: tf.float32 tensor of shape (number_xy_pairs,) for function range.
  70   """
  71   tf_x = tf.convert_to_tensor([x_y_pair.x
  72                                for x_y_pair
  73                                in x_y_pairs_message.x_y_pair],
  74                               dtype=tf.float32)
  75   tf_y = tf.convert_to_tensor([x_y_pair.y
  76                                for x_y_pair
  77                                in x_y_pairs_message.x_y_pair],
  78                               dtype=tf.float32)
  79   return tf_x, tf_y

  82 def _get_class_id_function_dict(calibration_config):
  83   """Create a dictionary mapping class id to function approximations.
  84 
  85   Args:
  86     calibration_config: calibration_pb2 proto containing
  87       id_function_approximations.
  88   Returns:
  89     Dictionary mapping a class id to a tuple of TF tensors to be used for
  90     function approximation.
  91   """
  92   class_id_function_dict = {}
  93   class_id_xy_pairs_map = (
  94       calibration_config.class_id_function_approximations.class_id_xy_pairs_map)
  95   for class_id in class_id_xy_pairs_map:
  96     class_id_function_dict[class_id] = (
  97         _function_approximation_proto_to_tf_tensors(
  98             class_id_xy_pairs_map[class_id]))
  99 
 100   return class_id_function_dict

 103 def build(calibration_config):
 104   """Returns a function that calibrates Tensorflow model scores.
 105 
 106   All returned functions are expected to apply positive monotonic
 107   transformations to inputs (i.e. score ordering is strictly preserved or
 108   adjacent scores are mapped to the same score, but an input of lower value
 109   should never be exceed an input of higher value after transformation).  For
 110   class-agnostic calibration, positive monotonicity should hold across all
 111   scores. In class-specific cases, positive monotonicity should hold within each
 112   class.
 113 
 114   Args:
 115     calibration_config: calibration_pb2.CalibrationConfig proto.
 116   Returns:
 117     Function that that accepts class_predictions_with_background and calibrates
 118     the output based on calibration_config's parameters.
 119   Raises:
 120     ValueError: No calibration builder defined for "Oneof" in
 121       calibration_config.
 122   """
 123 
 124   # Linear Interpolation (usually used as a result of calibration via
 125   # isotonic regression).
 126   if calibration_config.WhichOneof('calibrator') == 'function_approximation':
 127 
 128     def calibration_fn(class_predictions_with_background):
 129       """Calibrate predictions via 1-d linear interpolation.
 130 
 131       Predictions scores are linearly interpolated based on a class-agnostic
 132       function approximation. Note that the 0-indexed background class is also
 133       transformed.
 134 
 135       Args:
 136         class_predictions_with_background: tf.float32 tensor of shape
 137           [batch_size, num_anchors, num_classes + 1] containing scores on the
 138           interval [0,1]. This is usually produced by a sigmoid or softmax layer
 139           and the result of calling the `predict` method of a detection model.
 140 
 141       Returns:
 142         tf.float32 tensor of the same shape as the input with values on the
 143         interval [0, 1].
 144       """
 145       # Flattening Tensors and then reshaping at the end.
 146       flat_class_predictions_with_background = tf.reshape(
 147           class_predictions_with_background, shape=[-1])
 148       fn_x, fn_y = _function_approximation_proto_to_tf_tensors(
 149           calibration_config.function_approximation.x_y_pairs)
 150       updated_scores = _tf_linear_interp1d(
 151           flat_class_predictions_with_background, fn_x, fn_y)
 152 
 153       # Un-flatten the scores
 154       original_detections_shape = shape_utils.combined_static_and_dynamic_shape(
 155           class_predictions_with_background)
 156       calibrated_class_predictions_with_background = tf.reshape(
 157           updated_scores,
 158           shape=original_detections_shape,
 159           name='calibrate_scores')
 160       return calibrated_class_predictions_with_background
 161 
 162   elif (calibration_config.WhichOneof('calibrator') ==
 163         'class_id_function_approximations'):
 164 
 165     def calibration_fn(class_predictions_with_background):
 166       """Calibrate predictions per class via 1-d linear interpolation.
 167 
 168       Prediction scores are linearly interpolated with class-specific function
 169       approximations. Note that after calibration, an anchor's class scores will
 170       not necessarily sum to 1, and score ordering may change, depending on each
 171       class' calibration parameters.
 172 
 173       Args:
 174         class_predictions_with_background: tf.float32 tensor of shape
 175           [batch_size, num_anchors, num_classes + 1] containing scores on the
 176           interval [0,1]. This is usually produced by a sigmoid or softmax layer
 177           and the result of calling the `predict` method of a detection model.
 178 
 179       Returns:
 180         tf.float32 tensor of the same shape as the input with values on the
 181         interval [0, 1].
 182 
 183       Raises:
 184         KeyError: Calibration parameters are not present for a class.
 185       """
 186       class_id_function_dict = _get_class_id_function_dict(calibration_config)
 187 
 188       # Tensors are split by class and then recombined at the end to recover
 189       # the input's original shape. If a class id does not have calibration
 190       # parameters, it is left unchanged.
 191       class_tensors = tf.unstack(class_predictions_with_background, axis=-1)
 192       calibrated_class_tensors = []
 193       for class_id, class_tensor in enumerate(class_tensors):
 194         flat_class_tensor = tf.reshape(class_tensor, shape=[-1])
 195         if class_id in class_id_function_dict:
 196           output_tensor = _tf_linear_interp1d(
 197               x_to_interpolate=flat_class_tensor,
 198               fn_x=class_id_function_dict[class_id][0],
 199               fn_y=class_id_function_dict[class_id][1])
 200         else:
 201           tf.logging.info(
 202               'Calibration parameters for class id `%d` not not found',
 203               class_id)
 204           output_tensor = flat_class_tensor
 205         calibrated_class_tensors.append(output_tensor)
 206 
 207       combined_calibrated_tensor = tf.stack(calibrated_class_tensors, axis=1)
 208       input_shape = shape_utils.combined_static_and_dynamic_shape(
 209           class_predictions_with_background)
 210       calibrated_class_predictions_with_background = tf.reshape(
 211           combined_calibrated_tensor,
 212           shape=input_shape,
 213           name='calibrate_scores')
 214       return calibrated_class_predictions_with_background
 215 
 216   elif (calibration_config.WhichOneof('calibrator') ==
 217         'temperature_scaling_calibration'):
 218 
 219     def calibration_fn(class_predictions_with_background):
 220       """Calibrate predictions via temperature scaling.
 221 
 222       Predictions logits scores are scaled by the temperature scaler. Note that
 223       the 0-indexed background class is also transformed.
 224 
 225       Args:
 226         class_predictions_with_background: tf.float32 tensor of shape
 227           [batch_size, num_anchors, num_classes + 1] containing logits scores.
 228           This is usually produced before a sigmoid or softmax layer.
 229 
 230       Returns:
 231         tf.float32 tensor of the same shape as the input.
 232 
 233       Raises:
 234         ValueError: If temperature scaler is of incorrect value.
 235       """
 236       scaler = calibration_config.temperature_scaling_calibration.scaler
 237       if scaler <= 0:
 238         raise ValueError('The scaler in temperature scaling must be positive.')
 239       calibrated_class_predictions_with_background = tf.math.divide(
 240           class_predictions_with_background,
 241           scaler,
 242           name='calibrate_score')
 243       return calibrated_class_predictions_with_background
 244 
 245   # TODO(zbeaver): Add sigmoid calibration.
 246   else:
 247     raise ValueError('No calibration builder defined for "Oneof" in '
 248                      'calibration_config.')
 249 
 250   return calibration_fn
