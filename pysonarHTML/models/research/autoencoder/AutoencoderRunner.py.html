   5 import numpy as np

   7 import tensorflow as tf

  12 mnist = input_data.read_data_sets('MNIST_data', one_hot=True)

  15 def standard_scale(X_train, X_test):
  16     preprocessor = prep.StandardScaler().fit(X_train)
  17     X_train = preprocessor.transform(X_train)
  18     X_test = preprocessor.transform(X_test)
  19     return X_train, X_test
  20 
  21 
  22 def get_random_block_from_data(data, batch_size):
  23     start_index = np.random.randint(0, len(data) - batch_size)
  24     return data[start_index:(start_index + batch_size)]
  25 
  26 
  27 X_train, X_test = standard_scale(mnist.train.images, mnist.test.images)
  28 
  29 n_samples = int(mnist.train.num_examples)
  30 training_epochs = 20
  31 batch_size = 128
  32 display_step = 1
  33 
  34 autoencoder = Autoencoder(n_layers=[784, 200],
  35                           transfer_function = tf.nn.softplus,
  36                           optimizer = tf.train.AdamOptimizer(learning_rate = 0.001))
  37 
  38 for epoch in range(training_epochs):
  39     avg_cost = 0.
  40     total_batch = int(n_samples / batch_size)
  41     # Loop over all batches
  42     for i in range(total_batch):
  43         batch_xs = get_random_block_from_data(X_train, batch_size)
  44 
  45         # Fit training using batch data
  46         cost = autoencoder.partial_fit(batch_xs)
  47         # Compute average loss
  48         avg_cost += cost / n_samples * batch_size
  49 
  50     # Display logs per epoch step
  51     if epoch % display_step == 0:
  52         print("Epoch:", '%d,' % (epoch + 1),
  53               "Cost:", "{:.9f}".format(avg_cost))
  54
