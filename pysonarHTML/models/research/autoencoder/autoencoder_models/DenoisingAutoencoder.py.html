   3 class AdditiveGaussianNoiseAutoencoder(object):
   4     def __init__(self, n_input, n_hidden, transfer_function = tf.nn.softplus, optimizer = tf.train.AdamOptimizer(),
   5                  scale = 0.1):
   6         self.n_input = n_input
   7         self.n_hidden = n_hidden
   8         self.transfer = transfer_function
   9         self.scale = tf.placeholder(tf.float32)
  10         self.training_scale = scale
  11         network_weights = self._initialize_weights()
  12         self.weights = network_weights
  13 
  14         # model
  15         self.x = tf.placeholder(tf.float32, [None, self.n_input])
  16         self.hidden = self.transfer(tf.add(tf.matmul(self.x + scale * tf.random_normal((n_input,)),
  17                 self.weights['w1']),
  18                 self.weights['b1']))
  19         self.reconstruction = tf.add(tf.matmul(self.hidden, self.weights['w2']), self.weights['b2'])
  20 
  21         # cost
  22         self.cost = 0.5 * tf.reduce_sum(tf.pow(tf.subtract(self.reconstruction, self.x), 2.0))
  23         self.optimizer = optimizer.minimize(self.cost)
  24 
  25         init = tf.global_variables_initializer()
  26         self.sess = tf.Session()
  27         self.sess.run(init)
  28 
  29     def _initialize_weights(self):
  30         all_weights = dict()
  31         all_weights['w1'] = tf.get_variable("w1", shape=[self.n_input, self.n_hidden],
  32             initializer=tf.contrib.layers.xavier_initializer())
  33         all_weights['b1'] = tf.Variable(tf.zeros([self.n_hidden], dtype = tf.float32))
  34         all_weights['w2'] = tf.Variable(tf.zeros([self.n_hidden, self.n_input], dtype = tf.float32))
  35         all_weights['b2'] = tf.Variable(tf.zeros([self.n_input], dtype = tf.float32))
  36         return all_weights
  37

  39         cost, opt = self.sess.run((self.cost, self.optimizer), feed_dict = {self.x: X,

  41                                                                             })
