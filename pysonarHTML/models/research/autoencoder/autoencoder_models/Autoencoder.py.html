   7     def __init__(self, n_layers, transfer_function=tf.nn.softplus, optimizer=tf.train.AdamOptimizer()):
   8         self.n_layers = n_layers
   9         self.transfer = transfer_function
  10 
  11         network_weights = self._initialize_weights()
  12         self.weights = network_weights
  13 
  14         # model
  15         self.x = tf.placeholder(tf.float32, [None, self.n_layers[0]])
  16         self.hidden_encode = []
  17         h = self.x
  18         for layer in range(len(self.n_layers)-1):
  19             h = self.transfer(
  20                 tf.add(tf.matmul(h, self.weights['encode'][layer]['w']),
  21                        self.weights['encode'][layer]['b']))
  22             self.hidden_encode.append(h)
  23 
  24         self.hidden_recon = []
  25         for layer in range(len(self.n_layers)-1):
  26             h = self.transfer(
  27                 tf.add(tf.matmul(h, self.weights['recon'][layer]['w']),
  28                        self.weights['recon'][layer]['b']))
  29             self.hidden_recon.append(h)
  30         self.reconstruction = self.hidden_recon[-1]
  31 
  32         # cost
  33         self.cost = 0.5 * tf.reduce_sum(tf.pow(tf.subtract(self.reconstruction, self.x), 2.0))
  34         self.optimizer = optimizer.minimize(self.cost)
  35 
  36         init = tf.global_variables_initializer()
  37         self.sess = tf.Session()
  38         self.sess.run(init)

  41     def _initialize_weights(self):
  42         all_weights = dict()
  43         initializer = tf.contrib.layers.xavier_initializer()
  44         # Encoding network weights
  45         encoder_weights = []
  46         for layer in range(len(self.n_layers)-1):
  47             w = tf.Variable(
  48                 initializer((self.n_layers[layer], self.n_layers[layer + 1]),
  49                             dtype=tf.float32))
  50             b = tf.Variable(
  51                 tf.zeros([self.n_layers[layer + 1]], dtype=tf.float32))
  52             encoder_weights.append({'w': w, 'b': b})
  53         # Recon network weights
  54         recon_weights = []
  55         for layer in range(len(self.n_layers)-1, 0, -1):
  56             w = tf.Variable(
  57                 initializer((self.n_layers[layer], self.n_layers[layer - 1]),
  58                             dtype=tf.float32))
  59             b = tf.Variable(
  60                 tf.zeros([self.n_layers[layer - 1]], dtype=tf.float32))
  61             recon_weights.append({'w': w, 'b': b})
  62         all_weights['encode'] = encoder_weights
  63         all_weights['recon'] = recon_weights
  64         return all_weights
