  35 def _FSpecialGauss(size, sigma):
  36   """Function to mimic the 'fspecial' gaussian MATLAB function."""
  37   radius = size // 2
  38   offset = 0.0
  39   start, stop = -radius, radius + 1
  40   if size % 2 == 0:
  41     offset = 0.5
  42     stop -= 1
  43   x, y = np.mgrid[offset + start:stop, offset + start:stop]
  44   assert len(x) == size
  45   g = np.exp(-((x**2 + y**2)/(2.0 * sigma**2)))
  46   return g / g.sum()

  49 def _SSIMForMultiScale(img1, img2, max_val=255, filter_size=11,
  50                        filter_sigma=1.5, k1=0.01, k2=0.03):
  51   """Return the Structural Similarity Map between `img1` and `img2`.
  52 
  53   This function attempts to match the functionality of ssim_index_new.m by
  54   Zhou Wang: http://www.cns.nyu.edu/~lcv/ssim/msssim.zip
  55 
  56   Arguments:
  57     img1: Numpy array holding the first RGB image batch.
  58     img2: Numpy array holding the second RGB image batch.
  59     max_val: the dynamic range of the images (i.e., the difference between the
  60       maximum the and minimum allowed values).
  61     filter_size: Size of blur kernel to use (will be reduced for small images).
  62     filter_sigma: Standard deviation for Gaussian blur kernel (will be reduced
  63       for small images).
  64     k1: Constant used to maintain stability in the SSIM calculation (0.01 in
  65       the original paper).
  66     k2: Constant used to maintain stability in the SSIM calculation (0.03 in
  67       the original paper).
  68 
  69   Returns:
  70     Pair containing the mean SSIM and contrast sensitivity between `img1` and
  71     `img2`.
  72 
  73   Raises:
  74     RuntimeError: If input images don't have the same shape or don't have four
  75       dimensions: [batch_size, height, width, depth].
  76   """
  77   if img1.shape != img2.shape:
  78     raise RuntimeError('Input images must have the same shape (%s vs. %s).',
  79                        img1.shape, img2.shape)
  80   if img1.ndim != 4:
  81     raise RuntimeError('Input images must have four dimensions, not %d',
  82                        img1.ndim)
  83 
  84   img1 = img1.astype(np.float64)
  85   img2 = img2.astype(np.float64)
  86   _, height, width, _ = img1.shape
  87 
  88   # Filter size can't be larger than height or width of images.
  89   size = min(filter_size, height, width)
  90 
  91   # Scale down sigma if a smaller filter size is used.
  92   sigma = size * filter_sigma / filter_size if filter_size else 0
  93 
  94   if filter_size:
  95     window = np.reshape(_FSpecialGauss(size, sigma), (1, size, size, 1))
  96     mu1 = signal.fftconvolve(img1, window, mode='valid')
  97     mu2 = signal.fftconvolve(img2, window, mode='valid')
  98     sigma11 = signal.fftconvolve(img1 * img1, window, mode='valid')
  99     sigma22 = signal.fftconvolve(img2 * img2, window, mode='valid')
 100     sigma12 = signal.fftconvolve(img1 * img2, window, mode='valid')
 101   else:
 102     # Empty blur kernel so no need to convolve.
 103     mu1, mu2 = img1, img2
 104     sigma11 = img1 * img1
 105     sigma22 = img2 * img2
 106     sigma12 = img1 * img2
 107 
 108   mu11 = mu1 * mu1
 109   mu22 = mu2 * mu2
 110   mu12 = mu1 * mu2
 111   sigma11 -= mu11
 112   sigma22 -= mu22
 113   sigma12 -= mu12
 114 
 115   # Calculate intermediate values used by both ssim and cs_map.
 116   c1 = (k1 * max_val) ** 2
 117   c2 = (k2 * max_val) ** 2
 118   v1 = 2.0 * sigma12 + c2
 119   v2 = sigma11 + sigma22 + c2
 120   ssim = np.mean((((2.0 * mu12 + c1) * v1) / ((mu11 + mu22 + c1) * v2)))
 121   cs = np.mean(v1 / v2)
 122   return ssim, cs

 125 def MultiScaleSSIM(img1, img2, max_val=255, filter_size=11, filter_sigma=1.5,
 126                    k1=0.01, k2=0.03, weights=None):
 127   """Return the MS-SSIM score between `img1` and `img2`.
 128 
 129   This function implements Multi-Scale Structural Similarity (MS-SSIM) Image
 130   Quality Assessment according to Zhou Wang's paper, "Multi-scale structural
 131   similarity for image quality assessment" (2003).
 132   Link: https://ece.uwaterloo.ca/~z70wang/publications/msssim.pdf
 133 
 134   Author's MATLAB implementation:
 135   http://www.cns.nyu.edu/~lcv/ssim/msssim.zip
 136 
 137   Arguments:
 138     img1: Numpy array holding the first RGB image batch.
 139     img2: Numpy array holding the second RGB image batch.
 140     max_val: the dynamic range of the images (i.e., the difference between the
 141       maximum the and minimum allowed values).
 142     filter_size: Size of blur kernel to use (will be reduced for small images).
 143     filter_sigma: Standard deviation for Gaussian blur kernel (will be reduced
 144       for small images).
 145     k1: Constant used to maintain stability in the SSIM calculation (0.01 in
 146       the original paper).
 147     k2: Constant used to maintain stability in the SSIM calculation (0.03 in
 148       the original paper).
 149     weights: List of weights for each level; if none, use five levels and the
 150       weights from the original paper.
 151 
 152   Returns:
 153     MS-SSIM score between `img1` and `img2`.
 154 
 155   Raises:
 156     RuntimeError: If input images don't have the same shape or don't have four
 157       dimensions: [batch_size, height, width, depth].
 158   """
 159   if img1.shape != img2.shape:
 160     raise RuntimeError('Input images must have the same shape (%s vs. %s).',
 161                        img1.shape, img2.shape)
 162   if img1.ndim != 4:
 163     raise RuntimeError('Input images must have four dimensions, not %d',
 164                        img1.ndim)
 165 
 166   # Note: default weights don't sum to 1.0 but do match the paper / matlab code.
 167   weights = np.array(weights if weights else
 168                      [0.0448, 0.2856, 0.3001, 0.2363, 0.1333])
 169   levels = weights.size
 170   downsample_filter = np.ones((1, 2, 2, 1)) / 4.0
 171   im1, im2 = [x.astype(np.float64) for x in [img1, img2]]
 172   mssim = np.array([])
 173   mcs = np.array([])
 174   for _ in range(levels):
 175     ssim, cs = _SSIMForMultiScale(
 176         im1, im2, max_val=max_val, filter_size=filter_size,
 177         filter_sigma=filter_sigma, k1=k1, k2=k2)
 178     mssim = np.append(mssim, ssim)
 179     mcs = np.append(mcs, cs)
 180     filtered = [convolve(im, downsample_filter, mode='reflect')
 181                 for im in [im1, im2]]
 182     im1, im2 = [x[:, ::2, ::2, :] for x in filtered]
 183   return (np.prod(mcs[0:levels-1] ** weights[0:levels-1]) *
 184           (mssim[levels-1] ** weights[levels-1]))

 187 def main(_):
 188   if FLAGS.original_image is None or FLAGS.compared_image is None:
 189     print('\nUsage: python msssim.py --original_image=original.png '
 190           '--compared_image=distorted.png\n\n')
 191     return
 192 
 193   if not tf.gfile.Exists(FLAGS.original_image):
 194     print('\nCannot find --original_image.\n')
 195     return
 196 
 197   if not tf.gfile.Exists(FLAGS.compared_image):
 198     print('\nCannot find --compared_image.\n')
 199     return
 200 
 201   with tf.gfile.FastGFile(FLAGS.original_image) as image_file:
 202     img1_str = image_file.read('rb')
 203   with tf.gfile.FastGFile(FLAGS.compared_image) as image_file:
 204     img2_str = image_file.read('rb')
 205 
 206   input_img = tf.placeholder(tf.string)
 207   decoded_image = tf.expand_dims(tf.image.decode_png(input_img, channels=3), 0)
 208 
 209   with tf.Session() as sess:
 210     img1 = sess.run(decoded_image, feed_dict={input_img: img1_str})
 211     img2 = sess.run(decoded_image, feed_dict={input_img: img2_str})
 212 
 213   print((MultiScaleSSIM(img1, img2, max_val=255)))
 214 
 215 
 216 if __name__ == '__main__':
 217   tf.app.run()
