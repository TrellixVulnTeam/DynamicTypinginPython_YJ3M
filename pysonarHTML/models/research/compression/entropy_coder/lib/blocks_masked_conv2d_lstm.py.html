  27 class RasterScanConv2DLSTM(blocks_lstm.LSTMBase):
  28   """Convolutional LSTM implementation with optimizations inspired by [1].
  29 
  30   Note that when using the batch normalization feature, the bias initializer
  31   will not be used, since BN effectively cancels its effect out.
  32 
  33   [1] Zaremba, Sutskever, Vinyals. Recurrent Neural Network Regularization,
  34   2015. arxiv:1409.2329.
  35   """
  36 
  37   def __init__(self,
  38                depth,
  39                filter_size,
  40                hidden_filter_size,
  41                strides,
  42                padding,
  43                bias=blocks_lstm.LSTMBiasInit,
  44                initializer=block_util.RsqrtInitializer(dims=(0, 1, 2)),
  45                name=None):
  46     super(RasterScanConv2DLSTM, self).__init__([None, None, depth], name)
  47 
  48     with self._BlockScope():
  49       self._input_conv = blocks_masked_conv2d.RasterScanConv2D(
  50           4 * depth,
  51           filter_size,
  52           strides,
  53           padding,
  54           strict_order=False,
  55           bias=None,
  56           act=None,
  57           initializer=initializer,
  58           name='input_conv2d')
  59 
  60       self._hidden_conv = blocks_std.Conv2D(
  61           4 * depth,
  62           hidden_filter_size,
  63           [1, 1],
  64           'SAME',
  65           bias=None,
  66           act=None,
  67           initializer=initializer,
  68           name='hidden_conv2d')
  69 
  70       if bias is not None:
  71         self._bias = blocks_std.BiasAdd(bias, name='biases')
  72       else:
  73         self._bias = blocks_std.PassThrough()
  74 
  75   def _TransformInputs(self, x):
  76     return self._bias(self._input_conv(x))
  77 
  78   def _TransformHidden(self, h):
  79     return self._hidden_conv(h)
