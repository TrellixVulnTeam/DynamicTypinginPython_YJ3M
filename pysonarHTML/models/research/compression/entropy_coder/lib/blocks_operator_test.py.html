  16 """Tests of the block operators."""
  17 
  18 import numpy as np
  19 import tensorflow as tf
  20 
  21 import block_base
  22 import blocks_operator
  23 
  24 
  25 class AddOneBlock(block_base.BlockBase):
  26 
  27   def __init__(self, name=None):
  28     super(AddOneBlock, self).__init__(name)
  29 
  30   def _Apply(self, x):
  31     return x + 1.0
  32 
  33 
  34 class SquareBlock(block_base.BlockBase):
  35 
  36   def __init__(self, name=None):
  37     super(SquareBlock, self).__init__(name)
  38 
  39   def _Apply(self, x):
  40     return x * x
  41 
  42 
  43 class BlocksOperatorTest(tf.test.TestCase):
  44 
  45   def testComposition(self):
  46     x_value = np.array([[1.0, 2.0, 3.0],
  47                         [-1.0, -2.0, -3.0]])
  48     y_expected_value = np.array([[4.0, 9.0, 16.0],
  49                                  [0.0, 1.0, 4.0]])
  50 
  51     x = tf.placeholder(dtype=tf.float32, shape=[2, 3])
  52     complex_block = blocks_operator.CompositionOperator(
  53         [AddOneBlock(),
  54          SquareBlock()])
  55     y = complex_block(x)
  56 
  57     with self.test_session():
  58       y_value = y.eval(feed_dict={x: x_value})
  59 
  60     self.assertAllClose(y_expected_value, y_value)
  61 
  62 
  63 if __name__ == '__main__':
  64   tf.test.main()
