  25 class CompositionOperator(block_base.BlockBase):
  26   """Composition of several blocks."""
  27 
  28   def __init__(self, block_list, name=None):
  29     """Initialization of the composition operator.
  30 
  31     Args:
  32       block_list: List of blocks.BlockBase that are chained to create
  33         a new blocks.BlockBase.
  34       name: Name of this block.
  35     """
  36     super(CompositionOperator, self).__init__(name)
  37     self._blocks = block_list
  38 
  39   def _Apply(self, x):
  40     """Apply successively all the blocks on the given input tensor."""
  41     h = x
  42     for layer in self._blocks:
  43       h = layer(h)
  44     return h

  47 class LineOperator(block_base.BlockBase):
  48   """Repeat the same block over all the lines of an input tensor."""
  49 
  50   def __init__(self, block, name=None):
  51     super(LineOperator, self).__init__(name)
  52     self._block = block
  53 
  54   def _Apply(self, x):
  55     height = x.get_shape()[1].value
  56     if height is None:
  57       raise ValueError('Unknown tensor height')
  58     all_line_x = tf.split(value=x, num_or_size_splits=height, axis=1)
  59 
  60     y = []
  61     for line_x in all_line_x:
  62       y.append(self._block(line_x))
  63     y = tf.concat(values=y, axis=1)
  64 
  65     return y
  66

  68 class TowerOperator(block_base.BlockBase):
  69   """Parallel execution with concatenation of several blocks."""
  70 
  71   def __init__(self, block_list, dim=3, name=None):
  72     """Initialization of the parallel exec + concat (Tower).
  73 
  74     Args:
  75       block_list: List of blocks.BlockBase that are chained to create
  76         a new blocks.BlockBase.
  77       dim: the dimension on which to concat.
  78       name: Name of this block.
  79     """
  80     super(TowerOperator, self).__init__(name)
  81     self._blocks = block_list
  82     self._concat_dim = dim
  83 
  84   def _Apply(self, x):
  85     """Apply successively all the blocks on the given input tensor."""
  86     outputs = [layer(x) for layer in self._blocks]
  87     return tf.concat(outputs, self._concat_dim)
