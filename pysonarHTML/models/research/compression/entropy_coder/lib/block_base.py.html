  16 """Base class for Tensorflow building blocks."""
  17 
  18 import collections
  19 import contextlib
  20 import itertools
  21 
  22 import tensorflow as tf
  23 
  24 _block_stacks = collections.defaultdict(lambda: [])
  25 
  26 
  27 class BlockBase(object):
  28   """Base class for transform wrappers of Tensorflow.
  29 
  30   To implement a Tensorflow transform block, inherit this class.
  31 
  32   1. To create a variable, use NewVar() method. Do not overload this method!
  33      For example, use as follows.
  34          a_variable = self.NewVar(initial_value)
  35 
  36   2. All Tensorflow-related code must be done inside 'with self._BlockScope().'
  37      Otherwise, name scoping and block hierarchy will not work. An exception
  38      is _Apply() method, which is already called inside the context manager
  39      by __call__() method.
  40 
  41   3. Override and implement _Apply() method. This method is called by
  42      __call__() method.
  43 
  44   The users would use blocks like the following.
  45       nn1 = NN(128, bias=Bias(0), act=tf.nn.relu)
  46       y = nn1(x)
  47 
  48   Some things to consider.
  49 
  50   - Use lazy-initialization if possible. That is, initialize at first Apply()
  51     rather than at __init__().
  52 
  53   Note: if needed, the variables can be created on a specific parameter
  54   server by creating blocks in a scope like:
  55     with g.device(device):
  56       linear = Linear(...)
  57   """
  58 
  59   def __init__(self, name):
  60     self._variables = []
  61     self._subblocks = []
  62     self._called = False
  63 
  64     # Intentionally distinguishing empty string and None.
  65     # If name is an empty string, then do not use name scope.
  66     self.name = name if name is not None else self.__class__.__name__
  67     self._graph = tf.get_default_graph()
  68 
  69     if self.name:
  70       # Capture the scope string at the init time.
  71       with self._graph.name_scope(self.name) as scope:
  72         self._scope_str = scope
  73     else:
  74       self._scope_str = ''
  75 
  76     # Maintain hierarchy structure of blocks.
  77     self._stack = _block_stacks[self._graph]
  78     if self.__class__ is BlockBase:
  79       # This code is only executed to create the root, which starts in the
  80       # initialized state.
  81       assert not self._stack
  82       self._parent = None
  83       self._called = True  # The root is initialized.
  84       return
  85 
  86     # Create a fake root if a root is not already present.
  87     if not self._stack:
  88       self._stack.append(BlockBase('NoOpRoot'))
  89 
  90     self._parent = self._stack[-1]
  91     self._parent._subblocks.append(self)  # pylint: disable=protected-access
  92 
  93   def __repr__(self):
  94     return '"{}" ({})'.format(self._scope_str, self.__class__.__name__)
  95 
  96   @contextlib.contextmanager
  97   def _OptionalNameScope(self, scope_str):
  98     if scope_str:
  99       with self._graph.name_scope(scope_str):
 100         yield
 101     else:
 102       yield
 103 
 104   @contextlib.contextmanager
 105   def _BlockScope(self):
 106     """Context manager that handles graph, namescope, and nested blocks."""
 107     self._stack.append(self)
 108 
 109     try:
 110       with self._graph.as_default():
 111         with self._OptionalNameScope(self._scope_str):
 112           yield self
 113     finally:  # Pop from the stack no matter exception is raised or not.
 114       # The following line is executed when leaving 'with self._BlockScope()'
 115       self._stack.pop()
 116 
 117   def __call__(self, *args, **kwargs):
 118     assert self._stack is _block_stacks[self._graph]
 119 
 120     with self._BlockScope():
 121       ret = self._Apply(*args, **kwargs)
 122 
 123     self._called = True
 124     return ret
 125 
 126   def _Apply(self, *args, **kwargs):
 127     """Implementation of __call__()."""
 128     raise NotImplementedError()
 129 
 130   # Redirect all variable creation to this single function, so that we can
 131   # switch to better variable creation scheme.
 132   def NewVar(self, value, **kwargs):
 133     """Creates a new variable.
 134 
 135     This function creates a variable, then returns a local copy created by
 136     Identity operation. To get the Variable class object, use LookupRef()
 137     method.
 138 
 139     Note that each time Variable class object is used as an input to an
 140     operation, Tensorflow will create a new Send/Recv pair. This hurts
 141     performance.
 142 
 143     If not for assign operations, use the local copy returned by this method.
 144 
 145     Args:
 146       value: Initialization value of the variable. The shape and the data type
 147         of the variable is determined by this initial value.
 148       **kwargs: Extra named arguments passed to Variable.__init__().
 149 
 150     Returns:
 151       A local copy of the new variable.
 152     """
 153     v = tf.Variable(value, **kwargs)
 154 
 155     self._variables.append(v)
 156     return v
 157 
 158   @property
 159   def initialized(self):
 160     """Returns bool if the block is initialized.
 161 
 162     By default, BlockBase assumes that a block is initialized when __call__()
 163     is executed for the first time. If this is an incorrect assumption for some
 164     subclasses, override this property in those subclasses.
 165 
 166     Returns:
 167       True if initialized, False otherwise.
 168     """
 169     return self._called
 170 
 171   def AssertInitialized(self):
 172     """Asserts initialized property."""
 173     if not self.initialized:
 174       raise RuntimeError('{} has not been initialized.'.format(self))
 175 
 176   def VariableList(self):
 177     """Returns the list of all tensorflow variables used inside this block."""
 178     variables = list(itertools.chain(
 179         itertools.chain.from_iterable(
 180             t.VariableList() for t in self._subblocks),
 181         self._VariableList()))
 182     return variables
 183 
 184   def _VariableList(self):
 185     """Returns the list of all tensorflow variables owned by this block."""
 186     self.AssertInitialized()
 187     return self._variables
 188 
 189   def CreateWeightLoss(self):
 190     """Returns L2 loss list of (almost) all variables used inside this block.
 191 
 192     When this method needs to be overridden, there are two choices.
 193 
 194     1. Override CreateWeightLoss() to change the weight loss of all variables
 195        that belong to this block, both directly and indirectly.
 196     2. Override _CreateWeightLoss() to change the weight loss of all
 197        variables that directly belong to this block but not to the sub-blocks.
 198 
 199     Returns:
 200       A Tensor object or None.
 201     """
 202     losses = list(itertools.chain(
 203         itertools.chain.from_iterable(
 204             t.CreateWeightLoss() for t in self._subblocks),
 205         self._CreateWeightLoss()))
 206     return losses
 207 
 208   def _CreateWeightLoss(self):
 209     """Returns weight loss list of variables that belong to this block."""
 210     self.AssertInitialized()
 211     with self._BlockScope():
 212       return [tf.nn.l2_loss(v) for v in self._variables]
 213 
 214   def CreateUpdateOps(self):
 215     """Creates update operations for this block and its sub-blocks."""
 216     ops = list(itertools.chain(
 217         itertools.chain.from_iterable(
 218             t.CreateUpdateOps() for t in self._subblocks),
 219         self._CreateUpdateOps()))
 220     return ops
 221 
 222   def _CreateUpdateOps(self):
 223     """Creates update operations for this block."""
 224     self.AssertInitialized()
 225     return []
 226 
 227   def MarkAsNonTrainable(self):
 228     """Mark all the variables of this block as non-trainable.
 229 
 230     All the variables owned directly or indirectly (through subblocks) are
 231     marked as non trainable.
 232 
 233     This function along with CheckpointInitOp can be used to load a pretrained
 234     model that consists in only one part of the whole graph.
 235     """
 236     assert self._called
 237 
 238     all_variables = self.VariableList()
 239     collection = tf.get_collection_ref(tf.GraphKeys.TRAINABLE_VARIABLES)
 240     for v in all_variables:
 241       if v in collection:
 242         collection.remove(v)
 243 
 244 
 245 def CreateWeightLoss():
 246   """Returns all weight losses from the blocks in the graph."""
 247   stack = _block_stacks[tf.get_default_graph()]
 248   if not stack:
 249     return []
 250   return stack[0].CreateWeightLoss()
 251 
 252 
 253 def CreateBlockUpdates():
 254   """Combines all updates from the blocks in the graph."""
 255   stack = _block_stacks[tf.get_default_graph()]
 256   if not stack:
 257     return []
 258   return stack[0].CreateUpdateOps()
