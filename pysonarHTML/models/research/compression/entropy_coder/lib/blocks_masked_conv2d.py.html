  29 class RasterScanConv2D(blocks_std.Conv2DBase):
  30   """Conv2D with no dependency on future pixels (in raster scan order).
  31 
  32   For example, assuming a 5 x 5 kernel, the kernel is applied a spatial mask:
  33     T T T T T
  34     T T T T T
  35     T T x F F
  36     F F F F F
  37     F F F F F
  38   where 'T' are pixels which are available when computing the convolution
  39   for pixel 'x'. All the pixels marked with 'F' are not available.
  40   'x' itself is not available if strict_order is True, otherwise, it is
  41   available.
  42   """
  43 
  44   def __init__(self, depth, filter_size, strides, padding,
  45                strict_order=True,
  46                bias=None, act=None, initializer=None, name=None):
  47     super(RasterScanConv2D, self).__init__(
  48         depth, filter_size, strides, padding, bias, act, name=name)
  49 
  50     if (filter_size[0] % 2) != 1 or (filter_size[1] % 2) != 1:
  51       raise ValueError('Kernel size should be odd.')
  52 
  53     with self._BlockScope():
  54       if initializer is None:
  55         initializer = block_util.RsqrtInitializer(dims=(0, 1, 2))
  56       self._initializer = initializer
  57       self._strict_order = strict_order
  58 
  59   def _CreateKernel(self, shape, dtype):
  60     init = self._initializer(shape, dtype)
  61     kernel = self.NewVar(init)
  62 
  63     mask = np.ones(shape[:2], dtype=dtype.as_numpy_dtype)
  64     center = shape[:2] // 2
  65     mask[center[0] + 1:, :] = 0
  66     if not self._strict_order:
  67       mask[center[0], center[1] + 1:] = 0
  68     else:
  69       mask[center[0], center[1]:] = 0
  70     mask = mask.reshape(mask.shape + (1, 1))
  71 
  72     return tf.convert_to_tensor(mask, dtype) * kernel

  75 class DepthOrderConv2D(blocks_std.Conv2DBase):
  76   """Conv2D with no dependency on higher depth dimensions.
  77 
  78   More precisely, the output depth #n has only dependencies on input depths #k
  79   for k < n (if strict_order is True) or for k <= n (if strict_order is False).
  80   """
  81 
  82   def __init__(self, depth, filter_size, strides, padding,
  83                strict_order=True,
  84                bias=None, act=None, initializer=None, name=None):
  85     super(DepthOrderConv2D, self).__init__(
  86         depth, filter_size, strides, padding, bias, act, name=name)
  87 
  88     with self._BlockScope():
  89       if initializer is None:
  90         initializer = block_util.RsqrtInitializer(dims=(0, 1, 2))
  91       self._initializer = initializer
  92       self._strict_order = strict_order
  93 
  94   def _CreateKernel(self, shape, dtype):
  95     init = self._initializer(shape, dtype)
  96     kernel = self.NewVar(init)
  97 
  98     mask = np.ones(shape[2:], dtype=dtype.as_numpy_dtype)
  99     depth_output = shape[3]
 100     for d in xrange(depth_output):
 101       if self._strict_order:
 102         mask[d:, d] = 0
 103       else:
 104         mask[d + 1:, d] = 0
 105     mask = mask.reshape((1, 1) + mask.shape)
 106 
 107     return tf.convert_to_tensor(mask, dtype) * kernel

 110 class GroupRasterScanConv2D(blocks_std.Conv2DBase):
 111   """Conv2D with no dependency on future pixels (in raster scan order).
 112 
 113   This version only introduces dependencies on previous pixels in raster scan
 114   order. It can also introduce some dependencies on previous depth positions
 115   of the current pixel (current pixel = center pixel of the kernel) in the
 116   following way:
 117   the depth dimension of the input is split into Ki groups of size
 118   |input_group_size|, the output dimension is split into Ko groups of size
 119   |output_group_size| (usually Ki == Ko). Each output group ko of the current
 120   pixel position can only depend on previous input groups ki
 121   (i.e. ki < ko if strict_order is True or ki <= ko if strict_order is False).
 122 
 123   Notes:
 124   - Block RasterScanConv2D is a special case of GroupRasterScanConv2D
 125     where Ki == Ko == 1 (i.e. input_group_size == input_depth and
 126     output_group_size == output_depth).
 127   - For 1x1 convolution, block DepthOrderConv2D is a special case of
 128     GroupRasterScanConv2D where input_group_size == 1 and
 129     output_group_size == 1.
 130   """
 131 
 132   def __init__(self, depth, filter_size, strides, padding,
 133                strict_order=True,
 134                input_group_size=1,
 135                output_group_size=1,
 136                bias=None, act=None, initializer=None, name=None):
 137     super(GroupRasterScanConv2D, self).__init__(
 138         depth, filter_size, strides, padding, bias, act, name=name)
 139 
 140     if (filter_size[0] % 2) != 1 or (filter_size[1] % 2) != 1:
 141       raise ValueError('Kernel size should be odd.')
 142 
 143     with self._BlockScope():
 144       if initializer is None:
 145         initializer = block_util.RsqrtInitializer(dims=(0, 1, 2))
 146       self._initializer = initializer
 147       self._input_group_size = input_group_size
 148       self._output_group_size = output_group_size
 149       self._strict_order = strict_order
 150 
 151       if depth % self._output_group_size != 0:
 152         raise ValueError(
 153             'Invalid depth group size: {} for depth {}'.format(
 154                 self._output_group_size, depth))
 155       self._output_group_count = depth // self._output_group_size
 156 
 157   def _CreateKernel(self, shape, dtype):
 158     init = self._initializer(shape, dtype)
 159     kernel = self.NewVar(init)
 160 
 161     depth_input = shape[2]
 162     if depth_input % self._input_group_size != 0:
 163       raise ValueError(
 164           'Invalid depth group size: {} for depth {}'.format(
 165               self._input_group_size, depth_input))
 166     input_group_count = depth_input // self._input_group_size
 167     output_group_count = self._output_group_count
 168 
 169     # Set the mask to 0 for future pixels in raster scan order.
 170     center = shape[:2] // 2
 171     mask = np.ones([shape[0], shape[1],
 172                     input_group_count, self._input_group_size,
 173                     output_group_count, self._output_group_size],
 174                    dtype=dtype.as_numpy_dtype)
 175     mask[center[0] + 1:, :, :, :, :, :] = 0
 176     mask[center[0], center[1] + 1:, :, :, :, :] = 0
 177 
 178     # Adjust the mask for the current position (the center position).
 179     depth_output = shape[3]
 180     for d in xrange(output_group_count):
 181       mask[center[0], center[1], d + 1:, :, d:d + 1, :] = 0
 182       if self._strict_order:
 183         mask[center[0], center[1], d, :, d:d + 1, :] = 0
 184 
 185     mask = mask.reshape([shape[0], shape[1], depth_input, depth_output])
 186     return tf.convert_to_tensor(mask, dtype) * kernel

 189 class InFillingConv2D(blocks_std.Conv2DBase):
 190   """Conv2D with kernel having no dependency on the current pixel.
 191 
 192   For example, assuming a 5 x 5 kernel, the kernel is applied a spatial mask:
 193     T T T T T
 194     T T T T T
 195     T T x T T
 196     T T T T T
 197     T T T T T
 198   where 'T' marks a pixel which is available when computing the convolution
 199   for pixel 'x'. 'x' itself is not available.
 200   """
 201 
 202   def __init__(self, depth, filter_size, strides, padding,
 203                bias=None, act=None, initializer=None, name=None):
 204     super(InFillingConv2D, self).__init__(
 205         depth, filter_size, strides, padding, bias, act, name=name)
 206 
 207     if (filter_size[0] % 2) != 1 or (filter_size[1] % 2) != 1:
 208       raise ValueError('Kernel size should be odd.')
 209     if filter_size[0] == 1 and filter_size[1] == 1:
 210       raise ValueError('Kernel size should be larger than 1x1.')
 211 
 212     with self._BlockScope():
 213       if initializer is None:
 214         initializer = block_util.RsqrtInitializer(dims=(0, 1, 2))
 215       self._initializer = initializer
 216 
 217   def _CreateKernel(self, shape, dtype):
 218     init = self._initializer(shape, dtype)
 219     kernel = self.NewVar(init)
 220 
 221     mask = np.ones(shape[:2], dtype=dtype.as_numpy_dtype)
 222     center = shape[:2] // 2
 223     mask[center[0], center[1]] = 0
 224     mask = mask.reshape(mask.shape + (1, 1))
 225 
 226     return tf.convert_to_tensor(mask, dtype) * kernel
