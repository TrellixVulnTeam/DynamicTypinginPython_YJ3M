  16 """Tests for LSTM tensorflow blocks."""
  17 from __future__ import division
  18 
  19 import numpy as np

  27 class BlocksLSTMTest(tf.test.TestCase):
  28 
  29   def CheckUnary(self, y, op_type):
  30     self.assertEqual(op_type, y.op.type)
  31     self.assertEqual(1, len(y.op.inputs))
  32     return y.op.inputs[0]
  33 
  34   def CheckBinary(self, y, op_type):
  35     self.assertEqual(op_type, y.op.type)
  36     self.assertEqual(2, len(y.op.inputs))
  37     return y.op.inputs
  38 
  39   def testLSTM(self):
  40     lstm = blocks_lstm.LSTM(10)
  41     lstm.hidden = tf.zeros(shape=[10, 10], dtype=tf.float32)
  42     lstm.cell = tf.zeros(shape=[10, 10], dtype=tf.float32)
  43     x = tf.placeholder(dtype=tf.float32, shape=[10, 11])
  44     y = lstm(x)
  45 
  46     o, tanhc = self.CheckBinary(y, 'Mul')
  47     self.assertEqual(self.CheckUnary(o, 'Sigmoid').name, 'LSTM/split:3')
  48 
  49     self.assertIs(lstm.cell, self.CheckUnary(tanhc, 'Tanh'))
  50     fc, ij = self.CheckBinary(lstm.cell, 'Add')
  51 
  52     f, _ = self.CheckBinary(fc, 'Mul')
  53     self.assertEqual(self.CheckUnary(f, 'Sigmoid').name, 'LSTM/split:0')
  54 
  55     i, j = self.CheckBinary(ij, 'Mul')
  56     self.assertEqual(self.CheckUnary(i, 'Sigmoid').name, 'LSTM/split:1')
  57     j = self.CheckUnary(j, 'Tanh')
  58     self.assertEqual(j.name, 'LSTM/split:2')
  59 
  60   def testLSTMBiasInit(self):
  61     lstm = blocks_lstm.LSTM(9)
  62     x = tf.placeholder(dtype=tf.float32, shape=[15, 7])
  63     lstm(x)
  64     b = lstm._nn._bias
  65 
  66     with self.test_session():
  67       tf.global_variables_initializer().run()
  68       bias_var = b._bias.eval()
  69 
  70       comp = ([1.0] * 9) + ([0.0] * 27)
  71       self.assertAllEqual(bias_var, comp)
  72 
  73   def testConv2DLSTM(self):
  74     lstm = blocks_lstm.Conv2DLSTM(depth=10,
  75                                   filter_size=[1, 1],
  76                                   hidden_filter_size=[1, 1],
  77                                   strides=[1, 1],
  78                                   padding='SAME')
  79     lstm.hidden = tf.zeros(shape=[10, 11, 11, 10], dtype=tf.float32)
  80     lstm.cell = tf.zeros(shape=[10, 11, 11, 10], dtype=tf.float32)
  81     x = tf.placeholder(dtype=tf.float32, shape=[10, 11, 11, 1])
  82     y = lstm(x)
  83 
  84     o, tanhc = self.CheckBinary(y, 'Mul')
  85     self.assertEqual(self.CheckUnary(o, 'Sigmoid').name, 'Conv2DLSTM/split:3')
  86 
  87     self.assertIs(lstm.cell, self.CheckUnary(tanhc, 'Tanh'))
  88     fc, ij = self.CheckBinary(lstm.cell, 'Add')
  89 
  90     f, _ = self.CheckBinary(fc, 'Mul')
  91     self.assertEqual(self.CheckUnary(f, 'Sigmoid').name, 'Conv2DLSTM/split:0')
  92 
  93     i, j = self.CheckBinary(ij, 'Mul')
  94     self.assertEqual(self.CheckUnary(i, 'Sigmoid').name, 'Conv2DLSTM/split:1')
  95     j = self.CheckUnary(j, 'Tanh')
  96     self.assertEqual(j.name, 'Conv2DLSTM/split:2')
  97 
  98   def testConv2DLSTMBiasInit(self):
