  28 def HandleConvPaddingModes(x, padding, kernel_shape, strides):
  29   """Returns an updated tensor and padding type for REFLECT and SYMMETRIC.
  30 
  31   Args:
  32     x: A 4D tensor with shape [batch_size, height, width, depth].
  33     padding: Padding mode (SAME, VALID, REFLECT, or SYMMETRIC).
  34     kernel_shape: Shape of convolution kernel that will be applied.
  35     strides: Convolution stride that will be used.
  36 
  37   Returns:
  38     x and padding after adjustments for REFLECT and SYMMETRIC.
  39   """
  40   # For 1x1 convolution, all padding modes are the same.
  41   if np.all(kernel_shape[:2] == 1):
  42     return x, 'VALID'
  43 
  44   if padding == 'REFLECT' or padding == 'SYMMETRIC':
  45     # We manually compute the number of paddings as if 'SAME'.
  46     # From Tensorflow kernel, the formulas are as follows.
  47     #   output_shape = ceil(input_shape / strides)
  48     #   paddings = (output_shape - 1) * strides + filter_size - input_shape
  49     # Let x, y, s be a shorthand notations for input_shape, output_shape, and
  50     # strides, respectively. Let (x - 1) = sn + r where 0 <= r < s. Note that
  51     #   y - 1 = ceil(x / s) - 1 = floor((x - 1) / s) = n
  52     # provided that x > 0. Therefore
  53     #   paddings = n * s + filter_size - (sn + r + 1)
  54     #            = filter_size - r - 1.
  55     input_shape = x.get_shape()  # shape at graph construction time
  56     img_shape = tf.shape(x)[1:3]  # image shape (no batch) at run time
  57     remainder = tf.mod(img_shape - 1, strides[1:3])
  58     pad_sizes = kernel_shape[:2] - remainder - 1
  59 
  60     pad_rows = pad_sizes[0]
  61     pad_cols = pad_sizes[1]
  62     pad = tf.stack([[0, 0], tf.stack([pad_rows // 2, (pad_rows + 1) // 2]),
  63                     tf.stack([pad_cols // 2, (pad_cols + 1) // 2]), [0, 0]])
  64 
  65     # Manually pad the input and switch the padding mode to 'VALID'.
  66     x = tf.pad(x, pad, mode=padding)
  67     x.set_shape([input_shape[0], x.get_shape()[1],
  68                  x.get_shape()[2], input_shape[3]])
  69     padding = 'VALID'
  70 
  71   return x, padding

  74 class PassThrough(block_base.BlockBase):
  75   """A dummy transform block that does nothing."""
  76 
  77   def __init__(self):
  78     # Pass an empty string to disable name scoping.
  79     super(PassThrough, self).__init__(name='')
  80 
  81   def _Apply(self, inp):
  82     return inp
  83 
  84   @property
  85   def initialized(self):
  86     """Always returns True."""
  87     return True

  97 class BiasAdd(block_base.BlockBase):
  98   """A tf.nn.bias_add wrapper.
  99 
 100   This wrapper may act as a PassThrough block depending on the initializer
 101   provided, to make easier optional bias applications in NN blocks, etc.
 102   See __init__() for the details.
 103   """
 104 
 105   def __init__(self, initializer=Bias(0), name=None):
 106     """Initializes Bias block.
 107 
 108     |initializer| parameter have two special cases.
 109 
 110     1. If initializer is None, then this block works as a PassThrough.
 111     2. If initializer is a Bias class object, then tf.constant_initializer is
 112        used with the stored value.
 113 
 114     Args:
 115       initializer: An initializer for the bias variable.
 116       name: Name of this block.
 117     """
 118     super(BiasAdd, self).__init__(name)
 119 
 120     with self._BlockScope():
 121       if isinstance(initializer, Bias):
 122         self._initializer = tf.constant_initializer(value=initializer.value)
 123       else:
 124         self._initializer = initializer
 125 
 126       self._bias = None
 127 
 128   def _Apply(self, x):
 129     if not self._bias:
 130       init = self._initializer([int(x.get_shape()[-1])], x.dtype)
 131       self._bias = self.NewVar(init)
 132 
 133     return tf.nn.bias_add(x, self._bias)
 134 
 135   def CreateWeightLoss(self):
 136     return []

 139 class LinearBase(block_base.BlockBase):
 140   """A matmul wrapper.
 141 
 142   Returns input * W, where matrix W can be customized through derivation.
 143   """
 144 
 145   def __init__(self, depth, name=None):
 146     super(LinearBase, self).__init__(name)
 147 
 148     with self._BlockScope():
 149       self._depth = depth
 150       self._matrix = None
 151 
 152   def _CreateKernel(self, shape, dtype):
 153     raise NotImplementedError('This method must be sub-classed.')
 154 
 155   def _Apply(self, x):
 156     if not self._matrix:
 157       shape = [int(x.get_shape()[-1]), self._depth]
 158       self._matrix = self._CreateKernel(shape, x.dtype)
 159 
 160     return tf.matmul(x, self._matrix)
 161

 163 class Linear(LinearBase):
 164   """A matmul wrapper.
 165 
 166   Returns input * W, where matrix W is learned.
 167   """
 168 
 169   def __init__(self,
 170                depth,
 171                initializer=block_util.RsqrtInitializer(),
 172                name=None):
 173     super(Linear, self).__init__(depth, name)
 174 
 175     with self._BlockScope():
 176       self._initializer = initializer
 177 
 178   def _CreateKernel(self, shape, dtype):
 179     init = self._initializer(shape, dtype)
 180     return self.NewVar(init)

 183 class NN(block_base.BlockBase):
 184   """A neural network layer wrapper.
 185 
 186   Returns act(input * W + b), where matrix W, bias b are learned, and act is an
 187   optional activation function (i.e., nonlinearity).
 188 
 189   This transform block can handle multiple inputs. If x_1, x_2, ..., x_m are
 190   the inputs, then returns act(x_1 * W_1 + ... + x_m * W_m + b).
 191 
 192   Attributes:
 193     nunits: The dimension of the output.
 194   """
 195 
 196   def __init__(self,
 197                depth,
 198                bias=Bias(0),
 199                act=None,  # e.g., tf.nn.relu
 200                initializer=block_util.RsqrtInitializer(),
 201                linear_block_factory=(lambda d, i: Linear(d, initializer=i)),
 202                name=None):
 203     """Initializes NN block.
 204 
 205     Args:
 206       depth: The depth of the output.
 207       bias: An initializer for the bias, or a Bias class object. If None, there
 208         will be no bias term for this NN block. See BiasAdd block.
 209       act: Optional activation function. If None, no activation is applied.
 210       initializer: The initialization method for the matrix weights.
 211       linear_block_factory: A function used to create a linear block.
 212       name: The name of this block.
 213     """
 214     super(NN, self).__init__(name)
 215 
 216     with self._BlockScope():
 217       self._linear_block_factory = linear_block_factory
 218       self._depth = depth
 219       self._initializer = initializer
 220       self._matrices = None
 221 
 222       self._bias = BiasAdd(bias) if bias else PassThrough()
 223       self._act = act if act else PassThrough()
 224 
 225   def _Apply(self, *args):
 226     if not self._matrices:
 227       self._matrices = [
 228           self._linear_block_factory(self._depth, self._initializer)
 229           for _ in args]
 230 
 231     if len(self._matrices) != len(args):
 232       raise ValueError('{} expected {} inputs, but observed {} inputs'.format(
 233           self.name, len(self._matrices), len(args)))
 234 
 235     if len(args) > 1:
 236       y = tf.add_n([m(x) for m, x in zip(self._matrices, args)])
 237     else:
 238       y = self._matrices[0](args[0])
 239 
 240     return self._act(self._bias(y))

 243 class Conv2DBase(block_base.BlockBase):
 244   """A tf.nn.conv2d operator."""
 245 
 246   def __init__(self, depth, filter_size, strides, padding,
 247                bias=None, act=None, atrous_rate=None, conv=tf.nn.conv2d,
 248                name=None):
 249     """Initializes a Conv2DBase block.
 250 
 251     Arguments:
 252       depth: The output depth of the block (i.e. #filters); if negative, the
 253         output depth will be set to be the same as the input depth.
 254       filter_size: The size of the 2D filter. If it's specified as an integer,
 255         it's going to create a square filter. Otherwise, this is a tuple
 256         specifying the height x width of the filter.
 257       strides: A tuple specifying the y and x stride.
 258       padding: One of the valid padding modes allowed by tf.nn.conv2d, or
 259         'REFLECT'/'SYMMETRIC' for mirror padding.
 260       bias: An initializer for the bias, or a Bias class object. If None, there
 261           will be no bias in this block. See BiasAdd block.
 262       act: Optional activation function applied to the output.
 263       atrous_rate: optional input rate for ATrous convolution. If not None, this
 264           will be used and the strides will be ignored.
 265       conv: The convolution function to use (e.g. tf.nn.conv2d).
 266       name: The name for this conv2d op.
 267     """
 268     super(Conv2DBase, self).__init__(name)
 269 
 270     with self._BlockScope():
 271       self._act = act if act else PassThrough()
 272       self._bias = BiasAdd(bias) if bias else PassThrough()
 273 
 274       self._kernel_shape = np.zeros((4,), dtype=np.int32)
 275       self._kernel_shape[:2] = filter_size
 276       self._kernel_shape[3] = depth
 277 
 278       self._strides = np.ones((4,), dtype=np.int32)
 279       self._strides[1:3] = strides
 280       self._strides = list(self._strides)
 281 
 282       self._padding = padding
 283 
 284       self._kernel = None
 285       self._conv = conv
 286 
 287       self._atrous_rate = atrous_rate
 288 
 289   def _CreateKernel(self, shape, dtype):
 290     raise NotImplementedError('This method must be sub-classed')
 291 
 292   def _Apply(self, x):
 293     """Apply the self._conv op.
 294 
 295     Arguments:
 296       x: input tensor. It needs to be a 4D tensor of the form
 297           [batch, height, width, channels].
 298     Returns:
 299       The output of the convolution of x with the current convolutional
 300       kernel.
 301     Raises:
 302       ValueError: if number of channels is not defined at graph construction.
 303     """
 304     input_shape = x.get_shape().with_rank(4)
 305     input_shape[3:].assert_is_fully_defined()  # channels must be defined
 306     if self._kernel is None:
 307       assert self._kernel_shape[2] == 0, self._kernel_shape
 308       self._kernel_shape[2] = input_shape[3].value
 309       if self._kernel_shape[3] < 0:
 310         # Make output depth be the same as input depth.
 311         self._kernel_shape[3] = self._kernel_shape[2]
 312       self._kernel = self._CreateKernel(self._kernel_shape, x.dtype)
 313 
 314     x, padding = HandleConvPaddingModes(
 315         x, self._padding, self._kernel_shape, self._strides)
 316     if self._atrous_rate is None:
 317       x = self._conv(x, self._kernel, strides=self._strides, padding=padding)
 318     else:
 319       x = self._conv(x, self._kernel, rate=self._atrous_rate, padding=padding)
 320 
 321     if self._padding != 'VALID':
 322       # Manually update shape. Known shape information can be lost by tf.pad().
 323       height = (1 + (input_shape[1].value - 1) // self._strides[1]
 324                 if input_shape[1].value else None)
 325       width = (1 + (input_shape[2].value - 1) // self._strides[2]
 326                if input_shape[2].value else None)
 327       shape = x.get_shape()
 328       x.set_shape([shape[0], height, width, shape[3]])
 329 
 330     return self._act(self._bias(x))

 333 class Conv2D(Conv2DBase):
 334   """A tf.nn.conv2d operator."""
 335 
 336   def __init__(self, depth, filter_size, strides, padding,
 337                bias=None, act=None, initializer=None, name=None):
 338     """Initializes a Conv2D block.
 339 
 340     Arguments:
 341       depth: The output depth of the block (i.e., #filters)
 342       filter_size: The size of the 2D filter. If it's specified as an integer,
 343         it's going to create a square filter. Otherwise, this is a tuple
 344         specifying the height x width of the filter.
 345       strides: A tuple specifying the y and x stride.
 346       padding: One of the valid padding modes allowed by tf.nn.conv2d, or
 347         'REFLECT'/'SYMMETRIC' for mirror padding.
 348       bias: An initializer for the bias, or a Bias class object. If None, there
 349           will be no bias in this block. See BiasAdd block.
 350       act: Optional activation function applied to the output.
 351       initializer: Optional initializer for weights.
 352       name: The name for this conv2d op.
 353     """
 354     super(Conv2D, self).__init__(depth, filter_size, strides, padding, bias,
 355                                  act, conv=tf.nn.conv2d, name=name)
 356 
 357     with self._BlockScope():
 358       if initializer is None:
 359         initializer = block_util.RsqrtInitializer(dims=(0, 1, 2))
 360       self._initializer = initializer
 361 
 362   def _CreateKernel(self, shape, dtype):
 363     return self.NewVar(self._initializer(shape, dtype))
