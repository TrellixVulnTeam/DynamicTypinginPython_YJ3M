  31 def _NumpyConv2D(x, f, strides, padding, rate=1):
  32   assert strides[0] == 1 and strides[3] == 1, strides
  33 
  34   if rate > 1:
  35     f_shape = f.shape
  36     expand_f = np.zeros([f_shape[0], ((f_shape[1] - 1) * rate + 1),
  37                          f_shape[2], f_shape[3]])
  38     expand_f[:, [y * rate for y in range(f_shape[1])], :, :] = f
  39     f = np.zeros([((f_shape[0] - 1) * rate + 1), expand_f.shape[1],
  40                   f_shape[2], f_shape[3]])
  41     f[[y * rate for y in range(f_shape[0])], :, :, :] = expand_f
  42 
  43   if padding != 'VALID':
  44     assert x.shape[1] > 0 and x.shape[2] > 0, x.shape
  45     # Compute the number of padded rows and cols.
  46     # See Conv2D block comments for a math explanation.
  47     remainder = ((x.shape[1] - 1) % strides[1], (x.shape[2] - 1) % strides[2])
  48     pad_rows = f.shape[0] - remainder[0] - 1
  49     pad_cols = f.shape[1] - remainder[1] - 1
  50     pad = ((0, 0),
  51            (pad_rows // 2, (pad_rows + 1) // 2),
  52            (pad_cols // 2, (pad_cols + 1) // 2),
  53            (0, 0))
  54 
  55     # Pad the input using numpy.pad().
  56     mode = None
  57     if padding == 'SAME':
  58       mode = str('constant')
  59     if padding == 'REFLECT':
  60       mode = str('reflect')
  61     if padding == 'SYMMETRIC':
  62       mode = str('symmetric')
  63     x = np.pad(x, pad, mode=mode)
  64 
  65   # Since x is now properly padded, proceed as if padding mode is VALID.
  66   x_window = np.empty(
  67       (x.shape[0],
  68        int(math.ceil((x.shape[1] - f.shape[0] + 1) / strides[1])),
  69        int(math.ceil((x.shape[2] - f.shape[1] + 1) / strides[2])),
  70        np.prod(f.shape[:3])))
  71 
  72   # The output at pixel location (i, j) is the result of linear transformation
  73   # applied to the window whose top-left corner is at
  74   # (i * row_stride, j * col_stride).
  75   for i in xrange(x_window.shape[1]):
  76     k = i * strides[1]
  77     for j in xrange(x_window.shape[2]):
  78       l = j * strides[2]
  79       x_window[:, i, j, :] = x[:,
  80                                k:(k + f.shape[0]),
  81                                l:(l + f.shape[1]),
  82                                :].reshape((x_window.shape[0], -1))
  83 
  84   y = np.tensordot(x_window, f.reshape((-1, f.shape[3])), axes=1)
  85   return y

  90   def CheckUnary(self, y, op_type):
  91     self.assertEqual(op_type, y.op.type)
  92     self.assertEqual(1, len(y.op.inputs))
  93     return y.op.inputs[0]
  94 
  95   def CheckBinary(self, y, op_type):
  96     self.assertEqual(op_type, y.op.type)
  97     self.assertEqual(2, len(y.op.inputs))
  98     return y.op.inputs
  99 
 100   def testPassThrough(self):
 101     p = blocks_std.PassThrough()
 102     x = tf.placeholder(dtype=tf.float32, shape=[1])
 103     self.assertIs(p(x), x)
 104 
 105   def CheckBiasAdd(self, y, b):
 106     x, u = self.CheckBinary(y, 'BiasAdd')
 107     self.assertIs(u, b._bias.value())
 108     self.assertEqual(x.dtype, u.dtype.base_dtype)
 109     return x
 110 
 111   def testBiasAdd(self):
 112     b = blocks_std.BiasAdd()
 113     x = tf.placeholder(dtype=tf.float32, shape=[4, 8])
 114     y = b(x)
 115     self.assertEqual(b._bias.get_shape(), x.get_shape()[-1:])
 116     self.assertIs(x, self.CheckBiasAdd(y, b))
 117 
 118   def testBiasRankTest(self):
 119     b = blocks_std.BiasAdd()
 120     x = tf.placeholder(dtype=tf.float32, shape=[10])
 121     with self.assertRaises(ValueError):
 122       b(x)
 123 
 124   def CheckLinear(self, y, m):
 125     x, w = self.CheckBinary(y, 'MatMul')
 126     self.assertIs(w, m._matrix.value())
 127     self.assertEqual(x.dtype, w.dtype.base_dtype)
 128     return x

 130   def testLinear(self):
 131     m = blocks_std.Linear(10)
 132     x = tf.placeholder(dtype=tf.float32, shape=[8, 9])
 133     y = m(x)
 134     self.assertEqual(m._matrix.get_shape(), [9, 10])
 135     self.assertIs(x, self.CheckLinear(y, m))

 137   def testLinearShared(self):
 138     # Create a linear map which is applied twice on different inputs
 139     # (i.e. the weights of the map are shared).
 140     linear_map = blocks_std.Linear(6)
 141     x1 = tf.random_normal(shape=[1, 5])
 142     x2 = tf.random_normal(shape=[1, 5])
 143     xs = x1 + x2
 144 
 145     # Apply the transform with the same weights.
 146     y1 = linear_map(x1)
 147     y2 = linear_map(x2)
 148     ys = linear_map(xs)
 149 
 150     with self.test_session() as sess:
 151       # Initialize all the variables of the graph.
 152       tf.global_variables_initializer().run()
 153 
 154       y1_res, y2_res, ys_res = sess.run([y1, y2, ys])
 155       self.assertAllClose(y1_res + y2_res, ys_res)
 156 
 157   def CheckNN(self, y, nn, act=None):
 158     if act:
 159       pre_act = self.CheckUnary(y, act)
 160     else:
 161       pre_act = y
 162 
 163     if not isinstance(nn._bias, blocks_std.PassThrough):
 164       pre_bias = self.CheckBiasAdd(pre_act, nn._bias)
 165     else:
 166       pre_bias = pre_act
 167 
 168     if len(nn._matrices) > 1:
 169       self.assertEqual('AddN', pre_bias.op.type)
 170       pre_bias = pre_bias.op.inputs
 171     else:
 172       pre_bias = [pre_bias]
 173 
 174     self.assertEqual(len(pre_bias), len(nn._matrices))
 175     return [self.CheckLinear(u, m) for u, m in zip(pre_bias, nn._matrices)]

 201   def testNNMultipleInputs(self):
 202     nn = blocks_std.NN(10, bias=blocks_std.Bias(0), act=tf.tanh)
 203     x = [tf.placeholder(dtype=tf.float32, shape=[5, 7]),
 204          tf.placeholder(dtype=tf.float32, shape=[5, 3]),
 205          tf.placeholder(dtype=tf.float32, shape=[5, 5])]
 206     y = nn(*x)
 207     xs = self.CheckNN(y, nn, 'Tanh')
 208     self.assertEqual(len(x), len(xs))
 209     for u, v in zip(x, xs):
 210       self.assertIs(u, v)

 212   def testConv2DSAME(self):
 213     np.random.seed(142536)
 214 
 215     x_shape = [4, 16, 11, 5]
 216     f_shape = [4, 3, 5, 6]
 217     strides = [1, 2, 2, 1]
 218     padding = 'SAME'
 219 
 220     conv = blocks_std.Conv2D(depth=f_shape[-1],
 221                              filter_size=f_shape[0:2],
 222                              strides=strides[1:3],
 223                              padding=padding,
 224                              act=None,
 225                              bias=None)
 226     x_value = np.random.normal(size=x_shape)
 227     x = tf.convert_to_tensor(x_value, dtype=tf.float32)
 228     y = conv(x)
 229 
 230     with self.test_session():
 231       tf.global_variables_initializer().run()
 232       f_value = conv._kernel.eval()
 233       y_value = y.eval()
 234 
 235     y_expected = _NumpyConv2D(x_value, f_value,
 236                               strides=strides, padding=padding)
 237     self.assertAllClose(y_expected, y_value)
