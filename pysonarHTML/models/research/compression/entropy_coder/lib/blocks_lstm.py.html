  29 def LSTMBiasInit(shape, dtype):
  30   """Returns ones for forget-gate, and zeros for the others."""
  31   shape = np.array(shape)
  32 
  33   # Check internal consistencies.
  34   assert shape.shape == (1,), shape
  35   assert shape[0] % 4 == 0, shape
  36 
  37   n = shape[0] // 4
  38   ones = tf.fill([n], tf.constant(1, dtype=dtype))
  39   zeros = tf.fill([3 * n], tf.constant(0, dtype=dtype))
  40   return tf.concat([ones, zeros], 0)

  43 class LSTMBase(block_base.BlockBase):
  44   """Base class for LSTM implementations.
  45 
  46   These LSTM implementations use the pattern found in [1]. No peephole
  47   connection, i.e., cell content is not used in recurrence computation.
  48   Hidden units are also output units.
  49 
  50   [1] Zaremba, Sutskever, Vinyals. Recurrent Neural Network Regularization,
  51   2015. arxiv:1409.2329.
  52   """
  53 
  54   def __init__(self, output_shape, name):
  55     """Initializes LSTMBase class object.
  56 
  57     Args:
  58       output_shape: List representing the LSTM output shape. This argument
  59         does not include batch dimension. For example, if the LSTM output has
  60         shape [batch, depth], then pass [depth].
  61       name: Name of this block.
  62     """
  63     super(LSTMBase, self).__init__(name)
  64 
  65     with self._BlockScope():
  66       self._output_shape = [None] + list(output_shape)
  67       self._hidden = None
  68       self._cell = None
  69 
  70   @property
  71   def hidden(self):
  72     """Returns the hidden units of this LSTM."""
  73     return self._hidden
  74 
  75   @hidden.setter
  76   def hidden(self, value):
  77     """Assigns to the hidden units of this LSTM.
  78 
  79     Args:
  80       value: The new value for the hidden units. If None, the hidden units are
  81         considered to be filled with zeros.
  82     """
  83     if value is not None:
  84       value.get_shape().assert_is_compatible_with(self._output_shape)
  85     self._hidden = value
  86 
  87   @property
  88   def cell(self):
  89     """Returns the cell units of this LSTM."""
  90     return self._cell
  91 
  92   @cell.setter
  93   def cell(self, value):
  94     """Assigns to the cell units of this LSTM.
  95 
  96     Args:
  97       value: The new value for the cell units. If None, the cell units are
  98         considered to be filled with zeros.
  99     """
 100     if value is not None:
 101       value.get_shape().assert_is_compatible_with(self._output_shape)
 102     self._cell = value
 103 
 104   # Consider moving bias terms to the base, and require this method to be
 105   # linear.
 106   def _TransformInputs(self, _):
 107     """Transforms the input units to (4 * depth) units.
 108 
 109     The forget-gate, input-gate, output-gate, and cell update is computed as
 110       f, i, j, o = T(h) + R(x)
 111     where h is hidden units, x is input units, and T, R are transforms of
 112     h, x, respectively.
 113 
 114     This method implements R. Note that T is strictly linear, so if LSTM is
 115     going to use bias, this method must include the bias to the transformation.
 116 
 117     Subclasses must implement this method. See _Apply() for more details.
 118     """
 119     raise NotImplementedError()
 120 
 121   def _TransformHidden(self, _):
 122     """Transforms the hidden units to (4 * depth) units.
 123 
 124     The forget-gate, input-gate, output-gate, and cell update is computed as
 125       f, i, j, o = T(h) + R(x)
 126     where h is hidden units, x is input units, and T, R are transforms of
 127     h, x, respectively.
 128 
 129     This method implements T in the equation. The method must implement a
 130     strictly linear transformation. For example, it may use MatMul or Conv2D,
 131     but must not add bias. This is because when hidden units are zeros, then
 132     the LSTM implementation will skip calling this method, instead of passing
 133     zeros to this function.
 134 
 135     Subclasses must implement this method. See _Apply() for more details.
 136     """
 137     raise NotImplementedError()
 138 
 139   def _Apply(self, *args):
 140     xtransform = self._TransformInputs(*args)
 141     depth_axis = len(self._output_shape) - 1
 142 
 143     if self.hidden is not None:
 144       htransform = self._TransformHidden(self.hidden)
 145       f, i, j, o = tf.split(
 146           value=htransform + xtransform, num_or_size_splits=4, axis=depth_axis)
 147     else:
 148       f, i, j, o = tf.split(
 149           value=xtransform, num_or_size_splits=4, axis=depth_axis)
 150 
 151     if self.cell is not None:
 152       self.cell = tf.sigmoid(f) * self.cell + tf.sigmoid(i) * tf.tanh(j)
 153     else:
 154       self.cell = tf.sigmoid(i) * tf.tanh(j)
 155 
 156     self.hidden = tf.sigmoid(o) * tf.tanh(self.cell)
 157     return self.hidden

 160 class LSTM(LSTMBase):
 161   """Efficient LSTM implementation used in [1].
 162 
 163   [1] Zaremba, Sutskever, Vinyals. Recurrent Neural Network Regularization,
 164   2015. arxiv:1409.2329.
 165   """
 166 
 167   def __init__(self,
 168                depth,
 169                bias=LSTMBiasInit,
 170                initializer=block_util.RsqrtInitializer(),
 171                name=None):
 172     super(LSTM, self).__init__([depth], name)
 173 
 174     with self._BlockScope():
 175       self._depth = depth
 176       self._nn = blocks_std.NN(
 177           4 * depth, bias=bias, act=None, initializer=initializer)
 178       self._hidden_linear = blocks_std.Linear(
 179           4 * depth, initializer=initializer)
 180 
 181   def _TransformInputs(self, *args):
 182     return self._nn(*args)
 183 
 184   def _TransformHidden(self, h):
 185     return self._hidden_linear(h)

 188 class Conv2DLSTM(LSTMBase):
 189   """Convolutional LSTM implementation with optimizations inspired by [1].
 190 
 191   Note that when using the batch normalization feature, the bias initializer
 192   will not be used, since BN effectively cancels its effect out.
 193 
 194   [1] Zaremba, Sutskever, Vinyals. Recurrent Neural Network Regularization,
 195   2015. arxiv:1409.2329.
 196   """
 197 
 198   def __init__(self,
 199                depth,
 200                filter_size,
 201                hidden_filter_size,
 202                strides,
 203                padding,
 204                bias=LSTMBiasInit,
 205                initializer=block_util.RsqrtInitializer(dims=(0, 1, 2)),
 206                use_moving_average=False,
 207                name=None):
 208     super(Conv2DLSTM, self).__init__([None, None, depth], name)
 209     self._iter = 0
 210 
 211     with self._BlockScope():
 212       self._input_conv = blocks_std.Conv2D(
 213           4 * depth,
 214           filter_size,
 215           strides,
 216           padding,
 217           bias=None,
 218           act=None,
 219           initializer=initializer,
 220           name='input_conv2d')
 221 
 222       self._hidden_conv = blocks_std.Conv2D(
 223           4 * depth,
 224           hidden_filter_size,
 225           [1, 1],
 226           'SAME',
 227           bias=None,
 228           act=None,
 229           initializer=initializer,
 230           name='hidden_conv2d')
 231 
 232       if bias is not None:
 233         self._bias = blocks_std.BiasAdd(bias, name='biases')
 234       else:
 235         self._bias = blocks_std.PassThrough()
 236 
 237   def _TransformInputs(self, x):
 238     return self._bias(self._input_conv(x))
 239 
 240   def _TransformHidden(self, h):
 241     return self._hidden_conv(h)
 242 
 243   def _Apply(self, *args):
 244     xtransform = self._TransformInputs(*args)
 245     depth_axis = len(self._output_shape) - 1
 246 
 247     if self.hidden is not None:
 248       htransform = self._TransformHidden(self.hidden)
 249       f, i, j, o = tf.split(
 250           value=htransform + xtransform, num_or_size_splits=4, axis=depth_axis)
 251     else:
 252       f, i, j, o = tf.split(
 253           value=xtransform, num_or_size_splits=4, axis=depth_axis)
 254 
 255     if self.cell is not None:
 256       self.cell = tf.sigmoid(f) * self.cell + tf.sigmoid(i) * tf.tanh(j)
 257     else:
 258       self.cell = tf.sigmoid(i) * tf.tanh(j)
 259 
 260     self.hidden = tf.sigmoid(o) * tf.tanh(self.cell)
 261 
 262     self._iter += 1
 263     return self.hidden
