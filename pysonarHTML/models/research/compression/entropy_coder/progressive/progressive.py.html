  30 class BrnnPredictor(blocks.BlockBase):
  31   """BRNN prediction applied on one layer."""
  32 
  33   def __init__(self, code_depth, name=None):
  34     super(BrnnPredictor, self).__init__(name)
  35 
  36     with self._BlockScope():
  37       hidden_depth = 2 * code_depth
  38 
  39       # What is coming from the previous layer/iteration
  40       # is going through a regular Conv2D layer as opposed to the binary codes
  41       # of the current layer/iteration which are going through a masked
  42       # convolution.
  43       self._adaptation0 = blocks.RasterScanConv2D(
  44           hidden_depth, [7, 7], [1, 1], 'SAME',
  45           strict_order=True,
  46           bias=blocks.Bias(0), act=tf.tanh)
  47       self._adaptation1 = blocks.Conv2D(
  48           hidden_depth, [3, 3], [1, 1], 'SAME',
  49           bias=blocks.Bias(0), act=tf.tanh)
  50       self._predictor = blocks.CompositionOperator([
  51           blocks.LineOperator(
  52               blocks.RasterScanConv2DLSTM(
  53                   depth=hidden_depth,
  54                   filter_size=[1, 3],
  55                   hidden_filter_size=[1, 3],
  56                   strides=[1, 1],
  57                   padding='SAME')),
  58           blocks.Conv2D(hidden_depth, [1, 1], [1, 1], 'SAME',
  59                         bias=blocks.Bias(0), act=tf.tanh),
  60           blocks.Conv2D(code_depth, [1, 1], [1, 1], 'SAME',
  61                         bias=blocks.Bias(0), act=tf.tanh)
  62       ])
  63 
  64   def _Apply(self, x, s):
  65     # Code estimation using both:
  66     # - the state from the previous iteration/layer,
  67     # - the binary codes that are before in raster scan order.
  68     h = tf.concat(values=[self._adaptation0(x), self._adaptation1(s)], axis=3)
  69 
  70     estimated_codes = self._predictor(h)
  71 
  72     return estimated_codes

  75 class LayerPrediction(blocks.BlockBase):
  76   """Binary code prediction for one layer."""
  77 
  78   def __init__(self, layer_count, code_depth, name=None):
  79     super(LayerPrediction, self).__init__(name)
  80 
  81     self._layer_count = layer_count
  82 
  83     # No previous layer.
  84     self._layer_state = None
  85     self._current_layer = 0
  86 
  87     with self._BlockScope():
  88       # Layers used to do the conditional code prediction.
  89       self._brnn_predictors = []
  90       for _ in xrange(layer_count):
  91         self._brnn_predictors.append(BrnnPredictor(code_depth))
  92 
  93       # Layers used to generate the input of the LSTM operating on the
  94       # iteration/depth domain.
  95       hidden_depth = 2 * code_depth
  96       self._state_blocks = []
  97       for _ in xrange(layer_count):
  98         self._state_blocks.append(blocks.CompositionOperator([
  99             blocks.Conv2D(
 100                 hidden_depth, [3, 3], [1, 1], 'SAME',
 101                 bias=blocks.Bias(0), act=tf.tanh),
 102             blocks.Conv2D(
 103                 code_depth, [3, 3], [1, 1], 'SAME',
 104                 bias=blocks.Bias(0), act=tf.tanh)
 105         ]))
 106 
 107       # Memory of the RNN is equivalent to the size of 2 layers of binary
 108       # codes.
 109       hidden_depth = 2 * code_depth
 110       self._layer_rnn = blocks.CompositionOperator([
 111           blocks.Conv2DLSTM(
 112               depth=hidden_depth,
 113               filter_size=[1, 1],
 114               hidden_filter_size=[1, 1],
 115               strides=[1, 1],
 116               padding='SAME'),
 117           blocks.Conv2D(hidden_depth, [1, 1], [1, 1], 'SAME',
 118                         bias=blocks.Bias(0), act=tf.tanh),
 119           blocks.Conv2D(code_depth, [1, 1], [1, 1], 'SAME',
 120                         bias=blocks.Bias(0), act=tf.tanh)
 121       ])
 122 
 123   def _Apply(self, x):
 124     assert self._current_layer < self._layer_count
 125 
 126     # Layer state is set to 0 when there is no previous iteration.
 127     if self._layer_state is None:
 128       self._layer_state = tf.zeros_like(x, dtype=tf.float32)
 129 
 130     # Code estimation using both:
 131     # - the state from the previous iteration/layer,
 132     # - the binary codes that are before in raster scan order.
 133     estimated_codes = self._brnn_predictors[self._current_layer](
 134         x, self._layer_state)
 135 
 136     # Compute the updated layer state.
 137     h = self._state_blocks[self._current_layer](x)
 138     self._layer_state = self._layer_rnn(h)
 139     self._current_layer += 1
 140 
 141     return estimated_codes

 144 class ProgressiveModel(entropy_coder_model.EntropyCoderModel):
 145   """Progressive BRNN entropy coder model."""
 146 
 147   def __init__(self):
 148     super(ProgressiveModel, self).__init__()
 149 
 150   def Initialize(self, global_step, optimizer, config_string):
 151     if config_string is None:
 152       raise ValueError('The progressive model requires a configuration.')
 153     config = json.loads(config_string)
 154     if 'coded_layer_count' not in config:
 155       config['coded_layer_count'] = 0
 156 
 157     self._config = config
 158     self._optimizer = optimizer
 159     self._global_step = global_step
 160 
 161   def BuildGraph(self, input_codes):
 162     """Build the graph corresponding to the progressive BRNN model."""
 163     layer_depth = self._config['layer_depth']
 164     layer_count = self._config['layer_count']
 165 
 166     code_shape = input_codes.get_shape()
 167     code_depth = code_shape[-1].value
 168     if self._config['coded_layer_count'] > 0:
 169       prefix_depth = self._config['coded_layer_count'] * layer_depth
 170       if code_depth < prefix_depth:
 171         raise ValueError('Invalid prefix depth: {} VS {}'.format(
 172             prefix_depth, code_depth))
 173       input_codes = input_codes[:, :, :, :prefix_depth]
 174 
 175     code_shape = input_codes.get_shape()
 176     code_depth = code_shape[-1].value
 177     if code_depth % layer_depth != 0:
 178       raise ValueError(
 179           'Code depth must be a multiple of the layer depth: {} vs {}'.format(
 180               code_depth, layer_depth))
 181     code_layer_count = code_depth // layer_depth
 182     if code_layer_count > layer_count:
 183       raise ValueError('Input codes have too many layers: {}, max={}'.format(
 184           code_layer_count, layer_count))
 185 
 186     # Block used to estimate binary codes.
 187     layer_prediction = LayerPrediction(layer_count, layer_depth)
 188 
 189     # Block used to compute code lengths.
 190     code_length_block = blocks.CodeLength()
 191 
 192     # Loop over all the layers.
 193     code_length = []
 194     code_layers = tf.split(
 195         value=input_codes, num_or_size_splits=code_layer_count, axis=3)
 196     for k in xrange(code_layer_count):
 197       x = code_layers[k]
 198       predicted_x = layer_prediction(x)
 199       # Saturate the prediction to avoid infinite code length.
 200       epsilon = 0.001
 201       predicted_x = tf.clip_by_value(
 202           predicted_x, -1 + epsilon, +1 - epsilon)
 203       code_length.append(code_length_block(
 204           blocks.ConvertSignCodeToZeroOneCode(x),
 205           blocks.ConvertSignCodeToZeroOneCode(predicted_x)))
 206       tf.summary.scalar('code_length_layer_{:02d}'.format(k), code_length[-1])
 207     code_length = tf.stack(code_length)
 208     self.loss = tf.reduce_mean(code_length)
 209     tf.summary.scalar('loss', self.loss)
 210 
 211     # Loop over all the remaining layers just to make sure they are
 212     # instantiated. Otherwise, loading model params could fail.
 213     dummy_x = tf.zeros_like(code_layers[0])
 214     for _ in xrange(layer_count - code_layer_count):
 215       dummy_predicted_x = layer_prediction(dummy_x)
 216 
 217     # Average bitrate over total_line_count.
 218     self.average_code_length = tf.reduce_mean(code_length)
 219 
 220     if self._optimizer:
 221       optim_op = self._optimizer.minimize(self.loss,
 222                                           global_step=self._global_step)
 223       block_updates = blocks.CreateBlockUpdates()
 224       if block_updates:
 225         with tf.get_default_graph().control_dependencies([optim_op]):
 226           self.train_op = tf.group(*block_updates)
 227       else:
 228         self.train_op = optim_op
 229     else:
 230       self.train_op = None
 231 
 232   def GetConfigStringForUnitTest(self):
 233     s = '{\n'
 234     s += '"layer_depth": 1,\n'
 235     s += '"layer_count": 8\n'
 236     s += '}\n'
 237     return s

 240 @model_factory.RegisterEntropyCoderModel('progressive')
 241 def CreateProgressiveModel():
 242   return ProgressiveModel()
