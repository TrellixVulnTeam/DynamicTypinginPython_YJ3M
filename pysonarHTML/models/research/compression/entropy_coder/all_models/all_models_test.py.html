  16 """Basic test of all registered models."""
  17 
  18 import tensorflow as tf
  19 
  20 # pylint: disable=unused-import
  21 import all_models
  22 # pylint: enable=unused-import
  23 from entropy_coder.model import model_factory
  24 
  25 
  26 class AllModelsTest(tf.test.TestCase):
  27 
  28   def testBuildModelForTraining(self):
  29     factory = model_factory.GetModelRegistry()
  30     model_names = factory.GetAvailableModels()
  31 
  32     for m in model_names:
  33       tf.reset_default_graph()
  34 
  35       global_step = tf.Variable(tf.zeros([], dtype=tf.int64),
  36                                 trainable=False,
  37                                 name='global_step')
  38 
  39       optimizer = tf.train.GradientDescentOptimizer(learning_rate=0.1)
  40 
  41       batch_size = 3
  42       height = 40
  43       width = 20
  44       depth = 5
  45       binary_codes = tf.placeholder(dtype=tf.float32,
  46                                     shape=[batch_size, height, width, depth])
  47 
  48       # Create a model with the default configuration.
  49       print('Creating model: {}'.format(m))
  50       model = factory.CreateModel(m)
  51       model.Initialize(global_step,
  52                        optimizer,
  53                        model.GetConfigStringForUnitTest())
  54       self.assertTrue(model.loss is None, 'model: {}'.format(m))
  55       self.assertTrue(model.train_op is None, 'model: {}'.format(m))
  56       self.assertTrue(model.average_code_length is None, 'model: {}'.format(m))
  57 
  58       # Build the Tensorflow graph corresponding to the model.
  59       model.BuildGraph(binary_codes)
  60       self.assertTrue(model.loss is not None, 'model: {}'.format(m))
  61       self.assertTrue(model.average_code_length is not None,
  62                       'model: {}'.format(m))
  63       if model.train_op is None:
  64         print('Model {} is not trainable'.format(m))
  65 
  66 
  67 if __name__ == '__main__':
  68   tf.test.main()
