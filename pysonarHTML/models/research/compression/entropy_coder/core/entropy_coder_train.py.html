  31 FLAGS = tf.app.flags.FLAGS

  49 tf.app.flags.DEFINE_string('input_config', None,

  55 def train():
  56   if FLAGS.train_dir is None:
  57     raise ValueError('Parameter train_dir must be provided')
  58   if FLAGS.task is None:
  59     raise ValueError('Parameter task must be provided')
  60   if FLAGS.model is None:
  61     raise ValueError('Parameter model must be provided')
  62 
  63   input_config_string = config_helper.GetConfigString(FLAGS.input_config)
  64   input_config = config_helper.InputConfig(input_config_string)
  65 
  66   # Training parameters.
  67   train_config_string = config_helper.GetConfigString(FLAGS.train_config)
  68   train_config = config_helper.TrainConfig(train_config_string)
  69 
  70   batch_size = train_config.batch_size
  71   initial_learning_rate = train_config.learning_rate
  72   decay_rate = train_config.decay_rate
  73   samples_per_decay = train_config.samples_per_decay
  74 
  75   # Parameters for learning-rate decay.
  76   # The formula is decay_rate ** floor(steps / decay_steps).
  77   decay_steps = samples_per_decay / batch_size
  78   decay_steps = max(decay_steps, 1)
  79 
  80   first_code = code_loader.ReadFirstCode(input_config.data)
  81   first_code_height = (
  82       first_code.features.feature['code_shape'].int64_list.value[0])
  83   first_code_width = (
  84       first_code.features.feature['code_shape'].int64_list.value[1])
  85   max_bit_depth = (
  86       first_code.features.feature['code_shape'].int64_list.value[2])
  87   print('Maximum code depth: {}'.format(max_bit_depth))
  88 
  89   with tf.Graph().as_default():
  90     ps_ops = ["Variable", "VariableV2", "AutoReloadVariable", "VarHandleOp"]
  91     with tf.device(tf.train.replica_device_setter(FLAGS.ps_tasks,
  92                                                   ps_ops=ps_ops)):
  93       codes = code_loader.LoadBinaryCode(
  94           input_config=input_config,
  95           batch_size=batch_size)
  96       if input_config.unique_code_size:
  97         print('Input code size: {} x {}'.format(first_code_height,
  98                                                 first_code_width))
  99         codes.set_shape(
 100             [batch_size, first_code_height, first_code_width, max_bit_depth])
 101       else:
 102         codes.set_shape([batch_size, None, None, max_bit_depth])
 103       codes_effective_shape = tf.shape(codes)
 104 
 105       global_step = tf.contrib.framework.create_global_step()
 106 
 107       # Apply learning-rate decay.
 108       learning_rate = tf.train.exponential_decay(
 109           learning_rate=initial_learning_rate,
 110           global_step=global_step,
 111           decay_steps=decay_steps,
 112           decay_rate=decay_rate,
 113           staircase=True)
 114       tf.summary.scalar('Learning Rate', learning_rate)
 115       optimizer = tf.train.AdamOptimizer(learning_rate=learning_rate,
 116                                          epsilon=1.0)
 117 
 118       # Create the entropy coder model.
 119       model = model_factory.GetModelRegistry().CreateModel(FLAGS.model)
 120       model_config_string = config_helper.GetConfigString(FLAGS.model_config)
 121       model.Initialize(global_step, optimizer, model_config_string)
 122       model.BuildGraph(codes)
 123 
 124       summary_op = tf.summary.merge_all()
 125 
 126       # Verify that the model can actually be trained.
 127       if model.train_op is None:
 128         raise ValueError('Input model {} is not trainable'.format(FLAGS.model))
 129 
 130       # We disable the summary thread run by Supervisor class by passing
 131       # summary_op=None. We still pass save_summaries_secs because it is used by
 132       # the global step counter thread.
 133       is_chief = (FLAGS.task == 0)
 134       sv = tf.train.Supervisor(logdir=FLAGS.train_dir,
 135                                is_chief=is_chief,
 136                                global_step=global_step,
 137                                # saver=model.saver,
 138                                summary_op=None,
 139                                save_summaries_secs=120,
 140                                save_model_secs=600,
 141                                recovery_wait_secs=30)
 142 
 143       sess = sv.PrepareSession(FLAGS.master)
 144       sv.StartQueueRunners(sess)
 145 
 146       step = sess.run(global_step)
 147       print('Trainer initial step: {}.'.format(step))
 148 
 149       # Once everything has been setup properly, save the configs.
 150       if is_chief:
 151         config_helper.SaveConfig(FLAGS.train_dir, 'input_config.json',
 152                                  input_config_string)
 153         config_helper.SaveConfig(FLAGS.train_dir, 'model_config.json',
 154                                  model_config_string)
 155         config_helper.SaveConfig(FLAGS.train_dir, 'train_config.json',
 156                                  train_config_string)
 157 
 158       # Train the model.
 159       next_summary_time = time.time()
 160       while not sv.ShouldStop():
 161         feed_dict = None
 162 
 163         # Once in a while, update the summaries on the chief worker.
 164         if is_chief and next_summary_time < time.time():
 165           summary_str = sess.run(summary_op, feed_dict=feed_dict)
 166           sv.SummaryComputed(sess, summary_str)
 167           next_summary_time = time.time() + sv.save_summaries_secs
 168         else:
 169           tf_tensors = {
 170               'train': model.train_op,
 171               'code_length': model.average_code_length
 172           }
 173           np_tensors = sess.run(tf_tensors, feed_dict=feed_dict)
 174           print(np_tensors['code_length'])
 175 
 176       sv.Stop()
 177 
 178 
 179 def main(argv=None):  # pylint: disable=unused-argument
 180   train()

 183 if __name__ == '__main__':
 184   tf.app.run()
