  16 """Load binary codes stored as tf.Example in a TFRecord table."""
  17 
  18 import tensorflow as tf
  19 
  20 
  21 def ReadFirstCode(dataset):
  22   """Read the first example from a binary code RecordIO table."""
  23   for record in tf.python_io.tf_record_iterator(dataset):
  24     tf_example = tf.train.Example()
  25     tf_example.ParseFromString(record)
  26     break
  27   return tf_example
  28

  30 def LoadBinaryCode(input_config, batch_size):
  31   """Load a batch of binary codes from a tf.Example dataset.
  32 
  33   Args:
  34     input_config: An InputConfig proto containing the input configuration.
  35     batch_size: Output batch size of examples.
  36 
  37   Returns:
  38     A batched tensor of binary codes.
  39   """
  40   data = input_config.data
  41 
  42   # TODO: Possibly use multiple files (instead of just one).
  43   file_list = [data]
  44   filename_queue = tf.train.string_input_producer(file_list,
  45                                                   capacity=4)
  46   reader = tf.TFRecordReader()
  47   _, values = reader.read(filename_queue)
  48 
  49   serialized_example = tf.reshape(values, shape=[1])
  50   serialized_features = {
  51       'code_shape': tf.FixedLenFeature([3],
  52                                        dtype=tf.int64),
  53       'code': tf.VarLenFeature(tf.float32),
  54   }
  55   example = tf.parse_example(serialized_example, serialized_features)
  56 
  57   # 3D shape: height x width x binary_code_depth
  58   z = example['code_shape']
  59   code_shape = tf.reshape(tf.cast(z, tf.int32), [3])
  60   # Un-flatten the binary codes.
  61   code = tf.reshape(tf.sparse_tensor_to_dense(example['code']), code_shape)
  62 
  63   queue_size = 10
  64   queue = tf.PaddingFIFOQueue(
  65       queue_size + 3 * batch_size,
  66       dtypes=[code.dtype],
  67       shapes=[[None, None, None]])
  68   enqueue_op = queue.enqueue([code])
  69   dequeue_code = queue.dequeue_many(batch_size)
  70   queue_runner = tf.train.queue_runner.QueueRunner(queue, [enqueue_op])
  71   tf.add_to_collection(tf.GraphKeys.QUEUE_RUNNERS, queue_runner)
  72 
  73   return dequeue_code
