  16 """Helper functions used in both train and inference."""
  17 
  18 import json
  19 import os.path
  20 
  21 import tensorflow as tf
  22 
  23 
  24 def GetConfigString(config_file):
  25   config_string = ''
  26   if config_file is not None:
  27     config_string = open(config_file).read()
  28   return config_string
  29 
  30 
  31 class InputConfig(object):
  32 
  33   def __init__(self, config_string):
  34     config = json.loads(config_string)
  35     self.data = config["data"]
  36     self.unique_code_size = config["unique_code_size"]
  37 
  38 
  39 class TrainConfig(object):
  40 
  41   def __init__(self, config_string):
  42     config = json.loads(config_string)
  43     self.batch_size = config["batch_size"]
  44     self.learning_rate = config["learning_rate"]
  45     self.decay_rate = config["decay_rate"]
  46     self.samples_per_decay = config["samples_per_decay"]
  47 
  48 
  49 def SaveConfig(directory, filename, config_string):
  50   path = os.path.join(directory, filename)
  51   with tf.gfile.Open(path, mode='w') as f:
  52     f.write(config_string)
