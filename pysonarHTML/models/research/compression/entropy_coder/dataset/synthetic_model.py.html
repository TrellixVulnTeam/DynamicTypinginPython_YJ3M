  16 """Binary code sample generator."""
  17 
  18 import numpy as np
  19 from six.moves import xrange
  20 
  21 
  22 _CRC_LINE = [
  23     [0, 1, 0],
  24     [1, 1, 0],
  25     [1, 0, 0]
  26 ]
  27 
  28 _CRC_DEPTH = [1, 1, 0, 1]
  29 
  30 
  31 def ComputeLineCrc(code, width, y, x, d):
  32   crc = 0
  33   for dy in xrange(len(_CRC_LINE)):
  34     i = y - 1 - dy
  35     if i < 0:
  36       continue
  37     for dx in xrange(len(_CRC_LINE[dy])):
  38       j = x - 2 + dx
  39       if j < 0 or j >= width:
  40         continue
  41       crc += 1 if (code[i, j, d] != _CRC_LINE[dy][dx]) else 0
  42   return crc
  43 
  44 
  45 def ComputeDepthCrc(code, y, x, d):
  46   crc = 0
  47   for delta in xrange(len(_CRC_DEPTH)):
  48     k = d - 1 - delta
  49     if k < 0:
  50       continue
  51     crc += 1 if (code[y, x, k] != _CRC_DEPTH[delta]) else 0
  52   return crc
  53 
  54 
  55 def GenerateSingleCode(code_shape):
  56   code = np.zeros(code_shape, dtype=np.int)
  57 
  58   keep_value_proba = 0.8
  59 
  60   height = code_shape[0]
  61   width = code_shape[1]
  62   depth = code_shape[2]
  63 
  64   for d in xrange(depth):
  65     for y in xrange(height):
  66       for x in xrange(width):
  67         v1 = ComputeLineCrc(code, width, y, x, d)
  68         v2 = ComputeDepthCrc(code, y, x, d)
  69         v = 1 if (v1 + v2 >= 6) else 0
  70         if np.random.rand() < keep_value_proba:
  71           code[y, x, d] = v
  72         else:
  73           code[y, x, d] = 1 - v
  74 
  75   return code
