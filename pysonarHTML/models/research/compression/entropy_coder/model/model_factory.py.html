  19 class ModelFactory(object):
  20   """Factory of encoder/decoder models."""
  21 
  22   def __init__(self):
  23     self._model_dictionary = dict()
  24 
  25   def RegisterModel(self,
  26                     entropy_coder_model_name,
  27                     entropy_coder_model_factory):
  28     self._model_dictionary[entropy_coder_model_name] = (
  29         entropy_coder_model_factory)
  30 
  31   def CreateModel(self, model_name):
  32     current_model_factory = self._model_dictionary[model_name]
  33     return current_model_factory()
  34 
  35   def GetAvailableModels(self):
  36     return self._model_dictionary.keys()
  37 
  38 
  39 _model_registry = ModelFactory()
  40 
  41 
  42 def GetModelRegistry():
  43   return _model_registry
