  16 """Model using memory component.
  17 
  18 The model embeds images using a standard CNN architecture.
  19 These embeddings are used as keys to the memory component,
  20 which returns nearest neighbors.
  21 """
  22 
  23 import tensorflow as tf
  24 
  25 import memory

  30 class BasicClassifier(object):
  31 
  32   def __init__(self, output_dim):
  33     self.output_dim = output_dim
  34 
  35   def core_builder(self, memory_val, x, y):
  36     del x, y
  37     y_pred = memory_val
  38     loss = 0.0
  39 
  40     return loss, y_pred

  43 class LeNet(object):
  44   """Standard CNN architecture."""
  45 
  46   def __init__(self, image_size, num_channels, hidden_dim):
  47     self.image_size = image_size
  48     self.num_channels = num_channels
  49     self.hidden_dim = hidden_dim
  50     self.matrix_init = tf.truncated_normal_initializer(stddev=0.1)
  51     self.vector_init = tf.constant_initializer(0.0)
  52 
  53   def core_builder(self, x):
  54     """Embeds x using standard CNN architecture.
  55 
  56     Args:
  57       x: Batch of images as a 2-d Tensor [batch_size, -1].
  58 
  59     Returns:
  60       A 2-d Tensor [batch_size, hidden_dim] of embedded images.
  61     """
  62 
  63     ch1 = 32 * 2  # number of channels in 1st layer
  64     ch2 = 64 * 2  # number of channels in 2nd layer
  65     conv1_weights = tf.get_variable('conv1_w',
  66                                     [3, 3, self.num_channels, ch1],
  67                                     initializer=self.matrix_init)
  68     conv1_biases = tf.get_variable('conv1_b', [ch1],
  69                                    initializer=self.vector_init)
  70     conv1a_weights = tf.get_variable('conv1a_w',
  71                                      [3, 3, ch1, ch1],
  72                                      initializer=self.matrix_init)
  73     conv1a_biases = tf.get_variable('conv1a_b', [ch1],
  74                                     initializer=self.vector_init)
  75 
  76     conv2_weights = tf.get_variable('conv2_w', [3, 3, ch1, ch2],
  77                                     initializer=self.matrix_init)
  78     conv2_biases = tf.get_variable('conv2_b', [ch2],
  79                                    initializer=self.vector_init)
  80     conv2a_weights = tf.get_variable('conv2a_w', [3, 3, ch2, ch2],
  81                                      initializer=self.matrix_init)
  82     conv2a_biases = tf.get_variable('conv2a_b', [ch2],
  83                                     initializer=self.vector_init)
  84 
  85     # fully connected
  86     fc1_weights = tf.get_variable(
  87         'fc1_w', [self.image_size // 4 * self.image_size // 4 * ch2,
  88                   self.hidden_dim], initializer=self.matrix_init)
  89     fc1_biases = tf.get_variable('fc1_b', [self.hidden_dim],
  90                                  initializer=self.vector_init)
  91 
  92     # define model
  93     x = tf.reshape(x,
  94                    [-1, self.image_size, self.image_size, self.num_channels])
  95     batch_size = tf.shape(x)[0]
  96 
  97     conv1 = tf.nn.conv2d(x, conv1_weights,
  98                          strides=[1, 1, 1, 1], padding='SAME')
  99     relu1 = tf.nn.relu(tf.nn.bias_add(conv1, conv1_biases))
 100     conv1 = tf.nn.conv2d(relu1, conv1a_weights,
 101                          strides=[1, 1, 1, 1], padding='SAME')
 102     relu1 = tf.nn.relu(tf.nn.bias_add(conv1, conv1a_biases))
 103 
 104     pool1 = tf.nn.max_pool(relu1, ksize=[1, 2, 2, 1],
 105                            strides=[1, 2, 2, 1], padding='SAME')
 106 
 107     conv2 = tf.nn.conv2d(pool1, conv2_weights,
 108                          strides=[1, 1, 1, 1], padding='SAME')
 109     relu2 = tf.nn.relu(tf.nn.bias_add(conv2, conv2_biases))
 110     conv2 = tf.nn.conv2d(relu2, conv2a_weights,
 111                          strides=[1, 1, 1, 1], padding='SAME')
 112     relu2 = tf.nn.relu(tf.nn.bias_add(conv2, conv2a_biases))
 113 
 114     pool2 = tf.nn.max_pool(relu2, ksize=[1, 2, 2, 1],
 115                            strides=[1, 2, 2, 1], padding='SAME')
 116 
 117     reshape = tf.reshape(pool2, [batch_size, -1])
 118     hidden = tf.matmul(reshape, fc1_weights) + fc1_biases
 119 
 120     return hidden

 123 class Model(object):
 124   """Model for coordinating between CNN embedder and Memory module."""
 125 
 126   def __init__(self, input_dim, output_dim, rep_dim, memory_size, vocab_size,
 127                learning_rate=0.0001, use_lsh=False):
 128     self.input_dim = input_dim
 129     self.output_dim = output_dim
 130     self.rep_dim = rep_dim
 131     self.memory_size = memory_size
 132     self.vocab_size = vocab_size
 133     self.learning_rate = learning_rate
 134     self.use_lsh = use_lsh
 135 
 136     self.embedder = self.get_embedder()
 137     self.memory = self.get_memory()
 138     self.classifier = self.get_classifier()
 139 
 140     self.global_step = tf.train.get_or_create_global_step()
 141 
 142   def get_embedder(self):
 143     return LeNet(int(self.input_dim ** 0.5), 1, self.rep_dim)
 144 
 145   def get_memory(self):
 146     cls = memory.LSHMemory if self.use_lsh else memory.Memory
 147     return cls(self.rep_dim, self.memory_size, self.vocab_size)
 148 
 149   def get_classifier(self):
 150     return BasicClassifier(self.output_dim)
 151 
 152   def core_builder(self, x, y, keep_prob, use_recent_idx=True):
 153     embeddings = self.embedder.core_builder(x)
 154     if keep_prob < 1.0:
 155       embeddings = tf.nn.dropout(embeddings, keep_prob)
 156     memory_val, _, teacher_loss = self.memory.query(
 157         embeddings, y, use_recent_idx=use_recent_idx)
 158     loss, y_pred = self.classifier.core_builder(memory_val, x, y)
 159 
 160     return loss + teacher_loss, y_pred
 161 
 162   def train(self, x, y):
 163     loss, _ = self.core_builder(x, y, keep_prob=0.3)
 164     gradient_ops = self.training_ops(loss)
 165     return loss, gradient_ops
 166 
 167   def eval(self, x, y):
 168     _, y_preds = self.core_builder(x, y, keep_prob=1.0,
 169                                    use_recent_idx=False)
 170     return y_preds
 171 
 172   def get_xy_placeholders(self):
 173     return (tf.placeholder(tf.float32, [None, self.input_dim]),
 174             tf.placeholder(tf.int32, [None]))
 175 
 176   def setup(self):
 177     """Sets up all components of the computation graph."""
 178 
 179     self.x, self.y = self.get_xy_placeholders()
 180 
 181     # This context creates variables
 182     with tf.variable_scope('core', reuse=None):
 183       self.loss, self.gradient_ops = self.train(self.x, self.y)
 184     # And this one re-uses them (thus the `reuse=True`)
 185     with tf.variable_scope('core', reuse=True):
 186       self.y_preds = self.eval(self.x, self.y)
 187 
 188   def training_ops(self, loss):
 189     opt = self.get_optimizer()
 190     params = tf.trainable_variables()
 191     gradients = tf.gradients(loss, params)
 192     clipped_gradients, _ = tf.clip_by_global_norm(gradients, 5.0)
 193     return opt.apply_gradients(zip(clipped_gradients, params),
 194                                global_step=self.global_step)
 195 
 196   def get_optimizer(self):
 197     return tf.train.AdamOptimizer(learning_rate=self.learning_rate,
 198                                   epsilon=1e-4)
 199 
 200   def one_step(self, sess, x, y):
 201     outputs = [self.loss, self.gradient_ops]
 202     return sess.run(outputs, feed_dict={self.x: x, self.y: y})
 203 
 204   def episode_step(self, sess, x, y, clear_memory=False):
 205     """Performs training steps on episodic input.
 206 
 207     Args:
 208       sess: A Tensorflow Session.
 209       x: A list of batches of images defining the episode.
 210       y: A list of batches of labels corresponding to x.
 211       clear_memory: Whether to clear the memory before the episode.
 212 
 213     Returns:
 214       List of losses the same length as the episode.
 215     """
 216 
 217     outputs = [self.loss, self.gradient_ops]
 218 
 219     if clear_memory:
 220       self.clear_memory(sess)
 221 
 222     losses = []
 223     for xx, yy in zip(x, y):
 224       out = sess.run(outputs, feed_dict={self.x: xx, self.y: yy})
 225       loss = out[0]
 226       losses.append(loss)
 227 
 228     return losses
 229 
 230   def predict(self, sess, x, y=None):
 231     """Predict the labels on a single batch of examples.
 232 
 233     Args:
 234       sess: A Tensorflow Session.
 235       x: A batch of images.
 236       y: The labels for the images in x.
 237         This allows for updating the memory.
 238 
 239     Returns:
 240       Predicted y.
 241     """
 242 
 243     # Storing current memory state to restore it after prediction
 244     mem_keys, mem_vals, mem_age, _ = self.memory.get()
 245     cur_memory = (
 246         tf.identity(mem_keys),
 247         tf.identity(mem_vals),
 248         tf.identity(mem_age),
 249         None,
 250     )
 251 
 252     outputs = [self.y_preds]
 253     if y is None:
 254       ret = sess.run(outputs, feed_dict={self.x: x})
 255     else:
 256       ret = sess.run(outputs, feed_dict={self.x: x, self.y: y})
 257 
 258     # Restoring memory state
 259     self.memory.set(*cur_memory)
 260 
 261     return ret
 262 
 263   def episode_predict(self, sess, x, y, clear_memory=False):
 264     """Predict the labels on an episode of examples.
 265 
 266     Args:
 267       sess: A Tensorflow Session.
 268       x: A list of batches of images.
 269       y: A list of labels for the images in x.
 270         This allows for updating the memory.
 271       clear_memory: Whether to clear the memory before the episode.
 272 
 273     Returns:
 274       List of predicted y.
 275     """
 276 
 277     # Storing current memory state to restore it after prediction
 278     mem_keys, mem_vals, mem_age, _ = self.memory.get()
 279     cur_memory = (
 280         tf.identity(mem_keys),
 281         tf.identity(mem_vals),
 282         tf.identity(mem_age),
 283         None,
 284     )
 285 
 286     if clear_memory:
 287       self.clear_memory(sess)
 288 
 289     outputs = [self.y_preds]
 290     y_preds = []
 291     for xx, yy in zip(x, y):
 292       out = sess.run(outputs, feed_dict={self.x: xx, self.y: yy})
 293       y_pred = out[0]
 294       y_preds.append(y_pred)
 295 
 296     # Restoring memory state
 297     self.memory.set(*cur_memory)
 298 
 299     return y_preds
 300 
 301   def clear_memory(self, sess):
 302     sess.run([self.memory.clear()])
