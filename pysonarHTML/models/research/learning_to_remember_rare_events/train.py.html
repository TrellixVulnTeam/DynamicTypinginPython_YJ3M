  17 
  18 Simple command to get up and running:
  19   python train.py --memory_size=8192 \
  20       --batch_size=16 --validation_length=50 \
  21       --episode_width=5 --episode_length=30

  24 import logging
  25 import os
  26 import random

  28 import numpy as np
  29 from six.moves import xrange
  30 import tensorflow as tf

  32 import data_utils
  33 import model

  35 FLAGS = tf.flags.FLAGS
  36 
  37 tf.flags.DEFINE_integer('rep_dim', 128,
  38                         'dimension of keys to use in memory')
  39 tf.flags.DEFINE_integer('episode_length', 100, 'length of episode')
  40 tf.flags.DEFINE_integer('episode_width', 5,
  41                         'number of distinct labels in a single episode')
  42 tf.flags.DEFINE_integer('memory_size', None, 'number of slots in memory. '
  43                         'Leave as None to default to episode length')
  44 tf.flags.DEFINE_integer('batch_size', 16, 'batch size')
  45 tf.flags.DEFINE_integer('num_episodes', 100000, 'number of training episodes')
  46 tf.flags.DEFINE_integer('validation_frequency', 20,
  47                         'every so many training episodes, '
  48                         'assess validation accuracy')
  49 tf.flags.DEFINE_integer('validation_length', 10,
  50                         'number of episodes to use to compute '
  51                         'validation accuracy')
  52 tf.flags.DEFINE_integer('seed', 888, 'random seed for training sampling')
  53 tf.flags.DEFINE_string('save_dir', '', 'directory to save model to')
  54 tf.flags.DEFINE_bool('use_lsh', False,
  55                      'use locality-sensitive hashing '
  56                      '(NOTE: not fully tested)')

  59 class Trainer(object):
  60   """Class that takes care of training, validating, and checkpointing model."""
  61 
  62   def __init__(self, train_data, valid_data, input_dim, output_dim=None):
  63     self.train_data = train_data
  64     self.valid_data = valid_data
  65     self.input_dim = input_dim
  66 
  67     self.rep_dim = FLAGS.rep_dim
  68     self.episode_length = FLAGS.episode_length
  69     self.episode_width = FLAGS.episode_width
  70     self.batch_size = FLAGS.batch_size
  71     self.memory_size = (self.episode_length * self.batch_size
  72                         if FLAGS.memory_size is None else FLAGS.memory_size)
  73     self.use_lsh = FLAGS.use_lsh
  74 
  75     self.output_dim = (output_dim if output_dim is not None
  76                        else self.episode_width)
  77 
  78   def get_model(self):
  79     # vocab size is the number of distinct values that
  80     # could go into the memory key-value storage
  81     vocab_size = self.episode_width * self.batch_size
  82     return model.Model(
  83         self.input_dim, self.output_dim, self.rep_dim, self.memory_size,
  84         vocab_size, use_lsh=self.use_lsh)
  85 
  86   def sample_episode_batch(self, data,
  87                            episode_length, episode_width, batch_size):
  88     """Generates a random batch for training or validation.
  89 
  90     Structures each element of the batch as an 'episode'.
  91     Each episode contains episode_length examples and
  92     episode_width distinct labels.
  93 
  94     Args:
  95       data: A dictionary mapping label to list of examples.
  96       episode_length: Number of examples in each episode.
  97       episode_width: Distinct number of labels in each episode.
  98       batch_size: Batch size (number of episodes).
  99 
 100     Returns:
 101       A tuple (x, y) where x is a list of batches of examples
 102       with size episode_length and y is a list of batches of labels.
 103     """
 104 
 105     episodes_x = [[] for _ in xrange(episode_length)]
 106     episodes_y = [[] for _ in xrange(episode_length)]
 107     assert len(data) >= episode_width
 108     keys = data.keys()
 109     for b in xrange(batch_size):
 110       episode_labels = random.sample(keys, episode_width)
 111       remainder = episode_length % episode_width
 112       remainders = [0] * (episode_width - remainder) + [1] * remainder
 113       episode_x = [
 114           random.sample(data[lab],
 115                         r + (episode_length - remainder) // episode_width)
 116           for lab, r in zip(episode_labels, remainders)]
 117       episode = sum([[(x, i, ii) for ii, x in enumerate(xx)]
 118                      for i, xx in enumerate(episode_x)], [])
 119       random.shuffle(episode)
 120       # Arrange episode so that each distinct label is seen before moving to
 121       # 2nd showing
 122       episode.sort(key=lambda elem: elem[2])
 123       assert len(episode) == episode_length
 124       for i in xrange(episode_length):
 125         episodes_x[i].append(episode[i][0])
 126         episodes_y[i].append(episode[i][1] + b * episode_width)
 127 
 128     return ([np.array(xx).astype('float32') for xx in episodes_x],
 129             [np.array(yy).astype('int32') for yy in episodes_y])
 130 
 131   def compute_correct(self, ys, y_preds):
 132     return np.mean(np.equal(y_preds, np.array(ys)))
 133 
 134   def individual_compute_correct(self, y, y_pred):
 135     return y_pred == y
 136 
 137   def run(self):
 138     """Performs training.
 139 
 140     Trains a model using episodic training.
 141     Every so often, runs some evaluations on validation data.
 142     """
 143 
 144     train_data, valid_data = self.train_data, self.valid_data
 145     input_dim, output_dim = self.input_dim, self.output_dim
 146     rep_dim, episode_length = self.rep_dim, self.episode_length
 147     episode_width, memory_size = self.episode_width, self.memory_size
 148     batch_size = self.batch_size
 149 
 150     train_size = len(train_data)
 151     valid_size = len(valid_data)
 152     logging.info('train_size (number of labels) %d', train_size)
 153     logging.info('valid_size (number of labels) %d', valid_size)
 154     logging.info('input_dim %d', input_dim)
 155     logging.info('output_dim %d', output_dim)
 156     logging.info('rep_dim %d', rep_dim)
 157     logging.info('episode_length %d', episode_length)
 158     logging.info('episode_width %d', episode_width)
 159     logging.info('memory_size %d', memory_size)
 160     logging.info('batch_size %d', batch_size)
 161 
 162     assert all(len(v) >= float(episode_length) / episode_width
 163                for v in train_data.values())
 164     assert all(len(v) >= float(episode_length) / episode_width
 165                for v in valid_data.values())
 166 
 167     output_dim = episode_width
 168     self.model = self.get_model()
 169     self.model.setup()
 170 
 171     sess = tf.Session()
 172     sess.run(tf.global_variables_initializer())
 173 
 174     saver = tf.train.Saver(max_to_keep=10)
 175     ckpt = None
 176     if FLAGS.save_dir:
 177       ckpt = tf.train.get_checkpoint_state(FLAGS.save_dir)
 178     if ckpt and ckpt.model_checkpoint_path:
 179       logging.info('restoring from %s', ckpt.model_checkpoint_path)
 180       saver.restore(sess, ckpt.model_checkpoint_path)
 181 
 182     logging.info('starting now')
 183     losses = []
 184     random.seed(FLAGS.seed)
 185     np.random.seed(FLAGS.seed)
 186     for i in xrange(FLAGS.num_episodes):
 187       x, y = self.sample_episode_batch(
 188           train_data, episode_length, episode_width, batch_size)
 189       outputs = self.model.episode_step(sess, x, y, clear_memory=True)
 190       loss = outputs
 191       losses.append(loss)
 192 
 193       if i % FLAGS.validation_frequency == 0:
 194         logging.info('episode batch %d, avg train loss %f',
 195                      i, np.mean(losses))
 196         losses = []
 197 
 198         # validation
 199         correct = []
 200         num_shots = episode_length // episode_width
 201         correct_by_shot = dict((k, []) for k in xrange(num_shots))
 202         for _ in xrange(FLAGS.validation_length):
 203           x, y = self.sample_episode_batch(
 204               valid_data, episode_length, episode_width, 1)
 205           outputs = self.model.episode_predict(
 206               sess, x, y, clear_memory=True)
 207           y_preds = outputs
 208           correct.append(self.compute_correct(np.array(y), y_preds))
 209 
 210           # compute per-shot accuracies
 211           seen_counts = [0] * episode_width
 212           # loop over episode steps
 213           for yy, yy_preds in zip(y, y_preds):
 214             # loop over batch examples
 215             yyy, yyy_preds = int(yy[0]), int(yy_preds[0])
 216             count = seen_counts[yyy % episode_width]
 217             if count in correct_by_shot:
 218               correct_by_shot[count].append(
 219                 self.individual_compute_correct(yyy, yyy_preds))
 220             seen_counts[yyy % episode_width] = count + 1
 221 
 222         logging.info('validation overall accuracy %f', np.mean(correct))
 223         logging.info('%d-shot: %.3f, ' * num_shots,
 224                      *sum([[k, np.mean(correct_by_shot[k])]
 225                            for k in xrange(num_shots)], []))
 226 
 227         if saver and FLAGS.save_dir:
 228           saved_file = saver.save(sess,
 229                                   os.path.join(FLAGS.save_dir, 'model.ckpt'),
 230                                   global_step=self.model.global_step)
 231           logging.info('saved model to %s', saved_file)
 232
