  16 """Memory module for storing "nearest neighbors".
  17 
  18 Implements a key-value memory for generalized one-shot learning
  19 as described in the paper
  20 "Learning to Remember Rare Events"
  21 by Lukasz Kaiser, Ofir Nachum, Aurko Roy, Samy Bengio,
  22 published as a conference paper at ICLR 2017.
  23 """
  24 
  25 import numpy as np
  26 from six.moves import xrange
  27 import tensorflow as tf
  28 
  29 
  30 class Memory(object):
  31   """Memory module."""
  32 
  33   def __init__(self, key_dim, memory_size, vocab_size,
  34                choose_k=256, alpha=0.1, correct_in_top=1, age_noise=8.0,
  35                var_cache_device='', nn_device=''):
  36     self.key_dim = key_dim
  37     self.memory_size = memory_size
  38     self.vocab_size = vocab_size
  39     self.choose_k = min(choose_k, memory_size)
  40     self.alpha = alpha
  41     self.correct_in_top = correct_in_top
  42     self.age_noise = age_noise
  43     self.var_cache_device = var_cache_device  # Variables are cached here.
  44     self.nn_device = nn_device  # Device to perform nearest neighbour matmul.
  45 
  46     caching_device = var_cache_device if var_cache_device else None
  47     self.update_memory = tf.constant(True)  # Can be fed "false" if needed.
  48     self.mem_keys = tf.get_variable(
  49         'memkeys', [self.memory_size, self.key_dim], trainable=False,
  50         initializer=tf.random_uniform_initializer(-0.0, 0.0),
  51         caching_device=caching_device)
  52     self.mem_vals = tf.get_variable(
  53         'memvals', [self.memory_size], dtype=tf.int32, trainable=False,
  54         initializer=tf.constant_initializer(0, tf.int32),
  55         caching_device=caching_device)
  56     self.mem_age = tf.get_variable(
  57         'memage', [self.memory_size], dtype=tf.float32, trainable=False,
  58         initializer=tf.constant_initializer(0.0), caching_device=caching_device)
  59     self.recent_idx = tf.get_variable(
  60         'recent_idx', [self.vocab_size], dtype=tf.int32, trainable=False,
  61         initializer=tf.constant_initializer(0, tf.int32))
  62 
  63     # variable for projecting query vector into memory key
  64     self.query_proj = tf.get_variable(
  65         'memory_query_proj', [self.key_dim, self.key_dim], dtype=tf.float32,
  66         initializer=tf.truncated_normal_initializer(0, 0.01),
  67         caching_device=caching_device)
  68 
  69   def get(self):
  70     return self.mem_keys, self.mem_vals, self.mem_age, self.recent_idx
  71 
  72   def set(self, k, v, a, r=None):
  73     return tf.group(
  74         self.mem_keys.assign(k),
  75         self.mem_vals.assign(v),
  76         self.mem_age.assign(a),
  77         (self.recent_idx.assign(r) if r is not None else tf.group()))
  78 
  79   def clear(self):
  80     return tf.variables_initializer([self.mem_keys, self.mem_vals, self.mem_age,
  81                                      self.recent_idx])
  82 
  83   def get_hint_pool_idxs(self, normalized_query):
  84     """Get small set of idxs to compute nearest neighbor queries on.
  85 
  86     This is an expensive look-up on the whole memory that is used to
  87     avoid more expensive operations later on.
  88 
  89     Args:
  90       normalized_query: A Tensor of shape [None, key_dim].
  91 
  92     Returns:
  93       A Tensor of shape [None, choose_k] of indices in memory
  94       that are closest to the queries.
  95 
  96     """
  97     # look up in large memory, no gradients
  98     with tf.device(self.nn_device):
  99       similarities = tf.matmul(tf.stop_gradient(normalized_query),
 100                                self.mem_keys, transpose_b=True, name='nn_mmul')
 101     _, hint_pool_idxs = tf.nn.top_k(
 102         tf.stop_gradient(similarities), k=self.choose_k, name='nn_topk')
 103     return hint_pool_idxs
 104 
 105   def make_update_op(self, upd_idxs, upd_keys, upd_vals,
 106                      batch_size, use_recent_idx, intended_output):
 107     """Function that creates all the update ops."""
 108     mem_age_incr = self.mem_age.assign_add(tf.ones([self.memory_size],
 109                                                    dtype=tf.float32))
 110     with tf.control_dependencies([mem_age_incr]):
 111       mem_age_upd = tf.scatter_update(
 112           self.mem_age, upd_idxs, tf.zeros([batch_size], dtype=tf.float32))
 113 
 114     mem_key_upd = tf.scatter_update(
 115         self.mem_keys, upd_idxs, upd_keys)
 116     mem_val_upd = tf.scatter_update(
 117         self.mem_vals, upd_idxs, upd_vals)
 118 
 119     if use_recent_idx:
 120       recent_idx_upd = tf.scatter_update(
 121           self.recent_idx, intended_output, upd_idxs)
 122     else:
 123       recent_idx_upd = tf.group()
 124 
 125     return tf.group(mem_age_upd, mem_key_upd, mem_val_upd, recent_idx_upd)
 126 
 127   def query(self, query_vec, intended_output, use_recent_idx=True):
 128     """Queries memory for nearest neighbor.
 129 
 130     Args:
 131       query_vec: A batch of vectors to query (embedding of input to model).
 132       intended_output: The values that would be the correct output of the
 133         memory.
 134       use_recent_idx: Whether to always insert at least one instance of a
 135         correct memory fetch.
 136 
 137     Returns:
 138       A tuple (result, mask, teacher_loss).
 139       result: The result of the memory look up.
 140       mask: The affinity of the query to the result.
 141       teacher_loss: The loss for training the memory module.
 142     """
 143 
 144     batch_size = tf.shape(query_vec)[0]
 145     output_given = intended_output is not None
 146 
 147     # prepare query for memory lookup
 148     query_vec = tf.matmul(query_vec, self.query_proj)
 149     normalized_query = tf.nn.l2_normalize(query_vec, dim=1)
 150 
 151     hint_pool_idxs = self.get_hint_pool_idxs(normalized_query)
 152 
 153     if output_given and use_recent_idx:  # add at least one correct memory
 154       most_recent_hint_idx = tf.gather(self.recent_idx, intended_output)
 155       hint_pool_idxs = tf.concat(
 156           axis=1,
 157           values=[hint_pool_idxs, tf.expand_dims(most_recent_hint_idx, 1)])
 158     choose_k = tf.shape(hint_pool_idxs)[1]
 159 
 160     with tf.device(self.var_cache_device):
 161       # create small memory and look up with gradients
 162       my_mem_keys = tf.stop_gradient(tf.gather(self.mem_keys, hint_pool_idxs,
 163                                                name='my_mem_keys_gather'))
 164       similarities = tf.matmul(tf.expand_dims(normalized_query, 1),
 165                                my_mem_keys, adjoint_b=True, name='batch_mmul')
 166       hint_pool_sims = tf.squeeze(similarities, [1], name='hint_pool_sims')
 167       hint_pool_mem_vals = tf.gather(self.mem_vals, hint_pool_idxs,
 168                                      name='hint_pool_mem_vals')
 169     # Calculate softmax mask on the top-k if requested.
 170     # Softmax temperature. Say we have K elements at dist x and one at (x+a).
 171     # Softmax of the last is e^tm(x+a)/Ke^tm*x + e^tm(x+a) = e^tm*a/K+e^tm*a.
 172     # To make that 20% we'd need to have e^tm*a ~= 0.2K, so tm = log(0.2K)/a.
 173     softmax_temp = max(1.0, np.log(0.2 * self.choose_k) / self.alpha)
 174     mask = tf.nn.softmax(hint_pool_sims[:, :choose_k - 1] * softmax_temp)
 175 
 176     # prepare returned values
 177     nearest_neighbor = tf.to_int32(
 178         tf.argmax(hint_pool_sims[:, :choose_k - 1], 1))
 179 
 180     no_teacher_idxs = tf.gather(
 181         tf.reshape(hint_pool_idxs, [-1]),
 182         nearest_neighbor + choose_k * tf.range(batch_size))
 183 
 184     with tf.device(self.var_cache_device):
 185       result = tf.gather(self.mem_vals, tf.reshape(no_teacher_idxs, [-1]))
 186 
 187     if not output_given:
 188         teacher_loss = None
 189         return result, mask, teacher_loss
 190 
 191     # prepare hints from the teacher on hint pool
 192     teacher_hints = tf.to_float(
 193         tf.abs(tf.expand_dims(intended_output, 1) - hint_pool_mem_vals))
 194     teacher_hints = 1.0 - tf.minimum(1.0, teacher_hints)
 195 
 196     teacher_vals, teacher_hint_idxs = tf.nn.top_k(
 197         hint_pool_sims * teacher_hints, k=1)
 198     neg_teacher_vals, _ = tf.nn.top_k(
 199         hint_pool_sims * (1 - teacher_hints), k=1)
 200 
 201     # bring back idxs to full memory
 202     teacher_idxs = tf.gather(
 203         tf.reshape(hint_pool_idxs, [-1]),
 204         teacher_hint_idxs[:, 0] + choose_k * tf.range(batch_size))
 205 
 206     # zero-out teacher_vals if there are no hints
 207     teacher_vals *= (
 208         1 - tf.to_float(tf.equal(0.0, tf.reduce_sum(teacher_hints, 1))))
 209 
 210     # we'll determine whether to do an update to memory based on whether
 211     # memory was queried correctly
 212     sliced_hints = tf.slice(teacher_hints, [0, 0], [-1, self.correct_in_top])
 213     incorrect_memory_lookup = tf.equal(0.0, tf.reduce_sum(sliced_hints, 1))
 214 
 215     # loss based on triplet loss
 216     teacher_loss = (tf.nn.relu(neg_teacher_vals - teacher_vals + self.alpha)
 217                     - self.alpha)
 218 
 219     # prepare memory updates
 220     update_keys = normalized_query
 221     update_vals = intended_output
 222 
 223     fetched_idxs = teacher_idxs  # correctly fetched from memory
 224     with tf.device(self.var_cache_device):
 225       fetched_keys = tf.gather(self.mem_keys, fetched_idxs, name='fetched_keys')
 226       fetched_vals = tf.gather(self.mem_vals, fetched_idxs, name='fetched_vals')
 227 
 228     # do memory updates here
 229     fetched_keys_upd = update_keys + fetched_keys  # Momentum-like update
 230     fetched_keys_upd = tf.nn.l2_normalize(fetched_keys_upd, dim=1)
 231     # Randomize age a bit, e.g., to select different ones in parallel workers.
 232     mem_age_with_noise = self.mem_age + tf.random_uniform(
 233         [self.memory_size], - self.age_noise, self.age_noise)
 234 
 235     _, oldest_idxs = tf.nn.top_k(mem_age_with_noise, k=batch_size, sorted=False)
 236 
 237     with tf.control_dependencies([result]):
 238       upd_idxs = tf.where(incorrect_memory_lookup,
 239                           oldest_idxs,
 240                           fetched_idxs)
 241       # upd_idxs = tf.Print(upd_idxs, [upd_idxs], "UPD IDX", summarize=8)
 242       upd_keys = tf.where(incorrect_memory_lookup,
 243                           update_keys,
 244                           fetched_keys_upd)
 245       upd_vals = tf.where(incorrect_memory_lookup,
 246                           update_vals,
 247                           fetched_vals)
 248 
 249     def make_update_op():
 250       return self.make_update_op(upd_idxs, upd_keys, upd_vals,
 251                                  batch_size, use_recent_idx, intended_output)
 252 
 253     update_op = tf.cond(self.update_memory, make_update_op, tf.no_op)
 254 
 255     with tf.control_dependencies([update_op]):
 256       result = tf.identity(result)
 257       mask = tf.identity(mask)
 258       teacher_loss = tf.identity(teacher_loss)
 259 
 260     return result, mask, tf.reduce_mean(teacher_loss)

 263 class LSHMemory(Memory):
 264   """Memory employing locality sensitive hashing.
 265 
 266   Note: Not fully tested.
 267   """
 268 
 269   def __init__(self, key_dim, memory_size, vocab_size,
 270                choose_k=256, alpha=0.1, correct_in_top=1, age_noise=8.0,
 271                var_cache_device='', nn_device='',
 272                num_hashes=None, num_libraries=None):
 273     super(LSHMemory, self).__init__(
 274         key_dim, memory_size, vocab_size,
 275         choose_k=choose_k, alpha=alpha, correct_in_top=1, age_noise=age_noise,
 276         var_cache_device=var_cache_device, nn_device=nn_device)
 277 
 278     self.num_libraries = num_libraries or int(self.choose_k ** 0.5)
 279     self.num_per_hash_slot = max(1, self.choose_k // self.num_libraries)
 280     self.num_hashes = (num_hashes or
 281                        int(np.log2(self.memory_size / self.num_per_hash_slot)))
 282     self.num_hashes = min(max(self.num_hashes, 1), 20)
 283     self.num_hash_slots = 2 ** self.num_hashes
 284 
 285     # hashing vectors
 286     self.hash_vecs = [
 287         tf.get_variable(
 288             'hash_vecs%d' % i, [self.num_hashes, self.key_dim],
 289             dtype=tf.float32, trainable=False,
 290             initializer=tf.truncated_normal_initializer(0, 1))
 291         for i in xrange(self.num_libraries)]
 292 
 293     # map representing which hash slots map to which mem keys
 294     self.hash_slots = [
 295         tf.get_variable(
 296             'hash_slots%d' % i, [self.num_hash_slots, self.num_per_hash_slot],
 297             dtype=tf.int32, trainable=False,
 298             initializer=tf.random_uniform_initializer(maxval=self.memory_size,
 299                                                       dtype=tf.int32))
 300         for i in xrange(self.num_libraries)]
 301 
 302   def get(self):  # not implemented
 303     return self.mem_keys, self.mem_vals, self.mem_age, self.recent_idx
 304 
 305   def set(self, k, v, a, r=None):  # not implemented
 306     return tf.group(
 307         self.mem_keys.assign(k),
 308         self.mem_vals.assign(v),
 309         self.mem_age.assign(a),
 310         (self.recent_idx.assign(r) if r is not None else tf.group()))
 311 
 312   def clear(self):
 313     return tf.variables_initializer([self.mem_keys, self.mem_vals, self.mem_age,
 314                                      self.recent_idx] + self.hash_slots)
 315 
 316   def get_hash_slots(self, query):
 317     """Gets hashed-to buckets for batch of queries.
 318 
 319     Args:
 320       query: 2-d Tensor of query vectors.
 321 
 322     Returns:
 323       A list of hashed-to buckets for each hash function.
 324     """
 325 
 326     binary_hash = [
 327         tf.less(tf.matmul(query, self.hash_vecs[i], transpose_b=True), 0)
 328         for i in xrange(self.num_libraries)]
 329     hash_slot_idxs = [
 330         tf.reduce_sum(
 331             tf.to_int32(binary_hash[i]) *
 332             tf.constant([[2 ** i for i in xrange(self.num_hashes)]],
 333                         dtype=tf.int32), 1)
 334         for i in xrange(self.num_libraries)]
 335     return hash_slot_idxs
 336 
 337   def get_hint_pool_idxs(self, normalized_query):
 338     """Get small set of idxs to compute nearest neighbor queries on.
 339 
 340     This is an expensive look-up on the whole memory that is used to
 341     avoid more expensive operations later on.
 342 
 343     Args:
 344       normalized_query: A Tensor of shape [None, key_dim].
 345 
 346     Returns:
 347       A Tensor of shape [None, choose_k] of indices in memory
 348       that are closest to the queries.
 349 
 350     """
 351     # get hash of query vecs
 352     hash_slot_idxs = self.get_hash_slots(normalized_query)
 353 
 354     # grab mem idxs in the hash slots
 355     hint_pool_idxs = [
 356         tf.maximum(tf.minimum(
 357             tf.gather(self.hash_slots[i], idxs),
 358             self.memory_size - 1), 0)
 359         for i, idxs in enumerate(hash_slot_idxs)]
 360 
 361     return tf.concat(axis=1, values=hint_pool_idxs)
 362 
 363   def make_update_op(self, upd_idxs, upd_keys, upd_vals,
 364                      batch_size, use_recent_idx, intended_output):
 365     """Function that creates all the update ops."""
 366     base_update_op = super(LSHMemory, self).make_update_op(
 367         upd_idxs, upd_keys, upd_vals,
 368         batch_size, use_recent_idx, intended_output)
 369 
 370     # compute hash slots to be updated
 371     hash_slot_idxs = self.get_hash_slots(upd_keys)
 372 
 373     # make updates
 374     update_ops = []
 375     with tf.control_dependencies([base_update_op]):
 376       for i, slot_idxs in enumerate(hash_slot_idxs):
 377         # for each slot, choose which entry to replace
 378         entry_idx = tf.random_uniform([batch_size],
 379                                       maxval=self.num_per_hash_slot,
 380                                       dtype=tf.int32)
 381         entry_mul = 1 - tf.one_hot(entry_idx, self.num_per_hash_slot,
 382                                    dtype=tf.int32)
 383         entry_add = (tf.expand_dims(upd_idxs, 1) *
 384                      tf.one_hot(entry_idx, self.num_per_hash_slot,
 385                                 dtype=tf.int32))
 386 
 387         mul_op = tf.scatter_mul(self.hash_slots[i], slot_idxs, entry_mul)
 388         with tf.control_dependencies([mul_op]):
 389           add_op = tf.scatter_add(self.hash_slots[i], slot_idxs, entry_add)
 390           update_ops.append(add_op)
 391 
 392     return tf.group(*update_ops)
