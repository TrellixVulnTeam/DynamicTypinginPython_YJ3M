  50 def get_data():
  51   """Get data in form suitable for episodic training.
  52 
  53   Returns:
  54     Train and test data as dictionaries mapping
  55     label to list of examples.
  56   """
  57   with tf.gfile.GFile(DATA_FILE_FORMAT % 'train', 'rb') as f:
  58     processed_train_data = pickle.load(f)
  59   with tf.gfile.GFile(DATA_FILE_FORMAT % 'test', 'rb') as f:
  60     processed_test_data = pickle.load(f)
  61 
  62   train_data = {}
  63   test_data = {}
  64 
  65   for data, processed_data in zip([train_data, test_data],
  66                                   [processed_train_data, processed_test_data]):
  67     for image, label in zip(processed_data['images'],
  68                             processed_data['labels']):
  69       if label not in data:
  70         data[label] = []
  71       data[label].append(image.reshape([-1]).astype('float32'))
  72 
  73   intersection = set(train_data.keys()) & set(test_data.keys())
  74   assert not intersection, 'Train and test data intersect.'
  75   ok_num_examples = [len(ll) == 20 for _, ll in train_data.items()]
  76   assert all(ok_num_examples), 'Bad number of examples in train data.'
  77   ok_num_examples = [len(ll) == 20 for _, ll in test_data.items()]
  78   assert all(ok_num_examples), 'Bad number of examples in test data.'
  79 
  80   logging.info('Number of labels in train data: %d.', len(train_data))
  81   logging.info('Number of labels in test data: %d.', len(test_data))
  82 
  83   return train_data, test_data

  86 def crawl_directory(directory, augment_with_rotations=False,
  87                     first_label=0):
  88   """Crawls data directory and returns stuff."""
  89   label_idx = first_label
  90   images = []
  91   labels = []
  92   info = []
  93 
  94   # traverse root directory
  95   for root, _, files in os.walk(directory):
  96     logging.info('Reading files from %s', root)
  97     fileflag = 0
  98     for file_name in files:
  99       full_file_name = os.path.join(root, file_name)
 100       img = imread(full_file_name, flatten=True)
 101       for i, angle in enumerate([0, 90, 180, 270]):
 102         if not augment_with_rotations and i > 0:
 103           break
 104 
 105         images.append(imrotate(img, angle))
 106         labels.append(label_idx + i)
 107         info.append(full_file_name)
 108 
 109       fileflag = 1
 110 
 111     if fileflag:
 112       label_idx += 4 if augment_with_rotations else 1
 113 
 114   return images, labels, info

 117 def resize_images(images, new_width, new_height):
 118   """Resize images to new dimensions."""
 119   resized_images = np.zeros([images.shape[0], new_width, new_height],
 120                             dtype=np.float32)
 121 
 122   for i in range(images.shape[0]):
 123     resized_images[i, :, :] = imresize(images[i, :, :],
 124                                        [new_width, new_height],
 125                                        interp='bilinear',
 126                                        mode=None)
 127   return resized_images

 130 def write_datafiles(directory, write_file,
 131                     resize=True, rotate=False,
 132                     new_width=IMAGE_NEW_SIZE, new_height=IMAGE_NEW_SIZE,
 133                     first_label=0):
 134   """Load and preprocess images from a directory and write them to a file.
 135 
 136   Args:
 137     directory: Directory of alphabet sub-directories.
 138     write_file: Filename to write to.
 139     resize: Whether to resize the images.
 140     rotate: Whether to augment the dataset with rotations.
 141     new_width: New resize width.
 142     new_height: New resize height.
 143     first_label: Label to start with.
 144 
 145   Returns:
 146     Number of new labels created.
 147   """
 148 
 149   # these are the default sizes for Omniglot:
 150   imgwidth = IMAGE_ORIGINAL_SIZE
 151   imgheight = IMAGE_ORIGINAL_SIZE
 152 
 153   logging.info('Reading the data.')
 154   images, labels, info = crawl_directory(directory,
 155                                          augment_with_rotations=rotate,
 156                                          first_label=first_label)
 157 
 158   images_np = np.zeros([len(images), imgwidth, imgheight], dtype=np.bool)
 159   labels_np = np.zeros([len(labels)], dtype=np.uint32)
 160   for i in xrange(len(images)):
 161     images_np[i, :, :] = images[i]
 162     labels_np[i] = labels[i]
 163 
 164   if resize:
 165     logging.info('Resizing images.')
 166     resized_images = resize_images(images_np, new_width, new_height)
 167 
 168     logging.info('Writing resized data in float32 format.')
 169     data = {'images': resized_images,
 170             'labels': labels_np,
 171             'info': info}
 172     with tf.gfile.GFile(write_file, 'w') as f:
 173       pickle.dump(data, f)
 174   else:
 175     logging.info('Writing original sized data in boolean format.')
 176     data = {'images': images_np,
 177             'labels': labels_np,
 178             'info': info}
 179     with tf.gfile.GFile(write_file, 'w') as f:
 180       pickle.dump(data, f)
 181 
 182   return len(np.unique(labels_np))

 185 def maybe_download_data():
 186   """Download Omniglot repo if it does not exist."""
 187   if os.path.exists(REPO_DIR):
 188     logging.info('It appears that Git repo already exists.')
 189   else:
 190     logging.info('It appears that Git repo does not exist.')
 191     logging.info('Cloning now.')
 192 
 193     subprocess.check_output('git clone %s' % REPO_LOCATION, shell=True)
 194 
 195   if os.path.exists(TRAIN_DIR):
 196     logging.info('It appears that train data has already been unzipped.')
 197   else:
 198     logging.info('It appears that train data has not been unzipped.')
 199     logging.info('Unzipping now.')
 200 
 201     subprocess.check_output('unzip %s.zip -d %s' % (TRAIN_DIR, DATA_DIR),
 202                             shell=True)
 203 
 204   if os.path.exists(TEST_DIR):
 205     logging.info('It appears that test data has already been unzipped.')
 206   else:
 207     logging.info('It appears that test data has not been unzipped.')
 208     logging.info('Unzipping now.')
 209 
 210     subprocess.check_output('unzip %s.zip -d %s' % (TEST_DIR, DATA_DIR),
 211                             shell=True)

 214 def preprocess_omniglot():
 215   """Download and prepare raw Omniglot data.
 216 
 217   Downloads the data from GitHub if it does not exist.
 218   Then load the images, augment with rotations if desired.
 219   Resize the images and write them to a pickle file.
 220   """
 221 
 222   maybe_download_data()
 223 
 224   directory = TRAIN_DIR
 225   write_file = DATA_FILE_FORMAT % 'train'
 226   num_labels = write_datafiles(
 227       directory, write_file, resize=True, rotate=TRAIN_ROTATIONS,
 228       new_width=IMAGE_NEW_SIZE, new_height=IMAGE_NEW_SIZE)
 229 
 230   directory = TEST_DIR
 231   write_file = DATA_FILE_FORMAT % 'test'
 232   write_datafiles(directory, write_file, resize=True, rotate=TEST_ROTATIONS,
 233                   new_width=IMAGE_NEW_SIZE, new_height=IMAGE_NEW_SIZE,
 234                   first_label=num_labels)
