  16 """Code for building the input for the prediction model."""
  17 
  18 import os
  19 
  20 import numpy as np
  21 import tensorflow as tf
  22 
  23 from tensorflow.python.platform import flags
  24 from tensorflow.python.platform import gfile
  25 
  26 
  27 FLAGS = flags.FLAGS
  28 
  29 # Original image dimensions
  30 ORIGINAL_WIDTH = 640
  31 ORIGINAL_HEIGHT = 512
  32 COLOR_CHAN = 3
  33 
  34 # Default image dimensions.
  35 IMG_WIDTH = 64
  36 IMG_HEIGHT = 64
  37 
  38 # Dimension of the state and action.
  39 STATE_DIM = 5
  40 
  41 
  42 def build_tfrecord_input(training=True):
  43   """Create input tfrecord tensors.
  44 
  45   Args:
  46     training: training or validation data.
  47   Returns:
  48     list of tensors corresponding to images, actions, and states. The images
  49     tensor is 5D, batch x time x height x width x channels. The state and
  50     action tensors are 3D, batch x time x dimension.
  51   Raises:
  52     RuntimeError: if no files found.
  53   """
  54   filenames = gfile.Glob(os.path.join(FLAGS.data_dir, '*'))
  55   if not filenames:
  56     raise RuntimeError('No data files found.')
  57   index = int(np.floor(FLAGS.train_val_split * len(filenames)))
  58   if training:
  59     filenames = filenames[:index]
  60   else:
  61     filenames = filenames[index:]
  62   filename_queue = tf.train.string_input_producer(filenames, shuffle=True)
  63   reader = tf.TFRecordReader()
  64   _, serialized_example = reader.read(filename_queue)
  65 
  66   image_seq, state_seq, action_seq = [], [], []
  67 
  68   for i in range(FLAGS.sequence_length):
  69     image_name = 'move/' + str(i) + '/image/encoded'
  70     action_name = 'move/' + str(i) + '/commanded_pose/vec_pitch_yaw'
  71     state_name = 'move/' + str(i) + '/endeffector/vec_pitch_yaw'
  72     if FLAGS.use_state:
  73       features = {image_name: tf.FixedLenFeature([1], tf.string),
  74                   action_name: tf.FixedLenFeature([STATE_DIM], tf.float32),
  75                   state_name: tf.FixedLenFeature([STATE_DIM], tf.float32)}
  76     else:
  77       features = {image_name: tf.FixedLenFeature([1], tf.string)}
  78     features = tf.parse_single_example(serialized_example, features=features)
  79 
  80     image_buffer = tf.reshape(features[image_name], shape=[])
  81     image = tf.image.decode_jpeg(image_buffer, channels=COLOR_CHAN)
  82     image.set_shape([ORIGINAL_HEIGHT, ORIGINAL_WIDTH, COLOR_CHAN])
  83 
  84     if IMG_HEIGHT != IMG_WIDTH:
  85       raise ValueError('Unequal height and width unsupported')
  86 
  87     crop_size = min(ORIGINAL_HEIGHT, ORIGINAL_WIDTH)
  88     image = tf.image.resize_image_with_crop_or_pad(image, crop_size, crop_size)
  89     image = tf.reshape(image, [1, crop_size, crop_size, COLOR_CHAN])
  90     image = tf.image.resize_bicubic(image, [IMG_HEIGHT, IMG_WIDTH])
  91     image = tf.cast(image, tf.float32) / 255.0
  92     image_seq.append(image)
  93 
  94     if FLAGS.use_state:
  95       state = tf.reshape(features[state_name], shape=[1, STATE_DIM])
  96       state_seq.append(state)
  97       action = tf.reshape(features[action_name], shape=[1, STATE_DIM])
  98       action_seq.append(action)
  99 
 100   image_seq = tf.concat(axis=0, values=image_seq)
 101 
 102   if FLAGS.use_state:
 103     state_seq = tf.concat(axis=0, values=state_seq)
 104     action_seq = tf.concat(axis=0, values=action_seq)
 105     [image_batch, action_batch, state_batch] = tf.train.batch(
 106         [image_seq, action_seq, state_seq],
 107         FLAGS.batch_size,
 108         num_threads=FLAGS.batch_size,
 109         capacity=100 * FLAGS.batch_size)
 110     return image_batch, action_batch, state_batch
 111   else:
 112     image_batch = tf.train.batch(
 113         [image_seq],
 114         FLAGS.batch_size,
 115         num_threads=FLAGS.batch_size,
 116         capacity=100 * FLAGS.batch_size)
 117     zeros_batch = tf.zeros([FLAGS.batch_size, FLAGS.sequence_length, STATE_DIM])
 118     return image_batch, zeros_batch, zeros_batch
