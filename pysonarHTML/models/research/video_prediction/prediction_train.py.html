  77 ## Helper functions
  78 def peak_signal_to_noise_ratio(true, pred):
  79   """Image quality metric based on maximal signal power vs. power of the noise.
  80 
  81   Args:
  82     true: the ground truth image.
  83     pred: the predicted image.
  84   Returns:
  85     peak signal to noise ratio (PSNR)
  86   """
  87   return 10.0 * (- tf.log(tf.maximum(mean_squared_error(true, pred), EPSILON))) / tf.log(10.0)
  88 
  89 
  90 def mean_squared_error(true, pred):
  91   """L2 distance between tensors true and pred.
  92 
  93   Args:
  94     true: the ground truth image.
  95     pred: the predicted image.
  96   Returns:
  97     mean squared error between ground truth and predicted image.
  98   """
  99   return tf.reduce_sum(tf.square(true - pred)) / tf.to_float(tf.size(pred))

 102 class Model(object):
 103 
 104   def __init__(self,
 105                images=None,
 106                actions=None,
 107                states=None,
 108                sequence_length=None,
 109                reuse_scope=None,
 110                prefix=None):
 111 
 112     if sequence_length is None:
 113       sequence_length = FLAGS.sequence_length
 114 
 115     if prefix is None:
 116         prefix = tf.placeholder(tf.string, [])
 117     self.prefix = prefix
 118     self.iter_num = tf.placeholder(tf.float32, [])
 119     summaries = []
 120 
 121     # Split into timesteps.
 122     actions = tf.split(axis=1, num_or_size_splits=int(actions.get_shape()[1]), value=actions)
 123     actions = [tf.squeeze(act) for act in actions]
 124     states = tf.split(axis=1, num_or_size_splits=int(states.get_shape()[1]), value=states)
 125     states = [tf.squeeze(st) for st in states]
 126     images = tf.split(axis=1, num_or_size_splits=int(images.get_shape()[1]), value=images)
 127     images = [tf.squeeze(img) for img in images]
 128 
 129     if reuse_scope is None:
 130       gen_images, gen_states = construct_model(
 131           images,
 132           actions,
 133           states,
 134           iter_num=self.iter_num,
 135           k=FLAGS.schedsamp_k,
 136           use_state=FLAGS.use_state,
 137           num_masks=FLAGS.num_masks,
 138           cdna=FLAGS.model == 'CDNA',
 139           dna=FLAGS.model == 'DNA',
 140           stp=FLAGS.model == 'STP',
 141           context_frames=FLAGS.context_frames)
 142     else:  # If it's a validation or test model.
 143       with tf.variable_scope(reuse_scope, reuse=True):
 144         gen_images, gen_states = construct_model(
 145             images,
 146             actions,
 147             states,
 148             iter_num=self.iter_num,
 149             k=FLAGS.schedsamp_k,
 150             use_state=FLAGS.use_state,
 151             num_masks=FLAGS.num_masks,
 152             cdna=FLAGS.model == 'CDNA',
 153             dna=FLAGS.model == 'DNA',
 154             stp=FLAGS.model == 'STP',
 155             context_frames=FLAGS.context_frames)
 156 
 157     # L2 loss, PSNR for eval.
 158     loss, psnr_all = 0.0, 0.0
 159     for i, x, gx in zip(
 160         range(len(gen_images)), images[FLAGS.context_frames:],
 161         gen_images[FLAGS.context_frames - 1:]):
 162       recon_cost = mean_squared_error(x, gx)
 163       psnr_i = peak_signal_to_noise_ratio(x, gx)
 164       psnr_all += psnr_i
 165       summaries.append(
 166           tf.summary.scalar(prefix + '_recon_cost' + str(i), recon_cost))
 167       summaries.append(tf.summary.scalar(prefix + '_psnr' + str(i), psnr_i))
 168       loss += recon_cost
 169 
 170     for i, state, gen_state in zip(
 171         range(len(gen_states)), states[FLAGS.context_frames:],
 172         gen_states[FLAGS.context_frames - 1:]):
 173       state_cost = mean_squared_error(state, gen_state) * 1e-4
 174       summaries.append(
 175           tf.summary.scalar(prefix + '_state_cost' + str(i), state_cost))
 176       loss += state_cost
 177     summaries.append(tf.summary.scalar(prefix + '_psnr_all', psnr_all))
 178     self.psnr_all = psnr_all
 179 
 180     self.loss = loss = loss / np.float32(len(images) - FLAGS.context_frames)
 181 
 182     summaries.append(tf.summary.scalar(prefix + '_loss', loss))
 183 
 184     self.lr = tf.placeholder_with_default(FLAGS.learning_rate, ())
 185 
 186     self.train_op = tf.train.AdamOptimizer(self.lr).minimize(loss)
 187     self.summ_op = tf.summary.merge(summaries)
 188

 190 def main(unused_argv):
 191 
 192   print('Constructing models and inputs.')
 193   with tf.variable_scope('model', reuse=None) as training_scope:
 194     images, actions, states = build_tfrecord_input(training=True)
 195     model = Model(images, actions, states, FLAGS.sequence_length,
 196                   prefix='train')
 197 
 198   with tf.variable_scope('val_model', reuse=None):
 199     val_images, val_actions, val_states = build_tfrecord_input(training=False)
 200     val_model = Model(val_images, val_actions, val_states,
 201                       FLAGS.sequence_length, training_scope, prefix='val')
 202 
 203   print('Constructing saver.')
 204   # Make saver.
 205   saver = tf.train.Saver(
 206       tf.get_collection(tf.GraphKeys.GLOBAL_VARIABLES), max_to_keep=0)
 207 
 208   # Make training session.
 209   sess = tf.InteractiveSession()
 210   sess.run(tf.global_variables_initializer())
 211 
 212   summary_writer = tf.summary.FileWriter(
 213       FLAGS.event_log_dir, graph=sess.graph, flush_secs=10)
 214 
 215   if FLAGS.pretrained_model:
 216     saver.restore(sess, FLAGS.pretrained_model)
 217 
 218   tf.train.start_queue_runners(sess)
 219 
 220   tf.logging.info('iteration number, cost')
 221 
 222   # Run training.
 223   for itr in range(FLAGS.num_iterations):
 224     # Generate new batch of data.
 225     feed_dict = {model.iter_num: np.float32(itr),
 226                  model.lr: FLAGS.learning_rate}
 227     cost, _, summary_str = sess.run([model.loss, model.train_op, model.summ_op],
 228                                     feed_dict)
 229 
 230     # Print info: iteration #, cost.
 231     tf.logging.info(str(itr) + ' ' + str(cost))
 232 
 233     if (itr) % VAL_INTERVAL == 2:
 234       # Run through validation set.
 235       feed_dict = {val_model.lr: 0.0,
 236                    val_model.iter_num: np.float32(itr)}
 237       _, val_summary_str = sess.run([val_model.train_op, val_model.summ_op],
 238                                      feed_dict)
 239       summary_writer.add_summary(val_summary_str, itr)
 240 
 241     if (itr) % SAVE_INTERVAL == 2:
 242       tf.logging.info('Saving model.')
 243       saver.save(sess, FLAGS.output_dir + '/model' + str(itr))
 244 
 245     if (itr) % SUMMARY_INTERVAL:
 246       summary_writer.add_summary(summary_str, itr)
 247 
 248   tf.logging.info('Saving model.')
 249   saver.save(sess, FLAGS.output_dir + '/model')
 250   tf.logging.info('Training complete')
 251   tf.logging.flush()
 252 
 253 
 254 if __name__ == '__main__':
 255   app.run()
