  34 def block35(net, scale=1.0, activation_fn=tf.nn.relu, scope=None, reuse=None):
  35   """Builds the 35x35 resnet block."""
  36   with tf.variable_scope(scope, 'Block35', [net], reuse=reuse):
  37     with tf.variable_scope('Branch_0'):
  38       tower_conv = slim.conv2d(net, 32, 1, scope='Conv2d_1x1')
  39     with tf.variable_scope('Branch_1'):
  40       tower_conv1_0 = slim.conv2d(net, 32, 1, scope='Conv2d_0a_1x1')
  41       tower_conv1_1 = slim.conv2d(tower_conv1_0, 32, 3, scope='Conv2d_0b_3x3')
  42     with tf.variable_scope('Branch_2'):
  43       tower_conv2_0 = slim.conv2d(net, 32, 1, scope='Conv2d_0a_1x1')
  44       tower_conv2_1 = slim.conv2d(tower_conv2_0, 48, 3, scope='Conv2d_0b_3x3')
  45       tower_conv2_2 = slim.conv2d(tower_conv2_1, 64, 3, scope='Conv2d_0c_3x3')
  46     mixed = tf.concat(axis=3, values=[tower_conv, tower_conv1_1, tower_conv2_2])
  47     up = slim.conv2d(mixed, net.get_shape()[3], 1, normalizer_fn=None,
  48                      activation_fn=None, scope='Conv2d_1x1')
  49     net += scale * up
  50     if activation_fn:
  51       net = activation_fn(net)
  52   return net

  55 def block17(net, scale=1.0, activation_fn=tf.nn.relu, scope=None, reuse=None):
  56   """Builds the 17x17 resnet block."""
  57   with tf.variable_scope(scope, 'Block17', [net], reuse=reuse):
  58     with tf.variable_scope('Branch_0'):
  59       tower_conv = slim.conv2d(net, 192, 1, scope='Conv2d_1x1')
  60     with tf.variable_scope('Branch_1'):
  61       tower_conv1_0 = slim.conv2d(net, 128, 1, scope='Conv2d_0a_1x1')
  62       tower_conv1_1 = slim.conv2d(tower_conv1_0, 160, [1, 7],
  63                                   scope='Conv2d_0b_1x7')
  64       tower_conv1_2 = slim.conv2d(tower_conv1_1, 192, [7, 1],
  65                                   scope='Conv2d_0c_7x1')
  66     mixed = tf.concat(axis=3, values=[tower_conv, tower_conv1_2])
  67     up = slim.conv2d(mixed, net.get_shape()[3], 1, normalizer_fn=None,
  68                      activation_fn=None, scope='Conv2d_1x1')
  69     net += scale * up
  70     if activation_fn:
  71       net = activation_fn(net)
  72   return net

  95 def inception_resnet_v2_base(inputs,
  96                              final_endpoint='Conv2d_7b_1x1',
  97                              output_stride=16,
  98                              align_feature_maps=False,
  99                              scope=None):
 100   """Inception model from  http://arxiv.org/abs/1602.07261.
 101 
 102   Constructs an Inception Resnet v2 network from inputs to the given final
 103   endpoint. This method can construct the network up to the final inception
 104   block Conv2d_7b_1x1.
 105 
 106   Args:
 107     inputs: a tensor of size [batch_size, height, width, channels].
 108     final_endpoint: specifies the endpoint to construct the network up to. It
 109       can be one of ['Conv2d_1a_3x3', 'Conv2d_2a_3x3', 'Conv2d_2b_3x3',
 110       'MaxPool_3a_3x3', 'Conv2d_3b_1x1', 'Conv2d_4a_3x3', 'MaxPool_5a_3x3',
 111       'Mixed_5b', 'Mixed_6a', 'PreAuxLogits', 'Mixed_7a', 'Conv2d_7b_1x1']
 112     output_stride: A scalar that specifies the requested ratio of input to
 113       output spatial resolution. Only supports 8 and 16.
 114     align_feature_maps: When true, changes all the VALID paddings in the network
 115       to SAME padding so that the feature maps are aligned.
 116     scope: Optional variable_scope.
 117 
 118   Returns:
 119     tensor_out: output tensor corresponding to the final_endpoint.
 120     end_points: a set of activations for external use, for example summaries or
 121                 losses.
 122 
 123   Raises:
 124     ValueError: if final_endpoint is not set to one of the predefined values,
 125       or if the output_stride is not 8 or 16, or if the output_stride is 8 and
 126       we request an end point after 'PreAuxLogits'.
 127   """
 128   if output_stride != 8 and output_stride != 16:
 129     raise ValueError('output_stride must be 8 or 16.')
 130 
 131   padding = 'SAME' if align_feature_maps else 'VALID'
 132 
 133   end_points = {}
 134 
 135   def add_and_check_final(name, net):
 136     end_points[name] = net
 137     return name == final_endpoint
 138 
 139   with tf.variable_scope(scope, 'InceptionResnetV2', [inputs]):
 140     with slim.arg_scope([slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
 141                         stride=1, padding='SAME'):
 142       # 149 x 149 x 32
 143       net = slim.conv2d(inputs, 32, 3, stride=2, padding=padding,
 144                         scope='Conv2d_1a_3x3')
 145       if add_and_check_final('Conv2d_1a_3x3', net): return net, end_points
 146 
 147       # 147 x 147 x 32
 148       net = slim.conv2d(net, 32, 3, padding=padding,
 149                         scope='Conv2d_2a_3x3')
 150       if add_and_check_final('Conv2d_2a_3x3', net): return net, end_points
 151       # 147 x 147 x 64
 152       net = slim.conv2d(net, 64, 3, scope='Conv2d_2b_3x3')
 153       if add_and_check_final('Conv2d_2b_3x3', net): return net, end_points
 154       # 73 x 73 x 64
 155       net = slim.max_pool2d(net, 3, stride=2, padding=padding,
 156                             scope='MaxPool_3a_3x3')
 157       if add_and_check_final('MaxPool_3a_3x3', net): return net, end_points
 158       # 73 x 73 x 80
 159       net = slim.conv2d(net, 80, 1, padding=padding,
 160                         scope='Conv2d_3b_1x1')
 161       if add_and_check_final('Conv2d_3b_1x1', net): return net, end_points
 162       # 71 x 71 x 192
 163       net = slim.conv2d(net, 192, 3, padding=padding,
 164                         scope='Conv2d_4a_3x3')
 165       if add_and_check_final('Conv2d_4a_3x3', net): return net, end_points
 166       # 35 x 35 x 192
 167       net = slim.max_pool2d(net, 3, stride=2, padding=padding,
 168                             scope='MaxPool_5a_3x3')
 169       if add_and_check_final('MaxPool_5a_3x3', net): return net, end_points
 170 
 171       # 35 x 35 x 320
 172       with tf.variable_scope('Mixed_5b'):
 173         with tf.variable_scope('Branch_0'):
 174           tower_conv = slim.conv2d(net, 96, 1, scope='Conv2d_1x1')
 175         with tf.variable_scope('Branch_1'):
 176           tower_conv1_0 = slim.conv2d(net, 48, 1, scope='Conv2d_0a_1x1')
 177           tower_conv1_1 = slim.conv2d(tower_conv1_0, 64, 5,
 178                                       scope='Conv2d_0b_5x5')
 179         with tf.variable_scope('Branch_2'):
 180           tower_conv2_0 = slim.conv2d(net, 64, 1, scope='Conv2d_0a_1x1')
 181           tower_conv2_1 = slim.conv2d(tower_conv2_0, 96, 3,
 182                                       scope='Conv2d_0b_3x3')
 183           tower_conv2_2 = slim.conv2d(tower_conv2_1, 96, 3,
 184                                       scope='Conv2d_0c_3x3')
 185         with tf.variable_scope('Branch_3'):
 186           tower_pool = slim.avg_pool2d(net, 3, stride=1, padding='SAME',
 187                                        scope='AvgPool_0a_3x3')
 188           tower_pool_1 = slim.conv2d(tower_pool, 64, 1,
 189                                      scope='Conv2d_0b_1x1')
 190         net = tf.concat(
 191             [tower_conv, tower_conv1_1, tower_conv2_2, tower_pool_1], 3)
 192 
 193       if add_and_check_final('Mixed_5b', net): return net, end_points
 194       # TODO(alemi): Register intermediate endpoints
 195       net = slim.repeat(net, 10, block35, scale=0.17)
 196 
 197       # 17 x 17 x 1088 if output_stride == 8,
 198       # 33 x 33 x 1088 if output_stride == 16
 199       use_atrous = output_stride == 8
 200 
 201       with tf.variable_scope('Mixed_6a'):
 202         with tf.variable_scope('Branch_0'):
 203           tower_conv = slim.conv2d(net, 384, 3, stride=1 if use_atrous else 2,
 204                                    padding=padding,
 205                                    scope='Conv2d_1a_3x3')
 206         with tf.variable_scope('Branch_1'):
 207           tower_conv1_0 = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 208           tower_conv1_1 = slim.conv2d(tower_conv1_0, 256, 3,
 209                                       scope='Conv2d_0b_3x3')
 210           tower_conv1_2 = slim.conv2d(tower_conv1_1, 384, 3,
 211                                       stride=1 if use_atrous else 2,
 212                                       padding=padding,
 213                                       scope='Conv2d_1a_3x3')
 214         with tf.variable_scope('Branch_2'):
 215           tower_pool = slim.max_pool2d(net, 3, stride=1 if use_atrous else 2,
 216                                        padding=padding,
 217                                        scope='MaxPool_1a_3x3')
 218         net = tf.concat([tower_conv, tower_conv1_2, tower_pool], 3)
 219 
 220       if add_and_check_final('Mixed_6a', net): return net, end_points
 221 
 222       # TODO(alemi): register intermediate endpoints
 223       with slim.arg_scope([slim.conv2d], rate=2 if use_atrous else 1):
 224         net = slim.repeat(net, 20, block17, scale=0.10)
 225       if add_and_check_final('PreAuxLogits', net): return net, end_points
 226 
 227       if output_stride == 8:
 228         # TODO(gpapan): Properly support output_stride for the rest of the net.
 229         raise ValueError('output_stride==8 is only supported up to the '
 230                          'PreAuxlogits end_point for now.')
 231 
 232       # 8 x 8 x 2080
 233       with tf.variable_scope('Mixed_7a'):
 234         with tf.variable_scope('Branch_0'):
 235           tower_conv = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 236           tower_conv_1 = slim.conv2d(tower_conv, 384, 3, stride=2,
 237                                      padding=padding,
 238                                      scope='Conv2d_1a_3x3')
 239         with tf.variable_scope('Branch_1'):
 240           tower_conv1 = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 241           tower_conv1_1 = slim.conv2d(tower_conv1, 288, 3, stride=2,
 242                                       padding=padding,
 243                                       scope='Conv2d_1a_3x3')
 244         with tf.variable_scope('Branch_2'):
 245           tower_conv2 = slim.conv2d(net, 256, 1, scope='Conv2d_0a_1x1')
 246           tower_conv2_1 = slim.conv2d(tower_conv2, 288, 3,
 247                                       scope='Conv2d_0b_3x3')
 248           tower_conv2_2 = slim.conv2d(tower_conv2_1, 320, 3, stride=2,
 249                                       padding=padding,
 250                                       scope='Conv2d_1a_3x3')
 251         with tf.variable_scope('Branch_3'):
 252           tower_pool = slim.max_pool2d(net, 3, stride=2,
 253                                        padding=padding,
 254                                        scope='MaxPool_1a_3x3')
 255         net = tf.concat(
 256             [tower_conv_1, tower_conv1_1, tower_conv2_2, tower_pool], 3)
 257 
 258       if add_and_check_final('Mixed_7a', net): return net, end_points
 259 
 260       # TODO(alemi): register intermediate endpoints
 261       net = slim.repeat(net, 9, block8, scale=0.20)
 262       net = block8(net, activation_fn=None)
 263 
 264       # 8 x 8 x 1536
 265       net = slim.conv2d(net, 1536, 1, scope='Conv2d_7b_1x1')
 266       if add_and_check_final('Conv2d_7b_1x1', net): return net, end_points
 267 
 268     raise ValueError('final_endpoint (%s) not recognized', final_endpoint)

 271 def inception_resnet_v2(inputs, num_classes=1001, is_training=True,
 272                         dropout_keep_prob=0.8,
 273                         reuse=None,
 274                         scope='InceptionResnetV2',
 275                         create_aux_logits=True):
 276   """Creates the Inception Resnet V2 model.
 277 
 278   Args:
 279     inputs: a 4-D tensor of size [batch_size, height, width, 3].
 280     num_classes: number of predicted classes.
 281     is_training: whether is training or not.
 282     dropout_keep_prob: float, the fraction to keep before final layer.
 283     reuse: whether or not the network and its variables should be reused. To be
 284       able to reuse 'scope' must be given.
 285     scope: Optional variable_scope.
 286     create_aux_logits: Whether to include the auxilliary logits.
 287 
 288   Returns:
 289     logits: the logits outputs of the model.
 290     end_points: the set of end_points from the inception model.
 291   """
 292   end_points = {}
 293 
 294   with tf.variable_scope(scope, 'InceptionResnetV2', [inputs, num_classes],
 295                          reuse=reuse) as scope:
 296     with slim.arg_scope([slim.batch_norm, slim.dropout],
 297                         is_training=is_training):
 298 
 299       net, end_points = inception_resnet_v2_base(inputs, scope=scope)
 300 
 301       if create_aux_logits:
 302         with tf.variable_scope('AuxLogits'):
 303           aux = end_points['PreAuxLogits']
 304           aux = slim.avg_pool2d(aux, 5, stride=3, padding='VALID',
 305                                 scope='Conv2d_1a_3x3')
 306           aux = slim.conv2d(aux, 128, 1, scope='Conv2d_1b_1x1')
 307           aux = slim.conv2d(aux, 768, aux.get_shape()[1:3],
 308                             padding='VALID', scope='Conv2d_2a_5x5')
 309           aux = slim.flatten(aux)
 310           aux = slim.fully_connected(aux, num_classes, activation_fn=None,
 311                                      scope='Logits')
 312           end_points['AuxLogits'] = aux
 313 
 314       with tf.variable_scope('Logits'):
 315         net = slim.avg_pool2d(net, net.get_shape()[1:3], padding='VALID',
 316                               scope='AvgPool_1a_8x8')
 317         net = slim.flatten(net)
 318 
 319         net = slim.dropout(net, dropout_keep_prob, is_training=is_training,
 320                            scope='Dropout')
 321 
 322         end_points['PreLogitsFlatten'] = net
 323         logits = slim.fully_connected(net, num_classes, activation_fn=None,
 324                                       scope='Logits')
 325         end_points['Logits'] = logits
 326         end_points['Predictions'] = tf.nn.softmax(logits, name='Predictions')
 327 
 328     return logits, end_points
 329 inception_resnet_v2.default_image_size = 299

 332 def inception_resnet_v2_arg_scope(weight_decay=0.00004,
 333                                   batch_norm_decay=0.9997,
 334                                   batch_norm_epsilon=0.001):
 335   """Returns the scope with the default parameters for inception_resnet_v2.
 336 
 337   Args:
 338     weight_decay: the weight decay for weights variables.
 339     batch_norm_decay: decay for the moving average of batch_norm momentums.
 340     batch_norm_epsilon: small float added to variance to avoid dividing by zero.
 341 
 342   Returns:
 343     a arg_scope with the parameters needed for inception_resnet_v2.
 344   """
 345   # Set weight_decay for weights in conv2d and fully_connected layers.
 346   with slim.arg_scope([slim.conv2d, slim.fully_connected],
 347                       weights_regularizer=slim.l2_regularizer(weight_decay),
 348                       biases_regularizer=slim.l2_regularizer(weight_decay)):
 349 
 350     batch_norm_params = {
 351         'decay': batch_norm_decay,
 352         'epsilon': batch_norm_epsilon,
 353     }
 354     # Set activation_fn and parameters for batch_norm.
 355     with slim.arg_scope([slim.conv2d], activation_fn=tf.nn.relu,
 356                         normalizer_fn=slim.batch_norm,
 357                         normalizer_params=batch_norm_params) as scope:
 358       return scope
