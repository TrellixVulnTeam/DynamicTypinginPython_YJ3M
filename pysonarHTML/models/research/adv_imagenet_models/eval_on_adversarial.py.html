  83 def preprocess_for_eval(image, height, width,
  84                         central_fraction=0.875, scope=None):
  85   """Prepare one image for evaluation.
  86 
  87   If height and width are specified it would output an image with that size by
  88   applying resize_bilinear.
  89   If central_fraction is specified it would crop the central fraction of the
  90   input image.
  91 
  92   Args:
  93     image: 3-D Tensor of image. If dtype is tf.float32 then the range should be
  94       [0, 1], otherwise it would converted to tf.float32 assuming that the range
  95       is [0, MAX], where MAX is largest positive representable number for
  96       int(8/16/32) data type (see `tf.image.convert_image_dtype` for details)
  97     height: integer
  98     width: integer
  99     central_fraction: Optional Float, fraction of the image to crop.
 100     scope: Optional scope for name_scope.
 101   Returns:
 102     3-D float Tensor of prepared image.
 103   """
 104   with tf.name_scope(scope, 'eval_image', [image, height, width]):
 105     if image.dtype != tf.float32:
 106       image = tf.image.convert_image_dtype(image, dtype=tf.float32)
 107     # Crop the central region of the image with an area containing 87.5% of
 108     # the original image.
 109     if central_fraction:
 110       image = tf.image.central_crop(image, central_fraction=central_fraction)
 111 
 112     if height and width:
 113       # Resize the image to the specified height and width.
 114       image = tf.expand_dims(image, 0)
 115       image = tf.image.resize_bilinear(image, [height, width],
 116                                        align_corners=False)
 117       image = tf.squeeze(image, [0])
 118     image = tf.subtract(image, 0.5)
 119     image = tf.multiply(image, 2.0)
 120     return image

 123 def create_model(x, reuse=None):
 124   """Create model graph.
 125 
 126   Args:
 127     x: input images
 128     reuse: reuse parameter which will be passed to underlying variable scopes.
 129       Should be None first call and True every subsequent call.
 130 
 131   Returns:
 132     (logits, end_points) - tuple of model logits and enpoints
 133 
 134   Raises:
 135     ValueError: if model type specified by --model_name flag is invalid.
 136   """
 137   if FLAGS.model_name == 'inception_v3':
 138     with slim.arg_scope(inception.inception_v3_arg_scope()):
 139       return inception.inception_v3(
 140           x, num_classes=NUM_CLASSES, is_training=False, reuse=reuse)
 141   elif FLAGS.model_name == 'inception_resnet_v2':
 142     with slim.arg_scope(inception_resnet_v2.inception_resnet_v2_arg_scope()):
 143       return inception_resnet_v2.inception_resnet_v2(
 144           x, num_classes=NUM_CLASSES, is_training=False, reuse=reuse)
 145   else:
 146     raise ValueError('Invalid model name: %s' % (FLAGS.model_name))

 149 def step_target_class_adversarial_images(x, eps, one_hot_target_class):
 150   """Base code for one step towards target class methods.
 151 
 152   Args:
 153     x: source images
 154     eps: size of adversarial perturbation
 155     one_hot_target_class: one hot encoded target classes for all images
 156 
 157   Returns:
 158     tensor with adversarial images
 159   """
 160   logits, end_points = create_model(x, reuse=True)
 161   cross_entropy = tf.losses.softmax_cross_entropy(one_hot_target_class,
 162                                                   logits,
 163                                                   label_smoothing=0.1,
 164                                                   weights=1.0)
 165   cross_entropy += tf.losses.softmax_cross_entropy(one_hot_target_class,
 166                                                    end_points['AuxLogits'],
 167                                                    label_smoothing=0.1,
 168                                                    weights=0.4)
 169   x_adv = x - eps * tf.sign(tf.gradients(cross_entropy, x)[0])
 170   x_adv = tf.clip_by_value(x_adv, -1.0, 1.0)
 171   return tf.stop_gradient(x_adv)

 174 def stepll_adversarial_images(x, eps):
 175   """One step towards least likely class (Step L.L.) adversarial examples.
 176 
 177   This method is an alternative to FGSM which does not use true classes.
 178   Method is described in the "Adversarial Machine Learning at Scale" paper,
 179   https://arxiv.org/abs/1611.01236
 180 
 181   Args:
 182     x: source images
 183     eps: size of adversarial perturbation
 184 
 185   Returns:
 186     adversarial images
 187   """
 188   logits, _ = create_model(x, reuse=True)
 189   least_likely_class = tf.argmin(logits, 1)
 190   one_hot_ll_class = tf.one_hot(least_likely_class, NUM_CLASSES)
 191   return step_target_class_adversarial_images(x, eps, one_hot_ll_class)

 194 def stepllnoise_adversarial_images(x, eps):
 195   """Step L.L. with noise method.
 196 
 197   This is an imporvement of Step L.L. method. This method is better against
 198   adversarially trained models which learn to mask gradient.
 199   Method is described in the section "New randomized one shot attack" of
 200   "Ensemble Adversarial Training: Attacks and Defenses" paper,
 201   https://arxiv.org/abs/1705.07204
 202 
 203   Args:
 204     x: source images
 205     eps: size of adversarial perturbation
 206 
 207   Returns:
 208     adversarial images
 209   """
 210   logits, _ = create_model(x, reuse=True)
 211   least_likely_class = tf.argmin(logits, 1)
 212   one_hot_ll_class = tf.one_hot(least_likely_class, NUM_CLASSES)
 213   x_noise = x + eps / 2 * tf.sign(tf.random_normal(x.shape))
 214   return step_target_class_adversarial_images(x_noise, eps / 2,
 215                                               one_hot_ll_class)

 218 def get_input_images(dataset_images):
 219   """Gets input images for the evaluation.
 220 
 221   Args:
 222     dataset_images: tensor with dataset images
 223 
 224   Returns:
 225     tensor with input images, which is either dataset images or adversarial
 226     images.
 227 
 228   Raises:
 229     ValueError: if adversarial method specified by --adversarial_method flag
 230       is invalid.
 231   """
 232   # adversarial_eps defines max difference of values of pixels if
 233   # pixels are in range [0, 255]. However values of dataset pixels are
 234   # in range [-1, 1], so converting epsilon.
 235   eps = FLAGS.adversarial_eps / 255 * 2.0
 236 
 237   if FLAGS.adversarial_method == 'stepll':
 238     return stepll_adversarial_images(dataset_images, eps)
 239   elif FLAGS.adversarial_method == 'stepllnoise':
 240     return stepllnoise_adversarial_images(dataset_images, eps)
 241   elif FLAGS.adversarial_method == 'none':
 242     return dataset_images
 243   else:
 244     raise ValueError('Invalid adversarial method: %s'
 245                      % (FLAGS.adversarial_method))

 248 def main(_):
 249   if not FLAGS.dataset_dir:
 250     raise ValueError('You must supply the dataset directory with --dataset_dir')
 251 
 252   tf.logging.set_verbosity(tf.logging.INFO)
 253   with tf.Graph().as_default():
 254     tf_global_step = tf.train.get_or_create_global_step()
 255 
 256     ###################
 257     # Prepare dataset #
 258     ###################
 259     dataset = imagenet.get_split(FLAGS.split_name, FLAGS.dataset_dir)
 260     provider = slim.dataset_data_provider.DatasetDataProvider(
 261         dataset,
 262         shuffle=False,
 263         common_queue_capacity=2 * FLAGS.batch_size,
 264         common_queue_min=FLAGS.batch_size)
 265     [dataset_image, label] = provider.get(['image', 'label'])
 266     dataset_image = preprocess_for_eval(dataset_image, IMAGE_SIZE, IMAGE_SIZE)
 267     dataset_images, labels = tf.train.batch(
 268         [dataset_image, label],
 269         batch_size=FLAGS.batch_size,
 270         num_threads=FLAGS.num_preprocessing_threads,
 271         capacity=5 * FLAGS.batch_size)
 272 
 273     ########################################
 274     # Define the model and input exampeles #
 275     ########################################
 276     create_model(tf.placeholder(tf.float32, shape=dataset_images.shape))
 277     input_images = get_input_images(dataset_images)
 278     logits, _ = create_model(input_images, reuse=True)
 279 
 280     if FLAGS.moving_average_decay > 0:
 281       variable_averages = tf.train.ExponentialMovingAverage(
 282           FLAGS.moving_average_decay, tf_global_step)
 283       variables_to_restore = variable_averages.variables_to_restore(
 284           slim.get_model_variables())
 285       variables_to_restore[tf_global_step.op.name] = tf_global_step
 286     else:
 287       variables_to_restore = slim.get_variables_to_restore()
 288 
 289     ######################
 290     # Define the metrics #
 291     ######################
 292     predictions = tf.argmax(logits, 1)
 293     labels = tf.squeeze(labels)
 294     names_to_values, names_to_updates = slim.metrics.aggregate_metric_map({
 295         'Accuracy': slim.metrics.streaming_accuracy(predictions, labels),
 296         'Recall_5': slim.metrics.streaming_sparse_recall_at_k(
 297             logits, tf.reshape(labels, [-1, 1]), 5),
 298     })
 299 
 300     ######################
 301     # Run evaluation     #
 302     ######################
 303     if FLAGS.max_num_batches:
 304       num_batches = FLAGS.max_num_batches
 305     else:
 306       # This ensures that we make a single pass over all of the data.
 307       num_batches = math.ceil(dataset.num_samples / float(FLAGS.batch_size))
 308 
 309     if tf.gfile.IsDirectory(FLAGS.checkpoint_path):
 310       checkpoint_path = tf.train.latest_checkpoint(FLAGS.checkpoint_path)
 311     else:
 312       checkpoint_path = FLAGS.checkpoint_path
 313 
 314     tf.logging.info('Evaluating %s' % checkpoint_path)
 315 
 316     top1_accuracy, top5_accuracy = slim.evaluation.evaluate_once(
 317         master=FLAGS.master,
 318         checkpoint_path=checkpoint_path,
 319         logdir=None,
 320         summary_op=None,
 321         num_evals=num_batches,
 322         eval_op=list(names_to_updates.values()),
 323         final_op=[names_to_values['Accuracy'], names_to_values['Recall_5']],
 324         variables_to_restore=variables_to_restore)
 325 
 326     print('Top1 Accuracy: ', top1_accuracy)
 327     print('Top5 Accuracy: ', top5_accuracy)
 328 
 329 
 330 if __name__ == '__main__':
 331   tf.app.run()
