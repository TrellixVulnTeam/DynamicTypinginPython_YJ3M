  28 class Inference(object):
  29   def __init__(self, config, inputs, pretrained_embeddings, tasks):
  30     with tf.variable_scope('encoder'):
  31       self.encoder = encoder.Encoder(config, inputs, pretrained_embeddings)
  32     self.modules = {}
  33     for task in tasks:
  34       with tf.variable_scope(task.name):
  35         self.modules[task.name] = task.get_module(inputs, self.encoder)
  36 
  37 
  38 class Model(object):
  39   def __init__(self, config, pretrained_embeddings, tasks):
  40     self._config = config
  41     self._tasks = tasks
  42 
  43     self._global_step, self._optimizer = self._get_optimizer()
  44     self._inputs = shared_inputs.Inputs(config)
  45     with tf.variable_scope('model', reuse=tf.AUTO_REUSE) as scope:
  46       inference = Inference(config, self._inputs, pretrained_embeddings,
  47                             tasks)
  48       self._trainer = inference
  49       self._tester = inference
  50       self._teacher = inference
  51       if config.ema_test or config.ema_teacher:
  52         ema = tf.train.ExponentialMovingAverage(config.ema_decay)
  53         model_vars = tf.get_collection("trainable_variables", "model")
  54         ema_op = ema.apply(model_vars)
  55         tf.add_to_collection(tf.GraphKeys.UPDATE_OPS, ema_op)
  56 
  57         def ema_getter(getter, name, *args, **kwargs):
  58           var = getter(name, *args, **kwargs)
  59           return ema.average(var)
  60 
  61         scope.set_custom_getter(ema_getter)
  62         inference_ema = Inference(
  63             config, self._inputs, pretrained_embeddings, tasks)
  64         if config.ema_teacher:
  65           self._teacher = inference_ema
  66         if config.ema_test:
  67           self._tester = inference_ema
  68 
  69     self._unlabeled_loss = self._get_consistency_loss(tasks)
  70     self._unlabeled_train_op = self._get_train_op(self._unlabeled_loss)
  71     self._labeled_train_ops = {}
  72     for task in self._tasks:
  73       task_loss = self._trainer.modules[task.name].supervised_loss
  74       self._labeled_train_ops[task.name] = self._get_train_op(task_loss)
  75 
  76   def _get_consistency_loss(self, tasks):
  77     return sum([self._trainer.modules[task.name].unsupervised_loss
  78                 for task in tasks])
  79 
  80   def _get_optimizer(self):
  81     global_step = tf.get_variable('global_step', initializer=0, trainable=False)
  82     warm_up_multiplier = (tf.minimum(tf.to_float(global_step),
  83                                      self._config.warm_up_steps)
  84                           / self._config.warm_up_steps)
  85     decay_multiplier = 1.0 / (1 + self._config.lr_decay *
  86                               tf.sqrt(tf.to_float(global_step)))
  87     lr = self._config.lr * warm_up_multiplier * decay_multiplier
  88     optimizer = tf.train.MomentumOptimizer(lr, self._config.momentum)
  89     return global_step, optimizer
  90 
  91   def _get_train_op(self, loss):
  92     grads, vs = zip(*self._optimizer.compute_gradients(loss))
  93     grads, _ = tf.clip_by_global_norm(grads, self._config.grad_clip)
  94     update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS)
  95     with tf.control_dependencies(update_ops):
  96       return self._optimizer.apply_gradients(
  97           zip(grads, vs), global_step=self._global_step)
  98 
  99   def _create_feed_dict(self, mb, model, is_training=True):
 100     feed = self._inputs.create_feed_dict(mb, is_training)
 101     if mb.task_name in model.modules:
 102       model.modules[mb.task_name].update_feed_dict(feed, mb)
 103     else:
 104       for module in model.modules.values():
 105         module.update_feed_dict(feed, mb)
 106     return feed
 107 
 108   def train_unlabeled(self, sess, mb):
 109     return sess.run([self._unlabeled_train_op, self._unlabeled_loss],
 110                     feed_dict=self._create_feed_dict(mb, self._trainer))[1]
 111 
 112   def train_labeled(self, sess, mb):
 113     return sess.run([self._labeled_train_ops[mb.task_name],
 114                      self._trainer.modules[mb.task_name].supervised_loss,],
 115                     feed_dict=self._create_feed_dict(mb, self._trainer))[1]
 116 
 117   def run_teacher(self, sess, mb):
 118     result = sess.run({task.name: self._teacher.modules[task.name].probs
 119                        for task in self._tasks},
 120                       feed_dict=self._create_feed_dict(mb, self._teacher,
 121                                                        False))
 122     for task_name, probs in result.iteritems():
 123       mb.teacher_predictions[task_name] = probs.astype('float16')
 124 
 125   def test(self, sess, mb):
 126     return sess.run(
 127         [self._tester.modules[mb.task_name].supervised_loss,
 128          self._tester.modules[mb.task_name].preds],
 129         feed_dict=self._create_feed_dict(mb, self._tester, False))
 130 
 131   def get_global_step(self, sess):
 132     return sess.run(self._global_step)
