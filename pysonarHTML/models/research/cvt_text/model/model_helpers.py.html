  25 def project(input_layers, size, name='projection'):
  26   return tf.add_n([tf.layers.dense(layer, size, name=name + '_' + str(i))
  27                    for i, layer in enumerate(input_layers)])

  30 def lstm_cell(cell_size, keep_prob, num_proj):
  31   return tf.contrib.rnn.DropoutWrapper(
  32       tf.contrib.rnn.LSTMCell(cell_size, num_proj=min(cell_size, num_proj)),
  33       output_keep_prob=keep_prob)
  34 
  35 
  36 def multi_lstm_cell(cell_sizes, keep_prob, num_proj):
  37   return tf.contrib.rnn.MultiRNNCell([lstm_cell(cell_size, keep_prob, num_proj)
  38                                       for cell_size in cell_sizes])

  41 def masked_ce_loss(logits, labels, mask, sparse=False, roll_direction=0):
  42   if roll_direction != 0:
  43     labels = _roll(labels, roll_direction, sparse)
  44     mask *= _roll(mask, roll_direction, True)
  45   ce = ((tf.nn.sparse_softmax_cross_entropy_with_logits if sparse
  46          else tf.nn.softmax_cross_entropy_with_logits_v2)
  47         (logits=logits, labels=labels))
  48   return tf.reduce_sum(mask * ce) / tf.to_float(tf.reduce_sum(mask))
  49 
  50 
  51 def _roll(arr, direction, sparse=False):
  52   if sparse:
  53     return tf.concat([arr[:, direction:], arr[:, :direction]], axis=1)
  54   return tf.concat([arr[:, direction:, :], arr[:, :direction, :]], axis=1)
