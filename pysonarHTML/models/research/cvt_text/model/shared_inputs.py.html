  16 """Placeholders for non-task-specific model inputs."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 
  24 
  25 class Inputs(object):
  26   def __init__(self, config):
  27     self._config = config
  28     self.keep_prob = tf.placeholder(tf.float32, name='keep_prob')
  29     self.label_smoothing = tf.placeholder(tf.float32, name='label_smoothing')
  30     self.lengths = tf.placeholder(tf.int32, shape=[None], name='lengths')
  31     self.mask = tf.placeholder(tf.float32, [None, None], name='mask')
  32     self.words = tf.placeholder(tf.int32, shape=[None, None], name='words')
  33     self.chars = tf.placeholder(tf.int32, shape=[None, None, None],
  34                                 name='chars')
  35 
  36   def create_feed_dict(self, mb, is_training):
  37     cvt = mb.task_name == 'unlabeled'
  38     return {
  39         self.keep_prob: 1.0 if not is_training else
  40                         (self._config.unlabeled_keep_prob if cvt else
  41                          self._config.labeled_keep_prob),
  42         self.label_smoothing: self._config.label_smoothing
  43                               if (is_training and not cvt) else 0.0,
  44         self.lengths: mb.lengths,
  45         self.words: mb.words,
  46         self.chars: mb.chars,
  47         self.mask: mb.mask.astype('float32')
  48     }
