  16 """CNN-BiLSTM sentence encoder."""

  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 from base import embeddings
  24 from model import model_helpers
  25 
  26 
  27 class Encoder(object):
  28   def __init__(self, config, inputs, pretrained_embeddings):
  29     self._config = config
  30     self._inputs = inputs
  31 
  32     self.word_reprs = self._get_word_reprs(pretrained_embeddings)
  33     self.uni_fw, self.uni_bw = self._get_unidirectional_reprs(self.word_reprs)
  34     self.uni_reprs = tf.concat([self.uni_fw, self.uni_bw], axis=-1)
  35     self.bi_fw, self.bi_bw, self.bi_reprs = self._get_bidirectional_reprs(
  36         self.uni_reprs)
  37 
  38   def _get_word_reprs(self, pretrained_embeddings):
  39     with tf.variable_scope('word_embeddings'):
  40       word_embedding_matrix = tf.get_variable(
  41           'word_embedding_matrix', initializer=pretrained_embeddings)
  42       word_embeddings = tf.nn.embedding_lookup(
  43           word_embedding_matrix, self._inputs.words)
  44       word_embeddings = tf.nn.dropout(word_embeddings, self._inputs.keep_prob)
  45       word_embeddings *= tf.get_variable('emb_scale', initializer=1.0)
  46 
  47     if not self._config.use_chars:
  48       return word_embeddings
  49 
  50     with tf.variable_scope('char_embeddings'):
  51       char_embedding_matrix = tf.get_variable(
  52           'char_embeddings',
  53           shape=[embeddings.NUM_CHARS, self._config.char_embedding_size])
  54       char_embeddings = tf.nn.embedding_lookup(char_embedding_matrix,
  55                                                self._inputs.chars)
  56       shape = tf.shape(char_embeddings)
  57       char_embeddings = tf.reshape(
  58           char_embeddings,
  59           shape=[-1, shape[-2], self._config.char_embedding_size])
  60       char_reprs = []
  61       for filter_width in self._config.char_cnn_filter_widths:
  62         conv = tf.layers.conv1d(
  63             char_embeddings, self._config.char_cnn_n_filters, filter_width)
  64         conv = tf.nn.relu(conv)
  65         conv = tf.nn.dropout(tf.reduce_max(conv, axis=1),
  66                              self._inputs.keep_prob)
  67         conv = tf.reshape(conv, shape=[-1, shape[1],
  68                                        self._config.char_cnn_n_filters])
  69         char_reprs.append(conv)
  70       return tf.concat([word_embeddings] + char_reprs, axis=-1)
  71 
  72   def _get_unidirectional_reprs(self, word_reprs):
  73     with tf.variable_scope('unidirectional_reprs'):
  74       word_lstm_input_size = (
  75           self._config.word_embedding_size if not self._config.use_chars else
  76           (self._config.word_embedding_size +
  77            len(self._config.char_cnn_filter_widths)
  78            * self._config.char_cnn_n_filters))
  79       word_reprs.set_shape([None, None, word_lstm_input_size])
  80       (outputs_fw, outputs_bw), _ = tf.nn.bidirectional_dynamic_rnn(
  81           model_helpers.multi_lstm_cell(self._config.unidirectional_sizes,
  82                                         self._inputs.keep_prob,
  83                                         self._config.projection_size),
  84           model_helpers.multi_lstm_cell(self._config.unidirectional_sizes,
  85                                         self._inputs.keep_prob,
  86                                         self._config.projection_size),
  87           word_reprs,
  88           dtype=tf.float32,
  89           sequence_length=self._inputs.lengths,
  90           scope='unilstm'
  91       )
  92       return outputs_fw, outputs_bw
  93 
  94   def _get_bidirectional_reprs(self, uni_reprs):
  95     with tf.variable_scope('bidirectional_reprs'):
  96       current_outputs = uni_reprs
  97       outputs_fw, outputs_bw = None, None
  98       for size in self._config.bidirectional_sizes:
  99         (outputs_fw, outputs_bw), _ = tf.nn.bidirectional_dynamic_rnn(
 100             model_helpers.lstm_cell(size, self._inputs.keep_prob,
 101                                     self._config.projection_size),
 102             model_helpers.lstm_cell(size, self._inputs.keep_prob,
 103                                     self._config.projection_size),
 104             current_outputs,
 105             dtype=tf.float32,
 106             sequence_length=self._inputs.lengths,
 107             scope='bilstm'
 108         )
 109         current_outputs = tf.concat([outputs_fw, outputs_bw], axis=-1)
 110       return outputs_fw, outputs_bw, current_outputs
