  22 import numpy as np
  23 import tensorflow as tf

  25 from corpus_processing import minibatching
  26 from model import model_helpers
  27 from model import task_module

  30 class TaggingModule(task_module.SemiSupervisedModule):
  31   def __init__(self, config, task_name, n_classes, inputs,
  32                encoder):
  33     super(TaggingModule, self).__init__()
  34     self.task_name = task_name
  35     self.n_classes = n_classes
  36     self.labels = labels = tf.placeholder(tf.float32, [None, None, None],
  37                                           name=task_name + '_labels')
  38 
  39     class PredictionModule(object):
  40       def __init__(self, name, input_reprs, roll_direction=0, activate=True):
  41         self.name = name
  42         with tf.variable_scope(name + '/predictions'):
  43           projected = model_helpers.project(input_reprs, config.projection_size)
  44           if activate:
  45             projected = tf.nn.relu(projected)
  46           self.logits = tf.layers.dense(projected, n_classes, name='predict')
  47 
  48         targets = labels
  49         targets *= (1 - inputs.label_smoothing)
  50         targets += inputs.label_smoothing / n_classes
  51         self.loss = model_helpers.masked_ce_loss(
  52             self.logits, targets, inputs.mask, roll_direction=roll_direction)
  53 
  54     primary = PredictionModule('primary',
  55                                ([encoder.uni_reprs, encoder.bi_reprs]))
  56     ps = [
  57         PredictionModule('full', ([encoder.uni_reprs, encoder.bi_reprs]),
  58                          activate=False),
  59         PredictionModule('forwards', [encoder.uni_fw]),
  60         PredictionModule('backwards', [encoder.uni_bw]),
  61         PredictionModule('future', [encoder.uni_fw], roll_direction=1),
  62         PredictionModule('past', [encoder.uni_bw], roll_direction=-1),
  63     ]
  64 
  65     self.unsupervised_loss = sum(p.loss for p in ps)
  66     self.supervised_loss = primary.loss
  67     self.probs = tf.nn.softmax(primary.logits)
  68     self.preds = tf.argmax(primary.logits, axis=-1)
  69 
  70   def update_feed_dict(self, feed, mb):
  71     if self.task_name in mb.teacher_predictions:
  72       feed[self.labels] = mb.teacher_predictions[self.task_name]
  73     elif mb.task_name != 'unlabeled':
  74       labels = minibatching.build_array(
  75           [[0] + e.labels + [0] for e in mb.examples])
  76       feed[self.labels] = np.eye(self.n_classes)[labels]
