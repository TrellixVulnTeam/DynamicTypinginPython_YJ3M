  28 class AccuracyScorer(word_level_scorer.WordLevelScorer):
  29   def __init__(self, auto_fail_label=None):
  30     super(AccuracyScorer, self).__init__()
  31     self._auto_fail_label = auto_fail_label
  32 
  33   def _get_results(self):
  34     correct, count = 0, 0
  35     for example, preds in zip(self._examples, self._preds):
  36       for y_true, y_pred in zip(example.labels, preds):
  37         count += 1
  38         correct += (1 if y_pred == y_true and y_true != self._auto_fail_label
  39                     else 0)
  40     return [
  41         ("accuracy", 100.0 * correct / count),
  42         ("loss", self.get_loss())
  43     ]

  46 class F1Scorer(word_level_scorer.WordLevelScorer):
  47   __metaclass__ = abc.ABCMeta
  48 
  49   def __init__(self):
  50     super(F1Scorer, self).__init__()
  51     self._n_correct, self._n_predicted, self._n_gold = 0, 0, 0
  52 
  53   def _get_results(self):
  54     if self._n_correct == 0:
  55       p, r, f1 = 0, 0, 0
  56     else:
  57       p = 100.0 * self._n_correct / self._n_predicted
  58       r = 100.0 * self._n_correct / self._n_gold
  59       f1 = 2 * p * r / (p + r)
  60     return [
  61         ("precision", p),
  62         ("recall", r),
  63         ("f1", f1),
  64         ("loss", self.get_loss()),
  65     ]

  68 class EntityLevelF1Scorer(F1Scorer):
  69   def __init__(self, label_mapping):
  70     super(EntityLevelF1Scorer, self).__init__()
  71     self._inv_label_mapping = {v: k for k, v in label_mapping.iteritems()}
  72 
  73   def _get_results(self):
  74     self._n_correct, self._n_predicted, self._n_gold = 0, 0, 0
  75     for example, preds in zip(self._examples, self._preds):
  76       sent_spans = set(tagging_utils.get_span_labels(
  77           example.labels, self._inv_label_mapping))
  78       span_preds = set(tagging_utils.get_span_labels(
  79           preds, self._inv_label_mapping))
  80       self._n_correct += len(sent_spans & span_preds)
  81       self._n_gold += len(sent_spans)
  82       self._n_predicted += len(span_preds)
  83     return super(EntityLevelF1Scorer, self)._get_results()
