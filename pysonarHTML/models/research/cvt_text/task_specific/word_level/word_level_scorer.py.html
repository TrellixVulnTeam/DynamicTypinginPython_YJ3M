  16 """Base class for word-level scorers."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import abc
  23 
  24 from corpus_processing import scorer
  25 
  26 
  27 class WordLevelScorer(scorer.Scorer):
  28   __metaclass__ = abc.ABCMeta
  29 
  30   def __init__(self):
  31     super(WordLevelScorer, self).__init__()
  32     self._total_loss = 0
  33     self._total_words = 0
  34     self._examples = []
  35     self._preds = []
  36 
  37   def update(self, examples, predictions, loss):
  38     super(WordLevelScorer, self).update(examples, predictions, loss)
  39     n_words = 0
  40     for example, preds in zip(examples, predictions):
  41       self._examples.append(example)
  42       self._preds.append(list(preds)[1:len(example.words) - 1])
  43       n_words += len(example.words) - 2
  44     self._total_loss += loss * n_words
  45     self._total_words += n_words
  46 
  47   def get_loss(self):
  48     return self._total_loss / max(1, self._total_words)
