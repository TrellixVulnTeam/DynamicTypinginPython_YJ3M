  16 """Dependency parsing evaluation (computes UAS/LAS)."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 from task_specific.word_level import word_level_scorer
  23 
  24 
  25 class DepparseScorer(word_level_scorer.WordLevelScorer):
  26   def __init__(self, n_relations, punctuation):
  27     super(DepparseScorer, self).__init__()
  28     self._n_relations = n_relations
  29     self._punctuation = punctuation if punctuation else None
  30 
  31   def _get_results(self):
  32     correct_unlabeled, correct_labeled, count = 0, 0, 0
  33     for example, preds in zip(self._examples, self._preds):
  34       for w, y_true, y_pred in zip(example.words[1:-1], example.labels, preds):
  35         if w in self._punctuation:
  36           continue
  37         count += 1
  38         correct_labeled += (1 if y_pred == y_true else 0)
  39         correct_unlabeled += (1 if int(y_pred // self._n_relations) ==
  40                               int(y_true // self._n_relations) else 0)
  41     return [
  42         ("las", 100.0 * correct_labeled / count),
  43         ("uas", 100.0 * correct_unlabeled / count),
  44         ("loss", self.get_loss()),
  45     ]
