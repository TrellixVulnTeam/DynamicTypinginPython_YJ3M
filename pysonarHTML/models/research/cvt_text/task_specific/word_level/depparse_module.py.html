  30 class DepparseModule(task_module.SemiSupervisedModule):
  31   def __init__(self, config, task_name, n_classes, inputs, encoder):
  32     super(DepparseModule, self).__init__()
  33 
  34     self.task_name = task_name
  35     self.n_classes = n_classes
  36     self.labels = labels = tf.placeholder(tf.float32, [None, None, None],
  37                                           name=task_name + '_labels')
  38 
  39     class PredictionModule(object):
  40       def __init__(self, name, dep_reprs, head_reprs, roll_direction=0):
  41         self.name = name
  42         with tf.variable_scope(name + '/predictions'):
  43           # apply hidden layers to the input representations
  44           arc_dep_hidden = model_helpers.project(
  45               dep_reprs, config.projection_size, 'arc_dep_hidden')
  46           arc_head_hidden = model_helpers.project(
  47               head_reprs, config.projection_size, 'arc_head_hidden')
  48           arc_dep_hidden = tf.nn.relu(arc_dep_hidden)
  49           arc_head_hidden = tf.nn.relu(arc_head_hidden)
  50           arc_head_hidden = tf.nn.dropout(arc_head_hidden, inputs.keep_prob)
  51           arc_dep_hidden = tf.nn.dropout(arc_dep_hidden, inputs.keep_prob)
  52 
  53           # bilinear classifier excluding the final dot product
  54           arc_head = tf.layers.dense(
  55               arc_head_hidden, config.depparse_projection_size, name='arc_head')
  56           W = tf.get_variable('shared_W',
  57                               shape=[config.projection_size, n_classes,
  58                                      config.depparse_projection_size])
  59           Wr = tf.get_variable('relation_specific_W',
  60                                shape=[config.projection_size,
  61                                       config.depparse_projection_size])
  62           Wr_proj = tf.tile(tf.expand_dims(Wr, axis=-2), [1, n_classes, 1])
  63           W += Wr_proj
  64           arc_dep = tf.tensordot(arc_dep_hidden, W, axes=[[-1], [0]])
  65           shape = tf.shape(arc_dep)
  66           arc_dep = tf.reshape(arc_dep,
  67                                [shape[0], -1, config.depparse_projection_size])
  68 
  69           # apply the transformer scaling trick to prevent dot products from
  70           # getting too large (possibly not necessary)
  71           scale = np.power(
  72               config.depparse_projection_size, 0.25).astype('float32')
  73           scale = tf.get_variable('scale', initializer=scale, dtype=tf.float32)
  74           arc_dep /= scale
  75           arc_head /= scale
  76 
  77           # compute the scores for each candidate arc
  78           word_scores = tf.matmul(arc_head, arc_dep, transpose_b=True)
  79           root_scores = tf.layers.dense(arc_head, n_classes, name='root_score')
  80           arc_scores = tf.concat([root_scores, word_scores], axis=-1)
  81 
  82           # disallow the model from making impossible predictions
  83           mask = inputs.mask
  84           mask_shape = tf.shape(mask)
  85           mask = tf.tile(tf.expand_dims(mask, -1), [1, 1, n_classes])
  86           mask = tf.reshape(mask, [-1, mask_shape[1] * n_classes])
  87           mask = tf.concat([tf.ones((mask_shape[0], 1)),
  88                             tf.zeros((mask_shape[0], n_classes - 1)), mask],
  89                            axis=1)
  90           mask = tf.tile(tf.expand_dims(mask, 1), [1, mask_shape[1], 1])
  91           arc_scores += (mask - 1) * 100.0
  92 
  93           self.logits = arc_scores
  94           self.loss = model_helpers.masked_ce_loss(
  95               self.logits, labels, inputs.mask,
  96               roll_direction=roll_direction)
  97 
  98     primary = PredictionModule(
  99         'primary',
 100         [encoder.uni_reprs, encoder.bi_reprs],
 101         [encoder.uni_reprs, encoder.bi_reprs])
 102     ps = [
 103         PredictionModule(
 104             'full',
 105             [encoder.uni_reprs, encoder.bi_reprs],
 106             [encoder.uni_reprs, encoder.bi_reprs]),
 107         PredictionModule('fw_fw', [encoder.uni_fw], [encoder.uni_fw]),
 108         PredictionModule('fw_bw', [encoder.uni_fw], [encoder.uni_bw]),
 109         PredictionModule('bw_fw', [encoder.uni_bw], [encoder.uni_fw]),
 110         PredictionModule('bw_bw', [encoder.uni_bw], [encoder.uni_bw]),
 111     ]
 112 
 113     self.unsupervised_loss = sum(p.loss for p in ps)
 114     self.supervised_loss = primary.loss
 115     self.probs = tf.nn.softmax(primary.logits)
 116     self.preds = tf.argmax(primary.logits, axis=-1)
 117 
 118   def update_feed_dict(self, feed, mb):
 119     if self.task_name in mb.teacher_predictions:
 120       feed[self.labels] = mb.teacher_predictions[self.task_name]
 121     elif mb.task_name != 'unlabeled':
 122       labels = minibatching.build_array(
 123           [[0] + e.labels + [0] for e in mb.examples])
 124       feed[self.labels] = np.eye(
 125           (1 + mb.words.shape[1]) * self.n_classes)[labels]
