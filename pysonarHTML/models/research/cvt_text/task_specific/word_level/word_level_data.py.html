  34 class TaggedDataLoader(object):
  35   def __init__(self, config, name, is_token_level):
  36     self._config = config
  37     self._task_name = name
  38     self._raw_data_path = os.path.join(config.raw_data_topdir, name)
  39     self._is_token_level = is_token_level
  40     self.label_mapping_path = os.path.join(
  41         config.preprocessed_data_topdir,
  42         (name if is_token_level else
  43          name + '_' + config.label_encoding) + '_label_mapping.pkl')
  44 
  45     if self.label_mapping:
  46       self._n_classes = len(set(self.label_mapping.values()))
  47     else:
  48       self._n_classes = None
  49 
  50   def get_dataset(self, split):
  51     if (split == 'train' and not self._config.for_preprocessing and
  52         tf.gfile.Exists(os.path.join(self._raw_data_path, 'train_subset.txt'))):
  53       split = 'train_subset'
  54     return minibatching.Dataset(
  55         self._config, self._get_examples(split), self._task_name)
  56 
  57   def get_labeled_sentences(self, split):
  58     sentences = []
  59     path = os.path.join(self._raw_data_path, split + '.txt')
  60     if not tf.gfile.Exists(path):
  61       if self._config.for_preprocessing:
  62         return []
  63       else:
  64         raise ValueError('Unable to load data from', path)
  65 
  66     with tf.gfile.GFile(path, 'r') as f:
  67       sentence = []
  68       for line in f:
  69         line = line.strip().split()
  70         if not line:
  71           if sentence:
  72             words, tags = zip(*sentence)
  73             sentences.append((words, tags))
  74             sentence = []
  75           continue
  76         if line[0] == '-DOCSTART-':
  77           continue
  78         word, tag = line[0], line[-1]
  79         sentence.append((word, tag))
  80     return sentences
  81 
  82   @property
  83   def label_mapping(self):
  84     if not self._config.for_preprocessing:
  85       return utils.load_cpickle(self.label_mapping_path)
  86 
  87     tag_counts = collections.Counter()
  88     train_tags = set()
  89     for split in ['train', 'dev', 'test']:
  90       for words, tags in self.get_labeled_sentences(split):
  91         if not self._is_token_level:
  92           span_labels = tagging_utils.get_span_labels(tags)
  93           tags = tagging_utils.get_tags(
  94               span_labels, len(words), self._config.label_encoding)
  95         for tag in tags:
  96           if self._task_name == 'depparse':
  97             tag = tag.split('-')[1]
  98           tag_counts[tag] += 1
  99           if split == 'train':
 100             train_tags.add(tag)
 101     if self._task_name == 'ccg':
 102       # for CCG, there are tags in the test sets that aren't in the train set
 103       # all tags not in the train set get mapped to a special label
 104       # the model will never predict this label because it never sees it in the
 105       # training set
 106       not_in_train_tags = []
 107       for tag, count in tag_counts.items():
 108         if tag not in train_tags:
 109           not_in_train_tags.append(tag)
 110       label_mapping = {
 111           label: i for i, label in enumerate(sorted(filter(
 112             lambda t: t not in not_in_train_tags, tag_counts.keys())))
 113       }
 114       n = len(label_mapping)
 115       for tag in not_in_train_tags:
 116         label_mapping[tag] = n
 117     else:
 118       labels = sorted(tag_counts.keys())
 119       if self._task_name == 'depparse':
 120         labels.remove('root')
 121         labels.insert(0, 'root')
 122       label_mapping = {label: i for i, label in enumerate(labels)}
 123     return label_mapping
 124 
 125   def _get_examples(self, split):
 126     word_vocab = embeddings.get_word_vocab(self._config)
 127     char_vocab = embeddings.get_char_vocab()
 128     examples = [
 129         TaggingExample(
 130             self._config, self._is_token_level, words, tags,
 131             word_vocab, char_vocab, self.label_mapping, self._task_name)
 132         for words, tags in self.get_labeled_sentences(split)]
 133     if self._config.train_set_percent < 100:
 134       utils.log('using reduced train set ({:}%)'.format(
 135           self._config.train_set_percent))
 136       random.shuffle(examples)
 137       examples = examples[:int(len(examples) *
 138                                self._config.train_set_percent / 100.0)]
 139     return examples

 142 class TaggingExample(example.Example):
 143   def __init__(self, config, is_token_level, words, original_tags,
 144                word_vocab, char_vocab, label_mapping, task_name):
 145     super(TaggingExample, self).__init__(words, word_vocab, char_vocab)
 146     if is_token_level:
 147       labels = original_tags
 148     else:
 149       span_labels = tagging_utils.get_span_labels(original_tags)
 150       labels = tagging_utils.get_tags(
 151           span_labels, len(words), config.label_encoding)
 152 
 153     if task_name == 'depparse':
 154       self.labels = []
 155       for l in labels:
 156         split = l.split('-')
 157         self.labels.append(
 158             len(label_mapping) * (0 if split[0] == '0' else 1 + int(split[0]))
 159             + label_mapping[split[1]])
 160     else:
 161       self.labels = [label_mapping[l] for l in labels]
