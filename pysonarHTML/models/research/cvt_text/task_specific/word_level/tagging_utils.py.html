  23 def get_span_labels(sentence_tags, inv_label_mapping=None):
  24   """Go from token-level labels to list of entities (start, end, class)."""
  25 
  26   if inv_label_mapping:
  27     sentence_tags = [inv_label_mapping[i] for i in sentence_tags]
  28   span_labels = []
  29   last = 'O'
  30   start = -1
  31   for i, tag in enumerate(sentence_tags):
  32     pos, _ = (None, 'O') if tag == 'O' else tag.split('-')
  33     if (pos == 'S' or pos == 'B' or tag == 'O') and last != 'O':
  34       span_labels.append((start, i - 1, last.split('-')[-1]))
  35     if pos == 'B' or pos == 'S' or last == 'O':
  36       start = i
  37     last = tag
  38   if sentence_tags[-1] != 'O':
  39     span_labels.append((start, len(sentence_tags) - 1,
  40                         sentence_tags[-1].split('-')[-1]))
  41   return span_labels

  44 def get_tags(span_labels, length, encoding):
  45   """Converts a list of entities to token-label labels based on the provided
  46   encoding (e.g., BIOES).
  47   """
  48 
  49   tags = ['O' for _ in range(length)]
  50   for s, e, t in span_labels:
  51     for i in range(s, e + 1):
  52       tags[i] = 'I-' + t
  53     if 'E' in encoding:
  54       tags[e] = 'E-' + t
  55     if 'B' in encoding:
  56       tags[s] = 'B-' + t
  57     if 'S' in encoding and s == e:
  58       tags[s] = 'S-' + t
  59   return tags
