  32 class Task(object):
  33   __metaclass__ = abc.ABCMeta
  34 
  35   def __init__(self, config, name, loader):
  36     self.config = config
  37     self.name = name
  38     self.loader = loader
  39     self.train_set = self.loader.get_dataset("train")
  40     self.val_set = self.loader.get_dataset("dev" if config.dev_set else "test")
  41 
  42   @abc.abstractmethod
  43   def get_module(self, inputs, encoder):
  44     pass
  45 
  46   @abc.abstractmethod
  47   def get_scorer(self):
  48     pass
  49 
  50 
  51 class Tagging(Task):
  52   def __init__(self, config, name, is_token_level=True):
  53     super(Tagging, self).__init__(
  54         config, name, word_level_data.TaggedDataLoader(
  55             config, name, is_token_level))
  56     self.n_classes = len(set(self.loader.label_mapping.values()))
  57     self.is_token_level = is_token_level
  58 
  59   def get_module(self, inputs, encoder):
  60     return tagging_module.TaggingModule(
  61         self.config, self.name, self.n_classes, inputs, encoder)
  62 
  63   def get_scorer(self):
  64     if self.is_token_level:
  65       return tagging_scorers.AccuracyScorer()
  66     else:
  67       return tagging_scorers.EntityLevelF1Scorer(self.loader.label_mapping)
  68 
  69 
  70 class DependencyParsing(Tagging):
  71   def __init__(self, config, name):
  72     super(DependencyParsing, self).__init__(config, name, True)
  73 
  74   def get_module(self, inputs, encoder):
  75     return depparse_module.DepparseModule(
  76         self.config, self.name, self.n_classes, inputs, encoder)
  77 
  78   def get_scorer(self):
  79     return depparse_scorer.DepparseScorer(
  80         self.n_classes, (embeddings.get_punctuation_ids(self.config)))
  81 
  82 
  83 def get_task(config, name):
  84   if name in ["ccg", "pos"]:
  85     return Tagging(config, name, True)
  86   elif name in ["chunk", "ner", "er"]:
  87     return Tagging(config, name, False)
  88   elif name == "depparse":
  89     return DependencyParsing(config, name)
  90   else:
  91     raise ValueError("Unknown task", name)
