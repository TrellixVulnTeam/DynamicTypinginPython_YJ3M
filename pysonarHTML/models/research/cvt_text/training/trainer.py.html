  32 class Trainer(object):
  33   def __init__(self, config):
  34     self._config = config
  35     self.tasks = [task_definitions.get_task(self._config, task_name)
  36                   for task_name in self._config.task_names]
  37 
  38     utils.log('Loading Pretrained Embeddings')
  39     pretrained_embeddings = utils.load_cpickle(self._config.word_embeddings)
  40 
  41     utils.log('Building Model')
  42     self._model = multitask_model.Model(
  43         self._config, pretrained_embeddings, self.tasks)
  44     utils.log()
  45 
  46   def train(self, sess, progress, summary_writer):
  47     heading = lambda s: utils.heading(s, '(' + self._config.model_name + ')')
  48     trained_on_sentences = 0
  49     start_time = time.time()
  50     unsupervised_loss_total, unsupervised_loss_count = 0, 0
  51     supervised_loss_total, supervised_loss_count = 0, 0
  52     for mb in self._get_training_mbs(progress.unlabeled_data_reader):
  53       if mb.task_name != 'unlabeled':
  54         loss = self._model.train_labeled(sess, mb)
  55         supervised_loss_total += loss
  56         supervised_loss_count += 1
  57 
  58       if mb.task_name == 'unlabeled':
  59         self._model.run_teacher(sess, mb)
  60         loss = self._model.train_unlabeled(sess, mb)
  61         unsupervised_loss_total += loss
  62         unsupervised_loss_count += 1
  63         mb.teacher_predictions.clear()
  64 
  65       trained_on_sentences += mb.size
  66       global_step = self._model.get_global_step(sess)
  67 
  68       if global_step % self._config.print_every == 0:
  69         utils.log('step {:} - '
  70                   'supervised loss: {:.2f} - '
  71                   'unsupervised loss: {:.2f} - '
  72                   '{:.1f} sentences per second'.format(
  73             global_step,
  74             supervised_loss_total / max(1, supervised_loss_count),
  75             unsupervised_loss_total / max(1, unsupervised_loss_count),
  76             trained_on_sentences / (time.time() - start_time)))
  77         unsupervised_loss_total, unsupervised_loss_count = 0, 0
  78         supervised_loss_total, supervised_loss_count = 0, 0
  79 
  80       if global_step % self._config.eval_dev_every == 0:
  81         heading('EVAL ON DEV')
  82         self.evaluate_all_tasks(sess, summary_writer, progress.history)
  83         progress.save_if_best_dev_model(sess, global_step)
  84         utils.log()
  85 
  86       if global_step % self._config.eval_train_every == 0:
  87         heading('EVAL ON TRAIN')
  88         self.evaluate_all_tasks(sess, summary_writer, progress.history, True)
  89         utils.log()
  90 
  91       if global_step % self._config.save_model_every == 0:
  92         heading('CHECKPOINTING MODEL')
  93         progress.write(sess, global_step)
  94         utils.log()
  95 
  96   def evaluate_all_tasks(self, sess, summary_writer, history, train_set=False):
  97     for task in self.tasks:
  98       results = self._evaluate_task(sess, task, summary_writer, train_set)
  99       if history is not None:
 100         results.append(('step', self._model.get_global_step(sess)))
 101         history.append(results)
 102     if history is not None:
 103       utils.write_cpickle(history, self._config.history_file)
 104 
 105   def _evaluate_task(self, sess, task, summary_writer, train_set):
 106     scorer = task.get_scorer()
 107     data = task.train_set if train_set else task.val_set
 108     for i, mb in enumerate(data.get_minibatches(self._config.test_batch_size)):
 109       loss, batch_preds = self._model.test(sess, mb)
 110       scorer.update(mb.examples, batch_preds, loss)
 111 
 112     results = scorer.get_results(task.name +
 113                                  ('_train_' if train_set else '_dev_'))
 114     utils.log(task.name.upper() + ': ' + scorer.results_str())
 115     write_summary(summary_writer, results,
 116                   global_step=self._model.get_global_step(sess))
 117     return results
 118 
 119   def _get_training_mbs(self, unlabeled_data_reader):
 120     datasets = [task.train_set for task in self.tasks]
 121     weights = [np.sqrt(dataset.size) for dataset in datasets]
 122     thresholds = np.cumsum([w / np.sum(weights) for w in weights])
 123 
 124     labeled_mbs = [dataset.endless_minibatches(self._config.train_batch_size)
 125                    for dataset in datasets]
 126     unlabeled_mbs = unlabeled_data_reader.endless_minibatches()
 127     while True:
 128       dataset_ind = bisect.bisect(thresholds, np.random.random())
 129       yield next(labeled_mbs[dataset_ind])
 130       if self._config.is_semisup:
 131         yield next(unlabeled_mbs)

 134 def write_summary(writer, results, global_step):
 135   for k, v in results:
 136     if 'f1' in k or 'acc' in k or 'loss' in k:
 137       writer.add_summary(tf.Summary(
 138           value=[tf.Summary.Value(tag=k, simple_value=v)]), global_step)
 139   writer.flush()
