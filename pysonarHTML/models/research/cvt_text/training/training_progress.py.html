  16 """
  17 Tracks and saves training progress (models and other data such as the current
  18 location in the lm1b corpus) for later reloading.
  19 """
  20 
  21 from __future__ import absolute_import
  22 from __future__ import division
  23 from __future__ import print_function
  24 
  25 import tensorflow as tf
  26 
  27 from base import utils
  28 from corpus_processing import unlabeled_data
  29 
  30 
  31 class TrainingProgress(object):
  32   def __init__(self, config, sess, checkpoint_saver, best_model_saver,
  33               restore_if_possible=True):
  34     self.config = config
  35     self.checkpoint_saver = checkpoint_saver
  36     self.best_model_saver = best_model_saver
  37 
  38     tf.gfile.MakeDirs(config.checkpoints_dir)
  39     if restore_if_possible and tf.gfile.Exists(config.progress):
  40       history, current_file, current_line = utils.load_cpickle(
  41           config.progress, memoized=False)
  42       self.history = history
  43       self.unlabeled_data_reader = unlabeled_data.UnlabeledDataReader(
  44           config, current_file, current_line)
  45       utils.log("Continuing from global step", dict(self.history[-1])["step"],
  46                 "(lm1b file {:}, line {:})".format(current_file, current_line))
  47       self.checkpoint_saver.restore(sess, tf.train.latest_checkpoint(
  48           self.config.checkpoints_dir))
  49     else:
  50       utils.log("No previous checkpoint found - starting from scratch")
  51       self.history = []
  52       self.unlabeled_data_reader = (
  53           unlabeled_data.UnlabeledDataReader(config))
  54 
  55   def write(self, sess, global_step):
  56     self.checkpoint_saver.save(sess, self.config.checkpoint,
  57                                global_step=global_step)
  58     utils.write_cpickle(
  59         (self.history, self.unlabeled_data_reader.current_file,
  60          self.unlabeled_data_reader.current_line),
  61         self.config.progress)
  62 
  63   def save_if_best_dev_model(self, sess, global_step):
  64     best_avg_score = 0
  65     for i, results in enumerate(self.history):
  66       if any("train" in metric for metric, value in results):
  67         continue
  68       total, count = 0, 0
  69       for metric, value in results:
  70         if "f1" in metric or "las" in metric or "accuracy" in metric:
  71           total += value
  72           count += 1
  73       avg_score = total / count
  74       if avg_score >= best_avg_score:
  75         best_avg_score = avg_score
  76         if i == len(self.history) - 1:
  77           utils.log("New best model! Saving...")
  78           self.best_model_saver.save(sess, self.config.best_model_checkpoint,
  79                                      global_step=global_step)
