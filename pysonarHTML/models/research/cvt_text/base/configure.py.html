  16 """Classes for storing hyperparameters, data locations, etc."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import json
  23 from os.path import join
  24 import tensorflow as tf
  25 
  26 
  27 class Config(object):
  28   """Stores everything needed to train a model."""
  29 
  30   def __init__(self, **kwargs):
  31     # general
  32     self.data_dir = './data'  # top directory for data (corpora, models, etc.)
  33     self.model_name = 'default_model'  # name identifying the current model
  34 
  35     # mode
  36     self.mode = 'train'  # either "train" or "eval"
  37     self.task_names = ['chunk']  # list of tasks this model will learn
  38                                  # more than one trains a multi-task model
  39     self.is_semisup = True  # whether to use CVT or train purely supervised
  40     self.for_preprocessing = False  # is this for the preprocessing script
  41 
  42     # embeddings
  43     self.pretrained_embeddings = 'glove.6B.300d.txt'  # which pretrained
  44                                                       # embeddings to use
  45     self.word_embedding_size = 300  # size of each word embedding
  46 
  47     # encoder
  48     self.use_chars = True  # whether to include a character-level cnn
  49     self.char_embedding_size = 50  # size of character embeddings
  50     self.char_cnn_filter_widths = [2, 3, 4]  # filter widths for the char cnn
  51     self.char_cnn_n_filters = 100  # number of filters for each filter width
  52     self.unidirectional_sizes = [1024]  # size of first Bi-LSTM
  53     self.bidirectional_sizes = [512]  # size of second Bi-LSTM
  54     self.projection_size = 512  # projections size for LSTMs and hidden layers
  55 
  56     # dependency parsing
  57     self.depparse_projection_size = 128  # size of the representations used in
  58                                          # the bilinear classifier for parsing
  59 
  60     # tagging
  61     self.label_encoding = 'BIOES'  # label encoding scheme for entity-level
  62                                    # tagging tasks
  63     self.label_smoothing = 0.1  # label smoothing rate for tagging tasks
  64 
  65     # optimization
  66     self.lr = 0.5  # base learning rate
  67     self.momentum = 0.9  # momentum
  68     self.grad_clip = 1.0  # maximum gradient norm during optimization
  69     self.warm_up_steps = 5000.0  # linearly ramp up the lr for this many steps
  70     self.lr_decay = 0.005  # factor for gradually decaying the lr
  71 
  72     # EMA
  73     self.ema_decay = 0.998  # EMA coefficient for averaged model weights
  74     self.ema_test = True  # whether to use EMA weights at test time
  75     self.ema_teacher = False  # whether to use EMA weights for the teacher model
  76 
  77     # regularization
  78     self.labeled_keep_prob = 0.5  # 1 - dropout on labeled examples
  79     self.unlabeled_keep_prob = 0.8  # 1 - dropout on unlabeled examples
  80 
  81     # sizing
  82     self.max_sentence_length = 100  # maximum length of unlabeled sentences
  83     self.max_word_length = 20  # maximum length of words for char cnn
  84     self.train_batch_size = 64  # train batch size
  85     self.test_batch_size = 64  # test batch size
  86     self.buckets = [(0, 15), (15, 40), (40, 1000)]  # buckets for binning
  87                                                     # sentences by length
  88 
  89     # training
  90     self.print_every = 25  # how often to print out training progress
  91     self.eval_dev_every = 500  # how often to evaluate on the dev set
  92     self.eval_train_every = 2000  # how often to evaluate on the train set
  93     self.save_model_every = 1000  # how often to checkpoint the model
  94 
  95     # data set
  96     self.train_set_percent = 100  # how much of the train set to use
  97 
  98     for k, v in kwargs.iteritems():
  99       if k not in self.__dict__:
 100         raise ValueError("Unknown argument", k)
 101       self.__dict__[k] = v
 102 
 103     self.dev_set = self.mode == "train"  # whether to evaluate on the dev or
 104                                          # test set
 105 
 106     # locations of various data files
 107     self.raw_data_topdir = join(self.data_dir, 'raw_data')
 108     self.unsupervised_data = join(
 109         self.raw_data_topdir,
 110         'unlabeled_data',
 111         '1-billion-word-language-modeling-benchmark-r13output',
 112         'training-monolingual.tokenized.shuffled')
 113     self.pretrained_embeddings_file = join(
 114         self.raw_data_topdir, 'pretrained_embeddings',
 115         self.pretrained_embeddings)
 116 
 117     self.preprocessed_data_topdir = join(self.data_dir, 'preprocessed_data')
 118     self.embeddings_dir = join(self.preprocessed_data_topdir,
 119                                self.pretrained_embeddings.rsplit('.', 1)[0])
 120     self.word_vocabulary = join(self.embeddings_dir, 'word_vocabulary.pkl')
 121     self.word_embeddings = join(self.embeddings_dir, 'word_embeddings.pkl')
 122 
 123     self.model_dir = join(self.data_dir, "models", self.model_name)
 124     self.checkpoints_dir = join(self.model_dir, 'checkpoints')
 125     self.checkpoint = join(self.checkpoints_dir, 'checkpoint.ckpt')
 126     self.best_model_checkpoints_dir = join(
 127         self.model_dir, 'best_model_checkpoints')
 128     self.best_model_checkpoint = join(
 129         self.best_model_checkpoints_dir, 'checkpoint.ckpt')
 130     self.progress = join(self.checkpoints_dir, 'progress.pkl')
 131     self.summaries_dir = join(self.model_dir, 'summaries')
 132     self.history_file = join(self.model_dir, 'history.pkl')
 133 
 134   def write(self):
 135     tf.gfile.MakeDirs(self.model_dir)
 136     with open(join(self.model_dir, 'config.json'), 'w') as f:
 137       f.write(json.dumps(self.__dict__, sort_keys=True, indent=4,
 138                          separators=(',', ': ')))
