  16 """Reads data from a large unlabeled corpus."""

  30 class UnlabeledDataReader(object):
  31   def __init__(self, config, starting_file=0, starting_line=0, one_pass=False):
  32     self.config = config
  33     self.current_file = starting_file
  34     self.current_line = starting_line
  35     self._one_pass = one_pass
  36 
  37   def endless_minibatches(self):
  38     for examples in self.get_unlabeled_examples():
  39       d = minibatching.Dataset(self.config, examples, 'unlabeled')
  40       for mb in d.get_minibatches(self.config.train_batch_size):
  41         yield mb
  42 
  43   def _make_examples(self, sentences):
  44     word_vocab = embeddings.get_word_vocab(self.config)
  45     char_vocab = embeddings.get_char_vocab()
  46     return [
  47         example.Example(sentence, word_vocab, char_vocab)
  48         for sentence in sentences
  49     ]
  50 
  51   def get_unlabeled_examples(self):
  52     lines = []
  53     for words in self.get_unlabeled_sentences():
  54       lines.append(words)
  55       if len(lines) >= 10000:
  56         yield self._make_examples(lines)
  57         lines = []
  58 
  59   def get_unlabeled_sentences(self):
  60     while True:
  61       file_ids_and_names = sorted([
  62           (int(fname.split('-')[1].replace('.txt', '')), fname) for fname in
  63           tf.gfile.ListDirectory(self.config.unsupervised_data)])
  64       for fid, fname in file_ids_and_names:
  65         if fid < self.current_file:
  66           continue
  67         self.current_file = fid
  68         self.current_line = 0
  69         with tf.gfile.FastGFile(os.path.join(self.config.unsupervised_data,
  70                                              fname), 'r') as f:
  71           for i, line in enumerate(f):
  72             if i < self.current_line:
  73               continue
  74             self.current_line = i
  75             words = line.strip().split()
  76             if len(words) < self.config.max_sentence_length:
  77               yield words
  78       self.current_file = 0
  79       self.current_line = 0
  80       if self._one_pass:
  81         break
