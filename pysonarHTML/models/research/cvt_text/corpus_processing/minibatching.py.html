  29 def get_bucket(config, l):
  30   for i, (s, e) in enumerate(config.buckets):
  31     if s <= l < e:
  32       return config.buckets[i]

  35 def build_array(nested_lists, dtype='int32'):
  36   depth_to_sizes = collections.defaultdict(set)
  37   _get_sizes(nested_lists, depth_to_sizes)
  38   shape = [max(depth_to_sizes[depth]) for depth in range(len(depth_to_sizes))]
  39 
  40   copy_depth = len(depth_to_sizes) - 1
  41   while copy_depth > 0 and len(depth_to_sizes[copy_depth]) == 1:
  42     copy_depth -= 1
  43 
  44   arr = np.zeros(shape, dtype=dtype)
  45   _fill_array(nested_lists, arr, copy_depth)
  46 
  47   return arr

  50 def _get_sizes(nested_lists, depth_to_sizes, depth=0):
  51   depth_to_sizes[depth].add(len(nested_lists))
  52   first_elem = nested_lists[0]
  53   if (isinstance(first_elem, collections.Sequence) or
  54       isinstance(first_elem, np.ndarray)):
  55     for sublist in nested_lists:
  56       _get_sizes(sublist, depth_to_sizes, depth + 1)
  57 
  58 
  59 def _fill_array(nested_lists, arr, copy_depth, depth=0):
  60   if depth == copy_depth:
  61     for i in range(len(nested_lists)):
  62       if isinstance(nested_lists[i], np.ndarray):
  63         arr[i] = nested_lists[i]
  64       else:
  65         arr[i] = np.array(nested_lists[i])
  66   else:
  67     for i in range(len(nested_lists)):
  68       _fill_array(nested_lists[i], arr[i], copy_depth, depth + 1)

  71 class Dataset(object):
  72   def __init__(self, config, examples, task_name='unlabeled', is_training=False):
  73     self._config = config
  74     self.examples = examples
  75     self.size = len(examples)
  76     self.task_name = task_name
  77     self.is_training = is_training
  78 
  79   def get_minibatches(self, minibatch_size):
  80     by_bucket = collections.defaultdict(list)
  81     for i, e in enumerate(self.examples):
  82       by_bucket[get_bucket(self._config, len(e.words))].append(i)
  83 
  84     # save memory by weighting examples so longer sentences have
  85     # smaller minibatches.
  86     weight = lambda ind: np.sqrt(len(self.examples[ind].words))
  87     total_weight = float(sum(weight(i) for i in range(len(self.examples))))
  88     weight_per_batch = minibatch_size * total_weight / len(self.examples)
  89     cumulative_weight = 0.0
  90     id_batches = []
  91     for _, ids in by_bucket.iteritems():
  92       ids = np.array(ids)
  93       np.random.shuffle(ids)
  94       curr_batch, curr_weight = [], 0.0
  95       for i, curr_id in enumerate(ids):
  96         curr_batch.append(curr_id)
  97         curr_weight += weight(curr_id)
  98         if (i == len(ids) - 1 or cumulative_weight + curr_weight >=
  99             (len(id_batches) + 1) * weight_per_batch):
 100           cumulative_weight += curr_weight
 101           id_batches.append(np.array(curr_batch))
 102           curr_batch, curr_weight = [], 0.0
 103     random.shuffle(id_batches)
 104 
 105     for id_batch in id_batches:
 106       yield self._make_minibatch(id_batch)
 107 
 108   def endless_minibatches(self, minibatch_size):
 109     while True:
 110       for mb in self.get_minibatches(minibatch_size):
 111         yield mb
 112 
 113   def _make_minibatch(self, ids):
 114     examples = [self.examples[i] for i in ids]
 115     sentence_lengths = np.array([len(e.words) for e in examples])
 116     max_word_length = min(max(max(len(word) for word in e.chars)
 117                               for e in examples),
 118                           self._config.max_word_length)
 119     characters = [[[embeddings.PAD] + [embeddings.START] + w[:max_word_length] +
 120                    [embeddings.END] + [embeddings.PAD] for w in e.chars]
 121                   for e in examples]
 122     # the first and last words are masked because they are start/end tokens
 123     mask = build_array([[0] + [1] * (length - 2) + [0]
 124                         for length in sentence_lengths])
 125     words = build_array([e.words for e in examples])
 126     chars = build_array(characters, dtype='int16')
 127     return Minibatch(
 128         task_name=self.task_name,
 129         size=ids.size,
 130         examples=examples,
 131         ids=ids,
 132         teacher_predictions={},
 133         words=words,
 134         chars=chars,
 135         lengths=sentence_lengths,
 136         mask=mask,
 137     )
