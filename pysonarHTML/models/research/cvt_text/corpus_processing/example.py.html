  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function

  22 from base import embeddings
  23 
  24 
  25 CONTRACTION_WORDS = set(w + 'n' for w in
  26                         ['do', 'does', 'did', 'is', 'are', 'was', 'were', 'has',
  27                          'have', 'had', 'could', 'would', 'should', 'ca', 'wo',
  28                          'ai', 'might'])
  29 
  30 
  31 class Example(object):
  32   def __init__(self, words, word_vocab, char_vocab):
  33     words = words[:]
  34     # Fix inconsistent tokenization between datasets
  35     for i in range(len(words)):
  36       if (words[i].lower() == '\'t' and i > 0 and
  37           words[i - 1].lower() in CONTRACTION_WORDS):
  38         words[i] = words[i - 1][-1] + words[i]
  39         words[i - 1] = words[i - 1][:-1]
  40 
  41     self.words = ([embeddings.START] +
  42                   [word_vocab[embeddings.normalize_word(w)] for w in words] +
  43                   [embeddings.END])
  44     self.chars = ([[embeddings.MISSING]] +
  45                   [[char_vocab[c] for c in embeddings.normalize_chars(w)]
  46                    for w in words] +
  47                   [[embeddings.MISSING]])
  48 
  49   def __repr__(self,):
  50     inv_char_vocab = embeddings.get_inv_char_vocab()
  51     return ' '.join([''.join([inv_char_vocab[c] for c in w])
  52                      for w in self.chars])
