  15 """Defines NeuMF model for NCF framework.
  16 
  17 Some abbreviations used in the code base:
  18 NeuMF: Neural Matrix Factorization
  19 NCF: Neural Collaborative Filtering
  20 GMF: Generalized Matrix Factorization
  21 MLP: Multi-Layer Perceptron
  22 
  23 GMF applies a linear kernel to model the latent feature interactions, and MLP
  24 uses a nonlinear kernel to learn the interaction function from data. NeuMF model
  25 is a fused model of GMF and MLP to better model the complex user-item
  26 interactions, and unifies the strengths of linearity of MF and non-linearity of
  27 MLP for modeling the user-item latent structures.
  28 
  29 In NeuMF model, it allows GMF and MLP to learn separate embeddings, and combine
  30 the two models by concatenating their last hidden layer.
  31 """

  49 def sparse_to_dense_grads(grads_and_vars):
  50   """Convert sparse gradients to dense gradients.
  51 
  52   All sparse gradients, which are represented as instances of tf.IndexedSlices,
  53   are converted to dense Tensors. Dense gradients, which are represents as
  54   Tensors, are unchanged.
  55 
  56   The purpose of this conversion is that for small embeddings, which are used by
  57   this model, applying dense gradients with the AdamOptimizer is faster than
  58   applying sparse gradients.
  59 
  60   Args
  61     grads_and_vars: A list of (gradient, variable) tuples. Each gradient can
  62       be a Tensor or an IndexedSlices. Tensors are unchanged, and IndexedSlices
  63       are converted to dense Tensors.
  64   Returns:
  65     The same list of (gradient, variable) as `grads_and_vars`, except each
  66     IndexedSlices gradient is converted to a Tensor.
  67   """
  68 
  69   # Calling convert_to_tensor changes IndexedSlices into Tensors, and leaves
  70   # Tensors unchanged.
  71   return [(tf.convert_to_tensor(g), v) for g, v in grads_and_vars]

  74 def neumf_model_fn(features, labels, mode, params):
  75   """Model Function for NeuMF estimator."""
  76   if params.get("use_seed"):
  77     tf.set_random_seed(stat_utils.random_int32())
  78 
  79   users = features[movielens.USER_COLUMN]
  80   items = features[movielens.ITEM_COLUMN]
  81 
  82   user_input = tf.keras.layers.Input(tensor=users)
  83   item_input = tf.keras.layers.Input(tensor=items)
  84   logits = construct_model(user_input, item_input, params).output
  85 
  86   # Softmax with the first column of zeros is equivalent to sigmoid.
  87   softmax_logits = ncf_common.convert_to_softmax_logits(logits)
  88 
  89   if mode == tf.estimator.ModeKeys.EVAL:
  90     duplicate_mask = tf.cast(features[rconst.DUPLICATE_MASK], tf.float32)
  91     return _get_estimator_spec_with_metrics(
  92         logits,
  93         softmax_logits,
  94         duplicate_mask,
  95         params["num_neg"],
  96         params["match_mlperf"],
  97         use_tpu_spec=params["use_tpu"])
  98 
  99   elif mode == tf.estimator.ModeKeys.TRAIN:
 100     labels = tf.cast(labels, tf.int32)
 101     valid_pt_mask = features[rconst.VALID_POINT_MASK]
 102 
 103     optimizer = tf.compat.v1.train.AdamOptimizer(
 104         learning_rate=params["learning_rate"],
 105         beta1=params["beta1"],
 106         beta2=params["beta2"],
 107         epsilon=params["epsilon"])
 108     if params["use_tpu"]:
 109       optimizer = tf.compat.v1.tpu.CrossShardOptimizer(optimizer)
 110 
 111     loss = tf.compat.v1.losses.sparse_softmax_cross_entropy(
 112         labels=labels,
 113         logits=softmax_logits,
 114         weights=tf.cast(valid_pt_mask, tf.float32)
 115     )
 116 
 117     tf.identity(loss, name="cross_entropy")
 118 
 119     global_step = tf.compat.v1.train.get_global_step()
 120     tvars = tf.compat.v1.trainable_variables()
 121     gradients = optimizer.compute_gradients(
 122         loss, tvars, colocate_gradients_with_ops=True)
 123     gradients = sparse_to_dense_grads(gradients)
 124     minimize_op = optimizer.apply_gradients(
 125         gradients, global_step=global_step, name="train")
 126     update_ops = tf.compat.v1.get_collection(tf.compat.v1.GraphKeys.UPDATE_OPS)
 127     train_op = tf.group(minimize_op, update_ops)
 128 
 129     return tf.estimator.EstimatorSpec(mode=mode, loss=loss, train_op=train_op)
 130 
 131   else:
 132     raise NotImplementedError

 139 def construct_model(user_input: tf.Tensor, item_input: tf.Tensor,
 140                     params: Dict[Text, Any]) -> tf.keras.Model:
 141   """Initialize NeuMF model.
 142 
 143   Args:
 144     user_input: keras input layer for users
 145     item_input: keras input layer for items
 146     params: Dict of hyperparameters.
 147 
 148   Raises:
 149     ValueError: if the first model layer is not even.
 150   Returns:
 151     model:  a keras Model for computing the logits
 152   """
 153   num_users = params["num_users"]
 154   num_items = params["num_items"]
 155 
 156   model_layers = params["model_layers"]
 157 
 158   mf_regularization = params["mf_regularization"]
 159   mlp_reg_layers = params["mlp_reg_layers"]
 160 
 161   mf_dim = params["mf_dim"]
 162 
 163   if model_layers[0] % 2 != 0:
 164     raise ValueError("The first layer size should be multiple of 2!")
 165 
 166   # Initializer for embedding layers
 167   embedding_initializer = "glorot_uniform"
 168 
 169   def mf_slice_fn(x):
 170     x = tf.squeeze(x, [1])
 171     return x[:, :mf_dim]
 172 
 173   def mlp_slice_fn(x):
 174     x = tf.squeeze(x, [1])
 175     return x[:, mf_dim:]
 176 
 177   # It turns out to be significantly more effecient to store the MF and MLP
 178   # embedding portions in the same table, and then slice as needed.
 179   embedding_user = tf.keras.layers.Embedding(
 180       num_users,
 181       mf_dim + model_layers[0] // 2,
 182       embeddings_initializer=embedding_initializer,
 183       embeddings_regularizer=tf.keras.regularizers.l2(mf_regularization),
 184       input_length=1,
 185       name="embedding_user")(
 186           user_input)
 187 
 188   embedding_item = tf.keras.layers.Embedding(
 189       num_items,
 190       mf_dim + model_layers[0] // 2,
 191       embeddings_initializer=embedding_initializer,
 192       embeddings_regularizer=tf.keras.regularizers.l2(mf_regularization),
 193       input_length=1,
 194       name="embedding_item")(
 195           item_input)
 196 
 197   # GMF part
 198   mf_user_latent = tf.keras.layers.Lambda(
 199       mf_slice_fn, name="embedding_user_mf")(embedding_user)
 200   mf_item_latent = tf.keras.layers.Lambda(
 201       mf_slice_fn, name="embedding_item_mf")(embedding_item)
 202 
 203   # MLP part
 204   mlp_user_latent = tf.keras.layers.Lambda(
 205       mlp_slice_fn, name="embedding_user_mlp")(embedding_user)
 206   mlp_item_latent = tf.keras.layers.Lambda(
 207       mlp_slice_fn, name="embedding_item_mlp")(embedding_item)
 208 
 209   # Element-wise multiply
 210   mf_vector = tf.keras.layers.multiply([mf_user_latent, mf_item_latent])
 211 
 212   # Concatenation of two latent features
 213   mlp_vector = tf.keras.layers.concatenate([mlp_user_latent, mlp_item_latent])
 214 
 215   num_layer = len(model_layers)  # Number of layers in the MLP
 216   for layer in xrange(1, num_layer):
 217     model_layer = tf.keras.layers.Dense(
 218         model_layers[layer],
 219         kernel_regularizer=tf.keras.regularizers.l2(mlp_reg_layers[layer]),
 220         activation="relu")
 221     mlp_vector = model_layer(mlp_vector)
 222 
 223   # Concatenate GMF and MLP parts
 224   predict_vector = tf.keras.layers.concatenate([mf_vector, mlp_vector])
 225 
 226   # Final prediction layer
 227   logits = tf.keras.layers.Dense(
 228       1, activation=None, kernel_initializer="lecun_uniform",
 229       name=movielens.RATING_COLUMN)(predict_vector)
 230 
 231   # Print model topology.
 232   model = tf.keras.models.Model([user_input, item_input], logits)
 233   model.summary()
 234   sys.stdout.flush()
 235 
 236   return model

 239 def _get_estimator_spec_with_metrics(logits: tf.Tensor,
 240                                      softmax_logits: tf.Tensor,
 241                                      duplicate_mask: tf.Tensor,
 242                                      num_training_neg: int,
 243                                      match_mlperf: bool = False,
 244                                      use_tpu_spec: bool = False):
 245   """Returns a EstimatorSpec that includes the metrics."""
 246   cross_entropy, \
 247   metric_fn, \
 248   in_top_k, \
 249   ndcg, \
 250   metric_weights = compute_eval_loss_and_metrics_helper(
 251       logits,
 252       softmax_logits,
 253       duplicate_mask,
 254       num_training_neg,
 255       match_mlperf)
 256 
 257   if use_tpu_spec:
 258     return tf.estimator.tpu.TPUEstimatorSpec(
 259         mode=tf.estimator.ModeKeys.EVAL,
 260         loss=cross_entropy,
 261         eval_metrics=(metric_fn, [in_top_k, ndcg, metric_weights]))
 262 
 263   return tf.estimator.EstimatorSpec(
 264       mode=tf.estimator.ModeKeys.EVAL,
 265       loss=cross_entropy,
 266       eval_metric_ops=metric_fn(in_top_k, ndcg, metric_weights)
 267   )

 270 def compute_eval_loss_and_metrics_helper(logits: tf.Tensor,
 271                                          softmax_logits: tf.Tensor,
 272                                          duplicate_mask: tf.Tensor,
 273                                          num_training_neg: int,
 274                                          match_mlperf: bool = False):
 275   """Model evaluation with HR and NDCG metrics.
 276 
 277   The evaluation protocol is to rank the test interacted item (truth items)
 278   among the randomly chosen 999 items that are not interacted by the user.
 279   The performance of the ranked list is judged by Hit Ratio (HR) and Normalized
 280   Discounted Cumulative Gain (NDCG).
 281 
 282   For evaluation, the ranked list is truncated at 10 for both metrics. As such,
 283   the HR intuitively measures whether the test item is present on the top-10
 284   list, and the NDCG accounts for the position of the hit by assigning higher
 285   scores to hits at top ranks. Both metrics are calculated for each test user,
 286   and the average scores are reported.
 287 
 288   If `match_mlperf` is True, then the HR and NDCG computations are done in a
 289   slightly unusual way to match the MLPerf reference implementation.
 290   Specifically, if the evaluation negatives contain duplicate items, it will be
 291   treated as if the item only appeared once. Effectively, for duplicate items in
 292   a row, the predicted score for all but one of the items will be set to
 293   -infinity
 294 
 295   For example, suppose we have that following inputs:
 296   logits_by_user:     [[ 2,  3,  3],
 297                        [ 5,  4,  4]]
 298 
 299   items_by_user:     [[10, 20, 20],
 300                       [30, 40, 40]]
 301 
 302   # Note: items_by_user is not explicitly present. Instead the relevant \
 303           information is contained within `duplicate_mask`
 304 
 305   top_k: 2
 306 
 307   Then with match_mlperf=True, the HR would be 2/2 = 1.0. With
 308   match_mlperf=False, the HR would be 1/2 = 0.5. This is because each user has
 309   predicted scores for only 2 unique items: 10 and 20 for the first user, and 30
 310   and 40 for the second. Therefore, with match_mlperf=True, it's guaranteed the
 311   first item's score is in the top 2. With match_mlperf=False, this function
 312   would compute the first user's first item is not in the top 2, because item 20
 313   has a higher score, and item 20 occurs twice.
 314 
 315   Args:
 316     logits: A tensor containing the predicted logits for each user. The shape of
 317       logits is (num_users_per_batch * (1 + NUM_EVAL_NEGATIVES),) Logits for a
 318       user are grouped, and the last element of the group is the true element.
 319     softmax_logits: The same tensor, but with zeros left-appended.
 320     duplicate_mask: A vector with the same shape as logits, with a value of 1 if
 321       the item corresponding to the logit at that position has already appeared
 322       for that user.
 323     num_training_neg: The number of negatives per positive during training.
 324     match_mlperf: Use the MLPerf reference convention for computing rank.
 325 
 326   Returns:
 327     cross_entropy: the loss
 328     metric_fn: the metrics function
 329     in_top_k: hit rate metric
 330     ndcg: ndcg metric
 331     metric_weights: metric weights
 332   """
 333   in_top_k, ndcg, metric_weights, logits_by_user = compute_top_k_and_ndcg(
 334       logits, duplicate_mask, match_mlperf)
 335 
 336   # Examples are provided by the eval Dataset in a structured format, so eval
 337   # labels can be reconstructed on the fly.
 338   eval_labels = tf.reshape(shape=(-1,), tensor=tf.one_hot(
 339       tf.zeros(shape=(logits_by_user.shape[0],), dtype=tf.int32) +
 340       rconst.NUM_EVAL_NEGATIVES, logits_by_user.shape[1], dtype=tf.int32))
 341 
 342   eval_labels_float = tf.cast(eval_labels, tf.float32)
 343 
 344   # During evaluation, the ratio of negatives to positives is much higher
 345   # than during training. (Typically 999 to 1 vs. 4 to 1) By adjusting the
 346   # weights for the negative examples we compute a loss which is consistent with
 347   # the training data. (And provides apples-to-apples comparison)
 348   negative_scale_factor = num_training_neg / rconst.NUM_EVAL_NEGATIVES
 349   example_weights = (
 350       (eval_labels_float + (1 - eval_labels_float) * negative_scale_factor) *
 351       (1 + rconst.NUM_EVAL_NEGATIVES) / (1 + num_training_neg))
 352 
 353   # Tile metric weights back to logit dimensions
 354   expanded_metric_weights = tf.reshape(tf.tile(
 355       metric_weights[:, tf.newaxis], (1, rconst.NUM_EVAL_NEGATIVES + 1)), (-1,))
 356 
 357   # ignore padded examples
 358   example_weights *= tf.cast(expanded_metric_weights, tf.float32)
 359 
 360   cross_entropy = tf.compat.v1.losses.sparse_softmax_cross_entropy(
 361       logits=softmax_logits, labels=eval_labels, weights=example_weights)
 362 
 363   def metric_fn(top_k_tensor, ndcg_tensor, weight_tensor):
 364     return {
 365         rconst.HR_KEY: tf.compat.v1.metrics.mean(top_k_tensor,
 366                                                  weights=weight_tensor,
 367                                                  name=rconst.HR_METRIC_NAME),
 368         rconst.NDCG_KEY: tf.compat.v1.metrics.mean(ndcg_tensor,
 369                                                    weights=weight_tensor,
 370                                                    name=rconst.NDCG_METRIC_NAME)
 371     }
 372 
 373   return cross_entropy, metric_fn, in_top_k, ndcg, metric_weights

 376 def compute_top_k_and_ndcg(logits: tf.Tensor,
 377                            duplicate_mask: tf.Tensor,
 378                            match_mlperf: bool = False):
 379   """Compute inputs of metric calculation.
 380 
 381   Args:
 382     logits: A tensor containing the predicted logits for each user. The shape of
 383       logits is (num_users_per_batch * (1 + NUM_EVAL_NEGATIVES),) Logits for a
 384       user are grouped, and the first element of the group is the true element.
 385     duplicate_mask: A vector with the same shape as logits, with a value of 1 if
 386       the item corresponding to the logit at that position has already appeared
 387       for that user.
 388     match_mlperf: Use the MLPerf reference convention for computing rank.
 389 
 390   Returns:
 391     is_top_k, ndcg and weights, all of which has size (num_users_in_batch,), and
 392     logits_by_user which has size
 393     (num_users_in_batch, (rconst.NUM_EVAL_NEGATIVES + 1)).
 394   """
 395   logits_by_user = tf.reshape(logits, (-1, rconst.NUM_EVAL_NEGATIVES + 1))
 396   duplicate_mask_by_user = tf.cast(
 397       tf.reshape(duplicate_mask, (-1, rconst.NUM_EVAL_NEGATIVES + 1)),
 398       logits_by_user.dtype)
 399 
 400   if match_mlperf:
 401     # Set duplicate logits to the min value for that dtype. The MLPerf
 402     # reference dedupes during evaluation.
 403     logits_by_user *= (1 - duplicate_mask_by_user)
 404     logits_by_user += duplicate_mask_by_user * logits_by_user.dtype.min
 405 
 406   # Determine the location of the first element in each row after the elements
 407   # are sorted.
 408   sort_indices = tf.argsort(
 409       logits_by_user, axis=1, direction="DESCENDING")
 410 
 411   # Use matrix multiplication to extract the position of the true item from the
 412   # tensor of sorted indices. This approach is chosen because both GPUs and TPUs
 413   # perform matrix multiplications very quickly. This is similar to np.argwhere.
 414   # However this is a special case because the target will only appear in
 415   # sort_indices once.
 416   one_hot_position = tf.cast(tf.equal(sort_indices, rconst.NUM_EVAL_NEGATIVES),
 417                              tf.int32)
 418   sparse_positions = tf.multiply(
 419       one_hot_position, tf.range(logits_by_user.shape[1])[tf.newaxis, :])
 420   position_vector = tf.reduce_sum(sparse_positions, axis=1)
 421 
 422   in_top_k = tf.cast(tf.less(position_vector, rconst.TOP_K), tf.float32)
 423   ndcg = tf.math.log(2.) / tf.math.log(
 424       tf.cast(position_vector, tf.float32) + 2)
 425   ndcg *= in_top_k
 426 
 427   # If a row is a padded row, all but the first element will be a duplicate.
 428   metric_weights = tf.not_equal(tf.reduce_sum(duplicate_mask_by_user, axis=1),
 429                                 rconst.NUM_EVAL_NEGATIVES)
 430 
 431   return in_top_k, ndcg, metric_weights, logits_by_user
