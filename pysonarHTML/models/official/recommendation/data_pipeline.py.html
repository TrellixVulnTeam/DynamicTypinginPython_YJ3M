  60 class DatasetManager(object):
  61   """Helper class for handling TensorFlow specific data tasks.
  62 
  63   This class takes the (relatively) framework agnostic work done by the data
  64   constructor classes and handles the TensorFlow specific portions (TFRecord
  65   management, tf.Dataset creation, etc.).
  66   """
  67 
  68   def __init__(self,
  69                is_training,
  70                stream_files,
  71                batches_per_epoch,
  72                shard_root=None,
  73                deterministic=False,
  74                num_train_epochs=None):
  75     # type: (bool, bool, int, typing.Optional[str], bool, int) -> None
  76     """Constructs a `DatasetManager` instance.
  77     Args:
  78       is_training: Boolean of whether the data provided is training or
  79         evaluation data. This determines whether to reuse the data
  80         (if is_training=False) and the exact structure to use when storing and
  81         yielding data.
  82       stream_files: Boolean indicating whether data should be serialized and
  83         written to file shards.
  84       batches_per_epoch: The number of batches in a single epoch.
  85       shard_root: The base directory to be used when stream_files=True.
  86       deterministic: Forgo non-deterministic speedups. (i.e. sloppy=True)
  87       num_train_epochs: Number of epochs to generate. If None, then each
  88         call to `get_dataset()` increments the number of epochs requested.
  89     """
  90     self._is_training = is_training
  91     self._deterministic = deterministic
  92     self._stream_files = stream_files
  93     self._writers = []
  94     self._write_locks = [threading.RLock() for _ in
  95                          range(rconst.NUM_FILE_SHARDS)] if stream_files else []
  96     self._batches_per_epoch = batches_per_epoch
  97     self._epochs_completed = 0
  98     self._epochs_requested = num_train_epochs if num_train_epochs else 0
  99     self._shard_root = shard_root
 100 
 101     self._result_queue = queue.Queue()
 102     self._result_reuse = []
 103 
 104   @property
 105   def current_data_root(self):
 106     subdir = (rconst.TRAIN_FOLDER_TEMPLATE.format(self._epochs_completed)
 107               if self._is_training else rconst.EVAL_FOLDER)
 108     return os.path.join(self._shard_root, subdir)
 109 
 110   def buffer_reached(self):
 111     # Only applicable for training.
 112     return (self._epochs_completed - self._epochs_requested >=
 113             rconst.CYCLES_TO_BUFFER and self._is_training)

 115   @staticmethod
 116   def serialize(data):
 117     """Convert NumPy arrays into a TFRecords entry."""
 118 
 119     def create_int_feature(values):
 120       return tf.train.Feature(int64_list=tf.train.Int64List(value=list(values)))
 121 
 122     feature_dict = {
 123         k: create_int_feature(v.astype(np.int64)) for k, v in data.items()
 124     }
 125 
 126     return tf.train.Example(
 127         features=tf.train.Features(feature=feature_dict)).SerializeToString()
 128 
 129   @staticmethod
 130   def deserialize(serialized_data, batch_size=None, is_training=True):
 131     """Convert serialized TFRecords into tensors.
 132 
 133     Args:
 134       serialized_data: A tensor containing serialized records.
 135       batch_size: The data arrives pre-batched, so batch size is needed to
 136         deserialize the data.
 137       is_training: Boolean, whether data to deserialize to training data
 138         or evaluation data.
 139     """
 140 
 141     def _get_feature_map(batch_size, is_training=True):
 142       """Returns data format of the serialized tf record file."""
 143 
 144       if is_training:
 145         return {
 146             movielens.USER_COLUMN:
 147                 tf.io.FixedLenFeature([batch_size, 1], dtype=tf.int64),
 148             movielens.ITEM_COLUMN:
 149                 tf.io.FixedLenFeature([batch_size, 1], dtype=tf.int64),
 150             rconst.VALID_POINT_MASK:
 151                 tf.io.FixedLenFeature([batch_size, 1], dtype=tf.int64),
 152             "labels":
 153                 tf.io.FixedLenFeature([batch_size, 1], dtype=tf.int64)
 154         }
 155       else:
 156         return {
 157             movielens.USER_COLUMN:
 158                 tf.io.FixedLenFeature([batch_size, 1], dtype=tf.int64),
 159             movielens.ITEM_COLUMN:
 160                 tf.io.FixedLenFeature([batch_size, 1], dtype=tf.int64),
 161             rconst.DUPLICATE_MASK:
 162                 tf.io.FixedLenFeature([batch_size, 1], dtype=tf.int64)
 163         }
 164 
 165     features = tf.io.parse_single_example(
 166         serialized_data, _get_feature_map(batch_size, is_training=is_training))
 167     users = tf.cast(features[movielens.USER_COLUMN], rconst.USER_DTYPE)
 168     items = tf.cast(features[movielens.ITEM_COLUMN], rconst.ITEM_DTYPE)
 169 
 170     if is_training:
 171       valid_point_mask = tf.cast(features[rconst.VALID_POINT_MASK], tf.bool)
 172       fake_dup_mask = tf.zeros_like(users)
 173       return {
 174           movielens.USER_COLUMN: users,
 175           movielens.ITEM_COLUMN: items,
 176           rconst.VALID_POINT_MASK: valid_point_mask,
 177           rconst.TRAIN_LABEL_KEY:
 178               tf.reshape(tf.cast(features["labels"], tf.bool),
 179                          (batch_size, 1)),
 180           rconst.DUPLICATE_MASK: fake_dup_mask
 181       }
 182     else:
 183       labels = tf.cast(tf.zeros_like(users), tf.bool)
 184       fake_valid_pt_mask = tf.cast(tf.zeros_like(users), tf.bool)
 185       return {
 186           movielens.USER_COLUMN:
 187               users,
 188           movielens.ITEM_COLUMN:
 189               items,
 190           rconst.DUPLICATE_MASK:
 191               tf.cast(features[rconst.DUPLICATE_MASK], tf.bool),
 192           rconst.VALID_POINT_MASK:
 193               fake_valid_pt_mask,
 194           rconst.TRAIN_LABEL_KEY:
 195               labels
 196       }
 197 
 198   def put(self, index, data):
 199     # type: (int, dict) -> None
 200     """Store data for later consumption.
 201 
 202     Because there are several paths for storing and yielding data (queues,
 203     lists, files) the data producer simply provides the data in a standard
 204     format at which point the dataset manager handles storing it in the correct
 205     form.
 206 
 207     Args:
 208       index: Used to select shards when writing to files.
 209       data: A dict of the data to be stored. This method mutates data, and
 210         therefore expects to be the only consumer.
 211     """
 212     if self._is_training:
 213       mask_start_index = data.pop(rconst.MASK_START_INDEX)
 214       batch_size = data[movielens.ITEM_COLUMN].shape[0]
 215       data[rconst.VALID_POINT_MASK] = np.expand_dims(
 216           np.less(np.arange(batch_size), mask_start_index), -1)
 217 
 218     if self._stream_files:
 219       example_bytes = self.serialize(data)
 220       with self._write_locks[index % rconst.NUM_FILE_SHARDS]:
 221         self._writers[index % rconst.NUM_FILE_SHARDS].write(example_bytes)
 222 
 223     else:
 224       self._result_queue.put((
 225           data, data.pop("labels")) if self._is_training else data)
 226 
 227   def start_construction(self):
 228     if self._stream_files:
 229       tf.io.gfile.makedirs(self.current_data_root)
 230       template = os.path.join(self.current_data_root, rconst.SHARD_TEMPLATE)
 231       self._writers = [tf.io.TFRecordWriter(template.format(i))
 232                        for i in range(rconst.NUM_FILE_SHARDS)]
 233 
 234   def end_construction(self):
 235     if self._stream_files:
 236       [writer.close() for writer in self._writers]
 237       self._writers = []
 238       self._result_queue.put(self.current_data_root)
 239 
 240     self._epochs_completed += 1
 241 
 242   def data_generator(self, epochs_between_evals):
 243     """Yields examples during local training."""
 244     assert not self._stream_files
 245     assert self._is_training or epochs_between_evals == 1
 246 
 247     if self._is_training:
 248       for _ in range(self._batches_per_epoch * epochs_between_evals):
 249         yield self._result_queue.get(timeout=300)
 250 
 251     else:
 252       if self._result_reuse:
 253         assert len(self._result_reuse) == self._batches_per_epoch
 254 
 255         for i in self._result_reuse:
 256           yield i
 257       else:
 258         # First epoch.
 259         for _ in range(self._batches_per_epoch * epochs_between_evals):
 260           result = self._result_queue.get(timeout=300)
 261           self._result_reuse.append(result)
 262           yield result
 263 
 264   def increment_request_epoch(self):
 265     self._epochs_requested += 1
 266 
 267   def get_dataset(self, batch_size, epochs_between_evals):
 268     """Construct the dataset to be used for training and eval.
 269 
 270     For local training, data is provided through Dataset.from_generator. For
 271     remote training (TPUs) the data is first serialized to files and then sent
 272     to the TPU through a StreamingFilesDataset.
 273 
 274     Args:
 275       batch_size: The per-replica batch size of the dataset.
 276       epochs_between_evals: How many epochs worth of data to yield.
 277         (Generator mode only.)
 278     """
 279     self.increment_request_epoch()
 280     if self._stream_files:
 281       if epochs_between_evals > 1:
 282         raise ValueError("epochs_between_evals > 1 not supported for file "
 283                          "based dataset.")
 284       epoch_data_dir = self._result_queue.get(timeout=300)
 285       if not self._is_training:
 286         self._result_queue.put(epoch_data_dir)  # Eval data is reused.
 287 
 288       file_pattern = os.path.join(
 289           epoch_data_dir, rconst.SHARD_TEMPLATE.format("*"))
 290       dataset = StreamingFilesDataset(
 291           files=file_pattern, worker_job=popen_helper.worker_job(),
 292           num_parallel_reads=rconst.NUM_FILE_SHARDS, num_epochs=1,
 293           sloppy=not self._deterministic)
 294       map_fn = functools.partial(
 295           self.deserialize,
 296           batch_size=batch_size,
 297           is_training=self._is_training)
 298       dataset = dataset.map(map_fn, num_parallel_calls=16)
 299 
 300     else:
 301       types = {movielens.USER_COLUMN: rconst.USER_DTYPE,
 302                movielens.ITEM_COLUMN: rconst.ITEM_DTYPE}
 303       shapes = {
 304           movielens.USER_COLUMN: tf.TensorShape([batch_size, 1]),
 305           movielens.ITEM_COLUMN: tf.TensorShape([batch_size, 1])
 306       }
 307 
 308       if self._is_training:
 309         types[rconst.VALID_POINT_MASK] = np.bool
 310         shapes[rconst.VALID_POINT_MASK] = tf.TensorShape([batch_size, 1])
 311 
 312         types = (types, np.bool)
 313         shapes = (shapes, tf.TensorShape([batch_size, 1]))
 314 
 315       else:
 316         types[rconst.DUPLICATE_MASK] = np.bool
 317         shapes[rconst.DUPLICATE_MASK] = tf.TensorShape([batch_size, 1])
 318 
 319       data_generator = functools.partial(
 320           self.data_generator, epochs_between_evals=epochs_between_evals)
 321       dataset = tf.data.Dataset.from_generator(
 322           generator=data_generator, output_types=types,
 323           output_shapes=shapes)
 324 
 325     return dataset.prefetch(16)
 326 
 327   def make_input_fn(self, batch_size):
 328     """Create an input_fn which checks for batch size consistency."""
 329 
 330     def input_fn(params):
 331       """Returns batches for training."""
 332 
 333       # Estimator passes batch_size during training and eval_batch_size during
 334       # eval.
 335       param_batch_size = (params["batch_size"] if self._is_training else
 336                           params.get("eval_batch_size") or params["batch_size"])
 337       if batch_size != param_batch_size:
 338         raise ValueError("producer batch size ({}) differs from params batch "
 339                          "size ({})".format(batch_size, param_batch_size))
 340 
 341       epochs_between_evals = (params.get("epochs_between_evals", 1)
 342                               if self._is_training else 1)
 343       return self.get_dataset(batch_size=batch_size,
 344                               epochs_between_evals=epochs_between_evals)
 345 
 346     return input_fn

 349 class BaseDataConstructor(threading.Thread):
 350   """Data constructor base class.
 351 
 352   This class manages the control flow for constructing data. It is not meant
 353   to be used directly, but instead subclasses should implement the following
 354   two methods:
 355 
 356     self.construct_lookup_variables
 357     self.lookup_negative_items
 358 
 359   """
 360 
 361   def __init__(
 362       self,
 363       maximum_number_epochs,  # type: int
 364       num_users,  # type: int
 365       num_items,  # type: int
 366       user_map,  # type: dict
 367       item_map,  # type: dict
 368       train_pos_users,  # type: np.ndarray
 369       train_pos_items,  # type: np.ndarray
 370       train_batch_size,  # type: int
 371       batches_per_train_step,  # type: int
 372       num_train_negatives,  # type: int
 373       eval_pos_users,  # type: np.ndarray
 374       eval_pos_items,  # type: np.ndarray
 375       eval_batch_size,  # type: int
 376       batches_per_eval_step,  # type: int
 377       stream_files,  # type: bool
 378       deterministic=False,  # type: bool
 379       epoch_dir=None,  # type: str
 380       num_train_epochs=None,  # type: int
 381       create_data_offline=False  # type: bool
 382   ):
 383     # General constants
 384     self._maximum_number_epochs = maximum_number_epochs
 385     self._num_users = num_users
 386     self._num_items = num_items
 387     self.user_map = user_map
 388     self.item_map = item_map
 389     self._train_pos_users = train_pos_users
 390     self._train_pos_items = train_pos_items
 391     self.train_batch_size = train_batch_size
 392     self._num_train_negatives = num_train_negatives
 393     self._batches_per_train_step = batches_per_train_step
 394     self._eval_pos_users = eval_pos_users
 395     self._eval_pos_items = eval_pos_items
 396     self.eval_batch_size = eval_batch_size
 397     self.num_train_epochs = num_train_epochs
 398     self.create_data_offline = create_data_offline
 399 
 400     # Training
 401     if self._train_pos_users.shape != self._train_pos_items.shape:
 402       raise ValueError(
 403           "User positives ({}) is different from item positives ({})".format(
 404               self._train_pos_users.shape, self._train_pos_items.shape))
 405 
 406     (self._train_pos_count,) = self._train_pos_users.shape
 407     self._elements_in_epoch = (1 + num_train_negatives) * self._train_pos_count
 408     self.train_batches_per_epoch = self._count_batches(
 409         self._elements_in_epoch, train_batch_size, batches_per_train_step)
 410 
 411     # Evaluation
 412     if eval_batch_size % (1 + rconst.NUM_EVAL_NEGATIVES):
 413       raise ValueError("Eval batch size {} is not divisible by {}".format(
 414           eval_batch_size, 1 + rconst.NUM_EVAL_NEGATIVES))
 415     self._eval_users_per_batch = int(
 416         eval_batch_size // (1 + rconst.NUM_EVAL_NEGATIVES))
 417     self._eval_elements_in_epoch = num_users * (1 + rconst.NUM_EVAL_NEGATIVES)
 418     self.eval_batches_per_epoch = self._count_batches(
 419         self._eval_elements_in_epoch, eval_batch_size, batches_per_eval_step)
 420 
 421     # Intermediate artifacts
 422     self._current_epoch_order = np.empty(shape=(0,))
 423     self._shuffle_iterator = None
 424 
 425     self._shuffle_with_forkpool = not stream_files
 426     if stream_files:
 427       self._shard_root = epoch_dir or tempfile.mkdtemp(prefix="ncf_")
 428       if not create_data_offline:
 429         atexit.register(tf.io.gfile.rmtree, self._shard_root)
 430     else:
 431       self._shard_root = None
 432 
 433     self._train_dataset = DatasetManager(True, stream_files,
 434                                          self.train_batches_per_epoch,
 435                                          self._shard_root, deterministic,
 436                                          num_train_epochs)
 437     self._eval_dataset = DatasetManager(False, stream_files,
 438                                         self.eval_batches_per_epoch,
 439                                         self._shard_root, deterministic,
 440                                         num_train_epochs)
 441 
 442     # Threading details
 443     super(BaseDataConstructor, self).__init__()
 444     self.daemon = True
 445     self._stop_loop = False
 446     self._fatal_exception = None
 447     self.deterministic = deterministic
 448 
 449   def __str__(self):
 450     multiplier = ("(x{} devices)".format(self._batches_per_train_step)
 451                   if self._batches_per_train_step > 1 else "")
 452     summary = SUMMARY_TEMPLATE.format(
 453         spacer="  ", num_users=self._num_users, num_items=self._num_items,
 454         train_pos_ct=self._train_pos_count,
 455         train_batch_size=self.train_batch_size,
 456         train_batch_ct=self.train_batches_per_epoch,
 457         eval_pos_ct=self._num_users, eval_batch_size=self.eval_batch_size,
 458         eval_batch_ct=self.eval_batches_per_epoch, multiplier=multiplier)
 459     return super(BaseDataConstructor, self).__str__() + "\n" + summary
 460 
 461   @staticmethod
 462   def _count_batches(example_count, batch_size, batches_per_step):
 463     """Determine the number of batches, rounding up to fill all devices."""
 464     x = (example_count + batch_size - 1) // batch_size
 465     return (x + batches_per_step - 1) // batches_per_step * batches_per_step
 466 
 467   def stop_loop(self):
 468     self._stop_loop = True
 469 
 470   def construct_lookup_variables(self):
 471     """Perform any one time pre-compute work."""
 472     raise NotImplementedError
 473 
 474   def lookup_negative_items(self, **kwargs):
 475     """Randomly sample negative items for given users."""
 476     raise NotImplementedError
 477 
 478   def _run(self):
 479     atexit.register(self.stop_loop)
 480     self._start_shuffle_iterator()
 481     self.construct_lookup_variables()
 482     self._construct_training_epoch()
 483     self._construct_eval_epoch()
 484     for _ in range(self._maximum_number_epochs - 1):
 485       self._construct_training_epoch()
 486     self.stop_loop()
 487 
 488   def run(self):
 489     try:
 490       self._run()
 491     except Exception as e:
 492       # The Thread base class swallows stack traces, so unfortunately it is
 493       # necessary to catch and re-raise to get debug output
 494       traceback.print_exc()
 495       self._fatal_exception = e
 496       sys.stderr.flush()
 497       raise
 498 
 499   def _start_shuffle_iterator(self):
 500     if self._shuffle_with_forkpool:
 501       pool = popen_helper.get_forkpool(3, closing=False)
 502     else:
 503       pool = popen_helper.get_threadpool(1, closing=False)
 504     atexit.register(pool.close)
 505     args = [(self._elements_in_epoch, stat_utils.random_int32())
 506             for _ in range(self._maximum_number_epochs)]
 507     imap = pool.imap if self.deterministic else pool.imap_unordered
 508     self._shuffle_iterator = imap(stat_utils.permutation, args)
 509 
 510   def _get_training_batch(self, i):
 511     """Construct a single batch of training data.
 512 
 513     Args:
 514       i: The index of the batch. This is used when stream_files=True to assign
 515         data to file shards.
 516     """
 517     batch_indices = self._current_epoch_order[i * self.train_batch_size:
 518                                               (i + 1) * self.train_batch_size]
 519     (mask_start_index,) = batch_indices.shape
 520 
 521     batch_ind_mod = np.mod(batch_indices, self._train_pos_count)
 522     users = self._train_pos_users[batch_ind_mod]
 523 
 524     negative_indices = np.greater_equal(batch_indices, self._train_pos_count)
 525     negative_users = users[negative_indices]
 526 
 527     negative_items = self.lookup_negative_items(negative_users=negative_users)
 528 
 529     items = self._train_pos_items[batch_ind_mod]
 530     items[negative_indices] = negative_items
 531 
 532     labels = np.logical_not(negative_indices)
 533 
 534     # Pad last partial batch
 535     pad_length = self.train_batch_size - mask_start_index
 536     if pad_length:
 537       # We pad with arange rather than zeros because the network will still
 538       # compute logits for padded examples, and padding with zeros would create
 539       # a very "hot" embedding key which can have performance implications.
 540       user_pad = np.arange(pad_length, dtype=users.dtype) % self._num_users
 541       item_pad = np.arange(pad_length, dtype=items.dtype) % self._num_items
 542       label_pad = np.zeros(shape=(pad_length,), dtype=labels.dtype)
 543       users = np.concatenate([users, user_pad])
 544       items = np.concatenate([items, item_pad])
 545       labels = np.concatenate([labels, label_pad])
 546 
 547     self._train_dataset.put(
 548         i, {
 549             movielens.USER_COLUMN:
 550                 np.reshape(users, (self.train_batch_size, 1)),
 551             movielens.ITEM_COLUMN:
 552                 np.reshape(items, (self.train_batch_size, 1)),
 553             rconst.MASK_START_INDEX:
 554                 np.array(mask_start_index, dtype=np.int32),
 555             "labels":
 556                 np.reshape(labels, (self.train_batch_size, 1)),
 557         })
 558 
 559   def _wait_to_construct_train_epoch(self):
 560     count = 0
 561     while self._train_dataset.buffer_reached() and not self._stop_loop:
 562       time.sleep(0.01)
 563       count += 1
 564       if count >= 100 and np.log10(count) == np.round(np.log10(count)):
 565         logging.info(
 566             "Waited {} times for training data to be consumed".format(count))

 590   @staticmethod
 591   def _assemble_eval_batch(users, positive_items, negative_items,
 592                            users_per_batch):
 593     """Construct duplicate_mask and structure data accordingly.
 594 
 595     The positive items should be last so that they lose ties. However, they
 596     should not be masked out if the true eval positive happens to be
 597     selected as a negative. So instead, the positive is placed in the first
 598     position, and then switched with the last element after the duplicate
 599     mask has been computed.
 600 
 601     Args:
 602       users: An array of users in a batch. (should be identical along axis 1)
 603       positive_items: An array (batch_size x 1) of positive item indices.
 604       negative_items: An array of negative item indices.
 605       users_per_batch: How many users should be in the batch. This is passed
 606         as an argument so that ncf_test.py can use this method.
 607 
 608     Returns:
 609       User, item, and duplicate_mask arrays.
 610     """
 611     items = np.concatenate([positive_items, negative_items], axis=1)
 612 
 613     # We pad the users and items here so that the duplicate mask calculation
 614     # will include padding. The metric function relies on all padded elements
 615     # except the positive being marked as duplicate to mask out padded points.
 616     if users.shape[0] < users_per_batch:
 617       pad_rows = users_per_batch - users.shape[0]
 618       padding = np.zeros(shape=(pad_rows, users.shape[1]), dtype=np.int32)
 619       users = np.concatenate([users, padding.astype(users.dtype)], axis=0)
 620       items = np.concatenate([items, padding.astype(items.dtype)], axis=0)
 621 
 622     duplicate_mask = stat_utils.mask_duplicates(items, axis=1).astype(np.bool)
 623 
 624     items[:, (0, -1)] = items[:, (-1, 0)]
 625     duplicate_mask[:, (0, -1)] = duplicate_mask[:, (-1, 0)]
 626 
 627     assert users.shape == items.shape == duplicate_mask.shape
 628     return users, items, duplicate_mask
 629 
 630   def _get_eval_batch(self, i):
 631     """Construct a single batch of evaluation data.
 632 
 633     Args:
 634       i: The index of the batch.
 635     """
 636     low_index = i * self._eval_users_per_batch
 637     high_index = (i + 1) * self._eval_users_per_batch
 638     users = np.repeat(self._eval_pos_users[low_index:high_index, np.newaxis],
 639                       1 + rconst.NUM_EVAL_NEGATIVES, axis=1)
 640     positive_items = self._eval_pos_items[low_index:high_index, np.newaxis]
 641     negative_items = (self.lookup_negative_items(negative_users=users[:, :-1])
 642                       .reshape(-1, rconst.NUM_EVAL_NEGATIVES))
 643 
 644     users, items, duplicate_mask = self._assemble_eval_batch(
 645         users, positive_items, negative_items, self._eval_users_per_batch)
 646 
 647     self._eval_dataset.put(
 648         i, {
 649             movielens.USER_COLUMN:
 650                 np.reshape(users.flatten(), (self.eval_batch_size, 1)),
 651             movielens.ITEM_COLUMN:
 652                 np.reshape(items.flatten(), (self.eval_batch_size, 1)),
 653             rconst.DUPLICATE_MASK:
 654                 np.reshape(duplicate_mask.flatten(), (self.eval_batch_size, 1)),
 655         })
 656

 767 class MaterializedDataConstructor(BaseDataConstructor):
 768   """Materialize a table of negative examples for fast negative generation.
 769 
 770   This class creates a table (num_users x num_items) containing all of the
 771   negative examples for each user. This table is conceptually ragged; that is to
 772   say the items dimension will have a number of unused elements at the end equal
 773   to the number of positive elements for a given user. For instance:
 774 
 775   num_users = 3
 776   num_items = 5
 777   positives = [[1, 3], [0], [1, 2, 3, 4]]
 778 
 779   will generate a negative table:
 780   [
 781     [0         2         4         int32max  int32max],
 782     [1         2         3         4         int32max],
 783     [0         int32max  int32max  int32max  int32max],
 784   ]
 785 
 786   and a vector of per-user negative counts, which in this case would be:
 787     [3, 4, 1]
 788 
 789   When sampling negatives, integers are (nearly) uniformly selected from the
 790   range [0, per_user_neg_count[user]) which gives a column_index, at which
 791   point the negative can be selected as:
 792     negative_table[user, column_index]
 793 
 794   This technique will not scale; however MovieLens is small enough that even
 795   a pre-compute which is quadratic in problem size will still fit in memory. A
 796   more scalable lookup method is in the works.
 797   """
 798   def __init__(self, *args, **kwargs):
 799     super(MaterializedDataConstructor, self).__init__(*args, **kwargs)
 800     self._negative_table = None
 801     self._per_user_neg_count = None
 802 
 803   def construct_lookup_variables(self):
 804     # Materialize negatives for fast lookup sampling.
 805     start_time = timeit.default_timer()
 806     inner_bounds = np.argwhere(self._train_pos_users[1:] -
 807                                self._train_pos_users[:-1])[:, 0] + 1
 808     (upper_bound,) = self._train_pos_users.shape
 809     index_bounds = [0] + inner_bounds.tolist() + [upper_bound]
 810     self._negative_table = np.zeros(shape=(self._num_users, self._num_items),
 811                                     dtype=rconst.ITEM_DTYPE)
 812 
 813     # Set the table to the max value to make sure the embedding lookup will fail
 814     # if we go out of bounds, rather than just overloading item zero.
 815     self._negative_table += np.iinfo(rconst.ITEM_DTYPE).max
 816     assert self._num_items < np.iinfo(rconst.ITEM_DTYPE).max
 817 
 818     # Reuse arange during generation. np.delete will make a copy.
 819     full_set = np.arange(self._num_items, dtype=rconst.ITEM_DTYPE)
 820 
 821     self._per_user_neg_count = np.zeros(
 822         shape=(self._num_users,), dtype=np.int32)
 823 
 824     # Threading does not improve this loop. For some reason, the np.delete
 825     # call does not parallelize well. Multiprocessing incurs too much
 826     # serialization overhead to be worthwhile.
 827     for i in range(self._num_users):
 828       positives = self._train_pos_items[index_bounds[i]:index_bounds[i+1]]
 829       negatives = np.delete(full_set, positives)
 830       self._per_user_neg_count[i] = self._num_items - positives.shape[0]
 831       self._negative_table[i, :self._per_user_neg_count[i]] = negatives
 832 
 833     logging.info("Negative sample table built. Time: {:.1f} seconds".format(
 834         timeit.default_timer() - start_time))
 835 
 836   def lookup_negative_items(self, negative_users, **kwargs):
 837     negative_item_choice = stat_utils.very_slightly_biased_randint(
 838         self._per_user_neg_count[negative_users])
 839     return self._negative_table[negative_users, negative_item_choice]

 842 class BisectionDataConstructor(BaseDataConstructor):
 843   """Use bisection to index within positive examples.
 844 
 845   This class tallies the number of negative items which appear before each
 846   positive item for a user. This means that in order to select the ith negative
 847   item for a user, it only needs to determine which two positive items bound
 848   it at which point the item id for the ith negative is a simply algebraic
 849   expression.
 850   """
 851   def __init__(self, *args, **kwargs):
 852     super(BisectionDataConstructor, self).__init__(*args, **kwargs)
 853     self.index_bounds = None
 854     self._sorted_train_pos_items = None
 855     self._total_negatives = None
 856 
 857   def _index_segment(self, user):
 858     lower, upper = self.index_bounds[user:user+2]
 859     items = self._sorted_train_pos_items[lower:upper]
 860 
 861     negatives_since_last_positive = np.concatenate(
 862         [items[0][np.newaxis], items[1:] - items[:-1] - 1])
 863 
 864     return np.cumsum(negatives_since_last_positive)
 865 
 866   def construct_lookup_variables(self):
 867     start_time = timeit.default_timer()
 868     inner_bounds = np.argwhere(self._train_pos_users[1:] -
 869                                self._train_pos_users[:-1])[:, 0] + 1
 870     (upper_bound,) = self._train_pos_users.shape
 871     self.index_bounds = np.array([0] + inner_bounds.tolist() + [upper_bound])
 872 
 873     # Later logic will assume that the users are in sequential ascending order.
 874     assert np.array_equal(self._train_pos_users[self.index_bounds[:-1]],
 875                           np.arange(self._num_users))
 876 
 877     self._sorted_train_pos_items = self._train_pos_items.copy()
 878 
 879     for i in range(self._num_users):
 880       lower, upper = self.index_bounds[i:i+2]
 881       self._sorted_train_pos_items[lower:upper].sort()
 882 
 883     self._total_negatives = np.concatenate([
 884         self._index_segment(i) for i in range(self._num_users)])
 885 
 886     logging.info("Negative total vector built. Time: {:.1f} seconds".format(
 887         timeit.default_timer() - start_time))
 888 
 889   def lookup_negative_items(self, negative_users, **kwargs):
 890     output = np.zeros(shape=negative_users.shape, dtype=rconst.ITEM_DTYPE) - 1
 891 
 892     left_index = self.index_bounds[negative_users]
 893     right_index = self.index_bounds[negative_users + 1] - 1
 894 
 895     num_positives = right_index - left_index + 1
 896     num_negatives = self._num_items - num_positives
 897     neg_item_choice = stat_utils.very_slightly_biased_randint(num_negatives)
 898 
 899     # Shortcuts:
 900     # For points where the negative is greater than or equal to the tally before
 901     # the last positive point there is no need to bisect. Instead the item id
 902     # corresponding to the negative item choice is simply:
 903     #   last_postive_index + 1 + (neg_choice - last_negative_tally)
 904     # Similarly, if the selection is less than the tally at the first positive
 905     # then the item_id is simply the selection.
 906     #
 907     # Because MovieLens organizes popular movies into low integers (which is
 908     # preserved through the preprocessing), the first shortcut is very
 909     # efficient, allowing ~60% of samples to bypass the bisection. For the same
 910     # reason, the second shortcut is rarely triggered (<0.02%) and is therefore
 911     # not worth implementing.
 912     use_shortcut = neg_item_choice >= self._total_negatives[right_index]
 913     output[use_shortcut] = (
 914         self._sorted_train_pos_items[right_index] + 1 +
 915         (neg_item_choice - self._total_negatives[right_index])
 916     )[use_shortcut]
 917 
 918     if np.all(use_shortcut):
 919       # The bisection code is ill-posed when there are no elements.
 920       return output
 921 
 922     not_use_shortcut = np.logical_not(use_shortcut)
 923     left_index = left_index[not_use_shortcut]
 924     right_index = right_index[not_use_shortcut]
 925     neg_item_choice = neg_item_choice[not_use_shortcut]
 926 
 927     num_loops = np.max(
 928         np.ceil(np.log2(num_positives[not_use_shortcut])).astype(np.int32))
 929 
 930     for i in range(num_loops):
 931       mid_index = (left_index + right_index) // 2
 932       right_criteria = self._total_negatives[mid_index] > neg_item_choice
 933       left_criteria = np.logical_not(right_criteria)
 934 
 935       right_index[right_criteria] = mid_index[right_criteria]
 936       left_index[left_criteria] = mid_index[left_criteria]
 937 
 938     # Expected state after bisection pass:
 939     #   The right index is the smallest index whose tally is greater than the
 940     #   negative item choice index.
 941 
 942     assert np.all((right_index - left_index) <= 1)
 943 
 944     output[not_use_shortcut] = (
 945         self._sorted_train_pos_items[right_index] -
 946         (self._total_negatives[right_index] - neg_item_choice)
 947     )
 948 
 949     assert np.all(output >= 0)
 950 
 951     return output
