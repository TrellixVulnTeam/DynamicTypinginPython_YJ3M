  34 def create_dataset_from_tf_record_files(input_file_pattern,
  35                                         pre_batch_size,
  36                                         batch_size,
  37                                         is_training=True):
  38   """Creates dataset from (tf)records files for training/evaluation."""
  39 
  40   files = tf.data.Dataset.list_files(input_file_pattern, shuffle=is_training)
  41 
  42   def make_dataset(files_dataset, shard_index):
  43     """Returns dataset for sharded tf record files."""
  44     if pre_batch_size != batch_size:
  45       raise ValueError("Pre-batch ({}) size is not equal to batch "
  46                        "size ({})".format(pre_batch_size, batch_size))
  47     files_dataset = files_dataset.shard(NUM_SHARDS, shard_index)
  48     dataset = files_dataset.interleave(
  49         tf.data.TFRecordDataset,
  50         num_parallel_calls=tf.data.experimental.AUTOTUNE)
  51     decode_fn = functools.partial(
  52         data_pipeline.DatasetManager.deserialize,
  53         batch_size=pre_batch_size,
  54         is_training=is_training)
  55     dataset = dataset.map(
  56         decode_fn, num_parallel_calls=tf.data.experimental.AUTOTUNE)
  57     return dataset
  58 
  59   dataset = tf.data.Dataset.range(NUM_SHARDS)
  60   map_fn = functools.partial(make_dataset, files)
  61   dataset = dataset.interleave(
  62       map_fn,
  63       cycle_length=NUM_SHARDS,
  64       num_parallel_calls=tf.data.experimental.AUTOTUNE)
  65   dataset = dataset.prefetch(tf.data.experimental.AUTOTUNE)
  66   return dataset

  69 def create_dataset_from_data_producer(producer, params):
  70   """Return dataset online-generating data."""
  71 
  72   def preprocess_train_input(features, labels):
  73     """Pre-process the training data.
  74 
  75     This is needed because
  76     - The label needs to be extended to be used in the loss fn
  77     - We need the same inputs for training and eval so adding fake inputs
  78       for DUPLICATE_MASK in training data.
  79 
  80     Args:
  81       features: Dictionary of features for training.
  82       labels: Training labels.
  83 
  84     Returns:
  85       Processed training features.
  86     """
  87     fake_dup_mask = tf.zeros_like(features[movielens.USER_COLUMN])
  88     features[rconst.DUPLICATE_MASK] = fake_dup_mask
  89     features[rconst.TRAIN_LABEL_KEY] = labels
  90     return features
  91 
  92   train_input_fn = producer.make_input_fn(is_training=True)
  93   train_input_dataset = train_input_fn(params).map(preprocess_train_input)
  94 
  95   def preprocess_eval_input(features):
  96     """Pre-process the eval data.
  97 
  98     This is needed because:
  99     - The label needs to be extended to be used in the loss fn
 100     - We need the same inputs for training and eval so adding fake inputs
 101       for VALID_PT_MASK in eval data.
 102 
 103     Args:
 104       features: Dictionary of features for evaluation.
 105 
 106     Returns:
 107       Processed evaluation features.
 108     """
 109     labels = tf.cast(tf.zeros_like(features[movielens.USER_COLUMN]), tf.bool)
 110     fake_valid_pt_mask = tf.cast(
 111         tf.zeros_like(features[movielens.USER_COLUMN]), tf.bool)
 112     features[rconst.VALID_POINT_MASK] = fake_valid_pt_mask
 113     features[rconst.TRAIN_LABEL_KEY] = labels
 114     return features
 115 
 116   eval_input_fn = producer.make_input_fn(is_training=False)
 117   eval_input_dataset = eval_input_fn(params).map(preprocess_eval_input)
 118 
 119   return train_input_dataset, eval_input_dataset

 122 def create_ncf_input_data(params,
 123                           producer=None,
 124                           input_meta_data=None,
 125                           strategy=None):
 126   """Creates NCF training/evaluation dataset.
 127 
 128   Args:
 129     params: Dictionary containing parameters for train/evaluation data.
 130     producer: Instance of BaseDataConstructor that generates data online. Must
 131       not be None when params['train_dataset_path'] or
 132       params['eval_dataset_path'] is not specified.
 133     input_meta_data: A dictionary of input metadata to be used when reading data
 134       from tf record files. Must be specified when params["train_input_dataset"]
 135       is specified.
 136     strategy: Distribution strategy used for distributed training. If specified,
 137       used to assert that evaluation batch size is correctly a multiple of
 138       total number of devices used.
 139 
 140   Returns:
 141     (training dataset, evaluation dataset, train steps per epoch,
 142     eval steps per epoch)
 143 
 144   Raises:
 145     ValueError: If data is being generated online for when using TPU's.
 146   """
 147   # NCF evaluation metric calculation logic assumes that evaluation data
 148   # sample size are in multiples of (1 + number of negative samples in
 149   # evaluation) for each device. As so, evaluation batch size must be a
 150   # multiple of (number of replicas * (1 + number of negative samples)).
 151   num_devices = strategy.num_replicas_in_sync if strategy else 1
 152   if (params["eval_batch_size"] % (num_devices *
 153                                    (1 + rconst.NUM_EVAL_NEGATIVES))):
 154     raise ValueError("Evaluation batch size must be divisible by {} "
 155                      "times {}".format(num_devices,
 156                                        (1 + rconst.NUM_EVAL_NEGATIVES)))
 157 
 158   if params["train_dataset_path"]:
 159     assert params["eval_dataset_path"]
 160 
 161     train_dataset = create_dataset_from_tf_record_files(
 162         params["train_dataset_path"],
 163         input_meta_data["train_prebatch_size"],
 164         params["batch_size"],
 165         is_training=True)
 166     eval_dataset = create_dataset_from_tf_record_files(
 167         params["eval_dataset_path"],
 168         input_meta_data["eval_prebatch_size"],
 169         params["eval_batch_size"],
 170         is_training=False)
 171 
 172     num_train_steps = int(input_meta_data["num_train_steps"])
 173     num_eval_steps = int(input_meta_data["num_eval_steps"])
 174   else:
 175     if params["use_tpu"]:
 176       raise ValueError("TPU training does not support data producer yet. "
 177                        "Use pre-processed data.")
 178 
 179     assert producer
 180     # Start retrieving data from producer.
 181     train_dataset, eval_dataset = create_dataset_from_data_producer(
 182         producer, params)
 183     num_train_steps = producer.train_batches_per_epoch
 184     num_eval_steps = producer.eval_batches_per_epoch
 185 
 186   return train_dataset, eval_dataset, num_train_steps, num_eval_steps
