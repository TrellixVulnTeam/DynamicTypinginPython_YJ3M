  43 def get_inputs(params):
  44   """Returns some parameters used by the model."""
  45   if FLAGS.download_if_missing and not FLAGS.use_synthetic_data:
  46     movielens.download(FLAGS.dataset, FLAGS.data_dir)
  47 
  48   if FLAGS.seed is not None:
  49     np.random.seed(FLAGS.seed)
  50 
  51   if FLAGS.use_synthetic_data:
  52     producer = data_pipeline.DummyConstructor()
  53     num_users, num_items = movielens.DATASET_TO_NUM_USERS_AND_ITEMS[
  54         FLAGS.dataset]
  55     num_train_steps = rconst.SYNTHETIC_BATCHES_PER_EPOCH
  56     num_eval_steps = rconst.SYNTHETIC_BATCHES_PER_EPOCH
  57   else:
  58     num_users, num_items, producer = data_preprocessing.instantiate_pipeline(
  59         dataset=FLAGS.dataset, data_dir=FLAGS.data_dir, params=params,
  60         constructor_type=FLAGS.constructor_type,
  61         deterministic=FLAGS.seed is not None)
  62     num_train_steps = producer.train_batches_per_epoch
  63     num_eval_steps = producer.eval_batches_per_epoch
  64 
  65   return num_users, num_items, num_train_steps, num_eval_steps, producer

  68 def parse_flags(flags_obj):
  69   """Convenience function to turn flags into params."""
  70   num_gpus = flags_core.get_num_gpus(flags_obj)
  71 
  72   batch_size = flags_obj.batch_size
  73   eval_batch_size = flags_obj.eval_batch_size or flags_obj.batch_size
  74 
  75   return {
  76       "train_epochs": flags_obj.train_epochs,
  77       "batches_per_step": 1,
  78       "use_seed": flags_obj.seed is not None,
  79       "batch_size": batch_size,
  80       "eval_batch_size": eval_batch_size,
  81       "learning_rate": flags_obj.learning_rate,
  82       "mf_dim": flags_obj.num_factors,
  83       "model_layers": [int(layer) for layer in flags_obj.layers],
  84       "mf_regularization": flags_obj.mf_regularization,
  85       "mlp_reg_layers": [float(reg) for reg in flags_obj.mlp_regularization],
  86       "num_neg": flags_obj.num_neg,
  87       "distribution_strategy": flags_obj.distribution_strategy,
  88       "num_gpus": num_gpus,
  89       "use_tpu": flags_obj.tpu is not None,
  90       "tpu": flags_obj.tpu,
  91       "tpu_zone": flags_obj.tpu_zone,
  92       "tpu_gcp_project": flags_obj.tpu_gcp_project,
  93       "beta1": flags_obj.beta1,
  94       "beta2": flags_obj.beta2,
  95       "epsilon": flags_obj.epsilon,
  96       "match_mlperf": flags_obj.ml_perf,
  97       "epochs_between_evals": FLAGS.epochs_between_evals,
  98       "keras_use_ctl": flags_obj.keras_use_ctl,
  99       "hr_threshold": flags_obj.hr_threshold,
 100       "stream_files": flags_obj.tpu is not None,
 101       "train_dataset_path": flags_obj.train_dataset_path,
 102       "eval_dataset_path": flags_obj.eval_dataset_path,
 103       "input_meta_data_path": flags_obj.input_meta_data_path,
 104   }

 107 def get_v1_distribution_strategy(params):
 108   """Returns the distribution strategy to use."""
 109   if params["use_tpu"]:
 110     # Some of the networking libraries are quite chatty.
 111     for name in ["googleapiclient.discovery", "googleapiclient.discovery_cache",
 112                  "oauth2client.transport"]:
 113       logging.getLogger(name).setLevel(logging.ERROR)
 114 
 115     tpu_cluster_resolver = tf.distribute.cluster_resolver.TPUClusterResolver(
 116         tpu=params["tpu"],
 117         zone=params["tpu_zone"],
 118         project=params["tpu_gcp_project"],
 119         coordinator_name="coordinator"
 120     )
 121 
 122     logging.info("Issuing reset command to TPU to ensure a clean state.")
 123     tf.Session.reset(tpu_cluster_resolver.get_master())
 124 
 125     # Estimator looks at the master it connects to for MonitoredTrainingSession
 126     # by reading the `TF_CONFIG` environment variable, and the coordinator
 127     # is used by StreamingFilesDataset.
 128     tf_config_env = {
 129         "session_master": tpu_cluster_resolver.get_master(),
 130         "eval_session_master": tpu_cluster_resolver.get_master(),
 131         "coordinator": tpu_cluster_resolver.cluster_spec()
 132                        .as_dict()["coordinator"]
 133     }
 134     os.environ["TF_CONFIG"] = json.dumps(tf_config_env)
 135 
 136     distribution = tf.distribute.experimental.TPUStrategy(
 137         tpu_cluster_resolver, steps_per_run=100)
 138 
 139   else:
 140     distribution = distribution_utils.get_distribution_strategy(
 141         num_gpus=params["num_gpus"])
 142 
 143   return distribution

 158       dynamic_loss_scale=True,
 159       enable_xla=True,
 160   )
 161   flags_core.define_device(tpu=True)

 169       model_dir="/tmp/ncf/",

 317 def convert_to_softmax_logits(logits):
 318   """Convert the logits returned by the base model to softmax logits.
 319 
 320   Args:
 321     logits: used to create softmax.
 322 
 323   Returns:
 324     Softmax with the first column of zeros is equivalent to sigmoid.
 325   """
 326   softmax_logits = tf.concat([logits * 0, logits], axis=1)
 327   return softmax_logits
