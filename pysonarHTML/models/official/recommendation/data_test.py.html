  54 # The forkpool used by data producers interacts badly with the threading
  55 # used by TestCase. Without this patch tests will hang, and no amount

  60   def setUp(self):
  61     tf.compat.v1.disable_eager_execution()
  62     self.temp_data_dir = self.get_temp_dir()
  63     ratings_folder = os.path.join(self.temp_data_dir, DATASET)
  64     tf.io.gfile.makedirs(ratings_folder)
  65     np.random.seed(0)
  66     raw_user_ids = np.arange(NUM_USERS * 3)
  67     np.random.shuffle(raw_user_ids)
  68     raw_user_ids = raw_user_ids[:NUM_USERS]
  69 
  70     raw_item_ids = np.arange(NUM_ITEMS * 3)
  71     np.random.shuffle(raw_item_ids)
  72     raw_item_ids = raw_item_ids[:NUM_ITEMS]
  73 
  74     users = np.random.choice(raw_user_ids, NUM_PTS)
  75     items = np.random.choice(raw_item_ids, NUM_PTS)
  76     scores = np.random.randint(low=0, high=5, size=NUM_PTS)
  77     times = np.random.randint(low=1000000000, high=1200000000, size=NUM_PTS)
  78 
  79     self.rating_file = os.path.join(ratings_folder, movielens.RATINGS_FILE)
  80     self.seen_pairs = set()
  81     self.holdout = {}
  82     with tf.io.gfile.GFile(self.rating_file, "w") as f:
  83       f.write("user_id,item_id,rating,timestamp\n")
  84       for usr, itm, scr, ts in zip(users, items, scores, times):
  85         pair = (usr, itm)
  86         if pair in self.seen_pairs:
  87           continue
  88         self.seen_pairs.add(pair)
  89         if usr not in self.holdout or (ts, itm) > self.holdout[usr]:
  90           self.holdout[usr] = (ts, itm)
  91 
  92         f.write("{},{},{},{}\n".format(usr, itm, scr, ts))
  93 
  94     movielens.download = mock_download
  95     movielens.NUM_RATINGS[DATASET] = NUM_PTS
  96     movielens.DATASET_TO_NUM_USERS_AND_ITEMS[DATASET] = (NUM_USERS, NUM_ITEMS)

 123   def drain_dataset(self, dataset, g):
 124     # type: (tf.data.Dataset, tf.Graph) -> list
 125     with self.session(graph=g) as sess:
 126       with g.as_default():
 127         batch = tf.compat.v1.data.make_one_shot_iterator(dataset).get_next()
 128       output = []
 129       while True:
 130         try:
 131           output.append(sess.run(batch))
 132         except tf.errors.OutOfRangeError:
 133           break
 134     return output

 136   def _test_end_to_end(self, constructor_type):
 137     params = self.make_params(train_epochs=1)
 138     _, _, producer = data_preprocessing.instantiate_pipeline(
 139         dataset=DATASET, data_dir=self.temp_data_dir, params=params,
 140         constructor_type=constructor_type, deterministic=True)
 141 
 142     producer.start()
 143     producer.join()
 144     assert producer._fatal_exception is None
 145 
 146     user_inv_map = {v: k for k, v in producer.user_map.items()}
 147     item_inv_map = {v: k for k, v in producer.item_map.items()}
 148 
 149     # ==========================================================================
 150     # == Training Data =========================================================
 151     # ==========================================================================
 152     g = tf.Graph()
 153     with g.as_default():
 154       input_fn = producer.make_input_fn(is_training=True)
 155       dataset = input_fn(params)
 156 
 157     first_epoch = self.drain_dataset(dataset=dataset, g=g)
 158 
 159     counts = defaultdict(int)
 160     train_examples = {
 161         True: set(),
 162         False: set(),
 163     }
 164 
 165     md5 = hashlib.md5()
 166     for features, labels in first_epoch:
 167       data_list = [
 168           features[movielens.USER_COLUMN].flatten(),
 169           features[movielens.ITEM_COLUMN].flatten(),
 170           features[rconst.VALID_POINT_MASK].flatten(),
 171           labels.flatten()
 172       ]
 173       for i in data_list:
 174         md5.update(i.tobytes())
 175 
 176       for u, i, v, l in zip(*data_list):
 177         if not v:
 178           continue  # ignore padding
 179 
 180         u_raw = user_inv_map[u]
 181         i_raw = item_inv_map[i]
 182         if ((u_raw, i_raw) in self.seen_pairs) != l:
 183           # The evaluation item is not considered during false negative
 184           # generation, so it will occasionally appear as a negative example
 185           # during training.
 186           assert not l
 187           self.assertEqual(i_raw, self.holdout[u_raw][1])
 188         train_examples[l].add((u_raw, i_raw))
 189         counts[(u_raw, i_raw)] += 1
 190 
 191     self.assertRegexpMatches(md5.hexdigest(), END_TO_END_TRAIN_MD5)
 192 
 193     num_positives_seen = len(train_examples[True])
 194     self.assertEqual(producer._train_pos_users.shape[0], num_positives_seen)
 195 
 196     # This check is more heuristic because negatives are sampled with
 197     # replacement. It only checks that negative generation is reasonably random.
 198     self.assertGreater(
 199         len(train_examples[False]) / NUM_NEG / num_positives_seen, 0.9)
 200 
 201     # This checks that the samples produced are independent by checking the
 202     # number of duplicate entries. If workers are not properly independent there
 203     # will be lots of repeated pairs.
 204     self.assertLess(np.mean(list(counts.values())), 1.1)
 205 
 206     # ==========================================================================
 207     # == Eval Data =============================================================
 208     # ==========================================================================
 209     with g.as_default():
 210       input_fn = producer.make_input_fn(is_training=False)
 211       dataset = input_fn(params)
 212 
 213     eval_data = self.drain_dataset(dataset=dataset, g=g)
 214 
 215     current_user = None
 216     md5 = hashlib.md5()
 217     for features in eval_data:
 218       data_list = [
 219           features[movielens.USER_COLUMN].flatten(),
 220           features[movielens.ITEM_COLUMN].flatten(),
 221           features[rconst.DUPLICATE_MASK].flatten()
 222       ]
 223       for i in data_list:
 224         md5.update(i.tobytes())
 225 
 226       for idx, (u, i, d) in enumerate(zip(*data_list)):
 227         u_raw = user_inv_map[u]
 228         i_raw = item_inv_map[i]
 229         if current_user is None:
 230           current_user = u
 231 
 232         # Ensure that users appear in blocks, as the evaluation logic expects
 233         # this structure.
 234         self.assertEqual(u, current_user)
 235 
 236         # The structure of evaluation data is 999 negative examples followed
 237         # by the holdout positive.
 238         if not (idx + 1) % (rconst.NUM_EVAL_NEGATIVES + 1):
 239           # Check that the last element in each chunk is the holdout item.
 240           self.assertEqual(i_raw, self.holdout[u_raw][1])
 241           current_user = None
 242 
 243         elif i_raw == self.holdout[u_raw][1]:
 244           # Because the holdout item is not given to the negative generation
 245           # process, it can appear as a negative. In that case, it should be
 246           # masked out as a duplicate. (Since the true positive is placed at
 247           # the end and would therefore lose the tie.)
 248           assert d
 249 
 250         else:
 251           # Otherwise check that the other 999 points for a user are selected
 252           # from the negatives.
 253           assert (u_raw, i_raw) not in self.seen_pairs
 254 
 255     self.assertRegexpMatches(md5.hexdigest(), END_TO_END_EVAL_MD5)

 257   def _test_fresh_randomness(self, constructor_type):
 258     train_epochs = 5
 259     params = self.make_params(train_epochs=train_epochs)
 260     _, _, producer = data_preprocessing.instantiate_pipeline(
 261         dataset=DATASET, data_dir=self.temp_data_dir, params=params,
 262         constructor_type=constructor_type, deterministic=True)
 263 
 264     producer.start()
 265 
 266     results = []
 267     g = tf.Graph()
 268     with g.as_default():
 269       for _ in range(train_epochs):
 270         input_fn = producer.make_input_fn(is_training=True)
 271         dataset = input_fn(params)
 272         results.extend(self.drain_dataset(dataset=dataset, g=g))
 273 
 274     producer.join()
 275     assert producer._fatal_exception is None
 276 
 277     positive_counts, negative_counts = defaultdict(int), defaultdict(int)
 278     md5 = hashlib.md5()
 279     for features, labels in results:
 280       data_list = [
 281           features[movielens.USER_COLUMN].flatten(),
 282           features[movielens.ITEM_COLUMN].flatten(),
 283           features[rconst.VALID_POINT_MASK].flatten(),
 284           labels.flatten()
 285       ]
 286       for i in data_list:
 287         md5.update(i.tobytes())
 288 
 289       for u, i, v, l in zip(*data_list):
 290         if not v:
 291           continue  # ignore padding
 292 
 293         if l:
 294           positive_counts[(u, i)] += 1
 295         else:
 296           negative_counts[(u, i)] += 1
 297 
 298     self.assertRegexpMatches(md5.hexdigest(), FRESH_RANDOMNESS_MD5)
 299 
 300     # The positive examples should appear exactly once each epoch
 301     self.assertAllEqual(list(positive_counts.values()),
 302                         [train_epochs for _ in positive_counts])
 303 
 304     # The threshold for the negatives is heuristic, but in general repeats are
 305     # expected, but should not appear too frequently.
 306 
 307     pair_cardinality = NUM_USERS * NUM_ITEMS
 308     neg_pair_cardinality = pair_cardinality - len(self.seen_pairs)
 309 
 310     # Approximation for the expectation number of times that a particular
 311     # negative will appear in a given epoch. Implicit in this calculation is the
 312     # treatment of all negative pairs as equally likely. Normally is not
 313     # necessarily reasonable; however the generation in self.setUp() will
 314     # approximate this behavior sufficiently for heuristic testing.
 315     e_sample = len(self.seen_pairs) * NUM_NEG / neg_pair_cardinality
 316 
 317     # The frequency of occurance of a given negative pair should follow an
 318     # approximately binomial distribution in the limit that the cardinality of
 319     # the negative pair set >> number of samples per epoch.
 320     approx_pdf = scipy.stats.binom.pmf(k=np.arange(train_epochs+1),
 321                                        n=train_epochs, p=e_sample)
 322 
 323     # Tally the actual observed counts.
 324     count_distribution = [0 for _ in range(train_epochs + 1)]
 325     for i in negative_counts.values():
 326       i = min([i, train_epochs])  # round down tail for simplicity.
 327       count_distribution[i] += 1
 328     count_distribution[0] = neg_pair_cardinality - sum(count_distribution[1:])
 329 
 330     # Check that the frequency of negative pairs is approximately binomial.
 331     for i in range(train_epochs + 1):
 332       if approx_pdf[i] < 0.05:
 333         continue  # Variance will be high at the tails.
 334 
 335       observed_fraction = count_distribution[i] / neg_pair_cardinality
 336       deviation = (2 * abs(observed_fraction - approx_pdf[i]) /
 337                    (observed_fraction + approx_pdf[i]))
 338 
 339       self.assertLess(deviation, 0.2)
