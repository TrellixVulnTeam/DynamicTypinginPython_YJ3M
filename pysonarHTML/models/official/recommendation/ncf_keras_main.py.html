  49 def metric_fn(logits, dup_mask, match_mlperf):
  50   dup_mask = tf.cast(dup_mask, tf.float32)
  51   logits = tf.slice(logits, [0, 1], [-1, -1])
  52   in_top_k, _, metric_weights, _ = neumf_model.compute_top_k_and_ndcg(
  53       logits,
  54       dup_mask,
  55       match_mlperf)
  56   metric_weights = tf.cast(metric_weights, tf.float32)
  57   return in_top_k, metric_weights

  60 class MetricLayer(tf.keras.layers.Layer):
  61   """Custom layer of metrics for NCF model."""
  62 
  63   def __init__(self, match_mlperf):
  64     super(MetricLayer, self).__init__()
  65     self.match_mlperf = match_mlperf
  66 
  67   def get_config(self):
  68     return {"match_mlperf": self.match_mlperf}
  69 
  70   @classmethod
  71   def from_config(cls, config, custom_objects=None):
  72     return cls(**config)
  73 
  74   def call(self, inputs, training=False):
  75     logits, dup_mask = inputs
  76 
  77     if training:
  78       hr_sum = 0.0
  79       hr_count = 0.0
  80     else:
  81       metric, metric_weights = metric_fn(logits, dup_mask, self.match_mlperf)
  82       hr_sum = tf.reduce_sum(metric * metric_weights)
  83       hr_count = tf.reduce_sum(metric_weights)
  84 
  85     self.add_metric(hr_sum, name="hr_sum", aggregation="mean")
  86     self.add_metric(hr_count, name="hr_count", aggregation="mean")
  87     return logits

  90 class LossLayer(tf.keras.layers.Layer):
  91   """Pass-through loss layer for NCF model."""
  92 
  93   def __init__(self, loss_normalization_factor):
  94     # The loss may overflow in float16, so we use float32 instead.
  95     super(LossLayer, self).__init__(dtype="float32")
  96     self.loss_normalization_factor = loss_normalization_factor
  97     self.loss = tf.keras.losses.SparseCategoricalCrossentropy(
  98         from_logits=True, reduction="sum")
  99 
 100   def get_config(self):
 101     return {"loss_normalization_factor": self.loss_normalization_factor}
 102 
 103   @classmethod
 104   def from_config(cls, config, custom_objects=None):
 105     return cls(**config)
 106 
 107   def call(self, inputs):
 108     logits, labels, valid_pt_mask_input = inputs
 109     loss = self.loss(
 110         y_true=labels, y_pred=logits, sample_weight=valid_pt_mask_input)
 111     loss = loss * (1.0 / self.loss_normalization_factor)
 112     self.add_loss(loss)
 113     return logits

 116 class IncrementEpochCallback(tf.keras.callbacks.Callback):
 117   """A callback to increase the requested epoch for the data producer.
 118 
 119   The reason why we need this is because we can only buffer a limited amount of
 120   data. So we keep a moving window to represent the buffer. This is to move the
 121   one of the window's boundaries for each epoch.
 122   """
 123 
 124   def __init__(self, producer):
 125     self._producer = producer
 126 
 127   def on_epoch_begin(self, epoch, logs=None):
 128     self._producer.increment_request_epoch()

 131 class CustomEarlyStopping(tf.keras.callbacks.Callback):
 132   """Stop training has reached a desired hit rate."""
 133 
 134   def __init__(self, monitor, desired_value):
 135     super(CustomEarlyStopping, self).__init__()
 136 
 137     self.monitor = monitor
 138     self.desired = desired_value
 139     self.stopped_epoch = 0
 140 
 141   def on_epoch_end(self, epoch, logs=None):
 142     current = self.get_monitor_value(logs)
 143     if current and current >= self.desired:
 144       self.stopped_epoch = epoch
 145       self.model.stop_training = True
 146 
 147   def on_train_end(self, logs=None):
 148     if self.stopped_epoch > 0:
 149       print("Epoch %05d: early stopping" % (self.stopped_epoch + 1))
 150 
 151   def get_monitor_value(self, logs):
 152     logs = logs or {}
 153     monitor_value = logs.get(self.monitor)
 154     if monitor_value is None:
 155       logging.warning("Early stopping conditioned on metric `%s` "
 156                       "which is not available. Available metrics are: %s",
 157                       self.monitor, ",".join(list(logs.keys())))
 158     return monitor_value

 161 def _get_keras_model(params):
 162   """Constructs and returns the model."""
 163   batch_size = params["batch_size"]
 164 
 165   user_input = tf.keras.layers.Input(
 166       shape=(1,), name=movielens.USER_COLUMN, dtype=tf.int32)
 167 
 168   item_input = tf.keras.layers.Input(
 169       shape=(1,), name=movielens.ITEM_COLUMN, dtype=tf.int32)
 170 
 171   valid_pt_mask_input = tf.keras.layers.Input(
 172       shape=(1,), name=rconst.VALID_POINT_MASK, dtype=tf.bool)
 173 
 174   dup_mask_input = tf.keras.layers.Input(
 175       shape=(1,), name=rconst.DUPLICATE_MASK, dtype=tf.int32)
 176 
 177   label_input = tf.keras.layers.Input(
 178       shape=(1,), name=rconst.TRAIN_LABEL_KEY, dtype=tf.bool)
 179 
 180   base_model = neumf_model.construct_model(user_input, item_input, params)
 181 
 182   logits = base_model.output
 183 
 184   zeros = tf.keras.layers.Lambda(
 185       lambda x: x * 0)(logits)
 186 
 187   softmax_logits = tf.keras.layers.concatenate(
 188       [zeros, logits],
 189       axis=-1)
 190 
 191   # Custom training loop calculates loss and metric as a part of
 192   # training/evaluation step function.
 193   if not params["keras_use_ctl"]:
 194     softmax_logits = MetricLayer(
 195         params["match_mlperf"])([softmax_logits, dup_mask_input])
 196     # TODO(b/134744680): Use model.add_loss() instead once the API is well
 197     # supported.
 198     softmax_logits = LossLayer(batch_size)(
 199         [softmax_logits, label_input, valid_pt_mask_input])
 200 
 201   keras_model = tf.keras.Model(
 202       inputs={
 203           movielens.USER_COLUMN: user_input,
 204           movielens.ITEM_COLUMN: item_input,
 205           rconst.VALID_POINT_MASK: valid_pt_mask_input,
 206           rconst.DUPLICATE_MASK: dup_mask_input,
 207           rconst.TRAIN_LABEL_KEY: label_input},
 208       outputs=softmax_logits)
 209 
 210   keras_model.summary()
 211   return keras_model

 214 def run_ncf(_):
 215   """Run NCF training and eval with Keras."""
 216 
 217   keras_utils.set_session_config(enable_xla=FLAGS.enable_xla)
 218 
 219   if FLAGS.seed is not None:
 220     print("Setting tf seed")
 221     tf.random.set_seed(FLAGS.seed)
 222 
 223   model_helpers.apply_clean(FLAGS)
 224 
 225   if FLAGS.dtype == "fp16" and FLAGS.fp16_implementation == "keras":
 226     policy = tf.keras.mixed_precision.experimental.Policy(
 227         "mixed_float16",
 228         loss_scale=flags_core.get_loss_scale(FLAGS, default_for_fp16="dynamic"))
 229     tf.keras.mixed_precision.experimental.set_policy(policy)
 230 
 231   strategy = distribution_utils.get_distribution_strategy(
 232       distribution_strategy=FLAGS.distribution_strategy,
 233       num_gpus=FLAGS.num_gpus,
 234       tpu_address=FLAGS.tpu)
 235 
 236   params = ncf_common.parse_flags(FLAGS)
 237   params["distribute_strategy"] = strategy
 238 
 239   if params["use_tpu"] and not params["keras_use_ctl"]:
 240     logging.error("Custom training loop must be used when using TPUStrategy.")
 241     return
 242 
 243   batch_size = params["batch_size"]
 244   time_callback = keras_utils.TimeHistory(batch_size, FLAGS.log_steps)
 245   callbacks = [time_callback]
 246 
 247   producer, input_meta_data = None, None
 248   generate_input_online = params["train_dataset_path"] is None
 249 
 250   if generate_input_online:
 251     # Start data producing thread.
 252     num_users, num_items, _, _, producer = ncf_common.get_inputs(params)
 253     producer.start()
 254     per_epoch_callback = IncrementEpochCallback(producer)
 255     callbacks.append(per_epoch_callback)
 256   else:
 257     assert params["eval_dataset_path"] and params["input_meta_data_path"]
 258     with tf.io.gfile.GFile(params["input_meta_data_path"], "rb") as reader:
 259       input_meta_data = json.loads(reader.read().decode("utf-8"))
 260       num_users = input_meta_data["num_users"]
 261       num_items = input_meta_data["num_items"]
 262 
 263   params["num_users"], params["num_items"] = num_users, num_items
 264 
 265   if FLAGS.early_stopping:
 266     early_stopping_callback = CustomEarlyStopping(
 267         "val_HR_METRIC", desired_value=FLAGS.hr_threshold)
 268     callbacks.append(early_stopping_callback)
 269 
 270   (train_input_dataset, eval_input_dataset,
 271    num_train_steps, num_eval_steps) = \
 272     (ncf_input_pipeline.create_ncf_input_data(
 273         params, producer, input_meta_data, strategy))
 274   steps_per_epoch = None if generate_input_online else num_train_steps
 275 
 276   with distribution_utils.get_strategy_scope(strategy):
 277     keras_model = _get_keras_model(params)
 278     optimizer = tf.keras.optimizers.Adam(
 279         learning_rate=params["learning_rate"],
 280         beta_1=params["beta1"],
 281         beta_2=params["beta2"],
 282         epsilon=params["epsilon"])
 283     if FLAGS.fp16_implementation == "graph_rewrite":
 284       optimizer = \
 285         tf.compat.v1.train.experimental.enable_mixed_precision_graph_rewrite(
 286             optimizer,
 287             loss_scale=flags_core.get_loss_scale(FLAGS,
 288                                                  default_for_fp16="dynamic"))
 289     elif FLAGS.dtype == "fp16" and params["keras_use_ctl"]:
 290       # When keras_use_ctl is False, instead Model.fit() automatically applies
 291       # loss scaling so we don't need to create a LossScaleOptimizer.
 292       optimizer = tf.keras.mixed_precision.experimental.LossScaleOptimizer(
 293           optimizer,
 294           tf.keras.mixed_precision.experimental.global_policy().loss_scale)
 295 
 296     if params["keras_use_ctl"]:
 297       train_loss, eval_results = run_ncf_custom_training(
 298           params,
 299           strategy,
 300           keras_model,
 301           optimizer,
 302           callbacks,
 303           train_input_dataset,
 304           eval_input_dataset,
 305           num_train_steps,
 306           num_eval_steps,
 307           generate_input_online=generate_input_online)
 308     else:
 309       keras_model.compile(optimizer=optimizer, run_eagerly=FLAGS.run_eagerly)
 310 
 311       if not FLAGS.ml_perf:
 312         # Create Tensorboard summary and checkpoint callbacks.
 313         summary_dir = os.path.join(FLAGS.model_dir, "summaries")
 314         summary_callback = tf.keras.callbacks.TensorBoard(summary_dir)
 315         checkpoint_path = os.path.join(FLAGS.model_dir, "checkpoint")
 316         checkpoint_callback = tf.keras.callbacks.ModelCheckpoint(
 317             checkpoint_path, save_weights_only=True)
 318 
 319         callbacks += [summary_callback, checkpoint_callback]
 320 
 321       history = keras_model.fit(
 322           train_input_dataset,
 323           epochs=FLAGS.train_epochs,
 324           steps_per_epoch=steps_per_epoch,
 325           callbacks=callbacks,
 326           validation_data=eval_input_dataset,
 327           validation_steps=num_eval_steps,
 328           verbose=2)
 329 
 330       logging.info("Training done. Start evaluating")
 331 
 332       eval_loss_and_metrics = keras_model.evaluate(
 333           eval_input_dataset, steps=num_eval_steps, verbose=2)
 334 
 335       logging.info("Keras evaluation is done.")
 336 
 337       # Keras evaluate() API returns scalar loss and metric values from
 338       # evaluation as a list. Here, the returned list would contain
 339       # [evaluation loss, hr sum, hr count].
 340       eval_hit_rate = eval_loss_and_metrics[1] / eval_loss_and_metrics[2]
 341 
 342       # Format evaluation result into [eval loss, eval hit accuracy].
 343       eval_results = [eval_loss_and_metrics[0], eval_hit_rate]
 344 
 345       if history and history.history:
 346         train_history = history.history
 347         train_loss = train_history["loss"][-1]
 348 
 349   stats = build_stats(train_loss, eval_results, time_callback)
 350   return stats

 353 def run_ncf_custom_training(params,
 354                             strategy,
 355                             keras_model,
 356                             optimizer,
 357                             callbacks,
 358                             train_input_dataset,
 359                             eval_input_dataset,
 360                             num_train_steps,
 361                             num_eval_steps,
 362                             generate_input_online=True):
 363   """Runs custom training loop.
 364 
 365   Args:
 366     params: Dictionary containing training parameters.
 367     strategy: Distribution strategy to be used for distributed training.
 368     keras_model: Model used for training.
 369     optimizer: Optimizer used for training.
 370     callbacks: Callbacks to be invoked between batches/epochs.
 371     train_input_dataset: tf.data.Dataset used for training.
 372     eval_input_dataset: tf.data.Dataset used for evaluation.
 373     num_train_steps: Total number of steps to run for training.
 374     num_eval_steps: Total number of steps to run for evaluation.
 375     generate_input_online: Whether input data was generated by data producer.
 376       When data is generated by data producer, then train dataset must be
 377       re-initialized after every epoch.
 378 
 379   Returns:
 380     A tuple of train loss and a list of training and evaluation results.
 381   """
 382   loss_object = tf.keras.losses.SparseCategoricalCrossentropy(
 383       reduction="sum", from_logits=True)
 384   train_input_iterator = iter(
 385       strategy.experimental_distribute_dataset(train_input_dataset))
 386 
 387   def train_step(train_iterator):
 388     """Called once per step to train the model."""
 389 
 390     def step_fn(features):
 391       """Computes loss and applied gradient per replica."""
 392       with tf.GradientTape() as tape:
 393         softmax_logits = keras_model(features)
 394         # The loss can overflow in float16, so we cast to float32.
 395         softmax_logits = tf.cast(softmax_logits, "float32")
 396         labels = features[rconst.TRAIN_LABEL_KEY]
 397         loss = loss_object(
 398             labels,
 399             softmax_logits,
 400             sample_weight=features[rconst.VALID_POINT_MASK])
 401         loss *= (1.0 / params["batch_size"])
 402         if FLAGS.dtype == "fp16":
 403           loss = optimizer.get_scaled_loss(loss)
 404 
 405       grads = tape.gradient(loss, keras_model.trainable_variables)
 406       if FLAGS.dtype == "fp16":
 407         grads = optimizer.get_unscaled_gradients(grads)
 408       # Converting gradients to dense form helps in perf on GPU for NCF
 409       grads = neumf_model.sparse_to_dense_grads(
 410           list(zip(grads, keras_model.trainable_variables)))
 411       optimizer.apply_gradients(grads)
 412       return loss
 413 
 414     per_replica_losses = strategy.run(
 415         step_fn, args=(next(train_iterator),))
 416     mean_loss = strategy.reduce(
 417         tf.distribute.ReduceOp.SUM, per_replica_losses, axis=None)
 418     return mean_loss
 419 
 420   def eval_step(eval_iterator):
 421     """Called once per eval step to compute eval metrics."""
 422 
 423     def step_fn(features):
 424       """Computes eval metrics per replica."""
 425       softmax_logits = keras_model(features)
 426       in_top_k, metric_weights = metric_fn(softmax_logits,
 427                                            features[rconst.DUPLICATE_MASK],
 428                                            params["match_mlperf"])
 429       hr_sum = tf.reduce_sum(in_top_k * metric_weights)
 430       hr_count = tf.reduce_sum(metric_weights)
 431       return hr_sum, hr_count
 432 
 433     per_replica_hr_sum, per_replica_hr_count = (
 434         strategy.run(
 435             step_fn, args=(next(eval_iterator),)))
 436     hr_sum = strategy.reduce(
 437         tf.distribute.ReduceOp.SUM, per_replica_hr_sum, axis=None)
 438     hr_count = strategy.reduce(
 439         tf.distribute.ReduceOp.SUM, per_replica_hr_count, axis=None)
 440     return hr_sum, hr_count
 441 
 442   if not FLAGS.run_eagerly:
 443     train_step = tf.function(train_step)
 444     eval_step = tf.function(eval_step)
 445 
 446   for callback in callbacks:
 447     callback.on_train_begin()
 448 
 449   # Not writing tensorboard summaries if running in MLPerf.
 450   if FLAGS.ml_perf:
 451     eval_summary_writer, train_summary_writer = None, None
 452   else:
 453     summary_dir = os.path.join(FLAGS.model_dir, "summaries")
 454     eval_summary_writer = tf.summary.create_file_writer(
 455         os.path.join(summary_dir, "eval"))
 456     train_summary_writer = tf.summary.create_file_writer(
 457         os.path.join(summary_dir, "train"))
 458 
 459   train_loss = 0
 460   for epoch in range(FLAGS.train_epochs):
 461     for cb in callbacks:
 462       cb.on_epoch_begin(epoch)
 463 
 464     # As NCF dataset is sampled with randomness, not repeating
 465     # data elements in each epoch has significant impact on
 466     # convergence. As so, offline-generated TF record files
 467     # contains all epoch worth of data. Thus we do not need
 468     # to initialize dataset when reading from tf record files.
 469     if generate_input_online:
 470       train_input_iterator = iter(
 471           strategy.experimental_distribute_dataset(train_input_dataset))
 472 
 473     train_loss = 0
 474     for step in range(num_train_steps):
 475       current_step = step + epoch * num_train_steps
 476       for c in callbacks:
 477         c.on_batch_begin(current_step)
 478 
 479       train_loss += train_step(train_input_iterator)
 480 
 481       # Write train loss once in every 1000 steps.
 482       if train_summary_writer and step % 1000 == 0:
 483         with train_summary_writer.as_default():
 484           tf.summary.scalar("training_loss", train_loss/(step + 1),
 485                             step=current_step)
 486 
 487       for c in callbacks:
 488         c.on_batch_end(current_step)
 489 
 490     train_loss /= num_train_steps
 491     logging.info("Done training epoch %s, epoch loss=%s.", epoch + 1,
 492                  train_loss)
 493 
 494     eval_input_iterator = iter(
 495         strategy.experimental_distribute_dataset(eval_input_dataset))
 496     hr_sum = 0
 497     hr_count = 0
 498     for _ in range(num_eval_steps):
 499       step_hr_sum, step_hr_count = eval_step(eval_input_iterator)
 500       hr_sum += step_hr_sum
 501       hr_count += step_hr_count
 502 
 503     logging.info("Done eval epoch %s, hit_rate=%s.", epoch + 1,
 504                  hr_sum / hr_count)
 505     if eval_summary_writer:
 506       with eval_summary_writer.as_default():
 507         tf.summary.scalar("hit_rate", hr_sum / hr_count, step=current_step)
 508 
 509     if (FLAGS.early_stopping and
 510         float(hr_sum / hr_count) > params["hr_threshold"]):
 511       break
 512 
 513   for c in callbacks:
 514     c.on_train_end()
 515 
 516   # Saving the model at the end of training.
 517   if not FLAGS.ml_perf:
 518     checkpoint = tf.train.Checkpoint(model=keras_model, optimizer=optimizer)
 519     checkpoint_path = os.path.join(FLAGS.model_dir, "ctl_checkpoint")
 520     checkpoint.save(checkpoint_path)
 521     logging.info("Saving model as TF checkpoint: %s", checkpoint_path)
 522 
 523   return train_loss, [None, hr_sum / hr_count]

 526 def build_stats(loss, eval_result, time_callback):
 527   """Normalizes and returns dictionary of stats.
 528 
 529   Args:
 530     loss: The final loss at training time.
 531     eval_result: Output of the eval step. Assumes first value is eval_loss and
 532       second value is accuracy_top_1.
 533     time_callback: Time tracking callback likely used during keras.fit.
 534 
 535   Returns:
 536     Dictionary of normalized results.
 537   """
 538   stats = {}
 539   if loss:
 540     stats["loss"] = loss
 541 
 542   if eval_result:
 543     stats["eval_loss"] = eval_result[0]
 544     stats["eval_hit_rate"] = eval_result[1]
 545 
 546   if time_callback:
 547     timestamp_log = time_callback.timestamp_log
 548     stats["step_timestamp_log"] = timestamp_log
 549     stats["train_finish_time"] = time_callback.train_finish_time
 550     if len(timestamp_log) > 1:
 551       stats["avg_exp_per_second"] = (
 552           time_callback.batch_size * time_callback.log_steps *
 553           (len(time_callback.timestamp_log)-1) /
 554           (timestamp_log[-1].timestamp - timestamp_log[0].timestamp))
 555 
 556   return stats
