  17 import dataclasses
  18 import tensorflow as tf
  19 from official.modeling.hyperparams import base_config
  20 from official.modeling.hyperparams import oneof
  21 
  22 
  23 @dataclasses.dataclass
  24 class ResNet(base_config.Config):
  25   model_depth: int = 50
  26 
  27 
  28 @dataclasses.dataclass
  29 class Backbone(oneof.OneOfConfig):
  30   type: str = 'resnet'
  31   resnet: ResNet = ResNet()
  32   not_resnet: int = 2
  33 
  34 
  35 @dataclasses.dataclass
  36 class OutputLayer(oneof.OneOfConfig):
  37   type: str = 'single'
  38   single: int = 1
  39   multi_head: int = 2
  40 
  41 
  42 @dataclasses.dataclass
  43 class Network(base_config.Config):
  44   backbone: Backbone = Backbone()
  45   output_layer: OutputLayer = OutputLayer()
  46 
  47 
  48 class OneOfTest(tf.test.TestCase):
  49 
  50   def test_to_dict(self):
  51     network_params = {'backbone': {'type': 'resnet',
  52                                    'resnet': {'model_depth': 50}
  53                                    },
  54                       'output_layer': {'type': 'single',
  55                                        'single': 1000}
  56                       }
  57     network_config = Network(network_params)
  58     self.assertEqual(network_config.as_dict(), network_params)
  59 
  60   def test_get_oneof(self):
  61     backbone = Backbone()
  62     self.assertIsInstance(backbone.get(), ResNet)
  63     self.assertEqual(backbone.get().as_dict(), {'model_depth': 50})
  64

  67   tf.test.main()
