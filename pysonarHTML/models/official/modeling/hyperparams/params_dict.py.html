  29 # regex pattern that matches on key-value pairs in a comma-separated
  30 # key-value pair string. It splits each k-v pair on the = sign, and
  31 # matches on values that are within single quotes, double quotes, single
  32 # values (e.g. floats, ints, etc.), and a lists within brackets.
  33 _PARAM_RE = re.compile(r"""
  34   (?P<name>[a-zA-Z][\w\.]*)    # variable name: "var" or "x"
  35   \s*=\s*
  36   ((?P<val>\'(.*?)\'           # single quote
  37   |
  38   \"(.*?)\"                    # double quote
  39   |
  40   [^,\[]*                      # single value
  41   |
  42   \[[^\]]*\]))                 # list of values
  43   ($|,\s*)""", re.VERBOSE)
  44 
  45 _CONST_VALUE_RE = re.compile(r'(\d.*|-\d.*|None)')

  48 class ParamsDict(object):
  49   """A hyperparameter container class."""
  50 
  51   RESERVED_ATTR = ['_locked', '_restrictions']
  52 
  53   def __init__(self, default_params=None, restrictions=None):
  54     """Instantiate a ParamsDict.
  55 
  56     Instantiate a ParamsDict given a set of default parameters and a list of
  57     restrictions. Upon initialization, it validates itself by checking all the
  58     defined restrictions, and raise error if it finds inconsistency.
  59 
  60     Args:
  61       default_params: a Python dict or another ParamsDict object including the
  62         default parameters to initialize.
  63       restrictions: a list of strings, which define a list of restrictions to
  64         ensure the consistency of different parameters internally. Each
  65         restriction string is defined as a binary relation with a set of
  66         operators, including {'==', '!=',  '<', '<=', '>', '>='}.
  67     """
  68     self._locked = False
  69     self._restrictions = []
  70     if restrictions:
  71       self._restrictions = restrictions
  72     if default_params is None:
  73       default_params = {}
  74     self.override(default_params, is_strict=False)
  75     self.validate()
  76 
  77   def _set(self, k, v):
  78     if isinstance(v, dict):
  79       self.__dict__[k] = ParamsDict(v)
  80     else:
  81       self.__dict__[k] = copy.deepcopy(v)
  82 
  83   def __setattr__(self, k, v):
  84     """Sets the value of the existing key.
  85 
  86     Note that this does not allow directly defining a new key. Use the
  87     `override` method with `is_strict=False` instead.
  88 
  89     Args:
  90       k: the key string.
  91       v: the value to be used to set the key `k`.
  92 
  93     Raises:
  94       KeyError: if k is not defined in the ParamsDict.
  95     """
  96     if k not in ParamsDict.RESERVED_ATTR:
  97       if k not in self.__dict__.keys():
  98         raise KeyError('The key `%{}` does not exist. '
  99                        'To extend the existing keys, use '
 100                        '`override` with `is_strict` = True.'.format(k))
 101       if self._locked:
 102         raise ValueError('The ParamsDict has been locked. '
 103                          'No change is allowed.')
 104     self._set(k, v)
 105 
 106   def __getattr__(self, k):
 107     """Gets the value of the existing key.
 108 
 109     Args:
 110       k: the key string.
 111 
 112     Returns:
 113       the value of the key.
 114 
 115     Raises:
 116       AttributeError: if k is not defined in the ParamsDict.
 117     """
 118     if k not in self.__dict__.keys():
 119       raise AttributeError('The key `{}` does not exist. '.format(k))
 120     return self.__dict__[k]
 121 
 122   def __contains__(self, key):
 123     """Implements the membership test operator."""
 124     return key in self.__dict__
 125 
 126   def get(self, key, value=None):
 127     """Accesses through built-in dictionary get method."""
 128     return self.__dict__.get(key, value)
 129 
 130   def __delattr__(self, k):
 131     """Deletes the key and removes its values.
 132 
 133     Args:
 134       k: the key string.
 135 
 136     Raises:
 137       AttributeError: if k is reserverd or not defined in the ParamsDict.
 138       ValueError: if the ParamsDict instance has been locked.
 139     """
 140     if k in ParamsDict.RESERVED_ATTR:
 141       raise AttributeError('The key `{}` is reserved. No change is allowes. '
 142                            .format(k))
 143     if k not in self.__dict__.keys():
 144       raise AttributeError('The key `{}` does not exist. '.format(k))
 145     if self._locked:
 146       raise ValueError('The ParamsDict has been locked. No change is allowed.')
 147     del self.__dict__[k]
 148 
 149   def override(self, override_params, is_strict=True):
 150     """Override the ParamsDict with a set of given params.
 151 
 152     Args:
 153       override_params: a dict or a ParamsDict specifying the parameters to
 154         be overridden.
 155       is_strict: a boolean specifying whether override is strict or not. If
 156         True, keys in `override_params` must be present in the ParamsDict.
 157         If False, keys in `override_params` can be different from what is
 158         currently defined in the ParamsDict. In this case, the ParamsDict will
 159         be extended to include the new keys.
 160     """
 161     if self._locked:
 162       raise ValueError('The ParamsDict has been locked. No change is allowed.')
 163     if isinstance(override_params, ParamsDict):
 164       override_params = override_params.as_dict()
 165     self._override(override_params, is_strict)  # pylint: disable=protected-access
 166 
 167   def _override(self, override_dict, is_strict=True):
 168     """The implementation of `override`."""
 169     for k, v in six.iteritems(override_dict):
 170       if k in ParamsDict.RESERVED_ATTR:
 171         raise KeyError('The key `%{}` is internally reserved. '
 172                        'Can not be overridden.')
 173       if k not in self.__dict__.keys():
 174         if is_strict:
 175           raise KeyError('The key `{}` does not exist. '
 176                          'To extend the existing keys, use '
 177                          '`override` with `is_strict` = False.'.format(k))
 178         else:
 179           self._set(k, v)
 180       else:
 181         if isinstance(v, dict):
 182           self.__dict__[k]._override(v, is_strict)  # pylint: disable=protected-access
 183         elif isinstance(v, ParamsDict):
 184           self.__dict__[k]._override(v.as_dict(), is_strict)  # pylint: disable=protected-access
 185         else:
 186           self.__dict__[k] = copy.deepcopy(v)
 187 
 188   def lock(self):
 189     """Makes the ParamsDict immutable."""
 190     self._locked = True
 191 
 192   def as_dict(self):
 193     """Returns a dict representation of ParamsDict.
 194 
 195     For the nested ParamsDict, a nested dict will be returned.
 196     """
 197     params_dict = {}
 198     for k, v in six.iteritems(self.__dict__):
 199       if k not in ParamsDict.RESERVED_ATTR:
 200         if isinstance(v, ParamsDict):
 201           params_dict[k] = v.as_dict()
 202         else:
 203           params_dict[k] = copy.deepcopy(v)
 204     return params_dict
 205 
 206   def validate(self):
 207     """Validate the parameters consistency based on the restrictions.
 208 
 209     This method validates the internal consistency using the pre-defined list of
 210     restrictions. A restriction is defined as a string which specfiies a binary
 211     operation. The supported binary operations are {'==', '!=', '<', '<=', '>',
 212     '>='}. Note that the meaning of these operators are consistent with the
 213     underlying Python immplementation. Users should make sure the define
 214     restrictions on their type make sense.
 215 
 216     For example, for a ParamsDict like the following
 217     ```
 218     a:
 219       a1: 1
 220       a2: 2
 221     b:
 222       bb:
 223         bb1: 10
 224         bb2: 20
 225       ccc:
 226         a1: 1
 227         a3: 3
 228     ```
 229     one can define two restrictions like this
 230     ['a.a1 == b.ccc.a1', 'a.a2 <= b.bb.bb2']
 231 
 232     What it enforces are:
 233      - a.a1 = 1 == b.ccc.a1 = 2
 234      - a.a2 = 2 <= b.bb.bb2 = 20
 235 
 236     Raises:
 237       KeyError: if any of the following happens
 238         (1) any of parameters in any of restrictions is not defined in
 239             ParamsDict,
 240         (2) any inconsistency violating the restriction is found.
 241       ValueError: if the restriction defined in the string is not supported.
 242     """
 243     def _get_kv(dotted_string, params_dict):
 244       """Get keys and values indicated by dotted_string."""
 245       if _CONST_VALUE_RE.match(dotted_string) is not None:
 246         const_str = dotted_string
 247         if const_str == 'None':
 248           constant = None
 249         else:
 250           constant = float(const_str)
 251         return None, constant
 252       else:
 253         tokenized_params = dotted_string.split('.')
 254         v = params_dict
 255         for t in tokenized_params:
 256           v = v[t]
 257         return tokenized_params[-1], v
 258 
 259     def _get_kvs(tokens, params_dict):
 260       if len(tokens) != 2:
 261         raise ValueError('Only support binary relation in restriction.')
 262       stripped_tokens = [t.strip() for t in tokens]
 263       left_k, left_v = _get_kv(stripped_tokens[0], params_dict)
 264       right_k, right_v = _get_kv(stripped_tokens[1], params_dict)
 265       return left_k, left_v, right_k, right_v
 266 
 267     params_dict = self.as_dict()
 268     for restriction in self._restrictions:
 269       if '==' in restriction:
 270         tokens = restriction.split('==')
 271         _, left_v, _, right_v = _get_kvs(tokens, params_dict)
 272         if left_v != right_v:
 273           raise KeyError('Found inconsistncy between key `{}` and key `{}`.'
 274                          .format(tokens[0], tokens[1]))
 275       elif '!=' in restriction:
 276         tokens = restriction.split('!=')
 277         _, left_v, _, right_v = _get_kvs(tokens, params_dict)
 278         if left_v == right_v:
 279           raise KeyError('Found inconsistncy between key `{}` and key `{}`.'
 280                          .format(tokens[0], tokens[1]))
 281       elif '<' in restriction:
 282         tokens = restriction.split('<')
 283         _, left_v, _, right_v = _get_kvs(tokens, params_dict)
 284         if left_v >= right_v:
 285           raise KeyError('Found inconsistncy between key `{}` and key `{}`.'
 286                          .format(tokens[0], tokens[1]))
 287       elif '<=' in restriction:
 288         tokens = restriction.split('<=')
 289         _, left_v, _, right_v = _get_kvs(tokens, params_dict)
 290         if left_v > right_v:
 291           raise KeyError('Found inconsistncy between key `{}` and key `{}`.'
 292                          .format(tokens[0], tokens[1]))
 293       elif '>' in restriction:
 294         tokens = restriction.split('>')
 295         _, left_v, _, right_v = _get_kvs(tokens, params_dict)
 296         if left_v <= right_v:
 297           raise KeyError('Found inconsistncy between key `{}` and key `{}`.'
 298                          .format(tokens[0], tokens[1]))
 299       elif '>=' in restriction:
 300         tokens = restriction.split('>=')
 301         _, left_v, _, right_v = _get_kvs(tokens, params_dict)
 302         if left_v < right_v:
 303           raise KeyError('Found inconsistncy between key `{}` and key `{}`.'
 304                          .format(tokens[0], tokens[1]))
 305       else:
 306         raise ValueError('Unsupported relation in restriction.')
 307

 309 def read_yaml_to_params_dict(file_path):
 310   """Reads a YAML file to a ParamsDict."""
 311   with tf.io.gfile.GFile(file_path, 'r') as f:
 312     params_dict = yaml.load(f)
 313     return ParamsDict(params_dict)

 316 def save_params_dict_to_yaml(params, file_path):
 317   """Saves the input ParamsDict to a YAML file."""
 318   with tf.io.gfile.GFile(file_path, 'w') as f:
 319     def _my_list_rep(dumper, data):
 320       # u'tag:yaml.org,2002:seq' is the YAML internal tag for sequence.
 321       return dumper.represent_sequence(
 322           u'tag:yaml.org,2002:seq', data, flow_style=True)
 323     yaml.add_representer(list, _my_list_rep)
 324     yaml.dump(params.as_dict(), f, default_flow_style=False)

 327 def nested_csv_str_to_json_str(csv_str):
 328   """Converts a nested (using '.') comma-separated k=v string to a JSON string.
 329 
 330   Converts a comma-separated string of key/value pairs that supports
 331   nesting of keys to a JSON string. Nesting is implemented using
 332   '.' between levels for a given key.
 333 
 334   Spacing between commas and = is supported (e.g. there is no difference between
 335   "a=1,b=2", "a = 1, b = 2", or "a=1, b=2") but there should be no spaces before
 336   keys or after values (e.g. " a=1,b=2" and "a=1,b=2 " are not supported).
 337 
 338   Note that this will only support values supported by CSV, meaning
 339   values such as nested lists (e.g. "a=[[1,2,3],[4,5,6]]") are not
 340   supported. Strings are supported as well, e.g. "a='hello'".
 341 
 342   An example conversion would be:
 343 
 344   "a=1, b=2, c.a=2, c.b=3, d.a.a=5"
 345 
 346   to
 347 
 348   "{ a: 1, b : 2, c: {a : 2, b : 3}, d: {a: {a : 5}}}"
 349 
 350   Args:
 351     csv_str: the comma separated string.
 352 
 353   Returns:
 354     the converted JSON string.
 355 
 356   Raises:
 357     ValueError: If csv_str is not in a comma separated string or
 358       if the string is formatted incorrectly.
 359   """
 360   if not csv_str:
 361     return ''
 362 
 363   formatted_entries = []
 364   nested_map = collections.defaultdict(list)
 365   pos = 0
 366   while pos < len(csv_str):
 367     m = _PARAM_RE.match(csv_str, pos)
 368     if not m:
 369       raise ValueError('Malformed hyperparameter value while parsing '
 370                        'CSV string: %s' % csv_str[pos:])
 371     pos = m.end()
 372     # Parse the values.
 373     m_dict = m.groupdict()
 374     name = m_dict['name']
 375     v = m_dict['val']
 376 
 377     # If a GCS path (e.g. gs://...) is provided, wrap this in quotes
 378     # as yaml.load would otherwise throw an exception
 379     if re.match(r'(?=[^\"\'])(?=[gs://])', v):
 380       v = '\'{}\''.format(v)
 381 
 382     name_nested = name.split('.')
 383     if len(name_nested) > 1:
 384       grouping = name_nested[0]
 385       value = '.'.join(name_nested[1:]) + '=' + v
 386       nested_map[grouping].append(value)
 387     else:
 388       formatted_entries.append('%s : %s' % (name, v))
 389 
 390   for grouping, value in nested_map.items():
 391     value = ','.join(value)
 392     value = nested_csv_str_to_json_str(value)
 393     formatted_entries.append('%s : %s' % (grouping, value))
 394   return '{' + ', '.join(formatted_entries) + '}'

 397 def override_params_dict(params, dict_or_string_or_yaml_file, is_strict):
 398   """Override a given ParamsDict using a dict, JSON/YAML/CSV string or YAML file.
 399 
 400   The logic of the function is outlined below:
 401   1. Test that the input is a dict. If not, proceed to 2.
 402   2. Tests that the input is a string. If not, raise unknown ValueError
 403   2.1. Test if the string is in a CSV format. If so, parse.
 404   If not, proceed to 2.2.
 405   2.2. Try loading the string as a YAML/JSON. If successful, parse to
 406   dict and use it to override. If not, proceed to 2.3.
 407   2.3. Try using the string as a file path and load the YAML file.
 408 
 409   Args:
 410     params: a ParamsDict object to be overridden.
 411     dict_or_string_or_yaml_file: a Python dict, JSON/YAML/CSV string or
 412       path to a YAML file specifying the parameters to be overridden.
 413     is_strict: a boolean specifying whether override is strict or not.
 414 
 415   Returns:
 416     params: the overridden ParamsDict object.
 417 
 418   Raises:
 419     ValueError: if failed to override the parameters.
 420   """
 421   if not dict_or_string_or_yaml_file:
 422     return params
 423   if isinstance(dict_or_string_or_yaml_file, dict):
 424     params.override(dict_or_string_or_yaml_file, is_strict)
 425   elif isinstance(dict_or_string_or_yaml_file, six.string_types):
 426     try:
 427       dict_or_string_or_yaml_file = (
 428           nested_csv_str_to_json_str(dict_or_string_or_yaml_file))
 429     except ValueError:
 430       pass
 431     params_dict = yaml.load(dict_or_string_or_yaml_file)
 432     if isinstance(params_dict, dict):
 433       params.override(params_dict, is_strict)
 434     else:
 435       with tf.io.gfile.GFile(dict_or_string_or_yaml_file) as f:
 436         params.override(yaml.load(f), is_strict)
 437   else:
 438     raise ValueError('Unknown input type to parse.')
 439   return params
