  28   def test_init_from_an_empty_dict(self):

  47   def test_lock(self):
  48     params = params_dict.ParamsDict({'a': 1, 'b': 2, 'c': 3})
  49     params.lock()
  50     with self.assertRaises(ValueError):
  51       params.a = 10
  52     with self.assertRaises(ValueError):
  53       params.override({'b': 20})
  54     with self.assertRaises(ValueError):
  55       del params.c

  74   def test_delattr(self):
  75     params = params_dict.ParamsDict()
  76     params.override(
  77         {'a': 'aa', 'b': 2, 'c': None, 'd': {'d1': 1, 'd2': 10}},
  78         is_strict=False)
  79     del params.c
  80     self.assertEqual(params.a, 'aa')
  81     self.assertEqual(params.b, 2)
  82     with self.assertRaises(AttributeError):
  83       _ = params.c
  84     del params.d
  85     with self.assertRaises(AttributeError):
  86       _ = params.d.d1

 103   def test_override_is_strict_true(self):
 104     params = params_dict.ParamsDict(
 105         {'a': 'aa', 'b': 2, 'c': {'c1': 'cc', 'c2': 20}})
 106     params.override({'a': 2, 'c': {'c1': 'ccc'}}, is_strict=True)
 107     self.assertEqual(params.a, 2)
 108     self.assertEqual(params.c.c1, 'ccc')
 109     with self.assertRaises(KeyError):
 110       params.override({'d': 'ddd'}, is_strict=True)
 111     with self.assertRaises(KeyError):
 112       params.override({'c': {'c3': 30}}, is_strict=True)
 113 
 114   def test_override_is_strict_false(self):
 115     params = params_dict.ParamsDict(
 116         {'a': 'aa', 'b': 2, 'c': {'c1': 10, 'c2': 20}})
 117     params.override({'a': 2, 'c': {'c3': 3000}}, is_strict=False)
 118     self.assertEqual(params.a, 2)
 119     self.assertEqual(params.c.c3, 3000)
 120     params.override({'d': 'ddd'}, is_strict=False)
 121     self.assertEqual(params.d, 'ddd')
 122     params.override({'c': {'c4': 4444}}, is_strict=False)
 123     self.assertEqual(params.c.c4, 4444)

 134   def test_validate(self):
 135     # Raise error due to the unknown parameter.
 136     with self.assertRaises(KeyError):
 137       params = params_dict.ParamsDict(
 138           {'a': 1, 'b': {'a': 11}}, ['a == c'])
 139 
 140     # OK to check equality of two nested dicts.
 141     params = params_dict.ParamsDict(
 142         {'a': 1, 'b': {'a': 10}, 'c': {'a': 10}}, ['b == c'])
 143 
 144     # Raise error due to inconsistency
 145     with self.assertRaises(KeyError):
 146       params = params_dict.ParamsDict(
 147           {'a': 1, 'c': {'a': 10}}, ['a == c.a'])
 148 
 149     # Valid rule.
 150     params = params_dict.ParamsDict(
 151         {'a': 1, 'c': {'a': 1}}, ['a == c.a'])
 152 
 153     # Overridding violates the existing rule, raise error upon validate.
 154     params.override({'a': 11})
 155     with self.assertRaises(KeyError):
 156       params.validate()
 157 
 158     # Valid restrictions with constant.
 159     params = params_dict.ParamsDict(
 160         {'a': None, 'c': {'a': 1}}, ['a == None', 'c.a == 1'])
 161     params.validate()
 162     with self.assertRaises(KeyError):
 163       params = params_dict.ParamsDict(
 164           {'a': 4, 'c': {'a': 1}}, ['a == None', 'c.a == 1'])

 169   def write_temp_file(self, filename, text):
 170     temp_file = os.path.join(self.get_temp_dir(), filename)
 171     with tf.io.gfile.GFile(temp_file, 'w') as writer:
 172       writer.write(text)
 173     return temp_file
 174 
 175   def test_save_params_dict_to_yaml(self):
 176     params = params_dict.ParamsDict(
 177         {'a': 'aa', 'b': 2, 'c': {'c1': 10, 'c2': 20}})
 178     output_yaml_file = os.path.join(self.get_temp_dir(), 'params.yaml')
 179     params_dict.save_params_dict_to_yaml(params, output_yaml_file)
 180 
 181     with tf.io.gfile.GFile(output_yaml_file, 'r') as f:
 182       params_d = yaml.load(f)
 183       self.assertEqual(params.a, params_d['a'])
 184       self.assertEqual(params.b, params_d['b'])
 185       self.assertEqual(params.c.c1, params_d['c']['c1'])
 186       self.assertEqual(params.c.c2, params_d['c']['c2'])
 187 
 188   def test_read_yaml_to_params_dict(self):
 189     input_yaml_file = self.write_temp_file(
 190         'params.yaml', r"""
 191         a: 'aa'
 192         b: 2
 193         c:
 194           c1: 10
 195           c2: 20
 196     """)
 197     params = params_dict.read_yaml_to_params_dict(input_yaml_file)
 198 
 199     self.assertEqual(params.a, 'aa')
 200     self.assertEqual(params.b, 2)
 201     self.assertEqual(params.c.c1, 10)
 202     self.assertEqual(params.c.c2, 20)

 204   def test_override_params_dict_using_dict(self):
 205     params = params_dict.ParamsDict({
 206         'a': 1, 'b': 2.5, 'c': [3, 4], 'd': 'hello', 'e': False})
 207     override_dict = {'b': 5.2, 'c': [30, 40]}
 208     params = params_dict.override_params_dict(
 209         params, override_dict, is_strict=True)
 210     self.assertEqual(1, params.a)
 211     self.assertEqual(5.2, params.b)
 212     self.assertEqual([30, 40], params.c)
 213     self.assertEqual('hello', params.d)
 214     self.assertEqual(False, params.e)
 215 
 216   def test_override_params_dict_using_yaml_string(self):
 217     params = params_dict.ParamsDict({
 218         'a': 1, 'b': 2.5, 'c': [3, 4], 'd': 'hello', 'e': False})
 219     override_yaml_string = "'b': 5.2\n'c': [30, 40]"
 220     params = params_dict.override_params_dict(
 221         params, override_yaml_string, is_strict=True)
 222     self.assertEqual(1, params.a)
 223     self.assertEqual(5.2, params.b)
 224     self.assertEqual([30, 40], params.c)
 225     self.assertEqual('hello', params.d)
 226     self.assertEqual(False, params.e)
 227 
 228   def test_override_params_dict_using_json_string(self):
 229     params = params_dict.ParamsDict({
 230         'a': 1, 'b': {'b1': 2, 'b2': [2, 3],},
 231         'd': {'d1': {'d2': 'hello'}}, 'e': False})
 232     override_json_string = "{ b: { b2: [3, 4] }, d: { d1: { d2: 'hi' } } }"
 233     params = params_dict.override_params_dict(
 234         params, override_json_string, is_strict=True)
 235     self.assertEqual(1, params.a)
 236     self.assertEqual(2, params.b.b1)
 237     self.assertEqual([3, 4], params.b.b2)
 238     self.assertEqual('hi', params.d.d1.d2)
 239     self.assertEqual(False, params.e)

 241   def test_override_params_dict_using_csv_string(self):
 242     params = params_dict.ParamsDict({
 243         'a': 1, 'b': {'b1': 2, 'b2': [2, 3],},
 244         'd': {'d1': {'d2': 'hello'}}, 'e': False})
 245     override_csv_string = "b.b2=[3,4], d.d1.d2='hi, world', e=gs://test"
 246     params = params_dict.override_params_dict(
 247         params, override_csv_string, is_strict=True)
 248     self.assertEqual(1, params.a)
 249     self.assertEqual(2, params.b.b1)
 250     self.assertEqual([3, 4], params.b.b2)
 251     self.assertEqual('hi, world', params.d.d1.d2)
 252     self.assertEqual('gs://test', params.e)

 271 class IOTest(tf.test.TestCase):
 272 
 273   def test_basic_csv_str_to_json_str(self):
 274     csv_str = 'a=1,b=2,c=3'
 275     json_str = '{a : 1, b : 2, c : 3}'
 276     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
 277     self.assertEqual(converted_csv_str, json_str)
 278 
 279   def test_basic_csv_str_load(self):
 280     csv_str = 'a=1,b=2,c=3'
 281     expected_output = {'a': 1, 'b': 2, 'c': 3}
 282     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
 283     converted_dict = yaml.load(converted_csv_str)
 284     self.assertDictEqual(converted_dict, expected_output)
 285 
 286   def test_basic_nested_csv_str_to_json_str(self):
 287     csv_str = 'a=1,b.b1=2'
 288     json_str = '{a : 1, b : {b1 : 2}}'
 289     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
 290     self.assertEqual(converted_csv_str, json_str)
 291 
 292   def test_basic_nested_csv_str_load(self):
 293     csv_str = 'a=1,b.b1=2,c.c1=3'
 294     expected_output = {'a': 1, 'b': {'b1': 2}, 'c': {'c1': 3}}
 295     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
 296     converted_dict = yaml.load(converted_csv_str)
 297     self.assertDictEqual(converted_dict, expected_output)
 298 
 299   def test_complex_nested_csv_str_to_json_str(self):
 300     csv_str = 'a.aa.aaa.aaaaa.a=1'
 301     json_str = '{a : {aa : {aaa : {aaaaa : {a : 1}}}}}'
 302     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
 303     self.assertEqual(converted_csv_str, json_str)
 304 
 305   def test_complex_nested_csv_str_load(self):
 306     csv_str = 'a.aa.aaa.aaaaa.a=1,a.a=2'
 307     expected_output = {'a': {'aa': {'aaa': {'aaaaa': {'a': 1}}}, 'a': 2}}
 308     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
 309     converted_dict = yaml.load(converted_csv_str)
 310     self.assertDictEqual(converted_dict, expected_output)
 311 
 312   def test_csv_str_load_supported_datatypes(self):
 313     csv_str = 'a=1,b=2.,c=[1,2,3],d=\'hello, there\',e=\"Hi.\"'
 314     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
 315     converted_dict = yaml.load(converted_csv_str)
 316     self.assertEqual(converted_dict['a'], 1)
 317     self.assertEqual(converted_dict['b'], 2.)
 318     self.assertEqual(converted_dict['c'], [1, 2, 3])
 319     self.assertEqual(converted_dict['d'], 'hello, there')
 320     self.assertEqual(converted_dict['e'], 'Hi.')
 321 
 322   def test_csv_str_load_unsupported_datatypes(self):
 323     csv_str = 'a=[[1,2,3],[4,5,6]]'
 324     self.assertRaises(ValueError,
 325                       params_dict.nested_csv_str_to_json_str,
 326                       csv_str)
 327 
 328   def test_csv_str_to_json_str_spacing(self):
 329     csv_str1 = 'a=1,b=2,c=3'
 330     csv_str2 = 'a = 1, b = 2, c = 3'
 331     json_str = '{a : 1, b : 2, c : 3}'
 332     converted_csv_str1 = params_dict.nested_csv_str_to_json_str(csv_str1)
 333     converted_csv_str2 = params_dict.nested_csv_str_to_json_str(csv_str2)
 334     self.assertEqual(converted_csv_str1, converted_csv_str2)
 335     self.assertEqual(converted_csv_str1, json_str)
 336     self.assertEqual(converted_csv_str2, json_str)

 338   def test_gcs_added_quotes(self):
 339     csv_str = 'a=gs://abc, b=gs://def'
 340     expected_output = '{a : \'gs://abc\', b : \'gs://def\'}'
 341     converted_csv_str = params_dict.nested_csv_str_to_json_str(csv_str)
