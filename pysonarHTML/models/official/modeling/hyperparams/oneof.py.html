  16 """Config class that supports oneof functionality."""
  17 
  18 from typing import Optional
  19 
  20 import dataclasses
  21 from official.modeling.hyperparams import base_config
  22 
  23 
  24 @dataclasses.dataclass
  25 class OneOfConfig(base_config.Config):
  26   """Configuration for configs with one of feature.
  27 
  28   Attributes:
  29     type: 'str', name of the field to select.
  30   """
  31   type: Optional[str] = None
  32 
  33   def as_dict(self):
  34     """Returns a dict representation of OneOfConfig.
  35 
  36     For the nested base_config.Config, a nested dict will be returned.
  37     """
  38     if self.type is None:
  39       return {'type': None}
  40     elif self.__dict__['type'] not in self.__dict__:
  41       raise ValueError(
  42           'type: {!r} is not a valid key!'.format(self.__dict__['type']))
  43     else:
  44       chosen_type = self.type
  45       chosen_value = self.__dict__[chosen_type]
  46       return {
  47           'type': self.type,
  48           chosen_type: self._export_config(chosen_value)
  49       }
  50 
  51   def get(self):
  52     """Returns selected config based on the value of type.
  53 
  54     If type is not set (None), None is returned.
  55     """
  56     chosen_type = self.type
  57     if chosen_type is None:
  58       return None
  59     if chosen_type not in self.__dict__:
  60       raise ValueError(
  61           'type: {!r} is not a valid key!'.format(self.type))
  62     return self.__dict__[chosen_type]
