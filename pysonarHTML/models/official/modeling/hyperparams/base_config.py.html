  34 @dataclasses.dataclass
  35 class Config(params_dict.ParamsDict):
  36   """The base configuration class that supports YAML/JSON based overrides.
  37 
  38   * It recursively enforces a whitelist of basic types and container types, so
  39     it avoids surprises with copy and reuse caused by unanticipated types.
  40   * It converts dict to Config even within sequences,
  41     e.g. for config = Config({'key': [([{'a': 42}],)]),
  42          type(config.key[0][0][0]) is Config rather than dict.
  43   """
  44 
  45   # It's safe to add bytes and other immutable types here.
  46   IMMUTABLE_TYPES = (str, int, float, bool, type(None))
  47   # It's safe to add set, frozenset and other collections here.
  48   SEQUENCE_TYPES = (list, tuple)
  49 
  50   default_params: dataclasses.InitVar[Optional[Mapping[str, Any]]] = None
  51   restrictions: dataclasses.InitVar[Optional[List[str]]] = None
  52 
  53   @classmethod
  54   def _isvalidsequence(cls, v):
  55     """Check if the input values are valid sequences.
  56 
  57     Args:
  58       v: Input sequence.
  59 
  60     Returns:
  61       True if the sequence is valid. Valid sequence includes the sequence
  62       type in cls.SEQUENCE_TYPES and element type is in cls.IMMUTABLE_TYPES or
  63       is dict or ParamsDict.
  64     """
  65     if not isinstance(v, cls.SEQUENCE_TYPES):
  66       return False
  67     return (all(isinstance(e, cls.IMMUTABLE_TYPES) for e in v) or
  68             all(isinstance(e, dict) for e in v) or
  69             all(isinstance(e, params_dict.ParamsDict) for e in v))
  70 
  71   @classmethod
  72   def _import_config(cls, v, subconfig_type):
  73     """Returns v with dicts converted to Configs, recursively."""
  74     if not issubclass(subconfig_type, params_dict.ParamsDict):
  75       raise TypeError(
  76           'Subconfig_type should be subclass of ParamsDict, found {!r}'.format(
  77               subconfig_type))
  78     if isinstance(v, cls.IMMUTABLE_TYPES):
  79       return v
  80     elif isinstance(v, cls.SEQUENCE_TYPES):
  81       # Only support one layer of sequence.
  82       if not cls._isvalidsequence(v):
  83         raise TypeError(
  84             'Invalid sequence: only supports single level {!r} of {!r} or '
  85             'dict or ParamsDict found: {!r}'.format(cls.SEQUENCE_TYPES,
  86                                                     cls.IMMUTABLE_TYPES, v))
  87       import_fn = functools.partial(
  88           cls._import_config, subconfig_type=subconfig_type)
  89       return type(v)(map(import_fn, v))
  90     elif isinstance(v, params_dict.ParamsDict):
  91       # Deepcopy here is a temporary solution for preserving type in nested
  92       # Config object.
  93       return copy.deepcopy(v)
  94     elif isinstance(v, dict):
  95       return subconfig_type(v)
  96     else:
  97       raise TypeError('Unknown type: {!r}'.format(type(v)))
  98 
  99   @classmethod
 100   def _export_config(cls, v):
 101     """Returns v with Configs converted to dicts, recursively."""
 102     if isinstance(v, cls.IMMUTABLE_TYPES):
 103       return v
 104     elif isinstance(v, cls.SEQUENCE_TYPES):
 105       return type(v)(map(cls._export_config, v))
 106     elif isinstance(v, params_dict.ParamsDict):
 107       return v.as_dict()
 108     elif isinstance(v, dict):
 109       raise TypeError('dict value not supported in converting.')
 110     else:
 111       raise TypeError('Unknown type: {!r}'.format(type(v)))
 112 
 113   @classmethod
 114   def _get_subconfig_type(cls, k) -> Type[params_dict.ParamsDict]:
 115     """Get element type by the field name.
 116 
 117     Args:
 118       k: the key/name of the field.
 119 
 120     Returns:
 121       Config as default. If a type annotation is found for `k`,
 122       1) returns the type of the annotation if it is subtype of ParamsDict;
 123       2) returns the element type if the annotation of `k` is List[SubType]
 124          or Tuple[SubType].
 125     """
 126     subconfig_type = Config
 127     if k in cls.__annotations__:
 128       # Directly Config subtype.
 129       type_annotation = cls.__annotations__[k]
 130       if (isinstance(type_annotation, type) and
 131           issubclass(type_annotation, Config)):
 132         subconfig_type = cls.__annotations__[k]
 133       else:
 134         # Check if the field is a sequence of subtypes.
 135         field_type = getattr(type_annotation, '__origin__', type(None))
 136         if (isinstance(field_type, type) and
 137             issubclass(field_type, cls.SEQUENCE_TYPES)):
 138           element_type = getattr(type_annotation, '__args__', [type(None)])[0]
 139           subconfig_type = (
 140               element_type if issubclass(element_type, params_dict.ParamsDict)
 141               else subconfig_type)
 142     return subconfig_type

 144   def __post_init__(self, default_params, restrictions, *args, **kwargs):
 145     super().__init__(default_params=default_params,
 146                      restrictions=restrictions,
 147                      *args,
 148                      **kwargs)
 149 
 150   def _set(self, k, v):
 151     """Overrides same method in ParamsDict.
 152 
 153     Also called by ParamsDict methods.
 154 
 155     Args:
 156       k: key to set.
 157       v: value.
 158 
 159     Raises:
 160       RuntimeError
 161     """
 162     subconfig_type = self._get_subconfig_type(k)
 163     if isinstance(v, dict):
 164       if k not in self.__dict__ or not self.__dict__[k]:
 165         # If the key not exist or the value is None, a new Config-family object
 166         # sould be created for the key.
 167         self.__dict__[k] = subconfig_type(v)
 168       else:
 169         self.__dict__[k].override(v)
 170     else:
 171       self.__dict__[k] = self._import_config(v, subconfig_type)
 172 
 173   def __setattr__(self, k, v):
 174     if k not in self.RESERVED_ATTR:
 175       if getattr(self, '_locked', False):
 176         raise ValueError('The Config has been locked. ' 'No change is allowed.')
 177     self._set(k, v)
 178 
 179   def _override(self, override_dict, is_strict=True):
 180     """Overrides same method in ParamsDict.
 181 
 182     Also called by ParamsDict methods.
 183 
 184     Args:
 185       override_dict: dictionary to write to .
 186       is_strict: If True, not allows to add new keys.
 187 
 188     Raises:
 189       KeyError: overriding reserved keys or keys not exist (is_strict=True).
 190     """
 191     for k, v in sorted(override_dict.items()):
 192       if k in self.RESERVED_ATTR:
 193         raise KeyError('The key {!r} is internally reserved. '
 194                        'Can not be overridden.'.format(k))
 195       if k not in self.__dict__:
 196         if is_strict:
 197           raise KeyError('The key {!r} does not exist in {!r}. '
 198                          'To extend the existing keys, use '
 199                          '`override` with `is_strict` = False.'.format(
 200                              k, type(self)))
 201         else:
 202           self._set(k, v)
 203       else:
 204         if isinstance(v, dict) and self.__dict__[k]:
 205           self.__dict__[k]._override(v, is_strict)  # pylint: disable=protected-access
 206         elif isinstance(v, params_dict.ParamsDict) and self.__dict__[k]:
 207           self.__dict__[k]._override(v.as_dict(), is_strict)  # pylint: disable=protected-access
 208         else:
 209           self._set(k, v)
 210 
 211   def as_dict(self):
 212     """Returns a dict representation of params_dict.ParamsDict.
 213 
 214     For the nested params_dict.ParamsDict, a nested dict will be returned.
 215     """
 216     return {
 217         k: self._export_config(v)
 218         for k, v in self.__dict__.items()
 219         if k not in self.RESERVED_ATTR
 220     }
 221 
 222   def replace(self, **kwargs):
 223     """Like `override`, but returns a copy with the current config unchanged."""
 224     params = self.__class__(self)
 225     params.override(kwargs, is_strict=True)
 226     return params
 227 
 228   @classmethod
 229   def from_yaml(cls, file_path: str):
 230     # Note: This only works if the Config has all default values.
 231     with tf.io.gfile.GFile(file_path, 'r') as f:
 232       loaded = yaml.load(f)
 233       config = cls()
 234       config.override(loaded)
 235       return config
 236 
 237   @classmethod
 238   def from_json(cls, file_path: str):
 239     """Wrapper for `from_yaml`."""
 240     return cls.from_yaml(file_path)
 241 
 242   @classmethod
 243   def from_args(cls, *args, **kwargs):
 244     """Builds a config from the given list of arguments."""
 245     attributes = list(cls.__annotations__.keys())
 246     default_params = {a: p for a, p in zip(attributes, args)}
 247     default_params.update(kwargs)
 248     return cls(default_params)
