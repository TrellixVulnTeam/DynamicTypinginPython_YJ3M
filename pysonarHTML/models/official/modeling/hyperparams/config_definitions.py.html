  28 @dataclasses.dataclass
  29 class DataConfig(base_config.Config):
  30   """The base configuration for building datasets.
  31 
  32   Attributes:
  33     input_path: The path to the input. It can be either (1) a file pattern, or
  34       (2) multiple file patterns separated by comma. It should not be specified
  35       when the following `tfds_name` is specified.
  36     tfds_name: The name of the tensorflow dataset (TFDS). It should not be
  37       specified when the above `input_path` is specified.
  38     tfds_split: A str indicating which split of the data to load from TFDS. It
  39       is required when above `tfds_name` is specified.
  40     global_batch_size: The global batch size across all replicas.
  41     is_training: Whether this data is used for training or not.
  42     drop_remainder: Whether the last batch should be dropped in the case it has
  43       fewer than `global_batch_size` elements.
  44     shuffle_buffer_size: The buffer size used for shuffling training data.
  45     cache: Whether to cache dataset examples. Can be used to avoid re-reading
  46       from disk on the second epoch. Requires significant memory overhead.
  47     cycle_length: The number of files that will be processed concurrently when
  48       interleaving files.
  49     block_length: The number of consecutive elements to produce from each input
  50       element before cycling to another input element when interleaving files.
  51     sharding: Whether sharding is used in the input pipeline.
  52     examples_consume: An `integer` specifying the number of examples it will
  53       produce. If positive, it only takes this number of examples and raises
  54       tf.error.OutOfRangeError after that. Default is -1, meaning it will
  55       exhaust all the examples in the dataset.
  56     tfds_data_dir: A str specifying the directory to read/write TFDS data.
  57     tfds_download: A bool to indicate whether to download data using TFDS.
  58     tfds_as_supervised: A bool. When loading dataset from TFDS, if True,
  59       the returned tf.data.Dataset will have a 2-tuple structure (input, label)
  60       according to builder.info.supervised_keys; if False, the default,
  61       the returned tf.data.Dataset will have a dictionary with all the features.
  62     tfds_skip_decoding_feature: A str to indicate which features are skipped
  63       for decoding when loading dataset from TFDS. Use comma to separate
  64       multiple features. The main use case is to skip the image/video decoding
  65       for better performance.
  66   """
  67   input_path: str = ""
  68   tfds_name: str = ""
  69   tfds_split: str = ""
  70   global_batch_size: int = 0
  71   is_training: bool = None
  72   drop_remainder: bool = True
  73   shuffle_buffer_size: int = 100
  74   cache: bool = False
  75   cycle_length: int = 8
  76   block_length: int = 1
  77   sharding: bool = True
  78   examples_consume: int = -1
  79   tfds_data_dir: str = ""
  80   tfds_download: bool = False
  81   tfds_as_supervised: bool = False
  82   tfds_skip_decoding_feature: str = ""

  85 @dataclasses.dataclass
  86 class RuntimeConfig(base_config.Config):
  87   """High-level configurations for Runtime.
  88 
  89   These include parameters that are not directly related to the experiment,
  90   e.g. directories, accelerator type, etc.
  91 
  92   Attributes:
  93     distribution_strategy: e.g. 'mirrored', 'tpu', etc.
  94     enable_xla: Whether or not to enable XLA.
  95     per_gpu_thread_count: thread count per GPU.
  96     gpu_thread_mode: Whether and how the GPU device uses its own threadpool.
  97     dataset_num_private_threads: Number of threads for a private threadpool
  98       created for all datasets computation.
  99     tpu: The address of the TPU to use, if any.
 100     num_gpus: The number of GPUs to use, if any.
 101     worker_hosts: comma-separated list of worker ip:port pairs for running
 102       multi-worker models with DistributionStrategy.
 103     task_index: If multi-worker training, the task index of this worker.
 104     all_reduce_alg: Defines the algorithm for performing all-reduce.
 105     num_packs: Sets `num_packs` in the cross device ops used in
 106       MirroredStrategy.  For details, see tf.distribute.NcclAllReduce.
 107     mixed_precision_dtype: dtype of mixed precision policy. It can be 'float32',
 108       'float16', or 'bfloat16'.
 109     loss_scale: The type of loss scale, or 'float' value. This is used when
 110       setting the mixed precision policy.
 111     run_eagerly: Whether or not to run the experiment eagerly.
 112     batchnorm_spatial_persistent: Whether or not to enable the spatial
 113       persistent mode for CuDNN batch norm kernel for improved GPU performance.
 114   """
 115   distribution_strategy: str = "mirrored"
 116   enable_xla: bool = False
 117   gpu_thread_mode: Optional[str] = None
 118   dataset_num_private_threads: Optional[int] = None
 119   per_gpu_thread_count: int = 0
 120   tpu: Optional[str] = None
 121   num_gpus: int = 0
 122   worker_hosts: Optional[str] = None
 123   task_index: int = -1
 124   all_reduce_alg: Optional[str] = None
 125   num_packs: int = 1
 126   loss_scale: Optional[Union[str, float]] = None
 127   mixed_precision_dtype: Optional[str] = None
 128   run_eagerly: bool = False
 129   batchnorm_spatial_persistent: bool = False

 163 @dataclasses.dataclass
 164 class TrainerConfig(base_config.Config):
 165   """Configuration for trainer.
 166 
 167   Attributes:
 168     optimizer_config: optimizer config, it includes optimizer, learning rate,
 169       and warmup schedule configs.
 170     train_tf_while_loop: whether or not to use tf while loop.
 171     train_tf_function: whether or not to use tf_function for training loop.
 172     eval_tf_function: whether or not to use tf_function for eval.
 173     steps_per_loop: number of steps per loop.
 174     summary_interval: number of steps between each summary.
 175     checkpoint_intervals: number of steps between checkpoints.
 176     max_to_keep: max checkpoints to keep.
 177     continuous_eval_timeout: maximum number of seconds to wait between
 178       checkpoints, if set to None, continuous eval will wait indefinetely.
 179   """
 180   optimizer_config: OptimizationConfig = OptimizationConfig()
 181   train_tf_while_loop: bool = True
 182   train_tf_function: bool = True
 183   eval_tf_function: bool = True
 184   steps_per_loop: int = 1000
 185   summary_interval: int = 1000
 186   checkpoint_interval: int = 1000
 187   max_to_keep: int = 5
 188   continuous_eval_timeout: Optional[int] = None

 213 def register_config_factory(name):
 214   """Register ExperimentConfig factory method."""
 215   return registry.register(_REGISTERED_CONFIGS, name)
 216 
 217 
 218 def get_exp_config_creater(exp_name: str):
 219   """Looks up ExperimentConfig factory methods."""
 220   exp_creater = registry.lookup(_REGISTERED_CONFIGS, exp_name)
 221   return exp_creater
