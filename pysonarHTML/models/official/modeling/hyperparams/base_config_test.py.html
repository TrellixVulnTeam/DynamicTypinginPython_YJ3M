  50   def assertHasSameTypes(self, c, d, msg=''):
  51     """Checks if a Config has the same structure as a given dict.
  52 
  53     Args:
  54       c: the Config object to be check.
  55       d: the reference dict object.
  56       msg: The error message to show when type mismatched.
  57     """
  58     # Make sure d is not a Config. Assume d is either
  59     # dictionary or primitive type and c is the Config or primitive types.
  60     self.assertNotIsInstance(d, base_config.Config)
  61     if isinstance(d, base_config.Config.IMMUTABLE_TYPES):
  62       self.assertEqual(pprint.pformat(c), pprint.pformat(d), msg=msg)
  63     elif isinstance(d, base_config.Config.SEQUENCE_TYPES):
  64       self.assertEqual(type(c), type(d), msg=msg)
  65       for i, v in enumerate(d):
  66         self.assertHasSameTypes(c[i], v, msg='{}[{!r}]'.format(msg, i))
  67     elif isinstance(d, dict):
  68       self.assertIsInstance(c, base_config.Config, msg=msg)
  69       for k, v in sorted(d.items()):
  70         self.assertHasSameTypes(getattr(c, k), v, msg='{}[{!r}]'.format(msg, k))
  71     else:
  72       raise TypeError('Unknown type: %r' % type(d))

  74   def assertImportExport(self, v):
  75     config = base_config.Config({'key': v})
  76     back = config.as_dict()['key']
  77     self.assertEqual(pprint.pformat(back), pprint.pformat(v))
  78     self.assertHasSameTypes(config.key, v, msg='=%s v' % pprint.pformat(v))

  85   def test_nested_config_types(self):
  86     config = DumpConfig3()
  87     self.assertIsInstance(config.e, DumpConfig1)
  88     self.assertIsInstance(config.h[0], DumpConfig1)
  89     self.assertIsInstance(config.h[1], DumpConfig1)
  90     self.assertIsInstance(config.g[0], DumpConfig1)
  91 
  92     config.override({'e': {'a': 2, 'b': 'new text'}})
  93     self.assertIsInstance(config.e, DumpConfig1)
  94     self.assertEqual(config.e.a, 2)
  95     self.assertEqual(config.e.b, 'new text')
  96 
  97     config.override({'h': [{'a': 3, 'b': 'new text 2'}]})
  98     self.assertIsInstance(config.h[0], DumpConfig1)
  99     self.assertLen(config.h, 1)
 100     self.assertEqual(config.h[0].a, 3)
 101     self.assertEqual(config.h[0].b, 'new text 2')
 102 
 103     config.override({'g': [{'a': 4, 'b': 'new text 3'}]})
 104     self.assertIsInstance(config.g[0], DumpConfig1)
 105     self.assertLen(config.g, 1)
 106     self.assertEqual(config.g[0].a, 4)
 107     self.assertEqual(config.g[0].b, 'new text 3')

 132   def test_override_is_strict_true(self):
 133     params = base_config.Config({
 134         'a': 'aa',
 135         'b': 2,
 136         'c': {
 137             'c1': 'cc',
 138             'c2': 20
 139         }
 140     })
 141     params.override({'a': 2, 'c': {'c1': 'ccc'}}, is_strict=True)
 142     self.assertEqual(params.a, 2)
 143     self.assertEqual(params.c.c1, 'ccc')
 144     with self.assertRaises(KeyError):
 145       params.override({'d': 'ddd'}, is_strict=True)
 146     with self.assertRaises(KeyError):
 147       params.override({'c': {'c3': 30}}, is_strict=True)
 148 
 149     config = base_config.Config({'key': [{'a': 42}]})
 150     config.override({'key': [{'b': 43}]})
 151     self.assertEqual(config.key[0].b, 43)
 152     with self.assertRaisesRegex(AttributeError, 'The key `a` does not exist'):
 153       _ = config.key[0].a

 256   def test_import_config(self):
 257     params = base_config.Config({'a': [{'b': 2}, {'c': {'d': 3}}]})
 258     self.assertLen(params.a, 2)
 259     self.assertEqual(params.a[0].b, 2)
 260     self.assertEqual(type(params.a[0]), base_config.Config)
 261     self.assertEqual(pprint.pformat(params.a[0].b), '2')
 262     self.assertEqual(type(params.a[1]), base_config.Config)
 263     self.assertEqual(type(params.a[1].c), base_config.Config)
 264     self.assertEqual(pprint.pformat(params.a[1].c.d), '3')

 266   def test_override(self):
 267     params = base_config.Config({'a': [{'b': 2}, {'c': {'d': 3}}]})
 268     params.override({'a': [{'b': 4}, {'c': {'d': 5}}]}, is_strict=False)
 269     self.assertEqual(type(params.a), list)
 270     self.assertEqual(type(params.a[0]), base_config.Config)
 271     self.assertEqual(pprint.pformat(params.a[0].b), '4')
 272     self.assertEqual(type(params.a[1]), base_config.Config)
 273     self.assertEqual(type(params.a[1].c), base_config.Config)
 274     self.assertEqual(pprint.pformat(params.a[1].c.d), '5')
