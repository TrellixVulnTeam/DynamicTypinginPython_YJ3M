  51 def _steps_to_run(current_step, total_steps, steps_per_loop):
  52   """Calculates steps to run on device."""
  53   if steps_per_loop <= 0:
  54     raise ValueError('steps_per_loop should be positive integer.')
  55   return min(total_steps - current_step, steps_per_loop)

  62 def metrics_as_dict(metric):
  63   """Puts input metric(s) into a list.
  64 
  65   Args:
  66     metric: metric(s) to be put into the list. `metric` could be a object, a
  67       list or a dict of tf.keras.metrics.Metric or has the `required_method`.
  68 
  69   Returns:
  70     A dictionary of valid metrics.
  71   """
  72   if isinstance(metric, tf.keras.metrics.Metric):
  73     metrics = {metric.name: metric}
  74   elif isinstance(metric, list):
  75     metrics = {m.name: m for m in metric}
  76   elif isinstance(metric, dict):
  77     metrics = metric
  78   elif not metric:
  79     return {}
  80   else:
  81     metrics = {'metric': metric}
  82   return metrics

  85 def metric_results(metric):
  86   """Collects results from the given metric(s)."""
  87   metrics = metrics_as_dict(metric)
  88   metric_result = {
  89       name: m.result().numpy().astype(float) for name, m in metrics.items()
  90   }
  91   return metric_result

  94 def reset_states(metric):
  95   """Resets states of the given metric(s)."""
  96   metrics = metrics_as_dict(metric)
  97   for m in metrics.values():
  98     m.reset_states()

 101 class SummaryWriter(object):
 102   """Simple SummaryWriter for writing dictionary of metrics.
 103 
 104   Attributes:
 105     writer: The tf.SummaryWriter.
 106   """
 107 
 108   def __init__(self, model_dir: Text, name: Text):
 109     """Inits SummaryWriter with paths.
 110 
 111     Arguments:
 112       model_dir: the model folder path.
 113       name: the summary subfolder name.
 114     """
 115     self.writer = tf.summary.create_file_writer(os.path.join(model_dir, name))
 116 
 117   def __call__(self, metrics: Union[Dict[Text, float], float], step: int):
 118     """Write metrics to summary with the given writer.
 119 
 120     Args:
 121       metrics: a dictionary of metrics values. Prefer dictionary.
 122       step: integer. The training step.
 123     """
 124     if not isinstance(metrics, dict):
 125       # Support scalar metric without name.
 126       logging.warning('Warning: summary writer prefer metrics as dictionary.')
 127       metrics = {'metric': metrics}
 128 
 129     with self.writer.as_default():
 130       for k, v in metrics.items():
 131         tf.summary.scalar(k, v, step=step)
 132       self.writer.flush()

 197   def _get_input_iterator(
 198       self, input_fn: Callable[..., tf.data.Dataset],
 199       strategy: tf.distribute.Strategy) -> Optional[Iterator[Any]]:
 200     """Returns distributed dataset iterator.
 201 
 202     Args:
 203       input_fn: (params: dict) -> tf.data.Dataset.
 204       strategy: an instance of tf.distribute.Strategy.
 205 
 206     Returns:
 207       An iterator that yields input tensors.
 208     """
 209 
 210     if input_fn is None:
 211       return None
 212     # When training with multiple TPU workers, datasets needs to be cloned
 213     # across workers. Since Dataset instance cannot be cloned in eager mode,
 214     # we instead pass callable that returns a dataset.
 215     if self._is_multi_host:
 216       return iter(
 217           strategy.experimental_distribute_datasets_from_function(input_fn))
 218     else:
 219       input_data = input_fn()
 220       return iter(strategy.experimental_distribute_dataset(input_data))

 222   def _create_replicated_step(self,
 223                               strategy,
 224                               model,
 225                               loss_fn,
 226                               optimizer,
 227                               metric=None):
 228     """Creates a single training step.
 229 
 230     Args:
 231       strategy: an instance of tf.distribute.Strategy.
 232       model: (Tensor, bool) -> Tensor. model function.
 233       loss_fn: (y_true: Tensor, y_pred: Tensor) -> Tensor.
 234       optimizer: tf.keras.optimizers.Optimizer.
 235       metric: tf.keras.metrics.Metric subclass.
 236 
 237     Returns:
 238       The training step callable.
 239     """
 240     metrics = metrics_as_dict(metric)
 241 
 242     def _replicated_step(inputs):
 243       """Replicated training step."""
 244       inputs, labels = inputs
 245 
 246       with tf.GradientTape() as tape:
 247         outputs = model(inputs, training=True)
 248         prediction_loss = loss_fn(labels, outputs)
 249         loss = tf.reduce_mean(prediction_loss)
 250         loss = loss / strategy.num_replicas_in_sync
 251         for m in metrics.values():
 252           m.update_state(labels, outputs)
 253 
 254       grads = tape.gradient(loss, model.trainable_variables)
 255       optimizer.apply_gradients(zip(grads, model.trainable_variables))
 256       return loss
 257 
 258     return _replicated_step
 259 
 260   def _create_train_step(self,
 261                          strategy,
 262                          model,
 263                          loss_fn,
 264                          optimizer,
 265                          metric=None):
 266     """Creates a distributed training step.
 267 
 268     Args:
 269       strategy: an instance of tf.distribute.Strategy.
 270       model: (Tensor, bool) -> Tensor. model function.
 271       loss_fn: (y_true: Tensor, y_pred: Tensor) -> Tensor.
 272       optimizer: tf.keras.optimizers.Optimizer.
 273       metric: tf.keras.metrics.Metric subclass.
 274 
 275     Returns:
 276       The training step callable.
 277     """
 278     replicated_step = self._create_replicated_step(strategy, model, loss_fn,
 279                                                    optimizer, metric)
 280 
 281     @tf.function
 282     def train_step(iterator, num_steps):
 283       """Performs a distributed training step.
 284 
 285       Args:
 286         iterator: an iterator that yields input tensors.
 287         num_steps: the number of steps in the loop.
 288 
 289       Returns:
 290         The loss tensor.
 291       """
 292       if not isinstance(num_steps, tf.Tensor):
 293         raise ValueError('steps should be an Tensor. Python object may cause '
 294                          'retracing.')
 295 
 296       per_replica_losses = strategy.run(
 297           replicated_step, args=(next(iterator),))
 298       for _ in tf.range(num_steps - 1):
 299         per_replica_losses = strategy.run(
 300             replicated_step, args=(next(iterator),))
 301 
 302       # For reporting, we returns the mean of losses.
 303       losses = tf.nest.map_structure(
 304           lambda x: strategy.reduce(tf.distribute.ReduceOp.MEAN, x, axis=None),
 305           per_replica_losses)
 306       return losses
 307 
 308     return train_step
 309 
 310   def _create_test_step(self, strategy, model, metric):
 311     """Creates a distributed test step."""
 312     metrics = metrics_as_dict(metric)
 313 
 314     @tf.function
 315     def test_step(iterator):
 316       """Calculates evaluation metrics on distributed devices."""
 317       if not metric:
 318         logging.info('Skip test_step because metric is None (%s)', metric)
 319         return None, None
 320 
 321       def _test_step_fn(inputs):
 322         """Replicated accuracy calculation."""
 323         inputs, labels = inputs
 324         model_outputs = model(inputs, training=False)
 325         for m in metrics.values():
 326           m.update_state(labels, model_outputs)
 327         return labels, model_outputs
 328 
 329       return strategy.run(_test_step_fn, args=(next(iterator),))
 330 
 331     return test_step

 333   def train(self,
 334             train_input_fn: Callable[[params_dict.ParamsDict], tf.data.Dataset],
 335             eval_input_fn: Callable[[params_dict.ParamsDict],
 336                                     tf.data.Dataset] = None,
 337             model_dir: Text = None,
 338             total_steps: int = 1,
 339             iterations_per_loop: int = 1,
 340             train_metric_fn: Callable[[], Any] = None,
 341             eval_metric_fn: Callable[[], Any] = None,
 342             summary_writer_fn: Callable[[Text, Text],
 343                                         SummaryWriter] = SummaryWriter,
 344             init_checkpoint: Callable[[tf.keras.Model], Any] = None,
 345             custom_callbacks: List[tf.keras.callbacks.Callback] = None,
 346             continuous_eval: bool = False,
 347             save_config: bool = True):
 348     """Runs distributed training.
 349 
 350     Args:
 351       train_input_fn: (params: dict) -> tf.data.Dataset training data input
 352         function.
 353       eval_input_fn: (Optional) same type as train_input_fn. If not None, will
 354         trigger evaluting metric on eval data. If None, will not run eval step.
 355       model_dir: the folder path for model checkpoints.
 356       total_steps: total training steps.
 357       iterations_per_loop: train steps per loop. After each loop, this job will
 358         update metrics like loss and save checkpoint.
 359       train_metric_fn: metric_fn for evaluation in train_step.
 360       eval_metric_fn: metric_fn for evaluation in test_step.
 361       summary_writer_fn: function to create summary writer.
 362       init_checkpoint: function to load checkpoint.
 363       custom_callbacks: A list of Keras Callbacks objects to run during
 364         training. More specifically, `on_batch_begin()`, `on_batch_end()`,
 365         methods are invoked during training.
 366       continuous_eval: If `True`, will continously run evaluation on every
 367         available checkpoints. If `False`, will do the evaluation once after the
 368         final step.
 369       save_config: bool. Whether to save params to model_dir.
 370     Returns:
 371       The training loss and eval metrics.
 372     """
 373     assert train_input_fn is not None
 374     if train_metric_fn and not callable(train_metric_fn):
 375       raise ValueError('if `train_metric_fn` is specified, '
 376                        'train_metric_fn must be a callable.')
 377     if eval_metric_fn and not callable(eval_metric_fn):
 378       raise ValueError('if `eval_metric_fn` is specified, '
 379                        'eval_metric_fn must be a callable.')
 380     train_metric_fn = train_metric_fn or _no_metric
 381     eval_metric_fn = eval_metric_fn or _no_metric
 382 
 383     if custom_callbacks and iterations_per_loop != 1:
 384       logging.warning(
 385           'It is sematically wrong to run callbacks when '
 386           'iterations_per_loop is not one (%s)', iterations_per_loop)
 387 
 388     custom_callbacks = custom_callbacks or []
 389 
 390     def _run_callbacks_on_batch_begin(batch):
 391       """Runs custom callbacks at the start of every step."""
 392       if not custom_callbacks:
 393         return
 394       for callback in custom_callbacks:
 395         if callback:
 396           callback.on_batch_begin(batch)
 397 
 398     def _run_callbacks_on_batch_end(batch):
 399       """Runs custom callbacks at the end of every step."""
 400       if not custom_callbacks:
 401         return
 402       for callback in custom_callbacks:
 403         if callback:
 404           callback.on_batch_end(batch)
 405 
 406     if save_config:
 407       self._save_config(model_dir)
 408 
 409     if FLAGS.save_checkpoint_freq:
 410       save_freq = FLAGS.save_checkpoint_freq
 411     else:
 412       save_freq = iterations_per_loop
 413 
 414     params = self._params
 415     strategy = self._strategy
 416     # To reduce unnecessary send/receive input pipeline operation, we place
 417     # input pipeline ops in worker task.
 418     train_iterator = self._get_input_iterator(train_input_fn, strategy)
 419     train_loss = None
 420     train_metric_result = None
 421     eval_metric_result = None
 422     tf.keras.backend.set_learning_phase(1)
 423     with strategy.scope():
 424       # To correctly place the model weights on accelerators,
 425       # model and optimizer should be created in scope.
 426       model = self.model_fn(params.as_dict())
 427       if not hasattr(model, 'optimizer'):
 428         raise ValueError('User should set optimizer attribute to model '
 429                          'inside `model_fn`.')
 430       optimizer = model.optimizer
 431 
 432       # Training loop starts here.
 433       checkpoint = tf.train.Checkpoint(model=model, optimizer=optimizer)
 434       latest_checkpoint_file = tf.train.latest_checkpoint(model_dir)
 435       initial_step = 0
 436       if latest_checkpoint_file:
 437         logging.info(
 438             'Checkpoint file %s found and restoring from '
 439             'checkpoint', latest_checkpoint_file)
 440         checkpoint.restore(latest_checkpoint_file)
 441         initial_step = optimizer.iterations.numpy()
 442         logging.info('Loading from checkpoint file completed. Init step %d',
 443                      initial_step)
 444       elif init_checkpoint:
 445         logging.info('Restoring from init checkpoint function')
 446         init_checkpoint(model)
 447         logging.info('Loading from init checkpoint file completed')
 448 
 449       current_step = optimizer.iterations.numpy()
 450       checkpoint_name = self.checkpoint_name
 451 
 452       eval_metric = eval_metric_fn()
 453       train_metric = train_metric_fn()
 454       train_summary_writer = summary_writer_fn(model_dir, 'eval_train')
 455       self.train_summary_writer = train_summary_writer.writer
 456 
 457       test_summary_writer = summary_writer_fn(model_dir, 'eval_test')
 458       self.eval_summary_writer = test_summary_writer.writer
 459 
 460     # Use training summary writer in TimeHistory if it's in use
 461     for cb in custom_callbacks:
 462       if isinstance(cb, keras_utils.TimeHistory):
 463         cb.summary_writer = self.train_summary_writer
 464 
 465     # Continue training loop.
 466     train_step = self._create_train_step(
 467         strategy=strategy,
 468         model=model,
 469         loss_fn=self.loss_fn(),
 470         optimizer=optimizer,
 471         metric=train_metric)
 472     test_step = None
 473     if eval_input_fn and eval_metric:
 474       self.global_train_step = model.optimizer.iterations
 475       test_step = self._create_test_step(strategy, model, metric=eval_metric)
 476 
 477     # Step-0 operations
 478     if current_step == 0 and not latest_checkpoint_file:
 479       _save_checkpoint(
 480           checkpoint, model_dir, checkpoint_name.format(step=current_step))
 481     if test_step:
 482       eval_iterator = self._get_input_iterator(eval_input_fn, strategy)
 483       eval_metric_result = self._run_evaluation(
 484           test_step, current_step, eval_metric, eval_iterator)
 485       logging.info(
 486           'Step: %s evalation metric = %s.', current_step, eval_metric_result)
 487       test_summary_writer(
 488           metrics=eval_metric_result, step=optimizer.iterations)
 489       reset_states(eval_metric)
 490 
 491     logging.info('Training started')
 492     last_save_checkpoint_step = current_step
 493     while current_step < total_steps:
 494 
 495       num_steps = _steps_to_run(current_step, total_steps, iterations_per_loop)
 496       _run_callbacks_on_batch_begin(current_step)
 497       train_loss = train_step(train_iterator,
 498                               tf.convert_to_tensor(num_steps, dtype=tf.int32))
 499       current_step += num_steps
 500 
 501       train_loss = tf.nest.map_structure(lambda x: x.numpy().astype(float),
 502                                          train_loss)
 503 
 504       _run_callbacks_on_batch_end(current_step - 1)
 505       if not isinstance(train_loss, dict):
 506         train_loss = {'total_loss': train_loss}
 507       if np.isnan(train_loss['total_loss']):
 508         raise ValueError('total loss is NaN.')
 509 
 510       if train_metric:
 511         train_metric_result = metric_results(train_metric)
 512         train_metric_result.update(train_loss)
 513       else:
 514         train_metric_result = train_loss
 515       if callable(optimizer.lr):
 516         train_metric_result.update(
 517             {'learning_rate': optimizer.lr(current_step).numpy()})
 518       else:
 519         train_metric_result.update({'learning_rate': optimizer.lr.numpy()})
 520       logging.info('Train Step: %d/%d  / loss = %s / training metric = %s',
 521                    current_step, total_steps, train_loss,
 522                    train_metric_result)
 523 
 524       train_summary_writer(
 525           metrics=train_metric_result, step=optimizer.iterations)
 526 
 527       # Saves model checkpoints and run validation steps at every
 528       # iterations_per_loop steps.
 529       # To avoid repeated model saving, we do not save after the last
 530       # step of training.
 531       if save_freq > 0 and current_step < total_steps and (
 532           current_step - last_save_checkpoint_step) >= save_freq:
 533         _save_checkpoint(checkpoint, model_dir,
 534                          checkpoint_name.format(step=current_step))
 535         last_save_checkpoint_step = current_step
 536 
 537       if continuous_eval and current_step < total_steps and test_step:
 538         eval_iterator = self._get_input_iterator(eval_input_fn, strategy)
 539         eval_metric_result = self._run_evaluation(test_step, current_step,
 540                                                   eval_metric, eval_iterator)
 541         logging.info('Step: %s evalation metric = %s.', current_step,
 542                      eval_metric_result)
 543         test_summary_writer(
 544             metrics=eval_metric_result, step=optimizer.iterations)
 545 
 546       # Re-initialize evaluation metric, except the last step.
 547       if eval_metric and current_step < total_steps:
 548         reset_states(eval_metric)
 549       if train_metric and current_step < total_steps:
 550         reset_states(train_metric)
 551 
 552     # Reaches the end of training and saves the last checkpoint.
 553     if last_save_checkpoint_step < total_steps:
 554       _save_checkpoint(checkpoint, model_dir,
 555                        checkpoint_name.format(step=current_step))
 556 
 557     if test_step:
 558       logging.info('Running final evaluation after training is complete.')
 559       eval_iterator = self._get_input_iterator(eval_input_fn, strategy)
 560       eval_metric_result = self._run_evaluation(test_step, current_step,
 561                                                 eval_metric, eval_iterator)
 562       logging.info('Final evaluation metric = %s.', eval_metric_result)
 563       test_summary_writer(
 564           metrics=eval_metric_result, step=optimizer.iterations)
 565 
 566     self.train_summary_writer.close()
 567     self.eval_summary_writer.close()
 568 
 569     return train_metric_result, eval_metric_result
 570 
 571   def _run_evaluation(self, test_step, current_training_step, metric,
 572                       test_iterator):
 573     """Runs validation steps and aggregate metrics."""
 574     if not test_iterator or not metric:
 575       logging.warning(
 576           'Both test_iterator (%s) and metrics (%s) must not be None.',
 577           test_iterator, metric)
 578       return None
 579     logging.info('Running evaluation after step: %s.', current_training_step)
 580     eval_step = 0
 581     while True:
 582       try:
 583         with tf.experimental.async_scope():
 584           test_step(test_iterator)
 585           eval_step += 1
 586       except (StopIteration, tf.errors.OutOfRangeError):
 587         tf.experimental.async_clear_error()
 588         break
 589 
 590     metric_result = metric_results(metric)
 591     logging.info('Total eval steps: [%d]', eval_step)
 592     logging.info('At training step: [%r] Validation metric = %r',
 593                  current_training_step, metric_result)
 594     return metric_result

 596   def evaluate_from_model_dir(
 597       self,
 598       model_dir: Text,
 599       eval_input_fn: Callable[[params_dict.ParamsDict], tf.data.Dataset],
 600       eval_metric_fn: Callable[[], Any],
 601       total_steps: int = -1,
 602       eval_timeout: int = None,
 603       min_eval_interval: int = 180,
 604       summary_writer_fn: Callable[[Text, Text], SummaryWriter] = SummaryWriter):
 605     """Runs distributed evaluation on model folder.
 606 
 607     Args:
 608       model_dir: the folder for storing model checkpoints.
 609       eval_input_fn: (Optional) same type as train_input_fn. If not None, will
 610         trigger evaluting metric on eval data. If None, will not run eval step.
 611       eval_metric_fn: metric_fn for evaluation in test_step.
 612       total_steps: total training steps. If the current step reaches the
 613         total_steps, the evaluation loop will stop.
 614       eval_timeout: The maximum number of seconds to wait between checkpoints.
 615         If left as None, then the process will wait indefinitely. Used by
 616         tf.train.checkpoints_iterator.
 617       min_eval_interval: The minimum number of seconds between yielding
 618         checkpoints. Used by tf.train.checkpoints_iterator.
 619       summary_writer_fn: function to create summary writer.
 620 
 621     Returns:
 622       Eval metrics dictionary of the last checkpoint.
 623     """
 624 
 625     if not model_dir:
 626       raise ValueError('model_dir must be set.')
 627 
 628     def terminate_eval():
 629       tf.logging.info('Terminating eval after %d seconds of no checkpoints' %
 630                       eval_timeout)
 631       return True
 632 
 633     summary_writer = summary_writer_fn(model_dir, 'eval')
 634     self.eval_summary_writer = summary_writer.writer
 635 
 636     # Read checkpoints from the given model directory
 637     # until `eval_timeout` seconds elapses.
 638     for checkpoint_path in tf.train.checkpoints_iterator(
 639         model_dir,
 640         min_interval_secs=min_eval_interval,
 641         timeout=eval_timeout,
 642         timeout_fn=terminate_eval):
 643       eval_metric_result, current_step = self.evaluate_checkpoint(
 644           checkpoint_path=checkpoint_path,
 645           eval_input_fn=eval_input_fn,
 646           eval_metric_fn=eval_metric_fn,
 647           summary_writer=summary_writer)
 648       if total_steps > 0 and current_step >= total_steps:
 649         logging.info('Evaluation finished after training step %d', current_step)
 650         break
 651     return eval_metric_result
 652 
 653   def evaluate_checkpoint(self,
 654                           checkpoint_path: Text,
 655                           eval_input_fn: Callable[[params_dict.ParamsDict],
 656                                                   tf.data.Dataset],
 657                           eval_metric_fn: Callable[[], Any],
 658                           summary_writer: SummaryWriter = None):
 659     """Runs distributed evaluation on the one checkpoint.
 660 
 661     Args:
 662       checkpoint_path: the checkpoint to evaluate.
 663       eval_input_fn: (Optional) same type as train_input_fn. If not None, will
 664         trigger evaluting metric on eval data. If None, will not run eval step.
 665       eval_metric_fn: metric_fn for evaluation in test_step.
 666       summary_writer: function to create summary writer.
 667 
 668     Returns:
 669       Eval metrics dictionary of the last checkpoint.
 670     """
 671     if not callable(eval_metric_fn):
 672       raise ValueError('if `eval_metric_fn` is specified, '
 673                        'eval_metric_fn must be a callable.')
 674 
 675     old_phrase = tf.keras.backend.learning_phase()
 676     tf.keras.backend.set_learning_phase(0)
 677     params = self._params
 678     strategy = self._strategy
 679     # To reduce unnecessary send/receive input pipeline operation, we place
 680     # input pipeline ops in worker task.
 681     with strategy.scope():
 682 
 683       # To correctly place the model weights on accelerators,
 684       # model and optimizer should be created in scope.
 685       model = self.model_fn(params.as_dict())
 686       checkpoint = tf.train.Checkpoint(model=model)
 687 
 688       eval_metric = eval_metric_fn()
 689       assert eval_metric, 'eval_metric does not exist'
 690       test_step = self._create_test_step(strategy, model, metric=eval_metric)
 691 
 692       logging.info('Starting to evaluate.')
 693       if not checkpoint_path:
 694         raise ValueError('checkpoint path is empty')
 695       reader = tf.compat.v1.train.NewCheckpointReader(checkpoint_path)
 696       current_step = reader.get_tensor(
 697           'optimizer/iter/.ATTRIBUTES/VARIABLE_VALUE')
 698       logging.info(
 699           'Checkpoint file %s found and restoring from '
 700           'checkpoint', checkpoint_path)
 701       checkpoint.restore(checkpoint_path)
 702 
 703       self.global_train_step = model.optimizer.iterations
 704       eval_iterator = self._get_input_iterator(eval_input_fn, strategy)
 705       eval_metric_result = self._run_evaluation(test_step, current_step,
 706                                                 eval_metric, eval_iterator)
 707       logging.info('Step: %s evalation metric = %s.', current_step,
 708                    eval_metric_result)
 709       summary_writer(metrics=eval_metric_result, step=current_step)
 710       reset_states(eval_metric)
 711 
 712     tf.keras.backend.set_learning_phase(old_phrase)
 713     return eval_metric_result, current_step

 719 class ExecutorBuilder(object):
 720   """Builder of DistributedExecutor.
 721 
 722   Example 1: Builds an executor with supported Strategy.
 723     builder = ExecutorBuilder(
 724         strategy_type='tpu',
 725         strategy_config={'tpu': '/bns/xxx'})
 726     dist_executor = builder.build_executor(
 727         params=params,
 728         model_fn=my_model_fn,
 729         loss_fn=my_loss_fn,
 730         metric_fn=my_metric_fn)
 731 
 732   Example 2: Builds an executor with customized Strategy.
 733     builder = ExecutorBuilder()
 734     builder.strategy = <some customized Strategy>
 735     dist_executor = builder.build_executor(
 736         params=params,
 737         model_fn=my_model_fn,
 738         loss_fn=my_loss_fn,
 739         metric_fn=my_metric_fn)
 740 
 741   Example 3: Builds a customized executor with customized Strategy.
 742     class MyDistributedExecutor(DistributedExecutor):
 743       # implementation ...
 744 
 745     builder = ExecutorBuilder()
 746     builder.strategy = <some customized Strategy>
 747     dist_executor = builder.build_executor(
 748         class_ctor=MyDistributedExecutor,
 749         params=params,
 750         model_fn=my_model_fn,
 751         loss_fn=my_loss_fn,
 752         metric_fn=my_metric_fn)
 753   """
 754 
 755   def __init__(self, strategy_type=None, strategy_config=None):
 756     _ = distribution_utils.configure_cluster(
 757         strategy_config.worker_hosts, strategy_config.task_index)
 758     """Constructor.
 759 
 760     Args:
 761       strategy_type: string. One of 'tpu', 'mirrored', 'multi_worker_mirrored'.
 762         If None. User is responsible to set the strategy before calling
 763         build_executor(...).
 764       strategy_config: necessary config for constructing the proper Strategy.
 765         Check strategy_flags_dict() for examples of the structure.
 766     """
 767     self._strategy = distribution_utils.get_distribution_strategy(
 768         distribution_strategy=strategy_type,
 769         num_gpus=strategy_config.num_gpus,
 770         all_reduce_alg=strategy_config.all_reduce_alg,
 771         num_packs=strategy_config.num_packs,
 772         tpu_address=strategy_config.tpu)
 773 
 774   @property
 775   def strategy(self):
 776     """Returns default checkpoint name."""
 777     return self._strategy
 778 
 779   @strategy.setter
 780   def strategy(self, new_strategy):
 781     """Sets default summary writer for the current thread."""
 782     self._strategy = new_strategy
 783 
 784   def build_executor(self,
 785                      class_ctor=DistributedExecutor,
 786                      params=None,
 787                      model_fn=None,
 788                      loss_fn=None,
 789                      **kwargs):
 790     """Creates an executor according to strategy type.
 791 
 792     See doc string of the DistributedExecutor.__init__ for more information of
 793     the
 794     input arguments.
 795 
 796     Args:
 797       class_ctor: A constructor of executor (default: DistributedExecutor).
 798       params: ParamsDict, all the model parameters and runtime parameters.
 799       model_fn: Keras model function.
 800       loss_fn: loss function.
 801       **kwargs: other arguments to the executor constructor.
 802 
 803     Returns:
 804       An instance of DistributedExecutor or its subclass.
 805     """
 806     if self._strategy is None:
 807       raise ValueError('`strategy` should not be None. You need to specify '
 808                        '`strategy_type` in the builder contructor or directly '
 809                        'set the `strategy` property of the builder.')
 810     return class_ctor(
 811         strategy=self._strategy,
 812         params=params,
 813         model_fn=model_fn,
 814         loss_fn=loss_fn,
 815         **kwargs)
