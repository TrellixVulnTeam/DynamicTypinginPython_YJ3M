  23 class LinearWarmup(tf.keras.optimizers.schedules.LearningRateSchedule):
  24   """Linear warmup schedule."""
  25 
  26   def __init__(self, after_warmup_lr_sched: Union[
  27       tf.keras.optimizers.schedules.LearningRateSchedule, float],
  28                warmup_steps: int, warmup_learning_rate: float,
  29                name: Optional[str] = None):
  30     """Add linear warmup schedule to a learning rate schedule.
  31 
  32     warmup_lr is the initial learning rate, the final learning rate of the
  33     init_warmup period is the initial learning rate of lr_schedule in use.
  34     The learning rate at each step linearly increased according to the following
  35     formula:
  36       learning_rate = warmup_lr + step / warmup_steps
  37                     * (final_warmup_lr - warmup_lr).
  38     Using warmup overrides the learning rate schedule by the number of warmup
  39     steps.
  40 
  41     Args:
  42       after_warmup_lr_sched: tf.keras.optimizers.schedules
  43                                 .LearningRateSchedule or a constant.
  44       warmup_steps: int. number of the warmup steps.
  45       warmup_learning_rate: floating point number. Initial learning rate for the
  46                       warmup.
  47       name: Optional, name of warmup schedule.
  48     """
  49     super(LinearWarmup, self).__init__()
  50     self._name = name
  51     self._after_warmup_lr_sched = after_warmup_lr_sched
  52     self._warmup_steps = warmup_steps
  53     self._init_warmup_lr = warmup_learning_rate
  54     if isinstance(after_warmup_lr_sched,
  55                   tf.keras.optimizers.schedules.LearningRateSchedule):
  56       self._final_warmup_lr = after_warmup_lr_sched(warmup_steps)
  57     else:
  58       self._final_warmup_lr = tf.cast(
  59           after_warmup_lr_sched, dtype=tf.float32)
  60 
  61   def __call__(self, step: int):
  62 
  63     global_step = tf.cast(step, dtype=tf.float32)
  64 
  65     linear_warmup_lr = (
  66         self._init_warmup_lr + global_step / self._warmup_steps *
  67         (self._final_warmup_lr - self._init_warmup_lr))
  68 
  69     if isinstance(self._after_warmup_lr_sched,
  70                   tf.keras.optimizers.schedules.LearningRateSchedule):
  71       after_warmup_lr = self._after_warmup_lr_sched(step)
  72     else:
  73       after_warmup_lr = tf.cast(self._after_warmup_lr_sched, dtype=tf.float32)
  74 
  75     lr = tf.cond(global_step < self._warmup_steps,
  76                  lambda: linear_warmup_lr,
  77                  lambda: after_warmup_lr)
  78     return lr
  79 
  80   def get_config(self) -> Mapping[str, Any]:
  81     if isinstance(self._after_warmup_lr_sched,
  82                   tf.keras.optimizers.schedules.LearningRateSchedule):
  83       config = {
  84           "after_warmup_lr_sched": self._after_warmup_lr_sched.get_config()}  # pytype: disable=attribute-error
  85     else:
  86       config = {"after_warmup_lr_sched": self._after_warmup_lr_sched}  # pytype: disable=attribute-error
  87 
  88     config.update({
  89         "warmup_steps": self._warmup_steps,
  90         "warmup_learning_rate": self._init_warmup_lr,
  91         "name": self._name
  92     })
  93     return config

  96 class PolynomialWarmUp(tf.keras.optimizers.schedules.LearningRateSchedule):
  97   """Applies polynomial warmup schedule on a given learning rate decay schedule.
  98   """

 100   def __init__(self,

 102                    tf.keras.optimizers.schedules.LearningRateSchedule, float],
 103                warmup_steps: int,
 104                power: float = 1.0,
 105                name: str = "PolynomialWarmup"):
 106     super(PolynomialWarmUp, self).__init__()
 107     if isinstance(after_warmup_lr_sched,
 108                   tf.keras.optimizers.schedules.LearningRateSchedule):
 109       self._initial_learning_rate = after_warmup_lr_sched(warmup_steps)
 110     else:
 111       self._initial_learning_rate = tf.cast(
 112           after_warmup_lr_sched, dtype=tf.float32)
 113 
 114     self._warmup_steps = warmup_steps
 115     self._power = power
 116     self._after_warmup_lr_sched = after_warmup_lr_sched
 117     self._name = name
 118 
 119   def __call__(self, step):
 120     with tf.name_scope(self._name or "PolynomialWarmUp") as name:
 121       # Implements polynomial warmup. i.e., if global_step < warmup_steps, the
 122       # learning rate will be `global_step/num_warmup_steps * init_lr`.
 123       global_step_float = tf.cast(step, tf.float32)
 124       warmup_steps_float = tf.cast(self._warmup_steps, tf.float32)
 125       warmup_percent_done = global_step_float / warmup_steps_float
 126       warmup_learning_rate = (
 127           self._initial_learning_rate *
 128           tf.math.pow(warmup_percent_done, self._power))
 129 
 130       if isinstance(self._after_warmup_lr_sched,
 131                     tf.keras.optimizers.schedules.LearningRateSchedule):
 132         after_warmup_lr = self._after_warmup_lr_sched(step)
 133       else:
 134         after_warmup_lr = tf.cast(self._after_warmup_lr_sched, dtype=tf.float32)
 135 
 136       return tf.cond(
 137           global_step_float < warmup_steps_float,
 138           lambda: warmup_learning_rate,
 139           lambda: after_warmup_lr,
 140           name=name)
 141

 143     if isinstance(self._after_warmup_lr_sched,
 144                   tf.keras.optimizers.schedules.LearningRateSchedule):
 145       config = {
 146           "after_warmup_lr_sched": self._after_warmup_lr_sched.get_config()}  # pytype: disable=attribute-error
 147     else:
 148       config = {"after_warmup_lr_sched": self._after_warmup_lr_sched}  # pytype: disable=attribute-error

 152         "power": self._power,
 153         "name": self._name
