  28   def test_sgd_optimizer(self):
  29     params = {
  30         'optimizer': {
  31             'type': 'sgd',
  32             'sgd': {'learning_rate': 0.1, 'momentum': 0.9}
  33         }
  34     }
  35     expected_optimizer_config = {
  36         'name': 'SGD',
  37         'learning_rate': 0.1,
  38         'decay': 0.0,
  39         'momentum': 0.9,
  40         'nesterov': False
  41     }
  42     opt_config = optimization_config.OptimizationConfig(params)
  43     opt_factory = optimizer_factory.OptimizerFactory(opt_config)
  44     lr = opt_factory.build_learning_rate()
  45     optimizer = opt_factory.build_optimizer(lr)
  46 
  47     self.assertIsInstance(optimizer, tf.keras.optimizers.SGD)
  48     self.assertEqual(expected_optimizer_config, optimizer.get_config())

  98   def test_stepwise_lr_schedule(self):
  99     params = {
 100         'optimizer': {
 101             'type': 'sgd',
 102             'sgd': {'learning_rate': 0.1, 'momentum': 0.9}
 103         },
 104         'learning_rate': {
 105             'type': 'stepwise',
 106             'stepwise': {'boundaries': [10000, 20000],
 107                          'values': [0.1, 0.01, 0.001]}
 108         }
 109     }
 110     expected_lr_step_values = [
 111         [0, 0.1],
 112         [5000, 0.1],
 113         [10000, 0.1],
 114         [10001, 0.01],
 115         [20000, 0.01],
 116         [20001, 0.001]
 117     ]
 118     opt_config = optimization_config.OptimizationConfig(params)
 119     opt_factory = optimizer_factory.OptimizerFactory(opt_config)
 120     lr = opt_factory.build_learning_rate()
 121 
 122     for step, value in expected_lr_step_values:
 123       self.assertAlmostEqual(lr(step).numpy(), value)
