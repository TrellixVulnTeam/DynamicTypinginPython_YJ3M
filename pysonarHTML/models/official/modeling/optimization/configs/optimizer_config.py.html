  23 @dataclasses.dataclass
  24 class SGDConfig(base_config.Config):
  25   """Configuration for SGD optimizer.
  26 
  27   The attributes for this class matches the arguments of tf.keras.optimizer.SGD.
  28 
  29   Attributes:
  30     name: name of the optimizer.
  31     learning_rate: learning_rate for SGD optimizer.
  32     decay: decay rate for SGD optimizer.
  33     nesterov: nesterov for SGD optimizer.
  34     momentum: momentum for SGD optimizer.
  35   """
  36   name: str = "SGD"
  37   learning_rate: float = 0.01
  38   decay: float = 0.0
  39   nesterov: bool = False
  40   momentum: float = 0.0

  43 @dataclasses.dataclass
  44 class AdamConfig(base_config.Config):
  45   """Configuration for Adam optimizer.
  46 
  47   The attributes for this class matches the arguments of
  48   tf.keras.optimizer.Adam.
  49 
  50   Attributes:
  51     name: name of the optimizer.
  52     learning_rate: learning_rate for Adam optimizer.
  53     beta_1: decay rate for 1st order moments.
  54     beta_2: decay rate for 2st order moments.
  55     epsilon: epsilon value used for numerical stability in Adam optimizer.
  56     amsgrad: boolean. Whether to apply AMSGrad variant of this algorithm from
  57     the paper "On the Convergence of Adam and beyond".
  58   """
  59   name: str = "Adam"
  60   learning_rate: float = 0.001
  61   beta_1: float = 0.9
  62   beta_2: float = 0.999
  63   epsilon: float = 1e-07
  64   amsgrad: bool = False
  65

  67 @dataclasses.dataclass
  68 class AdamWeightDecayConfig(base_config.Config):
  69   """Configuration for Adam optimizer with weight decay.
  70 
  71   Attributes:
  72     name: name of the optimizer.
  73     learning_rate: learning_rate for the optimizer.
  74     beta_1: decay rate for 1st order moments.
  75     beta_2: decay rate for 2st order moments.
  76     epsilon: epsilon value used for numerical stability in the optimizer.
  77     amsgrad: boolean. Whether to apply AMSGrad variant of this algorithm from
  78     the paper "On the Convergence of Adam and beyond".
  79     weight_decay_rate: float. Weight decay rate. Default to 0.
  80     include_in_weight_decay: list[str], or None. List of weight names to include
  81                              in weight decay.
  82     include_in_weight_decay: list[str], or None. List of weight names to not
  83                              include in weight decay.
  84   """
  85   name: str = "AdamWeightDecay"
  86   learning_rate: float = 0.001
  87   beta_1: float = 0.9
  88   beta_2: float = 0.999
  89   epsilon: float = 1e-07
  90   amsgrad: bool = False
  91   weight_decay_rate: float = 0.0
  92   include_in_weight_decay: Optional[List[str]] = None
  93   exclude_from_weight_decay: Optional[List[str]] = None

  96 @dataclasses.dataclass
  97 class LAMBConfig(base_config.Config):
  98   """Configuration for LAMB optimizer.
  99 
 100   The attributes for this class matches the arguments of
 101   tensorflow_addons.optimizers.LAMB.
 102 
 103   Attributes:
 104     name: name of the optimizer.
 105     learning_rate: learning_rate for Adam optimizer.
 106     beta_1: decay rate for 1st order moments.
 107     beta_2: decay rate for 2st order moments.
 108     epsilon: epsilon value used for numerical stability in LAMB optimizer.
 109     weight_decay_rate: float. Weight decay rate. Default to 0.
 110     exclude_from_weight_decay: List of regex patterns of variables excluded from
 111                                weight decay. Variables whose name contain a
 112                                substring matching the pattern will be excluded.
 113     exclude_from_layer_adaptation: List of regex patterns of variables excluded
 114                                    from layer adaptation. Variables whose name
 115                                    contain a substring matching the pattern will
 116                                    be excluded.
 117   """
 118   name: str = "LAMB"
 119   learning_rate: float = 0.001
 120   beta_1: float = 0.9
 121   beta_2: float = 0.999
 122   epsilon: float = 1e-6
 123   weight_decay_rate: float = 0.0
 124   exclude_from_weight_decay: Optional[List[str]] = None
 125   exclude_from_layer_adaptation: Optional[List[str]] = None
