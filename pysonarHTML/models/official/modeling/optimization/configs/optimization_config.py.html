  16 """Dataclasses for optimization configs.
  17 
  18 This file define the dataclass for optimization configs (OptimizationConfig).
  19 It also has two helper functions get_optimizer_config, and get_lr_config from
  20 an OptimizationConfig class.
  21 """
  22 from typing import Optional
  23 
  24 import dataclasses
  25 
  26 from official.modeling.hyperparams import base_config
  27 from official.modeling.hyperparams import oneof
  28 from official.modeling.optimization.configs import learning_rate_config as lr_cfg
  29 from official.modeling.optimization.configs import optimizer_config as opt_cfg
  30 
  31 
  32 @dataclasses.dataclass
  33 class OptimizerConfig(oneof.OneOfConfig):
  34   """Configuration for optimizer.
  35 
  36   Attributes:
  37     type: 'str', type of optimizer to be used, on the of fields below.
  38     sgd: sgd optimizer config.
  39     adam: adam optimizer config.
  40     adamw: adam with weight decay.
  41     lamb: lamb optimizer.
  42   """
  43   type: Optional[str] = None
  44   sgd: opt_cfg.SGDConfig = opt_cfg.SGDConfig()
  45   adam: opt_cfg.AdamConfig = opt_cfg.AdamConfig()
  46   adamw: opt_cfg.AdamWeightDecayConfig = opt_cfg.AdamWeightDecayConfig()
  47   lamb: opt_cfg.LAMBConfig = opt_cfg.LAMBConfig()
  48 
  49 
  50 @dataclasses.dataclass
  51 class LrConfig(oneof.OneOfConfig):
  52   """Configuration for lr schedule.
  53 
  54   Attributes:
  55     type: 'str', type of lr schedule to be used, on the of fields below.
  56     stepwise: stepwise learning rate config.
  57     exponential: exponential learning rate config.
  58     polynomial: polynomial learning rate config.
  59     cosine: cosine learning rate config.
  60   """
  61   type: Optional[str] = None
  62   stepwise: lr_cfg.StepwiseLrConfig = lr_cfg.StepwiseLrConfig()
  63   exponential: lr_cfg.ExponentialLrConfig = lr_cfg.ExponentialLrConfig()
  64   polynomial: lr_cfg.PolynomialLrConfig = lr_cfg.PolynomialLrConfig()
  65   cosine: lr_cfg.CosineLrConfig = lr_cfg.CosineLrConfig()
  66 
  67 
  68 @dataclasses.dataclass
  69 class WarmupConfig(oneof.OneOfConfig):
  70   """Configuration for lr schedule.
  71 
  72   Attributes:
  73     type: 'str', type of warmup schedule to be used, on the of fields below.
  74     linear: linear warmup config.
  75     polynomial: polynomial warmup config.
  76   """
  77   type: Optional[str] = None
  78   linear: lr_cfg.LinearWarmupConfig = lr_cfg.LinearWarmupConfig()
  79   polynomial: lr_cfg.PolynomialWarmupConfig = lr_cfg.PolynomialWarmupConfig()
  80 
  81 
  82 @dataclasses.dataclass
  83 class OptimizationConfig(base_config.Config):
  84   """Configuration for optimizer and learning rate schedule.
  85 
  86   Attributes:
  87     optimizer: optimizer oneof config.
  88     learning_rate: learning rate oneof config.
  89     warmup: warmup oneof config.
  90   """
  91   optimizer: OptimizerConfig = OptimizerConfig()
  92   learning_rate: LrConfig = LrConfig()
  93   warmup: WarmupConfig = WarmupConfig()
