  16 """Tests for optimization_config.py."""
  17 
  18 import tensorflow as tf
  19 
  20 from official.modeling.optimization.configs import learning_rate_config as lr_cfg
  21 from official.modeling.optimization.configs import optimization_config
  22 from official.modeling.optimization.configs import optimizer_config as opt_cfg
  23 
  24 
  25 class OptimizerConfigTest(tf.test.TestCase):
  26 
  27   def test_no_optimizer(self):
  28     optimizer = optimization_config.OptimizationConfig({}).optimizer.get()
  29     self.assertEqual(optimizer, None)
  30 
  31   def test_no_lr_schedule(self):
  32     lr = optimization_config.OptimizationConfig({}).learning_rate.get()
  33     self.assertEqual(lr, None)
  34 
  35   def test_no_warmup_schedule(self):
  36     warmup = optimization_config.OptimizationConfig({}).warmup.get()
  37     self.assertEqual(warmup, None)
  38 
  39   def test_config(self):
  40     opt_config = optimization_config.OptimizationConfig({
  41         'optimizer': {
  42             'type': 'sgd',
  43             'sgd': {}  # default config
  44         },
  45         'learning_rate': {
  46             'type': 'polynomial',
  47             'polynomial': {}
  48         },
  49         'warmup': {
  50             'type': 'linear'
  51         }
  52     })
  53     self.assertEqual(opt_config.optimizer.get(),
  54                      opt_cfg.SGDConfig())
  55     self.assertEqual(opt_config.learning_rate.get(),
  56                      lr_cfg.PolynomialLrConfig())
  57     self.assertEqual(opt_config.warmup.get(),
  58                      lr_cfg.LinearWarmupConfig())
  59 
  60 if __name__ == '__main__':
  61   tf.test.main()
