  28 def define_gin_flags():
  29   """Define common gin configurable flags."""
  30   flags.DEFINE_multi_string('gin_file', None,
  31                             'List of paths to the config files.')
  32   flags.DEFINE_multi_string(
  33       'gin_param', None, 'Newline separated list of Gin parameter bindings.')
  34 
  35 
  36 def define_common_hparams_flags():
  37   """Define the common flags across models."""
  38 
  39   flags.DEFINE_string(
  40       'model_dir',
  41       default=None,
  42       help=('The directory where the model and training/evaluation summaries'
  43             'are stored.'))
  44 
  45   flags.DEFINE_integer(
  46       'train_batch_size', default=None, help='Batch size for training.')
  47 
  48   flags.DEFINE_integer(
  49       'eval_batch_size', default=None, help='Batch size for evaluation.')
  50 
  51   flags.DEFINE_string(
  52       'precision',
  53       default=None,
  54       help=('Precision to use; one of: {bfloat16, float32}'))
  55 
  56   flags.DEFINE_string(
  57       'config_file',
  58       default=None,
  59       help=('A YAML file which specifies overrides. Note that this file can be '
  60             'used as an override template to override the default parameters '
  61             'specified in Python. If the same parameter is specified in both '
  62             '`--config_file` and `--params_override`, the one in '
  63             '`--params_override` will be used finally.'))
  64 
  65   flags.DEFINE_string(
  66       'params_override',
  67       default=None,
  68       help=('a YAML/JSON string or a YAML file which specifies additional '
  69             'overrides over the default parameters and those specified in '
  70             '`--config_file`. Note that this is supposed to be used only to '
  71             'override the model parameters, but not the parameters like TPU '
  72             'specific flags. One canonical use case of `--config_file` and '
  73             '`--params_override` is users first define a template config file '
  74             'using `--config_file`, then use `--params_override` to adjust the '
  75             'minimal set of tuning parameters, for example setting up different'
  76             ' `train_batch_size`. '
  77             'The final override order of parameters: default_model_params --> '
  78             'params from config_file --> params in params_override.'
  79             'See also the help message of `--config_file`.'))
  80   flags.DEFINE_integer('save_checkpoint_freq', None,
  81                        'Number of steps to save checkpoint.')
  82 
  83 
  84 def initialize_common_flags():
  85   """Define the common flags across models."""
  86   define_common_hparams_flags()
  87 
  88   flags_core.define_device(tpu=True)
  89   flags_core.define_base(
  90       num_gpu=True, model_dir=False, data_dir=False, batch_size=False)
  91   flags_core.define_distribution(worker_hosts=True, task_index=True)
  92   flags_core.define_performance(all_reduce_alg=True, num_packs=True)
  93 
  94   # Reset the default value of num_gpus to zero.
  95   FLAGS.num_gpus = 0
  96 
  97   flags.DEFINE_string(
  98       'strategy_type', 'mirrored', 'Type of distribute strategy.'
  99       'One of mirrored, tpu and multiworker.')
 100

 102 def strategy_flags_dict():
 103   """Returns TPU and/or GPU related flags in a dictionary."""
 104   return {
 105       'distribution_strategy': FLAGS.strategy_type,
 106       # TPUStrategy related flags.
 107       'tpu': FLAGS.tpu,
 108       # MultiWorkerMirroredStrategy related flags.
 109       'all_reduce_alg': FLAGS.all_reduce_alg,
 110       'worker_hosts': FLAGS.worker_hosts,
 111       'task_index': FLAGS.task_index,
 112       # MirroredStrategy and OneDeviceStrategy
 113       'num_gpus': FLAGS.num_gpus,
 114       'num_packs': FLAGS.num_packs,
 115   }
 116 
 117 
 118 def hparam_flags_dict():
 119   """Returns model params related flags in a dictionary."""
 120   return {
 121       'data_dir': FLAGS.data_dir,
 122       'model_dir': FLAGS.model_dir,
 123       'train_batch_size': FLAGS.train_batch_size,
 124       'eval_batch_size': FLAGS.eval_batch_size,
 125       'precision': FLAGS.precision,
 126       'config_file': FLAGS.config_file,
 127       'params_override': FLAGS.params_override,
 128   }
