  26 class PastStopThresholdTest(tf.test.TestCase):
  27   """Tests for past_stop_threshold."""
  28 
  29   def setUp(self):
  30     super(PastStopThresholdTest, self).setUp()
  31     tf.compat.v1.disable_eager_execution()
  32 
  33   def test_past_stop_threshold(self):
  34     """Tests for normal operating conditions."""
  35     self.assertTrue(model_helpers.past_stop_threshold(0.54, 1))
  36     self.assertTrue(model_helpers.past_stop_threshold(54, 100))
  37     self.assertFalse(model_helpers.past_stop_threshold(0.54, 0.1))
  38     self.assertFalse(model_helpers.past_stop_threshold(-0.54, -1.5))
  39     self.assertTrue(model_helpers.past_stop_threshold(-0.54, 0))
  40     self.assertTrue(model_helpers.past_stop_threshold(0, 0))
  41     self.assertTrue(model_helpers.past_stop_threshold(0.54, 0.54))
  42 
  43   def test_past_stop_threshold_none_false(self):
  44     """Tests that check None returns false."""
  45     self.assertFalse(model_helpers.past_stop_threshold(None, -1.5))
  46     self.assertFalse(model_helpers.past_stop_threshold(None, None))
  47     self.assertFalse(model_helpers.past_stop_threshold(None, 1.5))
  48     # Zero should be okay, though.
  49     self.assertTrue(model_helpers.past_stop_threshold(0, 1.5))
  50 
  51   def test_past_stop_threshold_not_number(self):
  52     """Tests for error conditions."""
  53     with self.assertRaises(ValueError):
  54       model_helpers.past_stop_threshold("str", 1)
  55 
  56     with self.assertRaises(ValueError):
  57       model_helpers.past_stop_threshold("str", tf.constant(5))
  58 
  59     with self.assertRaises(ValueError):
  60       model_helpers.past_stop_threshold("str", "another")
  61 
  62     with self.assertRaises(ValueError):
  63       model_helpers.past_stop_threshold(0, None)
  64 
  65     with self.assertRaises(ValueError):
  66       model_helpers.past_stop_threshold(0.7, "str")
  67 
  68     with self.assertRaises(ValueError):
  69       model_helpers.past_stop_threshold(tf.constant(4), None)

  72 class SyntheticDataTest(tf.test.TestCase):
  73   """Tests for generate_synthetic_data."""
  74 
  75   def test_generate_synethetic_data(self):
  76     input_element, label_element = tf.compat.v1.data.make_one_shot_iterator(
  77         model_helpers.generate_synthetic_data(input_shape=tf.TensorShape([5]),
  78                                               input_value=123,
  79                                               input_dtype=tf.float32,
  80                                               label_shape=tf.TensorShape([]),
  81                                               label_value=456,
  82                                               label_dtype=tf.int32)).get_next()
  83 
  84     with self.session() as sess:
  85       for n in range(5):
  86         inp, lab = sess.run((input_element, label_element))
  87         self.assertAllClose(inp, [123., 123., 123., 123., 123.])
  88         self.assertEquals(lab, 456)
  89 
  90   def test_generate_only_input_data(self):
  91     d = model_helpers.generate_synthetic_data(
  92         input_shape=tf.TensorShape([4]),
  93         input_value=43.5,
  94         input_dtype=tf.float32)
  95 
  96     element = tf.compat.v1.data.make_one_shot_iterator(d).get_next()
  97     self.assertFalse(isinstance(element, tuple))
  98 
  99     with self.session() as sess:
 100       inp = sess.run(element)
 101       self.assertAllClose(inp, [43.5, 43.5, 43.5, 43.5])
 102 
 103   def test_generate_nested_data(self):
 104     d = model_helpers.generate_synthetic_data(
 105         input_shape={'a': tf.TensorShape([2]),
 106                      'b': {'c': tf.TensorShape([3]), 'd': tf.TensorShape([])}},
 107         input_value=1.1)
 108 
 109     element = tf.compat.v1.data.make_one_shot_iterator(d).get_next()
 110     self.assertIn('a', element)
 111     self.assertIn('b', element)
 112     self.assertEquals(len(element['b']), 2)
 113     self.assertIn('c', element['b'])
 114     self.assertIn('d', element['b'])
 115     self.assertNotIn('c', element)
 116 
 117     with self.session() as sess:
 118       inp = sess.run(element)
 119       self.assertAllClose(inp['a'], [1.1, 1.1])
 120       self.assertAllClose(inp['b']['c'], [1.1, 1.1, 1.1])
 121       self.assertAllClose(inp['b']['d'], 1.1)
