  29 class BatchTimestamp(object):
  30   """A structure to store batch time stamp."""
  31 
  32   def __init__(self, batch_index, timestamp):
  33     self.batch_index = batch_index
  34     self.timestamp = timestamp
  35 
  36   def __repr__(self):
  37     return "'BatchTimestamp<batch_index: {}, timestamp: {}>'".format(
  38         self.batch_index, self.timestamp)

  41 class TimeHistory(tf.keras.callbacks.Callback):
  42   """Callback for Keras models."""
  43 
  44   def __init__(self, batch_size, log_steps, initial_step=0, logdir=None):
  45     """Callback for logging performance.
  46 
  47     Args:
  48       batch_size: Total batch size.
  49       log_steps: Interval of steps between logging of batch level stats.
  50       initial_step: Optional, initial step.
  51       logdir: Optional directory to write TensorBoard summaries.
  52     """
  53     # TODO(wcromar): remove this parameter and rely on `logs` parameter of
  54     # on_train_batch_end()
  55     self.batch_size = batch_size
  56     super(TimeHistory, self).__init__()
  57     self.log_steps = log_steps
  58     self.last_log_step = initial_step
  59     self.steps_before_epoch = initial_step
  60     self.steps_in_epoch = 0
  61     self.start_time = None
  62 
  63     if logdir:
  64       self.summary_writer = tf.summary.create_file_writer(logdir)
  65     else:
  66       self.summary_writer = None
  67 
  68     # Logs start of step 1 then end of each step based on log_steps interval.
  69     self.timestamp_log = []
  70 
  71     # Records the time each epoch takes to run from start to finish of epoch.
  72     self.epoch_runtime_log = []
  73 
  74   @property
  75   def global_steps(self):
  76     """The current 1-indexed global step."""
  77     return self.steps_before_epoch + self.steps_in_epoch
  78 
  79   @property
  80   def average_steps_per_second(self):
  81     """The average training steps per second across all epochs."""
  82     return self.global_steps / sum(self.epoch_runtime_log)
  83 
  84   @property
  85   def average_examples_per_second(self):
  86     """The average number of training examples per second across all epochs."""
  87     return self.average_steps_per_second * self.batch_size
  88 
  89   def get_examples_per_sec(self, warmup=1):
  90     """Calculates examples/sec through timestamp_log and skip warmup period."""
  91     # First entry in timestamp_log is the start of the step 1. The rest of the
  92     # entries are the end of each step recorded.
  93     time_log = self.timestamp_log
  94     seconds = time_log[-1].timestamp - time_log[warmup].timestamp
  95     steps = time_log[-1].batch_index - time_log[warmup].batch_index
  96     return self.batch_size * steps / seconds
  97 
  98   def get_startup_time(self, start_time_sec):
  99     return self.timestamp_log[0].timestamp - start_time_sec
 100 
 101   def on_train_end(self, logs=None):
 102     self.train_finish_time = time.time()
 103 
 104     if self.summary_writer:
 105       self.summary_writer.flush()
 106 
 107   def on_epoch_begin(self, epoch, logs=None):
 108     self.epoch_start = time.time()
 109 
 110   def on_batch_begin(self, batch, logs=None):
 111     if not self.start_time:
 112       self.start_time = time.time()
 113 
 114     # Record the timestamp of the first global step
 115     if not self.timestamp_log:
 116       self.timestamp_log.append(BatchTimestamp(self.global_steps,
 117                                                self.start_time))
 118 
 119   def on_batch_end(self, batch, logs=None):
 120     """Records elapse time of the batch and calculates examples per second."""
 121     self.steps_in_epoch = batch + 1
 122     steps_since_last_log = self.global_steps - self.last_log_step
 123     if steps_since_last_log >= self.log_steps:
 124       now = time.time()
 125       elapsed_time = now - self.start_time
 126       steps_per_second = steps_since_last_log / elapsed_time
 127       examples_per_second = steps_per_second * self.batch_size
 128 
 129       self.timestamp_log.append(BatchTimestamp(self.global_steps, now))
 130       logging.info(
 131           'TimeHistory: %.2f seconds, %.2f examples/second between steps %d '
 132           'and %d', elapsed_time, examples_per_second, self.last_log_step,
 133           self.global_steps)
 134 
 135       if self.summary_writer:
 136         with self.summary_writer.as_default():
 137           tf.summary.scalar('steps_per_second', steps_per_second,
 138                             self.global_steps)
 139           tf.summary.scalar('examples_per_second', examples_per_second,
 140                             self.global_steps)
 141 
 142       self.last_log_step = self.global_steps
 143       self.start_time = None
 144 
 145   def on_epoch_end(self, epoch, logs=None):
 146     epoch_run_time = time.time() - self.epoch_start
 147     self.epoch_runtime_log.append(epoch_run_time)
 148 
 149     self.steps_before_epoch += self.steps_in_epoch
 150     self.steps_in_epoch = 0

 153 class SimpleCheckpoint(tf.keras.callbacks.Callback):
 154   """Keras callback to save tf.train.Checkpoints."""
 155 
 156   def __init__(self, checkpoint_manager):
 157     super(SimpleCheckpoint, self).__init__()
 158     self.checkpoint_manager = checkpoint_manager
 159 
 160   def on_epoch_end(self, epoch, logs=None):
 161     step_counter = self.checkpoint_manager._step_counter.numpy()  # pylint: disable=protected-access
 162     self.checkpoint_manager.save(checkpoint_number=step_counter)

 174 def set_gpu_thread_mode_and_count(gpu_thread_mode,
 175                                   datasets_num_private_threads,
 176                                   num_gpus, per_gpu_thread_count):
 177   """Set GPU thread mode and count, and adjust dataset threads count."""
 178   cpu_count = multiprocessing.cpu_count()
 179   logging.info('Logical CPU cores: %s', cpu_count)
 180 
 181   # Allocate private thread pool for each GPU to schedule and launch kernels
 182   per_gpu_thread_count = per_gpu_thread_count or 2
 183   os.environ['TF_GPU_THREAD_MODE'] = gpu_thread_mode
 184   os.environ['TF_GPU_THREAD_COUNT'] = str(per_gpu_thread_count)
 185   logging.info('TF_GPU_THREAD_COUNT: %s',
 186                os.environ['TF_GPU_THREAD_COUNT'])
 187   logging.info('TF_GPU_THREAD_MODE: %s',
 188                os.environ['TF_GPU_THREAD_MODE'])
 189 
 190   # Limit data preprocessing threadpool to CPU cores minus number of total GPU
 191   # private threads and memory copy threads.
 192   total_gpu_thread_count = per_gpu_thread_count * num_gpus
 193   num_runtime_threads = num_gpus
 194   if not datasets_num_private_threads:
 195     datasets_num_private_threads = min(
 196         cpu_count - total_gpu_thread_count - num_runtime_threads,
 197         num_gpus * 8)
 198     logging.info('Set datasets_num_private_threads to %s',
 199                  datasets_num_private_threads)
