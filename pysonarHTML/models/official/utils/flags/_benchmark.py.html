  26 def define_log_steps():
  27   flags.DEFINE_integer(
  28       name="log_steps", default=100,
  29       help="Frequency with which to log timing information with TimeHistory.")
  30 
  31   return []

  34 def define_benchmark(benchmark_log_dir=True, bigquery_uploader=True):
  35   """Register benchmarking flags.
  36 
  37   Args:
  38     benchmark_log_dir: Create a flag to specify location for benchmark logging.
  39     bigquery_uploader: Create flags for uploading results to BigQuery.
  40 
  41   Returns:
  42     A list of flags for core.py to marks as key flags.
  43   """
  44 
  45   key_flags = []
  46 
  47   flags.DEFINE_enum(
  48       name="benchmark_logger_type", default="BaseBenchmarkLogger",
  49       enum_values=["BaseBenchmarkLogger", "BenchmarkFileLogger"],
  50       help=help_wrap("The type of benchmark logger to use. Defaults to using "
  51                      "BaseBenchmarkLogger which logs to STDOUT. Different "
  52                      "loggers will require other flags to be able to work."))
  53   flags.DEFINE_string(
  54       name="benchmark_test_id", short_name="bti", default=None,
  55       help=help_wrap("The unique test ID of the benchmark run. It could be the "
  56                      "combination of key parameters. It is hardware "
  57                      "independent and could be used compare the performance "
  58                      "between different test runs. This flag is designed for "
  59                      "human consumption, and does not have any impact within "
  60                      "the system."))
  61 
  62   define_log_steps()
  63 
  64   if benchmark_log_dir:
  65     flags.DEFINE_string(
  66         name="benchmark_log_dir", short_name="bld", default=None,
  67         help=help_wrap("The location of the benchmark logging.")
  68     )
  69 
  70   if bigquery_uploader:
  71     flags.DEFINE_string(
  72         name="gcp_project", short_name="gp", default=None,
  73         help=help_wrap(
  74             "The GCP project name where the benchmark will be uploaded."))
  75 
  76     flags.DEFINE_string(
  77         name="bigquery_data_set", short_name="bds", default="test_benchmark",
  78         help=help_wrap(
  79             "The Bigquery dataset name where the benchmark will be uploaded."))
  80 
  81     flags.DEFINE_string(
  82         name="bigquery_run_table", short_name="brt", default="benchmark_run",
  83         help=help_wrap("The Bigquery table name where the benchmark run "
  84                        "information will be uploaded."))
  85 
  86     flags.DEFINE_string(
  87         name="bigquery_run_status_table", short_name="brst",
  88         default="benchmark_run_status",
  89         help=help_wrap("The Bigquery table name where the benchmark run "
  90                        "status information will be uploaded."))
  91 
  92     flags.DEFINE_string(
  93         name="bigquery_metric_table", short_name="bmt",
  94         default="benchmark_metric",
  95         help=help_wrap("The Bigquery table name where the benchmark metric "
  96                        "information will be uploaded."))
  97 
  98   @flags.multi_flags_validator(
  99       ["benchmark_logger_type", "benchmark_log_dir"],
 100       message="--benchmark_logger_type=BenchmarkFileLogger will require "
 101               "--benchmark_log_dir being set")
 102   def _check_benchmark_log_dir(flags_dict):
 103     benchmark_logger_type = flags_dict["benchmark_logger_type"]
 104     if benchmark_logger_type == "BenchmarkFileLogger":
 105       return flags_dict["benchmark_log_dir"]
 106     return True
 107 
 108   return key_flags
