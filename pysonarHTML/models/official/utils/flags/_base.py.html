  26 def define_base(data_dir=True, model_dir=True, clean=False, train_epochs=False,
  27                 epochs_between_evals=False, stop_threshold=False,
  28                 batch_size=True, num_gpu=False, hooks=False, export_dir=False,
  29                 distribution_strategy=False, run_eagerly=False):
  30   """Register base flags.
  31 
  32   Args:
  33     data_dir: Create a flag for specifying the input data directory.
  34     model_dir: Create a flag for specifying the model file directory.
  35     clean: Create a flag for removing the model_dir.
  36     train_epochs: Create a flag to specify the number of training epochs.
  37     epochs_between_evals: Create a flag to specify the frequency of testing.
  38     stop_threshold: Create a flag to specify a threshold accuracy or other
  39       eval metric which should trigger the end of training.
  40     batch_size: Create a flag to specify the batch size.
  41     num_gpu: Create a flag to specify the number of GPUs used.
  42     hooks: Create a flag to specify hooks for logging.
  43     export_dir: Create a flag to specify where a SavedModel should be exported.
  44     distribution_strategy: Create a flag to specify which Distribution Strategy
  45       to use.
  46     run_eagerly: Create a flag to specify to run eagerly op by op.
  47   Returns:
  48     A list of flags for core.py to marks as key flags.
  49   """
  50   key_flags = []
  51 
  52   if data_dir:
  53     flags.DEFINE_string(
  54         name="data_dir", short_name="dd", default="/tmp",
  55         help=help_wrap("The location of the input data."))
  56     key_flags.append("data_dir")
  57 
  58   if model_dir:
  59     flags.DEFINE_string(
  60         name="model_dir", short_name="md", default="/tmp",
  61         help=help_wrap("The location of the model checkpoint files."))
  62     key_flags.append("model_dir")
  63 
  64   if clean:
  65     flags.DEFINE_boolean(
  66         name="clean", default=False,
  67         help=help_wrap("If set, model_dir will be removed if it exists."))
  68     key_flags.append("clean")
  69 
  70   if train_epochs:
  71     flags.DEFINE_integer(
  72         name="train_epochs", short_name="te", default=1,
  73         help=help_wrap("The number of epochs used to train."))
  74     key_flags.append("train_epochs")
  75 
  76   if epochs_between_evals:
  77     flags.DEFINE_integer(
  78         name="epochs_between_evals", short_name="ebe", default=1,
  79         help=help_wrap("The number of training epochs to run between "
  80                        "evaluations."))
  81     key_flags.append("epochs_between_evals")
  82 
  83   if stop_threshold:
  84     flags.DEFINE_float(
  85         name="stop_threshold", short_name="st",
  86         default=None,
  87         help=help_wrap("If passed, training will stop at the earlier of "
  88                        "train_epochs and when the evaluation metric is  "
  89                        "greater than or equal to stop_threshold."))
  90 
  91   if batch_size:
  92     flags.DEFINE_integer(
  93         name="batch_size", short_name="bs", default=32,
  94         help=help_wrap("Batch size for training and evaluation. When using "
  95                        "multiple gpus, this is the global batch size for "
  96                        "all devices. For example, if the batch size is 32 "
  97                        "and there are 4 GPUs, each GPU will get 8 examples on "
  98                        "each step."))
  99     key_flags.append("batch_size")
 100 
 101   if num_gpu:
 102     flags.DEFINE_integer(
 103         name="num_gpus", short_name="ng",
 104         default=1,
 105         help=help_wrap(
 106             "How many GPUs to use at each worker with the "
 107             "DistributionStrategies API. The default is 1."))
 108 
 109   if run_eagerly:
 110     flags.DEFINE_boolean(
 111         name="run_eagerly", default=False,
 112         help="Run the model op by op without building a model function.")
 113 
 114   if hooks:
 115     flags.DEFINE_list(
 116         name="hooks", short_name="hk", default="LoggingTensorHook",
 117         help=help_wrap(
 118             u"A list of (case insensitive) strings to specify the names of "
 119             u"training hooks. Example: `--hooks ProfilerHook,"
 120             u"ExamplesPerSecondHook`\n See hooks_helper "
 121             u"for details.")
 122     )
 123     key_flags.append("hooks")
 124 
 125   if export_dir:
 126     flags.DEFINE_string(
 127         name="export_dir", short_name="ed", default=None,
 128         help=help_wrap("If set, a SavedModel serialization of the model will "
 129                        "be exported to this directory at the end of training. "
 130                        "See the README for more details and relevant links.")
 131     )
 132     key_flags.append("export_dir")
 133 
 134   if distribution_strategy:
 135     flags.DEFINE_string(
 136         name="distribution_strategy", short_name="ds", default="mirrored",
 137         help=help_wrap("The Distribution Strategy to use for training. "
 138                        "Accepted values are 'off', 'one_device', "
 139                        "'mirrored', 'parameter_server', 'collective', "
 140                        "case insensitive. 'off' means not to use "
 141                        "Distribution Strategy; 'default' means to choose "
 142                        "from `MirroredStrategy` or `OneDeviceStrategy` "
 143                        "according to the number of GPUs.")
 144     )
 145 
 146 
 147   return key_flags

 150 def get_num_gpus(flags_obj):
 151   """Treat num_gpus=-1 as 'use all'."""
 152   if flags_obj.num_gpus != -1:
 153     return flags_obj.num_gpus
