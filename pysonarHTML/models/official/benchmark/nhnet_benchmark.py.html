  40 class NHNetBenchmark(perfzero_benchmark.PerfZeroBenchmark):
  41   """Base benchmark class for NHNet."""
  42 
  43   def __init__(self, output_dir=None, default_flags=None, tpu=None, **kwargs):
  44     self.default_flags = default_flags or {}
  45     flag_methods = trainer.define_flags()
  46     super(NHNetBenchmark, self).__init__(
  47         output_dir=output_dir,
  48         default_flags=default_flags,
  49         flag_methods=flag_methods,
  50         tpu=tpu,
  51         **kwargs)
  52 
  53   def _report_benchmark(self,
  54                         stats,
  55                         wall_time_sec,
  56                         max_value=None,
  57                         min_value=None):
  58     """Report benchmark results by writing to local protobuf file.
  59 
  60     Args:
  61       stats: dict returned from keras models with known entries.
  62       wall_time_sec: the during of the benchmark execution in seconds
  63       max_value: highest passing level.
  64       min_value: lowest passing level.
  65     """
  66 
  67     metrics = []
  68     metrics.append({
  69         'name': 'training_loss',
  70         'value': stats['training_loss'],
  71         'min_value': min_value,
  72         'max_value': max_value
  73     })
  74     # These metrics are placeholders to avoid PerfZero failure.
  75     metrics.append({
  76         'name': 'exp_per_second',
  77         'value': 0.0,
  78     })
  79     metrics.append({
  80         'name': 'startup_time',
  81         'value': 9999.,
  82     })
  83     flags_str = flags_core.get_nondefault_flags_as_str()
  84     self.report_benchmark(
  85         iters=-1,
  86         wall_time=wall_time_sec,
  87         metrics=metrics,
  88         extras={'flags': flags_str})

  91 class NHNetAccuracyBenchmark(NHNetBenchmark):
  92   """Benchmark accuracy tests for NHNet."""
  93 
  94   def __init__(self,
  95                output_dir: Optional[str] = None,
  96                tpu: Optional[str] = None,
  97                **kwargs):
  98     default_flags = dict(
  99         mode='train',
 100         train_file_pattern=NHNET_DATA,
 101         train_batch_size=1024,
 102         model_type='nhnet',
 103         len_title=15,
 104         len_passage=200,
 105         num_encoder_layers=12,
 106         num_decoder_layers=12,
 107         num_nhnet_articles=5,
 108         steps_per_loop=1000,
 109         params_override='init_from_bert2bert=false')
 110     super(NHNetAccuracyBenchmark, self).__init__(
 111         output_dir=output_dir, default_flags=default_flags, tpu=tpu, **kwargs)
 112 
 113   @benchmark_wrappers.enable_runtime_flags
 114   def _run_and_report_benchmark(self, max_value=MAX_LOSS, min_value=MIN_LOSS):
 115     """Runs and reports the benchmark given the provided configuration."""
 116     start_time_sec = time.time()
 117     stats = trainer.run()
 118     wall_time_sec = time.time() - start_time_sec
 119     self._report_benchmark(
 120         stats, wall_time_sec, max_value=max_value, min_value=min_value)
 121 
 122   @owner_utils.Owner('tf-model-garden')
 123   def benchmark_accuracy_4x4_tpu_f32_50k_steps(self):
 124     """Test bert pretraining with 4x4 TPU for 50k steps."""
 125     # This is used for accuracy test.
 126     self._setup()
 127     FLAGS.train_steps = 50000
 128     FLAGS.checkpoint_interval = FLAGS.train_steps
 129     FLAGS.distribution_strategy = 'tpu'
 130     FLAGS.init_checkpoint = PRETRAINED_CHECKPOINT_PATH
 131     FLAGS.model_dir = self._get_model_dir(
 132         'benchmark_accuracy_4x4_tpu_bf32_50k_steps')
 133     self._run_and_report_benchmark()
 134
