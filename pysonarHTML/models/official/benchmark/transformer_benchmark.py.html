  38 class TransformerBenchmark(PerfZeroBenchmark):
  39   """Methods common to executing transformer w/keras tests.
  40 
  41      Code under test for the Transformer Keras models report the same data and
  42      require the same FLAG setup.
  43   """
  44 
  45   def __init__(self, output_dir=None, default_flags=None, root_data_dir=None,
  46                flag_methods=None, tpu=None):
  47     root_data_dir = root_data_dir if root_data_dir else ''
  48 
  49     self.train_data_dir = os.path.join(root_data_dir,
  50                                        TRANSFORMER_EN2DE_DATA_DIR_NAME)
  51 
  52     self.vocab_file = os.path.join(root_data_dir,
  53                                    TRANSFORMER_EN2DE_DATA_DIR_NAME,
  54                                    'vocab.ende.32768')
  55 
  56     self.bleu_source = os.path.join(root_data_dir,
  57                                     EN2DE_2014_BLEU_DATA_DIR_NAME,
  58                                     'newstest2014.en')
  59 
  60     self.bleu_ref = os.path.join(root_data_dir,
  61                                  EN2DE_2014_BLEU_DATA_DIR_NAME,
  62                                  'newstest2014.de')
  63 
  64     if default_flags is None:
  65       default_flags = {}
  66     default_flags['data_dir'] = self.train_data_dir
  67     default_flags['vocab_file'] = self.vocab_file
  68 
  69     super(TransformerBenchmark, self).__init__(
  70         output_dir=output_dir,
  71         default_flags=default_flags,
  72         flag_methods=flag_methods,
  73         tpu=tpu)

  75   @benchmark_wrappers.enable_runtime_flags
  76   def _run_and_report_benchmark(self,
  77                                 bleu_max=None,
  78                                 bleu_min=None,
  79                                 log_steps=None,
  80                                 total_batch_size=None,
  81                                 warmup=1):
  82     """Report benchmark results by writing to local protobuf file.
  83 
  84     Args:
  85       bleu_max: highest passing level for bleu score.
  86       bleu_min: lowest passing level for bleu score.
  87       log_steps: How often the log was created for stats['step_timestamp_log'].
  88       total_batch_size: Global batch-size.
  89       warmup: number of entries in stats['step_timestamp_log'] to ignore.
  90     """
  91     start_time_sec = time.time()
  92     task = transformer_main.TransformerTask(FLAGS)
  93     stats = task.train()
  94     wall_time_sec = time.time() - start_time_sec
  95 
  96     metrics = []
  97     if 'bleu_uncased' in stats:
  98       if 'bleu_uncased_history' in stats:
  99         bleu_uncased_best = max(stats['bleu_uncased_history'],
 100                                 key=lambda x: x[1])
 101         metrics.append({'name': 'bleu_uncased',
 102                         'value': bleu_uncased_best[1],
 103                         'min_value': bleu_min,
 104                         'max_value': bleu_max})
 105         metrics.append({'name': 'bleu_best_score_iteration',
 106                         'value': bleu_uncased_best[0]})
 107         metrics.append({'name': 'bleu_uncased_last',
 108                         'value': stats['bleu_uncased']})
 109       else:
 110         metrics.append({'name': 'bleu_uncased',
 111                         'value': stats['bleu_uncased'],
 112                         'min_value': bleu_min,
 113                         'max_value': bleu_max})
 114 
 115     if (warmup and 'step_timestamp_log' in stats and
 116         len(stats['step_timestamp_log']) > warmup + 1):
 117       # first entry in the time_log is start of step 1. The rest of the
 118       # entries are the end of each step recorded
 119       time_log = stats['step_timestamp_log']
 120       elapsed = time_log[-1].timestamp - time_log[warmup].timestamp
 121       num_examples = (
 122           total_batch_size * log_steps * (len(time_log) - warmup - 1))
 123       examples_per_sec = num_examples / elapsed
 124       metrics.append({'name': 'exp_per_second',
 125                       'value': examples_per_sec})
 126 
 127     if 'avg_exp_per_second' in stats:
 128       metrics.append({'name': 'avg_exp_per_second',
 129                       'value': stats['avg_exp_per_second']})
 130 
 131     if 'step_timestamp_log' in stats:
 132       time_log = stats['step_timestamp_log']
 133       metrics.append({'name': 'startup_time',
 134                       'value': time_log[0].timestamp - start_time_sec})
 135 
 136     flags_str = flags_core.get_nondefault_flags_as_str()
 137     self.report_benchmark(iters=-1, wall_time=wall_time_sec, metrics=metrics,
 138                           extras={'flags': flags_str})

 161     """Benchmark 1 gpu.

 163       The paper uses 8 GPUs and a much larger effective batch size, this is will

 172     FLAGS['bleu_ref'].value = self.bleu_ref
 173     FLAGS.param_set = 'base'
 174     FLAGS.batch_size = 2048
 175     FLAGS.train_steps = 1000
 176     FLAGS.steps_between_evals = 500
 177     FLAGS.model_dir = self._get_model_dir('benchmark_1_gpu')
 178     # These bleu scores are based on test runs after at this limited
 179     # number of steps and batch size after verifying SOTA at 8xV100s.
 180     self._run_and_report_benchmark(total_batch_size=FLAGS.batch_size,
 181                                    log_steps=FLAGS.log_steps,
 182                                    bleu_min=25.3,
 183                                    bleu_max=26)
