 189   @benchmark_wrappers.enable_runtime_flags
 190   def _run_and_report_benchmark(self):
 191     start_time_sec = time.time()
 192     stats = resnet_cifar_main.run(FLAGS)
 193     wall_time_sec = time.time() - start_time_sec
 194 
 195     super(Resnet56KerasAccuracy, self)._report_benchmark(
 196         stats,
 197         wall_time_sec,
 198         top_1_min=MIN_TOP_1_ACCURACY,
 199         top_1_max=MAX_TOP_1_ACCURACY,
 200         total_batch_size=FLAGS.batch_size,
 201         log_steps=100)
