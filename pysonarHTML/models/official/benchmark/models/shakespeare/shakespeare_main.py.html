  88 def get_dataset(path_to_file, batch_size=None, seq_length=SEQ_LENGTH):
  89   """Creates a dataset from a given text file.
  90 
  91   Args:
  92     path_to_file: The path to the training data.
  93     batch_size: Batch size to use.
  94     seq_length: The length of the LSTM sequence.
  95 
  96   Returns:
  97     A tuple, consisting of the Dataset and the class to character mapping
  98     and character to class mapping.
  99   """
 100   with tf.io.gfile.GFile(path_to_file, 'rb') as train_data:
 101     text = train_data.read().decode(encoding='utf-8')
 102 
 103   # Create vocab
 104   vocab = sorted(set(text))
 105   char2idx = {u: i for i, u in enumerate(vocab)}
 106   idx2char = np.array(vocab)
 107 
 108   # Split text into sequence length + 1 chucks to create examples
 109   text_as_int = np.array([char2idx[c] for c in text])
 110   char_dataset = tf.data.Dataset.from_tensor_slices(text_as_int)
 111   sequences = char_dataset.batch(seq_length+1, drop_remainder=True)
 112 
 113   def split_input_target(chunk):
 114     input_text = chunk[:-1]
 115     target_text = chunk[1:]
 116     return input_text, tf.one_hot(target_text, len(vocab))
 117   dataset = sequences.map(split_input_target)
 118   dataset = dataset.shuffle(10000).repeat()
 119   dataset = dataset.batch(batch_size, drop_remainder=True)
 120 
 121   return dataset, idx2char, char2idx

 124 def build_model(vocab_size,
 125                 embedding_dim=EMBEDDING_DIM,
 126                 rnn_units=RNN_UNITS,
 127                 batch_size=None,
 128                 stateful=False,
 129                 use_cudnn=True):
 130   """Builds the Shakespeare model.
 131 
 132   Args:
 133     vocab_size: The number of character classes in the input.
 134     embedding_dim: The dimension of the embedding space for each class.
 135     rnn_units: The number of RNN units in the layer.
 136     batch_size: When predicting, the batch size of the predictions.
 137     stateful: If true, the LSTM is stateful.
 138 
 139   Returns:
 140     A Keras Model.
 141   """
 142   LSTM = functools.partial(tf.keras.layers.LSTM, implementation=2)
 143 
 144   # By indirecting the activation through a lambda layer, the logic to dispatch
 145   # to CuDNN in V2 doesn't trigger and we force the LSTM to run in non-CuDNN
 146   # mode.
 147   lstm_activation = ('tanh' if use_cudnn else
 148                      lambda x: tf.math.tanh(x))
 149 
 150   batch_shape = [batch_size if stateful else None, None]
 151   return tf.keras.Sequential([
 152       tf.keras.layers.Embedding(vocab_size, embedding_dim,
 153                                 batch_input_shape=batch_shape),
 154       LSTM(rnn_units,
 155            activation=lstm_activation,
 156            return_sequences=True,
 157            stateful=stateful,
 158            recurrent_initializer='glorot_uniform'),
 159       tf.keras.layers.Dense(vocab_size),
 160       tf.keras.layers.Softmax(dtype=tf.float32)])
 161

 163 def train_model(flags_obj, dataset, vocab_size, strategy, checkpoint_dir=None):
 164   """Trains a Shakespeare model.
 165 
 166   Args:
 167     flags_obj: An object containing parsed flag values.s
 168     dataset: the training data set.
 169     vocab_size: the number of unique character classes.
 170     strategy: distribution strategy to use.
 171     checkpoint_dir: if not None, the directory in which to make checkpoints.
 172 
 173   Returns:
 174     The training history and callbacks.
 175   """
 176   if flags_obj.train_steps:
 177     train_steps = flags_obj.train_steps
 178   else:
 179     train_steps = BATCHES_PER_EPOCH // flags_obj.batch_size
 180   strategy_scope = distribution_utils.get_strategy_scope(strategy)
 181 
 182   with strategy_scope:
 183     model = build_model(vocab_size=vocab_size, batch_size=flags_obj.batch_size,
 184                         use_cudnn=flags_obj.cudnn)
 185 
 186    # When keras_use_ctl is False, Model.fit() automatically applies
 187    # loss scaling so we don't need to create a LossScaleOptimizer.
 188     model.compile(
 189         optimizer=tf.keras.optimizers.Adam(),
 190         loss=tf.keras.losses.CategoricalCrossentropy(),
 191         metrics=[tf.keras.metrics.Recall(top_k=1, name='RecallAt1'),
 192                  tf.keras.metrics.Recall(top_k=5, name='RecallAt5')],
 193         run_eagerly=flags_obj.run_eagerly)
 194 
 195   callbacks = []
 196   if checkpoint_dir:
 197     checkpoint_prefix = os.path.join(checkpoint_dir, 'ckpt_{epoch}')
 198     checkpoint_callback = tf.keras.callbacks.ModelCheckpoint(
 199         filepath=checkpoint_prefix,
 200         save_weights_only=True)
 201     callbacks.append(checkpoint_callback)
 202   time_callback = keras_utils.TimeHistory(flags_obj.batch_size,
 203                                           flags_obj.log_steps)
 204   callbacks.append(time_callback)
 205   history = model.fit(dataset,
 206                       epochs=flags_obj.train_epochs,
 207                       steps_per_epoch=train_steps,
 208                       callbacks=callbacks,
 209                       verbose=2)
 210   return history, callbacks

 213 def make_prediction(checkpoint_dir, length, context, idx2char, char2idx):
 214   """Make predictions from a Shakespeare model.
 215 
 216   Args:
 217     checkpoint_dir: the directory from which to load checkpoints
 218     length: the total length of the generated text (including the context).
 219     context: the initial text with which the LSTM is primed.
 220     idx2char: the character class to character mapping.
 221     char2idx: the character to character class mapping.
 222 
 223   Returns:
 224     A generated string of text of the given length.
 225   """
 226   prediction_model = build_model(
 227       vocab_size=len(idx2char), batch_size=1, stateful=True)
 228   prediction_model.load_weights(tf.train.latest_checkpoint(checkpoint_dir))
 229   prediction_model.build(tf.TensorShape([1, None]))
 230 
 231   input_eval = [char2idx[s] for s in context]
 232   input_eval = tf.expand_dims(input_eval, 0)
 233 
 234   text_generated = []
 235 
 236   prediction_model.reset_states()
 237   for _ in range(length - len(context)):
 238     predictions = prediction_model(input_eval)
 239     predictions = tf.squeeze(predictions, 0)
 240 
 241     # We applied a softmax to the output of the model so that
 242     # tf.keras.metrics.Recall would work. We need logits for
 243     # tf.random.categorical, so we convert the probabilities back to log odds
 244     predictions = tf.math.log(predictions / (1 - predictions))
 245 
 246     random_output = tf.random.categorical(predictions, num_samples=1)
 247     selected_id = random_output[-1, 0].numpy()
 248     input_eval = tf.expand_dims([selected_id], 0)
 249     text_generated.append(idx2char[selected_id])
 250 
 251   return context + ''.join(text_generated)

 254 def run(flags_obj):
 255   """Run Shakespeare training and predict.
 256 
 257   Args:
 258     flags_obj: An object containing parsed flag values.
 259 
 260   Returns:
 261     Dictionary with status from the run.
 262   """
 263   if not flags_obj.training_data:
 264     raise ValueError(
 265         'Must set the path to a training data file. e.g download the following '
 266         'https://storage.googleapis.com/download.tensorflow.org/data/'
 267         'shakespeare.txt')
 268 
 269   if flags_obj.dtype == 'fp16':
 270     policy = tf.keras.mixed_precision.experimental.Policy(
 271         'mixed_float16',
 272         loss_scale=flags_core.get_loss_scale(flags_obj,
 273                                              default_for_fp16='dynamic'))
 274     tf.keras.mixed_precision.experimental.set_policy(policy)
 275 
 276   keras_utils.set_session_config(
 277       enable_xla=flags_obj.enable_xla)
 278 
 279   strategy = distribution_utils.get_distribution_strategy(
 280       distribution_strategy=flags_obj.distribution_strategy,
 281       num_gpus=flags_obj.num_gpus)
 282 
 283   dataset, idx2char, char2idx = get_dataset(flags_obj.training_data,
 284                                             batch_size=flags_obj.batch_size)
 285   stats = {}
 286   if flags_obj.train:
 287     history, callbacks = train_model(flags_obj, dataset,
 288                                      len(idx2char), strategy,
 289                                      checkpoint_dir=flags_obj.model_dir)
 290 
 291     stats['history'] = history.history
 292     stats['callbacks'] = callbacks
 293 
 294   if flags_obj.predict_context:
 295     if not flags_obj.model_dir:
 296       raise ValueError('Must set model_dir to get predictions.')
 297     print(make_prediction(flags_obj.model_dir,
 298                           flags_obj.predict_length,
 299                           flags_obj.predict_context,
 300                           idx2char,
 301                           char2idx))
 302 
 303   return stats
