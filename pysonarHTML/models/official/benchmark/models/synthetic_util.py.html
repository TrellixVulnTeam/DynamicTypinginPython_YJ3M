  28 # The `SyntheticDataset` is a temporary solution for generating synthetic data
  29 # directly on devices. It is only useful for Keras with Distribution
  30 # Strategies. We will have better support in `tf.data` or Distribution Strategy
  31 # later.
  32 class SyntheticDataset(object):
  33   """A dataset that generates synthetic data on each device."""
  34 
  35   def __init__(self, dataset, split_by=1):
  36     # dataset.take(1) doesn't have GPU kernel.
  37     with tf.device('device:CPU:0'):
  38       tensor = tf.data.experimental.get_single_element(dataset.take(1))
  39     flat_tensor = tf.nest.flatten(tensor)
  40     variable_data = []
  41     initializers = []
  42     for t in flat_tensor:
  43       rebatched_t = tf.split(t, num_or_size_splits=split_by, axis=0)[0]
  44       assert rebatched_t.shape.is_fully_defined(), rebatched_t.shape
  45       v = tf.compat.v1.get_local_variable(self._random_name(),
  46                                           initializer=rebatched_t)
  47       variable_data.append(v)
  48       initializers.append(v.initializer)
  49     input_data = tf.nest.pack_sequence_as(tensor, variable_data)
  50     self._iterator = SyntheticIterator(input_data, initializers)
  51 
  52   def _random_name(self, size=10, chars=string.ascii_uppercase + string.digits):
  53     return ''.join(random.choice(chars) for _ in range(size))
  54 
  55   def __iter__(self):
  56     return self._iterator
  57 
  58   def make_one_shot_iterator(self):
  59     return self._iterator
  60 
  61   def make_initializable_iterator(self):
  62     return self._iterator
  63

  65 class SyntheticIterator(object):
  66   """A dataset that generates synthetic data on each device."""
  67 
  68   def __init__(self, input_data, initializers):
  69     self._input_data = input_data
  70     self._initializers = initializers
  71 
  72   def get_next(self):
  73     return self._input_data
  74 
  75   def next(self):
  76     return self.__next__()
  77 
  78   def __next__(self):
  79     try:
  80       return self.get_next()
  81     except tf.errors.OutOfRangeError:
  82       raise StopIteration
  83 
  84   def initialize(self):
  85     if tf.executing_eagerly():
  86       return tf.no_op()
  87     else:
  88       return self._initializers

  91 def _monkey_patch_dataset_method(strategy):
  92   """Monkey-patch `strategy`'s `make_dataset_iterator` method."""
  93   def make_dataset(self, dataset):
  94     logging.info('Using pure synthetic data.')
  95     with self.scope():
  96       if self.extended._global_batch_size:  # pylint: disable=protected-access
  97         return SyntheticDataset(dataset, self.num_replicas_in_sync)
  98       else:
  99         return SyntheticDataset(dataset)
 100 
 101   def make_iterator(self, dataset):
 102     dist_dataset = make_dataset(self, dataset)
 103     return iter(dist_dataset)
 104 
 105   strategy.orig_make_dataset_iterator = strategy.make_dataset_iterator
 106   strategy.make_dataset_iterator = make_iterator
 107   strategy.orig_distribute_dataset = strategy.experimental_distribute_dataset
 108   strategy.experimental_distribute_dataset = make_dataset
 109 
 110 
 111 def _undo_monkey_patch_dataset_method(strategy):
 112   if hasattr(strategy, 'orig_make_dataset_iterator'):
 113     strategy.make_dataset_iterator = strategy.orig_make_dataset_iterator
 114   if hasattr(strategy, 'orig_distribute_dataset'):
 115     strategy.make_dataset_iterator = strategy.orig_distribute_dataset
 116 
 117 
 118 def set_up_synthetic_data():
 119   _monkey_patch_dataset_method(tf.distribute.OneDeviceStrategy)
 120   _monkey_patch_dataset_method(tf.distribute.MirroredStrategy)
 121   _monkey_patch_dataset_method(
 122       tf.distribute.experimental.MultiWorkerMirroredStrategy)
 123 
 124 
 125 def undo_set_up_synthetic_data():
 126   _undo_monkey_patch_dataset_method(tf.distribute.OneDeviceStrategy)
 127   _undo_monkey_patch_dataset_method(tf.distribute.MirroredStrategy)
 128   _undo_monkey_patch_dataset_method(
 129       tf.distribute.experimental.MultiWorkerMirroredStrategy)
