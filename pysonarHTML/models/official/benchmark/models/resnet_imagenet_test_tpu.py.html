  15 """Test the keras ResNet model with ImageNet data on TPU."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from absl.testing import parameterized
  22 import tensorflow as tf
  23 from official.benchmark.models import resnet_imagenet_main
  24 from official.utils.testing import integration
  25 from official.vision.image_classification.resnet import imagenet_preprocessing
  26 
  27 
  28 class KerasImagenetTest(tf.test.TestCase, parameterized.TestCase):
  29   """Unit tests for Keras Models with ImageNet."""
  30 
  31   _extra_flags_dict = {
  32       "resnet": [
  33           "-batch_size", "4",
  34           "-train_steps", "1",
  35           "-use_synthetic_data", "true"
  36           "-model", "resnet50_v1.5",
  37           "-optimizer", "resnet50_default",
  38       ],
  39       "resnet_polynomial_decay": [
  40           "-batch_size", "4",
  41           "-train_steps", "1",
  42           "-use_synthetic_data", "true",
  43           "-model", "resnet50_v1.5",
  44           "-optimizer", "resnet50_default",
  45           "-pruning_method", "polynomial_decay",
  46       ],
  47   }
  48   _tempdir = None
  49 
  50   @classmethod
  51   def setUpClass(cls):  # pylint: disable=invalid-name
  52     super(KerasImagenetTest, cls).setUpClass()
  53     resnet_imagenet_main.define_imagenet_keras_flags()
  54 
  55   def setUp(self):
  56     super(KerasImagenetTest, self).setUp()
  57     imagenet_preprocessing.NUM_IMAGES["validation"] = 4
  58     self.policy = \
  59         tf.keras.mixed_precision.experimental.global_policy()
  60 
  61   def tearDown(self):
  62     super(KerasImagenetTest, self).tearDown()
  63     tf.io.gfile.rmtree(self.get_temp_dir())
  64     tf.keras.mixed_precision.experimental.set_policy(self.policy)
  65 
  66   @parameterized.parameters([
  67       "resnet",
  68       # "resnet_polynomial_decay"  b/151854314
  69   ])
  70   def test_end_to_end_tpu(self, flags_key):
  71     """Test Keras model with TPU distribution strategy."""
  72 
  73     extra_flags = [
  74         "-distribution_strategy", "tpu",
  75         "-data_format", "channels_last",
  76         "-enable_checkpoint_and_export", "1",
  77     ]
  78     extra_flags = extra_flags + self._extra_flags_dict[flags_key]
  79 
  80     integration.run_synthetic(
  81         main=resnet_imagenet_main.run,
  82         tmp_root=self.get_temp_dir(),
  83         extra_flags=extra_flags
  84     )
  85 
  86   @parameterized.parameters(["resnet"])
  87   def test_end_to_end_tpu_bf16(self, flags_key):
  88     """Test Keras model with TPU and bfloat16 activation."""
  89 
  90     extra_flags = [
  91         "-distribution_strategy", "tpu",
  92         "-data_format", "channels_last",
  93         "-dtype", "bf16",
  94     ]
  95     extra_flags = extra_flags + self._extra_flags_dict[flags_key]
  96 
  97     integration.run_synthetic(
  98         main=resnet_imagenet_main.run,
  99         tmp_root=self.get_temp_dir(),
 100         extra_flags=extra_flags
 101     )
 102
