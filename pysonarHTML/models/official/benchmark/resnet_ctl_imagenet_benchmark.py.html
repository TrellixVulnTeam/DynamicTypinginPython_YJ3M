  38 class CtlBenchmark(PerfZeroBenchmark):
  39   """Base benchmark class with methods to simplify testing."""
  40 
  41   def __init__(self, output_dir=None, default_flags=None, flag_methods=None):
  42     self.default_flags = default_flags or {}
  43     self.flag_methods = flag_methods or {}
  44     super(CtlBenchmark, self).__init__(
  45         output_dir=output_dir,
  46         default_flags=self.default_flags,
  47         flag_methods=self.flag_methods)
  48 
  49   def _report_benchmark(self,
  50                         stats,
  51                         wall_time_sec,
  52                         top_1_max=None,
  53                         top_1_min=None,
  54                         total_batch_size=None,
  55                         log_steps=None,
  56                         warmup=1,
  57                         start_time_sec=None):
  58     """Report benchmark results by writing to local protobuf file.
  59 
  60     Args:
  61       stats: dict returned from keras models with known entries.
  62       wall_time_sec: the during of the benchmark execution in seconds
  63       top_1_max: highest passing level for top_1 accuracy.
  64       top_1_min: lowest passing level for top_1 accuracy.
  65       total_batch_size: Global batch-size.
  66       log_steps: How often the log was created for stats['step_timestamp_log'].
  67       warmup: number of entries in stats['step_timestamp_log'] to ignore.
  68       start_time_sec: the start time of the program in seconds since epoch.
  69     """
  70 
  71     metrics = []
  72     if 'eval_acc' in stats:
  73       metrics.append({
  74           'name': 'accuracy_top_1',
  75           'value': stats['eval_acc'],
  76           'min_value': top_1_min,
  77           'max_value': top_1_max
  78       })
  79       metrics.append({'name': 'eval_loss', 'value': stats['eval_loss']})
  80 
  81       metrics.append({
  82           'name': 'top_1_train_accuracy',
  83           'value': stats['train_acc']
  84       })
  85       metrics.append({'name': 'train_loss', 'value': stats['train_loss']})
  86 
  87     if (warmup and 'step_timestamp_log' in stats and
  88         len(stats['step_timestamp_log']) > warmup + 1):
  89       # first entry in the time_log is start of step 0. The rest of the
  90       # entries are the end of each step recorded
  91       time_log = stats['step_timestamp_log']
  92       steps_elapsed = time_log[-1].batch_index - time_log[warmup].batch_index
  93       time_elapsed = time_log[-1].timestamp - time_log[warmup].timestamp
  94       examples_per_sec = total_batch_size * (steps_elapsed / time_elapsed)
  95       metrics.append({'name': 'exp_per_second', 'value': examples_per_sec})
  96 
  97     if 'avg_exp_per_second' in stats:
  98       metrics.append({
  99           'name': 'avg_exp_per_second',
 100           'value': stats['avg_exp_per_second']
 101       })
 102 
 103     if start_time_sec and 'step_timestamp_log' in stats:
 104       time_log = stats['step_timestamp_log']
 105       # time_log[0] is recorded at the beginning of the first step.
 106       startup_time = time_log[0].timestamp - start_time_sec
 107       metrics.append({'name': 'startup_time', 'value': startup_time})
 108 
 109     flags_str = flags_core.get_nondefault_flags_as_str()
 110     self.report_benchmark(
 111         iters=-1,
 112         wall_time=wall_time_sec,
 113         metrics=metrics,
 114         extras={'flags': flags_str})

 117 class Resnet50CtlAccuracy(CtlBenchmark):
 118   """Benchmark accuracy tests for ResNet50 in CTL."""

 120   def __init__(self, output_dir=None, root_data_dir=None, **kwargs):
 121     """A benchmark class.
 122 
 123     Args:
 124       output_dir: directory where to output e.g. log files
 125       root_data_dir: directory under which to look for dataset
 126       **kwargs: arbitrary named arguments. This is needed to make the
 127         constructor forward compatible in case PerfZero provides more named
 128         arguments before updating the constructor.
 129     """
 130 
 131     flag_methods = [common.define_keras_flags]
 132 
 133     self.data_dir = os.path.join(root_data_dir, 'imagenet')
 134     super(Resnet50CtlAccuracy, self).__init__(
 135         output_dir=output_dir, flag_methods=flag_methods)

 174   @benchmark_wrappers.enable_runtime_flags
 175   def _run_and_report_benchmark(self):
 176     start_time_sec = time.time()
 177     stats = resnet_ctl_imagenet_main.run(flags.FLAGS)
 178     wall_time_sec = time.time() - start_time_sec
 179 
 180     super(Resnet50CtlAccuracy, self)._report_benchmark(
 181         stats,
 182         wall_time_sec,
 183         top_1_min=MIN_TOP_1_ACCURACY,
 184         top_1_max=MAX_TOP_1_ACCURACY,
 185         total_batch_size=FLAGS.batch_size,
 186         log_steps=100,
 187         start_time_sec=start_time_sec)

 201   @benchmark_wrappers.enable_runtime_flags
 202   def _run_and_report_benchmark(self):
 203     start_time_sec = time.time()
 204     stats = resnet_ctl_imagenet_main.run(FLAGS)
 205     wall_time_sec = time.time() - start_time_sec
 206 
 207     # Warmup means the number of logged step time entries that are excluded in
 208     # performance report. Default to exclude 1 FLAGS.log_steps time.
 209     super(Resnet50CtlBenchmarkBase, self)._report_benchmark(
 210         stats,
 211         wall_time_sec,
 212         total_batch_size=FLAGS.batch_size,
 213         log_steps=FLAGS.log_steps,
 214         warmup=1,
 215         start_time_sec=start_time_sec)

 377     FLAGS.enable_tensorboard = False
 378     FLAGS.distribution_strategy = 'tpu'
 379     FLAGS.report_accuracy_metrics = False
 380     FLAGS.log_steps = 50
 381     FLAGS.single_l2_loss_op = True
 382     FLAGS.use_tf_function = True
 383     FLAGS.enable_checkpoint_and_export = False
