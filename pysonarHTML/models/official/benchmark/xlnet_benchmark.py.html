  67 class XLNetClassifyAccuracy(XLNetBenchmarkBase):
  68   """Short accuracy test for XLNet classifier model.
  69 
  70   Tests XLNet classification task model accuracy. The naming
  71   convention of below test cases follow
  72   `benchmark_(number of gpus)_gpu_(dataset type)` format.
  73   """
  74 
  75   def __init__(self, output_dir=None, tpu=None, **kwargs):
  76     self.train_data_path = CLASSIFIER_TRAIN_DATA_PATH
  77     self.eval_data_path = CLASSIFIER_EVAL_DATA_PATH
  78     self.pretrained_checkpoint_path = PRETRAINED_CHECKPOINT_PATH
  79 
  80     super(XLNetClassifyAccuracy, self).__init__(output_dir=output_dir, tpu=tpu)
  81 
  82   @benchmark_wrappers.enable_runtime_flags
  83   def _run_and_report_benchmark(self,
  84                                 training_summary_path,
  85                                 min_accuracy=0.95,
  86                                 max_accuracy=0.97):
  87     """Starts XLNet accuracy benchmark test."""
  88 
  89     start_time_sec = time.time()
  90     self._run_xlnet_classifier()
  91     wall_time_sec = time.time() - start_time_sec
  92 
  93     with tf.io.gfile.GFile(training_summary_path, 'rb') as reader:
  94       summary = json.loads(reader.read().decode('utf-8'))
  95 
  96     super(XLNetClassifyAccuracy, self)._report_benchmark(
  97         stats=summary,
  98         wall_time_sec=wall_time_sec,
  99         min_accuracy=min_accuracy,
 100         max_accuracy=max_accuracy)
 101 
 102   def _setup(self):
 103     super(XLNetClassifyAccuracy, self)._setup()
 104     FLAGS.test_data_size = 25024
 105     FLAGS.train_batch_size = 16
 106     FLAGS.seq_len = 512
 107     FLAGS.mem_len = 0
 108     FLAGS.n_layer = 24
 109     FLAGS.d_model = 1024
 110     FLAGS.d_embed = 1024
 111     FLAGS.n_head = 16
 112     FLAGS.d_head = 64
 113     FLAGS.d_inner = 4096
 114     FLAGS.untie_r = True
 115     FLAGS.n_class = 2
 116     FLAGS.ff_activation = 'gelu'
 117     FLAGS.strategy_type = 'mirror'
 118     FLAGS.learning_rate = 2e-5
 119     FLAGS.train_steps = 4000
 120     FLAGS.warmup_steps = 500
 121     FLAGS.iterations = 200
 122     FLAGS.bi_data = False
 123     FLAGS.init_checkpoint = self.pretrained_checkpoint_path
 124     FLAGS.train_tfrecord_path = self.train_data_path
 125     FLAGS.test_tfrecord_path = self.eval_data_path
 126 
 127   @owner_utils.Owner('tf-model-garden')
 128   def benchmark_8_gpu_imdb(self):
 129     """Run XLNet model accuracy test with 8 GPUs."""
 130     self._setup()
 131     FLAGS.model_dir = self._get_model_dir('benchmark_8_gpu_imdb')
 132     # Sets timer_callback to None as we do not use it now.
 133     self.timer_callback = None
 134 
 135     summary_path = os.path.join(FLAGS.model_dir,
 136                                 'summaries/training_summary.txt')
 137     self._run_and_report_benchmark(summary_path)
 138 
 139   @owner_utils.Owner('tf-model-garden')
 140   def benchmark_2x2_tpu_imdb(self):
 141     """Run XLNet model accuracy test on 2x2 tpu."""
 142     self._setup()
 143     FLAGS.strategy_type = 'tpu'
 144     FLAGS.model_dir = self._get_model_dir('benchmark_2x2_tpu_imdb')
 145     # Sets timer_callback to None as we do not use it now.
 146     self.timer_callback = None
 147 
 148     summary_path = os.path.join(FLAGS.model_dir,
 149                                 'summaries/training_summary.txt')
 150     self._run_and_report_benchmark(summary_path)

 153 class XLNetSquadAccuracy(XLNetBenchmarkBase):
 154   """Short accuracy test for XLNet squad model.
 155 
 156   Tests XLNet squad task model accuracy. The naming
 157   convention of below test cases follow
 158   `benchmark_(number of gpus)_gpu_(dataset type)` format.
 159   """
 160 
 161   def __init__(self, output_dir=None, tpu=None, **kwargs):
 162     self.train_data_path = SQUAD_DATA_PATH
 163     self.predict_file = os.path.join(SQUAD_DATA_PATH, "dev-v2.0.json")
 164     self.test_data_path = os.path.join(SQUAD_DATA_PATH, "12048.eval.tf_record")
 165     self.spiece_model_file = os.path.join(SQUAD_DATA_PATH, "spiece.cased.model")
 166     self.pretrained_checkpoint_path = PRETRAINED_CHECKPOINT_PATH
 167 
 168     super(XLNetSquadAccuracy, self).__init__(output_dir=output_dir, tpu=tpu)
 169 
 170   @benchmark_wrappers.enable_runtime_flags
 171   def _run_and_report_benchmark(self,
 172                                 training_summary_path,
 173                                 min_accuracy=87.0,
 174                                 max_accuracy=89.0):
 175     """Starts XLNet accuracy benchmark test."""
 176 
 177     start_time_sec = time.time()
 178     self._run_xlnet_squad()
 179     wall_time_sec = time.time() - start_time_sec
 180 
 181     with tf.io.gfile.GFile(training_summary_path, 'rb') as reader:
 182       summary = json.loads(reader.read().decode('utf-8'))
 183 
 184     super(XLNetSquadAccuracy, self)._report_benchmark(
 185         stats=summary,
 186         wall_time_sec=wall_time_sec,
 187         min_accuracy=min_accuracy,
 188         max_accuracy=max_accuracy)
 189 
 190   def _setup(self):
 191     super(XLNetSquadAccuracy, self)._setup()
 192     FLAGS.train_batch_size = 16
 193     FLAGS.seq_len = 512
 194     FLAGS.mem_len = 0
 195     FLAGS.n_layer = 24
 196     FLAGS.d_model = 1024
 197     FLAGS.d_embed = 1024
 198     FLAGS.n_head = 16
 199     FLAGS.d_head = 64
 200     FLAGS.d_inner = 4096
 201     FLAGS.untie_r = True
 202     FLAGS.ff_activation = 'gelu'
 203     FLAGS.strategy_type = 'mirror'
 204     FLAGS.learning_rate = 3e-5
 205     FLAGS.train_steps = 8000
 206     FLAGS.warmup_steps = 1000
 207     FLAGS.iterations = 1000
 208     FLAGS.bi_data = False
 209     FLAGS.init_checkpoint = self.pretrained_checkpoint_path
 210     FLAGS.train_tfrecord_path = self.train_data_path
 211     FLAGS.test_tfrecord_path = self.test_data_path
 212     FLAGS.spiece_model_file = self.spiece_model_file
 213     FLAGS.predict_file = self.predict_file
 214     FLAGS.adam_epsilon = 1e-6
 215     FLAGS.lr_layer_decay_rate = 0.75
