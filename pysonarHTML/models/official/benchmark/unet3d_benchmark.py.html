  46                output_dir: Optional[str] = None,

  49     """A benchmark class.

  51     Args:

  54       **kwargs: arbitrary named arguments. This is needed to make the

  59     flag_methods = [unet_training_lib.define_unet3d_flags]

  61     # UNet3D model in Keras."""
  62     self.training_file_pattern = UNET_TRAINING_FILES

  70   def _set_benchmark_parameters(self, experiment_name):
  71     """Overrides training parameters for benchmark tests."""
  72     FLAGS.model_dir = self._get_model_dir(experiment_name)
  73     FLAGS.mode = 'train'
  74     FLAGS.training_file_pattern = self.training_file_pattern
  75     FLAGS.eval_file_pattern = self.eval_file_pattern
  76     FLAGS.config_file = self.config_file
  77     FLAGS.lr_init_value = 0.00005
  78     FLAGS.lr_decay_rate = 0.5
  79     FLAGS.epochs = 3
  80 
  81   @benchmark_wrappers.enable_runtime_flags
  82   def _run_and_report_benchmark(self,
  83                                 experiment_name: str,
  84                                 min_accuracy: float = UNET3D_MIN_ACCURACY,
  85                                 max_accuracy: float = UNET3D_MAX_ACCURACY,
  86                                 distribution_strategy: str = 'tpu',
  87                                 epochs: int = 10,
  88                                 steps: int = 0,
  89                                 epochs_between_evals: int = 1,
  90                                 dtype: str = 'float32',
  91                                 enable_xla: bool = False,
  92                                 run_eagerly: bool = False):
  93     """Runs and reports the benchmark given the provided configuration."""
  94     params = unet_training_lib.extract_params(FLAGS)
  95     strategy = unet_training_lib.create_distribution_strategy(params)
  96     if params.use_bfloat16:
  97       policy = tf.keras.mixed_precision.experimental.Policy('mixed_bfloat16')
  98       tf.keras.mixed_precision.experimental.set_policy(policy)
  99 
 100     stats = {}
 101     start_time_sec = time.time()
 102     with strategy.scope():
 103       unet_model = unet_model_lib.build_unet_model(params)
 104       history = unet_training_lib.train(
 105           params, strategy, unet_model,
 106           functools.partial(unet_training_lib.get_train_dataset, params),
 107           functools.partial(unet_training_lib.get_eval_dataset, params))
 108 
 109       stats['accuracy_top_1'] = history.history['val_metric_accuracy'][-1]
 110       stats['training_accuracy_top_1'] = history.history['metric_accuracy'][-1]
 111     wall_time_sec = time.time() - start_time_sec
 112 
 113     super(Unet3DAccuracyBenchmark, self)._report_benchmark(
 114         stats,
 115         wall_time_sec,
 116         top_1_min=min_accuracy,
 117         top_1_max=max_accuracy,
 118         total_batch_size=params.train_batch_size)
 119

 121     return os.path.join(self.output_dir, folder_name)
