  15 """Utils to set Owner annotations on benchmarks.
  16 
  17 @owner_utils.Owner('owner_team/user') can be set either at the benchmark class
  18 level / benchmark method level or both.
  19 
  20 Runner frameworks can use owner_utils.GetOwner(benchmark_method) to get the
  21 actual owner. Python inheritance for the owner attribute is respected.  (E.g
  22 method level owner takes precedence over class level).
  23 
  24 See owner_utils_test for associated tests and more examples.
  25 
  26 The decorator can be applied both at the method level and at the class level.
  27 
  28 Simple example:
  29 ===============
  30 
  31 class MLBenchmark:
  32 
  33   @Owner('example_id')
  34   def benchmark_method_1_gpu(self):
  35     return True
  36 """
  37 
  38 
  39 def Owner(owner_name):
  40   """Sets the owner attribute on a decorated method or class."""
  41 
  42   def _Wrapper(func_or_class):
  43     """Sets the benchmark owner attribute."""
  44     func_or_class.__benchmark__owner__ = owner_name
  45     return func_or_class
  46 
  47   return _Wrapper
  48 
  49 
  50 def GetOwner(benchmark_method_or_class):
  51   """Gets the inherited owner attribute for this benchmark.
  52 
  53   Checks for existence of __benchmark__owner__. If it's not present, looks for
  54   it in the parent class's attribute list.
  55 
  56   Args:
  57     benchmark_method_or_class: A benchmark method or class.
  58 
  59   Returns:
  60     string - the associated owner if present / None.
  61   """
  62   if hasattr(benchmark_method_or_class, '__benchmark__owner__'):
  63     return benchmark_method_or_class.__benchmark__owner__
  64   elif hasattr(benchmark_method_or_class, '__self__'):
  65     if hasattr(benchmark_method_or_class.__self__, '__benchmark__owner__'):
  66       return benchmark_method_or_class.__self__.__benchmark__owner__
  67   return None
