  39 class NCFKerasBenchmarkBase(PerfZeroBenchmark):
  40   """Base class for NCF model benchmark."""
  41 
  42   def __init__(self, output_dir=None, default_flags=None, **kwargs):
  43     super(NCFKerasBenchmarkBase, self).__init__(output_dir, default_flags,
  44                                                 **kwargs)
  45 
  46     # Run all benchmarks with ml_perf flag.
  47     self.default_flags['ml_perf'] = True
  48 
  49   def _setup(self):
  50     """Sets up and resets flags before each test."""
  51     logging.set_verbosity(logging.INFO)
  52     if NCFKerasBenchmarkBase.local_flags is None:
  53       ncf_common.define_ncf_flags()
  54       # Loads flags to get defaults to then override. List cannot be empty.
  55       flags.FLAGS(['foo'])
  56       core.set_defaults(**self.default_flags)
  57       saved_flag_values = flagsaver.save_flag_values()
  58       NCFKerasBenchmarkBase.local_flags = saved_flag_values
  59     else:
  60       flagsaver.restore_flag_values(NCFKerasBenchmarkBase.local_flags)
  61 
  62   @benchmark_wrappers.enable_runtime_flags
  63   def _run_and_report_benchmark(self, hr_at_10_min=0, hr_at_10_max=0):
  64     start_time_sec = time.time()
  65     stats = ncf_keras_main.run_ncf(FLAGS)
  66     wall_time_sec = time.time() - start_time_sec
  67 
  68     metrics = []
  69     metrics.append({'name': 'exp_per_second',
  70                     'value': stats['avg_exp_per_second']})
  71 
  72     if hr_at_10_min > 0:
  73       metrics.append({'name': 'hr_at_10',
  74                       'value': stats['eval_hit_rate'],
  75                       'min_value': hr_at_10_min,
  76                       'max_value': hr_at_10_max})
  77 
  78       metrics.append({'name': 'train_loss',
  79                       'value': stats['loss']})
  80 
  81     self.report_benchmark(iters=-1, wall_time=wall_time_sec, metrics=metrics)
