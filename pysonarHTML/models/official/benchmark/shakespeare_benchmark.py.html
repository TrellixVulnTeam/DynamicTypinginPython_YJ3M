  37 class ShakespeareBenchmarkBase(PerfZeroBenchmark):
  38   """Base class for Shakespeare (LSTM) benchmark and accuracy tests."""
  39 
  40   def __init__(self, output_dir=None, default_flags=None, root_data_dir=None):
  41     super(ShakespeareBenchmarkBase, self).__init__(
  42         output_dir=output_dir,
  43         default_flags=default_flags,
  44         flag_methods=[shakespeare_main.define_flags])
  45 
  46   @benchmark_wrappers.enable_runtime_flags
  47   def _run_and_report_benchmark(self,
  48                                 top_1_train_min=0.91,
  49                                 top_1_train_max=0.94,
  50                                 warmup=1,
  51                                 log_steps=100):
  52     """Report benchmark results by writing to local protobuf file.
  53 
  54     Average epoch time is calculated by skipping the first epoch. This average
  55     ignores time spent between epoch and is recorded by begin and end epoch. To
  56     skip accuracy check set `top_1_train_min=None`.
  57 
  58     Args:
  59       top_1_train_min: lowest passing value.
  60       top_1_train_max: highest passing value.
  61       warmup: number of entries in `timestamp_log` to ignore.
  62       log_steps: How often the log was created for `timestamp_log`.
  63     """
  64     total_batch_size = FLAGS.batch_size
  65     metrics = []
  66     start_time_sec = time.time()
  67     stats = shakespeare_main.run(FLAGS)
  68     wall_time_sec = time.time() - start_time_sec
  69 
  70     if top_1_train_min:
  71       metrics.append({'name': 'accuracy_top_1_train',
  72                       'value': stats['history']['RecallAt1'][-1],
  73                       'min_value': top_1_train_min,
  74                       'max_value': top_1_train_max})
  75 
  76     # Look for the time history callback which was used during keras.fit
  77     for callback in stats['callbacks']:
  78       if isinstance(callback, keras_utils.TimeHistory):
  79         epoch_timings = callback.epoch_runtime_log
  80         if len(epoch_timings) > 1:
  81           average_time = sum(epoch_timings[1:]) / len(epoch_timings[1:])
  82           metrics.append({'name': 'avg_epoch_time',
  83                           'value': average_time})
  84 
  85       # First entry in timestamp_log is the start of step 1. The rest of the
  86       # entries are the end of each step recorded.
  87       time_log = callback.timestamp_log
  88       elapsed = time_log[-1].timestamp - time_log[warmup].timestamp
  89       num_examples = (
  90           total_batch_size * log_steps * (len(time_log) - warmup - 1))
  91       if elapsed > 0:
  92         examples_per_sec = num_examples / elapsed
  93         metrics.append({'name': 'exp_per_second',
  94                         'value': examples_per_sec})
  95 
  96     flags_str = flags_core.get_nondefault_flags_as_str()
  97     self.report_benchmark(iters=-1, wall_time=wall_time_sec,
  98                           metrics=metrics,
  99                           extras={'flags': flags_str})

 102 class ShakespeareAccuracy(ShakespeareBenchmarkBase):
 103   """Shakespeare accuracy tests.
 104 
 105   This is not an ideal test. The best we can use for the accuracy check is to
 106   validate top_1 of the training set. At batch size 64 the top_1 training
 107   stabilizes to ~0.92 around 40-45 epochs.
 108   """
 109 
 110   def __init__(self, output_dir=None, root_data_dir=None, **kwargs):
 111     """Shakespeare accuracy tests.
 112 
 113     Args:
 114       output_dir: directory where to output e.g. log files
 115       root_data_dir: directory under which to look for dataset
 116       **kwargs: arbitrary named arguments. This is needed to make the
 117                 constructor forward compatible in case PerfZero provides more
 118                 named arguments before updating the constructor.
 119     """
 120     self.train_data = os.path.join(root_data_dir, SHAKESPEARE_TRAIN_DATA)
 121     super(ShakespeareAccuracy, self).__init__(
 122         output_dir=output_dir, root_data_dir=root_data_dir)
 123 
 124   def benchmark_cpu(self):
 125     """Benchmark cpu."""
 126     self._setup()
 127     FLAGS.num_gpus = 0
 128     FLAGS.training_data = self.train_data
 129     FLAGS.batch_size = 64
 130     FLAGS.train_epochs = 43
 131     FLAGS.model_dir = ''
 132     self._run_and_report_benchmark()
 133
