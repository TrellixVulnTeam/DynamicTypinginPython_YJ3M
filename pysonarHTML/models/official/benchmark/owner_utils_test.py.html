  15 """Tests for official.benchmark.owner_utils."""

  17 from absl.testing import absltest
  18 
  19 from official.benchmark import owner_utils
  20 
  21 
  22 @owner_utils.Owner('static_owner')
  23 def static_function(foo=5):
  24   return foo
  25 
  26 
  27 def static_function_without_owner(foo=5):
  28   return foo
  29 
  30 
  31 class BenchmarkClassWithoutOwner:
  32 
  33   def method_without_owner(self):
  34     return 100
  35 
  36   @owner_utils.Owner('method_owner')
  37   def method_with_owner(self):
  38     return 200
  39 
  40 
  41 @owner_utils.Owner('class_owner')
  42 class SomeBenchmarkClass:
  43 
  44   def method_inherited_owner(self):
  45     return 123
  46 
  47   @owner_utils.Owner('method_owner')
  48   def method_override_owner(self):
  49     return 345
  50 
  51 
  52 @owner_utils.Owner('new_class_owner')
  53 class InheritedClass(SomeBenchmarkClass):
  54 
  55   def method_inherited_owner(self):
  56     return 456
  57 
  58   @owner_utils.Owner('new_method_owner')
  59   def method_override_owner(self):
  60     return 567
  61 
  62 
  63 class OwnerUtilsTest(absltest.TestCase):
  64   """Tests to assert for owner decorator functionality."""
  65 
  66   def test_owner_tag_missing(self):
  67     self.assertEqual(None, owner_utils.GetOwner(static_function_without_owner))
  68 
  69     benchmark_class = BenchmarkClassWithoutOwner()
  70     self.assertEqual(None,
  71                      owner_utils.GetOwner(benchmark_class.method_without_owner))
  72     self.assertEqual(100, benchmark_class.method_without_owner())
  73 
  74     self.assertEqual('method_owner',
  75                      owner_utils.GetOwner(benchmark_class.method_with_owner))
  76     self.assertEqual(200, benchmark_class.method_with_owner())
  77 
  78   def test_owner_attributes_static(self):
  79     self.assertEqual('static_owner', owner_utils.GetOwner(static_function))
  80     self.assertEqual(5, static_function(5))
  81 
  82   def test_owner_attributes_per_class(self):
  83     level1 = SomeBenchmarkClass()
  84     self.assertEqual('class_owner',
  85                      owner_utils.GetOwner(level1.method_inherited_owner))
  86     self.assertEqual(123, level1.method_inherited_owner())
  87 
  88     self.assertEqual('method_owner',
  89                      owner_utils.GetOwner(level1.method_override_owner))
  90     self.assertEqual(345, level1.method_override_owner())
  91 
  92   def test_owner_attributes_inherited_class(self):
  93     level2 = InheritedClass()
  94     self.assertEqual('new_class_owner',
  95                      owner_utils.GetOwner(level2.method_inherited_owner))
  96     self.assertEqual(456, level2.method_inherited_owner())
  97 
  98     self.assertEqual('new_method_owner',
  99                      owner_utils.GetOwner(level2.method_override_owner))
 100     self.assertEqual(567, level2.method_override_owner())
