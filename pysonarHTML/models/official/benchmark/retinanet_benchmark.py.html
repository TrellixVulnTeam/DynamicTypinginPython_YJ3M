  47 class DetectionBenchmarkBase(perfzero_benchmark.PerfZeroBenchmark):
  48   """Base class to hold methods common to test classes."""
  49 
  50   def __init__(self, **kwargs):
  51     super(DetectionBenchmarkBase, self).__init__(**kwargs)
  52     self.timer_callback = None
  53 
  54   def _report_benchmark(self, stats, start_time_sec, wall_time_sec, min_ap,
  55                         max_ap, warmup):
  56     """Report benchmark results by writing to local protobuf file.
  57 
  58     Args:
  59       stats: dict returned from Detection models with known entries.
  60       start_time_sec: the start of the benchmark execution in seconds
  61       wall_time_sec: the duration of the benchmark execution in seconds
  62       min_ap: Minimum detection AP constraint to verify correctness of the
  63         model.
  64       max_ap: Maximum detection AP accuracy constraint to verify correctness of
  65         the model.
  66       warmup: Number of time log entries to ignore when computing examples/sec.
  67     """
  68     metrics = [{
  69         'name': 'total_loss',
  70         'value': stats['total_loss'],
  71     }]
  72     if self.timer_callback:
  73       metrics.append({
  74           'name': 'exp_per_second',
  75           'value': self.timer_callback.get_examples_per_sec(warmup)
  76       })
  77       metrics.append({
  78           'name': 'startup_time',
  79           'value': self.timer_callback.get_startup_time(start_time_sec)
  80       })
  81     else:
  82       metrics.append({
  83           'name': 'exp_per_second',
  84           'value': 0.0,
  85       })
  86 
  87     if 'eval_metrics' in stats:
  88       metrics.append({
  89           'name': 'AP',
  90           'value': stats['AP'],
  91           'min_value': min_ap,
  92           'max_value': max_ap,
  93       })
  94     flags_str = flags_core.get_nondefault_flags_as_str()
  95     self.report_benchmark(
  96         iters=stats['total_steps'],
  97         wall_time=wall_time_sec,
  98         metrics=metrics,
  99         extras={'flags': flags_str})

 102 class RetinanetBenchmarkBase(DetectionBenchmarkBase):
 103   """Base class to hold methods common to test classes in the module."""
 104 
 105   def __init__(self, **kwargs):
 106     self.train_data_path = COCO_TRAIN_DATA
 107     self.eval_data_path = COCO_EVAL_DATA
 108     self.eval_json_path = COCO_EVAL_JSON
 109     self.resnet_checkpoint_path = RESNET_CHECKPOINT_PATH
 110     super(RetinanetBenchmarkBase, self).__init__(**kwargs)
 111 
 112   def _run_detection_main(self):
 113     """Starts detection job."""
 114     if self.timer_callback:
 115       FLAGS.log_steps = 0  # prevent detection.run from adding the same callback
 116       return detection.run(callbacks=[self.timer_callback])
 117     else:
 118       return detection.run()

 121 class RetinanetAccuracy(RetinanetBenchmarkBase):
 122   """Accuracy test for RetinaNet model.
 123 
 124   Tests RetinaNet detection task model accuracy. The naming
 125   convention of below test cases follow
 126   `benchmark_(number of gpus)_gpu_(dataset type)` format.
 127   """
 128 
 129   @benchmark_wrappers.enable_runtime_flags
 130   def _run_and_report_benchmark(self,
 131                                 params,
 132                                 min_ap=0.325,
 133                                 max_ap=0.35,
 134                                 do_eval=True,
 135                                 warmup=1):
 136     """Starts RetinaNet accuracy benchmark test."""
 137     FLAGS.params_override = json.dumps(params)
 138     # Need timer callback to measure performance
 139     self.timer_callback = keras_utils.TimeHistory(
 140         batch_size=params['train']['batch_size'],
 141         log_steps=FLAGS.log_steps,
 142     )
 143 
 144     start_time_sec = time.time()
 145     FLAGS.mode = 'train'
 146     summary, _ = self._run_detection_main()
 147     wall_time_sec = time.time() - start_time_sec
 148 
 149     if do_eval:
 150       FLAGS.mode = 'eval'
 151       eval_metrics = self._run_detection_main()
 152       summary.update(eval_metrics)
 153 
 154     summary['total_steps'] = params['train']['total_steps']
 155     self._report_benchmark(summary, start_time_sec, wall_time_sec, min_ap,
 156                            max_ap, warmup)
 157 
 158   def _setup(self):
 159     super(RetinanetAccuracy, self)._setup()
 160     FLAGS.model = 'retinanet'
 161 
 162   def _params(self):
 163     return {
 164         'architecture': {
 165             'use_bfloat16': True,
 166         },
 167         'train': {
 168             'batch_size': 64,
 169             'iterations_per_loop': 100,
 170             'total_steps': 22500,
 171             'train_file_pattern': self.train_data_path,
 172             'checkpoint': {
 173                 'path': self.resnet_checkpoint_path,
 174                 'prefix': 'resnet50/'
 175             },
 176             # Speed up ResNet training when loading from the checkpoint.
 177             'frozen_variable_prefix': base_config.RESNET_FROZEN_VAR_PREFIX,
 178         },
 179         'eval': {
 180             'batch_size': 8,
 181             'eval_samples': 5000,
 182             'val_json_file': self.eval_json_path,
 183             'eval_file_pattern': self.eval_data_path,
 184         },
 185     }
 186 
 187   @flagsaver.flagsaver
 188   def benchmark_8_gpu_coco(self):
 189     """Run RetinaNet model accuracy test with 8 GPUs."""
 190     self._setup()
 191     params = self._params()
 192     FLAGS.num_gpus = 8
 193     FLAGS.model_dir = self._get_model_dir('benchmark_8_gpu_coco')
 194     FLAGS.strategy_type = 'mirrored'
 195     self._run_and_report_benchmark(params)

 198 class RetinanetBenchmarkReal(RetinanetAccuracy):
 199   """Short benchmark performance tests for RetinaNet model.
 200 
 201   Tests RetinaNet performance in different GPU configurations.
 202   The naming convention of below test cases follow
 203   `benchmark_(number of gpus)_gpu` format.
 204   """
 205 
 206   def _setup(self):
 207     super(RetinanetBenchmarkReal, self)._setup()
 208     # Use negative value to avoid saving checkpoints.
 209     FLAGS.save_checkpoint_freq = -1
 210 
 211   @flagsaver.flagsaver
 212   def benchmark_8_gpu_coco(self):
 213     """Run RetinaNet model accuracy test with 8 GPUs."""
 214     self._setup()
 215     params = self._params()
 216     params['architecture']['use_bfloat16'] = False
 217     params['train']['total_steps'] = 1875  # One epoch.
 218     # The iterations_per_loop must be one, otherwise the number of examples per
 219     # second would be wrong. Currently only support calling callback per batch
 220     # when each loop only runs on one batch, i.e. host loop for one step. The
 221     # performance of this situation might be lower than the case of
 222     # iterations_per_loop > 1.
 223     # Related bug: b/135933080
 224     params['train']['iterations_per_loop'] = 1
 225     params['eval']['eval_samples'] = 8
 226     FLAGS.num_gpus = 8
 227     FLAGS.model_dir = self._get_model_dir('real_benchmark_8_gpu_coco')
 228     FLAGS.strategy_type = 'mirrored'
 229     self._run_and_report_benchmark(params)
 230 
 231   @flagsaver.flagsaver
 232   def benchmark_1_gpu_coco(self):
 233     """Run RetinaNet model accuracy test with 1 GPU."""
 234     self._setup()
 235     params = self._params()
 236     params['architecture']['use_bfloat16'] = False
 237     params['train']['batch_size'] = 8
 238     params['train']['total_steps'] = 200
 239     params['train']['iterations_per_loop'] = 1
 240     params['eval']['eval_samples'] = 8
 241     FLAGS.num_gpus = 1
 242     FLAGS.model_dir = self._get_model_dir('real_benchmark_1_gpu_coco')
 243     FLAGS.strategy_type = 'one_device'
 244     self._run_and_report_benchmark(params)
 245 
 246   @flagsaver.flagsaver
 247   def benchmark_xla_1_gpu_coco(self):
 248     """Run RetinaNet model accuracy test with 1 GPU and XLA enabled."""
 249     self._setup()
 250     params = self._params()
 251     params['architecture']['use_bfloat16'] = False
 252     params['train']['batch_size'] = 8
 253     params['train']['total_steps'] = 200
 254     params['train']['iterations_per_loop'] = 1
 255     params['eval']['eval_samples'] = 8
 256     FLAGS.num_gpus = 1
 257     FLAGS.model_dir = self._get_model_dir('real_benchmark_xla_1_gpu_coco')
 258     FLAGS.strategy_type = 'one_device'
 259     FLAGS.enable_xla = True
 260     self._run_and_report_benchmark(params)
 261 
 262   @flagsaver.flagsaver
 263   def benchmark_2x2_tpu_coco(self):
 264     """Run RetinaNet model accuracy test with 4 TPUs."""
 265     self._setup()
 266     params = self._params()
 267     params['train']['batch_size'] = 64
 268     params['train']['total_steps'] = 1875  # One epoch.
 269     params['train']['iterations_per_loop'] = 500
 270     FLAGS.model_dir = self._get_model_dir('real_benchmark_2x2_tpu_coco')
 271     FLAGS.strategy_type = 'tpu'
 272     self._run_and_report_benchmark(params, do_eval=False, warmup=0)
