  37 def to_4d(image: tf.Tensor) -> tf.Tensor:
  38   """Converts an input Tensor to 4 dimensions.
  39 
  40   4D image => [N, H, W, C] or [N, C, H, W]
  41   3D image => [1, H, W, C] or [1, C, H, W]
  42   2D image => [1, H, W, 1]
  43 
  44   Args:
  45     image: The 2/3/4D input tensor.
  46 
  47   Returns:
  48     A 4D image tensor.
  49 
  50   Raises:
  51     `TypeError` if `image` is not a 2/3/4D tensor.
  52 
  53   """
  54   shape = tf.shape(image)
  55   original_rank = tf.rank(image)
  56   left_pad = tf.cast(tf.less_equal(original_rank, 3), dtype=tf.int32)
  57   right_pad = tf.cast(tf.equal(original_rank, 2), dtype=tf.int32)
  58   new_shape = tf.concat(
  59       [
  60           tf.ones(shape=left_pad, dtype=tf.int32),
  61           shape,
  62           tf.ones(shape=right_pad, dtype=tf.int32),
  63       ],
  64       axis=0,
  65   )
  66   return tf.reshape(image, new_shape)
  67 
  68 
  69 def from_4d(image: tf.Tensor, ndims: tf.Tensor) -> tf.Tensor:
  70   """Converts a 4D image back to `ndims` rank."""
  71   shape = tf.shape(image)
  72   begin = tf.cast(tf.less_equal(ndims, 3), dtype=tf.int32)
  73   end = 4 - tf.cast(tf.equal(ndims, 2), dtype=tf.int32)
  74   new_shape = shape[begin:end]
  75   return tf.reshape(image, new_shape)

  78 def _convert_translation_to_transform(translations: tf.Tensor) -> tf.Tensor:
  79   """Converts translations to a projective transform.
  80 
  81   The translation matrix looks like this:
  82     [[1 0 -dx]
  83      [0 1 -dy]
  84      [0 0 1]]
  85 
  86   Args:
  87     translations: The 2-element list representing [dx, dy], or a matrix of
  88       2-element lists representing [dx dy] to translate for each image. The
  89       shape must be static.
  90 
  91   Returns:
  92     The transformation matrix of shape (num_images, 8).
  93 
  94   Raises:
  95     `TypeError` if
  96       - the shape of `translations` is not known or
  97       - the shape of `translations` is not rank 1 or 2.
  98 
  99   """
 100   translations = tf.convert_to_tensor(translations, dtype=tf.float32)
 101   if translations.get_shape().ndims is None:
 102     raise TypeError('translations rank must be statically known')
 103   elif len(translations.get_shape()) == 1:
 104     translations = translations[None]
 105   elif len(translations.get_shape()) != 2:
 106     raise TypeError('translations should have rank 1 or 2.')
 107   num_translations = tf.shape(translations)[0]
 108 
 109   return tf.concat(
 110       values=[
 111           tf.ones((num_translations, 1), tf.dtypes.float32),
 112           tf.zeros((num_translations, 1), tf.dtypes.float32),
 113           -translations[:, 0, None],
 114           tf.zeros((num_translations, 1), tf.dtypes.float32),
 115           tf.ones((num_translations, 1), tf.dtypes.float32),
 116           -translations[:, 1, None],
 117           tf.zeros((num_translations, 2), tf.dtypes.float32),
 118       ],
 119       axis=1,
 120   )

 123 def _convert_angles_to_transform(
 124     angles: tf.Tensor,
 125     image_width: tf.Tensor,
 126     image_height: tf.Tensor) -> tf.Tensor:
 127   """Converts an angle or angles to a projective transform.
 128 
 129   Args:
 130     angles: A scalar to rotate all images, or a vector to rotate a batch of
 131       images. This must be a scalar.
 132     image_width: The width of the image(s) to be transformed.
 133     image_height: The height of the image(s) to be transformed.
 134 
 135   Returns:
 136     A tensor of shape (num_images, 8).
 137 
 138   Raises:
 139     `TypeError` if `angles` is not rank 0 or 1.
 140 
 141   """
 142   angles = tf.convert_to_tensor(angles, dtype=tf.float32)
 143   if len(angles.get_shape()) == 0:  # pylint:disable=g-explicit-length-test
 144     angles = angles[None]
 145   elif len(angles.get_shape()) != 1:
 146     raise TypeError('Angles should have a rank 0 or 1.')
 147   x_offset = ((image_width - 1) -
 148               (tf.math.cos(angles) * (image_width - 1) - tf.math.sin(angles) *
 149                (image_height - 1))) / 2.0
 150   y_offset = ((image_height - 1) -
 151               (tf.math.sin(angles) * (image_width - 1) + tf.math.cos(angles) *
 152                (image_height - 1))) / 2.0
 153   num_angles = tf.shape(angles)[0]
 154   return tf.concat(
 155       values=[
 156           tf.math.cos(angles)[:, None],
 157           -tf.math.sin(angles)[:, None],
 158           x_offset[:, None],
 159           tf.math.sin(angles)[:, None],
 160           tf.math.cos(angles)[:, None],
 161           y_offset[:, None],
 162           tf.zeros((num_angles, 2), tf.dtypes.float32),
 163       ],
 164       axis=1,
 165   )

 168 def transform(image: tf.Tensor, transforms) -> tf.Tensor:
 169   """Prepares input data for `image_ops.transform`."""
 170   original_ndims = tf.rank(image)
 171   transforms = tf.convert_to_tensor(transforms, dtype=tf.float32)
 172   if transforms.shape.rank == 1:
 173     transforms = transforms[None]
 174   image = to_4d(image)
 175   image = image_ops.transform(
 176       images=image,
 177       transforms=transforms,
 178       interpolation='nearest')
 179   return from_4d(image, original_ndims)

 182 def translate(image: tf.Tensor, translations) -> tf.Tensor:
 183   """Translates image(s) by provided vectors.
 184 
 185   Args:
 186     image: An image Tensor of type uint8.
 187     translations: A vector or matrix representing [dx dy].
 188 
 189   Returns:
 190     The translated version of the image.
 191 
 192   """
 193   transforms = _convert_translation_to_transform(translations)
 194   return transform(image, transforms=transforms)

 197 def rotate(image: tf.Tensor, degrees: float) -> tf.Tensor:
 198   """Rotates the image by degrees either clockwise or counterclockwise.
 199 
 200   Args:
 201     image: An image Tensor of type uint8.
 202     degrees: Float, a scalar angle in degrees to rotate all images by. If
 203       degrees is positive the image will be rotated clockwise otherwise it will
 204       be rotated counterclockwise.
 205 
 206   Returns:
 207     The rotated version of image.
 208 
 209   """
 210   # Convert from degrees to radians.
 211   degrees_to_radians = math.pi / 180.0
 212   radians = tf.cast(degrees * degrees_to_radians, tf.float32)
 213 
 214   original_ndims = tf.rank(image)
 215   image = to_4d(image)
 216 
 217   image_height = tf.cast(tf.shape(image)[1], tf.float32)
 218   image_width = tf.cast(tf.shape(image)[2], tf.float32)
 219   transforms = _convert_angles_to_transform(angles=radians,
 220                                             image_width=image_width,
 221                                             image_height=image_height)
 222   # In practice, we should randomize the rotation degrees by flipping
 223   # it negatively half the time, but that's done on 'degrees' outside
 224   # of the function.
 225   image = transform(image, transforms=transforms)
 226   return from_4d(image, original_ndims)

 229 def blend(image1: tf.Tensor, image2: tf.Tensor, factor: float) -> tf.Tensor:
 230   """Blend image1 and image2 using 'factor'.
 231 
 232   Factor can be above 0.0.  A value of 0.0 means only image1 is used.
 233   A value of 1.0 means only image2 is used.  A value between 0.0 and
 234   1.0 means we linearly interpolate the pixel values between the two
 235   images.  A value greater than 1.0 "extrapolates" the difference
 236   between the two pixel values, and we clip the results to values
 237   between 0 and 255.
 238 
 239   Args:
 240     image1: An image Tensor of type uint8.
 241     image2: An image Tensor of type uint8.
 242     factor: A floating point value above 0.0.
 243 
 244   Returns:
 245     A blended image Tensor of type uint8.
 246   """
 247   if factor == 0.0:
 248     return tf.convert_to_tensor(image1)
 249   if factor == 1.0:
 250     return tf.convert_to_tensor(image2)
 251 
 252   image1 = tf.cast(image1, tf.float32)
 253   image2 = tf.cast(image2, tf.float32)
 254 
 255   difference = image2 - image1
 256   scaled = factor * difference
 257 
 258   # Do addition in float.
 259   temp = tf.cast(image1, tf.float32) + scaled
 260 
 261   # Interpolate
 262   if factor > 0.0 and factor < 1.0:
 263     # Interpolation means we always stay within 0 and 255.
 264     return tf.cast(temp, tf.uint8)
 265 
 266   # Extrapolate:
 267   #
 268   # We need to clip and then cast.
 269   return tf.cast(tf.clip_by_value(temp, 0.0, 255.0), tf.uint8)

 272 def cutout(image: tf.Tensor, pad_size: int, replace: int = 0) -> tf.Tensor:
 273   """Apply cutout (https://arxiv.org/abs/1708.04552) to image.
 274 
 275   This operation applies a (2*pad_size x 2*pad_size) mask of zeros to
 276   a random location within `img`. The pixel values filled in will be of the
 277   value `replace`. The located where the mask will be applied is randomly
 278   chosen uniformly over the whole image.
 279 
 280   Args:
 281     image: An image Tensor of type uint8.
 282     pad_size: Specifies how big the zero mask that will be generated is that
 283       is applied to the image. The mask will be of size
 284       (2*pad_size x 2*pad_size).
 285     replace: What pixel value to fill in the image in the area that has
 286       the cutout mask applied to it.
 287 
 288   Returns:
 289     An image Tensor that is of type uint8.
 290   """
 291   image_height = tf.shape(image)[0]
 292   image_width = tf.shape(image)[1]
 293 
 294   # Sample the center location in the image where the zero mask will be applied.
 295   cutout_center_height = tf.random.uniform(
 296       shape=[], minval=0, maxval=image_height,
 297       dtype=tf.int32)
 298 
 299   cutout_center_width = tf.random.uniform(
 300       shape=[], minval=0, maxval=image_width,
 301       dtype=tf.int32)
 302 
 303   lower_pad = tf.maximum(0, cutout_center_height - pad_size)
 304   upper_pad = tf.maximum(0, image_height - cutout_center_height - pad_size)
 305   left_pad = tf.maximum(0, cutout_center_width - pad_size)
 306   right_pad = tf.maximum(0, image_width - cutout_center_width - pad_size)
 307 
 308   cutout_shape = [image_height - (lower_pad + upper_pad),
 309                   image_width - (left_pad + right_pad)]
 310   padding_dims = [[lower_pad, upper_pad], [left_pad, right_pad]]
 311   mask = tf.pad(
 312       tf.zeros(cutout_shape, dtype=image.dtype),
 313       padding_dims, constant_values=1)
 314   mask = tf.expand_dims(mask, -1)
 315   mask = tf.tile(mask, [1, 1, 3])
 316   image = tf.where(
 317       tf.equal(mask, 0),
 318       tf.ones_like(image, dtype=image.dtype) * replace,
 319       image)
 320   return image

 323 def solarize(image: tf.Tensor, threshold: int = 128) -> tf.Tensor:
 324   # For each pixel in the image, select the pixel
 325   # if the value is less than the threshold.
 326   # Otherwise, subtract 255 from the pixel.
 327   return tf.where(image < threshold, image, 255 - image)

 330 def solarize_add(image: tf.Tensor,
 331                  addition: int = 0,
 332                  threshold: int = 128) -> tf.Tensor:
 333   # For each pixel in the image less than threshold
 334   # we add 'addition' amount to it and then clip the
 335   # pixel value to be between 0 and 255. The value
 336   # of 'addition' is between -128 and 128.
 337   added_image = tf.cast(image, tf.int64) + addition
 338   added_image = tf.cast(tf.clip_by_value(added_image, 0, 255), tf.uint8)
 339   return tf.where(image < threshold, added_image, image)

 348 def contrast(image: tf.Tensor, factor: float) -> tf.Tensor:
 349   """Equivalent of PIL Contrast."""
 350   degenerate = tf.image.rgb_to_grayscale(image)
 351   # Cast before calling tf.histogram.
 352   degenerate = tf.cast(degenerate, tf.int32)
 353 
 354   # Compute the grayscale histogram, then compute the mean pixel value,
 355   # and create a constant image size of that value.  Use that as the
 356   # blending degenerate target of the original image.
 357   hist = tf.histogram_fixed_width(degenerate, [0, 255], nbins=256)
 358   mean = tf.reduce_sum(tf.cast(hist, tf.float32)) / 256.0
 359   degenerate = tf.ones_like(degenerate, dtype=tf.float32) * mean
 360   degenerate = tf.clip_by_value(degenerate, 0.0, 255.0)
 361   degenerate = tf.image.grayscale_to_rgb(tf.cast(degenerate, tf.uint8))
 362   return blend(degenerate, image, factor)

 417 def autocontrast(image: tf.Tensor) -> tf.Tensor:
 418   """Implements Autocontrast function from PIL using TF ops.
 419 
 420   Args:
 421     image: A 3D uint8 tensor.
 422 
 423   Returns:
 424     The image after it has had autocontrast applied to it and will be of type
 425     uint8.
 426   """
 427 
 428   def scale_channel(image: tf.Tensor) -> tf.Tensor:
 429     """Scale the 2D image using the autocontrast rule."""
 430     # A possibly cheaper version can be done using cumsum/unique_with_counts
 431     # over the histogram values, rather than iterating over the entire image.
 432     # to compute mins and maxes.
 433     lo = tf.cast(tf.reduce_min(image), tf.float32)
 434     hi = tf.cast(tf.reduce_max(image), tf.float32)
 435 
 436     # Scale the image, making the lowest value 0 and the highest value 255.
 437     def scale_values(im):
 438       scale = 255.0 / (hi - lo)
 439       offset = -lo * scale
 440       im = tf.cast(im, tf.float32) * scale + offset
 441       im = tf.clip_by_value(im, 0.0, 255.0)
 442       return tf.cast(im, tf.uint8)
 443 
 444     result = tf.cond(hi > lo, lambda: scale_values(image), lambda: image)
 445     return result
 446 
 447   # Assumes RGB for now.  Scales each channel independently
 448   # and then stacks the result.
 449   s1 = scale_channel(image[:, :, 0])
 450   s2 = scale_channel(image[:, :, 1])
 451   s3 = scale_channel(image[:, :, 2])
 452   image = tf.stack([s1, s2, s3], 2)
 453   return image

 456 def sharpness(image: tf.Tensor, factor: float) -> tf.Tensor:
 457   """Implements Sharpness function from PIL using TF ops."""
 458   orig_image = image
 459   image = tf.cast(image, tf.float32)
 460   # Make image 4D for conv operation.
 461   image = tf.expand_dims(image, 0)
 462   # SMOOTH PIL Kernel.
 463   kernel = tf.constant(
 464       [[1, 1, 1], [1, 5, 1], [1, 1, 1]], dtype=tf.float32,
 465       shape=[3, 3, 1, 1]) / 13.
 466   # Tile across channel dimension.
 467   kernel = tf.tile(kernel, [1, 1, 3, 1])
 468   strides = [1, 1, 1, 1]
 469   degenerate = tf.nn.depthwise_conv2d(
 470       image, kernel, strides, padding='VALID', dilations=[1, 1])
 471   degenerate = tf.clip_by_value(degenerate, 0.0, 255.0)
 472   degenerate = tf.squeeze(tf.cast(degenerate, tf.uint8), [0])
 473 
 474   # For the borders of the resulting image, fill in the values of the
 475   # original image.
 476   mask = tf.ones_like(degenerate)
 477   padded_mask = tf.pad(mask, [[1, 1], [1, 1], [0, 0]])
 478   padded_degenerate = tf.pad(degenerate, [[1, 1], [1, 1], [0, 0]])
 479   result = tf.where(tf.equal(padded_mask, 1), padded_degenerate, orig_image)
 480 
 481   # Blend the final result.
 482   return blend(result, orig_image, factor)

 485 def equalize(image: tf.Tensor) -> tf.Tensor:
 486   """Implements Equalize function from PIL using TF ops."""
 487   def scale_channel(im, c):
 488     """Scale the data in the channel to implement equalize."""
 489     im = tf.cast(im[:, :, c], tf.int32)
 490     # Compute the histogram of the image channel.
 491     histo = tf.histogram_fixed_width(im, [0, 255], nbins=256)
 492 
 493     # For the purposes of computing the step, filter out the nonzeros.
 494     nonzero = tf.where(tf.not_equal(histo, 0))
 495     nonzero_histo = tf.reshape(tf.gather(histo, nonzero), [-1])
 496     step = (tf.reduce_sum(nonzero_histo) - nonzero_histo[-1]) // 255
 497 
 498     def build_lut(histo, step):
 499       # Compute the cumulative sum, shifting by step // 2
 500       # and then normalization by step.
 501       lut = (tf.cumsum(histo) + (step // 2)) // step
 502       # Shift lut, prepending with 0.
 503       lut = tf.concat([[0], lut[:-1]], 0)
 504       # Clip the counts to be in range.  This is done
 505       # in the C code for image.point.
 506       return tf.clip_by_value(lut, 0, 255)
 507 
 508     # If step is zero, return the original image.  Otherwise, build
 509     # lut from the full histogram and step and then index from it.
 510     result = tf.cond(tf.equal(step, 0),
 511                      lambda: im,
 512                      lambda: tf.gather(build_lut(histo, step), im))
 513 
 514     return tf.cast(result, tf.uint8)
 515 
 516   # Assumes RGB for now.  Scales each channel independently
 517   # and then stacks the result.
 518   s1 = scale_channel(image, 0)
 519   s2 = scale_channel(image, 1)
 520   s3 = scale_channel(image, 2)
 521   image = tf.stack([s1, s2, s3], 2)
 522   return image

 531 def wrap(image: tf.Tensor) -> tf.Tensor:
 532   """Returns 'image' with an extra channel set to all 1s."""
 533   shape = tf.shape(image)
 534   extended_channel = tf.ones([shape[0], shape[1], 1], image.dtype)
 535   extended = tf.concat([image, extended_channel], axis=2)
 536   return extended

 539 def unwrap(image: tf.Tensor, replace: int) -> tf.Tensor:
 540   """Unwraps an image produced by wrap.
 541 
 542   Where there is a 0 in the last channel for every spatial position,
 543   the rest of the three channels in that spatial dimension are grayed
 544   (set to 128).  Operations like translate and shear on a wrapped
 545   Tensor will leave 0s in empty locations.  Some transformations look
 546   at the intensity of values to do preprocessing, and we want these
 547   empty pixels to assume the 'average' value, rather than pure black.
 548 
 549 
 550   Args:
 551     image: A 3D Image Tensor with 4 channels.
 552     replace: A one or three value 1D tensor to fill empty pixels.
 553 
 554   Returns:
 555     image: A 3D image Tensor with 3 channels.
 556   """
 557   image_shape = tf.shape(image)
 558   # Flatten the spatial dimensions.
 559   flattened_image = tf.reshape(image, [-1, image_shape[2]])
 560 
 561   # Find all pixels where the last channel is zero.
 562   alpha_channel = tf.expand_dims(flattened_image[:, 3], axis=-1)
 563 
 564   replace = tf.concat([replace, tf.ones([1], image.dtype)], 0)
 565 
 566   # Where they are zero, fill them in with 'replace'.
 567   flattened_image = tf.where(
 568       tf.equal(alpha_channel, 0),
 569       tf.ones_like(flattened_image, dtype=image.dtype) * replace,
 570       flattened_image)
 571 
 572   image = tf.reshape(flattened_image, image_shape)
 573   image = tf.slice(image, [0, 0, 0], [image_shape[0], image_shape[1], 3])
 574   return image

 577 def _randomly_negate_tensor(tensor):
 578   """With 50% prob turn the tensor negative."""
 579   should_flip = tf.cast(tf.floor(tf.random.uniform([]) + 0.5), tf.bool)
 580   final_tensor = tf.cond(should_flip, lambda: tensor, lambda: -tensor)
 581   return final_tensor
 582 
 583 
 584 def _rotate_level_to_arg(level: float):
 585   level = (level/_MAX_LEVEL) * 30.
 586   level = _randomly_negate_tensor(level)
 587   return (level,)
 588 
 589 
 590 def _shrink_level_to_arg(level: float):
 591   """Converts level to ratio by which we shrink the image content."""
 592   if level == 0:
 593     return (1.0,)  # if level is zero, do not shrink the image
 594   # Maximum shrinking ratio is 2.9.
 595   level = 2. / (_MAX_LEVEL / level) + 0.9
 596   return (level,)
 597 
 598 
 599 def _enhance_level_to_arg(level: float):
 600   return ((level/_MAX_LEVEL) * 1.8 + 0.1,)
 601 
 602 
 603 def _shear_level_to_arg(level: float):
 604   level = (level/_MAX_LEVEL) * 0.3
 605   # Flip level to negative with 50% chance.
 606   level = _randomly_negate_tensor(level)
 607   return (level,)
 608 
 609 
 610 def _translate_level_to_arg(level: float, translate_const: float):
 611   level = (level/_MAX_LEVEL) * float(translate_const)
 612   # Flip level to negative with 50% chance.
 613   level = _randomly_negate_tensor(level)
 614   return (level,)
 615 
 616 
 617 def _mult_to_arg(level: float, multiplier: float = 1.):
 618   return (int((level / _MAX_LEVEL) * multiplier),)
 619 
 620 
 621 def _apply_func_with_prob(func: Any,
 622                           image: tf.Tensor,
 623                           args: Any,
 624                           prob: float):
 625   """Apply `func` to image w/ `args` as input with probability `prob`."""
 626   assert isinstance(args, tuple)
 627 
 628   # Apply the function with probability `prob`.
 629   should_apply_op = tf.cast(
 630       tf.floor(tf.random.uniform([], dtype=tf.float32) + prob), tf.bool)
 631   augmented_image = tf.cond(
 632       should_apply_op,
 633       lambda: func(image, *args),
 634       lambda: image)
 635   return augmented_image
 636 
 637 
 638 def select_and_apply_random_policy(policies: Any, image: tf.Tensor):
 639   """Select a random policy from `policies` and apply it to `image`."""
 640   policy_to_select = tf.random.uniform([], maxval=len(policies), dtype=tf.int32)
 641   # Note that using tf.case instead of tf.conds would result in significantly
 642   # larger graphs and would even break export for some larger policies.
 643   for (i, policy) in enumerate(policies):
 644     image = tf.cond(
 645         tf.equal(i, policy_to_select),
 646         lambda selected_policy=policy: selected_policy(image),
 647         lambda: image)
 648   return image

 652     'AutoContrast': autocontrast,
 653     'Equalize': equalize,
 654     'Invert': invert,
 655     'Rotate': wrapped_rotate,
 656     'Posterize': posterize,
 657     'Solarize': solarize,
 658     'SolarizeAdd': solarize_add,
 659     'Color': color,

 663     'ShearX': shear_x,
 664     'ShearY': shear_y,
 665     'TranslateX': translate_x,
 666     'TranslateY': translate_y,
 667     'Cutout': cutout,
 668 }

 670 # Functions that have a 'replace' parameter
 671 REPLACE_FUNCS = frozenset({
 672     'Rotate',
 673     'TranslateX',
 674     'ShearX',
 675     'ShearY',
 676     'TranslateY',
 677     'Cutout',
 678 })

 681 def level_to_arg(cutout_const: float, translate_const: float):
 682   """Creates a dict mapping image operation names to their arguments."""
 683 
 684   no_arg = lambda level: ()
 685   posterize_arg = lambda level: _mult_to_arg(level, 4)
 686   solarize_arg = lambda level: _mult_to_arg(level, 256)
 687   solarize_add_arg = lambda level: _mult_to_arg(level, 110)
 688   cutout_arg = lambda level: _mult_to_arg(level, cutout_const)
 689   translate_arg = lambda level: _translate_level_to_arg(level, translate_const)
 690 
 691   args = {
 692       'AutoContrast': no_arg,
 693       'Equalize': no_arg,
 694       'Invert': no_arg,
 695       'Rotate': _rotate_level_to_arg,
 696       'Posterize': posterize_arg,
 697       'Solarize': solarize_arg,
 698       'SolarizeAdd': solarize_add_arg,
 699       'Color': _enhance_level_to_arg,
 700       'Contrast': _enhance_level_to_arg,
 701       'Brightness': _enhance_level_to_arg,
 702       'Sharpness': _enhance_level_to_arg,
 703       'ShearX': _shear_level_to_arg,
 704       'ShearY': _shear_level_to_arg,
 705       'Cutout': cutout_arg,
 706       'TranslateX': translate_arg,
 707       'TranslateY': translate_arg,
 708   }
 709   return args

 712 def _parse_policy_info(name: Text,
 713                        prob: float,
 714                        level: float,
 715                        replace_value: List[int],
 716                        cutout_const: float,
 717                        translate_const: float) -> Tuple[Any, float, Any]:
 718   """Return the function that corresponds to `name` and update `level` param."""
 719   func = NAME_TO_FUNC[name]
 720   args = level_to_arg(cutout_const, translate_const)[name](level)
 721 
 722   if name in REPLACE_FUNCS:
 723     # Add in replace arg if it is required for the function that is called.
 724     args = tuple(list(args) + [replace_value])
 725 
 726   return func, prob, args

 744 class AutoAugment(ImageAugment):
 745   """Applies the AutoAugment policy to images.
 746 
 747     AutoAugment is from the paper: https://arxiv.org/abs/1805.09501.
 748   """
 749 
 750   def __init__(self,
 751                augmentation_name: Text = 'v0',
 752                policies: Optional[Dict[Text, Any]] = None,
 753                cutout_const: float = 100,
 754                translate_const: float = 250):
 755     """Applies the AutoAugment policy to images.
 756 
 757     Args:
 758       augmentation_name: The name of the AutoAugment policy to use. The
 759         available options are `v0` and `test`. `v0` is the policy used for all
 760         of the results in the paper and was found to achieve the best results on
 761         the COCO dataset. `v1`, `v2` and `v3` are additional good policies found
 762         on the COCO dataset that have slight variation in what operations were
 763         used during the search procedure along with how many operations are
 764         applied in parallel to a single image (2 vs 3).
 765       policies: list of lists of tuples in the form `(func, prob, level)`,
 766         `func` is a string name of the augmentation function, `prob` is the
 767         probability of applying the `func` operation, `level` is the input
 768         argument for `func`.
 769       cutout_const: multiplier for applying cutout.
 770       translate_const: multiplier for applying translation.
 771     """
 772     super(AutoAugment, self).__init__()
 773 
 774     if policies is None:
 775       self.available_policies = {
 776           'v0': self.policy_v0(),
 777           'test': self.policy_test(),
 778           'simple': self.policy_simple(),
 779       }
 780 
 781     if augmentation_name not in self.available_policies:
 782       raise ValueError(
 783           'Invalid augmentation_name: {}'.format(augmentation_name))
 784 
 785     self.augmentation_name = augmentation_name
 786     self.policies = self.available_policies[augmentation_name]
 787     self.cutout_const = float(cutout_const)
 788     self.translate_const = float(translate_const)
 789 
 790   def distort(self, image: tf.Tensor) -> tf.Tensor:
 791     """Applies the AutoAugment policy to `image`.
 792 
 793     AutoAugment is from the paper: https://arxiv.org/abs/1805.09501.
 794 
 795     Args:
 796       image: `Tensor` of shape [height, width, 3] representing an image.
 797 
 798     Returns:
 799       A version of image that now has data augmentation applied to it based on
 800       the `policies` pass into the function.
 801     """
 802     input_image_type = image.dtype
 803 
 804     if input_image_type != tf.uint8:
 805       image = tf.clip_by_value(image, 0.0, 255.0)
 806       image = tf.cast(image, dtype=tf.uint8)
 807 
 808     replace_value = [128] * 3
 809 
 810     # func is the string name of the augmentation function, prob is the
 811     # probability of applying the operation and level is the parameter
 812     # associated with the tf op.
 813 
 814     # tf_policies are functions that take in an image and return an augmented
 815     # image.
 816     tf_policies = []
 817     for policy in self.policies:
 818       tf_policy = []
 819       # Link string name to the correct python function and make sure the
 820       # correct argument is passed into that function.
 821       for policy_info in policy:
 822         policy_info = list(policy_info) + [
 823             replace_value, self.cutout_const, self.translate_const
 824         ]
 825         tf_policy.append(_parse_policy_info(*policy_info))
 826       # Now build the tf policy that will apply the augmentation procedue
 827       # on image.
 828       def make_final_policy(tf_policy_):
 829 
 830         def final_policy(image_):
 831           for func, prob, args in tf_policy_:
 832             image_ = _apply_func_with_prob(func, image_, args, prob)
 833           return image_
 834 
 835         return final_policy
 836 
 837       tf_policies.append(make_final_policy(tf_policy))
 838 
 839     image = select_and_apply_random_policy(tf_policies, image)
 840     image = tf.cast(image, dtype=input_image_type)
 841     return image

 918 class RandAugment(ImageAugment):
 919   """Applies the RandAugment policy to images.
 920 
 921   RandAugment is from the paper https://arxiv.org/abs/1909.13719,
 922   """
 923 
 924   def __init__(self,
 925                num_layers: int = 2,
 926                magnitude: float = 10.,
 927                cutout_const: float = 40.,
 928                translate_const: float = 100.):
 929     """Applies the RandAugment policy to images.
 930 
 931     Args:
 932       num_layers: Integer, the number of augmentation transformations to apply
 933         sequentially to an image. Represented as (N) in the paper. Usually best
 934         values will be in the range [1, 3].
 935       magnitude: Integer, shared magnitude across all augmentation operations.
 936         Represented as (M) in the paper. Usually best values are in the range
 937         [5, 10].
 938       cutout_const: multiplier for applying cutout.
 939       translate_const: multiplier for applying translation.
 940     """
 941     super(RandAugment, self).__init__()
 942 
 943     self.num_layers = num_layers
 944     self.magnitude = float(magnitude)
 945     self.cutout_const = float(cutout_const)
 946     self.translate_const = float(translate_const)
 947     self.available_ops = [
 948         'AutoContrast', 'Equalize', 'Invert', 'Rotate', 'Posterize', 'Solarize',
 949         'Color', 'Contrast', 'Brightness', 'Sharpness', 'ShearX', 'ShearY',
 950         'TranslateX', 'TranslateY', 'Cutout', 'SolarizeAdd'
 951     ]
 952 
 953   def distort(self, image: tf.Tensor) -> tf.Tensor:
 954     """Applies the RandAugment policy to `image`.
 955 
 956     Args:
 957       image: `Tensor` of shape [height, width, 3] representing an image.
 958 
 959     Returns:
 960       The augmented version of `image`.
 961     """
 962     input_image_type = image.dtype
 963 
 964     if input_image_type != tf.uint8:
 965       image = tf.clip_by_value(image, 0.0, 255.0)
 966       image = tf.cast(image, dtype=tf.uint8)
 967 
 968     replace_value = [128] * 3
 969     min_prob, max_prob = 0.2, 0.8
 970 
 971     for _ in range(self.num_layers):
 972       op_to_select = tf.random.uniform(
 973           [], maxval=len(self.available_ops) + 1, dtype=tf.int32)
 974 
 975       branch_fns = []
 976       for (i, op_name) in enumerate(self.available_ops):
 977         prob = tf.random.uniform([],
 978                                  minval=min_prob,
 979                                  maxval=max_prob,
 980                                  dtype=tf.float32)
 981         func, _, args = _parse_policy_info(op_name,
 982                                            prob,
 983                                            self.magnitude,
 984                                            replace_value,
 985                                            self.cutout_const,
 986                                            self.translate_const)
 987         branch_fns.append((
 988             i,
 989             # pylint:disable=g-long-lambda
 990             lambda selected_func=func, selected_args=args: selected_func(
 991                 image, *selected_args)))
 992         # pylint:enable=g-long-lambda
 993 
 994       image = tf.switch_case(branch_index=op_to_select,
 995                              branch_fns=branch_fns,
 996                              default=lambda: tf.identity(image))
 997 
 998     image = tf.cast(image, dtype=input_image_type)
 999     return image
