  31 def get_callbacks(model_checkpoint: bool = True,
  32                   include_tensorboard: bool = True,
  33                   time_history: bool = True,
  34                   track_lr: bool = True,
  35                   write_model_weights: bool = True,
  36                   apply_moving_average: bool = False,
  37                   initial_step: int = 0,
  38                   batch_size: int = 0,
  39                   log_steps: int = 0,
  40                   model_dir: str = None) -> List[tf.keras.callbacks.Callback]:
  41   """Get all callbacks."""
  42   model_dir = model_dir or ''
  43   callbacks = []
  44   if model_checkpoint:
  45     ckpt_full_path = os.path.join(model_dir, 'model.ckpt-{epoch:04d}')
  46     callbacks.append(tf.keras.callbacks.ModelCheckpoint(
  47         ckpt_full_path, save_weights_only=True, verbose=1))
  48   if include_tensorboard:
  49     callbacks.append(
  50         CustomTensorBoard(
  51             log_dir=model_dir,
  52             track_lr=track_lr,
  53             initial_step=initial_step,
  54             write_images=write_model_weights))
  55   if time_history:
  56     callbacks.append(
  57         keras_utils.TimeHistory(
  58             batch_size,
  59             log_steps,
  60             logdir=model_dir if include_tensorboard else None))
  61   if apply_moving_average:
  62     # Save moving average model to a different file so that
  63     # we can resume training from a checkpoint
  64     ckpt_full_path = os.path.join(
  65         model_dir, 'average', 'model.ckpt-{epoch:04d}')
  66     callbacks.append(AverageModelCheckpoint(
  67         update_weights=False,
  68         filepath=ckpt_full_path,
  69         save_weights_only=True,
  70         verbose=1))
  71     callbacks.append(MovingAverageCallback())
  72   return callbacks

  75 def get_scalar_from_tensor(t: tf.Tensor) -> int:
  76   """Utility function to convert a Tensor to a scalar."""
  77   t = tf.keras.backend.get_value(t)
  78   if callable(t):
  79     return t()
  80   else:
  81     return t

  84 class CustomTensorBoard(tf.keras.callbacks.TensorBoard):
  85   """A customized TensorBoard callback that tracks additional datapoints.
  86 
  87   Metrics tracked:
  88   - Global learning rate
  89 
  90   Attributes:
  91     log_dir: the path of the directory where to save the log files to be parsed
  92       by TensorBoard.
  93     track_lr: `bool`, whether or not to track the global learning rate.
  94     initial_step: the initial step, used for preemption recovery.
  95     **kwargs: Additional arguments for backwards compatibility. Possible key is
  96       `period`.
  97   """
  98 
  99   # TODO(b/146499062): track params, flops, log lr, l2 loss,
 100   # classification loss
 101 
 102   def __init__(self,
 103                log_dir: str,
 104                track_lr: bool = False,
 105                initial_step: int = 0,
 106                **kwargs):
 107     super(CustomTensorBoard, self).__init__(log_dir=log_dir, **kwargs)
 108     self.step = initial_step
 109     self._track_lr = track_lr
 110 
 111   def on_batch_begin(self,
 112                      epoch: int,
 113                      logs: MutableMapping[str, Any] = None) -> None:
 114     self.step += 1
 115     if logs is None:
 116       logs = {}
 117     logs.update(self._calculate_metrics())
 118     super(CustomTensorBoard, self).on_batch_begin(epoch, logs)
 119 
 120   def on_epoch_begin(self,
 121                      epoch: int,
 122                      logs: MutableMapping[str, Any] = None) -> None:
 123     if logs is None:
 124       logs = {}
 125     metrics = self._calculate_metrics()
 126     logs.update(metrics)
 127     for k, v in metrics.items():
 128       logging.info('Current %s: %f', k, v)
 129     super(CustomTensorBoard, self).on_epoch_begin(epoch, logs)
 130 
 131   def on_epoch_end(self,
 132                    epoch: int,
 133                    logs: MutableMapping[str, Any] = None) -> None:
 134     if logs is None:
 135       logs = {}
 136     metrics = self._calculate_metrics()
 137     logs.update(metrics)
 138     super(CustomTensorBoard, self).on_epoch_end(epoch, logs)
 139 
 140   def _calculate_metrics(self) -> MutableMapping[str, Any]:
 141     logs = {}
 142     # TODO(b/149030439): disable LR reporting.
 143     # if self._track_lr:
 144     #   logs['learning_rate'] = self._calculate_lr()
 145     return logs
 146 
 147   def _calculate_lr(self) -> int:
 148     """Calculates the learning rate given the current step."""
 149     return get_scalar_from_tensor(
 150         self._get_base_optimizer()._decayed_lr(var_dtype=tf.float32))  # pylint:disable=protected-access
 151 
 152   def _get_base_optimizer(self) -> tf.keras.optimizers.Optimizer:
 153     """Get the base optimizer used by the current model."""
 154 
 155     optimizer = self.model.optimizer
 156 
 157     # The optimizer might be wrapped by another class, so unwrap it
 158     while hasattr(optimizer, '_optimizer'):
 159       optimizer = optimizer._optimizer  # pylint:disable=protected-access
 160 
 161     return optimizer

 164 class MovingAverageCallback(tf.keras.callbacks.Callback):
 165   """A Callback to be used with a `MovingAverage` optimizer.
 166 
 167   Applies moving average weights to the model during validation time to test
 168   and predict on the averaged weights rather than the current model weights.
 169   Once training is complete, the model weights will be overwritten with the
 170   averaged weights (by default).
 171 
 172   Attributes:
 173     overwrite_weights_on_train_end: Whether to overwrite the current model
 174       weights with the averaged weights from the moving average optimizer.
 175     **kwargs: Any additional callback arguments.
 176   """
 177 
 178   def __init__(self,
 179                overwrite_weights_on_train_end: bool = False,
 180                **kwargs):
 181     super(MovingAverageCallback, self).__init__(**kwargs)
 182     self.overwrite_weights_on_train_end = overwrite_weights_on_train_end
 183 
 184   def set_model(self, model: tf.keras.Model):
 185     super(MovingAverageCallback, self).set_model(model)
 186     assert isinstance(self.model.optimizer,
 187                       optimizer_factory.MovingAverage)
 188     self.model.optimizer.shadow_copy(self.model)
 189 
 190   def on_test_begin(self, logs: MutableMapping[Text, Any] = None):
 191     self.model.optimizer.swap_weights()
 192 
 193   def on_test_end(self, logs: MutableMapping[Text, Any] = None):
 194     self.model.optimizer.swap_weights()
 195 
 196   def on_train_end(self, logs: MutableMapping[Text, Any] = None):
 197     if self.overwrite_weights_on_train_end:
 198       self.model.optimizer.assign_average_vars(self.model.variables)

 201 class AverageModelCheckpoint(tf.keras.callbacks.ModelCheckpoint):
 202   """Saves and, optionally, assigns the averaged weights.
 203 
 204   Taken from tfa.callbacks.AverageModelCheckpoint.
 205 
 206   Attributes:
 207     update_weights: If True, assign the moving average weights
 208       to the model, and save them. If False, keep the old
 209       non-averaged weights, but the saved model uses the
 210       average weights.
 211     See `tf.keras.callbacks.ModelCheckpoint` for the other args.
 212   """
 213 
 214   def __init__(
 215       self,
 216       update_weights: bool,
 217       filepath: str,
 218       monitor: str = 'val_loss',
 219       verbose: int = 0,
 220       save_best_only: bool = False,
 221       save_weights_only: bool = False,
 222       mode: str = 'auto',
 223       save_freq: str = 'epoch',
 224       **kwargs):
 225     self.update_weights = update_weights
 226     super().__init__(
 227         filepath,
 228         monitor,
 229         verbose,
 230         save_best_only,
 231         save_weights_only,
 232         mode,
 233         save_freq,
 234         **kwargs)
 235 
 236   def set_model(self, model):
 237     if not isinstance(model.optimizer, optimizer_factory.MovingAverage):
 238       raise TypeError(
 239           'AverageModelCheckpoint is only used when training'
 240           'with MovingAverage')
 241     return super().set_model(model)
 242 
 243   def _save_model(self, epoch, logs):
 244     assert isinstance(self.model.optimizer, optimizer_factory.MovingAverage)
 245 
 246     if self.update_weights:
 247       self.model.optimizer.assign_average_vars(self.model.variables)
 248       return super()._save_model(epoch, logs)
 249     else:
 250       # Note: `model.get_weights()` gives us the weights (non-ref)
 251       # whereas `model.variables` returns references to the variables.
 252       non_avg_weights = self.model.get_weights()
 253       self.model.optimizer.assign_average_vars(self.model.variables)
 254       # result is currently None, since `super._save_model` doesn't
 255       # return anything, but this may change in the future.
 256       result = super()._save_model(epoch, logs)
 257       self.model.set_weights(non_avg_weights)
 258       return result
