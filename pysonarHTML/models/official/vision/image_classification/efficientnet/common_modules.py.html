  29 @tf.keras.utils.register_keras_serializable(package='Vision')
  30 class TpuBatchNormalization(tf.keras.layers.BatchNormalization):
  31   """Cross replica batch normalization."""
  32 
  33   def __init__(self, fused: Optional[bool] = False, **kwargs):
  34     if fused in (True, None):
  35       raise ValueError('TpuBatchNormalization does not support fused=True.')
  36     super(TpuBatchNormalization, self).__init__(fused=fused, **kwargs)
  37 
  38   def _cross_replica_average(self, t: tf.Tensor, num_shards_per_group: int):
  39     """Calculates the average value of input tensor across TPU replicas."""
  40     num_shards = tpu_function.get_tpu_context().number_of_shards
  41     group_assignment = None
  42     if num_shards_per_group > 1:
  43       if num_shards % num_shards_per_group != 0:
  44         raise ValueError(
  45             'num_shards: %d mod shards_per_group: %d, should be 0' %
  46             (num_shards, num_shards_per_group))
  47       num_groups = num_shards // num_shards_per_group
  48       group_assignment = [[
  49           x for x in range(num_shards) if x // num_shards_per_group == y
  50       ] for y in range(num_groups)]
  51     return tf1.tpu.cross_replica_sum(t, group_assignment) / tf.cast(
  52         num_shards_per_group, t.dtype)
  53 
  54   def _moments(self, inputs: tf.Tensor, reduction_axes: int, keep_dims: int):
  55     """Compute the mean and variance: it overrides the original _moments."""
  56     shard_mean, shard_variance = super(TpuBatchNormalization, self)._moments(
  57         inputs, reduction_axes, keep_dims=keep_dims)
  58 
  59     num_shards = tpu_function.get_tpu_context().number_of_shards or 1
  60     if num_shards <= 8:  # Skip cross_replica for 2x2 or smaller slices.
  61       num_shards_per_group = 1
  62     else:
  63       num_shards_per_group = max(8, num_shards // 8)
  64     if num_shards_per_group > 1:
  65       # Compute variance using: Var[X]= E[X^2] - E[X]^2.
  66       shard_square_of_mean = tf.math.square(shard_mean)
  67       shard_mean_of_square = shard_variance + shard_square_of_mean
  68       group_mean = self._cross_replica_average(shard_mean, num_shards_per_group)
  69       group_mean_of_square = self._cross_replica_average(
  70           shard_mean_of_square, num_shards_per_group)
  71       group_variance = group_mean_of_square - tf.math.square(group_mean)
  72       return (group_mean, group_variance)
  73     else:
  74       return (shard_mean, shard_variance)
  75

  77 def get_batch_norm(batch_norm_type: Text) -> tf.keras.layers.BatchNormalization:
  78   """A helper to create a batch normalization getter.
  79 
  80   Args:
  81     batch_norm_type: The type of batch normalization layer implementation. `tpu`
  82      will use `TpuBatchNormalization`.
  83 
  84   Returns:
  85     An instance of `tf.keras.layers.BatchNormalization`.
  86   """
  87   if batch_norm_type == 'tpu':
  88     return TpuBatchNormalization
  89 
  90   return tf.keras.layers.BatchNormalization

  93 def count_params(model, trainable_only=True):
  94   """Returns the count of all model parameters, or just trainable ones."""
  95   if not trainable_only:
  96     return model.count_params()
  97   else:
  98     return int(np.sum([tf.keras.backend.count_params(p)
  99                        for p in model.trainable_weights]))

 102 def load_weights(model: tf.keras.Model,
 103                  model_weights_path: Text,
 104                  weights_format: Text = 'saved_model'):
 105   """Load model weights from the given file path.
 106 
 107   Args:
 108     model: the model to load weights into
 109     model_weights_path: the path of the model weights
 110     weights_format: the model weights format. One of 'saved_model', 'h5',
 111        or 'checkpoint'.
 112   """
 113   if weights_format == 'saved_model':
 114     loaded_model = tf.keras.models.load_model(model_weights_path)
 115     model.set_weights(loaded_model.get_weights())
 116   else:
 117     model.load_weights(model_weights_path)
