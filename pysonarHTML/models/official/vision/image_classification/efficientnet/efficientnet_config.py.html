  36   Attributes:

  41     loss: The configuration for loss. Defaults to a categorical cross entropy
  42       implementation.
  43     optimizer: The configuration for optimizations. Defaults to an RMSProp
  44       configuration.
  45     learning_rate: The configuration for learning rate. Defaults to an
  46       exponential configuration.
  47   """
  48   name: str = 'EfficientNet'

  50   model_params: base_config.Config = dataclasses.field(

  53           'model_weights_path': '',
  54           'weights_format': 'saved_model',
  55           'overrides': {
  56               'batch_norm': 'default',
  57               'rescale_input': True,
  58               'num_classes': 1000,
  59               'activation': 'swish',
  60               'dtype': 'float32',
  61           }
  62       })
  63   loss: base_configs.LossConfig = base_configs.LossConfig(
  64       name='categorical_crossentropy', label_smoothing=0.1)
  65   optimizer: base_configs.OptimizerConfig = base_configs.OptimizerConfig(
  66       name='rmsprop',
  67       decay=0.9,
  68       epsilon=0.001,
  69       momentum=0.9,
  70       moving_average_decay=None)
  71   learning_rate: base_configs.LearningRateConfig = base_configs.LearningRateConfig(  # pylint: disable=line-too-long
  72       name='exponential',
  73       initial_lr=0.008,
  74       decay_epochs=2.4,
  75       decay_rate=0.97,
  76       warmup_epochs=5,
  77       scale_by_batch_size=1. / 128.,
  78       staircase=True)
