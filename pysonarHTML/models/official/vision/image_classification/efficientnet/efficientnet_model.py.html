 131 def round_filters(filters: int,
 132                   config: ModelConfig) -> int:
 133   """Round number of filters based on width coefficient."""
 134   width_coefficient = config.width_coefficient
 135   min_depth = config.min_depth
 136   divisor = config.depth_divisor
 137   orig_filters = filters
 138 
 139   if not width_coefficient:
 140     return filters
 141 
 142   filters *= width_coefficient
 143   min_depth = min_depth or divisor
 144   new_filters = max(min_depth, int(filters + divisor / 2) // divisor * divisor)
 145   # Make sure that round down does not go down by more than 10%.
 146   if new_filters < 0.9 * filters:
 147     new_filters += divisor
 148   logging.info('round_filter input=%s output=%s', orig_filters, new_filters)
 149   return int(new_filters)

 152 def round_repeats(repeats: int, depth_coefficient: float) -> int:
 153   """Round number of repeats based on depth coefficient."""
 154   return int(math.ceil(depth_coefficient * repeats))

 157 def conv2d_block(inputs: tf.Tensor,
 158                  conv_filters: Optional[int],
 159                  config: ModelConfig,
 160                  kernel_size: Any = (1, 1),
 161                  strides: Any = (1, 1),
 162                  use_batch_norm: bool = True,
 163                  use_bias: bool = False,
 164                  activation: Any = None,
 165                  depthwise: bool = False,
 166                  name: Text = None):
 167   """A conv2d followed by batch norm and an activation."""
 168   batch_norm = common_modules.get_batch_norm(config.batch_norm)
 169   bn_momentum = config.bn_momentum
 170   bn_epsilon = config.bn_epsilon
 171   data_format = tf.keras.backend.image_data_format()
 172   weight_decay = config.weight_decay
 173 
 174   name = name or ''
 175 
 176   # Collect args based on what kind of conv2d block is desired
 177   init_kwargs = {
 178       'kernel_size': kernel_size,
 179       'strides': strides,
 180       'use_bias': use_bias,
 181       'padding': 'same',
 182       'name': name + '_conv2d',
 183       'kernel_regularizer': tf.keras.regularizers.l2(weight_decay),
 184       'bias_regularizer': tf.keras.regularizers.l2(weight_decay),
 185   }
 186 
 187   if depthwise:
 188     conv2d = tf.keras.layers.DepthwiseConv2D
 189     init_kwargs.update({'depthwise_initializer': CONV_KERNEL_INITIALIZER})
 190   else:
 191     conv2d = tf.keras.layers.Conv2D
 192     init_kwargs.update({'filters': conv_filters,
 193                         'kernel_initializer': CONV_KERNEL_INITIALIZER})
 194 
 195   x = conv2d(**init_kwargs)(inputs)
 196 
 197   if use_batch_norm:
 198     bn_axis = 1 if data_format == 'channels_first' else -1
 199     x = batch_norm(axis=bn_axis,
 200                    momentum=bn_momentum,
 201                    epsilon=bn_epsilon,
 202                    name=name + '_bn')(x)
 203 
 204   if activation is not None:
 205     x = tf.keras.layers.Activation(activation,
 206                                    name=name + '_activation')(x)
 207   return x

 210 def mb_conv_block(inputs: tf.Tensor,
 211                   block: BlockConfig,
 212                   config: ModelConfig,
 213                   prefix: Text = None):
 214   """Mobile Inverted Residual Bottleneck.
 215 
 216   Args:
 217     inputs: the Keras input to the block
 218     block: BlockConfig, arguments to create a Block
 219     config: ModelConfig, a set of model parameters
 220     prefix: prefix for naming all layers
 221 
 222   Returns:
 223     the output of the block
 224   """
 225   use_se = config.use_se
 226   activation = tf_utils.get_activation(config.activation)
 227   drop_connect_rate = config.drop_connect_rate
 228   data_format = tf.keras.backend.image_data_format()
 229   use_depthwise = block.conv_type != 'no_depthwise'
 230   prefix = prefix or ''
 231 
 232   filters = block.input_filters * block.expand_ratio
 233 
 234   x = inputs
 235 
 236   if block.fused_conv:
 237     # If we use fused mbconv, skip expansion and use regular conv.
 238     x = conv2d_block(x,
 239                      filters,
 240                      config,
 241                      kernel_size=block.kernel_size,
 242                      strides=block.strides,
 243                      activation=activation,
 244                      name=prefix + 'fused')
 245   else:
 246     if block.expand_ratio != 1:
 247       # Expansion phase
 248       kernel_size = (1, 1) if use_depthwise else (3, 3)
 249       x = conv2d_block(x,
 250                        filters,
 251                        config,
 252                        kernel_size=kernel_size,
 253                        activation=activation,
 254                        name=prefix + 'expand')
 255 
 256     # Depthwise Convolution
 257     if use_depthwise:
 258       x = conv2d_block(x,
 259                        conv_filters=None,
 260                        config=config,
 261                        kernel_size=block.kernel_size,
 262                        strides=block.strides,
 263                        activation=activation,
 264                        depthwise=True,
 265                        name=prefix + 'depthwise')
 266 
 267   # Squeeze and Excitation phase
 268   if use_se:
 269     assert block.se_ratio is not None
 270     assert 0 < block.se_ratio <= 1
 271     num_reduced_filters = max(1, int(
 272         block.input_filters * block.se_ratio
 273     ))
 274 
 275     if data_format == 'channels_first':
 276       se_shape = (filters, 1, 1)
 277     else:
 278       se_shape = (1, 1, filters)
 279 
 280     se = tf.keras.layers.GlobalAveragePooling2D(name=prefix + 'se_squeeze')(x)
 281     se = tf.keras.layers.Reshape(se_shape, name=prefix + 'se_reshape')(se)
 282 
 283     se = conv2d_block(se,
 284                       num_reduced_filters,
 285                       config,
 286                       use_bias=True,
 287                       use_batch_norm=False,
 288                       activation=activation,
 289                       name=prefix + 'se_reduce')
 290     se = conv2d_block(se,
 291                       filters,
 292                       config,
 293                       use_bias=True,
 294                       use_batch_norm=False,
 295                       activation='sigmoid',
 296                       name=prefix + 'se_expand')
 297     x = tf.keras.layers.multiply([x, se], name=prefix + 'se_excite')
 298 
 299   # Output phase
 300   x = conv2d_block(x,
 301                    block.output_filters,
 302                    config,
 303                    activation=None,
 304                    name=prefix + 'project')
 305 
 306   # Add identity so that quantization-aware training can insert quantization
 307   # ops correctly.
 308   x = tf.keras.layers.Activation(tf_utils.get_activation('identity'),
 309                                  name=prefix + 'id')(x)
 310 
 311   if (block.id_skip
 312       and all(s == 1 for s in block.strides)
 313       and block.input_filters == block.output_filters):
 314     if drop_connect_rate and drop_connect_rate > 0:
 315       # Apply dropconnect
 316       # The only difference between dropout and dropconnect in TF is scaling by
 317       # drop_connect_rate during training. See:
 318       # https://github.com/keras-team/keras/pull/9898#issuecomment-380577612
 319       x = tf.keras.layers.Dropout(drop_connect_rate,
 320                                   noise_shape=(None, 1, 1, 1),
 321                                   name=prefix + 'drop')(x)
 322 
 323     x = tf.keras.layers.add([x, inputs], name=prefix + 'add')
 324 
 325   return x

 328 def efficientnet(image_input: tf.keras.layers.Input,
 329                  config: ModelConfig):
 330   """Creates an EfficientNet graph given the model parameters.
 331 
 332   This function is wrapped by the `EfficientNet` class to make a tf.keras.Model.
 333 
 334   Args:
 335     image_input: the input batch of images
 336     config: the model config
 337 
 338   Returns:
 339     the output of efficientnet
 340   """
 341   depth_coefficient = config.depth_coefficient
 342   blocks = config.blocks
 343   stem_base_filters = config.stem_base_filters
 344   top_base_filters = config.top_base_filters
 345   activation = tf_utils.get_activation(config.activation)
 346   dropout_rate = config.dropout_rate
 347   drop_connect_rate = config.drop_connect_rate
 348   num_classes = config.num_classes
 349   input_channels = config.input_channels
 350   rescale_input = config.rescale_input
 351   data_format = tf.keras.backend.image_data_format()
 352   dtype = config.dtype
 353   weight_decay = config.weight_decay
 354 
 355   x = image_input
 356   if data_format == 'channels_first':
 357     # Happens on GPU/TPU if available.
 358     x = tf.keras.layers.Permute((3, 1, 2))(x)
 359   if rescale_input:
 360     x = preprocessing.normalize_images(x,
 361                                        num_channels=input_channels,
 362                                        dtype=dtype,
 363                                        data_format=data_format)
 364 
 365   # Build stem
 366   x = conv2d_block(x,
 367                    round_filters(stem_base_filters, config),
 368                    config,
 369                    kernel_size=[3, 3],
 370                    strides=[2, 2],
 371                    activation=activation,
 372                    name='stem')
 373 
 374   # Build blocks
 375   num_blocks_total = sum(
 376       round_repeats(block.num_repeat, depth_coefficient) for block in blocks)
 377   block_num = 0
 378 
 379   for stack_idx, block in enumerate(blocks):
 380     assert block.num_repeat > 0
 381     # Update block input and output filters based on depth multiplier
 382     block = block.replace(
 383         input_filters=round_filters(block.input_filters, config),
 384         output_filters=round_filters(block.output_filters, config),
 385         num_repeat=round_repeats(block.num_repeat, depth_coefficient))
 386 
 387     # The first block needs to take care of stride and filter size increase
 388     drop_rate = drop_connect_rate * float(block_num) / num_blocks_total
 389     config = config.replace(drop_connect_rate=drop_rate)
 390     block_prefix = 'stack_{}/block_0/'.format(stack_idx)
 391     x = mb_conv_block(x, block, config, block_prefix)
 392     block_num += 1
 393     if block.num_repeat > 1:
 394       block = block.replace(
 395           input_filters=block.output_filters,
 396           strides=[1, 1]
 397       )
 398 
 399       for block_idx in range(block.num_repeat - 1):
 400         drop_rate = drop_connect_rate * float(block_num) / num_blocks_total
 401         config = config.replace(drop_connect_rate=drop_rate)
 402         block_prefix = 'stack_{}/block_{}/'.format(stack_idx, block_idx + 1)
 403         x = mb_conv_block(x, block, config, prefix=block_prefix)
 404         block_num += 1
 405 
 406   # Build top
 407   x = conv2d_block(x,
 408                    round_filters(top_base_filters, config),
 409                    config,
 410                    activation=activation,
 411                    name='top')
 412 
 413   # Build classifier
 414   x = tf.keras.layers.GlobalAveragePooling2D(name='top_pool')(x)
 415   if dropout_rate and dropout_rate > 0:
 416     x = tf.keras.layers.Dropout(dropout_rate, name='top_dropout')(x)
 417   x = tf.keras.layers.Dense(
 418       num_classes,
 419       kernel_initializer=DENSE_KERNEL_INITIALIZER,
 420       kernel_regularizer=tf.keras.regularizers.l2(weight_decay),
 421       bias_regularizer=tf.keras.regularizers.l2(weight_decay),
 422       name='logits')(x)
 423   x = tf.keras.layers.Activation('softmax', name='probs')(x)
 424 
 425   return x

 428 @tf.keras.utils.register_keras_serializable(package='Vision')
 429 class EfficientNet(tf.keras.Model):
 430   """Wrapper class for an EfficientNet Keras model.
 431 
 432   Contains helper methods to build, manage, and save metadata about the model.
 433   """
 434 
 435   def __init__(self,
 436                config: ModelConfig = None,
 437                overrides: Dict[Text, Any] = None):
 438     """Create an EfficientNet model.
 439 
 440     Args:
 441       config: (optional) the main model parameters to create the model
 442       overrides: (optional) a dict containing keys that can override
 443                  config
 444     """
 445     overrides = overrides or {}
 446     config = config or ModelConfig()
 447 
 448     self.config = config.replace(**overrides)
 449 
 450     input_channels = self.config.input_channels
 451     model_name = self.config.model_name
 452     input_shape = (None, None, input_channels)  # Should handle any size image
 453     image_input = tf.keras.layers.Input(shape=input_shape)
 454 
 455     output = efficientnet(image_input, self.config)
 456 
 457     # Cast to float32 in case we have a different model dtype
 458     output = tf.cast(output, tf.float32)
 459 
 460     logging.info('Building model %s with params %s',
 461                  model_name,
 462                  self.config)
 463 
 464     super(EfficientNet, self).__init__(
 465         inputs=image_input, outputs=output, name=model_name)
 466 
 467   @classmethod
 468   def from_name(cls,
 469                 model_name: Text,
 470                 model_weights_path: Text = None,
 471                 weights_format: Text = 'saved_model',
 472                 overrides: Dict[Text, Any] = None):
 473     """Construct an EfficientNet model from a predefined model name.
 474 
 475     E.g., `EfficientNet.from_name('efficientnet-b0')`.
 476 
 477     Args:
 478       model_name: the predefined model name
 479       model_weights_path: the path to the weights (h5 file or saved model dir)
 480       weights_format: the model weights format. One of 'saved_model', 'h5',
 481        or 'checkpoint'.
 482       overrides: (optional) a dict containing keys that can override config
 483 
 484     Returns:
 485       A constructed EfficientNet instance.
 486     """
 487     model_configs = dict(MODEL_CONFIGS)
 488     overrides = dict(overrides) if overrides else {}
 489 
 490     # One can define their own custom models if necessary
 491     model_configs.update(overrides.pop('model_config', {}))
 492 
 493     if model_name not in model_configs:
 494       raise ValueError('Unknown model name {}'.format(model_name))
 495 
 496     config = model_configs[model_name]
 497 
 498     model = cls(config=config, overrides=overrides)
 499 
 500     if model_weights_path:
 501       common_modules.load_weights(model,
 502                                   model_weights_path,
 503                                   weights_format=weights_format)
 504 
 505     return model
