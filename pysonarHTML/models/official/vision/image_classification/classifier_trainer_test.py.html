  56           'efficientnet',

  59       mode='eager',

 102 def get_trivial_model(num_classes: int) -> tf.keras.Model:
 103   """Creates and compiles trivial model for ImageNet dataset."""
 104   model = test_utils.trivial_model(num_classes=num_classes)
 105   lr = 0.01
 106   optimizer = tf.keras.optimizers.SGD(learning_rate=lr)
 107   loss_obj = tf.keras.losses.SparseCategoricalCrossentropy()
 108   model.compile(optimizer=optimizer,
 109                 loss=loss_obj,
 110                 run_eagerly=True)
 111   return model

 114 def get_trivial_data() -> tf.data.Dataset:
 115   """Gets trivial data in the ImageNet size."""
 116   def generate_data(_) -> tf.data.Dataset:
 117     image = tf.zeros(shape=(224, 224, 3), dtype=tf.float32)
 118     label = tf.zeros([1], dtype=tf.int32)
 119     return image, label
 120 
 121   dataset = tf.data.Dataset.range(1)
 122   dataset = dataset.repeat()
 123   dataset = dataset.map(generate_data,
 124                         num_parallel_calls=tf.data.experimental.AUTOTUNE)
 125   dataset = dataset.prefetch(buffer_size=1).batch(1)
 126   return dataset

 129 def run_end_to_end(main: Callable[[Any], None],
 130                    extra_flags: Optional[Iterable[str]] = None,
 131                    model_dir: Optional[str] = None):
 132   """Runs the classifier trainer end-to-end."""
 133   extra_flags = [] if extra_flags is None else extra_flags
 134   args = [sys.argv[0], '--model_dir', model_dir] + extra_flags
 135   flags_core.parse_flags(argv=args)
 136   main(flags.FLAGS)

 151   @combinations.generate(distribution_strategy_combinations())
 152   def test_end_to_end_train_and_eval(self, distribution, model, dataset):
 153     """Test train_and_eval and export for Keras classifier models."""
 154     # Some parameters are not defined as flags (e.g. cannot run
 155     # classifier_train.py --batch_size=...) by design, so use
 156     # "--params_override=..." instead
 157     model_dir = self.get_temp_dir()
 158     base_flags = [
 159         '--data_dir=not_used',
 160         '--model_type=' + model,
 161         '--dataset=' + dataset,
 162     ]
 163     train_and_eval_flags = base_flags + [
 164         get_params_override(basic_params_override()),
 165         '--mode=train_and_eval',
 166     ]
 167 
 168     run = functools.partial(classifier_trainer.run,
 169                             strategy_override=distribution)
 170     run_end_to_end(main=run,
 171                    extra_flags=train_and_eval_flags,
 172                    model_dir=model_dir)

 216                    model_dir=model_dir)

 218                    extra_flags=export_flags,
 219                    model_dir=model_dir)

 340   def test_resume_from_checkpoint(self):
 341     """Tests functionality for resuming from checkpoint."""
 342     # Set the keras policy
 343     policy = tf.keras.mixed_precision.experimental.Policy('mixed_bfloat16')
 344     tf.keras.mixed_precision.experimental.set_policy(policy)
 345 
 346     # Get the model, datasets, and compile it.
 347     model = get_trivial_model(10)
 348 
 349     # Create the checkpoint
 350     model_dir = self.get_temp_dir()
 351     train_epochs = 1
 352     train_steps = 10
 353     ds = get_trivial_data()
 354     callbacks = [
 355         tf.keras.callbacks.ModelCheckpoint(
 356             os.path.join(model_dir, 'model.ckpt-{epoch:04d}'),
 357             save_weights_only=True)
 358     ]
 359     model.fit(
 360         ds,
 361         callbacks=callbacks,
 362         epochs=train_epochs,
 363         steps_per_epoch=train_steps)
 364 
 365     # Test load from checkpoint
 366     clean_model = get_trivial_model(10)
 367     weights_before_load = copy.deepcopy(clean_model.get_weights())
 368     initial_epoch = classifier_trainer.resume_from_checkpoint(
 369         model=clean_model,
 370         model_dir=model_dir,
 371         train_steps=train_steps)
 372     self.assertEqual(initial_epoch, 1)
 373     self.assertNotAllClose(weights_before_load, clean_model.get_weights())
 374 
 375     tf.io.gfile.rmtree(model_dir)
