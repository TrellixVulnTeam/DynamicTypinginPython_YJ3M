  61 def _get_metrics(one_hot: bool) -> Mapping[Text, Any]:
  62   """Get a dict of available metrics to track."""
  63   if one_hot:
  64     return {
  65         # (name, metric_fn)
  66         'acc': tf.keras.metrics.CategoricalAccuracy(name='accuracy'),
  67         'accuracy': tf.keras.metrics.CategoricalAccuracy(name='accuracy'),
  68         'top_1': tf.keras.metrics.CategoricalAccuracy(name='accuracy'),
  69         'top_5': tf.keras.metrics.TopKCategoricalAccuracy(
  70             k=5,
  71             name='top_5_accuracy'),
  72     }
  73   else:
  74     return {
  75         # (name, metric_fn)
  76         'acc': tf.keras.metrics.SparseCategoricalAccuracy(name='accuracy'),
  77         'accuracy': tf.keras.metrics.SparseCategoricalAccuracy(name='accuracy'),
  78         'top_1': tf.keras.metrics.SparseCategoricalAccuracy(name='accuracy'),
  79         'top_5': tf.keras.metrics.SparseTopKCategoricalAccuracy(
  80             k=5,
  81             name='top_5_accuracy'),

  85 def get_image_size_from_model(
  86     params: base_configs.ExperimentConfig) -> Optional[int]:
  87   """If the given model has a preferred image size, return it."""
  88   if params.model_name == 'efficientnet':
  89     efficientnet_name = params.model.model_params.model_name
  90     if efficientnet_name in efficientnet_model.MODEL_CONFIGS:
  91       return efficientnet_model.MODEL_CONFIGS[efficientnet_name].resolution
  92   return None

  95 def _get_dataset_builders(params: base_configs.ExperimentConfig,
  96                           strategy: tf.distribute.Strategy,
  97                           one_hot: bool
  98                          ) -> Tuple[Any, Any]:
  99   """Create and return train and validation dataset builders."""
 100   if one_hot:
 101     logging.warning('label_smoothing > 0, so datasets will be one hot encoded.')
 102   else:
 103     logging.warning('label_smoothing not applied, so datasets will not be one '
 104                     'hot encoded.')
 105 
 106   num_devices = strategy.num_replicas_in_sync if strategy else 1
 107 
 108   image_size = get_image_size_from_model(params)
 109 
 110   dataset_configs = [
 111       params.train_dataset, params.validation_dataset
 112   ]
 113   builders = []
 114 
 115   for config in dataset_configs:
 116     if config is not None and config.has_data:
 117       builder = dataset_factory.DatasetBuilder(
 118           config,
 119           image_size=image_size or config.image_size,
 120           num_devices=num_devices,
 121           one_hot=one_hot)
 122     else:
 123       builder = None
 124     builders.append(builder)
 125 
 126   return builders

 129 def get_loss_scale(params: base_configs.ExperimentConfig,
 130                    fp16_default: float = 128.) -> float:
 131   """Returns the loss scale for initializations."""
 132   loss_scale = params.runtime.loss_scale
 133   if loss_scale == 'dynamic':
 134     return loss_scale
 135   elif loss_scale is not None:
 136     return float(loss_scale)
 137   elif (params.train_dataset.dtype == 'float32' or
 138         params.train_dataset.dtype == 'bfloat16'):
 139     return 1.
 140   else:
 141     assert params.train_dataset.dtype == 'float16'
 142     return fp16_default

 145 def _get_params_from_flags(flags_obj: flags.FlagValues):
 146   """Get ParamsDict from flags."""
 147   model = flags_obj.model_type.lower()
 148   dataset = flags_obj.dataset.lower()
 149   params = configs.get_config(model=model, dataset=dataset)
 150 
 151   flags_overrides = {
 152       'model_dir': flags_obj.model_dir,
 153       'mode': flags_obj.mode,
 154       'model': {
 155           'name': model,
 156       },
 157       'runtime': {
 158           'run_eagerly': flags_obj.run_eagerly,
 159           'tpu': flags_obj.tpu,
 160       },
 161       'train_dataset': {
 162           'data_dir': flags_obj.data_dir,
 163       },
 164       'validation_dataset': {
 165           'data_dir': flags_obj.data_dir,
 166       },
 167       'train': {
 168           'time_history': {
 169               'log_steps': flags_obj.log_steps,
 170           },
 171       },
 172   }
 173 
 174   overriding_configs = (flags_obj.config_file,
 175                         flags_obj.params_override,
 176                         flags_overrides)
 177 
 178   pp = pprint.PrettyPrinter()
 179 
 180   logging.info('Base params: %s', pp.pformat(params.as_dict()))
 181 
 182   for param in overriding_configs:
 183     logging.info('Overriding params: %s', param)
 184     params = hyperparams.override_params_dict(params, param, is_strict=True)
 185 
 186   params.validate()
 187   params.lock()
 188 
 189   logging.info('Final model parameters: %s', pp.pformat(params.as_dict()))
 190   return params

 193 def resume_from_checkpoint(model: tf.keras.Model,
 194                            model_dir: str,
 195                            train_steps: int) -> int:
 196   """Resumes from the latest checkpoint, if possible.
 197 
 198   Loads the model weights and optimizer settings from a checkpoint.
 199   This function should be used in case of preemption recovery.
 200 
 201   Args:
 202     model: The model whose weights should be restored.
 203     model_dir: The directory where model weights were saved.
 204     train_steps: The number of steps to train.
 205 
 206   Returns:
 207     The epoch of the latest checkpoint, or 0 if not restoring.
 208 
 209   """
 210   logging.info('Load from checkpoint is enabled.')
 211   latest_checkpoint = tf.train.latest_checkpoint(model_dir)
 212   logging.info('latest_checkpoint: %s', latest_checkpoint)
 213   if not latest_checkpoint:
 214     logging.info('No checkpoint detected.')
 215     return 0
 216 
 217   logging.info('Checkpoint file %s found and restoring from '
 218                'checkpoint', latest_checkpoint)
 219   model.load_weights(latest_checkpoint)
 220   initial_epoch = model.optimizer.iterations // train_steps
 221   logging.info('Completed loading from checkpoint.')
 222   logging.info('Resuming from epoch %d', initial_epoch)
 223   return int(initial_epoch)

 226 def initialize(params: base_configs.ExperimentConfig,
 227                dataset_builder: dataset_factory.DatasetBuilder):
 228   """Initializes backend related initializations."""
 229   keras_utils.set_session_config(
 230       enable_xla=params.runtime.enable_xla)
 231   performance.set_mixed_precision_policy(dataset_builder.dtype,
 232                                          get_loss_scale(params))
 233   if tf.config.list_physical_devices('GPU'):
 234     data_format = 'channels_first'
 235   else:
 236     data_format = 'channels_last'
 237   tf.keras.backend.set_image_data_format(data_format)
 238   distribution_utils.configure_cluster(
 239       params.runtime.worker_hosts,
 240       params.runtime.task_index)
 241   if params.runtime.run_eagerly:
 242     # Enable eager execution to allow step-by-step debugging
 243     tf.config.experimental_run_functions_eagerly(True)
 244   if tf.config.list_physical_devices('GPU'):
 245     if params.runtime.gpu_thread_mode:
 246       keras_utils.set_gpu_thread_mode_and_count(
 247           per_gpu_thread_count=params.runtime.per_gpu_thread_count,
 248           gpu_thread_mode=params.runtime.gpu_thread_mode,
 249           num_gpus=params.runtime.num_gpus,
 250           datasets_num_private_threads=params.runtime.dataset_num_private_threads)  # pylint:disable=line-too-long
 251     if params.runtime.batchnorm_spatial_persistent:
 252       os.environ['TF_USE_CUDNN_BATCHNORM_SPATIAL_PERSISTENT'] = '1'

 256   """Defines common flags for image classification."""
 257   hyperparams_flags.initialize_common_flags()
 258   flags.DEFINE_string(
 259       'data_dir',
 260       default=None,
 261       help='The location of the input data.')
 262   flags.DEFINE_string(
 263       'mode',
 264       default=None,
 265       help='Mode to run: `train`, `eval`, `train_and_eval` or `export`.')
 266   flags.DEFINE_bool(
 267       'run_eagerly',
 268       default=None,
 269       help='Use eager execution and disable autograph for debugging.')
 270   flags.DEFINE_string(
 271       'model_type',
 272       default=None,
 273       help='The type of the model, e.g. EfficientNet, etc.')
 274   flags.DEFINE_string(
 275       'dataset',
 276       default=None,
 277       help='The name of the dataset, e.g. ImageNet, etc.')
 278   flags.DEFINE_integer(
 279       'log_steps',
 280       default=100,
 281       help='The interval of steps between logging of batch level stats.')

 293 def train_and_eval(
 294     params: base_configs.ExperimentConfig,
 295     strategy_override: tf.distribute.Strategy) -> Mapping[str, Any]:
 296   """Runs the train and eval path using compile/fit."""
 297   logging.info('Running train and eval.')
 298 
 299   # Note: for TPUs, strategy and scope should be created before the dataset
 300   strategy = strategy_override or distribution_utils.get_distribution_strategy(
 301       distribution_strategy=params.runtime.distribution_strategy,
 302       all_reduce_alg=params.runtime.all_reduce_alg,
 303       num_gpus=params.runtime.num_gpus,
 304       tpu_address=params.runtime.tpu)
 305 
 306   strategy_scope = distribution_utils.get_strategy_scope(strategy)
 307 
 308   logging.info('Detected %d devices.',
 309                strategy.num_replicas_in_sync if strategy else 1)
 310 
 311   label_smoothing = params.model.loss.label_smoothing
 312   one_hot = label_smoothing and label_smoothing > 0
 313 
 314   builders = _get_dataset_builders(params, strategy, one_hot)
 315   datasets = [builder.build(strategy)
 316               if builder else None for builder in builders]
 317 
 318   # Unpack datasets and builders based on train/val/test splits
 319   train_builder, validation_builder = builders  # pylint: disable=unbalanced-tuple-unpacking
 320   train_dataset, validation_dataset = datasets
 321 
 322   train_epochs = params.train.epochs
 323   train_steps = params.train.steps or train_builder.num_steps
 324   validation_steps = params.evaluation.steps or validation_builder.num_steps
 325 
 326   initialize(params, train_builder)
 327 
 328   logging.info('Global batch size: %d', train_builder.global_batch_size)
 329 
 330   with strategy_scope:
 331     model_params = params.model.model_params.as_dict()
 332     model = get_models()[params.model.name](**model_params)
 333     learning_rate = optimizer_factory.build_learning_rate(
 334         params=params.model.learning_rate,
 335         batch_size=train_builder.global_batch_size,
 336         train_epochs=train_epochs,
 337         train_steps=train_steps)
 338     optimizer = optimizer_factory.build_optimizer(
 339         optimizer_name=params.model.optimizer.name,
 340         base_learning_rate=learning_rate,
 341         params=params.model.optimizer.as_dict())
 342 
 343     metrics_map = _get_metrics(one_hot)
 344     metrics = [metrics_map[metric] for metric in params.train.metrics]
 345     steps_per_loop = train_steps if params.train.set_epoch_loop else 1
 346 
 347     if one_hot:
 348       loss_obj = tf.keras.losses.CategoricalCrossentropy(
 349           label_smoothing=params.model.loss.label_smoothing)
 350     else:
 351       loss_obj = tf.keras.losses.SparseCategoricalCrossentropy()
 352     model.compile(optimizer=optimizer,
 353                   loss=loss_obj,
 354                   metrics=metrics,
 355                   experimental_steps_per_execution=steps_per_loop)
 356 
 357     initial_epoch = 0
 358     if params.train.resume_checkpoint:
 359       initial_epoch = resume_from_checkpoint(model=model,
 360                                              model_dir=params.model_dir,
 361                                              train_steps=train_steps)
 362 
 363     callbacks = custom_callbacks.get_callbacks(
 364         model_checkpoint=params.train.callbacks.enable_checkpoint_and_export,
 365         include_tensorboard=params.train.callbacks.enable_tensorboard,
 366         time_history=params.train.callbacks.enable_time_history,
 367         track_lr=params.train.tensorboard.track_lr,
 368         write_model_weights=params.train.tensorboard.write_model_weights,
 369         initial_step=initial_epoch * train_steps,
 370         batch_size=train_builder.global_batch_size,
 371         log_steps=params.train.time_history.log_steps,
 372         model_dir=params.model_dir)
 373 
 374   serialize_config(params=params, model_dir=params.model_dir)
 375 
 376   if params.evaluation.skip_eval:
 377     validation_kwargs = {}
 378   else:
 379     validation_kwargs = {
 380         'validation_data': validation_dataset,
 381         'validation_steps': validation_steps,
 382         'validation_freq': params.evaluation.epochs_between_evals,
 383     }
 384 
 385   history = model.fit(
 386       train_dataset,
 387       epochs=train_epochs,
 388       steps_per_epoch=train_steps,
 389       initial_epoch=initial_epoch,
 390       callbacks=callbacks,
 391       verbose=2,
 392       **validation_kwargs)
 393 
 394   validation_output = None
 395   if not params.evaluation.skip_eval:
 396     validation_output = model.evaluate(
 397         validation_dataset, steps=validation_steps, verbose=2)
 398 
 399   # TODO(dankondratyuk): eval and save final test accuracy
 400   stats = common.build_stats(history,
 401                              validation_output,
 402                              callbacks)
 403   return stats

 406 def export(params: base_configs.ExperimentConfig):
 407   """Runs the model export functionality."""
 408   logging.info('Exporting model.')
 409   model_params = params.model.model_params.as_dict()
 410   model = get_models()[params.model.name](**model_params)
 411   checkpoint = params.export.checkpoint
 412   if checkpoint is None:
 413     logging.info('No export checkpoint was provided. Using the latest '
 414                  'checkpoint from model_dir.')
 415     checkpoint = tf.train.latest_checkpoint(params.model_dir)
 416 
 417   model.load_weights(checkpoint)
 418   model.save(params.export.destination)

 421 def run(flags_obj: flags.FlagValues,
 422         strategy_override: tf.distribute.Strategy = None) -> Mapping[str, Any]:
 423   """Runs Image Classification model using native Keras APIs.
 424 
 425   Args:
 426     flags_obj: An object containing parsed flag values.
 427     strategy_override: A `tf.distribute.Strategy` object to use for model.
 428 
 429   Returns:
 430     Dictionary of training/eval stats
 431   """
 432   params = _get_params_from_flags(flags_obj)
 433   if params.mode == 'train_and_eval':
 434     return train_and_eval(params, strategy_override)
 435   elif params.mode == 'export_only':
 436     export(params)
 437   else:
 438     raise ValueError('{} is not a valid mode.'.format(params.mode))
