  32 class MovingAverage(tf.keras.optimizers.Optimizer):
  33   """Optimizer that computes a moving average of the variables.
  34 
  35   Empirically it has been found that using the moving average of the trained
  36   parameters of a deep network is better than using its trained parameters
  37   directly. This optimizer allows you to compute this moving average and swap
  38   the variables at save time so that any code outside of the training loop
  39   will use by default the average values instead of the original ones.
  40 
  41   Example of usage for training:
  42   ```python
  43   opt = tf.keras.optimizers.SGD(learning_rate)
  44   opt = MovingAverage(opt)
  45 
  46   opt.shadow_copy(model)
  47   ```
  48 
  49   At test time, swap the shadow variables to evaluate on the averaged weights:
  50   ```python
  51   opt.swap_weights()
  52   # Test eval the model here
  53   opt.swap_weights()
  54   ```
  55   """
  56 
  57   def __init__(self,
  58                optimizer: tf.keras.optimizers.Optimizer,
  59                average_decay: float = 0.99,
  60                start_step: int = 0,
  61                dynamic_decay: bool = True,
  62                name: Text = 'moving_average',
  63                **kwargs):
  64     """Construct a new MovingAverage optimizer.
  65 
  66     Args:
  67       optimizer: `tf.keras.optimizers.Optimizer` that will be
  68         used to compute and apply gradients.
  69       average_decay: float. Decay to use to maintain the moving averages
  70         of trained variables.
  71       start_step: int. What step to start the moving average.
  72       dynamic_decay: bool. Whether to change the decay based on the number
  73         of optimizer updates. Decay will start at 0.1 and gradually increase
  74         up to `average_decay` after each optimizer update. This behavior is
  75         similar to `tf.train.ExponentialMovingAverage` in TF 1.x.
  76       name: Optional name for the operations created when applying
  77         gradients. Defaults to "moving_average".
  78       **kwargs: keyword arguments. Allowed to be {`clipnorm`,
  79         `clipvalue`, `lr`, `decay`}.
  80     """
  81     super(MovingAverage, self).__init__(name, **kwargs)
  82     self._optimizer = optimizer
  83     self._average_decay = average_decay
  84     self._start_step = tf.constant(start_step, tf.float32)
  85     self._dynamic_decay = dynamic_decay
  86 
  87   def shadow_copy(self, model: tf.keras.Model):
  88     """Creates shadow variables for the given model weights."""
  89     for var in model.weights:
  90       self.add_slot(var, 'average', initializer='zeros')
  91     self._average_weights = [
  92         self.get_slot(var, 'average') for var in model.weights
  93     ]
  94     self._model_weights = model.weights
  95 
  96   @property
  97   def has_shadow_copy(self):
  98     """Whether this optimizer has created shadow variables."""
  99     return self._model_weights is not None
 100 
 101   def _create_slots(self, var_list):
 102     self._optimizer._create_slots(var_list=var_list)  # pylint: disable=protected-access
 103 
 104   def apply_gradients(self, grads_and_vars, name: Text = None):
 105     result = self._optimizer.apply_gradients(grads_and_vars, name)
 106     self.update_average(self._optimizer.iterations)
 107     return result
 108 
 109   @tf.function
 110   def update_average(self, step: tf.Tensor):
 111     step = tf.cast(step, tf.float32)
 112     if step < self._start_step:
 113       decay = tf.constant(0., tf.float32)
 114     elif self._dynamic_decay:
 115       decay = step - self._start_step
 116       decay = tf.minimum(self._average_decay, (1. + decay) / (10. + decay))
 117     else:
 118       decay = self._average_decay
 119 
 120     def _apply_moving(v_moving, v_normal):
 121       diff = v_moving - v_normal
 122       v_moving.assign_sub(tf.cast(1. - decay, v_moving.dtype) * diff)
 123       return v_moving
 124 
 125     def _update(strategy, v_moving_and_v_normal):
 126       for v_moving, v_normal in v_moving_and_v_normal:
 127         strategy.extended.update(v_moving, _apply_moving, args=(v_normal,))
 128 
 129     ctx = tf.distribute.get_replica_context()
 130     return ctx.merge_call(_update, args=(zip(self._average_weights,
 131                                              self._model_weights),))
 132 
 133   def swap_weights(self):
 134     """Swap the average and moving weights.
 135 
 136     This is a convenience method to allow one to evaluate the averaged weights
 137     at test time. Loads the weights stored in `self._average` into the model,
 138     keeping a copy of the original model weights. Swapping twice will return
 139     the original weights.
 140     """
 141     if tf.distribute.in_cross_replica_context():
 142       strategy = tf.distribute.get_strategy()
 143       strategy.run(self._swap_weights, args=())
 144     else:
 145       raise ValueError('Swapping weights must occur under a '
 146                        'tf.distribute.Strategy')
 147 
 148   @tf.function
 149   def _swap_weights(self):
 150     def fn_0(a, b):
 151       a.assign_add(b)
 152       return a
 153     def fn_1(b, a):
 154       b.assign(a - b)
 155       return b
 156     def fn_2(a, b):
 157       a.assign_sub(b)
 158       return a
 159 
 160     def swap(strategy, a_and_b):
 161       """Swap `a` and `b` and mirror to all devices."""
 162       for a, b in a_and_b:
 163         strategy.extended.update(a, fn_0, args=(b,))  # a = a + b
 164         strategy.extended.update(b, fn_1, args=(a,))  # b = a - b
 165         strategy.extended.update(a, fn_2, args=(b,))  # a = a - b
 166 
 167     ctx = tf.distribute.get_replica_context()
 168     return ctx.merge_call(
 169         swap, args=(zip(self._average_weights, self._model_weights),))
 170 
 171   def assign_average_vars(self, var_list: List[tf.Variable]):
 172     """Assign variables in var_list with their respective averages.
 173 
 174     Args:
 175       var_list: List of model variables to be assigned to their average.
 176     Returns:
 177       assign_op: The op corresponding to the assignment operation of
 178         variables to their average.
 179     """
 180     assign_op = tf.group([
 181         var.assign(self.get_slot(var, 'average')) for var in var_list
 182         if var.trainable
 183     ])
 184     return assign_op

 231   def get_config(self):
 232     config = {
 233         'optimizer': tf.keras.optimizers.serialize(self._optimizer),
 234         'average_decay': self._average_decay,
 235         'start_step': self._start_step,
 236         'dynamic_decay': self._dynamic_decay,
 237     }
 238     base_config = super(MovingAverage, self).get_config()
 239     return dict(list(base_config.items()) + list(config.items()))
 240 
 241   @classmethod
 242   def from_config(cls, config, custom_objects=None):
 243     optimizer = tf.keras.optimizers.deserialize(
 244         config.pop('optimizer'),
 245         custom_objects=custom_objects,
 246     )
 247     return cls(optimizer, **config)

 250 def build_optimizer(
 251     optimizer_name: Text,
 252     base_learning_rate: tf.keras.optimizers.schedules.LearningRateSchedule,
 253     params: Dict[Text, Any]):
 254   """Build the optimizer based on name.
 255 
 256   Args:
 257     optimizer_name: String representation of the optimizer name. Examples:
 258       sgd, momentum, rmsprop.
 259     base_learning_rate: `tf.keras.optimizers.schedules.LearningRateSchedule`
 260       base learning rate.
 261     params: String -> Any dictionary representing the optimizer params.
 262       This should contain optimizer specific parameters such as
 263       `base_learning_rate`, `decay`, etc.
 264 
 265   Returns:
 266     A tf.keras.Optimizer.
 267 
 268   Raises:
 269     ValueError if the provided optimizer_name is not supported.
 270 
 271   """
 272   optimizer_name = optimizer_name.lower()
 273   logging.info('Building %s optimizer with params %s', optimizer_name, params)
 274 
 275   if optimizer_name == 'sgd':
 276     logging.info('Using SGD optimizer')
 277     nesterov = params.get('nesterov', False)
 278     optimizer = tf.keras.optimizers.SGD(learning_rate=base_learning_rate,
 279                                         nesterov=nesterov)
 280   elif optimizer_name == 'momentum':
 281     logging.info('Using momentum optimizer')
 282     nesterov = params.get('nesterov', False)
 283     optimizer = tf.keras.optimizers.SGD(learning_rate=base_learning_rate,
 284                                         momentum=params['momentum'],
 285                                         nesterov=nesterov)
 286   elif optimizer_name == 'rmsprop':
 287     logging.info('Using RMSProp')
 288     rho = params.get('decay', None) or params.get('rho', 0.9)
 289     momentum = params.get('momentum', 0.9)
 290     epsilon = params.get('epsilon', 1e-07)
 291     optimizer = tf.keras.optimizers.RMSprop(learning_rate=base_learning_rate,
 292                                             rho=rho,
 293                                             momentum=momentum,
 294                                             epsilon=epsilon)
 295   elif optimizer_name == 'adam':
 296     logging.info('Using Adam')
 297     beta_1 = params.get('beta_1', 0.9)
 298     beta_2 = params.get('beta_2', 0.999)
 299     epsilon = params.get('epsilon', 1e-07)
 300     optimizer = tf.keras.optimizers.Adam(learning_rate=base_learning_rate,
 301                                          beta_1=beta_1,
 302                                          beta_2=beta_2,
 303                                          epsilon=epsilon)
 304   elif optimizer_name == 'adamw':
 305     logging.info('Using AdamW')
 306     weight_decay = params.get('weight_decay', 0.01)
 307     beta_1 = params.get('beta_1', 0.9)
 308     beta_2 = params.get('beta_2', 0.999)
 309     epsilon = params.get('epsilon', 1e-07)
 310     optimizer = tfa.optimizers.AdamW(weight_decay=weight_decay,
 311                                      learning_rate=base_learning_rate,
 312                                      beta_1=beta_1,
 313                                      beta_2=beta_2,
 314                                      epsilon=epsilon)
 315   else:
 316     raise ValueError('Unknown optimizer %s' % optimizer_name)
 317 
 318   if params.get('lookahead', None):
 319     logging.info('Using lookahead optimizer.')
 320     optimizer = tfa.optimizers.Lookahead(optimizer)
 321 
 322   # Moving average should be applied last, as it's applied at test time
 323   moving_average_decay = params.get('moving_average_decay', 0.)
 324   if moving_average_decay is not None and moving_average_decay > 0.:
 325     logging.info('Including moving average decay.')
 326     optimizer = MovingAverage(
 327         optimizer,
 328         average_decay=moving_average_decay)
 329   return optimizer

 332 def build_learning_rate(params: base_configs.LearningRateConfig,
 333                         batch_size: int = None,
 334                         train_epochs: int = None,
 335                         train_steps: int = None):
 336   """Build the learning rate given the provided configuration."""
 337   decay_type = params.name
 338   base_lr = params.initial_lr
 339   decay_rate = params.decay_rate
 340   if params.decay_epochs is not None:
 341     decay_steps = params.decay_epochs * train_steps
 342   else:
 343     decay_steps = 0
 344   if params.warmup_epochs is not None:
 345     warmup_steps = params.warmup_epochs * train_steps
 346   else:
 347     warmup_steps = 0
 348 
 349   lr_multiplier = params.scale_by_batch_size
 350 
 351   if lr_multiplier and lr_multiplier > 0:
 352     # Scale the learning rate based on the batch size and a multiplier
 353     base_lr *= lr_multiplier * batch_size
 354     logging.info('Scaling the learning rate based on the batch size '
 355                  'multiplier. New base_lr: %f', base_lr)
 356 
 357   if decay_type == 'exponential':
 358     logging.info('Using exponential learning rate with: '
 359                  'initial_learning_rate: %f, decay_steps: %d, '
 360                  'decay_rate: %f', base_lr, decay_steps, decay_rate)
 361     lr = tf.keras.optimizers.schedules.ExponentialDecay(
 362         initial_learning_rate=base_lr,
 363         decay_steps=decay_steps,
 364         decay_rate=decay_rate,
 365         staircase=params.staircase)
 366   elif decay_type == 'piecewise_constant_with_warmup':
 367     logging.info('Using Piecewise constant decay with warmup. '
 368                  'Parameters: batch_size: %d, epoch_size: %d, '
 369                  'warmup_epochs: %d, boundaries: %s, multipliers: %s',
 370                  batch_size, params.examples_per_epoch,
 371                  params.warmup_epochs, params.boundaries,
 372                  params.multipliers)
 373     lr = learning_rate.PiecewiseConstantDecayWithWarmup(
 374         batch_size=batch_size,
 375         epoch_size=params.examples_per_epoch,
 376         warmup_epochs=params.warmup_epochs,
 377         boundaries=params.boundaries,
 378         multipliers=params.multipliers)
 379   elif decay_type == 'cosine_with_warmup':
 380     lr = learning_rate.CosineDecayWithWarmup(
 381         batch_size=batch_size,
 382         total_steps=train_epochs * train_steps,
 383         warmup_steps=warmup_steps)
 384   if warmup_steps > 0:
 385     if decay_type not in [
 386         'piecewise_constant_with_warmup', 'cosine_with_warmup'
 387     ]:
 388       logging.info('Applying %d warmup steps to the learning rate',
 389                    warmup_steps)
 390       lr = learning_rate.WarmupDecaySchedule(lr, warmup_steps)
 391   return lr
