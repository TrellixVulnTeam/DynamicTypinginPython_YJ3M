  98 def get_config(model: str, dataset: str) -> base_configs.ExperimentConfig:
  99   """Given model and dataset names, return the ExperimentConfig."""
 100   dataset_model_config_map = {
 101       'imagenet': {
 102           'efficientnet': EfficientNetImageNetConfig(),
 103           'resnet': ResNetImagenetConfig(),
 104       }
 105   }
 106   try:
 107     return dataset_model_config_map[dataset][model]
 108   except KeyError:
 109     if dataset not in dataset_model_config_map:
 110       raise KeyError('Invalid dataset received. Received: {}. Supported '
 111                      'datasets include: {}'.format(
 112                          dataset,
 113                          ', '.join(dataset_model_config_map.keys())))
 114     raise KeyError('Invalid model received. Received: {}. Supported models for'
 115                    '{} include: {}'.format(
 116                        model,
 117                        dataset,
 118                        ', '.join(dataset_model_config_map[dataset].keys())))
