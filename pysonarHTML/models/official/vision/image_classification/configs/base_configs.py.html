  69 @dataclasses.dataclass
  70 class TrainConfig(hyperparams.Config):
  71   """Configuration for training.
  72 
  73   Attributes:
  74     resume_checkpoint: Whether or not to enable load checkpoint loading.
  75       Defaults to None.
  76     epochs: The number of training epochs to run. Defaults to None.
  77     steps: The number of steps to run per epoch. If None, then this will be
  78       inferred based on the number of images and batch size. Defaults to None.
  79     callbacks: An instance of CallbacksConfig.
  80     metrics: An instance of MetricsConfig.
  81     tensorboard: An instance of TensorboardConfig.
  82     set_epoch_loop: Whether or not to set `experimental_steps_per_execution` to
  83       equal the number of training steps in `model.compile`. This reduces the
  84       number of callbacks run per epoch which significantly improves end-to-end
  85       TPU training time.
  86   """
  87   resume_checkpoint: bool = None
  88   epochs: int = None
  89   steps: int = None
  90   callbacks: CallbacksConfig = CallbacksConfig()
  91   metrics: MetricsConfig = None
  92   tensorboard: TensorboardConfig = TensorboardConfig()
  93   time_history: TimeHistoryConfig = TimeHistoryConfig()
  94   set_epoch_loop: bool = False

 127 @dataclasses.dataclass
 128 class OptimizerConfig(hyperparams.Config):
 129   """Configuration for Optimizers.
 130 
 131   Attributes:
 132     name: The name of the optimizer. Defaults to None.
 133     decay: Decay or rho, discounting factor for gradient. Defaults to None.
 134     epsilon: Small value used to avoid 0 denominator. Defaults to None.
 135     momentum: Plain momentum constant. Defaults to None.
 136     nesterov: Whether or not to apply Nesterov momentum. Defaults to None.
 137     moving_average_decay: The amount of decay to apply. If 0 or None, then
 138       exponential moving average is not used. Defaults to None.
 139     lookahead: Whether or not to apply the lookahead optimizer. Defaults to
 140       None.
 141     beta_1: The exponential decay rate for the 1st moment estimates. Used in the
 142       Adam optimizers. Defaults to None.
 143     beta_2: The exponential decay rate for the 2nd moment estimates. Used in the
 144       Adam optimizers. Defaults to None.
 145     epsilon: Small value used to avoid 0 denominator. Defaults to 1e-7.
 146   """
 147   name: str = None
 148   decay: float = None
 149   epsilon: float = None
 150   momentum: float = None
 151   nesterov: bool = None
 152   moving_average_decay: Optional[float] = None
 153   lookahead: Optional[bool] = None
 154   beta_1: float = None
 155   beta_2: float = None
 156   epsilon: float = None

 159 @dataclasses.dataclass
 160 class LearningRateConfig(hyperparams.Config):
 161   """Configuration for learning rates.
 162 
 163   Attributes:
 164     name: The name of the learning rate. Defaults to None.
 165     initial_lr: The initial learning rate. Defaults to None.
 166     decay_epochs: The number of decay epochs. Defaults to None.
 167     decay_rate: The rate of decay. Defaults to None.
 168     warmup_epochs: The number of warmup epochs. Defaults to None.
 169     batch_lr_multiplier: The multiplier to apply to the base learning rate, if
 170       necessary. Defaults to None.
 171     examples_per_epoch: the number of examples in a single epoch. Defaults to
 172       None.
 173     boundaries: boundaries used in piecewise constant decay with warmup.
 174     multipliers: multipliers used in piecewise constant decay with warmup.
 175     scale_by_batch_size: Scale the learning rate by a fraction of the batch
 176       size. Set to 0 for no scaling (default).
 177     staircase: Apply exponential decay at discrete values instead of continuous.
 178   """
 179   name: str = None
 180   initial_lr: float = None
 181   decay_epochs: float = None
 182   decay_rate: float = None
 183   warmup_epochs: int = None
 184   examples_per_epoch: int = None
 185   boundaries: List[int] = None
 186   multipliers: List[float] = None
 187   scale_by_batch_size: float = 0.
 188   staircase: bool = None

 209 @dataclasses.dataclass
 210 class ExperimentConfig(hyperparams.Config):
 211   """Base configuration for an image classification experiment.
 212 
 213   Attributes:
 214     model_dir: The directory to use when running an experiment.
 215     mode: e.g. 'train_and_eval', 'export'
 216     runtime: A `RuntimeConfig` instance.
 217     train: A `TrainConfig` instance.
 218     evaluation: An `EvalConfig` instance.
 219     model: A `ModelConfig` instance.
 220     export: An `ExportConfig` instance.
 221   """
 222   model_dir: str = None
 223   model_name: str = None
 224   mode: str = None
 225   runtime: RuntimeConfig = None
 226   train_dataset: Any = None
 227   validation_dataset: Any = None
 228   train: TrainConfig = None
 229   evaluation: EvalConfig = None
 230   model: ModelConfig = None
 231   export: ExportConfig = None
