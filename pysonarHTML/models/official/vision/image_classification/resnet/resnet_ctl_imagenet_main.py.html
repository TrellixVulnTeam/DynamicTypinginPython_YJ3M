  45 def build_stats(runnable, time_callback):
  46   """Normalizes and returns dictionary of stats.
  47 
  48   Args:
  49     runnable: The module containing all the training and evaluation metrics.
  50     time_callback: Time tracking callback instance.
  51 
  52   Returns:
  53     Dictionary of normalized results.
  54   """
  55   stats = {}
  56 
  57   if not runnable.flags_obj.skip_eval:
  58     stats['eval_loss'] = runnable.test_loss.result().numpy()
  59     stats['eval_acc'] = runnable.test_accuracy.result().numpy()
  60 
  61     stats['train_loss'] = runnable.train_loss.result().numpy()
  62     stats['train_acc'] = runnable.train_accuracy.result().numpy()
  63 
  64   if time_callback:
  65     timestamp_log = time_callback.timestamp_log
  66     stats['step_timestamp_log'] = timestamp_log
  67     stats['train_finish_time'] = time_callback.train_finish_time
  68     if time_callback.epoch_runtime_log:
  69       stats['avg_exp_per_second'] = time_callback.average_examples_per_second
  70 
  71   return stats

  74 def get_num_train_iterations(flags_obj):
  75   """Returns the number of training steps, train and test epochs."""
  76   train_steps = (
  77       imagenet_preprocessing.NUM_IMAGES['train'] // flags_obj.batch_size)
  78   train_epochs = flags_obj.train_epochs
  79 
  80   if flags_obj.train_steps:
  81     train_steps = min(flags_obj.train_steps, train_steps)
  82     train_epochs = 1
  83 
  84   eval_steps = math.ceil(1.0 * imagenet_preprocessing.NUM_IMAGES['validation'] /
  85                          flags_obj.batch_size)
  86 
  87   return train_steps, train_epochs, eval_steps

  90 def _steps_to_run(steps_in_current_epoch, steps_per_epoch, steps_per_loop):
  91   """Calculates steps to run on device."""
  92   if steps_per_loop <= 0:
  93     raise ValueError('steps_per_loop should be positive integer.')
  94   if steps_per_loop == 1:
  95     return steps_per_loop
  96   return min(steps_per_loop, steps_per_epoch - steps_in_current_epoch)

  99 def run(flags_obj):
 100   """Run ResNet ImageNet training and eval loop using custom training loops.
 101 
 102   Args:
 103     flags_obj: An object containing parsed flag values.
 104 
 105   Raises:
 106     ValueError: If fp16 is passed as it is not currently supported.
 107 
 108   Returns:
 109     Dictionary of training and eval stats.
 110   """
 111   keras_utils.set_session_config(
 112       enable_xla=flags_obj.enable_xla)
 113   performance.set_mixed_precision_policy(flags_core.get_tf_dtype(flags_obj))
 114 
 115   if tf.config.list_physical_devices('GPU'):
 116     if flags_obj.tf_gpu_thread_mode:
 117       keras_utils.set_gpu_thread_mode_and_count(
 118           per_gpu_thread_count=flags_obj.per_gpu_thread_count,
 119           gpu_thread_mode=flags_obj.tf_gpu_thread_mode,
 120           num_gpus=flags_obj.num_gpus,
 121           datasets_num_private_threads=flags_obj.datasets_num_private_threads)
 122     common.set_cudnn_batchnorm_mode()
 123 
 124   # TODO(anj-s): Set data_format without using Keras.
 125   data_format = flags_obj.data_format
 126   if data_format is None:
 127     data_format = ('channels_first' if tf.config.list_physical_devices('GPU')
 128                    else 'channels_last')
 129   tf.keras.backend.set_image_data_format(data_format)
 130 
 131   strategy = distribution_utils.get_distribution_strategy(
 132       distribution_strategy=flags_obj.distribution_strategy,
 133       num_gpus=flags_obj.num_gpus,
 134       all_reduce_alg=flags_obj.all_reduce_alg,
 135       num_packs=flags_obj.num_packs,
 136       tpu_address=flags_obj.tpu)
 137 
 138   per_epoch_steps, train_epochs, eval_steps = get_num_train_iterations(
 139       flags_obj)
 140   steps_per_loop = min(flags_obj.steps_per_loop, per_epoch_steps)
 141 
 142   logging.info(
 143       'Training %d epochs, each epoch has %d steps, '
 144       'total steps: %d; Eval %d steps', train_epochs, per_epoch_steps,
 145       train_epochs * per_epoch_steps, eval_steps)
 146 
 147   time_callback = keras_utils.TimeHistory(
 148       flags_obj.batch_size,
 149       flags_obj.log_steps,
 150       logdir=flags_obj.model_dir if flags_obj.enable_tensorboard else None)
 151   with distribution_utils.get_strategy_scope(strategy):
 152     runnable = resnet_runnable.ResnetRunnable(flags_obj, time_callback,
 153                                               per_epoch_steps)
 154 
 155   eval_interval = flags_obj.epochs_between_evals * per_epoch_steps
 156   checkpoint_interval = (
 157       per_epoch_steps if flags_obj.enable_checkpoint_and_export else None)
 158   summary_interval = per_epoch_steps if flags_obj.enable_tensorboard else None
 159 
 160   checkpoint_manager = tf.train.CheckpointManager(
 161       runnable.checkpoint,
 162       directory=flags_obj.model_dir,
 163       max_to_keep=10,
 164       step_counter=runnable.global_step,
 165       checkpoint_interval=checkpoint_interval)
 166 
 167   resnet_controller = controller.Controller(
 168       strategy,
 169       runnable.train,
 170       runnable.evaluate if not flags_obj.skip_eval else None,
 171       global_step=runnable.global_step,
 172       steps_per_loop=steps_per_loop,
 173       train_steps=per_epoch_steps * train_epochs,
 174       checkpoint_manager=checkpoint_manager,
 175       summary_interval=summary_interval,
 176       eval_steps=eval_steps,
 177       eval_interval=eval_interval)
 178 
 179   time_callback.on_train_begin()
 180   resnet_controller.train(evaluate=not flags_obj.skip_eval)
 181   time_callback.on_train_end()
 182 
 183   stats = build_stats(runnable, time_callback)
 184   return stats

 189   stats = run(flags.FLAGS)
 190   logging.info('Run stats:\n%s', stats)
