  15 """Provides utilities to preprocess images.
  16 
  17 Training images are sampled using the provided bounding boxes, and subsequently
  18 cropped to the sampled bounding box. Images are additionally flipped randomly,
  19 then resized to the target output size (without aspect-ratio preservation).
  20 
  21 Images used during evaluation are resized (with aspect-ratio preservation) and
  22 centrally cropped.
  23 
  24 All images undergo mean color subtraction.
  25 
  26 Note that these steps are colloquially referred to as "ResNet preprocessing,"
  27 and they differ from "VGG preprocessing," which does not use bounding boxes
  28 and instead does an aspect-preserving resize followed by random crop during
  29 training. (These both differ from "Inception preprocessing," which introduces
  30 color distortion steps.)
  31 
  32 """

  65 def process_record_dataset(dataset,
  66                            is_training,
  67                            batch_size,
  68                            shuffle_buffer,
  69                            parse_record_fn,
  70                            dtype=tf.float32,
  71                            datasets_num_private_threads=None,
  72                            drop_remainder=False,
  73                            tf_data_experimental_slack=False):
  74   """Given a Dataset with raw records, return an iterator over the records.
  75 
  76   Args:
  77     dataset: A Dataset representing raw records
  78     is_training: A boolean denoting whether the input is for training.
  79     batch_size: The number of samples per batch.
  80     shuffle_buffer: The buffer size to use when shuffling records. A larger
  81       value results in better randomness, but smaller values reduce startup
  82       time and use less memory.
  83     parse_record_fn: A function that takes a raw record and returns the
  84       corresponding (image, label) pair.
  85     dtype: Data type to use for images/features.
  86     datasets_num_private_threads: Number of threads for a private
  87       threadpool created for all datasets computation.
  88     drop_remainder: A boolean indicates whether to drop the remainder of the
  89       batches. If True, the batch dimension will be static.
  90     tf_data_experimental_slack: Whether to enable tf.data's
  91       `experimental_slack` option.
  92 
  93   Returns:
  94     Dataset of (image, label) pairs ready for iteration.
  95   """
  96   # Defines a specific size thread pool for tf.data operations.
  97   if datasets_num_private_threads:
  98     options = tf.data.Options()
  99     options.experimental_threading.private_threadpool_size = (
 100         datasets_num_private_threads)
 101     dataset = dataset.with_options(options)
 102     logging.info(
 103         'datasets_num_private_threads: %s', datasets_num_private_threads)
 104 
 105   if is_training:
 106     # Shuffles records before repeating to respect epoch boundaries.
 107     dataset = dataset.shuffle(buffer_size=shuffle_buffer)
 108     # Repeats the dataset for the number of epochs to train.
 109     dataset = dataset.repeat()
 110 
 111   # Parses the raw records into images and labels.
 112   dataset = dataset.map(
 113       lambda value: parse_record_fn(value, is_training, dtype),
 114       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 115   dataset = dataset.batch(batch_size, drop_remainder=drop_remainder)
 116 
 117   # Operations between the final prefetch and the get_next call to the iterator
 118   # will happen synchronously during run time. We prefetch here again to
 119   # background all of the above processing work and keep it out of the
 120   # critical training path. Setting buffer_size to tf.data.experimental.AUTOTUNE
 121   # allows DistributionStrategies to adjust how many batches to fetch based
 122   # on how many devices are present.
 123   dataset = dataset.prefetch(buffer_size=tf.data.experimental.AUTOTUNE)
 124 
 125   options = tf.data.Options()
 126   options.experimental_slack = tf_data_experimental_slack
 127   dataset = dataset.with_options(options)
 128 
 129   return dataset

 144 def parse_example_proto(example_serialized):
 145   """Parses an Example proto containing a training example of an image.
 146 
 147   The output of the build_image_data.py image preprocessing script is a dataset
 148   containing serialized Example protocol buffers. Each Example proto contains
 149   the following fields (values are included as examples):
 150 
 151     image/height: 462
 152     image/width: 581
 153     image/colorspace: 'RGB'
 154     image/channels: 3
 155     image/class/label: 615
 156     image/class/synset: 'n03623198'
 157     image/class/text: 'knee pad'
 158     image/object/bbox/xmin: 0.1
 159     image/object/bbox/xmax: 0.9
 160     image/object/bbox/ymin: 0.2
 161     image/object/bbox/ymax: 0.6
 162     image/object/bbox/label: 615
 163     image/format: 'JPEG'
 164     image/filename: 'ILSVRC2012_val_00041207.JPEG'
 165     image/encoded: <JPEG encoded string>
 166 
 167   Args:
 168     example_serialized: scalar Tensor tf.string containing a serialized
 169       Example protocol buffer.
 170 
 171   Returns:
 172     image_buffer: Tensor tf.string containing the contents of a JPEG file.
 173     label: Tensor tf.int32 containing the label.
 174     bbox: 3-D float Tensor of bounding boxes arranged [1, num_boxes, coords]
 175       where each coordinate is [0, 1) and the coordinates are arranged as
 176       [ymin, xmin, ymax, xmax].
 177   """
 178   # Dense features in Example proto.
 179   feature_map = {
 180       'image/encoded': tf.io.FixedLenFeature([], dtype=tf.string,
 181                                              default_value=''),
 182       'image/class/label': tf.io.FixedLenFeature([], dtype=tf.int64,
 183                                                  default_value=-1),
 184       'image/class/text': tf.io.FixedLenFeature([], dtype=tf.string,
 185                                                 default_value=''),
 186   }
 187   sparse_float32 = tf.io.VarLenFeature(dtype=tf.float32)
 188   # Sparse features in Example proto.
 189   feature_map.update(
 190       {k: sparse_float32 for k in [
 191           'image/object/bbox/xmin', 'image/object/bbox/ymin',
 192           'image/object/bbox/xmax', 'image/object/bbox/ymax']})
 193 
 194   features = tf.io.parse_single_example(serialized=example_serialized,
 195                                         features=feature_map)
 196   label = tf.cast(features['image/class/label'], dtype=tf.int32)
 197 
 198   xmin = tf.expand_dims(features['image/object/bbox/xmin'].values, 0)
 199   ymin = tf.expand_dims(features['image/object/bbox/ymin'].values, 0)
 200   xmax = tf.expand_dims(features['image/object/bbox/xmax'].values, 0)
 201   ymax = tf.expand_dims(features['image/object/bbox/ymax'].values, 0)
 202 
 203   # Note that we impose an ordering of (y, x) just to make life difficult.
 204   bbox = tf.concat([ymin, xmin, ymax, xmax], 0)
 205 
 206   # Force the variable number of bounding boxes into the shape
 207   # [1, num_boxes, coords].
 208   bbox = tf.expand_dims(bbox, 0)
 209   bbox = tf.transpose(a=bbox, perm=[0, 2, 1])
 210 
 211   return features['image/encoded'], label, bbox

 214 def parse_record(raw_record, is_training, dtype):
 215   """Parses a record containing a training example of an image.
 216 
 217   The input record is parsed into a label and image, and the image is passed
 218   through preprocessing steps (cropping, flipping, and so on).
 219 
 220   Args:
 221     raw_record: scalar Tensor tf.string containing a serialized
 222       Example protocol buffer.
 223     is_training: A boolean denoting whether the input is for training.
 224     dtype: data type to use for images/features.
 225 
 226   Returns:
 227     Tuple with processed image tensor in a channel-last format and
 228     one-hot-encoded label tensor.
 229   """
 230   image_buffer, label, bbox = parse_example_proto(raw_record)
 231 
 232   image = preprocess_image(
 233       image_buffer=image_buffer,
 234       bbox=bbox,
 235       output_height=DEFAULT_IMAGE_SIZE,
 236       output_width=DEFAULT_IMAGE_SIZE,
 237       num_channels=NUM_CHANNELS,
 238       is_training=is_training)
 239   image = tf.cast(image, dtype)
 240 
 241   # Subtract one so that labels are in [0, 1000), and cast to float32 for
 242   # Keras model.
 243   label = tf.cast(tf.cast(tf.reshape(label, shape=[1]), dtype=tf.int32) - 1,
 244                   dtype=tf.float32)
 245   return image, label

 248 def get_parse_record_fn(use_keras_image_data_format=False):
 249   """Get a function for parsing the records, accounting for image format.
 250 
 251   This is useful by handling different types of Keras models. For instance,
 252   the current resnet_model.resnet50 input format is always channel-last,
 253   whereas the keras_applications mobilenet input format depends on
 254   tf.keras.backend.image_data_format(). We should set
 255   use_keras_image_data_format=False for the former and True for the latter.
 256 
 257   Args:
 258     use_keras_image_data_format: A boolean denoting whether data format is keras
 259       backend image data format. If False, the image format is channel-last. If
 260       True, the image format matches tf.keras.backend.image_data_format().
 261 
 262   Returns:
 263     Function to use for parsing the records.
 264   """
 265   def parse_record_fn(raw_record, is_training, dtype):
 266     image, label = parse_record(raw_record, is_training, dtype)
 267     if use_keras_image_data_format:
 268       if tf.keras.backend.image_data_format() == 'channels_first':
 269         image = tf.transpose(image, perm=[2, 0, 1])
 270     return image, label
 271   return parse_record_fn

 274 def input_fn(is_training,
 275              data_dir,
 276              batch_size,
 277              dtype=tf.float32,
 278              datasets_num_private_threads=None,
 279              parse_record_fn=parse_record,
 280              input_context=None,
 281              drop_remainder=False,
 282              tf_data_experimental_slack=False,
 283              training_dataset_cache=False,
 284              filenames=None):
 285   """Input function which provides batches for train or eval.
 286 
 287   Args:
 288     is_training: A boolean denoting whether the input is for training.
 289     data_dir: The directory containing the input data.
 290     batch_size: The number of samples per batch.
 291     dtype: Data type to use for images/features
 292     datasets_num_private_threads: Number of private threads for tf.data.
 293     parse_record_fn: Function to use for parsing the records.
 294     input_context: A `tf.distribute.InputContext` object passed in by
 295       `tf.distribute.Strategy`.
 296     drop_remainder: A boolean indicates whether to drop the remainder of the
 297       batches. If True, the batch dimension will be static.
 298     tf_data_experimental_slack: Whether to enable tf.data's
 299       `experimental_slack` option.
 300     training_dataset_cache: Whether to cache the training dataset on workers.
 301        Typically used to improve training performance when training data is in
 302        remote storage and can fit into worker memory.
 303     filenames: Optional field for providing the file names of the TFRecords.
 304 
 305   Returns:
 306     A dataset that can be used for iteration.
 307   """
 308   if filenames is None:
 309     filenames = get_filenames(is_training, data_dir)
 310   dataset = tf.data.Dataset.from_tensor_slices(filenames)
 311 
 312   if input_context:
 313     logging.info(
 314         'Sharding the dataset: input_pipeline_id=%d num_input_pipelines=%d',
 315         input_context.input_pipeline_id, input_context.num_input_pipelines)
 316     dataset = dataset.shard(input_context.num_input_pipelines,
 317                             input_context.input_pipeline_id)
 318 
 319   if is_training:
 320     # Shuffle the input files
 321     dataset = dataset.shuffle(buffer_size=_NUM_TRAIN_FILES)
 322 
 323   # Convert to individual records.
 324   # cycle_length = 10 means that up to 10 files will be read and deserialized in
 325   # parallel. You may want to increase this number if you have a large number of
 326   # CPU cores.
 327   dataset = dataset.interleave(
 328       tf.data.TFRecordDataset,
 329       cycle_length=10,
 330       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 331 
 332   if is_training and training_dataset_cache:
 333     # Improve training performance when training data is in remote storage and
 334     # can fit into worker memory.
 335     dataset = dataset.cache()
 336 
 337   return process_record_dataset(
 338       dataset=dataset,
 339       is_training=is_training,
 340       batch_size=batch_size,
 341       shuffle_buffer=_SHUFFLE_BUFFER,
 342       parse_record_fn=parse_record_fn,
 343       dtype=dtype,
 344       datasets_num_private_threads=datasets_num_private_threads,
 345       drop_remainder=drop_remainder,
 346       tf_data_experimental_slack=tf_data_experimental_slack,
 347   )

 350 def _decode_crop_and_flip(image_buffer, bbox, num_channels):
 351   """Crops the given image to a random part of the image, and randomly flips.
 352 
 353   We use the fused decode_and_crop op, which performs better than the two ops
 354   used separately in series, but note that this requires that the image be
 355   passed in as an un-decoded string Tensor.
 356 
 357   Args:
 358     image_buffer: scalar string Tensor representing the raw JPEG image buffer.
 359     bbox: 3-D float Tensor of bounding boxes arranged [1, num_boxes, coords]
 360       where each coordinate is [0, 1) and the coordinates are arranged as
 361       [ymin, xmin, ymax, xmax].
 362     num_channels: Integer depth of the image buffer for decoding.
 363 
 364   Returns:
 365     3-D tensor with cropped image.
 366 
 367   """
 368   # A large fraction of image datasets contain a human-annotated bounding box
 369   # delineating the region of the image containing the object of interest.  We
 370   # choose to create a new bounding box for the object which is a randomly
 371   # distorted version of the human-annotated bounding box that obeys an
 372   # allowed range of aspect ratios, sizes and overlap with the human-annotated
 373   # bounding box. If no box is supplied, then we assume the bounding box is
 374   # the entire image.
 375   sample_distorted_bounding_box = tf.image.sample_distorted_bounding_box(
 376       tf.image.extract_jpeg_shape(image_buffer),
 377       bounding_boxes=bbox,
 378       min_object_covered=0.1,
 379       aspect_ratio_range=[0.75, 1.33],
 380       area_range=[0.05, 1.0],
 381       max_attempts=100,
 382       use_image_if_no_bounding_boxes=True)
 383   bbox_begin, bbox_size, _ = sample_distorted_bounding_box
 384 
 385   # Reassemble the bounding box in the format the crop op requires.
 386   offset_y, offset_x, _ = tf.unstack(bbox_begin)
 387   target_height, target_width, _ = tf.unstack(bbox_size)
 388   crop_window = tf.stack([offset_y, offset_x, target_height, target_width])
 389 
 390   # Use the fused decode and crop op here, which is faster than each in series.
 391   cropped = tf.image.decode_and_crop_jpeg(
 392       image_buffer, crop_window, channels=num_channels)
 393 
 394   # Flip to add a little more random distortion in.
 395   cropped = tf.image.random_flip_left_right(cropped)
 396   return cropped

 399 def _central_crop(image, crop_height, crop_width):
 400   """Performs central crops of the given image list.
 401 
 402   Args:
 403     image: a 3-D image tensor
 404     crop_height: the height of the image following the crop.
 405     crop_width: the width of the image following the crop.
 406 
 407   Returns:
 408     3-D tensor with cropped image.
 409   """
 410   shape = tf.shape(input=image)
 411   height, width = shape[0], shape[1]
 412 
 413   amount_to_be_cropped_h = (height - crop_height)
 414   crop_top = amount_to_be_cropped_h // 2
 415   amount_to_be_cropped_w = (width - crop_width)
 416   crop_left = amount_to_be_cropped_w // 2
 417   return tf.slice(
 418       image, [crop_top, crop_left, 0], [crop_height, crop_width, -1])

 421 def _mean_image_subtraction(image, means, num_channels):
 422   """Subtracts the given means from each image channel.
 423 
 424   For example:
 425     means = [123.68, 116.779, 103.939]
 426     image = _mean_image_subtraction(image, means)
 427 
 428   Note that the rank of `image` must be known.
 429 
 430   Args:
 431     image: a tensor of size [height, width, C].
 432     means: a C-vector of values to subtract from each channel.
 433     num_channels: number of color channels in the image that will be distorted.
 434 
 435   Returns:
 436     the centered image.
 437 
 438   Raises:
 439     ValueError: If the rank of `image` is unknown, if `image` has a rank other
 440       than three or if the number of channels in `image` doesn't match the
 441       number of values in `means`.
 442   """
 443   if image.get_shape().ndims != 3:
 444     raise ValueError('Input must be of size [height, width, C>0]')
 445 
 446   if len(means) != num_channels:
 447     raise ValueError('len(means) must match the number of channels')
 448 
 449   # We have a 1-D tensor of means; convert to 3-D.
 450   # Note(b/130245863): we explicitly call `broadcast` instead of simply
 451   # expanding dimensions for better performance.
 452   means = tf.broadcast_to(means, tf.shape(image))
 453 
 454   return image - means

 457 def _smallest_size_at_least(height, width, resize_min):
 458   """Computes new shape with the smallest side equal to `smallest_side`.
 459 
 460   Computes new shape with the smallest side equal to `smallest_side` while
 461   preserving the original aspect ratio.
 462 
 463   Args:
 464     height: an int32 scalar tensor indicating the current height.
 465     width: an int32 scalar tensor indicating the current width.
 466     resize_min: A python integer or scalar `Tensor` indicating the size of
 467       the smallest side after resize.
 468 
 469   Returns:
 470     new_height: an int32 scalar tensor indicating the new height.
 471     new_width: an int32 scalar tensor indicating the new width.
 472   """
 473   resize_min = tf.cast(resize_min, tf.float32)
 474 
 475   # Convert to floats to make subsequent calculations go smoothly.
 476   height, width = tf.cast(height, tf.float32), tf.cast(width, tf.float32)
 477 
 478   smaller_dim = tf.minimum(height, width)
 479   scale_ratio = resize_min / smaller_dim
 480 
 481   # Convert back to ints to make heights and widths that TF ops will accept.
 482   new_height = tf.cast(height * scale_ratio, tf.int32)
 483   new_width = tf.cast(width * scale_ratio, tf.int32)
 484 
 485   return new_height, new_width

 488 def _aspect_preserving_resize(image, resize_min):
 489   """Resize images preserving the original aspect ratio.
 490 
 491   Args:
 492     image: A 3-D image `Tensor`.
 493     resize_min: A python integer or scalar `Tensor` indicating the size of
 494       the smallest side after resize.
 495 
 496   Returns:
 497     resized_image: A 3-D tensor containing the resized image.
 498   """
 499   shape = tf.shape(input=image)
 500   height, width = shape[0], shape[1]
 501 
 502   new_height, new_width = _smallest_size_at_least(height, width, resize_min)
 503 
 504   return _resize_image(image, new_height, new_width)

 507 def _resize_image(image, height, width):
 508   """Simple wrapper around tf.resize_images.
 509 
 510   This is primarily to make sure we use the same `ResizeMethod` and other
 511   details each time.
 512 
 513   Args:
 514     image: A 3-D image `Tensor`.
 515     height: The target height for the resized image.
 516     width: The target width for the resized image.
 517 
 518   Returns:
 519     resized_image: A 3-D tensor containing the resized image. The first two
 520       dimensions have the shape [height, width].
 521   """
 522   return tf.compat.v1.image.resize(
 523       image, [height, width], method=tf.image.ResizeMethod.BILINEAR,
 524       align_corners=False)

 527 def preprocess_image(image_buffer, bbox, output_height, output_width,
 528                      num_channels, is_training=False):
 529   """Preprocesses the given image.
 530 
 531   Preprocessing includes decoding, cropping, and resizing for both training
 532   and eval images. Training preprocessing, however, introduces some random
 533   distortion of the image to improve accuracy.
 534 
 535   Args:
 536     image_buffer: scalar string Tensor representing the raw JPEG image buffer.
 537     bbox: 3-D float Tensor of bounding boxes arranged [1, num_boxes, coords]
 538       where each coordinate is [0, 1) and the coordinates are arranged as
 539       [ymin, xmin, ymax, xmax].
 540     output_height: The height of the image after preprocessing.
 541     output_width: The width of the image after preprocessing.
 542     num_channels: Integer depth of the image buffer for decoding.
 543     is_training: `True` if we're preprocessing the image for training and
 544       `False` otherwise.
 545 
 546   Returns:
 547     A preprocessed image.
 548   """
 549   if is_training:
 550     # For training, we want to randomize some of the distortions.
 551     image = _decode_crop_and_flip(image_buffer, bbox, num_channels)
 552     image = _resize_image(image, output_height, output_width)
 553   else:
 554     # For validation, we want to decode, resize, then just crop the middle.
 555     image = tf.image.decode_jpeg(image_buffer, channels=num_channels)
 556     image = _aspect_preserving_resize(image, _RESIZE_MIN)
 557     image = _central_crop(image, output_height, output_width)
 558 
 559   image.set_shape([output_height, output_width, num_channels])
 560 
 561   return _mean_image_subtraction(image, CHANNEL_MEANS, num_channels)
