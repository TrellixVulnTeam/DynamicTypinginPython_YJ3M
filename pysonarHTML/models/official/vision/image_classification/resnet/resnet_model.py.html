  41 def _gen_l2_regularizer(use_l2_regularizer=True, l2_weight_decay=1e-4):
  42   return regularizers.l2(l2_weight_decay) if use_l2_regularizer else None

  45 def identity_block(input_tensor,
  46                    kernel_size,
  47                    filters,
  48                    stage,
  49                    block,
  50                    use_l2_regularizer=True,
  51                    batch_norm_decay=0.9,
  52                    batch_norm_epsilon=1e-5):
  53   """The identity block is the block that has no conv layer at shortcut.
  54 
  55   Args:
  56     input_tensor: input tensor
  57     kernel_size: default 3, the kernel size of middle conv layer at main path
  58     filters: list of integers, the filters of 3 conv layer at main path
  59     stage: integer, current stage label, used for generating layer names
  60     block: 'a','b'..., current block label, used for generating layer names
  61     use_l2_regularizer: whether to use L2 regularizer on Conv layer.
  62     batch_norm_decay: Moment of batch norm layers.
  63     batch_norm_epsilon: Epsilon of batch borm layers.
  64 
  65   Returns:
  66     Output tensor for the block.
  67   """
  68   filters1, filters2, filters3 = filters
  69   if backend.image_data_format() == 'channels_last':
  70     bn_axis = 3
  71   else:
  72     bn_axis = 1
  73   conv_name_base = 'res' + str(stage) + block + '_branch'
  74   bn_name_base = 'bn' + str(stage) + block + '_branch'
  75 
  76   x = layers.Conv2D(
  77       filters1, (1, 1),
  78       use_bias=False,
  79       kernel_initializer='he_normal',
  80       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
  81       name=conv_name_base + '2a')(
  82           input_tensor)
  83   x = layers.BatchNormalization(
  84       axis=bn_axis,
  85       momentum=batch_norm_decay,
  86       epsilon=batch_norm_epsilon,
  87       name=bn_name_base + '2a')(
  88           x)
  89   x = layers.Activation('relu')(x)
  90 
  91   x = layers.Conv2D(
  92       filters2,
  93       kernel_size,
  94       padding='same',
  95       use_bias=False,
  96       kernel_initializer='he_normal',
  97       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
  98       name=conv_name_base + '2b')(
  99           x)
 100   x = layers.BatchNormalization(
 101       axis=bn_axis,
 102       momentum=batch_norm_decay,
 103       epsilon=batch_norm_epsilon,
 104       name=bn_name_base + '2b')(
 105           x)
 106   x = layers.Activation('relu')(x)
 107 
 108   x = layers.Conv2D(
 109       filters3, (1, 1),
 110       use_bias=False,
 111       kernel_initializer='he_normal',
 112       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 113       name=conv_name_base + '2c')(
 114           x)
 115   x = layers.BatchNormalization(
 116       axis=bn_axis,
 117       momentum=batch_norm_decay,
 118       epsilon=batch_norm_epsilon,
 119       name=bn_name_base + '2c')(
 120           x)
 121 
 122   x = layers.add([x, input_tensor])
 123   x = layers.Activation('relu')(x)
 124   return x

 127 def conv_block(input_tensor,
 128                kernel_size,
 129                filters,
 130                stage,
 131                block,
 132                strides=(2, 2),
 133                use_l2_regularizer=True,
 134                batch_norm_decay=0.9,
 135                batch_norm_epsilon=1e-5):
 136   """A block that has a conv layer at shortcut.
 137 
 138   Note that from stage 3,
 139   the second conv layer at main path is with strides=(2, 2)
 140   And the shortcut should have strides=(2, 2) as well
 141 
 142   Args:
 143     input_tensor: input tensor
 144     kernel_size: default 3, the kernel size of middle conv layer at main path
 145     filters: list of integers, the filters of 3 conv layer at main path
 146     stage: integer, current stage label, used for generating layer names
 147     block: 'a','b'..., current block label, used for generating layer names
 148     strides: Strides for the second conv layer in the block.
 149     use_l2_regularizer: whether to use L2 regularizer on Conv layer.
 150     batch_norm_decay: Moment of batch norm layers.
 151     batch_norm_epsilon: Epsilon of batch borm layers.
 152 
 153   Returns:
 154     Output tensor for the block.
 155   """
 156   filters1, filters2, filters3 = filters
 157   if backend.image_data_format() == 'channels_last':
 158     bn_axis = 3
 159   else:
 160     bn_axis = 1
 161   conv_name_base = 'res' + str(stage) + block + '_branch'
 162   bn_name_base = 'bn' + str(stage) + block + '_branch'
 163 
 164   x = layers.Conv2D(
 165       filters1, (1, 1),
 166       use_bias=False,
 167       kernel_initializer='he_normal',
 168       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 169       name=conv_name_base + '2a')(
 170           input_tensor)
 171   x = layers.BatchNormalization(
 172       axis=bn_axis,
 173       momentum=batch_norm_decay,
 174       epsilon=batch_norm_epsilon,
 175       name=bn_name_base + '2a')(
 176           x)
 177   x = layers.Activation('relu')(x)
 178 
 179   x = layers.Conv2D(
 180       filters2,
 181       kernel_size,
 182       strides=strides,
 183       padding='same',
 184       use_bias=False,
 185       kernel_initializer='he_normal',
 186       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 187       name=conv_name_base + '2b')(
 188           x)
 189   x = layers.BatchNormalization(
 190       axis=bn_axis,
 191       momentum=batch_norm_decay,
 192       epsilon=batch_norm_epsilon,
 193       name=bn_name_base + '2b')(
 194           x)
 195   x = layers.Activation('relu')(x)
 196 
 197   x = layers.Conv2D(
 198       filters3, (1, 1),
 199       use_bias=False,
 200       kernel_initializer='he_normal',
 201       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 202       name=conv_name_base + '2c')(
 203           x)
 204   x = layers.BatchNormalization(
 205       axis=bn_axis,
 206       momentum=batch_norm_decay,
 207       epsilon=batch_norm_epsilon,
 208       name=bn_name_base + '2c')(
 209           x)
 210 
 211   shortcut = layers.Conv2D(
 212       filters3, (1, 1),
 213       strides=strides,
 214       use_bias=False,
 215       kernel_initializer='he_normal',
 216       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 217       name=conv_name_base + '1')(
 218           input_tensor)
 219   shortcut = layers.BatchNormalization(
 220       axis=bn_axis,
 221       momentum=batch_norm_decay,
 222       epsilon=batch_norm_epsilon,
 223       name=bn_name_base + '1')(
 224           shortcut)
 225 
 226   x = layers.add([x, shortcut])
 227   x = layers.Activation('relu')(x)
 228   return x

 231 def resnet50(num_classes,
 232              batch_size=None,
 233              use_l2_regularizer=True,
 234              rescale_inputs=False,
 235              batch_norm_decay=0.9,
 236              batch_norm_epsilon=1e-5):
 237   """Instantiates the ResNet50 architecture.
 238 
 239   Args:
 240     num_classes: `int` number of classes for image classification.
 241     batch_size: Size of the batches for each step.
 242     use_l2_regularizer: whether to use L2 regularizer on Conv/Dense layer.
 243     rescale_inputs: whether to rescale inputs from 0 to 1.
 244     batch_norm_decay: Moment of batch norm layers.
 245     batch_norm_epsilon: Epsilon of batch borm layers.
 246 
 247   Returns:
 248       A Keras model instance.
 249   """
 250   input_shape = (224, 224, 3)
 251   img_input = layers.Input(shape=input_shape, batch_size=batch_size)
 252   if rescale_inputs:
 253     # Hub image modules expect inputs in the range [0, 1]. This rescales these
 254     # inputs to the range expected by the trained model.
 255     x = layers.Lambda(
 256         lambda x: x * 255.0 - backend.constant(
 257             imagenet_preprocessing.CHANNEL_MEANS,
 258             shape=[1, 1, 3],
 259             dtype=x.dtype),
 260         name='rescale')(
 261             img_input)
 262   else:
 263     x = img_input
 264 
 265   if backend.image_data_format() == 'channels_first':
 266     x = layers.Permute((3, 1, 2))(x)
 267     bn_axis = 1
 268   else:  # channels_last
 269     bn_axis = 3
 270 
 271   block_config = dict(
 272       use_l2_regularizer=use_l2_regularizer,
 273       batch_norm_decay=batch_norm_decay,
 274       batch_norm_epsilon=batch_norm_epsilon)
 275   x = layers.ZeroPadding2D(padding=(3, 3), name='conv1_pad')(x)
 276   x = layers.Conv2D(
 277       64, (7, 7),
 278       strides=(2, 2),
 279       padding='valid',
 280       use_bias=False,
 281       kernel_initializer='he_normal',
 282       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 283       name='conv1')(
 284           x)
 285   x = layers.BatchNormalization(
 286       axis=bn_axis,
 287       momentum=batch_norm_decay,
 288       epsilon=batch_norm_epsilon,
 289       name='bn_conv1')(
 290           x)
 291   x = layers.Activation('relu')(x)
 292   x = layers.MaxPooling2D((3, 3), strides=(2, 2), padding='same')(x)
 293 
 294   x = conv_block(
 295       x, 3, [64, 64, 256], stage=2, block='a', strides=(1, 1), **block_config)
 296   x = identity_block(x, 3, [64, 64, 256], stage=2, block='b', **block_config)
 297   x = identity_block(x, 3, [64, 64, 256], stage=2, block='c', **block_config)
 298 
 299   x = conv_block(x, 3, [128, 128, 512], stage=3, block='a', **block_config)
 300   x = identity_block(x, 3, [128, 128, 512], stage=3, block='b', **block_config)
 301   x = identity_block(x, 3, [128, 128, 512], stage=3, block='c', **block_config)
 302   x = identity_block(x, 3, [128, 128, 512], stage=3, block='d', **block_config)
 303 
 304   x = conv_block(x, 3, [256, 256, 1024], stage=4, block='a', **block_config)
 305   x = identity_block(x, 3, [256, 256, 1024], stage=4, block='b', **block_config)
 306   x = identity_block(x, 3, [256, 256, 1024], stage=4, block='c', **block_config)
 307   x = identity_block(x, 3, [256, 256, 1024], stage=4, block='d', **block_config)
 308   x = identity_block(x, 3, [256, 256, 1024], stage=4, block='e', **block_config)
 309   x = identity_block(x, 3, [256, 256, 1024], stage=4, block='f', **block_config)
 310 
 311   x = conv_block(x, 3, [512, 512, 2048], stage=5, block='a', **block_config)
 312   x = identity_block(x, 3, [512, 512, 2048], stage=5, block='b', **block_config)
 313   x = identity_block(x, 3, [512, 512, 2048], stage=5, block='c', **block_config)
 314 
 315   x = layers.GlobalAveragePooling2D()(x)
 316   x = layers.Dense(
 317       num_classes,
 318       kernel_initializer=initializers.RandomNormal(stddev=0.01),
 319       kernel_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 320       bias_regularizer=_gen_l2_regularizer(use_l2_regularizer),
 321       name='fc1000')(
 322           x)
 323 
 324   # A softmax that is followed by the model loss must be done cannot be done
 325   # in float16 due to numeric issues. So we pass dtype=float32.
 326   x = layers.Activation('softmax', dtype='float32')(x)
 327 
 328   # Create model.
 329   return models.Model(img_input, x, name='resnet50')
