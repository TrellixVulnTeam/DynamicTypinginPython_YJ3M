  33 class ResnetRunnable(standard_runnable.StandardTrainable,
  34                      standard_runnable.StandardEvaluable):
  35   """Implements the training and evaluation APIs for Resnet model."""
  36 
  37   def __init__(self, flags_obj, time_callback, epoch_steps):
  38     standard_runnable.StandardTrainable.__init__(self,
  39                                                  flags_obj.use_tf_while_loop,
  40                                                  flags_obj.use_tf_function)
  41     standard_runnable.StandardEvaluable.__init__(self,
  42                                                  flags_obj.use_tf_function)
  43 
  44     self.strategy = tf.distribute.get_strategy()
  45     self.flags_obj = flags_obj
  46     self.dtype = flags_core.get_tf_dtype(flags_obj)
  47     self.time_callback = time_callback
  48 
  49     # Input pipeline related
  50     batch_size = flags_obj.batch_size
  51     if batch_size % self.strategy.num_replicas_in_sync != 0:
  52       raise ValueError(
  53           'Batch size must be divisible by number of replicas : {}'.format(
  54               self.strategy.num_replicas_in_sync))
  55 
  56     # As auto rebatching is not supported in
  57     # `experimental_distribute_datasets_from_function()` API, which is
  58     # required when cloning dataset to multiple workers in eager mode,
  59     # we use per-replica batch size.
  60     self.batch_size = int(batch_size / self.strategy.num_replicas_in_sync)
  61 
  62     if self.flags_obj.use_synthetic_data:
  63       self.input_fn = common.get_synth_input_fn(
  64           height=imagenet_preprocessing.DEFAULT_IMAGE_SIZE,
  65           width=imagenet_preprocessing.DEFAULT_IMAGE_SIZE,
  66           num_channels=imagenet_preprocessing.NUM_CHANNELS,
  67           num_classes=imagenet_preprocessing.NUM_CLASSES,
  68           dtype=self.dtype,
  69           drop_remainder=True)
  70     else:
  71       self.input_fn = imagenet_preprocessing.input_fn
  72 
  73     self.model = resnet_model.resnet50(
  74         num_classes=imagenet_preprocessing.NUM_CLASSES,
  75         use_l2_regularizer=not flags_obj.single_l2_loss_op)
  76 
  77     lr_schedule = common.PiecewiseConstantDecayWithWarmup(
  78         batch_size=flags_obj.batch_size,
  79         epoch_size=imagenet_preprocessing.NUM_IMAGES['train'],
  80         warmup_epochs=common.LR_SCHEDULE[0][1],
  81         boundaries=list(p[1] for p in common.LR_SCHEDULE[1:]),
  82         multipliers=list(p[0] for p in common.LR_SCHEDULE),
  83         compute_lr_on_cpu=True)
  84     self.optimizer = common.get_optimizer(lr_schedule)
  85     # Make sure iterations variable is created inside scope.
  86     self.global_step = self.optimizer.iterations
  87 
  88     use_graph_rewrite = flags_obj.fp16_implementation == 'graph_rewrite'
  89     if use_graph_rewrite and not flags_obj.use_tf_function:
  90       raise ValueError('--fp16_implementation=graph_rewrite requires '
  91                        '--use_tf_function to be true')
  92     self.optimizer = performance.configure_optimizer(
  93         self.optimizer,
  94         use_float16=self.dtype == tf.float16,
  95         use_graph_rewrite=use_graph_rewrite,
  96         loss_scale=flags_core.get_loss_scale(flags_obj, default_for_fp16=128))
  97 
  98     self.train_loss = tf.keras.metrics.Mean('train_loss', dtype=tf.float32)
  99     self.train_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(
 100         'train_accuracy', dtype=tf.float32)
 101     self.test_loss = tf.keras.metrics.Mean('test_loss', dtype=tf.float32)
 102     self.test_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(
 103         'test_accuracy', dtype=tf.float32)
 104 
 105     self.checkpoint = tf.train.Checkpoint(
 106         model=self.model, optimizer=self.optimizer)
 107 
 108     # Handling epochs.
 109     self.epoch_steps = epoch_steps
 110     self.epoch_helper = utils.EpochHelper(epoch_steps, self.global_step)
 111 
 112   def build_train_dataset(self):
 113     """See base class."""
 114     return utils.make_distributed_dataset(
 115         self.strategy,
 116         self.input_fn,
 117         is_training=True,
 118         data_dir=self.flags_obj.data_dir,
 119         batch_size=self.batch_size,
 120         parse_record_fn=imagenet_preprocessing.parse_record,
 121         datasets_num_private_threads=self.flags_obj
 122         .datasets_num_private_threads,
 123         dtype=self.dtype,
 124         drop_remainder=True)
 125 
 126   def build_eval_dataset(self):
 127     """See base class."""
 128     return utils.make_distributed_dataset(
 129         self.strategy,
 130         self.input_fn,
 131         is_training=False,
 132         data_dir=self.flags_obj.data_dir,
 133         batch_size=self.batch_size,
 134         parse_record_fn=imagenet_preprocessing.parse_record,
 135         dtype=self.dtype)
 136 
 137   def train_loop_begin(self):
 138     """See base class."""
 139     # Reset all metrics
 140     self.train_loss.reset_states()
 141     self.train_accuracy.reset_states()
 142 
 143     self._epoch_begin()
 144     self.time_callback.on_batch_begin(self.epoch_helper.batch_index)
 145 
 146   def train_step(self, iterator):
 147     """See base class."""
 148 
 149     def step_fn(inputs):
 150       """Function to run on the device."""
 151       images, labels = inputs
 152       with tf.GradientTape() as tape:
 153         logits = self.model(images, training=True)
 154 
 155         prediction_loss = tf.keras.losses.sparse_categorical_crossentropy(
 156             labels, logits)
 157         loss = tf.reduce_sum(prediction_loss) * (1.0 /
 158                                                  self.flags_obj.batch_size)
 159         num_replicas = self.strategy.num_replicas_in_sync
 160         l2_weight_decay = 1e-4
 161         if self.flags_obj.single_l2_loss_op:
 162           l2_loss = l2_weight_decay * 2 * tf.add_n([
 163               tf.nn.l2_loss(v)
 164               for v in self.model.trainable_variables
 165               if 'bn' not in v.name
 166           ])
 167 
 168           loss += (l2_loss / num_replicas)
 169         else:
 170           loss += (tf.reduce_sum(self.model.losses) / num_replicas)
 171 
 172       grad_utils.minimize_using_explicit_allreduce(
 173           tape, self.optimizer, loss, self.model.trainable_variables)
 174       self.train_loss.update_state(loss)
 175       self.train_accuracy.update_state(labels, logits)
 176 
 177     self.strategy.run(step_fn, args=(next(iterator),))
 178 
 179   def train_loop_end(self):
 180     """See base class."""
 181     metrics = {
 182         'train_loss': self.train_loss.result(),
 183         'train_accuracy': self.train_accuracy.result(),
 184     }
 185     self.time_callback.on_batch_end(self.epoch_helper.batch_index - 1)
 186     self._epoch_end()
 187     return metrics
 188 
 189   def eval_begin(self):
 190     """See base class."""
 191     self.test_loss.reset_states()
 192     self.test_accuracy.reset_states()
 193 
 194   def eval_step(self, iterator):
 195     """See base class."""
 196 
 197     def step_fn(inputs):
 198       """Function to run on the device."""
 199       images, labels = inputs
 200       logits = self.model(images, training=False)
 201       loss = tf.keras.losses.sparse_categorical_crossentropy(labels, logits)
 202       loss = tf.reduce_sum(loss) * (1.0 / self.flags_obj.batch_size)
 203       self.test_loss.update_state(loss)
 204       self.test_accuracy.update_state(labels, logits)
 205 
 206     self.strategy.run(step_fn, args=(next(iterator),))
 207 
 208   def eval_end(self):
 209     """See base class."""
 210     return {
 211         'test_loss': self.test_loss.result(),
 212         'test_accuracy': self.test_accuracy.result()
 213     }
 214 
 215   def _epoch_begin(self):
 216     if self.epoch_helper.epoch_begin():
 217       self.time_callback.on_epoch_begin(self.epoch_helper.current_epoch)
 218 
 219   def _epoch_end(self):
 220     if self.epoch_helper.epoch_end():
 221       self.time_callback.on_epoch_end(self.epoch_helper.current_epoch)
