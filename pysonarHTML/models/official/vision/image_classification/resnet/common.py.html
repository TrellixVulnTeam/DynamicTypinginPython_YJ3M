  38 class PiecewiseConstantDecayWithWarmup(
  39     tf.keras.optimizers.schedules.LearningRateSchedule):
  40   """Piecewise constant decay with warmup schedule."""
  41 
  42   def __init__(self, batch_size, epoch_size, warmup_epochs, boundaries,
  43                multipliers, compute_lr_on_cpu=True, name=None):
  44     super(PiecewiseConstantDecayWithWarmup, self).__init__()
  45     if len(boundaries) != len(multipliers) - 1:
  46       raise ValueError('The length of boundaries must be 1 less than the '
  47                        'length of multipliers')
  48 
  49     base_lr_batch_size = 256
  50     steps_per_epoch = epoch_size // batch_size
  51 
  52     self.rescaled_lr = BASE_LEARNING_RATE * batch_size / base_lr_batch_size
  53     self.step_boundaries = [float(steps_per_epoch) * x for x in boundaries]
  54     self.lr_values = [self.rescaled_lr * m for m in multipliers]
  55     self.warmup_steps = warmup_epochs * steps_per_epoch
  56     self.compute_lr_on_cpu = compute_lr_on_cpu
  57     self.name = name
  58 
  59     self.learning_rate_ops_cache = {}
  60 
  61   def __call__(self, step):
  62     if tf.executing_eagerly():
  63       return self._get_learning_rate(step)
  64 
  65     # In an eager function or graph, the current implementation of optimizer
  66     # repeatedly call and thus create ops for the learning rate schedule. To
  67     # avoid this, we cache the ops if not executing eagerly.
  68     graph = tf.compat.v1.get_default_graph()
  69     if graph not in self.learning_rate_ops_cache:
  70       if self.compute_lr_on_cpu:
  71         with tf.device('/device:CPU:0'):
  72           self.learning_rate_ops_cache[graph] = self._get_learning_rate(step)
  73       else:
  74         self.learning_rate_ops_cache[graph] = self._get_learning_rate(step)
  75     return self.learning_rate_ops_cache[graph]
  76 
  77   def _get_learning_rate(self, step):
  78     """Compute learning rate at given step."""
  79     with tf.name_scope('PiecewiseConstantDecayWithWarmup'):
  80       def warmup_lr(step):
  81         return self.rescaled_lr * (
  82             tf.cast(step, tf.float32) / tf.cast(self.warmup_steps, tf.float32))
  83       def piecewise_lr(step):
  84         return tf.compat.v1.train.piecewise_constant(
  85             step, self.step_boundaries, self.lr_values)
  86       return tf.cond(step < self.warmup_steps,
  87                      lambda: warmup_lr(step),
  88                      lambda: piecewise_lr(step))
  89 
  90   def get_config(self):
  91     return {
  92         'rescaled_lr': self.rescaled_lr,
  93         'step_boundaries': self.step_boundaries,
  94         'lr_values': self.lr_values,
  95         'warmup_steps': self.warmup_steps,
  96         'compute_lr_on_cpu': self.compute_lr_on_cpu,
  97         'name': self.name
  98     }

 101 def get_optimizer(learning_rate=0.1):
 102   """Returns optimizer to use."""
 103   # The learning_rate is overwritten at the beginning of each step by callback.
 104   return gradient_descent_v2.SGD(learning_rate=learning_rate, momentum=0.9)

 107 def get_callbacks(
 108     pruning_method=None,
 109     enable_checkpoint_and_export=False,
 110     model_dir=None):
 111   """Returns common callbacks."""
 112   time_callback = keras_utils.TimeHistory(
 113       FLAGS.batch_size,
 114       FLAGS.log_steps,
 115       logdir=FLAGS.model_dir if FLAGS.enable_tensorboard else None)
 116   callbacks = [time_callback]
 117 
 118   if FLAGS.enable_tensorboard:
 119     tensorboard_callback = tf.keras.callbacks.TensorBoard(
 120         log_dir=FLAGS.model_dir,
 121         profile_batch=FLAGS.profile_steps)
 122     callbacks.append(tensorboard_callback)
 123 
 124   is_pruning_enabled = pruning_method is not None
 125   if is_pruning_enabled:
 126     callbacks.append(tfmot.sparsity.keras.UpdatePruningStep())
 127     if model_dir is not None:
 128       callbacks.append(tfmot.sparsity.keras.PruningSummaries(
 129           log_dir=model_dir, profile_batch=0))
 130 
 131   if enable_checkpoint_and_export:
 132     if model_dir is not None:
 133       ckpt_full_path = os.path.join(model_dir, 'model.ckpt-{epoch:04d}')
 134       callbacks.append(
 135           tf.keras.callbacks.ModelCheckpoint(ckpt_full_path,
 136                                              save_weights_only=True))
 137   return callbacks

 140 def build_stats(history, eval_output, callbacks):
 141   """Normalizes and returns dictionary of stats.
 142 
 143   Args:
 144     history: Results of the training step. Supports both categorical_accuracy
 145       and sparse_categorical_accuracy.
 146     eval_output: Output of the eval step. Assumes first value is eval_loss and
 147       second value is accuracy_top_1.
 148     callbacks: a list of callbacks which might include a time history callback
 149       used during keras.fit.
 150 
 151   Returns:
 152     Dictionary of normalized results.
 153   """
 154   stats = {}
 155   if eval_output:
 156     stats['accuracy_top_1'] = float(eval_output[1])
 157     stats['eval_loss'] = float(eval_output[0])
 158   if history and history.history:
 159     train_hist = history.history
 160     # Gets final loss from training.
 161     stats['loss'] = float(train_hist['loss'][-1])
 162     # Gets top_1 training accuracy.
 163     if 'categorical_accuracy' in train_hist:
 164       stats[TRAIN_TOP_1] = float(train_hist['categorical_accuracy'][-1])
 165     elif 'sparse_categorical_accuracy' in train_hist:
 166       stats[TRAIN_TOP_1] = float(train_hist['sparse_categorical_accuracy'][-1])
 167     elif 'accuracy' in train_hist:
 168       stats[TRAIN_TOP_1] = float(train_hist['accuracy'][-1])
 169 
 170   if not callbacks:
 171     return stats
 172 
 173   # Look for the time history callback which was used during keras.fit
 174   for callback in callbacks:
 175     if isinstance(callback, keras_utils.TimeHistory):
 176       timestamp_log = callback.timestamp_log
 177       stats['step_timestamp_log'] = timestamp_log
 178       stats['train_finish_time'] = callback.train_finish_time
 179       if callback.epoch_runtime_log:
 180         stats['avg_exp_per_second'] = callback.average_examples_per_second
 181 
 182   return stats

 289 def get_synth_data(height, width, num_channels, num_classes, dtype):
 290   """Creates a set of synthetic random data.
 291 
 292   Args:
 293     height: Integer height that will be used to create a fake image tensor.
 294     width: Integer width that will be used to create a fake image tensor.
 295     num_channels: Integer depth that will be used to create a fake image tensor.
 296     num_classes: Number of classes that should be represented in the fake labels
 297       tensor
 298     dtype: Data type for features/images.
 299 
 300   Returns:
 301     A tuple of tensors representing the inputs and labels.
 302 
 303   """
 304   # Synthetic input should be within [0, 255].
 305   inputs = tf.random.truncated_normal([height, width, num_channels],
 306                                       dtype=dtype,
 307                                       mean=127,
 308                                       stddev=60,
 309                                       name='synthetic_inputs')
 310   labels = tf.random.uniform([1],
 311                              minval=0,
 312                              maxval=num_classes - 1,
 313                              dtype=tf.int32,
 314                              name='synthetic_labels')
 315   return inputs, labels

 335 def get_synth_input_fn(height, width, num_channels, num_classes,
 336                        dtype=tf.float32, drop_remainder=True):
 337   """Returns an input function that returns a dataset with random data.
 338 
 339   This input_fn returns a data set that iterates over a set of random data and
 340   bypasses all preprocessing, e.g. jpeg decode and copy. The host to device
 341   copy is still included. This used to find the upper throughput bound when
 342   tuning the full input pipeline.
 343 
 344   Args:
 345     height: Integer height that will be used to create a fake image tensor.
 346     width: Integer width that will be used to create a fake image tensor.
 347     num_channels: Integer depth that will be used to create a fake image tensor.
 348     num_classes: Number of classes that should be represented in the fake labels
 349       tensor
 350     dtype: Data type for features/images.
 351     drop_remainder: A boolean indicates whether to drop the remainder of the
 352       batches. If True, the batch dimension will be static.
 353 
 354   Returns:
 355     An input_fn that can be used in place of a real one to return a dataset
 356     that can be used for iteration.
 357   """
 358   # pylint: disable=unused-argument
 359   def input_fn(is_training, data_dir, batch_size, *args, **kwargs):
 360     """Returns dataset filled with random data."""
 361     inputs, labels = get_synth_data(height=height,
 362                                     width=width,
 363                                     num_channels=num_channels,
 364                                     num_classes=num_classes,
 365                                     dtype=dtype)
 366     # Cast to float32 for Keras model.
 367     labels = tf.cast(labels, dtype=tf.float32)
 368     data = tf.data.Dataset.from_tensors((inputs, labels)).repeat()
 369 
 370     # `drop_remainder` will make dataset produce outputs with known shapes.
 371     data = data.batch(batch_size, drop_remainder=drop_remainder)
 372     data = data.prefetch(buffer_size=tf.data.experimental.AUTOTUNE)
 373     return data
 374 
 375   return input_fn
