  36 def build_model():
  37   """Constructs the ML model used to predict handwritten digits."""
  38 
  39   image = tf.keras.layers.Input(shape=(28, 28, 1))
  40 
  41   y = tf.keras.layers.Conv2D(filters=32,
  42                              kernel_size=5,
  43                              padding='same',
  44                              activation='relu')(image)
  45   y = tf.keras.layers.MaxPooling2D(pool_size=(2, 2),
  46                                    strides=(2, 2),
  47                                    padding='same')(y)
  48   y = tf.keras.layers.Conv2D(filters=32,
  49                              kernel_size=5,
  50                              padding='same',
  51                              activation='relu')(y)
  52   y = tf.keras.layers.MaxPooling2D(pool_size=(2, 2),
  53                                    strides=(2, 2),
  54                                    padding='same')(y)
  55   y = tf.keras.layers.Flatten()(y)
  56   y = tf.keras.layers.Dense(1024, activation='relu')(y)
  57   y = tf.keras.layers.Dropout(0.4)(y)
  58 
  59   probs = tf.keras.layers.Dense(10, activation='softmax')(y)
  60 
  61   model = tf.keras.models.Model(image, probs, name='mnist')
  62 
  63   return model

  66 @tfds.decode.make_decoder(output_dtype=tf.float32)
  67 def decode_image(example, feature):
  68   """Convert image to float32 and normalize from [0, 255] to [0.0, 1.0]."""
  69   return tf.cast(feature.decode_example(example), dtype=tf.float32) / 255

  72 def run(flags_obj, datasets_override=None, strategy_override=None):
  73   """Run MNIST model training and eval loop using native Keras APIs.
  74 
  75   Args:
  76     flags_obj: An object containing parsed flag values.
  77     datasets_override: A pair of `tf.data.Dataset` objects to train the model,
  78                        representing the train and test sets.
  79     strategy_override: A `tf.distribute.Strategy` object to use for model.
  80 
  81   Returns:
  82     Dictionary of training and eval stats.
  83   """
  84   strategy = strategy_override or distribution_utils.get_distribution_strategy(
  85       distribution_strategy=flags_obj.distribution_strategy,
  86       num_gpus=flags_obj.num_gpus,
  87       tpu_address=flags_obj.tpu)
  88 
  89   strategy_scope = distribution_utils.get_strategy_scope(strategy)
  90 
  91   mnist = tfds.builder('mnist', data_dir=flags_obj.data_dir)
  92   if flags_obj.download:
  93     mnist.download_and_prepare()
  94 
  95   mnist_train, mnist_test = datasets_override or mnist.as_dataset(
  96       split=['train', 'test'],
  97       decoders={'image': decode_image()},  # pylint: disable=no-value-for-parameter
  98       as_supervised=True)
  99   train_input_dataset = mnist_train.cache().repeat().shuffle(
 100       buffer_size=50000).batch(flags_obj.batch_size)
 101   eval_input_dataset = mnist_test.cache().repeat().batch(flags_obj.batch_size)
 102 
 103   with strategy_scope:
 104     lr_schedule = tf.keras.optimizers.schedules.ExponentialDecay(
 105         0.05, decay_steps=100000, decay_rate=0.96)
 106     optimizer = tf.keras.optimizers.SGD(learning_rate=lr_schedule)
 107 
 108     model = build_model()
 109     model.compile(
 110         optimizer=optimizer,
 111         loss='sparse_categorical_crossentropy',
 112         metrics=['sparse_categorical_accuracy'])
 113 
 114   num_train_examples = mnist.info.splits['train'].num_examples
 115   train_steps = num_train_examples // flags_obj.batch_size
 116   train_epochs = flags_obj.train_epochs
 117 
 118   ckpt_full_path = os.path.join(flags_obj.model_dir, 'model.ckpt-{epoch:04d}')
 119   callbacks = [
 120       tf.keras.callbacks.ModelCheckpoint(
 121           ckpt_full_path, save_weights_only=True),
 122       tf.keras.callbacks.TensorBoard(log_dir=flags_obj.model_dir),
 123   ]
 124 
 125   num_eval_examples = mnist.info.splits['test'].num_examples
 126   num_eval_steps = num_eval_examples // flags_obj.batch_size
 127 
 128   history = model.fit(
 129       train_input_dataset,
 130       epochs=train_epochs,
 131       steps_per_epoch=train_steps,
 132       callbacks=callbacks,
 133       validation_steps=num_eval_steps,
 134       validation_data=eval_input_dataset,
 135       validation_freq=flags_obj.epochs_between_evals)
 136 
 137   export_path = os.path.join(flags_obj.model_dir, 'saved_model')
 138   model.save(export_path, include_optimizer=False)
 139 
 140   eval_output = model.evaluate(
 141       eval_input_dataset, steps=num_eval_steps, verbose=2)
 142 
 143   stats = common.build_stats(history, eval_output, callbacks)
 144   return stats

 147 def define_mnist_flags():
 148   """Define command line flags for MNIST model."""
 149   flags_core.define_base(
 150       clean=True,
 151       num_gpu=True,
 152       train_epochs=True,
 153       epochs_between_evals=True,
 154       distribution_strategy=True)
 155   flags_core.define_device()
 156   flags_core.define_distribution()
 157   flags.DEFINE_bool('download', False,
 158                     'Whether to download data to `--data_dir`.')
 159   FLAGS.set_default('batch_size', 1024)
