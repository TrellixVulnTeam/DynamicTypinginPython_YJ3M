  29 def get_dtype_test_cases():
  30   return [
  31       ('uint8', tf.uint8),
  32       ('int32', tf.int32),
  33       ('float16', tf.float16),
  34       ('float32', tf.float32),
  35   ]
  36 
  37 
  38 @parameterized.named_parameters(get_dtype_test_cases())
  39 class TransformsTest(parameterized.TestCase, tf.test.TestCase):
  40   """Basic tests for fundamental transformations."""
  41 
  42   def test_to_from_4d(self, dtype):
  43     for shape in [(10, 10), (10, 10, 10), (10, 10, 10, 10)]:
  44       original_ndims = len(shape)
  45       image = tf.zeros(shape, dtype=dtype)
  46       image_4d = augment.to_4d(image)
  47       self.assertEqual(4, tf.rank(image_4d))
  48       self.assertAllEqual(image, augment.from_4d(image_4d, original_ndims))
  49 
  50   def test_transform(self, dtype):
  51     image = tf.constant([[1, 2], [3, 4]], dtype=dtype)
  52     self.assertAllEqual(augment.transform(image, transforms=[1]*8),
  53                         [[4, 4], [4, 4]])
  54 
  55   def test_translate(self, dtype):
  56     image = tf.constant(
  57         [[1, 0, 1, 0],
  58          [0, 1, 0, 1],
  59          [1, 0, 1, 0],
  60          [0, 1, 0, 1]],
  61         dtype=dtype)
  62     translations = [-1, -1]
  63     translated = augment.translate(image=image,
  64                                    translations=translations)
  65     expected = [
  66         [1, 0, 1, 1],
  67         [0, 1, 0, 0],
  68         [1, 0, 1, 1],
  69         [1, 0, 1, 1]]
  70     self.assertAllEqual(translated, expected)
  71 
  72   def test_translate_shapes(self, dtype):
  73     translation = [0, 0]
  74     for shape in [(3, 3), (5, 5), (224, 224, 3)]:
  75       image = tf.zeros(shape, dtype=dtype)
  76       self.assertAllEqual(image, augment.translate(image, translation))
  77 
  78   def test_translate_invalid_translation(self, dtype):
  79     image = tf.zeros((1, 1), dtype=dtype)
  80     invalid_translation = [[[1, 1]]]
  81     with self.assertRaisesRegex(TypeError, 'rank 1 or 2'):
  82       _ = augment.translate(image, invalid_translation)
  83 
  84   def test_rotate(self, dtype):
  85     image = tf.reshape(tf.cast(tf.range(9), dtype), (3, 3))
  86     rotation = 90.
  87     transformed = augment.rotate(image=image, degrees=rotation)
  88     expected = [[2, 5, 8],
  89                 [1, 4, 7],
  90                 [0, 3, 6]]
  91     self.assertAllEqual(transformed, expected)
  92 
  93   def test_rotate_shapes(self, dtype):
  94     degrees = 0.
  95     for shape in [(3, 3), (5, 5), (224, 224, 3)]:
  96       image = tf.zeros(shape, dtype=dtype)
  97       self.assertAllEqual(image, augment.rotate(image, degrees))

 100 class AutoaugmentTest(tf.test.TestCase):
 101 
 102   def test_autoaugment(self):
 103     """Smoke test to be sure there are no syntax errors."""
 104     image = tf.zeros((224, 224, 3), dtype=tf.uint8)
 105 
 106     augmenter = augment.AutoAugment()
 107     aug_image = augmenter.distort(image)
 108 
 109     self.assertEqual((224, 224, 3), aug_image.shape)
 110 
 111   def test_randaug(self):
 112     """Smoke test to be sure there are no syntax errors."""
 113     image = tf.zeros((224, 224, 3), dtype=tf.uint8)
 114 
 115     augmenter = augment.RandAugment()
 116     aug_image = augmenter.distort(image)
 117 
 118     self.assertEqual((224, 224, 3), aug_image.shape)
 119 
 120   def test_all_policy_ops(self):
 121     """Smoke test to be sure all augmentation functions can execute."""
 122 
 123     prob = 1
 124     magnitude = 10
 125     replace_value = [128] * 3
 126     cutout_const = 100
 127     translate_const = 250
 128 
 129     image = tf.ones((224, 224, 3), dtype=tf.uint8)
 130 
 131     for op_name in augment.NAME_TO_FUNC:
 132       func, _, args = augment._parse_policy_info(op_name,
 133                                                  prob,
 134                                                  magnitude,
 135                                                  replace_value,
 136                                                  cutout_const,
 137                                                  translate_const)
 138       image = func(image, *args)
 139 
 140     self.assertEqual((224, 224, 3), image.shape)
