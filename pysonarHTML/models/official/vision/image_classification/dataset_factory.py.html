  41 class AugmentConfig(base_config.Config):
  42   """Configuration for image augmenters.
  43 
  44   Attributes:
  45     name: The name of the image augmentation to use. Possible options are
  46       None (default), 'autoaugment', or 'randaugment'.
  47     params: Any paramaters used to initialize the augmenter.
  48   """
  49   name: Optional[str] = None
  50   params: Optional[Mapping[str, Any]] = None
  51 
  52   def build(self) -> augment.ImageAugment:
  53     """Build the augmenter using this config."""
  54     params = self.params or {}
  55     augmenter = AUGMENTERS.get(self.name, None)
  56     return augmenter(**params) if augmenter is not None else None

  59 @dataclass
  60 class DatasetConfig(base_config.Config):
  61   """The base configuration for building datasets.
  62 
  63   Attributes:
  64     name: The name of the Dataset. Usually should correspond to a TFDS dataset.
  65     data_dir: The path where the dataset files are stored, if available.
  66     filenames: Optional list of strings representing the TFRecord names.
  67     builder: The builder type used to load the dataset. Value should be one of
  68       'tfds' (load using TFDS), 'records' (load from TFRecords), or 'synthetic'
  69       (generate dummy synthetic data without reading from files).
  70     split: The split of the dataset. Usually 'train', 'validation', or 'test'.
  71     image_size: The size of the image in the dataset. This assumes that
  72       `width` == `height`. Set to 'infer' to infer the image size from TFDS
  73       info. This requires `name` to be a registered dataset in TFDS.
  74     num_classes: The number of classes given by the dataset. Set to 'infer'
  75       to infer the image size from TFDS info. This requires `name` to be a
  76       registered dataset in TFDS.
  77     num_channels: The number of channels given by the dataset. Set to 'infer'
  78       to infer the image size from TFDS info. This requires `name` to be a
  79       registered dataset in TFDS.
  80     num_examples: The number of examples given by the dataset. Set to 'infer'
  81       to infer the image size from TFDS info. This requires `name` to be a
  82       registered dataset in TFDS.
  83     batch_size: The base batch size for the dataset.
  84     use_per_replica_batch_size: Whether to scale the batch size based on
  85       available resources. If set to `True`, the dataset builder will return
  86       batch_size multiplied by `num_devices`, the number of device replicas
  87       (e.g., the number of GPUs or TPU cores). This setting should be `True` if
  88       the strategy argument is passed to `build()` and `num_devices > 1`.
  89     num_devices: The number of replica devices to use. This should be set by
  90       `strategy.num_replicas_in_sync` when using a distribution strategy.
  91     dtype: The desired dtype of the dataset. This will be set during
  92       preprocessing.
  93     one_hot: Whether to apply one hot encoding. Set to `True` to be able to use
  94       label smoothing.
  95     augmenter: The augmenter config to use. No augmentation is used by default.
  96     download: Whether to download data using TFDS.
  97     shuffle_buffer_size: The buffer size used for shuffling training data.
  98     file_shuffle_buffer_size: The buffer size used for shuffling raw training
  99       files.
 100     skip_decoding: Whether to skip image decoding when loading from TFDS.
 101     cache: whether to cache to dataset examples. Can be used to avoid re-reading
 102       from disk on the second epoch. Requires significant memory overhead.
 103     tf_data_service: The URI of a tf.data service to offload preprocessing onto
 104       during training. The URI should be in the format "protocol://address",
 105       e.g. "grpc://tf-data-service:5050".
 106     mean_subtract: whether or not to apply mean subtraction to the dataset.
 107     standardize: whether or not to apply standardization to the dataset.
 108   """
 109   name: Optional[str] = None
 110   data_dir: Optional[str] = None
 111   filenames: Optional[List[str]] = None
 112   builder: str = 'tfds'
 113   split: str = 'train'
 114   image_size: Union[int, str] = 'infer'
 115   num_classes: Union[int, str] = 'infer'
 116   num_channels: Union[int, str] = 'infer'
 117   num_examples: Union[int, str] = 'infer'
 118   batch_size: int = 128
 119   use_per_replica_batch_size: bool = True
 120   num_devices: int = 1
 121   dtype: str = 'float32'
 122   one_hot: bool = True
 123   augmenter: AugmentConfig = AugmentConfig()
 124   download: bool = False
 125   shuffle_buffer_size: int = 10000
 126   file_shuffle_buffer_size: int = 1024
 127   skip_decoding: bool = True
 128   cache: bool = False
 129   tf_data_service: Optional[str] = None
 130   mean_subtract: bool = False
 131   standardize: bool = False
 132 
 133   @property
 134   def has_data(self):
 135     """Whether this dataset is has any data associated with it."""
 136     return self.name or self.data_dir or self.filenames

 159 class DatasetBuilder:
 160   """An object for building datasets.
 161 
 162   Allows building various pipelines fetching examples, preprocessing, etc.
 163   Maintains additional state information calculated from the dataset, i.e.,
 164   training set split, batch size, and number of steps (batches).
 165   """
 166 
 167   def __init__(self, config: DatasetConfig, **overrides: Any):
 168     """Initialize the builder from the config."""
 169     self.config = config.replace(**overrides)
 170     self.builder_info = None
 171 
 172     if self.config.augmenter is not None:
 173       logging.info('Using augmentation: %s', self.config.augmenter.name)
 174       self.augmenter = self.config.augmenter.build()
 175     else:
 176       self.augmenter = None
 177 
 178   @property
 179   def is_training(self) -> bool:
 180     """Whether this is the training set."""
 181     return self.config.split == 'train'
 182 
 183   @property
 184   def batch_size(self) -> int:
 185     """The batch size, multiplied by the number of replicas (if configured)."""
 186     if self.config.use_per_replica_batch_size:
 187       return self.config.batch_size * self.config.num_devices
 188     else:
 189       return self.config.batch_size
 190 
 191   @property
 192   def global_batch_size(self):
 193     """The global batch size across all replicas."""
 194     return self.batch_size
 195 
 196   @property
 197   def local_batch_size(self):
 198     """The base unscaled batch size."""
 199     if self.config.use_per_replica_batch_size:
 200       return self.config.batch_size
 201     else:
 202       return self.config.batch_size // self.config.num_devices
 203 
 204   @property
 205   def num_steps(self) -> int:
 206     """The number of steps (batches) to exhaust this dataset."""
 207     # Always divide by the global batch size to get the correct # of steps
 208     return self.num_examples // self.global_batch_size
 209 
 210   @property
 211   def dtype(self) -> tf.dtypes.DType:
 212     """Converts the config's dtype string to a tf dtype.
 213 
 214     Returns:
 215       A mapping from string representation of a dtype to the `tf.dtypes.DType`.
 216 
 217     Raises:
 218       ValueError if the config's dtype is not supported.
 219 
 220     """
 221     dtype_map = {
 222         'float32': tf.float32,
 223         'bfloat16': tf.bfloat16,
 224         'float16': tf.float16,
 225         'fp32': tf.float32,
 226         'bf16': tf.bfloat16,
 227     }
 228     try:
 229       return dtype_map[self.config.dtype]
 230     except:
 231       raise ValueError('Invalid DType provided. Supported types: {}'.format(
 232           dtype_map.keys()))
 233 
 234   @property
 235   def image_size(self) -> int:
 236     """The size of each image (can be inferred from the dataset)."""
 237 
 238     if self.config.image_size == 'infer':
 239       return self.info.features['image'].shape[0]
 240     else:
 241       return int(self.config.image_size)
 242 
 243   @property
 244   def num_channels(self) -> int:
 245     """The number of image channels (can be inferred from the dataset)."""
 246     if self.config.num_channels == 'infer':
 247       return self.info.features['image'].shape[-1]
 248     else:
 249       return int(self.config.num_channels)
 250 
 251   @property
 252   def num_examples(self) -> int:
 253     """The number of examples (can be inferred from the dataset)."""
 254     if self.config.num_examples == 'infer':
 255       return self.info.splits[self.config.split].num_examples
 256     else:
 257       return int(self.config.num_examples)
 258 
 259   @property
 260   def num_classes(self) -> int:
 261     """The number of classes (can be inferred from the dataset)."""
 262     if self.config.num_classes == 'infer':
 263       return self.info.features['label'].num_classes
 264     else:
 265       return int(self.config.num_classes)
 266 
 267   @property
 268   def info(self) -> tfds.core.DatasetInfo:
 269     """The TFDS dataset info, if available."""
 270     if self.builder_info is None:
 271       self.builder_info = tfds.builder(self.config.name).info
 272     return self.builder_info
 273 
 274   def build(self, strategy: tf.distribute.Strategy = None) -> tf.data.Dataset:
 275     """Construct a dataset end-to-end and return it using an optional strategy.
 276 
 277     Args:
 278       strategy: a strategy that, if passed, will distribute the dataset
 279         according to that strategy. If passed and `num_devices > 1`,
 280         `use_per_replica_batch_size` must be set to `True`.
 281 
 282     Returns:
 283       A TensorFlow dataset outputting batched images and labels.
 284     """
 285     if strategy:
 286       if strategy.num_replicas_in_sync != self.config.num_devices:
 287         logging.warn('Passed a strategy with %d devices, but expected'
 288                      '%d devices.',
 289                      strategy.num_replicas_in_sync,
 290                      self.config.num_devices)
 291       dataset = strategy.experimental_distribute_datasets_from_function(
 292           self._build)
 293     else:
 294       dataset = self._build()
 295 
 296     return dataset
 297 
 298   def _build(self, input_context: tf.distribute.InputContext = None
 299              ) -> tf.data.Dataset:
 300     """Construct a dataset end-to-end and return it.
 301 
 302     Args:
 303       input_context: An optional context provided by `tf.distribute` for
 304         cross-replica training.
 305 
 306     Returns:
 307       A TensorFlow dataset outputting batched images and labels.
 308     """
 309     builders = {
 310         'tfds': self.load_tfds,
 311         'records': self.load_records,
 312         'synthetic': self.load_synthetic,
 313     }
 314 
 315     builder = builders.get(self.config.builder, None)
 316 
 317     if builder is None:
 318       raise ValueError('Unknown builder type {}'.format(self.config.builder))
 319 
 320     self.input_context = input_context
 321     dataset = builder()
 322     dataset = self.pipeline(dataset)
 323 
 324     return dataset
 325 
 326   def load_tfds(self) -> tf.data.Dataset:
 327     """Return a dataset loading files from TFDS."""
 328 
 329     logging.info('Using TFDS to load data.')
 330 
 331     builder = tfds.builder(self.config.name,
 332                            data_dir=self.config.data_dir)
 333 
 334     if self.config.download:
 335       builder.download_and_prepare()
 336 
 337     decoders = {}
 338 
 339     if self.config.skip_decoding:
 340       decoders['image'] = tfds.decode.SkipDecoding()
 341 
 342     read_config = tfds.ReadConfig(
 343         interleave_cycle_length=10,
 344         interleave_block_length=1,
 345         input_context=self.input_context)
 346 
 347     dataset = builder.as_dataset(
 348         split=self.config.split,
 349         as_supervised=True,
 350         shuffle_files=True,
 351         decoders=decoders,
 352         read_config=read_config)
 353 
 354     return dataset
 355 
 356   def load_records(self) -> tf.data.Dataset:
 357     """Return a dataset loading files with TFRecords."""
 358     logging.info('Using TFRecords to load data.')
 359     if self.config.filenames is None:
 360       if self.config.data_dir is None:
 361         raise ValueError('Dataset must specify a path for the data files.')
 362 
 363       file_pattern = os.path.join(self.config.data_dir,
 364                                   '{}*'.format(self.config.split))
 365       dataset = tf.data.Dataset.list_files(file_pattern, shuffle=False)
 366     else:
 367       dataset = tf.data.Dataset.from_tensor_slices(self.config.filenames)
 368 
 369     return dataset
 370 
 371   def load_synthetic(self) -> tf.data.Dataset:
 372     """Return a dataset generating dummy synthetic data."""
 373     logging.info('Generating a synthetic dataset.')
 374 
 375     def generate_data(_):
 376       image = tf.zeros([self.image_size, self.image_size, self.num_channels],
 377                        dtype=self.dtype)
 378       label = tf.zeros([1], dtype=tf.int32)
 379       return image, label
 380 
 381     dataset = tf.data.Dataset.range(1)
 382     dataset = dataset.repeat()
 383     dataset = dataset.map(generate_data,
 384                           num_parallel_calls=tf.data.experimental.AUTOTUNE)
 385     return dataset
 386 
 387   def pipeline(self, dataset: tf.data.Dataset) -> tf.data.Dataset:
 388     """Build a pipeline fetching, shuffling, and preprocessing the dataset.
 389 
 390     Args:
 391       dataset: A `tf.data.Dataset` that loads raw files.
 392 
 393     Returns:
 394       A TensorFlow dataset outputting batched images and labels.
 395     """
 396     if (self.config.builder != 'tfds' and self.input_context
 397         and self.input_context.num_input_pipelines > 1):
 398       dataset = dataset.shard(self.input_context.num_input_pipelines,
 399                               self.input_context.input_pipeline_id)
 400       logging.info('Sharding the dataset: input_pipeline_id=%d '
 401                    'num_input_pipelines=%d',
 402                    self.input_context.num_input_pipelines,
 403                    self.input_context.input_pipeline_id)
 404 
 405     if self.is_training and self.config.builder == 'records':
 406       # Shuffle the input files.
 407       dataset.shuffle(buffer_size=self.config.file_shuffle_buffer_size)
 408 
 409     if self.is_training and not self.config.cache:
 410       dataset = dataset.repeat()
 411 
 412     if self.config.builder == 'records':
 413       # Read the data from disk in parallel
 414       dataset = dataset.interleave(
 415           tf.data.TFRecordDataset,
 416           cycle_length=10,
 417           block_length=1,
 418           num_parallel_calls=tf.data.experimental.AUTOTUNE)
 419 
 420     if self.config.cache:
 421       dataset = dataset.cache()
 422 
 423     if self.is_training:
 424       dataset = dataset.shuffle(self.config.shuffle_buffer_size)
 425       dataset = dataset.repeat()
 426 
 427     # Parse, pre-process, and batch the data in parallel
 428     if self.config.builder == 'records':
 429       preprocess = self.parse_record
 430     else:
 431       preprocess = self.preprocess
 432     dataset = dataset.map(preprocess,
 433                           num_parallel_calls=tf.data.experimental.AUTOTUNE)
 434 
 435     if self.input_context and self.config.num_devices > 1:
 436       if not self.config.use_per_replica_batch_size:
 437         raise ValueError(
 438             'The builder does not support a global batch size with more than '
 439             'one replica. Got {} replicas. Please set a '
 440             '`per_replica_batch_size` and enable '
 441             '`use_per_replica_batch_size=True`.'.format(
 442                 self.config.num_devices))
 443 
 444       # The batch size of the dataset will be multiplied by the number of
 445       # replicas automatically when strategy.distribute_datasets_from_function
 446       # is called, so we use local batch size here.
 447       dataset = dataset.batch(self.local_batch_size,
 448                               drop_remainder=self.is_training)
 449     else:
 450       dataset = dataset.batch(self.global_batch_size,
 451                               drop_remainder=self.is_training)
 452 
 453     # Prefetch overlaps in-feed with training
 454     dataset = dataset.prefetch(tf.data.experimental.AUTOTUNE)
 455 
 456     if self.config.tf_data_service:
 457       if not hasattr(tf.data.experimental, 'service'):
 458         raise ValueError('The tf_data_service flag requires Tensorflow version '
 459                          '>= 2.3.0, but the version is {}'.format(
 460                              tf.__version__))
 461       dataset = dataset.apply(
 462           tf.data.experimental.service.distribute(
 463               processing_mode='parallel_epochs',
 464               service=self.config.tf_data_service,
 465               job_name='resnet_train'))
 466       dataset = dataset.prefetch(buffer_size=tf.data.experimental.AUTOTUNE)
 467 
 468     return dataset

 470   def parse_record(self, record: tf.Tensor) -> Tuple[tf.Tensor, tf.Tensor]:
 471     """Parse an ImageNet record from a serialized string Tensor."""
 472     keys_to_features = {
 473         'image/encoded':
 474             tf.io.FixedLenFeature((), tf.string, ''),
 475         'image/format':
 476             tf.io.FixedLenFeature((), tf.string, 'jpeg'),
 477         'image/class/label':
 478             tf.io.FixedLenFeature([], tf.int64, -1),
 479         'image/class/text':
 480             tf.io.FixedLenFeature([], tf.string, ''),
 481         'image/object/bbox/xmin':
 482             tf.io.VarLenFeature(dtype=tf.float32),
 483         'image/object/bbox/ymin':
 484             tf.io.VarLenFeature(dtype=tf.float32),
 485         'image/object/bbox/xmax':
 486             tf.io.VarLenFeature(dtype=tf.float32),
 487         'image/object/bbox/ymax':
 488             tf.io.VarLenFeature(dtype=tf.float32),
 489         'image/object/class/label':
 490             tf.io.VarLenFeature(dtype=tf.int64),
 491     }
 492 
 493     parsed = tf.io.parse_single_example(record, keys_to_features)
 494 
 495     label = tf.reshape(parsed['image/class/label'], shape=[1])
 496 
 497     # Subtract one so that labels are in [0, 1000)
 498     label -= 1
 499 
 500     image_bytes = tf.reshape(parsed['image/encoded'], shape=[])
 501     image, label = self.preprocess(image_bytes, label)
 502 
 503     return image, label
 504 
 505   def preprocess(self, image: tf.Tensor, label: tf.Tensor
 506                 ) -> Tuple[tf.Tensor, tf.Tensor]:
 507     """Apply image preprocessing and augmentation to the image and label."""
 508     if self.is_training:
 509       image = preprocessing.preprocess_for_train(
 510           image,
 511           image_size=self.image_size,
 512           mean_subtract=self.config.mean_subtract,
 513           standardize=self.config.standardize,
 514           dtype=self.dtype,
 515           augmenter=self.augmenter)
 516     else:
 517       image = preprocessing.preprocess_for_eval(
 518           image,
 519           image_size=self.image_size,
 520           num_channels=self.num_channels,
 521           mean_subtract=self.config.mean_subtract,
 522           standardize=self.config.standardize,
 523           dtype=self.dtype)
 524 
 525     label = tf.cast(label, tf.int32)
 526     if self.config.one_hot:
 527       label = tf.one_hot(label, self.num_classes)
 528       label = tf.reshape(label, [self.num_classes])
 529 
 530     return image, label
 531 
 532   @classmethod
 533   def from_params(cls, *args, **kwargs):
 534     """Construct a dataset builder from a default config and any overrides."""
 535     config = DatasetConfig.from_args(*args, **kwargs)
 536     return cls(config)
