  29 class WarmupDecaySchedule(tf.keras.optimizers.schedules.LearningRateSchedule):
  30   """A wrapper for LearningRateSchedule that includes warmup steps."""
  31 
  32   def __init__(
  33       self,
  34       lr_schedule: tf.keras.optimizers.schedules.LearningRateSchedule,
  35       warmup_steps: int):
  36     """Add warmup decay to a learning rate schedule.
  37 
  38     Args:
  39       lr_schedule: base learning rate scheduler
  40       warmup_steps: number of warmup steps
  41 
  42     """
  43     super(WarmupDecaySchedule, self).__init__()
  44     self._lr_schedule = lr_schedule
  45     self._warmup_steps = warmup_steps
  46 
  47   def __call__(self, step: int):
  48     lr = self._lr_schedule(step)
  49     if self._warmup_steps:
  50       initial_learning_rate = tf.convert_to_tensor(
  51           self._lr_schedule.initial_learning_rate, name="initial_learning_rate")
  52       dtype = initial_learning_rate.dtype
  53       global_step_recomp = tf.cast(step, dtype)
  54       warmup_steps = tf.cast(self._warmup_steps, dtype)
  55       warmup_lr = initial_learning_rate * global_step_recomp / warmup_steps
  56       lr = tf.cond(global_step_recomp < warmup_steps,
  57                    lambda: warmup_lr,
  58                    lambda: lr)
  59     return lr
  60 
  61   def get_config(self) -> Mapping[str, Any]:
  62     config = self._lr_schedule.get_config()
  63     config.update({
  64         "warmup_steps": self._warmup_steps,
  65     })
  66     return config

  71 class PiecewiseConstantDecayWithWarmup(
  72     tf.keras.optimizers.schedules.LearningRateSchedule):
  73   """Piecewise constant decay with warmup schedule."""
  74 
  75   def __init__(self,
  76                batch_size: int,
  77                epoch_size: int,
  78                warmup_epochs: int,
  79                boundaries: List[int],
  80                multipliers: List[float]):
  81     """Piecewise constant decay with warmup.
  82 
  83     Args:
  84       batch_size: The training batch size used in the experiment.
  85       epoch_size: The size of an epoch, or the number of examples in an epoch.
  86       warmup_epochs: The number of warmup epochs to apply.
  87       boundaries: The list of floats with strictly increasing entries.
  88       multipliers: The list of multipliers/learning rates to use for the
  89         piecewise portion. The length must be 1 less than that of boundaries.
  90 
  91     """
  92     super(PiecewiseConstantDecayWithWarmup, self).__init__()
  93     if len(boundaries) != len(multipliers) - 1:
  94       raise ValueError("The length of boundaries must be 1 less than the "
  95                        "length of multipliers")
  96 
  97     base_lr_batch_size = 256
  98     steps_per_epoch = epoch_size // batch_size
  99 
 100     self._rescaled_lr = BASE_LEARNING_RATE * batch_size / base_lr_batch_size
 101     self._step_boundaries = [float(steps_per_epoch) * x for x in boundaries]
 102     self._lr_values = [self._rescaled_lr * m for m in multipliers]
 103     self._warmup_steps = warmup_epochs * steps_per_epoch
 104 
 105   def __call__(self, step: int):
 106     """Compute learning rate at given step."""
 107     def warmup_lr():
 108       return self._rescaled_lr * (
 109           step / tf.cast(self._warmup_steps, tf.float32))
 110     def piecewise_lr():
 111       return tf.compat.v1.train.piecewise_constant(
 112           tf.cast(step, tf.float32), self._step_boundaries, self._lr_values)
 113     return tf.cond(step < self._warmup_steps, warmup_lr, piecewise_lr)
 114 
 115   def get_config(self) -> Mapping[str, Any]:
 116     return {
 117         "rescaled_lr": self._rescaled_lr,
 118         "step_boundaries": self._step_boundaries,
 119         "lr_values": self._lr_values,
 120         "warmup_steps": self._warmup_steps,
 121     }

 124 class CosineDecayWithWarmup(tf.keras.optimizers.schedules.LearningRateSchedule):
 125   """Class to generate learning rate tensor."""
 126 
 127   def __init__(self, batch_size: int, total_steps: int, warmup_steps: int):
 128     """Creates the consine learning rate tensor with linear warmup.
 129 
 130     Args:
 131       batch_size: The training batch size used in the experiment.
 132       total_steps: Total training steps.
 133       warmup_steps: Steps for the warm up period.
 134     """
 135     super(CosineDecayWithWarmup, self).__init__()
 136     base_lr_batch_size = 256
 137     self._total_steps = total_steps
 138     self._init_learning_rate = BASE_LEARNING_RATE * batch_size / base_lr_batch_size
 139     self._warmup_steps = warmup_steps
 140 
 141   def __call__(self, global_step: int):
 142     global_step = tf.cast(global_step, dtype=tf.float32)
 143     warmup_steps = self._warmup_steps
 144     init_lr = self._init_learning_rate
 145     total_steps = self._total_steps
 146 
 147     linear_warmup = global_step / warmup_steps * init_lr
 148 
 149     cosine_learning_rate = init_lr * (tf.cos(np.pi *
 150                                              (global_step - warmup_steps) /
 151                                              (total_steps - warmup_steps)) +
 152                                       1.0) / 2.0
 153 
 154     learning_rate = tf.where(global_step < warmup_steps, linear_warmup,
 155                              cosine_learning_rate)
 156     return learning_rate
 157 
 158   def get_config(self):
 159     return {
 160         "total_steps": self._total_steps,
 161         "warmup_learning_rate": self._warmup_learning_rate,
 162         "warmup_steps": self._warmup_steps,
 163         "init_learning_rate": self._init_learning_rate,
 164     }
