  29 def visualize_images_with_bounding_boxes(images, box_outputs, step,
  30                                          summary_writer):
  31   """Records subset of evaluation images with bounding boxes."""
  32   image_shape = tf.shape(images[0])
  33   image_height = tf.cast(image_shape[0], tf.float32)
  34   image_width = tf.cast(image_shape[1], tf.float32)
  35   normalized_boxes = normalize_boxes(box_outputs, [image_height, image_width])
  36 
  37   bounding_box_color = tf.constant([[1.0, 1.0, 0.0, 1.0]])
  38   image_summary = tf.image.draw_bounding_boxes(images, normalized_boxes,
  39                                                bounding_box_color)
  40   with summary_writer.as_default():
  41     tf.summary.image('bounding_box_summary', image_summary, step=step)
  42     summary_writer.flush()

  45 def yxyx_to_xywh(boxes):
  46   """Converts boxes from ymin, xmin, ymax, xmax to xmin, ymin, width, height.
  47 
  48   Args:
  49     boxes: a numpy array whose last dimension is 4 representing the coordinates
  50       of boxes in ymin, xmin, ymax, xmax order.
  51 
  52   Returns:
  53     boxes: a numpy array whose shape is the same as `boxes` in new format.
  54 
  55   Raises:
  56     ValueError: If the last dimension of boxes is not 4.
  57   """
  58   if boxes.shape[-1] != 4:
  59     raise ValueError('boxes.shape[-1] is {:d}, but must be 4.'.format(
  60         boxes.shape[-1]))
  61 
  62   boxes_ymin = boxes[..., 0]
  63   boxes_xmin = boxes[..., 1]
  64   boxes_width = boxes[..., 3] - boxes[..., 1]
  65   boxes_height = boxes[..., 2] - boxes[..., 0]
  66   new_boxes = np.stack([boxes_xmin, boxes_ymin, boxes_width, boxes_height],
  67                        axis=-1)
  68 
  69   return new_boxes

  72 def jitter_boxes(boxes, noise_scale=0.025):
  73   """Jitter the box coordinates by some noise distribution.
  74 
  75   Args:
  76     boxes: a tensor whose last dimension is 4 representing the coordinates of
  77       boxes in ymin, xmin, ymax, xmax order.
  78     noise_scale: a python float which specifies the magnitude of noise. The rule
  79       of thumb is to set this between (0, 0.1]. The default value is found to
  80       mimic the noisy detections best empirically.
  81 
  82   Returns:
  83     jittered_boxes: a tensor whose shape is the same as `boxes` representing
  84       the jittered boxes.
  85 
  86   Raises:
  87     ValueError: If the last dimension of boxes is not 4.
  88   """
  89   if boxes.shape[-1] != 4:
  90     raise ValueError('boxes.shape[-1] is {:d}, but must be 4.'.format(
  91         boxes.shape[-1]))
  92 
  93   with tf.name_scope('jitter_boxes'):
  94     bbox_jitters = tf.random.normal(boxes.get_shape(), stddev=noise_scale)
  95     ymin = boxes[..., 0:1]
  96     xmin = boxes[..., 1:2]
  97     ymax = boxes[..., 2:3]
  98     xmax = boxes[..., 3:4]
  99     width = xmax - xmin
 100     height = ymax - ymin
 101     new_center_x = (xmin + xmax) / 2.0 + bbox_jitters[..., 0:1] * width
 102     new_center_y = (ymin + ymax) / 2.0 + bbox_jitters[..., 1:2] * height
 103     new_width = width * tf.math.exp(bbox_jitters[..., 2:3])
 104     new_height = height * tf.math.exp(bbox_jitters[..., 3:4])
 105     jittered_boxes = tf.concat([
 106         new_center_y - new_height * 0.5, new_center_x - new_width * 0.5,
 107         new_center_y + new_height * 0.5, new_center_x + new_width * 0.5
 108     ],
 109                                axis=-1)
 110 
 111     return jittered_boxes

 114 def normalize_boxes(boxes, image_shape):
 115   """Converts boxes to the normalized coordinates.
 116 
 117   Args:
 118     boxes: a tensor whose last dimension is 4 representing the coordinates
 119       of boxes in ymin, xmin, ymax, xmax order.
 120     image_shape: a list of two integers, a two-element vector or a tensor such
 121       that all but the last dimensions are `broadcastable` to `boxes`. The last
 122       dimension is 2, which represents [height, width].
 123 
 124   Returns:
 125     normalized_boxes: a tensor whose shape is the same as `boxes` representing
 126       the normalized boxes.
 127 
 128   Raises:
 129     ValueError: If the last dimension of boxes is not 4.
 130   """
 131   if boxes.shape[-1] != 4:
 132     raise ValueError('boxes.shape[-1] is {:d}, but must be 4.'.format(
 133         boxes.shape[-1]))
 134 
 135   with tf.name_scope('normalize_boxes'):
 136     if isinstance(image_shape, list) or isinstance(image_shape, tuple):
 137       height, width = image_shape
 138     else:
 139       image_shape = tf.cast(image_shape, dtype=boxes.dtype)
 140       height = image_shape[..., 0:1]
 141       width = image_shape[..., 1:2]
 142 
 143     ymin = boxes[..., 0:1] / height
 144     xmin = boxes[..., 1:2] / width
 145     ymax = boxes[..., 2:3] / height
 146     xmax = boxes[..., 3:4] / width
 147 
 148     normalized_boxes = tf.concat([ymin, xmin, ymax, xmax], axis=-1)
 149     return normalized_boxes

 152 def denormalize_boxes(boxes, image_shape):
 153   """Converts boxes normalized by [height, width] to pixel coordinates.
 154 
 155   Args:
 156     boxes: a tensor whose last dimension is 4 representing the coordinates
 157       of boxes in ymin, xmin, ymax, xmax order.
 158     image_shape: a list of two integers, a two-element vector or a tensor such
 159       that all but the last dimensions are `broadcastable` to `boxes`. The last
 160       dimension is 2, which represents [height, width].
 161 
 162   Returns:
 163     denormalized_boxes: a tensor whose shape is the same as `boxes` representing
 164       the denormalized boxes.
 165 
 166   Raises:
 167     ValueError: If the last dimension of boxes is not 4.
 168   """
 169   with tf.name_scope('denormalize_boxes'):
 170     if isinstance(image_shape, list) or isinstance(image_shape, tuple):
 171       height, width = image_shape
 172     else:
 173       image_shape = tf.cast(image_shape, dtype=boxes.dtype)
 174       height, width = tf.split(image_shape, 2, axis=-1)
 175 
 176     ymin, xmin, ymax, xmax = tf.split(boxes, 4, axis=-1)
 177     ymin = ymin * height
 178     xmin = xmin * width
 179     ymax = ymax * height
 180     xmax = xmax * width
 181 
 182     denormalized_boxes = tf.concat([ymin, xmin, ymax, xmax], axis=-1)
 183     return denormalized_boxes

 186 def clip_boxes(boxes, image_shape):
 187   """Clips boxes to image boundaries.
 188 
 189   Args:
 190     boxes: a tensor whose last dimension is 4 representing the coordinates
 191       of boxes in ymin, xmin, ymax, xmax order.
 192     image_shape: a list of two integers, a two-element vector or a tensor such
 193       that all but the last dimensions are `broadcastable` to `boxes`. The last
 194       dimension is 2, which represents [height, width].
 195 
 196   Returns:
 197     clipped_boxes: a tensor whose shape is the same as `boxes` representing the
 198       clipped boxes.
 199 
 200   Raises:
 201     ValueError: If the last dimension of boxes is not 4.
 202   """
 203   if boxes.shape[-1] != 4:
 204     raise ValueError('boxes.shape[-1] is {:d}, but must be 4.'.format(
 205         boxes.shape[-1]))
 206 
 207   with tf.name_scope('clip_boxes'):
 208     if isinstance(image_shape, list) or isinstance(image_shape, tuple):
 209       height, width = image_shape
 210       max_length = [height - 1.0, width - 1.0, height - 1.0, width - 1.0]
 211     else:
 212       image_shape = tf.cast(image_shape, dtype=boxes.dtype)
 213       height, width = tf.unstack(image_shape, axis=-1)
 214       max_length = tf.stack(
 215           [height - 1.0, width - 1.0, height - 1.0, width - 1.0], axis=-1)
 216 
 217     clipped_boxes = tf.math.maximum(tf.math.minimum(boxes, max_length), 0.0)
 218     return clipped_boxes

 221 def compute_outer_boxes(boxes, image_shape, scale=1.0):
 222   """Compute outer box encloses an object with a margin.
 223 
 224   Args:
 225     boxes: a tensor whose last dimension is 4 representing the coordinates of
 226       boxes in ymin, xmin, ymax, xmax order.
 227     image_shape: a list of two integers, a two-element vector or a tensor such
 228       that all but the last dimensions are `broadcastable` to `boxes`. The last
 229       dimension is 2, which represents [height, width].
 230     scale: a float number specifying the scale of output outer boxes to input
 231       `boxes`.
 232 
 233   Returns:
 234     outer_boxes: a tensor whose shape is the same as `boxes` representing the
 235       outer boxes.
 236   """
 237   if scale < 1.0:
 238     raise ValueError(
 239         'scale is {}, but outer box scale must be greater than 1.0.'.format(
 240             scale))
 241   centers_y = (boxes[..., 0] + boxes[..., 2]) / 2.0
 242   centers_x = (boxes[..., 1] + boxes[..., 3]) / 2.0
 243   box_height = (boxes[..., 2] - boxes[..., 0]) * scale
 244   box_width = (boxes[..., 3] - boxes[..., 1]) * scale
 245   outer_boxes = tf.stack([
 246       centers_y - box_height / 2.0, centers_x - box_width / 2.0,
 247       centers_y + box_height / 2.0, centers_x + box_width / 2.0
 248   ],
 249                          axis=1)
 250   outer_boxes = clip_boxes(outer_boxes, image_shape)
 251   return outer_boxes

 254 def encode_boxes(boxes, anchors, weights=None):
 255   """Encode boxes to targets.
 256 
 257   Args:
 258     boxes: a tensor whose last dimension is 4 representing the coordinates
 259       of boxes in ymin, xmin, ymax, xmax order.
 260     anchors: a tensor whose shape is the same as, or `broadcastable` to `boxes`,
 261       representing the coordinates of anchors in ymin, xmin, ymax, xmax order.
 262     weights: None or a list of four float numbers used to scale coordinates.
 263 
 264   Returns:
 265     encoded_boxes: a tensor whose shape is the same as `boxes` representing the
 266       encoded box targets.
 267 
 268   Raises:
 269     ValueError: If the last dimension of boxes is not 4.
 270   """
 271   if boxes.shape[-1] != 4:
 272     raise ValueError('boxes.shape[-1] is {:d}, but must be 4.'.format(
 273         boxes.shape[-1]))
 274 
 275   with tf.name_scope('encode_boxes'):
 276     boxes = tf.cast(boxes, dtype=anchors.dtype)
 277     ymin = boxes[..., 0:1]
 278     xmin = boxes[..., 1:2]
 279     ymax = boxes[..., 2:3]
 280     xmax = boxes[..., 3:4]
 281     box_h = ymax - ymin + 1.0
 282     box_w = xmax - xmin + 1.0
 283     box_yc = ymin + 0.5 * box_h
 284     box_xc = xmin + 0.5 * box_w
 285 
 286     anchor_ymin = anchors[..., 0:1]
 287     anchor_xmin = anchors[..., 1:2]
 288     anchor_ymax = anchors[..., 2:3]
 289     anchor_xmax = anchors[..., 3:4]
 290     anchor_h = anchor_ymax - anchor_ymin + 1.0
 291     anchor_w = anchor_xmax - anchor_xmin + 1.0
 292     anchor_yc = anchor_ymin + 0.5 * anchor_h
 293     anchor_xc = anchor_xmin + 0.5 * anchor_w
 294 
 295     encoded_dy = (box_yc - anchor_yc) / anchor_h
 296     encoded_dx = (box_xc - anchor_xc) / anchor_w
 297     encoded_dh = tf.math.log(box_h / anchor_h)
 298     encoded_dw = tf.math.log(box_w / anchor_w)
 299     if weights:
 300       encoded_dy *= weights[0]
 301       encoded_dx *= weights[1]
 302       encoded_dh *= weights[2]
 303       encoded_dw *= weights[3]
 304 
 305     encoded_boxes = tf.concat(
 306         [encoded_dy, encoded_dx, encoded_dh, encoded_dw],
 307         axis=-1)
 308     return encoded_boxes

 311 def decode_boxes(encoded_boxes, anchors, weights=None):
 312   """Decode boxes.
 313 
 314   Args:
 315     encoded_boxes: a tensor whose last dimension is 4 representing the
 316       coordinates of encoded boxes in ymin, xmin, ymax, xmax order.
 317     anchors: a tensor whose shape is the same as, or `broadcastable` to `boxes`,
 318       representing the coordinates of anchors in ymin, xmin, ymax, xmax order.
 319     weights: None or a list of four float numbers used to scale coordinates.
 320 
 321   Returns:
 322     encoded_boxes: a tensor whose shape is the same as `boxes` representing the
 323       decoded box targets.
 324   """
 325   if encoded_boxes.shape[-1] != 4:
 326     raise ValueError('encoded_boxes.shape[-1] is {:d}, but must be 4.'.format(
 327         encoded_boxes.shape[-1]))
 328 
 329   with tf.name_scope('decode_boxes'):
 330     encoded_boxes = tf.cast(encoded_boxes, dtype=anchors.dtype)
 331     dy = encoded_boxes[..., 0:1]
 332     dx = encoded_boxes[..., 1:2]
 333     dh = encoded_boxes[..., 2:3]
 334     dw = encoded_boxes[..., 3:4]
 335     if weights:
 336       dy /= weights[0]
 337       dx /= weights[1]
 338       dh /= weights[2]
 339       dw /= weights[3]
 340     dh = tf.math.minimum(dh, BBOX_XFORM_CLIP)
 341     dw = tf.math.minimum(dw, BBOX_XFORM_CLIP)
 342 
 343     anchor_ymin = anchors[..., 0:1]
 344     anchor_xmin = anchors[..., 1:2]
 345     anchor_ymax = anchors[..., 2:3]
 346     anchor_xmax = anchors[..., 3:4]
 347     anchor_h = anchor_ymax - anchor_ymin + 1.0
 348     anchor_w = anchor_xmax - anchor_xmin + 1.0
 349     anchor_yc = anchor_ymin + 0.5 * anchor_h
 350     anchor_xc = anchor_xmin + 0.5 * anchor_w
 351 
 352     decoded_boxes_yc = dy * anchor_h + anchor_yc
 353     decoded_boxes_xc = dx * anchor_w + anchor_xc
 354     decoded_boxes_h = tf.math.exp(dh) * anchor_h
 355     decoded_boxes_w = tf.math.exp(dw) * anchor_w
 356 
 357     decoded_boxes_ymin = decoded_boxes_yc - 0.5 * decoded_boxes_h
 358     decoded_boxes_xmin = decoded_boxes_xc - 0.5 * decoded_boxes_w
 359     decoded_boxes_ymax = decoded_boxes_ymin + decoded_boxes_h - 1.0
 360     decoded_boxes_xmax = decoded_boxes_xmin + decoded_boxes_w - 1.0
 361 
 362     decoded_boxes = tf.concat(
 363         [decoded_boxes_ymin, decoded_boxes_xmin,
 364          decoded_boxes_ymax, decoded_boxes_xmax],
 365         axis=-1)
 366     return decoded_boxes

 369 def filter_boxes(boxes, scores, image_shape, min_size_threshold):
 370   """Filter and remove boxes that are too small or fall outside the image.
 371 
 372   Args:
 373     boxes: a tensor whose last dimension is 4 representing the coordinates of
 374       boxes in ymin, xmin, ymax, xmax order.
 375     scores: a tensor whose shape is the same as tf.shape(boxes)[:-1]
 376       representing the original scores of the boxes.
 377     image_shape: a tensor whose shape is the same as, or `broadcastable` to
 378       `boxes` except the last dimension, which is 2, representing [height,
 379       width] of the scaled image.
 380     min_size_threshold: a float representing the minimal box size in each side
 381       (w.r.t. the scaled image). Boxes whose sides are smaller than it will be
 382       filtered out.
 383 
 384   Returns:
 385     filtered_boxes: a tensor whose shape is the same as `boxes` but with
 386       the position of the filtered boxes are filled with 0.
 387     filtered_scores: a tensor whose shape is the same as 'scores' but with
 388       the positinon of the filtered boxes filled with 0.
 389   """
 390   if boxes.shape[-1] != 4:
 391     raise ValueError('boxes.shape[1] is {:d}, but must be 4.'.format(
 392         boxes.shape[-1]))
 393 
 394   with tf.name_scope('filter_boxes'):
 395     if isinstance(image_shape, list) or isinstance(image_shape, tuple):
 396       height, width = image_shape
 397     else:
 398       image_shape = tf.cast(image_shape, dtype=boxes.dtype)
 399       height = image_shape[..., 0]
 400       width = image_shape[..., 1]
 401 
 402     ymin = boxes[..., 0]
 403     xmin = boxes[..., 1]
 404     ymax = boxes[..., 2]
 405     xmax = boxes[..., 3]
 406 
 407     h = ymax - ymin + 1.0
 408     w = xmax - xmin + 1.0
 409     yc = ymin + 0.5 * h
 410     xc = xmin + 0.5 * w
 411 
 412     min_size = tf.cast(
 413         tf.math.maximum(min_size_threshold, 1.0), dtype=boxes.dtype)
 414 
 415     filtered_size_mask = tf.math.logical_and(
 416         tf.math.greater(h, min_size), tf.math.greater(w, min_size))
 417     filtered_center_mask = tf.logical_and(
 418         tf.math.logical_and(tf.math.greater(yc, 0.0), tf.math.less(yc, height)),
 419         tf.math.logical_and(tf.math.greater(xc, 0.0), tf.math.less(xc, width)))
 420     filtered_mask = tf.math.logical_and(filtered_size_mask,
 421                                         filtered_center_mask)
 422 
 423     filtered_scores = tf.where(filtered_mask, scores, tf.zeros_like(scores))
 424     filtered_boxes = tf.cast(
 425         tf.expand_dims(filtered_mask, axis=-1), dtype=boxes.dtype) * boxes
 426 
 427     return filtered_boxes, filtered_scores

 430 def filter_boxes_by_scores(boxes, scores, min_score_threshold):
 431   """Filter and remove boxes whose scores are smaller than the threshold.
 432 
 433   Args:
 434     boxes: a tensor whose last dimension is 4 representing the coordinates of
 435       boxes in ymin, xmin, ymax, xmax order.
 436     scores: a tensor whose shape is the same as tf.shape(boxes)[:-1]
 437       representing the original scores of the boxes.
 438     min_score_threshold: a float representing the minimal box score threshold.
 439       Boxes whose score are smaller than it will be filtered out.
 440 
 441   Returns:
 442     filtered_boxes: a tensor whose shape is the same as `boxes` but with
 443       the position of the filtered boxes are filled with -1.
 444     filtered_scores: a tensor whose shape is the same as 'scores' but with
 445       the
 446   """
 447   if boxes.shape[-1] != 4:
 448     raise ValueError('boxes.shape[1] is {:d}, but must be 4.'.format(
 449         boxes.shape[-1]))
 450 
 451   with tf.name_scope('filter_boxes_by_scores'):
 452     filtered_mask = tf.math.greater(scores, min_score_threshold)
 453     filtered_scores = tf.where(filtered_mask, scores, -tf.ones_like(scores))
 454     filtered_boxes = tf.cast(
 455         tf.expand_dims(filtered_mask, axis=-1), dtype=boxes.dtype) * boxes
 456 
 457     return filtered_boxes, filtered_scores

 460 def top_k_boxes(boxes, scores, k):
 461   """Sort and select top k boxes according to the scores.
 462 
 463   Args:
 464     boxes: a tensor of shape [batch_size, N, 4] representing the coordiante of
 465       the boxes. N is the number of boxes per image.
 466     scores: a tensor of shsape [batch_size, N] representing the socre of the
 467       boxes.
 468     k: an integer or a tensor indicating the top k number.
 469 
 470   Returns:
 471     selected_boxes: a tensor of shape [batch_size, k, 4] representing the
 472       selected top k box coordinates.
 473     selected_scores: a tensor of shape [batch_size, k] representing the selected
 474       top k box scores.
 475   """
 476   with tf.name_scope('top_k_boxes'):
 477     selected_scores, top_k_indices = tf.nn.top_k(scores, k=k, sorted=True)
 478 
 479     batch_size, _ = scores.get_shape().as_list()
 480     if batch_size == 1:
 481       selected_boxes = tf.squeeze(
 482           tf.gather(boxes, top_k_indices, axis=1), axis=1)
 483     else:
 484       top_k_indices_shape = tf.shape(top_k_indices)
 485       batch_indices = (
 486           tf.expand_dims(tf.range(top_k_indices_shape[0]), axis=-1) *
 487           tf.ones([1, top_k_indices_shape[-1]], dtype=tf.int32))
 488       gather_nd_indices = tf.stack([batch_indices, top_k_indices], axis=-1)
 489       selected_boxes = tf.gather_nd(boxes, gather_nd_indices)
 490 
 491     return selected_boxes, selected_scores

 494 def bbox_overlap(boxes, gt_boxes):
 495   """Calculates the overlap between proposal and ground truth boxes.
 496 
 497   Some `gt_boxes` may have been padded.  The returned `iou` tensor for these
 498   boxes will be -1.
 499 
 500   Args:
 501     boxes: a tensor with a shape of [batch_size, N, 4]. N is the number of
 502       proposals before groundtruth assignment (e.g., rpn_post_nms_topn). The
 503       last dimension is the pixel coordinates in [ymin, xmin, ymax, xmax] form.
 504     gt_boxes: a tensor with a shape of [batch_size, MAX_NUM_INSTANCES, 4]. This
 505       tensor might have paddings with a negative value.
 506 
 507   Returns:
 508     iou: a tensor with as a shape of [batch_size, N, MAX_NUM_INSTANCES].
 509   """
 510   with tf.name_scope('bbox_overlap'):
 511     bb_y_min, bb_x_min, bb_y_max, bb_x_max = tf.split(
 512         value=boxes, num_or_size_splits=4, axis=2)
 513     gt_y_min, gt_x_min, gt_y_max, gt_x_max = tf.split(
 514         value=gt_boxes, num_or_size_splits=4, axis=2)
 515 
 516     # Calculates the intersection area.
 517     i_xmin = tf.math.maximum(bb_x_min, tf.transpose(gt_x_min, [0, 2, 1]))
 518     i_xmax = tf.math.minimum(bb_x_max, tf.transpose(gt_x_max, [0, 2, 1]))
 519     i_ymin = tf.math.maximum(bb_y_min, tf.transpose(gt_y_min, [0, 2, 1]))
 520     i_ymax = tf.math.minimum(bb_y_max, tf.transpose(gt_y_max, [0, 2, 1]))
 521     i_area = tf.math.maximum((i_xmax - i_xmin), 0) * tf.math.maximum(
 522         (i_ymax - i_ymin), 0)
 523 
 524     # Calculates the union area.
 525     bb_area = (bb_y_max - bb_y_min) * (bb_x_max - bb_x_min)
 526     gt_area = (gt_y_max - gt_y_min) * (gt_x_max - gt_x_min)
 527     # Adds a small epsilon to avoid divide-by-zero.
 528     u_area = bb_area + tf.transpose(gt_area, [0, 2, 1]) - i_area + 1e-8
 529 
 530     # Calculates IoU.
 531     iou = i_area / u_area
 532 
 533     # Fills -1 for IoU entries between the padded ground truth boxes.
 534     gt_invalid_mask = tf.less(
 535         tf.reduce_max(gt_boxes, axis=-1, keepdims=True), 0.0)
 536     padding_mask = tf.logical_or(
 537         tf.zeros_like(bb_x_min, dtype=tf.bool),
 538         tf.transpose(gt_invalid_mask, [0, 2, 1]))
 539     iou = tf.where(padding_mask, -tf.ones_like(iou), iou)
 540 
 541     return iou

 544 def get_non_empty_box_indices(boxes):
 545   """Get indices for non-empty boxes."""
 546   # Selects indices if box height or width is 0.
 547   height = boxes[:, 2] - boxes[:, 0]
 548   width = boxes[:, 3] - boxes[:, 1]
 549   indices = tf.where(tf.logical_and(tf.greater(height, 0),
 550                                     tf.greater(width, 0)))
 551   return indices[:, 0]
