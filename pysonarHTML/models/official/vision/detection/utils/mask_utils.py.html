  25 def paste_instance_masks(masks,
  26                          detected_boxes,
  27                          image_height,
  28                          image_width):
  29   """Paste instance masks to generate the image segmentation results.
  30 
  31   Args:
  32     masks: a numpy array of shape [N, mask_height, mask_width] representing the
  33       instance masks w.r.t. the `detected_boxes`.
  34     detected_boxes: a numpy array of shape [N, 4] representing the reference
  35       bounding boxes.
  36     image_height: an integer representing the height of the image.
  37     image_width: an integer representing the width of the image.
  38 
  39   Returns:
  40     segms: a numpy array of shape [N, image_height, image_width] representing
  41       the instance masks *pasted* on the image canvas.
  42   """
  43 
  44   def expand_boxes(boxes, scale):
  45     """Expands an array of boxes by a given scale."""
  46     # Reference: https://github.com/facebookresearch/Detectron/blob/master/detectron/utils/boxes.py#L227  # pylint: disable=line-too-long
  47     # The `boxes` in the reference implementation is in [x1, y1, x2, y2] form,
  48     # whereas `boxes` here is in [x1, y1, w, h] form
  49     w_half = boxes[:, 2] * .5
  50     h_half = boxes[:, 3] * .5
  51     x_c = boxes[:, 0] + w_half
  52     y_c = boxes[:, 1] + h_half
  53 
  54     w_half *= scale
  55     h_half *= scale
  56 
  57     boxes_exp = np.zeros(boxes.shape)
  58     boxes_exp[:, 0] = x_c - w_half
  59     boxes_exp[:, 2] = x_c + w_half
  60     boxes_exp[:, 1] = y_c - h_half
  61     boxes_exp[:, 3] = y_c + h_half
  62 
  63     return boxes_exp
  64 
  65   # Reference: https://github.com/facebookresearch/Detectron/blob/master/detectron/core/test.py#L812  # pylint: disable=line-too-long
  66   # To work around an issue with cv2.resize (it seems to automatically pad
  67   # with repeated border values), we manually zero-pad the masks by 1 pixel
  68   # prior to resizing back to the original image resolution. This prevents
  69   # "top hat" artifacts. We therefore need to expand the reference boxes by an
  70   # appropriate factor.
  71   _, mask_height, mask_width = masks.shape
  72   scale = max((mask_width + 2.0) / mask_width,
  73               (mask_height + 2.0) / mask_height)
  74 
  75   ref_boxes = expand_boxes(detected_boxes, scale)
  76   ref_boxes = ref_boxes.astype(np.int32)
  77   padded_mask = np.zeros((mask_height + 2, mask_width + 2), dtype=np.float32)
  78   segms = []
  79   for mask_ind, mask in enumerate(masks):
  80     im_mask = np.zeros((image_height, image_width), dtype=np.uint8)
  81     # Process mask inside bounding boxes.
  82     padded_mask[1:-1, 1:-1] = mask[:, :]
  83 
  84     ref_box = ref_boxes[mask_ind, :]
  85     w = ref_box[2] - ref_box[0] + 1
  86     h = ref_box[3] - ref_box[1] + 1
  87     w = np.maximum(w, 1)
  88     h = np.maximum(h, 1)
  89 
  90     mask = cv2.resize(padded_mask, (w, h))
  91     mask = np.array(mask > 0.5, dtype=np.uint8)
  92 
  93     x_0 = min(max(ref_box[0], 0), image_width)
  94     x_1 = min(max(ref_box[2] + 1, 0), image_width)
  95     y_0 = min(max(ref_box[1], 0), image_height)
  96     y_1 = min(max(ref_box[3] + 1, 0), image_height)
  97 
  98     im_mask[y_0:y_1, x_0:x_1] = mask[
  99         (y_0 - ref_box[1]):(y_1 - ref_box[1]),
 100         (x_0 - ref_box[0]):(x_1 - ref_box[0])
 101     ]
 102     segms.append(im_mask)
 103 
 104   segms = np.array(segms)
 105   assert masks.shape[0] == segms.shape[0]
 106   return segms

 109 def paste_instance_masks_v2(masks,
 110                             detected_boxes,
 111                             image_height,
 112                             image_width):
 113   """Paste instance masks to generate the image segmentation (v2).
 114 
 115   Args:
 116     masks: a numpy array of shape [N, mask_height, mask_width] representing the
 117       instance masks w.r.t. the `detected_boxes`.
 118     detected_boxes: a numpy array of shape [N, 4] representing the reference
 119       bounding boxes.
 120     image_height: an integer representing the height of the image.
 121     image_width: an integer representing the width of the image.
 122 
 123   Returns:
 124     segms: a numpy array of shape [N, image_height, image_width] representing
 125       the instance masks *pasted* on the image canvas.
 126   """
 127   _, mask_height, mask_width = masks.shape
 128 
 129   segms = []
 130   for i, mask in enumerate(masks):
 131     box = detected_boxes[i, :]
 132     xmin = box[0]
 133     ymin = box[1]
 134     xmax = xmin + box[2]
 135     ymax = ymin + box[3]
 136 
 137     # Sample points of the cropped mask w.r.t. the image grid.
 138     # Note that these coordinates may fall beyond the image.
 139     # Pixel clipping will happen after warping.
 140     xmin_int = int(math.floor(xmin))
 141     xmax_int = int(math.ceil(xmax))
 142     ymin_int = int(math.floor(ymin))
 143     ymax_int = int(math.ceil(ymax))
 144 
 145     alpha = box[2] / (1.0 * mask_width)
 146     beta = box[3] / (1.0 * mask_height)
 147     # pylint: disable=invalid-name
 148     # Transformation from mask pixel indices to image coordinate.
 149     M_mask_to_image = np.array(
 150         [[alpha, 0, xmin],
 151          [0, beta, ymin],
 152          [0, 0, 1]],
 153         dtype=np.float32)
 154     # Transformation from image to cropped mask coordinate.
 155     M_image_to_crop = np.array(
 156         [[1, 0, -xmin_int],
 157          [0, 1, -ymin_int],
 158          [0, 0, 1]],
 159         dtype=np.float32)
 160     M = np.dot(M_image_to_crop, M_mask_to_image)
 161     # Compensate the half pixel offset that OpenCV has in the
 162     # warpPerspective implementation: the top-left pixel is sampled
 163     # at (0,0), but we want it to be at (0.5, 0.5).
 164     M = np.dot(
 165         np.dot(
 166             np.array([[1, 0, -0.5],
 167                       [0, 1, -0.5],
 168                       [0, 0, 1]], np.float32),
 169             M),
 170         np.array([[1, 0, 0.5],
 171                   [0, 1, 0.5],
 172                   [0, 0, 1]], np.float32))
 173     # pylint: enable=invalid-name
 174     cropped_mask = cv2.warpPerspective(
 175         mask.astype(np.float32), M,
 176         (xmax_int - xmin_int, ymax_int - ymin_int))
 177     cropped_mask = np.array(cropped_mask > 0.5, dtype=np.uint8)
 178 
 179     img_mask = np.zeros((image_height, image_width))
 180     x0 = max(min(xmin_int, image_width), 0)
 181     x1 = max(min(xmax_int, image_width), 0)
 182     y0 = max(min(ymin_int, image_height), 0)
 183     y1 = max(min(ymax_int, image_height), 0)
 184     img_mask[y0:y1, x0:x1] = cropped_mask[
 185         (y0 - ymin_int):(y1 - ymin_int),
 186         (x0 - xmin_int):(x1 - xmin_int)]
 187 
 188     segms.append(img_mask)
 189 
 190   segms = np.array(segms)
 191   return segms
