  16 """Matcher interface and Match class.
  17 
  18 This module defines the Matcher interface and the Match object. The job of the
  19 matcher is to match row and column indices based on the similarity matrix and
  20 other optional parameters. Each column is matched to at most one row. There
  21 are three possibilities for the matching:
  22 
  23 1) match: A column matches a row.
  24 2) no_match: A column does not match any row.
  25 3) ignore: A column that is neither 'match' nor no_match.
  26 
  27 The ignore case is regularly encountered in object detection: when an anchor has
  28 a relatively small overlap with a ground-truth box, one neither wants to
  29 consider this box a positive example (match) nor a negative example (no match).
  30 
  31 The Match class is used to store the match results and it provides simple apis
  32 to query the results.
  33 """
  34 from abc import ABCMeta
  35 from abc import abstractmethod
  36 
  37 import tensorflow as tf
  38 
  39 
  40 class Match(object):
  41   """Class to store results from the matcher.
  42 
  43   This class is used to store the results from the matcher. It provides
  44   convenient methods to query the matching results.
  45   """
  46 
  47   def __init__(self, match_results):
  48     """Constructs a Match object.
  49 
  50     Args:
  51       match_results: Integer tensor of shape [N] with (1) match_results[i]>=0,
  52         meaning that column i is matched with row match_results[i].
  53         (2) match_results[i]=-1, meaning that column i is not matched.
  54         (3) match_results[i]=-2, meaning that column i is ignored.
  55 
  56     Raises:
  57       ValueError: if match_results does not have rank 1 or is not an
  58         integer int32 scalar tensor
  59     """
  60     if match_results.shape.ndims != 1:
  61       raise ValueError('match_results should have rank 1')
  62     if match_results.dtype != tf.int32:
  63       raise ValueError('match_results should be an int32 or int64 scalar '
  64                        'tensor')
  65     self._match_results = match_results
  66 
  67   @property
  68   def match_results(self):
  69     """The accessor for match results.
  70 
  71     Returns:
  72       the tensor which encodes the match results.
  73     """
  74     return self._match_results
  75 
  76   def matched_column_indices(self):
  77     """Returns column indices that match to some row.
  78 
  79     The indices returned by this op are always sorted in increasing order.
  80 
  81     Returns:
  82       column_indices: int32 tensor of shape [K] with column indices.
  83     """
  84     return self._reshape_and_cast(tf.where(tf.greater(self._match_results, -1)))
  85 
  86   def matched_column_indicator(self):
  87     """Returns column indices that are matched.
  88 
  89     Returns:
  90       column_indices: int32 tensor of shape [K] with column indices.
  91     """
  92     return tf.greater_equal(self._match_results, 0)
  93 
  94   def num_matched_columns(self):
  95     """Returns number (int32 scalar tensor) of matched columns."""
  96     return tf.size(input=self.matched_column_indices())
  97 
  98   def unmatched_column_indices(self):
  99     """Returns column indices that do not match any row.
 100 
 101     The indices returned by this op are always sorted in increasing order.
 102 
 103     Returns:
 104       column_indices: int32 tensor of shape [K] with column indices.
 105     """
 106     return self._reshape_and_cast(tf.where(tf.equal(self._match_results, -1)))
 107 
 108   def unmatched_column_indicator(self):
 109     """Returns column indices that are unmatched.
 110 
 111     Returns:
 112       column_indices: int32 tensor of shape [K] with column indices.
 113     """
 114     return tf.equal(self._match_results, -1)
 115 
 116   def num_unmatched_columns(self):
 117     """Returns number (int32 scalar tensor) of unmatched columns."""
 118     return tf.size(input=self.unmatched_column_indices())
 119 
 120   def ignored_column_indices(self):
 121     """Returns column indices that are ignored (neither Matched nor Unmatched).
 122 
 123     The indices returned by this op are always sorted in increasing order.
 124 
 125     Returns:
 126       column_indices: int32 tensor of shape [K] with column indices.
 127     """
 128     return self._reshape_and_cast(tf.where(self.ignored_column_indicator()))
 129 
 130   def ignored_column_indicator(self):
 131     """Returns boolean column indicator where True means the colum is ignored.
 132 
 133     Returns:
 134       column_indicator: boolean vector which is True for all ignored column
 135       indices.
 136     """
 137     return tf.equal(self._match_results, -2)
 138 
 139   def num_ignored_columns(self):
 140     """Returns number (int32 scalar tensor) of matched columns."""
 141     return tf.size(input=self.ignored_column_indices())
 142 
 143   def unmatched_or_ignored_column_indices(self):
 144     """Returns column indices that are unmatched or ignored.
 145 
 146     The indices returned by this op are always sorted in increasing order.
 147 
 148     Returns:
 149       column_indices: int32 tensor of shape [K] with column indices.
 150     """
 151     return self._reshape_and_cast(tf.where(tf.greater(0, self._match_results)))
 152 
 153   def matched_row_indices(self):
 154     """Returns row indices that match some column.
 155 
 156     The indices returned by this op are ordered so as to be in correspondence
 157     with the output of matched_column_indicator().  For example if
 158     self.matched_column_indicator() is [0,2], and self.matched_row_indices() is
 159     [7, 3], then we know that column 0 was matched to row 7 and column 2 was
 160     matched to row 3.
 161 
 162     Returns:
 163       row_indices: int32 tensor of shape [K] with row indices.
 164     """
 165     return self._reshape_and_cast(
 166         tf.gather(self._match_results, self.matched_column_indices()))
 167 
 168   def _reshape_and_cast(self, t):
 169     return tf.cast(tf.reshape(t, [-1]), tf.int32)
 170 
 171   def gather_based_on_match(self, input_tensor, unmatched_value,
 172                             ignored_value):
 173     """Gathers elements from `input_tensor` based on match results.
 174 
 175     For columns that are matched to a row, gathered_tensor[col] is set to
 176     input_tensor[match_results[col]]. For columns that are unmatched,
 177     gathered_tensor[col] is set to unmatched_value. Finally, for columns that
 178     are ignored gathered_tensor[col] is set to ignored_value.
 179 
 180     Note that the input_tensor.shape[1:] must match with unmatched_value.shape
 181     and ignored_value.shape
 182 
 183     Args:
 184       input_tensor: Tensor to gather values from.
 185       unmatched_value: Constant tensor value for unmatched columns.
 186       ignored_value: Constant tensor value for ignored columns.
 187 
 188     Returns:
 189       gathered_tensor: A tensor containing values gathered from input_tensor.
 190         The shape of the gathered tensor is [match_results.shape[0]] +
 191         input_tensor.shape[1:].
 192     """
 193     input_tensor = tf.concat([tf.stack([ignored_value, unmatched_value]),
 194                               input_tensor], axis=0)
 195     gather_indices = tf.maximum(self.match_results + 2, 0)
 196     gathered_tensor = tf.gather(input_tensor, gather_indices)
 197     return gathered_tensor

 200 class Matcher(object):
 201   """Abstract base class for matcher.
 202   """
 203   __metaclass__ = ABCMeta
 204 
 205   def match(self, similarity_matrix, scope=None, **params):
 206     """Computes matches among row and column indices and returns the result.
 207 
 208     Computes matches among the row and column indices based on the similarity
 209     matrix and optional arguments.
 210 
 211     Args:
 212       similarity_matrix: Float tensor of shape [N, M] with pairwise similarity
 213         where higher value means more similar.
 214       scope: Op scope name. Defaults to 'Match' if None.
 215       **params: Additional keyword arguments for specific implementations of
 216         the Matcher.
 217 
 218     Returns:
 219       A Match object with the results of matching.
 220     """
 221     if not scope:
 222       scope = 'Match'
 223     with tf.name_scope(scope) as scope:
 224       return Match(self._match(similarity_matrix, **params))
 225 
 226   @abstractmethod
 227   def _match(self, similarity_matrix, **params):

 230     Args:
 231       similarity_matrix: Float tensor of shape [N, M] with pairwise similarity
 232         where higher value means more similar.
 233       **params: Additional keyword arguments for specific implementations of
 234         the Matcher.
 235 
 236     Returns:
 237       match_results: Integer tensor of shape [M]: match_results[i]>=0 means
 238         that column i is matched to row match_results[i], match_results[i]=-1
 239         means that the column is not matched. match_results[i]=-2 means that
 240         the column is ignored (usually this happens when there is a very weak
 241         match which one neither wants as positive nor negative example).
 242     """
 243     pass
