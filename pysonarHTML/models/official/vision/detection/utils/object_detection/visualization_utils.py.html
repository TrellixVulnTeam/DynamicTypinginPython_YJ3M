  69 def save_image_array_as_png(image, output_path):
  70   """Saves an image (represented as a numpy array) to PNG.
  71 
  72   Args:
  73     image: a numpy array with shape [height, width, 3].
  74     output_path: path to which image should be written.
  75   """
  76   image_pil = Image.fromarray(np.uint8(image)).convert('RGB')
  77   with tf.io.gfile.GFile(output_path, 'w') as fid:
  78     image_pil.save(fid, 'PNG')

  81 def encode_image_array_as_png_str(image):
  82   """Encodes a numpy array into a PNG string.
  83 
  84   Args:
  85     image: a numpy array with shape [height, width, 3].
  86 
  87   Returns:
  88     PNG encoded image string.
  89   """
  90   image_pil = Image.fromarray(np.uint8(image))
  91   output = six.BytesIO()
  92   image_pil.save(output, format='PNG')
  93   png_string = output.getvalue()
  94   output.close()
  95   return png_string

  98 def visualize_images_with_bounding_boxes(images, box_outputs, step,
  99                                          summary_writer):
 100   """Records subset of evaluation images with bounding boxes."""
 101   if not isinstance(images, list):
 102     logging.warning('visualize_images_with_bounding_boxes expects list of '
 103                     'images but received type: %s and value: %s',
 104                     type(images), images)

 107   image_shape = tf.shape(images[0])
 108   image_height = tf.cast(image_shape[0], tf.float32)
 109   image_width = tf.cast(image_shape[1], tf.float32)

 116   with summary_writer.as_default():
 117     tf.summary.image('bounding_box_summary', image_summary, step=step)
 118     summary_writer.flush()

 156 def draw_bounding_box_on_image(image,
 157                                ymin,
 158                                xmin,
 159                                ymax,
 160                                xmax,
 161                                color='red',
 162                                thickness=4,
 163                                display_str_list=(),
 164                                use_normalized_coordinates=True):
 165   """Adds a bounding box to an image.
 166 
 167   Bounding box coordinates can be specified in either absolute (pixel) or
 168   normalized coordinates by setting the use_normalized_coordinates argument.
 169 
 170   Each string in display_str_list is displayed on a separate line above the
 171   bounding box in black text on a rectangle filled with the input 'color'.
 172   If the top of the bounding box extends to the edge of the image, the strings
 173   are displayed below the bounding box.
 174 
 175   Args:
 176     image: a PIL.Image object.
 177     ymin: ymin of bounding box.
 178     xmin: xmin of bounding box.
 179     ymax: ymax of bounding box.
 180     xmax: xmax of bounding box.
 181     color: color to draw bounding box. Default is red.
 182     thickness: line thickness. Default value is 4.
 183     display_str_list: list of strings to display in box
 184                       (each to be shown on its own line).
 185     use_normalized_coordinates: If True (default), treat coordinates
 186       ymin, xmin, ymax, xmax as relative to the image.  Otherwise treat
 187       coordinates as absolute.
 188   """
 189   draw = ImageDraw.Draw(image)
 190   im_width, im_height = image.size
 191   if use_normalized_coordinates:
 192     (left, right, top, bottom) = (xmin * im_width, xmax * im_width,
 193                                   ymin * im_height, ymax * im_height)
 194   else:
 195     (left, right, top, bottom) = (xmin, xmax, ymin, ymax)
 196   draw.line([(left, top), (left, bottom), (right, bottom),
 197              (right, top), (left, top)], width=thickness, fill=color)
 198   try:
 199     font = ImageFont.truetype('arial.ttf', 24)
 200   except IOError:
 201     font = ImageFont.load_default()
 202 
 203   # If the total height of the display strings added to the top of the bounding
 204   # box exceeds the top of the image, stack the strings below the bounding box
 205   # instead of above.
 206   display_str_heights = [font.getsize(ds)[1] for ds in display_str_list]
 207   # Each display_str has a top and bottom margin of 0.05x.
 208   total_display_str_height = (1 + 2 * 0.05) * sum(display_str_heights)
 209 
 210   if top > total_display_str_height:
 211     text_bottom = top
 212   else:
 213     text_bottom = bottom + total_display_str_height
 214   # Reverse list and print from bottom to top.
 215   for display_str in display_str_list[::-1]:
 216     text_width, text_height = font.getsize(display_str)
 217     margin = np.ceil(0.05 * text_height)
 218     draw.rectangle(
 219         [(left, text_bottom - text_height - 2 * margin), (left + text_width,
 220                                                           text_bottom)],
 221         fill=color)
 222     draw.text(
 223         (left + margin, text_bottom - text_height - margin),
 224         display_str,
 225         fill='black',
 226         font=font)
 227     text_bottom -= text_height - 2 * margin

 258 def draw_bounding_boxes_on_image(image,
 259                                  boxes,
 260                                  color='red',
 261                                  thickness=4,
 262                                  display_str_list_list=()):
 263   """Draws bounding boxes on image.
 264 
 265   Args:
 266     image: a PIL.Image object.
 267     boxes: a 2 dimensional numpy array of [N, 4]: (ymin, xmin, ymax, xmax).
 268            The coordinates are in normalized format between [0, 1].
 269     color: color to draw bounding box. Default is red.
 270     thickness: line thickness. Default value is 4.
 271     display_str_list_list: list of list of strings.
 272                            a list of strings for each bounding box.
 273                            The reason to pass a list of strings for a
 274                            bounding box is that it might contain
 275                            multiple labels.
 276 
 277   Raises:
 278     ValueError: if boxes is not a [N, 4] array
 279   """
 280   boxes_shape = boxes.shape
 281   if not boxes_shape:
 282     return
 283   if len(boxes_shape) != 2 or boxes_shape[1] != 4:
 284     raise ValueError('Input must be of size [N, 4]')
 285   for i in range(boxes_shape[0]):
 286     display_str_list = ()
 287     if display_str_list_list:
 288       display_str_list = display_str_list_list[i]
 289     draw_bounding_box_on_image(image, boxes[i, 0], boxes[i, 1], boxes[i, 2],
 290                                boxes[i, 3], color, thickness, display_str_list)

 342 def draw_bounding_boxes_on_image_tensors(images,
 343                                          boxes,
 344                                          classes,
 345                                          scores,
 346                                          category_index,
 347                                          original_image_spatial_shape=None,
 348                                          true_image_shape=None,
 349                                          instance_masks=None,
 350                                          keypoints=None,
 351                                          max_boxes_to_draw=20,
 352                                          min_score_thresh=0.2,
 353                                          use_normalized_coordinates=True):
 354   """Draws bounding boxes, masks, and keypoints on batch of image tensors.
 355 
 356   Args:
 357     images: A 4D uint8 image tensor of shape [N, H, W, C]. If C > 3, additional
 358       channels will be ignored. If C = 1, then we convert the images to RGB
 359       images.
 360     boxes: [N, max_detections, 4] float32 tensor of detection boxes.
 361     classes: [N, max_detections] int tensor of detection classes. Note that
 362       classes are 1-indexed.
 363     scores: [N, max_detections] float32 tensor of detection scores.
 364     category_index: a dict that maps integer ids to category dicts. e.g.
 365       {1: {1: 'dog'}, 2: {2: 'cat'}, ...}
 366     original_image_spatial_shape: [N, 2] tensor containing the spatial size of
 367       the original image.
 368     true_image_shape: [N, 3] tensor containing the spatial size of unpadded
 369       original_image.
 370     instance_masks: A 4D uint8 tensor of shape [N, max_detection, H, W] with
 371       instance masks.
 372     keypoints: A 4D float32 tensor of shape [N, max_detection, num_keypoints, 2]
 373       with keypoints.
 374     max_boxes_to_draw: Maximum number of boxes to draw on an image. Default 20.
 375     min_score_thresh: Minimum score threshold for visualization. Default 0.2.
 376     use_normalized_coordinates: Whether to assume boxes and kepoints are in
 377       normalized coordinates (as opposed to absolute coordiantes).
 378       Default is True.
 379 
 380   Returns:
 381     4D image tensor of type uint8, with boxes drawn on top.
 382   """
 383   # Additional channels are being ignored.
 384   if images.shape[3] > 3:
 385     images = images[:, :, :, 0:3]
 386   elif images.shape[3] == 1:
 387     images = tf.image.grayscale_to_rgb(images)
 388   visualization_keyword_args = {
 389       'use_normalized_coordinates': use_normalized_coordinates,
 390       'max_boxes_to_draw': max_boxes_to_draw,
 391       'min_score_thresh': min_score_thresh,
 392       'agnostic_mode': False,
 393       'line_thickness': 4
 394   }
 395   if true_image_shape is None:
 396     true_shapes = tf.constant(-1, shape=[images.shape.as_list()[0], 3])
 397   else:
 398     true_shapes = true_image_shape
 399   if original_image_spatial_shape is None:
 400     original_shapes = tf.constant(-1, shape=[images.shape.as_list()[0], 2])
 401   else:
 402     original_shapes = original_image_spatial_shape
 403 
 404   if instance_masks is not None and keypoints is None:
 405     visualize_boxes_fn = functools.partial(
 406         _visualize_boxes_and_masks,
 407         category_index=category_index,
 408         **visualization_keyword_args)
 409     elems = [
 410         true_shapes, original_shapes, images, boxes, classes, scores,
 411         instance_masks
 412     ]
 413   elif instance_masks is None and keypoints is not None:
 414     visualize_boxes_fn = functools.partial(
 415         _visualize_boxes_and_keypoints,
 416         category_index=category_index,
 417         **visualization_keyword_args)
 418     elems = [
 419         true_shapes, original_shapes, images, boxes, classes, scores, keypoints
 420     ]
 421   elif instance_masks is not None and keypoints is not None:
 422     visualize_boxes_fn = functools.partial(
 423         _visualize_boxes_and_masks_and_keypoints,
 424         category_index=category_index,
 425         **visualization_keyword_args)
 426     elems = [
 427         true_shapes, original_shapes, images, boxes, classes, scores,
 428         instance_masks, keypoints
 429     ]
 430   else:
 431     visualize_boxes_fn = functools.partial(
 432         _visualize_boxes,
 433         category_index=category_index,
 434         **visualization_keyword_args)
 435     elems = [
 436         true_shapes, original_shapes, images, boxes, classes, scores
 437     ]
 438 
 439   def draw_boxes(image_and_detections):
 440     """Draws boxes on image."""
 441     true_shape = image_and_detections[0]
 442     original_shape = image_and_detections[1]
 443     if true_image_shape is not None:
 444       image = shape_utils.pad_or_clip_nd(
 445           image_and_detections[2], [true_shape[0], true_shape[1], 3])
 446     if original_image_spatial_shape is not None:
 447       image_and_detections[2] = _resize_original_image(image, original_shape)
 448 
 449     image_with_boxes = tf.compat.v1.py_func(visualize_boxes_fn,
 450                                             image_and_detections[2:], tf.uint8)
 451     return image_with_boxes
 452 
 453   images = tf.map_fn(draw_boxes, elems, dtype=tf.uint8, back_prop=False)
 454   return images

 479                             keypoints,

 482                             use_normalized_coordinates=True):
 483   """Draws keypoints on an image.
 484 
 485   Args:
 486     image: a PIL.Image object.
 487     keypoints: a numpy array with shape [num_keypoints, 2].
 488     color: color to draw the keypoints with. Default is red.

 493   draw = ImageDraw.Draw(image)

 500   for keypoint_x, keypoint_y in zip(keypoints_x, keypoints_y):
 501     draw.ellipse([(keypoint_x - radius, keypoint_y - radius),
 502                   (keypoint_x + radius, keypoint_y + radius)],
 503                  outline=color, fill=color)

 506 def draw_mask_on_image_array(image, mask, color='red', alpha=0.4):
 507   """Draws mask on an image.
 508 
 509   Args:
 510     image: uint8 numpy array with shape (img_height, img_height, 3)
 511     mask: a uint8 numpy array of shape (img_height, img_height) with
 512       values between either 0 or 1.
 513     color: color to draw the keypoints with. Default is red.
 514     alpha: transparency value between 0 and 1. (default: 0.4)
 515 
 516   Raises:
 517     ValueError: On incorrect data type for image or masks.
 518   """
 519   if image.dtype != np.uint8:
 520     raise ValueError('`image` not of type np.uint8')
 521   if mask.dtype != np.uint8:
 522     raise ValueError('`mask` not of type np.uint8')
 523   if np.any(np.logical_and(mask != 1, mask != 0)):
 524     raise ValueError('`mask` elements should be in [0, 1]')
 525   if image.shape[:2] != mask.shape:
 526     raise ValueError('The image has spatial dimensions %s but the mask has '
 527                      'dimensions %s' % (image.shape[:2], mask.shape))
 528   rgb = ImageColor.getrgb(color)
 529   pil_image = Image.fromarray(image)
 530 
 531   solid_color = np.expand_dims(
 532       np.ones_like(mask), axis=2) * np.reshape(list(rgb), [1, 1, 3])
 533   pil_solid_color = Image.fromarray(np.uint8(solid_color)).convert('RGBA')
 534   pil_mask = Image.fromarray(np.uint8(255.0*alpha*mask)).convert('L')
 535   pil_image = Image.composite(pil_solid_color, pil_image, pil_mask)
 536   np.copyto(image, np.array(pil_image.convert('RGB')))

 539 def visualize_boxes_and_labels_on_image_array(
 540     image,

 554     skip_scores=False,
 555     skip_labels=False):
 556   """Overlay labeled boxes on an image with formatted scores and label names.
 557 
 558   This function groups boxes that correspond to the same location
 559   and creates a display string for each detection and overlays these
 560   on the image. Note that this function modifies the image in place, and returns
 561   that same image.
 562 
 563   Args:
 564     image: uint8 numpy array with shape (img_height, img_width, 3)
 565     boxes: a numpy array of shape [N, 4]
 566     classes: a numpy array of shape [N]. Note that class indices are 1-based,
 567       and match the keys in the label map.
 568     scores: a numpy array of shape [N] or None.  If scores=None, then
 569       this function assumes that the boxes to be plotted are groundtruth
 570       boxes and plot all boxes as black with no classes or scores.
 571     category_index: a dict containing category dictionaries (each holding
 572       category index `id` and category name `name`) keyed by category indices.
 573     instance_masks: a numpy array of shape [N, image_height, image_width] with
 574       values ranging between 0 and 1, can be None.
 575     instance_boundaries: a numpy array of shape [N, image_height, image_width]
 576       with values ranging between 0 and 1, can be None.
 577     keypoints: a numpy array of shape [N, num_keypoints, 2], can
 578       be None
 579     use_normalized_coordinates: whether boxes is to be interpreted as
 580       normalized coordinates or not.
 581     max_boxes_to_draw: maximum number of boxes to visualize.  If None, draw
 582       all boxes.
 583     min_score_thresh: minimum score threshold for a box to be visualized
 584     agnostic_mode: boolean (default: False) controlling whether to evaluate in
 585       class-agnostic mode or not.  This mode will display scores but ignore
 586       classes.
 587     line_thickness: integer (default: 4) controlling line width of the boxes.
 588     groundtruth_box_visualization_color: box color for visualizing groundtruth
 589       boxes
 590     skip_scores: whether to skip score when drawing a single detection
 591     skip_labels: whether to skip label when drawing a single detection
 592 
 593   Returns:
 594     uint8 numpy array with shape (img_height, img_width, 3) with overlaid boxes.
 595   """
 596   # Create a display string (and color) for every box location, group any boxes
 597   # that correspond to the same location.
 598   box_to_display_str_map = collections.defaultdict(list)
 599   box_to_color_map = collections.defaultdict(str)
 600   box_to_instance_masks_map = {}
 601   box_to_instance_boundaries_map = {}
 602   box_to_keypoints_map = collections.defaultdict(list)
 603   if not max_boxes_to_draw:
 604     max_boxes_to_draw = boxes.shape[0]
 605   for i in range(min(max_boxes_to_draw, boxes.shape[0])):
 606     if scores is None or scores[i] > min_score_thresh:
 607       box = tuple(boxes[i].tolist())
 608       if instance_masks is not None:
 609         box_to_instance_masks_map[box] = instance_masks[i]
 610       if instance_boundaries is not None:
 611         box_to_instance_boundaries_map[box] = instance_boundaries[i]
 612       if keypoints is not None:
 613         box_to_keypoints_map[box].extend(keypoints[i])
 614       if scores is None:
 615         box_to_color_map[box] = groundtruth_box_visualization_color
 616       else:
 617         display_str = ''
 618         if not skip_labels:
 619           if not agnostic_mode:
 620             if classes[i] in category_index.keys():
 621               class_name = category_index[classes[i]]['name']
 622             else:
 623               class_name = 'N/A'
 624             display_str = str(class_name)
 625         if not skip_scores:
 626           if not display_str:
 627             display_str = '{}%'.format(int(100*scores[i]))
 628           else:
 629             display_str = '{}: {}%'.format(display_str, int(100*scores[i]))
 630         box_to_display_str_map[box].append(display_str)
 631         if agnostic_mode:
 632           box_to_color_map[box] = 'DarkOrange'
 633         else:
 634           box_to_color_map[box] = STANDARD_COLORS[
 635               classes[i] % len(STANDARD_COLORS)]
 636 
 637   # Draw all boxes onto image.
 638   for box, color in box_to_color_map.items():
 639     ymin, xmin, ymax, xmax = box
 640     if instance_masks is not None:
 641       draw_mask_on_image_array(
 642           image,
 643           box_to_instance_masks_map[box],
 644           color=color
 645       )
 646     if instance_boundaries is not None:
 647       draw_mask_on_image_array(
 648           image,
 649           box_to_instance_boundaries_map[box],
 650           color='red',
 651           alpha=1.0
 652       )
 653     draw_bounding_box_on_image_array(
 654         image,
 655         ymin,
 656         xmin,
 657         ymax,
 658         xmax,
 659         color=color,
 660         thickness=line_thickness,
 661         display_str_list=box_to_display_str_map[box],
 662         use_normalized_coordinates=use_normalized_coordinates)
 663     if keypoints is not None:
 664       draw_keypoints_on_image_array(
 665           image,
 666           box_to_keypoints_map[box],
 667           color=color,
 668           radius=line_thickness / 2,
 669           use_normalized_coordinates=use_normalized_coordinates)
 670 
 671   return image

 674 def add_cdf_image_summary(values, name):
 675   """Adds a tf.summary.image for a CDF plot of the values.
 676 
 677   Normalizes `values` such that they sum to 1, plots the cumulative distribution
 678   function and creates a tf image summary.
 679 
 680   Args:
 681     values: a 1-D float32 tensor containing the values.
 682     name: name for the image summary.
 683   """
 684   def cdf_plot(values):
 685     """Numpy function to plot CDF."""
 686     normalized_values = values / np.sum(values)
 687     sorted_values = np.sort(normalized_values)
 688     cumulative_values = np.cumsum(sorted_values)
 689     fraction_of_examples = (np.arange(cumulative_values.size, dtype=np.float32)
 690                             / cumulative_values.size)
 691     fig = plt.figure(frameon=False)
 692     ax = fig.add_subplot('111')
 693     ax.plot(fraction_of_examples, cumulative_values)
 694     ax.set_ylabel('cumulative normalized values')
 695     ax.set_xlabel('fraction of examples')
 696     fig.canvas.draw()
 697     width, height = fig.get_size_inches() * fig.get_dpi()
 698     image = np.fromstring(fig.canvas.tostring_rgb(), dtype='uint8').reshape(
 699         1, int(height), int(width), 3)
 700     return image
 701 
 702   cdf_plot = tf.compat.v1.py_func(cdf_plot, [values], tf.uint8)
 703   tf.compat.v1.summary.image(name, cdf_plot)

 706 def add_hist_image_summary(values, bins, name):
 707   """Adds a tf.summary.image for a histogram plot of the values.
 708 
 709   Plots the histogram of values and creates a tf image summary.
 710 
 711   Args:
 712     values: a 1-D float32 tensor containing the values.
 713     bins: bin edges which will be directly passed to np.histogram.
 714     name: name for the image summary.
 715   """
 716 
 717   def hist_plot(values, bins):
 718     """Numpy function to plot hist."""
 719     fig = plt.figure(frameon=False)
 720     ax = fig.add_subplot('111')
 721     y, x = np.histogram(values, bins=bins)
 722     ax.plot(x[:-1], y)
 723     ax.set_ylabel('count')
 724     ax.set_xlabel('value')
 725     fig.canvas.draw()
 726     width, height = fig.get_size_inches() * fig.get_dpi()
 727     image = np.fromstring(
 728         fig.canvas.tostring_rgb(), dtype='uint8').reshape(
 729             1, int(height), int(width), 3)
 730     return image
 731 
 732   hist_plot = tf.compat.v1.py_func(hist_plot, [values, bins], tf.uint8)
 733   tf.compat.v1.summary.image(name, hist_plot)
