  16 """Class to subsample minibatches by balancing positives and negatives.
  17 
  18 Subsamples minibatches based on a pre-specified positive fraction in range
  19 [0,1]. The class presumes there are many more negatives than positive examples:
  20 if the desired batch_size cannot be achieved with the pre-specified positive
  21 fraction, it fills the rest with negative examples. If this is not sufficient
  22 for obtaining the desired batch_size, it returns fewer examples.
  23 
  24 The main function to call is Subsample(self, indicator, labels). For convenience
  25 one can also call SubsampleWeights(self, weights, labels) which is defined in
  26 the minibatch_sampler base class.
  27 
  28 When is_static is True, it implements a method that guarantees static shapes.
  29 It also ensures the length of output of the subsample is always batch_size, even
  30 when number of examples set to True in indicator is less than batch_size.
  31 
  32 This is originally implemented in TensorFlow Object Detection API.
  33 """
  34 
  35 import tensorflow as tf
  36 
  37 from official.vision.detection.utils.object_detection import minibatch_sampler
  38 from official.vision.detection.utils.object_detection import ops
  39 
  40 
  41 class BalancedPositiveNegativeSampler(minibatch_sampler.MinibatchSampler):
  42   """Subsamples minibatches to a desired balance of positives and negatives."""
  43 
  44   def __init__(self, positive_fraction=0.5, is_static=False):
  45     """Constructs a minibatch sampler.
  46 
  47     Args:
  48       positive_fraction: desired fraction of positive examples (scalar in [0,1])
  49         in the batch.
  50       is_static: If True, uses an implementation with static shape guarantees.
  51 
  52     Raises:
  53       ValueError: if positive_fraction < 0, or positive_fraction > 1
  54     """
  55     if positive_fraction < 0 or positive_fraction > 1:
  56       raise ValueError('positive_fraction should be in range [0,1]. '
  57                        'Received: %s.' % positive_fraction)
  58     self._positive_fraction = positive_fraction
  59     self._is_static = is_static
  60 
  61   def _get_num_pos_neg_samples(self, sorted_indices_tensor, sample_size):
  62     """Counts the number of positives and negatives numbers to be sampled.
  63 
  64     Args:
  65       sorted_indices_tensor: A sorted int32 tensor of shape [N] which contains
  66         the signed indices of the examples where the sign is based on the label
  67         value. The examples that cannot be sampled are set to 0. It samples
  68         atmost sample_size*positive_fraction positive examples and remaining
  69         from negative examples.
  70       sample_size: Size of subsamples.
  71 
  72     Returns:
  73       A tuple containing the number of positive and negative labels in the
  74       subsample.
  75     """
  76     input_length = tf.shape(input=sorted_indices_tensor)[0]
  77     valid_positive_index = tf.greater(sorted_indices_tensor,
  78                                       tf.zeros(input_length, tf.int32))
  79     num_sampled_pos = tf.reduce_sum(
  80         input_tensor=tf.cast(valid_positive_index, tf.int32))
  81     max_num_positive_samples = tf.constant(
  82         int(sample_size * self._positive_fraction), tf.int32)
  83     num_positive_samples = tf.minimum(max_num_positive_samples, num_sampled_pos)
  84     num_negative_samples = tf.constant(sample_size,
  85                                        tf.int32) - num_positive_samples
  86 
  87     return num_positive_samples, num_negative_samples
  88 
  89   def _get_values_from_start_and_end(self, input_tensor, num_start_samples,
  90                                      num_end_samples, total_num_samples):
  91     """slices num_start_samples and last num_end_samples from input_tensor.
  92 
  93     Args:
  94       input_tensor: An int32 tensor of shape [N] to be sliced.
  95       num_start_samples: Number of examples to be sliced from the beginning
  96         of the input tensor.
  97       num_end_samples: Number of examples to be sliced from the end of the
  98         input tensor.
  99       total_num_samples: Sum of is num_start_samples and num_end_samples. This
 100         should be a scalar.
 101 
 102     Returns:
 103       A tensor containing the first num_start_samples and last num_end_samples
 104       from input_tensor.
 105 
 106     """
 107     input_length = tf.shape(input=input_tensor)[0]
 108     start_positions = tf.less(tf.range(input_length), num_start_samples)
 109     end_positions = tf.greater_equal(
 110         tf.range(input_length), input_length - num_end_samples)
 111     selected_positions = tf.logical_or(start_positions, end_positions)
 112     selected_positions = tf.cast(selected_positions, tf.float32)
 113     indexed_positions = tf.multiply(tf.cumsum(selected_positions),
 114                                     selected_positions)
 115     one_hot_selector = tf.one_hot(tf.cast(indexed_positions, tf.int32) - 1,
 116                                   total_num_samples,
 117                                   dtype=tf.float32)
 118     return tf.cast(tf.tensordot(tf.cast(input_tensor, tf.float32),
 119                                 one_hot_selector, axes=[0, 0]), tf.int32)
 120 
 121   def _static_subsample(self, indicator, batch_size, labels):
 122     """Returns subsampled minibatch.
 123 
 124     Args:
 125       indicator: boolean tensor of shape [N] whose True entries can be sampled.
 126         N should be a complie time constant.
 127       batch_size: desired batch size. This scalar cannot be None.
 128       labels: boolean tensor of shape [N] denoting positive(=True) and negative
 129         (=False) examples. N should be a complie time constant.
 130 
 131     Returns:
 132       sampled_idx_indicator: boolean tensor of shape [N], True for entries which
 133         are sampled. It ensures the length of output of the subsample is always
 134         batch_size, even when number of examples set to True in indicator is
 135         less than batch_size.
 136 
 137     Raises:
 138       ValueError: if labels and indicator are not 1D boolean tensors.
 139     """
 140     # Check if indicator and labels have a static size.
 141     if not indicator.shape.is_fully_defined():
 142       raise ValueError('indicator must be static in shape when is_static is'
 143                        'True')
 144     if not labels.shape.is_fully_defined():
 145       raise ValueError('labels must be static in shape when is_static is'
 146                        'True')
 147     if not isinstance(batch_size, int):
 148       raise ValueError('batch_size has to be an integer when is_static is'
 149                        'True.')
 150 
 151     input_length = tf.shape(input=indicator)[0]
 152 
 153     # Set the number of examples set True in indicator to be at least
 154     # batch_size.
 155     num_true_sampled = tf.reduce_sum(
 156         input_tensor=tf.cast(indicator, tf.float32))
 157     additional_false_sample = tf.less_equal(
 158         tf.cumsum(tf.cast(tf.logical_not(indicator), tf.float32)),
 159         batch_size - num_true_sampled)
 160     indicator = tf.logical_or(indicator, additional_false_sample)
 161 
 162     # Shuffle indicator and label. Need to store the permutation to restore the
 163     # order post sampling.
 164     permutation = tf.random.shuffle(tf.range(input_length))
 165     indicator = ops.matmul_gather_on_zeroth_axis(
 166         tf.cast(indicator, tf.float32), permutation)
 167     labels = ops.matmul_gather_on_zeroth_axis(
 168         tf.cast(labels, tf.float32), permutation)
 169 
 170     # index (starting from 1) when indicator is True, 0 when False
 171     indicator_idx = tf.where(
 172         tf.cast(indicator, tf.bool), tf.range(1, input_length + 1),
 173         tf.zeros(input_length, tf.int32))
 174 
 175     # Replace -1 for negative, +1 for positive labels
 176     signed_label = tf.where(
 177         tf.cast(labels, tf.bool), tf.ones(input_length, tf.int32),
 178         tf.scalar_mul(-1, tf.ones(input_length, tf.int32)))
 179     # negative of index for negative label, positive index for positive label,
 180     # 0 when indicator is False.
 181     signed_indicator_idx = tf.multiply(indicator_idx, signed_label)
 182     sorted_signed_indicator_idx = tf.nn.top_k(
 183         signed_indicator_idx, input_length, sorted=True).values
 184 
 185     [num_positive_samples,
 186      num_negative_samples] = self._get_num_pos_neg_samples(
 187          sorted_signed_indicator_idx, batch_size)
 188 
 189     sampled_idx = self._get_values_from_start_and_end(
 190         sorted_signed_indicator_idx, num_positive_samples,
 191         num_negative_samples, batch_size)
 192 
 193     # Shift the indices to start from 0 and remove any samples that are set as
 194     # False.
 195     sampled_idx = tf.abs(sampled_idx) - tf.ones(batch_size, tf.int32)
 196     sampled_idx = tf.multiply(
 197         tf.cast(tf.greater_equal(sampled_idx, tf.constant(0)), tf.int32),
 198         sampled_idx)
 199 
 200     sampled_idx_indicator = tf.cast(
 201         tf.reduce_sum(
 202             input_tensor=tf.one_hot(sampled_idx, depth=input_length), axis=0),
 203         tf.bool)
 204 
 205     # project back the order based on stored permutations
 206     reprojections = tf.one_hot(permutation, depth=input_length,
 207                                dtype=tf.float32)
 208     return tf.cast(tf.tensordot(
 209         tf.cast(sampled_idx_indicator, tf.float32),
 210         reprojections, axes=[0, 0]), tf.bool)
 211 
 212   def subsample(self, indicator, batch_size, labels, scope=None):
 213     """Returns subsampled minibatch.
 214 
 215     Args:
 216       indicator: boolean tensor of shape [N] whose True entries can be sampled.
 217       batch_size: desired batch size. If None, keeps all positive samples and
 218         randomly selects negative samples so that the positive sample fraction
 219         matches self._positive_fraction. It cannot be None is is_static is True.
 220       labels: boolean tensor of shape [N] denoting positive(=True) and negative
 221           (=False) examples.
 222       scope: name scope.
 223 
 224     Returns:
 225       sampled_idx_indicator: boolean tensor of shape [N], True for entries which
 226         are sampled.
 227 
 228     Raises:
 229       ValueError: if labels and indicator are not 1D boolean tensors.
 230     """
 231     if len(indicator.get_shape().as_list()) != 1:
 232       raise ValueError('indicator must be 1 dimensional, got a tensor of '
 233                        'shape %s' % indicator.get_shape())
 234     if len(labels.get_shape().as_list()) != 1:
 235       raise ValueError('labels must be 1 dimensional, got a tensor of '
 236                        'shape %s' % labels.get_shape())
 237     if labels.dtype != tf.bool:
 238       raise ValueError('labels should be of type bool. Received: %s' %
 239                        labels.dtype)
 240     if indicator.dtype != tf.bool:
 241       raise ValueError('indicator should be of type bool. Received: %s' %
 242                        indicator.dtype)
 243     scope = scope or 'BalancedPositiveNegativeSampler'
 244     with tf.name_scope(scope):
 245       if self._is_static:
 246         return self._static_subsample(indicator, batch_size, labels)
 247 
 248       else:
 249         # Only sample from indicated samples
 250         negative_idx = tf.logical_not(labels)
 251         positive_idx = tf.logical_and(labels, indicator)
 252         negative_idx = tf.logical_and(negative_idx, indicator)
 253 
 254         # Sample positive and negative samples separately
 255         if batch_size is None:
 256           max_num_pos = tf.reduce_sum(
 257               input_tensor=tf.cast(positive_idx, dtype=tf.int32))
 258         else:
 259           max_num_pos = int(self._positive_fraction * batch_size)
 260         sampled_pos_idx = self.subsample_indicator(positive_idx, max_num_pos)
 261         num_sampled_pos = tf.reduce_sum(
 262             input_tensor=tf.cast(sampled_pos_idx, tf.int32))
 263         if batch_size is None:
 264           negative_positive_ratio = (
 265               1 - self._positive_fraction) / self._positive_fraction
 266           max_num_neg = tf.cast(
 267               negative_positive_ratio *
 268               tf.cast(num_sampled_pos, dtype=tf.float32),
 269               dtype=tf.int32)
 270         else:
 271           max_num_neg = batch_size - num_sampled_pos
 272         sampled_neg_idx = self.subsample_indicator(negative_idx, max_num_neg)
 273 
 274         return tf.logical_or(sampled_pos_idx, sampled_neg_idx)
