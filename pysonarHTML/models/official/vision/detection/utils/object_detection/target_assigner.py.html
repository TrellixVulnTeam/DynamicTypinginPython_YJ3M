  16 """Base target assigner module.
  17 
  18 The job of a TargetAssigner is, for a given set of anchors (bounding boxes) and
  19 groundtruth detections (bounding boxes), to assign classification and regression
  20 targets to each anchor as well as weights to each anchor (specifying, e.g.,
  21 which anchors should not contribute to training loss).
  22 
  23 It assigns classification/regression targets by performing the following steps:
  24 1) Computing pairwise similarity between anchors and groundtruth boxes using a
  25   provided RegionSimilarity Calculator
  26 2) Computing a matching based on the similarity matrix using a provided Matcher
  27 3) Assigning regression targets based on the matching and a provided BoxCoder
  28 4) Assigning classification targets based on the matching and groundtruth labels
  29 
  30 Note that TargetAssigners only operate on detections from a single
  31 image at a time, so any logic for applying a TargetAssigner to multiple
  32 images must be handled externally.
  33 """
  34 import tensorflow as tf
  35 
  36 from official.vision.detection.utils.object_detection import box_list
  37 from official.vision.detection.utils.object_detection import shape_utils
  38 
  39 
  40 KEYPOINTS_FIELD_NAME = 'keypoints'

  43 class TargetAssigner(object):
  44   """Target assigner to compute classification and regression targets."""
  45 
  46   def __init__(self, similarity_calc, matcher, box_coder,
  47                negative_class_weight=1.0, unmatched_cls_target=None):
  48     """Construct Object Detection Target Assigner.
  49 
  50     Args:
  51       similarity_calc: a RegionSimilarityCalculator
  52       matcher: Matcher used to match groundtruth to anchors.
  53       box_coder: BoxCoder used to encode matching groundtruth boxes with
  54         respect to anchors.
  55       negative_class_weight: classification weight to be associated to negative
  56         anchors (default: 1.0). The weight must be in [0., 1.].
  57       unmatched_cls_target: a float32 tensor with shape [d_1, d_2, ..., d_k]
  58         which is consistent with the classification target for each
  59         anchor (and can be empty for scalar targets).  This shape must thus be
  60         compatible with the groundtruth labels that are passed to the "assign"
  61         function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
  62         If set to None, unmatched_cls_target is set to be [0] for each anchor.
  63 
  64     Raises:
  65       ValueError: if similarity_calc is not a RegionSimilarityCalculator or
  66         if matcher is not a Matcher or if box_coder is not a BoxCoder
  67     """
  68     self._similarity_calc = similarity_calc
  69     self._matcher = matcher
  70     self._box_coder = box_coder
  71     self._negative_class_weight = negative_class_weight
  72     if unmatched_cls_target is None:
  73       self._unmatched_cls_target = tf.constant([0], tf.float32)
  74     else:
  75       self._unmatched_cls_target = unmatched_cls_target
  76 
  77   @property
  78   def box_coder(self):
  79     return self._box_coder
  80 
  81   def assign(self, anchors, groundtruth_boxes, groundtruth_labels=None,
  82              groundtruth_weights=None, **params):
  83     """Assign classification and regression targets to each anchor.
  84 
  85     For a given set of anchors and groundtruth detections, match anchors
  86     to groundtruth_boxes and assign classification and regression targets to
  87     each anchor as well as weights based on the resulting match (specifying,
  88     e.g., which anchors should not contribute to training loss).
  89 
  90     Anchors that are not matched to anything are given a classification target
  91     of self._unmatched_cls_target which can be specified via the constructor.
  92 
  93     Args:
  94       anchors: a BoxList representing N anchors
  95       groundtruth_boxes: a BoxList representing M groundtruth boxes
  96       groundtruth_labels:  a tensor of shape [M, d_1, ... d_k]
  97         with labels for each of the ground_truth boxes. The subshape
  98         [d_1, ... d_k] can be empty (corresponding to scalar inputs).  When set
  99         to None, groundtruth_labels assumes a binary problem where all
 100         ground_truth boxes get a positive label (of 1).
 101       groundtruth_weights: a float tensor of shape [M] indicating the weight to
 102         assign to all anchors match to a particular groundtruth box. The weights
 103         must be in [0., 1.]. If None, all weights are set to 1.
 104       **params: Additional keyword arguments for specific implementations of
 105               the Matcher.
 106 
 107     Returns:
 108       cls_targets: a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k],
 109         where the subshape [d_1, ..., d_k] is compatible with groundtruth_labels
 110         which has shape [num_gt_boxes, d_1, d_2, ... d_k].
 111       cls_weights: a float32 tensor with shape [num_anchors]
 112       reg_targets: a float32 tensor with shape [num_anchors, box_code_dimension]
 113       reg_weights: a float32 tensor with shape [num_anchors]
 114       match: a matcher.Match object encoding the match between anchors and
 115         groundtruth boxes, with rows corresponding to groundtruth boxes
 116         and columns corresponding to anchors.
 117 
 118     Raises:
 119       ValueError: if anchors or groundtruth_boxes are not of type
 120         box_list.BoxList
 121     """
 122     if not isinstance(anchors, box_list.BoxList):
 123       raise ValueError('anchors must be an BoxList')
 124     if not isinstance(groundtruth_boxes, box_list.BoxList):
 125       raise ValueError('groundtruth_boxes must be an BoxList')
 126 
 127     if groundtruth_labels is None:
 128       groundtruth_labels = tf.ones(tf.expand_dims(groundtruth_boxes.num_boxes(),
 129                                                   0))
 130       groundtruth_labels = tf.expand_dims(groundtruth_labels, -1)
 131     unmatched_shape_assert = shape_utils.assert_shape_equal(
 132         shape_utils.combined_static_and_dynamic_shape(groundtruth_labels)[1:],
 133         shape_utils.combined_static_and_dynamic_shape(
 134             self._unmatched_cls_target))
 135     labels_and_box_shapes_assert = shape_utils.assert_shape_equal(
 136         shape_utils.combined_static_and_dynamic_shape(
 137             groundtruth_labels)[:1],
 138         shape_utils.combined_static_and_dynamic_shape(
 139             groundtruth_boxes.get())[:1])
 140 
 141     if groundtruth_weights is None:
 142       num_gt_boxes = groundtruth_boxes.num_boxes_static()
 143       if not num_gt_boxes:
 144         num_gt_boxes = groundtruth_boxes.num_boxes()
 145       groundtruth_weights = tf.ones([num_gt_boxes], dtype=tf.float32)
 146     with tf.control_dependencies(
 147         [unmatched_shape_assert, labels_and_box_shapes_assert]):
 148       match_quality_matrix = self._similarity_calc.compare(groundtruth_boxes,
 149                                                            anchors)
 150       match = self._matcher.match(match_quality_matrix, **params)
 151       reg_targets = self._create_regression_targets(anchors,
 152                                                     groundtruth_boxes,
 153                                                     match)
 154       cls_targets = self._create_classification_targets(groundtruth_labels,
 155                                                         match)
 156       reg_weights = self._create_regression_weights(match, groundtruth_weights)
 157       cls_weights = self._create_classification_weights(match,
 158                                                         groundtruth_weights)
 159 
 160     num_anchors = anchors.num_boxes_static()
 161     if num_anchors is not None:
 162       reg_targets = self._reset_target_shape(reg_targets, num_anchors)
 163       cls_targets = self._reset_target_shape(cls_targets, num_anchors)
 164       reg_weights = self._reset_target_shape(reg_weights, num_anchors)
 165       cls_weights = self._reset_target_shape(cls_weights, num_anchors)
 166 
 167     return cls_targets, cls_weights, reg_targets, reg_weights, match
 168 
 169   def _reset_target_shape(self, target, num_anchors):
 170     """Sets the static shape of the target.
 171 
 172     Args:
 173       target: the target tensor. Its first dimension will be overwritten.
 174       num_anchors: the number of anchors, which is used to override the target's
 175         first dimension.
 176 
 177     Returns:
 178       A tensor with the shape info filled in.
 179     """
 180     target_shape = target.get_shape().as_list()
 181     target_shape[0] = num_anchors
 182     target.set_shape(target_shape)
 183     return target
 184 
 185   def _create_regression_targets(self, anchors, groundtruth_boxes, match):
 186     """Returns a regression target for each anchor.
 187 
 188     Args:
 189       anchors: a BoxList representing N anchors
 190       groundtruth_boxes: a BoxList representing M groundtruth_boxes
 191       match: a matcher.Match object
 192 
 193     Returns:
 194       reg_targets: a float32 tensor with shape [N, box_code_dimension]
 195     """
 196     matched_gt_boxes = match.gather_based_on_match(
 197         groundtruth_boxes.get(),
 198         unmatched_value=tf.zeros(4),
 199         ignored_value=tf.zeros(4))
 200     matched_gt_boxlist = box_list.BoxList(matched_gt_boxes)
 201     if groundtruth_boxes.has_field(KEYPOINTS_FIELD_NAME):
 202       groundtruth_keypoints = groundtruth_boxes.get_field(KEYPOINTS_FIELD_NAME)
 203       matched_keypoints = match.gather_based_on_match(
 204           groundtruth_keypoints,
 205           unmatched_value=tf.zeros(groundtruth_keypoints.get_shape()[1:]),
 206           ignored_value=tf.zeros(groundtruth_keypoints.get_shape()[1:]))
 207       matched_gt_boxlist.add_field(KEYPOINTS_FIELD_NAME, matched_keypoints)
 208     matched_reg_targets = self._box_coder.encode(matched_gt_boxlist, anchors)
 209     match_results_shape = shape_utils.combined_static_and_dynamic_shape(
 210         match.match_results)
 211 
 212     # Zero out the unmatched and ignored regression targets.
 213     unmatched_ignored_reg_targets = tf.tile(
 214         self._default_regression_target(), [match_results_shape[0], 1])
 215     matched_anchors_mask = match.matched_column_indicator()
 216     # To broadcast matched_anchors_mask to the same shape as
 217     # matched_reg_targets.
 218     matched_anchors_mask = tf.tile(
 219         tf.expand_dims(matched_anchors_mask, 1),
 220         [1, tf.shape(matched_reg_targets)[1]])
 221     reg_targets = tf.where(matched_anchors_mask, matched_reg_targets,
 222                            unmatched_ignored_reg_targets)
 223     return reg_targets
 224 
 225   def _default_regression_target(self):
 226     """Returns the default target for anchors to regress to.
 227 
 228     Default regression targets are set to zero (though in
 229     this implementation what these targets are set to should
 230     not matter as the regression weight of any box set to
 231     regress to the default target is zero).
 232 
 233     Returns:
 234       default_target: a float32 tensor with shape [1, box_code_dimension]
 235     """
 236     return tf.constant([self._box_coder.code_size*[0]], tf.float32)
 237 
 238   def _create_classification_targets(self, groundtruth_labels, match):
 239     """Create classification targets for each anchor.
 240 
 241     Assign a classification target of for each anchor to the matching
 242     groundtruth label that is provided by match.  Anchors that are not matched
 243     to anything are given the target self._unmatched_cls_target
 244 
 245     Args:
 246       groundtruth_labels:  a tensor of shape [num_gt_boxes, d_1, ... d_k]
 247         with labels for each of the ground_truth boxes. The subshape
 248         [d_1, ... d_k] can be empty (corresponding to scalar labels).
 249       match: a matcher.Match object that provides a matching between anchors
 250         and groundtruth boxes.
 251 
 252     Returns:
 253       a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k], where the
 254       subshape [d_1, ..., d_k] is compatible with groundtruth_labels which has
 255       shape [num_gt_boxes, d_1, d_2, ... d_k].
 256     """
 257     return match.gather_based_on_match(
 258         groundtruth_labels,
 259         unmatched_value=self._unmatched_cls_target,
 260         ignored_value=self._unmatched_cls_target)
 261 
 262   def _create_regression_weights(self, match, groundtruth_weights):
 263     """Set regression weight for each anchor.
 264 
 265     Only positive anchors are set to contribute to the regression loss, so this
 266     method returns a weight of 1 for every positive anchor and 0 for every
 267     negative anchor.
 268 
 269     Args:
 270       match: a matcher.Match object that provides a matching between anchors
 271         and groundtruth boxes.
 272       groundtruth_weights: a float tensor of shape [M] indicating the weight to
 273         assign to all anchors match to a particular groundtruth box.
 274 
 275     Returns:
 276       a float32 tensor with shape [num_anchors] representing regression weights.
 277     """
 278     return match.gather_based_on_match(
 279         groundtruth_weights, ignored_value=0., unmatched_value=0.)
 280 
 281   def _create_classification_weights(self,
 282                                      match,
 283                                      groundtruth_weights):
 284     """Create classification weights for each anchor.
 285 
 286     Positive (matched) anchors are associated with a weight of
 287     positive_class_weight and negative (unmatched) anchors are associated with
 288     a weight of negative_class_weight. When anchors are ignored, weights are set
 289     to zero. By default, both positive/negative weights are set to 1.0,
 290     but they can be adjusted to handle class imbalance (which is almost always
 291     the case in object detection).
 292 
 293     Args:
 294       match: a matcher.Match object that provides a matching between anchors
 295         and groundtruth boxes.
 296       groundtruth_weights: a float tensor of shape [M] indicating the weight to
 297         assign to all anchors match to a particular groundtruth box.
 298 
 299     Returns:
 300       a float32 tensor with shape [num_anchors] representing classification
 301       weights.
 302     """
 303     return match.gather_based_on_match(
 304         groundtruth_weights,
 305         ignored_value=0.,
 306         unmatched_value=self._negative_class_weight)
 307 
 308   def get_box_coder(self):
 309     """Get BoxCoder of this TargetAssigner.
 310 
 311     Returns:
 312       BoxCoder object.
 313     """
 314     return self._box_coder
