  48 def area(boxlist, scope=None):
  49   """Computes area of boxes.
  50 
  51   Args:
  52     boxlist: BoxList holding N boxes
  53     scope: name scope.
  54 
  55   Returns:
  56     a tensor with shape [N] representing box areas.
  57   """
  58   with tf.name_scope(scope, 'Area'):
  59     y_min, x_min, y_max, x_max = tf.split(
  60         value=boxlist.get(), num_or_size_splits=4, axis=1)
  61     return tf.squeeze((y_max - y_min) * (x_max - x_min), [1])

  64 def height_width(boxlist, scope=None):
  65   """Computes height and width of boxes in boxlist.
  66 
  67   Args:
  68     boxlist: BoxList holding N boxes
  69     scope: name scope.
  70 
  71   Returns:
  72     Height: A tensor with shape [N] representing box heights.
  73     Width: A tensor with shape [N] representing box widths.
  74   """
  75   with tf.name_scope(scope, 'HeightWidth'):
  76     y_min, x_min, y_max, x_max = tf.split(
  77         value=boxlist.get(), num_or_size_splits=4, axis=1)
  78     return tf.squeeze(y_max - y_min, [1]), tf.squeeze(x_max - x_min, [1])

  81 def scale(boxlist, y_scale, x_scale, scope=None):
  82   """scale box coordinates in x and y dimensions.
  83 
  84   Args:
  85     boxlist: BoxList holding N boxes
  86     y_scale: (float) scalar tensor
  87     x_scale: (float) scalar tensor
  88     scope: name scope.
  89 
  90   Returns:
  91     boxlist: BoxList holding N boxes
  92   """
  93   with tf.name_scope(scope, 'Scale'):
  94     y_scale = tf.cast(y_scale, tf.float32)
  95     x_scale = tf.cast(x_scale, tf.float32)
  96     y_min, x_min, y_max, x_max = tf.split(
  97         value=boxlist.get(), num_or_size_splits=4, axis=1)
  98     y_min = y_scale * y_min
  99     y_max = y_scale * y_max
 100     x_min = x_scale * x_min
 101     x_max = x_scale * x_max
 102     scaled_boxlist = box_list.BoxList(
 103         tf.concat([y_min, x_min, y_max, x_max], 1))
 104     return _copy_extra_fields(scaled_boxlist, boxlist)

 107 def clip_to_window(boxlist, window, filter_nonoverlapping=True, scope=None):
 108   """Clip bounding boxes to a window.
 109 
 110   This op clips any input bounding boxes (represented by bounding box
 111   corners) to a window, optionally filtering out boxes that do not
 112   overlap at all with the window.
 113 
 114   Args:
 115     boxlist: BoxList holding M_in boxes
 116     window: a tensor of shape [4] representing the [y_min, x_min, y_max, x_max]
 117       window to which the op should clip boxes.
 118     filter_nonoverlapping: whether to filter out boxes that do not overlap at
 119       all with the window.
 120     scope: name scope.
 121 
 122   Returns:
 123     a BoxList holding M_out boxes where M_out <= M_in
 124   """
 125   with tf.name_scope(scope, 'ClipToWindow'):
 126     y_min, x_min, y_max, x_max = tf.split(
 127         value=boxlist.get(), num_or_size_splits=4, axis=1)
 128     win_y_min, win_x_min, win_y_max, win_x_max = tf.unstack(window)
 129     y_min_clipped = tf.maximum(tf.minimum(y_min, win_y_max), win_y_min)
 130     y_max_clipped = tf.maximum(tf.minimum(y_max, win_y_max), win_y_min)
 131     x_min_clipped = tf.maximum(tf.minimum(x_min, win_x_max), win_x_min)
 132     x_max_clipped = tf.maximum(tf.minimum(x_max, win_x_max), win_x_min)
 133     clipped = box_list.BoxList(
 134         tf.concat([y_min_clipped, x_min_clipped, y_max_clipped, x_max_clipped],
 135                   1))
 136     clipped = _copy_extra_fields(clipped, boxlist)
 137     if filter_nonoverlapping:
 138       areas = area(clipped)
 139       nonzero_area_indices = tf.cast(
 140           tf.reshape(tf.where(tf.greater(areas, 0.0)), [-1]), tf.int32)
 141       clipped = gather(clipped, nonzero_area_indices)
 142     return clipped

 145 def prune_outside_window(boxlist, window, scope=None):
 146   """Prunes bounding boxes that fall outside a given window.
 147 
 148   This function prunes bounding boxes that even partially fall outside the given
 149   window. See also clip_to_window which only prunes bounding boxes that fall
 150   completely outside the window, and clips any bounding boxes that partially
 151   overflow.
 152 
 153   Args:
 154     boxlist: a BoxList holding M_in boxes.
 155     window: a float tensor of shape [4] representing [ymin, xmin, ymax, xmax]
 156       of the window
 157     scope: name scope.
 158 
 159   Returns:
 160     pruned_corners: a tensor with shape [M_out, 4] where M_out <= M_in
 161     valid_indices: a tensor with shape [M_out] indexing the valid bounding boxes
 162      in the input tensor.
 163   """
 164   with tf.name_scope(scope, 'PruneOutsideWindow'):
 165     y_min, x_min, y_max, x_max = tf.split(
 166         value=boxlist.get(), num_or_size_splits=4, axis=1)
 167     win_y_min, win_x_min, win_y_max, win_x_max = tf.unstack(window)
 168     coordinate_violations = tf.concat([
 169         tf.less(y_min, win_y_min), tf.less(x_min, win_x_min),
 170         tf.greater(y_max, win_y_max), tf.greater(x_max, win_x_max)
 171     ], 1)
 172     valid_indices = tf.reshape(
 173         tf.where(tf.logical_not(tf.reduce_any(coordinate_violations, 1))), [-1])
 174     return gather(boxlist, valid_indices), valid_indices

 177 def prune_completely_outside_window(boxlist, window, scope=None):
 178   """Prunes bounding boxes that fall completely outside of the given window.
 179 
 180   The function clip_to_window prunes bounding boxes that fall
 181   completely outside the window, but also clips any bounding boxes that
 182   partially overflow. This function does not clip partially overflowing boxes.
 183 
 184   Args:
 185     boxlist: a BoxList holding M_in boxes.
 186     window: a float tensor of shape [4] representing [ymin, xmin, ymax, xmax]
 187       of the window
 188     scope: name scope.
 189 
 190   Returns:
 191     pruned_boxlist: a new BoxList with all bounding boxes partially or fully in
 192       the window.
 193     valid_indices: a tensor with shape [M_out] indexing the valid bounding boxes
 194      in the input tensor.
 195   """
 196   with tf.name_scope(scope, 'PruneCompleteleyOutsideWindow'):
 197     y_min, x_min, y_max, x_max = tf.split(
 198         value=boxlist.get(), num_or_size_splits=4, axis=1)
 199     win_y_min, win_x_min, win_y_max, win_x_max = tf.unstack(window)
 200     coordinate_violations = tf.concat([
 201         tf.greater_equal(y_min, win_y_max), tf.greater_equal(x_min, win_x_max),
 202         tf.less_equal(y_max, win_y_min), tf.less_equal(x_max, win_x_min)
 203     ], 1)
 204     valid_indices = tf.reshape(
 205         tf.where(tf.logical_not(tf.reduce_any(coordinate_violations, 1))), [-1])
 206     return gather(boxlist, valid_indices), valid_indices

 209 def intersection(boxlist1, boxlist2, scope=None):
 210   """Compute pairwise intersection areas between boxes.
 211 
 212   Args:
 213     boxlist1: BoxList holding N boxes
 214     boxlist2: BoxList holding M boxes
 215     scope: name scope.
 216 
 217   Returns:
 218     a tensor with shape [N, M] representing pairwise intersections
 219   """
 220   with tf.name_scope(scope, 'Intersection'):
 221     y_min1, x_min1, y_max1, x_max1 = tf.split(
 222         value=boxlist1.get(), num_or_size_splits=4, axis=1)
 223     y_min2, x_min2, y_max2, x_max2 = tf.split(
 224         value=boxlist2.get(), num_or_size_splits=4, axis=1)
 225     all_pairs_min_ymax = tf.minimum(y_max1, tf.transpose(y_max2))
 226     all_pairs_max_ymin = tf.maximum(y_min1, tf.transpose(y_min2))
 227     intersect_heights = tf.maximum(0.0, all_pairs_min_ymax - all_pairs_max_ymin)
 228     all_pairs_min_xmax = tf.minimum(x_max1, tf.transpose(x_max2))
 229     all_pairs_max_xmin = tf.maximum(x_min1, tf.transpose(x_min2))
 230     intersect_widths = tf.maximum(0.0, all_pairs_min_xmax - all_pairs_max_xmin)
 231     return intersect_heights * intersect_widths

 234 def matched_intersection(boxlist1, boxlist2, scope=None):
 235   """Compute intersection areas between corresponding boxes in two boxlists.
 236 
 237   Args:
 238     boxlist1: BoxList holding N boxes
 239     boxlist2: BoxList holding N boxes
 240     scope: name scope.
 241 
 242   Returns:
 243     a tensor with shape [N] representing pairwise intersections
 244   """
 245   with tf.name_scope(scope, 'MatchedIntersection'):
 246     y_min1, x_min1, y_max1, x_max1 = tf.split(
 247         value=boxlist1.get(), num_or_size_splits=4, axis=1)
 248     y_min2, x_min2, y_max2, x_max2 = tf.split(
 249         value=boxlist2.get(), num_or_size_splits=4, axis=1)
 250     min_ymax = tf.minimum(y_max1, y_max2)
 251     max_ymin = tf.maximum(y_min1, y_min2)
 252     intersect_heights = tf.maximum(0.0, min_ymax - max_ymin)
 253     min_xmax = tf.minimum(x_max1, x_max2)
 254     max_xmin = tf.maximum(x_min1, x_min2)
 255     intersect_widths = tf.maximum(0.0, min_xmax - max_xmin)
 256     return tf.reshape(intersect_heights * intersect_widths, [-1])

 259 def iou(boxlist1, boxlist2, scope=None):
 260   """Computes pairwise intersection-over-union between box collections.
 261 
 262   Args:
 263     boxlist1: BoxList holding N boxes
 264     boxlist2: BoxList holding M boxes
 265     scope: name scope.
 266 
 267   Returns:
 268     a tensor with shape [N, M] representing pairwise iou scores.
 269   """
 270   with tf.name_scope(scope, 'IOU'):
 271     intersections = intersection(boxlist1, boxlist2)
 272     areas1 = area(boxlist1)
 273     areas2 = area(boxlist2)
 274     unions = (
 275         tf.expand_dims(areas1, 1) + tf.expand_dims(areas2, 0) - intersections)
 276     return tf.where(
 277         tf.equal(intersections, 0.0),
 278         tf.zeros_like(intersections), tf.truediv(intersections, unions))
 279

 281 def matched_iou(boxlist1, boxlist2, scope=None):
 282   """Compute intersection-over-union between corresponding boxes in boxlists.
 283 
 284   Args:
 285     boxlist1: BoxList holding N boxes
 286     boxlist2: BoxList holding N boxes
 287     scope: name scope.
 288 
 289   Returns:
 290     a tensor with shape [N] representing pairwise iou scores.
 291   """
 292   with tf.name_scope(scope, 'MatchedIOU'):
 293     intersections = matched_intersection(boxlist1, boxlist2)
 294     areas1 = area(boxlist1)
 295     areas2 = area(boxlist2)
 296     unions = areas1 + areas2 - intersections
 297     return tf.where(
 298         tf.equal(intersections, 0.0),
 299         tf.zeros_like(intersections), tf.truediv(intersections, unions))

 302 def ioa(boxlist1, boxlist2, scope=None):
 303   """Computes pairwise intersection-over-area between box collections.
 304 
 305   intersection-over-area (IOA) between two boxes box1 and box2 is defined as
 306   their intersection area over box2's area. Note that ioa is not symmetric,
 307   that is, ioa(box1, box2) != ioa(box2, box1).
 308 
 309   Args:
 310     boxlist1: BoxList holding N boxes
 311     boxlist2: BoxList holding M boxes
 312     scope: name scope.
 313 
 314   Returns:
 315     a tensor with shape [N, M] representing pairwise ioa scores.
 316   """
 317   with tf.name_scope(scope, 'IOA'):
 318     intersections = intersection(boxlist1, boxlist2)
 319     areas = tf.expand_dims(area(boxlist2), 0)
 320     return tf.truediv(intersections, areas)

 323 def prune_non_overlapping_boxes(
 324     boxlist1, boxlist2, min_overlap=0.0, scope=None):
 325   """Prunes the boxes in boxlist1 that overlap less than thresh with boxlist2.
 326 
 327   For each box in boxlist1, we want its IOA to be more than minoverlap with
 328   at least one of the boxes in boxlist2. If it does not, we remove it.
 329 
 330   Args:
 331     boxlist1: BoxList holding N boxes.
 332     boxlist2: BoxList holding M boxes.
 333     min_overlap: Minimum required overlap between boxes, to count them as
 334                 overlapping.
 335     scope: name scope.
 336 
 337   Returns:
 338     new_boxlist1: A pruned boxlist with size [N', 4].
 339     keep_inds: A tensor with shape [N'] indexing kept bounding boxes in the
 340       first input BoxList `boxlist1`.
 341   """
 342   with tf.name_scope(scope, 'PruneNonOverlappingBoxes'):
 343     ioa_ = ioa(boxlist2, boxlist1)  # [M, N] tensor
 344     ioa_ = tf.reduce_max(ioa_, reduction_indices=[0])  # [N] tensor
 345     keep_bool = tf.greater_equal(ioa_, tf.constant(min_overlap))
 346     keep_inds = tf.squeeze(tf.where(keep_bool), axis=[1])
 347     new_boxlist1 = gather(boxlist1, keep_inds)
 348     return new_boxlist1, keep_inds

 351 def prune_small_boxes(boxlist, min_side, scope=None):
 352   """Prunes small boxes in the boxlist which have a side smaller than min_side.
 353 
 354   Args:
 355     boxlist: BoxList holding N boxes.
 356     min_side: Minimum width AND height of box to survive pruning.
 357     scope: name scope.
 358 
 359   Returns:
 360     A pruned boxlist.
 361   """
 362   with tf.name_scope(scope, 'PruneSmallBoxes'):
 363     height, width = height_width(boxlist)
 364     is_valid = tf.logical_and(tf.greater_equal(width, min_side),
 365                               tf.greater_equal(height, min_side))
 366     return gather(boxlist, tf.reshape(tf.where(is_valid), [-1]))

 369 def change_coordinate_frame(boxlist, window, scope=None):
 370   """Change coordinate frame of the boxlist to be relative to window's frame.
 371 
 372   Given a window of the form [ymin, xmin, ymax, xmax],
 373   changes bounding box coordinates from boxlist to be relative to this window
 374   (e.g., the min corner maps to (0,0) and the max corner maps to (1,1)).
 375 
 376   An example use case is data augmentation: where we are given groundtruth
 377   boxes (boxlist) and would like to randomly crop the image to some
 378   window (window). In this case we need to change the coordinate frame of
 379   each groundtruth box to be relative to this new window.
 380 
 381   Args:
 382     boxlist: A BoxList object holding N boxes.
 383     window: A rank 1 tensor [4].
 384     scope: name scope.
 385 
 386   Returns:
 387     Returns a BoxList object with N boxes.
 388   """
 389   with tf.name_scope(scope, 'ChangeCoordinateFrame'):
 390     win_height = window[2] - window[0]
 391     win_width = window[3] - window[1]
 392     boxlist_new = scale(box_list.BoxList(
 393         boxlist.get() - [window[0], window[1], window[0], window[1]]),
 394                         1.0 / win_height, 1.0 / win_width)
 395     boxlist_new = _copy_extra_fields(boxlist_new, boxlist)
 396     return boxlist_new

 399 def sq_dist(boxlist1, boxlist2, scope=None):
 400   """Computes the pairwise squared distances between box corners.
 401 
 402   This op treats each box as if it were a point in a 4d Euclidean space and
 403   computes pairwise squared distances.
 404 
 405   Mathematically, we are given two matrices of box coordinates X and Y,
 406   where X(i,:) is the i'th row of X, containing the 4 numbers defining the
 407   corners of the i'th box in boxlist1. Similarly Y(j,:) corresponds to
 408   boxlist2.  We compute
 409   Z(i,j) = ||X(i,:) - Y(j,:)||^2
 410          = ||X(i,:)||^2 + ||Y(j,:)||^2 - 2 X(i,:)' * Y(j,:),
 411 
 412   Args:
 413     boxlist1: BoxList holding N boxes
 414     boxlist2: BoxList holding M boxes
 415     scope: name scope.
 416 
 417   Returns:
 418     a tensor with shape [N, M] representing pairwise distances
 419   """
 420   with tf.name_scope(scope, 'SqDist'):
 421     sqnorm1 = tf.reduce_sum(tf.square(boxlist1.get()), 1, keep_dims=True)
 422     sqnorm2 = tf.reduce_sum(tf.square(boxlist2.get()), 1, keep_dims=True)
 423     innerprod = tf.matmul(boxlist1.get(), boxlist2.get(),
 424                           transpose_a=False, transpose_b=True)
 425     return sqnorm1 + tf.transpose(sqnorm2) - 2.0 * innerprod

 428 def boolean_mask(boxlist, indicator, fields=None, scope=None,
 429                  use_static_shapes=False, indicator_sum=None):
 430   """Select boxes from BoxList according to indicator and return new BoxList.
 431 
 432   `boolean_mask` returns the subset of boxes that are marked as "True" by the
 433   indicator tensor. By default, `boolean_mask` returns boxes corresponding to
 434   the input index list, as well as all additional fields stored in the boxlist
 435   (indexing into the first dimension).  However one can optionally only draw
 436   from a subset of fields.
 437 
 438   Args:
 439     boxlist: BoxList holding N boxes
 440     indicator: a rank-1 boolean tensor
 441     fields: (optional) list of fields to also gather from.  If None (default),
 442       all fields are gathered from.  Pass an empty fields list to only gather
 443       the box coordinates.
 444     scope: name scope.
 445     use_static_shapes: Whether to use an implementation with static shape
 446       gurantees.
 447     indicator_sum: An integer containing the sum of `indicator` vector. Only
 448       required if `use_static_shape` is True.
 449 
 450   Returns:
 451     subboxlist: a BoxList corresponding to the subset of the input BoxList
 452       specified by indicator
 453   Raises:
 454     ValueError: if `indicator` is not a rank-1 boolean tensor.
 455   """
 456   with tf.name_scope(scope, 'BooleanMask'):
 457     if indicator.shape.ndims != 1:
 458       raise ValueError('indicator should have rank 1')
 459     if indicator.dtype != tf.bool:
 460       raise ValueError('indicator should be a boolean tensor')
 461     if use_static_shapes:
 462       if not (indicator_sum and isinstance(indicator_sum, int)):
 463         raise ValueError('`indicator_sum` must be a of type int')
 464       selected_positions = tf.cast(indicator, dtype=tf.float32)
 465       indexed_positions = tf.cast(
 466           tf.multiply(
 467               tf.cumsum(selected_positions), selected_positions),
 468           dtype=tf.int32)
 469       one_hot_selector = tf.one_hot(
 470           indexed_positions - 1, indicator_sum, dtype=tf.float32)
 471       sampled_indices = tf.cast(
 472           tf.tensordot(
 473               tf.cast(tf.range(tf.shape(indicator)[0]), dtype=tf.float32),
 474               one_hot_selector,
 475               axes=[0, 0]),
 476           dtype=tf.int32)
 477       return gather(boxlist, sampled_indices, use_static_shapes=True)
 478     else:
 479       subboxlist = box_list.BoxList(tf.boolean_mask(boxlist.get(), indicator))
 480       if fields is None:
 481         fields = boxlist.get_extra_fields()
 482       for field in fields:
 483         if not boxlist.has_field(field):
 484           raise ValueError('boxlist must contain all specified fields')
 485         subfieldlist = tf.boolean_mask(boxlist.get_field(field), indicator)
 486         subboxlist.add_field(field, subfieldlist)
 487       return subboxlist

 490 def gather(boxlist, indices, fields=None, scope=None, use_static_shapes=False):
 491   """Gather boxes from BoxList according to indices and return new BoxList.
 492 
 493   By default, `gather` returns boxes corresponding to the input index list, as
 494   well as all additional fields stored in the boxlist (indexing into the
 495   first dimension).  However one can optionally only gather from a
 496   subset of fields.
 497 
 498   Args:
 499     boxlist: BoxList holding N boxes
 500     indices: a rank-1 tensor of type int32 / int64
 501     fields: (optional) list of fields to also gather from.  If None (default),
 502       all fields are gathered from.  Pass an empty fields list to only gather
 503       the box coordinates.
 504     scope: name scope.
 505     use_static_shapes: Whether to use an implementation with static shape
 506       gurantees.
 507 
 508   Returns:
 509     subboxlist: a BoxList corresponding to the subset of the input BoxList
 510     specified by indices
 511   Raises:
 512     ValueError: if specified field is not contained in boxlist or if the
 513       indices are not of type int32
 514   """
 515   with tf.name_scope(scope, 'Gather'):
 516     if len(indices.shape.as_list()) != 1:
 517       raise ValueError('indices should have rank 1')
 518     if indices.dtype != tf.int32 and indices.dtype != tf.int64:
 519       raise ValueError('indices should be an int32 / int64 tensor')
 520     gather_op = tf.gather
 521     if use_static_shapes:
 522       gather_op = ops.matmul_gather_on_zeroth_axis
 523     subboxlist = box_list.BoxList(gather_op(boxlist.get(), indices))
 524     if fields is None:
 525       fields = boxlist.get_extra_fields()
 526     fields += ['boxes']
 527     for field in fields:
 528       if not boxlist.has_field(field):
 529         raise ValueError('boxlist must contain all specified fields')
 530       subfieldlist = gather_op(boxlist.get_field(field), indices)
 531       subboxlist.add_field(field, subfieldlist)
 532     return subboxlist

 535 def concatenate(boxlists, fields=None, scope=None):
 536   """Concatenate list of BoxLists.
 537 
 538   This op concatenates a list of input BoxLists into a larger BoxList.  It also
 539   handles concatenation of BoxList fields as long as the field tensor shapes
 540   are equal except for the first dimension.
 541 
 542   Args:
 543     boxlists: list of BoxList objects
 544     fields: optional list of fields to also concatenate.  By default, all
 545       fields from the first BoxList in the list are included in the
 546       concatenation.
 547     scope: name scope.
 548 
 549   Returns:
 550     a BoxList with number of boxes equal to
 551       sum([boxlist.num_boxes() for boxlist in BoxList])
 552   Raises:
 553     ValueError: if boxlists is invalid (i.e., is not a list, is empty, or
 554       contains non BoxList objects), or if requested fields are not contained in
 555       all boxlists
 556   """
 557   with tf.name_scope(scope, 'Concatenate'):
 558     if not isinstance(boxlists, list):
 559       raise ValueError('boxlists should be a list')
 560     if not boxlists:
 561       raise ValueError('boxlists should have nonzero length')
 562     for boxlist in boxlists:
 563       if not isinstance(boxlist, box_list.BoxList):
 564         raise ValueError('all elements of boxlists should be BoxList objects')
 565     concatenated = box_list.BoxList(
 566         tf.concat([boxlist.get() for boxlist in boxlists], 0))
 567     if fields is None:
 568       fields = boxlists[0].get_extra_fields()
 569     for field in fields:
 570       first_field_shape = boxlists[0].get_field(field).get_shape().as_list()
 571       first_field_shape[0] = -1
 572       if None in first_field_shape:
 573         raise ValueError('field %s must have fully defined shape except for the'
 574                          ' 0th dimension.' % field)
 575       for boxlist in boxlists:
 576         if not boxlist.has_field(field):
 577           raise ValueError('boxlist must contain all requested fields')
 578         field_shape = boxlist.get_field(field).get_shape().as_list()
 579         field_shape[0] = -1
 580         if field_shape != first_field_shape:
 581           raise ValueError('field %s must have same shape for all boxlists '
 582                            'except for the 0th dimension.' % field)
 583       concatenated_field = tf.concat(
 584           [boxlist.get_field(field) for boxlist in boxlists], 0)
 585       concatenated.add_field(field, concatenated_field)
 586     return concatenated

 589 def sort_by_field(boxlist, field, order=SortOrder.descend, scope=None):
 590   """Sort boxes and associated fields according to a scalar field.
 591 
 592   A common use case is reordering the boxes according to descending scores.
 593 
 594   Args:
 595     boxlist: BoxList holding N boxes.
 596     field: A BoxList field for sorting and reordering the BoxList.
 597     order: (Optional) descend or ascend. Default is descend.
 598     scope: name scope.
 599 
 600   Returns:
 601     sorted_boxlist: A sorted BoxList with the field in the specified order.
 602 
 603   Raises:
 604     ValueError: if specified field does not exist
 605     ValueError: if the order is not either descend or ascend
 606   """
 607   with tf.name_scope(scope, 'SortByField'):
 608     if order != SortOrder.descend and order != SortOrder.ascend:
 609       raise ValueError('Invalid sort order')
 610 
 611     field_to_sort = boxlist.get_field(field)
 612     if len(field_to_sort.shape.as_list()) != 1:
 613       raise ValueError('Field should have rank 1')
 614 
 615     num_boxes = boxlist.num_boxes()
 616     num_entries = tf.size(field_to_sort)
 617     length_assert = tf.Assert(
 618         tf.equal(num_boxes, num_entries),
 619         ['Incorrect field size: actual vs expected.', num_entries, num_boxes])
 620 
 621     with tf.control_dependencies([length_assert]):
 622       _, sorted_indices = tf.nn.top_k(field_to_sort, num_boxes, sorted=True)
 623 
 624     if order == SortOrder.ascend:
 625       sorted_indices = tf.reverse_v2(sorted_indices, [0])
 626 
 627     return gather(boxlist, sorted_indices)

 659 def filter_field_value_equals(boxlist, field, value, scope=None):
 660   """Filter to keep only boxes with field entries equal to the given value.
 661 
 662   Args:
 663     boxlist: BoxList holding N boxes.
 664     field: field name for filtering.
 665     value: scalar value.
 666     scope: name scope.
 667 
 668   Returns:
 669     a BoxList holding M boxes where M <= N
 670 
 671   Raises:
 672     ValueError: if boxlist not a BoxList object or if it does not have
 673       the specified field.
 674   """
 675   with tf.name_scope(scope, 'FilterFieldValueEquals'):
 676     if not isinstance(boxlist, box_list.BoxList):
 677       raise ValueError('boxlist must be a BoxList')
 678     if not boxlist.has_field(field):
 679       raise ValueError('boxlist must contain the specified field')
 680     filter_field = boxlist.get_field(field)
 681     gather_index = tf.reshape(tf.where(tf.equal(filter_field, value)), [-1])
 682     return gather(boxlist, gather_index)

 685 def filter_greater_than(boxlist, thresh, scope=None):
 686   """Filter to keep only boxes with score exceeding a given threshold.
 687 
 688   This op keeps the collection of boxes whose corresponding scores are
 689   greater than the input threshold.
 690 
 691   TODO(jonathanhuang): Change function name to filter_scores_greater_than
 692 
 693   Args:
 694     boxlist: BoxList holding N boxes.  Must contain a 'scores' field
 695       representing detection scores.
 696     thresh: scalar threshold
 697     scope: name scope.
 698 
 699   Returns:
 700     a BoxList holding M boxes where M <= N
 701 
 702   Raises:
 703     ValueError: if boxlist not a BoxList object or if it does not
 704       have a scores field
 705   """
 706   with tf.name_scope(scope, 'FilterGreaterThan'):
 707     if not isinstance(boxlist, box_list.BoxList):
 708       raise ValueError('boxlist must be a BoxList')
 709     if not boxlist.has_field('scores'):
 710       raise ValueError('input boxlist must have \'scores\' field')
 711     scores = boxlist.get_field('scores')
 712     if len(scores.shape.as_list()) > 2:
 713       raise ValueError('Scores should have rank 1 or 2')
 714     if len(scores.shape.as_list()) == 2 and scores.shape.as_list()[1] != 1:
 715       raise ValueError('Scores should have rank 1 or have shape '
 716                        'consistent with [None, 1]')
 717     high_score_indices = tf.cast(tf.reshape(
 718         tf.where(tf.greater(scores, thresh)),
 719         [-1]), tf.int32)
 720     return gather(boxlist, high_score_indices)

 723 def non_max_suppression(boxlist, thresh, max_output_size, scope=None):
 724   """Non maximum suppression.
 725 
 726   This op greedily selects a subset of detection bounding boxes, pruning
 727   away boxes that have high IOU (intersection over union) overlap (> thresh)
 728   with already selected boxes.  Note that this only works for a single class ---
 729   to apply NMS to multi-class predictions, use MultiClassNonMaxSuppression.
 730 
 731   Args:
 732     boxlist: BoxList holding N boxes.  Must contain a 'scores' field
 733       representing detection scores.
 734     thresh: scalar threshold
 735     max_output_size: maximum number of retained boxes
 736     scope: name scope.
 737 
 738   Returns:
 739     a BoxList holding M boxes where M <= max_output_size
 740   Raises:
 741     ValueError: if thresh is not in [0, 1]
 742   """
 743   with tf.name_scope(scope, 'NonMaxSuppression'):
 744     if not 0 <= thresh <= 1.0:
 745       raise ValueError('thresh must be between 0 and 1')
 746     if not isinstance(boxlist, box_list.BoxList):
 747       raise ValueError('boxlist must be a BoxList')
 748     if not boxlist.has_field('scores'):
 749       raise ValueError('input boxlist must have \'scores\' field')
 750     selected_indices = tf.image.non_max_suppression(
 751         boxlist.get(), boxlist.get_field('scores'),
 752         max_output_size, iou_threshold=thresh)
 753     return gather(boxlist, selected_indices)

 771 def to_normalized_coordinates(boxlist, height, width,
 772                               check_range=True, scope=None):
 773   """Converts absolute box coordinates to normalized coordinates in [0, 1].
 774 
 775   Usually one uses the dynamic shape of the image or conv-layer tensor:
 776     boxlist = box_list_ops.to_normalized_coordinates(boxlist,
 777                                                      tf.shape(images)[1],
 778                                                      tf.shape(images)[2]),
 779 
 780   This function raises an assertion failed error at graph execution time when
 781   the maximum coordinate is smaller than 1.01 (which means that coordinates are
 782   already normalized). The value 1.01 is to deal with small rounding errors.
 783 
 784   Args:
 785     boxlist: BoxList with coordinates in terms of pixel-locations.
 786     height: Maximum value for height of absolute box coordinates.
 787     width: Maximum value for width of absolute box coordinates.
 788     check_range: If True, checks if the coordinates are normalized or not.
 789     scope: name scope.
 790 
 791   Returns:
 792     boxlist with normalized coordinates in [0, 1].
 793   """
 794   with tf.name_scope(scope, 'ToNormalizedCoordinates'):
 795     height = tf.cast(height, tf.float32)
 796     width = tf.cast(width, tf.float32)
 797 
 798     if check_range:
 799       max_val = tf.reduce_max(boxlist.get())
 800       max_assert = tf.Assert(tf.greater(max_val, 1.01),
 801                              ['max value is lower than 1.01: ', max_val])
 802       with tf.control_dependencies([max_assert]):
 803         width = tf.identity(width)
 804 
 805     return scale(boxlist, 1 / height, 1 / width)

 808 def to_absolute_coordinates(boxlist,
 809                             height,
 810                             width,
 811                             check_range=True,
 812                             maximum_normalized_coordinate=1.1,
 813                             scope=None):
 814   """Converts normalized box coordinates to absolute pixel coordinates.
 815 
 816   This function raises an assertion failed error when the maximum box coordinate
 817   value is larger than maximum_normalized_coordinate (in which case coordinates
 818   are already absolute).
 819 
 820   Args:
 821     boxlist: BoxList with coordinates in range [0, 1].
 822     height: Maximum value for height of absolute box coordinates.
 823     width: Maximum value for width of absolute box coordinates.
 824     check_range: If True, checks if the coordinates are normalized or not.
 825     maximum_normalized_coordinate: Maximum coordinate value to be considered
 826       as normalized, default to 1.1.
 827     scope: name scope.
 828 
 829   Returns:
 830     boxlist with absolute coordinates in terms of the image size.
 831 
 832   """
 833   with tf.name_scope(scope, 'ToAbsoluteCoordinates'):
 834     height = tf.cast(height, tf.float32)
 835     width = tf.cast(width, tf.float32)
 836 
 837     # Ensure range of input boxes is correct.
 838     if check_range:
 839       box_maximum = tf.reduce_max(boxlist.get())

 843            'than %f: ' % maximum_normalized_coordinate, box_maximum])
 844       with tf.control_dependencies([max_assert]):
 845         width = tf.identity(width)
 846 
 847     return scale(boxlist, height, width)

 850 def refine_boxes_multi_class(pool_boxes,
 851                              num_classes,
 852                              nms_iou_thresh,
 853                              nms_max_detections,
 854                              voting_iou_thresh=0.5):
 855   """Refines a pool of boxes using non max suppression and box voting.
 856 
 857   Box refinement is done independently for each class.
 858 
 859   Args:
 860     pool_boxes: (BoxList) A collection of boxes to be refined. pool_boxes must
 861       have a rank 1 'scores' field and a rank 1 'classes' field.
 862     num_classes: (int scalar) Number of classes.
 863     nms_iou_thresh: (float scalar) iou threshold for non max suppression (NMS).
 864     nms_max_detections: (int scalar) maximum output size for NMS.
 865     voting_iou_thresh: (float scalar) iou threshold for box voting.
 866 
 867   Returns:
 868     BoxList of refined boxes.
 869 
 870   Raises:
 871     ValueError: if
 872       a) nms_iou_thresh or voting_iou_thresh is not in [0, 1].
 873       b) pool_boxes is not a BoxList.
 874       c) pool_boxes does not have a scores and classes field.
 875   """
 876   if not 0.0 <= nms_iou_thresh <= 1.0:
 877     raise ValueError('nms_iou_thresh must be between 0 and 1')
 878   if not 0.0 <= voting_iou_thresh <= 1.0:
 879     raise ValueError('voting_iou_thresh must be between 0 and 1')
 880   if not isinstance(pool_boxes, box_list.BoxList):
 881     raise ValueError('pool_boxes must be a BoxList')
 882   if not pool_boxes.has_field('scores'):
 883     raise ValueError('pool_boxes must have a \'scores\' field')
 884   if not pool_boxes.has_field('classes'):
 885     raise ValueError('pool_boxes must have a \'classes\' field')
 886 
 887   refined_boxes = []
 888   for i in range(num_classes):
 889     boxes_class = filter_field_value_equals(pool_boxes, 'classes', i)
 890     refined_boxes_class = refine_boxes(boxes_class, nms_iou_thresh,
 891                                        nms_max_detections, voting_iou_thresh)
 892     refined_boxes.append(refined_boxes_class)
 893   return sort_by_field(concatenate(refined_boxes), 'scores')

 932 def box_voting(selected_boxes, pool_boxes, iou_thresh=0.5):
 933   """Performs box voting as described in S. Gidaris and N. Komodakis, ICCV 2015.
 934 
 935   Performs box voting as described in 'Object detection via a multi-region &
 936   semantic segmentation-aware CNN model', Gidaris and Komodakis, ICCV 2015. For
 937   each box 'B' in selected_boxes, we find the set 'S' of boxes in pool_boxes
 938   with iou overlap >= iou_thresh. The location of B is set to the weighted
 939   average location of boxes in S (scores are used for weighting). And the score
 940   of B is set to the average score of boxes in S.
 941 
 942   Args:
 943     selected_boxes: BoxList containing a subset of boxes in pool_boxes. These
 944       boxes are usually selected from pool_boxes using non max suppression.
 945     pool_boxes: BoxList containing a set of (possibly redundant) boxes.
 946     iou_thresh: (float scalar) iou threshold for matching boxes in
 947       selected_boxes and pool_boxes.
 948 
 949   Returns:
 950     BoxList containing averaged locations and scores for each box in
 951     selected_boxes.
 952 
 953   Raises:
 954     ValueError: if
 955       a) selected_boxes or pool_boxes is not a BoxList.
 956       b) if iou_thresh is not in [0, 1].
 957       c) pool_boxes does not have a scores field.
 958   """
 959   if not 0.0 <= iou_thresh <= 1.0:
 960     raise ValueError('iou_thresh must be between 0 and 1')
 961   if not isinstance(selected_boxes, box_list.BoxList):
 962     raise ValueError('selected_boxes must be a BoxList')
 963   if not isinstance(pool_boxes, box_list.BoxList):
 964     raise ValueError('pool_boxes must be a BoxList')
 965   if not pool_boxes.has_field('scores'):
 966     raise ValueError('pool_boxes must have a \'scores\' field')
 967 
 968   iou_ = iou(selected_boxes, pool_boxes)
 969   match_indicator = tf.cast(tf.greater(iou_, iou_thresh), dtype=tf.float32)
 970   num_matches = tf.reduce_sum(match_indicator, 1)
 971   # TODO(kbanoop): Handle the case where some boxes in selected_boxes do not
 972   # match to any boxes in pool_boxes. For such boxes without any matches, we
 973   # should return the original boxes without voting.
 974   match_assert = tf.Assert(
 975       tf.reduce_all(tf.greater(num_matches, 0)),
 976       ['Each box in selected_boxes must match with at least one box '
 977        'in pool_boxes.'])
 978 
 979   scores = tf.expand_dims(pool_boxes.get_field('scores'), 1)
 980   scores_assert = tf.Assert(
 981       tf.reduce_all(tf.greater_equal(scores, 0)),
 982       ['Scores must be non negative.'])
 983 
 984   with tf.control_dependencies([scores_assert, match_assert]):
 985     sum_scores = tf.matmul(match_indicator, scores)
 986   averaged_scores = tf.reshape(sum_scores, [-1]) / num_matches
 987 
 988   box_locations = tf.matmul(match_indicator,
 989                             pool_boxes.get() * scores) / sum_scores
 990   averaged_boxes = box_list.BoxList(box_locations)
 991   _copy_extra_fields(averaged_boxes, selected_boxes)
 992   averaged_boxes.add_field('scores', averaged_scores)
 993   return averaged_boxes

 996 def get_minimal_coverage_box(boxlist,
 997                              default_box=None,
 998                              scope=None):
 999   """Creates a single bounding box which covers all boxes in the boxlist.
1000 
1001   Args:
1002     boxlist: A Boxlist.
1003     default_box: A [1, 4] float32 tensor. If no boxes are present in `boxlist`,
1004       this default box will be returned. If None, will use a default box of
1005       [[0., 0., 1., 1.]].
1006     scope: Name scope.
1007 
1008   Returns:
1009     A [1, 4] float32 tensor with a bounding box that tightly covers all the
1010     boxes in the box list. If the boxlist does not contain any boxes, the
1011     default box is returned.
1012   """
1013   with tf.name_scope(scope, 'CreateCoverageBox'):
1014     num_boxes = boxlist.num_boxes()
1015 
1016     def coverage_box(bboxes):
1017       y_min, x_min, y_max, x_max = tf.split(
1018           value=bboxes, num_or_size_splits=4, axis=1)
1019       y_min_coverage = tf.reduce_min(y_min, axis=0)
1020       x_min_coverage = tf.reduce_min(x_min, axis=0)
1021       y_max_coverage = tf.reduce_max(y_max, axis=0)
1022       x_max_coverage = tf.reduce_max(x_max, axis=0)
1023       return tf.stack(
1024           [y_min_coverage, x_min_coverage, y_max_coverage, x_max_coverage],
1025           axis=1)
1026 
1027     default_box = default_box or tf.constant([[0., 0., 1., 1.]])
1028     return tf.cond(
1029         tf.greater_equal(num_boxes, 1),
1030         true_fn=lambda: coverage_box(boxlist.get()),
1031         false_fn=lambda: default_box)

1034 def sample_boxes_by_jittering(boxlist,
1035                               num_boxes_to_sample,
1036                               stddev=0.1,
1037                               scope=None):
1038   """Samples num_boxes_to_sample boxes by jittering around boxlist boxes.
1039 
1040   It is possible that this function might generate boxes with size 0. The larger
1041   the stddev, this is more probable. For a small stddev of 0.1 this probability
1042   is very small.
1043 
1044   Args:
1045     boxlist: A boxlist containing N boxes in normalized coordinates.
1046     num_boxes_to_sample: A positive integer containing the number of boxes to
1047       sample.
1048     stddev: Standard deviation. This is used to draw random offsets for the
1049       box corners from a normal distribution. The offset is multiplied by the
1050       box size so will be larger in terms of pixels for larger boxes.
1051     scope: Name scope.
1052 
1053   Returns:
1054     sampled_boxlist: A boxlist containing num_boxes_to_sample boxes in
1055       normalized coordinates.
1056   """
1057   with tf.name_scope(scope, 'SampleBoxesByJittering'):
1058     num_boxes = boxlist.num_boxes()
1059     box_indices = tf.random_uniform(
1060         [num_boxes_to_sample],
1061         minval=0,
1062         maxval=num_boxes,
1063         dtype=tf.int32)
1064     sampled_boxes = tf.gather(boxlist.get(), box_indices)
1065     sampled_boxes_height = sampled_boxes[:, 2] - sampled_boxes[:, 0]
1066     sampled_boxes_width = sampled_boxes[:, 3] - sampled_boxes[:, 1]
1067     rand_miny_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1068     rand_minx_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1069     rand_maxy_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1070     rand_maxx_gaussian = tf.random_normal([num_boxes_to_sample], stddev=stddev)
1071     miny = rand_miny_gaussian * sampled_boxes_height + sampled_boxes[:, 0]
1072     minx = rand_minx_gaussian * sampled_boxes_width + sampled_boxes[:, 1]
1073     maxy = rand_maxy_gaussian * sampled_boxes_height + sampled_boxes[:, 2]
1074     maxx = rand_maxx_gaussian * sampled_boxes_width + sampled_boxes[:, 3]
1075     maxy = tf.maximum(miny, maxy)
1076     maxx = tf.maximum(minx, maxx)
1077     sampled_boxes = tf.stack([miny, minx, maxy, maxx], axis=1)
1078     sampled_boxes = tf.maximum(tf.minimum(sampled_boxes, 1.0), 0.0)
1079     return box_list.BoxList(sampled_boxes)
