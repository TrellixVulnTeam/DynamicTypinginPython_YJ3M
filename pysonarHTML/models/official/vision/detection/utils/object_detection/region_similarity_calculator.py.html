  16 """Region Similarity Calculators for BoxLists.
  17 
  18 Region Similarity Calculators compare a pairwise measure of similarity
  19 between the boxes in two BoxLists.
  20 """
  21 from abc import ABCMeta
  22 from abc import abstractmethod
  23 
  24 import tensorflow as tf

  27 def area(boxlist, scope=None):
  28   """Computes area of boxes.
  29 
  30   Args:
  31     boxlist: BoxList holding N boxes
  32     scope: name scope.
  33 
  34   Returns:
  35     a tensor with shape [N] representing box areas.
  36   """
  37   if not scope:
  38     scope = 'Area'
  39   with tf.name_scope(scope):
  40     y_min, x_min, y_max, x_max = tf.split(
  41         value=boxlist.get(), num_or_size_splits=4, axis=1)
  42     return tf.squeeze((y_max - y_min) * (x_max - x_min), [1])

  45 def intersection(boxlist1, boxlist2, scope=None):
  46   """Compute pairwise intersection areas between boxes.
  47 
  48   Args:
  49     boxlist1: BoxList holding N boxes
  50     boxlist2: BoxList holding M boxes
  51     scope: name scope.
  52 
  53   Returns:
  54     a tensor with shape [N, M] representing pairwise intersections
  55   """
  56   if not scope:
  57     scope = 'Intersection'
  58   with tf.name_scope(scope):
  59     y_min1, x_min1, y_max1, x_max1 = tf.split(
  60         value=boxlist1.get(), num_or_size_splits=4, axis=1)
  61     y_min2, x_min2, y_max2, x_max2 = tf.split(
  62         value=boxlist2.get(), num_or_size_splits=4, axis=1)
  63     all_pairs_min_ymax = tf.minimum(y_max1, tf.transpose(a=y_max2))
  64     all_pairs_max_ymin = tf.maximum(y_min1, tf.transpose(a=y_min2))
  65     intersect_heights = tf.maximum(0.0, all_pairs_min_ymax - all_pairs_max_ymin)
  66     all_pairs_min_xmax = tf.minimum(x_max1, tf.transpose(a=x_max2))
  67     all_pairs_max_xmin = tf.maximum(x_min1, tf.transpose(a=x_min2))
  68     intersect_widths = tf.maximum(0.0, all_pairs_min_xmax - all_pairs_max_xmin)
  69     return intersect_heights * intersect_widths

  72 def iou(boxlist1, boxlist2, scope=None):
  73   """Computes pairwise intersection-over-union between box collections.
  74 
  75   Args:
  76     boxlist1: BoxList holding N boxes
  77     boxlist2: BoxList holding M boxes
  78     scope: name scope.
  79 
  80   Returns:
  81     a tensor with shape [N, M] representing pairwise iou scores.
  82   """
  83   if not scope:
  84     scope = 'IOU'
  85   with tf.name_scope(scope):
  86     intersections = intersection(boxlist1, boxlist2)
  87     areas1 = area(boxlist1)
  88     areas2 = area(boxlist2)
  89     unions = (
  90         tf.expand_dims(areas1, 1) + tf.expand_dims(areas2, 0) - intersections)
  91     return tf.where(
  92         tf.equal(intersections, 0.0), tf.zeros_like(intersections),
  93         tf.truediv(intersections, unions))

  96 class RegionSimilarityCalculator(object):
  97   """Abstract base class for region similarity calculator."""
  98   __metaclass__ = ABCMeta
  99 
 100   def compare(self, boxlist1, boxlist2, scope=None):
 101     """Computes matrix of pairwise similarity between BoxLists.
 102 
 103     This op (to be overriden) computes a measure of pairwise similarity between
 104     the boxes in the given BoxLists. Higher values indicate more similarity.
 105 
 106     Note that this method simply measures similarity and does not explicitly
 107     perform a matching.
 108 
 109     Args:
 110       boxlist1: BoxList holding N boxes.
 111       boxlist2: BoxList holding M boxes.
 112       scope: Op scope name. Defaults to 'Compare' if None.
 113 
 114     Returns:
 115       a (float32) tensor of shape [N, M] with pairwise similarity score.
 116     """
 117     if not scope:
 118       scope = 'Compare'
 119     with tf.name_scope(scope) as scope:
 120       return self._compare(boxlist1, boxlist2)
 121 
 122   @abstractmethod
 123   def _compare(self, boxlist1, boxlist2):
 124     pass

 127 class IouSimilarity(RegionSimilarityCalculator):
 128   """Class to compute similarity based on Intersection over Union (IOU) metric.
 129 
 130   This class computes pairwise similarity between two BoxLists based on IOU.
 131   """
 132 
 133   def _compare(self, boxlist1, boxlist2):
 134     """Compute pairwise IOU similarity between the two BoxLists.
 135 
 136     Args:
 137       boxlist1: BoxList holding N boxes.
 138       boxlist2: BoxList holding M boxes.
 139 
 140     Returns:
 141       A tensor with shape [N, M] representing pairwise iou scores.
 142     """
 143     return iou(boxlist1, boxlist2)
