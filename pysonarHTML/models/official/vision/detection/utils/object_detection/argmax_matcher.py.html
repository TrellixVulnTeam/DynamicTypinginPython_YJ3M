  16 """Argmax matcher implementation.
  17 
  18 This class takes a similarity matrix and matches columns to rows based on the
  19 maximum value per column. One can specify matched_thresholds and
  20 to prevent columns from matching to rows (generally resulting in a negative
  21 training example) and unmatched_theshold to ignore the match (generally
  22 resulting in neither a positive or negative training example).
  23 
  24 This matcher is used in Fast(er)-RCNN.
  25 
  26 Note: matchers are used in TargetAssigners. There is a create_target_assigner
  27 factory function for popular implementations.
  28 """
  29 import tensorflow as tf
  30 
  31 from official.vision.detection.utils.object_detection import matcher
  32 from official.vision.detection.utils.object_detection import shape_utils
  33 
  34 
  35 class ArgMaxMatcher(matcher.Matcher):
  36   """Matcher based on highest value.
  37 
  38   This class computes matches from a similarity matrix. Each column is matched
  39   to a single row.
  40 
  41   To support object detection target assignment this class enables setting both
  42   matched_threshold (upper threshold) and unmatched_threshold (lower thresholds)
  43   defining three categories of similarity which define whether examples are
  44   positive, negative, or ignored:
  45   (1) similarity >= matched_threshold: Highest similarity. Matched/Positive!
  46   (2) matched_threshold > similarity >= unmatched_threshold: Medium similarity.
  47           Depending on negatives_lower_than_unmatched, this is either
  48           Unmatched/Negative OR Ignore.
  49   (3) unmatched_threshold > similarity: Lowest similarity. Depending on flag
  50           negatives_lower_than_unmatched, either Unmatched/Negative OR Ignore.
  51   For ignored matches this class sets the values in the Match object to -2.
  52   """
  53 
  54   def __init__(self,
  55                matched_threshold,
  56                unmatched_threshold=None,
  57                negatives_lower_than_unmatched=True,
  58                force_match_for_each_row=False):
  59     """Construct ArgMaxMatcher.
  60 
  61     Args:
  62       matched_threshold: Threshold for positive matches. Positive if
  63         sim >= matched_threshold, where sim is the maximum value of the
  64         similarity matrix for a given column. Set to None for no threshold.
  65       unmatched_threshold: Threshold for negative matches. Negative if
  66         sim < unmatched_threshold. Defaults to matched_threshold
  67         when set to None.
  68       negatives_lower_than_unmatched: Boolean which defaults to True. If True
  69         then negative matches are the ones below the unmatched_threshold,
  70         whereas ignored matches are in between the matched and umatched
  71         threshold. If False, then negative matches are in between the matched
  72         and unmatched threshold, and everything lower than unmatched is ignored.
  73       force_match_for_each_row: If True, ensures that each row is matched to
  74         at least one column (which is not guaranteed otherwise if the
  75         matched_threshold is high). Defaults to False. See
  76         argmax_matcher_test.testMatcherForceMatch() for an example.
  77 
  78     Raises:
  79       ValueError: if unmatched_threshold is set but matched_threshold is not set
  80         or if unmatched_threshold > matched_threshold.
  81     """
  82     if (matched_threshold is None) and (unmatched_threshold is not None):
  83       raise ValueError('Need to also define matched_threshold when'
  84                        'unmatched_threshold is defined')
  85     self._matched_threshold = matched_threshold
  86     if unmatched_threshold is None:
  87       self._unmatched_threshold = matched_threshold
  88     else:
  89       if unmatched_threshold > matched_threshold:
  90         raise ValueError('unmatched_threshold needs to be smaller or equal'
  91                          'to matched_threshold')
  92       self._unmatched_threshold = unmatched_threshold
  93     if not negatives_lower_than_unmatched:
  94       if self._unmatched_threshold == self._matched_threshold:
  95         raise ValueError('When negatives are in between matched and '
  96                          'unmatched thresholds, these cannot be of equal '
  97                          'value. matched: %s, unmatched: %s',
  98                          self._matched_threshold, self._unmatched_threshold)
  99     self._force_match_for_each_row = force_match_for_each_row
 100     self._negatives_lower_than_unmatched = negatives_lower_than_unmatched
 101 
 102   def _match(self, similarity_matrix):
 103     """Tries to match each column of the similarity matrix to a row.
 104 
 105     Args:
 106       similarity_matrix: tensor of shape [N, M] representing any similarity
 107         metric.
 108 
 109     Returns:
 110       Match object with corresponding matches for each of M columns.
 111     """
 112 
 113     def _match_when_rows_are_empty():
 114       """Performs matching when the rows of similarity matrix are empty.
 115 
 116       When the rows are empty, all detections are false positives. So we return
 117       a tensor of -1's to indicate that the columns do not match to any rows.
 118 
 119       Returns:
 120         matches:  int32 tensor indicating the row each column matches to.
 121       """
 122       similarity_matrix_shape = shape_utils.combined_static_and_dynamic_shape(
 123           similarity_matrix)
 124       return -1 * tf.ones([similarity_matrix_shape[1]], dtype=tf.int32)
 125 
 126     def _match_when_rows_are_non_empty():
 127       """Performs matching when the rows of similarity matrix are non empty.
 128 
 129       Returns:
 130         matches:  int32 tensor indicating the row each column matches to.
 131       """
 132       # Matches for each column
 133       matches = tf.argmax(input=similarity_matrix, axis=0, output_type=tf.int32)
 134 
 135       # Deal with matched and unmatched threshold
 136       if self._matched_threshold is not None:
 137         # Get logical indices of ignored and unmatched columns as tf.int64
 138         matched_vals = tf.reduce_max(input_tensor=similarity_matrix, axis=0)
 139         below_unmatched_threshold = tf.greater(self._unmatched_threshold,
 140                                                matched_vals)
 141         between_thresholds = tf.logical_and(
 142             tf.greater_equal(matched_vals, self._unmatched_threshold),
 143             tf.greater(self._matched_threshold, matched_vals))
 144 
 145         if self._negatives_lower_than_unmatched:
 146           matches = self._set_values_using_indicator(matches,
 147                                                      below_unmatched_threshold,
 148                                                      -1)
 149           matches = self._set_values_using_indicator(matches,
 150                                                      between_thresholds,
 151                                                      -2)
 152         else:
 153           matches = self._set_values_using_indicator(matches,
 154                                                      below_unmatched_threshold,
 155                                                      -2)
 156           matches = self._set_values_using_indicator(matches,
 157                                                      between_thresholds,
 158                                                      -1)
 159 
 160       if self._force_match_for_each_row:
 161         similarity_matrix_shape = shape_utils.combined_static_and_dynamic_shape(
 162             similarity_matrix)
 163         force_match_column_ids = tf.argmax(
 164             input=similarity_matrix, axis=1, output_type=tf.int32)
 165         force_match_column_indicators = tf.one_hot(
 166             force_match_column_ids, depth=similarity_matrix_shape[1])
 167         force_match_row_ids = tf.argmax(
 168             input=force_match_column_indicators, axis=0, output_type=tf.int32)
 169         force_match_column_mask = tf.cast(
 170             tf.reduce_max(input_tensor=force_match_column_indicators, axis=0),
 171             tf.bool)
 172         final_matches = tf.where(force_match_column_mask, force_match_row_ids,
 173                                  matches)
 174         return final_matches
 175       else:
 176         return matches
 177 
 178     if similarity_matrix.shape.is_fully_defined():
 179       if similarity_matrix.shape.dims[0].value == 0:
 180         return _match_when_rows_are_empty()
 181       else:
 182         return _match_when_rows_are_non_empty()
 183     else:
 184       return tf.cond(
 185           pred=tf.greater(tf.shape(input=similarity_matrix)[0], 0),
 186           true_fn=_match_when_rows_are_non_empty,
 187           false_fn=_match_when_rows_are_empty)
 188 
 189   def _set_values_using_indicator(self, x, indicator, val):
 190     """Set the indicated fields of x to val.
 191 
 192     Args:
 193       x: tensor.
 194       indicator: boolean with same shape as x.
 195       val: scalar with value to set.
 196 
 197     Returns:
 198       modified tensor.
 199     """
 200     indicator = tf.cast(indicator, x.dtype)
 201     return tf.add(tf.multiply(x, 1 - indicator), val * indicator)
