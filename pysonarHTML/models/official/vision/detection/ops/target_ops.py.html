  28 def box_matching(boxes, gt_boxes, gt_classes):
  29   """Match boxes to groundtruth boxes.
  30 
  31   Given the proposal boxes and the groundtruth boxes and classes, perform the
  32   groundtruth matching by taking the argmax of the IoU between boxes and
  33   groundtruth boxes.
  34 
  35   Args:
  36     boxes: a tensor of shape of [batch_size, N, 4] representing the box
  37       coordiantes to be matched to groundtruth boxes.
  38     gt_boxes: a tensor of shape of [batch_size, MAX_INSTANCES, 4] representing
  39       the groundtruth box coordinates. It is padded with -1s to indicate the
  40       invalid boxes.
  41     gt_classes: [batch_size, MAX_INSTANCES] representing the groundtruth box
  42       classes. It is padded with -1s to indicate the invalid classes.
  43 
  44   Returns:
  45     matched_gt_boxes: a tensor of shape of [batch_size, N, 4], representing
  46       the matched groundtruth box coordinates for each input box. If the box
  47       does not overlap with any groundtruth boxes, the matched boxes of it
  48       will be set to all 0s.
  49     matched_gt_classes: a tensor of shape of [batch_size, N], representing
  50       the matched groundtruth classes for each input box. If the box does not
  51       overlap with any groundtruth boxes, the matched box classes of it will
  52       be set to 0, which corresponds to the background class.
  53     matched_gt_indices: a tensor of shape of [batch_size, N], representing
  54       the indices of the matched groundtruth boxes in the original gt_boxes
  55       tensor. If the box does not overlap with any groundtruth boxes, the
  56       index of the matched groundtruth will be set to -1.
  57     matched_iou: a tensor of shape of [batch_size, N], representing the IoU
  58       between the box and its matched groundtruth box. The matched IoU is the
  59       maximum IoU of the box and all the groundtruth boxes.
  60     iou: a tensor of shape of [batch_size, N, K], representing the IoU matrix
  61       between boxes and the groundtruth boxes. The IoU between a box and the
  62       invalid groundtruth boxes whose coordinates are [-1, -1, -1, -1] is -1.
  63   """
  64   # Compute IoU between boxes and gt_boxes.
  65   # iou <- [batch_size, N, K]
  66   iou = box_utils.bbox_overlap(boxes, gt_boxes)
  67 
  68   # max_iou <- [batch_size, N]
  69   # 0.0 -> no match to gt, or -1.0 match to no gt
  70   matched_iou = tf.reduce_max(iou, axis=-1)
  71 
  72   # background_box_mask <- bool, [batch_size, N]
  73   background_box_mask = tf.less_equal(matched_iou, 0.0)
  74 
  75   argmax_iou_indices = tf.argmax(iou, axis=-1, output_type=tf.int32)
  76 
  77   argmax_iou_indices_shape = tf.shape(argmax_iou_indices)
  78   batch_indices = (
  79       tf.expand_dims(tf.range(argmax_iou_indices_shape[0]), axis=-1) *
  80       tf.ones([1, argmax_iou_indices_shape[-1]], dtype=tf.int32))
  81   gather_nd_indices = tf.stack([batch_indices, argmax_iou_indices], axis=-1)
  82 
  83   matched_gt_boxes = tf.gather_nd(gt_boxes, gather_nd_indices)
  84   matched_gt_boxes = tf.where(
  85       tf.tile(tf.expand_dims(background_box_mask, axis=-1), [1, 1, 4]),
  86       tf.zeros_like(matched_gt_boxes, dtype=matched_gt_boxes.dtype),
  87       matched_gt_boxes)
  88 
  89   matched_gt_classes = tf.gather_nd(gt_classes, gather_nd_indices)
  90   matched_gt_classes = tf.where(
  91       background_box_mask,
  92       tf.zeros_like(matched_gt_classes),
  93       matched_gt_classes)
  94 
  95   matched_gt_indices = tf.where(
  96       background_box_mask,
  97       -tf.ones_like(argmax_iou_indices),
  98       argmax_iou_indices)
  99 
 100   return (matched_gt_boxes, matched_gt_classes, matched_gt_indices,
 101           matched_iou, iou)

 104 def assign_and_sample_proposals(proposed_boxes,
 105                                 gt_boxes,
 106                                 gt_classes,
 107                                 num_samples_per_image=512,
 108                                 mix_gt_boxes=True,
 109                                 fg_fraction=0.25,
 110                                 fg_iou_thresh=0.5,
 111                                 bg_iou_thresh_hi=0.5,
 112                                 bg_iou_thresh_lo=0.0):
 113   """Assigns the proposals with groundtruth classes and performs subsmpling.
 114 
 115   Given `proposed_boxes`, `gt_boxes`, and `gt_classes`, the function uses the
 116   following algorithm to generate the final `num_samples_per_image` RoIs.
 117     1. Calculates the IoU between each proposal box and each gt_boxes.
 118     2. Assigns each proposed box with a groundtruth class and box by choosing
 119        the largest IoU overlap.
 120     3. Samples `num_samples_per_image` boxes from all proposed boxes, and
 121        returns box_targets, class_targets, and RoIs.
 122 
 123   Args:
 124     proposed_boxes: a tensor of shape of [batch_size, N, 4]. N is the number
 125       of proposals before groundtruth assignment. The last dimension is the
 126       box coordinates w.r.t. the scaled images in [ymin, xmin, ymax, xmax]
 127       format.
 128     gt_boxes: a tensor of shape of [batch_size, MAX_NUM_INSTANCES, 4].
 129       The coordinates of gt_boxes are in the pixel coordinates of the scaled
 130       image. This tensor might have padding of values -1 indicating the invalid
 131       box coordinates.
 132     gt_classes: a tensor with a shape of [batch_size, MAX_NUM_INSTANCES]. This
 133       tensor might have paddings with values of -1 indicating the invalid
 134       classes.
 135     num_samples_per_image: a integer represents RoI minibatch size per image.
 136     mix_gt_boxes: a bool indicating whether to mix the groundtruth boxes before
 137       sampling proposals.
 138     fg_fraction: a float represents the target fraction of RoI minibatch that
 139       is labeled foreground (i.e., class > 0).
 140     fg_iou_thresh: a float represents the IoU overlap threshold for an RoI to be
 141       considered foreground (if >= fg_iou_thresh).
 142     bg_iou_thresh_hi: a float represents the IoU overlap threshold for an RoI to
 143       be considered background (class = 0 if overlap in [LO, HI)).
 144     bg_iou_thresh_lo: a float represents the IoU overlap threshold for an RoI to
 145       be considered background (class = 0 if overlap in [LO, HI)).
 146 
 147   Returns:
 148     sampled_rois: a tensor of shape of [batch_size, K, 4], representing the
 149       coordinates of the sampled RoIs, where K is the number of the sampled
 150       RoIs, i.e. K = num_samples_per_image.
 151     sampled_gt_boxes: a tensor of shape of [batch_size, K, 4], storing the
 152       box coordinates of the matched groundtruth boxes of the samples RoIs.
 153     sampled_gt_classes: a tensor of shape of [batch_size, K], storing the
 154       classes of the matched groundtruth boxes of the sampled RoIs.
 155     sampled_gt_indices: a tensor of shape of [batch_size, K], storing the
 156       indices of the sampled groudntruth boxes in the original `gt_boxes`
 157       tensor, i.e. gt_boxes[sampled_gt_indices[:, i]] = sampled_gt_boxes[:, i].
 158   """
 159 
 160   with tf.name_scope('sample_proposals'):
 161     if mix_gt_boxes:
 162       boxes = tf.concat([proposed_boxes, gt_boxes], axis=1)
 163     else:
 164       boxes = proposed_boxes
 165 
 166     (matched_gt_boxes, matched_gt_classes, matched_gt_indices,
 167      matched_iou, _) = box_matching(boxes, gt_boxes, gt_classes)
 168 
 169     positive_match = tf.greater(matched_iou, fg_iou_thresh)
 170     negative_match = tf.logical_and(
 171         tf.greater_equal(matched_iou, bg_iou_thresh_lo),
 172         tf.less(matched_iou, bg_iou_thresh_hi))
 173     ignored_match = tf.less(matched_iou, 0.0)
 174 
 175     # re-assign negatively matched boxes to the background class.
 176     matched_gt_classes = tf.where(
 177         negative_match, tf.zeros_like(matched_gt_classes), matched_gt_classes)
 178     matched_gt_indices = tf.where(
 179         negative_match, tf.zeros_like(matched_gt_indices), matched_gt_indices)
 180 
 181     sample_candidates = tf.logical_and(
 182         tf.logical_or(positive_match, negative_match),
 183         tf.logical_not(ignored_match))
 184 
 185     sampler = (
 186         balanced_positive_negative_sampler.BalancedPositiveNegativeSampler(
 187             positive_fraction=fg_fraction, is_static=True))
 188 
 189     batch_size, _ = sample_candidates.get_shape().as_list()
 190     sampled_indicators = []
 191     for i in range(batch_size):
 192       sampled_indicator = sampler.subsample(
 193           sample_candidates[i], num_samples_per_image, positive_match[i])
 194       sampled_indicators.append(sampled_indicator)
 195     sampled_indicators = tf.stack(sampled_indicators)
 196     _, sampled_indices = tf.nn.top_k(
 197         tf.cast(sampled_indicators, dtype=tf.int32),
 198         k=num_samples_per_image,
 199         sorted=True)
 200 
 201     sampled_indices_shape = tf.shape(sampled_indices)
 202     batch_indices = (
 203         tf.expand_dims(tf.range(sampled_indices_shape[0]), axis=-1) *
 204         tf.ones([1, sampled_indices_shape[-1]], dtype=tf.int32))
 205     gather_nd_indices = tf.stack([batch_indices, sampled_indices], axis=-1)
 206 
 207     sampled_rois = tf.gather_nd(boxes, gather_nd_indices)
 208     sampled_gt_boxes = tf.gather_nd(matched_gt_boxes, gather_nd_indices)
 209     sampled_gt_classes = tf.gather_nd(
 210         matched_gt_classes, gather_nd_indices)
 211     sampled_gt_indices = tf.gather_nd(
 212         matched_gt_indices, gather_nd_indices)
 213 
 214     return (sampled_rois, sampled_gt_boxes, sampled_gt_classes,
 215             sampled_gt_indices)

 218 def sample_and_crop_foreground_masks(candidate_rois,
 219                                      candidate_gt_boxes,
 220                                      candidate_gt_classes,
 221                                      candidate_gt_indices,
 222                                      gt_masks,
 223                                      num_mask_samples_per_image=128,
 224                                      mask_target_size=28):
 225   """Samples and creates cropped foreground masks for training.
 226 
 227   Args:
 228     candidate_rois: a tensor of shape of [batch_size, N, 4], where N is the
 229       number of candidate RoIs to be considered for mask sampling. It includes
 230       both positive and negative RoIs. The `num_mask_samples_per_image` positive
 231       RoIs will be sampled to create mask training targets.
 232     candidate_gt_boxes: a tensor of shape of [batch_size, N, 4], storing the
 233       corresponding groundtruth boxes to the `candidate_rois`.
 234     candidate_gt_classes: a tensor of shape of [batch_size, N], storing the
 235       corresponding groundtruth classes to the `candidate_rois`. 0 in the tensor
 236       corresponds to the background class, i.e. negative RoIs.
 237     candidate_gt_indices: a tensor of shape [batch_size, N], storing the
 238       corresponding groundtruth instance indices to the `candidate_gt_boxes`,
 239       i.e. gt_boxes[candidate_gt_indices[:, i]] = candidate_gt_boxes[:, i] and
 240       gt_boxes which is of shape [batch_size, MAX_INSTANCES, 4], M >= N, is the
 241       superset of candidate_gt_boxes.
 242     gt_masks: a tensor of [batch_size, MAX_INSTANCES, mask_height, mask_width]
 243       containing all the groundtruth masks which sample masks are drawn from.
 244     num_mask_samples_per_image: an integer which specifies the number of masks
 245       to sample.
 246     mask_target_size: an integer which specifies the final cropped mask size
 247       after sampling. The output masks are resized w.r.t the sampled RoIs.
 248 
 249   Returns:
 250     foreground_rois: a tensor of shape of [batch_size, K, 4] storing the RoI
 251       that corresponds to the sampled foreground masks, where
 252       K = num_mask_samples_per_image.
 253     foreground_classes: a tensor of shape of [batch_size, K] storing the classes
 254       corresponding to the sampled foreground masks.
 255     cropoped_foreground_masks: a tensor of shape of
 256       [batch_size, K, mask_target_size, mask_target_size] storing the cropped
 257       foreground masks used for training.
 258   """
 259   with tf.name_scope('sample_and_crop_foreground_masks'):
 260     _, fg_instance_indices = tf.nn.top_k(
 261         tf.cast(tf.greater(candidate_gt_classes, 0), dtype=tf.int32),
 262         k=num_mask_samples_per_image)
 263 
 264     fg_instance_indices_shape = tf.shape(fg_instance_indices)
 265     batch_indices = (
 266         tf.expand_dims(tf.range(fg_instance_indices_shape[0]), axis=-1) *
 267         tf.ones([1, fg_instance_indices_shape[-1]], dtype=tf.int32))
 268 
 269     gather_nd_instance_indices = tf.stack(
 270         [batch_indices, fg_instance_indices], axis=-1)
 271     foreground_rois = tf.gather_nd(
 272         candidate_rois, gather_nd_instance_indices)
 273     foreground_boxes = tf.gather_nd(
 274         candidate_gt_boxes, gather_nd_instance_indices)
 275     foreground_classes = tf.gather_nd(
 276         candidate_gt_classes, gather_nd_instance_indices)
 277     foreground_gt_indices = tf.gather_nd(
 278         candidate_gt_indices, gather_nd_instance_indices)
 279 
 280     foreground_gt_indices_shape = tf.shape(foreground_gt_indices)
 281     batch_indices = (
 282         tf.expand_dims(tf.range(foreground_gt_indices_shape[0]), axis=-1) *
 283         tf.ones([1, foreground_gt_indices_shape[-1]], dtype=tf.int32))
 284     gather_nd_gt_indices = tf.stack(
 285         [batch_indices, foreground_gt_indices], axis=-1)
 286     foreground_masks = tf.gather_nd(gt_masks, gather_nd_gt_indices)
 287 
 288     cropped_foreground_masks = spatial_transform_ops.crop_mask_in_target_box(
 289         foreground_masks, foreground_boxes, foreground_rois, mask_target_size,
 290         sample_offset=0.5)
 291 
 292     return foreground_rois, foreground_classes, cropped_foreground_masks

 295 class ROISampler(object):
 296   """Samples RoIs and creates training targets."""
 297 
 298   def __init__(self, params):
 299     self._num_samples_per_image = params.num_samples_per_image
 300     self._fg_fraction = params.fg_fraction
 301     self._fg_iou_thresh = params.fg_iou_thresh
 302     self._bg_iou_thresh_hi = params.bg_iou_thresh_hi
 303     self._bg_iou_thresh_lo = params.bg_iou_thresh_lo
 304     self._mix_gt_boxes = params.mix_gt_boxes
 305 
 306   def __call__(self, rois, gt_boxes, gt_classes):
 307     """Sample and assign RoIs for training.
 308 
 309     Args:
 310       rois: a tensor of shape of [batch_size, N, 4]. N is the number
 311         of proposals before groundtruth assignment. The last dimension is the
 312         box coordinates w.r.t. the scaled images in [ymin, xmin, ymax, xmax]
 313         format.
 314       gt_boxes: a tensor of shape of [batch_size, MAX_NUM_INSTANCES, 4].
 315         The coordinates of gt_boxes are in the pixel coordinates of the scaled
 316         image. This tensor might have padding of values -1 indicating the
 317         invalid box coordinates.
 318       gt_classes: a tensor with a shape of [batch_size, MAX_NUM_INSTANCES]. This
 319         tensor might have paddings with values of -1 indicating the invalid
 320         classes.
 321 
 322     Returns:
 323       sampled_rois: a tensor of shape of [batch_size, K, 4], representing the
 324         coordinates of the sampled RoIs, where K is the number of the sampled
 325         RoIs, i.e. K = num_samples_per_image.
 326       sampled_gt_boxes: a tensor of shape of [batch_size, K, 4], storing the
 327         box coordinates of the matched groundtruth boxes of the samples RoIs.
 328       sampled_gt_classes: a tensor of shape of [batch_size, K], storing the
 329         classes of the matched groundtruth boxes of the sampled RoIs.
 330     """
 331     sampled_rois, sampled_gt_boxes, sampled_gt_classes, sampled_gt_indices = (
 332         assign_and_sample_proposals(
 333             rois,
 334             gt_boxes,
 335             gt_classes,
 336             num_samples_per_image=self._num_samples_per_image,
 337             mix_gt_boxes=self._mix_gt_boxes,
 338             fg_fraction=self._fg_fraction,
 339             fg_iou_thresh=self._fg_iou_thresh,
 340             bg_iou_thresh_hi=self._bg_iou_thresh_hi,
 341             bg_iou_thresh_lo=self._bg_iou_thresh_lo))
 342     return (sampled_rois, sampled_gt_boxes, sampled_gt_classes,
 343             sampled_gt_indices)

 346 class MaskSampler(object):
 347   """Samples and creates mask training targets."""
 348 
 349   def __init__(self, mask_target_size, num_mask_samples_per_image):
 350     self._mask_target_size = mask_target_size
 351     self._num_mask_samples_per_image = num_mask_samples_per_image
 352 
 353   def __call__(self,
 354                candidate_rois,
 355                candidate_gt_boxes,
 356                candidate_gt_classes,
 357                candidate_gt_indices,
 358                gt_masks):
 359     """Sample and create mask targets for training.
 360 
 361     Args:
 362       candidate_rois: a tensor of shape of [batch_size, N, 4], where N is the
 363         number of candidate RoIs to be considered for mask sampling. It includes
 364         both positive and negative RoIs. The `num_mask_samples_per_image`
 365         positive RoIs will be sampled to create mask training targets.
 366       candidate_gt_boxes: a tensor of shape of [batch_size, N, 4], storing the
 367         corresponding groundtruth boxes to the `candidate_rois`.
 368       candidate_gt_classes: a tensor of shape of [batch_size, N], storing the
 369         corresponding groundtruth classes to the `candidate_rois`. 0 in the
 370         tensor corresponds to the background class, i.e. negative RoIs.
 371       candidate_gt_indices: a tensor of shape [batch_size, N], storing the
 372         corresponding groundtruth instance indices to the `candidate_gt_boxes`,
 373         i.e. gt_boxes[candidate_gt_indices[:, i]] = candidate_gt_boxes[:, i],
 374         where gt_boxes which is of shape [batch_size, MAX_INSTANCES, 4], M >= N,
 375         is the superset of candidate_gt_boxes.
 376       gt_masks: a tensor of [batch_size, MAX_INSTANCES, mask_height, mask_width]
 377         containing all the groundtruth masks which sample masks are drawn from.
 378         after sampling. The output masks are resized w.r.t the sampled RoIs.
 379 
 380     Returns:
 381       foreground_rois: a tensor of shape of [batch_size, K, 4] storing the RoI
 382         that corresponds to the sampled foreground masks, where
 383         K = num_mask_samples_per_image.
 384       foreground_classes: a tensor of shape of [batch_size, K] storing the
 385         classes corresponding to the sampled foreground masks.
 386       cropoped_foreground_masks: a tensor of shape of
 387         [batch_size, K, mask_target_size, mask_target_size] storing the
 388         cropped foreground masks used for training.
 389     """
 390     foreground_rois, foreground_classes, cropped_foreground_masks = (
 391         sample_and_crop_foreground_masks(
 392             candidate_rois,
 393             candidate_gt_boxes,
 394             candidate_gt_classes,
 395             candidate_gt_indices,
 396             gt_masks,
 397             self._num_mask_samples_per_image,
 398             self._mask_target_size))
 399     return foreground_rois, foreground_classes, cropped_foreground_masks
