  29 def _self_suppression(iou, _, iou_sum):
  30   batch_size = tf.shape(iou)[0]
  31   can_suppress_others = tf.cast(
  32       tf.reshape(tf.reduce_max(iou, 1) <= 0.5, [batch_size, -1, 1]), iou.dtype)
  33   iou_suppressed = tf.reshape(
  34       tf.cast(tf.reduce_max(can_suppress_others * iou, 1) <= 0.5, iou.dtype),
  35       [batch_size, -1, 1]) * iou
  36   iou_sum_new = tf.reduce_sum(iou_suppressed, [1, 2])
  37   return [
  38       iou_suppressed,
  39       tf.reduce_any(iou_sum - iou_sum_new > 0.5), iou_sum_new
  40   ]

  43 def _cross_suppression(boxes, box_slice, iou_threshold, inner_idx):
  44   batch_size = tf.shape(boxes)[0]
  45   new_slice = tf.slice(boxes, [0, inner_idx * NMS_TILE_SIZE, 0],
  46                        [batch_size, NMS_TILE_SIZE, 4])
  47   iou = box_utils.bbox_overlap(new_slice, box_slice)
  48   ret_slice = tf.expand_dims(
  49       tf.cast(tf.reduce_all(iou < iou_threshold, [1]), box_slice.dtype),
  50       2) * box_slice
  51   return boxes, ret_slice, iou_threshold, inner_idx + 1

  54 def _suppression_loop_body(boxes, iou_threshold, output_size, idx):
  55   """Process boxes in the range [idx*NMS_TILE_SIZE, (idx+1)*NMS_TILE_SIZE).
  56 
  57   Args:
  58     boxes: a tensor with a shape of [batch_size, anchors, 4].
  59     iou_threshold: a float representing the threshold for deciding whether boxes
  60       overlap too much with respect to IOU.
  61     output_size: an int32 tensor of size [batch_size]. Representing the number
  62       of selected boxes for each batch.
  63     idx: an integer scalar representing induction variable.
  64 
  65   Returns:
  66     boxes: updated boxes.
  67     iou_threshold: pass down iou_threshold to the next iteration.
  68     output_size: the updated output_size.
  69     idx: the updated induction variable.
  70   """
  71   num_tiles = tf.shape(boxes)[1] // NMS_TILE_SIZE
  72   batch_size = tf.shape(boxes)[0]
  73 
  74   # Iterates over tiles that can possibly suppress the current tile.
  75   box_slice = tf.slice(boxes, [0, idx * NMS_TILE_SIZE, 0],
  76                        [batch_size, NMS_TILE_SIZE, 4])
  77   _, box_slice, _, _ = tf.while_loop(
  78       lambda _boxes, _box_slice, _threshold, inner_idx: inner_idx < idx,
  79       _cross_suppression, [boxes, box_slice, iou_threshold,
  80                            tf.constant(0)])
  81 
  82   # Iterates over the current tile to compute self-suppression.
  83   iou = box_utils.bbox_overlap(box_slice, box_slice)
  84   mask = tf.expand_dims(
  85       tf.reshape(tf.range(NMS_TILE_SIZE), [1, -1]) > tf.reshape(
  86           tf.range(NMS_TILE_SIZE), [-1, 1]), 0)
  87   iou *= tf.cast(tf.logical_and(mask, iou >= iou_threshold), iou.dtype)
  88   suppressed_iou, _, _ = tf.while_loop(
  89       lambda _iou, loop_condition, _iou_sum: loop_condition, _self_suppression,
  90       [iou, tf.constant(True),
  91        tf.reduce_sum(iou, [1, 2])])
  92   suppressed_box = tf.reduce_sum(suppressed_iou, 1) > 0
  93   box_slice *= tf.expand_dims(1.0 - tf.cast(suppressed_box, box_slice.dtype), 2)
  94 
  95   # Uses box_slice to update the input boxes.
  96   mask = tf.reshape(
  97       tf.cast(tf.equal(tf.range(num_tiles), idx), boxes.dtype), [1, -1, 1, 1])
  98   boxes = tf.tile(tf.expand_dims(
  99       box_slice, [1]), [1, num_tiles, 1, 1]) * mask + tf.reshape(
 100           boxes, [batch_size, num_tiles, NMS_TILE_SIZE, 4]) * (1 - mask)
 101   boxes = tf.reshape(boxes, [batch_size, -1, 4])
 102 
 103   # Updates output_size.
 104   output_size += tf.reduce_sum(
 105       tf.cast(tf.reduce_any(box_slice > 0, [2]), tf.int32), [1])
 106   return boxes, iou_threshold, output_size, idx + 1

 109 def sorted_non_max_suppression_padded(scores,
 110                                       boxes,
 111                                       max_output_size,
 112                                       iou_threshold):
 113   """A wrapper that handles non-maximum suppression.
 114 
 115   Assumption:
 116     * The boxes are sorted by scores unless the box is a dot (all coordinates
 117       are zero).
 118     * Boxes with higher scores can be used to suppress boxes with lower scores.
 119 
 120   The overal design of the algorithm is to handle boxes tile-by-tile:
 121 
 122   boxes = boxes.pad_to_multiply_of(tile_size)
 123   num_tiles = len(boxes) // tile_size
 124   output_boxes = []
 125   for i in range(num_tiles):
 126     box_tile = boxes[i*tile_size : (i+1)*tile_size]
 127     for j in range(i - 1):
 128       suppressing_tile = boxes[j*tile_size : (j+1)*tile_size]
 129       iou = bbox_overlap(box_tile, suppressing_tile)
 130       # if the box is suppressed in iou, clear it to a dot
 131       box_tile *= _update_boxes(iou)
 132     # Iteratively handle the diagnal tile.
 133     iou = _box_overlap(box_tile, box_tile)
 134     iou_changed = True
 135     while iou_changed:
 136       # boxes that are not suppressed by anything else
 137       suppressing_boxes = _get_suppressing_boxes(iou)
 138       # boxes that are suppressed by suppressing_boxes
 139       suppressed_boxes = _get_suppressed_boxes(iou, suppressing_boxes)
 140       # clear iou to 0 for boxes that are suppressed, as they cannot be used
 141       # to suppress other boxes any more
 142       new_iou = _clear_iou(iou, suppressed_boxes)
 143       iou_changed = (new_iou != iou)
 144       iou = new_iou
 145     # remaining boxes that can still suppress others, are selected boxes.
 146     output_boxes.append(_get_suppressing_boxes(iou))
 147     if len(output_boxes) >= max_output_size:
 148       break
 149 
 150   Args:
 151     scores: a tensor with a shape of [batch_size, anchors].
 152     boxes: a tensor with a shape of [batch_size, anchors, 4].
 153     max_output_size: a scalar integer `Tensor` representing the maximum number
 154       of boxes to be selected by non max suppression.
 155     iou_threshold: a float representing the threshold for deciding whether boxes
 156       overlap too much with respect to IOU.
 157 
 158   Returns:
 159     nms_scores: a tensor with a shape of [batch_size, anchors]. It has same
 160       dtype as input scores.
 161     nms_proposals: a tensor with a shape of [batch_size, anchors, 4]. It has
 162       same dtype as input boxes.
 163   """
 164   batch_size = tf.shape(boxes)[0]
 165   num_boxes = tf.shape(boxes)[1]
 166   pad = tf.cast(
 167       tf.math.ceil(tf.cast(num_boxes, tf.float32) / NMS_TILE_SIZE),
 168       tf.int32) * NMS_TILE_SIZE - num_boxes
 169   boxes = tf.pad(tf.cast(boxes, tf.float32), [[0, 0], [0, pad], [0, 0]])
 170   scores = tf.pad(
 171       tf.cast(scores, tf.float32), [[0, 0], [0, pad]], constant_values=-1)
 172   num_boxes += pad
 173 
 174   def _loop_cond(unused_boxes, unused_threshold, output_size, idx):
 175     return tf.logical_and(
 176         tf.reduce_min(output_size) < max_output_size,
 177         idx < num_boxes // NMS_TILE_SIZE)
 178 
 179   selected_boxes, _, output_size, _ = tf.while_loop(
 180       _loop_cond, _suppression_loop_body, [
 181           boxes, iou_threshold,
 182           tf.zeros([batch_size], tf.int32),
 183           tf.constant(0)
 184       ])
 185   idx = num_boxes - tf.cast(
 186       tf.nn.top_k(
 187           tf.cast(tf.reduce_any(selected_boxes > 0, [2]), tf.int32) *
 188           tf.expand_dims(tf.range(num_boxes, 0, -1), 0), max_output_size)[0],
 189       tf.int32)
 190   idx = tf.minimum(idx, num_boxes - 1)
 191   idx = tf.reshape(
 192       idx + tf.reshape(tf.range(batch_size) * num_boxes, [-1, 1]), [-1])
 193   boxes = tf.reshape(
 194       tf.gather(tf.reshape(boxes, [-1, 4]), idx),
 195       [batch_size, max_output_size, 4])
 196   boxes = boxes * tf.cast(
 197       tf.reshape(tf.range(max_output_size), [1, -1, 1]) < tf.reshape(
 198           output_size, [-1, 1, 1]), boxes.dtype)
 199   scores = tf.reshape(
 200       tf.gather(tf.reshape(scores, [-1, 1]), idx),
 201       [batch_size, max_output_size])
 202   scores = scores * tf.cast(
 203       tf.reshape(tf.range(max_output_size), [1, -1]) < tf.reshape(
 204           output_size, [-1, 1]), scores.dtype)
 205   return scores, boxes
