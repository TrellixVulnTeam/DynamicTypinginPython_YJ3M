  27 def multilevel_propose_rois(rpn_boxes,
  28                             rpn_scores,
  29                             anchor_boxes,
  30                             image_shape,
  31                             rpn_pre_nms_top_k=2000,
  32                             rpn_post_nms_top_k=1000,
  33                             rpn_nms_threshold=0.7,
  34                             rpn_score_threshold=0.0,
  35                             rpn_min_size_threshold=0.0,
  36                             decode_boxes=True,
  37                             clip_boxes=True,
  38                             use_batched_nms=False,
  39                             apply_sigmoid_to_score=True):
  40   """Proposes RoIs given a group of candidates from different FPN levels.
  41 
  42   The following describes the steps:
  43     1. For each individual level:
  44       a. Apply sigmoid transform if specified.
  45       b. Decode boxes if specified.
  46       c. Clip boxes if specified.
  47       d. Filter small boxes and those fall outside image if specified.
  48       e. Apply pre-NMS filtering including pre-NMS top k and score thresholding.
  49       f. Apply NMS.
  50     2. Aggregate post-NMS boxes from each level.
  51     3. Apply an overall top k to generate the final selected RoIs.
  52 
  53   Args:
  54     rpn_boxes: a dict with keys representing FPN levels and values representing
  55       box tenors of shape [batch_size, feature_h, feature_w, num_anchors * 4].
  56     rpn_scores: a dict with keys representing FPN levels and values representing
  57       logit tensors of shape [batch_size, feature_h, feature_w, num_anchors].
  58     anchor_boxes: a dict with keys representing FPN levels and values
  59       representing anchor box tensors of shape
  60       [batch_size, feature_h, feature_w, num_anchors * 4].
  61     image_shape: a tensor of shape [batch_size, 2] where the last dimension are
  62       [height, width] of the scaled image.
  63     rpn_pre_nms_top_k: an integer of top scoring RPN proposals *per level* to
  64       keep before applying NMS. Default: 2000.
  65     rpn_post_nms_top_k: an integer of top scoring RPN proposals *in total* to
  66       keep after applying NMS. Default: 1000.
  67     rpn_nms_threshold: a float between 0 and 1 representing the IoU threshold
  68       used for NMS. If 0.0, no NMS is applied. Default: 0.7.
  69     rpn_score_threshold: a float between 0 and 1 representing the minimal box
  70       score to keep before applying NMS. This is often used as a pre-filtering
  71       step for better performance. If 0, no filtering is applied. Default: 0.
  72     rpn_min_size_threshold: a float representing the minimal box size in each
  73       side (w.r.t. the scaled image) to keep before applying NMS. This is often
  74       used as a pre-filtering step for better performance. If 0, no filtering is
  75       applied. Default: 0.
  76     decode_boxes: a boolean indicating whether `rpn_boxes` needs to be decoded
  77       using `anchor_boxes`. If False, use `rpn_boxes` directly and ignore
  78       `anchor_boxes`. Default: True.
  79     clip_boxes: a boolean indicating whether boxes are first clipped to the
  80       scaled image size before appliying NMS. If False, no clipping is applied
  81       and `image_shape` is ignored. Default: True.
  82     use_batched_nms: a boolean indicating whether NMS is applied in batch using
  83       `tf.image.combined_non_max_suppression`. Currently only available in
  84       CPU/GPU. Default: False.
  85     apply_sigmoid_to_score: a boolean indicating whether apply sigmoid to
  86       `rpn_scores` before applying NMS. Default: True.
  87 
  88   Returns:
  89     selected_rois: a tensor of shape [batch_size, rpn_post_nms_top_k, 4],
  90       representing the box coordinates of the selected proposals w.r.t. the
  91       scaled image.
  92     selected_roi_scores: a tensor of shape [batch_size, rpn_post_nms_top_k, 1],
  93       representing the scores of the selected proposals.
  94   """
  95   with tf.name_scope('multilevel_propose_rois'):
  96     rois = []
  97     roi_scores = []
  98     image_shape = tf.expand_dims(image_shape, axis=1)
  99     for level in sorted(rpn_scores.keys()):
 100       with tf.name_scope('level_%d' % level):
 101         _, feature_h, feature_w, num_anchors_per_location = (
 102             rpn_scores[level].get_shape().as_list())
 103 
 104         num_boxes = feature_h * feature_w * num_anchors_per_location
 105         this_level_scores = tf.reshape(rpn_scores[level], [-1, num_boxes])
 106         this_level_boxes = tf.reshape(rpn_boxes[level], [-1, num_boxes, 4])
 107         this_level_anchors = tf.cast(
 108             tf.reshape(anchor_boxes[level], [-1, num_boxes, 4]),
 109             dtype=this_level_scores.dtype)
 110 
 111         if apply_sigmoid_to_score:
 112           this_level_scores = tf.sigmoid(this_level_scores)
 113 
 114         if decode_boxes:
 115           this_level_boxes = box_utils.decode_boxes(
 116               this_level_boxes, this_level_anchors)
 117         if clip_boxes:
 118           this_level_boxes = box_utils.clip_boxes(
 119               this_level_boxes, image_shape)
 120 
 121         if rpn_min_size_threshold > 0.0:
 122           this_level_boxes, this_level_scores = box_utils.filter_boxes(
 123               this_level_boxes,
 124               this_level_scores,
 125               image_shape,
 126               rpn_min_size_threshold)
 127 
 128         this_level_pre_nms_top_k = min(num_boxes, rpn_pre_nms_top_k)
 129         this_level_post_nms_top_k = min(num_boxes, rpn_post_nms_top_k)
 130         if rpn_nms_threshold > 0.0:
 131           if use_batched_nms:
 132             this_level_rois, this_level_roi_scores, _, _ = (
 133                 tf.image.combined_non_max_suppression(
 134                     tf.expand_dims(this_level_boxes, axis=2),
 135                     tf.expand_dims(this_level_scores, axis=-1),
 136                     max_output_size_per_class=this_level_pre_nms_top_k,
 137                     max_total_size=this_level_post_nms_top_k,
 138                     iou_threshold=rpn_nms_threshold,
 139                     score_threshold=rpn_score_threshold,
 140                     pad_per_class=False,
 141                     clip_boxes=False))
 142           else:
 143             if rpn_score_threshold > 0.0:
 144               this_level_boxes, this_level_scores = (
 145                   box_utils.filter_boxes_by_scores(
 146                       this_level_boxes, this_level_scores, rpn_score_threshold))
 147             this_level_boxes, this_level_scores = box_utils.top_k_boxes(
 148                 this_level_boxes, this_level_scores, k=this_level_pre_nms_top_k)
 149             this_level_roi_scores, this_level_rois = (
 150                 nms.sorted_non_max_suppression_padded(
 151                     this_level_scores,
 152                     this_level_boxes,
 153                     max_output_size=this_level_post_nms_top_k,
 154                     iou_threshold=rpn_nms_threshold))
 155         else:
 156           this_level_rois, this_level_roi_scores = box_utils.top_k_boxes(
 157               this_level_rois,
 158               this_level_scores,
 159               k=this_level_post_nms_top_k)
 160 
 161         rois.append(this_level_rois)
 162         roi_scores.append(this_level_roi_scores)
 163 
 164     all_rois = tf.concat(rois, axis=1)
 165     all_roi_scores = tf.concat(roi_scores, axis=1)
 166 
 167     with tf.name_scope('top_k_rois'):
 168       _, num_valid_rois = all_roi_scores.get_shape().as_list()
 169       overall_top_k = min(num_valid_rois, rpn_post_nms_top_k)
 170 
 171       selected_rois, selected_roi_scores = box_utils.top_k_boxes(
 172           all_rois, all_roi_scores, k=overall_top_k)
 173 
 174     return selected_rois, selected_roi_scores

 177 class ROIGenerator(object):
 178   """Proposes RoIs for the second stage processing."""
 179 
 180   def __init__(self, params):
 181     self._rpn_pre_nms_top_k = params.rpn_pre_nms_top_k
 182     self._rpn_post_nms_top_k = params.rpn_post_nms_top_k
 183     self._rpn_nms_threshold = params.rpn_nms_threshold
 184     self._rpn_score_threshold = params.rpn_score_threshold
 185     self._rpn_min_size_threshold = params.rpn_min_size_threshold
 186     self._test_rpn_pre_nms_top_k = params.test_rpn_pre_nms_top_k
 187     self._test_rpn_post_nms_top_k = params.test_rpn_post_nms_top_k
 188     self._test_rpn_nms_threshold = params.test_rpn_nms_threshold
 189     self._test_rpn_score_threshold = params.test_rpn_score_threshold
 190     self._test_rpn_min_size_threshold = params.test_rpn_min_size_threshold
 191     self._use_batched_nms = params.use_batched_nms
 192 
 193   def __call__(self, boxes, scores, anchor_boxes, image_shape, is_training):
 194     """Generates RoI proposals.
 195 
 196     Args:
 197       boxes: a dict with keys representing FPN levels and values representing
 198         box tenors of shape [batch_size, feature_h, feature_w, num_anchors * 4].
 199       scores: a dict with keys representing FPN levels and values representing
 200         logit tensors of shape [batch_size, feature_h, feature_w, num_anchors].
 201       anchor_boxes: a dict with keys representing FPN levels and values
 202         representing anchor box tensors of shape
 203         [batch_size, feature_h, feature_w, num_anchors * 4].
 204       image_shape: a tensor of shape [batch_size, 2] where the last dimension
 205         are [height, width] of the scaled image.
 206       is_training: a bool indicating whether it is in training or inference
 207         mode.
 208 
 209     Returns:
 210       proposed_rois: a tensor of shape [batch_size, rpn_post_nms_top_k, 4],
 211         representing the box coordinates of the proposed RoIs w.r.t. the
 212         scaled image.
 213       proposed_roi_scores: a tensor of shape
 214         [batch_size, rpn_post_nms_top_k, 1], representing the scores of the
 215         proposed RoIs.
 216 
 217     """
 218     proposed_rois, proposed_roi_scores = multilevel_propose_rois(
 219         boxes,
 220         scores,
 221         anchor_boxes,
 222         image_shape,
 223         rpn_pre_nms_top_k=(self._rpn_pre_nms_top_k if is_training
 224                            else self._test_rpn_pre_nms_top_k),
 225         rpn_post_nms_top_k=(self._rpn_post_nms_top_k if is_training
 226                             else self._test_rpn_post_nms_top_k),
 227         rpn_nms_threshold=(self._rpn_nms_threshold if is_training
 228                            else self._test_rpn_nms_threshold),
 229         rpn_score_threshold=(self._rpn_score_threshold if is_training
 230                              else self._test_rpn_score_threshold),
 231         rpn_min_size_threshold=(self._rpn_min_size_threshold if is_training
 232                                 else self._test_rpn_min_size_threshold),
 233         decode_boxes=True,
 234         clip_boxes=True,
 235         use_batched_nms=self._use_batched_nms,
 236         apply_sigmoid_to_score=True)
 237     return proposed_rois, proposed_roi_scores
