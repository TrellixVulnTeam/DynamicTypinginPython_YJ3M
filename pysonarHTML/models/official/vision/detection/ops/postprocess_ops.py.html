  28 def generate_detections_factory(params):
  29   """Factory to select function to generate detection."""
  30   if params.use_batched_nms:
  31     func = functools.partial(
  32         _generate_detections_batched,
  33         max_total_size=params.max_total_size,
  34         nms_iou_threshold=params.nms_iou_threshold,
  35         score_threshold=params.score_threshold)
  36   else:
  37     func = functools.partial(
  38         _generate_detections,
  39         max_total_size=params.max_total_size,
  40         nms_iou_threshold=params.nms_iou_threshold,
  41         score_threshold=params.score_threshold,
  42         pre_nms_num_boxes=params.pre_nms_num_boxes)
  43   return func

  46 def _select_top_k_scores(scores_in, pre_nms_num_detections):
  47   """Select top_k scores and indices for each class.
  48 
  49   Args:
  50     scores_in: a Tensor with shape [batch_size, N, num_classes], which stacks
  51       class logit outputs on all feature levels. The N is the number of total
  52       anchors on all levels. The num_classes is the number of classes predicted
  53       by the model.
  54     pre_nms_num_detections: Number of candidates before NMS.
  55 
  56   Returns:
  57     scores and indices: Tensors with shape [batch_size, pre_nms_num_detections,
  58       num_classes].
  59   """
  60   batch_size, num_anchors, num_class = scores_in.get_shape().as_list()
  61   scores_trans = tf.transpose(scores_in, perm=[0, 2, 1])
  62   scores_trans = tf.reshape(scores_trans, [-1, num_anchors])
  63 
  64   top_k_scores, top_k_indices = tf.nn.top_k(
  65       scores_trans, k=pre_nms_num_detections, sorted=True)
  66 
  67   top_k_scores = tf.reshape(top_k_scores,
  68                             [batch_size, num_class, pre_nms_num_detections])
  69   top_k_indices = tf.reshape(top_k_indices,
  70                              [batch_size, num_class, pre_nms_num_detections])
  71 
  72   return tf.transpose(top_k_scores,
  73                       [0, 2, 1]), tf.transpose(top_k_indices, [0, 2, 1])

  76 def _generate_detections(boxes,
  77                          scores,
  78                          max_total_size=100,
  79                          nms_iou_threshold=0.3,
  80                          score_threshold=0.05,
  81                          pre_nms_num_boxes=5000):
  82   """Generate the final detections given the model outputs.
  83 
  84   This uses classes unrolling with while loop based NMS, could be parralled
  85   at batch dimension.
  86 
  87   Args:
  88     boxes: a tensor with shape [batch_size, N, num_classes, 4] or [batch_size,
  89       N, 1, 4], which box predictions on all feature levels. The N is the number
  90       of total anchors on all levels.
  91     scores: a tensor with shape [batch_size, N, num_classes], which stacks class
  92       probability on all feature levels. The N is the number of total anchors on
  93       all levels. The num_classes is the number of classes predicted by the
  94       model. Note that the class_outputs here is the raw score.
  95     max_total_size: a scalar representing maximum number of boxes retained over
  96       all classes.
  97     nms_iou_threshold: a float representing the threshold for deciding whether
  98       boxes overlap too much with respect to IOU.
  99     score_threshold: a float representing the threshold for deciding when to
 100       remove boxes based on score.
 101     pre_nms_num_boxes: an int number of top candidate detections per class
 102       before NMS.
 103 
 104   Returns:
 105     nms_boxes: `float` Tensor of shape [batch_size, max_total_size, 4]
 106       representing top detected boxes in [y1, x1, y2, x2].
 107     nms_scores: `float` Tensor of shape [batch_size, max_total_size]
 108       representing sorted confidence scores for detected boxes. The values are
 109       between [0, 1].
 110     nms_classes: `int` Tensor of shape [batch_size, max_total_size] representing
 111       classes for detected boxes.
 112     valid_detections: `int` Tensor of shape [batch_size] only the top
 113       `valid_detections` boxes are valid detections.
 114   """
 115   with tf.name_scope('generate_detections'):
 116     nmsed_boxes = []
 117     nmsed_classes = []
 118     nmsed_scores = []
 119     valid_detections = []
 120     batch_size, _, num_classes_for_box, _ = boxes.get_shape().as_list()
 121     _, total_anchors, num_classes = scores.get_shape().as_list()
 122     # Selects top pre_nms_num scores and indices before NMS.
 123     scores, indices = _select_top_k_scores(
 124         scores, min(total_anchors, pre_nms_num_boxes))
 125     for i in range(num_classes):
 126       boxes_i = boxes[:, :, min(num_classes_for_box - 1, i), :]
 127       scores_i = scores[:, :, i]
 128       # Obtains pre_nms_num_boxes before running NMS.
 129       boxes_i = tf.gather(boxes_i, indices[:, :, i], batch_dims=1, axis=1)
 130 
 131       # Filter out scores.
 132       boxes_i, scores_i = box_utils.filter_boxes_by_scores(
 133           boxes_i, scores_i, min_score_threshold=score_threshold)
 134 
 135       (nmsed_scores_i, nmsed_boxes_i) = nms.sorted_non_max_suppression_padded(
 136           tf.cast(scores_i, tf.float32),
 137           tf.cast(boxes_i, tf.float32),
 138           max_total_size,
 139           iou_threshold=nms_iou_threshold)
 140       nmsed_classes_i = tf.fill([batch_size, max_total_size], i)
 141       nmsed_boxes.append(nmsed_boxes_i)
 142       nmsed_scores.append(nmsed_scores_i)
 143       nmsed_classes.append(nmsed_classes_i)
 144   nmsed_boxes = tf.concat(nmsed_boxes, axis=1)
 145   nmsed_scores = tf.concat(nmsed_scores, axis=1)
 146   nmsed_classes = tf.concat(nmsed_classes, axis=1)
 147   nmsed_scores, indices = tf.nn.top_k(
 148       nmsed_scores, k=max_total_size, sorted=True)
 149   nmsed_boxes = tf.gather(nmsed_boxes, indices, batch_dims=1, axis=1)
 150   nmsed_classes = tf.gather(nmsed_classes, indices, batch_dims=1)
 151   valid_detections = tf.reduce_sum(
 152       input_tensor=tf.cast(tf.greater(nmsed_scores, -1), tf.int32), axis=1)
 153   return nmsed_boxes, nmsed_scores, nmsed_classes, valid_detections

 156 def _generate_detections_per_image(boxes,
 157                                    scores,
 158                                    max_total_size=100,
 159                                    nms_iou_threshold=0.3,
 160                                    score_threshold=0.05,
 161                                    pre_nms_num_boxes=5000):
 162   """Generate the final detections per image given the model outputs.
 163 
 164   Args:
 165     boxes: a tensor with shape [N, num_classes, 4] or [N, 1, 4], which box
 166       predictions on all feature levels. The N is the number of total anchors on
 167       all levels.
 168     scores: a tensor with shape [N, num_classes], which stacks class probability
 169       on all feature levels. The N is the number of total anchors on all levels.
 170       The num_classes is the number of classes predicted by the model. Note that
 171       the class_outputs here is the raw score.
 172     max_total_size: a scalar representing maximum number of boxes retained over
 173       all classes.
 174     nms_iou_threshold: a float representing the threshold for deciding whether
 175       boxes overlap too much with respect to IOU.
 176     score_threshold: a float representing the threshold for deciding when to
 177       remove boxes based on score.
 178     pre_nms_num_boxes: an int number of top candidate detections per class
 179       before NMS.
 180 
 181   Returns:
 182     nms_boxes: `float` Tensor of shape [max_total_size, 4] representing top
 183       detected boxes in [y1, x1, y2, x2].
 184     nms_scores: `float` Tensor of shape [max_total_size] representing sorted
 185       confidence scores for detected boxes. The values are between [0, 1].
 186     nms_classes: `int` Tensor of shape [max_total_size] representing classes for
 187       detected boxes.
 188     valid_detections: `int` Tensor of shape [1] only the top `valid_detections`
 189       boxes are valid detections.
 190   """
 191   nmsed_boxes = []
 192   nmsed_scores = []
 193   nmsed_classes = []
 194   num_classes_for_box = boxes.get_shape().as_list()[1]
 195   num_classes = scores.get_shape().as_list()[1]
 196   for i in range(num_classes):
 197     boxes_i = boxes[:, min(num_classes_for_box - 1, i)]
 198     scores_i = scores[:, i]
 199 
 200     # Obtains pre_nms_num_boxes before running NMS.
 201     scores_i, indices = tf.nn.top_k(
 202         scores_i, k=tf.minimum(tf.shape(input=scores_i)[-1], pre_nms_num_boxes))
 203     boxes_i = tf.gather(boxes_i, indices)
 204 
 205     (nmsed_indices_i,
 206      nmsed_num_valid_i) = tf.image.non_max_suppression_padded(
 207          tf.cast(boxes_i, tf.float32),
 208          tf.cast(scores_i, tf.float32),
 209          max_total_size,
 210          iou_threshold=nms_iou_threshold,
 211          score_threshold=score_threshold,
 212          pad_to_max_output_size=True,
 213          name='nms_detections_' + str(i))
 214     nmsed_boxes_i = tf.gather(boxes_i, nmsed_indices_i)
 215     nmsed_scores_i = tf.gather(scores_i, nmsed_indices_i)
 216     # Sets scores of invalid boxes to -1.
 217     nmsed_scores_i = tf.where(
 218         tf.less(tf.range(max_total_size), [nmsed_num_valid_i]), nmsed_scores_i,
 219         -tf.ones_like(nmsed_scores_i))
 220     nmsed_classes_i = tf.fill([max_total_size], i)
 221     nmsed_boxes.append(nmsed_boxes_i)
 222     nmsed_scores.append(nmsed_scores_i)
 223     nmsed_classes.append(nmsed_classes_i)
 224 
 225   # Concats results from all classes and sort them.
 226   nmsed_boxes = tf.concat(nmsed_boxes, axis=0)
 227   nmsed_scores = tf.concat(nmsed_scores, axis=0)
 228   nmsed_classes = tf.concat(nmsed_classes, axis=0)
 229   nmsed_scores, indices = tf.nn.top_k(
 230       nmsed_scores, k=max_total_size, sorted=True)
 231   nmsed_boxes = tf.gather(nmsed_boxes, indices)
 232   nmsed_classes = tf.gather(nmsed_classes, indices)
 233   valid_detections = tf.reduce_sum(
 234       input_tensor=tf.cast(tf.greater(nmsed_scores, -1), tf.int32))
 235   return nmsed_boxes, nmsed_scores, nmsed_classes, valid_detections

 238 def _generate_detections_batched(boxes,
 239                                  scores,
 240                                  max_total_size,
 241                                  nms_iou_threshold,
 242                                  score_threshold):
 243   """Generates detected boxes with scores and classes for one-stage detector.
 244 
 245   The function takes output of multi-level ConvNets and anchor boxes and
 246   generates detected boxes. Note that this used batched nms, which is not
 247   supported on TPU currently.
 248 
 249   Args:
 250     boxes: a tensor with shape [batch_size, N, num_classes, 4] or
 251       [batch_size, N, 1, 4], which box predictions on all feature levels. The N
 252       is the number of total anchors on all levels.
 253     scores: a tensor with shape [batch_size, N, num_classes], which
 254       stacks class probability on all feature levels. The N is the number of
 255       total anchors on all levels. The num_classes is the number of classes
 256       predicted by the model. Note that the class_outputs here is the raw score.
 257     max_total_size: a scalar representing maximum number of boxes retained over
 258       all classes.
 259     nms_iou_threshold: a float representing the threshold for deciding whether
 260       boxes overlap too much with respect to IOU.
 261     score_threshold: a float representing the threshold for deciding when to
 262       remove boxes based on score.
 263   Returns:
 264     nms_boxes: `float` Tensor of shape [batch_size, max_total_size, 4]
 265       representing top detected boxes in [y1, x1, y2, x2].
 266     nms_scores: `float` Tensor of shape [batch_size, max_total_size]
 267       representing sorted confidence scores for detected boxes. The values are
 268       between [0, 1].
 269     nms_classes: `int` Tensor of shape [batch_size, max_total_size] representing
 270       classes for detected boxes.
 271     valid_detections: `int` Tensor of shape [batch_size] only the top
 272       `valid_detections` boxes are valid detections.
 273   """
 274   with tf.name_scope('generate_detections'):
 275     # TODO(tsungyi): Removes normalization/denomalization once the
 276     # tf.image.combined_non_max_suppression is coordinate system agnostic.
 277     # Normalizes maximum box cooridinates to 1.
 278     normalizer = tf.reduce_max(boxes)
 279     boxes /= normalizer
 280     (nmsed_boxes, nmsed_scores, nmsed_classes,
 281      valid_detections) = tf.image.combined_non_max_suppression(
 282          boxes,
 283          scores,
 284          max_output_size_per_class=max_total_size,
 285          max_total_size=max_total_size,
 286          iou_threshold=nms_iou_threshold,
 287          score_threshold=score_threshold,
 288          pad_per_class=False,)
 289     # De-normalizes box cooridinates.
 290     nmsed_boxes *= normalizer
 291   nmsed_classes = tf.cast(nmsed_classes, tf.int32)
 292   return nmsed_boxes, nmsed_scores, nmsed_classes, valid_detections

 295 class MultilevelDetectionGenerator(object):
 296   """Generates detected boxes with scores and classes for one-stage detector."""
 297 
 298   def __init__(self, min_level, max_level, params):
 299     self._min_level = min_level
 300     self._max_level = max_level
 301     self._generate_detections = generate_detections_factory(params)
 302 
 303   def __call__(self, box_outputs, class_outputs, anchor_boxes, image_shape):
 304     # Collects outputs from all levels into a list.
 305     boxes = []
 306     scores = []
 307     for i in range(self._min_level, self._max_level + 1):
 308       box_outputs_i_shape = tf.shape(box_outputs[i])
 309       batch_size = box_outputs_i_shape[0]
 310       num_anchors_per_locations = box_outputs_i_shape[-1] // 4
 311       num_classes = tf.shape(class_outputs[i])[-1] // num_anchors_per_locations
 312 
 313       # Applies score transformation and remove the implicit background class.
 314       scores_i = tf.sigmoid(
 315           tf.reshape(class_outputs[i], [batch_size, -1, num_classes]))
 316       scores_i = tf.slice(scores_i, [0, 0, 1], [-1, -1, -1])
 317 
 318       # Box decoding.
 319       # The anchor boxes are shared for all data in a batch.
 320       # One stage detector only supports class agnostic box regression.
 321       anchor_boxes_i = tf.reshape(anchor_boxes[i], [batch_size, -1, 4])
 322       box_outputs_i = tf.reshape(box_outputs[i], [batch_size, -1, 4])
 323       boxes_i = box_utils.decode_boxes(box_outputs_i, anchor_boxes_i)
 324 
 325       # Box clipping.
 326       boxes_i = box_utils.clip_boxes(boxes_i, image_shape)
 327 
 328       boxes.append(boxes_i)
 329       scores.append(scores_i)
 330     boxes = tf.concat(boxes, axis=1)
 331     scores = tf.concat(scores, axis=1)
 332 
 333     nmsed_boxes, nmsed_scores, nmsed_classes, valid_detections = (
 334         self._generate_detections(tf.expand_dims(boxes, axis=2), scores))
 335 
 336     # Adds 1 to offset the background class which has index 0.
 337     nmsed_classes += 1
 338     return nmsed_boxes, nmsed_scores, nmsed_classes, valid_detections

 341 class GenericDetectionGenerator(object):
 342   """Generates the final detected boxes with scores and classes."""
 343 
 344   def __init__(self, params):
 345     self._generate_detections = generate_detections_factory(params)
 346 
 347   def __call__(self, box_outputs, class_outputs, anchor_boxes, image_shape):
 348     """Generate final detections.
 349 
 350     Args:
 351       box_outputs: a tensor of shape of [batch_size, K, num_classes * 4]
 352         representing the class-specific box coordinates relative to anchors.
 353       class_outputs: a tensor of shape of [batch_size, K, num_classes]
 354         representing the class logits before applying score activiation.
 355       anchor_boxes: a tensor of shape of [batch_size, K, 4] representing the
 356         corresponding anchor boxes w.r.t `box_outputs`.
 357       image_shape: a tensor of shape of [batch_size, 2] storing the image height
 358         and width w.r.t. the scaled image, i.e. the same image space as
 359         `box_outputs` and `anchor_boxes`.
 360 
 361     Returns:
 362       nms_boxes: `float` Tensor of shape [batch_size, max_total_size, 4]
 363         representing top detected boxes in [y1, x1, y2, x2].
 364       nms_scores: `float` Tensor of shape [batch_size, max_total_size]
 365         representing sorted confidence scores for detected boxes. The values are
 366         between [0, 1].
 367       nms_classes: `int` Tensor of shape [batch_size, max_total_size]
 368         representing classes for detected boxes.
 369       valid_detections: `int` Tensor of shape [batch_size] only the top
 370         `valid_detections` boxes are valid detections.
 371     """
 372     class_outputs = tf.nn.softmax(class_outputs, axis=-1)
 373 
 374     # Removes the background class.
 375     class_outputs_shape = tf.shape(class_outputs)
 376     batch_size = class_outputs_shape[0]
 377     num_locations = class_outputs_shape[1]
 378     num_classes = class_outputs_shape[-1]
 379     num_detections = num_locations * (num_classes - 1)
 380 
 381     class_outputs = tf.slice(class_outputs, [0, 0, 1], [-1, -1, -1])
 382     box_outputs = tf.reshape(
 383         box_outputs,
 384         tf.stack([batch_size, num_locations, num_classes, 4], axis=-1))
 385     box_outputs = tf.slice(
 386         box_outputs, [0, 0, 1, 0], [-1, -1, -1, -1])
 387     anchor_boxes = tf.tile(
 388         tf.expand_dims(anchor_boxes, axis=2), [1, 1, num_classes - 1, 1])
 389     box_outputs = tf.reshape(
 390         box_outputs,
 391         tf.stack([batch_size, num_detections, 4], axis=-1))
 392     anchor_boxes = tf.reshape(
 393         anchor_boxes,
 394         tf.stack([batch_size, num_detections, 4], axis=-1))
 395 
 396     # Box decoding.
 397     decoded_boxes = box_utils.decode_boxes(
 398         box_outputs, anchor_boxes, weights=[10.0, 10.0, 5.0, 5.0])
 399 
 400     # Box clipping
 401     decoded_boxes = box_utils.clip_boxes(decoded_boxes, image_shape)
 402 
 403     decoded_boxes = tf.reshape(
 404         decoded_boxes,
 405         tf.stack([batch_size, num_locations, num_classes - 1, 4], axis=-1))
 406 
 407     nmsed_boxes, nmsed_scores, nmsed_classes, valid_detections = (
 408         self._generate_detections(decoded_boxes, class_outputs))
 409 
 410     # Adds 1 to offset the background class which has index 0.
 411     nmsed_classes += 1
 412 
 413     return nmsed_boxes, nmsed_scores, nmsed_classes, valid_detections
