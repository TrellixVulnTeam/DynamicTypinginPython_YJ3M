  27 def nearest_upsampling(data, scale):
  28   """Nearest neighbor upsampling implementation.
  29 
  30   Args:
  31     data: A tensor with a shape of [batch, height_in, width_in, channels].
  32     scale: An integer multiple to scale resolution of input data.
  33   Returns:
  34     data_up: A tensor with a shape of
  35       [batch, height_in*scale, width_in*scale, channels]. Same dtype as input
  36       data.
  37   """
  38   with tf.name_scope('nearest_upsampling'):
  39     bs, _, _, c = data.get_shape().as_list()
  40     shape = tf.shape(input=data)
  41     h = shape[1]
  42     w = shape[2]
  43     bs = -1 if bs is None else bs
  44     # Uses reshape to quickly upsample the input.  The nearest pixel is selected
  45     # implicitly via broadcasting.
  46     data = tf.reshape(data, [bs, h, 1, w, 1, c]) * tf.ones(
  47         [1, 1, scale, 1, scale, 1], dtype=data.dtype)
  48     return tf.reshape(data, [bs, h * scale, w * scale, c])

  51 def feature_bilinear_interpolation(features, kernel_y, kernel_x):
  52   """Feature bilinear interpolation.
  53 
  54   The RoIAlign feature f can be computed by bilinear interpolation
  55   of four neighboring feature points f0, f1, f2, and f3.
  56 
  57   f(y, x) = [hy, ly] * [[f00, f01], * [hx, lx]^T
  58                         [f10, f11]]
  59   f(y, x) = (hy*hx)f00 + (hy*lx)f01 + (ly*hx)f10 + (lx*ly)f11
  60   f(y, x) = w00*f00 + w01*f01 + w10*f10 + w11*f11
  61   kernel_y = [hy, ly]
  62   kernel_x = [hx, lx]
  63 
  64   Args:
  65     features: The features are in shape of [batch_size, num_boxes, output_size *
  66       2, output_size * 2, num_filters].
  67     kernel_y: Tensor of size [batch_size, boxes, output_size, 2, 1].
  68     kernel_x: Tensor of size [batch_size, boxes, output_size, 2, 1].
  69 
  70   Returns:
  71     A 5-D tensor representing feature crop of shape
  72     [batch_size, num_boxes, output_size, output_size, num_filters].
  73 
  74   """
  75   (batch_size, num_boxes, output_size, _,
  76    num_filters) = features.get_shape().as_list()
  77   output_size = output_size // 2
  78   kernel_y = tf.reshape(kernel_y, [batch_size, num_boxes, output_size * 2, 1])
  79   kernel_x = tf.reshape(kernel_x, [batch_size, num_boxes, 1, output_size * 2])
  80   # Use implicit broadcast to generate the interpolation kernel. The
  81   # multiplier `4` is for avg pooling.
  82   interpolation_kernel = kernel_y * kernel_x * 4
  83 
  84   # Interpolate the gathered features with computed interpolation kernels.
  85   features *= tf.cast(
  86       tf.expand_dims(interpolation_kernel, axis=-1), dtype=features.dtype)
  87   features = tf.reshape(
  88       features,
  89       [batch_size * num_boxes, output_size * 2, output_size * 2, num_filters])
  90   features = tf.nn.avg_pool(features, [1, 2, 2, 1], [1, 2, 2, 1], 'VALID')
  91   features = tf.reshape(
  92       features, [batch_size, num_boxes, output_size, output_size, num_filters])
  93   return features

  96 def compute_grid_positions(boxes, boundaries, output_size, sample_offset):
  97   """Compute the grid position w.r.t.
  98 
  99   the corresponding feature map.
 100 
 101   Args:
 102     boxes: a 3-D tensor of shape [batch_size, num_boxes, 4] encoding the
 103       information of each box w.r.t. the corresponding feature map.
 104       boxes[:, :, 0:2] are the grid position in (y, x) (float) of the top-left
 105       corner of each box. boxes[:, :, 2:4] are the box sizes in (h, w) (float)
 106         in terms of the number of pixels of the corresponding feature map size.
 107     boundaries: a 3-D tensor of shape [batch_size, num_boxes, 2] representing
 108       the boundary (in (y, x)) of the corresponding feature map for each box.
 109       Any resampled grid points that go beyond the bounary will be clipped.
 110     output_size: a scalar indicating the output crop size.
 111     sample_offset: a float number in [0, 1] indicates the subpixel sample offset
 112       from grid point.
 113 
 114   Returns:
 115     kernel_y: Tensor of size [batch_size, boxes, output_size, 2, 1].
 116     kernel_x: Tensor of size [batch_size, boxes, output_size, 2, 1].
 117     box_grid_y0y1: Tensor of size [batch_size, boxes, output_size, 2]
 118     box_grid_x0x1: Tensor of size [batch_size, boxes, output_size, 2]
 119   """
 120   batch_size, num_boxes, _ = boxes.get_shape().as_list()
 121   box_grid_x = []
 122   box_grid_y = []
 123   for i in range(output_size):
 124     box_grid_x.append(boxes[:, :, 1] +
 125                       (i + sample_offset) * boxes[:, :, 3] / output_size)
 126     box_grid_y.append(boxes[:, :, 0] +
 127                       (i + sample_offset) * boxes[:, :, 2] / output_size)
 128   box_grid_x = tf.stack(box_grid_x, axis=2)
 129   box_grid_y = tf.stack(box_grid_y, axis=2)
 130 
 131   box_grid_y0 = tf.floor(box_grid_y)
 132   box_grid_x0 = tf.floor(box_grid_x)
 133   box_grid_x0 = tf.maximum(0., box_grid_x0)
 134   box_grid_y0 = tf.maximum(0., box_grid_y0)
 135 
 136   box_grid_x0 = tf.minimum(box_grid_x0, tf.expand_dims(boundaries[:, :, 1], -1))
 137   box_grid_x1 = tf.minimum(box_grid_x0 + 1,
 138                            tf.expand_dims(boundaries[:, :, 1], -1))
 139   box_grid_y0 = tf.minimum(box_grid_y0, tf.expand_dims(boundaries[:, :, 0], -1))
 140   box_grid_y1 = tf.minimum(box_grid_y0 + 1,
 141                            tf.expand_dims(boundaries[:, :, 0], -1))
 142 
 143   box_gridx0x1 = tf.stack([box_grid_x0, box_grid_x1], axis=-1)
 144   box_gridy0y1 = tf.stack([box_grid_y0, box_grid_y1], axis=-1)
 145 
 146   # The RoIAlign feature f can be computed by bilinear interpolation of four
 147   # neighboring feature points f0, f1, f2, and f3.
 148   # f(y, x) = [hy, ly] * [[f00, f01], * [hx, lx]^T
 149   #                       [f10, f11]]
 150   # f(y, x) = (hy*hx)f00 + (hy*lx)f01 + (ly*hx)f10 + (lx*ly)f11
 151   # f(y, x) = w00*f00 + w01*f01 + w10*f10 + w11*f11
 152   ly = box_grid_y - box_grid_y0
 153   lx = box_grid_x - box_grid_x0
 154   hy = 1.0 - ly
 155   hx = 1.0 - lx
 156   kernel_y = tf.reshape(
 157       tf.stack([hy, ly], axis=3), [batch_size, num_boxes, output_size, 2, 1])
 158   kernel_x = tf.reshape(
 159       tf.stack([hx, lx], axis=3), [batch_size, num_boxes, output_size, 2, 1])
 160   return kernel_y, kernel_x, box_gridy0y1, box_gridx0x1

 163 def get_grid_one_hot(box_gridy0y1, box_gridx0x1, feature_height, feature_width):
 164   """Get grid_one_hot from indices and feature_size."""
 165   (batch_size, num_boxes, output_size, _) = box_gridx0x1.get_shape().as_list()
 166   y_indices = tf.cast(
 167       tf.reshape(box_gridy0y1, [batch_size, num_boxes, output_size, 2]),
 168       dtype=tf.int32)
 169   x_indices = tf.cast(
 170       tf.reshape(box_gridx0x1, [batch_size, num_boxes, output_size, 2]),
 171       dtype=tf.int32)
 172 
 173   # shape is [batch_size, num_boxes, output_size, 2, height]
 174   grid_y_one_hot = tf.one_hot(tf.cast(y_indices, tf.int32), feature_height)
 175   # shape is [batch_size, num_boxes, output_size, 2, width]
 176   grid_x_one_hot = tf.one_hot(tf.cast(x_indices, tf.int32), feature_width)
 177 
 178   return grid_y_one_hot, grid_x_one_hot

 181 def selective_crop_and_resize(features,
 182                               boxes,
 183                               box_levels,
 184                               boundaries,
 185                               output_size=7,
 186                               sample_offset=0.5,
 187                               use_einsum_gather=False):
 188   """Crop and resize boxes on a set of feature maps.
 189 
 190   Given multiple features maps indexed by different levels, and a set of boxes
 191   where each box is mapped to a certain level, it selectively crops and resizes
 192   boxes from the corresponding feature maps to generate the box features.
 193 
 194   We follow the ROIAlign technique (see https://arxiv.org/pdf/1703.06870.pdf,
 195   figure 3 for reference). Specifically, for each feature map, we select an
 196   (output_size, output_size) set of pixels corresponding to the box location,
 197   and then use bilinear interpolation to select the feature value for each
 198   pixel.
 199 
 200   For performance, we perform the gather and interpolation on all layers as a
 201   single operation. In this op the multi-level features are first stacked and
 202   gathered into [2*output_size, 2*output_size] feature points. Then bilinear
 203   interpolation is performed on the gathered feature points to generate
 204   [output_size, output_size] RoIAlign feature map.
 205 
 206   Here is the step-by-step algorithm:
 207     1. The multi-level features are gathered into a
 208        [batch_size, num_boxes, output_size*2, output_size*2, num_filters]
 209        Tensor. The Tensor contains four neighboring feature points for each
 210        vertice in the output grid.
 211     2. Compute the interpolation kernel of shape
 212        [batch_size, num_boxes, output_size*2, output_size*2]. The last 2 axis
 213        can be seen as stacking 2x2 interpolation kernels for all vertices in the
 214        output grid.
 215     3. Element-wise multiply the gathered features and interpolation kernel.
 216        Then apply 2x2 average pooling to reduce spatial dimension to
 217        output_size.
 218 
 219   Args:
 220     features: a 5-D tensor of shape [batch_size, num_levels, max_height,
 221       max_width, num_filters] where cropping and resizing are based.
 222     boxes: a 3-D tensor of shape [batch_size, num_boxes, 4] encoding the
 223       information of each box w.r.t. the corresponding feature map.
 224       boxes[:, :, 0:2] are the grid position in (y, x) (float) of the top-left
 225       corner of each box. boxes[:, :, 2:4] are the box sizes in (h, w) (float)
 226         in terms of the number of pixels of the corresponding feature map size.
 227     box_levels: a 3-D tensor of shape [batch_size, num_boxes, 1] representing
 228       the 0-based corresponding feature level index of each box.
 229     boundaries: a 3-D tensor of shape [batch_size, num_boxes, 2] representing
 230       the boundary (in (y, x)) of the corresponding feature map for each box.
 231       Any resampled grid points that go beyond the bounary will be clipped.
 232     output_size: a scalar indicating the output crop size.
 233     sample_offset: a float number in [0, 1] indicates the subpixel sample offset
 234       from grid point.
 235     use_einsum_gather: use einsum to replace gather or not. Replacing einsum
 236       with gather can improve performance when feature size is not large, einsum
 237       is friendly with model partition as well. Gather's performance is better
 238       when feature size is very large and there are multiple box levels.
 239 
 240   Returns:
 241     features_per_box: a 5-D tensor of shape
 242       [batch_size, num_boxes, output_size, output_size, num_filters]
 243       representing the cropped features.
 244   """
 245   (batch_size, num_levels, max_feature_height, max_feature_width,
 246    num_filters) = features.get_shape().as_list()
 247   _, num_boxes, _ = boxes.get_shape().as_list()
 248 
 249   kernel_y, kernel_x, box_gridy0y1, box_gridx0x1 = compute_grid_positions(
 250       boxes, boundaries, output_size, sample_offset)
 251   x_indices = tf.cast(
 252       tf.reshape(box_gridx0x1, [batch_size, num_boxes, output_size * 2]),
 253       dtype=tf.int32)
 254   y_indices = tf.cast(
 255       tf.reshape(box_gridy0y1, [batch_size, num_boxes, output_size * 2]),
 256       dtype=tf.int32)
 257 
 258   if use_einsum_gather:
 259     # Blinear interpolation is done during the last two gathers:
 260     #        f(y, x) = [hy, ly] * [[f00, f01], * [hx, lx]^T
 261     #                              [f10, f11]]
 262     #        [[f00, f01],
 263     #         [f10, f11]] = tf.einsum(tf.einsum(features, y_one_hot), x_one_hot)
 264     #       where [hy, ly] and [hx, lx] are the bilinear interpolation kernel.
 265 
 266     # shape is [batch_size, boxes, output_size, 2, 1]
 267     grid_y_one_hot, grid_x_one_hot = get_grid_one_hot(box_gridy0y1,
 268                                                       box_gridx0x1,
 269                                                       max_feature_height,
 270                                                       max_feature_width)
 271 
 272     # shape is [batch_size, num_boxes, output_size, height]
 273     grid_y_weight = tf.reduce_sum(
 274         tf.multiply(grid_y_one_hot, kernel_y), axis=-2)
 275     # shape is [batch_size, num_boxes, output_size, width]
 276     grid_x_weight = tf.reduce_sum(
 277         tf.multiply(grid_x_one_hot, kernel_x), axis=-2)
 278 
 279     # Gather for y_axis.
 280     # shape is [batch_size, num_boxes, output_size, width, features]
 281     features_per_box = tf.einsum('bmhwf,bmoh->bmowf', features,
 282                                  tf.cast(grid_y_weight, features.dtype))
 283     # Gather for x_axis.
 284     # shape is [batch_size, num_boxes, output_size, output_size, features]
 285     features_per_box = tf.einsum('bmhwf,bmow->bmhof', features_per_box,
 286                                  tf.cast(grid_x_weight, features.dtype))
 287   else:
 288     height_dim_offset = max_feature_width
 289     level_dim_offset = max_feature_height * height_dim_offset
 290     batch_dim_offset = num_levels * level_dim_offset
 291 
 292     batch_size_offset = tf.tile(
 293         tf.reshape(
 294             tf.range(batch_size) * batch_dim_offset, [batch_size, 1, 1, 1]),
 295         [1, num_boxes, output_size * 2, output_size * 2])
 296     box_levels_offset = tf.tile(
 297         tf.reshape(box_levels * level_dim_offset,
 298                    [batch_size, num_boxes, 1, 1]),
 299         [1, 1, output_size * 2, output_size * 2])
 300     y_indices_offset = tf.tile(
 301         tf.reshape(y_indices * height_dim_offset,
 302                    [batch_size, num_boxes, output_size * 2, 1]),
 303         [1, 1, 1, output_size * 2])
 304     x_indices_offset = tf.tile(
 305         tf.reshape(x_indices, [batch_size, num_boxes, 1, output_size * 2]),
 306         [1, 1, output_size * 2, 1])
 307 
 308     indices = tf.reshape(
 309         batch_size_offset + box_levels_offset + y_indices_offset +
 310         x_indices_offset, [-1])
 311 
 312     features = tf.reshape(features, [-1, num_filters])
 313     # TODO(wangtao): replace tf.gather with tf.gather_nd and try to get similar
 314     # performance.
 315     features_per_box = tf.reshape(
 316         tf.gather(features, indices),
 317         [batch_size, num_boxes, output_size * 2, output_size * 2, num_filters])
 318     features_per_box = feature_bilinear_interpolation(features_per_box,
 319                                                       kernel_y, kernel_x)
 320 
 321   return features_per_box

 324 def multilevel_crop_and_resize(features, boxes, output_size=7):
 325   """Crop and resize on multilevel feature pyramid.
 326 
 327   Generate the (output_size, output_size) set of pixels for each input box
 328   by first locating the box into the correct feature level, and then cropping
 329   and resizing it using the correspoding feature map of that level.
 330 
 331   Args:
 332     features: A dictionary with key as pyramid level and value as features. The
 333       features are in shape of [batch_size, height_l, width_l, num_filters].
 334     boxes: A 3-D Tensor of shape [batch_size, num_boxes, 4]. Each row represents
 335       a box with [y1, x1, y2, x2] in un-normalized coordinates.
 336     output_size: A scalar to indicate the output crop size.
 337 
 338   Returns:
 339     A 5-D tensor representing feature crop of shape
 340     [batch_size, num_boxes, output_size, output_size, num_filters].
 341   """
 342 
 343   with tf.name_scope('multilevel_crop_and_resize'):
 344     levels = list(features.keys())
 345     min_level = min(levels)
 346     max_level = max(levels)
 347     batch_size, max_feature_height, max_feature_width, num_filters = (
 348         features[min_level].get_shape().as_list())
 349     _, num_boxes, _ = boxes.get_shape().as_list()
 350 
 351     # Stack feature pyramid into a features_all of shape
 352     # [batch_size, levels, height, width, num_filters].
 353     features_all = []
 354     feature_heights = []
 355     feature_widths = []
 356     for level in range(min_level, max_level + 1):
 357       shape = features[level].get_shape().as_list()
 358       feature_heights.append(shape[1])
 359       feature_widths.append(shape[2])
 360       # Concat tensor of [batch_size, height_l * width_l, num_filters] for each
 361       # levels.
 362       features_all.append(
 363           tf.reshape(features[level], [batch_size, -1, num_filters]))
 364       features_r2 = tf.reshape(tf.concat(features_all, 1), [-1, num_filters])
 365 
 366     # Calculate height_l * width_l for each level.
 367     level_dim_sizes = [
 368         feature_widths[i] * feature_heights[i]
 369         for i in range(len(feature_widths))
 370     ]
 371     # level_dim_offsets is accumulated sum of level_dim_size.
 372     level_dim_offsets = [0]
 373     for i in range(len(feature_widths) - 1):
 374       level_dim_offsets.append(level_dim_offsets[i] + level_dim_sizes[i])
 375     batch_dim_size = level_dim_offsets[-1] + level_dim_sizes[-1]
 376     level_dim_offsets = tf.constant(level_dim_offsets, tf.int32)
 377     height_dim_sizes = tf.constant(feature_widths, tf.int32)
 378 
 379     # Assigns boxes to the right level.
 380     box_width = boxes[:, :, 3] - boxes[:, :, 1]
 381     box_height = boxes[:, :, 2] - boxes[:, :, 0]
 382     areas_sqrt = tf.sqrt(box_height * box_width)
 383     levels = tf.cast(
 384         tf.math.floordiv(
 385             tf.math.log(tf.divide(areas_sqrt, 224.0)), tf.math.log(2.0)) +
 386         4.0,
 387         dtype=tf.int32)
 388     # Maps levels between [min_level, max_level].
 389     levels = tf.minimum(max_level, tf.maximum(levels, min_level))
 390 
 391     # Projects box location and sizes to corresponding feature levels.
 392     scale_to_level = tf.cast(
 393         tf.pow(tf.constant(2.0), tf.cast(levels, tf.float32)),
 394         dtype=boxes.dtype)
 395     boxes /= tf.expand_dims(scale_to_level, axis=2)
 396     box_width /= scale_to_level
 397     box_height /= scale_to_level
 398     boxes = tf.concat([boxes[:, :, 0:2],
 399                        tf.expand_dims(box_height, -1),
 400                        tf.expand_dims(box_width, -1)], axis=-1)
 401 
 402     # Maps levels to [0, max_level-min_level].
 403     levels -= min_level
 404     level_strides = tf.pow([[2.0]], tf.cast(levels, tf.float32))
 405     boundary = tf.cast(
 406         tf.concat([
 407             tf.expand_dims(
 408                 [[tf.cast(max_feature_height, tf.float32)]] / level_strides - 1,
 409                 axis=-1),
 410             tf.expand_dims(
 411                 [[tf.cast(max_feature_width, tf.float32)]] / level_strides - 1,
 412                 axis=-1),
 413         ],
 414                   axis=-1), boxes.dtype)
 415 
 416     # Compute grid positions.
 417     kernel_y, kernel_x, box_gridy0y1, box_gridx0x1 = compute_grid_positions(
 418         boxes, boundary, output_size, sample_offset=0.5)
 419 
 420     x_indices = tf.cast(
 421         tf.reshape(box_gridx0x1, [batch_size, num_boxes, output_size * 2]),
 422         dtype=tf.int32)
 423     y_indices = tf.cast(
 424         tf.reshape(box_gridy0y1, [batch_size, num_boxes, output_size * 2]),
 425         dtype=tf.int32)
 426 
 427     batch_size_offset = tf.tile(
 428         tf.reshape(
 429             tf.range(batch_size) * batch_dim_size, [batch_size, 1, 1, 1]),
 430         [1, num_boxes, output_size * 2, output_size * 2])
 431     # Get level offset for each box. Each box belongs to one level.
 432     levels_offset = tf.tile(
 433         tf.reshape(
 434             tf.gather(level_dim_offsets, levels),
 435             [batch_size, num_boxes, 1, 1]),
 436         [1, 1, output_size * 2, output_size * 2])
 437     y_indices_offset = tf.tile(
 438         tf.reshape(
 439             y_indices * tf.expand_dims(tf.gather(height_dim_sizes, levels), -1),
 440             [batch_size, num_boxes, output_size * 2, 1]),
 441         [1, 1, 1, output_size * 2])
 442     x_indices_offset = tf.tile(
 443         tf.reshape(x_indices, [batch_size, num_boxes, 1, output_size * 2]),
 444         [1, 1, output_size * 2, 1])
 445     indices = tf.reshape(
 446         batch_size_offset + levels_offset + y_indices_offset + x_indices_offset,
 447         [-1])
 448 
 449     # TODO(wangtao): replace tf.gather with tf.gather_nd and try to get similar
 450     # performance.
 451     features_per_box = tf.reshape(
 452         tf.gather(features_r2, indices),
 453         [batch_size, num_boxes, output_size * 2, output_size * 2, num_filters])
 454 
 455     # Bilinear interpolation.
 456     features_per_box = feature_bilinear_interpolation(features_per_box,
 457                                                       kernel_y, kernel_x)
 458     return features_per_box

 461 def single_level_feature_crop(features, level_boxes, detection_prior_levels,
 462                               min_mask_level, mask_crop_size):
 463   """Crop the FPN features at the appropriate levels for each detection.
 464 
 465 
 466   Args:
 467     features: a float tensor of shape [batch_size, num_levels,
 468       max_feature_size, max_feature_size, num_downsample_channels].
 469     level_boxes: a float Tensor of the level boxes to crop from.
 470         [batch_size, num_instances, 4].
 471     detection_prior_levels: an int Tensor of instance assigned level of shape
 472         [batch_size, num_instances].
 473     min_mask_level: minimum FPN level to crop mask feature from.
 474     mask_crop_size: an int of mask crop size.
 475 
 476   Returns:
 477     crop_features: a float Tensor of shape [batch_size * num_instances,
 478         mask_crop_size, mask_crop_size, num_downsample_channels]. This is the
 479         instance feature crop.
 480   """
 481   (batch_size, num_levels, max_feature_size,
 482    _, num_downsample_channels) = features.get_shape().as_list()
 483   _, num_of_instances, _ = level_boxes.get_shape().as_list()
 484   level_boxes = tf.cast(level_boxes, tf.int32)
 485   assert num_of_instances == detection_prior_levels.get_shape().as_list()[1]
 486 
 487   x_start_indices = level_boxes[:, :, 1]
 488   y_start_indices = level_boxes[:, :, 0]
 489   # generate the full indices (not just the starting index)
 490   x_idx_list = []
 491   y_idx_list = []
 492   for i in range(mask_crop_size):
 493     x_idx_list.append(x_start_indices + i)
 494     y_idx_list.append(y_start_indices + i)
 495 
 496   x_indices = tf.stack(x_idx_list, axis=2)
 497   y_indices = tf.stack(y_idx_list, axis=2)
 498   levels = detection_prior_levels - min_mask_level
 499   height_dim_size = max_feature_size
 500   level_dim_size = max_feature_size * height_dim_size
 501   batch_dim_size = num_levels * level_dim_size
 502   # TODO(weicheng) change this to gather_nd for better readability.
 503   indices = tf.reshape(
 504       tf.tile(
 505           tf.reshape(
 506               tf.range(batch_size) * batch_dim_size,
 507               [batch_size, 1, 1, 1]),
 508           [1, num_of_instances,
 509            mask_crop_size, mask_crop_size]) +
 510       tf.tile(
 511           tf.reshape(levels * level_dim_size,
 512                      [batch_size, num_of_instances, 1, 1]),
 513           [1, 1, mask_crop_size, mask_crop_size]) +
 514       tf.tile(
 515           tf.reshape(y_indices * height_dim_size,
 516                      [batch_size, num_of_instances,
 517                       mask_crop_size, 1]),
 518           [1, 1, 1, mask_crop_size]) +
 519       tf.tile(
 520           tf.reshape(x_indices,
 521                      [batch_size, num_of_instances,
 522                       1, mask_crop_size]),
 523           [1, 1, mask_crop_size, 1]), [-1])
 524 
 525   features_r2 = tf.reshape(features,
 526                            [-1, num_downsample_channels])
 527   crop_features = tf.reshape(
 528       tf.gather(features_r2, indices),
 529       [batch_size * num_of_instances,
 530        mask_crop_size, mask_crop_size,
 531        num_downsample_channels])
 532 
 533   return crop_features
 534

 536 def crop_mask_in_target_box(masks,
 537                             boxes,
 538                             target_boxes,
 539                             output_size,
 540                             sample_offset=0,
 541                             use_einsum=True):
 542   """Crop masks in target boxes.
 543 
 544   Args:
 545     masks: A tensor with a shape of [batch_size, num_masks, height, width].
 546     boxes: a float tensor representing box cooridnates that tightly enclose
 547       masks with a shape of [batch_size, num_masks, 4] in un-normalized
 548       coordinates. A box is represented by [ymin, xmin, ymax, xmax].
 549     target_boxes: a float tensor representing target box cooridnates for
 550       masks with a shape of [batch_size, num_masks, 4] in un-normalized
 551       coordinates. A box is represented by [ymin, xmin, ymax, xmax].
 552     output_size: A scalar to indicate the output crop size. It currently only
 553       supports to output a square shape outputs.
 554     sample_offset: a float number in [0, 1] indicates the subpixel sample offset
 555       from grid point.
 556     use_einsum: Use einsum to replace gather in selective_crop_and_resize.
 557 
 558   Returns:
 559     A 4-D tensor representing feature crop of shape
 560     [batch_size, num_boxes, output_size, output_size].
 561   """
 562   with tf.name_scope('crop_mask_in_target_box'):
 563     batch_size, num_masks, height, width = masks.get_shape().as_list()
 564     masks = tf.reshape(masks, [batch_size*num_masks, height, width, 1])
 565     # Pad zeros on the boundary of masks.
 566     masks = tf.image.pad_to_bounding_box(masks, 2, 2, height + 4, width + 4)
 567     masks = tf.reshape(masks, [batch_size, num_masks, height+4, width+4, 1])
 568 
 569     # Projects target box locations and sizes to corresponding cropped
 570     # mask coordinates.
 571     gt_y_min, gt_x_min, gt_y_max, gt_x_max = tf.split(
 572         value=boxes, num_or_size_splits=4, axis=2)
 573     bb_y_min, bb_x_min, bb_y_max, bb_x_max = tf.split(
 574         value=target_boxes, num_or_size_splits=4, axis=2)
 575     y_transform = (bb_y_min - gt_y_min) * height / (
 576         gt_y_max - gt_y_min + _EPSILON) + 2
 577     x_transform = (bb_x_min - gt_x_min) * height / (
 578         gt_x_max - gt_x_min + _EPSILON) + 2
 579     h_transform = (bb_y_max - bb_y_min) * width / (
 580         gt_y_max - gt_y_min + _EPSILON)
 581     w_transform = (bb_x_max - bb_x_min) * width / (
 582         gt_x_max - gt_x_min + _EPSILON)
 583 
 584     boundaries = tf.concat([
 585         tf.cast(
 586             tf.ones_like(y_transform) * ((height + 4) - 1), dtype=tf.float32),
 587         tf.cast(
 588             tf.ones_like(x_transform) * ((width + 4) - 1), dtype=tf.float32)
 589     ],
 590                            axis=-1)
 591 
 592     # Reshape tensors to have the right shape for selective_crop_and_resize.
 593     trasnformed_boxes = tf.concat(
 594         [y_transform, x_transform, h_transform, w_transform], -1)
 595     levels = tf.tile(tf.reshape(tf.range(num_masks), [1, num_masks]),
 596                      [batch_size, 1])
 597 
 598     cropped_masks = selective_crop_and_resize(
 599         masks,
 600         trasnformed_boxes,
 601         levels,
 602         boundaries,
 603         output_size,
 604         sample_offset=sample_offset,
 605         use_einsum_gather=use_einsum)
 606     cropped_masks = tf.squeeze(cropped_masks, axis=-1)
 607 
 608   return cropped_masks
