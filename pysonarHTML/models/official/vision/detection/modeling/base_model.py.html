  30 def _make_filter_trainable_variables_fn(frozen_variable_prefix):
  31   """Creates a function for filtering trainable varialbes."""
  32 
  33   def _filter_trainable_variables(variables):
  34     """Filters trainable varialbes.
  35 
  36     Args:
  37       variables: a list of tf.Variable to be filtered.
  38 
  39     Returns:
  40       filtered_variables: a list of tf.Variable filtered out the frozen ones.
  41     """
  42     # frozen_variable_prefix: a regex string specifing the prefix pattern of
  43     # the frozen variables' names.
  44     filtered_variables = [
  45         v for v in variables
  46         if not frozen_variable_prefix or
  47         not re.match(frozen_variable_prefix, v.name)
  48     ]
  49     return filtered_variables
  50 
  51   return _filter_trainable_variables

  54 class Model(object):
  55   """Base class for model function."""
  56 
  57   __metaclass__ = abc.ABCMeta
  58 
  59   def __init__(self, params):
  60     self._use_bfloat16 = params.architecture.use_bfloat16
  61 
  62     if params.architecture.use_bfloat16:
  63       policy = tf.compat.v2.keras.mixed_precision.experimental.Policy(
  64           'mixed_bfloat16')
  65       tf.compat.v2.keras.mixed_precision.experimental.set_policy(policy)
  66 
  67     # Optimization.
  68     self._optimizer_fn = optimizers.OptimizerFactory(params.train.optimizer)
  69     self._learning_rate = learning_rates.learning_rate_generator(
  70         params.train.total_steps, params.train.learning_rate)
  71 
  72     self._frozen_variable_prefix = params.train.frozen_variable_prefix
  73     self._regularization_var_regex = params.train.regularization_variable_regex
  74     self._l2_weight_decay = params.train.l2_weight_decay
  75 
  76     # Checkpoint restoration.
  77     self._checkpoint = params.train.checkpoint.as_dict()
  78 
  79     # Summary.
  80     self._enable_summary = params.enable_summary
  81     self._model_dir = params.model_dir

 102   def model_outputs(self, inputs, mode):
 103     """Build the model outputs."""
 104     return self.build_outputs(inputs, mode)
 105 
 106   def build_optimizer(self):
 107     """Returns train_op to optimize total loss."""
 108     # Sets up the optimizer.
 109     return self._optimizer_fn(self._learning_rate)
 110 
 111   def make_filter_trainable_variables_fn(self):
 112     """Creates a function for filtering trainable varialbes."""
 113     return _make_filter_trainable_variables_fn(self._frozen_variable_prefix)
 114 
 115   def weight_decay_loss(self, trainable_variables):
 116     reg_variables = [
 117         v for v in trainable_variables
 118         if self._regularization_var_regex is None
 119         or re.match(self._regularization_var_regex, v.name)
 120     ]
 121 
 122     return self._l2_weight_decay * tf.add_n(
 123         [tf.nn.l2_loss(v) for v in reg_variables])
 124 
 125   def make_restore_checkpoint_fn(self):
 126     """Returns scaffold function to restore parameters from v1 checkpoint."""
 127     if 'skip_checkpoint_variables' in self._checkpoint:
 128       skip_regex = self._checkpoint['skip_checkpoint_variables']
 129     else:
 130       skip_regex = None
 131     return checkpoint_utils.make_restore_checkpoint_fn(
 132         self._checkpoint['path'],
 133         prefix=self._checkpoint['prefix'],
 134         skip_regex=skip_regex)
