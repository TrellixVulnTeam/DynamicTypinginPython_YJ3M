  34 class ShapeMaskModel(base_model.Model):
  35   """ShapeMask model function."""
  36 
  37   def __init__(self, params):
  38     super(ShapeMaskModel, self).__init__(params)
  39 
  40     self._params = params
  41     self._keras_model = None
  42 
  43     # Architecture generators.
  44     self._backbone_fn = factory.backbone_generator(params)
  45     self._fpn_fn = factory.multilevel_features_generator(params)
  46     self._retinanet_head_fn = factory.retinanet_head_generator(params)
  47     self._shape_prior_head_fn = factory.shapeprior_head_generator(params)
  48     self._coarse_mask_fn = factory.coarsemask_head_generator(params)
  49     self._fine_mask_fn = factory.finemask_head_generator(params)
  50 
  51     # Loss functions.
  52     self._cls_loss_fn = losses.RetinanetClassLoss(
  53         params.retinanet_loss, params.architecture.num_classes)
  54     self._box_loss_fn = losses.RetinanetBoxLoss(params.retinanet_loss)
  55     self._box_loss_weight = params.retinanet_loss.box_loss_weight
  56 
  57     # Mask loss function.
  58     self._shapemask_prior_loss_fn = losses.ShapemaskMseLoss()
  59     self._shapemask_loss_fn = losses.ShapemaskLoss()
  60     self._shape_prior_loss_weight = (
  61         params.shapemask_loss.shape_prior_loss_weight)
  62     self._coarse_mask_loss_weight = (
  63         params.shapemask_loss.coarse_mask_loss_weight)
  64     self._fine_mask_loss_weight = (
  65         params.shapemask_loss.fine_mask_loss_weight)
  66 
  67     # Predict function.
  68     self._generate_detections_fn = postprocess_ops.MultilevelDetectionGenerator(
  69         params.architecture.min_level,
  70         params.architecture.max_level,
  71         params.postprocess)
  72 
  73   def build_outputs(self, inputs, mode):
  74     is_training = mode == mode_keys.TRAIN
  75     images = inputs['image']
  76 
  77     if 'anchor_boxes' in inputs:
  78       anchor_boxes = inputs['anchor_boxes']
  79     else:
  80       anchor_boxes = anchor.Anchor(
  81           self._params.architecture.min_level,
  82           self._params.architecture.max_level,
  83           self._params.anchor.num_scales,
  84           self._params.anchor.aspect_ratios,
  85           self._params.anchor.anchor_size,
  86           images.get_shape().as_list()[1:3]).multilevel_boxes
  87 
  88       batch_size = tf.shape(images)[0]
  89       for level in anchor_boxes:
  90         anchor_boxes[level] = tf.tile(
  91             tf.expand_dims(anchor_boxes[level], 0), [batch_size, 1, 1, 1])
  92 
  93     backbone_features = self._backbone_fn(images, is_training=is_training)
  94     fpn_features = self._fpn_fn(backbone_features, is_training=is_training)
  95     cls_outputs, box_outputs = self._retinanet_head_fn(
  96         fpn_features, is_training=is_training)
  97 
  98     valid_boxes, valid_scores, valid_classes, valid_detections = (
  99         self._generate_detections_fn(box_outputs, cls_outputs,
 100                                      anchor_boxes,
 101                                      inputs['image_info'][:, 1:2, :]))
 102 
 103     image_size = images.get_shape().as_list()[1:3]
 104     valid_outer_boxes = box_utils.compute_outer_boxes(
 105         tf.reshape(valid_boxes, [-1, 4]),
 106         image_size,
 107         scale=self._params.shapemask_parser.outer_box_scale)
 108     valid_outer_boxes = tf.reshape(valid_outer_boxes, tf.shape(valid_boxes))
 109 
 110     # Wrapping if else code paths into a layer to make the checkpoint loadable
 111     # in prediction mode.
 112     class SampledBoxesLayer(tf.keras.layers.Layer):
 113       """ShapeMask model function."""
 114 
 115       def call(self, inputs, val_boxes, val_classes, val_outer_boxes, training):
 116         if training:
 117           boxes = inputs['mask_boxes']
 118           outer_boxes = inputs['mask_outer_boxes']
 119           classes = inputs['mask_classes']
 120         else:
 121           boxes = val_boxes
 122           classes = val_classes
 123           outer_boxes = val_outer_boxes
 124         return boxes, classes, outer_boxes
 125 
 126     boxes, classes, outer_boxes = SampledBoxesLayer()(
 127         inputs, valid_boxes, valid_classes,
 128         valid_outer_boxes, training=is_training)
 129 
 130     instance_features, prior_masks = self._shape_prior_head_fn(fpn_features,
 131                                                                boxes,
 132                                                                outer_boxes,
 133                                                                classes,
 134                                                                is_training)
 135     coarse_mask_logits = self._coarse_mask_fn(instance_features,
 136                                               prior_masks,
 137                                               classes,
 138                                               is_training)
 139     fine_mask_logits = self._fine_mask_fn(instance_features,
 140                                           coarse_mask_logits,
 141                                           classes,
 142                                           is_training)
 143 
 144     model_outputs = {
 145         'cls_outputs': cls_outputs,
 146         'box_outputs': box_outputs,
 147         'fine_mask_logits': fine_mask_logits,
 148         'coarse_mask_logits': coarse_mask_logits,
 149         'prior_masks': prior_masks,
 150     }
 151 
 152     if not is_training:
 153       model_outputs.update({
 154           'num_detections': valid_detections,
 155           'detection_boxes': valid_boxes,
 156           'detection_outer_boxes': valid_outer_boxes,
 157           'detection_masks': fine_mask_logits,
 158           'detection_classes': valid_classes,
 159           'detection_scores': valid_scores,
 160       })
 161 
 162     return model_outputs
 163 
 164   def build_loss_fn(self):
 165     if self._keras_model is None:
 166       raise ValueError('build_loss_fn() must be called after build_model().')
 167 
 168     filter_fn = self.make_filter_trainable_variables_fn()
 169     trainable_variables = filter_fn(self._keras_model.trainable_variables)
 170 
 171     def _total_loss_fn(labels, outputs):
 172       cls_loss = self._cls_loss_fn(outputs['cls_outputs'],
 173                                    labels['cls_targets'],
 174                                    labels['num_positives'])
 175       box_loss = self._box_loss_fn(outputs['box_outputs'],
 176                                    labels['box_targets'],
 177                                    labels['num_positives'])
 178 
 179       # Adds Shapemask model losses.
 180       shape_prior_loss = self._shapemask_prior_loss_fn(
 181           outputs['prior_masks'],
 182           labels['mask_targets'],
 183           labels['mask_is_valid'])
 184       coarse_mask_loss = self._shapemask_loss_fn(
 185           outputs['coarse_mask_logits'],
 186           labels['mask_targets'],
 187           labels['mask_is_valid'])
 188       fine_mask_loss = self._shapemask_loss_fn(
 189           outputs['fine_mask_logits'],
 190           labels['fine_mask_targets'],
 191           labels['mask_is_valid'])
 192 
 193       model_loss = (
 194           cls_loss + self._box_loss_weight * box_loss +
 195           shape_prior_loss * self._shape_prior_loss_weight +
 196           coarse_mask_loss * self._coarse_mask_loss_weight +
 197           fine_mask_loss * self._fine_mask_loss_weight)
 198 
 199       l2_regularization_loss = self.weight_decay_loss(trainable_variables)
 200       total_loss = model_loss + l2_regularization_loss
 201 
 202       shapemask_losses = {
 203           'total_loss': total_loss,
 204           'loss': total_loss,
 205           'retinanet_cls_loss': cls_loss,
 206           'l2_regularization_loss': l2_regularization_loss,
 207           'retinanet_box_loss': box_loss,
 208           'shapemask_prior_loss': shape_prior_loss,
 209           'shapemask_coarse_mask_loss': coarse_mask_loss,
 210           'shapemask_fine_mask_loss': fine_mask_loss,
 211           'model_loss': model_loss,
 212       }
 213       return shapemask_losses
 214 
 215     return _total_loss_fn
 216 
 217   def build_input_layers(self, params, mode):
 218     is_training = mode == mode_keys.TRAIN
 219     input_shape = (
 220         params.shapemask_parser.output_size +
 221         [params.shapemask_parser.num_channels])
 222     if is_training:
 223       batch_size = params.train.batch_size
 224       input_layer = {
 225           'image': tf.keras.layers.Input(
 226               shape=input_shape,
 227               batch_size=batch_size,
 228               name='image',
 229               dtype=tf.bfloat16 if self._use_bfloat16 else tf.float32),
 230           'image_info': tf.keras.layers.Input(
 231               shape=[4, 2],
 232               batch_size=batch_size,
 233               name='image_info'),
 234           'mask_classes': tf.keras.layers.Input(
 235               shape=[params.shapemask_parser.num_sampled_masks],
 236               batch_size=batch_size,
 237               name='mask_classes',
 238               dtype=tf.int64),
 239           'mask_outer_boxes': tf.keras.layers.Input(
 240               shape=[params.shapemask_parser.num_sampled_masks, 4],
 241               batch_size=batch_size,
 242               name='mask_outer_boxes',
 243               dtype=tf.float32),
 244           'mask_boxes': tf.keras.layers.Input(
 245               shape=[params.shapemask_parser.num_sampled_masks, 4],
 246               batch_size=batch_size,
 247               name='mask_boxes',
 248               dtype=tf.float32),
 249       }
 250     else:
 251       batch_size = params.eval.batch_size
 252       input_layer = {
 253           'image': tf.keras.layers.Input(
 254               shape=input_shape,
 255               batch_size=batch_size,
 256               name='image',
 257               dtype=tf.bfloat16 if self._use_bfloat16 else tf.float32),
 258           'image_info': tf.keras.layers.Input(
 259               shape=[4, 2],
 260               batch_size=batch_size,
 261               name='image_info'),
 262       }
 263     return input_layer
 264 
 265   def build_model(self, params, mode):
 266     if self._keras_model is None:
 267       input_layers = self.build_input_layers(self._params, mode)
 268       with backend.get_graph().as_default():
 269         outputs = self.model_outputs(input_layers, mode)
 270 
 271         model = tf.keras.models.Model(
 272             inputs=input_layers, outputs=outputs, name='shapemask')
 273         assert model is not None, 'Fail to build tf.keras.Model.'
 274         model.optimizer = self.build_optimizer()
 275         self._keras_model = model
 276 
 277     return self._keras_model

 279   def post_processing(self, labels, outputs):
 280     required_output_fields = ['num_detections', 'detection_boxes',
 281                               'detection_classes', 'detection_masks',
 282                               'detection_scores']
 283 
 284     for field in required_output_fields:
 285       if field not in outputs:
 286         raise ValueError(
 287             '"{}" is missing in outputs, requried {} found {}'.format(
 288                 field, required_output_fields, outputs.keys()))
 289 
 290     required_label_fields = ['image_info']
 291     for field in required_label_fields:
 292       if field not in labels:
 293         raise ValueError(
 294             '"{}" is missing in labels, requried {} found {}'.format(
 295                 field, required_label_fields, labels.keys()))
 296 
 297     predictions = {
 298         'image_info': labels['image_info'],
 299         'num_detections': outputs['num_detections'],
 300         'detection_boxes': outputs['detection_boxes'],
 301         'detection_outer_boxes': outputs['detection_outer_boxes'],
 302         'detection_classes': outputs['detection_classes'],
 303         'detection_scores': outputs['detection_scores'],
 304         'detection_masks': outputs['detection_masks'],
 305     }
 306 
 307     if 'groundtruths' in labels:
 308       predictions['source_id'] = labels['groundtruths']['source_id']
 309       labels = labels['groundtruths']
 310 
 311     return labels, predictions
 312 
 313   def eval_metrics(self):
 314     return eval_factory.evaluator_generator(self._params.eval)
