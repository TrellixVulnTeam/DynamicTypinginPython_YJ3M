  36 class Fpn(object):
  37   """Feature pyramid networks."""
  38 
  39   def __init__(self,
  40                min_level=3,
  41                max_level=7,
  42                fpn_feat_dims=256,
  43                use_separable_conv=False,
  44                activation='relu',
  45                use_batch_norm=True,
  46                norm_activation=nn_ops.norm_activation_builder(
  47                    activation='relu')):
  48     """FPN initialization function.
  49 
  50     Args:
  51       min_level: `int` minimum level in FPN output feature maps.
  52       max_level: `int` maximum level in FPN output feature maps.
  53       fpn_feat_dims: `int` number of filters in FPN layers.
  54       use_separable_conv: `bool`, if True use separable convolution for
  55         convolution in FPN layers.
  56       use_batch_norm: 'bool', indicating whether batchnorm layers are added.
  57       norm_activation: an operation that includes a normalization layer
  58         followed by an optional activation layer.
  59     """
  60     self._min_level = min_level
  61     self._max_level = max_level
  62     self._fpn_feat_dims = fpn_feat_dims
  63     if use_separable_conv:
  64       self._conv2d_op = functools.partial(
  65           tf.keras.layers.SeparableConv2D, depth_multiplier=1)
  66     else:
  67       self._conv2d_op = tf.keras.layers.Conv2D
  68     if activation == 'relu':
  69       self._activation_op = tf.nn.relu
  70     elif activation == 'swish':
  71       self._activation_op = tf.nn.swish
  72     else:
  73       raise ValueError('Unsupported activation `{}`.'.format(activation))
  74     self._use_batch_norm = use_batch_norm
  75     self._norm_activation = norm_activation
  76 
  77     self._norm_activations = {}
  78     self._lateral_conv2d_op = {}
  79     self._post_hoc_conv2d_op = {}
  80     self._coarse_conv2d_op = {}
  81     for level in range(self._min_level, self._max_level + 1):
  82       if self._use_batch_norm:
  83         self._norm_activations[level] = norm_activation(
  84             use_activation=False, name='p%d-bn' % level)
  85       self._lateral_conv2d_op[level] = self._conv2d_op(
  86           filters=self._fpn_feat_dims,
  87           kernel_size=(1, 1),
  88           padding='same',
  89           name='l%d' % level)
  90       self._post_hoc_conv2d_op[level] = self._conv2d_op(
  91           filters=self._fpn_feat_dims,
  92           strides=(1, 1),
  93           kernel_size=(3, 3),
  94           padding='same',
  95           name='post_hoc_d%d' % level)
  96       self._coarse_conv2d_op[level] = self._conv2d_op(
  97           filters=self._fpn_feat_dims,
  98           strides=(2, 2),
  99           kernel_size=(3, 3),
 100           padding='same',
 101           name='p%d' % level)
 102 
 103   def __call__(self, multilevel_features, is_training=None):
 104     """Returns the FPN features for a given multilevel features.
 105 
 106     Args:
 107       multilevel_features: a `dict` containing `int` keys for continuous feature
 108         levels, e.g., [2, 3, 4, 5]. The values are corresponding features with
 109         shape [batch_size, height_l, width_l, num_filters].
 110       is_training: `bool` if True, the model is in training mode.
 111 
 112     Returns:
 113       a `dict` containing `int` keys for continuous feature levels
 114       [min_level, min_level + 1, ..., max_level]. The values are corresponding
 115       FPN features with shape [batch_size, height_l, width_l, fpn_feat_dims].
 116     """
 117     input_levels = list(multilevel_features.keys())
 118     if min(input_levels) > self._min_level:
 119       raise ValueError(
 120           'The minimum backbone level %d should be '%(min(input_levels)) +
 121           'less or equal to FPN minimum level %d.:'%(self._min_level))
 122     backbone_max_level = min(max(input_levels), self._max_level)
 123     with backend.get_graph().as_default(), tf.name_scope('fpn'):
 124       # Adds lateral connections.
 125       feats_lateral = {}
 126       for level in range(self._min_level, backbone_max_level + 1):
 127         feats_lateral[level] = self._lateral_conv2d_op[level](
 128             multilevel_features[level])
 129 
 130       # Adds top-down path.
 131       feats = {backbone_max_level: feats_lateral[backbone_max_level]}
 132       for level in range(backbone_max_level - 1, self._min_level - 1, -1):
 133         feats[level] = spatial_transform_ops.nearest_upsampling(
 134             feats[level + 1], 2) + feats_lateral[level]
 135 
 136       # Adds post-hoc 3x3 convolution kernel.
 137       for level in range(self._min_level, backbone_max_level + 1):
 138         feats[level] = self._post_hoc_conv2d_op[level](feats[level])
 139 
 140       # Adds coarser FPN levels introduced for RetinaNet.
 141       for level in range(backbone_max_level + 1, self._max_level + 1):
 142         feats_in = feats[level - 1]
 143         if level > backbone_max_level + 1:
 144           feats_in = self._activation_op(feats_in)
 145         feats[level] = self._coarse_conv2d_op[level](feats_in)
 146       if self._use_batch_norm:
 147         # Adds batch_norm layer.
 148         for level in range(self._min_level, self._max_level + 1):
 149           feats[level] = self._norm_activations[level](
 150               feats[level], is_training=is_training)
 151     return feats
