  28 def norm_activation_generator(params):
  29   return nn_ops.norm_activation_builder(
  30       momentum=params.batch_norm_momentum,
  31       epsilon=params.batch_norm_epsilon,
  32       trainable=params.batch_norm_trainable,
  33       activation=params.activation)

  36 def backbone_generator(params):
  37   """Generator function for various backbone models."""
  38   if params.architecture.backbone == 'resnet':
  39     resnet_params = params.resnet
  40     backbone_fn = resnet.Resnet(
  41         resnet_depth=resnet_params.resnet_depth,
  42         activation=params.norm_activation.activation,
  43         norm_activation=norm_activation_generator(
  44             params.norm_activation))
  45   else:
  46     raise ValueError('Backbone model `{}` is not supported.'
  47                      .format(params.architecture.backbone))
  48 
  49   return backbone_fn

  52 def multilevel_features_generator(params):
  53   """Generator function for various FPN models."""
  54   if params.architecture.multilevel_features == 'fpn':
  55     fpn_params = params.fpn
  56     fpn_fn = fpn.Fpn(
  57         min_level=params.architecture.min_level,
  58         max_level=params.architecture.max_level,
  59         fpn_feat_dims=fpn_params.fpn_feat_dims,
  60         use_separable_conv=fpn_params.use_separable_conv,
  61         activation=params.norm_activation.activation,
  62         use_batch_norm=fpn_params.use_batch_norm,
  63         norm_activation=norm_activation_generator(
  64             params.norm_activation))
  65   elif params.architecture.multilevel_features == 'identity':
  66     fpn_fn = identity.Identity()
  67   else:
  68     raise ValueError('The multi-level feature model `{}` is not supported.'
  69                      .format(params.architecture.multilevel_features))
  70   return fpn_fn

  73 def retinanet_head_generator(params):
  74   """Generator function for RetinaNet head architecture."""
  75   head_params = params.retinanet_head
  76   return heads.RetinanetHead(
  77       params.architecture.min_level,
  78       params.architecture.max_level,
  79       params.architecture.num_classes,
  80       head_params.anchors_per_location,
  81       head_params.num_convs,
  82       head_params.num_filters,
  83       head_params.use_separable_conv,
  84       norm_activation=norm_activation_generator(params.norm_activation))
