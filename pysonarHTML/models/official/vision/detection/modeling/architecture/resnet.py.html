  32 class Resnet(object):
  33   """Class to build ResNet family model."""
  34 
  35   def __init__(self,
  36                resnet_depth,
  37                activation='relu',
  38                norm_activation=nn_ops.norm_activation_builder(
  39                    activation='relu'),
  40                data_format='channels_last'):
  41     """ResNet initialization function.
  42 
  43     Args:
  44       resnet_depth: `int` depth of ResNet backbone model.
  45       norm_activation: an operation that includes a normalization layer
  46         followed by an optional activation layer.
  47       data_format: `str` either "channels_first" for `[batch, channels, height,
  48         width]` or "channels_last for `[batch, height, width, channels]`.
  49     """
  50     self._resnet_depth = resnet_depth
  51     if activation == 'relu':
  52       self._activation_op = tf.nn.relu
  53     elif activation == 'swish':
  54       self._activation_op = tf.nn.swish
  55     else:
  56       raise ValueError('Unsupported activation `{}`.'.format(activation))
  57     self._norm_activation = norm_activation
  58     self._data_format = data_format
  59 
  60     model_params = {
  61         10: {'block': self.residual_block, 'layers': [1, 1, 1, 1]},
  62         18: {'block': self.residual_block, 'layers': [2, 2, 2, 2]},
  63         34: {'block': self.residual_block, 'layers': [3, 4, 6, 3]},
  64         50: {'block': self.bottleneck_block, 'layers': [3, 4, 6, 3]},
  65         101: {'block': self.bottleneck_block, 'layers': [3, 4, 23, 3]},
  66         152: {'block': self.bottleneck_block, 'layers': [3, 8, 36, 3]},
  67         200: {'block': self.bottleneck_block, 'layers': [3, 24, 36, 3]}
  68     }
  69 
  70     if resnet_depth not in model_params:
  71       valid_resnet_depths = ', '.join(
  72           [str(depth) for depth in sorted(model_params.keys())])
  73       raise ValueError(
  74           'The resnet_depth should be in [%s]. Not a valid resnet_depth:'%(
  75               valid_resnet_depths), self._resnet_depth)
  76     params = model_params[resnet_depth]
  77     self._resnet_fn = self.resnet_v1_generator(
  78         params['block'], params['layers'])
  79 
  80   def __call__(self, inputs, is_training=None):
  81     """Returns the ResNet model for a given size and number of output classes.
  82 
  83     Args:
  84       inputs: a `Tesnor` with shape [batch_size, height, width, 3] representing
  85         a batch of images.
  86       is_training: `bool` if True, the model is in training mode.
  87 
  88     Returns:
  89       a `dict` containing `int` keys for continuous feature levels [2, 3, 4, 5].
  90       The values are corresponding feature hierarchy in ResNet with shape
  91       [batch_size, height_l, width_l, num_filters].
  92     """
  93     with backend.get_graph().as_default():
  94       with tf.name_scope('resnet%s' % self._resnet_depth):
  95         return self._resnet_fn(inputs, is_training)
  96 
  97   def fixed_padding(self, inputs, kernel_size):
  98     """Pads the input along the spatial dimensions independently of input size.
  99 
 100     Args:
 101       inputs: `Tensor` of size `[batch, channels, height, width]` or
 102           `[batch, height, width, channels]` depending on `data_format`.
 103       kernel_size: `int` kernel size to be used for `conv2d` or max_pool2d`
 104           operations. Should be a positive integer.
 105 
 106     Returns:
 107       A padded `Tensor` of the same `data_format` with size either intact
 108       (if `kernel_size == 1`) or padded (if `kernel_size > 1`).
 109     """
 110     pad_total = kernel_size - 1
 111     pad_beg = pad_total // 2
 112     pad_end = pad_total - pad_beg
 113     if self._data_format == 'channels_first':
 114       padded_inputs = tf.pad(
 115           tensor=inputs,
 116           paddings=[[0, 0], [0, 0], [pad_beg, pad_end], [pad_beg, pad_end]])
 117     else:
 118       padded_inputs = tf.pad(
 119           tensor=inputs,
 120           paddings=[[0, 0], [pad_beg, pad_end], [pad_beg, pad_end], [0, 0]])
 121 
 122     return padded_inputs
 123 
 124   def conv2d_fixed_padding(self, inputs, filters, kernel_size, strides):
 125     """Strided 2-D convolution with explicit padding.
 126 
 127     The padding is consistent and is based only on `kernel_size`, not on the
 128     dimensions of `inputs` (as opposed to using `tf.layers.conv2d` alone).
 129 
 130     Args:
 131       inputs: `Tensor` of size `[batch, channels, height_in, width_in]`.
 132       filters: `int` number of filters in the convolution.
 133       kernel_size: `int` size of the kernel to be used in the convolution.
 134       strides: `int` strides of the convolution.
 135 
 136     Returns:
 137       A `Tensor` of shape `[batch, filters, height_out, width_out]`.
 138     """
 139     if strides > 1:
 140       inputs = self.fixed_padding(inputs, kernel_size)
 141 
 142     return tf.keras.layers.Conv2D(
 143         filters=filters,
 144         kernel_size=kernel_size,
 145         strides=strides,
 146         padding=('SAME' if strides == 1 else 'VALID'),
 147         use_bias=False,
 148         kernel_initializer=tf.initializers.VarianceScaling(),
 149         data_format=self._data_format)(
 150             inputs=inputs)
 151 
 152   def residual_block(self,
 153                      inputs,
 154                      filters,
 155                      strides,
 156                      use_projection=False,
 157                      is_training=None):
 158     """Standard building block for residual networks with BN after convolutions.
 159 
 160     Args:
 161       inputs: `Tensor` of size `[batch, channels, height, width]`.
 162       filters: `int` number of filters for the first two convolutions. Note that
 163           the third and final convolution will use 4 times as many filters.
 164       strides: `int` block stride. If greater than 1, this block will ultimately
 165           downsample the input.
 166       use_projection: `bool` for whether this block should use a projection
 167           shortcut (versus the default identity shortcut). This is usually
 168           `True` for the first block of a block group, which may change the
 169           number of filters and the resolution.
 170       is_training: `bool` if True, the model is in training mode.
 171     Returns:
 172       The output `Tensor` of the block.
 173     """
 174     shortcut = inputs
 175     if use_projection:
 176       # Projection shortcut in first layer to match filters and strides
 177       shortcut = self.conv2d_fixed_padding(
 178           inputs=inputs, filters=filters, kernel_size=1, strides=strides)
 179       shortcut = self._norm_activation(use_activation=False)(
 180           shortcut, is_training=is_training)
 181 
 182     inputs = self.conv2d_fixed_padding(
 183         inputs=inputs, filters=filters, kernel_size=3, strides=strides)
 184     inputs = self._norm_activation()(inputs, is_training=is_training)
 185 
 186     inputs = self.conv2d_fixed_padding(
 187         inputs=inputs, filters=filters, kernel_size=3, strides=1)
 188     inputs = self._norm_activation(use_activation=False, init_zero=True)(
 189         inputs, is_training=is_training)
 190 
 191     return self._activation_op(inputs + shortcut)
 192 
 193   def bottleneck_block(self,
 194                        inputs,
 195                        filters,
 196                        strides,
 197                        use_projection=False,
 198                        is_training=None):
 199     """Bottleneck block variant for residual networks with BN after convolutions.
 200 
 201     Args:
 202       inputs: `Tensor` of size `[batch, channels, height, width]`.
 203       filters: `int` number of filters for the first two convolutions. Note that
 204           the third and final convolution will use 4 times as many filters.
 205       strides: `int` block stride. If greater than 1, this block will ultimately
 206           downsample the input.
 207       use_projection: `bool` for whether this block should use a projection
 208           shortcut (versus the default identity shortcut). This is usually
 209           `True` for the first block of a block group, which may change the
 210           number of filters and the resolution.
 211       is_training: `bool` if True, the model is in training mode.
 212 
 213     Returns:
 214       The output `Tensor` of the block.
 215     """
 216     shortcut = inputs
 217     if use_projection:
 218       # Projection shortcut only in first block within a group. Bottleneck
 219       # blocks end with 4 times the number of filters.
 220       filters_out = 4 * filters
 221       shortcut = self.conv2d_fixed_padding(
 222           inputs=inputs, filters=filters_out, kernel_size=1, strides=strides)
 223       shortcut = self._norm_activation(use_activation=False)(
 224           shortcut, is_training=is_training)
 225 
 226     inputs = self.conv2d_fixed_padding(
 227         inputs=inputs, filters=filters, kernel_size=1, strides=1)
 228     inputs = self._norm_activation()(inputs, is_training=is_training)
 229 
 230     inputs = self.conv2d_fixed_padding(
 231         inputs=inputs, filters=filters, kernel_size=3, strides=strides)
 232     inputs = self._norm_activation()(inputs, is_training=is_training)
 233 
 234     inputs = self.conv2d_fixed_padding(
 235         inputs=inputs, filters=4 * filters, kernel_size=1, strides=1)
 236     inputs = self._norm_activation(use_activation=False, init_zero=True)(
 237         inputs, is_training=is_training)
 238 
 239     return self._activation_op(inputs + shortcut)
 240 
 241   def block_group(self, inputs, filters, block_fn, blocks, strides, name,
 242                   is_training):
 243     """Creates one group of blocks for the ResNet model.
 244 
 245     Args:
 246       inputs: `Tensor` of size `[batch, channels, height, width]`.
 247       filters: `int` number of filters for the first convolution of the layer.
 248       block_fn: `function` for the block to use within the model
 249       blocks: `int` number of blocks contained in the layer.
 250       strides: `int` stride to use for the first convolution of the layer. If
 251           greater than 1, this layer will downsample the input.
 252       name: `str`name for the Tensor output of the block layer.
 253       is_training: `bool` if True, the model is in training mode.
 254 
 255     Returns:
 256       The output `Tensor` of the block layer.
 257     """
 258     # Only the first block per block_group uses projection shortcut and strides.
 259     inputs = block_fn(inputs, filters, strides, use_projection=True,
 260                       is_training=is_training)
 261 
 262     for _ in range(1, blocks):
 263       inputs = block_fn(inputs, filters, 1, is_training=is_training)
 264 
 265     return tf.identity(inputs, name)
 266 
 267   def resnet_v1_generator(self, block_fn, layers):
 268     """Generator for ResNet v1 models.
 269 
 270     Args:
 271       block_fn: `function` for the block to use within the model. Either
 272           `residual_block` or `bottleneck_block`.
 273       layers: list of 4 `int`s denoting the number of blocks to include in each
 274         of the 4 block groups. Each group consists of blocks that take inputs of
 275         the same resolution.
 276 
 277     Returns:
 278       Model `function` that takes in `inputs` and `is_training` and returns the
 279       output `Tensor` of the ResNet model.
 280     """
 281 
 282     def model(inputs, is_training=None):
 283       """Creation of the model graph."""
 284       inputs = self.conv2d_fixed_padding(
 285           inputs=inputs, filters=64, kernel_size=7, strides=2)
 286       inputs = tf.identity(inputs, 'initial_conv')
 287       inputs = self._norm_activation()(inputs, is_training=is_training)
 288 
 289       inputs = tf.keras.layers.MaxPool2D(
 290           pool_size=3, strides=2, padding='SAME',
 291           data_format=self._data_format)(
 292               inputs)
 293       inputs = tf.identity(inputs, 'initial_max_pool')
 294 
 295       c2 = self.block_group(
 296           inputs=inputs, filters=64, block_fn=block_fn, blocks=layers[0],
 297           strides=1, name='block_group1', is_training=is_training)
 298       c3 = self.block_group(
 299           inputs=c2, filters=128, block_fn=block_fn, blocks=layers[1],
 300           strides=2, name='block_group2', is_training=is_training)
 301       c4 = self.block_group(
 302           inputs=c3, filters=256, block_fn=block_fn, blocks=layers[2],
 303           strides=2, name='block_group3', is_training=is_training)
 304       c5 = self.block_group(
 305           inputs=c4, filters=512, block_fn=block_fn, blocks=layers[3],
 306           strides=2, name='block_group4', is_training=is_training)
 307       return {2: c2, 3: c3, 4: c4, 5: c5}
 308 
 309     return model
