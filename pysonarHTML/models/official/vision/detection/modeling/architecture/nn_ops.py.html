  25 class NormActivation(tf.keras.layers.Layer):
  26   """Combined Normalization and Activation layers."""
  27 
  28   def __init__(self,
  29                momentum=0.997,
  30                epsilon=1e-4,
  31                trainable=True,
  32                init_zero=False,
  33                use_activation=True,
  34                activation='relu',
  35                fused=True,
  36                name=None):
  37     """A class to construct layers for a batch normalization followed by a ReLU.
  38 
  39     Args:
  40       momentum: momentum for the moving average.
  41       epsilon: small float added to variance to avoid dividing by zero.
  42       trainable: `bool`, if True also add variables to the graph collection
  43         GraphKeys.TRAINABLE_VARIABLES. If False, freeze batch normalization
  44         layer.
  45       init_zero: `bool` if True, initializes scale parameter of batch
  46           normalization with 0. If False, initialize it with 1.
  47       fused: `bool` fused option in batch normalziation.
  48       use_actiation: `bool`, whether to add the optional activation layer after
  49         the batch normalization layer.
  50       activation: 'string', the type of the activation layer. Currently support
  51         `relu` and `swish`.
  52       name: `str` name for the operation.
  53     """
  54     super(NormActivation, self).__init__(trainable=trainable)
  55     if init_zero:
  56       gamma_initializer = tf.keras.initializers.Zeros()
  57     else:
  58       gamma_initializer = tf.keras.initializers.Ones()
  59     self._normalization_op = tf.keras.layers.BatchNormalization(
  60         momentum=momentum,
  61         epsilon=epsilon,
  62         center=True,
  63         scale=True,
  64         trainable=trainable,
  65         fused=fused,
  66         gamma_initializer=gamma_initializer,
  67         name=name)
  68     self._use_activation = use_activation
  69     if activation == 'relu':
  70       self._activation_op = tf.nn.relu
  71     elif activation == 'swish':
  72       self._activation_op = tf.nn.swish
  73     else:
  74       raise ValueError('Unsupported activation `{}`.'.format(activation))
  75 
  76   def __call__(self, inputs, is_training=None):
  77     """Builds the normalization layer followed by an optional activation layer.
  78 
  79     Args:
  80       inputs: `Tensor` of shape `[batch, channels, ...]`.
  81       is_training: `boolean`, if True if model is in training mode.
  82 
  83     Returns:
  84       A normalized `Tensor` with the same `data_format`.
  85     """
  86     # We will need to keep training=None by default, so that it can be inherit
  87     # from keras.Model.training
  88     if is_training and self.trainable:
  89       is_training = True
  90     inputs = self._normalization_op(inputs, training=is_training)
  91 
  92     if self._use_activation:
  93       inputs = self._activation_op(inputs)
  94     return inputs
  95 
  96 
  97 def norm_activation_builder(momentum=0.997,
  98                             epsilon=1e-4,
  99                             trainable=True,
 100                             activation='relu',
 101                             **kwargs):
 102   return functools.partial(
 103       NormActivation,
 104       momentum=momentum,
 105       epsilon=epsilon,
 106       trainable=trainable,
 107       activation=activation,
 108       **kwargs)
