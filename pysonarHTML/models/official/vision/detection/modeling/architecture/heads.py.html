  30 class RpnHead(tf.keras.layers.Layer):
  31   """Region Proposal Network head."""
  32 
  33   def __init__(self,
  34                min_level,
  35                max_level,
  36                anchors_per_location,
  37                num_convs=2,
  38                num_filters=256,
  39                use_separable_conv=False,
  40                activation='relu',
  41                use_batch_norm=True,
  42                norm_activation=nn_ops.norm_activation_builder(
  43                    activation='relu')):
  44     """Initialize params to build Region Proposal Network head.
  45 
  46     Args:
  47       min_level: `int` number of minimum feature level.
  48       max_level: `int` number of maximum feature level.
  49       anchors_per_location: `int` number of number of anchors per pixel
  50         location.
  51       num_convs: `int` number that represents the number of the intermediate
  52         conv layers before the prediction.
  53       num_filters: `int` number that represents the number of filters of the
  54         intermediate conv layers.
  55       use_separable_conv: `bool`, indicating whether the separable conv layers
  56         is used.
  57       activation: activation function. Support 'relu' and 'swish'.
  58       use_batch_norm: 'bool', indicating whether batchnorm layers are added.
  59       norm_activation: an operation that includes a normalization layer
  60         followed by an optional activation layer.
  61     """
  62     self._min_level = min_level
  63     self._max_level = max_level
  64     self._anchors_per_location = anchors_per_location
  65     if activation == 'relu':
  66       self._activation_op = tf.nn.relu
  67     elif activation == 'swish':
  68       self._activation_op = tf.nn.swish
  69     else:
  70       raise ValueError('Unsupported activation `{}`.'.format(activation))
  71     self._use_batch_norm = use_batch_norm
  72 
  73     if use_separable_conv:
  74       self._conv2d_op = functools.partial(
  75           tf.keras.layers.SeparableConv2D,
  76           depth_multiplier=1,
  77           bias_initializer=tf.zeros_initializer())
  78     else:
  79       self._conv2d_op = functools.partial(
  80           tf.keras.layers.Conv2D,
  81           kernel_initializer=tf.keras.initializers.RandomNormal(stddev=0.01),
  82           bias_initializer=tf.zeros_initializer())
  83 
  84     self._rpn_conv = self._conv2d_op(
  85         num_filters,
  86         kernel_size=(3, 3),
  87         strides=(1, 1),
  88         activation=(None if self._use_batch_norm else self._activation_op),
  89         padding='same',
  90         name='rpn')
  91     self._rpn_class_conv = self._conv2d_op(
  92         anchors_per_location,
  93         kernel_size=(1, 1),
  94         strides=(1, 1),
  95         padding='valid',
  96         name='rpn-class')
  97     self._rpn_box_conv = self._conv2d_op(
  98         4 * anchors_per_location,
  99         kernel_size=(1, 1),
 100         strides=(1, 1),
 101         padding='valid',
 102         name='rpn-box')
 103 
 104     self._norm_activations = {}
 105     if self._use_batch_norm:
 106       for level in range(self._min_level, self._max_level + 1):
 107         self._norm_activations[level] = norm_activation(name='rpn-l%d-bn' %
 108                                                         level)
 109 
 110   def _shared_rpn_heads(self, features, anchors_per_location, level,
 111                         is_training):
 112     """Shared RPN heads."""
 113     features = self._rpn_conv(features)
 114     if self._use_batch_norm:
 115       # The batch normalization layers are not shared between levels.
 116       features = self._norm_activations[level](
 117           features, is_training=is_training)
 118     # Proposal classification scores
 119     scores = self._rpn_class_conv(features)
 120     # Proposal bbox regression deltas
 121     bboxes = self._rpn_box_conv(features)
 122 
 123     return scores, bboxes
 124 
 125   def __call__(self, features, is_training=None):
 126 
 127     scores_outputs = {}
 128     box_outputs = {}
 129 
 130     with backend.get_graph().as_default(), tf.name_scope('rpn_head'):
 131       for level in range(self._min_level, self._max_level + 1):
 132         scores_output, box_output = self._shared_rpn_heads(
 133             features[level], self._anchors_per_location, level, is_training)
 134         scores_outputs[level] = scores_output
 135         box_outputs[level] = box_output
 136       return scores_outputs, box_outputs

 139 class FastrcnnHead(tf.keras.layers.Layer):
 140   """Fast R-CNN box head."""
 141 
 142   def __init__(self,
 143                num_classes,
 144                num_convs=0,
 145                num_filters=256,
 146                use_separable_conv=False,
 147                num_fcs=2,
 148                fc_dims=1024,
 149                activation='relu',
 150                use_batch_norm=True,
 151                norm_activation=nn_ops.norm_activation_builder(
 152                    activation='relu')):
 153     """Initialize params to build Fast R-CNN box head.
 154 
 155     Args:
 156       num_classes: a integer for the number of classes.
 157       num_convs: `int` number that represents the number of the intermediate
 158         conv layers before the FC layers.
 159       num_filters: `int` number that represents the number of filters of the
 160         intermediate conv layers.
 161       use_separable_conv: `bool`, indicating whether the separable conv layers
 162         is used.
 163       num_fcs: `int` number that represents the number of FC layers before the
 164         predictions.
 165       fc_dims: `int` number that represents the number of dimension of the FC
 166         layers.
 167       activation: activation function. Support 'relu' and 'swish'.
 168       use_batch_norm: 'bool', indicating whether batchnorm layers are added.
 169       norm_activation: an operation that includes a normalization layer
 170         followed by an optional activation layer.
 171     """
 172     self._num_classes = num_classes
 173 
 174     self._num_convs = num_convs
 175     self._num_filters = num_filters
 176     if use_separable_conv:
 177       self._conv2d_op = functools.partial(
 178           tf.keras.layers.SeparableConv2D,
 179           depth_multiplier=1,
 180           bias_initializer=tf.zeros_initializer())
 181     else:
 182       self._conv2d_op = functools.partial(
 183           tf.keras.layers.Conv2D,
 184           kernel_initializer=tf.keras.initializers.VarianceScaling(
 185               scale=2, mode='fan_out', distribution='untruncated_normal'),
 186           bias_initializer=tf.zeros_initializer())
 187 
 188     self._num_fcs = num_fcs
 189     self._fc_dims = fc_dims
 190     if activation == 'relu':
 191       self._activation_op = tf.nn.relu
 192     elif activation == 'swish':
 193       self._activation_op = tf.nn.swish
 194     else:
 195       raise ValueError('Unsupported activation `{}`.'.format(activation))
 196     self._use_batch_norm = use_batch_norm
 197     self._norm_activation = norm_activation
 198 
 199     self._conv_ops = []
 200     self._conv_bn_ops = []
 201     for i in range(self._num_convs):
 202       self._conv_ops.append(
 203           self._conv2d_op(
 204               self._num_filters,
 205               kernel_size=(3, 3),
 206               strides=(1, 1),
 207               padding='same',
 208               dilation_rate=(1, 1),
 209               activation=(None if self._use_batch_norm else self._activation_op),
 210               name='conv_{}'.format(i)))
 211       if self._use_batch_norm:
 212         self._conv_bn_ops.append(self._norm_activation())
 213 
 214     self._fc_ops = []
 215     self._fc_bn_ops = []
 216     for i in range(self._num_fcs):
 217       self._fc_ops.append(
 218           tf.keras.layers.Dense(
 219               units=self._fc_dims,
 220               activation=(None if self._use_batch_norm else self._activation_op),
 221               name='fc{}'.format(i)))
 222       if self._use_batch_norm:
 223         self._fc_bn_ops.append(self._norm_activation(fused=False))
 224 
 225     self._class_predict = tf.keras.layers.Dense(
 226         self._num_classes,
 227         kernel_initializer=tf.keras.initializers.RandomNormal(stddev=0.01),
 228         bias_initializer=tf.zeros_initializer(),
 229         name='class-predict')
 230     self._box_predict = tf.keras.layers.Dense(
 231         self._num_classes * 4,
 232         kernel_initializer=tf.keras.initializers.RandomNormal(stddev=0.001),
 233         bias_initializer=tf.zeros_initializer(),
 234         name='box-predict')
 235 
 236   def __call__(self, roi_features, is_training=None):
 237     """Box and class branches for the Mask-RCNN model.
 238 
 239     Args:
 240       roi_features: A ROI feature tensor of shape
 241         [batch_size, num_rois, height_l, width_l, num_filters].
 242       is_training: `boolean`, if True if model is in training mode.
 243 
 244     Returns:
 245       class_outputs: a tensor with a shape of
 246         [batch_size, num_rois, num_classes], representing the class predictions.
 247       box_outputs: a tensor with a shape of
 248         [batch_size, num_rois, num_classes * 4], representing the box
 249         predictions.
 250     """
 251 
 252     with backend.get_graph().as_default(), tf.name_scope('fast_rcnn_head'):
 253       # reshape inputs beofre FC.
 254       _, num_rois, height, width, filters = roi_features.get_shape().as_list()
 255 
 256       net = tf.reshape(roi_features, [-1, height, width, filters])
 257       for i in range(self._num_convs):
 258         net = self._conv_ops[i](net)
 259         if self._use_batch_norm:
 260           net = self._conv_bn_ops[i](net, is_training=is_training)
 261 
 262       filters = self._num_filters if self._num_convs > 0 else filters
 263       net = tf.reshape(net, [-1, num_rois, height * width * filters])
 264 
 265       for i in range(self._num_fcs):
 266         net = self._fc_ops[i](net)
 267         if self._use_batch_norm:
 268           net = self._fc_bn_ops[i](net, is_training=is_training)
 269 
 270       class_outputs = self._class_predict(net)
 271       box_outputs = self._box_predict(net)
 272       return class_outputs, box_outputs

 275 class MaskrcnnHead(tf.keras.layers.Layer):
 276   """Mask R-CNN head."""
 277 
 278   def __init__(self,
 279                num_classes,
 280                mask_target_size,
 281                num_convs=4,
 282                num_filters=256,
 283                use_separable_conv=False,
 284                activation='relu',
 285                use_batch_norm=True,
 286                norm_activation=nn_ops.norm_activation_builder(
 287                    activation='relu')):
 288     """Initialize params to build Fast R-CNN head.
 289 
 290     Args:
 291       num_classes: a integer for the number of classes.
 292       mask_target_size: a integer that is the resolution of masks.
 293       num_convs: `int` number that represents the number of the intermediate
 294         conv layers before the prediction.
 295       num_filters: `int` number that represents the number of filters of the
 296         intermediate conv layers.
 297       use_separable_conv: `bool`, indicating whether the separable conv layers
 298         is used.
 299       activation: activation function. Support 'relu' and 'swish'.
 300       use_batch_norm: 'bool', indicating whether batchnorm layers are added.
 301       norm_activation: an operation that includes a normalization layer
 302         followed by an optional activation layer.
 303     """
 304     self._num_classes = num_classes
 305     self._mask_target_size = mask_target_size
 306 
 307     self._num_convs = num_convs
 308     self._num_filters = num_filters
 309     if use_separable_conv:
 310       self._conv2d_op = functools.partial(
 311           tf.keras.layers.SeparableConv2D,
 312           depth_multiplier=1,
 313           bias_initializer=tf.zeros_initializer())
 314     else:
 315       self._conv2d_op = functools.partial(
 316           tf.keras.layers.Conv2D,
 317           kernel_initializer=tf.keras.initializers.VarianceScaling(
 318               scale=2, mode='fan_out', distribution='untruncated_normal'),
 319           bias_initializer=tf.zeros_initializer())
 320     if activation == 'relu':
 321       self._activation_op = tf.nn.relu
 322     elif activation == 'swish':
 323       self._activation_op = tf.nn.swish
 324     else:
 325       raise ValueError('Unsupported activation `{}`.'.format(activation))
 326     self._use_batch_norm = use_batch_norm
 327     self._norm_activation = norm_activation
 328     self._conv2d_ops = []
 329     for i in range(self._num_convs):
 330       self._conv2d_ops.append(
 331           self._conv2d_op(
 332               self._num_filters,
 333               kernel_size=(3, 3),
 334               strides=(1, 1),
 335               padding='same',
 336               dilation_rate=(1, 1),
 337               activation=(None if self._use_batch_norm else self._activation_op),
 338               name='mask-conv-l%d' % i))
 339     self._mask_conv_transpose = tf.keras.layers.Conv2DTranspose(
 340         self._num_filters,
 341         kernel_size=(2, 2),
 342         strides=(2, 2),
 343         padding='valid',
 344         activation=(None if self._use_batch_norm else self._activation_op),
 345         kernel_initializer=tf.keras.initializers.VarianceScaling(
 346             scale=2, mode='fan_out', distribution='untruncated_normal'),
 347         bias_initializer=tf.zeros_initializer(),
 348         name='conv5-mask')
 349 
 350   def __call__(self, roi_features, class_indices, is_training=None):
 351     """Mask branch for the Mask-RCNN model.
 352 
 353     Args:
 354       roi_features: A ROI feature tensor of shape
 355         [batch_size, num_rois, height_l, width_l, num_filters].
 356       class_indices: a Tensor of shape [batch_size, num_rois], indicating
 357         which class the ROI is.
 358       is_training: `boolean`, if True if model is in training mode.
 359 
 360     Returns:
 361       mask_outputs: a tensor with a shape of
 362         [batch_size, num_masks, mask_height, mask_width, num_classes],
 363         representing the mask predictions.
 364       fg_gather_indices: a tensor with a shape of [batch_size, num_masks, 2],
 365         representing the fg mask targets.
 366     Raises:
 367       ValueError: If boxes is not a rank-3 tensor or the last dimension of
 368         boxes is not 4.
 369     """
 370 
 371     with backend.get_graph().as_default():
 372       with tf.name_scope('mask_head'):
 373         _, num_rois, height, width, filters = roi_features.get_shape().as_list()
 374         net = tf.reshape(roi_features, [-1, height, width, filters])
 375 
 376         for i in range(self._num_convs):
 377           net = self._conv2d_ops[i](net)
 378           if self._use_batch_norm:
 379             net = self._norm_activation()(net, is_training=is_training)
 380 
 381         net = self._mask_conv_transpose(net)
 382         if self._use_batch_norm:
 383           net = self._norm_activation()(net, is_training=is_training)
 384 
 385         mask_outputs = self._conv2d_op(
 386             self._num_classes,
 387             kernel_size=(1, 1),
 388             strides=(1, 1),
 389             padding='valid',
 390             name='mask_fcn_logits')(
 391                 net)
 392         mask_outputs = tf.reshape(mask_outputs, [
 393             -1, num_rois, self._mask_target_size, self._mask_target_size,
 394             self._num_classes
 395         ])
 396 
 397         with tf.name_scope('masks_post_processing'):
 398           # TODO(pengchong): Figure out the way not to use the static inferred
 399           # batch size.
 400           batch_size, num_masks = class_indices.get_shape().as_list()
 401           mask_outputs = tf.transpose(a=mask_outputs, perm=[0, 1, 4, 2, 3])
 402           # Contructs indices for gather.
 403           batch_indices = tf.tile(
 404               tf.expand_dims(tf.range(batch_size), axis=1), [1, num_masks])
 405           mask_indices = tf.tile(
 406               tf.expand_dims(tf.range(num_masks), axis=0), [batch_size, 1])
 407           gather_indices = tf.stack(
 408               [batch_indices, mask_indices, class_indices], axis=2)
 409           mask_outputs = tf.gather_nd(mask_outputs, gather_indices)
 410       return mask_outputs

 413 class RetinanetHead(object):
 414   """RetinaNet head."""
 415 
 416   def __init__(self,
 417                min_level,
 418                max_level,
 419                num_classes,
 420                anchors_per_location,
 421                num_convs=4,
 422                num_filters=256,
 423                use_separable_conv=False,
 424                norm_activation=nn_ops.norm_activation_builder(
 425                    activation='relu')):
 426     """Initialize params to build RetinaNet head.
 427 
 428     Args:
 429       min_level: `int` number of minimum feature level.
 430       max_level: `int` number of maximum feature level.
 431       num_classes: `int` number of classification categories.
 432       anchors_per_location: `int` number of anchors per pixel location.
 433       num_convs: `int` number of stacked convolution before the last prediction
 434         layer.
 435       num_filters: `int` number of filters used in the head architecture.
 436       use_separable_conv: `bool` to indicate whether to use separable
 437         convoluation.
 438       norm_activation: an operation that includes a normalization layer
 439         followed by an optional activation layer.
 440     """
 441     self._min_level = min_level
 442     self._max_level = max_level
 443 
 444     self._num_classes = num_classes
 445     self._anchors_per_location = anchors_per_location
 446 
 447     self._num_convs = num_convs
 448     self._num_filters = num_filters
 449     self._use_separable_conv = use_separable_conv
 450     with tf.name_scope('class_net') as scope_name:
 451       self._class_name_scope = tf.name_scope(scope_name)
 452     with tf.name_scope('box_net') as scope_name:
 453       self._box_name_scope = tf.name_scope(scope_name)
 454     self._build_class_net_layers(norm_activation)
 455     self._build_box_net_layers(norm_activation)
 456 
 457   def _class_net_batch_norm_name(self, i, level):
 458     return 'class-%d-%d' % (i, level)
 459 
 460   def _box_net_batch_norm_name(self, i, level):
 461     return 'box-%d-%d' % (i, level)
 462 
 463   def _build_class_net_layers(self, norm_activation):
 464     """Build re-usable layers for class prediction network."""
 465     if self._use_separable_conv:
 466       self._class_predict = tf.keras.layers.SeparableConv2D(
 467           self._num_classes * self._anchors_per_location,
 468           kernel_size=(3, 3),
 469           bias_initializer=tf.constant_initializer(-np.log((1 - 0.01) / 0.01)),
 470           padding='same',
 471           name='class-predict')
 472     else:
 473       self._class_predict = tf.keras.layers.Conv2D(
 474           self._num_classes * self._anchors_per_location,
 475           kernel_size=(3, 3),
 476           bias_initializer=tf.constant_initializer(-np.log((1 - 0.01) / 0.01)),
 477           kernel_initializer=tf.keras.initializers.RandomNormal(stddev=1e-5),
 478           padding='same',
 479           name='class-predict')
 480     self._class_conv = []
 481     self._class_norm_activation = {}
 482     for i in range(self._num_convs):
 483       if self._use_separable_conv:
 484         self._class_conv.append(
 485             tf.keras.layers.SeparableConv2D(
 486                 self._num_filters,
 487                 kernel_size=(3, 3),
 488                 bias_initializer=tf.zeros_initializer(),
 489                 activation=None,
 490                 padding='same',
 491                 name='class-' + str(i)))
 492       else:
 493         self._class_conv.append(
 494             tf.keras.layers.Conv2D(
 495                 self._num_filters,
 496                 kernel_size=(3, 3),
 497                 bias_initializer=tf.zeros_initializer(),
 498                 kernel_initializer=tf.keras.initializers.RandomNormal(
 499                     stddev=0.01),
 500                 activation=None,
 501                 padding='same',
 502                 name='class-' + str(i)))
 503       for level in range(self._min_level, self._max_level + 1):
 504         name = self._class_net_batch_norm_name(i, level)
 505         self._class_norm_activation[name] = norm_activation(name=name)
 506 
 507   def _build_box_net_layers(self, norm_activation):
 508     """Build re-usable layers for box prediction network."""
 509     if self._use_separable_conv:
 510       self._box_predict = tf.keras.layers.SeparableConv2D(
 511           4 * self._anchors_per_location,
 512           kernel_size=(3, 3),
 513           bias_initializer=tf.zeros_initializer(),
 514           padding='same',
 515           name='box-predict')
 516     else:
 517       self._box_predict = tf.keras.layers.Conv2D(
 518           4 * self._anchors_per_location,
 519           kernel_size=(3, 3),
 520           bias_initializer=tf.zeros_initializer(),
 521           kernel_initializer=tf.keras.initializers.RandomNormal(stddev=1e-5),
 522           padding='same',
 523           name='box-predict')
 524     self._box_conv = []
 525     self._box_norm_activation = {}
 526     for i in range(self._num_convs):
 527       if self._use_separable_conv:
 528         self._box_conv.append(
 529             tf.keras.layers.SeparableConv2D(
 530                 self._num_filters,
 531                 kernel_size=(3, 3),
 532                 activation=None,
 533                 bias_initializer=tf.zeros_initializer(),
 534                 padding='same',
 535                 name='box-' + str(i)))
 536       else:
 537         self._box_conv.append(
 538             tf.keras.layers.Conv2D(
 539                 self._num_filters,
 540                 kernel_size=(3, 3),
 541                 activation=None,
 542                 bias_initializer=tf.zeros_initializer(),
 543                 kernel_initializer=tf.keras.initializers.RandomNormal(
 544                     stddev=0.01),
 545                 padding='same',
 546                 name='box-' + str(i)))
 547       for level in range(self._min_level, self._max_level + 1):
 548         name = self._box_net_batch_norm_name(i, level)
 549         self._box_norm_activation[name] = norm_activation(name=name)
 550 
 551   def __call__(self, fpn_features, is_training=None):
 552     """Returns outputs of RetinaNet head."""
 553     class_outputs = {}
 554     box_outputs = {}
 555     with backend.get_graph().as_default(), tf.name_scope('retinanet_head'):
 556       for level in range(self._min_level, self._max_level + 1):
 557         features = fpn_features[level]
 558 
 559         class_outputs[level] = self.class_net(
 560             features, level, is_training=is_training)
 561         box_outputs[level] = self.box_net(
 562             features, level, is_training=is_training)
 563     return class_outputs, box_outputs
 564 
 565   def class_net(self, features, level, is_training):
 566     """Class prediction network for RetinaNet."""
 567     with self._class_name_scope:
 568       for i in range(self._num_convs):
 569         features = self._class_conv[i](features)
 570         # The convolution layers in the class net are shared among all levels,
 571         # but each level has its batch normlization to capture the statistical
 572         # difference among different levels.
 573         name = self._class_net_batch_norm_name(i, level)
 574         features = self._class_norm_activation[name](
 575             features, is_training=is_training)
 576 
 577       classes = self._class_predict(features)
 578     return classes
 579 
 580   def box_net(self, features, level, is_training=None):
 581     """Box regression network for RetinaNet."""
 582     with self._box_name_scope:
 583       for i in range(self._num_convs):
 584         features = self._box_conv[i](features)
 585         # The convolution layers in the box net are shared among all levels, but
 586         # each level has its batch normlization to capture the statistical
 587         # difference among different levels.
 588         name = self._box_net_batch_norm_name(i, level)

 590             features, is_training=is_training)
 591 
 592       boxes = self._box_predict(features)

 597 class ShapemaskPriorHead(object):
 598   """ShapeMask Prior head."""
 599 
 600   def __init__(self,
 601                num_classes,
 602                num_downsample_channels,
 603                mask_crop_size,
 604                use_category_for_mask,
 605                shape_prior_path):
 606     """Initialize params to build RetinaNet head.
 607 
 608     Args:
 609       num_classes: Number of output classes.
 610       num_downsample_channels: number of channels in mask branch.
 611       mask_crop_size: feature crop size.
 612       use_category_for_mask: use class information in mask branch.
 613       shape_prior_path: the path to load shape priors.
 614     """
 615     self._mask_num_classes = num_classes if use_category_for_mask else 1
 616     self._num_downsample_channels = num_downsample_channels
 617     self._mask_crop_size = mask_crop_size
 618     self._shape_prior_path = shape_prior_path
 619     self._use_category_for_mask = use_category_for_mask
 620 
 621     self._shape_prior_fc = tf.keras.layers.Dense(
 622         self._num_downsample_channels, name='shape-prior-fc')
 623 
 624   def __call__(self, fpn_features, boxes, outer_boxes, classes, is_training):
 625     """Generate the detection priors from the box detections and FPN features.
 626 
 627     This corresponds to the Fig. 4 of the ShapeMask paper at
 628     https://arxiv.org/pdf/1904.03239.pdf
 629 
 630     Args:
 631       fpn_features: a dictionary of FPN features.
 632       boxes: a float tensor of shape [batch_size, num_instances, 4]
 633         representing the tight gt boxes from dataloader/detection.
 634       outer_boxes: a float tensor of shape [batch_size, num_instances, 4]
 635         representing the loose gt boxes from dataloader/detection.
 636       classes: a int Tensor of shape [batch_size, num_instances]
 637         of instance classes.
 638       is_training: training mode or not.
 639 
 640     Returns:
 641       instance_features: a float Tensor of shape [batch_size * num_instances,
 642           mask_crop_size, mask_crop_size, num_downsample_channels]. This is the
 643           instance feature crop.
 644       detection_priors: A float Tensor of shape [batch_size * num_instances,
 645         mask_size, mask_size, 1].
 646     """
 647     with backend.get_graph().as_default(), tf.name_scope('prior_mask'):
 648       batch_size, num_instances, _ = boxes.get_shape().as_list()
 649       outer_boxes = tf.cast(outer_boxes, tf.float32)
 650       boxes = tf.cast(boxes, tf.float32)
 651       instance_features = spatial_transform_ops.multilevel_crop_and_resize(
 652           fpn_features, outer_boxes, output_size=self._mask_crop_size)
 653       instance_features = self._shape_prior_fc(instance_features)
 654 
 655       shape_priors = self._get_priors()
 656 
 657       # Get uniform priors for each outer box.
 658       uniform_priors = tf.ones([batch_size, num_instances, self._mask_crop_size,
 659                                 self._mask_crop_size])
 660       uniform_priors = spatial_transform_ops.crop_mask_in_target_box(
 661           uniform_priors, boxes, outer_boxes, self._mask_crop_size)
 662 
 663       # Classify shape priors using uniform priors + instance features.
 664       prior_distribution = self._classify_shape_priors(
 665           tf.cast(instance_features, tf.float32), uniform_priors, classes)
 666 
 667       instance_priors = tf.gather(shape_priors, classes)
 668       instance_priors *= tf.expand_dims(tf.expand_dims(
 669           tf.cast(prior_distribution, tf.float32), axis=-1), axis=-1)
 670       instance_priors = tf.reduce_sum(instance_priors, axis=2)
 671       detection_priors = spatial_transform_ops.crop_mask_in_target_box(
 672           instance_priors, boxes, outer_boxes, self._mask_crop_size)
 673 
 674       return instance_features, detection_priors
 675 
 676   def _get_priors(self):
 677     """Load shape priors from file."""
 678     # loads class specific or agnostic shape priors
 679     if self._shape_prior_path:
 680       # Priors are loaded into shape [mask_num_classes, num_clusters, 32, 32].
 681       priors = np.load(tf.io.gfile.GFile(self._shape_prior_path, 'rb'))
 682       priors = tf.convert_to_tensor(priors, dtype=tf.float32)
 683       self._num_clusters = priors.get_shape().as_list()[1]
 684     else:
 685       # If prior path does not exist, do not use priors, i.e., pirors equal to
 686       # uniform empty 32x32 patch.
 687       self._num_clusters = 1
 688       priors = tf.zeros([self._mask_num_classes, self._num_clusters,
 689                          self._mask_crop_size, self._mask_crop_size])
 690     return priors
 691 
 692   def _classify_shape_priors(self, features, uniform_priors, classes):
 693     """Classify the uniform prior by predicting the shape modes.
 694 
 695     Classify the object crop features into K modes of the clusters for each
 696     category.
 697 
 698     Args:
 699       features: A float Tensor of shape [batch_size, num_instances,
 700         mask_size, mask_size, num_channels].
 701       uniform_priors: A float Tensor of shape [batch_size, num_instances,
 702         mask_size, mask_size] representing the uniform detection priors.
 703       classes: A int Tensor of shape [batch_size, num_instances]
 704         of detection class ids.
 705 
 706     Returns:
 707       prior_distribution: A float Tensor of shape
 708         [batch_size, num_instances, num_clusters] representing the classifier
 709         output probability over all possible shapes.
 710     """
 711 
 712     batch_size, num_instances, _, _, _ = features.get_shape().as_list()
 713     features *= tf.expand_dims(uniform_priors, axis=-1)
 714     # Reduce spatial dimension of features. The features have shape
 715     # [batch_size, num_instances, num_channels].
 716     features = tf.reduce_mean(features, axis=(2, 3))
 717     logits = tf.keras.layers.Dense(
 718         self._mask_num_classes * self._num_clusters,
 719         kernel_initializer=tf.random_normal_initializer(stddev=0.01))(features)
 720     logits = tf.reshape(logits,
 721                         [batch_size, num_instances,
 722                          self._mask_num_classes, self._num_clusters])
 723     if self._use_category_for_mask:
 724       logits = tf.gather(logits, tf.expand_dims(classes, axis=-1), batch_dims=2)
 725       logits = tf.squeeze(logits, axis=2)
 726     else:
 727       logits = logits[:, :, 0, :]
 728 
 729     distribution = tf.nn.softmax(logits, name='shape_prior_weights')
 730     return distribution

 736   def __init__(self,
 737                num_classes,

 739                mask_crop_size,
 740                use_category_for_mask,
 741                num_convs,
 742                norm_activation=nn_ops.norm_activation_builder()):
 743     """Initialize params to build ShapeMask coarse and fine prediction head.
 744 
 745     Args:
 746       num_classes: `int` number of mask classification categories.
 747       num_downsample_channels: `int` number of filters at mask head.
 748       mask_crop_size: feature crop size.
 749       use_category_for_mask: use class information in mask branch.
 750       num_convs: `int` number of stacked convolution before the last prediction
 751         layer.
 752       norm_activation: an operation that includes a normalization layer
 753         followed by an optional activation layer.
 754     """
 755     self._mask_num_classes = num_classes if use_category_for_mask else 1
 756     self._use_category_for_mask = use_category_for_mask
 757     self._num_downsample_channels = num_downsample_channels
 758     self._mask_crop_size = mask_crop_size
 759     self._num_convs = num_convs
 760     self._norm_activation = norm_activation
 761 
 762     self._coarse_mask_fc = tf.keras.layers.Dense(
 763         self._num_downsample_channels, name='coarse-mask-fc')
 764 
 765     self._class_conv = []
 766     self._class_norm_activation = []
 767 
 768     for i in range(self._num_convs):
 769       self._class_conv.append(tf.keras.layers.Conv2D(
 770           self._num_downsample_channels,
 771           kernel_size=(3, 3),
 772           bias_initializer=tf.zeros_initializer(),
 773           kernel_initializer=tf.keras.initializers.RandomNormal(stddev=0.01),
 774           padding='same',
 775           name='coarse-mask-class-%d' % i))
 776 
 777       self._class_norm_activation.append(
 778           norm_activation(name='coarse-mask-class-%d-bn' % i))
 779 
 780     self._class_predict = tf.keras.layers.Conv2D(
 781         self._mask_num_classes,
 782         kernel_size=(1, 1),
 783         # Focal loss bias initialization to have foreground 0.01 probability.
 784         bias_initializer=tf.constant_initializer(-np.log((1 - 0.01) / 0.01)),
 785         kernel_initializer=tf.keras.initializers.RandomNormal(stddev=0.01),
 786         padding='same',
 787         name='coarse-mask-class-predict')
 788 
 789   def __call__(self, features, detection_priors, classes, is_training):
 790     """Generate instance masks from FPN features and detection priors.
 791 
 792     This corresponds to the Fig. 5-6 of the ShapeMask paper at
 793     https://arxiv.org/pdf/1904.03239.pdf
 794 
 795     Args:
 796       features: a float Tensor of shape [batch_size, num_instances,
 797         mask_crop_size, mask_crop_size, num_downsample_channels]. This is the
 798         instance feature crop.
 799       detection_priors: a float Tensor of shape [batch_size, num_instances,
 800         mask_crop_size, mask_crop_size, 1]. This is the detection prior for
 801         the instance.
 802       classes: a int Tensor of shape [batch_size, num_instances]
 803         of instance classes.
 804       is_training: a bool indicating whether in training mode.
 805 
 806     Returns:
 807       mask_outputs: instance mask prediction as a float Tensor of shape
 808         [batch_size, num_instances, mask_size, mask_size].
 809     """
 810     with backend.get_graph().as_default(), tf.name_scope('coarse_mask'):
 811       # Transform detection priors to have the same dimension as features.
 812       detection_priors = tf.expand_dims(detection_priors, axis=-1)
 813       detection_priors = self._coarse_mask_fc(detection_priors)
 814 
 815       features += detection_priors
 816       mask_logits = self.decoder_net(features, is_training)
 817       # Gather the logits with right input class.
 818       if self._use_category_for_mask:
 819         mask_logits = tf.transpose(mask_logits, [0, 1, 4, 2, 3])
 820         mask_logits = tf.gather(mask_logits, tf.expand_dims(classes, -1),
 821                                 batch_dims=2)
 822         mask_logits = tf.squeeze(mask_logits, axis=2)
 823       else:
 824         mask_logits = mask_logits[..., 0]
 825 
 826       return mask_logits
 827 
 828   def decoder_net(self, features, is_training=False):
 829     """Coarse mask decoder network architecture.
 830 
 831     Args:
 832       features: A tensor of size [batch, height_in, width_in, channels_in].
 833       is_training: Whether batch_norm layers are in training mode.
 834 
 835     Returns:
 836       images: A feature tensor of size [batch, output_size, output_size,
 837         num_channels]
 838     """
 839     (batch_size, num_instances, height, width,
 840      num_channels) = features.get_shape().as_list()
 841     features = tf.reshape(features, [batch_size * num_instances, height, width,
 842                                      num_channels])
 843     for i in range(self._num_convs):
 844       features = self._class_conv[i](features)
 845       features = self._class_norm_activation[i](features,
 846                                                 is_training=is_training)
 847 
 848     mask_logits = self._class_predict(features)
 849     mask_logits = tf.reshape(mask_logits, [batch_size, num_instances, height,
 850                                            width, self._mask_num_classes])
 851     return mask_logits

 860                mask_crop_size,
 861                use_category_for_mask,
 862                num_convs,
 863                upsample_factor,
 864                norm_activation=nn_ops.norm_activation_builder()):

 867     Args:
 868       num_classes: `int` number of mask classification categories.
 869       num_downsample_channels: `int` number of filters at mask head.
 870       mask_crop_size: feature crop size.
 871       use_category_for_mask: use class information in mask branch.
 872       num_convs: `int` number of stacked convolution before the last prediction
 873         layer.
 874       upsample_factor: `int` number of fine mask upsampling factor.
 875       norm_activation: an operation that includes a batch normalization layer
 876         followed by a relu layer(optional).
 877     """
 878     self._use_category_for_mask = use_category_for_mask
 879     self._mask_num_classes = num_classes if use_category_for_mask else 1

 884 
 885     self._fine_mask_fc = tf.keras.layers.Dense(
 886         self._num_downsample_channels, name='fine-mask-fc')
 887 
 888     self._upsample_conv = tf.keras.layers.Conv2DTranspose(
 889         self._num_downsample_channels,
 890         (self.up_sample_factor, self.up_sample_factor),
 891         (self.up_sample_factor, self.up_sample_factor),
 892         name='fine-mask-conv2d-tran')
 893 
 894     self._fine_class_conv = []
 895     self._fine_class_bn = []
 896     for i in range(self._num_convs):
 897       self._fine_class_conv.append(
 898           tf.keras.layers.Conv2D(
 899               self._num_downsample_channels,
 900               kernel_size=(3, 3),
 901               bias_initializer=tf.zeros_initializer(),
 902               kernel_initializer=tf.keras.initializers.RandomNormal(
 903                   stddev=0.01),
 904               activation=None,
 905               padding='same',
 906               name='fine-mask-class-%d' % i))
 907       self._fine_class_bn.append(norm_activation(
 908           name='fine-mask-class-%d-bn' % i))
 909 
 910     self._class_predict_conv = tf.keras.layers.Conv2D(
 911         self._mask_num_classes,
 912         kernel_size=(1, 1),
 913         # Focal loss bias initialization to have foreground 0.01 probability.
 914         bias_initializer=tf.constant_initializer(-np.log((1 - 0.01) / 0.01)),
 915         kernel_initializer=tf.keras.initializers.RandomNormal(stddev=0.01),
 916         padding='same',
 917         name='fine-mask-class-predict')
 918 
 919   def __call__(self, features, mask_logits, classes, is_training):
 920     """Generate instance masks from FPN features and detection priors.
 921 
 922     This corresponds to the Fig. 5-6 of the ShapeMask paper at
 923     https://arxiv.org/pdf/1904.03239.pdf
 924 
 925     Args:
 926       features: a float Tensor of shape
 927         [batch_size, num_instances, mask_crop_size, mask_crop_size,
 928         num_downsample_channels]. This is the instance feature crop.
 929       mask_logits: a float Tensor of shape
 930         [batch_size, num_instances, mask_crop_size, mask_crop_size] indicating
 931         predicted mask logits.
 932       classes: a int Tensor of shape [batch_size, num_instances]
 933         of instance classes.
 934       is_training: a bool indicating whether in training mode.
 935 
 936     Returns:
 937       mask_outputs: instance mask prediction as a float Tensor of shape
 938         [batch_size, num_instances, mask_size, mask_size].
 939     """
 940     # Extract the foreground mean features
 941     # with tf.variable_scope('fine_mask', reuse=tf.AUTO_REUSE):
 942     with backend.get_graph().as_default(), tf.name_scope('fine_mask'):
 943       mask_probs = tf.nn.sigmoid(mask_logits)
 944       # Compute instance embedding for hard average.
 945       binary_mask = tf.cast(tf.greater(mask_probs, 0.5), features.dtype)
 946       instance_embedding = tf.reduce_sum(
 947           features * tf.expand_dims(binary_mask, axis=-1), axis=(2, 3))
 948       instance_embedding /= tf.expand_dims(
 949           tf.reduce_sum(binary_mask, axis=(2, 3)) + 1e-20, axis=-1)
 950       # Take the difference between crop features and mean instance features.
 951       features -= tf.expand_dims(
 952           tf.expand_dims(instance_embedding, axis=2), axis=2)
 953 
 954       features += self._fine_mask_fc(tf.expand_dims(mask_probs, axis=-1))
 955 
 956       # Decoder to generate upsampled segmentation mask.
 957       mask_logits = self.decoder_net(features, is_training)
 958       if self._use_category_for_mask:
 959         mask_logits = tf.transpose(mask_logits, [0, 1, 4, 2, 3])
 960         mask_logits = tf.gather(mask_logits,
 961                                 tf.expand_dims(classes, -1), batch_dims=2)
 962         mask_logits = tf.squeeze(mask_logits, axis=2)
 963       else:
 964         mask_logits = mask_logits[..., 0]
 965 
 966     return mask_logits
 967 
 968   def decoder_net(self, features, is_training=False):
 969     """Fine mask decoder network architecture.
 970 
 971     Args:
 972       features: A tensor of size [batch, height_in, width_in, channels_in].
 973       is_training: Whether batch_norm layers are in training mode.
 974 
 975     Returns:
 976       images: A feature tensor of size [batch, output_size, output_size,
 977         num_channels], where output size is self._gt_upsample_scale times
 978         that of input.
 979     """
 980     (batch_size, num_instances, height, width,
 981      num_channels) = features.get_shape().as_list()
 982     features = tf.reshape(features, [batch_size * num_instances, height, width,
 983                                      num_channels])
 984     for i in range(self._num_convs):
 985       features = self._fine_class_conv[i](features)
 986       features = self._fine_class_bn[i](features, is_training=is_training)
 987 
 988     if self.up_sample_factor > 1:
 989       features = self._upsample_conv(features)
 990 
 991     # Predict per-class instance masks.
 992     mask_logits = self._class_predict_conv(features)
 993 
 994     mask_logits = tf.reshape(mask_logits,
 995                              [batch_size, num_instances,
 996                               height * self.up_sample_factor,
 997                               width * self.up_sample_factor,
 998                               self._mask_num_classes])
 999     return mask_logits
