  32 class RetinanetModel(base_model.Model):
  33   """RetinaNet model function."""
  34 
  35   def __init__(self, params):
  36     super(RetinanetModel, self).__init__(params)
  37 
  38     # For eval metrics.
  39     self._params = params
  40 
  41     # Architecture generators.
  42     self._backbone_fn = factory.backbone_generator(params)
  43     self._fpn_fn = factory.multilevel_features_generator(params)
  44     self._head_fn = factory.retinanet_head_generator(params)
  45 
  46     # Loss function.
  47     self._cls_loss_fn = losses.RetinanetClassLoss(
  48         params.retinanet_loss, params.architecture.num_classes)
  49     self._box_loss_fn = losses.RetinanetBoxLoss(params.retinanet_loss)
  50     self._box_loss_weight = params.retinanet_loss.box_loss_weight
  51     self._keras_model = None
  52 
  53     # Predict function.
  54     self._generate_detections_fn = postprocess_ops.MultilevelDetectionGenerator(
  55         params.architecture.min_level,
  56         params.architecture.max_level,
  57         params.postprocess)
  58 
  59     self._transpose_input = params.train.transpose_input
  60     assert not self._transpose_input, 'Transpose input is not supportted.'
  61     # Input layer.
  62     input_shape = (
  63         params.retinanet_parser.output_size +
  64         [params.retinanet_parser.num_channels])
  65     self._input_layer = tf.keras.layers.Input(
  66         shape=input_shape, name='',
  67         dtype=tf.bfloat16 if self._use_bfloat16 else tf.float32)
  68 
  69   def build_outputs(self, inputs, mode):
  70     # If the input image is transposed (from NHWC to HWCN), we need to revert it
  71     # back to the original shape before it's used in the computation.
  72     if self._transpose_input:
  73       inputs = tf.transpose(inputs, [3, 0, 1, 2])
  74 
  75     backbone_features = self._backbone_fn(
  76         inputs, is_training=(mode == mode_keys.TRAIN))
  77     fpn_features = self._fpn_fn(
  78         backbone_features, is_training=(mode == mode_keys.TRAIN))
  79     cls_outputs, box_outputs = self._head_fn(
  80         fpn_features, is_training=(mode == mode_keys.TRAIN))
  81 
  82     if self._use_bfloat16:
  83       levels = cls_outputs.keys()
  84       for level in levels:
  85         cls_outputs[level] = tf.cast(cls_outputs[level], tf.float32)
  86         box_outputs[level] = tf.cast(box_outputs[level], tf.float32)
  87 
  88     model_outputs = {
  89         'cls_outputs': cls_outputs,
  90         'box_outputs': box_outputs,
  91     }
  92     return model_outputs
  93 
  94   def build_loss_fn(self):
  95     if self._keras_model is None:
  96       raise ValueError('build_loss_fn() must be called after build_model().')
  97 
  98     filter_fn = self.make_filter_trainable_variables_fn()
  99     trainable_variables = filter_fn(self._keras_model.trainable_variables)
 100 
 101     def _total_loss_fn(labels, outputs):
 102       cls_loss = self._cls_loss_fn(outputs['cls_outputs'],
 103                                    labels['cls_targets'],
 104                                    labels['num_positives'])
 105       box_loss = self._box_loss_fn(outputs['box_outputs'],
 106                                    labels['box_targets'],
 107                                    labels['num_positives'])
 108       model_loss = cls_loss + self._box_loss_weight * box_loss
 109       l2_regularization_loss = self.weight_decay_loss(trainable_variables)
 110       total_loss = model_loss + l2_regularization_loss
 111       return {
 112           'total_loss': total_loss,
 113           'cls_loss': cls_loss,
 114           'box_loss': box_loss,
 115           'model_loss': model_loss,
 116           'l2_regularization_loss': l2_regularization_loss,
 117       }
 118 
 119     return _total_loss_fn
 120 
 121   def build_model(self, params, mode=None):
 122     if self._keras_model is None:
 123       with backend.get_graph().as_default():
 124         outputs = self.model_outputs(self._input_layer, mode)
 125 
 126         model = tf.keras.models.Model(
 127             inputs=self._input_layer, outputs=outputs, name='retinanet')
 128         assert model is not None, 'Fail to build tf.keras.Model.'
 129         model.optimizer = self.build_optimizer()
 130         self._keras_model = model
 131 
 132     return self._keras_model
 133 
 134   def post_processing(self, labels, outputs):
 135     # TODO(yeqing): Moves the output related part into build_outputs.
 136     required_output_fields = ['cls_outputs', 'box_outputs']
 137     for field in required_output_fields:
 138       if field not in outputs:
 139         raise ValueError('"%s" is missing in outputs, requried %s found %s',
 140                          field, required_output_fields, outputs.keys())
 141     required_label_fields = ['image_info', 'groundtruths']
 142     for field in required_label_fields:
 143       if field not in labels:
 144         raise ValueError('"%s" is missing in outputs, requried %s found %s',
 145                          field, required_label_fields, labels.keys())
 146     boxes, scores, classes, valid_detections = self._generate_detections_fn(
 147         outputs['box_outputs'], outputs['cls_outputs'],
 148         labels['anchor_boxes'], labels['image_info'][:, 1:2, :])
 149     # Discards the old output tensors to save memory. The `cls_outputs` and
 150     # `box_outputs` are pretty big and could potentiall lead to memory issue.
 151     outputs = {
 152         'source_id': labels['groundtruths']['source_id'],
 153         'image_info': labels['image_info'],
 154         'num_detections': valid_detections,
 155         'detection_boxes': boxes,
 156         'detection_classes': classes,
 157         'detection_scores': scores,
 158     }
 159 
 160     if 'groundtruths' in labels:
 161       labels['source_id'] = labels['groundtruths']['source_id']
 162       labels['boxes'] = labels['groundtruths']['boxes']
 163       labels['classes'] = labels['groundtruths']['classes']
 164       labels['areas'] = labels['groundtruths']['areas']
 165       labels['is_crowds'] = labels['groundtruths']['is_crowds']
 166 
 167     return labels, outputs
 168 
 169   def eval_metrics(self):
 170     return eval_factory.evaluator_generator(self._params.eval)
