  25 def focal_loss(logits, targets, alpha, gamma, normalizer):
  26   """Compute the focal loss between `logits` and the golden `target` values.
  27 
  28   Focal loss = -(1-pt)^gamma * log(pt)
  29   where pt is the probability of being classified to the true class.
  30 
  31   Args:
  32     logits: A float32 tensor of size
  33       [batch, height_in, width_in, num_predictions].
  34     targets: A float32 tensor of size
  35       [batch, height_in, width_in, num_predictions].
  36     alpha: A float32 scalar multiplying alpha to the loss from positive examples
  37       and (1-alpha) to the loss from negative examples.
  38     gamma: A float32 scalar modulating loss from hard and easy examples.
  39     normalizer: A float32 scalar normalizes the total loss from all examples.
  40 
  41   Returns:
  42     loss: A float32 Tensor of size [batch, height_in, width_in, num_predictions]
  43       representing normalized loss on the prediction map.
  44   """
  45   with tf.name_scope('focal_loss'):
  46     positive_label_mask = tf.math.equal(targets, 1.0)
  47     cross_entropy = (
  48         tf.nn.sigmoid_cross_entropy_with_logits(labels=targets, logits=logits))
  49     # Below are comments/derivations for computing modulator.
  50     # For brevity, let x = logits,  z = targets, r = gamma, and p_t = sigmod(x)
  51     # for positive samples and 1 - sigmoid(x) for negative examples.
  52     #
  53     # The modulator, defined as (1 - P_t)^r, is a critical part in focal loss
  54     # computation. For r > 0, it puts more weights on hard examples, and less
  55     # weights on easier ones. However if it is directly computed as (1 - P_t)^r,
  56     # its back-propagation is not stable when r < 1. The implementation here
  57     # resolves the issue.
  58     #
  59     # For positive samples (labels being 1),
  60     #    (1 - p_t)^r
  61     #  = (1 - sigmoid(x))^r
  62     #  = (1 - (1 / (1 + exp(-x))))^r
  63     #  = (exp(-x) / (1 + exp(-x)))^r
  64     #  = exp(log((exp(-x) / (1 + exp(-x)))^r))
  65     #  = exp(r * log(exp(-x)) - r * log(1 + exp(-x)))
  66     #  = exp(- r * x - r * log(1 + exp(-x)))
  67     #
  68     # For negative samples (labels being 0),
  69     #    (1 - p_t)^r
  70     #  = (sigmoid(x))^r
  71     #  = (1 / (1 + exp(-x)))^r
  72     #  = exp(log((1 / (1 + exp(-x)))^r))
  73     #  = exp(-r * log(1 + exp(-x)))
  74     #
  75     # Therefore one unified form for positive (z = 1) and negative (z = 0)
  76     # samples is:
  77     #      (1 - p_t)^r = exp(-r * z * x - r * log(1 + exp(-x))).
  78     neg_logits = -1.0 * logits
  79     modulator = tf.math.exp(gamma * targets * neg_logits -
  80                             gamma * tf.math.log1p(tf.math.exp(neg_logits)))
  81     loss = modulator * cross_entropy
  82     weighted_loss = tf.where(positive_label_mask, alpha * loss,
  83                              (1.0 - alpha) * loss)
  84     weighted_loss /= normalizer
  85   return weighted_loss

  88 class RpnScoreLoss(object):
  89   """Region Proposal Network score loss function."""
  90 
  91   def __init__(self, params):
  92     self._rpn_batch_size_per_im = params.rpn_batch_size_per_im
  93     self._binary_crossentropy = tf.keras.losses.BinaryCrossentropy(
  94         reduction=tf.keras.losses.Reduction.SUM, from_logits=True)
  95 
  96   def __call__(self, score_outputs, labels):
  97     """Computes total RPN detection loss.
  98 
  99     Computes total RPN detection loss including box and score from all levels.
 100 
 101     Args:
 102       score_outputs: an OrderDict with keys representing levels and values
 103         representing scores in [batch_size, height, width, num_anchors].
 104       labels: the dictionary that returned from dataloader that includes
 105         groundturth targets.
 106 
 107     Returns:
 108       rpn_score_loss: a scalar tensor representing total score loss.
 109     """
 110     with tf.name_scope('rpn_loss'):
 111       levels = sorted(score_outputs.keys())
 112 
 113       score_losses = []
 114       for level in levels:
 115         score_losses.append(
 116             self._rpn_score_loss(
 117                 score_outputs[level],
 118                 labels[level],
 119                 normalizer=tf.cast(
 120                     tf.shape(score_outputs[level])[0] *
 121                     self._rpn_batch_size_per_im, dtype=tf.float32)))
 122 
 123       # Sums per level losses to total loss.
 124       return tf.math.add_n(score_losses)
 125 
 126   def _rpn_score_loss(self, score_outputs, score_targets, normalizer=1.0):
 127     """Computes score loss."""
 128     # score_targets has three values:
 129     # (1) score_targets[i]=1, the anchor is a positive sample.
 130     # (2) score_targets[i]=0, negative.
 131     # (3) score_targets[i]=-1, the anchor is don't care (ignore).
 132     with tf.name_scope('rpn_score_loss'):
 133       mask = tf.math.logical_or(tf.math.equal(score_targets, 1),
 134                                 tf.math.equal(score_targets, 0))
 135 
 136       score_targets = tf.math.maximum(score_targets,
 137                                       tf.zeros_like(score_targets))
 138 
 139       score_targets = tf.expand_dims(score_targets, axis=-1)
 140       score_outputs = tf.expand_dims(score_outputs, axis=-1)
 141       score_loss = self._binary_crossentropy(
 142           score_targets, score_outputs, sample_weight=mask)
 143 
 144       score_loss /= normalizer
 145       return score_loss

 148 class RpnBoxLoss(object):
 149   """Region Proposal Network box regression loss function."""
 150 
 151   def __init__(self, params):
 152     logging.info('RpnBoxLoss huber_loss_delta %s', params.huber_loss_delta)
 153     # The delta is typically around the mean value of regression target.
 154     # for instances, the regression targets of 512x512 input with 6 anchors on
 155     # P2-P6 pyramid is about [0.1, 0.1, 0.2, 0.2].
 156     self._huber_loss = tf.keras.losses.Huber(
 157         delta=params.huber_loss_delta, reduction=tf.keras.losses.Reduction.SUM)
 158 
 159   def __call__(self, box_outputs, labels):
 160     """Computes total RPN detection loss.
 161 
 162     Computes total RPN detection loss including box and score from all levels.
 163 
 164     Args:
 165       box_outputs: an OrderDict with keys representing levels and values
 166         representing box regression targets in
 167         [batch_size, height, width, num_anchors * 4].
 168       labels: the dictionary that returned from dataloader that includes
 169         groundturth targets.
 170 
 171     Returns:
 172       rpn_box_loss: a scalar tensor representing total box regression loss.
 173     """
 174     with tf.name_scope('rpn_loss'):
 175       levels = sorted(box_outputs.keys())
 176 
 177       box_losses = []
 178       for level in levels:
 179         box_losses.append(self._rpn_box_loss(box_outputs[level], labels[level]))
 180 
 181       # Sum per level losses to total loss.
 182       return tf.add_n(box_losses)
 183 
 184   def _rpn_box_loss(self, box_outputs, box_targets, normalizer=1.0):
 185     """Computes box regression loss."""
 186     with tf.name_scope('rpn_box_loss'):
 187       mask = tf.cast(tf.not_equal(box_targets, 0.0), dtype=tf.float32)
 188       box_targets = tf.expand_dims(box_targets, axis=-1)
 189       box_outputs = tf.expand_dims(box_outputs, axis=-1)
 190       box_loss = self._huber_loss(box_targets, box_outputs, sample_weight=mask)
 191       # The loss is normalized by the sum of non-zero weights and additional
 192       # normalizer provided by the function caller. Using + 0.01 here to avoid
 193       # division by zero.
 194       box_loss /= normalizer * (tf.reduce_sum(mask) + 0.01)
 195       return box_loss

 198 class FastrcnnClassLoss(object):
 199   """Fast R-CNN classification loss function."""
 200 
 201   def __init__(self):
 202     self._categorical_crossentropy = tf.keras.losses.CategoricalCrossentropy(
 203         reduction=tf.keras.losses.Reduction.SUM, from_logits=True)
 204 
 205   def __call__(self, class_outputs, class_targets):
 206     """Computes the class loss (Fast-RCNN branch) of Mask-RCNN.
 207 
 208     This function implements the classification loss of the Fast-RCNN.
 209 
 210     The classification loss is softmax on all RoIs.
 211     Reference: https://github.com/facebookresearch/Detectron/blob/master/detectron/modeling/fast_rcnn_heads.py  # pylint: disable=line-too-long
 212 
 213     Args:
 214       class_outputs: a float tensor representing the class prediction for each box
 215         with a shape of [batch_size, num_boxes, num_classes].
 216       class_targets: a float tensor representing the class label for each box
 217         with a shape of [batch_size, num_boxes].
 218 
 219     Returns:
 220       a scalar tensor representing total class loss.
 221     """
 222     with tf.name_scope('fast_rcnn_loss'):
 223       batch_size, num_boxes, num_classes = class_outputs.get_shape().as_list()
 224       class_targets = tf.cast(class_targets, dtype=tf.int32)
 225       class_targets_one_hot = tf.one_hot(class_targets, num_classes)
 226       return self._fast_rcnn_class_loss(class_outputs, class_targets_one_hot,
 227                                         normalizer=batch_size * num_boxes / 2.0)
 228 
 229   def _fast_rcnn_class_loss(self, class_outputs, class_targets_one_hot,
 230                             normalizer):
 231     """Computes classification loss."""
 232     with tf.name_scope('fast_rcnn_class_loss'):
 233       class_loss = self._categorical_crossentropy(class_targets_one_hot,
 234                                                   class_outputs)
 235 
 236       class_loss /= normalizer
 237       return class_loss

 240 class FastrcnnBoxLoss(object):
 241   """Fast R-CNN box regression loss function."""
 242 
 243   def __init__(self, params):
 244     logging.info('FastrcnnBoxLoss huber_loss_delta %s', params.huber_loss_delta)
 245     # The delta is typically around the mean value of regression target.
 246     # for instances, the regression targets of 512x512 input with 6 anchors on
 247     # P2-P6 pyramid is about [0.1, 0.1, 0.2, 0.2].
 248     self._huber_loss = tf.keras.losses.Huber(
 249         delta=params.huber_loss_delta, reduction=tf.keras.losses.Reduction.SUM)
 250 
 251   def __call__(self, box_outputs, class_targets, box_targets):
 252     """Computes the box loss (Fast-RCNN branch) of Mask-RCNN.
 253 
 254     This function implements the box regression loss of the Fast-RCNN. As the
 255     `box_outputs` produces `num_classes` boxes for each RoI, the reference model
 256     expands `box_targets` to match the shape of `box_outputs` and selects only
 257     the target that the RoI has a maximum overlap. (Reference: https://github.com/facebookresearch/Detectron/blob/master/detectron/roi_data/fast_rcnn.py)  # pylint: disable=line-too-long
 258     Instead, this function selects the `box_outputs` by the `class_targets` so
 259     that it doesn't expand `box_targets`.
 260 
 261     The box loss is smooth L1-loss on only positive samples of RoIs.
 262     Reference: https://github.com/facebookresearch/Detectron/blob/master/detectron/modeling/fast_rcnn_heads.py  # pylint: disable=line-too-long
 263 
 264     Args:
 265       box_outputs: a float tensor representing the box prediction for each box
 266         with a shape of [batch_size, num_boxes, num_classes * 4].
 267       class_targets: a float tensor representing the class label for each box
 268         with a shape of [batch_size, num_boxes].
 269       box_targets: a float tensor representing the box label for each box
 270         with a shape of [batch_size, num_boxes, 4].
 271 
 272     Returns:
 273       box_loss: a scalar tensor representing total box regression loss.
 274     """
 275     with tf.name_scope('fast_rcnn_loss'):
 276       class_targets = tf.cast(class_targets, dtype=tf.int32)
 277 
 278       # Selects the box from `box_outputs` based on `class_targets`, with which
 279       # the box has the maximum overlap.
 280       (batch_size, num_rois,
 281        num_class_specific_boxes) = box_outputs.get_shape().as_list()
 282       num_classes = num_class_specific_boxes // 4
 283       box_outputs = tf.reshape(box_outputs,
 284                                [batch_size, num_rois, num_classes, 4])
 285 
 286       box_indices = tf.reshape(
 287           class_targets + tf.tile(
 288               tf.expand_dims(
 289                   tf.range(batch_size) * num_rois * num_classes, 1),
 290               [1, num_rois]) + tf.tile(
 291                   tf.expand_dims(tf.range(num_rois) * num_classes, 0),
 292                   [batch_size, 1]), [-1])
 293 
 294       box_outputs = tf.matmul(
 295           tf.one_hot(
 296               box_indices,
 297               batch_size * num_rois * num_classes,
 298               dtype=box_outputs.dtype), tf.reshape(box_outputs, [-1, 4]))
 299       box_outputs = tf.reshape(box_outputs, [batch_size, -1, 4])
 300 
 301       return self._fast_rcnn_box_loss(box_outputs, box_targets, class_targets)
 302 
 303   def _fast_rcnn_box_loss(self, box_outputs, box_targets, class_targets,
 304                           normalizer=1.0):
 305     """Computes box regression loss."""
 306     with tf.name_scope('fast_rcnn_box_loss'):
 307       mask = tf.tile(tf.expand_dims(tf.greater(class_targets, 0), axis=2),
 308                      [1, 1, 4])
 309       mask = tf.cast(mask, dtype=tf.float32)
 310       box_targets = tf.expand_dims(box_targets, axis=-1)
 311       box_outputs = tf.expand_dims(box_outputs, axis=-1)
 312       box_loss = self._huber_loss(box_targets, box_outputs, sample_weight=mask)
 313       # The loss is normalized by the number of ones in mask,
 314       # additianal normalizer provided by the user and using 0.01 here to avoid
 315       # division by 0.
 316       box_loss /= normalizer * (tf.reduce_sum(mask) + 0.01)
 317       return box_loss

 320 class MaskrcnnLoss(object):
 321   """Mask R-CNN instance segmentation mask loss function."""
 322 
 323   def __init__(self):
 324     self._binary_crossentropy = tf.keras.losses.BinaryCrossentropy(
 325         reduction=tf.keras.losses.Reduction.SUM, from_logits=True)
 326 
 327   def __call__(self, mask_outputs, mask_targets, select_class_targets):
 328     """Computes the mask loss of Mask-RCNN.
 329 
 330     This function implements the mask loss of Mask-RCNN. As the `mask_outputs`
 331     produces `num_classes` masks for each RoI, the reference model expands
 332     `mask_targets` to match the shape of `mask_outputs` and selects only the
 333     target that the RoI has a maximum overlap. (Reference: https://github.com/facebookresearch/Detectron/blob/master/detectron/roi_data/mask_rcnn.py)  # pylint: disable=line-too-long
 334     Instead, this implementation selects the `mask_outputs` by the `class_targets`
 335     so that it doesn't expand `mask_targets`. Note that the selection logic is
 336     done in the post-processing of mask_rcnn_fn in mask_rcnn_architecture.py.
 337 
 338     Args:
 339       mask_outputs: a float tensor representing the prediction for each mask,
 340         with a shape of
 341         [batch_size, num_masks, mask_height, mask_width].
 342       mask_targets: a float tensor representing the binary mask of ground truth
 343         labels for each mask with a shape of
 344         [batch_size, num_masks, mask_height, mask_width].
 345       select_class_targets: a tensor with a shape of [batch_size, num_masks],
 346         representing the foreground mask targets.
 347 
 348     Returns:
 349       mask_loss: a float tensor representing total mask loss.
 350     """
 351     with tf.name_scope('mask_rcnn_loss'):
 352       (batch_size, num_masks, mask_height,
 353        mask_width) = mask_outputs.get_shape().as_list()
 354 
 355       weights = tf.tile(
 356           tf.reshape(tf.greater(select_class_targets, 0),
 357                      [batch_size, num_masks, 1, 1]),
 358           [1, 1, mask_height, mask_width])
 359       weights = tf.cast(weights, dtype=tf.float32)
 360 
 361       mask_targets = tf.expand_dims(mask_targets, axis=-1)
 362       mask_outputs = tf.expand_dims(mask_outputs, axis=-1)
 363       mask_loss = self._binary_crossentropy(mask_targets, mask_outputs,
 364                                             sample_weight=weights)
 365 
 366       # The loss is normalized by the number of 1's in weights and
 367       # + 0.01 is used to avoid division by zero.
 368       return mask_loss / (tf.reduce_sum(weights) + 0.01)

 371 class RetinanetClassLoss(object):
 372   """RetinaNet class loss."""
 373 
 374   def __init__(self, params, num_classes):
 375     self._num_classes = num_classes
 376     self._focal_loss_alpha = params.focal_loss_alpha
 377     self._focal_loss_gamma = params.focal_loss_gamma
 378 
 379   def __call__(self, cls_outputs, labels, num_positives):
 380     """Computes total detection loss.
 381 
 382     Computes total detection loss including box and class loss from all levels.
 383 
 384     Args:
 385       cls_outputs: an OrderDict with keys representing levels and values
 386         representing logits in [batch_size, height, width,
 387         num_anchors * num_classes].
 388       labels: the dictionary that returned from dataloader that includes
 389         class groundturth targets.
 390       num_positives: number of positive examples in the minibatch.
 391 
 392     Returns:
 393       an integar tensor representing total class loss.
 394     """
 395     # Sums all positives in a batch for normalization and avoids zero
 396     # num_positives_sum, which would lead to inf loss during training
 397     num_positives_sum = tf.reduce_sum(input_tensor=num_positives) + 1.0
 398 
 399     cls_losses = []
 400     for level in cls_outputs.keys():
 401       cls_losses.append(self.class_loss(
 402           cls_outputs[level], labels[level], num_positives_sum))
 403     # Sums per level losses to total loss.
 404     return tf.add_n(cls_losses)
 405 
 406   def class_loss(self, cls_outputs, cls_targets, num_positives,
 407                  ignore_label=-2):
 408     """Computes RetinaNet classification loss."""
 409     # Onehot encoding for classification labels.
 410     cls_targets_one_hot = tf.one_hot(cls_targets, self._num_classes)
 411     bs, height, width, _, _ = cls_targets_one_hot.get_shape().as_list()
 412     cls_targets_one_hot = tf.reshape(cls_targets_one_hot,
 413                                      [bs, height, width, -1])
 414     loss = focal_loss(tf.cast(cls_outputs, dtype=tf.float32),
 415                       tf.cast(cls_targets_one_hot, dtype=tf.float32),
 416                       self._focal_loss_alpha,
 417                       self._focal_loss_gamma,
 418                       num_positives)
 419 
 420     ignore_loss = tf.where(
 421         tf.equal(cls_targets, ignore_label),
 422         tf.zeros_like(cls_targets, dtype=tf.float32),
 423         tf.ones_like(cls_targets, dtype=tf.float32),
 424     )
 425     ignore_loss = tf.expand_dims(ignore_loss, -1)
 426     ignore_loss = tf.tile(ignore_loss, [1, 1, 1, 1, self._num_classes])
 427     ignore_loss = tf.reshape(ignore_loss, tf.shape(input=loss))
 428     return tf.reduce_sum(input_tensor=ignore_loss * loss)

 431 class RetinanetBoxLoss(object):
 432   """RetinaNet box loss."""
 433 
 434   def __init__(self, params):
 435     self._huber_loss = tf.keras.losses.Huber(
 436         delta=params.huber_loss_delta, reduction=tf.keras.losses.Reduction.SUM)
 437 
 438   def __call__(self, box_outputs, labels, num_positives):
 439     """Computes box detection loss.
 440 
 441     Computes total detection loss including box and class loss from all levels.
 442 
 443     Args:
 444       box_outputs: an OrderDict with keys representing levels and values
 445         representing box regression targets in [batch_size, height, width,
 446         num_anchors * 4].
 447       labels: the dictionary that returned from dataloader that includes
 448         box groundturth targets.
 449       num_positives: number of positive examples in the minibatch.
 450 
 451     Returns:
 452       an integar tensor representing total box regression loss.
 453     """
 454     # Sums all positives in a batch for normalization and avoids zero
 455     # num_positives_sum, which would lead to inf loss during training
 456     num_positives_sum = tf.reduce_sum(input_tensor=num_positives) + 1.0
 457 
 458     box_losses = []
 459     for level in box_outputs.keys():
 460       # Onehot encoding for classification labels.
 461       box_targets_l = labels[level]
 462       box_losses.append(
 463           self.box_loss(box_outputs[level], box_targets_l, num_positives_sum))
 464     # Sums per level losses to total loss.
 465     return tf.add_n(box_losses)
 466 
 467   def box_loss(self, box_outputs, box_targets, num_positives):
 468     """Computes RetinaNet box regression loss."""
 469     # The delta is typically around the mean value of regression target.
 470     # for instances, the regression targets of 512x512 input with 6 anchors on
 471     # P3-P7 pyramid is about [0.1, 0.1, 0.2, 0.2].
 472     normalizer = num_positives * 4.0
 473     mask = tf.cast(tf.not_equal(box_targets, 0.0), dtype=tf.float32)
 474     box_targets = tf.expand_dims(box_targets, axis=-1)
 475     box_outputs = tf.expand_dims(box_outputs, axis=-1)
 476     box_loss = self._huber_loss(box_targets, box_outputs, sample_weight=mask)
 477     box_loss /= normalizer
 478     return box_loss

 481 class ShapemaskMseLoss(object):
 482   """ShapeMask mask Mean Squared Error loss function wrapper."""
 483 
 484   def __call__(self, probs, labels, valid_mask):
 485     """Compute instance segmentation loss.
 486 
 487     Args:
 488       probs: A Tensor of shape [batch_size * num_points, height, width,
 489         num_classes]. The logits are not necessarily between 0 and 1.
 490       labels: A float32/float16 Tensor of shape [batch_size, num_instances,
 491           mask_size, mask_size], where mask_size =
 492           mask_crop_size * gt_upsample_scale for fine mask, or mask_crop_size
 493           for coarse masks and shape priors.
 494       valid_mask: a binary mask indicating valid training masks.
 495 
 496     Returns:
 497       loss: an float tensor representing total mask classification loss.
 498     """
 499     with tf.name_scope('shapemask_prior_loss'):
 500       batch_size, num_instances = valid_mask.get_shape().as_list()[:2]
 501       diff = (tf.cast(labels, dtype=tf.float32) -
 502               tf.cast(probs, dtype=tf.float32))
 503       diff *= tf.cast(
 504           tf.reshape(valid_mask, [batch_size, num_instances, 1, 1]),
 505           tf.float32)
 506       # Adding 0.001 in the denominator to avoid division by zero.
 507       loss = tf.nn.l2_loss(diff) / (tf.reduce_sum(labels) + 0.001)
 508     return loss

 511 class ShapemaskLoss(object):
 512   """ShapeMask mask loss function wrapper."""
 513 
 514   def __init__(self):
 515     self._binary_crossentropy = tf.keras.losses.BinaryCrossentropy(
 516         reduction=tf.keras.losses.Reduction.SUM, from_logits=True)
 517 
 518   def __call__(self, logits, labels, valid_mask):
 519     """ShapeMask mask cross entropy loss function wrapper.
 520 
 521     Args:
 522       logits: A Tensor of shape [batch_size * num_instances, height, width,
 523         num_classes]. The logits are not necessarily between 0 and 1.
 524       labels: A float16/float32 Tensor of shape [batch_size, num_instances,
 525         mask_size, mask_size], where mask_size =
 526         mask_crop_size * gt_upsample_scale for fine mask, or mask_crop_size
 527         for coarse masks and shape priors.
 528       valid_mask: a binary mask of shape [batch_size, num_instances]
 529         indicating valid training masks.
 530     Returns:
 531       loss: an float tensor representing total mask classification loss.
 532     """
 533     with tf.name_scope('shapemask_loss'):
 534       batch_size, num_instances = valid_mask.get_shape().as_list()[:2]
 535       labels = tf.cast(labels, tf.float32)
 536       logits = tf.cast(logits, tf.float32)
 537       loss = self._binary_crossentropy(labels, logits)
 538       loss *= tf.cast(tf.reshape(
 539           valid_mask, [batch_size, num_instances, 1, 1]), loss.dtype)
 540       # Adding 0.001 in the denominator to avoid division by zero.
 541       loss = tf.reduce_sum(loss) / (tf.reduce_sum(labels) + 0.001)
 542     return loss
