  37 class MaskrcnnModel(base_model.Model):
  38   """Mask R-CNN model function."""
  39 
  40   def __init__(self, params):
  41     super(MaskrcnnModel, self).__init__(params)
  42 
  43     # For eval metrics.
  44     self._params = params
  45     self._keras_model = None
  46 
  47     self._include_mask = params.architecture.include_mask
  48 
  49     # Architecture generators.
  50     self._backbone_fn = factory.backbone_generator(params)
  51     self._fpn_fn = factory.multilevel_features_generator(params)
  52     self._rpn_head_fn = factory.rpn_head_generator(params)
  53     self._generate_rois_fn = roi_ops.ROIGenerator(params.roi_proposal)
  54     self._sample_rois_fn = target_ops.ROISampler(params.roi_sampling)
  55     self._sample_masks_fn = target_ops.MaskSampler(
  56         params.architecture.mask_target_size,
  57         params.mask_sampling.num_mask_samples_per_image)
  58 
  59     self._frcnn_head_fn = factory.fast_rcnn_head_generator(params)
  60     if self._include_mask:
  61       self._mrcnn_head_fn = factory.mask_rcnn_head_generator(params)
  62 
  63     # Loss function.
  64     self._rpn_score_loss_fn = losses.RpnScoreLoss(params.rpn_score_loss)
  65     self._rpn_box_loss_fn = losses.RpnBoxLoss(params.rpn_box_loss)
  66     self._frcnn_class_loss_fn = losses.FastrcnnClassLoss()
  67     self._frcnn_box_loss_fn = losses.FastrcnnBoxLoss(params.frcnn_box_loss)
  68     if self._include_mask:
  69       self._mask_loss_fn = losses.MaskrcnnLoss()
  70 
  71     self._generate_detections_fn = postprocess_ops.GenericDetectionGenerator(
  72         params.postprocess)
  73 
  74     self._transpose_input = params.train.transpose_input
  75     assert not self._transpose_input, 'Transpose input is not supportted.'
  76 
  77   def build_outputs(self, inputs, mode):
  78     is_training = mode == mode_keys.TRAIN
  79     model_outputs = {}
  80 
  81     image = inputs['image']
  82     _, image_height, image_width, _ = image.get_shape().as_list()
  83     backbone_features = self._backbone_fn(image, is_training)
  84     fpn_features = self._fpn_fn(backbone_features, is_training)
  85 
  86     rpn_score_outputs, rpn_box_outputs = self._rpn_head_fn(
  87         fpn_features, is_training)
  88     model_outputs.update({
  89         'rpn_score_outputs':
  90             tf.nest.map_structure(lambda x: tf.cast(x, tf.float32),
  91                                   rpn_score_outputs),
  92         'rpn_box_outputs':
  93             tf.nest.map_structure(lambda x: tf.cast(x, tf.float32),
  94                                   rpn_box_outputs),
  95     })
  96     input_anchor = anchor.Anchor(self._params.architecture.min_level,
  97                                  self._params.architecture.max_level,
  98                                  self._params.anchor.num_scales,
  99                                  self._params.anchor.aspect_ratios,
 100                                  self._params.anchor.anchor_size,
 101                                  (image_height, image_width))
 102     rpn_rois, _ = self._generate_rois_fn(rpn_box_outputs, rpn_score_outputs,
 103                                          input_anchor.multilevel_boxes,
 104                                          inputs['image_info'][:, 1, :],
 105                                          is_training)
 106     if is_training:
 107       rpn_rois = tf.stop_gradient(rpn_rois)
 108 
 109       # Sample proposals.
 110       rpn_rois, matched_gt_boxes, matched_gt_classes, matched_gt_indices = (
 111           self._sample_rois_fn(rpn_rois, inputs['gt_boxes'],
 112                                inputs['gt_classes']))
 113 
 114       # Create bounding box training targets.
 115       box_targets = box_utils.encode_boxes(
 116           matched_gt_boxes, rpn_rois, weights=[10.0, 10.0, 5.0, 5.0])
 117       # If the target is background, the box target is set to all 0s.
 118       box_targets = tf.where(
 119           tf.tile(
 120               tf.expand_dims(tf.equal(matched_gt_classes, 0), axis=-1),
 121               [1, 1, 4]),
 122           tf.zeros_like(box_targets),
 123           box_targets)
 124       model_outputs.update({
 125           'class_targets': matched_gt_classes,
 126           'box_targets': box_targets,
 127       })
 128 
 129     roi_features = spatial_transform_ops.multilevel_crop_and_resize(
 130         fpn_features, rpn_rois, output_size=7)
 131 
 132     class_outputs, box_outputs = self._frcnn_head_fn(roi_features, is_training)
 133 
 134     model_outputs.update({
 135         'class_outputs':
 136             tf.nest.map_structure(lambda x: tf.cast(x, tf.float32),
 137                                   class_outputs),
 138         'box_outputs':
 139             tf.nest.map_structure(lambda x: tf.cast(x, tf.float32),
 140                                   box_outputs),
 141     })
 142 
 143     # Add this output to train to make the checkpoint loadable in predict mode.
 144     # If we skip it in train mode, the heads will be out-of-order and checkpoint
 145     # loading will fail.
 146     boxes, scores, classes, valid_detections = self._generate_detections_fn(
 147         box_outputs, class_outputs, rpn_rois, inputs['image_info'][:, 1:2, :])
 148     model_outputs.update({
 149         'num_detections': valid_detections,
 150         'detection_boxes': boxes,
 151         'detection_classes': classes,
 152         'detection_scores': scores,
 153     })
 154 
 155     if not self._include_mask:
 156       return model_outputs
 157 
 158     if is_training:
 159       rpn_rois, classes, mask_targets = self._sample_masks_fn(
 160           rpn_rois, matched_gt_boxes, matched_gt_classes, matched_gt_indices,
 161           inputs['gt_masks'])
 162       mask_targets = tf.stop_gradient(mask_targets)
 163 
 164       classes = tf.cast(classes, dtype=tf.int32)
 165 
 166       model_outputs.update({
 167           'mask_targets': mask_targets,
 168           'sampled_class_targets': classes,
 169       })
 170     else:
 171       rpn_rois = boxes
 172       classes = tf.cast(classes, dtype=tf.int32)
 173 
 174     mask_roi_features = spatial_transform_ops.multilevel_crop_and_resize(
 175         fpn_features, rpn_rois, output_size=14)
 176 
 177     mask_outputs = self._mrcnn_head_fn(mask_roi_features, classes, is_training)
 178 
 179     if is_training:
 180       model_outputs.update({
 181           'mask_outputs':
 182               tf.nest.map_structure(lambda x: tf.cast(x, tf.float32),
 183                                     mask_outputs),
 184       })
 185     else:
 186       model_outputs.update({
 187           'detection_masks': tf.nn.sigmoid(mask_outputs)
 188       })
 189 
 190     return model_outputs
 191 
 192   def build_loss_fn(self):
 193     if self._keras_model is None:
 194       raise ValueError('build_loss_fn() must be called after build_model().')
 195 
 196     filter_fn = self.make_filter_trainable_variables_fn()
 197     trainable_variables = filter_fn(self._keras_model.trainable_variables)
 198 
 199     def _total_loss_fn(labels, outputs):
 200       rpn_score_loss = self._rpn_score_loss_fn(outputs['rpn_score_outputs'],
 201                                                labels['rpn_score_targets'])
 202       rpn_box_loss = self._rpn_box_loss_fn(outputs['rpn_box_outputs'],
 203                                            labels['rpn_box_targets'])
 204 
 205       frcnn_class_loss = self._frcnn_class_loss_fn(outputs['class_outputs'],
 206                                                    outputs['class_targets'])
 207       frcnn_box_loss = self._frcnn_box_loss_fn(outputs['box_outputs'],
 208                                                outputs['class_targets'],
 209                                                outputs['box_targets'])
 210 
 211       if self._include_mask:
 212         mask_loss = self._mask_loss_fn(outputs['mask_outputs'],
 213                                        outputs['mask_targets'],
 214                                        outputs['sampled_class_targets'])
 215       else:
 216         mask_loss = 0.0
 217 
 218       model_loss = (
 219           rpn_score_loss + rpn_box_loss + frcnn_class_loss + frcnn_box_loss +
 220           mask_loss)
 221 
 222       l2_regularization_loss = self.weight_decay_loss(trainable_variables)
 223       total_loss = model_loss + l2_regularization_loss
 224       return {
 225           'total_loss': total_loss,
 226           'loss': total_loss,
 227           'fast_rcnn_class_loss': frcnn_class_loss,
 228           'fast_rcnn_box_loss': frcnn_box_loss,
 229           'mask_loss': mask_loss,
 230           'model_loss': model_loss,
 231           'l2_regularization_loss': l2_regularization_loss,
 232           'rpn_score_loss': rpn_score_loss,
 233           'rpn_box_loss': rpn_box_loss,
 234       }
 235 
 236     return _total_loss_fn

 238   def build_input_layers(self, params, mode):
 239     is_training = mode == mode_keys.TRAIN
 240     input_shape = (
 241         params.maskrcnn_parser.output_size +
 242         [params.maskrcnn_parser.num_channels])
 243     if is_training:
 244       batch_size = params.train.batch_size
 245       input_layer = {
 246           'image':
 247               tf.keras.layers.Input(
 248                   shape=input_shape,
 249                   batch_size=batch_size,
 250                   name='image',
 251                   dtype=tf.bfloat16 if self._use_bfloat16 else tf.float32),
 252           'image_info':
 253               tf.keras.layers.Input(
 254                   shape=[4, 2],
 255                   batch_size=batch_size,
 256                   name='image_info',
 257               ),
 258           'gt_boxes':
 259               tf.keras.layers.Input(
 260                   shape=[params.maskrcnn_parser.max_num_instances, 4],
 261                   batch_size=batch_size,
 262                   name='gt_boxes'),
 263           'gt_classes':
 264               tf.keras.layers.Input(
 265                   shape=[params.maskrcnn_parser.max_num_instances],
 266                   batch_size=batch_size,
 267                   name='gt_classes',
 268                   dtype=tf.int64),
 269       }
 270       if self._include_mask:
 271         input_layer['gt_masks'] = tf.keras.layers.Input(
 272             shape=[
 273                 params.maskrcnn_parser.max_num_instances,
 274                 params.maskrcnn_parser.mask_crop_size,
 275                 params.maskrcnn_parser.mask_crop_size
 276             ],
 277             batch_size=batch_size,
 278             name='gt_masks')
 279     else:
 280       batch_size = params.eval.batch_size
 281       input_layer = {
 282           'image':
 283               tf.keras.layers.Input(
 284                   shape=input_shape,
 285                   batch_size=batch_size,
 286                   name='image',
 287                   dtype=tf.bfloat16 if self._use_bfloat16 else tf.float32),
 288           'image_info':
 289               tf.keras.layers.Input(
 290                   shape=[4, 2],
 291                   batch_size=batch_size,
 292                   name='image_info',
 293               ),
 294       }
 295     return input_layer

 297   def build_model(self, params, mode):
 298     if self._keras_model is None:
 299       input_layers = self.build_input_layers(self._params, mode)
 300       with backend.get_graph().as_default():
 301         outputs = self.model_outputs(input_layers, mode)
 302 
 303         model = tf.keras.models.Model(
 304             inputs=input_layers, outputs=outputs, name='maskrcnn')
 305         assert model is not None, 'Fail to build tf.keras.Model.'
 306         model.optimizer = self.build_optimizer()
 307         self._keras_model = model
 308 
 309     return self._keras_model

 311   def post_processing(self, labels, outputs):
 312     required_output_fields = ['class_outputs', 'box_outputs']
 313     for field in required_output_fields:
 314       if field not in outputs:
 315         raise ValueError('"%s" is missing in outputs, requried %s found %s'
 316                          %(field, required_output_fields, outputs.keys()))
 317     predictions = {
 318         'image_info': labels['image_info'],
 319         'num_detections': outputs['num_detections'],
 320         'detection_boxes': outputs['detection_boxes'],
 321         'detection_classes': outputs['detection_classes'],
 322         'detection_scores': outputs['detection_scores'],
 323     }
 324     if self._include_mask:
 325       predictions.update({
 326           'detection_masks': outputs['detection_masks'],
 327       })
 328 
 329     if 'groundtruths' in labels:
 330       predictions['source_id'] = labels['groundtruths']['source_id']
 331       predictions['gt_source_id'] = labels['groundtruths']['source_id']
 332       predictions['gt_height'] = labels['groundtruths']['height']
 333       predictions['gt_width'] = labels['groundtruths']['width']
 334       predictions['gt_image_info'] = labels['image_info']
 335       predictions['gt_num_detections'] = (
 336           labels['groundtruths']['num_detections'])
 337       predictions['gt_boxes'] = labels['groundtruths']['boxes']
 338       predictions['gt_classes'] = labels['groundtruths']['classes']
 339       predictions['gt_areas'] = labels['groundtruths']['areas']
 340       predictions['gt_is_crowds'] = labels['groundtruths']['is_crowds']
 341     return labels, predictions
 342 
 343   def eval_metrics(self):
 344     return eval_factory.evaluator_generator(self._params.eval)
