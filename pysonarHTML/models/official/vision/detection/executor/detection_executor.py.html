  29 class DetectionDistributedExecutor(executor.DistributedExecutor):
  30   """Detection specific customer training loop executor.
  31 
  32   Subclasses the DistributedExecutor and adds support for numpy based metrics.
  33   """
  34 
  35   def __init__(self,
  36                predict_post_process_fn=None,
  37                trainable_variables_filter=None,
  38                **kwargs):
  39     super(DetectionDistributedExecutor, self).__init__(**kwargs)
  40     if predict_post_process_fn:
  41       assert callable(predict_post_process_fn)
  42     if trainable_variables_filter:
  43       assert callable(trainable_variables_filter)
  44     self._predict_post_process_fn = predict_post_process_fn
  45     self._trainable_variables_filter = trainable_variables_filter
  46     self.eval_steps = tf.Variable(
  47         0,
  48         trainable=False,
  49         dtype=tf.int32,
  50         synchronization=tf.VariableSynchronization.ON_READ,
  51         aggregation=tf.VariableAggregation.ONLY_FIRST_REPLICA,
  52         shape=[])
  53 
  54   def _create_replicated_step(self,
  55                               strategy,
  56                               model,
  57                               loss_fn,
  58                               optimizer,
  59                               metric=None):
  60     trainable_variables = model.trainable_variables
  61     if self._trainable_variables_filter:
  62       trainable_variables = self._trainable_variables_filter(
  63           trainable_variables)
  64     logging.info('Filter trainable variables from %d to %d',
  65                  len(model.trainable_variables), len(trainable_variables))
  66     _update_state = lambda labels, outputs: None
  67     if isinstance(metric, tf.keras.metrics.Metric):
  68       _update_state = lambda labels, outputs: metric.update_state(
  69           labels, outputs)
  70     else:
  71       logging.error('Detection: train metric is not an instance of '
  72                     'tf.keras.metrics.Metric.')
  73 
  74     def _replicated_step(inputs):
  75       """Replicated training step."""
  76       inputs, labels = inputs
  77 
  78       with tf.GradientTape() as tape:
  79         outputs = model(inputs, training=True)
  80         all_losses = loss_fn(labels, outputs)
  81         losses = {}
  82         for k, v in all_losses.items():
  83           losses[k] = tf.reduce_mean(v)
  84         per_replica_loss = losses['total_loss'] / strategy.num_replicas_in_sync
  85         _update_state(labels, outputs)
  86 
  87       grads = tape.gradient(per_replica_loss, trainable_variables)
  88       optimizer.apply_gradients(zip(grads, trainable_variables))
  89       return losses
  90 
  91     return _replicated_step
  92 
  93   def _create_test_step(self, strategy, model, metric):
  94     """Creates a distributed test step."""
  95 
  96     @tf.function
  97     def test_step(iterator, eval_steps):
  98       """Calculates evaluation metrics on distributed devices."""
  99 
 100       def _test_step_fn(inputs, eval_steps):
 101         """Replicated accuracy calculation."""
 102         inputs, labels = inputs
 103         model_outputs = model(inputs, training=False)
 104         if self._predict_post_process_fn:
 105           labels, prediction_outputs = self._predict_post_process_fn(
 106               labels, model_outputs)
 107           num_remaining_visualizations = (
 108               self._params.eval.num_images_to_visualize - eval_steps)
 109           # If there are remaining number of visualizations that needs to be
 110           # done, add next batch outputs for visualization.
 111           #
 112           # TODO(hongjunchoi): Once dynamic slicing is supported on TPU, only
 113           # write correct slice of outputs to summary file.
 114           if num_remaining_visualizations > 0:
 115             visualization_utils.visualize_images_with_bounding_boxes(
 116                 inputs, prediction_outputs['detection_boxes'],
 117                 self.global_train_step, self.eval_summary_writer)
 118 
 119         return labels, prediction_outputs
 120 
 121       labels, outputs = strategy.run(
 122           _test_step_fn, args=(
 123               next(iterator),
 124               eval_steps,
 125           ))
 126       outputs = tf.nest.map_structure(strategy.experimental_local_results,
 127                                       outputs)
 128       labels = tf.nest.map_structure(strategy.experimental_local_results,
 129                                      labels)
 130 
 131       eval_steps.assign_add(self._params.eval.batch_size)
 132       return labels, outputs
 133 
 134     return test_step
 135 
 136   def _run_evaluation(self, test_step, current_training_step, metric,
 137                       test_iterator):
 138     """Runs validation steps and aggregate metrics."""
 139     self.eval_steps.assign(0)
 140     if not test_iterator or not metric:
 141       logging.warning(
 142           'Both test_iterator (%s) and metrics (%s) must not be None.',
 143           test_iterator, metric)
 144       return None
 145     logging.info('Running evaluation after step: %s.', current_training_step)
 146     while True:
 147       try:
 148         labels, outputs = test_step(test_iterator, self.eval_steps)
 149         if metric:
 150           metric.update_state(labels, outputs)
 151       except (StopIteration, tf.errors.OutOfRangeError):
 152         break
 153 
 154     metric_result = metric.result()
 155     if isinstance(metric, tf.keras.metrics.Metric):
 156       metric_result = tf.nest.map_structure(lambda x: x.numpy().astype(float),
 157                                             metric_result)
 158     logging.info('Step: [%d] Validation metric = %s', current_training_step,
 159                  metric_result)
 160     return metric_result
