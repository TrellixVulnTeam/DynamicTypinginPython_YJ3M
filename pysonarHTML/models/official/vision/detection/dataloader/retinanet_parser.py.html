  33 def process_source_id(source_id):
  34   """Processes source_id to the right format."""
  35   if source_id.dtype == tf.string:
  36     source_id = tf.cast(tf.strings.to_number(source_id), tf.int32)
  37   with tf.control_dependencies([source_id]):
  38     source_id = tf.cond(
  39         pred=tf.equal(tf.size(input=source_id), 0),
  40         true_fn=lambda: tf.cast(tf.constant(-1), tf.int32),
  41         false_fn=lambda: tf.identity(source_id))
  42   return source_id

  45 def pad_groundtruths_to_fixed_size(gt, n):
  46   """Pads the first dimension of groundtruths labels to the fixed size."""
  47   gt['boxes'] = input_utils.pad_to_fixed_size(gt['boxes'], n, -1)
  48   gt['is_crowds'] = input_utils.pad_to_fixed_size(gt['is_crowds'], n, 0)
  49   gt['areas'] = input_utils.pad_to_fixed_size(gt['areas'], n, -1)
  50   gt['classes'] = input_utils.pad_to_fixed_size(gt['classes'], n, -1)
  51   return gt

  57   def __init__(self,
  58                output_size,
  59                min_level,
  60                max_level,
  61                num_scales,
  62                aspect_ratios,
  63                anchor_size,
  64                match_threshold=0.5,
  65                unmatched_threshold=0.5,
  66                aug_rand_hflip=False,
  67                aug_scale_min=1.0,
  68                aug_scale_max=1.0,
  69                use_autoaugment=False,
  70                autoaugment_policy_name='v0',
  71                skip_crowd_during_training=True,
  72                max_num_instances=100,
  73                use_bfloat16=True,
  74                mode=None):
  75     """Initializes parameters for parsing annotations in the dataset.
  76 
  77     Args:
  78       output_size: `Tensor` or `list` for [height, width] of output image. The
  79         output_size should be divided by the largest feature stride 2^max_level.
  80       min_level: `int` number of minimum level of the output feature pyramid.
  81       max_level: `int` number of maximum level of the output feature pyramid.
  82       num_scales: `int` number representing intermediate scales added
  83         on each level. For instances, num_scales=2 adds one additional
  84         intermediate anchor scales [2^0, 2^0.5] on each level.
  85       aspect_ratios: `list` of float numbers representing the aspect raito
  86         anchors added on each level. The number indicates the ratio of width to
  87         height. For instances, aspect_ratios=[1.0, 2.0, 0.5] adds three anchors
  88         on each scale level.
  89       anchor_size: `float` number representing the scale of size of the base
  90         anchor to the feature stride 2^level.
  91       match_threshold: `float` number between 0 and 1 representing the
  92         lower-bound threshold to assign positive labels for anchors. An anchor
  93         with a score over the threshold is labeled positive.
  94       unmatched_threshold: `float` number between 0 and 1 representing the
  95         upper-bound threshold to assign negative labels for anchors. An anchor
  96         with a score below the threshold is labeled negative.
  97       aug_rand_hflip: `bool`, if True, augment training with random
  98         horizontal flip.
  99       aug_scale_min: `float`, the minimum scale applied to `output_size` for
 100         data augmentation during training.
 101       aug_scale_max: `float`, the maximum scale applied to `output_size` for
 102         data augmentation during training.
 103       use_autoaugment: `bool`, if True, use the AutoAugment augmentation policy
 104         during training.
 105       autoaugment_policy_name: `string` that specifies the name of the
 106         AutoAugment policy that will be used during training.
 107       skip_crowd_during_training: `bool`, if True, skip annotations labeled with
 108         `is_crowd` equals to 1.
 109       max_num_instances: `int` number of maximum number of instances in an
 110         image. The groundtruth data will be padded to `max_num_instances`.
 111       use_bfloat16: `bool`, if True, cast output image to tf.bfloat16.
 112       mode: a ModeKeys. Specifies if this is training, evaluation, prediction
 113         or prediction with groundtruths in the outputs.
 114     """
 115     self._mode = mode
 116     self._max_num_instances = max_num_instances
 117     self._skip_crowd_during_training = skip_crowd_during_training
 118     self._is_training = (mode == ModeKeys.TRAIN)
 119 
 120     self._example_decoder = tf_example_decoder.TfExampleDecoder(
 121         include_mask=False)
 122 
 123     # Anchor.
 124     self._output_size = output_size
 125     self._min_level = min_level
 126     self._max_level = max_level
 127     self._num_scales = num_scales
 128     self._aspect_ratios = aspect_ratios
 129     self._anchor_size = anchor_size
 130     self._match_threshold = match_threshold
 131     self._unmatched_threshold = unmatched_threshold
 132 
 133     # Data augmentation.
 134     self._aug_rand_hflip = aug_rand_hflip
 135     self._aug_scale_min = aug_scale_min
 136     self._aug_scale_max = aug_scale_max
 137 
 138     # Data Augmentation with AutoAugment.
 139     self._use_autoaugment = use_autoaugment
 140     self._autoaugment_policy_name = autoaugment_policy_name
 141 
 142     # Device.
 143     self._use_bfloat16 = use_bfloat16
 144 
 145     # Data is parsed depending on the model Modekey.
 146     if mode == ModeKeys.TRAIN:
 147       self._parse_fn = self._parse_train_data
 148     elif mode == ModeKeys.EVAL:
 149       self._parse_fn = self._parse_eval_data
 150     elif mode == ModeKeys.PREDICT or mode == ModeKeys.PREDICT_WITH_GT:
 151       self._parse_fn = self._parse_predict_data
 152     else:
 153       raise ValueError('mode is not defined.')
 154 
 155   def __call__(self, value):
 156     """Parses data to an image and associated training labels.
 157 
 158     Args:
 159       value: a string tensor holding a serialized tf.Example proto.
 160 
 161     Returns:
 162       image: image tensor that is preproessed to have normalized value and
 163         dimension [output_size[0], output_size[1], 3]
 164       labels:
 165         cls_targets: ordered dictionary with keys
 166           [min_level, min_level+1, ..., max_level]. The values are tensor with
 167           shape [height_l, width_l, anchors_per_location]. The height_l and
 168           width_l represent the dimension of class logits at l-th level.
 169         box_targets: ordered dictionary with keys
 170           [min_level, min_level+1, ..., max_level]. The values are tensor with
 171           shape [height_l, width_l, anchors_per_location * 4]. The height_l and
 172           width_l represent the dimension of bounding box regression output at
 173           l-th level.
 174         num_positives: number of positive anchors in the image.
 175         anchor_boxes: ordered dictionary with keys
 176           [min_level, min_level+1, ..., max_level]. The values are tensor with
 177           shape [height_l, width_l, 4] representing anchor boxes at each level.
 178         image_info: a 2D `Tensor` that encodes the information of the image and
 179           the applied preprocessing. It is in the format of
 180           [[original_height, original_width], [scaled_height, scaled_width],
 181            [y_scale, x_scale], [y_offset, x_offset]].
 182         groundtruths:
 183           source_id: source image id. Default value -1 if the source id is empty
 184             in the groundtruth annotation.
 185           boxes: groundtruth bounding box annotations. The box is represented in
 186             [y1, x1, y2, x2] format. The tennsor is padded with -1 to the fixed
 187             dimension [self._max_num_instances, 4].
 188           classes: groundtruth classes annotations. The tennsor is padded with
 189             -1 to the fixed dimension [self._max_num_instances].
 190           areas: groundtruth areas annotations. The tennsor is padded with -1
 191             to the fixed dimension [self._max_num_instances].
 192           is_crowds: groundtruth annotations to indicate if an annotation
 193             represents a group of instances by value {0, 1}. The tennsor is
 194             padded with 0 to the fixed dimension [self._max_num_instances].
 195     """
 196     with tf.name_scope('parser'):
 197       data = self._example_decoder.decode(value)
 198       return self._parse_fn(data)

 200   def _parse_train_data(self, data):
 201     """Parses data for training and evaluation."""
 202     classes = data['groundtruth_classes']
 203     boxes = data['groundtruth_boxes']
 204     is_crowds = data['groundtruth_is_crowd']
 205     # Skips annotations with `is_crowd` = True.
 206     if self._skip_crowd_during_training and self._is_training:
 207       num_groundtrtuhs = tf.shape(input=classes)[0]
 208       with tf.control_dependencies([num_groundtrtuhs, is_crowds]):
 209         indices = tf.cond(
 210             pred=tf.greater(tf.size(input=is_crowds), 0),
 211             true_fn=lambda: tf.where(tf.logical_not(is_crowds))[:, 0],
 212             false_fn=lambda: tf.cast(tf.range(num_groundtrtuhs), tf.int64))
 213       classes = tf.gather(classes, indices)
 214       boxes = tf.gather(boxes, indices)
 215 
 216     # Gets original image and its size.
 217     image = data['image']
 218 
 219     image_shape = tf.shape(input=image)[0:2]
 220 
 221     # Normalizes image with mean and std pixel values.
 222     image = input_utils.normalize_image(image)
 223 
 224     # Flips image randomly during training.
 225     if self._aug_rand_hflip:
 226       image, boxes = input_utils.random_horizontal_flip(image, boxes)
 227 
 228     # Converts boxes from normalized coordinates to pixel coordinates.
 229     boxes = box_utils.denormalize_boxes(boxes, image_shape)
 230 
 231     # Resizes and crops image.
 232     image, image_info = input_utils.resize_and_crop_image(
 233         image,
 234         self._output_size,
 235         padded_size=input_utils.compute_padded_size(
 236             self._output_size, 2 ** self._max_level),
 237         aug_scale_min=self._aug_scale_min,
 238         aug_scale_max=self._aug_scale_max)
 239     image_height, image_width, _ = image.get_shape().as_list()
 240 
 241     # Resizes and crops boxes.
 242     image_scale = image_info[2, :]
 243     offset = image_info[3, :]
 244     boxes = input_utils.resize_and_crop_boxes(
 245         boxes, image_scale, image_info[1, :], offset)
 246     # Filters out ground truth boxes that are all zeros.
 247     indices = box_utils.get_non_empty_box_indices(boxes)
 248     boxes = tf.gather(boxes, indices)
 249     classes = tf.gather(classes, indices)
 250 
 251     # Assigns anchors.
 252     input_anchor = anchor.Anchor(
 253         self._min_level, self._max_level, self._num_scales,
 254         self._aspect_ratios, self._anchor_size, (image_height, image_width))
 255     anchor_labeler = anchor.AnchorLabeler(
 256         input_anchor, self._match_threshold, self._unmatched_threshold)
 257     (cls_targets, box_targets, num_positives) = anchor_labeler.label_anchors(
 258         boxes,
 259         tf.cast(tf.expand_dims(classes, axis=1), tf.float32))
 260 
 261     # If bfloat16 is used, casts input image to tf.bfloat16.
 262     if self._use_bfloat16:
 263       image = tf.cast(image, dtype=tf.bfloat16)
 264 
 265     # Packs labels for model_fn outputs.
 266     labels = {
 267         'cls_targets': cls_targets,
 268         'box_targets': box_targets,
 269         'anchor_boxes': input_anchor.multilevel_boxes,
 270         'num_positives': num_positives,
 271         'image_info': image_info,
 272     }
 273     return image, labels

 275   def _parse_eval_data(self, data):
 276     """Parses data for training and evaluation."""
 277     groundtruths = {}
 278     classes = data['groundtruth_classes']
 279     boxes = data['groundtruth_boxes']
 280 
 281     # Gets original image and its size.
 282     image = data['image']
 283     image_shape = tf.shape(input=image)[0:2]
 284 
 285     # Normalizes image with mean and std pixel values.
 286     image = input_utils.normalize_image(image)
 287 
 288     # Converts boxes from normalized coordinates to pixel coordinates.
 289     boxes = box_utils.denormalize_boxes(boxes, image_shape)
 290 
 291     # Resizes and crops image.
 292     image, image_info = input_utils.resize_and_crop_image(
 293         image,
 294         self._output_size,
 295         padded_size=input_utils.compute_padded_size(
 296             self._output_size, 2 ** self._max_level),
 297         aug_scale_min=1.0,
 298         aug_scale_max=1.0)
 299     image_height, image_width, _ = image.get_shape().as_list()
 300 
 301     # Resizes and crops boxes.
 302     image_scale = image_info[2, :]
 303     offset = image_info[3, :]
 304     boxes = input_utils.resize_and_crop_boxes(
 305         boxes, image_scale, image_info[1, :], offset)
 306     # Filters out ground truth boxes that are all zeros.
 307     indices = box_utils.get_non_empty_box_indices(boxes)
 308     boxes = tf.gather(boxes, indices)
 309     classes = tf.gather(classes, indices)
 310 
 311     # Assigns anchors.
 312     input_anchor = anchor.Anchor(
 313         self._min_level, self._max_level, self._num_scales,
 314         self._aspect_ratios, self._anchor_size, (image_height, image_width))
 315     anchor_labeler = anchor.AnchorLabeler(
 316         input_anchor, self._match_threshold, self._unmatched_threshold)
 317     (cls_targets, box_targets, num_positives) = anchor_labeler.label_anchors(
 318         boxes,
 319         tf.cast(tf.expand_dims(classes, axis=1), tf.float32))
 320 
 321     # If bfloat16 is used, casts input image to tf.bfloat16.
 322     if self._use_bfloat16:
 323       image = tf.cast(image, dtype=tf.bfloat16)
 324 
 325     # Sets up groundtruth data for evaluation.
 326     groundtruths = {
 327         'source_id': data['source_id'],
 328         'num_groundtrtuhs': tf.shape(data['groundtruth_classes']),
 329         'image_info': image_info,
 330         'boxes': box_utils.denormalize_boxes(
 331             data['groundtruth_boxes'], image_shape),
 332         'classes': data['groundtruth_classes'],
 333         'areas': data['groundtruth_area'],
 334         'is_crowds': tf.cast(data['groundtruth_is_crowd'], tf.int32),
 335     }
 336     groundtruths['source_id'] = process_source_id(groundtruths['source_id'])
 337     groundtruths = pad_groundtruths_to_fixed_size(
 338         groundtruths, self._max_num_instances)
 339 
 340     # Packs labels for model_fn outputs.
 341     labels = {
 342         'cls_targets': cls_targets,
 343         'box_targets': box_targets,
 344         'anchor_boxes': input_anchor.multilevel_boxes,
 345         'num_positives': num_positives,
 346         'image_info': image_info,
 347         'groundtruths': groundtruths,
 348     }
 349     return image, labels

 351   def _parse_predict_data(self, data):
 352     """Parses data for prediction."""
 353     # Gets original image and its size.
 354     image = data['image']
 355     image_shape = tf.shape(input=image)[0:2]
 356 
 357     # Normalizes image with mean and std pixel values.
 358     image = input_utils.normalize_image(image)
 359 
 360     # Resizes and crops image.
 361     image, image_info = input_utils.resize_and_crop_image(
 362         image,
 363         self._output_size,
 364         padded_size=input_utils.compute_padded_size(
 365             self._output_size, 2 ** self._max_level),
 366         aug_scale_min=1.0,
 367         aug_scale_max=1.0)
 368     image_height, image_width, _ = image.get_shape().as_list()
 369 
 370     # If bfloat16 is used, casts input image to tf.bfloat16.
 371     if self._use_bfloat16:
 372       image = tf.cast(image, dtype=tf.bfloat16)
 373 
 374     # Compute Anchor boxes.
 375     input_anchor = anchor.Anchor(
 376         self._min_level, self._max_level, self._num_scales,
 377         self._aspect_ratios, self._anchor_size, (image_height, image_width))
 378 
 379     labels = {
 380         'anchor_boxes': input_anchor.multilevel_boxes,
 381         'image_info': image_info,
 382     }
 383     # If mode is PREDICT_WITH_GT, returns groundtruths and training targets
 384     # in labels.
 385     if self._mode == ModeKeys.PREDICT_WITH_GT:
 386       # Converts boxes from normalized coordinates to pixel coordinates.
 387       boxes = box_utils.denormalize_boxes(
 388           data['groundtruth_boxes'], image_shape)
 389       groundtruths = {
 390           'source_id': data['source_id'],
 391           'num_detections': tf.shape(data['groundtruth_classes']),
 392           'boxes': boxes,
 393           'classes': data['groundtruth_classes'],
 394           'areas': data['groundtruth_area'],
 395           'is_crowds': tf.cast(data['groundtruth_is_crowd'], tf.int32),
 396       }
 397       groundtruths['source_id'] = process_source_id(groundtruths['source_id'])
 398       groundtruths = pad_groundtruths_to_fixed_size(
 399           groundtruths, self._max_num_instances)
 400       labels['groundtruths'] = groundtruths
 401 
 402       # Computes training objective for evaluation loss.
 403       classes = data['groundtruth_classes']
 404 
 405       image_scale = image_info[2, :]
 406       offset = image_info[3, :]
 407       boxes = input_utils.resize_and_crop_boxes(
 408           boxes, image_scale, image_info[1, :], offset)
 409       # Filters out ground truth boxes that are all zeros.
 410       indices = box_utils.get_non_empty_box_indices(boxes)
 411       boxes = tf.gather(boxes, indices)
 412 
 413       # Assigns anchors.
 414       anchor_labeler = anchor.AnchorLabeler(
 415           input_anchor, self._match_threshold, self._unmatched_threshold)
 416       (cls_targets, box_targets, num_positives) = anchor_labeler.label_anchors(
 417           boxes,
 418           tf.cast(tf.expand_dims(classes, axis=1), tf.float32))
 419       labels['cls_targets'] = cls_targets
 420       labels['box_targets'] = box_targets
 421       labels['num_positives'] = num_positives
 422     return image, labels
