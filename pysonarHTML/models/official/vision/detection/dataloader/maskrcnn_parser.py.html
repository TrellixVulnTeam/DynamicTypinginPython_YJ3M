  27 class Parser(object):
  28   """Parser to parse an image and its annotations into a dictionary of tensors."""
  29 
  30   def __init__(self,
  31                output_size,
  32                min_level,
  33                max_level,
  34                num_scales,
  35                aspect_ratios,
  36                anchor_size,
  37                rpn_match_threshold=0.7,
  38                rpn_unmatched_threshold=0.3,
  39                rpn_batch_size_per_im=256,
  40                rpn_fg_fraction=0.5,
  41                aug_rand_hflip=False,
  42                aug_scale_min=1.0,
  43                aug_scale_max=1.0,
  44                skip_crowd_during_training=True,
  45                max_num_instances=100,
  46                include_mask=False,
  47                mask_crop_size=112,
  48                use_bfloat16=True,
  49                mode=None):
  50     """Initializes parameters for parsing annotations in the dataset.
  51 
  52     Args:
  53       output_size: `Tensor` or `list` for [height, width] of output image. The
  54         output_size should be divided by the largest feature stride 2^max_level.
  55       min_level: `int` number of minimum level of the output feature pyramid.
  56       max_level: `int` number of maximum level of the output feature pyramid.
  57       num_scales: `int` number representing intermediate scales added
  58         on each level. For instances, num_scales=2 adds one additional
  59         intermediate anchor scales [2^0, 2^0.5] on each level.
  60       aspect_ratios: `list` of float numbers representing the aspect raito
  61         anchors added on each level. The number indicates the ratio of width to
  62         height. For instances, aspect_ratios=[1.0, 2.0, 0.5] adds three anchors
  63         on each scale level.
  64       anchor_size: `float` number representing the scale of size of the base
  65         anchor to the feature stride 2^level.
  66       rpn_match_threshold:
  67       rpn_unmatched_threshold:
  68       rpn_batch_size_per_im:
  69       rpn_fg_fraction:
  70       aug_rand_hflip: `bool`, if True, augment training with random
  71         horizontal flip.
  72       aug_scale_min: `float`, the minimum scale applied to `output_size` for
  73         data augmentation during training.
  74       aug_scale_max: `float`, the maximum scale applied to `output_size` for
  75         data augmentation during training.
  76       skip_crowd_during_training: `bool`, if True, skip annotations labeled with
  77         `is_crowd` equals to 1.
  78       max_num_instances: `int` number of maximum number of instances in an
  79         image. The groundtruth data will be padded to `max_num_instances`.
  80       include_mask: a bool to indicate whether parse mask groundtruth.
  81       mask_crop_size: the size which groundtruth mask is cropped to.
  82       use_bfloat16: `bool`, if True, cast output image to tf.bfloat16.
  83       mode: a ModeKeys. Specifies if this is training, evaluation, prediction
  84         or prediction with groundtruths in the outputs.
  85     """
  86     self._mode = mode
  87     self._max_num_instances = max_num_instances
  88     self._skip_crowd_during_training = skip_crowd_during_training
  89     self._is_training = (mode == ModeKeys.TRAIN)
  90 
  91     self._example_decoder = tf_example_decoder.TfExampleDecoder(
  92         include_mask=include_mask)
  93 
  94     # Anchor.
  95     self._output_size = output_size
  96     self._min_level = min_level
  97     self._max_level = max_level
  98     self._num_scales = num_scales
  99     self._aspect_ratios = aspect_ratios
 100     self._anchor_size = anchor_size
 101 
 102     # Target assigning.
 103     self._rpn_match_threshold = rpn_match_threshold
 104     self._rpn_unmatched_threshold = rpn_unmatched_threshold
 105     self._rpn_batch_size_per_im = rpn_batch_size_per_im
 106     self._rpn_fg_fraction = rpn_fg_fraction
 107 
 108     # Data augmentation.
 109     self._aug_rand_hflip = aug_rand_hflip
 110     self._aug_scale_min = aug_scale_min
 111     self._aug_scale_max = aug_scale_max
 112 
 113     # Mask.
 114     self._include_mask = include_mask
 115     self._mask_crop_size = mask_crop_size
 116 
 117     # Device.
 118     self._use_bfloat16 = use_bfloat16
 119 
 120     # Data is parsed depending on the model Modekey.
 121     if mode == ModeKeys.TRAIN:
 122       self._parse_fn = self._parse_train_data
 123     elif mode == ModeKeys.EVAL:
 124       self._parse_fn = self._parse_eval_data
 125     elif mode == ModeKeys.PREDICT or mode == ModeKeys.PREDICT_WITH_GT:
 126       self._parse_fn = self._parse_predict_data
 127     else:
 128       raise ValueError('mode is not defined.')
 129 
 130   def __call__(self, value):
 131     """Parses data to an image and associated training labels.
 132 
 133     Args:
 134       value: a string tensor holding a serialized tf.Example proto.
 135 
 136     Returns:
 137       image, labels: if mode == ModeKeys.TRAIN. see _parse_train_data.
 138       {'images': image, 'labels': labels}: if mode == ModeKeys.PREDICT
 139         or ModeKeys.PREDICT_WITH_GT.
 140     """
 141     with tf.name_scope('parser'):
 142       data = self._example_decoder.decode(value)
 143       return self._parse_fn(data)
 144 
 145   def _parse_train_data(self, data):
 146     """Parses data for training.
 147 
 148     Args:
 149       data: the decoded tensor dictionary from TfExampleDecoder.
 150 
 151     Returns:
 152       image: image tensor that is preproessed to have normalized value and
 153         dimension [output_size[0], output_size[1], 3]
 154       labels: a dictionary of tensors used for training. The following describes
 155         {key: value} pairs in the dictionary.
 156         image_info: a 2D `Tensor` that encodes the information of the image and
 157           the applied preprocessing. It is in the format of
 158           [[original_height, original_width], [scaled_height, scaled_width],
 159         anchor_boxes: ordered dictionary with keys
 160           [min_level, min_level+1, ..., max_level]. The values are tensor with
 161           shape [height_l, width_l, 4] representing anchor boxes at each level.
 162         rpn_score_targets: ordered dictionary with keys
 163           [min_level, min_level+1, ..., max_level]. The values are tensor with
 164           shape [height_l, width_l, anchors_per_location]. The height_l and
 165           width_l represent the dimension of class logits at l-th level.
 166         rpn_box_targets: ordered dictionary with keys
 167           [min_level, min_level+1, ..., max_level]. The values are tensor with
 168           shape [height_l, width_l, anchors_per_location * 4]. The height_l and
 169           width_l represent the dimension of bounding box regression output at
 170           l-th level.
 171         gt_boxes: Groundtruth bounding box annotations. The box is represented
 172            in [y1, x1, y2, x2] format. The coordinates are w.r.t the scaled
 173            image that is fed to the network. The tennsor is padded with -1 to
 174            the fixed dimension [self._max_num_instances, 4].
 175         gt_classes: Groundtruth classes annotations. The tennsor is padded
 176           with -1 to the fixed dimension [self._max_num_instances].
 177         gt_masks: groundtrugh masks cropped by the bounding box and
 178           resized to a fixed size determined by mask_crop_size.
 179     """
 180     classes = data['groundtruth_classes']
 181     boxes = data['groundtruth_boxes']
 182     if self._include_mask:
 183       masks = data['groundtruth_instance_masks']
 184 
 185     is_crowds = data['groundtruth_is_crowd']
 186     # Skips annotations with `is_crowd` = True.
 187     if self._skip_crowd_during_training and self._is_training:
 188       num_groundtrtuhs = tf.shape(classes)[0]
 189       with tf.control_dependencies([num_groundtrtuhs, is_crowds]):
 190         indices = tf.cond(
 191             tf.greater(tf.size(is_crowds), 0),
 192             lambda: tf.where(tf.logical_not(is_crowds))[:, 0],
 193             lambda: tf.cast(tf.range(num_groundtrtuhs), tf.int64))
 194       classes = tf.gather(classes, indices)
 195       boxes = tf.gather(boxes, indices)
 196       if self._include_mask:
 197         masks = tf.gather(masks, indices)
 198 
 199     # Gets original image and its size.
 200     image = data['image']
 201     image_shape = tf.shape(image)[0:2]
 202 
 203     # Normalizes image with mean and std pixel values.
 204     image = input_utils.normalize_image(image)
 205 
 206     # Flips image randomly during training.
 207     if self._aug_rand_hflip:
 208       if self._include_mask:
 209         image, boxes, masks = input_utils.random_horizontal_flip(
 210             image, boxes, masks)
 211       else:
 212         image, boxes = input_utils.random_horizontal_flip(
 213             image, boxes)
 214 
 215     # Converts boxes from normalized coordinates to pixel coordinates.
 216     # Now the coordinates of boxes are w.r.t. the original image.
 217     boxes = box_utils.denormalize_boxes(boxes, image_shape)
 218 
 219     # Resizes and crops image.
 220     image, image_info = input_utils.resize_and_crop_image(
 221         image,
 222         self._output_size,
 223         padded_size=input_utils.compute_padded_size(
 224             self._output_size, 2 ** self._max_level),
 225         aug_scale_min=self._aug_scale_min,
 226         aug_scale_max=self._aug_scale_max)
 227     image_height, image_width, _ = image.get_shape().as_list()
 228 
 229     # Resizes and crops boxes.
 230     # Now the coordinates of boxes are w.r.t the scaled image.
 231     image_scale = image_info[2, :]
 232     offset = image_info[3, :]
 233     boxes = input_utils.resize_and_crop_boxes(
 234         boxes, image_scale, image_info[1, :], offset)
 235 
 236     # Filters out ground truth boxes that are all zeros.
 237     indices = box_utils.get_non_empty_box_indices(boxes)
 238     boxes = tf.gather(boxes, indices)
 239     classes = tf.gather(classes, indices)
 240     if self._include_mask:
 241       masks = tf.gather(masks, indices)
 242       # Transfer boxes to the original image space and do normalization.
 243       cropped_boxes = boxes + tf.tile(tf.expand_dims(offset, axis=0), [1, 2])
 244       cropped_boxes /= tf.tile(tf.expand_dims(image_scale, axis=0), [1, 2])
 245       cropped_boxes = box_utils.normalize_boxes(cropped_boxes, image_shape)
 246       num_masks = tf.shape(masks)[0]
 247       masks = tf.image.crop_and_resize(
 248           tf.expand_dims(masks, axis=-1),
 249           cropped_boxes,
 250           box_indices=tf.range(num_masks, dtype=tf.int32),
 251           crop_size=[self._mask_crop_size, self._mask_crop_size],
 252           method='bilinear')
 253       masks = tf.squeeze(masks, axis=-1)
 254 
 255     # Assigns anchor targets.
 256     # Note that after the target assignment, box targets are absolute pixel
 257     # offsets w.r.t. the scaled image.
 258     input_anchor = anchor.Anchor(
 259         self._min_level,
 260         self._max_level,
 261         self._num_scales,
 262         self._aspect_ratios,
 263         self._anchor_size,
 264         (image_height, image_width))
 265     anchor_labeler = anchor.RpnAnchorLabeler(
 266         input_anchor,
 267         self._rpn_match_threshold,
 268         self._rpn_unmatched_threshold,
 269         self._rpn_batch_size_per_im,
 270         self._rpn_fg_fraction)
 271     rpn_score_targets, rpn_box_targets = anchor_labeler.label_anchors(
 272         boxes, tf.cast(tf.expand_dims(classes, axis=-1), dtype=tf.float32))
 273 
 274     # If bfloat16 is used, casts input image to tf.bfloat16.
 275     if self._use_bfloat16:
 276       image = tf.cast(image, dtype=tf.bfloat16)
 277 
 278     inputs = {
 279         'image': image,
 280         'image_info': image_info,
 281     }
 282     # Packs labels for model_fn outputs.
 283     labels = {
 284         'anchor_boxes': input_anchor.multilevel_boxes,
 285         'image_info': image_info,
 286         'rpn_score_targets': rpn_score_targets,
 287         'rpn_box_targets': rpn_box_targets,
 288     }
 289     inputs['gt_boxes'] = input_utils.pad_to_fixed_size(boxes,
 290                                                        self._max_num_instances,
 291                                                        -1)
 292     inputs['gt_classes'] = input_utils.pad_to_fixed_size(
 293         classes, self._max_num_instances, -1)
 294     if self._include_mask:
 295       inputs['gt_masks'] = input_utils.pad_to_fixed_size(
 296           masks, self._max_num_instances, -1)
 297 
 298     return inputs, labels
 299 
 300   def _parse_eval_data(self, data):
 301     """Parses data for evaluation."""
 302     raise NotImplementedError('Not implemented!')

 304   def _parse_predict_data(self, data):
 305     """Parses data for prediction.
 306 
 307     Args:
 308       data: the decoded tensor dictionary from TfExampleDecoder.
 309 
 310     Returns:
 311       A dictionary of {'images': image, 'labels': labels} where
 312         image: image tensor that is preproessed to have normalized value and
 313           dimension [output_size[0], output_size[1], 3]
 314         labels: a dictionary of tensors used for training. The following
 315           describes {key: value} pairs in the dictionary.
 316           source_ids: Source image id. Default value -1 if the source id is
 317             empty in the groundtruth annotation.
 318           image_info: a 2D `Tensor` that encodes the information of the image
 319             and the applied preprocessing. It is in the format of
 320             [[original_height, original_width], [scaled_height, scaled_width],
 321           anchor_boxes: ordered dictionary with keys
 322             [min_level, min_level+1, ..., max_level]. The values are tensor with
 323             shape [height_l, width_l, 4] representing anchor boxes at each
 324             level.
 325     """
 326     # Gets original image and its size.
 327     image = data['image']
 328     image_shape = tf.shape(image)[0:2]
 329 
 330     # Normalizes image with mean and std pixel values.
 331     image = input_utils.normalize_image(image)
 332 
 333     # Resizes and crops image.
 334     image, image_info = input_utils.resize_and_crop_image(
 335         image,
 336         self._output_size,
 337         padded_size=input_utils.compute_padded_size(
 338             self._output_size, 2 ** self._max_level),
 339         aug_scale_min=1.0,
 340         aug_scale_max=1.0)
 341     image_height, image_width, _ = image.get_shape().as_list()
 342 
 343     # If bfloat16 is used, casts input image to tf.bfloat16.
 344     if self._use_bfloat16:
 345       image = tf.cast(image, dtype=tf.bfloat16)
 346 
 347     # Compute Anchor boxes.
 348     input_anchor = anchor.Anchor(
 349         self._min_level,
 350         self._max_level,
 351         self._num_scales,
 352         self._aspect_ratios,
 353         self._anchor_size,
 354         (image_height, image_width))
 355 
 356     labels = {
 357         'image_info': image_info,
 358     }
 359 
 360     if self._mode == ModeKeys.PREDICT_WITH_GT:
 361       # Converts boxes from normalized coordinates to pixel coordinates.
 362       boxes = box_utils.denormalize_boxes(
 363           data['groundtruth_boxes'], image_shape)
 364       groundtruths = {
 365           'source_id': data['source_id'],
 366           'height': data['height'],
 367           'width': data['width'],
 368           'num_detections': tf.shape(data['groundtruth_classes']),
 369           'boxes': boxes,
 370           'classes': data['groundtruth_classes'],
 371           'areas': data['groundtruth_area'],
 372           'is_crowds': tf.cast(data['groundtruth_is_crowd'], tf.int32),
 373       }
 374       groundtruths['source_id'] = dataloader_utils.process_source_id(
 375           groundtruths['source_id'])
 376       groundtruths = dataloader_utils.pad_groundtruths_to_fixed_size(
 377           groundtruths, self._max_num_instances)
 378       # TODO(yeqing):  Remove the `groundtrtuh` layer key (no longer needed).
 379       labels['groundtruths'] = groundtruths
 380     inputs = {
 381         'image': image,
 382         'image_info': image_info,
 383     }
 384 
 385     return inputs, labels
