  36 def pad_to_size(input_tensor, size):
  37   """Pads data with zeros to a given length at the first dimension if needed.
  38 
  39   Args:
  40     input_tensor: `Tensor` with any dimension.
  41     size: `int` number for the first dimension of output Tensor.
  42 
  43   Returns:
  44     `Tensor` with the first dimension padded to `size` if the first diemsion
  45     is less than `size`, otherwise no padding.
  46   """
  47   input_shape = tf.shape(input_tensor)
  48   padding_shape = []
  49 
  50   # Computes the padding length on the first dimension.
  51   padding_length = tf.maximum(0, size - tf.shape(input_tensor)[0])
  52   assert_length = tf.Assert(
  53       tf.greater_equal(padding_length, 0), [padding_length])
  54   with tf.control_dependencies([assert_length]):
  55     padding_shape.append(padding_length)
  56 
  57   # Copies shapes of the rest of input shape dimensions.
  58   for i in range(1, len(input_shape)):
  59     padding_shape.append(tf.shape(input=input_tensor)[i])
  60 
  61   # Pads input tensor to the fixed first dimension.
  62   paddings = tf.cast(tf.zeros(padding_shape), input_tensor.dtype)
  63   padded_tensor = tf.concat([input_tensor, paddings], axis=0)
  64   return padded_tensor

  70   def __init__(self,
  71                output_size,
  72                min_level,
  73                max_level,
  74                num_scales,
  75                aspect_ratios,
  76                anchor_size,
  77                use_category=True,
  78                outer_box_scale=1.0,
  79                box_jitter_scale=0.025,
  80                num_sampled_masks=8,
  81                mask_crop_size=32,
  82                mask_min_level=3,
  83                mask_max_level=5,
  84                upsample_factor=4,
  85                match_threshold=0.5,
  86                unmatched_threshold=0.5,
  87                aug_rand_hflip=False,
  88                aug_scale_min=1.0,
  89                aug_scale_max=1.0,
  90                skip_crowd_during_training=True,
  91                max_num_instances=100,
  92                use_bfloat16=True,
  93                mask_train_class='all',
  94                mode=None):
  95     """Initializes parameters for parsing annotations in the dataset.
  96 
  97     Args:
  98       output_size: `Tensor` or `list` for [height, width] of output image. The
  99         output_size should be divided by the largest feature stride 2^max_level.
 100       min_level: `int` number of minimum level of the output feature pyramid.
 101       max_level: `int` number of maximum level of the output feature pyramid.
 102       num_scales: `int` number representing intermediate scales added
 103         on each level. For instances, num_scales=2 adds one additional
 104         intermediate anchor scales [2^0, 2^0.5] on each level.
 105       aspect_ratios: `list` of float numbers representing the aspect raito
 106         anchors added on each level. The number indicates the ratio of width to
 107         height. For instances, aspect_ratios=[1.0, 2.0, 0.5] adds three anchors
 108         on each scale level.
 109       anchor_size: `float` number representing the scale of size of the base
 110         anchor to the feature stride 2^level.
 111       use_category: if `False`, treat all object in all classes in one
 112         foreground category.
 113       outer_box_scale: `float` number in a range of [1.0, inf) representing
 114         the scale from object box to outer box. The mask branch predicts
 115         instance mask enclosed in outer box.
 116       box_jitter_scale: `float` number representing the noise magnitude to
 117         jitter the training groundtruth boxes for mask branch.
 118       num_sampled_masks: `int` number of sampled masks for training.
 119       mask_crop_size: `list` for [height, width] of output training masks.
 120       mask_min_level: `int` number indicating the minimum feature level to
 121         obtain instance features.
 122       mask_max_level: `int` number indicating the maximum feature level to
 123         obtain instance features.
 124       upsample_factor: `int` factor of upsampling the fine mask predictions.
 125       match_threshold: `float` number between 0 and 1 representing the
 126         lower-bound threshold to assign positive labels for anchors. An anchor
 127         with a score over the threshold is labeled positive.
 128       unmatched_threshold: `float` number between 0 and 1 representing the
 129         upper-bound threshold to assign negative labels for anchors. An anchor
 130         with a score below the threshold is labeled negative.
 131       aug_rand_hflip: `bool`, if True, augment training with random
 132         horizontal flip.
 133       aug_scale_min: `float`, the minimum scale applied to `output_size` for
 134         data augmentation during training.
 135       aug_scale_max: `float`, the maximum scale applied to `output_size` for
 136         data augmentation during training.
 137       skip_crowd_during_training: `bool`, if True, skip annotations labeled with
 138         `is_crowd` equals to 1.
 139       max_num_instances: `int` number of maximum number of instances in an
 140         image. The groundtruth data will be padded to `max_num_instances`.
 141       use_bfloat16: `bool`, if True, cast output image to tf.bfloat16.
 142       mask_train_class: a string of experiment mode: `all`, `voc` or `nonvoc`.
 143       mode: a ModeKeys. Specifies if this is training, evaluation, prediction
 144         or prediction with groundtruths in the outputs.
 145     """
 146     self._mode = mode
 147     self._mask_train_class = mask_train_class
 148     self._max_num_instances = max_num_instances
 149     self._skip_crowd_during_training = skip_crowd_during_training
 150     self._is_training = (mode == ModeKeys.TRAIN)
 151 
 152     self._example_decoder = tf_example_decoder.TfExampleDecoder(
 153         include_mask=True)
 154 
 155     # Anchor.
 156     self._output_size = output_size
 157     self._min_level = min_level
 158     self._max_level = max_level
 159     self._num_scales = num_scales
 160     self._aspect_ratios = aspect_ratios
 161     self._anchor_size = anchor_size
 162     self._match_threshold = match_threshold
 163     self._unmatched_threshold = unmatched_threshold
 164 
 165     # Data augmentation.
 166     self._aug_rand_hflip = aug_rand_hflip
 167     self._aug_scale_min = aug_scale_min
 168     self._aug_scale_max = aug_scale_max
 169 
 170     # Device.
 171     self._use_bfloat16 = use_bfloat16
 172 
 173     # ShapeMask specific.
 174     # Control of which category to use.
 175     self._use_category = use_category
 176     self._num_sampled_masks = num_sampled_masks
 177     self._mask_crop_size = mask_crop_size
 178     self._mask_min_level = mask_min_level
 179     self._mask_max_level = mask_max_level
 180     self._outer_box_scale = outer_box_scale
 181     self._box_jitter_scale = box_jitter_scale
 182     self._up_sample_factor = upsample_factor
 183 
 184     # Data is parsed depending on the model Modekey.
 185     if mode == ModeKeys.TRAIN:
 186       self._parse_fn = self._parse_train_data
 187     elif mode == ModeKeys.EVAL:
 188       self._parse_fn = self._parse_eval_data
 189     elif mode == ModeKeys.PREDICT or mode == ModeKeys.PREDICT_WITH_GT:
 190       self._parse_fn = self._parse_predict_data
 191     else:
 192       raise ValueError('mode is not defined.')
 193 
 194   def __call__(self, value):
 195     """Parses data to an image and associated training labels.
 196 
 197     Args:
 198       value: a string tensor holding a serialized tf.Example proto.
 199 
 200     Returns:
 201       inputs:
 202         image: image tensor that is preproessed to have normalized value and
 203           dimension [output_size[0], output_size[1], 3]
 204         mask_boxes: sampled boxes that tightly enclose the training masks. The
 205           box is represented in [y1, x1, y2, x2] format. The tensor is sampled
 206           to the fixed dimension [self._num_sampled_masks, 4].
 207         mask_outer_boxes: loose box that enclose sampled tight box. The
 208           box is represented in [y1, x1, y2, x2] format. The tensor is sampled
 209           to the fixed dimension [self._num_sampled_masks, 4].
 210         mask_classes: the class ids of sampled training masks. The tensor has
 211           shape [self._num_sampled_masks].
 212       labels:
 213         cls_targets: ordered dictionary with keys
 214           [min_level, min_level+1, ..., max_level]. The values are tensor with
 215           shape [height_l, width_l, anchors_per_location]. The height_l and
 216           width_l represent the dimension of class logits at l-th level.
 217         box_targets: ordered dictionary with keys
 218           [min_level, min_level+1, ..., max_level]. The values are tensor with
 219           shape [height_l, width_l, anchors_per_location * 4]. The height_l and
 220           width_l represent the dimension of bounding box regression output at
 221           l-th level.
 222         num_positives: number of positive anchors in the image.
 223         anchor_boxes: ordered dictionary with keys
 224           [min_level, min_level+1, ..., max_level]. The values are tensor with
 225           shape [height_l, width_l, 4] representing anchor boxes at each level.
 226         image_scale: 2D float `Tensor` representing scale factors that apply
 227           to [height, width] of input image.
 228         mask_targets: training binary mask targets. The tensor has shape
 229           [self._num_sampled_masks, self._mask_crop_size, self._mask_crop_size].
 230         mask_is_valid: the binary tensor to indicate if the sampled masks are
 231           valide. The sampled masks are invalid when no mask annotations are
 232           included in the image. The tensor has shape [1].
 233         groundtruths:
 234           source_id: source image id. Default value -1 if the source id is empty
 235             in the groundtruth annotation.
 236           boxes: groundtruth bounding box annotations. The box is represented in
 237             [y1, x1, y2, x2] format. The tensor is padded with -1 to the fixed
 238             dimension [self._max_num_instances, 4].
 239           classes: groundtruth classes annotations. The tensor is padded with
 240             -1 to the fixed dimension [self._max_num_instances].
 241           areas: groundtruth areas annotations. The tensor is padded with -1
 242             to the fixed dimension [self._max_num_instances].
 243           is_crowds: groundtruth annotations to indicate if an annotation
 244             represents a group of instances by value {0, 1}. The tensor is
 245             padded with 0 to the fixed dimension [self._max_num_instances].
 246     """
 247     with tf.name_scope('parser'):
 248       data = self._example_decoder.decode(value)
 249       return self._parse_fn(data)

 251   def _parse_train_data(self, data):
 252     """Parse data for ShapeMask training."""
 253     classes = data['groundtruth_classes']
 254     boxes = data['groundtruth_boxes']
 255     masks = data['groundtruth_instance_masks']
 256     is_crowds = data['groundtruth_is_crowd']
 257     # Skips annotations with `is_crowd` = True.
 258     if self._skip_crowd_during_training and self._is_training:
 259       num_groundtrtuhs = tf.shape(classes)[0]
 260       with tf.control_dependencies([num_groundtrtuhs, is_crowds]):
 261         indices = tf.cond(
 262             tf.greater(tf.size(is_crowds), 0),
 263             lambda: tf.where(tf.logical_not(is_crowds))[:, 0],
 264             lambda: tf.cast(tf.range(num_groundtrtuhs), tf.int64))
 265       classes = tf.gather(classes, indices)
 266       boxes = tf.gather(boxes, indices)
 267       masks = tf.gather(masks, indices)
 268 
 269     # Gets original image and its size.
 270     image = data['image']
 271     image_shape = tf.shape(image)[0:2]
 272 
 273     # If not using category, makes all categories with id = 0.
 274     if not self._use_category:
 275       classes = tf.cast(tf.greater(classes, 0), dtype=tf.float32)
 276 
 277     # Normalizes image with mean and std pixel values.
 278     image = input_utils.normalize_image(image)
 279 
 280     # Flips image randomly during training.
 281     if self._aug_rand_hflip:
 282       image, boxes, masks = input_utils.random_horizontal_flip(
 283           image, boxes, masks)
 284 
 285     # Converts boxes from normalized coordinates to pixel coordinates.
 286     boxes = box_utils.denormalize_boxes(boxes, image_shape)
 287 
 288     # Resizes and crops image.
 289     image, image_info = input_utils.resize_and_crop_image(
 290         image,
 291         self._output_size,
 292         self._output_size,
 293         aug_scale_min=self._aug_scale_min,
 294         aug_scale_max=self._aug_scale_max)
 295     image_scale = image_info[2, :]
 296     offset = image_info[3, :]
 297 
 298     # Resizes and crops boxes and masks.
 299     boxes = input_utils.resize_and_crop_boxes(
 300         boxes, image_scale, image_info[1, :], offset)
 301 
 302     # Filters out ground truth boxes that are all zeros.
 303     indices = box_utils.get_non_empty_box_indices(boxes)
 304     boxes = tf.gather(boxes, indices)
 305     classes = tf.gather(classes, indices)
 306     masks = tf.gather(masks, indices)
 307 
 308     # Assigns anchors.
 309     input_anchor = anchor.Anchor(
 310         self._min_level, self._max_level, self._num_scales,
 311         self._aspect_ratios, self._anchor_size, self._output_size)
 312     anchor_labeler = anchor.AnchorLabeler(
 313         input_anchor, self._match_threshold, self._unmatched_threshold)
 314     (cls_targets,
 315      box_targets,
 316      num_positives) = anchor_labeler.label_anchors(
 317          boxes,
 318          tf.cast(tf.expand_dims(classes, axis=1), tf.float32))
 319 
 320     # Sample groundtruth masks/boxes/classes for mask branch.
 321     num_masks = tf.shape(masks)[0]
 322     mask_shape = tf.shape(masks)[1:3]
 323 
 324     # Pad sampled boxes/masks/classes to a constant batch size.
 325     padded_boxes = pad_to_size(boxes, self._num_sampled_masks)
 326     padded_classes = pad_to_size(classes, self._num_sampled_masks)
 327     padded_masks = pad_to_size(masks, self._num_sampled_masks)
 328 
 329     # Randomly sample groundtruth masks for mask branch training. For the image
 330     # without groundtruth masks, it will sample the dummy padded tensors.
 331     rand_indices = tf.random.shuffle(
 332         tf.range(tf.maximum(num_masks, self._num_sampled_masks)))
 333     rand_indices = tf.math.mod(rand_indices, tf.maximum(num_masks, 1))
 334     rand_indices = rand_indices[0:self._num_sampled_masks]
 335     rand_indices = tf.reshape(rand_indices, [self._num_sampled_masks])
 336 
 337     sampled_boxes = tf.gather(padded_boxes, rand_indices)
 338     sampled_classes = tf.gather(padded_classes, rand_indices)
 339     sampled_masks = tf.gather(padded_masks, rand_indices)
 340     # Jitter the sampled boxes to mimic the noisy detections.
 341     sampled_boxes = box_utils.jitter_boxes(
 342         sampled_boxes, noise_scale=self._box_jitter_scale)
 343     sampled_boxes = box_utils.clip_boxes(sampled_boxes, self._output_size)
 344     # Compute mask targets in feature crop. A feature crop fully contains a
 345     # sampled box.
 346     mask_outer_boxes = box_utils.compute_outer_boxes(
 347         sampled_boxes, tf.shape(image)[0:2], scale=self._outer_box_scale)
 348     mask_outer_boxes = box_utils.clip_boxes(mask_outer_boxes, self._output_size)
 349     # Compensate the offset of mask_outer_boxes to map it back to original image
 350     # scale.
 351     mask_outer_boxes_ori = mask_outer_boxes
 352     mask_outer_boxes_ori += tf.tile(tf.expand_dims(offset, axis=0), [1, 2])
 353     mask_outer_boxes_ori /= tf.tile(tf.expand_dims(image_scale, axis=0), [1, 2])
 354     norm_mask_outer_boxes_ori = box_utils.normalize_boxes(
 355         mask_outer_boxes_ori, mask_shape)
 356 
 357     # Set sampled_masks shape to [batch_size, height, width, 1].
 358     sampled_masks = tf.cast(tf.expand_dims(sampled_masks, axis=-1), tf.float32)
 359     mask_targets = tf.image.crop_and_resize(
 360         sampled_masks,
 361         norm_mask_outer_boxes_ori,
 362         box_indices=tf.range(self._num_sampled_masks),
 363         crop_size=[self._mask_crop_size, self._mask_crop_size],
 364         method='bilinear',
 365         extrapolation_value=0,
 366         name='train_mask_targets')
 367     mask_targets = tf.where(tf.greater_equal(mask_targets, 0.5),
 368                             tf.ones_like(mask_targets),
 369                             tf.zeros_like(mask_targets))
 370     mask_targets = tf.squeeze(mask_targets, axis=-1)
 371     if self._up_sample_factor > 1:
 372       fine_mask_targets = tf.image.crop_and_resize(
 373           sampled_masks,
 374           norm_mask_outer_boxes_ori,
 375           box_indices=tf.range(self._num_sampled_masks),
 376           crop_size=[
 377               self._mask_crop_size * self._up_sample_factor,
 378               self._mask_crop_size * self._up_sample_factor
 379           ],
 380           method='bilinear',
 381           extrapolation_value=0,
 382           name='train_mask_targets')
 383       fine_mask_targets = tf.where(
 384           tf.greater_equal(fine_mask_targets, 0.5),
 385           tf.ones_like(fine_mask_targets), tf.zeros_like(fine_mask_targets))
 386       fine_mask_targets = tf.squeeze(fine_mask_targets, axis=-1)
 387     else:
 388       fine_mask_targets = mask_targets
 389 
 390     # If bfloat16 is used, casts input image to tf.bfloat16.
 391     if self._use_bfloat16:
 392       image = tf.cast(image, dtype=tf.bfloat16)
 393 
 394     valid_image = tf.cast(tf.not_equal(num_masks, 0), tf.int32)
 395     if self._mask_train_class == 'all':
 396       mask_is_valid = valid_image * tf.ones_like(sampled_classes, tf.int32)
 397     else:
 398       # Get the intersection of sampled classes with training splits.
 399       mask_valid_classes = tf.cast(
 400           tf.expand_dims(
 401               class_utils.coco_split_class_ids(self._mask_train_class), 1),
 402           sampled_classes.dtype)
 403       match = tf.reduce_any(
 404           tf.equal(tf.expand_dims(sampled_classes, 0), mask_valid_classes), 0)
 405       mask_is_valid = valid_image * tf.cast(match, tf.int32)
 406 
 407     # Packs labels for model_fn outputs.
 408     labels = {
 409         'cls_targets': cls_targets,
 410         'box_targets': box_targets,
 411         'anchor_boxes': input_anchor.multilevel_boxes,
 412         'num_positives': num_positives,
 413         'image_info': image_info,
 414         # For ShapeMask.
 415         'mask_targets': mask_targets,
 416         'fine_mask_targets': fine_mask_targets,
 417         'mask_is_valid': mask_is_valid,
 418     }
 419 
 420     inputs = {
 421         'image': image,
 422         'image_info': image_info,
 423         'mask_boxes': sampled_boxes,
 424         'mask_outer_boxes': mask_outer_boxes,
 425         'mask_classes': sampled_classes,
 426     }
 427     return inputs, labels
 428 
 429   def _parse_predict_data(self, data):
 430     """Parse data for ShapeMask training."""
 431     classes = data['groundtruth_classes']
 432     boxes = data['groundtruth_boxes']
 433     masks = data['groundtruth_instance_masks']
 434 
 435     # Gets original image and its size.
 436     image = data['image']
 437     image_shape = tf.shape(image)[0:2]
 438 
 439     # If not using category, makes all categories with id = 0.
 440     if not self._use_category:
 441       classes = tf.cast(tf.greater(classes, 0), dtype=tf.float32)
 442 
 443     # Normalizes image with mean and std pixel values.
 444     image = input_utils.normalize_image(image)
 445 
 446     # Converts boxes from normalized coordinates to pixel coordinates.
 447     boxes = box_utils.denormalize_boxes(boxes, image_shape)
 448 
 449     # Resizes and crops image.
 450     image, image_info = input_utils.resize_and_crop_image(
 451         image,
 452         self._output_size,
 453         self._output_size,
 454         aug_scale_min=1.0,
 455         aug_scale_max=1.0)
 456     image_scale = image_info[2, :]
 457     offset = image_info[3, :]
 458 
 459     # Resizes and crops boxes and masks.
 460     boxes = input_utils.resize_and_crop_boxes(
 461         boxes, image_scale, image_info[1, :], offset)
 462     masks = input_utils.resize_and_crop_masks(
 463         tf.expand_dims(masks, axis=-1), image_scale, self._output_size, offset)
 464 
 465     # Filters out ground truth boxes that are all zeros.
 466     indices = box_utils.get_non_empty_box_indices(boxes)
 467     boxes = tf.gather(boxes, indices)
 468     classes = tf.gather(classes, indices)
 469 
 470     # Assigns anchors.
 471     input_anchor = anchor.Anchor(
 472         self._min_level, self._max_level, self._num_scales,
 473         self._aspect_ratios, self._anchor_size, self._output_size)
 474     anchor_labeler = anchor.AnchorLabeler(
 475         input_anchor, self._match_threshold, self._unmatched_threshold)
 476 
 477     # If bfloat16 is used, casts input image to tf.bfloat16.
 478     if self._use_bfloat16:
 479       image = tf.cast(image, dtype=tf.bfloat16)
 480 
 481     labels = {
 482         'anchor_boxes': input_anchor.multilevel_boxes,
 483         'image_info': image_info,
 484     }
 485     if self._mode == ModeKeys.PREDICT_WITH_GT:
 486       # Converts boxes from normalized coordinates to pixel coordinates.
 487       groundtruths = {
 488           'source_id': data['source_id'],
 489           'height': data['height'],
 490           'width': data['width'],
 491           'num_detections': tf.shape(data['groundtruth_classes']),
 492           'boxes': box_utils.denormalize_boxes(
 493               data['groundtruth_boxes'], image_shape),
 494           'classes': data['groundtruth_classes'],
 495           # 'masks': tf.squeeze(masks, axis=-1),
 496           'areas': data['groundtruth_area'],
 497           'is_crowds': tf.cast(data['groundtruth_is_crowd'], tf.int32),
 498       }
 499       groundtruths['source_id'] = dataloader_utils.process_source_id(
 500           groundtruths['source_id'])
 501       groundtruths = dataloader_utils.pad_groundtruths_to_fixed_size(
 502           groundtruths, self._max_num_instances)
 503       # Computes training labels.
 504       (cls_targets,
 505        box_targets,
 506        num_positives) = anchor_labeler.label_anchors(
 507            boxes,
 508            tf.cast(tf.expand_dims(classes, axis=1), tf.float32))
 509       # Packs labels for model_fn outputs.
 510       labels.update({
 511           'cls_targets': cls_targets,
 512           'box_targets': box_targets,
 513           'num_positives': num_positives,
 514           'groundtruths': groundtruths,
 515       })
 516 
 517     inputs = {
 518         'image': image,
 519         'image_info': image_info,
 520     }
 521 
 522     return inputs, labels
