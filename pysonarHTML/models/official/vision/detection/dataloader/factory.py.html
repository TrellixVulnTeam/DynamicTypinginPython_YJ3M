  21 from official.vision.detection.dataloader import maskrcnn_parser
  22 from official.vision.detection.dataloader import retinanet_parser
  23 from official.vision.detection.dataloader import shapemask_parser

  26 def parser_generator(params, mode):
  27   """Generator function for various dataset parser."""
  28   if params.architecture.parser == 'retinanet_parser':
  29     anchor_params = params.anchor
  30     parser_params = params.retinanet_parser
  31     parser_fn = retinanet_parser.Parser(
  32         output_size=parser_params.output_size,
  33         min_level=params.architecture.min_level,
  34         max_level=params.architecture.max_level,
  35         num_scales=anchor_params.num_scales,
  36         aspect_ratios=anchor_params.aspect_ratios,
  37         anchor_size=anchor_params.anchor_size,
  38         match_threshold=parser_params.match_threshold,
  39         unmatched_threshold=parser_params.unmatched_threshold,
  40         aug_rand_hflip=parser_params.aug_rand_hflip,
  41         aug_scale_min=parser_params.aug_scale_min,
  42         aug_scale_max=parser_params.aug_scale_max,
  43         use_autoaugment=parser_params.use_autoaugment,
  44         autoaugment_policy_name=parser_params.autoaugment_policy_name,
  45         skip_crowd_during_training=parser_params.skip_crowd_during_training,
  46         max_num_instances=parser_params.max_num_instances,
  47         use_bfloat16=params.architecture.use_bfloat16,
  48         mode=mode)
  49   elif params.architecture.parser == 'maskrcnn_parser':
  50     anchor_params = params.anchor
  51     parser_params = params.maskrcnn_parser
  52     parser_fn = maskrcnn_parser.Parser(
  53         output_size=parser_params.output_size,
  54         min_level=params.architecture.min_level,
  55         max_level=params.architecture.max_level,
  56         num_scales=anchor_params.num_scales,
  57         aspect_ratios=anchor_params.aspect_ratios,
  58         anchor_size=anchor_params.anchor_size,
  59         rpn_match_threshold=parser_params.rpn_match_threshold,
  60         rpn_unmatched_threshold=parser_params.rpn_unmatched_threshold,
  61         rpn_batch_size_per_im=parser_params.rpn_batch_size_per_im,
  62         rpn_fg_fraction=parser_params.rpn_fg_fraction,
  63         aug_rand_hflip=parser_params.aug_rand_hflip,
  64         aug_scale_min=parser_params.aug_scale_min,
  65         aug_scale_max=parser_params.aug_scale_max,
  66         skip_crowd_during_training=parser_params.skip_crowd_during_training,
  67         max_num_instances=parser_params.max_num_instances,
  68         include_mask=params.architecture.include_mask,
  69         mask_crop_size=parser_params.mask_crop_size,
  70         use_bfloat16=params.architecture.use_bfloat16,
  71         mode=mode)
  72   elif params.architecture.parser == 'shapemask_parser':
  73     anchor_params = params.anchor
  74     parser_params = params.shapemask_parser
  75     parser_fn = shapemask_parser.Parser(
  76         output_size=parser_params.output_size,
  77         min_level=params.architecture.min_level,
  78         max_level=params.architecture.max_level,
  79         num_scales=anchor_params.num_scales,
  80         aspect_ratios=anchor_params.aspect_ratios,
  81         anchor_size=anchor_params.anchor_size,
  82         use_category=parser_params.use_category,
  83         outer_box_scale=parser_params.outer_box_scale,
  84         box_jitter_scale=parser_params.box_jitter_scale,
  85         num_sampled_masks=parser_params.num_sampled_masks,
  86         mask_crop_size=parser_params.mask_crop_size,
  87         mask_min_level=parser_params.mask_min_level,
  88         mask_max_level=parser_params.mask_max_level,
  89         upsample_factor=parser_params.upsample_factor,
  90         match_threshold=parser_params.match_threshold,
  91         unmatched_threshold=parser_params.unmatched_threshold,
  92         aug_rand_hflip=parser_params.aug_rand_hflip,
  93         aug_scale_min=parser_params.aug_scale_min,
  94         aug_scale_max=parser_params.aug_scale_max,
  95         skip_crowd_during_training=parser_params.skip_crowd_during_training,
  96         max_num_instances=parser_params.max_num_instances,
  97         use_bfloat16=params.architecture.use_bfloat16,
  98         mask_train_class=parser_params.mask_train_class,
  99         mode=mode)
 100   else:
 101     raise ValueError('Parser %s is not supported.' % params.architecture.parser)
 102 
 103   return parser_fn
