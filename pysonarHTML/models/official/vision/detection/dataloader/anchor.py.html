  31 class Anchor(object):
  32   """Anchor class for anchor-based object detectors."""
  33 
  34   def __init__(self,
  35                min_level,
  36                max_level,
  37                num_scales,
  38                aspect_ratios,
  39                anchor_size,
  40                image_size):
  41     """Constructs multiscale anchors.
  42 
  43     Args:
  44       min_level: integer number of minimum level of the output feature pyramid.
  45       max_level: integer number of maximum level of the output feature pyramid.
  46       num_scales: integer number representing intermediate scales added
  47         on each level. For instances, num_scales=2 adds one additional
  48         intermediate anchor scales [2^0, 2^0.5] on each level.
  49       aspect_ratios: list of float numbers representing the aspect raito anchors
  50         added on each level. The number indicates the ratio of width to height.
  51         For instances, aspect_ratios=[1.0, 2.0, 0.5] adds three anchors on each
  52         scale level.
  53       anchor_size: float number representing the scale of size of the base
  54         anchor to the feature stride 2^level.
  55       image_size: a list of integer numbers or Tensors representing
  56         [height, width] of the input image size.The image_size should be divided
  57         by the largest feature stride 2^max_level.
  58     """
  59     self.min_level = min_level
  60     self.max_level = max_level
  61     self.num_scales = num_scales
  62     self.aspect_ratios = aspect_ratios
  63     self.anchor_size = anchor_size
  64     self.image_size = image_size
  65     self.boxes = self._generate_boxes()
  66 
  67   def _generate_boxes(self):
  68     """Generates multiscale anchor boxes.
  69 
  70     Returns:
  71       a Tensor of shape [N, 4], represneting anchor boxes of all levels
  72       concatenated together.
  73     """
  74     boxes_all = []
  75     for level in range(self.min_level, self.max_level + 1):
  76       boxes_l = []
  77       for scale in range(self.num_scales):
  78         for aspect_ratio in self.aspect_ratios:
  79           stride = 2 ** level
  80           intermidate_scale = 2 ** (scale / float(self.num_scales))
  81           base_anchor_size = self.anchor_size * stride * intermidate_scale
  82           aspect_x = aspect_ratio ** 0.5
  83           aspect_y = aspect_ratio ** -0.5
  84           half_anchor_size_x = base_anchor_size * aspect_x / 2.0
  85           half_anchor_size_y = base_anchor_size * aspect_y / 2.0
  86           x = tf.range(stride / 2, self.image_size[1], stride)
  87           y = tf.range(stride / 2, self.image_size[0], stride)
  88           xv, yv = tf.meshgrid(x, y)
  89           xv = tf.cast(tf.reshape(xv, [-1]), dtype=tf.float32)
  90           yv = tf.cast(tf.reshape(yv, [-1]), dtype=tf.float32)
  91           # Tensor shape Nx4.
  92           boxes = tf.stack([yv - half_anchor_size_y, xv - half_anchor_size_x,
  93                             yv + half_anchor_size_y, xv + half_anchor_size_x],
  94                            axis=1)
  95           boxes_l.append(boxes)
  96       # Concat anchors on the same level to tensor shape NxAx4.
  97       boxes_l = tf.stack(boxes_l, axis=1)
  98       boxes_l = tf.reshape(boxes_l, [-1, 4])
  99       boxes_all.append(boxes_l)
 100     return tf.concat(boxes_all, axis=0)
 101 
 102   def unpack_labels(self, labels):
 103     """Unpacks an array of labels into multiscales labels."""
 104     unpacked_labels = collections.OrderedDict()
 105     count = 0
 106     for level in range(self.min_level, self.max_level + 1):
 107       feat_size_y = tf.cast(self.image_size[0] / 2 ** level, tf.int32)
 108       feat_size_x = tf.cast(self.image_size[1] / 2 ** level, tf.int32)
 109       steps = feat_size_y * feat_size_x * self.anchors_per_location
 110       unpacked_labels[level] = tf.reshape(
 111           labels[count:count + steps], [feat_size_y, feat_size_x, -1])
 112       count += steps
 113     return unpacked_labels
 114 
 115   @property
 116   def anchors_per_location(self):
 117     return self.num_scales * len(self.aspect_ratios)
 118 
 119   @property
 120   def multilevel_boxes(self):
 121     return self.unpack_labels(self.boxes)

 124 class AnchorLabeler(object):
 125   """Labeler for dense object detector."""
 126 
 127   def __init__(self,
 128                anchor,
 129                match_threshold=0.5,
 130                unmatched_threshold=0.5):
 131     """Constructs anchor labeler to assign labels to anchors.
 132 
 133     Args:
 134       anchor: an instance of class Anchors.
 135       match_threshold: a float number between 0 and 1 representing the
 136         lower-bound threshold to assign positive labels for anchors. An anchor
 137         with a score over the threshold is labeled positive.
 138       unmatched_threshold: a float number between 0 and 1 representing the
 139         upper-bound threshold to assign negative labels for anchors. An anchor
 140         with a score below the threshold is labeled negative.
 141     """
 142     similarity_calc = region_similarity_calculator.IouSimilarity()
 143     matcher = argmax_matcher.ArgMaxMatcher(
 144         match_threshold,
 145         unmatched_threshold=unmatched_threshold,
 146         negatives_lower_than_unmatched=True,
 147         force_match_for_each_row=True)
 148     box_coder = faster_rcnn_box_coder.FasterRcnnBoxCoder()
 149 
 150     self._target_assigner = target_assigner.TargetAssigner(
 151         similarity_calc, matcher, box_coder)
 152     self._anchor = anchor
 153     self._match_threshold = match_threshold
 154     self._unmatched_threshold = unmatched_threshold
 155 
 156   def label_anchors(self, gt_boxes, gt_labels):
 157     """Labels anchors with ground truth inputs.
 158 
 159     Args:
 160       gt_boxes: A float tensor with shape [N, 4] representing groundtruth boxes.
 161         For each row, it stores [y0, x0, y1, x1] for four corners of a box.
 162       gt_labels: A integer tensor with shape [N, 1] representing groundtruth
 163         classes.
 164     Returns:
 165       cls_targets_dict: ordered dictionary with keys
 166         [min_level, min_level+1, ..., max_level]. The values are tensor with
 167         shape [height_l, width_l, num_anchors_per_location]. The height_l and
 168         width_l represent the dimension of class logits at l-th level.
 169       box_targets_dict: ordered dictionary with keys
 170         [min_level, min_level+1, ..., max_level]. The values are tensor with
 171         shape [height_l, width_l, num_anchors_per_location * 4]. The height_l
 172         and width_l represent the dimension of bounding box regression output at
 173         l-th level.
 174       num_positives: scalar tensor storing number of positives in an image.
 175     """
 176     gt_box_list = box_list.BoxList(gt_boxes)
 177     anchor_box_list = box_list.BoxList(self._anchor.boxes)
 178 
 179     # The cls_weights, box_weights are not used.
 180     cls_targets, _, box_targets, _, matches = self._target_assigner.assign(
 181         anchor_box_list, gt_box_list, gt_labels)
 182 
 183     # Labels definition in matches.match_results:
 184     # (1) match_results[i]>=0, meaning that column i is matched with row
 185     #     match_results[i].
 186     # (2) match_results[i]=-1, meaning that column i is not matched.
 187     # (3) match_results[i]=-2, meaning that column i is ignored.
 188     match_results = tf.expand_dims(matches.match_results, axis=1)
 189     cls_targets = tf.cast(cls_targets, tf.int32)
 190     cls_targets = tf.where(
 191         tf.equal(match_results, -1), -tf.ones_like(cls_targets), cls_targets)
 192     cls_targets = tf.where(
 193         tf.equal(match_results, -2), -2 * tf.ones_like(cls_targets),
 194         cls_targets)
 195 
 196     # Unpacks labels into multi-level representations.
 197     cls_targets_dict = self._anchor.unpack_labels(cls_targets)
 198     box_targets_dict = self._anchor.unpack_labels(box_targets)
 199     num_positives = tf.reduce_sum(
 200         input_tensor=tf.cast(tf.greater(matches.match_results, -1), tf.float32))
 201 
 202     return cls_targets_dict, box_targets_dict, num_positives

 205 class RpnAnchorLabeler(AnchorLabeler):
 206   """Labeler for Region Proposal Network."""
 207 
 208   def __init__(self, anchor, match_threshold=0.7,
 209                unmatched_threshold=0.3, rpn_batch_size_per_im=256,
 210                rpn_fg_fraction=0.5):
 211     AnchorLabeler.__init__(self, anchor, match_threshold=0.7,
 212                            unmatched_threshold=0.3)
 213     self._rpn_batch_size_per_im = rpn_batch_size_per_im
 214     self._rpn_fg_fraction = rpn_fg_fraction
 215 
 216   def _get_rpn_samples(self, match_results):
 217     """Computes anchor labels.
 218 
 219     This function performs subsampling for foreground (fg) and background (bg)
 220     anchors.
 221     Args:
 222       match_results: A integer tensor with shape [N] representing the
 223         matching results of anchors. (1) match_results[i]>=0,
 224         meaning that column i is matched with row match_results[i].
 225         (2) match_results[i]=-1, meaning that column i is not matched.
 226         (3) match_results[i]=-2, meaning that column i is ignored.
 227     Returns:
 228       score_targets: a integer tensor with the a shape of [N].
 229         (1) score_targets[i]=1, the anchor is a positive sample.
 230         (2) score_targets[i]=0, negative. (3) score_targets[i]=-1, the anchor is
 231         don't care (ignore).
 232     """
 233     sampler = (
 234         balanced_positive_negative_sampler.BalancedPositiveNegativeSampler(
 235             positive_fraction=self._rpn_fg_fraction, is_static=False))
 236     # indicator includes both positive and negative labels.
 237     # labels includes only positives labels.
 238     # positives = indicator & labels.
 239     # negatives = indicator & !labels.
 240     # ignore = !indicator.
 241     indicator = tf.greater(match_results, -2)
 242     labels = tf.greater(match_results, -1)
 243 
 244     samples = sampler.subsample(
 245         indicator, self._rpn_batch_size_per_im, labels)
 246     positive_labels = tf.where(
 247         tf.logical_and(samples, labels),
 248         tf.constant(2, dtype=tf.int32, shape=match_results.shape),
 249         tf.constant(0, dtype=tf.int32, shape=match_results.shape))
 250     negative_labels = tf.where(
 251         tf.logical_and(samples, tf.logical_not(labels)),
 252         tf.constant(1, dtype=tf.int32, shape=match_results.shape),
 253         tf.constant(0, dtype=tf.int32, shape=match_results.shape))
 254     ignore_labels = tf.fill(match_results.shape, -1)
 255 
 256     return (ignore_labels + positive_labels + negative_labels,
 257             positive_labels, negative_labels)
 258 
 259   def label_anchors(self, gt_boxes, gt_labels):
 260     """Labels anchors with ground truth inputs.
 261 
 262     Args:
 263       gt_boxes: A float tensor with shape [N, 4] representing groundtruth boxes.
 264         For each row, it stores [y0, x0, y1, x1] for four corners of a box.
 265       gt_labels: A integer tensor with shape [N, 1] representing groundtruth
 266         classes.
 267     Returns:
 268       score_targets_dict: ordered dictionary with keys
 269         [min_level, min_level+1, ..., max_level]. The values are tensor with
 270         shape [height_l, width_l, num_anchors]. The height_l and width_l
 271         represent the dimension of class logits at l-th level.
 272       box_targets_dict: ordered dictionary with keys
 273         [min_level, min_level+1, ..., max_level]. The values are tensor with
 274         shape [height_l, width_l, num_anchors * 4]. The height_l and
 275         width_l represent the dimension of bounding box regression output at
 276         l-th level.
 277     """
 278     gt_box_list = box_list.BoxList(gt_boxes)
 279     anchor_box_list = box_list.BoxList(self._anchor.boxes)
 280 
 281     # cls_targets, cls_weights, box_weights are not used.
 282     _, _, box_targets, _, matches = self._target_assigner.assign(
 283         anchor_box_list, gt_box_list, gt_labels)
 284 
 285     # score_targets contains the subsampled positive and negative anchors.
 286     score_targets, _, _ = self._get_rpn_samples(matches.match_results)
 287 
 288     # Unpacks labels.
 289     score_targets_dict = self._anchor.unpack_labels(score_targets)
 290     box_targets_dict = self._anchor.unpack_labels(box_targets)
 291 
 292     return score_targets_dict, box_targets_dict
