  16 """Tensorflow Example proto decoder for object detection.
  17 
  18 A decoder to decode string tensors containing serialized tensorflow.Example
  19 protos for object detection.
  20 """
  21 import tensorflow as tf
  22 
  23 
  24 class TfExampleDecoder(object):
  25   """Tensorflow Example proto decoder."""
  26 
  27   def __init__(self, include_mask=False):
  28     self._include_mask = include_mask
  29     self._keys_to_features = {
  30         'image/encoded':
  31             tf.io.FixedLenFeature((), tf.string),
  32         'image/source_id':
  33             tf.io.FixedLenFeature((), tf.string),
  34         'image/height':
  35             tf.io.FixedLenFeature((), tf.int64),
  36         'image/width':
  37             tf.io.FixedLenFeature((), tf.int64),
  38         'image/object/bbox/xmin':
  39             tf.io.VarLenFeature(tf.float32),
  40         'image/object/bbox/xmax':
  41             tf.io.VarLenFeature(tf.float32),
  42         'image/object/bbox/ymin':
  43             tf.io.VarLenFeature(tf.float32),
  44         'image/object/bbox/ymax':
  45             tf.io.VarLenFeature(tf.float32),
  46         'image/object/class/label':
  47             tf.io.VarLenFeature(tf.int64),
  48         'image/object/area':
  49             tf.io.VarLenFeature(tf.float32),
  50         'image/object/is_crowd':
  51             tf.io.VarLenFeature(tf.int64),
  52     }
  53     if include_mask:
  54       self._keys_to_features.update({
  55           'image/object/mask':
  56               tf.io.VarLenFeature(tf.string),
  57       })
  58 
  59   def _decode_image(self, parsed_tensors):
  60     """Decodes the image and set its static shape."""
  61     image = tf.io.decode_image(parsed_tensors['image/encoded'], channels=3)
  62     image.set_shape([None, None, 3])
  63     return image
  64 
  65   def _decode_boxes(self, parsed_tensors):
  66     """Concat box coordinates in the format of [ymin, xmin, ymax, xmax]."""
  67     xmin = parsed_tensors['image/object/bbox/xmin']
  68     xmax = parsed_tensors['image/object/bbox/xmax']
  69     ymin = parsed_tensors['image/object/bbox/ymin']
  70     ymax = parsed_tensors['image/object/bbox/ymax']
  71     return tf.stack([ymin, xmin, ymax, xmax], axis=-1)
  72 
  73   def _decode_masks(self, parsed_tensors):
  74     """Decode a set of PNG masks to the tf.float32 tensors."""
  75     def _decode_png_mask(png_bytes):
  76       mask = tf.squeeze(
  77           tf.io.decode_png(png_bytes, channels=1, dtype=tf.uint8), axis=-1)
  78       mask = tf.cast(mask, dtype=tf.float32)
  79       mask.set_shape([None, None])
  80       return mask
  81 
  82     height = parsed_tensors['image/height']
  83     width = parsed_tensors['image/width']
  84     masks = parsed_tensors['image/object/mask']
  85     return tf.cond(
  86         pred=tf.greater(tf.size(input=masks), 0),
  87         true_fn=lambda: tf.map_fn(_decode_png_mask, masks, dtype=tf.float32),
  88         false_fn=lambda: tf.zeros([0, height, width], dtype=tf.float32))
  89 
  90   def _decode_areas(self, parsed_tensors):
  91     xmin = parsed_tensors['image/object/bbox/xmin']
  92     xmax = parsed_tensors['image/object/bbox/xmax']
  93     ymin = parsed_tensors['image/object/bbox/ymin']
  94     ymax = parsed_tensors['image/object/bbox/ymax']
  95     return tf.cond(
  96         tf.greater(tf.shape(parsed_tensors['image/object/area'])[0], 0),
  97         lambda: parsed_tensors['image/object/area'],
  98         lambda: (xmax - xmin) * (ymax - ymin))
  99 
 100   def decode(self, serialized_example):
 101     """Decode the serialized example.
 102 
 103     Args:
 104       serialized_example: a single serialized tf.Example string.
 105 
 106     Returns:
 107       decoded_tensors: a dictionary of tensors with the following fields:
 108         - image: a uint8 tensor of shape [None, None, 3].
 109         - source_id: a string scalar tensor.
 110         - height: an integer scalar tensor.
 111         - width: an integer scalar tensor.
 112         - groundtruth_classes: a int64 tensor of shape [None].
 113         - groundtruth_is_crowd: a bool tensor of shape [None].
 114         - groundtruth_area: a float32 tensor of shape [None].
 115         - groundtruth_boxes: a float32 tensor of shape [None, 4].
 116         - groundtruth_instance_masks: a float32 tensor of shape
 117             [None, None, None].
 118         - groundtruth_instance_masks_png: a string tensor of shape [None].
 119     """
 120     parsed_tensors = tf.io.parse_single_example(
 121         serialized=serialized_example, features=self._keys_to_features)
 122     for k in parsed_tensors:
 123       if isinstance(parsed_tensors[k], tf.SparseTensor):
 124         if parsed_tensors[k].dtype == tf.string:
 125           parsed_tensors[k] = tf.sparse.to_dense(
 126               parsed_tensors[k], default_value='')
 127         else:
 128           parsed_tensors[k] = tf.sparse.to_dense(
 129               parsed_tensors[k], default_value=0)
 130 
 131     image = self._decode_image(parsed_tensors)
 132     boxes = self._decode_boxes(parsed_tensors)
 133     areas = self._decode_areas(parsed_tensors)
 134     is_crowds = tf.cond(
 135         tf.greater(tf.shape(parsed_tensors['image/object/is_crowd'])[0], 0),
 136         lambda: tf.cast(parsed_tensors['image/object/is_crowd'], dtype=tf.bool),
 137         lambda: tf.zeros_like(parsed_tensors['image/object/class/label'], dtype=tf.bool))  # pylint: disable=line-too-long
 138     if self._include_mask:
 139       masks = self._decode_masks(parsed_tensors)
 140 
 141     decoded_tensors = {
 142         'image': image,
 143         'source_id': parsed_tensors['image/source_id'],
 144         'height': parsed_tensors['image/height'],
 145         'width': parsed_tensors['image/width'],
 146         'groundtruth_classes': parsed_tensors['image/object/class/label'],
 147         'groundtruth_is_crowd': is_crowds,
 148         'groundtruth_area': areas,
 149         'groundtruth_boxes': boxes,
 150     }
 151     if self._include_mask:
 152       decoded_tensors.update({
 153           'groundtruth_instance_masks': masks,
 154           'groundtruth_instance_masks_png': parsed_tensors['image/object/mask'],
 155       })
 156     return decoded_tensors
