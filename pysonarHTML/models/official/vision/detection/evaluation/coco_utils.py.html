  37 class COCOWrapper(coco.COCO):
  38   """COCO wrapper class.
  39 
  40   This class wraps COCO API object, which provides the following additional
  41   functionalities:
  42     1. Support string type image id.
  43     2. Support loading the groundtruth dataset using the external annotation
  44        dictionary.
  45     3. Support loading the prediction results using the external annotation
  46        dictionary.
  47   """
  48 
  49   def __init__(self, eval_type='box', annotation_file=None, gt_dataset=None):
  50     """Instantiates a COCO-style API object.
  51 
  52     Args:
  53       eval_type: either 'box' or 'mask'.
  54       annotation_file: a JSON file that stores annotations of the eval dataset.
  55         This is required if `gt_dataset` is not provided.
  56       gt_dataset: the groundtruth eval datatset in COCO API format.
  57     """
  58     if ((annotation_file and gt_dataset) or
  59         ((not annotation_file) and (not gt_dataset))):
  60       raise ValueError('One and only one of `annotation_file` and `gt_dataset` '
  61                        'needs to be specified.')
  62 
  63     if eval_type not in ['box', 'mask']:
  64       raise ValueError('The `eval_type` can only be either `box` or `mask`.')
  65 
  66     coco.COCO.__init__(self, annotation_file=annotation_file)
  67     self._eval_type = eval_type
  68     if gt_dataset:
  69       self.dataset = gt_dataset
  70       self.createIndex()
  71 
  72   def loadRes(self, predictions):
  73     """Loads result file and return a result api object.
  74 
  75     Args:
  76       predictions: a list of dictionary each representing an annotation in COCO
  77         format. The required fields are `image_id`, `category_id`, `score`,
  78         `bbox`, `segmentation`.
  79 
  80     Returns:
  81       res: result COCO api object.
  82 
  83     Raises:
  84       ValueError: if the set of image id from predctions is not the subset of
  85         the set of image id of the groundtruth dataset.
  86     """
  87     res = coco.COCO()
  88     res.dataset['images'] = copy.deepcopy(self.dataset['images'])
  89     res.dataset['categories'] = copy.deepcopy(self.dataset['categories'])
  90 
  91     image_ids = [ann['image_id'] for ann in predictions]
  92     if set(image_ids) != (set(image_ids) & set(self.getImgIds())):
  93       raise ValueError('Results do not correspond to the current dataset!')
  94     for ann in predictions:
  95       x1, x2, y1, y2 = [ann['bbox'][0], ann['bbox'][0] + ann['bbox'][2],
  96                         ann['bbox'][1], ann['bbox'][1] + ann['bbox'][3]]
  97       if self._eval_type == 'box':
  98         ann['area'] = ann['bbox'][2] * ann['bbox'][3]
  99         ann['segmentation'] = [
 100             [x1, y1, x1, y2, x2, y2, x2, y1]]
 101       elif self._eval_type == 'mask':
 102         ann['area'] = mask_api.area(ann['segmentation'])
 103 
 104     res.dataset['annotations'] = copy.deepcopy(predictions)
 105     res.createIndex()
 106     return res

 109 def convert_predictions_to_coco_annotations(predictions):
 110   """Converts a batch of predictions to annotations in COCO format.
 111 
 112   Args:
 113     predictions: a dictionary of lists of numpy arrays including the following
 114       fields. K below denotes the maximum number of instances per image.
 115       Required fields:
 116         - source_id: a list of numpy arrays of int or string of shape
 117             [batch_size].
 118         - num_detections: a list of numpy arrays of int of shape [batch_size].
 119         - detection_boxes: a list of numpy arrays of float of shape
 120             [batch_size, K, 4], where coordinates are in the original image
 121             space (not the scaled image space).
 122         - detection_classes: a list of numpy arrays of int of shape
 123             [batch_size, K].
 124         - detection_scores: a list of numpy arrays of float of shape
 125             [batch_size, K].
 126       Optional fields:
 127         - detection_masks: a list of numpy arrays of float of shape
 128             [batch_size, K, mask_height, mask_width].
 129 
 130   Returns:
 131     coco_predictions: prediction in COCO annotation format.
 132   """
 133   coco_predictions = []
 134   num_batches = len(predictions['source_id'])
 135   batch_size = predictions['source_id'][0].shape[0]
 136   max_num_detections = predictions['detection_classes'][0].shape[1]
 137   use_outer_box = 'detection_outer_boxes' in predictions
 138   for i in range(num_batches):
 139     predictions['detection_boxes'][i] = box_utils.yxyx_to_xywh(
 140         predictions['detection_boxes'][i])
 141     if use_outer_box:
 142       predictions['detection_outer_boxes'][i] = box_utils.yxyx_to_xywh(
 143           predictions['detection_outer_boxes'][i])
 144       mask_boxes = predictions['detection_outer_boxes']
 145     else:
 146       mask_boxes = predictions['detection_boxes']
 147 
 148     for j in range(batch_size):
 149       if 'detection_masks' in predictions:
 150         image_masks = mask_utils.paste_instance_masks(
 151             predictions['detection_masks'][i][j],
 152             mask_boxes[i][j],
 153             int(predictions['image_info'][i][j, 0, 0]),
 154             int(predictions['image_info'][i][j, 0, 1]))
 155         binary_masks = (image_masks > 0.0).astype(np.uint8)
 156         encoded_masks = [
 157             mask_api.encode(np.asfortranarray(binary_mask))
 158             for binary_mask in list(binary_masks)]
 159       for k in range(max_num_detections):
 160         ann = {}
 161         ann['image_id'] = predictions['source_id'][i][j]
 162         ann['category_id'] = predictions['detection_classes'][i][j, k]
 163         ann['bbox'] = predictions['detection_boxes'][i][j, k]
 164         ann['score'] = predictions['detection_scores'][i][j, k]
 165         if 'detection_masks' in predictions:
 166           ann['segmentation'] = encoded_masks[k]
 167         coco_predictions.append(ann)
 168 
 169   for i, ann in enumerate(coco_predictions):
 170     ann['id'] = i + 1
 171 
 172   return coco_predictions

 175 def convert_groundtruths_to_coco_dataset(groundtruths, label_map=None):
 176   """Converts groundtruths to the dataset in COCO format.
 177 
 178   Args:
 179     groundtruths: a dictionary of numpy arrays including the fields below.
 180       Note that each element in the list represent the number for a single
 181       example without batch dimension. K below denotes the actual number of
 182       instances for each image.
 183       Required fields:
 184         - source_id: a list of numpy arrays of int or string of shape
 185           [batch_size].
 186         - height: a list of numpy arrays of int of shape [batch_size].
 187         - width: a list of numpy arrays of int of shape [batch_size].
 188         - num_detections: a list of numpy arrays of int of shape [batch_size].
 189         - boxes: a list of numpy arrays of float of shape [batch_size, K, 4],
 190             where coordinates are in the original image space (not the
 191             normalized coordinates).
 192         - classes: a list of numpy arrays of int of shape [batch_size, K].
 193       Optional fields:
 194         - is_crowds: a list of numpy arrays of int of shape [batch_size, K]. If
 195             th field is absent, it is assumed that this instance is not crowd.
 196         - areas: a list of numy arrays of float of shape [batch_size, K]. If the
 197             field is absent, the area is calculated using either boxes or
 198             masks depending on which one is available.
 199         - masks: a list of numpy arrays of string of shape [batch_size, K],
 200     label_map: (optional) a dictionary that defines items from the category id
 201       to the category name. If `None`, collect the category mappping from the
 202       `groundtruths`.
 203 
 204   Returns:
 205     coco_groundtruths: the groundtruth dataset in COCO format.
 206   """
 207   source_ids = np.concatenate(groundtruths['source_id'], axis=0)
 208   heights = np.concatenate(groundtruths['height'], axis=0)
 209   widths = np.concatenate(groundtruths['width'], axis=0)
 210   gt_images = [{'id': int(i), 'height': int(h), 'width': int(w)} for i, h, w
 211                in zip(source_ids, heights, widths)]
 212 
 213   gt_annotations = []
 214   num_batches = len(groundtruths['source_id'])
 215   batch_size = groundtruths['source_id'][0].shape[0]
 216   for i in range(num_batches):
 217     for j in range(batch_size):
 218       num_instances = groundtruths['num_detections'][i][j]
 219       for k in range(num_instances):
 220         ann = {}
 221         ann['image_id'] = int(groundtruths['source_id'][i][j])
 222         if 'is_crowds' in groundtruths:
 223           ann['iscrowd'] = int(groundtruths['is_crowds'][i][j, k])
 224         else:
 225           ann['iscrowd'] = 0
 226         ann['category_id'] = int(groundtruths['classes'][i][j, k])
 227         boxes = groundtruths['boxes'][i]
 228         ann['bbox'] = [
 229             float(boxes[j, k, 1]),
 230             float(boxes[j, k, 0]),
 231             float(boxes[j, k, 3] - boxes[j, k, 1]),
 232             float(boxes[j, k, 2] - boxes[j, k, 0])]
 233         if 'areas' in groundtruths:
 234           ann['area'] = float(groundtruths['areas'][i][j, k])
 235         else:
 236           ann['area'] = float(
 237               (boxes[j, k, 3] - boxes[j, k, 1]) *
 238               (boxes[j, k, 2] - boxes[j, k, 0]))
 239         if 'masks' in groundtruths:
 240           mask = Image.open(six.StringIO(groundtruths['masks'][i][j, k]))
 241           width, height = mask.size
 242           np_mask = (
 243               np.array(mask.getdata()).reshape(height, width).astype(np.uint8))
 244           np_mask[np_mask > 0] = 255
 245           encoded_mask = mask_api.encode(np.asfortranarray(np_mask))
 246           ann['segmentation'] = encoded_mask
 247           if 'areas' not in groundtruths:
 248             ann['area'] = mask_api.area(encoded_mask)
 249         gt_annotations.append(ann)
 250 
 251   for i, ann in enumerate(gt_annotations):
 252     ann['id'] = i + 1
 253 
 254   if label_map:
 255     gt_categories = [{'id': i, 'name': label_map[i]} for i in label_map]
 256   else:
 257     category_ids = [gt['category_id'] for gt in gt_annotations]
 258     gt_categories = [{'id': i} for i in set(category_ids)]
 259 
 260   gt_dataset = {
 261       'images': gt_images,
 262       'categories': gt_categories,
 263       'annotations': copy.deepcopy(gt_annotations),
 264   }
 265   return gt_dataset

 268 class COCOGroundtruthGenerator(object):
 269   """Generates the groundtruth annotations from a single example."""
 270 
 271   def __init__(self, file_pattern, num_examples, include_mask):
 272     self._file_pattern = file_pattern
 273     self._num_examples = num_examples
 274     self._include_mask = include_mask
 275     self._dataset_fn = tf.data.TFRecordDataset
 276 
 277   def _parse_single_example(self, example):
 278     """Parses a single serialized tf.Example proto.
 279 
 280     Args:
 281       example: a serialized tf.Example proto string.
 282 
 283     Returns:
 284       A dictionary of groundtruth with the following fields:
 285         source_id: a scalar tensor of int64 representing the image source_id.
 286         height: a scalar tensor of int64 representing the image height.
 287         width: a scalar tensor of int64 representing the image width.
 288         boxes: a float tensor of shape [K, 4], representing the groundtruth
 289           boxes in absolute coordinates with respect to the original image size.
 290         classes: a int64 tensor of shape [K], representing the class labels of
 291           each instances.
 292         is_crowds: a bool tensor of shape [K], indicating whether the instance
 293           is crowd.
 294         areas: a float tensor of shape [K], indicating the area of each
 295           instance.
 296         masks: a string tensor of shape [K], containing the bytes of the png
 297           mask of each instance.
 298     """
 299     decoder = tf_example_decoder.TfExampleDecoder(
 300         include_mask=self._include_mask)
 301     decoded_tensors = decoder.decode(example)
 302 
 303     image = decoded_tensors['image']
 304     image_size = tf.shape(image)[0:2]
 305     boxes = box_utils.denormalize_boxes(
 306         decoded_tensors['groundtruth_boxes'], image_size)
 307     groundtruths = {
 308         'source_id': tf.string_to_number(
 309             decoded_tensors['source_id'], out_type=tf.int64),
 310         'height': decoded_tensors['height'],
 311         'width': decoded_tensors['width'],
 312         'num_detections': tf.shape(decoded_tensors['groundtruth_classes'])[0],
 313         'boxes': boxes,
 314         'classes': decoded_tensors['groundtruth_classes'],
 315         'is_crowds': decoded_tensors['groundtruth_is_crowd'],
 316         'areas': decoded_tensors['groundtruth_area'],
 317     }
 318     if self._include_mask:
 319       groundtruths.update({
 320           'masks': decoded_tensors['groundtruth_instance_masks_png'],
 321       })
 322     return groundtruths
 323 
 324   def _build_pipeline(self):
 325     """Builds data pipeline to generate groundtruth annotations."""
 326     dataset = tf.data.Dataset.list_files(self._file_pattern, shuffle=False)
 327     dataset = dataset.apply(
 328         tf.data.experimental.parallel_interleave(
 329             lambda filename: self._dataset_fn(filename).prefetch(1),
 330             cycle_length=32,
 331             sloppy=False))
 332     dataset = dataset.map(self._parse_single_example, num_parallel_calls=64)
 333     dataset = dataset.prefetch(tf.data.experimental.AUTOTUNE)
 334     dataset = dataset.batch(1, drop_remainder=False)
 335     return dataset
 336 
 337   def __call__(self):
 338     with tf.Graph().as_default():
 339       dataset = self._build_pipeline()
 340       groundtruth = dataset.make_one_shot_iterator().get_next()
 341 
 342       with tf.Session() as sess:
 343         for _ in range(self._num_examples):
 344           groundtruth_result = sess.run(groundtruth)
 345           yield groundtruth_result

 348 def scan_and_generator_annotation_file(file_pattern,
 349                                        num_samples,
 350                                        include_mask,
 351                                        annotation_file):
 352   """Scans and generate the COCO-style annotation JSON file given a dataset."""
 353   groundtruth_generator = COCOGroundtruthGenerator(
 354       file_pattern, num_samples, include_mask)
 355   generate_annotation_file(groundtruth_generator, annotation_file)
 356 
 357 
 358 def generate_annotation_file(groundtruth_generator,
 359                              annotation_file):
 360   """Generates COCO-style annotation JSON file given a groundtruth generator."""
 361   groundtruths = {}
 362   logging.info('Loading groundtruth annotations from dataset to memory...')
 363   for groundtruth in groundtruth_generator():
 364     for k, v in six.iteritems(groundtruth):
 365       if k not in groundtruths:
 366         groundtruths[k] = [v]
 367       else:
 368         groundtruths[k].append(v)
 369   gt_dataset = convert_groundtruths_to_coco_dataset(groundtruths)
 370 
 371   logging.info('Saving groundtruth annotations to the JSON file...')
 372   with tf.io.gfile.GFile(annotation_file, 'w') as f:
 373     f.write(json.dumps(gt_dataset))
 374   logging.info('Done saving the JSON file...')
