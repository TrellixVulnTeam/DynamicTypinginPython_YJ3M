  15 """The COCO-style evaluator.
  16 
  17 The following snippet demonstrates the use of interfaces:
  18 
  19   evaluator = COCOEvaluator(...)
  20   for _ in range(num_evals):
  21     for _ in range(num_batches_per_eval):
  22       predictions, groundtruth = predictor.predict(...)  # pop a batch.
  23       evaluator.update(predictions, groundtruths)  # aggregate internal stats.
  24     evaluator.evaluate()  # finish one full eval.
  25 
  26 See also: https://github.com/cocodataset/cocoapi/
  27 """

  45 class MetricWrapper(object):
  46   # This is only a wrapper for COCO metric and works on for numpy array. So it
  47   # doesn't inherit from tf.keras.layers.Layer or tf.keras.metrics.Metric.
  48 
  49   def __init__(self, evaluator):
  50     self._evaluator = evaluator
  51 
  52   def update_state(self, y_true, y_pred):
  53     labels = tf.nest.map_structure(lambda x: x.numpy(), y_true)
  54     outputs = tf.nest.map_structure(lambda x: x.numpy(), y_pred)
  55     groundtruths = {}
  56     predictions = {}
  57     for key, val in outputs.items():
  58       if isinstance(val, tuple):
  59         val = np.concatenate(val)
  60       predictions[key] = val
  61     for key, val in labels.items():
  62       if isinstance(val, tuple):
  63         val = np.concatenate(val)
  64       groundtruths[key] = val
  65     self._evaluator.update(predictions, groundtruths)
  66 
  67   def result(self):
  68     return self._evaluator.evaluate()
  69 
  70   def reset_states(self):
  71     return self._evaluator.reset()

  74 class COCOEvaluator(object):
  75   """COCO evaluation metric class."""
  76 
  77   def __init__(self, annotation_file, include_mask, need_rescale_bboxes=True):
  78     """Constructs COCO evaluation class.
  79 
  80     The class provides the interface to metrics_fn in TPUEstimator. The
  81     _update_op() takes detections from each image and push them to
  82     self.detections. The _evaluate() loads a JSON file in COCO annotation format
  83     as the groundtruths and runs COCO evaluation.
  84 
  85     Args:
  86       annotation_file: a JSON file that stores annotations of the eval dataset.
  87         If `annotation_file` is None, groundtruth annotations will be loaded
  88         from the dataloader.
  89       include_mask: a boolean to indicate whether or not to include the mask
  90         eval.
  91       need_rescale_bboxes: If true bboxes in `predictions` will be rescaled back
  92         to absolute values (`image_info` is needed in this case).
  93     """
  94     if annotation_file:
  95       if annotation_file.startswith('gs://'):
  96         _, local_val_json = tempfile.mkstemp(suffix='.json')
  97         tf.io.gfile.remove(local_val_json)
  98 
  99         tf.io.gfile.copy(annotation_file, local_val_json)
 100         atexit.register(tf.io.gfile.remove, local_val_json)
 101       else:
 102         local_val_json = annotation_file
 103       self._coco_gt = coco_utils.COCOWrapper(
 104           eval_type=('mask' if include_mask else 'box'),
 105           annotation_file=local_val_json)
 106     self._annotation_file = annotation_file
 107     self._include_mask = include_mask
 108     self._metric_names = [
 109         'AP', 'AP50', 'AP75', 'APs', 'APm', 'APl', 'ARmax1', 'ARmax10',
 110         'ARmax100', 'ARs', 'ARm', 'ARl'
 111     ]
 112     self._required_prediction_fields = [
 113         'source_id', 'num_detections', 'detection_classes', 'detection_scores',
 114         'detection_boxes'
 115     ]
 116     self._need_rescale_bboxes = need_rescale_bboxes
 117     if self._need_rescale_bboxes:
 118       self._required_prediction_fields.append('image_info')
 119     self._required_groundtruth_fields = [
 120         'source_id', 'height', 'width', 'classes', 'boxes'
 121     ]
 122     if self._include_mask:
 123       mask_metric_names = ['mask_' + x for x in self._metric_names]
 124       self._metric_names.extend(mask_metric_names)
 125       self._required_prediction_fields.extend(['detection_masks'])
 126       self._required_groundtruth_fields.extend(['masks'])
 127 
 128     self.reset()
 129 
 130   def reset(self):
 131     """Resets internal states for a fresh run."""
 132     self._predictions = {}
 133     if not self._annotation_file:
 134       self._groundtruths = {}
 135 
 136   def evaluate(self):
 137     """Evaluates with detections from all images with COCO API.
 138 
 139     Returns:
 140       coco_metric: float numpy array with shape [24] representing the
 141         coco-style evaluation metrics (box and mask).
 142     """
 143     if not self._annotation_file:
 144       logging.info('Thre is no annotation_file in COCOEvaluator.')
 145       gt_dataset = coco_utils.convert_groundtruths_to_coco_dataset(
 146           self._groundtruths)
 147       coco_gt = coco_utils.COCOWrapper(
 148           eval_type=('mask' if self._include_mask else 'box'),
 149           gt_dataset=gt_dataset)
 150     else:
 151       logging.info('Using annotation file: %s', self._annotation_file)
 152       coco_gt = self._coco_gt
 153     coco_predictions = coco_utils.convert_predictions_to_coco_annotations(
 154         self._predictions)
 155     coco_dt = coco_gt.loadRes(predictions=coco_predictions)
 156     image_ids = [ann['image_id'] for ann in coco_predictions]
 157 
 158     coco_eval = cocoeval.COCOeval(coco_gt, coco_dt, iouType='bbox')
 159     coco_eval.params.imgIds = image_ids
 160     coco_eval.evaluate()
 161     coco_eval.accumulate()
 162     coco_eval.summarize()
 163     coco_metrics = coco_eval.stats
 164 
 165     if self._include_mask:
 166       mcoco_eval = cocoeval.COCOeval(coco_gt, coco_dt, iouType='segm')
 167       mcoco_eval.params.imgIds = image_ids
 168       mcoco_eval.evaluate()
 169       mcoco_eval.accumulate()
 170       mcoco_eval.summarize()
 171       mask_coco_metrics = mcoco_eval.stats
 172 
 173     if self._include_mask:
 174       metrics = np.hstack((coco_metrics, mask_coco_metrics))
 175     else:
 176       metrics = coco_metrics
 177 
 178     # Cleans up the internal variables in order for a fresh eval next time.
 179     self.reset()
 180 
 181     metrics_dict = {}
 182     for i, name in enumerate(self._metric_names):
 183       metrics_dict[name] = metrics[i].astype(np.float32)
 184     return metrics_dict
 185 
 186   def _process_predictions(self, predictions):
 187     image_scale = np.tile(predictions['image_info'][:, 2:3, :], (1, 1, 2))
 188     predictions['detection_boxes'] = (
 189         predictions['detection_boxes'].astype(np.float32))
 190     predictions['detection_boxes'] /= image_scale
 191     if 'detection_outer_boxes' in predictions:
 192       predictions['detection_outer_boxes'] = (
 193           predictions['detection_outer_boxes'].astype(np.float32))
 194       predictions['detection_outer_boxes'] /= image_scale
 195 
 196   def update(self, predictions, groundtruths=None):
 197     """Update and aggregate detection results and groundtruth data.
 198 
 199     Args:
 200       predictions: a dictionary of numpy arrays including the fields below.
 201         See different parsers under `../dataloader` for more details.
 202         Required fields:
 203           - source_id: a numpy array of int or string of shape [batch_size].
 204           - image_info [if `need_rescale_bboxes` is True]: a numpy array of
 205             float of shape [batch_size, 4, 2].
 206           - num_detections: a numpy array of
 207             int of shape [batch_size].
 208           - detection_boxes: a numpy array of float of shape [batch_size, K, 4].
 209           - detection_classes: a numpy array of int of shape [batch_size, K].
 210           - detection_scores: a numpy array of float of shape [batch_size, K].
 211         Optional fields:
 212           - detection_masks: a numpy array of float of shape
 213               [batch_size, K, mask_height, mask_width].
 214       groundtruths: a dictionary of numpy arrays including the fields below.
 215         See also different parsers under `../dataloader` for more details.
 216         Required fields:
 217           - source_id: a numpy array of int or string of shape [batch_size].
 218           - height: a numpy array of int of shape [batch_size].
 219           - width: a numpy array of int of shape [batch_size].
 220           - num_detections: a numpy array of int of shape [batch_size].
 221           - boxes: a numpy array of float of shape [batch_size, K, 4].
 222           - classes: a numpy array of int of shape [batch_size, K].
 223         Optional fields:
 224           - is_crowds: a numpy array of int of shape [batch_size, K]. If the
 225               field is absent, it is assumed that this instance is not crowd.
 226           - areas: a numy array of float of shape [batch_size, K]. If the
 227               field is absent, the area is calculated using either boxes or
 228               masks depending on which one is available.
 229           - masks: a numpy array of float of shape
 230               [batch_size, K, mask_height, mask_width],
 231 
 232     Raises:
 233       ValueError: if the required prediction or groundtruth fields are not
 234         present in the incoming `predictions` or `groundtruths`.
 235     """
 236     for k in self._required_prediction_fields:
 237       if k not in predictions:
 238         raise ValueError(
 239             'Missing the required key `{}` in predictions!'.format(k))
 240     if self._need_rescale_bboxes:
 241       self._process_predictions(predictions)
 242     for k, v in six.iteritems(predictions):
 243       if k not in self._predictions:
 244         self._predictions[k] = [v]
 245       else:
 246         self._predictions[k].append(v)
 247 
 248     if not self._annotation_file:
 249       assert groundtruths
 250       for k in self._required_groundtruth_fields:
 251         if k not in groundtruths:
 252           raise ValueError(
 253               'Missing the required key `{}` in groundtruths!'.format(k))
 254       for k, v in six.iteritems(groundtruths):
 255         if k not in self._groundtruths:
 256           self._groundtruths[k] = [v]
 257         else:
 258           self._groundtruths[k].append(v)
 259

 261 class ShapeMaskCOCOEvaluator(COCOEvaluator):
 262   """COCO evaluation metric class for ShapeMask."""
 263 
 264   def __init__(self, mask_eval_class, **kwargs):
 265     """Constructs COCO evaluation class.
 266 
 267     The class provides the interface to metrics_fn in TPUEstimator. The
 268     _update_op() takes detections from each image and push them to
 269     self.detections. The _evaluate() loads a JSON file in COCO annotation format
 270     as the groundtruths and runs COCO evaluation.
 271 
 272     Args:
 273       mask_eval_class: the set of classes for mask evaluation.
 274       **kwargs: other keyword arguments passed to the parent class initializer.
 275     """
 276     super(ShapeMaskCOCOEvaluator, self).__init__(**kwargs)
 277     self._mask_eval_class = mask_eval_class
 278     self._eval_categories = class_utils.coco_split_class_ids(mask_eval_class)
 279     if mask_eval_class != 'all':
 280       self._metric_names = [
 281           x.replace('mask', 'novel_mask') for x in self._metric_names
 282       ]
 283 
 284   def evaluate(self):
 285     """Evaluates with detections from all images with COCO API.
 286 
 287     Returns:
 288       coco_metric: float numpy array with shape [24] representing the
 289         coco-style evaluation metrics (box and mask).
 290     """
 291     if not self._annotation_file:
 292       gt_dataset = coco_utils.convert_groundtruths_to_coco_dataset(
 293           self._groundtruths)
 294       coco_gt = coco_utils.COCOWrapper(
 295           eval_type=('mask' if self._include_mask else 'box'),
 296           gt_dataset=gt_dataset)
 297     else:
 298       coco_gt = self._coco_gt
 299     coco_predictions = coco_utils.convert_predictions_to_coco_annotations(
 300         self._predictions)
 301     coco_dt = coco_gt.loadRes(predictions=coco_predictions)
 302     image_ids = [ann['image_id'] for ann in coco_predictions]
 303 
 304     coco_eval = cocoeval.COCOeval(coco_gt, coco_dt, iouType='bbox')
 305     coco_eval.params.imgIds = image_ids
 306     coco_eval.evaluate()
 307     coco_eval.accumulate()
 308     coco_eval.summarize()
 309     coco_metrics = coco_eval.stats
 310 
 311     if self._include_mask:
 312       mcoco_eval = cocoeval.COCOeval(coco_gt, coco_dt, iouType='segm')
 313       mcoco_eval.params.imgIds = image_ids
 314       mcoco_eval.evaluate()
 315       mcoco_eval.accumulate()
 316       mcoco_eval.summarize()
 317       if self._mask_eval_class == 'all':
 318         metrics = np.hstack((coco_metrics, mcoco_eval.stats))
 319       else:
 320         mask_coco_metrics = mcoco_eval.category_stats
 321         val_catg_idx = np.isin(mcoco_eval.params.catIds,
 322                                self._eval_categories)
 323         # Gather the valid evaluation of the eval categories.
 324         if np.any(val_catg_idx):
 325           mean_val_metrics = []
 326           for mid in range(len(self._metric_names) // 2):
 327             mean_val_metrics.append(
 328                 np.nanmean(mask_coco_metrics[mid][val_catg_idx]))
 329 
 330           mean_val_metrics = np.array(mean_val_metrics)
 331         else:
 332           mean_val_metrics = np.zeros(len(self._metric_names) // 2)
 333         metrics = np.hstack((coco_metrics, mean_val_metrics))
 334     else:
 335       metrics = coco_metrics
 336 
 337     # Cleans up the internal variables in order for a fresh eval next time.
 338     self.reset()
 339 
 340     metrics_dict = {}
 341     for i, name in enumerate(self._metric_names):
 342       metrics_dict[name] = metrics[i].astype(np.float32)
 343     return metrics_dict
