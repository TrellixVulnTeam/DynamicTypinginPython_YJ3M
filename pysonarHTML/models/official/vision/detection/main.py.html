  66 def run_executor(params,
  67                  mode,
  68                  checkpoint_path=None,
  69                  train_input_fn=None,
  70                  eval_input_fn=None,
  71                  callbacks=None,
  72                  prebuilt_strategy=None):
  73   """Runs the object detection model on distribution strategy defined by the user."""
  74 
  75   if params.architecture.use_bfloat16:
  76     policy = tf.compat.v2.keras.mixed_precision.experimental.Policy(
  77         'mixed_bfloat16')
  78     tf.compat.v2.keras.mixed_precision.experimental.set_policy(policy)
  79 
  80   model_builder = model_factory.model_generator(params)
  81 
  82   if prebuilt_strategy is not None:
  83     strategy = prebuilt_strategy
  84   else:
  85     strategy_config = params.strategy_config
  86     distribution_utils.configure_cluster(strategy_config.worker_hosts,
  87                                          strategy_config.task_index)
  88     strategy = distribution_utils.get_distribution_strategy(
  89         distribution_strategy=params.strategy_type,
  90         num_gpus=strategy_config.num_gpus,
  91         all_reduce_alg=strategy_config.all_reduce_alg,
  92         num_packs=strategy_config.num_packs,
  93         tpu_address=strategy_config.tpu)
  94 
  95   num_workers = int(strategy.num_replicas_in_sync + 7) // 8
  96   is_multi_host = (int(num_workers) >= 2)
  97 
  98   if mode == 'train':
  99 
 100     def _model_fn(params):
 101       return model_builder.build_model(params, mode=ModeKeys.TRAIN)
 102 
 103     logging.info(
 104         'Train num_replicas_in_sync %d num_workers %d is_multi_host %s',
 105         strategy.num_replicas_in_sync, num_workers, is_multi_host)
 106 
 107     dist_executor = DetectionDistributedExecutor(
 108         strategy=strategy,
 109         params=params,
 110         model_fn=_model_fn,
 111         loss_fn=model_builder.build_loss_fn,
 112         is_multi_host=is_multi_host,
 113         predict_post_process_fn=model_builder.post_processing,
 114         trainable_variables_filter=model_builder
 115         .make_filter_trainable_variables_fn())
 116 
 117     if is_multi_host:
 118       train_input_fn = functools.partial(
 119           train_input_fn,
 120           batch_size=params.train.batch_size // strategy.num_replicas_in_sync)
 121 
 122     return dist_executor.train(
 123         train_input_fn=train_input_fn,
 124         model_dir=params.model_dir,
 125         iterations_per_loop=params.train.iterations_per_loop,
 126         total_steps=params.train.total_steps,
 127         init_checkpoint=model_builder.make_restore_checkpoint_fn(),
 128         custom_callbacks=callbacks,
 129         save_config=True)
 130   elif mode == 'eval' or mode == 'eval_once':
 131 
 132     def _model_fn(params):
 133       return model_builder.build_model(params, mode=ModeKeys.PREDICT_WITH_GT)
 134 
 135     logging.info('Eval num_replicas_in_sync %d num_workers %d is_multi_host %s',
 136                  strategy.num_replicas_in_sync, num_workers, is_multi_host)
 137 
 138     if is_multi_host:
 139       eval_input_fn = functools.partial(
 140           eval_input_fn,
 141           batch_size=params.eval.batch_size // strategy.num_replicas_in_sync)
 142 
 143     dist_executor = DetectionDistributedExecutor(
 144         strategy=strategy,
 145         params=params,
 146         model_fn=_model_fn,
 147         loss_fn=model_builder.build_loss_fn,
 148         is_multi_host=is_multi_host,
 149         predict_post_process_fn=model_builder.post_processing,
 150         trainable_variables_filter=model_builder
 151         .make_filter_trainable_variables_fn())
 152 
 153     if mode == 'eval':
 154       results = dist_executor.evaluate_from_model_dir(
 155           model_dir=params.model_dir,
 156           eval_input_fn=eval_input_fn,
 157           eval_metric_fn=model_builder.eval_metrics,
 158           eval_timeout=params.eval.eval_timeout,
 159           min_eval_interval=params.eval.min_eval_interval,
 160           total_steps=params.train.total_steps)
 161     else:
 162       # Run evaluation once for a single checkpoint.
 163       if not checkpoint_path:
 164         raise ValueError('checkpoint_path cannot be empty.')
 165       if tf.io.gfile.isdir(checkpoint_path):
 166         checkpoint_path = tf.train.latest_checkpoint(checkpoint_path)
 167       summary_writer = executor.SummaryWriter(params.model_dir, 'eval')
 168       results, _ = dist_executor.evaluate_checkpoint(
 169           checkpoint_path=checkpoint_path,
 170           eval_input_fn=eval_input_fn,
 171           eval_metric_fn=model_builder.eval_metrics,
 172           summary_writer=summary_writer)
 173     for k, v in results.items():
 174       logging.info('Final eval metric %s: %f', k, v)
 175     return results
 176   else:
 177     raise ValueError('Mode not found: %s.' % mode)

 180 def run(callbacks=None):
 181   keras_utils.set_session_config(enable_xla=FLAGS.enable_xla)
 182 
 183   params = config_factory.config_generator(FLAGS.model)
 184 
 185   params = params_dict.override_params_dict(
 186       params, FLAGS.config_file, is_strict=True)
 187 
 188   params = params_dict.override_params_dict(
 189       params, FLAGS.params_override, is_strict=True)
 190   params.override(
 191       {
 192           'strategy_type': FLAGS.strategy_type,
 193           'model_dir': FLAGS.model_dir,
 194           'strategy_config': executor.strategy_flags_dict(),
 195       },
 196       is_strict=False)
 197   params.validate()
 198   params.lock()
 199   pp = pprint.PrettyPrinter()
 200   params_str = pp.pformat(params.as_dict())
 201   logging.info('Model Parameters: %s', params_str)
 202 
 203   train_input_fn = None
 204   eval_input_fn = None
 205   training_file_pattern = FLAGS.training_file_pattern or params.train.train_file_pattern
 206   eval_file_pattern = FLAGS.eval_file_pattern or params.eval.eval_file_pattern
 207   if not training_file_pattern and not eval_file_pattern:
 208     raise ValueError('Must provide at least one of training_file_pattern and '
 209                      'eval_file_pattern.')
 210 
 211   if training_file_pattern:
 212     # Use global batch size for single host.
 213     train_input_fn = input_reader.InputFn(
 214         file_pattern=training_file_pattern,
 215         params=params,
 216         mode=input_reader.ModeKeys.TRAIN,
 217         batch_size=params.train.batch_size)
 218 
 219   if eval_file_pattern:
 220     eval_input_fn = input_reader.InputFn(
 221         file_pattern=eval_file_pattern,
 222         params=params,
 223         mode=input_reader.ModeKeys.PREDICT_WITH_GT,
 224         batch_size=params.eval.batch_size,
 225         num_examples=params.eval.eval_samples)
 226 
 227   if callbacks is None:
 228     callbacks = []
 229 
 230   if FLAGS.log_steps:
 231     callbacks.append(
 232         keras_utils.TimeHistory(
 233             batch_size=params.train.batch_size,
 234             log_steps=FLAGS.log_steps,
 235         ))
 236 
 237   return run_executor(
 238       params,
 239       FLAGS.mode,
 240       checkpoint_path=FLAGS.checkpoint_path,
 241       train_input_fn=train_input_fn,
 242       eval_input_fn=eval_input_fn,
 243       callbacks=callbacks)
