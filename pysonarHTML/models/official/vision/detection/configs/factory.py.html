  15 """Factory to provide model configs."""
  16 
  17 from official.modeling.hyperparams import params_dict
  18 from official.vision.detection.configs import maskrcnn_config
  19 from official.vision.detection.configs import retinanet_config
  20 from official.vision.detection.configs import shapemask_config
  21 
  22 
  23 def config_generator(model):
  24   """Model function generator."""
  25   if model == 'retinanet':
  26     default_config = retinanet_config.RETINANET_CFG
  27     restrictions = retinanet_config.RETINANET_RESTRICTIONS
  28   elif model == 'mask_rcnn':
  29     default_config = maskrcnn_config.MASKRCNN_CFG
  30     restrictions = maskrcnn_config.MASKRCNN_RESTRICTIONS
  31   elif model == 'shapemask':
  32     default_config = shapemask_config.SHAPEMASK_CFG
  33     restrictions = shapemask_config.SHAPEMASK_RESTRICTIONS
  34   else:
  35     raise ValueError('Model %s is not supported.' % model)
  36 
  37   return params_dict.ParamsDict(default_config, restrictions)
