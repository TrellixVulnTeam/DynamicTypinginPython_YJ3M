  27 # For ResNet, this freezes the variables of the first conv1 and conv2_x
  28 # layers [1], which leads to higher training speed and slightly better testing
  29 # accuracy. The intuition is that the low-level architecture (e.g., ResNet-50)
  30 # is able to capture low-level features such as edges; therefore, it does not
  31 # need to be fine-tuned for the detection task.
  32 # Note that we need to trailing `/` to avoid the incorrect match.
  33 # [1]: https://github.com/facebookresearch/Detectron/blob/master/detectron/core/config.py#L198
  34 RESNET_FROZEN_VAR_PREFIX = r'(resnet\d+)\/(conv2d(|_([1-9]|10))|batch_normalization(|_([1-9]|10)))\/'
