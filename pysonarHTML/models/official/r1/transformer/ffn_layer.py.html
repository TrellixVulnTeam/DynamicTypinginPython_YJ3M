  15 """Implementation of fully connected network."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import tensorflow.compat.v1 as tf
  22 
  23 
  24 class FeedFowardNetwork(tf.layers.Layer):
  25   """Fully connected feedforward network."""
  26 
  27   def __init__(self, hidden_size, filter_size, relu_dropout, train, allow_pad):
  28     super(FeedFowardNetwork, self).__init__()
  29     self.hidden_size = hidden_size
  30     self.filter_size = filter_size
  31     self.relu_dropout = relu_dropout
  32     self.train = train
  33     self.allow_pad = allow_pad
  34 
  35     self.filter_dense_layer = tf.layers.Dense(
  36         filter_size, use_bias=True, activation=tf.nn.relu, name="filter_layer")
  37     self.output_dense_layer = tf.layers.Dense(
  38         hidden_size, use_bias=True, name="output_layer")
  39 
  40   def call(self, x, padding=None):
  41     """Return outputs of the feedforward network.
  42 
  43     Args:
  44       x: tensor with shape [batch_size, length, hidden_size]
  45       padding: (optional) If set, the padding values are temporarily removed
  46         from x (provided self.allow_pad is set). The padding values are placed
  47         back in the output tensor in the same locations.
  48         shape [batch_size, length]
  49 
  50     Returns:
  51       Output of the feedforward network.
  52       tensor with shape [batch_size, length, hidden_size]
  53     """
  54     padding = None if not self.allow_pad else padding
  55 
  56     # Retrieve dynamically known shapes
  57     batch_size = tf.shape(x)[0]
  58     length = tf.shape(x)[1]
  59 
  60     if padding is not None:
  61       with tf.name_scope("remove_padding"):
  62         # Flatten padding to [batch_size*length]
  63         pad_mask = tf.reshape(padding, [-1])
  64 
  65         nonpad_ids = tf.to_int32(tf.where(pad_mask < 1e-9))
  66 
  67         # Reshape x to [batch_size*length, hidden_size] to remove padding
  68         x = tf.reshape(x, [-1, self.hidden_size])
  69         x = tf.gather_nd(x, indices=nonpad_ids)
  70 
  71         # Reshape x from 2 dimensions to 3 dimensions.
  72         x.set_shape([None, self.hidden_size])
  73         x = tf.expand_dims(x, axis=0)
  74 
  75     output = self.filter_dense_layer(x)
  76     if self.train:
  77       output = tf.nn.dropout(output, 1.0 - self.relu_dropout)
  78     output = self.output_dense_layer(output)
  79 
  80     if padding is not None:
  81       with tf.name_scope("re_add_padding"):
  82         output = tf.squeeze(output, axis=0)
  83         output = tf.scatter_nd(
  84             indices=nonpad_ids,
  85             updates=output,
  86             shape=[batch_size * length, self.hidden_size]
  87         )
  88         output = tf.reshape(output, [batch_size, length, self.hidden_size])
  89     return output
