  15 """Implementation of multiheaded attention and self-attention layers."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import tensorflow.compat.v1 as tf
  22 
  23 
  24 class Attention(tf.layers.Layer):
  25   """Multi-headed attention layer."""
  26 
  27   def __init__(self, hidden_size, num_heads, attention_dropout, train):
  28     if hidden_size % num_heads != 0:
  29       raise ValueError("Hidden size must be evenly divisible by the number of "
  30                        "heads.")
  31 
  32     super(Attention, self).__init__()
  33     self.hidden_size = hidden_size
  34     self.num_heads = num_heads
  35     self.attention_dropout = attention_dropout
  36     self.train = train
  37 
  38     # Layers for linearly projecting the queries, keys, and values.
  39     self.q_dense_layer = tf.layers.Dense(hidden_size, use_bias=False, name="q")
  40     self.k_dense_layer = tf.layers.Dense(hidden_size, use_bias=False, name="k")
  41     self.v_dense_layer = tf.layers.Dense(hidden_size, use_bias=False, name="v")
  42 
  43     self.output_dense_layer = tf.layers.Dense(hidden_size, use_bias=False,
  44                                               name="output_transform")
  45 
  46   def split_heads(self, x):
  47     """Split x into different heads, and transpose the resulting value.
  48 
  49     The tensor is transposed to insure the inner dimensions hold the correct
  50     values during the matrix multiplication.
  51 
  52     Args:
  53       x: A tensor with shape [batch_size, length, hidden_size]
  54 
  55     Returns:
  56       A tensor with shape [batch_size, num_heads, length, hidden_size/num_heads]
  57     """
  58     with tf.name_scope("split_heads"):
  59       batch_size = tf.shape(x)[0]
  60       length = tf.shape(x)[1]
  61 
  62       # Calculate depth of last dimension after it has been split.
  63       depth = (self.hidden_size // self.num_heads)
  64 
  65       # Split the last dimension
  66       x = tf.reshape(x, [batch_size, length, self.num_heads, depth])
  67 
  68       # Transpose the result
  69       return tf.transpose(x, [0, 2, 1, 3])
  70 
  71   def combine_heads(self, x):
  72     """Combine tensor that has been split.
  73 
  74     Args:
  75       x: A tensor [batch_size, num_heads, length, hidden_size/num_heads]
  76 
  77     Returns:
  78       A tensor with shape [batch_size, length, hidden_size]
  79     """
  80     with tf.name_scope("combine_heads"):
  81       batch_size = tf.shape(x)[0]
  82       length = tf.shape(x)[2]
  83       x = tf.transpose(x, [0, 2, 1, 3])  # --> [batch, length, num_heads, depth]
  84       return tf.reshape(x, [batch_size, length, self.hidden_size])
  85 
  86   def call(self, x, y, bias, cache=None):
  87     """Apply attention mechanism to x and y.
  88 
  89     Args:
  90       x: a tensor with shape [batch_size, length_x, hidden_size]
  91       y: a tensor with shape [batch_size, length_y, hidden_size]
  92       bias: attention bias that will be added to the result of the dot product.
  93       cache: (Used during prediction) dictionary with tensors containing results
  94         of previous attentions. The dictionary must have the items:
  95             {"k": tensor with shape [batch_size, i, key_channels],
  96              "v": tensor with shape [batch_size, i, value_channels]}
  97         where i is the current decoded length.
  98 
  99     Returns:
 100       Attention layer output with shape [batch_size, length_x, hidden_size]
 101     """
 102     # Linearly project the query (q), key (k) and value (v) using different
 103     # learned projections. This is in preparation of splitting them into
 104     # multiple heads. Multi-head attention uses multiple queries, keys, and
 105     # values rather than regular attention (which uses a single q, k, v).
 106     q = self.q_dense_layer(x)
 107     k = self.k_dense_layer(y)
 108     v = self.v_dense_layer(y)
 109 
 110     if cache is not None:
 111       # Combine cached keys and values with new keys and values.
 112       k = tf.concat([cache["k"], k], axis=1)
 113       v = tf.concat([cache["v"], v], axis=1)
 114 
 115       # Update cache
 116       cache["k"] = k
 117       cache["v"] = v
 118 
 119     # Split q, k, v into heads.
 120     q = self.split_heads(q)
 121     k = self.split_heads(k)
 122     v = self.split_heads(v)
 123 
 124     # Scale q to prevent the dot product between q and k from growing too large.
 125     depth = (self.hidden_size // self.num_heads)
 126     q *= depth ** -0.5
 127 
 128     # Calculate dot product attention
 129     logits = tf.matmul(q, k, transpose_b=True)
 130     logits += bias
 131     weights = tf.nn.softmax(logits, name="attention_weights")
 132     if self.train:
 133       weights = tf.nn.dropout(weights, 1.0 - self.attention_dropout)
 134     attention_output = tf.matmul(weights, v)
 135 
 136     # Recombine heads --> [batch_size, length, hidden_size]
 137     attention_output = self.combine_heads(attention_output)
 138 
 139     # Run the combined outputs through another linear projection layer.
 140     attention_output = self.output_dense_layer(attention_output)
 141     return attention_output
 142 
 143 
 144 class SelfAttention(Attention):
 145   """Multiheaded self-attention layer."""
 146 
 147   def call(self, x, bias, cache=None):
 148     return super(SelfAttention, self).call(x, x, bias, cache)
