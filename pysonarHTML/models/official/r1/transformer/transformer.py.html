  37 class Transformer(object):
  38   """Transformer model for sequence to sequence data.
  39 
  40   Implemented as described in: https://arxiv.org/pdf/1706.03762.pdf
  41 
  42   The Transformer model consists of an encoder and decoder. The input is an int
  43   sequence (or a batch of sequences). The encoder produces a continous
  44   representation, and the decoder uses the encoder output to generate
  45   probabilities for the output sequence.
  46   """
  47 
  48   def __init__(self, params, train):
  49     """Initialize layers to build Transformer model.
  50 
  51     Args:
  52       params: hyperparameter object defining layer sizes, dropout values, etc.
  53       train: boolean indicating whether the model is in training mode. Used to
  54         determine if dropout layers should be added.
  55     """
  56     self.train = train
  57     self.params = params
  58 
  59     self.embedding_softmax_layer = embedding_layer.EmbeddingSharedWeights(
  60         params["vocab_size"], params["hidden_size"],
  61         method="matmul" if params["tpu"] else "gather")
  62     self.encoder_stack = EncoderStack(params, train)
  63     self.decoder_stack = DecoderStack(params, train)
  64 
  65   def __call__(self, inputs, targets=None):
  66     """Calculate target logits or inferred target sequences.
  67 
  68     Args:
  69       inputs: int tensor with shape [batch_size, input_length].
  70       targets: None or int tensor with shape [batch_size, target_length].
  71 
  72     Returns:
  73       If targets is defined, then return logits for each word in the target
  74       sequence. float tensor with shape [batch_size, target_length, vocab_size]
  75       If target is none, then generate output sequence one token at a time.
  76         returns a dictionary {
  77           output: [batch_size, decoded length]
  78           score: [batch_size, float]}
  79     """
  80     # Variance scaling is used here because it seems to work in many problems.
  81     # Other reasonable initializers may also work just as well.
  82     initializer = tf.variance_scaling_initializer(
  83         self.params["initializer_gain"], mode="fan_avg", distribution="uniform")
  84     with tf.variable_scope("Transformer", initializer=initializer):
  85       # Calculate attention bias for encoder self-attention and decoder
  86       # multi-headed attention layers.
  87       attention_bias = model_utils.get_padding_bias(inputs)
  88 
  89       # Run the inputs through the encoder layer to map the symbol
  90       # representations to continuous representations.
  91       encoder_outputs = self.encode(inputs, attention_bias)
  92 
  93       # Generate output sequence if targets is None, or return logits if target
  94       # sequence is known.
  95       if targets is None:
  96         return self.predict(encoder_outputs, attention_bias)
  97       else:
  98         logits = self.decode(targets, encoder_outputs, attention_bias)
  99         return logits
 100 
 101   def encode(self, inputs, attention_bias):
 102     """Generate continuous representation for inputs.
 103 
 104     Args:
 105       inputs: int tensor with shape [batch_size, input_length].
 106       attention_bias: float tensor with shape [batch_size, 1, 1, input_length]
 107 
 108     Returns:
 109       float tensor with shape [batch_size, input_length, hidden_size]
 110     """
 111     with tf.name_scope("encode"):
 112       # Prepare inputs to the layer stack by adding positional encodings and
 113       # applying dropout.
 114       embedded_inputs = self.embedding_softmax_layer(inputs)
 115       inputs_padding = model_utils.get_padding(inputs)
 116 
 117       with tf.name_scope("add_pos_encoding"):
 118         length = tf.shape(embedded_inputs)[1]
 119         pos_encoding = model_utils.get_position_encoding(
 120             length, self.params["hidden_size"])
 121         encoder_inputs = embedded_inputs + pos_encoding
 122 
 123       if self.train:
 124         encoder_inputs = tf.nn.dropout(
 125             encoder_inputs, 1 - self.params["layer_postprocess_dropout"])
 126 
 127       return self.encoder_stack(encoder_inputs, attention_bias, inputs_padding)
 128 
 129   def decode(self, targets, encoder_outputs, attention_bias):
 130     """Generate logits for each value in the target sequence.
 131 
 132     Args:
 133       targets: target values for the output sequence.
 134         int tensor with shape [batch_size, target_length]
 135       encoder_outputs: continuous representation of input sequence.
 136         float tensor with shape [batch_size, input_length, hidden_size]
 137       attention_bias: float tensor with shape [batch_size, 1, 1, input_length]
 138 
 139     Returns:
 140       float32 tensor with shape [batch_size, target_length, vocab_size]
 141     """
 142     with tf.name_scope("decode"):
 143       # Prepare inputs to decoder layers by shifting targets, adding positional
 144       # encoding and applying dropout.
 145       decoder_inputs = self.embedding_softmax_layer(targets)
 146       with tf.name_scope("shift_targets"):
 147         # Shift targets to the right, and remove the last element
 148         decoder_inputs = tf.pad(
 149             decoder_inputs, [[0, 0], [1, 0], [0, 0]])[:, :-1, :]
 150       with tf.name_scope("add_pos_encoding"):
 151         length = tf.shape(decoder_inputs)[1]
 152         decoder_inputs += model_utils.get_position_encoding(
 153             length, self.params["hidden_size"])
 154       if self.train:
 155         decoder_inputs = tf.nn.dropout(
 156             decoder_inputs, 1 - self.params["layer_postprocess_dropout"])
 157 
 158       # Run values
 159       decoder_self_attention_bias = model_utils.get_decoder_self_attention_bias(
 160           length)
 161       outputs = self.decoder_stack(
 162           decoder_inputs, encoder_outputs, decoder_self_attention_bias,
 163           attention_bias)
 164       logits = self.embedding_softmax_layer.linear(outputs)
 165       return logits
 166 
 167   def _get_symbols_to_logits_fn(self, max_decode_length):
 168     """Returns a decoding function that calculates logits of the next tokens."""
 169 
 170     timing_signal = model_utils.get_position_encoding(
 171         max_decode_length + 1, self.params["hidden_size"])
 172     decoder_self_attention_bias = model_utils.get_decoder_self_attention_bias(
 173         max_decode_length)
 174 
 175     def symbols_to_logits_fn(ids, i, cache):
 176       """Generate logits for next potential IDs.
 177 
 178       Args:
 179         ids: Current decoded sequences.
 180           int tensor with shape [batch_size * beam_size, i + 1]
 181         i: Loop index
 182         cache: dictionary of values storing the encoder output, encoder-decoder
 183           attention bias, and previous decoder attention values.
 184 
 185       Returns:
 186         Tuple of
 187           (logits with shape [batch_size * beam_size, vocab_size],
 188            updated cache values)
 189       """
 190       # Set decoder input to the last generated IDs
 191       decoder_input = ids[:, -1:]
 192 
 193       # Preprocess decoder input by getting embeddings and adding timing signal.
 194       decoder_input = self.embedding_softmax_layer(decoder_input)
 195       decoder_input += timing_signal[i:i + 1]
 196 
 197       self_attention_bias = decoder_self_attention_bias[:, :, i:i + 1, :i + 1]
 198       decoder_outputs = self.decoder_stack(
 199           decoder_input, cache.get("encoder_outputs"), self_attention_bias,
 200           cache.get("encoder_decoder_attention_bias"), cache)
 201       logits = self.embedding_softmax_layer.linear(decoder_outputs)
 202       logits = tf.squeeze(logits, axis=[1])
 203       return logits, cache
 204     return symbols_to_logits_fn
 205 
 206   def predict(self, encoder_outputs, encoder_decoder_attention_bias):
 207     """Return predicted sequence."""
 208     batch_size = tf.shape(encoder_outputs)[0]
 209     input_length = tf.shape(encoder_outputs)[1]
 210     max_decode_length = input_length + self.params["extra_decode_length"]
 211 
 212     symbols_to_logits_fn = self._get_symbols_to_logits_fn(max_decode_length)
 213 
 214     # Create initial set of IDs that will be passed into symbols_to_logits_fn.
 215     initial_ids = tf.zeros([batch_size], dtype=tf.int32)
 216 
 217     # Create cache storing decoder attention values for each layer.
 218     cache = {
 219         "layer_%d" % layer: {
 220             "k": tf.zeros([batch_size, 0, self.params["hidden_size"]]),
 221             "v": tf.zeros([batch_size, 0, self.params["hidden_size"]]),
 222         } for layer in range(self.params["num_hidden_layers"])}
 223 
 224     # Add encoder output and attention bias to the cache.
 225     cache["encoder_outputs"] = encoder_outputs
 226     cache["encoder_decoder_attention_bias"] = encoder_decoder_attention_bias
 227 
 228     # Use beam search to find the top beam_size sequences and scores.
 229     decoded_ids, scores = beam_search.sequence_beam_search(
 230         symbols_to_logits_fn=symbols_to_logits_fn,
 231         initial_ids=initial_ids,
 232         initial_cache=cache,
 233         vocab_size=self.params["vocab_size"],
 234         beam_size=self.params["beam_size"],
 235         alpha=self.params["alpha"],
 236         max_decode_length=max_decode_length,
 237         eos_id=EOS_ID)
 238 
 239     # Get the top sequence for each batch element
 240     top_decoded_ids = decoded_ids[:, 0, 1:]
 241     top_scores = scores[:, 0]
 242 
 243     return {"outputs": top_decoded_ids, "scores": top_scores}

 246 class LayerNormalization(tf.layers.Layer):
 247   """Applies layer normalization."""
 248 
 249   def __init__(self, hidden_size):
 250     super(LayerNormalization, self).__init__()
 251     self.hidden_size = hidden_size
 252 
 253   def build(self, _):
 254     self.scale = tf.get_variable("layer_norm_scale", [self.hidden_size],
 255                                  initializer=tf.ones_initializer())
 256     self.bias = tf.get_variable("layer_norm_bias", [self.hidden_size],
 257                                 initializer=tf.zeros_initializer())
 258     self.built = True
 259 
 260   def call(self, x, epsilon=1e-6):
 261     mean = tf.reduce_mean(x, axis=[-1], keepdims=True)
 262     variance = tf.reduce_mean(tf.square(x - mean), axis=[-1], keepdims=True)
 263     norm_x = (x - mean) * tf.rsqrt(variance + epsilon)
 264     return norm_x * self.scale + self.bias

 267 class PrePostProcessingWrapper(object):
 268   """Wrapper class that applies layer pre-processing and post-processing."""
 269 
 270   def __init__(self, layer, params, train):
 271     self.layer = layer
 272     self.postprocess_dropout = params["layer_postprocess_dropout"]
 273     self.train = train
 274 
 275     # Create normalization layer
 276     self.layer_norm = LayerNormalization(params["hidden_size"])
 277 
 278   def __call__(self, x, *args, **kwargs):
 279     # Preprocessing: apply layer normalization
 280     y = self.layer_norm(x)
 281 
 282     # Get layer output
 283     y = self.layer(y, *args, **kwargs)
 284 
 285     # Postprocessing: apply dropout and residual connection
 286     if self.train:
 287       y = tf.nn.dropout(y, 1 - self.postprocess_dropout)
 288     return x + y

 291 class EncoderStack(tf.layers.Layer):
 292   """Transformer encoder stack.
 293 
 294   The encoder stack is made up of N identical layers. Each layer is composed
 295   of the sublayers:
 296     1. Self-attention layer
 297     2. Feedforward network (which is 2 fully-connected layers)
 298   """
 299 
 300   def __init__(self, params, train):
 301     super(EncoderStack, self).__init__()
 302     self.layers = []
 303     for _ in range(params["num_hidden_layers"]):
 304       # Create sublayers for each layer.
 305       self_attention_layer = attention_layer.SelfAttention(
 306           params["hidden_size"], params["num_heads"],
 307           params["attention_dropout"], train)
 308       feed_forward_network = ffn_layer.FeedFowardNetwork(
 309           params["hidden_size"], params["filter_size"],
 310           params["relu_dropout"], train, params["allow_ffn_pad"])
 311 
 312       self.layers.append([
 313           PrePostProcessingWrapper(self_attention_layer, params, train),
 314           PrePostProcessingWrapper(feed_forward_network, params, train)])
 315 
 316     # Create final layer normalization layer.
 317     self.output_normalization = LayerNormalization(params["hidden_size"])
 318 
 319   def call(self, encoder_inputs, attention_bias, inputs_padding):
 320     """Return the output of the encoder layer stacks.
 321 
 322     Args:
 323       encoder_inputs: tensor with shape [batch_size, input_length, hidden_size]
 324       attention_bias: bias for the encoder self-attention layer.
 325         [batch_size, 1, 1, input_length]
 326       inputs_padding: P
 327 
 328     Returns:
 329       Output of encoder layer stack.
 330       float32 tensor with shape [batch_size, input_length, hidden_size]
 331     """
 332     for n, layer in enumerate(self.layers):
 333       # Run inputs through the sublayers.
 334       self_attention_layer = layer[0]
 335       feed_forward_network = layer[1]
 336 
 337       with tf.variable_scope("layer_%d" % n):
 338         with tf.variable_scope("self_attention"):
 339           encoder_inputs = self_attention_layer(encoder_inputs, attention_bias)
 340         with tf.variable_scope("ffn"):
 341           encoder_inputs = feed_forward_network(encoder_inputs, inputs_padding)
 342 
 343     return self.output_normalization(encoder_inputs)

 346 class DecoderStack(tf.layers.Layer):
 347   """Transformer decoder stack.
 348 
 349   Like the encoder stack, the decoder stack is made up of N identical layers.
 350   Each layer is composed of the sublayers:
 351     1. Self-attention layer
 352     2. Multi-headed attention layer combining encoder outputs with results from
 353        the previous self-attention layer.
 354     3. Feedforward network (2 fully-connected layers)
 355   """
 356 
 357   def __init__(self, params, train):
 358     super(DecoderStack, self).__init__()
 359     self.layers = []
 360     for _ in range(params["num_hidden_layers"]):
 361       self_attention_layer = attention_layer.SelfAttention(
 362           params["hidden_size"], params["num_heads"],
 363           params["attention_dropout"], train)
 364       enc_dec_attention_layer = attention_layer.Attention(
 365           params["hidden_size"], params["num_heads"],
 366           params["attention_dropout"], train)
 367       feed_forward_network = ffn_layer.FeedFowardNetwork(
 368           params["hidden_size"], params["filter_size"],
 369           params["relu_dropout"], train, params["allow_ffn_pad"])
 370 
 371       self.layers.append([
 372           PrePostProcessingWrapper(self_attention_layer, params, train),
 373           PrePostProcessingWrapper(enc_dec_attention_layer, params, train),
 374           PrePostProcessingWrapper(feed_forward_network, params, train)])
 375 
 376     self.output_normalization = LayerNormalization(params["hidden_size"])
 377 
 378   def call(self, decoder_inputs, encoder_outputs, decoder_self_attention_bias,
 379            attention_bias, cache=None):
 380     """Return the output of the decoder layer stacks.
 381 
 382     Args:
 383       decoder_inputs: tensor with shape [batch_size, target_length, hidden_size]
 384       encoder_outputs: tensor with shape [batch_size, input_length, hidden_size]
 385       decoder_self_attention_bias: bias for decoder self-attention layer.
 386         [1, 1, target_len, target_length]
 387       attention_bias: bias for encoder-decoder attention layer.
 388         [batch_size, 1, 1, input_length]
 389       cache: (Used for fast decoding) A nested dictionary storing previous
 390         decoder self-attention values. The items are:
 391           {layer_n: {"k": tensor with shape [batch_size, i, key_channels],
 392                      "v": tensor with shape [batch_size, i, value_channels]},
 393            ...}
 394 
 395     Returns:
 396       Output of decoder layer stack.
 397       float32 tensor with shape [batch_size, target_length, hidden_size]
 398     """
 399     for n, layer in enumerate(self.layers):
 400       self_attention_layer = layer[0]
 401       enc_dec_attention_layer = layer[1]
 402       feed_forward_network = layer[2]
 403 
 404       # Run inputs through the sublayers.
 405       layer_name = "layer_%d" % n
 406       layer_cache = cache[layer_name] if cache is not None else None
 407       with tf.variable_scope(layer_name):
 408         with tf.variable_scope("self_attention"):
 409           decoder_inputs = self_attention_layer(
 410               decoder_inputs, decoder_self_attention_bias, cache=layer_cache)
 411         with tf.variable_scope("encdec_attention"):
 412           decoder_inputs = enc_dec_attention_layer(
 413               decoder_inputs, encoder_outputs, attention_bias)
 414         with tf.variable_scope("ffn"):
 415           decoder_inputs = feed_forward_network(decoder_inputs)
 416 
 417     return self.output_normalization(decoder_inputs)
