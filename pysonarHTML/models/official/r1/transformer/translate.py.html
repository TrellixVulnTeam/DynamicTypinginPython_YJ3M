  32 _DECODE_BATCH_SIZE = 32
  33 _EXTRA_DECODE_LENGTH = 100
  34 _BEAM_SIZE = 4
  35 _ALPHA = 0.6

  38 def _get_sorted_inputs(filename):
  39   """Read and sort lines from the file sorted by decreasing length.
  40 
  41   Args:
  42     filename: String name of file to read inputs from.
  43   Returns:
  44     Sorted list of inputs, and dictionary mapping original index->sorted index
  45     of each element.
  46   """
  47   with tf.io.gfile.GFile(filename) as f:
  48     records = f.read().split("\n")
  49     inputs = [record.strip() for record in records]
  50     if not inputs[-1]:
  51       inputs.pop()
  52 
  53   input_lens = [(i, len(line.split())) for i, line in enumerate(inputs)]
  54   sorted_input_lens = sorted(input_lens, key=lambda x: x[1], reverse=True)
  55 
  56   sorted_inputs = [None] * len(sorted_input_lens)
  57   sorted_keys = [0] * len(sorted_input_lens)
  58   for i, (index, _) in enumerate(sorted_input_lens):
  59     sorted_inputs[i] = inputs[index]
  60     sorted_keys[index] = i
  61 
  62   return sorted_inputs, sorted_keys

  65 def _encode_and_add_eos(line, subtokenizer):
  66   """Encode line with subtokenizer, and add EOS id to the end."""
  67   return subtokenizer.encode(line) + [tokenizer.EOS_ID]

  70 def _trim_and_decode(ids, subtokenizer):
  71   """Trim EOS and PAD tokens from ids, and decode to return a string."""
  72   try:
  73     index = list(ids).index(tokenizer.EOS_ID)
  74     return subtokenizer.decode(ids[:index])
  75   except ValueError:  # No EOS found in sequence
  76     return subtokenizer.decode(ids)

  79 def translate_file(
  80     estimator, subtokenizer, input_file, output_file=None,
  81     print_all_translations=True):
  82   """Translate lines in file, and save to output file if specified.
  83 
  84   Args:
  85     estimator: tf.Estimator used to generate the translations.
  86     subtokenizer: Subtokenizer object for encoding and decoding source and
  87        translated lines.
  88     input_file: file containing lines to translate
  89     output_file: file that stores the generated translations.
  90     print_all_translations: If true, all translations are printed to stdout.
  91 
  92   Raises:
  93     ValueError: if output file is invalid.
  94   """
  95   batch_size = _DECODE_BATCH_SIZE
  96 
  97   # Read and sort inputs by length. Keep dictionary (original index-->new index
  98   # in sorted list) to write translations in the original order.
  99   sorted_inputs, sorted_keys = _get_sorted_inputs(input_file)
 100   num_decode_batches = (len(sorted_inputs) - 1) // batch_size + 1
 101 
 102   def input_generator():
 103     """Yield encoded strings from sorted_inputs."""
 104     for i, line in enumerate(sorted_inputs):
 105       if i % batch_size == 0:
 106         batch_num = (i // batch_size) + 1
 107 
 108         tf.logging.info("Decoding batch %d out of %d." %
 109                         (batch_num, num_decode_batches))
 110       yield _encode_and_add_eos(line, subtokenizer)
 111 
 112   def input_fn():
 113     """Created batched dataset of encoded inputs."""
 114     ds = tf.data.Dataset.from_generator(
 115         input_generator, tf.int64, tf.TensorShape([None]))
 116     ds = ds.padded_batch(batch_size, [None])
 117     return ds
 118 
 119   translations = []
 120   for i, prediction in enumerate(estimator.predict(input_fn)):
 121     translation = _trim_and_decode(prediction["outputs"], subtokenizer)
 122     translations.append(translation)
 123 
 124     if print_all_translations:
 125       tf.logging.info("Translating:\n\tInput: %s\n\tOutput: %s" %
 126                       (sorted_inputs[i], translation))
 127 
 128   # Write translations in the order they appeared in the original file.
 129   if output_file is not None:
 130     if tf.io.gfile.isdir(output_file):
 131       raise ValueError("File output is a directory, will not save outputs to "
 132                        "file.")
 133     tf.logging.info("Writing to file %s" % output_file)
 134     with tf.io.gfile.GFile(output_file, "w") as f:
 135       for i in sorted_keys:
 136         f.write("%s\n" % translations[i])

 139 def translate_text(estimator, subtokenizer, txt):
 140   """Translate a single string."""
 141   encoded_txt = _encode_and_add_eos(txt, subtokenizer)
 142 
 143   def input_fn():
 144     ds = tf.data.Dataset.from_tensors(encoded_txt)
 145     ds = ds.batch(_DECODE_BATCH_SIZE)
 146     return ds
 147 
 148   predictions = estimator.predict(input_fn)
 149   translation = next(predictions)["outputs"]
 150   translation = _trim_and_decode(translation, subtokenizer)
 151   tf.logging.info("Translation of \"%s\": \"%s\"" % (txt, translation))

 154 def main(unused_argv):
 155   from official.transformer import transformer_main
 156 
 157   tf.logging.set_verbosity(tf.logging.INFO)
 158 
 159   if FLAGS.text is None and FLAGS.file is None:
 160     tf.logging.warn("Nothing to translate. Make sure to call this script using "
 161                     "flags --text or --file.")
 162     return
 163 
 164   subtokenizer = tokenizer.Subtokenizer(FLAGS.vocab_file)
 165 
 166   # Set up estimator and params
 167   params = transformer_main.PARAMS_MAP[FLAGS.param_set]
 168   params["beam_size"] = _BEAM_SIZE
 169   params["alpha"] = _ALPHA
 170   params["extra_decode_length"] = _EXTRA_DECODE_LENGTH
 171   params["batch_size"] = _DECODE_BATCH_SIZE
 172   estimator = tf.estimator.Estimator(
 173       model_fn=transformer_main.model_fn, model_dir=FLAGS.model_dir,
 174       params=params)
 175 
 176   if FLAGS.text is not None:
 177     tf.logging.info("Translating text: %s" % FLAGS.text)
 178     translate_text(estimator, subtokenizer, FLAGS.text)
 179 
 180   if FLAGS.file is not None:
 181     input_file = os.path.abspath(FLAGS.file)
 182     tf.logging.info("Translating file: %s" % input_file)
 183     if not tf.gfile.Exists(FLAGS.file):
 184       raise ValueError("File does not exist: %s" % input_file)
 185 
 186     output_file = None
 187     if FLAGS.file_out is not None:
 188       output_file = os.path.abspath(FLAGS.file_out)
 189       tf.logging.info("File output specified: %s" % output_file)
 190 
 191     translate_file(estimator, subtokenizer, input_file, output_file)

 194 def define_translate_flags():
 195   """Define flags used for translation script."""
 196   # Model flags
 197   flags.DEFINE_string(
 198       name="model_dir", short_name="md", default="/tmp/transformer_model",
 199       help=flags_core.help_wrap(
 200           "Directory containing Transformer model checkpoints."))
 201   flags.DEFINE_enum(
 202       name="param_set", short_name="mp", default="big",
 203       enum_values=["base", "big"],
 204       help=flags_core.help_wrap(
 205           "Parameter set to use when creating and training the model. The "
 206           "parameters define the input shape (batch size and max length), "
 207           "model configuration (size of embedding, # of hidden layers, etc.), "
 208           "and various other settings. The big parameter set increases the "
 209           "default batch size, embedding/hidden size, and filter size. For a "
 210           "complete list of parameters, please see model/model_params.py."))
 211   flags.DEFINE_string(
 212       name="vocab_file", short_name="vf", default=None,
 213       help=flags_core.help_wrap(
 214           "Path to subtoken vocabulary file. If data_download.py was used to "
 215           "download and encode the training data, look in the data_dir to find "
 216           "the vocab file."))
 217   flags.mark_flag_as_required("vocab_file")
 218 
 219   flags.DEFINE_string(
 220       name="text", default=None,
 221       help=flags_core.help_wrap(
 222           "Text to translate. Output will be printed to console."))
 223   flags.DEFINE_string(
 224       name="file", default=None,
 225       help=flags_core.help_wrap(
 226           "File containing text to translate. Translation will be printed to "
 227           "console and, if --file_out is provided, saved to an output file."))
 228   flags.DEFINE_string(
 229       name="file_out", default=None,
 230       help=flags_core.help_wrap(
 231           "If --file flag is specified, save translation to this file."))
 232 
 233 
 234 if __name__ == "__main__":
 235   define_translate_flags()
 236   FLAGS = flags.FLAGS
 237   absl_app.run(main)
