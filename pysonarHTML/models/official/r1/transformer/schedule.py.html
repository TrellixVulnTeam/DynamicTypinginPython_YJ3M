  26 _TRAIN, _EVAL = tf.estimator.ModeKeys.TRAIN, tf.estimator.ModeKeys.EVAL

  29 NUM_EXAMPLES = {
  30     tf.estimator.ModeKeys.TRAIN: 4572160,
  31     # # Examples that are too long are filtered out, thus the total is less
  32     # # than the total number of lines.
  33     # 2399123 +  # news-commentary-v12.de-en
  34     # 1920209 +  # commoncrawl.de-en
  35     # 270769,    # europarl-v7.de-en
  36     tf.estimator.ModeKeys.EVAL: 3000,  # newstest2013
  37 }

  40 class Manager(object):
  41   """Container for convenience functions to abstract step or epoch basis.
  42   Transformer allows users to specify an epoch basis (generally recommended for
  43   full training) or a number of steps basis (convenient since epochs are rather
  44   large). TPUs furthermore require a step basis; however epochs are the norm in
  45   the machine learning community and it is desirable to allow users to specify
  46   epochs even when running with TPUS which requires behind the scenes
  47   conversions.
  48   This container simply groups what are largely mundane checks and conversions
  49   rather than interspersing them throughout the run loop code.
  50   """
  51 
  52   def __init__(self, train_steps, steps_between_evals, train_epochs,
  53                epochs_between_evals, default_train_epochs, batch_size,
  54                max_length, use_tpu=False, num_tpu_shards=8):
  55     if train_steps and train_epochs:
  56       raise ValueError("Both train_steps or train_epochs were be defined.")
  57 
  58     # Determine training schedule based on flags.
  59     if train_steps:
  60       self.train_eval_iterations = train_steps // steps_between_evals
  61       self._single_iteration_train_steps = steps_between_evals
  62       self._single_iteration_train_epochs = None
  63     else:
  64       train_epochs = train_epochs or default_train_epochs
  65       self.train_eval_iterations = train_epochs // epochs_between_evals
  66       self._single_iteration_train_steps = None
  67       self._single_iteration_train_epochs = epochs_between_evals
  68 
  69     self.max_length = max_length
  70     self.batch_size = batch_size
  71     self.use_tpu = use_tpu
  72     self.num_tpu_shards = num_tpu_shards
  73 
  74     if self.use_tpu:
  75       assert (self.batch_size // self.max_length) % self.num_tpu_shards == 0
  76 
  77   @property
  78   def single_iteration_train_steps(self):
  79     if self._single_iteration_train_steps or not self.use_tpu:
  80       return self._single_iteration_train_steps
  81 
  82     return self.epochs_to_steps(
  83         num_epochs=self._single_iteration_train_epochs, mode=_TRAIN)
  84 
  85   @property
  86   def single_iteration_eval_steps(self):
  87     if not self.use_tpu:
  88       return None
  89 
  90     return self.epochs_to_steps(num_epochs=1, mode=_EVAL)
  91 
  92   @property
  93   def train_increment_str(self):
  94     if self._single_iteration_train_steps:
  95       return "{} steps.".format(self._single_iteration_train_steps)
  96 
  97     if not self.use_tpu:
  98       return "{} epochs.".format(self._single_iteration_train_epochs)
  99 
 100     return "~{} epochs. ({} steps)".format(
 101         self._single_iteration_train_epochs,
 102         self.single_iteration_train_steps)
 103 
 104   @property
 105   def repeat_dataset(self):
 106     if (self._single_iteration_train_epochs is None and
 107         self._single_iteration_train_steps > NUM_EXAMPLES[_TRAIN]):
 108       return math.ceil(self._single_iteration_train_steps /
 109                        NUM_EXAMPLES[_TRAIN])
 110     return self._single_iteration_train_epochs
 111 
 112   def epochs_to_steps(self, num_epochs, mode):
 113     """Converts a number of epochs to a number of training steps.
 114 
 115     TPU only: This function assumes that static_batch is True.
 116 
 117       TPU can not tolerate an OutOfRange error from a dataset. As a result the
 118     number of examples to be processed must be known ahead of time. TPUs also
 119     do not allow partial batches, so this function rounds down.
 120 
 121     Args:
 122       num_epochs: An integer of the number of epochs to convert to steps.
 123       mode: The estimator ModeKey of the computation
 124 
 125     Returns:
 126       An integer of the number of equivalent steps rounded down.
 127     """
 128     assert self.use_tpu, "epochs_to_steps should only be reached when using TPU"
 129     total_num_tokens = NUM_EXAMPLES[mode] * self.max_length * num_epochs
 130     return total_num_tokens // self.batch_size
