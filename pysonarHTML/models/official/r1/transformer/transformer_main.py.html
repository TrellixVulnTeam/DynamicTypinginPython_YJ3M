  69 def model_fn(features, labels, mode, params):
  70   """Defines how to train, evaluate and predict from the transformer model."""
  71   with tf.variable_scope("model"):
  72     inputs, targets = features, labels
  73 
  74     # Create model and get output logits.
  75     model = transformer.Transformer(params, mode == tf.estimator.ModeKeys.TRAIN)
  76 
  77     logits = model(inputs, targets)
  78 
  79     # When in prediction mode, the labels/targets is None. The model output
  80     # is the prediction
  81     if mode == tf.estimator.ModeKeys.PREDICT:
  82       if params["use_tpu"]:
  83         raise NotImplementedError("Prediction is not yet supported on TPUs.")
  84       return tf.estimator.EstimatorSpec(
  85           tf.estimator.ModeKeys.PREDICT,
  86           predictions=logits,
  87           export_outputs={
  88               "translate": tf.estimator.export.PredictOutput(logits)
  89           })
  90 
  91     # Explicitly set the shape of the logits for XLA (TPU). This is needed
  92     # because the logits are passed back to the host VM CPU for metric
  93     # evaluation, and the shape of [?, ?, vocab_size] is too vague. However
  94     # it is known from Transformer that the first two dimensions of logits
  95     # are the dimensions of targets. Note that the ambiguous shape of logits is
  96     # not a problem when computing xentropy, because padded_cross_entropy_loss
  97     # resolves the shape on the TPU.
  98     logits.set_shape(targets.shape.as_list() + logits.shape.as_list()[2:])
  99 
 100     # Calculate model loss.
 101     # xentropy contains the cross entropy loss of every nonpadding token in the
 102     # targets.
 103     xentropy, weights = metrics.padded_cross_entropy_loss(
 104         logits, targets, params["label_smoothing"], params["vocab_size"])
 105     loss = tf.reduce_sum(xentropy) / tf.reduce_sum(weights)
 106 
 107     # Save loss as named tensor that will be logged with the logging hook.
 108     tf.identity(loss, "cross_entropy")
 109 
 110     if mode == tf.estimator.ModeKeys.EVAL:
 111       if params["use_tpu"]:
 112         # host call functions should only have tensors as arguments.
 113         # This lambda pre-populates params so that metric_fn is
 114         # TPUEstimator compliant.
 115         metric_fn = lambda logits, labels: (
 116             metrics.get_eval_metrics(logits, labels, params=params))
 117         eval_metrics = (metric_fn, [logits, labels])
 118         return tf.estimator.tpu.TPUEstimatorSpec(
 119             mode=mode,
 120             loss=loss,
 121             predictions={"predictions": logits},
 122             eval_metrics=eval_metrics)
 123       return tf.estimator.EstimatorSpec(
 124           mode=mode, loss=loss, predictions={"predictions": logits},
 125           eval_metric_ops=metrics.get_eval_metrics(logits, labels, params))
 126     else:
 127       train_op, metric_dict = get_train_op_and_metrics(loss, params)
 128 
 129       # Epochs can be quite long. This gives some intermediate information
 130       # in TensorBoard.
 131       metric_dict["minibatch_loss"] = loss
 132       if params["use_tpu"]:
 133         return tf.estimator.tpu.TPUEstimatorSpec(
 134             mode=mode,
 135             loss=loss,
 136             train_op=train_op,
 137             host_call=tpu_util.construct_scalar_host_call(
 138                 metric_dict=metric_dict,
 139                 model_dir=params["model_dir"],
 140                 prefix="training/"))
 141       record_scalars(metric_dict)
 142       return tf.estimator.EstimatorSpec(mode=mode, loss=loss, train_op=train_op)

 150 def get_learning_rate(learning_rate, hidden_size, learning_rate_warmup_steps):
 151   """Calculate learning rate with linear warmup and rsqrt decay."""
 152   with tf.name_scope("learning_rate"):
 153     warmup_steps = tf.to_float(learning_rate_warmup_steps)
 154     step = tf.to_float(tf.train.get_or_create_global_step())
 155 
 156     learning_rate *= (hidden_size ** -0.5)
 157     # Apply linear warmup
 158     learning_rate *= tf.minimum(1.0, step / warmup_steps)
 159     # Apply rsqrt decay
 160     learning_rate *= tf.rsqrt(tf.maximum(step, warmup_steps))
 161 
 162     # Create a named tensor that will be logged using the logging hook.
 163     # The full name includes variable and names scope. In this case, the name
 164     # is model/get_train_op/learning_rate/learning_rate
 165     tf.identity(learning_rate, "learning_rate")
 166 
 167     return learning_rate

 170 def get_train_op_and_metrics(loss, params):
 171   """Generate training op and metrics to save in TensorBoard."""
 172   with tf.variable_scope("get_train_op"):
 173     learning_rate = get_learning_rate(
 174         learning_rate=params["learning_rate"],
 175         hidden_size=params["hidden_size"],
 176         learning_rate_warmup_steps=params["learning_rate_warmup_steps"])
 177 
 178     # Create optimizer. Use LazyAdamOptimizer from TF contrib, which is faster
 179     # than the TF core Adam optimizer.
 180     from tensorflow.contrib import opt as contrib_opt  # pylint: disable=g-import-not-at-top
 181     optimizer = contrib_opt.LazyAdamOptimizer(
 182         learning_rate,
 183         beta1=params["optimizer_adam_beta1"],
 184         beta2=params["optimizer_adam_beta2"],
 185         epsilon=params["optimizer_adam_epsilon"])
 186 
 187     if params["use_tpu"] and params["tpu"] != tpu_util.LOCAL:
 188       optimizer = tf.compat.v1.tpu.CrossShardOptimizer(optimizer)
 189 
 190     # Uses automatic mixed precision FP16 training if on GPU.
 191     if params["dtype"] == "fp16":
 192       optimizer = tf.train.experimental.enable_mixed_precision_graph_rewrite(
 193           optimizer)
 194 
 195     # Calculate and apply gradients using LazyAdamOptimizer.
 196     global_step = tf.train.get_global_step()
 197     tvars = tf.trainable_variables()
 198     gradients = optimizer.compute_gradients(
 199         loss, tvars, colocate_gradients_with_ops=True)
 200     minimize_op = optimizer.apply_gradients(
 201         gradients, global_step=global_step, name="train")
 202     update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS)
 203     train_op = tf.group(minimize_op, update_ops)
 204 
 205     train_metrics = {"learning_rate": learning_rate}
 206 
 207     if not params["use_tpu"]:
 208       # gradient norm is not included as a summary when running on TPU, as
 209       # it can cause instability between the TPU and the host controller.
 210       gradient_norm = tf.global_norm(list(zip(*gradients))[0])
 211       train_metrics["global_norm/gradient_norm"] = gradient_norm
 212 
 213     return train_op, train_metrics

 216 def translate_and_compute_bleu(estimator, subtokenizer, bleu_source, bleu_ref):
 217   """Translate file and report the cased and uncased bleu scores."""
 218   # Create temporary file to store translation.
 219   tmp = tempfile.NamedTemporaryFile(delete=False)
 220   tmp_filename = tmp.name
 221 
 222   translate.translate_file(
 223       estimator, subtokenizer, bleu_source, output_file=tmp_filename,
 224       print_all_translations=False)
 225 
 226   # Compute uncased and cased bleu scores.
 227   uncased_score = compute_bleu.bleu_wrapper(bleu_ref, tmp_filename, False)
 228   cased_score = compute_bleu.bleu_wrapper(bleu_ref, tmp_filename, True)
 229   os.remove(tmp_filename)
 230   return uncased_score, cased_score

 238 def evaluate_and_log_bleu(estimator, bleu_source, bleu_ref, vocab_file):
 239   """Calculate and record the BLEU score."""
 240   subtokenizer = tokenizer.Subtokenizer(vocab_file)
 241 
 242   uncased_score, cased_score = translate_and_compute_bleu(
 243       estimator, subtokenizer, bleu_source, bleu_ref)
 244 
 245   tf.logging.info("Bleu score (uncased): %f", uncased_score)
 246   tf.logging.info("Bleu score (cased): %f", cased_score)
 247   return uncased_score, cased_score

 256 def run_loop(
 257     estimator, schedule_manager, train_hooks=None, benchmark_logger=None,
 258     bleu_source=None, bleu_ref=None, bleu_threshold=None, vocab_file=None):
 259   """Train and evaluate model, and optionally compute model's BLEU score.
 260 
 261   **Step vs. Epoch vs. Iteration**
 262 
 263   Steps and epochs are canonical terms used in TensorFlow and general machine
 264   learning. They are used to describe running a single process (train/eval):
 265     - Step refers to running the process through a single or batch of examples.
 266     - Epoch refers to running the process through an entire dataset.
 267 
 268   E.g. training a dataset with 100 examples. The dataset is
 269   divided into 20 batches with 5 examples per batch. A single training step
 270   trains the model on one batch. After 20 training steps, the model will have
 271   trained on every batch in the dataset, or, in other words, one epoch.
 272 
 273   Meanwhile, iteration is used in this implementation to describe running
 274   multiple processes (training and eval).
 275     - A single iteration:
 276       1. trains the model for a specific number of steps or epochs.
 277       2. evaluates the model.
 278       3. (if source and ref files are provided) compute BLEU score.
 279 
 280   This function runs through multiple train+eval+bleu iterations.
 281 
 282   Args:
 283     estimator: tf.Estimator containing model to train.
 284     schedule_manager: A schedule.Manager object to guide the run loop.
 285     train_hooks: List of hooks to pass to the estimator during training.
 286     benchmark_logger: a BenchmarkLogger object that logs evaluation data
 287     bleu_source: File containing text to be translated for BLEU calculation.
 288     bleu_ref: File containing reference translations for BLEU calculation.
 289     bleu_threshold: minimum BLEU score before training is stopped.
 290     vocab_file: Path to vocab file that will be used to subtokenize bleu_source.
 291 
 292   Returns:
 293     Dict of results of the run.  Contains the keys `eval_results`,
 294     `train_hooks`, `bleu_cased`, and `bleu_uncased`. `train_hooks` is a list the
 295     instances of hooks used during training.
 296 
 297   Raises:
 298     ValueError: if both or none of single_iteration_train_steps and
 299       single_iteration_train_epochs were defined.
 300     NotFoundError: if the vocab file or bleu files don't exist.
 301   """
 302   if bleu_source:
 303     _validate_file(bleu_source)
 304   if bleu_ref:
 305     _validate_file(bleu_ref)
 306   if vocab_file:
 307     _validate_file(vocab_file)
 308 
 309   evaluate_bleu = bleu_source is not None and bleu_ref is not None
 310   if evaluate_bleu and schedule_manager.use_tpu:
 311     raise ValueError("BLEU score can not be computed when training with a TPU, "
 312                      "as it requires estimator.predict which is not yet "
 313                      "supported.")
 314 
 315   # Print details of training schedule.
 316   tf.logging.info("Training schedule:")
 317   tf.logging.info(
 318       "\t1. Train for {}".format(schedule_manager.train_increment_str))
 319   tf.logging.info("\t2. Evaluate model.")
 320   if evaluate_bleu:
 321     tf.logging.info("\t3. Compute BLEU score.")
 322     if bleu_threshold is not None:
 323       tf.logging.info("Repeat above steps until the BLEU score reaches %f" %
 324                       bleu_threshold)
 325   if not evaluate_bleu or bleu_threshold is None:
 326     tf.logging.info("Repeat above steps %d times." %
 327                     schedule_manager.train_eval_iterations)
 328 
 329   if evaluate_bleu:
 330     # Create summary writer to log bleu score (values can be displayed in
 331     # Tensorboard).
 332     bleu_writer = tf.summary.FileWriter(
 333         os.path.join(estimator.model_dir, BLEU_DIR))
 334     if bleu_threshold is not None:
 335       # Change loop stopping condition if bleu_threshold is defined.
 336       schedule_manager.train_eval_iterations = INF
 337 
 338   # Loop training/evaluation/bleu cycles
 339   stats = {}
 340   for i in xrange(schedule_manager.train_eval_iterations):
 341     tf.logging.info("Starting iteration %d" % (i + 1))
 342 
 343     # Train the model for single_iteration_train_steps or until the input fn
 344     # runs out of examples (if single_iteration_train_steps is None).
 345     estimator.train(
 346         dataset.train_input_fn,
 347         steps=schedule_manager.single_iteration_train_steps,
 348         hooks=train_hooks)
 349 
 350     eval_results = None
 351     eval_results = estimator.evaluate(
 352         input_fn=dataset.eval_input_fn,
 353         steps=schedule_manager.single_iteration_eval_steps)
 354 
 355     tf.logging.info("Evaluation results (iter %d/%d):" %
 356                     (i + 1, schedule_manager.train_eval_iterations))
 357     tf.logging.info(eval_results)
 358     benchmark_logger.log_evaluation_result(eval_results)
 359 
 360     # The results from estimator.evaluate() are measured on an approximate
 361     # translation, which utilize the target golden values provided. The actual
 362     # bleu score must be computed using the estimator.predict() path, which
 363     # outputs translations that are not based on golden values. The translations
 364     # are compared to reference file to get the actual bleu score.
 365     if evaluate_bleu:
 366       uncased_score, cased_score = evaluate_and_log_bleu(
 367           estimator, bleu_source, bleu_ref, vocab_file)
 368 
 369       stats["bleu_uncased"] = uncased_score
 370       stats["bleu_cased"] = cased_score
 371 
 372       # Write actual bleu scores using summary writer and benchmark logger
 373       global_step = get_global_step(estimator)
 374       summary = tf.Summary(value=[
 375           tf.Summary.Value(tag="bleu/uncased", simple_value=uncased_score),
 376           tf.Summary.Value(tag="bleu/cased", simple_value=cased_score),
 377       ])
 378       bleu_writer.add_summary(summary, global_step)
 379       bleu_writer.flush()
 380       benchmark_logger.log_metric(
 381           "bleu_uncased", uncased_score, global_step=global_step)
 382       benchmark_logger.log_metric(
 383           "bleu_cased", cased_score, global_step=global_step)
 384 
 385       # Stop training if bleu stopping threshold is met.
 386       if model_helpers.past_stop_threshold(bleu_threshold, uncased_score):
 387         bleu_writer.close()
 388         break
 389 
 390   stats["eval_results"] = eval_results
 391   stats["train_hooks"] = train_hooks
 392 
 393   return stats

 517 def construct_estimator(flags_obj, params, schedule_manager):
 518   """Construct an estimator from either Estimator or TPUEstimator.
 519 
 520   Args:
 521     flags_obj: The FLAGS object parsed from command line.
 522     params: A dict of run specific parameters.
 523     schedule_manager: A schedule.Manager object containing the run schedule.
 524 
 525   Returns:
 526     An estimator object to be used for training and eval.
 527   """
 528   if not params["use_tpu"]:
 529     distribution_strategy = distribution_utils.get_distribution_strategy(
 530         distribution_strategy=flags_obj.distribution_strategy,
 531         num_gpus=flags_core.get_num_gpus(flags_obj),
 532         all_reduce_alg=flags_obj.all_reduce_alg)
 533     return tf.estimator.Estimator(
 534         model_fn=model_fn, model_dir=flags_obj.model_dir, params=params,
 535         config=tf.estimator.RunConfig(train_distribute=distribution_strategy))
 536 
 537   tpu_cluster_resolver = tf.compat.v1.cluster_resolver.TPUClusterResolver(
 538       tpu=flags_obj.tpu,
 539       zone=flags_obj.tpu_zone,
 540       project=flags_obj.tpu_gcp_project
 541   )
 542 
 543   tpu_config = tf.estimator.tpu.TPUConfig(
 544       iterations_per_loop=schedule_manager.single_iteration_train_steps,
 545       num_shards=flags_obj.num_tpu_shards)
 546 
 547   run_config = tf.estimator.tpu.RunConfig(
 548       cluster=tpu_cluster_resolver,
 549       model_dir=flags_obj.model_dir,
 550       session_config=tf.ConfigProto(
 551           allow_soft_placement=True, log_device_placement=True),
 552       tpu_config=tpu_config)
 553 
 554   return tf.estimator.tpu.TPUEstimator(
 555       model_fn=model_fn,
 556       use_tpu=params["use_tpu"] and flags_obj.tpu != tpu_util.LOCAL,
 557       train_batch_size=schedule_manager.batch_size,
 558       eval_batch_size=schedule_manager.batch_size,
 559       params={
 560           # TPUEstimator needs to populate batch_size itself due to sharding.
 561           key: value for key, value in params.items() if key != "batch_size"
 562       },
 563       config=run_config)
 564 
 565 def per_replica_batch_size(batch_size, num_gpus):
 566   """For multi-gpu, batch-size must be a multiple of the number of GPUs.
 567 
 568 
 569   Note that distribution strategy handles this automatically when used with
 570   Keras. For using with Estimator, we need to get per GPU batch.
 571 
 572   Args:
 573     batch_size: Global batch size to be divided among devices. This should be
 574       equal to num_gpus times the single-GPU batch_size for multi-gpu training.
 575     num_gpus: How many GPUs are used with DistributionStrategies.
 576 
 577   Returns:
 578     Batch size per device.
 579 
 580   Raises:
 581     ValueError: if batch_size is not divisible by number of devices
 582   """
 583   if num_gpus <= 1:
 584     return batch_size
 585 
 586   remainder = batch_size % num_gpus
 587   if remainder:
 588     err = ('When running with multiple GPUs, batch size '
 589            'must be a multiple of the number of available GPUs. Found {} '
 590            'GPUs with a batch size of {}; try --batch_size={} instead.'
 591           ).format(num_gpus, batch_size, batch_size - remainder)
 592     raise ValueError(err)
 593   return int(batch_size / num_gpus)

 596 def run_transformer(flags_obj):
 597   """Create tf.Estimator to train and evaluate transformer model.
 598 
 599   Args:
 600     flags_obj: Object containing parsed flag values.
 601 
 602   Returns:
 603     Dict of results of the run.  Contains the keys `eval_results`,
 604     `train_hooks`, `bleu_cased`, and `bleu_uncased`. `train_hooks` is a list the
 605     instances of hooks used during training.
 606   """
 607   num_gpus = flags_core.get_num_gpus(flags_obj)
 608 
 609   # Add flag-defined parameters to params object
 610   params = PARAMS_MAP[flags_obj.param_set]
 611   if num_gpus > 1:
 612     if flags_obj.param_set == "big":
 613       params = model_params.BIG_MULTI_GPU_PARAMS
 614     elif flags_obj.param_set == "base":
 615       params = model_params.BASE_MULTI_GPU_PARAMS
 616 
 617   params["data_dir"] = flags_obj.data_dir
 618   params["model_dir"] = flags_obj.model_dir
 619   params["num_parallel_calls"] = flags_obj.num_parallel_calls
 620 
 621   params["tpu"] = flags_obj.tpu
 622   params["use_tpu"] = bool(flags_obj.tpu)  # was a tpu specified.
 623   params["static_batch"] = flags_obj.static_batch or params["use_tpu"]
 624   params["allow_ffn_pad"] = not params["use_tpu"]
 625 
 626   params["max_length"] = flags_obj.max_length or params["max_length"]
 627 
 628   params["use_synthetic_data"] = flags_obj.use_synthetic_data
 629 
 630   # Set batch size parameter, which depends on the availability of
 631   # TPU and GPU, and distribution settings.
 632   params["batch_size"] = (flags_obj.batch_size or (
 633       params["default_batch_size_tpu"] if params["use_tpu"]
 634       else params["default_batch_size"]))
 635 
 636   total_batch_size = params["batch_size"]
 637   if not params["use_tpu"]:
 638     params["batch_size"] = per_replica_batch_size(params["batch_size"],
 639                                                   num_gpus)
 640 
 641   schedule_manager = schedule.Manager(
 642       train_steps=flags_obj.train_steps,
 643       steps_between_evals=flags_obj.steps_between_evals,
 644       train_epochs=flags_obj.train_epochs,
 645       epochs_between_evals=flags_obj.epochs_between_evals,
 646       default_train_epochs=DEFAULT_TRAIN_EPOCHS,
 647       batch_size=params["batch_size"],
 648       max_length=params["max_length"],
 649       use_tpu=params["use_tpu"],
 650       num_tpu_shards=flags_obj.num_tpu_shards
 651   )
 652 
 653   params["repeat_dataset"] = schedule_manager.repeat_dataset
 654 
 655   model_helpers.apply_clean(flags.FLAGS)
 656 
 657   # Create hooks that log information about the training and metric values
 658   train_hooks = hooks_helper.get_train_hooks(
 659       flags_obj.hooks,
 660       model_dir=flags_obj.model_dir,
 661       tensors_to_log=TENSORS_TO_LOG,  # used for logging hooks
 662       batch_size=total_batch_size,  # for ExamplesPerSecondHook
 663       use_tpu=params["use_tpu"]  # Not all hooks can run with TPUs
 664   )
 665   benchmark_logger = logger.get_benchmark_logger()
 666   benchmark_logger.log_run_info(
 667       model_name="transformer",
 668       dataset_name="wmt_translate_ende",
 669       run_params=params,
 670       test_id=flags_obj.benchmark_test_id)
 671 
 672   # Train and evaluate transformer model
 673   estimator = construct_estimator(flags_obj, params, schedule_manager)
 674   stats = run_loop(
 675       estimator=estimator,
 676       # Training arguments
 677       schedule_manager=schedule_manager,
 678       train_hooks=train_hooks,
 679       benchmark_logger=benchmark_logger,
 680       # BLEU calculation arguments
 681       bleu_source=flags_obj.bleu_source,
 682       bleu_ref=flags_obj.bleu_ref,
 683       bleu_threshold=flags_obj.stop_threshold,
 684       vocab_file=flags_obj.vocab_file)
 685 
 686   if flags_obj.export_dir and not params["use_tpu"]:
 687     serving_input_fn = export.build_tensor_serving_input_receiver_fn(
 688         shape=[None], dtype=tf.int64, batch_size=None)
 689     # Export saved model, and save the vocab file as an extra asset. The vocab
 690     # file is saved to allow consistent input encoding and output decoding.
 691     # (See the "Export trained model" section in the README for an example of
 692     # how to use the vocab file.)
 693     # Since the model itself does not use the vocab file, this file is saved as
 694     # an extra asset rather than a core asset.
 695     estimator.export_savedmodel(
 696         flags_obj.export_dir, serving_input_fn,
 697         assets_extra={"vocab.txt": flags_obj.vocab_file},
 698         strip_default_attrs=True)
 699   return stats
 700

 702 def main(_):
 703   with logger.benchmark_context(flags.FLAGS):
 704     run_transformer(flags.FLAGS)

 707 if __name__ == "__main__":
 708   tf.logging.set_verbosity(tf.logging.INFO)
 709   define_transformer_flags()
