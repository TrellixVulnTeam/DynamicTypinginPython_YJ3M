  29 def dummy_input_fn():
  30   image = tf.random.uniform([BATCH_SIZE, 784])
  31   labels = tf.random.uniform([BATCH_SIZE, 1], maxval=9, dtype=tf.int32)
  32   return image, labels

  35 def make_estimator():
  36   data_format = 'channels_last'
  37   if tf.test.is_built_with_cuda():
  38     data_format = 'channels_first'
  39   return tf.estimator.Estimator(
  40       model_fn=mnist.model_fn, params={
  41           'data_format': data_format
  42       })

  45 class Tests(tf.test.TestCase):
  46   """Run tests for MNIST model.
  47 
  48   MNIST uses contrib and will not work with TF 2.0.  All tests are disabled if
  49   using TF 2.0.
  50   """
  51 
  52   def test_mnist(self):
  53     classifier = make_estimator()
  54     classifier.train(input_fn=dummy_input_fn, steps=2)
  55     eval_results = classifier.evaluate(input_fn=dummy_input_fn, steps=1)
  56 
  57     loss = eval_results['loss']
  58     global_step = eval_results['global_step']
  59     accuracy = eval_results['accuracy']
  60     self.assertEqual(loss.shape, ())
  61     self.assertEqual(2, global_step)
  62     self.assertEqual(accuracy.shape, ())
  63 
  64     input_fn = lambda: tf.random.uniform([3, 784])
  65     predictions_generator = classifier.predict(input_fn)
  66     for _ in range(3):
  67       predictions = next(predictions_generator)
  68       self.assertEqual(predictions['probabilities'].shape, (10,))
  69       self.assertEqual(predictions['classes'].shape, ())
  70 
  71   def mnist_model_fn_helper(self, mode, multi_gpu=False):
  72     features, labels = dummy_input_fn()
  73     image_count = features.shape[0]
  74     spec = mnist.model_fn(features, labels, mode, {
  75         'data_format': 'channels_last',
  76         'multi_gpu': multi_gpu
  77     })
  78 
  79     if mode == tf.estimator.ModeKeys.PREDICT:
  80       predictions = spec.predictions
  81       self.assertAllEqual(predictions['probabilities'].shape, (image_count, 10))
  82       self.assertEqual(predictions['probabilities'].dtype, tf.float32)
  83       self.assertAllEqual(predictions['classes'].shape, (image_count,))
  84       self.assertEqual(predictions['classes'].dtype, tf.int64)
  85 
  86     if mode != tf.estimator.ModeKeys.PREDICT:
  87       loss = spec.loss
  88       self.assertAllEqual(loss.shape, ())
  89       self.assertEqual(loss.dtype, tf.float32)
  90 
  91     if mode == tf.estimator.ModeKeys.EVAL:
  92       eval_metric_ops = spec.eval_metric_ops
  93       self.assertAllEqual(eval_metric_ops['accuracy'][0].shape, ())
  94       self.assertAllEqual(eval_metric_ops['accuracy'][1].shape, ())
  95       self.assertEqual(eval_metric_ops['accuracy'][0].dtype, tf.float32)
  96       self.assertEqual(eval_metric_ops['accuracy'][1].dtype, tf.float32)
  97 
  98   def test_mnist_model_fn_train_mode(self):
  99     self.mnist_model_fn_helper(tf.estimator.ModeKeys.TRAIN)

 105     self.mnist_model_fn_helper(tf.estimator.ModeKeys.EVAL)

 108     self.mnist_model_fn_helper(tf.estimator.ModeKeys.PREDICT)

 111 class Benchmarks(tf.test.Benchmark):
 112   """Simple speed benchmarking for MNIST."""
 113 
 114   def benchmark_train_step_time(self):
 115     classifier = make_estimator()
 116     # Run one step to warmup any use of the GPU.
 117     classifier.train(input_fn=dummy_input_fn, steps=1)
 118 
 119     have_gpu = tf.test.is_gpu_available()
 120     num_steps = 1000 if have_gpu else 100
 121     name = 'train_step_time_%s' % ('gpu' if have_gpu else 'cpu')
 122 
 123     start = time.time()
 124     classifier.train(input_fn=dummy_input_fn, steps=num_steps)
 125     end = time.time()
 126 
 127     wall_time = (end - start) / num_steps
 128     self.report_benchmark(
 129         iters=num_steps,
 130         wall_time=wall_time,
 131         name=name,
 132         extras={
 133             'examples_per_sec': BATCH_SIZE / wall_time
 134         })
