  16 
  17 This program demonstrates training of the convolutional neural network model
  18 defined in mnist.py on Google Cloud TPUs (https://cloud.google.com/tpu/).

  20 If you are not interested in TPUs, you should ignore this file.
  21 """

  40 # Cloud TPU Cluster Resolver flags
  41 tf.flags.DEFINE_string(
  42     "tpu", default=None,
  43     help="The Cloud TPU to use for training. This should be either the name "
  44     "used when creating the Cloud TPU, or a grpc://ip.address.of.tpu:8470 "

  46 tf.flags.DEFINE_string(
  47     "tpu_zone", default=None,

  58 tf.flags.DEFINE_string("data_dir", "",

  79 def metric_fn(labels, logits):
  80   accuracy = tf.metrics.accuracy(
  81       labels=labels, predictions=tf.argmax(logits, axis=1))
  82   return {"accuracy": accuracy}
  83 
  84 
  85 def model_fn(features, labels, mode, params):
  86   """model_fn constructs the ML model used to predict handwritten digits."""
  87 
  88   del params
  89   image = features
  90   if isinstance(image, dict):
  91     image = features["image"]
  92 
  93   model = mnist.create_model("channels_last")
  94 
  95   if mode == tf.estimator.ModeKeys.PREDICT:
  96     logits = model(image, training=False)
  97     predictions = {
  98         'class_ids': tf.argmax(logits, axis=1),
  99         'probabilities': tf.nn.softmax(logits),
 100     }
 101     return tf.estimator.tpu.TPUEstimatorSpec(mode, predictions=predictions)
 102 
 103   logits = model(image, training=(mode == tf.estimator.ModeKeys.TRAIN))
 104   loss = tf.losses.sparse_softmax_cross_entropy(labels=labels, logits=logits)
 105 
 106   if mode == tf.estimator.ModeKeys.TRAIN:
 107     learning_rate = tf.train.exponential_decay(
 108         FLAGS.learning_rate,
 109         tf.train.get_global_step(),
 110         decay_steps=100000,
 111         decay_rate=0.96)
 112     optimizer = tf.train.GradientDescentOptimizer(learning_rate=learning_rate)
 113     if FLAGS.use_tpu:
 114       optimizer = tf.tpu.CrossShardOptimizer(optimizer)
 115     return tf.estimator.tpu.TPUEstimatorSpec(
 116         mode=mode,
 117         loss=loss,
 118         train_op=optimizer.minimize(loss, tf.train.get_global_step()))
 119 
 120   if mode == tf.estimator.ModeKeys.EVAL:
 121     return tf.estimator.tpu.TPUEstimatorSpec(
 122         mode=mode, loss=loss, eval_metrics=(metric_fn, [labels, logits]))

 125 def train_input_fn(params):
 126   """train_input_fn defines the input pipeline used for training."""
 127   batch_size = params["batch_size"]
 128   data_dir = params["data_dir"]
 129   # Retrieves the batch size for the current shard. The # of shards is
 130   # computed according to the input pipeline deployment. See
 131   # `tf.estimator.tpu.RunConfig` for details.
 132   ds = dataset.train(data_dir).cache().repeat().shuffle(
 133       buffer_size=50000).batch(batch_size, drop_remainder=True)
 134   return ds
 135 
 136 
 137 def eval_input_fn(params):
 138   batch_size = params["batch_size"]
 139   data_dir = params["data_dir"]
 140   ds = dataset.test(data_dir).batch(batch_size, drop_remainder=True)
 141   return ds
 142 
 143 
 144 def predict_input_fn(params):
 145   batch_size = params["batch_size"]
 146   data_dir = params["data_dir"]
 147   # Take out top 10 samples from test data to make the predictions.
 148   ds = dataset.test(data_dir).take(10).batch(batch_size)
 149   return ds

 152 def main(argv):
 153   del argv  # Unused.
 154   tf.logging.set_verbosity(tf.logging.INFO)
 155 
 156   tpu_cluster_resolver = tf.distribute.cluster_resolver.TPUClusterResolver(
 157       FLAGS.tpu,
 158       zone=FLAGS.tpu_zone,
 159       project=FLAGS.gcp_project
 160   )
 161 
 162   run_config = tf.estimator.tpu.RunConfig(
 163       cluster=tpu_cluster_resolver,
 164       model_dir=FLAGS.model_dir,
 165       session_config=tf.ConfigProto(
 166           allow_soft_placement=True, log_device_placement=True),
 167       tpu_config=tf.estimator.tpu.TPUConfig(FLAGS.iterations, FLAGS.num_shards),
 168   )
 169 
 170   estimator = tf.estimator.tpu.TPUEstimator(
 171       model_fn=model_fn,
 172       use_tpu=FLAGS.use_tpu,
 173       train_batch_size=FLAGS.batch_size,
 174       eval_batch_size=FLAGS.batch_size,
 175       predict_batch_size=FLAGS.batch_size,
 176       params={"data_dir": FLAGS.data_dir},
 177       config=run_config)
 178   # TPUEstimator.train *requires* a max_steps argument.
 179   estimator.train(input_fn=train_input_fn, max_steps=FLAGS.train_steps)
 180   # TPUEstimator.evaluate *requires* a steps argument.
 181   # Note that the number of examples used during evaluation is
 182   # --eval_steps * --batch_size.
 183   # So if you change --batch_size then change --eval_steps too.
 184   if FLAGS.eval_steps:
 185     estimator.evaluate(input_fn=eval_input_fn, steps=FLAGS.eval_steps)
 186 
 187   # Run prediction on top few samples of test data.
 188   if FLAGS.enable_predict:
 189     predictions = estimator.predict(input_fn=predict_input_fn)
 190 
 191     for pred_dict in predictions:
 192       template = ('Prediction is "{}" ({:.1f}%).')
 193 
 194       class_id = pred_dict['class_ids']
 195       probability = pred_dict['probabilities'][class_id]
 196 
 197       print(template.format(class_id, 100 * probability))
 198 
 199 
 200 if __name__ == "__main__":
 201   tf.disable_v2_behavior()
 202   absl_app.run(main)
