  36 def create_model(data_format):
  37   """Model to recognize digits in the MNIST dataset.
  38 
  39   Network structure is equivalent to:
  40   https://github.com/tensorflow/tensorflow/blob/r1.5/tensorflow/examples/tutorials/mnist/mnist_deep.py
  41 
  42   But uses the tf.keras API.
  43 
  44   Args:
  45     data_format: Either 'channels_first' or 'channels_last'. 'channels_first' is
  46       typically faster on GPUs while 'channels_last' is typically faster on
  47       CPUs. See
  48       https://www.tensorflow.org/performance/performance_guide#data_formats
  49 
  50   Returns:
  51     A tf.keras.Model.
  52   """
  53   if data_format == 'channels_first':
  54     input_shape = [1, 28, 28]
  55   else:
  56     assert data_format == 'channels_last'
  57     input_shape = [28, 28, 1]
  58 
  59   l = tf.keras.layers
  60   max_pool = l.MaxPooling2D(
  61       (2, 2), (2, 2), padding='same', data_format=data_format)
  62   # The model consists of a sequential chain of layers, so tf.keras.Sequential
  63   # (a subclass of tf.keras.Model) makes for a compact description.
  64   return tf.keras.Sequential(
  65       [
  66           l.Reshape(
  67               target_shape=input_shape,
  68               input_shape=(28 * 28,)),
  69           l.Conv2D(
  70               32,
  71               5,
  72               padding='same',
  73               data_format=data_format,
  74               activation=tf.nn.relu),
  75           max_pool,
  76           l.Conv2D(
  77               64,
  78               5,
  79               padding='same',
  80               data_format=data_format,
  81               activation=tf.nn.relu),
  82           max_pool,
  83           l.Flatten(),
  84           l.Dense(1024, activation=tf.nn.relu),
  85           l.Dropout(0.4),
  86           l.Dense(10)
  87       ])

 107 def model_fn(features, labels, mode, params):
 108   """The model_fn argument for creating an Estimator."""
 109   model = create_model(params['data_format'])
 110   image = features
 111   if isinstance(image, dict):
 112     image = features['image']
 113 
 114   if mode == tf.estimator.ModeKeys.PREDICT:
 115     logits = model(image, training=False)
 116     predictions = {
 117         'classes': tf.argmax(logits, axis=1),
 118         'probabilities': tf.nn.softmax(logits),
 119     }
 120     return tf.estimator.EstimatorSpec(
 121         mode=tf.estimator.ModeKeys.PREDICT,
 122         predictions=predictions,
 123         export_outputs={
 124             'classify': tf.estimator.export.PredictOutput(predictions)
 125         })
 126   if mode == tf.estimator.ModeKeys.TRAIN:
 127     optimizer = tf.compat.v1.train.AdamOptimizer(learning_rate=LEARNING_RATE)
 128 
 129     logits = model(image, training=True)
 130     loss = tf.compat.v1.losses.sparse_softmax_cross_entropy(labels=labels,
 131                                                             logits=logits)
 132     accuracy = tf.compat.v1.metrics.accuracy(
 133         labels=labels, predictions=tf.argmax(logits, axis=1))
 134 
 135     # Name tensors to be logged with LoggingTensorHook.
 136     tf.identity(LEARNING_RATE, 'learning_rate')
 137     tf.identity(loss, 'cross_entropy')
 138     tf.identity(accuracy[1], name='train_accuracy')
 139 
 140     # Save accuracy scalar to Tensorboard output.
 141     tf.summary.scalar('train_accuracy', accuracy[1])
 142 
 143     return tf.estimator.EstimatorSpec(
 144         mode=tf.estimator.ModeKeys.TRAIN,
 145         loss=loss,
 146         train_op=optimizer.minimize(
 147             loss,
 148             tf.compat.v1.train.get_or_create_global_step()))
 149   if mode == tf.estimator.ModeKeys.EVAL:
 150     logits = model(image, training=False)
 151     loss = tf.losses.sparse_softmax_cross_entropy(labels=labels, logits=logits)
 152     return tf.estimator.EstimatorSpec(
 153         mode=tf.estimator.ModeKeys.EVAL,
 154         loss=loss,
 155         eval_metric_ops={
 156             'accuracy':
 157                 tf.metrics.accuracy(
 158                     labels=labels, predictions=tf.argmax(logits, axis=1)),
 159         })

 162 def run_mnist(flags_obj):
 163   """Run MNIST training and eval loop.
 164 
 165   Args:
 166     flags_obj: An object containing parsed flag values.
 167   """
 168   model_helpers.apply_clean(flags_obj)
 169   model_function = model_fn
 170 
 171   session_config = tf.compat.v1.ConfigProto(
 172       inter_op_parallelism_threads=flags_obj.inter_op_parallelism_threads,
 173       intra_op_parallelism_threads=flags_obj.intra_op_parallelism_threads,
 174       allow_soft_placement=True)
 175 
 176   distribution_strategy = distribution_utils.get_distribution_strategy(
 177       distribution_strategy=flags_obj.distribution_strategy,
 178       num_gpus=flags_core.get_num_gpus(flags_obj),
 179       all_reduce_alg=flags_obj.all_reduce_alg)
 180 
 181   run_config = tf.estimator.RunConfig(
 182       train_distribute=distribution_strategy, session_config=session_config)
 183 
 184   data_format = flags_obj.data_format
 185   if data_format is None:
 186     data_format = ('channels_first' if tf.config.list_physical_devices('GPU')
 187                    else 'channels_last')
 188   mnist_classifier = tf.estimator.Estimator(
 189       model_fn=model_function,
 190       model_dir=flags_obj.model_dir,
 191       config=run_config,
 192       params={
 193           'data_format': data_format,
 194       })
 195 
 196   # Set up training and evaluation input functions.
 197   def train_input_fn():
 198     """Prepare data for training."""
 199 
 200     # When choosing shuffle buffer sizes, larger sizes result in better
 201     # randomness, while smaller sizes use less memory. MNIST is a small
 202     # enough dataset that we can easily shuffle the full epoch.
 203     ds = dataset.train(flags_obj.data_dir)
 204     ds = ds.cache().shuffle(buffer_size=50000).batch(flags_obj.batch_size)
 205 
 206     # Iterate through the dataset a set number (`epochs_between_evals`) of times
 207     # during each training session.
 208     ds = ds.repeat(flags_obj.epochs_between_evals)
 209     return ds
 210 
 211   def eval_input_fn():
 212     return dataset.test(flags_obj.data_dir).batch(
 213         flags_obj.batch_size).make_one_shot_iterator().get_next()
 214 
 215   # Set up hook that outputs training logs every 100 steps.
 216   train_hooks = hooks_helper.get_train_hooks(
 217       flags_obj.hooks, model_dir=flags_obj.model_dir,
 218       batch_size=flags_obj.batch_size)
 219 
 220   # Train and evaluate model.
 221   for _ in range(flags_obj.train_epochs // flags_obj.epochs_between_evals):
 222     mnist_classifier.train(input_fn=train_input_fn, hooks=train_hooks)
 223     eval_results = mnist_classifier.evaluate(input_fn=eval_input_fn)
 224     print('\nEvaluation results:\n\t%s\n' % eval_results)
 225 
 226     if model_helpers.past_stop_threshold(flags_obj.stop_threshold,
 227                                          eval_results['accuracy']):
 228       break
 229 
 230   # Export the model
 231   if flags_obj.export_dir is not None:
 232     image = tf.compat.v1.placeholder(tf.float32, [None, 28, 28])
 233     input_fn = tf.estimator.export.build_raw_serving_input_receiver_fn({
 234         'image': image,
 235     })
 236     mnist_classifier.export_savedmodel(flags_obj.export_dir, input_fn,
 237                                        strip_default_attrs=True)

 241   run_mnist(flags.FLAGS)
