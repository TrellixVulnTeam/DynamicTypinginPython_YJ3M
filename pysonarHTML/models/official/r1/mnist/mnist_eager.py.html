  47 def loss(logits, labels):
  48   return tf.reduce_mean(
  49       tf.nn.sparse_softmax_cross_entropy_with_logits(
  50           logits=logits, labels=labels))
  51 
  52 
  53 def compute_accuracy(logits, labels):
  54   predictions = tf.argmax(logits, axis=1, output_type=tf.int64)
  55   labels = tf.cast(labels, tf.int64)
  56   batch_size = int(logits.shape[0])
  57   return tf.reduce_sum(
  58       tf.cast(tf.equal(predictions, labels), dtype=tf.float32)) / batch_size
  59

  61 def train(model, optimizer, dataset, step_counter, log_interval=None):
  62   """Trains model on `dataset` using `optimizer`."""
  63   from tensorflow.contrib import summary as contrib_summary  # pylint: disable=g-import-not-at-top
  64 
  65   start = time.time()
  66   for (batch, (images, labels)) in enumerate(dataset):
  67     with contrib_summary.record_summaries_every_n_global_steps(
  68         10, global_step=step_counter):
  69       # Record the operations used to compute the loss given the input,
  70       # so that the gradient of the loss with respect to the variables
  71       # can be computed.
  72       with tf.GradientTape() as tape:
  73         logits = model(images, training=True)
  74         loss_value = loss(logits, labels)
  75         contrib_summary.scalar('loss', loss_value)
  76         contrib_summary.scalar('accuracy',
  77                                     compute_accuracy(logits, labels))
  78       grads = tape.gradient(loss_value, model.variables)
  79       optimizer.apply_gradients(
  80           list(zip(grads, model.variables)), global_step=step_counter)
  81       if log_interval and batch % log_interval == 0:
  82         rate = log_interval / (time.time() - start)
  83         print('Step #%d\tLoss: %.6f (%d steps/sec)' % (batch, loss_value, rate))
  84         start = time.time()

  87 def test(model, dataset):
  88   """Perform an evaluation of `model` on the examples from `dataset`."""
  89   from tensorflow.contrib import summary as contrib_summary  # pylint: disable=g-import-not-at-top
  90   avg_loss = tf.keras.metrics.Mean('loss', dtype=tf.float32)
  91   accuracy = tf.keras.metrics.Accuracy('accuracy', dtype=tf.float32)
  92 
  93   for (images, labels) in dataset:
  94     logits = model(images, training=False)
  95     avg_loss.update_state(loss(logits, labels))
  96     accuracy.update_state(
  97         tf.argmax(logits, axis=1, output_type=tf.int64),
  98         tf.cast(labels, tf.int64))
  99   print('Test set: Average loss: %.4f, Accuracy: %4f%%\n' %
 100         (avg_loss.result(), 100 * accuracy.result()))
 101   with contrib_summary.always_record_summaries():
 102     contrib_summary.scalar('loss', avg_loss.result())
 103     contrib_summary.scalar('accuracy', accuracy.result())

 106 def run_mnist_eager(flags_obj):
 107   """Run MNIST training and eval loop in eager mode.
 108 
 109   Args:
 110     flags_obj: An object containing parsed flag values.
 111   """
 112   tf.enable_eager_execution()
 113   model_helpers.apply_clean(flags.FLAGS)
 114 
 115   # Automatically determine device and data_format
 116   (device, data_format) = ('/gpu:0', 'channels_first')
 117   if flags_obj.no_gpu or not tf.test.is_gpu_available():
 118     (device, data_format) = ('/cpu:0', 'channels_last')
 119   # If data_format is defined in FLAGS, overwrite automatically set value.
 120   if flags_obj.data_format is not None:
 121     data_format = flags_obj.data_format
 122   print('Using device %s, and data format %s.' % (device, data_format))
 123 
 124   # Load the datasets
 125   train_ds = mnist_dataset.train(flags_obj.data_dir).shuffle(60000).batch(
 126       flags_obj.batch_size)
 127   test_ds = mnist_dataset.test(flags_obj.data_dir).batch(
 128       flags_obj.batch_size)
 129 
 130   # Create the model and optimizer
 131   model = mnist.create_model(data_format)
 132   optimizer = tf.train.MomentumOptimizer(flags_obj.lr, flags_obj.momentum)
 133 
 134   # Create file writers for writing TensorBoard summaries.
 135   if flags_obj.output_dir:
 136     # Create directories to which summaries will be written
 137     # tensorboard --logdir=<output_dir>
 138     # can then be used to see the recorded summaries.
 139     train_dir = os.path.join(flags_obj.output_dir, 'train')
 140     test_dir = os.path.join(flags_obj.output_dir, 'eval')
 141     tf.gfile.MakeDirs(flags_obj.output_dir)
 142   else:
 143     train_dir = None
 144     test_dir = None
 145   summary_writer = tf.compat.v2.summary.create_file_writer(
 146       train_dir, flush_millis=10000)
 147   test_summary_writer = tf.compat.v2.summary.create_file_writer(
 148       test_dir, flush_millis=10000, name='test')
 149 
 150   # Create and restore checkpoint (if one exists on the path)
 151   checkpoint_prefix = os.path.join(flags_obj.model_dir, 'ckpt')
 152   step_counter = tf.train.get_or_create_global_step()
 153   checkpoint = tf.train.Checkpoint(
 154       model=model, optimizer=optimizer, step_counter=step_counter)
 155   # Restore variables on creation if a checkpoint exists.
 156   checkpoint.restore(tf.train.latest_checkpoint(flags_obj.model_dir))
 157 
 158   # Train and evaluate for a set number of epochs.
 159   with tf.device(device):
 160     for _ in range(flags_obj.train_epochs):
 161       start = time.time()
 162       with summary_writer.as_default():
 163         train(model, optimizer, train_ds, step_counter,
 164               flags_obj.log_interval)
 165       end = time.time()
 166       print('\nTrain time for epoch #%d (%d total steps): %f' %
 167             (checkpoint.save_counter.numpy() + 1,
 168              step_counter.numpy(),
 169              end - start))
 170       with test_summary_writer.as_default():
 171         test(model, test_ds)
 172       checkpoint.save(checkpoint_prefix)

 175 def define_mnist_eager_flags():
 176   """Defined flags and defaults for MNIST in eager mode."""
 177   flags_core.define_base(clean=True, train_epochs=True, export_dir=True,
 178                          distribution_strategy=True)
 179   flags_core.define_image()
 180   flags.adopt_module_key_flags(flags_core)
 181 
 182   flags.DEFINE_integer(
 183       name='log_interval', short_name='li', default=10,
 184       help=flags_core.help_wrap('batches between logging training status'))
 185 
 186   flags.DEFINE_string(
 187       name='output_dir', short_name='od', default=None,
 188       help=flags_core.help_wrap('Directory to write TensorBoard summaries'))
 189 
 190   flags.DEFINE_float(name='learning_rate', short_name='lr', default=0.01,
 191                      help=flags_core.help_wrap('Learning rate.'))
 192 
 193   flags.DEFINE_float(name='momentum', short_name='m', default=0.5,
 194                      help=flags_core.help_wrap('SGD momentum.'))
 195 
 196   flags.DEFINE_bool(name='no_gpu', short_name='nogpu', default=False,
 197                     help=flags_core.help_wrap(
 198                         'disables GPU usage even if a GPU is available'))
 199 
 200   flags_core.set_defaults(
 201       data_dir='/tmp/tensorflow/mnist/input_data',
 202       model_dir='/tmp/tensorflow/mnist/checkpoints/',
 203       batch_size=100,
 204       train_epochs=10,
 205   )
