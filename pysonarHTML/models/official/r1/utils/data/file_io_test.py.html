  86 @contextlib.contextmanager
  87 def fixed_core_count(cpu_count):
  88   """Override CPU count.
  89 
  90   file_io.py uses the cpu_count function to scale to the size of the instance.
  91   However, this is not desirable for testing because it can make the test flaky.
  92   Instead, this context manager fixes the count for more robust testing.
  93 
  94   Args:
  95     cpu_count: How many cores multiprocessing claims to have.
  96 
  97   Yields:
  98     Nothing. (for context manager only)
  99   """
 100   old_count_fn = multiprocessing.cpu_count
 101   multiprocessing.cpu_count = lambda: cpu_count
 102   yield
 103   multiprocessing.cpu_count = old_count_fn

 112   def _test_sharding(self, row_count, cpu_count, expected):
 113     df = pd.DataFrame({_DUMMY_COL: list(range(row_count))})
 114     with fixed_core_count(cpu_count):
 115       shards = list(file_io.iter_shard_dataframe(df, _ROWS_PER_CORE))
 116     result = [[j[_DUMMY_COL].tolist() for j in i] for i in shards]
 117     self.assertAllEqual(expected, result)

 143   def _serialize_deserialize(self, num_cores=1, num_rows=20):
 144     np.random.seed(1)
 145     df = pd.DataFrame({
 146         # Serialization order is only deterministic for num_cores=1. raw_row is
 147         # used in validation after the deserialization.
 148         _RAW_ROW: np.array(range(num_rows), dtype=np.int64),
 149         _DUMMY_COL: np.random.randint(0, 35, size=(num_rows,)),
 150         _DUMMY_VEC_COL: [
 151             np.array([np.random.random() for _ in range(_DUMMY_VEC_LEN)])
 152             for i in range(num_rows)  # pylint: disable=unused-variable
 153         ]
 154     })
 155 
 156     with fixed_core_count(num_cores):
 157       buffer_path = file_io.write_to_temp_buffer(
 158           df, self.get_temp_dir(), [_RAW_ROW, _DUMMY_COL, _DUMMY_VEC_COL])
 159 
 160     with self.session(graph=tf.Graph()) as sess:
 161       dataset = tf.data.TFRecordDataset(buffer_path)
 162       dataset = dataset.batch(1).map(
 163           lambda x: tf.io.parse_example(serialized=x, features=_FEATURE_MAP))
 164 
 165       data_iter = tf.compat.v1.data.make_one_shot_iterator(dataset)
 166       seen_rows = set()
 167       for i in range(num_rows+5):
 168         row = data_iter.get_next()
 169         try:
 170           row_id, val_0, val_1 = sess.run(
 171               [row[_RAW_ROW], row[_DUMMY_COL], row[_DUMMY_VEC_COL]])
 172           row_id, val_0, val_1 = row_id[0][0], val_0[0][0], val_1[0]
 173           assert row_id not in seen_rows
 174           seen_rows.add(row_id)
 175 
 176           self.assertEqual(val_0, df[_DUMMY_COL][row_id])
 177           self.assertAllClose(val_1, df[_DUMMY_VEC_COL][row_id])
 178 
 179           self.assertLess(i, num_rows, msg="Too many rows.")
 180         except tf.errors.OutOfRangeError:
 181           self.assertGreaterEqual(i, num_rows, msg="Too few rows.")
 182 
 183     file_io._GARBAGE_COLLECTOR.purge()
 184     assert not tf.io.gfile.exists(buffer_path)
 185
