  35 class _GarbageCollector(object):
  36   """Deletes temporary buffer files at exit.
  37 
  38   Certain tasks (such as NCF Recommendation) require writing buffers to
  39   temporary files. (Which may be local or distributed.) It is not generally safe
  40   to delete these files during operation, but they should be cleaned up. This
  41   class keeps track of temporary files created, and deletes them at exit.
  42   """
  43   def __init__(self):
  44     self.temp_buffers = []
  45 
  46   def register(self, filepath):
  47     self.temp_buffers.append(filepath)
  48 
  49   def purge(self):
  50     try:
  51       for i in self.temp_buffers:
  52         if tf.io.gfile.exists(i):
  53           tf.io.gfile.remove(i)
  54           logging.info("Buffer file {} removed".format(i))
  55     except Exception as e:
  56       logging.error("Failed to cleanup buffer files: {}".format(e))
  57 
  58 
  59 _GARBAGE_COLLECTOR = _GarbageCollector()
  60 atexit.register(_GARBAGE_COLLECTOR.purge)
  61 
  62 _ROWS_PER_CORE = 50000

  65 def write_to_temp_buffer(dataframe, buffer_folder, columns):
  66   if buffer_folder is None:
  67     _, buffer_path = tempfile.mkstemp()
  68   else:
  69     tf.io.gfile.makedirs(buffer_folder)
  70     buffer_path = os.path.join(buffer_folder, str(uuid.uuid4()))
  71   _GARBAGE_COLLECTOR.register(buffer_path)
  72 
  73   return write_to_buffer(dataframe, buffer_path, columns)

  76 def iter_shard_dataframe(df, rows_per_core=1000):
  77   """Two way shard of a dataframe.
  78 
  79   This function evenly shards a dataframe so that it can be mapped efficiently.
  80   It yields a list of dataframes with length equal to the number of CPU cores,
  81   with each dataframe having rows_per_core rows. (Except for the last batch
  82   which may have fewer rows in the dataframes.) Passing vectorized inputs to
  83   a pool is more effecient than iterating through a dataframe in serial and
  84   passing a list of inputs to the pool.
  85 
  86   Args:
  87     df: Pandas dataframe to be sharded.
  88     rows_per_core: Number of rows in each shard.
  89 
  90   Returns:
  91     A list of dataframe shards.
  92   """
  93   n = len(df)
  94   num_cores = min([multiprocessing.cpu_count(), n])
  95 
  96   num_blocks = int(np.ceil(n / num_cores / rows_per_core))
  97   max_batch_size = num_cores * rows_per_core
  98   for i in range(num_blocks):
  99     min_index = i * max_batch_size
 100     max_index = min([(i + 1) * max_batch_size, n])
 101     df_shard = df[min_index:max_index]
 102     n_shard = len(df_shard)
 103     boundaries = np.linspace(0, n_shard, num_cores + 1, dtype=np.int64)
 104     yield [df_shard[boundaries[j]:boundaries[j+1]] for j in range(num_cores)]

 107 def _shard_dict_to_examples(shard_dict):
 108   """Converts a dict of arrays into a list of example bytes."""
 109   n = [i for i in shard_dict.values()][0].shape[0]
 110   feature_list = [{} for _ in range(n)]
 111   for column, values in shard_dict.items():
 112     if len(values.shape) == 1:
 113       values = np.reshape(values, values.shape + (1,))
 114 
 115     if values.dtype.kind == "i":
 116       feature_map = lambda x: tf.train.Feature(
 117           int64_list=tf.train.Int64List(value=x))
 118     elif values.dtype.kind == "f":
 119       feature_map = lambda x: tf.train.Feature(
 120           float_list=tf.train.FloatList(value=x))
 121     else:
 122       raise ValueError("Invalid dtype")
 123     for i in range(n):
 124       feature_list[i][column] = feature_map(values[i])
 125   examples = [
 126       tf.train.Example(features=tf.train.Features(feature=example_features))
 127       for example_features in feature_list
 128   ]
 129 
 130   return [e.SerializeToString() for e in examples]

 133 def _serialize_shards(df_shards, columns, pool, writer):
 134   """Map sharded dataframes to bytes, and write them to a buffer.
 135 
 136   Args:
 137     df_shards: A list of pandas dataframes. (Should be of similar size)
 138     columns: The dataframe columns to be serialized.
 139     pool: A pool to serialize in parallel.
 140     writer: A TFRecordWriter to write the serialized shards.
 141   """
 142   # Pandas does not store columns of arrays as nd arrays. stack remedies this.
 143   map_inputs = [{c: np.stack(shard[c].values, axis=0) for c in columns}
 144                 for shard in df_shards]
 145 
 146   # Failure within pools is very irksome. Thus, it is better to thoroughly check
 147   # inputs in the main process.
 148   for inp in map_inputs:
 149     # Check that all fields have the same number of rows.
 150     assert len(set([v.shape[0] for v in inp.values()])) == 1
 151     for val in inp.values():
 152       assert hasattr(val, "dtype")
 153       assert hasattr(val.dtype, "kind")
 154       assert val.dtype.kind in ("i", "f")
 155       assert len(val.shape) in (1, 2)
 156   shard_bytes = pool.map(_shard_dict_to_examples, map_inputs)
 157   for s in shard_bytes:
 158     for example in s:
 159       writer.write(example)

 162 def write_to_buffer(dataframe, buffer_path, columns, expected_size=None):
 163   """Write a dataframe to a binary file for a dataset to consume.
 164 
 165   Args:
 166     dataframe: The pandas dataframe to be serialized.
 167     buffer_path: The path where the serialized results will be written.
 168     columns: The dataframe columns to be serialized.
 169     expected_size: The size in bytes of the serialized results. This is used to
 170       lazily construct the buffer.
 171 
 172   Returns:
 173     The path of the buffer.
 174   """
 175   if (tf.io.gfile.exists(buffer_path) and
 176       tf.io.gfile.stat(buffer_path).length > 0):
 177     actual_size = tf.io.gfile.stat(buffer_path).length
 178     if expected_size == actual_size:
 179       return buffer_path
 180     logging.warning(
 181         "Existing buffer {} has size {}. Expected size {}. Deleting and "
 182         "rebuilding buffer.".format(buffer_path, actual_size, expected_size))
 183     tf.io.gfile.remove(buffer_path)
 184 
 185   if dataframe is None:
 186     raise ValueError(
 187         "dataframe was None but a valid existing buffer was not found.")
 188 
 189   tf.io.gfile.makedirs(os.path.split(buffer_path)[0])
 190 
 191   logging.info("Constructing TFRecordDataset buffer: {}".format(buffer_path))
 192 
 193   count = 0
 194   pool = multiprocessing.dummy.Pool(multiprocessing.cpu_count())
 195   try:
 196     with tf.io.TFRecordWriter(buffer_path) as writer:
 197       for df_shards in iter_shard_dataframe(df=dataframe,
 198                                             rows_per_core=_ROWS_PER_CORE):
 199         _serialize_shards(df_shards, columns, pool, writer)
 200         count += sum([len(s) for s in df_shards])
 201         logging.info("{}/{} examples written.".format(
 202             str(count).ljust(8), len(dataframe)))
 203   finally:
 204     pool.terminate()
 205 
 206   logging.info("Buffer write complete.")
 207   return buffer_path
