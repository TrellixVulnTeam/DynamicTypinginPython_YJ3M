  33 class ExamplesPerSecondHookTest(tf.test.TestCase):
  34   """Tests for the ExamplesPerSecondHook.
  35 
  36   In the test, we explicitly run global_step tensor after train_op in order to
  37   keep the global_step value and the train_op (which increase the glboal_step
  38   by 1) consistent. This is to correct the discrepancies in reported global_step
  39   value when running on GPUs.
  40   """
  41 
  42   def setUp(self):
  43     """Mock out logging calls to verify if correct info is being monitored."""
  44     self._logger = mock_lib.MockBenchmarkLogger()
  45 
  46     self.graph = tf.Graph()
  47     with self.graph.as_default():
  48       tf.compat.v1.train.create_global_step()
  49       self.train_op = tf.compat.v1.assign_add(
  50           tf.compat.v1.train.get_global_step(), 1)
  51       self.global_step = tf.compat.v1.train.get_global_step()
  52 
  53   def test_raise_in_both_secs_and_steps(self):
  54     with self.assertRaises(ValueError):
  55       hooks.ExamplesPerSecondHook(
  56           batch_size=256,
  57           every_n_steps=10,
  58           every_n_secs=20,
  59           metric_logger=self._logger)
  60 
  61   def test_raise_in_none_secs_and_steps(self):
  62     with self.assertRaises(ValueError):
  63       hooks.ExamplesPerSecondHook(
  64           batch_size=256,
  65           every_n_steps=None,
  66           every_n_secs=None,
  67           metric_logger=self._logger)
  68 
  69   def _validate_log_every_n_steps(self, every_n_steps, warm_steps):
  70     hook = hooks.ExamplesPerSecondHook(
  71         batch_size=256,
  72         every_n_steps=every_n_steps,
  73         warm_steps=warm_steps,
  74         metric_logger=self._logger)
  75 
  76     with tf.compat.v1.train.MonitoredSession(
  77         tf.compat.v1.train.ChiefSessionCreator(), [hook]) as mon_sess:
  78       for _ in range(every_n_steps):
  79         # Explicitly run global_step after train_op to get the accurate
  80         # global_step value
  81         mon_sess.run(self.train_op)
  82         mon_sess.run(self.global_step)
  83         # Nothing should be in the list yet
  84         self.assertFalse(self._logger.logged_metric)
  85 
  86       mon_sess.run(self.train_op)
  87       global_step_val = mon_sess.run(self.global_step)
  88 
  89       if global_step_val > warm_steps:
  90         self._assert_metrics()
  91       else:
  92         # Nothing should be in the list yet
  93         self.assertFalse(self._logger.logged_metric)
  94 
  95       # Add additional run to verify proper reset when called multiple times.
  96       prev_log_len = len(self._logger.logged_metric)
  97       mon_sess.run(self.train_op)
  98       global_step_val = mon_sess.run(self.global_step)
  99 
 100       if every_n_steps == 1 and global_step_val > warm_steps:
 101         # Each time, we log two additional metrics. Did exactly 2 get added?
 102         self.assertEqual(len(self._logger.logged_metric), prev_log_len + 2)
 103       else:
 104         # No change in the size of the metric list.
 105         self.assertEqual(len(self._logger.logged_metric), prev_log_len)
 106 
 107   def test_examples_per_sec_every_1_steps(self):
 108     with self.graph.as_default():
 109       self._validate_log_every_n_steps(1, 0)
 110 
 111   def test_examples_per_sec_every_5_steps(self):

 113       self._validate_log_every_n_steps(5, 0)
 114 
 115   def test_examples_per_sec_every_1_steps_with_warm_steps(self):
 116     with self.graph.as_default():
 117       self._validate_log_every_n_steps(1, 10)
 118 
 119   def test_examples_per_sec_every_5_steps_with_warm_steps(self):
 120     with self.graph.as_default():
 121       self._validate_log_every_n_steps(5, 10)

 123   def _validate_log_every_n_secs(self, every_n_secs):
 124     hook = hooks.ExamplesPerSecondHook(
 125         batch_size=256,
 126         every_n_steps=None,
 127         every_n_secs=every_n_secs,
 128         metric_logger=self._logger)
 129 
 130     with tf.compat.v1.train.MonitoredSession(
 131         tf.compat.v1.train.ChiefSessionCreator(), [hook]) as mon_sess:
 132       # Explicitly run global_step after train_op to get the accurate
 133       # global_step value
 134       mon_sess.run(self.train_op)
 135       mon_sess.run(self.global_step)
 136       # Nothing should be in the list yet
 137       self.assertFalse(self._logger.logged_metric)
 138       time.sleep(every_n_secs)
 139 
 140       mon_sess.run(self.train_op)
 141       mon_sess.run(self.global_step)
 142       self._assert_metrics()
 143 
 144   def test_examples_per_sec_every_1_secs(self):
 145     with self.graph.as_default():
 146       self._validate_log_every_n_secs(1)
 147 
 148   def test_examples_per_sec_every_5_secs(self):
 149     with self.graph.as_default():
 150       self._validate_log_every_n_secs(5)
 151 
 152   def _assert_metrics(self):
 153     metrics = self._logger.logged_metric
 154     self.assertEqual(metrics[-2]["name"], "average_examples_per_sec")
 155     self.assertEqual(metrics[-1]["name"], "current_examples_per_sec")
