  56 def config_benchmark_logger(flag_obj=None):
  57   """Config the global benchmark logger."""
  58   _logger_lock.acquire()
  59   try:
  60     global _benchmark_logger
  61     if not flag_obj:
  62       flag_obj = FLAGS
  63 
  64     if (not hasattr(flag_obj, "benchmark_logger_type") or
  65         flag_obj.benchmark_logger_type == "BaseBenchmarkLogger"):
  66       _benchmark_logger = BaseBenchmarkLogger()
  67     elif flag_obj.benchmark_logger_type == "BenchmarkFileLogger":
  68       _benchmark_logger = BenchmarkFileLogger(flag_obj.benchmark_log_dir)
  69     else:
  70       raise ValueError("Unrecognized benchmark_logger_type: %s"
  71                        % flag_obj.benchmark_logger_type)
  72 
  73   finally:
  74     _logger_lock.release()
  75   return _benchmark_logger
  76

  78 def get_benchmark_logger():
  79   if not _benchmark_logger:
  80     config_benchmark_logger()
  81   return _benchmark_logger

  84 @contextlib.contextmanager
  85 def benchmark_context(flag_obj):
  86   """Context of benchmark, which will update status of the run accordingly."""
  87   benchmark_logger = config_benchmark_logger(flag_obj)
  88   try:
  89     yield
  90     benchmark_logger.on_finish(RUN_STATUS_SUCCESS)
  91   except Exception:  # pylint: disable=broad-except
  92     # Catch all the exception, update the run status to be failure, and re-raise
  93     benchmark_logger.on_finish(RUN_STATUS_FAILURE)
  94     raise

  97 class BaseBenchmarkLogger(object):
  98   """Class to log the benchmark information to STDOUT."""
  99 
 100   def log_evaluation_result(self, eval_results):
 101     """Log the evaluation result.
 102 
 103     The evaluate result is a dictionary that contains metrics defined in
 104     model_fn. It also contains a entry for global_step which contains the value
 105     of the global step when evaluation was performed.
 106 
 107     Args:
 108       eval_results: dict, the result of evaluate.
 109     """
 110     if not isinstance(eval_results, dict):
 111       logging.warning("eval_results should be dictionary for logging. Got %s",
 112                       type(eval_results))
 113       return
 114     global_step = eval_results[tf.compat.v1.GraphKeys.GLOBAL_STEP]
 115     for key in sorted(eval_results):
 116       if key != tf.compat.v1.GraphKeys.GLOBAL_STEP:
 117         self.log_metric(key, eval_results[key], global_step=global_step)
 118 
 119   def log_metric(self, name, value, unit=None, global_step=None, extras=None):
 120     """Log the benchmark metric information to local file.
 121 
 122     Currently the logging is done in a synchronized way. This should be updated
 123     to log asynchronously.
 124 
 125     Args:
 126       name: string, the name of the metric to log.
 127       value: number, the value of the metric. The value will not be logged if it
 128         is not a number type.
 129       unit: string, the unit of the metric, E.g "image per second".
 130       global_step: int, the global_step when the metric is logged.
 131       extras: map of string:string, the extra information about the metric.
 132     """
 133     metric = _process_metric_to_json(name, value, unit, global_step, extras)
 134     if metric:
 135       logging.info("Benchmark metric: %s", metric)
 136 
 137   def log_run_info(self, model_name, dataset_name, run_params, test_id=None):
 138     logging.info(
 139         "Benchmark run: %s",
 140         _gather_run_info(model_name, dataset_name, run_params, test_id))
 141 
 142   def on_finish(self, status):
 143     pass

 146 class BenchmarkFileLogger(BaseBenchmarkLogger):
 147   """Class to log the benchmark information to local disk."""
 148 
 149   def __init__(self, logging_dir):
 150     super(BenchmarkFileLogger, self).__init__()
 151     self._logging_dir = logging_dir
 152     if not tf.io.gfile.isdir(self._logging_dir):
 153       tf.io.gfile.makedirs(self._logging_dir)
 154     self._metric_file_handler = tf.io.gfile.GFile(
 155         os.path.join(self._logging_dir, METRIC_LOG_FILE_NAME), "a")
 156 
 157   def log_metric(self, name, value, unit=None, global_step=None, extras=None):
 158     """Log the benchmark metric information to local file.
 159 
 160     Currently the logging is done in a synchronized way. This should be updated
 161     to log asynchronously.
 162 
 163     Args:
 164       name: string, the name of the metric to log.
 165       value: number, the value of the metric. The value will not be logged if it
 166         is not a number type.
 167       unit: string, the unit of the metric, E.g "image per second".
 168       global_step: int, the global_step when the metric is logged.
 169       extras: map of string:string, the extra information about the metric.
 170     """
 171     metric = _process_metric_to_json(name, value, unit, global_step, extras)
 172     if metric:
 173       try:
 174         json.dump(metric, self._metric_file_handler)
 175         self._metric_file_handler.write("\n")
 176         self._metric_file_handler.flush()
 177       except (TypeError, ValueError) as e:
 178         logging.warning(
 179             "Failed to dump metric to log file: name %s, value %s, error %s",
 180             name, value, e)
 181 
 182   def log_run_info(self, model_name, dataset_name, run_params, test_id=None):
 183     """Collect most of the TF runtime information for the local env.
 184 
 185     The schema of the run info follows official/benchmark/datastore/schema.
 186 
 187     Args:
 188       model_name: string, the name of the model.
 189       dataset_name: string, the name of dataset for training and evaluation.
 190       run_params: dict, the dictionary of parameters for the run, it could
 191         include hyperparameters or other params that are important for the run.
 192       test_id: string, the unique name of the test run by the combination of key
 193         parameters, eg batch size, num of GPU. It is hardware independent.
 194     """
 195     run_info = _gather_run_info(model_name, dataset_name, run_params, test_id)
 196 
 197     with tf.io.gfile.GFile(os.path.join(
 198         self._logging_dir, BENCHMARK_RUN_LOG_FILE_NAME), "w") as f:
 199       try:
 200         json.dump(run_info, f)
 201         f.write("\n")
 202       except (TypeError, ValueError) as e:
 203         logging.warning("Failed to dump benchmark run info to log file: %s", e)
 204 
 205   def on_finish(self, status):
 206     self._metric_file_handler.flush()
 207     self._metric_file_handler.close()

 210 def _gather_run_info(model_name, dataset_name, run_params, test_id):
 211   """Collect the benchmark run information for the local environment."""
 212   run_info = {
 213       "model_name": model_name,
 214       "dataset": {"name": dataset_name},
 215       "machine_config": {},
 216       "test_id": test_id,
 217       "run_date": datetime.datetime.utcnow().strftime(
 218           _DATE_TIME_FORMAT_PATTERN)}
 219   _collect_tensorflow_info(run_info)
 220   _collect_tensorflow_environment_variables(run_info)
 221   _collect_run_params(run_info, run_params)
 222   _collect_memory_info(run_info)
 223   _collect_test_environment(run_info)
 224   return run_info

 227 def _process_metric_to_json(
 228     name, value, unit=None, global_step=None, extras=None):
 229   """Validate the metric data and generate JSON for insert."""
 230   if not isinstance(value, numbers.Number):
 231     logging.warning("Metric value to log should be a number. Got %s",
 232                     type(value))
 233     return None
 234 
 235   extras = _convert_to_json_dict(extras)
 236   return {
 237       "name": name,
 238       "value": float(value),
 239       "unit": unit,
 240       "global_step": global_step,
 241       "timestamp": datetime.datetime.utcnow().strftime(
 242           _DATE_TIME_FORMAT_PATTERN),
 243       "extras": extras}

 247   run_info["tensorflow_version"] = {
 248       "version": tf.version.VERSION, "git_hash": tf.version.GIT_VERSION}

 251 def _collect_run_params(run_info, run_params):
 252   """Log the parameter information for the benchmark run."""
 253   def process_param(name, value):
 254     type_check = {
 255         str: {"name": name, "string_value": value},
 256         int: {"name": name, "long_value": value},
 257         bool: {"name": name, "bool_value": str(value)},
 258         float: {"name": name, "float_value": value},
 259     }
 260     return type_check.get(type(value),
 261                           {"name": name, "string_value": str(value)})
 262   if run_params:
 263     run_info["run_parameters"] = [
 264         process_param(k, v) for k, v in sorted(run_params.items())]
 265

 267 def _collect_tensorflow_environment_variables(run_info):
 268   run_info["tensorflow_environment_variables"] = [
 269       {"name": k, "value": v}
 270       for k, v in sorted(os.environ.items()) if k.startswith("TF_")]

 273 def _collect_memory_info(run_info):
 274   try:
 275     # Note: psutil is not installed in the TensorFlow OSS tree.
 276     # It is installable via pip.
 277     import psutil   # pylint: disable=g-import-not-at-top
 278     vmem = psutil.virtual_memory()
 279     run_info["machine_config"]["memory_total"] = vmem.total
 280     run_info["machine_config"]["memory_available"] = vmem.available
 281   except ImportError:
 282     logging.warn("'psutil' not imported. Memory info will not be logged.")

 292 def _parse_gpu_model(physical_device_desc):
 293   # Assume all the GPU connected are same model
 294   for kv in physical_device_desc.split(","):
 295     k, _, v = kv.partition(":")
 296     if k.strip() == "name":
 297       return v.strip()
 298   return None

 301 def _convert_to_json_dict(input_dict):
 302   if input_dict:
 303     return [{"name": k, "value": v} for k, v in sorted(input_dict.items())]
 304   else:
 305     return []
