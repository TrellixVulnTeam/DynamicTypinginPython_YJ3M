  16 """Wrapper for the mlperf logging utils.
  17 
  18 MLPerf compliance logging is only desired under a limited set of circumstances.
  19 This module is intended to keep users from needing to consider logging (or
  20 install the module) unless they are performing mlperf runs.
  21 """

  24 from __future__ import division

  38 _MIN_VERSION = (0, 0, 10)
  39 _STACK_OFFSET = 2
  40 
  41 SUDO = "sudo" if os.geteuid() else ""
  42 
  43 # This indirection is used in docker.
  44 DROP_CACHE_LOC = os.getenv("DROP_CACHE_LOC", "/proc/sys/vm/drop_caches")
  45 
  46 _NCF_PREFIX = "NCF_RAW_"
  47 
  48 # TODO(robieta): move line parsing to mlperf util
  49 _PREFIX = r"(?:{})?:::MLPv([0-9]+).([0-9]+).([0-9]+)".format(_NCF_PREFIX)
  50 _BENCHMARK = r"([a-zA-Z0-9_]+)"
  51 _TIMESTAMP = r"([0-9]+\.[0-9]+)"
  52 _CALLSITE = r"\((.+):([0-9]+)\)"
  53 _TAG = r"([a-zA-Z0-9_]+)"
  54 _VALUE = r"(.*)"
  55 
  56 ParsedLine = namedtuple("ParsedLine", ["version", "benchmark", "timestamp",
  57                                        "callsite", "tag", "value"])
  58 
  59 LINE_PATTERN = re.compile(
  60     "^{prefix} {benchmark} {timestamp} {callsite} {tag}(: |$){value}?$".format(
  61         prefix=_PREFIX, benchmark=_BENCHMARK, timestamp=_TIMESTAMP,
  62         callsite=_CALLSITE, tag=_TAG, value=_VALUE))
  63 
  64 
  65 def parse_line(line): # type: (str) -> typing.Optional[ParsedLine]
  66   match = LINE_PATTERN.match(line.strip())
  67   if not match:
  68     return
  69 
  70   major, minor, micro, benchmark, timestamp = match.groups()[:5]
  71   call_file, call_line, tag, _, value = match.groups()[5:]
  72 
  73   return ParsedLine(version=(int(major), int(minor), int(micro)),
  74                     benchmark=benchmark, timestamp=timestamp,
  75                     callsite=(call_file, call_line), tag=tag, value=value)
  76 
  77 
  78 def unparse_line(parsed_line): # type: (ParsedLine) -> str
  79   version_str = "{}.{}.{}".format(*parsed_line.version)
  80   callsite_str = "({}:{})".format(*parsed_line.callsite)
  81   value_str = ": {}".format(parsed_line.value) if parsed_line.value else ""
  82   return ":::MLPv{} {} {} {} {} {}".format(
  83       version_str, parsed_line.benchmark, parsed_line.timestamp, callsite_str,
  84       parsed_line.tag, value_str)

  87 def get_mlperf_log():
  88   """Shielded import of mlperf_log module."""
  89   try:
  90     import mlperf_compliance
  91 
  92     def test_mlperf_log_pip_version():
  93       """Check that mlperf_compliance is up to date."""
  94       import pkg_resources
  95       version = pkg_resources.get_distribution("mlperf_compliance")
  96       version = tuple(int(i) for i in version.version.split("."))
  97       if version < _MIN_VERSION:
  98         logging.warning("mlperf_compliance is version {}, must be >= {}".format(
  99             ".".join([str(i) for i in version]),
 100             ".".join([str(i) for i in _MIN_VERSION])))
 101         raise ImportError
 102       return mlperf_compliance.mlperf_log
 103 
 104     mlperf_log = test_mlperf_log_pip_version()
 105 
 106   except ImportError:
 107     mlperf_log = None
 108 
 109   return mlperf_log

 112 class Logger(object):
 113   """MLPerf logger indirection class.
 114 
 115   This logger only logs for MLPerf runs, and prevents various errors associated
 116   with not having the mlperf_compliance package installed.
 117   """
 118   class Tags(object):
 119     def __init__(self, mlperf_log):
 120       self._enabled = False
 121       self._mlperf_log = mlperf_log
 122 
 123     def __getattr__(self, item):
 124       if self._mlperf_log is None or not self._enabled:
 125         return
 126       return getattr(self._mlperf_log, item)
 127 
 128   def __init__(self):
 129     self._enabled = False
 130     self._mlperf_log = get_mlperf_log()
 131     self.tags = self.Tags(self._mlperf_log)
 132 
 133   def __call__(self, enable=False):
 134     if enable and self._mlperf_log is None:
 135       raise ImportError("MLPerf logging was requested, but mlperf_compliance "
 136                         "module could not be loaded.")
 137 
 138     self._enabled = enable
 139     self.tags._enabled = enable
 140     return self
 141 
 142   def __enter__(self):
 143     pass
 144 
 145   def __exit__(self, exc_type, exc_val, exc_tb):
 146     self._enabled = False
 147     self.tags._enabled = False
 148 
 149   @property
 150   def log_file(self):
 151     if self._mlperf_log is None:
 152       return
 153     return self._mlperf_log.LOG_FILE
 154 
 155   @property
 156   def enabled(self):
 157     return self._enabled
 158 
 159   def ncf_print(self, key, value=None, stack_offset=_STACK_OFFSET,
 160                 deferred=False, extra_print=False, prefix=_NCF_PREFIX):
 161     if self._mlperf_log is None or not self.enabled:
 162       return
 163     self._mlperf_log.ncf_print(key=key, value=value, stack_offset=stack_offset,
 164                                deferred=deferred, extra_print=extra_print,
 165                                prefix=prefix)
 166 
 167   def set_ncf_root(self, path):
 168     if self._mlperf_log is None:
 169       return
 170     self._mlperf_log.ROOT_DIR_NCF = path
 171 
 172 
 173 LOGGER = Logger()
 174 ncf_print, set_ncf_root = LOGGER.ncf_print, LOGGER.set_ncf_root
 175 TAGS = LOGGER.tags
 176

 178 def clear_system_caches():
 179   if not LOGGER.enabled:
 180     return
 181   ret_code = subprocess.call(
 182       ["sync && echo 3 | {} tee {}".format(SUDO, DROP_CACHE_LOC)],
 183       shell=True)
 184 
 185   if ret_code:
 186     raise ValueError("Failed to clear caches")
 187 
 188 
 189 if __name__ == "__main__":
 190   logging.set_verbosity(logging.INFO)
 191   with LOGGER(True):
 192     ncf_print(key=TAGS.RUN_START)
