  44   def test_illegal_args(self):
  45     with self.assertRaisesRegexp(ValueError, "nvalid every_n_iter"):
  46       metric_hook.LoggingMetricHook(tensors=["t"], every_n_iter=0)
  47     with self.assertRaisesRegexp(ValueError, "nvalid every_n_iter"):
  48       metric_hook.LoggingMetricHook(tensors=["t"], every_n_iter=-10)
  49     with self.assertRaisesRegexp(ValueError, "xactly one of"):
  50       metric_hook.LoggingMetricHook(
  51           tensors=["t"], every_n_iter=5, every_n_secs=5)
  52     with self.assertRaisesRegexp(ValueError, "xactly one of"):
  53       metric_hook.LoggingMetricHook(tensors=["t"])
  54     with self.assertRaisesRegexp(ValueError, "metric_logger"):
  55       metric_hook.LoggingMetricHook(tensors=["t"], every_n_iter=5)

  57   def test_print_at_end_only(self):
  58     with tf.Graph().as_default(), tf.compat.v1.Session() as sess:
  59       tf.compat.v1.train.get_or_create_global_step()
  60       t = tf.constant(42.0, name="foo")
  61       train_op = tf.constant(3)
  62       hook = metric_hook.LoggingMetricHook(
  63           tensors=[t.name], at_end=True, metric_logger=self._logger)
  64       hook.begin()
  65       mon_sess = monitored_session._HookedSession(sess, [hook])  # pylint: disable=protected-access
  66       sess.run(tf.compat.v1.global_variables_initializer())
  67 
  68       for _ in range(3):
  69         mon_sess.run(train_op)
  70         self.assertEqual(self._logger.logged_metric, [])
  71 
  72       hook.end(sess)
  73       self.assertEqual(len(self._logger.logged_metric), 1)
  74       metric = self._logger.logged_metric[0]
  75       self.assertRegexpMatches(metric["name"], "foo")
  76       self.assertEqual(metric["value"], 42.0)
  77       self.assertEqual(metric["unit"], None)
  78       self.assertEqual(metric["global_step"], 0)

  90   def test_log_tensors(self):

  92       tf.compat.v1.train.get_or_create_global_step()

  95       train_op = tf.constant(3)
  96       hook = metric_hook.LoggingMetricHook(
  97           tensors=[t1, t2], at_end=True, metric_logger=self._logger)
  98       hook.begin()

 100       sess.run(tf.compat.v1.global_variables_initializer())

 104         self.assertEqual(self._logger.logged_metric, [])
 105

 107       self.assertEqual(len(self._logger.logged_metric), 2)
 108       metric1 = self._logger.logged_metric[0]

 110       self.assertEqual(metric1["value"], 42.0)

 120   def _validate_print_every_n_steps(self, sess, at_end):
 121     t = tf.constant(42.0, name="foo")
 122 
 123     train_op = tf.constant(3)
 124     hook = metric_hook.LoggingMetricHook(
 125         tensors=[t.name], every_n_iter=10, at_end=at_end,
 126         metric_logger=self._logger)
 127     hook.begin()
 128     mon_sess = monitored_session._HookedSession(sess, [hook])  # pylint: disable=protected-access
 129     sess.run(tf.compat.v1.global_variables_initializer())
 130     mon_sess.run(train_op)
 131     self.assertRegexpMatches(str(self._logger.logged_metric), t.name)
 132     for _ in range(3):
 133       self._logger.logged_metric = []
 134       for _ in range(9):
 135         mon_sess.run(train_op)
 136         # assertNotRegexpMatches is not supported by python 3.1 and later
 137         self.assertEqual(str(self._logger.logged_metric).find(t.name), -1)
 138       mon_sess.run(train_op)
 139       self.assertRegexpMatches(str(self._logger.logged_metric), t.name)
 140 
 141     # Add additional run to verify proper reset when called multiple times.
 142     self._logger.logged_metric = []
 143     mon_sess.run(train_op)
 144     # assertNotRegexpMatches is not supported by python 3.1 and later
 145     self.assertEqual(str(self._logger.logged_metric).find(t.name), -1)
 146 
 147     self._logger.logged_metric = []
 148     hook.end(sess)
 149     if at_end:
 150       self.assertRegexpMatches(str(self._logger.logged_metric), t.name)
 151     else:
 152       # assertNotRegexpMatches is not supported by python 3.1 and later
 153       self.assertEqual(str(self._logger.logged_metric).find(t.name), -1)

 169   def _validate_print_every_n_secs(self, sess, at_end):
 170     t = tf.constant(42.0, name="foo")
 171     train_op = tf.constant(3)
 172 
 173     hook = metric_hook.LoggingMetricHook(
 174         tensors=[t.name], every_n_secs=1.0, at_end=at_end,
 175         metric_logger=self._logger)
 176     hook.begin()
 177     mon_sess = monitored_session._HookedSession(sess, [hook])  # pylint: disable=protected-access
 178     sess.run(tf.compat.v1.global_variables_initializer())
 179 
 180     mon_sess.run(train_op)
 181     self.assertRegexpMatches(str(self._logger.logged_metric), t.name)
 182 
 183     # assertNotRegexpMatches is not supported by python 3.1 and later
 184     self._logger.logged_metric = []
 185     mon_sess.run(train_op)
 186     self.assertEqual(str(self._logger.logged_metric).find(t.name), -1)
 187     time.sleep(1.0)
 188 
 189     self._logger.logged_metric = []
 190     mon_sess.run(train_op)
 191     self.assertRegexpMatches(str(self._logger.logged_metric), t.name)
 192 
 193     self._logger.logged_metric = []
 194     hook.end(sess)
 195     if at_end:
 196       self.assertRegexpMatches(str(self._logger.logged_metric), t.name)
 197     else:
 198       # assertNotRegexpMatches is not supported by python 3.1 and later
 199       self.assertEqual(str(self._logger.logged_metric).find(t.name), -1)
