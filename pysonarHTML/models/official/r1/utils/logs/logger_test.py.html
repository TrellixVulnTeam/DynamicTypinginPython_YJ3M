  64 class BaseBenchmarkLoggerTest(tf.test.TestCase):
  65 
  66   def setUp(self):
  67     super(BaseBenchmarkLoggerTest, self).setUp()
  68     self._actual_log = logging.info
  69     self.logged_message = None
  70 
  71     def mock_log(*args, **kwargs):
  72       self.logged_message = args
  73       self._actual_log(*args, **kwargs)
  74 
  75     logging.info = mock_log
  76 
  77   def tearDown(self):
  78     super(BaseBenchmarkLoggerTest, self).tearDown()
  79     logging.info = self._actual_log
  80 
  81   def test_log_metric(self):
  82     log = logger.BaseBenchmarkLogger()
  83     log.log_metric("accuracy", 0.999, global_step=1e4, extras={"name": "value"})
  84 
  85     expected_log_prefix = "Benchmark metric:"
  86     self.assertRegexpMatches(str(self.logged_message), expected_log_prefix)

  89 class BenchmarkFileLoggerTest(tf.test.TestCase):
  90 
  91   def setUp(self):
  92     super(BenchmarkFileLoggerTest, self).setUp()
  93     # Avoid pulling extra env vars from test environment which affects the test
  94     # result, eg. Kokoro test has a TF_PKG env which affect the test case
  95     # test_collect_tensorflow_environment_variables()
  96     self.original_environ = dict(os.environ)
  97     os.environ.clear()
  98 
  99   def tearDown(self):
 100     super(BenchmarkFileLoggerTest, self).tearDown()
 101     tf.io.gfile.rmtree(self.get_temp_dir())
 102     os.environ.clear()
 103     os.environ.update(self.original_environ)

 105   def test_create_logging_dir(self):
 106     non_exist_temp_dir = os.path.join(self.get_temp_dir(), "unknown_dir")
 107     self.assertFalse(tf.io.gfile.isdir(non_exist_temp_dir))
 108 
 109     logger.BenchmarkFileLogger(non_exist_temp_dir)
 110     self.assertTrue(tf.io.gfile.isdir(non_exist_temp_dir))
 111 
 112   def test_log_metric(self):
 113     log_dir = tempfile.mkdtemp(dir=self.get_temp_dir())
 114     log = logger.BenchmarkFileLogger(log_dir)
 115     log.log_metric("accuracy", 0.999, global_step=1e4, extras={"name": "value"})
 116 
 117     metric_log = os.path.join(log_dir, "metric.log")
 118     self.assertTrue(tf.io.gfile.exists(metric_log))
 119     with tf.io.gfile.GFile(metric_log) as f:
 120       metric = json.loads(f.readline())
 121       self.assertEqual(metric["name"], "accuracy")
 122       self.assertEqual(metric["value"], 0.999)
 123       self.assertEqual(metric["unit"], None)
 124       self.assertEqual(metric["global_step"], 1e4)
 125       self.assertEqual(metric["extras"], [{"name": "name", "value": "value"}])

 127   def test_log_multiple_metrics(self):
 128     log_dir = tempfile.mkdtemp(dir=self.get_temp_dir())
 129     log = logger.BenchmarkFileLogger(log_dir)
 130     log.log_metric("accuracy", 0.999, global_step=1e4, extras={"name": "value"})
 131     log.log_metric("loss", 0.02, global_step=1e4)
 132 
 133     metric_log = os.path.join(log_dir, "metric.log")
 134     self.assertTrue(tf.io.gfile.exists(metric_log))
 135     with tf.io.gfile.GFile(metric_log) as f:
 136       accuracy = json.loads(f.readline())
 137       self.assertEqual(accuracy["name"], "accuracy")
 138       self.assertEqual(accuracy["value"], 0.999)
 139       self.assertEqual(accuracy["unit"], None)
 140       self.assertEqual(accuracy["global_step"], 1e4)
 141       self.assertEqual(accuracy["extras"], [{"name": "name", "value": "value"}])
 142 
 143       loss = json.loads(f.readline())
 144       self.assertEqual(loss["name"], "loss")
 145       self.assertEqual(loss["value"], 0.02)
 146       self.assertEqual(loss["unit"], None)
 147       self.assertEqual(loss["global_step"], 1e4)
 148       self.assertEqual(loss["extras"], [])

 172       self.assertEqual(accuracy["value"], 0.9285)

 200   def test_collect_run_params(self):
 201     run_info = {}
 202     run_parameters = {
 203         "batch_size": 32,
 204         "synthetic_data": True,
 205         "train_epochs": 100.00,
 206         "dtype": "fp16",
 207         "resnet_size": 50,
 208         "random_tensor": tf.constant(2.0)
 209     }
 210     logger._collect_run_params(run_info, run_parameters)
 211     self.assertEqual(len(run_info["run_parameters"]), 6)
 212     self.assertEqual(run_info["run_parameters"][0],
 213                      {"name": "batch_size", "long_value": 32})
 214     self.assertEqual(run_info["run_parameters"][1],
 215                      {"name": "dtype", "string_value": "fp16"})
 216     v1_tensor = {"name": "random_tensor", "string_value":
 217                      "Tensor(\"Const:0\", shape=(), dtype=float32)"}
 218     v2_tensor = {"name": "random_tensor", "string_value":
 219                      "tf.Tensor(2.0, shape=(), dtype=float32)"}
 220     self.assertIn(run_info["run_parameters"][2], [v1_tensor, v2_tensor])
 221 
 222 
 223     self.assertEqual(run_info["run_parameters"][3],
 224                      {"name": "resnet_size", "long_value": 50})
 225     self.assertEqual(run_info["run_parameters"][4],
 226                      {"name": "synthetic_data", "bool_value": "True"})
 227     self.assertEqual(run_info["run_parameters"][5],
 228                      {"name": "train_epochs", "float_value": 100.00})

 230   def test_collect_tensorflow_environment_variables(self):
