  57 def construct_estimator(model_dir, params):
  58   """Construct either an Estimator for NCF.
  59 
  60   Args:
  61     model_dir: The model directory for the estimator
  62     params: The params dict for the estimator
  63 
  64   Returns:
  65     An Estimator.
  66   """
  67   distribution = ncf_common.get_v1_distribution_strategy(params)
  68   run_config = tf.estimator.RunConfig(train_distribute=distribution,
  69                                       eval_distribute=distribution)
  70   model_fn = neumf_model.neumf_model_fn
  71   estimator = tf.estimator.Estimator(model_fn=model_fn, model_dir=model_dir,
  72                                      config=run_config, params=params)
  73   return estimator

  76 def log_and_get_hooks(eval_batch_size):
  77   """Convenience function for hook and logger creation."""
  78   # Create hooks that log information about the training and metric values
  79   train_hooks = hooks_helper.get_train_hooks(
  80       FLAGS.hooks,
  81       model_dir=FLAGS.model_dir,
  82       batch_size=FLAGS.batch_size,  # for ExamplesPerSecondHook
  83       tensors_to_log={"cross_entropy": "cross_entropy"}
  84   )
  85   run_params = {
  86       "batch_size": FLAGS.batch_size,
  87       "eval_batch_size": eval_batch_size,
  88       "number_factors": FLAGS.num_factors,
  89       "hr_threshold": FLAGS.hr_threshold,
  90       "train_epochs": FLAGS.train_epochs,
  91   }
  92   benchmark_logger = logger.get_benchmark_logger()
  93   benchmark_logger.log_run_info(
  94       model_name="recommendation",
  95       dataset_name=FLAGS.dataset,
  96       run_params=run_params,
  97       test_id=FLAGS.benchmark_test_id)
  98 
  99   return benchmark_logger, train_hooks

 102 def main(_):
 103   with logger.benchmark_context(FLAGS), \
 104        mlperf_helper.LOGGER(FLAGS.output_ml_perf_compliance_logging):
 105     mlperf_helper.set_ncf_root(os.path.split(os.path.abspath(__file__))[0])
 106     run_ncf(FLAGS)

 109 def run_ncf(_):
 110   """Run NCF training and eval loop."""
 111   params = ncf_common.parse_flags(FLAGS)
 112 
 113   num_users, num_items, num_train_steps, num_eval_steps, producer = (
 114       ncf_common.get_inputs(params))
 115 
 116   params["num_users"], params["num_items"] = num_users, num_items
 117   producer.start()
 118   model_helpers.apply_clean(flags.FLAGS)
 119 
 120   estimator = construct_estimator(model_dir=FLAGS.model_dir, params=params)
 121 
 122   benchmark_logger, train_hooks = log_and_get_hooks(params["eval_batch_size"])
 123   total_training_cycle = FLAGS.train_epochs // FLAGS.epochs_between_evals
 124 
 125   target_reached = False
 126   mlperf_helper.ncf_print(key=mlperf_helper.TAGS.TRAIN_LOOP)
 127   for cycle_index in range(total_training_cycle):
 128     assert FLAGS.epochs_between_evals == 1 or not mlperf_helper.LOGGER.enabled
 129     logging.info("Starting a training cycle: {}/{}".format(
 130         cycle_index + 1, total_training_cycle))
 131 
 132     mlperf_helper.ncf_print(key=mlperf_helper.TAGS.TRAIN_EPOCH,
 133                             value=cycle_index)
 134 
 135     train_input_fn = producer.make_input_fn(is_training=True)
 136     estimator.train(input_fn=train_input_fn, hooks=train_hooks,
 137                     steps=num_train_steps)
 138 
 139     logging.info("Beginning evaluation.")
 140     eval_input_fn = producer.make_input_fn(is_training=False)
 141 
 142     mlperf_helper.ncf_print(key=mlperf_helper.TAGS.EVAL_START,
 143                             value=cycle_index)
 144     eval_results = estimator.evaluate(eval_input_fn, steps=num_eval_steps)
 145     logging.info("Evaluation complete.")
 146 
 147     hr = float(eval_results[rconst.HR_KEY])
 148     ndcg = float(eval_results[rconst.NDCG_KEY])
 149     loss = float(eval_results["loss"])
 150 
 151     mlperf_helper.ncf_print(
 152         key=mlperf_helper.TAGS.EVAL_TARGET,
 153         value={"epoch": cycle_index, "value": FLAGS.hr_threshold})
 154     mlperf_helper.ncf_print(key=mlperf_helper.TAGS.EVAL_ACCURACY,
 155                             value={"epoch": cycle_index, "value": hr})
 156     mlperf_helper.ncf_print(
 157         key=mlperf_helper.TAGS.EVAL_HP_NUM_NEG,
 158         value={"epoch": cycle_index, "value": rconst.NUM_EVAL_NEGATIVES})
 159 
 160     mlperf_helper.ncf_print(key=mlperf_helper.TAGS.EVAL_STOP, value=cycle_index)
 161 
 162     # Benchmark the evaluation results
 163     benchmark_logger.log_evaluation_result(eval_results)
 164     # Log the HR and NDCG results.
 165     logging.info(
 166         "Iteration {}: HR = {:.4f}, NDCG = {:.4f}, Loss = {:.4f}".format(
 167             cycle_index + 1, hr, ndcg, loss))
 168 
 169     # If some evaluation threshold is met
 170     if model_helpers.past_stop_threshold(FLAGS.hr_threshold, hr):
 171       target_reached = True
 172       break
 173 
 174   mlperf_helper.ncf_print(key=mlperf_helper.TAGS.RUN_STOP,
 175                           value={"success": target_reached})
 176   producer.stop_loop()
 177   producer.join()
 178 
 179   # Clear the session explicitly to avoid session delete error
 180   tf.keras.backend.clear_session()
 181   mlperf_helper.ncf_print(key=mlperf_helper.TAGS.RUN_FINAL)
 182 
 183 
 184 if __name__ == "__main__":
 185   logging.set_verbosity(logging.INFO)
 186   ncf_common.define_ncf_flags()
 187   absl_app.run(main)
