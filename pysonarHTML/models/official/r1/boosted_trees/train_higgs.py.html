  15 r"""A script that builds boosted trees over higgs data.
  16 
  17 If you haven't, please run data_download.py beforehand to prepare the data.
  18 
  19 For some more details on this example, please refer to README.md as well.
  20 
  21 Note that the model_dir is cleaned up before starting the training.
  22 
  23 Usage:
  24 $ python train_higgs.py --n_trees=100 --max_depth=6 --learning_rate=0.1 \
  25     --model_dir=/tmp/higgs_model
  26 
  27 Note that BoostedTreesClassifier is available since Tensorflow 1.8.0.
  28 So you need to install recent enough version of Tensorflow to use this example.
  29 
  30 The training data is by default the first million examples out of 11M examples,
  31 and eval data is by default the last million examples.
  32 They are controlled by --train_start, --train_count, --eval_start, --eval_count.
  33 e.g. to train over the first 10 million examples instead of 1 million:
  34 $ python train_higgs.py --n_trees=100 --max_depth=6 --learning_rate=0.1 \
  35     --model_dir=/tmp/higgs_model --train_count=10000000
  36 
  37 Training history and metrics can be inspected using tensorboard.
  38 Set --logdir as the --model_dir set by flag when training

  40 $ tensorboard --logdir=/tmp/higgs_model

  60 def read_higgs_data(data_dir, train_start, train_count, eval_start, eval_count):
  61   """Reads higgs data from csv and returns train and eval data.
  62 
  63   Args:
  64     data_dir: A string, the directory of higgs dataset.
  65     train_start: An integer, the start index of train examples within the data.
  66     train_count: An integer, the number of train examples within the data.
  67     eval_start: An integer, the start index of eval examples within the data.
  68     eval_count: An integer, the number of eval examples within the data.
  69 
  70   Returns:
  71     Numpy array of train data and eval data.
  72   """
  73   npz_filename = os.path.join(data_dir, NPZ_FILE)
  74   try:
  75     # gfile allows numpy to read data from network data sources as well.
  76     with tf.gfile.Open(npz_filename, "rb") as npz_file:
  77       with np.load(npz_file) as npz:
  78         data = npz["data"]
  79   except tf.errors.NotFoundError as e:
  80     raise RuntimeError(
  81         "Error loading data; use data_download.py to prepare the data.\n{}: {}"
  82         .format(type(e).__name__, e))
  83   return (data[train_start:train_start+train_count],
  84           data[eval_start:eval_start+eval_count])

  87 # This showcases how to make input_fn when the input data is available in the
  88 # form of numpy arrays.
  89 def make_inputs_from_np_arrays(features_np, label_np):
  90   """Makes and returns input_fn and feature_columns from numpy arrays.
  91 
  92   The generated input_fn will return tf.data.Dataset of feature dictionary and a
  93   label, and feature_columns will consist of the list of
  94   tf.feature_column.BucketizedColumn.
  95 
  96   Note, for in-memory training, tf.data.Dataset should contain the whole data
  97   as a single tensor. Don't use batch.
  98 
  99   Args:
 100     features_np: A numpy ndarray (shape=[batch_size, num_features]) for
 101         float32 features.
 102     label_np: A numpy ndarray (shape=[batch_size, 1]) for labels.
 103 
 104   Returns:
 105     input_fn: A function returning a Dataset of feature dict and label.
 106     feature_names: A list of feature names.
 107     feature_column: A list of tf.feature_column.BucketizedColumn.
 108   """
 109   num_features = features_np.shape[1]
 110   features_np_list = np.split(features_np, num_features, axis=1)
 111   # 1-based feature names.
 112   feature_names = ["feature_%02d" % (i + 1) for i in range(num_features)]
 113 
 114   # Create source feature_columns and bucketized_columns.
 115   def get_bucket_boundaries(feature):
 116     """Returns bucket boundaries for feature by percentiles."""
 117     return np.unique(np.percentile(feature, range(0, 100))).tolist()
 118   source_columns = [
 119       tf.feature_column.numeric_column(
 120           feature_name, dtype=tf.float32,
 121           # Although higgs data have no missing values, in general, default
 122           # could be set as 0 or some reasonable value for missing values.
 123           default_value=0.0)
 124       for feature_name in feature_names
 125   ]
 126   bucketized_columns = [
 127       tf.feature_column.bucketized_column(
 128           source_columns[i],
 129           boundaries=get_bucket_boundaries(features_np_list[i]))
 130       for i in range(num_features)
 131   ]
 132 
 133   # Make an input_fn that extracts source features.
 134   def input_fn():
 135     """Returns features as a dictionary of numpy arrays, and a label."""
 136     features = {
 137         feature_name: tf.constant(features_np_list[i])
 138         for i, feature_name in enumerate(feature_names)
 139     }
 140     return tf.data.Dataset.zip((tf.data.Dataset.from_tensors(features),
 141                                 tf.data.Dataset.from_tensors(label_np),))
 142 
 143   return input_fn, feature_names, bucketized_columns

 146 def make_eval_inputs_from_np_arrays(features_np, label_np):
 147   """Makes eval input as streaming batches."""
 148   num_features = features_np.shape[1]
 149   features_np_list = np.split(features_np, num_features, axis=1)
 150   # 1-based feature names.
 151   feature_names = ["feature_%02d" % (i + 1) for i in range(num_features)]
 152 
 153   def input_fn():
 154     features = {
 155         feature_name: tf.constant(features_np_list[i])
 156         for i, feature_name in enumerate(feature_names)
 157     }
 158     return tf.data.Dataset.zip((
 159         tf.data.Dataset.from_tensor_slices(features),
 160         tf.data.Dataset.from_tensor_slices(label_np),)).batch(1000)
 161 
 162   return input_fn

 165 def _make_csv_serving_input_receiver_fn(column_names, column_defaults):
 166   """Returns serving_input_receiver_fn for csv.
 167 
 168   The input arguments are relevant to `tf.decode_csv()`.
 169 
 170   Args:
 171     column_names: a list of column names in the order within input csv.
 172     column_defaults: a list of default values with the same size of
 173         column_names. Each entity must be either a list of one scalar, or an
 174         empty list to denote the corresponding column is required.
 175         e.g. [[""], [2.5], []] indicates the third column is required while
 176             the first column must be string and the second must be float/double.
 177 
 178   Returns:
 179     a serving_input_receiver_fn that handles csv for serving.
 180   """
 181   def serving_input_receiver_fn():
 182     csv = tf.placeholder(dtype=tf.string, shape=[None], name="csv")
 183     features = dict(zip(column_names, tf.decode_csv(csv, column_defaults)))
 184     receiver_tensors = {"inputs": csv}
 185     return tf.estimator.export.ServingInputReceiver(features, receiver_tensors)
 186 
 187   return serving_input_receiver_fn

 190 def train_boosted_trees(flags_obj):
 191   """Train boosted_trees estimator on HIGGS data.
 192 
 193   Args:
 194     flags_obj: An object containing parsed flag values.
 195   """
 196   # Clean up the model directory if present.
 197   if tf.gfile.Exists(flags_obj.model_dir):
 198     tf.gfile.DeleteRecursively(flags_obj.model_dir)
 199   tf.logging.info("## Data loading...")
 200   train_data, eval_data = read_higgs_data(
 201       flags_obj.data_dir, flags_obj.train_start, flags_obj.train_count,
 202       flags_obj.eval_start, flags_obj.eval_count)
 203   tf.logging.info("## Data loaded; train: {}{}, eval: {}{}".format(
 204       train_data.dtype, train_data.shape, eval_data.dtype, eval_data.shape))
 205   # Data consists of one label column followed by 28 feature columns.
 206   train_input_fn, feature_names, feature_columns = make_inputs_from_np_arrays(
 207       features_np=train_data[:, 1:], label_np=train_data[:, 0:1])
 208   eval_input_fn = make_eval_inputs_from_np_arrays(
 209       features_np=eval_data[:, 1:], label_np=eval_data[:, 0:1])
 210   tf.logging.info("## Features prepared. Training starts...")
 211 
 212   # Create benchmark logger to log info about the training and metric values
 213   run_params = {
 214       "train_start": flags_obj.train_start,
 215       "train_count": flags_obj.train_count,
 216       "eval_start": flags_obj.eval_start,
 217       "eval_count": flags_obj.eval_count,
 218       "n_trees": flags_obj.n_trees,
 219       "max_depth": flags_obj.max_depth,
 220   }
 221   benchmark_logger = logger.config_benchmark_logger(flags_obj)
 222   benchmark_logger.log_run_info(
 223       model_name="boosted_trees",
 224       dataset_name="higgs",
 225       run_params=run_params,
 226       test_id=flags_obj.benchmark_test_id)
 227 
 228   # Though BoostedTreesClassifier is under tf.estimator, faster in-memory
 229   # training is yet provided as a contrib library.
 230   from tensorflow.contrib import estimator as contrib_estimator  # pylint: disable=g-import-not-at-top
 231   classifier = contrib_estimator.boosted_trees_classifier_train_in_memory(
 232       train_input_fn,
 233       feature_columns,
 234       model_dir=flags_obj.model_dir or None,
 235       n_trees=flags_obj.n_trees,
 236       max_depth=flags_obj.max_depth,
 237       learning_rate=flags_obj.learning_rate)
 238 
 239   # Evaluation.
 240   eval_results = classifier.evaluate(eval_input_fn)
 241   # Benchmark the evaluation results
 242   benchmark_logger.log_evaluation_result(eval_results)
 243 
 244   # Exporting the savedmodel with csv parsing.
 245   if flags_obj.export_dir is not None:
 246     classifier.export_savedmodel(
 247         flags_obj.export_dir,
 248         _make_csv_serving_input_receiver_fn(
 249             column_names=feature_names,
 250             # columns are all floats.
 251             column_defaults=[[0.0]] * len(feature_names)),
 252         strip_default_attrs=True)
 253

 256   train_boosted_trees(flags.FLAGS)

 259 def define_train_higgs_flags():
 260   """Add tree related flags as well as training/eval configuration."""
 261   flags_core.define_base(clean=False, stop_threshold=False, batch_size=False,
 262                          num_gpu=False, export_dir=True)
 263   flags_core.define_benchmark()
 264   flags.adopt_module_key_flags(flags_core)
 265 
 266   flags.DEFINE_integer(
 267       name="train_start", default=0,
 268       help=help_wrap("Start index of train examples within the data."))
 269   flags.DEFINE_integer(
 270       name="train_count", default=1000000,
 271       help=help_wrap("Number of train examples within the data."))
 272   flags.DEFINE_integer(
 273       name="eval_start", default=10000000,
 274       help=help_wrap("Start index of eval examples within the data."))
 275   flags.DEFINE_integer(
 276       name="eval_count", default=1000000,
 277       help=help_wrap("Number of eval examples within the data."))
 278 
 279   flags.DEFINE_integer(
 280       "n_trees", default=100, help=help_wrap("Number of trees to build."))
 281   flags.DEFINE_integer(
 282       "max_depth", default=6, help=help_wrap("Maximum depths of each tree."))
 283   flags.DEFINE_float(
 284       "learning_rate", default=0.1,
 285       help=help_wrap("The learning rate."))
 286 
 287   flags_core.set_defaults(data_dir="/tmp/higgs_data",
 288                           model_dir="/tmp/higgs_model")
 289 
 290 
 291 if __name__ == "__main__":
 292   # Training progress and eval results are shown as logging.INFO; so enables it.
 293   tf.logging.set_verbosity(tf.logging.INFO)
 294   define_train_higgs_flags()
 295   absl_app.run(main)
