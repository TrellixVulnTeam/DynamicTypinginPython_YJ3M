  38 class EstimatorBenchmark(tf.test.Benchmark):
  39   """Base class to hold methods common to test classes in the module.
  40 
  41      Code under test for Estimator models (ResNet50 and 56) report mostly the
  42      same data and require the same FLAG setup.
  43   """
  44 
  45   local_flags = None
  46 
  47   def __init__(self, output_dir=None, default_flags=None, flag_methods=None):
  48     if not output_dir:
  49       output_dir = '/tmp'
  50     self.output_dir = output_dir
  51     self.default_flags = default_flags or {}
  52     self.flag_methods = flag_methods or {}
  53 
  54   def _get_model_dir(self, folder_name):
  55     """Returns directory to store info, e.g. saved model and event log."""
  56     return os.path.join(self.output_dir, folder_name)
  57 
  58   def _setup(self):
  59     """Sets up and resets flags before each test."""
  60     logging.set_verbosity(logging.INFO)
  61     if EstimatorBenchmark.local_flags is None:
  62       for flag_method in self.flag_methods:
  63         flag_method()
  64       # Loads flags to get defaults to then override. List cannot be empty.
  65       flags.FLAGS(['foo'])
  66       # Overrides flag values with defaults for the class of tests.
  67       for k, v in self.default_flags.items():
  68         setattr(FLAGS, k, v)
  69       saved_flag_values = flagsaver.save_flag_values()
  70       EstimatorBenchmark.local_flags = saved_flag_values
  71     else:
  72       flagsaver.restore_flag_values(EstimatorBenchmark.local_flags)
  73 
  74   def _report_benchmark(self,
  75                         stats,
  76                         wall_time_sec,
  77                         top_1_max=None,
  78                         top_1_min=None):
  79     """Report benchmark results by writing to local protobuf file.
  80 
  81     Args:
  82       stats: dict returned from estimator models with known entries.
  83       wall_time_sec: the during of the benchmark execution in seconds
  84       top_1_max: highest passing level for top_1 accuracy.
  85       top_1_min: lowest passing level for top_1 accuracy.
  86     """
  87 
  88     examples_per_sec_hook = None
  89     for hook in stats['train_hooks']:
  90       if isinstance(hook, hooks.ExamplesPerSecondHook):
  91         examples_per_sec_hook = hook
  92         break
  93 
  94     eval_results = stats['eval_results']
  95     metrics = []
  96     if 'accuracy' in eval_results:
  97       metrics.append({'name': 'accuracy_top_1',
  98                       'value': float(eval_results['accuracy']),
  99                       'min_value': top_1_min,
 100                       'max_value': top_1_max})
 101     if 'accuracy_top_5' in eval_results:
 102       metrics.append({'name': 'accuracy_top_5',
 103                       'value': float(eval_results['accuracy_top_5'])})
 104 
 105     if examples_per_sec_hook:
 106       exp_per_second_list = examples_per_sec_hook.current_examples_per_sec_list
 107       # ExamplesPerSecondHook skips the first 10 steps.
 108       exp_per_sec = sum(exp_per_second_list) / (len(exp_per_second_list))
 109       metrics.append({'name': 'exp_per_second',
 110                       'value': exp_per_sec})
 111     flags_str = flags_core.get_nondefault_flags_as_str()
 112     self.report_benchmark(
 113         iters=eval_results.get('global_step', None),
 114         wall_time=wall_time_sec,
 115         metrics=metrics,
 116         extras={'flags': flags_str})
