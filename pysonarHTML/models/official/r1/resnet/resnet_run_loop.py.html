  48 def process_record_dataset(dataset,
  49                            is_training,
  50                            batch_size,
  51                            shuffle_buffer,
  52                            parse_record_fn,
  53                            num_epochs=1,
  54                            dtype=tf.float32,
  55                            datasets_num_private_threads=None,
  56                            drop_remainder=False,
  57                            tf_data_experimental_slack=False):
  58   """Given a Dataset with raw records, return an iterator over the records.
  59 
  60   Args:
  61     dataset: A Dataset representing raw records
  62     is_training: A boolean denoting whether the input is for training.
  63     batch_size: The number of samples per batch.
  64     shuffle_buffer: The buffer size to use when shuffling records. A larger
  65       value results in better randomness, but smaller values reduce startup
  66       time and use less memory.
  67     parse_record_fn: A function that takes a raw record and returns the
  68       corresponding (image, label) pair.
  69     num_epochs: The number of epochs to repeat the dataset.
  70     dtype: Data type to use for images/features.
  71     datasets_num_private_threads: Number of threads for a private
  72       threadpool created for all datasets computation.
  73     drop_remainder: A boolean indicates whether to drop the remainder of the
  74       batches. If True, the batch dimension will be static.
  75     tf_data_experimental_slack: Whether to enable tf.data's
  76       `experimental_slack` option.
  77 
  78   Returns:
  79     Dataset of (image, label) pairs ready for iteration.
  80   """
  81   # Defines a specific size thread pool for tf.data operations.
  82   if datasets_num_private_threads:
  83     options = tf.data.Options()
  84     options.experimental_threading.private_threadpool_size = (
  85         datasets_num_private_threads)
  86     dataset = dataset.with_options(options)
  87     logging.info('datasets_num_private_threads: %s',
  88                  datasets_num_private_threads)
  89 
  90   # Disable intra-op parallelism to optimize for throughput instead of latency.
  91   options = tf.data.Options()
  92   options.experimental_threading.max_intra_op_parallelism = 1
  93   dataset = dataset.with_options(options)
  94 
  95   # Prefetches a batch at a time to smooth out the time taken to load input
  96   # files for shuffling and processing.
  97   dataset = dataset.prefetch(buffer_size=batch_size)
  98   if is_training:
  99     # Shuffles records before repeating to respect epoch boundaries.
 100     dataset = dataset.shuffle(buffer_size=shuffle_buffer)
 101 
 102   # Repeats the dataset for the number of epochs to train.
 103   dataset = dataset.repeat(num_epochs)
 104 
 105   # Parses the raw records into images and labels.
 106   dataset = dataset.map(
 107       lambda value: parse_record_fn(value, is_training, dtype),
 108       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 109   dataset = dataset.batch(batch_size, drop_remainder=drop_remainder)
 110 
 111   # Operations between the final prefetch and the get_next call to the iterator
 112   # will happen synchronously during run time. We prefetch here again to
 113   # background all of the above processing work and keep it out of the
 114   # critical training path. Setting buffer_size to tf.data.experimental.AUTOTUNE
 115   # allows DistributionStrategies to adjust how many batches to fetch based
 116   # on how many devices are present.
 117   dataset = dataset.prefetch(buffer_size=tf.data.experimental.AUTOTUNE)
 118 
 119   if tf_data_experimental_slack:
 120     options = tf.data.Options()
 121     options.experimental_slack = True
 122     dataset = dataset.with_options(options)
 123 
 124   return dataset

 127 def get_synth_input_fn(height, width, num_channels, num_classes,
 128                        dtype=tf.float32):
 129   """Returns an input function that returns a dataset with random data.
 130 
 131   This input_fn returns a data set that iterates over a set of random data and
 132   bypasses all preprocessing, e.g. jpeg decode and copy. The host to device
 133   copy is still included. This used to find the upper throughput bound when
 134   tunning the full input pipeline.
 135 
 136   Args:
 137     height: Integer height that will be used to create a fake image tensor.
 138     width: Integer width that will be used to create a fake image tensor.
 139     num_channels: Integer depth that will be used to create a fake image tensor.
 140     num_classes: Number of classes that should be represented in the fake labels
 141       tensor
 142     dtype: Data type for features/images.
 143 
 144   Returns:
 145     An input_fn that can be used in place of a real one to return a dataset
 146     that can be used for iteration.
 147   """
 148   # pylint: disable=unused-argument
 149   def input_fn(is_training, data_dir, batch_size, *args, **kwargs):
 150     """Returns dataset filled with random data."""
 151     # Synthetic input should be within [0, 255].
 152     inputs = tf.random.truncated_normal(
 153         [batch_size] + [height, width, num_channels],
 154         dtype=dtype,
 155         mean=127,
 156         stddev=60,
 157         name='synthetic_inputs')
 158 
 159     labels = tf.random.uniform(
 160         [batch_size],
 161         minval=0,
 162         maxval=num_classes - 1,
 163         dtype=tf.int32,
 164         name='synthetic_labels')
 165     data = tf.data.Dataset.from_tensors((inputs, labels)).repeat()
 166     data = data.prefetch(buffer_size=tf.data.experimental.AUTOTUNE)
 167     return data
 168 
 169   return input_fn

 172 def image_bytes_serving_input_fn(image_shape, dtype=tf.float32):
 173   """Serving input fn for raw jpeg images."""
 174 
 175   def _preprocess_image(image_bytes):
 176     """Preprocess a single raw image."""
 177     # Bounding box around the whole image.
 178     bbox = tf.constant([0.0, 0.0, 1.0, 1.0], dtype=dtype, shape=[1, 1, 4])
 179     height, width, num_channels = image_shape
 180     image = imagenet_preprocessing.preprocess_image(
 181         image_bytes, bbox, height, width, num_channels, is_training=False)
 182     return image
 183 
 184   image_bytes_list = tf.compat.v1.placeholder(
 185       shape=[None], dtype=tf.string, name='input_tensor')
 186   images = tf.map_fn(
 187       _preprocess_image, image_bytes_list, back_prop=False, dtype=dtype)
 188   return tf.estimator.export.TensorServingInputReceiver(
 189       images, {'image_bytes': image_bytes_list})

 192 def override_flags_and_set_envars_for_gpu_thread_pool(flags_obj):
 193   """Override flags and set env_vars for performance.
 194 
 195   These settings exist to test the difference between using stock settings
 196   and manual tuning. It also shows some of the ENV_VARS that can be tweaked to
 197   squeeze a few extra examples per second.  These settings are defaulted to the
 198   current platform of interest, which changes over time.
 199 
 200   On systems with small numbers of cpu cores, e.g. under 8 logical cores,
 201   setting up a gpu thread pool with `tf_gpu_thread_mode=gpu_private` may perform
 202   poorly.
 203 
 204   Args:
 205     flags_obj: Current flags, which will be adjusted possibly overriding
 206     what has been set by the user on the command-line.
 207   """
 208   cpu_count = multiprocessing.cpu_count()
 209   logging.info('Logical CPU cores: %s', cpu_count)
 210 
 211   # Sets up thread pool for each GPU for op scheduling.
 212   per_gpu_thread_count = 1
 213   total_gpu_thread_count = per_gpu_thread_count * flags_obj.num_gpus
 214   os.environ['TF_GPU_THREAD_MODE'] = flags_obj.tf_gpu_thread_mode
 215   os.environ['TF_GPU_THREAD_COUNT'] = str(per_gpu_thread_count)
 216   logging.info('TF_GPU_THREAD_COUNT: %s', os.environ['TF_GPU_THREAD_COUNT'])
 217   logging.info('TF_GPU_THREAD_MODE: %s', os.environ['TF_GPU_THREAD_MODE'])
 218 
 219   # Reduces general thread pool by number of threads used for GPU pool.
 220   main_thread_count = cpu_count - total_gpu_thread_count
 221   flags_obj.inter_op_parallelism_threads = main_thread_count
 222 
 223   # Sets thread count for tf.data. Logical cores minus threads assign to the
 224   # private GPU pool along with 2 thread per GPU for event monitoring and
 225   # sending / receiving tensors.
 226   num_monitoring_threads = 2 * flags_obj.num_gpus
 227   flags_obj.datasets_num_private_threads = (cpu_count - total_gpu_thread_count
 228                                             - num_monitoring_threads)

 234 def learning_rate_with_decay(
 235     batch_size, batch_denom, num_images, boundary_epochs, decay_rates,
 236     base_lr=0.1, warmup=False):
 237   """Get a learning rate that decays step-wise as training progresses.
 238 
 239   Args:
 240     batch_size: the number of examples processed in each training batch.
 241     batch_denom: this value will be used to scale the base learning rate.
 242       `0.1 * batch size` is divided by this number, such that when
 243       batch_denom == batch_size, the initial learning rate will be 0.1.
 244     num_images: total number of images that will be used for training.
 245     boundary_epochs: list of ints representing the epochs at which we
 246       decay the learning rate.
 247     decay_rates: list of floats representing the decay rates to be used
 248       for scaling the learning rate. It should have one more element
 249       than `boundary_epochs`, and all elements should have the same type.
 250     base_lr: Initial learning rate scaled based on batch_denom.
 251     warmup: Run a 5 epoch warmup to the initial lr.
 252   Returns:
 253     Returns a function that takes a single argument - the number of batches
 254     trained so far (global_step)- and returns the learning rate to be used
 255     for training the next batch.
 256   """
 257   initial_learning_rate = base_lr * batch_size / batch_denom
 258   batches_per_epoch = num_images / batch_size
 259 
 260   # Reduce the learning rate at certain epochs.
 261   # CIFAR-10: divide by 10 at epoch 100, 150, and 200
 262   # ImageNet: divide by 10 at epoch 30, 60, 80, and 90
 263   boundaries = [int(batches_per_epoch * epoch) for epoch in boundary_epochs]
 264   vals = [initial_learning_rate * decay for decay in decay_rates]
 265 
 266   def learning_rate_fn(global_step):
 267     """Builds scaled learning rate function with 5 epoch warm up."""
 268     lr = tf.compat.v1.train.piecewise_constant(global_step, boundaries, vals)
 269     if warmup:
 270       warmup_steps = int(batches_per_epoch * 5)
 271       warmup_lr = (
 272           initial_learning_rate * tf.cast(global_step, tf.float32) / tf.cast(
 273               warmup_steps, tf.float32))
 274       return tf.cond(pred=global_step < warmup_steps,
 275                      true_fn=lambda: warmup_lr,
 276                      false_fn=lambda: lr)
 277     return lr
 278 
 279   def poly_rate_fn(global_step):
 280     """Handles linear scaling rule, gradual warmup, and LR decay.
 281 
 282     The learning rate starts at 0, then it increases linearly per step.  After
 283     FLAGS.poly_warmup_epochs, we reach the base learning rate (scaled to account
 284     for batch size). The learning rate is then decayed using a polynomial rate
 285     decay schedule with power 2.0.
 286 
 287     Args:
 288       global_step: the current global_step
 289 
 290     Returns:
 291       returns the current learning rate
 292     """
 293 
 294     # Learning rate schedule for LARS polynomial schedule
 295     if flags.FLAGS.batch_size < 8192:
 296       plr = 5.0
 297       w_epochs = 5
 298     elif flags.FLAGS.batch_size < 16384:
 299       plr = 10.0
 300       w_epochs = 5
 301     elif flags.FLAGS.batch_size < 32768:
 302       plr = 25.0
 303       w_epochs = 5
 304     else:
 305       plr = 32.0
 306       w_epochs = 14
 307 
 308     w_steps = int(w_epochs * batches_per_epoch)
 309     wrate = (plr * tf.cast(global_step, tf.float32) / tf.cast(
 310         w_steps, tf.float32))
 311 
 312     # TODO(pkanwar): use a flag to help calc num_epochs.
 313     num_epochs = 90
 314     train_steps = batches_per_epoch * num_epochs
 315 
 316     min_step = tf.constant(1, dtype=tf.int64)
 317     decay_steps = tf.maximum(min_step, tf.subtract(global_step, w_steps))
 318     poly_rate = tf.train.polynomial_decay(
 319         plr,
 320         decay_steps,
 321         train_steps - w_steps + 1,
 322         power=2.0)
 323     return tf.where(global_step <= w_steps, wrate, poly_rate)
 324 
 325   # For LARS we have a new learning rate schedule
 326   if flags.FLAGS.enable_lars:
 327     return poly_rate_fn
 328 
 329   return learning_rate_fn

 332 def per_replica_batch_size(batch_size, num_gpus):
 333   """For multi-gpu, batch-size must be a multiple of the number of GPUs.
 334 
 335 
 336   Note that distribution strategy handles this automatically when used with
 337   Keras. For using with Estimator, we need to get per GPU batch.
 338 
 339   Args:
 340     batch_size: Global batch size to be divided among devices. This should be
 341       equal to num_gpus times the single-GPU batch_size for multi-gpu training.
 342     num_gpus: How many GPUs are used with DistributionStrategies.
 343 
 344   Returns:
 345     Batch size per device.
 346 
 347   Raises:
 348     ValueError: if batch_size is not divisible by number of devices
 349   """
 350   if num_gpus <= 1:
 351     return batch_size
 352 
 353   remainder = batch_size % num_gpus
 354   if remainder:
 355     err = ('When running with multiple GPUs, batch size '
 356            'must be a multiple of the number of available GPUs. Found {} '
 357            'GPUs with a batch size of {}; try --batch_size={} instead.'
 358           ).format(num_gpus, batch_size, batch_size - remainder)
 359     raise ValueError(err)
 360   return int(batch_size / num_gpus)

 363 def resnet_model_fn(features, labels, mode, model_class,
 364                     resnet_size, weight_decay, learning_rate_fn, momentum,
 365                     data_format, resnet_version, loss_scale,
 366                     loss_filter_fn=None, dtype=resnet_model.DEFAULT_DTYPE,
 367                     fine_tune=False, label_smoothing=0.0):
 368   """Shared functionality for different resnet model_fns.
 369 
 370   Initializes the ResnetModel representing the model layers
 371   and uses that model to build the necessary EstimatorSpecs for
 372   the `mode` in question. For training, this means building losses,
 373   the optimizer, and the train op that get passed into the EstimatorSpec.
 374   For evaluation and prediction, the EstimatorSpec is returned without
 375   a train op, but with the necessary parameters for the given mode.
 376 
 377   Args:
 378     features: tensor representing input images
 379     labels: tensor representing class labels for all input images
 380     mode: current estimator mode; should be one of
 381       `tf.estimator.ModeKeys.TRAIN`, `EVALUATE`, `PREDICT`
 382     model_class: a class representing a TensorFlow model that has a __call__
 383       function. We assume here that this is a subclass of ResnetModel.
 384     resnet_size: A single integer for the size of the ResNet model.
 385     weight_decay: weight decay loss rate used to regularize learned variables.
 386     learning_rate_fn: function that returns the current learning rate given
 387       the current global_step
 388     momentum: momentum term used for optimization
 389     data_format: Input format ('channels_last', 'channels_first', or None).
 390       If set to None, the format is dependent on whether a GPU is available.
 391     resnet_version: Integer representing which version of the ResNet network to
 392       use. See README for details. Valid values: [1, 2]
 393     loss_scale: The factor to scale the loss for numerical stability. A detailed
 394       summary is present in the arg parser help text.
 395     loss_filter_fn: function that takes a string variable name and returns
 396       True if the var should be included in loss calculation, and False
 397       otherwise. If None, batch_normalization variables will be excluded
 398       from the loss.
 399     dtype: the TensorFlow dtype to use for calculations.
 400     fine_tune: If True only train the dense layers(final layers).
 401     label_smoothing: If greater than 0 then smooth the labels.
 402 
 403   Returns:
 404     EstimatorSpec parameterized according to the input params and the
 405     current mode.
 406   """
 407 
 408   # Generate a summary node for the images
 409   tf.compat.v1.summary.image('images', features, max_outputs=6)
 410   # Checks that features/images have same data type being used for calculations.
 411   assert features.dtype == dtype
 412 
 413   model = model_class(resnet_size, data_format, resnet_version=resnet_version,
 414                       dtype=dtype)
 415 
 416   logits = model(features, mode == tf.estimator.ModeKeys.TRAIN)
 417 
 418   # This acts as a no-op if the logits are already in fp32 (provided logits are
 419   # not a SparseTensor). If dtype is is low precision, logits must be cast to
 420   # fp32 for numerical stability.
 421   logits = tf.cast(logits, tf.float32)
 422 
 423   predictions = {
 424       'classes': tf.argmax(input=logits, axis=1),
 425       'probabilities': tf.nn.softmax(logits, name='softmax_tensor')
 426   }
 427 
 428   if mode == tf.estimator.ModeKeys.PREDICT:
 429     # Return the predictions and the specification for serving a SavedModel
 430     return tf.estimator.EstimatorSpec(
 431         mode=mode,
 432         predictions=predictions,
 433         export_outputs={
 434             'predict': tf.estimator.export.PredictOutput(predictions)
 435         })
 436 
 437   # Calculate loss, which includes softmax cross entropy and L2 regularization.
 438   if label_smoothing != 0.0:
 439     one_hot_labels = tf.one_hot(labels, 1001)
 440     cross_entropy = tf.losses.softmax_cross_entropy(
 441         logits=logits, onehot_labels=one_hot_labels,
 442         label_smoothing=label_smoothing)
 443   else:
 444     cross_entropy = tf.compat.v1.losses.sparse_softmax_cross_entropy(
 445         logits=logits, labels=labels)
 446 
 447   # Create a tensor named cross_entropy for logging purposes.
 448   tf.identity(cross_entropy, name='cross_entropy')
 449   tf.compat.v1.summary.scalar('cross_entropy', cross_entropy)
 450 
 451   # If no loss_filter_fn is passed, assume we want the default behavior,
 452   # which is that batch_normalization variables are excluded from loss.
 453   def exclude_batch_norm(name):
 454     return 'batch_normalization' not in name
 455   loss_filter_fn = loss_filter_fn or exclude_batch_norm
 456 
 457   # Add weight decay to the loss.
 458   l2_loss = weight_decay * tf.add_n(
 459       # loss is computed using fp32 for numerical stability.
 460       [
 461           tf.nn.l2_loss(tf.cast(v, tf.float32))
 462           for v in tf.compat.v1.trainable_variables()
 463           if loss_filter_fn(v.name)
 464       ])
 465   tf.compat.v1.summary.scalar('l2_loss', l2_loss)
 466   loss = cross_entropy + l2_loss
 467 
 468   if mode == tf.estimator.ModeKeys.TRAIN:
 469     global_step = tf.compat.v1.train.get_or_create_global_step()
 470 
 471     learning_rate = learning_rate_fn(global_step)
 472 
 473     # Create a tensor named learning_rate for logging purposes
 474     tf.identity(learning_rate, name='learning_rate')
 475     tf.compat.v1.summary.scalar('learning_rate', learning_rate)
 476 
 477     if flags.FLAGS.enable_lars:
 478       from tensorflow.contrib import opt as contrib_opt  # pylint: disable=g-import-not-at-top
 479       optimizer = contrib_opt.LARSOptimizer(
 480           learning_rate,
 481           momentum=momentum,
 482           weight_decay=weight_decay,
 483           skip_list=['batch_normalization', 'bias'])
 484     else:
 485       optimizer = tf.compat.v1.train.MomentumOptimizer(
 486           learning_rate=learning_rate,
 487           momentum=momentum
 488       )
 489 
 490     fp16_implementation = getattr(flags.FLAGS, 'fp16_implementation', None)
 491     if fp16_implementation == 'graph_rewrite':
 492       optimizer = (
 493           tf.compat.v1.train.experimental.enable_mixed_precision_graph_rewrite(
 494               optimizer, loss_scale=loss_scale))
 495 
 496     def _dense_grad_filter(gvs):
 497       """Only apply gradient updates to the final layer.
 498 
 499       This function is used for fine tuning.
 500 
 501       Args:
 502         gvs: list of tuples with gradients and variable info
 503       Returns:
 504         filtered gradients so that only the dense layer remains
 505       """
 506       return [(g, v) for g, v in gvs if 'dense' in v.name]
 507 
 508     if loss_scale != 1 and fp16_implementation != 'graph_rewrite':
 509       # When computing fp16 gradients, often intermediate tensor values are
 510       # so small, they underflow to 0. To avoid this, we multiply the loss by
 511       # loss_scale to make these tensor values loss_scale times bigger.
 512       scaled_grad_vars = optimizer.compute_gradients(loss * loss_scale)
 513 
 514       if fine_tune:
 515         scaled_grad_vars = _dense_grad_filter(scaled_grad_vars)
 516 
 517       # Once the gradient computation is complete we can scale the gradients
 518       # back to the correct scale before passing them to the optimizer.
 519       unscaled_grad_vars = [(grad / loss_scale, var)
 520                             for grad, var in scaled_grad_vars]
 521       minimize_op = optimizer.apply_gradients(unscaled_grad_vars, global_step)
 522     else:
 523       grad_vars = optimizer.compute_gradients(loss)
 524       if fine_tune:
 525         grad_vars = _dense_grad_filter(grad_vars)
 526       minimize_op = optimizer.apply_gradients(grad_vars, global_step)
 527 
 528     update_ops = tf.compat.v1.get_collection(tf.compat.v1.GraphKeys.UPDATE_OPS)
 529     train_op = tf.group(minimize_op, update_ops)
 530   else:
 531     train_op = None
 532 
 533   accuracy = tf.compat.v1.metrics.accuracy(labels, predictions['classes'])
 534   accuracy_top_5 = tf.compat.v1.metrics.mean(
 535       tf.nn.in_top_k(predictions=logits, targets=labels, k=5, name='top_5_op'))
 536   metrics = {'accuracy': accuracy,
 537              'accuracy_top_5': accuracy_top_5}
 538 
 539   # Create a tensor named train_accuracy for logging purposes
 540   tf.identity(accuracy[1], name='train_accuracy')
 541   tf.identity(accuracy_top_5[1], name='train_accuracy_top_5')
 542   tf.compat.v1.summary.scalar('train_accuracy', accuracy[1])
 543   tf.compat.v1.summary.scalar('train_accuracy_top_5', accuracy_top_5[1])
 544 
 545   return tf.estimator.EstimatorSpec(
 546       mode=mode,
 547       predictions=predictions,
 548       loss=loss,
 549       train_op=train_op,
 550       eval_metric_ops=metrics)

 553 def resnet_main(
 554     flags_obj, model_function, input_function, dataset_name, shape=None):
 555   """Shared main loop for ResNet Models.
 556 
 557   Args:
 558     flags_obj: An object containing parsed flags. See define_resnet_flags()
 559       for details.
 560     model_function: the function that instantiates the Model and builds the
 561       ops for train/eval. This will be passed directly into the estimator.
 562     input_function: the function that processes the dataset and returns a
 563       dataset that the estimator can train on. This will be wrapped with
 564       all the relevant flags for running and passed to estimator.
 565     dataset_name: the name of the dataset for training and evaluation. This is
 566       used for logging purpose.
 567     shape: list of ints representing the shape of the images used for training.
 568       This is only used if flags_obj.export_dir is passed.
 569 
 570   Returns:
 571      Dict of results of the run.  Contains the keys `eval_results` and
 572     `train_hooks`. `eval_results` contains accuracy (top_1) and accuracy_top_5.
 573     `train_hooks` is a list the instances of hooks used during training.
 574   """
 575 
 576   model_helpers.apply_clean(flags.FLAGS)
 577 
 578   # Ensures flag override logic is only executed if explicitly triggered.
 579   if flags_obj.tf_gpu_thread_mode:
 580     override_flags_and_set_envars_for_gpu_thread_pool(flags_obj)
 581 
 582   # Configures cluster spec for distribution strategy.
 583   num_workers = distribution_utils.configure_cluster(flags_obj.worker_hosts,
 584                                                      flags_obj.task_index)
 585 
 586   # Creates session config. allow_soft_placement = True, is required for
 587   # multi-GPU and is not harmful for other modes.
 588   session_config = tf.compat.v1.ConfigProto(
 589       inter_op_parallelism_threads=flags_obj.inter_op_parallelism_threads,
 590       intra_op_parallelism_threads=flags_obj.intra_op_parallelism_threads,
 591       allow_soft_placement=True)
 592 
 593   distribution_strategy = distribution_utils.get_distribution_strategy(
 594       distribution_strategy=flags_obj.distribution_strategy,
 595       num_gpus=flags_core.get_num_gpus(flags_obj),
 596       all_reduce_alg=flags_obj.all_reduce_alg,
 597       num_packs=flags_obj.num_packs)
 598 
 599   # Creates a `RunConfig` that checkpoints every 24 hours which essentially
 600   # results in checkpoints determined only by `epochs_between_evals`.
 601   run_config = tf.estimator.RunConfig(
 602       train_distribute=distribution_strategy,
 603       session_config=session_config,
 604       save_checkpoints_secs=60*60*24,
 605       save_checkpoints_steps=None)
 606 
 607   # Initializes model with all but the dense layer from pretrained ResNet.
 608   if flags_obj.pretrained_model_checkpoint_path is not None:
 609     warm_start_settings = tf.estimator.WarmStartSettings(
 610         flags_obj.pretrained_model_checkpoint_path,
 611         vars_to_warm_start='^(?!.*dense)')
 612   else:
 613     warm_start_settings = None
 614 
 615   classifier = tf.estimator.Estimator(
 616       model_fn=model_function, model_dir=flags_obj.model_dir, config=run_config,
 617       warm_start_from=warm_start_settings, params={
 618           'resnet_size': int(flags_obj.resnet_size),
 619           'data_format': flags_obj.data_format,
 620           'batch_size': flags_obj.batch_size,
 621           'resnet_version': int(flags_obj.resnet_version),
 622           'loss_scale': flags_core.get_loss_scale(flags_obj,
 623                                                   default_for_fp16=128),
 624           'dtype': flags_core.get_tf_dtype(flags_obj),
 625           'fine_tune': flags_obj.fine_tune,
 626           'num_workers': num_workers,
 627       })
 628 
 629   run_params = {
 630       'batch_size': flags_obj.batch_size,
 631       'dtype': flags_core.get_tf_dtype(flags_obj),
 632       'resnet_size': flags_obj.resnet_size,
 633       'resnet_version': flags_obj.resnet_version,
 634       'synthetic_data': flags_obj.use_synthetic_data,
 635       'train_epochs': flags_obj.train_epochs,
 636       'num_workers': num_workers,
 637   }
 638   if flags_obj.use_synthetic_data:
 639     dataset_name = dataset_name + '-synthetic'
 640 
 641   benchmark_logger = logger.get_benchmark_logger()
 642   benchmark_logger.log_run_info('resnet', dataset_name, run_params,
 643                                 test_id=flags_obj.benchmark_test_id)
 644 
 645   train_hooks = hooks_helper.get_train_hooks(
 646       flags_obj.hooks,
 647       model_dir=flags_obj.model_dir,
 648       batch_size=flags_obj.batch_size)
 649 
 650   def input_fn_train(num_epochs, input_context=None):
 651     return input_function(
 652         is_training=True,
 653         data_dir=flags_obj.data_dir,
 654         batch_size=per_replica_batch_size(
 655             flags_obj.batch_size, flags_core.get_num_gpus(flags_obj)),
 656         num_epochs=num_epochs,
 657         dtype=flags_core.get_tf_dtype(flags_obj),
 658         datasets_num_private_threads=flags_obj.datasets_num_private_threads,
 659         input_context=input_context)
 660 
 661   def input_fn_eval():
 662     return input_function(
 663         is_training=False,
 664         data_dir=flags_obj.data_dir,
 665         batch_size=per_replica_batch_size(
 666             flags_obj.batch_size, flags_core.get_num_gpus(flags_obj)),
 667         num_epochs=1,
 668         dtype=flags_core.get_tf_dtype(flags_obj))
 669 
 670   train_epochs = (0 if flags_obj.eval_only or not flags_obj.train_epochs else
 671                   flags_obj.train_epochs)
 672 
 673   use_train_and_evaluate = flags_obj.use_train_and_evaluate or num_workers > 1
 674   if use_train_and_evaluate:
 675     train_spec = tf.estimator.TrainSpec(
 676         input_fn=lambda input_context=None: input_fn_train(
 677             train_epochs, input_context=input_context),
 678         hooks=train_hooks,
 679         max_steps=flags_obj.max_train_steps)
 680     eval_spec = tf.estimator.EvalSpec(input_fn=input_fn_eval)
 681     logging.info('Starting to train and evaluate.')
 682     tf.estimator.train_and_evaluate(classifier, train_spec, eval_spec)
 683     # tf.estimator.train_and_evalute doesn't return anything in multi-worker
 684     # case.
 685     eval_results = {}
 686   else:
 687     if train_epochs == 0:
 688       # If --eval_only is set, perform a single loop with zero train epochs.
 689       schedule, n_loops = [0], 1
 690     else:
 691       # Compute the number of times to loop while training. All but the last
 692       # pass will train for `epochs_between_evals` epochs, while the last will
 693       # train for the number needed to reach `training_epochs`. For instance if
 694       #   train_epochs = 25 and epochs_between_evals = 10
 695       # schedule will be set to [10, 10, 5]. That is to say, the loop will:
 696       #   Train for 10 epochs and then evaluate.
 697       #   Train for another 10 epochs and then evaluate.
 698       #   Train for a final 5 epochs (to reach 25 epochs) and then evaluate.
 699       n_loops = math.ceil(train_epochs / flags_obj.epochs_between_evals)
 700       schedule = [flags_obj.epochs_between_evals for _ in range(int(n_loops))]
 701       schedule[-1] = train_epochs - sum(schedule[:-1])  # over counting.
 702 
 703     for cycle_index, num_train_epochs in enumerate(schedule):
 704       logging.info('Starting cycle: %d/%d', cycle_index, int(n_loops))
 705 
 706       if num_train_epochs:
 707         # Since we are calling classifier.train immediately in each loop, the
 708         # value of num_train_epochs in the lambda function will not be changed
 709         # before it is used. So it is safe to ignore the pylint error here
 710         # pylint: disable=cell-var-from-loop
 711         classifier.train(
 712             input_fn=lambda input_context=None: input_fn_train(
 713                 num_train_epochs, input_context=input_context),
 714             hooks=train_hooks,
 715             max_steps=flags_obj.max_train_steps)
 716 
 717       # flags_obj.max_train_steps is generally associated with testing and
 718       # profiling. As a result it is frequently called with synthetic data,
 719       # which will iterate forever. Passing steps=flags_obj.max_train_steps
 720       # allows the eval (which is generally unimportant in those circumstances)
 721       # to terminate.  Note that eval will run for max_train_steps each loop,
 722       # regardless of the global_step count.
 723       logging.info('Starting to evaluate.')
 724       eval_results = classifier.evaluate(input_fn=input_fn_eval,
 725                                          steps=flags_obj.max_train_steps)
 726 
 727       benchmark_logger.log_evaluation_result(eval_results)
 728 
 729       if model_helpers.past_stop_threshold(
 730           flags_obj.stop_threshold, eval_results['accuracy']):
 731         break
 732 
 733   if flags_obj.export_dir is not None:
 734     # Exports a saved model for the given classifier.
 735     export_dtype = flags_core.get_tf_dtype(flags_obj)
 736     if flags_obj.image_bytes_as_serving_input:
 737       input_receiver_fn = functools.partial(
 738           image_bytes_serving_input_fn, shape, dtype=export_dtype)
 739     else:
 740       input_receiver_fn = export.build_tensor_serving_input_receiver_fn(
 741           shape, batch_size=flags_obj.batch_size, dtype=export_dtype)
 742     classifier.export_savedmodel(flags_obj.export_dir, input_receiver_fn,
 743                                  strip_default_attrs=True)
 744 
 745   stats = {}
 746   stats['eval_results'] = eval_results
 747   stats['train_hooks'] = train_hooks
 748 
 749   return stats

 752 def define_resnet_flags(resnet_size_choices=None, dynamic_loss_scale=False,
 753                         fp16_implementation=False):
 754   """Add flags and validators for ResNet."""
 755   flags_core.define_base(clean=True, train_epochs=True,
 756                          epochs_between_evals=True, stop_threshold=True,
 757                          num_gpu=True, hooks=True, export_dir=True,
 758                          distribution_strategy=True)
 759   flags_core.define_performance(num_parallel_calls=False,
 760                                 inter_op=True,
 761                                 intra_op=True,
 762                                 synthetic_data=True,
 763                                 dtype=True,
 764                                 all_reduce_alg=True,
 765                                 num_packs=True,
 766                                 tf_gpu_thread_mode=True,
 767                                 datasets_num_private_threads=True,
 768                                 dynamic_loss_scale=dynamic_loss_scale,
 769                                 fp16_implementation=fp16_implementation,
 770                                 loss_scale=True,
 771                                 tf_data_experimental_slack=True,
 772                                 max_train_steps=True)
 773   flags_core.define_image()
 774   flags_core.define_benchmark()
 775   flags_core.define_distribution()
 776   flags.adopt_module_key_flags(flags_core)
 777 
 778   flags.DEFINE_enum(
 779       name='resnet_version', short_name='rv', default='1',
 780       enum_values=['1', '2'],
 781       help=flags_core.help_wrap(
 782           'Version of ResNet. (1 or 2) See README.md for details.'))
 783   flags.DEFINE_bool(
 784       name='fine_tune', short_name='ft', default=False,
 785       help=flags_core.help_wrap(
 786           'If True do not train any parameters except for the final layer.'))
 787   flags.DEFINE_string(
 788       name='pretrained_model_checkpoint_path', short_name='pmcp', default=None,
 789       help=flags_core.help_wrap(
 790           'If not None initialize all the network except the final layer with '
 791           'these values'))
 792   flags.DEFINE_boolean(
 793       name='eval_only', default=False,
 794       help=flags_core.help_wrap('Skip training and only perform evaluation on '
 795                                 'the latest checkpoint.'))
 796   flags.DEFINE_boolean(
 797       name='image_bytes_as_serving_input', default=False,
 798       help=flags_core.help_wrap(
 799           'If True exports savedmodel with serving signature that accepts '
 800           'JPEG image bytes instead of a fixed size [HxWxC] tensor that '
 801           'represents the image. The former is easier to use for serving at '
 802           'the expense of image resize/cropping being done as part of model '
 803           'inference. Note, this flag only applies to ImageNet and cannot '
 804           'be used for CIFAR.'))
 805   flags.DEFINE_boolean(
 806       name='use_train_and_evaluate', default=False,
 807       help=flags_core.help_wrap(
 808           'If True, uses `tf.estimator.train_and_evaluate` for the training '
 809           'and evaluation loop, instead of separate calls to `classifier.train '
 810           'and `classifier.evaluate`, which is the default behavior.'))
 811   flags.DEFINE_bool(
 812       name='enable_lars', default=False,
 813       help=flags_core.help_wrap(
 814           'Enable LARS optimizer for large batch training.'))
 815   flags.DEFINE_float(
 816       name='label_smoothing', default=0.0,
 817       help=flags_core.help_wrap(
 818           'Label smoothing parameter used in the softmax_cross_entropy'))
 819   flags.DEFINE_float(
 820       name='weight_decay', default=1e-4,
 821       help=flags_core.help_wrap(
 822           'Weight decay coefficiant for l2 regularization.'))
 823 
 824   choice_kwargs = dict(
 825       name='resnet_size', short_name='rs', default='50',
 826       help=flags_core.help_wrap('The size of the ResNet model to use.'))
 827 
 828   if resnet_size_choices is None:
 829     flags.DEFINE_string(**choice_kwargs)
 830   else:
 831     flags.DEFINE_enum(enum_values=resnet_size_choices, **choice_kwargs)
