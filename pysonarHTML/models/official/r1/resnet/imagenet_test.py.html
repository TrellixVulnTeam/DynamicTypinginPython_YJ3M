  54   def _tensor_shapes_helper(self, resnet_size, resnet_version, dtype, with_gpu):
  55     """Checks the tensor shapes after each phase of the ResNet model."""
  56     def reshape(shape):
  57       """Returns the expected dimensions depending on if a GPU is being used."""
  58 
  59       # If a GPU is used for the test, the shape is returned (already in NCHW
  60       # form). When GPU is not used, the shape is converted to NHWC.
  61       if with_gpu:
  62         return shape
  63       return shape[0], shape[2], shape[3], shape[1]
  64 
  65     graph = tf.Graph()
  66 
  67     with graph.as_default(), self.test_session(
  68         graph=graph, use_gpu=with_gpu, force_gpu=with_gpu):
  69       model = imagenet_main.ImagenetModel(
  70           resnet_size=resnet_size,
  71           data_format='channels_first' if with_gpu else 'channels_last',
  72           resnet_version=resnet_version,
  73           dtype=dtype
  74       )
  75       inputs = tf.random.uniform([1, 224, 224, 3])
  76       output = model(inputs, training=True)
  77 
  78       initial_conv = graph.get_tensor_by_name('resnet_model/initial_conv:0')
  79       max_pool = graph.get_tensor_by_name('resnet_model/initial_max_pool:0')
  80       block_layer1 = graph.get_tensor_by_name('resnet_model/block_layer1:0')
  81       block_layer2 = graph.get_tensor_by_name('resnet_model/block_layer2:0')
  82       block_layer3 = graph.get_tensor_by_name('resnet_model/block_layer3:0')
  83       block_layer4 = graph.get_tensor_by_name('resnet_model/block_layer4:0')
  84       reduce_mean = graph.get_tensor_by_name('resnet_model/final_reduce_mean:0')
  85       dense = graph.get_tensor_by_name('resnet_model/final_dense:0')
  86 
  87       self.assertAllEqual(initial_conv.shape, reshape((1, 64, 112, 112)))
  88       self.assertAllEqual(max_pool.shape, reshape((1, 64, 56, 56)))
  89 
  90       # The number of channels after each block depends on whether we're
  91       # using the building_block or the bottleneck_block.
  92       if resnet_size < 50:
  93         self.assertAllEqual(block_layer1.shape, reshape((1, 64, 56, 56)))
  94         self.assertAllEqual(block_layer2.shape, reshape((1, 128, 28, 28)))
  95         self.assertAllEqual(block_layer3.shape, reshape((1, 256, 14, 14)))
  96         self.assertAllEqual(block_layer4.shape, reshape((1, 512, 7, 7)))
  97         self.assertAllEqual(reduce_mean.shape, reshape((1, 512, 1, 1)))
  98       else:
  99         self.assertAllEqual(block_layer1.shape, reshape((1, 256, 56, 56)))
 100         self.assertAllEqual(block_layer2.shape, reshape((1, 512, 28, 28)))
 101         self.assertAllEqual(block_layer3.shape, reshape((1, 1024, 14, 14)))
 102         self.assertAllEqual(block_layer4.shape, reshape((1, 2048, 7, 7)))
 103         self.assertAllEqual(reduce_mean.shape, reshape((1, 2048, 1, 1)))
 104 
 105       self.assertAllEqual(dense.shape, (1, _LABEL_CLASSES))
 106       self.assertAllEqual(output.shape, (1, _LABEL_CLASSES))

 200   def resnet_model_fn_helper(self, mode, resnet_version, dtype):
 201     """Tests that the EstimatorSpec is given the appropriate arguments."""
 202     tf.compat.v1.train.create_global_step()
 203 
 204     input_fn = imagenet_main.get_synth_input_fn(dtype)
 205     dataset = input_fn(True, '', _BATCH_SIZE)
 206     iterator = tf.compat.v1.data.make_initializable_iterator(dataset)
 207     features, labels = iterator.get_next()
 208     spec = imagenet_main.imagenet_model_fn(
 209         features, labels, mode, {
 210             'dtype': dtype,
 211             'resnet_size': 50,
 212             'data_format': 'channels_last',
 213             'batch_size': _BATCH_SIZE,
 214             'resnet_version': resnet_version,
 215             'loss_scale': 128 if dtype == tf.float16 else 1,
 216             'fine_tune': False,
 217         })
 218 
 219     predictions = spec.predictions
 220     self.assertAllEqual(predictions['probabilities'].shape,
 221                         (_BATCH_SIZE, _LABEL_CLASSES))
 222     self.assertEqual(predictions['probabilities'].dtype, tf.float32)
 223     self.assertAllEqual(predictions['classes'].shape, (_BATCH_SIZE,))
 224     self.assertEqual(predictions['classes'].dtype, tf.int64)
 225 
 226     if mode != tf.estimator.ModeKeys.PREDICT:
 227       loss = spec.loss
 228       self.assertAllEqual(loss.shape, ())
 229       self.assertEqual(loss.dtype, tf.float32)
 230 
 231     if mode == tf.estimator.ModeKeys.EVAL:
 232       eval_metric_ops = spec.eval_metric_ops
 233       self.assertAllEqual(eval_metric_ops['accuracy'][0].shape, ())
 234       self.assertAllEqual(eval_metric_ops['accuracy'][1].shape, ())
 235       self.assertEqual(eval_metric_ops['accuracy'][0].dtype, tf.float32)
 236       self.assertEqual(eval_metric_ops['accuracy'][1].dtype, tf.float32)

 262   def _test_imagenetmodel_shape(self, resnet_version):
 263     batch_size = 135
 264     num_classes = 246
 265 
 266     model = imagenet_main.ImagenetModel(
 267         50, data_format='channels_last', num_classes=num_classes,
 268         resnet_version=resnet_version)
 269 
 270     fake_input = tf.random.uniform([batch_size, 224, 224, 3])
 271     output = model(fake_input, training=True)
 272 
 273     self.assertAllEqual(output.shape, (batch_size, num_classes))
