  52 def get_filenames(is_training, data_dir):
  53   """Return filenames for dataset."""
  54   if is_training:
  55     return [
  56         os.path.join(data_dir, 'train-%05d-of-01024' % i)
  57         for i in range(_NUM_TRAIN_FILES)]
  58   else:
  59     return [
  60         os.path.join(data_dir, 'validation-%05d-of-00128' % i)
  61         for i in range(128)]

  64 def _parse_example_proto(example_serialized):
  65   """Parses an Example proto containing a training example of an image.
  66 
  67   The output of the build_image_data.py image preprocessing script is a dataset
  68   containing serialized Example protocol buffers. Each Example proto contains
  69   the following fields (values are included as examples):
  70 
  71     image/height: 462
  72     image/width: 581
  73     image/colorspace: 'RGB'
  74     image/channels: 3
  75     image/class/label: 615
  76     image/class/synset: 'n03623198'
  77     image/class/text: 'knee pad'
  78     image/object/bbox/xmin: 0.1
  79     image/object/bbox/xmax: 0.9
  80     image/object/bbox/ymin: 0.2
  81     image/object/bbox/ymax: 0.6
  82     image/object/bbox/label: 615
  83     image/format: 'JPEG'
  84     image/filename: 'ILSVRC2012_val_00041207.JPEG'
  85     image/encoded: <JPEG encoded string>
  86 
  87   Args:
  88     example_serialized: scalar Tensor tf.string containing a serialized
  89       Example protocol buffer.
  90 
  91   Returns:
  92     image_buffer: Tensor tf.string containing the contents of a JPEG file.
  93     label: Tensor tf.int32 containing the label.
  94     bbox: 3-D float Tensor of bounding boxes arranged [1, num_boxes, coords]
  95       where each coordinate is [0, 1) and the coordinates are arranged as
  96       [ymin, xmin, ymax, xmax].
  97   """
  98   # Dense features in Example proto.
  99   feature_map = {
 100       'image/encoded': tf.io.FixedLenFeature([], dtype=tf.string,
 101                                              default_value=''),
 102       'image/class/label': tf.io.FixedLenFeature([], dtype=tf.int64,
 103                                                  default_value=-1),
 104       'image/class/text': tf.io.FixedLenFeature([], dtype=tf.string,
 105                                                 default_value=''),
 106   }
 107   sparse_float32 = tf.io.VarLenFeature(dtype=tf.float32)
 108   # Sparse features in Example proto.
 109   feature_map.update(
 110       {k: sparse_float32 for k in ['image/object/bbox/xmin',
 111                                    'image/object/bbox/ymin',
 112                                    'image/object/bbox/xmax',
 113                                    'image/object/bbox/ymax']})
 114 
 115   features = tf.io.parse_single_example(serialized=example_serialized,
 116                                         features=feature_map)
 117   label = tf.cast(features['image/class/label'], dtype=tf.int32)
 118 
 119   xmin = tf.expand_dims(features['image/object/bbox/xmin'].values, 0)
 120   ymin = tf.expand_dims(features['image/object/bbox/ymin'].values, 0)
 121   xmax = tf.expand_dims(features['image/object/bbox/xmax'].values, 0)
 122   ymax = tf.expand_dims(features['image/object/bbox/ymax'].values, 0)
 123 
 124   # Note that we impose an ordering of (y, x) just to make life difficult.
 125   bbox = tf.concat([ymin, xmin, ymax, xmax], 0)
 126 
 127   # Force the variable number of bounding boxes into the shape
 128   # [1, num_boxes, coords].
 129   bbox = tf.expand_dims(bbox, 0)
 130   bbox = tf.transpose(a=bbox, perm=[0, 2, 1])
 131 
 132   return features['image/encoded'], label, bbox

 135 def parse_record(raw_record, is_training, dtype):
 136   """Parses a record containing a training example of an image.
 137 
 138   The input record is parsed into a label and image, and the image is passed
 139   through preprocessing steps (cropping, flipping, and so on).
 140 
 141   Args:
 142     raw_record: scalar Tensor tf.string containing a serialized
 143       Example protocol buffer.
 144     is_training: A boolean denoting whether the input is for training.
 145     dtype: data type to use for images/features.
 146 
 147   Returns:
 148     Tuple with processed image tensor and one-hot-encoded label tensor.
 149   """
 150   image_buffer, label, bbox = _parse_example_proto(raw_record)
 151 
 152   image = imagenet_preprocessing.preprocess_image(
 153       image_buffer=image_buffer,
 154       bbox=bbox,
 155       output_height=DEFAULT_IMAGE_SIZE,
 156       output_width=DEFAULT_IMAGE_SIZE,
 157       num_channels=NUM_CHANNELS,
 158       is_training=is_training)
 159   image = tf.cast(image, dtype)
 160 
 161   return image, label

 164 def input_fn(is_training,
 165              data_dir,
 166              batch_size,
 167              num_epochs=1,
 168              dtype=tf.float32,
 169              datasets_num_private_threads=None,
 170              parse_record_fn=parse_record,
 171              input_context=None,
 172              drop_remainder=False,
 173              tf_data_experimental_slack=False):
 174   """Input function which provides batches for train or eval.
 175 
 176   Args:
 177     is_training: A boolean denoting whether the input is for training.
 178     data_dir: The directory containing the input data.
 179     batch_size: The number of samples per batch.
 180     num_epochs: The number of epochs to repeat the dataset.
 181     dtype: Data type to use for images/features
 182     datasets_num_private_threads: Number of private threads for tf.data.
 183     parse_record_fn: Function to use for parsing the records.
 184     input_context: A `tf.distribute.InputContext` object passed in by
 185       `tf.distribute.Strategy`.
 186     drop_remainder: A boolean indicates whether to drop the remainder of the
 187       batches. If True, the batch dimension will be static.
 188     tf_data_experimental_slack: Whether to enable tf.data's
 189       `experimental_slack` option.
 190 
 191   Returns:
 192     A dataset that can be used for iteration.
 193   """
 194   filenames = get_filenames(is_training, data_dir)
 195   dataset = tf.data.Dataset.from_tensor_slices(filenames)
 196 
 197   if input_context:
 198     logging.info(
 199         'Sharding the dataset: input_pipeline_id=%d num_input_pipelines=%d',
 200         input_context.input_pipeline_id, input_context.num_input_pipelines)
 201     dataset = dataset.shard(input_context.num_input_pipelines,
 202                             input_context.input_pipeline_id)
 203 
 204   if is_training:
 205     # Shuffle the input files
 206     dataset = dataset.shuffle(buffer_size=_NUM_TRAIN_FILES)
 207 
 208   # Convert to individual records.
 209   # cycle_length = 10 means that up to 10 files will be read and deserialized in
 210   # parallel. You may want to increase this number if you have a large number of
 211   # CPU cores.
 212   dataset = dataset.interleave(
 213       tf.data.TFRecordDataset,
 214       cycle_length=10,
 215       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 216 
 217   return resnet_run_loop.process_record_dataset(
 218       dataset=dataset,
 219       is_training=is_training,
 220       batch_size=batch_size,
 221       shuffle_buffer=_SHUFFLE_BUFFER,
 222       parse_record_fn=parse_record_fn,
 223       num_epochs=num_epochs,
 224       dtype=dtype,
 225       datasets_num_private_threads=datasets_num_private_threads,
 226       drop_remainder=drop_remainder,
 227       tf_data_experimental_slack=tf_data_experimental_slack,
 228   )

 240 class ImagenetModel(resnet_model.Model):
 241   """Model class with appropriate defaults for Imagenet data."""
 242 
 243   def __init__(self, resnet_size, data_format=None, num_classes=NUM_CLASSES,
 244                resnet_version=resnet_model.DEFAULT_VERSION,
 245                dtype=resnet_model.DEFAULT_DTYPE):
 246     """These are the parameters that work for Imagenet data.
 247 
 248     Args:
 249       resnet_size: The number of convolutional layers needed in the model.
 250       data_format: Either 'channels_first' or 'channels_last', specifying which
 251         data format to use when setting up the model.
 252       num_classes: The number of output classes needed from the model. This
 253         enables users to extend the same model to their own datasets.
 254       resnet_version: Integer representing which version of the ResNet network
 255         to use. See README for details. Valid values: [1, 2]
 256       dtype: The TensorFlow dtype to use for calculations.
 257     """
 258 
 259     # For bigger models, we want to use "bottleneck" layers
 260     if resnet_size < 50:
 261       bottleneck = False
 262     else:
 263       bottleneck = True
 264 
 265     super(ImagenetModel, self).__init__(
 266         resnet_size=resnet_size,
 267         bottleneck=bottleneck,
 268         num_classes=num_classes,
 269         num_filters=64,
 270         kernel_size=7,
 271         conv_stride=2,
 272         first_pool_size=3,
 273         first_pool_stride=2,
 274         block_sizes=_get_block_sizes(resnet_size),
 275         block_strides=[1, 2, 2, 2],
 276         resnet_version=resnet_version,
 277         data_format=data_format,
 278         dtype=dtype
 279     )

 282 def _get_block_sizes(resnet_size):
 283   """Retrieve the size of each block_layer in the ResNet model.
 284 
 285   The number of block layers used for the Resnet model varies according
 286   to the size of the model. This helper grabs the layer set we want, throwing
 287   an error if a non-standard size has been selected.
 288 
 289   Args:
 290     resnet_size: The number of convolutional layers needed in the model.
 291 
 292   Returns:
 293     A list of block sizes to use in building the model.
 294 
 295   Raises:
 296     KeyError: if invalid resnet_size is received.
 297   """
 298   choices = {
 299       18: [2, 2, 2, 2],
 300       34: [3, 4, 6, 3],
 301       50: [3, 4, 6, 3],
 302       101: [3, 4, 23, 3],
 303       152: [3, 8, 36, 3],
 304       200: [3, 24, 36, 3]
 305   }
 306 
 307   try:
 308     return choices[resnet_size]
 309   except KeyError:
 310     err = ('Could not find layers for selected Resnet size.\n'
 311            'Size received: {}; sizes allowed: {}.'.format(
 312                resnet_size, list(choices.keys())))
 313     raise ValueError(err)

 316 def imagenet_model_fn(features, labels, mode, params):
 317   """Our model_fn for ResNet to be used with our Estimator."""
 318 
 319   # Warmup and higher lr may not be valid for fine tuning with small batches
 320   # and smaller numbers of training images.
 321   if params['fine_tune']:
 322     warmup = False
 323     base_lr = .1
 324   else:
 325     warmup = True
 326     base_lr = .128
 327 
 328   learning_rate_fn = resnet_run_loop.learning_rate_with_decay(
 329       batch_size=params['batch_size'] * params.get('num_workers', 1),
 330       batch_denom=256, num_images=NUM_IMAGES['train'],
 331       boundary_epochs=[30, 60, 80, 90], decay_rates=[1, 0.1, 0.01, 0.001, 1e-4],
 332       warmup=warmup, base_lr=base_lr)
 333 
 334   return resnet_run_loop.resnet_model_fn(
 335       features=features,
 336       labels=labels,
 337       mode=mode,
 338       model_class=ImagenetModel,
 339       resnet_size=params['resnet_size'],
 340       weight_decay=flags.FLAGS.weight_decay,
 341       learning_rate_fn=learning_rate_fn,
 342       momentum=0.9,
 343       data_format=params['data_format'],
 344       resnet_version=params['resnet_version'],
 345       loss_scale=params['loss_scale'],
 346       loss_filter_fn=None,
 347       dtype=params['dtype'],
 348       fine_tune=params['fine_tune'],
 349       label_smoothing=flags.FLAGS.label_smoothing
 350   )

 362 def run_imagenet(flags_obj):
 363   """Run ResNet ImageNet training and eval loop.
 364 
 365   Args:
 366     flags_obj: An object containing parsed flag values.
 367 
 368   Returns:
 369     Dict of results of the run.  Contains the keys `eval_results` and
 370       `train_hooks`. `eval_results` contains accuracy (top_1) and
 371       accuracy_top_5. `train_hooks` is a list the instances of hooks used during
 372       training.
 373   """
 374   input_function = (flags_obj.use_synthetic_data and
 375                     get_synth_input_fn(flags_core.get_tf_dtype(flags_obj)) or
 376                     input_fn)
 377 
 378   result = resnet_run_loop.resnet_main(
 379       flags_obj, imagenet_model_fn, input_function, DATASET_NAME,
 380       shape=[DEFAULT_IMAGE_SIZE, DEFAULT_IMAGE_SIZE, NUM_CHANNELS])
 381 
 382   return result
