  15 """Provides utilities to preprocess images.
  16 
  17 Training images are sampled using the provided bounding boxes, and subsequently
  18 cropped to the sampled bounding box. Images are additionally flipped randomly,
  19 then resized to the target output size (without aspect-ratio preservation).
  20 
  21 Images used during evaluation are resized (with aspect-ratio preservation) and
  22 centrally cropped.
  23 
  24 All images undergo mean color subtraction.
  25 
  26 Note that these steps are colloquially referred to as "ResNet preprocessing,"
  27 and they differ from "VGG preprocessing," which does not use bounding boxes
  28 and instead does an aspect-preserving resize followed by random crop during
  29 training. (These both differ from "Inception preprocessing," which introduces
  30 color distortion steps.)
  31 
  32 """
  33 
  34 from __future__ import absolute_import
  35 from __future__ import division
  36 from __future__ import print_function

  38 import tensorflow as tf
  39 
  40 _R_MEAN = 123.68
  41 _G_MEAN = 116.78
  42 _B_MEAN = 103.94
  43 _CHANNEL_MEANS = [_R_MEAN, _G_MEAN, _B_MEAN]
  44 
  45 # The lower bound for the smallest side of the image for aspect-preserving
  46 # resizing. For example, if an image is 500 x 1000, it will be resized to
  47 # _RESIZE_MIN x (_RESIZE_MIN * 2).
  48 _RESIZE_MIN = 256

  51 def _decode_crop_and_flip(image_buffer, bbox, num_channels):
  52   """Crops the given image to a random part of the image, and randomly flips.
  53 
  54   We use the fused decode_and_crop op, which performs better than the two ops
  55   used separately in series, but note that this requires that the image be
  56   passed in as an un-decoded string Tensor.
  57 
  58   Args:
  59     image_buffer: scalar string Tensor representing the raw JPEG image buffer.
  60     bbox: 3-D float Tensor of bounding boxes arranged [1, num_boxes, coords]
  61       where each coordinate is [0, 1) and the coordinates are arranged as
  62       [ymin, xmin, ymax, xmax].
  63     num_channels: Integer depth of the image buffer for decoding.
  64 
  65   Returns:
  66     3-D tensor with cropped image.
  67 
  68   """
  69   # A large fraction of image datasets contain a human-annotated bounding box
  70   # delineating the region of the image containing the object of interest.  We
  71   # choose to create a new bounding box for the object which is a randomly
  72   # distorted version of the human-annotated bounding box that obeys an
  73   # allowed range of aspect ratios, sizes and overlap with the human-annotated
  74   # bounding box. If no box is supplied, then we assume the bounding box is
  75   # the entire image.
  76   sample_distorted_bounding_box = tf.image.sample_distorted_bounding_box(
  77       tf.image.extract_jpeg_shape(image_buffer),
  78       bounding_boxes=bbox,
  79       min_object_covered=0.1,
  80       aspect_ratio_range=[0.75, 1.33],
  81       area_range=[0.05, 1.0],
  82       max_attempts=100,
  83       use_image_if_no_bounding_boxes=True)
  84   bbox_begin, bbox_size, _ = sample_distorted_bounding_box
  85 
  86   # Reassemble the bounding box in the format the crop op requires.
  87   offset_y, offset_x, _ = tf.unstack(bbox_begin)
  88   target_height, target_width, _ = tf.unstack(bbox_size)
  89   crop_window = tf.stack([offset_y, offset_x, target_height, target_width])
  90 
  91   # Use the fused decode and crop op here, which is faster than each in series.
  92   cropped = tf.image.decode_and_crop_jpeg(
  93       image_buffer, crop_window, channels=num_channels)
  94 
  95   # Flip to add a little more random distortion in.
  96   cropped = tf.image.random_flip_left_right(cropped)
  97   return cropped

 100 def _central_crop(image, crop_height, crop_width):
 101   """Performs central crops of the given image list.
 102 
 103   Args:
 104     image: a 3-D image tensor
 105     crop_height: the height of the image following the crop.
 106     crop_width: the width of the image following the crop.
 107 
 108   Returns:
 109     3-D tensor with cropped image.
 110   """
 111   shape = tf.shape(input=image)
 112   height, width = shape[0], shape[1]
 113 
 114   amount_to_be_cropped_h = (height - crop_height)
 115   crop_top = amount_to_be_cropped_h // 2
 116   amount_to_be_cropped_w = (width - crop_width)
 117   crop_left = amount_to_be_cropped_w // 2
 118   return tf.slice(
 119       image, [crop_top, crop_left, 0], [crop_height, crop_width, -1])

 122 def _mean_image_subtraction(image, means, num_channels):
 123   """Subtracts the given means from each image channel.
 124 
 125   For example:
 126     means = [123.68, 116.779, 103.939]
 127     image = _mean_image_subtraction(image, means)
 128 
 129   Note that the rank of `image` must be known.
 130 
 131   Args:
 132     image: a tensor of size [height, width, C].
 133     means: a C-vector of values to subtract from each channel.
 134     num_channels: number of color channels in the image that will be distorted.
 135 
 136   Returns:
 137     the centered image.
 138 
 139   Raises:
 140     ValueError: If the rank of `image` is unknown, if `image` has a rank other
 141       than three or if the number of channels in `image` doesn't match the
 142       number of values in `means`.
 143   """
 144   if image.get_shape().ndims != 3:
 145     raise ValueError('Input must be of size [height, width, C>0]')
 146 
 147   if len(means) != num_channels:
 148     raise ValueError('len(means) must match the number of channels')
 149 
 150   # We have a 1-D tensor of means; convert to 3-D.
 151   # Note(b/130245863): we explicitly call `broadcast` instead of simply
 152   # expanding dimensions for better performance.
 153   means = tf.broadcast_to(means, tf.shape(image))
 154 
 155   return image - means

 158 def _smallest_size_at_least(height, width, resize_min):
 159   """Computes new shape with the smallest side equal to `smallest_side`.
 160 
 161   Computes new shape with the smallest side equal to `smallest_side` while
 162   preserving the original aspect ratio.
 163 
 164   Args:
 165     height: an int32 scalar tensor indicating the current height.
 166     width: an int32 scalar tensor indicating the current width.
 167     resize_min: A python integer or scalar `Tensor` indicating the size of
 168       the smallest side after resize.
 169 
 170   Returns:
 171     new_height: an int32 scalar tensor indicating the new height.
 172     new_width: an int32 scalar tensor indicating the new width.
 173   """
 174   resize_min = tf.cast(resize_min, tf.float32)
 175 
 176   # Convert to floats to make subsequent calculations go smoothly.
 177   height, width = tf.cast(height, tf.float32), tf.cast(width, tf.float32)
 178 
 179   smaller_dim = tf.minimum(height, width)
 180   scale_ratio = resize_min / smaller_dim
 181 
 182   # Convert back to ints to make heights and widths that TF ops will accept.
 183   new_height = tf.cast(height * scale_ratio, tf.int32)
 184   new_width = tf.cast(width * scale_ratio, tf.int32)
 185 
 186   return new_height, new_width

 189 def _aspect_preserving_resize(image, resize_min):
 190   """Resize images preserving the original aspect ratio.
 191 
 192   Args:
 193     image: A 3-D image `Tensor`.
 194     resize_min: A python integer or scalar `Tensor` indicating the size of
 195       the smallest side after resize.
 196 
 197   Returns:
 198     resized_image: A 3-D tensor containing the resized image.
 199   """
 200   shape = tf.shape(input=image)
 201   height, width = shape[0], shape[1]
 202 
 203   new_height, new_width = _smallest_size_at_least(height, width, resize_min)
 204 
 205   return _resize_image(image, new_height, new_width)

 208 def _resize_image(image, height, width):
 209   """Simple wrapper around tf.resize_images.
 210 
 211   This is primarily to make sure we use the same `ResizeMethod` and other
 212   details each time.
 213 
 214   Args:
 215     image: A 3-D image `Tensor`.
 216     height: The target height for the resized image.
 217     width: The target width for the resized image.
 218 
 219   Returns:
 220     resized_image: A 3-D tensor containing the resized image. The first two
 221       dimensions have the shape [height, width].
 222   """
 223   return tf.compat.v1.image.resize(
 224       image, [height, width], method=tf.image.ResizeMethod.BILINEAR,
 225       align_corners=False)

 228 def preprocess_image(image_buffer, bbox, output_height, output_width,
 229                      num_channels, is_training=False):
 230   """Preprocesses the given image.
 231 
 232   Preprocessing includes decoding, cropping, and resizing for both training
 233   and eval images. Training preprocessing, however, introduces some random
 234   distortion of the image to improve accuracy.
 235 
 236   Args:
 237     image_buffer: scalar string Tensor representing the raw JPEG image buffer.
 238     bbox: 3-D float Tensor of bounding boxes arranged [1, num_boxes, coords]
 239       where each coordinate is [0, 1) and the coordinates are arranged as
 240       [ymin, xmin, ymax, xmax].
 241     output_height: The height of the image after preprocessing.
 242     output_width: The width of the image after preprocessing.
 243     num_channels: Integer depth of the image buffer for decoding.
 244     is_training: `True` if we're preprocessing the image for training and
 245       `False` otherwise.
 246 
 247   Returns:
 248     A preprocessed image.
 249   """
 250   if is_training:
 251     # For training, we want to randomize some of the distortions.
 252     image = _decode_crop_and_flip(image_buffer, bbox, num_channels)
 253     image = _resize_image(image, output_height, output_width)
 254   else:
 255     # For validation, we want to decode, resize, then just crop the middle.
 256     image = tf.image.decode_jpeg(image_buffer, channels=num_channels)
 257     image = _aspect_preserving_resize(image, _RESIZE_MIN)
 258     image = _central_crop(image, output_height, output_width)
 259 
 260   image.set_shape([output_height, output_width, num_channels])
 261 
 262   return _mean_image_subtraction(image, _CHANNEL_MEANS, num_channels)
