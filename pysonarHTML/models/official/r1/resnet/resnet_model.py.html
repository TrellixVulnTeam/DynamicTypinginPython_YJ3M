  15 """Contains definitions for Residual Networks.
  16 
  17 Residual networks ('v1' ResNets) were originally proposed in:
  18 [1] Kaiming He, Xiangyu Zhang, Shaoqing Ren, Jian Sun
  19     Deep Residual Learning for Image Recognition. arXiv:1512.03385
  20 
  21 The full preactivation 'v2' ResNet variant was introduced by:
  22 [2] Kaiming He, Xiangyu Zhang, Shaoqing Ren, Jian Sun
  23     Identity Mappings in Deep Residual Networks. arXiv: 1603.05027
  24 
  25 The key difference of the full preactivation 'v2' variant compared to the
  26 'v1' variant in [1] is the use of batch normalization before every weight layer
  27 rather than after.
  28 """

  47 def batch_norm(inputs, training, data_format):
  48   """Performs a batch normalization using a standard set of parameters."""
  49   # We set fused=True for a significant performance boost. See
  50   # https://www.tensorflow.org/performance/performance_guide#common_fused_ops
  51   return tf.compat.v1.layers.batch_normalization(
  52       inputs=inputs, axis=1 if data_format == 'channels_first' else 3,
  53       momentum=_BATCH_NORM_DECAY, epsilon=_BATCH_NORM_EPSILON, center=True,
  54       scale=True, training=training, fused=True)
  55

  57 def fixed_padding(inputs, kernel_size, data_format):
  58   """Pads the input along the spatial dimensions independently of input size.
  59 
  60   Args:
  61     inputs: A tensor of size [batch, channels, height_in, width_in] or
  62       [batch, height_in, width_in, channels] depending on data_format.
  63     kernel_size: The kernel to be used in the conv2d or max_pool2d operation.
  64                  Should be a positive integer.
  65     data_format: The input format ('channels_last' or 'channels_first').
  66 
  67   Returns:
  68     A tensor with the same format as the input with the data either intact
  69     (if kernel_size == 1) or padded (if kernel_size > 1).
  70   """
  71   pad_total = kernel_size - 1
  72   pad_beg = pad_total // 2
  73   pad_end = pad_total - pad_beg
  74 
  75   if data_format == 'channels_first':
  76     padded_inputs = tf.pad(tensor=inputs,
  77                            paddings=[[0, 0], [0, 0], [pad_beg, pad_end],
  78                                      [pad_beg, pad_end]])
  79   else:
  80     padded_inputs = tf.pad(tensor=inputs,
  81                            paddings=[[0, 0], [pad_beg, pad_end],
  82                                      [pad_beg, pad_end], [0, 0]])
  83   return padded_inputs
  84 
  85 
  86 def conv2d_fixed_padding(inputs, filters, kernel_size, strides, data_format):
  87   """Strided 2-D convolution with explicit padding."""
  88   # The padding is consistent and is based only on `kernel_size`, not on the
  89   # dimensions of `inputs` (as opposed to using `tf.layers.conv2d` alone).
  90   if strides > 1:
  91     inputs = fixed_padding(inputs, kernel_size, data_format)
  92 
  93   return tf.compat.v1.layers.conv2d(
  94       inputs=inputs, filters=filters, kernel_size=kernel_size, strides=strides,
  95       padding=('SAME' if strides == 1 else 'VALID'), use_bias=False,
  96       kernel_initializer=tf.compat.v1.variance_scaling_initializer(),
  97       data_format=data_format)

 103 def _building_block_v1(inputs, filters, training, projection_shortcut, strides,
 104                        data_format):
 105   """A single block for ResNet v1, without a bottleneck.
 106 
 107   Convolution then batch normalization then ReLU as described by:
 108     Deep Residual Learning for Image Recognition
 109     https://arxiv.org/pdf/1512.03385.pdf
 110     by Kaiming He, Xiangyu Zhang, Shaoqing Ren, and Jian Sun, Dec 2015.
 111 
 112   Args:
 113     inputs: A tensor of size [batch, channels, height_in, width_in] or
 114       [batch, height_in, width_in, channels] depending on data_format.
 115     filters: The number of filters for the convolutions.
 116     training: A Boolean for whether the model is in training or inference
 117       mode. Needed for batch normalization.
 118     projection_shortcut: The function to use for projection shortcuts
 119       (typically a 1x1 convolution when downsampling the input).
 120     strides: The block's stride. If greater than 1, this block will ultimately
 121       downsample the input.
 122     data_format: The input format ('channels_last' or 'channels_first').
 123 
 124   Returns:
 125     The output tensor of the block; shape should match inputs.
 126   """
 127   shortcut = inputs
 128 
 129   if projection_shortcut is not None:
 130     shortcut = projection_shortcut(inputs)
 131     shortcut = batch_norm(inputs=shortcut, training=training,
 132                           data_format=data_format)
 133 
 134   inputs = conv2d_fixed_padding(
 135       inputs=inputs, filters=filters, kernel_size=3, strides=strides,
 136       data_format=data_format)
 137   inputs = batch_norm(inputs, training, data_format)
 138   inputs = tf.nn.relu(inputs)
 139 
 140   inputs = conv2d_fixed_padding(
 141       inputs=inputs, filters=filters, kernel_size=3, strides=1,
 142       data_format=data_format)
 143   inputs = batch_norm(inputs, training, data_format)
 144   inputs += shortcut
 145   inputs = tf.nn.relu(inputs)
 146 
 147   return inputs
 148

 150 def _building_block_v2(inputs, filters, training, projection_shortcut, strides,
 151                        data_format):
 152   """A single block for ResNet v2, without a bottleneck.
 153 
 154   Batch normalization then ReLu then convolution as described by:
 155     Identity Mappings in Deep Residual Networks
 156     https://arxiv.org/pdf/1603.05027.pdf
 157     by Kaiming He, Xiangyu Zhang, Shaoqing Ren, and Jian Sun, Jul 2016.
 158 
 159   Args:
 160     inputs: A tensor of size [batch, channels, height_in, width_in] or
 161       [batch, height_in, width_in, channels] depending on data_format.
 162     filters: The number of filters for the convolutions.
 163     training: A Boolean for whether the model is in training or inference
 164       mode. Needed for batch normalization.
 165     projection_shortcut: The function to use for projection shortcuts
 166       (typically a 1x1 convolution when downsampling the input).
 167     strides: The block's stride. If greater than 1, this block will ultimately
 168       downsample the input.
 169     data_format: The input format ('channels_last' or 'channels_first').
 170 
 171   Returns:
 172     The output tensor of the block; shape should match inputs.
 173   """
 174   shortcut = inputs
 175   inputs = batch_norm(inputs, training, data_format)
 176   inputs = tf.nn.relu(inputs)
 177 
 178   # The projection shortcut should come after the first batch norm and ReLU
 179   # since it performs a 1x1 convolution.
 180   if projection_shortcut is not None:
 181     shortcut = projection_shortcut(inputs)
 182 
 183   inputs = conv2d_fixed_padding(
 184       inputs=inputs, filters=filters, kernel_size=3, strides=strides,
 185       data_format=data_format)
 186 
 187   inputs = batch_norm(inputs, training, data_format)
 188   inputs = tf.nn.relu(inputs)
 189   inputs = conv2d_fixed_padding(
 190       inputs=inputs, filters=filters, kernel_size=3, strides=1,
 191       data_format=data_format)
 192 
 193   return inputs + shortcut

 196 def _bottleneck_block_v1(inputs, filters, training, projection_shortcut,
 197                          strides, data_format):
 198   """A single block for ResNet v1, with a bottleneck.
 199 
 200   Similar to _building_block_v1(), except using the "bottleneck" blocks
 201   described in:
 202     Convolution then batch normalization then ReLU as described by:
 203       Deep Residual Learning for Image Recognition
 204       https://arxiv.org/pdf/1512.03385.pdf
 205       by Kaiming He, Xiangyu Zhang, Shaoqing Ren, and Jian Sun, Dec 2015.
 206 
 207   Args:
 208     inputs: A tensor of size [batch, channels, height_in, width_in] or
 209       [batch, height_in, width_in, channels] depending on data_format.
 210     filters: The number of filters for the convolutions.
 211     training: A Boolean for whether the model is in training or inference
 212       mode. Needed for batch normalization.
 213     projection_shortcut: The function to use for projection shortcuts
 214       (typically a 1x1 convolution when downsampling the input).
 215     strides: The block's stride. If greater than 1, this block will ultimately
 216       downsample the input.
 217     data_format: The input format ('channels_last' or 'channels_first').
 218 
 219   Returns:
 220     The output tensor of the block; shape should match inputs.
 221   """
 222   shortcut = inputs
 223 
 224   if projection_shortcut is not None:
 225     shortcut = projection_shortcut(inputs)
 226     shortcut = batch_norm(inputs=shortcut, training=training,
 227                           data_format=data_format)
 228 
 229   inputs = conv2d_fixed_padding(
 230       inputs=inputs, filters=filters, kernel_size=1, strides=1,
 231       data_format=data_format)
 232   inputs = batch_norm(inputs, training, data_format)
 233   inputs = tf.nn.relu(inputs)
 234 
 235   inputs = conv2d_fixed_padding(
 236       inputs=inputs, filters=filters, kernel_size=3, strides=strides,
 237       data_format=data_format)
 238   inputs = batch_norm(inputs, training, data_format)
 239   inputs = tf.nn.relu(inputs)
 240 
 241   inputs = conv2d_fixed_padding(
 242       inputs=inputs, filters=4 * filters, kernel_size=1, strides=1,
 243       data_format=data_format)
 244   inputs = batch_norm(inputs, training, data_format)
 245   inputs += shortcut
 246   inputs = tf.nn.relu(inputs)
 247 
 248   return inputs

 251 def _bottleneck_block_v2(inputs, filters, training, projection_shortcut,
 252                          strides, data_format):
 253   """A single block for ResNet v2, with a bottleneck.
 254 
 255   Similar to _building_block_v2(), except using the "bottleneck" blocks
 256   described in:
 257     Convolution then batch normalization then ReLU as described by:
 258       Deep Residual Learning for Image Recognition
 259       https://arxiv.org/pdf/1512.03385.pdf
 260       by Kaiming He, Xiangyu Zhang, Shaoqing Ren, and Jian Sun, Dec 2015.
 261 
 262   Adapted to the ordering conventions of:
 263     Batch normalization then ReLu then convolution as described by:
 264       Identity Mappings in Deep Residual Networks
 265       https://arxiv.org/pdf/1603.05027.pdf
 266       by Kaiming He, Xiangyu Zhang, Shaoqing Ren, and Jian Sun, Jul 2016.
 267 
 268   Args:
 269     inputs: A tensor of size [batch, channels, height_in, width_in] or
 270       [batch, height_in, width_in, channels] depending on data_format.
 271     filters: The number of filters for the convolutions.
 272     training: A Boolean for whether the model is in training or inference
 273       mode. Needed for batch normalization.
 274     projection_shortcut: The function to use for projection shortcuts
 275       (typically a 1x1 convolution when downsampling the input).
 276     strides: The block's stride. If greater than 1, this block will ultimately
 277       downsample the input.
 278     data_format: The input format ('channels_last' or 'channels_first').
 279 
 280   Returns:
 281     The output tensor of the block; shape should match inputs.
 282   """
 283   shortcut = inputs
 284   inputs = batch_norm(inputs, training, data_format)
 285   inputs = tf.nn.relu(inputs)
 286 
 287   # The projection shortcut should come after the first batch norm and ReLU
 288   # since it performs a 1x1 convolution.
 289   if projection_shortcut is not None:
 290     shortcut = projection_shortcut(inputs)
 291 
 292   inputs = conv2d_fixed_padding(
 293       inputs=inputs, filters=filters, kernel_size=1, strides=1,
 294       data_format=data_format)
 295 
 296   inputs = batch_norm(inputs, training, data_format)
 297   inputs = tf.nn.relu(inputs)
 298   inputs = conv2d_fixed_padding(
 299       inputs=inputs, filters=filters, kernel_size=3, strides=strides,
 300       data_format=data_format)
 301 
 302   inputs = batch_norm(inputs, training, data_format)
 303   inputs = tf.nn.relu(inputs)
 304   inputs = conv2d_fixed_padding(
 305       inputs=inputs, filters=4 * filters, kernel_size=1, strides=1,
 306       data_format=data_format)
 307 
 308   return inputs + shortcut

 311 def block_layer(inputs, filters, bottleneck, block_fn, blocks, strides,
 312                 training, name, data_format):
 313   """Creates one layer of blocks for the ResNet model.
 314 
 315   Args:
 316     inputs: A tensor of size [batch, channels, height_in, width_in] or
 317       [batch, height_in, width_in, channels] depending on data_format.
 318     filters: The number of filters for the first convolution of the layer.
 319     bottleneck: Is the block created a bottleneck block.
 320     block_fn: The block to use within the model, either `building_block` or
 321       `bottleneck_block`.
 322     blocks: The number of blocks contained in the layer.
 323     strides: The stride to use for the first convolution of the layer. If
 324       greater than 1, this layer will ultimately downsample the input.
 325     training: Either True or False, whether we are currently training the
 326       model. Needed for batch norm.
 327     name: A string name for the tensor output of the block layer.
 328     data_format: The input format ('channels_last' or 'channels_first').
 329 
 330   Returns:
 331     The output tensor of the block layer.
 332   """
 333 
 334   # Bottleneck blocks end with 4x the number of filters as they start with
 335   filters_out = filters * 4 if bottleneck else filters
 336 
 337   def projection_shortcut(inputs):
 338     return conv2d_fixed_padding(
 339         inputs=inputs, filters=filters_out, kernel_size=1, strides=strides,
 340         data_format=data_format)
 341 
 342   # Only the first block per block_layer uses projection_shortcut and strides
 343   inputs = block_fn(inputs, filters, training, projection_shortcut, strides,
 344                     data_format)
 345 
 346   for _ in range(1, blocks):
 347     inputs = block_fn(inputs, filters, training, None, 1, data_format)
 348 
 349   return tf.identity(inputs, name)

 352 class Model(object):
 353   """Base class for building the Resnet Model."""
 354 
 355   def __init__(self, resnet_size, bottleneck, num_classes, num_filters,
 356                kernel_size,
 357                conv_stride, first_pool_size, first_pool_stride,
 358                block_sizes, block_strides,
 359                resnet_version=DEFAULT_VERSION, data_format=None,
 360                dtype=DEFAULT_DTYPE):
 361     """Creates a model for classifying an image.
 362 
 363     Args:
 364       resnet_size: A single integer for the size of the ResNet model.
 365       bottleneck: Use regular blocks or bottleneck blocks.
 366       num_classes: The number of classes used as labels.
 367       num_filters: The number of filters to use for the first block layer
 368         of the model. This number is then doubled for each subsequent block
 369         layer.
 370       kernel_size: The kernel size to use for convolution.
 371       conv_stride: stride size for the initial convolutional layer
 372       first_pool_size: Pool size to be used for the first pooling layer.
 373         If none, the first pooling layer is skipped.
 374       first_pool_stride: stride size for the first pooling layer. Not used
 375         if first_pool_size is None.
 376       block_sizes: A list containing n values, where n is the number of sets of
 377         block layers desired. Each value should be the number of blocks in the
 378         i-th set.
 379       block_strides: List of integers representing the desired stride size for
 380         each of the sets of block layers. Should be same length as block_sizes.
 381       resnet_version: Integer representing which version of the ResNet network
 382         to use. See README for details. Valid values: [1, 2]
 383       data_format: Input format ('channels_last', 'channels_first', or None).
 384         If set to None, the format is dependent on whether a GPU is available.
 385       dtype: The TensorFlow dtype to use for calculations. If not specified
 386         tf.float32 is used.
 387 
 388     Raises:
 389       ValueError: if invalid version is selected.
 390     """
 391     self.resnet_size = resnet_size
 392 
 393     if not data_format:
 394       data_format = ('channels_first' if tf.config.list_physical_devices('GPU')
 395                      else 'channels_last')
 396 
 397     self.resnet_version = resnet_version
 398     if resnet_version not in (1, 2):
 399       raise ValueError(
 400           'Resnet version should be 1 or 2. See README for citations.')
 401 
 402     self.bottleneck = bottleneck
 403     if bottleneck:
 404       if resnet_version == 1:
 405         self.block_fn = _bottleneck_block_v1
 406       else:
 407         self.block_fn = _bottleneck_block_v2
 408     else:
 409       if resnet_version == 1:
 410         self.block_fn = _building_block_v1
 411       else:
 412         self.block_fn = _building_block_v2
 413 
 414     if dtype not in ALLOWED_TYPES:
 415       raise ValueError('dtype must be one of: {}'.format(ALLOWED_TYPES))
 416 
 417     self.data_format = data_format
 418     self.num_classes = num_classes
 419     self.num_filters = num_filters
 420     self.kernel_size = kernel_size
 421     self.conv_stride = conv_stride
 422     self.first_pool_size = first_pool_size
 423     self.first_pool_stride = first_pool_stride
 424     self.block_sizes = block_sizes
 425     self.block_strides = block_strides
 426     self.dtype = dtype
 427     self.pre_activation = resnet_version == 2
 428 
 429   def _custom_dtype_getter(self, getter, name, shape=None, dtype=DEFAULT_DTYPE,
 430                            *args, **kwargs):
 431     """Creates variables in fp32, then casts to fp16 if necessary.
 432 
 433     This function is a custom getter. A custom getter is a function with the
 434     same signature as tf.get_variable, except it has an additional getter
 435     parameter. Custom getters can be passed as the `custom_getter` parameter of
 436     tf.variable_scope. Then, tf.get_variable will call the custom getter,
 437     instead of directly getting a variable itself. This can be used to change
 438     the types of variables that are retrieved with tf.get_variable.
 439     The `getter` parameter is the underlying variable getter, that would have
 440     been called if no custom getter was used. Custom getters typically get a
 441     variable with `getter`, then modify it in some way.
 442 
 443     This custom getter will create an fp32 variable. If a low precision
 444     (e.g. float16) variable was requested it will then cast the variable to the
 445     requested dtype. The reason we do not directly create variables in low
 446     precision dtypes is that applying small gradients to such variables may
 447     cause the variable not to change.
 448 
 449     Args:
 450       getter: The underlying variable getter, that has the same signature as
 451         tf.get_variable and returns a variable.
 452       name: The name of the variable to get.
 453       shape: The shape of the variable to get.
 454       dtype: The dtype of the variable to get. Note that if this is a low
 455         precision dtype, the variable will be created as a tf.float32 variable,
 456         then cast to the appropriate dtype
 457       *args: Additional arguments to pass unmodified to getter.
 458       **kwargs: Additional keyword arguments to pass unmodified to getter.
 459 
 460     Returns:
 461       A variable which is cast to fp16 if necessary.
 462     """
 463 
 464     if dtype in CASTABLE_TYPES:
 465       var = getter(name, shape, tf.float32, *args, **kwargs)
 466       return tf.cast(var, dtype=dtype, name=name + '_cast')
 467     else:
 468       return getter(name, shape, dtype, *args, **kwargs)
 469 
 470   def _model_variable_scope(self):
 471     """Returns a variable scope that the model should be created under.
 472 
 473     If self.dtype is a castable type, model variable will be created in fp32
 474     then cast to self.dtype before being used.
 475 
 476     Returns:
 477       A variable scope for the model.
 478     """
 479 
 480     return tf.compat.v1.variable_scope('resnet_model',
 481                                        custom_getter=self._custom_dtype_getter)
 482 
 483   def __call__(self, inputs, training):
 484     """Add operations to classify a batch of input images.
 485 
 486     Args:
 487       inputs: A Tensor representing a batch of input images.
 488       training: A boolean. Set to True to add operations required only when
 489         training the classifier.
 490 
 491     Returns:
 492       A logits Tensor with shape [<batch_size>, self.num_classes].
 493     """
 494 
 495     with self._model_variable_scope():
 496       if self.data_format == 'channels_first':
 497         # Convert the inputs from channels_last (NHWC) to channels_first (NCHW).
 498         # This provides a large performance boost on GPU. See
 499         # https://www.tensorflow.org/performance/performance_guide#data_formats
 500         inputs = tf.transpose(a=inputs, perm=[0, 3, 1, 2])
 501 
 502       inputs = conv2d_fixed_padding(
 503           inputs=inputs, filters=self.num_filters, kernel_size=self.kernel_size,
 504           strides=self.conv_stride, data_format=self.data_format)
 505       inputs = tf.identity(inputs, 'initial_conv')
 506 
 507       # We do not include batch normalization or activation functions in V2
 508       # for the initial conv1 because the first ResNet unit will perform these
 509       # for both the shortcut and non-shortcut paths as part of the first
 510       # block's projection. Cf. Appendix of [2].
 511       if self.resnet_version == 1:
 512         inputs = batch_norm(inputs, training, self.data_format)
 513         inputs = tf.nn.relu(inputs)
 514 
 515       if self.first_pool_size:
 516         inputs = tf.compat.v1.layers.max_pooling2d(
 517             inputs=inputs, pool_size=self.first_pool_size,
 518             strides=self.first_pool_stride, padding='SAME',
 519             data_format=self.data_format)
 520         inputs = tf.identity(inputs, 'initial_max_pool')
 521 
 522       for i, num_blocks in enumerate(self.block_sizes):
 523         num_filters = self.num_filters * (2**i)
 524         inputs = block_layer(
 525             inputs=inputs, filters=num_filters, bottleneck=self.bottleneck,
 526             block_fn=self.block_fn, blocks=num_blocks,
 527             strides=self.block_strides[i], training=training,
 528             name='block_layer{}'.format(i + 1), data_format=self.data_format)
 529 
 530       # Only apply the BN and ReLU for model that does pre_activation in each
 531       # building/bottleneck block, eg resnet V2.
 532       if self.pre_activation:
 533         inputs = batch_norm(inputs, training, self.data_format)
 534         inputs = tf.nn.relu(inputs)
 535 
 536       # The current top layer has shape
 537       # `batch_size x pool_size x pool_size x final_size`.
 538       # ResNet does an Average Pooling layer over pool_size,
 539       # but that is the same as doing a reduce_mean. We do a reduce_mean
 540       # here because it performs better than AveragePooling2D.
 541       axes = [2, 3] if self.data_format == 'channels_first' else [1, 2]
 542       inputs = tf.reduce_mean(input_tensor=inputs, axis=axes, keepdims=True)
 543       inputs = tf.identity(inputs, 'final_reduce_mean')
 544 
 545       inputs = tf.squeeze(inputs, axes)
 546       inputs = tf.compat.v1.layers.dense(inputs=inputs, units=self.num_classes)
 547       inputs = tf.identity(inputs, 'final_dense')
 548       return inputs
