  34 HEIGHT = 32
  35 WIDTH = 32
  36 NUM_CHANNELS = 3
  37 _DEFAULT_IMAGE_BYTES = HEIGHT * WIDTH * NUM_CHANNELS
  38 # The record is the image plus a one-byte label
  39 _RECORD_BYTES = _DEFAULT_IMAGE_BYTES + 1

  55 def get_filenames(is_training, data_dir):
  56   """Returns a list of filenames."""
  57   assert tf.io.gfile.exists(data_dir), (
  58       'Run cifar10_download_and_extract.py first to download and extract the '
  59       'CIFAR-10 data.')
  60 
  61   if is_training:
  62     return [
  63         os.path.join(data_dir, 'data_batch_%d.bin' % i)
  64         for i in range(1, _NUM_DATA_FILES + 1)
  65     ]
  66   else:
  67     return [os.path.join(data_dir, 'test_batch.bin')]

  70 def parse_record(raw_record, is_training, dtype):
  71   """Parse CIFAR-10 image and label from a raw record."""
  72   # Convert bytes to a vector of uint8 that is record_bytes long.
  73   record_vector = tf.io.decode_raw(raw_record, tf.uint8)
  74 
  75   # The first byte represents the label, which we convert from uint8 to int32
  76   # and then to one-hot.
  77   label = tf.cast(record_vector[0], tf.int32)
  78 
  79   # The remaining bytes after the label represent the image, which we reshape
  80   # from [depth * height * width] to [depth, height, width].
  81   depth_major = tf.reshape(record_vector[1:_RECORD_BYTES],
  82                            [NUM_CHANNELS, HEIGHT, WIDTH])
  83 
  84   # Convert from [depth, height, width] to [height, width, depth], and cast as
  85   # float32.
  86   image = tf.cast(tf.transpose(a=depth_major, perm=[1, 2, 0]), tf.float32)
  87 
  88   image = preprocess_image(image, is_training)
  89   image = tf.cast(image, dtype)
  90 
  91   return image, label
  92

  94 def preprocess_image(image, is_training):
  95   """Preprocess a single image of layout [height, width, depth]."""
  96   if is_training:
  97     # Resize the image to add four extra pixels on each side.
  98     image = tf.image.resize_with_crop_or_pad(
  99         image, HEIGHT + 8, WIDTH + 8)
 100 
 101     # Randomly crop a [HEIGHT, WIDTH] section of the image.
 102     image = tf.image.random_crop(image, [HEIGHT, WIDTH, NUM_CHANNELS])
 103 
 104     # Randomly flip the image horizontally.
 105     image = tf.image.random_flip_left_right(image)
 106 
 107   # Subtract off the mean and divide by the variance of the pixels.
 108   image = tf.image.per_image_standardization(image)
 109   return image

 112 def input_fn(is_training,
 113              data_dir,
 114              batch_size,
 115              num_epochs=1,
 116              dtype=tf.float32,
 117              datasets_num_private_threads=None,
 118              parse_record_fn=parse_record,
 119              input_context=None,
 120              drop_remainder=False):
 121   """Input function which provides batches for train or eval.
 122 
 123   Args:
 124     is_training: A boolean denoting whether the input is for training.
 125     data_dir: The directory containing the input data.
 126     batch_size: The number of samples per batch.
 127     num_epochs: The number of epochs to repeat the dataset.
 128     dtype: Data type to use for images/features
 129     datasets_num_private_threads: Number of private threads for tf.data.
 130     parse_record_fn: Function to use for parsing the records.
 131     input_context: A `tf.distribute.InputContext` object passed in by
 132       `tf.distribute.Strategy`.
 133     drop_remainder: A boolean indicates whether to drop the remainder of the
 134       batches. If True, the batch dimension will be static.
 135 
 136   Returns:
 137     A dataset that can be used for iteration.
 138   """
 139   filenames = get_filenames(is_training, data_dir)
 140   dataset = tf.data.FixedLengthRecordDataset(filenames, _RECORD_BYTES)
 141 
 142   if input_context:
 143     logging.info(
 144         'Sharding the dataset: input_pipeline_id=%d num_input_pipelines=%d',
 145         input_context.input_pipeline_id, input_context.num_input_pipelines)
 146     dataset = dataset.shard(input_context.num_input_pipelines,
 147                             input_context.input_pipeline_id)
 148 
 149   return resnet_run_loop.process_record_dataset(
 150       dataset=dataset,
 151       is_training=is_training,
 152       batch_size=batch_size,
 153       shuffle_buffer=NUM_IMAGES['train'],
 154       parse_record_fn=parse_record_fn,
 155       num_epochs=num_epochs,
 156       dtype=dtype,
 157       datasets_num_private_threads=datasets_num_private_threads,
 158       drop_remainder=drop_remainder
 159   )

 170 class Cifar10Model(resnet_model.Model):
 171   """Model class with appropriate defaults for CIFAR-10 data."""
 172 
 173   def __init__(self, resnet_size, data_format=None, num_classes=NUM_CLASSES,
 174                resnet_version=resnet_model.DEFAULT_VERSION,
 175                dtype=resnet_model.DEFAULT_DTYPE):
 176     """These are the parameters that work for CIFAR-10 data.
 177 
 178     Args:
 179       resnet_size: The number of convolutional layers needed in the model.
 180       data_format: Either 'channels_first' or 'channels_last', specifying which
 181         data format to use when setting up the model.
 182       num_classes: The number of output classes needed from the model. This
 183         enables users to extend the same model to their own datasets.
 184       resnet_version: Integer representing which version of the ResNet network
 185       to use. See README for details. Valid values: [1, 2]
 186       dtype: The TensorFlow dtype to use for calculations.
 187 
 188     Raises:
 189       ValueError: if invalid resnet_size is chosen
 190     """
 191     if resnet_size % 6 != 2:
 192       raise ValueError('resnet_size must be 6n + 2:', resnet_size)
 193 
 194     num_blocks = (resnet_size - 2) // 6
 195 
 196     super(Cifar10Model, self).__init__(
 197         resnet_size=resnet_size,
 198         bottleneck=False,
 199         num_classes=num_classes,
 200         num_filters=16,
 201         kernel_size=3,
 202         conv_stride=1,
 203         first_pool_size=None,
 204         first_pool_stride=None,
 205         block_sizes=[num_blocks] * 3,
 206         block_strides=[1, 2, 2],
 207         resnet_version=resnet_version,
 208         data_format=data_format,
 209         dtype=dtype
 210     )

 213 def cifar10_model_fn(features, labels, mode, params):
 214   """Model function for CIFAR-10."""
 215   features = tf.reshape(features, [-1, HEIGHT, WIDTH, NUM_CHANNELS])
 216   # Learning rate schedule follows arXiv:1512.03385 for ResNet-56 and under.
 217   learning_rate_fn = resnet_run_loop.learning_rate_with_decay(
 218       batch_size=params['batch_size'] * params.get('num_workers', 1),
 219       batch_denom=128, num_images=NUM_IMAGES['train'],
 220       boundary_epochs=[91, 136, 182], decay_rates=[1, 0.1, 0.01, 0.001])
 221 
 222   # Weight decay of 2e-4 diverges from 1e-4 decay used in the ResNet paper
 223   # and seems more stable in testing. The difference was nominal for ResNet-56.
 224   weight_decay = 2e-4
 225 
 226   # Empirical testing showed that including batch_normalization variables
 227   # in the calculation of regularized loss helped validation accuracy
 228   # for the CIFAR-10 dataset, perhaps because the regularization prevents
 229   # overfitting on the small data set. We therefore include all vars when
 230   # regularizing and computing loss during training.
 231   def loss_filter_fn(_):
 232     return True
 233 
 234   return resnet_run_loop.resnet_model_fn(
 235       features=features,
 236       labels=labels,
 237       mode=mode,
 238       model_class=Cifar10Model,
 239       resnet_size=params['resnet_size'],
 240       weight_decay=weight_decay,
 241       learning_rate_fn=learning_rate_fn,
 242       momentum=0.9,
 243       data_format=params['data_format'],
 244       resnet_version=params['resnet_version'],
 245       loss_scale=params['loss_scale'],
 246       loss_filter_fn=loss_filter_fn,
 247       dtype=params['dtype'],
 248       fine_tune=params['fine_tune']
 249   )

 264 def run_cifar(flags_obj):
 265   """Run ResNet CIFAR-10 training and eval loop.
 266 
 267   Args:
 268     flags_obj: An object containing parsed flag values.
 269 
 270   Returns:
 271     Dictionary of results. Including final accuracy.
 272   """
 273   if flags_obj.image_bytes_as_serving_input:
 274     logging.fatal(
 275         '--image_bytes_as_serving_input cannot be set to True for CIFAR. '
 276         'This flag is only applicable to ImageNet.')
 277     return
 278 
 279   input_function = (flags_obj.use_synthetic_data and
 280                     get_synth_input_fn(flags_core.get_tf_dtype(flags_obj)) or
 281                     input_fn)
 282   result = resnet_run_loop.resnet_main(
 283       flags_obj, cifar10_model_fn, input_function, DATASET_NAME,
 284       shape=[HEIGHT, WIDTH, NUM_CHANNELS])
 285 
 286   return result
