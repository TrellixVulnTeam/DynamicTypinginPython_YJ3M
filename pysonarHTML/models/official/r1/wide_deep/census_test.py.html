  50 class BaseTest(tf.test.TestCase):
  51   """Tests for Wide Deep model."""

  53   @classmethod
  54   def setUpClass(cls):  # pylint: disable=invalid-name
  55     super(BaseTest, cls).setUpClass()
  56     census_main.define_census_flags()

  58   def setUp(self):
  59     # Create temporary CSV file
  60     self.temp_dir = self.get_temp_dir()
  61     self.input_csv = os.path.join(self.temp_dir, 'test.csv')
  62     with tf.io.gfile.GFile(self.input_csv, 'w') as temp_csv:
  63       temp_csv.write(TEST_INPUT)
  64 
  65     with tf.io.gfile.GFile(TEST_CSV, 'r') as temp_csv:
  66       test_csv_contents = temp_csv.read()
  67 
  68     # Used for end-to-end tests.
  69     for fname in [census_dataset.TRAINING_FILE, census_dataset.EVAL_FILE]:
  70       with tf.io.gfile.GFile(
  71           os.path.join(self.temp_dir, fname), 'w') as test_csv:
  72         test_csv.write(test_csv_contents)
  73 
  74   def test_input_fn(self):
  75     dataset = census_dataset.input_fn(self.input_csv, 1, False, 1)
  76     features, labels = dataset.make_one_shot_iterator().get_next()
  77 
  78     with self.test_session() as sess:
  79       features, labels = sess.run((features, labels))
  80 
  81       # Compare the two features dictionaries.
  82       for key in TEST_INPUT_VALUES:
  83         self.assertTrue(key in features)
  84         self.assertEqual(len(features[key]), 1)
  85         feature_value = features[key][0]
  86 
  87         # Convert from bytes to string for Python 3.
  88         if isinstance(feature_value, bytes):
  89           feature_value = feature_value.decode()
  90 
  91         self.assertEqual(TEST_INPUT_VALUES[key], feature_value)
  92 
  93       self.assertFalse(labels)
  94 
  95   def build_and_test_estimator(self, model_type):
  96     """Ensure that model trains and minimizes loss."""
  97     model = census_main.build_estimator(
  98         self.temp_dir, model_type,
  99         model_column_fn=census_dataset.build_model_columns,
 100         inter_op=0, intra_op=0)
 101 
 102     # Train for 1 step to initialize model and evaluate initial loss
 103     def get_input_fn(num_epochs, shuffle, batch_size):
 104       def input_fn():
 105         return census_dataset.input_fn(
 106             TEST_CSV, num_epochs=num_epochs, shuffle=shuffle,
 107             batch_size=batch_size)
 108       return input_fn
 109 
 110     model.train(input_fn=get_input_fn(1, True, 1), steps=1)
 111     initial_results = model.evaluate(input_fn=get_input_fn(1, False, 1))
 112 
 113     # Train for 100 epochs at batch size 3 and evaluate final loss
 114     model.train(input_fn=get_input_fn(100, True, 3))
 115     final_results = model.evaluate(input_fn=get_input_fn(1, False, 1))
 116 
 117     print('%s initial results:' % model_type, initial_results)
 118     print('%s final results:' % model_type, final_results)
 119 
 120     # Ensure loss has decreased, while accuracy and both AUCs have increased.
 121     self.assertLess(final_results['loss'], initial_results['loss'])
 122     self.assertGreater(final_results['auc'], initial_results['auc'])
 123     self.assertGreater(final_results['auc_precision_recall'],
 124                        initial_results['auc_precision_recall'])
 125     self.assertGreater(final_results['accuracy'], initial_results['accuracy'])
