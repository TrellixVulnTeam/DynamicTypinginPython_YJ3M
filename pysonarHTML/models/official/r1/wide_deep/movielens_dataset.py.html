  36 _BUFFER_SUBDIR = "wide_deep_buffer"
  37 _FEATURE_MAP = {
  38     movielens.USER_COLUMN: tf.compat.v1.FixedLenFeature([1], dtype=tf.int64),
  39     movielens.ITEM_COLUMN: tf.compat.v1.FixedLenFeature([1], dtype=tf.int64),
  40     movielens.TIMESTAMP_COLUMN: tf.compat.v1.FixedLenFeature([1],
  41                                                              dtype=tf.int64),
  42     movielens.GENRE_COLUMN: tf.compat.v1.FixedLenFeature(
  43         [movielens.N_GENRE], dtype=tf.int64),
  44     movielens.RATING_COLUMN: tf.compat.v1.FixedLenFeature([1],
  45                                                           dtype=tf.float32),
  46 }
  47 
  48 _BUFFER_SIZE = {
  49     movielens.ML_1M: {"train": 107978119, "eval": 26994538},
  50     movielens.ML_20M: {"train": 2175203810, "eval": 543802008}
  51 }

  53 _USER_EMBEDDING_DIM = 16
  54 _ITEM_EMBEDDING_DIM = 64
  55 
  56 def build_model_columns(dataset):
  57   """Builds a set of wide and deep feature columns."""
  58   user_id = tf.feature_column.categorical_column_with_vocabulary_list(
  59       movielens.USER_COLUMN, range(1, movielens.NUM_USER_IDS[dataset]))
  60   user_embedding = tf.feature_column.embedding_column(
  61       user_id, _USER_EMBEDDING_DIM, max_norm=np.sqrt(_USER_EMBEDDING_DIM))
  62 
  63   item_id = tf.feature_column.categorical_column_with_vocabulary_list(
  64       movielens.ITEM_COLUMN, range(1, movielens.NUM_ITEM_IDS))
  65   item_embedding = tf.feature_column.embedding_column(
  66       item_id, _ITEM_EMBEDDING_DIM, max_norm=np.sqrt(_ITEM_EMBEDDING_DIM))
  67 
  68   time = tf.feature_column.numeric_column(movielens.TIMESTAMP_COLUMN)
  69   genres = tf.feature_column.numeric_column(
  70       movielens.GENRE_COLUMN, shape=(movielens.N_GENRE,), dtype=tf.uint8)
  71 
  72   deep_columns = [user_embedding, item_embedding, time, genres]
  73   wide_columns = []
  74 
  75   return wide_columns, deep_columns

  78 def _deserialize(examples_serialized):
  79   features = tf.parse_example(examples_serialized, _FEATURE_MAP)
  80   return features, features[movielens.RATING_COLUMN] / movielens.MAX_RATING

  83 def _buffer_path(data_dir, dataset, name):
  84   return os.path.join(data_dir, _BUFFER_SUBDIR,
  85                       "{}_{}_buffer".format(dataset, name))
  86 
  87 
  88 def _df_to_input_fn(df, name, dataset, data_dir, batch_size, repeat, shuffle):
  89   """Serialize a dataframe and write it to a buffer file."""
  90   buffer_path = _buffer_path(data_dir, dataset, name)
  91   expected_size = _BUFFER_SIZE[dataset].get(name)
  92 
  93   file_io.write_to_buffer(
  94       dataframe=df, buffer_path=buffer_path,
  95       columns=list(_FEATURE_MAP.keys()), expected_size=expected_size)
  96 
  97   def input_fn():
  98     dataset = tf.data.TFRecordDataset(buffer_path)
  99     # batch comes before map because map can deserialize multiple examples.
 100     dataset = dataset.batch(batch_size)
 101     dataset = dataset.map(_deserialize, num_parallel_calls=16)
 102     if shuffle:
 103       dataset = dataset.shuffle(shuffle)
 104 
 105     dataset = dataset.repeat(repeat)
 106     return dataset.prefetch(1)
 107 
 108   return input_fn

 111 def _check_buffers(data_dir, dataset):
 112   train_path = os.path.join(data_dir, _BUFFER_SUBDIR,
 113                             "{}_{}_buffer".format(dataset, "train"))
 114   eval_path = os.path.join(data_dir, _BUFFER_SUBDIR,
 115                            "{}_{}_buffer".format(dataset, "eval"))
 116 
 117   if not tf.gfile.Exists(train_path) or not tf.gfile.Exists(eval_path):
 118     return False
 119 
 120   return all([
 121       tf.gfile.Stat(_buffer_path(data_dir, dataset, "train")).length ==
 122       _BUFFER_SIZE[dataset]["train"],
 123       tf.gfile.Stat(_buffer_path(data_dir, dataset, "eval")).length ==
 124       _BUFFER_SIZE[dataset]["eval"],
 125   ])

 128 def construct_input_fns(dataset, data_dir, batch_size=16, repeat=1):
 129   """Construct train and test input functions, as well as the column fn."""
 130   if _check_buffers(data_dir, dataset):
 131     train_df, eval_df = None, None
 132   else:
 133     df = movielens.csv_to_joint_dataframe(dataset=dataset, data_dir=data_dir)
 134     df = movielens.integerize_genres(dataframe=df)
 135     df = df.drop(columns=[movielens.TITLE_COLUMN])
 136 
 137     train_df = df.sample(frac=0.8, random_state=0)
 138     eval_df = df.drop(train_df.index)
 139 
 140     train_df = train_df.reset_index(drop=True)
 141     eval_df = eval_df.reset_index(drop=True)
 142 
 143   train_input_fn = _df_to_input_fn(
 144       df=train_df, name="train", dataset=dataset, data_dir=data_dir,
 145       batch_size=batch_size, repeat=repeat,
 146       shuffle=movielens.NUM_RATINGS[dataset])
 147   eval_input_fn = _df_to_input_fn(
 148       df=eval_df, name="eval", dataset=dataset, data_dir=data_dir,
 149       batch_size=batch_size, repeat=repeat, shuffle=None)
 150   model_column_fn = functools.partial(build_model_columns, dataset=dataset)
 151 
 152   train_input_fn()
 153   return train_input_fn, eval_input_fn, model_column_fn
