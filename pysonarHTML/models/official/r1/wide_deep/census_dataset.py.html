  60 def _download_and_clean_file(filename, url):
  61   """Downloads data from url, and makes changes to match the CSV format."""
  62   temp_file, _ = urllib.request.urlretrieve(url)
  63   with tf.gfile.Open(temp_file, 'r') as temp_eval_file:
  64     with tf.gfile.Open(filename, 'w') as eval_file:
  65       for line in temp_eval_file:
  66         line = line.strip()
  67         line = line.replace(', ', ',')
  68         if not line or ',' not in line:
  69           continue
  70         if line[-1] == '.':
  71           line = line[:-1]
  72         line += '\n'
  73         eval_file.write(line)
  74   tf.gfile.Remove(temp_file)

  77 def download(data_dir):
  78   """Download census data if it is not already present."""
  79   tf.gfile.MakeDirs(data_dir)
  80 
  81   training_file_path = os.path.join(data_dir, TRAINING_FILE)
  82   if not tf.gfile.Exists(training_file_path):
  83     _download_and_clean_file(training_file_path, TRAINING_URL)
  84 
  85   eval_file_path = os.path.join(data_dir, EVAL_FILE)
  86   if not tf.gfile.Exists(eval_file_path):
  87     _download_and_clean_file(eval_file_path, EVAL_URL)

  90 def build_model_columns():
  91   """Builds a set of wide and deep feature columns."""
  92   # Continuous variable columns
  93   age = tf.feature_column.numeric_column('age')
  94   education_num = tf.feature_column.numeric_column('education_num')
  95   capital_gain = tf.feature_column.numeric_column('capital_gain')
  96   capital_loss = tf.feature_column.numeric_column('capital_loss')
  97   hours_per_week = tf.feature_column.numeric_column('hours_per_week')
  98 
  99   education = tf.feature_column.categorical_column_with_vocabulary_list(
 100       'education', [
 101           'Bachelors', 'HS-grad', '11th', 'Masters', '9th', 'Some-college',
 102           'Assoc-acdm', 'Assoc-voc', '7th-8th', 'Doctorate', 'Prof-school',
 103           '5th-6th', '10th', '1st-4th', 'Preschool', '12th'])
 104 
 105   marital_status = tf.feature_column.categorical_column_with_vocabulary_list(
 106       'marital_status', [
 107           'Married-civ-spouse', 'Divorced', 'Married-spouse-absent',
 108           'Never-married', 'Separated', 'Married-AF-spouse', 'Widowed'])
 109 
 110   relationship = tf.feature_column.categorical_column_with_vocabulary_list(
 111       'relationship', [
 112           'Husband', 'Not-in-family', 'Wife', 'Own-child', 'Unmarried',
 113           'Other-relative'])
 114 
 115   workclass = tf.feature_column.categorical_column_with_vocabulary_list(
 116       'workclass', [
 117           'Self-emp-not-inc', 'Private', 'State-gov', 'Federal-gov',
 118           'Local-gov', '?', 'Self-emp-inc', 'Without-pay', 'Never-worked'])
 119 
 120   # To show an example of hashing:
 121   occupation = tf.feature_column.categorical_column_with_hash_bucket(
 122       'occupation', hash_bucket_size=_HASH_BUCKET_SIZE)
 123 
 124   # Transformations.
 125   age_buckets = tf.feature_column.bucketized_column(
 126       age, boundaries=[18, 25, 30, 35, 40, 45, 50, 55, 60, 65])
 127 
 128   # Wide columns and deep columns.
 129   base_columns = [
 130       education, marital_status, relationship, workclass, occupation,
 131       age_buckets,
 132   ]
 133 
 134   crossed_columns = [
 135       tf.feature_column.crossed_column(
 136           ['education', 'occupation'], hash_bucket_size=_HASH_BUCKET_SIZE),
 137       tf.feature_column.crossed_column(
 138           [age_buckets, 'education', 'occupation'],
 139           hash_bucket_size=_HASH_BUCKET_SIZE),
 140   ]
 141 
 142   wide_columns = base_columns + crossed_columns
 143 
 144   deep_columns = [
 145       age,
 146       education_num,
 147       capital_gain,
 148       capital_loss,
 149       hours_per_week,
 150       tf.feature_column.indicator_column(workclass),
 151       tf.feature_column.indicator_column(education),
 152       tf.feature_column.indicator_column(marital_status),
 153       tf.feature_column.indicator_column(relationship),
 154       # To show an example of embedding
 155       tf.feature_column.embedding_column(occupation, dimension=8),
 156   ]
 157 
 158   return wide_columns, deep_columns

 161 def input_fn(data_file, num_epochs, shuffle, batch_size):
 162   """Generate an input function for the Estimator."""
 163   assert tf.gfile.Exists(data_file), (
 164       '%s not found. Please make sure you have run census_dataset.py and '
 165       'set the --data_dir argument to the correct path.' % data_file)
 166 
 167   def parse_csv(value):
 168     tf.logging.info('Parsing {}'.format(data_file))
 169     columns = tf.decode_csv(value, record_defaults=_CSV_COLUMN_DEFAULTS)
 170     features = dict(list(zip(_CSV_COLUMNS, columns)))
 171     labels = features.pop('income_bracket')
 172     classes = tf.equal(labels, '>50K')  # binary classification
 173     return features, classes
 174 
 175   # Extract lines from input files using the Dataset API.
 176   dataset = tf.data.TextLineDataset(data_file)
 177 
 178   if shuffle:
 179     dataset = dataset.shuffle(buffer_size=_NUM_EXAMPLES['train'])
 180 
 181   dataset = dataset.map(parse_csv, num_parallel_calls=5)
 182 
 183   # We call repeat after shuffling, rather than before, to prevent separate
 184   # epochs from blending together.
 185   dataset = dataset.repeat(num_epochs)
 186   dataset = dataset.batch(batch_size)
 187   return dataset
