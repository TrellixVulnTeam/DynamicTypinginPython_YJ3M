  15 """Sets up TensorFlow Official Models."""
  16 import datetime
  17 import os
  18 import sys
  19 
  20 from setuptools import find_packages
  21 from setuptools import setup
  22 
  23 version = '2.2.0'
  24 
  25 project_name = 'tf-models-official'
  26 
  27 long_description = """The TensorFlow official models are a collection of
  28 models that use TensorFlow's high-level APIs.
  29 They are intended to be well-maintained, tested, and kept up to date with the
  30 latest TensorFlow API. They should also be reasonably optimized for fast
  31 performance while still being easy to read."""
  32 
  33 if '--project_name' in sys.argv:
  34   project_name_idx = sys.argv.index('--project_name')
  35   project_name = sys.argv[project_name_idx + 1]
  36   sys.argv.remove('--project_name')
  37   sys.argv.pop(project_name_idx)
  38 
  39 
  40 def _get_requirements():
  41   """Parses requirements.txt file."""
  42   install_requires_tmp = []
  43   dependency_links_tmp = []
  44   with open(
  45       os.path.join(os.path.dirname(__file__), '../requirements.txt'), 'r') as f:
  46     for line in f:
  47       package_name = line.strip()
  48       if package_name.startswith('-e '):
  49         dependency_links_tmp.append(package_name[3:].strip())
  50       else:
  51         install_requires_tmp.append(package_name)
  52   return install_requires_tmp, dependency_links_tmp
  53 
  54 install_requires, dependency_links = _get_requirements()
  55 
  56 if project_name == 'tf-models-nightly':
  57   version += '.dev' + datetime.datetime.now().strftime('%Y%m%d')
  58   install_requires.append('tf-nightly')
  59 else:
  60   install_requires.append('tensorflow>=2.2.0')
  61 
  62 print('install_requires: ', install_requires)
  63 print('dependency_links: ', dependency_links)
  64 
  65 setup(
  66     name=project_name,
  67     version=version,
  68     description='TensorFlow Official Models',
  69     long_description=long_description,
  70     author='Google Inc.',
  71     author_email='no-reply@google.com',
  72     url='https://github.com/tensorflow/models',
  73     license='Apache 2.0',
  74     packages=find_packages(exclude=[
  75         'research*',
  76         'tutorials*',
  77         'samples*',
  78         'official.r1*',
  79         'official.pip_package*',
  80         'official.benchmark*',
  81         'official.colab*',
  82     ]),
  83     exclude_package_data={
  84         '': ['*_test.py',],
  85     },
  86     install_requires=install_requires,
  87     dependency_links=dependency_links,
  88     python_requires='>=3.6',
  89 )
