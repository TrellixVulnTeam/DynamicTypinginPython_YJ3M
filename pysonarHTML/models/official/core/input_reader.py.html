  16 """A common dataset reader."""
  17 
  18 from typing import Any, Callable, List, Optional
  19 
  20 import tensorflow as tf
  21 import tensorflow_datasets as tfds
  22 
  23 from official.modeling.hyperparams import config_definitions as cfg
  24 
  25 
  26 class InputReader:
  27   """Input reader that returns a tf.data.Dataset instance."""
  28 
  29   def __init__(self,
  30                params: cfg.DataConfig,
  31                shards: Optional[List[str]] = None,
  32                dataset_fn=tf.data.TFRecordDataset,
  33                decoder_fn: Optional[Callable[..., Any]] = None,
  34                parser_fn: Optional[Callable[..., Any]] = None,
  35                dataset_transform_fn: Optional[Callable[[tf.data.Dataset],
  36                                                        tf.data.Dataset]] = None,
  37                postprocess_fn: Optional[Callable[..., Any]] = None):
  38     """Initializes an InputReader instance.
  39 
  40     Args:
  41       params: A config_definitions.DataConfig object.
  42       shards: A list of files to be read. If given, read from these files.
  43         Otherwise, read from params.input_path.
  44       dataset_fn: A `tf.data.Dataset` that consumes the input files. For
  45         example, it can be `tf.data.TFRecordDataset`.
  46       decoder_fn: An optional `callable` that takes the serialized data string
  47         and decodes them into the raw tensor dictionary.
  48       parser_fn: An optional `callable` that takes the decoded raw tensors dict
  49         and parse them into a dictionary of tensors that can be consumed by the
  50         model. It will be executed after decoder_fn.
  51       dataset_transform_fn: An optional `callable` that takes a
  52         `tf.data.Dataset` object and returns a `tf.data.Dataset`. It will be
  53         executed after parser_fn.
  54       postprocess_fn: A optional `callable` that processes batched tensors. It
  55         will be executed after batching.
  56     """
  57     if params.input_path and params.tfds_name:
  58       raise ValueError('At most one of `input_path` and `tfds_name` can be '
  59                        'specified, but got %s and %s.' % (
  60                            params.input_path, params.tfds_name))
  61     self._shards = shards
  62     self._tfds_builder = None
  63     if self._shards:
  64       self._num_files = len(self._shards)
  65     elif not params.tfds_name:
  66       self._input_patterns = params.input_path.strip().split(',')
  67       self._num_files = 0
  68       for input_pattern in self._input_patterns:
  69         input_pattern = input_pattern.strip()
  70         if not input_pattern:
  71           continue
  72         matched_files = tf.io.gfile.glob(input_pattern)
  73         if not matched_files:
  74           raise ValueError('%s does not match any files.' % input_pattern)
  75         else:
  76           self._num_files += len(matched_files)
  77       if self._num_files == 0:
  78         raise ValueError('%s does not match any files.' % params.input_path)
  79     else:
  80       if not params.tfds_split:
  81         raise ValueError(
  82             '`tfds_name` is %s, but `tfds_split` is not specified.' %
  83             params.tfds_name)
  84       self._tfds_builder = tfds.builder(
  85           params.tfds_name, data_dir=params.tfds_data_dir)
  86 
  87     self._global_batch_size = params.global_batch_size
  88     self._is_training = params.is_training
  89     self._drop_remainder = params.drop_remainder
  90     self._shuffle_buffer_size = params.shuffle_buffer_size
  91     self._cache = params.cache
  92     self._cycle_length = params.cycle_length
  93     self._block_length = params.block_length
  94     self._sharding = params.sharding
  95     self._examples_consume = params.examples_consume
  96     self._tfds_split = params.tfds_split
  97     self._tfds_download = params.tfds_download
  98     self._tfds_as_supervised = params.tfds_as_supervised
  99     self._tfds_skip_decoding_feature = params.tfds_skip_decoding_feature
 100 
 101     self._dataset_fn = dataset_fn
 102     self._decoder_fn = decoder_fn
 103     self._parser_fn = parser_fn
 104     self._dataset_transform_fn = dataset_transform_fn
 105     self._postprocess_fn = postprocess_fn
 106 
 107   def _read_sharded_files(
 108       self,
 109       input_context: Optional[tf.distribute.InputContext] = None):
 110     """Reads a dataset from sharded files."""
 111     # Read from `self._shards` if it is provided.
 112     if self._shards:
 113       dataset = tf.data.Dataset.from_tensor_slices(self._shards)
 114     else:
 115       dataset = tf.data.Dataset.list_files(
 116           self._input_patterns, shuffle=self._is_training)
 117     if self._sharding and input_context and (
 118         input_context.num_input_pipelines > 1):
 119       dataset = dataset.shard(input_context.num_input_pipelines,
 120                               input_context.input_pipeline_id)
 121     if self._is_training:
 122       dataset = dataset.repeat()
 123 
 124     dataset = dataset.interleave(
 125         map_func=self._dataset_fn,
 126         cycle_length=self._cycle_length,
 127         block_length=self._block_length,
 128         num_parallel_calls=tf.data.experimental.AUTOTUNE)
 129     return dataset
 130 
 131   def _read_single_file(
 132       self,
 133       input_context: Optional[tf.distribute.InputContext] = None):
 134     """Reads a dataset from a single file."""
 135     # Read from `self._shards` if it is provided.
 136     dataset = self._dataset_fn(self._shards or self._input_patterns)
 137 
 138     # When `input_file` is a path to a single file, disable auto sharding
 139     # so that same input file is sent to all workers.
 140     options = tf.data.Options()
 141     options.experimental_distribute.auto_shard_policy = (
 142         tf.data.experimental.AutoShardPolicy.OFF)
 143     dataset = dataset.with_options(options)
 144     if self._sharding and input_context and (
 145         input_context.num_input_pipelines > 1):
 146       dataset = dataset.shard(input_context.num_input_pipelines,
 147                               input_context.input_pipeline_id)
 148     if self._is_training:
 149       dataset = dataset.repeat()
 150     return dataset
 151 
 152   def _read_tfds(
 153       self,
 154       input_context: Optional[tf.distribute.InputContext] = None
 155   ) -> tf.data.Dataset:
 156     """Reads a dataset from tfds."""
 157     if self._tfds_download:
 158       self._tfds_builder.download_and_prepare()
 159 
 160     read_config = tfds.ReadConfig(
 161         interleave_cycle_length=self._cycle_length,
 162         interleave_block_length=self._block_length,
 163         input_context=input_context)
 164     decoders = {}
 165     if self._tfds_skip_decoding_feature:
 166       for skip_feature in self._tfds_skip_decoding_feature.split(','):
 167         decoders[skip_feature.strip()] = tfds.decode.SkipDecoding()
 168     dataset = self._tfds_builder.as_dataset(
 169         split=self._tfds_split,
 170         shuffle_files=self._is_training,
 171         as_supervised=self._tfds_as_supervised,
 172         decoders=decoders,
 173         read_config=read_config)
 174     return dataset
 175 
 176   @property
 177   def tfds_info(self) -> tfds.core.DatasetInfo:
 178     """Returns TFDS dataset info, if available."""
 179     if self._tfds_builder:
 180       return self._tfds_builder.info
 181     else:
 182       raise ValueError('tfds_info is not available, because the dataset '
 183                        'is not loaded from tfds.')
 184 
 185   def read(
 186       self,
 187       input_context: Optional[tf.distribute.InputContext] = None
 188   ) -> tf.data.Dataset:
 189     """Generates a tf.data.Dataset object."""
 190     if self._tfds_builder:
 191       dataset = self._read_tfds(input_context)
 192     elif self._num_files > 1:
 193       dataset = self._read_sharded_files(input_context)
 194     else:
 195       assert self._num_files == 1
 196       dataset = self._read_single_file(input_context)
 197 
 198     if self._cache:
 199       dataset = dataset.cache()
 200 
 201     if self._is_training:
 202       dataset = dataset.shuffle(self._shuffle_buffer_size)
 203 
 204     if self._examples_consume > 0:
 205       dataset = dataset.take(self._examples_consume)
 206 
 207     def maybe_map_fn(dataset, fn):
 208       return dataset if fn is None else dataset.map(
 209           fn, num_parallel_calls=tf.data.experimental.AUTOTUNE)
 210 
 211     dataset = maybe_map_fn(dataset, self._decoder_fn)
 212     dataset = maybe_map_fn(dataset, self._parser_fn)
 213 
 214     if self._dataset_transform_fn is not None:
 215       dataset = self._dataset_transform_fn(dataset)
 216 
 217     per_replica_batch_size = input_context.get_per_replica_batch_size(
 218         self._global_batch_size) if input_context else self._global_batch_size
 219 
 220     dataset = dataset.batch(
 221         per_replica_batch_size, drop_remainder=self._drop_remainder)
 222     dataset = maybe_map_fn(dataset, self._postprocess_fn)
 223     return dataset.prefetch(tf.data.experimental.AUTOTUNE)
