  28 @six.add_metaclass(abc.ABCMeta)
  29 class Task(tf.Module):
  30   """A single-replica view of training procedure.
  31 
  32   Tasks provide artifacts for training/evalution procedures, including
  33   loading/iterating over Datasets, initializing the model, calculating the loss
  34   and customized metrics with reduction.
  35   """
  36 
  37   # Special keys in train/validate step returned logs.
  38   loss = "loss"
  39 
  40   def __init__(self, params: cfg.TaskConfig):
  41     self._task_config = params
  42 
  43   @property
  44   def task_config(self) -> cfg.TaskConfig:
  45     return self._task_config

  52     will not be called. You can use this callback function to load a pretrained
  53     checkpoint, saved under a directory other than the model_dir.
  54 
  55     Args:
  56       model: The keras.Model built or used by this task.
  57     """
  58     pass

  68   def compile_model(self,
  69                     model: tf.keras.Model,
  70                     optimizer: tf.keras.optimizers.Optimizer,
  71                     loss=None,
  72                     train_step: Optional[Callable[..., Any]] = None,
  73                     validation_step: Optional[Callable[..., Any]] = None,
  74                     **kwargs) -> tf.keras.Model:
  75     """Compiles the model with objects created by the task.
  76 
  77     The method should not be used in any customized training implementation.
  78 
  79     Args:
  80       model: a keras.Model.
  81       optimizer: the keras optimizer.
  82       loss: a callable/list of losses.
  83       train_step: optional train step function defined by the task.
  84       validation_step: optional validation_step step function defined by the
  85         task.
  86       **kwargs: other kwargs consumed by keras.Model compile().
  87 
  88     Returns:
  89       a compiled keras.Model.
  90     """
  91     if bool(loss is None) == bool(train_step is None):
  92       raise ValueError("`loss` and `train_step` should be exclusive to "
  93                        "each other.")
  94     model.compile(optimizer=optimizer, loss=loss, **kwargs)
  95 
  96     if train_step:
  97       model.train_step = functools.partial(
  98           train_step, model=model, optimizer=model.optimizer)
  99     if validation_step:
 100       model.test_step = functools.partial(validation_step, model=model)
 101     return model

 119   def build_losses(self, labels, model_outputs, aux_losses=None) -> tf.Tensor:
 120     """Standard interface to compute losses.
 121 
 122     Args:
 123       labels: optional label tensors.
 124       model_outputs: a nested structure of output tensors.
 125       aux_losses: auxiliarly loss tensors, i.e. `losses` in keras.Model.
 126 
 127     Returns:
 128       The total loss tensor.
 129     """
 130     del model_outputs, labels
 131 
 132     if aux_losses is None:
 133       losses = [tf.constant(0.0, dtype=tf.float32)]
 134     else:
 135       losses = aux_losses
 136     total_loss = tf.add_n(losses)
 137     return total_loss

 144   def process_metrics(self, metrics, labels, model_outputs):
 145     """Process and update metrics. Called when using custom training loop API.
 146 
 147     Args:
 148       metrics: a nested structure of metrics objects.
 149         The return of function self.build_metrics.
 150       labels: a tensor or a nested structure of tensors.
 151       model_outputs: a tensor or a nested structure of tensors.
 152         For example, output of the keras model built by self.build_model.
 153     """
 154     for metric in metrics:
 155       metric.update_state(labels, model_outputs)

 157   def process_compiled_metrics(self, compiled_metrics, labels, model_outputs):
 158     """Process and update compiled_metrics. call when using compile/fit API.
 159 
 160     Args:
 161       compiled_metrics: the compiled metrics (model.compiled_metrics).
 162       labels: a tensor or a nested structure of tensors.
 163       model_outputs: a tensor or a nested structure of tensors.
 164         For example, output of the keras model built by self.build_model.
 165     """
 166     compiled_metrics.update_state(labels, model_outputs)

 168   def train_step(self,
 169                  inputs,
 170                  model: tf.keras.Model,
 171                  optimizer: tf.keras.optimizers.Optimizer,
 172                  metrics=None):
 173     """Does forward and backward.
 174 
 175     Args:
 176       inputs: a dictionary of input tensors.
 177       model: the model, forward pass definition.
 178       optimizer: the optimizer for this training step.
 179       metrics: a nested structure of metrics objects.
 180 
 181     Returns:
 182       A dictionary of logs.
 183     """
 184     if isinstance(inputs, tuple) and len(inputs) == 2:
 185       features, labels = inputs
 186     else:
 187       features, labels = inputs, inputs
 188     with tf.GradientTape() as tape:
 189       outputs = model(features, training=True)
 190       # Computes per-replica loss.
 191       loss = self.build_losses(
 192           labels=labels, model_outputs=outputs, aux_losses=model.losses)
 193       # Scales loss as the default gradients allreduce performs sum inside the
 194       # optimizer.
 195       scaled_loss = loss / tf.distribute.get_strategy().num_replicas_in_sync
 196 
 197       # For mixed precision, when a LossScaleOptimizer is used, the loss is
 198       # scaled to avoid numeric underflow.
 199       if isinstance(optimizer,
 200                     tf.keras.mixed_precision.experimental.LossScaleOptimizer):
 201         scaled_loss = optimizer.get_scaled_loss(scaled_loss)
 202 
 203     tvars = model.trainable_variables
 204     grads = tape.gradient(scaled_loss, tvars)
 205 
 206     if isinstance(optimizer,
 207                   tf.keras.mixed_precision.experimental.LossScaleOptimizer):
 208       grads = optimizer.get_unscaled_gradients(grads)
 209     optimizer.apply_gradients(list(zip(grads, tvars)))
 210     logs = {self.loss: loss}
 211     if metrics:
 212       self.process_metrics(metrics, labels, outputs)
 213       logs.update({m.name: m.result() for m in metrics})
 214     elif model.compiled_metrics:
 215       self.process_compiled_metrics(model.compiled_metrics, labels, outputs)
 216       logs.update({m.name: m.result() for m in model.metrics})
 217     return logs
 218 
 219   def validation_step(self, inputs, model: tf.keras.Model, metrics=None):
 220     """Validatation step.
 221 
 222     Args:
 223       inputs: a dictionary of input tensors.
 224       model: the keras.Model.
 225       metrics: a nested structure of metrics objects.
 226 
 227     Returns:
 228       A dictionary of logs.
 229     """
 230     if isinstance(inputs, tuple) and len(inputs) == 2:
 231       features, labels = inputs
 232     else:
 233       features, labels = inputs, inputs
 234     outputs = self.inference_step(features, model)
 235     loss = self.build_losses(
 236         labels=labels, model_outputs=outputs, aux_losses=model.losses)
 237     logs = {self.loss: loss}
 238     if metrics:
 239       self.process_metrics(metrics, labels, outputs)
 240       logs.update({m.name: m.result() for m in metrics})
 241     elif model.compiled_metrics:
 242       self.process_compiled_metrics(model.compiled_metrics, labels, outputs)
 243       logs.update({m.name: m.result() for m in model.metrics})
 244     return logs
 245 
 246   def inference_step(self, inputs, model: tf.keras.Model):
 247     """Performs the forward step."""
 248     return model(inputs, training=False)

 254 # TODO(b/158268740): Move these outside the base class file.

 256 def register_task_cls(task_config_cls):
 257   """Decorates a factory of Tasks for lookup by a subclass of TaskConfig.
 258 
 259   This decorator supports registration of tasks as follows:
 260 
 261   ```
 262   @dataclasses.dataclass
 263   class MyTaskConfig(TaskConfig):
 264     # Add fields here.
 265     pass
 266 
 267   @register_task_cls(MyTaskConfig)
 268   class MyTask(Task):
 269     # Inherits def __init__(self, task_config).
 270     pass
 271 
 272   my_task_config = MyTaskConfig()
 273   my_task = get_task(my_task_config)  # Returns MyTask(my_task_config).
 274   ```
 275 
 276   Besisdes a class itself, other callables that create a Task from a TaskConfig
 277   can be decorated by the result of this function, as long as there is at most
 278   one registration for each config class.
 279 
 280   Args:
 281     task_config_cls: a subclass of TaskConfig (*not* an instance of TaskConfig).
 282       Each task_config_cls can only be used for a single registration.
 283 
 284   Returns:
 285     A callable for use as class decorator that registers the decorated class
 286     for creation from an instance of task_config_cls.
 287   """
 288   return registry.register(_REGISTERED_TASK_CLS, task_config_cls)

 291 # The user-visible get_task() is defined after classes have been registered.
 292 # TODO(b/158741360): Add type annotations once pytype checks across modules.
 293 def get_task_cls(task_config_cls):
 294   task_cls = registry.lookup(_REGISTERED_TASK_CLS, task_config_cls)
 295   return task_cls
