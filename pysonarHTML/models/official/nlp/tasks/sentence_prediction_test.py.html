  16 """Tests for official.nlp.tasks.sentence_prediction."""
  17 import functools

  21 from official.nlp.bert import configs

  28 class SentencePredictionTaskTest(tf.test.TestCase):
  29 
  30   def _run_task(self, config):
  31     task = sentence_prediction.SentencePredictionTask(config)
  32     model = task.build_model()
  33     metrics = task.build_metrics()
  34 
  35     strategy = tf.distribute.get_strategy()
  36     dataset = strategy.experimental_distribute_datasets_from_function(
  37         functools.partial(task.build_inputs, config.train_data))
  38 
  39     iterator = iter(dataset)
  40     optimizer = tf.keras.optimizers.SGD(lr=0.1)
  41     task.train_step(next(iterator), model, optimizer, metrics=metrics)
  42     task.validation_step(next(iterator), model, metrics=metrics)
  43 
  44   def test_task(self):
  45     config = sentence_prediction.SentencePredictionConfig(
  46         init_checkpoint=self.get_temp_dir(),
  47         network=bert.BertPretrainerConfig(
  48             encoder=encoders.TransformerEncoderConfig(
  49                 vocab_size=30522, num_layers=1),
  50             num_masked_tokens=0,
  51             cls_heads=[
  52                 bert.ClsHeadConfig(
  53                     inner_dim=10, num_classes=3, name="sentence_prediction")
  54             ]),
  55         train_data=bert.BertSentencePredictionDataConfig(
  56             input_path="dummy", seq_length=128, global_batch_size=1))
  57     task = sentence_prediction.SentencePredictionTask(config)
  58     model = task.build_model()
  59     metrics = task.build_metrics()
  60     dataset = task.build_inputs(config.train_data)
  61 
  62     iterator = iter(dataset)
  63     optimizer = tf.keras.optimizers.SGD(lr=0.1)
  64     task.train_step(next(iterator), model, optimizer, metrics=metrics)
  65     task.validation_step(next(iterator), model, metrics=metrics)
  66 
  67     # Saves a checkpoint.
  68     pretrain_cfg = bert.BertPretrainerConfig(
  69         encoder=encoders.TransformerEncoderConfig(
  70             vocab_size=30522, num_layers=1),
  71         num_masked_tokens=20,
  72         cls_heads=[
  73             bert.ClsHeadConfig(
  74                 inner_dim=10, num_classes=3, name="next_sentence")
  75         ])
  76     pretrain_model = bert.instantiate_from_cfg(pretrain_cfg)
  77     ckpt = tf.train.Checkpoint(
  78         model=pretrain_model, **pretrain_model.checkpoint_items)
  79     ckpt.save(config.init_checkpoint)
  80     task.initialize(model)
  81 
  82   def _export_bert_tfhub(self):
  83     bert_config = configs.BertConfig(
  84         vocab_size=30522,
  85         hidden_size=16,
  86         intermediate_size=32,
  87         max_position_embeddings=128,
  88         num_attention_heads=2,
  89         num_hidden_layers=1)
  90     _, encoder = export_tfhub.create_bert_model(bert_config)
  91     model_checkpoint_dir = os.path.join(self.get_temp_dir(), "checkpoint")
  92     checkpoint = tf.train.Checkpoint(model=encoder)
  93     checkpoint.save(os.path.join(model_checkpoint_dir, "test"))
  94     model_checkpoint_path = tf.train.latest_checkpoint(model_checkpoint_dir)
  95 
  96     vocab_file = os.path.join(self.get_temp_dir(), "uncased_vocab.txt")
  97     with tf.io.gfile.GFile(vocab_file, "w") as f:
  98       f.write("dummy content")
  99 
 100     hub_destination = os.path.join(self.get_temp_dir(), "hub")
 101     export_tfhub.export_bert_tfhub(bert_config, model_checkpoint_path,
 102                                    hub_destination, vocab_file)
 103     return hub_destination
 104 
 105   def test_task_with_hub(self):
 106     hub_module_url = self._export_bert_tfhub()
 107     config = sentence_prediction.SentencePredictionConfig(
 108         hub_module_url=hub_module_url,
 109         network=bert.BertPretrainerConfig(
 110             encoders.TransformerEncoderConfig(vocab_size=30522, num_layers=1),
 111             num_masked_tokens=0,
 112             cls_heads=[
 113                 bert.ClsHeadConfig(
 114                     inner_dim=10, num_classes=3, name="sentence_prediction")
 115             ]),
 116         train_data=bert.BertSentencePredictionDataConfig(
 117             input_path="dummy", seq_length=128, global_batch_size=10))
 118     self._run_task(config)
