  29 @dataclasses.dataclass
  30 class SentencePredictionConfig(cfg.TaskConfig):
  31   """The model config."""
  32   # At most one of `init_checkpoint` and `hub_module_url` can
  33   # be specified.
  34   init_checkpoint: str = ''
  35   hub_module_url: str = ''
  36   network: bert.BertPretrainerConfig = bert.BertPretrainerConfig(
  37       num_masked_tokens=0,
  38       cls_heads=[
  39           bert.ClsHeadConfig(
  40               inner_dim=768,
  41               num_classes=3,
  42               dropout_rate=0.1,
  43               name='sentence_prediction')
  44       ])
  45   train_data: cfg.DataConfig = cfg.DataConfig()
  46   validation_data: cfg.DataConfig = cfg.DataConfig()

  49 @base_task.register_task_cls(SentencePredictionConfig)
  50 class SentencePredictionTask(base_task.Task):
  51   """Task object for sentence_prediction."""
  52 
  53   def __init__(self, params=cfg.TaskConfig):
  54     super(SentencePredictionTask, self).__init__(params)
  55     if params.hub_module_url and params.init_checkpoint:
  56       raise ValueError('At most one of `hub_module_url` and '
  57                        '`pretrain_checkpoint_dir` can be specified.')
  58     if params.hub_module_url:
  59       self._hub_module = hub.load(params.hub_module_url)
  60     else:
  61       self._hub_module = None
  62 
  63   def build_model(self):
  64     if self._hub_module:
  65       input_word_ids = tf.keras.layers.Input(
  66           shape=(None,), dtype=tf.int32, name='input_word_ids')
  67       input_mask = tf.keras.layers.Input(
  68           shape=(None,), dtype=tf.int32, name='input_mask')
  69       input_type_ids = tf.keras.layers.Input(
  70           shape=(None,), dtype=tf.int32, name='input_type_ids')
  71       bert_model = hub.KerasLayer(self._hub_module, trainable=True)
  72       pooled_output, sequence_output = bert_model(
  73           [input_word_ids, input_mask, input_type_ids])
  74       encoder_from_hub = tf.keras.Model(
  75           inputs=[input_word_ids, input_mask, input_type_ids],
  76           outputs=[sequence_output, pooled_output])
  77       return bert.instantiate_from_cfg(
  78           self.task_config.network, encoder_network=encoder_from_hub)
  79     else:
  80       return bert.instantiate_from_cfg(self.task_config.network)
  81 
  82   def build_losses(self, labels, model_outputs, aux_losses=None) -> tf.Tensor:
  83     loss = loss_lib.weighted_sparse_categorical_crossentropy_loss(
  84         labels=labels,
  85         predictions=tf.nn.log_softmax(
  86             model_outputs['sentence_prediction'], axis=-1))
  87 
  88     if aux_losses:
  89       loss += tf.add_n(aux_losses)
  90     return loss
  91 
  92   def build_inputs(self, params, input_context=None):
  93     """Returns tf.data.Dataset for sentence_prediction task."""
  94     if params.input_path == 'dummy':
  95 
  96       def dummy_data(_):
  97         dummy_ids = tf.zeros((1, params.seq_length), dtype=tf.int32)
  98         x = dict(
  99             input_word_ids=dummy_ids,
 100             input_mask=dummy_ids,
 101             input_type_ids=dummy_ids)
 102         y = tf.ones((1, 1), dtype=tf.int32)
 103         return (x, y)
 104 
 105       dataset = tf.data.Dataset.range(1)
 106       dataset = dataset.repeat()
 107       dataset = dataset.map(
 108           dummy_data, num_parallel_calls=tf.data.experimental.AUTOTUNE)
 109       return dataset
 110 
 111     return sentence_prediction_dataloader.SentencePredictionDataLoader(
 112         params).load(input_context)
 113 
 114   def build_metrics(self, training=None):
 115     del training
 116     metrics = [tf.keras.metrics.SparseCategoricalAccuracy(name='cls_accuracy')]
 117     return metrics
 118 
 119   def process_metrics(self, metrics, labels, model_outputs):
 120     for metric in metrics:
 121       metric.update_state(labels, model_outputs['sentence_prediction'])
 122 
 123   def process_compiled_metrics(self, compiled_metrics, labels, model_outputs):
 124     compiled_metrics.update_state(labels, model_outputs['sentence_prediction'])
 125 
 126   def initialize(self, model):
 127     """Load a pretrained checkpoint (if exists) and then train from iter 0."""
 128     ckpt_dir_or_file = self.task_config.init_checkpoint
 129     if tf.io.gfile.isdir(ckpt_dir_or_file):
 130       ckpt_dir_or_file = tf.train.latest_checkpoint(ckpt_dir_or_file)
 131     if not ckpt_dir_or_file:
 132       return
 133 
 134     pretrain2finetune_mapping = {
 135         'encoder':
 136             model.checkpoint_items['encoder'],
 137         'next_sentence.pooler_dense':
 138             model.checkpoint_items['sentence_prediction.pooler_dense'],
 139     }
 140     ckpt = tf.train.Checkpoint(**pretrain2finetune_mapping)
 141     status = ckpt.restore(ckpt_dir_or_file)
 142     status.expect_partial().assert_existing_objects_matched()
 143     logging.info('finished loading pretrained checkpoint from %s',
 144                  ckpt_dir_or_file)
