  16 """Tests for official.nlp.tasks.masked_lm."""
  17 
  18 import tensorflow as tf
  19 
  20 from official.nlp.configs import bert
  21 from official.nlp.configs import encoders
  22 from official.nlp.tasks import masked_lm
  23 
  24 
  25 class MLMTaskTest(tf.test.TestCase):
  26 
  27   def test_task(self):
  28     config = masked_lm.MaskedLMConfig(
  29         network=bert.BertPretrainerConfig(
  30             encoders.TransformerEncoderConfig(vocab_size=30522, num_layers=1),
  31             num_masked_tokens=20,
  32             cls_heads=[
  33                 bert.ClsHeadConfig(
  34                     inner_dim=10, num_classes=2, name="next_sentence")
  35             ]),
  36         train_data=bert.BertPretrainDataConfig(
  37             input_path="dummy",
  38             max_predictions_per_seq=20,
  39             seq_length=128,
  40             global_batch_size=1))
  41     task = masked_lm.MaskedLMTask(config)
  42     model = task.build_model()
  43     metrics = task.build_metrics()
  44     dataset = task.build_inputs(config.train_data)
  45 
  46     iterator = iter(dataset)
  47     optimizer = tf.keras.optimizers.SGD(lr=0.1)
  48     task.train_step(next(iterator), model, optimizer, metrics=metrics)
  49     task.validation_step(next(iterator), model, metrics=metrics)
  50 
  51 
  52 if __name__ == "__main__":
  53   tf.test.main()
