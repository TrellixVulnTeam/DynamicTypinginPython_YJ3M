  45   def build_losses(self,
  46                    labels,
  47                    model_outputs,
  48                    metrics,
  49                    aux_losses=None) -> tf.Tensor:
  50     metrics = dict([(metric.name, metric) for metric in metrics])
  51     lm_output = tf.nn.log_softmax(model_outputs['lm_output'], axis=-1)
  52     mlm_loss = loss_lib.weighted_sparse_categorical_crossentropy_loss(
  53         labels=labels['masked_lm_ids'],
  54         predictions=lm_output,
  55         weights=labels['masked_lm_weights'])
  56     metrics['lm_example_loss'].update_state(mlm_loss)
  57     if 'next_sentence_labels' in labels:
  58       policy = tf.keras.mixed_precision.experimental.global_policy()
  59       if policy.name == 'mixed_bfloat16':  # b/158514794: bf16 is not stable.
  60         policy = tf.float32
  61       predictions = tf.keras.layers.Activation(
  62           tf.nn.log_softmax, dtype=policy)(model_outputs['next_sentence'])
  63 
  64       sentence_labels = labels['next_sentence_labels']
  65       sentence_loss = loss_lib.weighted_sparse_categorical_crossentropy_loss(
  66           labels=sentence_labels,
  67           predictions=predictions)
  68       metrics['next_sentence_loss'].update_state(sentence_loss)
  69       total_loss = mlm_loss + sentence_loss
  70     else:
  71       total_loss = mlm_loss
  72 
  73     if aux_losses:
  74       total_loss += tf.add_n(aux_losses)
  75     return total_loss

  77   def build_inputs(self, params, input_context=None):
  78     """Returns tf.data.Dataset for pretraining."""
  79     if params.input_path == 'dummy':
  80       def dummy_data(_):
  81         dummy_ids = tf.zeros((1, params.seq_length), dtype=tf.int32)
  82         dummy_lm = tf.zeros((1, params.max_predictions_per_seq), dtype=tf.int32)
  83         return dict(
  84             input_word_ids=dummy_ids,
  85             input_mask=dummy_ids,
  86             input_type_ids=dummy_ids,
  87             masked_lm_positions=dummy_lm,
  88             masked_lm_ids=dummy_lm,
  89             masked_lm_weights=tf.cast(dummy_lm, dtype=tf.float32),
  90             next_sentence_labels=tf.zeros((1, 1), dtype=tf.int32))
  91 
  92       dataset = tf.data.Dataset.range(1)
  93       dataset = dataset.repeat()
  94       dataset = dataset.map(
  95           dummy_data, num_parallel_calls=tf.data.experimental.AUTOTUNE)
  96       return dataset
  97 
  98     return pretrain_dataloader.BertPretrainDataLoader(params).load(
  99         input_context)

 101   def build_metrics(self, training=None):
 102     del training
 103     metrics = [
 104         tf.keras.metrics.SparseCategoricalAccuracy(name='masked_lm_accuracy'),
 105         tf.keras.metrics.Mean(name='lm_example_loss')
 106     ]
 107     # TODO(hongkuny): rethink how to manage metrics creation with heads.
 108     if self.task_config.train_data.use_next_sentence_label:
 109       metrics.append(
 110           tf.keras.metrics.SparseCategoricalAccuracy(
 111               name='next_sentence_accuracy'))
 112       metrics.append(tf.keras.metrics.Mean(name='next_sentence_loss'))
 113     return metrics

 115   def process_metrics(self, metrics, labels, model_outputs):
 116     metrics = dict([(metric.name, metric) for metric in metrics])
 117     if 'masked_lm_accuracy' in metrics:
 118       metrics['masked_lm_accuracy'].update_state(labels['masked_lm_ids'],
 119                                                  model_outputs['lm_output'],
 120                                                  labels['masked_lm_weights'])
 121     if 'next_sentence_accuracy' in metrics:
 122       metrics['next_sentence_accuracy'].update_state(
 123           labels['next_sentence_labels'], model_outputs['next_sentence'])

 125   def train_step(self, inputs, model: tf.keras.Model,
 126                  optimizer: tf.keras.optimizers.Optimizer, metrics):
 127     """Does forward and backward.
 128 
 129     Args:
 130       inputs: a dictionary of input tensors.
 131       model: the model, forward pass definition.
 132       optimizer: the optimizer for this training step.
 133       metrics: a nested structure of metrics objects.
 134 
 135     Returns:
 136       A dictionary of logs.
 137     """
 138     with tf.GradientTape() as tape:
 139       outputs = model(inputs, training=True)
 140       # Computes per-replica loss.
 141       loss = self.build_losses(
 142           labels=inputs,
 143           model_outputs=outputs,
 144           metrics=metrics,
 145           aux_losses=model.losses)
 146       # Scales loss as the default gradients allreduce performs sum inside the
 147       # optimizer.
 148       # TODO(b/154564893): enable loss scaling.
 149       # scaled_loss = loss / tf.distribute.get_strategy().num_replicas_in_sync
 150     tvars = model.trainable_variables
 151     grads = tape.gradient(loss, tvars)
 152     optimizer.apply_gradients(list(zip(grads, tvars)))
 153     self.process_metrics(metrics, inputs, outputs)
 154     return {self.loss: loss}
 155 
 156   def validation_step(self, inputs, model: tf.keras.Model, metrics):
 157     """Validatation step.
 158 
 159     Args:
 160       inputs: a dictionary of input tensors.
 161       model: the keras.Model.
 162       metrics: a nested structure of metrics objects.
 163 
 164     Returns:
 165       A dictionary of logs.
 166     """
 167     outputs = self.inference_step(inputs, model)
 168     loss = self.build_losses(
 169         labels=inputs,
 170         model_outputs=outputs,
 171         metrics=metrics,
 172         aux_losses=model.losses)
 173     self.process_metrics(metrics, inputs, outputs)
 174     return {self.loss: loss}
