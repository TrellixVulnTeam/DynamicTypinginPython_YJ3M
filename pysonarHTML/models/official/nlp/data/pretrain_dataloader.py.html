  16 """Loads dataset for the BERT pretraining task."""
  17 from typing import Mapping, Optional
  18 
  19 import tensorflow as tf
  20 
  21 from official.core import input_reader
  22 
  23 
  24 class BertPretrainDataLoader:
  25   """A class to load dataset for bert pretraining task."""
  26 
  27   def __init__(self, params):
  28     """Inits `BertPretrainDataLoader` class.
  29 
  30     Args:
  31       params: A `BertPretrainDataConfig` object.
  32     """
  33     self._params = params
  34     self._seq_length = params.seq_length
  35     self._max_predictions_per_seq = params.max_predictions_per_seq
  36     self._use_next_sentence_label = params.use_next_sentence_label
  37     self._use_position_id = params.use_position_id
  38 
  39   def _decode(self, record: tf.Tensor):
  40     """Decodes a serialized tf.Example."""
  41     name_to_features = {
  42         'input_ids':
  43             tf.io.FixedLenFeature([self._seq_length], tf.int64),
  44         'input_mask':
  45             tf.io.FixedLenFeature([self._seq_length], tf.int64),
  46         'segment_ids':
  47             tf.io.FixedLenFeature([self._seq_length], tf.int64),
  48         'masked_lm_positions':
  49             tf.io.FixedLenFeature([self._max_predictions_per_seq], tf.int64),
  50         'masked_lm_ids':
  51             tf.io.FixedLenFeature([self._max_predictions_per_seq], tf.int64),
  52         'masked_lm_weights':
  53             tf.io.FixedLenFeature([self._max_predictions_per_seq], tf.float32),
  54     }
  55     if self._use_next_sentence_label:
  56       name_to_features['next_sentence_labels'] = tf.io.FixedLenFeature([1],
  57                                                                        tf.int64)
  58     if self._use_position_id:
  59       name_to_features['position_ids'] = tf.io.FixedLenFeature(
  60           [self._seq_length], tf.int64)
  61 
  62     example = tf.io.parse_single_example(record, name_to_features)
  63 
  64     # tf.Example only supports tf.int64, but the TPU only supports tf.int32.
  65     # So cast all int64 to int32.
  66     for name in list(example.keys()):
  67       t = example[name]
  68       if t.dtype == tf.int64:
  69         t = tf.cast(t, tf.int32)
  70       example[name] = t
  71 
  72     return example
  73 
  74   def _parse(self, record: Mapping[str, tf.Tensor]):
  75     """Parses raw tensors into a dict of tensors to be consumed by the model."""
  76     x = {
  77         'input_word_ids': record['input_ids'],
  78         'input_mask': record['input_mask'],
  79         'input_type_ids': record['segment_ids'],
  80         'masked_lm_positions': record['masked_lm_positions'],
  81         'masked_lm_ids': record['masked_lm_ids'],
  82         'masked_lm_weights': record['masked_lm_weights'],
  83     }
  84     if self._use_next_sentence_label:
  85       x['next_sentence_labels'] = record['next_sentence_labels']
  86     if self._use_position_id:
  87       x['position_ids'] = record['position_ids']
  88 
  89     return x
  90 
  91   def load(self, input_context: Optional[tf.distribute.InputContext] = None):
  92     """Returns a tf.dataset.Dataset."""
  93     reader = input_reader.InputReader(
  94         params=self._params,
  95         decoder_fn=self._decode,
  96         parser_fn=self._parse)
  97     return reader.read(input_context)
