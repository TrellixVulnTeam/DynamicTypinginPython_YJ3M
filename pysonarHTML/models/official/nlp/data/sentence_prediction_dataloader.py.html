  16 """Loads dataset for the sentence prediction (classification) task."""
  17 from typing import Mapping, Optional
  18 import tensorflow as tf
  19 
  20 from official.core import input_reader
  21 
  22 
  23 class SentencePredictionDataLoader:
  24   """A class to load dataset for sentence prediction (classification) task."""
  25 
  26   def __init__(self, params):
  27     self._params = params
  28     self._seq_length = params.seq_length
  29 
  30   def _decode(self, record: tf.Tensor):
  31     """Decodes a serialized tf.Example."""
  32     name_to_features = {
  33         'input_ids': tf.io.FixedLenFeature([self._seq_length], tf.int64),
  34         'input_mask': tf.io.FixedLenFeature([self._seq_length], tf.int64),
  35         'segment_ids': tf.io.FixedLenFeature([self._seq_length], tf.int64),
  36         'label_ids': tf.io.FixedLenFeature([], tf.int64),
  37     }
  38     example = tf.io.parse_single_example(record, name_to_features)
  39 
  40     # tf.Example only supports tf.int64, but the TPU only supports tf.int32.
  41     # So cast all int64 to int32.
  42     for name in example:
  43       t = example[name]
  44       if t.dtype == tf.int64:
  45         t = tf.cast(t, tf.int32)
  46       example[name] = t
  47 
  48     return example
  49 
  50   def _parse(self, record: Mapping[str, tf.Tensor]):
  51     """Parses raw tensors into a dict of tensors to be consumed by the model."""
  52     x = {
  53         'input_word_ids': record['input_ids'],
  54         'input_mask': record['input_mask'],
  55         'input_type_ids': record['segment_ids']
  56     }
  57     y = record['label_ids']
  58     return (x, y)
  59 
  60   def load(self, input_context: Optional[tf.distribute.InputContext] = None):
  61     """Returns a tf.dataset.Dataset."""
  62     reader = input_reader.InputReader(
  63         params=self._params, decoder_fn=self._decode, parser_fn=self._parse)
  64     return reader.read(input_context)
