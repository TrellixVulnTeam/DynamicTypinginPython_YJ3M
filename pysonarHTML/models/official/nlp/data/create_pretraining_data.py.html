  74 class TrainingInstance(object):
  75   """A single training instance (sentence pair)."""
  76 
  77   def __init__(self, tokens, segment_ids, masked_lm_positions, masked_lm_labels,
  78                is_random_next):
  79     self.tokens = tokens
  80     self.segment_ids = segment_ids
  81     self.is_random_next = is_random_next
  82     self.masked_lm_positions = masked_lm_positions
  83     self.masked_lm_labels = masked_lm_labels
  84 
  85   def __str__(self):
  86     s = ""
  87     s += "tokens: %s\n" % (" ".join(
  88         [tokenization.printable_text(x) for x in self.tokens]))
  89     s += "segment_ids: %s\n" % (" ".join([str(x) for x in self.segment_ids]))
  90     s += "is_random_next: %s\n" % self.is_random_next
  91     s += "masked_lm_positions: %s\n" % (" ".join(
  92         [str(x) for x in self.masked_lm_positions]))
  93     s += "masked_lm_labels: %s\n" % (" ".join(
  94         [tokenization.printable_text(x) for x in self.masked_lm_labels]))
  95     s += "\n"
  96     return s
  97 
  98   def __repr__(self):
  99     return self.__str__()

 102 def write_instance_to_example_files(instances, tokenizer, max_seq_length,
 103                                     max_predictions_per_seq, output_files,
 104                                     gzip_compress):
 105   """Create TF example files from `TrainingInstance`s."""
 106   writers = []
 107   for output_file in output_files:
 108     writers.append(
 109         tf.io.TFRecordWriter(
 110             output_file, options="GZIP" if gzip_compress else ""))
 111 
 112   writer_index = 0
 113 
 114   total_written = 0
 115   for (inst_index, instance) in enumerate(instances):
 116     input_ids = tokenizer.convert_tokens_to_ids(instance.tokens)
 117     input_mask = [1] * len(input_ids)
 118     segment_ids = list(instance.segment_ids)
 119     assert len(input_ids) <= max_seq_length
 120 
 121     while len(input_ids) < max_seq_length:
 122       input_ids.append(0)
 123       input_mask.append(0)
 124       segment_ids.append(0)
 125 
 126     assert len(input_ids) == max_seq_length
 127     assert len(input_mask) == max_seq_length
 128     assert len(segment_ids) == max_seq_length
 129 
 130     masked_lm_positions = list(instance.masked_lm_positions)
 131     masked_lm_ids = tokenizer.convert_tokens_to_ids(instance.masked_lm_labels)
 132     masked_lm_weights = [1.0] * len(masked_lm_ids)
 133 
 134     while len(masked_lm_positions) < max_predictions_per_seq:
 135       masked_lm_positions.append(0)
 136       masked_lm_ids.append(0)
 137       masked_lm_weights.append(0.0)
 138 
 139     next_sentence_label = 1 if instance.is_random_next else 0
 140 
 141     features = collections.OrderedDict()
 142     features["input_ids"] = create_int_feature(input_ids)
 143     features["input_mask"] = create_int_feature(input_mask)
 144     features["segment_ids"] = create_int_feature(segment_ids)
 145     features["masked_lm_positions"] = create_int_feature(masked_lm_positions)
 146     features["masked_lm_ids"] = create_int_feature(masked_lm_ids)
 147     features["masked_lm_weights"] = create_float_feature(masked_lm_weights)
 148     features["next_sentence_labels"] = create_int_feature([next_sentence_label])
 149 
 150     tf_example = tf.train.Example(features=tf.train.Features(feature=features))
 151 
 152     writers[writer_index].write(tf_example.SerializeToString())
 153     writer_index = (writer_index + 1) % len(writers)
 154 
 155     total_written += 1
 156 
 157     if inst_index < 20:
 158       logging.info("*** Example ***")
 159       logging.info("tokens: %s", " ".join(
 160           [tokenization.printable_text(x) for x in instance.tokens]))
 161 
 162       for feature_name in features.keys():
 163         feature = features[feature_name]
 164         values = []
 165         if feature.int64_list.value:
 166           values = feature.int64_list.value
 167         elif feature.float_list.value:
 168           values = feature.float_list.value
 169         logging.info("%s: %s", feature_name, " ".join([str(x) for x in values]))
 170 
 171   for writer in writers:
 172     writer.close()
 173 
 174   logging.info("Wrote %d total instances", total_written)
 175

 177 def create_int_feature(values):
 178   feature = tf.train.Feature(int64_list=tf.train.Int64List(value=list(values)))
 179   return feature
 180 
 181 
 182 def create_float_feature(values):
 183   feature = tf.train.Feature(float_list=tf.train.FloatList(value=list(values)))
 184   return feature

 187 def create_training_instances(input_files,
 188                               tokenizer,
 189                               max_seq_length,
 190                               dupe_factor,
 191                               short_seq_prob,
 192                               masked_lm_prob,
 193                               max_predictions_per_seq,
 194                               rng,
 195                               do_whole_word_mask=False):
 196   """Create `TrainingInstance`s from raw text."""
 197   all_documents = [[]]
 198 
 199   # Input file format:
 200   # (1) One sentence per line. These should ideally be actual sentences, not
 201   # entire paragraphs or arbitrary spans of text. (Because we use the
 202   # sentence boundaries for the "next sentence prediction" task).
 203   # (2) Blank lines between documents. Document boundaries are needed so
 204   # that the "next sentence prediction" task doesn't span between documents.
 205   for input_file in input_files:
 206     with tf.io.gfile.GFile(input_file, "rb") as reader:
 207       while True:
 208         line = tokenization.convert_to_unicode(reader.readline())
 209         if not line:
 210           break
 211         line = line.strip()
 212 
 213         # Empty lines are used as document delimiters
 214         if not line:
 215           all_documents.append([])
 216         tokens = tokenizer.tokenize(line)
 217         if tokens:
 218           all_documents[-1].append(tokens)
 219 
 220   # Remove empty documents
 221   all_documents = [x for x in all_documents if x]
 222   rng.shuffle(all_documents)
 223 
 224   vocab_words = list(tokenizer.vocab.keys())
 225   instances = []
 226   for _ in range(dupe_factor):
 227     for document_index in range(len(all_documents)):
 228       instances.extend(
 229           create_instances_from_document(
 230               all_documents, document_index, max_seq_length, short_seq_prob,
 231               masked_lm_prob, max_predictions_per_seq, vocab_words, rng,
 232               do_whole_word_mask))
 233 
 234   rng.shuffle(instances)
 235   return instances

 238 def create_instances_from_document(
 239     all_documents, document_index, max_seq_length, short_seq_prob,
 240     masked_lm_prob, max_predictions_per_seq, vocab_words, rng,
 241     do_whole_word_mask=False):
 242   """Creates `TrainingInstance`s for a single document."""
 243   document = all_documents[document_index]
 244 
 245   # Account for [CLS], [SEP], [SEP]
 246   max_num_tokens = max_seq_length - 3
 247 
 248   # We *usually* want to fill up the entire sequence since we are padding
 249   # to `max_seq_length` anyways, so short sequences are generally wasted
 250   # computation. However, we *sometimes*
 251   # (i.e., short_seq_prob == 0.1 == 10% of the time) want to use shorter
 252   # sequences to minimize the mismatch between pre-training and fine-tuning.
 253   # The `target_seq_length` is just a rough target however, whereas
 254   # `max_seq_length` is a hard limit.
 255   target_seq_length = max_num_tokens
 256   if rng.random() < short_seq_prob:
 257     target_seq_length = rng.randint(2, max_num_tokens)
 258 
 259   # We DON'T just concatenate all of the tokens from a document into a long
 260   # sequence and choose an arbitrary split point because this would make the
 261   # next sentence prediction task too easy. Instead, we split the input into
 262   # segments "A" and "B" based on the actual "sentences" provided by the user
 263   # input.
 264   instances = []
 265   current_chunk = []
 266   current_length = 0
 267   i = 0
 268   while i < len(document):
 269     segment = document[i]
 270     current_chunk.append(segment)
 271     current_length += len(segment)
 272     if i == len(document) - 1 or current_length >= target_seq_length:
 273       if current_chunk:
 274         # `a_end` is how many segments from `current_chunk` go into the `A`
 275         # (first) sentence.
 276         a_end = 1
 277         if len(current_chunk) >= 2:
 278           a_end = rng.randint(1, len(current_chunk) - 1)
 279 
 280         tokens_a = []
 281         for j in range(a_end):
 282           tokens_a.extend(current_chunk[j])
 283 
 284         tokens_b = []
 285         # Random next
 286         is_random_next = False
 287         if len(current_chunk) == 1 or rng.random() < 0.5:
 288           is_random_next = True
 289           target_b_length = target_seq_length - len(tokens_a)
 290 
 291           # This should rarely go for more than one iteration for large
 292           # corpora. However, just to be careful, we try to make sure that
 293           # the random document is not the same as the document
 294           # we're processing.
 295           for _ in range(10):
 296             random_document_index = rng.randint(0, len(all_documents) - 1)
 297             if random_document_index != document_index:
 298               break
 299 
 300           random_document = all_documents[random_document_index]
 301           random_start = rng.randint(0, len(random_document) - 1)
 302           for j in range(random_start, len(random_document)):
 303             tokens_b.extend(random_document[j])
 304             if len(tokens_b) >= target_b_length:
 305               break
 306           # We didn't actually use these segments so we "put them back" so
 307           # they don't go to waste.
 308           num_unused_segments = len(current_chunk) - a_end
 309           i -= num_unused_segments
 310         # Actual next
 311         else:
 312           is_random_next = False
 313           for j in range(a_end, len(current_chunk)):
 314             tokens_b.extend(current_chunk[j])
 315         truncate_seq_pair(tokens_a, tokens_b, max_num_tokens, rng)
 316 
 317         assert len(tokens_a) >= 1
 318         assert len(tokens_b) >= 1
 319 
 320         tokens = []
 321         segment_ids = []
 322         tokens.append("[CLS]")
 323         segment_ids.append(0)
 324         for token in tokens_a:
 325           tokens.append(token)
 326           segment_ids.append(0)
 327 
 328         tokens.append("[SEP]")
 329         segment_ids.append(0)
 330 
 331         for token in tokens_b:
 332           tokens.append(token)
 333           segment_ids.append(1)
 334         tokens.append("[SEP]")
 335         segment_ids.append(1)
 336 
 337         (tokens, masked_lm_positions,
 338          masked_lm_labels) = create_masked_lm_predictions(
 339              tokens, masked_lm_prob, max_predictions_per_seq, vocab_words, rng,
 340              do_whole_word_mask)
 341         instance = TrainingInstance(
 342             tokens=tokens,
 343             segment_ids=segment_ids,
 344             is_random_next=is_random_next,
 345             masked_lm_positions=masked_lm_positions,
 346             masked_lm_labels=masked_lm_labels)
 347         instances.append(instance)
 348       current_chunk = []
 349       current_length = 0
 350     i += 1
 351 
 352   return instances

 359 def create_masked_lm_predictions(tokens, masked_lm_prob,
 360                                  max_predictions_per_seq, vocab_words, rng,
 361                                  do_whole_word_mask):
 362   """Creates the predictions for the masked LM objective."""
 363 
 364   cand_indexes = []
 365   for (i, token) in enumerate(tokens):
 366     if token == "[CLS]" or token == "[SEP]":
 367       continue
 368     # Whole Word Masking means that if we mask all of the wordpieces
 369     # corresponding to an original word. When a word has been split into
 370     # WordPieces, the first token does not have any marker and any subsequence
 371     # tokens are prefixed with ##. So whenever we see the ## token, we
 372     # append it to the previous set of word indexes.
 373     #
 374     # Note that Whole Word Masking does *not* change the training code
 375     # at all -- we still predict each WordPiece independently, softmaxed
 376     # over the entire vocabulary.
 377     if (do_whole_word_mask and len(cand_indexes) >= 1 and
 378         token.startswith("##")):
 379       cand_indexes[-1].append(i)
 380     else:
 381       cand_indexes.append([i])
 382 
 383   rng.shuffle(cand_indexes)
 384 
 385   output_tokens = list(tokens)
 386 
 387   num_to_predict = min(max_predictions_per_seq,
 388                        max(1, int(round(len(tokens) * masked_lm_prob))))
 389 
 390   masked_lms = []
 391   covered_indexes = set()
 392   for index_set in cand_indexes:
 393     if len(masked_lms) >= num_to_predict:
 394       break
 395     # If adding a whole-word mask would exceed the maximum number of
 396     # predictions, then just skip this candidate.
 397     if len(masked_lms) + len(index_set) > num_to_predict:
 398       continue
 399     is_any_index_covered = False
 400     for index in index_set:
 401       if index in covered_indexes:
 402         is_any_index_covered = True
 403         break
 404     if is_any_index_covered:
 405       continue
 406     for index in index_set:
 407       covered_indexes.add(index)
 408 
 409       masked_token = None
 410       # 80% of the time, replace with [MASK]
 411       if rng.random() < 0.8:
 412         masked_token = "[MASK]"
 413       else:
 414         # 10% of the time, keep original
 415         if rng.random() < 0.5:
 416           masked_token = tokens[index]
 417         # 10% of the time, replace with random word
 418         else:
 419           masked_token = vocab_words[rng.randint(0, len(vocab_words) - 1)]
 420 
 421       output_tokens[index] = masked_token
 422 
 423       masked_lms.append(MaskedLmInstance(index=index, label=tokens[index]))
 424   assert len(masked_lms) <= num_to_predict
 425   masked_lms = sorted(masked_lms, key=lambda x: x.index)
 426 
 427   masked_lm_positions = []
 428   masked_lm_labels = []
 429   for p in masked_lms:
 430     masked_lm_positions.append(p.index)
 431     masked_lm_labels.append(p.label)
 432 
 433   return (output_tokens, masked_lm_positions, masked_lm_labels)

 436 def truncate_seq_pair(tokens_a, tokens_b, max_num_tokens, rng):
 437   """Truncates a pair of sequences to a maximum sequence length."""
 438   while True:
 439     total_length = len(tokens_a) + len(tokens_b)
 440     if total_length <= max_num_tokens:
 441       break
 442 
 443     trunc_tokens = tokens_a if len(tokens_a) > len(tokens_b) else tokens_b
 444     assert len(trunc_tokens) >= 1
 445 
 446     # We want to sometimes truncate from the front and sometimes from the
 447     # back to add more randomness and avoid biases.
 448     if rng.random() < 0.5:
 449       del trunc_tokens[0]
 450     else:
 451       trunc_tokens.pop()

 454 def main(_):
 455   tokenizer = tokenization.FullTokenizer(
 456       vocab_file=FLAGS.vocab_file, do_lower_case=FLAGS.do_lower_case)
 457 
 458   input_files = []
 459   for input_pattern in FLAGS.input_file.split(","):
 460     input_files.extend(tf.io.gfile.glob(input_pattern))
 461 
 462   logging.info("*** Reading from input files ***")
 463   for input_file in input_files:
 464     logging.info("  %s", input_file)
 465 
 466   rng = random.Random(FLAGS.random_seed)
 467   instances = create_training_instances(
 468       input_files, tokenizer, FLAGS.max_seq_length, FLAGS.dupe_factor,
 469       FLAGS.short_seq_prob, FLAGS.masked_lm_prob, FLAGS.max_predictions_per_seq,
 470       rng, FLAGS.do_whole_word_mask)
 471 
 472   output_files = FLAGS.output_file.split(",")
 473   logging.info("*** Writing to output files ***")
 474   for output_file in output_files:
 475     logging.info("  %s", output_file)
 476 
 477   write_instance_to_example_files(instances, tokenizer, FLAGS.max_seq_length,
 478                                   FLAGS.max_predictions_per_seq, output_files,
 479                                   FLAGS.gzip_compress)
 480 
 481 
 482 if __name__ == "__main__":
 483   flags.mark_flag_as_required("input_file")
 484   flags.mark_flag_as_required("output_file")
 485   flags.mark_flag_as_required("vocab_file")
 486   app.run(main)
