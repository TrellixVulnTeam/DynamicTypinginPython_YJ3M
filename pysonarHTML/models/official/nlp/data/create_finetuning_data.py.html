 134 def generate_classifier_dataset():
 135   """Generates classifier dataset and returns input meta data."""
 136   assert (FLAGS.input_data_dir and FLAGS.classification_task_name
 137           or FLAGS.tfds_params)
 138 
 139   if FLAGS.tokenizer_impl == "word_piece":
 140     tokenizer = tokenization.FullTokenizer(
 141         vocab_file=FLAGS.vocab_file, do_lower_case=FLAGS.do_lower_case)
 142     processor_text_fn = tokenization.convert_to_unicode
 143   else:
 144     assert FLAGS.tokenizer_impl == "sentence_piece"
 145     tokenizer = tokenization.FullSentencePieceTokenizer(FLAGS.sp_model_file)
 146     processor_text_fn = functools.partial(
 147         tokenization.preprocess_text, lower=FLAGS.do_lower_case)
 148 
 149   if FLAGS.tfds_params:
 150     processor = classifier_data_lib.TfdsProcessor(
 151         tfds_params=FLAGS.tfds_params,
 152         process_text_fn=processor_text_fn)
 153     return classifier_data_lib.generate_tf_record_from_data_file(
 154         processor,
 155         None,
 156         tokenizer,
 157         train_data_output_path=FLAGS.train_data_output_path,
 158         eval_data_output_path=FLAGS.eval_data_output_path,
 159         test_data_output_path=FLAGS.test_data_output_path,
 160         max_seq_length=FLAGS.max_seq_length)
 161   else:
 162     processors = {
 163         "cola":
 164             classifier_data_lib.ColaProcessor,
 165         "mnli":
 166             classifier_data_lib.MnliProcessor,
 167         "mrpc":
 168             classifier_data_lib.MrpcProcessor,
 169         "qnli":
 170             classifier_data_lib.QnliProcessor,
 171         "qqp": classifier_data_lib.QqpProcessor,
 172         "sst-2":
 173             classifier_data_lib.SstProcessor,
 174         "xnli":
 175             functools.partial(classifier_data_lib.XnliProcessor,
 176                               language=FLAGS.xnli_language),
 177         "paws-x":
 178             functools.partial(classifier_data_lib.PawsxProcessor,
 179                               language=FLAGS.pawsx_language)
 180     }
 181     task_name = FLAGS.classification_task_name.lower()
 182     if task_name not in processors:
 183       raise ValueError("Task not found: %s" % (task_name))
 184 
 185     processor = processors[task_name](process_text_fn=processor_text_fn)
 186     return classifier_data_lib.generate_tf_record_from_data_file(
 187         processor,
 188         FLAGS.input_data_dir,
 189         tokenizer,
 190         train_data_output_path=FLAGS.train_data_output_path,
 191         eval_data_output_path=FLAGS.eval_data_output_path,
 192         test_data_output_path=FLAGS.test_data_output_path,
 193         max_seq_length=FLAGS.max_seq_length)

 196 def generate_regression_dataset():
 197   """Generates regression dataset and returns input meta data."""
 198   if FLAGS.tokenizer_impl == "word_piece":
 199     tokenizer = tokenization.FullTokenizer(
 200         vocab_file=FLAGS.vocab_file, do_lower_case=FLAGS.do_lower_case)
 201     processor_text_fn = tokenization.convert_to_unicode
 202   else:
 203     assert FLAGS.tokenizer_impl == "sentence_piece"
 204     tokenizer = tokenization.FullSentencePieceTokenizer(FLAGS.sp_model_file)
 205     processor_text_fn = functools.partial(
 206         tokenization.preprocess_text, lower=FLAGS.do_lower_case)
 207 
 208   if FLAGS.tfds_params:
 209     processor = classifier_data_lib.TfdsProcessor(
 210         tfds_params=FLAGS.tfds_params,
 211         process_text_fn=processor_text_fn)
 212     return classifier_data_lib.generate_tf_record_from_data_file(
 213         processor,
 214         None,
 215         tokenizer,
 216         train_data_output_path=FLAGS.train_data_output_path,
 217         eval_data_output_path=FLAGS.eval_data_output_path,
 218         test_data_output_path=FLAGS.test_data_output_path,
 219         max_seq_length=FLAGS.max_seq_length)
 220   else:
 221     raise ValueError("No data processor found for the given regression task.")

 224 def generate_squad_dataset():
 225   """Generates squad training dataset and returns input meta data."""
 226   assert FLAGS.squad_data_file
 227   if FLAGS.tokenizer_impl == "word_piece":
 228     return squad_lib_wp.generate_tf_record_from_json_file(
 229         FLAGS.squad_data_file, FLAGS.vocab_file, FLAGS.train_data_output_path,
 230         FLAGS.max_seq_length, FLAGS.do_lower_case, FLAGS.max_query_length,
 231         FLAGS.doc_stride, FLAGS.version_2_with_negative)
 232   else:
 233     assert FLAGS.tokenizer_impl == "sentence_piece"
 234     return squad_lib_sp.generate_tf_record_from_json_file(
 235         FLAGS.squad_data_file, FLAGS.sp_model_file,
 236         FLAGS.train_data_output_path, FLAGS.max_seq_length, FLAGS.do_lower_case,
 237         FLAGS.max_query_length, FLAGS.doc_stride, FLAGS.version_2_with_negative)

 240 def main(_):
 241   if FLAGS.tokenizer_impl == "word_piece":
 242     if not FLAGS.vocab_file:
 243       raise ValueError(
 244           "FLAG vocab_file for word-piece tokenizer is not specified.")
 245   else:
 246     assert FLAGS.tokenizer_impl == "sentence_piece"
 247     if not FLAGS.sp_model_file:
 248       raise ValueError(
 249           "FLAG sp_model_file for sentence-piece tokenizer is not specified.")
 250 
 251   if FLAGS.fine_tuning_task_type == "classification":
 252     input_meta_data = generate_classifier_dataset()
 253   elif FLAGS.fine_tuning_task_type == "regression":
 254     input_meta_data = generate_regression_dataset()
 255   else:
 256     input_meta_data = generate_squad_dataset()
 257 
 258   tf.io.gfile.makedirs(os.path.dirname(FLAGS.meta_data_file_path))
 259   with tf.io.gfile.GFile(FLAGS.meta_data_file_path, "w") as writer:
 260     writer.write(json.dumps(input_meta_data, indent=4) + "\n")
 261 
 262 
 263 if __name__ == "__main__":
 264   flags.mark_flag_as_required("train_data_output_path")
 265   flags.mark_flag_as_required("meta_data_file_path")
 266   app.run(main)
