  35 class SquadExample(object):
  36   """A single training/test example for simple sequence classification.
  37 
  38   For examples without an answer, the start and end position are -1.
  39 
  40   Attributes:
  41     qas_id: ID of the question-answer pair.
  42     question_text: Original text for the question.
  43     doc_tokens: The list of tokens in the context obtained by splitting
  44       on whitespace only.
  45     orig_answer_text: Original text for the answer.
  46     start_position: Starting index of the answer in `doc_tokens`.
  47     end_position: Ending index of the answer in `doc_tokens`.
  48     is_impossible: Whether the question is impossible to answer given the
  49       context. Only used in SQuAD 2.0.
  50   """
  51 
  52   def __init__(self,
  53                qas_id,
  54                question_text,
  55                doc_tokens,
  56                orig_answer_text=None,
  57                start_position=None,
  58                end_position=None,
  59                is_impossible=False):
  60     self.qas_id = qas_id
  61     self.question_text = question_text
  62     self.doc_tokens = doc_tokens
  63     self.orig_answer_text = orig_answer_text
  64     self.start_position = start_position
  65     self.end_position = end_position
  66     self.is_impossible = is_impossible
  67 
  68   def __str__(self):
  69     return self.__repr__()
  70 
  71   def __repr__(self):
  72     s = ""
  73     s += "qas_id: %s" % (tokenization.printable_text(self.qas_id))
  74     s += ", question_text: %s" % (
  75         tokenization.printable_text(self.question_text))
  76     s += ", doc_tokens: [%s]" % (" ".join(self.doc_tokens))
  77     if self.start_position:
  78       s += ", start_position: %d" % (self.start_position)
  79     if self.start_position:
  80       s += ", end_position: %d" % (self.end_position)
  81     if self.start_position:
  82       s += ", is_impossible: %r" % (self.is_impossible)
  83     return s

 116 class FeatureWriter(object):
 117   """Writes InputFeature to TF example file."""
 118 
 119   def __init__(self, filename, is_training):
 120     self.filename = filename
 121     self.is_training = is_training
 122     self.num_features = 0
 123     tf.io.gfile.makedirs(os.path.dirname(filename))
 124     self._writer = tf.io.TFRecordWriter(filename)
 125 
 126   def process_feature(self, feature):
 127     """Write a InputFeature to the TFRecordWriter as a tf.train.Example."""
 128     self.num_features += 1
 129 
 130     def create_int_feature(values):
 131       feature = tf.train.Feature(
 132           int64_list=tf.train.Int64List(value=list(values)))
 133       return feature
 134 
 135     features = collections.OrderedDict()
 136     features["unique_ids"] = create_int_feature([feature.unique_id])
 137     features["input_ids"] = create_int_feature(feature.input_ids)
 138     features["input_mask"] = create_int_feature(feature.input_mask)
 139     features["segment_ids"] = create_int_feature(feature.segment_ids)
 140 
 141     if self.is_training:
 142       features["start_positions"] = create_int_feature([feature.start_position])
 143       features["end_positions"] = create_int_feature([feature.end_position])
 144       impossible = 0
 145       if feature.is_impossible:
 146         impossible = 1
 147       features["is_impossible"] = create_int_feature([impossible])
 148 
 149     tf_example = tf.train.Example(features=tf.train.Features(feature=features))
 150     self._writer.write(tf_example.SerializeToString())
 151 
 152   def close(self):
 153     self._writer.close()

 156 def read_squad_examples(input_file, is_training, version_2_with_negative):
 157   """Read a SQuAD json file into a list of SquadExample."""
 158   with tf.io.gfile.GFile(input_file, "r") as reader:
 159     input_data = json.load(reader)["data"]
 160 
 161   def is_whitespace(c):
 162     if c == " " or c == "\t" or c == "\r" or c == "\n" or ord(c) == 0x202F:
 163       return True
 164     return False
 165 
 166   examples = []
 167   for entry in input_data:
 168     for paragraph in entry["paragraphs"]:
 169       paragraph_text = paragraph["context"]
 170       doc_tokens = []
 171       char_to_word_offset = []
 172       prev_is_whitespace = True
 173       for c in paragraph_text:
 174         if is_whitespace(c):
 175           prev_is_whitespace = True
 176         else:
 177           if prev_is_whitespace:
 178             doc_tokens.append(c)
 179           else:
 180             doc_tokens[-1] += c
 181           prev_is_whitespace = False
 182         char_to_word_offset.append(len(doc_tokens) - 1)
 183 
 184       for qa in paragraph["qas"]:
 185         qas_id = qa["id"]
 186         question_text = qa["question"]
 187         start_position = None
 188         end_position = None
 189         orig_answer_text = None
 190         is_impossible = False
 191         if is_training:
 192 
 193           if version_2_with_negative:
 194             is_impossible = qa["is_impossible"]
 195           if (len(qa["answers"]) != 1) and (not is_impossible):
 196             raise ValueError(
 197                 "For training, each question should have exactly 1 answer.")
 198           if not is_impossible:
 199             answer = qa["answers"][0]
 200             orig_answer_text = answer["text"]
 201             answer_offset = answer["answer_start"]
 202             answer_length = len(orig_answer_text)
 203             start_position = char_to_word_offset[answer_offset]
 204             end_position = char_to_word_offset[answer_offset + answer_length -
 205                                                1]
 206             # Only add answers where the text can be exactly recovered from the
 207             # document. If this CAN'T happen it's likely due to weird Unicode
 208             # stuff so we will just skip the example.
 209             #
 210             # Note that this means for training mode, every example is NOT
 211             # guaranteed to be preserved.
 212             actual_text = " ".join(
 213                 doc_tokens[start_position:(end_position + 1)])
 214             cleaned_answer_text = " ".join(
 215                 tokenization.whitespace_tokenize(orig_answer_text))
 216             if actual_text.find(cleaned_answer_text) == -1:
 217               logging.warning("Could not find answer: '%s' vs. '%s'",
 218                               actual_text, cleaned_answer_text)
 219               continue
 220           else:
 221             start_position = -1
 222             end_position = -1
 223             orig_answer_text = ""
 224 
 225         example = SquadExample(
 226             qas_id=qas_id,
 227             question_text=question_text,
 228             doc_tokens=doc_tokens,
 229             orig_answer_text=orig_answer_text,
 230             start_position=start_position,
 231             end_position=end_position,
 232             is_impossible=is_impossible)
 233         examples.append(example)
 234 
 235   return examples

 238 def convert_examples_to_features(examples,
 239                                  tokenizer,
 240                                  max_seq_length,
 241                                  doc_stride,
 242                                  max_query_length,
 243                                  is_training,
 244                                  output_fn,
 245                                  batch_size=None):
 246   """Loads a data file into a list of `InputBatch`s."""
 247 
 248   base_id = 1000000000
 249   unique_id = base_id
 250   feature = None
 251   for (example_index, example) in enumerate(examples):
 252     query_tokens = tokenizer.tokenize(example.question_text)
 253 
 254     if len(query_tokens) > max_query_length:
 255       query_tokens = query_tokens[0:max_query_length]
 256 
 257     tok_to_orig_index = []
 258     orig_to_tok_index = []
 259     all_doc_tokens = []
 260     for (i, token) in enumerate(example.doc_tokens):
 261       orig_to_tok_index.append(len(all_doc_tokens))
 262       sub_tokens = tokenizer.tokenize(token)
 263       for sub_token in sub_tokens:
 264         tok_to_orig_index.append(i)
 265         all_doc_tokens.append(sub_token)
 266 
 267     tok_start_position = None
 268     tok_end_position = None
 269     if is_training and example.is_impossible:
 270       tok_start_position = -1
 271       tok_end_position = -1
 272     if is_training and not example.is_impossible:
 273       tok_start_position = orig_to_tok_index[example.start_position]
 274       if example.end_position < len(example.doc_tokens) - 1:
 275         tok_end_position = orig_to_tok_index[example.end_position + 1] - 1
 276       else:
 277         tok_end_position = len(all_doc_tokens) - 1
 278       (tok_start_position, tok_end_position) = _improve_answer_span(
 279           all_doc_tokens, tok_start_position, tok_end_position, tokenizer,
 280           example.orig_answer_text)
 281 
 282     # The -3 accounts for [CLS], [SEP] and [SEP]
 283     max_tokens_for_doc = max_seq_length - len(query_tokens) - 3
 284 
 285     # We can have documents that are longer than the maximum sequence length.
 286     # To deal with this we do a sliding window approach, where we take chunks
 287     # of the up to our max length with a stride of `doc_stride`.
 288     _DocSpan = collections.namedtuple(  # pylint: disable=invalid-name
 289         "DocSpan", ["start", "length"])
 290     doc_spans = []
 291     start_offset = 0
 292     while start_offset < len(all_doc_tokens):
 293       length = len(all_doc_tokens) - start_offset
 294       if length > max_tokens_for_doc:
 295         length = max_tokens_for_doc
 296       doc_spans.append(_DocSpan(start=start_offset, length=length))
 297       if start_offset + length == len(all_doc_tokens):
 298         break
 299       start_offset += min(length, doc_stride)
 300 
 301     for (doc_span_index, doc_span) in enumerate(doc_spans):
 302       tokens = []
 303       token_to_orig_map = {}
 304       token_is_max_context = {}
 305       segment_ids = []
 306       tokens.append("[CLS]")
 307       segment_ids.append(0)
 308       for token in query_tokens:
 309         tokens.append(token)
 310         segment_ids.append(0)
 311       tokens.append("[SEP]")
 312       segment_ids.append(0)
 313 
 314       for i in range(doc_span.length):
 315         split_token_index = doc_span.start + i
 316         token_to_orig_map[len(tokens)] = tok_to_orig_index[split_token_index]
 317 
 318         is_max_context = _check_is_max_context(doc_spans, doc_span_index,
 319                                                split_token_index)
 320         token_is_max_context[len(tokens)] = is_max_context
 321         tokens.append(all_doc_tokens[split_token_index])
 322         segment_ids.append(1)
 323       tokens.append("[SEP]")
 324       segment_ids.append(1)
 325 
 326       input_ids = tokenizer.convert_tokens_to_ids(tokens)
 327 
 328       # The mask has 1 for real tokens and 0 for padding tokens. Only real
 329       # tokens are attended to.
 330       input_mask = [1] * len(input_ids)
 331 
 332       # Zero-pad up to the sequence length.
 333       while len(input_ids) < max_seq_length:
 334         input_ids.append(0)
 335         input_mask.append(0)
 336         segment_ids.append(0)
 337 
 338       assert len(input_ids) == max_seq_length
 339       assert len(input_mask) == max_seq_length
 340       assert len(segment_ids) == max_seq_length
 341 
 342       start_position = None
 343       end_position = None
 344       if is_training and not example.is_impossible:
 345         # For training, if our document chunk does not contain an annotation
 346         # we throw it out, since there is nothing to predict.
 347         doc_start = doc_span.start
 348         doc_end = doc_span.start + doc_span.length - 1
 349         out_of_span = False
 350         if not (tok_start_position >= doc_start and
 351                 tok_end_position <= doc_end):
 352           out_of_span = True
 353         if out_of_span:
 354           start_position = 0
 355           end_position = 0
 356         else:
 357           doc_offset = len(query_tokens) + 2
 358           start_position = tok_start_position - doc_start + doc_offset
 359           end_position = tok_end_position - doc_start + doc_offset
 360 
 361       if is_training and example.is_impossible:
 362         start_position = 0
 363         end_position = 0
 364 
 365       if example_index < 20:
 366         logging.info("*** Example ***")
 367         logging.info("unique_id: %s", (unique_id))
 368         logging.info("example_index: %s", (example_index))
 369         logging.info("doc_span_index: %s", (doc_span_index))
 370         logging.info("tokens: %s",
 371                      " ".join([tokenization.printable_text(x) for x in tokens]))
 372         logging.info(
 373             "token_to_orig_map: %s", " ".join([
 374                 "%d:%d" % (x, y) for (x, y) in six.iteritems(token_to_orig_map)
 375             ]))
 376         logging.info(
 377             "token_is_max_context: %s", " ".join([
 378                 "%d:%s" % (x, y)
 379                 for (x, y) in six.iteritems(token_is_max_context)
 380             ]))
 381         logging.info("input_ids: %s", " ".join([str(x) for x in input_ids]))
 382         logging.info("input_mask: %s", " ".join([str(x) for x in input_mask]))
 383         logging.info("segment_ids: %s", " ".join([str(x) for x in segment_ids]))
 384         if is_training and example.is_impossible:
 385           logging.info("impossible example")
 386         if is_training and not example.is_impossible:
 387           answer_text = " ".join(tokens[start_position:(end_position + 1)])
 388           logging.info("start_position: %d", (start_position))
 389           logging.info("end_position: %d", (end_position))
 390           logging.info("answer: %s", tokenization.printable_text(answer_text))
 391 
 392       feature = InputFeatures(
 393           unique_id=unique_id,
 394           example_index=example_index,
 395           doc_span_index=doc_span_index,
 396           tokens=tokens,
 397           token_to_orig_map=token_to_orig_map,
 398           token_is_max_context=token_is_max_context,
 399           input_ids=input_ids,
 400           input_mask=input_mask,
 401           segment_ids=segment_ids,
 402           start_position=start_position,
 403           end_position=end_position,
 404           is_impossible=example.is_impossible)
 405 
 406       # Run callback
 407       if is_training:
 408         output_fn(feature)
 409       else:
 410         output_fn(feature, is_padding=False)
 411 
 412       unique_id += 1
 413 
 414   if not is_training and feature:
 415     assert batch_size
 416     num_padding = 0
 417     num_examples = unique_id - base_id
 418     if unique_id % batch_size != 0:
 419       num_padding = batch_size - (num_examples % batch_size)
 420     logging.info("Adding padding examples to make sure no partial batch.")
 421     logging.info("Adds %d padding examples for inference.", num_padding)
 422     dummy_feature = copy.deepcopy(feature)
 423     for _ in range(num_padding):
 424       dummy_feature.unique_id = unique_id
 425 
 426       # Run callback
 427       output_fn(feature, is_padding=True)
 428       unique_id += 1
 429   return unique_id - base_id

 432 def _improve_answer_span(doc_tokens, input_start, input_end, tokenizer,
 433                          orig_answer_text):
 434   """Returns tokenized answer spans that better match the annotated answer."""
 435 
 436   # The SQuAD annotations are character based. We first project them to
 437   # whitespace-tokenized words. But then after WordPiece tokenization, we can
 438   # often find a "better match". For example:
 439   #
 440   #   Question: What year was John Smith born?
 441   #   Context: The leader was John Smith (1895-1943).
 442   #   Answer: 1895
 443   #
 444   # The original whitespace-tokenized answer will be "(1895-1943).". However
 445   # after tokenization, our tokens will be "( 1895 - 1943 ) .". So we can match
 446   # the exact answer, 1895.
 447   #
 448   # However, this is not always possible. Consider the following:
 449   #
 450   #   Question: What country is the top exporter of electornics?
 451   #   Context: The Japanese electronics industry is the lagest in the world.
 452   #   Answer: Japan
 453   #
 454   # In this case, the annotator chose "Japan" as a character sub-span of
 455   # the word "Japanese". Since our WordPiece tokenizer does not split
 456   # "Japanese", we just use "Japanese" as the annotation. This is fairly rare
 457   # in SQuAD, but does happen.
 458   tok_answer_text = " ".join(tokenizer.tokenize(orig_answer_text))
 459 
 460   for new_start in range(input_start, input_end + 1):
 461     for new_end in range(input_end, new_start - 1, -1):
 462       text_span = " ".join(doc_tokens[new_start:(new_end + 1)])
 463       if text_span == tok_answer_text:
 464         return (new_start, new_end)
 465 
 466   return (input_start, input_end)

 469 def _check_is_max_context(doc_spans, cur_span_index, position):
 470   """Check if this is the 'max context' doc span for the token."""
 471 
 472   # Because of the sliding window approach taken to scoring documents, a single
 473   # token can appear in multiple documents. E.g.
 474   #  Doc: the man went to the store and bought a gallon of milk
 475   #  Span A: the man went to the
 476   #  Span B: to the store and bought
 477   #  Span C: and bought a gallon of
 478   #  ...
 479   #
 480   # Now the word 'bought' will have two scores from spans B and C. We only
 481   # want to consider the score with "maximum context", which we define as
 482   # the *minimum* of its left and right context (the *sum* of left and
 483   # right context will always be the same, of course).
 484   #
 485   # In the example the maximum context for 'bought' would be span C since
 486   # it has 1 left context and 3 right context, while span B has 4 left context
 487   # and 0 right context.
 488   best_score = None
 489   best_span_index = None
 490   for (span_index, doc_span) in enumerate(doc_spans):
 491     end = doc_span.start + doc_span.length - 1
 492     if position < doc_span.start:
 493       continue
 494     if position > end:
 495       continue
 496     num_left_context = position - doc_span.start
 497     num_right_context = end - position
 498     score = min(num_left_context, num_right_context) + 0.01 * doc_span.length
 499     if best_score is None or score > best_score:
 500       best_score = score
 501       best_span_index = span_index
 502 
 503   return cur_span_index == best_span_index

 506 def write_predictions(all_examples,
 507                       all_features,
 508                       all_results,

 510                       max_answer_length,
 511                       do_lower_case,

 539 def postprocess_output(all_examples,
 540                        all_features,
 541                        all_results,
 542                        n_best_size,
 543                        max_answer_length,
 544                        do_lower_case,
 545                        version_2_with_negative=False,
 546                        null_score_diff_threshold=0.0,
 547                        verbose=False):
 548   """Postprocess model output, to form predicton results."""
 549 
 550   example_index_to_features = collections.defaultdict(list)
 551   for feature in all_features:
 552     example_index_to_features[feature.example_index].append(feature)
 553   unique_id_to_result = {}
 554   for result in all_results:
 555     unique_id_to_result[result.unique_id] = result
 556 
 557   _PrelimPrediction = collections.namedtuple(  # pylint: disable=invalid-name
 558       "PrelimPrediction",
 559       ["feature_index", "start_index", "end_index", "start_logit", "end_logit"])
 560 
 561   all_predictions = collections.OrderedDict()
 562   all_nbest_json = collections.OrderedDict()
 563   scores_diff_json = collections.OrderedDict()
 564 
 565   for (example_index, example) in enumerate(all_examples):
 566     features = example_index_to_features[example_index]
 567 
 568     prelim_predictions = []
 569     # keep track of the minimum score of null start+end of position 0
 570     score_null = 1000000  # large and positive
 571     min_null_feature_index = 0  # the paragraph slice with min mull score
 572     null_start_logit = 0  # the start logit at the slice with min null score
 573     null_end_logit = 0  # the end logit at the slice with min null score
 574     for (feature_index, feature) in enumerate(features):
 575       result = unique_id_to_result[feature.unique_id]
 576       start_indexes = _get_best_indexes(result.start_logits, n_best_size)
 577       end_indexes = _get_best_indexes(result.end_logits, n_best_size)
 578       # if we could have irrelevant answers, get the min score of irrelevant
 579       if version_2_with_negative:
 580         feature_null_score = result.start_logits[0] + result.end_logits[0]
 581         if feature_null_score < score_null:
 582           score_null = feature_null_score
 583           min_null_feature_index = feature_index
 584           null_start_logit = result.start_logits[0]
 585           null_end_logit = result.end_logits[0]
 586       for start_index in start_indexes:
 587         for end_index in end_indexes:
 588           # We could hypothetically create invalid predictions, e.g., predict
 589           # that the start of the span is in the question. We throw out all
 590           # invalid predictions.
 591           if start_index >= len(feature.tokens):
 592             continue
 593           if end_index >= len(feature.tokens):
 594             continue
 595           if start_index not in feature.token_to_orig_map:
 596             continue
 597           if end_index not in feature.token_to_orig_map:
 598             continue
 599           if not feature.token_is_max_context.get(start_index, False):
 600             continue
 601           if end_index < start_index:
 602             continue
 603           length = end_index - start_index + 1
 604           if length > max_answer_length:
 605             continue
 606           prelim_predictions.append(
 607               _PrelimPrediction(
 608                   feature_index=feature_index,
 609                   start_index=start_index,
 610                   end_index=end_index,
 611                   start_logit=result.start_logits[start_index],
 612                   end_logit=result.end_logits[end_index]))
 613 
 614     if version_2_with_negative:
 615       prelim_predictions.append(
 616           _PrelimPrediction(
 617               feature_index=min_null_feature_index,
 618               start_index=0,
 619               end_index=0,
 620               start_logit=null_start_logit,
 621               end_logit=null_end_logit))
 622     prelim_predictions = sorted(
 623         prelim_predictions,
 624         key=lambda x: (x.start_logit + x.end_logit),
 625         reverse=True)
 626 
 627     _NbestPrediction = collections.namedtuple(  # pylint: disable=invalid-name
 628         "NbestPrediction", ["text", "start_logit", "end_logit"])
 629 
 630     seen_predictions = {}
 631     nbest = []
 632     for pred in prelim_predictions:
 633       if len(nbest) >= n_best_size:
 634         break
 635       feature = features[pred.feature_index]
 636       if pred.start_index > 0:  # this is a non-null prediction
 637         tok_tokens = feature.tokens[pred.start_index:(pred.end_index + 1)]
 638         orig_doc_start = feature.token_to_orig_map[pred.start_index]
 639         orig_doc_end = feature.token_to_orig_map[pred.end_index]
 640         orig_tokens = example.doc_tokens[orig_doc_start:(orig_doc_end + 1)]
 641         tok_text = " ".join(tok_tokens)
 642 
 643         # De-tokenize WordPieces that have been split off.
 644         tok_text = tok_text.replace(" ##", "")
 645         tok_text = tok_text.replace("##", "")
 646 
 647         # Clean whitespace
 648         tok_text = tok_text.strip()
 649         tok_text = " ".join(tok_text.split())
 650         orig_text = " ".join(orig_tokens)
 651 
 652         final_text = get_final_text(
 653             tok_text, orig_text, do_lower_case, verbose=verbose)
 654         if final_text in seen_predictions:
 655           continue
 656 
 657         seen_predictions[final_text] = True
 658       else:
 659         final_text = ""
 660         seen_predictions[final_text] = True
 661 
 662       nbest.append(
 663           _NbestPrediction(
 664               text=final_text,
 665               start_logit=pred.start_logit,
 666               end_logit=pred.end_logit))
 667 
 668     # if we didn't inlude the empty option in the n-best, inlcude it
 669     if version_2_with_negative:
 670       if "" not in seen_predictions:
 671         nbest.append(
 672             _NbestPrediction(
 673                 text="", start_logit=null_start_logit,
 674                 end_logit=null_end_logit))
 675     # In very rare edge cases we could have no valid predictions. So we
 676     # just create a nonce prediction in this case to avoid failure.
 677     if not nbest:
 678       nbest.append(
 679           _NbestPrediction(text="empty", start_logit=0.0, end_logit=0.0))
 680 
 681     assert len(nbest) >= 1
 682 
 683     total_scores = []
 684     best_non_null_entry = None
 685     for entry in nbest:
 686       total_scores.append(entry.start_logit + entry.end_logit)
 687       if not best_non_null_entry:
 688         if entry.text:
 689           best_non_null_entry = entry
 690 
 691     probs = _compute_softmax(total_scores)
 692 
 693     nbest_json = []
 694     for (i, entry) in enumerate(nbest):
 695       output = collections.OrderedDict()
 696       output["text"] = entry.text
 697       output["probability"] = probs[i]
 698       output["start_logit"] = entry.start_logit
 699       output["end_logit"] = entry.end_logit
 700       nbest_json.append(output)
 701 
 702     assert len(nbest_json) >= 1
 703 
 704     if not version_2_with_negative:
 705       all_predictions[example.qas_id] = nbest_json[0]["text"]
 706     else:
 707       # pytype: disable=attribute-error
 708       # predict "" iff the null score - the score of best non-null > threshold
 709       if best_non_null_entry is not None:
 710         score_diff = score_null - best_non_null_entry.start_logit - (
 711             best_non_null_entry.end_logit)
 712         scores_diff_json[example.qas_id] = score_diff
 713         if score_diff > null_score_diff_threshold:
 714           all_predictions[example.qas_id] = ""
 715         else:
 716           all_predictions[example.qas_id] = best_non_null_entry.text
 717       else:
 718         logging.warning("best_non_null_entry is None")
 719         scores_diff_json[example.qas_id] = score_null
 720         all_predictions[example.qas_id] = ""
 721       # pytype: enable=attribute-error
 722 
 723     all_nbest_json[example.qas_id] = nbest_json
 724 
 725   return all_predictions, all_nbest_json, scores_diff_json

 733 def get_final_text(pred_text, orig_text, do_lower_case, verbose=False):
 734   """Project the tokenized prediction back to the original text."""
 735 
 736   # When we created the data, we kept track of the alignment between original
 737   # (whitespace tokenized) tokens and our WordPiece tokenized tokens. So
 738   # now `orig_text` contains the span of our original text corresponding to the
 739   # span that we predicted.
 740   #
 741   # However, `orig_text` may contain extra characters that we don't want in
 742   # our prediction.
 743   #
 744   # For example, let's say:
 745   #   pred_text = steve smith
 746   #   orig_text = Steve Smith's
 747   #
 748   # We don't want to return `orig_text` because it contains the extra "'s".
 749   #
 750   # We don't want to return `pred_text` because it's already been normalized
 751   # (the SQuAD eval script also does punctuation stripping/lower casing but
 752   # our tokenizer does additional normalization like stripping accent
 753   # characters).
 754   #
 755   # What we really want to return is "Steve Smith".
 756   #
 757   # Therefore, we have to apply a semi-complicated alignment heruistic between
 758   # `pred_text` and `orig_text` to get a character-to-charcter alignment. This
 759   # can fail in certain cases in which case we just return `orig_text`.
 760 
 761   def _strip_spaces(text):
 762     ns_chars = []
 763     ns_to_s_map = collections.OrderedDict()
 764     for (i, c) in enumerate(text):
 765       if c == " ":
 766         continue
 767       ns_to_s_map[len(ns_chars)] = i
 768       ns_chars.append(c)
 769     ns_text = "".join(ns_chars)
 770     return (ns_text, ns_to_s_map)
 771 
 772   # We first tokenize `orig_text`, strip whitespace from the result
 773   # and `pred_text`, and check if they are the same length. If they are
 774   # NOT the same length, the heuristic has failed. If they are the same
 775   # length, we assume the characters are one-to-one aligned.
 776   tokenizer = tokenization.BasicTokenizer(do_lower_case=do_lower_case)
 777 
 778   tok_text = " ".join(tokenizer.tokenize(orig_text))
 779 
 780   start_position = tok_text.find(pred_text)
 781   if start_position == -1:
 782     if verbose:
 783       logging.info("Unable to find text: '%s' in '%s'", pred_text, orig_text)
 784     return orig_text
 785   end_position = start_position + len(pred_text) - 1
 786 
 787   (orig_ns_text, orig_ns_to_s_map) = _strip_spaces(orig_text)
 788   (tok_ns_text, tok_ns_to_s_map) = _strip_spaces(tok_text)
 789 
 790   if len(orig_ns_text) != len(tok_ns_text):
 791     if verbose:
 792       logging.info("Length not equal after stripping spaces: '%s' vs '%s'",
 793                    orig_ns_text, tok_ns_text)
 794     return orig_text
 795 
 796   # We then project the characters in `pred_text` back to `orig_text` using
 797   # the character-to-character alignment.
 798   tok_s_to_ns_map = {}
 799   for (i, tok_index) in six.iteritems(tok_ns_to_s_map):
 800     tok_s_to_ns_map[tok_index] = i
 801 
 802   orig_start_position = None
 803   if start_position in tok_s_to_ns_map:
 804     ns_start_position = tok_s_to_ns_map[start_position]
 805     if ns_start_position in orig_ns_to_s_map:
 806       orig_start_position = orig_ns_to_s_map[ns_start_position]
 807 
 808   if orig_start_position is None:
 809     if verbose:
 810       logging.info("Couldn't map start position")
 811     return orig_text
 812 
 813   orig_end_position = None
 814   if end_position in tok_s_to_ns_map:
 815     ns_end_position = tok_s_to_ns_map[end_position]
 816     if ns_end_position in orig_ns_to_s_map:
 817       orig_end_position = orig_ns_to_s_map[ns_end_position]
 818 
 819   if orig_end_position is None:
 820     if verbose:
 821       logging.info("Couldn't map end position")
 822     return orig_text
 823 
 824   output_text = orig_text[orig_start_position:(orig_end_position + 1)]
 825   return output_text

 828 def _get_best_indexes(logits, n_best_size):
 829   """Get the n-best logits from a list."""
 830   index_and_score = sorted(enumerate(logits), key=lambda x: x[1], reverse=True)
 831 
 832   best_indexes = []
 833   for i in range(len(index_and_score)):  # pylint: disable=consider-using-enumerate
 834     if i >= n_best_size:
 835       break
 836     best_indexes.append(index_and_score[i][0])
 837   return best_indexes

 840 def _compute_softmax(scores):
 841   """Compute softmax probability over raw logits."""
 842   if not scores:
 843     return []
 844 
 845   max_score = None
 846   for score in scores:
 847     if max_score is None or score > max_score:
 848       max_score = score
 849 
 850   exp_scores = []
 851   total_sum = 0.0
 852   for score in scores:
 853     x = math.exp(score - max_score)
 854     exp_scores.append(x)
 855     total_sum += x
 856 
 857   probs = []
 858   for score in exp_scores:
 859     probs.append(score / total_sum)
 860   return probs
