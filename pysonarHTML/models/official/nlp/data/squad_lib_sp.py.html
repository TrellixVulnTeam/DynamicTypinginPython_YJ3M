  37 class SquadExample(object):
  38   """A single training/test example for simple sequence classification.
  39 
  40      For examples without an answer, the start and end position are -1.
  41   """
  42 
  43   def __init__(self,
  44                qas_id,
  45                question_text,
  46                paragraph_text,
  47                orig_answer_text=None,
  48                start_position=None,
  49                end_position=None,
  50                is_impossible=False):
  51     self.qas_id = qas_id
  52     self.question_text = question_text
  53     self.paragraph_text = paragraph_text
  54     self.orig_answer_text = orig_answer_text
  55     self.start_position = start_position
  56     self.end_position = end_position
  57     self.is_impossible = is_impossible
  58 
  59   def __str__(self):
  60     return self.__repr__()
  61 
  62   def __repr__(self):
  63     s = ""
  64     s += "qas_id: %s" % (tokenization.printable_text(self.qas_id))
  65     s += ", question_text: %s" % (
  66         tokenization.printable_text(self.question_text))
  67     s += ", paragraph_text: [%s]" % (" ".join(self.paragraph_text))
  68     if self.start_position:
  69       s += ", start_position: %d" % (self.start_position)
  70     if self.start_position:
  71       s += ", end_position: %d" % (self.end_position)
  72     if self.start_position:
  73       s += ", is_impossible: %r" % (self.is_impossible)
  74     return s

 111 def read_squad_examples(input_file, is_training, version_2_with_negative):
 112   """Read a SQuAD json file into a list of SquadExample."""
 113   del version_2_with_negative
 114   with tf.io.gfile.GFile(input_file, "r") as reader:
 115     input_data = json.load(reader)["data"]
 116 
 117   examples = []
 118   for entry in input_data:
 119     for paragraph in entry["paragraphs"]:
 120       paragraph_text = paragraph["context"]
 121 
 122       for qa in paragraph["qas"]:
 123         qas_id = qa["id"]
 124         question_text = qa["question"]
 125         start_position = None
 126         orig_answer_text = None
 127         is_impossible = False
 128 
 129         if is_training:
 130           is_impossible = qa.get("is_impossible", False)
 131           if (len(qa["answers"]) != 1) and (not is_impossible):
 132             raise ValueError(
 133                 "For training, each question should have exactly 1 answer.")
 134           if not is_impossible:
 135             answer = qa["answers"][0]
 136             orig_answer_text = answer["text"]
 137             start_position = answer["answer_start"]
 138           else:
 139             start_position = -1
 140             orig_answer_text = ""
 141 
 142         example = SquadExample(
 143             qas_id=qas_id,
 144             question_text=question_text,
 145             paragraph_text=paragraph_text,
 146             orig_answer_text=orig_answer_text,
 147             start_position=start_position,
 148             is_impossible=is_impossible)
 149         examples.append(example)
 150 
 151   return examples

 154 def _convert_index(index, pos, m=None, is_start=True):
 155   """Converts index."""
 156   if index[pos] is not None:
 157     return index[pos]
 158   n = len(index)
 159   rear = pos
 160   while rear < n - 1 and index[rear] is None:
 161     rear += 1
 162   front = pos
 163   while front > 0 and index[front] is None:
 164     front -= 1
 165   assert index[front] is not None or index[rear] is not None
 166   if index[front] is None:
 167     if index[rear] >= 1:
 168       if is_start:
 169         return 0
 170       else:
 171         return index[rear] - 1
 172     return index[rear]
 173   if index[rear] is None:
 174     if m is not None and index[front] < m - 1:
 175       if is_start:
 176         return index[front] + 1
 177       else:
 178         return m - 1
 179     return index[front]
 180   if is_start:
 181     if index[rear] > index[front] + 1:
 182       return index[front] + 1
 183     else:
 184       return index[rear]
 185   else:
 186     if index[rear] > index[front] + 1:
 187       return index[rear] - 1
 188     else:
 189       return index[front]

 192 def convert_examples_to_features(examples,
 193                                  tokenizer,
 194                                  max_seq_length,
 195                                  doc_stride,
 196                                  max_query_length,
 197                                  is_training,
 198                                  output_fn,
 199                                  do_lower_case,
 200                                  batch_size=None):
 201   """Loads a data file into a list of `InputBatch`s."""
 202   cnt_pos, cnt_neg = 0, 0
 203   base_id = 1000000000
 204   unique_id = base_id
 205   max_n, max_m = 1024, 1024
 206   f = np.zeros((max_n, max_m), dtype=np.float32)
 207 
 208   for (example_index, example) in enumerate(examples):
 209 
 210     if example_index % 100 == 0:
 211       logging.info("Converting %d/%d pos %d neg %d", example_index,
 212                    len(examples), cnt_pos, cnt_neg)
 213 
 214     query_tokens = tokenization.encode_ids(
 215         tokenizer.sp_model,
 216         tokenization.preprocess_text(
 217             example.question_text, lower=do_lower_case))
 218 
 219     if len(query_tokens) > max_query_length:
 220       query_tokens = query_tokens[0:max_query_length]
 221 
 222     paragraph_text = example.paragraph_text
 223     para_tokens = tokenization.encode_pieces(
 224         tokenizer.sp_model,
 225         tokenization.preprocess_text(
 226             example.paragraph_text, lower=do_lower_case))
 227 
 228     chartok_to_tok_index = []
 229     tok_start_to_chartok_index = []
 230     tok_end_to_chartok_index = []
 231     char_cnt = 0
 232     for i, token in enumerate(para_tokens):
 233       new_token = token.replace(tokenization.SPIECE_UNDERLINE, " ")
 234       chartok_to_tok_index.extend([i] * len(new_token))
 235       tok_start_to_chartok_index.append(char_cnt)
 236       char_cnt += len(new_token)
 237       tok_end_to_chartok_index.append(char_cnt - 1)
 238 
 239     tok_cat_text = "".join(para_tokens).replace(tokenization.SPIECE_UNDERLINE,
 240                                                 " ")
 241     n, m = len(paragraph_text), len(tok_cat_text)
 242 
 243     if n > max_n or m > max_m:
 244       max_n = max(n, max_n)
 245       max_m = max(m, max_m)
 246       f = np.zeros((max_n, max_m), dtype=np.float32)
 247 
 248     g = {}
 249     # pylint: disable=cell-var-from-loop
 250     def _lcs_match(max_dist, n=n, m=m):
 251       """Longest-common-substring algorithm."""
 252       f.fill(0)
 253       g.clear()
 254 
 255       ### longest common sub sequence
 256       # f[i, j] = max(f[i - 1, j], f[i, j - 1], f[i - 1, j - 1] + match(i, j))
 257       for i in range(n):
 258 
 259         # unlike standard LCS, this is specifically optimized for the setting
 260         # because the mismatch between sentence pieces and original text will
 261         # be small
 262         for j in range(i - max_dist, i + max_dist):
 263           if j >= m or j < 0:
 264             continue
 265 
 266           if i > 0:
 267             g[(i, j)] = 0
 268             f[i, j] = f[i - 1, j]
 269 
 270           if j > 0 and f[i, j - 1] > f[i, j]:
 271             g[(i, j)] = 1
 272             f[i, j] = f[i, j - 1]
 273 
 274           f_prev = f[i - 1, j - 1] if i > 0 and j > 0 else 0
 275           if (tokenization.preprocess_text(
 276               paragraph_text[i], lower=do_lower_case,
 277               remove_space=False) == tok_cat_text[j] and f_prev + 1 > f[i, j]):
 278             g[(i, j)] = 2
 279             f[i, j] = f_prev + 1
 280     # pylint: enable=cell-var-from-loop
 281 
 282     max_dist = abs(n - m) + 5
 283     for _ in range(2):
 284       _lcs_match(max_dist)
 285       if f[n - 1, m - 1] > 0.8 * n:
 286         break
 287       max_dist *= 2
 288 
 289     orig_to_chartok_index = [None] * n
 290     chartok_to_orig_index = [None] * m
 291     i, j = n - 1, m - 1
 292     while i >= 0 and j >= 0:
 293       if (i, j) not in g:
 294         break
 295       if g[(i, j)] == 2:
 296         orig_to_chartok_index[i] = j
 297         chartok_to_orig_index[j] = i
 298         i, j = i - 1, j - 1
 299       elif g[(i, j)] == 1:
 300         j = j - 1
 301       else:
 302         i = i - 1
 303 
 304     if (all(v is None for v in orig_to_chartok_index) or
 305         f[n - 1, m - 1] < 0.8 * n):
 306       logging.info("MISMATCH DETECTED!")
 307       continue
 308 
 309     tok_start_to_orig_index = []
 310     tok_end_to_orig_index = []
 311     for i in range(len(para_tokens)):
 312       start_chartok_pos = tok_start_to_chartok_index[i]
 313       end_chartok_pos = tok_end_to_chartok_index[i]
 314       start_orig_pos = _convert_index(
 315           chartok_to_orig_index, start_chartok_pos, n, is_start=True)
 316       end_orig_pos = _convert_index(
 317           chartok_to_orig_index, end_chartok_pos, n, is_start=False)
 318 
 319       tok_start_to_orig_index.append(start_orig_pos)
 320       tok_end_to_orig_index.append(end_orig_pos)
 321 
 322     if not is_training:
 323       tok_start_position = tok_end_position = None
 324 
 325     if is_training and example.is_impossible:
 326       tok_start_position = 0
 327       tok_end_position = 0
 328 
 329     if is_training and not example.is_impossible:
 330       start_position = example.start_position
 331       end_position = start_position + len(example.orig_answer_text) - 1
 332 
 333       start_chartok_pos = _convert_index(
 334           orig_to_chartok_index, start_position, is_start=True)
 335       tok_start_position = chartok_to_tok_index[start_chartok_pos]
 336 
 337       end_chartok_pos = _convert_index(
 338           orig_to_chartok_index, end_position, is_start=False)
 339       tok_end_position = chartok_to_tok_index[end_chartok_pos]
 340       assert tok_start_position <= tok_end_position
 341 
 342     def _piece_to_id(x):
 343       return tokenizer.sp_model.PieceToId(x)
 344 
 345     all_doc_tokens = list(map(_piece_to_id, para_tokens))
 346 
 347     # The -3 accounts for [CLS], [SEP] and [SEP]
 348     max_tokens_for_doc = max_seq_length - len(query_tokens) - 3
 349 
 350     # We can have documents that are longer than the maximum sequence length.
 351     # To deal with this we do a sliding window approach, where we take chunks
 352     # of the up to our max length with a stride of `doc_stride`.
 353     _DocSpan = collections.namedtuple(  # pylint: disable=invalid-name
 354         "DocSpan", ["start", "length"])
 355     doc_spans = []
 356     start_offset = 0
 357     while start_offset < len(all_doc_tokens):
 358       length = len(all_doc_tokens) - start_offset
 359       if length > max_tokens_for_doc:
 360         length = max_tokens_for_doc
 361       doc_spans.append(_DocSpan(start=start_offset, length=length))
 362       if start_offset + length == len(all_doc_tokens):
 363         break
 364       start_offset += min(length, doc_stride)
 365 
 366     for (doc_span_index, doc_span) in enumerate(doc_spans):
 367       tokens = []
 368       token_is_max_context = {}
 369       segment_ids = []
 370 
 371       cur_tok_start_to_orig_index = []
 372       cur_tok_end_to_orig_index = []
 373 
 374       tokens.append(tokenizer.sp_model.PieceToId("[CLS]"))
 375       segment_ids.append(0)
 376       for token in query_tokens:
 377         tokens.append(token)
 378         segment_ids.append(0)
 379       tokens.append(tokenizer.sp_model.PieceToId("[SEP]"))
 380       segment_ids.append(0)
 381 
 382       for i in range(doc_span.length):
 383         split_token_index = doc_span.start + i
 384 
 385         cur_tok_start_to_orig_index.append(
 386             tok_start_to_orig_index[split_token_index])
 387         cur_tok_end_to_orig_index.append(
 388             tok_end_to_orig_index[split_token_index])
 389 
 390         is_max_context = _check_is_max_context(doc_spans, doc_span_index,
 391                                                split_token_index)
 392         token_is_max_context[len(tokens)] = is_max_context
 393         tokens.append(all_doc_tokens[split_token_index])
 394         segment_ids.append(1)
 395       tokens.append(tokenizer.sp_model.PieceToId("[SEP]"))
 396       segment_ids.append(1)
 397 
 398       paragraph_len = len(tokens)
 399       input_ids = tokens
 400 
 401       # The mask has 1 for real tokens and 0 for padding tokens. Only real
 402       # tokens are attended to.
 403       input_mask = [1] * len(input_ids)
 404 
 405       # Zero-pad up to the sequence length.
 406       while len(input_ids) < max_seq_length:
 407         input_ids.append(0)
 408         input_mask.append(0)
 409         segment_ids.append(0)
 410 
 411       assert len(input_ids) == max_seq_length
 412       assert len(input_mask) == max_seq_length
 413       assert len(segment_ids) == max_seq_length
 414 
 415       span_is_impossible = example.is_impossible
 416       start_position = None
 417       end_position = None
 418       if is_training and not span_is_impossible:
 419         # For training, if our document chunk does not contain an annotation
 420         # we throw it out, since there is nothing to predict.
 421         doc_start = doc_span.start
 422         doc_end = doc_span.start + doc_span.length - 1
 423         out_of_span = False
 424         if not (tok_start_position >= doc_start and
 425                 tok_end_position <= doc_end):
 426           out_of_span = True
 427         if out_of_span:
 428           # continue
 429           start_position = 0
 430           end_position = 0
 431           span_is_impossible = True
 432         else:
 433           doc_offset = len(query_tokens) + 2
 434           start_position = tok_start_position - doc_start + doc_offset
 435           end_position = tok_end_position - doc_start + doc_offset
 436 
 437       if is_training and span_is_impossible:
 438         start_position = 0
 439         end_position = 0
 440 
 441       if example_index < 20:
 442         logging.info("*** Example ***")
 443         logging.info("unique_id: %s", (unique_id))
 444         logging.info("example_index: %s", (example_index))
 445         logging.info("doc_span_index: %s", (doc_span_index))
 446         logging.info("tok_start_to_orig_index: %s",
 447                      " ".join([str(x) for x in cur_tok_start_to_orig_index]))
 448         logging.info("tok_end_to_orig_index: %s",
 449                      " ".join([str(x) for x in cur_tok_end_to_orig_index]))
 450         logging.info(
 451             "token_is_max_context: %s", " ".join(
 452                 ["%d:%s" % (x, y) for (x, y) in token_is_max_context.items()]))
 453         logging.info(
 454             "input_pieces: %s",
 455             " ".join([tokenizer.sp_model.IdToPiece(x) for x in tokens]))
 456         logging.info("input_ids: %s", " ".join([str(x) for x in input_ids]))
 457         logging.info("input_mask: %s", " ".join([str(x) for x in input_mask]))
 458         logging.info("segment_ids: %s", " ".join([str(x) for x in segment_ids]))
 459 
 460         if is_training and span_is_impossible:
 461           logging.info("impossible example span")
 462 
 463         if is_training and not span_is_impossible:
 464           pieces = [
 465               tokenizer.sp_model.IdToPiece(token)
 466               for token in tokens[start_position:(end_position + 1)]
 467           ]
 468           answer_text = tokenizer.sp_model.DecodePieces(pieces)
 469           logging.info("start_position: %d", (start_position))
 470           logging.info("end_position: %d", (end_position))
 471           logging.info("answer: %s", (tokenization.printable_text(answer_text)))
 472 
 473           # With multi processing, the example_index is actually the index
 474           # within the current process therefore we use example_index=None
 475           # to avoid being used in the future.
 476           # The current code does not use example_index of training data.
 477       if is_training:
 478         feat_example_index = None
 479       else:
 480         feat_example_index = example_index
 481 
 482       feature = InputFeatures(
 483           unique_id=unique_id,
 484           example_index=feat_example_index,
 485           doc_span_index=doc_span_index,
 486           tok_start_to_orig_index=cur_tok_start_to_orig_index,
 487           tok_end_to_orig_index=cur_tok_end_to_orig_index,
 488           token_is_max_context=token_is_max_context,
 489           tokens=[tokenizer.sp_model.IdToPiece(x) for x in tokens],
 490           input_ids=input_ids,
 491           input_mask=input_mask,
 492           segment_ids=segment_ids,
 493           paragraph_len=paragraph_len,
 494           start_position=start_position,
 495           end_position=end_position,
 496           is_impossible=span_is_impossible)
 497 
 498       # Run callback
 499       if is_training:
 500         output_fn(feature)
 501       else:
 502         output_fn(feature, is_padding=False)
 503 
 504       unique_id += 1
 505       if span_is_impossible:
 506         cnt_neg += 1
 507       else:
 508         cnt_pos += 1
 509 
 510   if not is_training and feature:
 511     assert batch_size
 512     num_padding = 0
 513     num_examples = unique_id - base_id
 514     if unique_id % batch_size != 0:
 515       num_padding = batch_size - (num_examples % batch_size)
 516     dummy_feature = copy.deepcopy(feature)
 517     for _ in range(num_padding):
 518       dummy_feature.unique_id = unique_id
 519 
 520       # Run callback
 521       output_fn(feature, is_padding=True)
 522       unique_id += 1
 523 
 524   logging.info("Total number of instances: %d = pos %d neg %d",
 525                cnt_pos + cnt_neg, cnt_pos, cnt_neg)
 526   return unique_id - base_id

 529 def _check_is_max_context(doc_spans, cur_span_index, position):
 530   """Check if this is the 'max context' doc span for the token."""
 531 
 532   # Because of the sliding window approach taken to scoring documents, a single
 533   # token can appear in multiple documents. E.g.
 534   #  Doc: the man went to the store and bought a gallon of milk
 535   #  Span A: the man went to the
 536   #  Span B: to the store and bought
 537   #  Span C: and bought a gallon of
 538   #  ...
 539   #
 540   # Now the word 'bought' will have two scores from spans B and C. We only
 541   # want to consider the score with "maximum context", which we define as
 542   # the *minimum* of its left and right context (the *sum* of left and
 543   # right context will always be the same, of course).
 544   #
 545   # In the example the maximum context for 'bought' would be span C since
 546   # it has 1 left context and 3 right context, while span B has 4 left context
 547   # and 0 right context.
 548   best_score = None
 549   best_span_index = None
 550   for (span_index, doc_span) in enumerate(doc_spans):
 551     end = doc_span.start + doc_span.length - 1
 552     if position < doc_span.start:
 553       continue
 554     if position > end:
 555       continue
 556     num_left_context = position - doc_span.start
 557     num_right_context = end - position
 558     score = min(num_left_context, num_right_context) + 0.01 * doc_span.length
 559     if best_score is None or score > best_score:
 560       best_score = score
 561       best_span_index = span_index
 562 
 563   return cur_span_index == best_span_index

 566 def write_predictions(all_examples,
 567                       all_features,
 568                       all_results,
 569                       n_best_size,
 570                       max_answer_length,
 571                       do_lower_case,
 572                       output_prediction_file,
 573                       output_nbest_file,
 574                       output_null_log_odds_file,
 575                       version_2_with_negative=False,
 576                       null_score_diff_threshold=0.0,
 577                       verbose=False):
 578   """Write final predictions to the json file and log-odds of null if needed."""
 579   logging.info("Writing predictions to: %s", (output_prediction_file))
 580   logging.info("Writing nbest to: %s", (output_nbest_file))
 581 
 582   all_predictions, all_nbest_json, scores_diff_json = (
 583       postprocess_output(all_examples=all_examples,
 584                          all_features=all_features,
 585                          all_results=all_results,
 586                          n_best_size=n_best_size,
 587                          max_answer_length=max_answer_length,
 588                          do_lower_case=do_lower_case,
 589                          version_2_with_negative=version_2_with_negative,
 590                          null_score_diff_threshold=null_score_diff_threshold,
 591                          verbose=verbose))
 592 
 593   write_to_json_files(all_predictions, output_prediction_file)
 594   write_to_json_files(all_nbest_json, output_nbest_file)
 595   if version_2_with_negative:
 596     write_to_json_files(scores_diff_json, output_null_log_odds_file)

 599 def postprocess_output(all_examples,
 600                        all_features,
 601                        all_results,
 602                        n_best_size,
 603                        max_answer_length,
 604                        do_lower_case,
 605                        version_2_with_negative=False,
 606                        null_score_diff_threshold=0.0,
 607                        verbose=False):
 608   """Postprocess model output, to form predicton results."""
 609 
 610   del do_lower_case, verbose
 611 
 612   example_index_to_features = collections.defaultdict(list)
 613   for feature in all_features:
 614     example_index_to_features[feature.example_index].append(feature)
 615 
 616   unique_id_to_result = {}
 617   for result in all_results:
 618     unique_id_to_result[result.unique_id] = result
 619 
 620   _PrelimPrediction = collections.namedtuple(  # pylint: disable=invalid-name
 621       "PrelimPrediction",
 622       ["feature_index", "start_index", "end_index", "start_logit", "end_logit"])
 623 
 624   all_predictions = collections.OrderedDict()
 625   all_nbest_json = collections.OrderedDict()
 626   scores_diff_json = collections.OrderedDict()
 627 
 628   for (example_index, example) in enumerate(all_examples):
 629     features = example_index_to_features[example_index]
 630 
 631     prelim_predictions = []
 632     # keep track of the minimum score of null start+end of position 0
 633     score_null = 1000000  # large and positive
 634     min_null_feature_index = 0  # the paragraph slice with min mull score
 635     null_start_logit = 0  # the start logit at the slice with min null score
 636     null_end_logit = 0  # the end logit at the slice with min null score
 637     for (feature_index, feature) in enumerate(features):
 638       result = unique_id_to_result[feature.unique_id]
 639       start_indexes = _get_best_indexes(result.start_logits, n_best_size)
 640       end_indexes = _get_best_indexes(result.end_logits, n_best_size)
 641       # if we could have irrelevant answers, get the min score of irrelevant
 642       if version_2_with_negative:
 643         feature_null_score = result.start_logits[0] + result.end_logits[0]
 644         if feature_null_score < score_null:
 645           score_null = feature_null_score
 646           min_null_feature_index = feature_index
 647           null_start_logit = result.start_logits[0]
 648           null_end_logit = result.end_logits[0]
 649       for start_index in start_indexes:
 650         for end_index in end_indexes:
 651           doc_offset = feature.tokens.index("[SEP]") + 1
 652           # We could hypothetically create invalid predictions, e.g., predict
 653           # that the start of the span is in the question. We throw out all
 654           # invalid predictions.
 655           if start_index - doc_offset >= len(feature.tok_start_to_orig_index):
 656             continue
 657           if end_index - doc_offset >= len(feature.tok_end_to_orig_index):
 658             continue
 659           # if start_index not in feature.tok_start_to_orig_index:
 660           #   continue
 661           # if end_index not in feature.tok_end_to_orig_index:
 662           #   continue
 663           if not feature.token_is_max_context.get(start_index, False):
 664             continue
 665           if end_index < start_index:
 666             continue
 667           length = end_index - start_index + 1
 668           if length > max_answer_length:
 669             continue
 670           prelim_predictions.append(
 671               _PrelimPrediction(
 672                   feature_index=feature_index,
 673                   start_index=start_index - doc_offset,
 674                   end_index=end_index - doc_offset,
 675                   start_logit=result.start_logits[start_index],
 676                   end_logit=result.end_logits[end_index]))
 677 
 678     if version_2_with_negative:
 679       prelim_predictions.append(
 680           _PrelimPrediction(
 681               feature_index=min_null_feature_index,
 682               start_index=-1,
 683               end_index=-1,
 684               start_logit=null_start_logit,
 685               end_logit=null_end_logit))
 686     prelim_predictions = sorted(
 687         prelim_predictions,
 688         key=lambda x: (x.start_logit + x.end_logit),
 689         reverse=True)
 690 
 691     _NbestPrediction = collections.namedtuple(  # pylint: disable=invalid-name
 692         "NbestPrediction", ["text", "start_logit", "end_logit"])
 693 
 694     seen_predictions = {}
 695     nbest = []
 696     for pred in prelim_predictions:
 697       if len(nbest) >= n_best_size:
 698         break
 699       feature = features[pred.feature_index]
 700       if pred.start_index >= 0:  # this is a non-null prediction
 701         tok_start_to_orig_index = feature.tok_start_to_orig_index
 702         tok_end_to_orig_index = feature.tok_end_to_orig_index
 703         start_orig_pos = tok_start_to_orig_index[pred.start_index]
 704         end_orig_pos = tok_end_to_orig_index[pred.end_index]
 705 
 706         paragraph_text = example.paragraph_text
 707         final_text = paragraph_text[start_orig_pos:end_orig_pos + 1].strip()
 708         if final_text in seen_predictions:
 709           continue
 710 
 711         seen_predictions[final_text] = True
 712       else:
 713         final_text = ""
 714         seen_predictions[final_text] = True
 715 
 716       nbest.append(
 717           _NbestPrediction(
 718               text=final_text,
 719               start_logit=pred.start_logit,
 720               end_logit=pred.end_logit))
 721 
 722     # if we didn't inlude the empty option in the n-best, inlcude it
 723     if version_2_with_negative:
 724       if "" not in seen_predictions:
 725         nbest.append(
 726             _NbestPrediction(
 727                 text="", start_logit=null_start_logit,
 728                 end_logit=null_end_logit))
 729     # In very rare edge cases we could have no valid predictions. So we
 730     # just create a nonce prediction in this case to avoid failure.
 731     if not nbest:
 732       nbest.append(
 733           _NbestPrediction(text="empty", start_logit=0.0, end_logit=0.0))
 734 
 735     assert len(nbest) >= 1
 736 
 737     total_scores = []
 738     best_non_null_entry = None
 739     for entry in nbest:
 740       total_scores.append(entry.start_logit + entry.end_logit)
 741       if not best_non_null_entry:
 742         if entry.text:
 743           best_non_null_entry = entry
 744 
 745     probs = _compute_softmax(total_scores)
 746 
 747     nbest_json = []
 748     for (i, entry) in enumerate(nbest):
 749       output = collections.OrderedDict()
 750       output["text"] = entry.text
 751       output["probability"] = probs[i]
 752       output["start_logit"] = entry.start_logit
 753       output["end_logit"] = entry.end_logit
 754       nbest_json.append(output)
 755 
 756     assert len(nbest_json) >= 1
 757 
 758     if not version_2_with_negative:
 759       all_predictions[example.qas_id] = nbest_json[0]["text"]
 760     else:
 761       assert best_non_null_entry is not None
 762       # predict "" iff the null score - the score of best non-null > threshold
 763       score_diff = score_null - best_non_null_entry.start_logit - (
 764           best_non_null_entry.end_logit)
 765       scores_diff_json[example.qas_id] = score_diff
 766       if score_diff > null_score_diff_threshold:
 767         all_predictions[example.qas_id] = ""
 768       else:
 769         all_predictions[example.qas_id] = best_non_null_entry.text
 770 
 771     all_nbest_json[example.qas_id] = nbest_json
 772 
 773   return all_predictions, all_nbest_json, scores_diff_json

 781 def _get_best_indexes(logits, n_best_size):
 782   """Get the n-best logits from a list."""
 783   index_and_score = sorted(enumerate(logits), key=lambda x: x[1], reverse=True)
 784 
 785   best_indexes = []
 786   for i in range(len(index_and_score)):
 787     if i >= n_best_size:
 788       break
 789     best_indexes.append(index_and_score[i][0])
 790   return best_indexes
 791 
 792 
 793 def _compute_softmax(scores):
 794   """Compute softmax probability over raw logits."""
 795   if not scores:
 796     return []
 797 
 798   max_score = None
 799   for score in scores:
 800     if max_score is None or score > max_score:
 801       max_score = score
 802 
 803   exp_scores = []
 804   total_sum = 0.0
 805   for score in scores:
 806     x = math.exp(score - max_score)
 807     exp_scores.append(x)
 808     total_sum += x
 809 
 810   probs = []
 811   for score in exp_scores:
 812     probs.append(score / total_sum)
 813   return probs

 816 class FeatureWriter(object):
 817   """Writes InputFeature to TF example file."""
 818 
 819   def __init__(self, filename, is_training):
 820     self.filename = filename
 821     self.is_training = is_training
 822     self.num_features = 0
 823     tf.io.gfile.makedirs(os.path.dirname(filename))
 824     self._writer = tf.io.TFRecordWriter(filename)
 825 
 826   def process_feature(self, feature):
 827     """Write a InputFeature to the TFRecordWriter as a tf.train.Example."""
 828     self.num_features += 1
 829 
 830     def create_int_feature(values):
 831       feature = tf.train.Feature(
 832           int64_list=tf.train.Int64List(value=list(values)))
 833       return feature
 834 
 835     features = collections.OrderedDict()
 836     features["unique_ids"] = create_int_feature([feature.unique_id])
 837     features["input_ids"] = create_int_feature(feature.input_ids)
 838     features["input_mask"] = create_int_feature(feature.input_mask)
 839     features["segment_ids"] = create_int_feature(feature.segment_ids)
 840 
 841     if self.is_training:
 842       features["start_positions"] = create_int_feature([feature.start_position])
 843       features["end_positions"] = create_int_feature([feature.end_position])
 844       impossible = 0
 845       if feature.is_impossible:
 846         impossible = 1
 847       features["is_impossible"] = create_int_feature([impossible])
 848 
 849     tf_example = tf.train.Example(features=tf.train.Features(feature=features))
 850     self._writer.write(tf_example.SerializeToString())
 851 
 852   def close(self):
 853     self._writer.close()
