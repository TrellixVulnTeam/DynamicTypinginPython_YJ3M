  23 flags.DEFINE_string("master", default=None, help="master")
  24 flags.DEFINE_string(
  25     "tpu",
  26     default=None,
  27     help="The Cloud TPU to use for training. This should be "
  28     "either the name used when creating the Cloud TPU, or a "
  29     "url like grpc://ip.address.of.tpu:8470.")

  31     "use_tpu", default=True, help="Use TPUs rather than plain CPUs.")

  38     "init_checkpoint",
