  38 def _save_checkpoint(checkpoint, model_dir, checkpoint_prefix):
  39   """Saves model to with provided checkpoint prefix."""
  40 
  41   checkpoint_path = os.path.join(model_dir, checkpoint_prefix)
  42   saved_path = checkpoint.save(checkpoint_path)
  43   logging.info("Saving model as TF checkpoint: %s", saved_path)
  44   return

  52 def train(
  53     strategy: tf.distribute.Strategy,
  54     model_fn: Callable,
  55     input_meta_data: Dict,
  56     train_input_fn: Callable,
  57     total_training_steps: int,
  58     steps_per_loop: int,
  59     optimizer: tf.keras.optimizers.Optimizer,
  60     learning_rate_fn: tf.keras.optimizers.schedules.LearningRateSchedule,
  61     eval_fn: Optional[Callable[[tf.keras.Model, int, tf.summary.SummaryWriter],
  62                                Any]] = None,
  63     metric_fn: Optional[Callable[[], tf.keras.metrics.Metric]] = None,
  64     init_checkpoint: Optional[Text] = None,
  65     init_from_transformerxl: Optional[bool] = False,
  66     model_dir: Optional[Text] = None,
  67     save_steps: Optional[int] = None,
  68     run_eagerly: Optional[bool] = False):
  69   """Runs customized training.
  70 
  71   Args:
  72       strategy: Distribution strategy on which to run low level training loop.
  73       model_fn: The function returns a keras.Model.
  74       input_meta_data: A dictionary of params: `mem_len`, `lr_layer_decay_rate`,
  75         `n_layer`, `batch_size_per_core` and `d_model`.
  76       train_input_fn: Function returns a tf.data.Dataset used for training.
  77       total_training_steps: Number of steps to train in total.
  78       steps_per_loop: Number of steps per graph-mode loop. In order to reduce
  79         communication in eager context, training logs are printed every
  80         steps_per_loop.
  81       optimizer: The optimizer for model.
  82       learning_rate_fn: the learning rate schedule.
  83       eval_fn: A callback of evaluation function, that takes a keras.Model,
  84         current step and evaluation summary writer.
  85       metric_fn: A metrics function returns a Keras Metric object to record
  86         evaluation result using evaluation dataset or with training dataset
  87         after every epoch.
  88       init_checkpoint: Optional checkpoint to load to `sub_model` returned by
  89         `model_fn`.
  90       init_from_transformerxl: Whether to load to `transformerxl_model` of
  91         `model_fn`.
  92       model_dir: The directory of model (checkpoints, summaries).
  93       save_steps: The frequency to save checkpoints. Every save_steps, we save a
  94         model checkpoint. Model checkpoint will be saved and evaluation will be
  95         conducted if evaluation dataset is provided.
  96       run_eagerly: Whether to run training eagerly.
  97 
  98   Returns:
  99       Last training step logits if training happens, otherwise returns None.
 100   Raises:
 101     TypeError: if model directory is not specified.
 102   """
 103   required_arguments = [
 104       train_input_fn, total_training_steps, steps_per_loop, optimizer,
 105       learning_rate_fn, save_steps
 106   ]
 107   if [arg for arg in required_arguments if arg is None]:
 108     raise ValueError("`train_input_fn`, `total_training_steps`, "
 109                      "`steps_per_loop`, `optimizer`, `save_steps` and "
 110                      "`learning_rate_fn` are required parameters.")
 111   if not model_dir:
 112     raise TypeError("Model directory must be specified.")
 113   train_iterator = data_utils.get_input_iterator(train_input_fn, strategy)
 114   if not tf.io.gfile.exists(model_dir):
 115     tf.io.gfile.mkdir(model_dir)
 116   # Create summary writers
 117   summary_dir = os.path.join(model_dir, "summaries")
 118   if not tf.io.gfile.exists(summary_dir):
 119     tf.io.gfile.mkdir(summary_dir)
 120   train_summary_writer = None
 121   eval_summary_writer = None
 122   if eval_fn:
 123     eval_summary_writer = tf.summary.create_file_writer(
 124         os.path.join(summary_dir, "eval"))
 125   if steps_per_loop >= _MIN_SUMMARY_STEPS:
 126     # Only writes summary when the stats are collected sufficiently over
 127     # enough steps.
 128     train_summary_writer = tf.summary.create_file_writer(
 129         os.path.join(summary_dir, "train"))
 130 
 131   with strategy.scope():
 132     model = model_fn()
 133 
 134     if init_checkpoint:
 135       logging.info("restore from %s", init_checkpoint)
 136       if init_from_transformerxl:
 137         checkpoint = tf.train.Checkpoint(
 138             transformer_xl=model.transformerxl_model)
 139       else:
 140         checkpoint = tf.train.Checkpoint(model=model)
 141       checkpoint.restore(init_checkpoint)
 142 
 143     model.optimizer = optimizer
 144 
 145     if not hasattr(model, "optimizer"):
 146       raise ValueError("User should set optimizer attribute to model.")
 147 
 148     train_loss_metric = tf.keras.metrics.Mean("training_loss", dtype=tf.float32)
 149     train_metric = None
 150     if metric_fn:
 151       train_metric = metric_fn()
 152 
 153     def _replicated_step(inputs, mem=None):
 154       """Replicated training step."""
 155 
 156       inputs["mems"] = mem
 157       with tf.GradientTape() as tape:
 158         mem, logits = model(inputs, training=True)
 159         loss = model.losses
 160         train_loss_metric.update_state(loss)
 161         if train_metric:
 162           train_metric.update_state(inputs["label_ids"], logits)
 163         scaled_loss = loss[0] * 1.0 / float(strategy.num_replicas_in_sync)
 164 
 165       # Collects training variables.
 166       tvars = model.trainable_variables
 167       grads = tape.gradient(scaled_loss, tvars)
 168       clipped, _ = tf.clip_by_global_norm(grads, clip_norm=1.0)
 169 
 170       if input_meta_data["lr_layer_decay_rate"] != 1.0:
 171         n_layer = 0
 172         for i in range(len(clipped)):
 173           m = re.search(r"model/transformer/layer_(\d+?)/", tvars[i].name)
 174           if not m:
 175             continue
 176           n_layer = max(n_layer, int(m.group(1)) + 1)
 177 
 178         for i in range(len(clipped)):
 179           for l in range(n_layer):
 180             if "model/transformer/layer_{}/".format(l) in tvars[i].name:
 181               abs_rate = input_meta_data["lr_layer_decay_rate"]**(
 182                   n_layer - 1 - l)
 183               clipped[i] *= abs_rate
 184               logging.info("Apply mult {:.4f} to layer-{} grad of {}".format(
 185                   abs_rate, l, tvars[i].name))
 186               break
 187 
 188       optimizer.apply_gradients(zip(clipped, tvars))
 189       if input_meta_data["mem_len"] > 0:
 190         return mem
 191 
 192     def train_steps(iterator, steps):
 193       """Performs distributed training steps in a loop.
 194 
 195       Args:
 196         iterator: the distributed iterator of training datasets.
 197         steps: an tf.int32 integer tensor to specify number of steps to run
 198           inside host training loop.
 199 
 200       Raises:
 201         ValueError: Any of the arguments or tensor shapes are invalid.
 202 
 203       Returns:
 204         logits: logits computed.
 205       """
 206       if not isinstance(steps, tf.Tensor):
 207         raise ValueError("steps should be an Tensor. Python object may cause "
 208                          "retracing.")
 209 
 210       def cache_fn():
 211         """Initializes memory tensor used in XLNet pretraining."""
 212         mems = []
 213         if input_meta_data["mem_len"] > 0:
 214           for _ in range(input_meta_data["n_layer"]):
 215             zeros = tf.zeros([
 216                 input_meta_data["mem_len"],
 217                 input_meta_data["batch_size_per_core"],
 218                 input_meta_data["d_model"]
 219             ],
 220                              dtype=tf.float32)
 221             mems.append(zeros)
 222         return mems
 223 
 224       if input_meta_data["mem_len"] > 0:
 225         mem = strategy.run(cache_fn)
 226         for _ in tf.range(steps):
 227           mem = strategy.run(
 228               _replicated_step, args=(
 229                   next(iterator),
 230                   mem,
 231               ))
 232       else:
 233         for _ in tf.range(steps):
 234           strategy.run(_replicated_step, args=(next(iterator),))
 235 
 236     if not run_eagerly:
 237       train_steps = tf.function(train_steps)
 238 
 239     logging.info("Start training...")
 240     checkpoint = tf.train.Checkpoint(model=model, optimizer=optimizer)
 241     latest_checkpoint_file = tf.train.latest_checkpoint(model_dir)
 242     if latest_checkpoint_file:
 243       logging.info("Checkpoint file %s found and restoring from checkpoint",
 244                    latest_checkpoint_file)
 245       checkpoint.restore(latest_checkpoint_file)
 246       logging.info("Loading from checkpoint file completed")
 247 
 248     current_step = optimizer.iterations.numpy()
 249     checkpoint_name = "xlnet_step_{step}.ckpt"
 250 
 251     while current_step < total_training_steps:
 252       train_loss_metric.reset_states()
 253       if train_metric:
 254         train_metric.reset_states()
 255 
 256       steps = model_training_utils.steps_to_run(current_step, save_steps,
 257                                                 steps_per_loop)
 258       train_steps(train_iterator, tf.convert_to_tensor(steps, dtype=tf.int32))
 259       current_step += steps
 260       train_loss = _float_metric_value(train_loss_metric)
 261       log_stream = "Train step: %d/%d  /  lr = %.9f  /  loss = %.7f" % (
 262           current_step, total_training_steps, learning_rate_fn(current_step),
 263           train_loss)
 264       if train_metric:
 265         log_stream += "  /  %s = %f" % (train_metric.name,
 266                                         _float_metric_value(train_metric))
 267       logging.info(log_stream)
 268       if train_summary_writer:
 269         with train_summary_writer.as_default():
 270           tf.summary.scalar(
 271               "learning_rate",
 272               learning_rate_fn(current_step),
 273               step=current_step)
 274           tf.summary.scalar(
 275               train_loss_metric.name, train_loss, step=current_step)
 276           if train_metric:
 277             tf.summary.scalar(
 278                 train_metric.name,
 279                 _float_metric_value(train_metric),
 280                 step=current_step)
 281           train_summary_writer.flush()
 282       if model_dir and current_step % save_steps == 0:
 283         _save_checkpoint(checkpoint, model_dir,
 284                          checkpoint_name.format(step=current_step))
 285 
 286       if eval_fn and current_step % save_steps == 0:
 287 
 288         logging.info("Running evaluation after step: %s.", current_step)
 289 
 290         eval_fn(model, current_step, eval_summary_writer)
 291     if model_dir:
 292       _save_checkpoint(checkpoint, model_dir,
 293                        checkpoint_name.format(step=current_step))
 294     if eval_fn:
 295       logging.info("Running final evaluation after training is complete.")
 296       eval_metric = eval_fn(model, current_step, eval_summary_writer)
 297 
 298     training_summary = {
 299         "total_training_steps": total_training_steps,
 300         "train_loss": _float_metric_value(train_loss_metric),
 301     }
 302     if train_metric:
 303       training_summary["last_train_metrics"] = _float_metric_value(train_metric)
 304     if eval_fn:
 305       # eval_metric is supposed to be a float.
 306       training_summary["eval_metrics"] = eval_metric
 307 
 308     model_training_utils.write_txt_summary(training_summary, summary_dir)
 309 
 310     return model
