  20 SEG_ID_A = 0

  24 class PaddingInputExample(object):
  25   """Fake example so the num input examples is a multiple of the batch size.
  26 
  27   When running eval/predict on the TPU, we need to pad the number of examples
  28   to be a multiple of the batch size, because the TPU requires a fixed batch
  29   size. The alternative is to drop the last batch, which is bad because it means
  30   the entire output data won't be generated.
  31   We use this class instead of `None` because treating `None` as padding
  32   battches could cause silent errors.
  33   """
  34

  36 class InputFeatures(object):
  37   """A single set of features of data."""
  38 
  39   def __init__(self,
  40                input_ids,
  41                input_mask,
  42                segment_ids,
  43                label_id,
  44                is_real_example=True):
  45     self.input_ids = input_ids
  46     self.input_mask = input_mask
  47     self.segment_ids = segment_ids
  48     self.label_id = label_id
  49     self.is_real_example = is_real_example

  52 def _truncate_seq_pair(tokens_a, tokens_b, max_length):
  53   """Truncates a sequence pair in place to the maximum length."""
  54 
  55   # This is a simple heuristic which will always truncate the longer sequence
  56   # one token at a time. This makes more sense than truncating an equal percent
  57   # of tokens from each, since if one sequence is very short then each token
  58   # that's truncated likely contains more information than a longer sequence.
  59   while True:
  60     total_length = len(tokens_a) + len(tokens_b)
  61     if total_length <= max_length:
  62       break
  63     if len(tokens_a) > len(tokens_b):
  64       tokens_a.pop()
  65     else:
  66       tokens_b.pop()
  67

  69 def convert_single_example(example_index, example, label_list, max_seq_length,
  70                            tokenize_fn, use_bert_format):
  71   """Converts a single `InputExample` into a single `InputFeatures`."""
  72 
  73   if isinstance(example, PaddingInputExample):
  74     return InputFeatures(
  75         input_ids=[0] * max_seq_length,
  76         input_mask=[1] * max_seq_length,
  77         segment_ids=[0] * max_seq_length,
  78         label_id=0,
  79         is_real_example=False)
  80 
  81   if label_list is not None:
  82     label_map = {}
  83     for (i, label) in enumerate(label_list):
  84       label_map[label] = i
  85 
  86   tokens_a = tokenize_fn(example.text_a)
  87   tokens_b = None
  88   if example.text_b:
  89     tokens_b = tokenize_fn(example.text_b)
  90 
  91   if tokens_b:
  92     # Modifies `tokens_a` and `tokens_b` in place so that the total
  93     # length is less than the specified length.
  94     # Account for two [SEP] & one [CLS] with "- 3"
  95     _truncate_seq_pair(tokens_a, tokens_b, max_seq_length - 3)
  96   else:
  97     # Account for one [SEP] & one [CLS] with "- 2"
  98     if len(tokens_a) > max_seq_length - 2:
  99       tokens_a = tokens_a[:max_seq_length - 2]
 100 
 101   tokens = []
 102   segment_ids = []
 103   for token in tokens_a:
 104     tokens.append(token)
 105     segment_ids.append(SEG_ID_A)
 106   tokens.append(data_utils.SEP_ID)
 107   segment_ids.append(SEG_ID_A)
 108 
 109   if tokens_b:
 110     for token in tokens_b:
 111       tokens.append(token)
 112       segment_ids.append(SEG_ID_B)
 113     tokens.append(data_utils.SEP_ID)
 114     segment_ids.append(SEG_ID_B)
 115 
 116   if use_bert_format:
 117     tokens.insert(0, data_utils.CLS_ID)
 118     segment_ids.insert(0, data_utils.SEG_ID_CLS)
 119   else:
 120     tokens.append(data_utils.CLS_ID)
 121     segment_ids.append(data_utils.SEG_ID_CLS)
 122 
 123   input_ids = tokens
 124 
 125   # The mask has 0 for real tokens and 1 for padding tokens. Only real
 126   # tokens are attended to.
 127   input_mask = [0] * len(input_ids)
 128 
 129   # Zero-pad up to the sequence length.
 130   if len(input_ids) < max_seq_length:
 131     delta_len = max_seq_length - len(input_ids)
 132     if use_bert_format:
 133       input_ids = input_ids + [0] * delta_len
 134       input_mask = input_mask + [1] * delta_len
 135       segment_ids = segment_ids + [data_utils.SEG_ID_PAD] * delta_len
 136     else:
 137       input_ids = [0] * delta_len + input_ids
 138       input_mask = [1] * delta_len + input_mask
 139       segment_ids = [data_utils.SEG_ID_PAD] * delta_len + segment_ids
 140 
 141   assert len(input_ids) == max_seq_length
 142   assert len(input_mask) == max_seq_length
 143   assert len(segment_ids) == max_seq_length
 144 
 145   if label_list is not None:
 146     label_id = label_map[example.label]
 147   else:
 148     label_id = example.label
 149   if example_index < 5:
 150     logging.info("*** Example ***")
 151     logging.info("guid: %s", (example.guid))
 152     logging.info("input_ids: %s", " ".join([str(x) for x in input_ids]))
 153     logging.info("input_mask: %s", " ".join([str(x) for x in input_mask]))
 154     logging.info("segment_ids: %s", " ".join([str(x) for x in segment_ids]))
 155     logging.info("label: %d (id = %d)", example.label, label_id)
 156 
 157   feature = InputFeatures(
 158       input_ids=input_ids,
 159       input_mask=input_mask,
 160       segment_ids=segment_ids,
 161       label_id=label_id)
 162   return feature
