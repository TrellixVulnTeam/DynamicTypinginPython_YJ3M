  56 def run_evaluation(strategy,
  57                    test_input_fn,
  58                    eval_steps,
  59                    model,
  60                    step,
  61                    eval_summary_writer=None):
  62   """Run evaluation for classification task.
  63 
  64   Args:
  65     strategy: distribution strategy.
  66     test_input_fn: input function for evaluation data.
  67     eval_steps: total number of evaluation steps.
  68     model: keras model object.
  69     step: current train step.
  70     eval_summary_writer: summary writer used to record evaluation metrics.  As
  71       there are fake data samples in validation set, we use mask to get rid of
  72       them when calculating the accuracy. For the reason that there will be
  73       dynamic-shape tensor, we first collect logits, labels and masks from TPU
  74       and calculate the accuracy via numpy locally.
  75 
  76   Returns:
  77     A float metric, accuracy.
  78   """
  79 
  80   def _test_step_fn(inputs):
  81     """Replicated validation step."""
  82 
  83     inputs["mems"] = None
  84     _, logits = model(inputs, training=False)
  85     return logits, inputs["label_ids"], inputs["is_real_example"]
  86 
  87   @tf.function
  88   def _run_evaluation(test_iterator):
  89     """Runs validation steps."""
  90     logits, labels, masks = strategy.run(
  91         _test_step_fn, args=(next(test_iterator),))
  92     return logits, labels, masks
  93 
  94   test_iterator = data_utils.get_input_iterator(test_input_fn, strategy)
  95   correct = 0
  96   total = 0
  97   for _ in range(eval_steps):
  98     logits, labels, masks = _run_evaluation(test_iterator)
  99     logits = strategy.experimental_local_results(logits)
 100     labels = strategy.experimental_local_results(labels)
 101     masks = strategy.experimental_local_results(masks)
 102     merged_logits = []
 103     merged_labels = []
 104     merged_masks = []
 105 
 106     for i in range(strategy.num_replicas_in_sync):
 107       merged_logits.append(logits[i].numpy())
 108       merged_labels.append(labels[i].numpy())
 109       merged_masks.append(masks[i].numpy())
 110     merged_logits = np.vstack(np.array(merged_logits))
 111     merged_labels = np.hstack(np.array(merged_labels))
 112     merged_masks = np.hstack(np.array(merged_masks))
 113     real_index = np.where(np.equal(merged_masks, 1))
 114     correct += np.sum(
 115         np.equal(
 116             np.argmax(merged_logits[real_index], axis=-1),
 117             merged_labels[real_index]))
 118     total += np.shape(real_index)[-1]
 119   accuracy = float(correct) / float(total)
 120   logging.info("Train step: %d  /  acc = %d/%d = %f", step, correct, total,
 121                accuracy)
 122   if eval_summary_writer:
 123     with eval_summary_writer.as_default():
 124       tf.summary.scalar("eval_acc", float(correct) / float(total), step=step)
 125       eval_summary_writer.flush()
 126   return accuracy

 129 def get_metric_fn():
 130   train_acc_metric = tf.keras.metrics.SparseCategoricalAccuracy(
 131       "acc", dtype=tf.float32)
 132   return train_acc_metric

 135 def main(unused_argv):
 136   del unused_argv
 137   if FLAGS.strategy_type == "mirror":
 138     strategy = tf.distribute.MirroredStrategy()
 139   elif FLAGS.strategy_type == "tpu":
 140     cluster_resolver = tpu_lib.tpu_initialize(FLAGS.tpu)
 141     strategy = tf.distribute.experimental.TPUStrategy(cluster_resolver)
 142   else:
 143     raise ValueError("The distribution strategy type is not supported: %s" %
 144                      FLAGS.strategy_type)
 145   if strategy:
 146     logging.info("***** Number of cores used : %d",
 147                  strategy.num_replicas_in_sync)
 148   train_input_fn = functools.partial(data_utils.get_classification_input_data,
 149                                      FLAGS.train_batch_size, FLAGS.seq_len,
 150                                      strategy, True, FLAGS.train_tfrecord_path)
 151   test_input_fn = functools.partial(data_utils.get_classification_input_data,
 152                                     FLAGS.test_batch_size, FLAGS.seq_len,
 153                                     strategy, False, FLAGS.test_tfrecord_path)
 154 
 155   total_training_steps = FLAGS.train_steps
 156   steps_per_loop = FLAGS.iterations
 157   eval_steps = int(FLAGS.test_data_size / FLAGS.test_batch_size)
 158   eval_fn = functools.partial(run_evaluation, strategy, test_input_fn,
 159                               eval_steps)
 160   optimizer, learning_rate_fn = optimization.create_optimizer(
 161       FLAGS.learning_rate,
 162       total_training_steps,
 163       FLAGS.warmup_steps,
 164       adam_epsilon=FLAGS.adam_epsilon)
 165   model_config = xlnet_config.XLNetConfig(FLAGS)
 166   run_config = xlnet_config.create_run_config(True, False, FLAGS)
 167   model_fn = functools.partial(get_classificationxlnet_model, model_config,
 168                                run_config, FLAGS.n_class, FLAGS.summary_type)
 169   input_meta_data = {}
 170   input_meta_data["d_model"] = FLAGS.d_model
 171   input_meta_data["mem_len"] = FLAGS.mem_len
 172   input_meta_data["batch_size_per_core"] = int(FLAGS.train_batch_size /
 173                                                strategy.num_replicas_in_sync)
 174   input_meta_data["n_layer"] = FLAGS.n_layer
 175   input_meta_data["lr_layer_decay_rate"] = FLAGS.lr_layer_decay_rate
 176   input_meta_data["n_class"] = FLAGS.n_class
 177 
 178   training_utils.train(
 179       strategy=strategy,
 180       model_fn=model_fn,
 181       input_meta_data=input_meta_data,
 182       eval_fn=eval_fn,
 183       metric_fn=get_metric_fn,
 184       train_input_fn=train_input_fn,
 185       init_checkpoint=FLAGS.init_checkpoint,
 186       init_from_transformerxl=FLAGS.init_from_transformerxl,
 187       total_training_steps=total_training_steps,
 188       steps_per_loop=steps_per_loop,
 189       optimizer=optimizer,
 190       learning_rate_fn=learning_rate_fn,
 191       model_dir=FLAGS.model_dir,
 192       save_steps=FLAGS.save_steps)
 193 
 194 
 195 if __name__ == "__main__":
 196   app.run(main)
