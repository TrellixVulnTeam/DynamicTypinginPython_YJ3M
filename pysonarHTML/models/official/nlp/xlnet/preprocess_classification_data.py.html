  66 class InputExample(object):
  67   """A single training/test example for simple sequence classification."""
  68 
  69   def __init__(self, guid, text_a, text_b=None, label=None):
  70     """Constructs a InputExample.
  71 
  72     Args:
  73       guid: Unique id for the example.
  74       text_a: string. The untokenized text of the first sequence. For single
  75         sequence tasks, only this sequence must be specified.
  76       text_b: (Optional) string. The untokenized text of the second sequence.
  77         Only must be specified for sequence pair tasks.
  78       label: (Optional) string. The label of the example. This should be
  79         specified for train and dev examples, but not for test examples.
  80     """
  81     self.guid = guid
  82     self.text_a = text_a
  83     self.text_b = text_b
  84     self.label = label

  87 class DataProcessor(object):
  88   """Base class for data converters for sequence classification data sets."""
  89 
  90   def get_train_examples(self, data_dir):
  91     """Gets a collection of `InputExample`s for the train set."""
  92     raise NotImplementedError()
  93 
  94   def get_dev_examples(self, data_dir):
  95     """Gets a collection of `InputExample`s for the dev set."""
  96     raise NotImplementedError()
  97 
  98   def get_test_examples(self, data_dir):
  99     """Gets a collection of `InputExample`s for prediction."""
 100     raise NotImplementedError()
 101 
 102   def get_labels(self):
 103     """Gets the list of labels for this data set."""
 104     raise NotImplementedError()
 105 
 106   @classmethod
 107   def _read_tsv(cls, input_file, quotechar=None):
 108     """Reads a tab separated value file."""
 109     with tf.io.gfile.GFile(input_file, "r") as f:
 110       reader = csv.reader(f, delimiter="\t", quotechar=quotechar)
 111       lines = []
 112       for line in reader:
 113         # pylint: disable=g-explicit-length-test
 114         if len(line) == 0:
 115           continue
 116         lines.append(line)
 117       return lines

 120 class GLUEProcessor(DataProcessor):
 121   """GLUEProcessor."""
 122 
 123   def __init__(self):
 124     self.train_file = "train.tsv"
 125     self.dev_file = "dev.tsv"
 126     self.test_file = "test.tsv"
 127     self.label_column = None
 128     self.text_a_column = None
 129     self.text_b_column = None
 130     self.contains_header = True
 131     self.test_text_a_column = None
 132     self.test_text_b_column = None
 133     self.test_contains_header = True
 134 
 135   def get_train_examples(self, data_dir):
 136     """See base class."""
 137     return self._create_examples(
 138         self._read_tsv(os.path.join(data_dir, self.train_file)), "train")
 139 
 140   def get_dev_examples(self, data_dir):
 141     """See base class."""
 142     return self._create_examples(
 143         self._read_tsv(os.path.join(data_dir, self.dev_file)), "dev")
 144 
 145   def get_test_examples(self, data_dir):
 146     """See base class."""
 147     if self.test_text_a_column is None:
 148       self.test_text_a_column = self.text_a_column
 149     if self.test_text_b_column is None:
 150       self.test_text_b_column = self.text_b_column
 151 
 152     return self._create_examples(
 153         self._read_tsv(os.path.join(data_dir, self.test_file)), "test")
 154 
 155   def get_labels(self):
 156     """See base class."""
 157     return ["0", "1"]
 158 
 159   def _create_examples(self, lines, set_type):
 160     """Creates examples for the training and dev sets."""
 161     examples = []
 162     for (i, line) in enumerate(lines):
 163       if i == 0 and self.contains_header and set_type != "test":
 164         continue
 165       if i == 0 and self.test_contains_header and set_type == "test":
 166         continue
 167       guid = "%s-%s" % (set_type, i)
 168 
 169       a_column = (
 170           self.text_a_column if set_type != "test" else self.test_text_a_column)
 171       b_column = (
 172           self.text_b_column if set_type != "test" else self.test_text_b_column)
 173 
 174       # there are some incomplete lines in QNLI
 175       if len(line) <= a_column:
 176         logging.warning("Incomplete line, ignored.")
 177         continue
 178       text_a = line[a_column]
 179 
 180       if b_column is not None:
 181         if len(line) <= b_column:
 182           logging.warning("Incomplete line, ignored.")
 183           continue
 184         text_b = line[b_column]
 185       else:
 186         text_b = None
 187 
 188       if set_type == "test":
 189         label = self.get_labels()[0]
 190       else:
 191         if len(line) <= self.label_column:
 192           logging.warning("Incomplete line, ignored.")
 193           continue
 194         label = line[self.label_column]
 195       examples.append(
 196           InputExample(guid=guid, text_a=text_a, text_b=text_b, label=label))
 197     return examples

 200 class Yelp5Processor(DataProcessor):
 201   """Yelp5Processor."""
 202 
 203   def get_train_examples(self, data_dir):
 204     return self._create_examples(os.path.join(data_dir, "train.csv"))
 205 
 206   def get_dev_examples(self, data_dir):
 207     return self._create_examples(os.path.join(data_dir, "test.csv"))
 208 
 209   def get_labels(self):
 210     """See base class."""
 211     return ["1", "2", "3", "4", "5"]
 212 
 213   def _create_examples(self, input_file):
 214     """Creates examples for the training and dev sets."""
 215     examples = []
 216     with tf.io.gfile.GFile(input_file) as f:
 217       reader = csv.reader(f)
 218       for i, line in enumerate(reader):
 219 
 220         label = line[0]
 221         text_a = line[1].replace('""', '"').replace('\\"', '"')
 222         examples.append(
 223             InputExample(guid=str(i), text_a=text_a, text_b=None, label=label))
 224     return examples

 227 class ImdbProcessor(DataProcessor):
 228   """ImdbProcessor."""
 229 
 230   def get_labels(self):
 231     return ["neg", "pos"]
 232 
 233   def get_train_examples(self, data_dir):
 234     return self._create_examples(os.path.join(data_dir, "train"))
 235 
 236   def get_dev_examples(self, data_dir):
 237     return self._create_examples(os.path.join(data_dir, "test"))
 238 
 239   def _create_examples(self, data_dir):
 240     """Creates examples."""
 241     examples = []
 242     for label in ["neg", "pos"]:
 243       cur_dir = os.path.join(data_dir, label)
 244       for filename in tf.io.gfile.listdir(cur_dir):
 245         if not filename.endswith("txt"):
 246           continue
 247 
 248         if len(examples) % 1000 == 0:
 249           logging.info("Loading dev example %d", len(examples))
 250 
 251         path = os.path.join(cur_dir, filename)
 252         with tf.io.gfile.GFile(path) as f:
 253           text = f.read().strip().replace("<br />", " ")
 254         examples.append(
 255             InputExample(
 256                 guid="unused_id", text_a=text, text_b=None, label=label))
 257     return examples

 337 def file_based_convert_examples_to_features(examples,
 338                                             label_list,
 339                                             max_seq_length,
 340                                             tokenize_fn,
 341                                             output_file,
 342                                             num_passes=1):
 343   """Convert a set of `InputExample`s to a TFRecord file."""
 344 
 345   # do not create duplicated records
 346   if tf.io.gfile.exists(output_file) and not FLAGS.overwrite_data:
 347     logging.info("Do not overwrite tfrecord %s exists.", output_file)
 348     return
 349 
 350   logging.info("Create new tfrecord %s.", output_file)
 351 
 352   writer = tf.io.TFRecordWriter(output_file)
 353 
 354   examples *= num_passes
 355 
 356   for (ex_index, example) in enumerate(examples):
 357     if ex_index % 10000 == 0:
 358       logging.info("Writing example %d of %d", ex_index, len(examples))
 359 
 360     feature = classifier_utils.convert_single_example(ex_index, example,
 361                                                       label_list,
 362                                                       max_seq_length,
 363                                                       tokenize_fn,
 364                                                       FLAGS.use_bert_format)
 365 
 366     def create_int_feature(values):
 367       f = tf.train.Feature(int64_list=tf.train.Int64List(value=list(values)))
 368       return f
 369 
 370     def create_float_feature(values):
 371       f = tf.train.Feature(float_list=tf.train.FloatList(value=list(values)))
 372       return f
 373 
 374     features = collections.OrderedDict()
 375     features["input_ids"] = create_int_feature(feature.input_ids)
 376     features["input_mask"] = create_float_feature(feature.input_mask)
 377     features["segment_ids"] = create_int_feature(feature.segment_ids)
 378     if label_list is not None:
 379       features["label_ids"] = create_int_feature([feature.label_id])
 380     else:
 381       features["label_ids"] = create_float_feature([float(feature.label_id)])
 382     features["is_real_example"] = create_int_feature(
 383         [int(feature.is_real_example)])
 384 
 385     tf_example = tf.train.Example(features=tf.train.Features(feature=features))
 386     writer.write(tf_example.SerializeToString())
 387   writer.close()

 390 def main(_):
 391   logging.set_verbosity(logging.INFO)
 392   processors = {
 393       "mnli_matched": MnliMatchedProcessor,
 394       "mnli_mismatched": MnliMismatchedProcessor,
 395       "sts-b": StsbProcessor,
 396       "imdb": ImdbProcessor,
 397       "yelp5": Yelp5Processor
 398   }
 399 
 400   task_name = FLAGS.task_name.lower()
 401 
 402   if task_name not in processors:
 403     raise ValueError("Task not found: %s" % (task_name))
 404 
 405   processor = processors[task_name]()
 406   label_list = processor.get_labels() if not FLAGS.is_regression else None
 407 
 408   sp = spm.SentencePieceProcessor()
 409   sp.Load(FLAGS.spiece_model_file)
 410 
 411   def tokenize_fn(text):
 412     text = preprocess_utils.preprocess_text(text, lower=FLAGS.uncased)
 413     return preprocess_utils.encode_ids(sp, text)
 414 
 415   spm_basename = os.path.basename(FLAGS.spiece_model_file)
 416 
 417   train_file_base = "{}.len-{}.train.tf_record".format(spm_basename,
 418                                                        FLAGS.max_seq_length)
 419   train_file = os.path.join(FLAGS.output_dir, train_file_base)
 420   logging.info("Use tfrecord file %s", train_file)
 421 
 422   train_examples = processor.get_train_examples(FLAGS.data_dir)
 423   np.random.shuffle(train_examples)
 424   logging.info("Num of train samples: %d", len(train_examples))
 425 
 426   file_based_convert_examples_to_features(train_examples, label_list,
 427                                           FLAGS.max_seq_length, tokenize_fn,
 428                                           train_file, FLAGS.num_passes)
 429   if FLAGS.eval_split == "dev":
 430     eval_examples = processor.get_dev_examples(FLAGS.data_dir)
 431   else:
 432     eval_examples = processor.get_test_examples(FLAGS.data_dir)
 433 
 434   logging.info("Num of eval samples: %d", len(eval_examples))
 435 
 436   # TPU requires a fixed batch size for all batches, therefore the number
 437   # of examples must be a multiple of the batch size, or else examples
 438   # will get dropped. So we pad with fake examples which are ignored
 439   # later on. These do NOT count towards the metric (all tf.metrics
 440   # support a per-instance weight, and these get a weight of 0.0).
 441   #
 442   # Modified in XL: We also adopt the same mechanism for GPUs.
 443   while len(eval_examples) % FLAGS.eval_batch_size != 0:
 444     eval_examples.append(classifier_utils.PaddingInputExample())
 445 
 446   eval_file_base = "{}.len-{}.{}.eval.tf_record".format(spm_basename,
 447                                                         FLAGS.max_seq_length,
 448                                                         FLAGS.eval_split)
 449   eval_file = os.path.join(FLAGS.output_dir, eval_file_base)
 450 
 451   file_based_convert_examples_to_features(eval_examples, label_list,
 452                                           FLAGS.max_seq_length, tokenize_fn,
 453                                           eval_file)
 454 
 455 
 456 if __name__ == "__main__":
 457   app.run(main)
