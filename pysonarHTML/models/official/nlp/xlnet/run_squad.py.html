 102 def run_evaluation(strategy, test_input_fn, eval_examples, eval_features,
 103                    original_data, eval_steps, input_meta_data, model,
 104                    current_step, eval_summary_writer):
 105   """Run evaluation for SQUAD task.
 106 
 107   Args:
 108     strategy: distribution strategy.
 109     test_input_fn: input function for evaluation data.
 110     eval_examples: tf.Examples of the evaluation set.
 111     eval_features: Feature objects of the evaluation set.
 112     original_data: The original json data for the evaluation set.
 113     eval_steps: total number of evaluation steps.
 114     input_meta_data: input meta data.
 115     model: keras model object.
 116     current_step: current training step.
 117     eval_summary_writer: summary writer used to record evaluation metrics.
 118 
 119   Returns:
 120     A float metric, F1 score.
 121   """
 122 
 123   def _test_step_fn(inputs):
 124     """Replicated validation step."""
 125 
 126     inputs["mems"] = None
 127     res = model(inputs, training=False)
 128     return res, inputs["unique_ids"]
 129 
 130   @tf.function
 131   def _run_evaluation(test_iterator):
 132     """Runs validation steps."""
 133     res, unique_ids = strategy.run(
 134         _test_step_fn, args=(next(test_iterator),))
 135     return res, unique_ids
 136 
 137   test_iterator = data_utils.get_input_iterator(test_input_fn, strategy)
 138   cur_results = []
 139   for _ in range(eval_steps):
 140     results, unique_ids = _run_evaluation(test_iterator)
 141     unique_ids = strategy.experimental_local_results(unique_ids)
 142 
 143     for result_key in results:
 144       results[result_key] = (
 145           strategy.experimental_local_results(results[result_key]))
 146     for core_i in range(strategy.num_replicas_in_sync):
 147       bsz = int(input_meta_data["test_batch_size"] /
 148                 strategy.num_replicas_in_sync)
 149       for j in range(bsz):
 150         result = {}
 151         for result_key in results:
 152           result[result_key] = results[result_key][core_i].numpy()[j]
 153         result["unique_ids"] = unique_ids[core_i].numpy()[j]
 154         # We appended a fake example into dev set to make data size can be
 155         # divided by test_batch_size. Ignores this fake example during
 156         # evaluation.
 157         if result["unique_ids"] == 1000012047:
 158           continue
 159         unique_id = int(result["unique_ids"])
 160 
 161         start_top_log_probs = ([
 162             float(x) for x in result["start_top_log_probs"].flat
 163         ])
 164         start_top_index = [int(x) for x in result["start_top_index"].flat]
 165         end_top_log_probs = ([
 166             float(x) for x in result["end_top_log_probs"].flat
 167         ])
 168         end_top_index = [int(x) for x in result["end_top_index"].flat]
 169 
 170         cls_logits = float(result["cls_logits"].flat[0])
 171         cur_results.append(
 172             squad_utils.RawResult(
 173                 unique_id=unique_id,
 174                 start_top_log_probs=start_top_log_probs,
 175                 start_top_index=start_top_index,
 176                 end_top_log_probs=end_top_log_probs,
 177                 end_top_index=end_top_index,
 178                 cls_logits=cls_logits))
 179         if len(cur_results) % 1000 == 0:
 180           logging.info("Processing example: %d", len(cur_results))
 181 
 182   output_prediction_file = os.path.join(input_meta_data["predict_dir"],
 183                                         "predictions.json")
 184   output_nbest_file = os.path.join(input_meta_data["predict_dir"],
 185                                    "nbest_predictions.json")
 186   output_null_log_odds_file = os.path.join(input_meta_data["predict_dir"],
 187                                            "null_odds.json")
 188 
 189   results = squad_utils.write_predictions(
 190       eval_examples, eval_features, cur_results, input_meta_data["n_best_size"],
 191       input_meta_data["max_answer_length"], output_prediction_file,
 192       output_nbest_file, output_null_log_odds_file, original_data,
 193       input_meta_data["start_n_top"], input_meta_data["end_n_top"])
 194 
 195   # Log current results.
 196   log_str = "Result | "
 197   for key, val in results.items():
 198     log_str += "{} {} | ".format(key, val)
 199   logging.info(log_str)
 200   with eval_summary_writer.as_default():
 201     tf.summary.scalar("best_f1", results["best_f1"], step=current_step)
 202     tf.summary.scalar("best_exact", results["best_exact"], step=current_step)
 203     eval_summary_writer.flush()
 204   return results["best_f1"]

 207 def get_qaxlnet_model(model_config, run_config, start_n_top, end_n_top):
 208   model = modeling.QAXLNetModel(
 209       model_config,
 210       run_config,
 211       start_n_top=start_n_top,
 212       end_n_top=end_n_top,
 213       name="model")
 214   return model

 217 def main(unused_argv):
 218   del unused_argv
 219   if FLAGS.strategy_type == "mirror":
 220     strategy = tf.distribute.MirroredStrategy()
 221   elif FLAGS.strategy_type == "tpu":
 222     cluster_resolver = tpu_lib.tpu_initialize(FLAGS.tpu)
 223     strategy = tf.distribute.experimental.TPUStrategy(cluster_resolver)
 224   else:
 225     raise ValueError("The distribution strategy type is not supported: %s" %
 226                      FLAGS.strategy_type)
 227   if strategy:
 228     logging.info("***** Number of cores used : %d",
 229                  strategy.num_replicas_in_sync)
 230   train_input_fn = functools.partial(data_utils.get_squad_input_data,
 231                                      FLAGS.train_batch_size, FLAGS.seq_len,
 232                                      FLAGS.query_len, strategy, True,
 233                                      FLAGS.train_tfrecord_path)
 234 
 235   test_input_fn = functools.partial(data_utils.get_squad_input_data,
 236                                     FLAGS.test_batch_size, FLAGS.seq_len,
 237                                     FLAGS.query_len, strategy, False,
 238                                     FLAGS.test_tfrecord_path)
 239 
 240   total_training_steps = FLAGS.train_steps
 241   steps_per_loop = FLAGS.iterations
 242   eval_steps = int(FLAGS.test_data_size / FLAGS.test_batch_size)
 243 
 244   optimizer, learning_rate_fn = optimization.create_optimizer(
 245       FLAGS.learning_rate,
 246       total_training_steps,
 247       FLAGS.warmup_steps,
 248       adam_epsilon=FLAGS.adam_epsilon)
 249   model_config = xlnet_config.XLNetConfig(FLAGS)
 250   run_config = xlnet_config.create_run_config(True, False, FLAGS)
 251   input_meta_data = {}
 252   input_meta_data["start_n_top"] = FLAGS.start_n_top
 253   input_meta_data["end_n_top"] = FLAGS.end_n_top
 254   input_meta_data["lr_layer_decay_rate"] = FLAGS.lr_layer_decay_rate
 255   input_meta_data["predict_dir"] = FLAGS.predict_dir
 256   input_meta_data["n_best_size"] = FLAGS.n_best_size
 257   input_meta_data["max_answer_length"] = FLAGS.max_answer_length
 258   input_meta_data["test_batch_size"] = FLAGS.test_batch_size
 259   input_meta_data["batch_size_per_core"] = int(FLAGS.train_batch_size /
 260                                                strategy.num_replicas_in_sync)
 261   input_meta_data["mem_len"] = FLAGS.mem_len
 262   model_fn = functools.partial(get_qaxlnet_model, model_config, run_config,
 263                                FLAGS.start_n_top, FLAGS.end_n_top)
 264   eval_examples = squad_utils.read_squad_examples(
 265       FLAGS.predict_file, is_training=False)
 266   if FLAGS.test_feature_path:
 267     logging.info("start reading pickle file...")
 268     with tf.io.gfile.GFile(FLAGS.test_feature_path, "rb") as f:
 269       eval_features = pickle.load(f)
 270     logging.info("finishing reading pickle file...")
 271   else:
 272     sp_model = spm.SentencePieceProcessor()
 273     sp_model.LoadFromSerializedProto(
 274         tf.io.gfile.GFile(FLAGS.spiece_model_file, "rb").read())
 275     spm_basename = os.path.basename(FLAGS.spiece_model_file)
 276     eval_features = squad_utils.create_eval_data(
 277         spm_basename, sp_model, eval_examples, FLAGS.max_seq_length,
 278         FLAGS.max_query_length, FLAGS.doc_stride, FLAGS.uncased)
 279 
 280   with tf.io.gfile.GFile(FLAGS.predict_file) as f:
 281     original_data = json.load(f)["data"]
 282   eval_fn = functools.partial(run_evaluation, strategy, test_input_fn,
 283                               eval_examples, eval_features, original_data,
 284                               eval_steps, input_meta_data)
 285 
 286   training_utils.train(
 287       strategy=strategy,
 288       model_fn=model_fn,
 289       input_meta_data=input_meta_data,
 290       eval_fn=eval_fn,
 291       metric_fn=None,
 292       train_input_fn=train_input_fn,
 293       init_checkpoint=FLAGS.init_checkpoint,
 294       init_from_transformerxl=FLAGS.init_from_transformerxl,
 295       total_training_steps=total_training_steps,
 296       steps_per_loop=steps_per_loop,
 297       optimizer=optimizer,
 298       learning_rate_fn=learning_rate_fn,
 299       model_dir=FLAGS.model_dir,
 300       save_steps=FLAGS.save_steps)
 301 
 302 
 303 if __name__ == "__main__":
 304   app.run(main)
