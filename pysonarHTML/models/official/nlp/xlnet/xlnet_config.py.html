  28 def create_run_config(is_training, is_finetune, flags):
  29   """Helper function for creating RunConfig."""
  30   kwargs = dict(
  31       is_training=is_training,
  32       use_tpu=flags.use_tpu,
  33       dropout=flags.dropout,
  34       dropout_att=flags.dropout_att,
  35       init_method=flags.init_method,
  36       init_range=flags.init_range,
  37       init_std=flags.init_std,
  38       clamp_len=flags.clamp_len)
  39 
  40   if not is_finetune:
  41     kwargs.update(dict(
  42         mem_len=flags.mem_len,
  43         reuse_len=flags.reuse_len,
  44         bi_data=flags.bi_data,
  45         clamp_len=flags.clamp_len,
  46         same_length=flags.same_length))
  47 
  48   return RunConfig(**kwargs)

  52 class XLNetConfig(object):
  53   """Configs for XLNet model.
  54 
  55   XLNetConfig contains hyperparameters that are specific to a model checkpoint;
  56   i.e., these hyperparameters should be the same between
  57   pretraining and finetuning.
  58 
  59   The following hyperparameters are defined:
  60     n_layer: int, the number of layers.
  61     d_model: int, the hidden size.
  62     n_head: int, the number of attention heads.
  63     d_head: int, the dimension size of each attention head.
  64     d_inner: int, the hidden size in feed-forward layers.
  65     ff_activation: str, "relu" or "gelu".
  66     untie_r: bool, whether to untie the biases in attention.
  67     n_token: int, the vocab size.
  68   """
  69 
  70   def __init__(self, FLAGS=None, json_path=None, args_dict=None):
  71     """Constructing an XLNetConfig.
  72 
  73     One of FLAGS or json_path should be provided.
  74 
  75     Args:
  76       FLAGS: An FLAGS instance.
  77       json_path: A path to a json config file.
  78       args_dict: A dict for args.
  79     """
  80 
  81     assert FLAGS is not None or json_path is not None or args_dict is not None
  82 
  83     self.keys = ['n_layer', 'd_model', 'n_head', 'd_head', 'd_inner',
  84                  'ff_activation', 'untie_r', 'n_token']
  85 
  86     if FLAGS is not None:
  87       self.init_from_flags(FLAGS)
  88 
  89     if json_path is not None:
  90       self.init_from_json(json_path)
  91 
  92     if args_dict is not None:
  93       self.init_from_dict(args_dict)
  94 
  95   def init_from_dict(self, args_dict):
  96     """Constructs a `BertConfig` from a Python dictionary of parameters."""
  97     for key in self.keys:
  98       setattr(self, key, args_dict[key])
  99 
 100   def init_from_flags(self, flags):
 101     for key in self.keys:
 102       setattr(self, key, getattr(flags, key))
 103 
 104   def init_from_json(self, json_path):
 105     with tf.io.gfile.GFile(json_path) as f:
 106       json_data = json.load(f)
 107       self.init_from_dict(json_data)
 108 
 109   def to_json(self, json_path):
 110     """Save XLNetConfig to a json file."""
 111     json_data = {}
 112     for key in self.keys:
 113       json_data[key] = getattr(self, key)
 114 
 115     json_dir = os.path.dirname(json_path)
 116     if not tf.io.gfile.exists(json_dir):
 117       tf.io.gfile.makedirs(json_dir)
 118     with tf.io.gfile.GFile(json_path, 'w') as f:
 119       json.dump(json_data, f, indent=4, sort_keys=True)

 122 class RunConfig(object):
 123   """Class of RunConfig.
 124 
 125   RunConfig contains hyperparameters that could be different
 126   between pretraining and finetuning.
 127   These hyperparameters can also be changed from run to run.
 128   We store them separately from XLNetConfig for flexibility.
 129   """
 130 
 131   def __init__(self,
 132                is_training,
 133                use_tpu,
 134                dropout,
 135                dropout_att,
 136                init_method='normal',
 137                init_range=0.1,
 138                init_std=0.02,
 139                mem_len=None,
 140                reuse_len=None,
 141                bi_data=False,
 142                clamp_len=-1,
 143                same_length=False,
 144                use_cls_mask=True):
 145     """Initializes RunConfig.
 146 
 147     Args:
 148       is_training: bool, whether in training mode.
 149       use_tpu: bool, whether TPUs are used.
 150       dropout: float, dropout rate.
 151       dropout_att: float, dropout rate on attention probabilities.
 152       init_method: str, the initialization scheme, either "normal" or "uniform".
 153       init_range: float, initialize the parameters with a uniform distribution
 154         in [-init_range, init_range]. Only effective when init="uniform".
 155       init_std: float, initialize the parameters with a normal distribution
 156         with mean 0 and stddev init_std. Only effective when init="normal".
 157       mem_len: int, the number of tokens to cache.
 158       reuse_len: int, the number of tokens in the currect batch to be cached
 159         and reused in the future.
 160       bi_data: bool, whether to use bidirectional input pipeline.
 161         Usually set to True during pretraining and False during finetuning.
 162       clamp_len: int, clamp all relative distances larger than clamp_len.
 163         -1 means no clamping.
 164       same_length: bool, whether to use the same attention length
 165                    for each token.
 166       use_cls_mask: bool, whether to introduce cls mask.
 167     """
 168 
 169     self.init_method = init_method
 170     self.init_range = init_range
 171     self.init_std = init_std
 172     self.is_training = is_training
 173     self.dropout = dropout
 174     self.dropout_att = dropout_att
 175     self.use_tpu = use_tpu
 176     self.mem_len = mem_len
 177     self.reuse_len = reuse_len
 178     self.bi_data = bi_data
 179     self.clamp_len = clamp_len
 180     self.same_length = same_length
 181     self.use_cls_mask = use_cls_mask
