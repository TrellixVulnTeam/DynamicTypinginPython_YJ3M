  28 def printable_text(text):
  29   """Returns text encoded in a way suitable for print or `tf.logging`."""
  30 
  31   # These functions want `str` for both Python2 and Python3, but in one case
  32   # it's a Unicode string and in the other it's a byte string.
  33   if six.PY3:
  34     if isinstance(text, str):
  35       return text
  36     elif isinstance(text, bytes):
  37       return text.decode('utf-8', 'ignore')
  38     else:
  39       raise ValueError('Unsupported string type: %s' % (type(text)))
  40   elif six.PY2:
  41     if isinstance(text, str):
  42       return text
  43     elif isinstance(text, unicode):
  44       return text.encode('utf-8')
  45     else:
  46       raise ValueError('Unsupported string type: %s' % (type(text)))
  47   else:
  48     raise ValueError('Not running on Python2 or Python 3?')

  51 def print_(*args):
  52   new_args = []
  53   for arg in args:
  54     if isinstance(arg, list):
  55       s = [printable_text(i) for i in arg]
  56       s = ' '.join(s)
  57       new_args.append(s)
  58     else:
  59       new_args.append(printable_text(arg))
  60   print(*new_args)

  63 def preprocess_text(inputs, lower=False, remove_space=True, keep_accents=False):
  64   """Preprocesses texts."""
  65   if remove_space:
  66     outputs = ' '.join(inputs.strip().split())
  67   else:
  68     outputs = inputs
  69 
  70   outputs = outputs.replace('``', '"').replace("''", '"')
  71 
  72   if six.PY2 and isinstance(outputs, str):
  73     outputs = outputs.decode('utf-8')
  74 
  75   if not keep_accents:
  76     outputs = unicodedata.normalize('NFKD', outputs)
  77     outputs = ''.join([c for c in outputs if not unicodedata.combining(c)])
  78   if lower:
  79     outputs = outputs.lower()
  80 
  81   return outputs

  84 def encode_pieces(sp_model, text, return_unicode=True, sample=False):
  85   """Encodes pieces."""
  86   # return_unicode is used only for py2
  87 
  88   if six.PY2 and isinstance(text, unicode):
  89     text = text.encode('utf-8')
  90 
  91   if not sample:
  92     pieces = sp_model.EncodeAsPieces(text)
  93   else:
  94     pieces = sp_model.SampleEncodeAsPieces(text, 64, 0.1)
  95   new_pieces = []
  96   for piece in pieces:
  97     if len(piece) > 1 and piece[-1] == ',' and piece[-2].isdigit():
  98       cur_pieces = sp_model.EncodeAsPieces(
  99           piece[:-1].replace(SPIECE_UNDERLINE, ''))
 100       if piece[0] != SPIECE_UNDERLINE and cur_pieces[0][0] == SPIECE_UNDERLINE:
 101         if len(cur_pieces[0]) == 1:
 102           cur_pieces = cur_pieces[1:]
 103         else:
 104           cur_pieces[0] = cur_pieces[0][1:]
 105       cur_pieces.append(piece[-1])
 106       new_pieces.extend(cur_pieces)
 107     else:
 108       new_pieces.append(piece)
 109 
 110   # note(zhiliny): convert back to unicode for py2
 111   if six.PY2 and return_unicode:
 112     ret_pieces = []
 113     for piece in new_pieces:
 114       if isinstance(piece, str):
 115         piece = piece.decode('utf-8')
 116       ret_pieces.append(piece)
 117     new_pieces = ret_pieces
 118 
 119   return new_pieces

 122 def encode_ids(sp_model, text, sample=False):
 123   pieces = encode_pieces(sp_model, text, return_unicode=False, sample=sample)
 124   ids = [sp_model.PieceToId(piece) for piece in pieces]
 125   return ids
