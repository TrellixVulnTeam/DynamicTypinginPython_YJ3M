  29 def gelu(x):
  30   """Gaussian Error Linear Unit.
  31 
  32   This is a smoother version of the RELU.
  33   Original paper: https://arxiv.org/abs/1606.08415
  34   Args:
  35     x: float Tensor to perform activation.
  36 
  37   Returns:
  38     `x` with the GELU activation applied.
  39   """
  40   cdf = 0.5 * (1.0 + tf.tanh(
  41       (np.sqrt(2 / np.pi) * (x + 0.044715 * tf.pow(x, 3)))))
  42   return x * cdf

  45 def rel_shift(x, klen=-1):
  46   """Performs relative shift to form the relative attention score."""
  47   x_size = tf.shape(x)
  48 
  49   x = tf.reshape(x, [x_size[1], x_size[0], x_size[2], x_size[3]])
  50   x = tf.slice(x, [1, 0, 0, 0], [-1, -1, -1, -1])
  51   x = tf.reshape(x, [x_size[0], x_size[1] - 1, x_size[2], x_size[3]])
  52   x = tf.slice(x, [0, 0, 0, 0], [-1, klen, -1, -1])
  53 
  54   return x

  57 def _get_initializer(flags):
  58   """Get variable intializer."""
  59   if flags.init_method == 'uniform':
  60     initializer = tf.keras.initializers.RandomUniform(
  61         minval=-flags.init_range, maxval=flags.init_range)
  62   elif flags.init_method == 'normal':
  63     initializer = tf.keras.initializers.RandomNormal(stddev=flags.init_std)
  64   else:
  65     raise ValueError('Initializer {} not supported'.format(flags.init_method))
  66   return initializer

  69 def _create_mask(qlen, mlen, dtype=tf.float32, same_length=False):
  70   """Creates attention mask when single-side context allowed only."""
  71   attn_mask = tf.ones([qlen, qlen], dtype=dtype)
  72   mask_u = tf.linalg.band_part(attn_mask, 0, -1)
  73   mask_dia = tf.linalg.band_part(attn_mask, 0, 0)
  74   attn_mask_pad = tf.zeros([qlen, mlen], dtype=dtype)
  75   ret = tf.concat([attn_mask_pad, mask_u - mask_dia], 1)
  76   if same_length:
  77     mask_l = tf.linalg.band_part(attn_mask, -1, 0)
  78     ret = tf.concat([ret[:, :qlen] + mask_l - mask_dia, ret[:, qlen:]], 1)
  79 
  80   return ret

  83 def _cache_mem(curr_out, prev_mem, mem_len, reuse_len=None):
  84   """cache hidden states into memory."""
  85 
  86   if mem_len is None or mem_len == 0:
  87     return None
  88   else:
  89     if reuse_len is not None and reuse_len > 0:
  90       curr_out = curr_out[:reuse_len]
  91 
  92     if prev_mem is None:
  93       new_mem = curr_out[-mem_len:]
  94     else:
  95       new_mem = tf.concat([prev_mem, curr_out], 0)[-mem_len:]
  96 
  97   return tf.keras.backend.stop_gradient(new_mem)

 105 class PositionalEmbedding(tf.keras.layers.Layer):
 106   """Generates relative positional embeddings used in Transformer-XL and XLNet."""
 107 
 108   def __init__(self, dim, **kwargs):
 109     super(PositionalEmbedding, self).__init__(**kwargs)
 110     self.dim = dim
 111 
 112   def build(self, unused_input_shapes):
 113     """Constructs inversed frequency vector for positional embedding layer."""
 114     self.inv_freq = 1.0 / (10000.0**(tf.range(0, self.dim, 2.0) / self.dim))
 115     super(PositionalEmbedding, self).build(unused_input_shapes)
 116 
 117   def call(self, pos_seq, batch_size):
 118     """Implements call() for the layer."""
 119     sinusoid_inp = tf.einsum('i,d->id', pos_seq, self.inv_freq)
 120     pos_emb = tf.concat([tf.sin(sinusoid_inp), tf.cos(sinusoid_inp)], -1)
 121     pos_emb = pos_emb[:, None, :]
 122 
 123     if batch_size is not None:
 124       pos_emb = tf.tile(pos_emb, [1, batch_size, 1])
 125 
 126     return pos_emb

 129 class RelativeAttention(tf.keras.layers.Layer):
 130   """Core calculations for relative attention."""
 131 
 132   def __init__(self, dropout_att, scale):
 133     super(RelativeAttention, self).__init__()
 134     self.scale = scale
 135     self.dropout_att = dropout_att
 136 
 137   def build(self, unused_input_shapes):
 138     """Implements build() for the layer."""
 139 
 140     self.attention_probs_dropout = tf.keras.layers.Dropout(
 141         rate=self.dropout_att)
 142 
 143     super(RelativeAttention, self).build(unused_input_shapes)
 144 
 145   def call(self, q_head, k_head_h, v_head_h, k_head_r, seg_embed, seg_mat,
 146            r_w_bias, r_r_bias, r_s_bias, attn_mask):
 147     """Implements call() for the layer."""
 148 
 149     # content based attention score
 150     ac = tf.einsum('ibnd,jbnd->ijbn', q_head + r_w_bias, k_head_h)
 151 
 152     # position based attention score
 153     bd = tf.einsum('ibnd,jbnd->ijbn', q_head + r_r_bias, k_head_r)
 154     bd = rel_shift(bd, klen=tf.shape(ac)[1])
 155 
 156     # segment-based attention score
 157     if seg_mat is None:
 158       ef = 0
 159     else:
 160       ef = tf.einsum('ibnd,snd->isbn', q_head + r_s_bias, seg_embed)
 161       tgt_shape = tf.shape(bd)
 162       ef = tf.where(
 163           tf.broadcast_to(tf.expand_dims(seg_mat, 3), tgt_shape),
 164           tf.broadcast_to(ef[:, 1:, :, :], tgt_shape),
 165           tf.broadcast_to(ef[:, :1, :, :], tgt_shape))
 166 
 167     # merges attention scores and performs masking
 168     attn_score = (ac + bd + ef) * self.scale
 169     if attn_mask is not None:
 170       attn_score = attn_score - 1e30 * attn_mask
 171 
 172     # attention probability
 173     attn_prob = tf.nn.softmax(attn_score, 1)
 174     attn_prob = self.attention_probs_dropout(attn_prob)
 175 
 176     # attention output
 177     attn_vec = tf.einsum('ijbn,jbnd->ibnd', attn_prob, v_head_h)
 178 
 179     return attn_vec

 182 class PositionwiseFF(tf.keras.layers.Layer):
 183   """Positionwise feed-forward layer."""
 184 
 185   def __init__(self, d_model, d_inner, dropout, kernel_initializer,
 186                activation_type, **kwargs):
 187     super(PositionwiseFF, self).__init__(**kwargs)
 188     self.d_model = d_model
 189     self.d_inner = d_inner
 190     self.dropout = dropout
 191     self.activation_type = activation_type
 192     self.kernel_initializer = kernel_initializer
 193 
 194   def build(self, unused_input_shapes):
 195     """Implements build() for the layer."""
 196     if self.activation_type == 'relu':
 197       activation = tf.nn.relu
 198     elif self.activation_type == 'gelu':
 199       activation = gelu
 200     else:
 201       raise (ValueError('Unsupported activation type {}'.format(
 202           self.activation_type)))
 203     self.inner_projection_layer = (
 204         tf.keras.layers.Dense(
 205             units=self.d_inner,
 206             activation=activation,
 207             kernel_initializer=self.kernel_initializer,
 208             name='layer_1'))
 209     self.output_projection_layer = (
 210         tf.keras.layers.Dense(
 211             units=self.d_model,
 212             kernel_initializer=self.kernel_initializer,
 213             name='layer_2'))
 214     self.output_dropout = tf.keras.layers.Dropout(
 215         rate=self.dropout, name='drop_2')
 216     self.output_layer_norm = (
 217         tf.keras.layers.LayerNormalization(
 218             name='LayerNorm', axis=-1, epsilon=1e-12))
 219     super(PositionwiseFF, self).build(unused_input_shapes)
 220 
 221   def call(self, inp):
 222     """Implements call() for the layer."""
 223 
 224     output = self.inner_projection_layer(inp)
 225     output = self.output_projection_layer(output)
 226     output = self.output_dropout(output)
 227     output = self.output_layer_norm(output + inp)
 228     return output

 254 class RelativeMultiheadAttention(tf.keras.layers.Layer):
 255   """Multi-head attention with relative embedding."""
 256 
 257   def __init__(self, d_model, n_head, d_head, dropout, dropout_att,
 258                kernel_initializer, **kwargs):
 259     super(RelativeMultiheadAttention, self).__init__(**kwargs)
 260     self.d_model = d_model
 261     self.n_head = n_head
 262     self.d_head = d_head
 263     self.dropout = dropout
 264     self.dropout_att = dropout_att
 265     self.initializer = kernel_initializer
 266 
 267   def build(self, unused_input_shapes):
 268     """Implements build() for the layer."""
 269     self.scale = 1.0 / (self.d_head**0.5)
 270 
 271     self.output_layer_norm = tf.keras.layers.LayerNormalization(
 272         name='LayerNorm', axis=-1, epsilon=1e-12)
 273 
 274     self.kh_projection_layer = self.add_weight(
 275         'k/kernel',
 276         shape=[self.d_model, self.n_head, self.d_head],
 277         initializer=self.initializer)
 278     self.vh_projection_layer = self.add_weight(
 279         'v/kernel',
 280         shape=[self.d_model, self.n_head, self.d_head],
 281         initializer=self.initializer)
 282     self.kr_projection_layer = self.add_weight(
 283         'r/kernel',
 284         shape=[self.d_model, self.n_head, self.d_head],
 285         initializer=self.initializer)
 286     self.qh_projection_layer = self.add_weight(
 287         'q/kernel',
 288         shape=[self.d_model, self.n_head, self.d_head],
 289         initializer=self.initializer)
 290 
 291     self.relative_attention_layer = RelativeAttention(
 292         dropout_att=self.dropout_att, scale=self.scale)
 293 
 294     self.proj_o = self.add_weight(
 295         'o/kernel',
 296         shape=[self.d_model, self.n_head, self.d_head],
 297         initializer=self.initializer)
 298 
 299     self.attention_dropout = tf.keras.layers.Dropout(rate=self.dropout)
 300 
 301     super(RelativeMultiheadAttention, self).build(unused_input_shapes)
 302 
 303   def call(self, h, g, r, r_w_bias, r_r_bias, seg_mat, r_s_bias, seg_embed,
 304            attn_mask_h, attn_mask_g, mems, target_mapping):
 305     """Implements call() for the layer."""
 306 
 307     if mems is not None and mems.shape.ndims > 1:
 308       cat = tf.concat([mems, h], 0)
 309     else:
 310       cat = h
 311 
 312     # content heads
 313     q_head_h = tf.einsum('ibh,hnd->ibnd', h, self.qh_projection_layer)
 314     k_head_h = tf.einsum('ibh,hnd->ibnd', cat, self.kh_projection_layer)
 315     v_head_h = tf.einsum('ibh,hnd->ibnd', cat, self.vh_projection_layer)
 316 
 317     # positional heads
 318     k_head_r = tf.einsum('ibh,hnd->ibnd', r, self.kr_projection_layer)
 319 
 320     # core attention ops
 321     attn_vec_h = self.relative_attention_layer(q_head_h, k_head_h, v_head_h,
 322                                                k_head_r, seg_embed, seg_mat,
 323                                                r_w_bias, r_r_bias, r_s_bias,
 324                                                attn_mask_h)
 325 
 326     # post processing
 327     output_h = tf.einsum('ibnd,hnd->ibh', attn_vec_h, self.proj_o)
 328     output_h = self.attention_dropout(output_h)
 329     output_h = self.output_layer_norm(output_h + h)
 330 
 331     output_g = None
 332     if g is not None:  # enable two-stream attention
 333       # g-stream
 334       q_head_g = tf.einsum('ibh,hnd->ibnd', g, self.qh_projection_layer)
 335       if target_mapping is not None:
 336         q_head_g = tf.einsum('mbnd,mlb->lbnd', q_head_g, target_mapping)
 337         attn_vec_g = self.relative_attention_layer(q_head_g, k_head_h, v_head_h,
 338                                                    k_head_r, seg_embed, seg_mat,
 339                                                    r_w_bias, r_r_bias, r_s_bias,
 340                                                    attn_mask_g)
 341         attn_vec_g = tf.einsum('lbnd,mlb->mbnd', attn_vec_g, target_mapping)
 342 
 343       else:
 344         attn_vec_g = self.relative_attention_layer(q_head_g, k_head_h, v_head_h,
 345                                                    k_head_r, seg_embed, seg_mat,
 346                                                    r_w_bias, r_r_bias, r_s_bias,
 347                                                    attn_mask_g)
 348 
 349       # post processing
 350       output_g = tf.einsum('ibnd,hnd->ibh', attn_vec_g, self.proj_o)
 351       output_g = self.attention_dropout(output_g)
 352       output_g = self.output_layer_norm(output_g + g)
 353 
 354     return (output_h, output_g)

 357 class TransformerXLModel(tf.keras.layers.Layer):
 358   """Defines a Transformer-XL computation graph with additional support for XLNet."""
 359 
 360   def __init__(self,
 361                n_token,
 362                n_layer,
 363                d_model,
 364                n_head,
 365                d_head,
 366                d_inner,
 367                dropout,
 368                dropout_att,
 369                attn_type,
 370                bi_data,
 371                is_training,
 372                initializer,
 373                mem_len=None,
 374                same_length=False,
 375                clamp_len=-1,
 376                untie_r=False,
 377                use_tpu=True,
 378                reuse_len=None,
 379                ff_activation='relu',
 380                use_cls_mask=False,
 381                **kwargs):
 382     """Initializes TransformerXLModel.
 383 
 384     Args:
 385       n_token: int, the number of tokens in vocabulary.
 386       n_layer: int, the number of layers.
 387       d_model: int, the hidden size.
 388       n_head: int, the number of attention heads.
 389       d_head: int, the dimension size of each attention head.
 390       d_inner: int, the hidden size in feed-forward layers.
 391       dropout: float, dropout rate.
 392       dropout_att: float, dropout rate on attention probabilities.
 393       attn_type: str, "uni" or "bi".
 394       bi_data: bool, whether to use bidirectional input pipeline. Usually set to
 395         True during pretraining and False during finetuning.
 396       is_training: bool, whether in training mode.
 397       initializer: A tf initializer.
 398       mem_len: int, the number of tokens to cache.
 399       same_length: bool, whether to use the same attention length for each
 400         token.
 401       clamp_len: int, clamp all relative distances larger than clamp_len. -1
 402         means no clamping.
 403       untie_r: bool, whether to untie the biases in attention.
 404       use_tpu: bool, whether TPUs are used.
 405       reuse_len: int, the number of tokens in the currect batch to be cached and
 406         reused in the future.
 407       ff_activation: str, "relu" or "gelu".
 408       use_cls_mask: bool, whether to introduce cls mask.
 409       **kwargs: Other parameters.
 410     """
 411 
 412     super(TransformerXLModel, self).__init__(**kwargs)
 413 
 414     self.n_token = n_token
 415     self.initializer = initializer
 416     self.attn_type = attn_type
 417     self.n_layer = n_layer
 418     self.d_model = d_model
 419     self.n_head = n_head
 420     self.d_head = d_head
 421     self.d_inner = d_inner
 422     self.ff_activation = ff_activation
 423     self.untie_r = untie_r
 424     self.use_tpu = use_tpu
 425     self.dropout = dropout
 426     self.dropout_att = dropout_att
 427 
 428     self.mem_len = mem_len
 429     self.reuse_len = reuse_len
 430     self.bi_data = bi_data
 431     self.clamp_len = clamp_len
 432     self.same_length = same_length
 433     self.use_cls_mask = use_cls_mask
 434 
 435   def build(self, unused_input_shapes):
 436     """Implements build() for the layer."""
 437     self.tf_float = tf.float32
 438 
 439     self.embedding_lookup = EmbeddingLookup(
 440         n_token=self.n_token,
 441         d_embed=self.d_model,
 442         initializer=self.initializer,
 443         dtype=self.tf_float,
 444         name='word_embedding')
 445 
 446     self.h_dropout = tf.keras.layers.Dropout(rate=self.dropout)
 447     self.g_dropout = tf.keras.layers.Dropout(rate=self.dropout)
 448 
 449     if self.untie_r:
 450       self.r_w_bias = (
 451           self.add_weight(
 452               'r_w_bias',
 453               shape=[self.n_layer, self.n_head, self.d_head],
 454               dtype=self.tf_float,
 455               initializer=self.initializer))
 456       self.r_r_bias = (
 457           self.add_weight(
 458               'r_r_bias',
 459               shape=[self.n_layer, self.n_head, self.d_head],
 460               dtype=self.tf_float,
 461               initializer=self.initializer))
 462       self.r_s_bias = (
 463           self.add_weight(
 464               'r_s_bias',
 465               shape=[self.n_layer, self.n_head, self.d_head],
 466               dtype=self.tf_float,
 467               initializer=self.initializer))
 468     else:
 469       self.r_w_bias = (
 470           self.add_weight(
 471               'r_w_bias',
 472               shape=[self.n_head, self.d_head],
 473               dtype=self.tf_float,
 474               initializer=self.initializer))
 475       self.r_r_bias = (
 476           self.add_weight(
 477               'r_r_bias',
 478               shape=[self.n_head, self.d_head],
 479               dtype=self.tf_float,
 480               initializer=self.initializer))
 481       self.r_s_bias = (
 482           self.add_weight(
 483               'r_s_bias', [self.n_head, self.d_head],
 484               dtype=self.tf_float,
 485               initializer=self.initializer))
 486 
 487     self.seg_embed = self.add_weight(
 488         'seg_embed', [self.n_layer, 2, self.n_head, self.d_head],
 489         dtype=self.tf_float,
 490         initializer=self.initializer)
 491 
 492     self.mask_emb = self.add_weight(
 493         'mask_emb/mask_emb', shape=[1, 1, self.d_model], dtype=self.tf_float)
 494 
 495     self.emb_dropout = tf.keras.layers.Dropout(rate=self.dropout)
 496     self.fwd_position_embedding = PositionalEmbedding(self.d_model)
 497     self.bwd_position_embedding = PositionalEmbedding(self.d_model)
 498 
 499     self.rel_multihead_layers = []
 500     self.h_positionwise_ffn_layers = []
 501     for i in range(self.n_layer):
 502       self.rel_multihead_layers.append(
 503           RelativeMultiheadAttention(
 504               d_model=self.d_model,
 505               dropout=self.dropout,
 506               n_head=self.n_head,
 507               d_head=self.d_head,
 508               dropout_att=self.dropout_att,
 509               kernel_initializer=self.initializer,
 510               name='layer_%d/rel_attn' % (i)))
 511       self.h_positionwise_ffn_layers.append(
 512           PositionwiseFF(
 513               d_model=self.d_model,
 514               d_inner=self.d_inner,
 515               dropout=self.dropout,
 516               kernel_initializer=self.initializer,
 517               activation_type=self.ff_activation,
 518               name='layer_%d/ff' % (i)))
 519 
 520     self.output_dropout = tf.keras.layers.Dropout(rate=self.dropout)
 521 
 522     super(TransformerXLModel, self).build(unused_input_shapes)
 523 
 524   def __call__(self,
 525                inp_k,
 526                seg_id=None,
 527                input_mask=None,
 528                mems=None,
 529                perm_mask=None,
 530                target_mapping=None,
 531                inp_q=None,
 532                **kwargs):
 533     # Uses dict to feed inputs into call() in order to keep mems as a python
 534     # list.
 535     inputs = {
 536         'inp_k': inp_k,
 537         'seg_id': seg_id,
 538         'input_mask': input_mask,
 539         'mems': mems,
 540         'perm_mask': perm_mask,
 541         'target_mapping': target_mapping,
 542         'inp_q': inp_q
 543     }
 544     return super(TransformerXLModel, self).__call__(inputs, **kwargs)
 545 
 546   def call(self, inputs):
 547     """Implements call() for the layer."""
 548     inp_k = inputs['inp_k']
 549     seg_id = inputs['seg_id']
 550     input_mask = inputs['input_mask']
 551     mems = inputs['mems']
 552     perm_mask = inputs['perm_mask']
 553     target_mapping = inputs['target_mapping']
 554     inp_q = inputs['inp_q']
 555 
 556     new_mems = []
 557 
 558     bsz = tf.shape(inp_k)[1]
 559 
 560     qlen = inp_k.shape.as_list()[0]
 561 
 562     mlen = mems[0].shape.as_list()[0] if mems is not None else 0
 563     klen = mlen + qlen
 564 
 565     ##### Attention mask
 566     # causal attention mask
 567     if self.attn_type == 'uni':
 568       attn_mask = _create_mask(qlen, mlen, self.tf_float, self.same_length)
 569       # pylint: enable=protected-access
 570       attn_mask = attn_mask[:, :, None, None]
 571     elif self.attn_type == 'bi':
 572       attn_mask = None
 573     else:
 574       raise ValueError('Unsupported attention type: {}'.format(self.attn_type))
 575 
 576     # data mask: input mask & perm mask
 577     if input_mask is not None and perm_mask is not None:
 578       data_mask = input_mask[None] + perm_mask
 579 
 580     elif input_mask is not None and perm_mask is None:
 581       data_mask = input_mask[None]
 582     elif input_mask is None and perm_mask is not None:
 583       data_mask = perm_mask
 584     else:
 585       data_mask = None
 586 
 587     if data_mask is not None:
 588       # all mems can be attended to
 589       mems_mask = tf.zeros([tf.shape(data_mask)[0], mlen, bsz],
 590                            dtype=self.tf_float)
 591       data_mask = tf.concat([mems_mask, data_mask], 1)
 592       if attn_mask is None:
 593         attn_mask = data_mask[:, :, :, None]
 594       else:
 595         attn_mask += data_mask[:, :, :, None]
 596 
 597     if attn_mask is not None:
 598       attn_mask = tf.cast(attn_mask > 0, dtype=self.tf_float)
 599 
 600     if attn_mask is not None:
 601       non_tgt_mask = -tf.eye(qlen, dtype=self.tf_float)
 602       non_tgt_mask = tf.concat(
 603           [tf.zeros([qlen, mlen], dtype=self.tf_float), non_tgt_mask], axis=-1)
 604       non_tgt_mask = tf.cast(
 605           (attn_mask + non_tgt_mask[:, :, None, None]) > 0, dtype=self.tf_float)
 606     else:
 607       non_tgt_mask = None
 608 
 609     word_emb_k = self.embedding_lookup(inp_k)
 610 
 611     if inp_q is not None:
 612       if target_mapping is not None:
 613         word_emb_q = tf.tile(self.mask_emb,
 614                              [tf.shape(target_mapping)[0], bsz, 1])
 615       else:
 616         inp_q_ext = inp_q[:, :, None]
 617         word_emb_q = inp_q_ext * self.mask_emb + (1 - inp_q_ext) * word_emb_k
 618 
 619     output_h = self.h_dropout(word_emb_k)
 620     output_g = None
 621     if inp_q is not None:
 622       output_g = self.g_dropout(word_emb_q)
 623 
 624     ##### Segment embedding
 625     if seg_id is not None:
 626 
 627       # Convert `seg_id` to one-hot `seg_mat`
 628 
 629       mem_pad = tf.zeros([mlen, bsz], dtype=tf.int32)
 630 
 631       cat_id = tf.concat([mem_pad, seg_id], 0)
 632 
 633       if self.use_cls_mask:
 634         # `1` indicates not in the same segment [qlen x klen x bsz]
 635         # seg_id: [qlen x bsz] & cat_id: [klen x bsz]
 636         cls_mat = tf.logical_or(
 637             tf.equal(seg_id, tf.constant([data_utils.SEG_ID_CLS]))[:, None],
 638             tf.equal(cat_id, tf.constant([data_utils.SEG_ID_CLS]))[None, :])
 639         seg_mat = tf.equal(seg_id[:, None], cat_id[None, :])
 640         seg_mat = tf.logical_or(cls_mat, seg_mat)
 641       else:
 642         seg_mat = tf.logical_not(tf.equal(seg_id[:, None], cat_id[None, :]))
 643     else:
 644       seg_mat = None
 645 
 646     dtype = self.tf_float
 647     freq_seq = tf.range(0, self.d_model, 2.0)
 648     if dtype is not None and dtype != tf.float32:
 649       freq_seq = tf.cast(freq_seq, dtype=self.dtype)
 650 
 651     if self.attn_type == 'bi':
 652       beg, end = klen, -qlen
 653     elif self.attn_type == 'uni':
 654       beg, end = klen, -1
 655     else:
 656       raise ValueError('Unknown `attn_type` {}.'.format(self.attn_type))
 657 
 658     if self.bi_data:
 659       fwd_pos_seq = tf.range(beg, end, -1.0)
 660       bwd_pos_seq = tf.range(-beg, -end, 1.0)
 661 
 662       if dtype is not None and dtype != tf.float32:
 663         fwd_pos_seq = tf.cast(fwd_pos_seq, dtype=dtype)
 664         bwd_pos_seq = tf.cast(bwd_pos_seq, dtype=dtype)
 665 
 666       if self.clamp_len > 0:
 667         fwd_pos_seq = tf.clip_by_value(fwd_pos_seq, -self.clamp_len,
 668                                        self.clamp_len)
 669         bwd_pos_seq = tf.clip_by_value(bwd_pos_seq, -self.clamp_len,
 670                                        self.clamp_len)
 671 
 672       if bsz is not None:
 673         fwd_pos_emb = self.fwd_position_embedding(fwd_pos_seq, bsz // 2)
 674         bwd_pos_emb = self.bwd_position_embedding(bwd_pos_seq, bsz // 2)
 675       else:
 676         fwd_pos_emb = self.fwd_position_embedding(fwd_pos_seq, None)
 677         bwd_pos_emb = self.bwd_position_embedding(bwd_pos_seq, None)
 678 
 679       pos_emb = tf.concat([fwd_pos_emb, bwd_pos_emb], axis=1)
 680     else:
 681       fwd_pos_seq = tf.range(beg, end, -1.0)
 682       if dtype is not None and dtype != tf.float32:
 683         fwd_pos_seq = tf.cast(fwd_pos_seq, dtype=dtype)
 684       if self.clamp_len > 0:
 685         fwd_pos_seq = tf.clip_by_value(fwd_pos_seq, -self.clamp_len,
 686                                        self.lamp_len)
 687 
 688       pos_emb = self.fwd_position_embedding(fwd_pos_seq, bsz)
 689 
 690     pos_emb = self.emb_dropout(pos_emb)
 691 
 692     if mems is None:
 693       mems = [None] * self.n_layer
 694     for i in range(self.n_layer):
 695       # cache new mems
 696       new_mems.append(
 697           _cache_mem(output_h, mems[i], self.mem_len, self.reuse_len))
 698       # pylint: enable=protected-access
 699 
 700       # segment bias
 701       if seg_id is None:
 702         r_s_bias_i = None
 703         seg_embed_i = None
 704       else:
 705         r_s_bias_i = self.r_s_bias if not self.untie_r else self.r_s_bias[i]
 706         seg_embed_i = self.seg_embed[i]
 707 
 708       ffn_layer = self.h_positionwise_ffn_layers[i]
 709       attention_layer = self.rel_multihead_layers[i]
 710       output_h, output_g = attention_layer(
 711           h=output_h,
 712           g=output_g,
 713           r=pos_emb,
 714           r_w_bias=self.r_w_bias if not self.untie_r else self.r_w_bias[i],
 715           r_r_bias=self.r_r_bias if not self.untie_r else self.r_r_bias[i],
 716           seg_mat=seg_mat,
 717           r_s_bias=r_s_bias_i,
 718           seg_embed=seg_embed_i,
 719           attn_mask_h=non_tgt_mask,
 720           attn_mask_g=attn_mask,
 721           mems=mems[i],
 722           target_mapping=target_mapping)
 723       output_h = ffn_layer(output_h)
 724       if output_g is not None:
 725         output_g = ffn_layer(output_g)
 726 
 727     if inp_q is not None:
 728       output = output_g
 729     else:
 730       output = output_h
 731 
 732     return output, new_mems, None

 887 class LMLossLayer(tf.keras.layers.Layer):
 888   """Layer computing cross entropy loss for language modeling."""
 889 
 890   def __init__(self,
 891                n_token,
 892                d_model,
 893                initializer,
 894                tie_weight=False,
 895                bi_data=True,
 896                use_tpu=False,
 897                use_proj=False,
 898                **kwargs):
 899     """Constructs LMLoss layer.
 900 
 901     Args:
 902       n_token: Number of tokens in vocabulary.
 903       d_model: The dimension of model hidden state.
 904       initializer: Initializer used for parameters.
 905       tie_weight: Whether to share weights between embedding lookup layer and
 906         next-token prediction layer.
 907       bi_data: Whether to use bidirectional input pipeline. Usually set to True
 908         during pretraining and False during finetuning.
 909       use_tpu: bool, whether to use TPU.
 910       use_proj: bool, whether to add a projection layer before LM prediction.
 911       **kwargs: Other parameters.
 912     """
 913     super(LMLossLayer, self).__init__(**kwargs)
 914     self.n_token = n_token
 915     self.d_model = d_model
 916     self.initializer = initializer
 917 
 918     self.tie_weight = tie_weight
 919     self.bi_data = bi_data
 920     self.use_tpu = use_tpu
 921     self.use_proj = use_proj
 922 
 923   def build(self, unused_input_shapes):
 924     """Implements build() for the layer."""
 925     if self.use_proj:
 926       self.proj_layer = tf.keras.layers.Dense(
 927           units=self.d_model,
 928           kernel_initializer=self.initializer,
 929           activation=gelu,
 930           name='lm_projection/dense')
 931       self.proj_layer_norm = tf.keras.layers.LayerNormalization(
 932           axis=-1, epsilon=1e-12, name='lm_projection/LayerNorm')
 933     if not self.tie_weight:
 934       self.softmax_w = self.add_weight(
 935           'weight',
 936           shape=[self.n_token, self.d_model],
 937           initializer=self.initializer)
 938 
 939     self.softmax_b = self.add_weight(
 940         'bias', shape=[self.n_token], initializer=tf.zeros_initializer())
 941 
 942     super(LMLossLayer, self).build(unused_input_shapes)
 943 
 944   def call(self, hidden, target, lookup_table, target_mask):
 945     """Implements call() for the layer."""
 946     if self.use_proj:
 947       hidden = self.proj_layer_norm(self.proj_layer(hidden))
 948     if self.tie_weight:
 949       logits = tf.einsum('ibd,nd->ibn', hidden, lookup_table) + self.softmax_b
 950     else:
 951       logits = tf.einsum('ibd,nd->ibn', hidden, self.softmax_w) + self.softmax_b
 952 
 953     if self.use_tpu:
 954       one_hot_target = tf.one_hot(target, self.n_token, dtype=logits.dtype)
 955       loss = -tf.reduce_sum(tf.nn.log_softmax(logits) * one_hot_target, -1)
 956     else:
 957       loss = tf.nn.sparse_softmax_cross_entropy_with_logits(
 958           labels=target, logits=logits)
 959 
 960     total_loss = tf.reduce_sum(loss * target_mask) / tf.reduce_sum(target_mask)
 961 
 962     return total_loss, logits

 965 class Summarization(tf.keras.layers.Layer):
 966   """The layer to pool the output from XLNet model into a vector."""
 967 
 968   def __init__(self,
 969                d_model,
 970                n_head,
 971                d_head,
 972                dropout,
 973                dropout_att,
 974                initializer,
 975                use_proj=True,
 976                summary_type='last',
 977                **kwargs):
 978     """Constructs Summarization layer.
 979 
 980     Args:
 981       d_model: int, the dimension of model hidden state.
 982       n_head: int, the number of attention heads.
 983       d_head: int, the dimension size of each attention head.
 984       dropout: float, dropout rate.
 985       dropout_att: float, dropout rate on attention probabilities.
 986       initializer: Initializer used for parameters.
 987       use_proj: bool, whether to use projection layer for summarization.
 988       summary_type: Method used to summarize a sequence into a compact vector.
 989       **kwargs: Other parameters.
 990     """
 991     super(Summarization, self).__init__(**kwargs)
 992     self.d_model = d_model
 993     self.n_head = n_head
 994     self.d_head = d_head
 995     self.initializer = initializer
 996 
 997     self.dropout = dropout
 998     self.dropout_att = dropout_att
 999     self.use_proj = use_proj
1000     self.summary_type = summary_type
1001 
1002   def build(self, unused_input_shapes):
1003     """Implements build() for the layer."""
1004     if self.use_proj:
1005       self.proj_layer = tf.keras.layers.Dense(
1006           units=self.d_model,
1007           kernel_initializer=self.initializer,
1008           activation=tf.nn.tanh,
1009           name='summary')
1010     self.dropout_layer = tf.keras.layers.Dropout(rate=self.dropout)
1011 
1012     super(Summarization, self).build(unused_input_shapes)
1013 
1014   def call(self, inputs):
1015     """Implements call() for the layer."""
1016     if self.summary_type == 'last':
1017       summary = inputs[-1]
1018     elif self.summary_type == 'first':
1019       summary = inputs[0]
1020     else:
1021       raise ValueError('Invalid summary type provided: %s' % self.summary_type)
1022     if self.use_proj:
1023       summary = self.proj_layer(summary)
1024     summary = self.dropout_layer(summary)
1025     return summary

1028 class ClassificationLossLayer(tf.keras.layers.Layer):
1029   """Layer computing cross entropy loss for classification task."""
1030 
1031   def __init__(self, n_class, initializer, **kwargs):
1032     """Constructs Summarization layer.
1033 
1034     Args:
1035       n_class: Number of tokens in vocabulary.
1036       initializer: Initializer used for parameters.
1037       **kwargs: Other parameters.
1038     """
1039     super(ClassificationLossLayer, self).__init__(**kwargs)
1040 
1041     self.n_class = n_class
1042     self.initializer = initializer
1043 
1044   def build(self, unused_input_shapes):
1045     """Implements build() for the layer."""
1046     self.proj_layer = tf.keras.layers.Dense(
1047         units=self.n_class, kernel_initializer=self.initializer, name='logit')
1048 
1049     super(ClassificationLossLayer, self).build(unused_input_shapes)
1050 
1051   def call(self, hidden, labels):
1052     """Implements call() for the layer."""
1053 
1054     logits = self.proj_layer(hidden)
1055     one_hot_target = tf.one_hot(labels, self.n_class, dtype=hidden.dtype)  # pytype: disable=attribute-error
1056     loss = -tf.reduce_sum(tf.nn.log_softmax(logits) * one_hot_target, -1)
1057 
1058     return loss, logits

1126       return new_mems, logits

1133 class QALossLayer(tf.keras.layers.Layer):
1134   """Layer computing position and regression loss for question answering task."""
1135 
1136   def __init__(self, d_model, start_n_top, end_n_top, initializer, dropout,
1137                **kwargs):
1138     """Constructs Summarization layer.
1139 
1140     Args:
1141       d_model: Int, the hidden size.
1142       start_n_top: Beam size for span start.
1143       end_n_top: Beam size for span end.
1144       initializer: Initializer used for parameters.
1145       dropout: float, dropout rate.
1146       **kwargs: Other parameters.
1147     """
1148     super(QALossLayer, self).__init__(**kwargs)
1149     self.d_model = d_model
1150     self.start_n_top = start_n_top
1151     self.end_n_top = end_n_top
1152     self.initializer = initializer
1153     self.dropout = dropout
1154 
1155   def build(self, unused_input_shapes):
1156     """Implements build() for the layer."""
1157     self.start_logits_proj_layer = tf.keras.layers.Dense(
1158         units=1, kernel_initializer=self.initializer, name='start_logits/dense')
1159     self.end_logits_proj_layer0 = tf.keras.layers.Dense(
1160         units=self.d_model,
1161         kernel_initializer=self.initializer,
1162         activation=tf.nn.tanh,
1163         name='end_logits/dense_0')
1164     self.end_logits_proj_layer1 = tf.keras.layers.Dense(
1165         units=1, kernel_initializer=self.initializer, name='end_logits/dense_1')
1166     self.end_logits_layer_norm = tf.keras.layers.LayerNormalization(
1167         axis=-1, epsilon=1e-12, name='end_logits/LayerNorm')
1168     self.answer_class_proj_layer0 = tf.keras.layers.Dense(
1169         units=self.d_model,
1170         kernel_initializer=self.initializer,
1171         activation=tf.nn.tanh,
1172         name='answer_class/dense_0')
1173     self.answer_class_proj_layer1 = tf.keras.layers.Dense(
1174         units=1,
1175         kernel_initializer=self.initializer,
1176         use_bias=False,
1177         name='answer_class/dense_1')
1178     self.ans_feature_dropout = tf.keras.layers.Dropout(rate=self.dropout)
1179     super(QALossLayer, self).build(unused_input_shapes)
1180 
1181   def __call__(self, hidden, p_mask, cls_index, **kwargs):
1182     return super(QALossLayer, self).__call__(
1183         (hidden, p_mask, cls_index, kwargs))
1184 
1185   def call(self, inputs, training=False):
1186     """Implements call() for the layer."""
1187     hidden, p_mask, cls_index, kwargs = inputs
1188     return_dict = {}
1189     seq_len = tf.shape(hidden)[0]
1190 
1191     start_logits = self.start_logits_proj_layer(hidden)
1192     start_logits = tf.transpose(tf.squeeze(start_logits, -1), [1, 0])
1193     start_logits_masked = start_logits * (1 - p_mask) - 1e30 * p_mask
1194     start_log_probs = tf.nn.log_softmax(start_logits_masked, -1)
1195     if training:
1196       start_positions = kwargs['start_positions']
1197       end_positions = kwargs['end_positions']
1198       is_impossible = kwargs['is_impossible']
1199       start_positions = tf.reshape(start_positions, [-1])
1200       start_index = tf.one_hot(
1201           start_positions, depth=seq_len, axis=-1, dtype=tf.float32)
1202       start_features = tf.einsum('lbh,bl->bh', hidden, start_index)
1203       start_features = tf.tile(start_features[None], [seq_len, 1, 1])
1204       end_logits = self.end_logits_proj_layer0(
1205           tf.concat([hidden, start_features], axis=-1))
1206 
1207       end_logits = self.end_logits_layer_norm(end_logits)
1208 
1209       end_logits = self.end_logits_proj_layer1(end_logits)
1210       end_logits = tf.transpose(tf.squeeze(end_logits, -1), [1, 0])
1211       end_logits_masked = end_logits * (1 - p_mask) - 1e30 * p_mask
1212       end_log_probs = tf.nn.log_softmax(end_logits_masked, -1)
1213     else:
1214       # during inference, compute the end logits based on beam search
1215 
1216       start_top_log_probs, start_top_index = tf.nn.top_k(
1217           start_log_probs, k=self.start_n_top)
1218       start_index = tf.one_hot(
1219           start_top_index, depth=seq_len, axis=-1, dtype=tf.float32)
1220       start_features = tf.einsum('lbh,bkl->bkh', hidden, start_index)
1221       end_input = tf.tile(hidden[:, :, None], [1, 1, self.start_n_top, 1])
1222       start_features = tf.tile(start_features[None], [seq_len, 1, 1, 1])
1223       end_input = tf.concat([end_input, start_features], axis=-1)
1224       end_logits = self.end_logits_proj_layer0(end_input)
1225       end_logits = tf.reshape(end_logits, [seq_len, -1, self.d_model])
1226       end_logits = self.end_logits_layer_norm(end_logits)
1227 
1228       end_logits = tf.reshape(end_logits,
1229                               [seq_len, -1, self.start_n_top, self.d_model])
1230 
1231       end_logits = self.end_logits_proj_layer1(end_logits)
1232       end_logits = tf.reshape(end_logits, [seq_len, -1, self.start_n_top])
1233       end_logits = tf.transpose(end_logits, [1, 2, 0])
1234       end_logits_masked = end_logits * (
1235           1 - p_mask[:, None]) - 1e30 * p_mask[:, None]
1236       end_log_probs = tf.nn.log_softmax(end_logits_masked, -1)
1237       end_top_log_probs, end_top_index = tf.nn.top_k(
1238           end_log_probs, k=self.end_n_top)
1239       end_top_log_probs = tf.reshape(end_top_log_probs,
1240                                      [-1, self.start_n_top * self.end_n_top])
1241       end_top_index = tf.reshape(end_top_index,
1242                                  [-1, self.start_n_top * self.end_n_top])
1243 
1244     if training:
1245       return_dict['start_log_probs'] = start_log_probs
1246       return_dict['end_log_probs'] = end_log_probs
1247     else:
1248       return_dict['start_top_log_probs'] = start_top_log_probs
1249       return_dict['start_top_index'] = start_top_index
1250       return_dict['end_top_log_probs'] = end_top_log_probs
1251       return_dict['end_top_index'] = end_top_index
1252     # an additional layer to predict answerability
1253 
1254     # get the representation of CLS
1255     cls_index = tf.one_hot(cls_index, seq_len, axis=-1, dtype=tf.float32)
1256     cls_feature = tf.einsum('lbh,bl->bh', hidden, cls_index)
1257 
1258     # get the representation of START
1259     start_p = tf.nn.softmax(start_logits_masked, axis=-1, name='softmax_start')
1260     start_feature = tf.einsum('lbh,bl->bh', hidden, start_p)
1261 
1262     ans_feature = tf.concat([start_feature, cls_feature], -1)
1263     ans_feature = self.answer_class_proj_layer0(ans_feature)
1264     ans_feature = self.ans_feature_dropout(ans_feature)
1265     cls_logits = self.answer_class_proj_layer1(ans_feature)
1266     cls_logits = tf.squeeze(cls_logits, -1)
1267     return_dict['cls_logits'] = cls_logits
1268 
1269     if not training:
1270       return return_dict
1271 
1272     def compute_loss(log_probs, positions):
1273       one_hot_positions = tf.one_hot(positions, depth=seq_len, dtype=tf.float32)
1274 
1275       loss = -tf.reduce_sum(one_hot_positions * log_probs, axis=-1)
1276       loss = tf.reduce_mean(loss)
1277       return loss
1278 
1279     start_loss = compute_loss(start_log_probs, start_positions)
1280     end_loss = compute_loss(end_log_probs, end_positions)
1281 
1282     total_loss = (start_loss + end_loss) * 0.5
1283 
1284     is_impossible = tf.reshape(is_impossible, [-1])
1285     regression_loss = tf.nn.sigmoid_cross_entropy_with_logits(
1286         labels=is_impossible, logits=cls_logits)
1287     regression_loss = tf.reduce_mean(regression_loss)
1288 
1289     total_loss += regression_loss * 0.5
1290     return total_loss, cls_logits
