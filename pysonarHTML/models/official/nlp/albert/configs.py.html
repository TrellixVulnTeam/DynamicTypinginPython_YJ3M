  26 class AlbertConfig(configs.BertConfig):
  27   """Configuration for `ALBERT`."""
  28 
  29   def __init__(self,
  30                num_hidden_groups=1,
  31                inner_group_num=1,
  32                **kwargs):
  33     """Constructs AlbertConfig.
  34 
  35     Args:
  36       num_hidden_groups: Number of group for the hidden layers, parameters in
  37         the same group are shared. Note that this value and also the following
  38         'inner_group_num' has to be 1 for now, because all released ALBERT
  39         models set them to 1. We may support arbitary valid values in future.
  40       inner_group_num: Number of inner repetition of attention and ffn.
  41       **kwargs: The remaining arguments are the same as above 'BertConfig'.
  42     """
  43     super(AlbertConfig, self).__init__(**kwargs)
  44 
  45     # TODO(chendouble): 'inner_group_num' and 'num_hidden_groups' are always 1
  46     # in the released ALBERT. Support other values in AlbertTransformerEncoder
  47     # if needed.
  48     if inner_group_num != 1 or num_hidden_groups != 1:
  49       raise ValueError("We only support 'inner_group_num' and "
  50                        "'num_hidden_groups' as 1.")
  51 
  52   @classmethod
  53   def from_dict(cls, json_object):
  54     """Constructs a `AlbertConfig` from a Python dictionary of parameters."""
  55     config = AlbertConfig(vocab_size=None)
  56     for (key, value) in six.iteritems(json_object):
  57       config.__dict__[key] = value
  58     return config
