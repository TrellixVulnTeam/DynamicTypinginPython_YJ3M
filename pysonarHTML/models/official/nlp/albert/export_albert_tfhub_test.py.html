  15 """Tests official.nlp.albert.export_albert_tfhub."""
  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function
  19 
  20 import os
  21 
  22 import numpy as np
  23 
  24 import tensorflow as tf
  25 import tensorflow_hub as hub
  26 
  27 from official.nlp.albert import configs
  28 from official.nlp.albert import export_albert_tfhub
  29 
  30 
  31 class ExportAlbertTfhubTest(tf.test.TestCase):
  32 
  33   def test_export_albert_tfhub(self):
  34     # Exports a savedmodel for TF-Hub
  35     albert_config = configs.AlbertConfig(
  36         vocab_size=100,
  37         embedding_size=8,
  38         hidden_size=16,
  39         intermediate_size=32,
  40         max_position_embeddings=128,
  41         num_attention_heads=2,
  42         num_hidden_layers=1)
  43     bert_model, encoder = export_albert_tfhub.create_albert_model(albert_config)
  44     model_checkpoint_dir = os.path.join(self.get_temp_dir(), "checkpoint")
  45     checkpoint = tf.train.Checkpoint(model=encoder)
  46     checkpoint.save(os.path.join(model_checkpoint_dir, "test"))
  47     model_checkpoint_path = tf.train.latest_checkpoint(model_checkpoint_dir)
  48 
  49     sp_model_file = os.path.join(self.get_temp_dir(), "sp_tokenizer.model")
  50     with tf.io.gfile.GFile(sp_model_file, "w") as f:
  51       f.write("dummy content")
  52 
  53     hub_destination = os.path.join(self.get_temp_dir(), "hub")
  54     export_albert_tfhub.export_albert_tfhub(
  55         albert_config,
  56         model_checkpoint_path,
  57         hub_destination,
  58         sp_model_file=sp_model_file)
  59 
  60     # Restores a hub KerasLayer.
  61     hub_layer = hub.KerasLayer(hub_destination, trainable=True)
  62 
  63     if hasattr(hub_layer, "resolved_object"):
  64       with tf.io.gfile.GFile(
  65           hub_layer.resolved_object.sp_model_file.asset_path.numpy()) as f:
  66         self.assertEqual("dummy content", f.read())
  67     # Checks the hub KerasLayer.
  68     for source_weight, hub_weight in zip(bert_model.trainable_weights,
  69                                          hub_layer.trainable_weights):
  70       self.assertAllClose(source_weight.numpy(), hub_weight.numpy())
  71 
  72     dummy_ids = np.zeros((2, 10), dtype=np.int32)
  73     hub_outputs = hub_layer([dummy_ids, dummy_ids, dummy_ids])
  74     source_outputs = bert_model([dummy_ids, dummy_ids, dummy_ids])
  75 
  76     # The outputs of hub module are "pooled_output" and "sequence_output",
  77     # while the outputs of encoder is in reversed order, i.e.,
  78     # "sequence_output" and "pooled_output".
  79     encoder_outputs = reversed(encoder([dummy_ids, dummy_ids, dummy_ids]))
  80     self.assertEqual(hub_outputs[0].shape, (2, 16))
  81     self.assertEqual(hub_outputs[1].shape, (2, 10, 16))
  82     for source_output, hub_output, encoder_output in zip(
  83         source_outputs, hub_outputs, encoder_outputs):
  84       self.assertAllClose(source_output.numpy(), hub_output.numpy())
  85       self.assertAllClose(source_output.numpy(), encoder_output.numpy())
  86 
  87 
  88 if __name__ == "__main__":
  89   tf.test.main()
