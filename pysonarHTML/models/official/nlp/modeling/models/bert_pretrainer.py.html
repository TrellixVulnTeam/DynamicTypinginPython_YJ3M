  32 @tf.keras.utils.register_keras_serializable(package='Text')
  33 class BertPretrainer(tf.keras.Model):
  34   """BERT network training model.
  35 
  36   This is an implementation of the network structure surrounding a transformer
  37   encoder as described in "BERT: Pre-training of Deep Bidirectional Transformers
  38   for Language Understanding" (https://arxiv.org/abs/1810.04805).
  39 
  40   The BertPretrainer allows a user to pass in a transformer stack, and
  41   instantiates the masked language model and classification networks that are
  42   used to create the training objectives.
  43 
  44   Arguments:
  45     network: A transformer network. This network should output a sequence output
  46       and a classification output.
  47     num_classes: Number of classes to predict from the classification network.
  48     num_token_predictions: Number of tokens to predict from the masked LM.
  49     embedding_table: Embedding table of a network. If None, the
  50       "network.get_embedding_table()" is used.
  51     activation: The activation (if any) to use in the masked LM network. If
  52       None, no activation will be used.
  53     initializer: The initializer (if any) to use in the masked LM and
  54       classification networks. Defaults to a Glorot uniform initializer.
  55     output: The output style for this network. Can be either 'logits' or
  56       'predictions'.
  57   """
  58 
  59   def __init__(self,
  60                network,
  61                num_classes,
  62                num_token_predictions,
  63                embedding_table=None,
  64                activation=None,
  65                initializer='glorot_uniform',
  66                output='logits',
  67                **kwargs):
  68     self._self_setattr_tracking = False
  69     self._config = {
  70         'network': network,
  71         'num_classes': num_classes,
  72         'num_token_predictions': num_token_predictions,
  73         'activation': activation,
  74         'initializer': initializer,
  75         'output': output,
  76     }
  77     self.encoder = network
  78     # We want to use the inputs of the passed network as the inputs to this
  79     # Model. To do this, we need to keep a copy of the network inputs for use
  80     # when we construct the Model object at the end of init. (We keep a copy
  81     # because we'll be adding another tensor to the copy later.)
  82     network_inputs = self.encoder.inputs
  83     inputs = copy.copy(network_inputs)
  84 
  85     # Because we have a copy of inputs to create this Model object, we can
  86     # invoke the Network object with its own input tensors to start the Model.
  87     # Note that, because of how deferred construction happens, we can't use
  88     # the copy of the list here - by the time the network is invoked, the list
  89     # object contains the additional input added below.
  90     sequence_output, cls_output = self.encoder(network_inputs)
  91 
  92     # The encoder network may get outputs from all layers.
  93     if isinstance(sequence_output, list):
  94       sequence_output = sequence_output[-1]
  95     if isinstance(cls_output, list):
  96       cls_output = cls_output[-1]
  97     sequence_output_length = sequence_output.shape.as_list()[1]
  98     if sequence_output_length < num_token_predictions:
  99       raise ValueError(
 100           "The passed network's output length is %s, which is less than the "
 101           'requested num_token_predictions %s.' %
 102           (sequence_output_length, num_token_predictions))
 103 
 104     masked_lm_positions = tf.keras.layers.Input(
 105         shape=(num_token_predictions,),
 106         name='masked_lm_positions',
 107         dtype=tf.int32)
 108     inputs.append(masked_lm_positions)
 109 
 110     if embedding_table is None:
 111       embedding_table = self.encoder.get_embedding_table()
 112     self.masked_lm = layers.MaskedLM(
 113         embedding_table=embedding_table,
 114         activation=activation,
 115         initializer=initializer,
 116         output=output,
 117         name='cls/predictions')
 118     lm_outputs = self.masked_lm(
 119         sequence_output, masked_positions=masked_lm_positions)
 120 
 121     self.classification = networks.Classification(
 122         input_width=cls_output.shape[-1],
 123         num_classes=num_classes,
 124         initializer=initializer,
 125         output=output,
 126         name='classification')
 127     sentence_outputs = self.classification(cls_output)
 128 
 129     super(BertPretrainer, self).__init__(
 130         inputs=inputs,
 131         outputs=dict(masked_lm=lm_outputs, classification=sentence_outputs),
 132         **kwargs)
 133 
 134   def get_config(self):
 135     return self._config
 136 
 137   @classmethod
 138   def from_config(cls, config, custom_objects=None):
 139     return cls(**config)

 145 class BertPretrainerV2(tf.keras.Model):
 146   """BERT pretraining model V2.
 147 
 148   (Experimental).
 149   Adds the masked language model head and optional classification heads upon the
 150   transformer encoder. When num_masked_tokens == 0, there won't be MaskedLM
 151   head.
 152 
 153   Arguments:
 154     num_masked_tokens: Number of tokens to predict from the masked LM.
 155     encoder_network: A transformer network. This network should output a
 156       sequence output and a classification output.
 157     mlm_activation: The activation (if any) to use in the masked LM network. If
 158       None, no activation will be used.
 159     mlm_initializer: The initializer (if any) to use in the masked LM. Default
 160       to a Glorot uniform initializer.
 161     classification_heads: A list of optional head layers to transform on encoder
 162       sequence outputs.
 163     name: The name of the model.
 164   Inputs: Inputs defined by the encoder network, plus `masked_lm_positions` as a
 165     dictionary.
 166   Outputs: A dictionary of `lm_output` and classification head outputs keyed by
 167     head names.
 168   """
 169 
 170   def __init__(
 171       self,
 172       num_masked_tokens: int,
 173       encoder_network: tf.keras.Model,
 174       mlm_activation=None,
 175       mlm_initializer='glorot_uniform',
 176       classification_heads: Optional[List[tf.keras.layers.Layer]] = None,
 177       name: str = 'bert',
 178       **kwargs):
 179     self._self_setattr_tracking = False
 180     self._config = {
 181         'encoder_network': encoder_network,
 182         'num_masked_tokens': num_masked_tokens,
 183         'mlm_initializer': mlm_initializer,
 184         'classification_heads': classification_heads,
 185         'name': name,
 186     }
 187 
 188     self.encoder_network = encoder_network
 189     inputs = copy.copy(self.encoder_network.inputs)
 190     sequence_output, _ = self.encoder_network(inputs)
 191 
 192     self.classification_heads = classification_heads or []
 193     if len(set([cls.name for cls in self.classification_heads])) != len(
 194         self.classification_heads):
 195       raise ValueError('Classification heads should have unique names.')
 196 
 197     outputs = dict()
 198     if num_masked_tokens > 0:
 199       self.masked_lm = layers.MaskedLM(
 200           embedding_table=self.encoder_network.get_embedding_table(),
 201           activation=mlm_activation,
 202           initializer=mlm_initializer,
 203           name='cls/predictions')
 204       masked_lm_positions = tf.keras.layers.Input(
 205           shape=(num_masked_tokens,),
 206           name='masked_lm_positions',
 207           dtype=tf.int32)
 208       inputs.append(masked_lm_positions)
 209       outputs['lm_output'] = self.masked_lm(
 210           sequence_output, masked_positions=masked_lm_positions)
 211     for cls_head in self.classification_heads:
 212       outputs[cls_head.name] = cls_head(sequence_output)
 213 
 214     super(BertPretrainerV2, self).__init__(
 215         inputs=inputs, outputs=outputs, name=name, **kwargs)
 216 
 217   @property
 218   def checkpoint_items(self):
 219     """Returns a dictionary of items to be additionally checkpointed."""
 220     items = dict(encoder=self.encoder_network)
 221     for head in self.classification_heads:
 222       for key, item in head.checkpoint_items.items():
 223         items['.'.join([head.name, key])] = item
 224     return items
 225 
 226   def get_config(self):
 227     return self._config
 228 
 229   @classmethod
 230   def from_config(cls, config, custom_objects=None):
 231     return cls(**config)
