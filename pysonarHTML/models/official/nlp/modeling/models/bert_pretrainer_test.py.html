  28 # This decorator runs the test in V1, V2-Eager, and V2-Functional mode. It

  33   def test_bert_pretrainer(self):
  34     """Validate that the Keras object can be created."""
  35     # Build a transformer network to use within the BERT trainer.
  36     vocab_size = 100
  37     sequence_length = 512
  38     test_network = networks.TransformerEncoder(
  39         vocab_size=vocab_size, num_layers=2, sequence_length=sequence_length)
  40 
  41     # Create a BERT trainer with the created network.
  42     num_classes = 3
  43     num_token_predictions = 2
  44     bert_trainer_model = bert_pretrainer.BertPretrainer(
  45         test_network,
  46         num_classes=num_classes,
  47         num_token_predictions=num_token_predictions)
  48 
  49     # Create a set of 2-dimensional inputs (the first dimension is implicit).
  50     word_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  51     mask = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  52     type_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  53     masked_lm_positions = tf.keras.Input(
  54         shape=(num_token_predictions,), dtype=tf.int32)
  55 
  56     # Invoke the trainer model on the inputs. This causes the layer to be built.
  57     outputs = bert_trainer_model(
  58         [word_ids, mask, type_ids, masked_lm_positions])
  59 
  60     # Validate that the outputs are of the expected shape.
  61     expected_lm_shape = [None, num_token_predictions, vocab_size]
  62     expected_classification_shape = [None, num_classes]
  63     self.assertAllEqual(expected_lm_shape, outputs['masked_lm'].shape.as_list())
  64     self.assertAllEqual(expected_classification_shape,
  65                         outputs['classification'].shape.as_list())

  67   def test_bert_trainer_tensor_call(self):
  68     """Validate that the Keras object can be invoked."""
  69     # Build a transformer network to use within the BERT trainer. (Here, we use
  70     # a short sequence_length for convenience.)
  71     test_network = networks.TransformerEncoder(
  72         vocab_size=100, num_layers=2, sequence_length=2)
  73 
  74     # Create a BERT trainer with the created network.
  75     bert_trainer_model = bert_pretrainer.BertPretrainer(
  76         test_network, num_classes=2, num_token_predictions=2)
  77 
  78     # Create a set of 2-dimensional data tensors to feed into the model.
  79     word_ids = tf.constant([[1, 1], [2, 2]], dtype=tf.int32)
  80     mask = tf.constant([[1, 1], [1, 0]], dtype=tf.int32)
  81     type_ids = tf.constant([[1, 1], [2, 2]], dtype=tf.int32)
  82     lm_mask = tf.constant([[1, 1], [1, 0]], dtype=tf.int32)
  83 
  84     # Invoke the trainer model on the tensors. In Eager mode, this does the
  85     # actual calculation. (We can't validate the outputs, since the network is
  86     # too complex: this simply ensures we're not hitting runtime errors.)
  87     _ = bert_trainer_model([word_ids, mask, type_ids, lm_mask])

  89   def test_serialize_deserialize(self):
  90     """Validate that the BERT trainer can be serialized and deserialized."""
  91     # Build a transformer network to use within the BERT trainer. (Here, we use
  92     # a short sequence_length for convenience.)
  93     test_network = networks.TransformerEncoder(
  94         vocab_size=100, num_layers=2, sequence_length=5)
  95 
  96     # Create a BERT trainer with the created network. (Note that all the args
  97     # are different, so we can catch any serialization mismatches.)
  98     bert_trainer_model = bert_pretrainer.BertPretrainer(
  99         test_network, num_classes=4, num_token_predictions=3)
 100 
 101     # Create another BERT trainer via serialization and deserialization.
 102     config = bert_trainer_model.get_config()
 103     new_bert_trainer_model = bert_pretrainer.BertPretrainer.from_config(config)
 104 
 105     # Validate that the config can be forced to JSON.
 106     _ = new_bert_trainer_model.to_json()
 107 
 108     # If the serialization was successful, the new config should match the old.
 109     self.assertAllEqual(bert_trainer_model.get_config(),
 110                         new_bert_trainer_model.get_config())
