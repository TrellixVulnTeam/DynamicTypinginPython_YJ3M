  31 @tf.keras.utils.register_keras_serializable(package='Text')
  32 @gin.configurable
  33 class EncoderScaffold(tf.keras.Model):
  34   """Bi-directional Transformer-based encoder network scaffold.
  35 
  36   This network allows users to flexibly implement an encoder similar to the one
  37   described in "BERT: Pre-training of Deep Bidirectional Transformers for
  38   Language Understanding" (https://arxiv.org/abs/1810.04805).
  39 
  40   In this network, users can choose to provide a custom embedding subnetwork
  41   (which will replace the standard embedding logic) and/or a custom hidden layer
  42   class (which will replace the Transformer instantiation in the encoder). For
  43   each of these custom injection points, users can pass either a class or a
  44   class instance. If a class is passed, that class will be instantiated using
  45   the 'embedding_cfg' or 'hidden_cfg' argument, respectively; if an instance
  46   is passed, that instance will be invoked. (In the case of hidden_cls, the
  47   instance will be invoked 'num_hidden_instances' times.
  48 
  49   If the hidden_cls is not overridden, a default transformer layer will be
  50   instantiated.
  51 
  52   Arguments:
  53     pooled_output_dim: The dimension of pooled output.
  54     pooler_layer_initializer: The initializer for the classification
  55       layer.
  56     embedding_cls: The class or instance to use to embed the input data. This
  57       class or instance defines the inputs to this encoder and outputs
  58       (1) embeddings tensor with shape [batch_size, seq_length, hidden_size] and
  59       (2) attention masking with tensor [batch_size, seq_length, seq_length].
  60       If embedding_cls is not set, a default embedding network
  61       (from the original BERT paper) will be created.
  62     embedding_cfg: A dict of kwargs to pass to the embedding_cls, if it needs to
  63       be instantiated. If embedding_cls is not set, a config dict must be
  64       passed to 'embedding_cfg' with the following values:
  65       "vocab_size": The size of the token vocabulary.
  66       "type_vocab_size": The size of the type vocabulary.
  67       "hidden_size": The hidden size for this encoder.
  68       "max_seq_length": The maximum sequence length for this encoder.
  69       "seq_length": The sequence length for this encoder.
  70       "initializer": The initializer for the embedding portion of this encoder.
  71       "dropout_rate": The dropout rate to apply before the encoding layers.
  72     embedding_data: A reference to the embedding weights that will be used to
  73       train the masked language model, if necessary. This is optional, and only
  74       needed if (1) you are overriding embedding_cls and (2) are doing standard
  75       pretraining.
  76     num_hidden_instances: The number of times to instantiate and/or invoke the
  77       hidden_cls.
  78     hidden_cls: The class or instance to encode the input data. If hidden_cls is
  79       not set, a KerasBERT transformer layer will be used as the encoder class.
  80     hidden_cfg: A dict of kwargs to pass to the hidden_cls, if it needs to be
  81       instantiated. If hidden_cls is not set, a config dict must be passed to
  82       'hidden_cfg' with the following values:
  83         "num_attention_heads": The number of attention heads. The hidden size
  84           must be divisible by num_attention_heads.
  85         "intermediate_size": The intermediate size of the transformer.
  86         "intermediate_activation": The activation to apply in the transfomer.
  87         "dropout_rate": The overall dropout rate for the transformer layers.
  88         "attention_dropout_rate": The dropout rate for the attention layers.
  89         "kernel_initializer": The initializer for the transformer layers.
  90     return_all_layer_outputs: Whether to output sequence embedding outputs of
  91       all encoder transformer layers.
  92   """
  93 
  94   def __init__(
  95       self,
  96       pooled_output_dim,
  97       pooler_layer_initializer=tf.keras.initializers.TruncatedNormal(
  98           stddev=0.02),
  99       embedding_cls=None,
 100       embedding_cfg=None,
 101       embedding_data=None,
 102       num_hidden_instances=1,
 103       hidden_cls=layers.Transformer,
 104       hidden_cfg=None,
 105       return_all_layer_outputs=False,
 106       **kwargs):
 107     self._self_setattr_tracking = False
 108     self._hidden_cls = hidden_cls
 109     self._hidden_cfg = hidden_cfg
 110     self._num_hidden_instances = num_hidden_instances
 111     self._pooled_output_dim = pooled_output_dim
 112     self._pooler_layer_initializer = pooler_layer_initializer
 113     self._embedding_cls = embedding_cls
 114     self._embedding_cfg = embedding_cfg
 115     self._embedding_data = embedding_data
 116     self._return_all_layer_outputs = return_all_layer_outputs
 117     self._kwargs = kwargs
 118 
 119     if embedding_cls:
 120       if inspect.isclass(embedding_cls):
 121         self._embedding_network = embedding_cls(
 122             **embedding_cfg) if embedding_cfg else embedding_cls()
 123       else:
 124         self._embedding_network = embedding_cls
 125       inputs = self._embedding_network.inputs
 126       embeddings, attention_mask = self._embedding_network(inputs)
 127     else:
 128       self._embedding_network = None
 129       word_ids = tf.keras.layers.Input(
 130           shape=(embedding_cfg['seq_length'],),
 131           dtype=tf.int32,
 132           name='input_word_ids')
 133       mask = tf.keras.layers.Input(
 134           shape=(embedding_cfg['seq_length'],),
 135           dtype=tf.int32,
 136           name='input_mask')
 137       type_ids = tf.keras.layers.Input(
 138           shape=(embedding_cfg['seq_length'],),
 139           dtype=tf.int32,
 140           name='input_type_ids')
 141       inputs = [word_ids, mask, type_ids]
 142 
 143       self._embedding_layer = layers.OnDeviceEmbedding(
 144           vocab_size=embedding_cfg['vocab_size'],
 145           embedding_width=embedding_cfg['hidden_size'],
 146           initializer=embedding_cfg['initializer'],
 147           name='word_embeddings')
 148 
 149       word_embeddings = self._embedding_layer(word_ids)
 150 
 151       # Always uses dynamic slicing for simplicity.
 152       self._position_embedding_layer = layers.PositionEmbedding(
 153           initializer=embedding_cfg['initializer'],
 154           use_dynamic_slicing=True,
 155           max_sequence_length=embedding_cfg['max_seq_length'],
 156           name='position_embedding')
 157       position_embeddings = self._position_embedding_layer(word_embeddings)
 158 
 159       type_embeddings = (
 160           layers.OnDeviceEmbedding(
 161               vocab_size=embedding_cfg['type_vocab_size'],
 162               embedding_width=embedding_cfg['hidden_size'],
 163               initializer=embedding_cfg['initializer'],
 164               use_one_hot=True,
 165               name='type_embeddings')(type_ids))
 166 
 167       embeddings = tf.keras.layers.Add()(
 168           [word_embeddings, position_embeddings, type_embeddings])
 169       embeddings = (
 170           tf.keras.layers.LayerNormalization(
 171               name='embeddings/layer_norm',
 172               axis=-1,
 173               epsilon=1e-12,
 174               dtype=tf.float32)(embeddings))
 175       embeddings = (
 176           tf.keras.layers.Dropout(
 177               rate=embedding_cfg['dropout_rate'])(embeddings))
 178 
 179       attention_mask = layers.SelfAttentionMask()([embeddings, mask])
 180 
 181     data = embeddings
 182 
 183     layer_output_data = []
 184     self._hidden_layers = []
 185     for _ in range(num_hidden_instances):
 186       if inspect.isclass(hidden_cls):
 187         layer = hidden_cls(**hidden_cfg) if hidden_cfg else hidden_cls()
 188       else:
 189         layer = hidden_cls
 190       data = layer([data, attention_mask])
 191       layer_output_data.append(data)
 192       self._hidden_layers.append(layer)
 193 
 194     first_token_tensor = (
 195         tf.keras.layers.Lambda(lambda x: tf.squeeze(x[:, 0:1, :], axis=1))(
 196             layer_output_data[-1]))
 197     self._pooler_layer = tf.keras.layers.Dense(
 198         units=pooled_output_dim,
 199         activation='tanh',
 200         kernel_initializer=pooler_layer_initializer,
 201         name='cls_transform')
 202     cls_output = self._pooler_layer(first_token_tensor)
 203 
 204     if return_all_layer_outputs:
 205       outputs = [layer_output_data, cls_output]
 206     else:
 207       outputs = [layer_output_data[-1], cls_output]
 208 
 209     super(EncoderScaffold, self).__init__(
 210         inputs=inputs, outputs=outputs, **kwargs)
 211 
 212   def get_config(self):
 213     config_dict = {
 214         'num_hidden_instances':
 215             self._num_hidden_instances,
 216         'pooled_output_dim':
 217             self._pooled_output_dim,
 218         'pooler_layer_initializer':
 219             self._pooler_layer_initializer,
 220         'embedding_cls':
 221             self._embedding_network,
 222         'embedding_cfg':
 223             self._embedding_cfg,
 224         'hidden_cfg':
 225             self._hidden_cfg,
 226         'return_all_layer_outputs':
 227             self._return_all_layer_outputs,
 228     }
 229     if inspect.isclass(self._hidden_cls):
 230       config_dict['hidden_cls_string'] = tf.keras.utils.get_registered_name(
 231           self._hidden_cls)
 232     else:
 233       config_dict['hidden_cls'] = self._hidden_cls
 234 
 235     config_dict.update(self._kwargs)
 236     return config_dict
 237 
 238   @classmethod
 239   def from_config(cls, config, custom_objects=None):
 240     if 'hidden_cls_string' in config:
 241       config['hidden_cls'] = tf.keras.utils.get_registered_object(
 242           config['hidden_cls_string'], custom_objects=custom_objects)
 243       del config['hidden_cls_string']
 244     return cls(**config)

 246   def get_embedding_table(self):
 247     if self._embedding_network is None:
 248       # In this case, we don't have a custom embedding network and can return
 249       # the standard embedding data.
 250       return self._embedding_layer.embeddings
 251 
 252     if self._embedding_data is None:
 253       raise RuntimeError(('The EncoderScaffold %s does not have a reference '
 254                           'to the embedding data. This is required when you '
 255                           'pass a custom embedding network to the scaffold. '
 256                           'It is also possible that you are trying to get '
 257                           'embedding data from an embedding scaffold with a '
 258                           'custom embedding network where the scaffold has '
 259                           'been serialized and deserialized. Unfortunately, '
 260                           'accessing custom embedding references after '
 261                           'serialization is not yet supported.') % self.name)
 262     else:
 263       return self._embedding_data
