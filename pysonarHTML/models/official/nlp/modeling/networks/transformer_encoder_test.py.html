  38   def test_network_creation(self):
  39     hidden_size = 32
  40     sequence_length = 21
  41     # Create a small TransformerEncoder for testing.
  42     test_network = transformer_encoder.TransformerEncoder(
  43         vocab_size=100,
  44         hidden_size=hidden_size,
  45         sequence_length=sequence_length,
  46         num_attention_heads=2,
  47         num_layers=3)
  48     # Create the inputs (note that the first dimension is implicit).
  49     word_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  50     mask = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  51     type_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  52     data, pooled = test_network([word_ids, mask, type_ids])
  53 
  54     self.assertIsInstance(test_network.transformer_layers, list)
  55     self.assertLen(test_network.transformer_layers, 3)
  56     self.assertIsInstance(test_network.pooler_layer, tf.keras.layers.Dense)
  57 
  58     expected_data_shape = [None, sequence_length, hidden_size]
  59     expected_pooled_shape = [None, hidden_size]
  60     self.assertAllEqual(expected_data_shape, data.shape.as_list())
  61     self.assertAllEqual(expected_pooled_shape, pooled.shape.as_list())
  62 
  63     # The default output dtype is float32.
  64     self.assertAllEqual(tf.float32, data.dtype)
  65     self.assertAllEqual(tf.float32, pooled.dtype)

  67   def test_all_encoder_outputs_network_creation(self):
  68     hidden_size = 32
  69     sequence_length = 21
  70     # Create a small TransformerEncoder for testing.
  71     test_network = transformer_encoder.TransformerEncoder(
  72         vocab_size=100,
  73         hidden_size=hidden_size,
  74         sequence_length=sequence_length,
  75         num_attention_heads=2,
  76         num_layers=3,
  77         return_all_encoder_outputs=True)
  78     # Create the inputs (note that the first dimension is implicit).
  79     word_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  80     mask = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  81     type_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  82     all_encoder_outputs, pooled = test_network([word_ids, mask, type_ids])
  83 
  84     expected_data_shape = [None, sequence_length, hidden_size]
  85     expected_pooled_shape = [None, hidden_size]

  87     for data in all_encoder_outputs:
  88       self.assertAllEqual(expected_data_shape, data.shape.as_list())
  89     self.assertAllEqual(expected_pooled_shape, pooled.shape.as_list())

  91     # The default output dtype is float32.
  92     self.assertAllEqual(tf.float32, all_encoder_outputs[-1].dtype)
  93     self.assertAllEqual(tf.float32, pooled.dtype)

 126   def test_network_invocation(self, output_range, out_seq_len):
 127     hidden_size = 32
 128     sequence_length = 21
 129     vocab_size = 57
 130     num_types = 7
 131     # Create a small TransformerEncoder for testing.
 132     test_network = transformer_encoder.TransformerEncoder(
 133         vocab_size=vocab_size,
 134         hidden_size=hidden_size,
 135         sequence_length=sequence_length,
 136         num_attention_heads=2,
 137         num_layers=3,
 138         type_vocab_size=num_types,
 139         output_range=output_range)
 140     self.assertTrue(
 141         test_network._position_embedding_layer._use_dynamic_slicing)
 142     # Create the inputs (note that the first dimension is implicit).
 143     word_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
 144     mask = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
 145     type_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
 146     data, pooled = test_network([word_ids, mask, type_ids])
 147 
 148     # Create a model based off of this network:
 149     model = tf.keras.Model([word_ids, mask, type_ids], [data, pooled])
 150 
 151     # Invoke the model. We can't validate the output data here (the model is too
 152     # complex) but this will catch structural runtime errors.
 153     batch_size = 3
 154     word_id_data = np.random.randint(
 155         vocab_size, size=(batch_size, sequence_length))
 156     mask_data = np.random.randint(2, size=(batch_size, sequence_length))
 157     type_id_data = np.random.randint(
 158         num_types, size=(batch_size, sequence_length))
 159     _ = model.predict([word_id_data, mask_data, type_id_data])
 160 
 161     # Creates a TransformerEncoder with max_sequence_length != sequence_length
 162     max_sequence_length = 128
 163     test_network = transformer_encoder.TransformerEncoder(
 164         vocab_size=vocab_size,
 165         hidden_size=hidden_size,
 166         sequence_length=sequence_length,
 167         max_sequence_length=max_sequence_length,
 168         num_attention_heads=2,
 169         num_layers=3,
 170         type_vocab_size=num_types)
 171     self.assertTrue(test_network._position_embedding_layer._use_dynamic_slicing)
 172     model = tf.keras.Model([word_ids, mask, type_ids], [data, pooled])
 173     outputs = model.predict([word_id_data, mask_data, type_id_data])
 174     self.assertEqual(outputs[0].shape[1], out_seq_len)
 175 
 176     # Creates a TransformerEncoder with embedding_width != hidden_size
 177     test_network = transformer_encoder.TransformerEncoder(
 178         vocab_size=vocab_size,
 179         hidden_size=hidden_size,
 180         sequence_length=sequence_length,
 181         max_sequence_length=max_sequence_length,
 182         num_attention_heads=2,
 183         num_layers=3,
 184         type_vocab_size=num_types,
 185         embedding_width=16)
 186     model = tf.keras.Model([word_ids, mask, type_ids], [data, pooled])
 187     outputs = model.predict([word_id_data, mask_data, type_id_data])
 188     self.assertEqual(outputs[0].shape[-1], hidden_size)
 189     self.assertTrue(hasattr(test_network, "_embedding_projection"))

 191   def test_serialize_deserialize(self):
 192     tf.keras.mixed_precision.experimental.set_policy("mixed_float16")
 193     # Create a network object that sets all of its config options.
 194     kwargs = dict(
 195         vocab_size=100,
 196         hidden_size=32,
 197         num_layers=3,
 198         num_attention_heads=2,
 199         sequence_length=21,
 200         max_sequence_length=21,
 201         type_vocab_size=12,
 202         intermediate_size=1223,
 203         activation="relu",
 204         dropout_rate=0.05,
 205         attention_dropout_rate=0.22,
 206         initializer="glorot_uniform",
 207         return_all_encoder_outputs=False,
 208         output_range=-1,
 209         embedding_width=16)
 210     network = transformer_encoder.TransformerEncoder(**kwargs)
 211 
 212     expected_config = dict(kwargs)
 213     expected_config["activation"] = tf.keras.activations.serialize(
 214         tf.keras.activations.get(expected_config["activation"]))
 215     expected_config["initializer"] = tf.keras.initializers.serialize(
 216         tf.keras.initializers.get(expected_config["initializer"]))
 217     self.assertEqual(network.get_config(), expected_config)
 218 
 219     # Create another network object from the first object's config.
 220     new_network = transformer_encoder.TransformerEncoder.from_config(
 221         network.get_config())
 222 
 223     # Validate that the config can be forced to JSON.
 224     _ = new_network.to_json()
 225 
 226     # If the serialization was successful, the new config should match the old.
 227     self.assertAllEqual(network.get_config(), new_network.get_config())
