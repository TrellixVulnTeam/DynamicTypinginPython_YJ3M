  34   def tearDown(self):
  35     super(AlbertTransformerEncoderTest, self).tearDown()
  36     tf.keras.mixed_precision.experimental.set_policy("float32")

  38   @parameterized.named_parameters(
  39       dict(testcase_name="default", expected_dtype=tf.float32),
  40       dict(
  41           testcase_name="with_float16_dtype",
  42           expected_dtype=tf.float16),
  43   )
  44   def test_network_creation(self, expected_dtype):
  45     hidden_size = 32
  46     sequence_length = 21
  47 
  48     kwargs = dict(
  49         vocab_size=100,
  50         hidden_size=hidden_size,
  51         sequence_length=sequence_length,
  52         num_attention_heads=2,
  53         num_layers=3)
  54     if expected_dtype == tf.float16:
  55       tf.keras.mixed_precision.experimental.set_policy("mixed_float16")
  56 
  57     # Create a small TransformerEncoder for testing.
  58     test_network = albert_transformer_encoder.AlbertTransformerEncoder(**kwargs)
  59 
  60     # Create the inputs (note that the first dimension is implicit).
  61     word_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  62     mask = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  63     type_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  64     data, pooled = test_network([word_ids, mask, type_ids])
  65 
  66     expected_data_shape = [None, sequence_length, hidden_size]
  67     expected_pooled_shape = [None, hidden_size]
  68     self.assertAllEqual(expected_data_shape, data.shape.as_list())
  69     self.assertAllEqual(expected_pooled_shape, pooled.shape.as_list())
  70 
  71     # If float_dtype is set to float16, the data output is float32 (from a layer
  72     # norm) and pool output should be float16.
  73     self.assertEqual(tf.float32, data.dtype)
  74     self.assertEqual(expected_dtype, pooled.dtype)
  75 
  76     # ALBERT has additonal 'embedding_hidden_mapping_in' weights and
  77     # it shares transformer weights.
  78     self.assertNotEmpty(
  79         [x for x in test_network.weights if "embedding_projection/" in x.name])
  80     self.assertNotEmpty(
  81         [x for x in test_network.weights if "transformer/" in x.name])
  82     self.assertEmpty(
  83         [x for x in test_network.weights if "transformer/layer" in x.name])

  85   def test_network_invocation(self):
  86     hidden_size = 32
  87     sequence_length = 21
  88     vocab_size = 57
  89     num_types = 7
  90     # Create a small TransformerEncoder for testing.
  91     test_network = albert_transformer_encoder.AlbertTransformerEncoder(
  92         vocab_size=vocab_size,
  93         embedding_width=8,
  94         hidden_size=hidden_size,
  95         sequence_length=sequence_length,
  96         num_attention_heads=2,
  97         num_layers=3,
  98         type_vocab_size=num_types)
  99     self.assertTrue(
 100         test_network._position_embedding_layer._use_dynamic_slicing)
 101     # Create the inputs (note that the first dimension is implicit).
 102     word_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
 103     mask = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
 104     type_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
 105     data, pooled = test_network([word_ids, mask, type_ids])
 106 
 107     # Create a model based off of this network:
 108     model = tf.keras.Model([word_ids, mask, type_ids], [data, pooled])
 109 
 110     # Invoke the model. We can't validate the output data here (the model is too
 111     # complex) but this will catch structural runtime errors.
 112     batch_size = 3
 113     word_id_data = np.random.randint(
 114         vocab_size, size=(batch_size, sequence_length))
 115     mask_data = np.random.randint(2, size=(batch_size, sequence_length))
 116     type_id_data = np.random.randint(
 117         num_types, size=(batch_size, sequence_length))
 118     _ = model.predict([word_id_data, mask_data, type_id_data])
 119 
 120     # Creates a TransformerEncoder with max_sequence_length != sequence_length
 121     max_sequence_length = 128
 122     test_network = albert_transformer_encoder.AlbertTransformerEncoder(
 123         vocab_size=vocab_size,
 124         embedding_width=8,
 125         hidden_size=hidden_size,
 126         sequence_length=sequence_length,
 127         max_sequence_length=max_sequence_length,
 128         num_attention_heads=2,
 129         num_layers=3,
 130         type_vocab_size=num_types)
 131     self.assertTrue(test_network._position_embedding_layer._use_dynamic_slicing)
 132     model = tf.keras.Model([word_ids, mask, type_ids], [data, pooled])
 133     _ = model.predict([word_id_data, mask_data, type_id_data])

 135   def test_serialize_deserialize(self):
 136     tf.keras.mixed_precision.experimental.set_policy("mixed_float16")
 137     # Create a network object that sets all of its config options.
 138     kwargs = dict(
 139         vocab_size=100,
 140         embedding_width=8,
 141         hidden_size=32,
 142         num_layers=3,
 143         num_attention_heads=2,
 144         sequence_length=21,
 145         max_sequence_length=21,
 146         type_vocab_size=12,
 147         intermediate_size=1223,
 148         activation="relu",
 149         dropout_rate=0.05,
 150         attention_dropout_rate=0.22,
 151         initializer="glorot_uniform")
 152     network = albert_transformer_encoder.AlbertTransformerEncoder(**kwargs)
 153 
 154     expected_config = dict(kwargs)
 155     expected_config["activation"] = tf.keras.activations.serialize(
 156         tf.keras.activations.get(expected_config["activation"]))
 157     expected_config["initializer"] = tf.keras.initializers.serialize(
 158         tf.keras.initializers.get(expected_config["initializer"]))
 159     self.assertEqual(network.get_config(), expected_config)
 160 
 161     # Create another network object from the first object's config.
 162     new_network = (
 163         albert_transformer_encoder.AlbertTransformerEncoder.from_config(
 164             network.get_config()))
 165 
 166     # Validate that the config can be forced to JSON.
 167     _ = new_network.to_json()
 168 
 169     # If the serialization was successful, the new config should match the old.
 170     self.assertAllEqual(network.get_config(), new_network.get_config())
