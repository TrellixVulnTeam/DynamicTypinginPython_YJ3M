  49   def test_network_invocation(self):
  50     """Validate that the Keras object can be invoked."""
  51     sequence_length = 15
  52     input_width = 512
  53     test_network = span_labeling.SpanLabeling(input_width=input_width)
  54 
  55     # Create a 3-dimensional input (the first dimension is implicit).
  56     sequence_data = tf.keras.Input(
  57         shape=(sequence_length, input_width), dtype=tf.float32)
  58     outputs = test_network(sequence_data)
  59     model = tf.keras.Model(sequence_data, outputs)
  60 
  61     # Invoke the network as part of a Model.
  62     batch_size = 3
  63     input_data = 10 * np.random.random_sample(
  64         (batch_size, sequence_length, input_width))
  65     start_outputs, end_outputs = model.predict(input_data)
  66 
  67     # Validate that the outputs are of the expected shape.
  68     expected_output_shape = (batch_size, sequence_length)
  69     self.assertEqual(expected_output_shape, start_outputs.shape)
  70     self.assertEqual(expected_output_shape, end_outputs.shape)

  72   def test_network_invocation_with_internal_logit_output(self):
  73     """Validate that the logit outputs are correct."""
  74     sequence_length = 15
  75     input_width = 512
  76     test_network = span_labeling.SpanLabeling(
  77         input_width=input_width, output='predictions')
  78     # Create a 3-dimensional input (the first dimension is implicit).
  79     sequence_data = tf.keras.Input(
  80         shape=(sequence_length, input_width), dtype=tf.float32)
  81     output = test_network(sequence_data)
  82     model = tf.keras.Model(sequence_data, output)
  83     logit_model = tf.keras.Model(
  84         test_network.inputs,
  85         [test_network.start_logits, test_network.end_logits])
  86 
  87     batch_size = 3
  88     input_data = 10 * np.random.random_sample(
  89         (batch_size, sequence_length, input_width))
  90     start_outputs, end_outputs = model.predict(input_data)
  91     start_logits, end_logits = logit_model.predict(input_data)
  92 
  93     # Ensure that the tensor shapes are correct.
  94     expected_output_shape = (batch_size, sequence_length)
  95     self.assertEqual(expected_output_shape, start_outputs.shape)
  96     self.assertEqual(expected_output_shape, end_outputs.shape)
  97     self.assertEqual(expected_output_shape, start_logits.shape)
  98     self.assertEqual(expected_output_shape, end_logits.shape)
  99 
 100     # Ensure that the logits, when softmaxed, create the outputs.
 101     input_tensor = tf.keras.Input(expected_output_shape[1:])
 102     output_tensor = tf.keras.layers.Activation(tf.nn.log_softmax)(input_tensor)
 103     softmax_model = tf.keras.Model(input_tensor, output_tensor)
 104 
 105     start_softmax = softmax_model.predict(start_logits)
 106     self.assertAllClose(start_outputs, start_softmax)
 107     end_softmax = softmax_model.predict(end_logits)
 108     self.assertAllClose(end_outputs, end_softmax)

 110   def test_network_invocation_with_external_logit_output(self):
 111     """Validate that the logit outputs are correct."""
 112     sequence_length = 15
 113     input_width = 512
 114     test_network = span_labeling.SpanLabeling(
 115         input_width=input_width, output='predictions')
 116     logit_network = span_labeling.SpanLabeling(
 117         input_width=input_width, output='logits')
 118     logit_network.set_weights(test_network.get_weights())
 119 
 120     # Create a 3-dimensional input (the first dimension is implicit).
 121     sequence_data = tf.keras.Input(
 122         shape=(sequence_length, input_width), dtype=tf.float32)
 123     output = test_network(sequence_data)
 124     logit_output = logit_network(sequence_data)
 125     model = tf.keras.Model(sequence_data, output)
 126     logit_model = tf.keras.Model(sequence_data, logit_output)
 127 
 128     batch_size = 3
 129     input_data = 10 * np.random.random_sample(
 130         (batch_size, sequence_length, input_width))
 131     start_outputs, end_outputs = model.predict(input_data)
 132     start_logits, end_logits = logit_model.predict(input_data)
 133 
 134     # Ensure that the tensor shapes are correct.
 135     expected_output_shape = (batch_size, sequence_length)
 136     self.assertEqual(expected_output_shape, start_outputs.shape)
 137     self.assertEqual(expected_output_shape, end_outputs.shape)
 138     self.assertEqual(expected_output_shape, start_logits.shape)
 139     self.assertEqual(expected_output_shape, end_logits.shape)
 140 
 141     # Ensure that the logits, when softmaxed, create the outputs.
 142     input_tensor = tf.keras.Input(expected_output_shape[1:])
 143     output_tensor = tf.keras.layers.Activation(tf.nn.log_softmax)(input_tensor)
 144     softmax_model = tf.keras.Model(input_tensor, output_tensor)
 145 
 146     start_softmax = softmax_model.predict(start_logits)
 147     self.assertAllClose(start_outputs, start_softmax)
 148     end_softmax = softmax_model.predict(end_logits)
 149     self.assertAllClose(end_outputs, end_softmax)

 151   def test_serialize_deserialize(self):
 152     # Create a network object that sets all of its config options.
 153     network = span_labeling.SpanLabeling(
 154         input_width=128,
 155         activation='relu',
 156         initializer='zeros',
 157         output='predictions')
 158 
 159     # Create another network object from the first object's config.
 160     new_network = span_labeling.SpanLabeling.from_config(network.get_config())
 161 
 162     # Validate that the config can be forced to JSON.
 163     _ = new_network.to_json()
 164 
 165     # If the serialization was successful, the new config should match the old.
 166     self.assertAllEqual(network.get_config(), new_network.get_config())
