  31 # Test class that wraps a standard transformer layer. If this layer is called
  32 # at any point, the list passed to the config object will be filled with a
  33 # boolean 'True'. We register this class as a Keras serializable so we can
  34 # test serialization below.
  35 @tf.keras.utils.register_keras_serializable(package="TestOnly")
  36 class ValidatedTransformerLayer(layers.Transformer):
  37 
  38   def __init__(self, call_list, **kwargs):
  39     super(ValidatedTransformerLayer, self).__init__(**kwargs)
  40     self.list = call_list
  41 
  42   def call(self, inputs):
  43     self.list.append(True)
  44     return super(ValidatedTransformerLayer, self).call(inputs)
  45 
  46   def get_config(self):
  47     config = super(ValidatedTransformerLayer, self).get_config()
  48     config["call_list"] = []
  49     return config
