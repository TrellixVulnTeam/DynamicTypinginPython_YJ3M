  66   def test_network_invocation_with_internal_logits(self):
  67     """Validate that the logit outputs are correct."""
  68     sequence_length = 5
  69     input_width = 512
  70     num_classes = 10
  71     test_object = token_classification.TokenClassification(
  72         input_width=input_width, num_classes=num_classes, output='predictions')
  73 
  74     # Create a 3-dimensional input (the first dimension is implicit).
  75     sequence_data = tf.keras.Input(shape=(sequence_length, input_width),
  76                                    dtype=tf.float32)
  77     output = test_object(sequence_data)
  78     model = tf.keras.Model(sequence_data, output)
  79     logits_model = tf.keras.Model(test_object.inputs, test_object.logits)
  80 
  81     batch_size = 3
  82     input_data = 10 * np.random.random_sample(
  83         (batch_size, sequence_length, input_width))
  84     outputs = model.predict(input_data)
  85     logits = logits_model.predict(input_data)
  86 
  87     # Ensure that the tensor shapes are correct.
  88     expected_output_shape = (batch_size, sequence_length, num_classes)
  89     self.assertEqual(expected_output_shape, outputs.shape)
  90     self.assertEqual(expected_output_shape, logits.shape)
  91 
  92     # Ensure that the logits, when softmaxed, create the outputs.
  93     input_tensor = tf.keras.Input(expected_output_shape[1:])
  94     output_tensor = tf.keras.layers.Activation(tf.nn.log_softmax)(input_tensor)
  95     softmax_model = tf.keras.Model(input_tensor, output_tensor)
  96 
  97     calculated_softmax = softmax_model.predict(logits)
  98     self.assertAllClose(outputs, calculated_softmax)

 128   def test_network_invocation_with_logit_output(self):
 129     """Validate that the logit outputs are correct."""
 130     sequence_length = 5
 131     input_width = 512
 132     num_classes = 10
 133     test_object = token_classification.TokenClassification(
 134         input_width=input_width, num_classes=num_classes, output='predictions')
 135     logit_object = token_classification.TokenClassification(
 136         input_width=input_width, num_classes=num_classes, output='logits')
 137     logit_object.set_weights(test_object.get_weights())
 138 
 139     # Create a 3-dimensional input (the first dimension is implicit).
 140     sequence_data = tf.keras.Input(shape=(sequence_length, input_width),
 141                                    dtype=tf.float32)
 142     output = test_object(sequence_data)
 143     logit_output = logit_object(sequence_data)
 144 
 145     model = tf.keras.Model(sequence_data, output)
 146     logits_model = tf.keras.Model(sequence_data, logit_output)
 147 
 148     batch_size = 3
 149     input_data = 10 * np.random.random_sample(
 150         (batch_size, sequence_length, input_width))
 151     outputs = model.predict(input_data)
 152     logits = logits_model.predict(input_data)
 153 
 154     # Ensure that the tensor shapes are correct.
 155     expected_output_shape = (batch_size, sequence_length, num_classes)
 156     self.assertEqual(expected_output_shape, outputs.shape)
 157     self.assertEqual(expected_output_shape, logits.shape)
 158 
 159     # Ensure that the logits, when softmaxed, create the outputs.
 160     input_tensor = tf.keras.Input(expected_output_shape[1:])
 161     output_tensor = tf.keras.layers.Activation(tf.nn.log_softmax)(input_tensor)
 162     softmax_model = tf.keras.Model(input_tensor, output_tensor)
 163 
 164     calculated_softmax = softmax_model.predict(logits)
 165     self.assertAllClose(outputs, calculated_softmax)

 167   def test_serialize_deserialize(self):
 168     # Create a network object that sets all of its config options.
 169     network = token_classification.TokenClassification(
 170         input_width=128,
 171         num_classes=10,
 172         initializer='zeros',
 173         output='predictions')
 174 
 175     # Create another network object from the first object's config.
 176     new_network = token_classification.TokenClassification.from_config(
 177         network.get_config())
 178 
 179     # Validate that the config can be forced to JSON.
 180     _ = new_network.to_json()
 181 
 182     # If the serialization was successful, the new config should match the old.
 183     self.assertAllEqual(network.get_config(), new_network.get_config())
