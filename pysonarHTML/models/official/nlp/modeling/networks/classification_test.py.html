  63   def test_network_invocation_with_internal_logits(self):
  64     """Validate that the logit outputs are correct."""
  65     input_width = 512
  66     num_classes = 10
  67     test_object = classification.Classification(
  68         input_width=input_width, num_classes=num_classes, output='predictions')
  69 
  70     # Create a 2-dimensional input (the first dimension is implicit).
  71     cls_data = tf.keras.Input(shape=(input_width,), dtype=tf.float32)
  72     output = test_object(cls_data)
  73     model = tf.keras.Model(cls_data, output)
  74     logits_model = tf.keras.Model(test_object.inputs, test_object.logits)
  75 
  76     batch_size = 3
  77     input_data = 10 * np.random.random_sample((batch_size, input_width))
  78     outputs = model.predict(input_data)
  79     logits = logits_model.predict(input_data)
  80 
  81     # Ensure that the tensor shapes are correct.
  82     expected_output_shape = (batch_size, num_classes)
  83     self.assertEqual(expected_output_shape, outputs.shape)
  84     self.assertEqual(expected_output_shape, logits.shape)
  85 
  86     # Ensure that the logits, when softmaxed, create the outputs.
  87     input_tensor = tf.keras.Input(expected_output_shape[1:])
  88     output_tensor = tf.keras.layers.Activation(tf.nn.log_softmax)(input_tensor)
  89     softmax_model = tf.keras.Model(input_tensor, output_tensor)
  90 
  91     calculated_softmax = softmax_model.predict(logits)
  92     self.assertAllClose(outputs, calculated_softmax)

 120   def test_network_invocation_with_logit_output(self):
 121     """Validate that the logit outputs are correct."""
 122     input_width = 512
 123     num_classes = 10
 124     test_object = classification.Classification(
 125         input_width=input_width, num_classes=num_classes, output='predictions')
 126     logit_object = classification.Classification(
 127         input_width=input_width, num_classes=num_classes, output='logits')
 128     logit_object.set_weights(test_object.get_weights())
 129 
 130     # Create a 2-dimensional input (the first dimension is implicit).
 131     cls_data = tf.keras.Input(shape=(input_width,), dtype=tf.float32)
 132     output = test_object(cls_data)
 133     logit_output = logit_object(cls_data)
 134 
 135     model = tf.keras.Model(cls_data, output)
 136     logits_model = tf.keras.Model(cls_data, logit_output)
 137 
 138     batch_size = 3
 139     input_data = 10 * np.random.random_sample((batch_size, input_width))
 140     outputs = model.predict(input_data)
 141     logits = logits_model.predict(input_data)
 142 
 143     # Ensure that the tensor shapes are correct.
 144     expected_output_shape = (batch_size, num_classes)
 145     self.assertEqual(expected_output_shape, outputs.shape)
 146     self.assertEqual(expected_output_shape, logits.shape)
 147 
 148     # Ensure that the logits, when softmaxed, create the outputs.
 149     input_tensor = tf.keras.Input(expected_output_shape[1:])
 150     output_tensor = tf.keras.layers.Activation(tf.nn.log_softmax)(input_tensor)
 151     softmax_model = tf.keras.Model(input_tensor, output_tensor)
 152 
 153     calculated_softmax = softmax_model.predict(logits)
 154     self.assertAllClose(outputs, calculated_softmax)

 156   def test_serialize_deserialize(self):
 157     # Create a network object that sets all of its config options.
 158     network = classification.Classification(
 159         input_width=128,
 160         num_classes=10,
 161         initializer='zeros',
 162         output='predictions')
 163 
 164     # Create another network object from the first object's config.
 165     new_network = classification.Classification.from_config(
 166         network.get_config())
 167 
 168     # Validate that the config can be forced to JSON.
 169     _ = new_network.to_json()
 170 
 171     # If the serialization was successful, the new config should match the old.
 172     self.assertAllEqual(network.get_config(), new_network.get_config())
