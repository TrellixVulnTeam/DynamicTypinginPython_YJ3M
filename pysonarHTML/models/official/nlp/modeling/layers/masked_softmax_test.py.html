  15 """Tests for Keras-based masked softmax layer."""

  21 import numpy as np
  22 import tensorflow as tf

  24 from tensorflow.python.keras import keras_parameterized  # pylint: disable=g-direct-tensorflow-import
  25 from official.nlp.modeling.layers import masked_softmax

  28 # This decorator runs the test in V1, V2-Eager, and V2-Functional mode. It
  29 # guarantees forward compatibility of this code for the V2 switchover.
  30 @keras_parameterized.run_all_keras_modes
  31 class MaskedSoftmaxLayerTest(keras_parameterized.TestCase):
  32 
  33   def test_non_masked_softmax(self):
  34     test_layer = masked_softmax.MaskedSoftmax()
  35     input_tensor = tf.keras.Input(shape=(4, 8))
  36     output = test_layer(input_tensor)
  37     model = tf.keras.Model(input_tensor, output)
  38 
  39     input_data = 10 * np.random.random_sample((3, 4, 8))
  40     output_data = model.predict(input_data)
  41     expected_data = tf.nn.softmax(input_data)
  42     self.assertAllClose(expected_data, output_data)
  43 
  44   def test_masked_softmax(self):
  45     test_layer = masked_softmax.MaskedSoftmax()
  46     input_tensor = tf.keras.Input(shape=(4, 8))
  47     mask_tensor = tf.keras.Input(shape=(4, 8))
  48     output = test_layer(input_tensor, mask_tensor)
  49     model = tf.keras.Model([input_tensor, mask_tensor], output)
  50 
  51     input_data = 10 * np.random.random_sample((3, 4, 8))
  52     mask_data = np.random.randint(2, size=(3, 4, 8))
  53 
  54     output_data = model.predict([input_data, mask_data])
  55     expected_zeros = np.greater(mask_data, 0)
  56     is_zeros = np.greater(output_data, 0)
  57     self.assertAllEqual(expected_zeros, is_zeros)
  58 
  59   def test_masked_softmax_with_none_mask(self):
  60     test_layer = masked_softmax.MaskedSoftmax()
  61     input_tensor = tf.keras.Input(shape=(4, 8))
  62     output = test_layer(input_tensor, None)
  63     model = tf.keras.Model(input_tensor, output)
  64 
  65     input_data = 10 * np.random.random_sample((3, 4, 8))
  66     output_data = model.predict(input_data)
  67     expected_data = tf.nn.softmax(input_data)
  68     self.assertAllClose(expected_data, output_data)
  69 
  70   def test_softmax_with_axes_expansion(self):
  71     test_layer = masked_softmax.MaskedSoftmax(mask_expansion_axes=[1])
  72     input_tensor = tf.keras.Input(shape=(4, 8))
  73     mask_tensor = tf.keras.Input(shape=(8))
  74     output = test_layer(input_tensor, mask_tensor)
  75     model = tf.keras.Model([input_tensor, mask_tensor], output)
  76 
  77     input_data = 10 * np.random.random_sample((3, 4, 8))
  78     mask_data = np.random.randint(2, size=(3, 8))
  79 
  80     output_data = model.predict([input_data, mask_data])
  81     expanded_mask = np.expand_dims(mask_data, axis=1) * np.ones_like(input_data)
  82     expected_zeros = np.greater(expanded_mask, 0)
  83     is_zeros = np.greater(output_data, 0)
  84     self.assertAllEqual(expected_zeros, is_zeros)
  85 
  86   def test_masked_softmax_high_dims(self):
  87     test_layer = masked_softmax.MaskedSoftmax(
  88         mask_expansion_axes=[1], normalization_axes=[6, 7])
  89     input_shape = [2, 3, 4, 5, 6, 7, 8]
  90     mask_shape = [5, 6, 7, 8]
  91     input_tensor = tf.keras.Input(shape=input_shape)
  92     mask_tensor = tf.keras.Input(shape=mask_shape)
  93     output = test_layer(input_tensor, mask_tensor)
  94     model = tf.keras.Model([input_tensor, mask_tensor], output)
  95 
  96     input_data = 10 * np.random.random_sample([3] + input_shape)
  97     mask_data = np.random.randint(2, size=[3] + mask_shape)
  98 
  99     output_data = model.predict([input_data, mask_data])
 100     expanded_mask = np.expand_dims(mask_data, axis=1)
 101     expanded_mask = np.expand_dims(expanded_mask, axis=1)
 102     expanded_mask = np.expand_dims(
 103         expanded_mask, axis=1) * np.ones_like(input_data)
 104     expected_zeros = np.greater(expanded_mask, 0)
 105     is_zeros = np.greater(output_data, 0)
 106     self.assertAllEqual(expected_zeros, is_zeros)
 107 
 108   def test_serialize_deserialize(self):
 109     test_layer = masked_softmax.MaskedSoftmax(
 110         mask_expansion_axes=[1], normalization_axes=[6, 7])
 111     new_layer = masked_softmax.MaskedSoftmax.from_config(
 112         test_layer.get_config())
 113 
 114     # If the serialization was successful, the new config should match the old.
 115     self.assertAllEqual(test_layer.get_config(), new_layer.get_config())
