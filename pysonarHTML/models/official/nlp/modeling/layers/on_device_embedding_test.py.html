  28 # This decorator runs the test in V1, V2-Eager, and V2-Functional mode. It
  29 # guarantees forward compatibility of this code for the V2 switchover.
  30 @keras_parameterized.run_all_keras_modes
  31 class OnDeviceEmbeddingTest(keras_parameterized.TestCase):
  32 
  33   def test_layer_creation(self):
  34     vocab_size = 31
  35     embedding_width = 27
  36     test_layer = on_device_embedding.OnDeviceEmbedding(
  37         vocab_size=vocab_size, embedding_width=embedding_width)
  38     # Create a 2-dimensional input (the first dimension is implicit).
  39     sequence_length = 23
  40     input_tensor = tf.keras.Input(shape=(sequence_length), dtype=tf.int32)
  41     output_tensor = test_layer(input_tensor)
  42 
  43     # The output should be the same as the input, save that it has an extra
  44     # embedding_width dimension on the end.
  45     expected_output_shape = [None, sequence_length, embedding_width]
  46     self.assertEqual(expected_output_shape, output_tensor.shape.as_list())
  47     self.assertEqual(output_tensor.dtype, tf.float32)
  48

  66   def test_layer_invocation(self):
  67     vocab_size = 31
  68     embedding_width = 27
  69     test_layer = on_device_embedding.OnDeviceEmbedding(
  70         vocab_size=vocab_size, embedding_width=embedding_width)
  71     # Create a 2-dimensional input (the first dimension is implicit).
  72     sequence_length = 23
  73     input_tensor = tf.keras.Input(shape=(sequence_length), dtype=tf.int32)
  74     output_tensor = test_layer(input_tensor)
  75 
  76     # Create a model from the test layer.
  77     model = tf.keras.Model(input_tensor, output_tensor)
  78 
  79     # Invoke the model on test data. We can't validate the output data itself
  80     # (the NN is too complex) but this will rule out structural runtime errors.
  81     batch_size = 3
  82     input_data = np.random.randint(
  83         vocab_size, size=(batch_size, sequence_length))
  84     output = model.predict(input_data)
  85     self.assertEqual(tf.float32, output.dtype)
