  16 """Tests for cls_head."""
  17 
  18 import tensorflow as tf
  19 
  20 from official.nlp.modeling.layers import cls_head
  21 
  22 
  23 class ClassificationHead(tf.test.TestCase):
  24 
  25   def test_layer_invocation(self):
  26     test_layer = cls_head.ClassificationHead(inner_dim=5, num_classes=2)
  27     features = tf.zeros(shape=(2, 10, 10), dtype=tf.float32)
  28     output = test_layer(features)
  29     self.assertAllClose(output, [[0., 0.], [0., 0.]])
  30     self.assertSameElements(test_layer.checkpoint_items.keys(),
  31                             ["pooler_dense"])
  32 
  33   def test_layer_serialization(self):
  34     layer = cls_head.ClassificationHead(10, 2)
  35     new_layer = cls_head.ClassificationHead.from_config(layer.get_config())
  36 
  37     # If the serialization was successful, the new config should match the old.
  38     self.assertAllEqual(layer.get_config(), new_layer.get_config())
  39 
  40 
  41 if __name__ == "__main__":
  42   tf.test.main()
