  27 @tf.keras.utils.register_keras_serializable(package='Text')
  28 class MaskedLM(tf.keras.layers.Layer):
  29   """Masked language model network head for BERT modeling.
  30 
  31   This network implements a masked language model based on the provided network.
  32   It assumes that the network being passed has a "get_embedding_table()" method.
  33 
  34   Arguments:
  35     embedding_table: The embedding table of the targets.
  36     activation: The activation, if any, for the dense layer.
  37     initializer: The intializer for the dense layer. Defaults to a Glorot
  38       uniform initializer.
  39     output: The output style for this network. Can be either 'logits' or
  40       'predictions'.
  41   """
  42 
  43   def __init__(self,
  44                embedding_table,
  45                activation=None,
  46                initializer='glorot_uniform',
  47                output='logits',
  48                name='cls/predictions',
  49                **kwargs):
  50     super(MaskedLM, self).__init__(name=name, **kwargs)
  51     self.embedding_table = embedding_table
  52     self.activation = activation
  53     self.initializer = tf.keras.initializers.get(initializer)
  54 
  55     if output not in ('predictions', 'logits'):
  56       raise ValueError(
  57           ('Unknown `output` value "%s". `output` can be either "logits" or '
  58            '"predictions"') % output)
  59     self._output_type = output
  60 
  61   def build(self, input_shape):
  62     self._vocab_size, hidden_size = self.embedding_table.shape
  63     self.dense = tf.keras.layers.Dense(
  64         hidden_size,
  65         activation=self.activation,
  66         kernel_initializer=self.initializer,
  67         name='transform/dense')
  68     self.layer_norm = tf.keras.layers.LayerNormalization(
  69         axis=-1, epsilon=1e-12, name='transform/LayerNorm')
  70     self.bias = self.add_weight(
  71         'output_bias/bias',
  72         shape=(self._vocab_size,),
  73         initializer='zeros',
  74         trainable=True)
  75 
  76     super(MaskedLM, self).build(input_shape)
  77 
  78   def call(self, sequence_data, masked_positions):
  79     masked_lm_input = self._gather_indexes(sequence_data, masked_positions)
  80     lm_data = self.dense(masked_lm_input)
  81     lm_data = self.layer_norm(lm_data)
  82     lm_data = tf.matmul(lm_data, self.embedding_table, transpose_b=True)
  83     logits = tf.nn.bias_add(lm_data, self.bias)
  84 
  85     masked_positions_shape = tf_utils.get_shape_list(
  86         masked_positions, name='masked_positions_tensor')
  87     logits = tf.reshape(logits,
  88                         [-1, masked_positions_shape[1], self._vocab_size])
  89     if self._output_type == 'logits':
  90       return logits
  91     return tf.nn.log_softmax(logits)
  92 
  93   def get_config(self):
  94     raise NotImplementedError('MaskedLM cannot be directly serialized because '
  95                               'it has variable sharing logic.')
  96 
  97   def _gather_indexes(self, sequence_tensor, positions):
  98     """Gathers the vectors at the specific positions.
  99 
 100     Args:
 101         sequence_tensor: Sequence output of `BertModel` layer of shape
 102           (`batch_size`, `seq_length`, num_hidden) where num_hidden is number of
 103           hidden units of `BertModel` layer.
 104         positions: Positions ids of tokens in sequence to mask for pretraining
 105           of with dimension (batch_size, num_predictions) where
 106           `num_predictions` is maximum number of tokens to mask out and predict
 107           per each sequence.
 108 
 109     Returns:
 110         Masked out sequence tensor of shape (batch_size * num_predictions,
 111         num_hidden).
 112     """
 113     sequence_shape = tf_utils.get_shape_list(
 114         sequence_tensor, name='sequence_output_tensor')
 115     batch_size, seq_length, width = sequence_shape
 116 
 117     flat_offsets = tf.reshape(
 118         tf.range(0, batch_size, dtype=tf.int32) * seq_length, [-1, 1])
 119     flat_positions = tf.reshape(positions + flat_offsets, [-1])
 120     flat_sequence_tensor = tf.reshape(sequence_tensor,
 121                                       [batch_size * seq_length, width])
 122     output_tensor = tf.gather(flat_sequence_tensor, flat_positions)
 123 
 124     return output_tensor
