  28 # This decorator runs the test in V1, V2-Eager, and V2-Functional mode. It
  29 # guarantees forward compatibility of this code for the V2 switchover.
  30 @keras_parameterized.run_all_keras_modes
  31 class PositionEmbeddingLayerTest(keras_parameterized.TestCase):

  33   def test_static_layer_output_shape(self):
  34     test_layer = position_embedding.PositionEmbedding()
  35     # Create a 3-dimensional input (the first dimension is implicit).
  36     sequence_length = 21
  37     width = 30
  38     input_tensor = tf.keras.Input(shape=(sequence_length, width))
  39     output_tensor = test_layer(input_tensor)
  40 
  41     # When using static positional embedding shapes, the output is expected
  42     # to be the same as the input shape in all dimensions save batch.
  43     expected_output_shape = [None, sequence_length, width]
  44     self.assertEqual(expected_output_shape, output_tensor.shape.as_list())
  45     # The default output dtype for this layer should be tf.float32.
  46     self.assertEqual(tf.float32, output_tensor.dtype)

  63   def test_dynamic_layer_output_shape(self):
  64     max_sequence_length = 40
  65     test_layer = position_embedding.PositionEmbedding(
  66         use_dynamic_slicing=True, max_sequence_length=max_sequence_length)
  67     # Create a 3-dimensional input (the first dimension is implicit).
  68     width = 30
  69     input_tensor = tf.keras.Input(shape=(None, width))
  70     output_tensor = test_layer(input_tensor)
  71 
  72     # When using dynamic positional embedding shapes, the output is expected
  73     # to be the same as the input shape in all dimensions - but may be None if
  74     # the input shape is None there.
  75     expected_output_shape = [None, None, width]
  76     self.assertEqual(expected_output_shape, output_tensor.shape.as_list())
  77 
  78   def test_dynamic_layer_slicing(self):
  79     max_sequence_length = 40
  80     test_layer = position_embedding.PositionEmbedding(
  81         use_dynamic_slicing=True, max_sequence_length=max_sequence_length)
  82     # Create a 3-dimensional input (the first dimension is implicit).
  83     width = 30
  84     input_tensor = tf.keras.Input(shape=(None, width))
  85     output_tensor = test_layer(input_tensor)
  86 
  87     model = tf.keras.Model(input_tensor, output_tensor)
  88 
  89     # Create input data that is shorter than max_sequence_length, which should
  90     # trigger a down-slice.
  91     input_length = 17
  92     # Note: This test explicitly uses a batch size of 1. This is to get around
  93     # Keras' restriction on Model invocations: inputs are expected to have the
  94     # same batch cardinality as outputs. In practice, this layer should be used
  95     # inside a model, where it can be projected when added to another tensor.
  96     input_data = np.ones((1, input_length, width))
  97     output_data = model.predict(input_data)
  98 
  99     self.assertAllEqual([1, input_length, width], output_data.shape)

 101   def test_relative_tensor_input(self):
 102     hidden_size = 8
 103     test_layer = position_embedding.RelativePositionEmbedding(
 104         hidden_size=hidden_size)
 105 
 106     # create a 3-dimensional input for test_layer to infer length as 1.
 107     input_tensor = tf.constant([[[0] * hidden_size]])
 108     output_tensor = test_layer(input_tensor)
 109 
 110     # expected output is the theoretical result of the input based on
 111     # sine cosine relative position embedding formula.
 112     expected_output_tensor = tf.constant([[0, 0, 0, 0, 1, 1, 1, 1]])
 113     self.assertAllEqual(output_tensor, expected_output_tensor)
 114 
 115   def test_relative_length_input(self):
 116     hidden_size = 8
 117 
 118     # When we do not have tensor as input, we explicitly specify length
 119     # value when initializing test_layer.
 120     test_layer = position_embedding.RelativePositionEmbedding(
 121         hidden_size=hidden_size)
 122     input_tensor = None
 123     output_tensor = test_layer(input_tensor, length=1)
 124 
 125     # expected output is the theoretical result of the input based on
 126     # sine cosine relative position embedding formula.
 127     expected_output_tensor = tf.constant([[0, 0, 0, 0, 1, 1, 1, 1]])
 128     self.assertAllEqual(output_tensor, expected_output_tensor)
