  16 """A Classification head layer which is common used with sequence encoders."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 # from __future__ import google_type_annotations
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 
  24 from official.modeling import tf_utils
  25 
  26 
  27 class ClassificationHead(tf.keras.layers.Layer):
  28   """Pooling head for sentence-level classification tasks."""
  29 
  30   def __init__(self,
  31                inner_dim,
  32                num_classes,
  33                cls_token_idx=0,
  34                activation="tanh",
  35                dropout_rate=0.0,
  36                initializer="glorot_uniform",
  37                **kwargs):
  38     """Initializes the `ClassificationHead`.
  39 
  40     Args:
  41       inner_dim: The dimensionality of inner projection layer.
  42       num_classes: Number of output classes.
  43       cls_token_idx: The index inside the sequence to pool.
  44       activation: Dense layer activation.
  45       dropout_rate: Dropout probability.
  46       initializer: Initializer for dense layer kernels.
  47       **kwargs: Keyword arguments.
  48     """
  49     super(ClassificationHead, self).__init__(**kwargs)
  50     self.dropout_rate = dropout_rate
  51     self.inner_dim = inner_dim
  52     self.num_classes = num_classes
  53     self.activation = tf_utils.get_activation(activation)
  54     self.initializer = tf.keras.initializers.get(initializer)
  55     self.cls_token_idx = cls_token_idx
  56 
  57     self.dense = tf.keras.layers.Dense(
  58         units=inner_dim,
  59         activation=self.activation,
  60         kernel_initializer=self.initializer,
  61         name="pooler_dense")
  62     self.dropout = tf.keras.layers.Dropout(rate=self.dropout_rate)
  63     self.out_proj = tf.keras.layers.Dense(
  64         units=num_classes, kernel_initializer=self.initializer, name="logits")
  65 
  66   def call(self, features):
  67     x = features[:, self.cls_token_idx, :]  # take <CLS> token.
  68     x = self.dense(x)
  69     x = self.dropout(x)
  70     x = self.out_proj(x)
  71     return x
  72 
  73   def get_config(self):
  74     config = {
  75         "dropout_rate": self.dropout_rate,
  76         "num_classes": self.num_classes,
  77         "inner_dim": self.inner_dim,
  78         "activation": tf.keras.activations.serialize(self.activation),
  79         "initializer": tf.keras.initializers.serialize(self.initializer),
  80     }
  81     config.update(super(ClassificationHead, self).get_config())
  82     return config
  83 
  84   @classmethod
  85   def from_config(cls, config, custom_objects=None):
  86     return cls(**config)
  87 
  88   @property
  89   def checkpoint_items(self):
  90     return {self.dense.name: self.dense}
