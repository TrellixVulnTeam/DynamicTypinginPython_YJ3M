  28 # This decorator runs the test in V1, V2-Eager, and V2-Functional mode. It
  29 # guarantees forward compatibility of this code for the V2 switchover.
  30 @keras_parameterized.run_all_keras_modes

  33   def test_3D_einsum_with_two_bound_dimensions(self):
  34     test_layer = dense_einsum.DenseEinsum(
  35         output_shape=(64,), num_summed_dimensions=2)
  36     # Create a 4-dimensional input (the first dimension is implicit).
  37     input_tensor = tf.keras.Input(shape=(None, 40, 80))
  38     _ = test_layer(input_tensor)
  39     self.assertEqual(test_layer._einsum_string, "abcd,cde->abe")
  40     self.assertEqual(test_layer._kernel_shape, (40, 80, 64))
  41

  60   def test_bias_term_can_be_disabled(self):
  61     # A layer created using the bias should have two weights.
  62     test_layer = dense_einsum.DenseEinsum(
  63         output_shape=64, num_summed_dimensions=1, use_bias=True)
  64     input_tensor = tf.keras.Input(shape=(None, 80))
  65     _ = test_layer(input_tensor)
  66     self.assertEqual(2, len(test_layer.get_weights()))
  67 
  68     # A layer created without the bias should have only one weight.
  69     test_layer = dense_einsum.DenseEinsum(
  70         output_shape=64, num_summed_dimensions=1, use_bias=False)
  71     input_tensor = tf.keras.Input(shape=(None, 80))
  72     _ = test_layer(input_tensor)
  73     self.assertEqual(1, len(test_layer.get_weights()))
  74 
  75   def test_activation(self):
  76     # Create a model that does not use an activation.
  77     no_activation_layer = dense_einsum.DenseEinsum(
  78         output_shape=64, num_summed_dimensions=1, activation=None)
  79     input_tensor = tf.keras.Input(shape=(None, 80))
  80     output_tensor = no_activation_layer(input_tensor)
  81     no_activation_model = tf.keras.Model(input_tensor, output_tensor)
  82 
  83     # Create a model that uses a softmax activation.
  84     activation_layer = dense_einsum.DenseEinsum(
  85         output_shape=64, num_summed_dimensions=1, activation="softmax")
  86     input_tensor = tf.keras.Input(shape=(None, 80))
  87     output_tensor = activation_layer(input_tensor)
  88     activation_model = tf.keras.Model(input_tensor, output_tensor)
  89 
  90     # Make sure the models' weights are identical.
  91     activation_model.set_weights(no_activation_model.get_weights())
  92 
  93     # Predict using each model on the same input data. The output should be
  94     # different, since one is using a softmax - even though the models' weights
  95     # are the same.
  96     input_values = 10 * np.random.random_sample((10, 4, 80))
  97     non_activated_data = no_activation_model.predict(input_values)
  98     activated_data = activation_model.predict(input_values)
  99     self.assertNotAllClose(activated_data, non_activated_data)
