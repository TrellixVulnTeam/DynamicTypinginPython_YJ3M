  15 """Keras-based one-hot embedding layer."""
  16 # pylint: disable=g-classes-have-attributes
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 # from __future__ import google_type_annotations
  20 from __future__ import print_function
  21 
  22 import tensorflow as tf
  23 
  24 
  25 @tf.keras.utils.register_keras_serializable(package="Text")
  26 class OnDeviceEmbedding(tf.keras.layers.Layer):
  27   """Performs an embedding lookup suitable for accelerator devices.
  28 
  29   This layer uses either tf.gather or tf.one_hot to translate integer indices to
  30   float embeddings.
  31 
  32   Arguments:
  33     vocab_size: Number of elements in the vocabulary.
  34     embedding_width: Output size of the embedding layer.
  35     initializer: The initializer to use for the embedding weights. Defaults to
  36       "glorot_uniform".
  37     use_one_hot: Whether to use tf.one_hot over tf.gather for the embedding
  38       lookup. Defaults to False (that is, using tf.gather). Setting this option
  39       to True may improve performance, especially on small vocabulary sizes, but
  40       will generally require more memory.
  41   """
  42 
  43   def __init__(self,
  44                vocab_size,
  45                embedding_width,
  46                initializer="glorot_uniform",
  47                use_one_hot=False,
  48                **kwargs):
  49 
  50     super(OnDeviceEmbedding, self).__init__(**kwargs)
  51     self._vocab_size = vocab_size
  52     self._embedding_width = embedding_width
  53     self._initializer = initializer
  54     self._use_one_hot = use_one_hot
  55 
  56   def get_config(self):
  57     config = {
  58         "vocab_size": self._vocab_size,
  59         "embedding_width": self._embedding_width,
  60         "initializer": self._initializer,
  61         "use_one_hot": self._use_one_hot,
  62     }
  63     base_config = super(OnDeviceEmbedding, self).get_config()
  64     return dict(list(base_config.items()) + list(config.items()))
  65 
  66   def build(self, input_shape):
  67     self.embeddings = self.add_weight(
  68         "embeddings",
  69         shape=[self._vocab_size, self._embedding_width],
  70         initializer=self._initializer,
  71         dtype=tf.float32)
  72 
  73     super(OnDeviceEmbedding, self).build(input_shape)
  74 
  75   def call(self, inputs):
  76     flat_inputs = tf.reshape(inputs, [-1])
  77     if self._use_one_hot:
  78       one_hot_data = tf.one_hot(
  79           flat_inputs, depth=self._vocab_size, dtype=self.embeddings.dtype)
  80       embeddings = tf.matmul(one_hot_data, self.embeddings)
  81     else:
  82       embeddings = tf.gather(self.embeddings, flat_inputs)
  83     embeddings = tf.reshape(
  84         embeddings,
  85         # Work around b/142213824: prefer concat to shape over a Python list.
  86         tf.concat([tf.shape(inputs), [self._embedding_width]], axis=0))
  87     embeddings.set_shape(inputs.shape.as_list() + [self._embedding_width])
  88     return embeddings
