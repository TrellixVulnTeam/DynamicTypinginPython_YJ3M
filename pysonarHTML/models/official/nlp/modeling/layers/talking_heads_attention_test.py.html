  29 # This decorator runs the test in V1, V2-Eager, and V2-Functional mode. It

  35   @parameterized.named_parameters(
  36       ("key_value_same_proj", None, None, [40, 80]),
  37       ("key_value_different_proj", 32, 60, [40, 60]),
  38   )
  39   def test_non_masked_attention(self, value_size, output_shape, output_dims):
  40     """Test that the attention layer can be created without a mask tensor."""
  41     test_layer = talking_heads_attention.TalkingHeadsAttention(
  42         num_heads=12,
  43         key_size=64,
  44         value_size=value_size,
  45         output_shape=output_shape)
  46     # Create a 3-dimensional input (the first dimension is implicit).
  47     query = tf.keras.Input(shape=(40, 80))
  48     value = tf.keras.Input(shape=(20, 80))
  49     output = test_layer([query, value])
  50     self.assertEqual(output.shape.as_list(), [None] + output_dims)

  62     """Test attention outputs with coefficients."""

  71   @parameterized.named_parameters(("with_bias", True), ("no_bias", False))
  72   def test_masked_attention(self, use_bias):
  73     """Test with a mask tensor."""
  74     test_layer = talking_heads_attention.TalkingHeadsAttention(
  75         num_heads=12, key_size=2, use_bias=use_bias)
  76     # Create a 3-dimensional input (the first dimension is implicit).
  77     batch_size = 3
  78     query = tf.keras.Input(shape=(4, 8))
  79     value = tf.keras.Input(shape=(2, 8))
  80     mask_tensor = tf.keras.Input(shape=(4, 2))
  81     output = test_layer([query, value], mask_tensor)
  82 
  83     # Create a model containing the test layer.
  84     model = tf.keras.Model([query, value, mask_tensor], output)
  85 
  86     # Generate data for the input (non-mask) tensors.
  87     from_data = 10 * np.random.random_sample((batch_size, 4, 8))
  88     to_data = 10 * np.random.random_sample((batch_size, 2, 8))
  89 
  90     # Invoke the data with a random set of mask data. This should mask at least
  91     # one element.
  92     mask_data = np.random.randint(2, size=(batch_size, 4, 2))
  93     masked_output_data = model.predict([from_data, to_data, mask_data])
  94 
  95     # Invoke the same data, but with a null mask (where no elements are masked).
  96     null_mask_data = np.ones((batch_size, 4, 2))
  97     unmasked_output_data = model.predict([from_data, to_data, null_mask_data])
  98 
  99     # Because one data is masked and one is not, the outputs should not be the
 100     # same.
 101     self.assertNotAllClose(masked_output_data, unmasked_output_data)
 102 
 103     # Tests the layer with three inputs: Q, K, V.
 104     key = tf.keras.Input(shape=(2, 8))
 105     output = test_layer([query, value, key], mask_tensor)
 106     model = tf.keras.Model([query, value, key, mask_tensor], output)
 107 
 108     masked_output_data = model.predict([from_data, to_data, to_data, mask_data])
 109     unmasked_output_data = model.predict(
 110         [from_data, to_data, to_data, null_mask_data])
 111     # Because one data is masked and one is not, the outputs should not be the
 112     # same.
 113     self.assertNotAllClose(masked_output_data, unmasked_output_data)
 114 
 115     if use_bias:
 116       self.assertLen(test_layer._query_dense.trainable_variables, 2)
 117       self.assertLen(test_layer._output_dense.trainable_variables, 2)
 118     else:
 119       self.assertLen(test_layer._query_dense.trainable_variables, 1)
 120       self.assertLen(test_layer._output_dense.trainable_variables, 1)

 133   @parameterized.named_parameters(
 134       ("4d_inputs_one_free_batch", [3, 4], [3, 2], [4, 2], (2,)),
 135       ("4D_inputs_2D_attention", [3, 4], [3, 2], [3, 4, 3, 2], (1, 2)),
 136       ("5D_inputs_2D_attention", [5, 3, 4], [5, 3, 2], [3, 4, 3, 2], (2, 3)))
 137   def test_high_dim_attention(self, q_dims, v_dims, mask_dims, attention_axes):
 138     """Test with a mask tensor."""
 139     test_layer = talking_heads_attention.TalkingHeadsAttention(
 140         num_heads=12, key_size=2, attention_axes=attention_axes)
 141     batch_size, hidden_size = 3, 8
 142     # Generate data for the input (non-mask) tensors.
 143     query_shape = [batch_size] + q_dims + [hidden_size]
 144     value_shape = [batch_size] + v_dims + [hidden_size]
 145     mask_shape = [batch_size] + mask_dims
 146     query = 10 * np.random.random_sample(query_shape)
 147     value = 10 * np.random.random_sample(value_shape)
 148 
 149     # Invoke the data with a random set of mask data. This should mask at least
 150     # one element.
 151     mask_data = np.random.randint(2, size=mask_shape).astype("bool")
 152     output = test_layer([query, value], mask_data)
 153 
 154     # Invoke the same data, but with a null mask (where no elements are masked).
 155     null_mask_data = np.ones(mask_shape)
 156     unmasked_output = test_layer([query, value], null_mask_data)
 157     # Because one data is masked and one is not, the outputs should not be the
 158     # same.
 159     self.assertNotAllClose(output, unmasked_output)
