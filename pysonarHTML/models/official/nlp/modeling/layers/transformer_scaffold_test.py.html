  31 # Test class that wraps a standard attention layer. If this layer is called
  32 # at any point, the list passed to the config object will be filled with a
  33 # boolean 'True'. We register this class as a Keras serializable so we can
  34 # test serialization below.
  35 @tf.keras.utils.register_keras_serializable(package='TestOnlyAttention')
  36 class ValidatedAttentionLayer(attention.MultiHeadAttention):
  37 
  38   def __init__(self, call_list, **kwargs):
  39     super(ValidatedAttentionLayer, self).__init__(**kwargs)
  40     self.list = call_list
  41 
  42   def call(self, inputs, attention_mask=None):
  43     self.list.append(True)
  44     return super(ValidatedAttentionLayer, self).call(
  45         inputs, attention_mask=attention_mask)
  46 
  47   def get_config(self):
  48     config = super(ValidatedAttentionLayer, self).get_config()
  49     config['call_list'] = []
  50     return config

  53 # Test class implements a simple feedforward layer. If this layer is called
  54 # at any point, the list passed to the config object will be filled with a
  55 # boolean 'True'. We register this class as a Keras serializable so we can
  56 # test serialization below.
  57 @tf.keras.utils.register_keras_serializable(package='TestOnlyFeedforward')
  58 class ValidatedFeedforwardLayer(tf.keras.layers.Layer):
  59 
  60   def __init__(self, call_list, activation, **kwargs):
  61     super(ValidatedFeedforwardLayer, self).__init__(**kwargs)
  62     self.list = call_list
  63     self.activation = activation
  64 
  65   def build(self, input_shape):
  66     hidden_size = input_shape.as_list()[-1]
  67     self._feedforward_dense = tf.keras.layers.experimental.EinsumDense(
  68         '...x,xy->...y',
  69         output_shape=hidden_size,
  70         bias_axes='y',
  71         activation=self.activation,
  72         name='feedforward')
  73 
  74   def call(self, inputs):
  75     self.list.append(True)
  76     return self._feedforward_dense(inputs)
  77 
  78   def get_config(self):
  79     config = super(ValidatedFeedforwardLayer, self).get_config()
  80     config['call_list'] = []
  81     config['activation'] = self.activation
  82     return config

 410   def test_layer_restoration_from_config(self):
 411     sequence_length = 21
 412     width = 80
 413 
 414     call_list = []
 415     attention_layer_cfg = {
 416         'num_heads': 10,
 417         'key_size': 8,
 418         'call_list': call_list,
 419         'name': 'test_layer',
 420     }
 421     test_layer = transformer_scaffold.TransformerScaffold(
 422         attention_cls=ValidatedAttentionLayer,
 423         attention_cfg=attention_layer_cfg,
 424         num_attention_heads=10,
 425         intermediate_size=2048,
 426         intermediate_activation='relu')
 427 
 428     # Create a 3-dimensional input (the first dimension is implicit).
 429     data_tensor = tf.keras.Input(shape=(sequence_length, width))
 430     # Create a 2-dimensional input (the first dimension is implicit).
 431     mask_tensor = tf.keras.Input(shape=(sequence_length, sequence_length))
 432     output_tensor = test_layer([data_tensor, mask_tensor])
 433 
 434     # Create a model from the test layer.
 435     model = tf.keras.Model([data_tensor, mask_tensor], output_tensor)
 436 
 437     # Invoke the model on test data. We can't validate the output data itself
 438     # (the NN is too complex) but this will rule out structural runtime errors.
 439     batch_size = 6
 440     input_data = 10 * np.random.random_sample(
 441         (batch_size, sequence_length, width))
 442     # The attention mask should be of shape (batch, from_seq_len, to_seq_len),
 443     # which here is (batch, sequence_length, sequence_length)
 444     mask_data = np.random.randint(
 445         2, size=(batch_size, sequence_length, sequence_length))
 446     pre_serialization_output = model.predict([input_data, mask_data])
 447 
 448     # Serialize the model config. Pass the serialized data through json to
 449     # ensure that we can serialize this layer to disk.
 450     serialized_data = json.dumps(model.get_config())
 451     post_string_serialized_data = json.loads(serialized_data)
 452 
 453     # Create a new model from the old config, and copy the weights. These models
 454     # should have identical outputs.
 455     new_model = tf.keras.Model.from_config(post_string_serialized_data)
 456     new_model.set_weights(model.get_weights())
 457     output = new_model.predict([input_data, mask_data])
 458 
 459     self.assertAllClose(pre_serialization_output, output)
 460 
 461     # If the layer was configured correctly, it should have a list attribute
 462     # (since it should have the custom class and config passed to it).
 463     new_model.summary()
 464     new_call_list = new_model.get_layer(
 465         name='transformer_scaffold')._attention_layer.list
 466     self.assertNotEmpty(new_call_list)
 467     self.assertTrue(new_call_list[0],
 468                     "The passed layer class wasn't instantiated.")
