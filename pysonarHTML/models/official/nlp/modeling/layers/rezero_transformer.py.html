  29 @tf.keras.utils.register_keras_serializable(package="Text")

  31 class ReZeroTransformer(tf.keras.layers.Layer):
  32   """Transformer layer with ReZero.
  33 
  34   This layer implements the Transformer from "Attention Is All You Need".
  35   (https://arxiv.org/abs/1706.03762).
  36   The residual connection implements the ReZero method.
  37   (https://arxiv.org/abs/2003.04887)
  38 
  39   Arguments:
  40     num_attention_heads: Number of attention heads.
  41     intermediate_size: Size of the intermediate layer.
  42     intermediate_activation: Activation for the intermediate layer.
  43     dropout_rate: Dropout probability for the post-attention and output dropout.
  44     attention_dropout_rate: Dropout probability for within the attention layer.
  45     output_range: the sequence output range, [0, output_range) by slicing the
  46       target sequence. `None` means the target sequence is not sliced.
  47     kernel_initializer: Initializer for dense layer kernels.
  48     bias_initializer: Initializer for dense layer biases.
  49     kernel_regularizer: Regularizer for dense layer kernels.
  50     bias_regularizer: Regularizer for dense layer biases.
  51     activity_regularizer: Regularizer for dense layer activity.
  52     kernel_constraint: Constraint for dense layer kernels.
  53     bias_constraint: Constraint for dense layer kernels.
  54     use_layer_norm: If add layer_norm on top of the ReZero.
  55   """
  56 
  57   def __init__(self,
  58                num_attention_heads,
  59                intermediate_size,
  60                intermediate_activation,
  61                dropout_rate=0.0,
  62                attention_dropout_rate=0.0,
  63                output_range=None,
  64                kernel_initializer="glorot_uniform",
  65                bias_initializer="zeros",
  66                kernel_regularizer=None,
  67                bias_regularizer=None,
  68                activity_regularizer=None,
  69                kernel_constraint=None,
  70                bias_constraint=None,
  71                use_layer_norm=False,
  72                **kwargs):
  73     super(ReZeroTransformer, self).__init__(**kwargs)
  74 
  75     self._num_heads = num_attention_heads
  76     self._intermediate_size = intermediate_size
  77     self._intermediate_activation = intermediate_activation
  78     self._attention_dropout_rate = attention_dropout_rate
  79     self._dropout_rate = dropout_rate
  80     self._output_range = output_range
  81     self._kernel_initializer = tf.keras.initializers.get(kernel_initializer)
  82     self._bias_initializer = tf.keras.initializers.get(bias_initializer)
  83     self._kernel_regularizer = tf.keras.regularizers.get(kernel_regularizer)
  84     self._bias_regularizer = tf.keras.regularizers.get(bias_regularizer)
  85     self._kernel_constraint = tf.keras.constraints.get(kernel_constraint)
  86     self._bias_constraint = tf.keras.constraints.get(bias_constraint)
  87     self._use_layer_norm = use_layer_norm
  88 
  89   def build(self, input_shape):
  90     input_tensor = input_shape[0] if len(input_shape) == 2 else input_shape
  91     input_tensor_shape = tf.TensorShape(input_tensor)
  92     if len(input_tensor_shape) != 3:
  93       raise ValueError("TransformerLayer expects a three-dimensional input of "
  94                        "shape [batch, sequence, width].")
  95     batch_size, sequence_length, hidden_size = input_tensor_shape
  96 
  97     if len(input_shape) == 2:
  98       mask_tensor_shape = tf.TensorShape(input_shape[1])
  99       expected_mask_tensor_shape = tf.TensorShape(
 100           [batch_size, sequence_length, sequence_length])
 101       if not expected_mask_tensor_shape.is_compatible_with(mask_tensor_shape):
 102         raise ValueError("When passing a mask tensor to TransformerLayer, the "
 103                          "mask tensor must be of shape [batch, "
 104                          "sequence_length, sequence_length] (here %s). Got a "
 105                          "mask tensor of shape %s." %
 106                          (expected_mask_tensor_shape, mask_tensor_shape))
 107     if hidden_size % self._num_heads != 0:
 108       raise ValueError(
 109           "The input size (%d) is not a multiple of the number of attention "
 110           "heads (%d)" % (hidden_size, self._num_heads))
 111     self._attention_head_size = int(hidden_size // self._num_heads)
 112 
 113     self._attention_layer = attention.MultiHeadAttention(
 114         num_heads=self._num_heads,
 115         key_size=self._attention_head_size,
 116         dropout=self._attention_dropout_rate,
 117         kernel_initializer=self._kernel_initializer,
 118         bias_initializer=self._bias_initializer,
 119         kernel_regularizer=self._kernel_regularizer,
 120         bias_regularizer=self._bias_regularizer,
 121         activity_regularizer=self._activity_regularizer,
 122         kernel_constraint=self._kernel_constraint,
 123         bias_constraint=self._bias_constraint,
 124         name="self_attention")
 125     self._attention_dropout = tf.keras.layers.Dropout(rate=self._dropout_rate)
 126     if self._use_layer_norm:
 127       # Use float32 in layernorm for numeric stability.
 128       # It is probably safe in mixed_float16, but we haven't validated this yet.
 129       self._attention_layer_norm = (
 130           tf.keras.layers.LayerNormalization(
 131               name="self_attention_layer_norm",
 132               axis=-1,
 133               epsilon=1e-12,
 134               dtype=tf.float32))
 135     self._intermediate_dense = dense_einsum.DenseEinsum(
 136         output_shape=self._intermediate_size,
 137         activation=None,
 138         kernel_initializer=self._kernel_initializer,
 139         bias_initializer=self._bias_initializer,
 140         kernel_regularizer=self._kernel_regularizer,
 141         bias_regularizer=self._bias_regularizer,
 142         activity_regularizer=self._activity_regularizer,
 143         kernel_constraint=self._kernel_constraint,
 144         bias_constraint=self._bias_constraint,
 145         name="intermediate")
 146     policy = tf.keras.mixed_precision.experimental.global_policy()
 147     if policy.name == "mixed_bfloat16":
 148       # bfloat16 causes BERT with the LAMB optimizer to not converge
 149       # as well, so we use float32.
 150       # TODO(b/154538392): Investigate this.
 151       policy = tf.float32
 152     self._intermediate_activation_layer = tf.keras.layers.Activation(
 153         self._intermediate_activation, dtype=policy)
 154     self._output_dense = dense_einsum.DenseEinsum(
 155         output_shape=hidden_size,
 156         kernel_initializer=self._kernel_initializer,
 157         bias_initializer=self._bias_initializer,
 158         kernel_regularizer=self._kernel_regularizer,
 159         bias_regularizer=self._bias_regularizer,
 160         activity_regularizer=self._activity_regularizer,
 161         kernel_constraint=self._kernel_constraint,
 162         bias_constraint=self._bias_constraint,
 163         name="output")
 164     self._output_dropout = tf.keras.layers.Dropout(rate=self._dropout_rate)
 165     if self._use_layer_norm:
 166       # Use float32 in layernorm for numeric stability.
 167       self._output_layer_norm = tf.keras.layers.LayerNormalization(
 168           name="output_layer_norm", axis=-1, epsilon=1e-12, dtype=tf.float32)
 169 
 170     self._rezero_a = self.add_weight(
 171         name="rezero_alpha",
 172         initializer=tf.keras.initializers.Zeros(),
 173         trainable=True, dtype=tf.float32)
 174 
 175     super(ReZeroTransformer, self).build(input_shape)
 176 
 177   def get_config(self):
 178     config = {
 179         "num_attention_heads":
 180             self._num_heads,
 181         "intermediate_size":
 182             self._intermediate_size,
 183         "intermediate_activation":
 184             self._intermediate_activation,
 185         "dropout_rate":
 186             self._dropout_rate,
 187         "attention_dropout_rate":
 188             self._attention_dropout_rate,
 189         "output_range":
 190             self._output_range,
 191         "use_layer_norm":
 192             self._use_layer_norm,
 193         "kernel_initializer":
 194             tf.keras.initializers.serialize(self._kernel_initializer),
 195         "bias_initializer":
 196             tf.keras.initializers.serialize(self._bias_initializer),
 197         "kernel_regularizer":
 198             tf.keras.regularizers.serialize(self._kernel_regularizer),
 199         "bias_regularizer":
 200             tf.keras.regularizers.serialize(self._bias_regularizer),
 201         "activity_regularizer":
 202             tf.keras.regularizers.serialize(self._activity_regularizer),
 203         "kernel_constraint":
 204             tf.keras.constraints.serialize(self._kernel_constraint),
 205         "bias_constraint":
 206             tf.keras.constraints.serialize(self._bias_constraint),
 207     }
 208     base_config = super(ReZeroTransformer, self).get_config()
 209     return dict(list(base_config.items()) + list(config.items()))
 210 
 211   def reset_rezero(self):
 212     self._rezero_a.assign(0.)
 213 
 214   def call(self, inputs):
 215     if isinstance(inputs, (list, tuple)) and len(inputs) == 2:
 216       input_tensor, attention_mask = inputs
 217     else:
 218       input_tensor, attention_mask = (inputs, None)
 219 
 220     if self._output_range:
 221       target_tensor = input_tensor[:, 0:self._output_range, :]
 222       attention_mask = attention_mask[:, 0:self._output_range, :]
 223     else:
 224       target_tensor = input_tensor
 225     attention_inputs = [target_tensor, input_tensor]
 226 
 227     attention_output = self._attention_layer(attention_inputs, attention_mask)
 228     attention_output = self._attention_dropout(attention_output)
 229     attention_output = target_tensor + self._rezero_a * attention_output
 230     if self._use_layer_norm:
 231       attention_output = self._attention_layer_norm(attention_output)
 232     else:
 233       attention_output = tf.cast(attention_output, tf.float32)
 234 
 235     intermediate_output = self._intermediate_dense(attention_output)
 236     intermediate_output = self._intermediate_activation_layer(
 237         intermediate_output)
 238     layer_output = self._output_dense(intermediate_output)
 239     layer_output = self._output_dropout(layer_output)
 240     # During mixed precision training, attention_output is from layer norm and
 241     # is always fp32 for now. Cast layer_output to fp32 for the subsequent add.
 242     layer_output = attention_output + tf.cast(self._rezero_a * layer_output,
 243                                               tf.float32)
 244     if self._use_layer_norm:
 245       layer_output = self._output_layer_norm(layer_output)
 246 
 247     return layer_output
