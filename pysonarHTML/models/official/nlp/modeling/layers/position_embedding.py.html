  29 @tf.keras.utils.register_keras_serializable(package="Text")
  30 class PositionEmbedding(tf.keras.layers.Layer):
  31   """Creates a positional embedding.
  32 
  33   This layer creates a positional embedding as described in "BERT: Pre-training
  34   of Deep Bidirectional Transformers for Language Understanding"
  35   (https://arxiv.org/abs/1810.04805).
  36 
  37   This layer can be set up to either create a statically shaped slice or a
  38   dynamically shaped slice. If `use_dynamic_slicing` is True, the input tensor
  39   can have a dynamic 1st dimension, while if `use_dynamic_slicing` is False the
  40   input size must be fixed.
  41 
  42   Arguments:
  43     use_dynamic_slicing: Whether to use the dynamic slicing path.
  44     max_sequence_length: The maximum size of the dynamic sequence. Only
  45       applicable if `use_dynamic_slicing` is True.
  46     initializer: The initializer to use for the embedding weights. Defaults to
  47       "glorot_uniform".
  48   """
  49 
  50   def __init__(self,
  51                initializer="glorot_uniform",
  52                use_dynamic_slicing=False,
  53                max_sequence_length=None,
  54                **kwargs):
  55     # We need to have a default dtype of float32, since the inputs (which Keras
  56     # usually uses to infer the dtype) will always be int32.
  57     if "dtype" not in kwargs:
  58       kwargs["dtype"] = "float32"
  59 
  60     super(PositionEmbedding, self).__init__(**kwargs)
  61     if use_dynamic_slicing and max_sequence_length is None:
  62       raise ValueError(
  63           "If `use_dynamic_slicing` is True, `max_sequence_length` must be set."
  64       )
  65     self._max_sequence_length = max_sequence_length
  66     self._initializer = tf.keras.initializers.get(initializer)
  67     self._use_dynamic_slicing = use_dynamic_slicing
  68 
  69   def get_config(self):
  70     config = {
  71         "max_sequence_length": self._max_sequence_length,
  72         "initializer": tf.keras.initializers.serialize(self._initializer),
  73         "use_dynamic_slicing": self._use_dynamic_slicing,
  74     }
  75     base_config = super(PositionEmbedding, self).get_config()
  76     return dict(list(base_config.items()) + list(config.items()))
  77 
  78   def build(self, input_shape):
  79     """Implements build() for the layer."""
  80     dimension_list = input_shape.as_list()
  81 
  82     if len(dimension_list) != 3:
  83       raise ValueError("PositionEmbedding expects a 3-dimensional input tensor "
  84                        "of shape [batch, sequence, width]")
  85     seq_length = dimension_list[1]
  86     width = dimension_list[2]
  87 
  88     # If we are not using dynamic slicing, we must assume that the sequence
  89     # length is fixed and max_sequence_length should not be specified.
  90     if not self._use_dynamic_slicing:
  91       if seq_length is None:
  92         raise ValueError(
  93             "PositionEmbedding must have `use_dynamic_slicing` set "
  94             "to True (and max_sequence_length set) when the "
  95             "sequence (1st) dimension of the input is None.")
  96       if self._max_sequence_length is not None:
  97         raise ValueError(
  98             "When `use_dynamic_slicing` is False, max_sequence_length should "
  99             "not be specified and we ought to use seq_length to get the "
 100             "variable shape.")
 101 
 102     if self._max_sequence_length is not None:
 103       weight_sequence_length = self._max_sequence_length
 104     else:
 105       weight_sequence_length = seq_length
 106 
 107     self._position_embeddings = self.add_weight(
 108         "embeddings",
 109         shape=[weight_sequence_length, width],
 110         initializer=self._initializer)
 111 
 112     super(PositionEmbedding, self).build(input_shape)
 113 
 114   def call(self, inputs):
 115     """Implements call() for the layer."""
 116     input_shape = tf_utils.get_shape_list(inputs, expected_rank=3)
 117     if self._use_dynamic_slicing:
 118       position_embeddings = self._position_embeddings[:input_shape[1], :]
 119     else:
 120       position_embeddings = self._position_embeddings
 121 
 122     return tf.broadcast_to(position_embeddings, input_shape)

 125 @tf.keras.utils.register_keras_serializable(package="Text")
 126 class RelativePositionEmbedding(tf.keras.layers.Layer):
 127   """Creates a positional embedding.
 128 
 129   This layer calculates the position encoding as a mix of sine and cosine
 130   functions with geometrically increasing wavelengths. Defined and formulized in
 131    "Attention is All You Need", section 3.5.
 132   (https://arxiv.org/abs/1706.03762).
 133 
 134   Arguments:
 135     hidden_size: Size of the hidden layer.
 136     min_timescale: Minimum scale that will be applied at each position
 137     max_timescale: Maximum scale that will be applied at each position.
 138   """
 139 
 140   def __init__(self,
 141                hidden_size,
 142                min_timescale=1.0,
 143                max_timescale=1.0e4,
 144                **kwargs):
 145     # We need to have a default dtype of float32, since the inputs (which Keras
 146     # usually uses to infer the dtype) will always be int32.
 147     # We compute the positional encoding in float32 even if the model uses
 148     # float16, as many of the ops used, like log and exp, are numerically
 149     # unstable in float16.
 150     if "dtype" not in kwargs:
 151       kwargs["dtype"] = "float32"
 152 
 153     super(RelativePositionEmbedding, self).__init__(**kwargs)
 154     self._hidden_size = hidden_size
 155     self._min_timescale = min_timescale
 156     self._max_timescale = max_timescale
 157 
 158   def get_config(self):
 159     config = {
 160         "hidden_size": self._hidden_size,
 161         "min_timescale": self._min_timescale,
 162         "max_timescale": self._max_timescale,
 163         "length": self._length,
 164     }
 165     base_config = super(RelativePositionEmbedding, self).get_config()
 166     return dict(list(base_config.items()) + list(config.items()))
 167 
 168   def call(self, inputs, length=None):
 169     """Implements call() for the layer.
 170 
 171     Args:
 172       inputs: An tensor whose second dimension will be used as `length`. If
 173         `None`, the other `length` argument must be specified.
 174       length: An optional integer specifying the number of positions. If both
 175         `inputs` and `length` are spcified, `length` must be equal to the
 176         second dimension of `inputs`.
 177 
 178     Returns:
 179       A tensor in shape of [length, hidden_size].
 180     """
 181     if inputs is None and length is None:
 182       raise ValueError(
 183           "If inputs is None, `length` must be set in "
 184           "RelativePositionEmbedding().")
 185     if inputs is not None:
 186       input_shape = tf_utils.get_shape_list(inputs)
 187       if length is not None and length != input_shape[1]:
 188         raise ValueError(
 189             "If inputs is not None, `length` must equal to input_shape[1]."
 190         )
 191       length = input_shape[1]
 192     position = tf.cast(tf.range(length), tf.float32)
 193     num_timescales = self._hidden_size // 2
 194     min_timescale, max_timescale = self._min_timescale, self._max_timescale
 195     log_timescale_increment = (
 196         math.log(float(max_timescale) / float(min_timescale)) /
 197         (tf.cast(num_timescales, tf.float32) - 1))
 198     inv_timescales = min_timescale * tf.exp(
 199         tf.cast(tf.range(num_timescales), tf.float32) *
 200         -log_timescale_increment)
 201     scaled_time = tf.expand_dims(position, 1) * tf.expand_dims(inv_timescales,
 202                                                                0)
 203     position_embeddings = tf.concat([tf.sin(scaled_time), tf.cos(scaled_time)],
 204                                     axis=1)
 205     return position_embeddings
