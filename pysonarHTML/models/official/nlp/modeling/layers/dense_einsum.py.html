  27 @tf.keras.utils.register_keras_serializable(package="Text")
  28 class DenseEinsum(tf.keras.layers.Layer):
  29   """A densely connected layer that uses tf.einsum as the backing computation.
  30 
  31   This layer can perform einsum calculations of arbitrary dimensionality.
  32 
  33   Arguments:
  34     output_shape: Positive integer or tuple, dimensionality of the output space.
  35     num_summed_dimensions: The number of dimensions to sum over. Standard 2D
  36       matmul should use 1, 3D matmul should use 2, and so forth.
  37     activation: Activation function to use. If you don't specify anything, no
  38       activation is applied
  39       (ie. "linear" activation: `a(x) = x`).
  40     use_bias: Boolean, whether the layer uses a bias vector.
  41     kernel_initializer: Initializer for the `kernel` weights matrix.
  42     bias_initializer: Initializer for the bias vector.
  43     kernel_regularizer: Regularizer function applied to the `kernel` weights
  44       matrix.
  45     bias_regularizer: Regularizer function applied to the bias vector.
  46     activity_regularizer: Regularizer function applied to the output of the
  47       layer (its "activation")..
  48     kernel_constraint: Constraint function applied to the `kernel` weights
  49       matrix.
  50     bias_constraint: Constraint function applied to the bias vector.
  51   Input shape:
  52     N-D tensor with shape: `(batch_size, ..., input_dim)`. The most common
  53       situation would be a 2D input with shape `(batch_size, input_dim)`.
  54   Output shape:
  55     N-D tensor with shape: `(batch_size, ..., units)`. For instance, for a 2D
  56       input with shape `(batch_size, input_dim)`, the output would have shape
  57       `(batch_size, units)`.
  58   """
  59 
  60   def __init__(self,
  61                output_shape,
  62                num_summed_dimensions=1,
  63                activation=None,
  64                use_bias=True,
  65                kernel_initializer="glorot_uniform",
  66                bias_initializer="zeros",
  67                kernel_regularizer=None,
  68                bias_regularizer=None,
  69                activity_regularizer=None,
  70                kernel_constraint=None,
  71                bias_constraint=None,
  72                **kwargs):
  73     super(DenseEinsum, self).__init__(**kwargs)
  74     self._output_shape = output_shape if isinstance(
  75         output_shape, (list, tuple)) else (output_shape,)
  76     self._activation = tf.keras.activations.get(activation)
  77     self._use_bias = use_bias
  78     self._kernel_initializer = tf.keras.initializers.get(kernel_initializer)
  79     self._bias_initializer = tf.keras.initializers.get(bias_initializer)
  80     self._kernel_regularizer = tf.keras.regularizers.get(kernel_regularizer)
  81     self._bias_regularizer = tf.keras.regularizers.get(bias_regularizer)
  82     self._kernel_constraint = tf.keras.constraints.get(kernel_constraint)
  83     self._bias_constraint = tf.keras.constraints.get(bias_constraint)
  84     self._num_summed_dimensions = num_summed_dimensions
  85     self._einsum_string = None
  86 
  87   def _build_einsum_string(self, free_input_dims, bound_dims, output_dims):
  88     input_str = ""
  89     kernel_str = ""
  90     output_str = ""
  91     letter_offset = 0
  92     for i in range(free_input_dims):
  93       char = _CHR_IDX[i + letter_offset]
  94       input_str += char
  95       output_str += char
  96 
  97     letter_offset += free_input_dims
  98     for i in range(bound_dims):
  99       char = _CHR_IDX[i + letter_offset]
 100       input_str += char
 101       kernel_str += char
 102 
 103     letter_offset += bound_dims
 104     for i in range(output_dims):
 105       char = _CHR_IDX[i + letter_offset]
 106       kernel_str += char
 107       output_str += char
 108 
 109     return input_str + "," + kernel_str + "->" + output_str
 110 
 111   def build(self, input_shape):
 112     input_shape = tf.TensorShape(input_shape)
 113     input_rank = input_shape.rank
 114     free_input_dims = input_rank - self._num_summed_dimensions
 115     output_dims = len(self._output_shape)
 116 
 117     self._einsum_string = self._build_einsum_string(free_input_dims,
 118                                                     self._num_summed_dimensions,
 119                                                     output_dims)
 120 
 121     # This is only saved for testing purposes.
 122     self._kernel_shape = (
 123         input_shape[free_input_dims:].concatenate(self._output_shape))
 124 
 125     self._kernel = self.add_weight(
 126         "kernel",
 127         shape=self._kernel_shape,
 128         initializer=self._kernel_initializer,
 129         regularizer=self._kernel_regularizer,
 130         constraint=self._kernel_constraint,
 131         dtype=self.dtype,
 132         trainable=True)
 133     if self._use_bias:
 134       self._bias = self.add_weight(
 135           "bias",
 136           shape=self._output_shape,
 137           initializer=self._bias_initializer,
 138           regularizer=self._bias_regularizer,
 139           constraint=self._bias_constraint,
 140           dtype=self.dtype,
 141           trainable=True)
 142     else:
 143       self._bias = None
 144     super(DenseEinsum, self).build(input_shape)
 145 
 146   def get_config(self):
 147     config = {
 148         "output_shape":
 149             self._output_shape,
 150         "num_summed_dimensions":
 151             self._num_summed_dimensions,
 152         "activation":
 153             tf.keras.activations.serialize(self._activation),
 154         "use_bias":
 155             self._use_bias,
 156         "kernel_initializer":
 157             tf.keras.initializers.serialize(self._kernel_initializer),
 158         "bias_initializer":
 159             tf.keras.initializers.serialize(self._bias_initializer),
 160         "kernel_regularizer":
 161             tf.keras.regularizers.serialize(self._kernel_regularizer),
 162         "bias_regularizer":
 163             tf.keras.regularizers.serialize(self._bias_regularizer),
 164         "activity_regularizer":
 165             tf.keras.regularizers.serialize(self._activity_regularizer),
 166         "kernel_constraint":
 167             tf.keras.constraints.serialize(self._kernel_constraint),
 168         "bias_constraint":
 169             tf.keras.constraints.serialize(self._bias_constraint)
 170     }
 171     base_config = super(DenseEinsum, self).get_config()
 172     return dict(list(base_config.items()) + list(config.items()))
 173 
 174   def call(self, inputs):
 175     ret = tf.einsum(self._einsum_string, inputs, self._kernel)
 176     if self._use_bias:
 177       ret += self._bias
 178     if self._activation is not None:
 179       ret = self._activation(ret)
 180     return ret
