  40   def test_layer_creation(self, transformer_cls):
  41     test_layer = transformer_cls(
  42         num_attention_heads=10,
  43         intermediate_size=2048,
  44         intermediate_activation='relu')
  45     sequence_length = 21
  46     width = 80
  47     # Create a 3-dimensional input (the first dimension is implicit).
  48     data_tensor = tf.keras.Input(shape=(sequence_length, width))
  49     output_tensor = test_layer(data_tensor)
  50     # The default output of a transformer layer should be the same as the input.
  51     self.assertEqual(data_tensor.shape.as_list(), output_tensor.shape.as_list())

 103   def test_layer_invocation_with_mask(self, transformer_cls):
 104     test_layer = transformer_cls(
 105         num_attention_heads=10,
 106         intermediate_size=2048,
 107         intermediate_activation='relu')
 108     sequence_length = 21
 109     width = 80
 110     # Create a 3-dimensional input (the first dimension is implicit).
 111     data_tensor = tf.keras.Input(shape=(sequence_length, width))
 112     # Create a 2-dimensional input (the first dimension is implicit).
 113     mask_tensor = tf.keras.Input(shape=(sequence_length, sequence_length))
 114     output_tensor = test_layer([data_tensor, mask_tensor])
 115 
 116     # Create a model from the test layer.
 117     model = tf.keras.Model([data_tensor, mask_tensor], output_tensor)
 118 
 119     # Invoke the model on test data. We can't validate the output data itself
 120     # (the NN is too complex) but this will rule out structural runtime errors.
 121     batch_size = 6
 122     input_data = 10 * np.random.random_sample(
 123         (batch_size, sequence_length, width))
 124     # The attention mask should be of shape (batch, from_seq_len, to_seq_len),
 125     # which here is (batch, sequence_length, sequence_length)
 126     mask_data = np.random.randint(
 127         2, size=(batch_size, sequence_length, sequence_length))
 128     _ = model.predict([input_data, mask_data])

 130   def test_layer_output_range(self, transformer_cls):
 131     test_layer = transformer_cls(
 132         num_attention_heads=10,
 133         intermediate_size=2048,
 134         intermediate_activation='relu')
 135     sequence_length = 21
 136     width = 80
 137 
 138     batch_size = 6
 139     input_data = 10 * np.random.random_sample(
 140         (batch_size, sequence_length, width))
 141     mask_data = np.random.randint(
 142         2, size=(batch_size, sequence_length, sequence_length))
 143     output_tensor = test_layer([input_data, mask_data])
 144 
 145     # The layer only attends to the first token and outputs the first token
 146     # embeeding.
 147     new_layer = transformer_cls(
 148         num_attention_heads=10,
 149         intermediate_size=2048,
 150         intermediate_activation='relu',
 151         output_range=1)
 152     _ = new_layer([input_data, mask_data])
 153     new_layer.set_weights(test_layer.get_weights())
 154     new_output_tensor = new_layer([input_data, mask_data])
 155     self.assertAllClose(new_output_tensor, output_tensor[:, 0:1, :])

 199   def test_dynamic_layer_sequence(self, transformer_cls):
 200     test_layer = transformer_cls(
 201         num_attention_heads=10,
 202         intermediate_size=2048,
 203         intermediate_activation='relu',
 204         kernel_initializer=tf.keras.initializers.TruncatedNormal(stddev=0.02))
 205     # Create a 3-dimensional input (the first dimension is implicit).
 206     width = 30
 207     input_tensor = tf.keras.Input(shape=(None, width))
 208     output_tensor = test_layer(input_tensor)
 209     model = tf.keras.Model(input_tensor, output_tensor)
 210 
 211     input_length = 17
 212     input_data = np.ones((1, input_length, width))
 213     output_data = model.predict(input_data)
 214 
 215     self.assertAllEqual([1, input_length, width], output_data.shape)
