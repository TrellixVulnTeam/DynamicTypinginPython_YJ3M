  37   def test_layer_invocation_with_float16_dtype(self):
  38     tf.keras.mixed_precision.experimental.set_policy('mixed_float16')
  39     test_layer = rezero_transformer.ReZeroTransformer(
  40         num_attention_heads=10,
  41         intermediate_size=2048,
  42         intermediate_activation='relu')
  43     sequence_length = 21
  44     width = 80
  45     # Create a 3-dimensional input (the first dimension is implicit).
  46     data_tensor = tf.keras.Input(shape=(sequence_length, width))
  47     # Create a 2-dimensional input (the first dimension is implicit).
  48     mask_tensor = tf.keras.Input(shape=(sequence_length, sequence_length))
  49     output_tensor = test_layer([data_tensor, mask_tensor])
  50 
  51     # Create a model from the test layer.
  52     model = tf.keras.Model([data_tensor, mask_tensor], output_tensor)
  53 
  54     # Invoke the model on test data. We can't validate the output data itself
  55     # (the NN is too complex) but this will rule out structural runtime errors.
  56     batch_size = 6
  57     input_data = (10 * np.random.random_sample(
  58         (batch_size, sequence_length, width)))
  59     # The attention mask should be of shape (batch, from_seq_len, to_seq_len),
  60     # which here is (batch, sequence_length, sequence_length)
  61     mask_data = np.random.randint(
  62         2, size=(batch_size, sequence_length, sequence_length))
  63     _ = model.predict([input_data, mask_data])
  64 
  65   def test_rezero_without_layer_norm(self):
  66     test_layer = rezero_transformer.ReZeroTransformer(
  67         num_attention_heads=10,
  68         intermediate_size=2048,
  69         intermediate_activation='relu',
  70         use_layer_norm=False)
  71 
  72     input_length, width = 16, 30
  73     input_tensor = tf.keras.Input(shape=(input_length, width))
  74     output_tensor = test_layer(input_tensor)
  75     model = tf.keras.Model(input_tensor, output_tensor)
  76 
  77     input_data = np.random.rand(2, input_length, width)
  78     test_layer._rezero_a.assign(1.0)
  79     test_layer.reset_rezero()
  80     output_data = model.predict(input_data)
  81 
  82     self.assertAllClose(input_data, output_data)
  83 
  84   def test_rezero_with_layer_norm(self):
  85     test_layer = rezero_transformer.ReZeroTransformer(
  86         num_attention_heads=10,
  87         intermediate_size=2048,
  88         intermediate_activation='relu',
  89         use_layer_norm=True)
  90 
  91     input_length, width = 16, 30
  92     input_tensor = tf.keras.Input(shape=(input_length, width))
  93     output_tensor = test_layer(input_tensor)
  94     model = tf.keras.Model(input_tensor, output_tensor)
  95 
  96     input_data = np.random.rand(2, input_length, width) + 2.0
  97     output_data = model.predict(input_data)
  98     input_data_normed = (
  99         input_data - np.mean(input_data, axis=-1, keepdims=True)) / (
 100             np.std(input_data, axis=-1, keepdims=True))
 101 
 102     self.assertAllClose(input_data_normed, output_data)

 104   def test_layer_output_range(self):
 105     test_layer = rezero_transformer.ReZeroTransformer(
 106         num_attention_heads=10,
 107         intermediate_size=2048,
 108         intermediate_activation='relu')
 109     sequence_length = 21
 110     width = 80
 111 
 112     batch_size = 6
 113     input_data = 10 * np.random.random_sample(
 114         (batch_size, sequence_length, width))
 115     mask_data = np.random.randint(
 116         2, size=(batch_size, sequence_length, sequence_length))
 117     output_tensor = test_layer([input_data, mask_data])
 118 
 119     # The layer only attends to the first token and outputs the first token
 120     # embeeding.
 121     new_layer = rezero_transformer.ReZeroTransformer(
 122         num_attention_heads=10,
 123         intermediate_size=2048,
 124         intermediate_activation='relu',
 125         output_range=1)
 126     _ = new_layer([input_data, mask_data])
 127     new_layer.set_weights(test_layer.get_weights())
 128     new_output_tensor = new_layer([input_data, mask_data])
 129     self.assertAllClose(new_output_tensor, output_tensor[:, 0:1, :])
