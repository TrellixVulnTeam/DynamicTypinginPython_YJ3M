  30 class TransformerScaffold(tf.keras.layers.Layer):
  31   """Transformer scaffold layer.
  32 
  33   This layer implements the Transformer from "Attention Is All You Need".
  34   (https://arxiv.org/abs/1706.03762), with a customizable attention layer and
  35   feedforward layer option. Users can pass a class to
  36   `attention_cls`/`feedforward_cls` and associated config to
  37   `attention_cfg`/`feedforward_cfg`, in which case the scaffold will
  38   instantiate the class with the config, or pass a class instance to
  39   `attention_cls`/`feedforward_cls`.
  40 
  41   Arguments:
  42     num_attention_heads: Number of attention heads.
  43     intermediate_size: Size of the intermediate layer.
  44     intermediate_activation: Activation for the intermediate layer.
  45     attention_cls: A class to instantiate attention layer, or a layer instance.
  46     attention_cfg: The config with which to instantiate `attention_cls`. Ignored
  47       if attention_cls is a layer instance or None. If `attention_cls` is a
  48       class, but `attention_cfg` is None, following kwargs will be used to
  49       instantiate the attention instance:
  50       {
  51         "num_heads": num_attention_heads,
  52         "key_size": int(hidden_size // num_attention_heads),
  53         "dropout": attention_dropout_rate,
  54         "name": "self_attention"
  55       }, where `hidden_size` is the input tensor's last dimension.
  56     feedforward_cls: A class to instantiate feedforward layer, or a layer
  57       instance. If None, will use the standard feedforward layer as described
  58       in "Attention Is All You Need" paper. If not None, the instantiated
  59       feedforward layer is expected to take the output of attention as input
  60       and its output is this transformer layer's output.
  61     feedforward_cfg: The config with which to instantiate `feedforward_cls`.
  62       Ignored if feedforward_cls is a layer instance or is None.
  63       If `feedforward_cls` is a class, but `feedforward_cfg` is None, following
  64       kwargs will be used to instantiate the feedforward instance:
  65       {
  66         "intermediate_size": intermediate_size,
  67         "intermediate_activation": intermediate_activation,
  68         "dropout": dropout_rate,
  69         "name": "feedforward"
  70       }.
  71     dropout_rate: Dropout probability for the post-attention and output dropout.
  72     attention_dropout_rate: Dropout probability for within the attention layer.
  73     kernel_initializer: Initializer for dense layer kernels.
  74     bias_initializer: Initializer for dense layer biases.
  75     kernel_regularizer: Regularizer for dense layer kernels.
  76     bias_regularizer: Regularizer for dense layer biases.
  77     activity_regularizer: Regularizer for dense layer activity.
  78     kernel_constraint: Constraint for dense layer kernels.
  79     bias_constraint: Constraint for dense layer kernels.
  80   """
  81 
  82   def __init__(self,
  83                num_attention_heads,
  84                intermediate_size,
  85                intermediate_activation,
  86                attention_cls=attention.MultiHeadAttention,
  87                attention_cfg=None,
  88                feedforward_cls=None,
  89                feedforward_cfg=None,
  90                dropout_rate=0.0,
  91                attention_dropout_rate=0.0,
  92                kernel_initializer="glorot_uniform",
  93                bias_initializer="zeros",
  94                kernel_regularizer=None,
  95                bias_regularizer=None,
  96                activity_regularizer=None,
  97                kernel_constraint=None,
  98                bias_constraint=None,
  99                **kwargs):
 100     super(TransformerScaffold, self).__init__(**kwargs)
 101 
 102     self._attention_cfg = attention_cfg
 103     self._attention_cls = attention_cls
 104     self._feedforward_cls = feedforward_cls
 105     self._feedforward_cfg = feedforward_cfg
 106     self._num_heads = num_attention_heads
 107     self._intermediate_size = intermediate_size
 108     self._intermediate_activation = intermediate_activation
 109     self._attention_dropout_rate = attention_dropout_rate
 110     self._dropout_rate = dropout_rate
 111     self._kernel_initializer = tf.keras.initializers.get(kernel_initializer)
 112     self._bias_initializer = tf.keras.initializers.get(bias_initializer)
 113     self._kernel_regularizer = tf.keras.regularizers.get(kernel_regularizer)
 114     self._bias_regularizer = tf.keras.regularizers.get(bias_regularizer)
 115     self._kernel_constraint = tf.keras.constraints.get(kernel_constraint)
 116     self._bias_constraint = tf.keras.constraints.get(bias_constraint)
 117 
 118   def build(self, input_shape):
 119     input_tensor = input_shape[0] if len(input_shape) == 2 else input_shape
 120     input_tensor_shape = tf.TensorShape(input_tensor)
 121     if len(input_tensor_shape) != 3:
 122       raise ValueError(
 123           "TransformerScaffold expects a three-dimensional input of "
 124           "shape [batch, sequence, width].")
 125     batch_size, sequence_length, hidden_size = input_tensor_shape
 126 
 127     if len(input_shape) == 2:
 128       mask_tensor_shape = tf.TensorShape(input_shape[1])
 129       expected_mask_tensor_shape = tf.TensorShape(
 130           [batch_size, sequence_length, sequence_length])
 131       if not expected_mask_tensor_shape.is_compatible_with(mask_tensor_shape):
 132         raise ValueError("When passing a mask tensor to TransformerLayer, the "
 133                          "mask tensor must be of shape [batch, "
 134                          "sequence_length, sequence_length] (here %s). Got a "
 135                          "mask tensor of shape %s." %
 136                          (expected_mask_tensor_shape, mask_tensor_shape))
 137     if hidden_size % self._num_heads != 0:
 138       raise ValueError(
 139           "The input size (%d) is not a multiple of the number of attention "
 140           "heads (%d)" % (hidden_size, self._num_heads))
 141     self._attention_head_size = int(hidden_size // self._num_heads)
 142 
 143     common_kwargs = dict(
 144         kernel_initializer=self._kernel_initializer,
 145         bias_initializer=self._bias_initializer,
 146         kernel_regularizer=self._kernel_regularizer,
 147         bias_regularizer=self._bias_regularizer,
 148         activity_regularizer=self._activity_regularizer,
 149         kernel_constraint=self._kernel_constraint,
 150         bias_constraint=self._bias_constraint)
 151 
 152     def get_layer_instance(instance_or_cls, config, default_config):
 153       if isinstance(instance_or_cls, tf.keras.layers.Layer):
 154         return instance_or_cls
 155       else:
 156         if config is None:
 157           return instance_or_cls(**default_config)
 158         else:
 159           return instance_or_cls(**config)
 160 
 161     default_attention_cfg = {
 162         "num_heads": self._num_heads,
 163         "key_size": self._attention_head_size,
 164         "dropout": self._attention_dropout_rate,
 165         "name": "self_attention"
 166     }
 167     default_attention_cfg.update(common_kwargs)
 168     self._attention_layer = get_layer_instance(
 169         self._attention_cls,
 170         config=self._attention_cfg,
 171         default_config=default_attention_cfg)
 172 
 173     if self._feedforward_cls is not None:
 174       default_feedforward_cfg = {
 175           "intermediate_size": self._intermediate_size,
 176           "intermediate_activation": self._intermediate_activation,
 177           "dropout": self._dropout_rate,
 178           "name": "feedforward",
 179       }
 180       default_feedforward_cfg.update(common_kwargs)
 181       self._feedforward_block = get_layer_instance(
 182           self._feedforward_cls,
 183           config=self._feedforward_cfg,
 184           default_config=default_feedforward_cfg)
 185     else:
 186       self._feedforward_block = None
 187 
 188     self._attention_dropout = tf.keras.layers.Dropout(rate=self._dropout_rate)
 189     # Use float32 in layernorm for numeric stability.
 190     # It is probably safe in mixed_float16, but we haven't validated this yet.
 191     self._attention_layer_norm = (
 192         tf.keras.layers.LayerNormalization(
 193             name="self_attention_layer_norm", axis=-1, epsilon=1e-12,
 194             dtype=tf.float32))
 195 
 196     if self._feedforward_block is None:
 197       self._intermediate_dense = tf.keras.layers.experimental.EinsumDense(
 198           "abc,cd->abd",
 199           output_shape=(None, self._intermediate_size),
 200           bias_axes="d",
 201           name="intermediate",
 202           **common_kwargs)
 203       policy = tf.keras.mixed_precision.experimental.global_policy()
 204       if policy.name == "mixed_bfloat16":
 205         # bfloat16 causes BERT with the LAMB optimizer to not converge
 206         # as well, so we use float32.
 207         # TODO(b/154538392): Investigate this.
 208         policy = tf.float32
 209       self._intermediate_activation_layer = tf.keras.layers.Activation(
 210           self._intermediate_activation, dtype=policy)
 211       self._output_dense = tf.keras.layers.experimental.EinsumDense(
 212           "abc,cd->abd",
 213           output_shape=(None, hidden_size),
 214           bias_axes="d",
 215           name="output",
 216           **common_kwargs)
 217 
 218     self._output_dropout = tf.keras.layers.Dropout(rate=self._dropout_rate)
 219     # Use float32 in layernorm for numeric stability.
 220     self._output_layer_norm = tf.keras.layers.LayerNormalization(
 221         name="output_layer_norm", axis=-1, epsilon=1e-12, dtype=tf.float32)
 222 
 223     super(TransformerScaffold, self).build(input_shape)
 224 
 225   def get_config(self):
 226     config = {
 227         "attention_cls":
 228             self._attention_layer,
 229         "feedforward_cls":
 230             self._feedforward_block,
 231         "num_attention_heads":
 232             self._num_heads,
 233         "intermediate_size":
 234             self._intermediate_size,
 235         "intermediate_activation":
 236             self._intermediate_activation,
 237         "dropout_rate":
 238             self._dropout_rate,
 239         "attention_dropout_rate":
 240             self._attention_dropout_rate,
 241         "kernel_initializer":
 242             tf.keras.initializers.serialize(self._kernel_initializer),
 243         "bias_initializer":
 244             tf.keras.initializers.serialize(self._bias_initializer),
 245         "kernel_regularizer":
 246             tf.keras.regularizers.serialize(self._kernel_regularizer),
 247         "bias_regularizer":
 248             tf.keras.regularizers.serialize(self._bias_regularizer),
 249         "activity_regularizer":
 250             tf.keras.regularizers.serialize(self._activity_regularizer),
 251         "kernel_constraint":
 252             tf.keras.constraints.serialize(self._kernel_constraint),
 253         "bias_constraint":
 254             tf.keras.constraints.serialize(self._bias_constraint)
 255     }
 256     base_config = super(TransformerScaffold, self).get_config()
 257     return dict(list(base_config.items()) + list(config.items()))
 258 
 259   def call(self, inputs):
 260     if isinstance(inputs, (list, tuple)) and len(inputs) == 2:
 261       input_tensor, attention_mask = inputs
 262     else:
 263       input_tensor, attention_mask = (inputs, None)
 264 
 265     attention_inputs = [input_tensor, input_tensor]
 266 
 267     attention_output = self._attention_layer(attention_inputs, attention_mask)
 268     attention_output = self._attention_dropout(attention_output)
 269     attention_output = self._attention_layer_norm(input_tensor +
 270                                                   attention_output)
 271     if self._feedforward_block is None:
 272       intermediate_output = self._intermediate_dense(attention_output)
 273       intermediate_output = self._intermediate_activation_layer(
 274           intermediate_output)
 275       layer_output = self._output_dense(intermediate_output)
 276       layer_output = self._output_dropout(layer_output)
 277       # During mixed precision training, attention_output is from layer norm
 278       # and is always fp32 for now. Cast layer_output to fp32 for the subsequent
 279       # add.
 280       layer_output = tf.cast(layer_output, tf.float32)
 281       layer_output = self._output_layer_norm(layer_output + attention_output)
 282     else:
 283       layer_output = self._feedforward_block(attention_output)
 284 
 285     return layer_output
