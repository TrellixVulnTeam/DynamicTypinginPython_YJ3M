  28 class GatedFeedforward(tf.keras.layers.Layer):
  29   """Gated linear feedforward layer.
  30 
  31   This layer follows the paper "GLU Variants Improve Transformer"
  32   (https://arxiv.org/abs/2002.05202). In additional, it allows to stack
  33   multiple feedforward blocks and specify the position of dropout layer.
  34 
  35   Arguments:
  36     intermediate_size: Size of the intermediate layer.
  37     intermediate_activation: Activation for the intermediate layer.
  38     dropout: Dropout probability for the output dropout.
  39     use_gate: Whether to use gated linear units. If True, assuming `GELU` as
  40       the activation and omitting bias, will apply
  41       `GEGLU(x, W, V, W_2) = (GEGLU(xW) * xV)W2`; if False, will follow
  42       "Attention Is All You Need" (https://arxiv.org/abs/1706.03762) paper
  43       and apply `FFN(x, W, W_2) = GELU(xW_1)W_2.`
  44     num_blocks: The number of feedforward blocks to stack. Each block contains
  45       a (gated) linear layer and a fully connected layer followed by dropout,
  46       layer norm and residual.
  47     dropout_position: Where to apply the dropout, the value can be either
  48       `before_residual` or `after_residual`. If `before_residual`, will apply
  49       `layer_output = layer_norm(dropout(layer_output) + layer_input)`;
  50       if `after residual`, will apply
  51       `layer_output = dropout(layer_norm(layer_output + layer_input))`.
  52     kernel_initializer: Initializer for dense layer kernels.
  53     bias_initializer: Initializer for dense layer biases.
  54     kernel_regularizer: Regularizer for dense layer kernels.
  55     bias_regularizer: Regularizer for dense layer biases.
  56     activity_regularizer: Regularizer for dense layer activity.
  57     kernel_constraint: Constraint for dense layer kernels.
  58     bias_constraint: Constraint for dense layer kernels.
  59   """
  60 
  61   def __init__(self,
  62                intermediate_size,
  63                intermediate_activation,
  64                dropout,
  65                use_gate=True,
  66                num_blocks=1,
  67                dropout_position="before_residual",
  68                kernel_initializer="glorot_uniform",
  69                bias_initializer="zeros",
  70                kernel_regularizer=None,
  71                bias_regularizer=None,
  72                activity_regularizer=None,
  73                kernel_constraint=None,
  74                bias_constraint=None,
  75                **kwargs):
  76     super(GatedFeedforward, self).__init__(**kwargs)
  77     self._intermediate_size = intermediate_size
  78     self._intermediate_activation = intermediate_activation
  79     self._dropout = dropout
  80     self._use_gate = use_gate
  81     self._num_blocks = num_blocks
  82     self._dropout_position = dropout_position
  83     if self._dropout_position not in ("before_residual", "after_residual"):
  84       raise ValueError(
  85           "The dropout_position should be either `before_residual` or"
  86           "`after_residual`, got: %s" % self._dropout_position)
  87 
  88     self._kernel_initializer = tf.keras.initializers.get(kernel_initializer)
  89     self._bias_initializer = tf.keras.initializers.get(bias_initializer)
  90     self._kernel_regularizer = tf.keras.regularizers.get(kernel_regularizer)
  91     self._bias_regularizer = tf.keras.regularizers.get(bias_regularizer)
  92     self._activity_regularizer = tf.keras.regularizers.get(activity_regularizer)
  93     self._kernel_constraint = tf.keras.constraints.get(kernel_constraint)
  94     self._bias_constraint = tf.keras.constraints.get(bias_constraint)
  95 
  96   def build(self, input_shape):
  97     hidden_size = input_shape.as_list()[-1]
  98 
  99     common_kwargs = dict(
 100         kernel_initializer=self._kernel_initializer,
 101         bias_initializer=self._bias_initializer,
 102         kernel_regularizer=self._kernel_regularizer,
 103         bias_regularizer=self._bias_regularizer,
 104         activity_regularizer=self._activity_regularizer,
 105         kernel_constraint=self._kernel_constraint,
 106         bias_constraint=self._bias_constraint)
 107     self._intermediate_dense = []
 108     self._intermediate_activation_layers = []
 109     self._gate_dense = []
 110     self._output_dense = []
 111     self._output_dropout = []
 112     self._output_layer_norm = []
 113     activation_policy = tf.keras.mixed_precision.experimental.global_policy()
 114     if activation_policy.name == "mixed_bfloat16":
 115       # bfloat16 causes BERT with the LAMB optimizer to not converge
 116       # as well, so we use float32.
 117       # TODO(b/154538392): Investigate this.
 118       activation_policy = tf.float32
 119     for i in range(self._num_blocks):
 120       self._intermediate_dense.append(
 121           tf.keras.layers.experimental.EinsumDense(
 122               "abc,cd->abd",
 123               output_shape=(None, self._intermediate_size),
 124               bias_axes="d",
 125               name="intermediate_%d" % i,
 126               **common_kwargs))
 127       self._intermediate_activation_layers.append(tf.keras.layers.Activation(
 128           self._intermediate_activation, dtype=activation_policy))
 129       if self._use_gate:
 130         self._gate_dense.append(
 131             tf.keras.layers.experimental.EinsumDense(
 132                 "abc,cd->abd",
 133                 output_shape=(None, self._intermediate_size),
 134                 bias_axes="d",
 135                 name="gate_%d" % i,
 136                 **common_kwargs))
 137       self._output_dense.append(
 138           tf.keras.layers.experimental.EinsumDense(
 139               "abc,cd->abd",
 140               output_shape=(None, hidden_size),
 141               bias_axes="d",
 142               name="output_%d" % i,
 143               **common_kwargs))
 144       self._output_dropout.append(
 145           tf.keras.layers.Dropout(rate=self._dropout))
 146       # Use float32 in layernorm for numeric stability.
 147       self._output_layer_norm.append(
 148           tf.keras.layers.LayerNormalization(
 149               name="output_layer_norm_%d" % i,
 150               axis=-1,
 151               epsilon=1e-12,
 152               dtype=tf.float32))
 153 
 154   def get_config(self):
 155     config = {
 156         "intermediate_size":
 157             self._intermediate_size,
 158         "intermediate_activation":
 159             self._intermediate_activation,
 160         "dropout":
 161             self._dropout,
 162         "use_gate":
 163             self._use_gate,
 164         "num_blocks":
 165             self._num_blocks,
 166         "dropout_position":
 167             self._dropout_position,
 168         "kernel_initializer":
 169             tf.keras.initializers.serialize(self._kernel_initializer),
 170         "bias_initializer":
 171             tf.keras.initializers.serialize(self._bias_initializer),
 172         "kernel_regularizer":
 173             tf.keras.regularizers.serialize(self._kernel_regularizer),
 174         "bias_regularizer":
 175             tf.keras.regularizers.serialize(self._bias_regularizer),
 176         "activity_regularizer":
 177             tf.keras.regularizers.serialize(self._activity_regularizer),
 178         "kernel_constraint":
 179             tf.keras.constraints.serialize(self._kernel_constraint),
 180         "bias_constraint":
 181             tf.keras.constraints.serialize(self._bias_constraint)
 182     }
 183     base_config = super(GatedFeedforward, self).get_config()
 184     return dict(list(base_config.items()) + list(config.items()))
 185 
 186   def call(self, inputs):
 187     layer_output = inputs
 188     for i in range(self._num_blocks):
 189       layer_input = layer_output
 190       intermediate_output = self._intermediate_dense[i](layer_input)
 191       intermediate_output = self._intermediate_activation_layers[i](
 192           intermediate_output)
 193       if self._use_gate:
 194         gated_linear = self._gate_dense[i](layer_input)
 195         intermediate_output = intermediate_output * gated_linear
 196 
 197       layer_output = self._output_dense[i](intermediate_output)
 198       if self._dropout_position == "before_residual":
 199         layer_output = self._output_dropout[i](layer_output)
 200 
 201       # During mixed precision training, `layer_input` may be from layer norm.
 202       # If so, it is always fp32. Cast layer_output to fp32 for the subsequent
 203       # add.
 204       if layer_input.dtype == tf.float32:
 205         layer_output = tf.cast(layer_output, tf.float32)
 206       layer_output = self._output_layer_norm[i](layer_output + layer_input)
 207       if self._dropout_position == "after_residual":
 208         layer_output = self._output_dropout[i](layer_output)
 209 
 210     return layer_output
