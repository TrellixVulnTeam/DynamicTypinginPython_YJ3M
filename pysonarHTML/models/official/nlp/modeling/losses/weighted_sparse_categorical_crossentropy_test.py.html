  34   def create_lm_model(self,
  35                       vocab_size,
  36                       sequence_length,
  37                       hidden_size,
  38                       num_predictions,
  39                       output="predictions"):
  40     # First, create a transformer stack that we can use to get the LM's
  41     # vocabulary weight.
  42     xformer_stack = networks.TransformerEncoder(
  43         vocab_size=vocab_size,
  44         num_layers=1,
  45         sequence_length=sequence_length,
  46         hidden_size=hidden_size,
  47         num_attention_heads=4,
  48     )
  49     word_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  50     mask = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  51     type_ids = tf.keras.Input(shape=(sequence_length,), dtype=tf.int32)
  52     _ = xformer_stack([word_ids, mask, type_ids])
  53 
  54     # Create a maskedLM from the transformer stack.
  55     test_layer = layers.MaskedLM(
  56         embedding_table=xformer_stack.get_embedding_table(),
  57         output=output)
  58 
  59     # Create a model from the masked LM layer.
  60     lm_input_tensor = tf.keras.Input(shape=(sequence_length, hidden_size))
  61     masked_lm_positions = tf.keras.Input(
  62         shape=(num_predictions,), dtype=tf.int32)
  63     output = test_layer(lm_input_tensor, masked_positions=masked_lm_positions)
  64     return tf.keras.Model([lm_input_tensor, masked_lm_positions], output)
  65 
  66   def create_classification_model(self, input_width, num_classes):
  67     test_object = networks.Classification(
  68         input_width=input_width, num_classes=num_classes)
  69     # Create a 2-dimensional input (the first dimension is implicit).
  70     pooled_data = tf.keras.Input(shape=(input_width,), dtype=tf.float32)
  71     output = test_object(pooled_data)
  72     return tf.keras.Model(pooled_data, output)

  74   def test_per_example_loss_3d_input(self):
  75     """Test per-example loss with a 3-dimensional input, from a masked LM."""
  76     vocab_size = 100
  77     sequence_length = 32
  78     hidden_size = 64
  79     num_predictions = 21
  80     model = self.create_lm_model(
  81         vocab_size=vocab_size,
  82         sequence_length=sequence_length,
  83         hidden_size=hidden_size,
  84         num_predictions=num_predictions)
  85 
  86     # Get the output of the masked LM.
  87     batch_size = 3
  88     lm_input_data = 10 * np.random.random_sample(
  89         (batch_size, sequence_length, hidden_size))
  90     masked_position_data = np.random.randint(
  91         2, size=(batch_size, num_predictions))
  92     output_data = model.predict([lm_input_data, masked_position_data])
  93 
  94     # Calculate per-example loss.
  95     labels = np.random.randint(vocab_size, size=(batch_size, num_predictions))
  96     per_example_loss_data = weighted_sparse_categorical_crossentropy.per_example_loss(
  97         predictions=output_data, labels=labels)
  98 
  99     # Per-example loss data should have one value per prediction, and those
 100     # values shouldn't be zero in this case (as we're using random data).
 101     expected_shape = [batch_size, num_predictions]
 102     self.assertEqual(expected_shape, per_example_loss_data.shape.as_list())
 103     self.assertNotAllClose(
 104         tf.zeros_like(per_example_loss_data), per_example_loss_data)

 295   def test_mismatched_weights_and_labels_ranks_fail(self):
 296     """Test that the loss asserts when rank(predictions) != rank(labels)."""
 297     batch_size = 3
 298     output_data = np.random.random_sample((batch_size, 10, 15))
 299     labels = np.random.randint(10, size=(batch_size, 10))
 300     weights = np.random.randint(2, size=(batch_size))
 301 
 302     with self.assertRaisesRegex(RuntimeError, ".*of the same rank.*"):
 303       _ = weighted_sparse_categorical_crossentropy.per_example_loss(
 304           predictions=output_data, labels=labels, weights=weights)
 305     with self.assertRaisesRegex(RuntimeError, ".*of the same rank.*"):
 306       _ = weighted_sparse_categorical_crossentropy.loss(
 307           predictions=output_data, labels=labels, weights=weights)

 309   def test_tf_tensor_inputs(self):
 310     """Test that tf.Tensors can be used as inputs to the loss function."""
 311     batch_size = 3
 312     output_data = tf.convert_to_tensor(
 313         np.random.random_sample((batch_size, 10, 15)))
 314     labels = tf.convert_to_tensor(np.random.randint(10, size=(batch_size, 10)))
 315     weights = tf.convert_to_tensor(np.random.randint(2, size=(batch_size, 10)))
 316 
 317     # We're not trying to validate numerical correctness, just ensure that
 318     # we can in fact pass tensors to these functions without causing runtime
 319     # errors from the shape checking code.
 320     _ = weighted_sparse_categorical_crossentropy.per_example_loss(
 321         predictions=output_data, labels=labels, weights=weights)
 322     _ = weighted_sparse_categorical_crossentropy.loss(
 323         predictions=output_data, labels=labels, weights=weights)

 325   def test_legacy_lm_loss_compatibility(self):
 326     """Test to validate computational correctness during refactors."""
 327     # This is the empirical output of a masked LM with the following parameters:
 328     #   batch_size = 3
 329     #   vocab_size = 5
 330     #   sequence_length = 4
 331     #   num_predictions = 2
 332     output_data = np.array(
 333         [[[-2.5286622, -1.0963473, -1.4925185, -2.4451098, -1.2923571],
 334           [-2.7117882, -1.1205841, -4.02187, -0.9966936, -1.5119683]],
 335          [[-2.5379114, -0.82479054, -2.287932, -1.3747153, -2.053741],
 336           [-2.5379114, -0.82479054, -2.287932, -1.3747153, -2.053741]],
 337          [[-2.7760355, -1.8219438, -3.0924666, -1.0779881, -0.9407509],
 338           [-2.7760355, -1.8219438, -3.0924666, -1.0779881, -0.9407509]]])
 339     labels = np.array([[4, 0], [2, 2], [2, 1]])
 340 
 341     # Validate that per_example loss calculations are the same.
 342     per_example_loss_data = weighted_sparse_categorical_crossentropy.per_example_loss(
 343         predictions=output_data, labels=labels)
 344     expected_per_example_loss_data = [[1.2923571, 2.7117882],
 345                                       [2.287932, 2.287932],
 346                                       [3.0924666, 1.8219438]]
 347     self.assertAllClose(expected_per_example_loss_data, per_example_loss_data)
 348 
 349     # Validate that overall loss calculations are the same.
 350     weights = np.array([[1, 0], [0, 0], [0, 0]])
 351     loss_data = weighted_sparse_categorical_crossentropy.loss(
 352         predictions=output_data, labels=labels, weights=weights)
 353     expected_loss_data = 1.2923441
 354     self.assertAllClose(expected_loss_data, loss_data)
