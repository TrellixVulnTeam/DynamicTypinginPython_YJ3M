  40   return (not strategy) or strategy.extended.should_save_summary

  43 def _save_checkpoint(strategy, checkpoint, model_dir, checkpoint_prefix):
  44   """Saves model to with provided checkpoint prefix."""
  45 
  46   if _should_export_checkpoint(strategy):
  47     checkpoint_path = os.path.join(model_dir, checkpoint_prefix)
  48     saved_path = checkpoint.save(checkpoint_path)
  49     logging.info('Saving model as TF checkpoint: %s', saved_path)
  50   else:
  51     # In multi worker training we need every worker to save checkpoint, because
  52     # variables can trigger synchronization on read and synchronization needs
  53     # all workers to participate. To avoid workers overriding each other we save
  54     # to a temporary directory on non-chief workers.
  55     tmp_dir = tempfile.mkdtemp()
  56     checkpoint.save(os.path.join(tmp_dir, 'ckpt'))
  57     tf.io.gfile.rmtree(tmp_dir)
  58   return

  61 def _get_input_iterator(input_fn, strategy):
  62   """Returns distributed dataset iterator."""
  63   # When training with TPU pods, datasets needs to be cloned across
  64   # workers. Since Dataset instance cannot be cloned in eager mode, we instead
  65   # pass callable that returns a dataset.
  66   if not callable(input_fn):
  67     raise ValueError('`input_fn` should be a closure that returns a dataset.')
  68   iterator = iter(
  69       strategy.experimental_distribute_datasets_from_function(input_fn))
  70   return iterator

  73 def _float_metric_value(metric):
  74   """Gets the value of a float-value keras metric."""
  75   return metric.result().numpy().astype(float)

  78 def steps_to_run(current_step, steps_per_epoch, steps_per_loop):
  79   """Calculates steps to run on device."""
  80   if steps_per_loop <= 0:
  81     raise ValueError('steps_per_loop should be positive integer.')
  82   if steps_per_loop == 1:
  83     return steps_per_loop
  84   remainder_in_epoch = current_step % steps_per_epoch
  85   if remainder_in_epoch != 0:
  86     return min(steps_per_epoch - remainder_in_epoch, steps_per_loop)
  87   else:
  88     return steps_per_loop

  91 def write_txt_summary(training_summary, summary_dir):
  92   """Writes a summary text file to record stats."""
  93   if not tf.io.gfile.exists(summary_dir):
  94     tf.io.gfile.mkdir(summary_dir)
  95   summary_path = os.path.join(summary_dir, _SUMMARY_TXT)
  96   with tf.io.gfile.GFile(summary_path, 'wb') as f:
  97     logging.info('Training Summary: \n%s', str(training_summary))
  98     f.write(json.dumps(training_summary, indent=4))

 101 @deprecation.deprecated(
 102     None, 'This function is deprecated. Please use Keras compile/fit instead.')
 103 def run_customized_training_loop(
 104     # pylint: disable=invalid-name
 105     _sentinel=None,
 106     # pylint: enable=invalid-name
 107     strategy=None,
 108     model_fn=None,
 109     loss_fn=None,
 110     scale_loss=True,
 111     model_dir=None,
 112     train_input_fn=None,
 113     steps_per_epoch=None,
 114     num_eval_per_epoch=1,
 115     steps_per_loop=None,
 116     epochs=1,
 117     eval_input_fn=None,
 118     eval_steps=None,
 119     metric_fn=None,
 120     init_checkpoint=None,
 121     custom_callbacks=None,
 122     run_eagerly=False,
 123     sub_model_export_name=None,
 124     explicit_allreduce=False,
 125     pre_allreduce_callbacks=None,
 126     post_allreduce_callbacks=None,
 127     train_summary_interval=0):
 128   """Run BERT pretrain model training using low-level API.
 129 
 130   Arguments:
 131       _sentinel: Used to prevent positional parameters. Internal, do not use.
 132       strategy: Distribution strategy on which to run low level training loop.
 133       model_fn: Function that returns a tuple (model, sub_model). Caller of this
 134         function should add optimizer to the `model` via calling
 135         `model.compile()` API or manually setting `model.optimizer` attribute.
 136         Second element of the returned tuple(sub_model) is an optional sub model
 137         to be used for initial checkpoint -- if provided.
 138       loss_fn: Function with signature func(labels, logits) and returns a loss
 139         tensor.
 140       scale_loss: Whether to divide the raw loss by number of replicas before
 141         gradients calculation.
 142       model_dir: Model directory used during training for restoring/saving model
 143         weights.
 144       train_input_fn: Function that returns a tf.data.Dataset used for training.
 145       steps_per_epoch: Number of steps to run per epoch. At the end of each
 146         epoch, model checkpoint will be saved and evaluation will be conducted
 147         if evaluation dataset is provided.
 148       num_eval_per_epoch: Number of evaluations per epoch.
 149       steps_per_loop: Number of steps per graph-mode loop. In order to reduce
 150         communication in eager context, training logs are printed every
 151         steps_per_loop.
 152       epochs: Number of epochs to train.
 153       eval_input_fn: Function that returns evaluation dataset. If none,
 154         evaluation is skipped.
 155       eval_steps: Number of steps to run evaluation. Required if `eval_input_fn`
 156         is not none.
 157       metric_fn: A metrics function that returns a Keras Metric object to record
 158         evaluation result using evaluation dataset or with training dataset
 159         after every epoch.
 160       init_checkpoint: Optional checkpoint to load to `sub_model` returned by
 161         `model_fn`.
 162       custom_callbacks: A list of Keras Callbacks objects to run during
 163         training. More specifically, `on_train_begin(), on_train_end(),
 164         on_batch_begin()`, `on_batch_end()`, `on_epoch_begin()`,
 165         `on_epoch_end()` methods are invoked during training.
 166         Note that some metrics may be missing from `logs`.
 167       run_eagerly: Whether to run model training in pure eager execution. This
 168         should be disable for TPUStrategy.
 169       sub_model_export_name: If not None, will export `sub_model` returned by
 170         `model_fn` into checkpoint files. The name of intermediate checkpoint
 171         file is {sub_model_export_name}_step_{step}.ckpt and the last
 172         checkpint's name is {sub_model_export_name}.ckpt; if None, `sub_model`
 173         will not be exported as checkpoint.
 174       explicit_allreduce: Whether to explicitly perform gradient allreduce,
 175         instead of relying on implicit allreduce in optimizer.apply_gradients().
 176         default is False. For now, if training using FP16 mixed precision,
 177         explicit allreduce will aggregate gradients in FP16 format. For TPU and
 178         GPU training using FP32, explicit allreduce will aggregate gradients in
 179         FP32 format.
 180       pre_allreduce_callbacks: A list of callback functions that takes gradients
 181         and model variables pairs as input, manipulate them, and returns a new
 182         gradients and model variables paris. The callback functions will be
 183         invoked in the list order and before gradients are allreduced. With
 184         mixed precision training, the pre_allreduce_allbacks will be applied on
 185         scaled_gradients. Default is no callbacks. Only used when
 186         explicit_allreduce=True.
 187       post_allreduce_callbacks: A list of callback functions that takes
 188         gradients and model variables pairs as input, manipulate them, and
 189         returns a new gradients and model variables paris. The callback
 190         functions will be invoked in the list order and right before gradients
 191         are applied to variables for updates. Default is no callbacks. Only used
 192         when explicit_allreduce=True.
 193       train_summary_interval: Step interval for training summaries. If the value
 194         is a negative number, then training summaries are not enabled.
 195 
 196   Returns:
 197       Trained model.
 198 
 199   Raises:
 200       ValueError: (1) When model returned by `model_fn` does not have optimizer
 201         attribute or when required parameters are set to none. (2) eval args are
 202         not specified correctly. (3) metric_fn must be a callable if specified.
 203         (4) sub_model_checkpoint_name is specified, but `sub_model` returned
 204         by `model_fn` is None.
 205   """
 206 
 207   if _sentinel is not None:
 208     raise ValueError('only call `run_customized_training_loop()` '
 209                      'with named arguments.')
 210 
 211   required_arguments = [
 212       strategy, model_fn, loss_fn, model_dir, steps_per_epoch, train_input_fn
 213   ]
 214 
 215   steps_between_evals = int(steps_per_epoch / num_eval_per_epoch)
 216   if [arg for arg in required_arguments if arg is None]:
 217     raise ValueError('`strategy`, `model_fn`, `loss_fn`, `model_dir`, '
 218                      '`steps_per_epoch` and `train_input_fn` are required '
 219                      'parameters.')
 220   if not steps_per_loop:
 221     if tf.config.list_logical_devices('TPU'):
 222       # One can't fully utilize a TPU with steps_per_loop=1, so in this case
 223       # default users to a more useful value.
 224       steps_per_loop = min(1000, steps_between_evals)
 225     else:
 226       steps_per_loop = 1
 227     logging.info('steps_per_loop not specified. Using steps_per_loop=%d',
 228                  steps_per_loop)
 229   if steps_per_loop > steps_between_evals:
 230     logging.warning(
 231         'steps_per_loop: %d is specified to be greater than '
 232         ' steps_between_evals: %d, we will use steps_between_evals as'
 233         ' steps_per_loop.', steps_per_loop, steps_between_evals)
 234     steps_per_loop = steps_between_evals
 235   assert tf.executing_eagerly()
 236 
 237   if run_eagerly:
 238     if isinstance(strategy, tf.distribute.experimental.TPUStrategy):
 239       raise ValueError(
 240           'TPUStrategy should not run eagerly as it heavily relies on graph'
 241           ' optimization for the distributed system.')
 242 
 243   if eval_input_fn and eval_steps is None:
 244     raise ValueError(
 245         '`eval_step` is required when `eval_input_fn ` is not none.')
 246   if metric_fn and not callable(metric_fn):
 247     raise ValueError(
 248         'if `metric_fn` is specified, metric_fn must be a callable.')
 249 
 250   total_training_steps = steps_per_epoch * epochs
 251   train_iterator = _get_input_iterator(train_input_fn, strategy)
 252   eval_loss_metric = tf.keras.metrics.Mean('training_loss', dtype=tf.float32)
 253 
 254   with distribution_utils.get_strategy_scope(strategy):
 255     # To correctly place the model weights on accelerators,
 256     # model and optimizer should be created in scope.
 257     model, sub_model = model_fn()
 258     if not hasattr(model, 'optimizer'):
 259       raise ValueError('User should set optimizer attribute to model '
 260                        'inside `model_fn`.')
 261     if sub_model_export_name and sub_model is None:
 262       raise ValueError('sub_model_export_name is specified as %s, but '
 263                        'sub_model is None.' % sub_model_export_name)
 264 
 265     callback_list = tf.keras.callbacks.CallbackList(
 266         callbacks=custom_callbacks, model=model)
 267 
 268     optimizer = model.optimizer
 269 
 270     if init_checkpoint:
 271       logging.info(
 272           'Checkpoint file %s found and restoring from '
 273           'initial checkpoint for core model.', init_checkpoint)
 274       checkpoint = tf.train.Checkpoint(model=sub_model)
 275       checkpoint.restore(init_checkpoint).assert_existing_objects_matched()
 276       logging.info('Loading from checkpoint file completed')
 277 
 278     train_loss_metric = tf.keras.metrics.Mean('training_loss', dtype=tf.float32)
 279     eval_metrics = [metric_fn()] if metric_fn else []
 280     # If evaluation is required, make a copy of metric as it will be used by
 281     # both train and evaluation.
 282     train_metrics = [
 283         metric.__class__.from_config(metric.get_config())
 284         for metric in eval_metrics
 285     ]
 286 
 287     # Create summary writers
 288     if _should_export_summary(strategy):
 289       summary_dir = os.path.join(model_dir, 'summaries')
 290     else:
 291       # In multi worker training we need every worker to write summary, because
 292       # variables can trigger synchronization on read and synchronization needs
 293       # all workers to participate.
 294       summary_dir = tempfile.mkdtemp()
 295     eval_summary_writer = tf.summary.create_file_writer(
 296         os.path.join(summary_dir, 'eval'))
 297     last_summary_step = 0
 298     if steps_per_loop >= _MIN_SUMMARY_STEPS and train_summary_interval >= 0:
 299       # Only writes summary when the stats are collected sufficiently over
 300       # enough steps.
 301       train_summary_writer = tf.summary.create_file_writer(
 302           os.path.join(summary_dir, 'train'))
 303     else:
 304       train_summary_writer = tf.summary.create_noop_writer()
 305 
 306     # Collects training variables.
 307     training_vars = model.trainable_variables
 308 
 309     def _replicated_step(inputs):
 310       """Replicated training step."""
 311 
 312       inputs, labels = inputs
 313       with tf.GradientTape() as tape:
 314         model_outputs = model(inputs, training=True)
 315         loss = loss_fn(labels, model_outputs)
 316         # Raw loss is used for reporting in metrics/logs.
 317         raw_loss = loss
 318         if scale_loss:
 319           # Scales down the loss for gradients to be invariant from replicas.
 320           loss = loss / strategy.num_replicas_in_sync
 321 
 322       if explicit_allreduce:
 323         grad_utils.minimize_using_explicit_allreduce(tape, optimizer, loss,
 324                                                      training_vars,
 325                                                      pre_allreduce_callbacks,
 326                                                      post_allreduce_callbacks)
 327       else:
 328         if isinstance(optimizer,
 329                       tf.keras.mixed_precision.experimental.LossScaleOptimizer):
 330           with tape:
 331             scaled_loss = optimizer.get_scaled_loss(loss)
 332           scaled_grads = tape.gradient(scaled_loss, training_vars)
 333           grads = optimizer.get_unscaled_gradients(scaled_grads)
 334         else:
 335           grads = tape.gradient(loss, training_vars)
 336         optimizer.apply_gradients(zip(grads, training_vars))
 337       # For reporting, the metric takes the mean of losses.
 338       train_loss_metric.update_state(raw_loss)
 339       for metric in train_metrics:
 340         metric.update_state(labels, model_outputs)
 341 
 342     @tf.function
 343     def train_steps(iterator, steps):
 344       """Performs distributed training steps in a loop.
 345 
 346       Args:
 347         iterator: the distributed iterator of training datasets.
 348         steps: an tf.int32 integer tensor to specify number of steps to run
 349           inside host training loop.
 350 
 351       Raises:
 352         ValueError: Any of the arguments or tensor shapes are invalid.
 353       """
 354       if not isinstance(steps, tf.Tensor):
 355         raise ValueError('steps should be an Tensor. Python object may cause '
 356                          'retracing.')
 357 
 358       for _ in tf.range(steps):
 359         strategy.run(_replicated_step, args=(next(iterator),))
 360 
 361     def train_single_step(iterator):
 362       """Performs a distributed training step.
 363 
 364       Args:
 365         iterator: the distributed iterator of training datasets.
 366 
 367       Raises:
 368         ValueError: Any of the arguments or tensor shapes are invalid.
 369       """
 370       strategy.run(_replicated_step, args=(next(iterator),))
 371 
 372     def test_step(iterator):
 373       """Calculates evaluation metrics on distributed devices."""
 374 
 375       def _test_step_fn(inputs):
 376         """Replicated accuracy calculation."""
 377 
 378         inputs, labels = inputs
 379         model_outputs = model(inputs, training=False)
 380         for metric in eval_metrics:
 381           metric.update_state(labels, model_outputs)
 382         return model_outputs, labels
 383 
 384       outputs, labels = strategy.run(_test_step_fn, args=(next(iterator),))
 385       outputs = tf.nest.map_structure(strategy.experimental_local_results,
 386                                       outputs)
 387       labels = tf.nest.map_structure(strategy.experimental_local_results,
 388                                      labels)
 389       return outputs, labels
 390 
 391     if not run_eagerly:
 392       train_single_step = tf.function(train_single_step)
 393       test_step = tf.function(test_step)
 394 
 395     def _run_evaluation(current_training_step, test_iterator):
 396       """Runs validation steps and aggregate metrics.
 397 
 398       Args:
 399         current_training_step: tf.int32 tensor containing the current step.
 400         test_iterator: distributed iterator of test datasets.
 401 
 402       Returns:
 403         A dict of metic names and values.
 404       """
 405       # The last batch of the evaluation is often smaller than previous ones.
 406       # Moreover, in some distributed pieces it might even be empty. Therefore,
 407       # different from the way training_loss is calculated, it is needed to
 408       # gather all the logits and labels here to calculate the evaluation loss
 409       # outside.
 410       loss_list, loss_weights = list(), list()
 411       for _ in range(eval_steps):
 412         outputs, labels = test_step(test_iterator)
 413         for cur_logits, cur_labels in zip(outputs, labels):
 414           # This is to handle cases when cur_labels is not a single tensor,
 415           # but a dict of tensors.
 416           cur_weight = tf.shape(tf.nest.flatten(cur_labels)[0])[0]
 417           if cur_weight != 0:
 418             loss_list.append(loss_fn(cur_labels, cur_logits).numpy())
 419             loss_weights.append(cur_weight)
 420       # The sample_weights are the actual number of examples in each batch,
 421       # a summation of numbers of examples in each replica if using
 422       # distributed training.
 423       eval_loss_metric.update_state(loss_list, sample_weight=loss_weights)
 424 
 425       logs = {}
 426       with eval_summary_writer.as_default():
 427         for metric in [eval_loss_metric] + eval_metrics + model.metrics:
 428           metric_value = _float_metric_value(metric)
 429           logs[metric.name] = metric_value
 430           logging.info('Step: [%d] Validation %s = %f', current_training_step,
 431                        metric.name, metric_value)
 432           tf.summary.scalar(
 433               metric.name, metric_value, step=current_training_step)
 434         eval_summary_writer.flush()
 435 
 436       return logs
 437 
 438     # Training loop starts here.
 439     checkpoint = tf.train.Checkpoint(
 440         model=model, optimizer=optimizer, global_step=optimizer.iterations)
 441     sub_model_checkpoint = tf.train.Checkpoint(
 442         model=sub_model,
 443         global_step=optimizer.iterations) if sub_model_export_name else None
 444 
 445     latest_checkpoint_file = tf.train.latest_checkpoint(model_dir)
 446     if latest_checkpoint_file:
 447       logging.info('Checkpoint file %s found and restoring from '
 448                    'checkpoint', latest_checkpoint_file)
 449       checkpoint.restore(latest_checkpoint_file)
 450       logging.info('Loading from checkpoint file completed')
 451 
 452     current_step = optimizer.iterations.numpy()
 453     checkpoint_name = 'ctl_step_{step}.ckpt'
 454 
 455     logs = {}
 456     callback_list.on_train_begin()
 457     while current_step < total_training_steps and not model.stop_training:
 458       if current_step % steps_per_epoch == 0:
 459         callback_list.on_epoch_begin(
 460             int(current_step / steps_per_epoch) + 1)
 461 
 462       # Training loss/metric are taking average over steps inside micro
 463       # training loop. We reset the their values before each round.
 464       train_loss_metric.reset_states()
 465       for metric in train_metrics + model.metrics:
 466         metric.reset_states()
 467 
 468       callback_list.on_batch_begin(current_step)
 469       # Runs several steps in the host while loop.
 470       steps = steps_to_run(current_step, steps_between_evals, steps_per_loop)
 471 
 472       if tf.config.list_physical_devices('GPU'):
 473         # TODO(zongweiz): merge with train_steps once tf.while_loop
 474         # GPU performance bugs are fixed.
 475         for _ in range(steps):
 476           train_single_step(train_iterator)
 477       else:
 478         # Converts steps to a Tensor to avoid tf.function retracing.
 479         train_steps(train_iterator, tf.convert_to_tensor(steps, dtype=tf.int32))
 480       train_loss = _float_metric_value(train_loss_metric)
 481       current_step += steps
 482 
 483       # Updates training logging.
 484       training_status = 'Train Step: %d/%d  / loss = %s' % (
 485           current_step, total_training_steps, train_loss)
 486 
 487       if current_step >= last_summary_step + train_summary_interval:
 488         summary_writer = train_summary_writer
 489         last_summary_step = current_step
 490       else:
 491         summary_writer = tf.summary.create_noop_writer()
 492 
 493       with summary_writer.as_default():
 494         if callable(optimizer.learning_rate):
 495           tf.summary.scalar(
 496               'learning_rate',
 497               optimizer.learning_rate(current_step),
 498               step=current_step)
 499         tf.summary.scalar(train_loss_metric.name, train_loss, step=current_step)
 500         for metric in train_metrics + model.metrics:
 501           metric_value = _float_metric_value(metric)
 502           training_status += '  %s = %f' % (metric.name, metric_value)
 503           tf.summary.scalar(metric.name, metric_value, step=current_step)
 504         summary_writer.flush()
 505       logging.info(training_status)
 506 
 507       # If no need for evaluation, we only call on_batch_end with train_loss,
 508       # this is to ensure we get granular global_step/sec on Tensorboard.
 509       if current_step % steps_between_evals:
 510         callback_list.on_batch_end(current_step - 1, {'loss': train_loss})
 511       else:
 512         # Save a submodel with the step in the file name after each epoch.
 513         if sub_model_export_name:
 514           _save_checkpoint(
 515               strategy, sub_model_checkpoint, model_dir,
 516               '%s_step_%d.ckpt' % (sub_model_export_name, current_step))
 517 
 518         # Save model checkpoints and run validation steps after each epoch
 519         # (with the exception of the final epoch which is handled after the
 520         # training loop).
 521         if current_step < total_training_steps:
 522           _save_checkpoint(strategy, checkpoint, model_dir,
 523                            checkpoint_name.format(step=current_step))
 524           if eval_input_fn:
 525             logging.info('Running evaluation after step: %s.', current_step)
 526             logs = _run_evaluation(current_step,
 527                                    _get_input_iterator(eval_input_fn, strategy))
 528             # Re-initialize evaluation metric.
 529             eval_loss_metric.reset_states()
 530             for metric in eval_metrics + model.metrics:
 531               metric.reset_states()
 532         # We add train_loss here rather than call on_batch_end twice to make
 533         # sure that no duplicated values are generated.
 534         logs['loss'] = train_loss
 535         callback_list.on_batch_end(current_step - 1, logs)
 536 
 537       # Calls on_epoch_end after each real epoch ends to prevent mis-calculation
 538       # of training steps.
 539       if current_step % steps_per_epoch == 0:
 540         callback_list.on_epoch_end(int(current_step / steps_per_epoch), logs)
 541 
 542     if sub_model_export_name:
 543       _save_checkpoint(strategy, sub_model_checkpoint, model_dir,
 544                        '%s.ckpt' % sub_model_export_name)
 545 
 546     _save_checkpoint(strategy, checkpoint, model_dir,
 547                      checkpoint_name.format(step=current_step))
 548     if eval_input_fn:
 549       logging.info('Running final evaluation after training is complete.')
 550       logs = _run_evaluation(current_step,
 551                              _get_input_iterator(eval_input_fn, strategy))
 552     callback_list.on_epoch_end(int(current_step / steps_per_epoch), logs)
 553     training_summary = {
 554         'total_training_steps': total_training_steps,
 555         'train_loss': _float_metric_value(train_loss_metric),
 556     }
 557     for metric in model.metrics:
 558       training_summary[metric.name] = _float_metric_value(metric)
 559     if eval_metrics:
 560       # TODO(hongkuny): Cleans up summary reporting in text.
 561       training_summary['last_train_metrics'] = _float_metric_value(
 562           train_metrics[0])
 563       training_summary['eval_metrics'] = _float_metric_value(eval_metrics[0])
 564 
 565     write_txt_summary(training_summary, summary_dir)
 566 
 567     if not _should_export_summary(strategy):
 568       tf.io.gfile.rmtree(summary_dir)
 569 
 570     callback_list.on_train_end()
 571 
 572     return model
