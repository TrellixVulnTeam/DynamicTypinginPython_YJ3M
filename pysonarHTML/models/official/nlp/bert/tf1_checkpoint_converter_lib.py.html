  67 def _bert_name_replacement(var_name, name_replacements):
  68   """Gets the variable name replacement."""
  69   for src_pattern, tgt_pattern in name_replacements:
  70     if src_pattern in var_name:
  71       old_var_name = var_name
  72       var_name = var_name.replace(src_pattern, tgt_pattern)
  73       tf.logging.info("Converted: %s --> %s", old_var_name, var_name)
  74   return var_name
  75

  77 def _has_exclude_patterns(name, exclude_patterns):
  78   """Checks if a string contains substrings that match patterns to exclude."""
  79   for p in exclude_patterns:
  80     if p in name:
  81       return True
  82   return False
  83

  85 def _get_permutation(name, permutations):
  86   """Checks whether a variable requires transposition by pattern matching."""
  87   for src_pattern, permutation in permutations:
  88     if src_pattern in name:
  89       tf.logging.info("Permuted: %s --> %s", name, permutation)
  90       return permutation
  91 
  92   return None
  93

  95 def _get_new_shape(name, shape, num_heads):
  96   """Checks whether a variable requires reshape by pattern matching."""
  97   if "self_attention/attention_output/kernel" in name:
  98     return tuple([num_heads, shape[0] // num_heads, shape[1]])
  99   if "self_attention/attention_output/bias" in name:
 100     return shape
 101 
 102   patterns = [
 103       "self_attention/query", "self_attention/value", "self_attention/key"
 104   ]
 105   for pattern in patterns:
 106     if pattern in name:
 107       if "kernel" in name:
 108         return tuple([shape[0], num_heads, shape[1] // num_heads])
 109       if "bias" in name:
 110         return tuple([num_heads, shape[0] // num_heads])
 111   return None

 114 def create_v2_checkpoint(model, src_checkpoint, output_path):
 115   """Converts a name-based matched TF V1 checkpoint to TF V2 checkpoint."""
 116   # Uses streaming-restore in eager model to read V1 name-based checkpoints.
 117   model.load_weights(src_checkpoint).assert_existing_objects_matched()
 118   checkpoint = tf.train.Checkpoint(model=model)
 119   checkpoint.save(output_path)

 122 def convert(checkpoint_from_path,
 123             checkpoint_to_path,
 124             num_heads,
 125             name_replacements,
 126             permutations,
 127             exclude_patterns=None):
 128   """Migrates the names of variables within a checkpoint.
 129 
 130   Args:
 131     checkpoint_from_path: Path to source checkpoint to be read in.
 132     checkpoint_to_path: Path to checkpoint to be written out.
 133     num_heads: The number of heads of the model.
 134     name_replacements: A list of tuples of the form (match_str, replace_str)
 135       describing variable names to adjust.
 136     permutations: A list of tuples of the form (match_str, permutation)
 137       describing permutations to apply to given variables. Note that match_str
 138       should match the original variable name, not the replaced one.
 139     exclude_patterns: A list of string patterns to exclude variables from
 140       checkpoint conversion.
 141 
 142   Returns:
 143     A dictionary that maps the new variable names to the Variable objects.
 144     A dictionary that maps the old variable names to the new variable names.
 145   """
 146   with tf.Graph().as_default():
 147     tf.logging.info("Reading checkpoint_from_path %s", checkpoint_from_path)
 148     reader = tf.train.NewCheckpointReader(checkpoint_from_path)
 149     name_shape_map = reader.get_variable_to_shape_map()
 150     new_variable_map = {}
 151     conversion_map = {}
 152     for var_name in name_shape_map:
 153       if exclude_patterns and _has_exclude_patterns(var_name, exclude_patterns):
 154         continue
 155       # Get the original tensor data.
 156       tensor = reader.get_tensor(var_name)
 157 
 158       # Look up the new variable name, if any.
 159       new_var_name = _bert_name_replacement(var_name, name_replacements)
 160 
 161       # See if we need to reshape the underlying tensor.
 162       new_shape = None
 163       if num_heads > 0:
 164         new_shape = _get_new_shape(new_var_name, tensor.shape, num_heads)
 165       if new_shape:
 166         tf.logging.info("Veriable %s has a shape change from %s to %s",
 167 
 168                         var_name, tensor.shape, new_shape)
 169         tensor = np.reshape(tensor, new_shape)
 170 
 171       # See if we need to permute the underlying tensor.
 172       permutation = _get_permutation(var_name, permutations)
 173       if permutation:
 174         tensor = np.transpose(tensor, permutation)
 175 
 176       # Create a new variable with the possibly-reshaped or transposed tensor.
 177       var = tf.Variable(tensor, name=var_name)
 178 
 179       # Save the variable into the new variable map.
 180       new_variable_map[new_var_name] = var
 181 
 182       # Keep a list of converter variables for sanity checking.
 183       if new_var_name != var_name:
 184         conversion_map[var_name] = new_var_name
 185 
 186     saver = tf.train.Saver(new_variable_map)
 187 
 188     with tf.Session() as sess:
 189       sess.run(tf.global_variables_initializer())
 190       tf.logging.info("Writing checkpoint_to_path %s", checkpoint_to_path)
 191       saver.save(sess, checkpoint_to_path, write_meta_graph=False)
 192 
 193   tf.logging.info("Summary:")
 194   tf.logging.info("  Converted %d variable name(s).", len(new_variable_map))
 195   tf.logging.info("  Converted: %s", str(conversion_map))
