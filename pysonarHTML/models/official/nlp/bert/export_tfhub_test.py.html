  15 """Tests official.nlp.bert.export_tfhub."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import os
  22 
  23 import numpy as np
  24 
  25 import tensorflow as tf
  26 import tensorflow_hub as hub
  27 from official.nlp.bert import configs
  28 from official.nlp.bert import export_tfhub
  29 
  30 
  31 class ExportTfhubTest(tf.test.TestCase):
  32 
  33   def test_export_tfhub(self):
  34     # Exports a savedmodel for TF-Hub
  35     hidden_size = 16
  36     bert_config = configs.BertConfig(
  37         vocab_size=100,
  38         hidden_size=hidden_size,
  39         intermediate_size=32,
  40         max_position_embeddings=128,
  41         num_attention_heads=2,
  42         num_hidden_layers=1)
  43     bert_model, encoder = export_tfhub.create_bert_model(bert_config)
  44     model_checkpoint_dir = os.path.join(self.get_temp_dir(), "checkpoint")
  45     checkpoint = tf.train.Checkpoint(model=encoder)
  46     checkpoint.save(os.path.join(model_checkpoint_dir, "test"))
  47     model_checkpoint_path = tf.train.latest_checkpoint(model_checkpoint_dir)
  48 
  49     vocab_file = os.path.join(self.get_temp_dir(), "uncased_vocab.txt")
  50     with tf.io.gfile.GFile(vocab_file, "w") as f:
  51       f.write("dummy content")
  52 
  53     hub_destination = os.path.join(self.get_temp_dir(), "hub")
  54     export_tfhub.export_bert_tfhub(bert_config, model_checkpoint_path,
  55                                    hub_destination, vocab_file)
  56 
  57     # Restores a hub KerasLayer.
  58     hub_layer = hub.KerasLayer(hub_destination, trainable=True)
  59 
  60     if hasattr(hub_layer, "resolved_object"):
  61       # Checks meta attributes.
  62       self.assertTrue(hub_layer.resolved_object.do_lower_case.numpy())
  63       with tf.io.gfile.GFile(
  64           hub_layer.resolved_object.vocab_file.asset_path.numpy()) as f:
  65         self.assertEqual("dummy content", f.read())
  66     # Checks the hub KerasLayer.
  67     for source_weight, hub_weight in zip(bert_model.trainable_weights,
  68                                          hub_layer.trainable_weights):
  69       self.assertAllClose(source_weight.numpy(), hub_weight.numpy())
  70 
  71     seq_length = 10
  72     dummy_ids = np.zeros((2, seq_length), dtype=np.int32)
  73     hub_outputs = hub_layer([dummy_ids, dummy_ids, dummy_ids])
  74     source_outputs = bert_model([dummy_ids, dummy_ids, dummy_ids])
  75 
  76     # The outputs of hub module are "pooled_output" and "sequence_output",
  77     # while the outputs of encoder is in reversed order, i.e.,
  78     # "sequence_output" and "pooled_output".
  79     encoder_outputs = reversed(encoder([dummy_ids, dummy_ids, dummy_ids]))
  80     self.assertEqual(hub_outputs[0].shape, (2, hidden_size))
  81     self.assertEqual(hub_outputs[1].shape, (2, seq_length, hidden_size))
  82     for source_output, hub_output, encoder_output in zip(
  83         source_outputs, hub_outputs, encoder_outputs):
  84       self.assertAllClose(source_output.numpy(), hub_output.numpy())
  85       self.assertAllClose(source_output.numpy(), encoder_output.numpy())
  86 
  87     # Test that training=True makes a difference (activates dropout).
  88     def _dropout_mean_stddev(training, num_runs=20):
  89       input_ids = np.array([[14, 12, 42, 95, 99]], np.int32)
  90       inputs = [input_ids, np.ones_like(input_ids), np.zeros_like(input_ids)]
  91       outputs = np.concatenate(
  92           [hub_layer(inputs, training=training)[0] for _ in range(num_runs)])
  93       return np.mean(np.std(outputs, axis=0))
  94     self.assertLess(_dropout_mean_stddev(training=False), 1e-6)
  95     self.assertGreater(_dropout_mean_stddev(training=True), 1e-3)
  96 
  97     # Test propagation of seq_length in shape inference.
  98     input_word_ids = tf.keras.layers.Input(shape=(seq_length,), dtype=tf.int32)
  99     input_mask = tf.keras.layers.Input(shape=(seq_length,), dtype=tf.int32)
 100     input_type_ids = tf.keras.layers.Input(shape=(seq_length,), dtype=tf.int32)
 101     pooled_output, sequence_output = hub_layer(
 102         [input_word_ids, input_mask, input_type_ids])
 103     self.assertEqual(pooled_output.shape.as_list(), [None, hidden_size])
 104     self.assertEqual(sequence_output.shape.as_list(),
 105                      [None, seq_length, hidden_size])
