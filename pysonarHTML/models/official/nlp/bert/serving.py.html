  16 """Examples of SavedModel export for tf-serving."""

  18 from absl import app
  19 from absl import flags
  20 import tensorflow as tf
  21 
  22 from official.nlp.bert import bert_models
  23 from official.nlp.bert import configs
  24 
  25 flags.DEFINE_integer("sequence_length", None,
  26                      "Sequence length to parse the tf.Example. If "
  27                      "sequence_length > 0, add a signature for serialized "
  28                      "tf.Example and define the parsing specification by the "
  29                      "sequence_length.")
  30 flags.DEFINE_string("bert_config_file", None,
  31                     "Bert configuration file to define core bert layers.")
  32 flags.DEFINE_string("model_checkpoint_path", None,
  33                     "File path to TF model checkpoint.")
  34 flags.DEFINE_string("export_path", None,
  35                     "Destination folder to export the serving SavedModel.")
  36 
  37 FLAGS = flags.FLAGS
  38 
  39 
  40 class BertServing(tf.keras.Model):
  41   """Bert transformer encoder model for serving."""
  42 
  43   def __init__(self, bert_config, name_to_features=None, name="serving_model"):
  44     super(BertServing, self).__init__(name=name)
  45     self.encoder = bert_models.get_transformer_encoder(
  46         bert_config, sequence_length=None)
  47     self.name_to_features = name_to_features
  48 
  49   def call(self, inputs):
  50     input_word_ids = inputs["input_ids"]
  51     input_mask = inputs["input_mask"]
  52     input_type_ids = inputs["segment_ids"]
  53 
  54     encoder_outputs, _ = self.encoder(
  55         [input_word_ids, input_mask, input_type_ids])
  56     return encoder_outputs
  57 
  58   def serve_body(self, input_ids, input_mask=None, segment_ids=None):
  59     if segment_ids is None:
  60       # Requires CLS token is the first token of inputs.
  61       segment_ids = tf.zeros_like(input_ids)
  62     if input_mask is None:
  63       # The mask has 1 for real tokens and 0 for padding tokens.
  64       input_mask = tf.where(
  65           tf.equal(input_ids, 0), tf.zeros_like(input_ids),
  66           tf.ones_like(input_ids))
  67 
  68     inputs = dict(
  69         input_ids=input_ids, input_mask=input_mask, segment_ids=segment_ids)
  70     return self.call(inputs)
  71 
  72   @tf.function
  73   def serve(self, input_ids, input_mask=None, segment_ids=None):
  74     outputs = self.serve_body(input_ids, input_mask, segment_ids)
  75     # Returns a dictionary to control SignatureDef output signature.
  76     return {"outputs": outputs[-1]}
  77 
  78   @tf.function
  79   def serve_examples(self, inputs):
  80     features = tf.io.parse_example(inputs, self.name_to_features)
  81     for key in list(features.keys()):
  82       t = features[key]
  83       if t.dtype == tf.int64:
  84         t = tf.cast(t, tf.int32)
  85       features[key] = t
  86     return self.serve(
  87         features["input_ids"],
  88         input_mask=features["input_mask"] if "input_mask" in features else None,
  89         segment_ids=features["segment_ids"]
  90         if "segment_ids" in features else None)
  91 
  92   @classmethod
  93   def export(cls, model, export_dir):
  94     if not isinstance(model, cls):
  95       raise ValueError("Invalid model instance: %s, it should be a %s" %
  96                        (model, cls))
  97 
  98     signatures = {
  99         "serving_default":
 100             model.serve.get_concrete_function(
 101                 input_ids=tf.TensorSpec(
 102                     shape=[None, None], dtype=tf.int32, name="inputs")),
 103     }
 104     if model.name_to_features:
 105       signatures[
 106           "serving_examples"] = model.serve_examples.get_concrete_function(
 107               tf.TensorSpec(shape=[None], dtype=tf.string, name="examples"))
 108     tf.saved_model.save(model, export_dir=export_dir, signatures=signatures)
 109 
 110 
 111 def main(_):
 112   sequence_length = FLAGS.sequence_length
 113   if sequence_length is not None and sequence_length > 0:
 114     name_to_features = {
 115         "input_ids": tf.io.FixedLenFeature([sequence_length], tf.int64),
 116         "input_mask": tf.io.FixedLenFeature([sequence_length], tf.int64),
 117         "segment_ids": tf.io.FixedLenFeature([sequence_length], tf.int64),
 118     }
 119   else:
 120     name_to_features = None
 121   bert_config = configs.BertConfig.from_json_file(FLAGS.bert_config_file)
 122   serving_model = BertServing(
 123       bert_config=bert_config, name_to_features=name_to_features)
 124   checkpoint = tf.train.Checkpoint(model=serving_model.encoder)
 125   checkpoint.restore(FLAGS.model_checkpoint_path
 126                     ).assert_existing_objects_matched().run_restore_ops()
 127   BertServing.export(serving_model, FLAGS.export_path)
 128 
 129 
 130 if __name__ == "__main__":
 131   flags.mark_flag_as_required("bert_config_file")
 132   flags.mark_flag_as_required("model_checkpoint_path")
 133   flags.mark_flag_as_required("export_path")
 134   app.run(main)
