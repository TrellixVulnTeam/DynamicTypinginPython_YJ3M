  61 def get_pretrain_dataset_fn(input_file_pattern, seq_length,
  62                             max_predictions_per_seq, global_batch_size,
  63                             use_next_sentence_label=True):
  64   """Returns input dataset from input file string."""
  65   def _dataset_fn(ctx=None):
  66     """Returns tf.data.Dataset for distributed BERT pretraining."""
  67     input_patterns = input_file_pattern.split(',')
  68     batch_size = ctx.get_per_replica_batch_size(global_batch_size)
  69     train_dataset = input_pipeline.create_pretrain_dataset(
  70         input_patterns,
  71         seq_length,
  72         max_predictions_per_seq,
  73         batch_size,
  74         is_training=True,
  75         input_pipeline_context=ctx,
  76         use_next_sentence_label=use_next_sentence_label)
  77     return train_dataset
  78 
  79   return _dataset_fn

  82 def get_loss_fn():
  83   """Returns loss function for BERT pretraining."""
  84 
  85   def _bert_pretrain_loss_fn(unused_labels, losses, **unused_args):
  86     return tf.reduce_mean(losses)
  87 
  88   return _bert_pretrain_loss_fn

  91 def run_customized_training(strategy,
  92                             bert_config,
  93                             init_checkpoint,
  94                             max_seq_length,
  95                             max_predictions_per_seq,
  96                             model_dir,
  97                             steps_per_epoch,
  98                             steps_per_loop,
  99                             epochs,
 100                             initial_lr,
 101                             warmup_steps,
 102                             end_lr,
 103                             optimizer_type,
 104                             input_files,
 105                             train_batch_size,
 106                             use_next_sentence_label=True,
 107                             train_summary_interval=0,
 108                             custom_callbacks=None):
 109   """Run BERT pretrain model training using low-level API."""
 110 
 111   train_input_fn = get_pretrain_dataset_fn(input_files, max_seq_length,
 112                                            max_predictions_per_seq,
 113                                            train_batch_size,
 114                                            use_next_sentence_label)
 115 
 116   def _get_pretrain_model():
 117     """Gets a pretraining model."""
 118     pretrain_model, core_model = bert_models.pretrain_model(
 119         bert_config, max_seq_length, max_predictions_per_seq,
 120         use_next_sentence_label=use_next_sentence_label)
 121     optimizer = optimization.create_optimizer(
 122         initial_lr, steps_per_epoch * epochs, warmup_steps,
 123         end_lr, optimizer_type)
 124     pretrain_model.optimizer = performance.configure_optimizer(
 125         optimizer,
 126         use_float16=common_flags.use_float16(),
 127         use_graph_rewrite=common_flags.use_graph_rewrite())
 128     return pretrain_model, core_model
 129 
 130   trained_model = model_training_utils.run_customized_training_loop(
 131       strategy=strategy,
 132       model_fn=_get_pretrain_model,
 133       loss_fn=get_loss_fn(),
 134       scale_loss=FLAGS.scale_loss,
 135       model_dir=model_dir,
 136       init_checkpoint=init_checkpoint,
 137       train_input_fn=train_input_fn,
 138       steps_per_epoch=steps_per_epoch,
 139       steps_per_loop=steps_per_loop,
 140       epochs=epochs,
 141       sub_model_export_name='pretrained/bert_model',
 142       train_summary_interval=train_summary_interval,
 143       custom_callbacks=custom_callbacks)
 144 
 145   return trained_model

 148 def run_bert_pretrain(strategy, custom_callbacks=None):
 149   """Runs BERT pre-training."""
 150 
 151   bert_config = configs.BertConfig.from_json_file(FLAGS.bert_config_file)
 152   if not strategy:
 153     raise ValueError('Distribution strategy is not specified.')
 154 
 155   # Runs customized training loop.
 156   logging.info('Training using customized training loop TF 2.0 with distributed'
 157                'strategy.')
 158 
 159   performance.set_mixed_precision_policy(common_flags.dtype())
 160 
 161   return run_customized_training(
 162       strategy,
 163       bert_config,
 164       FLAGS.init_checkpoint,  # Used to initialize only the BERT submodel.
 165       FLAGS.max_seq_length,
 166       FLAGS.max_predictions_per_seq,
 167       FLAGS.model_dir,
 168       FLAGS.num_steps_per_epoch,
 169       FLAGS.steps_per_loop,
 170       FLAGS.num_train_epochs,
 171       FLAGS.learning_rate,
 172       FLAGS.warmup_steps,
 173       FLAGS.end_lr,
 174       FLAGS.optimizer_type,
 175       FLAGS.input_files,
 176       FLAGS.train_batch_size,
 177       FLAGS.use_next_sentence_label,
 178       FLAGS.train_summary_interval,
 179       custom_callbacks=custom_callbacks)

 182 def main(_):
 183   gin.parse_config_files_and_bindings(FLAGS.gin_file, FLAGS.gin_param)
 184   if not FLAGS.model_dir:
 185     FLAGS.model_dir = '/tmp/bert20/'
 186   strategy = distribution_utils.get_distribution_strategy(
 187       distribution_strategy=FLAGS.distribution_strategy,
 188       num_gpus=FLAGS.num_gpus,
 189       tpu_address=FLAGS.tpu)
 190   if strategy:
 191     print('***** Number of cores used : ', strategy.num_replicas_in_sync)
 192 
 193   run_bert_pretrain(strategy)
