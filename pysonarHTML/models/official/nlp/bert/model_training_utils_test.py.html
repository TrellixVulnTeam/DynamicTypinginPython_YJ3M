  59 def create_fake_data_input_fn(batch_size, features_shape, num_classes):
  60   """Creates a dummy input function with the given feature and label shapes.
  61 
  62   Args:
  63     batch_size: integer.
  64     features_shape: list[int]. Feature shape for an individual example.
  65     num_classes: integer. Number of labels.
  66 
  67   Returns:
  68     An input function that is usable in the executor.
  69   """
  70 
  71   def _dataset_fn(input_context=None):
  72     """An input function for generating fake data."""
  73     local_batch_size = input_context.get_per_replica_batch_size(batch_size)
  74     features = np.random.rand(64, *features_shape)
  75     labels = np.random.randint(2, size=[64, num_classes])
  76     # Convert the inputs to a Dataset.
  77     dataset = tf.data.Dataset.from_tensor_slices((features, labels))
  78     dataset = dataset.shard(input_context.num_input_pipelines,
  79                             input_context.input_pipeline_id)
  80 
  81     def _assign_dtype(features, labels):
  82       features = tf.cast(features, tf.float32)
  83       labels = tf.cast(labels, tf.float32)
  84       return features, labels
  85 
  86     # Shuffle, repeat, and batch the examples.
  87     dataset = dataset.map(_assign_dtype)
  88     dataset = dataset.shuffle(64).repeat()
  89     dataset = dataset.batch(local_batch_size, drop_remainder=True)
  90     dataset = dataset.prefetch(buffer_size=64)
  91     return dataset
  92 
  93   return _dataset_fn

  96 def create_model_fn(input_shape, num_classes, use_float16=False):
  97 
  98   def _model_fn():
  99     """A one-layer softmax model suitable for testing."""
 100     input_layer = tf.keras.layers.Input(shape=input_shape)
 101     x = tf.keras.layers.Dense(num_classes, activation='relu')(input_layer)
 102     output_layer = tf.keras.layers.Dense(num_classes, activation='softmax')(x)
 103     sub_model = tf.keras.models.Model(input_layer, x, name='sub_model')
 104     model = tf.keras.models.Model(input_layer, output_layer, name='model')
 105     model.add_metric(
 106         tf.reduce_mean(input_layer), name='mean_input', aggregation='mean')
 107     model.optimizer = tf.keras.optimizers.SGD(learning_rate=0.1, momentum=0.9)
 108     if use_float16:
 109       model.optimizer = (
 110           tf.keras.mixed_precision.experimental.LossScaleOptimizer(
 111               model.optimizer, loss_scale='dynamic'))
 112     return model, sub_model
 113 
 114   return _model_fn

 122 def summaries_with_matching_keyword(keyword, summary_dir):
 123   """Yields summary protos matching given keyword from event file."""
 124   event_paths = tf.io.gfile.glob(os.path.join(summary_dir, 'events*'))
 125   for event in tf.compat.v1.train.summary_iterator(event_paths[-1]):
 126     if event.summary is not None:
 127       for value in event.summary.value:
 128         if keyword in value.tag:
 129           logging.error(event)
 130           yield event.summary
 131 
 132 
 133 def check_eventfile_for_keyword(keyword, summary_dir):
 134   """Checks event files for the keyword."""
 135   return any(summaries_with_matching_keyword(keyword, summary_dir))

 138 class RecordingCallback(tf.keras.callbacks.Callback):
 139 
 140   def __init__(self):
 141     self.batch_begin = []  # (batch, logs)
 142     self.batch_end = []    # (batch, logs)
 143     self.epoch_begin = []  # (epoch, logs)
 144     self.epoch_end = []    # (epoch, logs)
 145 
 146   def on_batch_begin(self, batch, logs=None):
 147     self.batch_begin.append((batch, logs))
 148 
 149   def on_batch_end(self, batch, logs=None):
 150     self.batch_end.append((batch, logs))
 151 
 152   def on_epoch_begin(self, epoch, logs=None):
 153     self.epoch_begin.append((epoch, logs))
 154 
 155   def on_epoch_end(self, epoch, logs=None):
 156     self.epoch_end.append((epoch, logs))

 163     self._model_fn = create_model_fn(input_shape=[128], num_classes=3)

 165   def run_training(self, strategy, model_dir, steps_per_loop, run_eagerly):
 166     input_fn = create_fake_data_input_fn(

 169         strategy=strategy,
 170         model_fn=self._model_fn,
 171         loss_fn=tf.keras.losses.categorical_crossentropy,
 172         model_dir=model_dir,
 173         steps_per_epoch=20,
 174         steps_per_loop=steps_per_loop,
 175         epochs=2,
 176         train_input_fn=input_fn,
 177         eval_input_fn=input_fn,
 178         eval_steps=10,
 179         init_checkpoint=None,
 180         sub_model_export_name='my_submodel_name',
 181         metric_fn=metric_fn,
 182         custom_callbacks=None,
 183         run_eagerly=run_eagerly)

 185   @combinations.generate(eager_strategy_combinations())
 186   def test_train_eager_single_step(self, distribution):
 187     model_dir = self.get_temp_dir()

 189       with self.assertRaises(ValueError):
 190         self.run_training(
 191             distribution, model_dir, steps_per_loop=1, run_eagerly=True)
 192     else:
 193       self.run_training(
 194           distribution, model_dir, steps_per_loop=1, run_eagerly=True)

 206   @combinations.generate(eager_strategy_combinations())
 207   def test_train_check_artifacts(self, distribution):
 208     model_dir = self.get_temp_dir()
 209     self.run_training(
 210         distribution, model_dir, steps_per_loop=10, run_eagerly=False)
 211 
 212     # Two checkpoints should be saved after two epochs.
 213     files = map(os.path.basename,
 214                 tf.io.gfile.glob(os.path.join(model_dir, 'ctl_step_*index')))
 215     self.assertCountEqual(['ctl_step_20.ckpt-1.index',
 216                            'ctl_step_40.ckpt-2.index'], files)
 217 
 218     # Three submodel checkpoints should be saved after two epochs (one after
 219     # each epoch plus one final).
 220     files = map(os.path.basename,
 221                 tf.io.gfile.glob(os.path.join(model_dir,
 222                                               'my_submodel_name*index')))
 223     self.assertCountEqual(['my_submodel_name.ckpt-3.index',
 224                            'my_submodel_name_step_20.ckpt-1.index',
 225                            'my_submodel_name_step_40.ckpt-2.index'], files)
 226 
 227     self.assertNotEmpty(
 228         tf.io.gfile.glob(
 229             os.path.join(model_dir, 'summaries/training_summary*')))
 230 
 231     # Loss and accuracy values should be written into summaries.
 232     self.assertTrue(
 233         check_eventfile_for_keyword('loss',
 234                                     os.path.join(model_dir, 'summaries/train')))
 235     self.assertTrue(
 236         check_eventfile_for_keyword('accuracy',
 237                                     os.path.join(model_dir, 'summaries/train')))
 238     self.assertTrue(
 239         check_eventfile_for_keyword('mean_input',
 240                                     os.path.join(model_dir, 'summaries/train')))
 241     self.assertTrue(
 242         check_eventfile_for_keyword('accuracy',
 243                                     os.path.join(model_dir, 'summaries/eval')))
 244     self.assertTrue(
 245         check_eventfile_for_keyword('mean_input',
 246                                     os.path.join(model_dir, 'summaries/eval')))

 248   @combinations.generate(eager_strategy_combinations())
 249   def test_train_check_callbacks(self, distribution):
 250     model_dir = self.get_temp_dir()
 251     callback = RecordingCallback()
 252     callbacks = [callback]
 253     input_fn = create_fake_data_input_fn(
 254         batch_size=8, features_shape=[128], num_classes=3)
 255     model_training_utils.run_customized_training_loop(
 256         strategy=distribution,
 257         model_fn=self._model_fn,
 258         loss_fn=tf.keras.losses.categorical_crossentropy,
 259         model_dir=model_dir,
 260         steps_per_epoch=20,
 261         num_eval_per_epoch=4,
 262         steps_per_loop=10,
 263         epochs=2,
 264         train_input_fn=input_fn,
 265         eval_input_fn=input_fn,
 266         eval_steps=10,
 267         init_checkpoint=None,
 268         metric_fn=metric_fn,
 269         custom_callbacks=callbacks,
 270         run_eagerly=False)
 271     self.assertEqual(callback.epoch_begin, [(1, {}), (2, {})])
 272     epoch_ends, epoch_end_infos = zip(*callback.epoch_end)
 273     self.assertEqual(list(epoch_ends), [1, 2, 2])
 274     for info in epoch_end_infos:
 275       self.assertIn('accuracy', info)
 276 
 277     self.assertEqual(callback.batch_begin, [(0, {}), (5, {}), (10, {}),
 278                                             (15, {}), (20, {}), (25, {}),
 279                                             (30, {}), (35, {})])
 280     batch_ends, batch_end_infos = zip(*callback.batch_end)
 281     self.assertEqual(list(batch_ends), [4, 9, 14, 19, 24, 29, 34, 39])
 282     for info in batch_end_infos:
 283       self.assertIn('loss', info)

 285   @combinations.generate(
 286       combinations.combine(
 287           distribution=[
 288               strategy_combinations.one_device_strategy_gpu,
 289           ],
 290           mode='eager',
 291       ))
 292   def test_train_check_artifacts_non_chief(self, distribution):
 293     # We shouldn't export artifacts on non-chief workers. Since there's no easy
 294     # way to test with real MultiWorkerMirroredStrategy, we patch the strategy
 295     # to make it as if it's MultiWorkerMirroredStrategy on non-chief workers.
 296     extended = distribution.extended
 297     with mock.patch.object(extended.__class__, 'should_checkpoint',
 298                            new_callable=mock.PropertyMock, return_value=False), \
 299          mock.patch.object(extended.__class__, 'should_save_summary',
 300                            new_callable=mock.PropertyMock, return_value=False):
 301       model_dir = self.get_temp_dir()
 302       self.run_training(
 303           distribution, model_dir, steps_per_loop=10, run_eagerly=False)
 304       self.assertEmpty(tf.io.gfile.listdir(model_dir))
