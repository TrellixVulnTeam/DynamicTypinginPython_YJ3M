  14 """Evaluation script for SQuAD version 2.0.
  15 
  16 The functions are copied and modified from
  17 https://raw.githubusercontent.com/white127/SQUAD-2.0-bidaf/master/evaluate-v2.0.py
  18 
  19 In addition to basic functionality, we also compute additional statistics and
  20 plot precision-recall curves if an additional na_prob.json file is provided.
  21 This file is expected to map question ID's to the model's predicted probability
  22 that a question is unanswerable.

  36 def _make_qid_to_has_ans(dataset):
  37   qid_to_has_ans = {}
  38   for article in dataset:
  39     for p in article['paragraphs']:
  40       for qa in p['qas']:
  41         qid_to_has_ans[qa['id']] = bool(qa['answers'])
  42   return qid_to_has_ans

  45 def _normalize_answer(s):
  46   """Lower text and remove punctuation, articles and extra whitespace."""
  47   def remove_articles(text):
  48     regex = re.compile(r'\b(a|an|the)\b', re.UNICODE)
  49     return re.sub(regex, ' ', text)
  50   def white_space_fix(text):
  51     return ' '.join(text.split())
  52   def remove_punc(text):
  53     exclude = set(string.punctuation)
  54     return ''.join(ch for ch in text if ch not in exclude)
  55   def lower(text):
  56     return text.lower()
  57   return white_space_fix(remove_articles(remove_punc(lower(s))))

  60 def _get_tokens(s):
  61   if not s: return []
  62   return _normalize_answer(s).split()
  63 
  64 
  65 def _compute_exact(a_gold, a_pred):
  66   return int(_normalize_answer(a_gold) == _normalize_answer(a_pred))

  69 def _compute_f1(a_gold, a_pred):
  70   """Compute F1-score."""
  71   gold_toks = _get_tokens(a_gold)
  72   pred_toks = _get_tokens(a_pred)
  73   common = collections.Counter(gold_toks) & collections.Counter(pred_toks)
  74   num_same = sum(common.values())
  75   if not gold_toks or not pred_toks:
  76     # If either is no-answer, then F1 is 1 if they agree, 0 otherwise
  77     return int(gold_toks == pred_toks)
  78   if num_same == 0:
  79     return 0
  80   precision = 1.0 * num_same / len(pred_toks)
  81   recall = 1.0 * num_same / len(gold_toks)
  82   f1 = (2 * precision * recall) / (precision + recall)
  83   return f1

  86 def _get_raw_scores(dataset, predictions):
  87   """Compute raw scores."""
  88   exact_scores = {}
  89   f1_scores = {}
  90   for article in dataset:
  91     for p in article['paragraphs']:
  92       for qa in p['qas']:
  93         qid = qa['id']
  94         gold_answers = [a['text'] for a in qa['answers']
  95                         if _normalize_answer(a['text'])]
  96         if not gold_answers:
  97           # For unanswerable questions, only correct answer is empty string
  98           gold_answers = ['']
  99         if qid not in predictions:
 100           logging.error('Missing prediction for %s', qid)
 101           continue
 102         a_pred = predictions[qid]
 103         # Take max over all gold answers
 104         exact_scores[qid] = max(_compute_exact(a, a_pred) for a in gold_answers)
 105         f1_scores[qid] = max(_compute_f1(a, a_pred) for a in gold_answers)
 106   return exact_scores, f1_scores

 109 def _apply_no_ans_threshold(
 110     scores, na_probs, qid_to_has_ans, na_prob_thresh=1.0):
 111   new_scores = {}
 112   for qid, s in scores.items():
 113     pred_na = na_probs[qid] > na_prob_thresh
 114     if pred_na:
 115       new_scores[qid] = float(not qid_to_has_ans[qid])
 116     else:
 117       new_scores[qid] = s
 118   return new_scores

 121 def _make_eval_dict(exact_scores, f1_scores, qid_list=None):
 122   """Make evaluation result dictionary."""
 123   if not qid_list:
 124     total = len(exact_scores)
 125     return collections.OrderedDict([
 126         ('exact', 100.0 * sum(exact_scores.values()) / total),
 127         ('f1', 100.0 * sum(f1_scores.values()) / total),
 128         ('total', total),
 129     ])
 130   else:
 131     total = len(qid_list)
 132     return collections.OrderedDict([
 133         ('exact', 100.0 * sum(exact_scores[k] for k in qid_list) / total),
 134         ('f1', 100.0 * sum(f1_scores[k] for k in qid_list) / total),
 135         ('total', total),
 136     ])

 139 def _merge_eval(main_eval, new_eval, prefix):
 140   for k in new_eval:
 141     main_eval['%s_%s' % (prefix, k)] = new_eval[k]

 144 def _make_precision_recall_eval(scores, na_probs, num_true_pos, qid_to_has_ans):
 145   """Make evaluation dictionary containing average recision recall."""
 146   qid_list = sorted(na_probs, key=lambda k: na_probs[k])
 147   true_pos = 0.0
 148   cur_p = 1.0
 149   cur_r = 0.0
 150   precisions = [1.0]
 151   recalls = [0.0]
 152   avg_prec = 0.0
 153   for i, qid in enumerate(qid_list):
 154     if qid_to_has_ans[qid]:
 155       true_pos += scores[qid]
 156     cur_p = true_pos / float(i+1)
 157     cur_r = true_pos / float(num_true_pos)
 158     if i == len(qid_list) - 1 or na_probs[qid] != na_probs[qid_list[i+1]]:
 159       # i.e., if we can put a threshold after this point
 160       avg_prec += cur_p * (cur_r - recalls[-1])
 161       precisions.append(cur_p)
 162       recalls.append(cur_r)
 163   return {'ap': 100.0 * avg_prec}

 166 def _run_precision_recall_analysis(
 167     main_eval, exact_raw, f1_raw, na_probs, qid_to_has_ans):
 168   """Run precision recall analysis and return result dictionary."""
 169   num_true_pos = sum(1 for v in qid_to_has_ans.values() if v)
 170   if num_true_pos == 0:
 171     return
 172   pr_exact = _make_precision_recall_eval(
 173       exact_raw, na_probs, num_true_pos, qid_to_has_ans)
 174   pr_f1 = _make_precision_recall_eval(
 175       f1_raw, na_probs, num_true_pos, qid_to_has_ans)
 176   oracle_scores = {k: float(v) for k, v in qid_to_has_ans.items()}
 177   pr_oracle = _make_precision_recall_eval(
 178       oracle_scores, na_probs, num_true_pos, qid_to_has_ans)
 179   _merge_eval(main_eval, pr_exact, 'pr_exact')
 180   _merge_eval(main_eval, pr_f1, 'pr_f1')
 181   _merge_eval(main_eval, pr_oracle, 'pr_oracle')

 184 def _find_best_thresh(predictions, scores, na_probs, qid_to_has_ans):
 185   """Find the best threshold for no answer probability."""
 186   num_no_ans = sum(1 for k in qid_to_has_ans if not qid_to_has_ans[k])
 187   cur_score = num_no_ans
 188   best_score = cur_score
 189   best_thresh = 0.0
 190   qid_list = sorted(na_probs, key=lambda k: na_probs[k])
 191   for qid in qid_list:
 192     if qid not in scores: continue
 193     if qid_to_has_ans[qid]:
 194       diff = scores[qid]
 195     else:
 196       if predictions[qid]:
 197         diff = -1
 198       else:
 199         diff = 0
 200     cur_score += diff
 201     if cur_score > best_score:
 202       best_score = cur_score
 203       best_thresh = na_probs[qid]
 204   return 100.0 * best_score / len(scores), best_thresh

 207 def _find_all_best_thresh(
 208     main_eval, predictions, exact_raw, f1_raw, na_probs, qid_to_has_ans):
 209   best_exact, exact_thresh = _find_best_thresh(
 210       predictions, exact_raw, na_probs, qid_to_has_ans)
 211   best_f1, f1_thresh = _find_best_thresh(
 212       predictions, f1_raw, na_probs, qid_to_has_ans)
 213   main_eval['final_exact'] = best_exact
 214   main_eval['final_exact_thresh'] = exact_thresh
 215   main_eval['final_f1'] = best_f1
 216   main_eval['final_f1_thresh'] = f1_thresh

 219 def evaluate(dataset, predictions, na_probs=None):
 220   """Evaluate prediction results."""
 221   new_orig_data = []
 222   for article in dataset:
 223     for p in article['paragraphs']:
 224       for qa in p['qas']:
 225         if qa['id'] in predictions:
 226           new_para = {'qas': [qa]}
 227           new_article = {'paragraphs': [new_para]}
 228           new_orig_data.append(new_article)
 229   dataset = new_orig_data
 230 
 231   if na_probs is None:
 232     na_probs = {k: 0.0 for k in predictions}
 233   qid_to_has_ans = _make_qid_to_has_ans(dataset)  # maps qid to True/False
 234   has_ans_qids = [k for k, v in qid_to_has_ans.items() if v]
 235   no_ans_qids = [k for k, v in qid_to_has_ans.items() if not v]
 236   exact_raw, f1_raw = _get_raw_scores(dataset, predictions)
 237   exact_thresh = _apply_no_ans_threshold(exact_raw, na_probs, qid_to_has_ans)
 238   f1_thresh = _apply_no_ans_threshold(f1_raw, na_probs, qid_to_has_ans)
 239   out_eval = _make_eval_dict(exact_thresh, f1_thresh)
 240   if has_ans_qids:
 241     has_ans_eval = _make_eval_dict(
 242         exact_thresh, f1_thresh, qid_list=has_ans_qids)
 243     _merge_eval(out_eval, has_ans_eval, 'HasAns')
 244   if no_ans_qids:
 245     no_ans_eval = _make_eval_dict(exact_thresh, f1_thresh, qid_list=no_ans_qids)
 246     _merge_eval(out_eval, no_ans_eval, 'NoAns')
 247 
 248   _find_all_best_thresh(
 249       out_eval, predictions, exact_raw, f1_raw, na_probs, qid_to_has_ans)
 250   _run_precision_recall_analysis(
 251       out_eval, exact_raw, f1_raw, na_probs, qid_to_has_ans)
 252   return out_eval
