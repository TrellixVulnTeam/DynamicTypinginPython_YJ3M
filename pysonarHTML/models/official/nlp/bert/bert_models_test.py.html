  15 from __future__ import absolute_import
  16 from __future__ import division
  17 from __future__ import print_function
  18 
  19 import tensorflow as tf
  20 
  21 from official.nlp.bert import bert_models
  22 from official.nlp.bert import configs as bert_configs
  23 from official.nlp.modeling import networks
  24 
  25 
  26 class BertModelsTest(tf.test.TestCase):
  27 
  28   def setUp(self):
  29     super(BertModelsTest, self).setUp()
  30     self._bert_test_config = bert_configs.BertConfig(
  31         attention_probs_dropout_prob=0.0,
  32         hidden_act='gelu',
  33         hidden_dropout_prob=0.0,
  34         hidden_size=16,
  35         initializer_range=0.02,
  36         intermediate_size=32,
  37         max_position_embeddings=128,
  38         num_attention_heads=2,
  39         num_hidden_layers=2,
  40         type_vocab_size=2,
  41         vocab_size=30522)
  42 
  43   def test_pretrain_model(self):
  44     model, encoder = bert_models.pretrain_model(
  45         self._bert_test_config,
  46         seq_length=5,
  47         max_predictions_per_seq=2,
  48         initializer=None,
  49         use_next_sentence_label=True)
  50     self.assertIsInstance(model, tf.keras.Model)
  51     self.assertIsInstance(encoder, networks.TransformerEncoder)
  52 
  53     # model has one scalar output: loss value.
  54     self.assertEqual(model.output.shape.as_list(), [None,])
  55 
  56     # Expect two output from encoder: sequence and classification output.
  57     self.assertIsInstance(encoder.output, list)
  58     self.assertLen(encoder.output, 2)
  59     # shape should be [batch size, seq_length, hidden_size]
  60     self.assertEqual(encoder.output[0].shape.as_list(), [None, 5, 16])
  61     # shape should be [batch size, hidden_size]
  62     self.assertEqual(encoder.output[1].shape.as_list(), [None, 16])
  63 
  64   def test_squad_model(self):
  65     model, core_model = bert_models.squad_model(
  66         self._bert_test_config,
  67         max_seq_length=5,
  68         initializer=None,
  69         hub_module_url=None,
  70         hub_module_trainable=None)
  71     self.assertIsInstance(model, tf.keras.Model)
  72     self.assertIsInstance(core_model, tf.keras.Model)
  73 
  74     # Expect two output from model: start positions and end positions
  75     self.assertIsInstance(model.output, list)
  76     self.assertLen(model.output, 2)
  77     # shape should be [batch size, seq_length]
  78     self.assertEqual(model.output[0].shape.as_list(), [None, 5])
  79     # shape should be [batch size, seq_length]
  80     self.assertEqual(model.output[1].shape.as_list(), [None, 5])
  81 
  82     # Expect two output from core_model: sequence and classification output.
  83     self.assertIsInstance(core_model.output, list)
  84     self.assertLen(core_model.output, 2)
  85     # shape should be [batch size, seq_length, hidden_size]
  86     self.assertEqual(core_model.output[0].shape.as_list(), [None, 5, 16])
  87     # shape should be [batch size, hidden_size]
  88     self.assertEqual(core_model.output[1].shape.as_list(), [None, 16])
  89 
  90   def test_classifier_model(self):
  91     model, core_model = bert_models.classifier_model(
  92         self._bert_test_config,
  93         num_labels=3,
  94         max_seq_length=5,
  95         final_layer_initializer=None,
  96         hub_module_url=None,
  97         hub_module_trainable=None)
  98     self.assertIsInstance(model, tf.keras.Model)
  99     self.assertIsInstance(core_model, tf.keras.Model)
 100 
 101     # model has one classification output with num_labels=3.
 102     self.assertEqual(model.output.shape.as_list(), [None, 3])
 103 
 104     # Expect two output from core_model: sequence and classification output.
 105     self.assertIsInstance(core_model.output, list)
 106     self.assertLen(core_model.output, 2)
 107     # shape should be [batch size, 1, hidden_size]
 108     self.assertEqual(core_model.output[0].shape.as_list(), [None, 1, 16])
 109     # shape should be [batch size, hidden_size]
 110     self.assertEqual(core_model.output[1].shape.as_list(), [None, 16])
 111

 113 if __name__ == '__main__':
 114   tf.test.main()
