  24 def decode_record(record, name_to_features):
  25   """Decodes a record to a TensorFlow example."""
  26   example = tf.io.parse_single_example(record, name_to_features)
  27 
  28   # tf.Example only supports tf.int64, but the TPU only supports tf.int32.
  29   # So cast all int64 to int32.
  30   for name in list(example.keys()):
  31     t = example[name]
  32     if t.dtype == tf.int64:
  33       t = tf.cast(t, tf.int32)
  34     example[name] = t
  35 
  36   return example

  39 def single_file_dataset(input_file, name_to_features):
  40   """Creates a single-file dataset to be passed for BERT custom training."""
  41   # For training, we want a lot of parallel reading and shuffling.
  42   # For eval, we want no shuffling and parallel reading doesn't matter.
  43   d = tf.data.TFRecordDataset(input_file)
  44   d = d.map(
  45       lambda record: decode_record(record, name_to_features),
  46       num_parallel_calls=tf.data.experimental.AUTOTUNE)
  47 
  48   # When `input_file` is a path to a single file or a list
  49   # containing a single path, disable auto sharding so that
  50   # same input file is sent to all workers.
  51   if isinstance(input_file, str) or len(input_file) == 1:
  52     options = tf.data.Options()
  53     options.experimental_distribute.auto_shard_policy = (
  54         tf.data.experimental.AutoShardPolicy.OFF)
  55     d = d.with_options(options)
  56   return d

  59 def create_pretrain_dataset(input_patterns,
  60                             seq_length,
  61                             max_predictions_per_seq,
  62                             batch_size,
  63                             is_training=True,
  64                             input_pipeline_context=None,
  65                             use_next_sentence_label=True,
  66                             use_position_id=False,
  67                             output_fake_labels=True):
  68   """Creates input dataset from (tf)records files for pretraining."""
  69   name_to_features = {
  70       'input_ids':
  71           tf.io.FixedLenFeature([seq_length], tf.int64),
  72       'input_mask':
  73           tf.io.FixedLenFeature([seq_length], tf.int64),
  74       'segment_ids':
  75           tf.io.FixedLenFeature([seq_length], tf.int64),
  76       'masked_lm_positions':
  77           tf.io.FixedLenFeature([max_predictions_per_seq], tf.int64),
  78       'masked_lm_ids':
  79           tf.io.FixedLenFeature([max_predictions_per_seq], tf.int64),
  80       'masked_lm_weights':
  81           tf.io.FixedLenFeature([max_predictions_per_seq], tf.float32),
  82   }
  83   if use_next_sentence_label:
  84     name_to_features['next_sentence_labels'] = tf.io.FixedLenFeature([1],
  85                                                                      tf.int64)
  86   if use_position_id:
  87     name_to_features['position_ids'] = tf.io.FixedLenFeature([seq_length],
  88                                                              tf.int64)
  89   for input_pattern in input_patterns:
  90     if not tf.io.gfile.glob(input_pattern):
  91       raise ValueError('%s does not match any files.' % input_pattern)
  92 
  93   dataset = tf.data.Dataset.list_files(input_patterns, shuffle=is_training)
  94 
  95   if input_pipeline_context and input_pipeline_context.num_input_pipelines > 1:
  96     dataset = dataset.shard(input_pipeline_context.num_input_pipelines,
  97                             input_pipeline_context.input_pipeline_id)
  98   if is_training:
  99     dataset = dataset.repeat()
 100 
 101     # We set shuffle buffer to exactly match total number of
 102     # training files to ensure that training data is well shuffled.
 103     input_files = []
 104     for input_pattern in input_patterns:
 105       input_files.extend(tf.io.gfile.glob(input_pattern))
 106     dataset = dataset.shuffle(len(input_files))
 107 
 108   # In parallel, create tf record dataset for each train files.
 109   # cycle_length = 8 means that up to 8 files will be read and deserialized in
 110   # parallel. You may want to increase this number if you have a large number of
 111   # CPU cores.
 112   dataset = dataset.interleave(
 113       tf.data.TFRecordDataset,
 114       cycle_length=8,
 115       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 116 
 117   if is_training:
 118     dataset = dataset.shuffle(100)
 119 
 120   decode_fn = lambda record: decode_record(record, name_to_features)
 121   dataset = dataset.map(
 122       decode_fn, num_parallel_calls=tf.data.experimental.AUTOTUNE)
 123 
 124   def _select_data_from_record(record):
 125     """Filter out features to use for pretraining."""
 126     x = {
 127         'input_word_ids': record['input_ids'],
 128         'input_mask': record['input_mask'],
 129         'input_type_ids': record['segment_ids'],
 130         'masked_lm_positions': record['masked_lm_positions'],
 131         'masked_lm_ids': record['masked_lm_ids'],
 132         'masked_lm_weights': record['masked_lm_weights'],
 133     }
 134     if use_next_sentence_label:
 135       x['next_sentence_labels'] = record['next_sentence_labels']
 136     if use_position_id:
 137       x['position_ids'] = record['position_ids']
 138 
 139     # TODO(hongkuny): Remove the fake labels after migrating bert pretraining.
 140     if output_fake_labels:
 141       return (x, record['masked_lm_weights'])
 142     else:
 143       return x
 144 
 145   dataset = dataset.map(
 146       _select_data_from_record,
 147       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 148   dataset = dataset.batch(batch_size, drop_remainder=is_training)
 149   dataset = dataset.prefetch(tf.data.experimental.AUTOTUNE)
 150   return dataset

 153 def create_classifier_dataset(file_path,
 154                               seq_length,
 155                               batch_size,
 156                               is_training=True,
 157                               input_pipeline_context=None,
 158                               label_type=tf.int64,
 159                               include_sample_weights=False):
 160   """Creates input dataset from (tf)records files for train/eval."""
 161   name_to_features = {
 162       'input_ids': tf.io.FixedLenFeature([seq_length], tf.int64),
 163       'input_mask': tf.io.FixedLenFeature([seq_length], tf.int64),
 164       'segment_ids': tf.io.FixedLenFeature([seq_length], tf.int64),
 165       'label_ids': tf.io.FixedLenFeature([], label_type),
 166   }
 167   if include_sample_weights:
 168     name_to_features['weight'] = tf.io.FixedLenFeature([], tf.float32)
 169   dataset = single_file_dataset(file_path, name_to_features)
 170 
 171   # The dataset is always sharded by number of hosts.
 172   # num_input_pipelines is the number of hosts rather than number of cores.
 173   if input_pipeline_context and input_pipeline_context.num_input_pipelines > 1:
 174     dataset = dataset.shard(input_pipeline_context.num_input_pipelines,
 175                             input_pipeline_context.input_pipeline_id)
 176 
 177   def _select_data_from_record(record):
 178     x = {
 179         'input_word_ids': record['input_ids'],
 180         'input_mask': record['input_mask'],
 181         'input_type_ids': record['segment_ids']
 182     }
 183     y = record['label_ids']
 184     if include_sample_weights:
 185       w = record['weight']
 186       return (x, y, w)
 187     return (x, y)
 188 
 189   if is_training:
 190     dataset = dataset.shuffle(100)
 191     dataset = dataset.repeat()
 192 
 193   dataset = dataset.map(
 194       _select_data_from_record,
 195       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 196   dataset = dataset.batch(batch_size, drop_remainder=is_training)
 197   dataset = dataset.prefetch(tf.data.experimental.AUTOTUNE)
 198   return dataset

 201 def create_squad_dataset(file_path,
 202                          seq_length,
 203                          batch_size,
 204                          is_training=True,
 205                          input_pipeline_context=None):
 206   """Creates input dataset from (tf)records files for train/eval."""
 207   name_to_features = {
 208       'input_ids': tf.io.FixedLenFeature([seq_length], tf.int64),
 209       'input_mask': tf.io.FixedLenFeature([seq_length], tf.int64),
 210       'segment_ids': tf.io.FixedLenFeature([seq_length], tf.int64),
 211   }
 212   if is_training:
 213     name_to_features['start_positions'] = tf.io.FixedLenFeature([], tf.int64)
 214     name_to_features['end_positions'] = tf.io.FixedLenFeature([], tf.int64)
 215   else:
 216     name_to_features['unique_ids'] = tf.io.FixedLenFeature([], tf.int64)
 217 
 218   dataset = single_file_dataset(file_path, name_to_features)
 219 
 220   # The dataset is always sharded by number of hosts.
 221   # num_input_pipelines is the number of hosts rather than number of cores.
 222   if input_pipeline_context and input_pipeline_context.num_input_pipelines > 1:
 223     dataset = dataset.shard(input_pipeline_context.num_input_pipelines,
 224                             input_pipeline_context.input_pipeline_id)
 225 
 226   def _select_data_from_record(record):
 227     """Dispatches record to features and labels."""
 228     x, y = {}, {}
 229     for name, tensor in record.items():
 230       if name in ('start_positions', 'end_positions'):
 231         y[name] = tensor
 232       elif name == 'input_ids':
 233         x['input_word_ids'] = tensor
 234       elif name == 'segment_ids':
 235         x['input_type_ids'] = tensor
 236       else:
 237         x[name] = tensor
 238     return (x, y)
 239 
 240   if is_training:
 241     dataset = dataset.shuffle(100)
 242     dataset = dataset.repeat()
 243 
 244   dataset = dataset.map(
 245       _select_data_from_record,
 246       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 247   dataset = dataset.batch(batch_size, drop_remainder=True)
 248   dataset = dataset.prefetch(tf.data.experimental.AUTOTUNE)
 249   return dataset
