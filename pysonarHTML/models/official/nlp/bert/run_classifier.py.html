  73 def get_loss_fn(num_classes):
  74   """Gets the classification loss function."""
  75 
  76   def classification_loss_fn(labels, logits):
  77     """Classification loss."""
  78     labels = tf.squeeze(labels)
  79     log_probs = tf.nn.log_softmax(logits, axis=-1)
  80     one_hot_labels = tf.one_hot(
  81         tf.cast(labels, dtype=tf.int32), depth=num_classes, dtype=tf.float32)
  82     per_example_loss = -tf.reduce_sum(
  83         tf.cast(one_hot_labels, dtype=tf.float32) * log_probs, axis=-1)
  84     return tf.reduce_mean(per_example_loss)
  85 
  86   return classification_loss_fn

  89 def get_dataset_fn(input_file_pattern,
  90                    max_seq_length,
  91                    global_batch_size,
  92                    is_training,
  93                    label_type=tf.int64,
  94                    include_sample_weights=False):
  95   """Gets a closure to create a dataset."""
  96 
  97   def _dataset_fn(ctx=None):
  98     """Returns tf.data.Dataset for distributed BERT pretraining."""
  99     batch_size = ctx.get_per_replica_batch_size(
 100         global_batch_size) if ctx else global_batch_size
 101     dataset = input_pipeline.create_classifier_dataset(
 102         tf.io.gfile.glob(input_file_pattern),
 103         max_seq_length,
 104         batch_size,
 105         is_training=is_training,
 106         input_pipeline_context=ctx,
 107         label_type=label_type,
 108         include_sample_weights=include_sample_weights)
 109     return dataset
 110 
 111   return _dataset_fn

 114 def run_bert_classifier(strategy,
 115                         bert_config,
 116                         input_meta_data,
 117                         model_dir,
 118                         epochs,
 119                         steps_per_epoch,
 120                         steps_per_loop,
 121                         eval_steps,
 122                         warmup_steps,
 123                         initial_lr,
 124                         init_checkpoint,
 125                         train_input_fn,
 126                         eval_input_fn,
 127                         training_callbacks=True,
 128                         custom_callbacks=None,
 129                         custom_metrics=None):
 130   """Run BERT classifier training using low-level API."""
 131   max_seq_length = input_meta_data['max_seq_length']
 132   num_classes = input_meta_data.get('num_labels', 1)
 133   is_regression = num_classes == 1
 134 
 135   def _get_classifier_model():
 136     """Gets a classifier model."""
 137     classifier_model, core_model = (
 138         bert_models.classifier_model(
 139             bert_config,
 140             num_classes,
 141             max_seq_length,
 142             hub_module_url=FLAGS.hub_module_url,
 143             hub_module_trainable=FLAGS.hub_module_trainable))
 144     optimizer = optimization.create_optimizer(initial_lr,
 145                                               steps_per_epoch * epochs,
 146                                               warmup_steps, FLAGS.end_lr,
 147                                               FLAGS.optimizer_type)
 148     classifier_model.optimizer = performance.configure_optimizer(
 149         optimizer,
 150         use_float16=common_flags.use_float16(),
 151         use_graph_rewrite=common_flags.use_graph_rewrite())
 152     return classifier_model, core_model
 153 
 154   # tf.keras.losses objects accept optional sample_weight arguments (eg. coming
 155   # from the dataset) to compute weighted loss, as used for the regression
 156   # tasks. The classification tasks, using the custom get_loss_fn don't accept
 157   # sample weights though.
 158   loss_fn = (tf.keras.losses.MeanSquaredError() if is_regression
 159              else get_loss_fn(num_classes))
 160 
 161   # Defines evaluation metrics function, which will create metrics in the
 162   # correct device and strategy scope.
 163   if custom_metrics:
 164     metric_fn = custom_metrics
 165   elif is_regression:
 166     metric_fn = functools.partial(
 167         tf.keras.metrics.MeanSquaredError,
 168         'mean_squared_error',
 169         dtype=tf.float32)
 170   else:
 171     metric_fn = functools.partial(
 172         tf.keras.metrics.SparseCategoricalAccuracy,
 173         'accuracy',
 174         dtype=tf.float32)
 175 
 176   # Start training using Keras compile/fit API.
 177   logging.info('Training using TF 2.x Keras compile/fit API with '
 178                'distribution strategy.')
 179   return run_keras_compile_fit(
 180       model_dir,
 181       strategy,
 182       _get_classifier_model,
 183       train_input_fn,
 184       eval_input_fn,
 185       loss_fn,
 186       metric_fn,
 187       init_checkpoint,
 188       epochs,
 189       steps_per_epoch,
 190       steps_per_loop,
 191       eval_steps,
 192       training_callbacks=training_callbacks,
 193       custom_callbacks=custom_callbacks)

 196 def run_keras_compile_fit(model_dir,
 197                           strategy,
 198                           model_fn,
 199                           train_input_fn,
 200                           eval_input_fn,
 201                           loss_fn,
 202                           metric_fn,
 203                           init_checkpoint,
 204                           epochs,
 205                           steps_per_epoch,
 206                           steps_per_loop,
 207                           eval_steps,
 208                           training_callbacks=True,
 209                           custom_callbacks=None):
 210   """Runs BERT classifier model using Keras compile/fit API."""
 211 
 212   with strategy.scope():
 213     training_dataset = train_input_fn()
 214     evaluation_dataset = eval_input_fn() if eval_input_fn else None
 215     bert_model, sub_model = model_fn()
 216     optimizer = bert_model.optimizer
 217 
 218     if init_checkpoint:
 219       checkpoint = tf.train.Checkpoint(model=sub_model)
 220       checkpoint.restore(init_checkpoint).assert_existing_objects_matched()
 221 
 222     if not isinstance(metric_fn, (list, tuple)):
 223       metric_fn = [metric_fn]
 224     bert_model.compile(
 225         optimizer=optimizer,
 226         loss=loss_fn,
 227         metrics=[fn() for fn in metric_fn],
 228         experimental_steps_per_execution=steps_per_loop)
 229 
 230     summary_dir = os.path.join(model_dir, 'summaries')
 231     summary_callback = tf.keras.callbacks.TensorBoard(summary_dir)
 232     checkpoint = tf.train.Checkpoint(model=bert_model, optimizer=optimizer)
 233     checkpoint_manager = tf.train.CheckpointManager(
 234         checkpoint,
 235         directory=model_dir,
 236         max_to_keep=None,
 237         step_counter=optimizer.iterations,
 238         checkpoint_interval=0)
 239     checkpoint_callback = keras_utils.SimpleCheckpoint(checkpoint_manager)
 240 
 241     if training_callbacks:
 242       if custom_callbacks is not None:
 243         custom_callbacks += [summary_callback, checkpoint_callback]
 244       else:
 245         custom_callbacks = [summary_callback, checkpoint_callback]
 246 
 247     history = bert_model.fit(
 248         x=training_dataset,
 249         validation_data=evaluation_dataset,
 250         steps_per_epoch=steps_per_epoch,
 251         epochs=epochs,
 252         validation_steps=eval_steps,
 253         callbacks=custom_callbacks)
 254     stats = {'total_training_steps': steps_per_epoch * epochs}
 255     if 'loss' in history.history:
 256       stats['train_loss'] = history.history['loss'][-1]
 257     if 'val_accuracy' in history.history:
 258       stats['eval_metrics'] = history.history['val_accuracy'][-1]
 259     return bert_model, stats

 262 def get_predictions_and_labels(strategy,
 263                                trained_model,
 264                                eval_input_fn,
 265                                return_probs=False):
 266   """Obtains predictions of trained model on evaluation data.
 267 
 268   Note that list of labels is returned along with the predictions because the
 269   order changes on distributing dataset over TPU pods.
 270 
 271   Args:
 272     strategy: Distribution strategy.
 273     trained_model: Trained model with preloaded weights.
 274     eval_input_fn: Input function for evaluation data.
 275     return_probs: Whether to return probabilities of classes.
 276 
 277   Returns:
 278     predictions: List of predictions.
 279     labels: List of gold labels corresponding to predictions.
 280   """
 281 
 282   @tf.function
 283   def test_step(iterator):
 284     """Computes predictions on distributed devices."""
 285 
 286     def _test_step_fn(inputs):
 287       """Replicated predictions."""
 288       inputs, labels = inputs
 289       logits = trained_model(inputs, training=False)
 290       probabilities = tf.nn.softmax(logits)
 291       return probabilities, labels
 292 
 293     outputs, labels = strategy.run(_test_step_fn, args=(next(iterator),))
 294     # outputs: current batch logits as a tuple of shard logits
 295     outputs = tf.nest.map_structure(strategy.experimental_local_results,
 296                                     outputs)
 297     labels = tf.nest.map_structure(strategy.experimental_local_results, labels)
 298     return outputs, labels
 299 
 300   def _run_evaluation(test_iterator):
 301     """Runs evaluation steps."""
 302     preds, golds = list(), list()
 303     try:
 304       with tf.experimental.async_scope():
 305         while True:
 306           probabilities, labels = test_step(test_iterator)
 307           for cur_probs, cur_labels in zip(probabilities, labels):
 308             if return_probs:
 309               preds.extend(cur_probs.numpy().tolist())
 310             else:
 311               preds.extend(tf.math.argmax(cur_probs, axis=1).numpy())
 312             golds.extend(cur_labels.numpy().tolist())
 313     except (StopIteration, tf.errors.OutOfRangeError):
 314       tf.experimental.async_clear_error()
 315     return preds, golds
 316 
 317   test_iter = iter(
 318       strategy.experimental_distribute_datasets_from_function(eval_input_fn))
 319   predictions, labels = _run_evaluation(test_iter)
 320 
 321   return predictions, labels

 324 def export_classifier(model_export_path, input_meta_data, bert_config,
 325                       model_dir):
 326   """Exports a trained model as a `SavedModel` for inference.
 327 
 328   Args:
 329     model_export_path: a string specifying the path to the SavedModel directory.
 330     input_meta_data: dictionary containing meta data about input and model.
 331     bert_config: Bert configuration file to define core bert layers.
 332     model_dir: The directory where the model weights and training/evaluation
 333       summaries are stored.
 334 
 335   Raises:
 336     Export path is not specified, got an empty string or None.
 337   """
 338   if not model_export_path:
 339     raise ValueError('Export path is not specified: %s' % model_export_path)
 340   if not model_dir:
 341     raise ValueError('Export path is not specified: %s' % model_dir)
 342 
 343   # Export uses float32 for now, even if training uses mixed precision.
 344   tf.keras.mixed_precision.experimental.set_policy('float32')
 345   classifier_model = bert_models.classifier_model(
 346       bert_config, input_meta_data.get('num_labels', 1))[0]
 347 
 348   model_saving_utils.export_bert_model(
 349       model_export_path, model=classifier_model, checkpoint_dir=model_dir)

 352 def run_bert(strategy,
 353              input_meta_data,
 354              model_config,
 355              train_input_fn=None,
 356              eval_input_fn=None,
 357              init_checkpoint=None,
 358              custom_callbacks=None,
 359              custom_metrics=None):
 360   """Run BERT training."""
 361   # Enables XLA in Session Config. Should not be set for TPU.
 362   keras_utils.set_session_config(FLAGS.enable_xla)
 363   performance.set_mixed_precision_policy(common_flags.dtype())
 364 
 365   epochs = FLAGS.num_train_epochs * FLAGS.num_eval_per_epoch
 366   train_data_size = (
 367       input_meta_data['train_data_size'] // FLAGS.num_eval_per_epoch)
 368   steps_per_epoch = int(train_data_size / FLAGS.train_batch_size)
 369   warmup_steps = int(epochs * train_data_size * 0.1 / FLAGS.train_batch_size)
 370   eval_steps = int(
 371       math.ceil(input_meta_data['eval_data_size'] / FLAGS.eval_batch_size))
 372 
 373   if not strategy:
 374     raise ValueError('Distribution strategy has not been specified.')
 375 
 376   if not custom_callbacks:
 377     custom_callbacks = []
 378 
 379   if FLAGS.log_steps:
 380     custom_callbacks.append(
 381         keras_utils.TimeHistory(
 382             batch_size=FLAGS.train_batch_size,
 383             log_steps=FLAGS.log_steps,
 384             logdir=FLAGS.model_dir))
 385 
 386   trained_model, _ = run_bert_classifier(
 387       strategy,
 388       model_config,
 389       input_meta_data,
 390       FLAGS.model_dir,
 391       epochs,
 392       steps_per_epoch,
 393       FLAGS.steps_per_loop,
 394       eval_steps,
 395       warmup_steps,
 396       FLAGS.learning_rate,
 397       init_checkpoint or FLAGS.init_checkpoint,
 398       train_input_fn,
 399       eval_input_fn,
 400       custom_callbacks=custom_callbacks,
 401       custom_metrics=custom_metrics)
 402 
 403   if FLAGS.model_export_path:
 404     model_saving_utils.export_bert_model(
 405         FLAGS.model_export_path, model=trained_model)
 406   return trained_model

 409 def custom_main(custom_callbacks=None, custom_metrics=None):
 410   """Run classification or regression.
 411 
 412   Args:
 413     custom_callbacks: list of tf.keras.Callbacks passed to training loop.
 414     custom_metrics: list of metrics passed to the training loop.
 415   """
 416   gin.parse_config_files_and_bindings(FLAGS.gin_file, FLAGS.gin_param)
 417 
 418   with tf.io.gfile.GFile(FLAGS.input_meta_data_path, 'rb') as reader:
 419     input_meta_data = json.loads(reader.read().decode('utf-8'))
 420   label_type = LABEL_TYPES_MAP[input_meta_data.get('label_type', 'int')]
 421   include_sample_weights = input_meta_data.get('has_sample_weights', False)
 422 
 423   if not FLAGS.model_dir:
 424     FLAGS.model_dir = '/tmp/bert20/'
 425 
 426   bert_config = bert_configs.BertConfig.from_json_file(FLAGS.bert_config_file)
 427 
 428   if FLAGS.mode == 'export_only':
 429     export_classifier(FLAGS.model_export_path, input_meta_data, bert_config,
 430                       FLAGS.model_dir)
 431     return
 432 
 433   strategy = distribution_utils.get_distribution_strategy(
 434       distribution_strategy=FLAGS.distribution_strategy,
 435       num_gpus=FLAGS.num_gpus,
 436       tpu_address=FLAGS.tpu)
 437   eval_input_fn = get_dataset_fn(
 438       FLAGS.eval_data_path,
 439       input_meta_data['max_seq_length'],
 440       FLAGS.eval_batch_size,
 441       is_training=False,
 442       label_type=label_type,
 443       include_sample_weights=include_sample_weights)
 444 
 445   if FLAGS.mode == 'predict':
 446     with strategy.scope():
 447       classifier_model = bert_models.classifier_model(
 448           bert_config, input_meta_data['num_labels'])[0]
 449       checkpoint = tf.train.Checkpoint(model=classifier_model)
 450       latest_checkpoint_file = (
 451           FLAGS.predict_checkpoint_path or
 452           tf.train.latest_checkpoint(FLAGS.model_dir))
 453       assert latest_checkpoint_file
 454       logging.info('Checkpoint file %s found and restoring from '
 455                    'checkpoint', latest_checkpoint_file)
 456       checkpoint.restore(
 457           latest_checkpoint_file).assert_existing_objects_matched()
 458       preds, _ = get_predictions_and_labels(
 459           strategy, classifier_model, eval_input_fn, return_probs=True)
 460     output_predict_file = os.path.join(FLAGS.model_dir, 'test_results.tsv')
 461     with tf.io.gfile.GFile(output_predict_file, 'w') as writer:
 462       logging.info('***** Predict results *****')
 463       for probabilities in preds:
 464         output_line = '\t'.join(
 465             str(class_probability)
 466             for class_probability in probabilities) + '\n'
 467         writer.write(output_line)
 468     return
 469 
 470   if FLAGS.mode != 'train_and_eval':
 471     raise ValueError('Unsupported mode is specified: %s' % FLAGS.mode)
 472   train_input_fn = get_dataset_fn(
 473       FLAGS.train_data_path,
 474       input_meta_data['max_seq_length'],
 475       FLAGS.train_batch_size,
 476       is_training=True,
 477       label_type=label_type,
 478       include_sample_weights=include_sample_weights)
 479   run_bert(
 480       strategy,
 481       input_meta_data,
 482       bert_config,
 483       train_input_fn,
 484       eval_input_fn,
 485       custom_callbacks=custom_callbacks,
 486       custom_metrics=custom_metrics)
 487
