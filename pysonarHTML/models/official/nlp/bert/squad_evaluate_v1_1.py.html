  14 """Evaluation of SQuAD predictions (version 1.1).
  15 
  16 The functions are copied from
  17 https://worksheets.codalab.org/rest/bundles/0xbcd57bee090b421c982906709c8c27e1/contents/blob/.
  18 
  19 The SQuAD dataset is described in this paper:
  20 SQuAD: 100,000+ Questions for Machine Comprehension of Text
  21 Pranav Rajpurkar, Jian Zhang, Konstantin Lopyrev, Percy Liang
  22 https://nlp.stanford.edu/pubs/rajpurkar2016squad.pdf
  23 """
  24 
  25 from __future__ import absolute_import
  26 from __future__ import division
  27 from __future__ import print_function
  28 
  29 import collections
  30 import re
  31 import string
  32 
  33 # pylint: disable=g-bad-import-order
  34 from absl import logging
  35 # pylint: enable=g-bad-import-order
  36 
  37 
  38 def _normalize_answer(s):
  39   """Lowers text and remove punctuation, articles and extra whitespace."""
  40 
  41   def remove_articles(text):
  42     return re.sub(r"\b(a|an|the)\b", " ", text)
  43 
  44   def white_space_fix(text):
  45     return " ".join(text.split())
  46 
  47   def remove_punc(text):
  48     exclude = set(string.punctuation)
  49     return "".join(ch for ch in text if ch not in exclude)
  50 
  51   def lower(text):
  52     return text.lower()
  53 
  54   return white_space_fix(remove_articles(remove_punc(lower(s))))
  55 
  56 
  57 def _f1_score(prediction, ground_truth):
  58   """Computes F1 score by comparing prediction to ground truth."""
  59   prediction_tokens = _normalize_answer(prediction).split()
  60   ground_truth_tokens = _normalize_answer(ground_truth).split()
  61   prediction_counter = collections.Counter(prediction_tokens)
  62   ground_truth_counter = collections.Counter(ground_truth_tokens)
  63   common = prediction_counter & ground_truth_counter
  64   num_same = sum(common.values())
  65   if num_same == 0:
  66     return 0
  67   precision = 1.0 * num_same / len(prediction_tokens)
  68   recall = 1.0 * num_same / len(ground_truth_tokens)
  69   f1 = (2 * precision * recall) / (precision + recall)
  70   return f1
  71 
  72 
  73 def _exact_match_score(prediction, ground_truth):
  74   """Checks if predicted answer exactly matches ground truth answer."""
  75   return _normalize_answer(prediction) == _normalize_answer(ground_truth)
  76 
  77 
  78 def _metric_max_over_ground_truths(metric_fn, prediction, ground_truths):
  79   """Computes the max over all metric scores."""
  80   scores_for_ground_truths = []
  81   for ground_truth in ground_truths:
  82     score = metric_fn(prediction, ground_truth)
  83     scores_for_ground_truths.append(score)
  84   return max(scores_for_ground_truths)
  85 
  86 
  87 def evaluate(dataset, predictions):
  88   """Evaluates predictions for a dataset."""
  89   f1 = exact_match = total = 0
  90   for article in dataset:
  91     for paragraph in article["paragraphs"]:
  92       for qa in paragraph["qas"]:
  93         total += 1
  94         if qa["id"] not in predictions:
  95           message = "Unanswered question " + qa["id"] + " will receive score 0."
  96           logging.error(message)
  97           continue
  98         ground_truths = [entry["text"] for entry in qa["answers"]]
  99         prediction = predictions[qa["id"]]
 100         exact_match += _metric_max_over_ground_truths(_exact_match_score,
 101                                                       prediction, ground_truths)
 102         f1 += _metric_max_over_ground_truths(_f1_score, prediction,
 103                                              ground_truths)
 104 
 105   exact_match = exact_match / total
 106   f1 = f1 / total
 107 
 108   return {"exact_match": exact_match, "final_f1": f1}
