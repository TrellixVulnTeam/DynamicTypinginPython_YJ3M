  16 """Tests for nlp.nhnet.multi_channel_attention."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import numpy as np
  23 import tensorflow as tf
  24 
  25 from official.nlp.nhnet import multi_channel_attention
  26 
  27 
  28 class MultiChannelAttentionTest(tf.test.TestCase):
  29 
  30   def test_doc_attention(self):
  31     num_heads = 2
  32     doc_attention = multi_channel_attention.DocAttention(num_heads, head_size=8)
  33     num_docs = 3
  34     inputs = np.zeros((2, num_docs, 10, 16), dtype=np.float32)
  35     doc_mask = np.zeros((2, num_docs), dtype=np.float32)
  36     outputs = doc_attention(inputs, doc_mask)
  37     self.assertEqual(outputs.shape, (2, num_docs))
  38 
  39   def test_multi_channel_attention(self):
  40     num_heads = 2
  41     num_docs = 5
  42     attention_layer = multi_channel_attention.MultiChannelAttention(
  43         num_heads, key_size=2)
  44 
  45     from_data = 10 * np.random.random_sample((3, 4, 8))
  46     to_data = 10 * np.random.random_sample((3, num_docs, 2, 8))
  47     mask_data = np.random.randint(2, size=(3, num_docs, 4, 2))
  48     doc_probs = np.random.randint(
  49         2, size=(3, num_heads, 4, num_docs)).astype(float)
  50     outputs = attention_layer([from_data, to_data, doc_probs], mask_data)
  51     self.assertEqual(outputs.shape, (3, 4, 8))
  52 
  53 
  54 if __name__ == "__main__":
  55   tf.test.main()
