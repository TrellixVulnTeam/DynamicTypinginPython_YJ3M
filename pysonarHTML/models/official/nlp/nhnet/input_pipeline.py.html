  24 def decode_record(record, name_to_features):
  25   """Decodes a record to a TensorFlow example."""
  26   example = tf.io.parse_single_example(record, name_to_features)
  27 
  28   # tf.Example only supports tf.int64, but the TPU only supports tf.int32.
  29   # So cast all int64 to int32.
  30   for name in list(example.keys()):
  31     t = example[name]
  32     if t.dtype == tf.int64:
  33       t = tf.cast(t, tf.int32)
  34     example[name] = t
  35 
  36   return example

  39 def process_singledoc_dataset(dataset, batch_size, params):
  40   """Parses and batches single-doc dataset."""
  41   name_to_features = {
  42       "input_ids_a": tf.io.FixedLenFeature([params.len_title], tf.int64),
  43       "input_ids_b": tf.io.FixedLenFeature([params.len_passage], tf.int64),
  44       "input_mask_b": tf.io.FixedLenFeature([params.len_passage], tf.int64),
  45       "segment_ids_b": tf.io.FixedLenFeature([params.len_passage], tf.int64),
  46   }
  47   decode_fn = lambda record: decode_record(record, name_to_features)
  48   dataset = dataset.map(
  49       decode_fn, num_parallel_calls=tf.data.experimental.AUTOTUNE)
  50 
  51   def _select_data_from_record(record):
  52     """Filter out features to use for pretraining."""
  53     return {
  54         "input_ids": record["input_ids_b"],
  55         "input_mask": record["input_mask_b"],
  56         "segment_ids": record["segment_ids_b"],
  57         "target_ids": record["input_ids_a"],
  58     }
  59 
  60   dataset = dataset.map(
  61       _select_data_from_record,
  62       num_parallel_calls=tf.data.experimental.AUTOTUNE)
  63   dataset = dataset.batch(batch_size, drop_remainder=True)
  64   return dataset

  67 def decode_sparse_record(record, name_to_features):
  68   """Decodes a sparse record to a TensorFlow example."""
  69   example = tf.io.parse_single_example(record, name_to_features)
  70 
  71   # tf.Example only supports tf.int64, but the TPU only supports tf.int32.
  72   # So cast all int64 to int32.
  73   for name in list(example.keys()):
  74     t = example[name]
  75     if t.dtype == tf.int64:
  76       t = tf.cast(t, tf.int32)
  77     example[name] = tf.sparse.to_dense(t)
  78 
  79   return example

  82 def _filter_max_length(example, max_title_length=256):
  83   """Indicates whether the example's length is lower than the maximum length."""
  84   return tf.size(example["targets"]) <= max_title_length

  87 def process_singledoc_transformer_dataset(dataset, batch_size, params):
  88   """Parses, batches and pads single-doc dataset."""
  89   name_to_features = {
  90       "inputs": tf.io.VarLenFeature(tf.int64),
  91       "targets": tf.io.VarLenFeature(tf.int64),
  92   }
  93   decode_fn = lambda record: decode_sparse_record(record, name_to_features)
  94   dataset = dataset.map(
  95       decode_fn, num_parallel_calls=tf.data.experimental.AUTOTUNE)
  96 
  97   def _select_data_from_record(record):
  98     """Filter out features to use for pretraining."""
  99     input_ids = record["inputs"][:params.len_passage]
 100     target_ids = record["targets"]
 101     input_mask = tf.ones_like(input_ids)
 102     segment_ids = tf.zeros_like(input_ids)
 103     return {
 104         "input_ids": input_ids,
 105         "input_mask": input_mask,
 106         "segment_ids": segment_ids,
 107         "target_ids": target_ids,
 108     }
 109 
 110   dataset = dataset.filter(lambda x: _filter_max_length(x, params.len_title))
 111 
 112   dataset = dataset.map(
 113       _select_data_from_record,
 114       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 115 
 116   dataset = dataset.padded_batch(
 117       batch_size, {
 118           "input_ids": [params.len_passage],
 119           "input_mask": [params.len_passage],
 120           "segment_ids": [params.len_passage],
 121           "target_ids": [params.len_title],
 122       },
 123       padding_values={
 124           "input_ids": params.pad_token_id,
 125           "input_mask": 0,
 126           "segment_ids": 0,
 127           "target_ids": params.pad_token_id,
 128       },
 129       drop_remainder=True)
 130 
 131   return dataset

 134 def multidoc_parse_spec(params, training=True):
 135   """Gets the mutli-doc tf.Example parsing spec."""
 136   len_p = params.len_passage
 137   name_to_features = {}
 138   feature_list = ["input_ids", "input_mask", "segment_ids"]
 139   for idx in params.passage_list:
 140     for feature in feature_list:
 141       name_to_features["%s_%s" % (feature, idx)] = tf.io.FixedLenFeature(
 142           [len_p], tf.int64)
 143   if training:
 144     # Cluster title.
 145     name_to_features["input_ids_a"] = tf.io.FixedLenFeature([params.len_title],
 146                                                             tf.int64)
 147   return name_to_features, feature_list

 150 def process_multidoc_dataset(dataset, batch_size, params):
 151   """Parses, organizes and batches multi-doc dataset."""
 152   name_to_features, feature_list = multidoc_parse_spec(params)
 153   decode_fn = lambda record: decode_record(record, name_to_features)
 154   dataset = dataset.map(
 155       decode_fn, num_parallel_calls=tf.data.experimental.AUTOTUNE)
 156 
 157   def _select_data_from_record(record):
 158     """Filter out features to use for pretraining."""
 159     features = {"target_ids": record["input_ids_a"]}
 160     for feature in feature_list:
 161       tensors = [record["%s_%s" % (feature, i)] for i in params.passage_list]
 162       features[feature] = tf.stack(tensors)
 163     return features
 164 
 165   dataset = dataset.map(
 166       _select_data_from_record,
 167       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 168   dataset = dataset.batch(batch_size, drop_remainder=True)
 169   return dataset

 172 def create_dataset(file_paths,
 173                    batch_size,
 174                    params,
 175                    is_training=True,
 176                    input_pipeline_context=None):
 177   """Creates input dataset from (tf)records files for pretraining."""
 178   dataset = tf.data.Dataset.list_files(file_paths, shuffle=is_training)
 179 
 180   if input_pipeline_context and input_pipeline_context.num_input_pipelines > 1:
 181     if not is_training or params.input_sharding:
 182       dataset = dataset.shard(input_pipeline_context.num_input_pipelines,
 183                               input_pipeline_context.input_pipeline_id)
 184 
 185   if is_training:
 186     dataset = dataset.repeat()
 187     # We set shuffle buffer to exactly match total number of
 188     # training files to ensure that training data is well shuffled.
 189     dataset = dataset.shuffle(len(file_paths))
 190 
 191   # In parallel, create tf record dataset for each train files.
 192   # cycle_length = 8 means that up to 8 files will be read and deserialized in
 193   # parallel. You may want to increase this number if you have a large number of
 194   # CPU cores.
 195   dataset = dataset.interleave(
 196       tf.data.TFRecordDataset,
 197       cycle_length=8,
 198       num_parallel_calls=tf.data.experimental.AUTOTUNE)
 199 
 200   if is_training:
 201     dataset = dataset.shuffle(100)
 202 
 203   if params.get("multi_channel_cross_attention", value=False):
 204     dataset = process_multidoc_dataset(dataset, batch_size, params)
 205   else:
 206     if not params.input_data_not_padded:
 207       dataset = process_singledoc_dataset(dataset, batch_size, params)
 208     else:
 209       dataset = process_singledoc_transformer_dataset(dataset, batch_size,
 210                                                       params)
 211   dataset = dataset.prefetch(tf.data.experimental.AUTOTUNE)
 212   return dataset

 215 def get_input_dataset(input_file_pattern,
 216                       batch_size,
 217                       params,
 218                       is_training,
 219                       strategy=None):
 220   """Returns input dataset from input file string."""
 221 
 222   # When using TPU pods, we need to clone dataset across
 223   # workers and need to pass in function that returns the dataset rather
 224   # than passing dataset instance itself.
 225   use_dataset_fn = isinstance(strategy, tf.distribute.experimental.TPUStrategy)
 226   if use_dataset_fn:
 227     if batch_size % strategy.num_replicas_in_sync != 0:
 228       raise ValueError(
 229           "Batch size must be divisible by number of replicas : {}".format(
 230               strategy.num_replicas_in_sync))
 231 
 232     # As auto rebatching is not supported in
 233     # `experimental_distribute_datasets_from_function()` API, which is
 234     # required when cloning dataset to multiple workers in eager mode,
 235     # we use per-replica batch size.
 236     batch_size = int(batch_size / strategy.num_replicas_in_sync)
 237 
 238   def _dataset_fn(ctx=None):
 239     """Returns tf.data.Dataset for distributed BERT pretraining."""
 240     input_files = []
 241     for input_pattern in input_file_pattern.split(","):
 242       input_files.extend(tf.io.gfile.glob(input_pattern))
 243 
 244     return create_dataset(
 245         input_files,
 246         batch_size,
 247         params,
 248         is_training=is_training,
 249         input_pipeline_context=ctx)
 250 
 251   if use_dataset_fn:
 252     return strategy.experimental_distribute_datasets_from_function(_dataset_fn)
 253   else:
 254     return strategy.experimental_distribute_dataset(_dataset_fn())
