  33 def all_strategy_combinations():
  34   return combinations.combine(
  35       distribution=[
  36           strategy_combinations.default_strategy,
  37           strategy_combinations.tpu_strategy,
  38           strategy_combinations.one_device_strategy_gpu,
  39           strategy_combinations.mirrored_strategy_with_gpu_and_cpu,
  40           strategy_combinations.mirrored_strategy_with_two_gpus,
  41       ],
  42       mode="eager",
  43   )

  46 def distribution_forward_path(strategy,
  47                               model,
  48                               inputs,
  49                               batch_size,
  50                               mode="train"):
  51   dataset = tf.data.Dataset.from_tensor_slices((inputs))
  52   dataset = dataset.batch(batch_size)
  53   dataset = strategy.experimental_distribute_dataset(dataset)
  54 
  55   @tf.function
  56   def test_step(inputs):
  57     """Calculates evaluation metrics on distributed devices."""
  58 
  59     def _test_step_fn(inputs):
  60       """Replicated accuracy calculation."""
  61       return model(inputs, mode=mode, training=False)
  62 
  63     outputs = strategy.run(_test_step_fn, args=(inputs,))
  64     return tf.nest.map_structure(strategy.experimental_local_results, outputs)
  65 
  66   return [test_step(inputs) for inputs in dataset]

  69 def process_decoded_ids(predictions, end_token_id):
  70   """Transforms decoded tensors to lists ending with END_TOKEN_ID."""
  71   if isinstance(predictions, tf.Tensor):
  72     predictions = predictions.numpy()
  73   flatten_ids = predictions.reshape((-1, predictions.shape[-1]))
  74   results = []
  75   for ids in flatten_ids:
  76     ids = list(ids)
  77     if end_token_id in ids:
  78       ids = ids[:ids.index(end_token_id)]
  79       results.append(ids)
  80   return results

 121   def test_bert2bert_decoding(self):
 122     seq_length = 10
 123     self._config.override(
 124         {
 125             "beam_size": 3,
 126             "len_title": seq_length,
 127             "alpha": 0.6,
 128         },
 129         is_strict=False)
 130 
 131     batch_size = 2
 132     fake_ids = np.zeros((batch_size, seq_length), dtype=np.int32)
 133     fake_inputs = {
 134         "input_ids": fake_ids,
 135         "input_mask": fake_ids,
 136         "segment_ids": fake_ids,
 137     }
 138     self._config.override({
 139         "padded_decode": False,
 140         "use_cache": False,
 141     },
 142                           is_strict=False)
 143     model = models.create_bert2bert_model(params=self._config)
 144     ckpt = tf.train.Checkpoint(model=model)
 145 
 146     # Initializes variables from checkpoint to keep outputs deterministic.
 147     init_checkpoint = ckpt.save(os.path.join(self.get_temp_dir(), "ckpt"))
 148     ckpt.restore(init_checkpoint).assert_existing_objects_matched()
 149     top_ids, scores = model(fake_inputs, mode="predict")
 150 
 151     self._config.override({
 152         "padded_decode": False,
 153         "use_cache": True,
 154     },
 155                           is_strict=False)
 156     model = models.create_bert2bert_model(params=self._config)
 157     ckpt = tf.train.Checkpoint(model=model)
 158     ckpt.restore(init_checkpoint).assert_existing_objects_matched()
 159     cached_top_ids, cached_scores = model(fake_inputs, mode="predict")
 160     self.assertEqual(
 161         process_decoded_ids(top_ids, self._config.end_token_id),
 162         process_decoded_ids(cached_top_ids, self._config.end_token_id))
 163     self.assertAllClose(scores, cached_scores)
 164 
 165     self._config.override({
 166         "padded_decode": True,
 167         "use_cache": True,
 168     },
 169                           is_strict=False)
 170     model = models.create_bert2bert_model(params=self._config)
 171     ckpt = tf.train.Checkpoint(model=model)
 172     ckpt.restore(init_checkpoint).assert_existing_objects_matched()
 173     padded_top_ids, padded_scores = model(fake_inputs, mode="predict")
 174     self.assertEqual(
 175         process_decoded_ids(top_ids, self._config.end_token_id),
 176         process_decoded_ids(padded_top_ids, self._config.end_token_id))
 177     self.assertAllClose(scores, padded_scores)

 214   def setUp(self):
 215     super(NHNetTest, self).setUp()
 216     self._nhnet_config = configs.NHNetConfig()
 217     self._nhnet_config.override(utils.get_test_params().as_dict())
 218     self._bert2bert_config = configs.BERT2BERTConfig()
 219     self._bert2bert_config.override(utils.get_test_params().as_dict())

 221   def _count_params(self, layer, trainable_only=True):
 222     """Returns the count of all model parameters, or just trainable ones."""
 223     if not trainable_only:
 224       return layer.count_params()
 225     else:
 226       return int(
 227           np.sum([
 228               tf.keras.backend.count_params(p) for p in layer.trainable_weights
 229           ]))
 230 
 231   def test_create_nhnet_layers(self):
 232     single_doc_bert, single_doc_decoder = models.get_bert2bert_layers(
 233         self._bert2bert_config)
 234     multi_doc_bert, multi_doc_decoder = models.get_nhnet_layers(
 235         self._nhnet_config)
 236     # Expects multi-doc encoder/decoder have the same number of parameters as
 237     # single-doc encoder/decoder.
 238     self.assertEqual(
 239         self._count_params(multi_doc_bert), self._count_params(single_doc_bert))
 240     self.assertEqual(
 241         self._count_params(multi_doc_decoder),
 242         self._count_params(single_doc_decoder))
 243 
 244   def test_checkpoint_restore(self):
 245     bert2bert_model = models.create_bert2bert_model(self._bert2bert_config)
 246     ckpt = tf.train.Checkpoint(model=bert2bert_model)
 247     init_checkpoint = ckpt.save(os.path.join(self.get_temp_dir(), "ckpt"))
 248     nhnet_model = models.create_nhnet_model(
 249         params=self._nhnet_config, init_checkpoint=init_checkpoint)
 250     source_weights = (
 251         bert2bert_model.bert_layer.trainable_weights +
 252         bert2bert_model.decoder_layer.trainable_weights)
 253     dest_weights = (
 254         nhnet_model.bert_layer.trainable_weights +
 255         nhnet_model.decoder_layer.trainable_weights)
 256     for source_weight, dest_weight in zip(source_weights, dest_weights):
 257       self.assertAllClose(source_weight.numpy(), dest_weight.numpy())
