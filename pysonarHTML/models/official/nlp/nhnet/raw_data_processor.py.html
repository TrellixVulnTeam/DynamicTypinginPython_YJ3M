  29 class RawDataProcessor(object):
  30   """Data converter for story examples."""
  31 
  32   def __init__(self,
  33                vocab: str,
  34                do_lower_case: bool,
  35                len_title: int = 15,
  36                len_passage: int = 200,
  37                max_num_articles: int = 5,
  38                include_article_title_in_passage: bool = False,
  39                include_text_snippet_in_example: bool = False):
  40     """Constructs a RawDataProcessor.
  41 
  42     Args:
  43       vocab: Filepath of the BERT vocabulary.
  44       do_lower_case: Whether the vocabulary is uncased or not.
  45       len_title: Maximum number of tokens in story headline.
  46       len_passage: Maximum number of tokens in article passage.
  47       max_num_articles: Maximum number of articles in a story.
  48       include_article_title_in_passage: Whether to include article title in
  49         article passage.
  50       include_text_snippet_in_example: Whether to include text snippet
  51         (headline and article content) in generated tensorflow Examples, for
  52         debug usage. If include_article_title_in_passage=True, title and body
  53         will be separated by [SEP].
  54     """
  55     self.articles = dict()
  56     self.tokenizer = tokenization.FullTokenizer(
  57         vocab, do_lower_case=do_lower_case, split_on_punc=False)
  58     self.len_title = len_title
  59     self.len_passage = len_passage
  60     self.max_num_articles = max_num_articles
  61     self.include_article_title_in_passage = include_article_title_in_passage
  62     self.include_text_snippet_in_example = include_text_snippet_in_example
  63     # ex_index=5 deactivates printing inside convert_single_example.
  64     self.ex_index = 5
  65     # Parameters used in InputExample, not used in NHNet.
  66     self.label = 0
  67     self.guid = 0
  68     self.num_generated_examples = 0
  69 
  70   def read_crawled_articles(self, folder_path):
  71     """Reads crawled articles under folder_path."""
  72     for path, _, files in os.walk(folder_path):
  73       for name in files:
  74         if not name.endswith(".json"):
  75           continue
  76         url, article = self._get_article_content_from_json(
  77             os.path.join(path, name))
  78         if not article.text_a:
  79           continue
  80         self.articles[RawDataProcessor.normalize_url(url)] = article
  81         if len(self.articles) % 5000 == 0:
  82           print("Number of articles loaded: %d\r" % len(self.articles), end="")
  83     print()
  84     return len(self.articles)
  85 
  86   def generate_examples(self, input_file, output_files):
  87     """Loads story from input json file and exports examples in output_files."""
  88     writers = []
  89     story_partition = []
  90     for output_file in output_files:
  91       writers.append(tf.io.TFRecordWriter(output_file))
  92       story_partition.append(list())
  93     with tf.io.gfile.GFile(input_file, "r") as story_json_file:
  94       stories = json.load(story_json_file)
  95       writer_index = 0
  96       for story in stories:
  97         articles = []
  98         for url in story["urls"]:
  99           normalized_url = RawDataProcessor.normalize_url(url)
 100           if normalized_url in self.articles:
 101             articles.append(self.articles[normalized_url])
 102         if not articles:
 103           continue
 104         story_partition[writer_index].append((story["label"], articles))
 105         writer_index = (writer_index + 1) % len(writers)
 106     lock = multiprocessing.Lock()
 107     pool = multiprocessing.pool.ThreadPool(len(writers))
 108     data = [(story_partition[i], writers[i], lock) for i in range(len(writers))]
 109     pool.map(self._write_story_partition, data)
 110     return len(stories), self.num_generated_examples
 111 
 112   @classmethod
 113   def normalize_url(cls, url):
 114     """Normalize url for better matching."""
 115     url = urllib.parse.unquote(
 116         urllib.parse.urlsplit(url)._replace(query=None).geturl())
 117     output, part = [], None
 118     for part in url.split("//"):
 119       if part == "http:" or part == "https:":
 120         continue
 121       else:
 122         output.append(part)
 123     return "//".join(output)
 124 
 125   def _get_article_content_from_json(self, file_path):
 126     """Returns (url, InputExample) keeping content extracted from file_path."""
 127     with tf.io.gfile.GFile(file_path, "r") as article_json_file:
 128       article = json.load(article_json_file)
 129       if self.include_article_title_in_passage:
 130         return article["url"], classifier_data_lib.InputExample(
 131             guid=self.guid,
 132             text_a=article["title"],
 133             text_b=article["maintext"],
 134             label=self.label)
 135       else:
 136         return article["url"], classifier_data_lib.InputExample(
 137             guid=self.guid, text_a=article["maintext"], label=self.label)
 138 
 139   def _write_story_partition(self, data):
 140     """Writes stories in a partition into file."""
 141     for (story_headline, articles) in data[0]:
 142       story_example = tf.train.Example(
 143           features=tf.train.Features(
 144               feature=self._get_single_story_features(story_headline,
 145                                                       articles)))
 146       data[1].write(story_example.SerializeToString())
 147       data[2].acquire()
 148       try:
 149         self.num_generated_examples += 1
 150         if self.num_generated_examples % 1000 == 0:
 151           print(
 152               "Number of stories written: %d\r" % self.num_generated_examples,
 153               end="")
 154       finally:
 155         data[2].release()
 156 
 157   def _get_single_story_features(self, story_headline, articles):
 158     """Converts a list of articles to a tensorflow Example."""
 159     def get_text_snippet(article):
 160       if article.text_b:
 161         return " [SEP] ".join([article.text_a, article.text_b])
 162       else:
 163         return article.text_a
 164 
 165     story_features = collections.OrderedDict()
 166     story_headline_feature = classifier_data_lib.convert_single_example(
 167         ex_index=self.ex_index,
 168         example=classifier_data_lib.InputExample(
 169             guid=self.guid, text_a=story_headline, label=self.label),
 170         label_list=[self.label],
 171         max_seq_length=self.len_title,
 172         tokenizer=self.tokenizer)
 173     if self.include_text_snippet_in_example:
 174       story_headline_feature.label_id = story_headline
 175     self._add_feature_with_suffix(
 176         feature=story_headline_feature,
 177         suffix="a",
 178         story_features=story_features)
 179     for (article_index, article) in enumerate(articles):
 180       if article_index == self.max_num_articles:
 181         break
 182       article_feature = classifier_data_lib.convert_single_example(
 183           ex_index=self.ex_index,
 184           example=article,
 185           label_list=[self.label],
 186           max_seq_length=self.len_passage,
 187           tokenizer=self.tokenizer)
 188       if self.include_text_snippet_in_example:
 189         article_feature.label_id = get_text_snippet(article)
 190       suffix = chr(ord("b") + article_index)
 191       self._add_feature_with_suffix(
 192           feature=article_feature, suffix=suffix, story_features=story_features)
 193 
 194     # Adds empty features as placeholder.
 195     for article_index in range(len(articles), self.max_num_articles):
 196       suffix = chr(ord("b") + article_index)
 197       empty_article = classifier_data_lib.InputExample(
 198           guid=self.guid, text_a="", label=self.label)
 199       empty_feature = classifier_data_lib.convert_single_example(
 200           ex_index=self.ex_index,
 201           example=empty_article,
 202           label_list=[self.label],
 203           max_seq_length=self.len_passage,
 204           tokenizer=self.tokenizer)
 205       if self.include_text_snippet_in_example:
 206         empty_feature.label_id = ""
 207       self._add_feature_with_suffix(
 208           feature=empty_feature, suffix=suffix, story_features=story_features)
 209     return story_features
 210 
 211   def _add_feature_with_suffix(self, feature, suffix, story_features):
 212     """Appends suffix to feature names and fills in the corresponding values."""
 213 
 214     def _create_int_feature(values):
 215       return tf.train.Feature(int64_list=tf.train.Int64List(value=list(values)))
 216 
 217     def _create_string_feature(value):
 218       return tf.train.Feature(bytes_list=tf.train.BytesList(value=[value]))
 219 
 220     story_features["input_ids_%c" % suffix] = _create_int_feature(
 221         feature.input_ids)
 222     story_features["input_mask_%c" % suffix] = _create_int_feature(
 223         feature.input_mask)
 224     story_features["segment_ids_%c" % suffix] = _create_int_feature(
 225         feature.segment_ids)
 226     if self.include_text_snippet_in_example:
 227       story_features["text_snippet_%c" % suffix] = _create_string_feature(
 228           bytes(feature.label_id.encode()))
