  37 def embedding_linear(embedding_matrix, x):
  38   """Uses embeddings as linear transformation weights."""
  39   with tf.name_scope("presoftmax_linear"):
  40     batch_size = tf.shape(x)[0]
  41     length = tf.shape(x)[1]
  42     hidden_size = tf.shape(x)[2]
  43     vocab_size = tf.shape(embedding_matrix)[0]
  44 
  45     x = tf.reshape(x, [-1, hidden_size])
  46     logits = tf.matmul(x, embedding_matrix, transpose_b=True)
  47 
  48     return tf.reshape(logits, [batch_size, length, vocab_size])

  51 def _add_sos_to_seq(seq, start_token_id):
  52   """Add a start sequence token while keeping seq length."""
  53   batch_size = tf.shape(seq)[0]
  54   seq_len = tf.shape(seq)[1]
  55   sos_ids = tf.ones([batch_size], tf.int32) * start_token_id
  56   targets = tf.concat([tf.expand_dims(sos_ids, axis=1), seq], axis=1)
  57   targets = targets[:, :-1]
  58   tf.assert_equal(tf.shape(targets), (batch_size, seq_len))
  59   return targets
  60

  62 def remove_sos_from_seq(seq, pad_token_id):
  63   """Remove the start sequence token while keeping seq length."""
  64   batch_size, seq_len = tf_utils.get_shape_list(seq, expected_rank=2)
  65   # remove <s>
  66   targets = seq[:, 1:]
  67   # pad
  68   pad_ids = tf.ones([batch_size], tf.int32) * pad_token_id
  69   targets = tf.concat([targets, tf.expand_dims(pad_ids, axis=1)], axis=1)
  70   tf.assert_equal(tf.shape(targets), (batch_size, seq_len))
  71   return targets

  74 class Bert2Bert(tf.keras.Model):
  75   """Bert2Bert encoder decoder model for training."""
  76 
  77   def __init__(self, params, bert_layer, decoder_layer, name=None):
  78     super(Bert2Bert, self).__init__(name=name)
  79     self.params = params
  80     if not bert_layer.built:
  81       raise ValueError("bert_layer should be built.")
  82     if not decoder_layer.built:
  83       raise ValueError("decoder_layer should be built.")
  84     self.bert_layer = bert_layer
  85     self.decoder_layer = decoder_layer
  86 
  87   def get_config(self):
  88     return {"params": self.params.as_dict()}
  89 
  90   def get_decode_logits(self,
  91                         decoder_inputs,
  92                         ids,
  93                         decoder_self_attention_bias,
  94                         step,
  95                         cache=None):
  96     if cache:
  97       if self.params.get("padded_decode", False):
  98         bias_shape = decoder_self_attention_bias.shape.as_list()
  99         self_attention_bias = tf.slice(
 100             decoder_self_attention_bias, [0, 0, step, 0],
 101             [bias_shape[0], bias_shape[1], 1, bias_shape[3]])
 102       else:
 103         self_attention_bias = decoder_self_attention_bias[:, :, step:step +
 104                                                           1, :step + 1]
 105       # Sets decoder input to the last generated IDs.
 106       decoder_input = ids[:, -1:]
 107     else:
 108       self_attention_bias = decoder_self_attention_bias[:, :, :step + 1, :step +
 109                                                         1]
 110       decoder_input = ids
 111     decoder_inputs["target_ids"] = decoder_input
 112     decoder_inputs["self_attention_bias"] = self_attention_bias
 113     if cache:
 114       decoder_outputs = self.decoder_layer(
 115           decoder_inputs,
 116           cache,
 117           decode_loop_step=step,
 118           padded_decode=self.params.get("padded_decode", False))
 119     else:
 120       decoder_outputs = self.decoder_layer(decoder_inputs)
 121     logits = embedding_linear(self.decoder_layer.embedding_lookup.embeddings,
 122                               decoder_outputs[:, -1:, :])
 123     logits = tf.squeeze(logits, axis=[1])
 124     return logits

 126   def _get_symbols_to_logits_fn(self, max_decode_length):
 127     """Returns a decoding function that calculates logits of the next tokens."""
 128     # Max decode length should be smaller than the positional embedding max
 129     # sequence length.
 130     decoder_self_attention_bias = decoder.get_attention_bias(
 131         input_tensor=None,
 132         bias_type="decoder_self",
 133         max_length=max_decode_length)
 134 
 135     def _symbols_to_logits_fn(ids, i, cache):
 136       """Generate logits for next candidate IDs.
 137 
 138       Args:
 139         ids: Current decoded sequences. int tensor with shape [batch_size *
 140           beam_size, i + 1]
 141         i: Loop index
 142         cache: dictionary of values storing the encoder output, encoder-decoder
 143           attention bias, and previous decoder attention values.
 144 
 145       Returns:
 146         Tuple of
 147           (logits with shape [batch_size * beam_size, vocab_size],
 148            updated cache values)
 149       """
 150       decoder_inputs = dict(
 151           all_encoder_outputs=cache["all_encoder_outputs"],
 152           attention_bias=cache["attention_bias"])
 153       logits = self.get_decode_logits(
 154           decoder_inputs,
 155           ids,
 156           decoder_self_attention_bias,
 157           step=i,
 158           cache=cache if self.params.use_cache else None)
 159       return logits, cache
 160 
 161     return _symbols_to_logits_fn
 162 
 163   def train_decode(self, decode_outputs):
 164     logits = embedding_linear(self.decoder_layer.embedding_lookup.embeddings,
 165                               decode_outputs)
 166     decode_output_ids = tf.cast(tf.argmax(logits, axis=-1), tf.int32)
 167     output_log_probs = tf.nn.log_softmax(logits, axis=-1)
 168     return logits, decode_output_ids, output_log_probs
 169 
 170   def predict_decode(self, start_token_ids, cache):
 171     symbols_to_logits_fn = self._get_symbols_to_logits_fn(self.params.len_title)
 172     # Use beam search to find the top beam_size sequences and scores.
 173     decoded_ids, scores = beam_search.sequence_beam_search(
 174         symbols_to_logits_fn=symbols_to_logits_fn,
 175         initial_ids=start_token_ids,
 176         initial_cache=cache,
 177         vocab_size=self.params.vocab_size,
 178         beam_size=self.params.beam_size,
 179         alpha=self.params.alpha,
 180         max_decode_length=self.params.len_title,
 181         padded_decode=self.params.get("padded_decode", False),
 182         eos_id=self.params.end_token_id)
 183     return decoded_ids, scores
 184 
 185   def _get_logits_for_decode_ids(self, decoder_inputs, top_decoded_ids):
 186     """Returns the log probabilities for ids."""
 187     target_ids = _add_sos_to_seq(top_decoded_ids, self.params.start_token_id)
 188     decoder_inputs["self_attention_bias"] = decoder.get_attention_bias(
 189         target_ids, bias_type="decoder_self")
 190     decoder_inputs["target_ids"] = target_ids
 191     decoder_outputs = self.decoder_layer(decoder_inputs)
 192     logits = embedding_linear(self.decoder_layer.embedding_lookup.embeddings,
 193                               decoder_outputs)
 194     return logits
 195 
 196   def _init_cache(self, batch_size):
 197     num_heads = self.params.num_decoder_attn_heads
 198     dim_per_head = self.params.hidden_size // num_heads
 199     init_decode_length = (
 200         self.params.len_title if self.params.get("padded_decode", False) else 0)
 201     cache = {}
 202     for layer in range(self.params.num_decoder_layers):
 203       cache[str(layer)] = {
 204           "key":
 205               tf.zeros(
 206                   [batch_size, init_decode_length, num_heads, dim_per_head],
 207                   dtype=tf.float32),
 208           "value":
 209               tf.zeros(
 210                   [batch_size, init_decode_length, num_heads, dim_per_head],
 211                   dtype=tf.float32)
 212       }
 213     return cache

 215   def call(self, inputs, mode="train"):
 216     """Implements call().
 217 
 218     Args:
 219       inputs: a dictionary of tensors.
 220       mode: string, an enum for mode, train/eval.
 221 
 222     Returns:
 223       logits, decode_output_ids, output_log_probs for training. top_decoded_ids
 224       for eval.
 225     """
 226     input_ids = inputs["input_ids"]
 227     input_mask = inputs["input_mask"]
 228     segment_ids = inputs["segment_ids"]
 229     all_encoder_outputs, _ = self.bert_layer(
 230         [input_ids, input_mask, segment_ids])
 231 
 232     if mode not in ("train", "eval", "predict"):
 233       raise ValueError("Invalid call mode: %s" % mode)
 234     encoder_decoder_attention_bias = decoder.get_attention_bias(
 235         input_ids,
 236         bias_type="single_cross",
 237         padding_value=self.params.pad_token_id)
 238     if mode == "train":
 239       self_attention_bias = decoder.get_attention_bias(
 240           inputs["target_ids"], bias_type="decoder_self")
 241       decoder_inputs = dict(
 242           attention_bias=encoder_decoder_attention_bias,
 243           all_encoder_outputs=all_encoder_outputs,
 244           target_ids=inputs["target_ids"],
 245           self_attention_bias=self_attention_bias)
 246       decoder_outputs = self.decoder_layer(decoder_inputs)
 247       return self.train_decode(decoder_outputs)
 248 
 249     batch_size = tf.shape(input_ids)[0]
 250     start_token_ids = tf.ones([batch_size],
 251                               tf.int32) * self.params.start_token_id
 252     # Add encoder output and attention bias to the cache.
 253     if self.params.use_cache:
 254       cache = self._init_cache(batch_size)
 255     else:
 256       cache = {}
 257     cache["all_encoder_outputs"] = all_encoder_outputs
 258     cache["attention_bias"] = encoder_decoder_attention_bias
 259     decoded_ids, scores = self.predict_decode(start_token_ids, cache)
 260     if mode == "predict":
 261       return decoded_ids[:, :self.params.beam_size,
 262                          1:], scores[:, :self.params.beam_size]
 263 
 264     decoder_inputs = dict(
 265         attention_bias=encoder_decoder_attention_bias,
 266         all_encoder_outputs=all_encoder_outputs)
 267     top_decoded_ids = decoded_ids[:, 0, 1:]
 268     return self._get_logits_for_decode_ids(decoder_inputs, top_decoded_ids)

 271 class NHNet(Bert2Bert):
 272   """NHNet model which performs multi-doc decoding."""
 273 
 274   def __init__(self, params, bert_layer, decoder_layer, name=None):
 275     super(NHNet, self).__init__(params, bert_layer, decoder_layer, name=name)
 276     self.doc_attention = multi_channel_attention.DocAttention(
 277         num_heads=params.num_decoder_attn_heads,
 278         head_size=params.hidden_size // params.num_decoder_attn_heads)
 279 
 280   def _expand_doc_attention_probs(self, doc_attention_probs, target_length):
 281     """Expands doc attention probs to fit the decoding sequence length."""
 282     doc_attention_probs = tf.expand_dims(
 283         doc_attention_probs, axis=[1])  # [B, 1, A]
 284     doc_attention_probs = tf.expand_dims(
 285         doc_attention_probs, axis=[2])  # [B, 1, 1, A]
 286     return tf.tile(doc_attention_probs,
 287                    [1, self.params.num_decoder_attn_heads, target_length, 1])
 288 
 289   def _get_symbols_to_logits_fn(self, max_decode_length):
 290     """Returns a decoding function that calculates logits of the next tokens."""
 291     # Max decode length should be smaller than the positional embedding max
 292     # sequence length.
 293     decoder_self_attention_bias = decoder.get_attention_bias(
 294         input_tensor=None,
 295         bias_type="decoder_self",
 296         max_length=max_decode_length)
 297 
 298     def _symbols_to_logits_fn(ids, i, cache):
 299       """Generate logits for next candidate IDs."""
 300       if self.params.use_cache:
 301         target_length = 1
 302       else:
 303         target_length = i + 1
 304       decoder_inputs = dict(
 305           doc_attention_probs=self._expand_doc_attention_probs(
 306               cache["doc_attention_probs"], target_length),
 307           all_encoder_outputs=cache["all_encoder_outputs"],
 308           attention_bias=cache["attention_bias"])
 309       logits = self.get_decode_logits(
 310           decoder_inputs,
 311           ids,
 312           decoder_self_attention_bias,
 313           step=i,
 314           cache=cache if self.params.use_cache else None)
 315       return logits, cache
 316 
 317     return _symbols_to_logits_fn
 318 
 319   def call(self, inputs, mode="training"):
 320     input_shape = tf_utils.get_shape_list(inputs["input_ids"], expected_rank=3)
 321     batch_size, num_docs, len_passage = (input_shape[0], input_shape[1],
 322                                          input_shape[2])
 323     input_ids = tf.reshape(inputs["input_ids"], [-1, len_passage])
 324     input_mask = tf.reshape(inputs["input_mask"], [-1, len_passage])
 325     segment_ids = tf.reshape(inputs["segment_ids"], [-1, len_passage])
 326     all_encoder_outputs, _ = self.bert_layer(
 327         [input_ids, input_mask, segment_ids])
 328     encoder_outputs = tf.reshape(
 329         all_encoder_outputs[-1],
 330         [batch_size, num_docs, len_passage, self.params.hidden_size])
 331     doc_attention_mask = tf.reshape(
 332         tf.cast(
 333             tf.math.count_nonzero(input_mask, axis=1, dtype=tf.int32) > 2,
 334             tf.int32), [batch_size, num_docs])
 335 
 336     doc_attention_probs = self.doc_attention(encoder_outputs,
 337                                              doc_attention_mask)
 338     encoder_decoder_attention_bias = decoder.get_attention_bias(
 339         inputs["input_ids"],
 340         bias_type="multi_cross",
 341         padding_value=self.params.pad_token_id)
 342 
 343     if mode == "train":
 344       target_length = tf_utils.get_shape_list(
 345           inputs["target_ids"], expected_rank=2)[1]
 346       doc_attention_probs = self._expand_doc_attention_probs(
 347           doc_attention_probs, target_length)
 348       self_attention_bias = decoder.get_attention_bias(
 349           inputs["target_ids"], bias_type="decoder_self")
 350       decoder_inputs = dict(
 351           attention_bias=encoder_decoder_attention_bias,
 352           self_attention_bias=self_attention_bias,
 353           target_ids=inputs["target_ids"],
 354           all_encoder_outputs=encoder_outputs,
 355           doc_attention_probs=doc_attention_probs)
 356       decoder_outputs = self.decoder_layer(decoder_inputs)
 357       return self.train_decode(decoder_outputs)
 358 
 359     # Adds encoder output and attention bias to the cache.
 360     if self.params.use_cache:
 361       cache = self._init_cache(batch_size)
 362     else:
 363       cache = {}
 364     cache["all_encoder_outputs"] = [encoder_outputs]
 365     cache["attention_bias"] = encoder_decoder_attention_bias
 366     cache["doc_attention_probs"] = doc_attention_probs
 367 
 368     start_token_ids = tf.ones([batch_size],
 369                               tf.int32) * self.params.start_token_id
 370     decoded_ids, scores = self.predict_decode(start_token_ids, cache)
 371     if mode == "predict":
 372       return decoded_ids[:, :self.params.beam_size,
 373                          1:], scores[:, :self.params.beam_size]
 374 
 375     top_decoded_ids = decoded_ids[:, 0, 1:]
 376     target_length = tf_utils.get_shape_list(top_decoded_ids)[-1]
 377     decoder_inputs = dict(
 378         attention_bias=encoder_decoder_attention_bias,
 379         all_encoder_outputs=[encoder_outputs],
 380         doc_attention_probs=self._expand_doc_attention_probs(
 381             doc_attention_probs, target_length))
 382     return self._get_logits_for_decode_ids(decoder_inputs, top_decoded_ids)

 385 def get_bert2bert_layers(params: configs.BERT2BERTConfig):
 386   """Creates a Bert2Bert stem model and returns Bert encoder/decoder.
 387 
 388   We use funtional-style to create stem model because we need to make all layers
 389   built to restore variables in a customized way. The layers are called with
 390   placeholder inputs to make them fully built.
 391 
 392   Args:
 393     params: ParamsDict.
 394 
 395   Returns:
 396     two keras Layers, bert_model_layer and decoder_layer
 397   """
 398   input_ids = tf.keras.layers.Input(
 399       shape=(None,), name="input_ids", dtype=tf.int32)
 400   input_mask = tf.keras.layers.Input(
 401       shape=(None,), name="input_mask", dtype=tf.int32)
 402   segment_ids = tf.keras.layers.Input(
 403       shape=(None,), name="segment_ids", dtype=tf.int32)
 404   target_ids = tf.keras.layers.Input(
 405       shape=(None,), name="target_ids", dtype=tf.int32)
 406   bert_config = utils.get_bert_config_from_params(params)
 407   bert_model_layer = networks.TransformerEncoder(
 408       vocab_size=bert_config.vocab_size,
 409       hidden_size=bert_config.hidden_size,
 410       num_layers=bert_config.num_hidden_layers,
 411       num_attention_heads=bert_config.num_attention_heads,
 412       intermediate_size=bert_config.intermediate_size,
 413       activation=tf_utils.get_activation(bert_config.hidden_act),
 414       dropout_rate=bert_config.hidden_dropout_prob,
 415       attention_dropout_rate=bert_config.attention_probs_dropout_prob,
 416       sequence_length=None,
 417       max_sequence_length=bert_config.max_position_embeddings,
 418       type_vocab_size=bert_config.type_vocab_size,
 419       initializer=tf.keras.initializers.TruncatedNormal(
 420           stddev=bert_config.initializer_range),
 421       return_all_encoder_outputs=True,
 422       name="bert_encoder")
 423   all_encoder_outputs, _ = bert_model_layer(
 424       [input_ids, input_mask, segment_ids])
 425   # pylint: disable=protected-access
 426   decoder_layer = decoder.Decoder(params, bert_model_layer._embedding_layer)
 427   # pylint: enable=protected-access
 428   cross_attention_bias = decoder.AttentionBias(bias_type="single_cross")(
 429       input_ids)
 430   self_attention_bias = decoder.AttentionBias(bias_type="decoder_self")(
 431       target_ids)
 432   decoder_inputs = dict(
 433       attention_bias=cross_attention_bias,
 434       self_attention_bias=self_attention_bias,
 435       target_ids=target_ids,
 436       all_encoder_outputs=all_encoder_outputs)
 437   _ = decoder_layer(decoder_inputs)
 438 
 439   return bert_model_layer, decoder_layer

 442 def get_nhnet_layers(params: configs.NHNetConfig):
 443   """Creates a Mult-doc encoder/decoder.
 444 
 445   Args:
 446     params: ParamsDict.
 447 
 448   Returns:
 449     two keras Layers, bert_model_layer and decoder_layer
 450   """
 451   input_ids = tf.keras.layers.Input(
 452       shape=(None,), name="input_ids", dtype=tf.int32)
 453   input_mask = tf.keras.layers.Input(
 454       shape=(None,), name="input_mask", dtype=tf.int32)
 455   segment_ids = tf.keras.layers.Input(
 456       shape=(None,), name="segment_ids", dtype=tf.int32)
 457   bert_config = utils.get_bert_config_from_params(params)
 458   bert_model_layer = networks.TransformerEncoder(
 459       vocab_size=bert_config.vocab_size,
 460       hidden_size=bert_config.hidden_size,
 461       num_layers=bert_config.num_hidden_layers,
 462       num_attention_heads=bert_config.num_attention_heads,
 463       intermediate_size=bert_config.intermediate_size,
 464       activation=tf_utils.get_activation(bert_config.hidden_act),
 465       dropout_rate=bert_config.hidden_dropout_prob,
 466       attention_dropout_rate=bert_config.attention_probs_dropout_prob,
 467       sequence_length=None,
 468       max_sequence_length=bert_config.max_position_embeddings,
 469       type_vocab_size=bert_config.type_vocab_size,
 470       initializer=tf.keras.initializers.TruncatedNormal(
 471           stddev=bert_config.initializer_range),
 472       return_all_encoder_outputs=True,
 473       name="bert_encoder")
 474   bert_model_layer([input_ids, input_mask, segment_ids])
 475 
 476   input_ids = tf.keras.layers.Input(
 477       shape=(None, None), name="input_ids", dtype=tf.int32)
 478   all_encoder_outputs = tf.keras.layers.Input((None, None, params.hidden_size),
 479                                               dtype=tf.float32)
 480   target_ids = tf.keras.layers.Input(
 481       shape=(None,), name="target_ids", dtype=tf.int32)
 482   doc_attention_probs = tf.keras.layers.Input(
 483       (params.num_decoder_attn_heads, None, None), dtype=tf.float32)
 484   # pylint: disable=protected-access
 485   decoder_layer = decoder.Decoder(params, bert_model_layer._embedding_layer)
 486   # pylint: enable=protected-access
 487   cross_attention_bias = decoder.AttentionBias(bias_type="multi_cross")(
 488       input_ids)
 489   self_attention_bias = decoder.AttentionBias(bias_type="decoder_self")(
 490       target_ids)
 491   decoder_inputs = dict(
 492       attention_bias=cross_attention_bias,
 493       self_attention_bias=self_attention_bias,
 494       target_ids=target_ids,
 495       all_encoder_outputs=all_encoder_outputs,
 496       doc_attention_probs=doc_attention_probs)
 497   _ = decoder_layer(decoder_inputs)
 498 
 499   return bert_model_layer, decoder_layer

 502 def create_transformer_model(params,
 503                              init_checkpoint: Optional[Text] = None
 504                             ) -> tf.keras.Model:
 505   """A helper to create Transformer model."""
 506   bert_layer, decoder_layer = get_bert2bert_layers(params=params)
 507   model = Bert2Bert(
 508       params=params,
 509       bert_layer=bert_layer,
 510       decoder_layer=decoder_layer,
 511       name="transformer")
 512 
 513   if init_checkpoint:
 514     logging.info(
 515         "Checkpoint file %s found and restoring from "
 516         "initial checkpoint.", init_checkpoint)
 517     ckpt = tf.train.Checkpoint(model=model)
 518     ckpt.restore(init_checkpoint).expect_partial()
 519 
 520   return model

 535       decoder_layer=decoder_layer,

 550   if init_checkpoint:
 551     logging.info(
 552         "Checkpoint file %s found and restoring from "
 553         "initial checkpoint.", init_checkpoint)
 554     if params.init_from_bert2bert:
 555       ckpt = tf.train.Checkpoint(model=model)
 556       ckpt.restore(init_checkpoint).assert_existing_objects_matched()
 557     else:
 558       utils.initialize_bert2bert_from_pretrained_bert(bert_layer, decoder_layer,
 559                                                       init_checkpoint)
 560   return model
