  29 def _create_cache(batch_size, init_decode_length, num_heads, head_size):
  30   return {
  31       "key":
  32           tf.zeros([batch_size, init_decode_length, num_heads, head_size],
  33                    dtype=tf.float32),
  34       "value":
  35           tf.zeros([batch_size, init_decode_length, num_heads, head_size],
  36                    dtype=tf.float32)
  37   }
  38

  46   def test_transformer_decoder(self):
  47     decoder_block = decoder.TransformerDecoder(
  48         num_hidden_layers=self._config.num_hidden_layers,
  49         hidden_size=self._config.hidden_size,
  50         num_attention_heads=self._config.num_attention_heads,
  51         intermediate_size=self._config.intermediate_size,
  52         intermediate_activation=self._config.hidden_act,
  53         hidden_dropout_prob=self._config.hidden_dropout_prob,
  54         attention_probs_dropout_prob=self._config.attention_probs_dropout_prob,
  55         initializer_range=self._config.initializer_range)
  56     decoder_block.build(None)
  57     self.assertEqual(len(decoder_block.layers), self._config.num_hidden_layers)

  59   def test_decoder_block_with_cache(self):
  60     decoder_block = decoder.TransformerDecoderBlock(
  61         hidden_size=self._config.hidden_size,
  62         num_attention_heads=self._config.num_attention_heads,
  63         intermediate_size=self._config.intermediate_size,
  64         intermediate_activation=self._config.hidden_act,
  65         hidden_dropout_prob=self._config.hidden_dropout_prob,
  66         attention_probs_dropout_prob=self._config.attention_probs_dropout_prob,
  67         initializer_range=self._config.initializer_range)
  68     # Forward path.
  69     dummy_tensor = tf.zeros([2, 4, self._config.hidden_size], dtype=tf.float32)
  70     dummy_mask = tf.zeros([2, 4, 4], dtype=tf.float32)
  71     inputs = [dummy_tensor, dummy_tensor, dummy_mask, dummy_mask]
  72     cache = _create_cache(
  73         2, 0, self._config.num_attention_heads,
  74         self._config.hidden_size // self._config.num_attention_heads)
  75     output, cache = decoder_block(inputs, cache)
  76     self.assertEqual(output.shape, (2, 4, self._config.hidden_size))
  77     self.assertEqual(cache["value"].shape, (2, 4, 2, 8))

  79   def test_bert_decoder(self):
  80     seq_length = 10
  81     encoder_input_ids = tf.keras.layers.Input(
  82         shape=(seq_length,), name="encoder_input_ids", dtype=tf.int32)
  83     target_ids = tf.keras.layers.Input(
  84         shape=(seq_length,), name="target_ids", dtype=tf.int32)
  85     encoder_outputs = tf.keras.layers.Input(
  86         shape=(seq_length, self._config.hidden_size),
  87         name="all_encoder_outputs",
  88         dtype=tf.float32)
  89     embedding_lookup = layers.OnDeviceEmbedding(
  90         vocab_size=self._config.vocab_size,
  91         embedding_width=self._config.hidden_size,
  92         initializer=tf.keras.initializers.TruncatedNormal(
  93             stddev=self._config.initializer_range),
  94         name="word_embeddings")
  95     cross_attention_bias = decoder.AttentionBias(bias_type="single_cross")(
  96         encoder_input_ids)
  97     self_attention_bias = decoder.AttentionBias(bias_type="decoder_self")(
  98         target_ids)
  99     inputs = dict(
 100         attention_bias=cross_attention_bias,
 101         self_attention_bias=self_attention_bias,
 102         target_ids=target_ids,
 103         all_encoder_outputs=encoder_outputs)
 104     decoder_layer = decoder.Decoder(self._config, embedding_lookup)
 105     outputs = decoder_layer(inputs)
 106     model_inputs = dict(
 107         encoder_input_ids=encoder_input_ids,
 108         target_ids=target_ids,
 109         all_encoder_outputs=encoder_outputs)
 110     model = tf.keras.Model(inputs=model_inputs, outputs=outputs, name="test")
 111     self.assertLen(decoder_layer.trainable_weights, 30)
 112     # Forward path.
 113     fake_inputs = {
 114         "encoder_input_ids": np.zeros((2, 10), dtype=np.int32),
 115         "target_ids": np.zeros((2, 10), dtype=np.int32),
 116         "all_encoder_outputs": np.zeros((2, 10, 16), dtype=np.float32),
 117     }
 118     output_tensor = model(fake_inputs)
 119     self.assertEqual(output_tensor.shape, (2, 10, 16))

 121   def test_multi_doc_decoder(self):
 122     self._config = utils.get_test_params(cls=configs.NHNetConfig)
 123     seq_length = 10
 124     num_docs = 5
 125     encoder_input_ids = tf.keras.layers.Input(
 126         shape=(num_docs, seq_length), name="encoder_input_ids", dtype=tf.int32)
 127     target_ids = tf.keras.layers.Input(
 128         shape=(seq_length,), name="target_ids", dtype=tf.int32)
 129     encoder_outputs = tf.keras.layers.Input(
 130         shape=(num_docs, seq_length, self._config.hidden_size),
 131         name="all_encoder_outputs",
 132         dtype=tf.float32)
 133     embedding_lookup = layers.OnDeviceEmbedding(
 134         vocab_size=self._config.vocab_size,
 135         embedding_width=self._config.hidden_size,
 136         initializer=tf.keras.initializers.TruncatedNormal(
 137             stddev=self._config.initializer_range),
 138         name="word_embeddings")
 139     doc_attention_probs = tf.keras.layers.Input(
 140         shape=(self._config.num_decoder_attn_heads, seq_length, num_docs),
 141         name="doc_attention_probs",
 142         dtype=tf.float32)
 143     cross_attention_bias = decoder.AttentionBias(bias_type="multi_cross")(
 144         encoder_input_ids)
 145     self_attention_bias = decoder.AttentionBias(bias_type="decoder_self")(
 146         target_ids)
 147 
 148     inputs = dict(
 149         attention_bias=cross_attention_bias,
 150         self_attention_bias=self_attention_bias,
 151         target_ids=target_ids,
 152         all_encoder_outputs=encoder_outputs,
 153         doc_attention_probs=doc_attention_probs)
 154 
 155     decoder_layer = decoder.Decoder(self._config, embedding_lookup)
 156     outputs = decoder_layer(inputs)
 157     model_inputs = dict(
 158         encoder_input_ids=encoder_input_ids,
 159         target_ids=target_ids,
 160         all_encoder_outputs=encoder_outputs,
 161         doc_attention_probs=doc_attention_probs)
 162     model = tf.keras.Model(inputs=model_inputs, outputs=outputs, name="test")
 163     self.assertLen(decoder_layer.trainable_weights, 30)
 164     # Forward path.
 165     fake_inputs = {
 166         "encoder_input_ids":
 167             np.zeros((2, num_docs, seq_length), dtype=np.int32),
 168         "target_ids":
 169             np.zeros((2, seq_length), dtype=np.int32),
 170         "all_encoder_outputs":
 171             np.zeros((2, num_docs, seq_length, 16), dtype=np.float32),
 172         "doc_attention_probs":
 173             np.zeros(
 174                 (2, self._config.num_decoder_attn_heads, seq_length, num_docs),
 175                 dtype=np.float32)
 176     }
 177     output_tensor = model(fake_inputs)
 178     self.assertEqual(output_tensor.shape, (2, seq_length, 16))
