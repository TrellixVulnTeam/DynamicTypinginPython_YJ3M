  41 def define_flags():
  42   """Defines command line flags used by NHNet trainer."""
  43   ## Required parameters
  44   flags.DEFINE_enum("mode", "train", ["train", "eval", "train_and_eval"],
  45                     "Execution mode.")
  46   flags.DEFINE_string("train_file_pattern", "", "Train file pattern.")
  47   flags.DEFINE_string("eval_file_pattern", "", "Eval file pattern.")
  48   flags.DEFINE_string(
  49       "model_dir", None,
  50       "The output directory where the model checkpoints will be written.")
  51 
  52   # Model training specific flags.
  53   flags.DEFINE_enum(
  54       "distribution_strategy", "mirrored", ["tpu", "mirrored"],
  55       "Distribution Strategy type to use for training. `tpu` uses TPUStrategy "
  56       "for running on TPUs, `mirrored` uses GPUs with single host.")
  57   flags.DEFINE_string("tpu", "", "TPU address to connect to.")
  58   flags.DEFINE_string(
  59       "init_checkpoint", None,
  60       "Initial checkpoint (usually from a pre-trained BERT model).")
  61   flags.DEFINE_integer("train_steps", 100000, "Max train steps")
  62   flags.DEFINE_integer("eval_steps", 32, "Number of eval steps per run.")
  63   flags.DEFINE_integer("eval_timeout", 3000, "Timeout waiting for checkpoints.")
  64   flags.DEFINE_integer("train_batch_size", 32, "Total batch size for training.")
  65   flags.DEFINE_integer("eval_batch_size", 4, "Total batch size for evaluation.")
  66   flags.DEFINE_integer(
  67       "steps_per_loop", 1000,
  68       "Number of steps per graph-mode loop. Only training step "
  69       "happens inside the loop.")
  70   flags.DEFINE_integer("checkpoint_interval", 2000, "Checkpointing interval.")
  71   flags.DEFINE_integer("len_title", 15, "Title length.")
  72   flags.DEFINE_integer("len_passage", 200, "Passage length.")
  73   flags.DEFINE_integer("num_encoder_layers", 12,
  74                        "Number of hidden layers of encoder.")
  75   flags.DEFINE_integer("num_decoder_layers", 12,
  76                        "Number of hidden layers of decoder.")
  77   flags.DEFINE_string("model_type", "nhnet",
  78                       "Model type to choose a model configuration.")
  79   flags.DEFINE_integer(
  80       "num_nhnet_articles", 5,
  81       "Maximum number of articles in NHNet, only used when model_type=nhnet")
  82   flags.DEFINE_string(
  83       "params_override",
  84       default=None,
  85       help=("a YAML/JSON string or a YAML file which specifies additional "
  86             "overrides over the default parameters"))

  92 class Trainer(tf.keras.Model):
  93   """A training only model."""
  94 
  95   def __init__(self, model, params):
  96     super(Trainer, self).__init__()
  97     self.model = model
  98     self.params = params
  99     self._num_replicas_in_sync = tf.distribute.get_strategy(
 100     ).num_replicas_in_sync
 101 
 102   def call(self, inputs, mode="train"):
 103     return self.model(inputs, mode)
 104 
 105   def train_step(self, inputs):
 106     """The logic for one training step."""
 107     with tf.GradientTape() as tape:
 108       logits, _, _ = self(inputs, mode="train", training=True)
 109       targets = models.remove_sos_from_seq(inputs["target_ids"],
 110                                            self.params.pad_token_id)
 111       loss = transformer_metrics.transformer_loss(logits, targets,
 112                                                   self.params.label_smoothing,
 113                                                   self.params.vocab_size)
 114       # Scales the loss, which results in using the average loss across all
 115       # of the replicas for backprop.
 116       scaled_loss = loss / self._num_replicas_in_sync
 117 
 118     tvars = self.trainable_variables
 119     grads = tape.gradient(scaled_loss, tvars)
 120     self.optimizer.apply_gradients(list(zip(grads, tvars)))
 121     return {
 122         "training_loss": loss,
 123         "learning_rate": self.optimizer._decayed_lr(var_dtype=tf.float32)
 124     }
 125

 127 def train(params, strategy, dataset=None):
 128   """Runs training."""
 129 
 130   if not dataset:
 131     dataset = input_pipeline.get_input_dataset(
 132         FLAGS.train_file_pattern,
 133         FLAGS.train_batch_size,
 134         params,
 135         is_training=True,
 136         strategy=strategy)
 137 
 138   with strategy.scope():
 139     model = models.create_model(
 140         FLAGS.model_type, params, init_checkpoint=FLAGS.init_checkpoint)
 141     opt = optimizer.create_optimizer(params)
 142     trainer = Trainer(model, params)
 143     model.global_step = opt.iterations
 144 
 145     trainer.compile(
 146         optimizer=opt,
 147         experimental_steps_per_execution=FLAGS.steps_per_loop)
 148     summary_dir = os.path.join(FLAGS.model_dir, "summaries")
 149     summary_callback = tf.keras.callbacks.TensorBoard(
 150         summary_dir, update_freq=max(100, FLAGS.steps_per_loop))
 151     checkpoint = tf.train.Checkpoint(model=model, optimizer=opt)
 152     checkpoint_manager = tf.train.CheckpointManager(
 153         checkpoint,
 154         directory=FLAGS.model_dir,
 155         max_to_keep=10,
 156         step_counter=model.global_step,
 157         checkpoint_interval=FLAGS.checkpoint_interval)
 158     if checkpoint_manager.restore_or_initialize():
 159       logging.info("Training restored from the checkpoints in: %s",
 160                    FLAGS.model_dir)
 161     checkpoint_callback = keras_utils.SimpleCheckpoint(checkpoint_manager)
 162 
 163   # Trains the model.
 164   steps_per_epoch = min(FLAGS.train_steps, FLAGS.checkpoint_interval)
 165   epochs = FLAGS.train_steps // steps_per_epoch
 166   history = trainer.fit(
 167       x=dataset,
 168       steps_per_epoch=steps_per_epoch,
 169       epochs=epochs,
 170       callbacks=[summary_callback, checkpoint_callback],
 171       verbose=2)
 172   train_hist = history.history
 173   # Gets final loss from training.
 174   stats = dict(training_loss=float(train_hist["training_loss"][-1]))
 175   return stats

 178 def run():
 179   """Runs NHNet using Keras APIs."""
 180   strategy = distribution_utils.get_distribution_strategy(
 181       distribution_strategy=FLAGS.distribution_strategy, tpu_address=FLAGS.tpu)
 182   if strategy:
 183     logging.info("***** Number of cores used : %d",
 184                  strategy.num_replicas_in_sync)
 185 
 186   params = models.get_model_params(FLAGS.model_type)
 187   params = params_dict.override_params_dict(
 188       params, FLAGS.params_override, is_strict=True)
 189   params.override(
 190       {
 191           "len_title":
 192               FLAGS.len_title,
 193           "len_passage":
 194               FLAGS.len_passage,
 195           "num_hidden_layers":
 196               FLAGS.num_encoder_layers,
 197           "num_decoder_layers":
 198               FLAGS.num_decoder_layers,
 199           "passage_list":
 200               [chr(ord("b") + i) for i in range(FLAGS.num_nhnet_articles)],
 201       },
 202       is_strict=False)
 203   stats = {}
 204   if "train" in FLAGS.mode:
 205     stats = train(params, strategy)
 206   if "eval" in FLAGS.mode:
 207     timeout = 0 if FLAGS.mode == "train_and_eval" else FLAGS.eval_timeout
 208     # Uses padded decoding for TPU. Always uses cache.
 209     padded_decode = isinstance(strategy, tf.distribute.experimental.TPUStrategy)
 210     params.override({
 211         "padded_decode": padded_decode,
 212     }, is_strict=False)
 213     stats = evaluation.continuous_eval(
 214         strategy,
 215         params,
 216         model_type=FLAGS.model_type,
 217         eval_file_pattern=FLAGS.eval_file_pattern,
 218         batch_size=FLAGS.eval_batch_size,
 219         eval_steps=FLAGS.eval_steps,
 220         model_dir=FLAGS.model_dir,
 221         timeout=timeout)
 222   return stats
 223

 225 def main(_):
 226   stats = run()
 227   if stats:
 228     logging.info("Stats:\n%s", stats)

 230 if __name__ == "__main__":
 231   define_flags()
 232   app.run(main)
