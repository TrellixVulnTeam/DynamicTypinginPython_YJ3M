  29 class TransformerDecoderBlock(tf.keras.layers.Layer):
  30   """Single transformer layer for decoder.
  31 
  32   It has three sub-layers:
  33   (1) a multi-head self-attention mechanism.
  34   (2) a encoder-decoder attention.
  35   (3) a positionwise fully connected feed-forward network.
  36   """
  37 
  38   def __init__(self,
  39                hidden_size=768,
  40                num_attention_heads=12,
  41                intermediate_size=3072,
  42                intermediate_activation="gelu",
  43                hidden_dropout_prob=0.0,
  44                attention_probs_dropout_prob=0.0,
  45                initializer_range=0.02,
  46                multi_channel_cross_attention=False,
  47                **kwargs):
  48     super(TransformerDecoderBlock, self).__init__(**kwargs)
  49     self.hidden_size = hidden_size
  50     self.num_attention_heads = num_attention_heads
  51     self.intermediate_size = intermediate_size
  52     self.intermediate_activation = tf_utils.get_activation(
  53         intermediate_activation)
  54     self.hidden_dropout_prob = hidden_dropout_prob
  55     self.attention_probs_dropout_prob = attention_probs_dropout_prob
  56     self.multi_channel_cross_attention = multi_channel_cross_attention
  57     self._kernel_initializer = tf.keras.initializers.TruncatedNormal(
  58         stddev=initializer_range)
  59     self._bias_initializer = tf.keras.initializers.get("zeros")
  60     if self.multi_channel_cross_attention:
  61       self._cross_attention_cls = multi_channel_attention.MultiChannelAttention
  62     else:
  63       self._cross_attention_cls = layers.MultiHeadAttention
  64 
  65     if self.hidden_size % self.num_attention_heads != 0:
  66       raise ValueError(
  67           "The hidden size (%d) is not a multiple of the number of attention "
  68           "heads (%d)" % (self.hidden_size, self.num_attention_heads))
  69     self.attention_head_size = int(self.hidden_size / self.num_attention_heads)
  70 
  71   def build(self, input_shape):
  72     # Self attention.
  73     self.self_attention = layers.CachedAttention(
  74         num_heads=self.num_attention_heads,
  75         key_size=self.attention_head_size,
  76         dropout=self.attention_probs_dropout_prob,
  77         kernel_initializer=self._kernel_initializer,
  78         name="self_attention")
  79     self.self_attention_output_dense = layers.DenseEinsum(
  80         output_shape=self.hidden_size,
  81         num_summed_dimensions=2,
  82         kernel_initializer=self._kernel_initializer,
  83         bias_initializer=self._bias_initializer,
  84         name="self_attention_output")
  85     self.self_attention_dropout = tf.keras.layers.Dropout(
  86         rate=self.hidden_dropout_prob)
  87     self.self_attention_layer_norm = (
  88         tf.keras.layers.LayerNormalization(
  89             name="self_attention_layer_norm", axis=-1, epsilon=1e-12))
  90     # Encoder-decoder attention.
  91     self.encdec_attention = self._cross_attention_cls(
  92         num_heads=self.num_attention_heads,
  93         key_size=self.attention_head_size,
  94         dropout=self.attention_probs_dropout_prob,
  95         output_shape=self.hidden_size,
  96         kernel_initializer=self._kernel_initializer,
  97         name="attention/encdec")
  98 
  99     self.encdec_attention_dropout = tf.keras.layers.Dropout(
 100         rate=self.hidden_dropout_prob)
 101     self.encdec_attention_layer_norm = (
 102         tf.keras.layers.LayerNormalization(
 103             name="attention/encdec_output_layer_norm", axis=-1, epsilon=1e-12))
 104 
 105     # Feed-forward projection.
 106     self.intermediate_dense = layers.DenseEinsum(
 107         output_shape=self.intermediate_size,
 108         activation=None,
 109         kernel_initializer=self._kernel_initializer,
 110         bias_initializer=self._bias_initializer,
 111         name="intermediate")
 112     self.intermediate_activation_layer = tf.keras.layers.Activation(
 113         self.intermediate_activation)
 114     self.output_dense = layers.DenseEinsum(
 115         output_shape=self.hidden_size,
 116         kernel_initializer=self._kernel_initializer,
 117         bias_initializer=self._bias_initializer,
 118         name="output")
 119     self.output_dropout = tf.keras.layers.Dropout(rate=self.hidden_dropout_prob)
 120     self.output_layer_norm = tf.keras.layers.LayerNormalization(
 121         name="output_layer_norm", axis=-1, epsilon=1e-12)
 122     super(TransformerDecoderBlock, self).build(input_shape)
 123 
 124   def common_layers_with_encoder(self):
 125     """Gets layer objects that can make a Transformer encoder block."""
 126     return [
 127         self.self_attention, self.self_attention_layer_norm,
 128         self.intermediate_dense, self.output_dense, self.output_layer_norm
 129     ]
 130 
 131   def call(self, inputs, cache=None, decode_loop_step=None):
 132     if self.multi_channel_cross_attention:
 133       if len(inputs) != 5:
 134         raise ValueError(
 135             "TransformerDecoderBlock must have 5 inputs, when it uses "
 136             "multi_channel_cross_attention. But it got: %d" % len(inputs))
 137     elif len(inputs) != 4:
 138       raise ValueError(
 139           "TransformerDecoderBlock must have 4 inputs, but it got: %d" %
 140           len(inputs))
 141     input_tensor, memory, attention_mask, self_attention_mask = inputs[:4]
 142     self_attention_inputs = [input_tensor, input_tensor]
 143     self_attention_output, cache = self.self_attention(
 144         self_attention_inputs,
 145         attention_mask=self_attention_mask,
 146         cache=cache,
 147         decode_loop_step=decode_loop_step)
 148     self_attention_output = self.self_attention_dropout(self_attention_output)
 149     self_attention_output = self.self_attention_layer_norm(
 150         input_tensor + self_attention_output)
 151 
 152     cross_attn_inputs = [self_attention_output, memory]
 153     if self.multi_channel_cross_attention:
 154       # Accesses the 5-th input tensor for the doc-attention probabilities.
 155       cross_attn_inputs.append(inputs[-1])
 156     attention_output = self.encdec_attention(cross_attn_inputs, attention_mask)
 157     attention_output = self.encdec_attention_dropout(attention_output)
 158     attention_output = self.encdec_attention_layer_norm(self_attention_output +
 159                                                         attention_output)
 160 
 161     intermediate_output = self.intermediate_dense(attention_output)
 162     intermediate_output = self.intermediate_activation_layer(
 163         intermediate_output)
 164     layer_output = self.output_dense(intermediate_output)
 165     layer_output = self.output_dropout(layer_output)
 166     layer_output = self.output_layer_norm(layer_output + attention_output)
 167     return layer_output, cache

 170 class TransformerDecoder(tf.keras.layers.Layer):
 171   """Transformer decoder stack."""
 172 
 173   def __init__(self,
 174                num_hidden_layers=12,
 175                hidden_size=768,
 176                num_attention_heads=12,
 177                intermediate_size=3072,
 178                intermediate_activation="gelu",
 179                hidden_dropout_prob=0.0,
 180                attention_probs_dropout_prob=0.0,
 181                initializer_range=0.02,
 182                attend_to_last_layer=True,
 183                multi_channel_cross_attention=False,
 184                **kwargs):
 185     super(TransformerDecoder, self).__init__(**kwargs)
 186     self.num_hidden_layers = num_hidden_layers
 187     self.hidden_size = hidden_size
 188     self.num_attention_heads = num_attention_heads
 189     self.intermediate_size = intermediate_size
 190     self.intermediate_activation = tf_utils.get_activation(
 191         intermediate_activation)
 192     self.hidden_dropout_prob = hidden_dropout_prob
 193     self.attention_probs_dropout_prob = attention_probs_dropout_prob
 194     self.initializer_range = initializer_range
 195     self.attend_to_last_layer = attend_to_last_layer
 196     self.multi_channel_cross_attention = multi_channel_cross_attention
 197 
 198   def build(self, unused_input_shapes):
 199     """Implements build() for the layer."""
 200     self.layers = []
 201     for i in range(self.num_hidden_layers):
 202       self.layers.append(
 203           TransformerDecoderBlock(
 204               hidden_size=self.hidden_size,
 205               num_attention_heads=self.num_attention_heads,
 206               intermediate_size=self.intermediate_size,
 207               intermediate_activation=self.intermediate_activation,
 208               hidden_dropout_prob=self.hidden_dropout_prob,
 209               attention_probs_dropout_prob=self.attention_probs_dropout_prob,
 210               initializer_range=self.initializer_range,
 211               multi_channel_cross_attention=self.multi_channel_cross_attention,
 212               name=("layer_%d" % i)))
 213     super(TransformerDecoder, self).build(unused_input_shapes)
 214 
 215   def call(self, inputs, cache=None, decode_loop_step=None):
 216     """Return the output of the decoder layer stacks.
 217 
 218     Args:
 219       inputs: A dictionary of inputs. `decoder_inputs` is a tf.int32 tensor for
 220         input ids. `encoder_outputs` is a list of tensors with shape
 221         [batch_size, input_length, hidden_size]. `self_attention_mask` is the
 222         bias for decoder self-attention layer. [1, 1, target_length,
 223         target_length]. `attention_mask` is the bias for encoder-decoder
 224         attention layer, [batch_size, 1, 1, input_length].
 225       cache: A dictionary of cache tensors, including key & value attentions.
 226       decode_loop_step: an integer to indicate the step inside a decoding loop.
 227 
 228     Returns:
 229       Output of decoder layer stack.
 230       float32 tensor with shape [batch_size, target_length, hidden_size]
 231     """
 232     decoder_inputs = inputs["decoder_inputs"]
 233     encoder_outputs = inputs["encoder_outputs"]
 234     self_attention_mask = inputs["self_attention_mask"]
 235     attention_mask = inputs["attention_mask"]
 236     decoder_shape = tf_utils.get_shape_list(decoder_inputs, expected_rank=3)
 237     batch_size = decoder_shape[0]
 238     decoder_length = decoder_shape[1]
 239 
 240     def _to_bert_self_attention_mask(matrix):
 241       """[1, 1, target_len, target_len] -> [bs, target_len, target_len]."""
 242       matrix = tf.squeeze(matrix, axis=[1])
 243       matrix = tf.tile(matrix, [batch_size, 1, 1])
 244       return matrix
 245 
 246     def _to_bert_encdec_attention_mask(matrix):
 247       """[bs, 1, 1, input_len] -> [bs, target_len, input_len]."""
 248       if self.multi_channel_cross_attention:
 249         matrix = tf.expand_dims(matrix, axis=2)
 250         matrix = tf.tile(matrix, [1, 1, decoder_length, 1])
 251       else:
 252         matrix = tf.squeeze(matrix, axis=[1])
 253         matrix = tf.tile(matrix, [1, decoder_length, 1])
 254       return matrix
 255 
 256     attention_mask = _to_bert_encdec_attention_mask(attention_mask)
 257     self_attention_mask = _to_bert_self_attention_mask(self_attention_mask)
 258 
 259     output_tensor = decoder_inputs
 260     for layer_idx in range(self.num_hidden_layers):
 261       if self.attend_to_last_layer:
 262         memory = encoder_outputs[-1]
 263       else:
 264         memory = encoder_outputs[layer_idx]
 265       if self.multi_channel_cross_attention:
 266         transformer_inputs = [
 267             output_tensor, memory, attention_mask, self_attention_mask,
 268             inputs["doc_attention_probs"]
 269         ]
 270       else:
 271         transformer_inputs = [
 272             output_tensor, memory, attention_mask, self_attention_mask
 273         ]
 274       # Gets the cache for decoding.
 275       if cache is None:
 276         output_tensor, _ = self.layers[layer_idx](transformer_inputs)
 277       else:
 278         cache_layer_idx = str(layer_idx)
 279         output_tensor, cache[cache_layer_idx] = self.layers[layer_idx](
 280             transformer_inputs,
 281             cache=cache[cache_layer_idx],
 282             decode_loop_step=decode_loop_step)
 283     return output_tensor, cache

 286 def get_attention_bias(input_tensor,
 287                        bias_type,
 288                        padding_value=0,
 289                        max_length=None):
 290   """A helper function to get various attention bias tensors."""
 291   if bias_type not in ("single_cross", "multi_cross", "decoder_self"):
 292     raise ValueError("Invalid attention bias type: %s" % bias_type)
 293   if bias_type == "single_cross":
 294     length = tf_utils.get_shape_list(input_tensor, expected_rank=2)[1]
 295     bias = transformer_utils.get_padding_bias(
 296         input_tensor, padding_value=padding_value)
 297   elif bias_type == "multi_cross":
 298     length = tf_utils.get_shape_list(input_tensor, expected_rank=3)[2]
 299     padding = transformer_utils.get_padding(
 300         input_tensor, padding_value=padding_value)
 301     bias = padding * -1e9
 302   else:
 303     if max_length is not None:
 304       length = max_length
 305     else:
 306       length = tf_utils.get_shape_list(input_tensor, expected_rank=2)[1]
 307     bias = transformer_utils.get_decoder_self_attention_bias(length)
 308 
 309   return tf.where(bias < 0, tf.zeros_like(bias), tf.ones_like(bias))

 322 class EmbeddingPostprocessor(tf.keras.layers.Layer):
 323   """Performs various post-processing on a word embedding tensor."""
 324 
 325   def __init__(self,
 326                use_type_embeddings=False,
 327                token_type_vocab_size=None,
 328                use_position_embeddings=True,
 329                max_position_embeddings=512,
 330                dropout_prob=0.0,
 331                initializer_range=0.02,
 332                initializer=None,
 333                **kwargs):
 334     super(EmbeddingPostprocessor, self).__init__(**kwargs)
 335     self.use_type_embeddings = use_type_embeddings
 336     self.token_type_vocab_size = token_type_vocab_size
 337     self.use_position_embeddings = use_position_embeddings
 338     self.max_position_embeddings = max_position_embeddings
 339     self.dropout_prob = dropout_prob
 340     self.initializer_range = initializer_range
 341 
 342     if not initializer:
 343       self.initializer = tf.keras.initializers.TruncatedNormal(
 344           stddev=initializer_range)
 345     else:
 346       self.initializer = initializer
 347 
 348     if self.use_type_embeddings and not self.token_type_vocab_size:
 349       raise ValueError("If `use_type_embeddings` is True, then "
 350                        "`token_type_vocab_size` must be specified.")
 351 
 352   def build(self, input_shapes):
 353     """Implements build() for the layer."""
 354     (word_embeddings_shape, _) = input_shapes
 355     width = word_embeddings_shape.as_list()[-1]
 356     self.type_embeddings = None
 357     if self.use_type_embeddings:
 358       self.type_embeddings = self.add_weight(
 359           "type_embeddings",
 360           shape=[self.token_type_vocab_size, width],
 361           initializer=tf.keras.initializers.TruncatedNormal(
 362               stddev=self.initializer_range),
 363           dtype=self.dtype)
 364 
 365     self.position_embeddings = None
 366     if self.use_position_embeddings:
 367       self.position_embeddings = self.add_weight(
 368           "position_embeddings",
 369           shape=[self.max_position_embeddings, width],
 370           initializer=tf.keras.initializers.TruncatedNormal(
 371               stddev=self.initializer_range),
 372           dtype=self.dtype)
 373 
 374     self.output_layer_norm = tf.keras.layers.LayerNormalization(
 375         name="layer_norm", axis=-1, epsilon=1e-12, dtype=tf.float32)
 376     self.output_dropout = tf.keras.layers.Dropout(
 377         rate=self.dropout_prob, dtype=tf.float32)
 378     super(EmbeddingPostprocessor, self).build(input_shapes)
 379 
 380   def __call__(self, word_embeddings, token_type_ids=None, **kwargs):
 381     inputs = tf_utils.pack_inputs([word_embeddings, token_type_ids])
 382     return super(EmbeddingPostprocessor, self).__call__(inputs, **kwargs)
 383 
 384   def call(self, inputs):
 385     """Implements call() for the layer."""
 386     unpacked_inputs = tf_utils.unpack_inputs(inputs)
 387     word_embeddings = unpacked_inputs[0]
 388     token_type_ids = unpacked_inputs[1]
 389     input_shape = tf_utils.get_shape_list(word_embeddings, expected_rank=3)
 390     batch_size = input_shape[0]
 391     seq_length = input_shape[1]
 392     width = input_shape[2]
 393 
 394     output = word_embeddings
 395     if self.use_type_embeddings:
 396       flat_token_type_ids = tf.reshape(token_type_ids, [-1])
 397       token_type_embeddings = tf.gather(self.type_embeddings,
 398                                         flat_token_type_ids)
 399       token_type_embeddings = tf.reshape(token_type_embeddings,
 400                                          [batch_size, seq_length, width])
 401       output += token_type_embeddings
 402 
 403     if self.use_position_embeddings:
 404       position_embeddings = tf.expand_dims(
 405           tf.slice(self.position_embeddings, [0, 0], [seq_length, width]),
 406           axis=0)
 407 
 408       output += position_embeddings
 409 
 410     output = self.output_layer_norm(output)
 411     output = self.output_dropout(output)
 412 
 413     return output

 416 class Decoder(tf.keras.layers.Layer):
 417   """The decoder network which can reuse encoder embeddings for target."""
 418 
 419   def __init__(self, config, embedding_lookup=None, **kwargs):
 420     super(Decoder, self).__init__(**kwargs)
 421     self.config = config
 422     # Shares vocabulary embedding.
 423     self.embedding_lookup = None
 424     if embedding_lookup:
 425       self.embedding_lookup = embedding_lookup
 426 
 427   def build(self, unused_input_shapes):
 428     """Implements build() for the layer."""
 429     if self.embedding_lookup is None:
 430       self.embedding_lookup = layers.OnDeviceEmbedding(
 431           vocab_size=self.config.vocab_size,
 432           embedding_width=self.config.hidden_size,
 433           initializer=tf.keras.initializers.TruncatedNormal(
 434               stddev=self.config.initializer_range),
 435           name="target_embeddings")
 436     self.embedding_postprocessor = EmbeddingPostprocessor(
 437         use_type_embeddings=False,
 438         use_position_embeddings=True,
 439         max_position_embeddings=self.config.max_position_embeddings,
 440         dropout_prob=self.config.hidden_dropout_prob,
 441         initializer=tf.keras.initializers.VarianceScaling(
 442             scale=self.config.initializer_gain,
 443             mode="fan_avg",
 444             distribution="uniform"),
 445         name="embedding_postprocessor")
 446     # Decoder can use a different intermediate size.
 447     self.multi_channel_cross_attention = self.config.get(
 448         "multi_channel_cross_attention", False)
 449     self.decoder = TransformerDecoder(
 450         num_hidden_layers=self.config.num_decoder_layers,
 451         hidden_size=self.config.hidden_size,
 452         num_attention_heads=self.config.num_decoder_attn_heads,
 453         intermediate_size=self.config.decoder_intermediate_size,
 454         intermediate_activation=self.config.hidden_act,
 455         hidden_dropout_prob=self.config.hidden_dropout_prob,
 456         attention_probs_dropout_prob=self.config.attention_probs_dropout_prob,
 457         initializer_range=self.config.initializer_range,
 458         multi_channel_cross_attention=self.multi_channel_cross_attention,
 459         name="decoder")
 460     super(Decoder, self).build(unused_input_shapes)
 461 
 462   def _decoding_step_time_signal(self, target_embeds, decode_loop_step):
 463     """Applies time signal (positional embeddings) for decoded embeddings."""
 464     # TODO(hongkuny): migrate to keras bert and design a module to handle this.
 465     output = target_embeds
 466     if self.embedding_postprocessor.use_position_embeddings:
 467       position_embeddings = tf.gather(
 468           self.embedding_postprocessor.position_embeddings, [decode_loop_step])
 469       # Broadcasts to all sequences inside a batch.
 470       output += position_embeddings
 471 
 472     output = self.embedding_postprocessor.output_layer_norm(output)
 473     output = self.embedding_postprocessor.output_dropout(output)
 474     return output
 475 
 476   def call(self,
 477            inputs,
 478            cache=None,
 479            decode_loop_step=None,
 480            padded_decode=False):
 481     """Implements call() for the layer.
 482 
 483     Args:
 484       inputs: a list of input tensors.
 485       cache: A dictionary of cache tensors, including key & value attentions.
 486         Due to the limit of keras, we uses the side effect to update cache and
 487         states of tensors will be mutated.
 488       decode_loop_step: an integer to indicate the step inside a decoding loop.
 489       padded_decode: a boolean indicates if the pass is for padded decoding.
 490 
 491     Returns:
 492       Decoder output tensors.
 493     """
 494     attention_bias = inputs["attention_bias"]
 495     target_ids = inputs["target_ids"]
 496     all_encoder_outputs = inputs["all_encoder_outputs"]
 497     self_attention_bias = inputs["self_attention_bias"]
 498     if not isinstance(all_encoder_outputs, list):
 499       all_encoder_outputs = [all_encoder_outputs]
 500 
 501     target_embeds = self.embedding_lookup(target_ids)
 502     if decode_loop_step is None:
 503       target_embeds = self.embedding_postprocessor(target_embeds)
 504     else:
 505       target_embeds = self._decoding_step_time_signal(target_embeds,
 506                                                       decode_loop_step)
 507     decoder_inputs = dict(
 508         decoder_inputs=target_embeds,
 509         encoder_outputs=all_encoder_outputs,
 510         self_attention_mask=self_attention_bias,
 511         attention_mask=attention_bias)
 512     if self.multi_channel_cross_attention:
 513       decoder_inputs["doc_attention_probs"] = inputs["doc_attention_probs"]
 514     decode_outputs, cache = self.decoder(
 515         decoder_inputs, cache, decode_loop_step if padded_decode else None)
 516     return decode_outputs
