 107 class ConfigsTest(tf.test.TestCase):

 109   def test_configs(self):
 110     cfg = configs.BERT2BERTConfig()

 112     self.assertEqual(cfg.as_dict(), BERT2BERT_CONFIG)

 114   def test_nhnet_config(self):
