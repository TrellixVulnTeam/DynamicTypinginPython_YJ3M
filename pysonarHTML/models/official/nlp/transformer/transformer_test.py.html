  15 """Test Transformer model."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import tensorflow as tf
  22 
  23 from official.nlp.transformer import model_params
  24 from official.nlp.transformer import transformer
  25 
  26 
  27 class TransformerV2Test(tf.test.TestCase):
  28 
  29   def setUp(self):
  30     self.params = params = model_params.TINY_PARAMS
  31     params["batch_size"] = params["default_batch_size"] = 16
  32     params["use_synthetic_data"] = True
  33     params["hidden_size"] = 12
  34     params["num_hidden_layers"] = 2
  35     params["filter_size"] = 14
  36     params["num_heads"] = 2
  37     params["vocab_size"] = 41
  38     params["extra_decode_length"] = 2
  39     params["beam_size"] = 3
  40     params["dtype"] = tf.float32
  41 
  42   def test_create_model_train(self):
  43     model = transformer.create_model(self.params, True)
  44     inputs, outputs = model.inputs, model.outputs
  45     self.assertEqual(len(inputs), 2)
  46     self.assertEqual(len(outputs), 1)
  47     self.assertEqual(inputs[0].shape.as_list(), [None, None])
  48     self.assertEqual(inputs[0].dtype, tf.int64)
  49     self.assertEqual(inputs[1].shape.as_list(), [None, None])
  50     self.assertEqual(inputs[1].dtype, tf.int64)
  51     self.assertEqual(outputs[0].shape.as_list(), [None, None, 41])
  52     self.assertEqual(outputs[0].dtype, tf.float32)
  53 
  54   def test_create_model_not_train(self):
  55     model = transformer.create_model(self.params, False)
  56     inputs, outputs = model.inputs, model.outputs
  57     self.assertEqual(len(inputs), 1)
  58     self.assertEqual(len(outputs), 2)
  59     self.assertEqual(inputs[0].shape.as_list(), [None, None])
  60     self.assertEqual(inputs[0].dtype, tf.int64)
  61     self.assertEqual(outputs[0].shape.as_list(), [None, None])
  62     self.assertEqual(outputs[0].dtype, tf.int32)
  63     self.assertEqual(outputs[1].shape.as_list(), [None])
  64     self.assertEqual(outputs[1].dtype, tf.float32)
  65 
  66 
  67 if __name__ == "__main__":
  68   tf.test.main()
