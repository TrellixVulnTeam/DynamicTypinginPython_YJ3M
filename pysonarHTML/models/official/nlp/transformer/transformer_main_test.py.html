  39 def _generate_file(filepath, lines):

  45 class TransformerTaskTest(tf.test.TestCase):
  46   local_flags = None

  48   def setUp(self):
  49     temp_dir = self.get_temp_dir()
  50     if TransformerTaskTest.local_flags is None:
  51       misc.define_transformer_flags()
  52       # Loads flags, array cannot be blank.
  53       flags.FLAGS(['foo'])
  54       TransformerTaskTest.local_flags = flagsaver.save_flag_values()
  55     else:
  56       flagsaver.restore_flag_values(TransformerTaskTest.local_flags)
  57     FLAGS.model_dir = os.path.join(temp_dir, FIXED_TIMESTAMP)
  58     FLAGS.param_set = 'tiny'
  59     FLAGS.use_synthetic_data = True
  60     FLAGS.steps_between_evals = 1
  61     FLAGS.train_steps = 2
  62     FLAGS.validation_steps = 1
  63     FLAGS.batch_size = 8
  64     FLAGS.max_length = 1
  65     FLAGS.num_gpus = 1
  66     FLAGS.distribution_strategy = 'off'
  67     FLAGS.dtype = 'fp32'
  68     self.model_dir = FLAGS.model_dir
  69     self.temp_dir = temp_dir
  70     self.vocab_file = os.path.join(temp_dir, 'vocab')
  71     self.vocab_size = misc.get_model_params(FLAGS.param_set, 0)['vocab_size']
  72     self.bleu_source = os.path.join(temp_dir, 'bleu_source')
  73     self.bleu_ref = os.path.join(temp_dir, 'bleu_ref')
  74     self.orig_policy = (
  75         tf.compat.v2.keras.mixed_precision.experimental.global_policy())
  76 
  77   def tearDown(self):
  78     tf.compat.v2.keras.mixed_precision.experimental.set_policy(self.orig_policy)

 139   def _prepare_files_and_flags(self, *extra_flags):
 140     # Make log dir.
 141     if not os.path.exists(self.temp_dir):
 142       os.makedirs(self.temp_dir)
 143 
 144     # Fake vocab, bleu_source and bleu_ref.
 145     tokens = [
 146         "'<pad>'", "'<EOS>'", "'_'", "'a'", "'b'", "'c'", "'d'", "'a_'", "'b_'",
 147         "'c_'", "'d_'"
 148     ]
 149     tokens += ["'{}'".format(i) for i in range(self.vocab_size - len(tokens))]
 150     _generate_file(self.vocab_file, tokens)
 151     _generate_file(self.bleu_source, ['a b', 'c d'])
 152     _generate_file(self.bleu_ref, ['a b', 'd c'])
 153 
 154     # Update flags.
 155     update_flags = [
 156         'ignored_program_name',
 157         '--vocab_file={}'.format(self.vocab_file),
 158         '--bleu_source={}'.format(self.bleu_source),
 159         '--bleu_ref={}'.format(self.bleu_ref),
 160     ]
 161     if extra_flags:
 162       update_flags.extend(extra_flags)
 163     FLAGS(update_flags)
