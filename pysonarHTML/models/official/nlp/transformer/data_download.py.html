 100 def find_file(path, filename, max_depth=5):
 101   """Returns full filepath if the file is in path or a subdirectory."""
 102   for root, dirs, files in os.walk(path):
 103     if filename in files:
 104       return os.path.join(root, filename)
 105 
 106     # Don't search past max_depth
 107     depth = root[len(path) + 1:].count(os.sep)
 108     if depth > max_depth:
 109       del dirs[:]  # Clear dirs
 110   return None

 116 def get_raw_files(raw_dir, data_source):
 117   """Return raw files from source. Downloads/extracts if needed.
 118 
 119   Args:
 120     raw_dir: string directory to store raw files
 121     data_source: dictionary with
 122       {"url": url of compressed dataset containing input and target files
 123        "input": file with data in input language
 124        "target": file with data in target language}
 125 
 126   Returns:
 127     dictionary with
 128       {"inputs": list of files containing data in input language
 129        "targets": list of files containing corresponding data in target language
 130       }
 131   """
 132   raw_files = {
 133       "inputs": [],
 134       "targets": [],
 135   }  # keys
 136   for d in data_source:
 137     input_file, target_file = download_and_extract(
 138         raw_dir, d["url"], d["input"], d["target"])
 139     raw_files["inputs"].append(input_file)
 140     raw_files["targets"].append(target_file)
 141   return raw_files

 144 def download_report_hook(count, block_size, total_size):
 145   """Report hook for download progress.
 146 
 147   Args:
 148     count: current block number
 149     block_size: block size
 150     total_size: total size
 151   """
 152   percent = int(count * block_size * 100 / total_size)
 153   print(six.ensure_str("\r%d%%" % percent) + " completed", end="\r")

 156 def download_from_url(path, url):
 157   """Download content from a url.
 158 
 159   Args:
 160     path: string directory where file will be downloaded
 161     url: string url
 162 
 163   Returns:
 164     Full path to downloaded file
 165   """
 166   filename = six.ensure_str(url).split("/")[-1]
 167   found_file = find_file(path, filename, max_depth=0)
 168   if found_file is None:
 169     filename = os.path.join(path, filename)
 170     logging.info("Downloading from %s to %s." % (url, filename))
 171     inprogress_filepath = six.ensure_str(filename) + ".incomplete"
 172     inprogress_filepath, _ = urllib.request.urlretrieve(
 173         url, inprogress_filepath, reporthook=download_report_hook)
 174     # Print newline to clear the carriage return from the download progress.
 175     print()
 176     tf.gfile.Rename(inprogress_filepath, filename)
 177     return filename
 178   else:
 179     logging.info("Already downloaded: %s (at %s)." % (url, found_file))
 180     return found_file

 183 def download_and_extract(path, url, input_filename, target_filename):
 184   """Extract files from downloaded compressed archive file.
 185 
 186   Args:
 187     path: string directory where the files will be downloaded
 188     url: url containing the compressed input and target files
 189     input_filename: name of file containing data in source language
 190     target_filename: name of file containing data in target language
 191 
 192   Returns:
 193     Full paths to extracted input and target files.
 194 
 195   Raises:
 196     OSError: if the the download/extraction fails.
 197   """
 198   # Check if extracted files already exist in path
 199   input_file = find_file(path, input_filename)
 200   target_file = find_file(path, target_filename)
 201   if input_file and target_file:
 202     logging.info("Already downloaded and extracted %s." % url)
 203     return input_file, target_file
 204 
 205   # Download archive file if it doesn't already exist.
 206   compressed_file = download_from_url(path, url)
 207 
 208   # Extract compressed files
 209   logging.info("Extracting %s." % compressed_file)
 210   with tarfile.open(compressed_file, "r:gz") as corpus_tar:
 211     corpus_tar.extractall(path)
 212 
 213   # Return file paths of the requested files.
 214   input_file = find_file(path, input_filename)
 215   target_file = find_file(path, target_filename)
 216 
 217   if input_file and target_file:
 218     return input_file, target_file
 219 
 220   raise OSError("Download/extraction failed for url %s to path %s" %
 221                 (url, path))

 224 def txt_line_iterator(path):
 225   """Iterate through lines of file."""
 226   with tf.io.gfile.GFile(path) as f:
 227     for line in f:
 228       yield line.strip()

 231 def compile_files(raw_dir, raw_files, tag):
 232   """Compile raw files into a single file for each language.
 233 
 234   Args:
 235     raw_dir: Directory containing downloaded raw files.
 236     raw_files: Dict containing filenames of input and target data.
 237       {"inputs": list of files containing data in input language
 238        "targets": list of files containing corresponding data in target language
 239       }
 240     tag: String to append to the compiled filename.
 241 
 242   Returns:
 243     Full path of compiled input and target files.
 244   """
 245   logging.info("Compiling files with tag %s." % tag)
 246   filename = "%s-%s" % (_PREFIX, tag)
 247   input_compiled_file = os.path.join(raw_dir,
 248                                      six.ensure_str(filename) + ".lang1")
 249   target_compiled_file = os.path.join(raw_dir,
 250                                       six.ensure_str(filename) + ".lang2")
 251 
 252   with tf.io.gfile.GFile(input_compiled_file, mode="w") as input_writer:
 253     with tf.io.gfile.GFile(target_compiled_file, mode="w") as target_writer:
 254       for i in range(len(raw_files["inputs"])):
 255         input_file = raw_files["inputs"][i]
 256         target_file = raw_files["targets"][i]
 257 
 258         logging.info("Reading files %s and %s." % (input_file, target_file))
 259         write_file(input_writer, input_file)
 260         write_file(target_writer, target_file)
 261   return input_compiled_file, target_compiled_file

 264 def write_file(writer, filename):
 265   """Write all of lines from file using the writer."""
 266   for line in txt_line_iterator(filename):
 267     writer.write(line)
 268     writer.write("\n")

 274 def encode_and_save_files(
 275     subtokenizer, data_dir, raw_files, tag, total_shards):
 276   """Save data from files as encoded Examples in TFrecord format.
 277 
 278   Args:
 279     subtokenizer: Subtokenizer object that will be used to encode the strings.
 280     data_dir: The directory in which to write the examples
 281     raw_files: A tuple of (input, target) data files. Each line in the input and
 282       the corresponding line in target file will be saved in a tf.Example.
 283     tag: String that will be added onto the file names.
 284     total_shards: Number of files to divide the data into.
 285 
 286   Returns:
 287     List of all files produced.
 288   """
 289   # Create a file for each shard.
 290   filepaths = [shard_filename(data_dir, tag, n + 1, total_shards)
 291                for n in range(total_shards)]
 292 
 293   if all_exist(filepaths):
 294     logging.info("Files with tag %s already exist." % tag)
 295     return filepaths
 296 
 297   logging.info("Saving files with tag %s." % tag)
 298   input_file = raw_files[0]
 299   target_file = raw_files[1]
 300 
 301   # Write examples to each shard in round robin order.
 302   tmp_filepaths = [six.ensure_str(fname) + ".incomplete" for fname in filepaths]
 303   writers = [tf.python_io.TFRecordWriter(fname) for fname in tmp_filepaths]
 304   counter, shard = 0, 0
 305   for counter, (input_line, target_line) in enumerate(zip(
 306       txt_line_iterator(input_file), txt_line_iterator(target_file))):
 307     if counter > 0 and counter % 100000 == 0:
 308       logging.info("\tSaving case %d." % counter)
 309     example = dict_to_example(
 310         {"inputs": subtokenizer.encode(input_line, add_eos=True),
 311          "targets": subtokenizer.encode(target_line, add_eos=True)})
 312     writers[shard].write(example.SerializeToString())
 313     shard = (shard + 1) % total_shards
 314   for writer in writers:
 315     writer.close()
 316 
 317   for tmp_name, final_name in zip(tmp_filepaths, filepaths):
 318     tf.gfile.Rename(tmp_name, final_name)
 319 
 320   logging.info("Saved %d Examples", counter + 1)
 321   return filepaths

 324 def shard_filename(path, tag, shard_num, total_shards):
 325   """Create filename for data shard."""
 326   return os.path.join(
 327       path, "%s-%s-%.5d-of-%.5d" % (_PREFIX, tag, shard_num, total_shards))

 330 def shuffle_records(fname):
 331   """Shuffle records in a single file."""
 332   logging.info("Shuffling records in file %s" % fname)
 333 
 334   # Rename file prior to shuffling
 335   tmp_fname = six.ensure_str(fname) + ".unshuffled"
 336   tf.gfile.Rename(fname, tmp_fname)
 337 
 338   reader = tf.io.tf_record_iterator(tmp_fname)
 339   records = []
 340   for record in reader:
 341     records.append(record)
 342     if len(records) % 100000 == 0:
 343       logging.info("\tRead: %d", len(records))
 344 
 345   random.shuffle(records)
 346 
 347   # Write shuffled records to original file name
 348   with tf.python_io.TFRecordWriter(fname) as w:
 349     for count, record in enumerate(records):
 350       w.write(record)
 351       if count > 0 and count % 100000 == 0:
 352         logging.info("\tWriting record: %d" % count)
 353 
 354   tf.gfile.Remove(tmp_fname)

 357 def dict_to_example(dictionary):
 358   """Converts a dictionary of string->int to a tf.Example."""
 359   features = {}
 360   for k, v in six.iteritems(dictionary):
 361     features[k] = tf.train.Feature(int64_list=tf.train.Int64List(value=v))
 362   return tf.train.Example(features=tf.train.Features(feature=features))

 365 def all_exist(filepaths):
 366   """Returns true if all files in the list exist."""
 367   for fname in filepaths:
 368     if not tf.gfile.Exists(fname):
 369       return False
 370   return True

 379 def main(unused_argv):
 380   """Obtain training and evaluation data for the Transformer model."""
 381   make_dir(FLAGS.raw_dir)
 382   make_dir(FLAGS.data_dir)
 383 
 384   # Download test_data
 385   logging.info("Step 1/5: Downloading test data")
 386   get_raw_files(FLAGS.data_dir, _TEST_DATA_SOURCES)
 387 
 388   # Get paths of download/extracted training and evaluation files.
 389   logging.info("Step 2/5: Downloading data from source")
 390   train_files = get_raw_files(FLAGS.raw_dir, _TRAIN_DATA_SOURCES)
 391   eval_files = get_raw_files(FLAGS.raw_dir, _EVAL_DATA_SOURCES)
 392 
 393   # Create subtokenizer based on the training files.
 394   logging.info("Step 3/5: Creating subtokenizer and building vocabulary")
 395   train_files_flat = train_files["inputs"] + train_files["targets"]
 396   vocab_file = os.path.join(FLAGS.data_dir, VOCAB_FILE)
 397   subtokenizer = tokenizer.Subtokenizer.init_from_files(
 398       vocab_file, train_files_flat, _TARGET_VOCAB_SIZE, _TARGET_THRESHOLD,
 399       min_count=None if FLAGS.search else _TRAIN_DATA_MIN_COUNT)
 400 
 401   logging.info("Step 4/5: Compiling training and evaluation data")
 402   compiled_train_files = compile_files(FLAGS.raw_dir, train_files, _TRAIN_TAG)
 403   compiled_eval_files = compile_files(FLAGS.raw_dir, eval_files, _EVAL_TAG)
 404 
 405   # Tokenize and save data as Examples in the TFRecord format.
 406   logging.info("Step 5/5: Preprocessing and saving data")
 407   train_tfrecord_files = encode_and_save_files(
 408       subtokenizer, FLAGS.data_dir, compiled_train_files, _TRAIN_TAG,
 409       _TRAIN_SHARDS)
 410   encode_and_save_files(
 411       subtokenizer, FLAGS.data_dir, compiled_eval_files, _EVAL_TAG,
 412       _EVAL_SHARDS)
 413 
 414   for fname in train_tfrecord_files:
 415     shuffle_records(fname)
