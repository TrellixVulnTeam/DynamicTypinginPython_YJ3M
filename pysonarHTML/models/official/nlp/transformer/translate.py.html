  32 def _get_sorted_inputs(filename):
  33   """Read and sort lines from the file sorted by decreasing length.
  34 
  35   Args:
  36     filename: String name of file to read inputs from.
  37   Returns:
  38     Sorted list of inputs, and dictionary mapping original index->sorted index
  39     of each element.
  40   """
  41   with tf.io.gfile.GFile(filename) as f:
  42     records = f.read().split("\n")
  43     inputs = [record.strip() for record in records]
  44     if not inputs[-1]:
  45       inputs.pop()
  46 
  47   input_lens = [(i, len(line.split())) for i, line in enumerate(inputs)]
  48   sorted_input_lens = sorted(input_lens, key=lambda x: x[1], reverse=True)
  49 
  50   sorted_inputs = [None] * len(sorted_input_lens)
  51   sorted_keys = [0] * len(sorted_input_lens)
  52   for i, (index, _) in enumerate(sorted_input_lens):
  53     sorted_inputs[i] = inputs[index]
  54     sorted_keys[index] = i
  55   return sorted_inputs, sorted_keys

  58 def _encode_and_add_eos(line, subtokenizer):
  59   """Encode line with subtokenizer, and add EOS id to the end."""
  60   return subtokenizer.encode(line) + [tokenizer.EOS_ID]

  63 def _trim_and_decode(ids, subtokenizer):
  64   """Trim EOS and PAD tokens from ids, and decode to return a string."""
  65   try:
  66     index = list(ids).index(tokenizer.EOS_ID)
  67     return subtokenizer.decode(ids[:index])
  68   except ValueError:  # No EOS found in sequence
  69     return subtokenizer.decode(ids)

  72 def translate_file(model,
  73                    params,
  74                    subtokenizer,
  75                    input_file,
  76                    output_file=None,
  77                    print_all_translations=True,
  78                    distribution_strategy=None):
  79   """Translate lines in file, and save to output file if specified.
  80 
  81   Args:
  82     model: A Keras model, used to generate the translations.
  83     params: A dictionary, containing the translation related parameters.
  84     subtokenizer: A subtokenizer object, used for encoding and decoding source
  85       and translated lines.
  86     input_file: A file containing lines to translate.
  87     output_file: A file that stores the generated translations.
  88     print_all_translations: A bool. If true, all translations are printed to
  89       stdout.
  90     distribution_strategy: A distribution strategy, used to perform inference
  91       directly with tf.function instead of Keras model.predict().
  92 
  93   Raises:
  94     ValueError: if output file is invalid.
  95   """
  96   batch_size = params["decode_batch_size"]
  97 
  98   # Read and sort inputs by length. Keep dictionary (original index-->new index
  99   # in sorted list) to write translations in the original order.
 100   sorted_inputs, sorted_keys = _get_sorted_inputs(input_file)
 101   total_samples = len(sorted_inputs)
 102   num_decode_batches = (total_samples - 1) // batch_size + 1
 103 
 104   def input_generator():
 105     """Yield encoded strings from sorted_inputs."""
 106     for i in range(num_decode_batches):
 107       lines = [
 108           sorted_inputs[j + i * batch_size]
 109           for j in range(batch_size)
 110           if j + i * batch_size < total_samples
 111       ]
 112       lines = [_encode_and_add_eos(l, subtokenizer) for l in lines]
 113       if distribution_strategy:
 114         for j in range(batch_size - len(lines)):
 115           lines.append([tokenizer.EOS_ID])
 116       batch = tf.keras.preprocessing.sequence.pad_sequences(
 117           lines,
 118           maxlen=params["decode_max_length"],
 119           dtype="int32",
 120           padding="post")
 121       logging.info("Decoding batch %d out of %d.", i, num_decode_batches)
 122       yield batch
 123 
 124   @tf.function
 125   def predict_step(inputs):
 126     """Decoding step function for TPU runs."""
 127 
 128     def _step_fn(inputs):
 129       """Per replica step function."""
 130       tag = inputs[0]
 131       val_inputs = inputs[1]
 132       val_outputs, _ = model([val_inputs], training=False)
 133       return tag, val_outputs
 134 
 135     return distribution_strategy.run(_step_fn, args=(inputs,))
 136 
 137   translations = []
 138   if distribution_strategy:
 139     num_replicas = distribution_strategy.num_replicas_in_sync
 140     local_batch_size = params["decode_batch_size"] // num_replicas
 141   for i, text in enumerate(input_generator()):
 142     if distribution_strategy:
 143       text = np.reshape(text, [num_replicas, local_batch_size, -1])
 144       # Add tag to the input of each replica with the reordering logic after
 145       # outputs, to ensure the output order matches the input order.
 146       text = tf.constant(text)
 147 
 148       @tf.function
 149       def text_as_per_replica():
 150         replica_context = tf.distribute.get_replica_context()
 151         replica_id = replica_context.replica_id_in_sync_group
 152         return replica_id, text[replica_id]
 153 
 154       text = distribution_strategy.run(text_as_per_replica)
 155       outputs = distribution_strategy.experimental_local_results(
 156           predict_step(text))
 157       tags, unordered_val_outputs = outputs[0]
 158       tags = [tag.numpy() for tag in tags._values]
 159       unordered_val_outputs = [
 160           val_output.numpy() for val_output in unordered_val_outputs._values]
 161       # pylint: enable=protected-access
 162       val_outputs = [None] * len(tags)
 163       for k in range(len(tags)):
 164         val_outputs[tags[k]] = unordered_val_outputs[k]
 165       val_outputs = np.reshape(val_outputs, [params["decode_batch_size"], -1])
 166     else:
 167       val_outputs, _ = model.predict(text)
 168 
 169     length = len(val_outputs)
 170     for j in range(length):
 171       if j + i * batch_size < total_samples:
 172         translation = _trim_and_decode(val_outputs[j], subtokenizer)
 173         translations.append(translation)
 174         if print_all_translations:
 175           logging.info("Translating:\n\tInput: %s\n\tOutput: %s",
 176                        sorted_inputs[j + i * batch_size], translation)
 177 
 178   # Write translations in the order they appeared in the original file.
 179   if output_file is not None:
 180     if tf.io.gfile.isdir(output_file):
 181       raise ValueError("File output is a directory, will not save outputs to "
 182                        "file.")
 183     logging.info("Writing to file %s", output_file)
 184     with tf.compat.v1.gfile.Open(output_file, "w") as f:
 185       for i in sorted_keys:
 186         f.write("%s\n" % translations[i])

 189 def translate_from_text(model, subtokenizer, txt):
 190   encoded_txt = _encode_and_add_eos(txt, subtokenizer)
 191   result = model.predict(encoded_txt)
 192   outputs = result["outputs"]
 193   logging.info("Original: \"%s\"", txt)
 194   translate_from_input(outputs, subtokenizer)
 195

 197 def translate_from_input(outputs, subtokenizer):
 198   translation = _trim_and_decode(outputs, subtokenizer)
 199   logging.info("Translation: \"%s\"", translation)
