  15 """Test functions in compute_blue.py."""

  17 import tempfile
  18 
  19 import tensorflow as tf
  20 
  21 from official.nlp.transformer import compute_bleu
  22 
  23 
  24 class ComputeBleuTest(tf.test.TestCase):
  25 
  26   def _create_temp_file(self, text):
  27     temp_file = tempfile.NamedTemporaryFile(delete=False)
  28     with tf.io.gfile.GFile(temp_file.name, "w") as w:
  29       w.write(text)
  30     return temp_file.name
  31 
  32   def test_bleu_same(self):
  33     ref = self._create_temp_file("test 1 two 3\nmore tests!")
  34     hyp = self._create_temp_file("test 1 two 3\nmore tests!")
  35 
  36     uncased_score = compute_bleu.bleu_wrapper(ref, hyp, False)
  37     cased_score = compute_bleu.bleu_wrapper(ref, hyp, True)
  38     self.assertEqual(100, uncased_score)
  39     self.assertEqual(100, cased_score)
  40 
  41   def test_bleu_same_different_case(self):
  42     ref = self._create_temp_file("Test 1 two 3\nmore tests!")
  43     hyp = self._create_temp_file("test 1 two 3\nMore tests!")
  44     uncased_score = compute_bleu.bleu_wrapper(ref, hyp, False)
  45     cased_score = compute_bleu.bleu_wrapper(ref, hyp, True)
  46     self.assertEqual(100, uncased_score)
  47     self.assertLess(cased_score, 100)
  48 
  49   def test_bleu_different(self):
  50     ref = self._create_temp_file("Testing\nmore tests!")
  51     hyp = self._create_temp_file("Dog\nCat")
  52     uncased_score = compute_bleu.bleu_wrapper(ref, hyp, False)
  53     cased_score = compute_bleu.bleu_wrapper(ref, hyp, True)
  54     self.assertLess(uncased_score, 100)
  55     self.assertLess(cased_score, 100)
  56 
  57   def test_bleu_tokenize(self):
  58     s = "Test0, 1 two, 3"
  59     tokenized = compute_bleu.bleu_tokenize(s)
  60     self.assertEqual(["Test0", ",", "1", "two", ",", "3"], tokenized)

  64   tf.test.main()
