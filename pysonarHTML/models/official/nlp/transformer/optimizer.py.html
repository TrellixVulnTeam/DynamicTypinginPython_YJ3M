  26 class LearningRateSchedule(tf.keras.optimizers.schedules.LearningRateSchedule):
  27   """Learning rate schedule."""
  28 
  29   def __init__(self, initial_learning_rate, hidden_size, warmup_steps):
  30     """Initialize configuration of the learning rate schedule.
  31 
  32     Args:
  33       initial_learning_rate: A float, the initial learning rate.
  34       hidden_size: An integer, the model dimension in the hidden layers.
  35       warmup_steps: An integer, the number of steps required for linear warmup.
  36     """
  37     super(LearningRateSchedule, self).__init__()
  38     self.initial_learning_rate = initial_learning_rate
  39     self.hidden_size = hidden_size
  40     self.warmup_steps = tf.cast(warmup_steps, tf.float32)
  41 
  42   def __call__(self, global_step):
  43     """Calculate learning rate with linear warmup and rsqrt decay.
  44 
  45     Args:
  46       global_step: An integer, the current global step used for learning rate
  47         calculation.
  48 
  49     Returns:
  50       A float, the learning rate needs to be used for current global step.
  51     """
  52     with tf.name_scope('learning_rate_schedule'):
  53       global_step = tf.cast(global_step, tf.float32)
  54       learning_rate = self.initial_learning_rate
  55       learning_rate *= (self.hidden_size**-0.5)
  56       # Apply linear warmup
  57       learning_rate *= tf.minimum(1.0, global_step / self.warmup_steps)
  58       # Apply rsqrt decay
  59       learning_rate /= tf.sqrt(tf.maximum(global_step, self.warmup_steps))
  60       return learning_rate
  61 
  62   def get_config(self):
  63     """Get the configuration of the learning rate schedule."""
  64     return {
  65         'initial_learning_rate': self.initial_learning_rate,
  66         'hidden_size': self.hidden_size,
  67         'warmup_steps': self.warmup_steps,
  68     }

  71 class LearningRateFn(object):
  72   """Creates learning rate function."""
  73 
  74   def __init__(self, learning_rate, hidden_size, warmup_steps):
  75     self.learning_rate = learning_rate
  76     self.hidden_size = hidden_size
  77     self.warmup_steps = float(warmup_steps)
  78 
  79   def __call__(self, global_step):
  80     """Calculate learning rate with linear warmup and rsqrt decay."""
  81     step = float(global_step)
  82     learning_rate = self.learning_rate
  83     learning_rate *= (self.hidden_size ** -0.5)
  84     # Apply linear warmup
  85     learning_rate *= np.minimum(1.0, step / self.warmup_steps)
  86     # Apply rsqrt decay
  87     learning_rate /= np.sqrt(np.maximum(step, self.warmup_steps))
  88     return learning_rate

  91 class LearningRateScheduler(tf.keras.callbacks.Callback):
  92   """Keras callback to schedule learning rate.
  93 
  94   TODO(tianlin): Refactor this scheduler and LearningRateBatchScheduler in
  95   official/resnet/keras/keras_common.py.
  96   """
  97 
  98   def __init__(self, schedule, init_steps=None, verbose=False):
  99     super(LearningRateScheduler, self).__init__()
 100     self.schedule = schedule
 101     self.verbose = verbose
 102     if init_steps is None:
 103       init_steps = 0.0
 104     self.steps = float(init_steps)   # Total steps during training.
 105 
 106   def on_epoch_begin(self, epoch, logs=None):
 107     if not hasattr(self.model.optimizer, 'lr'):
 108       raise ValueError('Optimizer must have a "lr" attribute.')
 109     if not hasattr(self.model.optimizer, 'iterations'):
 110       raise ValueError('Optimizer must have a "iterations" attribute.')
 111 
 112   def on_train_batch_begin(self, batch, logs=None):
 113     """Adjusts learning rate for each train batch."""
 114     if self.verbose > 0:
 115       iterations = K.get_value(self.model.optimizer.iterations)
 116       print('Original iteration %d' % iterations)
 117 
 118     self.steps += 1.0
 119     try:  # new API
 120       lr = float(K.get_value(self.model.optimizer.lr))
 121       lr = self.schedule(self.steps, lr)
 122     except TypeError:  # Support for old API for backward compatibility
 123       lr = self.schedule(self.steps)
 124     if not isinstance(lr, (float, np.float32, np.float64)):
 125       raise ValueError('The output of the "schedule" function '
 126                        'should be float.')
 127     K.set_value(self.model.optimizer.lr, lr)
 128     K.set_value(self.model.optimizer.iterations, self.steps)
 129 
 130     if self.verbose > 0:
 131       print('Batch %05d Step %05d: LearningRateScheduler setting learning '
 132             'rate to %s.' % (batch + 1, self.steps, lr))
 133 
 134   def on_epoch_end(self, epoch, logs=None):
 135     logs = logs or {}
 136     logs['lr'] = K.get_value(self.model.optimizer.lr)
 137     logs['steps'] = self.steps
