  15 """Test beam search helper methods."""
  16 
  17 import tensorflow.compat.v1 as tf
  18 
  19 from official.nlp.transformer import beam_search_v1 as beam_search
  20 
  21 
  22 class BeamSearchHelperTests(tf.test.TestCase):
  23 
  24   def setUp(self):
  25     super(BeamSearchHelperTests, self).setUp()
  26     tf.compat.v1.disable_eager_execution()
  27 
  28   def test_expand_to_beam_size(self):
  29     x = tf.ones([7, 4, 2, 5])
  30     x = beam_search._expand_to_beam_size(x, 3)
  31     with self.session() as sess:
  32       shape = sess.run(tf.shape(x))
  33     self.assertAllEqual([7, 3, 4, 2, 5], shape)
  34 
  35   def test_shape_list(self):
  36     y = tf.compat.v1.placeholder(dtype=tf.int32, shape=[])
  37     x = tf.ones([7, y, 2, 5])
  38     shape = beam_search._shape_list(x)
  39     self.assertIsInstance(shape[0], int)
  40     self.assertIsInstance(shape[1], tf.Tensor)
  41     self.assertIsInstance(shape[2], int)
  42     self.assertIsInstance(shape[3], int)
  43 
  44   def test_get_shape_keep_last_dim(self):
  45     y = tf.constant(4.0)
  46     x = tf.ones([7, tf.cast(tf.sqrt(y), tf.int32), 2, 5])
  47     shape = beam_search._get_shape_keep_last_dim(x)
  48     self.assertAllEqual([None, None, None, 5],
  49                         shape.as_list())
  50 
  51   def test_flatten_beam_dim(self):
  52     x = tf.ones([7, 4, 2, 5])
  53     x = beam_search._flatten_beam_dim(x)
  54     with self.session() as sess:
  55       shape = sess.run(tf.shape(x))
  56     self.assertAllEqual([28, 2, 5], shape)
  57 
  58   def test_unflatten_beam_dim(self):
  59     x = tf.ones([28, 2, 5])
  60     x = beam_search._unflatten_beam_dim(x, 7, 4)
  61     with self.session() as sess:
  62       shape = sess.run(tf.shape(x))
  63     self.assertAllEqual([7, 4, 2, 5], shape)
  64 
  65   def test_gather_beams(self):
  66     x = tf.reshape(tf.range(24), [2, 3, 4])
  67     # x looks like:  [[[ 0  1  2  3]
  68     #                  [ 4  5  6  7]
  69     #                  [ 8  9 10 11]]
  70     #
  71     #                 [[12 13 14 15]
  72     #                  [16 17 18 19]
  73     #                  [20 21 22 23]]]
  74 
  75     y = beam_search._gather_beams(x, [[1, 2], [0, 2]], 2, 2)
  76     with self.session() as sess:
  77       y = sess.run(y)
  78 
  79     self.assertAllEqual([[[4, 5, 6, 7],
  80                           [8, 9, 10, 11]],
  81                          [[12, 13, 14, 15],
  82                           [20, 21, 22, 23]]],
  83                         y)
  84 
  85   def test_gather_topk_beams(self):
  86     x = tf.reshape(tf.range(24), [2, 3, 4])
  87     x_scores = [[0, 1, 1], [1, 0, 1]]
  88 
  89     y = beam_search._gather_topk_beams(x, x_scores, 2, 2)
  90     with self.session() as sess:
  91       y = sess.run(y)
  92 
  93     self.assertAllEqual([[[4, 5, 6, 7],
  94                           [8, 9, 10, 11]],
  95                          [[12, 13, 14, 15],
  96                           [20, 21, 22, 23]]],
  97                         y)
