  25 class SubtokenizerTest(tf.test.TestCase):
  26 
  27   def _init_subtokenizer(self, vocab_list):
  28     temp_file = tempfile.NamedTemporaryFile(delete=False)
  29     with tf.io.gfile.GFile(temp_file.name, "w") as w:
  30       for subtoken in vocab_list:
  31         w.write("'%s'" % subtoken)
  32         w.write("\n")
  33     return tokenizer.Subtokenizer(temp_file.name, reserved_tokens=[])
  34 
  35   def test_encode(self):
  36     vocab_list = ["123_", "test", "ing_"]
  37     subtokenizer = self._init_subtokenizer(vocab_list)
  38     s = "testing 123"
  39     encoded_list = subtokenizer.encode(s)
  40     self.assertEqual([1, 2, 0], encoded_list)
  41 
  42   def test_decode(self):
  43     vocab_list = ["123_", "test", "ing_"]
  44     subtokenizer = self._init_subtokenizer(vocab_list)
  45     encoded_list = [1, 2, 0]  # testing 123
  46     decoded_str = subtokenizer.decode(encoded_list)
  47     self.assertEqual("testing 123", decoded_str)
  48 
  49   def test_subtoken_ids_to_tokens(self):
  50     vocab_list = ["123_", "test", "ing_"]
  51     subtokenizer = self._init_subtokenizer(vocab_list)
  52     encoded_list = [1, 2, 0]  # testing 123
  53     token_list = subtokenizer._subtoken_ids_to_tokens(encoded_list)
  54     self.assertEqual([u"testing", u"123"], token_list)

  73   def test_escape_token(self):
  74     token = u"abc_\\4"
  75     alphabet = set("abc_\\u;")
  76 
  77     escaped_token = tokenizer._escape_token(token, alphabet)
  78     self.assertEqual("abc\\u\\\\\\52;_", escaped_token)

  84     self.assertEqual("Underline: _, Backslash: \\, Unicode: 4", unescaped_token)

 117   def test_count_and_gen_subtokens(self):
 118     token_counts = {"abc": 5}
 119     alphabet = set("abc_")
 120     subtoken_dict = {"a": 0, "b": 1, "c": 2, "_": 3}
 121     max_subtoken_length = 2
 122 
 123     subtoken_counts = tokenizer._count_and_gen_subtokens(
 124         token_counts, alphabet, subtoken_dict, max_subtoken_length)

 126     self.assertIsInstance(subtoken_counts, collections.defaultdict)

 128         {
 129             "a": 5,
 130             "b": 5,
 131             "c": 5,
 132             "_": 5,
 133             "ab": 5,
 134             "bc": 5,
 135             "c_": 5,
 136             "abc": 5,
 137             "bc_": 5,
 138             "abc_": 5

 141   def test_filter_and_bucket_subtokens(self):
 142     subtoken_counts = collections.defaultdict(int, {
 143         "a": 2,
 144         "b": 4,
 145         "c": 1,
 146         "ab": 6,
 147         "ac": 3,
 148         "abbc": 5
 149     })
 150     min_count = 3
 151 
 152     subtoken_buckets = tokenizer._filter_and_bucket_subtokens(
 153         subtoken_counts, min_count)
 154 
 155     self.assertEqual(len(subtoken_buckets[0]), 0)
 156     self.assertEqual(set("b"), subtoken_buckets[1])
 157     self.assertEqual(set(["ab", "ac"]), subtoken_buckets[2])
 158     self.assertEqual(len(subtoken_buckets[3]), 0)
 159     self.assertEqual(set(["abbc"]), subtoken_buckets[4])

 161   def test_gen_new_subtoken_list(self):
 162     subtoken_counts = collections.defaultdict(int, {
 163         "translate": 10,
 164         "t": 40,
 165         "tr": 16,
 166         "tra": 12
 167     })
 168     min_count = 5
 169     alphabet = set("translate")
 170     reserved_tokens = ["reserved", "tokens"]
 171 
 172     subtoken_list, max_token_length = tokenizer._gen_new_subtoken_list(
 173         subtoken_counts, min_count, alphabet, reserved_tokens)
 174 
 175     # Check that "tra" isn"t in the list (its count should be decremented to 2,
 176     # so it should not be added to the canddiate list).
 177     self.assertNotIn("tra", subtoken_list)
 178 
 179     self.assertIn("tr", subtoken_list)
 180     self.assertIn("t", subtoken_list)
 181 
 182     self.assertEqual(len("translate"), max_token_length)

 184   def test_generate_subtokens(self):
 185     token_counts = {"ab": 1, "bc": 3, "abc": 5}
 186     alphabet = set("abc_")
 187     min_count = 100
 188     num_iterations = 1
 189     reserved_tokens = ["reserved", "tokens"]
 190 
 191     vocab_list = tokenizer._generate_subtokens(token_counts, alphabet,
 192                                                min_count, num_iterations,
 193                                                reserved_tokens)
 194 
 195     # Check that reserved tokens are at the front of the list
 196     self.assertEqual(vocab_list[:2], reserved_tokens)
 197 
 198     # Check that each character in alphabet is in the vocab list
 199     for c in alphabet:
 200       self.assertIn(c, vocab_list)
