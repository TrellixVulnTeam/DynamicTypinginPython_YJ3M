  39 def _pad_tensors_to_same_length(x, y):
  40   """Pad x and y so that the results have the same length (second dimension)."""
  41   with tf.name_scope("pad_to_same_length"):
  42     x_length = tf.shape(x)[1]
  43     y_length = tf.shape(y)[1]
  44 
  45     max_length = tf.maximum(x_length, y_length)
  46 
  47     x = tf.pad(x, [[0, 0], [0, max_length - x_length], [0, 0]])
  48     y = tf.pad(y, [[0, 0], [0, max_length - y_length]])
  49     return x, y

  52 def padded_cross_entropy_loss(logits, labels, smoothing, vocab_size):
  53   """Calculate cross entropy loss while ignoring padding.
  54 
  55   Args:
  56     logits: Tensor of size [batch_size, length_logits, vocab_size]
  57     labels: Tensor of size [batch_size, length_labels]
  58     smoothing: Label smoothing constant, used to determine the on and off values
  59     vocab_size: int size of the vocabulary
  60   Returns:
  61     Returns the cross entropy loss and weight tensors: float32 tensors with
  62       shape [batch_size, max(length_logits, length_labels)]
  63   """
  64   with tf.name_scope("loss", values=[logits, labels]):
  65     logits, labels = _pad_tensors_to_same_length(logits, labels)
  66 
  67     # Calculate smoothing cross entropy
  68     with tf.name_scope("smoothing_cross_entropy", values=[logits, labels]):
  69       confidence = 1.0 - smoothing
  70       low_confidence = (1.0 - confidence) / tf.to_float(vocab_size - 1)
  71       soft_targets = tf.one_hot(
  72           tf.cast(labels, tf.int32),
  73           depth=vocab_size,
  74           on_value=confidence,
  75           off_value=low_confidence)
  76       xentropy = tf.nn.softmax_cross_entropy_with_logits_v2(
  77           logits=logits, labels=soft_targets)
  78 
  79       # Calculate the best (lowest) possible value of cross entropy, and
  80       # subtract from the cross entropy loss.
  81       normalizing_constant = -(
  82           confidence * tf.log(confidence) + tf.to_float(vocab_size - 1) *
  83           low_confidence * tf.log(low_confidence + 1e-20))
  84       xentropy -= normalizing_constant
  85 
  86     weights = tf.to_float(tf.not_equal(labels, 0))
  87     return xentropy * weights, weights

  90 def _convert_to_eval_metric(metric_fn):
  91   """Wrap a metric fn that returns scores and weights as an eval metric fn.
  92 
  93   The input metric_fn returns values for the current batch. The wrapper
  94   aggregates the return values collected over all of the batches evaluated.
  95 
  96   Args:
  97     metric_fn: function that returns scores and weights for the current batch's
  98       logits and predicted labels.
  99 
 100   Returns:
 101     function that aggregates the scores and weights from metric_fn.
 102   """
 103   def problem_metric_fn(*args):
 104     """Returns an aggregation of the metric_fn's returned values."""
 105     (scores, weights) = metric_fn(*args)
 106 
 107     # The tf.metrics.mean function assures correct aggregation.
 108     return tf.metrics.mean(scores, weights)
 109   return problem_metric_fn

 112 def get_eval_metrics(logits, labels, params):
 113   """Return dictionary of model evaluation metrics."""
 114   metrics = {
 115       "accuracy": _convert_to_eval_metric(padded_accuracy)(logits, labels),
 116       "accuracy_top5": _convert_to_eval_metric(padded_accuracy_top5)(
 117           logits, labels),
 118       "accuracy_per_sequence": _convert_to_eval_metric(
 119           padded_sequence_accuracy)(logits, labels),
 120       "neg_log_perplexity": _convert_to_eval_metric(padded_neg_log_perplexity)(
 121           logits, labels, params["vocab_size"]),
 122   }
 123 
 124   if not params["use_tpu"]:
 125     # TPU does not support tf.py_func
 126     metrics.update({
 127         "approx_bleu_score": _convert_to_eval_metric(
 128             bleu_score)(logits, labels),
 129         "rouge_2_fscore": _convert_to_eval_metric(
 130             rouge_2_fscore)(logits, labels),
 131         "rouge_L_fscore": _convert_to_eval_metric(
 132             rouge_l_fscore)(logits, labels),
 133     })
 134 
 135   # Prefix each of the metric names with "metrics/". This allows the metric
 136   # graphs to display under the "metrics" category in TensorBoard.
 137   metrics = {"metrics/%s" % k: v for k, v in six.iteritems(metrics)}
 138   return metrics

 141 def padded_accuracy(logits, labels):
 142   """Percentage of times that predictions matches labels on non-0s."""
 143   with tf.variable_scope("padded_accuracy", values=[logits, labels]):
 144     logits, labels = _pad_tensors_to_same_length(logits, labels)
 145     weights = tf.to_float(tf.not_equal(labels, 0))
 146     outputs = tf.to_int32(tf.argmax(logits, axis=-1))
 147     padded_labels = tf.to_int32(labels)
 148     return tf.to_float(tf.equal(outputs, padded_labels)), weights
 149 
 150 
 151 def padded_accuracy_topk(logits, labels, k):
 152   """Percentage of times that top-k predictions matches labels on non-0s."""
 153   with tf.variable_scope("padded_accuracy_topk", values=[logits, labels]):
 154     logits, labels = _pad_tensors_to_same_length(logits, labels)
 155     weights = tf.to_float(tf.not_equal(labels, 0))
 156     effective_k = tf.minimum(k, tf.shape(logits)[-1])
 157     _, outputs = tf.nn.top_k(logits, k=effective_k)
 158     outputs = tf.to_int32(outputs)
 159     padded_labels = tf.to_int32(labels)
 160     padded_labels = tf.expand_dims(padded_labels, axis=-1)
 161     padded_labels += tf.zeros_like(outputs)  # Pad to same shape.
 162     same = tf.to_float(tf.equal(outputs, padded_labels))
 163     same_topk = tf.reduce_sum(same, axis=-1)
 164     return same_topk, weights
 165 
 166 
 167 def padded_accuracy_top5(logits, labels):
 168   return padded_accuracy_topk(logits, labels, 5)

 171 def padded_sequence_accuracy(logits, labels):
 172   """Percentage of times that predictions matches labels everywhere (non-0)."""
 173   with tf.variable_scope("padded_sequence_accuracy", values=[logits, labels]):
 174     logits, labels = _pad_tensors_to_same_length(logits, labels)
 175     weights = tf.to_float(tf.not_equal(labels, 0))
 176     outputs = tf.to_int32(tf.argmax(logits, axis=-1))
 177     padded_labels = tf.to_int32(labels)
 178     not_correct = tf.to_float(tf.not_equal(outputs, padded_labels)) * weights
 179     axis = list(range(1, len(outputs.get_shape())))
 180     correct_seq = 1.0 - tf.minimum(1.0, tf.reduce_sum(not_correct, axis=axis))
 181     return correct_seq, tf.constant(1.0)

 184 def padded_neg_log_perplexity(logits, labels, vocab_size):
 185   """Average log-perplexity excluding padding 0s. No smoothing."""
 186   num, den = padded_cross_entropy_loss(logits, labels, 0, vocab_size)
 187   return -num, den

 190 def bleu_score(logits, labels):
 191   """Approximate BLEU score computation between labels and predictions.
 192 
 193   An approximate BLEU scoring method since we do not glue word pieces or
 194   decode the ids and tokenize the output. By default, we use ngram order of 4
 195   and use brevity penalty. Also, this does not have beam search.
 196 
 197   Args:
 198     logits: Tensor of size [batch_size, length_logits, vocab_size]
 199     labels: Tensor of size [batch-size, length_labels]
 200 
 201   Returns:
 202     bleu: int, approx bleu score
 203   """
 204   predictions = tf.to_int32(tf.argmax(logits, axis=-1))
 205   # TODO: Look into removing use of py_func
 206   bleu = tf.py_func(compute_bleu, (labels, predictions), tf.float32)
 207   return bleu, tf.constant(1.0)

 210 def _get_ngrams_with_counter(segment, max_order):
 211   """Extracts all n-grams up to a given maximum order from an input segment.
 212 
 213   Args:
 214     segment: text segment from which n-grams will be extracted.
 215     max_order: maximum length in tokens of the n-grams returned by this
 216         methods.
 217 
 218   Returns:
 219     The Counter containing all n-grams upto max_order in segment
 220     with a count of how many times each n-gram occurred.
 221   """
 222   ngram_counts = collections.Counter()
 223   for order in xrange(1, max_order + 1):
 224     for i in xrange(0, len(segment) - order + 1):
 225       ngram = tuple(segment[i:i + order])
 226       ngram_counts[ngram] += 1
 227   return ngram_counts

 230 def compute_bleu(reference_corpus, translation_corpus, max_order=4,
 231                  use_bp=True):
 232   """Computes BLEU score of translated segments against one or more references.
 233 
 234   Args:
 235     reference_corpus: list of references for each translation. Each
 236         reference should be tokenized into a list of tokens.
 237     translation_corpus: list of translations to score. Each translation
 238         should be tokenized into a list of tokens.
 239     max_order: Maximum n-gram order to use when computing BLEU score.
 240     use_bp: boolean, whether to apply brevity penalty.
 241 
 242   Returns:
 243     BLEU score.
 244   """
 245   reference_length = 0
 246   translation_length = 0
 247   bp = 1.0
 248   geo_mean = 0
 249 
 250   matches_by_order = [0] * max_order
 251   possible_matches_by_order = [0] * max_order
 252   precisions = []
 253 
 254   for (references, translations) in zip(reference_corpus, translation_corpus):
 255     reference_length += len(references)
 256     translation_length += len(translations)
 257     ref_ngram_counts = _get_ngrams_with_counter(references, max_order)
 258     translation_ngram_counts = _get_ngrams_with_counter(translations, max_order)
 259 
 260     overlap = dict((ngram,
 261                     min(count, translation_ngram_counts[ngram]))
 262                    for ngram, count in ref_ngram_counts.items())
 263 
 264     for ngram in overlap:
 265       matches_by_order[len(ngram) - 1] += overlap[ngram]
 266     for ngram in translation_ngram_counts:
 267       possible_matches_by_order[len(ngram) - 1] += translation_ngram_counts[
 268           ngram]
 269 
 270   precisions = [0] * max_order
 271   smooth = 1.0
 272 
 273   for i in xrange(0, max_order):
 274     if possible_matches_by_order[i] > 0:
 275       precisions[i] = float(matches_by_order[i]) / possible_matches_by_order[i]
 276       if matches_by_order[i] > 0:
 277         precisions[i] = float(matches_by_order[i]) / possible_matches_by_order[
 278             i]
 279       else:
 280         smooth *= 2
 281         precisions[i] = 1.0 / (smooth * possible_matches_by_order[i])
 282     else:
 283       precisions[i] = 0.0
 284 
 285   if max(precisions) > 0:
 286     p_log_sum = sum(math.log(p) for p in precisions if p)
 287     geo_mean = math.exp(p_log_sum / max_order)
 288 
 289   if use_bp:
 290     ratio = translation_length / reference_length
 291     bp = math.exp(1 - 1. / ratio) if ratio < 1.0 else 1.0
 292   bleu = geo_mean * bp
 293   return np.float32(bleu)

 306   Returns:

 315 def _get_ngrams(n, text):
 316   """Calculates n-grams.
 317 
 318   Args:
 319     n: which n-grams to calculate
 320     text: An array of tokens
 321 
 322   Returns:
 323     A set of n-grams
 324   """
 325   ngram_set = set()
 326   text_length = len(text)
 327   max_index_ngram_start = text_length - n
 328   for i in range(max_index_ngram_start + 1):
 329     ngram_set.add(tuple(text[i:i + n]))
 330   return ngram_set

 333 def rouge_n(eval_sentences, ref_sentences, n=2):
 334   """Computes ROUGE-N f1 score of two text collections of sentences.
 335 
 336   Source: https://www.microsoft.com/en-us/research/publication/
 337   rouge-a-package-for-automatic-evaluation-of-summaries/
 338 
 339   Args:
 340     eval_sentences: Predicted sentences.
 341     ref_sentences: Sentences from the reference set
 342     n: Size of ngram.  Defaults to 2.
 343 
 344   Returns:
 345     f1 score for ROUGE-N
 346   """
 347   f1_scores = []
 348   for eval_sentence, ref_sentence in zip(eval_sentences, ref_sentences):
 349     eval_ngrams = _get_ngrams(n, eval_sentence)
 350     ref_ngrams = _get_ngrams(n, ref_sentence)
 351     ref_count = len(ref_ngrams)
 352     eval_count = len(eval_ngrams)
 353 
 354     # Count the overlapping ngrams between evaluated and reference
 355     overlapping_ngrams = eval_ngrams.intersection(ref_ngrams)
 356     overlapping_count = len(overlapping_ngrams)
 357 
 358     # Handle edge case. This isn't mathematically correct, but it's good enough
 359     if eval_count == 0:
 360       precision = 0.0
 361     else:
 362       precision = float(overlapping_count) / eval_count
 363     if ref_count == 0:
 364       recall = 0.0
 365     else:
 366       recall = float(overlapping_count) / ref_count
 367     f1_scores.append(2.0 * ((precision * recall) / (precision + recall + 1e-8)))
 368 
 369   # return overlapping_count / reference_count
 370   return np.mean(f1_scores, dtype=np.float32)

 392 def rouge_l_sentence_level(eval_sentences, ref_sentences):
 393   """Computes ROUGE-L (sentence level) of two collections of sentences.
 394 
 395   Source: https://www.microsoft.com/en-us/research/publication/
 396   rouge-a-package-for-automatic-evaluation-of-summaries/
 397 
 398   Calculated according to:
 399   R_lcs = LCS(X,Y)/m
 400   P_lcs = LCS(X,Y)/n
 401   F_lcs = ((1 + beta^2)*R_lcs*P_lcs) / (R_lcs + (beta^2) * P_lcs)
 402 
 403   where:
 404   X = reference summary
 405   Y = Candidate summary
 406   m = length of reference summary
 407   n = length of candidate summary
 408 
 409   Args:
 410     eval_sentences: The sentences that have been picked by the summarizer
 411     ref_sentences: The sentences from the reference set
 412 
 413   Returns:
 414     A float: F_lcs
 415   """
 416 
 417   f1_scores = []
 418   for eval_sentence, ref_sentence in zip(eval_sentences, ref_sentences):
 419     m = float(len(ref_sentence))
 420     n = float(len(eval_sentence))
 421     lcs = _len_lcs(eval_sentence, ref_sentence)
 422     f1_scores.append(_f_lcs(lcs, m, n))
 423   return np.mean(f1_scores, dtype=np.float32)

 426 def _len_lcs(x, y):
 427   """Returns the length of the Longest Common Subsequence between two seqs.
 428 
 429   Source: http://www.algorithmist.com/index.php/Longest_Common_Subsequence
 430 
 431   Args:
 432     x: sequence of words
 433     y: sequence of words
 434 
 435   Returns
 436     integer: Length of LCS between x and y
 437   """
 438   table = _lcs(x, y)
 439   n, m = len(x), len(y)
 440   return table[n, m]

 443 def _lcs(x, y):
 444   """Computes the length of the LCS between two seqs.
 445 
 446   The implementation below uses a DP programming algorithm and runs
 447   in O(nm) time where n = len(x) and m = len(y).
 448   Source: http://www.algorithmist.com/index.php/Longest_Common_Subsequence
 449 
 450   Args:
 451     x: collection of words
 452     y: collection of words
 453 
 454   Returns:
 455     Table of dictionary of coord and len lcs
 456   """
 457   n, m = len(x), len(y)
 458   table = dict()
 459   for i in range(n + 1):
 460     for j in range(m + 1):
 461       if i == 0 or j == 0:
 462         table[i, j] = 0
 463       elif x[i - 1] == y[j - 1]:
 464         table[i, j] = table[i - 1, j - 1] + 1
 465       else:
 466         table[i, j] = max(table[i - 1, j], table[i, j - 1])
 467   return table

 470 def _f_lcs(llcs, m, n):
 471   """Computes the LCS-based F-measure score.
 472 
 473   Source: http://research.microsoft.com/en-us/um/people/cyl/download/papers/
 474   rouge-working-note-v1.3.1.pdf
 475 
 476   Args:
 477     llcs: Length of LCS
 478     m: number of words in reference summary
 479     n: number of words in candidate summary
 480 
 481   Returns:
 482     Float. LCS-based F-measure score
 483   """
 484   r_lcs = llcs / m
 485   p_lcs = llcs / n
 486   beta = p_lcs / (r_lcs + 1e-12)
 487   num = (1 + (beta ** 2)) * r_lcs * p_lcs
 488   denom = r_lcs + ((beta ** 2) * p_lcs)
 489   f_lcs = num / (denom + 1e-12)
 490   return f_lcs
