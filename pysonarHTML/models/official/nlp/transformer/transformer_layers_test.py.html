  15 """Tests for layers in Transformer."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import tensorflow as tf
  22 
  23 from official.nlp.transformer import attention_layer
  24 from official.nlp.transformer import embedding_layer
  25 from official.nlp.transformer import ffn_layer
  26 from official.nlp.transformer import metrics
  27 
  28 
  29 class TransformerLayersTest(tf.test.TestCase):
  30 
  31   def test_attention_layer(self):
  32     hidden_size = 64
  33     num_heads = 4
  34     dropout = 0.5
  35     dim_per_head = hidden_size // num_heads
  36     layer = attention_layer.SelfAttention(hidden_size, num_heads, dropout)
  37     self.assertDictEqual(layer.get_config(), {
  38         "hidden_size": hidden_size,
  39         "num_heads": num_heads,
  40         "attention_dropout": dropout,
  41     })
  42     length = 2
  43     x = tf.ones([1, length, hidden_size])
  44     bias = tf.ones([1])
  45     cache = {
  46         "k": tf.zeros([1, 0, num_heads, dim_per_head]),
  47         "v": tf.zeros([1, 0, num_heads, dim_per_head]),
  48     }
  49     y = layer(x, bias, training=True, cache=cache)
  50     self.assertEqual(y.shape, (1, length, 64,))
  51     self.assertEqual(cache["k"].shape, (1, length, num_heads, dim_per_head,))
  52     self.assertEqual(cache["v"].shape, (1, length, num_heads, dim_per_head,))
  53 
  54   def test_embedding_shared_weights(self):
  55     vocab_size = 50
  56     hidden_size = 64
  57     length = 2
  58     layer = embedding_layer.EmbeddingSharedWeights(vocab_size, hidden_size)
  59     self.assertDictEqual(layer.get_config(), {
  60         "vocab_size": 50,
  61         "hidden_size": 64,
  62     })
  63 
  64     idx = tf.ones([1, length], dtype="int32")
  65     y = layer(idx)
  66     self.assertEqual(y.shape, (1, length, hidden_size,))
  67     x = tf.ones([1, length, hidden_size])
  68     output = layer(x, "linear")
  69     self.assertEqual(output.shape, (1, length, vocab_size,))
  70 
  71   def test_feed_forward_network(self):
  72     hidden_size = 64
  73     filter_size = 32
  74     relu_dropout = 0.5
  75     layer = ffn_layer.FeedForwardNetwork(hidden_size, filter_size, relu_dropout)
  76     self.assertDictEqual(layer.get_config(), {
  77         "hidden_size": hidden_size,
  78         "filter_size": filter_size,
  79         "relu_dropout": relu_dropout,
  80     })
  81     length = 2
  82     x = tf.ones([1, length, hidden_size])
  83     y = layer(x, training=True)
  84     self.assertEqual(y.shape, (1, length, hidden_size,))
  85 
  86   def test_metric_layer(self):
  87     vocab_size = 50
  88     logits = tf.keras.layers.Input((None, vocab_size),
  89                                    dtype="float32",
  90                                    name="logits")
  91     targets = tf.keras.layers.Input((None,), dtype="int64", name="targets")
  92     output_logits = metrics.MetricLayer(vocab_size)([logits, targets])
  93     self.assertEqual(output_logits.shape.as_list(), [None, None, vocab_size,])
