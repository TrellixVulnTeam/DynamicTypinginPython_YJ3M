  50 def translate_and_compute_bleu(model,
  51                                params,
  52                                subtokenizer,
  53                                bleu_source,
  54                                bleu_ref,
  55                                distribution_strategy=None):
  56   """Translate file and report the cased and uncased bleu scores.
  57 
  58   Args:
  59     model: A Keras model, used to generate the translations.
  60     params: A dictionary, containing the translation related parameters.
  61     subtokenizer: A subtokenizer object, used for encoding and decoding source
  62       and translated lines.
  63     bleu_source: A file containing source sentences for translation.
  64     bleu_ref: A file containing the reference for the translated sentences.
  65     distribution_strategy: A platform distribution strategy, used for TPU based
  66       translation.
  67 
  68   Returns:
  69     uncased_score: A float, the case insensitive BLEU score.
  70     cased_score: A float, the case sensitive BLEU score.
  71   """
  72   # Create temporary file to store translation.
  73   tmp = tempfile.NamedTemporaryFile(delete=False)
  74   tmp_filename = tmp.name
  75 
  76   translate.translate_file(
  77       model,
  78       params,
  79       subtokenizer,
  80       bleu_source,
  81       output_file=tmp_filename,
  82       print_all_translations=False,
  83       distribution_strategy=distribution_strategy)
  84 
  85   # Compute uncased and cased bleu scores.
  86   uncased_score = compute_bleu.bleu_wrapper(bleu_ref, tmp_filename, False)
  87   cased_score = compute_bleu.bleu_wrapper(bleu_ref, tmp_filename, True)
  88   os.remove(tmp_filename)
  89   return uncased_score, cased_score

  92 def evaluate_and_log_bleu(model,
  93                           params,
  94                           bleu_source,
  95                           bleu_ref,
  96                           vocab_file,
  97                           distribution_strategy=None):
  98   """Calculate and record the BLEU score.
  99 
 100   Args:
 101     model: A Keras model, used to generate the translations.
 102     params: A dictionary, containing the translation related parameters.
 103     bleu_source: A file containing source sentences for translation.
 104     bleu_ref: A file containing the reference for the translated sentences.
 105     vocab_file: A file containing the vocabulary for translation.
 106     distribution_strategy: A platform distribution strategy, used for TPU based
 107       translation.
 108 
 109   Returns:
 110     uncased_score: A float, the case insensitive BLEU score.
 111     cased_score: A float, the case sensitive BLEU score.
 112   """
 113   subtokenizer = tokenizer.Subtokenizer(vocab_file)
 114 
 115   uncased_score, cased_score = translate_and_compute_bleu(
 116       model, params, subtokenizer, bleu_source, bleu_ref, distribution_strategy)
 117 
 118   logging.info("Bleu score (uncased): %s", uncased_score)
 119   logging.info("Bleu score (cased): %s", cased_score)
 120   return uncased_score, cased_score

 123 class TransformerTask(object):
 124   """Main entry of Transformer model."""

 126   def __init__(self, flags_obj):
 127     """Init function of TransformerMain.
 128 
 129     Args:
 130       flags_obj: Object containing parsed flag values, i.e., FLAGS.
 131 
 132     Raises:
 133       ValueError: if not using static batch for input data on TPU.
 134     """
 135     self.flags_obj = flags_obj
 136     self.predict_model = None
 137 
 138     # Add flag-defined parameters to params object
 139     num_gpus = flags_core.get_num_gpus(flags_obj)
 140     self.params = params = misc.get_model_params(flags_obj.param_set, num_gpus)
 141 
 142     params["num_gpus"] = num_gpus
 143     params["use_ctl"] = flags_obj.use_ctl
 144     params["data_dir"] = flags_obj.data_dir
 145     params["model_dir"] = flags_obj.model_dir
 146     params["static_batch"] = flags_obj.static_batch
 147     params["max_length"] = flags_obj.max_length
 148     params["decode_batch_size"] = flags_obj.decode_batch_size
 149     params["decode_max_length"] = flags_obj.decode_max_length
 150     params["padded_decode"] = flags_obj.padded_decode
 151     params["max_io_parallelism"] = (
 152         flags_obj.num_parallel_calls or tf.data.experimental.AUTOTUNE)
 153 
 154     params["use_synthetic_data"] = flags_obj.use_synthetic_data
 155     params["batch_size"] = flags_obj.batch_size or params["default_batch_size"]
 156     params["repeat_dataset"] = None
 157     params["dtype"] = flags_core.get_tf_dtype(flags_obj)
 158     params["enable_tensorboard"] = flags_obj.enable_tensorboard
 159     params["enable_metrics_in_training"] = flags_obj.enable_metrics_in_training
 160     params["steps_between_evals"] = flags_obj.steps_between_evals
 161     params["enable_checkpointing"] = flags_obj.enable_checkpointing
 162 
 163     self.distribution_strategy = distribution_utils.get_distribution_strategy(
 164         distribution_strategy=flags_obj.distribution_strategy,
 165         num_gpus=num_gpus,
 166         all_reduce_alg=flags_obj.all_reduce_alg,
 167         num_packs=flags_obj.num_packs,
 168         tpu_address=flags_obj.tpu or "")
 169     if self.use_tpu:
 170       params["num_replicas"] = self.distribution_strategy.num_replicas_in_sync
 171     else:
 172       logging.info("Running transformer with num_gpus = %d", num_gpus)
 173 
 174     if self.distribution_strategy:
 175       logging.info("For training, using distribution strategy: %s",
 176                    self.distribution_strategy)
 177     else:
 178       logging.info("Not using any distribution strategy.")
 179 
 180     performance.set_mixed_precision_policy(
 181         params["dtype"],
 182         flags_core.get_loss_scale(flags_obj, default_for_fp16="dynamic"))

 184   @property
 185   def use_tpu(self):
 186     if self.distribution_strategy:
 187       return isinstance(self.distribution_strategy,
 188                         tf.distribute.experimental.TPUStrategy)
 189     return False
 190 
 191   def train(self):
 192     """Trains the model."""
 193     params = self.params
 194     flags_obj = self.flags_obj
 195     # Sets config options.
 196     keras_utils.set_session_config(enable_xla=flags_obj.enable_xla)
 197 
 198     _ensure_dir(flags_obj.model_dir)
 199     with distribution_utils.get_strategy_scope(self.distribution_strategy):
 200       model = transformer.create_model(params, is_train=True)
 201       opt = self._create_optimizer()
 202 
 203       current_step = 0
 204       checkpoint = tf.train.Checkpoint(model=model, optimizer=opt)
 205       latest_checkpoint = tf.train.latest_checkpoint(flags_obj.model_dir)
 206       if latest_checkpoint:
 207         checkpoint.restore(latest_checkpoint)
 208         logging.info("Loaded checkpoint %s", latest_checkpoint)
 209         current_step = opt.iterations.numpy()
 210 
 211       if params["use_ctl"]:
 212         train_loss_metric = tf.keras.metrics.Mean(
 213             "training_loss", dtype=tf.float32)
 214         if params["enable_tensorboard"]:
 215           summary_writer = tf.compat.v2.summary.create_file_writer(
 216               flags_obj.model_dir)
 217         else:
 218           summary_writer = tf.compat.v2.summary.create_noop_writer()
 219         train_metrics = [train_loss_metric]
 220         if params["enable_metrics_in_training"]:
 221           train_metrics = train_metrics + model.metrics
 222       else:
 223         model.compile(opt)
 224 
 225     model.summary()
 226 
 227     if self.use_tpu:
 228       # Different from experimental_distribute_dataset,
 229       # experimental_distribute_datasets_from_function requires
 230       # per-replica/local batch size.
 231       params["batch_size"] /= self.distribution_strategy.num_replicas_in_sync
 232       train_ds = (
 233           self.distribution_strategy
 234           .experimental_distribute_datasets_from_function(
 235               lambda ctx: data_pipeline.train_input_fn(params, ctx)))
 236     else:
 237       train_ds = data_pipeline.train_input_fn(params)
 238       map_data_fn = data_pipeline.map_data_for_transformer_fn
 239       train_ds = train_ds.map(
 240           map_data_fn, num_parallel_calls=tf.data.experimental.AUTOTUNE)
 241     if params["use_ctl"]:
 242       train_ds_iterator = iter(train_ds)
 243 
 244     callbacks = self._create_callbacks(flags_obj.model_dir, 0, params)
 245 
 246     # Only TimeHistory callback is supported for CTL
 247     if params["use_ctl"]:
 248       callbacks = [cb for cb in callbacks
 249                    if isinstance(cb, keras_utils.TimeHistory)]
 250 
 251     # TODO(b/139418525): Refactor the custom training loop logic.
 252     @tf.function
 253     def train_steps(iterator, steps):
 254       """Training steps function for TPU runs.
 255 
 256       Args:
 257         iterator: The input iterator of the training dataset.
 258         steps: An integer, the number of training steps.
 259 
 260       Returns:
 261         A float, the loss value.
 262       """
 263 
 264       def _step_fn(inputs):
 265         """Per-replica step function."""
 266         inputs, targets = inputs
 267         with tf.GradientTape() as tape:
 268           logits = model([inputs, targets], training=True)
 269           loss = metrics.transformer_loss(logits, targets,
 270                                           params["label_smoothing"],
 271                                           params["vocab_size"])
 272           # Scales the loss, which results in using the average loss across all
 273           # of the replicas for backprop.
 274           scaled_loss = loss / self.distribution_strategy.num_replicas_in_sync
 275 
 276         # De-dupes variables due to keras tracking issues.
 277         tvars = list({id(v): v for v in model.trainable_variables}.values())
 278         grads = tape.gradient(scaled_loss, tvars)
 279         opt.apply_gradients(zip(grads, tvars))
 280         # For reporting, the metric takes the mean of losses.
 281         train_loss_metric.update_state(loss)
 282 
 283       for _ in tf.range(steps):
 284         train_loss_metric.reset_states()
 285         self.distribution_strategy.run(
 286             _step_fn, args=(next(iterator),))
 287 
 288     cased_score, uncased_score = None, None
 289     cased_score_history, uncased_score_history = [], []
 290     while current_step < flags_obj.train_steps:
 291       remaining_steps = flags_obj.train_steps - current_step
 292       train_steps_per_eval = (
 293           remaining_steps if remaining_steps < flags_obj.steps_between_evals
 294           else flags_obj.steps_between_evals)
 295       current_iteration = current_step // flags_obj.steps_between_evals
 296 
 297       logging.info(
 298           "Start train iteration at global step:{}".format(current_step))
 299       history = None
 300       if params["use_ctl"]:
 301         if not self.use_tpu:
 302           raise NotImplementedError(
 303               "Custom training loop on GPUs is not implemented.")
 304 
 305         # Runs training steps.
 306         with summary_writer.as_default():
 307           for cb in callbacks:
 308             cb.on_epoch_begin(current_iteration)
 309             cb.on_batch_begin(0)
 310 
 311           train_steps(
 312               train_ds_iterator,
 313               tf.convert_to_tensor(train_steps_per_eval, dtype=tf.int32))
 314           current_step += train_steps_per_eval
 315           train_loss = train_loss_metric.result().numpy().astype(float)
 316           logging.info("Train Step: %d/%d / loss = %s", current_step,
 317                        flags_obj.train_steps, train_loss)
 318 
 319           for cb in callbacks:
 320             cb.on_batch_end(train_steps_per_eval - 1)
 321             cb.on_epoch_end(current_iteration)
 322 
 323           if params["enable_tensorboard"]:
 324             for metric_obj in train_metrics:
 325               tf.compat.v2.summary.scalar(metric_obj.name, metric_obj.result(),
 326                                           current_step)
 327               summary_writer.flush()
 328 
 329         for cb in callbacks:
 330           cb.on_train_end()
 331 
 332         if flags_obj.enable_checkpointing:
 333           # avoid check-pointing when running for benchmarking.
 334           checkpoint_name = checkpoint.save(
 335               os.path.join(flags_obj.model_dir,
 336                            "ctl_step_{}.ckpt".format(current_step)))
 337           logging.info("Saved checkpoint to %s", checkpoint_name)
 338       else:
 339         if self.use_tpu:
 340           raise NotImplementedError(
 341               "Keras model.fit on TPUs is not implemented.")
 342         history = model.fit(
 343             train_ds,
 344             initial_epoch=current_iteration,
 345             epochs=current_iteration + 1,
 346             steps_per_epoch=train_steps_per_eval,
 347             callbacks=callbacks,
 348             # If TimeHistory is enabled, progress bar would be messy. Increase
 349             # the verbose level to get rid of it.
 350             verbose=(2 if flags_obj.enable_time_history else 1))
 351         current_step += train_steps_per_eval
 352         logging.info("Train history: {}".format(history.history))
 353 
 354       logging.info("End train iteration at global step:{}".format(current_step))
 355 
 356       if (flags_obj.bleu_source and flags_obj.bleu_ref):
 357         uncased_score, cased_score = self.eval()
 358         cased_score_history.append([current_iteration + 1, cased_score])
 359         uncased_score_history.append([current_iteration + 1, uncased_score])
 360 
 361     stats = ({
 362         "loss": train_loss
 363     } if history is None else {})
 364     misc.update_stats(history, stats, callbacks)
 365     if uncased_score and cased_score:
 366       stats["bleu_uncased"] = uncased_score
 367       stats["bleu_cased"] = cased_score
 368       stats["bleu_uncased_history"] = uncased_score_history
 369       stats["bleu_cased_history"] = cased_score_history
 370     return stats
 371 
 372   def eval(self):
 373     """Evaluates the model."""
 374     distribution_strategy = self.distribution_strategy if self.use_tpu else None
 375 
 376     # We only want to create the model under DS scope for TPU case.
 377     # When 'distribution_strategy' is None, a no-op DummyContextManager will
 378     # be used.
 379     with distribution_utils.get_strategy_scope(distribution_strategy):
 380       if not self.predict_model:
 381         self.predict_model = transformer.create_model(self.params, False)
 382       self._load_weights_if_possible(
 383           self.predict_model,
 384           tf.train.latest_checkpoint(self.flags_obj.model_dir))
 385       self.predict_model.summary()
 386     return evaluate_and_log_bleu(
 387         self.predict_model, self.params, self.flags_obj.bleu_source,
 388         self.flags_obj.bleu_ref, self.flags_obj.vocab_file,
 389         distribution_strategy)
 390 
 391   def predict(self):
 392     """Predicts result from the model."""
 393     params = self.params
 394     flags_obj = self.flags_obj
 395 
 396     with tf.name_scope("model"):
 397       model = transformer.create_model(params, is_train=False)
 398       self._load_weights_if_possible(
 399           model, tf.train.latest_checkpoint(self.flags_obj.model_dir))
 400       model.summary()
 401     subtokenizer = tokenizer.Subtokenizer(flags_obj.vocab_file)
 402 
 403     ds = data_pipeline.eval_input_fn(params)
 404     ds = ds.map(lambda x, y: x).take(_SINGLE_SAMPLE)
 405     ret = model.predict(ds)
 406     val_outputs, _ = ret
 407     length = len(val_outputs)
 408     for i in range(length):
 409       translate.translate_from_input(val_outputs[i], subtokenizer)
 410 
 411   def _create_callbacks(self, cur_log_dir, init_steps, params):
 412     """Creates a list of callbacks."""
 413     sfunc = optimizer.LearningRateFn(params["learning_rate"],
 414                                      params["hidden_size"],
 415                                      params["learning_rate_warmup_steps"])
 416     scheduler_callback = optimizer.LearningRateScheduler(sfunc, init_steps)
 417     callbacks = misc.get_callbacks()
 418     callbacks.append(scheduler_callback)
 419     if params["enable_checkpointing"]:
 420       ckpt_full_path = os.path.join(cur_log_dir, "cp-{epoch:04d}.ckpt")
 421       callbacks.append(
 422           tf.keras.callbacks.ModelCheckpoint(
 423               ckpt_full_path, save_weights_only=True))
 424     return callbacks
 425 
 426   def _load_weights_if_possible(self, model, init_weight_path=None):
 427     """Loads model weights when it is provided."""
 428     if init_weight_path:
 429       logging.info("Load weights: {}".format(init_weight_path))
 430       # TODO(b/139414977): Having the same variable restoring method for both
 431       # TPU and GPU.
 432       if self.use_tpu:
 433         checkpoint = tf.train.Checkpoint(
 434             model=model, optimizer=self._create_optimizer())
 435         checkpoint.restore(init_weight_path)
 436       else:
 437         model.load_weights(init_weight_path)
 438     else:
 439       logging.info("Weights not loaded from path:{}".format(init_weight_path))
 440 
 441   def _create_optimizer(self):
 442     """Creates optimizer."""
 443     params = self.params
 444     lr_schedule = optimizer.LearningRateSchedule(
 445         params["learning_rate"], params["hidden_size"],
 446         params["learning_rate_warmup_steps"])
 447     opt = tf.keras.optimizers.Adam(
 448         lr_schedule if self.use_tpu else params["learning_rate"],
 449         params["optimizer_adam_beta1"],
 450         params["optimizer_adam_beta2"],
 451         epsilon=params["optimizer_adam_epsilon"])
 452 
 453     opt = performance.configure_optimizer(
 454         opt,
 455         use_float16=params["dtype"] == tf.float16,
 456         use_graph_rewrite=self.flags_obj.fp16_implementation == "graph_rewrite",
 457         loss_scale=flags_core.get_loss_scale(
 458             self.flags_obj, default_for_fp16="dynamic"))
 459 
 460     return opt

 463 def _ensure_dir(log_dir):
 464   """Makes log dir if not existed."""
 465   if not tf.io.gfile.exists(log_dir):
 466     tf.io.gfile.makedirs(log_dir)

 469 def main(_):
 470   flags_obj = flags.FLAGS
 471   if flags_obj.enable_mlir_bridge:
 472     tf.config.experimental.enable_mlir_bridge()
 473   task = TransformerTask(flags_obj)
 474 
 475   # Execute flag override logic for better model performance
 476   if flags_obj.tf_gpu_thread_mode:
 477     keras_utils.set_gpu_thread_mode_and_count(
 478         per_gpu_thread_count=flags_obj.per_gpu_thread_count,
 479         gpu_thread_mode=flags_obj.tf_gpu_thread_mode,
 480         num_gpus=flags_obj.num_gpus,
 481         datasets_num_private_threads=flags_obj.datasets_num_private_threads)
 482 
 483   if flags_obj.mode == "train":
 484     task.train()
 485   elif flags_obj.mode == "predict":
 486     task.predict()
 487   elif flags_obj.mode == "eval":
 488     task.eval()
 489   else:
 490     raise ValueError("Invalid mode {}".format(flags_obj.mode))
 491 
 492 
 493 if __name__ == "__main__":
 494   logging.set_verbosity(logging.INFO)
 495   misc.define_transformer_flags()
 496   app.run(main)
