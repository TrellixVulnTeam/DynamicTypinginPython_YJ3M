  15 """Functions for calculating loss, accuracy, and other model metrics.
  16 
  17 Metrics:

  34 def _pad_tensors_to_same_length(x, y):
  35   """Pad x and y so that the results have the same length (second dimension)."""
  36   with tf.name_scope("pad_to_same_length"):
  37     x_length = tf.shape(x)[1]
  38     y_length = tf.shape(y)[1]
  39 
  40     max_length = tf.maximum(x_length, y_length)
  41 
  42     x = tf.pad(x, [[0, 0], [0, max_length - x_length], [0, 0]])
  43     y = tf.pad(y, [[0, 0], [0, max_length - y_length]])
  44     return x, y

  47 def padded_cross_entropy_loss(logits, labels, smoothing, vocab_size):
  48   """Calculate cross entropy loss while ignoring padding.
  49 
  50   Args:
  51     logits: Tensor of size [batch_size, length_logits, vocab_size]
  52     labels: Tensor of size [batch_size, length_labels]
  53     smoothing: Label smoothing constant, used to determine the on and off values
  54     vocab_size: int size of the vocabulary
  55 
  56   Returns:
  57     Returns the cross entropy loss and weight tensors: float32 tensors with
  58       shape [batch_size, max(length_logits, length_labels)]
  59   """
  60   with tf.name_scope("loss"):
  61     logits, labels = _pad_tensors_to_same_length(logits, labels)
  62 
  63     # Calculate smoothing cross entropy
  64     with tf.name_scope("smoothing_cross_entropy"):
  65       confidence = 1.0 - smoothing
  66       low_confidence = (1.0 - confidence) / tf.cast(vocab_size - 1, tf.float32)
  67       soft_targets = tf.one_hot(
  68           tf.cast(labels, tf.int32),
  69           depth=vocab_size,
  70           on_value=confidence,
  71           off_value=low_confidence)
  72       xentropy = tf.nn.softmax_cross_entropy_with_logits(
  73           logits=logits, labels=soft_targets)
  74 
  75       # Calculate the best (lowest) possible value of cross entropy, and
  76       # subtract from the cross entropy loss.
  77       normalizing_constant = -(
  78           confidence * tf.math.log(confidence) +
  79           tf.cast(vocab_size - 1, tf.float32) * low_confidence *
  80           tf.math.log(low_confidence + 1e-20))
  81       xentropy -= normalizing_constant
  82 
  83     weights = tf.cast(tf.not_equal(labels, 0), tf.float32)
  84     return xentropy * weights, weights
  85

  87 def padded_accuracy(logits, labels):
  88   """Percentage of times that predictions matches labels on non-0s."""
  89   with tf.name_scope("padded_accuracy"):
  90     logits, labels = _pad_tensors_to_same_length(logits, labels)
  91     weights = tf.cast(tf.not_equal(labels, 0), tf.float32)
  92     outputs = tf.cast(tf.argmax(logits, axis=-1), tf.int32)
  93     padded_labels = tf.cast(labels, tf.int32)
  94     return tf.cast(tf.equal(outputs, padded_labels), tf.float32), weights
  95 
  96 
  97 def padded_accuracy_topk(logits, labels, k):
  98   """Percentage of times that top-k predictions matches labels on non-0s."""
  99   with tf.name_scope("padded_accuracy_topk"):
 100     logits, labels = _pad_tensors_to_same_length(logits, labels)
 101     weights = tf.cast(tf.not_equal(labels, 0), tf.float32)
 102     effective_k = tf.minimum(k, tf.shape(logits)[-1])
 103     _, outputs = tf.nn.top_k(logits, k=effective_k)
 104     outputs = tf.cast(outputs, tf.int32)
 105     padded_labels = tf.cast(labels, tf.int32)
 106     padded_labels = tf.expand_dims(padded_labels, axis=-1)
 107     padded_labels += tf.zeros_like(outputs)  # Pad to same shape.
 108     same = tf.cast(tf.equal(outputs, padded_labels), tf.float32)
 109     same_topk = tf.reduce_sum(same, axis=-1)
 110     return same_topk, weights
 111 
 112 
 113 def padded_accuracy_top5(logits, labels):
 114   return padded_accuracy_topk(logits, labels, 5)

 117 def padded_sequence_accuracy(logits, labels):
 118   """Percentage of times that predictions matches labels everywhere (non-0)."""
 119   with tf.name_scope("padded_sequence_accuracy"):
 120     logits, labels = _pad_tensors_to_same_length(logits, labels)
 121     weights = tf.cast(tf.not_equal(labels, 0), tf.float32)
 122     outputs = tf.cast(tf.argmax(logits, axis=-1), tf.int32)
 123     padded_labels = tf.cast(labels, tf.int32)
 124     not_correct = tf.cast(tf.not_equal(outputs, padded_labels),
 125                           tf.float32) * weights
 126     axis = list(range(1, len(outputs.get_shape())))
 127     correct_seq = 1.0 - tf.minimum(1.0, tf.reduce_sum(not_correct, axis=axis))
 128     return correct_seq, tf.constant(1.0)

 131 def padded_neg_log_perplexity(logits, labels, vocab_size):
 132   """Average log-perplexity excluding padding 0s. No smoothing."""
 133   num, den = padded_cross_entropy_loss(logits, labels, 0, vocab_size)
 134   return -num, den

 137 class MetricLayer(tf.keras.layers.Layer):
 138   """Custom a layer of metrics for Transformer model."""
 139 
 140   def __init__(self, vocab_size):
 141     super(MetricLayer, self).__init__()
 142     self.vocab_size = vocab_size
 143     self.metric_mean_fns = []
 144 
 145   def build(self, input_shape):
 146     """"Builds metric layer."""
 147     neg_log_perplexity = functools.partial(
 148         padded_neg_log_perplexity, vocab_size=self.vocab_size)
 149     self.metric_mean_fns = [
 150         (tf.keras.metrics.Mean("accuracy"), padded_accuracy),
 151         (tf.keras.metrics.Mean("accuracy_top5"), padded_accuracy_top5),
 152         (tf.keras.metrics.Mean("accuracy_per_sequence"),
 153          padded_sequence_accuracy),
 154         (tf.keras.metrics.Mean("neg_log_perplexity"), neg_log_perplexity),
 155     ]
 156     super(MetricLayer, self).build(input_shape)
 157 
 158   def get_config(self):
 159     return {"vocab_size": self.vocab_size}
 160 
 161   def call(self, inputs):
 162     logits, targets = inputs[0], inputs[1]
 163     for mean, fn in self.metric_mean_fns:
 164       m = mean(*fn(logits, targets))
 165       self.add_metric(m)
 166     return logits

 169 def transformer_loss(logits, labels, smoothing, vocab_size):
 170   """Calculates total loss containing cross entropy with padding ignored.
 171 
 172   Args:
 173     logits: Tensor of size [batch_size, length_logits, vocab_size]
 174     labels: Tensor of size [batch_size, length_labels]
 175     smoothing: Label smoothing constant, used to determine the on and off values
 176     vocab_size: int size of the vocabulary
 177 
 178   Returns:
 179     A scalar float tensor for loss.
 180   """
 181   xentropy, weights = padded_cross_entropy_loss(logits, labels, smoothing,
 182                                                 vocab_size)
 183   return tf.reduce_sum(xentropy) / tf.reduce_sum(weights)
