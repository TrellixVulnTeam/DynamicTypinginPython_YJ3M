  15 """Defines Transformer model parameters."""
  16 
  17 from collections import defaultdict
  18 
  19 
  20 BASE_PARAMS = defaultdict(
  21     lambda: None,  # Set default value to None.
  22 
  23     # Input params
  24     default_batch_size=2048,  # Maximum number of tokens per batch of examples.
  25     default_batch_size_tpu=32768,
  26     max_length=256,  # Maximum number of tokens per example.
  27 
  28     # Model params
  29     initializer_gain=1.0,  # Used in trainable variable initialization.
  30     vocab_size=33708,  # Number of tokens defined in the vocabulary file.
  31     hidden_size=512,  # Model dimension in the hidden layers.
  32     num_hidden_layers=6,  # Number of layers in the encoder and decoder stacks.
  33     num_heads=8,  # Number of heads to use in multi-headed attention.
  34     filter_size=2048,  # Inner layer dimension in the feedforward network.
  35 
  36     # Dropout values (only used when training)
  37     layer_postprocess_dropout=0.1,
  38     attention_dropout=0.1,
  39     relu_dropout=0.1,
  40 
  41     # Training params
  42     label_smoothing=0.1,
  43     learning_rate=2.0,
  44     learning_rate_decay_rate=1.0,
  45     learning_rate_warmup_steps=16000,
  46 
  47     # Optimizer params
  48     optimizer_adam_beta1=0.9,
  49     optimizer_adam_beta2=0.997,
  50     optimizer_adam_epsilon=1e-09,
  51 
  52     # Default prediction params
  53     extra_decode_length=50,
  54     beam_size=4,
  55     alpha=0.6,  # used to calculate length normalization in beam search
  56 
  57     # TPU specific parameters
  58     use_tpu=False,
  59     static_batch=False,
  60     allow_ffn_pad=True,
  61 )
  62 
  63 BIG_PARAMS = BASE_PARAMS.copy()
  64 BIG_PARAMS.update(
  65     default_batch_size=4096,
  66 
  67     # default batch size is smaller than for BASE_PARAMS due to memory limits.
  68     default_batch_size_tpu=16384,
  69 
  70     hidden_size=1024,
  71     filter_size=4096,
  72     num_heads=16,
  73 )
  74 
  75 # Parameters for running the model in multi gpu. These should not change the
  76 # params that modify the model shape (such as the hidden_size or num_heads).
  77 BASE_MULTI_GPU_PARAMS = BASE_PARAMS.copy()
  78 BASE_MULTI_GPU_PARAMS.update(
  79     learning_rate_warmup_steps=8000
  80 )
  81 
  82 BIG_MULTI_GPU_PARAMS = BIG_PARAMS.copy()
  83 BIG_MULTI_GPU_PARAMS.update(
  84     layer_postprocess_dropout=0.3,
  85     learning_rate_warmup_steps=8000
  86 )
  87 
  88 # Parameters for testing the model
  89 TINY_PARAMS = BASE_PARAMS.copy()
  90 TINY_PARAMS.update(
  91     default_batch_size=1024,
  92     default_batch_size_tpu=1024,
  93     hidden_size=32,
  94     num_heads=4,
  95     filter_size=256,
  96 )
