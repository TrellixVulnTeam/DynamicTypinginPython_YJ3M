  15 """Implementation of fully connected network."""

  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import tensorflow as tf
  22 
  23 
  24 class FeedForwardNetwork(tf.keras.layers.Layer):
  25   """Fully connected feedforward network."""
  26 
  27   def __init__(self, hidden_size, filter_size, relu_dropout):
  28     """Initialize FeedForwardNetwork.
  29 
  30     Args:
  31       hidden_size: int, output dim of hidden layer.
  32       filter_size: int, filter size for the inner (first) dense layer.
  33       relu_dropout: float, dropout rate for training.
  34     """
  35     super(FeedForwardNetwork, self).__init__()
  36     self.hidden_size = hidden_size
  37     self.filter_size = filter_size
  38     self.relu_dropout = relu_dropout
  39 
  40   def build(self, input_shape):
  41     self.filter_dense_layer = tf.keras.layers.Dense(
  42         self.filter_size,
  43         use_bias=True,
  44         activation=tf.nn.relu,
  45         name="filter_layer")
  46     self.output_dense_layer = tf.keras.layers.Dense(
  47         self.hidden_size, use_bias=True, name="output_layer")
  48     super(FeedForwardNetwork, self).build(input_shape)
  49 
  50   def get_config(self):
  51     return {
  52         "hidden_size": self.hidden_size,
  53         "filter_size": self.filter_size,
  54         "relu_dropout": self.relu_dropout,
  55     }
  56 
  57   def call(self, x, training):
  58     """Return outputs of the feedforward network.
  59 
  60     Args:
  61       x: tensor with shape [batch_size, length, hidden_size]
  62       training: boolean, whether in training mode or not.
  63 
  64     Returns:
  65       Output of the feedforward network.
  66       tensor with shape [batch_size, length, hidden_size]
  67     """
  68     # Retrieve dynamically known shapes
  69     batch_size = tf.shape(x)[0]
  70     length = tf.shape(x)[1]
  71 
  72     output = self.filter_dense_layer(x)
  73     if training:
  74       output = tf.nn.dropout(output, rate=self.relu_dropout)
  75     output = self.output_dense_layer(output)
  76 
  77     return output
