  28 class WarmUp(tf.keras.optimizers.schedules.LearningRateSchedule):
  29   """Applies a warmup schedule on a given learning rate decay schedule."""
  30 
  31   def __init__(self,
  32                initial_learning_rate,
  33                decay_schedule_fn,
  34                warmup_steps,
  35                power=1.0,
  36                name=None):
  37     super(WarmUp, self).__init__()
  38     self.initial_learning_rate = initial_learning_rate
  39     self.warmup_steps = warmup_steps
  40     self.power = power
  41     self.decay_schedule_fn = decay_schedule_fn
  42     self.name = name
  43 
  44   def __call__(self, step):
  45     with tf.name_scope(self.name or 'WarmUp') as name:
  46       # Implements polynomial warmup. i.e., if global_step < warmup_steps, the
  47       # learning rate will be `global_step/num_warmup_steps * init_lr`.
  48       global_step_float = tf.cast(step, tf.float32)
  49       warmup_steps_float = tf.cast(self.warmup_steps, tf.float32)
  50       warmup_percent_done = global_step_float / warmup_steps_float
  51       warmup_learning_rate = (
  52           self.initial_learning_rate *
  53           tf.math.pow(warmup_percent_done, self.power))
  54       return tf.cond(
  55           global_step_float < warmup_steps_float,
  56           lambda: warmup_learning_rate,
  57           lambda: self.decay_schedule_fn(step),
  58           name=name)
  59 
  60   def get_config(self):
  61     return {
  62         'initial_learning_rate': self.initial_learning_rate,
  63         'decay_schedule_fn': self.decay_schedule_fn,
  64         'warmup_steps': self.warmup_steps,
  65         'power': self.power,
  66         'name': self.name
  67     }
  68 
  69 
  70 @gin.configurable
  71 def create_optimizer(init_lr,
  72                      num_train_steps,
  73                      num_warmup_steps,
  74                      end_lr=0.0,
  75                      optimizer_type='adamw'):
  76   """Creates an optimizer with learning rate schedule."""
  77   # Implements linear decay of the learning rate.
  78   lr_schedule = tf.keras.optimizers.schedules.PolynomialDecay(
  79       initial_learning_rate=init_lr,
  80       decay_steps=num_train_steps,
  81       end_learning_rate=end_lr)
  82   if num_warmup_steps:
  83     lr_schedule = WarmUp(
  84         initial_learning_rate=init_lr,
  85         decay_schedule_fn=lr_schedule,
  86         warmup_steps=num_warmup_steps)
  87 
  88   if optimizer_type == 'adamw':
  89     logging.info('using Adamw optimizer')
  90     optimizer = AdamWeightDecay(
  91         learning_rate=lr_schedule,
  92         weight_decay_rate=0.01,
  93         beta_1=0.9,
  94         beta_2=0.999,
  95         epsilon=1e-6,
  96         exclude_from_weight_decay=['LayerNorm', 'layer_norm', 'bias'])
  97   elif optimizer_type == 'lamb':
  98     logging.info('using Lamb optimizer')
  99     optimizer = tfa_optimizers.LAMB(
 100         learning_rate=lr_schedule,
 101         weight_decay_rate=0.01,
 102         beta_1=0.9,
 103         beta_2=0.999,
 104         epsilon=1e-6,
 105         exclude_from_weight_decay=['LayerNorm', 'layer_norm', 'bias'])
 106   else:
 107     raise ValueError('Unsupported optimizer type: ', optimizer_type)
 108 
 109   return optimizer

 112 class AdamWeightDecay(tf.keras.optimizers.Adam):
 113   """Adam enables L2 weight decay and clip_by_global_norm on gradients.
 114 
 115   Just adding the square of the weights to the loss function is *not* the
 116   correct way of using L2 regularization/weight decay with Adam, since that will
 117   interact with the m and v parameters in strange ways.
 118 
 119   Instead we want ot decay the weights in a manner that doesn't interact with
 120   the m/v parameters. This is equivalent to adding the square of the weights to
 121   the loss with plain (non-momentum) SGD.
 122   """
 123 
 124   def __init__(self,
 125                learning_rate=0.001,
 126                beta_1=0.9,
 127                beta_2=0.999,
 128                epsilon=1e-7,
 129                amsgrad=False,
 130                weight_decay_rate=0.0,
 131                include_in_weight_decay=None,
 132                exclude_from_weight_decay=None,
 133                name='AdamWeightDecay',
 134                **kwargs):
 135     super(AdamWeightDecay, self).__init__(learning_rate, beta_1, beta_2,
 136                                           epsilon, amsgrad, name, **kwargs)
 137     self.weight_decay_rate = weight_decay_rate
 138     self._include_in_weight_decay = include_in_weight_decay
 139     self._exclude_from_weight_decay = exclude_from_weight_decay
 140 
 141   @classmethod
 142   def from_config(cls, config):
 143     """Creates an optimizer from its config with WarmUp custom object."""
 144     custom_objects = {'WarmUp': WarmUp}
 145     return super(AdamWeightDecay, cls).from_config(
 146         config, custom_objects=custom_objects)
 147 
 148   def _prepare_local(self, var_device, var_dtype, apply_state):
 149     super(AdamWeightDecay, self)._prepare_local(var_device, var_dtype,
 150                                                 apply_state)
 151     apply_state[(var_device, var_dtype)]['weight_decay_rate'] = tf.constant(
 152         self.weight_decay_rate, name='adam_weight_decay_rate')
 153 
 154   def _decay_weights_op(self, var, learning_rate, apply_state):
 155     do_decay = self._do_use_weight_decay(var.name)
 156     if do_decay:
 157       return var.assign_sub(
 158           learning_rate * var *
 159           apply_state[(var.device, var.dtype.base_dtype)]['weight_decay_rate'],
 160           use_locking=self._use_locking)
 161     return tf.no_op()
 162 
 163   def apply_gradients(self,
 164                       grads_and_vars,
 165                       name=None,
 166                       experimental_aggregate_gradients=True):
 167     grads, tvars = list(zip(*grads_and_vars))
 168     if experimental_aggregate_gradients:
 169       # when experimental_aggregate_gradients = False, apply_gradients() no
 170       # longer implicitly allreduce gradients, users manually allreduce gradient
 171       # and passed the allreduced grads_and_vars. For now, the
 172       # clip_by_global_norm will be moved to before the explicit allreduce to
 173       # keep the math the same as TF 1 and pre TF 2.2 implementation.
 174       (grads, _) = tf.clip_by_global_norm(grads, clip_norm=1.0)
 175     return super(AdamWeightDecay, self).apply_gradients(
 176         zip(grads, tvars),
 177         name=name,
 178         experimental_aggregate_gradients=experimental_aggregate_gradients)
 179 
 180   def _get_lr(self, var_device, var_dtype, apply_state):
 181     """Retrieves the learning rate with the given state."""
 182     if apply_state is None:
 183       return self._decayed_lr_t[var_dtype], {}
 184 
 185     apply_state = apply_state or {}
 186     coefficients = apply_state.get((var_device, var_dtype))
 187     if coefficients is None:
 188       coefficients = self._fallback_apply_state(var_device, var_dtype)
 189       apply_state[(var_device, var_dtype)] = coefficients
 190 
 191     return coefficients['lr_t'], dict(apply_state=apply_state)
 192 
 193   def _resource_apply_dense(self, grad, var, apply_state=None):
 194     lr_t, kwargs = self._get_lr(var.device, var.dtype.base_dtype, apply_state)
 195     decay = self._decay_weights_op(var, lr_t, apply_state)
 196     with tf.control_dependencies([decay]):
 197       return super(AdamWeightDecay,
 198                    self)._resource_apply_dense(grad, var, **kwargs)
 199 
 200   def _resource_apply_sparse(self, grad, var, indices, apply_state=None):
 201     lr_t, kwargs = self._get_lr(var.device, var.dtype.base_dtype, apply_state)
 202     decay = self._decay_weights_op(var, lr_t, apply_state)
 203     with tf.control_dependencies([decay]):
 204       return super(AdamWeightDecay,
 205                    self)._resource_apply_sparse(grad, var, indices, **kwargs)
 206 
 207   def get_config(self):
 208     config = super(AdamWeightDecay, self).get_config()
 209     config.update({
 210         'weight_decay_rate': self.weight_decay_rate,
 211     })
 212     return config
 213 
 214   def _do_use_weight_decay(self, param_name):
 215     """Whether to use L2 weight decay for `param_name`."""
 216     if self.weight_decay_rate == 0:
 217       return False
 218 
 219     if self._include_in_weight_decay:
 220       for r in self._include_in_weight_decay:
 221         if re.search(r, param_name) is not None:
 222           return True
 223 
 224     if self._exclude_from_weight_decay:
 225       for r in self._exclude_from_weight_decay:
 226         if re.search(r, param_name) is not None:
 227           return False
 228     return True
