 138 def detect_vendored_libs(vendor_dir):
 139     retval = []
 140     for item in vendor_dir.iterdir():
 141         if item.is_dir():
 142             retval.append(item.name)
 143         elif "LICENSE" in item.name or "COPYING" in item.name:
 144             continue
 145         elif item.name.endswith(".pyi"):
 146             continue
 147         elif item.name not in FILE_WHITE_LIST:
 148             retval.append(item.name[:-3])
 149     return retval

 160 def rewrite_file_imports(item, vendored_libs, vendor_dir):
 161     """Rewrite 'import xxx' and 'from xxx import' for vendored_libs"""
 162     # log('Reading file: %s' % item)
 163     try:
 164         text = item.read_text(encoding="utf-8")
 165     except UnicodeDecodeError:
 166         text = item.read_text(encoding="cp1252")
 167     renames = LIBRARY_RENAMES
 168     for k in LIBRARY_RENAMES.keys():
 169         if k not in vendored_libs:
 170             vendored_libs.append(k)
 171     for lib in vendored_libs:
 172         to_lib = lib
 173         if lib in renames:
 174             to_lib = renames[lib]
 175         text = re.sub(
 176             r"([\n\s]*)import %s([\n\s\.]+)" % lib, r"\1import %s\2" % to_lib, text,
 177         )
 178         text = re.sub(r"([\n\s]*)from %s([\s\.])+" % lib, r"\1from %s\2" % to_lib, text,)
 179         text = re.sub(
 180             r"(\n\s*)__import__\('%s([\s'\.])+" % lib,
 181             r"\1__import__('%s\2" % to_lib,
 182             text,
 183         )
 184     item.write_text(text, encoding="utf-8")

 192 def _recursive_write_to_zip(zf, path, root=None):
 193     if path == Path(zf.filename):
 194         return
 195     if root is None:
 196         if not path.is_dir():
 197             raise ValueError('root is required for non-directory path')
 198         root = path
 199     if not path.is_dir():
 200         zf.write(str(path), str(path.relative_to(root)))
 201         return
 202     for c in path.iterdir():
 203         _recursive_write_to_zip(zf, c, root)

 328 def rename_if_needed(ctx, vendor_dir, item):
 329     rename_dict = LIBRARY_RENAMES if vendor_dir.name != "patched" else PATCHED_RENAMES
 330     new_path = None

 333         new_path = item.parent / new_name

 358 def _ensure_package_in_requirements(ctx, requirements_file, package):
 359     requirement = None
 360     log("using requirements file: %s" % requirements_file)
 361     req_file_lines = [l for l in requirements_file.read_text().splitlines()]
 362     if package:
 363         match = [r for r in req_file_lines if r.strip().lower().startswith(package)]
 364         matched_req = None
 365         if match:
 366             for m in match:
 367                 specifiers = [m.index(s) for s in [">", "<", "=", "~"] if s in m]
 368                 if m.lower() == package or (
 369                     specifiers and m[: min(specifiers)].lower() == package
 370                 ):
 371                     matched_req = "{0}".format(m)
 372                     requirement = matched_req
 373                     log("Matched req: %r" % matched_req)
 374         if not matched_req:
 375             req_file_lines.append("{0}".format(package))
 376             log("Writing requirements file: %s" % requirements_file)
 377             requirements_file.write_text("\n".join(req_file_lines))
 378             requirement = "{0}".format(package)
 379     return requirement

 415 def install(ctx, vendor_dir, package=None):
 416     requirements_file = vendor_dir / "{0}.txt".format(vendor_dir.name)
 417     requirement = "-r {0}".format(requirements_file.as_posix())
 418     log("Using requirements file: %s" % requirement)
 419     if package:
 420         requirement = _ensure_package_in_requirements(ctx, requirements_file, package)
 421     # We use --no-deps because we want to ensure that all of our dependencies
 422     # are added to vendor.txt, this includes all dependencies recursively up
 423     # the chain.
 424     ctx.run(
 425         "pip install -t {0} --no-compile --no-deps --upgrade {1}".format(
 426             vendor_dir.as_posix(), requirement,
 427         )
 428     )
 429     # read licenses from distinfo files if possible
 430     for path in vendor_dir.glob("*.dist-info"):
 431         pkg, _, _ = path.stem.rpartition("-")
 432         license_file = path / "LICENSE"
 433         if not license_file.exists():
 434             continue
 435         if vendor_dir.joinpath(pkg).exists():
 436             vendor_dir.joinpath(pkg).joinpath("LICENSE").write_text(
 437                 license_file.read_text()
 438             )
 439         elif vendor_dir.joinpath("{0}.py".format(pkg)).exists():
 440             vendor_dir.joinpath("{0}.LICENSE".format(pkg)).write_text(
 441                 license_file.read_text()
 442             )
 443         else:
 444             pkg = pkg.replace("-", "?").replace("_", "?")
 445             matched_path = next(
 446                 iter(pth for pth in vendor_dir.glob("{0}*".format(pkg))), None
 447             )
 448             if matched_path is not None:
 449                 if matched_path.is_dir():
 450                     target = vendor_dir.joinpath(matched_path).joinpath("LICENSE")
 451                 else:
 452                     target = vendor_dir.joinpath("{0}.LICENSE".format(matched_path))
 453                 target.write_text(
 454                     license_file.read_text()
 455                 )

 489 def vendor(ctx, vendor_dir, package=None, rewrite=True):
 490     log("Reinstalling vendored libraries")
 491     is_patched = vendor_dir.name == "patched"
 492     install(ctx, vendor_dir, package=package)
 493     log("Running post-install cleanup...")
 494     post_install_cleanup(ctx, vendor_dir)
 495     # Detect the vendored packages/modules
 496     vendored_libs = detect_vendored_libs(_get_vendor_dir(ctx))
 497     log("Detected vendored libraries: %s" % ", ".join(vendored_libs))
 498 
 499     # Apply pre-patches
 500     log("Applying pre-patches...")
 501     if is_patched:
 502         apply_patches(ctx, patched=is_patched, pre=True)
 503     log("Removing scandir library files...")
 504     for extension in ("*.so", "*.pyd", "*.egg-info", "*.dist-info"):
 505         remove_all(vendor_dir.glob(extension))
 506     for dirname in ("setuptools", "pkg_resources/_vendor", "pkg_resources/extern", "bin"):
 507         drop_dir(vendor_dir / dirname)
 508 
 509     # Global import rewrites
 510     log("Renaming specified libs...")
 511     for item in vendor_dir.iterdir():
 512         if item.is_dir():

 563 @invoke.task
 564 def packages_missing_licenses(
 565     ctx, vendor_dir=None, requirements_file="vendor.txt", package=None
 566 ):
 567     if not vendor_dir:
 568         vendor_dir = _get_vendor_dir(ctx)
 569     requirements = vendor_dir.joinpath(requirements_file).read_text().splitlines()
 570     new_requirements = []
 571     LICENSE_EXTS = ("rst", "txt", "APACHE", "BSD", "md")
 572     LICENSES = [
 573         ".".join(lic)
 574         for lic in itertools.product(("LICENSE", "LICENSE-MIT"), LICENSE_EXTS)
 575     ]
 576     for i, req in enumerate(requirements):
 577         if req.startswith("git+"):
 578             pkg = req.strip().split("#egg=")[1]
 579         else:
 580             pkg = req.strip().split("=")[0]
 581         possible_pkgs = [pkg, pkg.replace("-", "_")]
 582         match_found = False
 583         if pkg in PY2_DOWNLOAD:
 584             match_found = True
 585             # print("pkg ===> %s" % pkg)
 586         if pkg in LIBRARY_DIRNAMES:
 587             possible_pkgs.append(LIBRARY_DIRNAMES[pkg])
 588         for pkgpath in possible_pkgs:
 589             pkgpath = vendor_dir.joinpath(pkgpath)
 590             py_path = pkgpath.parent / "{0}.py".format(pkgpath.stem)
 591             if pkgpath.exists() and pkgpath.is_dir():
 592                 for license_path in LICENSES:
 593                     license_path = pkgpath.joinpath(license_path)
 594                     if license_path.exists():
 595                         match_found = True
 596                         # log("%s: Trying path %s... FOUND" % (pkg, license_path))
 597                         break
 598             elif pkgpath.exists() or py_path.exists():
 599                 for license_path in LICENSES:
 600                     license_name = "{0}.{1}".format(pkgpath.stem, license_path)
 601                     license_path = pkgpath.parent / license_name
 602                     if license_path.exists():
 603                         match_found = True
 604                         # log("%s: Trying path %s... FOUND" % (pkg, license_path))
 605                         break
 606             if match_found:
 607                 break
 608         if match_found:
 609             continue
 610         else:
 611             #  log("%s: No license found in %s" % (pkg, pkgpath))
 612             new_requirements.append(req)
 613     return new_requirements

 681 def extract_license(vendor_dir, sdist):
 682     if sdist.stem.endswith(".tar"):
 683         ext = sdist.suffix[1:]
 684         with tarfile.open(sdist, mode="r:{}".format(ext)) as tar:
 685             found = find_and_extract_license(vendor_dir, tar, tar.getmembers())
 686     elif sdist.suffix in (".zip", ".whl"):
 687         with zipfile.ZipFile(sdist) as zip:
 688             found = find_and_extract_license(vendor_dir, zip, zip.infolist())
 689     else:
 690         raise NotImplementedError("new sdist type!")
 691 
 692     if not found:
 693         log("License not found in {}, will download".format(sdist.name))
 694         license_fallback(vendor_dir, sdist.name)

 697 def find_and_extract_license(vendor_dir, tar, members):
 698     found = False
 699     for member in members:
 700         try:
 701             name = member.name
 702         except AttributeError:  # zipfile
 703             name = member.filename
 704         if "LICENSE" in name or "COPYING" in name:
 705             if "/test" in name:
 706                 # some testing licenses in hml5lib and distlib
 707                 log("Ignoring {}".format(name))
 708                 continue
 709             found = True
 710             extract_license_member(vendor_dir, tar, member, name)
 711     return found

 714 def license_fallback(vendor_dir, sdist_name):
 715     """Hardcoded license URLs. Check when updating if those are still needed"""
 716     libname = libname_from_dir(sdist_name)
 717     if libname not in HARDCODED_LICENSE_URLS:
 718         raise ValueError("No hardcoded URL for {} license".format(libname))
 719 
 720     url = HARDCODED_LICENSE_URLS[libname]
 721     _, _, name = url.rpartition("/")
 722     dest = license_destination(vendor_dir, libname, name)
 723     r = requests.get(url, allow_redirects=True)
 724     log("Downloading {}".format(url))
 725     r.raise_for_status()
 726     dest.write_bytes(r.content)

 729 def libname_from_dir(dirname):
 730     """Reconstruct the library name without it's version"""
 731     parts = []
 732     for part in dirname.split("-"):
 733         if part[0].isdigit():
 734             break
 735         parts.append(part)
 736     return "-".join(parts)

 739 def license_destination(vendor_dir, libname, filename):
 740     """Given the (reconstructed) library name, find appropriate destination"""
 741     normal = vendor_dir / libname
 742     if normal.is_dir():
 743         return normal / filename
 744     lowercase = vendor_dir / libname.lower().replace("-", "_")
 745     if lowercase.is_dir():
 746         return lowercase / filename
 747     rename_dict = LIBRARY_RENAMES if vendor_dir.name != "patched" else PATCHED_RENAMES
 748     # Short circuit all logic if we are renaming the whole library
 749     if libname in rename_dict:
 750         return vendor_dir / rename_dict[libname] / filename
 751     if libname in LIBRARY_DIRNAMES:
 752         override = vendor_dir / LIBRARY_DIRNAMES[libname]
 753         if not override.exists() and override.parent.exists():
 754             # for flattened subdeps, specifically backports/weakref.py
 755             return (vendor_dir / override.parent) / "{0}.{1}".format(
 756                 override.name, filename
 757             )
 758         license_path = Path(LIBRARY_DIRNAMES[libname]) / filename
 759         if license_path.as_posix() in LICENSE_RENAMES:
 760             return vendor_dir / LICENSE_RENAMES[license_path.as_posix()]
 761         return vendor_dir / LIBRARY_DIRNAMES[libname] / filename
 762     # fallback to libname.LICENSE (used for nondirs)
 763     return vendor_dir / "{}.{}".format(libname, filename)

 766 def extract_license_member(vendor_dir, tar, member, name):
 767     mpath = Path(name)  # relative path inside the sdist
 768     dirname = list(mpath.parents)[-2].name  # -1 is .
 769     libname = libname_from_dir(dirname)
 770     dest = license_destination(vendor_dir, libname, mpath.name)
 771     log("Extracting {} into {}".format(name, dest))
 772     try:
 773         fileobj = tar.extractfile(member)
 774         dest.write_bytes(fileobj.read())
 775     except AttributeError:  # zipfile
 776         dest.write_bytes(tar.read(member))

 822 def unpin_file(contents):
 823     requirements = []
 824     for line in contents.splitlines():
 825         if "==" in line:
 826             line, _, _ = line.strip().partition("=")
 827         if not line.startswith("#"):
 828             requirements.append(line)
 829     return "\n".join(sorted(requirements))

 832 def unpin_and_copy_requirements(ctx, requirement_file, name="requirements.txt"):
 833     with TemporaryDirectory() as tempdir:
 834         target = Path(tempdir.name).joinpath("requirements.txt")
 835         contents = unpin_file(requirement_file.read_text())
 836         target.write_text(contents)
 837         env = {
 838             "PIPENV_IGNORE_VIRTUALENVS": "1",
 839             "PIPENV_NOSPIN": "1",
 840             "PIPENV_PYTHON": "2.7",
 841         }
 842         with ctx.cd(tempdir.name):
 843             ctx.run("pipenv install -r {0}".format(target.as_posix()), env=env, hide=True)
 844             result = ctx.run("pipenv lock -r", env=env, hide=True).stdout.strip()
 845             ctx.run("pipenv --rm", env=env, hide=True)
 846             result = list(sorted([line.strip() for line in result.splitlines()[1:]]))
 847             new_requirements = requirement_file.parent.joinpath(name)
 848             requirement_file.rename(
 849                 requirement_file.parent.joinpath("{}.bak".format(name))
 850             )
 851             new_requirements.write_text("\n".join(result))
