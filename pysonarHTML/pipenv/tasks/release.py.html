  33 def find_version(ctx):
  34     version_file = get_version_file(ctx).read_text()
  35     version_match = re.search(r"^__version__ = ['\"]([^'\"]*)['\"]", version_file, re.M)
  36     if version_match:
  37         return version_match.group(1)
  38     raise RuntimeError("Unable to find version string.")

  53 def _render_log():
  54     """Totally tap into Towncrier internals to get an in-memory result.
  55     """
  56     config = load_config(ROOT)
  57     definitions = config["types"]
  58     fragments, fragment_filenames = find_fragments(
  59         pathlib.Path(config["directory"]).absolute(),
  60         config["sections"],
  61         None,
  62         definitions,
  63     )
  64     rendered = render_fragments(
  65         pathlib.Path(config["template"]).read_text(encoding="utf-8"),
  66         config["issue_format"],
  67         split_fragments(fragments, definitions),
  68         definitions,
  69         config["underlines"][1:],
  70         False,  # Don't add newlines to wrapped text.
  71     )
  72     return rendered

  85 def release(ctx, manual=False, local=False, dry_run=False, pre=False, tag=None, month_offset="0"):
  86     trunc_month = False
  87     if pre:
  88         trunc_month = True
  89     drop_dist_dirs(ctx)
  90     bump_version(
  91         ctx,
  92         dry_run=dry_run,
  93         pre=pre,
  94         tag=tag,
  95         month_offset=month_offset,
  96         trunc_month=trunc_month
  97     )
  98     version = find_version(ctx)
  99     tag_content = _render_log()
 100     if dry_run:
 101         ctx.run("towncrier --draft > CHANGELOG.draft.rst")
 102         log("would remove: news/*")
 103         log("would remove: CHANGELOG.draft.rst")
 104         log("would update: pipenv/pipenv.1")
 105         log(f'Would commit with message: "Release v{version}"')
 106     else:
 107         if pre:
 108             log("generating towncrier draft...")
 109             ctx.run("towncrier --draft > CHANGELOG.draft.rst")
 110             ctx.run("git add {0}".format(get_version_file(ctx).as_posix()))
 111         else:
 112             ctx.run("towncrier")
 113             ctx.run(
 114                 "git add CHANGELOG.rst news/ {0}".format(get_version_file(ctx).as_posix())
 115             )
 116             log("removing changelog draft if present")
 117             draft_changelog = pathlib.Path("CHANGELOG.draft.rst")
 118             if draft_changelog.exists():
 119                 draft_changelog.unlink()
 120         log("generating man files...")
 121         generate_manual(ctx)
 122         ctx.run("git add pipenv/pipenv.1")
 123         ctx.run(f'git commit -m "Release v{version}"')
 124 
 125     tag_content = tag_content.replace('"', '\\"')
 126     if dry_run or pre:
 127         log(f"Generated tag content: {tag_content}")
 128         draft_rstfile = "CHANGELOG.draft.rst"
 129         markdown_path = pathlib.Path(draft_rstfile).with_suffix(".md")
 130         generate_markdown(ctx, source_rstfile=draft_rstfile)
 131         content = clean_mdchangelog(ctx, markdown_path.as_posix())
 132         log(f"would generate markdown: {markdown_path.read_text()}")
 133         if pre and not dry_run:
 134             ctx.run(f'git tag -a v{version} -m "Version v{version}\n\n{tag_content}"')
 135     else:
 136         generate_markdown(ctx)
 137         clean_mdchangelog(ctx)
 138         ctx.run(f'git tag -a v{version} -m "Version v{version}\n\n{tag_content}"')
 139     if local:
 140         build_dists(ctx)
 141         dist_pattern = f'{PACKAGE_NAME.replace("-", "[-_]")}-*'
 142         artifacts = list(ROOT.joinpath("dist").glob(dist_pattern))
 143         if dry_run:
 144             filename_display = "\n".join(f"  {a}" for a in artifacts)
 145             log(f"Would upload dists: {filename_display}")
 146         else:
 147             upload_dists(ctx)
 148             bump_version(ctx, dev=True)

 158 @invoke.task
 159 def build_dists(ctx):
 160     drop_dist_dirs(ctx)
 161     py_version = ".".join(str(v) for v in sys.version_info[:2])
 162     env = {"PIPENV_PYTHON": py_version}
 163     with ctx.cd(ROOT.as_posix()), temp_environ():
 164         executable = ctx.run(
 165             "python -c 'import sys; print(sys.executable)'", hide=True
 166         ).stdout.strip()
 167         log("Building sdist using %s ...." % executable)
 168         os.environ["PIPENV_PYTHON"] = py_version
 169         ctx.run("pipenv install --dev", env=env)
 170         ctx.run(
 171             "pipenv run pip install -e . --upgrade --upgrade-strategy=eager", env=env
 172         )
 173         log("Building wheel using python %s ...." % py_version)
 174         ctx.run(f"pipenv run python setup.py sdist bdist_wheel", env=env)

 242 @invoke.task
 243 def clean_mdchangelog(ctx, filename=None, content=None):
 244     changelog = None
 245     if not content:
 246         if filename is not None:
 247             changelog = pathlib.Path(filename)
 248         else:
 249             changelog = _get_git_root(ctx) / "CHANGELOG.md"
 250         content = changelog.read_text()
 251     content = re.sub(
 252         r"([^\n]+)\n?\s+\[[\\]+(#\d+)\]\(https://github\.com/pypa/[\w\-]+/issues/\d+\)",
 253         r"\1 \2",
 254         content,
 255         flags=re.MULTILINE,
 256     )
 257     if changelog:
 258         changelog.write_text(content)
 259     else:
 260         return content

 279 def date_offset(dt, month_offset=0, day_offset=0, truncate=False):
 280     new_month = (dt.month + month_offset) % 12
 281     year_offset = new_month // 12
 282     replace_args = {
 283         "month": dt.month + month_offset,
 284         "year": dt.year + year_offset,
 285     }
 286     log("Getting updated date from date: {0} using month offset: {1} and year offset {2}".format(
 287         dt, new_month, replace_args["year"]
 288     ))
 289     if day_offset:
 290         dt = dt + datetime.timedelta(days=day_offset)
 291         log("updated date using day offset: {0} => {1}".format(day_offset, dt))
 292     if truncate:
 293         log("Truncating...")
 294         replace_args["day"] = 1
 295     return dt.replace(**replace_args)

 299 def bump_version(ctx, dry_run=False, dev=False, pre=False, tag=None, commit=False, month_offset="0", trunc_month=False):
 300     current_version = Version.parse(__version__)
 301     today = datetime.date.today()
 302     day_offset = 0
 303     tomorrow = today + datetime.timedelta(days=1)
 304     month_offset = int(month_offset)
 305     if month_offset:
 306         # if we are offsetting by a month, grab the first day of the month
 307         trunc_month = True
 308     else:
 309         target_day = today
 310         if dev or pre:
 311             target_day = date_offset(today, day_offset=1)
 312     target_day = date_offset(
 313         today,
 314         month_offset=month_offset,
 315         day_offset=day_offset,
 316         truncate=trunc_month
 317     )
 318     log("target_day: {0}".format(target_day))
 319     target_timetuple = target_day.timetuple()[:3]
 320     new_version = current_version.replace(release=target_timetuple)
 321     if pre and dev:
 322         raise RuntimeError("Can't use 'pre' and 'dev' together!")
 323     if dev:
 324         new_version = new_version.replace(pre=None).bump_dev()
 325     elif pre:
 326         if not tag:
 327             print('Using "pre" requires a corresponding tag.')
 328             return
 329         tag_version = re.match(
 330             r"(?P<tag>alpha|a|beta|b|c|preview|pre|rc)(?P<version>[0-9]+)?", tag
 331         )
 332         tag_dict = tag_version.groupdict()
 333         tag = tag_dict.get("tag", tag)
 334         tag_version = int(tag_dict["version"]) if tag_dict["version"] is not None else 0
 335         if new_version.dev is not None:
 336             new_version = new_version.replace(dev=None)
 337         if new_version.pre_tag:
 338             if new_version.pre_tag != tag:
 339                 log("Swapping prerelease tag: {0} for {1}".format(new_version.pre_tag, tag))
 340                 new_version = new_version.replace(pre_tag=tag, pre=tag_version)
 341         else:
 342             new_version = new_version.replace(pre_tag=tag, pre=tag_version)
 343         if tag_version == 0:
 344             new_version = new_version.bump_pre(tag=tag)
 345     else:
 346         new_version = new_version.replace(pre=None, dev=None)
 347     log("Updating version to %s" % new_version.normalize())
 348     version = find_version(ctx)
 349     log("Found current version: %s" % version)
 350     if dry_run:
 351         log("Would update to: %s" % new_version.normalize())
 352     else:
 353         log("Updating to: %s" % new_version.normalize())
 354         version_file = get_version_file(ctx)
 355         file_contents = version_file.read_text()
 356         version_file.write_text(
 357             file_contents.replace(version, str(new_version.normalize()))
 358         )
 359         if commit:
 360             ctx.run("git add {0}".format(version_file.as_posix()))
 361             log("Committing...")
 362             ctx.run('git commit -s -m "Bumped version."')
