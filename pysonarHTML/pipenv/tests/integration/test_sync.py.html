  27 def test_mirror_lock_sync(PipenvInstance):
  28     with temp_environ(), PipenvInstance(chdir=True) as p:
  29         mirror_url = os.environ.pop('PIPENV_TEST_INDEX', "https://pypi.kennethreitz.org/simple")
  30         assert 'pypi.org' not in mirror_url
  31         with open(p.pipfile_path, 'w') as f:
  32             f.write("""
  33 [[source]]
  34 name = "pypi"
  35 url = "https://pypi.org/simple"
  36 verify_ssl = true
  37 
  38 [packages]
  39 six = "*"
  40             """.strip())
  41         c = p.pipenv('lock --pypi-mirror {0}'.format(mirror_url))
  42         assert c.return_code == 0
  43         c = p.pipenv('sync --pypi-mirror {0}'.format(mirror_url))
  44         assert c.return_code == 0

  47 @pytest.mark.sync
  48 @pytest.mark.lock
  49 def test_sync_should_not_lock(PipenvInstance):
  50     """Sync should not touch the lock file, even if Pipfile is changed.
  51     """
  52     with PipenvInstance(chdir=True) as p:
  53         with open(p.pipfile_path, 'w') as f:
  54             f.write("""
  55 [packages]
  56             """.strip())
  57 
  58         # Perform initial lock.
  59         c = p.pipenv('lock')
  60         assert c.return_code == 0
  61         lockfile_content = p.lockfile
  62         assert lockfile_content
  63 
  64         # Make sure sync does not trigger lockfile update.
  65         with open(p.pipfile_path, 'w') as f:
  66             f.write("""
  67 [packages]
  68 six = "*"
  69             """.strip())
  70         c = p.pipenv('sync')
  71         assert c.return_code == 0
  72         assert lockfile_content == p.lockfile

  75 @pytest.mark.sync
  76 @pytest.mark.lock
  77 def test_sync_sequential_detect_errors(PipenvInstance):
  78     with PipenvInstance() as p:
  79         with open(p.pipfile_path, 'w') as f:
  80             contents = """
  81 [packages]
  82 requests = "*"
  83         """.strip()
  84             f.write(contents)
  85 
  86         c = p.pipenv('lock')
  87         assert c.return_code == 0
  88 
  89         # Force hash mismatch when installing `requests`
  90         lock = p.lockfile
  91         lock['default']['requests']['hashes'] = ['sha256:' + '0' * 64]
  92         with open(p.lockfile_path, 'w') as f:
  93             json.dump(lock, f)
  94 
  95         c = p.pipenv('sync --sequential')
  96         assert c.return_code != 0
