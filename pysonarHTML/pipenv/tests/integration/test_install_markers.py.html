  36 @flaky
  37 @pytest.mark.markers
  38 def test_platform_python_implementation_marker(PipenvInstance):
  39     """Markers should be converted during locking to help users who input this
  40     incorrectly.
  41     """
  42     with PipenvInstance() as p:
  43         with open(p.pipfile_path, 'w') as f:
  44             contents = """
  45 [packages]
  46 depends-on-marked-package = "*"
  47             """.strip()
  48             f.write(contents)
  49 
  50         c = p.pipenv('install')
  51         assert c.return_code == 0
  52 
  53         # depends-on-marked-package has an install_requires of
  54         # 'pytz; platform_python_implementation=="CPython"'
  55         # Verify that that marker shows up in our lockfile unaltered.
  56         assert 'pytz' in p.lockfile['default']
  57         assert p.lockfile['default']['pytz'].get('markers') == \
  58             "platform_python_implementation == 'CPython'"

 105 @flaky
 106 @pytest.mark.markers
 107 @pytest.mark.install
 108 def test_global_overrides_environment_markers(PipenvInstance):
 109     """Empty (unconditional) dependency should take precedence.
 110     If a dependency is specified without environment markers, it should
 111     override dependencies with environment markers. In this example,
 112     APScheduler requires funcsigs only on Python 2, but since funcsigs is
 113     also specified as an unconditional dep, its markers should be empty.
 114     """
 115     with PipenvInstance() as p:
 116         with open(p.pipfile_path, 'w') as f:
 117             contents = """
 118 [packages]
 119 apscheduler = "*"
 120 funcsigs = "*"
 121             """.strip()
 122             f.write(contents)
 123 
 124         c = p.pipenv('install')
 125         assert c.return_code == 0
 126 
 127         assert p.lockfile['default']['funcsigs'].get('markers', '') == ''

 133 @pytest.mark.py3_only
 134 @pytest.mark.skip("test is flaky on CI")
 135 def test_resolver_unique_markers(PipenvInstance):
 136     """vcrpy has a dependency on `yarl` which comes with a marker
 137     of 'python version in "3.4, 3.5, 3.6" - this marker duplicates itself:
 138 
 139     'yarl; python version in "3.4, 3.5, 3.6"; python version in "3.4, 3.5, 3.6"'
 140 
 141     This verifies that we clean that successfully.
 142     """
 143     with PipenvInstance(chdir=True) as p:
 144         c = p.pipenv('install vcrpy==2.0.1')
 145         assert c.return_code == 0
 146         c = p.pipenv('lock')
 147         assert c.return_code == 0
 148         assert 'yarl' in p.lockfile['default']
 149         yarl = p.lockfile['default']['yarl']
 150         assert 'markers' in yarl
 151         # Two possible marker sets are ok here
 152         assert yarl['markers'] in [
 153             "python_version in '3.4, 3.5, 3.6'",
 154             "python_version >= '3.4'",
 155             "python_version >= '3.5'",  # yarl 1.3.0 requires python 3.5.3
 156         ]

 159 @flaky

 161 def test_environment_variable_value_does_not_change_hash(PipenvInstance):
 162     with PipenvInstance(chdir=True) as p:
 163         with temp_environ():
 164             with open(p.pipfile_path, 'w') as f:
 165                 f.write("""
 166 [[source]]
 167 url = 'https://${PYPI_USERNAME}:${PYPI_PASSWORD}@pypi.org/simple'
 168 verify_ssl = true
 169 name = 'pypi'
 170 
 171 [packages]
 172 six = "*"
 173 """)
 174             project = Project()
 175 
 176             os.environ['PYPI_USERNAME'] = 'whatever'
 177             os.environ['PYPI_PASSWORD'] = 'pass'
 178             assert project.get_lockfile_hash() is None
 179 
 180             c = p.pipenv('install')
 181             assert c.return_code == 0
 182             lock_hash = project.get_lockfile_hash()
 183             assert lock_hash is not None
 184             assert lock_hash == project.calculate_pipfile_hash()
 185 
 186             # sanity check on pytest
 187             assert 'PYPI_USERNAME' not in str(pipfile.load(p.pipfile_path))
 188             assert c.return_code == 0
 189             assert project.get_lockfile_hash() == project.calculate_pipfile_hash()
 190 
 191             os.environ['PYPI_PASSWORD'] = 'pass2'
 192             assert project.get_lockfile_hash() == project.calculate_pipfile_hash()
 193 
 194             with open(p.pipfile_path, 'a') as f:
 195                 f.write('requests = "==2.14.0"\n')
 196             assert project.get_lockfile_hash() != project.calculate_pipfile_hash()
