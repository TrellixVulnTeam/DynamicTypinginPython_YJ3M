  11 @pytest.mark.run

  13 def test_env(PipenvInstance):
  14     with PipenvInstance(pipfile=False, chdir=True) as p:
  15         with open('.env', 'w') as f:
  16             f.write('HELLO=WORLD')
  17 
  18         c = p.pipenv('run python -c "import os; print(os.environ[\'HELLO\'])"')
  19         assert c.return_code == 0
  20         assert 'WORLD' in c.out

  23 @pytest.mark.run
  24 def test_scripts(PipenvInstance):
  25     with PipenvInstance(chdir=True) as p:
  26         with open(p.pipfile_path, 'w') as f:
  27             f.write(r"""
  28 [scripts]
  29 printfoo = "python -c \"print('foo')\""
  30 notfoundscript = "randomthingtotally"
  31 appendscript = "cmd arg1"
  32 multicommand = "bash -c \"cd docs && make html\""
  33             """)
  34             if os.name == "nt":
  35                 f.write('scriptwithenv = "echo %HELLO%"\n')
  36             else:
  37                 f.write('scriptwithenv = "echo $HELLO"\n')
  38         c = p.pipenv('install')
  39         assert c.return_code == 0
  40 
  41         c = p.pipenv('run printfoo')
  42         assert c.return_code == 0
  43         assert c.out == 'foo\n'
  44         assert c.err == ''
  45 
  46         c = p.pipenv('run notfoundscript')
  47         assert c.return_code == 1
  48         assert c.out == ''
  49         if os.name != 'nt':     # TODO: Implement this message for Windows.
  50             assert 'Error' in c.err
  51             assert 'randomthingtotally (from notfoundscript)' in c.err
  52 
  53         project = Project()
  54 
  55         script = project.build_script('multicommand')
  56         assert script.command == 'bash'
  57         assert script.args == ['-c', 'cd docs && make html']
  58 
  59         script = project.build_script('appendscript', ['a', 'b'])
  60         assert script.command == 'cmd'
  61         assert script.args == ['arg1', 'a', 'b']
  62 
  63         with temp_environ():
  64             os.environ['HELLO'] = 'WORLD'
  65             c = p.pipenv("run scriptwithenv")
  66             assert c.ok
  67             if os.name != "nt":  # This doesn't work on CI windows.
  68                 assert c.out.strip() == "WORLD"
