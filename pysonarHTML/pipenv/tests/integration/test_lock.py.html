 393 def test_outdated_setuptools_with_pep517_legacy_build_meta_is_updated(PipenvInstance):
 394     """
 395     This test ensures we are using build isolation and a pep517 backend
 396     because the package in question includes ``pyproject.toml`` but lacks
 397     a ``build-backend`` declaration. In this case, ``pip`` defaults to using
 398     ``setuptools.build_meta:__legacy__`` as a builder, but without ``pep517``
 399     enabled and with ``setuptools==40.2.0`` installed, this build backend was
 400     not yet available. ``setuptools<40.8`` will not be aware of this backend.
 401 
 402     If pip is able to build in isolation with a pep517 backend, this will not
 403     matter and the test will still pass as pip will by default install a more
 404     recent version of ``setuptools``.
 405     """
 406     with PipenvInstance(chdir=True) as p:
 407         c = p.pipenv('run pip install "setuptools<=40.2"')
 408         assert c.return_code == 0
 409         c = p.pipenv("run python -c 'import setuptools; print(setuptools.__version__)'")
 410         assert c.return_code == 0
 411         assert c.out.strip() == "40.2.0"
 412         c = p.pipenv("install legacy-backend-package")
 413         assert c.return_code == 0
 414         assert "vistir" in p.lockfile["default"]
