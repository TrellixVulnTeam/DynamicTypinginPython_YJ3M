  11 @pytest.mark.uninstall
  12 @pytest.mark.install
  13 def test_uninstall_requests(PipenvInstance):
  14     # Uninstalling requests can fail even when uninstall Django below
  15     # succeeds, if requests was de-vendored.
  16     # See https://github.com/pypa/pipenv/issues/3644 for problems
  17     # caused by devendoring
  18     with PipenvInstance() as p:
  19         c = p.pipenv("install requests")
  20         assert c.return_code == 0
  21         assert "requests" in p.pipfile["packages"]
  22 
  23         c = p.pipenv("run python -m requests.help")
  24         assert c.return_code == 0
  25 
  26         c = p.pipenv("uninstall requests")
  27         assert c.return_code == 0
  28         assert "requests" not in p.pipfile["dev-packages"]
  29 
  30         c = p.pipenv("run python -m requests.help")
  31         assert c.return_code > 0

  95 @pytest.mark.files
  96 @pytest.mark.install
  97 @pytest.mark.uninstall
  98 def test_uninstall_all_local_files(PipenvInstance, testsroot):
  99     file_name = "tablib-0.12.1.tar.gz"
 100     # Not sure where travis/appveyor run tests from
 101     source_path = os.path.abspath(os.path.join(testsroot, "pypi", "tablib", file_name))
 102 
 103     with PipenvInstance(chdir=True) as p:
 104         shutil.copy(source_path, os.path.join(p.path, file_name))
 105         os.mkdir(os.path.join(p.path, "tablib"))
 106         c = p.pipenv("install {}".format(file_name))
 107         assert c.return_code == 0
 108         c = p.pipenv("uninstall --all")
 109         assert c.return_code == 0
 110         assert "tablib" in c.out
 111         # Uninstall --all is not supposed to remove things from the pipfile
 112         # Note that it didn't before, but that instead local filenames showed as hashes
 113         assert "tablib" in p.pipfile["packages"]

 151 @pytest.mark.uninstall
 152 def test_normalize_name_uninstall(PipenvInstance):
 153     with PipenvInstance() as p:
 154         with open(p.pipfile_path, "w") as f:
 155             contents = """
 156 # Pre comment
 157 [packages]
 158 Requests = "*"
 159 python_DateUtil = "*"   # Inline comment
 160 """
 161             f.write(contents)
 162 
 163         c = p.pipenv("install")
 164         assert c.return_code == 0
 165 
 166         c = p.pipenv("uninstall python_dateutil")
 167         assert "Requests" in p.pipfile["packages"]
 168         assert "python_DateUtil" not in p.pipfile["packages"]
 169         with open(p.pipfile_path) as f:
 170             contents = f.read()
 171             assert "# Pre comment" in contents
 172             assert "# Inline comment" in contents
