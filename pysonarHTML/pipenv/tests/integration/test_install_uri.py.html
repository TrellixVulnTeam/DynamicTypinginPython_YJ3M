 238 @pytest.mark.vcs
 239 @pytest.mark.urls
 240 @pytest.mark.install
 241 @pytest.mark.needs_internet
 242 @pytest.mark.skip_py27_win
 243 @pytest.mark.skip_py38
 244 def test_vcs_entry_supersedes_non_vcs(PipenvInstance):
 245     """See issue #2181 -- non-editable VCS dep was specified, but not showing up
 246     in the lockfile -- due to not running pip install before locking and not locking
 247     the resolution graph of non-editable vcs dependencies.
 248     """
 249     with PipenvInstance(chdir=True) as p:
 250         # pyinstaller_path = p._pipfile.get_fixture_path("git/pyinstaller")
 251         pyinstaller_uri = "https://github.com/pyinstaller/pyinstaller.git"
 252         with open(p.pipfile_path, "w") as f:
 253             f.write(
 254                 """
 255 [[source]]
 256 url = "https://pypi.org/simple"
 257 verify_ssl = true
 258 name = "pypi"
 259 
 260 [packages]
 261 PyUpdater = "*"
 262 PyInstaller = {{ref = "v3.6", git = "{0}"}}
 263             """.format(pyinstaller_uri).strip()
 264             )
 265         c = p.pipenv("install")
 266         assert c.return_code == 0
 267         installed_packages = ["PyUpdater", "PyInstaller"]
 268         assert all([k in p.pipfile["packages"] for k in installed_packages])
 269         assert all([k.lower() in p.lockfile["default"] for k in installed_packages])
 270         assert all([k in p.lockfile["default"]["pyinstaller"] for k in ["ref", "git"]]), str(p.lockfile["default"])
 271         assert p.lockfile["default"]["pyinstaller"].get("ref") is not None
 272         assert (
 273             p.lockfile["default"]["pyinstaller"]["git"]
 274             == pyinstaller_uri
 275         )
