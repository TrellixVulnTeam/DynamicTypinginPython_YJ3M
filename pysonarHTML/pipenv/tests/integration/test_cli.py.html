 100 @pytest.mark.cli
 101 def test_pipenv_graph_reverse(PipenvInstance):
 102     with PipenvInstance() as p:
 103         c = p.pipenv('install tablib==0.13.0')
 104         assert c.ok
 105         c = p.pipenv('graph --reverse')
 106         assert c.ok
 107         output = c.out
 108 
 109         c = p.pipenv('graph --reverse --json')
 110         assert c.return_code == 1
 111         assert 'Warning: Using both --reverse and --json together is not supported.' in c.err
 112 
 113         requests_dependency = [
 114             ('backports.csv', 'backports.csv'),
 115             ('odfpy', 'odfpy'),
 116             ('openpyxl', 'openpyxl>=2.4.0'),
 117             ('pyyaml', 'pyyaml'),
 118             ('xlrd', 'xlrd'),
 119             ('xlwt', 'xlwt'),
 120         ]
 121 
 122         for dep_name, dep_constraint in requests_dependency:
 123             pat = r'^[ -]*{}==[\d.]+'.format(dep_name)
 124             dep_match = re.search(pat, output, flags=re.MULTILINE)
 125             assert dep_match is not None, '{} not found in {}'.format(pat, output)
 126 
 127             # openpyxl should be indented
 128             if dep_name == 'openpyxl':
 129                 openpyxl_dep = re.search(r'^openpyxl', output, flags=re.MULTILINE)
 130                 assert openpyxl_dep is None, 'openpyxl should not appear at begining of lines in {}'.format(output)
 131 
 132                 assert '  - openpyxl==2.5.4 [requires: et-xmlfile]' in output
 133             else:
 134                 dep_match = re.search(r'^[ -]*{}==[\d.]+$'.format(dep_name), output, flags=re.MULTILINE)
 135                 assert dep_match is not None, '{} not found at beginning of line in {}'.format(dep_name, output)
 136 
 137             dep_requests_match = re.search(r'^ +- tablib==0.13.0 \[requires: {}\]$'.format(dep_constraint), output, flags=re.MULTILINE)
 138             assert dep_requests_match is not None, 'constraint {} not found in {}'.format(dep_constraint, output)
 139             assert dep_requests_match.start() > dep_match.start()

 213 @pytest.mark.cli
 214 def test_install_parse_error(PipenvInstance):
 215     with PipenvInstance() as p:
 216 
 217         # Make sure unparseable packages don't wind up in the pipfile
 218         # Escape $ for shell input
 219         with open(p.pipfile_path, 'w') as f:
 220             contents = """
 221 [packages]
 222 
 223 [dev-packages]
 224             """.strip()
 225             f.write(contents)
 226         c = p.pipenv('install requests u/\\/p@r\\$34b13+pkg')
 227         assert c.return_code != 0
 228         assert 'u/\\/p@r$34b13+pkg' not in p.pipfile['packages']
