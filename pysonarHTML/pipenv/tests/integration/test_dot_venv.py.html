  11 @pytest.mark.dotvenv
  12 def test_venv_in_project(PipenvInstance):
  13     with temp_environ():
  14         os.environ['PIPENV_VENV_IN_PROJECT'] = '1'
  15         with PipenvInstance() as p:
  16             c = p.pipenv('install requests')
  17             assert c.return_code == 0
  18             assert normalize_drive(p.path) in p.pipenv('--venv').out

  28             assert normalize_drive(p.path) in p.pipenv('--venv').out

  45 @pytest.mark.dotvenv
  46 @pytest.mark.parametrize('venv_name', ('test-venv', os.path.join('foo', 'test-venv')))
  47 def test_venv_file(venv_name, PipenvInstance):
  48     """Tests virtualenv creation when a .venv file exists at the project root
  49     and contains a venv name.
  50     """
  51     with PipenvInstance(chdir=True) as p:
  52         file_path = os.path.join(p.path, '.venv')
  53         with open(file_path, 'w') as f:
  54             f.write(venv_name)
  55 
  56         with temp_environ(), TemporaryDirectory(
  57             prefix='pipenv-', suffix='temp_workon_home'
  58         ) as workon_home:
  59             os.environ['WORKON_HOME'] = workon_home.name
  60             if 'PIPENV_VENV_IN_PROJECT' in os.environ:
  61                 del os.environ['PIPENV_VENV_IN_PROJECT']
  62 
  63             c = p.pipenv('install')
  64             assert c.return_code == 0
  65 
  66             c = p.pipenv('--venv')
  67             assert c.return_code == 0
  68             venv_loc = Path(c.out.strip()).absolute()
  69             assert venv_loc.exists()
  70             assert venv_loc.joinpath('.project').exists()
  71             venv_path = normalize_drive(venv_loc.as_posix())
  72             if os.path.sep in venv_name:
  73                 venv_expected_path = Path(p.path).joinpath(venv_name).absolute().as_posix()
  74             else:
  75                 venv_expected_path = Path(workon_home.name).joinpath(venv_name).absolute().as_posix()
  76             assert venv_path == normalize_drive(venv_expected_path)
