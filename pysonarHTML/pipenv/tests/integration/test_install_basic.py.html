 352     """Ensure environment variables are not expanded in lock file.

 395 def test_install_venv_project_directory(PipenvInstance):
 396     """Test the project functionality during virtualenv creation.
 397     """
 398     with PipenvInstance(chdir=True) as p:
 399         with temp_environ(), TemporaryDirectory(
 400             prefix="pipenv-", suffix="temp_workon_home"
 401         ) as workon_home:
 402             os.environ["WORKON_HOME"] = workon_home.name
 403             if "PIPENV_VENV_IN_PROJECT" in os.environ:
 404                 del os.environ["PIPENV_VENV_IN_PROJECT"]
 405

 407             assert c.return_code == 0
 408 
 409             venv_loc = None
 410             for line in c.err.splitlines():
 411                 if line.startswith("Virtualenv location:"):
 412                     venv_loc = Path(line.split(":", 1)[-1].strip())
 413             assert venv_loc is not None
 414             assert venv_loc.joinpath(".project").exists()
