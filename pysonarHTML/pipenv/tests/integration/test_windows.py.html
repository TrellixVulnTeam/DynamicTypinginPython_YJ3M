  14 @pytest.mark.project
  15 def test_case_changes_windows(PipenvInstance):
  16     """Test project matching for case changes on Windows.
  17     """
  18     with PipenvInstance(chdir=True) as p:
  19         c = p.pipenv('install pytz')
  20         assert c.return_code == 0
  21 
  22         # Canonical venv location.
  23         c = p.pipenv('--venv')
  24         assert c.return_code == 0
  25         virtualenv_location = c.out.strip()
  26 
  27         # Dance around to change the casing of the project directory.
  28         target = p.path.upper()
  29         if target == p.path:
  30             target = p.path.lower()
  31         os.chdir('..')
  32         os.chdir(target)
  33         assert os.path.abspath(os.curdir) != p.path
  34 
  35         # Ensure the incorrectly-cased project can find the correct venv.
  36         c = p.pipenv('--venv')
  37         assert c.return_code == 0
  38         assert c.out.strip().lower() == virtualenv_location.lower()

  73 @pytest.mark.cli
  74 def test_pipenv_clean_windows(PipenvInstance):
  75     with PipenvInstance(chdir=True) as p:
  76         c = p.pipenv('install requests')
  77         assert c.return_code == 0
  78         c = p.pipenv('run pip install click')
  79         assert c.return_code == 0
  80 
  81         c = p.pipenv('clean --dry-run')
  82         assert c.return_code == 0
  83         assert 'click' in c.out.strip()
