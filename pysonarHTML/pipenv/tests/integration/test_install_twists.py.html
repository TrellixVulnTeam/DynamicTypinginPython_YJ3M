  19 def test_local_extras_install(PipenvInstance):
  20     """Ensure -e .[extras] installs.
  21     """
  22     with PipenvInstance(chdir=True) as p:
  23         setup_py = os.path.join(p.path, "setup.py")
  24         with open(setup_py, "w") as fh:
  25             contents = """
  26 from setuptools import setup, find_packages
  27 setup(
  28     name='testpipenv',
  29     version='0.1',
  30     description='Pipenv Test Package',
  31     author='Pipenv Test',
  32     author_email='test@pipenv.package',
  33     license='MIT',
  34     packages=find_packages(),
  35     install_requires=[],
  36     extras_require={'dev': ['six']},
  37     zip_safe=False
  38 )
  39             """.strip()
  40             fh.write(contents)
  41         line = "-e .[dev]"
  42         with open(os.path.join(p.path, 'Pipfile'), 'w') as fh:
  43             fh.write("""
  44 [packages]
  45 testpipenv = {path = ".", editable = true, extras = ["dev"]}
  46 
  47 [dev-packages]
  48             """.strip())
  49         # project.write_toml({"packages": pipfile, "dev-packages": {}})
  50         c = p.pipenv("install")
  51         assert c.return_code == 0
  52         assert "testpipenv" in p.lockfile["default"]
  53         assert p.lockfile["default"]["testpipenv"]["extras"] == ["dev"]
  54         assert "six" in p.lockfile["default"]
  55         c = p.pipenv("uninstall --all")
  56         assert c.return_code == 0
  57         print("Current directory: {0}".format(os.getcwd()), file=sys.stderr)
  58         c = p.pipenv("install {0}".format(line))
  59         assert c.return_code == 0
  60         assert "testpipenv" in p.pipfile["packages"]
  61         assert p.pipfile["packages"]["testpipenv"]["path"] == "."
  62         assert p.pipfile["packages"]["testpipenv"]["extras"] == ["dev"]
  63         assert "six" in p.lockfile["default"]

 340 @pytest.mark.run
 341 @pytest.mark.files
 342 @pytest.mark.install
 343 def test_multiple_editable_packages_should_not_race(PipenvInstance, testsroot):
 344     """Test for a race condition that can occur when installing multiple 'editable' packages at
 345     once, and which causes some of them to not be importable.
 346 
 347     This issue had been fixed for VCS packages already, but not local 'editable' packages.
 348 
 349     So this test locally installs packages from tarballs that have already been committed in
 350     the local `pypi` dir to avoid using VCS packages.
 351     """
 352     pkgs = ["requests", "flask", "six", "jinja2"]
 353 
 354     pipfile_string = """
 355 [dev-packages]
 356 
 357 [packages]
 358 """
 359 
 360     with PipenvInstance(chdir=True) as p:
 361         for pkg_name in pkgs:
 362             source_path = p._pipfile.get_fixture_path("git/{0}/".format(pkg_name)).as_posix()
 363             c = delegator.run("git clone {0} ./{1}".format(source_path, pkg_name))
 364             assert c.return_code == 0
 365 
 366             pipfile_string += '"{0}" = {{path = "./{0}", editable = true}}\n'.format(pkg_name)
 367 
 368         with open(p.pipfile_path, 'w') as f:
 369             f.write(pipfile_string.strip())
 370 
 371         c = p.pipenv('install')
 372         assert c.return_code == 0
 373 
 374         c = p.pipenv('run python -c "import requests, flask, six, jinja2"')
 375         assert c.return_code == 0, c.err
