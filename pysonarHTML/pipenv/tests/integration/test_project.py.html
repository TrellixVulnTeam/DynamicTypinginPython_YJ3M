  19 def test_pipfile_envvar_expansion(PipenvInstance):
  20     with PipenvInstance(chdir=True) as p:
  21         with temp_environ():
  22             with open(p.pipfile_path, 'w') as f:
  23                 f.write("""
  24 [[source]]
  25 url = 'https://${TEST_HOST}/simple'
  26 verify_ssl = false
  27 name = "pypi"
  28 
  29 [packages]
  30 pytz = "*"
  31                 """.strip())

  35             assert 'localhost:5000' not in str(pipfile.load(p.pipfile_path))

  39 @pytest.mark.sources
  40 @pytest.mark.parametrize('lock_first', [True, False])
  41 def test_get_source(PipenvInstance, lock_first):
  42     with PipenvInstance(chdir=True) as p:
  43         with open(p.pipfile_path, 'w') as f:
  44             contents = """
  45 [[source]]
  46 url = "{0}"
  47 verify_ssl = false
  48 name = "testindex"
  49 
  50 [[source]]
  51 url = "https://pypi.org/simple"
  52 verify_ssl = "true"
  53 name = "pypi"
  54 
  55 [packages]
  56 pytz = "*"
  57 six = {{version = "*", index = "pypi"}}
  58 
  59 [dev-packages]
  60             """.format(os.environ['PIPENV_TEST_INDEX']).strip()
  61             f.write(contents)
  62 
  63         if lock_first:
  64             # force source to be cached
  65             c = p.pipenv('lock')
  66             assert c.return_code == 0
  67         project = Project()
  68         sources = [
  69             ['pypi', 'https://pypi.org/simple'],
  70             ['testindex', os.environ.get('PIPENV_TEST_INDEX')]
  71         ]
  72         for src in sources:
  73             name, url = src
  74             source = [s for s in project.pipfile_sources if s.get('name') == name]
  75             assert source
  76             source = source[0]
  77             assert source['name'] == name
  78             assert source['url'] == url
  79             assert sorted(source.items()) == sorted(project.get_source(name=name).items())
  80             assert sorted(source.items()) == sorted(project.get_source(url=url).items())
  81             assert sorted(source.items()) == sorted(project.find_source(name).items())
  82             assert sorted(source.items()) == sorted(project.find_source(url).items())

  85 @pytest.mark.install
  86 @pytest.mark.project
  87 @pytest.mark.parametrize('newlines', [u'\n', u'\r\n'])
  88 def test_maintain_file_line_endings(PipenvInstance, newlines):
  89     with PipenvInstance(chdir=True) as p:
  90         # Initial pipfile + lockfile generation
  91         c = p.pipenv('install pytz')
  92         assert c.return_code == 0
  93 
  94         # Rewrite each file with parameterized newlines
  95         for fn in [p.pipfile_path, p.lockfile_path]:
  96             with io.open(fn) as f:
  97                 contents = f.read()
  98                 written_newlines = f.newlines
  99 
 100             assert written_newlines == u'\n', '{0!r} != {1!r} for {2}'.format(
 101                 written_newlines, u'\n', fn,
 102             )
 103             # message because of  https://github.com/pytest-dev/pytest/issues/3443
 104             with io.open(fn, 'w', newline=newlines) as f:
 105                 f.write(contents)
 106 
 107         # Run pipenv install to programatically rewrite
 108         c = p.pipenv('install chardet')
 109         assert c.return_code == 0
 110 
 111         # Make sure we kept the right newlines
 112         for fn in [p.pipfile_path, p.lockfile_path]:
 113             with io.open(fn) as f:
 114                 f.read()    # Consumes the content to detect newlines.
 115                 actual_newlines = f.newlines
 116             assert actual_newlines == newlines, '{0!r} != {1!r} for {2}'.format(
 117                 actual_newlines, newlines, fn,
 118             )
 119             # message because of  https://github.com/pytest-dev/pytest/issues/3443

 156 def test_include_editable_packages(PipenvInstance, testsroot, pathlib_tmpdir):
 157     file_name = "tablib-0.12.1.tar.gz"
 158     package = pathlib_tmpdir.joinpath("tablib-0.12.1")

 168             for package in project.environment.get_installed_packages()

 172 @pytest.mark.project
 173 @pytest.mark.virtualenv
 174 def test_run_in_virtualenv_with_global_context(PipenvInstance, virtualenv):
 175     with PipenvInstance(chdir=True, venv_root=virtualenv.as_posix(), ignore_virtualenvs=False, venv_in_project=False) as p:
 176         c = delegator_run(
 177             "pipenv run pip freeze", cwd=os.path.abspath(p.path),
 178             env=os.environ.copy()
 179         )
 180         assert c.return_code == 0, (c.out, c.err)
 181         assert 'Creating a virtualenv' not in c.err, c.err
 182         project = Project()
 183         assert project.virtualenv_location == virtualenv.as_posix(), (
 184             project.virtualenv_location, virtualenv.as_posix()
 185         )
 186         c = delegator_run(
 187             "pipenv run pip install click", cwd=os.path.abspath(p.path),
 188             env=os.environ.copy()
 189         )
 190         assert c.return_code == 0, (c.out, c.err)
 191         assert "Courtesy Notice" in c.err, (c.out, c.err)
 192         c = delegator_run(
 193             "pipenv install six", cwd=os.path.abspath(p.path), env=os.environ.copy()
 194         )
 195         assert c.return_code == 0, (c.out, c.err)
 196         c = delegator_run(
 197             'pipenv run python -c "import click;print(click.__file__)"',
 198             cwd=os.path.abspath(p.path), env=os.environ.copy()
 199         )
 200         assert c.return_code == 0, (c.out, c.err)
 201         assert is_in_path(c.out.strip(), str(virtualenv)), (c.out.strip(), str(virtualenv))
 202         c = delegator_run(
 203             "pipenv clean --dry-run", cwd=os.path.abspath(p.path),
 204             env=os.environ.copy()
 205         )
 206         assert c.return_code == 0, (c.out, c.err)
 207         assert "click" in c.out, c.out

 210 @pytest.mark.project
 211 @pytest.mark.virtualenv
 212 def test_run_in_virtualenv(PipenvInstance):
 213     with PipenvInstance(chdir=True) as p:
 214         c = p.pipenv('run pip freeze')
 215         assert c.return_code == 0
 216         assert 'Creating a virtualenv' in c.err
 217         project = Project()
 218         c = p.pipenv("run pip install click")
 219         assert c.return_code == 0
 220         c = p.pipenv("install six")
 221         assert c.return_code == 0
 222         c = p.pipenv('run python -c "import click;print(click.__file__)"')
 223         assert c.return_code == 0
 224         assert normalize_path(c.out.strip()).startswith(
 225             normalize_path(str(project.virtualenv_location))
 226         )
 227         c = p.pipenv("clean --dry-run")
 228         assert c.return_code == 0
 229         assert "click" in c.out
