  28 @pytest.mark.lock
  29 @pytest.mark.deploy
  30 def test_deploy_works(PipenvInstance):
  31 
  32     with PipenvInstance(chdir=True) as p:
  33         with open(p.pipfile_path, 'w') as f:
  34             contents = """
  35 [packages]
  36 requests = "==2.19.1"
  37 flask = "==1.1.2"
  38 
  39 [dev-packages]
  40 pytest = "==4.6.9"
  41             """.strip()
  42             f.write(contents)
  43         c = p.pipenv('install --verbose')
  44         if c.return_code != 0:
  45             assert c.err == '' or c.err is None
  46             assert c.out == ''
  47         assert c.return_code == 0
  48         c = p.pipenv('lock')
  49         assert c.return_code == 0
  50         with open(p.pipfile_path, 'w') as f:
  51             contents = """
  52 [packages]
  53 requests = "==2.19.1"
  54             """.strip()
  55             f.write(contents)
  56 
  57         c = p.pipenv('install --deploy')
  58         assert c.return_code > 0

  61 @pytest.mark.update
  62 @pytest.mark.lock
  63 def test_update_locks(PipenvInstance):
  64     with PipenvInstance() as p:
  65         c = p.pipenv('install jdcal==1.3')
  66         assert c.return_code == 0
  67         assert p.lockfile['default']['jdcal']['version'] == '==1.3'
  68         with open(p.pipfile_path, 'r') as fh:
  69             pipfile_contents = fh.read()
  70         assert '==1.3' in pipfile_contents
  71         pipfile_contents = pipfile_contents.replace('==1.3', '*')
  72         with open(p.pipfile_path, 'w') as fh:
  73             fh.write(pipfile_contents)
  74         c = p.pipenv('update jdcal')
  75         assert c.return_code == 0
  76         assert p.lockfile['default']['jdcal']['version'] == '==1.4'
  77         c = p.pipenv('run pip freeze')
  78         assert c.return_code == 0
  79         lines = c.out.splitlines()
  80         assert 'jdcal==1.4' in [l.strip() for l in lines]
