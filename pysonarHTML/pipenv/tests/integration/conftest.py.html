  64 def check_github_ssh():
  65     res = False
  66     try:
  67         # `ssh -T git@github.com` will return successfully with return_code==1
  68         # and message 'Hi <username>! You've successfully authenticated, but
  69         # GitHub does not provide shell access.' if ssh keys are available and
  70         # registered with GitHub. Otherwise, the command will fail with
  71         # return_code=255 and say 'Permission denied (publickey).'
  72         c = delegator.run('ssh -o StrictHostKeyChecking=accept-new -o CheckHostIP=no -T git@github.com', timeout=30)
  73         res = True if c.return_code == 1 else False
  74     except KeyboardInterrupt:
  75         warnings.warn(
  76             "KeyboardInterrupt while checking GitHub ssh access", RuntimeWarning
  77         )
  78     except Exception:
  79         pass
  80     global HAS_WARNED_GITHUB
  81     if not res and not HAS_WARNED_GITHUB:
  82         warnings.warn(
  83             'Cannot connect to GitHub via SSH', RuntimeWarning
  84         )
  85         warnings.warn(
  86             'Will skip tests requiring SSH access to GitHub', RuntimeWarning
  87         )
  88         HAS_WARNED_GITHUB = True
  89     return res

 228 class _Pipfile(object):
 229     def __init__(self, path):
 230         self.path = path
 231         if self.path.exists():
 232             self.loads()
 233         else:
 234             self.document = tomlkit.document()
 235         self.document["source"] = self.document.get("source", tomlkit.aot())
 236         self.document["requires"] = self.document.get("requires", tomlkit.table())
 237         self.document["packages"] = self.document.get("packages", tomlkit.table())
 238         self.document["dev_packages"] = self.document.get("dev_packages", tomlkit.table())
 239         super(_Pipfile, self).__init__()
 240 
 241     def install(self, package, value, dev=False):
 242         section = "packages" if not dev else "dev_packages"
 243         if isinstance(value, dict):
 244             table = tomlkit.inline_table()
 245             table.update(value)
 246             self.document[section][package] = table
 247         else:
 248             self.document[section][package] = value
 249         self.write()
 250 
 251     def remove(self, package, dev=False):
 252         section = "packages" if not dev else "dev_packages"
 253         if not dev and package not in self.document[section]:
 254             if package in self.document["dev_packages"]:
 255                 section = "dev_packages"
 256         del self.document[section][package]
 257         self.write()
 258 
 259     def add(self, package, value, dev=False):
 260         self.install(package, value, dev=dev)
 261 
 262     def update(self, package, value, dev=False):
 263         self.install(package, value, dev=dev)
 264 
 265     def loads(self):
 266         self.document = tomlkit.loads(self.path.read_text())
 267 
 268     def dumps(self):
 269         source_table = tomlkit.table()
 270         pypi_url = os.environ.get("PIPENV_PYPI_URL", "https://pypi.org/simple")
 271         source_table["url"] = os.environ.get("PIPENV_TEST_INDEX", pypi_url)
 272         source_table["verify_ssl"] = False
 273         source_table["name"] = "pipenv_test_index"
 274         self.document["source"].append(source_table)
 275         return tomlkit.dumps(self.document)

 277     def write(self):
 278         self.path.write_text(self.dumps())

 386     def pipenv(self, cmd, block=True):
 387         if self.pipfile_path and os.path.isfile(self.pipfile_path):
 388             os.environ['PIPENV_PIPFILE'] = fs_str(self.pipfile_path)
 389         # a bit of a hack to make sure the virtualenv is created
 390 
 391         with TemporaryDirectory(prefix='pipenv-', suffix='-cache') as tempdir:
 392             os.environ['PIPENV_CACHE_DIR'] = fs_str(tempdir.name)
 393             c = delegator.run(
 394                 'pipenv {0}'.format(cmd), block=block,
 395                 cwd=os.path.abspath(self.path), env=os.environ.copy()
 396             )

 431 def _rmtree_func(path, ignore_errors=True, onerror=None):
 432     directory = fs_encode(path)
 433     shutil_rmtree = _rmtree
 434     if onerror is None:
 435         onerror = handle_remove_readonly
 436     try:
 437         shutil_rmtree(directory, ignore_errors=ignore_errors, onerror=onerror)
 438     except (IOError, OSError, FileNotFoundError, PermissionError) as exc:
 439         # Ignore removal failures where the file doesn't exist
 440         if exc.errno != errno.ENOENT:
 441             raise

 499 class VirtualEnv(object):
 500     def __init__(self, name="venv", base_dir=None):
 501         if base_dir is None:
 502             base_dir = Path(_create_tracked_dir())
 503         self.base_dir = base_dir
 504         self.name = name
 505         self.path = base_dir / name
 506 
 507     def __enter__(self):
 508         self._old_environ = os.environ.copy()
 509         self.create()
 510         return self.activate()
 511 
 512     def __exit__(self, *args, **kwargs):
 513         os.environ = self._old_environ
 514 
 515     def create(self):
 516         python = Path(sys.executable).absolute().as_posix()
 517         cmd = [
 518             python, "-m", "virtualenv", self.path.absolute().as_posix()
 519         ]
 520         c = run(
 521             cmd, verbose=False, return_object=True, write_to_stdout=False,
 522             combine_stderr=False, block=True, nospin=True,
 523         )
 524         # cmd = "{0} -m virtualenv {1}".format(python, self.path.as_posix())
 525         # c = delegator.run(cmd, block=True)
 526         assert c.returncode == 0
 527 
 528     def activate(self):
 529         script_path = "Scripts" if os.name == "nt" else "bin"
 530         activate_this = self.path / script_path / "activate_this.py"
 531         if activate_this.exists():
 532             with open(str(activate_this)) as f:
 533                 code = compile(f.read(), str(activate_this), "exec")
 534                 exec(code, dict(__file__=str(activate_this)))
 535             os.environ["VIRTUAL_ENV"] = str(self.path)
 536             try:
 537                 return self.path.absolute().resolve()
 538             except OSError:
 539                 return self.path.absolute()
 540         else:
 541             raise VirtualenvActivationException("Can't find the activate_this.py script.")
