  30 class Handler(WSGIRequestHandler):
  31 
  32     def handle(self):
  33         """Handle a single HTTP request"""
  34 
  35         self.raw_requestline = self.rfile.readline()
  36         if not self.parse_request():  # An error code has been sent, just exit
  37             return
  38 
  39         handler = ServerHandler(
  40             self.rfile, self.wfile, self.get_stderr(), self.get_environ()
  41         )
  42         handler.request_handler = self      # backpointer for logging
  43         handler.run(self.server.get_app())
  44 
  45     def get_environ(self):
  46         """
  47         wsgiref simple server adds content-type text/plain to everything, this
  48         removes it if it's not actually in the headers.
  49         """
  50         # Note: Can't use super since this is an oldstyle class in python 2.x
  51         environ = WSGIRequestHandler.get_environ(self).copy()
  52         if self.headers.get('content-type') is None:
  53             del environ['CONTENT_TYPE']
  54         return environ

  57 class SecureWSGIServer(WSGIServer):
  58 
  59     def finish_request(self, request, client_address):
  60         """
  61         Negotiates SSL and then mimics BaseServer behavior.
  62         """
  63         request.settimeout(1.0)
  64         try:
  65             ssock = ssl.wrap_socket(
  66                 request,
  67                 keyfile=os.path.join(CERT_DIR, 'key.pem'),
  68                 certfile=os.path.join(CERT_DIR, 'cert.pem'),
  69                 server_side=True,
  70                 suppress_ragged_eofs=False,
  71             )
  72             self.RequestHandlerClass(ssock, client_address, self)
  73         except Exception as e:
  74             print("pytest-httpbin server hit an exception serving request: %s" % e)
  75             print("attempting to ignore so the rest of the tests can run")
  76         # WSGIRequestHandler seems to close the socket for us.
  77         # Thanks, WSGIRequestHandler!!

  80 class Server(object):
  81     """
  82     HTTP server running a WSGI application in its own thread.
  83     """
  84 
  85     port_envvar = 'HTTPBIN_HTTP_PORT'
  86 
  87     def __init__(self, host='127.0.0.1', port=0, application=None, **kwargs):
  88         self.app = application
  89         if self.port_envvar in os.environ:
  90             port = int(os.environ[self.port_envvar])
  91         self._server = make_server(
  92             host,
  93             port,
  94             self.app,
  95             handler_class=Handler,
  96             **kwargs
  97         )
  98         self.host = self._server.server_address[0]
  99         self.port = self._server.server_address[1]
 100         self.protocol = 'http'
 101 
 102         self._thread = threading.Thread(
 103             name=self.__class__,
 104             target=self._server.serve_forever,
 105         )
 106 
 107     def __del__(self):
 108         if hasattr(self, '_server'):
 109             self.stop()
 110 
 111     def start(self):
 112         self._thread.start()
 113 
 114     def __add__(self, other):
 115         return self.url + other
 116 
 117     def stop(self):
 118         self._server.shutdown()
 119 
 120     @property
 121     def url(self):
 122         return '{0}://{1}:{2}'.format(self.protocol, self.host, self.port)
 123 
 124     def join(self, url, allow_fragments=True):
 125         return urljoin(self.url, url, allow_fragments=allow_fragments)
