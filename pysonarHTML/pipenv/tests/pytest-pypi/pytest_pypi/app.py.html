  45 class Package(object):
  46     """Package represents a collection of releases from one or more directories"""
  47 
  48     def __init__(self, name):
  49         super(Package, self).__init__()
  50         self.name = name
  51         self.releases = {}
  52         self._package_dirs = set()
  53 
  54     @property
  55     def json(self):
  56         for path, _ in self._package_dirs:
  57             try:
  58                 with open(os.path.join(path, 'api.json')) as f:
  59                     return json.load(f)
  60             except FileNotFoundError:
  61                 r = session.get('https://pypi.org/pypi/{0}/json'.format(self.name))
  62                 response = r.json()
  63                 releases = response["releases"]
  64                 files = {
  65                     pkg for pkg_dir in self._package_dirs
  66                     for pkg in os.listdir(pkg_dir.path)
  67                 }
  68                 for release in list(releases.keys()):
  69                     values = (
  70                         r for r in releases[release] if r["filename"] in files
  71                     )
  72                     values = list(values)
  73                     if values:
  74                         releases[release] = values
  75                     else:
  76                         del releases[release]
  77                 response["releases"] = releases
  78                 with io.open(os.path.join(path, "api.json"), "w") as fh:
  79                     json.dump(response, fh, indent=4)
  80                 return response
  81 
  82     def __repr__(self):
  83         return "<Package name={0!r} releases={1!r}".format(self.name, len(self.releases))

  85     def add_release(self, path_to_binary):
  86         path_to_binary = os.path.abspath(path_to_binary)
  87         path, release = os.path.split(path_to_binary)
  88         requires_python = ""
  89         if path_to_binary.endswith(".whl"):
  90             pkg = distlib.wheel.Wheel(path_to_binary)
  91             md_dict = pkg.metadata.todict()
  92             requires_python = md_dict.get("requires_python", "")
  93             if requires_python.count(".") > 1:
  94                 requires_python, _, _ = requires_python.rpartition(".")
  95         self.releases[release] = ReleaseTuple(path_to_binary, requires_python)
  96         self._package_dirs.add(ReleaseTuple(path, requires_python))

 118 def prepare_fixtures(path):
 119     path = os.path.abspath(path)
 120     if not (os.path.exists(path) and os.path.isdir(path)):
 121         raise ValueError("{} is not a directory!".format(path))
 122     for root, dirs, files in os.walk(path):
 123         package_name, _, _ = os.path.relpath(root, start=path).partition(os.path.sep)
 124         if package_name not in ARTIFACTS:
 125             ARTIFACTS[package_name] = Artifact(package_name)
 126         for file in files:
 127             file_path = os.path.join(root, file)
 128             rel_path = os.path.relpath(file_path, start=path)
 129             _, _, subpkg = rel_path.partition(os.path.sep)
 130             subpkg, _, _ = subpkg.partition(os.path.sep)
 131             pkg, ext = os.path.splitext(subpkg)
 132             if not (is_tarfile(file_path) or is_zipfile(file_path) or ext == ".git"):
 133                 continue
 134             if subpkg not in ARTIFACTS[package_name].files:
 135                 ARTIFACTS[package_name].add_file(os.path.join(root, file))
 136             ARTIFACTS[package_name].add_file(os.path.join(root, file))

 139 def prepare_packages(path):
 140     """Add packages in path to the registry."""
 141     path = os.path.abspath(path)
 142     if not (os.path.exists(path) and os.path.isdir(path)):
 143         raise ValueError("{} is not a directory!".format(path))
 144     for root, dirs, files in os.walk(path):
 145         if all([setup_file in list(files) for setup_file in ("setup.py", "setup.cfg")]):
 146             continue
 147         for file in files:
 148             if not file.startswith('.') and not file.endswith('.json'):
 149                 package_name = os.path.basename(root)
 150                 if package_name and package_name == "fixtures":
 151                     prepare_fixtures(root)
 152                     continue
 153                 package_name = package_name.replace("_", "-")
 154                 if package_name not in packages:
 155                     packages[package_name] = Package(package_name)
 156 
 157                 packages[package_name].add_release(os.path.join(root, file))
 158     remaining = get_pypi_package_names() - set(list(packages.keys()))
 159     for pypi_pkg in remaining:
 160         packages[pypi_pkg] = Package(pypi_pkg)
