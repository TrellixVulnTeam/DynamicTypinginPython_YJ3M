  22 @pytest.mark.core
  23 def test_load_dot_env_from_environment_variable_location(monkeypatch, capsys):
  24     with temp_environ(), monkeypatch.context() as m, TemporaryDirectory(prefix='pipenv-', suffix='') as tempdir:
  25         if os.name == "nt":
  26             import click
  27             is_console = False
  28             m.setattr(click._winconsole, "_is_console", lambda x: is_console)
  29         dotenv_path = os.path.join(tempdir.name, 'test.env')
  30         key, val = 'SOME_KEY', 'some_value'
  31         with open(dotenv_path, 'w') as f:
  32             f.write('{}={}'.format(key, val))
  33 
  34         m.setenv("PIPENV_DOTENV_LOCATION", str(dotenv_path))
  35         m.setattr("pipenv.environments.PIPENV_DOTENV_LOCATION", str(dotenv_path))
  36         load_dot_env()
  37         assert os.environ[key] == val
