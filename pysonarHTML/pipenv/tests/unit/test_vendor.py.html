  18     def test_inject_environment_variables(self):
  19         os.environ['PYTEST_PIPFILE_TEST'] = "XYZ"
  20         p = PipfileParser()
  21 
  22         parsed_dict = p.inject_environment_variables({
  23             "a_string": "https://$PYTEST_PIPFILE_TEST@something.com",
  24             "another_string": "https://${PYTEST_PIPFILE_TEST}@something.com",
  25             "nested": {
  26                 "a_string": "https://$PYTEST_PIPFILE_TEST@something.com",
  27                 "another_string": "${PYTEST_PIPFILE_TEST}",
  28             },
  29             "list": [
  30                 {
  31                     "a_string": "https://$PYTEST_PIPFILE_TEST@something.com",
  32                     "another_string": "${PYTEST_PIPFILE_TEST}"
  33                 },
  34                 {},
  35             ],
  36             "bool": True,
  37             "none": None,
  38         })
  39 
  40         assert parsed_dict["a_string"] == "https://XYZ@something.com"
  41         assert parsed_dict["another_string"] == "https://XYZ@something.com"
  42         assert parsed_dict["nested"]["another_string"] == "XYZ"
  43         assert parsed_dict["list"][0]["a_string"] == "https://XYZ@something.com"
  44         assert parsed_dict["list"][1] == {}
  45         assert parsed_dict["bool"] is True
  46         assert parsed_dict["none"] is None
