   8 @pytest.mark.environments
   9 @pytest.mark.parametrize(
  10     "arg, prefix, use_negation",
  11     list(itertools.product(("ENABLE_SOMETHING",), ("FAKEPREFIX", None), (True, False))),
  12 )
  13 def test_get_from_env(arg, prefix, use_negation):
  14     negated_arg = "NO_{0}".format(arg)
  15     positive_var = arg
  16     negative_var = negated_arg
  17     if prefix:
  18         negative_var = "{0}_{1}".format(prefix, negative_var)
  19         positive_var = "{0}_{1}".format(prefix, positive_var)
  20     # set the positive first
  21     for var_to_set, opposite_var in ((arg, negated_arg), (negated_arg, arg)):
  22         os.environ.pop(var_to_set, None)
  23         os.environ.pop(opposite_var, None)
  24         with temp_environ():
  25             is_positive = var_to_set == arg
  26             is_negative = not is_positive
  27             envvar = positive_var if is_positive else negative_var
  28             os.environ[envvar] = "true"
  29             main_expected_value = True if is_positive else None
  30             if use_negation and not is_positive:
  31                 main_expected_value = False
  32             # use negation means if the normal variable isnt set we will check
  33             # for the negated version
  34             negative_expected_value = (
  35                 True if is_negative else None
  36             )
  37             if is_positive:
  38                 assert (
  39                     environments.get_from_env(
  40                         var_to_set, prefix, check_for_negation=use_negation
  41                     )
  42                     is main_expected_value
  43                 )
  44                 assert (
  45                     environments.get_from_env(
  46                         opposite_var, prefix, check_for_negation=use_negation
  47                     )
  48                     is negative_expected_value
  49                 )
  50             else:
  51                 # var_to_set = negative version i.e. NO_xxxx
  52                 # opposite_var = positive_version i.e. XXXX
  53 
  54                 # get NO_BLAH -- expecting this to be True
  55                 assert (
  56                     environments.get_from_env(
  57                         var_to_set, prefix, check_for_negation=use_negation
  58                     )
  59                     is negative_expected_value
  60                 )
  61                 # get BLAH -- expecting False if checking for negation
  62                 # but otherwise should be None
  63                 assert (
  64                     environments.get_from_env(
  65                         opposite_var, prefix, check_for_negation=use_negation
  66                     )
  67                     is main_expected_value
  68                 )
