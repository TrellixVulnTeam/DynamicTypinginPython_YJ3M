  27 def read_requirements(fh, resolve=False):
  28     """
  29     Reads requirements from a file like object and (optionally) from referenced files.
  30     :param fh: file like object to read from
  31     :param resolve: boolean. resolves referenced files.
  32     :return: generator
  33     """
  34     is_temp_file = not hasattr(fh, 'name')
  35     for num, line in enumerate(iter_lines(fh)):
  36         line = line.strip()
  37         if not line:
  38             # skip empty lines
  39             continue
  40         if line.startswith('#') or \
  41             line.startswith('-i') or \
  42             line.startswith('--index-url') or \
  43             line.startswith('--extra-index-url') or \
  44             line.startswith('-f') or line.startswith('--find-links') or \
  45             line.startswith('--no-index') or line.startswith('--allow-external') or \
  46             line.startswith('--allow-unverified') or line.startswith('-Z') or \
  47             line.startswith('--always-unzip'):
  48             # skip unsupported lines
  49             continue
  50         elif line.startswith('-r') or line.startswith('--requirement'):
  51             # got a referenced file here, try to resolve the path
  52             # if this is a tempfile, skip
  53             if is_temp_file:
  54                 continue
  55             filename = line.strip("-r ").strip("--requirement").strip()
  56             # if there is a comment, remove it
  57             if " #" in filename:
  58                 filename = filename.split(" #")[0].strip()
  59             req_file_path = os.path.join(os.path.dirname(fh.name), filename)
  60             if resolve:
  61                 # recursively yield the resolved requirements
  62                 if os.path.exists(req_file_path):
  63                     with open(req_file_path) as _fh:
  64                         for req in read_requirements(_fh, resolve=True):
  65                             yield req
  66             else:
  67                 yield RequirementFile(path=req_file_path)
  68         else:
  69             try:
  70                 parseable_line = line
  71                 # multiline requirements are not parseable
  72                 if "\\" in line:
  73                     parseable_line = line.replace("\\", "")
  74                     for next_line in iter_lines(fh, num + 1):
  75                         parseable_line += next_line.strip().replace("\\", "")
  76                         line += "\n" + next_line
  77                         if "\\" in next_line:
  78                             continue
  79                         break
  80                 req, = parse_line(parseable_line)
  81                 if len(req.specifier._specs) == 1 and \
  82                         next(iter(req.specifier._specs))._spec[0] == "==":
  83                     yield Package(key=req.name, version=next(iter(req.specifier._specs))._spec[1])
  84                 else:
  85                     try:
  86                         fname = fh.name
  87                     except AttributeError:
  88                         fname = line
  89 
  90                     click.secho(
  91                         "Warning: unpinned requirement '{req}' found in {fname}, "
  92                         "unable to check.".format(req=req.name,
  93                                                   fname=fname),
  94                         fg="yellow",
  95                         file=sys.stderr
  96                     )
  97             except ValueError:
  98                 continue
