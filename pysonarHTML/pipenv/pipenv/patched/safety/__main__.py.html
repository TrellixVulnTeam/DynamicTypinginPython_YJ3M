  14 def get_site_packages():
  15     prefixes = {sys.prefix, sysconfig.get_config_var('prefix')}
  16     try:
  17         prefixes.add(sys.real_prefix)
  18     except AttributeError:
  19         pass
  20     form = sysconfig.get_path('purelib', expand=False)
  21     py_version_short = '{0[0]}.{0[1]}'.format(sys.version_info)
  22     return {
  23         form.format(base=prefix, py_version_short=py_version_short)
  24         for prefix in prefixes
  25     }

  28 def insert_before_site_packages(*paths):
  29     site_packages = get_site_packages()
  30     index = None
  31     for i, path in enumerate(sys.path):
  32         if path in site_packages:
  33             index = i
  34             break
  35     if index is None:
  36         sys.path += list(paths)
  37     else:
  38         sys.path = sys.path[:index] + list(paths) + sys.path[index:]
  39

  41 def insert_pipenv_dirs():
  42     insert_before_site_packages(os.path.dirname(PIPENV_DIR), PATCHED_DIR, VENDORED_DIR)
