  92     @staticmethod
  93     def render(vulns, full, checked_packages, used_db):
  94         db_format_str = '{: <' + str(51 - len(str(checked_packages))) + '}'
  95         status = "│ checked {packages} packages, using {db} │".format(
  96             packages=checked_packages,
  97             db=db_format_str.format(used_db),
  98             section=SheetReport.REPORT_SECTION
  99         )
 100         if vulns:
 101             table = []
 102             for n, vuln in enumerate(vulns):
 103                 table.append("│ {:26} │ {:9} │ {:24} │ {:8} │".format(
 104                     vuln.name[:26],
 105                     vuln.version[:9],
 106                     vuln.spec[:24],
 107                     vuln.vuln_id
 108                 ))
 109                 if full:
 110                     table.append(SheetReport.REPORT_SECTION)
 111 
 112                     descr = get_advisory(vuln)
 113 
 114                     for pn, paragraph in enumerate(descr.replace('\r', '').split('\n\n')):
 115                         if pn:
 116                             table.append("│ {:76} │".format(''))
 117                         for line in textwrap.wrap(paragraph, width=76):
 118                             try:
 119                                 table.append("│ {:76} │".format(line.encode('utf-8')))
 120                             except TypeError:
 121                                 table.append("│ {:76} │".format(line))
 122                     # append the REPORT_SECTION only if this isn't the last entry
 123                     if n + 1 < len(vulns):
 124                         table.append(SheetReport.REPORT_SECTION)
 125             return "\n".join(
 126                 [SheetReport.REPORT_BANNER, SheetReport.REPORT_HEADING, status, SheetReport.TABLE_HEADING,
 127                  "\n".join(table), SheetReport.REPORT_FOOTER]
 128             )
 129         else:
 130             content = "│ {:76} │".format("No known security vulnerabilities found.")
 131             return "\n".join(
 132                     [SheetReport.REPORT_BANNER, SheetReport.REPORT_HEADING, status, SheetReport.REPORT_SECTION,
 133                      content, SheetReport.REPORT_FOOTER]
 134                 )

 137 class BasicReport(object):
 138     """Basic report, intented to be used for terminals with < 80 columns"""
 139 
 140     @staticmethod
 141     def render(vulns, full, checked_packages, used_db):
 142         table = [
 143             "safety report",
 144             "checked {packages} packages, using {db}".format(
 145                 packages=checked_packages,
 146                 db=used_db
 147             ),
 148             "---"
 149         ]
 150         if vulns:
 151 
 152             for vuln in vulns:
 153                 table.append("-> {}, installed {}, affected {}, id {}".format(
 154                     vuln.name,
 155                     vuln.version[:13],
 156                     vuln.spec[:27],
 157                     vuln.vuln_id
 158                 ))
 159                 if full:
 160                     table.append(get_advisory(vuln))
 161                     table.append("--")
 162         else:
 163             table.append("No known security vulnerabilities found.")
 164         return "\n".join(
 165             table
 166         )

 184 def get_used_db(key, db):
 185     key = key if key else os.environ.get("SAFETY_API_KEY", False)
 186     if key:
 187         return "pyup.io's DB"
 188     if db == '':
 189         return 'default DB'
 190     return "local DB"

 193 def report(vulns, full=False, json_report=False, bare_report=False, checked_packages=0, db=None, key=None):
 194     if bare_report:
 195         return BareReport.render(vulns, full=full)
 196     if json_report:
 197         return JsonReport.render(vulns, full=full)
 198     size = get_terminal_size()
 199     used_db = get_used_db(key=key, db=db)
 200     if size.columns >= 80:
 201         return SheetReport.render(vulns, full=full, checked_packages=checked_packages, used_db=used_db)
 202     return BasicReport.render(vulns, full=full, checked_packages=checked_packages, used_db=used_db)
