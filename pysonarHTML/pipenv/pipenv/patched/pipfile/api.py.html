  19 def format_full_version(info):
  20     version = '{0.major}.{0.minor}.{0.micro}'.format(info)
  21     kind = info.releaselevel
  22     if kind != 'final':
  23         version += kind[0] + str(info.serial)
  24     return version

  27 def walk_up(bottom):
  28     """mimic os.walk, but walk 'up' instead of down the directory tree.
  29     From: https://gist.github.com/zdavkeos/1098474
  30     """
  31 
  32     bottom = os.path.realpath(bottom)
  33 
  34     # get files in current dir
  35     try:
  36         names = os.listdir(bottom)
  37     except Exception:
  38         return
  39 
  40     dirs, nondirs = [], []
  41     for name in names:
  42         if os.path.isdir(os.path.join(bottom, name)):
  43             dirs.append(name)
  44         else:
  45             nondirs.append(name)
  46 
  47     yield bottom, dirs, nondirs
  48 
  49     new_path = os.path.realpath(os.path.join(bottom, '..'))
  50 
  51     # see if we are at the top
  52     if new_path == bottom:
  53         return
  54 
  55     for x in walk_up(new_path):
  56         yield x

  59 class PipfileParser(object):
  60     def __init__(self, filename='Pipfile'):
  61         self.filename = filename
  62         self.sources = []
  63         self.groups = {
  64             'default': [],
  65             'develop': []
  66         }
  67         self.group_stack = ['default']
  68         self.requirements = []
  69 
  70     def __repr__(self):
  71         return '<PipfileParser path={0!r}'.format(self.filename)
  72 
  73     def inject_environment_variables(self, d):
  74         """
  75         Recursively injects environment variables into TOML values
  76         """
  77 
  78         if not d:
  79             return d
  80         if isinstance(d, six.string_types):
  81             return os.path.expandvars(d)
  82         for k, v in d.items():
  83             if isinstance(v, six.string_types):
  84                 d[k] = os.path.expandvars(v)
  85             elif isinstance(v, dict):
  86                 d[k] = self.inject_environment_variables(v)
  87             elif isinstance(v, list):
  88                 d[k] = [self.inject_environment_variables(e) for e in v]
  89 
  90         return d
  91 
  92     def parse(self, inject_env=True):
  93         # Open the Pipfile.
  94         with open(self.filename) as f:
  95             content = f.read()
  96 
  97         # Load the default configuration.
  98         default_config = {
  99             u'source': [DEFAULT_SOURCE],
 100             u'packages': {},
 101             u'requires': {},
 102             u'dev-packages': {}
 103         }
 104 
 105         config = {}
 106         config.update(default_config)
 107 
 108         # Deserialize the TOML, and parse for Environment Variables
 109         parsed = toml.loads(content)
 110 
 111         if inject_env:
 112             injected_toml = self.inject_environment_variables(parsed)
 113 
 114             # Load the Pipfile's configuration.
 115             config.update(injected_toml)
 116         else:
 117             config.update(parsed)
 118 
 119         # Structure the data for output.
 120         data = {
 121             '_meta': {
 122                 'sources': config['source'],
 123                 'requires': config['requires']
 124             },
 125         }
 126 
 127         # TODO: Validate given data here.
 128         self.groups['default'] = config['packages']
 129         self.groups['develop'] = config['dev-packages']
 130 
 131         # Update the data structure with group information.
 132         data.update(self.groups)
 133         return data

 136 class Pipfile(object):
 137     def __init__(self, filename):
 138         super(Pipfile, self).__init__()
 139         self.filename = filename
 140         self.data = None
 141 
 142     @staticmethod
 143     def find(max_depth=3):
 144         """Returns the path of a Pipfile in parent directories."""
 145         i = 0
 146         for c, d, f in walk_up(os.getcwd()):
 147             i += 1
 148 
 149             if i < max_depth:
 150                 if 'Pipfile':
 151                     p = os.path.join(c, 'Pipfile')
 152                     if os.path.isfile(p):
 153                         return p
 154         raise RuntimeError('No Pipfile found!')
 155 
 156     @classmethod
 157     def load(klass, filename, inject_env=True):
 158         """Load a Pipfile from a given filename."""
 159         p = PipfileParser(filename=filename)
 160         pipfile = klass(filename=filename)
 161         pipfile.data = p.parse(inject_env=inject_env)
 162         return pipfile
 163 
 164     @property
 165     def hash(self):
 166         """Returns the SHA256 of the pipfile's data."""
 167         content = json.dumps(self.data, sort_keys=True, separators=(",", ":"))
 168         return hashlib.sha256(content.encode("utf8")).hexdigest()
 169 
 170     @property
 171     def contents(self):
 172         """Returns the contents of the pipfile."""
 173         with codecs.open(self.filename, 'r', 'utf-8') as f:
 174             return f.read()
 175 
 176     def lock(self):
 177         """Returns a JSON representation of the Pipfile."""
 178         data = self.data
 179         data['_meta']['hash'] = {"sha256": self.hash}
 180         data['_meta']['pipfile-spec'] = 6
 181         return json.dumps(data, indent=4, separators=(',', ': '))

 183     def assert_requirements(self):
 184         """"Asserts PEP 508 specifiers."""
 185 
 186         # Support for 508's implementation_version.
 187         if hasattr(sys, 'implementation'):
 188             implementation_version = format_full_version(sys.implementation.version)
 189         else:
 190             implementation_version = "0"
 191 
 192         # Default to cpython for 2.7.
 193         if hasattr(sys, 'implementation'):
 194             implementation_name = sys.implementation.name
 195         else:
 196             implementation_name = 'cpython'
 197 
 198         lookup = {
 199             'os_name': os.name,
 200             'sys_platform': sys.platform,
 201             'platform_machine': platform.machine(),
 202             'platform_python_implementation': platform.python_implementation(),
 203             'platform_release': platform.release(),
 204             'platform_system': platform.system(),
 205             'platform_version': platform.version(),
 206             'python_version': platform.python_version()[:3],
 207             'python_full_version': platform.python_version(),
 208             'implementation_name': implementation_name,
 209             'implementation_version': implementation_version
 210         }
 211 
 212         # Assert each specified requirement.
 213         for marker, specifier in self.data['_meta']['requires'].items():
 214 
 215             if marker in lookup:
 216                 try:
 217                     assert lookup[marker] == specifier
 218                 except AssertionError:
 219                     raise AssertionError('Specifier {!r} does not match {!r}.'.format(marker, specifier))

 222 def load(pipfile_path=None, inject_env=True):
 223     """Loads a pipfile from a given path.
 224     If none is provided, one will try to be found.
 225     """
 226 
 227     if pipfile_path is None:
 228         pipfile_path = Pipfile.find()
 229 
 230     return Pipfile.load(filename=pipfile_path, inject_env=inject_env)
