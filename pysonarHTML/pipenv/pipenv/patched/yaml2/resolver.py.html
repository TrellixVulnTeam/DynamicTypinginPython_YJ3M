  25     def add_implicit_resolver(cls, tag, regexp, first):
  26         if not 'yaml_implicit_resolvers' in cls.__dict__:
  27             implicit_resolvers = {}
  28             for key in cls.yaml_implicit_resolvers:
  29                 implicit_resolvers[key] = cls.yaml_implicit_resolvers[key][:]
  30             cls.yaml_implicit_resolvers = implicit_resolvers
  31         if first is None:
  32             first = [None]
  33         for ch in first:
  34             cls.yaml_implicit_resolvers.setdefault(ch, []).append((tag, regexp))

  37     def add_path_resolver(cls, tag, path, kind=None):
  38         # Note: `add_path_resolver` is experimental.  The API could be changed.
  39         # `new_path` is a pattern that is matched against the path from the
  40         # root to the node that is being considered.  `node_path` elements are
  41         # tuples `(node_check, index_check)`.  `node_check` is a node class:
  42         # `ScalarNode`, `SequenceNode`, `MappingNode` or `None`.  `None`
  43         # matches any kind of a node.  `index_check` could be `None`, a boolean
  44         # value, a string value, or a number.  `None` and `False` match against
  45         # any _value_ of sequence and mapping nodes.  `True` matches against
  46         # any _key_ of a mapping node.  A string `index_check` matches against
  47         # a mapping value that corresponds to a scalar key which content is
  48         # equal to the `index_check` value.  An integer `index_check` matches
  49         # against a sequence value with the index equal to `index_check`.
  50         if not 'yaml_path_resolvers' in cls.__dict__:
  51             cls.yaml_path_resolvers = cls.yaml_path_resolvers.copy()
  52         new_path = []
  53         for element in path:
  54             if isinstance(element, (list, tuple)):
  55                 if len(element) == 2:
  56                     node_check, index_check = element
  57                 elif len(element) == 1:
  58                     node_check = element[0]
  59                     index_check = True
  60                 else:
  61                     raise ResolverError("Invalid path element: %s" % element)
  62             else:
  63                 node_check = None
  64                 index_check = element
  65             if node_check is str:
  66                 node_check = ScalarNode
  67             elif node_check is list:
  68                 node_check = SequenceNode
  69             elif node_check is dict:
  70                 node_check = MappingNode
  71             elif node_check not in [ScalarNode, SequenceNode, MappingNode]  \
  72                     and not isinstance(node_check, basestring)  \
  73                     and node_check is not None:
  74                 raise ResolverError("Invalid node checker: %s" % node_check)
  75             if not isinstance(index_check, (basestring, int))   \
  76                     and index_check is not None:
  77                 raise ResolverError("Invalid index checker: %s" % index_check)
  78             new_path.append((node_check, index_check))
  79         if kind is str:
  80             kind = ScalarNode
  81         elif kind is list:
  82             kind = SequenceNode
  83         elif kind is dict:
  84             kind = MappingNode
  85         elif kind not in [ScalarNode, SequenceNode, MappingNode]    \
  86                 and kind is not None:
  87             raise ResolverError("Invalid node kind: %s" % kind)
  88         cls.yaml_path_resolvers[tuple(new_path), kind] = tag
  89     add_path_resolver = classmethod(add_path_resolver)

  91     def descend_resolver(self, current_node, current_index):
  92         if not self.yaml_path_resolvers:
  93             return
  94         exact_paths = {}
  95         prefix_paths = []
  96         if current_node:
  97             depth = len(self.resolver_prefix_paths)
  98             for path, kind in self.resolver_prefix_paths[-1]:
  99                 if self.check_resolver_prefix(depth, path, kind,
 100                         current_node, current_index):
 101                     if len(path) > depth:
 102                         prefix_paths.append((path, kind))
 103                     else:
 104                         exact_paths[kind] = self.yaml_path_resolvers[path, kind]
 105         else:
 106             for path, kind in self.yaml_path_resolvers:
 107                 if not path:
 108                     exact_paths[kind] = self.yaml_path_resolvers[path, kind]
 109                 else:
 110                     prefix_paths.append((path, kind))
 111         self.resolver_exact_paths.append(exact_paths)
 112         self.resolver_prefix_paths.append(prefix_paths)

 114     def ascend_resolver(self):
 115         if not self.yaml_path_resolvers:
 116             return
 117         self.resolver_exact_paths.pop()
 118         self.resolver_prefix_paths.pop()

 120     def check_resolver_prefix(self, depth, path, kind,
 121             current_node, current_index):
 122         node_check, index_check = path[depth-1]
 123         if isinstance(node_check, basestring):
 124             if current_node.tag != node_check:
 125                 return
 126         elif node_check is not None:
 127             if not isinstance(current_node, node_check):
 128                 return
 129         if index_check is True and current_index is not None:
 130             return
 131         if (index_check is False or index_check is None)    \
 132                 and current_index is None:
 133             return
 134         if isinstance(index_check, basestring):
 135             if not (isinstance(current_index, ScalarNode)
 136                     and index_check == current_index.value):
 137                 return
 138         elif isinstance(index_check, int) and not isinstance(index_check, bool):
 139             if index_check != current_index:
 140                 return
 141         return True

 143     def resolve(self, kind, value, implicit):
 144         if kind is ScalarNode and implicit[0]:
 145             if value == u'':
 146                 resolvers = self.yaml_implicit_resolvers.get(u'', [])
 147             else:
 148                 resolvers = self.yaml_implicit_resolvers.get(value[0], [])
 149             resolvers += self.yaml_implicit_resolvers.get(None, [])
 150             for tag, regexp in resolvers:
 151                 if regexp.match(value):
 152                     return tag
 153             implicit = implicit[1]
 154         if self.yaml_path_resolvers:
 155             exact_paths = self.resolver_exact_paths[-1]
 156             if kind in exact_paths:
 157                 return exact_paths[kind]
 158             if None in exact_paths:
 159                 return exact_paths[None]
 160         if kind is ScalarNode:
 161             return self.DEFAULT_SCALAR_TAG
 162         elif kind is SequenceNode:
 163             return self.DEFAULT_SEQUENCE_TAG
 164         elif kind is MappingNode:
 165             return self.DEFAULT_MAPPING_TAG
