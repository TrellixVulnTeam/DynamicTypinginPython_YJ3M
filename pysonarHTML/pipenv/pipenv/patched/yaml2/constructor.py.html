  59     def check_state_key(self, key):
  60         """Block special attributes/methods from being set in a newly created
  61         object, to prevent user-controlled methods from being called during
  62         deserialization"""
  63         if self.get_state_keys_blacklist_regexp().match(key):
  64             raise ConstructorError(None, None,
  65                 "blacklisted key '%s' in instance state found" % (key,), None)

  79     def construct_document(self, node):
  80         data = self.construct_object(node)
  81         while self.state_generators:
  82             state_generators = self.state_generators
  83             self.state_generators = []
  84             for generator in state_generators:
  85                 for dummy in generator:
  86                     pass
  87         self.constructed_objects = {}
  88         self.recursive_objects = {}
  89         self.deep_construct = False
  90         return data

  92     def construct_object(self, node, deep=False):
  93         if node in self.constructed_objects:
  94             return self.constructed_objects[node]
  95         if deep:
  96             old_deep = self.deep_construct
  97             self.deep_construct = True
  98         if node in self.recursive_objects:
  99             raise ConstructorError(None, None,
 100                     "found unconstructable recursive node", node.start_mark)
 101         self.recursive_objects[node] = None
 102         constructor = None
 103         tag_suffix = None
 104         if node.tag in self.yaml_constructors:
 105             constructor = self.yaml_constructors[node.tag]
 106         else:
 107             for tag_prefix in self.yaml_multi_constructors:
 108                 if tag_prefix is not None and node.tag.startswith(tag_prefix):
 109                     tag_suffix = node.tag[len(tag_prefix):]
 110                     constructor = self.yaml_multi_constructors[tag_prefix]
 111                     break
 112             else:
 113                 if None in self.yaml_multi_constructors:
 114                     tag_suffix = node.tag
 115                     constructor = self.yaml_multi_constructors[None]
 116                 elif None in self.yaml_constructors:
 117                     constructor = self.yaml_constructors[None]
 118                 elif isinstance(node, ScalarNode):
 119                     constructor = self.__class__.construct_scalar
 120                 elif isinstance(node, SequenceNode):
 121                     constructor = self.__class__.construct_sequence
 122                 elif isinstance(node, MappingNode):
 123                     constructor = self.__class__.construct_mapping
 124         if tag_suffix is None:
 125             data = constructor(self, node)
 126         else:
 127             data = constructor(self, tag_suffix, node)
 128         if isinstance(data, types.GeneratorType):
 129             generator = data
 130             data = generator.next()
 131             if self.deep_construct:
 132                 for dummy in generator:
 133                     pass
 134             else:
 135                 self.state_generators.append(generator)
 136         self.constructed_objects[node] = data
 137         del self.recursive_objects[node]
 138         if deep:
 139             self.deep_construct = old_deep
 140         return data

 142     def construct_scalar(self, node):
 143         if not isinstance(node, ScalarNode):
 144             raise ConstructorError(None, None,
 145                     "expected a scalar node, but found %s" % node.id,
 146                     node.start_mark)
 147         return node.value

 149     def construct_sequence(self, node, deep=False):
 150         if not isinstance(node, SequenceNode):
 151             raise ConstructorError(None, None,
 152                     "expected a sequence node, but found %s" % node.id,
 153                     node.start_mark)
 154         return [self.construct_object(child, deep=deep)
 155                 for child in node.value]
 156 
 157     def construct_mapping(self, node, deep=False):
 158         if not isinstance(node, MappingNode):
 159             raise ConstructorError(None, None,
 160                     "expected a mapping node, but found %s" % node.id,
 161                     node.start_mark)
 162         mapping = {}
 163         for key_node, value_node in node.value:
 164             key = self.construct_object(key_node, deep=deep)
 165             try:
 166                 hash(key)
 167             except TypeError, exc:
 168                 raise ConstructorError("while constructing a mapping", node.start_mark,
 169                         "found unacceptable key (%s)" % exc, key_node.start_mark)
 170             value = self.construct_object(value_node, deep=deep)
 171             mapping[key] = value
 172         return mapping
 173 
 174     def construct_pairs(self, node, deep=False):
 175         if not isinstance(node, MappingNode):
 176             raise ConstructorError(None, None,
 177                     "expected a mapping node, but found %s" % node.id,
 178                     node.start_mark)
 179         pairs = []
 180         for key_node, value_node in node.value:
 181             key = self.construct_object(key_node, deep=deep)
 182             value = self.construct_object(value_node, deep=deep)
 183             pairs.append((key, value))
 184         return pairs

 207     def flatten_mapping(self, node):
 208         merge = []
 209         index = 0
 210         while index < len(node.value):
 211             key_node, value_node = node.value[index]
 212             if key_node.tag == u'tag:yaml.org,2002:merge':
 213                 del node.value[index]
 214                 if isinstance(value_node, MappingNode):
 215                     self.flatten_mapping(value_node)
 216                     merge.extend(value_node.value)
 217                 elif isinstance(value_node, SequenceNode):
 218                     submerge = []
 219                     for subnode in value_node.value:
 220                         if not isinstance(subnode, MappingNode):
 221                             raise ConstructorError("while constructing a mapping",
 222                                     node.start_mark,
 223                                     "expected a mapping for merging, but found %s"
 224                                     % subnode.id, subnode.start_mark)
 225                         self.flatten_mapping(subnode)
 226                         submerge.append(subnode.value)
 227                     submerge.reverse()
 228                     for value in submerge:
 229                         merge.extend(value)
 230                 else:
 231                     raise ConstructorError("while constructing a mapping", node.start_mark,
 232                             "expected a mapping or list of mappings for merging, but found %s"
 233                             % value_node.id, value_node.start_mark)
 234             elif key_node.tag == u'tag:yaml.org,2002:value':
 235                 key_node.tag = u'tag:yaml.org,2002:str'
 236                 index += 1
 237             else:
 238                 index += 1
 239         if merge:
 240             node.value = merge + node.value

 264     def construct_yaml_int(self, node):
 265         value = str(self.construct_scalar(node))
 266         value = value.replace('_', '')
 267         sign = +1
 268         if value[0] == '-':
 269             sign = -1
 270         if value[0] in '+-':
 271             value = value[1:]
 272         if value == '0':
 273             return 0
 274         elif value.startswith('0b'):
 275             return sign*int(value[2:], 2)
 276         elif value.startswith('0x'):
 277             return sign*int(value[2:], 16)
 278         elif value[0] == '0':
 279             return sign*int(value, 8)
 280         elif ':' in value:
 281             digits = [int(part) for part in value.split(':')]
 282             digits.reverse()
 283             base = 1
 284             value = 0
 285             for digit in digits:
 286                 value += digit*base
 287                 base *= 60
 288             return sign*value
 289         else:
 290             return sign*int(value)

 297     def construct_yaml_float(self, node):
 298         value = str(self.construct_scalar(node))
 299         value = value.replace('_', '').lower()
 300         sign = +1
 301         if value[0] == '-':
 302             sign = -1
 303         if value[0] in '+-':
 304             value = value[1:]
 305         if value == '.inf':
 306             return sign*self.inf_value
 307         elif value == '.nan':
 308             return self.nan_value
 309         elif ':' in value:
 310             digits = [float(part) for part in value.split(':')]
 311             digits.reverse()
 312             base = 1
 313             value = 0.0
 314             for digit in digits:
 315                 value += digit*base
 316                 base *= 60
 317             return sign*value
 318         else:
 319             return sign*float(value)

 329     timestamp_regexp = re.compile(
 330             ur'''^(?P<year>[0-9][0-9][0-9][0-9])
 331                 -(?P<month>[0-9][0-9]?)
 332                 -(?P<day>[0-9][0-9]?)
 333                 (?:(?:[Tt]|[ \t]+)
 334                 (?P<hour>[0-9][0-9]?)
 335                 :(?P<minute>[0-9][0-9])
 336                 :(?P<second>[0-9][0-9])
 337                 (?:\.(?P<fraction>[0-9]*))?
 338                 (?:[ \t]*(?P<tz>Z|(?P<tz_sign>[-+])(?P<tz_hour>[0-9][0-9]?)
 339                 (?::(?P<tz_minute>[0-9][0-9]))?))?)?$''', re.X)
 340 
 341     def construct_yaml_timestamp(self, node):
 342         value = self.construct_scalar(node)
 343         match = self.timestamp_regexp.match(node.value)
 344         values = match.groupdict()
 345         year = int(values['year'])
 346         month = int(values['month'])
 347         day = int(values['day'])
 348         if not values['hour']:
 349             return datetime.date(year, month, day)
 350         hour = int(values['hour'])
 351         minute = int(values['minute'])
 352         second = int(values['second'])
 353         fraction = 0
 354         tzinfo = None
 355         if values['fraction']:
 356             fraction = values['fraction'][:6]
 357             while len(fraction) < 6:
 358                 fraction += '0'
 359             fraction = int(fraction)
 360         if values['tz_sign']:
 361             tz_hour = int(values['tz_hour'])
 362             tz_minute = int(values['tz_minute'] or 0)
 363             delta = datetime.timedelta(hours=tz_hour, minutes=tz_minute)
 364             if values['tz_sign'] == '-':
 365                 delta = -delta
 366             tzinfo = timezone(delta)
 367         elif values['tz']:
 368             tzinfo = timezone(datetime.timedelta(0))
 369         return datetime.datetime(year, month, day, hour, minute, second, fraction,
 370                                  tzinfo=tzinfo)

 372     def construct_yaml_omap(self, node):
 373         # Note: we do not check for duplicate keys, because it's too
 374         # CPU-expensive.
 375         omap = []
 376         yield omap
 377         if not isinstance(node, SequenceNode):
 378             raise ConstructorError("while constructing an ordered map", node.start_mark,
 379                     "expected a sequence, but found %s" % node.id, node.start_mark)
 380         for subnode in node.value:
 381             if not isinstance(subnode, MappingNode):
 382                 raise ConstructorError("while constructing an ordered map", node.start_mark,
 383                         "expected a mapping of length 1, but found %s" % subnode.id,
 384                         subnode.start_mark)
 385             if len(subnode.value) != 1:
 386                 raise ConstructorError("while constructing an ordered map", node.start_mark,
 387                         "expected a single mapping item, but found %d items" % len(subnode.value),
 388                         subnode.start_mark)
 389             key_node, value_node = subnode.value[0]
 390             key = self.construct_object(key_node)
 391             value = self.construct_object(value_node)
 392             omap.append((key, value))

 439     def construct_yaml_object(self, node, cls):
 440         data = cls.__new__(cls)
 441         yield data
 442         if hasattr(data, '__setstate__'):
 443             state = self.construct_mapping(node, deep=True)

 446             state = self.construct_mapping(node)

 532     def find_python_module(self, name, mark, unsafe=False):
 533         if not name:
 534             raise ConstructorError("while constructing a Python module", mark,
 535                     "expected non-empty name appended to the tag", mark)
 536         if unsafe:
 537             try:
 538                 __import__(name)
 539             except ImportError, exc:
 540                 raise ConstructorError("while constructing a Python module", mark,
 541                         "cannot find module %r (%s)" % (name.encode('utf-8'), exc), mark)
 542         if name not in sys.modules:
 543             raise ConstructorError("while constructing a Python module", mark,
 544                     "module %r is not imported" % name.encode('utf-8'), mark)
 545         return sys.modules[name]
 546 
 547     def find_python_name(self, name, mark, unsafe=False):
 548         if not name:
 549             raise ConstructorError("while constructing a Python object", mark,
 550                     "expected non-empty name appended to the tag", mark)
 551         if u'.' in name:
 552             module_name, object_name = name.rsplit('.', 1)
 553         else:
 554             module_name = '__builtin__'
 555             object_name = name
 556         if unsafe:
 557             try:
 558                 __import__(module_name)
 559             except ImportError, exc:
 560                 raise ConstructorError("while constructing a Python object", mark,
 561                         "cannot find module %r (%s)" % (module_name.encode('utf-8'), exc), mark)
 562         if module_name not in sys.modules:
 563             raise ConstructorError("while constructing a Python object", mark,
 564                     "module %r is not imported" % module_name.encode('utf-8'), mark)
 565         module = sys.modules[module_name]
 566         if not hasattr(module, object_name):
 567             raise ConstructorError("while constructing a Python object", mark,
 568                     "cannot find %r in the module %r" % (object_name.encode('utf-8'),
 569                         module.__name__), mark)
 570         return getattr(module, object_name)

 590     def make_python_instance(self, suffix, node,
 591             args=None, kwds=None, newobj=False, unsafe=False):
 592         if not args:
 593             args = []
 594         if not kwds:
 595             kwds = {}
 596         cls = self.find_python_name(suffix, node.start_mark)
 597         if not (unsafe or isinstance(cls, type) or isinstance(cls, type(self.classobj))):
 598             raise ConstructorError("while constructing a Python instance", node.start_mark,
 599                     "expected a class, but found %r" % type(cls),
 600                     node.start_mark)
 601         if newobj and isinstance(cls, type(self.classobj))  \
 602                 and not args and not kwds:
 603             instance = self.classobj()
 604             instance.__class__ = cls
 605             return instance
 606         elif newobj and isinstance(cls, type):
 607             return cls.__new__(cls, *args, **kwds)
 608         else:
 609             return cls(*args, **kwds)
 610 
 611     def set_python_instance_state(self, instance, state, unsafe=False):
 612         if hasattr(instance, '__setstate__'):
 613             instance.__setstate__(state)
 614         else:
 615             slotstate = {}
 616             if isinstance(state, tuple) and len(state) == 2:
 617                 state, slotstate = state
 618             if hasattr(instance, '__dict__'):
 619                 if not unsafe and state:
 620                     for key in state.keys():
 621                         self.check_state_key(key)
 622                 instance.__dict__.update(state)
 623             elif state:
 624                 slotstate.update(state)
 625             for key, value in slotstate.items():
 626                 if not unsafe:
 627                     self.check_state_key(key)
 628                 setattr(instance, key, value)
 629 
 630     def construct_python_object(self, suffix, node):
 631         # Format:
 632         #   !!python/object:module.name { ... state ... }
 633         instance = self.make_python_instance(suffix, node, newobj=True)
 634         yield instance
 635         deep = hasattr(instance, '__setstate__')
 636         state = self.construct_mapping(node, deep=deep)
 637         self.set_python_instance_state(instance, state)
 638 
 639     def construct_python_object_apply(self, suffix, node, newobj=False):
 640         # Format:
 641         #   !!python/object/apply       # (or !!python/object/new)
 642         #   args: [ ... arguments ... ]
 643         #   kwds: { ... keywords ... }
 644         #   state: ... state ...
 645         #   listitems: [ ... listitems ... ]
 646         #   dictitems: { ... dictitems ... }
 647         # or short format:
 648         #   !!python/object/apply [ ... arguments ... ]
 649         # The difference between !!python/object/apply and !!python/object/new
 650         # is how an object is created, check make_python_instance for details.
 651         if isinstance(node, SequenceNode):
 652             args = self.construct_sequence(node, deep=True)
 653             kwds = {}
 654             state = {}
 655             listitems = []
 656             dictitems = {}
 657         else:
 658             value = self.construct_mapping(node, deep=True)
 659             args = value.get('args', [])
 660             kwds = value.get('kwds', {})
 661             state = value.get('state', {})
 662             listitems = value.get('listitems', [])
 663             dictitems = value.get('dictitems', {})
 664         instance = self.make_python_instance(suffix, node, args, kwds, newobj)
 665         if state:
 666             self.set_python_instance_state(instance, state)
 667         if listitems:
 668             instance.extend(listitems)
 669         if dictitems:
 670             for key in dictitems:
 671                 instance[key] = dictitems[key]
 672         return instance
 673 
 674     def construct_python_object_new(self, suffix, node):
 675         return self.construct_python_object_apply(suffix, node, newobj=True)
