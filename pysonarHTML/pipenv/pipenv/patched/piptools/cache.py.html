  17 def _implementation_name():
  18     """similar to PEP 425, however the minor version is separated from the
  19     major to differentation "3.10" and "31.0".
  20     """
  21     implementation_name = platform.python_implementation().lower()
  22     implementation = _PEP425_PY_TAGS.get(implementation_name, "??")
  23     return "{}{}.{}".format(implementation, *sys.version_info)

  26 class CorruptCacheError(PipToolsError):
  27     def __init__(self, path):
  28         self.path = path
  29 
  30     def __str__(self):
  31         lines = [
  32             "The dependency cache seems to have been corrupted.",
  33             "Inspect, or delete, the following file:",
  34             "  {}".format(self.path),
  35         ]
  36         return os.linesep.join(lines)

  39 def read_cache_file(cache_file_path):
  40     with open(cache_file_path, "r") as cache_file:
  41         try:
  42             doc = json.load(cache_file)
  43         except ValueError:
  44             raise CorruptCacheError(cache_file_path)
  45 
  46         # Check version and load the contents
  47         if doc["__format__"] != 1:
  48             raise AssertionError("Unknown cache file format")
  49         return doc["dependencies"]

  52 class DependencyCache(object):
  53     """
  54     Creates a new persistent dependency cache for the current Python version.
  55     The cache file is written to the appropriate user cache dir for the
  56     current platform, i.e.
  57 
  58         ~/.cache/pip-tools/depcache-pyX.Y.json
  59 
  60     Where py indicates the Python implementation.
  61     Where X.Y indicates the Python version.
  62     """
  63 
  64     def __init__(self, cache_dir):
  65         if not os.path.isdir(cache_dir):
  66             os.makedirs(cache_dir)
  67         cache_filename = "depcache-{}.json".format(_implementation_name())
  68 
  69         self._cache_file = os.path.join(cache_dir, cache_filename)
  70         self._cache = None
  71 
  72     @property
  73     def cache(self):
  74         """
  75         The dictionary that is the actual in-memory cache.  This property
  76         lazily loads the cache from disk.
  77         """
  78         if self._cache is None:
  79             self.read_cache()
  80         return self._cache
  81 
  82     def as_cache_key(self, ireq):
  83         """
  84         Given a requirement, return its cache key. This behavior is a little weird
  85         in order to allow backwards compatibility with cache files. For a requirement
  86         without extras, this will return, for example:
  87 
  88         ("ipython", "2.1.0")
  89 
  90         For a requirement with extras, the extras will be comma-separated and appended
  91         to the version, inside brackets, like so:
  92 
  93         ("ipython", "2.1.0[nbconvert,notebook]")
  94         """
  95         name, version, extras = as_tuple(ireq)
  96         if not extras:
  97             extras_string = ""
  98         else:
  99             extras_string = "[{}]".format(",".join(extras))
 100         return name, "{}{}".format(version, extras_string)
 101 
 102     def read_cache(self):
 103         """Reads the cached contents into memory."""
 104         if os.path.exists(self._cache_file):
 105             self._cache = read_cache_file(self._cache_file)
 106         else:
 107             self._cache = {}
 108 
 109     def write_cache(self):
 110         """Writes the cache to disk as JSON."""
 111         doc = {"__format__": 1, "dependencies": self._cache}
 112         with open(self._cache_file, "w") as f:
 113             json.dump(doc, f, sort_keys=True)
 114 
 115     def clear(self):
 116         self._cache = {}
 117         self.write_cache()
 118 
 119     def __contains__(self, ireq):
 120         pkgname, pkgversion_and_extras = self.as_cache_key(ireq)
 121         return pkgversion_and_extras in self.cache.get(pkgname, {})
 122 
 123     def __getitem__(self, ireq):
 124         pkgname, pkgversion_and_extras = self.as_cache_key(ireq)
 125         return self.cache[pkgname][pkgversion_and_extras]
 126 
 127     def __setitem__(self, ireq, values):
 128         pkgname, pkgversion_and_extras = self.as_cache_key(ireq)
 129         self.cache.setdefault(pkgname, {})
 130         self.cache[pkgname][pkgversion_and_extras] = values
 131         self.write_cache()
 132 
 133     def reverse_dependencies(self, ireqs):
 134         """
 135         Returns a lookup table of reverse dependencies for all the given ireqs.
 136 
 137         Since this is all static, it only works if the dependency cache
 138         contains the complete data, otherwise you end up with a partial view.
 139         This is typically no problem if you use this function after the entire
 140         dependency tree is resolved.
 141         """
 142         ireqs_as_cache_values = [self.as_cache_key(ireq) for ireq in ireqs]
 143         return self._reverse_dependencies(ireqs_as_cache_values)
 144 
 145     def _reverse_dependencies(self, cache_keys):
 146         """
 147         Returns a lookup table of reverse dependencies for all the given cache keys.
 148 
 149         Example input:
 150 
 151             [('pep8', '1.5.7'),
 152              ('flake8', '2.4.0'),
 153              ('mccabe', '0.3'),
 154              ('pyflakes', '0.8.1')]
 155 
 156         Example output:
 157 
 158             {'pep8': ['flake8'],
 159              'flake8': [],
 160              'mccabe': ['flake8'],
 161              'pyflakes': ['flake8']}
 162 
 163         """
 164         # First, collect all the dependencies into a sequence of (parent, child)
 165         # tuples, like [('flake8', 'pep8'), ('flake8', 'mccabe'), ...]
 166         return lookup_table(
 167             (key_from_req(Requirement(dep_name)), name)
 168             for name, version_and_extras in cache_keys
 169             for dep_name in self.cache[name][version_and_extras]
 170         )
