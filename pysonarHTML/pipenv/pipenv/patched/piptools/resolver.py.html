  31     Summary of a requirement's properties for comparison purposes.

  34     def __init__(self, ireq):
  35         self.req = ireq.req
  36         self.key = key_from_ireq(ireq)
  37         self.extras = str(sorted(ireq.extras))
  38         self.markers = ireq.markers
  39         self.specifier = str(ireq.specifier)

  51 def combine_install_requirements(ireqs):
  52     """
  53     Return a single install requirement that reflects a combination of
  54     all the inputs.
  55     """
  56     # We will store the source ireqs in a _source_ireqs attribute;
  57     # if any of the inputs have this, then use those sources directly.
  58     source_ireqs = []
  59     for ireq in ireqs:
  60         source_ireqs.extend(getattr(ireq, "_source_ireqs", [ireq]))
  61 
  62     # deepcopy the accumulator so as to not modify the inputs
  63     combined_ireq = copy.deepcopy(source_ireqs[0])
  64     for ireq in source_ireqs[1:]:
  65         # NOTE we may be losing some info on dropped reqs here
  66         if combined_ireq.req is not None and ireq.req is not None:
  67             combined_ireq.req.specifier &= ireq.req.specifier
  68         combined_ireq.constraint &= ireq.constraint
  69         if ireq.markers and not combined_ireq.markers:
  70             combined_ireq.markers = copy.deepcopy(ireq.markers)
  71         elif ireq.markers and combined_ireq.markers:
  72             _markers = []  # type: List[Marker]
  73             for marker in [ireq.markers, combined_ireq.markers]:
  74                 if isinstance(marker, str):
  75                     _markers.append(Marker(marker))
  76                 else:
  77                     _markers.append(marker)
  78             marker_str = " and ".join([normalize_marker_str(m) for m in _markers if m])
  79             combined_ireq.markers = Marker(marker_str)
  80         # Return a sorted, de-duped tuple of extras
  81         combined_ireq.extras = tuple(
  82             sorted(set(tuple(combined_ireq.extras) + tuple(ireq.extras)))
  83         )
  84 
  85     # InstallRequirements objects are assumed to come from only one source, and
  86     # so they support only a single comes_from entry. This function breaks this
  87     # model. As a workaround, we deterministically choose a single source for
  88     # the comes_from entry, and add an extra _source_ireqs attribute to keep
  89     # track of multiple sources for use within pip-tools.
  90     if len(source_ireqs) > 1:
  91         if any(ireq.comes_from is None for ireq in source_ireqs):
  92             # None indicates package was directly specified.
  93             combined_ireq.comes_from = None
  94         else:
  95             # Populate the comes_from field from one of the sources.
  96             # Requirement input order is not stable, so we need to sort:
  97             # We choose the shortest entry in order to keep the printed
  98             # representation as concise as possible.
  99             combined_ireq.comes_from = min(
 100                 (ireq.comes_from for ireq in source_ireqs),
 101                 key=lambda x: (len(str(x)), str(x)),
 102             )
 103         combined_ireq._source_ireqs = source_ireqs
 104     return combined_ireq

 107 class Resolver(object):
 108     def __init__(
 109         self,
 110         constraints,
 111         repository,
 112         cache,
 113         prereleases=False,
 114         clear_caches=False,
 115         allow_unsafe=False,
 116     ):
 117         """
 118         This class resolves a given set of constraints (a collection of
 119         InstallRequirement objects) by consulting the given Repository and the
 120         DependencyCache.
 121         """
 122         self.our_constraints = set(constraints)
 123         self.their_constraints = set()
 124         self.repository = repository
 125         self.dependency_cache = cache
 126         self.prereleases = prereleases
 127         self.clear_caches = clear_caches
 128         self.allow_unsafe = allow_unsafe
 129         self.unsafe_constraints = set()
 130 
 131     @property
 132     def constraints(self):
 133         return set(
 134             self._group_constraints(
 135                 chain(
 136                     sorted(self.our_constraints, key=str),
 137                     sorted(self.their_constraints, key=str),
 138                 )
 139             )
 140         )
 141 
 142     def resolve_hashes(self, ireqs):
 143         """
 144         Finds acceptable hashes for all of the given InstallRequirements.
 145         """
 146         log.debug("")
 147         log.debug("Generating hashes:")
 148         with self.repository.allow_all_wheels():
 149             return {ireq: self.repository.get_hashes(ireq) for ireq in ireqs}
 150 
 151     def resolve(self, max_rounds=10):
 152         """
 153         Finds concrete package versions for all the given InstallRequirements
 154         and their recursive dependencies.  The end result is a flat list of
 155         (name, version) tuples.  (Or an editable package.)
 156 
 157         Resolves constraints one round at a time, until they don't change
 158         anymore.  Protects against infinite loops by breaking out after a max
 159         number rounds.
 160         """
 161         if self.clear_caches:
 162             self.dependency_cache.clear()
 163             self.repository.clear_caches()
 164 
 165         # Ignore existing packages
 166         os.environ[str("PIP_EXISTS_ACTION")] = str(
 167             "i"
 168         )  # NOTE: str() wrapping necessary for Python 2/3 compat
 169         for current_round in count(start=1):  # pragma: no branch
 170             if current_round > max_rounds:
 171                 raise RuntimeError(
 172                     "No stable configuration of concrete packages "
 173                     "could be found for the given constraints after "
 174                     "{max_rounds} rounds of resolving.\n"
 175                     "This is likely a bug.".format(max_rounds=max_rounds)
 176                 )
 177 
 178             log.debug("")
 179             log.debug(magenta("{:^60}".format("ROUND {}".format(current_round))))
 180             has_changed, best_matches = self._resolve_one_round()
 181             log.debug("-" * 60)
 182             log.debug(
 183                 "Result of round {}: {}".format(
 184                     current_round, "not stable" if has_changed else "stable, done"
 185                 )
 186             )
 187             if not has_changed:
 188                 break
 189 
 190             # If a package version (foo==2.0) was built in a previous round,
 191             # and in this round a different version of foo needs to be built
 192             # (i.e. foo==1.0), the directory will exist already, which will
 193             # cause a pip build failure.  The trick is to start with a new
 194             # build cache dir for every round, so this can never happen.
 195             self.repository.freshen_build_caches()
 196 
 197         del os.environ["PIP_EXISTS_ACTION"]
 198 
 199         # Only include hard requirements and not pip constraints
 200         results = {req for req in best_matches if not req.constraint}
 201 
 202         # Filter out unsafe requirements.
 203         self.unsafe_constraints = set()
 204         if not self.allow_unsafe:
 205             # reverse_dependencies is used to filter out packages that are only
 206             # required by unsafe packages. This logic is incomplete, as it would
 207             # fail to filter sub-sub-dependencies of unsafe packages. None of the
 208             # UNSAFE_PACKAGES currently have any dependencies at all (which makes
 209             # sense for installation tools) so this seems sufficient.
 210             reverse_dependencies = self.reverse_dependencies(results)
 211             for req in results.copy():
 212                 required_by = reverse_dependencies.get(req.name.lower(), [])
 213                 if req.name in UNSAFE_PACKAGES or (
 214                     required_by and all(name in UNSAFE_PACKAGES for name in required_by)
 215                 ):
 216                     self.unsafe_constraints.add(req)
 217                     results.remove(req)
 218 
 219         return results
 220 
 221     def _group_constraints(self, constraints):
 222         """
 223         Groups constraints (remember, InstallRequirements!) by their key name,
 224         and combining their SpecifierSets into a single InstallRequirement per
 225         package.  For example, given the following constraints:
 226 
 227             Django<1.9,>=1.4.2
 228             django~=1.5
 229             Flask~=0.7
 230 
 231         This will be combined into a single entry per package:
 232 
 233             django~=1.5,<1.9,>=1.4.2
 234             flask~=0.7
 235 
 236         """
 237         for _, ireqs in full_groupby(constraints, key=key_from_ireq):
 238             yield combine_install_requirements(ireqs)
 239 
 240     def _resolve_one_round(self):
 241         """
 242         Resolves one level of the current constraints, by finding the best
 243         match for each package in the repository and adding all requirements
 244         for those best package versions.  Some of these constraints may be new
 245         or updated.
 246 
 247         Returns whether new constraints appeared in this round.  If no
 248         constraints were added or changed, this indicates a stable
 249         configuration.
 250         """
 251         # Sort this list for readability of terminal output
 252         constraints = sorted(self.constraints, key=key_from_ireq)
 253 
 254         log.debug("Current constraints:")
 255         for constraint in constraints:
 256             log.debug("  {}".format(constraint))
 257 
 258         log.debug("")
 259         log.debug("Finding the best candidates:")
 260         best_matches = {self.get_best_match(ireq) for ireq in constraints}
 261 
 262         # Find the new set of secondary dependencies
 263         log.debug("")
 264         log.debug("Finding secondary dependencies:")
 265 
 266         their_constraints = []
 267         for best_match in best_matches:
 268             their_constraints.extend(self._iter_dependencies(best_match))
 269         # Grouping constraints to make clean diff between rounds
 270         theirs = set(self._group_constraints(sorted(their_constraints, key=str)))
 271 
 272         # NOTE: We need to compare RequirementSummary objects, since
 273         # InstallRequirement does not define equality
 274         diff = {RequirementSummary(t) for t in theirs} - {
 275             RequirementSummary(t) for t in self.their_constraints
 276         }
 277         removed = {RequirementSummary(t) for t in self.their_constraints} - {
 278             RequirementSummary(t) for t in theirs
 279         }
 280 
 281         has_changed = len(diff) > 0 or len(removed) > 0
 282         if has_changed:
 283             log.debug("")
 284             log.debug("New dependencies found in this round:")
 285             for new_dependency in sorted(diff, key=key_from_ireq):
 286                 log.debug("  adding {}".format(new_dependency))
 287             log.debug("Removed dependencies in this round:")
 288             for removed_dependency in sorted(removed, key=key_from_ireq):
 289                 log.debug("  removing {}".format(removed_dependency))
 290 
 291         # Store the last round's results in the their_constraints
 292         self.their_constraints = theirs
 293         return has_changed, best_matches

 295     def get_best_match(self, ireq):
 296         """
 297         Returns a (pinned or editable) InstallRequirement, indicating the best
 298         match to use for the given InstallRequirement (in the form of an
 299         InstallRequirement).
 300 
 301         Example:
 302         Given the constraint Flask>=0.10, may return Flask==0.10.1 at
 303         a certain moment in time.
 304 
 305         Pinned requirements will always return themselves, i.e.
 306 
 307             Flask==0.10.1 => Flask==0.10.1
 308 
 309         """
 310         if ireq.editable or is_url_requirement(ireq):
 311             # NOTE: it's much quicker to immediately return instead of
 312             # hitting the index server
 313             best_match = ireq
 314         elif is_pinned_requirement(ireq):
 315             # NOTE: it's much quicker to immediately return instead of
 316             # hitting the index server
 317             best_match = ireq
 318         else:
 319             best_match = self.repository.find_best_match(
 320                 ireq, prereleases=self.prereleases
 321             )
 322 
 323         # Format the best match
 324         log.debug(
 325             "  found candidate {} (constraint was {})".format(
 326                 format_requirement(best_match), format_specifier(ireq)
 327             )
 328         )
 329         best_match.comes_from = ireq.comes_from
 330         if hasattr(ireq, "_source_ireqs"):
 331             best_match._source_ireqs = ireq._source_ireqs
 332         return best_match

 334     def _iter_dependencies(self, ireq):
 335         """
 336         Given a pinned, url, or editable InstallRequirement, collects all the
 337         secondary dependencies for them, either by looking them up in a local
 338         cache, or by reaching out to the repository.
 339 
 340         Editable requirements will never be looked up, as they may have
 341         changed at any time.
 342         """
 343         # Pip does not resolve dependencies of constraints. We skip handling
 344         # constraints here as well to prevent the cache from being polluted.
 345         # Constraints that are later determined to be dependencies will be
 346         # marked as non-constraints in later rounds by
 347         # `combine_install_requirements`, and will be properly resolved.
 348         # See https://github.com/pypa/pip/
 349         # blob/6896dfcd831330c13e076a74624d95fa55ff53f4/src/pip/_internal/
 350         # legacy_resolve.py#L325
 351         if ireq.constraint:
 352             return
 353 
 354         if ireq.editable or (is_url_requirement(ireq) and not ireq.link.is_wheel):
 355             for dependency in self.repository.get_dependencies(ireq):
 356                 yield dependency
 357             return
 358 
 359         # fix our malformed extras
 360         if ireq.extras:
 361             if getattr(ireq, "extra", None):
 362                 if ireq.extras:
 363                     ireq.extras.extend(ireq.extra)
 364                 else:
 365                     ireq.extras = ireq.extra
 366 
 367         elif not is_pinned_requirement(ireq):
 368             raise TypeError(
 369                 "Expected pinned or editable requirement, got {}".format(ireq)
 370             )
 371 
 372         # Now, either get the dependencies from the dependency cache (for
 373         # speed), or reach out to the external repository to
 374         # download and inspect the package version and get dependencies
 375         # from there
 376         if ireq not in self.dependency_cache:
 377             log.debug(
 378                 "  {} not in cache, need to check index".format(
 379                     format_requirement(ireq)
 380                 ),
 381                 fg="yellow",
 382             )
 383             dependencies = self.repository.get_dependencies(ireq)
 384             self.dependency_cache[ireq] = sorted(set(format_requirement(ireq) for ireq in dependencies))
 385 
 386         # Example: ['Werkzeug>=0.9', 'Jinja2>=2.4']
 387         dependency_strings = self.dependency_cache[ireq]
 388         log.debug(
 389             "  {:25} requires {}".format(
 390                 format_requirement(ireq),
 391                 ", ".join(sorted(dependency_strings, key=lambda s: s.lower())) or "-",
 392             )
 393         )
 394         for dependency_string in dependency_strings:
 395             yield install_req_from_line(
 396                 dependency_string, constraint=ireq.constraint, comes_from=ireq
 397             )
 398 
 399     def reverse_dependencies(self, ireqs):
 400         is_non_wheel_url = lambda r: is_url_requirement(r) and not r.link.is_wheel
 401         non_editable = [
 402             ireq for ireq in ireqs if not (ireq.editable or is_non_wheel_url(ireq))
 403         ]
 404         return self.dependency_cache.reverse_dependencies(non_editable)
