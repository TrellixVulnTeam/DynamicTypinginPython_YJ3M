  33 def simplify_markers(ireq):
  34     """simplify_markers "This code cleans up markers for a specific :class:`~InstallRequirement`"
  35 
  36     Clean and deduplicate markers.
  37 
  38     :param ireq: An InstallRequirement to clean
  39     :type ireq: :class:`~pip._internal.req.req_install.InstallRequirement`
  40     :return: An InstallRequirement with cleaned Markers
  41     :rtype: :class:`~pip._internal.req.req_install.InstallRequirement`
  42     """
  43 
  44     if not getattr(ireq, 'markers', None):
  45         return ireq
  46     markers = ireq.markers
  47     marker_list = []
  48     if isinstance(markers, six.string_types):
  49         if ';' in markers:
  50             markers = [Marker(m_str.strip()) for m_str in markers.split(';')]
  51         else:
  52             markers = Marker(markers)
  53     for m in markers._markers:
  54         _single_marker = []
  55         if isinstance(m[0], six.string_types):
  56             continue
  57         if not isinstance(m[0], (list, tuple)):
  58             marker_list.append(''.join([_piece.serialize() for _piece in m]))
  59             continue
  60         for _marker_part in m:
  61             if isinstance(_marker_part, six.string_types):
  62                 _single_marker.append(_marker_part)
  63                 continue
  64             _single_marker.append(''.join([_piece.serialize() for _piece in _marker_part]))
  65         _single_marker = [_m.strip() for _m in _single_marker]
  66         marker_list.append(tuple(_single_marker,))
  67     marker_str = ' and '.join(list(dedup(tuple(marker_list,)))) if marker_list else ''
  68     new_markers = Marker(marker_str)
  69     ireq.markers = new_markers
  70     new_ireq = install_req_from_line(format_requirement(ireq))
  71     if ireq.constraint:
  72         new_ireq.constraint = ireq.constraint
  73     return new_ireq

  76 def clean_requires_python(candidates):
  77     """Get a cleaned list of all the candidates with valid specifiers in the `requires_python` attributes."""
  78     all_candidates = []
  79     py_version = parse_version(os.environ.get('PIPENV_REQUESTED_PYTHON_VERSION', '.'.join(map(str, sys.version_info[:3]))))
  80     for c in candidates:
  81         if getattr(c, "requires_python", None):
  82             # Old specifications had people setting this to single digits
  83             # which is effectively the same as '>=digit,<digit+1'
  84             if len(c.requires_python) == 1 and c.requires_python in ("2", "3"):
  85                 c.requires_python = '>={0},<{1!s}'.format(c.requires_python, int(c.requires_python) + 1)
  86             try:
  87                 specifierset = SpecifierSet(c.requires_python)
  88             except InvalidSpecifier:
  89                 continue
  90             else:
  91                 if not specifierset.contains(py_version):
  92                     continue
  93         all_candidates.append(c)
  94     return all_candidates

 105 def key_from_req(req):
 106     """Get an all-lowercase version of the requirement's name."""
 107     if hasattr(req, "key"):
 108         # from pkg_resources, such as installed dists for pip-sync
 109         key = req.key
 110     else:
 111         # from packaging, such as install requirements from requirements.txt
 112         key = req.name
 113 
 114     key = key.replace("_", "-").lower()
 115     return key

 122 def make_install_requirement(name, version, extras, markers, constraint=False):
 123     # If no extras are specified, the extras string is blank
 124     extras_string = ""
 125     if extras:
 126         # Sort extras for stability
 127         extras_string = "[{}]".format(",".join(sorted(extras)))
 128 
 129     if not markers:
 130         return install_req_from_line(
 131             str('{}{}=={}'.format(name, extras_string, version)),
 132             constraint=constraint)
 133     else:
 134         return install_req_from_line(
 135             str('{}{}=={}; {}'.format(name, extras_string, version, str(markers))),
 136             constraint=constraint)

 139 def _requirement_to_str_lowercase_name(requirement):
 140     """
 141     Formats a packaging.requirements.Requirement with a lowercase name.
 142 
 143     This is simply a copy of
 144     https://github.com/pypa/pipenv/patched/packaging/blob/pipenv/patched/16.8/packaging/requirements.py#L109-L124
 145     modified to lowercase the dependency name.
 146 
 147     Previously, we were invoking the original Requirement.__str__ method and
 148     lowercasing the entire result, which would lowercase the name, *and* other,
 149     important stuff that should not be lowercased (such as the marker). See
 150     this issue for more information: https://github.com/pypa/pipenv/patched/pipenv/issues/2113.
 151     """
 152     parts = [requirement.name.lower()]
 153 
 154     if requirement.extras:
 155         parts.append("[{0}]".format(",".join(sorted(requirement.extras))))
 156 
 157     if requirement.specifier:
 158         parts.append(str(requirement.specifier))
 159 
 160     if requirement.url:
 161         parts.append("@ {0}".format(requirement.url))
 162 
 163     if requirement.marker:
 164         parts.append("; {0}".format(requirement.marker))
 165 
 166     return "".join(parts)

 177 def format_requirement(ireq, marker=None, hashes=None):
 178     """
 179     Generic formatter for pretty printing InstallRequirements to the terminal
 180     in a less verbose way than using its `__str__` method.
 181     """
 182     if ireq.editable:
 183         line = "-e {}".format(ireq.link.url)
 184     elif ireq.link and ireq.link.is_vcs:
 185         line = str(ireq.req)
 186     elif is_url_requirement(ireq):
 187         line = ireq.link.url
 188     else:
 189         line = _requirement_to_str_lowercase_name(ireq.req)
 190 
 191     if marker and ';' not in line:
 192         line = "{}; {}".format(line, marker)
 193 
 194     if hashes:
 195         for hash_ in sorted(hashes):
 196             line += " \\\n    --hash={}".format(hash_)
 197 
 198     return line

 212 def is_pinned_requirement(ireq):
 213     """
 214     Returns whether an InstallRequirement is a "pinned" requirement.
 215 
 216     An InstallRequirement is considered pinned if:
 217 
 218     - Is not editable
 219     - It has exactly one specifier
 220     - That specifier is "=="
 221     - The version does not contain a wildcard
 222 
 223     Examples:
 224         django==1.8   # pinned
 225         django>1.8    # NOT pinned
 226         django~=1.8   # NOT pinned
 227         django==1.*   # NOT pinned
 228     """
 229     if ireq.editable:
 230         return False
 231 
 232     if ireq.req is None or len(ireq.specifier._specs) != 1:
 233         return False
 234 
 235     op, version = next(iter(ireq.specifier._specs))._spec
 236     return (op == "==" or op == "===") and not version.endswith(".*")

 239 def as_tuple(ireq):
 240     """
 241     Pulls out the (name: str, version:str, extras:(str)) tuple from
 242     the pinned InstallRequirement.
 243     """
 244     if not is_pinned_requirement(ireq):
 245         raise TypeError("Expected a pinned InstallRequirement, got {}".format(ireq))
 246 
 247     name = key_from_ireq(ireq)
 248     version = next(iter(ireq.specifier._specs))._spec[1]
 249     extras = tuple(sorted(ireq.extras))
 250     return name, version, extras

 263 def lookup_table(values, key=None, keyval=None, unique=False, use_lists=False):
 264     """
 265     Builds a dict-based lookup table (index) elegantly.
 266 
 267     Supports building normal and unique lookup tables.  For example:
 268 
 269     >>> assert lookup_table(
 270     ...     ['foo', 'bar', 'baz', 'qux', 'quux'], lambda s: s[0]) == {
 271     ...     'b': {'bar', 'baz'},
 272     ...     'f': {'foo'},
 273     ...     'q': {'quux', 'qux'}
 274     ... }
 275 
 276     For key functions that uniquely identify values, set unique=True:
 277 
 278     >>> assert lookup_table(
 279     ...     ['foo', 'bar', 'baz', 'qux', 'quux'], lambda s: s[0],
 280     ...     unique=True) == {
 281     ...     'b': 'baz',
 282     ...     'f': 'foo',
 283     ...     'q': 'quux'
 284     ... }
 285 
 286     For the values represented as lists, set use_lists=True:
 287 
 288     >>> assert lookup_table(
 289     ...     ['foo', 'bar', 'baz', 'qux', 'quux'], lambda s: s[0],
 290     ...     use_lists=True) == {
 291     ...     'b': ['bar', 'baz'],
 292     ...     'f': ['foo'],
 293     ...     'q': ['qux', 'quux']
 294     ... }
 295 
 296     The values of the resulting lookup table will be lists, not sets.
 297 
 298     For extra power, you can even change the values while building up the LUT.
 299     To do so, use the `keyval` function instead of the `key` arg:
 300 
 301     >>> assert lookup_table(
 302     ...     ['foo', 'bar', 'baz', 'qux', 'quux'],
 303     ...     keyval=lambda s: (s[0], s[1:])) == {
 304     ...     'b': {'ar', 'az'},
 305     ...     'f': {'oo'},
 306     ...     'q': {'uux', 'ux'}
 307     ... }
 308 
 309     """
 310     if keyval is None:
 311         if key is None:
 312 
 313             def keyval(v):
 314                 return v
 315 
 316         else:
 317 
 318             def keyval(v):
 319                 return (key(v), v)
 320 
 321     if unique:
 322         return dict(keyval(v) for v in values)
 323 
 324     lut = {}
 325     for value in values:
 326         k, v = keyval(value)
 327         try:
 328             s = lut[k]
 329         except KeyError:
 330             if use_lists:
 331                 s = lut[k] = list()
 332             else:
 333                 s = lut[k] = set()
 334         if use_lists:
 335             s.append(v)
 336         else:
 337             s.add(v)
 338     return dict(lut)

 358 def fs_str(string):
 359     """
 360     Convert given string to a correctly encoded filesystem string.
 361 
 362     On Python 2, if the input string is unicode, converts it to bytes
 363     encoded with the filesystem encoding.
 364 
 365     On Python 3 returns the string as is, since Python 3 uses unicode
 366     paths and the input string shouldn't be bytes.
 367 
 368     :type string: str|unicode
 369     :rtype: str
 370     """
 371     if isinstance(string, str):
 372         return string
 373     if isinstance(string, bytes):
 374         raise AssertionError
 375     return string.encode(_fs_encoding)
 376 
 377 
 378 _fs_encoding = sys.getfilesystemencoding() or sys.getdefaultencoding()

 381 def get_hashes_from_ireq(ireq):
 382     """
 383     Given an InstallRequirement, return a list of string hashes in
 384     the format "{algorithm}:{hash}". Return an empty list if there are no hashes
 385     in the requirement options.
 386     """
 387     result = []
 388     if PIP_VERSION[:2] <= (20, 0):
 389         ireq_hashes = ireq.options.get("hashes", {})
 390     else:
 391         ireq_hashes = ireq.hash_options
 392     for algorithm, hexdigests in ireq_hashes.items():
 393         for hash_ in hexdigests:
 394             result.append("{}:{}".format(algorithm, hash_))
 395     return result

 398 def force_text(s):
 399     """
 400     Return a string representing `s`.
 401     """
 402     if s is None:
 403         return ""
 404     if not isinstance(s, six.string_types):
 405         return six.text_type(s)
 406     return s

 409 def get_compile_command(click_ctx):
 410     """
 411     Returns a normalized compile command depending on cli context.
 412 
 413     The command will be normalized by:
 414         - expanding options short to long
 415         - removing values that are already default
 416         - sorting the arguments
 417         - removing one-off arguments like '--upgrade'
 418         - removing arguments that don't change build behaviour like '--verbose'
 419     """
 420     from piptools.scripts.compile import cli
 421 
 422     # Map of the compile cli options (option name -> click.Option)
 423     compile_options = {option.name: option for option in cli.params}
 424 
 425     left_args = []
 426     right_args = []
 427 
 428     for option_name, value in click_ctx.params.items():
 429         option = compile_options[option_name]
 430 
 431         # Get the latest option name (usually it'll be a long name)
 432         option_long_name = option.opts[-1]
 433 
 434         # Collect variadic args separately, they will be added
 435         # at the end of the command later
 436         if option.nargs < 0:
 437             # These will necessarily be src_files
 438             # Re-add click-stripped '--' if any start with '-'
 439             if any(val.startswith("-") and val != "-" for val in value):
 440                 right_args.append("--")
 441             right_args.extend([shlex_quote(force_text(val)) for val in value])
 442             continue
 443 
 444         # Exclude one-off options (--upgrade/--upgrade-package/--rebuild/...)
 445         # or options that don't change compile behaviour (--verbose/--dry-run/...)
 446         if option_long_name in COMPILE_EXCLUDE_OPTIONS:
 447             continue
 448 
 449         # Skip options without a value
 450         if option.default is None and not value:
 451             continue
 452 
 453         # Skip options with a default value
 454         if option.default == value:
 455             continue
 456 
 457         # Use a file name for file-like objects
 458         if isinstance(value, LazyFile):
 459             value = value.name
 460 
 461         # Convert value to the list
 462         if not isinstance(value, (tuple, list)):
 463             value = [value]
 464 
 465         for val in value:
 466             # Flags don't have a value, thus add to args true or false option long name
 467             if option.is_flag:
 468                 # If there are false-options, choose an option name depending on a value
 469                 if option.secondary_opts:
 470                     # Get the latest false-option
 471                     secondary_option_long_name = option.secondary_opts[-1]
 472                     arg = option_long_name if val else secondary_option_long_name
 473                 # There are no false-options, use true-option
 474                 else:
 475                     arg = option_long_name
 476                 left_args.append(shlex_quote(arg))
 477             # Append to args the option with a value
 478             else:
 479                 if option.name == "pip_args":
 480                     # shlex_quote would produce functional but noisily quoted results,
 481                     # e.g. --pip-args='--cache-dir='"'"'/tmp/with spaces'"'"''
 482                     # Instead, we try to get more legible quoting via repr:
 483                     left_args.append(
 484                         "{option}={value}".format(
 485                             option=option_long_name, value=repr(fs_str(force_text(val)))
 486                         )
 487                     )
 488                 else:
 489                     left_args.append(
 490                         "{option}={value}".format(
 491                             option=option_long_name, value=shlex_quote(force_text(val))
 492                         )
 493                     )
 494 
 495     return " ".join(["pip-compile"] + sorted(left_args) + sorted(right_args))
