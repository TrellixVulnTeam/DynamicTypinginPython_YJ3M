  18 def do_import(module_path, subimport=None, old_path=None):
  19     old_path = old_path or module_path
  20     pip_path = os.environ.get("PIP_SHIMS_BASE_MODULE", "pip")
  21     prefixes = ["{}._internal".format(pip_path), pip_path]
  22     paths = [module_path, old_path]
  23     search_order = [
  24         "{0}.{1}".format(p, pth) for p in prefixes for pth in paths if pth is not None
  25     ]
  26     package = subimport if subimport else None
  27     for to_import in search_order:
  28         if not subimport:
  29             to_import, _, package = to_import.rpartition(".")
  30         try:
  31             imported = importlib.import_module(to_import)
  32         except ImportError:
  33             continue
  34         else:
  35             return getattr(imported, package)
