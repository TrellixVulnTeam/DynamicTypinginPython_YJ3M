  88     def _sort_key(self, ireq):
  89         return (not ireq.editable, str(ireq.req).lower())

  91     def write_header(self):
  92         if self.emit_header:
  93             yield comment("#")
  94             yield comment("# This file is autogenerated by pip-compile")
  95             yield comment("# To update, run:")
  96             yield comment("#")
  97             compile_command = os.environ.get(
  98                 "CUSTOM_COMPILE_COMMAND"
  99             ) or get_compile_command(self.click_ctx)
 100             yield comment("#    {}".format(compile_command))
 101             yield comment("#")

 103     def write_index_options(self):
 104         if self.emit_index:
 105             for index, index_url in enumerate(dedup(self.index_urls)):
 106                 if index_url.rstrip("/") == self.default_index_url:
 107                     continue
 108                 flag = "--index-url" if index == 0 else "--extra-index-url"
 109                 yield "{} {}".format(flag, index_url)
 110 
 111     def write_trusted_hosts(self):
 112         if self.emit_trusted_host:
 113             for trusted_host in dedup(self.trusted_hosts):
 114                 yield "--trusted-host {}".format(trusted_host)
 115 
 116     def write_format_controls(self):
 117         for nb in dedup(sorted(self.format_control.no_binary)):
 118             yield "--no-binary {}".format(nb)
 119         for ob in dedup(sorted(self.format_control.only_binary)):
 120             yield "--only-binary {}".format(ob)
 121 
 122     def write_find_links(self):
 123         if self.emit_find_links:
 124             for find_link in dedup(self.find_links):
 125                 yield "--find-links {}".format(find_link)
 126 
 127     def write_flags(self):
 128         emitted = False
 129         for line in chain(
 130             self.write_index_options(),
 131             self.write_find_links(),
 132             self.write_trusted_hosts(),
 133             self.write_format_controls(),
 134         ):
 135             emitted = True
 136             yield line
 137         if emitted:
 138             yield ""

 140     def _iter_lines(self, results, unsafe_requirements=None, markers=None, hashes=None):
 141         # default values
 142         unsafe_requirements = unsafe_requirements or []
 143         markers = markers or {}
 144         hashes = hashes or {}
 145 
 146         # Check for unhashed or unpinned packages if at least one package does have
 147         # hashes, which will trigger pip install's --require-hashes mode.
 148         warn_uninstallable = False
 149         has_hashes = hashes and any(hash for hash in hashes.values())
 150 
 151         yielded = False
 152 
 153         for line in self.write_header():
 154             yield line
 155             yielded = True
 156         for line in self.write_flags():
 157             yield line
 158             yielded = True
 159 
 160         unsafe_requirements = (
 161             {r for r in results if r.name in UNSAFE_PACKAGES}
 162             if not unsafe_requirements
 163             else unsafe_requirements
 164         )
 165         packages = {r for r in results if r.name not in UNSAFE_PACKAGES}
 166 
 167         if packages:
 168             packages = sorted(packages, key=self._sort_key)
 169             for ireq in packages:
 170                 if has_hashes and not hashes.get(ireq):
 171                     yield MESSAGE_UNHASHED_PACKAGE
 172                     warn_uninstallable = True
 173                 line = self._format_requirement(
 174                     ireq, markers.get(key_from_ireq(ireq)), hashes=hashes
 175                 )
 176                 yield line
 177             yielded = True
 178 
 179         if unsafe_requirements:
 180             unsafe_requirements = sorted(unsafe_requirements, key=self._sort_key)
 181             yield ""
 182             yielded = True
 183             if has_hashes and not self.allow_unsafe:
 184                 yield MESSAGE_UNSAFE_PACKAGES_UNPINNED
 185                 warn_uninstallable = True
 186             else:
 187                 yield MESSAGE_UNSAFE_PACKAGES
 188 
 189             for ireq in unsafe_requirements:
 190                 ireq_key = key_from_ireq(ireq)
 191                 if not self.allow_unsafe:
 192                     yield comment("# {}".format(ireq_key))
 193                 else:
 194                     line = self._format_requirement(
 195                         ireq, marker=markers.get(ireq_key), hashes=hashes
 196                     )
 197                     yield line
 198 
 199         # Yield even when there's no real content, so that blank files are written
 200         if not yielded:
 201             yield ""
 202 
 203         if warn_uninstallable:
 204             log.warning(MESSAGE_UNINSTALLABLE)
 205 
 206     def write(self, results, unsafe_requirements, markers, hashes):
 207 
 208         for line in self._iter_lines(results, unsafe_requirements, markers, hashes):
 209             log.info(line)
 210             if not self.dry_run:
 211                 self.dst_file.write(unstyle(line).encode("utf-8"))
 212                 self.dst_file.write(os.linesep.encode("utf-8"))

 214     def _format_requirement(self, ireq, marker=None, hashes=None):
 215         ireq_hashes = (hashes if hashes is not None else {}).get(ireq)
 216 
 217         line = format_requirement(ireq, marker=marker, hashes=ireq_hashes)
 218 
 219         if not self.annotate:
 220             return line
 221 
 222         # Annotate what packages or reqs-ins this package is required by
 223         required_by = set()
 224         if hasattr(ireq, "_source_ireqs"):
 225             required_by |= {
 226                 _comes_from_as_string(src_ireq)
 227                 for src_ireq in ireq._source_ireqs
 228                 if src_ireq.comes_from
 229             }
 230         elif ireq.comes_from:
 231             required_by.add(_comes_from_as_string(ireq))
 232         if required_by:
 233             annotation = ", ".join(sorted(required_by))
 234             line = "{:24}{}{}".format(
 235                 line,
 236                 " \\\n    " if ireq_hashes else "  ",
 237                 comment("# via " + annotation),
 238             )
 239         return line
