   5 class NoCandidateFound(PipToolsError):
   6     def __init__(self, ireq, candidates_tried, finder):
   7         self.ireq = ireq
   8         self.candidates_tried = candidates_tried
   9         self.finder = finder
  10 
  11     def __str__(self):
  12         versions = []
  13         pre_versions = []
  14 
  15         for candidate in sorted(self.candidates_tried):
  16             version = str(candidate.version)
  17             if candidate.version.is_prerelease:
  18                 pre_versions.append(version)
  19             else:
  20                 versions.append(version)
  21 
  22         lines = ["Could not find a version that matches {}".format(self.ireq)]
  23 
  24         if versions:
  25             lines.append("Tried: {}".format(", ".join(versions)))
  26 
  27         if pre_versions:
  28             if self.finder.allow_all_prereleases:
  29                 line = "Tried"
  30             else:
  31                 line = "Skipped"
  32 
  33             line += " pre-versions: {}".format(", ".join(pre_versions))
  34             lines.append(line)
  35 
  36         if versions or pre_versions:
  37             lines.append(
  38                 "There are incompatible versions in the resolved dependencies:"
  39             )
  40             source_ireqs = getattr(self.ireq, "_source_ireqs", [])
  41             lines.extend("  {}".format(ireq) for ireq in source_ireqs)
  42         else:
  43             lines.append("No versions found")
  44             lines.append(
  45                 "{} {} reachable?".format(
  46                     "Were" if len(self.finder.index_urls) > 1 else "Was",
  47                     " or ".join(self.finder.index_urls),
  48                 )
  49             )
  50         return "\n".join(lines)
