  93     """Synchronize virtual environment with requirements.txt."""
  94     if not src_files:
  95         if os.path.exists(DEFAULT_REQUIREMENTS_FILE):
  96             src_files = (DEFAULT_REQUIREMENTS_FILE,)
  97         else:
  98             msg = "No requirement files given and no {} found in the current directory"
  99             log.error(msg.format(DEFAULT_REQUIREMENTS_FILE))
 100             sys.exit(2)
 101 
 102     if any(src_file.endswith(".in") for src_file in src_files):
 103         msg = (
 104             "Some input files have the .in extension, which is most likely an error "
 105             "and can cause weird behaviour. You probably meant to use "
 106             "the corresponding *.txt file?"
 107         )
 108         if force:
 109             log.warning("WARNING: " + msg)
 110         else:
 111             log.error("ERROR: " + msg)
 112             sys.exit(2)
 113 
 114     install_command = InstallCommand()
 115     options, _ = install_command.parse_args([])
 116     session = install_command._build_session(options)
 117     finder = install_command._build_package_finder(options=options, session=session)
 118 
 119     # Parse requirements file. Note, all options inside requirements file
 120     # will be collected by the finder.
 121     requirements = flat_map(
 122         lambda src: parse_requirements(src, finder=finder, session=session), src_files
 123     )
 124 
 125     try:
 126         requirements = sync.merge(requirements, ignore_conflicts=force)
 127     except PipToolsError as e:
 128         log.error(str(e))
 129         sys.exit(2)
 130 
 131     installed_dists = get_installed_distributions(skip=[], user_only=user_only)
 132     to_install, to_uninstall = sync.diff(requirements, installed_dists)
 133 
 134     install_flags = _compose_install_flags(
 135         finder,
 136         no_index=no_index,
 137         index_url=index_url,
 138         extra_index_url=extra_index_url,
 139         trusted_host=trusted_host,
 140         find_links=find_links,
 141         user_only=user_only,
 142         cert=cert,
 143         client_cert=client_cert,
 144     ) + shlex.split(pip_args or "")
 145     sys.exit(
 146         sync.sync(
 147             to_install,
 148             to_uninstall,
 149             verbose=(not quiet),
 150             dry_run=dry_run,
 151             install_flags=install_flags,
 152             ask=ask,
 153         )
 154     )

 157 def _compose_install_flags(
 158     finder,
 159     no_index=False,
 160     index_url=None,
 161     extra_index_url=None,
 162     trusted_host=None,
 163     find_links=None,
 164     user_only=False,
 165     cert=None,
 166     client_cert=None,
 167 ):
 168     """
 169     Compose install flags with the given finder and CLI options.
 170     """
 171     result = []
 172 
 173     # Build --index-url/--extra-index-url/--no-index
 174     if no_index:
 175         result.append("--no-index")
 176     elif index_url:
 177         result.extend(["--index-url", index_url])
 178     elif finder.index_urls:
 179         finder_index_url = finder.index_urls[0]
 180         if finder_index_url != PyPIRepository.DEFAULT_INDEX_URL:
 181             result.extend(["--index-url", finder_index_url])
 182         for extra_index in finder.index_urls[1:]:
 183             result.extend(["--extra-index-url", extra_index])
 184     else:
 185         result.append("--no-index")
 186 
 187     for extra_index in extra_index_url or []:
 188         result.extend(["--extra-index-url", extra_index])
 189 
 190     # Build --trusted-hosts
 191     for host in itertools.chain(trusted_host or [], finder.trusted_hosts):
 192         result.extend(["--trusted-host", host])
 193 
 194     # Build --find-links
 195     for link in itertools.chain(find_links or [], finder.find_links):
 196         result.extend(["--find-links", link])
 197 
 198     # Build format controls --no-binary/--only-binary
 199     for format_control in ("no_binary", "only_binary"):
 200         formats = getattr(finder.format_control, format_control)
 201         if not formats:
 202             continue
 203         result.extend(
 204             ["--" + format_control.replace("_", "-"), ",".join(sorted(formats))]
 205         )
 206 
 207     if user_only:
 208         result.append("--user")
 209 
 210     if cert:
 211         result.extend(["--cert", cert])
 212 
 213     if client_cert:
 214         result.extend(["--client-cert", client_cert])
 215 
 216     return result
