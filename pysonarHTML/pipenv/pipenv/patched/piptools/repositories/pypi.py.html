  60 class HashCache(SafeFileCache):
  61     """Caches hashes of PyPI artifacts so we do not need to re-download them
  62 
  63     Hashes are only cached when the URL appears to contain a hash in it and the cache key includes
  64     the hash value returned from the server). This ought to avoid ssues where the location on the
  65     server changes."""
  66     def __init__(self, *args, **kwargs):
  67         session = kwargs.pop('session')
  68         self.session = session
  69         kwargs.setdefault('directory', os.path.join(CACHE_DIR, 'hash-cache'))
  70         super(HashCache, self).__init__(*args, **kwargs)
  71 
  72     def get_hash(self, location):
  73         # if there is no location hash (i.e., md5 / sha256 / etc) we on't want to store it
  74         hash_value = None
  75         vcs = VcsSupport()
  76         orig_scheme = location.scheme
  77         new_location = copy.deepcopy(location)
  78         if orig_scheme in vcs.all_schemes:
  79             new_location.url = new_location.url.split("+", 1)[-1]
  80         can_hash = new_location.hash
  81         if can_hash:
  82             # hash url WITH fragment
  83             hash_value = self.get(new_location.url)
  84         if not hash_value:
  85             hash_value = self._get_file_hash(new_location) if not new_location.url.startswith("ssh") else None
  86             hash_value = hash_value.encode('utf8') if hash_value else None
  87         if can_hash:
  88             self.set(new_location.url, hash_value)
  89         return hash_value.decode('utf8') if hash_value else None
  90 
  91     def _get_file_hash(self, location):
  92         h = hashlib.new(FAVORITE_HASH)
  93         with open_local_or_remote_file(location, self.session) as (fp, size):
  94             for chunk in iter(lambda: fp.read(8096), b""):
  95                 h.update(chunk)
  96         return ":".join([FAVORITE_HASH, h.hexdigest()])

 109     def __init__(self, pip_args, cache_dir=CACHE_DIR, session=None, build_isolation=False, use_json=False):

 114         # Use pip's parser for pip.conf management and defaults.

 117         self.command = InstallCommand()
 118         self.options, _ = self.command.parse_args(pip_args)

 120             self.options.build_isolation = build_isolation

 181     def find_best_match(self, ireq, prereleases=None):
 182         """
 183         Returns a Version object that indicates the best match for the given
 184         InstallRequirement according to the external repository.
 185         """
 186         if ireq.editable or is_url_requirement(ireq):
 187             return ireq  # return itself as the best match
 188 
 189         all_candidates = clean_requires_python(self.find_all_candidates(ireq.name))
 190         candidates_by_version = lookup_table(
 191             all_candidates, key=lambda c: c.version, unique=True
 192         )
 193         try:
 194             matching_versions = ireq.specifier.filter((candidate.version for candidate in all_candidates),
 195                                                       prereleases=prereleases)
 196         except TypeError:
 197             matching_versions = [candidate.version for candidate in all_candidates]
 198 
 199         # Reuses pip's internal candidate sort key to sort
 200         matching_candidates = [candidates_by_version[ver] for ver in matching_versions]
 201         if not matching_candidates:
 202             raise NoCandidateFound(ireq, all_candidates, self.finder)
 203 
 204         evaluator = self.finder.make_candidate_evaluator(ireq.name)
 205         best_candidate_result = evaluator.compute_best_candidate(matching_candidates)
 206         best_candidate = best_candidate_result.best_candidate
 207 
 208         # Turn the candidate into a pinned InstallRequirement
 209         return make_install_requirement(
 210             best_candidate.name,
 211             best_candidate.version,
 212             ireq.extras,
 213             ireq.markers,
 214             constraint=ireq.constraint,
 215         )

 231     def get_json_dependencies(self, ireq):
 232 
 233         if not (is_pinned_requirement(ireq)):
 234             raise TypeError('Expected pinned InstallRequirement, got {}'.format(ireq))
 235 
 236         def gen(ireq):
 237             if self.DEFAULT_INDEX_URL not in self.finder.index_urls:
 238                 return
 239 
 240             url = 'https://pypi.org/pypi/{0}/json'.format(ireq.req.name)
 241             releases = self.session.get(url).json()['releases']
 242 
 243             matches = [
 244                 r for r in releases
 245                 if '=={0}'.format(r) == str(ireq.req.specifier)
 246             ]
 247             if not matches:
 248                 return
 249 
 250             release_requires = self.session.get(
 251                 'https://pypi.org/pypi/{0}/{1}/json'.format(
 252                     ireq.req.name, matches[0],
 253                 ),
 254             ).json()
 255             try:
 256                 requires_dist = release_requires['info']['requires_dist']
 257             except KeyError:
 258                 return
 259 
 260             for requires in requires_dist:
 261                 i = InstallRequirement.from_line(requires)
 262                 if 'extra' not in repr(i.markers):
 263                     yield i
 264 
 265         try:
 266             if ireq not in self._json_dep_cache:
 267                 self._json_dep_cache[ireq] = [g for g in gen(ireq)]
 268 
 269             return set(self._json_dep_cache[ireq])
 270         except Exception:
 271             return set()

 273     def resolve_reqs(self, download_dir, ireq, wheel_cache):
 274         with get_requirement_tracker() as req_tracker, TempDirectory(
 275             kind="resolver"
 276         ) as temp_dir:
 277             preparer = self.command.make_requirement_preparer(
 278                 temp_build_dir=temp_dir,
 279                 options=self.options,
 280                 req_tracker=req_tracker,
 281                 session=self.session,
 282                 finder=self.finder,
 283                 use_user_site=False,
 284                 download_dir=download_dir,
 285                 wheel_download_dir=self._wheel_download_dir,
 286             )
 287 
 288             reqset = RequirementSet()
 289             ireq.is_direct = True
 290             reqset.add_requirement(ireq)
 291 
 292             resolver = self.command.make_resolver(
 293                 preparer=preparer,
 294                 finder=self.finder,
 295                 options=self.options,
 296                 wheel_cache=wheel_cache,
 297                 use_user_site=False,
 298                 ignore_installed=True,
 299                 ignore_requires_python=True,
 300                 force_reinstall=False,
 301                 upgrade_strategy="to-satisfy-only",
 302             )
 303             results = resolver._resolve_one(reqset, ireq)
 304 
 305             if PIP_VERSION[:2] <= (20, 0):
 306                 reqset.cleanup_files()
 307         results = set(results) if results else set()
 308 
 309         return results, ireq

 311     def get_legacy_dependencies(self, ireq):
 312         """
 313         Given a pinned, URL, or editable InstallRequirement, returns a set of
 314         dependencies (also InstallRequirements, but not necessarily pinned).
 315         They indicate the secondary dependencies for the given requirement.
 316         """
 317         if not (
 318             ireq.editable or is_url_requirement(ireq) or is_pinned_requirement(ireq)
 319         ):
 320             raise TypeError(
 321                 "Expected url, pinned or editable InstallRequirement, got {}".format(
 322                     ireq
 323                 )
 324             )
 325 
 326         if ireq not in self._dependencies_cache:
 327             if ireq.editable and (ireq.source_dir and os.path.exists(ireq.source_dir)):
 328                 # No download_dir for locally available editable requirements.
 329                 # If a download_dir is passed, pip will  unnecessarely
 330                 # archive the entire source directory
 331                 download_dir = None
 332             elif ireq.link and ireq.link.is_vcs:
 333                 # No download_dir for VCS sources.  This also works around pip
 334                 # using git-checkout-index, which gets rid of the .git dir.
 335                 download_dir = None
 336             else:
 337                 download_dir = self._download_dir
 338                 if not os.path.isdir(download_dir):
 339                     os.makedirs(download_dir)
 340             if not os.path.isdir(self._wheel_download_dir):
 341                 os.makedirs(self._wheel_download_dir)
 342 
 343             with global_tempdir_manager():
 344                 wheel_cache = WheelCache(self._cache_dir, self.options.format_control)
 345                 prev_tracker = os.environ.get("PIP_REQ_TRACKER")
 346                 try:
 347                     results, ireq = self.resolve_reqs(
 348                         download_dir, ireq, wheel_cache
 349                     )
 350                     self._dependencies_cache[ireq] = results
 351                 finally:
 352                     if "PIP_REQ_TRACKER" in os.environ:
 353                         if prev_tracker:
 354                             os.environ["PIP_REQ_TRACKER"] = prev_tracker
 355                         else:
 356                             del os.environ["PIP_REQ_TRACKER"]
 357 
 358                     if PIP_VERSION[:2] <= (20, 0):
 359                         wheel_cache.cleanup()
 360 
 361         return self._dependencies_cache[ireq]

 363     def get_hashes(self, ireq):
 364         """
 365         Given an InstallRequirement, return a set of hashes that represent all
 366         of the files for a given requirement. Unhashable requirements return an
 367         empty set. Unpinned requirements raise a TypeError.
 368         """
 369 
 370         if ireq.link:
 371             link = ireq.link
 372 
 373             if link.is_vcs or (link.is_file and link.is_existing_dir()):
 374                 # Return empty set for unhashable requirements.
 375                 # Unhashable logic modeled on pip's
 376                 # RequirementPreparer.prepare_linked_requirement
 377                 return set()
 378 
 379             if is_url_requirement(ireq):
 380                 # Directly hash URL requirements.
 381                 # URL requirements may have been previously downloaded and cached
 382                 # locally by self.resolve_reqs()
 383                 cached_path = os.path.join(self._download_dir, link.filename)
 384                 if os.path.exists(cached_path):
 385                     cached_link = Link(path_to_url(cached_path))
 386                 else:
 387                     cached_link = link
 388                 return {self._hash_cache._get_file_hash(cached_link)}
 389 
 390         if not is_pinned_requirement(ireq):
 391             raise TypeError("Expected pinned requirement, got {}".format(ireq))
 392 
 393         # We need to get all of the candidates that match our current version
 394         # pin, these will represent all of the files that could possibly
 395         # satisfy this constraint.
 396 
 397         result = {}
 398         with self.allow_all_links():
 399             matching_candidates = (
 400                 c for c in clean_requires_python(self.find_all_candidates(ireq.name))
 401                 if c.version in ireq.specifier
 402             )
 403             log.debug("  {}".format(ireq.name))
 404             result = {
 405                 h for h in
 406                 map(lambda c: self._hash_cache.get_hash(c.link), matching_candidates)
 407                 if h is not None
 408             }
 409         return result

 419     @contextmanager
 420     def allow_all_wheels(self):
 421         """
 422         Monkey patches pip.Wheel to allow wheels from all platforms and Python versions.
 423 
 424         This also saves the candidate cache and set a new one, or else the results from
 425         the previous non-patched calls will interfere.
 426         """
 427 
 428         def _wheel_supported(self, tags=None):
 429             # Ignore current platform. Support everything.
 430             return True
 431 
 432         def _wheel_support_index_min(self, tags=None):
 433             # All wheels are equal priority for sorting.
 434             return 0
 435 
 436         original_wheel_supported = Wheel.supported
 437         original_support_index_min = Wheel.support_index_min
 438         original_cache = self._available_candidates_cache
 439 
 440         Wheel.supported = _wheel_supported
 441         Wheel.support_index_min = _wheel_support_index_min
 442         self._available_candidates_cache = {}
 443 
 444         try:
 445             yield
 446         finally:
 447             Wheel.supported = original_wheel_supported
 448             Wheel.support_index_min = original_support_index_min
 449             self._available_candidates_cache = original_cache

 452 @contextmanager
 453 def open_local_or_remote_file(link, session):
 454     """
 455     Open local or remote file for reading.
 456 
 457     :type link: pip.index.Link
 458     :type session: requests.Session
 459     :raises ValueError: If link points to a local directory.
 460     :return: a context manager to a FileStream with the opened file-like object
 461     """
 462     url = link.url_without_fragment
 463 
 464     if link.is_file:
 465         # Local URL
 466         local_path = url_to_path(url)
 467         if os.path.isdir(local_path):
 468             raise ValueError("Cannot open directory for read: {}".format(url))
 469         else:
 470             st = os.stat(local_path)
 471             with open(local_path, "rb") as local_file:
 472                 yield FileStream(stream=local_file, size=st.st_size)
 473     else:
 474         # Remote URL
 475         headers = {"Accept-Encoding": "identity"}
 476         response = session.get(url, headers=headers, stream=True)
 477 
 478         # Content length must be int or None
 479         try:
 480             content_length = int(response.headers["content-length"])
 481         except (ValueError, KeyError, TypeError):
 482             content_length = None
 483 
 484         try:
 485             yield FileStream(stream=response.raw, size=content_length)
 486         finally:
 487             response.close()
