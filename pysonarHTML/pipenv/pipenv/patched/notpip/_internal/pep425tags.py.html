  29 def version_info_to_nodot(version_info):
  30     # type: (Tuple[int, ...]) -> str
  31     # Only use up to the first two numbers.
  32     return ''.join(map(str, version_info[:2]))

  35 def _mac_platforms(arch):
  36     # type: (str) -> List[str]
  37     match = _osx_arch_pat.match(arch)
  38     if match:
  39         name, major, minor, actual_arch = match.groups()
  40         mac_version = (int(major), int(minor))
  41         arches = [
  42             # Since we have always only checked that the platform starts
  43             # with "macosx", for backwards-compatibility we extract the
  44             # actual prefix provided by the user in case they provided
  45             # something like "macosxcustom_". It may be good to remove
  46             # this as undocumented or deprecate it in the future.
  47             '{}_{}'.format(name, arch[len('macosx_'):])
  48             for arch in mac_platforms(mac_version, actual_arch)
  49         ]
  50     else:
  51         # arch pattern didn't match (?!)
  52         arches = [arch]
  53     return arches

  56 def _custom_manylinux_platforms(arch):
  57     # type: (str) -> List[str]
  58     arches = [arch]
  59     arch_prefix, arch_sep, arch_suffix = arch.partition('_')
  60     if arch_prefix == 'manylinux2014':
  61         # manylinux1/manylinux2010 wheels run on most manylinux2014 systems
  62         # with the exception of wheels depending on ncurses. PEP 599 states
  63         # manylinux1/manylinux2010 wheels should be considered
  64         # manylinux2014 wheels:
  65         # https://www.python.org/dev/peps/pep-0599/#backwards-compatibility-with-manylinux2010-wheels
  66         if arch_suffix in {'i686', 'x86_64'}:
  67             arches.append('manylinux2010' + arch_sep + arch_suffix)
  68             arches.append('manylinux1' + arch_sep + arch_suffix)
  69     elif arch_prefix == 'manylinux2010':
  70         # manylinux1 wheels run on most manylinux2010 systems with the
  71         # exception of wheels depending on ncurses. PEP 571 states
  72         # manylinux1 wheels should be considered manylinux2010 wheels:
  73         # https://www.python.org/dev/peps/pep-0571/#backwards-compatibility-with-manylinux1-wheels
  74         arches.append('manylinux1' + arch_sep + arch_suffix)
  75     return arches

  78 def _get_custom_platforms(arch):
  79     # type: (str) -> List[str]
  80     arch_prefix, arch_sep, arch_suffix = arch.partition('_')
  81     if arch.startswith('macosx'):
  82         arches = _mac_platforms(arch)
  83     elif arch_prefix in ['manylinux2014', 'manylinux2010']:
  84         arches = _custom_manylinux_platforms(arch)
  85     else:
  86         arches = [arch]
  87     return arches

  90 def _get_python_version(version):
  91     # type: (str) -> PythonVersion
  92     if len(version) > 1:
  93         return int(version[0]), int(version[1:])
  94     else:
  95         return (int(version[0]),)

  98 def _get_custom_interpreter(implementation=None, version=None):
  99     # type: (Optional[str], Optional[str]) -> str
 100     if implementation is None:
 101         implementation = interpreter_name()
 102     if version is None:
 103         version = interpreter_version()
 104     return "{}{}".format(implementation, version)

 107 def get_supported(
 108     version=None,  # type: Optional[str]
 109     platform=None,  # type: Optional[str]
 110     impl=None,  # type: Optional[str]
 111     abi=None  # type: Optional[str]
 112 ):
 113     # type: (...) -> List[Tag]
 114     """Return a list of supported tags for each version specified in
 115     `versions`.
 116 
 117     :param version: a string version, of the form "33" or "32",
 118         or None. The version will be assumed to support our ABI.
 119     :param platform: specify the exact platform you want valid
 120         tags for, or None. If None, use the local system platform.
 121     :param impl: specify the exact implementation you want valid
 122         tags for, or None. If None, use the local interpreter impl.
 123     :param abi: specify the exact abi you want valid
 124         tags for, or None. If None, use the local interpreter abi.
 125     """
 126     supported = []  # type: List[Tag]
 127 
 128     python_version = None  # type: Optional[PythonVersion]
 129     if version is not None:
 130         python_version = _get_python_version(version)
 131 
 132     interpreter = _get_custom_interpreter(impl, version)
 133 
 134     abis = None  # type: Optional[List[str]]
 135     if abi is not None:
 136         abis = [abi]
 137 
 138     platforms = None  # type: Optional[List[str]]
 139     if platform is not None:
 140         platforms = _get_custom_platforms(platform)
 141 
 142     is_cpython = (impl or interpreter_name()) == "cp"
 143     if is_cpython:
 144         supported.extend(
 145             cpython_tags(
 146                 python_version=python_version,
 147                 abis=abis,
 148                 platforms=platforms,
 149             )
 150         )
 151     else:
 152         supported.extend(
 153             generic_tags(
 154                 interpreter=interpreter,
 155                 abis=abis,
 156                 platforms=platforms,
 157             )
 158         )
 159     supported.extend(
 160         compatible_tags(
 161             python_version=python_version,
 162             interpreter=interpreter,
 163             platforms=platforms,
 164         )
 165     )
 166 
 167     return supported
