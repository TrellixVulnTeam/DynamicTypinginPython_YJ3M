  27     def __init__(self, check_supported_wheels=True, ignore_compatibility=True):
  28         # type: (bool, bool) -> None
  29         """Create a RequirementSet.
  30         """
  31 
  32         self.requirements = OrderedDict()  # type: Dict[str, InstallRequirement]  # noqa: E501
  33         self.check_supported_wheels = check_supported_wheels
  34 
  35         self.unnamed_requirements = []  # type: List[InstallRequirement]
  36         self.successfully_downloaded = []  # type: List[InstallRequirement]
  37         self.reqs_to_cleanup = []  # type: List[InstallRequirement]

  42     def __str__(self):
  43         # type: () -> str
  44         requirements = sorted(
  45             (req for req in self.requirements.values() if not req.comes_from),
  46             key=lambda req: canonicalize_name(req.name),
  47         )
  48         return ' '.join(str(req.req) for req in requirements)
  49 
  50     def __repr__(self):
  51         # type: () -> str
  52         requirements = sorted(
  53             self.requirements.values(),
  54             key=lambda req: canonicalize_name(req.name),
  55         )
  56 
  57         format_string = '<{classname} object; {count} requirement(s): {reqs}>'
  58         return format_string.format(
  59             classname=self.__class__.__name__,
  60             count=len(requirements),
  61             reqs=', '.join(str(req.req) for req in requirements),
  62         )

  76     def add_requirement(
  77         self,
  78         install_req,  # type: InstallRequirement
  79         parent_req_name=None,  # type: Optional[str]
  80         extras_requested=None  # type: Optional[Iterable[str]]
  81     ):
  82         # type: (...) -> Tuple[List[InstallRequirement], Optional[InstallRequirement]]  # noqa: E501
  83         """Add install_req as a requirement to install.
  84 
  85         :param parent_req_name: The name of the requirement that needed this
  86             added. The name is used because when multiple unnamed requirements
  87             resolve to the same name, we could otherwise end up with dependency
  88             links that point outside the Requirements set. parent_req must
  89             already be added. Note that None implies that this is a user
  90             supplied requirement, vs an inferred one.
  91         :param extras_requested: an iterable of extras used to evaluate the
  92             environment markers.
  93         :return: Additional requirements to scan. That is either [] if
  94             the requirement is not applicable, or [install_req] if the
  95             requirement is applicable and has just been added.
  96         """
  97         # If the markers do not match, ignore this requirement.
  98         if not install_req.match_markers(extras_requested):
  99             logger.info(
 100                 "Ignoring %s: markers '%s' don't match your environment",
 101                 install_req.name, install_req.markers,
 102             )
 103             return [], None
 104 
 105         # If the wheel is not supported, raise an error.
 106         # Should check this after filtering out based on environment markers to
 107         # allow specifying different wheels based on the environment/OS, in a
 108         # single requirements file.
 109         if install_req.link and install_req.link.is_wheel:
 110             wheel = Wheel(install_req.link.filename)
 111             tags = pep425tags.get_supported()
 112             if (self.check_supported_wheels and not wheel.supported(tags)):
 113                 raise InstallationError(
 114                     "%s is not a supported wheel on this platform." %
 115                     wheel.filename
 116                 )
 117 
 118         # This next bit is really a sanity check.
 119         assert install_req.is_direct == (parent_req_name is None), (
 120             "a direct req shouldn't have a parent and also, "
 121             "a non direct req should have a parent"
 122         )
 123 
 124         # Unnamed requirements are scanned again and the requirement won't be
 125         # added as a dependency until after scanning.
 126         if not install_req.name:
 127             self.add_unnamed_requirement(install_req)
 128             return [install_req], None
 129 
 130         try:
 131             existing_req = self.get_requirement(install_req.name)
 132         except KeyError:
 133             existing_req = None
 134 
 135         has_conflicting_requirement = (
 136             parent_req_name is None and
 137             existing_req and
 138             not existing_req.constraint and
 139             existing_req.extras == install_req.extras and
 140             existing_req.req.specifier != install_req.req.specifier
 141         )
 142         if has_conflicting_requirement:
 143             raise InstallationError(
 144                 "Double requirement given: %s (already in %s, name=%r)"
 145                 % (install_req, existing_req, install_req.name)
 146             )
 147 
 148         # When no existing requirement exists, add the requirement as a
 149         # dependency and it will be scanned again after.
 150         if not existing_req:
 151             self.add_named_requirement(install_req)
 152             # We'd want to rescan this requirement later
 153             return [install_req], install_req
 154 
 155         # Assume there's no need to scan, and that we've already
 156         # encountered this for scanning.
 157         if install_req.constraint or not existing_req.constraint:
 158             return [], existing_req
 159 
 160         does_not_satisfy_constraint = (
 161             install_req.link and
 162             not (
 163                 existing_req.link and
 164                 install_req.link.path == existing_req.link.path
 165             )
 166         )
 167         if does_not_satisfy_constraint:
 168             self.reqs_to_cleanup.append(install_req)
 169             raise InstallationError(
 170                 "Could not satisfy constraints for '%s': "
 171                 "installation from path or url cannot be "
 172                 "constrained to a version" % install_req.name,
 173             )
 174         # If we're now installing a constraint, mark the existing
 175         # object for real installation.
 176         existing_req.constraint = False
 177         existing_req.extras = tuple(sorted(
 178             set(existing_req.extras) | set(install_req.extras)
 179         ))
 180         logger.debug(
 181             "Setting %s extras to: %s",
 182             existing_req, existing_req.extras,
 183         )
 184         # Return the existing requirement for addition to the parent and
 185         # scanning again.
 186         return [existing_req], existing_req
 187 
 188     def has_requirement(self, name):
 189         # type: (str) -> bool
 190         project_name = canonicalize_name(name)
 191 
 192         return (
 193             project_name in self.requirements and
 194             not self.requirements[project_name].constraint
 195         )
 196 
 197     def get_requirement(self, name):
 198         # type: (str) -> InstallRequirement
 199         project_name = canonicalize_name(name)
 200 
 201         if project_name in self.requirements:
 202             return self.requirements[project_name]
 203 
 204         pass

 206     def cleanup_files(self):
 207         # type: () -> None
 208         """Clean up files, remove builds."""
 209         logger.debug('Cleaning up...')
 210         with indent_log():
 211             for req in self.reqs_to_cleanup:
 212                 req.remove_temporary_source()
