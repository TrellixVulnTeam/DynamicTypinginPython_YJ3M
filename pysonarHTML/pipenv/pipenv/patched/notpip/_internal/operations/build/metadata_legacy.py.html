  22 def _find_egg_info(source_directory, is_editable):
  23     # type: (str, bool) -> str
  24     """Find an .egg-info in `source_directory`, based on `is_editable`.
  25     """
  26 
  27     def looks_like_virtual_env(path):
  28         # type: (str) -> bool
  29         return (
  30             os.path.lexists(os.path.join(path, 'bin', 'python')) or
  31             os.path.exists(os.path.join(path, 'Scripts', 'Python.exe'))
  32         )
  33 
  34     def locate_editable_egg_info(base):
  35         # type: (str) -> List[str]
  36         candidates = []  # type: List[str]
  37         for root, dirs, files in os.walk(base):
  38             for dir_ in vcs.dirnames:
  39                 if dir_ in dirs:
  40                     dirs.remove(dir_)
  41             # Iterate over a copy of ``dirs``, since mutating
  42             # a list while iterating over it can cause trouble.
  43             # (See https://github.com/pypa/pip/pull/462.)
  44             for dir_ in list(dirs):
  45                 if looks_like_virtual_env(os.path.join(root, dir_)):
  46                     dirs.remove(dir_)
  47                 # Also don't search through tests
  48                 elif dir_ == 'test' or dir_ == 'tests':
  49                     dirs.remove(dir_)
  50             candidates.extend(os.path.join(root, dir_) for dir_ in dirs)
  51         return [f for f in candidates if f.endswith('.egg-info')]
  52 
  53     def depth_of_directory(dir_):
  54         # type: (str) -> int
  55         return (
  56             dir_.count(os.path.sep) +
  57             (os.path.altsep and dir_.count(os.path.altsep) or 0)
  58         )
  59 
  60     base = source_directory
  61     if is_editable:
  62         filenames = locate_editable_egg_info(base)
  63     else:
  64         base = os.path.join(base, 'pip-egg-info')
  65         filenames = os.listdir(base)
  66 
  67     if not filenames:
  68         raise InstallationError(
  69             "Files/directories not found in {}".format(base)
  70         )
  71 
  72     # If we have more than one match, we pick the toplevel one.  This
  73     # can easily be the case if there is a dist folder which contains
  74     # an extracted tarball for testing purposes.
  75     if len(filenames) > 1:
  76         filenames.sort(key=depth_of_directory)
  77 
  78     return os.path.join(base, filenames[0])

  81 def generate_metadata(
  82     build_env,  # type: BuildEnvironment
  83     setup_py_path,  # type: str
  84     source_dir,  # type: str
  85     editable,  # type: bool
  86     isolated,  # type: bool
  87     details,  # type: str
  88 ):
  89     # type: (...) -> str
  90     """Generate metadata using setup.py-based defacto mechanisms.
  91 
  92     Returns the generated metadata directory.
  93     """
  94     logger.debug(
  95         'Running setup.py (path:%s) egg_info for package %s',
  96         setup_py_path, details,
  97     )
  98 
  99     egg_info_dir = None  # type: Optional[str]
 100     # For non-editable installs, don't put the .egg-info files at the root,
 101     # to avoid confusion due to the source code being considered an installed
 102     # egg.
 103     if not editable:
 104         egg_info_dir = os.path.join(source_dir, 'pip-egg-info')
 105         # setuptools complains if the target directory does not exist.
 106         ensure_dir(egg_info_dir)
 107 
 108     args = make_setuptools_egg_info_args(
 109         setup_py_path,
 110         egg_info_dir=egg_info_dir,
 111         no_user_config=isolated,
 112     )
 113 
 114     with build_env:
 115         call_subprocess(
 116             args,
 117             cwd=source_dir,
 118             command_desc='python setup.py egg_info',
 119         )
 120 
 121     # Return the .egg-info directory.
 122     return _find_egg_info(source_dir, editable)
