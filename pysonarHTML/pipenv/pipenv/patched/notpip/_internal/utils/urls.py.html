  31 def url_to_path(url):
  32     # type: (str) -> str
  33     """
  34     Convert a file: URL to a path.
  35     """
  36     assert url.startswith('file:'), (
  37         "You can only turn file: urls into filenames (not %r)" % url)
  38 
  39     _, netloc, path, _, _ = urllib_parse.urlsplit(url)
  40 
  41     if not netloc or netloc == 'localhost':
  42         # According to RFC 8089, same as empty authority.
  43         netloc = ''

  50             % url
  51         )

  53     path = urllib_request.url2pathname(netloc + path)
