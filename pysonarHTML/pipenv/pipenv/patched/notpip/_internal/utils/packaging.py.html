  23 def check_requires_python(requires_python, version_info):
  24     # type: (Optional[str], Tuple[int, ...]) -> bool
  25     """
  26     Check if the given Python version matches a "Requires-Python" specifier.
  27 
  28     :param version_info: A 3-tuple of ints representing a Python
  29         major-minor-micro version to check (e.g. `sys.version_info[:3]`).
  30 
  31     :return: `True` if the given Python version satisfies the requirement.
  32         Otherwise, return `False`.
  33 
  34     :raises InvalidSpecifier: If `requires_python` has an invalid format.
  35     """
  36     if requires_python is None:
  37         # The package provides no information
  38         return True
  39     requires_python_specifier = specifiers.SpecifierSet(requires_python)
  40 
  41     python_version = version.parse('{0}.{1}.{2}'.format(*sys.version_info[:3]))
  42     return python_version in requires_python_specifier

  45 def get_metadata(dist):
  46     # type: (Distribution) -> Message
  47     """
  48     :raises NoneMetadataError: if the distribution reports `has_metadata()`
  49         True but `get_metadata()` returns None.
  50     """
  51     metadata_name = 'METADATA'
  52     if (isinstance(dist, pkg_resources.DistInfoDistribution) and
  53             dist.has_metadata(metadata_name)):
  54         metadata = dist.get_metadata(metadata_name)
  55     elif dist.has_metadata('PKG-INFO'):
  56         metadata_name = 'PKG-INFO'
  57         metadata = dist.get_metadata(metadata_name)
  58     else:
  59         logger.warning("No metadata found in %s", display_path(dist.location))
  60         metadata = ''
  61 
  62     if metadata is None:
  63         raise NoneMetadataError(dist, metadata_name)
  64 
  65     feed_parser = FeedParser()
  66     # The following line errors out if with a "NoneType" TypeError if
  67     # passed metadata=None.
  68     feed_parser.feed(metadata)
  69     return feed_parser.close()

  72 def get_requires_python(dist):
  73     # type: (pkg_resources.Distribution) -> Optional[str]
  74     """
  75     Return the "Requires-Python" metadata for a distribution, or None
  76     if not present.
  77     """
  78     pkg_info_dict = get_metadata(dist)
  79     requires_python = pkg_info_dict.get('Requires-Python')
  80 
  81     if requires_python is not None:
  82         # Convert to a str to satisfy the type checker, since requires_python
  83         # can be a Header object.
  84         requires_python = str(requires_python)
  85 
  86     return requires_python

  89 def get_installer(dist):
  90     # type: (Distribution) -> str
  91     if dist.has_metadata('INSTALLER'):
  92         for line in dist.get_metadata_lines('INSTALLER'):
  93             if line.strip():
  94                 return line.strip()
  95     return ''
