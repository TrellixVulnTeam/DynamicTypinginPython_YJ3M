  82 class TempDirectory(object):
  83     """Helper class that owns and cleans up a temporary directory.
  84 
  85     This class can be used as a context manager or as an OO representation of a
  86     temporary directory.
  87 
  88     Attributes:
  89         path
  90             Location to the created temporary directory
  91         delete
  92             Whether the directory should be deleted when exiting
  93             (when used as a contextmanager)
  94 
  95     Methods:
  96         cleanup()
  97             Deletes the temporary directory
  98 
  99     When used as a context manager, if the delete attribute is True, on
 100     exiting the context the temporary directory is deleted.
 101     """
 102 
 103     def __init__(
 104         self,
 105         path=None,    # type: Optional[str]
 106         delete=None,  # type: Optional[bool]
 107         kind="temp",  # type: str
 108         globally_managed=False,  # type: bool
 109     ):
 110         super(TempDirectory, self).__init__()
 111 
 112         # If we were given an explicit directory, resolve delete option now.
 113         # Otherwise we wait until cleanup and see what tempdir_registry says.
 114         if path is not None and delete is None:
 115             delete = False
 116 
 117         if path is None:
 118             path = self._create(kind)
 119 
 120         self._path = path
 121         self._deleted = False
 122         self.delete = delete
 123         self.kind = kind
 124         self._finalizer = None
 125         if self._path:
 126             self._register_finalizer()
 127 
 128         if globally_managed:
 129             assert _tempdir_manager is not None
 130             _tempdir_manager.enter_context(self)
 131 
 132     def _register_finalizer(self):
 133         if self.delete and self._path:
 134             self._finalizer = finalize(
 135                 self,
 136                 self._cleanup,
 137                 self._path,
 138                 warn_message = None
 139             )
 140         else:
 141             self._finalizer = None
 142 
 143     @property
 144     def path(self):
 145         # type: () -> str
 146         assert not self._deleted, (
 147             "Attempted to access deleted path: {}".format(self._path)
 148         )
 149         return self._path
 150 
 151     def __repr__(self):
 152         # type: () -> str
 153         return "<{} {!r}>".format(self.__class__.__name__, self.path)
 154 
 155     def __enter__(self):
 156         # type: (_T) -> _T
 157         return self
 158 
 159     def __exit__(self, exc, value, tb):
 160         # type: (Any, Any, Any) -> None
 161         if self.delete is not None:
 162             delete = self.delete
 163         elif _tempdir_registry:
 164             delete = _tempdir_registry.get_delete(self.kind)
 165         else:
 166             delete = True
 167 
 168         if delete:
 169             self.cleanup()
 170 
 171     def _create(self, kind):
 172         # type: (str) -> str
 173         """Create a temporary directory and store its path in self.path
 174         """
 175         # We realpath here because some systems have their default tmpdir
 176         # symlinked to another directory.  This tends to confuse build
 177         # scripts, so we canonicalize the path by traversing potential
 178         # symlinks here.
 179         path = os.path.realpath(
 180             tempfile.mkdtemp(prefix="pip-{}-".format(kind))
 181         )
 182         logger.debug("Created temporary directory: {}".format(path))
 183         return path
 184 
 185     @classmethod
 186     def _cleanup(cls, name, warn_message=None):
 187         if not os.path.exists(name):
 188             return
 189         try:
 190             rmtree(name)
 191         except OSError:
 192             pass
 193         else:
 194             if warn_message:
 195                 warnings.warn(warn_message, ResourceWarning)
 196 
 197     def cleanup(self):
 198         # type: () -> None
 199         """Remove the temporary directory created and reset state
 200         """
 201         if getattr(self._finalizer, "detach", None) and self._finalizer.detach():
 202             if os.path.exists(self._path):
 203                 self._deleted = True
 204                 try:
 205                     rmtree(self._path)
 206                 except OSError:
 207                     pass

 210 class AdjacentTempDirectory(TempDirectory):
 211     """Helper class that creates a temporary directory adjacent to a real one.
 212 
 213     Attributes:
 214         original
 215             The original directory to create a temp directory for.
 216         path
 217             After calling create() or entering, contains the full
 218             path to the temporary directory.
 219         delete
 220             Whether the directory should be deleted when exiting
 221             (when used as a contextmanager)
 222 
 223     """
 224     # The characters that may be used to name the temp directory
 225     # We always prepend a ~ and then rotate through these until
 226     # a usable name is found.
 227     # pkg_resources raises a different error for .dist-info folder
 228     # with leading '-' and invalid metadata
 229     LEADING_CHARS = "-~.=%0123456789"
 230 
 231     def __init__(self, original, delete=None):
 232         # type: (str, Optional[bool]) -> None
 233         self.original = original.rstrip('/\\')
 234         super(AdjacentTempDirectory, self).__init__(delete=delete)
 235 
 236     @classmethod
 237     def _generate_names(cls, name):
 238         # type: (str) -> Iterator[str]
 239         """Generates a series of temporary names.
 240 
 241         The algorithm replaces the leading characters in the name
 242         with ones that are valid filesystem characters, but are not
 243         valid package names (for both Python and pip definitions of
 244         package).
 245         """
 246         for i in range(1, len(name)):
 247             for candidate in itertools.combinations_with_replacement(
 248                     cls.LEADING_CHARS, i - 1):
 249                 new_name = '~' + ''.join(candidate) + name[i:]
 250                 if new_name != name:
 251                     yield new_name
 252 
 253         # If we make it this far, we will have to make a longer name
 254         for i in range(len(cls.LEADING_CHARS)):
 255             for candidate in itertools.combinations_with_replacement(
 256                     cls.LEADING_CHARS, i):
 257                 new_name = '~' + ''.join(candidate) + name
 258                 if new_name != name:
 259                     yield new_name
 260 
 261     def _create(self, kind):
 262         # type: (str) -> str
 263         root, name = os.path.split(self.original)
 264         for candidate in self._generate_names(name):
 265             path = os.path.join(root, candidate)
 266             try:
 267                 os.mkdir(path)
 268             except OSError as ex:
 269                 # Continue if the name exists already
 270                 if ex.errno != errno.EEXIST:
 271                     raise
 272             else:
 273                 path = os.path.realpath(path)
 274                 break
 275         else:
 276             # Final fallback on the default behavior.
 277             path = os.path.realpath(
 278                 tempfile.mkdtemp(prefix="pip-{}-".format(kind))
 279             )
 280 
 281         logger.debug("Created temporary directory: {}".format(path))
 282         return path
