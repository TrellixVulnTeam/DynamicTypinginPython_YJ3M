  61 def _check_link_requires_python(
  62     link,  # type: Link
  63     version_info,  # type: Tuple[int, int, int]
  64     ignore_requires_python=False,  # type: bool
  65 ):
  66     # type: (...) -> bool
  67     """
  68     Return whether the given Python version is compatible with a link's
  69     "Requires-Python" value.
  70 
  71     :param version_info: A 3-tuple of ints representing the Python
  72         major-minor-micro version to check.
  73     :param ignore_requires_python: Whether to ignore the "Requires-Python"
  74         value if the given Python version isn't compatible.
  75     """
  76     try:
  77         is_compatible = check_requires_python(
  78             link.requires_python, version_info=version_info,
  79         )
  80     except specifiers.InvalidSpecifier:
  81         logger.debug(
  82             "Ignoring invalid Requires-Python (%r) for link: %s",
  83             link.requires_python, link,
  84         )
  85     else:
  86         if not is_compatible:
  87             version = '.'.join(map(str, version_info))
  88             if not ignore_requires_python:
  89                 logger.debug(
  90                     'Link requires a different Python (%s not in: %r): %s',
  91                     version, link.requires_python, link,
  92                 )
  93                 return False
  94 
  95             logger.debug(
  96                 'Ignoring failed Requires-Python check (%s not in: %r) '
  97                 'for link: %s',
  98                 version, link.requires_python, link,
  99             )
 100 
 101     return True

 104 class LinkEvaluator(object):
 105 
 106     """
 107     Responsible for evaluating links for a particular project.
 108     """
 109 
 110     _py_version_re = re.compile(r'-py([123]\.?[0-9]?)$')
 111 
 112     # Don't include an allow_yanked default value to make sure each call
 113     # site considers whether yanked releases are allowed. This also causes
 114     # that decision to be made explicit in the calling code, which helps
 115     # people when reading the code.
 116     def __init__(
 117         self,
 118         project_name,    # type: str
 119         canonical_name,  # type: str
 120         formats,         # type: FrozenSet[str]
 121         target_python,   # type: TargetPython
 122         allow_yanked,    # type: bool
 123         ignore_requires_python=None,  # type: Optional[bool]
 124         ignore_compatibility=None,  # type: Optional[bool]
 125     ):
 126         # type: (...) -> None
 127         """
 128         :param project_name: The user supplied package name.
 129         :param canonical_name: The canonical package name.
 130         :param formats: The formats allowed for this package. Should be a set
 131             with 'binary' or 'source' or both in it.
 132         :param target_python: The target Python interpreter to use when
 133             evaluating link compatibility. This is used, for example, to
 134             check wheel compatibility, as well as when checking the Python
 135             version, e.g. the Python version embedded in a link filename
 136             (or egg fragment) and against an HTML link's optional PEP 503
 137             "data-requires-python" attribute.
 138         :param allow_yanked: Whether files marked as yanked (in the sense
 139             of PEP 592) are permitted to be candidates for install.
 140         :param ignore_requires_python: Whether to ignore incompatible
 141             PEP 503 "data-requires-python" values in HTML links. Defaults
 142             to False.
 143         :param Optional[bool] ignore_compatibility: Whether to ignore
 144             compatibility of python versions and allow all versions of packages.
 145         """
 146         if ignore_requires_python is None:
 147             ignore_requires_python = False
 148         if ignore_compatibility is None:
 149             ignore_compatibility = True
 150 
 151         self._allow_yanked = allow_yanked
 152         self._canonical_name = canonical_name
 153         self._ignore_requires_python = ignore_requires_python
 154         self._formats = formats
 155         self._target_python = target_python
 156         self._ignore_compatibility = ignore_compatibility
 157 
 158         self.project_name = project_name
 159 
 160     def evaluate_link(self, link):
 161         # type: (Link) -> Tuple[bool, Optional[Text]]
 162         """
 163         Determine whether a link is a candidate for installation.
 164 
 165         :return: A tuple (is_candidate, result), where `result` is (1) a
 166             version string if `is_candidate` is True, and (2) if
 167             `is_candidate` is False, an optional string to log the reason
 168             the link fails to qualify.
 169         """
 170         version = None
 171         if link.is_yanked and not self._allow_yanked:
 172             reason = link.yanked_reason or '<none given>'
 173             # Mark this as a unicode string to prevent "UnicodeEncodeError:
 174             # 'ascii' codec can't encode character" in Python 2 when
 175             # the reason contains non-ascii characters.
 176             return (False, u'yanked for reason: {}'.format(reason))
 177 
 178         if link.egg_fragment:
 179             egg_info = link.egg_fragment
 180             ext = link.ext
 181         else:
 182             egg_info, ext = link.splitext()
 183             if not ext:
 184                 return (False, 'not a file')
 185             if ext not in SUPPORTED_EXTENSIONS:
 186                 return (False, 'unsupported archive format: %s' % ext)
 187             if "binary" not in self._formats and ext == WHEEL_EXTENSION and not self._ignore_compatibility:
 188                 reason = 'No binaries permitted for %s' % self.project_name
 189                 return (False, reason)
 190             if "macosx10" in link.path and ext == '.zip' and not self._ignore_compatibility:
 191                 return (False, 'macosx10 one')
 192             if ext == WHEEL_EXTENSION:
 193                 try:
 194                     wheel = Wheel(link.filename)
 195                 except InvalidWheelFilename:
 196                     return (False, 'invalid wheel filename')
 197                 if canonicalize_name(wheel.name) != self._canonical_name:
 198                     reason = 'wrong project name (not %s)' % self.project_name
 199                     return (False, reason)
 200 
 201                 supported_tags = self._target_python.get_tags()
 202                 if not wheel.supported(supported_tags) and not self._ignore_compatibility:
 203                     # Include the wheel's tags in the reason string to
 204                     # simplify troubleshooting compatibility issues.
 205                     file_tags = wheel.get_formatted_file_tags()
 206                     reason = (
 207                         "none of the wheel's tags match: {}".format(
 208                             ', '.join(file_tags)
 209                         )
 210                     )
 211                     return (False, reason)
 212 
 213                 version = wheel.version
 214 
 215         # This should be up by the self.ok_binary check, but see issue 2700.
 216         if "source" not in self._formats and ext != WHEEL_EXTENSION:
 217             return (False, 'No sources permitted for %s' % self.project_name)
 218 
 219         if not version:
 220             version = _extract_version_from_fragment(
 221                 egg_info, self._canonical_name,
 222             )
 223         if not version:
 224             return (
 225                 False, 'Missing project version for %s' % self.project_name,
 226             )
 227 
 228         match = self._py_version_re.search(version)
 229         if match:
 230             version = version[:match.start()]
 231             py_version = match.group(1)
 232             if py_version != self._target_python.py_version:
 233                 return (False, 'Python version is incorrect')
 234 
 235         supports_python = _check_link_requires_python(
 236             link, version_info=self._target_python.py_version_info,
 237             ignore_requires_python=self._ignore_requires_python,
 238         )
 239         if not supports_python and not self._ignore_compatibility:
 240             # Return None for the reason text to suppress calling
 241             # _log_skipped_link().
 242             return (False, None)
 243 
 244         logger.debug('Found link %s, version: %s', link, version)
 245 
 246         return (True, version)

 249 def filter_unallowed_hashes(
 250     candidates,    # type: List[InstallationCandidate]
 251     hashes,        # type: Hashes
 252     project_name,  # type: str
 253 ):
 254     # type: (...) -> List[InstallationCandidate]
 255     """
 256     Filter out candidates whose hashes aren't allowed, and return a new
 257     list of candidates.
 258 
 259     If at least one candidate has an allowed hash, then all candidates with
 260     either an allowed hash or no hash specified are returned.  Otherwise,
 261     the given candidates are returned.
 262 
 263     Including the candidates with no hash specified when there is a match
 264     allows a warning to be logged if there is a more preferred candidate
 265     with no hash specified.  Returning all candidates in the case of no
 266     matches lets pip report the hash of the candidate that would otherwise
 267     have been installed (e.g. permitting the user to more easily update
 268     their requirements file with the desired hash).
 269     """
 270     if not hashes:
 271         logger.debug(
 272             'Given no hashes to check %s links for project %r: '
 273             'discarding no candidates',
 274             len(candidates),
 275             project_name,
 276         )
 277         # Make sure we're not returning back the given value.
 278         return list(candidates)
 279 
 280     matches_or_no_digest = []
 281     # Collect the non-matches for logging purposes.
 282     non_matches = []
 283     match_count = 0
 284     for candidate in candidates:
 285         link = candidate.link
 286         if not link.has_hash:
 287             pass
 288         elif link.is_hash_allowed(hashes=hashes):
 289             match_count += 1
 290         else:
 291             non_matches.append(candidate)
 292             continue
 293 
 294         matches_or_no_digest.append(candidate)
 295 
 296     if match_count:
 297         filtered = matches_or_no_digest
 298     else:
 299         # Make sure we're not returning back the given value.
 300         filtered = list(candidates)
 301 
 302     if len(filtered) == len(candidates):
 303         discard_message = 'discarding no candidates'
 304     else:
 305         discard_message = 'discarding {} non-matches:\n  {}'.format(
 306             len(non_matches),
 307             '\n  '.join(str(candidate.link) for candidate in non_matches)
 308         )
 309 
 310     logger.debug(
 311         'Checked %s links for project %r against %s hashes '
 312         '(%s matches, %s no digest): %s',
 313         len(candidates),
 314         project_name,
 315         hashes.digest_count,
 316         match_count,
 317         len(matches_or_no_digest) - match_count,
 318         discard_message
 319     )
 320 
 321     return filtered

 344 class BestCandidateResult(object):
 345     """A collection of candidates, returned by `PackageFinder.find_best_candidate`.

 347     This class is only intended to be instantiated by CandidateEvaluator's
 348     `compute_best_candidate()` method.

 351     def __init__(
 352         self,

 356     ):
 357         # type: (...) -> None
 358         """

 360         :param applicable_candidates: The applicable candidates.

 374         self.best_candidate = best_candidate

 389 class CandidateEvaluator(object):
 390 
 391     """
 392     Responsible for filtering and sorting candidates for installation based
 393     on what tags are valid.
 394     """
 395 
 396     @classmethod
 397     def create(
 398         cls,
 399         project_name,         # type: str
 400         target_python=None,   # type: Optional[TargetPython]
 401         prefer_binary=False,  # type: bool
 402         allow_all_prereleases=False,  # type: bool
 403         specifier=None,       # type: Optional[specifiers.BaseSpecifier]
 404         hashes=None,          # type: Optional[Hashes]
 405     ):
 406         # type: (...) -> CandidateEvaluator
 407         """Create a CandidateEvaluator object.
 408 
 409         :param target_python: The target Python interpreter to use when
 410             checking compatibility. If None (the default), a TargetPython
 411             object will be constructed from the running Python.
 412         :param specifier: An optional object implementing `filter`
 413             (e.g. `packaging.specifiers.SpecifierSet`) to filter applicable
 414             versions.
 415         :param hashes: An optional collection of allowed hashes.
 416         """
 417         if target_python is None:
 418             target_python = TargetPython()
 419         if specifier is None:
 420             specifier = specifiers.SpecifierSet()
 421 
 422         supported_tags = target_python.get_tags()
 423 
 424         return cls(
 425             project_name=project_name,
 426             supported_tags=supported_tags,
 427             specifier=specifier,
 428             prefer_binary=prefer_binary,
 429             allow_all_prereleases=allow_all_prereleases,
 430             hashes=hashes,
 431         )
 432 
 433     def __init__(
 434         self,
 435         project_name,         # type: str
 436         supported_tags,       # type: List[Tag]
 437         specifier,            # type: specifiers.BaseSpecifier
 438         prefer_binary=False,  # type: bool
 439         allow_all_prereleases=False,  # type: bool
 440         hashes=None,                  # type: Optional[Hashes]
 441     ):
 442         # type: (...) -> None
 443         """
 444         :param supported_tags: The PEP 425 tags supported by the target
 445             Python in order of preference (most preferred first).
 446         """
 447         self._allow_all_prereleases = allow_all_prereleases
 448         self._hashes = hashes
 449         self._prefer_binary = prefer_binary
 450         self._project_name = project_name
 451         self._specifier = specifier
 452         self._supported_tags = supported_tags
 453 
 454     def get_applicable_candidates(
 455         self,
 456         candidates,  # type: List[InstallationCandidate]
 457     ):
 458         # type: (...) -> List[InstallationCandidate]
 459         """
 460         Return the applicable candidates from a list of candidates.
 461         """
 462         # Using None infers from the specifier instead.
 463         allow_prereleases = self._allow_all_prereleases or None
 464         specifier = self._specifier
 465         versions = {
 466             str(v) for v in specifier.filter(
 467                 # We turn the version object into a str here because otherwise
 468                 # when we're debundled but setuptools isn't, Python will see
 469                 # packaging.version.Version and
 470                 # pkg_resources._vendor.packaging.version.Version as different
 471                 # types. This way we'll use a str as a common data interchange
 472                 # format. If we stop using the pkg_resources provided specifier
 473                 # and start using our own, we can drop the cast to str().
 474                 (str(c.version) for c in candidates),
 475                 prereleases=allow_prereleases,
 476             )
 477         }
 478 
 479         # Again, converting version to str to deal with debundling.
 480         applicable_candidates = [
 481             c for c in candidates if str(c.version) in versions
 482         ]
 483 
 484         filtered_applicable_candidates = filter_unallowed_hashes(
 485             candidates=applicable_candidates,
 486             hashes=self._hashes,
 487             project_name=self._project_name,
 488         )
 489 
 490         return sorted(filtered_applicable_candidates, key=self._sort_key)
 491 
 492     def _sort_key(self, candidate, ignore_compatibility=True):
 493         # type: (InstallationCandidate, bool) -> CandidateSortingKey
 494         """
 495         Function to pass as the `key` argument to a call to sorted() to sort
 496         InstallationCandidates by preference.
 497 
 498         Returns a tuple such that tuples sorting as greater using Python's
 499         default comparison operator are more preferred.
 500 
 501         The preference is as follows:
 502 
 503         First and foremost, candidates with allowed (matching) hashes are
 504         always preferred over candidates without matching hashes. This is
 505         because e.g. if the only candidate with an allowed hash is yanked,
 506         we still want to use that candidate.
 507 
 508         Second, excepting hash considerations, candidates that have been
 509         yanked (in the sense of PEP 592) are always less preferred than
 510         candidates that haven't been yanked. Then:
 511 
 512         If not finding wheels, they are sorted by version only.
 513         If finding wheels, then the sort order is by version, then:
 514           1. existing installs
 515           2. wheels ordered via Wheel.support_index_min(self._supported_tags)
 516           3. source archives
 517         If prefer_binary was set, then all wheels are sorted above sources.
 518 
 519         Note: it was considered to embed this logic into the Link
 520               comparison operators, but then different sdist links
 521               with the same version, would have to be considered equal
 522         """
 523         valid_tags = self._supported_tags
 524         support_num = len(valid_tags)
 525         build_tag = ()  # type: BuildTag
 526         binary_preference = 0
 527         link = candidate.link
 528         if link.is_wheel:
 529             # can raise InvalidWheelFilename
 530             wheel = Wheel(link.filename)
 531             if not wheel.supported(valid_tags) and not ignore_compatibility:
 532                 raise UnsupportedWheel(
 533                     "%s is not a supported wheel for this platform. It "
 534                     "can't be sorted." % wheel.filename
 535                 )
 536             if self._prefer_binary:
 537                 binary_preference = 1
 538             tags = valid_tags
 539             try:
 540                 pri = -(wheel.support_index_min(tags=tags))
 541             except TypeError:
 542                 pri = -(support_num)
 543             if wheel.build_tag is not None:
 544                 match = re.match(r'^(\d+)(.*)$', wheel.build_tag)
 545                 build_tag_groups = match.groups()
 546                 build_tag = (int(build_tag_groups[0]), build_tag_groups[1])
 547         else:  # sdist
 548             pri = -(support_num)
 549         has_allowed_hash = int(link.is_hash_allowed(self._hashes))
 550         yank_value = -1 * int(link.is_yanked)  # -1 for yanked.
 551         return (
 552             has_allowed_hash, yank_value, binary_preference, candidate.version,
 553             build_tag, pri,
 554         )
 555 
 556     def sort_best_candidate(
 557         self,
 558         candidates,    # type: List[InstallationCandidate]
 559     ):
 560         # type: (...) -> Optional[InstallationCandidate]
 561         """
 562         Return the best candidate per the instance's sort order, or None if
 563         no candidate is acceptable.
 564         """
 565         if not candidates:
 566             return None
 567 
 568         best_candidate = max(candidates, key=self._sort_key)
 569 
 570         # Log a warning per PEP 592 if necessary before returning.
 571         link = best_candidate.link
 572         if link.is_yanked:
 573             reason = link.yanked_reason or '<none given>'
 574             msg = (
 575                 # Mark this as a unicode string to prevent
 576                 # "UnicodeEncodeError: 'ascii' codec can't encode character"
 577                 # in Python 2 when the reason contains non-ascii characters.
 578                 u'The candidate selected for download or install is a '
 579                 'yanked version: {candidate}\n'
 580                 'Reason for being yanked: {reason}'
 581             ).format(candidate=best_candidate, reason=reason)
 582             logger.warning(msg)
 583 
 584         return best_candidate
 585 
 586     def compute_best_candidate(
 587         self,
 588         candidates,      # type: List[InstallationCandidate]
 589     ):
 590         # type: (...) -> BestCandidateResult
 591         """
 592         Compute and return a `BestCandidateResult` instance.
 593         """
 594         applicable_candidates = self.get_applicable_candidates(candidates)
 595 
 596         best_candidate = self.sort_best_candidate(applicable_candidates)
 597 
 598         return BestCandidateResult(
 599             candidates,
 600             applicable_candidates=applicable_candidates,
 601             best_candidate=best_candidate,
 602         )

 605 class PackageFinder(object):
 606     """This finds packages.
 607 
 608     This is meant to match easy_install's technique for looking for
 609     packages, by reading pages and looking for appropriate links.
 610     """
 611 
 612     def __init__(
 613         self,
 614         link_collector,       # type: LinkCollector
 615         target_python,        # type: TargetPython
 616         allow_yanked,         # type: bool
 617         format_control=None,  # type: Optional[FormatControl]
 618         candidate_prefs=None,         # type: CandidatePreferences
 619         ignore_requires_python=None,  # type: Optional[bool]
 620         ignore_compatibility=None,  # type: Optional[bool]
 621     ):
 622         # type: (...) -> None
 623         """
 624         This constructor is primarily meant to be used by the create() class
 625         method and from tests.
 626 
 627         :param format_control: A FormatControl object, used to control
 628             the selection of source packages / binary packages when consulting
 629             the index and links.
 630         :param candidate_prefs: Options to use when creating a
 631             CandidateEvaluator object.
 632         """
 633         if candidate_prefs is None:
 634             candidate_prefs = CandidatePreferences()
 635         if ignore_compatibility is None:
 636             ignore_compatibility = False
 637 
 638         format_control = format_control or FormatControl(set(), set())
 639 
 640         self._allow_yanked = allow_yanked

 759     def _sort_links(self, links):
 760         # type: (Iterable[Link]) -> List[Link]
 761         """
 762         Returns elements of links in order, non-egg links first, egg links
 763         second, while eliminating duplicates
 764         """
 765         eggs, no_eggs = [], []
 766         seen = set()  # type: Set[Link]
 767         for link in links:
 768             if link not in seen:
 769                 seen.add(link)
 770                 if link.egg_fragment:
 771                     eggs.append(link)
 772                 else:
 773                     no_eggs.append(link)
 774         return no_eggs + eggs

 776     def _log_skipped_link(self, link, reason):
 777         # type: (Link, Text) -> None

 787     def get_install_candidate(self, link_evaluator, link):
 788         # type: (LinkEvaluator, Link) -> Optional[InstallationCandidate]
 789         """
 790         If the link is a candidate for install, convert it to an
 791         InstallationCandidate and return it. Otherwise, return None.
 792         """
 793         is_candidate, result = link_evaluator.evaluate_link(link)
 794         if not is_candidate:
 795             if result:
 796                 self._log_skipped_link(link, reason=result)
 797             return None
 798 
 799         return InstallationCandidate(
 800             name=link_evaluator.project_name,
 801             link=link,
 802             # Convert the Text result to str since InstallationCandidate
 803             # accepts str.
 804             version=str(result),
 805             requires_python=getattr(link, "requires_python", None)
 806         )

 810         """

 840     def find_all_candidates(self, project_name):
 841         # type: (str) -> List[InstallationCandidate]
 842         """Find all available InstallationCandidate for project_name
 843 
 844         This checks index_urls and find_links.
 845         All versions found are returned as an InstallationCandidate list.
 846 
 847         See LinkEvaluator.evaluate_link() for details on which files
 848         are accepted.
 849         """
 850         collected_links = self._link_collector.collect_links(project_name)
 851 
 852         link_evaluator = self.make_link_evaluator(project_name)
 853 
 854         find_links_versions = self.evaluate_links(
 855             link_evaluator,
 856             links=collected_links.find_links,
 857         )
 858 
 859         page_versions = []
 860         for project_url in collected_links.project_urls:
 861             package_links = self.process_project_url(
 862                 project_url, link_evaluator=link_evaluator,
 863             )
 864             page_versions.extend(package_links)
 865 
 866         file_versions = self.evaluate_links(
 867             link_evaluator,
 868             links=collected_links.files,
 869         )
 870         if file_versions:
 871             file_versions.sort(reverse=True)
 872             logger.debug(
 873                 'Local files found: %s',
 874                 ', '.join([
 875                     url_to_path(candidate.link.url)
 876                     for candidate in file_versions
 877                 ])
 878             )
 879 
 880         # This is an intentional priority ordering
 881         return file_versions + find_links_versions + page_versions

 925     def find_requirement(self, req, upgrade):
 926         # type: (InstallRequirement, bool) -> Optional[Link]
 927         """Try to find a Link matching req
 928 
 929         Expects req, an InstallRequirement and upgrade, a boolean
 930         Returns a Link if found,
 931         Raises DistributionNotFound or BestVersionAlreadyInstalled otherwise
 932         """
 933         hashes = req.hashes(trust_internet=False)
 934         best_candidate_result = self.find_best_candidate(
 935             req.name, specifier=req.specifier, hashes=hashes,
 936         )
 937         best_candidate = best_candidate_result.best_candidate
 938 
 939         installed_version = None    # type: Optional[_BaseVersion]
 940         if req.satisfied_by is not None:
 941             installed_version = parse_version(req.satisfied_by.version)
 942 
 943         def _format_versions(cand_iter):
 944             # type: (Iterable[InstallationCandidate]) -> str
 945             # This repeated parse_version and str() conversion is needed to
 946             # handle different vendoring sources from pipenv.patched.notpip and pkg_resources.
 947             # If we stop using the pkg_resources provided specifier and start
 948             # using our own, we can drop the cast to str().
 949             return ", ".join(sorted(
 950                 {str(c.version) for c in cand_iter},
 951                 key=parse_version,
 952             )) or "none"
 953 
 954         if installed_version is None and best_candidate is None:
 955             logger.critical(
 956                 'Could not find a version that satisfies the requirement %s '
 957                 '(from versions: %s)',
 958                 req,
 959                 _format_versions(best_candidate_result.iter_all()),
 960             )
 961 
 962             raise DistributionNotFound(
 963                 'No matching distribution found for %s' % req
 964             )
 965 
 966         best_installed = False
 967         if installed_version and (
 968                 best_candidate is None or
 969                 best_candidate.version <= installed_version):
 970             best_installed = True
 971 
 972         if not upgrade and installed_version is not None:
 973             if best_installed:
 974                 logger.debug(
 975                     'Existing installed version (%s) is most up-to-date and '
 976                     'satisfies requirement',
 977                     installed_version,
 978                 )
 979             else:
 980                 logger.debug(
 981                     'Existing installed version (%s) satisfies requirement '
 982                     '(most up-to-date version is %s)',
 983                     installed_version,
 984                     best_candidate.version,
 985                 )
 986             return None
 987 
 988         if best_installed:
 989             # We have an existing version, and its the best version
 990             logger.debug(
 991                 'Installed version (%s) is most up-to-date (past versions: '
 992                 '%s)',
 993                 installed_version,
 994                 _format_versions(best_candidate_result.iter_applicable()),
 995             )
 996             raise BestVersionAlreadyInstalled
 997 
 998         logger.debug(
 999             'Using version %s (newest of versions: %s)',
1000             best_candidate.version,
1001             _format_versions(best_candidate_result.iter_applicable()),
1002         )
1003         return best_candidate.link

1006 def _find_name_version_sep(fragment, canonical_name):
1007     # type: (str, str) -> int
1008     """Find the separator's index based on the package's canonical name.
1009 
1010     :param fragment: A <package>+<version> filename "fragment" (stem) or
1011         egg fragment.
1012     :param canonical_name: The package's canonical name.
1013 
1014     This function is needed since the canonicalized name does not necessarily
1015     have the same length as the egg info's name part. An example::
1016 
1017     >>> fragment = 'foo__bar-1.0'
1018     >>> canonical_name = 'foo-bar'
1019     >>> _find_name_version_sep(fragment, canonical_name)
1020     8
1021     """
1022     # Project name and version must be separated by one single dash. Find all
1023     # occurrences of dashes; if the string in front of it matches the canonical
1024     # name, this is the one separating the name and version parts.
1025     for i, c in enumerate(fragment):
1026         if c != "-":
1027             continue
1028         if canonicalize_name(fragment[:i]) == canonical_name:
1029             return i
1030     raise ValueError("{} does not match {}".format(fragment, canonical_name))

1033 def _extract_version_from_fragment(fragment, canonical_name):
1034     # type: (str, str) -> Optional[str]
1035     """Parse the version string from a <package>+<version> filename
1036     "fragment" (stem) or egg fragment.
1037 
1038     :param fragment: The string to parse. E.g. foo-2.1
1039     :param canonical_name: The canonicalized name of the package this
1040         belongs to.
1041     """
1042     try:
1043         version_start = _find_name_version_sep(fragment, canonical_name) + 1
1044     except ValueError:
1045         return None
1046     version = fragment[version_start:]
1047     if not version:
1048         return None
1049     return version
