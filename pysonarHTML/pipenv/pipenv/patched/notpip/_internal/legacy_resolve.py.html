   1 """Dependency Resolution
   2 
   3 The dependency resolution in pip is performed as follows:
   4 
   5 for top-level requirements:
   6     a. only one spec allowed per project, regardless of conflicts or not.
   7        otherwise a "double requirement" exception is raised
   8     b. they override sub-dependency requirements.
   9 for sub-dependencies
  10     a. "first found, wins" (where the order is breadth first)
  11 """

  57 def _check_dist_requires_python(
  58     dist,  # type: pkg_resources.Distribution
  59     version_info,  # type: Tuple[int, int, int]
  60     ignore_requires_python=False,  # type: bool
  61 ):
  62     # type: (...) -> None
  63     """
  64     Check whether the given Python version is compatible with a distribution's
  65     "Requires-Python" value.
  66 
  67     :param version_info: A 3-tuple of ints representing the Python
  68         major-minor-micro version to check.
  69     :param ignore_requires_python: Whether to ignore the "Requires-Python"
  70         value if the given Python version isn't compatible.
  71 
  72     :raises UnsupportedPythonVersion: When the given Python version isn't
  73         compatible.
  74     """
  75     requires_python = get_requires_python(dist)
  76     try:
  77         is_compatible = check_requires_python(
  78             requires_python, version_info=version_info,
  79         )
  80     except specifiers.InvalidSpecifier as exc:
  81         logger.warning(
  82             "Package %r has an invalid Requires-Python: %s",
  83             dist.project_name, exc,
  84         )
  85         return
  86 
  87     if is_compatible:
  88         return
  89 
  90     version = '.'.join(map(str, version_info))
  91     if ignore_requires_python:
  92         logger.debug(
  93             'Ignoring failed Requires-Python check for package %r: '
  94             '%s not in %r',
  95             dist.project_name, version, requires_python,
  96         )
  97         return
  98 
  99     raise UnsupportedPythonVersion(
 100         'Package {!r} requires a different Python: {} not in {!r}'.format(
 101             dist.project_name, version, requires_python,
 102         ))

 105 class Resolver(object):
 106     """Resolves which packages need to be installed/uninstalled to perform \
 107     the requested operation without breaking the requirements of any package.
 108     """
 109 
 110     _allowed_strategies = {"eager", "only-if-needed", "to-satisfy-only"}
 111 
 112     def __init__(
 113         self,
 114         preparer,  # type: RequirementPreparer
 115         finder,  # type: PackageFinder
 116         make_install_req,  # type: InstallRequirementProvider
 117         use_user_site,  # type: bool
 118         ignore_dependencies,  # type: bool
 119         ignore_installed,  # type: bool
 120         ignore_requires_python,  # type: bool
 121         force_reinstall,  # type: bool
 122         upgrade_strategy,  # type: str
 123         py_version_info=None,  # type: Optional[Tuple[int, ...]]
 124         ignore_compatibility=False,  # type: bool
 125     ):
 126         # type: (...) -> None
 127         super(Resolver, self).__init__()
 128         assert upgrade_strategy in self._allowed_strategies
 129 
 130         if py_version_info is None:
 131             py_version_info = sys.version_info[:3]
 132         else:
 133             py_version_info = normalize_version_info(py_version_info)
 134 
 135         self._py_version_info = py_version_info
 136 
 137         self.preparer = preparer
 138         self.finder = finder
 139 
 140         self.upgrade_strategy = upgrade_strategy
 141         self.force_reinstall = force_reinstall
 142         self.ignore_dependencies = ignore_dependencies
 143         self.ignore_installed = ignore_installed
 144         self.ignore_requires_python = ignore_requires_python
 145         self.use_user_site = use_user_site
 146         self._make_install_req = make_install_req
 147         self.ignore_compatibility = ignore_compatibility
 148         self.requires_python = None
 149         if self.ignore_compatibility:
 150             self.ignore_requires_python = True
 151 
 152         self._discovered_dependencies = \
 153             defaultdict(list)  # type: DiscoveredDependencies

 155     def resolve(self, requirement_set):
 156         # type: (RequirementSet) -> None
 157         """Resolve what operations need to be done
 158 
 159         As a side-effect of this method, the packages (and their dependencies)
 160         are downloaded, unpacked and prepared for installation. This
 161         preparation is done by ``pip.operations.prepare``.
 162 
 163         Once PyPI has static dependency metadata available, it would be
 164         possible to move the preparation to become a step separated from
 165         dependency resolution.
 166         """
 167         # If any top-level requirement has a hash specified, enter
 168         # hash-checking mode, which requires hashes from all.
 169         root_reqs = (
 170             requirement_set.unnamed_requirements +
 171             list(requirement_set.requirements.values())
 172         )
 173 
 174         # Actually prepare the files, and collect any exceptions. Most hash
 175         # exceptions cannot be checked ahead of time, because
 176         # req.populate_link() needs to be called before we can make decisions
 177         # based on link type.
 178         discovered_reqs = []  # type: List[InstallRequirement]
 179         hash_errors = HashErrors()
 180         for req in chain(root_reqs, discovered_reqs):
 181             try:
 182                 discovered_reqs.extend(self._resolve_one(requirement_set, req))
 183             except HashError as exc:
 184                 exc.req = req
 185                 hash_errors.append(exc)
 186 
 187         if hash_errors:
 188             raise hash_errors

 190     def _is_upgrade_allowed(self, req):
 191         # type: (InstallRequirement) -> bool
 192         if self.upgrade_strategy == "to-satisfy-only":
 193             return False
 194         elif self.upgrade_strategy == "eager":
 195             return True
 196         else:
 197             assert self.upgrade_strategy == "only-if-needed"
 198             return req.is_direct
 199 
 200     def _set_req_to_reinstall(self, req):
 201         # type: (InstallRequirement) -> None
 202         """
 203         Set a requirement to be installed.
 204         """
 205         # Don't uninstall the conflict if doing a user install and the
 206         # conflict is not a user install.
 207         if not self.use_user_site or dist_in_usersite(req.satisfied_by):
 208             req.should_reinstall = True
 209         req.satisfied_by = None
 210 
 211     def _check_skip_installed(self, req_to_install):
 212         # type: (InstallRequirement) -> Optional[str]
 213         """Check if req_to_install should be skipped.
 214 
 215         This will check if the req is installed, and whether we should upgrade
 216         or reinstall it, taking into account all the relevant user options.
 217 
 218         After calling this req_to_install will only have satisfied_by set to
 219         None if the req_to_install is to be upgraded/reinstalled etc. Any
 220         other value will be a dist recording the current thing installed that
 221         satisfies the requirement.
 222 
 223         Note that for vcs urls and the like we can't assess skipping in this
 224         routine - we simply identify that we need to pull the thing down,
 225         then later on it is pulled down and introspected to assess upgrade/
 226         reinstalls etc.
 227 
 228         :return: A text reason for why it was skipped, or None.
 229         """
 230         if self.ignore_installed:
 231             return None
 232 
 233         req_to_install.check_if_exists(self.use_user_site)
 234         if not req_to_install.satisfied_by:
 235             return None
 236 
 237         if self.force_reinstall:
 238             self._set_req_to_reinstall(req_to_install)
 239             return None
 240 
 241         if not self._is_upgrade_allowed(req_to_install):
 242             if self.upgrade_strategy == "only-if-needed":
 243                 return 'already satisfied, skipping upgrade'
 244             return 'already satisfied'
 245 
 246         # Check for the possibility of an upgrade.  For link-based
 247         # requirements we have to pull the tree down and inspect to assess
 248         # the version #, so it's handled way down.
 249         if not req_to_install.link:
 250             try:
 251                 self.finder.find_requirement(req_to_install, upgrade=True)
 252             except BestVersionAlreadyInstalled:
 253                 # Then the best version is installed.
 254                 return 'already up-to-date'
 255             except DistributionNotFound:
 256                 # No distribution found, so we squash the error.  It will
 257                 # be raised later when we re-try later to do the install.
 258                 # Why don't we just raise here?
 259                 pass
 260 
 261         self._set_req_to_reinstall(req_to_install)
 262         return None
 263 
 264     def _get_abstract_dist_for(self, req):
 265         # type: (InstallRequirement) -> AbstractDistribution
 266         """Takes a InstallRequirement and returns a single AbstractDist \
 267         representing a prepared variant of the same.
 268         """
 269         if req.editable:
 270             return self.preparer.prepare_editable_requirement(req)
 271 
 272         # satisfied_by is only evaluated by calling _check_skip_installed,
 273         # so it must be None here.
 274         assert req.satisfied_by is None
 275         skip_reason = self._check_skip_installed(req)
 276 
 277         if req.satisfied_by:
 278             return self.preparer.prepare_installed_requirement(
 279                 req, skip_reason
 280             )
 281 
 282         upgrade_allowed = self._is_upgrade_allowed(req)
 283 
 284         # We eagerly populate the link, since that's our "legacy" behavior.
 285         require_hashes = self.preparer.require_hashes
 286         req.populate_link(self.finder, upgrade_allowed, require_hashes)
 287         abstract_dist = self.preparer.prepare_linked_requirement(req)
 288 
 289         # NOTE
 290         # The following portion is for determining if a certain package is
 291         # going to be re-installed/upgraded or not and reporting to the user.
 292         # This should probably get cleaned up in a future refactor.
 293 
 294         # req.req is only avail after unpack for URL
 295         # pkgs repeat check_if_exists to uninstall-on-upgrade
 296         # (#14)
 297         if not self.ignore_installed:
 298             req.check_if_exists(self.use_user_site)
 299 
 300         if req.satisfied_by:
 301             should_modify = (
 302                 self.upgrade_strategy != "to-satisfy-only" or
 303                 self.force_reinstall or
 304                 self.ignore_installed or
 305                 req.link.scheme == 'file'
 306             )
 307             if should_modify:
 308                 self._set_req_to_reinstall(req)
 309             else:
 310                 logger.info(
 311                     'Requirement already satisfied (use --upgrade to upgrade):'
 312                     ' %s', req,
 313                 )
 314 
 315         return abstract_dist

 317     def _resolve_one(
 318         self,
 319         requirement_set,  # type: RequirementSet
 320         req_to_install,  # type: InstallRequirement
 321         ignore_requires_python=False,  # type: bool
 322     ):
 323         # type: (...) -> List[InstallRequirement]
 324         """Prepare a single requirements file.
 325 
 326         :return: A list of additional InstallRequirements to also install.
 327         """
 328         # Tell user what we are doing for this requirement:
 329         # obtain (editable), skipping, processing (local url), collecting
 330         # (remote url or package name)
 331         if req_to_install.constraint or req_to_install.prepared:
 332             return []
 333 
 334         req_to_install.prepared = True
 335 
 336         # register tmp src for cleanup in case something goes wrong
 337         requirement_set.reqs_to_cleanup.append(req_to_install)
 338 
 339         abstract_dist = self._get_abstract_dist_for(req_to_install)
 340 
 341         # Parse and return dependencies
 342         dist = abstract_dist.get_pkg_resources_distribution()
 343         # This will raise UnsupportedPythonVersion if the given Python
 344         # version isn't compatible with the distribution's Requires-Python.
 345         ignore_requires_python = (
 346             ignore_requires_python or self.ignore_requires_python or
 347             self.ignore_compatibility
 348         )
 349         _check_dist_requires_python(
 350             dist, version_info=self._py_version_info,
 351             ignore_requires_python=ignore_requires_python,
 352         )
 353         # Patched in - lets get the python version on here then
 354         # FIXME: Does this patch even work? it puts the python version
 355         # on the resolver... why?
 356         try:
 357             self.requires_python = get_requires_python(dist)
 358         except TypeError:
 359             self.requires_python = None
 360         more_reqs = []  # type: List[InstallRequirement]
 361 
 362         def add_req(subreq, extras_requested):
 363             sub_install_req = self._make_install_req(
 364                 str(subreq),
 365                 req_to_install,
 366             )
 367             parent_req_name = req_to_install.name
 368             to_scan_again, add_to_parent = requirement_set.add_requirement(
 369                 sub_install_req,
 370                 parent_req_name=parent_req_name,
 371                 extras_requested=extras_requested,
 372             )
 373             if parent_req_name and add_to_parent:
 374                 self._discovered_dependencies[parent_req_name].append(
 375                     add_to_parent
 376                 )
 377             more_reqs.extend(to_scan_again)
 378 
 379         with indent_log():
 380             # We add req_to_install before its dependencies, so that we
 381             # can refer to it when adding dependencies.
 382             if not requirement_set.has_requirement(req_to_install.name):
 383                 # 'unnamed' requirements will get added here
 384                 # 'unnamed' requirements can only come from being directly
 385                 # provided by the user.
 386                 req_to_install.is_direct = True
 387                 assert req_to_install.is_direct
 388                 available_requested = sorted(
 389                     set(dist.extras) & set(req_to_install.extras)
 390                 )
 391                 requirement_set.add_requirement(
 392                     req_to_install, parent_req_name=None,
 393                     extras_requested=available_requested,
 394                 )
 395 
 396             if not self.ignore_dependencies:
 397                 if req_to_install.extras:
 398                     logger.debug(
 399                         "Installing extra requirements: %r",
 400                         ','.join(req_to_install.extras),
 401                     )
 402                 missing_requested = sorted(
 403                     set(req_to_install.extras) - set(dist.extras)
 404                 )
 405                 for missing in missing_requested:
 406                     logger.warning(
 407                         '%s does not provide the extra \'%s\'',
 408                         dist, missing
 409                     )
 410 
 411                 available_requested = sorted(
 412                     set(dist.extras) & set(req_to_install.extras)
 413                 )
 414                 for subreq in dist.requires(available_requested):
 415                     add_req(subreq, extras_requested=available_requested)
 416 
 417             if not req_to_install.editable and not req_to_install.satisfied_by:
 418                 # XXX: --no-install leads this to report 'Successfully
 419                 # downloaded' for only non-editable reqs, even though we took
 420                 # action on them.
 421                 requirement_set.successfully_downloaded.append(req_to_install)
 422 
 423         return more_reqs

 425     def get_installation_order(self, req_set):
 426         # type: (RequirementSet) -> List[InstallRequirement]
 427         """Create the installation order.
 428 
 429         The installation order is topological - requirements are installed
 430         before the requiring thing. We break cycles at an arbitrary point,
 431         and make no other guarantees.
 432         """
 433         # The current implementation, which we may change at any point
 434         # installs the user specified things in the order given, except when
 435         # dependencies must come earlier to achieve topological order.
 436         order = []
 437         ordered_reqs = set()  # type: Set[InstallRequirement]
 438 
 439         def schedule(req):
 440             if req.satisfied_by or req in ordered_reqs:
 441                 return
 442             if req.constraint:
 443                 return
 444             ordered_reqs.add(req)
 445             for dep in self._discovered_dependencies[req.name]:
 446                 schedule(dep)
 447             order.append(req)
 448 
 449         for install_req in req_set.requirements.values():
 450             schedule(install_req)
 451         return order
