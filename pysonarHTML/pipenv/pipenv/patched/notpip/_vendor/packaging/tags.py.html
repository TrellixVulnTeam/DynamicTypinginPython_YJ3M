  60 class Tag(object):
  61 
  62     __slots__ = ["_interpreter", "_abi", "_platform"]
  63 
  64     def __init__(self, interpreter, abi, platform):
  65         # type: (str, str, str) -> None
  66         self._interpreter = interpreter.lower()
  67         self._abi = abi.lower()
  68         self._platform = platform.lower()
  69 
  70     @property
  71     def interpreter(self):
  72         # type: () -> str
  73         return self._interpreter
  74 
  75     @property
  76     def abi(self):
  77         # type: () -> str
  78         return self._abi
  79 
  80     @property
  81     def platform(self):
  82         # type: () -> str
  83         return self._platform
  84 
  85     def __eq__(self, other):
  86         # type: (object) -> bool
  87         if not isinstance(other, Tag):
  88             return NotImplemented
  89 
  90         return (
  91             (self.platform == other.platform)
  92             and (self.abi == other.abi)
  93             and (self.interpreter == other.interpreter)
  94         )
  95 
  96     def __hash__(self):
  97         # type: () -> int
  98         return hash((self._interpreter, self._abi, self._platform))
  99 
 100     def __str__(self):
 101         # type: () -> str
 102         return "{}-{}-{}".format(self._interpreter, self._abi, self._platform)
 103 
 104     def __repr__(self):
 105         # type: () -> str
 106         return "<{self} @ {self_id}>".format(self=self, self_id=id(self))

 109 def parse_tag(tag):
 110     # type: (str) -> FrozenSet[Tag]
 111     tags = set()
 112     interpreters, abis, platforms = tag.split("-")
 113     for interpreter in interpreters.split("."):
 114         for abi in abis.split("."):
 115             for platform_ in platforms.split("."):
 116                 tags.add(Tag(interpreter, abi, platform_))
 117     return frozenset(tags)

 120 def _warn_keyword_parameter(func_name, kwargs):
 121     # type: (str, Dict[str, bool]) -> bool
 122     """
 123     Backwards-compatibility with Python 2.7 to allow treating 'warn' as keyword-only.
 124     """
 125     if not kwargs:
 126         return False
 127     elif len(kwargs) > 1 or "warn" not in kwargs:
 128         kwargs.pop("warn", None)
 129         arg = next(iter(kwargs.keys()))
 130         raise TypeError(
 131             "{}() got an unexpected keyword argument {!r}".format(func_name, arg)
 132         )
 133     return kwargs["warn"]

 161 def _cpython_abis(py_version, warn=False):
 162     # type: (PythonVersion, bool) -> List[str]
 163     py_version = tuple(py_version)  # To allow for version comparison.
 164     abis = []
 165     version = "{}{}".format(*py_version[:2])
 166     debug = pymalloc = ucs4 = ""
 167     with_debug = _get_config_var("Py_DEBUG", warn)
 168     has_refcount = hasattr(sys, "gettotalrefcount")
 169     # Windows doesn't set Py_DEBUG, so checking for support of debug-compiled
 170     # extension modules is the best option.
 171     # https://github.com/pypa/pip/issues/3383#issuecomment-173267692
 172     has_ext = "_d.pyd" in EXTENSION_SUFFIXES
 173     if with_debug or (with_debug is None and (has_refcount or has_ext)):
 174         debug = "d"
 175     if py_version < (3, 8):
 176         with_pymalloc = _get_config_var("WITH_PYMALLOC", warn)
 177         if with_pymalloc or with_pymalloc is None:
 178             pymalloc = "m"
 179         if py_version < (3, 3):
 180             unicode_size = _get_config_var("Py_UNICODE_SIZE", warn)
 181             if unicode_size == 4 or (
 182                 unicode_size is None and sys.maxunicode == 0x10FFFF
 183             ):
 184                 ucs4 = "u"
 185     elif debug:
 186         # Debug builds can also load "normal" extension modules.
 187         # We can also assume no UCS-4 or pymalloc requirement.
 188         abis.append("cp{version}".format(version=version))
 189     abis.insert(
 190         0,
 191         "cp{version}{debug}{pymalloc}{ucs4}".format(
 192             version=version, debug=debug, pymalloc=pymalloc, ucs4=ucs4
 193         ),
 194     )
 195     return abis

 198 def cpython_tags(
 199     python_version=None,  # type: Optional[PythonVersion]
 200     abis=None,  # type: Optional[Iterable[str]]
 201     platforms=None,  # type: Optional[Iterable[str]]
 202     **kwargs  # type: bool
 203 ):
 204     # type: (...) -> Iterator[Tag]
 205     """
 206     Yields the tags for a CPython interpreter.
 207 
 208     The tags consist of:
 209     - cp<python_version>-<abi>-<platform>
 210     - cp<python_version>-abi3-<platform>
 211     - cp<python_version>-none-<platform>
 212     - cp<less than python_version>-abi3-<platform>  # Older Python versions down to 3.2.
 213 
 214     If python_version only specifies a major version then user-provided ABIs and
 215     the 'none' ABItag will be used.
 216 
 217     If 'abi3' or 'none' are specified in 'abis' then they will be yielded at
 218     their normal position and not at the beginning.
 219     """
 220     warn = _warn_keyword_parameter("cpython_tags", kwargs)
 221     if not python_version:
 222         python_version = sys.version_info[:2]
 223 
 224     if len(python_version) < 2:
 225         interpreter = "cp{}".format(python_version[0])
 226     else:
 227         interpreter = "cp{}{}".format(*python_version[:2])
 228 
 229     if abis is None:
 230         if len(python_version) > 1:
 231             abis = _cpython_abis(python_version, warn)
 232         else:
 233             abis = []
 234     abis = list(abis)
 235     # 'abi3' and 'none' are explicitly handled later.
 236     for explicit_abi in ("abi3", "none"):
 237         try:
 238             abis.remove(explicit_abi)
 239         except ValueError:
 240             pass
 241 
 242     platforms = list(platforms or _platform_tags())
 243     for abi in abis:
 244         for platform_ in platforms:
 245             yield Tag(interpreter, abi, platform_)
 246     if _abi3_applies(python_version):
 247         for tag in (Tag(interpreter, "abi3", platform_) for platform_ in platforms):
 248             yield tag
 249     for tag in (Tag(interpreter, "none", platform_) for platform_ in platforms):
 250         yield tag
 251 
 252     if _abi3_applies(python_version):
 253         for minor_version in range(python_version[1] - 1, 1, -1):
 254             for platform_ in platforms:
 255                 interpreter = "cp{major}{minor}".format(
 256                     major=python_version[0], minor=minor_version
 257                 )
 258                 yield Tag(interpreter, "abi3", platform_)

 268 def generic_tags(
 269     interpreter=None,  # type: Optional[str]
 270     abis=None,  # type: Optional[Iterable[str]]
 271     platforms=None,  # type: Optional[Iterable[str]]
 272     **kwargs  # type: bool
 273 ):
 274     # type: (...) -> Iterator[Tag]
 275     """
 276     Yields the tags for a generic interpreter.
 277 
 278     The tags consist of:
 279     - <interpreter>-<abi>-<platform>
 280 
 281     The "none" ABI will be added if it was not explicitly provided.
 282     """
 283     warn = _warn_keyword_parameter("generic_tags", kwargs)
 284     if not interpreter:
 285         interp_name = interpreter_name()
 286         interp_version = interpreter_version(warn=warn)
 287         interpreter = "".join([interp_name, interp_version])
 288     if abis is None:
 289         abis = _generic_abi()
 290     platforms = list(platforms or _platform_tags())
 291     abis = list(abis)
 292     if "none" not in abis:
 293         abis.append("none")
 294     for abi in abis:
 295         for platform_ in platforms:
 296             yield Tag(interpreter, abi, platform_)

 299 def _py_interpreter_range(py_version):
 300     # type: (PythonVersion) -> Iterator[str]
 301     """
 302     Yields Python versions in descending order.
 303 
 304     After the latest version, the major-only version will be yielded, and then
 305     all previous versions of that major version.
 306     """
 307     if len(py_version) > 1:
 308         yield "py{major}{minor}".format(major=py_version[0], minor=py_version[1])
 309     yield "py{major}".format(major=py_version[0])
 310     if len(py_version) > 1:
 311         for minor in range(py_version[1] - 1, -1, -1):
 312             yield "py{major}{minor}".format(major=py_version[0], minor=minor)

 315 def compatible_tags(
 316     python_version=None,  # type: Optional[PythonVersion]
 317     interpreter=None,  # type: Optional[str]
 318     platforms=None,  # type: Optional[Iterable[str]]
 319 ):
 320     # type: (...) -> Iterator[Tag]
 321     """
 322     Yields the sequence of tags that are compatible with a specific version of Python.
 323 
 324     The tags consist of:
 325     - py*-none-<platform>
 326     - <interpreter>-none-any  # ... if `interpreter` is provided.
 327     - py*-none-any
 328     """
 329     if not python_version:
 330         python_version = sys.version_info[:2]
 331     platforms = list(platforms or _platform_tags())
 332     for version in _py_interpreter_range(python_version):
 333         for platform_ in platforms:
 334             yield Tag(version, "none", platform_)
 335     if interpreter:
 336         yield Tag(interpreter, "none", "any")
 337     for version in _py_interpreter_range(python_version):
 338         yield Tag(version, "none", "any")

 341 def _mac_arch(arch, is_32bit=_32_BIT_INTERPRETER):
 342     # type: (str, bool) -> str
 343     if not is_32bit:
 344         return arch
 345 
 346     if arch.startswith("ppc"):
 347         return "ppc"
 348 
 349     return "i386"

 352 def _mac_binary_formats(version, cpu_arch):
 353     # type: (MacVersion, str) -> List[str]
 354     formats = [cpu_arch]
 355     if cpu_arch == "x86_64":
 356         if version < (10, 4):
 357             return []
 358         formats.extend(["intel", "fat64", "fat32"])
 359 
 360     elif cpu_arch == "i386":
 361         if version < (10, 4):
 362             return []
 363         formats.extend(["intel", "fat32", "fat"])
 364 
 365     elif cpu_arch == "ppc64":
 366         # TODO: Need to care about 32-bit PPC for ppc64 through 10.2?
 367         if version > (10, 5) or version < (10, 4):
 368             return []
 369         formats.append("fat64")
 370 
 371     elif cpu_arch == "ppc":
 372         if version > (10, 6):
 373             return []
 374         formats.extend(["fat32", "fat"])
 375 
 376     formats.append("universal")
 377     return formats

 380 def mac_platforms(version=None, arch=None):
 381     # type: (Optional[MacVersion], Optional[str]) -> Iterator[str]
 382     """
 383     Yields the platform tags for a macOS system.
 384 
 385     The `version` parameter is a two-item tuple specifying the macOS version to
 386     generate platform tags for. The `arch` parameter is the CPU architecture to
 387     generate platform tags for. Both parameters default to the appropriate value
 388     for the current system.
 389     """
 390     version_str, _, cpu_arch = platform.mac_ver()  # type: ignore
 391     if version is None:
 392         version = cast("MacVersion", tuple(map(int, version_str.split(".")[:2])))
 393     else:
 394         version = version
 395     if arch is None:
 396         arch = _mac_arch(cpu_arch)
 397     else:
 398         arch = arch
 399     for minor_version in range(version[1], -1, -1):
 400         compat_version = version[0], minor_version
 401         binary_formats = _mac_binary_formats(compat_version, arch)
 402         for binary_format in binary_formats:
 403             yield "macosx_{major}_{minor}_{binary_format}".format(
 404                 major=compat_version[0],
 405                 minor=compat_version[1],
 406                 binary_format=binary_format,
 407             )

 431 def _glibc_version_string_confstr():
 432     # type: () -> Optional[str]
 433     """
 434     Primary implementation of glibc_version_string using os.confstr.
 435     """
 436     # os.confstr is quite a bit faster than ctypes.DLL. It's also less likely
 437     # to be broken or missing. This strategy is used in the standard library
 438     # platform module.
 439     # https://github.com/python/cpython/blob/fcf1d003bf4f0100c9d0921ff3d70e1127ca1b71/Lib/platform.py#L175-L183
 440     try:
 441         # os.confstr("CS_GNU_LIBC_VERSION") returns a string like "glibc 2.17".
 442         version_string = os.confstr(  # type: ignore[attr-defined] # noqa: F821
 443             "CS_GNU_LIBC_VERSION"
 444         )
 445         assert version_string is not None
 446         _, version = version_string.split()  # type: Tuple[str, str]
 447     except (AssertionError, AttributeError, OSError, ValueError):
 448         # os.confstr() or CS_GNU_LIBC_VERSION not available (or a bad value)...
 449         return None
 450     return version

 453 def _glibc_version_string_ctypes():
 454     # type: () -> Optional[str]
 455     """
 456     Fallback implementation of glibc_version_string using ctypes.
 457     """
 458     try:
 459         import ctypes
 460     except ImportError:
 461         return None
 462 
 463     # ctypes.CDLL(None) internally calls dlopen(NULL), and as the dlopen
 464     # manpage says, "If filename is NULL, then the returned handle is for the
 465     # main program". This way we can let the linker do the work to figure out
 466     # which libc our process is actually using.
 467     #
 468     # Note: typeshed is wrong here so we are ignoring this line.
 469     process_namespace = ctypes.CDLL(None)  # type: ignore
 470     try:
 471         gnu_get_libc_version = process_namespace.gnu_get_libc_version
 472     except AttributeError:
 473         # Symbol doesn't exist -> therefore, we are not linked to
 474         # glibc.
 475         return None
 476 
 477     # Call gnu_get_libc_version, which returns a string like "2.5"
 478     gnu_get_libc_version.restype = ctypes.c_char_p
 479     version_str = gnu_get_libc_version()  # type: str
 480     # py2 / py3 compatibility:
 481     if not isinstance(version_str, str):
 482         version_str = version_str.decode("ascii")
 483 
 484     return version_str

 487 # Separated out from have_compatible_glibc for easier unit testing.
 488 def _check_glibc_version(version_str, required_major, minimum_minor):
 489     # type: (str, int, int) -> bool
 490     # Parse string and check against requested version.
 491     #
 492     # We use a regexp instead of str.split because we want to discard any
 493     # random junk that might come after the minor version -- this might happen
 494     # in patched/forked versions of glibc (e.g. Linaro's version of glibc
 495     # uses version strings like "2.20-2014.11"). See gh-3588.
 496     m = re.match(r"(?P<major>[0-9]+)\.(?P<minor>[0-9]+)", version_str)
 497     if not m:
 498         warnings.warn(
 499             "Expected glibc version with 2 components major.minor,"
 500             " got: %s" % version_str,
 501             RuntimeWarning,
 502         )
 503         return False
 504     return (
 505         int(m.group("major")) == required_major
 506         and int(m.group("minor")) >= minimum_minor
 507     )

 512     version_str = _glibc_version_string()

 515     return _check_glibc_version(version_str, required_major, minimum_minor)

 518 # Python does not provide platform information at sufficient granularity to
 519 # identify the architecture of the running executable in some cases, so we
 520 # determine it dynamically by reading the information from the running
 521 # process. This only applies on Linux, which uses the ELF format.
 522 class _ELFFileHeader(object):
 523     # https://en.wikipedia.org/wiki/Executable_and_Linkable_Format#File_header
 524     class _InvalidELFFileHeader(ValueError):
 525         """
 526         An invalid ELF file header was found.
 527         """
 528 
 529     ELF_MAGIC_NUMBER = 0x7F454C46
 530     ELFCLASS32 = 1
 531     ELFCLASS64 = 2
 532     ELFDATA2LSB = 1
 533     ELFDATA2MSB = 2
 534     EM_386 = 3
 535     EM_S390 = 22
 536     EM_ARM = 40
 537     EM_X86_64 = 62
 538     EF_ARM_ABIMASK = 0xFF000000
 539     EF_ARM_ABI_VER5 = 0x05000000
 540     EF_ARM_ABI_FLOAT_HARD = 0x00000400
 541 
 542     def __init__(self, file):
 543         # type: (IO[bytes]) -> None
 544         def unpack(fmt):
 545             # type: (str) -> int
 546             try:
 547                 result, = struct.unpack(
 548                     fmt, file.read(struct.calcsize(fmt))
 549                 )  # type: (int, )
 550             except struct.error:
 551                 raise _ELFFileHeader._InvalidELFFileHeader()
 552             return result
 553 
 554         self.e_ident_magic = unpack(">I")
 555         if self.e_ident_magic != self.ELF_MAGIC_NUMBER:
 556             raise _ELFFileHeader._InvalidELFFileHeader()
 557         self.e_ident_class = unpack("B")
 558         if self.e_ident_class not in {self.ELFCLASS32, self.ELFCLASS64}:
 559             raise _ELFFileHeader._InvalidELFFileHeader()
 560         self.e_ident_data = unpack("B")
 561         if self.e_ident_data not in {self.ELFDATA2LSB, self.ELFDATA2MSB}:
 562             raise _ELFFileHeader._InvalidELFFileHeader()
 563         self.e_ident_version = unpack("B")
 564         self.e_ident_osabi = unpack("B")
 565         self.e_ident_abiversion = unpack("B")
 566         self.e_ident_pad = file.read(7)
 567         format_h = "<H" if self.e_ident_data == self.ELFDATA2LSB else ">H"
 568         format_i = "<I" if self.e_ident_data == self.ELFDATA2LSB else ">I"
 569         format_q = "<Q" if self.e_ident_data == self.ELFDATA2LSB else ">Q"
 570         format_p = format_i if self.e_ident_class == self.ELFCLASS32 else format_q
 571         self.e_type = unpack(format_h)
 572         self.e_machine = unpack(format_h)
 573         self.e_version = unpack(format_i)
 574         self.e_entry = unpack(format_p)
 575         self.e_phoff = unpack(format_p)
 576         self.e_shoff = unpack(format_p)
 577         self.e_flags = unpack(format_i)
 578         self.e_ehsize = unpack(format_h)
 579         self.e_phentsize = unpack(format_h)
 580         self.e_phnum = unpack(format_h)
 581         self.e_shentsize = unpack(format_h)
 582         self.e_shnum = unpack(format_h)
 583         self.e_shstrndx = unpack(format_h)
 584

 586 def _get_elf_header():
 587     # type: () -> Optional[_ELFFileHeader]
 588     try:
 589         with open(sys.executable, "rb") as f:
 590             elf_header = _ELFFileHeader(f)

 596 def _is_linux_armhf():
 597     # type: () -> bool
 598     # hard-float ABI can be detected from the ELF header of the running
 599     # process
 600     # https://static.docs.arm.com/ihi0044/g/aaelf32.pdf
 601     elf_header = _get_elf_header()
 602     if elf_header is None:
 603         return False
 604     result = elf_header.e_ident_class == elf_header.ELFCLASS32
 605     result &= elf_header.e_ident_data == elf_header.ELFDATA2LSB
 606     result &= elf_header.e_machine == elf_header.EM_ARM
 607     result &= (
 608         elf_header.e_flags & elf_header.EF_ARM_ABIMASK
 609     ) == elf_header.EF_ARM_ABI_VER5
 610     result &= (
 611         elf_header.e_flags & elf_header.EF_ARM_ABI_FLOAT_HARD
 612     ) == elf_header.EF_ARM_ABI_FLOAT_HARD
 613     return result
 614 
 615 
 616 def _is_linux_i686():
 617     # type: () -> bool
 618     elf_header = _get_elf_header()
 619     if elf_header is None:
 620         return False
 621     result = elf_header.e_ident_class == elf_header.ELFCLASS32
 622     result &= elf_header.e_ident_data == elf_header.ELFDATA2LSB
 623     result &= elf_header.e_machine == elf_header.EM_386
 624     return result

 627 def _have_compatible_manylinux_abi(arch):
 628     # type: (str) -> bool
 629     if arch == "armv7l":
 630         return _is_linux_armhf()
 631     if arch == "i686":
 632         return _is_linux_i686()
 633     return True

 636 def _linux_platforms(is_32bit=_32_BIT_INTERPRETER):
 637     # type: (bool) -> Iterator[str]
 638     linux = _normalize_string(distutils.util.get_platform())
 639     if linux == "linux_x86_64" and is_32bit:
 640         linux = "linux_i686"
 641     manylinux_support = []
 642     _, arch = linux.split("_", 1)
 643     if _have_compatible_manylinux_abi(arch):
 644         if arch in {"x86_64", "i686", "aarch64", "armv7l", "ppc64", "ppc64le", "s390x"}:
 645             manylinux_support.append(
 646                 ("manylinux2014", (2, 17))
 647             )  # CentOS 7 w/ glibc 2.17 (PEP 599)
 648         if arch in {"x86_64", "i686"}:
 649             manylinux_support.append(
 650                 ("manylinux2010", (2, 12))
 651             )  # CentOS 6 w/ glibc 2.12 (PEP 571)
 652             manylinux_support.append(
 653                 ("manylinux1", (2, 5))
 654             )  # CentOS 5 w/ glibc 2.5 (PEP 513)
 655     manylinux_support_iter = iter(manylinux_support)
 656     for name, glibc_version in manylinux_support_iter:
 657         if _is_manylinux_compatible(name, glibc_version):
 658             yield linux.replace("linux", name)
 659             break
 660     # Support for a later manylinux implies support for an earlier version.
 661     for name, _ in manylinux_support_iter:
 662         yield linux.replace("linux", name)
 663     yield linux

 666 def _generic_platforms():
 667     # type: () -> Iterator[str]
 668     yield _normalize_string(distutils.util.get_platform())

 671 def _platform_tags():
 672     # type: () -> Iterator[str]
 673     """
 674     Provides the platform tags for this installation.
 675     """
 676     if platform.system() == "Darwin":
 677         return mac_platforms()
 678     elif platform.system() == "Linux":
 679         return _linux_platforms()
 680     else:
 681         return _generic_platforms()

 684 def interpreter_name():
 685     # type: () -> str
 686     """
 687     Returns the name of the running interpreter.
 688     """
 689     try:
 690         name = sys.implementation.name  # type: ignore
 691     except AttributeError:  # pragma: no cover
 692         # Python 2.7 compatibility.
 693         name = platform.python_implementation().lower()
 694     return INTERPRETER_SHORT_NAMES.get(name) or name

 697 def interpreter_version(**kwargs):
 698     # type: (bool) -> str
 699     """
 700     Returns the version of the running interpreter.
 701     """
 702     warn = _warn_keyword_parameter("interpreter_version", kwargs)
 703     version = _get_config_var("py_version_nodot", warn=warn)
 704     if version:
 705         version = str(version)
 706     else:
 707         version = "".join(map(str, sys.version_info[:2]))
 708     return version

 711 def sys_tags(**kwargs):
 712     # type: (bool) -> Iterator[Tag]
 713     """
 714     Returns the sequence of tag triples for the running interpreter.
 715 
 716     The order of the sequence corresponds to priority order for the
 717     interpreter, from most to least important.
 718     """
 719     warn = _warn_keyword_parameter("sys_tags", kwargs)
 720 
 721     interp_name = interpreter_name()
 722     if interp_name == "cp":
 723         for tag in cpython_tags(warn=warn):
 724             yield tag
 725     else:
 726         for tag in generic_tags():
 727             yield tag
 728 
 729     for tag in compatible_tags():
 730         yield tag
