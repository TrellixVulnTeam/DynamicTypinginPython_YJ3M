  30                            for attr, value in token["data"].items()])

  32             if type == "EmptyTag":
  33                 type = "EndTag"

  39                 name = token["name"]

  41             yield END, QName(name), (None, -1, -1)
  42 
  43         elif type == "Comment":
  44             yield COMMENT, token["data"], (None, -1, -1)
  45 
  46         elif type == "Doctype":

  48                             token["systemId"]), (None, -1, -1)
