  15 def _secure_open_write(filename, fmode):
  16     # We only want to write to this file, so open it in write only mode
  17     flags = os.O_WRONLY
  18 
  19     # os.O_CREAT | os.O_EXCL will fail if the file already exists, so we only
  20     #  will open *new* files.
  21     # We specify this because we want to ensure that the mode we pass is the
  22     # mode of the file.
  23     flags |= os.O_CREAT | os.O_EXCL
  24 
  25     # Do not follow symlinks to prevent someone from making a symlink that
  26     # we follow and insecurely open a cache file.
  27     if hasattr(os, "O_NOFOLLOW"):
  28         flags |= os.O_NOFOLLOW
  29 
  30     # On Windows we'll mark this file as binary
  31     if hasattr(os, "O_BINARY"):
  32         flags |= os.O_BINARY
  33 
  34     # Before we open our file, we want to delete any existing file that is
  35     # there
  36     try:
  37         os.remove(filename)
  38     except (IOError, OSError):
  39         # The file must not exist already, so we can just skip ahead to opening
  40         pass
  41 
  42     # Open our file, the use of os.O_CREAT | os.O_EXCL will ensure that if a
  43     # race condition happens between the os.remove and this line, that an
  44     # error will be raised. Because we utilize a lockfile this should only
  45     # happen if someone is attempting to attack us.
  46     fd = os.open(filename, flags, fmode)
  47     try:
  48         return os.fdopen(fd, "wb")
  49 
  50     except:
  51         # An error occurred wrapping our FD in a file object
  52         os.close(fd)
  53         raise

  56 class FileCache(BaseCache):
  57 
  58     def __init__(
  59         self,
  60         directory,
  61         forever=False,
  62         filemode=0o0600,
  63         dirmode=0o0700,
  64         use_dir_lock=None,
  65         lock_class=None,
  66     ):
  67 
  68         if use_dir_lock is not None and lock_class is not None:
  69             raise ValueError("Cannot use use_dir_lock and lock_class together")
  70 
  71         try:
  72             from lockfile import LockFile
  73             from lockfile.mkdirlockfile import MkdirLockFile
  74         except ImportError:
  75             notice = dedent(
  76                 """
  77             NOTE: In order to use the FileCache you must have
  78             lockfile installed. You can install it via pip:
  79               pip install lockfile
  80             """
  81             )
  82             raise ImportError(notice)
  83 
  84         else:
  85             if use_dir_lock:
  86                 lock_class = MkdirLockFile
  87 
  88             elif lock_class is None:
  89                 lock_class = LockFile
  90 
  91         self.directory = directory
  92         self.forever = forever
  93         self.filemode = filemode
  94         self.dirmode = dirmode
  95         self.lock_class = lock_class
  96 
  97     @staticmethod
  98     def encode(x):
  99         return hashlib.sha224(x.encode()).hexdigest()
 100 
 101     def _fn(self, name):
 102         # NOTE: This method should not change as some may depend on it.
 103         #       See: https://github.com/ionrock/cachecontrol/issues/63
 104         hashed = self.encode(name)
 105         parts = list(hashed[:5]) + [hashed]
 106         return os.path.join(self.directory, *parts)
 107 
 108     def get(self, key):
 109         name = self._fn(key)
 110         try:
 111             with open(name, "rb") as fh:
 112                 return fh.read()
 113 
 114         except FileNotFoundError:
 115             return None
 116 
 117     def set(self, key, value):
 118         name = self._fn(key)
 119 
 120         # Make sure the directory exists
 121         try:
 122             os.makedirs(os.path.dirname(name), self.dirmode)
 123         except (IOError, OSError):
 124             pass
 125 
 126         with self.lock_class(name) as lock:
 127             # Write our actual file
 128             with _secure_open_write(lock.path, self.filemode) as fh:
 129                 fh.write(value)
 130 
 131     def delete(self, key):
 132         name = self._fn(key)
 133         if not self.forever:
 134             try:
 135                 os.remove(name)
 136             except FileNotFoundError:
 137                 pass

 140 def url_to_file_path(url, filecache):
 141     """Return the file cache path based on the URL.
 142 
 143     This does not ensure the file exists!
 144     """
 145     key = CacheController.cache_url(url)
 146     return filecache._fn(key)
