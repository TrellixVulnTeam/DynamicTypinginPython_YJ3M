  76     def safe_import(self, name):
  77         # type: (str) -> ModuleType
  78         """Helper utility for reimporting previously imported modules while inside the env"""
  79         module = None
  80         if name not in self._modules:
  81             self._modules[name] = importlib.import_module(name)
  82         module = self._modules[name]
  83         if not module:
  84             dist = next(iter(
  85                 dist for dist in self.base_working_set if dist.project_name == name
  86             ), None)
  87             if dist:
  88                 dist.activate()
  89             module = importlib.import_module(name)
  90         if name in sys.modules:
  91             try:
  92                 six.moves.reload_module(module)
  93                 six.moves.reload_module(sys.modules[name])
  94             except TypeError:
  95                 del sys.modules[name]
  96                 sys.modules[name] = self._modules[name]
  97                 return self._modules[name]
  98         return module

 100     @classmethod
 101     def resolve_dist(cls, dist, working_set):
 102         # type: (pkg_resources.Distribution, pkg_resources.WorkingSet) -> Set[pkg_resources.Distribution]
 103         """Given a local distribution and a working set, returns all dependencies from the set.
 104 
 105         :param dist: A single distribution to find the dependencies of
 106         :type dist: :class:`pkg_resources.Distribution`
 107         :param working_set: A working set to search for all packages
 108         :type working_set: :class:`pkg_resources.WorkingSet`
 109         :return: A set of distributions which the package depends on, including the package
 110         :rtype: set(:class:`pkg_resources.Distribution`)
 111         """
 112 
 113         deps = set()
 114         deps.add(dist)
 115         try:
 116             reqs = dist.requires()
 117         # KeyError = limited metadata can be found
 118         except (KeyError, AttributeError, OSError, IOError):  # The METADATA file can't be found
 119             return deps
 120         for req in reqs:
 121             dist = working_set.find(req)
 122             deps |= cls.resolve_dist(dist, working_set)
 123         return deps
 124

 150     @property
 151     def python_info(self):
 152         # type: () -> Dict[str, str]
 153         include_dir = self.prefix / "include"
 154         if not os.path.exists(include_dir):
 155             include_dirs = self.get_include_path()

 158                 if not include_path:

 162         if python_path and python_path.name.startswith("python"):

 164             py_version_short, abiflags = python_version[:3], python_version[3:]

 168     def _replace_parent_version(self, path, replace_version):
 169         # type: (str, str) -> str
 170         if not os.path.exists(path):
 171             base, leaf = os.path.split(path)
 172             base, parent = os.path.split(base)
 173             leaf = os.path.join(parent, leaf).replace(
 174                 replace_version, self.python_info.get("py_version_short", get_python_version())
 175             )
 176             return os.path.join(base, leaf)
 177         return path

 179     @cached_property
 180     def base_paths(self):
 181         # type: () -> Dict[str, str]
 182         """
 183         Returns the context appropriate paths for the environment.
 184 
 185         :return: A dictionary of environment specific paths to be used for installation operations
 186         :rtype: dict
 187 
 188         .. note:: The implementation of this is borrowed from a combination of pip and
 189            virtualenv and is likely to change at some point in the future.
 190 
 191         >>> from pipenv.core import project
 192         >>> from pipenv.environment import Environment
 193         >>> env = Environment(prefix=project.virtualenv_location, is_venv=True, sources=project.sources)
 194         >>> import pprint
 195         >>> pprint.pprint(env.base_paths)
 196         {'PATH': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW/bin::/bin:/usr/bin',
 197         'PYTHONPATH': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW/lib/python3.7/site-packages',
 198         'data': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW',
 199         'include': '/home/hawk/.pyenv/versions/3.7.1/include/python3.7m',
 200         'libdir': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW/lib/python3.7/site-packages',
 201         'platinclude': '/home/hawk/.pyenv/versions/3.7.1/include/python3.7m',
 202         'platlib': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW/lib/python3.7/site-packages',
 203         'platstdlib': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW/lib/python3.7',
 204         'prefix': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW',
 205         'purelib': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW/lib/python3.7/site-packages',
 206         'scripts': '/home/hawk/.virtualenvs/pipenv-MfOPs1lW/bin',
 207         'stdlib': '/home/hawk/.pyenv/versions/3.7.1/lib/python3.7'}
 208         """
 209 
 210         prefix = make_posix(self.prefix.as_posix())
 211         paths = {}
 212         if self._base_paths:
 213             paths = self._base_paths.copy()
 214         else:
 215             try:
 216                 paths = self.get_paths()
 217             except Exception:
 218                 install_scheme = 'nt' if (os.name == 'nt') else 'posix_prefix'
 219                 paths = get_paths(install_scheme, vars={
 220                     'base': prefix,
 221                     'platbase': prefix,
 222                 })
 223                 current_version = get_python_version()
 224                 try:
 225                     for k in list(paths.keys()):
 226                         if not os.path.exists(paths[k]):
 227                             paths[k] = self._replace_parent_version(paths[k], current_version)
 228                 except OSError:
 229                     # Sometimes virtualenvs are made using virtualenv interpreters and there is no
 230                     # include directory, which will cause this approach to fail. This failsafe
 231                     # will make sure we fall back to the shell execution to find the real include path
 232                     paths = self.get_include_path()
 233                     paths.update(self.get_lib_paths())
 234                     paths["scripts"] = self.script_basedir
 235         if not paths:
 236             install_scheme = 'nt' if (os.name == 'nt') else 'posix_prefix'
 237             paths = get_paths(install_scheme, vars={
 238                 'base': prefix,
 239                 'platbase': prefix,
 240             })
 241         if not os.path.exists(paths["purelib"]) and not os.path.exists(paths["platlib"]):
 242             lib_paths = self.get_lib_paths()
 243             paths.update(lib_paths)
 244         paths["PATH"] = paths["scripts"] + os.pathsep + os.defpath
 245         if "prefix" not in paths:
 246             paths["prefix"] = prefix
 247         purelib = paths["purelib"] = make_posix(paths["purelib"])
 248         platlib = paths["platlib"] = make_posix(paths["platlib"])
 249         if purelib == platlib:
 250             lib_dirs = purelib
 251         else:
 252             lib_dirs = purelib + os.pathsep + platlib
 253         paths["libdir"] = purelib
 254         paths['PYTHONPATH'] = os.pathsep.join(["", ".", lib_dirs])
 255         paths["libdirs"] = lib_dirs
 256         return paths

 285     @cached_property
 286     def sys_path(self):
 287         # type: () -> List[str]
 288         """
 289         The system path inside the environment
 290 
 291         :return: The :data:`sys.path` from the environment
 292         :rtype: list
 293         """
 294 
 295         from .vendor.vistir.compat import JSONDecodeError
 296         current_executable = vistir.compat.Path(sys.executable).as_posix()
 297         if not self.python or self.python == current_executable:
 298             return sys.path
 299         elif any([sys.prefix == self.prefix, not self.is_venv]):
 300             return sys.path
 301         cmd_args = [self.python, "-c", "import json, sys; print(json.dumps(sys.path))"]
 302         path, _ = vistir.misc.run(cmd_args, return_object=False, nospin=True, block=True, combine_stderr=False, write_to_stdout=False)
 303         try:
 304             path = json.loads(path.strip())
 305         except JSONDecodeError:
 306             path = sys.path
 307         return path

 309     def build_command(self, python_lib=False, python_inc=False, scripts=False, py_version=False):
 310         # type: (bool, bool, bool, bool) -> str
 311         """Build the text for running a command in the given environment
 312 
 313         :param python_lib: Whether to include the python lib dir commands, defaults to False
 314         :type python_lib: bool, optional
 315         :param python_inc: Whether to include the python include dir commands, defaults to False
 316         :type python_inc: bool, optional
 317         :param scripts: Whether to include the scripts directory, defaults to False
 318         :type scripts: bool, optional
 319         :param py_version: Whether to include the python version info, defaults to False
 320         :type py_version: bool, optional
 321         :return: A string representing the command to run
 322         """
 323         pylib_lines = []
 324         pyinc_lines = []
 325         py_command = (
 326             "import sysconfig, distutils.sysconfig, io, json, sys; paths = {{"
 327             "%s }}; value = u'{{0}}'.format(json.dumps(paths));"
 328             "fh = io.open('{0}', 'w'); fh.write(value); fh.close()"
 329         )
 330         distutils_line = "distutils.sysconfig.get_python_{0}(plat_specific={1})"
 331         sysconfig_line = "sysconfig.get_path('{0}')"
 332         if python_lib:
 333             for key, var, val in (("pure", "lib", "0"), ("plat", "lib", "1")):
 334                 dist_prefix = "{0}lib".format(key)
 335                 # XXX: We need to get 'stdlib' or 'platstdlib'
 336                 sys_prefix = "{0}stdlib".format("" if key == "pure" else key)
 337                 pylib_lines.append("u'%s': u'{{0}}'.format(%s)" % (dist_prefix, distutils_line.format(var, val)))
 338                 pylib_lines.append("u'%s': u'{{0}}'.format(%s)" % (sys_prefix, sysconfig_line.format(sys_prefix)))
 339         if python_inc:
 340             for key, var, val in (("include", "inc", "0"), ("platinclude", "inc", "1")):
 341                 pylib_lines.append("u'%s': u'{{0}}'.format(%s)" % (key, distutils_line.format(var, val)))
 342         lines = pylib_lines + pyinc_lines
 343         if scripts:
 344             lines.append("u'scripts': u'{{0}}'.format(%s)" % sysconfig_line.format("scripts"))
 345         if py_version:
 346             lines.append("u'py_version_short': u'{{0}}'.format(distutils.sysconfig.get_python_version()),")
 347         lines_as_str = u",".join(lines)
 348         py_command = py_command % lines_as_str
 349         return py_command

 351     def get_paths(self):
 352         # type: () -> Optional[Dict[str, str]]
 353         """
 354         Get the paths for the environment by running a subcommand
 355 
 356         :return: The python paths for the environment
 357         :rtype: Dict[str, str]
 358         """
 359         tmpfile = vistir.path.create_tracked_tempfile(suffix=".json")
 360         tmpfile.close()
 361         tmpfile_path = make_posix(tmpfile.name)
 362         py_command = self.build_command(python_lib=True, python_inc=True, scripts=True, py_version=True)
 363         command = [self.python, "-c", py_command.format(tmpfile_path)]
 364         c = vistir.misc.run(
 365             command, return_object=True, block=True, nospin=True, write_to_stdout=False
 366         )
 367         if c.returncode == 0:
 368             paths = {}
 369             with io.open(tmpfile_path, "r", encoding="utf-8") as fh:
 370                 paths = json.load(fh)
 371             if "purelib" in paths:
 372                 paths["libdir"] = paths["purelib"] = make_posix(paths["purelib"])
 373             for key in ("platlib", "scripts", "platstdlib", "stdlib", "include", "platinclude"):
 374                 if key in paths:
 375                     paths[key] = make_posix(paths[key])
 376             return paths
 377         else:
 378             vistir.misc.echo("Failed to load paths: {0}".format(c.err), fg="yellow")
 379             vistir.misc.echo("Output: {0}".format(c.out), fg="yellow")
 380         return None

 383         # type: () -> Dict[str, str]

 396         )
 397         paths = None
 398         if c.returncode == 0:
 399             paths = {}
 400             with io.open(tmpfile_path, "r", encoding="utf-8") as fh:
 401                 paths = json.load(fh)

 521     def expand_egg_links(self):
 522         # type: () -> None
 523         """
 524         Expand paths specified in egg-link files to prevent pip errors during
 525         reinstall
 526         """
 527         prefixes = [
 528             vistir.compat.Path(prefix)
 529             for prefix in self.base_paths["libdirs"].split(os.pathsep)
 530             if vistir.path.is_in_path(prefix, self.prefix.as_posix())
 531         ]
 532         for loc in prefixes:
 533             if not loc.exists():
 534                 continue
 535             for pth in loc.iterdir():
 536                 if not pth.suffix == ".egg-link":
 537                     continue
 538                 contents = [
 539                     vistir.path.normalize_path(line.strip())
 540                     for line in pth.read_text().splitlines()
 541                 ]
 542                 pth.write_text("\n".join(contents))

 559     def find_egg(self, egg_dist):
 560         # type: (pkg_resources.Distribution) -> str
 561         """Find an egg by name in the given environment"""
 562         site_packages = self.libdir[1]
 563         search_filename = "{0}.egg-link".format(egg_dist.project_name)
 564         try:
 565             user_site = site.getusersitepackages()
 566         except AttributeError:
 567             user_site = site.USER_SITE
 568         search_locations = [site_packages, user_site]
 569         for site_directory in search_locations:
 570             egg = os.path.join(site_directory, search_filename)
 571             if os.path.isfile(egg):
 572                 return egg

 584         # type: (pkg_resources.Distribution) -> bool
 585         """Determine whether the supplied distribution is in the environment."""
 586         from .project import _normalized
 587         prefixes = [
 588             _normalized(prefix) for prefix in self.base_paths["libdirs"].split(os.pathsep)

 624     def get_package_info(self, pre=False):
 625         # type: (bool) -> Generator[pkg_resources.Distribution, None, None]
 626         from .vendor.pip_shims.shims import pip_version, parse_version
 627         dependency_links = []
 628         packages = self.get_installed_packages()
 629         # This code is borrowed from pip's current implementation
 630         if parse_version(pip_version) < parse_version("19.0"):
 631             for dist in packages:
 632                 if dist.has_metadata('dependency_links.txt'):
 633                     dependency_links.extend(
 634                         dist.get_metadata_lines('dependency_links.txt')
 635                     )
 636 
 637         with self.get_finder() as finder:
 638             if parse_version(pip_version) < parse_version("19.0"):
 639                 finder.add_dependency_links(dependency_links)
 640 
 641             for dist in packages:
 642                 typ = 'unknown'
 643                 all_candidates = finder.find_all_candidates(dist.key)
 644                 if not self.pipfile.get("pre", finder.allow_all_prereleases):
 645                     # Remove prereleases
 646                     all_candidates = [
 647                         candidate for candidate in all_candidates
 648                         if not candidate.version.is_prerelease
 649                     ]
 650 
 651                 if not all_candidates:
 652                     continue
 653                 candidate_evaluator = finder.make_candidate_evaluator(project_name=dist.key)
 654                 best_candidate_result = candidate_evaluator.compute_best_candidate(all_candidates)
 655                 remote_version = best_candidate_result.best_candidate.version
 656                 if best_candidate_result.best_candidate.link.is_wheel:
 657                     typ = 'wheel'
 658                 else:
 659                     typ = 'sdist'
 660                 # This is dirty but makes the rest of the code much cleaner
 661                 dist.latest_version = remote_version
 662                 dist.latest_filetype = typ
 663                 yield dist

 672     @classmethod
 673     def _get_requirements_for_package(cls, node, key_tree, parent=None, chain=None):
 674         if chain is None:
 675             chain = [node.project_name]
 676 
 677         d = node.as_dict()
 678         if parent:
 679             d['required_version'] = node.version_spec if node.version_spec else 'Any'
 680         else:
 681             d['required_version'] = d['installed_version']
 682 
 683         get_children = lambda n: key_tree.get(n.key, [])    # noqa
 684 
 685         d['dependencies'] = [
 686             cls._get_requirements_for_package(c, key_tree, parent=node,
 687                                               chain=chain+[c.project_name])
 688             for c in get_children(node)
 689             if c.project_name not in chain
 690         ]
 691 
 692         return d

 694     def get_package_requirements(self, pkg=None):
 695         from .vendor.pipdeptree import flatten, sorted_tree, build_dist_index, construct_tree
 696         packages = self.get_installed_packages()
 697         if pkg:
 698             packages = [p for p in packages if p.key == pkg]
 699         dist_index = build_dist_index(packages)
 700         tree = sorted_tree(construct_tree(dist_index))
 701         branch_keys = set(r.key for r in flatten(tree.values()))
 702         if pkg is not None:
 703             nodes = [p for p in tree.keys() if p.key == pkg]
 704         else:
 705             nodes = [p for p in tree.keys() if p.key not in branch_keys]
 706         key_tree = dict((k.key, v) for k, v in tree.items())
 707 
 708         return [self._get_requirements_for_package(p, key_tree) for p in nodes]

 710     @classmethod
 711     def reverse_dependency(cls, node):
 712         new_node = {
 713             "package_name": node["package_name"],
 714             "installed_version": node["installed_version"],
 715             "required_version": node["required_version"]
 716         }
 717         for dependency in node.get("dependencies", []):
 718             for dep in cls.reverse_dependency(dependency):
 719                 new_dep = dep.copy()
 720                 new_dep["parent"] = (node["package_name"], node["installed_version"])
 721                 yield new_dep
 722         yield new_node

 724     def reverse_dependencies(self):
 725         from vistir.misc import unnest, chunked
 726         rdeps = {}
 727         for req in self.get_package_requirements():
 728             for d in self.reverse_dependency(req):
 729                 parents = None
 730                 name = d["package_name"]
 731                 pkg = {
 732                     name: {
 733                         "installed": d["installed_version"],
 734                         "required": d["required_version"]
 735                     }
 736                 }
 737                 parents = tuple(d.get("parent", ()))
 738                 pkg[name]["parents"] = parents
 739                 if rdeps.get(name):
 740                     if not (rdeps[name].get("required") or rdeps[name].get("installed")):
 741                         rdeps[name].update(pkg[name])
 742                     rdeps[name]["parents"] = rdeps[name].get("parents", ()) + parents
 743                 else:
 744                     rdeps[name] = pkg[name]
 745         for k in list(rdeps.keys()):
 746             entry = rdeps[k]
 747             if entry.get("parents"):
 748                 rdeps[k]["parents"] = set([
 749                    p for p, version in chunked(2, unnest(entry["parents"]))
 750                 ])
 751         return rdeps

 773     def is_satisfied(self, req):
 774         match = next(
 775             iter(
 776                 d for d in self.get_distributions()
 777                 if canonicalize_name(d.project_name) == req.normalized_name
 778             ), None
 779         )
 780         if match is not None:
 781             if req.editable and req.line_instance.is_local and self.find_egg(match):
 782                 requested_path = req.line_instance.path
 783                 return requested_path and vistir.compat.samefile(requested_path, match.location)
 784             elif match.has_metadata("direct_url.json"):
 785                 direct_url_metadata = json.loads(match.get_metadata("direct_url.json"))
 786                 commit_id = direct_url_metadata.get("vcs_info", {}).get("commit_id", "")
 787                 vcs_type = direct_url_metadata.get("vcs_info", {}).get("vcs", "")
 788                 _, pipfile_part = req.as_pipfile().popitem()
 789                 return (
 790                     vcs_type == req.vcs and commit_id == req.commit_hash
 791                     and direct_url_metadata["url"] == pipfile_part[req.vcs]
 792                 )
 793             elif req.line_instance.specifiers is not None:
 794                 return req.line_instance.specifiers.contains(
 795                     match.version, prereleases=True
 796                 )
 797             return True
 798         return False

 845     @contextlib.contextmanager
 846     def activated(self, include_extras=True, extra_dists=None):
 847         """Helper context manager to activate the environment.
 848 
 849         This context manager will set the following variables for the duration
 850         of its activation:
 851 
 852             * sys.prefix
 853             * sys.path
 854             * os.environ["VIRTUAL_ENV"]
 855             * os.environ["PATH"]
 856 
 857         In addition, it will make any distributions passed into `extra_dists` available
 858         on `sys.path` while inside the context manager, as well as making `passa` itself
 859         available.
 860 
 861         The environment's `prefix` as well as `scripts_dir` properties are both prepended
 862         to `os.environ["PATH"]` to ensure that calls to `~Environment.run()` use the
 863         environment's path preferentially.
 864         """
 865 
 866         if not extra_dists:
 867             extra_dists = []
 868         original_path = sys.path
 869         original_prefix = sys.prefix
 870         parent_path = vistir.compat.Path(__file__).absolute().parent
 871         vendor_dir = parent_path.joinpath("vendor").as_posix()
 872         patched_dir = parent_path.joinpath("patched").as_posix()
 873         parent_path = parent_path.as_posix()
 874         self.add_dist("pip")
 875         prefix = self.prefix.as_posix()
 876         with vistir.contextmanagers.temp_environ(), vistir.contextmanagers.temp_path():
 877             os.environ["PATH"] = os.pathsep.join([
 878                 vistir.compat.fs_str(self.script_basedir),
 879                 vistir.compat.fs_str(self.prefix.as_posix()),
 880                 os.environ.get("PATH", "")
 881             ])
 882             os.environ["PYTHONIOENCODING"] = vistir.compat.fs_str("utf-8")
 883             os.environ["PYTHONDONTWRITEBYTECODE"] = vistir.compat.fs_str("1")
 884             from .environments import PIPENV_USE_SYSTEM
 885             if self.is_venv:
 886                 os.environ["PYTHONPATH"] = self.base_paths["PYTHONPATH"]
 887                 os.environ["VIRTUAL_ENV"] = vistir.compat.fs_str(prefix)
 888             else:
 889                 if not PIPENV_USE_SYSTEM and not os.environ.get("VIRTUAL_ENV"):
 890                     os.environ["PYTHONPATH"] = self.base_paths["PYTHONPATH"]
 891                     os.environ.pop("PYTHONHOME", None)
 892             sys.path = self.sys_path
 893             sys.prefix = self.sys_prefix
 894             site.addsitedir(self.base_paths["purelib"])
 895             pip = self.safe_import("pip")       # noqa
 896             pip_vendor = self.safe_import("pip._vendor")
 897             pep517_dir = os.path.join(os.path.dirname(pip_vendor.__file__), "pep517")
 898             site.addsitedir(pep517_dir)
 899             os.environ["PYTHONPATH"] = os.pathsep.join([
 900                 os.environ.get("PYTHONPATH", self.base_paths["PYTHONPATH"]), pep517_dir
 901             ])
 902             if include_extras:
 903                 site.addsitedir(parent_path)
 904                 sys.path.extend([parent_path, patched_dir, vendor_dir])
 905                 extra_dists = list(self.extra_dists) + extra_dists
 906                 for extra_dist in extra_dists:
 907                     if extra_dist not in self.get_working_set():
 908                         extra_dist.activate(self.sys_path)
 909             try:
 910                 yield
 911             finally:
 912                 sys.path = original_path
 913                 sys.prefix = original_prefix
 914                 six.moves.reload_module(pkg_resources)

 981     @contextlib.contextmanager
 982     def uninstall(self, pkgname, *args, **kwargs):
 983         """A context manager which allows uninstallation of packages from the environment
 984 
 985         :param str pkgname: The name of a package to uninstall
 986 
 987         >>> env = Environment("/path/to/env/root")
 988         >>> with env.uninstall("pytz", auto_confirm=True, verbose=False) as uninstaller:
 989                 cleaned = uninstaller.paths
 990         >>> if cleaned:
 991                 print("uninstalled packages: %s" % cleaned)
 992         """
 993 
 994         auto_confirm = kwargs.pop("auto_confirm", True)
 995         verbose = kwargs.pop("verbose", False)
 996         with self.activated():
 997             monkey_patch = next(iter(
 998                 dist for dist in self.base_working_set
 999                 if dist.project_name == "recursive-monkey-patch"
1000             ), None)
1001             if monkey_patch:
1002                 monkey_patch.activate()
1003             pip_shims = self.safe_import("pip_shims")
1004             pathset_base = pip_shims.UninstallPathSet
1005             pathset_base._permitted = PatchedUninstaller._permitted
1006             dist = next(
1007                 iter(d for d in self.get_working_set() if d.project_name == pkgname),
1008                 None
1009             )
1010             pathset = pathset_base.from_dist(dist)
1011             if pathset is not None:
1012                 pathset.remove(auto_confirm=auto_confirm, verbose=verbose)
1013             try:
1014                 yield pathset
1015             except Exception:
1016                 if pathset is not None:
1017                     pathset.rollback()
1018             else:
1019                 if pathset is not None:
1020                     pathset.commit()
1021             if pathset is None:
1022                 return
