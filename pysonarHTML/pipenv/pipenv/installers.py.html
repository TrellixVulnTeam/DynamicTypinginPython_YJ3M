  13 @attr.s
  14 class Version(object):
  15 
  16     major = attr.ib()
  17     minor = attr.ib()
  18     patch = attr.ib()
  19 
  20     def __str__(self):
  21         parts = [self.major, self.minor]
  22         if self.patch is not None:
  23             parts.append(self.patch)
  24         return '.'.join(str(p) for p in parts)
  25 
  26     @classmethod
  27     def parse(cls, name):
  28         """Parse an X.Y.Z or X.Y string into a version tuple.
  29         """
  30         match = re.match(r'^(\d+)\.(\d+)(?:\.(\d+))?$', name)
  31         if not match:
  32             raise ValueError('invalid version name {0!r}'.format(name))
  33         major = int(match.group(1))
  34         minor = int(match.group(2))
  35         patch = match.group(3)
  36         if patch is not None:
  37             patch = int(patch)
  38         return cls(major, minor, patch)
  39 
  40     @property
  41     def cmpkey(self):
  42         """Make the version a comparable tuple.
  43 
  44         Some old Python versions does not have a patch part, e.g. 2.7.0 is
  45         named "2.7" in pyenv. Fix that, otherwise `None` will fail to compare
  46         with int.
  47         """
  48         return (self.major, self.minor, self.patch or 0)
  49 
  50     def matches_minor(self, other):
  51         """Check whether this version matches the other in (major, minor).
  52         """
  53         return (self.major, self.minor) == (other.major, other.minor)

  81     @staticmethod
  82     def _find_python_installer_by_name_and_env(name, env_var):
  83         """
  84         Given a python installer (pyenv or asdf), try to locate the binary for that
  85         installer.
  86 
  87         pyenv/asdf are not always present on PATH. Both installers also support a
  88         custom environment variable (PYENV_ROOT or ASDF_DIR) which alows them to
  89         be installed into a non-default location (the default/suggested source
  90         install location is in ~/.pyenv or ~/.asdf).
  91 
  92         For systems without the installers on PATH, and with a custom location
  93         (e.g. /opt/pyenv), Pipenv can use those installers without modifications to
  94         PATH, if an installer's respective environment variable is present in an
  95         environment's .env file.
  96 
  97         This function searches for installer binaries in the following locations,
  98         by precedence:
  99             1. On PATH, equivalent to which(1).
 100             2. In the "bin" subdirectory of PYENV_ROOT or ASDF_DIR, depending on the
 101                installer.
 102             3. In ~/.pyenv/bin or ~/.asdf/bin, depending on the installer.
 103         """
 104         for candidate in (
 105             # Look for the Python installer using the equivalent of 'which'. On
 106             # Homebrew-installed systems, the env var may not be set, but this
 107             # strategy will work.
 108             find_windows_executable('', name),
 109             # Check for explicitly set install locations (e.g. PYENV_ROOT, ASDF_DIR).
 110             os.path.join(os.path.expanduser(os.getenv(env_var, '/dev/null')), 'bin', name),
 111             # Check the pyenv/asdf-recommended from-source install locations
 112             os.path.join(os.path.expanduser('~/.{}'.format(name)), 'bin', name),
 113         ):
 114             if candidate is not None and os.path.isfile(candidate) and os.access(candidate, os.X_OK):
 115                 return candidate
 116         raise InstallerNotFound()

 118     def _run(self, *args, **kwargs):
 119         timeout = kwargs.pop('timeout', delegator.TIMEOUT)
 120         if kwargs:
 121             k = list(kwargs.keys())[0]
 122             raise TypeError('unexpected keyword argument {0!r}'.format(k))
 123         args = (self.cmd,) + tuple(args)
 124         c = delegator.run(args, block=False, timeout=timeout)
 125         c.block()
 126         if c.return_code != 0:
 127             raise InstallerError('failed to run {0}'.format(args), c)
 128         return c

 136     def find_version_to_install(self, name):
 137         """Find a version in the installer from the version supplied.
 138 
 139         A ValueError is raised if a matching version cannot be found.
 140         """
 141         version = Version.parse(name)
 142         if version.patch is not None:
 143             return name
 144         try:
 145             best_match = max((
 146                 inst_version
 147                 for inst_version in self.iter_installable_versions()
 148                 if inst_version.matches_minor(version)
 149             ), key=operator.attrgetter('cmpkey'))
 150         except ValueError:
 151             raise ValueError(
 152                 'no installable version found for {0!r}'.format(name),
 153             )
 154         return best_match

 169 class Pyenv(Installer):
 170 
 171     def _find_installer(self):
 172         return self._find_python_installer_by_name_and_env('pyenv', 'PYENV_ROOT')
 173 
 174     def iter_installable_versions(self):
 175         """Iterate through CPython versions available for Pipenv to install.
 176         """
 177         for name in self._run('install', '--list').out.splitlines():
 178             try:
 179                 version = Version.parse(name.strip())
 180             except ValueError:
 181                 continue
 182             yield version
 183 
 184     def install(self, version):
 185         """Install the given version with pyenv.
 186         The version must be a ``Version`` instance representing a version
 187         found in pyenv.
 188         A ValueError is raised if the given version does not have a match in
 189         pyenv. A InstallerError is raised if the pyenv command fails.
 190         """
 191         c = self._run(
 192             'install', '-s', str(version),
 193             timeout=PIPENV_INSTALL_TIMEOUT,
 194         )
 195         return c
