  32 def _get_activate_script(cmd, venv):
  33     """Returns the string to activate a virtualenv.
  34 
  35     This is POSIX-only at the moment since the compat (pexpect-based) shell
  36     does not work elsewhere anyway.
  37     """
  38     # Suffix and source command for other shells.
  39     # Support for fish shell.
  40     if "fish" in cmd:
  41         suffix = ".fish"
  42         command = "source"
  43     # Support for csh shell.
  44     elif "csh" in cmd:
  45         suffix = ".csh"
  46         command = "source"
  47     else:
  48         suffix = ""
  49         command = "."
  50     # Escape any spaces located within the virtualenv path to allow
  51     # for proper activation.
  52     venv_location = str(venv).replace(" ", r"\ ")
  53     # The leading space can make history cleaner in some shells.
  54     return " {2} {0}/bin/activate{1}".format(venv_location, suffix, command)

  57 def _handover(cmd, args):
  58     args = [cmd] + args
  59     if os.name != "nt":
  60         os.execvp(cmd, args)
  61     else:
  62         sys.exit(subprocess.call(args, shell=True, universal_newlines=True))

  86     def fork(self, venv, cwd, args):
  87         # FIXME: This isn't necessarily the correct prompt. We should read the
  88         # actual prompt by peeking into the activation script.
  89         name = os.path.basename(venv)
  90         os.environ["VIRTUAL_ENV"] = str(venv)
  91         if "PROMPT" in os.environ:
  92             os.environ["PROMPT"] = "({0}) {1}".format(name, os.environ["PROMPT"])
  93         if "PS1" in os.environ:
  94             os.environ["PS1"] = "({0}) {1}".format(name, os.environ["PS1"])
  95         with self.inject_path(venv):
  96             os.chdir(cwd)
  97             _handover(self.cmd, self.args + list(args))

  99     def fork_compat(self, venv, cwd, args):
 100         from .vendor import pexpect
 101 
 102         # Grab current terminal dimensions to replace the hardcoded default
 103         # dimensions of pexpect.
 104         dims = get_terminal_size()
 105         with temp_environ():
 106             c = pexpect.spawn(self.cmd, ["-i"], dimensions=(dims.lines, dims.columns))
 107         c.sendline(_get_activate_script(self.cmd, venv))
 108         if args:
 109             c.sendline(" ".join(args))
 110 
 111         # Handler for terminal resizing events
 112         # Must be defined here to have the shell process in its context, since
 113         # we can't pass it as an argument
 114         def sigwinch_passthrough(sig, data):
 115             dims = get_terminal_size()
 116             c.setwinsize(dims.lines, dims.columns)
 117 
 118         signal.signal(signal.SIGWINCH, sigwinch_passthrough)
 119 
 120         # Interact with the new shell.
 121         c.interact(escape_character=None)
 122         c.close()
 123         sys.exit(c.exitstatus)

 136 class Bash(Shell):
 137     def _format_path(self, python):
 138         return python.parent.as_posix()
 139 
 140     # The usual PATH injection technique does not work with Bash.
 141     # https://github.com/berdario/pew/issues/58#issuecomment-102182346
 142     @contextlib.contextmanager
 143     def inject_path(self, venv):
 144         from ._compat import NamedTemporaryFile
 145 
 146         bashrc_path = Path.home().joinpath(".bashrc")
 147         with NamedTemporaryFile("w+") as rcfile:
 148             if bashrc_path.is_file():
 149                 base_rc_src = 'source "{0}"\n'.format(bashrc_path.as_posix())
 150                 rcfile.write(base_rc_src)
 151 
 152             export_path = 'export PATH="{0}:$PATH"\n'.format(":".join(
 153                 self._format_path(python)
 154                 for python in _iter_python(venv)
 155             ))
 156             rcfile.write(export_path)
 157             rcfile.flush()
 158             self.args.extend(["--rcfile", rcfile.name])
 159             yield

 162 class MsysBash(Bash):
 163     def _format_path(self, python):
 164         s = super(MsysBash, self)._format_path(python)
 165         if not python.drive:
 166             return s
 167         # Convert "C:/something" to "/c/something".
 168         return '/{drive}{path}'.format(drive=s[0].lower(), path=s[2:])

 204 # Two dimensional dict. First is the shell type, second is the emulator type.
 205 # Example: SHELL_LOOKUP['powershell']['cmder'] => CmderPowershell.
 206 SHELL_LOOKUP = collections.defaultdict(
 207     lambda: collections.defaultdict(lambda: Shell),
 208     {
 209         "bash": collections.defaultdict(
 210             lambda: Bash, {"msys": MsysBash},
 211         ),
 212         "cmd": collections.defaultdict(
 213             lambda: Shell, {"cmder": CmderCommandPrompt},
 214         ),
 215         "powershell": collections.defaultdict(
 216             lambda: Shell, {"cmder": CmderPowershell},
 217         ),
 218         "pwsh": collections.defaultdict(
 219             lambda: Shell, {"cmder": CmderPowershell},
 220         ),
 221     },

 225 def _detect_emulator():
 226     keys = []
 227     if os.environ.get("CMDER_ROOT"):
 228         keys.append("cmder")
 229     if os.environ.get("MSYSTEM"):
 230         keys.append("msys")
 231     return ",".join(keys)

 234 def choose_shell():
 235     emulator = PIPENV_EMULATOR.lower() or _detect_emulator()
 236     type_, command = detect_info()
 237     shell_types = SHELL_LOOKUP[type_]
 238     for key in emulator.split(","):
 239         key = key.strip().lower()
 240         if key in shell_types:
 241             return shell_types[key](command)
 242     return shell_types[""](command)
