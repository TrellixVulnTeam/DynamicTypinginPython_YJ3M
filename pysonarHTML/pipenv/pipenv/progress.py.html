  12 import os
  13 import sys
  14 import time

  16 import crayons
  17 
  18 from .environments import PIPENV_COLORBLIND, PIPENV_HIDE_EMOJIS

  21 STREAM = sys.stderr
  22 MILL_TEMPLATE = "%s %s %i/%i\r"
  23 DOTS_CHAR = "."
  24 if PIPENV_HIDE_EMOJIS:
  25     if PIPENV_COLORBLIND:
  26         BAR_FILLED_CHAR = "="
  27         BAR_EMPTY_CHAR = "-"
  28     else:
  29         BAR_FILLED_CHAR = str(crayons.green("=", bold=True))
  30         BAR_EMPTY_CHAR = str(crayons.black("-"))
  31 else:
  32     if PIPENV_COLORBLIND:
  33         BAR_FILLED_CHAR = "▉"
  34         BAR_EMPTY_CHAR = " "
  35     else:
  36         BAR_FILLED_CHAR = str(crayons.green("▉", bold=True))
  37         BAR_EMPTY_CHAR = str(crayons.black("▉"))
  38 
  39 if (sys.version_info[0] >= 3) and (os.name != "nt"):
  40     BAR_TEMPLATE = u"  %s%s%s %i/%i — {0}\r".format(crayons.black("%s"))
  41 else:
  42     if os.name == "nt":
  43         BAR_TEMPLATE = "  %s%s%s %i/%i - %s\r"
  44     else:
  45         BAR_TEMPLATE = "  %s%s%s %i/%i — %s\r"
  46 MILL_CHARS = ["|", "/", "-", "\\"]
  47 # How long to wait before recalculating the ETA
  48 ETA_INTERVAL = 1
  49 # How many intervals (excluding the current one) to calculate the simple moving
  50 # average
  51 ETA_SMA_WINDOW = 9
  52 
  53 
  54 class Bar(object):
  55     def __enter__(self):
  56         return self
  57 
  58     def __exit__(self, exc_type, exc_val, exc_tb):
  59         self.done()
  60         return False  # we're not suppressing exceptions
  61 
  62     def __init__(
  63         self,
  64         label="",
  65         width=32,
  66         hide=None,
  67         empty_char=BAR_EMPTY_CHAR,
  68         filled_char=BAR_FILLED_CHAR,
  69         expected_size=None,
  70         every=1,
  71     ):
  72         self.label = label
  73         self.width = width
  74         self.hide = hide
  75         # Only show bar in terminals by default (better for piping, logging etc.)
  76         if hide is None:
  77             try:
  78                 self.hide = not STREAM.isatty()
  79             except AttributeError:  # output does not support isatty()
  80                 self.hide = True
  81         self.empty_char = empty_char
  82         self.filled_char = filled_char
  83         self.expected_size = expected_size
  84         self.every = every
  85         self.start = time.time()
  86         self.ittimes = []
  87         self.eta = 0
  88         self.etadelta = time.time()
  89         self.etadisp = self.format_time(self.eta)
  90         self.last_progress = 0
  91         if self.expected_size:
  92             self.show(0)
  93 
  94     def show(self, progress, count=None):
  95         if count is not None:
  96             self.expected_size = count
  97         if self.expected_size is None:
  98             raise Exception("expected_size not initialized")
  99 
 100         self.last_progress = progress
 101         if (time.time() - self.etadelta) > ETA_INTERVAL:
 102             self.etadelta = time.time()
 103             self.ittimes = self.ittimes[-ETA_SMA_WINDOW:] + [
 104                 -(self.start - time.time()) / (progress + 1)
 105             ]
 106             self.eta = (
 107                 sum(self.ittimes)
 108                 / float(len(self.ittimes))
 109                 * (self.expected_size - progress)
 110             )
 111             self.etadisp = self.format_time(self.eta)
 112         x = int(self.width * progress / self.expected_size)
 113         if not self.hide:
 114             if (
 115                 progress % self.every == 0  # True every "every" updates
 116                 or progress == self.expected_size  # And when we're done
 117             ):
 118                 STREAM.write(
 119                     BAR_TEMPLATE
 120                     % (
 121                         self.label,
 122                         self.filled_char * x,
 123                         self.empty_char * (self.width - x),
 124                         progress,
 125                         self.expected_size,
 126                         self.etadisp,
 127                     )
 128                 )
 129                 STREAM.flush()
 130 
 131     def done(self):
 132         self.elapsed = time.time() - self.start
 133         elapsed_disp = self.format_time(self.elapsed)
 134         if not self.hide:
 135             # Print completed bar with elapsed time
 136             STREAM.write(
 137                 BAR_TEMPLATE
 138                 % (
 139                     self.label,
 140                     self.filled_char * self.width,
 141                     self.empty_char * 0,
 142                     self.last_progress,
 143                     self.expected_size,
 144                     elapsed_disp,
 145                 )
 146             )
 147             STREAM.write("\n")
 148             STREAM.flush()
 149 
 150     def format_time(self, seconds):
 151         return time.strftime("%H:%M:%S", time.gmtime(seconds))

 154 def bar(
 155     it,
 156     label="",
 157     width=32,
 158     hide=None,
 159     empty_char=BAR_EMPTY_CHAR,
 160     filled_char=BAR_FILLED_CHAR,
 161     expected_size=None,
 162     every=1,
 163 ):
 164     """Progress iterator. Wrap your iterables with it."""
 165     count = len(it) if expected_size is None else expected_size
 166     with Bar(
 167         label=label,
 168         width=width,
 169         hide=hide,
 170         empty_char=BAR_EMPTY_CHAR,
 171         filled_char=BAR_FILLED_CHAR,
 172         expected_size=count,
 173         every=every,
 174     ) as bar:
 175         for i, item in enumerate(it):
 176             yield item
 177 
 178             bar.show(i + 1)

 181 def dots(it, label="", hide=None, every=1):
 182     """Progress iterator. Prints a dot for each item being iterated"""
 183     count = 0
 184     if not hide:
 185         STREAM.write(label)
 186     for i, item in enumerate(it):
 187         if not hide:
 188             if i % every == 0:  # True every "every" updates
 189                 STREAM.write(DOTS_CHAR)
 190                 sys.stderr.flush()
 191         count += 1
 192         yield item
 193 
 194     STREAM.write("\n")
 195     STREAM.flush()

 198 def mill(it, label="", hide=None, expected_size=None, every=1):
 199     """Progress iterator. Prints a mill while iterating over the items."""
 200 
 201     def _mill_char(_i):
 202         if _i >= count:
 203             return " "
 204 
 205         else:
 206             return MILL_CHARS[(_i // every) % len(MILL_CHARS)]
 207 
 208     def _show(_i):
 209         if not hide:
 210             if (
 211                 _i % every == 0  # True every "every" updates
 212                 or _i == count  # And when we're done
 213             ):
 214                 STREAM.write(MILL_TEMPLATE % (label, _mill_char(_i), _i, count))
 215                 STREAM.flush()
 216 
 217     count = len(it) if expected_size is None else expected_size
 218     if count:
 219         _show(0)
 220     for i, item in enumerate(it):
 221         yield item
 222 
 223         _show(i + 1)
 224     if not hide:
 225         STREAM.write("\n")
 226         STREAM.flush()
