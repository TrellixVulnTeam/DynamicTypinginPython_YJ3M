  18 class Script(object):
  19     """Parse a script line (in Pipfile's [scripts] section).
  20 
  21     This always works in POSIX mode, even on Windows.
  22     """
  23 
  24     def __init__(self, command, args=None):
  25         self._parts = [command]
  26         if args:
  27             self._parts.extend(args)

  29     @classmethod

  31         if isinstance(value, six.string_types):
  32             value = shlex.split(value)
  33         if not value:
  34             raise ScriptEmptyError(value)
  35         return cls(value[0], value[1:])

  51     def cmdify(self):
  52         """Encode into a cmd-executable string.
  53 
  54         This re-implements CreateProcess's quoting logic to turn a list of
  55         arguments into one single string for the shell to interpret.
  56 
  57         * All double quotes are escaped with a backslash.
  58         * Existing backslashes before a quote are doubled, so they are all
  59           escaped properly.
  60         * Backslashes elsewhere are left as-is; cmd will interpret them
  61           literally.
  62 
  63         The result is then quoted into a pair of double quotes to be grouped.
  64 
  65         An argument is intentionally not quoted if it does not contain
  66         foul characters. This is done to be compatible with Windows built-in
  67         commands that don't work well with quotes, e.g. everything with `echo`,
  68         and DOS-style (forward slash) switches.
  69 
  70         Foul characters include:
  71 
  72         * Whitespaces.
  73         * Carets (^). (pypa/pipenv#3307)
  74         * Parentheses in the command. (pypa/pipenv#3168)
  75 
  76         Carets introduce a difficult situation since they are essentially
  77         "lossy" when parsed. Consider this in cmd.exe::
  78 
  79             > echo "foo^bar"
  80             "foo^bar"
  81             > echo foo^^bar
  82             foo^bar
  83 
  84         The two commands produce different results, but are both parsed by the
  85         shell as `foo^bar`, and there's essentially no sensible way to tell
  86         what was actually passed in. This implementation assumes the quoted
  87         variation (the first) since it is easier to implement, and arguably
  88         the more common case.
  89 
  90         The intended use of this function is to pre-process an argument list
  91         before passing it into ``subprocess.Popen(..., shell=True)``.
  92 
  93         See also: https://docs.python.org/3/library/subprocess.html#converting-argument-sequence
  94         """
  95         return " ".join(itertools.chain(
  96             [_quote_if_contains(self.command, r'[\s^()]')],
  97             (_quote_if_contains(arg, r'[\s^]') for arg in self.args),
  98         ))
