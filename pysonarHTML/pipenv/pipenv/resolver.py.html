  13 def find_site_path(pkg, site_dir=None):
  14     import pkg_resources
  15     if site_dir is not None:
  16         site_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
  17     working_set = pkg_resources.WorkingSet([site_dir] + sys.path[:])
  18     for dist in working_set:
  19         root = dist.location
  20         base_name = dist.project_name if dist.project_name else dist.key
  21         name = None
  22         if "top_level.txt" in dist.metadata_listdir(""):
  23             name = next(iter([l.strip() for l in dist.get_metadata_lines("top_level.txt") if l is not None]), None)
  24         if name is None:
  25             name = pkg_resources.safe_name(base_name).replace("-", "_")
  26         if not any(pkg == _ for _ in [base_name, name]):
  27             continue
  28         path_options = [name, "{0}.py".format(name)]
  29         path_options = [os.path.join(root, p) for p in path_options if p is not None]
  30         path = next(iter(p for p in path_options if os.path.exists(p)), None)
  31         if path is not None:
  32             return (dist, path)
  33     return (None, None)

  36 def _patch_path(pipenv_site=None):
  37     import site
  38     pipenv_libdir = os.path.dirname(os.path.abspath(__file__))
  39     pipenv_site_dir = os.path.dirname(pipenv_libdir)
  40     pipenv_dist = None
  41     if pipenv_site is not None:
  42         pipenv_dist, pipenv_path = find_site_path("pipenv", site_dir=pipenv_site)
  43     else:
  44         pipenv_dist, pipenv_path = find_site_path("pipenv", site_dir=pipenv_site_dir)
  45     if pipenv_dist is not None:
  46         pipenv_dist.activate()
  47     else:
  48         site.addsitedir(next(iter(
  49             sitedir for sitedir in (pipenv_site, pipenv_site_dir)
  50             if sitedir is not None
  51         ), None))
  52     if pipenv_path is not None:
  53         pipenv_libdir = pipenv_path
  54     for _dir in ("vendor", "patched", pipenv_libdir):
  55         sys.path.insert(0, os.path.join(pipenv_libdir, _dir))

 147     @classmethod
 148     def parse_pyparsing_exprs(cls, expr_iterable):
 149         from pipenv.vendor.pyparsing import Literal, MatchFirst
 150         keys = []
 151         expr_list = []
 152         expr = expr_iterable.copy()
 153         if isinstance(expr, Literal) or (
 154             expr.__class__.__name__ == Literal.__name__
 155         ):
 156             keys.append(expr.match)
 157         elif isinstance(expr, MatchFirst) or (
 158             expr.__class__.__name__ == MatchFirst.__name__
 159         ):
 160             expr_list = expr.exprs
 161         elif isinstance(expr, list):
 162             expr_list = expr
 163         if expr_list:
 164             for part in expr_list:
 165                 keys.extend(cls.parse_pyparsing_exprs(part))
 166         return keys

 168     @classmethod
 169     def get_markers_from_dict(cls, entry_dict):
 170         from pipenv.vendor.packaging import markers as packaging_markers
 171         from pipenv.vendor.requirementslib.models.markers import normalize_marker_str
 172         marker_keys = cls.parse_pyparsing_exprs(packaging_markers.VARIABLE)
 173         markers = set()
 174         keys_in_dict = [k for k in marker_keys if k in entry_dict]
 175         markers = {
 176             normalize_marker_str("{k} {v}".format(k=k, v=entry_dict.pop(k)))
 177             for k in keys_in_dict
 178         }
 179         if "markers" in entry_dict:
 180             markers.add(normalize_marker_str(entry_dict["markers"]))
 181         if None in markers:
 182             markers.remove(None)
 183         if markers:
 184             entry_dict["markers"] = " and ".join(list(markers))
 185         else:
 186             markers = None
 187         return markers, entry_dict

 214     @staticmethod
 215     def marker_to_str(marker):
 216         from pipenv.vendor.requirementslib.models.markers import normalize_marker_str
 217         if not marker:
 218             return None
 219         from pipenv.vendor import six
 220         from pipenv.vendor.vistir.compat import Mapping
 221         marker_str = None
 222         if isinstance(marker, Mapping):
 223             marker_dict, _ = Entry.get_markers_from_dict(marker)
 224             if marker_dict:
 225                 marker_str = "{0}".format(marker_dict.popitem()[1])
 226         elif isinstance(marker, (list, set, tuple)):
 227             marker_str = " and ".join([normalize_marker_str(m) for m in marker if m])
 228         elif isinstance(marker, six.string_types):
 229             marker_str = "{0}".format(normalize_marker_str(marker))
 230         if isinstance(marker_str, six.string_types):
 231             return marker_str
 232         return None

 234     def get_cleaned_dict(self, keep_outdated=False):
 235         if keep_outdated and self.is_updated:
 236             self.validate_constraints()
 237             self.ensure_least_updates_possible()
 238         elif not keep_outdated:
 239             self.validate_constraints()
 240         if self.entry.extras != self.lockfile_entry.extras:
 241             entry_extras = list(self.entry.extras)
 242             if self.lockfile_entry.extras:
 243                 entry_extras.extend(list(self.lockfile_entry.extras))
 244             self._entry.req.extras = entry_extras
 245             self.entry_dict["extras"] = self.entry.extras
 246         if self.original_markers and not self.markers:
 247             original_markers = self.marker_to_str(self.original_markers)
 248             self.markers = original_markers
 249             self.entry_dict["markers"] = self.marker_to_str(original_markers)
 250         entry_hashes = set(self.entry.hashes)
 251         locked_hashes = set(self.lockfile_entry.hashes)
 252         if entry_hashes != locked_hashes and not self.is_updated:
 253             self.entry_dict["hashes"] = list(entry_hashes | locked_hashes)
 254         self.entry_dict["name"] = self.name
 255         if "version" in self.entry_dict:
 256             self.entry_dict["version"] = self.strip_version(self.entry_dict["version"])
 257         _, self.entry_dict = self.get_markers_from_dict(self.entry_dict)
 258         return self.entry_dict

 314     @staticmethod
 315     def clean_specifier(specifier):
 316         from pipenv.vendor.packaging.specifiers import Specifier
 317         if not any(specifier.startswith(k) for k in Specifier._operators.keys()):
 318             if specifier.strip().lower() in ["any", "<any>", "*"]:
 319                 return "*"
 320             specifier = "=={0}".format(specifier)
 321         elif specifier.startswith("==") and specifier.count("=") > 3:
 322             specifier = "=={0}".format(specifier.lstrip("="))
 323         return specifier

 325     @staticmethod
 326     def strip_version(specifier):
 327         from pipenv.vendor.packaging.specifiers import Specifier
 328         op = next(iter(
 329             k for k in Specifier._operators.keys() if specifier.startswith(k)
 330         ), None)
 331         if op:
 332             specifier = specifier[len(op):]
 333         while op:
 334             op = next(iter(
 335                 k for k in Specifier._operators.keys() if specifier.startswith(k)
 336             ), None)
 337             if op:
 338                 specifier = specifier[len(op):]
 339         return specifier

 408     def get_parent_deps(self, unnest=False):
 409         from pipenv.vendor.packaging.specifiers import Specifier
 410         parents = []
 411         for spec in self.reverse_deps.get(self.normalized_name, {}).get("parents", set()):
 412             spec_match = next(iter(c for c in Specifier._operators if c in spec), None)
 413             name = spec
 414             parent = None
 415             if spec_match is not None:
 416                 spec_index = spec.index(spec_match)
 417                 specifier = self.clean_specifier(spec[spec_index:len(spec_match)]).strip()
 418                 name_start = spec_index + len(spec_match)
 419                 name = spec[name_start:].strip()
 420                 parent = self.create_parent(name, specifier)
 421             else:
 422                 name = spec
 423                 parent = self.create_parent(name)
 424             if parent is not None:
 425                 parents.append(parent)
 426             if not unnest or parent.pipfile_name is not None:
 427                 continue
 428             if self.reverse_deps.get(parent.normalized_name, {}).get("parents", set()):
 429                 parents.extend(parent.flattened_parents)
 430         return parents

 432     def ensure_least_updates_possible(self):
 433         """
 434         Mutate the current entry to ensure that we are making the smallest amount of
 435         changes possible to the existing lockfile -- this will keep the old locked
 436         versions of packages if they satisfy new constraints.
 437 
 438         :return: None
 439         """
 440         constraints = self.get_constraints()
 441         can_use_original = True
 442         can_use_updated = True
 443         satisfied_by_versions = set()
 444         for constraint in constraints:
 445             if not constraint.specifier.contains(self.original_version):
 446                 self.can_use_original = False
 447             if not constraint.specifier.contains(self.updated_version):
 448                 self.can_use_updated = False
 449             satisfied_by_value = getattr(constraint, "satisfied_by", None)
 450             if satisfied_by_value:
 451                 satisfied_by = "{0}".format(
 452                     self.clean_specifier(str(satisfied_by_value.version))
 453                 )
 454                 satisfied_by_versions.add(satisfied_by)
 455         if can_use_original:
 456             self.entry_dict = self.lockfile_dict.copy()
 457         elif can_use_updated:
 458             if len(satisfied_by_versions) == 1:
 459                 self.entry_dict["version"] = next(iter(
 460                     sat_by for sat_by in satisfied_by_versions if sat_by
 461                 ), None)
 462                 hashes = None
 463                 if self.lockfile_entry.specifiers == satisfied_by:
 464                     ireq = self.lockfile_entry.as_ireq()
 465                     if not self.lockfile_entry.hashes and self.resolver._should_include_hash(ireq):
 466                         hashes = self.resolver.get_hash(ireq)
 467                     else:
 468                         hashes = self.lockfile_entry.hashes
 469                 else:
 470                     if self.resolver._should_include_hash(constraint):
 471                         hashes = self.resolver.get_hash(constraint)
 472                 if hashes:
 473                     self.entry_dict["hashes"] = list(hashes)
 474                     self._entry.hashes = frozenset(hashes)
 475         else:
 476             # check for any parents, since they depend on this and the current
 477             # installed versions are not compatible with the new version, so
 478             # we will need to update the top level dependency if possible
 479             self.check_flattened_parents()

 481     def get_constraints(self):
 482         """
 483         Retrieve all of the relevant constraints, aggregated from the pipfile, resolver,
 484         and parent dependencies and their respective conflict resolution where possible.
 485 
 486         :return: A set of **InstallRequirement** instances representing constraints
 487         :rtype: Set
 488         """
 489         constraints = {
 490             c for c in self.resolver.parsed_constraints
 491             if c and c.name == self.entry.name
 492         }
 493         pipfile_constraint = self.get_pipfile_constraint()
 494         if pipfile_constraint and not (self.pipfile_entry.editable or pipfile_constraint.editable):
 495             constraints.add(pipfile_constraint)
 496         return constraints
 497 
 498     def get_pipfile_constraint(self):
 499         """
 500         Retrieve the version constraint from the pipfile if it is specified there,
 501         otherwise check the constraints of the parent dependencies and their conflicts.
 502 
 503         :return: An **InstallRequirement** instance representing a version constraint
 504         """
 505         if self.is_in_pipfile:
 506             return self.pipfile_entry.as_ireq()
 507         return self.constraint_from_parent_conflicts()

 509     def constraint_from_parent_conflicts(self):
 510         """
 511         Given a resolved entry with multiple parent dependencies with different
 512         constraints, searches for the resolution that satisfies all of the parent
 513         constraints.
 514 
 515         :return: A new **InstallRequirement** satisfying all parent constraints
 516         :raises: :exc:`~pipenv.exceptions.DependencyConflict` if resolution is impossible
 517         """
 518         # ensure that we satisfy the parent dependencies of this dep
 519         parent_dependencies = set()
 520         has_mismatch = False
 521         can_use_original = True
 522         for p in self.parent_deps:
 523             # updated dependencies should be satisfied since they were resolved already
 524             if p.is_updated:
 525                 continue
 526             # parents with no requirements can't conflict
 527             if not p.requirements:
 528                 continue
 529             entry_ref = p.get_dependency(self.name)
 530             required = entry_ref.get("required_version", "*")
 531             required = self.clean_specifier(required)
 532             parent_requires = self.make_requirement(self.name, required)
 533             parent_dependencies.add("{0} => {1} ({2})".format(p.name, self.name, required))
 534             # use pre=True here or else prereleases dont satisfy constraints
 535             if parent_requires.requirement.specifier and (
 536                 not parent_requires.requirement.specifier.contains(self.original_version, prereleases=True)
 537             ):
 538                 can_use_original = False
 539             if parent_requires.requirement.specifier and (
 540                 not parent_requires.requirement.specifier.contains(self.updated_version, prereleases=True)
 541             ):
 542                 if not self.entry.editable and self.updated_version != self.original_version:
 543                     has_mismatch = True
 544         if has_mismatch and not can_use_original:
 545             from pipenv.exceptions import DependencyConflict
 546             msg = (
 547                 "Cannot resolve {0} ({1}) due to conflicting parent dependencies: "
 548                 "\n\t{2}".format(
 549                     self.name, self.updated_version, "\n\t".join(parent_dependencies)
 550                 )
 551             )
 552             raise DependencyConflict(msg)
 553         elif can_use_original:
 554             return self.lockfile_entry.as_ireq()
 555         return self.entry.as_ireq()

 557     def validate_constraints(self):
 558         """
 559         Retrieves the full set of available constraints and iterate over them, validating
 560         that they exist and that they are not causing unresolvable conflicts.
 561 
 562         :return: True if the constraints are satisfied by the resolution provided
 563         :raises: :exc:`pipenv.exceptions.DependencyConflict` if the constraints dont exist
 564         """
 565         constraints = self.get_constraints()
 566         for constraint in constraints:
 567             try:
 568                 constraint.check_if_exists(False)
 569             except Exception:
 570                 from pipenv.exceptions import DependencyConflict
 571                 from pipenv.environments import is_verbose
 572                 if is_verbose():
 573                     print("Tried constraint: {0!r}".format(constraint), file=sys.stderr)
 574                 msg = (
 575                     "Cannot resolve conflicting version {0}{1} while {2}{3} is "
 576                     "locked.".format(
 577                         self.name, self.updated_specifier, self.old_name, self.old_specifiers
 578                     )
 579                 )
 580                 raise DependencyConflict(msg)
 581         return True

 583     def check_flattened_parents(self):
 584         for parent in self.parents_in_pipfile:
 585             if not parent.updated_specifier:
 586                 continue
 587             if not parent.validate_specifiers():
 588                 from pipenv.exceptions import DependencyConflict
 589                 msg = (
 590                     "Cannot resolve conflicting versions: (Root: {0}) {1}{2} (Pipfile) "
 591                     "Incompatible with {3}{4} (resolved)\n".format(
 592                         self.name, parent.pipfile_name,
 593                         parent.pipfile_entry.requirement.specifiers, parent.name,
 594                         parent.updated_specifiers
 595                     )
 596                 )
 597                 raise DependencyConflict(msg)

 599     def __getattribute__(self, key):
 600         result = None
 601         old_version = ["was_", "had_", "old_"]
 602         new_version = ["is_", "has_", "new_"]
 603         if any(key.startswith(v) for v in new_version):
 604             entry = Entry.__getattribute__(self, "entry")
 605             try:
 606                 keystart = key.index("_") + 1
 607                 try:
 608                     result = getattr(entry, key[keystart:])
 609                 except AttributeError:
 610                     result = getattr(entry, key)
 611             except AttributeError:
 612                 result = super(Entry, self).__getattribute__(key)
 613             return result
 614         if any(key.startswith(v) for v in old_version):
 615             lockfile_entry = Entry.__getattribute__(self, "lockfile_entry")
 616             try:
 617                 keystart = key.index("_") + 1
 618                 try:
 619                     result = getattr(lockfile_entry, key[keystart:])
 620                 except AttributeError:
 621                     result = getattr(lockfile_entry, key)
 622             except AttributeError:
 623                 result = super(Entry, self).__getattribute__(key)
 624             return result
 625         return super(Entry, self).__getattribute__(key)

 628 def clean_results(results, resolver, project, dev=False):
 629     from pipenv.utils import translate_markers
 630     if not project.lockfile_exists:
 631         return results
 632     lockfile = project.lockfile_content
 633     section = "develop" if dev else "default"
 634     reverse_deps = project.environment.reverse_dependencies()
 635     new_results = [r for r in results if r["name"] not in lockfile[section]]
 636     for result in results:
 637         name = result.get("name")
 638         entry_dict = result.copy()
 639         entry = Entry(name, entry_dict, project, resolver, reverse_deps=reverse_deps, dev=dev)
 640         entry_dict = translate_markers(entry.get_cleaned_dict(keep_outdated=False))
 641         new_results.append(entry_dict)
 642     return new_results

 645 def clean_outdated(results, resolver, project, dev=False):
 646     if not project.lockfile_exists:
 647         return results
 648     lockfile = project.lockfile_content
 649     section = "develop" if dev else "default"
 650     reverse_deps = project.environment.reverse_dependencies()
 651     new_results = [r for r in results if r["name"] not in lockfile[section]]
 652     for result in results:
 653         name = result.get("name")
 654         entry_dict = result.copy()
 655         entry = Entry(name, entry_dict, project, resolver, reverse_deps=reverse_deps, dev=dev)
 656         # The old entry was editable but this one isnt; prefer the old one
 657         # TODO: Should this be the case for all locking?
 658         if entry.was_editable and not entry.is_editable:
 659             continue
 660         lockfile_entry = lockfile[section].get(name, None)
 661         if not lockfile_entry:
 662             alternate_section = "develop" if not dev else "default"
 663             if name in lockfile[alternate_section]:
 664                 lockfile_entry = lockfile[alternate_section][name]
 665         if lockfile_entry and not entry.is_updated:
 666             old_markers = next(iter(m for m in (
 667                 entry.lockfile_entry.markers, lockfile_entry.get("markers", None)
 668             ) if m is not None), None)
 669             new_markers = entry_dict.get("markers", None)
 670             if old_markers:
 671                 old_markers = Entry.marker_to_str(old_markers)
 672             if old_markers and not new_markers:
 673                 entry.markers = old_markers
 674             elif new_markers and not old_markers:
 675                 del entry.entry_dict["markers"]
 676                 entry._entry.req.req.marker = None
 677                 entry._entry.markers = None
 678             # if the entry has not changed versions since the previous lock,
 679             # don't introduce new markers since that is more restrictive
 680             # if entry.has_markers and not entry.had_markers and not entry.is_updated:
 681             # do make sure we retain the original markers for entries that are not changed
 682         entry_dict = entry.get_cleaned_dict(keep_outdated=True)
 683         new_results.append(entry_dict)
 684     return new_results

 687 def parse_packages(packages, pre, clear, system, requirements_dir=None):
 688     from pipenv.vendor.requirementslib.models.requirements import Requirement
 689     from pipenv.vendor.vistir.contextmanagers import cd, temp_path
 690     from pipenv.utils import parse_indexes
 691     parsed_packages = []
 692     for package in packages:
 693         indexes, trusted_hosts, line = parse_indexes(package)
 694         line = " ".join(line)
 695         pf = dict()
 696         req = Requirement.from_line(line)
 697         if not req.name:
 698             with temp_path(), cd(req.req.setup_info.base_dir):
 699                 sys.path.insert(0, req.req.setup_info.base_dir)
 700                 req.req._setup_info.get_info()
 701                 req.update_name_from_path(req.req.setup_info.base_dir)
 702         try:
 703             name, entry = req.pipfile_entry
 704         except Exception:
 705             continue
 706         else:
 707             if name is not None and entry is not None:
 708                 pf[name] = entry
 709                 parsed_packages.append(pf)
 710     print("RESULTS:")
 711     if parsed_packages:
 712         print(json.dumps(parsed_packages))
 713     else:
 714         print(json.dumps([]))

 717 def resolve_packages(pre, clear, verbose, system, write, requirements_dir, packages):
 718     from pipenv.utils import create_mirror_source, resolve_deps, replace_pypi_sources
 719     pypi_mirror_source = (
 720         create_mirror_source(os.environ["PIPENV_PYPI_MIRROR"])
 721         if "PIPENV_PYPI_MIRROR" in os.environ
 722         else None
 723     )
 724 
 725     def resolve(packages, pre, project, sources, clear, system, requirements_dir=None):
 726         from pipenv.patched.piptools import logging as piptools_logging
 727         piptools_logging.log.verbosity = 1 if verbose else 0
 728         return resolve_deps(
 729             packages,
 730             which,
 731             project=project,
 732             pre=pre,
 733             sources=sources,
 734             clear=clear,
 735             allow_global=system,
 736             req_dir=requirements_dir
 737         )
 738 
 739     from pipenv.core import project
 740     sources = (
 741         replace_pypi_sources(project.pipfile_sources, pypi_mirror_source)
 742         if pypi_mirror_source
 743         else project.pipfile_sources
 744     )
 745     keep_outdated = os.environ.get("PIPENV_KEEP_OUTDATED", False)
 746     results, resolver = resolve(
 747         packages,
 748         pre=pre,
 749         project=project,
 750         sources=sources,
 751         clear=clear,
 752         system=system,
 753         requirements_dir=requirements_dir,
 754     )
 755     if keep_outdated:
 756         results = clean_outdated(results, resolver, project)
 757     else:
 758         results = clean_results(results, resolver, project)
 759     if write:
 760         with open(write, "w") as fh:
 761             if not results:
 762                 json.dump([], fh)
 763             else:
 764                 json.dump(results, fh)
 765     else:
 766         print("RESULTS:")
 767         if results:
 768             print(json.dumps(results))
 769         else:
 770             print(json.dumps([]))
