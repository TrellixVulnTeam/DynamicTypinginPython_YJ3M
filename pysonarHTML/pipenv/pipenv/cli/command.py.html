 319 @cli.command(short_help="Generates Pipfile.lock.", context_settings=CONTEXT_SETTINGS)
 320 @lock_options
 321 @pass_state
 322 @pass_context
 323 def lock(
 324     ctx,
 325     state,
 326     **kwargs
 327 ):
 328     """Generates Pipfile.lock."""
 329     from ..core import ensure_project, do_init, do_lock
 330     # Ensure that virtualenv is available.
 331     # Note that we don't pass clear on to ensure_project as it is also
 332     # handled in do_lock
 333     ensure_project(
 334         three=state.three, python=state.python, pypi_mirror=state.pypi_mirror,
 335         warn=(not state.quiet), site_packages=state.site_packages,
 336     )
 337     emit_requirements = state.lockoptions.emit_requirements
 338     dev = state.installstate.dev
 339     dev_only = state.lockoptions.dev_only
 340     pre = state.installstate.pre
 341     if emit_requirements:
 342         # Emit requirements file header (unless turned off with --no-header)
 343         if state.lockoptions.emit_requirements_header:
 344             header_options = ["--requirements"]
 345             if dev_only:
 346                 header_options.append("--dev-only")
 347             elif dev:
 348                 header_options.append("--dev")
 349             echo(LOCK_HEADER.format(options=" ".join(header_options)))
 350             # TODO: Emit pip-compile style header
 351             if dev and not dev_only:
 352                 echo(LOCK_DEV_NOTE)
 353         # Setting "emit_requirements=True" means do_init() just emits the
 354         # install requirements file to stdout, it doesn't install anything
 355         do_init(
 356             dev=dev,
 357             dev_only=dev_only,
 358             emit_requirements=emit_requirements,
 359             pypi_mirror=state.pypi_mirror,
 360             pre=pre,
 361         )
 362     elif state.lockoptions.dev_only:
 363         raise PipenvOptionsError(
 364             "--dev-only",
 365             "--dev-only is only permitted in combination with --requirements. "
 366             "Aborting."
 367         )
 368     do_lock(
 369         ctx=ctx,
 370         clear=state.clear,
 371         pre=pre,
 372         keep_outdated=state.installstate.keep_outdated,
 373         pypi_mirror=state.pypi_mirror,
 374         write=not state.quiet,
 375     )

 631     """View a given module in your editor.
 632 
 633     This uses the EDITOR environment variable. You can temporarily override it,
 634     for example:
