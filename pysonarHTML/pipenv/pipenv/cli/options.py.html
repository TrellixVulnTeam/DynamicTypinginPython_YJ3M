  23 class PipenvGroup(DYMMixin, Group):
  24     """Custom Group class provides formatted main help"""
  25 
  26     def get_help_option(self, ctx):
  27         from ..core import format_help
  28 
  29         """Override for showing formatted main help via --help and -h options"""
  30         help_options = self.get_help_option_names(ctx)
  31         if not help_options or not self.add_help_option:
  32             return
  33 
  34         def show_help(ctx, param, value):
  35             if value and not ctx.resilient_parsing:
  36                 if not ctx.invoked_subcommand:
  37                     # legit main help
  38                     echo(format_help(ctx.get_help()))
  39                 else:
  40                     # legit sub-command help
  41                     echo(ctx.get_help(), color=ctx.color)
  42                 ctx.exit()
  43 
  44         return Option(
  45             help_options,
  46             is_flag=True,
  47             is_eager=True,
  48             expose_value=False,
  49             callback=show_help,
  50             help="Show this message and exit.",
  51         )

 376 def setup_verbosity(ctx, param, value):
 377     if not value:
 378         return
 379     import logging
 380     loggers = ("pip", "piptools")
 381     if value == 1:
 382         for logger in loggers:
 383             logging.getLogger(logger).setLevel(logging.INFO)
 384     elif value == -1:
 385         for logger in loggers:
 386             logging.getLogger(logger).setLevel(logging.CRITICAL)
 387     environments.PIPENV_VERBOSITY = value

 390 def validate_python_path(ctx, param, value):
 391     # Validating the Python path is complicated by accepting a number of
 392     # friendly options: the default will be boolean False to enable
 393     # autodetection but it may also be a value which will be searched in
 394     # the path or an absolute path. To report errors as early as possible
 395     # we'll report absolute paths which do not exist:
 396     if isinstance(value, (str, bytes)):
 397         if os.path.isabs(value) and not os.path.isfile(value):
 398             raise BadParameter("Expected Python at path %s does not exist" % value)
 399     return value
 400 
 401 
 402 def validate_bool_or_none(ctx, param, value):
 403     if value is not None:
 404         return click.types.BOOL(value)
 405     return False

 408 def validate_pypi_mirror(ctx, param, value):
 409     if value and not is_valid_url(value):
 410         raise BadParameter("Invalid PyPI mirror URL: %s" % value)
 411     return value
