  34 def getpreferredencoding():
  35     import locale
  36     # Borrowed from Invoke
  37     # (see https://github.com/pyinvoke/invoke/blob/93af29d/invoke/runners.py#L881)
  38     _encoding = locale.getpreferredencoding(False)
  39     if six.PY2 and not sys.platform == "win32":
  40         _default_encoding = locale.getdefaultlocale()[1]
  41         if _default_encoding is not None:
  42             _encoding = _default_encoding
  43     return _encoding

  49 def canonical_encoding_name(name):
  50     import codecs
  51     try:
  52         codec = codecs.lookup(name)
  53     except LookupError:
  54         return name
  55     else:
  56         return codec.name

  61 def force_encoding():
  62     try:
  63         stdout_isatty = sys.stdout.isatty
  64         stderr_isatty = sys.stderr.isatty
  65     except AttributeError:
  66         return DEFAULT_ENCODING, DEFAULT_ENCODING
  67     else:
  68         if not (stdout_isatty() and stderr_isatty()):
  69             return DEFAULT_ENCODING, DEFAULT_ENCODING
  70     stdout_encoding = canonical_encoding_name(sys.stdout.encoding)
  71     stderr_encoding = canonical_encoding_name(sys.stderr.encoding)
  72     if sys.platform == "win32" and sys.version_info >= (3, 1):
  73         return DEFAULT_ENCODING, DEFAULT_ENCODING
  74     if stdout_encoding != "utf-8" or stderr_encoding != "utf-8":
  75 
  76         try:
  77             from ctypes import pythonapi, py_object, c_char_p
  78         except ImportError:
  79             return DEFAULT_ENCODING, DEFAULT_ENCODING
  80         try:
  81             PyFile_SetEncoding = pythonapi.PyFile_SetEncoding
  82         except AttributeError:
  83             return DEFAULT_ENCODING, DEFAULT_ENCODING
  84         else:
  85             PyFile_SetEncoding.argtypes = (py_object, c_char_p)
  86             if stdout_encoding != "utf-8":
  87                 try:
  88                     was_set = PyFile_SetEncoding(sys.stdout, "utf-8")
  89                 except OSError:
  90                     was_set = False
  91                 if not was_set:
  92                     stdout_encoding = DEFAULT_ENCODING
  93                 else:
  94                     stdout_encoding = "utf-8"
  95 
  96             if stderr_encoding != "utf-8":
  97                 try:
  98                     was_set = PyFile_SetEncoding(sys.stderr, "utf-8")
  99                 except OSError:
 100                     was_set = False
 101                 if not was_set:
 102                     stderr_encoding = DEFAULT_ENCODING
 103                 else:
 104                     stderr_encoding = "utf-8"
 105 
 106     return stdout_encoding, stderr_encoding

 126 def decode_output(output):
 127     if not isinstance(output, six.string_types):
 128         return output
 129     try:
 130         output = output.encode(DEFAULT_ENCODING)
 131     except (AttributeError, UnicodeDecodeError, UnicodeEncodeError):
 132         if six.PY2:
 133             output = unicode.translate(vistir.misc.to_text(output),  # noqa
 134                                        UNICODE_TO_ASCII_TRANSLATION_MAP)
 135         else:
 136             output = output.translate(UNICODE_TO_ASCII_TRANSLATION_MAP)
 137         output = output.encode(DEFAULT_ENCODING, "replace")
 138     return vistir.misc.to_text(output, encoding=DEFAULT_ENCODING, errors="replace")

 141 def fix_utf8(text):
 142     if not isinstance(text, six.string_types):
 143         return text
 144     try:
 145         text = decode_output(text)
 146     except UnicodeDecodeError:
 147         if six.PY2:
 148             text = unicode.translate(vistir.misc.to_text(text), UNICODE_TO_ASCII_TRANSLATION_MAP)   # noqa
 149     return text
