  66 def cleanup_toml(tml):
  67     toml = tml.split("\n")
  68     new_toml = []
  69     # Remove all empty lines from TOML.
  70     for line in toml:
  71         if line.strip():
  72             new_toml.append(line)
  73     toml = "\n".join(new_toml)
  74     new_toml = []
  75     # Add newlines between TOML sections.
  76     for i, line in enumerate(toml.split("\n")):
  77         # Skip the first line.
  78         if line.startswith("["):
  79             if i > 0:
  80                 # Insert a newline before the heading.
  81                 new_toml.append("")
  82         new_toml.append(line)
  83     # adding new line at the end of the TOML file
  84     new_toml.append("")
  85     toml = "\n".join(new_toml)
  86     return toml

 124 def run_command(cmd, *args, **kwargs):
 125     """
 126     Take an input command and run it, handling exceptions and error codes and returning
 127     its stdout and stderr.
 128 
 129     :param cmd: The list of command and arguments.
 130     :type cmd: list
 131     :returns: A 2-tuple of the output and error from the command
 132     :rtype: Tuple[str, str]
 133     :raises: exceptions.PipenvCmdError
 134     """
 135 
 136     from pipenv.vendor import delegator
 137     from ._compat import decode_for_output
 138     from .cmdparse import Script
 139     catch_exceptions = kwargs.pop("catch_exceptions", True)
 140     if isinstance(cmd, (six.string_types, list, tuple)):
 141         cmd = Script.parse(cmd)
 142     if not isinstance(cmd, Script):
 143         raise TypeError("Command input must be a string, list or tuple")
 144     if "env" not in kwargs:
 145         kwargs["env"] = os.environ.copy()
 146     kwargs["env"]["PYTHONIOENCODING"] = "UTF-8"
 147     try:
 148         cmd_string = cmd.cmdify()
 149     except TypeError:
 150         click_echo("Error turning command into string: {0}".format(cmd), err=True)
 151         sys.exit(1)
 152     if environments.is_verbose():
 153         click_echo("Running command: $ {0}".format(cmd_string, err=True))
 154     c = delegator.run(cmd_string, *args, **kwargs)
 155     return_code = c.return_code
 156     if environments.is_verbose():
 157         click_echo("Command output: {0}".format(
 158             crayons.blue(decode_for_output(c.out))
 159         ), err=True)
 160     if not c.ok and catch_exceptions:
 161         raise PipenvCmdError(cmd_string, c.out, c.err, return_code)
 162     return c

 165 def parse_python_version(output):
 166     """Parse a Python version output returned by `python --version`.
 167 
 168     Return a dict with three keys: major, minor, and micro. Each value is a
 169     string containing a version part.
 170 
 171     Note: The micro part would be `'0'` if it's missing from the input string.
 172     """
 173     version_line = output.split("\n", 1)[0]
 174     version_pattern = re.compile(
 175         r"""
 176         ^                   # Beginning of line.
 177         Python              # Literally "Python".
 178         \s                  # Space.
 179         (?P<major>\d+)      # Major = one or more digits.
 180         \.                  # Dot.
 181         (?P<minor>\d+)      # Minor = one or more digits.
 182         (?:                 # Unnamed group for dot-micro.
 183             \.              # Dot.
 184             (?P<micro>\d+)  # Micro = one or more digit.
 185         )?                  # Micro is optional because pypa/pipenv#1893.
 186         .*                  # Trailing garbage.
 187         $                   # End of line.
 188     """,
 189         re.VERBOSE,
 190     )
 191 
 192     match = version_pattern.match(version_line)
 193     if not match:
 194         return None
 195     return match.groupdict(default="0")

 278                     pip_args.extend(

 555             # Allow users to toggle resolution off for non-editable VCS packages
 556             # but leave it on for local, installable folders on the filesystem
 557             if environments.PIPENV_RESOLVE_VCS or (
 558                 req.editable or parsed_line.is_wheel or (
 559                     req.is_file_or_url and parsed_line.is_local
 560                     and is_installable_dir(parsed_line.path)
 561                 )
 562             ):
 563                 requirements = [v for v in getattr(setup_info, "requires", {}).values()]
 564             for r in requirements:
 565                 if getattr(r, "url", None) and not getattr(r, "editable", False):
 566                     if r is not None:
 567                         if not r.url:
 568                             continue
 569                         line = _requirement_to_str_lowercase_name(r)
 570                         new_req, _, _ = cls.parse_line(line)
 571                         if r.marker and not r.marker.evaluate():
 572                             new_constraints = {}
 573                             _, new_entry = req.pipfile_entry
 574                             new_lock = {
 575                                 pep423_name(new_req.normalized_name): new_entry
 576                             }
 577                         else:
 578                             new_constraints, new_lock = cls.get_deps_from_req(
 579                                 new_req, resolver
 580                             )
 581                         locked_deps.update(new_lock)
 582                         constraints |= new_constraints
 583                 # if there is no marker or there is a valid marker, add the constraint line
 584                 elif r and (not r.marker or (r.marker and r.marker.evaluate())):
 585                     line = _requirement_to_str_lowercase_name(r)
 586                     constraints.add(line)
 587             # ensure the top level entry remains as provided
 588             # note that we shouldn't pin versions for editable vcs deps
 589             if not req.is_vcs:
 590                 if req.specifiers:
 591                     locked_deps[name]["version"] = req.specifiers
 592                 elif parsed_line.setup_info and parsed_line.setup_info.version:
 593                     locked_deps[name]["version"] = "=={}".format(
 594                         parsed_line.setup_info.version
 595                     )
 596             # if not req.is_vcs:
 597             locked_deps.update({name: entry})
 598             if req.is_vcs and req.editable:
 599                 constraints.add(req.constraint_line)
 600             if req.is_file_or_url and req.req.is_local and req.editable and (
 601                     req.req.setup_path is not None and os.path.exists(req.req.setup_path)):
 602                 constraints.add(req.constraint_line)
 603         else:
 604             # if the dependency isn't installable, don't add it to constraints
 605             # and instead add it directly to the lock
 606             if req and req.requirement and (
 607                 req.requirement.marker and not req.requirement.marker.evaluate()
 608             ):
 609                 pypi = resolver.repository if resolver else None
 610                 best_match = pypi.find_best_match(req.ireq) if pypi else None
 611                 if best_match:
 612                     hashes = resolver.collect_hashes(best_match) if resolver else []
 613                     new_req = Requirement.from_ireq(best_match)
 614                     new_req = new_req.add_hashes(hashes)
 615                     name, entry = new_req.pipfile_entry
 616                     locked_deps[pep423_name(name)] = translate_markers(entry)
 617                 return constraints, locked_deps
 618             constraints.add(req.constraint_line)
 619             return constraints, locked_deps
 620         return constraints, locked_deps

 866     def collect_hashes(self, ireq):
 867         from .vendor.requests import ConnectionError
 868         collected_hashes = []
 869         if ireq in self.hashes:
 870             collected_hashes += list(self.hashes.get(ireq, []))
 871         if self._should_include_hash(ireq):
 872             try:
 873                 hash_map = self.get_hash(ireq)
 874                 collected_hashes += list(hash_map)
 875             except (ValueError, KeyError, IndexError, ConnectionError):
 876                 pass
 877         elif any(
 878             "python.org" in source["url"] or "pypi.org" in source["url"]
 879             for source in self.sources
 880         ):
 881             pkg_url = "https://pypi.org/pypi/{0}/json".format(ireq.name)
 882             session = _get_requests_session()

 888                 for api_version, api_info in api_releases.items():
 889                     api_version = clean_pkg_version(api_version)
 890                     cleaned_releases[api_version] = api_info

 892                 if ireq.specifier:

 897                     collected_hashes.append(release["digests"]["sha256"])

 908     @staticmethod
 909     def _should_include_hash(ireq):
 910         from pipenv.vendor.vistir.compat import Path, to_native_string
 911         from pipenv.vendor.vistir.path import url_to_path
 912 
 913         # We can only hash artifacts.
 914         try:
 915             if not ireq.link.is_artifact:
 916                 return False
 917         except AttributeError:
 918             return False
 919 
 920         # But we don't want normal pypi artifcats since the normal resolver
 921         # handles those
 922         if is_pypi_url(ireq.link.url):
 923             return False
 924 
 925         # We also don't want to try to hash directories as this will fail
 926         # as these are editable deps and are not hashable.
 927         if (
 928             ireq.link.scheme == "file"
 929             and Path(to_native_string(url_to_path(ireq.link.url))).is_dir()
 930         ):
 931             return False
 932         return True

 934     def get_hash(self, ireq, ireq_hashes=None):
 935         """
 936         Retrieve hashes for a specific ``InstallRequirement`` instance.
 937 
 938         :param ireq: An ``InstallRequirement`` to retrieve hashes for
 939         :type ireq: :class:`~pip_shims.InstallRequirement`
 940         :return: A set of hashes.
 941         :rtype: Set
 942         """
 943 
 944         # We _ALWAYS MUST PRIORITIZE_ the inclusion of hashes from local sources
 945         # PLEASE *DO NOT MODIFY THIS* TO CHECK WHETHER AN IREQ ALREADY HAS A HASH
 946         # RESOLVED. The resolver will pull hashes from PyPI and only from PyPI.
 947         # The entire purpose of this approach is to include missing hashes.
 948         # This fixes a race condition in resolution for missing dependency caches
 949         # see pypa/pipenv#3289
 950         if not self._should_include_hash(ireq):
 951             return add_to_set(set(), ireq_hashes)
 952         elif self._should_include_hash(ireq) and (
 953             not ireq_hashes or ireq.link.scheme == "file"
 954         ):
 955             if not ireq_hashes:
 956                 ireq_hashes = set()
 957             new_hashes = self.resolver.repository._hash_cache.get_hash(ireq.link)
 958             ireq_hashes = add_to_set(ireq_hashes, new_hashes)
 959         else:
 960             ireq_hashes = set(ireq_hashes)
 961         # The _ONLY CASE_ where we flat out set the value is if it isn't present
 962         # It's a set, so otherwise we *always* need to do a union update
 963         if ireq not in self.hashes:
 964             return ireq_hashes
 965         else:
 966             return self.hashes[ireq] | ireq_hashes

 972                 self.hashes[ireq] = self.get_hash(ireq, ireq_hashes=ireq_hashes)

 975     def _clean_skipped_result(self, req, value):
 976         ref = None
 977         if req.is_vcs:
 978             ref = req.commit_hash
 979         ireq = req.as_ireq()
 980         entry = value.copy()
 981         entry["name"] = req.name
 982         if entry.get("editable", False) and entry.get("version"):
 983             del entry["version"]
 984         ref = ref if ref is not None else entry.get("ref")
 985         if ref:
 986             entry["ref"] = ref
 987         if self._should_include_hash(ireq):
 988             collected_hashes = self.collect_hashes(ireq)
 989             if collected_hashes:
 990                 entry["hashes"] = sorted(set(collected_hashes))
 991         return req.name, entry

 993     def clean_results(self):
 994         from pipenv.vendor.requirementslib.models.requirements import Requirement
 995         reqs = [(Requirement.from_ireq(ireq), ireq) for ireq in self.resolved_tree]
 996         results = {}
 997         for req, ireq in reqs:
 998             if (req.vcs and req.editable and not req.is_direct_url):
 999                 continue
1000             elif req.normalized_name in self.skipped.keys():
1001                 continue
1002             collected_hashes = self.collect_hashes(ireq)
1003             req = req.add_hashes(collected_hashes)
1004             if not collected_hashes and self._should_include_hash(ireq):
1005                 discovered_hashes = self.hashes.get(ireq, set()) | self.get_hash(ireq)
1006                 if discovered_hashes:
1007                     req = req.add_hashes(discovered_hashes)
1008                 self.hashes[ireq] = collected_hashes = discovered_hashes
1009             if collected_hashes:
1010                 collected_hashes = sorted(set(collected_hashes))
1011             name, entry = format_requirement_for_lockfile(
1012                 req, self.markers_lookup, self.index_lookup, collected_hashes
1013             )
1014             entry = translate_markers(entry)
1015             if name in results:
1016                 results[name].update(entry)
1017             else:
1018                 results[name] = entry
1019         for k in list(self.skipped.keys()):
1020             req = Requirement.from_pipfile(k, self.skipped[k])
1021             name, entry = self._clean_skipped_result(req, self.skipped[k])
1022             entry = translate_markers(entry)
1023             if name in results:
1024                 results[name].update(entry)
1025             else:
1026                 results[name] = entry
1027         results = list(results.values())
1028         return results

1031 def format_requirement_for_lockfile(req, markers_lookup, index_lookup, hashes=None):
1032     if req.specifiers:
1033         version = str(req.get_version())
1034     else:
1035         version = None
1036     index = index_lookup.get(req.normalized_name)
1037     markers = markers_lookup.get(req.normalized_name)
1038     req.index = index
1039     name, pf_entry = req.pipfile_entry
1040     name = pep423_name(req.name)
1041     entry = {}
1042     if isinstance(pf_entry, six.string_types):
1043         entry["version"] = pf_entry.lstrip("=")
1044     else:
1045         entry.update(pf_entry)
1046         if version is not None and not req.is_vcs:
1047             entry["version"] = version
1048         if req.line_instance.is_direct_url and not req.is_vcs:
1049             entry["file"] = req.req.uri
1050     if hashes:
1051         entry["hashes"] = sorted(set(hashes))
1052     entry["name"] = name
1053     if index:
1054         entry.update({"index": index})
1055     if markers:
1056         entry.update({"markers": markers})
1057     entry = translate_markers(entry)
1058     if req.vcs or req.editable:
1059         for key in ("index", "version", "file"):
1060             try:
1061                 del entry[key]
1062             except KeyError:
1063                 pass
1064     return name, entry

1450 def mkdir_p(newdir):
1451     """works the way a good mkdir should :)
1452         - already exists, silently complete
1453         - regular file in the way, raise an exception
1454         - parent directory(ies) does not exist, make them as well
1455         From: http://code.activestate.com/recipes/82465-a-friendly-mkdir/
1456     """
1457     if os.path.isdir(newdir):
1458         pass
1459     elif os.path.isfile(newdir):
1460         raise OSError(
1461             "a file with the same name as the desired dir, '{0}', already exists.".format(
1462                 newdir
1463             )
1464         )
1465 
1466     else:
1467         head, tail = os.path.split(newdir)
1468         if head and not os.path.isdir(head):
1469             mkdir_p(head)
1470         if tail:
1471             # Even though we've checked that the directory doesn't exist above, it might exist
1472             # now if some other process has created it between now and the time we checked it.
1473             try:
1474                 os.mkdir(newdir)
1475             except OSError as exn:
1476                 # If we failed because the directory does exist, that's not a problem -
1477                 # that's what we were trying to do anyway. Only re-raise the exception
1478                 # if we failed for some other reason.
1479                 if exn.errno != errno.EEXIST:
1480                     raise

1646 def walk_up(bottom):
1647     """Mimic os.walk, but walk 'up' instead of down the directory tree.
1648     From: https://gist.github.com/zdavkeos/1098474
1649     """
1650     bottom = os.path.realpath(bottom)
1651     # Get files in current dir.
1652     try:
1653         names = os.listdir(bottom)
1654     except Exception:
1655         return
1656 
1657     dirs, nondirs = [], []
1658     for name in names:
1659         if os.path.isdir(os.path.join(bottom, name)):
1660             dirs.append(name)
1661         else:
1662             nondirs.append(name)
1663     yield bottom, dirs, nondirs
1664 
1665     new_path = os.path.realpath(os.path.join(bottom, ".."))
1666     # See if we are at the top.
1667     if new_path == bottom:
1668         return
1669 
1670     for x in walk_up(new_path):
1671         yield x

1928 def clean_resolved_dep(dep, is_top_level=False, pipfile_entry=None):
1929     from .vendor.requirementslib.utils import is_vcs
1930     name = pep423_name(dep["name"])
1931     lockfile = {}
1932     # We use this to determine if there are any markers on top level packages
1933     # So we can make sure those win out during resolution if the packages reoccur
1934     if "version" in dep and dep["version"] and not dep.get("editable", False):
1935         version = "{0}".format(dep["version"])
1936         if not version.startswith("=="):
1937             version = "=={0}".format(version)
1938         lockfile["version"] = version
1939     if is_vcs(dep):
1940         ref = dep.get("ref", None)
1941         if ref is not None:
1942             lockfile["ref"] = ref

1947             lockfile["subdirectory"] = dep["subdirectory"]
1948     for key in ["hashes", "index", "extras", "editable"]:
1949         if key in dep:
1950             lockfile[key] = dep[key]
1951     # In case we lock a uri or a file when the user supplied a path
1952     # remove the uri or file keys from the entry and keep the path
1953     fs_key = next(iter(k for k in ["path", "file"] if k in dep), None)
1954     pipfile_fs_key = None
1955     if pipfile_entry:
1956         pipfile_fs_key = next(iter(k for k in ["path", "file"] if k in pipfile_entry), None)
1957     if fs_key and pipfile_fs_key and fs_key != pipfile_fs_key:
1958         lockfile[pipfile_fs_key] = pipfile_entry[pipfile_fs_key]
1959     elif fs_key is not None:
1960         lockfile[fs_key] = dep[fs_key]
1961 
1962     # If a package is **PRESENT** in the pipfile but has no markers, make sure we
1963     # **NEVER** include markers in the lockfile
1964     if "markers" in dep and dep.get("markers", "").strip():
1965         # First, handle the case where there is no top level dependency in the pipfile
1966         if not is_top_level:
1967             translated = translate_markers(dep).get("markers", "").strip()
1968             if translated:
1969                 try:
1970                     lockfile["markers"] = translated
1971                 except TypeError:
1972                     pass
1973         # otherwise make sure we are prioritizing whatever the pipfile says about the markers
1974         # If the pipfile says nothing, then we should put nothing in the lockfile
1975         else:
1976             try:
1977                 pipfile_entry = translate_markers(pipfile_entry)
1978                 lockfile["markers"] = pipfile_entry.get("markers")
1979             except TypeError:
1980                 pass
1981     return {name: lockfile}
