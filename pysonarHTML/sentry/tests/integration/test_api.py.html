  13 class AuthenticationTest(AuthProviderTestCase):
  14     def test_sso_auth_required(self):
  15         user = self.create_user("foo@example.com", is_superuser=False)
  16         organization = self.create_organization(name="foo")
  17         team = self.create_team(name="bar", organization=organization)
  18         project = self.create_project(name="baz", organization=organization, teams=[team])
  19         member = self.create_member(user=user, organization=organization, teams=[team])
  20         setattr(member.flags, "sso:linked", True)
  21         member.save()
  22         event = self.store_event(data={}, project_id=project.id)
  23         group_id = event.group_id
  24         auth_provider = AuthProvider.objects.create(
  25             organization=organization, provider="dummy", flags=0
  26         )
  27 
  28         AuthIdentity.objects.create(auth_provider=auth_provider, user=user)
  29 
  30         self.login_as(user)
  31 
  32         paths = (
  33             u"/api/0/organizations/{}/".format(organization.slug),
  34             u"/api/0/projects/{}/{}/".format(organization.slug, project.slug),
  35             u"/api/0/teams/{}/{}/".format(organization.slug, team.slug),
  36             u"/api/0/issues/{}/".format(group_id),
  37             # this uses the internal API, which once upon a time was broken
  38             u"/api/0/issues/{}/events/latest/".format(group_id),
  39         )
  40 
  41         for path in paths:
  42             # we should be redirecting the user to the authentication form as they
  43             # haven't verified this specific organization
  44             resp = self.client.get(path)
  45             assert resp.status_code == 401, (resp.status_code, resp.content)
  46 
  47         # superuser should still require SSO as they're a member of the org
  48         user.update(is_superuser=True)
  49         for path in paths:
  50             resp = self.client.get(path)
  51             assert resp.status_code == 401, (resp.status_code, resp.content)
  52 
  53         # XXX(dcramer): using internal API as exposing a request object is hard
  54         self.session[SSO_SESSION_KEY] = six.text_type(organization.id)
  55         self.save_session()
  56 
  57         # now that SSO is marked as complete, we should be able to access dash
  58         for path in paths:
  59             resp = self.client.get(path)
  60             assert resp.status_code == 200, (path, resp.status_code, resp.content)
