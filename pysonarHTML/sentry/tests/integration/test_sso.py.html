  10 class OrganizationAuthLoginTest(AuthProviderTestCase):
  11     def test_sso_auth_required(self):
  12         user = self.create_user("foo@example.com", is_superuser=False)
  13         organization = self.create_organization(name="foo")
  14         member = self.create_member(user=user, organization=organization)
  15         setattr(member.flags, "sso:linked", True)
  16         member.save()
  17 
  18         auth_provider = AuthProvider.objects.create(
  19             organization=organization, provider="dummy", flags=0
  20         )
  21 
  22         AuthIdentity.objects.create(auth_provider=auth_provider, user=user)
  23 
  24         self.login_as(user)
  25 
  26         path = u"/{}/".format(organization.slug)
  27         redirect_uri = u"/auth/login/{}/".format(organization.slug)
  28 
  29         # we should be redirecting the user to the authentication form as they
  30         # haven't verified this specific organization
  31         resp = self.client.get(path)
  32         self.assertRedirects(resp, redirect_uri)
  33 
  34         # superuser should still require SSO as they're a member of the org
  35         user.update(is_superuser=True)
  36         resp = self.client.get(path)
  37         self.assertRedirects(resp, redirect_uri)
  38 
  39         # XXX(dcramer): using internal API as exposing a request object is hard
  40         self.session[SSO_SESSION_KEY] = six.text_type(organization.id)
  41         self.save_session()
  42 
  43         # now that SSO is marked as complete, we should be able to access dash
  44         resp = self.client.get(path)
  45         assert resp.status_code == 200
