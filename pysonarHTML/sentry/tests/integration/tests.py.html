 104     def configure_sentry_errors(self):
 105         # delay raising of assertion errors to make sure they do not get
 106         # swallowed again
 107         failures = []
 108 
 109         class AssertHandler(logging.Handler):
 110             def emit(self, entry):
 111                 failures.append(entry)
 112 
 113         assert_handler = AssertHandler()
 114 
 115         for name in "sentry.errors", "sentry_sdk.errors":
 116             sentry_errors = logging.getLogger(name)
 117             sentry_errors.addHandler(assert_handler)
 118             sentry_errors.setLevel(logging.DEBUG)
 119 
 120             @self.addCleanup
 121             def remove_handler(sentry_errors=sentry_errors):
 122                 sentry_errors.handlers.pop(sentry_errors.handlers.index(assert_handler))
 123 
 124         @self.addCleanup
 125         def reraise_failures():
 126             for entry in failures:
 127                 raise AssertionError(entry.message)

 129     def send_event(self, method, url, body, headers):
 130         from sentry.app import buffer
 131 
 132         with self.tasks():
 133             content_type = headers.pop("Content-Type", None)
 134             headers = {"HTTP_" + k.replace("-", "_").upper(): v for k, v in six.iteritems(headers)}
 135             resp = self.client.post(
 136                 reverse("sentry-api-store", args=[self.pk.project_id]),
 137                 data=body,
 138                 content_type=content_type,
 139                 **headers
 140             )
 141             assert resp.status_code == 200
 142 
 143             buffer.process_pending()

 470 @pytest.mark.obsolete("Functionality not relevant in Relay store")
 471 class SentryWsgiRemoteTest(TransactionTestCase):
 472     @override_settings(ALLOWED_HOSTS=["localhost"])
 473     def test_traceparent_header_wsgi(self):
 474         # Assert that posting something to store will not create another
 475         # (transaction) event under any circumstances.
 476         #
 477         # We use Werkzeug's test client because Django's test client bypasses a
 478         # lot of request handling code that we want to test implicitly (such as
 479         # all our WSGI middlewares and the entire Django instrumentation by
 480         # sentry-sdk).
 481         #
 482         # XXX(markus): Ideally methods such as `_postWithHeader` would always
 483         # call the WSGI application => swap out Django's test client with e.g.
 484         # Werkzeug's.
 485         client = WerkzeugClient(application)
 486 
 487         calls = []
 488 
 489         def new_disable_transaction_events():
 490             with configure_scope() as scope:
 491                 assert scope.span.sampled
 492                 assert scope.span.transaction
 493                 disable_transaction_events()
 494                 assert not scope.span.sampled
 495 
 496             calls.append(1)
 497 
 498         events = []
 499 
 500         auth = get_auth_header(
 501             "_postWithWerkzeug/0.0.0", self.projectkey.public_key, self.projectkey.secret_key, "7"
 502         )
 503 
 504         with mock.patch(
 505             "sentry.web.api.disable_transaction_events", new_disable_transaction_events
 506         ):
 507             with self.tasks():
 508                 with Hub(
 509                     Client(
 510                         transport=events.append,
 511                         integrations=[CeleryIntegration(), DjangoIntegration()],
 512                     )
 513                 ):
 514                     app_iter, status, headers = client.post(
 515                         reverse("sentry-api-store"),
 516                         data=b'{"message": "hello"}',
 517                         headers={
 518                             "x-sentry-auth": auth,
 519                             "sentry-trace": "1",
 520                             "content-type": "application/octet-stream",
 521                         },
 522                         environ_base={"REMOTE_ADDR": "127.0.0.1"},
 523                     )
 524 
 525                     body = "".join(app_iter)
 526 
 527         assert status == "200 OK", body
 528         assert set((e.get("type"), e.get("transaction")) for e in events) == {
 529             ("transaction", "rule_processor_apply")
 530         }
 531         assert calls == [1]

 584         assert resp.status_code == 201, resp.content
 585         # XXX: there appears to be a race condition between the 201 return and get_events,

 587         # Modifying the timestamp of store_event, like how it's done in other snuba tests,

 589         sleep(0.5)
 590         events = eventstore.get_events(
 591             filter=eventstore.Filter(
 592                 project_ids=[self.project.id], conditions=[["type", "=", "csp"]]
 593             )
 594         )
 595         assert len(events) == 1

 598         for key, value in six.iteritems(output["tags"]):
 599             assert e.get_tag(key) == value
 600         for key, value in six.iteritems(output["data"]):
 601             assert e.data[key] == value
