 148     def test_shutdown(self):
 149         self.producer.produce(self.topic, json.dumps(self.valid_wrapper))
 150         valid_wrapper_2 = deepcopy(self.valid_wrapper)
 151         valid_wrapper_2["payload"]["result"]["hello"] = 25
 152         valid_wrapper_3 = deepcopy(valid_wrapper_2)
 153         valid_wrapper_3["payload"]["result"]["hello"] = 5000
 154         self.producer.produce(self.topic, json.dumps(valid_wrapper_2))
 155         self.producer.flush()
 156 
 157         counts = [0]
 158 
 159         def mock_callback(*args, **kwargs):
 160             counts[0] += 1
 161             if counts[0] > 1:
 162                 raise KeyboardInterrupt()
 163 
 164         mock = Mock()
 165         mock.side_effect = mock_callback
 166 
 167         register_subscriber(self.registration_key)(mock)
 168         sub = self.create_subscription()
 169         consumer = QuerySubscriptionConsumer("hi", topic=self.topic, commit_batch_size=100)
 170         consumer.run()
 171         valid_payload = self.valid_payload
 172         valid_payload["values"] = valid_payload["result"]
 173         valid_payload["timestamp"] = parse_date(valid_payload["timestamp"]).replace(tzinfo=pytz.utc)
 174         valid_wrapper_2["payload"]["values"] = valid_wrapper_2["payload"]["result"]
 175         valid_wrapper_2["payload"]["timestamp"] = parse_date(
 176             valid_wrapper_2["payload"]["timestamp"]
 177         ).replace(tzinfo=pytz.utc)
 178         mock.assert_has_calls([call(valid_payload, sub), call(valid_wrapper_2["payload"], sub)])
 179         # Offset should be committed for the first message, so second run should process
 180         # the second message again
 181         self.producer.produce(self.topic, json.dumps(valid_wrapper_3))
 182         self.producer.flush()
 183         mock.reset_mock()
 184         counts[0] = 0
 185         consumer.run()
 186         valid_wrapper_3["payload"]["values"] = valid_wrapper_3["payload"]["result"]
 187         valid_wrapper_3["payload"]["timestamp"] = parse_date(
 188             valid_wrapper_3["payload"]["timestamp"]
 189         ).replace(tzinfo=pytz.utc)
 190 
 191         mock.assert_has_calls(
 192             [call(valid_wrapper_2["payload"], sub), call(valid_wrapper_3["payload"], sub)]
 193         )
