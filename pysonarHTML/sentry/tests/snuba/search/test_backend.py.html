 120     def store_event(self, data, *args, **kwargs):
 121         event = super(EventsSnubaSearchTest, self).store_event(data, *args, **kwargs)
 122         environment_name = data.get("environment")
 123         if environment_name:
 124             GroupEnvironment.objects.filter(
 125                 group_id=event.group_id,
 126                 environment__name=environment_name,
 127                 first_seen__gt=event.datetime,
 128             ).update(first_seen=event.datetime)
 129         return event

 157     def make_query(
 158         self,
 159         projects=None,
 160         search_filter_query=None,
 161         environments=None,
 162         sort_by="date",
 163         limit=None,
 164         count_hits=False,
 165         date_from=None,
 166         date_to=None,
 167     ):
 168         search_filters = []
 169         projects = projects if projects is not None else [self.project]
 170         if search_filter_query is not None:
 171             search_filters = self.build_search_filter(
 172                 search_filter_query, projects, environments=environments
 173             )
 174 
 175         kwargs = {}
 176         if limit is not None:
 177             kwargs["limit"] = limit
 178 
 179         return self.backend.query(
 180             projects,
 181             search_filters=search_filters,
 182             environments=environments,
 183             count_hits=count_hits,
 184             sort_by=sort_by,
 185             date_from=date_from,
 186             date_to=date_to,
 187             **kwargs
 188         )

1013     def test_hits_estimate(self):
1014         # 400 Groups/Events
1015         # Every 3rd one is Unresolved
1016         # Every 2nd one has tag match=1
1017         for i in range(400):
1018             event = self.store_event(
1019                 data={
1020                     "event_id": md5("event {}".format(i)).hexdigest(),
1021                     "fingerprint": ["put-me-in-group{}".format(i)],
1022                     "timestamp": iso_format(self.base_datetime - timedelta(days=21)),
1023                     "message": "group {} event".format(i),
1024                     "stacktrace": {"frames": [{"module": "module {}".format(i)}]},
1025                     "tags": {"match": "{}".format(i % 2)},
1026                     "environment": "production",
1027                 },
1028                 project_id=self.project.id,
1029             )
1030 
1031             group = event.group
1032             group.times_seen = 5
1033             group.status = GroupStatus.UNRESOLVED if i % 3 == 0 else GroupStatus.RESOLVED
1034             group.save()
1035             self.store_group(group)
1036 
1037         # Sample should estimate there are roughly 66 overall matching groups
1038         # based on a random sample of 100 (or $sample_size) of the total 200
1039         # snuba matches, of which 33% should pass the postgres filter.
1040         with self.options(
1041             {
1042                 # Too small to pass all django candidates down to snuba
1043                 "snuba.search.max-pre-snuba-candidates": 5,
1044                 "snuba.search.hits-sample-size": 50,
1045             }
1046         ):
1047             first_results = self.make_query(
1048                 search_filter_query="is:unresolved match:1", limit=10, count_hits=True
1049             )
1050 
1051             # Deliberately do not assert that the value is within some margin
1052             # of error, as this will fail tests at some rate corresponding to
1053             # our confidence interval.
1054             assert first_results.hits > 10
1055 
1056             # When searching for the same tags, we should get the same set of
1057             # hits as the sampling is based on the hash of the query.
1058             second_results = self.make_query(
1059                 search_filter_query="is:unresolved match:1", limit=10, count_hits=True
1060             )
1061 
1062             assert first_results.results == second_results.results
1063 
1064             # When using a different search, we should get a different sample
1065             # but still should have some hits.
1066             third_results = self.make_query(
1067                 search_filter_query="is:unresolved match:0", limit=10, count_hits=True
1068             )
1069 
1070             assert third_results.hits > 10
1071             assert third_results.results != second_results.results
