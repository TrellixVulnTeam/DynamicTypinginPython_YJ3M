  99     def test(self):
 100         # Full integration test to ensure that when a subscription receives an update
 101         # the `QuerySubscriptionConsumer` successfully retries the subscription and
 102         # calls the correct callback, which should result in an incident being created.
 103 
 104         callback = subscriber_registry[INCIDENTS_SNUBA_SUBSCRIPTION_TYPE]
 105 
 106         def exception_callback(*args, **kwargs):
 107             # We want to just error after the callback so that we can see the result of
 108             # processing. This means the offset won't be committed, but that's fine, we
 109             # can still check the results.
 110             callback(*args, **kwargs)
 111             raise KeyboardInterrupt()
 112 
 113         subscriber_registry[INCIDENTS_SNUBA_SUBSCRIPTION_TYPE] = exception_callback
 114         message = {
 115             "version": 1,
 116             "payload": {
 117                 "subscription_id": self.subscription.subscription_id,
 118                 "values": {"data": [{"some_col": self.trigger.alert_threshold + 1}]},
 119                 "timestamp": "2020-01-01T01:23:45.1234",
 120             },
 121         }
 122         self.producer.produce(self.topic, json.dumps(message))
 123         self.producer.flush()
 124 
 125         def active_incident():
 126             return Incident.objects.filter(
 127                 type=IncidentType.ALERT_TRIGGERED.value, alert_rule=self.rule
 128             ).exclude(status=IncidentStatus.CLOSED.value)
 129 
 130         consumer = QuerySubscriptionConsumer("hi", topic=self.topic)
 131         with self.assertChanges(
 132             lambda: active_incident().exists(), before=False, after=True
 133         ), self.tasks():
 134             consumer.run()
 135 
 136         assert len(mail.outbox) == 1
 137         handler = EmailActionHandler(self.action, active_incident().get(), self.project)
 138         message = handler.build_message(
 139             handler.generate_email_context(TriggerStatus.ACTIVE), TriggerStatus.ACTIVE, self.user.id
 140         )
 141 
 142         out = mail.outbox[0]
 143         assert out.to == [self.user.email]
 144         assert out.subject == message.subject
 145         built_message = message.build(self.user.email)
 146         assert out.body == built_message.body
