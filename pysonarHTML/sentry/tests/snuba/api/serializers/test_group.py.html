 180     def test_implicit_subscribed(self):
 181         user = self.create_user()
 182         group = self.create_group()
 183 
 184         combinations = (
 185             # ((default, project), (subscribed, details))
 186             ((UserOptionValue.all_conversations, None), (True, None)),
 187             ((UserOptionValue.all_conversations, UserOptionValue.all_conversations), (True, None)),
 188             (
 189                 (UserOptionValue.all_conversations, UserOptionValue.participating_only),
 190                 (False, None),
 191             ),
 192             (
 193                 (UserOptionValue.all_conversations, UserOptionValue.no_conversations),
 194                 (False, {"disabled": True}),
 195             ),
 196             ((None, None), (False, None)),
 197             ((UserOptionValue.participating_only, None), (False, None)),
 198             ((UserOptionValue.participating_only, UserOptionValue.all_conversations), (True, None)),
 199             (
 200                 (UserOptionValue.participating_only, UserOptionValue.participating_only),
 201                 (False, None),
 202             ),
 203             (
 204                 (UserOptionValue.participating_only, UserOptionValue.no_conversations),
 205                 (False, {"disabled": True}),
 206             ),
 207             ((UserOptionValue.no_conversations, None), (False, {"disabled": True})),
 208             ((UserOptionValue.no_conversations, UserOptionValue.all_conversations), (True, None)),
 209             ((UserOptionValue.no_conversations, UserOptionValue.participating_only), (False, None)),
 210             (
 211                 (UserOptionValue.no_conversations, UserOptionValue.no_conversations),
 212                 (False, {"disabled": True}),
 213             ),
 214         )
 215 
 216         def maybe_set_value(project, value):
 217             if value is not None:
 218                 UserOption.objects.set_value(
 219                     user=user, project=project, key="workflow:notifications", value=value
 220                 )
 221             else:
 222                 UserOption.objects.unset_value(
 223                     user=user, project=project, key="workflow:notifications"
 224                 )
 225 
 226         for options, (is_subscribed, subscription_details) in combinations:
 227             default_value, project_value = options
 228             UserOption.objects.clear_local_cache()
 229             maybe_set_value(None, default_value)
 230             maybe_set_value(group.project, project_value)
 231             result = serialize(group, user, serializer=GroupSerializerSnuba())
 232             assert result["isSubscribed"] is is_subscribed
 233             assert result.get("subscriptionDetails") == subscription_details

 279     def test_seen_stats(self):
 280         environment = self.create_environment(project=self.project)
 281         environment2 = self.create_environment(project=self.project)
 282 
 283         events = []
 284 
 285         for event_id, env, user_id, timestamp in [
 286             ("a" * 32, environment, 1, iso_format(self.min_ago)),
 287             ("b" * 32, environment, 2, iso_format(self.min_ago)),
 288             ("c" * 32, environment2, 3, iso_format(self.week_ago)),
 289         ]:
 290             events.append(
 291                 self.store_event(
 292                     data={
 293                         "event_id": event_id,
 294                         "fingerprint": ["put-me-in-group1"],
 295                         "timestamp": timestamp,
 296                         "environment": env.name,
 297                         "user": {"id": user_id},
 298                     },
 299                     project_id=self.project.id,
 300                 )
 301             )
 302 
 303         # Assert all events are in the same group
 304         (group_id,) = set(e.group.id for e in events)
 305 
 306         group = Group.objects.get(id=group_id)
 307         group.times_seen = 3
 308         group.first_seen = self.week_ago - timedelta(days=5)
 309         group.last_seen = self.week_ago
 310         group.save()
 311 
 312         # should use group columns when no environments arg passed
 313         result = serialize(group, serializer=GroupSerializerSnuba(environment_ids=[]))
 314         assert result["count"] == "3"
 315         assert iso_format(result["lastSeen"]) == iso_format(self.min_ago)
 316         assert result["firstSeen"] == group.first_seen
 317 
 318         # update this to something different to make sure it's being used
 319         group_env = GroupEnvironment.objects.get(group_id=group_id, environment_id=environment.id)
 320         group_env.first_seen = self.day_ago - timedelta(days=3)
 321         group_env.save()
 322 
 323         group_env2 = GroupEnvironment.objects.get(group_id=group_id, environment_id=environment2.id)
 324 
 325         result = serialize(
 326             group,
 327             serializer=GroupSerializerSnuba(environment_ids=[environment.id, environment2.id]),
 328         )
 329         assert result["count"] == "3"
 330         # result is rounded down to nearest second
 331         assert iso_format(result["lastSeen"]) == iso_format(self.min_ago)
 332         assert iso_format(result["firstSeen"]) == iso_format(group_env.first_seen)
 333         assert iso_format(group_env2.first_seen) > iso_format(group_env.first_seen)
 334         assert result["userCount"] == 3
 335 
 336         # test userCount, count, lastSeen filtering correctly by time
 337         # firstSeen should still be from GroupEnvironment
 338         result = serialize(
 339             group,
 340             serializer=GroupSerializerSnuba(
 341                 environment_ids=[environment.id, environment2.id],
 342                 start=self.week_ago - timedelta(hours=1),
 343                 end=self.week_ago + timedelta(hours=1),
 344             ),
 345         )
 346         assert result["userCount"] == 1
 347         assert iso_format(result["lastSeen"]) == iso_format(self.week_ago)
 348         assert iso_format(result["firstSeen"]) == iso_format(group_env.first_seen)
 349         assert result["count"] == "1"
