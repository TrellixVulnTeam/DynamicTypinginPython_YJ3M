 132     def test_access_non_member_project(self):
 133         # Add a new user to a project and then access events on project they are not part of.
 134         member_user = self.create_user()
 135         team = self.create_team(members=[member_user])
 136         self.create_project(organization=self.organization, teams=[team])
 137 
 138         # Enable open membership
 139         self.organization.flags.allow_joinleave = True
 140         self.organization.save()
 141 
 142         self.login_as(member_user)
 143 
 144         url = reverse(
 145             "sentry-api-0-organization-event-details",
 146             kwargs={
 147                 "organization_slug": self.organization.slug,
 148                 "project_slug": self.project.slug,
 149                 "event_id": "a" * 32,
 150             },
 151         )
 152         with self.feature("organizations:discover-basic"):
 153             response = self.client.get(url, format="json")
 154         assert response.status_code == 200, response.content
 155 
 156         # When open membership is off, access should be denied to non owner users
 157         self.organization.flags.allow_joinleave = False
 158         self.organization.save()
 159 
 160         with self.feature("organizations:discover-basic"):
 161             response = self.client.get(url, format="json")
 162         assert response.status_code == 404, response.content

 179     def test_long_trace_description(self):
 180         data = load_data("transaction")
 181         data["event_id"] = "d" * 32
 182         data["timestamp"] = iso_format(before_now(minutes=1))
 183         data["start_timestamp"] = iso_format(before_now(minutes=1) - timedelta(seconds=5))
 184         data["contexts"]["trace"]["description"] = "b" * 512
 185         self.store_event(data=data, project_id=self.project.id)
 186 
 187         url = reverse(
 188             "sentry-api-0-organization-event-details",
 189             kwargs={
 190                 "organization_slug": self.project.organization.slug,
 191                 "project_slug": self.project.slug,
 192                 "event_id": "d" * 32,
 193             },
 194         )
 195         with self.feature("organizations:discover-basic"):
 196             response = self.client.get(url, format="json")
 197 
 198         assert response.status_code == 200, response.content
 199         trace = response.data["contexts"]["trace"]
 200         original_trace = data["contexts"]["trace"]
 201         assert trace["trace_id"] == original_trace["trace_id"]
 202         assert trace["span_id"] == original_trace["span_id"]
 203         assert trace["parent_span_id"] == original_trace["parent_span_id"]
 204         assert trace["description"].endswith("...")
 205         assert trace["description"][:-3] in original_trace["description"]
