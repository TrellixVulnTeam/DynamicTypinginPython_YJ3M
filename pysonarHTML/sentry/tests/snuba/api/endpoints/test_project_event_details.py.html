 121 class ProjectEventJsonEndpointTest(APITestCase, SnubaTestCase):
 122     def setUp(self):
 123         super(ProjectEventJsonEndpointTest, self).setUp()
 124         self.login_as(user=self.user)
 125         self.event_id = "c" * 32
 126         self.fingerprint = ["group_2"]
 127         self.min_ago = iso_format(before_now(minutes=1))
 128         self.event = self.store_event(
 129             data={
 130                 "event_id": self.event_id,
 131                 "timestamp": self.min_ago,
 132                 "fingerprint": self.fingerprint,
 133                 "user": {"email": self.user.email},
 134             },
 135             project_id=self.project.id,
 136         )
 137         self.url = reverse(
 138             "sentry-api-0-event-json",
 139             kwargs={
 140                 "organization_slug": self.organization.slug,
 141                 "project_slug": self.project.slug,
 142                 "event_id": self.event_id,
 143             },
 144         )
 145 
 146     def assert_event(self, data):
 147         assert data["event_id"] == self.event_id
 148         assert data["user"]["email"] == self.user.email
 149         assert data["datetime"][:19] == self.min_ago
 150         assert data["fingerprint"] == self.fingerprint
 151 
 152     def test_simple(self):
 153         response = self.client.get(self.url, format="json")
 154         assert response.status_code == 200, response.content
 155         self.assert_event(response.data)
 156 
 157     def test_event_does_not_exist(self):
 158         self.url = reverse(
 159             "sentry-api-0-event-json",
 160             kwargs={
 161                 "organization_slug": self.organization.slug,
 162                 "project_slug": self.project.slug,
 163                 "event_id": "no" * 16,
 164             },
 165         )
 166         response = self.client.get(self.url, format="json")
 167         assert response.status_code == 404, response.content
 168         assert response.data == {"detail": "Event not found"}
 169 
 170     def test_user_unauthorized(self):
 171         user = self.create_user()
 172         self.login_as(user)
 173 
 174         response = self.client.get(self.url, format="json")
 175         assert response.status_code == 403, response.content
 176         assert response.data == {"detail": "You do not have permission to perform this action."}
