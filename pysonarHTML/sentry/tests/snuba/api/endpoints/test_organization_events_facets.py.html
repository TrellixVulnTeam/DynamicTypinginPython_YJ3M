  31     def assert_facet(self, response, key, expected):
  32         actual = None
  33         for facet in response.data:
  34             if facet["key"] == key:

  36                 break

  38         assert "topValues" in actual

 513     @mock.patch("sentry.utils.snuba.quantize_time")
 514     def test_quantize_dates(self, mock_quantize):
 515         mock_quantize.return_value = before_now(days=1).replace(tzinfo=utc)
 516         with self.feature("organizations:discover-basic"):
 517             # Don't quantize short time periods
 518             self.client.get(
 519                 self.url,
 520                 format="json",
 521                 data={"statsPeriod": "1h", "query": "", "field": ["id", "timestamp"]},
 522             )
 523             # Don't quantize absolute date periods
 524             self.client.get(
 525                 self.url,
 526                 format="json",
 527                 data={
 528                     "start": iso_format(before_now(days=20)),
 529                     "end": iso_format(before_now(days=15)),
 530                     "query": "",
 531                     "field": ["id", "timestamp"],
 532                 },
 533             )
 534 
 535             assert len(mock_quantize.mock_calls) == 0
 536 
 537             # Quantize long date periods
 538             self.client.get(
 539                 self.url,
 540                 format="json",
 541                 data={"field": ["id", "timestamp"], "statsPeriod": "90d", "query": ""},
 542             )
 543 
 544             assert len(mock_quantize.mock_calls) == 2
