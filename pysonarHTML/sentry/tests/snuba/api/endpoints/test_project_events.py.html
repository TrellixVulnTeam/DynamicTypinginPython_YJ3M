  10 class ProjectEventsTest(APITestCase, SnubaTestCase):
  11     def test_simple(self):
  12         self.login_as(user=self.user)
  13 
  14         project = self.create_project()
  15         event_1 = self.store_event(
  16             data={"timestamp": iso_format(before_now(minutes=1))}, project_id=project.id
  17         )
  18         event_2 = self.store_event(
  19             data={"timestamp": iso_format(before_now(minutes=1))}, project_id=project.id
  20         )
  21 
  22         url = reverse(
  23             "sentry-api-0-project-events",
  24             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
  25         )

  27 
  28         assert response.status_code == 200, response.content
  29         assert len(response.data) == 2
  30         assert sorted(map(lambda x: x["eventID"], response.data)) == sorted(
  31             [event_1.event_id, event_2.event_id]

  34     def test_message_search(self):
  35         self.login_as(user=self.user)
  36 
  37         project = self.create_project()
  38         self.store_event(
  39             data={"message": "how to make fast", "timestamp": iso_format(before_now(minutes=1))},
  40             project_id=project.id,
  41         )
  42         event_2 = self.store_event(
  43             data={"message": "Delet the Data", "timestamp": iso_format(before_now(minutes=1))},
  44             project_id=project.id,
  45         )
  46 
  47         url = reverse(
  48             "sentry-api-0-project-events",
  49             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
  50         )

  58     def test_filters_based_on_retention(self):
  59         self.login_as(user=self.user)
  60 
  61         project = self.create_project()

  63         event_2 = self.store_event(
  64             data={"timestamp": iso_format(before_now(minutes=1))}, project_id=project.id
  65         )
  66 
  67         with self.options({"system.event-retention-days": 1}):
  68             url = reverse(
  69                 "sentry-api-0-project-events",
  70                 kwargs={
  71                     "organization_slug": project.organization.slug,
  72                     "project_slug": project.slug,
  73                 },
  74             )
  75             response = self.client.get(url, format="json")
  76 
  77         assert response.status_code == 200, response.content
  78         assert len(response.data) == 1
  79         assert response.data[0]["eventID"] == event_2.event_id
