  33     def build_grouplink(self, group=None):
  34         group = self.group if group is None else group
  35         repo = Repository.objects.create(organization_id=self.org.id, name=group.project.name)
  36         commit = Commit.objects.create(
  37             organization_id=self.org.id, repository_id=repo.id, key=uuid1().hex
  38         )
  39         commit_2 = Commit.objects.create(
  40             organization_id=self.org.id, repository_id=repo.id, key=uuid1().hex
  41         )
  42         ReleaseCommit.objects.create(
  43             organization_id=self.org.id, release=self.release, commit=commit, order=commit.id
  44         )
  45         ReleaseCommit.objects.create(
  46             organization_id=self.org.id, release=self.release, commit=commit_2, order=commit_2.id
  47         )
  48         GroupLink.objects.create(
  49             group_id=group.id,
  50             project_id=group.project_id,
  51             linked_type=GroupLink.LinkedType.commit,
  52             relationship=GroupLink.Relationship.resolves,
  53             linked_id=commit.id,
  54         )
  55 
  56     def build_group_resolution(self, group=None):
  57         return GroupResolution.objects.create(
  58             group=self.group if group is None else group,
  59             release=self.release,
  60             type=GroupResolution.Type.in_release,
  61         )

  63     def run_test(self, expected_groups, project_ids=None, environment_names=None):
  64         params = {}
  65         if project_ids:
  66             params["project"] = project_ids
  67         if environment_names:
  68             params["environment"] = environment_names
  69 
  70         response = self.get_valid_response(self.org.slug, self.release.version, **params)
  71         assert len(response.data) == len(expected_groups)
  72         expected = set(map(six.text_type, [g.id for g in expected_groups]))
  73         assert set([item["id"] for item in response.data]) == expected

 101     def test_return_groups_from_both_types(self):
 102         """
 103         tests that the endpoint will correctly retrieve issues resolved
 104         in a release from both the GroupLink and GroupResolution model
 105         """
 106         self.build_grouplink()
 107         new_group = self.create_group(project=self.project)
 108         self.build_group_resolution(new_group)
 109         self.run_test([self.group, new_group])
 110 
 111     def test_multiple_projects(self):
 112         """
 113         Test that the endpoint will return issues resolved in a release across
 114         projects in the org, and that filtering by project works as expected
 115         """
 116         self.build_grouplink()
 117         self.build_grouplink(self.group_2)
 118         self.run_test([self.group, self.group_2])
 119         self.run_test([self.group], project_ids=[self.group.project_id])
 120         self.run_test([self.group_2], project_ids=[self.group_2.project_id])
 121         self.run_test(
 122             [self.group, self.group_2], project_ids=[self.group.project_id, self.group_2.project_id]
 123         )
