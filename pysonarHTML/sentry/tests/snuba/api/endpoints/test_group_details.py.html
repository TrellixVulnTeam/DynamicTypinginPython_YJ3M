  11 class GroupDetailsTest(APITestCase, SnubaTestCase):
  12     def test_multiple_environments(self):
  13         group = self.create_group()
  14         self.login_as(user=self.user)
  15 
  16         environment = Environment.get_or_create(group.project, "production")
  17         environment2 = Environment.get_or_create(group.project, "staging")
  18 
  19         url = u"/api/0/issues/{}/".format(group.id)
  20 
  21         from sentry.api.endpoints.group_details import tsdb
  22 
  23         with mock.patch(
  24             "sentry.api.endpoints.group_details.tsdb.get_range", side_effect=tsdb.get_range
  25         ) as get_range:
  26             response = self.client.get(
  27                 "%s?environment=production&environment=staging" % (url,), format="json"
  28             )
  29             assert response.status_code == 200
  30             assert get_range.call_count == 2
  31             for args, kwargs in get_range.call_args_list:
  32                 assert kwargs["environment_ids"] == [environment.id, environment2.id]
  33 
  34         response = self.client.get("%s?environment=invalid" % (url,), format="json")
  35         assert response.status_code == 404

  37     def test_with_first_last_release(self):
  38         self.login_as(user=self.user)
  39         first_release = {
  40             "firstEvent": before_now(minutes=3),
  41             "lastEvent": before_now(minutes=2, seconds=30),
  42         }
  43         last_release = {
  44             "firstEvent": before_now(minutes=1, seconds=30),
  45             "lastEvent": before_now(minutes=1),
  46         }
  47 
  48         for timestamp in first_release.values():
  49             self.store_event(
  50                 data={"release": "1.0", "timestamp": iso_format(timestamp)},
  51                 project_id=self.project.id,
  52             )
  53         self.store_event(
  54             data={"release": "1.1", "timestamp": iso_format(before_now(minutes=2))},
  55             project_id=self.project.id,
  56         )
  57         for timestamp in last_release.values():
  58             event = self.store_event(
  59                 data={"release": "1.0a", "timestamp": iso_format(timestamp)},
  60                 project_id=self.project.id,
  61             )
  62 
  63         group = event.group
  64 
  65         url = u"/api/0/issues/{}/".format(group.id)
  66         response = self.client.get(url, format="json")
  67 
  68         assert response.status_code == 200, response.content
  69         assert response.data["id"] == six.text_type(group.id)
  70         release = response.data["firstRelease"]
  71         assert release["version"] == "1.0"
  72         for event, timestamp in six.iteritems(first_release):
  73             assert release[event].ctime() == timestamp.ctime()
  74         release = response.data["lastRelease"]
  75         assert release["version"] == "1.0a"
  76         for event, timestamp in six.iteritems(last_release):
  77             assert release[event].ctime() == timestamp.ctime()

 102     def test_first_release_only(self):
 103         self.login_as(user=self.user)
 104 
 105         first_event = before_now(days=3)
 106 
 107         self.store_event(
 108             data={"release": "1.0", "timestamp": iso_format(first_event)},
 109             project_id=self.project.id,
 110         )
 111         event = self.store_event(
 112             data={"release": "1.1", "timestamp": iso_format(before_now(days=1))},
 113             project_id=self.project.id,
 114         )
 115         # Forcibly remove one of the releases
 116         Release.objects.get(version="1.1").delete()
 117 
 118         group = event.group
 119 
 120         url = u"/api/0/issues/{}/".format(group.id)
 121 
 122         response = self.client.get(url, format="json")
 123         assert response.status_code == 200, response.content
 124         assert response.data["firstRelease"]["version"] == "1.0"
 125         # only one event
 126         assert (
 127             response.data["firstRelease"]["firstEvent"]
 128             == response.data["firstRelease"]["lastEvent"]
 129         )
 130         assert response.data["firstRelease"]["firstEvent"].ctime() == first_event.ctime()
 131         assert response.data["lastRelease"] == {"version": "1.1"}
