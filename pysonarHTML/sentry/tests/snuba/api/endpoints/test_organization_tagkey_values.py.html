  40 class OrganizationTagKeyValuesTest(OrganizationTagKeyTestCase):
  41     def test_simple(self):
  42         self.store_event(
  43             data={"timestamp": iso_format(self.day_ago), "tags": {"fruit": "apple"}},
  44             project_id=self.project.id,
  45         )
  46         self.store_event(
  47             data={"timestamp": iso_format(self.min_ago), "tags": {"fruit": "orange"}},
  48             project_id=self.project.id,
  49         )
  50         self.store_event(
  51             data={"timestamp": iso_format(self.min_ago), "tags": {"some_tag": "some_value"}},
  52             project_id=self.project.id,
  53         )
  54         self.store_event(
  55             data={"timestamp": iso_format(self.min_ago), "tags": {"fruit": "orange"}},
  56             project_id=self.project.id,
  57         )
  58 
  59         url = reverse(
  60             "sentry-api-0-organization-tagkey-values",
  61             kwargs={"organization_slug": self.org.slug, "key": "fruit"},
  62         )
  63         response = self.client.get(url, format="json")
  64         assert response.status_code == 200, response.content
  65         self.run_test("fruit", expected=[("orange", 2), ("apple", 1)])

 160 class TransactionTagKeyValues(OrganizationTagKeyTestCase):
 161     def setUp(self):
 162         super(TransactionTagKeyValues, self).setUp()
 163         data = load_data("transaction", timestamp=before_now(minutes=1))
 164         self.store_event(data, project_id=self.project.id)
 165         self.transaction = data.copy()
 166         self.transaction.update(
 167             {
 168                 "transaction": "/city_by_code/",
 169                 "timestamp": iso_format(before_now(seconds=30)),
 170                 "start_timestamp": iso_format(before_now(seconds=35)),
 171             }
 172         )
 173         self.transaction["contexts"]["trace"].update(
 174             {"status": "unknown_error", "parent_span_id": "9000cec7cc0779c1", "op": "bar.server"}
 175         )
 176         self.store_event(
 177             self.transaction, project_id=self.project.id,
 178         )
 179

 185             expected=[("unknown_error", 1), ("ok", 1)],
 186         )

 188             "transaction.status", qs_params={"query": "ow"}, expected=[("unknown_error", 1)]
