 171

 174         assert response.data[0]["eventID"] == event_1.event_id
 175 
 176         # Multiple Word Query

 186     def test_environment(self):
 187         self.login_as(user=self.user)
 188         events = {}
 189 
 190         for name in ["production", "development"]:
 191             events[name] = self.store_event(
 192                 data={
 193                     "fingerprint": ["put-me-in-group1"],
 194                     "timestamp": iso_format(self.min_ago),
 195                     "environment": name,
 196                 },
 197                 project_id=self.project.id,
 198             )
 199 
 200         # Asserts that all are in the same group
 201         (group_id,) = set(e.group.id for e in events.values())
 202 
 203         url = u"/api/0/issues/{}/events/".format(group_id)
 204         response = self.client.get(url + "?environment=production", format="json")
 205 
 206         assert response.status_code == 200, response.content
 207         assert set(map(lambda x: x["eventID"], response.data)) == set(
 208             [six.text_type(events["production"].event_id)]
 209         )
 210 
 211         response = self.client.get(
 212             url, data={"environment": ["production", "development"]}, format="json"
 213         )
 214         assert response.status_code == 200, response.content
 215         assert set(map(lambda x: x["eventID"], response.data)) == set(
 216             [six.text_type(event.event_id) for event in events.values()]
 217         )
 218 
 219         response = self.client.get(url + "?environment=invalid", format="json")
 220 
 221         assert response.status_code == 200, response.content
 222         assert response.data == []
 223 
 224         response = self.client.get(
 225             url + "?environment=production&query=environment:development", format="json"
 226         )
 227 
 228         assert response.status_code == 200, response.content
 229         assert response.data == []
