 131     def test_conditional_fields(self):
 132         with self.feature("organizations:discover"):
 133             self.store_event(
 134                 data={
 135                     "platform": "javascript",
 136                     "environment": "production",
 137                     "tags": {"sentry:release": "bar"},
 138                     "timestamp": self.one_second_ago,
 139                 },
 140                 project_id=self.project.id,
 141             )
 142             self.store_event(
 143                 data={
 144                     "platform": "javascript",
 145                     "environment": "production",
 146                     "tags": {"sentry:release": "baz"},
 147                     "timestamp": self.one_second_ago,
 148                 },
 149                 project_id=self.project.id,
 150             )
 151             url = reverse("sentry-api-0-discover-query", args=[self.org.slug])
 152             response = self.client.post(
 153                 url,
 154                 {
 155                     "projects": [self.project.id],
 156                     "aggregations": [["count()", None, "count"]],
 157                     "conditionFields": [
 158                         [
 159                             "if",
 160                             [["in", ["release", "tuple", ["'foo'"]]], "release", "'other'"],
 161                             "release",
 162                         ]
 163                     ],
 164                     "start": iso_format(datetime.now() - timedelta(seconds=10)),
 165                     "end": iso_format(datetime.now()),
 166                     "groupby": ["time", "release"],
 167                     "rollup": 86400,
 168                     "limit": 1000,
 169                     "orderby": "-time",
 170                     "range": None,
 171                 },
 172             )
 173 
 174         assert response.status_code == 200, response.content
 175 
 176         # rollup is by one day and diff of start/end is 10 seconds, so we only have one day
 177         assert len(response.data["data"]) == 2
 178 
 179         for data in response.data["data"]:
 180             # note this "release" key represents the alias for the column condition
 181             # and is also used in `groupby`, it is NOT the release tag
 182             if data["release"] == "foo":
 183                 assert data["count"] == 1
 184             elif data["release"] == "other":
 185                 assert data["count"] == 2

 397     def test_zerofilled_dates_when_rollup_absolute(self):
 398         with self.feature("organizations:discover"):
 399             url = reverse("sentry-api-0-discover-query", args=[self.org.slug])
 400             response = self.client.post(
 401                 url,
 402                 {
 403                     "projects": [self.project.id],
 404                     "aggregations": [["count()", "", "count"]],
 405                     "fields": ["project.name"],
 406                     "groupby": ["time"],
 407                     "orderby": "-time",
 408                     "start": (self.now - timedelta(seconds=300)).strftime("%Y-%m-%dT%H:%M:%S"),
 409                     "end": self.now.strftime("%Y-%m-%dT%H:%M:%S"),
 410                     "rollup": 60,
 411                     "range": None,
 412                 },
 413             )
 414 
 415         assert response.status_code == 200, response.content
 416         assert len(response.data["data"]) == 6
 417         event_record = response.data["data"][0]
 418         # This test can span across an hour, where the start is in hour 1, end is in hour 2, and event is in hour 2.
 419         # That pushes the result to the second row.
 420         if "project.name" not in event_record:
 421             event_record = response.data["data"][1]
 422         assert (event_record["time"]) > response.data["data"][2]["time"]
 423         assert (event_record["project.name"]) == "bar"
 424         assert (event_record["count"]) == 1
