 110     @mock.patch("sentry.options.get", return_value=1.0)

 112     def test_different_times_caching(self, mock_snuba_query, mock_options):
 113         user = self.create_user()

 115         team = self.create_team(organization=org)
 116         self.create_member(organization=org, user=user, teams=[team])
 117         self.create_project(organization=org, teams=[team])
 118         self.login_as(user=user)
 119 
 120         start = iso_format(before_now(minutes=10))
 121         end = iso_format(before_now(minutes=5))
 122         url = reverse("sentry-api-0-organization-tags", kwargs={"organization_slug": org.slug})
 123         response = self.client.get(

 125         )
 126         assert response.status_code == 200, response.content
 127         assert mock_snuba_query.call_count == 1
 128 
 129         # 5 minutes later, cache_key should be different
 130         start = iso_format(before_now(minutes=5))
 131         end = iso_format(before_now(minutes=0))
 132         response = self.client.get(
 133             url, {"use_cache": "1", "start": start, "end": end}, format="json"
 134         )
 135         assert response.status_code == 200, response.content
 136         assert mock_snuba_query.call_count == 2

 151             data={"event_id": "a" * 32, "tags": {"fruit": "apple"}, "timestamp": middle},
 152             project_id=project.id,
 153         )

 157         response = self.client.get(
 158             url, {"use_cache": "1", "start": start, "end": end}, format="json"
 159         )
 160         original_data = response.data
 161 
 162         url = reverse("sentry-api-0-organization-tags", kwargs={"organization_slug": org.slug})
 163         response = self.client.get(
 164             url, {"use_cache": "1", "start": start, "end": end}, format="json"
 165         )
 166         cached_data = response.data

 168         assert original_data == cached_data
