 111     def test_get_sortby(self):
 112         query = {"fields": ["message"], "query": "", "limit": 10}
 113         model = DiscoverSavedQuery.objects.create(
 114             organization=self.org,
 115             created_by=self.user,
 116             name="My query",
 117             query=query,
 118             version=2,
 119             date_created=before_now(minutes=10),
 120             date_updated=before_now(minutes=10),
 121         )
 122         model.set_projects(self.project_ids)
 123 
 124         sort_options = {
 125             "dateCreated": True,
 126             "-dateCreated": False,
 127             "dateUpdated": True,
 128             "-dateUpdated": False,
 129             "name": True,
 130             "-name": False,
 131         }
 132         for sorting, forward_sort in sort_options.items():
 133             with self.feature(self.feature_name):
 134                 response = self.client.get(self.url, data={"sortBy": sorting})
 135             assert response.status_code == 200
 136 
 137             values = [row[sorting.strip("-")] for row in response.data]
 138             if not forward_sort:
 139                 values = list(reversed(values))
 140             assert list(sorted(values)) == values
