 119     @mock.patch("sentry.utils.snuba.quantize_time")
 120     def test_quantize_dates(self, mock_quantize):
 121         mock_quantize.return_value = before_now(days=1).replace(tzinfo=utc)
 122         with self.feature("organizations:discover-basic"):
 123             # Don't quantize short time periods
 124             self.client.get(
 125                 self.url,
 126                 format="json",
 127                 data={"statsPeriod": "1h", "query": "", "field": ["id", "timestamp"]},
 128             )
 129             # Don't quantize absolute date periods
 130             self.client.get(
 131                 self.url,
 132                 format="json",
 133                 data={
 134                     "start": iso_format(before_now(days=20)),
 135                     "end": iso_format(before_now(days=15)),
 136                     "query": "",
 137                     "field": ["id", "timestamp"],
 138                 },
 139             )
 140 
 141             assert len(mock_quantize.mock_calls) == 0
 142 
 143             # Quantize long date periods
 144             self.client.get(
 145                 self.url,
 146                 format="json",
 147                 data={"field": ["id", "timestamp"], "statsPeriod": "90d", "query": ""},
 148             )
 149 
 150             assert len(mock_quantize.mock_calls) == 2
