  25 def has_shape(data, shape, allow_empty=False):
  26     """
  27     Determine if a data object has the provided shape
  28 
  29     At any level, the object in `data` and in `shape` must have the same type.
  30     A dict is the same shape if all its keys and values have the same shape as the
  31     key/value in `shape`. The number of keys/values is not relevant.
  32     A list is the same shape if all its items have the same shape as the value
  33     in `shape`
  34     A tuple is the same shape if it has the same length as `shape` and all the
  35     values have the same shape as the corresponding value in `shape`
  36     Any other object simply has to have the same type.
  37     If `allow_empty` is set, lists and dicts in `data` will pass even if they are empty.
  38     """
  39     if not isinstance(data, type(shape)):
  40         return False
  41     if isinstance(data, dict):
  42         return (
  43             (allow_empty or len(data) > 0)
  44             and all(has_shape(k, shape.keys()[0]) for k in data.keys())
  45             and all(has_shape(v, shape.values()[0]) for v in data.values())
  46         )
  47     elif isinstance(data, list):
  48         return (allow_empty or len(data) > 0) and all(has_shape(v, shape[0]) for v in data)
  49     elif isinstance(data, tuple):
  50         return len(data) == len(shape) and all(
  51             has_shape(data[i], shape[i]) for i in range(len(data))
  52         )
  53     else:
  54         return True

 259     def test_range_rollups(self):
 260         # Daily
 261         daystart = self.now.replace(hour=0)  # day buckets start on day boundaries
 262         dts = [daystart + timedelta(days=i) for i in range(2)]
 263         assert self.db.get_range(
 264             TSDBModel.project, [self.proj1.id], dts[0], dts[-1], rollup=86400
 265         ) == {self.proj1.id: [(timestamp(dts[0]), 12), (timestamp(dts[1]), 0)]}
 266 
 267         # Minutely
 268         dts = [self.now + timedelta(minutes=i) for i in range(120)]
 269         # Expect every 20th minute to have a 1, else 0
 270         expected = [
 271             (to_timestamp(d), 1 if i % 10 == 0 and i % 20 != 0 else 0) for i, d in enumerate(dts)
 272         ]
 273         expected[0] = (expected[0][0], 0)
 274         assert self.db.get_range(
 275             TSDBModel.project, [self.proj1.id], dts[0], dts[-1], rollup=60
 276         ) == {self.proj1.id: expected}

 403     def test_result_shape(self):
 404         """
 405         Tests that the results from the different TSDB methods have the
 406         expected format.
 407         """
 408         project_id = self.proj1.id
 409         dts = [self.now + timedelta(hours=i) for i in range(4)]
 410 
 411         results = self.db.get_most_frequent(
 412             TSDBModel.frequent_issues_by_project, [project_id], dts[0], dts[0]
 413         )
 414         assert has_shape(results, {1: [(1, 1.0)]})
 415 
 416         results = self.db.get_most_frequent_series(
 417             TSDBModel.frequent_issues_by_project, [project_id], dts[0], dts[0]
 418         )
 419         assert has_shape(results, {1: [(1, {1: 1.0})]})
 420 
 421         items = {
 422             # {project_id: (issue_id, issue_id, ...)}
 423             project_id: (self.proj1group1.id, self.proj1group2.id)
 424         }
 425         results = self.db.get_frequency_series(
 426             TSDBModel.frequent_issues_by_project, items, dts[0], dts[-1]
 427         )
 428         assert has_shape(results, {1: [(1, {1: 1})]})
 429 
 430         results = self.db.get_frequency_totals(
 431             TSDBModel.frequent_issues_by_project, items, dts[0], dts[-1]
 432         )
 433         assert has_shape(results, {1: {1: 1}})
 434 
 435         results = self.db.get_range(TSDBModel.project, [project_id], dts[0], dts[-1])
 436         assert has_shape(results, {1: [(1, 1)]})
 437 
 438         results = self.db.get_distinct_counts_series(
 439             TSDBModel.users_affected_by_project, [project_id], dts[0], dts[-1]
 440         )
 441         assert has_shape(results, {1: [(1, 1)]})
 442 
 443         results = self.db.get_distinct_counts_totals(
 444             TSDBModel.users_affected_by_project, [project_id], dts[0], dts[-1]
 445         )
 446         assert has_shape(results, {1: 1})
 447 
 448         results = self.db.get_distinct_counts_union(
 449             TSDBModel.users_affected_by_project, [project_id], dts[0], dts[-1]
 450         )
 451         assert has_shape(results, 1)
