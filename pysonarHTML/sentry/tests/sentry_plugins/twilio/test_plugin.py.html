  14 class TwilioConfigurationFormTest(TestCase):
  15     def test_valid_form(self):
  16         form = TwilioConfigurationForm(
  17             data={
  18                 "sms_from": "3305093095",
  19                 "sms_to": "330-509-3095, (330)-509-3095, +13305093095, 4045550144",
  20                 "auth_token": "foo",
  21                 "account_sid": "bar",
  22             }
  23         )
  24 
  25         self.assertTrue(form.is_valid())
  26         self.assertDictEqual(
  27             form.clean(),
  28             {
  29                 "auth_token": u"foo",
  30                 "sms_to": u"+13305093095,+14045550144",
  31                 "sms_from": u"+13305093095",
  32                 "account_sid": u"bar",
  33             },
  34         )
  35 
  36     def test_invalid_form(self):
  37         form = TwilioConfigurationForm(data={"sms_from": "foobar", "sms_to": "911"})
  38         self.assertFalse(form.is_valid())
  39         errors = form.errors.as_data()
  40 
  41         # extracting the message from django.forms.ValidationError
  42         # is the easiest and simplest way I've found to assert as_data
  43         for e in errors:
  44             errors[e] = map(lambda x: x.message, errors[e])
  45 
  46         self.assertDictEqual(
  47             errors,
  48             {
  49                 "auth_token": [u"This field is required."],
  50                 "account_sid": [u"This field is required."],
  51                 "sms_from": [u"foobar is not a valid phone number."],
  52                 "sms_to": [u"911 is not a valid phone number."],
  53             },
  54         )
