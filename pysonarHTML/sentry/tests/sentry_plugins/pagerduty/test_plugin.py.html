 106     def test_no_secrets(self):
 107         self.user = self.create_user("foo@example.com")
 108         self.org = self.create_organization(owner=self.user, name="Rowdy Tiger")
 109         self.team = self.create_team(organization=self.org, name="Mariachi Band")
 110         self.project = self.create_project(organization=self.org, teams=[self.team], name="Bengal")
 111         self.login_as(self.user)
 112         self.plugin.set_option("service_key", "abcdef", self.project)
 113         url = reverse(
 114             "sentry-api-0-project-plugin-details",
 115             args=[self.org.slug, self.project.slug, "pagerduty"],
 116         )
 117         res = self.client.get(url)
 118         config = json.loads(res.content)["config"]
 119         key_config = [item for item in config if item["name"] == "service_key"][0]
 120         assert key_config.get("type") == "secret"
 121         assert key_config.get("value") is None
 122         assert key_config.get("hasSavedValue") is True
 123         assert key_config.get("prefix") == "abcd"
