  26     @responses.activate
  27     def test_simple_notification(self):
  28         responses.add("POST", "http://example.com/slack")
  29         self.plugin.set_option("webhook", "http://example.com/slack", self.project)
  30 
  31         event = self.store_event(
  32             data={"message": "Hello world", "level": "warning", "culprit": "foo.bar"},
  33             project_id=self.project.id,
  34         )
  35         group = event.group
  36 
  37         rule = Rule.objects.create(project=self.project, label="my rule")
  38 
  39         notification = Notification(event=event, rule=rule)
  40 
  41         with self.options({"system.url-prefix": "http://example.com"}):
  42             self.plugin.notify(notification)
  43 
  44         request = responses.calls[0].request
  45         payload = json.loads(parse_qs(request.body)["payload"][0])
  46         assert payload == {
  47             "username": "Sentry",

  49                 {
  50                     "color": "#f18500",
  51                     "fields": [

  53                         {"short": True, "value": "bar", "title": "Project"},

  55                     "fallback": "[bar] Hello world",
  56                     "title": "Hello world",
