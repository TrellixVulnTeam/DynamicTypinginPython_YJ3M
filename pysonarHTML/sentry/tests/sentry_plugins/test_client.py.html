  55 class AuthApiClientTest(TestCase):
  56     @responses.activate
  57     def test_without_authorization(self):
  58         responses.add(responses.GET, "http://example.com", json={})
  59 
  60         resp = AuthApiClient().get("http://example.com")
  61         assert resp.status_code == 200
  62 
  63         request = responses.calls[-1].request
  64         assert not request.headers.get("Authorization")
  65 
  66     @responses.activate
  67     def test_with_authorization(self):
  68         responses.add(responses.GET, "http://example.com", json={})
  69 
  70         auth = Mock()
  71         auth.tokens = {"access_token": "access-token"}
  72 
  73         resp = AuthApiClient(auth=auth).get("http://example.com")
  74         assert resp.status_code == 200
  75 
  76         request = responses.calls[-1].request
  77         assert request.headers.get("Authorization") == "Bearer access-token"
  78 
  79     @responses.activate
  80     def test_with_authorization_and_no_auth(self):
  81         responses.add(responses.GET, "http://example.com", json={})
  82 
  83         auth = Mock()
  84         auth.tokens = {"access_token": "access-token"}
  85 
  86         resp = AuthApiClient(auth=auth).get("http://example.com", auth=None)
  87         assert resp.status_code == 200
  88 
  89         request = responses.calls[-1].request
  90         assert not request.headers.get("Authorization")
  91 
  92     @responses.activate
  93     def test_invalid_host(self):
  94         with pytest.raises(ApiHostError):
  95             AuthApiClient().get("http://example.com")
  96 
  97     @responses.activate
  98     def test_unauthorized(self):
  99         responses.add(responses.GET, "http://example.com", status=404)
 100         with pytest.raises(ApiError):

 102
