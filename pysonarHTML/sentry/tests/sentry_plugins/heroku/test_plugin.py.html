  26 class SetRefsTest(TestCase):
  27     """
  28     tests that when finish_release is called on a release hook,
  29     we try to get the previous commits based on the version ref
  30     and that we create `ReleaseHeadCommit`s for the version
  31     """
  32 
  33     @patch("sentry.tasks.commits.fetch_commits")
  34     def test_minimal(self, mock_fetch_commits):
  35         project = self.create_project()
  36         version = "bbee5b51f84611e4b14834363b8514c2"
  37         data_list = [
  38             {
  39                 "id": "c7155651831549cf8a5e47889fce17eb",
  40                 "message": "foo",
  41                 "author_email": "jane@example.com",
  42             },
  43             {
  44                 "id": "62de626b7c7cfb8e77efb4273b1a3df4123e6216",
  45                 "message": "hello",
  46                 "author_name": "Jess",
  47             },
  48             {
  49                 "id": "58de626b7c7cfb8e77efb4273b1a3df4123e6345",
  50                 "message": "bar",
  51                 "author_name": "Joe^^",
  52             },
  53             {
  54                 "id": "bbee5b51f84611e4b14834363b8514c2",
  55                 "message": "blah",
  56                 "author_email": "katie@example.com",
  57             },
  58         ]
  59         user = User.objects.create(email="stebe@sentry.io")
  60         repo = Repository.objects.create(
  61             organization_id=project.organization_id, name=project.name, provider="dummy"
  62         )
  63         ProjectOption.objects.set_value(key="heroku:repository", project=project, value=repo.name)
  64         for data in data_list:
  65             Commit.objects.create(
  66                 key=data["id"], organization_id=self.project.organization_id, repository_id=repo.id
  67             )
  68 
  69         old_release = Release.objects.create(
  70             version="a" * 40,
  71             organization_id=project.organization_id,
  72             date_added=timezone.now() - timedelta(minutes=30),
  73         )
  74         old_release.add_project(project)
  75 
  76         ReleaseCommit.objects.create(
  77             organization_id=project.organization_id,
  78             project_id=project.id,
  79             release=old_release,
  80             commit=Commit.objects.get(key="c7155651831549cf8a5e47889fce17eb"),
  81             order=0,
  82         )
  83         ReleaseHeadCommit.objects.create(
  84             organization_id=project.organization_id,
  85             repository_id=repo.id,
  86             release=old_release,
  87             commit=Commit.objects.get(key="c7155651831549cf8a5e47889fce17eb"),
  88         )
  89         release_heads = ReleaseHeadCommit.objects.filter(
  90             organization_id=project.organization_id,
  91             repository_id=repo.id,
  92             commit=Commit.objects.get(key="bbee5b51f84611e4b14834363b8514c2"),
  93         )
  94 
  95         assert len(release_heads) == 0
  96         hook = HerokuReleaseHook(project)
  97         hook.finish_release(version=version, owner=user)
  98 
  99         release = Release.objects.get(projects=project, version=version)
 100 
 101         new_release_heads = ReleaseHeadCommit.objects.filter(
 102             organization_id=project.organization_id,
 103             repository_id=repo.id,
 104             release=release,
 105             commit=Commit.objects.get(key="bbee5b51f84611e4b14834363b8514c2"),
 106         )
 107         assert len(new_release_heads) == 1
 108         assert release.version == "bbee5b51f84611e4b14834363b8514c2"
 109 
 110         deploy = Deploy.objects.filter(
 111             organization_id=project.organization_id,
 112             release=release,
 113             environment_id=Environment.objects.get(
 114                 organization_id=project.organization_id, name="production"
 115             ).id,
 116         )
 117         assert len(deploy) == 1
 118 
 119         mock_fetch_commits.apply_async.assert_called_with(
 120             kwargs={
 121                 "release_id": release.id,
 122                 "user_id": user.id,
 123                 "refs": [{"commit": "bbee5b51f84611e4b14834363b8514c2", "repository": repo.name}],
 124                 "prev_release_id": old_release.id,
 125             }
 126         )

 129 class HookHandleTest(TestCase):
 130     def test_user_success(self):
 131         user = self.create_user()
 132         organization = self.create_organization(owner=user)
 133         project = self.create_project(organization=organization)
 134         hook = HerokuReleaseHook(project)
 135         hook.set_refs = Mock()
 136 
 137         req = Mock()
 138         req.POST = {"head_long": "abcd123", "url": "http://example.com", "user": user.email}
 139         hook.handle(req)
 140         assert Release.objects.filter(version=req.POST["head_long"]).exists()
 141         assert hook.set_refs.call_count == 1
 142 
 143     def test_actor_email_success(self):
 144         user = self.create_user()
 145         organization = self.create_organization(owner=user)
 146         project = self.create_project(organization=organization)
 147         hook = HerokuReleaseHook(project)
 148         hook.set_refs = Mock()
 149 
 150         req = Mock()
 151         req.POST = {
 152             "head_long": "v999",
 153             "url": "http://example.com",
 154             "actor": {"email": user.email},
 155         }
 156         hook.handle(req)
 157         assert Release.objects.filter(version=req.POST["head_long"]).exists()
 158         assert hook.set_refs.call_count == 1
 159 
 160     def test_email_mismatch(self):
 161         user = self.create_user()
 162         organization = self.create_organization(owner=user)
 163         project = self.create_project(organization=organization)
 164         hook = HerokuReleaseHook(project)
 165 
 166         req = Mock()
 167         req.POST = {"head_long": "v999", "url": "http://example.com", "user": "wrong@example.com"}
 168         hook.handle(req)
 169         assert Release.objects.filter(version=req.POST["head_long"]).exists()
 170 
 171     def test_bad_version(self):
 172         project = self.create_project()
 173         user = self.create_user()
 174         hook = HerokuReleaseHook(project)
 175 
 176         req = Mock()
 177         req.POST = {"head_long": "", "url": "http://example.com", "user": user.email}
 178         with self.assertRaises(HookValidationError):
 179             hook.handle(req)
