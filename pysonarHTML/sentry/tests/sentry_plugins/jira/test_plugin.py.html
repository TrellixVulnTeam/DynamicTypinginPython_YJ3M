 192 class JiraPluginTest(TestCase):
 193     @fixture
 194     def plugin(self):
 195         return JiraPlugin()
 196 
 197     @fixture
 198     def request(self):
 199         return RequestFactory()
 200 
 201     def test_conf_key(self):
 202         assert self.plugin.conf_key == "jira"
 203 
 204     def test_get_issue_label(self):
 205         group = self.create_group(message="Hello world", culprit="foo.bar")
 206         assert self.plugin.get_issue_label(group, "SEN-1") == "SEN-1"
 207 
 208     def test_get_issue_url(self):
 209         self.plugin.set_option("instance_url", "https://getsentry.atlassian.net", self.project)
 210         group = self.create_group(message="Hello world", culprit="foo.bar")
 211         assert (
 212             self.plugin.get_issue_url(group, "SEN-1")
 213             == "https://getsentry.atlassian.net/browse/SEN-1"
 214         )
 215 
 216     def test_is_configured(self):
 217         assert self.plugin.is_configured(None, self.project) is False
 218         self.plugin.set_option("default_project", "SEN", self.project)
 219         assert self.plugin.is_configured(None, self.project) is True
 220 
 221     @responses.activate
 222     def test_create_issue(self):
 223         responses.add(
 224             responses.GET,
 225             "https://getsentry.atlassian.net/rest/api/2/issue/createmeta",
 226             json=create_meta_response,
 227         )
 228         responses.add(
 229             responses.POST,
 230             "https://getsentry.atlassian.net/rest/api/2/issue",
 231             json={"key": "SEN-1"},
 232         )
 233         self.plugin.set_option("instance_url", "https://getsentry.atlassian.net", self.project)
 234         group = self.create_group(message="Hello world", culprit="foo.bar")
 235 
 236         request = self.request.get("/")
 237         request.user = AnonymousUser()
 238         form_data = {
 239             "title": "Hello",
 240             "description": "Fix this.",
 241             "issuetype": "bug",
 242             "project": "SEN",
 243         }
 244         assert self.plugin.create_issue(request, group, form_data) == "SEN-1"

 246     @responses.activate
 247     def test_link_issue(self):
 248         responses.add(
 249             responses.GET,
 250             "https://getsentry.atlassian.net/rest/api/2/issue/SEN-19",
 251             json=issue_response,
 252         )
 253         self.plugin.set_option("instance_url", "https://getsentry.atlassian.net", self.project)
 254         group = self.create_group(message="Hello world", culprit="foo.bar")
 255 
 256         request = self.request.get("/")
 257         request.user = AnonymousUser()
 258         form_data = {"issue_id": "SEN-19"}
 259         assert (
 260             self.plugin.link_issue(request, group, form_data)["title"]
 261             == issue_response["fields"]["summary"]
 262         )

 264     def test_no_secrets(self):
 265         self.user = self.create_user("foo@example.com")
 266         self.org = self.create_organization(owner=self.user, name="Rowdy Tiger")
 267         self.team = self.create_team(organization=self.org, name="Mariachi Band")
 268         self.project = self.create_project(organization=self.org, teams=[self.team], name="Bengal")
 269         self.login_as(self.user)
 270         self.plugin.set_option("password", "abcdef", self.project)
 271         url = reverse(
 272             "sentry-api-0-project-plugin-details", args=[self.org.slug, self.project.slug, "jira"]
 273         )
 274         res = self.client.get(url)
 275         config = json.loads(res.content)["config"]
 276         password_config = [item for item in config if item["name"] == "password"][0]
 277         assert password_config.get("type") == "secret"
 278         assert password_config.get("value") is None
 279         assert password_config.get("hasSavedValue") is True
 280         assert password_config.get("prefix") == ""

 282     def test_get_formatted_user(self):
 283         assert self.plugin._get_formatted_user(
 284             {"displayName": "Foo Bar", "emailAddress": "foo@sentry.io", "name": "foobar"}
 285         ) == {"text": "Foo Bar - foo@sentry.io (foobar)", "id": "foobar"}
 286 
 287         # test weird addon users that don't have email addresses
 288         assert self.plugin._get_formatted_user(
 289             {
 290                 "name": "robot",
 291                 "avatarUrls": {
 292                     "16x16": "https://avatar-cdn.atlassian.com/someid",
 293                     "24x24": "https://avatar-cdn.atlassian.com/someotherid",
 294                 },
 295                 "self": "https://something.atlassian.net/rest/api/2/user?username=someaddon",
 296             }
 297         ) == {"id": "robot", "text": "robot (robot)"}
