 111     def test_no_secrets(self):
 112         self.user = self.create_user("foo@example.com")
 113         self.org = self.create_organization(owner=self.user, name="Rowdy Tiger")
 114         self.team = self.create_team(organization=self.org, name="Mariachi Band")
 115         self.project = self.create_project(organization=self.org, teams=[self.team], name="Bengal")
 116         self.login_as(self.user)
 117         self.plugin.set_option("userkey", "abcdef", self.project)
 118         self.plugin.set_option("apikey", "abcdef", self.project)
 119         url = reverse(
 120             "sentry-api-0-project-plugin-details",
 121             args=[self.org.slug, self.project.slug, "pushover"],
 122         )
 123         res = self.client.get(url)
 124         config = json.loads(res.content)["config"]
 125         userkey_config = [item for item in config if item["name"] == "userkey"][0]
 126         apikey_config = [item for item in config if item["name"] == "apikey"][0]
 127         assert userkey_config.get("type") == "secret"
 128         assert userkey_config.get("value") is None
 129         assert userkey_config.get("hasSavedValue") is True
 130         assert userkey_config.get("prefix") == "abcd"
 131         assert apikey_config.get("type") == "secret"
 132         assert apikey_config.get("value") is None
 133         assert apikey_config.get("hasSavedValue") is True
 134         assert apikey_config.get("prefix") == "abcd"
