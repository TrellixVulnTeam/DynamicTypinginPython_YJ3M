  13 class AmazonSQSPluginTest(PluginTestCase):
  14     @fixture

  18     def test_conf_key(self):
  19         assert self.plugin.conf_key == "amazon-sqs"

  24     def run_test(self):
  25         self.plugin.set_option("access_key", "access-key", self.project)
  26         self.plugin.set_option("secret_key", "secret-key", self.project)
  27         self.plugin.set_option("region", "us-east-1", self.project)
  28         self.plugin.set_option(
  29             "queue_url", "https://sqs-us-east-1.amazonaws.com/12345678/myqueue", self.project
  30         )
  31 
  32         event = self.store_event(
  33             data={
  34                 "sentry.interfaces.Exception": {"type": "ValueError", "value": "foo bar"},
  35                 "sentry.interfaces.User": {"id": "1", "email": "foo@example.com"},
  36                 "type": "error",
  37                 "metadata": {"type": "ValueError", "value": "foo bar"},
  38             },
  39             project_id=self.project.id,
  40         )
  41 
  42         with self.options({"system.url-prefix": "http://example.com"}):
  43             self.plugin.post_process(event)
  44         return event

  46     @patch("boto3.client")

  60     @patch("sentry_plugins.amazon_sqs.plugin.logger")
  61     @patch("boto3.client")
  62     def test_token_error(self, mock_client, logger):
  63         mock_client.return_value.send_message.side_effect = ClientError(
  64             {"Error": {"Code": "Hello", "Message": "hello"}}, "SendMessage"
  65         )
  66         with self.assertRaises(ClientError):
  67             self.run_test()
  68         assert not logger.info.called
  69 
  70         mock_client.return_value.send_message.side_effect = ClientError(
  71             {"Error": {"Code": "AccessDenied", "Message": "Hello"}}, "SendMessage"
  72         )
  73         self.run_test()
  74         assert len(logger.info.call_args_list) == 1
  75         assert (
  76             logger.info.call_args_list[0][0][0] == "sentry_plugins.amazon_sqs.access_token_invalid"
  77         )
