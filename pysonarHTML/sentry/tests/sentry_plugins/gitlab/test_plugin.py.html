 110     def test_no_secrets(self):
 111         self.user = self.create_user("foo@example.com")
 112         self.org = self.create_organization(owner=self.user, name="Rowdy Tiger")
 113         self.team = self.create_team(organization=self.org, name="Mariachi Band")
 114         self.project = self.create_project(organization=self.org, teams=[self.team], name="Bengal")
 115         self.login_as(self.user)
 116         self.plugin.set_option("gitlab_url", "https://gitlab.com", self.project)
 117         self.plugin.set_option("gitlab_repo", "getsentry/sentry", self.project)
 118         self.plugin.set_option("gitlab_token", "abcdefg", self.project)
 119         url = reverse(
 120             "sentry-api-0-project-plugin-details", args=[self.org.slug, self.project.slug, "gitlab"]
 121         )
 122         res = self.client.get(url)
 123         config = json.loads(res.content)["config"]
 124         token_config = [item for item in config if item["name"] == "gitlab_token"][0]
 125         assert token_config.get("type") == "secret"
 126         assert token_config.get("value") is None
 127         assert token_config.get("hasSavedValue") is True
 128         assert token_config.get("prefix") == "abcd"
