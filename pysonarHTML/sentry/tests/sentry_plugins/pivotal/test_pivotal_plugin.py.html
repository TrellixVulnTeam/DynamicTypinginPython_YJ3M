  36     def test_no_secrets(self):
  37         self.user = self.create_user("foo@example.com")
  38         self.org = self.create_organization(owner=self.user, name="Rowdy Tiger")
  39         self.team = self.create_team(organization=self.org, name="Mariachi Band")
  40         self.project = self.create_project(organization=self.org, teams=[self.team], name="Bengal")
  41         self.login_as(self.user)
  42         self.plugin.set_option("token", "abcdef", self.project)
  43         url = reverse(
  44             "sentry-api-0-project-plugin-details",
  45             args=[self.org.slug, self.project.slug, "pivotal"],
  46         )
  47         res = self.client.get(url)
  48         config = json.loads(res.content)["config"]
  49         token_config = [item for item in config if item["name"] == "token"][0]
  50         assert token_config.get("type") == "secret"
  51         assert token_config.get("value") is None
  52         assert token_config.get("hasSavedValue") is True
  53         assert token_config.get("prefix") == "abcd"
