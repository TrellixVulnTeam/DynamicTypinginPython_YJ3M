  62 class VstsPluginTest(PluginTestCase):
  63     @fixture
  64     def plugin(self):
  65         return VstsPlugin()
  66 
  67     @fixture
  68     def request(self):

  70 
  71     def test_conf_key(self):

  73 
  74     def test_entry_point(self):

  77     def test_get_issue_label(self):

  79         assert self.plugin.get_issue_label(group, {"id": 309}) == "Bug 309"

  81     def test_get_issue_url(self):

 100     @responses.activate
 101     def test_create_issue(self):
 102         responses.add(
 103             responses.PATCH,
 104             "https://fabrikam-fiber-inc.visualstudio.com/DefaultProject/_apis/wit/workitems/$Bug",
 105             body=WORK_ITEM_RESPONSE,
 106             content_type="application/json",
 107         )
 108 
 109         self.plugin.set_option("instance", "fabrikam-fiber-inc.visualstudio.com", self.project)
 110         self.plugin.set_option("default_project", "DefaultProject", self.project)
 111         group = self.create_group(message="Hello world", culprit="foo.bar")
 112 
 113         request = self.request.get("/")
 114         request.user = AnonymousUser()
 115         form_data = {"title": "Hello", "description": "Fix this."}
 116         with self.assertRaises(PluginError):
 117             self.plugin.create_issue(request, group, form_data)
 118 
 119         request.user = self.user
 120         self.login_as(self.user)
 121         UserSocialAuth.objects.create(
 122             user=self.user,
 123             provider=self.plugin.auth_provider,
 124             uid="a89e7204-9ca0-4680-ba7a-cfcf6b3c7445",
 125             extra_data={"access_token": "foo", "refresh_token": "bar"},
 126         )
 127 
 128         assert self.plugin.create_issue(request, group, form_data) == {
 129             "id": 309,
 130             "url": "https://fabrikam-fiber-inc.visualstudio.com/web/wi.aspx?pcguid=d81542e4-cdfa-4333-b082-1ae2d6c3ad16&id=309",
 131             "title": "Hello",
 132         }
 133         request = responses.calls[-1].request

 135         payload = json.loads(request.body)
