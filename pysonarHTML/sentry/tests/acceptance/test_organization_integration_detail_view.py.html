  15 class OrganizationIntegrationDetailView(AcceptanceTestCase):
  16     """
  17     As a developer, I can create an integration, install it, and uninstall it
  18     """
  19 
  20     def setUp(self):
  21         super(OrganizationIntegrationDetailView, self).setUp()
  22         features.add("organizations:integrations-feature_flag_integration", OrganizationFeature)
  23         self.login_as(self.user)
  24 
  25     def load_page(self, slug, configuration_tab=False):
  26         self.dismiss_assistant("discover_sidebar")
  27         url = u"/settings/{}/integrations/{}/".format(self.organization.slug, slug)
  28         if configuration_tab:
  29             url += "?tab=configurations"
  30         self.browser.get(url)
  31         self.browser.wait_until_not(".loading-indicator")
  32 
  33     def test_example_installation(self):
  34         self.provider = mock.Mock()
  35         self.provider.key = "example"
  36         self.provider.name = "Example Installation"
  37 
  38         self.load_page("example")
  39         self.browser.snapshot("integrations - integration detail overview")
  40 
  41         detail_view_page = OrganizationIntegrationDetailViewPage(browser=self.browser)
  42         detail_view_page.click_install_button()
  43         detail_view_page.click_through_integration_setup(
  44             ExampleIntegrationSetupWindowElement, {"name": self.provider.name}
  45         )
  46 
  47         # provider_element might be rerendered
  48         assert Integration.objects.filter(
  49             provider=self.provider.key, external_id=self.provider.name
  50         ).exists()
  51 
  52         detail_view_page.switch_to_configuration_view()
  53         assert self.browser.element_exists('[aria-label="Configure"]')
  54 
  55     def test_uninstallation(self):
  56         model = Integration.objects.create(
  57             provider="slack",
  58             external_id="some_slack",
  59             name="Test Slack",
  60             metadata={"domain_name": "slack-test.slack.com"},
  61         )
  62 
  63         model.add_organization(self.organization, self.user)
  64         self.load_page("slack", configuration_tab=True)
  65         self.browser.snapshot(name="integrations - integration detail one configuration")
  66 
  67         detail_view_page = OrganizationIntegrationDetailViewPage(browser=self.browser)
  68         assert self.browser.element_exists('[aria-label="Configure"]')
  69         detail_view_page.uninstall()
  70         assert not self.browser.element_exists('[aria-label="Configure"]')
  71         self.browser.snapshot(name="integrations - integration detail no configurations")
