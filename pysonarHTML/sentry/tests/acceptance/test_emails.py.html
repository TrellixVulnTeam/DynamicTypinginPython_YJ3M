  50 class EmailTestCase(AcceptanceTestCase):
  51     def setUp(self):
  52         super(EmailTestCase, self).setUp()
  53         self.user = self.create_user("foo@example.com")
  54         self.login_as(self.user)
  55 
  56     def build_url(self, path, format="html"):
  57         return u"{}?{}".format(path, urlencode({"format": format, "seed": "123"}))
  58 
  59     def test_emails(self):
  60         for url, name in EMAILS:
  61             # HTML output is captured as a snapshot
  62             self.browser.get(self.build_url(url, "html"))
  63             self.browser.wait_until("#preview")
  64             self.browser.snapshot(u"{} email html".format(name))
  65 
  66             # Text output is asserted against static fixture files
  67             self.browser.get(self.build_url(url, "txt"))
  68             self.browser.wait_until("#preview")
  69             elem = self.browser.find_element_by_css_selector("#preview pre")
  70             text_src = elem.get_attribute("innerHTML")
  71 
  72             fixture_src = read_txt_email_fixture(name)
  73             assert fixture_src == text_src
