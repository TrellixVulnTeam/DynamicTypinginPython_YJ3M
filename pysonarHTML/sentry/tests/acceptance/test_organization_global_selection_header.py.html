  89         Global Selection Header should:
  90         1) load project from URL if it exists

 100

 106         assert self.issues_list.global_selection.get_selected_project_slug() == self.project_2.slug
 107

 219     @patch("django.utils.timezone.now")
 220     def test_issues_list_to_details_and_back_with_all_projects(self, mock_now):
 221         """
 222         If user has access to the `global-views` feature, which allows selecting multiple projects,
 223         they should be able to visit issues list with no project in URL and list issues
 224         for all projects they are members of.
 225 
 226         They should also be able to open an issue and then navigate back to still see
 227         "My Projects" in issues list.
 228         """
 229         with self.feature("organizations:global-views"):
 230             mock_now.return_value = datetime.utcnow().replace(tzinfo=pytz.utc)
 231             self.create_issues()
 232             self.issues_list.visit_issue_list(self.org.slug)
 233             self.issues_list.wait_for_issue()
 234 
 235             assert u"project=" not in self.browser.current_url
 236             assert self.issues_list.global_selection.get_selected_project_slug() == "My Projects"
 237 
 238             # select the issue
 239             self.issues_list.navigate_to_issue(1)
 240 
 241             # going back to issues list should not have the issue's project id in url
 242             self.issues_list.issue_details.go_back_to_issues()
 243             self.issues_list.wait_for_issue()
 244 
 245             # project id should remain *NOT* in URL
 246             assert u"project=" not in self.browser.current_url
 247             assert self.issues_list.global_selection.get_selected_project_slug() == "My Projects"
 248 
 249             # can select a different project
 250             self.issues_list.global_selection.select_project_by_slug(self.project_3.slug)
 251             self.issues_list.wait_until_loaded()
 252             assert u"project={}".format(self.project_3.id) in self.browser.current_url
 253             assert (
 254                 self.issues_list.global_selection.get_selected_project_slug() == self.project_3.slug
 255             )
