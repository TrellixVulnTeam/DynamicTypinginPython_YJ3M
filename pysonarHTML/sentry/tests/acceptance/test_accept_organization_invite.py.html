   9 class AcceptOrganizationInviteTest(AcceptanceTestCase):
  10     def setUp(self):
  11         super(AcceptOrganizationInviteTest, self).setUp()
  12         self.user = self.create_user("foo@example.com")
  13         self.org = self.create_organization(name="Rowdy Tiger", owner=None)
  14         self.team = self.create_team(organization=self.org, name="Mariachi Band")
  15         self.member = self.create_member(
  16             user=None,
  17             email="bar@example.com",
  18             organization=self.org,
  19             role="owner",
  20             teams=[self.team],
  21         )
  22 
  23     def test_invite_simple(self):
  24         self.login_as(self.user)
  25         self.browser.get(self.member.get_invite_link().split("/", 3)[-1])
  26         self.browser.wait_until('[data-test-id="accept-invite"]')
  27         self.browser.snapshot(name="accept organization invite")
  28         assert self.browser.element_exists('[aria-label="join-organization"]')
  29 
  30     def test_invite_not_authenticated(self):
  31         self.browser.get(self.member.get_invite_link().split("/", 3)[-1])
  32         self.browser.wait_until('[data-test-id="accept-invite"]')
  33         assert self.browser.element_exists('[aria-label="create-account"]')
  34 
  35     def test_invite_2fa_enforced_org(self):
  36         self.org.update(flags=F("flags").bitor(Organization.flags.require_2fa))
  37         self.browser.get(self.member.get_invite_link().split("/", 3)[-1])
  38         self.browser.wait_until('[data-test-id="accept-invite"]')
  39         assert not self.browser.element_exists_by_test_id("2fa-warning")
  40 
  41         self.login_as(self.user)
  42         self.org.update(flags=F("flags").bitor(Organization.flags.require_2fa))

  47     def test_invite_sso_org(self):
