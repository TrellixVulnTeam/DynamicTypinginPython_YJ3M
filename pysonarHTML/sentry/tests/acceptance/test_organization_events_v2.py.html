  61 def generate_transaction():
  62     start_datetime = before_now(minutes=1, milliseconds=500)
  63     end_datetime = before_now(minutes=1)
  64     event_data = load_data("transaction", timestamp=end_datetime, start_timestamp=start_datetime)
  65     event_data.update({"event_id": "a" * 32})
  66 
  67     # generate and build up span tree
  68     reference_span = event_data["spans"][0]
  69     parent_span_id = reference_span["parent_span_id"]
  70 
  71     span_tree_blueprint = {
  72         "a": {},
  73         "b": {"bb": {"bbb": {"bbbb": "bbbbb"}}},
  74         "c": {},
  75         "d": {},
  76         "e": {},
  77     }
  78 
  79     time_offsets = {
  80         "a": (timedelta(), timedelta(milliseconds=10)),
  81         "b": (timedelta(milliseconds=120), timedelta(milliseconds=250)),
  82         "bb": (timedelta(milliseconds=130), timedelta(milliseconds=10)),
  83         "bbb": (timedelta(milliseconds=140), timedelta(milliseconds=10)),
  84         "bbbb": (timedelta(milliseconds=150), timedelta(milliseconds=10)),
  85         "bbbbb": (timedelta(milliseconds=160), timedelta(milliseconds=90)),
  86         "c": (timedelta(milliseconds=260), timedelta(milliseconds=100)),
  87         "d": (timedelta(milliseconds=375), timedelta(milliseconds=50)),
  88         "e": (timedelta(milliseconds=400), timedelta(milliseconds=100)),
  89     }
  90 
  91     def build_span_tree(span_tree, spans, parent_span_id):
  92         for span_id, child in span_tree.items():
  93             span = copy.deepcopy(reference_span)
  94             # non-leaf node span
  95             span["parent_span_id"] = parent_span_id.ljust(16, "0")
  96             span["span_id"] = span_id.ljust(16, "0")
  97 
  98             (start_delta, span_length) = time_offsets.get(span_id, (timedelta(), timedelta()))
  99 
 100             span_start_time = start_datetime + start_delta
 101             span["start_timestamp"] = timestamp_format(span_start_time)
 102             span["timestamp"] = timestamp_format(span_start_time + span_length)
 103             spans.append(span)
 104 
 105             if isinstance(child, dict):
 106                 spans = build_span_tree(child, spans, span_id)
 107             elif isinstance(child, six.string_types):
 108                 parent_span_id = span_id
 109                 span_id = child
 110 
 111                 span = copy.deepcopy(reference_span)
 112                 # leaf node span
 113                 span["parent_span_id"] = parent_span_id.ljust(16, "0")
 114                 span["span_id"] = span_id.ljust(16, "0")
 115 
 116                 (start_delta, span_length) = time_offsets.get(span_id, (timedelta(), timedelta()))
 117 
 118                 span_start_time = start_datetime + start_delta
 119                 span["start_timestamp"] = timestamp_format(span_start_time)
 120                 span["timestamp"] = timestamp_format(span_start_time + span_length)
 121                 spans.append(span)
 122 
 123         return spans
 124 
 125     event_data["spans"] = build_span_tree(span_tree_blueprint, [], parent_span_id)
 126 
 127     return event_data
