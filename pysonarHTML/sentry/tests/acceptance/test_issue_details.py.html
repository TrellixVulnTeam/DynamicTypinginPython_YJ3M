  31     def create_sample_event(self, platform, default=None, sample_name=None, time=None):
  32         event_data = load_data(platform, default=default, sample_name=sample_name)
  33         event_data["event_id"] = "d964fdbd649a4cf8bfc35d18082b6b0e"
  34 
  35         # Only set these properties if we were given a time.
  36         # event processing will mark old time values as processing errors.
  37         if time:
  38             event_data["received"] = time.isoformat()
  39 
  40         # We need a fallback datetime for the event
  41         if time is None:
  42             time = now - timedelta(days=2)
  43             time = time.replace(hour=0, minute=0, second=0, microsecond=0)
  44 
  45         event_data["timestamp"] = time.isoformat()
  46         event = self.store_event(
  47             data=event_data, project_id=self.project.id, assert_no_errors=False
  48         )
  49         event.group.update(
  50             first_seen=datetime(2015, 8, 13, 3, 8, 25, tzinfo=timezone.utc), last_seen=time
  51         )
  52         return event
