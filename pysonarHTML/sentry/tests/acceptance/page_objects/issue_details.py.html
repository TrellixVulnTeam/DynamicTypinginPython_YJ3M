   7 class IssueDetailsPage(BasePage):
   8     def __init__(self, browser, client):
   9         super(IssueDetailsPage, self).__init__(browser)
  10         self.client = client
  11         self.global_selection = GlobalSelectionPage(browser)
  12 
  13     def visit_issue(self, org, groupid):
  14         self.browser.get(u"/organizations/{}/issues/{}/".format(org, groupid))
  15         self.wait_until_loaded()
  16 
  17     def visit_issue_in_environment(self, org, groupid, environment):
  18         self.browser.get(
  19             u"/organizations/{}/issues/{}/?environment={}".format(org, groupid, environment)
  20         )
  21         self.browser.wait_until(".group-detail")
  22 
  23     def visit_tag_values(self, org, groupid, tag):
  24         self.browser.get(u"/organizations/{}/issues/{}/tags/{}".format(org, groupid, tag))
  25         self.browser.wait_until_not(".loading-indicator")
  26 
  27     def get_environment(self):
  28         return self.browser.find_element_by_css_selector('[data-test-id="env-label"').text.lower()
  29 
  30     def go_back_to_issues(self):
  31         self.global_selection.go_back_to_issues()
  32 
  33     def api_issue_get(self, groupid):
  34         return self.client.get(u"/api/0/issues/{}/".format(groupid))
  35 
  36     def go_to_subtab(self, name):
  37         tabs = self.browser.find_element_by_css_selector(".group-detail .nav-tabs")
  38         tabs.find_element_by_partial_link_text(name).click()
  39         self.browser.wait_until_not(".loading-indicator")
  40 
  41     def open_issue_errors(self):

  43         self.browser.wait_until(".entries > .errors ul")
  44 
  45     def open_curl(self):

  48     def resolve_issue(self):
  49         self.browser.click('[data-test-id="action-link-resolve"]')
  50         # Resolve should become unresolve
  51         self.browser.wait_until('[data-test-id="button-unresolve"]')

  53     def ignore_issue(self):
  54         self.browser.click('[data-test-id="action-link-ignore"]')
  55         # Ignore should become unresolve
  56         self.browser.wait_until('[data-test-id="button-unresolve"]')
  57 
  58     def bookmark_issue(self):

  60         self.browser.wait_until(".group-bookmark.active")
  61 
  62     def assign_to(self, user):
  63         assignee = self.browser.find_element_by_css_selector(".assigned-to")
  64 
  65         # Open the assignee picker
  66         assignee.find_element_by_css_selector('[role="button"]').click()
  67         assignee.find_element_by_tag_name("input").send_keys(user)
  68 
  69         # Click the member/team
  70         options = assignee.find_elements_by_css_selector('[data-test-id="assignee-option"]')
  71         assert len(options) > 0, "No assignees could be found."
  72         options[0].click()
  73 
  74         self.browser.wait_until_not(".loading-indicator")
  75 
  76     def find_comment_form(self):
  77         return self.browser.find_element_by_css_selector('[data-test-id="note-input-form"]')
  78 
  79     def has_comment(self, text):
  80         element = self.browser.element('[data-test-id="activity-note-body"]')
  81         return text in element.text
  82 
  83     def wait_until_loaded(self):
  84         self.browser.wait_until_not(".loading-indicator")
  85         self.browser.wait_until_test_id("event-entries")
  86         self.browser.wait_until_test_id("linked-issues")
  87         self.browser.wait_until_test_id("loaded-device-name")
  88         if self.browser.element_exists("#grouping-info"):
  89             self.browser.wait_until_test_id("loaded-grouping-info")
