   8 class DiscoverProcessorTest(TestCase, SnubaTestCase):
   9     def setUp(self):
  10         super(DiscoverProcessorTest, self).setUp()
  11         self.user = self.create_user()
  12         self.org = self.create_organization(owner=self.user)
  13         self.project1 = self.create_project(organization=self.org)
  14         self.project2 = self.create_project(organization=self.org)
  15         self.group = self.create_group(project=self.project1)
  16         self.discover_query = {
  17             "statsPeriod": "14d",
  18             "project": [self.project1.id, self.project2.id],
  19             "field": ["count(id)", "fake(field)", "issue"],
  20             "query": "",
  21         }
  22 
  23     def test_get_projects(self):
  24         project = DiscoverProcessor.get_projects(
  25             organization_id=self.org.id, query={"project": [self.project1.id]}
  26         )
  27         assert isinstance(project, list)
  28         assert project[0] == self.project1
  29         projects = DiscoverProcessor.get_projects(
  30             organization_id=self.org.id, query={"project": [self.project1.id, self.project2.id]}
  31         )
  32         assert sorted([p.id for p in projects]) == sorted([self.project1.id, self.project2.id])
  33         with self.assertRaises(ExportError):
  34             DiscoverProcessor.get_projects(organization_id=self.org.id, query={"project": [-1]})
  35 
  36     def test_handle_fields(self):
  37         processor = DiscoverProcessor(
  38             organization_id=self.org.id, discover_query=self.discover_query
  39         )
  40         assert processor.header_fields == ["count_id", "fake_field", "issue"]
  41         result_list = [{"issue": self.group.id, "issue.id": self.group.id}]
  42         new_result_list = processor.handle_fields(result_list)
  43         assert new_result_list[0] != result_list
  44         assert new_result_list[0]["issue"] == self.group.qualified_short_id
