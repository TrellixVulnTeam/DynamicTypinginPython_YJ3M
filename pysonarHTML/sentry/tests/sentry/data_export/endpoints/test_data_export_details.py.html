  15 class DataExportDetailsTest(APITestCase):
  16     endpoint = "sentry-api-0-organization-data-export-details"
  17 
  18     def setUp(self):
  19         self.user = self.create_user()
  20         self.organization = self.create_organization(owner=self.user)
  21         self.login_as(user=self.user)
  22         self.data_export = ExportedData.objects.create(
  23             user=self.user, organization=self.organization, query_type=0, query_info={"env": "test"}
  24         )
  25 
  26     def test_content(self):
  27         with self.feature("organizations:data-export"):
  28             response = self.get_valid_response(self.organization.slug, self.data_export.id)
  29         assert response.data["id"] == self.data_export.id
  30         assert response.data["user"] == {
  31             "id": six.binary_type(self.user.id),
  32             "email": self.user.email,
  33             "username": self.user.username,
  34         }
  35         assert response.data["dateCreated"] == self.data_export.date_added
  36         assert response.data["query"] == {
  37             "type": ExportQueryType.as_str(self.data_export.query_type),
  38             "info": self.data_export.query_info,
  39         }
  40 
  41     def test_early(self):
  42         with self.feature("organizations:data-export"):
  43             response = self.get_valid_response(self.organization.slug, self.data_export.id)
  44         assert response.data["dateFinished"] is None
  45         assert response.data["dateExpired"] is None
  46         assert response.data["status"] == ExportStatus.Early
  47 
  48     def test_valid(self):
  49         self.data_export.update(
  50             date_finished=timezone.now() - timedelta(weeks=2),
  51             date_expired=timezone.now() + timedelta(weeks=1),
  52         )
  53         with self.feature("organizations:data-export"):
  54             response = self.get_valid_response(self.organization.slug, self.data_export.id)
  55         assert response.data["dateFinished"] is not None
  56         assert response.data["dateFinished"] == self.data_export.date_finished
  57         assert response.data["dateExpired"] is not None
  58         assert response.data["dateExpired"] == self.data_export.date_expired
  59         assert response.data["status"] == ExportStatus.Valid
  60 
  61     def test_expired(self):
  62         self.data_export.update(
  63             date_finished=timezone.now() - timedelta(weeks=2),
  64             date_expired=timezone.now() - timedelta(weeks=1),
  65         )
  66         with self.feature("organizations:data-export"):
  67             response = self.get_valid_response(self.organization.slug, self.data_export.id)
  68         assert response.data["dateFinished"] is not None
  69         assert response.data["dateFinished"] == self.data_export.date_finished
  70         assert response.data["dateExpired"] is not None
  71         assert response.data["dateExpired"] == self.data_export.date_expired
  72         assert response.data["status"] == ExportStatus.Expired
  73 
  74     def test_no_file(self):
  75         with self.feature("organizations:data-export"):
  76             response = self.get_valid_response(self.organization.slug, self.data_export.id)
  77         assert response.data["checksum"] is None
  78         assert response.data["fileName"] is None
  79 
  80     def test_file(self):
  81         contents = b"test"
  82         file = File.objects.create(
  83             name="test.csv", type="export.csv", headers={"Content-Type": "text/csv"}
  84         )
  85         file.putfile(six.BytesIO(contents))
  86         self.data_export.update(file=file)
  87         with self.feature("organizations:data-export"):
  88             response = self.get_valid_response(self.organization.slug, self.data_export.id)
  89         assert response.data["checksum"] == sha1(contents).hexdigest()
  90         assert response.data["fileName"] == "test.csv"
