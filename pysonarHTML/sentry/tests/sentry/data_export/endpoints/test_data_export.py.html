  23     def test_authorization(self):
  24         # Without the data-export feature, the endpoint should 404
  25         self.get_valid_response(self.organization.slug, status_code=404, **self.payload)
  26         # Without project permissions, the endpoint should 403
  27         modified_payload = {"query_type": self.payload["query_type"], "query_info": {"project": -5}}
  28         with self.feature("organizations:data-export"):
  29             self.get_valid_response(self.organization.slug, status_code=403, **modified_payload)
  30         # Without the discover-basic feature, the endpoint should 403
  31         discover_payload = {"query_type": "Discover", "query_info": self.payload["query_info"]}
  32         with self.feature("organizations:data-export"):
  33             self.get_valid_response(self.organization.slug, status_code=403, **discover_payload)
  34         # With both, the endpoint should 201
  35         with self.feature(["organizations:data-export", "organizations:discover-basic"]):
  36             self.get_valid_response(self.organization.slug, status_code=201, **discover_payload)

  38     def test_new_export(self):

  40         Ensures that a request to this endpoint returns a 201 status code
  41         and an appropriate response object
  42         """
  43         with self.feature("organizations:data-export"):

  64     def test_progress_export(self):
  65         """
  66         Checks to make sure that identical requests (same payload, organization, user)
  67         are routed to the same ExportedData object, with a 200 status code
  68         """
  69         with self.feature("organizations:data-export"):
  70             response1 = self.get_response(self.organization.slug, **self.payload)
  71         data_export = ExportedData.objects.get(id=response1.data["id"])
  72         with self.feature("organizations:data-export"):
  73             response2 = self.get_valid_response(self.organization.slug, **self.payload)
  74         assert response2.data == {
  75             "id": data_export.id,
  76             "user": {
  77                 "id": six.binary_type(self.user.id),
  78                 "email": self.user.email,

  81             "dateCreated": data_export.date_added,

  93     def test_export_too_many_fields(self):
  94         """
  95         Ensures that if too many fields are requested, returns a 400 status code with the
  96         corresponding error message.
  97         """
  98         payload = {
  99             "query_type": ExportQueryType.DISCOVER_STR,
 100             "query_info": {"env": "test", "field": ["id"] * (MAX_FIELDS + 1)},
 101         }
 102         with self.feature(
 103             {"organizations:data-export": True, "organizations:discover-basic": True}
 104         ):
 105             response = self.get_valid_response(self.organization.slug, status_code=400, **payload)
 106         assert response.data == {
 107             "detail": "You can export up to 20 fields at a time. Please delete some and try again."
 108         }

 110     @freeze_time("2020-05-19 14:00:00")
 111     def test_converts_stats_period(self):
 112         """
 113         Ensures that statsPeriod is converted to start/end.
 114         """
 115         payload = {
 116             "query_type": ExportQueryType.DISCOVER_STR,
 117             "query_info": {"env": "test", "statsPeriod": "24h"},
 118         }
 119         with self.feature(
 120             {"organizations:data-export": True, "organizations:discover-basic": True}
 121         ):
 122             response = self.get_valid_response(self.organization.slug, status_code=201, **payload)
 123         data_export = ExportedData.objects.get(id=response.data["id"])
 124         query_info = data_export.query_info
 125         assert parse_datetime_string(query_info["start"]) == parse_datetime_string(
 126             "2020-05-18T14:00:00"
 127         )
 128         assert parse_datetime_string(query_info["end"]) == parse_datetime_string(
 129             "2020-05-19T14:00:00"
 130         )
 131         assert "statsPeriod" not in query_info
 132         assert "statsPeriodStart" not in query_info
 133         assert "statsPeriodSEnd" not in query_info
