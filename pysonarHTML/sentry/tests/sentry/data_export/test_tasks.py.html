 421     @patch("sentry.data_export.tasks.MAX_BATCH_SIZE", 200)
 422     @patch("sentry.data_export.models.ExportedData.email_success")
 423     def test_discover_large_batch(self, emailer):
 424         """
 425         Each row in this export requires exactly 13 bytes, with batch_size=3 and
 426         MAX_BATCH_SIZE=200, this means that each batch can export 6 batch fragments,
 427         each containing 3 rows for a total of 3 * 6 * 13 = 234 bytes per batch before
 428         it stops the current batch and starts another. This runs for 2 batches and
 429         during the 3rd batch, it will finish exporting all 50 rows.
 430         """
 431         de = ExportedData.objects.create(
 432             user=self.user,
 433             organization=self.org,
 434             query_type=ExportQueryType.DISCOVER,
 435             query_info={"project": [self.project.id], "field": ["title"], "query": ""},
 436         )
 437         with self.tasks():
 438             assemble_download(de.id, batch_size=3)
 439         de = ExportedData.objects.get(id=de.id)
 440         assert de.date_finished is not None
 441         assert de.date_expired is not None
 442         assert de.file is not None
 443         assert isinstance(de.file, File)
 444 
 445         assert emailer.called
