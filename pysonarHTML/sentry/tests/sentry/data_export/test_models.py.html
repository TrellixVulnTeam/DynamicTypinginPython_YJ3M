  36     def test_status_property(self):

  38         self.data_export.update(

  61     def test_delete_file(self):
  62         # Empty call should have no effect
  63         assert self.data_export.file is None
  64         self.data_export.delete_file()
  65         assert self.data_export.file is None
  66         # Real call should delete the file
  67         assert File.objects.filter(id=self.file1.id).exists()
  68         self.data_export.update(file=self.file1)
  69         assert isinstance(self.data_export.file, File)
  70         self.data_export.delete_file()
  71         assert not File.objects.filter(id=self.file1.id).exists()
  72         # The ExportedData should be unaffected
  73         assert ExportedData.objects.filter(id=self.data_export.id).exists()
  74         assert ExportedData.objects.get(id=self.data_export.id).file is None

  76     def test_delete(self):

  80         self.data_export.delete()

  84     def test_finalize_upload(self):
  85         # With default expiration
  86         with tempfile.TemporaryFile() as tf:
  87             tf.write(self.TEST_STRING)
  88             tf.seek(0)
  89             self.file1.putfile(tf)
  90         self.data_export.finalize_upload(file=self.file1)
  91         assert self.data_export.file.getfile().read() == self.TEST_STRING
  92         assert self.data_export.date_finished is not None
  93         assert self.data_export.date_expired is not None
  94         assert self.data_export.date_expired == self.data_export.date_finished + DEFAULT_EXPIRATION
  95         # With custom expiration
  96         with tempfile.TemporaryFile() as tf:
  97             tf.write(self.TEST_STRING + self.TEST_STRING)
  98             tf.seek(0)
  99             self.file2.putfile(tf)
 100         self.data_export.finalize_upload(file=self.file2, expiration=timedelta(weeks=2))
 101         assert self.data_export.file.getfile().read() == self.TEST_STRING + self.TEST_STRING
 102         # Ensure the first file is deleted
 103         assert not File.objects.filter(id=self.file1.id).exists()
 104         assert self.data_export.date_expired == self.data_export.date_finished + timedelta(weeks=2)

 106     def test_email_success(self):
 107         # Shouldn't send if ExportedData is incomplete
 108         with self.tasks():
 109             self.data_export.email_success()
 110         assert len(mail.outbox) == 0
 111         # Should send one email if complete
 112         self.data_export.finalize_upload(file=self.file1)

 114             self.data_export.email_success()
