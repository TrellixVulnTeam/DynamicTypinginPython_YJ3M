   9 def test_builtin_types():
  10     encoder = Encoder()
  11     values = [
  12         1,
  13         1.1,
  14         b"\x00\x01\x02",
  15         u"\N{SNOWMAN}",
  16         ("a", "b", "c"),
  17         ["a", "b", "c"],
  18         {"a": 1, "b": 2, "c": 3},
  19         set(["a", "b", "c"]),
  20         frozenset(["a", "b", "c"]),
  21         [{"a": 1}, set("b"), ["c"], u"text"],
  22     ]
  23 
  24     try:
  25         values.append(long(1))  # noqa
  26     except NameError:
  27         pass
  28 
  29     for value in values:
  30         encoded = encoder.dumps(value)
  31         assert isinstance(encoded, six.binary_type)
  32 
  33     with pytest.raises(TypeError):
  34         encoder.dumps(object())
