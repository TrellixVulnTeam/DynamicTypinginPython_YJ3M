  11 class MinHashSignatureBuilderTestCase(TestCase):
  12     def test_signatures(self):
  13         n = 32
  14         r = 0xFFFF
  15         get_signature = MinHashSignatureBuilder(n, r)
  16         get_signature(set(["foo", "bar", "baz"])) == get_signature(set(["foo", "bar", "baz"]))
  17 
  18         assert len(get_signature("hello world")) == n
  19         for value in get_signature("hello world"):
  20             assert 0 <= value < r
  21 
  22         a = set("the quick grown box jumps over the hazy fog".split())
  23         b = set("the quick brown fox jumps over the lazy dog".split())
  24 
  25         results = Counter(
  26             map(lambda l__r: l__r[0] == l__r[1], zip(get_signature(a), get_signature(b)))
  27         )
  28 
  29         similarity = len(a & b) / float(len(a | b))
  30         estimation = results[True] / float(sum(results.values()))
  31 
  32         self.assertAlmostEqual(
  33             similarity, estimation, delta=0.1  # totally made up constant, seems reasonable
