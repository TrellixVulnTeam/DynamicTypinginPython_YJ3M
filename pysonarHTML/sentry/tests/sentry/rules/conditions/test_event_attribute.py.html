 254     def test_stacktrace_filename(self):
 255         """Stacktrace.filename should match frames anywhere in the stack."""
 256 
 257         event = self.get_event(
 258             exception={
 259                 "values": [
 260                     {
 261                         "type": "SyntaxError",
 262                         "value": "hello world",
 263                         "stacktrace": {
 264                             "frames": [
 265                                 {"filename": "example.php", "module": "example"},
 266                                 {"filename": "somecode.php", "module": "somecode"},
 267                                 {"filename": "othercode.php", "module": "othercode"},
 268                             ]
 269                         },
 270                     }
 271                 ]
 272             }
 273         )
 274 
 275         # correctly matching filenames, at various locations in the stacktrace
 276         for value in ["example.php", "somecode.php", "othercode.php"]:
 277             rule = self.get_rule(
 278                 data={"match": MatchType.EQUAL, "attribute": "stacktrace.filename", "value": value}
 279             )
 280             self.assertPasses(rule, event)
 281 
 282         # non-matching filename
 283         rule = self.get_rule(
 284             data={"match": MatchType.EQUAL, "attribute": "stacktrace.filename", "value": "foo.php"}
 285         )
 286         self.assertDoesNotPass(rule, event)

 322     def test_stacktrace_code(self):
 323         """Stacktrace.code should match frames anywhere in the stack."""
 324 
 325         event = self.get_event(
 326             exception={
 327                 "values": [
 328                     {
 329                         "type": "NameError",
 330                         "value": "name 'hi' is not defined",
 331                         "stacktrace": {
 332                             "frames": [
 333                                 {
 334                                     "filename": "example.py",
 335                                     "module": "example",
 336                                     "function": "foo",
 337                                     "context_line": "somecode.bar()",
 338                                 },
 339                                 {
 340                                     "filename": "somecode.py",
 341                                     "module": "somecode",
 342                                     "function": "bar",
 343                                     "context_line": "othercode.baz()",
 344                                 },
 345                                 {
 346                                     "filename": "othercode.py",
 347                                     "module": "othercode",
 348                                     "function": "baz",
 349                                     "context_line": "hi()",
 350                                 },
 351                             ]
 352                         },
 353                     }
 354                 ]
 355             }
 356         )
 357 
 358         # correctly matching code, at various locations in the stacktrace
 359         for value in ["somecode.bar()", "othercode.baz()", "hi()"]:
 360             rule = self.get_rule(
