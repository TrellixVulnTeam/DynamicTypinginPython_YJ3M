 143 class EventUniqueUserFrequencyConditionTestCase(FrequencyConditionMixin, RuleTestCase):
 144     rule_cls = EventUniqueUserFrequencyCondition
 145 
 146     sequence = itertools.count()  # generates unique values, class scope doesn't matter
 147 
 148     def increment(self, event, count, environment_id, timestamp=None):
 149         tsdb.record(
 150             tsdb.models.users_affected_by_group,
 151             event.group_id,
 152             [next(self.sequence) for _ in xrange(0, count)],
 153             environment_id=environment_id,
 154             timestamp=timestamp,
 155         )
