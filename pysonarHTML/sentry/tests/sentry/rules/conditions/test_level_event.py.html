  10     def test_render_label(self):

  14     def test_equals(self):

  43     # This simulates the following case:
  44     # - Rule is setup to accept >= error
  45     # - error event finishes the save_event task, group has a level of error
  46     # - warning event finishes the save event, group now has a level of warning
  47     # - error event starts post_process_group should pass even though the group
  48     #   has a warning level set
  49     #
  50     # Specifically here to make sure the check is properly checking the event's level
  51     def test_differing_levels(self):
  52         eevent = self.store_event(data={"level": "error"}, project_id=self.project.id)
  53         wevent = self.store_event(data={"level": "warning"}, project_id=self.project.id)
  54         assert wevent.event_id != eevent.event_id
  55         assert wevent.group.id == eevent.group.id
  56 
  57         rule = self.get_rule(data={"match": MatchType.GREATER_OR_EQUAL, "level": "40"})
  58         self.assertDoesNotPass(rule, wevent)
  59         self.assertPasses(rule, eevent)
