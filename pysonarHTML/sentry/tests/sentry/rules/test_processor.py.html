  14 class RuleProcessorTest(TestCase):
  15     def setUp(self):
  16         self.event = self.store_event(data={}, project_id=self.project.id)
  17 
  18         action_data = {
  19             "id": "sentry.mail.actions.NotifyEmailAction",
  20             "targetType": ActionTargetType.ISSUE_OWNERS.value,
  21             "targetIdentifier": None,
  22         }
  23         condition_data = {"id": "sentry.rules.conditions.every_event.EveryEventCondition"}
  24 
  25         Rule.objects.filter(project=self.event.project).delete()
  26         self.rule = Rule.objects.create(
  27             project=self.event.project,
  28             data={"conditions": [condition_data], "actions": [action_data]},
  29         )
  30 
  31     # this test relies on a few other tests passing
  32     def test_integrated(self):
  33         rp = RuleProcessor(
  34             self.event,
  35             is_new=True,
  36             is_regression=True,
  37             is_new_group_environment=True,
  38             has_reappeared=True,
  39         )
  40         results = list(rp.apply())
  41         assert len(results) == 1
  42         callback, futures = results[0]
  43         assert len(futures) == 1
  44         assert futures[0].rule == self.rule
  45         assert futures[0].kwargs == {}
  46 
  47         # should not apply twice due to default frequency
  48         results = list(rp.apply())
  49         assert len(results) == 0
  50 
  51         # now ensure that moving the last update backwards
  52         # in time causes the rule to trigger again
  53         GroupRuleStatus.objects.filter(rule=self.rule).update(
  54             last_active=timezone.now() - timedelta(minutes=Rule.DEFAULT_FREQUENCY + 1)
  55         )
  56 
  57         results = list(rp.apply())
  58         assert len(results) == 1
  59 
  60     def test_ignored_issue(self):

  70         results = list(rp.apply())
