  10 def dump_obj(obj):
  11     if not isinstance(getattr(obj, "__dict__", None), dict):
  12         return obj
  13     rv = {}
  14     for (key, value) in six.iteritems(obj.__dict__):
  15         if isinstance(value, list):
  16             rv[key] = [dump_obj(x) for x in value]
  17         elif isinstance(value, dict):
  18             rv[key] = {k: dump_obj(v) for k, v in six.iteritems(value)}
  19         else:
  20             rv[key] = value
  21     return rv

  24 def test_basic_parsing(insta_snapshot):
  25     enhancement = Enhancements.from_config_string(
  26         """
  27 # This is a config
  28 path:*/code/game/whatever/*                     +app
  29 function:panic_handler                          ^-group -group
  30 function:ThreadStartWin32                       v-group
  31 function:ThreadStartLinux                       v-group
  32 function:ThreadStartMac                         v-group
  33 family:native module:std::*                     -app
  34 module:core::*                                  -app
  35 family:javascript path:*/test.js                -app
  36 family:javascript app:1 path:*/test.js          -app
  37 family:native                                   max-frames=3
  38 """,
  39         bases=["common:v1"],
  40     )
  41 
  42     dumped = enhancement.dumps()
  43     insta_snapshot(dump_obj(enhancement))
  44     assert Enhancements.loads(dumped).dumps() == dumped
  45     assert Enhancements.loads(dumped)._to_config_structure() == enhancement._to_config_structure()
  46     assert isinstance(dumped, six.string_types)

  49 def test_basic_path_matching():
  50     enhancement = Enhancements.from_config_string(
  51         """
  52         path:**/test.js              +app
  53     """
  54     )
  55     js_rule = enhancement.rules[0]
  56 
  57     assert bool(
  58         js_rule.get_matching_frame_actions(
  59             {"abs_path": "http://example.com/foo/test.js", "filename": "/foo/test.js"}, "javascript"
  60         )
  61     )
  62 
  63     assert not bool(
  64         js_rule.get_matching_frame_actions(
  65             {"abs_path": "http://example.com/foo/bar.js", "filename": "/foo/bar.js"}, "javascript"
  66         )
  67     )
  68 
  69     assert bool(
  70         js_rule.get_matching_frame_actions(
  71             {"abs_path": "http://example.com/foo/test.js"}, "javascript"
  72         )
  73     )
  74 
  75     assert not bool(js_rule.get_matching_frame_actions({"filename": "/foo/bar.js"}, "javascript"))
  76 
  77     assert bool(
  78         js_rule.get_matching_frame_actions(
  79             {"abs_path": "http://example.com/foo/TEST.js"}, "javascript"
  80         )
  81     )
  82 
  83     assert not bool(
  84         js_rule.get_matching_frame_actions(
  85             {"abs_path": "http://example.com/foo/bar.js"}, "javascript"

  90 def test_family_matching():
  91     enhancement = Enhancements.from_config_string(
  92         """
  93         family:javascript path:**/test.js              +app
  94         family:native function:std::*                  -app
  95     """
  96     )
  97     js_rule, native_rule = enhancement.rules
  98 
  99     assert bool(
 100         js_rule.get_matching_frame_actions(
 101             {"abs_path": "http://example.com/foo/TEST.js"}, "javascript"
 102         )
 103     )
 104 
 105     assert not bool(
 106         js_rule.get_matching_frame_actions({"abs_path": "http://example.com/foo/TEST.js"}, "native")
 107     )
 108 
 109     assert not bool(
 110         native_rule.get_matching_frame_actions(
 111             {"abs_path": "http://example.com/foo/TEST.js", "function": "std::whatever"},
 112             "javascript",
 113         )
 114     )
 115 
 116     assert bool(native_rule.get_matching_frame_actions({"function": "std::whatever"}, "native"))

 119 def test_app_matching():
 120     enhancement = Enhancements.from_config_string(
 121         """
 122         family:javascript path:**/test.js app:yes       +app
 123         family:native path:**/test.c app:no            -group
 124     """
 125     )
 126     app_yes_rule, app_no_rule = enhancement.rules
 127 
 128     assert bool(
 129         app_yes_rule.get_matching_frame_actions(
 130             {"abs_path": "http://example.com/foo/TEST.js", "in_app": True}, "javascript"
 131         )
 132     )
 133     assert not bool(
 134         app_yes_rule.get_matching_frame_actions(
 135             {"abs_path": "http://example.com/foo/TEST.js", "in_app": False}, "javascript"
 136         )
 137     )
 138 
 139     assert bool(
 140         app_no_rule.get_matching_frame_actions({"abs_path": "/test.c", "in_app": False}, "native")
 141     )
 142     assert not bool(
 143         app_no_rule.get_matching_frame_actions({"abs_path": "/test.c", "in_app": True}, "native")
 144     )

 147 def test_package_matching():
 148     # This tests a bunch of different rules from the default in-app logic that
 149     # was ported from the former native plugin.
 150     enhancement = Enhancements.from_config_string(
 151         """
 152         family:native package:/var/**/Frameworks/**                  -app
 153         family:native package:**/*.app/Contents/**                   +app
 154         family:native package:linux-gate.so                          -app
 155         family:native package:?:/Windows/**                          -app
 156     """
 157     )
 158 
 159     bundled_rule, macos_rule, linux_rule, windows_rule = enhancement.rules
 160 
 161     assert bool(
 162         bundled_rule.get_matching_frame_actions(
 163             {"package": "/var/containers/MyApp/Frameworks/libsomething"}, "native"
 164         )
 165     )
 166 
 167     assert bool(
 168         macos_rule.get_matching_frame_actions(
 169             {"package": "/Applications/MyStuff.app/Contents/MacOS/MyStuff"}, "native"
 170         )
 171     )
 172 
 173     assert bool(linux_rule.get_matching_frame_actions({"package": "linux-gate.so"}, "native"))
 174 
 175     assert bool(
 176         windows_rule.get_matching_frame_actions(
 177             {"package": "D:\\Windows\\System32\\kernel32.dll"}, "native"
 178         )
 179     )
 180 
 181     assert bool(
 182         windows_rule.get_matching_frame_actions(
 183             {"package": "d:\\windows\\System32\\kernel32.dll"}, "native"
 184         )
 185     )
 186 
 187     assert not bool(
 188         bundled_rule.get_matching_frame_actions(
 189             {"package": "/var2/containers/MyApp/Frameworks/libsomething"}, "native"
 190         )
 191     )
 192 
 193     assert not bool(
 194         bundled_rule.get_matching_frame_actions(
 195             {"package": "/var/containers/MyApp/MacOs/MyApp"}, "native"
 196         )
 197     )
 198 
 199     assert not bool(
 200         bundled_rule.get_matching_frame_actions({"package": "/usr/lib/linux-gate.so"}, "native")
 201     )
