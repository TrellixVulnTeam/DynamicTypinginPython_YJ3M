  16 def dump_variant(variant, lines=None, indent=0):
  17     if lines is None:
  18         lines = []
  19 
  20     def _dump_component(component, indent):
  21         if not component.hint and not component.values:
  22             return
  23         lines.append(
  24             "%s%s%s%s"
  25             % (
  26                 "  " * indent,
  27                 component.id,
  28                 component.contributes and "*" or "",
  29                 component.hint and " (%s)" % component.hint or "",
  30             )
  31         )
  32         for value in component.values:
  33             if isinstance(value, GroupingComponent):
  34                 _dump_component(value, indent + 1)
  35             else:
  36                 lines.append("%s%s" % ("  " * (indent + 1), json.dumps(value)))
  37 
  38     lines.append("%shash: %s" % ("  " * indent, json.dumps(variant.get_hash())))
  39     for (key, value) in sorted(variant.__dict__.items()):
  40         if isinstance(value, GroupingComponent):
  41             lines.append("%s%s:" % ("  " * indent, key))
  42             _dump_component(value, indent + 1)
  43         elif key == "config":
  44             # We do not want to dump the config
  45             continue
  46         else:
  47             lines.append("%s%s: %s" % ("  " * indent, key, json.dumps(value)))
  48 
  49     return lines
