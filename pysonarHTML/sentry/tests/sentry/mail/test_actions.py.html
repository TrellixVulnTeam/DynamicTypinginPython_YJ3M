 105     def test_full_integration(self):
 106         one_min_ago = iso_format(before_now(minutes=1))
 107         event = self.store_event(
 108             data={
 109                 "message": "hello",
 110                 "exception": {"type": "Foo", "value": "uh oh"},
 111                 "level": "error",
 112                 "timestamp": one_min_ago,
 113             },
 114             project_id=self.project.id,
 115             assert_no_errors=False,
 116         )
 117         action_data = {
 118             "id": "sentry.mail.actions.NotifyEmailAction",
 119             "targetType": "Member",
 120             "targetIdentifier": six.text_type(self.user.id),
 121         }
 122         condition_data = {"id": "sentry.rules.conditions.first_seen_event.FirstSeenEventCondition"}
 123 
 124         Rule.objects.filter(project=event.project).delete()
 125         Rule.objects.create(
 126             project=event.project, data={"conditions": [condition_data], "actions": [action_data]}
 127         )
 128 
 129         with self.tasks():
 130             post_process_group(
 131                 event=event, is_new=True, is_regression=False, is_new_group_environment=False
 132             )
 133 
 134         assert len(mail.outbox) == 1
 135         sent = mail.outbox[0]
 136         assert sent.to == [self.user.email]
 137         assert "uh oh" in sent.subject
