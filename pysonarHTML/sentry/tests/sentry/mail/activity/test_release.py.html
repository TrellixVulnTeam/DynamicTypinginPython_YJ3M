 171     def test_simple(self):
 172         email = ReleaseActivityEmail(
 173             Activity(
 174                 project=self.project,
 175                 user=self.user,
 176                 type=Activity.RELEASE,
 177                 data={"version": self.release.version, "deploy_id": self.deploy.id},
 178             )
 179         )
 180         # user is included because they committed
 181         # user2 committed but isn't in a team associated with the project.
 182         # user3 is included because they oped into all deploy emails
 183         # user4 committed but isn't included because they opted out of all deploy emails
 184         # for that org -- also tests to make sure org overrides default preference
 185         # user5 committed with another email address and is still included.
 186 
 187         assert len(email.get_participants()) == 3
 188 
 189         assert email.get_participants() == {
 190             self.user: GroupSubscriptionReason.committed,
 191             self.user3: GroupSubscriptionReason.deploy_setting,
 192             self.user5: GroupSubscriptionReason.committed,
 193         }
 194 
 195         context = email.get_context()
 196         assert context["environment"] == "production"
 197         assert context["repos"][0]["commits"] == [
 198             (self.commit, self.user),
 199             (self.commit2, self.user2),
 200             (self.commit3, self.user4),
 201             (self.commit4, self.user5),
 202         ]
 203 
 204         user_context = email.get_user_context(self.user)
 205         # make sure this only includes projects user has access to
 206         assert len(user_context["projects"]) == 1
 207         assert user_context["projects"][0][0] == self.project
 208 
 209         with self.tasks():
 210             email.send()
 211 
 212         assert len(mail.outbox) == 3
 213 
 214         sent_email_addresses = {msg.to[0] for msg in mail.outbox}
 215 
 216         assert sent_email_addresses == {self.user.email, self.user3.email, self.user5.email}
