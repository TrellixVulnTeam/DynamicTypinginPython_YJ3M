 244     def test_cache_unavailable(self):
 245         self.manager.set("foo", "bar")
 246         self.store.flush_local_cache()
 247 
 248         with patch.object(self.store.cache, "get", side_effect=Exception()):
 249             assert self.manager.get("foo") == "bar"
 250             self.store.flush_local_cache()
 251 
 252             with patch.object(self.store.cache, "set", side_effect=Exception()):
 253                 assert self.manager.get("foo") == "bar"
 254                 self.store.flush_local_cache()
 255 
 256                 # we should still be able to write a new value
 257                 self.manager.set("foo", "baz")
 258                 self.store.flush_local_cache()
 259 
 260         # the cache should be incorrect now, but sync_options will eventually
 261         # correct the state
 262         assert self.manager.get("foo") == "bar"
 263         self.store.flush_local_cache()
 264 
 265         # when the cache poofs, the db will be return the most-true answer
 266         with patch.object(self.store.cache, "get", side_effect=Exception()):
 267             assert self.manager.get("foo") == "baz"
 268             self.store.flush_local_cache()
 269 
 270             with patch.object(self.store.cache, "set", side_effect=Exception()):
 271                 assert self.manager.get("foo") == "baz"
 272                 self.store.flush_local_cache()

 287         self.manager.register("nostorerequired", flags=FLAG_NOSTORE | FLAG_REQUIRED)
