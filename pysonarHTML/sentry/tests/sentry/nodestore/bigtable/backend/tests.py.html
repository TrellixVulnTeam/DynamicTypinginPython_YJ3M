  55     def test_cache(self):
  56         node_1 = ("a" * 32, {"foo": "a"})
  57         node_2 = ("b" * 32, {"foo": "b"})
  58         node_3 = ("c" * 32, {"foo": "c"})
  59 
  60         for node_id, data in [node_1, node_2, node_3]:
  61             self.ns.set(node_id, data)
  62 
  63         # Get / get multi populates cache
  64         assert self.ns.get(node_1[0]) == node_1[1]
  65         assert self.ns.get_multi([node_2[0], node_3[0]]) == {
  66             node_2[0]: node_2[1],
  67             node_3[0]: node_3[1],
  68         }
  69         with mock.patch.object(self.ns.connection, "read_row") as mock_read_row:
  70             assert self.ns.get(node_1[0]) == node_1[1]
  71             assert self.ns.get(node_2[0]) == node_2[1]
  72             assert self.ns.get(node_3[0]) == node_3[1]
  73             assert mock_read_row.call_count == 0
  74 
  75         with mock.patch.object(self.ns.connection, "read_rows") as mock_read_rows:
  76             assert self.ns.get_multi([node_1[0], node_2[0], node_3[0]])
  77             assert mock_read_rows.call_count == 0
  78 
  79         # Manually deleted item should still retreivable from cache
  80         row = self.ns.connection.row(node_1[0])
  81         row.delete()
  82         row.commit()
  83         assert self.ns.get(node_1[0]) == node_1[1]
  84         assert self.ns.get_multi([node_1[0], node_2[0]]) == {
  85             node_1[0]: node_1[1],
  86             node_2[0]: node_2[1],
  87         }
  88 
  89         # Deletion clars cache
  90         self.ns.delete(node_1[0])
  91         assert self.ns.get_multi([node_1[0], node_2[0]]) == {node_1[0]: None, node_2[0]: node_2[1]}
  92         self.ns.delete_multi([node_1[0], node_2[0]])
  93         assert self.ns.get_multi([node_1[0], node_2[0]]) == {node_1[0]: None, node_2[0]: None}
  94 
  95         # Setting the item updates cache
  96         new_value = {"event_id": "d" * 32}
  97         self.ns.set(node_1[0], new_value)
  98         with mock.patch.object(self.ns.connection, "read_row") as mock_read_row:
  99             assert self.ns.get(node_1[0]) == new_value
 100             assert mock_read_row.call_count == 0
 101 
 102         # Missing rows are never cached
 103         assert self.ns.get("node_4") is None
 104         with mock.patch.object(self.ns.connection, "read_row") as mock_read_row:
 105             mock_read_row.return_value = None
 106             self.ns.get("node_4")
 107             self.ns.get("node_4")
 108             assert mock_read_row.call_count == 2
