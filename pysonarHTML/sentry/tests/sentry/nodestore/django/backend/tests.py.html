  82     def test_cache(self):
  83         node_1 = ("a" * 32, {"foo": "a"})
  84         node_2 = ("b" * 32, {"foo": "b"})
  85         node_3 = ("c" * 32, {"foo": "c"})
  86 
  87         for node_id, data in [node_1, node_2, node_3]:
  88             Node.objects.create(id=node_id, data=data)
  89 
  90         # Get / get multi populates cache
  91         assert self.ns.get(node_1[0]) == node_1[1]
  92         assert self.ns.get_multi([node_2[0], node_3[0]]) == {
  93             node_2[0]: node_2[1],
  94             node_3[0]: node_3[1],
  95         }
  96         with mock.patch.object(Node.objects, "get") as mock_get:
  97             assert self.ns.get(node_1[0]) == node_1[1]
  98             assert self.ns.get(node_2[0]) == node_2[1]
  99             assert self.ns.get(node_3[0]) == node_3[1]
 100             assert mock_get.call_count == 0
 101 
 102         with mock.patch.object(Node.objects, "filter") as mock_filter:
 103             assert self.ns.get_multi([node_1[0], node_2[0], node_3[0]])
 104             assert mock_filter.call_count == 0
 105 
 106         # Manually deleted item should still retreivable from cache
 107         Node.objects.get(id=node_1[0]).delete()
 108         assert self.ns.get(node_1[0]) == node_1[1]
 109         assert self.ns.get_multi([node_1[0], node_2[0]]) == {
 110             node_1[0]: node_1[1],
 111             node_2[0]: node_2[1],
 112         }
 113 
 114         # Deletion clars cache
 115         self.ns.delete(node_1[0])
 116         assert self.ns.get_multi([node_1[0], node_2[0]]) == {node_2[0]: node_2[1]}
 117         self.ns.delete_multi([node_1[0], node_2[0]])
 118         assert self.ns.get_multi([node_1[0], node_2[0]]) == {}
 119 
 120         # Setting the item updates cache
 121         new_value = {"event_id": "d" * 32}
 122         self.ns.set(node_1[0], new_value)
 123         with mock.patch.object(Node.objects, "get") as mock_get:
 124             assert self.ns.get(node_1[0]) == new_value
 125             assert mock_get.call_count == 0
 126 
 127         # Missing rows are never cached
 128         assert self.ns.get("node_4") is None
 129         with mock.patch.object(Node.objects, "get") as mock_get:
 130             mock_get.side_effect = Node.DoesNotExist
 131             self.ns.get("node_4")
 132             self.ns.get("node_4")
 133             assert mock_get.call_count == 2
