 119     def create_dif(self, fileobj=None, file=None, **kwargs):
 120         args = {
 121             "file_format": "macho",
 122             "arch": "x86_64",

 126         }

 128         args.update(kwargs)

 186         file = self.create_file(
 187             name="crash.dsym", checksum="dc1e3f3e411979d336c3057cce64294f3420f93a"

 237 class DebugFilesClearTest(APITestCase):
 238     def test_simple_cache_clear(self):
 239         project = self.create_project(name="foo")
 240 
 241         url = reverse(
 242             "sentry-api-0-dsym-files",
 243             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
 244         )
 245 
 246         self.login_as(user=self.user)
 247 
 248         out = BytesIO()
 249         f = zipfile.ZipFile(out, "w")
 250         f.writestr("proguard/%s.txt" % PROGUARD_UUID, PROGUARD_SOURCE)
 251         f.writestr("ignored-file.txt", b"This is just some stuff")
 252         f.close()
 253 
 254         response = self.client.post(
 255             url,
 256             {
 257                 "file": SimpleUploadedFile(
 258                     "symbols.zip", out.getvalue(), content_type="application/zip"
 259                 )
 260             },
 261             format="multipart",
 262         )
 263 
 264         assert response.status_code == 201, response.content
 265         assert len(response.data) == 1
 266         assert response.data[0]["headers"] == {"Content-Type": "text/x-proguard+plain"}
 267         assert response.data[0]["sha1"] == "e6d3c5185dac63eddfdc1a5edfffa32d46103b44"
 268         assert response.data[0]["uuid"] == PROGUARD_UUID
 269         assert response.data[0]["objectName"] == "proguard-mapping"
 270         assert response.data[0]["cpuName"] == "any"
 271         assert response.data[0]["symbolType"] == "proguard"
 272 
 273         difs = ProjectDebugFile.difcache.fetch_difs(
 274             project=project, debug_ids=[PROGUARD_UUID], features=["mapping"]
 275         )
 276         assert len(difs) == 1
 277         assert os.path.isfile(difs[PROGUARD_UUID])
 278 
 279         # if we clear now, nothing happens
 280         ProjectDebugFile.difcache.clear_old_entries()
 281         assert os.path.isfile(difs[PROGUARD_UUID])
 282 
 283         # Put the time into the future
 284         real_time = time.time
 285         time.time = lambda: real_time() + 60 * 60 * 48
 286         try:
 287             ProjectDebugFile.difcache.clear_old_entries()
 288         finally:
 289             time.time = real_time
 290 
 291         # But it's gone now
 292         assert not os.path.isfile(difs[PROGUARD_UUID])
