 176             key="alksdflskdfjsldkfajsflkslk",
 177         )
 178         commit2 = Commit.objects.create(
 179             organization_id=org.id,
 180             repository_id=repo.id,

 194                 {"id": commit2.key, "repository": repo.name},

 210         assert release.authors == []

 229             [
 230                 {

 232                     "repository": repo.name,

 234                     "author_name": "foo bar baz",
 235                     "message": "i fixed a bug",

 237                 {

 239                     "repository": repo.name,

 267

 270         assert commit.message == "i fixed another bug"
 271         assert commit.author_id == author.id

 276         assert ReleaseCommit.objects.filter(
 277             commit__key="b" * 40, commit__repository_id=repo.id, release=release

 279         assert ReleaseCommit.objects.filter(
 280             commit__key="c" * 40, commit__repository_id=repo.id, release=release
 281         ).exists()

 284             group_id=group.id, linked_type=GroupLink.LinkedType.commit, linked_id=commit_c.id

 302     def test_using_saved_data(self):
 303         org = self.create_organization()
 304         project = self.create_project(organization=org, name="foo")
 305 
 306         repo = Repository.objects.create(organization_id=org.id, name="test/repo")
 307 
 308         author = CommitAuthor.objects.create(
 309             name="foo bar baz", email="foo@example.com", organization_id=org.id
 310         )
 311 
 312         Commit.objects.create(
 313             repository_id=repo.id,
 314             organization_id=org.id,
 315             key="b" * 40,
 316             author=author,
 317             date_added="2019-03-01 12:00:00",
 318             message="fixed a thing",
 319         )
 320 
 321         release = Release.objects.create(version="abcdabc", organization=org)
 322         release.add_project(project)
