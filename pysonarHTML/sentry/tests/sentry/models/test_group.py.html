 112     def test_get_group_with_redirect(self):
 113         group = self.create_group()
 114         assert get_group_with_redirect(group.id) == (group, False)
 115 
 116         duplicate_id = self.create_group().id
 117         Group.objects.filter(id=duplicate_id).delete()
 118         GroupRedirect.objects.create(group_id=group.id, previous_group_id=duplicate_id)
 119 
 120         assert get_group_with_redirect(duplicate_id) == (group, True)
 121 
 122         # We shouldn't end up in a case where the redirect points to a bad
 123         # reference, but testing this path for completeness.
 124         group.delete()
 125 
 126         with pytest.raises(Group.DoesNotExist):
 127             get_group_with_redirect(duplicate_id)
