   9 class ProjectOptionManagerTest(TestCase):
  10     def test_set_value(self):
  11         ProjectOption.objects.set_value(self.project, "foo", "bar")
  12         assert ProjectOption.objects.get(project=self.project, key="foo").value == "bar"
  13 
  14     def test_get_value(self):
  15         result = ProjectOption.objects.get_value(self.project, "foo")
  16         assert result is None
  17 
  18         ProjectOption.objects.create(project=self.project, key="foo", value="bar")
  19         result = ProjectOption.objects.get_value(self.project, "foo")
  20         assert result == "bar"
  21 
  22     def test_unset_value(self):
  23         ProjectOption.objects.unset_value(self.project, "foo")
  24         ProjectOption.objects.create(project=self.project, key="foo", value="bar")
  25         ProjectOption.objects.unset_value(self.project, "foo")
  26         assert not ProjectOption.objects.filter(project=self.project, key="foo").exists()
  27 
  28     def test_get_value_bulk(self):
  29         result = ProjectOption.objects.get_value_bulk([self.project], "foo")
  30         assert result == {self.project: None}
  31 
  32         ProjectOption.objects.create(project=self.project, key="foo", value="bar")
  33         result = ProjectOption.objects.get_value_bulk([self.project], "foo")
  34         assert result == {self.project: "bar"}
