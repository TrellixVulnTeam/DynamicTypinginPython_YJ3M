  12 class MonitorTestCase(TestCase):
  13     def test_next_run_crontab_implicit(self):
  14         monitor = Monitor(
  15             last_checkin=datetime(2019, 1, 1, 1, 10, 20, tzinfo=timezone.utc),
  16             config={"schedule": "* * * * *"},
  17         )
  18         assert monitor.get_next_scheduled_checkin() == datetime(
  19             2019, 1, 1, 1, 11, tzinfo=timezone.utc
  20         )
