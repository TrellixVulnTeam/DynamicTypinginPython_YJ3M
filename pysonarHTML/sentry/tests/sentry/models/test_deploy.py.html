   8 class DeployNotifyTest(TestCase):
   9     def test_notify_if_ready_long_release(self):
  10         org = self.create_organization()
  11         project = self.create_project(organization=org)
  12         release = Release.objects.create(version="a" * 200, organization=org)
  13         release.add_project(project)
  14         env = Environment.objects.create(name="production", organization_id=org.id)
  15         deploy = Deploy.objects.create(
  16             release=release, organization_id=org.id, environment_id=env.id
  17         )
  18         Deploy.notify_if_ready(deploy.id)
  19 
  20         # make sure activity has been created
  21         record = Activity.objects.get(type=Activity.DEPLOY, project=project)
  22         assert release.version.startswith(record.ident)

  69     def test_head_commits_fetch_not_complete(self):
  70         # case where there are not commits, but there are head
  71         # commits, indicating we should wait to notify
  72         org = self.create_organization()
  73         project = self.create_project(organization=org)
  74         release = Release.objects.create(version="a" * 40, organization=org)
  75         release.add_project(project)
  76         ReleaseHeadCommit.objects.create(
  77             release=release,
  78             organization_id=org.id,
  79             repository_id=5,
  80             commit=Commit.objects.create(key="b" * 40, repository_id=5, organization_id=org.id),
  81         )
  82         env = Environment.objects.create(name="production", organization_id=org.id)
  83 
  84         deploy = Deploy.objects.create(
  85             release=release, organization_id=org.id, environment_id=env.id
  86         )
  87 
  88         Deploy.notify_if_ready(deploy.id)
  89 
  90         # make sure activity has been created
  91         assert not Activity.objects.filter(
  92             type=Activity.DEPLOY, project=project, ident=release.version
  93         ).exists()
  94         assert Deploy.objects.get(id=deploy.id).notified is False

  96     def test_no_commits_fetch_complete(self):

  98         # we've tried to fetch commits, but there
  99         # weren't any
 100         org = self.create_organization()
 101         project = self.create_project(organization=org)
 102         release = Release.objects.create(version="a" * 40, organization=org)

 105 
 106         deploy = Deploy.objects.create(
 107             release=release, organization_id=org.id, environment_id=env.id

 122         assert Deploy.objects.get(id=deploy.id).notified is True
