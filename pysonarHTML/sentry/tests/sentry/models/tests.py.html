  61     def test_send_recover_mail(self):
  62         request = HttpRequest()
  63         request.method = "GET"
  64         request.META["REMOTE_ADDR"] = "1.1.1.1"
  65 
  66         with self.options({"system.url-prefix": "http://testserver"}), self.tasks():
  67             self.password_hash.send_email(request)
  68 
  69         assert len(mail.outbox) == 1
  70         msg = mail.outbox[0]
  71         assert msg.to == [self.user.email]
  72         assert msg.subject == "[Sentry] Password Recovery"
  73         url = "http://testserver" + reverse(
  74             "sentry-account-recover-confirm",
  75             args=[self.password_hash.user_id, self.password_hash.hash],
  76         )
  77         assert url in msg.body

  90 class EventNodeStoreTest(TestCase):
  91     def test_event_node_id(self):
  92         # Create an event without specifying node_id. A node_id should be generated
  93         e1 = Event(project_id=1, event_id="abc", data={"foo": "bar"})
  94         assert e1.data.id is not None, "We should have generated a node_id for this event"
  95         e1_node_id = e1.data.id
  96         e1.data.save()
  97         e1_body = nodestore.get(e1_node_id)
  98         assert e1_body == {"foo": "bar"}, "The event body should be in nodestore"
  99 
 100         e1 = Event(project_id=1, event_id="abc")
 101 
 102         assert e1.data.data == {"foo": "bar"}, "The event body should be loaded from nodestore"
 103         assert e1.data.id == e1_node_id, "The event's node_id should be the same after load"
 104 
 105         # Event with no data should not be saved to nodestore
 106         e2 = Event(project_id=1, event_id="mno", data=None)
 107         e2_node_id = e2.data.id
 108         assert e2.data.data == {}  # NodeData returns {} by default
 109         eventstore.bind_nodes([e2], "data")
 110         assert e2.data.data == {}
 111         e2_body = nodestore.get(e2_node_id)
 112         assert e2_body is None

 114     def test_screams_bloody_murder_when_ref_fails(self):
 115         project1 = self.create_project()
 116         project2 = self.create_project()
 117         invalid_event = self.store_event(
 118             data={
 119                 "event_id": "a" * 32,
 120                 "timestamp": iso_format(before_now(minutes=1)),
 121                 "fingerprint": ["group-1"],
 122             },
 123             project_id=project1.id,
 124         )
 125         event = self.store_event(
 126             data={
 127                 "event_id": "b" * 32,
 128                 "timestamp": iso_format(before_now(minutes=1)),
 129                 "fingerprint": ["group-2"],
 130             },
 131             project_id=project2.id,
 132         )
 133         event.data.bind_ref(invalid_event)
 134         event.data.save()
 135 
 136         assert event.data.get_ref(event) != event.data.get_ref(invalid_event)
 137 
 138         # Unload node data to force reloading from nodestore
 139         event.data._node_data = None
 140 
 141         with pytest.raises(NodeIntegrityFailure):
 142             eventstore.bind_nodes([event])
