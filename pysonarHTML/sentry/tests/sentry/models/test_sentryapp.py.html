   8 class SentryAppTest(TestCase):
   9     def setUp(self):
  10         self.user = self.create_user()
  11         self.org = self.create_organization(owner=self.user)
  12         self.proxy = self.create_user()
  13         self.application = ApiApplication.objects.create(owner=self.proxy)
  14 
  15         self.sentry_app = SentryApp(
  16             application=self.application,
  17             name="NullDB",
  18             proxy_user=self.proxy,
  19             owner=self.org,
  20             scope_list=("project:read",),
  21             webhook_url="http://example.com",
  22         )
  23 
  24     def test_paranoid(self):

  28         assert self.sentry_app not in SentryApp.objects.all()
  29 
  30     def test_date_updated(self):

  32         date_updated = self.sentry_app.date_updated

  35 
  36     def test_related_names(self):

  39         assert self.sentry_app.proxy_user.sentry_app == self.sentry_app

  42     def test_is_unpublished(self):
