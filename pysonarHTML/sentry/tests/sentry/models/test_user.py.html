  22         UserEmail.objects.create_or_update(
  23             user=from_user, email=from_user.email, values={"is_verified": True}

  68 class GetUsersFromTeamsTest(TestCase):
  69     def test(self):
  70         user = self.create_user()
  71         org = self.create_organization(name="foo", owner=user)
  72         team = self.create_team(organization=org)
  73         org2 = self.create_organization(name="bar", owner=None)
  74         team2 = self.create_team(organization=org2)
  75         user2 = self.create_user("foo@example.com")
  76         self.create_member(user=user2, organization=org, role="admin", teams=[team])
  77 
  78         assert list(User.objects.get_from_teams(org, [team])) == [user2]
  79         user3 = self.create_user("bar@example.com")
  80         self.create_member(user=user3, organization=org, role="admin", teams=[team])
  81         assert set(list(User.objects.get_from_teams(org, [team]))) == set([user2, user3])
  82         assert list(User.objects.get_from_teams(org2, [team])) == []
  83         assert list(User.objects.get_from_teams(org2, [team2])) == []
  84         self.create_member(user=user, organization=org2, role="member", teams=[team2])
  85         assert list(User.objects.get_from_teams(org2, [team2])) == [user]
