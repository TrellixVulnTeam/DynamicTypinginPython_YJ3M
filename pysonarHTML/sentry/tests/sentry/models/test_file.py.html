  41 class FileTest(TestCase):
  42     def test_file_handling(self):
  43         fileobj = ContentFile("foo bar".encode("utf-8"))
  44         file1 = File.objects.create(name="baz.js", type="default", size=7)
  45         results = file1.putfile(fileobj, 3)
  46         assert len(results) == 3
  47         assert results[0].offset == 0
  48         assert results[1].offset == 3
  49         assert results[2].offset == 6
  50 
  51         fp = None
  52         with file1.getfile() as fp:
  53             assert fp.read().decode("utf-8") == "foo bar"
  54             fp.seek(2)
  55             assert fp.tell() == 2
  56             assert fp.read().decode("utf-8") == "o bar"
  57             fp.seek(0)
  58             assert fp.tell() == 0
  59             assert fp.read().decode("utf-8") == "foo bar"
  60             fp.seek(4)
  61             assert fp.tell() == 4
  62             assert fp.read().decode("utf-8") == "bar"
  63             fp.seek(1000)
  64             assert fp.tell() == 1000
  65 
  66             with self.assertRaises(IOError):
  67                 fp.seek(-1)
  68 
  69         with self.assertRaises(ValueError):
  70             fp.seek(0)
  71 
  72         with self.assertRaises(ValueError):
  73             fp.tell()
  74 
  75         with self.assertRaises(ValueError):
  76             fp.read()
