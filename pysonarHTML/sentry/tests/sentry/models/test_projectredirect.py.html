   5 from sentry.models import ProjectRedirect
   6 from sentry.testutils import TestCase

   9 class ProjectRedirectTest(TestCase):
  10     def test_record(self):
  11         org = self.create_organization()
  12         project = self.create_project(organization=org)
  13         ProjectRedirect.record(project, "old_slug")
  14 
  15         assert ProjectRedirect.objects.filter(redirect_slug="old_slug", project=project).exists()
  16 
  17         # Recording the same historic slug on a different project updates the
  18         # project pointer.
  19         project2 = self.create_project(organization=org)
  20         ProjectRedirect.record(project2, "old_slug")
  21 
  22         assert not ProjectRedirect.objects.filter(
  23             redirect_slug="old_slug", project=project
  24         ).exists()
  25 
  26         assert ProjectRedirect.objects.filter(redirect_slug="old_slug", project=project2).exists()
