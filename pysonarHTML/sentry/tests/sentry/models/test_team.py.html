  54 class TransferTest(TestCase):
  55     def test_simple(self):
  56         user = self.create_user()
  57         org = self.create_organization(name="foo", owner=user)
  58         org2 = self.create_organization(name="bar", owner=None)
  59         team = self.create_team(organization=org)
  60         project = self.create_project(teams=[team])
  61         user2 = self.create_user("foo@example.com")
  62         self.create_member(user=user2, organization=org, role="admin", teams=[team])
  63         self.create_member(user=user2, organization=org2, role="member", teams=[])
  64         team.transfer_to(org2)
  65 
  66         assert team.organization == org2
  67         team = Team.objects.get(id=team.id)
  68         assert team.organization == org2
  69 
  70         project = Project.objects.get(id=project.id)
  71         assert project.organization == org2
  72 
  73         # owner does not exist on new org, so should not be transferred
  74         assert not OrganizationMember.objects.filter(user=user, organization=org2).exists()
  75 
  76         # existing member should now have access
  77         member = OrganizationMember.objects.get(user=user2, organization=org2)
  78         assert list(member.teams.all()) == [team]
  79         # role should not automatically upgrade
  80         assert member.role == "member"
  81 
  82         # old member row should still exist
  83         assert OrganizationMember.objects.filter(user=user2, organization=org).exists()
  84 
  85         # no references to old org for this team should exist
  86         assert not OrganizationMemberTeam.objects.filter(
  87             organizationmember__organization=org, team=team
  88         ).exists()
