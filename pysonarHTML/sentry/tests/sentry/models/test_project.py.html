 206 class CopyProjectSettingsTest(TestCase):
 207     def setUp(self):
 208         super(CopyProjectSettingsTest, self).setUp()
 209         self.login_as(user=self.user)
 210 
 211         self.options_dict = {
 212             "sentry:resolve_age": 1,
 213             "sentry:scrub_data": False,
 214             "sentry:scrub_defaults": False,
 215         }
 216         self.other_project = self.create_project()
 217         for key, value in six.iteritems(self.options_dict):
 218             self.other_project.update_option(key=key, value=value)
 219 
 220         self.teams = [self.create_team(), self.create_team(), self.create_team()]
 221 
 222         for team in self.teams:
 223             ProjectTeam.objects.create(team=team, project=self.other_project)
 224 
 225         self.environments = [
 226             self.create_environment(project=self.other_project),
 227             self.create_environment(project=self.other_project),
 228         ]
 229 
 230         self.ownership = ProjectOwnership.objects.create(
 231             project=self.other_project, raw='{"hello":"hello"}', schema={"hello": "hello"}
 232         )
 233 
 234         Rule.objects.create(project=self.other_project, label="rule1")
 235         Rule.objects.create(project=self.other_project, label="rule2")
 236         Rule.objects.create(project=self.other_project, label="rule3")
 237         # there is a default rule added to project
 238         self.rules = Rule.objects.filter(project_id=self.other_project.id).order_by("label")
 239 
 240     def assert_other_project_settings_not_changed(self):
 241         # other_project should not have changed. This should check that.
 242         self.assert_settings_copied(self.other_project)
 243 
 244     def assert_settings_copied(self, project):
 245         for key, value in six.iteritems(self.options_dict):
 246             assert project.get_option(key) == value
 247 
 248         project_teams = ProjectTeam.objects.filter(project_id=project.id, team__in=self.teams)
 249         assert len(project_teams) == len(self.teams)
 250 
 251         project_env = EnvironmentProject.objects.filter(
 252             project_id=project.id, environment__in=self.environments
 253         )
 254         assert len(project_env) == len(self.environments)
 255 
 256         ownership = ProjectOwnership.objects.get(project_id=project.id)
 257         assert ownership.raw == self.ownership.raw
 258         assert ownership.schema == self.ownership.schema
 259 
 260         rules = Rule.objects.filter(project_id=project.id).order_by("label")
 261         for rule, other_rule in zip(rules, self.rules):
 262             assert rule.label == other_rule.label
 263 
 264     def assert_settings_not_copied(self, project, teams=()):
 265         for key in six.iterkeys(self.options_dict):
 266             assert project.get_option(key) is None
 267 
 268         project_teams = ProjectTeam.objects.filter(project_id=project.id, team__in=teams)
 269         assert len(project_teams) == len(teams)
 270 
 271         project_envs = EnvironmentProject.objects.filter(project_id=project.id)
 272         assert len(project_envs) == 0
 273 
 274         assert not ProjectOwnership.objects.filter(project_id=project.id).exists()
 275 
 276         # default rule
 277         rules = Rule.objects.filter(project_id=project.id)
 278         assert len(rules) == 1
 279         assert rules[0].label == "Send a notification for new issues"
