  42             project=self.group.project, group=self.group, type=Activity.ASSIGNED

 186     def test_assignee_sync_inbound_assign(self):
 187         group = self.group
 188         user_no_access = self.create_user()
 189         user_w_access = self.user
 190         integration = Integration.objects.create(provider="example", external_id="123456")
 191         integration.add_organization(group.organization, user_no_access)
 192 
 193         OrganizationIntegration.objects.filter(
 194             integration_id=integration.id, organization_id=group.organization.id
 195         ).update(
 196             config={
 197                 "sync_comments": True,
 198                 "sync_status_outbound": True,
 199                 "sync_status_inbound": True,
 200                 "sync_assignee_outbound": True,
 201                 "sync_assignee_inbound": True,
 202             }
 203         )
 204 
 205         external_issue = ExternalIssue.objects.create(
 206             organization_id=group.organization.id, integration_id=integration.id, key="APP-123"
 207         )
 208 
 209         GroupLink.objects.create(
 210             group_id=group.id,
 211             project_id=group.project_id,
 212             linked_type=GroupLink.LinkedType.issue,
 213             linked_id=external_issue.id,
 214             relationship=GroupLink.Relationship.references,
 215         )
 216 
 217         with self.feature("organizations:integrations-issue-sync"):
 218             # no permissions
 219             groups_updated = sync_group_assignee_inbound(
 220                 integration, user_no_access.email, "APP-123"
 221             )
 222 
 223             assert not groups_updated
 224 
 225             # w permissions
 226             groups_updated = sync_group_assignee_inbound(
 227                 integration, user_w_access.email, "APP-123"
 228             )
 229 
 230             assert groups_updated[0] == group
 231             assert GroupAssignee.objects.filter(
 232                 project=group.project, group=group, user=user_w_access, team__isnull=True
 233             ).exists()
