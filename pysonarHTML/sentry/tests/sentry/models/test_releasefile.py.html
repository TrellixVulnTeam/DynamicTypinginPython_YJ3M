  12 class ReleaseFileTestCase(TestCase):
  13     def test_normalize(self):
  14         n = ReleaseFile.normalize
  15 
  16         assert n("http://example.com") == ["http://example.com", "~"]
  17         assert n("http://example.com/foo.js") == ["http://example.com/foo.js", "~/foo.js"]
  18         assert n("http://example.com/foo.js?bar") == [
  19             "http://example.com/foo.js?bar",
  20             "http://example.com/foo.js",
  21             "~/foo.js?bar",
  22             "~/foo.js",
  23         ]
  24         assert n("/foo.js") == ["/foo.js", "~/foo.js"]
  25 
  26         assert n("http://example.com/foo.js?bar#baz") == [
  27             "http://example.com/foo.js?bar",
  28             "http://example.com/foo.js",
  29             "~/foo.js?bar",
  30             "~/foo.js",
  31         ]
  32 
  33         # This is the current behavior, but seems weird to me.
  34         # unclear if we actually experience this case in the real
  35         # world, but worth documenting the behavior
  36         assert n("foo.js") == ["foo.js", "~foo.js"]

  39 class ReleaseFileCacheTest(TestCase):
  40     def test_getfile_fs_cache(self):
  41         file_content = b"this is a test"
  42 
  43         file = self.create_file(name="dummy.txt")
  44         file.putfile(six.BytesIO(file_content))
  45         release_file = self.create_release_file(file=file)
  46 
  47         expected_path = os.path.join(
  48             options.get("releasefile.cache-path"),
  49             six.text_type(self.organization.id),
  50             six.text_type(file.id),
  51         )
  52 
  53         # Set the threshold to zero to force caching on the file system
  54         options.set("releasefile.cache-limit", 0)
  55         with ReleaseFile.cache.getfile(release_file) as f:
  56             assert f.read() == file_content
  57             assert f.name == expected_path
  58 
  59         # Check that the file was cached
  60         os.stat(expected_path)

  62     def test_getfile_streaming(self):
  63         file_content = b"this is a test"
  64 
  65         file = self.create_file(name="dummy.txt")
  66         file.putfile(six.BytesIO(file_content))
  67         release_file = self.create_release_file(file=file)
  68 
  69         expected_path = os.path.join(
  70             options.get("releasefile.cache-path"),
  71             six.text_type(self.organization.id),
  72             six.text_type(file.id),
  73         )
  74 
  75         # Set the threshold larger than the file size to force streaming
  76         options.set("releasefile.cache-limit", 1024)
  77         with ReleaseFile.cache.getfile(release_file) as f:
  78             assert f.read() == file_content
  79 
  80         # Check that the file was not cached
  81         try:
  82             os.stat(expected_path)
  83         except OSError as e:
  84             assert e.errno == errno.ENOENT
  85         else:
  86             assert False, "file should not exist"
