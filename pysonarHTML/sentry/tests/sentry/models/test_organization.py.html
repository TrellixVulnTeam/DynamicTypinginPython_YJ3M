 204     def setUp(self):
 205         self.owner = self.create_user("foo@example.com")
 206         TotpInterface().enroll(self.owner)
 207         self.org = self.create_organization(owner=self.owner)

 210     def _create_user(self, has_email=True):
 211         if not has_email:
 212             return self.create_user("")
 213         return self.create_user()
 214 
 215     def _create_user_and_member(self, has_2fa=False, has_user_email=True, has_member_email=False):
 216         user = self._create_user(has_email=has_user_email)
 217         if has_2fa:
 218             TotpInterface().enroll(user)
 219         if has_member_email:
 220             email = uuid4().hex
 221             member = self.create_member(organization=self.org, user=user, email=email)
 222         else:
 223             member = self.create_member(organization=self.org, user=user)
 224         return user, member

 226     def is_organization_member(self, user_id, member_id):
 227         member = OrganizationMember.objects.get(id=member_id)
 228         user = User.objects.get(id=user_id)
 229         assert not member.is_pending
 230         assert not member.email
 231         assert member.user == user

 233     def is_pending_organization_member(self, user_id, member_id, was_booted=True):
 234         member = OrganizationMember.objects.get(id=member_id)
 235         assert User.objects.filter(id=user_id).exists()
 236         assert member.is_pending
 237         assert member.email
 238         if was_booted:
 239             assert member.token
 240             assert member.token_expires_at
 241         else:
 242             assert member.token is None
 243             assert member.token_expires_at is None

 245     @mock.patch("sentry.utils.email.logger")
 246     def test_handle_2fa_required__compliant_and_non_compliant_members(self, email_log):
 247         compliant_user, compliant_member = self._create_user_and_member(has_2fa=True)
 248         non_compliant_user, non_compliant_member = self._create_user_and_member()
 249 
 250         with self.options({"system.url-prefix": "http://example.com"}), self.tasks():
 251             self.org.handle_2fa_required(self.request)
 252 
 253         self.is_organization_member(compliant_user.id, compliant_member.id)
 254         self.is_pending_organization_member(non_compliant_user.id, non_compliant_member.id)
 255 
 256         assert len(mail.outbox) == 1
 257         assert mail.outbox[0].to == [non_compliant_user.email]
 258         assert email_log.info.call_count == 2  # mail.queued, mail.sent
 259 
 260         audit_logs = AuditLogEntry.objects.filter(
 261             event=AuditLogEntryEvent.MEMBER_PENDING, organization=self.org, actor=self.owner
 262         )
 263         assert audit_logs.count() == 1
 264         assert audit_logs[0].data["email"] == non_compliant_user.email
 265         assert audit_logs[0].target_user_id == non_compliant_user.id
