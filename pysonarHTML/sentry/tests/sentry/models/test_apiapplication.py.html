   7 class ApiApplicationTest(TestCase):
   8     def test_is_valid_redirect_uri(self):
   9         app = ApiApplication.objects.create(
  10             owner=self.user, redirect_uris="http://example.com\nhttp://sub.example.com/path"
  11         )
  12 
  13         assert app.is_valid_redirect_uri("http://example.com/")

  15         assert app.is_valid_redirect_uri("http://example.com/biz/baz")
  16         assert not app.is_valid_redirect_uri("https://example.com/")

  18         assert not app.is_valid_redirect_uri("http://example.com.foo.com")
  19 
  20         assert app.is_valid_redirect_uri("http://sub.example.com/path")
  21         assert app.is_valid_redirect_uri("http://sub.example.com/path/bar")
  22         assert not app.is_valid_redirect_uri("http://sub.example.com")
  23         assert not app.is_valid_redirect_uri("https://sub.example.com")
