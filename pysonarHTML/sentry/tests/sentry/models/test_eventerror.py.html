   1 from __future__ import absolute_import

   3 import pytest

   5 from sentry.models import EventError

   8 @pytest.mark.parametrize(
   9     "error,type,message,data",
  10     (
  11         ({"type": "unknown_error"}, "unknown_error", "Unknown error", {}),
  12         ({"type": "unknown_error", "foo": "bar"}, "unknown_error", "Unknown error", {"foo": "bar"}),
  13         (
  14             {"type": "invalid_data", "name": "foo"},
  15             "invalid_data",
  16             "Discarded invalid value",
  17             {"name": "foo"},
  18         ),
  19         ({"type": "invalid_data"}, "invalid_data", "Discarded invalid value", {}),
  20         ({"type": "INVALID_ERROR_TYPE"}, "INVALID_ERROR_TYPE", "Unknown error", {}),
  21     ),
  22 )
  23 def test_event_error(error, type, message, data):
  24     assert EventError.get_message(error) == message
  25     assert EventError(error).type == type
  26     assert EventError(error).message == message
  27     assert EventError(error).data == data
  28 
  29 
  30 def test_api_context():
  31     error = {"type": "unknown_error", "foo": "bar"}
  32     assert EventError(error).get_api_context() == {
  33         "type": "unknown_error",
  34         "message": "Unknown error",
  35         "data": {"foo": "bar"},
  36     }
