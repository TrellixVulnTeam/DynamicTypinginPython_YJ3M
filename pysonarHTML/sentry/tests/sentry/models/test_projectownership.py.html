  27     def test_get_owners_basic(self):
  28         rule_a = Rule(Matcher("path", "*.py"), [Owner("team", self.team.slug)])
  29         rule_b = Rule(Matcher("path", "src/*"), [Owner("user", self.user.email)])
  30 
  31         ProjectOwnership.objects.create(
  32             project_id=self.project.id, schema=dump_schema([rule_a, rule_b]), fallthrough=True
  33         )
  34 
  35         # No data matches
  36         assert ProjectOwnership.get_owners(self.project.id, {}) == (ProjectOwnership.Everyone, None)
  37 
  38         # Match only rule_a
  39         self.assert_ownership_equals(
  40             ProjectOwnership.get_owners(
  41                 self.project.id, {"stacktrace": {"frames": [{"filename": "foo.py"}]}}
  42             ),
  43             ([Actor(self.team.id, Team)], [rule_a]),
  44         )
  45 
  46         # Match only rule_b
  47         self.assert_ownership_equals(
  48             ProjectOwnership.get_owners(
  49                 self.project.id, {"stacktrace": {"frames": [{"filename": "src/thing.txt"}]}}
  50             ),
  51             ([Actor(self.user.id, User)], [rule_b]),
  52         )
  53 
  54         # Matches both rule_a and rule_b
  55         self.assert_ownership_equals(
  56             ProjectOwnership.get_owners(
  57                 self.project.id, {"stacktrace": {"frames": [{"filename": "src/foo.py"}]}}
  58             ),
  59             ([Actor(self.team.id, Team), Actor(self.user.id, User)], [rule_a, rule_b]),
  60         )
  61 
  62         assert ProjectOwnership.get_owners(
  63             self.project.id, {"stacktrace": {"frames": [{"filename": "xxxx"}]}}
  64         ) == (ProjectOwnership.Everyone, None)
  65 
  66         # When fallthrough = False, we don't implicitly assign to Everyone

  68         owner.fallthrough = False
  69         owner.save()
  70 
  71         assert ProjectOwnership.get_owners(
  72             self.project.id, {"stacktrace": {"frames": [{"filename": "xxxx"}]}}
  73         ) == ([], None)

  79             ([Actor(self.team.id, Team), Actor(self.user.id, User)], [rule_a, rule_b]),

  83 class ResolveActorsTestCase(TestCase):
  84     def test_no_actors(self):
  85         assert resolve_actors([], self.project.id) == {}
  86 
  87     def test_basic(self):
  88         owners = [Owner("user", self.user.email), Owner("team", self.team.slug)]
  89         assert resolve_actors(owners, self.project.id) == {
  90             owners[0]: Actor(self.user.id, User),
  91             owners[1]: Actor(self.team.id, Team),
  92         }

  94     def test_teams(self):
  95         # Normal team
  96         owner1 = Owner("team", self.team.slug)
  97         actor1 = Actor(self.team.id, Team)
  98 
  99         # Team that doesn't exist
 100         owner2 = Owner("team", "nope")
 101         actor2 = None
 102 
 103         # A team that's not ours
 104         otherteam = Team.objects.exclude(projectteam__project_id=self.project.id)[0]
 105         owner3 = Owner("team", otherteam.slug)
 106         actor3 = None
 107 
 108         assert resolve_actors([owner1, owner2, owner3], self.project.id) == {
 109             owner1: actor1,
 110             owner2: actor2,
 111             owner3: actor3,
 112         }

 114     def test_users(self):
 115         # Normal user
 116         owner1 = Owner("user", self.user.email)
 117         actor1 = Actor(self.user.id, User)
 118 
 119         # An extra secondary email
 120         email1 = self.create_useremail(self.user, None, is_verified=True).email
 121         owner2 = Owner("user", email1)
 122         actor2 = actor1  # They map to the same user since it's just a secondary email
 123 
 124         # Another secondary email, that isn't verified
 125         email2 = self.create_useremail(self.user, None, is_verified=False).email
 126         owner3 = Owner("user", email2)
 127         # Intentionally allow unverified emails
 128         # actor3 = None
 129         actor3 = actor1
 130 
 131         # An entirely unknown user
 132         owner4 = Owner("user", "nope")
 133         actor4 = None
 134 
 135         # A user that doesn't belong with us
 136         otheruser = self.create_user()
 137         owner5 = Owner("user", otheruser.email)
 138         actor5 = None
 139 
 140         # Case-insensitive for user
 141         owner6 = Owner("user", self.user.email.upper())
 142         actor6 = actor1
 143 
 144         assert resolve_actors(
 145             [owner1, owner2, owner3, owner4, owner5, owner6], self.project.id
 146         ) == {
 147             owner1: actor1,
 148             owner2: actor2,
 149             owner3: actor3,
 150             owner4: actor4,
 151             owner5: actor5,
 152             owner6: actor6,
 153         }
