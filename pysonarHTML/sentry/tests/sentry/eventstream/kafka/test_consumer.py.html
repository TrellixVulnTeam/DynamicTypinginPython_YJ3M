 484 def consume_until_constraints_met(consumer, constraints, iterations, timeout=1):
 485     constraints = set(constraints)
 486 
 487     for i in xrange(iterations):
 488         message = consumer.poll(timeout)
 489         for constraint in list(constraints):
 490             if constraint(message):
 491                 constraints.remove(constraint)
 492 
 493         if not constraints:
 494             break
 495 
 496     if constraints:
 497         raise AssertionError(
 498             "Completed {} iterations with {} unmet constraints: {!r}".format(
 499                 iterations, len(constraints), constraints
 500             )
 501         )

 504 def collect_messages_received(count):
 505     messages = []
 506 
 507     def messages_received_constraint(message):
 508         if message is not None:
 509             messages.append(message)
 510             if len(messages) == count:
 511                 return True
 512 
 513     return messages_received_constraint

 520 def test_consumer_rebalance_from_uncommitted_offset(requires_kafka):
 521     consumer_group = "consumer-{}".format(uuid.uuid1().hex)
 522     synchronize_commit_group = "consumer-{}".format(uuid.uuid1().hex)
 523 
 524     messages_delivered = defaultdict(list)
 525 
 526     def record_message_delivered(error, message):
 527         assert error is None
 528         messages_delivered[message.topic()].append(message)
 529 
 530     producer = Producer(
 531         {
 532             "bootstrap.servers": os.environ["SENTRY_KAFKA_HOSTS"],
 533             "on_delivery": record_message_delivered,
 534         }
 535     )
 536 
 537     with create_topic(partitions=2) as topic, create_topic() as commit_log_topic:
 538 
 539         # Produce some messages into the topic.
 540         for i in range(4):
 541             producer.produce(topic, "{}".format(i).encode("utf8"), partition=i % 2)
 542 
 543         assert producer.flush(5) == 0, "producer did not successfully flush queue"
 544 
 545         for (topic, partition), offset in {
 546             (message.topic(), message.partition()): message.offset()
 547             for message in messages_delivered[topic]
 548         }.items():
 549             producer.produce(
 550                 commit_log_topic,
 551                 key="{}:{}:{}".format(topic, partition, synchronize_commit_group).encode("utf8"),
 552                 value="{}".format(offset + 1).encode("utf8"),
 553             )
 554 
 555         assert producer.flush(5) == 0, "producer did not successfully flush queue"
 556 
 557         consumer_a = SynchronizedConsumer(
 558             bootstrap_servers=os.environ["SENTRY_KAFKA_HOSTS"],
 559             consumer_group=consumer_group,
 560             commit_log_topic=commit_log_topic,
 561             synchronize_commit_group=synchronize_commit_group,
 562             initial_offset_reset="earliest",
 563         )
 564 
 565         assignments_received = defaultdict(list)
 566 
 567         def on_assign(consumer, assignment):
 568             assignments_received[consumer].append(assignment)
 569 
 570         consumer_a.subscribe([topic], on_assign=on_assign)
 571 
 572         consume_until_constraints_met(
 573             consumer_a,
 574             [lambda message: assignments_received[consumer_a], collect_messages_received(4)],
 575             10,
 576         )
 577 
 578         assert (
 579             len(assignments_received[consumer_a]) == 1
 580         ), "expected to receive partition assignment"
 581         assert set((i.topic, i.partition) for i in assignments_received[consumer_a][0]) == set(
 582             [(topic, 0), (topic, 1)]
 583         )
 584         assignments_received[consumer_a].pop()
 585 
 586         message = consumer_a.poll(1)
 587         assert (
 588             message is None or message.error() is KafkaError._PARTITION_EOF
 589         ), "there should be no more messages to receive"
 590 
 591         consumer_b = SynchronizedConsumer(
 592             bootstrap_servers=os.environ["SENTRY_KAFKA_HOSTS"],
 593             consumer_group=consumer_group,
 594             commit_log_topic=commit_log_topic,
 595             synchronize_commit_group=synchronize_commit_group,
 596             initial_offset_reset="earliest",
 597         )
 598 
 599         consumer_b.subscribe([topic], on_assign=on_assign)
 600 
 601         consume_until_constraints_met(
 602             consumer_a, [lambda message: assignments_received[consumer_a]], 10
 603         )
 604 
 605         consume_until_constraints_met(
 606             consumer_b,
 607             [lambda message: assignments_received[consumer_b], collect_messages_received(2)],
 608             10,
 609         )
 610 
 611         for consumer in [consumer_a, consumer_b]:
 612             assert len(assignments_received[consumer][0]) == 1
 613 
 614         message = consumer_a.poll(1)
 615         assert (
 616             message is None or message.error() is KafkaError._PARTITION_EOF
 617         ), "there should be no more messages to receive"
 618 
 619         message = consumer_b.poll(1)
 620         assert (
 621             message is None or message.error() is KafkaError._PARTITION_EOF
 622         ), "there should be no more messages to receive"
