  98 class SetCommitsTest(TestCase):
  99     def test_minimal(self):
 100         project = self.create_project()
 101         version = "bbee5b51f84611e4b14834363b8514c2"
 102         data_list = [
 103             {
 104                 "id": "c7155651831549cf8a5e47889fce17eb",
 105                 "message": "foo",
 106                 "author_email": "jane@example.com",
 107             },
 108             {"id": "bbee5b51f84611e4b14834363b8514c2", "message": "bar", "author_name": "Joe^^"},
 109         ]
 110 
 111         hook = ReleaseHook(project)
 112         hook.set_commits(version, data_list)
 113 
 114         release = Release.objects.get(projects=project, version=version)
 115         commit_list = list(
 116             Commit.objects.filter(releasecommit__release=release)
 117             .select_related("author")
 118             .order_by("releasecommit__order")
 119         )
 120 
 121         assert len(commit_list) == 2
 122         assert commit_list[0].key == "c7155651831549cf8a5e47889fce17eb"
 123         assert commit_list[0].message == "foo"
 124         assert commit_list[0].author.name is None
 125         assert commit_list[0].author.email == "jane@example.com"
 126         assert commit_list[1].key == "bbee5b51f84611e4b14834363b8514c2"
 127         assert commit_list[1].message == "bar"
 128         assert commit_list[1].author.name == "Joe^^"
 129         assert commit_list[1].author.email == "joe@localhost"
