  14 class GetAuthForUserTest(TestCase):
  15     def _get_mock_user(self):
  16         user = mock.Mock(spec=User())
  17         user.id = 1
  18         user.is_authenticated.return_value = False
  19         return user
  20 
  21     def test_requires_auth_provider(self):
  22         user = self._get_mock_user()
  23         p = IssueTrackingPlugin()
  24         self.assertRaises(AssertionError, p.get_auth_for_user, user)
  25 
  26     def test_returns_none_on_missing_identity(self):
  27         user = self._get_mock_user()
  28         p = IssueTrackingPlugin()
  29         p.auth_provider = "test"
  30         self.assertEquals(p.get_auth_for_user(user), None)
  31 
  32     def test_returns_identity(self):
  33         user = User.objects.create(username="test", email="test@example.com")
  34         auth = UserSocialAuth.objects.create(provider="test", user=user)
  35         p = IssueTrackingPlugin()
  36         p.auth_provider = "test"
  37         self.assertEquals(p.get_auth_for_user(user), auth)
