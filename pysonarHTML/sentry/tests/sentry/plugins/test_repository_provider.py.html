  10 class RepositoryProviderTest(TestCase):
  11     def test_needs_auth_for_user(self):
  12         user = self.create_user()
  13         provider = DummyRepositoryProvider(id="dummy")
  14 
  15         # if no org is provided, user needs auth
  16         assert provider.needs_auth(user) is True
  17 
  18         UserSocialAuth.objects.create(provider="dummy", user=user)
  19

  22     def test_needs_auth_for_organization(self):
  23         user = self.create_user()
  24         provider = DummyRepositoryProvider(id="dummy")
  25 
  26         org = self.create_organization()
  27         integration = Integration.objects.create(provider="dummy", external_id="123456")
  28         integration.add_organization(org, user)
  29 
  30         assert provider.needs_auth(user, organization=org) is False
  31 
  32     def test_get_auth_for_user(self):
