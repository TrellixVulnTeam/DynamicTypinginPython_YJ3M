   3 from sentry.utils.compat.mock import patch

   5 from sentry.constants import (

  12 mock_integration_ids = {
  13     "java": {},
  14     "java-log4j": {},
  15     "java-log4j2": {},
  16     "java-android": {},

  21 def test_marketing_slug_to_integration_id():
  22     with patch.dict(INTEGRATION_ID_TO_PLATFORM_DATA, mock_integration_ids):
  23         assert get_integration_id_for_marketing_slug("java") == "java"
  24         # kotlin uses the java library, too
  25         assert get_integration_id_for_marketing_slug("kotlin") == "java"
  26         assert get_integration_id_for_marketing_slug("android") == "java-android"
  27         assert get_integration_id_for_marketing_slug("foobar") is None
  28

  30 def test_integration_id_for_event():
  31     with patch.dict(INTEGRATION_ID_TO_PLATFORM_DATA, mock_integration_ids):
  32         assert get_integration_id_for_event("java", "sentry-java", None) == "java"
  33         assert get_integration_id_for_event("java", "raven-java", None) == "java"
  34         assert get_integration_id_for_event("java", "raven-java:log4j", None) == "java-log4j"
  35         assert get_integration_id_for_event("java", "sentry-java", ["android"]) == "java-android"
  36         assert (
  37             get_integration_id_for_event("java", "sentry-java", ["foobar", "log4j2"])
  38             == "java-log4j2"
  39         )
  40         assert get_integration_id_for_event("foobar", "sentry-java", None) == "java"
  41         assert get_integration_id_for_event("java", "foobar", None) == "java"
  42         assert get_integration_id_for_event("foobar", "foobar", None) is None
