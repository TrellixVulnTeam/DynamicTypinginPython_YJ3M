  62     def test_build_identity(self):
  63         self.provider.config = dummy_provider_config
  64         attrs = {
  65             "id": ["123"],
  66             "email": ["valid@example.com"],
  67             "first": ["Morty"],
  68             "last": ["Smith"],
  69         }
  70 
  71         state = {"auth_attributes": attrs}
  72         identity = self.provider.build_identity(state)
  73 
  74         assert identity["id"] == "123"
  75         assert identity["email"] == "valid@example.com"
  76         assert identity["name"] == "Morty Smith"
