  59     def test_mixed_access(self):
  60         user = self.create_user()
  61         organization = self.create_organization(flags=0)  # disable default allow_joinleave
  62         team = self.create_team(organization=organization)
  63         team_no_access = self.create_team(organization=organization)
  64         project = self.create_project(organization=organization, teams=[team])
  65         project_no_access = self.create_project(organization=organization, teams=[team_no_access])
  66         self.create_member(organization=organization, user=user, teams=[team])
  67         result = access.from_user(user, organization)
  68         assert result.has_project_access(project)

  70         assert not result.has_projects_access([project, project_no_access])
