  17         request = RequestFactory().post("/auth/sso/")
  18         request.user = AnonymousUser()
  19 
  20         auth_provider = AuthProvider.objects.create(
  21             organization=self.organization, provider=provider

  80     def test_associate_pending_invite(self):
  81         provider = AuthProvider.objects.create(organization=self.organization, provider="dummy")
  82         identity = {"id": "1234", "email": "test@example.com", "name": "Morty"}
  83 
  84         # The org member invite should have a non matching email, but the
  85         # member id and token will match from the cookie, allowing association
  86         member = OrganizationMember.objects.create(
  87             organization=self.organization, email="different.email@example.com", token="abc"
  88         )
  89 
  90         request = RequestFactory().post("/auth/sso/")
  91         request.user = AnonymousUser()
  92         request.COOKIES["pending-invite"] = urlencode(
  93             {"memberId": member.id, "token": member.token, "url": ""}
  94         )
  95 
  96         auth_identity = handle_new_user(provider, self.organization, request, identity)
  97 
  98         assigned_member = OrganizationMember.objects.get(
  99             organization=self.organization, user=auth_identity.user
 100         )
 101 
 102         assert assigned_member.id == member.id
