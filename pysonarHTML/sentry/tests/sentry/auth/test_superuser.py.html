  37     def build_request(
  38         self,
  39         cookie_token=UNSET,
  40         session_token=UNSET,
  41         expires=UNSET,
  42         idle_expires=UNSET,
  43         uid=UNSET,
  44         session_data=True,
  45         user=None,
  46     ):
  47         if user is None:
  48             user = self.create_user("foo@example.com", is_superuser=True)
  49         current_datetime = self.current_datetime
  50         request = self.make_request(user=user)
  51         if cookie_token is not None:
  52             request.COOKIES[COOKIE_NAME] = signing.get_cookie_signer(
  53                 salt=COOKIE_NAME + COOKIE_SALT
  54             ).sign(self.default_token if cookie_token is UNSET else cookie_token)
  55         if session_data:
  56             request.session[SESSION_KEY] = {
  57                 "exp": (
  58                     current_datetime + timedelta(hours=6) if expires is UNSET else expires
  59                 ).strftime("%s"),
  60                 "idl": (
  61                     current_datetime + timedelta(minutes=15)
  62                     if idle_expires is UNSET
  63                     else idle_expires
  64                 ).strftime("%s"),
  65                 "tok": self.default_token if session_token is UNSET else session_token,
  66                 "uid": six.text_type(user.id) if uid is UNSET else uid,
  67             }
  68         return request

  70     def test_ips(self):
  71         user = User(is_superuser=True)
  72         request = self.make_request(user=user)
  73         request.META["REMOTE_ADDR"] = "10.0.0.1"
  74 
  75         # no ips = any host
  76         superuser = Superuser(request, allowed_ips=())
  77         superuser.set_logged_in(request.user)
  78         assert superuser.is_active is True
  79 
  80         superuser = Superuser(request, allowed_ips=("127.0.0.1",))
  81         superuser.set_logged_in(request.user)
  82         assert superuser.is_active is False
  83 
  84         superuser = Superuser(request, allowed_ips=("10.0.0.1",))
  85         superuser.set_logged_in(request.user)
  86         assert superuser.is_active is True

 146     def test_login_saves_session(self):
 147         user = self.create_user("foo@example.com", is_superuser=True)
 148         request = self.make_request()
 149         superuser = Superuser(request, allowed_ips=(), current_datetime=self.current_datetime)
 150         superuser.set_logged_in(user, current_datetime=self.current_datetime)
 151 
 152         # request.user wasn't set
 153         assert not superuser.is_active
 154 
 155         request.user = user
 156         assert superuser.is_active
 157 
 158         data = request.session.get(SESSION_KEY)
 159         assert data
 160         assert data["exp"] == (self.current_datetime + MAX_AGE).strftime("%s")
 161         assert data["idl"] == (self.current_datetime + IDLE_MAX_AGE).strftime("%s")
 162         assert len(data["tok"]) == 12
 163         assert data["uid"] == six.text_type(user.id)

 173     def test_middleware_as_superuser(self):
 174         request = self.build_request()
 175 
 176         delattr(request, "superuser")
 177         delattr(request, "is_superuser")
 178 
 179         middleware = SuperuserMiddleware()
 180         middleware.process_request(request)
 181         assert request.superuser.is_active
 182         assert request.is_superuser()

 184         response = Mock()

 197     def test_changed_user(self):
