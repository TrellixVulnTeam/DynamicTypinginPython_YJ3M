  14 def floor_to_hour_epoch(value):
  15     value = value.replace(minute=0, second=0, microsecond=0)
  16     return int(to_timestamp(value))
  17 
  18 
  19 def floor_to_10s_epoch(value):
  20     seconds = value.second
  21     floored_second = 10 * (seconds // 10)
  22 
  23     value = value.replace(second=floored_second, microsecond=0)
  24     return int(to_timestamp(value))

  81 
  82             for time, count in response[self.organization.id]:

 217     def test_outcomes_have_a_10s_setting(self):
 218         exceptions = [
 219             TSDBModel.project_total_forwarded  # this is not outcomes and will be moved separately
 220         ]
 221 
 222         def is_an_outcome(model):
 223             if model in exceptions:
 224                 return False
 225 
 226             # 100 - 200: project outcomes
 227             # 200 - 300: organization outcomes
 228             # 500 - 600: key outcomes
 229             # 600 - 700: filtered project based outcomes
 230             return (
 231                 (100 <= model.value < 200)
 232                 or (200 <= model.value < 300)
 233                 or (500 <= model.value < 600)
 234                 or (600 <= model.value < 700)
 235             )
 236 
 237         models = [x for x in list(TSDBModel) if is_an_outcome(x)]
 238         for model in models:
 239             assert model in SnubaTSDB.lower_rollup_query_settings
