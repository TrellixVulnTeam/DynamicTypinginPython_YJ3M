   8 def get_callargs(model):
   9     """
  10     Represents for all possible ways that a model could be passed to ``selector_func`` through the callargs
  11     """
  12     return {
  13         "model": model,
  14         "models": [model],
  15         "items": [(model, "key", ["values"])],
  16         "requests": [(model, "data")],
  17     }

  20 def test_redissnuba_connects_to_correct_backend():
  21     should_resolve_to_redis = set(list(TSDBModel)) - set(
  22         SnubaTSDB.non_outcomes_query_settings.keys()
  23     )
  24     should_resolve_to_snuba = SnubaTSDB.non_outcomes_query_settings.keys()
  25 
  26     methods = set(method_specifications.keys()) - set(["flush"])
  27 
  28     for method in methods:
  29         for model in should_resolve_to_redis:
  30             assert "redis" == selector_func(method, get_callargs(model))
  31 
  32         for model in should_resolve_to_snuba:
  33             read_or_write, _ = method_specifications.get(method)
  34 
  35             if read_or_write == READ:
  36                 assert "snuba" == selector_func(method, get_callargs(model))
  37             else:
  38                 assert "dummy" == selector_func(method, get_callargs(model))
