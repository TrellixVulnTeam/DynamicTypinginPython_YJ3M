  79     @mock.patch("django.utils.timezone.now")
  80     def test_get_optimal_rollup_series_offset_intervals(self, now):
  81         # This test is a funny one (notice it doesn't return a range that
  82         # includes the start position.) This occurs because the algorithm for
  83         # determining the series to be returned will attempt to return the same
  84         # duration of time as represented by the start and end timestamps, but
  85         # doesn't necessarily return data *from that specific interval* (the
  86         # end timestamp is always included.)
  87 
  88         now.return_value = datetime(2016, 8, 1, 0, 0, 15, tzinfo=pytz.utc)
  89         start = now() - timedelta(seconds=19)
  90         assert self.tsdb.get_optimal_rollup_series(start, rollup=10) == (
  91             10,
  92             [
  93                 to_timestamp(datetime(2016, 8, 1, 0, 0, 0, tzinfo=pytz.utc)),
  94                 to_timestamp(datetime(2016, 8, 1, 0, 0, 10, tzinfo=pytz.utc)),
  95             ],
  96         )
  97 
  98         now.return_value = datetime(2016, 8, 1, 0, 0, 30, tzinfo=pytz.utc)
  99         start = now() - timedelta(seconds=ONE_MINUTE - 1)
 100         assert self.tsdb.get_optimal_rollup_series(start, rollup=ONE_MINUTE) == (
 101             ONE_MINUTE,
 102             [to_timestamp(datetime(2016, 8, 1, 0, 0, 0, tzinfo=pytz.utc))],
 103         )
 104 
 105         now.return_value = datetime(2016, 8, 1, 12, tzinfo=pytz.utc)
 106         start = now() - timedelta(seconds=ONE_DAY - 1)
 107         assert self.tsdb.get_optimal_rollup_series(start, rollup=ONE_DAY) == (
 108             ONE_DAY,
 109             [to_timestamp(datetime(2016, 8, 1, 0, tzinfo=pytz.utc))],
 110         )

 112     @mock.patch("django.utils.timezone.now")
 113     def test_make_series_aligned_intervals(self, now):
 114         now.return_value = datetime(2016, 8, 1, tzinfo=pytz.utc)
 115 
 116         start = now() - timedelta(seconds=30)
 117         assert self.tsdb.make_series(0, start) == [
 118             (to_timestamp(start + timedelta(seconds=10) * i), 0) for i in xrange(4)
 119         ]
 120 
 121         start = now() - timedelta(minutes=30)
 122         assert self.tsdb.make_series(lambda timestamp: 1, start) == [
 123             (to_timestamp(start + timedelta(minutes=1) * i), 1) for i in xrange(31)
 124         ]
 125 
 126         counter = itertools.count()
 127         start = now() - timedelta(hours=5)
 128         assert self.tsdb.make_series(lambda timestamp: next(counter), start) == [
 129             (to_timestamp(start + timedelta(hours=1) * i), i) for i in xrange(6)
 130         ]
 131 
 132         start = now() - timedelta(days=7)
 133         assert self.tsdb.make_series(0, start) == [
 134             (to_timestamp(start + timedelta(hours=24) * i), 0) for i in xrange(8)
 135         ]
