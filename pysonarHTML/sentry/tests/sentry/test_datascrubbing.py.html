  11 def merge_pii_configs(prefixes_and_configs):
  12     from sentry.datascrubbing import merge_pii_configs as f
  13 
  14     prefixes_and_configs_bak = copy.deepcopy(prefixes_and_configs)
  15     rv = f(prefixes_and_configs)
  16 
  17     # No accidental mutation
  18     assert prefixes_and_configs == prefixes_and_configs_bak
  19     return rv

  85 def test_merge_pii_configs_simple():
  86     assert merge_pii_configs([("p:", {}), ("o:", {})]) == {}
  87 
  88     assert merge_pii_configs(
  89         [("p:", {"applications": {"$string": ["@ip:remove"]}}), ("o:", {})]
  90     ) == {"applications": {"$string": ["@ip:remove"]}}

  93 def test_merge_pii_configs_rule_references():
  94     my_rules = {
  95         "remove_ips_alias": {
  96             "type": "alias",
  97             "rule": "@ip",
  98             "hide_rule": False,
  99             "redaction": {"method": "remove"},
 100         },
 101         "remove_ips_and_macs": {
 102             "type": "multiple",
 103             "rules": ["remove_ips_alias", "@mac"],
 104             "hide_rule": False,
 105             "redaction": {"method": "remove"},
 106         },
 107     }
 108 
 109     assert merge_pii_configs(
 110         [

 122             },

 128             },
 129             "p:remove_ips_and_macs": {
 130                 "hide_rule": False,
