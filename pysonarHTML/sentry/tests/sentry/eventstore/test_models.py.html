  15     def test_pickling_compat(self):
  16         event = self.store_event(
  17             data={
  18                 "message": "Hello World!",
  19                 "tags": {"logger": "foobar", "site": "foo", "server_name": "bar"},
  20             },
  21             project_id=self.project.id,
  22         )
  23 
  24         # Ensure we load and memoize the interfaces as well.
  25         assert len(event.interfaces) > 0
  26 
  27         # When we pickle an event we need to make sure our canonical code
  28         # does not appear here or it breaks old workers.
  29         data = pickle.dumps(event, protocol=2)
  30         assert "canonical" not in data
  31 
  32         # For testing we remove the backwards compat support in the
  33         # `NodeData` as well.
  34         nodedata_getstate = NodeData.__getstate__
  35         del NodeData.__getstate__
  36 
  37         # Old worker loading
  38         try:
  39             event2 = pickle.loads(data)
  40             assert event2.data == event.data
  41         finally:
  42             NodeData.__getstate__ = nodedata_getstate
  43 
  44         # New worker loading
  45         event2 = pickle.loads(data)
  46         assert event2.data == event.data

 184     def test_snuba_data(self):
 185         self.store_event(
 186             data={

 188                 "message": "Hello World!",
 189                 "tags": {"logger": "foobar", "site": "foo", "server_name": "bar"},
 190                 "user": {"id": "test", "email": "test@test.com"},
 191                 "timestamp": iso_format(before_now(seconds=1)),
 192             },
 193             project_id=self.project.id,
 194         )
 195 
 196         event_from_nodestore = Event(project_id=self.project.id, event_id="a" * 32)
 197 
 198         event_from_snuba = Event(
 199             project_id=self.project.id,
 200             event_id="a" * 32,

 203                     "event_id",
 204                     "project_id",
 205                     "group_id",
 206                     "timestamp",
 207                     "culprit",
 208                     "location",
 209                     "message",
 210                     "title",
 211                     "type",
 212                     "transaction",
 213                     "tags.key",
 214                     "tags.value",
 215                     "email",
 216                     "ip_address",
 217                     "user_id",
 218                     "username",
 219                 ],
 220                 filter_keys={"project_id": [self.project.id], "event_id": ["a" * 32]},
 221             )["data"][0],
 222         )
 223 
 224         assert event_from_nodestore.event_id == event_from_snuba.event_id
 225         assert event_from_nodestore.project_id == event_from_snuba.project_id
 226         assert event_from_nodestore.project == event_from_snuba.project
 227         assert event_from_nodestore.timestamp == event_from_snuba.timestamp
 228         assert event_from_nodestore.datetime == event_from_snuba.datetime
 229         assert event_from_nodestore.title == event_from_snuba.title
 230         assert event_from_nodestore.message == event_from_snuba.message
 231         assert event_from_nodestore.platform == event_from_snuba.platform
 232         assert event_from_nodestore.location == event_from_snuba.location
 233         assert event_from_nodestore.culprit == event_from_snuba.culprit

 236         assert event_from_nodestore.ip_address == event_from_snuba.ip_address

 239         # Group ID must be fetched from Snuba since it is not present in nodestore
 240         assert event_from_snuba.group_id
 241         assert event_from_snuba.group

 248 def test_renormalization(monkeypatch, factories, task_runner, default_project):
 249     from sentry_relay.processing import StoreNormalizer
 250 
 251     old_normalize = StoreNormalizer.normalize_event
 252     normalize_mock_calls = []
 253 
 254     def normalize(*args, **kwargs):
 255         normalize_mock_calls.append(1)
 256         return old_normalize(*args, **kwargs)
 257 
 258     monkeypatch.setattr("sentry_relay.processing.StoreNormalizer.normalize_event", normalize)
 259 
 260     with task_runner():
 261         factories.store_event(
 262             data={"event_id": "a" * 32, "environment": "production"}, project_id=default_project.id
 263         )
 264 
 265     # Assert we only renormalize this once. If this assertion fails it's likely
 266     # that you will encounter severe performance issues during event processing
 267     # or postprocessing.
 268     assert len(normalize_mock_calls) == 1
