 126     def test_get_next_prev_event_id(self):
 127         event = self.eventstore.get_event_by_id(self.project2.id, "b" * 32)
 128 
 129         _filter = Filter(project_ids=[self.project1.id, self.project2.id])
 130 
 131         prev_event = self.eventstore.get_prev_event_id(event, filter=_filter)
 132 
 133         next_event = self.eventstore.get_next_event_id(event, filter=_filter)
 134 
 135         assert prev_event == (six.text_type(self.project1.id), "a" * 32)
 136 
 137         # Events with the same timestamp are sorted by event_id
 138         assert next_event == (six.text_type(self.project2.id), "c" * 32)
 139 
 140         # Returns None if no event
 141         assert self.eventstore.get_prev_event_id(None, filter=_filter) is None
 142         assert self.eventstore.get_next_event_id(None, filter=_filter) is None

 144     def test_get_latest_or_oldest_event_id(self):
 145         # Returns a latest/oldest event
 146         event = self.eventstore.get_event_by_id(self.project2.id, "b" * 32)
 147         _filter = Filter(project_ids=[self.project1.id, self.project2.id])
 148         oldest_event = self.eventstore.get_earliest_event_id(event, filter=_filter)
 149         latest_event = self.eventstore.get_latest_event_id(event, filter=_filter)
 150         assert oldest_event == (six.text_type(self.project1.id), "a" * 32)
 151         assert latest_event == (six.text_type(self.project2.id), "e" * 32)
 152 
 153         # Returns none when no latest/oldest event that meets conditions
 154         event = self.eventstore.get_event_by_id(self.project2.id, "b" * 32)
 155         _filter = Filter(project_ids=[self.project1.id], group_ids=[self.event2.group_id])
 156         oldest_event = self.eventstore.get_earliest_event_id(event, filter=_filter)
 157         latest_event = self.eventstore.get_latest_event_id(event, filter=_filter)
 158         assert oldest_event is None
 159         assert latest_event is None
