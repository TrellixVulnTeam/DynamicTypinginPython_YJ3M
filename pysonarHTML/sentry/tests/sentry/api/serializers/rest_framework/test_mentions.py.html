   9 class ExtractUserIdsFromMentionsTest(TestCase):
  10     def test_users(self):
  11         actor = Actor(self.user.id, User)
  12         result = extract_user_ids_from_mentions(self.organization.id, [actor])
  13         assert result["users"] == set([self.user.id])
  14         assert result["team_users"] == set()
  15 
  16         other_user = self.create_user()
  17         result = extract_user_ids_from_mentions(
  18             self.organization.id, [actor, Actor(other_user.id, User)]
  19         )
  20         assert result["users"] == set([self.user.id, other_user.id])
  21         assert result["team_users"] == set()
  22 
  23     def test_teams(self):
  24         member_user = self.create_user()
  25         self.create_member(
  26             user=member_user, organization=self.organization, role="member", teams=[self.team]
  27         )
  28         actor = Actor(self.team.id, Team)
  29         result = extract_user_ids_from_mentions(self.organization.id, [actor])
  30         assert result["users"] == set()
  31         assert result["team_users"] == set([self.user.id, member_user.id])
  32 
  33         # Explicitly mentioned users shouldn't be included in team_users
  34         result = extract_user_ids_from_mentions(
  35             self.organization.id, [Actor(member_user.id, User), actor]
  36         )
  37         assert result["users"] == set([member_user.id])
  38         assert result["team_users"] == set([self.user.id])
