  14 class BaseAlertRuleTriggerSerializerTest(object):
  15     def assert_alert_rule_trigger_serialized(self, trigger, result):
  16         assert result["id"] == six.text_type(trigger.id)
  17         assert result["alertRuleId"] == six.text_type(trigger.alert_rule_id)
  18         assert result["label"] == trigger.label
  19         assert result["thresholdType"] == trigger.threshold_type
  20         assert result["alertThreshold"] == trigger.alert_threshold

  53     def test_excluded_projects(self):

  55         alert_rule = self.create_alert_rule(projects=excluded)
  56         trigger = create_alert_rule_trigger(
  57             alert_rule, "hi", AlertRuleThresholdType.ABOVE, 1000, 200, excluded_projects=excluded
