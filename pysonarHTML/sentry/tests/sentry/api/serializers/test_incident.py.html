  18 class IncidentSerializerTest(TestCase):
  19     @freeze_time()
  20     def test_simple(self):
  21         incident = self.create_incident(date_started=timezone.now() - timedelta(minutes=5))
  22         result = serialize(incident)
  23 
  24         assert result["id"] == six.text_type(incident.id)
  25         assert result["identifier"] == six.text_type(incident.identifier)
  26         assert result["organizationId"] == six.text_type(incident.organization_id)
  27         assert result["projects"] == [p.slug for p in incident.projects.all()]
  28         assert result["status"] == incident.status
  29         assert result["statusMethod"] == incident.status_method
  30         assert result["type"] == incident.type
  31         assert result["title"] == incident.title
  32         assert result["dateStarted"] == incident.date_started
  33         assert result["dateDetected"] == incident.date_detected
  34         assert result["dateCreated"] == incident.date_added
  35         assert result["dateClosed"] == incident.date_closed

  38 class DetailedIncidentSerializerTest(TestCase):
  39     def test_subscribed(self):
  40         incident = self.create_incident(date_started=timezone.now() - timedelta(minutes=5))
  41         serializer = DetailedIncidentSerializer()
  42         result = serialize(incident, serializer=serializer, user=self.user)
  43         assert not result["isSubscribed"]
  44         subscribe_to_incident(incident, self.user)
  45         result = serialize(incident, serializer=serializer, user=self.user)
  46         assert result["isSubscribed"]
  47
