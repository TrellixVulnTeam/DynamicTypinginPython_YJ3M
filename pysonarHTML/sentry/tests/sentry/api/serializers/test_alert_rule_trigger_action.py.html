  14 class AlertRuleTriggerActionSerializerTest(TestCase):
  15     def assert_action_serialized(self, action, result):
  16         assert result["id"] == six.text_type(action.id)
  17         assert result["alertRuleTriggerId"] == six.text_type(action.alert_rule_trigger_id)
  18         assert (
  19             result["type"]
  20             == AlertRuleTriggerAction.get_registered_type(
  21                 AlertRuleTriggerAction.Type(action.type)
  22             ).slug
  23         )
  24         assert (
  25             result["targetType"]
  26             == action_target_type_to_string[AlertRuleTriggerAction.TargetType(action.target_type)]
  27         )
  28         assert result["targetIdentifier"] == action.target_identifier
  29         assert result["integrationId"] == action.integration_id
  30         assert result["dateCreated"] == action.date_added
  31 
  32     def test_simple(self):
  33         alert_rule = self.create_alert_rule()
  34         trigger = create_alert_rule_trigger(
  35             alert_rule, "hi", AlertRuleThresholdType.ABOVE, 1000, 200
  36         )
  37         action = create_alert_rule_trigger_action(
  38             trigger,
  39             AlertRuleTriggerAction.Type.EMAIL,
  40             AlertRuleTriggerAction.TargetType.SPECIFIC,
  41             "hello",
  42         )
  43         result = serialize(action)
  44         self.assert_action_serialized(action, result)
