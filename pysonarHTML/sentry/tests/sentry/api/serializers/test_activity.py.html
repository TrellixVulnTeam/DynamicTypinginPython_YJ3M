   6 from sentry.api.serializers import serialize
   7 from sentry.models import Activity, PullRequest, Commit, GroupStatus
   8 from sentry.testutils import TestCase

  11 class GroupActivityTestCase(TestCase):
  12     def test_pr_activity(self):
  13         self.org = self.create_organization(name="Rowdy Tiger")
  14         user = self.create_user()
  15         group = self.create_group(status=GroupStatus.UNRESOLVED)
  16         repo = self.create_repo(self.project, name="organization-bar")
  17         pr = PullRequest.objects.create(
  18             organization_id=self.org.id,
  19             repository_id=repo.id,
  20             key=5,
  21             title="aaaa",
  22             message="kartoffel",
  23         )
  24 
  25         activity = Activity.objects.create(
  26             project_id=group.project_id,
  27             group=group,
  28             type=Activity.SET_RESOLVED_IN_PULL_REQUEST,
  29             ident=pr.id,
  30             user=user,
  31             data={"pull_request": pr.id},
  32         )
  33 
  34         result = serialize([activity], user)[0]["data"]
  35         pull_request = result["pullRequest"]
  36         assert pull_request["repository"]["name"] == "organization-bar"
  37         assert pull_request["message"] == "kartoffel"
