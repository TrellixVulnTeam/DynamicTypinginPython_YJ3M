  18 class BaseAlertRuleSerializerTest(object):
  19     def assert_alert_rule_serialized(self, alert_rule, result, skip_dates=False):
  20         assert result["id"] == six.text_type(alert_rule.id)
  21         assert result["organizationId"] == six.text_type(alert_rule.organization_id)
  22         assert result["name"] == alert_rule.name
  23         assert result["dataset"] == alert_rule.snuba_query.dataset
  24         assert result["query"] == alert_rule.snuba_query.query
  25         assert result["aggregate"] == alert_rule.snuba_query.aggregate
  26         assert result["timeWindow"] == alert_rule.snuba_query.time_window / 60
  27         assert result["resolution"] == alert_rule.snuba_query.resolution / 60
  28         assert result["thresholdPeriod"] == alert_rule.threshold_period
  29         assert result["includeAllProjects"] == alert_rule.include_all_projects
  30         if not skip_dates:
  31             assert result["dateModified"] == alert_rule.date_modified
  32             assert result["dateCreated"] == alert_rule.date_added
  33         if alert_rule.snuba_query.environment:
  34             assert result["environment"] == alert_rule.snuba_query.environment.name
  35         else:
  36             assert result["environment"] is None

  38     def create_issue_alert_rule(self, data):
  39         """data format
  40         {
  41             "project": project
  42             "environment": environment
  43             "name": "My rule name",
  44             "conditions": [],
  45             "actions": [],
  46             "actionMatch": "all"
  47         }
  48         """
  49         rule = Rule()
  50         rule.project = data["project"]
  51         if "environment" in data:
  52             environment = data["environment"]
  53             rule.environment_id = int(environment) if environment else environment
  54         if data.get("name"):
  55             rule.label = data["name"]
  56         if data.get("actionMatch"):
  57             rule.data["action_match"] = data["actionMatch"]
  58         if data.get("actions") is not None:
  59             rule.data["actions"] = data["actions"]
  60         if data.get("conditions") is not None:
  61             rule.data["conditions"] = data["conditions"]
  62         if data.get("frequency"):
  63             rule.data["frequency"] = data["frequency"]
  64         if data.get("date_added"):
  65             rule.date_added = data["date_added"]
  66 
  67         rule.save()
  68         return rule
