  10 class AppPlatformEventSerializerTest(TestCase):
  11     def setUp(self):
  12         self.user = self.create_user(username="foo")
  13         self.organization = self.create_organization(owner=self.user)
  14         self.sentry_app = self.create_sentry_app(organization=self.organization)
  15         self.install = self.create_sentry_app_installation(
  16             organization=self.organization, slug=self.sentry_app.slug
  17         )
  18 
  19     def test_no_actor(self):
  20         result = AppPlatformEvent(
  21             resource="event_alert", action="triggered", install=self.install, data={}
  22         )
  23 
  24         assert result.body == json.dumps(
  25             {
  26                 "action": "triggered",
  27                 "installation": {"uuid": self.install.uuid},
  28                 "data": {},
  29                 "actor": {"type": "application", "id": "sentry", "name": "Sentry"},
  30             }
  31         )
  32 
  33         signature = self.sentry_app.build_signature(result.body)
  34 
  35         assert result.headers["Content-Type"] == "application/json"
  36         assert result.headers["Sentry-Hook-Resource"] == "event_alert"
  37         assert result.headers["Sentry-Hook-Signature"] == signature
  38 
  39     def test_sentry_app_actor(self):
  40         result = AppPlatformEvent(
  41             resource="issue",
  42             action="assigned",
  43             install=self.install,
  44             data={},
  45             actor=self.sentry_app.proxy_user,
  46         )
  47 
  48         assert json.loads(result.body)["actor"] == {
  49             "type": "application",
  50             "id": self.sentry_app.uuid,
  51             "name": self.sentry_app.name,
  52         }
  53 
  54         signature = self.sentry_app.build_signature(result.body)
  55 
  56         assert result.headers["Content-Type"] == "application/json"
  57         assert result.headers["Sentry-Hook-Resource"] == "issue"
  58         assert result.headers["Sentry-Hook-Signature"] == signature
  59
