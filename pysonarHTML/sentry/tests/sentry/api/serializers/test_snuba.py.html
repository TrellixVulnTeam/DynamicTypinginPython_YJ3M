  12 class ZeroFillTest(unittest.TestCase):
  13     def run_test(self, filled_buckets, irregular_buckets, start, end, rollup, zerofilled_buckets):
  14         filled_buckets = [(start + (rollup * bucket), val) for bucket, val in filled_buckets]
  15         buckets = [(to_timestamp(date), val) for date, val in filled_buckets + irregular_buckets]
  16         sort_key = lambda row: row[0]
  17         buckets.sort(key=sort_key)
  18         zerofilled_buckets = [
  19             (to_timestamp(start + (rollup * bucket)), []) for bucket in zerofilled_buckets
  20         ]
  21         expected = buckets + zerofilled_buckets
  22         expected.sort(key=sort_key)
  23 
  24         assert zerofill(buckets, start, end, int(rollup.total_seconds())) == expected
  25 
  26     def test_missing_buckets(self):
  27         start = timezone.now().replace(minute=0, second=0, microsecond=0)
  28         rollup = timedelta(minutes=10)
  29         self.run_test(
  30             [(0, [0]), (1, [1])], [], start, start + timedelta(minutes=60), rollup, [2, 3, 4, 5, 6]
  31         )
  32         self.run_test(
  33             [(0, [0]), (2, [1]), (4, [4])],
  34             [],
  35             start,
  36             start + timedelta(minutes=60),
  37             rollup,
  38             [1, 3, 5, 6],
  39         )
  40 
  41     def test_non_rollup_buckets(self):
