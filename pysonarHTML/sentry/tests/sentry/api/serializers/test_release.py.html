 309     def test_deduplicate_users(self):
 310         """
 311         Tests that the same user is not returned more than once
 312         if there are commits associated with multiple of their
 313         emails
 314         """
 315         user = User.objects.create(email="stebe@sentry.io")
 316         email = UserEmail.objects.get(user=user, email="stebe@sentry.io")
 317         otheremail = UserEmail.objects.create(email="alsostebe@sentry.io", user=user)
 318         project = self.create_project()
 319         self.create_member(user=user, organization=project.organization)
 320         release = Release.objects.create(
 321             organization_id=project.organization_id, version=uuid4().hex, new_groups=1
 322         )
 323         release.add_project(project)
 324         commit_author1 = CommitAuthor.objects.create(
 325             name="stebe", email=email.email, organization_id=project.organization_id
 326         )
 327         commit_author2 = CommitAuthor.objects.create(
 328             name="stebe", email=otheremail.email, organization_id=project.organization_id
 329         )
 330         commit1 = Commit.objects.create(
 331             organization_id=project.organization_id,
 332             repository_id=1,
 333             key="abc",
 334             author=commit_author1,
 335             message="waddap",
 336         )
 337         commit2 = Commit.objects.create(
 338             organization_id=project.organization_id,
 339             repository_id=1,
 340             key="cde",
 341             author=commit_author2,
 342             message="oh hi",
 343         )
 344         ReleaseCommit.objects.create(
 345             organization_id=project.organization_id,
 346             project_id=project.id,
 347             release=release,
 348             commit=commit1,
 349             order=1,
 350         )
 351         ReleaseCommit.objects.create(
 352             organization_id=project.organization_id,
 353             project_id=project.id,
 354             release=release,
 355             commit=commit2,
 356             order=2,
 357         )
 358         ReleaseProject.objects.filter(release=release, project=project).update(new_groups=1)
 359         release.update(
 360             authors=[six.text_type(commit_author1.id), six.text_type(commit_author2.id)],
 361             commit_count=2,
 362             last_commit_id=commit2.id,
 363         )
 364         result = serialize(release, user)
 365         assert len(result["authors"]) == 1
 366         assert result["authors"][0]["email"] == "stebe@sentry.io"
