 231 class SimpleEventSerializerTest(TestCase):
 232     def test_user(self):
 233         """
 234         Use the SimpleEventSerializer to serialize an event
 235         """
 236         event = self.store_event(
 237             data={
 238                 "event_id": "a" * 32,
 239                 "timestamp": iso_format(before_now(minutes=1)),
 240                 "user": {"email": "test@test.com"},
 241             },
 242             project_id=self.project.id,
 243         )
 244         result = serialize(event, None, SimpleEventSerializer())
 245 
 246         assert result["eventID"] == event.event_id
 247         assert result["projectID"] == six.text_type(event.project_id)
 248         assert result["groupID"] == six.text_type(event.group.id)
 249         assert result["message"] == event.message
 250         assert result["title"] == event.title
 251         assert result["location"] == event.location
 252         assert result["culprit"] == event.culprit
 253         assert result["dateCreated"] == event.datetime
 254         assert result["user"]["id"] == event.get_minimal_user().id
 255         assert result["user"]["email"] == event.get_minimal_user().email
 256         assert result["user"]["username"] == event.get_minimal_user().username
 257         assert result["user"]["ip_address"] == event.get_minimal_user().ip_address
 258         assert result["tags"] == [
 259             {"key": "level", "value": "error"},
 260             {"key": "user", "value": "email:test@test.com", "query": 'user.email:"test@test.com"'},
 261         ]
