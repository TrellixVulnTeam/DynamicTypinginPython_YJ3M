 147     def test_project_features(self):
 148         early_flag = "projects:TEST_early"
 149         red_flag = "projects:TEST_red"
 150         blue_flag = "projects:TEST_blue"
 151 
 152         early_adopter = self.create_organization()
 153         early_red = self.create_project(organization=early_adopter)
 154         early_blue = self.create_project(organization=early_adopter)
 155 
 156         late_adopter = self.create_organization()
 157         late_red = self.create_project(organization=late_adopter)
 158         late_blue = self.create_project(organization=late_adopter)
 159 
 160         class EarlyAdopterFeatureHandler(features.BatchFeatureHandler):
 161             features = {early_flag}
 162 
 163             def _check_for_batch(self, feature_name, organization, actor):
 164                 return organization == early_adopter
 165 
 166         def create_color_handler(color_flag, included_projects):
 167             class ProjectColorFeatureHandler(features.FeatureHandler):
 168                 features = {color_flag}
 169 
 170                 def has(self, feature, actor):
 171                     return feature.project in included_projects
 172 
 173             return ProjectColorFeatureHandler()
 174 
 175         features.add(early_flag, features.ProjectFeature)
 176         features.add(red_flag, features.ProjectFeature)
 177         features.add(blue_flag, features.ProjectFeature)
 178         red_handler = create_color_handler(red_flag, [early_red, late_red])
 179         blue_handler = create_color_handler(blue_flag, [early_blue, late_blue])
 180         for handler in (EarlyAdopterFeatureHandler(), red_handler, blue_handler):
 181             features.add_handler(handler)
 182 
 183         def api_form(flag):
 184             return flag[len("projects:") :]
 185 
 186         flags_to_find = set(api_form(f) for f in [early_flag, red_flag, blue_flag])
 187 
 188         def assert_has_features(project, expected_features):
 189             serialized = serialize(project)
 190             actual_features = set(f for f in serialized["features"] if f in flags_to_find)
 191             assert actual_features == set(api_form(f) for f in expected_features)
 192 
 193         assert_has_features(early_red, [early_flag, red_flag])
 194         assert_has_features(early_blue, [early_flag, blue_flag])
 195         assert_has_features(late_red, [red_flag])
 196         assert_has_features(late_blue, [blue_flag])

 433 class BulkFetchProjectLatestReleases(TestCase):
 434     @fixture
 435     def project(self):

 438     @fixture
 439     def other_project(self):
 440         return self.create_project(teams=[self.team], organization=self.organization)
 441 
 442     def test_single_no_release(self):
 443         assert bulk_fetch_project_latest_releases([self.project]) == []
 444 
 445     def test_single_release(self):
 446         release = self.create_release(
 447             self.project, date_added=timezone.now() - timedelta(minutes=5)
 448         )
 449         assert bulk_fetch_project_latest_releases([self.project]) == [release]
 450         newer_release = self.create_release(self.project)
 451         assert bulk_fetch_project_latest_releases([self.project]) == [newer_release]
 452 
 453     def test_multi_no_release(self):
 454         assert bulk_fetch_project_latest_releases([self.project, self.other_project]) == []
 455 
 456     def test_multi_mixed_releases(self):
 457         release = self.create_release(self.project)
 458         assert set(bulk_fetch_project_latest_releases([self.project, self.other_project])) == set(
 459             [release]
 460         )
 461 
 462     def test_multi_releases(self):
 463         release = self.create_release(
 464             self.project, date_added=timezone.now() - timedelta(minutes=5)
 465         )
 466         other_project_release = self.create_release(self.other_project)
 467         assert set(bulk_fetch_project_latest_releases([self.project, self.other_project])) == set(
 468             [release, other_project_release]
 469         )
 470         release_2 = self.create_release(self.project)
 471         assert set(bulk_fetch_project_latest_releases([self.project, self.other_project])) == set(
 472             [release_2, other_project_release]
