  12 class CommitFileChangeSerializerTest(TestCase):
  13     def test_simple(self):
  14         user = self.create_user()
  15         project = self.create_project()
  16         release = Release.objects.create(
  17             organization_id=project.organization_id, version=uuid4().hex
  18         )
  19         release.add_project(project)
  20         repository = Repository.objects.create(
  21             organization_id=project.organization_id, name="test/test"
  22         )
  23         commit_author = CommitAuthor.objects.create(
  24             name="stebe", email="stebe@sentry.io", organization_id=project.organization_id
  25         )
  26         commit = Commit.objects.create(
  27             organization_id=project.organization_id,
  28             repository_id=repository.id,
  29             key="abc",
  30             author=commit_author,
  31             message="waddap",
  32         )
  33         ReleaseCommit.objects.create(
  34             organization_id=project.organization_id,
  35             project_id=project.id,
  36             release=release,
  37             commit=commit,
  38             order=1,
  39         )
  40         cfc = CommitFileChange.objects.create(
  41             organization_id=project.organization_id, commit=commit, filename=".gitignore", type="M"
  42         )
  43         result = serialize(cfc, user)
  44 
  45         assert result["filename"] == ".gitignore"
  46         assert result["commitMessage"] == "waddap"
  47         assert result["author"] == {"name": "stebe", "email": "stebe@sentry.io"}
