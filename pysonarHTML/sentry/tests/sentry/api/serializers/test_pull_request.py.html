  14 class PullRequestSerializerTest(TestCase):
  15     def test_simple(self):
  16         user = self.create_user()
  17         project = self.create_project()
  18         release = Release.objects.create(
  19             organization_id=project.organization_id, version=uuid4().hex
  20         )
  21         release.add_project(project)
  22         repository = Repository.objects.create(
  23             organization_id=project.organization_id, name="test/test"
  24         )
  25         commit_author = CommitAuthor.objects.create(
  26             name="stebe", email="stebe@sentry.io", organization_id=project.organization_id
  27         )
  28         pull_request = PullRequest.objects.create(
  29             organization_id=project.organization_id,
  30             repository_id=repository.id,
  31             key="9",
  32             author=commit_author,
  33             message="waddap",
  34             title="cool pr",
  35         )
  36 
  37         result = serialize(pull_request, user)
  38 
  39         assert result["message"] == "waddap"
  40         assert result["title"] == "cool pr"
  41         assert result["repository"]["name"] == "test/test"
  42         assert result["author"] == {"name": "stebe", "email": "stebe@sentry.io"}

  65     def test_integration_repository(self):

  71         user = self.create_user()

  74             organization_id=project.organization_id, version=uuid4().hex

  78             organization_id=project.organization_id,

  85         )
  86         pull_request = PullRequest.objects.create(
  87             organization_id=project.organization_id,
  88             repository_id=repository.id,
  89             key="9",
