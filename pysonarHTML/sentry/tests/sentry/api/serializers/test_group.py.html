 166     def test_implicit_subscribed(self):
 167         user = self.create_user()
 168         group = self.create_group()
 169 
 170         combinations = (
 171             # ((default, project), (subscribed, details))
 172             ((UserOptionValue.all_conversations, None), (True, None)),
 173             ((UserOptionValue.all_conversations, UserOptionValue.all_conversations), (True, None)),
 174             (
 175                 (UserOptionValue.all_conversations, UserOptionValue.participating_only),
 176                 (False, None),
 177             ),
 178             (
 179                 (UserOptionValue.all_conversations, UserOptionValue.no_conversations),
 180                 (False, {"disabled": True}),
 181             ),
 182             ((None, None), (False, None)),
 183             ((UserOptionValue.participating_only, None), (False, None)),
 184             ((UserOptionValue.participating_only, UserOptionValue.all_conversations), (True, None)),
 185             (
 186                 (UserOptionValue.participating_only, UserOptionValue.participating_only),
 187                 (False, None),
 188             ),
 189             (
 190                 (UserOptionValue.participating_only, UserOptionValue.no_conversations),
 191                 (False, {"disabled": True}),
 192             ),
 193             ((UserOptionValue.no_conversations, None), (False, {"disabled": True})),
 194             ((UserOptionValue.no_conversations, UserOptionValue.all_conversations), (True, None)),
 195             ((UserOptionValue.no_conversations, UserOptionValue.participating_only), (False, None)),
 196             (
 197                 (UserOptionValue.no_conversations, UserOptionValue.no_conversations),
 198                 (False, {"disabled": True}),
 199             ),
 200         )
 201 
 202         def maybe_set_value(project, value):
 203             if value is not None:
 204                 UserOption.objects.set_value(
 205                     user=user, project=project, key="workflow:notifications", value=value
 206                 )
 207             else:
 208                 UserOption.objects.unset_value(
 209                     user=user, project=project, key="workflow:notifications"
 210                 )
 211 
 212         for options, (is_subscribed, subscription_details) in combinations:
 213             default_value, project_value = options
 214             UserOption.objects.clear_local_cache()
 215             maybe_set_value(None, default_value)
 216             maybe_set_value(group.project, project_value)
 217             result = serialize(group, user)
 218             assert result["isSubscribed"] is is_subscribed
 219             assert result.get("subscriptionDetails") == subscription_details

 266 class StreamGroupSerializerTestCase(TestCase):
 267     def test_environment(self):
 268         group = self.group
 269 
 270         environment = Environment.get_or_create(group.project, "production")
 271 
 272         from sentry.api.serializers.models.group import tsdb
 273 
 274         with mock.patch(
 275             "sentry.api.serializers.models.group.tsdb.get_range", side_effect=tsdb.get_range
 276         ) as get_range:
 277             serialize(
 278                 [group],
 279                 serializer=StreamGroupSerializer(
 280                     environment_func=lambda: environment, stats_period="14d"
 281                 ),
 282             )
 283             assert get_range.call_count == 1
 284             for args, kwargs in get_range.call_args_list:
 285                 assert kwargs["environment_ids"] == [environment.id]
 286 
 287         def get_invalid_environment():
 288             raise Environment.DoesNotExist()
 289 
 290         with mock.patch(
 291             "sentry.api.serializers.models.group.tsdb.make_series", side_effect=tsdb.make_series
 292         ) as make_series:
 293             serialize(
 294                 [group],
 295                 serializer=StreamGroupSerializer(
 296                     environment_func=get_invalid_environment, stats_period="14d"
 297                 ),
 298             )
 299             assert make_series.call_count == 1
