  12 class SentryAppSerializerTest(TestCase):
  13     def test_published_app(self):
  14         user = self.create_user()
  15         organization = self.create_organization(owner=user)
  16         sentry_app = self.create_sentry_app(
  17             name="Tesla App",
  18             organization=organization,
  19             published=True,
  20             scopes=("org:write", "team:admin"),
  21         )
  22         result = serialize(sentry_app, None, SentryAppSerializer(), access=None)
  23 
  24         assert result["name"] == "Tesla App"
  25         assert result["featureData"] == [
  26             {
  27                 "description": "Tesla App can **utilize the Sentry API** to pull data or update resources in Sentry (with permissions granted, of course).",
  28                 "featureGate": "integrations-api",
  29             }
  30         ]
  31         assert result["scopes"] == ["org:write", "team:admin"]
  32         assert result.get("clientSecret") is None
