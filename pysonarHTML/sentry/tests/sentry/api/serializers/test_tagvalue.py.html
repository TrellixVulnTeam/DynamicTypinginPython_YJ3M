  12 class TagValueSerializerTest(TestCase):
  13     def test_with_user(self):
  14         user = self.create_user()
  15         tagvalue = TagValue(
  16             key="sentry:user",
  17             value="username:ted",
  18             times_seen=1,
  19             first_seen=datetime(2018, 1, 1),
  20             last_seen=datetime(2018, 1, 1),
  21         )
  22 
  23         result = serialize(tagvalue, user)
  24         assert result["key"] == "user"
  25         assert result["value"] == "username:ted"
  26         assert result["name"] == "ted"
  27         assert result["query"] == 'user.username:"ted"'
