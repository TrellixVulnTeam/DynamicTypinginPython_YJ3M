  44     def test_member_count_does_not_include_invite_requests(self):
  45         org = self.create_organization(owner=self.user)
  46         team = self.create_team(organization=org)
  47         self.create_member(user=self.create_user(), organization=org, teams=[team])  # member
  48         self.create_member(email="1@example.com", organization=org, teams=[team])  # pending invite
  49 
  50         result = serialize(team, self.user)
  51         assert result["memberCount"] == 2
  52 
  53         # invite requests
  54         self.create_member(
  55             email="2@example.com",
  56             organization=org,
  57             invite_status=InviteStatus.REQUESTED_TO_BE_INVITED.value,
  58             teams=[team],
  59         )
  60         self.create_member(
  61             email="3@gmail.com",
  62             organization=org,
  63             invite_status=InviteStatus.REQUESTED_TO_JOIN.value,
  64             teams=[team],
  65         )
  66 
  67         result = serialize(team, self.user)
  68         assert result["memberCount"] == 2

  70     def test_member_access(self):
  71         user = self.create_user(username="foo")
  72         organization = self.create_organization()
  73         self.create_member(user=user, organization=organization)
  74         team = self.create_team(organization=organization)
  75 
  76         result = serialize(team, user)
  77         result.pop("dateCreated")
  78 
  79         assert result["hasAccess"] is True
  80         assert result["isMember"] is False
  81 
  82         organization.flags.allow_joinleave = False
  83         organization.save()
  84         result = serialize(team, user)
  85         # after changing to allow_joinleave=False
  86         assert result["hasAccess"] is False
  87         assert result["isMember"] is False
  88 
  89         self.create_team_membership(user=user, team=team)
  90         result = serialize(team, user)
  91         # after giving them access to team
  92         assert result["hasAccess"] is True
  93         assert result["isMember"] is True

 171 class TeamWithProjectsSerializerTest(TestCase):
 172     def test_simple(self):
 173         user = self.create_user(username="foo")
 174         organization = self.create_organization(owner=user)
 175         team = self.create_team(organization=organization)
 176         project = self.create_project(teams=[team], organization=organization, name="foo")
 177         project2 = self.create_project(teams=[team], organization=organization, name="bar")
 178 
 179         result = serialize(team, user, TeamWithProjectsSerializer())
 180         serialized_projects = serialize([project2, project], user)
 181 
 182         assert result == {
 183             "slug": team.slug,
 184             "name": team.name,
 185             "hasAccess": True,
 186             "isPending": False,
 187             "isMember": False,
 188             "id": six.text_type(team.id),
 189             "projects": serialized_projects,
 190             "avatar": {"avatarType": "letter_avatar", "avatarUuid": None},
 191             "memberCount": 0,
 192             "dateCreated": team.date_added,
 193         }
