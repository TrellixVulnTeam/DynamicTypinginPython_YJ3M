   9 class OrganizationMemberWithProjectsSerializerTest(TestCase):
  10     def setUp(self):
  11         self.owner_user = self.create_user("foo@localhost", username="foo")
  12         self.user_2 = self.create_user("bar@localhost", username="bar")
  13 
  14         self.org = self.create_organization(owner=self.owner_user)
  15         self.org.member_set.create(user=self.user_2)
  16         self.team = self.create_team(organization=self.org, members=[self.owner_user, self.user_2])
  17         self.team_2 = self.create_team(organization=self.org, members=[self.user_2])
  18         self.project = self.create_project(teams=[self.team])
  19         self.project_2 = self.create_project(teams=[self.team_2])
  20 
  21     def test_simple(self):
  22         projects_ids = [self.project.id, self.project_2.id]
  23         org_members = list(
  24             self.org.member_set.filter(user__in=[self.owner_user, self.user_2]).order_by(
  25                 "user__email"
  26             )
  27         )
  28         result = serialize(
  29             org_members,
  30             self.user_2,
  31             OrganizationMemberWithProjectsSerializer(project_ids=projects_ids),
  32         )
  33         expected_projects = [[self.project.slug, self.project_2.slug], [self.project.slug]]
  34         expected_projects[0].sort()
  35         assert [r["projects"] for r in result] == expected_projects
  36 
  37         projects_ids = [self.project_2.id]
  38         result = serialize(
  39             org_members,
  40             self.user_2,
  41             OrganizationMemberWithProjectsSerializer(project_ids=projects_ids),
  42         )
  43         expected_projects = [[self.project_2.slug], []]
  44         assert [r["projects"] for r in result] == expected_projects
