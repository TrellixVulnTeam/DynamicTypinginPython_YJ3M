  19 class IncidentActivitySerializerTest(TestCase, SnubaTestCase):
  20     def test_simple(self):
  21         activity = create_incident_activity(
  22             incident=self.create_incident(),
  23             activity_type=IncidentActivityType.COMMENT,
  24             user=self.user,
  25             comment="hello",
  26         )
  27         result = serialize(activity)

  33         assert result["value"] is None
  34         assert result["previousValue"] is None

  56     def test_event_stats(self):
  57         now = datetime.now()
  58         with freeze_time((now - timedelta(days=1)).replace(hour=12, minute=30, second=25)):
  59             for _ in range(2):
  60                 self.store_event(
  61                     data={
  62                         "event_id": uuid4().hex,
  63                         "fingerprint": ["group1"],
  64                         "timestamp": iso_format(before_now(seconds=1)),
  65                     },
  66                     project_id=self.project.id,
  67                 )
  68             incident = self.create_incident(
  69                 date_started=timezone.now() - timedelta(hours=2), projects=[self.project], query=""
  70             )
  71             activity = create_incident_activity(
  72                 incident=incident,
  73                 activity_type=IncidentActivityType.COMMENT,
  74                 user=self.user,
  75                 comment="hello",
  76             )
  77             result = serialize(activity)
  78
