   8 class TeamPermissionBase(TestCase):
   9     def setUp(self):
  10         self.org = self.create_organization(flags=0)
  11         self.team = self.create_team(organization=self.org)
  12         super(TeamPermissionBase, self).setUp()
  13 
  14     def has_object_perm(self, method, obj, auth=None, user=None, is_superuser=None):
  15         perm = TeamPermission()
  16         request = self.make_request(user=user, auth=auth, method=method)
  17         if is_superuser:
  18             request.superuser.set_logged_in(request.user)
  19         return perm.has_permission(request, None) and perm.has_object_permission(request, None, obj)
