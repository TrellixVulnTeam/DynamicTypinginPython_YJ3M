  31     def has_object_perm(self, method, obj, auth=None, user=None, is_superuser=None):
  32         perm = OrganizationPermission()
  33         request = self.make_request(user=user, auth=auth, method=method)
  34         if is_superuser:
  35             request.superuser.set_logged_in(request.user)
  36         return perm.has_permission(request, None) and perm.has_object_permission(request, None, obj)

 126     def build_request(self, user=None, active_superuser=False, **params):
 127         request = RequestFactory().get("/", params)
 128         request.session = {}
 129         if active_superuser:
 130             request.superuser = MockSuperUser()
 131         if user is None:
 132             user = self.user
 133         request.user = user
 134         request.access = from_request(request, self.org)
 135         return request

 287 class GetFilterParamsTest(BaseOrganizationEndpointTest):
 288     def setUp(self):
 289         self.team_1 = self.create_team(organization=self.org)
 290         self.project_1 = self.create_project(organization=self.org, teams=[self.team_1])
 291         self.project_2 = self.create_project(organization=self.org, teams=[self.team_1])
 292         self.env_1 = self.create_environment(project=self.project_1)
 293         self.env_2 = self.create_environment(project=self.project_1)
 294 
 295     def run_test(
 296         self,
 297         expected_projects,
 298         expected_envs=None,
 299         expected_start=None,
 300         expected_end=None,
 301         env_names=None,
 302         user=None,
 303         date_filter_optional=False,
 304         project_ids=None,
 305         start=None,
 306         end=None,
 307         stats_period=None,
 308         active_superuser=False,
 309     ):
 310         request_args = {}
 311         if env_names:
 312             request_args["environment"] = env_names
 313         if project_ids:
 314             request_args["project"] = project_ids
 315         if start and end:
 316             request_args["start"] = start
 317             request_args["end"] = end
 318         if stats_period:
 319             request_args["statsPeriod"] = stats_period
 320 
 321         result = self.endpoint.get_filter_params(
 322             self.build_request(user=user, active_superuser=active_superuser, **request_args),
 323             self.org,
 324             date_filter_optional=date_filter_optional,
 325         )
 326 
 327         assert set([p.id for p in expected_projects]) == set(result["project_id"])
 328         assert expected_start == result["start"]
 329         assert expected_end == result["end"]
 330         if expected_envs:
 331             assert set([e.name for e in expected_envs]) == set(result["environment"])
 332         else:
 333             assert "environment" not in result
