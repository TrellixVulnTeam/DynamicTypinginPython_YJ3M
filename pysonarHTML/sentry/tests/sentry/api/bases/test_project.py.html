  15     def has_object_perm(self, method, obj, auth=None, user=None, is_superuser=None):
  16         perm = ProjectPermission()
  17         request = self.make_request(user=user, auth=auth, method=method)
  18         if is_superuser:
  19             request.superuser.set_logged_in(request.user)
  20         return perm.has_permission(request, None) and perm.has_object_permission(request, None, obj)
