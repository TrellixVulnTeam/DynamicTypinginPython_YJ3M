  11 class IntegrationEndpointTest(APITestCase):
  12     def setUp(self):
  13         self.endpoint = IntegrationEndpoint()
  14 
  15     def test_handle_exception(self):
  16         exc = APIException("There was a problem!")
  17         exc.status_code = 400  # set the status code to 400 not possible to set in init
  18         exc.code = 400  # rest framework APIError is not compatible with integration APIError exception type
  19         resp = self.endpoint.handle_exception(HttpRequest(), exc)
  20         assert resp.status_code == 400
  21         assert resp.exception is True
  22 
  23     def test_handle_exception_503(self):
  24         resp = IntegrationEndpoint().handle_exception(
  25             HttpRequest(), ApiError("This is an error", code=503)
  26         )
  27         assert resp.status_code == 503
  28         assert resp.exception is True
  29 
  30     def test_handle_exception_stdlib(self):
  31         resp = IntegrationEndpoint().handle_exception(HttpRequest(), ValueError("This is an error"))
  32         assert resp.status_code == 500
  33         assert resp.exception is True
