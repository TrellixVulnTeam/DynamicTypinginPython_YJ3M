 196             assert activity.group_id == group.id

 333     def test_default_project(self):
 334         def assert_default_project(path, action, expected_project_field):
 335             response = self.client.get(path)
 336             assert response.status_code == 200
 337             if action == "create":
 338                 fields = response.data["createIssueConfig"]
 339             else:
 340                 fields = response.data["linkIssueConfig"]
 341             assert response.data["id"] == six.text_type(integration.id)
 342             for field in fields:
 343                 if field["name"] == "project":
 344                     project_field = field
 345                     break
 346 
 347             assert project_field == expected_project_field
 348 
 349         self.login_as(user=self.user)
 350         org = self.organization
 351         event = self.store_event(
 352             data={"event_id": "b" * 32, "timestamp": iso_format(self.min_ago)},
 353             project_id=self.project.id,
 354         )
 355         group = event.group
 356         integration = Integration.objects.create(provider="example", name="Example")
 357         org_integration = integration.add_organization(org, self.user)
 358         org_integration.config = {"project_issue_defaults": {group.project_id: {"project": "2"}}}
 359         org_integration.save()
 360         create_path = u"/api/0/issues/{}/integrations/{}/?action=create".format(
 361             group.id, integration.id
 362         )
 363         link_path = u"/api/0/issues/{}/integrations/{}/?action=link".format(
 364             group.id, integration.id
 365         )
 366         project_field = {
 367             "name": "project",
 368             "label": "Project",
 369             "choices": [("1", "Project 1"), ("2", "Project 2")],
 370             "type": "select",
 371             "default": "2",
 372         }
 373         with self.feature("organizations:integrations-issue-basic"):
 374             assert_default_project(create_path, "create", project_field)
 375             assert_default_project(link_path, "link", project_field)
