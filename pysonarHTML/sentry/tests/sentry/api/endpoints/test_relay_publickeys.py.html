  94     def _call_endpoint(self, calling_relay):
  95 
  96         raw_json, signature = self.private_key.pack(
  97             {
  98                 "relay_ids": [
  99                     six.text_type(self.relay_a.relay_id),
 100                     six.text_type(self.relay_b.relay_id),
 101                     self.non_existing_key,
 102                 ]
 103             }
 104         )
 105 
 106         resp = self.client.post(
 107             self.path,
 108             data=raw_json,
 109             content_type="application/json",
 110             HTTP_X_SENTRY_RELAY_ID=calling_relay.relay_id,
 111             HTTP_X_SENTRY_RELAY_SIGNATURE=signature,
 112         )
 113 
 114         result = json.loads(resp.content)
 115         return result
