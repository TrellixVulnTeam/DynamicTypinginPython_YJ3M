  10 class SentryInternalAppTokenCreationTest(APITestCase):
  11     def setUp(self):
  12         self.user = self.create_user(email="boop@example.com")
  13         self.org = self.create_organization(owner=self.user, name="My Org")
  14         self.project = self.create_project(organization=self.org)
  15 
  16         self.internal_sentry_app = self.create_internal_integration(
  17             name="My Internal App", organization=self.org
  18         )
  19 
  20         self.api_token = ApiToken.objects.get(application=self.internal_sentry_app.application)
  21 
  22         self.url = reverse(
  23             "sentry-api-0-sentry-internal-app-token-details",
  24             args=[self.internal_sentry_app.slug, self.api_token.token],
  25         )
  26 
  27     def test_delete_token(self):
  28         self.login_as(user=self.user)
  29         response = self.client.delete(self.url, format="json")
  30         assert response.status_code == 204
  31         assert not ApiToken.objects.filter(pk=self.api_token.id).exists()

  58     def test_non_internal_app(self):

  60 
  61         install = self.create_sentry_app_installation(
  62             slug=sentry_app.slug, organization=self.org, user=self.user
  63         )
  64 
  65         client_id = install.sentry_app.application.client_id
  66         user = install.sentry_app.proxy_user
  67 
  68         api_token = GrantExchanger.run(
  69             install=install, code=install.api_grant.code, client_id=client_id, user=user
  70         )
  71 
  72         url = reverse(
  73             "sentry-api-0-sentry-internal-app-token-details",
  74             args=[install.sentry_app.slug, api_token.token],
  75         )
  76 
  77         self.login_as(user=self.user)
  78         response = self.client.delete(url, format="json")
  79 
  80         assert response.status_code == 403
  81         assert response.data == "This route is limited to internal integrations only"
