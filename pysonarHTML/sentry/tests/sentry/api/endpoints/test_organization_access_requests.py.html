  12     def test_owner_can_list_access_requests(self):
  13         self.login_as(user=self.user)
  14 
  15         organization = self.create_organization(name="foo", owner=self.user)
  16         user = self.create_user("bar@example.com")
  17         member = self.create_member(organization=organization, user=user, role="member")
  18         team = self.create_team(name="foo", organization=organization)
  19 
  20         OrganizationAccessRequest.objects.create(member=member, team=team)
  21 
  22         path = reverse("sentry-api-0-organization-access-requests", args=[organization.slug])

  30     def test_admin_can_list_access_requests(self):
  31         organization = self.create_organization(
  32             name="foo", owner=self.user, flags=0  # kill default allow_joinleave
  33         )
  34         team_1 = self.create_team(name="foo", organization=organization)
  35 
  36         # team which this team_admin user is not an admin of
  37         team_2 = self.create_team(name="bar", organization=organization)
  38 
  39         team_admin = self.create_user("admin@example.com")
  40         self.create_member(organization=organization, user=team_admin, role="admin", teams=[team_1])
  41 
  42         other_user = self.create_user("bar@example.com")
  43         other_member = self.create_member(
  44             organization=organization, user=other_user, role="member", teams=[]
  45         )
  46 
  47         request_1 = OrganizationAccessRequest.objects.create(member=other_member, team=team_1)
  48         OrganizationAccessRequest.objects.create(member=other_member, team=team_2)
  49 
  50         path = reverse("sentry-api-0-organization-access-requests", args=[organization.slug])
  51 
  52         self.login_as(team_admin)
  53 
  54         resp = self.client.get(path)
  55 
  56         assert resp.status_code == 200
  57 
  58         # There are requests to two diff teams, make sure we only get the one user is team admin for
  59         assert len(resp.data) == 1
  60         assert resp.data[0]["member"]["id"] == six.text_type(other_member.id)
  61         assert resp.data[0]["member"]["id"] == six.text_type(request_1.member_id)
  62         assert resp.data[0]["team"]["id"] == six.text_type(request_1.team_id)
