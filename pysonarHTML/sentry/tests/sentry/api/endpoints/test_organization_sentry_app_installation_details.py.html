  47 class GetSentryAppInstallationDetailsTest(SentryAppInstallationDetailsTest):
  48     def test_access_within_installs_organization(self):
  49         self.login_as(user=self.user)
  50         response = self.client.get(self.url, format="json")
  51 
  52         assert response.status_code == 200, response.content
  53         assert response.data == {
  54             "app": {"uuid": self.unpublished_app.uuid, "slug": self.unpublished_app.slug},
  55             "organization": {"slug": self.org.slug},
  56             "uuid": self.installation2.uuid,
  57             "code": self.installation2.api_grant.code,
  58             "status": "pending",
  59         }

  88 class MarkInstalledSentryAppInstallationsTest(SentryAppInstallationDetailsTest):
  89     def setUp(self):
  90         super(MarkInstalledSentryAppInstallationsTest, self).setUp()
  91         self.token = GrantExchanger.run(
  92             install=self.installation,
  93             code=self.installation.api_grant.code,
  94             client_id=self.published_app.application.client_id,
  95             user=self.published_app.proxy_user,
  96         )

  98     @patch("sentry.analytics.record")
  99     def test_sentry_app_installation_mark_installed(self, record):
 100         self.url = reverse(
 101             "sentry-api-0-sentry-app-installation-details", args=[self.installation.uuid]
 102         )
 103         response = self.client.put(
 104             self.url,
 105             data={"status": "installed"},
 106             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 107             format="json",
 108         )
 109         assert response.status_code == 200
 110         assert response.data["status"] == "installed"
 111 
 112         record.assert_called_with(
 113             "sentry_app_installation.updated",
 114             sentry_app_installation_id=self.installation.id,
 115             sentry_app_id=self.installation.sentry_app.id,
 116             organization_id=self.installation.organization.id,
 117         )
