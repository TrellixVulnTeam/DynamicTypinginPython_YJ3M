   9 class ApiApplicationDetailsTest(APITestCase):
  10     def test_simple(self):
  11         app = ApiApplication.objects.create(owner=self.user, name="a")
  12 
  13         self.login_as(self.user)
  14         url = reverse("sentry-api-0-api-application-details", args=[app.client_id])
