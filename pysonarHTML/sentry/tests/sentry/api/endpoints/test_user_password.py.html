  12 class UserPasswordTest(APITestCase):
  13     def setUp(self):
  14         self.user = self.create_user(email="a@example.com", is_managed=False, name="example name")
  15         self.user.set_password("helloworld!")
  16         self.user.save()
  17 
  18         self.login_as(user=self.user)
  19         self.url = reverse("sentry-api-0-user-password", kwargs={"user_id": "me"})
  20 
  21     def test_change_password(self):
  22         old_password = self.user.password
  23         response = self.client.put(
  24             self.url,
  25             data={
  26                 "password": "helloworld!",
  27                 "passwordNew": "testpassword",
  28                 "passwordVerify": "testpassword",
  29             },
  30         )
  31         user = User.objects.get(id=self.user.id)
  32         assert response.status_code == 204
  33         assert old_password != user.password
