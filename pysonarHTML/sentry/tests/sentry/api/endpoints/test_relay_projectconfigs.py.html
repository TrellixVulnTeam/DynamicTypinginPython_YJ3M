  24 def _get_all_keys(config):
  25     for key in config:
  26         yield key
  27         if isinstance(config[key], dict):
  28             for key in _get_all_keys(config[key]):

 280 @pytest.mark.django_db
 281 def test_relay_projectconfig_cache_full_config(
 282     call_endpoint, default_project, projectconfig_cache_set, task_runner
 283 ):
 284     """
 285     When a relay fetches a full config, that config should end up in Redis.
 286     """
 287 
 288     with task_runner():
 289         result, status_code = call_endpoint(full_config=True)
 290         assert status_code < 400
 291 
 292     (http_cfg,) = six.itervalues(result["configs"])
 293     (call,) = projectconfig_cache_set
 294     assert len(call) == 1
 295     redis_cfg = call[six.text_type(default_project.id)]
 296 
 297     del http_cfg["lastFetch"]
 298     del http_cfg["lastChange"]
 299     del redis_cfg["lastFetch"]
 300     del redis_cfg["lastChange"]

 302     assert redis_cfg == http_cfg
