  15         )
  16 
  17         self.login_as(user=self.user)
  18 
  19         url = reverse(
  20             "sentry-api-0-project-tagkey-values",
  21             kwargs={

  23                 "project_slug": project.slug,
  24                 "key": "foo",
  25             },
  26         )
