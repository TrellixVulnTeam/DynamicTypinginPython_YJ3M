  10 class ProjectKeyStatsTest(APITestCase):
  11     def setUp(self):
  12         self.project = self.create_project()
  13         self.key = ProjectKey.objects.create(project=self.project)
  14         self.login_as(user=self.user)
  15         self.path = u"/api/0/projects/{}/{}/keys/{}/stats/".format(
  16             self.project.organization.slug, self.project.slug, self.key.public_key
  17         )
  18 
  19     def test_simple(self):
  20         tsdb.incr(tsdb.models.key_total_received, self.key.id, count=3)
  21         tsdb.incr(tsdb.models.key_total_blacklisted, self.key.id, count=1)
  22 
  23         response = self.client.get(self.path)
  24         assert response.status_code == 200
  25 
  26         assert response.status_code == 200, response.content
  27         assert response.data[-1]["total"] == 3, response.data
  28         assert response.data[-1]["filtered"] == 1, response.data
  29         for point in response.data[:-1]:
  30             assert point["total"] == 0
  31         assert len(response.data) == 24
  32 
  33     def test_invalid_parameters(self):
  34         url = self.path + "?resolution=1x"
  35         response = self.client.get(url)
  36         assert response.status_code == 400
  37 
  38     # This test can be removed once the TSDB metrics that were stored
  39     # under str(key_id) have expired out of redis.
  40     def test_str_key_id(self):
  41         tsdb.incr(tsdb.models.key_total_received, self.key.id, count=1)
  42         tsdb.incr(tsdb.models.key_total_received, six.text_type(self.key.id), count=1)
  43 
  44         response = self.client.get(self.path)
  45         assert response.status_code == 200
  46 
  47         assert response.status_code == 200, response.content
  48         assert response.data[-1]["total"] == 2, response.data
  49         for point in response.data[:-1]:
  50             assert point["total"] == 0
  51         assert len(response.data) == 24
