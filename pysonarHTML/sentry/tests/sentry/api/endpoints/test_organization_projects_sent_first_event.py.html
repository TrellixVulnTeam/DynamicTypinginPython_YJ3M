  10 class OrganizationProjectsSentFirstEventEndpointTest(APITestCase):
  11     def setUp(self):
  12         self.foo = self.create_user("foo@example.com")
  13         self.org = self.create_organization(owner=self.user)
  14         self.team = self.create_team(organization=self.org)
  15         self.url = reverse(
  16             "sentry-api-0-organization-sent-first-event",
  17             kwargs={"organization_slug": self.org.slug},
  18         )
  19 
  20     def test_simple_sent_first_event(self):
  21         self.create_project(teams=[self.team], first_event=datetime.now())
  22         self.create_member(organization=self.org, user=self.foo, teams=[self.team])
  23 
  24         self.login_as(user=self.foo)
  25 
  26         response = self.client.get(self.url)
  27         assert response.status_code == 200
  28 
  29         assert response.data["sentFirstEvent"]
  30 
  31     def test_simple_no_first_event(self):
