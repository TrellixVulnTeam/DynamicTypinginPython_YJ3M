 133     def test_environments(self):
 134         self.login_as(user=self.user)
 135 
 136         base_url = u"/api/0/projects/{}/{}/user-feedback/".format(
 137             self.project.organization.slug, self.project.slug
 138         )
 139 
 140         # Specify environment
 141         response = self.client.get(base_url + "?environment=production")
 142 
 143         assert response.status_code == 200, response.content
 144         assert len(response.data) == 1
 145         assert response.data[0]["eventID"] == "a" * 32
 146 
 147         # No environment
 148         response = self.client.get(base_url + "?environment=")
 149         assert response.status_code == 200
 150         assert response.data == []
 151 
 152         # All environments
 153         response = self.client.get(base_url)
 154 
 155         assert response.status_code == 200, response.content
 156         assert len(response.data) == 2
 157         assert set([report["eventID"] for report in response.data]) == set(["a" * 32, "b" * 32])
 158 
 159         # Invalid environment
 160         response = self.client.get(base_url + "?environment=invalid_env")
 161         assert response.status_code == 200
 162         assert response.data == []
