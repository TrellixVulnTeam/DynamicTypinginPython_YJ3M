   9 class SentryAppsStatsTest(APITestCase):
  10     def setUp(self):
  11         self.superuser = self.create_user(email="a@example.com", is_superuser=True)
  12         self.user = self.create_user(email="boop@example.com")
  13         self.org = self.create_organization(owner=self.user)
  14         self.super_org = self.create_organization(owner=self.superuser)
  15 
  16         self.app_1 = self.create_sentry_app(
  17             name="Test", organization=self.super_org, published=True
  18         )
  19 
  20         self.app_2 = self.create_sentry_app(name="Testin", organization=self.org)
  21 
  22         self.create_sentry_app_installation(slug=self.app_1.slug, organization=self.org)
  23         self.create_sentry_app_installation(slug=self.app_2.slug, organization=self.org)
  24 
  25         self.url = reverse("sentry-api-0-sentry-apps-stats")
  26 
  27     def test_superuser_has_access(self):
  28         self.login_as(user=self.superuser, superuser=True)
  29 
  30         response = self.client.get(self.url, format="json")
  31 
  32         assert response.status_code == 200
  33         assert {
  34             "id": self.app_2.id,
  35             "slug": self.app_2.slug,
  36             "name": self.app_2.name,
  37             "installs": 1,
  38         } in json.loads(response.content)
  39 
  40         assert {
  41             "id": self.app_1.id,
  42             "slug": self.app_1.slug,
  43             "name": self.app_1.name,
  44             "installs": 1,
  45         } in json.loads(response.content)
  46 
  47     def test_nonsuperusers_have_no_access(self):
  48         self.login_as(user=self.user)
  49 
  50         response = self.client.get(self.url, format="json")
  51 
  52         assert response.status_code == 403
  53 
  54     def test_per_page(self):
  55         self.login_as(user=self.superuser, superuser=True)
  56 
  57         self.create_sentry_app_installation(
  58             slug=self.app_1.slug, organization=self.create_organization()
  59         )
  60 
  61         for i in range(15):
  62             app = self.create_sentry_app(
  63                 name="Test {}".format(i), organization=self.super_org, published=True
  64             )
  65 
  66             self.create_sentry_app_installation(slug=app.slug, organization=self.org)
  67 
  68         response = self.client.get(self.url + "?per_page=10", format="json")
  69         integrations = json.loads(response.content)
  70 
  71         assert len(integrations) == 10  # honors per_page
  72         assert integrations[0]["installs"] == 2  # sorted by installs
