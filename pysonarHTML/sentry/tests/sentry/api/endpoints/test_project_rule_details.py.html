  84 class UpdateProjectRuleTest(APITestCase):
  85     def test_simple(self):
  86         self.login_as(user=self.user)
  87 
  88         project = self.create_project()
  89 
  90         rule = Rule.objects.create(project=project, label="foo")
  91 
  92         conditions = [
  93             {
  94                 "id": "sentry.rules.conditions.first_seen_event.FirstSeenEventCondition",
  95                 "key": "foo",
  96                 "match": "eq",
  97                 "value": "bar",
  98             }
  99         ]
 100 
 101         url = reverse(
 102             "sentry-api-0-project-rule-details",
 103             kwargs={
 104                 "organization_slug": project.organization.slug,
 105                 "project_slug": project.slug,
 106                 "rule_id": rule.id,
 107             },
 108         )
 109         response = self.client.put(
 110             url,
 111             data={
 112                 "name": "hello world",
 113                 "actionMatch": "any",
 114                 "actions": [{"id": "sentry.rules.actions.notify_event.NotifyEventAction"}],
 115                 "conditions": conditions,
 116             },
 117             format="json",
 118         )
 119 
 120         assert response.status_code == 200, response.content
 121         assert response.data["id"] == six.text_type(rule.id)
 122 
 123         rule = Rule.objects.get(id=rule.id)
 124         assert rule.label == "hello world"
 125         assert rule.environment_id is None
 126         assert rule.data["action_match"] == "any"
 127         assert rule.data["actions"] == [
 128             {"id": "sentry.rules.actions.notify_event.NotifyEventAction"}
 129         ]
 130         assert rule.data["conditions"] == conditions
