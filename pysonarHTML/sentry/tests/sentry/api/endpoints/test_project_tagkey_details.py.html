  13 class ProjectTagKeyDetailsTest(APITestCase, SnubaTestCase):
  14     def test_simple(self):
  15         project = self.create_project()
  16 
  17         def make_event(i):
  18             self.store_event(
  19                 data={
  20                     "tags": {"foo": "val{}".format(i)},
  21                     "timestamp": iso_format(before_now(seconds=1)),
  22                 },
  23                 project_id=project.id,
  24             )
  25 
  26         for i in range(0, 16):
  27             make_event(i)
  28 
  29         self.login_as(user=self.user)
  30 
  31         url = reverse(
  32             "sentry-api-0-project-tagkey-details",
  33             kwargs={
  34                 "organization_slug": project.organization.slug,
  35                 "project_slug": project.slug,
  36                 "key": "foo",
  37             },
  38         )
  39 
  40         response = self.client.get(url)
  41 
  42         assert response.status_code == 200
  43         assert response.data["uniqueValues"] == 16
