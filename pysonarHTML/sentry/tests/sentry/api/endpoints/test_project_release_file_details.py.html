  49 
  50         from six import BytesIO
  51 
  52         f = File.objects.create(name="applicatiosn.js", type="release.file")
  53         f.putfile(BytesIO("File contents here"))
  54 
  55         releasefile = ReleaseFile.objects.create(
  56             organization_id=project.organization_id,
  57             project_id=project.id,
  58             release=release,
  59             file=f,
  60             name="  http://example.com/appli\n\rcatios n.js\n\n\r  ",
  61         )
  62 
  63         url = reverse(
  64             "sentry-api-0-project-release-file-details",
  65             kwargs={
  66                 "organization_slug": project.organization.slug,
  67                 "project_slug": project.slug,
  68                 "version": release.version,
  69                 "file_id": releasefile.id,
  70             },
  71         )
  72 
  73         response = self.client.get(url + "?download=1")
  74         assert response.status_code == 200, response.content
  75         assert response.get("Content-Disposition") == 'attachment; filename="appli catios n.js"'
  76         assert response.get("Content-Length") == six.text_type(f.size)
  77         assert response.get("Content-Type") == "application/octet-stream"
  78         assert "File contents here" == BytesIO(b"".join(response.streaming_content)).getvalue()
  79 
  80         user_no_permission = self.create_user("baz@localhost", username="baz")
  81         self.login_as(user=user_no_permission)
  82         response = self.client.get(url + "?download=1")
  83         assert response.status_code == 403, response.content
