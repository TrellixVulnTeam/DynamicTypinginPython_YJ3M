  61     def test_assemble_check(self):
  62         content = "foo bar".encode("utf-8")
  63         fileobj = ContentFile(content)
  64         file1 = File.objects.create(name="baz.dSYM", type="default", size=7)
  65         file1.putfile(fileobj, 3)
  66         checksum = sha1(content).hexdigest()
  67 
  68         blobs = FileBlob.objects.all()
  69         checksums = []
  70         for blob in blobs:
  71             checksums.append(blob.checksum)
  72 
  73         # Request to see of file is there
  74         # file exists but we have no overship for the chunks
  75         response = self.client.post(
  76             self.url,
  77             data={checksum: {"name": "dif", "chunks": checksums}},
  78             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
  79         )
  80 
  81         assert response.status_code == 200, response.content
  82         assert response.data[checksum]["state"] == ChunkFileState.NOT_FOUND
  83         assert set(response.data[checksum]["missingChunks"]) == set(checksums)
  84 
  85         # Now we add ownership to the blob
  86         blobs = FileBlob.objects.all()
  87         for blob in blobs:
  88             FileBlobOwner.objects.create(blob=blob, organization=self.organization)
  89 
  90         # The request will start the job to assemble the file
  91         response = self.client.post(
  92             self.url,
  93             data={checksum: {"name": "dif", "chunks": checksums}},
  94             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
  95         )
  96 
  97         assert response.status_code == 200, response.content
  98         assert response.data[checksum]["state"] == ChunkFileState.CREATED
  99         assert response.data[checksum]["missingChunks"] == []
 100 
 101         # Finally, we simulate a successful job
 102         ProjectDebugFile.objects.create(
 103             file=file1,
 104             object_name="baz.dSYM",
 105             cpu_name="x86_64",
 106             project=self.project,
 107             debug_id="df449af8-0dcd-4320-9943-ec192134d593",
 108             code_id="DF449AF80DCD43209943EC192134D593",
 109         )
 110         set_assemble_status(AssembleTask.DIF, self.project.id, checksum, None)
 111 
 112         # Request now tells us that everything is alright
 113         response = self.client.post(
 114             self.url,
 115             data={checksum: {"name": "dif", "chunks": checksums}},
 116             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 117         )
 118 
 119         assert response.status_code == 200, response.content
 120         assert response.data[checksum]["state"] == ChunkFileState.OK
 121         assert response.data[checksum]["missingChunks"] == []
 122 
 123         not_found_checksum = sha1("1").hexdigest()
 124 
 125         response = self.client.post(
 126             self.url,
 127             data={not_found_checksum: {"name": "dif", "chunks": [not_found_checksum]}},
 128             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 129         )
 130 
 131         assert response.status_code == 200, response.content
 132         assert response.data[not_found_checksum]["state"] == ChunkFileState.NOT_FOUND
 133         assert set(response.data[not_found_checksum]["missingChunks"]) == set([not_found_checksum])

 135     @patch("sentry.tasks.assemble.assemble_dif")
 136     def test_assemble(self, mock_assemble_dif):
 137         content1 = "foo".encode("utf-8")
 138         fileobj1 = ContentFile(content1)
 139         checksum1 = sha1(content1).hexdigest()
 140 
 141         content2 = "bar".encode("utf-8")
 142         fileobj2 = ContentFile(content2)
 143         checksum2 = sha1(content2).hexdigest()
 144 
 145         content3 = "baz".encode("utf-8")
 146         fileobj3 = ContentFile(content3)
 147         checksum3 = sha1(content3).hexdigest()
 148 
 149         total_checksum = sha1(content2 + content1 + content3).hexdigest()
 150 
 151         # The order here is on purpose because we check for the order of checksums
 152         blob1 = FileBlob.from_file(fileobj1)
 153         FileBlobOwner.objects.get_or_create(organization=self.organization, blob=blob1)
 154         blob3 = FileBlob.from_file(fileobj3)
 155         FileBlobOwner.objects.get_or_create(organization=self.organization, blob=blob3)
 156         blob2 = FileBlob.from_file(fileobj2)
 157 
 158         # we make a request now but we are missing ownership for chunk 2
 159         response = self.client.post(
 160             self.url,
 161             data={total_checksum: {"name": "test", "chunks": [checksum2, checksum1, checksum3]}},
 162             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 163         )
 164         assert response.status_code == 200, response.content
 165         assert response.data[total_checksum]["state"] == ChunkFileState.NOT_FOUND
 166         assert response.data[total_checksum]["missingChunks"] == [checksum2]
 167 
 168         # we add ownership to chunk 2
 169         FileBlobOwner.objects.get_or_create(organization=self.organization, blob=blob2)
 170 
 171         # new request, ownership for all chunks is there but file does not exist yet
 172         response = self.client.post(
 173             self.url,
 174             data={total_checksum: {"name": "test", "chunks": [checksum2, checksum1, checksum3]}},
 175             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 176         )
 177         assert response.status_code == 200, response.content
 178         assert response.data[total_checksum]["state"] == ChunkFileState.CREATED
 179         assert response.data[total_checksum]["missingChunks"] == []
 180 
 181         chunks = [checksum2, checksum1, checksum3]
 182         mock_assemble_dif.apply_async.assert_called_once_with(
 183             kwargs={
 184                 "project_id": self.project.id,
 185                 "name": "test",
 186                 "chunks": chunks,
 187                 "checksum": total_checksum,
 188                 "debug_id": None,
 189             }
 190         )
 191 
 192         file = assemble_file(
 193             AssembleTask.DIF, self.project, "test", total_checksum, chunks, "project.dif"
 194         )[0]
 195         status, _ = get_assemble_status(AssembleTask.DIF, self.project.id, total_checksum)
 196         assert status != ChunkFileState.ERROR
 197         assert file.checksum == total_checksum
 198 
 199         file_blob_index = FileBlobIndex.objects.all()
 200         assert len(file_blob_index) == 3

 202     def test_dif_response(self):
 203         sym_file = self.load_fixture("crash.sym")
 204         blob1 = FileBlob.from_file(ContentFile(sym_file))
 205         total_checksum = sha1(sym_file).hexdigest()
 206         chunks = [blob1.checksum]
 207 
 208         assemble_dif(
 209             project_id=self.project.id, name="crash.sym", checksum=total_checksum, chunks=chunks
 210         )
 211 
 212         response = self.client.post(
 213             self.url,
 214             data={total_checksum: {"name": "test.sym", "chunks": chunks}},
 215             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 216         )
 217 
 218         assert response.status_code == 200, response.content

 221         assert (
 222             response.data[total_checksum]["dif"]["uuid"] == "67e9247c-814e-392b-a027-dbde6748fcbf"
