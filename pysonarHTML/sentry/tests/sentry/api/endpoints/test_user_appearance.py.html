   9 class UserAppearanceTest(APITestCase):

  11         self.user = self.create_user(email="a@example.com")
  12         self.login_as(user=self.user)
  13         self.url = reverse("sentry-api-0-user-appearance", kwargs={"user_id": self.user.id})
  14 
  15     def test_default_options(self):
  16         resp = self.client.get(self.url)
  17 
  18         assert resp.status_code == 200, resp.content
  19         assert resp.data["timezone"] == "UTC"
  20         assert resp.data["stacktraceOrder"] == -1
  21         assert resp.data["language"] == "en"
  22         assert not resp.data["clock24Hours"]
  23 
  24     def test_update(self):

  26             self.url,
  27             data={
  28                 "timezone": "UTC",
  29                 "stacktraceOrder": "2",
  30                 "language": "fr",
  31                 "clock24Hours": True,
  32                 "extra": True,
  33             },
  34         )
  35 
  36         assert resp.status_code == 204
  37 
  38         assert UserOption.objects.get_value(user=self.user, key="timezone") == "UTC"
  39         assert UserOption.objects.get_value(user=self.user, key="stacktrace_order") == "2"
  40         assert UserOption.objects.get_value(user=self.user, key="language") == "fr"
