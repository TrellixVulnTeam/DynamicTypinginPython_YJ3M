 155 class OrganizationRepositoriesCreateTest(APITestCase):
 156     def test_simple(self):
 157         self.login_as(user=self.user)
 158 
 159         org = self.create_organization(owner=self.user, name="baz")
 160 
 161         with patch.object(DummyRepositoryProvider, "needs_auth", return_value=False):
 162             url = reverse("sentry-api-0-organization-repositories", args=[org.slug])
 163             response = self.client.post(url, data={"provider": "dummy", "name": "getsentry/sentry"})
 164 
 165         assert response.status_code == 201, (response.status_code, response.content)
 166         assert response.data["id"]
 167 
 168         repo = Repository.objects.get(id=response.data["id"])
 169         assert repo.provider == "dummy"
 170         assert repo.name == "getsentry/sentry"
 171 
 172     def test_admin_ok(self):

 174         team = self.create_team(name="people", organization=org)

 176         user = self.create_user(email="admin@example.org")

 182             url = reverse("sentry-api-0-organization-repositories", args=[org.slug])
