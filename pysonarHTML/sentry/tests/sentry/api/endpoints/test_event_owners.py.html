 198     def test_owners_of_different_types_ordered_correctly(self):
 199         owners = [self.user, self.team3, self.user2, self.team2, self.user3, self.team]
 200         rules = [
 201             Rule(Matcher("path", "*.py"), [Owner("user", owners[0].email)]),
 202             Rule(Matcher("path", "*py"), [Owner("team", owners[1].slug)]),
 203             Rule(Matcher("path", "*foo*"), [Owner("user", owners[2].email)]),
 204             Rule(Matcher("path", "*y"), [Owner("team", owners[3].slug)]),
 205             Rule(Matcher("path", "*"), [Owner("user", owners[4].email)]),
 206             Rule(Matcher("path", "*o.py"), [Owner("team", owners[5].slug)]),
 207         ]
 208 
 209         ProjectOwnership.objects.create(
 210             project_id=self.project.id, schema=dump_schema(rules), fallthrough=True
 211         )
 212 
 213         event1 = self.store_event(
 214             data={"stacktrace": {"frames": [{"filename": "foo.py"}]}}, project_id=self.project.id
 215         )
 216 
 217         self.path = reverse(
 218             "sentry-api-0-event-owners",
 219             kwargs={
 220                 "organization_slug": self.organization.slug,
 221                 "project_slug": self.project.slug,
 222                 "event_id": event1.event_id,
 223             },
 224         )
 225 
 226         resp = self.client.get(self.path)
 227         assert resp.status_code == 200
 228         assert len(resp.data["owners"]) == 6
 229         assert [o["id"] for o in resp.data["owners"]] == [six.text_type(o.id) for o in owners]
 230         assert [o["type"] for o in resp.data["owners"]] == ["user", "team"] * 3
 231         assert resp.data["rule"] == Matcher("path", "*.py")
 232         assert len(resp.data["rules"]) == 6
