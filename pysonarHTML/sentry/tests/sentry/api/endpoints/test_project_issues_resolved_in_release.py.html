  17     def setUp(self):

  28     def build_grouplink(self):
  29         repo = Repository.objects.create(organization_id=self.org.id, name=self.project.name)
  30         commit = Commit.objects.create(
  31             organization_id=self.org.id, repository_id=repo.id, key=uuid1().hex
  32         )
  33         commit2 = Commit.objects.create(
  34             organization_id=self.org.id, repository_id=repo.id, key=uuid1().hex
  35         )
  36         ReleaseCommit.objects.create(
  37             organization_id=self.org.id, release=self.release, commit=commit, order=1
  38         )
  39         ReleaseCommit.objects.create(
  40             organization_id=self.org.id, release=self.release, commit=commit2, order=0
  41         )
  42         GroupLink.objects.create(
  43             group_id=self.group.id,
  44             project_id=self.group.project_id,
  45             linked_type=GroupLink.LinkedType.commit,
  46             relationship=GroupLink.Relationship.resolves,
  47             linked_id=commit.id,
  48         )

  50     def build_group_resolution(self, group=None):
  51         return GroupResolution.objects.create(
  52             group=self.group if group is None else group,
  53             release=self.release,
  54             type=GroupResolution.Type.in_release,
  55         )

  57     def run_test(self, expected_groups):
  58         response = self.get_valid_response(self.org.slug, self.project.slug, self.release.version)
  59         assert len(response.data) == len(expected_groups)
  60         expected = set(map(six.text_type, [g.id for g in expected_groups]))
  61         assert set([item["id"] for item in response.data]) == expected
