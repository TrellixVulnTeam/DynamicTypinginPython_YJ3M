  12 class ReleaseFilesListTest(APITestCase):
  13     def test_simple(self):
  14         project = self.create_project(name="foo")
  15 
  16         release = Release.objects.create(organization_id=project.organization_id, version="1")
  17         release.add_project(project)
  18 
  19         releasefile = ReleaseFile.objects.create(
  20             organization_id=project.organization_id,
  21             release=release,
  22             file=File.objects.create(name="application.js", type="release.file"),
  23             name="http://example.com/application.js",
  24         )
  25 
  26         url = reverse(
  27             "sentry-api-0-organization-release-files",
  28             kwargs={"organization_slug": project.organization.slug, "version": release.version},
  29         )
  30 
  31         self.login_as(user=self.user)
  32 
  33         response = self.client.get(url)
  34 
  35         assert response.status_code == 200, response.content
  36         assert len(response.data) == 1
  37         assert response.data[0]["id"] == six.text_type(releasefile.id)

  40 class ReleaseFileCreateTest(APITestCase):
  41     def test_simple(self):
  42         project = self.create_project(name="foo")
  43 
  44         release = Release.objects.create(organization_id=project.organization_id, version="1")
  45         release.add_project(project)
  46 
  47         url = reverse(
  48             "sentry-api-0-organization-release-files",
  49             kwargs={"organization_slug": project.organization.slug, "version": release.version},
  50         )
  51 
  52         self.login_as(user=self.user)
  53 
  54         response = self.client.post(
  55             url,
  56             {
  57                 "name": "http://example.com/application.js",
  58                 "header": "X-SourceMap: http://example.com",
  59                 "file": SimpleUploadedFile(
  60                     "application.js", b"function() { }", content_type="application/javascript"
  61                 ),
  62             },
  63             format="multipart",
  64         )
  65 
  66         assert response.status_code == 201, response.content
  67 
  68         releasefile = ReleaseFile.objects.get(release=release)
  69         assert releasefile.name == "http://example.com/application.js"
  70         assert releasefile.ident == ReleaseFile.get_ident("http://example.com/application.js")
  71         assert releasefile.file.headers == {
  72             "Content-Type": "application/javascript",
  73             "X-SourceMap": "http://example.com",
  74         }

 189     def test_duplicate_file(self):
 190         project = self.create_project(name="foo")
 191 
 192         release = Release.objects.create(
 193             project_id=project.id, organization_id=project.organization_id, version="1"
 194         )
 195         release.add_project(project)
 196 
 197         url = reverse(
 198             "sentry-api-0-organization-release-files",
 199             kwargs={"organization_slug": project.organization.slug, "version": release.version},
 200         )
 201 
 202         self.login_as(user=self.user)
 203 
 204         data = {
 205             "name": "http://example.com/application.js",
 206             "header": "X-SourceMap: http://example.com",
 207             "file": SimpleUploadedFile(
 208                 "application.js", b"function() { }", content_type="application/javascript"
 209             ),
 210         }
 211 
 212         response = self.client.post(url, data, format="multipart")
 213 
 214         assert response.status_code == 201, response.content
 215 
 216         releasefile = ReleaseFile.objects.get(release=release)
 217         assert releasefile.name == "http://example.com/application.js"
 218         assert releasefile.file.headers == {
 219             "Content-Type": "application/javascript",
 220             "X-SourceMap": "http://example.com",
 221         }
 222 
 223         # Now upload it again!
 224         response = self.client.post(url, data, format="multipart")
 225 
 226         assert response.status_code == 409, response.content
