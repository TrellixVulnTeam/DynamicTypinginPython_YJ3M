  60     def test_with_mentions(self):
  61         user = self.create_user(email="hello@meow.com")
  62 
  63         self.org = self.create_organization(name="Gnarly Org", owner=None)
  64         self.team = self.create_team(organization=self.org, name="Ultra Rad Team")
  65 
  66         # member that IS NOT part of the team
  67         self.create_member(user=user, organization=self.org, role="member", teams=[])
  68         # member that IS part of the team
  69         self.create_member(user=self.user, organization=self.org, role="member", teams=[self.team])
  70         group = self.group
  71 
  72         self.login_as(user=self.user)
  73 
  74         url = u"/api/0/issues/{}/comments/".format(group.id)
  75 
  76         # mentioning a member that does not exist returns 400
  77         response = self.client.post(
  78             url,
  79             format="json",
  80             data={"text": "**meredith@getsentry.com** is fun", "mentions": [u"8"]},
  81         )
  82         assert response.status_code == 400, response.content
  83 
  84         user_id = six.text_type(self.user.id)
  85 
  86         # mentioning a member in the correct team returns 201
  87         response = self.client.post(
  88             url,
  89             format="json",
  90             data={"text": "**meredith@getsentry.com** is so fun", "mentions": [u"%s" % user_id]},
  91         )
  92         assert response.status_code == 201, response.content
  93 
  94         user_id = six.text_type(user.id)
  95 
  96         # mentioning a member that exists but NOT in the team returns
  97         # validation error
  98         response = self.client.post(
  99             url,
 100             format="json",
 101             data={"text": "**hello@meow.com** is not so fun", "mentions": [u"%s" % user_id]},
 102         )
 103 
 104         assert response.data == {"mentions": ["Cannot mention a non team member"]}

 106     def test_with_team_mentions(self):
 107         user = self.create_user(email="redTeamUser@example.com")
 108 
 109         self.org = self.create_organization(name="Gnarly Org", owner=None)
 110         # team that IS part of the project
 111         self.team = self.create_team(organization=self.org, name="Red Team", members=[user])
 112         # team that IS NOT part of the project
 113         self.team2 = self.create_team(organization=self.org, name="Blue Team")
 114 
 115         self.create_member(user=self.user, organization=self.org, role="member", teams=[self.team])
 116 
 117         group = self.group
 118 
 119         self.login_as(user=self.user)
 120 
 121         url = u"/api/0/issues/{}/comments/".format(group.id)
 122 
 123         # mentioning a team that does not exist returns 400
 124         response = self.client.post(
 125             url,
 126             format="json",
 127             data={
 128                 "text": "hey **blue-team** fix this bug",
 129                 "mentions": [u"team:%s" % self.team2.id],
 130             },
 131         )
 132         assert response.status_code == 400, response.content
 133 
 134         assert response.data == {
 135             "mentions": ["Mentioned team not found or not associated with project"]
 136         }
 137 
 138         # mentioning a team in the project returns 201
 139         response = self.client.post(
 140             url,
 141             format="json",
 142             data={"text": "hey **red-team** fix this bug", "mentions": [u"team:%s" % self.team.id]},
 143         )
 144         assert response.status_code == 201, response.content
 145         assert (
 146             len(
 147                 GroupSubscription.objects.filter(
 148                     group=group, reason=GroupSubscriptionReason.team_mentioned
 149                 )
 150             )
 151             == 1
 152         )

 154     def test_with_group_link(self):
 155         group = self.group
 156 
 157         integration = Integration.objects.create(provider="example", external_id="123456")
 158         integration.add_organization(group.organization, self.user)
 159 
 160         OrganizationIntegration.objects.filter(
 161             integration_id=integration.id, organization_id=group.organization.id
 162         ).update(
 163             config={
 164                 "sync_comments": True,
 165                 "sync_status_outbound": True,
 166                 "sync_status_inbound": True,
 167                 "sync_assignee_outbound": True,
 168                 "sync_assignee_inbound": True,
 169             }
 170         )
 171 
 172         external_issue = ExternalIssue.objects.create(
 173             organization_id=group.organization.id, integration_id=integration.id, key="APP-123"
 174         )
 175 
 176         GroupLink.objects.create(
 177             group_id=group.id,
 178             project_id=group.project_id,
 179             linked_type=GroupLink.LinkedType.issue,
 180             linked_id=external_issue.id,
 181             relationship=GroupLink.Relationship.references,
 182         )
 183 
 184         self.user.name = "Sentry Admin"
 185         self.user.save()
 186         self.login_as(user=self.user)
 187 
 188         url = u"/api/0/issues/{}/comments/".format(group.id)
 189 
 190         with self.feature({"organizations:integrations-issue-sync": True}):
 191             with self.tasks():
 192                 comment = "hello world"
 193                 response = self.client.post(url, format="json", data={"text": comment})
 194                 assert response.status_code == 201, response.content
 195 
 196                 activity = Activity.objects.get(id=response.data["id"])
 197                 assert activity.user == self.user
 198                 assert activity.group == group
 199                 assert activity.data == {"text": comment, "external_id": "123456789"}
