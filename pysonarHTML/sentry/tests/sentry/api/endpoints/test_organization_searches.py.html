 122     def test_pinned(self):
 123         included = self.create_base_data()
 124         pinned_query = SavedSearch.objects.create(
 125             organization=self.organization,
 126             owner=self.user,
 127             name="My Pinned Query",
 128             query="pinned junk",
 129             date_added=timezone.now(),
 130         )
 131         included.append(pinned_query)
 132         self.check_results(included)
 133         # Check a pinned query that uses an existing query correctly filters
 134         # the existing query
 135         to_be_pinned = included.pop()
 136         to_be_pinned.is_pinned = True
 137         pinned_query.query = to_be_pinned.query
 138         pinned_query.save()
 139         included[0] = to_be_pinned
 140         self.check_results(included)

 143 class CreateOrganizationSearchesTest(APITestCase):
 144     endpoint = "sentry-api-0-organization-searches"
 145     method = "post"
 146 
 147     @fixture
 148     def manager(self):
 149         user = self.create_user("test@test.com")
 150         self.create_member(organization=self.organization, user=user, role="manager")
 151         return user
 152 
 153     @fixture
 154     def member(self):
 155         user = self.create_user("test@test.com")
 156         self.create_member(organization=self.organization, user=user)
 157         return user
 158 
 159     def test_simple(self):
 160         search_type = SearchType.ISSUE.value
 161         name = "test"
 162         query = "hello"
 163         self.login_as(user=self.manager)
 164         resp = self.get_valid_response(
 165             self.organization.slug, type=search_type, name=name, query=query
 166         )
 167         assert resp.data["name"] == name
 168         assert resp.data["query"] == query
 169         assert resp.data["type"] == search_type
 170         assert SavedSearch.objects.filter(id=resp.data["id"]).exists()
 171 
 172     def test_perms(self):
 173         self.login_as(user=self.member)
 174         resp = self.get_response(
 175             self.organization.slug, type=SearchType.ISSUE.value, name="hello", query="test"
 176         )
 177         assert resp.status_code == 403
 178 
 179     def test_exists(self):
 180         global_search = SavedSearch.objects.create(
 181             type=SearchType.ISSUE.value,
 182             name="Some global search",
 183             query="is:unresolved",
 184             is_global=True,
 185         )
 186         self.login_as(user=self.manager)
 187         resp = self.get_response(
 188             self.organization.slug,
 189             type=SearchType.ISSUE.value,
 190             name="hello",
 191             query=global_search.query,
 192         )
 193         assert resp.status_code == 400
 194         assert "already exists" in resp.data["detail"]
 195 
 196         org_search = SavedSearch.objects.create(
 197             organization=self.organization,
 198             type=SearchType.ISSUE.value,
 199             name="Some org search",
 200             query="org search",
 201         )
 202         resp = self.get_response(
 203             self.organization.slug,
 204             type=SearchType.ISSUE.value,
 205             name="hello",
 206             query=org_search.query,
 207         )
 208         assert resp.status_code == 400
 209         assert "already exists" in resp.data["detail"]
