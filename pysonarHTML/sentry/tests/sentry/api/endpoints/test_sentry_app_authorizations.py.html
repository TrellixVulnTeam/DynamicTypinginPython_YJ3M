  38     def _run_request(self, *args, **kwargs):
  39         data = {
  40             "client_id": self.sentry_app.application.client_id,
  41             "client_secret": self.sentry_app.application.client_secret,
  42             "grant_type": "authorization_code",
  43             "code": self.install.api_grant.code,
  44         }
  45         data.update(**kwargs)
  46         return self.client.post(self.url, data, headers={"Content-Type": "application/json"})

  48     def test_exchanges_for_token_successfully(self):
  49         expected_expires_at = (timezone.now() + timedelta(hours=8)).replace(second=0, microsecond=0)
  50 
  51         response = self._run_request()
  52 
  53         token = ApiToken.objects.get(application=self.sentry_app.application)
  54 
  55         assert response.status_code == 201, response.content
  56         assert response.data["scopes"] == self.sentry_app.scope_list
  57         assert response.data["token"] == token.token
  58         assert response.data["refreshToken"] == token.refresh_token
  59 
  60         expires_at = response.data["expiresAt"].replace(second=0, microsecond=0)
  61 
  62         assert expires_at == expected_expires_at
