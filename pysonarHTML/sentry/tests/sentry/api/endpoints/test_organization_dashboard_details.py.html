  89         return reverse(
  90             "sentry-api-0-organization-dashboard-details",

  94     def sort_by_order(self, widgets):
  95         def get_order(x):
  96             try:
  97                 return x["order"]
  98             except TypeError:
  99                 return x.order
 100 
 101         return sorted(widgets, key=get_order)

 125         response = self.client.get(self.url(self.dashboard.id))
 126         assert response.status_code == 200, response.content
 127 
 128         self.assert_dashboard(response.data, self.dashboard)

 130         widgets = self.sort_by_order(response.data["widgets"])
 131         self.assert_widget(widgets[0], self.widget_1)
 132         self.assert_widget(widgets[1], self.widget_2)
 133 
 134         widget_1_data_sources = self.sort_by_order(widgets[0]["dataSources"])
 135         assert len(widget_1_data_sources) == 2
 136         self.assert_widget_data_source(widget_1_data_sources[0], self.widget_1_data_1)
 137         self.assert_widget_data_source(widget_1_data_sources[1], self.widget_1_data_2)
 138 
 139         assert len(widgets[1]["dataSources"]) == 1
 140         self.assert_widget_data_source(widgets[1]["dataSources"][0], self.widget_2_data_1)

 160 class OrganizationDashboardDetailsPutTest(OrganizationDashboardDetailsTestCase):
 161     def setUp(self):
 162         super(OrganizationDashboardDetailsPutTest, self).setUp()
 163         self.widget_3 = Widget.objects.create(
 164             dashboard=self.dashboard,
 165             order=3,
 166             title="Widget 3",
 167             display_type=WidgetDisplayTypes.LINE_CHART,
 168         )
 169         self.widget_4 = Widget.objects.create(
 170             dashboard=self.dashboard,
 171             order=4,
 172             title="Widget 4",
 173             display_type=WidgetDisplayTypes.LINE_CHART,
 174         )
 175         self.widget_ids = [self.widget_1.id, self.widget_2.id, self.widget_3.id, self.widget_4.id]
 176 
 177     def assert_no_changes(self):
 178         self.assert_dashboard_and_widgets(self.widget_ids, [1, 2, 3, 4])
 179 
 180     def assert_dashboard_and_widgets(self, widget_ids, order):
 181         assert Dashboard.objects.filter(
 182             organization=self.organization, id=self.dashboard.id
 183         ).exists()
 184 
 185         widgets = self.sort_by_order(
 186             Widget.objects.filter(dashboard_id=self.dashboard.id, status=ObjectStatus.VISIBLE)
 187         )
 188         assert len(widgets) == len(list(widget_ids))
 189 
 190         for widget, id, order in zip(widgets, widget_ids, order):
 191             assert widget.id == id
 192             assert widget.order == order

 194     def test_put(self):
 195         response = self.client.put(
 196             self.url(self.dashboard.id),
 197             data={
 198                 "title": "Changed the title",

 204                 ],
 205             },
 206         )

 253     def test_partial_reordering_deletes_widgets(self):

 268         assert not WidgetDataSource.objects.filter(widget_id__in=deleted_widget_ids).exists()
