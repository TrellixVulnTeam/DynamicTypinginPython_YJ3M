  11 class SentryAppInstallationExternalRequestsEndpointTest(APITestCase):
  12     def setUp(self):
  13         self.user = self.create_user(email="boop@example.com")
  14         self.org = self.create_organization(owner=self.user)
  15         self.project = self.create_project(organization=self.org)
  16 
  17         self.sentry_app = self.create_sentry_app(
  18             name="Testin", organization=self.org, webhook_url="https://example.com"
  19         )
  20 
  21         self.install = self.create_sentry_app_installation(
  22             organization=self.org, slug=self.sentry_app.slug, user=self.user
  23         )
  24 
  25         self.url = reverse(
  26             "sentry-api-0-sentry-app-installation-external-requests", args=[self.install.uuid]
  27         )
  28 
  29     @responses.activate
  30     def test_makes_external_request(self):
  31         self.login_as(user=self.user)
  32         options = [{"label": "Project Name", "value": "1234"}]
  33         responses.add(
  34             method=responses.GET,
  35             url=u"https://example.com/get-projects?projectSlug={}&installationId={}&query=proj".format(
  36                 self.project.slug, self.install.uuid
  37             ),
  38             json=options,
  39             status=200,
  40             content_type="application/json",
  41             match_querystring=True,
  42         )
  43         url = self.url + u"?projectId={}&uri={}&query={}".format(
  44             self.project.id, "/get-projects", "proj"
  45         )
  46         response = self.client.get(url, format="json")
  47         assert response.status_code == 200
  48         assert response.data == {"choices": [["1234", "Project Name"]]}

  50     @responses.activate
  51     def test_makes_external_request_with_dependent_data(self):
  52         self.login_as(user=self.user)
  53         options = [{"label": "Project Name", "value": "1234"}]
  54         query = {
  55             "projectSlug": self.project.slug,
  56             "installationId": self.install.uuid,
  57             "query": "proj",
  58             "dependentData": json.dumps({"org_id": "A"}),
  59         }
  60         responses.add(
  61             method=responses.GET,
  62             url=u"https://example.com/get-projects?%s" % urlencode(query),
  63             json=options,
  64             status=200,
  65             content_type="application/json",
  66             match_querystring=True,
  67         )
  68         query = {
  69             "projectId": self.project.id,
  70             "uri": "/get-projects",
  71             "query": "proj",
  72             "dependentData": json.dumps({"org_id": "A"}),
  73         }
  74         url = u"%s?%s" % (self.url, urlencode(query))
  75         response = self.client.get(url, format="json")
  76         assert response.status_code == 200
  77         assert response.data == {"choices": [["1234", "Project Name"]]}
