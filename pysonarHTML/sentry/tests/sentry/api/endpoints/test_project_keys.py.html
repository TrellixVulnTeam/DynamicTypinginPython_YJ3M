   9 class ListProjectKeysTest(APITestCase):
  10     def test_simple(self):
  11         project = self.create_project()
  12         key = ProjectKey.objects.get_or_create(project=project)[0]
  13         self.login_as(user=self.user)
  14         url = reverse(
  15             "sentry-api-0-project-keys",
  16             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
  17         )
  18         response = self.client.get(url)
  19         assert response.status_code == 200
  20         assert len(response.data) == 1
  21         assert response.data[0]["public"] == key.public_key
