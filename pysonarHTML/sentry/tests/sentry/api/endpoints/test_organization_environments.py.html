  20     def test_simple(self):

  28     def test_visibility(self):
  29         visible = self.create_environment(name="visible", project=self.project, is_hidden=False)
  30         hidden = self.create_environment(name="not visible", project=self.project, is_hidden=True)
  31         not_set = self.create_environment(name="null visible", project=self.project)
  32         response = self.get_valid_response(self.project.organization.slug, visibility="visible")
  33         assert response.data == serialize([not_set, visible])
  34         response = self.get_valid_response(self.project.organization.slug, visibility="hidden")
  35         assert response.data == serialize([hidden])
  36         response = self.get_valid_response(self.project.organization.slug, visibility="all")
  37         assert response.data == serialize([hidden, not_set, visible])
