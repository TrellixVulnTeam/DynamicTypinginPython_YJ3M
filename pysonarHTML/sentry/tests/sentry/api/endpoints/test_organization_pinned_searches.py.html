 113 class DeleteOrganizationPinnedSearchTest(APITestCase):
 114     endpoint = "sentry-api-0-organization-pinned-searches"
 115     method = "delete"
 116 
 117     @fixture
 118     def member(self):
 119         user = self.create_user("test@test.com")
 120         self.create_member(organization=self.organization, user=user)
 121         return user
 122 
 123     def get_response(self, *args, **params):
 124         return super(DeleteOrganizationPinnedSearchTest, self).get_response(
 125             *((self.organization.slug,) + args), **params
 126         )
 127 
 128     def test(self):
 129         saved_search = SavedSearch.objects.create(
 130             organization=self.organization,
 131             owner=self.member,
 132             type=SearchType.ISSUE.value,
 133             query="wat",
 134         )
 135         other_saved_search = SavedSearch.objects.create(
 136             organization=self.organization,
 137             owner=self.user,
 138             type=SearchType.ISSUE.value,
 139             query="wat",
 140         )
 141 
 142         self.login_as(self.member)
 143         self.get_valid_response(type=saved_search.type, status_code=204)
 144         assert not SavedSearch.objects.filter(id=saved_search.id).exists()
 145         assert SavedSearch.objects.filter(id=other_saved_search.id).exists()
 146 
 147         # Test calling multiple times works ok, doesn't cause other rows to
 148         # delete
 149         self.get_valid_response(type=saved_search.type, status_code=204)
 150         assert SavedSearch.objects.filter(id=other_saved_search.id).exists()
 151
