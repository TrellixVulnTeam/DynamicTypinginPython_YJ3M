  38 class GetSentryAppStatsTest(SentryAppStatsTest):
  39     def test_superuser_sees_unowned_published_stats(self):
  40         self.login_as(user=self.superuser, superuser=True)
  41 
  42         url = reverse("sentry-api-0-sentry-app-stats", args=[self.unowned_published_app.slug])
  43         response = self.client.get(url, format="json")
  44         assert response.status_code == 200
  45         assert response.data["totalInstalls"] == 1
  46         assert response.data["totalUninstalls"] == 0
  47         install_epoch = int(
  48             to_timestamp(
  49                 self.unowned_published_app_install.date_added.replace(
  50                     microsecond=0, second=0, minute=0
  51                 )
  52             )
  53         )
  54         assert (install_epoch, 1) in response.data["installStats"]
  55
