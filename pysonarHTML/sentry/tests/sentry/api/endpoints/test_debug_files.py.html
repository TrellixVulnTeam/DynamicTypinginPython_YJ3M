 192 
 193         (dsym,) = response.data

 198         assert dsym["symbolType"] == "proguard"
 199         assert dsym["uuid"] == "6dc7fdb0-d2fb-4c8e-9d6b-bb1aa98929b1"
 200         download_id = dsym["id"]
 201 
 202         # Test download
 203         response = self.client.get(url + "?id=" + download_id)
 204 
 205         assert response.status_code == 200, response.content
 206         assert (
 207             response.get("Content-Disposition")

 212         assert PROGUARD_SOURCE == BytesIO(b"".join(response.streaming_content)).getvalue()

 237     def test_dsyms_search(self):
 238         project = self.create_project(name="foo")
 239 
 240         url = reverse(
 241             "sentry-api-0-dsym-files",
 242             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
 243         )
 244 
 245         self.login_as(user=self.user)
 246 
 247         first_uuid = None
 248         last_uuid = None
 249         for i in range(25):
 250             last_uuid = text_type(uuid4())
 251             if first_uuid is None:
 252                 first_uuid = last_uuid
 253             self._upload_proguard(url, last_uuid)
 254 
 255         # Test max 20 per page
 256         response = self.client.get(url)
 257         assert response.status_code == 200, response.content
 258         dsyms = response.data
 259         assert len(dsyms) == 20
 260 
 261         # Test should return last
 262         response = self.client.get(url + "?query=" + last_uuid)
 263         assert response.status_code == 200, response.content
 264         dsyms = response.data
 265         assert len(dsyms) == 1
 266 
 267         response = self.client.get(url + "?query=proguard")
 268         assert response.status_code == 200, response.content
 269         dsyms = response.data
 270         assert len(dsyms) == 20

 272     def test_source_maps(self):

 286             organization_id=project.organization_id,
 287             release=release,
 288             file=File.objects.create(name="application2.js", type="release.file"),
