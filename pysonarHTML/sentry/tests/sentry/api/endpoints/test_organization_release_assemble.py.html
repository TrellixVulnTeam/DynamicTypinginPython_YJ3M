  54     @patch("sentry.tasks.assemble.assemble_artifacts")
  55     def test_assemble(self, mock_assemble_artifacts):
  56         bundle_file = self.create_artifact_bundle()
  57         total_checksum = sha1(bundle_file).hexdigest()
  58 
  59         blob1 = FileBlob.from_file(ContentFile(bundle_file))
  60         FileBlobOwner.objects.get_or_create(organization=self.organization, blob=blob1)
  61 
  62         response = self.client.post(
  63             self.url,

  66         )
  67 
  68         assert response.status_code == 200, response.content
  69         assert response.data["state"] == ChunkFileState.CREATED
  70         assert set(response.data["missingChunks"]) == set([])
  71 
  72         mock_assemble_artifacts.apply_async.assert_called_once_with(
  73             kwargs={
  74                 "org_id": self.organization.id,
  75                 "version": self.release.version,
  76                 "chunks": [blob1.checksum],
