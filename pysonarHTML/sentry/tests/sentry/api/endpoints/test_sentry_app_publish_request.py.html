  10 class SentryAppPublishRequestTest(APITestCase):
  11     def setUp(self):
  12         # create user as superuser
  13         self.user = self.create_user(email="boop@example.com", is_superuser=True)
  14         self.org = self.create_organization(owner=self.user, name="My Org")
  15         self.project = self.create_project(organization=self.org)
  16 
  17         self.sentry_app = self.create_sentry_app(name="Testin", organization=self.org)
  18 
  19         self.url = reverse("sentry-api-0-sentry-app-publish-request", args=[self.sentry_app.slug])
  20 
  21     @mock.patch("sentry.utils.email.send_mail")
  22     def test_publish_request(self, send_mail):
  23         self.login_as(user=self.user)
  24         response = self.client.post(
  25             self.url,
  26             format="json",
  27             data={
  28                 "questionnaire": [
  29                     {"question": "First question", "answer": "First response"},
  30                     {"question": "Second question", "answer": "Second response"},
  31                 ]
  32             },
  33         )
  34         assert response.status_code == 201
  35         message = (
  36             "User boop@example.com of organization my-org wants to publish testin"
  37             "\n\n\n>First question\nFirst response"
  38             "\n\n>Second question\nSecond response"
  39         )
  40 
  41         send_mail.assert_called_with(
  42             "Sentry Integration Publication Request from my-org",
  43             message,
  44             "root@localhost",
  45             ["partners@sentry.io"],
  46             reply_to=[self.user.email],
  47         )
  48 
  49     @mock.patch("sentry.utils.email.send_mail")
  50     def test_publish_already_published(self, send_mail):
  51         self.sentry_app.update(status=SentryAppStatus.PUBLISHED)
  52         self.login_as(user=self.user)
  53         response = self.client.post(self.url, format="json")
  54         assert response.status_code == 400
  55         assert response.data["detail"] == "Cannot publish already published integration"
  56         send_mail.asssert_not_called()
