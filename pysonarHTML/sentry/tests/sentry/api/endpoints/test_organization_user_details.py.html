   9 class OrganizationUserDetailsTest(APITestCase):
  10     def setUp(self):
  11         self.owner_user = self.create_user("foo@localhost", username="foo")
  12         self.user = self.create_user("bar@localhost", username="bar")
  13 
  14         self.org = self.create_organization(owner=self.owner_user)

  18         )

  20     def test_gets_info_for_user_in_org(self):
