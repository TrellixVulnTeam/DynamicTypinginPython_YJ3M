  21 class BroadcastUpdateTest(APITestCase):
  22     def test_regular_user(self):
  23         broadcast1 = Broadcast.objects.create(message="bar", is_active=True)
  24         broadcast2 = Broadcast.objects.create(message="foo", is_active=False)
  25 
  26         self.add_user_permission(user=self.user, permission="broadcasts.admin")
  27         self.login_as(user=self.user)
  28 
  29         response = self.client.put(
  30             u"/api/0/broadcasts/{}/".format(broadcast1.id), {"hasSeen": "1", "message": "foobar"}
  31         )
  32         assert response.status_code == 200
  33         assert response.data["hasSeen"]
  34 
  35         assert BroadcastSeen.objects.filter(user=self.user, broadcast=broadcast1).exists()
  36         assert not BroadcastSeen.objects.filter(user=self.user, broadcast=broadcast2).exists()
  37         broadcast1 = Broadcast.objects.get(id=broadcast1.id)
  38         assert broadcast1.message == "bar"
  39         broadcast2 = Broadcast.objects.get(id=broadcast2.id)
  40         assert broadcast2.message == "foo"
