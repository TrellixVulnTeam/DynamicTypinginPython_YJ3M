 605         super(ReleaseSerializerTest, self).setUp()
 606         self.version = "1234567890"
 607         self.repo_name = "repo/name"
 608         self.repo2_name = "repo2/name"
 609         self.commits = [{"id": "a" * 40}, {"id": "b" * 40}]

 614     def test_simple(self):
 615         serializer = ReleaseWithVersionSerializer(
 616             data={
 617                 "version": self.version,
 618                 "owner": self.user.username,
 619                 "ref": self.ref,
 620                 "url": self.url,
 621                 "dateReleased": self.dateReleased,
 622                 "commits": self.commits,
 623             }
 624         )
 625 
 626         assert serializer.is_valid()
 627         assert sorted(serializer.fields.keys()) == sorted(
 628             ["version", "owner", "ref", "url", "dateReleased", "commits"]
 629         )
 630 
 631         result = serializer.validated_data
 632         assert result["version"] == self.version
 633         assert result["owner"] == self.user
 634         assert result["ref"] == self.ref
 635         assert result["url"] == self.url
 636         assert result["dateReleased"] == datetime(1000, 10, 10, 6, 6, tzinfo=pytz.UTC)
 637         assert result["commits"] == self.commits
