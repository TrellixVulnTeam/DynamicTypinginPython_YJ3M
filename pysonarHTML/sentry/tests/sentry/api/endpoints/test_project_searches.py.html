  11 class ProjectSearchListTest(APITestCase):

  14 
  15         team = self.create_team()

  36     def test_user_searches_visible__before_and_after_project_write_permissions(self):
  37         user = self.create_user()
  38         # user without project-write permissions
  39         member = self.create_member(user=user, organization=self.organization, role="member")
  40         self.login_as(user=user)
  41 
  42         team = self.create_team()
  43         project1 = self.create_project(teams=[team], name="foo")
  44         SavedSearch.objects.filter(project=project1).delete()
  45 
  46         url = reverse(
  47             "sentry-api-0-project-searches",
  48             kwargs={"organization_slug": project1.organization.slug, "project_slug": project1.slug},
  49         )
  50 
  51         resp = self.client.post(
  52             url, format="json", data={"name": "Latest Release", "query": "release:[latest]"}
  53         )
  54         assert resp.status_code == 201, resp.content
  55         search1 = SavedSearch.objects.get(project=project1, owner_id=user.id)
  56 
  57         resp = self.client.get(url, format="json")
  58         assert resp.status_code == 200, resp.content
  59         assert len(resp.data) == 1
  60         assert resp.data[0]["id"] == six.text_type(search1.id)
  61 
  62         # update permissions
  63         member.role = "manager"
  64         member.save()
  65 
  66         resp = self.client.post(
  67             url, format="json", data={"name": "New Yesterday", "query": "age:[-48h]"}
  68         )
  69 
  70         assert resp.status_code == 201, resp.content
  71         search2 = SavedSearch.objects.get(project=project1, owner_id__isnull=True)
  72 
  73         resp = self.client.get(url, format="json")
  74         assert resp.status_code == 200, resp.content
  75         assert len(resp.data) == 2
  76         assert resp.data[0]["id"] == six.text_type(search1.id)
  77         assert resp.data[1]["id"] == six.text_type(search2.id)
