  11 class OrganizationMemberUnreleasedCommitsTest(APITestCase):
  12     def test_simple(self):
  13         user = self.create_user("foo@example.com")
  14         org = self.create_organization(name="foo")
  15         team = self.create_team(name="foo", organization=org)
  16         self.create_member(organization=org, user=user, role="admin", teams=[team])
  17         project = self.create_project(name="foo", organization=org, teams=[team])
  18 
  19         repo = self.create_repo(project)
  20         repo2 = self.create_repo(project)
  21 
  22         # we first need to create a release attached to a repository, or that repository
  23         # will never be included
  24         release = self.create_release(project)
  25         author = self.create_commit_author(project=project, user=user)
  26 
  27         # note: passing 'release' to create_commit causes it to bind ReleaseCommit
  28         self.create_commit(
  29             project=project,
  30             repo=repo,
  31             release=release,
  32             author=author,
  33             date_added=datetime(2015, 1, 1, tzinfo=timezone.utc),
  34         )
  35 
  36         # create a commit associated with an unreleased repo -- which should not appear
  37         self.create_commit(
  38             project=project,
  39             repo=repo2,
  40             author=author,
  41             date_added=datetime(2015, 1, 2, tzinfo=timezone.utc),
  42         )
  43 
  44         # create several unreleased commits associated with repo
  45         unreleased_commit = self.create_commit(
  46             project=project,
  47             repo=repo,
  48             author=author,
  49             date_added=datetime(2015, 1, 2, tzinfo=timezone.utc),
  50         )
  51         unreleased_commit2 = self.create_commit(
  52             project=project,
  53             repo=repo,
  54             author=author,
  55             date_added=datetime(2015, 1, 3, tzinfo=timezone.utc),
  56         )
  57         self.create_commit(
  58             project=project, repo=repo, date_added=datetime(2015, 1, 3, tzinfo=timezone.utc)
  59         )
  60 
  61         path = u"/api/0/organizations/{}/members/me/unreleased-commits/".format(org.slug)
  62 
  63         self.login_as(user)
  64 
  65         resp = self.client.get(path)
  66 
  67         assert resp.status_code == 200
  68         assert len(resp.data["commits"]) == 2
  69         assert resp.data["commits"][0]["id"] == unreleased_commit2.key
  70         assert resp.data["commits"][1]["id"] == unreleased_commit.key
  71         assert len(resp.data["repositories"]) == 1
  72         assert six.text_type(repo.id) in resp.data["repositories"]
