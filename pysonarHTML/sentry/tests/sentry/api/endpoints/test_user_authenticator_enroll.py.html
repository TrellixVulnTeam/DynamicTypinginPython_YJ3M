 180     def setUp(self):
 181         self.organization = self.create_organization(owner=self.create_user("foo@example.com"))
 182         self.user = self.create_user("bar@example.com", is_superuser=False)

 188     def require_2fa_for_organization(self):
 189         self.organization.update(flags=F("flags").bitor(Organization.flags.require_2fa))
 190         self.assertTrue(self.organization.flags.require_2fa.is_set)

 192     def _assert_pending_invite_cookie_set(self, response, om):
 193         invite_link = om.get_invite_link()
 194         invite_data = dict(parse_qsl(response.client.cookies["pending-invite"].value))
 195         assert invite_data.get("url") in invite_link

 202     def get_om_and_init_invite(self):
 203         om = OrganizationMember.objects.create(
 204             email="newuser@example.com", role="member", token="abc", organization=self.organization
 205         )
 206 
 207         resp = self.client.get(
 208             reverse("sentry-api-0-accept-organization-invite", args=[om.id, om.token])
 209         )
 210         assert resp.status_code == 200
 211         self._assert_pending_invite_cookie_set(resp, om)
 212 
 213         return om

 215     def assert_invite_accepted(self, response, member_id):
 216         om = OrganizationMember.objects.get(id=member_id)
 217         assert om.user == self.user
 218         assert om.email is None
 219 
 220         AuditLogEntry.objects.get(
 221             organization=self.organization,
 222             target_object=om.id,
 223             target_user=self.user,
 224             event=AuditLogEntryEvent.MEMBER_ACCEPT,
 225             data=om.get_audit_log_data(),
 226         )
 227 
 228         self.assertFalse(response.client.cookies["pending-invite"].value)
