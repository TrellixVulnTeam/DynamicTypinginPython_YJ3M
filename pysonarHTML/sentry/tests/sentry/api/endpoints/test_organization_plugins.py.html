   9 class OrganizationPluginsTest(APITestCase):
  10     def setUp(self):
  11         self.projectA = self.create_project()
  12         self.projectB = self.create_project(organization=self.projectA.organization)
  13 
  14         plugins.get("webhooks").enable(self.projectA)
  15         plugins.get("slack").enable(self.projectB)
  16 
  17         self.login_as(user=self.user)
  18 
  19     def test_exposes_all_plugins_available_no_enabled_state(self):
  20         url = reverse(
  21             "sentry-api-0-organization-plugins",
  22             kwargs={"organization_slug": self.projectA.organization.slug},
  23         )
  24 
  25         url = u"{}?{}".format(url, "plugins=_all")
  26 
  27         response = self.client.get(url)
  28 
  29         assert response.status_code == 200, (response.status_code, response.content)
  30 
  31         # should have these plugins:
  32         # issuetrackingplugin2, webhooks, mail
  33         # none of the plugins should have an 'enabled' key since these
  34         # plugins are not bound by a project
  35         plugins = [p for p in [p for p in response.data if "enabled" in p]]
  36         assert len(plugins) == 0
  37         assert len(response.data) > 0
  38 
  39     def test_exposes_plugins_across_all_org_projects(self):
  40         url = reverse(
  41             "sentry-api-0-organization-plugins",
  42             kwargs={"organization_slug": self.projectA.organization.slug},
  43         )
  44 
  45         url = u"{}?{}".format(url, "plugins=slack&plugins=webhooks")
  46 
  47         response = self.client.get(url)
  48 
  49         assert response.status_code == 200, (response.status_code, response.content)
  50 
  51         enabled_plugins = [
  52             (p["project"]["id"], p["slug"]) for p in [p for p in response.data if p["enabled"]]
  53         ]
  54 
  55         assert (self.projectA.id, "webhooks") in enabled_plugins
  56         assert (self.projectB.id, "slack") in enabled_plugins
  57 
  58     def test_exposes_specific_plugins_across_all_org_projects(self):
