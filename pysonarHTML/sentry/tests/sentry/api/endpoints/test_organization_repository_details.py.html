 170     def test_put_cancel_deletion(self):
 171         self.login_as(user=self.user)
 172 
 173         org = self.create_organization(owner=self.user, name="baz")
 174         integration = Integration.objects.create(provider="example", name="example")
 175         integration.add_organization(org)
 176 
 177         repo = Repository.objects.create(
 178             name="uuid-name",
 179             external_id="uuid-external-id",
 180             organization_id=org.id,
 181             status=ObjectStatus.PENDING_DELETION,
 182             config={"pending_deletion_name": "example-name"},
 183         )
 184 
 185         OrganizationOption.objects.create(
 186             organization_id=org.id,
 187             key=repo.build_pending_deletion_key(),
 188             value={
 189                 "name": "example-name",
 190                 "external_id": "example-external-id",
 191                 "id": repo.id,
 192                 "model": Repository.__name__,
 193             },
 194         )
 195 
 196         url = reverse("sentry-api-0-organization-repository-details", args=[org.slug, repo.id])
 197         response = self.client.put(url, data={"status": "visible", "integrationId": integration.id})
 198 
 199         assert response.status_code == 200
 200 
 201         repo = Repository.objects.get(id=repo.id)
 202         assert repo.status == ObjectStatus.VISIBLE
 203         assert repo.integration_id == integration.id
 204         assert repo.provider == "integrations:example"
 205         assert repo.name == "example-name"
 206         assert repo.external_id == "example-external-id"
 207         assert repo.config == {}
 208 
 209         assert not OrganizationOption.objects.filter(
 210             organization_id=org.id, key=repo.build_pending_deletion_key()
 211         ).exists()
