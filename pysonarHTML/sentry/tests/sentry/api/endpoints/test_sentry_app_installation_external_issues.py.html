  11 class SentryAppInstallationExternalIssuesEndpointTest(APITestCase):
  12     def setUp(self):
  13         self.superuser = self.create_user(email="a@example.com", is_superuser=True)
  14         self.user = self.create_user(email="boop@example.com")
  15         self.org = self.create_organization(owner=self.user)
  16         self.project = self.create_project(organization=self.org)
  17         self.group = self.create_group(project=self.project)
  18 
  19         self.sentry_app = self.create_sentry_app(
  20             name="Testin", organization=self.org, webhook_url="https://example.com"
  21         )
  22 
  23         self.install = self.create_sentry_app_installation(
  24             organization=self.org, slug=self.sentry_app.slug, user=self.user
  25         )
  26 
  27         self.url = reverse(
  28             "sentry-api-0-sentry-app-installation-external-issues", args=[self.install.uuid]
  29         )
  30 
  31     @responses.activate
  32     def test_creates_external_issue(self):
  33         self.login_as(user=self.superuser, superuser=True)
  34         data = {
  35             "groupId": self.group.id,
  36             "action": "create",
  37             "fields": {"title": "Hello"},
  38             "uri": "/create-issues",
  39         }
  40         responses.add(
  41             method=responses.POST,
  42             url="https://example.com/create-issues",
  43             json={
  44                 "project": "ProjectName",
  45                 "webUrl": "https://example.com/project/issue-id",
  46                 "identifier": "issue-1",
  47             },
  48             status=200,
  49             content_type="application/json",
  50         )
  51 
  52         response = self.client.post(self.url, data=data, format="json")
  53         external_issue = PlatformExternalIssue.objects.first()
  54 
  55         assert response.status_code == 200
  56         assert response.data == {
  57             "id": six.text_type(external_issue.id),
  58             "groupId": six.text_type(self.group.id),
  59             "serviceType": self.sentry_app.slug,
  60             "displayName": "ProjectName#issue-1",
  61             "webUrl": "https://example.com/project/issue-id",
  62         }
