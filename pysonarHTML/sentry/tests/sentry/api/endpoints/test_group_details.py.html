 512 class GroupDeleteTest(APITestCase):
 513     def test_delete(self):
 514         self.login_as(user=self.user)
 515 
 516         group = self.create_group()
 517         hash = "x" * 32
 518         GroupHash.objects.create(project=group.project, hash=hash, group=group)
 519 
 520         url = u"/api/0/issues/{}/".format(group.id)
 521 
 522         response = self.client.delete(url, format="json")
 523         assert response.status_code == 202, response.content
 524 
 525         # Deletion was deferred, so it should still exist
 526         assert Group.objects.get(id=group.id).status == GroupStatus.PENDING_DELETION
 527         # BUT the hash should be gone
 528         assert not GroupHash.objects.filter(group_id=group.id).exists()
 529 
 530         Group.objects.filter(id=group.id).update(status=GroupStatus.UNRESOLVED)
 531 
 532         url = u"/api/0/issues/{}/".format(group.id)
 533 
 534         with self.tasks():
 535             response = self.client.delete(url, format="json")
 536 
 537         assert response.status_code == 202, response.content
 538 
 539         # Now we killed everything with fire
 540         assert not Group.objects.filter(id=group.id).exists()
 541         assert not GroupHash.objects.filter(group_id=group.id).exists()
