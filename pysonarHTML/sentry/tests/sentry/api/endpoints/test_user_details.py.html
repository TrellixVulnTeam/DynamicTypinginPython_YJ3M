 160     def test_close_account(self):
 161         self.login_as(user=self.user)
 162         org_single_owner = self.create_organization(name="A", owner=self.user)
 163         user2 = self.create_user(email="user2@example.com")
 164         org_with_other_owner = self.create_organization(name="B", owner=self.user)
 165         org_as_other_owner = self.create_organization(name="C", owner=user2)
 166         not_owned_org = self.create_organization(name="D", owner=user2)
 167 
 168         self.create_member(user=user2, organization=org_with_other_owner, role="owner")
 169 
 170         self.create_member(user=self.user, organization=org_as_other_owner, role="owner")
 171 
 172         url = reverse("sentry-api-0-user-details", kwargs={"user_id": self.user.id})
 173 
 174         # test validations
 175         response = self.client.delete(url, data={})
 176         assert response.status_code == 400
 177         response = self.client.delete(url, data={"organizations": None})
 178         assert response.status_code == 400
 179 
 180         # test actual delete
 181         response = self.client.delete(
 182             url,
 183             data={
 184                 "organizations": [
 185                     org_with_other_owner.slug,
 186                     org_as_other_owner.slug,
 187                     not_owned_org.slug,
 188                 ]
 189             },
 190         )
 191 
 192         # deletes org_single_owner even though it wasn't specified in array
 193         # because it has a single owner
 194         assert (
 195             Organization.objects.get(id=org_single_owner.id).status
 196             == OrganizationStatus.PENDING_DELETION
 197         )
 198         # should delete org_with_other_owner, and org_as_other_owner
 199         assert (
 200             Organization.objects.get(id=org_with_other_owner.id).status
 201             == OrganizationStatus.PENDING_DELETION
 202         )
 203         assert (
 204             Organization.objects.get(id=org_as_other_owner.id).status
 205             == OrganizationStatus.PENDING_DELETION
 206         )
 207         # should NOT delete `not_owned_org`
 208         assert Organization.objects.get(id=not_owned_org.id).status == OrganizationStatus.ACTIVE
 209 
 210         assert response.status_code == 204

 247     def test_hard_delete_account(self):
 248         self.login_as(user=self.user)
 249         user2 = self.create_user(email="user2@example.com")
 250 
 251         url = reverse("sentry-api-0-user-details", kwargs={"user_id": user2.id})
 252 
 253         # failed authorization, user does not have permissions to delete another
 254         # user
 255         response = self.client.delete(url, data={"hardDelete": True, "organizations": []})
 256         assert response.status_code == 403
 257 
 258         # Reauthenticate as super user to hard delete an account
 259         self.user.update(is_superuser=True)
 260         self.login_as(user=self.user, superuser=True)
 261 
 262         response = self.client.delete(url, data={"hardDelete": True, "organizations": []})
 263         assert response.status_code == 204
