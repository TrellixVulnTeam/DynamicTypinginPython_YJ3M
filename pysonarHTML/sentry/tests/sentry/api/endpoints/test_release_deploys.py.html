  52 class ReleaseDeploysCreateTest(APITestCase):
  53     def test_simple(self):
  54         project = self.create_project(name="foo")
  55         release = Release.objects.create(
  56             organization_id=project.organization_id, version="1", total_deploys=0
  57         )
  58         release.add_project(project)
  59 
  60         environment = Environment.objects.create(
  61             organization_id=project.organization_id, name="production"
  62         )
  63 
  64         url = reverse(
  65             "sentry-api-0-organization-release-deploys",
  66             kwargs={"organization_slug": project.organization.slug, "version": release.version},
  67         )
  68 
  69         self.login_as(user=self.user)
  70 
  71         response = self.client.post(
  72             url, data={"name": "foo", "environment": "production", "url": "https://www.example.com"}
  73         )
  74         assert response.status_code == 201, response.content
  75         assert response.data["name"] == "foo"
  76         assert response.data["url"] == "https://www.example.com"
  77         assert response.data["environment"] == "production"
  78 
  79         deploy = Deploy.objects.get(id=response.data["id"])
  80 
  81         assert deploy.name == "foo"
  82         assert deploy.environment_id == environment.id
  83         assert deploy.url == "https://www.example.com"
  84         assert deploy.release == release
  85 
  86         release = Release.objects.get(id=release.id)
  87         assert release.total_deploys == 1
  88         assert release.last_deploy_id == deploy.id
  89 
  90         rpe = ReleaseProjectEnvironment.objects.get(
  91             project=project, release=release, environment=environment
  92         )
  93         assert rpe.last_deploy_id == deploy.id
