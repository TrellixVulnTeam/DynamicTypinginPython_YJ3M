  10 DEFAULT_SCOPES = ["project:read", "event:read", "team:read", "org:read", "member:read"]
  11 
  12 
  13 class OrganizationApiKeyDetails(APITestCase):
  14     def test_api_key_no_exist(self):
  15         self.login_as(user=self.user)
  16         organization = self.create_organization(name="foo", owner=self.user)
  17 
  18         path = reverse("sentry-api-0-organization-api-key-details", args=[organization.slug, 2])
  19 
  20         resp = self.client.get(path)
  21 
  22         assert resp.status_code == 404
  23 
  24     def test_get_api_details(self):
  25         self.login_as(user=self.user)
  26         organization = self.create_organization(name="foo", owner=self.user)
  27 
  28         api_key = ApiKey.objects.create(organization=organization, scope_list=DEFAULT_SCOPES)
  29 
  30         path = reverse(
  31             "sentry-api-0-organization-api-key-details", args=[organization.slug, api_key.id]
  32         )
  33 
  34         resp = self.client.get(path)
  35 
  36         assert resp.status_code == 200
  37         assert resp.data.get("id") == six.text_type(api_key.id)
  38 
  39     def test_update_api_key_details(self):
  40         self.login_as(user=self.user)
  41         organization = self.create_organization(name="foo", owner=self.user)
  42 
  43         api_key = ApiKey.objects.create(organization=organization, scope_list=DEFAULT_SCOPES)
  44 
  45         path = reverse(
  46             "sentry-api-0-organization-api-key-details", args=[organization.slug, api_key.id]
  47         )

  50 
  51         assert resp.status_code == 200
  52 
  53         api_key = ApiKey.objects.get(id=api_key.id, organization_id=organization.id)
  54 
  55         assert api_key.label == "New Label"
  56         assert api_key.allowed_origins == "sentry.io"
  57
