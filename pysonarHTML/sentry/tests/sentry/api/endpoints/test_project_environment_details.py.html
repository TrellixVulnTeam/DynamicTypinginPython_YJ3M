   9 class ProjectEnvironmentsTest(APITestCase):
  10     def test_get(self):
  11         project = self.create_project()
  12 
  13         environment = Environment.objects.create(
  14             project_id=project.id, organization_id=project.organization_id, name="production"
  15         )
  16         environment.add_project(project)
  17 
  18         self.login_as(user=self.user)
  19 
  20         url = reverse(
  21             "sentry-api-0-project-environment-details",
  22             kwargs={
  23                 "organization_slug": project.organization.slug,
  24                 "project_slug": project.slug,
  25                 "environment": "production",
  26             },
  27         )

  29         assert response.status_code == 200, response.content

  31             "id": u"{}".format(
  32                 EnvironmentProject.objects.get(environment__name="production", project=project).id
  33             ),
  34             "name": "production",
  35             "isHidden": False,
  36         }
  37 
  38         assert (
  39             self.client.get(
  40                 reverse(
  41                     "sentry-api-0-project-environment-details",
  42                     kwargs={
  43                         "organization_slug": project.organization.slug,
