  11 class OrganizationDashboardsTest(APITestCase):
  12     def setUp(self):
  13         super(OrganizationDashboardsTest, self).setUp()
  14         self.login_as(self.user)
  15         self.url = reverse(
  16             "sentry-api-0-organization-dashboards",
  17             kwargs={"organization_slug": self.organization.slug},
  18         )
  19         self.dashboard_1 = Dashboard.objects.create(
  20             title="Dashboard 1", created_by=self.user, organization=self.organization
  21         )
  22         self.dashboard_2 = Dashboard.objects.create(
  23             title="Dashboard 2", created_by=self.user, organization=self.organization
  24         )
  25 
  26     def assert_equal_dashboards(self, dashboard, data):
  27         assert data["id"] == six.text_type(dashboard.id)
  28         assert data["organization"] == six.text_type(dashboard.organization.id)
  29         assert data["title"] == dashboard.title
  30         assert data["createdBy"] == six.text_type(dashboard.created_by.id)
  31 
  32     def test_get(self):
  33         response = self.client.get(self.url)
  34         assert response.status_code == 200, response.content
  35         assert len(response.data) == 2
  36 
  37         self.assert_equal_dashboards(self.dashboard_1, response.data[0])
  38         self.assert_equal_dashboards(self.dashboard_2, response.data[1])
  39 
  40     def test_post(self):
  41         response = self.client.post(self.url, data={"title": "Dashboard from Post"})
  42         assert response.status_code == 201
  43         dashboard = Dashboard.objects.get(
  44             organization=self.organization, title="Dashboard from Post"
  45         )
  46         assert dashboard.created_by == self.user
  47 
  48     def test_query(self):
  49         dashboard = Dashboard.objects.create(
  50             title="Dashboard 11", created_by=self.user, organization=self.organization
  51         )
  52         response = self.client.get(self.url, data={"query": "1"})
  53         assert response.status_code == 200, response.content
  54         assert len(response.data) == 2
  55         self.assert_equal_dashboards(self.dashboard_1, response.data[0])
  56         self.assert_equal_dashboards(dashboard, response.data[1])
  57 
  58     def test_query_no_results(self):
  59         response = self.client.get(self.url, data={"query": "not-in-there"})
  60         assert response.status_code == 200, response.content
  61         assert len(response.data) == 0
  62 
  63     def test_invalid_data(self):
  64         response = self.client.post(self.url, data={"malformed-data": "Dashboard from Post"})
  65         assert response.status_code == 400
  66 
  67     def test_integrity_error(self):
  68         response = self.client.post(self.url, data={"title": self.dashboard_1.title})
  69         assert response.status_code == 409
  70         assert response.data == "This dashboard already exists"
