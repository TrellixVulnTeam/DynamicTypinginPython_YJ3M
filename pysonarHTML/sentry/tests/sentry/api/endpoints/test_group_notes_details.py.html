  75     @responses.activate

  78         self.login_as(user=self.user)
  79 
  80         with self.tasks():
  81             with self.feature("organizations:integrations-issue-sync"):
  82                 response = self.client.put(
  83                     self.url,
  84                     format="json",
  85                     data={
  86                         "text": "hi **@{}**".format(self.user.username),
  87                         "mentions": ["user:{}".format(self.user.id)],
  88                     },
  89                 )
  90         assert response.status_code == 200, response.content
  91 
  92         activity = Activity.objects.get(id=response.data["id"])

  94         assert activity.group == self.group
