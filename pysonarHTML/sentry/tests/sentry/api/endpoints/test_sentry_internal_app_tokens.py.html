  56     def test_token_limit(self):
  57         self.login_as(user=self.user)
  58 
  59         # we already have one token created so just need to make 19 more first
  60         for i in range(19):
  61             response = self.client.post(self.url, format="json")
  62             assert response.status_code == 201
  63 
  64         response = self.client.post(self.url, format="json")

  69 class GetSentryInternalAppTokenTest(SentryInternalAppTokenTest):
  70     def test_get_tokens(self):
  71         self.login_as(self.user)
  72 
  73         self.create_internal_integration(name="OtherInternal", organization=self.org)
  74 
  75         token = ApiToken.objects.get(application_id=self.internal_sentry_app.application_id)
  76 
  77         response = self.client.get(self.url, format="json")
  78 
  79         assert response.status_code == 200
  80         response_content = json.loads(response.content)
  81 
  82         # should not include tokens from other internal app
  83         assert len(response_content) == 1
  84 
  85         assert response_content[0]["id"] == six.text_type(token.id)
  86         assert response_content[0]["token"] == token.token
