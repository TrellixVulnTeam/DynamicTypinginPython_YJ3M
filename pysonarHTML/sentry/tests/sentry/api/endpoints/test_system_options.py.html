   9 class SystemOptionsTest(APITestCase):
  10     url = reverse("sentry-api-0-system-options")
  11 
  12     def test_without_superuser(self):
  13         self.login_as(user=self.user, superuser=False)
  14         response = self.client.get(self.url)
  15         assert response.status_code == 403
  16 
  17     def test_simple(self):
  18         self.login_as(user=self.user, superuser=True)
  19         response = self.client.get(self.url)
  20         assert response.status_code == 200
  21         assert "system.secret-key" in response.data
  22         assert "system.url-prefix" in response.data
  23         assert "system.admin-email" in response.data
  24 
  25     def test_bad_query(self):
  26         self.login_as(user=self.user, superuser=True)
  27         response = self.client.get(self.url, {"query": "nonsense"})
  28         assert response.status_code == 400
  29         assert "nonsense" in response.data
  30 
  31     def test_required(self):
  32         self.login_as(user=self.user, superuser=True)
  33         response = self.client.get(self.url, {"query": "is:required"})
  34         assert response.status_code == 200
  35         assert "system.rate-limit" not in response.data
  36         assert "system.url-prefix" in response.data
  37 
  38     def test_not_logged_in(self):
  39         response = self.client.get(self.url)
  40         assert response.status_code == 401
  41         response = self.client.put(self.url)
  42         assert response.status_code == 401

  44     def test_disabled_smtp(self):
  45         self.login_as(user=self.user, superuser=True)
  46 
  47         with self.options({"mail.backend": "smtp"}):
  48             response = self.client.get(self.url)
  49             assert response.status_code == 200
  50             assert response.data["mail.host"]["field"]["disabled"] is False
  51             assert response.data["mail.host"]["field"]["disabledReason"] is None
  52 
  53         with self.options({"mail.backend": "dummy"}):
  54             response = self.client.get(self.url)
  55             assert response.status_code == 200
  56             assert response.data["mail.host"]["field"]["disabled"] is True
  57             assert response.data["mail.host"]["field"]["disabledReason"] == "smtpDisabled"

  70     def test_put_simple(self):
  71         self.login_as(user=self.user, superuser=True)
