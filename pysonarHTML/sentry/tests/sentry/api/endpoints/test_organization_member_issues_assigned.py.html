  13 class OrganizationMemberIssuesAssignedTest(APITestCase):
  14     def test_simple(self):
  15         now = timezone.now()
  16         user = self.create_user("foo@example.com")
  17         org = self.create_organization(name="foo")
  18         team = self.create_team(name="foo", organization=org)
  19         self.create_member(organization=org, user=user, role="admin", teams=[team])
  20         project1 = self.create_project(name="foo", organization=org, teams=[team])
  21         group1 = self.create_group(project=project1)
  22         group2 = self.create_group(project=project1)
  23         project2 = self.create_project(
  24             name="bar", organization=org, teams=[team], status=ProjectStatus.PENDING_DELETION
  25         )
  26         group3 = self.create_group(project=project2)
  27         GroupAssignee.objects.create(group=group1, project=project1, user=user, date_added=now)
  28         GroupAssignee.objects.create(
  29             group=group2, project=project1, user=user, date_added=now + timedelta(seconds=1)
  30         )
  31         # should not show up as project is pending removal
  32         GroupAssignee.objects.create(
  33             group=group3, project=project2, user=user, date_added=now + timedelta(seconds=2)
  34         )
  35 
  36         path = reverse("sentry-api-0-organization-member-issues-assigned", args=[org.slug, "me"])
  37 
  38         self.login_as(user)
  39 
  40         resp = self.client.get(path)
  41 
  42         assert resp.status_code == 200
  43         assert len(resp.data) == 2
  44         assert resp.data[0]["id"] == six.text_type(group2.id)
  45         assert resp.data[1]["id"] == six.text_type(group1.id)
