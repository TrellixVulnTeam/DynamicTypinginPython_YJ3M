  79     def test_valid_env(self):
  80         event = self.store_event(
  81             data={
  82                 "tags": {"foo": "bar", "biz": "baz"},
  83                 "environment": "prod",
  84                 "timestamp": iso_format(before_now(minutes=1)),
  85             },
  86             project_id=self.project.id,
  87         )
  88         group = event.group
  89 
  90         self.login_as(user=self.user)
  91         url = u"/api/0/issues/{}/tags/".format(group.id)
  92         response = self.client.get(url, {"environment": "prod"}, format="json")
  93         assert response.status_code == 200
  94         assert len(response.data) == 4
  95         assert set([tag["key"] for tag in response.data]) == set(
  96             ["foo", "biz", "environment", "level"]

  99     def test_multi_env(self):
 100         min_ago = before_now(minutes=1)
 101         env = self.create_environment(project=self.project, name="prod")
 102         env2 = self.create_environment(project=self.project, name="staging")
 103         self.store_event(
 104             data={
 105                 "fingerprint": ["put-me-in-group1"],
 106                 "timestamp": iso_format(min_ago),
 107                 "environment": env.name,
 108                 "tags": {"foo": "bar"},
 109             },
 110             project_id=self.project.id,
 111         )
 112         event2 = self.store_event(
 113             data={
 114                 "fingerprint": ["put-me-in-group1"],
 115                 "timestamp": iso_format(min_ago),
 116                 "environment": env2.name,
 117                 "tags": {"biz": "baz"},
 118             },
 119             project_id=self.project.id,
 120         )
 121 
 122         self.login_as(user=self.user)
 123         url = u"/api/0/issues/{}/tags/".format(event2.group.id)
 124         response = self.client.get(
 125             "%s?environment=%s&environment=%s" % (url, env.name, env2.name), format="json"
 126         )
 127         assert response.status_code == 200
 128         assert set([tag["key"] for tag in response.data]) >= set(["biz", "environment", "foo"])
