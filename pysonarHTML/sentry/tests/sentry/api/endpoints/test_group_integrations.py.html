  13         group = self.create_group()
  14         integration = Integration.objects.create(provider="example", name="Example")
  15         integration.add_organization(org, self.user)
  16         external_issue = ExternalIssue.objects.create(
  17             organization_id=org.id,
  18             integration_id=integration.id,
  19             key="APP-123",
  20             title="this is an example title",
  21             description="this is an example description",
  22         )
  23         GroupLink.objects.create(
  24             group_id=group.id,
  25             project_id=group.project_id,
  26             linked_type=GroupLink.LinkedType.issue,
  27             linked_id=external_issue.id,
  28             relationship=GroupLink.Relationship.references,
  29         )
  30 
  31         path = u"/api/0/issues/{}/integrations/".format(group.id)
  32

  36

  38                 "id": six.text_type(integration.id),

  43                 "status": integration.get_status_display(),

  48                     "canAdd": provider.can_add,
