  26     def test_chunk_parameters(self):

  29         )
  30 
  31         endpoint = options.get("system.upload-url-prefix")

  65     def test_upload(self):
  66         data1 = "1 this is my testString"
  67         data2 = "2 this is my testString"
  68         checksum1 = sha1(data1).hexdigest()
  69         checksum2 = sha1(data2).hexdigest()
  70         blob1 = SimpleUploadedFile(checksum1, data1, content_type="multipart/form-data")
  71         blob2 = SimpleUploadedFile(checksum2, data2, content_type="multipart/form-data")
  72 
  73         response = self.client.post(
  74             self.url,
  75             data={"file": [blob1, blob2]},
  76             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
  77             # this tells drf to select the MultiPartParser. We use that instead of
  78             # FileUploadParser because we have our own specific file chunking mechanism
  79             # in the chunk endpoint that has requirements like blob/chunk's filename = checksum.
  80             format="multipart",
  81         )
  82 
  83         assert response.status_code == 200, response.content
  84 
  85         file_blobs = FileBlob.objects.all()
  86         assert len(file_blobs) == 2
  87         assert file_blobs[0].checksum == checksum1
  88         assert file_blobs[1].checksum == checksum2

  99     def test_too_many_chunks(self):
 100         files = []
 101 
 102         # Exactly the limit
 103         for x in range(0, MAX_CHUNKS_PER_REQUEST + 1):
 104             content = "x"
 105             files.append(SimpleUploadedFile(sha1(content).hexdigest(), content))
 106 
 107         response = self.client.post(
 108             self.url,
 109             data={"file": files},
 110             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 111             format="multipart",
 112         )
 113 
 114         assert response.status_code == 400, response.content

 116     def test_too_large_request(self):
 117         files = []
 118 
 119         # Exactly the limit
 120         for x in range(0, MAX_CHUNKS_PER_REQUEST):
 121             content = "x" * (MAX_REQUEST_SIZE // MAX_CHUNKS_PER_REQUEST)
 122             files.append(SimpleUploadedFile(sha1(content).hexdigest(), content))
 123 
 124         response = self.client.post(
 125             self.url,
 126             data={"file": files},
 127             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 128             format="multipart",
 129         )
 130 
 131         assert response.status_code == 200, response.content
 132 
 133         # We overflow the request here
 134         files.append(SimpleUploadedFile(sha1("content").hexdigest(), "content"))
 135         response = self.client.post(
 136             self.url,
 137             data={"file": files},
 138             HTTP_AUTHORIZATION=u"Bearer {}".format(self.token.token),
 139             format="multipart",
 140         )
 141         assert response.status_code == 400, response.content
