   9 class ProjectCreateSampleTest(APITestCase):

  14     def test_simple(self):
  15         project = self.create_project(teams=[self.team], name="foo")
  16 
  17         url = reverse(
  18             "sentry-api-0-project-create-sample",
  19             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
  20         )
  21         response = self.client.post(url, format="json")
