 113     def test_member_can_override_their_default(self):
 114         project = self.create_project(name="foo")
 115 
 116         member = self.create_user("member@example.com", is_superuser=False)
 117         self.create_member(
 118             user=member,
 119             role="member",
 120             organization=project.organization,
 121             teams=[project.teams.first()],
 122         )
 123 
 124         search = SavedSearch.objects.create(project=project, name="foo", query="")
 125 
 126         SavedSearch.objects.create(project=project, name="bar", query="", is_default=True)
 127 
 128         self.login_as(user=member)
 129 
 130         url = reverse(
 131             "sentry-api-0-project-search-details",
 132             kwargs={
 133                 "organization_slug": project.organization.slug,
 134                 "project_slug": project.slug,
 135                 "search_id": search.id,
 136             },
 137         )
 138         response = self.client.put(
 139             url,
 140             {
 141                 # these params get ignored barring isUserDefault
 142                 "name": "baz",
 143                 "isUserDefault": True,
 144                 "isDefault": True,
 145             },
 146         )
 147 
 148         assert response.status_code == 200, response.content
 149         assert response.data["id"] == six.text_type(search.id)
 150 
 151         search = SavedSearch.objects.get(id=search.id)
 152         assert search.name == "foo"
 153         assert not search.is_default
 154 
 155         assert SavedSearchUserDefault.objects.filter(
 156             user=member, project=project, savedsearch=search
 157         ).exists()
