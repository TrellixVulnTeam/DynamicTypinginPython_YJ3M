  36 class GetSentryAppRequestsTest(SentryAppRequestsTest):
  37     def test_superuser_sees_unowned_published_requests(self):
  38         self.login_as(user=self.superuser, superuser=True)
  39 
  40         buffer = SentryAppWebhookRequestsBuffer(self.unowned_published_app)
  41         buffer.add_request(
  42             response_code=200,
  43             org_id=self.org.id,
  44             event="issue.assigned",
  45             url=self.unowned_published_app.webhook_url,
  46         )
  47         buffer.add_request(
  48             response_code=500,
  49             org_id=self.org.id,
  50             event="issue.assigned",
  51             url=self.unowned_published_app.webhook_url,
  52         )
  53 
  54         url = reverse("sentry-api-0-sentry-app-requests", args=[self.unowned_published_app.slug])
  55         response = self.client.get(url, format="json")
  56         assert response.status_code == 200
  57         assert len(response.data) == 2
  58         assert response.data[0]["organization"]["slug"] == self.org.slug
  59         assert response.data[0]["sentryAppSlug"] == self.unowned_published_app.slug
  60         assert response.data[0]["responseCode"] == 500

 200     def test_linked_error_id_converts_to_url(self):
 201         self.login_as(user=self.user)
 202 
 203         event = self.store_event(
 204             data={"event_id": self.event_id, "timestamp": iso_format(before_now(minutes=1))},
 205             project_id=self.project.id,
 206         )
 207 
 208         buffer = SentryAppWebhookRequestsBuffer(self.published_app)
 209         buffer.add_request(
 210             response_code=200,
 211             org_id=self.org.id,
 212             event="issue.assigned",
 213             url=self.unpublished_app.webhook_url,
 214             error_id=self.event_id,
 215             project_id=self.project.id,
 216         )
 217 
 218         url = reverse("sentry-api-0-sentry-app-requests", args=[self.published_app.slug])
 219         response = self.client.get(url, format="json")
 220         assert response.status_code == 200
 221         assert len(response.data) == 1
 222         assert response.data[0]["organization"]["slug"] == self.org.slug
 223         assert response.data[0]["sentryAppSlug"] == self.published_app.slug
 224         assert response.data[0]["errorUrl"] == reverse(
 225             "sentry-organization-event-detail", args=[self.org.slug, event.group_id, event.event_id]
