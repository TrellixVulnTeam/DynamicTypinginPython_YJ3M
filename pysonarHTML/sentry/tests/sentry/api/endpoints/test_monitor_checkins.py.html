  12 class CreateMonitorCheckInTest(APITestCase):
  13     def test_passing(self):
  14         user = self.create_user()
  15         org = self.create_organization(owner=user)
  16         team = self.create_team(organization=org, members=[user])
  17         project = self.create_project(teams=[team])
  18 
  19         monitor = Monitor.objects.create(
  20             organization_id=org.id,
  21             project_id=project.id,
  22             next_checkin=timezone.now() - timedelta(minutes=1),
  23             type=MonitorType.CRON_JOB,
  24             config={"schedule": "* * * * *"},
  25         )
  26 
  27         self.login_as(user=user)
  28         with self.feature({"organizations:monitors": True}):
  29             resp = self.client.post(
  30                 "/api/0/monitors/{}/checkins/".format(monitor.guid), data={"status": "ok"}
  31             )
  32 
  33         assert resp.status_code == 201, resp.content
  34 
  35         checkin = MonitorCheckIn.objects.get(guid=resp.data["id"])
  36         assert checkin.status == CheckInStatus.OK
  37 
  38         monitor = Monitor.objects.get(id=monitor.id)
  39         assert monitor.status == MonitorStatus.OK
  40         assert monitor.last_checkin == checkin.date_added
  41         assert monitor.next_checkin == monitor.get_next_scheduled_checkin(checkin.date_added)
