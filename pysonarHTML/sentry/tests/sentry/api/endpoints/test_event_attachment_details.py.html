  10 class CreateAttachmentMixin(object):
  11     def create_attachment(self):
  12         self.project = self.create_project()
  13         self.release = self.create_release(self.project, self.user)
  14         min_ago = iso_format(before_now(minutes=1))
  15         self.event = self.store_event(
  16             data={
  17                 "fingerprint": ["group1"],
  18                 "timestamp": min_ago,
  19                 "tags": {"sentry:release": self.release.version},
  20             },
  21             project_id=self.project.id,
  22         )
  23 
  24         self.file = File.objects.create(name="hello.png", type="image/png")
  25         self.file.putfile(six.BytesIO(b"File contents here"))
  26 
  27         self.attachment = EventAttachment.objects.create(
  28             event_id=self.event.event_id,
  29             project_id=self.event.project_id,
  30             file=self.file,
  31             name="hello.png",
  32         )
  33 
  34         return self.attachment

  84 class EventAttachmentDetailsPermissionTest(PermissionTestCase, CreateAttachmentMixin):

  87         self.create_attachment()
  88         self.path = u"/api/0/projects/{}/{}/events/{}/attachments/{}/?download".format(

  91 
  92     def test_member_can_access_by_default(self):
  93         with self.feature("organizations:event-attachments"):
  94             self.assert_member_can_access(self.path)
  95             self.assert_can_access(self.owner, self.path)
  96 
  97     def test_member_cannot_access_for_owner_role(self):
