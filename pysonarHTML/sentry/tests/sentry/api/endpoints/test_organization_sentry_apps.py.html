   9 def assert_response_json(response, data):
  10     """
  11     Normalizes unicode strings by encoding/decoding expected output
  12     """
  13     assert json.loads(response.content) == json.loads(json.dumps(data))

  16 class OrganizationSentryAppsTest(APITestCase):

  18         self.superuser = self.create_user(email="a@example.com", is_superuser=True)
  19         self.user = self.create_user(email="boop@example.com")
  20         self.org = self.create_organization(owner=self.user)
