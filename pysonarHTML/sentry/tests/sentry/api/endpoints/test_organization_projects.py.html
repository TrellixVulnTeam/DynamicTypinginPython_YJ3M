 100     def test_bookmarks_appear_first_across_pages(self):
 101         self.login_as(user=self.user)
 102 
 103         projects = [self.create_project(teams=[self.team], name=i, slug=i) for i in range(3)]
 104         projects.sort(key=lambda project: project.slug)
 105 
 106         response = self.client.get(self.path)
 107         self.check_valid_response(response, [project for project in projects])
 108 
 109         response = self.client.get(self.path + "?per_page=2")
 110         self.check_valid_response(response, [project for project in projects[:2]])
 111 
 112         self.create_project_bookmark(projects[-1], user=self.user)
 113         # Move the bookmarked project to the front
 114         projects.insert(0, projects.pop())
 115         response = self.client.get(self.path)
 116         self.check_valid_response(response, [project for project in projects])
 117 
 118         # Make sure that it's at the front when on the second page as well
 119         response = self.client.get(self.path + "?per_page=2")
 120         self.check_valid_response(response, [project for project in projects[:2]])
 121 
 122         # Make sure that other user's bookmarks don't interfere with this user
 123         other_user = self.create_user()
 124         self.create_project_bookmark(projects[1], user=other_user)
 125         response = self.client.get(self.path)
 126         self.check_valid_response(response, [project for project in projects])
