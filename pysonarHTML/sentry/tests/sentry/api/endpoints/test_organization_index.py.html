 149 class OrganizationIndex2faTest(TwoFactorAPITestCase):
 150     def setUp(self):
 151         self.org_2fa = self.create_organization(owner=self.create_user())
 152         self.enable_org_2fa(self.org_2fa)
 153         self.no_2fa_user = self.create_user()
 154         self.create_member(organization=self.org_2fa, user=self.no_2fa_user, role="member")
 155 
 156     @fixture
 157     def path(self):
 158         return reverse("sentry-organization-home", kwargs={"organization_slug": self.org_2fa.slug})
 159 
 160     def assert_can_access_org_home(self):
 161         response = self.client.get(self.path)
 162         assert response.status_code == 200
 163 
 164     def assert_redirected_to_2fa(self):
 165         response = self.client.get(self.path)
 166         assert response.status_code == 302
 167         assert self.path_2fa in response.url
 168 
 169     def test_preexisting_members_must_enable_2fa(self):
 170         self.login_as(self.no_2fa_user)
 171         self.assert_redirected_to_2fa()
 172 
 173         TotpInterface().enroll(self.no_2fa_user)
 174         self.assert_can_access_org_home()
 175 
 176     def test_new_member_must_enable_2fa(self):
 177         new_user = self.create_user()
 178         self.create_member(organization=self.org_2fa, user=new_user, role="member")
 179         self.login_as(new_user)
 180 
 181         self.assert_redirected_to_2fa()
 182 
 183         TotpInterface().enroll(new_user)
 184         self.assert_can_access_org_home()
 185 
 186     def test_member_disable_all_2fa_blocked(self):
 187         TotpInterface().enroll(self.no_2fa_user)
 188         self.login_as(self.no_2fa_user)
 189         self.assert_can_access_org_home()
 190 
 191         Authenticator.objects.get(user=self.no_2fa_user).delete()
 192         self.assert_redirected_to_2fa()
 193 
 194     def test_superuser_can_access_org_home(self):
 195         user = self.create_user(is_superuser=True)
 196         self.login_as(user, superuser=True)
 197         self.assert_can_access_org_home()
