  10 class OrganizationUserTeamsTest(APITestCase):
  11     def setUp(self):
  12         self.foo = self.create_user("foo@example.com")
  13         self.bar = self.create_user("bar@example.com", is_superuser=True)
  14         self.org = self.create_organization(owner=self.user)
  15         self.team1 = self.create_team(organization=self.org)
  16         self.team2 = self.create_team(organization=self.org)
  17         self.team3 = self.create_team(organization=self.org)
  18         self.project1 = self.create_project(teams=[self.team1])
  19         self.project2 = self.create_project(teams=[self.team2])
  20         self.create_member(organization=self.org, user=self.foo, teams=[self.team1, self.team2])
  21         self.create_member(organization=self.org, user=self.bar, teams=[self.team2])
  22 
  23         self.url = reverse(
  24             "sentry-api-0-organization-user-teams", kwargs={"organization_slug": self.org.slug}
  25         )
  26 
  27     def test_simple(self):
  28         self.login_as(user=self.foo)
  29 
  30         response = self.client.get(self.url)
  31         assert response.status_code == 200
  32 
  33         # Verify that only teams that the user is a member of, are returned
  34         assert len(response.data) == 2
  35 
  36         # Sort teams so there is a guaranteed ordering
  37         response.data.sort(key=lambda x: x["id"])
  38         assert response.data[0]["id"] == six.text_type(self.team1.id)
  39         assert response.data[0]["isMember"]
  40         assert response.data[0]["projects"][0]["id"] == six.text_type(self.project1.id)
  41 
  42         assert response.data[1]["id"] == six.text_type(self.team2.id)
  43         assert response.data[1]["isMember"]
  44         assert response.data[1]["projects"][0]["id"] == six.text_type(self.project2.id)
  45 
  46     def test_super_user(self):
  47         self.login_as(user=self.bar, superuser=True)
  48 
  49         response = self.client.get(self.url)
  50         assert response.status_code == 200
  51 
  52         # Verify that all teams are returned
  53         assert len(response.data) == 3
  54 
  55         # Sort teams so there is a guaranteed ordering
  56         response.data.sort(key=lambda x: x["id"])
  57         assert response.data[0]["id"] == six.text_type(self.team1.id)
  58         assert not response.data[0]["isMember"]
  59         assert response.data[0]["projects"][0]["id"] == six.text_type(self.project1.id)
  60 
  61         assert response.data[1]["id"] == six.text_type(self.team2.id)
  62         assert response.data[1]["isMember"]
  63         assert response.data[1]["projects"][0]["id"] == six.text_type(self.project2.id)
  64 
  65         assert response.data[2]["id"] == six.text_type(self.team3.id)
  66         assert not response.data[2]["isMember"]
