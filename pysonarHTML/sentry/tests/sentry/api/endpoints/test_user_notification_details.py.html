  43     def test_returns_correct_defaults(self):
  44         user = self.create_user(email="a@example.com")
  45         org = self.create_organization(name="Org Name", owner=user)
  46 
  47         # Adding existing UserOptions for a project or org to test that defaults are correct
  48         # default is 3
  49         UserOption.objects.create(
  50             user=user, project=None, organization=org, key="deploy-emails", value=1
  51         )
  52 
  53         # default is UserOptionValue.participating_only
  54         UserOption.objects.create(
  55             user=user,
  56             project=None,
  57             organization=org,
  58             key="workflow:notifications",
  59             value=UserOptionValue.all_conversations,
  60         )
  61 
  62         self.login_as(user=user)
  63 
  64         url = reverse("sentry-api-0-user-notifications", kwargs={"user_id": "me"})
  65         resp = self.client.get(url, format="json")
  66 
  67         assert resp.data.get("deployNotifications") == 3
  68         assert resp.data.get("personalActivityNotifications") is False
  69         assert resp.data.get("selfAssignOnResolve") is False
  70         assert resp.data.get("subscribeByDefault") is True
  71         assert resp.data.get("workflowNotifications") == int(UserOptionValue.participating_only)
