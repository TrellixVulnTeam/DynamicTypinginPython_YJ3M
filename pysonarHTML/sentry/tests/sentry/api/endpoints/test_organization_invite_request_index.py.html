  79     def test_simple(self):
  80         self.login_as(user=self.user)
  81         with self.tasks():
  82             response = self.client.post(
  83                 self.url, {"email": "eric@localhost", "role": "member", "teams": [self.team.slug]}
  84             )
  85 
  86         assert response.status_code == 201
  87         assert response.data["email"] == "eric@localhost"

  89         assert len(mail.outbox) == 1
  90 
  91         member = OrganizationMember.objects.get(organization=self.org, email=response.data["email"])
  92         assert member.user is None
  93         assert member.role == "member"
  94         assert member.inviter == self.user
  95         assert member.invite_status == InviteStatus.REQUESTED_TO_BE_INVITED.value
  96 
  97         teams = OrganizationMemberTeam.objects.filter(organizationmember=member)
  98
