  11 class SentryAppInstallationsTest(APITestCase):
  12     def setUp(self):
  13         self.superuser = self.create_user(email="a@example.com", is_superuser=True)
  14         self.user = self.create_user(email="boop@example.com")
  15         self.org = self.create_organization(owner=self.user)
  16         self.super_org = self.create_organization(owner=self.superuser)
  17 
  18         self.published_app = self.create_sentry_app(
  19             name="Test", organization=self.super_org, published=True
  20         )
  21         self.unpublished_app = self.create_sentry_app(name="Testin", organization=self.org)
  22 
  23         self.installation = self.create_sentry_app_installation(
  24             slug=self.published_app.slug, organization=self.super_org, user=self.superuser
  25         )
  26 
  27         self.installation2 = self.create_sentry_app_installation(
  28             slug=self.unpublished_app.slug, organization=self.org, user=self.user
  29         )
  30 
  31         self.url = reverse("sentry-api-0-sentry-app-installations", args=[self.org.slug])

  87 class PostSentryAppInstallationsTest(SentryAppInstallationsTest):
  88     def test_install_unpublished_app(self):
  89         self.login_as(user=self.user)
  90         app = self.create_sentry_app(name="Sample", organization=self.org)
  91         response = self.client.post(self.url, data={"slug": app.slug}, format="json")
  92         expected = {
  93             "app": {"slug": app.slug, "uuid": app.uuid},
  94             "organization": {"slug": self.org.slug},
  95         }
  96 
  97         assert response.status_code == 200, response.content
  98         assert six.viewitems(expected) <= six.viewitems(response.data)
  99 
 100     def test_install_published_app(self):

 112     def test_members_cannot_install_apps(self):
 113         user = self.create_user("bar@example.com")
 114         self.create_member(organization=self.org, user=user, role="member")
 115         self.login_as(user)
 116         app = self.create_sentry_app(name="Sample", organization=self.org, published=True)
 117         response = self.client.post(self.url, data={"slug": app.slug}, format="json")
 118         assert response.status_code == 403

 120     def test_install_twice(self):
 121         self.login_as(user=self.user)
 122         app = self.create_sentry_app(name="Sample", organization=self.org)
 123         self.client.post(self.url, data={"slug": app.slug}, format="json")
 124         response = self.client.post(self.url, data={"slug": app.slug}, format="json")
 125         assert SentryAppInstallation.objects.filter(sentry_app=app).count() == 1
 126         assert response.status_code == 200
