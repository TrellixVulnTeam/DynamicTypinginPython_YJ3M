  57     def test_only_configuable_plugins(self):
  58         response = self.client.get(self.url)
  59         assert [x for x in response.data if not x["hasConfiguration"]] == []

  61     def test_enabled_not_configured(self):
  62         plugins.get("webhooks").enable(self.projectA)
  63         response = self.client.get(self.url)
  64         assert filter(lambda x: x["slug"] == "webhooks", response.data)[0]["projectList"] == []

  66     def test_configured_not_enabled(self):

  68         plugins.get("trello").set_option("key", "some_value", self.projectA)

  96     def test_configured_multiple_projects(self):
  97         plugins.get("trello").set_option("key", "some_value", self.projectA)
  98         plugins.get("trello").set_option("key", "another_value", self.projectB)
  99         response = self.client.get(self.url)
 100         projectList = filter(lambda x: x["slug"] == "trello", response.data)[0]["projectList"]
 101         assert filter(lambda x: x["projectId"] == self.projectA.id, projectList)[0] == {
 102             "projectId": self.projectA.id,
 103             "projectSlug": self.projectA.slug,
 104             "projectName": self.projectA.name,
 105             "enabled": False,
 106             "configured": True,
 107             "projectPlatform": None,
 108         }
 109         assert filter(lambda x: x["projectId"] == self.projectB.id, projectList)[0] == {
 110             "projectId": self.projectB.id,
 111             "projectSlug": self.projectB.slug,
 112             "projectName": self.projectB.name,
 113             "enabled": False,
 114             "configured": True,
 115             "projectPlatform": "react",
 116         }

 118     def test_query_parameter(self):
 119         url = self.url + "?plugins=trello"
 120         response = self.client.get(url)
 121         assert len(response.data) == 1
 122         assert response.data[0]["id"] == "trello"

 124     def test_query_parameter_bad_slug(self):
 125         url = self.url + "?plugins=bad_plugin"
 126         response = self.client.get(url)
 127         assert response.status_code == 404
 128         assert response.data["detail"] == "Plugin bad_plugin not found"

 130     def test_sort_by_slug(self):
 131         another = self.create_project(slug="another")
 132         plugins.get("trello").set_option("key", "some_value", self.projectA)
 133         plugins.get("trello").set_option("key", "some_value", self.projectB)
 134         plugins.get("trello").set_option("key", "some_value", another)
 135         url = self.url + "?plugins=trello"
 136         response = self.client.get(url)
 137         assert map(lambda x: x["projectSlug"], response.data[0]["projectList"]) == [
 138             "another",
 139             "proj_a",
