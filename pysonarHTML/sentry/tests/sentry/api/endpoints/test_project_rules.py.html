  30     def test_simple(self):

  32 
  33         project = self.create_project()
  34 
  35         conditions = [
  36             {
  37                 "id": "sentry.rules.conditions.first_seen_event.FirstSeenEventCondition",

  39                 "match": "eq",

  42         ]

  52             data={
  53                 "name": "hello world",
