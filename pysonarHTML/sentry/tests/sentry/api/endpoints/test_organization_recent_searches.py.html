 123 class RecentSearchesCreateTest(APITestCase):
 124     endpoint = "sentry-api-0-organization-recent-searches"
 125     method = "post"
 126 
 127     @fixture
 128     def organization(self):
 129         return self.create_organization()
 130 
 131     @fixture
 132     def user(self):
 133         user = self.create_user("test@test.com")
 134         self.create_team(members=[user], organization=self.organization)
 135         return user
 136 
 137     def test(self):
 138         self.login_as(self.user)
 139         search_type = 1
 140         query = "something"
 141         the_date = datetime(2019, 1, 1, 1, 1, 1)
 142         with freeze_time(the_date):
 143             response = self.get_response(self.organization.slug, type=search_type, query=query)
 144             assert response.status_code == 201
 145             assert RecentSearch.objects.filter(
 146                 organization=self.organization,
 147                 user=self.user,
 148                 type=search_type,
 149                 query=query,
 150                 last_seen=the_date,
 151             ).exists()
 152         the_date = datetime(2019, 1, 1, 2, 2, 2)
 153         with freeze_time(the_date):
 154             response = self.get_response(self.organization.slug, type=search_type, query=query)
 155             assert response.status_code == 204, response.content
 156             assert RecentSearch.objects.filter(
 157                 organization=self.organization,
 158                 user=self.user,
 159                 type=search_type,
 160                 query=query,
 161                 last_seen=the_date,
 162             ).exists()
