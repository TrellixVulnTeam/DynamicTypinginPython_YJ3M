  40 class TeamDeleteTest(APITestCase):
  41     def assert_team_deleted(self, team_id, mock_delete_team, transaction_id):
  42         """Checks team status, membership in DeletedTeams table, org
  43            audit log, and to see that delete function has been called"""
  44 
  45         team = Team.objects.get(id=team_id)
  46 
  47         assert team.status == TeamStatus.PENDING_DELETION
  48 
  49         deleted_team = DeletedTeam.objects.get(slug=team.slug)
  50         # in spite of the name, this checks the DeletedTeam object, not the
  51         # audit log
  52         self.assert_valid_deleted_log(deleted_team, team)
  53 
  54         # *this* actually checks the audit log
  55         audit_log_entry = AuditLogEntry.objects.filter(
  56             event=AuditLogEntryEvent.TEAM_REMOVE, target_object=team.id
  57         )
  58         assert audit_log_entry
  59 
  60         mock_delete_team.apply_async.assert_called_once_with(
  61             kwargs={"object_id": team.id, "transaction_id": transaction_id}
  62         )

  64     def assert_team_not_deleted(self, team_id, mock_delete_team):
  65         """Checks team status, membership in DeletedTeams table, org
  66            audit log, and to see that delete function has not been called"""
  67 
  68         team = Team.objects.get(id=team_id)

 113     @patch("sentry.api.endpoints.team_details.uuid4")
 114     @patch("sentry.api.endpoints.team_details.delete_team")
 115     def test_remove_as_admin_not_in_team(self, mock_delete_team, mock_uuid4):
 116         """Admins can't remove teams of which they're not a part, unless
 117            open membership is on."""
 118 
 119         # mock the transaction_id when mock_delete_team is called
 120         class uuid(object):
 121             hex = "abc123"
 122 
 123         mock_uuid4.return_value = uuid
 124 
 125         # an org with closed membership (byproduct of flags=0)
 126         org = self.create_organization(owner=self.user, flags=0)
 127         team = self.create_team(organization=org)
 128         admin_user = self.create_user(email="foo@example.com", is_superuser=False)
 129 
 130         self.create_member(
 131             organization=org,
 132             user=admin_user,
 133             role="admin",
 134             teams=[],  # note that admin_user isn't a member of `team`
 135         )
 136 
 137         self.login_as(admin_user)
 138 
 139         url = reverse(
 140             "sentry-api-0-team-details",
 141             kwargs={"organization_slug": team.organization.slug, "team_slug": team.slug},
 142         )
 143 
 144         # first, try deleting the team with open membership off
 145         response = self.client.delete(url)
 146 
 147         assert response.status_code == 403
 148         self.assert_team_not_deleted(team.id, mock_delete_team)
 149 
 150         # now, with open membership on
 151         org.flags.allow_joinleave = True
 152         org.save()
 153         response = self.client.delete(url)
 154 
 155         assert response.status_code == 204
 156         self.assert_team_deleted(team.id, mock_delete_team, "abc123")
