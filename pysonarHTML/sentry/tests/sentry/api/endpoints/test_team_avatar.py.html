  13 class TeamAvatarTest(APITestCase):
  14     def test_get(self):
  15         team = self.team  # force creation
  16         self.login_as(user=self.user)
  17         url = reverse(
  18             "sentry-api-0-team-avatar",
  19             kwargs={"organization_slug": team.organization.slug, "team_slug": team.slug},
  20         )
  21         response = self.client.get(url)
  22         assert response.status_code == 200
  23         assert response.data["id"] == six.text_type(team.id)
  24         assert response.data["avatar"]["avatarType"] == "letter_avatar"
  25         assert response.data["avatar"]["avatarUuid"] is None
