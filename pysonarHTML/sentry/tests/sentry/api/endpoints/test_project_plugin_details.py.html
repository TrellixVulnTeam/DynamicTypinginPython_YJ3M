  74 class EnableProjectPluginTest(APITestCase):
  75     @mock.patch.object(NotificationPlugin, "test_configuration", side_effect="test_configuration")
  76     def test_simple(self, test_configuration):
  77         project = self.create_project()
  78 
  79         self.login_as(user=self.user)
  80 
  81         plugins.get("webhooks").disable(project)
  82 
  83         url = reverse(
  84             "sentry-api-0-project-plugin-details",
  85             kwargs={
  86                 "organization_slug": project.organization.slug,
  87                 "project_slug": project.slug,
  88                 "plugin_id": "webhooks",
  89             },
  90         )
  91         audit = AuditLogEntry.objects.filter(target_object=project.id)
  92 
  93         assert not audit
  94 
  95         response = self.client.post(url)
  96         audit = AuditLogEntry.objects.get(target_object=project.id)
  97 
  98         assert audit.event == 110
  99         assert response.status_code == 201, (response.status_code, response.content)
 100         assert ProjectOption.objects.get(key="webhooks:enabled", project=project).value is True
 101         audit.delete()
 102 
 103         # Testing the Plugin
 104         response = self.client.post(url, {"test": True})
 105         test_configuration.assert_called_once_with(project)
 106         assert response.status_code == 200, (response.status_code, response.content)
 107 
 108         # Reset the plugin
 109         response = self.client.post(url, {"reset": True})
 110         audit = AuditLogEntry.objects.get(target_object=project.id)
 111         test_configuration.assert_called_once_with(project)
 112         assert audit.event == 111
 113         assert response.status_code == 200, (response.status_code, response.content)
 114 
 115         configs = response.data.get("config")
 116 
 117         for config in configs:
 118             assert config.get("value") is None
