   8 class SentryAppInteractionTest(APITestCase):
   9     def setUp(self):
  10         self.superuser = self.create_user(email="superuser@example.com", is_superuser=True)
  11         self.user = self.create_user(email="user@example.com")
  12         self.org = self.create_organization(owner=self.user)
  13 
  14         self.published_app = self.create_sentry_app(
  15             name="Published App",
  16             organization=self.org,
  17             published=True,
  18             schema={"elements": [self.create_issue_link_schema()]},
  19         )
  20         self.unowned_published_app = self.create_sentry_app(
  21             name="Unowned Published App", organization=self.create_organization(), published=True
  22         )
  23         self.owned_url = reverse(
  24             "sentry-api-0-sentry-app-interaction", args=[self.published_app.slug]
  25         )
  26         self.unowned_url = reverse(
  27             "sentry-api-0-sentry-app-interaction", args=[self.unowned_published_app.slug]

  31 class GetSentryAppInteractionTest(SentryAppInteractionTest):
  32     def test_superuser_sees_unowned_interactions(self):
  33         self.login_as(user=self.superuser, superuser=True)
  34 
  35         response = self.client.get(self.unowned_url, format="json")

  37         assert len(response.data["views"]) > 0
  38         assert response.data["componentInteractions"] == {}
