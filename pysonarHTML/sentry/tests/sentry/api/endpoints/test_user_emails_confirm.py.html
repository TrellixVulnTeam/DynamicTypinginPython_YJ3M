  11 class UserEmailsConfirmTest(APITestCase):
  12     def setUp(self):
  13         self.user = self.create_user(email="foo@example.com")
  14         self.login_as(user=self.user)
  15 
  16     @mock.patch("sentry.models.User.send_confirm_email_singular")
  17     def test_can_confirm(self, send_confirm_email):
  18         email = UserEmail.objects.create(email="bar@example.com", is_verified=False, user=self.user)
  19         email.save()
  20         self.path = reverse("sentry-api-0-user-emails-confirm", kwargs={"user_id": self.user.id})
  21         resp = self.client.post(self.path, {"email": "bar@example.com"})
  22         assert resp.status_code == 204
  23         send_confirm_email.assert_called_once_with(UserEmail.objects.get(email="bar@example.com"))
  24 
  25     @mock.patch("sentry.models.User.send_confirm_email_singular")
  26     def test_cant_confirm_verified_email(self, send_confirm_email):
  27         email = UserEmail.objects.create(email="bar@example.com", is_verified=True, user=self.user)
  28         email.save()
  29         self.path = reverse("sentry-api-0-user-emails-confirm", kwargs={"user_id": self.user.id})
  30         resp = self.client.post(self.path, {"email": "bar@example.com"})
  31         assert resp.status_code == 400
  32         assert send_confirm_email.call_count == 0
  33 
  34     @mock.patch("sentry.models.User.send_confirm_email_singular")
  35     def test_validate_email(self, send_confirm_email):
  36         self.path = reverse("sentry-api-0-user-emails-confirm", kwargs={"user_id": self.user.id})
  37         resp = self.client.post(self.path, {"email": ""})
  38         assert resp.status_code == 400
  39         assert send_confirm_email.call_count == 0
