 216 class ReleaseSerializerTest(unittest.TestCase):
 217     def setUp(self):
 218         super(ReleaseSerializerTest, self).setUp()
 219         self.commits = [{"id": "a" * 40}, {"id": "b" * 40}]
 220         self.ref = "master"
 221         self.url = "https://example.com"
 222         self.dateReleased = "1000-10-10T06:06"
 223 
 224     def test_simple(self):
 225         serializer = ReleaseSerializer(
 226             data={
 227                 "ref": self.ref,
 228                 "url": self.url,
 229                 "dateReleased": self.dateReleased,
 230                 "commits": self.commits,
 231             }
 232         )
 233 
 234         assert serializer.is_valid()
 235         assert sorted(serializer.fields.keys()) == sorted(["ref", "url", "dateReleased", "commits"])
 236 
 237         result = serializer.validated_data
 238         assert result["ref"] == self.ref
 239         assert result["url"] == self.url
 240         assert result["dateReleased"] == datetime(1000, 10, 10, 6, 6, tzinfo=pytz.UTC)
 241         assert result["commits"] == self.commits
 242 
 243     def test_fields_not_required(self):
 244         serializer = ReleaseSerializer(data={})
 245         assert serializer.is_valid()
 246 
 247     def test_do_not_allow_null_commits(self):
 248         serializer = ReleaseSerializer(data={"commits": None})
 249         assert not serializer.is_valid()
 250 
 251     def test_ref_limited_by_max_version_length(self):
 252         serializer = ReleaseSerializer(data={"ref": "a" * MAX_VERSION_LENGTH})
 253         assert serializer.is_valid()
 254         serializer = ReleaseSerializer(data={"ref": "a" * (MAX_VERSION_LENGTH + 1)})
 255         assert not serializer.is_valid()
