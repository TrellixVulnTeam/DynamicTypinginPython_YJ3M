  11 class ApiAuthorizationsListTest(APITestCase):
  12     def test_simple(self):
  13         app = ApiApplication.objects.create(name="test", owner=self.user)
  14         auth = ApiAuthorization.objects.create(application=app, user=self.user)
  15         ApiAuthorization.objects.create(
  16             application=app, user=self.create_user("example@example.com")
  17         )
  18         self.login_as(self.user)
  19         url = reverse("sentry-api-0-api-authorizations")
  20         response = self.client.get(url)
  21         assert response.status_code == 200, response.content
  22         assert len(response.data) == 1
  23         assert response.data[0]["id"] == six.text_type(auth.id)

  26 class ApiAuthorizationsDeleteTest(APITestCase):
  27     def test_simple(self):
  28         app = ApiApplication.objects.create(name="test", owner=self.user)
  29         auth = ApiAuthorization.objects.create(application=app, user=self.user)
  30         token = ApiToken.objects.create(application=app, user=self.user)
  31         self.login_as(self.user)
  32         url = reverse("sentry-api-0-api-authorizations")
  33         response = self.client.delete(url, data={"authorization": auth.id})
  34         assert response.status_code == 204
  35         assert not ApiAuthorization.objects.filter(id=auth.id).exists()
  36         assert not ApiToken.objects.filter(id=token.id).exists()
