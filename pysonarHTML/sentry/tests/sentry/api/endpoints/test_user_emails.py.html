   9 class UserEmailsTest(APITestCase):
  10     def setUp(self):
  11         super(UserEmailsTest, self).setUp()
  12         self.user = self.create_user(email="foo@example.com")
  13         self.login_as(user=self.user)
  14         self.url = reverse("sentry-api-0-user-emails", kwargs={"user_id": self.user.id})
  15 
  16     def test_get_emails(self):
  17         UserEmail.objects.create(user=self.user, email="altemail1@example.com")
  18         UserEmail.objects.create(user=self.user, email="altemail2@example.com")
  19 
  20         response = self.client.get(self.url)
  21         assert response.status_code == 200, response.content
  22         assert len(response.data) == 3
  23 
  24         primary_email = [n for n in response.data if n["isPrimary"]]
  25         assert len(primary_email) == 1
  26         assert primary_email[0]["email"] == "foo@example.com"
  27 
  28         secondary_emails = [n for n in response.data if not n["isPrimary"]]
  29         assert len(secondary_emails) == 2

  31     def test_add_secondary_email(self):
  32         # test invalid email
  33         response = self.client.post(self.url, data={"email": "invalidemail"})
  34         assert response.status_code == 400, response.data
  35         assert not UserEmail.objects.filter(user=self.user, email="invalidemail").exists()
  36 
  37         # valid secondary email
  38         response = self.client.post(self.url, data={"email": "altemail1@example.com"})
  39 
  40         assert response.status_code == 201, response.data
  41         assert UserEmail.objects.filter(user=self.user, email="altemail1@example.com").exists()
  42 
  43         # duplicate email allows still succeeds, but has a diff response code
  44         response = self.client.post(self.url, data={"email": "altemail1@example.com"})
  45         assert response.status_code == 200, response.data

  47     def test_change_verified_secondary_to_primary(self):
  48         UserEmail.objects.create(user=self.user, email="altemail1@example.com", is_verified=True)
  49         response = self.client.put(self.url, data={"email": "altemail1@example.com"})
  50         assert response.status_code == 200, response.data
  51 
  52         user = User.objects.get(id=self.user.id)
  53         assert user.email == "altemail1@example.com"
  54         assert user.username == "altemail1@example.com"
  55 
  56     def test_change_unverified_secondary_to_primary(self):
  57         UserEmail.objects.create(user=self.user, email="altemail1@example.com", is_verified=False)
  58         response = self.client.put(self.url, data={"email": "altemail1@example.com"})
  59         assert response.status_code == 400, response.data
  60 
  61         user = User.objects.get(id=self.user.id)
  62         assert user.email != "altemail1@example.com"
  63         assert user.username != "altemail1@example.com"
  64 
  65     def test_remove_email(self):

  67 
  68         response = self.client.delete(self.url, data={"email": "altemail1@example.com"})
  69         assert response.status_code == 204, response.data
  70         assert not len(UserEmail.objects.filter(user=self.user, email="altemail1@example.com"))

  72     def test_cant_remove_primary_email(self):
