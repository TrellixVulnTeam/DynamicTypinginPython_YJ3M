  10 class ShortIdLookupEndpointTest(APITestCase):
  11     def test_simple(self):
  12         org = self.create_organization(owner=self.user)
  13         project = self.create_project(organization=org)
  14         group = self.create_group(project=project, short_id=project.next_short_id())
  15 
  16         self.login_as(user=self.user)
  17         url = reverse(
  18             "sentry-api-0-short-id-lookup",
  19             kwargs={"organization_slug": org.slug, "short_id": group.qualified_short_id},
  20         )
  21         response = self.client.get(url, format="json")
  22 
  23         assert response.status_code == 200, response.content
  24         assert response.data["organizationSlug"] == org.slug
  25         assert response.data["projectSlug"] == project.slug
  26         assert response.data["groupId"] == six.text_type(group.id)
