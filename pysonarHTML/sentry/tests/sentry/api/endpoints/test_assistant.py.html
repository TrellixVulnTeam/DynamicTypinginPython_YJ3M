  14 class AssistantActivityTest(APITestCase):
  15     def setUp(self):
  16         super(AssistantActivityTest, self).setUp()
  17         self.login_as(user=self.user)
  18         self.path = reverse("sentry-api-0-assistant")
  19         self.guides = manager.all()
  20 
  21     def test_invalid_inputs(self):
  22         # Missing status
  23         resp = self.client.put(self.path, {"guide_id": 1})
  24         assert resp.status_code == 400
  25 
  26         # Invalid guide id
  27         resp = self.client.put(self.path, {"guide_id": 1938, "status": "dismissed"})
  28         assert resp.status_code == 400
  29 
  30         # Invalid status
  31         resp = self.client.put(self.path, {"guide_id": 1, "status": "whats_my_name_again"})
  32         assert resp.status_code == 400
  33 
  34     def test_activity(self):
  35         guides_with_seen = deepcopy(manager.all())
  36         for g in guides_with_seen:
  37             guides_with_seen[g]["seen"] = False
  38 
  39         resp = self.client.get(self.path)
  40         assert resp.status_code == 200
  41         assert resp.data == guides_with_seen
  42 
  43         # Dismiss the guide and make sure it is not returned again.
  44         resp = self.client.put(self.path, {"guide_id": 3, "status": "dismissed"})
  45         assert resp.status_code == 201
  46         resp = self.client.get(self.path)
  47         guides_with_seen["issue_stream"]["seen"] = True
  48         assert resp.status_code == 200
  49         assert resp.data == guides_with_seen

  51     def test_validate_guides(self):
  52         # Steps in different guides should not have the same target.
  53         guides = self.guides.values()
  54         for i in range(len(guides)):
  55             for j in range(0, i):
  56                 steps_i = set(s["target"] for s in guides[i]["steps"])
  57                 steps_j = set(s["target"] for s in guides[j]["steps"])
  58                 assert not (steps_i & steps_j)
