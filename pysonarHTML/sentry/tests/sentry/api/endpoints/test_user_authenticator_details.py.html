 232     def test_recovery_codes_regenerate(self, email_log):
 233         interface = RecoveryCodeInterface()
 234         interface.enroll(self.user)
 235 
 236         url = reverse(
 237             "sentry-api-0-user-authenticator-details",
 238             kwargs={"user_id": self.user.id, "auth_id": interface.authenticator.id},
 239         )
 240 
 241         resp = self.client.get(url)
 242         assert resp.status_code == 200
 243         old_codes = resp.data["codes"]
 244         old_created_at = resp.data["createdAt"]
 245 
 246         resp = self.client.get(url)
 247         assert old_codes == resp.data["codes"]
 248         assert old_created_at == resp.data["createdAt"]
 249 
 250         # regenerate codes
 251         tomorrow = timezone.now() + datetime.timedelta(days=1)
 252         with mock.patch.object(timezone, "now", return_value=tomorrow):
 253             resp = self.client.put(url)
 254 
 255             resp = self.client.get(url)
 256             assert old_codes != resp.data["codes"]
 257             assert old_created_at != resp.data["createdAt"]
 258 
 259         self._assert_security_email_sent("recovery-codes-regenerated", email_log)

 264         new_options["sms.twilio-account"] = "twilio-account"

 266

 271             )
 272

 275 
 276             url = reverse(
 277                 "sentry-api-0-user-authenticator-details",
