   9 class ProjectEnvironmentsTest(APITestCase):
  10     def test_simple(self):
  11         project = self.create_project()
  12 
  13         env1 = Environment.objects.create(
  14             project_id=project.id, organization_id=project.organization_id, name="production"
  15         )
  16         env1.add_project(project)
  17 
  18         env2 = Environment.objects.create(
  19             project_id=project.id, organization_id=project.organization_id, name="staging"
  20         )
  21         env2.add_project(project)
  22 
  23         self.login_as(user=self.user)
  24 
  25         url = reverse(
  26             "sentry-api-0-project-environments",
  27             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
  28         )
  29         response = self.client.get(url, format="json")
  30         assert response.status_code == 200, response.content
  31         assert len(response.data) == 2
  32         assert response.data[0]["name"] == "production"
  33         assert response.data[1]["name"] == "staging"

  35     def test_visibility_filtering(self):
  36         project = self.create_project()
  37 
  38         env1 = Environment.objects.create(
  39             project_id=project.id, organization_id=project.organization_id, name="production"
  40         )
  41 
  42         EnvironmentProject.objects.create(project=project, environment=env1, is_hidden=False)
  43 
  44         env2 = Environment.objects.create(
  45             project_id=project.id, organization_id=project.organization_id, name="staging"
  46         )
  47 
  48         EnvironmentProject.objects.create(project=project, environment=env2, is_hidden=True)
  49 
  50         self.login_as(user=self.user)
  51 
  52         url = reverse(
  53             "sentry-api-0-project-environments",
  54             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
  55         )
  56         response = self.client.get(url, format="json")
  57         assert response.status_code == 200, response.content
  58         assert len(response.data) == 1
  59         assert response.data[0]["name"] == "production"
  60 
  61         response = self.client.get(url + "?visibility=all", format="json")
  62         assert response.status_code == 200, response.content
  63         assert len(response.data) == 2
  64         assert response.data[0]["name"] == "production"
  65         assert response.data[1]["name"] == "staging"
  66 
  67         response = self.client.get(url + "?visibility=hidden", format="json")
  68         assert response.status_code == 200, response.content
  69         assert len(response.data) == 1
  70         assert response.data[0]["name"] == "staging"
