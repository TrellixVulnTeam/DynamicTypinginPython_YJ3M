  22 class AcceptTransferProjectTest(APITestCase):
  23     def setUp(self):
  24         super(AcceptTransferProjectTest, self).setUp()
  25         self.owner = self.create_user(email="example@example.com", is_superuser=False)
  26         self.from_organization = self.create_organization(owner=self.owner)
  27         self.to_organization = self.create_organization(owner=self.owner)
  28         self.from_team = self.create_team(organization=self.from_organization)
  29         self.to_team = self.create_team(organization=self.to_organization)
  30         user = self.create_user("admin@example.com")
  31         self.member = self.create_member(
  32             organization=self.from_organization, user=user, role="admin", teams=[self.from_team]
  33         )
  34         self.project = self.create_project(name="proj", teams=[self.from_team])
  35         self.transaction_id = uuid4().hex
  36         self.path = reverse("sentry-api-0-accept-project-transfer")

  38     def test_requires_authentication(self):
  39         response = self.client.get(self.path)
  40         assert response.status_code == 403
  41         assert response.data == {"detail": "Authentication credentials were not provided."}

  43     def test_handle_incorrect_url_data(self):
  44         self.login_as(self.owner)
  45         url_data = sign(
  46             actor_id=self.member.id,
  47             # This is bad data
  48             from_organization_id=9999999,
  49             project_id=self.project.id,
  50             user_id=self.owner.id,
  51             transaction_id=self.transaction_id,
  52         )
  53         resp = self.client.get(self.path + "?" + urlencode({"data": url_data}))
  54         assert resp.status_code == 400
  55         assert resp.data["detail"] == "Project no longer exists"
  56         resp = self.client.get(self.path)
  57         assert resp.status_code == 404

  59     def test_returns_org_options_with_signed_link(self):
  60         self.login_as(self.owner)
  61         url_data = sign(
  62             actor_id=self.member.user_id,
  63             from_organization_id=self.from_organization.id,
  64             project_id=self.project.id,
  65             user_id=self.owner.id,
  66             transaction_id=self.transaction_id,
  67         )
  68 
  69         resp = self.client.get(self.path + "?" + urlencode({"data": url_data}))
  70         assert resp.status_code == 200
  71         assert resp.data["project"]["slug"] == self.project.slug
  72         assert resp.data["project"]["id"] == self.project.id
  73         assert len(resp.data["organizations"]) == 2
  74         org_slugs = {o["slug"] for o in resp.data["organizations"]}
  75         assert self.from_organization.slug in org_slugs
  76         assert self.to_organization.slug in org_slugs

  78     def test_transfers_project_to_correct_team(self):
