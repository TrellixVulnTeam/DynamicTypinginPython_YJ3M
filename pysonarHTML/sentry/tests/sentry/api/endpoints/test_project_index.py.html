 142     def test_valid_with_internal_integration(self):
 143         project = self.create_project(organization=self.organization, teams=[self.team])
 144         self.create_internal_integration(
 145             name="my_app",
 146             organization=self.organization,
 147             scopes=("project:read",),
 148             webhook_url="http://example.com",
 149         )
 150         # there should only be one record created so just grab the first one
 151         token = SentryAppInstallationToken.objects.first()
 152         response = self.client.get(
 153             u"{}".format(self.path), HTTP_AUTHORIZATION=u"Bearer {}".format(token.api_token.token)
 154         )
 155         assert project.name in response.content
