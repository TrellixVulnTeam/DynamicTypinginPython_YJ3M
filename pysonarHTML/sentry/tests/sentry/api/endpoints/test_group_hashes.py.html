  53     def test_return_multiple_hashes(self):
  54         self.login_as(user=self.user)
  55 
  56         min_ago = iso_format(before_now(minutes=1))
  57         two_min_ago = iso_format(before_now(minutes=2))
  58 
  59         event1 = self.store_event(
  60             data={
  61                 "event_id": "a" * 32,
  62                 "message": "message",
  63                 "timestamp": two_min_ago,
  64                 "stacktrace": copy.deepcopy(DEFAULT_EVENT_DATA["stacktrace"]),
  65                 "fingerprint": ["group-1"],
  66             },
  67             project_id=self.project.id,
  68         )
  69 
  70         event2 = self.store_event(
  71             data={
  72                 "event_id": "b" * 32,
  73                 "message": "message2",
  74                 "timestamp": min_ago,
  75                 "fingerprint": ["group-2"],
  76             },
  77             project_id=self.project.id,
  78         )
  79 
  80         # Merge the events
  81         eventstream = SnubaEventStream()
  82         state = eventstream.start_merge(self.project.id, [event2.group_id], event1.group_id)
  83 
  84         eventstream.end_merge(state)
  85 
  86         url = u"/api/0/issues/{}/hashes/".format(event1.group_id)
  87         response = self.client.get(url, format="json")
  88 
  89         assert response.status_code == 200, response.content
  90         assert len(response.data) == 2
  91 
  92         primary_hashes = [hash["id"] for hash in response.data]
  93         assert primary_hashes == [event2.get_primary_hash(), event1.get_primary_hash()]

  95     def test_unmerge(self):
  96         self.login_as(user=self.user)
  97 
  98         group = self.create_group()
  99 
 100         hashes = [
 101             GroupHash.objects.create(project=group.project, group=group, hash=hash)
 102             for hash in ["a" * 32, "b" * 32]
 103         ]
 104 
 105         url = "?".join(
 106             [
 107                 u"/api/0/issues/{}/hashes/".format(group.id),
 108                 urlencode({"id": [h.hash for h in hashes]}, True),
 109             ]
 110         )
 111 
 112         response = self.client.delete(url, format="json")
 113         assert response.status_code == 202, response.content
