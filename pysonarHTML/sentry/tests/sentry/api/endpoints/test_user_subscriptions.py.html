  13 @pytest.mark.skipIf(
  14     lambda x: settings.SENTRY_NEWSLETTER != "sentry.newsletter.dummy.DummyNewsletter"
  15 )
  16 class UserSubscriptionsNewsletterTest(APITestCase):
  17     def setUp(self):
  18         self.user = self.create_user(email="foo@example.com")
  19         self.login_as(self.user)
  20         self.url = reverse("sentry-api-0-user-subscriptions", kwargs={"user_id": self.user.id})
  21 
  22         def disable_newsletter():
  23             newsletter.backend.disable()
  24 
  25         self.addCleanup(disable_newsletter)
  26         newsletter.backend.enable()
  27 
  28     def test_get_subscriptions(self):
  29         response = self.client.get(self.url)
  30         assert response.status_code == 200, response.content
  31 
  32     def test_subscribe(self):
  33         response = self.client.put(self.url, data={"listId": "123", "subscribed": True})
  34         assert response.status_code == 204, response.content
  35         results = newsletter.get_subscriptions(self.user)["subscriptions"]
  36         assert len(results) == 1
  37         assert results[0].list_id == 123
  38         assert results[0].subscribed
  39         assert results[0].verified
  40 
  41     def test_requires_subscribed(self):
  42         response = self.client.put(self.url, data={"listId": "123"})
  43         assert response.status_code == 400, response.content
  44 
  45     def test_unverified_emails(self):
  46         UserEmail.objects.get(email=self.user.email).update(is_verified=False)
  47         response = self.client.put(self.url, data={"listId": "123", "subscribed": True})
  48         assert response.status_code == 204, response.content
  49 
  50     def test_unsubscribe(self):
  51         response = self.client.put(self.url, data={"listId": "123", "subscribed": False})
  52         assert response.status_code == 204, response.content
  53         results = newsletter.get_subscriptions(self.user)["subscriptions"]
  54         assert len(results) == 1
  55         assert results[0].list_id == 123
  56         assert not results[0].subscribed
  57         assert results[0].verified
  58 
  59     def test_default_subscription(self):
  60         response = self.client.post(self.url, data={"subscribed": True})
  61         assert response.status_code == 204, response.content
  62         results = newsletter.get_subscriptions(self.user)["subscriptions"]
  63         assert len(results) == 1
  64         assert results[0].list_id == newsletter.get_default_list_id()
  65         assert results[0].subscribed
  66         assert results[0].verified
