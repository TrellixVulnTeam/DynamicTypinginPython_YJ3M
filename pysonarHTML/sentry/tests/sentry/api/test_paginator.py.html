 309     def test_same_row_updated(self):
 310         joined = timezone.now()
 311         res1 = self.create_user("foo@example.com", date_joined=joined)
 312         queryset = User.objects.all()
 313 
 314         paginator = DateTimePaginator(queryset, "-date_joined")
 315         result1 = paginator.get_result(limit=3, cursor=None)
 316         assert len(result1) == 1, result1
 317         assert result1[0] == res1
 318 
 319         # Prev page should return no results
 320         result2 = paginator.get_result(limit=3, cursor=result1.prev)
 321         assert len(result2) == 0, result2
 322 
 323         # If the same row has an updated join date then it should
 324         # show up on the prev page
 325         res1.update(date_joined=joined + timedelta(seconds=1))
 326         result3 = paginator.get_result(limit=3, cursor=result1.prev)
 327         assert len(result3) == 1, result3
 328         assert result3[0] == res1
 329 
 330         # Make sure updates work as expected with extra rows
 331         res1.update(date_joined=res1.date_joined + timedelta(seconds=1))
 332         res2 = self.create_user(
 333             "bar@example.com", date_joined=res1.date_joined + timedelta(seconds=1)
 334         )
 335         res3 = self.create_user(
 336             "baz@example.com", date_joined=res1.date_joined + timedelta(seconds=2)
 337         )
 338         res4 = self.create_user(
 339             "bat@example.com", date_joined=res1.date_joined + timedelta(seconds=3)
 340         )
 341         result4 = paginator.get_result(limit=1, cursor=result3.prev)
 342         assert len(result4) == 1, result4
 343         assert result4[0] == res1
 344 
 345         result5 = paginator.get_result(limit=3, cursor=result3.prev)
 346         assert len(result5) == 3, result5
 347         assert result5[0] == res3
 348         assert result5[1] == res2
 349         assert result5[2] == res1
 350 
 351         result6 = paginator.get_result(limit=3, cursor=result5.prev)
 352         assert len(result6) == 1, result6
 353         assert result6[0] == res4
 354 
 355         res4.update(date_joined=res4.date_joined + timedelta(seconds=1))
 356         result7 = paginator.get_result(limit=3, cursor=result6.prev)
 357         assert len(result7) == 1, result7
 358         assert result7[0] == res4
