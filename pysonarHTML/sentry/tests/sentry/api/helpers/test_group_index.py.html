  69 class UpdateGroupsTest(TestCase):
  70     @patch("sentry.signals.issue_ignored.send_robust")
  71     def test_unresolving_resolved_group(self, send_robust):
  72         resolved_group = self.create_group(status=GroupStatus.RESOLVED)
  73         assert resolved_group.status == GroupStatus.RESOLVED
  74 
  75         request = self.make_request(user=self.user, method="GET")
  76         request.user = self.user
  77         request.data = {"status": "unresolved"}
  78         request.GET = QueryDict(query_string="id={}".format(resolved_group.id))
  79 
  80         search_fn = Mock()
  81         update_groups(request, [self.project], self.organization.id, search_fn)
  82 
  83         resolved_group.refresh_from_db()
  84 
  85         assert resolved_group.status == GroupStatus.UNRESOLVED
  86         assert not send_robust.called
