  12 class GetDateRangeFromParamsTest(TestCase):
  13     def test_stats_period(self):
  14         start, end = get_date_range_from_params({"statsPeriod": "14h"})
  15         assert end - datetime.timedelta(hours=14) == start
  16 
  17         start, end = get_date_range_from_params({"statsPeriod": "14d"})
  18         assert end - datetime.timedelta(days=14) == start
  19 
  20         start, end = get_date_range_from_params({"statsPeriod": "60m"})
  21         assert end - datetime.timedelta(minutes=60) == start
  22 
  23         start, end = get_date_range_from_params({"statsPeriod": "3600s"})
  24         assert end - datetime.timedelta(seconds=3600) == start
  25 
  26         start, end = get_date_range_from_params({"statsPeriod": "91d"})
  27         assert end - datetime.timedelta(days=91) == start

  29     def test_date_range(self):
  30         start, end = get_date_range_from_params({"start": "2018-11-01", "end": "2018-11-07"})
  31 
  32         assert start == datetime.datetime(2018, 11, 1, tzinfo=timezone.utc)
  33         assert end == datetime.datetime(2018, 11, 7, tzinfo=timezone.utc)

  35     @freeze_time("2018-12-11 03:21:34")
