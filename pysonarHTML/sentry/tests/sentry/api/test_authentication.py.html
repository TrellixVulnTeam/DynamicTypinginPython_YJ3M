  71 class TestDSNAuthentication(TestCase):
  72     def setUp(self):
  73         super(TestDSNAuthentication, self).setUp()
  74 
  75         self.auth = DSNAuthentication()
  76         self.org = self.create_organization(owner=self.user)
  77         self.project = self.create_project(organization=self.org)
  78         self.project_key = self.create_project_key(project=self.project)
  79 
  80     def test_authenticate(self):
  81         request = HttpRequest()
  82         request.META["HTTP_AUTHORIZATION"] = u"DSN {}".format(self.project_key.dsn_public)
  83 
  84         result = self.auth.authenticate(request)
  85         assert result is not None
  86 
  87         user, auth = result
  88         assert user.is_anonymous()
  89         assert auth == self.project_key
  90 
  91     def test_inactive_key(self):

  95 
  96         with pytest.raises(AuthenticationFailed):
