  25 class ParseSearchQueryTest(TestCase):
  26     def test_key_mappings(self):
  27         # Test a couple of keys to ensure things are working as expected
  28         assert parse_search_query("bookmarks:123") == [
  29             SearchFilter(
  30                 key=SearchKey(name="bookmarked_by"), operator="=", value=SearchValue("123")
  31             )
  32         ]
  33         assert parse_search_query("first-release:123") == [
  34             SearchFilter(
  35                 key=SearchKey(name="first_release"), operator="=", value=SearchValue("123")
  36             )
  37         ]
  38         assert parse_search_query("first-release:123 non_mapped:456") == [
  39             SearchFilter(
  40                 key=SearchKey(name="first_release"), operator="=", value=SearchValue("123")
  41             ),
  42             SearchFilter(key=SearchKey(name="non_mapped"), operator="=", value=SearchValue("456")),
  43         ]

  45     def test_is_query_unassigned(self):
  46         assert parse_search_query("is:unassigned") == [
  47             SearchFilter(key=SearchKey(name="unassigned"), operator="=", value=SearchValue(True))
  48         ]
  49         assert parse_search_query("is:assigned") == [
  50             SearchFilter(key=SearchKey(name="unassigned"), operator="=", value=SearchValue(False))
  51         ]
  52 
  53         assert parse_search_query("!is:unassigned") == [
  54             SearchFilter(key=SearchKey(name="unassigned"), operator="!=", value=SearchValue(True))
  55         ]
  56         assert parse_search_query("!is:assigned") == [
  57             SearchFilter(key=SearchKey(name="unassigned"), operator="!=", value=SearchValue(False))
  58         ]

  60     def test_is_query_status(self):
  61         for status_string, status_val in STATUS_CHOICES.items():
  62             assert parse_search_query("is:%s" % status_string) == [
  63                 SearchFilter(
  64                     key=SearchKey(name="status"), operator="=", value=SearchValue(status_val)
  65                 )
  66             ]
  67             assert parse_search_query("!is:%s" % status_string) == [
  68                 SearchFilter(
  69                     key=SearchKey(name="status"), operator="!=", value=SearchValue(status_val)
  70                 )
  71             ]

  73     def test_is_query_invalid(self):

  78             'Invalid value for "is" search, valid values are'

  81     def test_numeric_filter(self):
  82         # test numeric format
  83         assert parse_search_query("times_seen:500") == [
  84             SearchFilter(
  85                 key=SearchKey(name="times_seen"), operator="=", value=SearchValue(raw_value=500)
  86             )
  87         ]
  88         assert parse_search_query("times_seen:>500") == [
  89             SearchFilter(
  90                 key=SearchKey(name="times_seen"), operator=">", value=SearchValue(raw_value=500)
  91             )
  92         ]
  93         assert parse_search_query("times_seen:<500") == [
  94             SearchFilter(
  95                 key=SearchKey(name="times_seen"), operator="<", value=SearchValue(raw_value=500)
  96             )
  97         ]
  98         invalid_queries = [
  99             "times_seen:<hello",
 100             "times_seen:<512.1.0",
 101             "times_seen:2018-01-01",
 102             "times_seen:+7d",
 103             "times_seen:>2018-01-01",
 104             'times_seen:"<10"',
 105         ]
 106         for invalid_query in invalid_queries:
 107             with self.assertRaises(
 108                 InvalidSearchQuery, expected_regex="Invalid format for numeric search"
 109             ):
 110                 parse_search_query(invalid_query)

 112     def test_boolean_operators_not_allowed(self):
 113         invalid_queries = [
 114             "user.email:foo@example.com OR user.email:bar@example.com",
 115             "user.email:foo@example.com AND user.email:bar@example.com",
 116             "user.email:foo@example.com OR user.email:bar@example.com OR user.email:foobar@example.com",
 117             "user.email:foo@example.com AND user.email:bar@example.com AND user.email:foobar@example.com",
 118         ]
 119         for invalid_query in invalid_queries:
 120             with self.assertRaises(
 121                 InvalidSearchQuery,
 122                 expected_regex='Boolean statements containing "OR" or "AND" are not supported in this search',
 123             ):
 124                 parse_search_query(invalid_query)

 127 class ConvertQueryValuesTest(TestCase):
 128     def test_valid_converter(self):
 129         filters = [SearchFilter(SearchKey("assigned_to"), "=", SearchValue("me"))]
 130         expected = value_converters["assigned_to"](
 131             filters[0].value.raw_value, [self.project], self.user, None
 132         )
 133         filters = convert_query_values(filters, [self.project], self.user, None)
 134         assert filters[0].value.raw_value == expected
 135 
 136     def test_no_converter(self):
 137         search_val = SearchValue("me")

 139         filters = convert_query_values(filters, [self.project], self.user, None)
 140         assert filters[0].value.raw_value == search_val.raw_value

 143 class ConvertStatusValueTest(TestCase):
 144     def test_valid(self):
 145         for status_string, status_val in STATUS_CHOICES.items():
 146             filters = [SearchFilter(SearchKey("status"), "=", SearchValue(status_string))]
 147             result = convert_query_values(filters, [self.project], self.user, None)
 148             assert result[0].value.raw_value == status_val
 149 
 150             filters = [SearchFilter(SearchKey("status"), "=", SearchValue(status_val))]
 151             result = convert_query_values(filters, [self.project], self.user, None)
 152             assert result[0].value.raw_value == status_val
 153 
 154     def test_invalid(self):
 155         filters = [SearchFilter(SearchKey("status"), "=", SearchValue("wrong"))]
 156         with self.assertRaises(InvalidSearchQuery, expected_regex="invalid status value"):
 157             convert_query_values(filters, [self.project], self.user, None)
 158 
 159         filters = [AggregateFilter(AggregateKey("count_unique(user)"), ">", SearchValue("1"))]
 160         with self.assertRaises(
 161             InvalidSearchQuery,
 162             expected_regex="Aggregate filters (count_unique(user)) are not supported in issue searches.",
 163         ):
 164             convert_query_values(filters, [self.project], self.user, None)
