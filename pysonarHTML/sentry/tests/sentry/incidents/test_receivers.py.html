  15 class AddProjectToIncludeAllRulesTest(TestCase):
  16     def test_include_all_projects_enabled(self):
  17         alert_rule = self.create_alert_rule(include_all_projects=True)
  18         new_project = self.create_project()

  35 class PreSaveIncidentTriggerTest(TestCase):
  36     def test_update_date_modified(self):
  37         org = Organization.objects.create(name="chris' test org")
  38         alert_rule = self.create_alert_rule(include_all_projects=False)
  39         alert_rule.query = "event.type:error"
  40         trigger = AlertRuleTrigger.objects.create(
  41             alert_rule=alert_rule,
  42             label="warning",
  43             threshold_type=0,
  44             alert_threshold=100,
  45             resolve_threshold=50,
  46         )
  47         incident = Incident.objects.create(
  48             organization=org,
  49             detection_uuid=None,
  50             status=IncidentStatus.WARNING.value,
  51             type=2,
  52             title="a custom incident title",
  53             date_started=datetime.utcnow().replace(tzinfo=pytz.utc),
  54             date_detected=datetime.utcnow().replace(tzinfo=pytz.utc),
  55             alert_rule=alert_rule,
  56         )
  57         incident_trigger = IncidentTrigger.objects.create(
  58             incident=incident, alert_rule_trigger=trigger, status=TriggerStatus.ACTIVE.value
  59         )
  60         date_modified = incident_trigger.date_modified
  61         incident_trigger.status = TriggerStatus.RESOLVED.value
  62         incident_trigger.save()
  63         incident_trigger.refresh_from_db()
  64         assert date_modified < incident_trigger.date_modified
