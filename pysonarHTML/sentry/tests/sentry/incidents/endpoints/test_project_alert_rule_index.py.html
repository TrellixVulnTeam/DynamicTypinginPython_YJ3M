 131 class ProjectCombinedRuleIndexEndpointTest(BaseAlertRuleSerializerTest, TestCase):
 132     def setup_project_and_rules(self):
 133         self.org = self.create_organization(owner=self.user, name="Rowdy Tiger")
 134         self.team = self.create_team(organization=self.org, name="Mariachi Band")
 135         self.project = self.create_project(organization=self.org, teams=[self.team], name="Bengal")
 136         self.login_as(self.user)
 137         self.projects = [self.project, self.create_project()]
 138         self.alert_rule = self.create_alert_rule(
 139             projects=self.projects, date_added=before_now(minutes=6).replace(tzinfo=pytz.UTC)
 140         )
 141         self.other_alert_rule = self.create_alert_rule(
 142             projects=self.projects, date_added=before_now(minutes=5).replace(tzinfo=pytz.UTC)
 143         )
 144         self.issue_rule = self.create_issue_alert_rule(
 145             data={
 146                 "project": self.project,
 147                 "name": "Issue Rule Test",
 148                 "conditions": [],
 149                 "actions": [],
 150                 "actionMatch": "all",
 151                 "date_added": before_now(minutes=4).replace(tzinfo=pytz.UTC),
 152             }
 153         )
 154         self.yet_another_alert_rule = self.create_alert_rule(
 155             projects=self.projects, date_added=before_now(minutes=3).replace(tzinfo=pytz.UTC)
 156         )
 157         self.combined_rules_url = "/api/0/projects/{0}/{1}/combined-rules/".format(
 158             self.org.slug, self.project.slug
 159         )

 269     def test_offset_pagination(self):
 270         self.setup_project_and_rules()
 271 
 272         date_added = before_now(minutes=1)
 273         self.one_alert_rule = self.create_alert_rule(
 274             projects=self.projects, date_added=date_added.replace(tzinfo=pytz.UTC)
 275         )
 276         self.two_alert_rule = self.create_alert_rule(
 277             projects=self.projects, date_added=date_added.replace(tzinfo=pytz.UTC)
 278         )
 279         self.three_alert_rule = self.create_alert_rule(projects=self.projects)
 280 
 281         with self.feature("organizations:incidents"):
 282             request_data = {"per_page": "2"}
 283             response = self.client.get(
 284                 path=self.combined_rules_url, data=request_data, content_type="application/json"
 285             )
 286         assert response.status_code == 200
 287 
 288         result = json.loads(response.content)
 289         assert len(result) == 2
 290         self.assert_alert_rule_serialized(self.three_alert_rule, result[0], skip_dates=True)
 291         self.assert_alert_rule_serialized(self.one_alert_rule, result[1], skip_dates=True)
 292 
 293         links = requests.utils.parse_header_links(
 294             response.get("link").rstrip(">").replace(">,<", ",<")
 295         )
 296         next_cursor = links[1]["cursor"]
 297         assert next_cursor.split(":")[1] == "1"  # Assert offset is properly calculated.
 298 
 299         with self.feature("organizations:incidents"):
 300             request_data = {"cursor": next_cursor, "per_page": "2"}
 301             response = self.client.get(
 302                 path=self.combined_rules_url, data=request_data, content_type="application/json"
 303             )
 304         assert response.status_code == 200
 305 
 306         result = json.loads(response.content)
 307         assert len(result) == 2
 308 
 309         self.assert_alert_rule_serialized(self.two_alert_rule, result[0], skip_dates=True)
 310         self.assert_alert_rule_serialized(self.yet_another_alert_rule, result[1], skip_dates=True)
