 262     def test_invalid_slack_channel(self):
 263         # We had an error where an invalid slack channel was spitting out unclear
 264         # error for the user, and CREATING THE RULE. So the next save (after fixing slack action)
 265         # says "Name already in use". This test makes sure that is not happening anymore.
 266         # We save a rule with an invalid slack, make sure we get back a useful error
 267         # and that the rule is not created.
 268         base_params = self.valid_params.copy()
 269         base_params["name"] = "Aun1qu3n4m3"
 270         integration = Integration.objects.create(
 271             external_id="1",
 272             provider="slack",
 273             metadata={"access_token": "xoxp-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx"},
 274         )
 275         integration.add_organization(self.organization, self.user)
 276         base_params["triggers"][0]["actions"].append(
 277             {
 278                 "type": AlertRuleTriggerAction.get_registered_type(
 279                     AlertRuleTriggerAction.Type.SLACK
 280                 ).slug,
 281                 "targetType": action_target_type_to_string[
 282                     AlertRuleTriggerAction.TargetType.SPECIFIC
 283                 ],
 284                 "targetIdentifier": "123",
 285                 "integration": six.text_type(integration.id),
 286             }
 287         )
 288         serializer = AlertRuleSerializer(context=self.context, data=base_params)
 289         assert serializer.is_valid()
 290         with self.assertRaises(serializers.ValidationError):
 291             serializer.save()
 292 
 293         # Make sure the rule was not created.
 294         assert len(list(AlertRule.objects.filter(name="Aun1qu3n4m3"))) == 0
 295 
 296         # Make sure the action was not created.
 297         alert_rule_trigger_actions = list(
 298             AlertRuleTriggerAction.objects.filter(integration=integration)
 299         )
 300         assert len(alert_rule_trigger_actions) == 0
