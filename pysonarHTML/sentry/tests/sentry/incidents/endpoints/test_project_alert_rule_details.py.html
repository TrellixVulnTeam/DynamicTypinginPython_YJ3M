 160     def test_not_updated_fields(self):

 162         test_params["aggregate"] = self.alert_rule.snuba_query.aggregate
 163 
 164         self.create_member(
 165             user=self.user, organization=self.organization, role="owner", teams=[self.team]

 183     def test_update_snapshot(self):

 186         )
 187         self.login_as(self.user)
 188         alert_rule = self.alert_rule

 192         # Archive the rule so that the endpoint 404s:

 197             self.get_valid_response(
 198                 self.organization.slug,
 199                 self.project.slug,

 223     def test_snapshot_and_create_new_with_same_name(self):
 224         with self.tasks():
 225             self.create_member(
 226                 user=self.user, organization=self.organization, role="owner", teams=[self.team]
 227             )
 228             self.login_as(self.user)
 229 
 230             # We attach the rule to an incident so the rule is snapshotted instead of deleted.
 231             incident = self.create_incident(alert_rule=self.alert_rule)
 232 
 233             with self.feature("organizations:incidents"):
 234                 self.get_valid_response(
 235                     self.organization.slug, self.project.slug, self.alert_rule.id, status_code=204
 236                 )
 237 
 238             alert_rule = AlertRule.objects_with_snapshots.get(id=self.alert_rule.id)
 239 
 240             assert not AlertRule.objects.filter(id=alert_rule.id).exists()
 241             assert AlertRule.objects_with_snapshots.filter(id=alert_rule.id).exists()
 242             assert alert_rule.status == AlertRuleStatus.SNAPSHOT.value
 243 
 244             # We also confirm that the incident is automatically resolved.
 245             assert Incident.objects.get(id=incident.id).status == IncidentStatus.CLOSED.value
