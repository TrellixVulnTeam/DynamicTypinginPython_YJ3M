 280 class IncidentCreationTest(TestCase):
 281     def test_simple(self):
 282         title = "hello"
 283         alert_rule = self.create_alert_rule()
 284         incident = Incident.objects.create(
 285             self.organization,
 286             title=title,
 287             type=IncidentType.ALERT_TRIGGERED.value,
 288             alert_rule=alert_rule,
 289         )
 290         assert incident.identifier == 1
 291         assert incident.title == title
 292 
 293         # Check identifier correctly increments
 294         incident = Incident.objects.create(
 295             self.organization,
 296             title=title,
 297             type=IncidentType.ALERT_TRIGGERED.value,
 298             alert_rule=alert_rule,
 299         )
 300         assert incident.identifier == 2
 301 
 302     def test_identifier_conflict(self):
 303         create_method = BaseManager.create
 304         call_count = [0]
 305         alert_rule = self.create_alert_rule()
 306 
 307         def mock_base_create(*args, **kwargs):
 308             if not call_count[0]:
 309                 call_count[0] += 1
 310                 # This incident will take the identifier we already fetched and
 311                 # use it, which will cause the database to throw an integrity
 312                 # error.
 313                 with transaction.atomic():
 314                     incident = Incident.objects.create(
 315                         self.organization,
 316                         status=IncidentStatus.OPEN.value,
 317                         title="Conflicting Incident",
 318                         type=IncidentType.ALERT_TRIGGERED.value,
 319                         alert_rule=alert_rule,
 320                     )
 321                 assert incident.identifier == kwargs["identifier"]
 322                 try:
 323                     create_method(*args, **kwargs)
 324                 except IntegrityError:
 325                     raise
 326                 else:
 327                     self.fail("Expected an integrity error")
 328             else:
 329                 call_count[0] += 1
 330 
 331             return create_method(*args, **kwargs)
 332 
 333         self.organization
 334         with patch.object(BaseManager, "create", new=mock_base_create):
 335             incident = Incident.objects.create(
 336                 self.organization,
 337                 alert_rule=alert_rule,
 338                 status=IncidentStatus.OPEN.value,
 339                 title="hi",
 340                 type=IncidentType.ALERT_TRIGGERED.value,
 341             )
 342             # We should have 3 calls - one for initial create, one for conflict,
 343             # then the final one for the retry we get due to the conflict
 344             assert call_count[0] == 3
 345             # Ideally this would be 2, but because we create the conflicting
 346             # row inside of a transaction it ends up rolled back. We just want
 347             # to verify that it was created successfully.
 348             assert incident.identifier == 1
