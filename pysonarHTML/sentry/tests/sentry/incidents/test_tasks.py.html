  45 class TestSendSubscriberNotifications(BaseIncidentActivityTest, TestCase):
  46     send_async = patcher("sentry.utils.email.MessageBuilder.send_async")
  47 
  48     def test_simple(self):
  49         activity = create_incident_activity(
  50             self.incident, IncidentActivityType.COMMENT, user=self.user, comment="hello"
  51         )
  52         send_subscriber_notifications(activity.id)
  53         # User shouldn't receive an email for their own activity
  54         self.send_async.assert_not_called()  # NOQA
  55 
  56         self.send_async.reset_mock()
  57         non_member_user = self.create_user(email="non_member@test.com")
  58         subscribe_to_incident(activity.incident, non_member_user)
  59 
  60         member_user = self.create_user(email="member@test.com")
  61         self.create_member([self.team], user=member_user, organization=self.organization)
  62         subscribe_to_incident(activity.incident, member_user)
  63         send_subscriber_notifications(activity.id)
  64         self.send_async.assert_called_once_with([member_user.email])
  65         assert not IncidentSubscription.objects.filter(
  66             incident=activity.incident, user=non_member_user
  67         ).exists()
  68         assert IncidentSubscription.objects.filter(
  69             incident=activity.incident, user=member_user
  70         ).exists()

  96 class TestBuildActivityContext(BaseIncidentActivityTest, TestCase):
  97     def run_test(
  98         self, activity, expected_username, expected_action, expected_comment, expected_recipient
  99     ):
 100         incident = activity.incident
 101         context = build_activity_context(activity, expected_recipient)
 102         assert context["user_name"] == expected_username
 103         assert context["action"] == "%s on alert %s (#%s)" % (
 104             expected_action,
 105             activity.incident.title,
 106             activity.incident.identifier,
 107         )
 108         assert (
 109             context["link"]
 110             == absolute_uri(
 111                 reverse(
 112                     "sentry-metric-alert",
 113                     kwargs={
 114                         "organization_slug": incident.organization.slug,
 115                         "incident_id": incident.identifier,
 116                     },
 117                 )
 118             )
 119             + "?referrer=incident_activity_email"
 120         )
 121         assert context["comment"] == expected_comment

 204 class ProcessPendingIncidentSnapshots(TestCase):
 205     def test_simple(self):
 206         incident = self.create_incident(title="incident", status=IncidentStatus.CLOSED.value)
 207         pending = PendingIncidentSnapshot.objects.create(
 208             incident=incident, target_run_date=timezone.now()
 209         )
 210 
 211         assert IncidentSnapshot.objects.all().count() == 0
 212 
 213         with self.tasks():
 214             process_pending_incident_snapshots()
 215 
 216         assert not PendingIncidentSnapshot.objects.filter(id=pending.id).exists()
 217         assert IncidentSnapshot.objects.filter(incident=incident).exists()
 218 
 219     def test_skip_open_incident(self):
 220         incident = self.create_incident(title="incident", status=IncidentStatus.OPEN.value)
 221         pending = PendingIncidentSnapshot.objects.create(
 222             incident=incident, target_run_date=timezone.now()
 223         )
 224         assert IncidentSnapshot.objects.all().count() == 0
 225 
 226         with self.tasks():
 227             process_pending_incident_snapshots()
 228 
 229         # The PendingSnapshot should be deleted, and a Snapshot should not be created because the incident is open.
 230         assert not PendingIncidentSnapshot.objects.filter(id=pending.id).exists()
 231         assert not IncidentSnapshot.objects.filter(incident=incident).exists()
 232 
 233     def test_skip_future_run_date(self):
 234         incident_1 = self.create_incident(title="incident1", status=IncidentStatus.CLOSED.value)
 235         incident_2 = self.create_incident(title="incident2", status=IncidentStatus.CLOSED.value)
 236         pending_1 = PendingIncidentSnapshot.objects.create(
 237             incident=incident_1, target_run_date=timezone.now()
 238         )
 239         pending_2 = PendingIncidentSnapshot.objects.create(
 240             incident=incident_2, target_run_date=timezone.now() + timedelta(minutes=5)
 241         )
 242 
 243         assert IncidentSnapshot.objects.all().count() == 0
 244 
 245         with self.tasks():
 246             process_pending_incident_snapshots()
 247 
 248         # Should only process the one with target_run_date <= timezone.now()
 249         assert not PendingIncidentSnapshot.objects.filter(id=pending_1.id).exists()
 250         assert PendingIncidentSnapshot.objects.filter(id=pending_2.id).exists()
 251 
 252         assert IncidentSnapshot.objects.filter(incident=incident_1).exists()
 253         assert not IncidentSnapshot.objects.filter(incident=incident_2).exists()
 254 
 255     def test_skip_because_existing_snapshot(self):
 256         incident = self.create_incident(title="incident1", status=IncidentStatus.CLOSED.value)
 257         pending_1 = PendingIncidentSnapshot.objects.create(
 258             incident=incident, target_run_date=timezone.now()
 259         )
 260 
 261         assert IncidentSnapshot.objects.all().count() == 0
 262 
 263         with self.tasks():
 264             process_pending_incident_snapshots()
 265 
 266         assert not PendingIncidentSnapshot.objects.filter(id=pending_1.id).exists()
 267         assert IncidentSnapshot.objects.filter(incident=incident).exists()
 268         assert IncidentSnapshot.objects.all().count() == 1
 269 
 270         # Have to create it here otherwise the unique constraint will cause this to fail:
 271         pending_2 = PendingIncidentSnapshot.objects.create(
 272             incident=incident, target_run_date=timezone.now()
 273         )
 274         with self.tasks():
 275             process_pending_incident_snapshots()
 276 
 277         assert not PendingIncidentSnapshot.objects.filter(id=pending_2.id).exists()
 278         assert IncidentSnapshot.objects.filter(incident=incident).exists()
 279         assert IncidentSnapshot.objects.filter(incident=incident).count() == 1
 280         assert IncidentSnapshot.objects.all().count() == 1
