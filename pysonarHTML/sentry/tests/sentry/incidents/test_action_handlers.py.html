 140 class EmailActionHandlerFireTest(TestCase):
 141     def test_user(self):
 142         action = self.create_alert_rule_trigger_action(
 143             target_identifier=six.text_type(self.user.id)
 144         )
 145         incident = self.create_incident(status=IncidentStatus.CRITICAL.value)
 146         handler = EmailActionHandler(action, incident, self.project)
 147         with self.tasks():
 148             handler.fire(1000)
 149         out = mail.outbox[0]
 150         assert out.to == [self.user.email]
 151         assert out.subject == u"[Critical] {} - {}".format(incident.title, self.project.slug)

 158             target_identifier=six.text_type(self.user.id)

 170 @freeze_time()
 171 class SlackActionHandlerBaseTest(object):
 172     @responses.activate
 173     def run_test(self, incident, method):
 174         token = "xoxp-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx"
 175         integration = Integration.objects.create(
 176             external_id="1", provider="slack", metadata={"access_token": token}
 177         )
 178         integration.add_organization(self.organization, self.user)
 179         channel_id = "some_id"
 180         channel_name = "#hello"
 181         responses.add(
 182             method=responses.GET,
 183             url="https://slack.com/api/channels.list",
 184             status=200,
 185             content_type="application/json",
 186             body=json.dumps(
 187                 {"ok": "true", "channels": [{"name": channel_name[1:], "id": channel_id}]}
 188             ),
 189         )
 190 
 191         action = self.create_alert_rule_trigger_action(
 192             target_identifier=channel_name,
 193             type=AlertRuleTriggerAction.Type.SLACK,
 194             target_type=AlertRuleTriggerAction.TargetType.SPECIFIC,
 195             integration=integration,
 196         )
 197         responses.add(
 198             method=responses.POST,
 199             url="https://slack.com/api/chat.postMessage",
 200             status=200,
 201             content_type="application/json",
 202             body='{"ok": true}',
 203         )
 204         handler = SlackActionHandler(action, incident, self.project)
 205         metric_value = 1000
 206         with self.tasks():
 207             getattr(handler, method)(metric_value)
 208         data = parse_qs(responses.calls[1].request.body)
 209         assert data["channel"] == [channel_id]
 210         assert data["token"] == [token]
 211         assert json.loads(data["attachments"][0])[0] == build_incident_attachment(
 212             incident, metric_value
 213         )
