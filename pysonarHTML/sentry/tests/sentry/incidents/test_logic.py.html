 220         event_id = uuid4().hex

 265     def validate_result(self, incident, result, expected_results, start, end, windowed_stats):
 266         # Duration of 300s, but no alert rule
 267         time_window = incident.alert_rule.snuba_query.time_window if incident.alert_rule else 60
 268         assert result.rollup == time_window
 269         expected_start = start if start else incident.date_started - timedelta(seconds=time_window)
 270         expected_end = end if end else incident.current_end_date + timedelta(seconds=time_window)
 271 
 272         if windowed_stats:
 273             now = timezone.now()
 274             expected_end = expected_start + timedelta(
 275                 seconds=time_window * (WINDOWED_STATS_DATA_POINTS / 2)
 276             )
 277             expected_start = expected_start - timedelta(
 278                 seconds=time_window * (WINDOWED_STATS_DATA_POINTS / 2)
 279             )
 280             if expected_end > now:
 281                 expected_end = now
 282                 expected_start = now - timedelta(seconds=time_window * WINDOWED_STATS_DATA_POINTS)
 283 
 284         assert result.start == expected_start
 285         assert result.end == expected_end
 286         assert [r["count"] for r in result.data["data"]] == expected_results
