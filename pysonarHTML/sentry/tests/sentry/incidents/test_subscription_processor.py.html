 110     def build_subscription_update(self, subscription, time_delta=None, value=None):
 111         if time_delta is not None:
 112             timestamp = timezone.now() + time_delta
 113         else:
 114             timestamp = timezone.now()
 115         timestamp = timestamp.replace(tzinfo=pytz.utc, microsecond=0)
 116 
 117         data = {}
 118 
 119         if subscription:
 120             data = {"some_col_name": randint(0, 100) if value is None else value}
 121         values = {"data": [data]}
 122         return {
 123             "subscription_id": subscription.subscription_id if subscription else uuid4().hex,
 124             "values": values,
 125             "timestamp": timestamp,
 126             "interval": 1,
 127             "partition": 1,
 128             "offset": 1,
 129         }

 131     def send_update(self, rule, value, time_delta=None, subscription=None):
 132         self.email_action_handler.reset_mock()
 133         if time_delta is None:
 134             time_delta = timedelta()
 135         if subscription is None:
 136             subscription = self.sub
 137         processor = SubscriptionProcessor(subscription)
 138         message = self.build_subscription_update(subscription, value=value, time_delta=time_delta)
 139         processor.process_update(message)
 140         return processor

 432     def test_multiple_subscriptions_do_not_conflict(self):
 433         # Verify that multiple subscriptions associated with a rule don't conflict with
 434         # each other
 435         rule = self.rule
 436         rule.update(threshold_period=2)
 437         trigger = self.trigger
 438 
 439         # Send an update through for the first subscription. This shouldn't trigger an
 440         # incident, since we need two consecutive updates that are over the threshold.
 441         processor = self.send_update(
 442             rule, trigger.alert_threshold + 1, timedelta(minutes=-10), subscription=self.sub
 443         )
 444         self.assert_trigger_counts(processor, self.trigger, 1, 0)
 445         self.assert_no_active_incident(rule, self.sub)
 446         self.assert_trigger_does_not_exist(self.trigger)
 447         self.assert_action_handler_called_with_actions(None, [])
 448 
 449         # Have an update come through for the other sub. This shouldn't influence the original
 450         processor = self.send_update(
 451             rule, trigger.alert_threshold + 1, timedelta(minutes=-9), subscription=self.other_sub
 452         )
 453         self.assert_trigger_counts(processor, self.trigger, 1, 0)
 454         self.assert_no_active_incident(rule, self.sub)
 455         self.assert_trigger_does_not_exist(self.trigger)
 456         self.assert_no_active_incident(rule, self.other_sub)
 457         self.assert_trigger_does_not_exist(self.trigger)
 458         self.assert_action_handler_called_with_actions(None, [])
 459 
 460         # Send another update through for the first subscription. This should trigger an
 461         # incident for just this subscription.
 462         processor = self.send_update(
 463             rule, trigger.alert_threshold + 1, timedelta(minutes=-9), subscription=self.sub
 464         )
 465         self.assert_trigger_counts(processor, self.trigger, 0, 0)
 466         incident = self.assert_active_incident(rule, self.sub)
 467         self.assert_trigger_exists_with_status(incident, self.trigger, TriggerStatus.ACTIVE)
 468         self.assert_actions_fired_for_incident(incident, [self.action])
 469         self.assert_no_active_incident(rule, self.other_sub)
 470         self.assert_trigger_does_not_exist(self.trigger, [incident])
 471 
 472         # Send another update through for the second subscription. This should trigger an
 473         # incident for just this subscription.
 474         processor = self.send_update(
 475             rule, trigger.alert_threshold + 1, timedelta(minutes=-8), subscription=self.other_sub
 476         )
 477         self.assert_trigger_counts(processor, self.trigger, 0, 0)
 478         incident = self.assert_active_incident(rule, self.sub)
 479         self.assert_trigger_exists_with_status(incident, self.trigger, TriggerStatus.ACTIVE)
 480         self.assert_action_handler_called_with_actions(incident, [])
 481         other_incident = self.assert_active_incident(rule, self.other_sub)
 482         self.assert_trigger_exists_with_status(other_incident, self.trigger, TriggerStatus.ACTIVE)
 483         self.assert_actions_fired_for_incident(other_incident, [self.action], self.other_project)
 484 
 485         # Now we want to test that resolving is isolated. Send another update through
 486         # for the first subscription.
 487         processor = self.send_update(
 488             rule, trigger.resolve_threshold - 1, timedelta(minutes=-7), subscription=self.sub
 489         )
 490         self.assert_trigger_counts(processor, self.trigger, 0, 1)
 491         incident = self.assert_active_incident(rule, self.sub)
 492         self.assert_trigger_exists_with_status(incident, self.trigger, TriggerStatus.ACTIVE)
 493         self.assert_action_handler_called_with_actions(incident, [])
 494         other_incident = self.assert_active_incident(rule, self.other_sub)

 615         self.assert_trigger_exists_with_status(incident, trigger, TriggerStatus.RESOLVED)
 616         self.assert_trigger_exists_with_status(incident, other_trigger, TriggerStatus.RESOLVED)
