   7 class ErrorEventTest(TestCase):
   8     def test_get_metadata(self):
   9         inst = ErrorEvent()
  10         data = {"exception": {"values": [{"type": "Exception", "value": "Foo"}]}}
  11         assert inst.get_metadata(data) == {"type": "Exception", "value": "Foo"}

  13     def test_get_metadata_none(self):

  16         assert inst.get_metadata(data) == {"type": "Error", "value": ""}
