 295     def test_generates_list_ids_for_registered_types(self):
 296         build_message = functools.partial(
 297             MessageBuilder, subject="Test", body="hello world", html_body="<b>hello world</b>"
 298         )
 299 
 300         expected = u"{event.project.slug}.{event.organization.slug}.{namespace}".format(
 301             event=self.event, namespace=options.get("mail.list-namespace")
 302         )
 303 
 304         references = (self.event.group, self.event.project, self.activity)
 305 
 306         for reference in references:
 307             (message,) = build_message(reference=reference).get_built_messages(["foo@example.com"])
 308             assert message.message()["List-Id"] == expected
