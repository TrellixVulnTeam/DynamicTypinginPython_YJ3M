 282 def test_format_bytes():
 283     assert format_bytes(50) == "50 B"
 284     assert format_bytes(1024) == "1.00 KB"
 285     assert format_bytes(3000) == "2.93 KB"
 286     assert format_bytes(30000) == "29.30 KB"
 287     assert format_bytes(3000000) == "2.86 MB"
 288     assert format_bytes(3000000000) == "2.79 GB"
 289     assert format_bytes(3000000000000) == "2.73 TB"
 290 
 291     assert format_bytes(3000000000000, units=["B", "KB", "MB", "GB"]) == "2793.97 GB"
