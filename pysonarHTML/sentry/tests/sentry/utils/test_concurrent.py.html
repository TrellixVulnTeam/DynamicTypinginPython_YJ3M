 174 def test_threaded_executor():
 175     executor = ThreadedExecutor(worker_count=1, maxsize=3)
 176 
 177     def waiter(ready, waiting, result):
 178         ready.set()
 179         waiting.wait()
 180         return result
 181 
 182     initial_ready = Event()
 183     initial_waiting = Event()
 184     initial_future = executor.submit(
 185         lambda: waiter(initial_ready, initial_waiting, 1), block=True, timeout=1
 186     )
 187 
 188     # wait until the worker has removed this item from the queue
 189     assert initial_ready.wait(timeout=1), "waiter not ready"
 190     assert initial_future.running(), "waiter did not get marked as started"
 191 
 192     low_priority_ready = Event()
 193     low_priority_waiting = Event()
 194     low_priority_future = executor.submit(
 195         lambda: waiter(low_priority_ready, low_priority_waiting, 2),
 196         block=True,
 197         timeout=1,
 198         priority=10,
 199     )
 200     assert not low_priority_future.done(), "future should not be done (indicative of a full queue)"
 201 
 202     cancelled_future = executor.submit(lambda: None, block=True, timeout=1, priority=5)
 203     assert not cancelled_future.done(), "future should not be done (indicative of a full queue)"
 204     assert cancelled_future.cancel(), "future should be able to be cancelled"
 205     assert cancelled_future.done(), "future should be completed"
 206     with pytest.raises(CancelledError):
 207         cancelled_future.result()
 208 
 209     high_priority_ready = Event()
 210     high_priority_waiting = Event()
 211     high_priority_future = executor.submit(
 212         lambda: waiter(high_priority_ready, high_priority_waiting, 3),
 213         block=True,
 214         timeout=1,
 215         priority=0,
 216     )
 217     assert not high_priority_future.done(), "future should not be done (indicative of a full queue)"
 218 
 219     queue_full_future = executor.submit(lambda: None, block=False)
 220     assert queue_full_future.done()
 221     with pytest.raises(Full):
 222         queue_full_future.result()  # will not block if completed
 223 
 224     initial_waiting.set()  # let the task finish
 225     assert initial_future.result(timeout=1) == 1
 226     assert initial_future.done()
 227 
 228     assert high_priority_ready.wait(timeout=1)  # this should be the next task to execute
 229     assert high_priority_future.running()
 230     assert not low_priority_future.running()
 231 
 232     high_priority_waiting.set()  # let the task finish
 233     assert high_priority_future.result(timeout=1) == 3
 234     assert high_priority_future.done()
 235 
 236     assert low_priority_ready.wait(timeout=1)  # this should be the next task to execute
 237     assert low_priority_future.running()
 238 
 239     low_priority_waiting.set()  # let the task finish
 240     assert low_priority_future.result(timeout=1) == 2
 241     assert low_priority_future.done()
