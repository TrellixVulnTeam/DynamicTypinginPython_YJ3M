  20 def test_codec_lookup():
  21     def assert_match(enc, ref=None):
  22         if ref is None:
  23             ref = enc
  24         rv = codec_lookup(enc)
  25         if rv is None:
  26             assert ref is rv is None
  27         else:
  28             assert rv.name == ref
  29 
  30     assert codec_lookup("utf-8").name == "utf-8"
  31     assert codec_lookup("utf8").name == "utf-8"
  32     if sys.version_info[:3] >= (2, 7, 12):
  33         assert codec_lookup("zlib").name == "utf-8"
  34     assert codec_lookup("utf16").name == "utf-16"
  35     assert codec_lookup("undefined").name == "utf-8"
  36     assert codec_lookup("undefined", default=None) is None
  37     assert codec_lookup("undefined", default="latin1").name == "iso8859-1"
  38     if sys.version_info[:3] >= (2, 7, 12):
  39         assert codec_lookup("zlib", default="latin1").name == "iso8859-1"
  40     assert codec_lookup("unknownshit", default="latin1").name == "iso8859-1"
