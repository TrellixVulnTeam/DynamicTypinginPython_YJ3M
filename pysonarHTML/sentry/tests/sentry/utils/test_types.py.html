   5 from sentry.utils.types import InvalidTypeError, Any, Bool, Int, Float, String, Dict, Sequence
   6 from unittest import TestCase

   8 
   9 class OptionsTypesTest(TestCase):
  10     def test_any(self):
  11         assert Any("foo") == "foo"
  12         assert Any(1) == 1
  13         assert Any(None) is None
  14         assert Any() is None
  15         assert Any.test(None)
  16         assert Any.test("foo")
  17         assert Any.test("bar")
  18 
  19     def test_bool(self):
  20         assert Bool(True) is True
  21         assert Bool(1) is True
  22         assert Bool("y") is True
  23         assert Bool("YES") is True
  24         assert Bool("t") is True
  25         assert Bool("true") is True
  26         assert Bool("1") is True
  27         assert Bool("on") is True
  28         assert Bool(False) is False
  29         assert Bool(0) is False
  30         assert Bool("n") is False
  31         assert Bool("NO") is False
  32         assert Bool("f") is False
  33         assert Bool("false") is False
  34         assert Bool("0") is False
  35         assert Bool("off") is False
  36         assert Bool() is False
  37         assert Bool.test(None) is False
  38         assert Bool(True) is True
  39         assert Bool.test("foo") is False
  40         with self.assertRaises(InvalidTypeError):
  41             Bool("foo")
  42 
  43     def test_int(self):
  44         assert Int(1) == 1
  45         assert Int("1") == 1
  46         assert Int("-1") == -1
  47         assert Int() == 0
  48         with self.assertRaises(InvalidTypeError):
  49             Int("foo")
  50         with self.assertRaises(InvalidTypeError):
  51             Int("1.1")
  52 
  53     def test_float(self):
  54         assert Float(1.0) == 1.0
  55         assert Float("1") == 1.0
  56         assert Float("-1.1") == -1.1
  57         assert Float(1) == 1.0
  58         assert Float() == 0.0
  59         with self.assertRaises(InvalidTypeError):
  60             Float("foo")
  61 
  62     def test_string(self):
  63         assert String("foo") == "foo"
  64         assert String(u"foo") == u"foo"
  65         assert String() == u""
  66         with self.assertRaises(InvalidTypeError):
  67             String(0)
  68 
  69     def test_dict(self):
  70         assert Dict({}) == {}
  71         assert Dict({"foo": "bar"}) == {"foo": "bar"}
  72         assert Dict("{foo: bar}") == {"foo": "bar"}
  73 
  74         assert Dict() == {}
  75         with self.assertRaises(InvalidTypeError):
  76             assert Dict("[]")
  77         with self.assertRaises(InvalidTypeError):
  78             assert Dict([])
  79         with self.assertRaises(InvalidTypeError):
  80             assert Dict("")
  81         with self.assertRaises(InvalidTypeError):
  82             # malformed yaml/json (a plain scalar, "b: ar", cannot contain ": ")
  83             assert Dict("{foo: b: ar}")
  84 
  85     def test_sequence(self):
  86         assert Sequence(()) == ()
  87         assert Sequence([]) == []
  88         assert Sequence((1, 2, 3)) == (1, 2, 3)
  89         assert Sequence([1, 2, 3]) == [1, 2, 3]
  90         assert Sequence("[1,2,3]") == (1, 2, 3)
  91         with self.assertRaises(InvalidTypeError):
  92             Sequence("{}")
  93         with self.assertRaises(InvalidTypeError):
  94             Sequence({})
  95         with self.assertRaises(InvalidTypeError):
  96             Sequence("")
  97         with self.assertRaises(InvalidTypeError):
  98             # malformed yaml/json
  99             Sequence("[1,")
