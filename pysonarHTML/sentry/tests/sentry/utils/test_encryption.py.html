   9 class EncryptionManagerTest(TestCase):
  10     def test_simple(self):
  11         manager = EncryptionManager(
  12             schemes=(("1", Fernet("J5NxyG0w1OyZEDdEOX0Nyv2upm5H3J35rTEb1jEiVbs=")),)
  13         )
  14         value = manager.encrypt("hello world")
  15         assert value.startswith(u"{}1$".format(MARKER))
  16         result = manager.decrypt(value)
  17         assert result == "hello world"
  18 
  19         manager = EncryptionManager(
  20             schemes=(
  21                 ("2", Fernet(Fernet.generate_key())),
  22                 ("1", Fernet("J5NxyG0w1OyZEDdEOX0Nyv2upm5H3J35rTEb1jEiVbs=")),
  23             )
  24         )
  25 
  26         # this should use the first scheme
  27         result = manager.decrypt(value)
  28         assert result == "hello world"
  29 
  30         value2 = manager.encrypt("hello world")
  31         assert value2 != value
  32         assert value2.startswith(u"{}2$".format(MARKER))
  33 
  34     def test_no_schemes(self):
  35         manager = EncryptionManager(schemes=())
  36         value = manager.encrypt("hello world")
