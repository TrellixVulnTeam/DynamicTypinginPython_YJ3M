  77     def test_audit_entry_org_restore_log(self):
  78         Organization.objects.filter(id=self.organization.id).update(
  79             status=OrganizationStatus.PENDING_DELETION
  80         )
  81 
  82         org = Organization.objects.get(id=self.organization.id)
  83 
  84         Organization.objects.filter(id=self.organization.id).update(
  85             status=OrganizationStatus.DELETION_IN_PROGRESS
  86         )
  87 
  88         org2 = Organization.objects.get(id=self.organization.id)
  89 
  90         Organization.objects.filter(id=self.organization.id).update(
  91             status=OrganizationStatus.VISIBLE
  92         )
  93 
  94         org3 = Organization.objects.get(id=self.organization.id)
  95 
  96         orgs = [org, org2, org3]
  97 
  98         entry = create_audit_entry(
  99             request=self.req,
 100             organization=self.org,
 101             target_object=self.org.id,
 102             event=AuditLogEntryEvent.ORG_RESTORE,
 103             data=self.org.get_audit_log_data(),
 104         )
 105 
 106         entry2 = create_audit_entry(
 107             request=self.req,
 108             organization=self.org,
 109             target_object=self.org.id,
 110             event=AuditLogEntryEvent.ORG_EDIT,
 111             data=self.org.get_audit_log_data(),
 112         )
 113 
 114         for i in orgs:
 115             if (
 116                 i.status == OrganizationStatus.PENDING_DELETION
 117                 or i.status == OrganizationStatus.DELETION_IN_PROGRESS
 118             ):
 119                 assert i.status != OrganizationStatus.VISIBLE
 120                 assert ("restored") in entry.get_note()
 121                 assert entry.actor == self.user
 122                 assert entry.target_object == self.org.id
 123                 assert entry.event == AuditLogEntryEvent.ORG_RESTORE
 124             else:
 125                 assert i.status == OrganizationStatus.VISIBLE
 126                 assert ("edited") in entry2.get_note()
 127                 assert entry2.actor == self.user
 128                 assert entry2.target_object == self.org.id
 129                 assert entry2.event == AuditLogEntryEvent.ORG_EDIT
