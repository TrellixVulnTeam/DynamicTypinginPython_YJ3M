  52     def test_translation(self):
  53         # Case 1: No translation
  54         filter_keys = {"sdk": ["python", "js"]}
  55         forward, reverse = get_snuba_translators(filter_keys)
  56         assert forward(filter_keys) == filter_keys
  57         result = [{"sdk": "python", "count": 123}, {"sdk": "js", "count": 234}]
  58         assert all(reverse(row) == row for row in result)
  59 
  60         # Case 2: Environment ID -> Name and back
  61         filter_keys = {"environment": [self.proj1env1.id]}
  62         forward, reverse = get_snuba_translators(filter_keys)
  63         assert forward(filter_keys) == {"environment": [self.proj1env1.name]}
  64         row = {"environment": self.proj1env1.name, "count": 123}
  65         assert reverse(row) == {"environment": self.proj1env1.id, "count": 123}
  66 
  67         # Case 3, both Environment and Release
  68         filter_keys = {
  69             "environment": [self.proj1env1.id],
  70             "tags[sentry:release]": [self.release1.id],
  71         }
  72         forward, reverse = get_snuba_translators(filter_keys)
  73         assert forward(filter_keys) == {
  74             "environment": [self.proj1env1.name],
  75             "tags[sentry:release]": [self.release1.version],
  76         }
  77         row = {
  78             "environment": self.proj1env1.name,
  79             "tags[sentry:release]": self.release1.version,
  80             "count": 123,
  81         }
  82         assert reverse(row) == {
  83             "environment": self.proj1env1.id,
  84             "tags[sentry:release]": self.release1.id,
  85             "count": 123,
  86         }
  87 
  88         # Case 4: 2 Groups, many-to-many mapping of Groups
  89         # to Releases. Reverse translation depends on multiple
  90         # fields.
  91         filter_keys = {
  92             "group_id": [self.proj1group1.id, self.proj1group2.id],
  93             "tags[sentry:release]": [
  94                 self.group1release1.id,
  95                 self.group1release2.id,
  96                 self.group2release1.id,
  97             ],
  98         }
  99         forward, reverse = get_snuba_translators(filter_keys, is_grouprelease=True)
 100         assert forward(filter_keys) == {
 101             "group_id": [self.proj1group1.id, self.proj1group2.id],
 102             "tags[sentry:release]": [
 103                 self.release1.version,
 104                 self.release2.version,
 105                 self.release1.version,  # Duplicated because 2 GroupReleases refer to it
 106             ],
 107         }
 108         result = [
 109             {
 110                 "group_id": self.proj1group1.id,
 111                 "tags[sentry:release]": self.release1.version,
 112                 "count": 1,
 113             },

 210     def test_cache_suffix_time(self):
 211         starting_key = quantize_time(self.now, 0)
 212         finishing_key = quantize_time(self.now + timedelta(seconds=300), 0)
 213

 216     def test_quantize_hour_edges(self):
 217         """ a suffix should still behave correctly around the end of the hour
 218 
 219             At a duration of 10 only one key between 0-10 should flip on the hour, the other 9
 220             should flip at different times.
 221         """
 222         before = datetime(2019, 9, 5, 17, 59, 59)
 223         on_hour = datetime(2019, 9, 5, 18, 0, 0)
 224         changed_on_hour = 0
 225         # Check multiple keyhashes so that this test doesn't depend on implementation
 226         for key_hash in range(10):
 227             before_key = quantize_time(before, key_hash, duration=10)
 228             on_key = quantize_time(on_hour, key_hash, duration=10)
 229             if before_key != on_key:
 230                 changed_on_hour += 1
 231 
 232         assert changed_on_hour == 1
 233 
 234     def test_quantize_day_edges(self):
 235         """ a suffix should still behave correctly around the end of a day
 236 
 237             This test is nearly identical to test_quantize_hour_edges, but is to confirm that date changes don't
 238             cause a different behaviour
 239         """
 240         before = datetime(2019, 9, 5, 23, 59, 59)
 241         next_day = datetime(2019, 9, 6, 0, 0, 0)
 242         changed_on_hour = 0
 243         for key_hash in range(10):
 244             before_key = quantize_time(before, key_hash, duration=10)
 245             next_key = quantize_time(next_day, key_hash, duration=10)
 246             if before_key != next_key:
 247                 changed_on_hour += 1
 248 
 249         assert changed_on_hour == 1
 250 
 251     def test_quantize_time_matches_duration(self):
 252         """ The number of seconds between keys changing should match duration """
 253         previous_key = quantize_time(self.now, 0, duration=10)
 254         changes = []
 255         for i in range(21):
 256             current_time = self.now + timedelta(seconds=i)
 257             current_key = quantize_time(current_time, 0, duration=10)
 258             if current_key != previous_key:
 259                 changes.append(current_time)
 260                 previous_key = current_key
 261 
 262         assert len(changes) == 2
 263         assert (changes[1] - changes[0]).total_seconds() == 10
 264 
 265     def test_quantize_time_jitter(self):
 266         """ Different key hashes should change keys at different times
 267 
 268             While starting_key and other_key might begin as the same values they should change at different times
 269         """
 270         starting_key = quantize_time(self.now, 0, duration=10)
 271         for i in range(11):

 277         for j in range(11):
 278             current_key = quantize_time(self.now + timedelta(seconds=j), 5, duration=10)
 279             if current_key != other_key:
 280                 break
