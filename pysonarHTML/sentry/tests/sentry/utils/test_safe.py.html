  16 class TrimTest(unittest.TestCase):
  17     def test_simple_string(self):
  18         assert trim(a_very_long_string) == a_very_long_string[:509] + "..."
  19 
  20     def test_list_of_strings(self):
  21         assert trim([a_very_long_string, a_very_long_string]) == [a_very_long_string[:507] + "..."]
  22 
  23     def test_nonascii(self):
  24         assert trim({"x": "\xc3\xbc"}) == {"x": "\xc3\xbc"}
  25         assert trim(["x", "\xc3\xbc"]) == ["x", "\xc3\xbc"]
  26 
  27     def test_idempotent(self):
  28         trm = partial(trim, max_depth=2)
  29         a = {"a": {"b": {"c": {"d": 1}}}}
  30         assert trm(a) == {"a": {"b": {"c": '{"d":1}'}}}
  31         assert trm(trm(trm(trm(a)))) == trm(a)
  32 
  33     def test_sorted_trim(self):
  34         # Trim should always trim the keys in alpha order
  35         # regardless of the original order.
  36         alpha = OrderedDict([("a", "12345"), ("z", "12345")])
  37         reverse = OrderedDict([("z", "12345"), ("a", "12345")])
  38         trm = partial(trim, max_size=12)
  39         expected = {"a": "12345", "z": "1..."}
  40 
  41         assert trm(alpha) == expected
  42         assert trm(reverse) == expected
  43 
  44     def test_max_depth(self):
  45         trm = partial(trim, max_depth=2)
  46         a = {"a": {"b": {"c": "d"}}}
  47         assert trm(a) == a
  48 
  49         a = {"a": {"b": {"c": u"d"}}}
  50         assert trm(a) == {"a": {"b": {"c": "d"}}}
  51 
  52         a = {"a": {"b": {"c": {u"d": u"e"}}}}
  53         assert trm(a) == {"a": {"b": {"c": '{"d":"e"}'}}}
  54 
  55         a = {"a": {"b": {"c": []}}}
  56         assert trm(a) == {"a": {"b": {"c": "[]"}}}

 108 class GetPathTest(unittest.TestCase):
 109     def test_get_none(self):
 110         assert get_path(None, "foo") is None
 111         assert get_path("foo", "foo") is None
 112         assert get_path(42, "foo") is None
 113         assert get_path(ValueError(), "foo") is None
 114         assert get_path(True, "foo") is None
 115 
 116     def test_get_path_dict(self):
 117         assert get_path({}, "a") is None
 118         assert get_path({"a": 2}, "a") == 2
 119         assert get_path({"a": 2}, "b") is None
 120         assert get_path({"a": {"b": []}}, "a", "b") == []
 121         assert get_path({"a": []}, "a", "b") is None
 122         assert get_path(CanonicalKeyDict({"a": 2}), "a") == 2
 123 
 124     def test_get_default(self):
 125         assert get_path({"a": 2}, "b", default=1) == 1
 126         assert get_path({"a": 2}, "a", default=1) == 2
 127         assert get_path({"a": None}, "a", default=1) == 1
 128 
 129     def test_get_path_list(self):
 130         arr = [1, 2]
 131         assert get_path(arr, 1) == 2
 132         assert get_path(arr, -1) == 2
 133         assert get_path(arr, 2) is None
 134         assert get_path(arr, "1") is None
 135         assert get_path([], 1) is None
 136 
 137     def test_filter_list(self):
 138         data = {"a": [False, 1, None]}
 139         assert get_path(data, "a", filter=True) == [False, 1]
 140         assert get_path(data, "a", filter=lambda x: x) == [1]
 141 
 142     def test_filter_tuple(self):
 143         data = {"a": (False, 1, None)}
 144         assert get_path(data, "a", filter=True) == [False, 1]
 145         assert get_path(data, "a", filter=lambda x: x) == [1]
 146 
 147     def test_filter_other(self):
 148         assert get_path({"a": 42}, "a", filter=True) == 42
 149         assert get_path({"a": True}, "a", filter=True) is True
 150         assert get_path({"a": {"b": 42}}, "a", filter=True) == {"b": 42}
 151         assert get_path({"a": 42}, "b", filter=True) is None
 152 
 153     def test_kwargs(self):
 154         with pytest.raises(TypeError):
 155             get_path({}, "foo", unknown=True)

 158 class SetPathTest(unittest.TestCase):
 159     def test_set_none(self):
 160         assert not set_path(None, "foo", value=42)
 161         assert not set_path("foo", "foo", value=42)
 162         assert not set_path(42, "foo", value=42)
 163         assert not set_path(ValueError(), "foo", value=42)
 164         assert not set_path(True, "foo", value=42)
 165 
 166     def test_set_dict(self):
 167         data = {}
 168         assert set_path(data, "a", value=42)
 169         assert data == {"a": 42}
 170 
 171         data = {"a": 2}
 172         assert set_path(data, "a", value=42)
 173         assert data == {"a": 42}
 174 
 175         data = {}
 176         assert set_path(data, "a", "b", value=42)
 177         assert data == {"a": {"b": 42}}
 178 
 179         data = CanonicalKeyDict({})
 180         assert set_path(data, "a", value=42)
 181         assert data == {"a": 42}
 182 
 183     def test_set_default(self):
 184         data = {"a": {"b": 2}}
 185         assert not setdefault_path(data, "a", "b", value=42)
 186         assert data == {"a": {"b": 2}}
 187 
 188         data = {}
 189         assert setdefault_path(data, "a", "b", value=42)
 190         assert data == {"a": {"b": 42}}
 191 
 192     def test_kwargs(self):
 193         with pytest.raises(TypeError):
 194             set_path({}, "foo")
 195 
 196         with pytest.raises(TypeError):
 197             set_path({}, "foo", value=1, unknown=True)
