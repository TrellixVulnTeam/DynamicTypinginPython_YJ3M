  64 class TokenizePathTestCase(unittest.TestCase):
  65     def test_forward_slash(self):
  66         assert list(tokenize_path("foo/bar")) == ["bar", "foo"]
  67 
  68     def test_back_slash(self):
  69         assert list(tokenize_path("foo\\bar")) == ["bar", "foo"]
  70 
  71     def test_dot_does_not_separate(self):
  72         assert list(tokenize_path("foo.bar")) == ["foo.bar"]
  73 
  74     def test_additional_slash_in_front(self):
  75         assert list(tokenize_path("/foo/bar")) == ["bar", "foo"]
  76         assert list(tokenize_path("\\foo\\bar")) == ["bar", "foo"]
  77 
  78     def test_relative_paths(self):
  79         assert list(tokenize_path("./")) == ["."]
  80         assert list(tokenize_path("./../")) == ["..", "."]
  81         assert list(tokenize_path("./foo/bar")) == ["bar", "foo", "."]
  82         assert list(tokenize_path(".\\foo\\bar")) == ["bar", "foo", "."]
  83 
  84     def test_path_with_spaces(self):
  85         assert list(tokenize_path("\\foo bar\\bar")) == ["bar", "foo bar"]
  86 
  87     def test_no_path(self):
  88         assert list(tokenize_path("/")) == []

 155 class MatchCommitsPathTestCase(CommitTestCase):
 156     def test_simple(self):
 157         file_change = self.create_commitfilechange(filename="hello/app.py", type="A")
 158         file_changes = [
 159             file_change,
 160             self.create_commitfilechange(filename="goodbye/app.js", type="A"),
 161         ]
 162         assert [(file_change.commit, 2)] == _match_commits_path(file_changes, "hello/app.py")
 163 
 164     def test_skip_one_score_match_longer_than_one_token(self):
 165         file_changes = [
 166             self.create_commitfilechange(filename="hello/app.py", type="A"),
 167             self.create_commitfilechange(filename="hello/world/app.py", type="A"),
 168             self.create_commitfilechange(filename="hello/world/template/app.py", type="A"),
 169         ]
 170         assert [] == _match_commits_path(file_changes, "app.py")
 171 
 172     def test_similar_paths(self):
 173         file_changes = [
 174             self.create_commitfilechange(filename="hello/app.py", type="A"),
 175             self.create_commitfilechange(filename="world/hello/app.py", type="A"),
 176             self.create_commitfilechange(filename="template/hello/app.py", type="A"),
 177         ]
 178 
 179         commits = sorted([(fc.commit, 2) for fc in file_changes], key=lambda fc: fc[0].id)
 180         assert commits == sorted(
 181             _match_commits_path(file_changes, "hello/app.py"), key=lambda fc: fc[0].id
 182         )
