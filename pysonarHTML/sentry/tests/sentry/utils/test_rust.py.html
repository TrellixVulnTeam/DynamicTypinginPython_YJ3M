 106 def get_exc_info(rust_info):
 107     exc = ValueError("hello world")
 108     if rust_info is not None:
 109         exc.rust_info = rust_info
 110     return type(exc), exc, None

 113 def test_merge_rust_info():
 114     event = get_event(STACKTRACE)
 115     exc_info = get_exc_info(STACKTRACE)
 116 
 117     merge_rust_info_frames(event, {"exc_info": exc_info})
 118 
 119     assert event["platform"] == "native"
 120     assert event["logentry"]["formatted"] == "invalid debug identifier"
 121 
 122     exception = event["exception"]["values"][0]
 123     assert exception["value"] == "invalid debug identifier"
 124 
 125     frames = exception["stacktrace"]["frames"]
 126     assert len(frames) == 8
 127     assert frames[0]["platform"] == "python"
 128 
 129     # Top frame
 130     assert frames[7]["instruction_addr"] == "0x11163e27c"
 131     assert frames[7]["function"] == "<failure::error::Error as core::convert::From<F>>::from"
 132     assert frames[7]["package"] == "failure"
 133     assert frames[7]["in_app"] is False
 134     assert frames[7]["filename"] == "mod.rs"
 135     assert frames[7]["lineno"] == 36

 211 def test_starts_with():
 212     # Basic functions
 213     assert starts_with("__rust_maybe_catch_panic", "__rust")
 214     assert starts_with("futures::task_impl::std::set", "futures::")
 215     assert not starts_with("futures::task_impl::std::set", "tokio::")
 216 
 217     # Generics
 218     assert starts_with("_<futures..task_impl..Spawn<T>>::enter::_{{closure}}", "futures::")
 219     assert not starts_with("_<futures..task_impl..Spawn<T>>::enter::_{{closure}}", "tokio::")
 220     assert starts_with("<futures::task_impl::Spawn<T>>::enter::{{closure}}", "futures::")
 221     assert not starts_with("<futures::task_impl::Spawn<T>>::enter::{{closure}}", "tokio::")
 222 
 223     # Trait implementations
 224     assert starts_with("<failure::error::Error as core::convert::From<F>>::from", "failure::")
 225     assert starts_with("_<failure::error::Error as core::convert::From<F>>::from", "failure::")
 226 
 227     # Blanket implementations
 228     assert starts_with("<T as core::convert::Into<U>>::into", "core::")

 231 def test_strip_symbol():
 232     assert strip_symbol("") == ""
 233     assert strip_symbol("_ffi_call_unix64") == "_ffi_call_unix64"
 234     assert (
 235         strip_symbol("backtrace::backtrace::trace::h1c213d29ba950696")
 236         == "backtrace::backtrace::trace"
 237     )
 238     assert (
 239         strip_symbol("<T as core::convert::Into<U>>::into::h58e05f056150874e")
 240         == "<T as core::convert::Into<U>>::into"
 241     )
 242     assert strip_symbol("symbolic_symcache_from_object") == "symbolic_symcache_from_object"
