  21 def test_retries():
  22     """
  23     Tests that, even if I set up 5 retries, there is only one request
  24     made since it times out.
  25     """
  26     connection_mock = mock.Mock()
  27     connection_mock.request.side_effect = ReadTimeoutError(None, "test.com", "Timeout")
  28 
  29     snuba_pool = FakeConnectionPool(
  30         connection=connection_mock,
  31         host="www.test.com",
  32         port=80,
  33         retries=RetrySkipTimeout(total=5, method_whitelist={"GET", "POST"}),
  34         timeout=30,
  35         maxsize=10,
  36     )
  37 
  38     with pytest.raises(HTTPError):
  39         snuba_pool.urlopen("POST", "/query", body="{}")
  40 
  41     assert connection_mock.request.call_count == 1
