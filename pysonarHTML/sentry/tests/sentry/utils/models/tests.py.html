   3 from django.db import models
   4 from sentry.db.models import (
   5     Model,
   6     BoundedIntegerField,

   9 )
  10 from sentry.testutils import TestCase

  13 # There's a good chance this model wont get created in the db, so avoid
  14 # assuming it exists in these tests.
  15 class DummyModel(Model):
  16     __core__ = False  # needs defined for Sentry to not yell at you
  17 
  18     foo = models.CharField(max_length=32)
  19     normint = BoundedIntegerField(null=True)
  20     bigint = BoundedBigIntegerField(null=True)
  21     posint = BoundedPositiveIntegerField(null=True)
  22 
  23 
  24 class ModelTest(TestCase):
  25     def test_foo_hasnt_changed_on_init(self):
  26         inst = DummyModel(id=1, foo="bar")
  27         self.assertFalse(inst.has_changed("foo"))
  28 
  29     def test_foo_has_changes_before_save(self):
  30         inst = DummyModel(id=1, foo="bar")
  31         inst.foo = "baz"
  32         self.assertTrue(inst.has_changed("foo"))
  33         self.assertEquals(inst.old_value("foo"), "bar")
  34 
  35     def test_foo_hasnt_changed_after_save(self):
  36         inst = DummyModel(id=1, foo="bar")
  37         inst.foo = "baz"
  38         self.assertTrue(inst.has_changed("foo"))
  39         self.assertEquals(inst.old_value("foo"), "bar")
  40         models.signals.post_save.send(instance=inst, sender=type(inst), created=False)
  41         self.assertFalse(inst.has_changed("foo"))
  42 
  43     def test_large_int(self):
  44         with self.assertRaises(AssertionError):
  45             DummyModel.objects.create(normint=int(9223372036854775807), foo="bar")
  46 
  47         with self.assertRaises(AssertionError):
  48             DummyModel.objects.create(bigint=int(9223372036854775808), foo="bar")
  49 
  50         with self.assertRaises(AssertionError):
  51             DummyModel.objects.create(posint=int(9223372036854775808), foo="bar")
