   9 class TimedRetryPolicyTestCase(TestCase):
  10     def test_policy_success(self):
  11         bomb = Exception("Boom!")
  12         callable = mock.MagicMock(side_effect=[bomb, mock.sentinel.OK])
  13 
  14         retry = TimedRetryPolicy(0.3, delay=lambda i: 0.1)
  15         retry.clock = mock.Mock()
  16         retry.clock.sleep = mock.MagicMock()
  17         retry.clock.time = mock.MagicMock(side_effect=[0, 0.15])
  18 
  19         assert retry(callable) is mock.sentinel.OK
  20         assert callable.call_count == 2
  21 
  22     def test_policy_failure(self):
  23         bomb = Exception("Boom!")
  24         callable = mock.MagicMock(side_effect=bomb)
  25 
  26         retry = TimedRetryPolicy(0.3, delay=lambda i: 0.1)
  27         retry.clock = mock.Mock()
  28         retry.clock.sleep = mock.MagicMock()
  29         retry.clock.time = mock.MagicMock(side_effect=[0, 0.15, 0.25])
  30 
  31         try:
  32             retry(callable)
  33         except RetryException as exception:
  34             assert exception.exception is bomb
  35         else:
  36             self.fail(u"Expected {!r}!".format(RetryException))
  37 
  38         assert callable.call_count == 2
  39 
  40     def test_decorator(self):
