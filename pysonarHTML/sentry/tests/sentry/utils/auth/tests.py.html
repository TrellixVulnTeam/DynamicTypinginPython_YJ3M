  61 class LoginTest(TestCase):
  62     def make_request(self, next=None):
  63         request = HttpRequest()
  64         request.META["REMOTE_ADDR"] = "127.0.0.1"
  65         request.session = self.session
  66         request.user = AnonymousUser()
  67         if next:
  68             request.session["_next"] = next

  71     def test_simple(self):
