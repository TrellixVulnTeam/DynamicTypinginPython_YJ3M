  14 class JSONTest(TestCase):
  15     def test_uuid(self):
  16         res = uuid.uuid4()
  17         self.assertEquals(json.dumps(res), '"%s"' % res.hex)
  18 
  19     def test_datetime(self):
  20         res = datetime.datetime(day=1, month=1, year=2011, hour=1, minute=1, second=1)
  21         self.assertEquals(json.dumps(res), '"2011-01-01T01:01:01.000000Z"')
  22 
  23     def test_set(self):
  24         res = set(["foo"])
  25         self.assertEquals(json.dumps(res), '["foo"]')
  26 
  27     def test_frozenset(self):
  28         res = frozenset(["foo"])
  29         self.assertEquals(json.dumps(res), '["foo"]')
  30 
  31     def test_escape(self):
  32         res = "<script>alert('&');</script>"
  33         assert json.dumps(res) == "\"<script>alert('&');</script>\""
  34         assert (
  35             json.dumps(res, escape=True)
  36             == '"\\u003cscript\\u003ealert(\\u0027\u0026\\u0027);\\u003c/script\\u003e"'
  37         )
  38         assert (
  39             json.dumps_htmlsafe(res)
  40             == '"\\u003cscript\\u003ealert(\\u0027\u0026\\u0027);\\u003c/script\\u003e"'
  41         )
  42 
  43     def test_inf(self):
  44         res = float("inf")
  45         self.assertEquals(json.dumps(res), "null")
  46 
  47     def test_enum(self):
  48         enum = Enum("foo", "a b c")
  49         res = enum.a
  50         self.assertEquals(json.dumps(res), "1")
