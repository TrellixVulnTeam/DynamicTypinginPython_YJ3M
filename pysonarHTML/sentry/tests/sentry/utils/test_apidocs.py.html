   5 def test_simplify_regex():
   6     out = apidocs.simplify_regex(r"^/organizations/(?P<org_id>\w+)/$")
   7     assert out == "/organizations/{org_id}/"
   8 
   9     out = apidocs.simplify_regex(r"^/orgs/(?P<org_id>[\w]+)/project/(?P<id>[\d]+)/$")
  10     assert out == "/orgs/{org_id}/project/{id}/"

  13 def test_parse_doc_string():
  14     text = """
  15 Glory!
  16 ~~~~~~
  17 
  18 This is a glorious function.
  19 
  20 It has a multi-line description.
  21 
  22 :param int id: An ``id``.
  23 :qparam bool louder: Make it **louder**?
  24 :pparam string orgid: The *orgid*.
  25                       On a second line.
  26 :auth: required
  27 """
  28     result = apidocs.parse_doc_string(text)
  29     assert result[0] == "Glory!"
  30     expected = ["", "This is a glorious function.", "", "It has a multi-line description.", ""]
  31     assert result[1] == expected
  32     assert result[2] is None, "no warning present"
  33 
  34     params = result[3]
  35     assert "path" in params, "Should have path params"
  36     assert params["path"][0] == dict(
  37         name="orgid", type="string", description="The *orgid*. On a second line."
  38     )
  39     assert "query" in params, "Should have query params"
  40     assert params["query"][0] == dict(name="louder", type="bool", description="Make it **louder**?")
  41     assert "auth" in params, "Should have auth"
  42     assert params["auth"][0] == dict(name="", type="", description="required")
  43 
  44     assert "param" in params, "Should have regular param"
  45     assert params["param"][0] == dict(name="id", type="int", description="An `id`.")

  48 def test_parse_doc_string_warning():
  49     text = """
  50 Danger!
  51 ~~~~~~~
  52 
  53 This is a dangerous function.
  54 
  55 .. warning::
  56     This is a warning
  57 
  58     It has multiple lines
  59 
  60 It has a multi-line description.
  61 
  62 :param int id: An id.
  63 """
  64     result = apidocs.parse_doc_string(text)
  65     assert result[0] == "Danger!"
  66     expected = ["", "This is a dangerous function.", "", "It has a multi-line description.", ""]
  67     assert result[1] == expected
  68 
  69     expected = "This is a warning\n\nIt has multiple lines"
  70     assert result[2] == expected
  71 
  72     params = result[3]
  73     assert "param" in params, "Should have regular param"
  74     assert params["param"][0] == dict(name="id", type="int", description="An id.")
