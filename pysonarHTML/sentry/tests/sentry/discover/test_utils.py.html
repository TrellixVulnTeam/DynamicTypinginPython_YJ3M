   8 class TransformAliasesAndQueryTest(SnubaTestCase, TestCase):
   9     def setUp(self):
  10         super(TransformAliasesAndQueryTest, self).setUp()
  11         self.environment = self.create_environment(self.project, name="prod")

  14         self.store_event(

  26     def test_field_aliasing_in_selected_columns(self):
  27         result = transform_aliases_and_query(
  28             selected_columns=["project.id", "user.email", "release"],
  29             filter_keys={"project_id": [self.project.id]},
  30         )
  31         data = result["data"]
  32         assert len(data) == 1
  33         assert data[0]["project.id"] == self.project.id

  37     def test_field_aliasing_in_aggregate_functions_and_groupby(self):
  38         result = transform_aliases_and_query(
  39             selected_columns=["project.id"],
  40             aggregations=[["uniq", "user.email", "uniq_email"]],
  41             filter_keys={"project_id": [self.project.id]},

  44         data = result["data"]
  45         assert len(data) == 1

  47         assert data[0]["uniq_email"] == 1

  49     def test_field_aliasing_in_conditions(self):

  53             filter_keys={"project_id": [self.project.id]},

  60     def test_autoconversion_of_time_column(self):
  61         result = transform_aliases_and_query(
  62             aggregations=[["count", None, "count"]],
  63             filter_keys={"project_id": [self.project.id]},
  64             start=before_now(minutes=10),
  65             end=before_now(minutes=-1),
  66             groupby=["time"],
  67             orderby=["time"],
  68             rollup=3600,
  69         )
  70 
  71         # If the date range spans across two hours, then one row will have results
  72         # and the other one won't.
  73         for row in result["data"]:
  74             assert isinstance(row["time"], int)
  75             if "count" in row:
  76                 assert row["count"] == 1
