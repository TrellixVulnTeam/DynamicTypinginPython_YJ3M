  11 class BulkDeleteQueryTest(TestCase):
  12     def test_project_restriction(self):
  13         project1 = self.create_project()
  14         group1_1 = self.create_group(project1)
  15         group1_2 = self.create_group(project1)
  16         project2 = self.create_project()
  17         group2_1 = self.create_group(project2)
  18         group2_2 = self.create_group(project2)
  19         BulkDeleteQuery(model=Group, project_id=project1.id).execute()
  20         assert Project.objects.filter(id=project1.id).exists()
  21         assert Project.objects.filter(id=project2.id).exists()
  22         assert Group.objects.filter(id=group2_1.id).exists()
  23         assert Group.objects.filter(id=group2_2.id).exists()
  24         assert not Group.objects.filter(id=group1_1.id).exists()
  25         assert not Group.objects.filter(id=group1_2.id).exists()

  27     def test_datetime_restriction(self):
  28         now = timezone.now()
  29         project1 = self.create_project()
  30         group1_1 = self.create_group(project1, last_seen=now - timedelta(days=1))
  31         group1_2 = self.create_group(project1, last_seen=now - timedelta(days=1))
  32         group1_3 = self.create_group(project1, last_seen=now)
  33         BulkDeleteQuery(model=Group, dtfield="last_seen", days=1).execute()
  34         assert not Group.objects.filter(id=group1_1.id).exists()
  35         assert not Group.objects.filter(id=group1_2.id).exists()
  36         assert Group.objects.filter(id=group1_3.id).exists()
