   3 from sentry.utils.compat.mock import patch

   5 from sentry.models import OrganizationOption, Repository
   6 from sentry.testutils import TestCase
   7 
   8 
   9 class RenamePendingDeleteTest(TestCase):
  10     def setUp(self):
  11         super(RenamePendingDeleteTest, self).setUp()
  12         self.repository = Repository.objects.create(
  13             organization_id=self.organization.id,
  14             name="example/name",
  15             provider="provider",
  16             external_id="external_id",
  17         )
  18 
  19         class MockUuid4:
  20             hex = "1234567"
  21 
  22         self.mock_uuid4 = MockUuid4
  23 
  24     def assert_organization_option(self, repo):
  25         option = OrganizationOption.objects.get(
  26             organization_id=repo.organization_id, key=repo.build_pending_deletion_key()
  27         )
  28         assert option.value == {
  29             "id": repo.id,
  30             "model": Repository.__name__,
  31             "name": "example/name",
  32             "external_id": "external_id",
  33         }
  34 
  35     def test_rename_on_pending_deletion(self):
  36         with patch("sentry.db.mixin.uuid4", new=self.mock_uuid4):
  37             self.repository.rename_on_pending_deletion()
  38         repo = Repository.objects.get(id=self.repository.id)
  39         assert repo.name == "1234567"
  40         assert repo.external_id == "1234567"
  41         self.assert_organization_option(repo)
  42 
  43     def test_reset_pending_deletion_field_names(self):
  44         self.repository.rename_on_pending_deletion()
  45         self.repository.reset_pending_deletion_field_names()
  46         repo = Repository.objects.get(id=self.repository.id)
  47         assert repo.name == "example/name"
  48         assert repo.external_id == "external_id"
  49         self.assert_organization_option(repo)
  50 
  51     def test_delete_pending_deletion_option(self):
  52         self.repository.rename_on_pending_deletion()
  53         self.repository.delete()
  54         assert not OrganizationOption.objects.filter(
  55             organization_id=self.organization.id, key=self.repository.build_pending_deletion_key()
  56         ).exists()
