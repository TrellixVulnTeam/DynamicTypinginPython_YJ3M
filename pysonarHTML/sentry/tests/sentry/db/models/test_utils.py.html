   1 from __future__ import absolute_import

   3 from sentry.testutils import TestCase
   4 from sentry.db.models.utils import slugify_instance
   5 from sentry.models import Organization
   6 
   7 
   8 class SlugifyInstanceTest(TestCase):
   9     def test_no_conflict(self):
  10         org = Organization(name="matt")
  11         slugify_instance(org, "matt")
  12         assert org.slug == "matt"
  13         assert not Organization.objects.filter(slug="matt").exists()
  14 
  15     def test_conflict(self):
  16         base_slug = self.organization.slug
  17         org = Organization(name="foo")
  18         slugify_instance(org, base_slug)
  19         assert org.slug.startswith(base_slug + "-"), org.slug
  20         assert not Organization.objects.filter(slug=org.slug).exists()
  21 
  22     def test_reserved(self):
  23         base_slug = self.organization.slug
  24         org = Organization(name="foo")
  25         slugify_instance(org, base_slug, reserved=(base_slug,))
  26         assert not org.slug.startswith(base_slug + "-"), org.slug
  27         assert not Organization.objects.filter(slug=org.slug).exists()
  28 
  29     def test_max_length(self):
  30         org = Organization(name="matt")
  31         slugify_instance(org, "matt", max_length=2)
  32         assert org.slug == "ma", org.slug
  33         assert not Organization.objects.filter(slug="ma").exists()
