 123     def test_query_object(self):
 124         JSONFieldTestModel.objects.create(json={})
 125         JSONFieldTestModel.objects.create(json={"foo": "bar"})
 126         self.assertEqual(2, JSONFieldTestModel.objects.all().count())
 127         self.assertEqual(1, JSONFieldTestModel.objects.exclude(json={}).count())
 128         self.assertEqual(1, JSONFieldTestModel.objects.filter(json={}).count())
 129         self.assertEqual(1, JSONFieldTestModel.objects.filter(json={"foo": "bar"}).count())
 130         self.assertEqual(
 131             1, JSONFieldTestModel.objects.filter(json__contains={"foo": "bar"}).count()
 132         )
 133         JSONFieldTestModel.objects.create(json={"foo": "bar", "baz": "bing"})
 134         self.assertEqual(
 135             2, JSONFieldTestModel.objects.filter(json__contains={"foo": "bar"}).count()
