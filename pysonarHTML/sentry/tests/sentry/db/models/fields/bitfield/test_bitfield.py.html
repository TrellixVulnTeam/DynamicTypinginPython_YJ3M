 112 class BitTest(TestCase):
 113     def test_int(self):
 114         bit = Bit(0)
 115         self.assertEquals(int(bit), 1)
 116         self.assertEquals(bool(bit), True)
 117         self.assertFalse(not bit)
 118 
 119     def test_comparison(self):
 120         self.assertEquals(Bit(0), Bit(0))
 121         self.assertNotEquals(Bit(1), Bit(0))
 122         self.assertNotEquals(Bit(0, 0), Bit(0, 1))
 123         self.assertEquals(Bit(0, 1), Bit(0, 1))
 124         self.assertEquals(Bit(0), 1)
 125 
 126     def test_and(self):
 127         self.assertEquals(1 & Bit(2), 0)
 128         self.assertEquals(1 & Bit(0), 1)
 129         self.assertEquals(1 & ~Bit(0), 0)
 130         self.assertEquals(Bit(0) & Bit(2), 0)
 131         self.assertEquals(Bit(0) & Bit(0), 1)
 132         self.assertEquals(Bit(0) & ~Bit(0), 0)
 133 
 134     def test_or(self):
 135         self.assertEquals(1 | Bit(2), 5)
 136         self.assertEquals(1 | Bit(5), 33)
 137         self.assertEquals(1 | ~Bit(2), -5)
 138         self.assertEquals(Bit(0) | Bit(2), 5)
 139         self.assertEquals(Bit(0) | Bit(5), 33)
 140         self.assertEquals(Bit(0) | ~Bit(2), -5)
 141 
 142     def test_xor(self):
 143         self.assertEquals(1 ^ Bit(2), 5)
 144         self.assertEquals(1 ^ Bit(0), 0)
 145         self.assertEquals(1 ^ Bit(1), 3)
 146         self.assertEquals(1 ^ Bit(5), 33)
 147         self.assertEquals(1 ^ ~Bit(2), -6)
 148         self.assertEquals(Bit(0) ^ Bit(2), 5)
 149         self.assertEquals(Bit(0) ^ Bit(0), 0)
 150         self.assertEquals(Bit(0) ^ Bit(1), 3)
 151         self.assertEquals(Bit(0) ^ Bit(5), 33)
 152         self.assertEquals(Bit(0) ^ ~Bit(2), -6)

 164     def test_regression_1425(self):
 165         # Creating new instances shouldn't allow negative values.
 166         instance = BitFieldTestModel.objects.create(flags=-1)
 167         self.assertEqual(instance.flags._value, 15)
 168         self.assertTrue(instance.flags.FLAG_0)
 169         self.assertTrue(instance.flags.FLAG_1)
 170         self.assertTrue(instance.flags.FLAG_2)
 171         self.assertTrue(instance.flags.FLAG_3)
 172 
 173         cursor = connection.cursor()
 174         flags_field = BitFieldTestModel._meta.get_field("flags")
 175         flags_db_column = flags_field.db_column or flags_field.name
 176         cursor.execute(
 177             "INSERT INTO %s (%s) VALUES (-1)" % (BitFieldTestModel._meta.db_table, flags_db_column)
 178         )
 179         # There should only be the one row we inserted through the cursor.
 180         instance = BitFieldTestModel.objects.get(flags=-1)
 181         self.assertTrue(instance.flags.FLAG_0)
 182         self.assertTrue(instance.flags.FLAG_1)
 183         self.assertTrue(instance.flags.FLAG_2)
 184         self.assertTrue(instance.flags.FLAG_3)
 185         instance.save()
 186 
 187         self.assertEqual(BitFieldTestModel.objects.filter(flags=15).count(), 2)
 188         self.assertEqual(BitFieldTestModel.objects.filter(flags__lt=0).count(), 0)

 295     def test_binary_capacity(self):
 296         import math
 297         from django.db.models.fields import BigIntegerField
 298 
 299         # Local maximum value, slow canonical algorithm
 300         MAX_COUNT = int(math.floor(math.log(BigIntegerField.MAX_BIGINT, 2)))
 301 
 302         # Big flags list
 303         flags = ["f" + six.text_type(i) for i in range(100)]
 304 
 305         try:
 306             BitField(flags=flags[:MAX_COUNT])
 307         except ValueError:
 308             self.fail("It should work well with these flags")
 309 
 310         self.assertRaises(ValueError, BitField, flags=flags[: (MAX_COUNT + 1)])
