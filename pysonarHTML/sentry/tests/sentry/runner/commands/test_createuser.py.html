  15     def test_superuser(self):
  16         rv = self.invoke("--email=you@somewhereawesome.com", "--password=awesome", "--superuser")
  17         assert rv.exit_code == 0, rv.output
  18         assert "you@somewhereawesome.com" in rv.output
