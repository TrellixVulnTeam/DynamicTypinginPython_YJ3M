  15     def test_simple(self):
  16         with self.runner.isolated_filesystem():
  17             rv = self.invoke("config")
  18             assert rv.exit_code == 0, rv.output
  19             contents = os.listdir("config")
  20             assert set(contents) == {"sentry.conf.py", "config.yml"}
  21 
  22             # Make sure the python file is valid
  23             ctx = {"__file__": "sentry.conf.py"}
  24             with open("config/sentry.conf.py") as fp:
  25                 six.exec_(fp.read(), ctx)
  26             assert "DEBUG" in ctx
  27 
  28             # Make sure the yaml file is valid
  29             from sentry.utils.yaml import safe_load
  30 
  31             with open("config/config.yml", "rb") as fp:
  32                 ctx = safe_load(fp)
  33             assert "system.secret-key" in ctx
