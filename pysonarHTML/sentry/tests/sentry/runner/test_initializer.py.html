  30 def test_bootstrap_options_simple(settings, config_yml):
  31     "Config options are specified in both places, but config.yml should prevail"
  32     settings.SECRET_KEY = "xxx"
  33     settings.EMAIL_BACKEND = "xxx"
  34     settings.EMAIL_HOST = "xxx"
  35     settings.EMAIL_PORT = 6969
  36     settings.EMAIL_HOST_USER = "xxx"
  37     settings.EMAIL_HOST_PASSWORD = "xxx"
  38     settings.EMAIL_USE_TLS = False
  39     settings.SERVER_EMAIL = "xxx"
  40     settings.EMAIL_SUBJECT_PREFIX = "xxx"
  41     settings.SENTRY_OPTIONS = {"something.else": True}
  42 
  43     config_yml.write(
  44         """\
  45 foo.bar: my-foo-bar
  46 system.secret-key: my-system-secret-key
  47 mail.backend: my-mail-backend
  48 mail.host: my-mail-host
  49 mail.port: 123
  50 mail.username: my-mail-username
  51 mail.password: my-mail-password
  52 mail.use-tls: true
  53 mail.from: my-mail-from
  54 mail.subject-prefix: my-mail-subject-prefix
  55 """
  56     )
  57 
  58     bootstrap_options(settings, six.text_type(config_yml))
  59     assert settings.SENTRY_OPTIONS == {
  60         "something.else": True,
  61         "foo.bar": "my-foo-bar",
  62         "system.secret-key": "my-system-secret-key",
  63         "mail.backend": "my-mail-backend",
  64         "mail.host": "my-mail-host",
  65         "mail.port": 123,
  66         "mail.username": "my-mail-username",
  67         "mail.password": "my-mail-password",
  68         "mail.use-tls": True,
  69         "mail.from": "my-mail-from",
  70         "mail.subject-prefix": "my-mail-subject-prefix",
  71     }
  72     assert settings.SECRET_KEY == "my-system-secret-key"
  73     assert settings.EMAIL_BACKEND == "my-mail-backend"
  74     assert settings.EMAIL_HOST == "my-mail-host"
  75     assert settings.EMAIL_PORT == 123
  76     assert settings.EMAIL_HOST_USER == "my-mail-username"
  77     assert settings.EMAIL_HOST_PASSWORD == "my-mail-password"
  78     assert settings.EMAIL_USE_TLS is True
  79     assert settings.SERVER_EMAIL == "my-mail-from"
  80     assert settings.EMAIL_SUBJECT_PREFIX == "my-mail-subject-prefix"

 163 def test_apply_legacy_settings(settings):
 164     settings.ALLOWED_HOSTS = []
 165     settings.SENTRY_USE_QUEUE = True
 166     settings.SENTRY_ALLOW_REGISTRATION = True
 167     settings.SENTRY_ADMIN_EMAIL = "admin-email"
 168     settings.SENTRY_URL_PREFIX = "http://url-prefix"
 169     settings.SENTRY_SYSTEM_MAX_EVENTS_PER_MINUTE = 10
 170     settings.SENTRY_REDIS_OPTIONS = {"foo": "bar"}
 171     settings.SENTRY_ENABLE_EMAIL_REPLIES = True
 172     settings.SENTRY_SMTP_HOSTNAME = "reply-hostname"
 173     settings.MAILGUN_API_KEY = "mailgun-api-key"
 174     settings.SENTRY_OPTIONS = {"system.secret-key": "secret-key", "mail.from": "mail-from"}
 175     settings.SENTRY_FILESTORE = "some-filestore"
 176     settings.SENTRY_FILESTORE_OPTIONS = {"filestore-foo": "filestore-bar"}
 177     apply_legacy_settings(settings)
 178     assert settings.CELERY_ALWAYS_EAGER is False
 179     assert settings.SENTRY_FEATURES["auth:register"] is True
 180     assert settings.SENTRY_OPTIONS == {
 181         "system.admin-email": "admin-email",
 182         "system.url-prefix": "http://url-prefix",
 183         "system.rate-limit": 10,
 184         "system.secret-key": "secret-key",
 185         "redis.clusters": {"default": {"foo": "bar"}},
 186         "mail.from": "mail-from",
 187         "mail.enable-replies": True,
 188         "mail.reply-hostname": "reply-hostname",
 189         "mail.mailgun-api-key": "mailgun-api-key",
 190         "filestore.backend": "some-filestore",
 191         "filestore.options": {"filestore-foo": "filestore-bar"},
 192     }
 193     assert settings.DEFAULT_FROM_EMAIL == "mail-from"
 194     assert settings.ALLOWED_HOSTS == ["*"]
