  19 def sort_records(records):
  20     """
  21     Sorts records for fetch_state method
  22     fetch_state is expecting these records to be ordered from newest to oldest
  23     """
  24     return sorted(records, key=lambda r: r.value.event.datetime, reverse=True)

 224     def create_events_from_filenames(self, project, filenames=None):
 225         events = []
 226         for index, label in enumerate(filenames):
 227             event_data = {
 228                 "stacktrace": {"frames": [{"filename": label}]},
 229                 "fingerprint": [label],
 230                 "timestamp": iso_format(before_now(minutes=1)),
 231             }
 232             event = self.store_event(data=event_data, project_id=project.id, assert_no_errors=False)
 233             events.append(event)
 234         return events

 236     def assert_get_personalized_digests(
 237         self, project, digest, user_ids, expected_result, target_type=ActionTargetType.ISSUE_OWNERS
 238     ):
 239         result_user_ids = []
 240         for user_id, user_digest in get_personalized_digests(
 241             target_type, project.id, digest, user_ids
 242         ):
 243             assert user_id in expected_result
 244             assert set([e.event_id for e in get_event_from_groups_in_digest(user_digest)]) == set(
 245                 [e.event_id for e in expected_result[user_id]]
 246             )
 247             result_user_ids.append(user_id)
 248 
 249         assert sorted(expected_result.keys()) == sorted(result_user_ids)

 289     def test_team_without_members(self):
 290         team = self.create_team()
 291         project = self.create_project(teams=[team], fire_project_created=True)
 292         ProjectOwnership.objects.create(
 293             project_id=project.id,
 294             schema=dump_schema([Rule(Matcher("path", "*.cpp"), [Owner("team", team.slug)])]),
 295             fallthrough=True,
 296         )
 297         rule = project.rule_set.all()[0]
 298         records = [
 299             event_to_record(event, (rule,))
 300             for event in self.create_events_from_filenames(
 301                 project, ["hello.py", "goodbye.py", "hola.py", "adios.py"]
 302             )
 303         ]
 304         digest = build_digest(project, sort_records(records))
 305         user_ids = [member.user_id for member in team.member_set]
 306         assert not user_ids
 307         for user_id, user_digest in get_personalized_digests(
 308             ActionTargetType.ISSUE_OWNERS, project.id, digest, user_ids
 309         ):
 310             assert False  # no users in this team no digests should be processed
