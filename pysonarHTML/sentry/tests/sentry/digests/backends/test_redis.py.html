  14 class RedisBackendTestCase(TestCase):
  15     def test_basic(self):
  16         backend = RedisBackend()
  17 
  18         # The first item should return "true", indicating that this timeline
  19         # can be immediately dispatched to be digested.
  20         record_1 = Record("record:1", "value", time.time())
  21         assert backend.add("timeline", record_1) is True
  22 
  23         # The second item should return "false", since it's ready to be
  24         # digested but dispatching again would cause it to be sent twice.
  25         record_2 = Record("record:2", "value", time.time())
  26         assert backend.add("timeline", record_2) is False
  27 
  28         # There's nothing to move between sets, so scheduling should return nothing.
  29         assert set(backend.schedule(time.time())) == set()
  30 
  31         with backend.digest("timeline", 0) as records:
  32             assert set(records) == set([record_1, record_2])
  33 
  34         # The schedule should now contain the timeline.
  35         assert set(entry.key for entry in backend.schedule(time.time())) == set(["timeline"])
  36 
  37         # We didn't add any new records so there's nothing to do here.
  38         with backend.digest("timeline", 0) as records:
  39             assert set(records) == set([])
  40 
  41         # There's nothing to move between sets since the timeline contents no
  42         # longer exist at this point.
  43         assert set(backend.schedule(time.time())) == set()

  45     def test_truncation(self):
  46         backend = RedisBackend(capacity=2, truncation_chance=1.0)
  47 
  48         records = [Record(u"record:{}".format(i), "value", time.time()) for i in xrange(4)]
  49         for record in records:
  50             backend.add("timeline", record)
  51 
  52         with backend.digest("timeline", 0) as records:
  53             assert set(records) == set(records[-2:])

  55     def test_maintenance_failure_recovery(self):
  56         backend = RedisBackend()
  57 
  58         record_1 = Record("record:1", "value", time.time())
  59         backend.add("timeline", record_1)
  60 
  61         try:
  62             with backend.digest("timeline", 0) as records:
  63                 raise Exception("This causes the digest to not be closed.")
  64         except Exception:
  65             pass
  66 
  67         # Maintenance should move the timeline back to the waiting state, ...
  68         backend.maintenance(time.time())
  69 
  70         # ...and you can't send a digest in the waiting state.
  71         with pytest.raises(InvalidState):
  72             with backend.digest("timeline", 0) as records:
  73                 pass
  74 
  75         record_2 = Record("record:2", "value", time.time())
  76         backend.add("timeline", record_2)
  77 
  78         # The schedule should now contain the timeline.
  79         assert set(entry.key for entry in backend.schedule(time.time())) == set(["timeline"])
  80 
  81         # The existing and new record should be there because the timeline
  82         # contents were merged back into the digest.
  83         with backend.digest("timeline", 0) as records:
  84             assert set(records) == set([record_1, record_2])

  86     def test_maintenance_failure_recovery_with_capacity(self):
  87         backend = RedisBackend(capacity=10, truncation_chance=0.0)
  88 
  89         t = time.time()
  90 
  91         # Add 10 items to the timeline.
  92         for i in xrange(10):
  93             backend.add("timeline", Record(u"record:{}".format(i), u"{}".format(i), t + i))
  94 
  95         try:
  96             with backend.digest("timeline", 0) as records:
  97                 raise Exception("This causes the digest to not be closed.")
  98         except Exception:
  99             pass
 100 
 101         # The 10 existing items should now be in the digest set (the exception
 102         # prevented the close operation from occurring, so they were never
 103         # deleted from Redis or removed from the digest set.) If we add 10 more
 104         # items, they should be added to the timeline set (not the digest set.)
 105         for i in xrange(10, 20):
 106             backend.add("timeline", Record(u"record:{}".format(i), u"{}".format(i), t + i))
 107 
 108         # Maintenance should move the timeline back to the waiting state, ...
 109         backend.maintenance(time.time())
 110 
 111         # The schedule should now contain the timeline.
 112         assert set(entry.key for entry in backend.schedule(time.time())) == set(["timeline"])
 113 
 114         # Only the new records should exist -- the older one should have been
 115         # trimmed to avoid the digest growing beyond the timeline capacity.
 116         with backend.digest("timeline", 0) as records:
 117             expected_keys = set(u"record:{}".format(i) for i in xrange(10, 20))
 118             assert set(record.key for record in records) == expected_keys

 120     def test_delete(self):
 121         backend = RedisBackend()
 122         backend.add("timeline", Record("record:1", "value", time.time()))
 123         backend.delete("timeline")
 124 
 125         with pytest.raises(InvalidState):
 126             with backend.digest("timeline", 0) as records:
 127                 assert set(records) == set([])
 128 
 129         assert set(backend.schedule(time.time())) == set()
 130         assert len(backend._get_connection("timeline").keys("d:*")) == 0
