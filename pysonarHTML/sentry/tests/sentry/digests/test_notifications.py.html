  23 class RewriteRecordTestCase(TestCase):
  24     @fixture
  25     def rule(self):
  26         return self.event.project.rule_set.all()[0]
  27 
  28     @fixture
  29     def record(self):
  30         return event_to_record(self.event, (self.rule,))
  31 
  32     def test_success(self):
  33         assert rewrite_record(
  34             self.record,
  35             project=self.event.project,
  36             groups={self.event.group.id: self.event.group},
  37             rules={self.rule.id: self.rule},
  38         ) == Record(
  39             self.record.key,
  40             Notification(self.record.value.event, [self.rule]),
  41             self.record.timestamp,
  42         )
  43 
  44     def test_without_group(self):
  45         # If the record can't be associated with a group, it should be returned as None.
  46         assert (
  47             rewrite_record(
  48                 self.record, project=self.event.project, groups={}, rules={self.rule.id: self.rule}
  49             )
  50             is None
  51         )
  52 
  53     def test_filters_invalid_rules(self):
  54         assert rewrite_record(
  55             self.record,
  56             project=self.event.project,
  57             groups={self.event.group.id: self.event.group},
  58             rules={},
  59         ) == Record(
  60             self.record.key, Notification(self.record.value.event, []), self.record.timestamp
  61         )

  64 class GroupRecordsTestCase(TestCase):
  65     @fixture
  66     def rule(self):
  67         return self.project.rule_set.all()[0]
  68 
  69     def test_success(self):
  70         events = [
  71             self.store_event(data={"fingerprint": ["group-1"]}, project_id=self.project.id)
  72             for i in range(3)
  73         ]
  74         group = events[0].group
  75         records = [
  76             Record(event.event_id, Notification(event, [self.rule]), event.datetime)
  77             for event in events
  78         ]
  79         assert reduce(group_records, records, defaultdict(lambda: defaultdict(list))) == {
  80             self.rule: {group: records}

 120 class SplitKeyTestCase(TestCase):
 121     def test_old_style_key(self):
 122         assert split_key("mail:p:{}".format(self.project.id)) == (
 123             self.project,
 124             ActionTargetType.ISSUE_OWNERS,
 125             None,
 126         )

 128     def test_new_style_key_no_identifier(self):

 130             "mail:p:{}:{}:".format(self.project.id, ActionTargetType.ISSUE_OWNERS.value)

 132 
 133     def test_new_style_key_identifier(self):
