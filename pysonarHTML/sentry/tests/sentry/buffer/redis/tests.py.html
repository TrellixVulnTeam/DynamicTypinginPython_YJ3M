  99         model = mock.Mock()

 109         assert result == {"i+times_seen": "1", "m": "mock.mock.Mock"}
 110 
 111         pending = client.zrange("b:p", 0, -1)
 112         assert pending == ["foo"]
 113         self.buf.incr(model, columns, filters, extra={"foo": "baz", "datetime": now})
 114         result = client.hgetall("foo")
 115         f = result.pop("f")
 116         assert pickle.loads(f) == {"pk": 1, "datetime": now}
 117         assert pickle.loads(result.pop("e+datetime")) == now

 127     def test_process_pending_partitions_none(self, process_pending, process_incr):
 128         self.buf.pending_partitions = 2
 129         with self.buf.cluster.map() as client:
 130             client.zadd("b:p:0", 1, "foo")
 131             client.zadd("b:p:1", 1, "bar")
 132             client.zadd("b:p", 1, "baz")
 133 
 134         # On first pass, we are expecting to do:
 135         # * process the buffer that doesn't have a partition (b:p)
 136         # * queue up 2 jobs, one for each partition to process.
 137         self.buf.process_pending()
 138         assert len(process_incr.apply_async.mock_calls) == 1
 139         process_incr.apply_async.assert_any_call(kwargs={"batch_keys": ["baz"]})
 140         assert len(process_pending.apply_async.mock_calls) == 2
 141         process_pending.apply_async.mock_calls == [
 142             mock.call(kwargs={"partition": 0}),
 143             mock.call(kwargs={"partition": 1}),
 144         ]
 145 
 146         # Confirm that we've only processed the unpartitioned buffer
 147         client = self.buf.cluster.get_routing_client()
 148         assert client.zrange("b:p", 0, -1) == []
 149         assert client.zrange("b:p:0", 0, -1) != []
 150         assert client.zrange("b:p:1", 0, -1) != []
 151 
 152         # partition 0
 153         self.buf.process_pending(partition=0)
 154         assert len(process_incr.apply_async.mock_calls) == 2
 155         process_incr.apply_async.assert_any_call(kwargs={"batch_keys": ["foo"]})
 156         assert client.zrange("b:p:0", 0, -1) == []
 157 
 158         # Make sure we didn't queue up more
 159         assert len(process_pending.apply_async.mock_calls) == 2
 160 
 161         # partition 1
 162         self.buf.process_pending(partition=1)
 163         assert len(process_incr.apply_async.mock_calls) == 3
 164         process_incr.apply_async.assert_any_call(kwargs={"batch_keys": ["bar"]})
 165         assert client.zrange("b:p:1", 0, -1) == []
