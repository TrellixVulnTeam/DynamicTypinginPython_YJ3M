  27     def test_process_saves_data(self):
  28         group = Group.objects.create(project=Project(id=1))
  29         columns = {"times_seen": 1}
  30         filters = {"id": group.id, "project_id": 1}
  31         self.buf.process(Group, columns, filters)
  32         assert Group.objects.get(id=group.id).times_seen == group.times_seen + 1
  33 
  34     def test_process_saves_data_without_existing_row(self):
  35         columns = {"times_seen": 1}
  36         filters = {"message": "foo bar", "project_id": 1}
  37         self.buf.process(Group, columns, filters)
  38         group = Group.objects.get(message="foo bar")
  39         # the default value for times_seen is 1, so we actually end up
  40         # incrementing it to 2 here
  41         assert group.times_seen == 2
  42         assert group.project_id == 1

  44     def test_process_saves_extra(self):
  45         group = Group.objects.create(project=Project(id=1))
  46         columns = {"times_seen": 1}
  47         filters = {"id": group.id, "project_id": 1}
  48         the_date = timezone.now() + timedelta(days=5)
  49         self.buf.process(Group, columns, filters, {"last_seen": the_date})
  50         group_ = Group.objects.get(id=group.id)
  51         assert group_.times_seen == group.times_seen + 1
  52         assert group_.last_seen == the_date

  54     def test_increments_when_null(self):
  55         org = Organization.objects.create(slug="test-org")
  56         team = Team.objects.create(organization=org, slug="test-team")
  57         project = Project.objects.create(organization=org, slug="test-project")
  58         project.add_team(team)
  59         release = Release.objects.create(organization=org, version="abcdefg")
  60         release_project = ReleaseProject.objects.create(project=project, release=release)
  61         assert release_project.new_groups == 0
  62 
  63         columns = {"new_groups": 1}
  64         filters = {"id": release_project.id}
  65         self.buf.process(ReleaseProject, columns, filters)
  66         release_project_ = ReleaseProject.objects.get(id=release_project.id)
  67         assert release_project_.new_groups == 1
