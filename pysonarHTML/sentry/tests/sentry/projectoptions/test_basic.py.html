  12 @contextmanager
  13 def latest_epoch(value):
  14     old = defaults.LATEST_EPOCH
  15     try:
  16         defaults.LATEST_EPOCH = value
  17         yield
  18     finally:
  19         defaults.LATEST_EPOCH = old
