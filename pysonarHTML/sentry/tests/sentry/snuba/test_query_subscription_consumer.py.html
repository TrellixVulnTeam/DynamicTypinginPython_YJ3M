  62 class HandleMessageTest(BaseQuerySubscriptionTest, TestCase):
  63     metrics = patcher("sentry.snuba.query_subscription_consumer.metrics")
  64 
  65     def test_no_subscription(self):
  66         with mock.patch("sentry.snuba.tasks._snuba_pool") as pool:
  67             pool.urlopen.return_value.status = 202
  68             self.consumer.handle_message(
  69                 self.build_mock_message(
  70                     self.valid_wrapper, topic=settings.KAFKA_EVENTS_SUBSCRIPTIONS_RESULTS
  71                 )
  72             )
  73             pool.urlopen.assert_called_once_with(
  74                 "DELETE",
  75                 "/{}/subscriptions/{}".format(
  76                     QueryDatasets.EVENTS.value, self.valid_payload["subscription_id"]
  77                 ),
  78             )
  79         self.metrics.incr.assert_called_once_with(
  80             "snuba_query_subscriber.subscription_doesnt_exist"
  81         )
  82 
  83     def test_subscription_not_registered(self):
  84         sub = QuerySubscription.objects.create(
  85             project=self.project, type="unregistered", subscription_id="an_id"
  86         )
  87         data = self.valid_wrapper
  88         data["payload"]["subscription_id"] = sub.subscription_id
  89         self.consumer.handle_message(self.build_mock_message(data))
  90         self.metrics.incr.assert_called_once_with(
  91             "snuba_query_subscriber.subscription_type_not_registered"
  92         )
  93 
  94     def test_subscription_registered(self):
  95         registration_key = "registered_test"
  96         mock_callback = Mock()
  97         register_subscriber(registration_key)(mock_callback)
  98         with self.tasks():
  99             snuba_query = create_snuba_query(
 100                 QueryDatasets.EVENTS,
 101                 "hello",
 102                 "count()",
 103                 timedelta(minutes=10),
 104                 timedelta(minutes=1),
 105                 None,
 106             )
 107             sub = create_snuba_subscription(self.project, registration_key, snuba_query)
 108         sub.refresh_from_db()
 109 
 110         data = self.valid_wrapper
 111         data["payload"]["subscription_id"] = sub.subscription_id
 112         self.consumer.handle_message(self.build_mock_message(data))
 113         data = deepcopy(data)
 114         data["payload"]["values"] = data["payload"]["result"]
 115         data["payload"]["timestamp"] = parse_date(data["payload"]["timestamp"]).replace(
 116             tzinfo=pytz.utc
 117         )
 118         mock_callback.assert_called_once_with(data["payload"], sub)

 121 class ParseMessageValueTest(BaseQuerySubscriptionTest, unittest.TestCase):
 122     def run_test(self, message):
 123         self.consumer.parse_message_value(json.dumps(message))
 124 
 125     def run_invalid_schema_test(self, message):
 126         with self.assertRaises(InvalidSchemaError):
 127             self.run_test(message)
 128 
 129     def run_invalid_payload_test(self, remove_fields=None, update_fields=None):
 130         payload = deepcopy(self.valid_payload)
 131         if remove_fields:
 132             for field in remove_fields:
 133                 payload.pop(field)
 134         if update_fields:
 135             payload.update(update_fields)
 136         self.run_invalid_schema_test({"version": 2, "payload": payload})
 137 
 138     def test_invalid_payload(self):
 139         self.run_invalid_payload_test(remove_fields=["subscription_id"])
 140         self.run_invalid_payload_test(remove_fields=["result"])
 141         self.run_invalid_payload_test(remove_fields=["timestamp"])
 142         self.run_invalid_payload_test(update_fields={"subscription_id": ""})
 143         self.run_invalid_payload_test(update_fields={"result": {}})
 144         self.run_invalid_payload_test(update_fields={"result": {"hello": "hi"}})
 145         self.run_invalid_payload_test(update_fields={"timestamp": -1})
 146 
 147     def test_invalid_version(self):
 148         with self.assertRaises(InvalidMessageError) as cm:
 149             self.run_test({"version": 50, "payload": {}})
 150         assert six.text_type(cm.exception) == "Version specified in wrapper has no schema"
 151 
 152     def test_valid(self):
 153         self.run_test({"version": 2, "payload": self.valid_payload})
 154 
 155     def test_old_version(self):
 156         self.run_test({"version": 1, "payload": self.old_payload})
 157 
 158     def test_invalid_wrapper(self):
 159         self.run_invalid_schema_test({})
 160         self.run_invalid_schema_test({"version": 1})
 161         self.run_invalid_schema_test({"payload": self.valid_payload})

 164 class RegisterSubscriberTest(unittest.TestCase):
 165     def setUp(self):
 166         self.orig_registry = deepcopy(subscriber_registry)
 167 
 168     def tearDown(self):
 169         subscriber_registry.clear()
 170         subscriber_registry.update(self.orig_registry)
 171 
 172     def test_register(self):
 173         callback = object()
 174         other_callback = object()
 175         register_subscriber("hello")(callback)
 176         assert subscriber_registry["hello"] == callback
 177         register_subscriber("goodbye")(other_callback)
 178         assert subscriber_registry["goodbye"] == other_callback
 179 
 180     def test_already_registered(self):
 181         callback = object()
 182         other_callback = object()
 183         register_subscriber("hello")(callback)
 184         assert subscriber_registry["hello"] == callback
 185         with self.assertRaises(Exception) as cm:
 186             register_subscriber("hello")(other_callback)
 187         assert six.text_type(cm.exception) == "Handler already registered for hello"
