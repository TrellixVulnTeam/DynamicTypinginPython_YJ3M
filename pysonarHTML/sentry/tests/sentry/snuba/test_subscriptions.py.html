 171 class UpdateSnubaSubscriptionTest(TestCase):
 172     def test(self):
 173         old_dataset = QueryDatasets.EVENTS
 174         with self.tasks():
 175             snuba_query = create_snuba_query(
 176                 old_dataset,
 177                 "level:error",
 178                 "count()",
 179                 timedelta(minutes=10),
 180                 timedelta(minutes=1),
 181                 None,
 182             )
 183             subscription = create_snuba_subscription(self.project, "something", snuba_query)
 184 
 185         dataset = QueryDatasets.TRANSACTIONS
 186         query = "level:warning"
 187         aggregate = "count_unique(tags[sentry:user])"
 188         time_window = timedelta(minutes=20)
 189         resolution = timedelta(minutes=2)
 190         subscription = QuerySubscription.objects.get(id=subscription.id)
 191         subscription_id = subscription.subscription_id
 192         snuba_query.update(
 193             dataset=dataset.value,
 194             query=query,
 195             time_window=int(time_window.total_seconds()),
 196             resolution=int(resolution.total_seconds()),
 197             environment=self.environment,
 198             aggregate=aggregate,
 199         )
 200         assert subscription_id is not None
 201         update_snuba_subscription(subscription, old_dataset)
 202         assert subscription.status == QuerySubscription.Status.UPDATING.value
 203         assert subscription.subscription_id == subscription_id
 204         assert subscription.snuba_query.dataset == dataset.value
 205         assert subscription.snuba_query.query == query
 206         assert subscription.snuba_query.aggregate == aggregate
 207         assert subscription.snuba_query.time_window == int(time_window.total_seconds())
 208         assert subscription.snuba_query.resolution == int(resolution.total_seconds())

 246 class BulkDeleteSnubaSubscriptionTest(TestCase):
 247     def test(self):
 248         with self.tasks():
 249             snuba_query = create_snuba_query(
 250                 QueryDatasets.EVENTS,
 251                 "level:error",
 252                 "count()",
 253                 timedelta(minutes=10),
 254                 timedelta(minutes=1),
 255                 None,
 256             )
 257             subscription = create_snuba_subscription(self.project, "something", snuba_query)
 258             snuba_query = create_snuba_query(
 259                 QueryDatasets.EVENTS,
 260                 "level:error",
 261                 "count()",
 262                 timedelta(minutes=10),
 263                 timedelta(minutes=1),
 264                 None,
 265             )
 266             other_subscription = create_snuba_subscription(
 267                 self.create_project(organization=self.organization), "something", snuba_query
 268             )
 269         subscription_ids = [subscription.id, other_subscription.id]
 270         bulk_delete_snuba_subscriptions([subscription, other_subscription])
 271         assert (
 272             QuerySubscription.objects.filter(
 273                 id__in=subscription_ids,
 274                 status=QuerySubscription.Status.DELETING.value,
 275                 subscription_id__isnull=False,
 276             ).count()
 277             == 2
 278         )
