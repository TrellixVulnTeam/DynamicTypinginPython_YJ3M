  22 @add_metaclass(abc.ABCMeta)
  23 class BaseSnubaTaskTest(object):
  24     metrics = patcher("sentry.snuba.tasks.metrics")
  25 
  26     status_translations = {
  27         QuerySubscription.Status.CREATING: "create",
  28         QuerySubscription.Status.UPDATING: "update",
  29         QuerySubscription.Status.DELETING: "delete",
  30     }

  32     @abc.abstractproperty
  33     def expected_status(self):
  34         pass

  37     def task(self):
  38         pass

  40     def create_subscription(self, status=None, subscription_id=None):
  41         if status is None:
  42             status = self.expected_status
  43         dataset = QueryDatasets.EVENTS.value
  44         aggregate = "count_unique(tags[sentry:user])"
  45         query = "hello"
  46         time_window = 60
  47         resolution = 60
  48 
  49         snuba_query = SnubaQuery.objects.create(
  50             dataset=dataset,
  51             aggregate=aggregate,
  52             query=query,
  53             time_window=time_window,
  54             resolution=resolution,
  55         )
  56         return QuerySubscription.objects.create(
  57             snuba_query=snuba_query,
  58             status=status.value,
  59             subscription_id=subscription_id,
  60             project=self.project,
  61             type="something",
  62         )
  63 
  64     def test_no_subscription(self):
  65         self.task(12345)
  66         self.metrics.incr.assert_called_once_with(
  67             "snuba.subscriptions.{}.subscription_does_not_exist".format(
  68                 self.status_translations[self.expected_status]
  69             )
  70         )
  71 
  72     def test_invalid_status(self):
  73         sub = self.create_subscription(QuerySubscription.Status.ACTIVE)
  74         self.task(sub.id)
  75         self.metrics.incr.assert_called_once_with(
  76             "snuba.subscriptions.{}.incorrect_status".format(
  77                 self.status_translations[self.expected_status]
  78             )
  79         )

  83     expected_status = QuerySubscription.Status.CREATING

  86     def test_already_created(self):
  87         sub = self.create_subscription(
  88             QuerySubscription.Status.CREATING, subscription_id=uuid4().hex
  89         )
  90         create_subscription_in_snuba(sub.id)

  92             "snuba.subscriptions.create.already_created_in_snuba"
  93         )
  94 
  95     def test(self):
  96         sub = self.create_subscription(QuerySubscription.Status.CREATING)
  97         create_subscription_in_snuba(sub.id)
  98         sub = QuerySubscription.objects.get(id=sub.id)
  99         assert sub.status == QuerySubscription.Status.ACTIVE.value
 100         assert sub.subscription_id is not None

 102     @responses.activate
 103     def test_adds_type(self):
 104         sub = self.create_subscription(QuerySubscription.Status.CREATING)
 105         with patch("sentry.snuba.tasks._snuba_pool") as pool:
 106             resp = Mock()
 107             resp.status = 202
 108             resp.data = json.dumps({"subscription_id": "123"})
 109             pool.urlopen.return_value = resp
 110 
 111             create_subscription_in_snuba(sub.id)
 112             request_body = json.loads(pool.urlopen.call_args[1]["body"])
 113             assert ["type", "=", "error"] in request_body["conditions"]

 116 class UpdateSubscriptionInSnubaTest(BaseSnubaTaskTest, TestCase):
 117     expected_status = QuerySubscription.Status.UPDATING
 118     task = update_subscription_in_snuba
 119 
 120     def test(self):
 121         subscription_id = "1/{}".format(uuid4().hex)
 122         sub = self.create_subscription(
 123             QuerySubscription.Status.UPDATING, subscription_id=subscription_id
 124         )
 125         update_subscription_in_snuba(sub.id)
 126         sub = QuerySubscription.objects.get(id=sub.id)
 127         assert sub.status == QuerySubscription.Status.ACTIVE.value
 128         assert sub.subscription_id is not None
 129         assert sub.subscription_id != subscription_id
 130 
 131     def test_no_subscription_id(self):
 132         sub = self.create_subscription(QuerySubscription.Status.UPDATING)
 133         assert sub.subscription_id is None
 134         update_subscription_in_snuba(sub.id)
 135         sub = QuerySubscription.objects.get(id=sub.id)
 136         assert sub.status == QuerySubscription.Status.ACTIVE.value
 137         assert sub.subscription_id is not None

 159 class BuildSnubaFilterTest(TestCase):
 160     def test_simple_events(self):
 161         snuba_filter = build_snuba_filter(QueryDatasets.EVENTS, "", "count_unique(user)", None,)
 162         assert snuba_filter
 163         assert snuba_filter.conditions == [["type", "=", "error"]]
 164         assert snuba_filter.aggregations == [["uniq", "tags[sentry:user]", u"count_unique_user"]]
 165 
 166     def test_simple_transactions(self):
 167         snuba_filter = build_snuba_filter(
 168             QueryDatasets.TRANSACTIONS, "", "count_unique(user)", None,
 169         )
 170         assert snuba_filter
 171         assert snuba_filter.conditions == []
 172         assert snuba_filter.aggregations == [["uniq", "user", u"count_unique_user"]]
 173 
 174     def test_aliased_query_events(self):
 175         snuba_filter = build_snuba_filter(
 176             QueryDatasets.EVENTS, "release:latest", "count_unique(user)", None,
 177         )
 178         assert snuba_filter
 179         assert snuba_filter.conditions == [
 180             ["tags[sentry:release]", "=", "latest"],
 181             ["type", "=", "error"],
 182         ]
 183         assert snuba_filter.aggregations == [["uniq", "tags[sentry:user]", u"count_unique_user"]]
 184 
 185     def test_aliased_query_transactions(self):
 186         snuba_filter = build_snuba_filter(
 187             QueryDatasets.TRANSACTIONS,
 188             "release:latest",
 189             "percentile(transaction.duration,.95)",
 190             None,
 191         )
 192         assert snuba_filter
 193         assert snuba_filter.conditions == [["release", "=", "latest"]]
 194         assert snuba_filter.aggregations == [
 195             [u"quantile(0.95)", "duration", u"percentile_transaction_duration__95"]
 196         ]
 197 
 198     def test_user_query(self):
 199         snuba_filter = build_snuba_filter(
 200             QueryDatasets.EVENTS, "user:anengineer@work.io", "count()", None,
 201         )
 202         assert snuba_filter
 203         assert snuba_filter.conditions == [
 204             [
 205                 ["email", "=", "anengineer@work.io"],
 206                 ["username", "=", "anengineer@work.io"],
 207                 ["ip_address", "=", "anengineer@work.io"],
 208                 ["user_id", "=", "anengineer@work.io"],
 209             ],
 210             ["type", "=", "error"],
 211         ]
 212         assert snuba_filter.aggregations == [[u"count", None, u"count"]]
 213

 216             QueryDatasets.TRANSACTIONS, "user:anengineer@work.io", "p95()", None,

 219         assert snuba_filter.conditions == [
 220             [
 221                 ["user_email", "=", "anengineer@work.io"],

 223                 ["ip_address", "=", "anengineer@work.io"],
 224                 ["user_id", "=", "anengineer@work.io"],
