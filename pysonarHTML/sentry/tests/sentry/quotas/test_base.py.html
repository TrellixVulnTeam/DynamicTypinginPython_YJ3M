  13 class QuotaTest(TestCase):
  14     def setUp(self):
  15         self.backend = Quota()

  17     def test_get_project_quota(self):
  18         org = self.create_organization()
  19         project = self.create_project(organization=org)
  20 
  21         with self.settings(SENTRY_DEFAULT_MAX_EVENTS_PER_MINUTE=0):
  22             with self.options({"system.rate-limit": 0}):
  23                 assert self.backend.get_project_quota(project) == (None, 60)
  24 
  25             OrganizationOption.objects.set_value(org, "sentry:project-rate-limit", 80)
  26 
  27             with self.options({"system.rate-limit": 100}):
  28                 assert self.backend.get_project_quota(project) == (80, 60)
  29 
  30             with self.options({"system.rate-limit": 0}):
  31                 assert self.backend.get_project_quota(project) == (None, 60)

  33     def test_get_key_quota(self):
  34         key = ProjectKey.objects.create(
  35             project=self.project, rate_limit_window=5, rate_limit_count=60
  36         )
  37         assert self.backend.get_key_quota(key) == (60, 5)
  38 
  39     def test_get_key_quota_empty(self):
  40         key = ProjectKey.objects.create(
  41             project=self.project, rate_limit_window=None, rate_limit_count=None
  42         )
  43         assert self.backend.get_key_quota(key) == (None, 0)
  44 
  45     def test_get_key_quota_multiple_keys(self):
  46         # This checks for a regression where we'd cache key quotas per project
  47         # rather than per key.
  48         key = ProjectKey.objects.create(
  49             project=self.project, rate_limit_window=None, rate_limit_count=None
  50         )
  51         rate_limited_key = ProjectKey.objects.create(
  52             project=self.project, rate_limit_window=200, rate_limit_count=86400
  53         )
  54         assert self.backend.get_key_quota(key) == (None, 0)
  55         assert self.backend.get_key_quota(rate_limited_key) == (86400, 200)
