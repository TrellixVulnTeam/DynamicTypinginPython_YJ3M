  20 def test_is_rate_limited_script():
  21     now = int(time.time())
  22 
  23     cluster = clusters.get("default")
  24     client = cluster.get_local_client(six.next(iter(cluster.hosts)))
  25 
  26     # The item should not be rate limited by either key.
  27     assert list(
  28         map(
  29             bool,
  30             is_rate_limited(client, ("foo", "r:foo", "bar", "r:bar"), (1, now + 60, 2, now + 120)),
  31         )
  32     ) == [False, False]
  33 
  34     # The item should be rate limited by the first key (1).
  35     assert list(
  36         map(
  37             bool,
  38             is_rate_limited(client, ("foo", "r:foo", "bar", "r:bar"), (1, now + 60, 2, now + 120)),
  39         )
  40     ) == [True, False]
  41 
  42     # The item should still be rate limited by the first key (1), but *not*
  43     # rate limited by the second key (2) even though this is the third time
  44     # we've checked the quotas. This ensures items that are rejected by a lower
  45     # quota don't affect unrelated items that share a parent quota.
  46     assert list(
  47         map(
  48             bool,
  49             is_rate_limited(client, ("foo", "r:foo", "bar", "r:bar"), (1, now + 60, 2, now + 120)),
  50         )
  51     ) == [True, False]
  52 
  53     assert client.get("foo") == "1"
  54     assert 59 <= client.ttl("foo") <= 60
  55 
  56     assert client.get("bar") == "1"
  57     assert 119 <= client.ttl("bar") <= 120
  58 
  59     # make sure "refund/negative" keys haven't been incremented
  60     assert client.get("r:foo") is None
  61     assert client.get("r:bar") is None
  62 
  63     # Test that refunded quotas work
  64     client.set("apple", 5)
  65     # increment
  66     is_rate_limited(client, ("orange", "baz"), (1, now + 60))
  67     # test that it's rate limited without refund
  68     assert map(bool, is_rate_limited(client, ("orange", "baz"), (1, now + 60))) == [True]
  69     # test that refund key is used
  70     assert map(bool, is_rate_limited(client, ("orange", "apple"), (1, now + 60))) == [False]

 183     def test_get_usage(self):
 184         timestamp = time.time()
 185 
 186         self.get_project_quota.return_value = (200, 60)
 187         self.get_organization_quota.return_value = (300, 60)
 188 
 189         n = 10
 190         for _ in xrange(n):
 191             self.quota.is_rate_limited(self.project, timestamp=timestamp)
 192 
 193         quotas = self.quota.get_quotas(self.project)
 194         all_quotas = quotas + [
 195             QuotaConfig(id="unlimited", limit=None, window=60, reason_code="unlimited"),
 196             QuotaConfig(id="dummy", limit=10, window=60, reason_code="dummy"),
 197         ]
 198 
 199         usage = self.quota.get_usage(self.project.organization_id, all_quotas, timestamp=timestamp)
 200 
 201         # Only quotas with an ID are counted in Redis (via this ID). Assume the
 202         # count for these quotas and None for the others.
 203         assert usage == [n if q.id else None for q in quotas] + [0, 0]

 240     def test_get_usage_uses_refund(self):
 241         timestamp = time.time()
 242 
 243         self.get_project_quota.return_value = (200, 60)
 244         self.get_organization_quota.return_value = (300, 60)
 245 
 246         n = 10
 247         for _ in xrange(n):
 248             self.quota.is_rate_limited(self.project, timestamp=timestamp)
 249 
 250         self.quota.refund(self.project, timestamp=timestamp)
 251 
 252         quotas = self.quota.get_quotas(self.project)
 253         all_quotas = quotas + [
 254             QuotaConfig(id="unlimited", limit=None, window=60, reason_code="unlimited"),
 255             QuotaConfig(id="dummy", limit=10, window=60, reason_code="dummy"),
 256         ]
 257 
 258         usage = self.quota.get_usage(self.project.organization_id, all_quotas, timestamp=timestamp)
 259 
 260         # Only quotas with an ID are counted in Redis (via this ID). Assume the
 261         # count for these quotas and None for the others.
 262         # The ``- 1`` is because we refunded once.
 263         assert usage == [n - 1 if q.id else None for q in quotas] + [0, 0]
