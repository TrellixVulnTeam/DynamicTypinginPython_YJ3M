   7 class FeatureManagerTest(TestCase):
   8     def test_feature_registry(self):
   9         manager = features.FeatureManager()
  10         assert manager.all() == {}

  13         manager.add("projects:feature2", features.ProjectFeature)

  21     def test_handlers(self):
  22         project_flag = "projects:test_handlers"
  23         test_user = self.create_user()
  24 
  25         class TestProjectHandler(features.FeatureHandler):
  26             features = {project_flag}
  27 
  28             def __init__(self, true_set, false_set):
  29                 self.true_set = frozenset(true_set)
  30                 self.false_set = frozenset(false_set)
  31 
  32             def has(self, feature, actor):
  33                 assert actor == test_user
  34 
  35                 if feature.project in self.true_set:
  36                     return True
  37                 if feature.project in self.false_set:
  38                     return False
  39                 return None
  40 
  41         p1 = self.create_project()
  42         p2 = self.create_project()
  43         p3 = self.create_project()
  44         p4 = self.create_project()
  45 
  46         handlers = [
  47             TestProjectHandler([], []),
  48             TestProjectHandler([p1, p3], []),
  49             TestProjectHandler([], [p2, p3]),
  50         ]
  51 
  52         manager = features.FeatureManager()
  53         manager.add(project_flag, features.ProjectFeature)
  54         for handler in handlers:
  55             manager.add_handler(handler)
  56 
  57         assert manager.has(project_flag, p1, actor=test_user) is True
  58         assert manager.has(project_flag, p2, actor=test_user) is False
  59         assert manager.has(project_flag, p3, actor=test_user) is True
  60         assert manager.has(project_flag, p4, actor=test_user) is False
  61 
  62         organization = None  # Doesn't matter because no BatchFeatureHandlers
  63         assert manager.has_for_batch(
  64             project_flag, organization, [p1, p2, p3, p4], actor=test_user
  65         ) == {p1: True, p2: False, p3: True, p4: False}

  67     def test_has_for_batch(self):
  68         test_user = self.create_user()
  69         test_org = self.create_organization()
  70 
  71         projects = [self.create_project(organization=test_org) for i in range(5)]
  72 
  73         def create_handler(flags, result):
  74             class OrganizationTestHandler(features.BatchFeatureHandler):
  75                 features = frozenset(flags)
  76 
  77                 def __init__(self):
  78                     self.hit_counter = 0
  79 
  80                 def _check_for_batch(self, feature_name, organization, actor):
  81                     assert feature_name in self.features
  82                     assert organization == test_org
  83                     assert actor == test_user
  84 
  85                     self.hit_counter += 1
  86                     return result
  87 
  88             return OrganizationTestHandler()
  89 
  90         yes_flag = "organizations:yes"
  91         no_flag = "organizations:no"
  92 
  93         null_handler = create_handler([yes_flag, no_flag], None)
  94         yes_handler = create_handler([yes_flag], True)
  95         after_yes_handler = create_handler([yes_flag], False)
  96         no_handler = create_handler([no_flag], False)
  97         after_no_handler = create_handler([no_flag], True)
  98 
  99         manager = features.FeatureManager()
 100         for flag in (yes_flag, no_flag):
 101             manager.add(flag, features.OrganizationFeature)
 102 
 103         for handler in (null_handler, yes_handler, after_yes_handler, no_handler, after_no_handler):
 104             manager.add_handler(handler)
 105 
 106         assert manager.has_for_batch(yes_flag, test_org, projects, actor=test_user) == {
 107             p: True for p in projects
 108         }
 109         assert yes_handler.hit_counter == 1  # as opposed to len(projects)
 110         assert after_yes_handler.hit_counter == 0
 111 
 112         assert manager.has_for_batch(no_flag, test_org, projects, actor=test_user) == {
 113             p: False for p in projects
 114         }
 115         assert no_handler.hit_counter == 1
 116         assert after_no_handler.hit_counter == 0
 117 
 118         assert null_handler.hit_counter == 2
