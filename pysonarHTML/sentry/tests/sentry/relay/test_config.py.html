  32 def test_get_project_config(default_project, insta_snapshot, full):
  33     # We could use the default_project fixture here, but we would like to avoid 1) hitting the db 2) creating a mock
  34     default_project.update_option("sentry:relay_pii_config", PII_CONFIG)
  35     default_project.organization.update_option("sentry:relay_pii_config", PII_CONFIG)
  36     keys = ProjectKey.objects.filter(project=default_project)
  37 
  38     cfg = get_project_config(default_project, full_config=full, project_keys=keys)
  39     cfg = cfg.to_dict()
  40 
  41     # Remove keys that change everytime
  42     cfg.pop("lastChange")
  43     cfg.pop("lastFetch")
  44     cfg.pop("rev")
  45 
  46     # public keys change every time
  47     assert cfg.pop("projectId") == default_project.id
  48     assert len(cfg.pop("publicKeys")) == len(keys)
  49     assert cfg.pop("organizationId") == default_project.organization.id
  50
