 163     def test_first_event_received(self):
 164         now = timezone.now()
 165         project = self.create_project(first_event=now)
 166         project_created.send(project=project, user=self.user, sender=type(project))
 167         event = self.store_event(
 168             data={"platform": "javascript", "message": "javascript error message"},
 169             project_id=project.id,
 170         )
 171         first_event_received.send(project=project, event=event, sender=type(project))
 172 
 173         task = OrganizationOnboardingTask.objects.get(
 174             organization=project.organization,
 175             task=OnboardingTask.FIRST_EVENT,
 176             status=OnboardingTaskStatus.COMPLETE,
 177         )
 178         assert task is not None
 179         assert "platform" in task.data
 180         assert task.data["platform"] == "javascript"
 181 
 182         second_project = self.create_project(first_event=now)
 183         project_created.send(project=second_project, user=self.user, sender=type(second_project))
 184         second_task = OrganizationOnboardingTask.objects.get(
 185             organization=second_project.organization,
 186             task=OnboardingTask.SECOND_PLATFORM,
 187             status=OnboardingTaskStatus.PENDING,
 188         )
 189         assert second_task is not None
 190 
 191         second_event = self.store_event(
 192             data={"platform": "python", "message": "python error message"},
 193             project_id=second_project.id,
 194         )
 195         first_event_received.send(
 196             project=second_project, event=second_event, sender=type(second_project)
 197         )
 198         second_task = OrganizationOnboardingTask.objects.get(
 199             organization=second_project.organization,
 200             task=OnboardingTask.SECOND_PLATFORM,
 201             status=OnboardingTaskStatus.COMPLETE,
 202         )
 203         assert second_task is not None
 204         assert "platform" in second_task.data
 205         assert second_task.data["platform"] == "python"
 206         assert task.data["platform"] != second_task.data["platform"]

 285     def test_onboarding_complete(self):
 286         now = timezone.now()
 287         user = self.create_user(email="test@example.org")
 288         project = self.create_project(first_event=now)
 289         second_project = self.create_project(first_event=now)
 290         second_event = self.store_event(
 291             data={"platform": "python", "message": "python error message"},
 292             project_id=second_project.id,
 293         )
 294         event = self.store_event(
 295             data={
 296                 "event_id": "a" * 32,
 297                 "platform": "javascript",
 298                 "timestamp": iso_format(before_now(minutes=1)),
 299                 "tags": {
 300                     "sentry:release": "e1b5d1900526feaf20fe2bc9cad83d392136030a",
 301                     "sentry:user": "id:41656",
 302                 },
 303                 "user": {"ip_address": "0.0.0.0", "id": "41656", "email": "test@example.com"},
 304                 "exception": {
 305                     "values": [
 306                         {
 307                             "stacktrace": {
 308                                 "frames": [
 309                                     {
 310                                         "data": {
 311                                             "sourcemap": "https://media.sentry.io/_static/29e365f8b0d923bc123e8afa38d890c3/sentry/dist/vendor.js.map"
 312                                         }
 313                                     }
 314                                 ]
 315                             },
 316                             "type": "TypeError",
 317                         }
 318                     ]
 319                 },
 320             },
 321             project_id=project.id,
 322         )
 323         member = self.create_member(organization=self.organization, teams=[self.team], user=user)
 324 
 325         event_processed.send(project=project, event=event, sender=type(project))
 326         project_created.send(project=project, user=user, sender=type(project))
 327         project_created.send(project=second_project, user=user, sender=type(second_project))
 328 
 329         first_event_received.send(project=project, event=event, sender=type(project))
 330         first_event_received.send(
 331             project=second_project, event=second_event, sender=type(second_project)
 332         )
 333         member_joined.send(member=member, organization=self.organization, sender=type(member))
 334         plugin_enabled.send(
 335             plugin=IssueTrackingPlugin(),
 336             project=project,
 337             user=user,
 338             sender=type(IssueTrackingPlugin),
 339         )
 340         issue_tracker_used.send(
 341             plugin=IssueTrackingPlugin(),
 342             project=project,
 343             user=user,
 344             sender=type(IssueTrackingPlugin),
 345         )
 346         alert_rule_created.send(
 347             rule=Rule(id=1),
 348             project=self.project,
 349             user=self.user,
 350             rule_type="issue",
 351             sender=type(Rule),
 352         )

 356                 organization=self.organization, key="onboarding:complete"
 357             ).count()
