  17

  19

  21             organization=self.organization, slug=self.sentry_app.slug
  22         )

  30     def update_issue(self, _data=None):

 157 @patch("sentry.tasks.sentry_apps.workflow_notification.delay")
 158 class TestIssueAssigned(APITestCase):
 159     def setUp(self):
 160         self.issue = self.create_group(project=self.project)
 161 
 162         self.sentry_app = self.create_sentry_app(events=["issue.assigned"])
 163 
 164         self.install = self.create_sentry_app_installation(
 165             organization=self.organization, slug=self.sentry_app.slug
 166         )
 167 
 168         self.assignee = self.create_user(name="Bert", email="bert@example.com")
 169 
 170     def test_after_issue_assigned(self, delay):
 171         GroupAssignee.objects.assign(self.issue, self.assignee, self.user)
 172 
 173         assert faux(delay).called_with(
 174             installation_id=self.install.id,
 175             issue_id=self.issue.id,
 176             type="assigned",
 177             user_id=self.user.id,
 178             data={
 179                 "assignee": {
 180                     "type": "user",
 181                     "name": self.assignee.name,
 182                     "email": self.assignee.email,
 183                     "id": self.assignee.id,
 184                 }
 185             },
 186         )
 187 
 188     def test_after_issue_assigned_with_enhanced_privacy(self, delay):
 189         org = self.issue.project.organization
 190         org.flags.enhanced_privacy = True
 191         org.save()
 192 
 193         GroupAssignee.objects.assign(self.issue, self.assignee, self.user)
 194 
 195         assert faux(delay).called_with(
 196             installation_id=self.install.id,
 197             issue_id=self.issue.id,
 198             type="assigned",
 199             user_id=self.user.id,
 200             data={
 201                 # Excludes email address
 202                 "assignee": {"type": "user", "name": self.assignee.name, "id": self.assignee.id}
 203             },
 204         )
