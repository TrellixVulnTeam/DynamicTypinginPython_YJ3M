 135     def test_matching_author_with_assignment(self):
 136         group = self.create_group()
 137         user = self.create_user(name="Foo Bar", email="foo@example.com", is_active=True)
 138         email = UserEmail.get_primary_email(user=user)
 139         email.is_verified = True
 140         email.save()
 141         repo = Repository.objects.create(name="example", organization_id=self.group.organization.id)
 142         OrganizationMember.objects.create(organization=group.project.organization, user=user)
 143         UserOption.objects.set_value(user=user, key="self_assign_issue", value="1")
 144 
 145         commit = Commit.objects.create(
 146             key=sha1(uuid4().hex).hexdigest(),
 147             organization_id=group.organization.id,
 148             repository_id=repo.id,
 149             message=u"Foo Biz\n\nFixes {}".format(group.qualified_short_id),
 150             author=CommitAuthor.objects.create(
 151                 organization_id=group.organization.id, name=user.name, email=user.email
 152             ),
 153         )
 154 
 155         self.assertResolvedFromCommit(group, commit)
 156 
 157         assert GroupAssignee.objects.filter(group=group, user=user).exists()
 158 
 159         assert Activity.objects.filter(
 160             project=group.project, group=group, type=Activity.ASSIGNED, user=user
 161         )[0].data == {
 162             "assignee": six.text_type(user.id),
 163             "assigneeEmail": user.email,
 164             "assigneeType": "user",
 165         }
 166 
 167         assert GroupSubscription.objects.filter(group=group, user=user).exists()
