  14 def test_is_url():
  15     assert is_url("http://example.org/") is True
  16     assert is_url("https://example.org/") is True
  17     assert is_url("file:///tmp/filename") is True
  18     assert is_url("applewebdata://00000000-0000-1000-8080-808080808080") is True
  19     assert is_url("app:///index.bundle") is False  # react native
  20     assert is_url("webpack:///./app/index.jsx") is False  # webpack bundle
  21     assert is_url("data:,") is False
  22     assert is_url("blob:\x00") is False
