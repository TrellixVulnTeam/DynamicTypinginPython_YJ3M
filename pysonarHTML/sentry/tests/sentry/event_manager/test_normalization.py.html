 137 def test_moves_stacktrace_to_exception():
 138     manager = EventManager(
 139         make_event(
 140             exception={"type": "MyException"},
 141             stacktrace={
 142                 "frames": [{"lineno": 1, "filename": "foo.py"}, {"lineno": 1, "filename": "bar.py"}]
 143             },
 144         )
 145     )
 146     manager.normalize()
 147     data = manager.get_data()
 148 
 149     frames = data["exception"]["values"][0]["stacktrace"]["frames"]
 150     assert frames[0]["lineno"] == 1
 151     assert frames[0]["filename"] == "foo.py"
 152     assert frames[1]["lineno"] == 1
 153     assert frames[1]["filename"] == "bar.py"
 154     assert "stacktrace" not in data
