  10 def validate_and_normalize(report, client_ip="198.51.100.0", user_agent="Awesome Browser"):
  11     manager = EventManager(report, client_ip=client_ip, user_agent=user_agent)
  12     manager.process_csp_report()
  13     manager.normalize()
  14     return manager.get_data()

  42 @pytest.mark.parametrize(
  43     "effective_directive,violated_directive,culprit_element",
  44     (
  45         ("img-src", "img-src https://45.55.25.245:8123/", "img-src"),
  46         ("img-src", "default-src https://45.55.25.245:8123/", "default-src"),
  47         # build a report without the effective-directive key
  48         (None, "img-src https://45.55.25.245:8123/", "img-src"),
  49     ),
  50     ids=(
  51         "directives match",
  52         "prefer effective-directive",
  53         "parse effective-directive from violated-directive",
  54     ),
  55 )
  56 def test_csp_validate(effective_directive, violated_directive, culprit_element):
  57     report = _build_test_report(effective_directive, violated_directive)
  58     result = validate_and_normalize(report)
  59     assert result["logger"] == "csp"
  60     assert result["release"] == "abc123"
  61     assert result["environment"] == "production"
  62     assert "errors" not in result
  63     assert "logentry" in result
  64     assert result["culprit"] == culprit_element + " 'self'"
  65     assert map(tuple, result["tags"]) == [
  66         ("effective-directive", "img-src"),
  67         ("blocked-uri", "http://google.com"),
  68     ]
  69     assert result["user"] == {"ip_address": "198.51.100.0"}
  70     assert result["request"]["url"] == "http://45.55.25.245:8123/csp"
  71     assert dict(result["request"]["headers"]) == {
  72         "User-Agent": "Awesome Browser",
  73         "Referer": "http://example.com",
  74     }

 144 def test_hpkp_validate_basic():
 145     report = {
 146         "release": "abc123",
 147         "interface": "hpkp",
 148         "report": {
 149             "date-time": "2014-04-06T13:00:50Z",
 150             "hostname": "www.example.com",
 151             "port": 443,
 152             "effective-expiration-date": "2014-05-01T12:40:50Z",
 153             "include-subdomains": False,
 154             "served-certificate-chain": ["-----BEGIN CERTIFICATE-----\n-----END CERTIFICATE-----"],
 155             "validated-certificate-chain": [
 156                 "-----BEGIN CERTIFICATE-----\n-----END CERTIFICATE-----"
 157             ],
 158             "known-pins": ['pin-sha256="E9CZ9INDbd+2eRQozYqqbQ2yXLVKB9+xcprMF+44U1g="'],
 159         },
 160     }
 161     result = validate_and_normalize(report)
 162     assert result["release"] == "abc123"
 163     assert "errors" not in result
 164     assert "logentry" in result
 165     assert not result.get("culprit")
 166     assert sorted(map(tuple, result["tags"])) == [
 167         ("hostname", "www.example.com"),
 168         ("include-subdomains", "false"),
 169         ("port", "443"),
 170     ]
 171     assert result["user"] == {"ip_address": "198.51.100.0"}
 172     expected_headers = [["User-Agent", "Awesome Browser"]]
 173 
 174     assert result["request"] == {"url": "www.example.com", "headers": expected_headers}
