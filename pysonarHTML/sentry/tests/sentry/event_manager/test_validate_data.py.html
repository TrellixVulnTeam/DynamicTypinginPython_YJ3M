  11 def validate_and_normalize(data):
  12     manager = EventManager(data)
  13     manager.normalize()
  14     return manager.get_data()

  18     data = validate_and_normalize({"timestamp": "not-a-timestamp"})
  19     assert len(data["errors"]) == 1

 284 def test_fingerprints():

 286     assert not data.get("fingerprint")
 287     assert data["errors"][0]["type"] == "invalid_data"
 288     assert data["errors"][0]["name"] == "fingerprint"

 290     data = validate_and_normalize({"fingerprint": ["foo", ["bar"]]})
 291     assert data.get("fingerprint") == ["foo"]
 292     # With rust, there will be errors emitted
 293 
 294     data = validate_and_normalize({"fingerprint": ["foo", None, "bar"]})
 295     assert data.get("fingerprint") == ["foo", "bar"]

 300     assert "errors" not in data

 317 def test_messages():
 318     # Just 'message': wrap it in interface
 319     data = validate_and_normalize({"message": "foo is bar"})
 320     assert data["logentry"] == {"formatted": "foo is bar"}
 321 
 322     # both 'message' and interface with no 'formatted' value, put 'message'
 323     # into 'formatted'.
 324     data = validate_and_normalize(
 325         {"message": "foo is bar", "logentry": {"message": "something else"}}
 326     )
 327     assert data["logentry"] == {"formatted": "something else"}
 328 
 329     # both 'message' and complete interface, 'message' is discarded
 330     data = validate_and_normalize(
 331         {
 332             "message": "foo is bar",
 333             "logentry": {"message": "something else", "formatted": "something else formatted"},
 334         }
 335     )
 336     assert "errors" not in data
 337     assert data["logentry"] == {
 338         "message": "something else",
 339         "formatted": "something else formatted",
 340     }

 344 def test_messages_old_behavior():
