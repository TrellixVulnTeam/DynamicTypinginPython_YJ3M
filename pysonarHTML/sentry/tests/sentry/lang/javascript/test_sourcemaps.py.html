  57 class FindSourceTest(TestCase):
  58     def test_simple(self):
  59         smap_view = SourceMapView.from_json_bytes(sourcemap)
  60 
  61         result = smap_view.lookup(0, 56)
  62         assert result == SourceMapTokenMatch(
  63             dst_line=0,
  64             dst_col=50,
  65             src="foo/file2.js",
  66             src_line=0,
  67             src_col=9,
  68             src_id=1,
  69             name="multiply",
  70         )
  71 
  72         # Start of minified file (exact match first line/col tuple)
  73         result = smap_view.lookup(0, 0)
  74         assert result == SourceMapTokenMatch(
  75             dst_line=0, dst_col=0, src="foo/file1.js", src_line=0, src_col=0, src_id=0, name=None
  76         )
  77 
  78         # Last character in mapping
  79         result = smap_view.lookup(0, 36)
  80         assert result == SourceMapTokenMatch(
  81             dst_line=0, dst_col=30, src="foo/file1.js", src_line=2, src_col=1, src_id=0, name=None
  82         )
  83 
  84         # First character in mapping (exact match line/col tuple)
  85         result = smap_view.lookup(0, 37)
  86         assert result == SourceMapTokenMatch(
  87             dst_line=0, dst_col=37, src="foo/file1.js", src_line=2, src_col=8, src_id=0, name="a"
  88         )
  89 
  90         # End of minified file (character *beyond* last line/col tuple)
  91         result = smap_view.lookup(0, 192)
  92         assert result == SourceMapTokenMatch(
  93             dst_line=0, dst_col=191, src="foo/file2.js", src_line=9, src_col=25, src_id=1, name="e"
  94         )

 445 class ParseIndexedSourcemapTest(TestCase):
 446     # Tests lookups that fall exactly on source map token boundaries
 447     # https://github.com/mozilla/source-map/blob/master/test/test-source-map-consumer.js#138
 448     def test_exact_mappings(self):
 449         smap_view = SourceMapView.from_json_bytes(indexed_sourcemap_example)
 450 
 451         # one.js
 452         assert smap_view.lookup(0, 1) == SourceMapTokenMatch(
 453             dst_line=0,
 454             dst_col=1,
 455             src="/the/root/one.js",
 456             src_line=0,
 457             src_col=1,
 458             src_id=0,
 459             name=None,
 460         )
 461         assert smap_view.lookup(0, 18) == SourceMapTokenMatch(
 462             dst_line=0,
 463             dst_col=18,
 464             src="/the/root/one.js",
 465             src_line=0,
 466             src_col=21,
 467             src_id=0,
 468             name="bar",
 469         )
 470         assert smap_view.lookup(0, 28) == SourceMapTokenMatch(
 471             dst_line=0,
 472             dst_col=28,
 473             src="/the/root/one.js",
 474             src_line=1,
 475             src_col=10,
 476             src_id=0,
 477             name="baz",
 478         )
 479 
 480         # two.js
 481         assert smap_view.lookup(1, 18) == SourceMapTokenMatch(
 482             dst_line=1,
 483             dst_col=18,
 484             src="/the/root/two.js",
 485             src_line=0,
 486             src_col=21,
 487             src_id=1,
 488             name="n",
 489         )
 490         assert smap_view.lookup(1, 21) == SourceMapTokenMatch(
 491             dst_line=1,
 492             dst_col=21,
 493             src="/the/root/two.js",
 494             src_line=1,
 495             src_col=3,
 496             src_id=1,
 497             name=None,
 498         )
 499         assert smap_view.lookup(1, 21) == SourceMapTokenMatch(
 500             dst_line=1,
 501             dst_col=21,
 502             src="/the/root/two.js",
 503             src_line=1,
 504             src_col=3,
 505             src_id=1,
 506             name=None,
 507         )
