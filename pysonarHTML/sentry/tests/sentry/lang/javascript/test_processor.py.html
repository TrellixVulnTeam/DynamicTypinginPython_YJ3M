 349     def test_max_and_min(self):
 350         headers = {
 351             "content-type": "application/json",

 353         }
 354         assert get_max_age(headers) == CACHE_CONTROL_MAX
 355 
 356         headers = {
 357             "content-type": "application/json",
 358             "cache-control": "max-age=%s" % CACHE_CONTROL_MIN,

 360         assert get_max_age(headers) == CACHE_CONTROL_MIN
 361 
 362     def test_out_of_bounds(self):
 363         greater_than_max = CACHE_CONTROL_MAX + 1
 364         headers = {
 365             "content-type": "application/json",

 367         }
 368         assert get_max_age(headers) == CACHE_CONTROL_MAX
 369 
 370         less_than_min = CACHE_CONTROL_MIN - 1
 371         headers = {
 372             "content-type": "application/json",
 373             "cache-control": "max-age=%s" % less_than_min,
 374         }
 375         assert get_max_age(headers) == CACHE_CONTROL_MIN

 822 class CacheSourceTest(TestCase):
 823     def test_file_no_source_records_error(self):
 824         """
 825         If we can't find a given file, either on the release or by scraping, an
 826         error should be recorded.
 827         """
 828 
 829         project = self.create_project()
 830 
 831         processor = JavaScriptStacktraceProcessor(data={}, stacktrace_infos=None, project=project)
 832 
 833         # no release on the event, so won't find file in database
 834         assert processor.release is None
 835 
 836         # not a real url, so won't find file on the internet
 837         abs_path = "app:///i/dont/exist.js"
 838 
 839         # before caching, no errors
 840         assert len(processor.cache.get_errors(abs_path)) == 0
 841 
 842         processor.cache_source(abs_path)
 843 
 844         # now we have an error
 845         assert len(processor.cache.get_errors(abs_path)) == 1
 846         assert processor.cache.get_errors(abs_path)[0] == {"url": abs_path, "type": "js_no_source"}
 847 
 848     def test_node_modules_file_no_source_no_error(self):
 849         """
 850         If someone hasn't uploaded node_modules (which most people don't), it
 851         shouldn't complain about a source file being missing.
 852         """
 853 
 854         project = self.create_project()
 855         processor = JavaScriptStacktraceProcessor(data={}, stacktrace_infos=None, project=project)
 856 
 857         # no release on the event, so won't find file in database
 858         assert processor.release is None
 859 
 860         # not a real url, so won't find file on the internet
 861         abs_path = "app:///../node_modules/i/dont/exist.js"
 862 
 863         processor.cache_source(abs_path)
 864 
 865         # no errors, even though the file can't have been found
 866         assert len(processor.cache.get_errors(abs_path)) == 0

 868     def test_node_modules_file_with_source_is_used(self):
 869         """
 870         If someone has uploaded node_modules, files in there should be treated like
 871         any other files (in other words, they should land in the cache with no errors).
 872         """
 873 
 874         project = self.create_project()
 875         release = self.create_release(project=project, version="12.31.12")
 876 
 877         abs_path = "app:///../node_modules/some-package/index.js"
 878         self.create_release_file(release=release, name=abs_path)
 879 
 880         processor = JavaScriptStacktraceProcessor(
 881             data={"release": release.version}, stacktrace_infos=None, project=project
 882         )
 883         # in real life the preprocess step will pull release out of the data
 884         # dictionary passed to the JavaScriptStacktraceProcessor constructor,
 885         # but since this is just a unit test, we have to set it manually
 886         processor.release = release
 887 
 888         processor.cache_source(abs_path)
 889 
 890         # file is cached, no errors are generated
 891         assert processor.cache.get(abs_path)
 892         assert len(processor.cache.get_errors(abs_path)) == 0
