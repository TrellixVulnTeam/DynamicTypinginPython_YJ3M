 125 def test_merge_attached_breadcrumbs_capped():
 126     # Crumbs are capped by the largest file
 127     event = {}
 128 
 129     crumbs_file1 = bytearray()
 130     for i in range(0, 2):
 131         crumbs_file1.extend(msgpack.packb({"timestamp": "0001-01-01T01:00:01Z"}))
 132 
 133     merge_attached_breadcrumbs(MockFile(crumbs_file1), event)
 134     assert len(event["breadcrumbs"]) == 2
 135     assert event["breadcrumbs"][0]["timestamp"] == "0001-01-01T01:00:01Z"
 136     assert event["breadcrumbs"][1]["timestamp"] == "0001-01-01T01:00:01Z"
 137 
 138     crumbs_file2 = bytearray()
 139     for i in range(0, 3):
 140         crumbs_file2.extend(msgpack.packb({"timestamp": "0001-01-01T01:00:02Z"}))
 141 
 142     merge_attached_breadcrumbs(MockFile(crumbs_file2), event)
 143     assert len(event["breadcrumbs"]) == 3
 144     assert event["breadcrumbs"][0]["timestamp"] == "0001-01-01T01:00:02Z"
 145     assert event["breadcrumbs"][1]["timestamp"] == "0001-01-01T01:00:02Z"
 146     assert event["breadcrumbs"][2]["timestamp"] == "0001-01-01T01:00:02Z"
 147 
 148     crumbs_file3 = msgpack.packb({"timestamp": "0001-01-01T01:00:03Z"})
 149     merge_attached_breadcrumbs(MockFile(crumbs_file3), event)
 150     assert len(event["breadcrumbs"]) == 3
 151     assert event["breadcrumbs"][0]["timestamp"] == "0001-01-01T01:00:02Z"
 152     assert event["breadcrumbs"][1]["timestamp"] == "0001-01-01T01:00:02Z"
 153     assert event["breadcrumbs"][2]["timestamp"] == "0001-01-01T01:00:03Z"
 154 
 155     crumbs_file4 = msgpack.packb({"timestamp": "0001-01-01T01:00:04Z"})
 156     merge_attached_breadcrumbs(MockFile(crumbs_file4), event)
 157     assert len(event["breadcrumbs"]) == 3

 159     assert event["breadcrumbs"][1]["timestamp"] == "0001-01-01T01:00:03Z"

 163 def test_is_minidump():
 164     assert is_minidump_event({"exception": {"values": [{"mechanism": {"type": "minidump"}}]}})
 165     assert not is_minidump_event({"exception": {"values": [{"mechanism": {"type": "other"}}]}})
 166     assert not is_minidump_event({"exception": {"values": [{"mechanism": {"type": None}}]}})
 167     assert not is_minidump_event({"exception": {"values": [{"mechanism": None}]}})
 168     assert not is_minidump_event({"exception": {"values": [None]}})
 169     assert not is_minidump_event({"exception": {"values": []}})
 170     assert not is_minidump_event({"exception": {"values": None}})
 171     assert not is_minidump_event({"exception": None})
