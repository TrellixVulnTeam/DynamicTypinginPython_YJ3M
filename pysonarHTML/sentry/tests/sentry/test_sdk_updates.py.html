 181 def test_more_specific_dotnet_sdk(some_dotnet_sdk):
 182     setup = SdkSetupState(
 183         sdk_name=some_dotnet_sdk,
 184         sdk_version="1.2.0",
 185         integrations=[],
 186         modules={
 187             "Microsoft.AspNetCore.Hosting": "2.1.0",
 188             "Microsoft.Extensions.Logging.Configuration": "2.1.0",
 189         },
 190     )
 191 
 192     suggestions = list(get_suggested_updates(setup, DOTNET_INDEX_STATE))
 193     if some_dotnet_sdk == "sentry.dotnet.aspnetcore":
 194         assert not suggestions
 195     else:
 196         # Do not show suggestion for sentry.dotnet.extensions.logging. The user
 197         # has a aspnetcore application and the aspnetcore SDK instruments
 198         # logging too.
 199         assert suggestions == [
 200             {
 201                 "enables": [],
 202                 "newSdkName": "sentry.dotnet.aspnetcore",
 203                 "sdkUrl": None,
 204                 "type": "changeSdk",
 205             }
 206         ]
