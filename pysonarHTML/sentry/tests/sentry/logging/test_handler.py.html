  20 def make_logrecord(**extra):
  21     kwargs = dict(
  22         name="name",
  23         level=logging.INFO,
  24         pathname="pathname",
  25         lineno=10,
  26         msg="msg",
  27         args=None,
  28         exc_info=None,
  29     )
  30     kwargs.update(extra or {})
  31     return logging.LogRecord(**kwargs)

  34 @pytest.mark.parametrize(
  35     "record,out",
  36     (
  37         ({}, {}),

  39         ({"args": ({"a": 1},)}, {"positional_args": ({"a": 1},)}),

  43 def test_emit(record, out, handler, logger):
