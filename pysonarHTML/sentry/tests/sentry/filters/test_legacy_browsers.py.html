 225     def _get_project_config(self, filter_opt=None):
 226         """
 227         Constructs a test project_config with the provided legacy_browsers filter setting
 228         :param filter_opt: the value for 'filters:legacy-browsers' project options (may be None)
 229         :return: a ProjectConfig object with the filter option set and the project taken from
 230         the TestCase
 231         """
 232         ret_val = ProjectConfig(self.project, config={})
 233         config = ret_val.config
 234         filter_settings = {}
 235         config["filterSettings"] = filter_settings
 236         if filter_opt is not None:
 237             key = get_filter_key(_legacy_browsers_filter)
 238             filter_settings[key] = _filter_option_to_config_setting(
 239                 _legacy_browsers_filter, filter_opt
 240             )
 241         return ret_val
