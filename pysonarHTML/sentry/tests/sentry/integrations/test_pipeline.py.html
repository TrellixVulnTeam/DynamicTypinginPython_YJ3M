 151     def test_default_identity_does_update(self, *args):
 152         self.provider.needs_default_identity = True
 153         old_identity_id = 234567
 154         integration = Integration.objects.create(
 155             provider=self.provider.key,
 156             external_id=self.external_id,
 157             metadata={"url": "https://example.com"},
 158         )
 159         OrganizationIntegration.objects.create(
 160             organization=self.organization, integration=integration, default_auth_id=old_identity_id
 161         )
 162         self.pipeline.state.data = {
 163             "external_id": self.external_id,
 164             "name": "Name",
 165             "metadata": {"url": "https://example.com"},
 166             "user_identity": {
 167                 "type": "plugin",
 168                 "external_id": "AccountId",
 169                 "scopes": [],
 170                 "data": {
 171                     "access_token": "token12345",
 172                     "expires_in": "123456789",
 173                     "refresh_token": "refresh12345",
 174                     "token_type": "typetype",
 175                 },
 176             },
 177         }
 178 
 179         resp = self.pipeline.finish_pipeline()
 180         self.assertDialogSuccess(resp)

 183             organization_id=self.organization.id, integration_id=integration.id
 184         )

 188     def test_existing_identity_becomes_default_auth_on_new_orgintegration(self, *args):
 189         # The reinstall flow will result in an existing identity provider, identity
 190         # and integration records. Ensure that the new organizationintegration gets
 191         # a default_auth_id set.
 192         self.provider.needs_default_identity = True
 193         integration = Integration.objects.create(
 194             provider=self.provider.key,
 195             external_id=self.external_id,
 196             metadata={"url": "https://example.com"},
 197         )
 198         identity_provider = IdentityProvider.objects.create(
 199             external_id=self.external_id, type="plugin"
 200         )
 201         identity = Identity.objects.create(
 202             idp_id=identity_provider.id, external_id="AccountId", user_id=self.user.id
 203         )
 204         self.pipeline.state.data = {
 205             "external_id": self.external_id,
 206             "name": "Name",
 207             "metadata": {"url": "https://example.com"},
 208             "user_identity": {
 209                 "type": "plugin",
 210                 "external_id": "AccountId",
 211                 "scopes": [],
 212                 "data": {
 213                     "access_token": "token12345",
 214                     "expires_in": "123456789",
 215                     "refresh_token": "refresh12345",
 216                     "token_type": "typetype",
 217                 },
 218             },
 219         }
 220         resp = self.pipeline.finish_pipeline()
 221         self.assertDialogSuccess(resp)
 222 
 223         org_integration = OrganizationIntegration.objects.get(
 224             organization_id=self.organization.id, integration_id=integration.id
 225         )
 226         assert org_integration.default_auth_id == identity.id
