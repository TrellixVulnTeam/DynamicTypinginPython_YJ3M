  95     @responses.activate
  96     def test_build_repository_config(self):
  97         project = u"laurynsentry"
  98         repo = u"helloworld"
  99         full_repo_name = u"%s/%s" % (project, repo)
 100 
 101         webhook_id = 79
 102         responses.add(
 103             responses.GET,
 104             "https://bitbucket.example.com/rest/api/1.0/projects/%s/repos/%s" % (project, repo),
 105             json=REPO,
 106         )
 107         responses.add(
 108             responses.POST,
 109             "https://bitbucket.example.com/rest/api/1.0/projects/%s/repos/%s/webhooks"
 110             % (project, repo),
 111             json={"id": webhook_id},
 112             status=201,
 113         )
 114 
 115         organization = self.organization
 116         integration = self.integration
 117 
 118         data = {
 119             "provider": "integrations:bitbucket_server",
 120             "identifier": project + "/" + repo,
 121             "installation": integration.id,
 122         }
 123         data = self.provider.get_repository_data(organization, data)
 124         assert data == {
 125             "provider": "integrations:bitbucket_server",
 126             "repo": repo,
 127             "project": project,
 128             "identifier": project + "/" + repo,
 129             "name": full_repo_name,
 130             "installation": integration.id,
 131             "external_id": six.text_type(REPO["id"]),
 132         }
 133 
 134         data["identifier"] = full_repo_name
 135         data = self.provider.build_repository_config(organization, data)
 136 
 137         assert data == {
 138             "name": full_repo_name,
 139             "external_id": six.text_type(REPO["id"]),
 140             "url": "https://bitbucket.example.com/projects/laurynsentry/repos/helloworld/browse",
 141             "integration_id": integration.id,
 142             "config": {
 143                 "name": full_repo_name,
 144                 "project": project,
 145                 "repo": repo,
 146                 "webhook_id": webhook_id,
 147             },
 148         }
