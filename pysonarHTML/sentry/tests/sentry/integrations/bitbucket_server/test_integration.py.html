 239         responses.add(
 240             responses.POST,
 241             "https://bitbucket.example.com/plugins/servlet/oauth/request-token",
 242             status=200,
 243             content_type="text/plain",
 244             body="oauth_token=abc123&oauth_token_secret=def456",
 245         )
 246         responses.add(
 247             responses.POST,
 248             "https://bitbucket.example.com/plugins/servlet/oauth/access-token",
 249             status=200,
 250             content_type="text/plain",
 251             body="oauth_token=valid-token&oauth_token_secret=valid-secret",
 252         )
 253         responses.add(
 254             responses.POST,
 255             "https://bitbucket.example.com/rest/webhooks/1.0/webhook",
 256             status=204,
 257             body="",
 258         )
 259 
 260         self.install_integration()
 261 
 262         integration = Integration.objects.get()
 263         assert integration.name == "sentry-bot"
 264         assert integration.metadata["domain_name"] == "bitbucket.example.com"
 265         assert integration.metadata["base_url"] == "https://bitbucket.example.com"
 266         assert integration.metadata["verify_ssl"] is False
 267 
 268         org_integration = OrganizationIntegration.objects.get(
 269             integration=integration, organization=self.organization
 270         )
 271         assert org_integration.config == {}
 272 
 273         idp = IdentityProvider.objects.get(type="bitbucket_server")
 274         identity = Identity.objects.get(
 275             idp=idp, user=self.user, external_id="bitbucket.example.com:sentry-bot"
 276         )
 277         assert identity.data["consumer_key"] == "sentry-bot"
 278         assert identity.data["access_token"] == "valid-token"
 279         assert identity.data["access_token_secret"] == "valid-secret"
 280         assert identity.data["private_key"] == EXAMPLE_PRIVATE_KEY
