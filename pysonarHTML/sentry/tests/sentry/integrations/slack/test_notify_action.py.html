  32     @responses.activate
  33     def test_applies_correctly(self):
  34         event = self.get_event()
  35 
  36         rule = self.get_rule(data={"workspace": self.integration.id, "channel": "#my-channel"})
  37 
  38         results = list(rule.after(event=event, state=self.get_state()))
  39         assert len(results) == 1
  40 
  41         responses.add(
  42             method=responses.POST,
  43             url="https://slack.com/api/chat.postMessage",
  44             body='{"ok": true}',
  45             status=200,
  46             content_type="application/json",
  47         )
  48 
  49         # Trigger rule callback
  50         results[0].callback(event, futures=[])
  51         data = parse_qs(responses.calls[0].request.body)
  52 
  53         assert "attachments" in data
  54         attachments = json.loads(data["attachments"][0])
  55 
  56         assert len(attachments) == 1
  57         assert attachments[0]["title"] == event.title
