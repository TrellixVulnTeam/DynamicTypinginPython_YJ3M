  50 class BaseEventTest(APITestCase):
  51     def setUp(self):
  52         super(BaseEventTest, self).setUp()
  53         self.user = self.create_user(is_superuser=False)
  54         self.org = self.create_organization(owner=None)
  55         self.integration = Integration.objects.create(
  56             provider="slack",
  57             external_id="TXXXXXXX1",
  58             metadata={"access_token": "xoxp-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx"},
  59         )
  60         OrganizationIntegration.objects.create(organization=self.org, integration=self.integration)
  61 
  62     def post_webhook(
  63         self, event_data=None, type="event_callback", data=None, token=UNSET, team_id="TXXXXXXX1"
  64     ):
  65         if token is UNSET:
  66             token = options.get("slack.verification-token")
  67         payload = {
  68             "token": token,
  69             "team_id": team_id,
  70             "api_app_id": "AXXXXXXXX1",
  71             "type": type,
  72             "authed_users": [],
  73             "event_id": "Ev08MFMKH6",
  74             "event_time": 123456789,
  75         }
  76         if data:
  77             payload.update(data)
  78         if event_data:
  79             payload.setdefault("event", {}).update(event_data)
  80         return self.client.post("/extensions/slack/event/", payload)

 106 class LinkSharedEventTest(BaseEventTest):
 107     @responses.activate
 108     def test_valid_token(self):
 109         responses.add(responses.POST, "https://slack.com/api/chat.unfurl", json={"ok": True})
 110         org2 = self.create_organization(name="biz")
 111         project1 = self.create_project(organization=self.org)
 112         project2 = self.create_project(organization=org2)
 113         group1 = self.create_group(project=project1)
 114         group2 = self.create_group(project=project2)
 115         alert_rule = self.create_alert_rule()
 116         incident = self.create_incident(
 117             status=2, organization=self.org, projects=[project1], alert_rule=alert_rule
 118         )
 119         incident.update(identifier=123)
 120         resp = self.post_webhook(
 121             event_data=json.loads(
 122                 LINK_SHARED_EVENT
 123                 % {
 124                     "group1": group1.id,
 125                     "group2": group2.id,
 126                     "incident": incident.identifier,
 127                     "org1": self.org.slug,
 128                     "org2": org2.slug,
 129                 }
 130             )
 131         )
 132         assert resp.status_code == 200, resp.content
 133         data = dict(parse_qsl(responses.calls[0].request.body))
 134         unfurls = json.loads(data["unfurls"])
 135         issue_url = "http://testserver/organizations/%s/issues/%s/bar/" % (self.org.slug, group1.id)
 136         incident_url = "http://testserver/organizations/%s/incidents/%s/" % (
 137             self.org.slug,
 138             incident.identifier,
 139         )
 140         assert unfurls == {
 141             issue_url: build_group_attachment(group1),
 142             incident_url: build_incident_attachment(incident),
 143         }
 144         assert data["token"] == "xoxp-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx"
