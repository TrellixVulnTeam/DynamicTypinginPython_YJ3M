  98     def setUp(self):

 113     def set_signature(self, secret, data):
 114         timestamp = six.text_type(int(time.mktime(datetime.utcnow().timetuple())))
 115         req = six.binary_type("v0:%s:%s" % (timestamp, six.binary_type(data)))
 116 
 117         signature = "v0=" + hmac.new(six.binary_type(secret), req, sha256).hexdigest()
 118         self.request.META["HTTP_X_SLACK_REQUEST_TIMESTAMP"] = timestamp
 119         self.request.META["HTTP_X_SLACK_SIGNATURE"] = signature
 120 
 121     @memoize
 122     def slack_request(self):
 123         return SlackEventRequest(self.request)
 124

 130         }
 131

 133         # the `event` key.

 160     def test_signing_secret(self):
 161         with override_options({"slack.signing-secret": "secret"}):
 162             self.request.data = {"challenge": "abc123", "type": "url_verification"}
 163 
 164             # we get a url encoded body with Slack
 165             self.request.body = urlencode(self.request.data)

 168             self.slack_request.validate()

 180     def test_signing_secret_bad(self):
 181         with override_options({"slack.signing-secret": "secret"}):
 182             # even though we provide the token, should still fail
 183             self.request.data = {
 184                 "token": options.get("slack.verification-token"),
 185                 "challenge": "abc123",
 186                 "type": "url_verification",
 187             }
 188             self.request.body = urlencode(self.request.data)
 189 
 190             self.set_signature("bad_key", self.request.body)
 191             with self.assertRaises(SlackRequestError) as e:
 192                 self.slack_request.validate()
 193                 assert e.status == 401
