  58     def add_list_response(self, list_type, channels, result_name="channels"):
  59         self.resp.add(
  60             method=responses.GET,
  61             url="https://slack.com/api/%s.list" % list_type,
  62             status=200,
  63             content_type="application/json",
  64             body=json.dumps({"ok": "true", result_name: channels}),
  65         )

  93 class BuildIncidentAttachmentTest(TestCase):
  94     def test_simple(self):
  95         logo_url = absolute_uri(get_asset_url("sentry", "images/sentry-email-avatar.png"))
  96         alert_rule = self.create_alert_rule()
  97         incident = self.create_incident(alert_rule=alert_rule, status=2)
  98         title = u"{}: {}".format("Resolved", alert_rule.name)
  99         assert build_incident_attachment(incident) == {
 100             "fallback": title,
 101             "title": title,
 102             "title_link": absolute_uri(
 103                 reverse(
 104                     "sentry-metric-alert",
 105                     kwargs={
 106                         "organization_slug": incident.organization.slug,
 107                         "incident_id": incident.identifier,
 108                     },
 109                 )
 110             ),
 111             "text": "0 events in the last 10 minutes\nFilter: level:error",
 112             "fields": [],
 113             "mrkdwn_in": ["text"],
 114             "footer_icon": logo_url,
 115             "footer": "Sentry Incident",
 116             "ts": to_timestamp(incident.date_started),
 117             "color": RESOLVED_COLOR,
 118             "actions": [],
 119         }
