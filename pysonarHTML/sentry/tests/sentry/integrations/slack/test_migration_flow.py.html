  54     def assert_setup_flow(
  55         self,
  56         team_id="TXXXXXXX1",
  57         authorizing_user_id="UXXXXXXX1",
  58         expected_client_id="slack-client-id",
  59         expected_client_secret="slack-client-secret",
  60         check_channels=True,
  61     ):
  62         responses.reset()
  63 
  64         resp = self.client.get(
  65             u"{}?{}".format(self.init_path, urlencode({"integration_id": self.integration.id}))
  66         )
  67         assert resp.status_code == 200
  68 
  69         responses.add(
  70             responses.POST,
  71             "https://slack.com/api/conversations.info",
  72             json={"ok": True, "channel": {"is_private": True}},
  73         )
  74         resp = self.client.get(self.init_path_verification_results)
  75         assert resp.status_code == 200
  76         resp = self.client.get(self.init_path_channels)
  77         assert resp.status_code == 302
  78 
  79         redirect = urlparse(resp["Location"])
  80         assert redirect.scheme == "https"
  81         assert redirect.netloc == "slack.com"
  82         assert redirect.path == "/oauth/v2/authorize"
  83         params = parse_qs(redirect.query)
  84         scopes = self.provider.identity_oauth_scopes
  85         assert params["scope"] == [" ".join(scopes)]
  86         assert params["state"]
  87         assert params["redirect_uri"] == ["http://testserver/extensions/slack/setup/"]
  88         assert params["response_type"] == ["code"]
  89         assert params["client_id"] == [expected_client_id]
  90 
  91         assert params.get("user_scope") == ["links:read"]
  92         # once we've asserted on it, switch to a singular values to make life
  93         # easier
  94         authorize_params = {k: v[0] for k, v in six.iteritems(params)}
  95 
  96         access_json = {
  97             "ok": True,
  98             "access_token": "xoxb-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx",
  99             "team": {"id": team_id, "name": "Example"},
 100             "authed_user": {"id": authorizing_user_id},
 101         }
 102         responses.add(responses.POST, "https://slack.com/api/oauth.v2.access", json=access_json)
 103 
 104         responses.add(
 105             responses.GET,
 106             "https://slack.com/api/team.info",
 107             json={
 108                 "ok": True,
 109                 "team": {
 110                     "domain": "test-slack-workspace",
 111                     "icon": {"image_132": "http://example.com/ws_icon.jpg"},
 112                 },
 113             },
 114         )
 115 
 116         resp = self.client.get(
 117             u"{}?{}".format(
 118                 self.setup_path,
 119                 urlencode({"code": "oauth-code", "state": authorize_params["state"]}),
 120             )
 121         )
 122 
 123         if check_channels:
 124             mock_request = responses.calls[1].request
 125             req_params = parse_qs(mock_request.body)
 126             assert req_params["grant_type"] == ["authorization_code"]
 127             assert req_params["code"] == ["oauth-code"]
 128             assert req_params["redirect_uri"] == ["http://testserver/extensions/slack/setup/"]
 129             assert req_params["client_id"] == [expected_client_id]
 130             assert req_params["client_secret"] == [expected_client_secret]
 131 
 132         assert resp.status_code == 200
 133         self.assertDialogSuccess(resp)

 137     def test_migration_flow(self, run_post_migration):
 138         self.assert_setup_flow()
 139 
 140         integration = Integration.objects.get(provider=self.provider.key)
 141

 143         assert integration.name == "Example"

 146         assert md["old_access_token"] == "xoxa-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx"
 147         assert md["scopes"] == sorted(self.provider.identity_oauth_scopes)
 148         assert md["icon"] == "http://example.com/ws_icon.jpg"
 149         assert md["domain_name"] == "test-slack-workspace.slack.com"
 150         assert md["installation_type"] == "migrated_to_bot"
 151         assert md["migrated_at"]
 152 
 153         oi = OrganizationIntegration.objects.get(
 154             integration=integration, organization=self.organization
 155         )
 156         assert oi.config == {}
 157 
 158         idp = IdentityProvider.objects.get(type="slack", external_id="TXXXXXXX1")
 159         identity = Identity.objects.get(idp=idp, user=self.user, external_id="UXXXXXXX1")
 160         assert identity.status == IdentityStatus.VALID
 161         run_post_migration.apply_async.assert_called_with(
 162             kwargs={
 163                 "integration_id": integration.id,
 164                 "organization_id": self.organization.id,
 165                 "user_id": self.user.id,
 166                 "private_channels": [{"name": "#general", "id": "XXXXX"}],
 167                 "missing_channels": [],
 168             }
 169         )
 170 
 171         audit_entry = AuditLogEntry.objects.get(event=AuditLogEntryEvent.INTEGRATION_UPGRADE)
