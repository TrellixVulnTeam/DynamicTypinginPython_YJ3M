  25     def assert_setup_flow(
  26         self,
  27         team_id="TXXXXXXX1",
  28         authorizing_user_id="UXXXXXXX1",
  29         expected_client_id="slack-client-id",
  30         expected_client_secret="slack-client-secret",
  31     ):
  32         responses.reset()
  33 
  34         resp = self.client.get(self.init_path)
  35         assert resp.status_code == 302
  36         redirect = urlparse(resp["Location"])
  37         assert redirect.scheme == "https"
  38         assert redirect.netloc == "slack.com"
  39         assert redirect.path == "/oauth/v2/authorize"
  40         params = parse_qs(redirect.query)
  41         scopes = self.provider.identity_oauth_scopes
  42         assert params["scope"] == [" ".join(scopes)]
  43         assert params["state"]
  44         assert params["redirect_uri"] == ["http://testserver/extensions/slack/setup/"]
  45         assert params["response_type"] == ["code"]
  46         assert params["client_id"] == [expected_client_id]
  47 
  48         assert params.get("user_scope") == ["links:read"]
  49         # once we've asserted on it, switch to a singular values to make life
  50         # easier
  51         authorize_params = {k: v[0] for k, v in six.iteritems(params)}
  52 
  53         access_json = {
  54             "ok": True,
  55             "access_token": "xoxb-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx",
  56             "team": {"id": team_id, "name": "Example"},
  57             "authed_user": {"id": authorizing_user_id},
  58         }
  59         responses.add(responses.POST, "https://slack.com/api/oauth.v2.access", json=access_json)
  60 
  61         responses.add(
  62             responses.GET,
  63             "https://slack.com/api/team.info",
  64             json={
  65                 "ok": True,
  66                 "team": {
  67                     "domain": "test-slack-workspace",
  68                     "icon": {"image_132": "http://example.com/ws_icon.jpg"},
  69                 },
  70             },
  71         )
  72 
  73         resp = self.client.get(
  74             u"{}?{}".format(
  75                 self.setup_path,
  76                 urlencode({"code": "oauth-code", "state": authorize_params["state"]}),
  77             )
  78         )
  79 
  80         mock_request = responses.calls[0].request
  81         req_params = parse_qs(mock_request.body)
  82         assert req_params["grant_type"] == ["authorization_code"]
  83         assert req_params["code"] == ["oauth-code"]
  84         assert req_params["redirect_uri"] == ["http://testserver/extensions/slack/setup/"]
  85         assert req_params["client_id"] == [expected_client_id]
  86         assert req_params["client_secret"] == [expected_client_secret]
  87 
  88         assert resp.status_code == 200
  89         self.assertDialogSuccess(resp)
