  59     def post_webhook(
  60         self,
  61         action_data=None,
  62         type="event_callback",
  63         data=None,
  64         token=None,
  65         team_id="TXXXXXXX1",
  66         callback_id=None,
  67         slack_user=None,
  68         original_message=None,
  69     ):
  70         if token is None:
  71             token = options.get("slack.verification-token")
  72 
  73         if slack_user is None:
  74             slack_user = {"id": self.identity.external_id, "domain": "example"}
  75 
  76         if callback_id is None:
  77             callback_id = json.dumps({"issue": self.group1.id})
  78 
  79         if original_message is None:
  80             original_message = {}
  81 
  82         payload = {
  83             "token": token,
  84             "team": {"id": team_id, "domain": "example.com"},
  85             "channel": {"id": "C065W1189", "domain": "forgotten-works"},
  86             "user": slack_user,
  87             "callback_id": callback_id,
  88             "action_ts": "1458170917.164398",
  89             "message_ts": "1458170866.000004",
  90             "original_message": original_message,
  91             "trigger_id": self.trigger_id,
  92             "response_url": self.response_url,
  93             "attachment_id": "1",
  94             "actions": action_data or [],
  95             "type": type,
  96         }
  97         if data:
  98             payload.update(data)
  99 
 100         payload = {"payload": json.dumps(payload)}
 101 
 102         return self.client.post("/extensions/slack/action/", data=payload)

 261     @responses.activate
 262     def test_resolve_issue(self):
 263         status_action = {"name": "resolve_dialog", "value": "resolve_dialog"}
 264 
 265         # Expect request to open dialog on slack
 266         responses.add(
 267             method=responses.POST,
 268             url="https://slack.com/api/dialog.open",
 269             body='{"ok": true}',
 270             status=200,
 271             content_type="application/json",
 272         )
 273 
 274         resp = self.post_webhook(action_data=[status_action])
 275         assert resp.status_code == 200, resp.content
 276 
 277         # Opening dialog should *not* cause the current message to be updated
 278         assert resp.content == b""
 279 
 280         data = parse_qs(responses.calls[0].request.body)
 281         assert data["token"][0] == self.integration.metadata["access_token"]
 282         assert data["trigger_id"][0] == self.trigger_id
 283         assert "dialog" in data
 284 
 285         dialog = json.loads(data["dialog"][0])
 286         callback_data = json.loads(dialog["callback_id"])
 287         assert int(callback_data["issue"]) == self.group1.id
 288         assert callback_data["orig_response_url"] == self.response_url
 289 
 290         # Completing the dialog will update the message
 291         responses.add(
 292             method=responses.POST,
 293             url=self.response_url,
 294             body='{"ok": true}',
 295             status=200,
 296             content_type="application/json",
 297         )
 298 
 299         resp = self.post_webhook(
 300             type="dialog_submission",
 301             callback_id=dialog["callback_id"],
 302             data={"submission": {"resolve_type": "resolved"}},
 303         )
 304         self.group1 = Group.objects.get(id=self.group1.id)
 305 
 306         assert resp.status_code == 200, resp.content
 307         assert self.group1.get_status() == GroupStatus.RESOLVED
 308 
 309         update_data = json.loads(responses.calls[1].request.body)
 310 
 311         expect_status = u"*Issue resolved by <@{}>*".format(self.identity.external_id)
 312         assert update_data["text"].endswith(expect_status)
