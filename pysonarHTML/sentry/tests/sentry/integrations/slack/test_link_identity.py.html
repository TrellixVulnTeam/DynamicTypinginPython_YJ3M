  37     @responses.activate

  40         unsign.return_value = {
  41             "integration_id": self.integration.id,
  42             "organization_id": self.org.id,
  43             "slack_id": "new-slack-id",
  44             "channel_id": "my-channel",
  45             "response_url": "http://example.slack.com/response_url",
  46         }
  47 
  48         linking_url = build_linking_url(
  49             self.integration,

  51             "new-slack-id",
  52             "my-channel",
  53             "http://example.slack.com/response_url",
  54         )
  55 
  56         resp = self.client.get(linking_url)
  57 
  58         assert resp.status_code == 200
  59         self.assertTemplateUsed(resp, "sentry/auth-link-identity.html")
  60 
  61         responses.add(
  62             method=responses.POST,
  63             url="http://example.slack.com/response_url",
  64             body='{"ok": true}',
  65             status=200,
  66             content_type="application/json",
  67         )
  68 
  69         # Link identity of user
  70         resp = self.client.post(linking_url)
  71 
  72         identity = Identity.objects.filter(external_id="new-slack-id", user=self.user1)

  74         assert len(identity) == 1

  79     @responses.activate

  95         }
  96 
  97         linking_url = build_linking_url(
  98             self.integration,
  99             self.org,
 100             "slack-id2",
 101             "my-channel",
 102             "http://example.slack.com/response_url",
 103         )
 104         responses.add(
 105             method=responses.POST,
 106             url="http://example.slack.com/response_url",
 107             body='{"ok": true}',
 108             status=200,
 109             content_type="application/json",
 110         )
 111 
 112         self.client.post(linking_url)
