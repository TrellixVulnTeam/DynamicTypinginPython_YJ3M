 113 class JiraWebhooksTest(APITestCase):
 114     @patch("sentry.integrations.jira.webhooks.sync_group_assignee_inbound")
 115     def test_simple_assign(self, mock_sync_group_assignee_inbound):
 116         org = self.organization
 117 
 118         integration = Integration.objects.create(provider="jira", name="Example Jira")
 119         integration.add_organization(org, self.user)
 120 
 121         path = reverse("sentry-extensions-jira-issue-updated")
 122 
 123         with patch(
 124             "sentry.integrations.jira.webhooks.get_integration_from_jwt", return_value=integration
 125         ):
 126             resp = self.client.post(
 127                 path,
 128                 data=json.loads(SAMPLE_EDIT_ISSUE_PAYLOAD_ASSIGNEE.strip()),
 129                 HTTP_AUTHORIZATION="JWT anexampletoken",
 130             )
 131             assert resp.status_code == 200
 132             mock_sync_group_assignee_inbound.assert_called_with(
 133                 integration, "jess@sentry.io", "APP-123", assign=True
 134             )

 136     @override_settings(JIRA_USE_EMAIL_SCOPE=True)
 137     @patch("sentry.integrations.jira.webhooks.sync_group_assignee_inbound")
 138     @responses.activate
 139     def test_assign_use_email_api(self, mock_sync_group_assignee_inbound):
 140         org = self.organization
 141 
 142         integration = Integration.objects.create(
 143             provider="jira",
 144             name="Example Jira",
 145             metadata={
 146                 "oauth_client_id": "oauth-client-id",
 147                 "shared_secret": "a-super-secret-key-from-atlassian",
 148                 "base_url": "https://example.atlassian.net",
 149                 "domain_name": "example.atlassian.net",
 150             },
 151         )
 152         integration.add_organization(org, self.user)
 153 
 154         path = reverse("sentry-extensions-jira-issue-updated")
 155 
 156         responses.add(
 157             responses.GET,
 158             "https://example.atlassian.net/rest/api/3/user/email",
 159             json={"accountId": "deadbeef123", "email": self.user.email},
 160             match_querystring=False,
 161         )
 162 
 163         with patch(
 164             "sentry.integrations.jira.webhooks.get_integration_from_jwt", return_value=integration
 165         ):
 166             data = json.loads(SAMPLE_EDIT_ISSUE_PAYLOAD_ASSIGNEE.strip())
 167             data["issue"]["fields"]["assignee"]["emailAddress"] = ""
 168             resp = self.client.post(path, data=data, HTTP_AUTHORIZATION="JWT anexampletoken")
 169             assert resp.status_code == 200
 170             assert mock_sync_group_assignee_inbound.called
 171             assert len(responses.calls) == 1
