  48 class JiraSearchEndpointTest(APITestCase):
  49     @fixture
  50     def integration(self):
  51         integration = Integration.objects.create(
  52             provider="jira",
  53             name="Jira Cloud",
  54             metadata={
  55                 "oauth_client_id": "oauth-client-id",
  56                 "shared_secret": "a-super-secret-key-from-atlassian",
  57                 "base_url": "https://example.atlassian.net",
  58                 "domain_name": "example.atlassian.net",
  59             },
  60         )
  61         integration.add_organization(self.organization, self.user)
  62         return integration
  63 
  64     @responses.activate
  65     def test_issue_search_text(self):
  66         responses.add(
  67             responses.GET,
  68             "https://example.atlassian.net/rest/api/2/search/",
  69             body=SAMPLE_SEARCH_RESPONSE,
  70             content_type="json",
  71             match_querystring=False,
  72         )
  73         org = self.organization
  74         self.login_as(self.user)
  75 
  76         path = reverse("sentry-extensions-jira-search", args=[org.slug, self.integration.id])
  77 
  78         resp = self.client.get("%s?field=externalIssue&query=test" % (path,))
  79         assert resp.status_code == 200
  80         assert resp.data == [{"label": "(HSP-1) this is a test issue summary", "value": "HSP-1"}]
  81 
  82     @responses.activate
  83     def test_issue_search_id(self):
  84         def responder(request):
  85             query = parse_qs(urlparse(request.url).query)
  86             assert 'id="hsp-1"' == query["jql"][0]
  87             return (200, {}, SAMPLE_SEARCH_RESPONSE)
  88 
  89         responses.add_callback(
  90             responses.GET,
  91             "https://example.atlassian.net/rest/api/2/search/",
  92             callback=responder,
  93             match_querystring=False,
  94         )
  95         org = self.organization
  96         self.login_as(self.user)
  97 
  98         path = reverse("sentry-extensions-jira-search", args=[org.slug, self.integration.id])
  99 
 100         # queries come through from the front end lowercased, so HSP-1 -> hsp-1
 101         resp = self.client.get("%s?field=externalIssue&query=hsp-1" % (path,))
 102         assert resp.status_code == 200
 103         assert resp.data == [{"label": "(HSP-1) this is a test issue summary", "value": "HSP-1"}]

 123     @responses.activate
 124     def test_assignee_search(self):
 125         responses.add(
 126             responses.GET,
 127             "https://example.atlassian.net/rest/api/2/project",
 128             json=[{"key": "HSP", "id": "10000"}],
 129             match_querystring=False,
 130         )
 131 
 132         def responder(request):
 133             query = parse_qs(urlparse(request.url).query)
 134             assert "HSP" == query["project"][0]
 135             assert "bob" == query["query"][0]
 136             return (200, {}, SAMPLE_USER_SEARCH_RESPONSE)
 137 
 138         responses.add_callback(
 139             responses.GET,
 140             "https://example.atlassian.net/rest/api/2/user/assignable/search",
 141             callback=responder,
 142             content_type="json",
 143             match_querystring=False,
 144         )
 145         org = self.organization
 146         self.login_as(self.user)
 147 
 148         path = reverse("sentry-extensions-jira-search", args=[org.slug, self.integration.id])
 149 
 150         resp = self.client.get("%s?project=10000&field=assignee&query=bob" % (path,))
 151         assert resp.status_code == 200
