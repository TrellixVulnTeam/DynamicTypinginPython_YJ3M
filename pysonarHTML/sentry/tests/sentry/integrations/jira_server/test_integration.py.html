 284     @responses.activate
 285     def test_setup_create_webhook(self):
 286         responses.add(
 287             responses.POST,
 288             "https://jira.example.com/plugins/servlet/oauth/request-token",
 289             status=200,
 290             content_type="text/plain",
 291             body="oauth_token=abc123&oauth_token_secret=def456",
 292         )
 293         responses.add(
 294             responses.POST,
 295             "https://jira.example.com/plugins/servlet/oauth/access-token",
 296             status=200,
 297             content_type="text/plain",
 298             body="oauth_token=valid-token&oauth_token_secret=valid-secret",
 299         )
 300 
 301         expected_id = "jira.example.com:sentry-bot"
 302 
 303         def webhook_response(request):
 304             # Ensure the webhook token contains our integration
 305             # external id
 306             data = json.loads(request.body)
 307             url = data["url"]
 308             token = url.split("/")[-2]
 309             token_data = jwt.decode(token, verify=False)
 310             assert "id" in token_data
 311             assert token_data["id"] == expected_id
 312 
 313             return (204, {}, "")
 314 
 315         responses.add_callback(
 316             responses.POST,
 317             "https://jira.example.com/rest/webhooks/1.0/webhook",
 318             callback=webhook_response,
 319         )
 320         self.install_integration()
 321 
 322         integration = Integration.objects.get()
 323         assert integration.external_id == expected_id
