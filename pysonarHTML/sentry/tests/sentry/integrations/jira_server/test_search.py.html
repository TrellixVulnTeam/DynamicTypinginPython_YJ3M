 102     @responses.activate
 103     def test_assignee_search(self):
 104         responses.add(
 105             responses.GET,
 106             "https://jira.example.org/rest/api/2/project",
 107             json=[{"key": "HSP", "id": "10000"}],
 108             match_querystring=False,
 109         )
 110 
 111         def responder(request):
 112             query = parse_qs(urlparse(request.url).query)
 113             assert "HSP" == query["project"][0]
 114             assert "bob" == query["username"][0]
 115             return (200, {}, EXAMPLE_USER_SEARCH_RESPONSE)
 116 
 117         responses.add_callback(
 118             responses.GET,
 119             "https://jira.example.org/rest/api/2/user/assignable/search",
 120             callback=responder,
 121             content_type="json",
 122             match_querystring=False,
 123         )
 124         org = self.organization
 125         self.login_as(self.user)
 126 
 127         path = reverse("sentry-extensions-jiraserver-search", args=[org.slug, self.integration.id])
 128 
 129         resp = self.client.get("%s?project=10000&field=assignee&query=bob" % (path,))
 130         assert resp.status_code == 200
 131         assert resp.data == [{"value": "bob", "label": "Bobby - bob@example.org (bob)"}]
