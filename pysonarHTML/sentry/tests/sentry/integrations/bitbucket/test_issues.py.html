  72     @responses.activate
  73     def test_after_link_issue(self):
  74         issue_id = 3
  75         repo = "myaccount/myrepo"
  76         comment = {"comment": "hello I'm a comment"}
  77         responses.add(
  78             responses.POST,
  79             u"https://api.bitbucket.org/2.0/repositories/{repo}/issues/{issue_id}/comments".format(
  80                 repo=repo, issue_id=issue_id
  81             ),
  82             status=201,
  83             json={"content": {"raw": comment}},
  84         )
  85 
  86         external_issue = ExternalIssue.objects.create(
  87             organization_id=self.organization.id,
  88             integration_id=self.integration.id,
  89             key="%s#%d" % (repo, issue_id),
  90         )
  91 
  92         self.integration.get_installation(external_issue.organization_id).after_link_issue(
  93             external_issue, data=comment
  94         )
  95 
  96         request = responses.calls[0].request
  97         assert responses.calls[0].response.status_code == 201
  98         payload = json.loads(request.body)
  99         assert payload == {"content": {"raw": comment["comment"]}}

 102     def test_default_repo_link_fields(self):
 103         responses.add(
 104             responses.GET,
 105             "https://api.bitbucket.org/2.0/repositories/myaccount",

 109                     {"full_name": "myaccount/repo2"}
 110                 ]
 111             }""",
 112             content_type="application/json",
 113         )
 114         self.org_integration.config = {
 115             "project_issue_defaults": {
 116                 six.text_type(self.group.project_id): {"repo": "myaccount/repo1"}
 117             }
 118         }
 119         self.org_integration.save()
 120         installation = self.integration.get_installation(self.organization.id)
 121         fields = installation.get_link_issue_config(self.group)
 122         repo_field = [field for field in fields if field["name"] == "repo"][0]
 123         assert repo_field["default"] == "myaccount/repo1"

 277     @responses.activate
 278     def test_create_issue(self):
 279         repo = "myaccount/repo1"
 280         id = "112"
 281         title = "hello"
 282         content = {"html": "This is the description"}
 283 
 284         responses.add(
 285             responses.POST,
 286             u"https://api.bitbucket.org/2.0/repositories/{repo}/issues".format(repo=repo),
 287             json={"id": id, "title": title, "content": {"html": content}},
 288         )
 289         installation = self.integration.get_installation(self.organization.id)
 290         result = installation.create_issue(
