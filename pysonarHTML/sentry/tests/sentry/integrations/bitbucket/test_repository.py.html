 139 class BitbucketCreateRepositoryTestCase(IntegrationRepositoryTestCase):
 140     provider_name = "integrations:bitbucket"
 141 
 142     def setUp(self):
 143         super(BitbucketCreateRepositoryTestCase, self).setUp()
 144         self.base_url = "https://api.bitbucket.org"
 145         self.shared_secret = "234567890"
 146         self.subject = "connect:1234567"
 147         self.integration = Integration.objects.create(
 148             provider="bitbucket",
 149             external_id=self.subject,
 150             name="MyBitBucket",
 151             metadata={
 152                 "base_url": self.base_url,
 153                 "shared_secret": self.shared_secret,
 154                 "subject": self.subject,
 155             },
 156         )
 157         self.integration.get_provider().setup()
 158         self.integration.add_organization(self.organization, self.user)
 159         self.repo = Repository.objects.create(
 160             provider="bitbucket",
 161             name="sentryuser/newsdiffs",
 162             organization_id=self.organization.id,
 163             config={"name": "sentryuser/newsdiffs"},
 164             integration_id=self.integration.id,
 165         )
 166         self.default_repository_config = {"full_name": "getsentry/example-repo", "id": "123"}
 167 
 168     def add_create_repository_responses(self, repository_config):
 169         responses.add(
 170             responses.GET,
 171             "%s/2.0/repositories/%s" % (self.base_url, self.repo.name),
 172             json=repository_config,
 173         )
 174         responses.add(
 175             responses.POST,
 176             u"%s/2.0/repositories/%s/hooks" % (self.base_url, self.repo.name),
 177             json={"uuid": "99"},
 178         )
 179 
 180     def test_create_repository_data_no_installation_id(self):
 181         response = self.create_repository(self.default_repository_config, None)
 182         assert response.status_code == 400
 183         self.assert_error_message(response, "validation", "requires an integration id")
 184 
 185     def test_create_repository_data_integration_does_not_exist(self):
 186         integration_id = self.integration.id
 187         self.integration.delete()
 188 
 189         response = self.create_repository(self.default_repository_config, integration_id)
 190         assert response.status_code == 404
 191         self.assert_error_message(
 192             response, "not found", "Integration matching query does not exist."
 193         )
