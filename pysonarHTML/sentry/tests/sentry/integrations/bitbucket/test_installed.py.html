  94     @responses.activate
  95     def test_plugin_migration(self):
  96         accessible_repo = Repository.objects.create(
  97             organization_id=self.organization.id,
  98             name="sentryuser/repo",
  99             url="https://bitbucket.org/sentryuser/repo",
 100             provider="bitbucket",
 101             external_id="123456",
 102             config={"name": "sentryuser/repo"},
 103         )
 104 
 105         inaccessible_repo = Repository.objects.create(
 106             organization_id=self.organization.id,
 107             name="otheruser/otherrepo",
 108             url="https://bitbucket.org/otheruser/otherrepo",
 109             provider="bitbucket",
 110             external_id="654321",
 111             config={"name": "otheruser/otherrepo"},
 112         )
 113 
 114         self.client.post(self.path, data=self.data_from_bitbucket)
 115 
 116         integration = Integration.objects.get(provider=self.provider, external_id=self.client_key)
 117 
 118         responses.add(
 119             responses.GET,
 120             u"https://api.bitbucket.org/2.0/repositories/{}/hooks".format(accessible_repo.name),
 121             json={"values": [{"description": "sentry-bitbucket-repo-hook"}]},
 122         )
 123 
 124         with self.tasks():
 125             BitbucketIntegrationProvider().post_install(integration, self.organization)
 126 
 127             assert Repository.objects.get(id=accessible_repo.id).integration_id == integration.id
 128 
 129             assert (
 130                 Repository.objects.get(id=accessible_repo.id).provider == "integrations:bitbucket"
 131             )
 132 
 133             assert Repository.objects.get(id=inaccessible_repo.id).integration_id is None

 135     @responses.activate
 136     def test_disable_plugin_when_fully_migrated(self):
 137         project = Project.objects.create(organization_id=self.organization.id)
 138 
 139         plugin = plugins.get("bitbucket")
 140         plugin.enable(project)
 141 
 142         # Accessible to new Integration
 143         Repository.objects.create(
 144             organization_id=self.organization.id,
 145             name="sentryuser/repo",
 146             url="https://bitbucket.org/sentryuser/repo",
 147             provider="bitbucket",
 148             external_id="123456",
 149             config={"name": "sentryuser/repo"},
 150         )
 151 
 152         self.client.post(self.path, data=self.data_from_bitbucket)
 153 
 154         integration = Integration.objects.get(provider=self.provider, external_id=self.client_key)
 155 
 156         responses.add(
 157             responses.GET,
 158             u"https://api.bitbucket.org/2.0/repositories/sentryuser/repo/hooks",
 159             json={"values": [{"description": "sentry-bitbucket-repo-hook"}]},
 160         )
 161 
 162         assert "bitbucket" in [p.slug for p in plugins.for_project(project)]
 163 
 164         with self.tasks():
 165             BitbucketIntegrationProvider().post_install(integration, self.organization)
 166 
 167             assert "bitbucket" not in [p.slug for p in plugins.for_project(project)]

 169     def test_installed_without_public_key(self):
 170         integration = Integration.objects.get_or_create(
 171             provider=self.provider,
 172             external_id=self.client_key,
 173             defaults={"name": self.username, "metadata": self.metadata},
 174         )[0]
 175 
 176         response = self.client.post(self.path, data=self.data_from_bitbucket)
 177         assert response.status_code == 200
 178 
 179         # assert no changes have been made to the integration
 180         integration_after = Integration.objects.get(
 181             provider=self.provider, external_id=self.client_key
 182         )
 183         assert integration.name == integration_after.name
 184         assert integration.metadata == integration_after.metadata
