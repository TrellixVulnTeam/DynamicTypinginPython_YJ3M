  11 class BitbucketSearchEndpointTest(APITestCase):
  12     def setUp(self):
  13         self.base_url = "https://api.bitbucket.org"
  14         self.shared_secret = "234567890"
  15         self.subject = "connect:1234567"
  16         self.integration = Integration.objects.create(
  17             provider="bitbucket",
  18             external_id=self.subject,
  19             name="meredithanya",
  20             metadata={
  21                 "base_url": self.base_url,
  22                 "shared_secret": self.shared_secret,
  23                 "subject": self.subject,
  24             },
  25         )
  26 
  27         self.login_as(self.user)
  28         self.integration.add_organization(self.organization, self.user)
  29         self.path = reverse(
  30             "sentry-extensions-bitbucket-search", args=[self.organization.slug, self.integration.id]
  31         )
  32 
  33     @responses.activate
  34     def test_search_issues(self):
  35         responses.add(
  36             responses.GET,
  37             "https://api.bitbucket.org/2.0/repositories/meredithanya/apples/issues",
  38             json={
  39                 "values": [
  40                     {"id": "123", "title": "Issue Title 123"},
  41                     {"id": "456", "title": "Issue Title 456"},
  42                 ]
  43             },
  44         )
  45         resp = self.client.get(
  46             self.path,
  47             data={"field": "externalIssue", "query": "issue", "repo": "meredithanya/apples"},
  48         )
  49 
  50         assert resp.status_code == 200
  51         assert resp.data == [
  52             {"label": "#123 Issue Title 123", "value": "123"},
  53             {"label": "#456 Issue Title 456", "value": "456"},
  54         ]
  55 
  56     @responses.activate
  57     def test_search_repositories(self):
  58         responses.add(
  59             responses.GET,
  60             "https://api.bitbucket.org/2.0/repositories/meredithanya",
  61             json={"values": [{"full_name": "meredithanya/apples"}]},
  62         )
  63         resp = self.client.get(self.path, data={"field": "repo", "query": "apple"})
  64 
  65         assert resp.status_code == 200
  66         assert resp.data == [{"label": "meredithanya/apples", "value": "meredithanya/apples"}]
  67 
  68     @responses.activate
  69     def test_search_repositories_no_issue_tracker(self):
  70         responses.add(
  71             responses.GET,
  72             "https://api.bitbucket.org/2.0/repositories/meredithanya/apples/issues",
  73             json={"type": "error", "error": {"message": "Repository has no issue tracker."}},
  74             status=404,
  75         )
  76         resp = self.client.get(
  77             self.path,
  78             data={"field": "externalIssue", "query": "issue", "repo": "meredithanya/apples"},
  79         )
  80         assert resp.status_code == 400
  81         assert resp.data == {"detail": "Bitbucket Repository has no issue tracker."}
