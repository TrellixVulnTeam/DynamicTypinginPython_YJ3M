 102 class IssueDefaultTest(TestCase):
 103     def setUp(self):
 104         self.group.status = GroupStatus.RESOLVED
 105         self.group.save()
 106 
 107         integration = Integration.objects.create(provider="example", external_id="123456")
 108         integration.add_organization(self.group.organization, self.user)
 109 
 110         self.external_issue = ExternalIssue.objects.create(
 111             organization_id=self.group.organization.id, integration_id=integration.id, key="APP-123"
 112         )
 113 
 114         self.group_link = GroupLink.objects.create(
 115             group_id=self.group.id,
 116             project_id=self.group.project_id,
 117             linked_type=GroupLink.LinkedType.issue,
 118             linked_id=self.external_issue.id,
 119             relationship=GroupLink.Relationship.references,
 120         )
 121 
 122         self.installation = integration.get_installation(self.group.organization.id)
 123 
 124     def test_get_repository_choices(self):
 125         default_repo, repo_choice = self.installation.get_repository_choices(self.group)
 126         assert default_repo == "user/repo"
 127         assert repo_choice == [("user/repo", "repo")]
 128 
 129     def test_get_repository_choices_no_repos(self):
 130         self.installation.get_repositories = lambda: []
 131         default_repo, repo_choice = self.installation.get_repository_choices(self.group)
 132         assert default_repo == ""
 133         assert repo_choice == []
 134 
 135     def test_get_repository_choices_default_repo(self):
 136         self.installation.org_integration.config = {
 137             "project_issue_defaults": {six.text_type(self.group.project_id): {"repo": "user/repo2"}}
 138         }
 139         self.installation.org_integration.save()
 140         self.installation.get_repositories = lambda: [
 141             {"name": "repo1", "identifier": "user/repo1"},
 142             {"name": "repo2", "identifier": "user/repo2"},
 143         ]
 144         default_repo, repo_choice = self.installation.get_repository_choices(self.group)
 145         assert default_repo == "user/repo2"
 146         assert repo_choice == [("user/repo1", "repo1"), ("user/repo2", "repo2")]
 147 
 148     def test_annotations(self):
 149         label = self.installation.get_issue_display_name(self.external_issue)
 150         link = self.installation.get_issue_url(self.external_issue.key)
 151 
 152         assert self.installation.get_annotations_for_group_list([self.group]) == {
 153             self.group.id: ['<a href="%s">%s</a>' % (link, label)]
 154         }
 155 
 156         integration = Integration.objects.create(provider="example", external_id="4444")
 157         integration.add_organization(self.group.organization, self.user)
 158         installation = integration.get_installation(self.group.organization.id)
 159 
 160         assert installation.get_annotations_for_group_list([self.group]) == {self.group.id: []}
