  18         self.user = self.create_user(is_superuser=False)

  28     def post_webhook(self, data, signature=UNSET, variant=UNSET, key=None):
  29         if key is None:
  30             key = options.get("cloudflare.secret-key")
  31         if not isinstance(data, six.string_types):
  32             body = json.dumps(data)
  33         else:
  34             body = data
  35         if signature is UNSET:
  36             signature = hmac.new(
  37                 key=key.encode("utf-8"), msg=body.encode("utf-8"), digestmod=sha256
  38             ).hexdigest()
  39         if variant is UNSET:
  40             variant = "1"
  41 
  42         headers = {
  43             "HTTP_X_SIGNATURE_HMAC_SHA256_HEX": signature,
  44             "HTTP_X_SIGNATURE_KEY_VARIANT": variant,
  45         }
  46 
  47         return self.client.post(
  48             "/extensions/cloudflare/webhook/", body, content_type="application/json", **headers
  49         )
