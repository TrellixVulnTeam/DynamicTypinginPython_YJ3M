 148     @responses.activate
 149     def test_update_organization_config(self):
 150         with self.tasks():
 151             self.assert_setup_flow()
 152         integration = Integration.objects.get(provider=self.provider.key)
 153         service_id = PagerDutyService.objects.get(integration_key="key1").id
 154         config_data = {
 155             "service_table": [
 156                 {"service": "Mleep", "integration_key": "xxxxxxxxxxxxxxxx", "id": None},
 157                 {"service": "new_service", "integration_key": "new_key", "id": service_id},
 158             ]
 159         }
 160         integration.get_installation(self.organization).update_organization_config(config_data)
 161         assert len(PagerDutyService.objects.filter()) == 2
 162         service_row = PagerDutyService.objects.get(id=service_id)
 163         assert service_row.service_name == "new_service"
 164         assert service_row.integration_key == "new_key"
