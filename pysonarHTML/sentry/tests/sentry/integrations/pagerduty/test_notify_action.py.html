  22 class PagerDutyNotifyActionTest(RuleTestCase):
  23     rule_cls = PagerDutyNotifyServiceAction

  25     def setUp(self):
  26         self.integration = Integration.objects.create(
  27             provider="pagerduty",
  28             name="Example",
  29             external_id=EXTERNAL_ID,
  30             metadata={"services": SERVICES},
  31         )
  32         self.integration.add_organization(self.organization, self.user)
  33         self.service = PagerDutyService.objects.create(
  34             service_name=SERVICES[0]["service_name"],
  35             integration_key=SERVICES[0]["integration_key"],
  36             organization_integration=self.integration.organizationintegration_set.first(),
  37         )
  38         self.installation = self.integration.get_installation(self.organization.id)

  40     @responses.activate
  41     def test_applies_correctly(self):
  42         event = self.get_event()
  43 
  44         rule = self.get_rule(data={"account": self.integration.id, "service": self.service.id})
  45 
  46         results = list(rule.after(event=event, state=self.get_state()))
  47         assert len(results) == 1
  48 
  49         responses.add(
  50             method=responses.POST,
  51             url="https://events.pagerduty.com/v2/enqueue/",
  52             body={},
  53             status=202,
  54             content_type="application/json",
  55         )
  56 
  57         # Trigger rule callback
  58         results[0].callback(event, futures=[])
  59         data = json.loads(responses.calls[0].request.body)
  60 
  61         assert data["event_action"] == "trigger"

  79     def test_valid_service_options(self):
  80         # create new org that has the same pd account but different a service added
  81         new_org = self.create_organization(name="New Org", owner=self.user)
  82 
  83         # need to create a new project and set it as self.project because rules are
  84         # project based and we want the project in this case to be associated with the
  85         # new organization when we call `self.get_rule()`
  86         new_project = self.create_project(name="new proj", organization=new_org)
  87         self.project = new_project
  88 
  89         self.integration.add_organization(new_org, self.user)
  90         oi = OrganizationIntegration.objects.get(organization=new_org)
  91         new_service = PagerDutyService.objects.create(
  92             service_name="New Service",
  93             integration_key="new_service_key",
  94             organization_integration=oi,
  95         )
  96 
  97         rule = self.get_rule(data={"account": self.integration.id})
  98 
  99         service_options = rule.get_services()
 100         assert service_options == [(new_service.id, new_service.service_name)]
