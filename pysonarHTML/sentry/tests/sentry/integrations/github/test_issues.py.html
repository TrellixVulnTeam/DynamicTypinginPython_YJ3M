 159     @responses.activate
 160     @patch("sentry.integrations.github.client.get_jwt", return_value="jwt_token_1")
 161     def test_repo_dropdown_choices(self, mock_get_jwt):
 162         event = self.store_event(
 163             data={"event_id": "a" * 32, "timestamp": self.min_ago}, project_id=self.project.id
 164         )
 165 
 166         responses.add(
 167             responses.POST,
 168             "https://api.github.com/installations/github_external_id/access_tokens",
 169             json={"token": "token_1", "expires_at": "2018-10-11T22:14:10Z"},
 170         )
 171 
 172         responses.add(
 173             responses.GET,
 174             "https://api.github.com/repos/getsentry/sentry/assignees",
 175             json=[{"login": "MeredithAnya"}],
 176         )
 177 
 178         responses.add(
 179             responses.GET,
 180             "https://api.github.com/installation/repositories",
 181             json={
 182                 "repositories": [
 183                     {"full_name": "getsentry/sentry", "name": "sentry"},
 184                     {"full_name": "getsentry/other", "name": "other", "archived": True},
 185                 ]
 186             },
 187         )
 188 
 189         resp = self.integration.get_create_issue_config(group=event.group)
 190         assert resp[0]["choices"] == [(u"getsentry/sentry", u"sentry")]
 191 
 192         responses.add(
 193             responses.GET,
 194             "https://api.github.com/repos/getsentry/hello/assignees",
 195             json=[{"login": "MeredithAnya"}],
 196         )
 197 
 198         # create an issue
 199         data = {"params": {"repo": "getsentry/hello"}}
 200         resp = self.integration.get_create_issue_config(group=event.group, **data)
 201         assert resp[0]["choices"] == [
 202             (u"getsentry/hello", u"hello"),
 203             (u"getsentry/sentry", u"sentry"),
 204         ]
 205         # link an issue
 206         data = {"params": {"repo": "getsentry/hello"}}
 207         resp = self.integration.get_link_issue_config(group=event.group, **data)
 208         assert resp[0]["choices"] == [
 209             (u"getsentry/hello", u"hello"),
 210             (u"getsentry/sentry", u"sentry"),
 211         ]

 245         "sentry.integrations.github.client.GitHubClientMixin.get_token", return_value="jwt_token_1"
 246     )
 247     def test_default_repo_link_fields(self, mock_get_jwt):
 248         responses.add(
 249             responses.GET,
 250             "https://api.github.com/installation/repositories",
 251             json={"repositories": [{"name": "sentry", "full_name": "getsentry/sentry"}]},
 252         )
 253         event = self.store_event(
 254             data={"event_id": "a" * 32, "timestamp": self.min_ago}, project_id=self.project.id
 255         )
 256         group = event.group
 257 
 258         org_integration = self.integration.org_integration
 259         org_integration.config = {
 260             "project_issue_defaults": {
 261                 six.text_type(group.project_id): {"repo": "getsentry/sentry"}
 262             }
 263         }
 264         org_integration.save()
 265         fields = self.integration.get_link_issue_config(group)
 266         for field in fields:
 267             if field["name"] == "repo":
 268                 repo_field = field
 269                 break
 270         assert repo_field["default"] == "getsentry/sentry"
