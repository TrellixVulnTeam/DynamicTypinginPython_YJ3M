 173     @mock.patch("sentry.integrations.github.client.get_jwt", return_value="jwt_token_1")
 174     @responses.activate
 175     def test_compare_commits_force_refresh(self, get_jwt):
 176         stub_installation_token()
 177         ten_hours = datetime.datetime.utcnow() + datetime.timedelta(hours=10)
 178         self.integration.metadata = {
 179             "access_token": "old-access-token",
 180             "expires_at": ten_hours.replace(microsecond=0).isoformat(),
 181         }
 182         self.integration.save()
 183         responses.add(
 184             responses.GET,
 185             "https://api.github.com/repos/getsentry/example-repo/compare/xyz123...abcdef",
 186             status=404,
 187             body="GitHub returned a 404 Not Found error.",
 188         )
 189         responses.add(
 190             responses.GET,
 191             "https://api.github.com/repos/getsentry/example-repo/compare/xyz123...abcdef",
 192             json=json.loads(COMPARE_COMMITS_EXAMPLE),
 193         )
 194         responses.add(
 195             responses.GET,
 196             "https://api.github.com/repos/getsentry/example-repo/commits/6dcb09b5b57875f334f61aebed695e2e4193db5e",
 197             json=json.loads(GET_COMMIT_EXAMPLE),
 198         )
 199 
 200         result = self.provider.compare_commits(self.repository, "xyz123", "abcdef")
 201         for commit in result:
 202             assert_commit_shape(commit)
 203 
 204         # assert token was refreshed
 205         assert (
 206             Integration.objects.get(id=self.integration.id).metadata["access_token"]
 207             == "v1.install-token"
 208         )
 209 
 210         # compare_commits gives 400, token was refreshed, and compare_commits gives 200
 211         assert (
 212             responses.calls[0].response.url
 213             == u"https://api.github.com/repos/getsentry/example-repo/compare/xyz123...abcdef"
 214         )
 215         assert responses.calls[0].response.status_code == 404
 216         assert (
 217             responses.calls[1].response.url
 218             == u"https://api.github.com/installations/654321/access_tokens"
 219         )
 220         assert responses.calls[1].response.status_code == 200
 221         assert (
 222             responses.calls[2].response.url
 223             == u"https://api.github.com/repos/getsentry/example-repo/compare/xyz123...abcdef"
 224         )
 225         assert responses.calls[2].response.status_code == 200
