  98     def test_plugin_migration(self):
  99         accessible_repo = Repository.objects.create(
 100             organization_id=self.organization.id,
 101             name="Test-Organization/foo",
 102             url="https://github.com/Test-Organization/foo",
 103             provider="github",
 104             external_id=123,
 105             config={"name": "Test-Organization/foo"},
 106         )
 107 
 108         inaccessible_repo = Repository.objects.create(
 109             organization_id=self.organization.id,
 110             name="Not-My-Org/other",
 111             provider="github",
 112             external_id=321,
 113             config={"name": "Not-My-Org/other"},
 114         )
 115 
 116         with self.tasks():
 117             self.assert_setup_flow()
 118 
 119         integration = Integration.objects.get(provider=self.provider.key)
 120 
 121         # Updates the existing Repository to belong to the new Integration
 122         assert Repository.objects.get(id=accessible_repo.id).integration_id == integration.id
 123 
 124         # Doesn't touch Repositories not accessible by the new Integration
 125         assert Repository.objects.get(id=inaccessible_repo.id).integration_id is None

 239     @responses.activate
 240     def test_get_message_from_error(self):
 241         self.assert_setup_flow()
 242         integration = Integration.objects.get(provider=self.provider.key)
 243         installation = integration.get_installation(self.organization)
 244         base_error = "Error Communicating with GitHub (HTTP 404): %s" % (API_ERRORS[404])
 245         assert (
 246             installation.message_from_error(
 247                 ApiError("Not Found", code=404, url="https://api.github.com/repos/scefali")
 248             )
 249             == base_error
 250         )
 251         url = "https://api.github.com/repos/scefali/sentry-integration-example/compare/2adcab794f6f57efa8aa84de68a724e728395792...e208ee2d71e8426522f95efbdae8630fa66499ab"
 252         assert (
 253             installation.message_from_error(ApiError("Not Found", code=404, url=url))
 254             == base_error
 255             + " Please also confirm that the commits associated with the following URL have been pushed to GitHub: %s"
 256             % url
