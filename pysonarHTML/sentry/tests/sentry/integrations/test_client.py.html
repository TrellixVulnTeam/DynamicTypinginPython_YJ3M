 118 class OAuth2ApiClientTest(TestCase):
 119     def setUp(self):
 120         self.user = self.create_user()
 121         self.organization = self.create_organization()
 122         self.access_token = "1234567890"
 123         self.identity_provider_model = IdentityProvider.objects.create(type="oauth")
 124         register(OAuthProvider)
 125 
 126     @responses.activate
 127     def test_check_auth(self):
 128         new_auth = {
 129             "access_token": "1234567890",
 130             "refresh_token": "0987654321",
 131             "expires_in": 45678988239,
 132         }
 133         responses.add(responses.POST, "https://example.com", json=new_auth)
 134         identity = Identity.objects.create(
 135             idp=self.identity_provider_model,
 136             user=self.user,
 137             external_id="oauth_base",
 138             data={
 139                 "access_token": "access_token",
 140                 "refresh_token": "refresh_token",
 141                 "expires": int(time()) - 3600,
 142             },
 143         )
 144 
 145         client = OAuth2ApiClient(identity)
 146         client.check_auth()
 147 
 148         assert client.identity.data["access_token"] == new_auth["access_token"]
 149         assert client.identity.data["refresh_token"] == new_auth["refresh_token"]
 150         assert client.identity.data["expires"] > int(time())
 151 
 152     @responses.activate
 153     def test_check_auth_no_refresh(self):
 154         new_auth = {
