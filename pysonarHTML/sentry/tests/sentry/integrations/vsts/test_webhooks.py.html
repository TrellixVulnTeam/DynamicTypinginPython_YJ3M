  85     def set_workitem_state(self, old_value, new_value):
  86         work_item = dict(WORK_ITEM_UPDATED_STATUS)
  87         state = work_item["resource"]["fields"]["System.State"]
  88 
  89         if old_value is None:
  90             del state["oldValue"]
  91         else:
  92             state["oldValue"] = old_value
  93         state["newValue"] = new_value
  94 
  95         return work_item

 144     @responses.activate
 145     def test_inbound_status_sync_resolve(self):
 146         responses.add(
 147             responses.GET,
 148             "https://instance.visualstudio.com/c0bf429a-c03c-4a99-9336-d45be74db5a6/_apis/wit/workitemtypes/Bug/states",
 149             json=WORK_ITEM_STATES,
 150         )
 151         work_item_id = 33
 152         num_groups = 5
 153         external_issue = ExternalIssue.objects.create(
 154             organization_id=self.organization.id, integration_id=self.model.id, key=work_item_id
 155         )
 156         groups = [
 157             self.create_linked_group(external_issue, self.project, GroupStatus.UNRESOLVED)
 158             for _ in range(num_groups)
 159         ]
 160 
 161         # Change so that state is changing from unresolved to resolved
 162         work_item = self.set_workitem_state("Active", "Resolved")
 163 
 164         with self.feature("organizations:integrations-issue-sync"):
 165             resp = self.client.post(
 166                 absolute_uri("/extensions/vsts/issue-updated/"),
 167                 data=work_item,
 168                 HTTP_SHARED_SECRET=self.shared_secret,
 169             )
 170             assert resp.status_code == 200
 171             group_ids = [g.id for g in groups]
 172             assert (
 173                 len(Group.objects.filter(id__in=group_ids, status=GroupStatus.RESOLVED))
 174                 == num_groups
 175             )
 176             assert len(Activity.objects.filter(group_id__in=group_ids)) == num_groups

 212     @responses.activate
 213     def test_inbound_status_sync_new_workitem(self):
 214         responses.add(
 215             responses.GET,
 216             "https://instance.visualstudio.com/c0bf429a-c03c-4a99-9336-d45be74db5a6/_apis/wit/workitemtypes/Bug/states",
 217             json=WORK_ITEM_STATES,
 218         )
 219         work_item_id = 33
 220         external_issue = ExternalIssue.objects.create(
 221             organization_id=self.organization.id, integration_id=self.model.id, key=work_item_id
 222         )
 223 
 224         group = self.create_linked_group(external_issue, self.project, GroupStatus.UNRESOLVED)
 225 
 226         # Change so that it is a new workitem
 227         work_item = self.set_workitem_state(None, "New")
 228         assert "oldValue" not in work_item["resource"]["fields"]["System.State"]
 229 
 230         with self.feature("organizations:integrations-issue-sync"):
 231             resp = self.client.post(
 232                 absolute_uri("/extensions/vsts/issue-updated/"),
 233                 data=work_item,
 234                 HTTP_SHARED_SECRET=self.shared_secret,
 235             )
 236             assert resp.status_code == 200
 237             assert Group.objects.get(id=group.id).status == GroupStatus.UNRESOLVED
 238             # no change happened. no activity should be created here
 239             assert len(Activity.objects.filter(group_id=group.id)) == 0
