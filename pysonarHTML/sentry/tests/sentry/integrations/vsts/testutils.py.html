 156     def make_init_request(self, path=None, body=None):
 157         return self.client.get(path or self.init_path, body or {})
 158 
 159     def make_oauth_redirect_request(self, state):
 160         return self.client.get(
 161             u"{}?{}".format(self.setup_path, urlencode({"code": "oauth-code", "state": state}))
 162         )
 163 
 164     def assert_vsts_oauth_redirect(self, redirect):
 165         assert redirect.scheme == "https"
 166         assert redirect.netloc == "app.vssps.visualstudio.com"
 167         assert redirect.path == "/oauth2/authorize"
 168 
 169     def assert_account_selection(self, response, account_id=None):
 170         account_id = account_id or self.vsts_account_id
 171         assert response.status_code == 200
 172         assert u'<option value="{}"'.format(account_id).encode("utf-8") in response.content
 173 
 174     def assert_installation(self):
 175         # Initial request to the installation URL for VSTS
 176         resp = self.make_init_request()
 177         redirect = urlparse(resp["Location"])
 178 
 179         assert resp.status_code == 302
 180         self.assert_vsts_oauth_redirect(redirect)
 181 
 182         query = parse_qs(redirect.query)
 183 
 184         # OAuth redirect back to Sentry (identity_pipeline_view)
 185         resp = self.make_oauth_redirect_request(query["state"][0])
 186         self.assert_account_selection(resp)
 187 
 188         # User choosing which VSTS Account to use (AccountConfigView)
 189         # Final step.
 190         return self.client.post(
 191             self.setup_path, {"account": self.vsts_account_id, "provider": "vsts"}
 192         )
