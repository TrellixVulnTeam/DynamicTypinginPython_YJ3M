  16     def test_exchange_token(self):

  18             return "https://app.vssps.visualstudio.com/oauth2/authorize"

  21             access_token_url="https://app.vssps.visualstudio.com/oauth2/token",
  22             client_id="vsts-client-id",
  23             client_secret="vsts-client-secret",
  24         )
  25         request = Mock()
  26         pipeline = Mock()
  27 
  28         pipeline.redirect_url = redirect_url
  29 
  30         responses.add(
  31             responses.POST,
  32             "https://app.vssps.visualstudio.com/oauth2/token",

  34                 "access_token": "xxxxxxxxx",
  35                 "token_type": "jwt-bearer",
  36                 "expires_in": "3599",
  37                 "refresh_token": "zzzzzzzzzz",
  38             },
  39         )
  40

  44

  49             "urn:ietf:params:oauth:client-assertion-type:jwt-bearer"
  50         ]
  51         assert req_params["client_assertion"] == ["vsts-client-secret"]
  52 
  53         assert result["access_token"] == "xxxxxxxxx"
  54         assert result["token_type"] == "jwt-bearer"

 167 class VstsIdentityProviderTest(TestCase):
 168     def setUp(self):
 169         self.identity_provider_model = IdentityProvider.objects.create(type="vsts")
 170         self.identity = Identity.objects.create(
 171             idp=self.identity_provider_model,
 172             user=self.user,
 173             external_id="vsts_id",
 174             data={
 175                 "access_token": "123456789",
 176                 "token_type": "token_type",
 177                 "expires": 12345678,
 178                 "refresh_token": "n354678",
 179             },
 180         )
 181         self.provider = VSTSIdentityProvider()
 182         self.client_secret = "12345678"
 183         self.provider.get_oauth_client_secret = lambda: self.client_secret
 184 
 185     def get_refresh_token_params(self):
 186         refresh_token = "wertyui"
 187         params = self.provider.get_refresh_token_params(refresh_token)
 188         assert params == {
 189             "client_assertion_type": "urn:ietf:params:oauth:client-assertion-type:jwt-bearer",
 190             "client_assertion": self.client_secret,
 191             "grant_type": "refresh_token",
 192             "assertion": refresh_token,
 193             "redirect_uri": absolute_uri(self.provider.oauth_redirect_url),
 194         }
 195 
 196     @responses.activate
 197     def test_refresh_identity(self):
 198         refresh_data = {
 199             "access_token": "access token for this user",
 200             "token_type": "type of token",
 201             "expires": 1234567,
 202             "refresh_token": "new refresh token to use when the token has timed out",
 203         }
 204         responses.add(
 205             responses.POST, "https://app.vssps.visualstudio.com/oauth2/token", json=refresh_data
 206         )
 207         self.provider.refresh_identity(self.identity, redirect_url="redirect_url")
 208 
 209         assert len(responses.calls) == 1
 210 
 211         new_identity = Identity.objects.get(id=self.identity.id)
 212         assert new_identity.data["access_token"] == refresh_data["access_token"]
 213         assert new_identity.data["token_type"] == refresh_data["token_type"]
 214         assert new_identity.data["expires"] <= int(time())
