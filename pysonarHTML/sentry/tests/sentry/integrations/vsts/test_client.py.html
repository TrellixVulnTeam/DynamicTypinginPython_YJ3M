  31 
  32         # Second to last request, before the Projects request, was to refresh

  35 
  36         # Then we request the Projects with the new token
  37         assert responses.calls[-1].request.url == u"{}_apis/projects?stateFilter=WellFormed".format(
  38             self.vsts_base_url.lower()
  39         )
  40 
  41         identity = Identity.objects.get(id=identity.id)
