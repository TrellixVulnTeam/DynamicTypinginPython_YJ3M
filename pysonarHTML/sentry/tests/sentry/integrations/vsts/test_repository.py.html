 184     def assert_repository(self, repository_config, organization_id=None):
 185         repo = Repository.objects.get(
 186             organization_id=organization_id or self.organization.id,
 187             provider=self.provider_name,
 188             external_id=repository_config["id"],
 189         )
 190         assert repo.name == repository_config["name"]
 191         assert repo.url == repository_config["_links"]["web"]["href"]
 192         assert repo.integration_id == self.integration.id
 193         assert repo.config == {
 194             "instance": self.base_url,
 195             "project": repository_config["project"]["name"],
 196             "name": repository_config["name"],
 197         }
