  11 class GitlabRefreshAuthTest(GitLabTestCase):

  14     def setUp(self):
  15         super(GitlabRefreshAuthTest, self).setUp()
  16         self.client = self.installation.get_client()
  17         self.request_data = {"id": "user_id"}
  18         self.request_url = "https://example.gitlab.com/api/v4/user"
  19         self.refresh_url = "https://example.gitlab.com/oauth/token"
  20         self.refresh_response = {
  21             "access_token": "123432sfh29uhs29347",
  22             "token_type": "bearer",
  23             "refresh_token": "29f43sdfsk22fsj929",
  24             "created_at": 1536798907,
  25             "scope": "api",
  26         }
  27         self.original_identity_data = dict(self.client.identity.data)

  29     def tearDown(self):
  30         responses.reset()

  32     def make_users_request(self):
  33         return self.client.get_user()
  34 
  35     def add_refresh_auth(self, success=True):
  36         responses.add(
  37             responses.POST,
  38             self.refresh_url,
  39             status=200 if success else 401,
  40             json=self.refresh_response if success else {},
  41         )
  42 
  43     def add_get_user_response(self, success):
  44         responses.add(
  45             responses.GET,
  46             self.request_url,
  47             json=self.request_data if success else {},
  48             status=200 if success else 401,
  49         )

  51     def assert_response_call(self, call, url, status):
  52         assert call.request.url == url
  53         assert call.response.status_code == status
  54 
  55     def assert_data(self, data, expected_data):
  56         assert data["access_token"] == expected_data["access_token"]
  57         assert data["refresh_token"] == expected_data["refresh_token"]
  58         assert data["created_at"] == expected_data["created_at"]
  59 
  60     def assert_request_failed_refresh(self):
  61         responses_calls = responses.calls
  62         assert len(responses_calls) == 2
  63 
  64         self.assert_response_call(responses_calls[0], self.request_url, 401)
  65         self.assert_response_call(responses_calls[1], self.refresh_url, 401)
  66 
  67     def assert_request_with_refresh(self):
  68         responses_calls = responses.calls
  69         assert len(responses_calls) == 3
  70 
  71         self.assert_response_call(responses_calls[0], self.request_url, 401)
  72         self.assert_response_call(responses_calls[1], self.refresh_url, 200)
  73         self.assert_response_call(responses_calls[2], self.request_url, 200)
  74 
  75         assert json.loads(responses_calls[2].response.text) == self.request_data
  76 
  77     def assert_identity_was_refreshed(self):
  78         data = self.client.identity.data
  79         self.assert_data(data, self.refresh_response)
  80 
  81         data = Identity.objects.get(id=self.client.identity.id).data
  82         self.assert_data(data, self.refresh_response)
  83 
  84     def assert_identity_was_not_refreshed(self):
  85         data = self.client.identity.data

  87 
  88         data = Identity.objects.get(id=self.client.identity.id).data
  89         self.assert_data(data, self.original_identity_data)

  91     @responses.activate
  92     def test_refresh_auth_flow(self):
  93         # Fail first then succeed
  94         self.add_get_user_response(success=False)
  95         self.add_get_user_response(success=True)
  96 
  97         self.add_refresh_auth(success=True)
  98 
  99         resp = self.make_users_request()
 100         self.assert_request_with_refresh()
 101         assert resp == self.request_data
 102         self.assert_identity_was_refreshed()
