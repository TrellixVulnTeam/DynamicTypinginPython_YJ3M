 141         project_id = "10"

 143         key = "%s#%s" % (project_name, issue_iid)

 149                 "iid": issue_iid,
 150                 "title": "hello",
 151                 "description": "This is the description",
 152                 "web_url": "https://example.gitlab.com/%s/issues/%s" % (project_name, issue_iid),
 153             },
 154         )
 155         responses.add(
 156             responses.GET,
 157             u"https://example.gitlab.com/api/v4/projects/%s" % project_id,
 158             json={"path_with_namespace": project_name, "id": 10},
 159         )
 160         form_data = {
 161             "project": project_id,
 162             "title": "hello",
 163             "description": "This is the description",
 164         }
 165 
 166         assert self.installation.create_issue(form_data) == {
 167             "key": key,
 168             "description": "This is the description",
 169             "title": "hello",
 170             "url": "https://example.gitlab.com/%s/issues/%s" % (project_name, issue_iid),
 171             "project": project_id,
 172             "metadata": {"display_name": key},
 173         }

 394     def test_after_link_issue_required_fields(self):

 396         external_issue = ExternalIssue.objects.create(
