  40     def assert_setup_flow(self, user_id="user_id_1"):
  41         resp = self.client.get(self.init_path)
  42         assert resp.status_code == 200
  43         self.assertContains(resp, "you will need to create a Sentry app in your GitLab instance")
  44         resp = self.client.get(self.init_path_without_guide)
  45         assert resp.status_code == 200
  46         resp = self.client.post(self.init_path_without_guide, data=self.config)
  47         assert resp.status_code == 302
  48         redirect = urlparse(resp["Location"])
  49         assert redirect.scheme == "https"
  50         assert redirect.netloc == "gitlab.example.com"
  51         assert redirect.path == "/oauth/authorize"
  52 
  53         params = parse_qs(redirect.query)
  54         assert params["state"]
  55         assert params["redirect_uri"] == ["http://testserver/extensions/gitlab/setup/"]
  56         assert params["response_type"] == ["code"]
  57         assert params["client_id"] == ["client_id"]
  58         # once we've asserted on it, switch to a singular values to make life
  59         # easier
  60         authorize_params = {k: v[0] for k, v in six.iteritems(params)}
  61 
  62         access_token = "xxxxx-xxxxxxxxx-xxxxxxxxxx-xxxxxxxxxxxx"
  63 
  64         responses.add(
  65             responses.POST,
  66             "https://gitlab.example.com/oauth/token",
  67             json={"access_token": access_token},
  68         )
  69         responses.add(responses.GET, "https://gitlab.example.com/api/v4/user", json={"id": user_id})
  70         responses.add(
  71             responses.GET,
  72             "https://gitlab.example.com/api/v4/groups/cool-group",
  73             json={
  74                 "id": self.default_group_id,
  75                 "full_name": "Cool",
  76                 "full_path": "cool-group",
  77                 "web_url": "https://gitlab.example.com/groups/cool-group",
  78                 "avatar_url": "https://gitlab.example.com/uploads/group/avatar/4/foo.jpg",
  79             },
  80         )
  81         responses.add(
  82             responses.POST, "https://gitlab.example.com/api/v4/hooks", json={"id": "webhook-id-1"}
  83         )
  84 
  85         resp = self.client.get(
  86             u"{}?{}".format(
  87                 self.setup_path,
  88                 urlencode({"code": "oauth-code", "state": authorize_params["state"]}),
  89             )
  90         )
  91 
  92         mock_access_token_request = responses.calls[0].request
  93         req_params = parse_qs(mock_access_token_request.body)
  94         assert req_params["grant_type"] == ["authorization_code"]
  95         assert req_params["code"] == ["oauth-code"]

  97         assert req_params["client_id"] == ["client_id"]
  98         assert req_params["client_secret"] == ["client_secret"]
  99

 140     def test_goback_to_instructions(self):
 141         # Go to instructions
 142         resp = self.client.get(self.init_path)
 143         assert resp.status_code == 200
 144         self.assertContains(resp, "Step 1")
 145 
 146         # Go to setup form
 147         resp = self.client.get(self.init_path_without_guide)
 148         assert resp.status_code == 200
 149         self.assertContains(resp, "Step 2")

 153         assert resp.status_code == 200

 157     def test_setup_missing_group(self):

 159         assert resp.status_code == 200
 160

 163

 166         assert redirect.netloc == "gitlab.example.com"
 167         assert redirect.path == "/oauth/authorize"
 168 
 169         params = parse_qs(redirect.query)
 170         authorize_params = {k: v[0] for k, v in six.iteritems(params)}
 171 
 172         responses.add(
