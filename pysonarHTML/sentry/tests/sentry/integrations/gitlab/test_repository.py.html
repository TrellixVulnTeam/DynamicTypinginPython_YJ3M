  74     def assert_repository(self, repository_config, organization_id=None):
  75         instance = self.integration.metadata["instance"]
  76 
  77         external_id = u"{}:{}".format(instance, repository_config["id"])
  78         repo = Repository.objects.get(
  79             organization_id=organization_id or self.organization.id,
  80             provider=self.provider_name,
  81             external_id=external_id,
  82         )
  83         assert repo.name == repository_config["name_with_namespace"]
  84         assert repo.url == repository_config["web_url"]
  85         assert repo.integration_id == self.integration.id
  86         assert repo.config == {
  87             "instance": instance,
  88             "path": repository_config["path_with_namespace"],

  90             "webhook_id": 99,

  99     @responses.activate
 100     def test_create_repository_verify_payload(self):
 101         def request_callback(request):
 102             payload = json.loads(request.body)
 103             assert "url" in payload
 104             assert payload["push_events"]
 105             assert payload["merge_requests_events"]
 106             expected_token = u"{}:{}".format(
 107                 self.integration.external_id, self.integration.metadata["webhook_secret"]
 108             )
 109             assert payload["token"] == expected_token
 110 
 111             return (201, {}, json.dumps({"id": 99}))
 112 
 113         responses.add_callback(
 114             responses.POST,
 115             u"https://example.gitlab.com/api/v4/projects/%s/hooks" % self.gitlab_id,
 116             callback=request_callback,
 117         )
 118         response = self.create_repository(self.default_repository_config, self.integration.id)
 119         assert response.status_code == 201
 120         self.assert_repository(self.default_repository_config)
