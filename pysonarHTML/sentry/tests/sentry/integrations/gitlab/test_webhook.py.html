 120         # Create a repo on the primary organization
 121         repo = self.create_repo("getsentry/sentry")
 122

 124         other_org = self.create_organization(owner=self.user)

 136

 214     def test_merge_event_no_last_commit(self):
 215         payload = json.loads(MERGE_REQUEST_OPENED_EVENT)
 216 
 217         # Remove required keys. There have been events in prod that are missing
 218         # these important attributes. GitLab docs don't explain why though.
 219         del payload["object_attributes"]["last_commit"]
 220 
 221         response = self.client.post(
 222             self.url,
 223             data=json.dumps(payload),
 224             content_type="application/json",
 225             HTTP_X_GITLAB_TOKEN=WEBHOOK_TOKEN,
 226             HTTP_X_GITLAB_EVENT="Merge Request Hook",
 227         )
 228         assert response.status_code == 204
 229         assert 0 == PullRequest.objects.count()

 249     def test_merge_event_update_pull_request(self):
 250         repo = self.create_repo("getsentry/sentry")
 251         group = self.create_group(project=self.project, short_id=9)
 252         PullRequest.objects.create(
 253             organization_id=self.organization.id,
 254             repository_id=repo.id,
 255             key=1,
 256             title="Old title",
 257             message="Old message",
 258         )
 259 
 260         response = self.client.post(
 261             self.url,
 262             data=MERGE_REQUEST_OPENED_EVENT,
 263             content_type="application/json",
 264             HTTP_X_GITLAB_TOKEN=WEBHOOK_TOKEN,
 265             HTTP_X_GITLAB_EVENT="Merge Request Hook",
 266         )
 267         assert response.status_code == 204
 268         author = CommitAuthor.objects.all().first()
 269         self.assert_commit_author(author)
 270 
 271         pull = PullRequest.objects.all().first()
 272         assert pull.title != "Old title"
 273         assert pull.message != "Old message"
 274 
 275         self.assert_pull_request(pull, author)
 276         self.assert_group_link(group, pull)
