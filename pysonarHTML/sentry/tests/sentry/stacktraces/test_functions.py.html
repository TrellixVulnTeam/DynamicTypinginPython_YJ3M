 139 def test_trim_function_name_cocoa():
 140     assert trim_function_name("+[foo:(bar)]", "objc") == "+[foo:(bar)]"
 141     assert trim_function_name("[foo:(bar)]", "objc") == "[foo:(bar)]"
 142     assert trim_function_name("-[foo:(bar)]", "objc") == "-[foo:(bar)]"
 143     assert (
 144         trim_function_name("(anonymous namespace)::foo(int)", "native")
 145         == "(anonymous namespace)::foo"
 146     )
 147     assert trim_function_name("foo::bar::foo(int)", "native") == "foo::bar::foo"
