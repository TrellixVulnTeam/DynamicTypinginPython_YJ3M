  30             responses.POST, "https://api.vercel.com/v2/oauth/access_token", json=access_json

  34             responses.GET,

  36             json={"user": {"name": "my_user_name"}},

  40             responses.GET, "https://api.vercel.com/v4/projects/", json={"projects": []},

  55     @responses.activate

  57         self.login_as(self.user)
