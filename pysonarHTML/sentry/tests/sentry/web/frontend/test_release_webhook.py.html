  15 class ReleaseWebhookTest(TestCase):

  18         self.organization = self.create_organization()

  20         self.project = self.create_project(teams=[self.team])

  65     @patch("sentry.plugins.base.plugins.get")
  66     def test_valid_signature(self, mock_plugin_get):
