   8 class ReactPageViewTest(TestCase):
   9     def test_redirects_unauthenticated_request(self):
  10         owner = self.create_user("bar@example.com")
  11         org = self.create_organization(owner=owner)
  12 
  13         path = reverse("sentry-organization-home", args=[org.slug])
  14         resp = self.client.get(path)
  15 
  16         self.assertRedirects(resp, reverse("sentry-auth-organization", args=[org.slug]))
  17         assert resp["X-Robots-Tag"] == "noindex, nofollow"
  18 
  19     def test_superuser_can_load(self):
  20         org = self.create_organization(owner=self.user)
  21         path = reverse("sentry-organization-home", args=[org.slug])
  22 
  23         self.login_as(self.user)
  24 
  25         resp = self.client.get(path)
  26 
  27         assert resp.status_code == 200
  28         self.assertTemplateUsed(resp, "sentry/bases/react.html")

  31     def test_redirects_user_to_auth_without_membership(self):
  32         owner = self.create_user("bar@example.com")
  33         org = self.create_organization(owner=owner)
  34         non_member = self.create_user("foo@example.com")
  35 
  36         path = reverse("sentry-organization-home", args=[org.slug])
  37 
  38         self.login_as(non_member)
  39 
  40         resp = self.client.get(path)
  41 
  42         self.assertRedirects(resp, reverse("sentry-auth-organization", args=[org.slug]))
  43 
  44         # ensure we dont redirect to auth if its not a valid org
  45         path = reverse("sentry-organization-home", args=["foobar"])
  46 
  47         resp = self.client.get(path)
  48 
  49         assert resp.status_code == 302
  50         assert resp["Location"] != reverse("sentry-auth-organization", args=[org.slug])
  51 
  52         # ensure we dont redirect with valid membership
  53         path = reverse("sentry-organization-home", args=[org.slug])
  54 
  55         self.login_as(owner)
  56 
  57         resp = self.client.get(path)
  58 
  59         assert resp.status_code == 200
  60         self.assertTemplateUsed(resp, "sentry/bases/react.html")
