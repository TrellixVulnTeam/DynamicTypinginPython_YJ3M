   1 from __future__ import absolute_import
   2 
   3 from django.core.urlresolvers import reverse
   4 
   5 from sentry.testutils import TestCase
   6 from sentry.models import OrganizationMember
   7 
   8 
   9 class VstsExtensionConfigurationTest(TestCase):
  10     @property
  11     def path(self):
  12         return reverse("vsts-extension-configuration")
  13 
  14     def setUp(self):
  15         self.user = self.create_user()
  16         self.org = self.create_organization()
  17 
  18         OrganizationMember.objects.create(user=self.user, organization=self.org)
  19 
  20     def test_logged_in_one_org(self):
  21         self.login_as(self.user)
  22 
  23         resp = self.client.get(self.path, {"targetId": "1", "targetName": "foo"})
  24 
  25         # Goes straight to VSTS OAuth
  26         assert resp.status_code == 302
  27         assert resp.url.startswith("https://app.vssps.visualstudio.com/oauth2/authorize")
  28 
  29     def test_logged_in_many_orgs(self):
  30         self.login_as(self.user)
  31 
  32         org = self.create_organization()
  33         OrganizationMember.objects.create(user=self.user, organization=org)
  34 
  35         resp = self.client.get(self.path, {"targetId": "1", "targetName": "foo"})
  36 
  37         assert resp.status_code == 302
  38         assert "/extensions/vsts/link/" in resp.url
  39 
  40     def test_choose_org(self):
  41         self.login_as(self.user)
  42 
  43         resp = self.client.post(
  44             self.path, {"vsts_id": "1", "vsts_name": "foo", "organization": self.org.slug}
  45         )
  46 
  47         assert resp.status_code == 302
  48         assert resp.url.startswith("https://app.vssps.visualstudio.com/oauth2/authorize")
  49 
  50     def test_logged_out(self):
  51         resp = self.client.get(self.path, {"targetId": "1", "targetName": "foo"})
  52 
  53         assert resp.status_code == 302
  54         assert "/auth/login/" in resp.url
  55         # URL encoded post-login redirect URL
  56         assert (
  57             "next=%2Fextensions%2Fvsts%2Fconfigure%2F%3FtargetName%3Dfoo%26targetId%3D1" in resp.url
  58         )
