  11 class JavaScriptSdkLoaderTest(TestCase):

  20     def test_noop_no_pub_key(self):
  21         resp = self.client.get(reverse("sentry-js-sdk-loader", args=["abc"]))
  22         assert resp.status_code == 200
  23         self.assertTemplateUsed(resp, "sentry/js-sdk-loader-noop.js.tmpl")

  25     def test_noop(self):
  26         settings.JS_SDK_LOADER_DEFAULT_SDK_URL = ""
  27         resp = self.client.get(reverse("sentry-js-sdk-loader", args=[self.projectkey.public_key]))
  28         assert resp.status_code == 200
  29         self.assertTemplateUsed(resp, "sentry/js-sdk-loader-noop.js.tmpl")

  31     def test_no_replace(self):
  32         settings.JS_SDK_LOADER_SDK_VERSION = "0.5.2"
  33         settings.JS_SDK_LOADER_DEFAULT_SDK_URL = (
  34             "https://s3.amazonaws.com/getsentry-cdn/@sentry/browser/0.0.0/bundle.min.js"
  35         )
  36         resp = self.client.get(reverse("sentry-js-sdk-loader", args=[self.projectkey.public_key]))
  37         assert resp.status_code == 200
  38         assert settings.JS_SDK_LOADER_DEFAULT_SDK_URL.encode("utf-8") in resp.content
  39         self.assertTemplateUsed(resp, "sentry/js-sdk-loader.js.tmpl")
  40 
  41     def test_renders_js_loader(self):
  42         resp = self.client.get(self.path)
  43         assert resp.status_code == 200
  44         self.assertTemplateUsed(resp, "sentry/js-sdk-loader.js.tmpl")
  45         assert self.projectkey.public_key.encode("utf-8") in resp.content
  46         assert b"bundle.min.js" in resp.content
  47 
  48     def test_minified(self):
  49         resp = self.client.get(self.path)
  50         assert resp.status_code == 200
  51         min_resp = self.client.get(
  52             reverse("sentry-js-sdk-loader", args=[self.projectkey.public_key, ".min"])
  53         )
  54         assert min_resp.status_code == 200
  55         self.assertTemplateUsed(min_resp, "sentry/js-sdk-loader.min.js.tmpl")
  56         assert self.projectkey.public_key.encode("utf-8") in min_resp.content
  57         assert b"bundle.min.js" in min_resp.content
  58         assert len(resp.content) > len(min_resp.content)
