 118     def test_valid_params(self):
 119         self.login_as(self.user)
 120 
 121         resp = self.client.post(
 122             self.path,
 123             {
 124                 "grant_type": "authorization_code",
 125                 "redirect_uri": self.application.get_default_redirect_uri(),
 126                 "code": self.grant.code,
 127                 "client_id": self.application.client_id,
 128             },
 129         )
 130 
 131         assert resp.status_code == 200
 132         data = json.loads(resp.content)
 133 
 134         token = ApiToken.objects.get(token=data["access_token"])
 135         assert token.application == self.application
 136         assert token.user == self.grant.user
 137         assert token.get_scopes() == self.grant.get_scopes()
 138 
 139         assert data["access_token"] == token.token
 140         assert data["refresh_token"] == token.refresh_token
 141         assert isinstance(data["expires_in"], int)
 142         assert data["token_type"] == "bearer"
 143         assert data["user"]["id"] == six.text_type(token.user_id)
