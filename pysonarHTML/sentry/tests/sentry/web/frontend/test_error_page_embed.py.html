  93     def test_xss(self):
  94         user_feedback_options = {}
  95 
  96         option_keys = [
  97             "errorFormEntry",
  98             "successMessage",
  99             "errorGeneric",
 100             "title",
 101             "subtitle",
 102             "subtitle2",
 103             "labelName",
 104             "labelEmail",
 105             "labelComments",
 106             "labelSubmit",
 107             "labelClose",
 108         ]
 109         for key in option_keys:
 110             user_feedback_options[key] = "<img src=x onerror=alert({0})>XSS_{0}".format(key).encode(
 111                 "utf-8"
 112             )
 113 
 114         user_feedback_options_qs = urlencode(user_feedback_options)
 115         path_with_qs = "%s?eventId=%s&dsn=%s&%s" % (
 116             self.path,
 117             quote(self.event_id),
 118             quote(self.key.dsn_public),
 119             user_feedback_options_qs,
 120         )
 121         resp = self.client.get(
 122             path_with_qs,
 123             HTTP_REFERER="http://example.com",
 124             HTTP_ACCEPT="text/html, text/javascript",
 125         )
 126         assert resp.status_code == 200, resp.content
 127         self.assertTemplateUsed(resp, "sentry/error-page-embed.html")
 128 
 129         for xss_payload in user_feedback_options.values():
 130             assert xss_payload not in resp.content
