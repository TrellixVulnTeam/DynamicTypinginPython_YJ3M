  56     def setUp(self):
  57         self.user = self.create_user("rick@onehundredyears.com")
  58         self.org = self.create_organization(owner=self.user, name="saml2-org")
  59 
  60         # enable require 2FA and enroll user
  61         TotpInterface().enroll(self.user)
  62         self.org.update(flags=models.F("flags").bitor(Organization.flags.require_2fa))
  63         assert self.org.flags.require_2fa.is_set
  64 
  65         self.auth_provider = AuthProvider.objects.create(
  66             provider=self.provider_name, config=dummy_provider_config, organization=self.org
  67         )
  68 
  69         # The system.url-prefix, which is used to generate absolute URLs, must
  70         # have a TLD for the SAML2 library to consider the URL generated for
  71         # the ACS endpoint valid.
  72         self.url_prefix = settings.SENTRY_OPTIONS.get("system.url-prefix")
  73 
  74         settings.SENTRY_OPTIONS.update({"system.url-prefix": "http://testserver.com"})
  75 
  76         super(AuthSAML2Test, self).setUp()

  79         # restore url-prefix config

 130     @mock.patch("sentry.auth.helper.logger")
 131     def test_auth_setup(self, auth_log):
 132         self.auth_provider.delete()
 133         self.login_as(self.user)
 134 
 135         data = {"init": True, "provider": self.provider_name}
 136 
 137         with Feature(["organizations:sso-basic", "organizations:sso-saml2"]):
 138             setup = self.client.post(self.setup_path, data)
 139 
 140         assert setup.status_code == 302
 141         redirect = urlparse(setup.get("Location", ""))
 142         assert redirect.path == "/sso_url"
 143 
 144         auth = self.accept_auth(follow=True)
 145 
 146         messages = map(lambda m: six.text_type(m), auth.context["messages"])
 147 
 148         assert len(messages) == 2
 149         assert messages[0] == "You have successfully linked your account to your SSO provider."
 150         assert messages[1].startswith("SSO has been configured for your organization")
 151 
 152         # require 2FA disabled when saml is enabled
 153         org = Organization.objects.get(id=self.org.id)
 154         assert not org.flags.require_2fa.is_set
 155 
 156         event = AuditLogEntry.objects.get(
 157             target_object=org.id, event=AuditLogEntryEvent.ORG_EDIT, actor=self.user
 158         )
 159         assert "require_2fa to False when enabling SSO" in event.get_note()
 160         auth_log.info.assert_called_once_with(
 161             "Require 2fa disabled during sso setup", extra={"organization_id": self.org.id}
 162         )
