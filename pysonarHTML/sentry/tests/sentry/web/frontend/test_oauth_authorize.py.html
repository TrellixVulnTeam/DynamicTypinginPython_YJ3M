 316

 318             u"{}?response_type=token&client_id={}".format(self.path, self.application.client_id)

 320 
 321         assert resp.status_code == 200
 322         self.assertTemplateUsed("sentry/oauth-authorize.html")
 323         assert resp.context["application"] == self.application
 324 
 325         resp = self.client.post(self.path, {"op": "approve"})
 326 
 327         assert not ApiGrant.objects.filter(user=self.user).exists()

 333

 335         location, fragment = resp["Location"].split("#", 1)
 336         assert location == "https://example.com"
 337         fragment = parse_qs(fragment)
 338         assert fragment["access_token"] == [token.token]
 339         assert fragment["token_type"] == ["bearer"]
 340         assert "refresh_token" not in fragment
 341         assert fragment["expires_in"]
 342         assert fragment["token_type"] == ["bearer"]
