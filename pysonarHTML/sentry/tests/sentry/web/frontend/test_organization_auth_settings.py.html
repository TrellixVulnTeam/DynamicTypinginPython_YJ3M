  99 class OrganizationAuthSettingsTest(AuthProviderTestCase):
 100     def enroll_user_and_require_2fa(self, user, organization):
 101         TotpInterface().enroll(user)
 102         organization.update(flags=models.F("flags").bitor(Organization.flags.require_2fa))
 103         assert organization.flags.require_2fa.is_set
 104 
 105     def assert_require_2fa_disabled(self, user, organization, logger):
 106         organization = Organization.objects.get(id=organization.id)
 107         assert not organization.flags.require_2fa.is_set
 108 
 109         event = AuditLogEntry.objects.get(
 110             target_object=organization.id, event=AuditLogEntryEvent.ORG_EDIT, actor=user
 111         )
 112         assert "require_2fa to False when enabling SSO" in event.get_note()
 113         logger.info.assert_called_once_with(
 114             "Require 2fa disabled during sso setup", extra={"organization_id": organization.id}
 115         )

 117     def assert_basic_flow(self, user, organization, expect_error=False):
 118         configure_path = reverse(
 119             "sentry-organization-auth-provider-settings", args=[organization.slug]
 120         )
 121 
 122         with self.feature("organizations:sso-basic"):
 123             resp = self.client.post(configure_path, {"provider": "dummy", "init": True})
 124             assert resp.status_code == 200
 125             assert self.provider.TEMPLATE in resp.content.decode("utf-8")
 126 
 127             path = reverse("sentry-auth-sso")
 128             resp = self.client.post(path, {"email": user.email})
 129 
 130         settings_path = reverse("sentry-organization-auth-settings", args=[organization.slug])
 131 
 132         if expect_error:
 133             self.assertRedirects(resp, settings_path)
 134             return
 135         else:
 136             self.assertRedirects(resp, configure_path)
 137 
 138         auth_provider = AuthProvider.objects.get(organization=organization, provider="dummy")
 139         auth_identity = AuthIdentity.objects.get(auth_provider=auth_provider)
 140         assert user == auth_identity.user
 141 
 142         member = OrganizationMember.objects.get(organization=organization, user=user)
 143 
 144         assert getattr(member.flags, "sso:linked")
 145         assert not getattr(member.flags, "sso:invalid")
