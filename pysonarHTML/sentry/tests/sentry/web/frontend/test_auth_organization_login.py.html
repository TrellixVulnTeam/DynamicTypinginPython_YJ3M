 413     def test_flow_duplicate_users_with_membership_and_verified(self):
 414         """
 415         Given an existing authenticated user, and an updated identity (e.g.
 416         the ident changed from the SSO provider), we should be re-linking
 417         the identity automatically (without prompt) assuming the user is
 418         a member of the org.
 419 
 420         This only works when the email is mapped to an identical identity.
 421         """
 422         auth_provider = AuthProvider.objects.create(
 423             organization=self.organization, provider="dummy"
 424         )
 425 
 426         # setup a 'previous' identity, such as when we migrated Google from
 427         # the old idents to the new
 428         user = self.create_user("bar@example.com", is_managed=True, is_active=False)
 429         auth_identity = AuthIdentity.objects.create(
 430             auth_provider=auth_provider, user=user, ident="bar@example.com"
 431         )
 432 
 433         # they must be a member for the auto merge to happen
 434         self.create_member(organization=self.organization, user=user)
 435 
 436         # user needs to be logged in
 437         self.login_as(user)
 438 
 439         resp = self.client.post(self.path, {"init": True})
 440 
 441         assert resp.status_code == 200
 442         assert self.provider.TEMPLATE in resp.content.decode("utf-8")
 443 
 444         path = reverse("sentry-auth-sso")
 445 
 446         # we're suggesting the identity changed (as if the Google ident was
 447         # updated to be something else)
 448         resp = self.client.post(
 449             path, {"email": "bar@example.com", "id": "123", "email_verified": "1"}, follow=True
 450         )
 451         assert resp.redirect_chain == [
 452             (reverse("sentry-login"), 302),
 453             ("/organizations/foo/issues/", 302),
 454             ("/auth/login/foo/", 302),
 455         ]
 456         # there should be no prompt as we auto merge the identity

 458         auth_identity = AuthIdentity.objects.get(id=auth_identity.id)

 461 
 462         new_user = auth_identity.user

 465         member = OrganizationMember.objects.get(organization=self.organization, user=new_user)
