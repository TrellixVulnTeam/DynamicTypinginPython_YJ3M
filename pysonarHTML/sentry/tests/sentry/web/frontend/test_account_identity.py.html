  12 class AccountIdentityTest(TestCase):

  18     def test_associate_identity(self):

  21         IdentityProvider.objects.create(type="dummy", external_id="1234", config={})
  22 
  23         self.login_as(user)
  24 
  25         path = reverse(
  26             "sentry-account-associate-identity", args=[organization.slug, "dummy", "1234"]
  27         )
  28         resp = self.client.get(path)
  29 
  30         assert resp.status_code == 200
  31         assert resp.context["organization"] == organization
  32         assert isinstance(resp.context["provider"], DummyProvider)
  33 
  34         resp = self.client.post(path)
  35 
  36         assert resp.status_code == 200
  37         assert resp.content == DummyProvider.TEMPLATE
  38 
  39         resp = self.client.post(path, data={"email": "rick@example.com"})
  40         ident = Identity.objects.get(user=user)
  41 
  42         assert resp.status_code == 302
  43         assert ident.external_id == "rick@example.com"
  44         assert ident.status == IdentityStatus.VALID
