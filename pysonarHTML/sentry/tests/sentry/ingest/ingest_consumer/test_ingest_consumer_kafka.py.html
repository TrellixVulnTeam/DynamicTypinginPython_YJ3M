  50                 },
  51             }

  72 @pytest.mark.django_db(transaction=True)
  73 @pytest.mark.parametrize(
  74     "inline_transactions", [True, False], ids=["inline_transactions", "worker_transactions"]
  75 )
  76 def test_ingest_consumer_reads_from_topic_and_calls_celery_task(
  77     task_runner,
  78     kafka_producer,
  79     kafka_admin,
  80     requires_kafka,
  81     default_project,
  82     get_test_message,
  83     inline_transactions,
  84 ):
  85     group_id = "test-consumer"
  86     topic_event_name = ConsumerType.get_topic_name(ConsumerType.Events)
  87 
  88     admin = kafka_admin(settings)
  89     admin.delete_topic(topic_event_name)
  90     producer = kafka_producer(settings)
  91 
  92     event_ids = set()
  93     for _ in range(3):
  94         message, event_id = get_test_message()
  95         producer.produce(topic_event_name, message)
  96         event_ids.add(event_id)
  97 
  98     consumer = get_ingest_consumer(
  99         max_batch_size=2,
 100         max_batch_time=5000,
 101         group_id=group_id,
 102         consumer_types=set([ConsumerType.Events]),
 103         auto_offset_reset="earliest",
 104     )
 105 
 106     options.set("store.transactions-celery", not inline_transactions)
 107     with task_runner():
 108         i = 0
 109         while i < MAX_POLL_ITERATIONS:
 110             if eventstore.get_event_by_id(default_project.id, event_id):
 111                 break
 112 
 113             consumer._run_once()
 114             i += 1
 115 
 116     # check that we got the messages
 117     for event_id in event_ids:
 118         message = eventstore.get_event_by_id(default_project.id, event_id)
 119         assert message is not None
 120         assert message.data["event_id"] == event_id
 121         if message.data["type"] == "transaction":
 122             assert message.data["spans"] == []
 123             assert message.data["contexts"]["trace"]
 124         else:
 125             assert message.data["extra"]["the_id"] == event_id
