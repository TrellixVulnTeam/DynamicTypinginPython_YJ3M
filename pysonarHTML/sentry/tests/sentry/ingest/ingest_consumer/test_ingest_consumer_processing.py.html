 136 @pytest.mark.django_db
 137 @pytest.mark.parametrize(
 138     "event_attachments", [True, False], ids=["with_feature", "without_feature"]
 139 )
 140 @pytest.mark.parametrize(
 141     "chunks", [(b"Hello ", b"World!"), (b"",), ()], ids=["basic", "zerolen", "nochunks"]
 142 )
 143 @pytest.mark.parametrize("with_group", [True, False], ids=["with_group", "without_group"])
 144 def test_individual_attachments(
 145     default_project, factories, monkeypatch, event_attachments, chunks, with_group
 146 ):
 147     monkeypatch.setattr("sentry.features.has", lambda *a, **kw: event_attachments)
 148 
 149     event_id = "515539018c9b4260a6f999572f1661ee"
 150     attachment_id = "ca90fb45-6dd9-40a0-a18f-8693aa621abb"
 151     project_id = default_project.id
 152     group_id = None
 153 
 154     if with_group:
 155         event = factories.store_event(
 156             data={"event_id": event_id, "message": "existence is pain"}, project_id=project_id
 157         )
 158 
 159         group_id = event.group.id
 160         assert group_id, "this test requires a group to work"
 161 
 162     for i, chunk in enumerate(chunks):
 163         process_attachment_chunk(
 164             {
 165                 "payload": chunk,
 166                 "event_id": event_id,
 167                 "project_id": project_id,
 168                 "id": attachment_id,
 169                 "chunk_index": i,
 170             },
 171             projects={default_project.id: default_project},
 172         )
 173 
 174     process_individual_attachment(
 175         {
 176             "type": "attachment",
 177             "attachment": {
 178                 "attachment_type": "event.attachment",
 179                 "chunks": len(chunks),
 180                 "content_type": "application/octet-stream",
 181                 "id": attachment_id,
 182                 "name": "foo.txt",
 183             },
 184             "event_id": event_id,
 185             "project_id": project_id,
 186         },
 187         projects={default_project.id: default_project},
 188     )
 189 
 190     attachments = list(
 191         EventAttachment.objects.filter(project_id=project_id, event_id=event_id).select_related(
 192             "file"
 193         )
 194     )
 195 
 196     if not event_attachments:
 197         assert not attachments
 198     else:
 199         (attachment,) = attachments
 200         assert attachment.file.type == "event.attachment"
 201         assert attachment.file.headers == {"Content-Type": "application/octet-stream"}
 202         assert attachment.group_id == group_id
 203         file = attachment.file.getfile()
 204         assert file.read() == b"".join(chunks)
 205         assert file.name == "foo.txt"

 210     """

 252 @pytest.mark.django_db
 253 def test_userreport_reverse_order(default_project, monkeypatch):
 254     """
 255     Test that ingesting a userreport before the event works. This is relevant
 256     for unreal crashes where the userreport is processed immediately in the
 257     ingest consumer while the rest of the event goes to processing tasks.
 258     """
 259     event_id = uuid.uuid4().hex
 260     start_time = time.time() - 3600
 261 
 262     assert process_userreport(
 263         {
 264             "type": "user_report",
 265             "start_time": start_time,
 266             "payload": json.dumps(
 267                 {
 268                     "name": "Hans Gans",
 269                     "event_id": event_id,
 270                     "comments": "hello world",
 271                     "email": "markus+dontatme@sentry.io",
 272                 }
 273             ),
 274             "project_id": default_project.id,
 275         },
 276         projects={default_project.id: default_project},
 277     )
 278 
 279     mgr = EventManager(data={"event_id": event_id, "user": {"email": "markus+dontatme@sentry.io"}})
 280 
 281     mgr.normalize()
 282     mgr.save(default_project.id)
 283 
 284     (report,) = UserReport.objects.all()
 285     assert report.comments == "hello world"
 286 
 287     (evtuser,) = EventUser.objects.all()
 288     # Event got saved after user report, and the sync only works in the
 289     # opposite direction. That's fine, we just accept it.
 290     assert evtuser.name is None
