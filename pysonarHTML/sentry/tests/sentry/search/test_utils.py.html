 488 class GetLatestReleaseTest(TestCase):
 489     def test(self):
 490         with pytest.raises(Release.DoesNotExist):
 491             # no releases exist period
 492             environment = None
 493             get_latest_release([self.project], environment)
 494 
 495         old = Release.objects.create(organization_id=self.project.organization_id, version="old")
 496         old.add_project(self.project)
 497 
 498         new_date = old.date_added + timedelta(minutes=1)
 499         new = Release.objects.create(
 500             version="new-but-in-environment",
 501             organization_id=self.project.organization_id,
 502             date_released=new_date,
 503         )
 504         new.add_project(self.project)
 505         ReleaseEnvironment.get_or_create(
 506             project=self.project, release=new, environment=self.environment, datetime=new_date
 507         )
 508         ReleaseProjectEnvironment.get_or_create(
 509             project=self.project, release=new, environment=self.environment, datetime=new_date
 510         )
 511 
 512         newest = Release.objects.create(
 513             version="newest-overall",
 514             organization_id=self.project.organization_id,
 515             date_released=old.date_added + timedelta(minutes=5),
 516         )
 517         newest.add_project(self.project)
 518 
 519         # latest overall (no environment filter)
 520         environment = None
 521         result = get_latest_release([self.project], environment)
 522         assert result == newest.version
 523 
 524         # latest in environment
 525         environment = self.environment
 526         result = get_latest_release([self.project], [environment])
 527         assert result == new.version
 528 
 529         assert get_latest_release([self.project.id], [environment]) == ""
 530         assert (
 531             get_latest_release([self.project.id], [environment], self.project.organization_id)
 532             == new.version
 533         )
 534 
 535         # Verify that not passing an environment correctly gets the latest one
 536         assert get_latest_release([self.project], None) == newest.version
 537         assert get_latest_release([self.project], []) == newest.version
 538 
 539         with pytest.raises(Release.DoesNotExist):
 540             # environment with no releases
 541             environment = self.create_environment()
 542             result = get_latest_release([self.project], [environment])
 543             assert result == new.version
