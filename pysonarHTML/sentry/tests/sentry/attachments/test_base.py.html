   8 class InMemoryCache(object):
   9     """
  10     In-memory mock cache that roughly works like Django cache. Extended with
  11     internal assertions to ensure correct use of `raw`.
  12     """
  13 
  14     def __init__(self):
  15         self.data = {}
  16         #: Used to check for consistent usage of `raw` param
  17         self.raw_map = {}
  18 
  19     def get(self, key, raw=False):
  20         assert key not in self.raw_map or raw == self.raw_map[key]
  21         return copy.deepcopy(self.data.get(key))
  22 
  23     def set(self, key, value, timeout=None, raw=False):
  24         # Attachment chunks MUST be bytestrings. Josh please don't change this
  25         # to unicode.
  26         assert isinstance(value, bytes) or not raw
  27         assert key not in self.raw_map or raw == self.raw_map[key]
  28         self.data[key] = value
  29 
  30     def delete(self, key):
  31         del self.data[key]
  32

  34 def test_basic_chunked():
  35     data = InMemoryCache()
  36     cache = BaseAttachmentCache(data)
  37 
  38     cache.set_chunk("c:foo", 123, 0, b"Hello World! ")
  39     cache.set_chunk("c:foo", 123, 1, b"")
  40     cache.set_chunk("c:foo", 123, 2, b"Bye.")
  41 
  42     att = CachedAttachment(key="c:foo", id=123, name="lol.txt", content_type="text/plain", chunks=3)
  43     cache.set("c:foo", [att])
  44 
  45     (att2,) = cache.get("c:foo")
  46     assert att2.key == att.key == "c:foo"
  47     assert att2.id == att.id == 123
  48     assert att2.data == att.data == b"Hello World! Bye."
  49 
  50     cache.delete("c:foo")
  51     assert not list(cache.get("c:foo"))
