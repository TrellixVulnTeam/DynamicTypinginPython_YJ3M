  14 class ScheduleAutoResolutionTest(TestCase):
  15     def test_task_persistent_name(self):
  16         assert schedule_auto_resolution.name == "sentry.tasks.schedule_auto_resolution"
  17 
  18     @patch("sentry.tasks.auto_resolve_issues.kick_off_status_syncs")
  19     def test_simple(self, mock_kick_off_status_syncs):
  20         project = self.create_project()
  21         project2 = self.create_project()
  22         project3 = self.create_project()
  23         project4 = self.create_project()
  24 
  25         current_ts = int(time()) - 1
  26 
  27         project.update_option("sentry:resolve_age", 1)
  28         project3.update_option("sentry:resolve_age", 1)
  29         project3.update_option("sentry:_last_auto_resolve", current_ts)
  30         project4.update_option("sentry:_last_auto_resolve", current_ts)
  31 
  32         group1 = self.create_group(
  33             project=project,
  34             status=GroupStatus.UNRESOLVED,
  35             last_seen=timezone.now() - timedelta(days=1),
  36         )
  37 
  38         group2 = self.create_group(
  39             project=project, status=GroupStatus.UNRESOLVED, last_seen=timezone.now()
  40         )
  41 
  42         group3 = self.create_group(
  43             project=project3,
  44             status=GroupStatus.UNRESOLVED,
  45             last_seen=timezone.now() - timedelta(days=1),
  46         )
  47 
  48         with self.tasks():
  49             schedule_auto_resolution()
  50 
  51         assert Group.objects.get(id=group1.id).status == GroupStatus.RESOLVED
  52 
  53         assert Group.objects.get(id=group2.id).status == GroupStatus.UNRESOLVED
  54 
  55         assert Group.objects.get(id=group3.id).status == GroupStatus.UNRESOLVED
  56 
  57         mock_kick_off_status_syncs.apply_async.assert_called_once_with(
  58             kwargs={"project_id": group1.project_id, "group_id": group1.id}
  59         )
  60 
  61         assert project.get_option("sentry:_last_auto_resolve") > current_ts
  62         assert not project2.get_option("sentry:_last_auto_resolve")
  63         assert project3.get_option("sentry:_last_auto_resolve") == current_ts
  64         # this should get cleaned up since it had no resolve age set
