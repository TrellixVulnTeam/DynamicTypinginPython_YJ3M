 102     def test_timeout_with_no_future_complete_checkin(self):
 103         org = self.create_organization()
 104         project = self.create_project(organization=org)
 105 
 106         current_datetime = timezone.now() - timedelta(hours=24)
 107 
 108         monitor = Monitor.objects.create(
 109             organization_id=org.id,
 110             project_id=project.id,
 111             next_checkin=current_datetime + timedelta(hours=12, minutes=1),
 112             last_checkin=current_datetime + timedelta(hours=12),
 113             type=MonitorType.CRON_JOB,
 114             config={"schedule": "0 0 * * *"},
 115             status=MonitorStatus.OK,
 116             date_added=current_datetime,
 117         )
 118         checkin = MonitorCheckIn.objects.create(
 119             monitor=monitor,
 120             project_id=project.id,
 121             status=CheckInStatus.IN_PROGRESS,
 122             date_added=current_datetime,
 123             date_updated=current_datetime,
 124         )
 125         checkin2 = MonitorCheckIn.objects.create(
 126             monitor=monitor,
 127             project_id=project.id,
 128             status=CheckInStatus.IN_PROGRESS,
 129             date_added=monitor.last_checkin,
 130             date_updated=monitor.last_checkin,
 131         )
 132 
 133         assert checkin.date_added == checkin.date_updated == current_datetime
 134 
 135         check_monitors(current_datetime=current_datetime + timedelta(hours=12, minutes=1))
 136 
 137         assert MonitorCheckIn.objects.filter(id=checkin.id, status=CheckInStatus.ERROR).exists()
 138 
 139         assert MonitorCheckIn.objects.filter(
 140             id=checkin2.id, status=CheckInStatus.IN_PROGRESS
 141         ).exists()
 142 
 143         assert Monitor.objects.filter(id=monitor.id, status=MonitorStatus.ERROR).exists()
