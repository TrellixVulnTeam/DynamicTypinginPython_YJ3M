   1 from __future__ import absolute_import
   2 
   3 from datetime import timedelta
   4 from django.utils import timezone
   5 
   6 from sentry.models import Activity, Group, GroupResolution, GroupStatus, Release
   7 from sentry.tasks.clear_expired_resolutions import clear_expired_resolutions
   8 from sentry.testutils import TestCase
   9 
  10 
  11 class ClearExpiredResolutionsTest(TestCase):
  12     def test_task_persistent_name(self):
  13         assert clear_expired_resolutions.name == "sentry.tasks.clear_expired_resolutions"
  14 
  15     def test_simple(self):
  16         project = self.create_project()
  17 
  18         old_release = Release.objects.create(organization_id=project.organization_id, version="a")
  19         old_release.add_project(project)
  20 
  21         group1 = self.create_group(
  22             project=project, status=GroupStatus.RESOLVED, active_at=timezone.now()
  23         )
  24         resolution1 = GroupResolution.objects.create(
  25             group=group1, release=old_release, type=GroupResolution.Type.in_next_release
  26         )
  27         activity1 = Activity.objects.create(
  28             group=group1,
  29             project=project,
  30             type=Activity.SET_RESOLVED_IN_RELEASE,
  31             ident=resolution1.id,
  32             data={"version": ""},
  33         )
  34 
  35         new_release = Release.objects.create(
  36             organization_id=project.organization_id,
  37             version="b",
  38             date_added=timezone.now() + timedelta(minutes=1),
  39         )
  40         new_release.add_project(project)
  41 
  42         group2 = self.create_group(status=GroupStatus.UNRESOLVED, active_at=timezone.now())
  43         resolution2 = GroupResolution.objects.create(
  44             group=group2, release=new_release, type=GroupResolution.Type.in_next_release
  45         )
  46         activity2 = Activity.objects.create(
  47             group=group2,
  48             project=project,
  49             type=Activity.SET_RESOLVED_IN_RELEASE,
  50             ident=resolution2.id,
  51             data={"version": ""},
  52         )
  53 
  54         clear_expired_resolutions(new_release.id)
  55 
  56         assert Group.objects.get(id=group1.id).status == GroupStatus.RESOLVED
  57 
  58         assert Group.objects.get(id=group2.id).status == GroupStatus.UNRESOLVED
  59 
  60         # row should be updated to the in_release type, and reflect
  61         # the release it was resolved in
  62         resolution1 = GroupResolution.objects.get(id=resolution1.id)
  63         assert resolution1.status == GroupResolution.Status.resolved
  64         assert resolution1.release == new_release
  65         assert resolution1.type == GroupResolution.Type.in_release
  66 
  67         resolution2 = GroupResolution.objects.get(id=resolution2.id)
  68         assert resolution2.status == GroupResolution.Status.pending
  69 
  70         activity1 = Activity.objects.get(id=activity1.id)
  71         assert activity1.data["version"] == new_release.version
  72 
  73         activity2 = Activity.objects.get(id=activity2.id)
  74         assert activity2.data["version"] == ""
