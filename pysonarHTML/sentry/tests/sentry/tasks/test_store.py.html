  17 class BasicPreprocessorPlugin(Plugin2):
  18     def get_event_preprocessors(self, data):
  19         def remove_extra(data):
  20             del data["extra"]
  21             return data
  22 
  23         def put_on_hold(data):
  24             data["unprocessed"] = True
  25             return data
  26 
  27         if data.get("platform") == "mattlang":
  28             return [remove_extra, lambda x: None]
  29 
  30         if data.get("platform") == "noop":
  31             return [lambda data: None]
  32 
  33         if data.get("platform") == "holdmeclose":
  34             return [put_on_hold]
  35 
  36         return []

 309     default_project,

 311     mock_save_event,
 312     register_plugin,
 313     mock_event_processing_store,
 314     setting_method,
 315     options_model,
 316 ):
 317     @register_plugin
 318     class TestPlugin(Plugin2):
 319         def get_event_preprocessors(self, data):
 320             # Right now we do not scrub data from event preprocessors
 321             def more_extra(data):
 322                 data["extra"]["aaa2"] = "event preprocessor"
 323                 return data
 324 
 325             return [more_extra]
 326 
 327         def is_enabled(self, project=None):
 328             return True
 329 
 330     if setting_method == "datascrubbers":
 331         options_model.update_option("sentry:sensitive_fields", ["a"])
 332         options_model.update_option("sentry:scrub_data", True)
 333     elif setting_method == "piiconfig":
 334         options_model.update_option(
 335             "sentry:relay_pii_config", '{"applications": {"extra.aaa": ["@anything:replace"]}}'
 336         )
 337     else:
 338         raise ValueError(setting_method)
 339 
 340     data = {
 341         "project": default_project.id,
 342         "platform": "python",
 343         "logentry": {"formatted": "test"},
 344         "event_id": EVENT_ID,
 345         "extra": {"aaa": "remove me"},
 346     }
 347 
 348     mock_event_processing_store.get.return_value = data
 349     mock_event_processing_store.store.return_value = "e:1"
 350 
 351     with Feature({"organizations:datascrubbers-v2": True}):
 352         # We pass data_has_changed=True to pretend that we've added "extra" attribute
 353         # to "data" shortly before (e.g. during symbolication).
 354         process_event(cache_key="e:1", start_time=1, data_has_changed=True)
 355
