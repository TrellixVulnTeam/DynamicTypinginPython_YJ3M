 179 class DeleteGroupTest(TestCase):
 180     def test_simple(self):
 181         event_id = "a" * 32
 182         event_id_2 = "b" * 32
 183         project = self.create_project()
 184 
 185         node_id = Event.generate_node_id(project.id, event_id)
 186         node_id_2 = Event.generate_node_id(project.id, event_id_2)
 187 
 188         event = self.store_event(
 189             data={
 190                 "event_id": event_id,
 191                 "timestamp": iso_format(before_now(minutes=1)),
 192                 "fingerprint": ["group1"],
 193             },
 194             project_id=project.id,
 195         )
 196 
 197         self.store_event(
 198             data={
 199                 "event_id": event_id_2,
 200                 "timestamp": iso_format(before_now(minutes=1)),
 201                 "fingerprint": ["group1"],
 202             },
 203             project_id=project.id,
 204         )
 205 
 206         group = event.group
 207         group.update(status=GroupStatus.PENDING_DELETION)
 208 
 209         GroupAssignee.objects.create(group=group, project=project, user=self.user)
 210         GroupHash.objects.create(project=project, group=group, hash=uuid4().hex)
 211         GroupMeta.objects.create(group=group, key="foo", value="bar")
 212         GroupRedirect.objects.create(group_id=group.id, previous_group_id=1)
 213 
 214         assert nodestore.get(node_id)
 215         assert nodestore.get(node_id_2)
 216 
 217         with self.tasks():
 218             delete_groups(object_ids=[group.id])
 219 
 220         assert not GroupRedirect.objects.filter(group_id=group.id).exists()
 221         assert not GroupHash.objects.filter(group_id=group.id).exists()
 222         assert not Group.objects.filter(id=group.id).exists()
 223         assert not nodestore.get(node_id)
 224         assert not nodestore.get(node_id_2)

 249             application=app, user=self.create_user("bar@example.com"), scopes=0

 261     def test_with_timestamp(self):
 262         cutoff = datetime(2017, 1, 1)
 263         app = ApiApplication.objects.create(owner=self.user)
 264         token1 = ApiToken.objects.create(
 265             application=app, user=self.create_user("bar@example.com"), scopes=0, date_added=cutoff
 266         )
 267         token2 = ApiToken.objects.create(
 268             application=app,
 269             user=self.create_user("foo@example.com"),
