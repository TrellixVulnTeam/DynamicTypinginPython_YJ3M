  33 class AssembleDifTest(BaseAssembleTest):
  34     def test_wrong_dif(self):
  35         content1 = "foo".encode("utf-8")
  36         fileobj1 = ContentFile(content1)
  37 
  38         content2 = "bar".encode("utf-8")
  39         fileobj2 = ContentFile(content2)
  40 
  41         content3 = "baz".encode("utf-8")
  42         fileobj3 = ContentFile(content3)
  43 
  44         total_checksum = sha1(content2 + content1 + content3).hexdigest()
  45 
  46         # The order here is on purpose because we check for the order of checksums
  47         blob1 = FileBlob.from_file(fileobj1)
  48         blob3 = FileBlob.from_file(fileobj3)
  49         blob2 = FileBlob.from_file(fileobj2)
  50 
  51         chunks = [blob2.checksum, blob1.checksum, blob3.checksum]
  52 
  53         assemble_dif(
  54             project_id=self.project.id, name="foo.sym", checksum=total_checksum, chunks=chunks
  55         )
  56 
  57         status, _ = get_assemble_status(AssembleTask.DIF, self.project.id, total_checksum)
  58         assert status == ChunkFileState.ERROR

  81     def test_assemble_from_files(self):
  82         files = []
  83         file_checksum = sha1()
  84         for _ in xrange(8):
  85             blob = os.urandom(1024 * 1024 * 8)
  86             hash = sha1(blob).hexdigest()
  87             file_checksum.update(blob)
  88             files.append((io.BytesIO(blob), hash))
  89 
  90         # upload all blobs
  91         FileBlob.from_files(files, organization=self.organization)
  92 
  93         # find all blobs
  94         for reference, checksum in files:
  95             blob = FileBlob.objects.get(checksum=checksum)
  96             ref_bytes = reference.getvalue()
  97             assert blob.getfile().read(len(ref_bytes)) == ref_bytes
  98             FileBlobOwner.objects.filter(blob=blob, organization=self.organization).get()
  99 
 100         rv = assemble_file(
 101             AssembleTask.DIF,
 102             self.project,
 103             "testfile",
 104             file_checksum.hexdigest(),
 105             [x[1] for x in files],
 106             "dummy.type",
 107         )
 108 
 109         assert rv is not None
 110         f, tmp = rv
 111         assert f.checksum == file_checksum.hexdigest()
 112         assert f.type == "dummy.type"
 113 
 114         # upload all blobs a second time
 115         for f, _ in files:
 116             f.seek(0)
 117         FileBlob.from_files(files, organization=self.organization)
 118 
 119         # assemble a second time
 120         f = assemble_file(
 121             AssembleTask.DIF,
 122             self.project,
 123             "testfile",
 124             file_checksum.hexdigest(),
 125             [x[1] for x in files],
 126             "dummy.type",
 127         )[0]
 128         assert f.checksum == file_checksum.hexdigest()
