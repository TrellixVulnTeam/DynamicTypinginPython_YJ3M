  51     def test_merge_with_event_integrity(self):
  52         project = self.create_project()
  53         event1 = self.store_event(
  54             data={
  55                 "event_id": "a" * 32,
  56                 "timestamp": iso_format(before_now(seconds=1)),
  57                 "fingerprint": ["group-1"],
  58                 "extra": {"foo": "bar"},
  59             },
  60             project_id=project.id,
  61         )
  62         group1 = event1.group
  63         event2 = self.store_event(
  64             data={
  65                 "event_id": "b" * 32,
  66                 "timestamp": iso_format(before_now(seconds=1)),
  67                 "fingerprint": ["group-2"],
  68                 "extra": {"foo": "baz"},
  69             },
  70             project_id=project.id,
  71         )
  72         group2 = event2.group
  73 
  74         with self.tasks():
  75             eventstream_state = eventstream.start_merge(project.id, [group1.id], group2.id)
  76             merge_groups([group1.id], group2.id)
  77             eventstream.end_merge(eventstream_state)
  78 
  79         assert not Group.objects.filter(id=group1.id).exists()
  80 
  81         event1 = eventstore.get_event_by_id(project.id, event1.event_id)
  82         assert event1.group_id == group2.id
  83         assert event1.data["extra"]["foo"] == "bar"
  84 
  85         event2 = eventstore.get_event_by_id(project.id, event2.event_id)
  86         assert event2.group_id == group2.id
  87         assert event2.data["extra"]["foo"] == "baz"

 139     def test_merge_with_group_meta(self):
 140         project1 = self.create_project()
 141         event1 = self.store_event(data={}, project_id=project1.id)
 142         group1 = event1.group
 143 
 144         project2 = self.create_project()
 145         event2 = self.store_event(data={}, project_id=project2.id)
 146         group2 = event2.group
 147 
 148         GroupMeta.objects.create(group=event1.group, key="github:tid", value="134")
 149 
 150         GroupMeta.objects.create(group=event1.group, key="other:tid", value="567")
 151 
 152         GroupMeta.objects.create(group=event2.group, key="other:tid", value="abc")
 153 
 154         GroupMeta.objects.populate_cache([group1, group2])
 155 
 156         assert GroupMeta.objects.get_value(group1, "github:tid") == "134"
 157         assert GroupMeta.objects.get_value(group2, "other:tid") == "abc"
 158         assert not GroupMeta.objects.get_value(group2, "github:tid")
 159         assert GroupMeta.objects.get_value(group1, "other:tid") == "567"
 160 
 161         with self.tasks():
 162             merge_groups([group1.id], group2.id)
 163 
 164         assert not Group.objects.filter(id=group1.id).exists()
 165 
 166         GroupMeta.objects.clear_local_cache()
 167         GroupMeta.objects.populate_cache([group1, group2])
 168 
 169         assert not GroupMeta.objects.get_value(group1, "github:tid")
 170         assert not GroupMeta.objects.get_value(group1, "other:tid")
 171         assert GroupMeta.objects.get_value(group2, "github:tid") == "134"
 172         assert GroupMeta.objects.get_value(group2, "other:tid") == "abc"
