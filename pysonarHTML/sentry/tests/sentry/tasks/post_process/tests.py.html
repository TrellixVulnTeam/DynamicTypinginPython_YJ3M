 133     def make_ownership(self):
 134         rule_a = Rule(Matcher("path", "src/*"), [Owner("user", self.user.email)])
 135         rule_b = Rule(Matcher("path", "tests/*"), [Owner("team", self.team.name)])
 136         rule_c = Rule(Matcher("path", "src/app/*"), [Owner("team", self.team.name)])
 137 
 138         ProjectOwnership.objects.create(
 139             project_id=self.project.id,
 140             schema=dump_schema([rule_a, rule_b, rule_c]),
 141             fallthrough=True,
 142             auto_assignment=True,
 143         )
