   9 class CeleryQueueRegisteredTest(TestCase):
  10     def test(self):
  11         queue_names = set([q.name for q in settings.CELERY_QUEUES])
  12         missing_queue_tasks = []
  13         for task in current_app.tasks.values():
  14             # Filter out any tasks that aren't sentry specific, or don't specify `queue`.
  15             if task.name.startswith("celery.") or not hasattr(task, "queue"):
  16                 continue
  17             if task.queue not in queue_names:
  18                 missing_queue_tasks.append(" - Task: {}, Queue: {}".format(task.name, task.queue))
  19 
  20         assert not missing_queue_tasks, (
  21             "Found tasks with queues that are undefined. These must be defined in "
  22             "settings.CELERY_QUEUES.\nTask Info:\n{}.".format("\n".join(missing_queue_tasks))
