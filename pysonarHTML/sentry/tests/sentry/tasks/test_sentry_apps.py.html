  59 class DictContaining(object):
  60     def __init__(self, *args, **kwargs):
  61         if len(args) == 1 and isinstance(args[0], dict):
  62             self.args = []
  63             self.kwargs = args[0]
  64         else:
  65             self.args = args
  66             self.kwargs = kwargs
  67 
  68     def __eq__(self, other):
  69         return self._args_match(other) and self._kwargs_match(other)
  70 
  71     def _args_match(self, other):
  72         for key in self.args:
  73             if key not in other.keys():
  74                 return False
  75         return True
  76 
  77     def _kwargs_match(self, other):
  78         for key, value in six.iteritems(self.kwargs):
  79             if self.kwargs[key] != other[key]:
  80                 return False
  81         return True
