 205     def test_integration(self):
 206         Project.objects.all().delete()
 207 
 208         now = datetime(2016, 9, 12, tzinfo=pytz.utc)
 209 
 210         project = self.create_project(
 211             organization=self.organization, teams=[self.team], date_added=now - timedelta(days=90)
 212         )
 213 
 214         tsdb.incr(tsdb.models.project, project.id, now - timedelta(days=1))
 215 
 216         member_set = set(project.teams.first().member_set.all())
 217 
 218         with self.tasks(), mock.patch.object(
 219             tsdb, "get_earliest_timestamp"
 220         ) as get_earliest_timestamp:
 221             # Ensure ``get_earliest_timestamp`` is relative to the fixed
 222             # "current" timestamp -- this prevents filtering out data points
 223             # that would be considered expired relative to the *actual* current
 224             # timestamp.
 225             get_earliest_timestamp.return_value = to_timestamp(now - timedelta(days=60))
 226 
 227             prepare_reports(timestamp=to_timestamp(now))
 228             assert len(mail.outbox) == len(member_set) == 1
 229 
 230             message = mail.outbox[0]
 231             assert self.organization.name in message.subject

 300     @mock.patch("sentry.tasks.reports.BATCH_SIZE", 1)
 301     def test_paginates_project_series_and_reassembles_result(self):
 302         self.login_as(user=self.user)
 303 
 304         now = timezone.now()
 305         two_days_ago = now - timedelta(days=2)
 306         three_days_ago = iso_format(now - timedelta(days=3))
 307         seven_days_back = now - timedelta(days=7)
 308 
 309         event1 = self.store_event(
 310             data={
 311                 "event_id": "a" * 32,
 312                 "message": "message",
 313                 "timestamp": three_days_ago,
 314                 "stacktrace": copy.deepcopy(DEFAULT_EVENT_DATA["stacktrace"]),
 315                 "fingerprint": ["group-1"],
 316             },
 317             project_id=self.project.id,
 318         )
 319 
 320         event2 = self.store_event(
 321             data={
 322                 "event_id": "b" * 32,
 323                 "message": "message",
 324                 "timestamp": three_days_ago,
 325                 "stacktrace": copy.deepcopy(DEFAULT_EVENT_DATA["stacktrace"]),
 326                 "fingerprint": ["group-2"],
 327             },
 328             project_id=self.project.id,
 329         )
 330 
 331         group1 = event1.group
 332         group2 = event2.group
 333 
 334         group1.status = GroupStatus.RESOLVED
 335         group1.resolved_at = two_days_ago
 336         group1.save()
 337 
 338         group2.status = GroupStatus.RESOLVED
 339         group2.resolved_at = two_days_ago
 340         group2.save()
 341 
 342         response = prepare_project_series(
 343             [floor_to_utc_day(seven_days_back), floor_to_utc_day(now)], self.project
 344         )
 345 
 346         assert any(
 347             map(lambda x: x[1] == (2, 0), response)
 348         ), "must show two issues resolved in one rollup window"
