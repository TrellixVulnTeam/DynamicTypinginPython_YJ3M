  22     def test_simple(self):
  23         self.login_as(user=self.user)
  24         org = self.create_organization(owner=self.user, name="baz")
  25 
  26         repo = Repository.objects.create(name="example", provider="dummy", organization_id=org.id)
  27         release = Release.objects.create(organization_id=org.id, version="abcabcabc")
  28 
  29         commit = Commit.objects.create(organization_id=org.id, repository_id=repo.id, key="a" * 40)
  30 
  31         ReleaseHeadCommit.objects.create(
  32             organization_id=org.id, repository_id=repo.id, release=release, commit=commit
  33         )
  34 
  35         refs = [{"repository": repo.name, "commit": "b" * 40}]
  36 
  37         release2 = Release.objects.create(organization_id=org.id, version="12345678")
  38 
  39         deploy = Deploy.objects.create(organization_id=org.id, release=release2, environment_id=5)
  40 
  41         with self.tasks():
  42             with patch.object(Deploy, "notify_if_ready") as mock_notify_if_ready:
  43                 fetch_commits(
  44                     release_id=release2.id,
  45                     user_id=self.user.id,
  46                     refs=refs,
  47                     previous_release_id=release.id,
  48                 )
  49 
  50         commit_list = list(
  51             Commit.objects.filter(releasecommit__release=release2).order_by("releasecommit__order")
  52         )
  53 
  54         # see DummyRepositoryProvider.compare_commits
  55         assert len(commit_list) == 3
  56         assert commit_list[0].repository_id == repo.id
  57         assert commit_list[0].organization_id == org.id
  58         assert commit_list[0].key == "62de626b7c7cfb8e77efb4273b1a3df4123e6216"
  59         assert commit_list[1].repository_id == repo.id
  60         assert commit_list[1].organization_id == org.id
  61         assert commit_list[1].key == "58de626b7c7cfb8e77efb4273b1a3df4123e6345"
  62         assert commit_list[2].repository_id == repo.id
  63         assert commit_list[2].organization_id == org.id
  64         assert commit_list[2].key == "b" * 40
  65 
  66         mock_notify_if_ready.assert_called_with(deploy.id, fetch_complete=True)
  67 
  68         latest_repo_release_environment = LatestRepoReleaseEnvironment.objects.get(
  69             repository_id=repo.id, environment_id=5
  70         )
  71         assert latest_repo_release_environment.deploy_id == deploy.id
  72         assert latest_repo_release_environment.release_id == release2.id
  73         assert latest_repo_release_environment.commit_id == commit_list[0].id

 248 class HandleInvalidIdentityTest(TestCase):
 249     def test_simple(self):
 250         usa = UserSocialAuth.objects.create(user=self.user, provider="dummy")
 251 
 252         with self.tasks():
 253             handle_invalid_identity(usa)
 254 
 255         assert not UserSocialAuth.objects.filter(id=usa.id).exists()
 256 
 257         msg = mail.outbox[-1]
 258         assert msg.subject == "Action Required"
 259         assert msg.to == [self.user.email]
 260 
 261     def test_commit_failure(self):
 262         usa = UserSocialAuth.objects.create(user=self.user, provider="dummy")
 263 
 264         with self.tasks():
 265             handle_invalid_identity(usa, commit_failure=True)
 266 
 267         assert not UserSocialAuth.objects.filter(id=usa.id).exists()
 268 
 269         msg = mail.outbox[-1]
 270         assert msg.subject == "Unable to Fetch Commits"
 271         assert msg.to == [self.user.email]
