   9 class AssetsTest(TestCase):
  10     TEMPLATE = engines["django"].from_string(
  11         """
  12         {% load sentry_assets %}
  13         {% locale_js_include %}
  14     """
  15     )
  16 
  17     def test_supported_foreign_lang(self):
  18         result = self.TEMPLATE.render(
  19             request=MagicMock(LANGUAGE_CODE="fr")  # French, in locale/catalogs.json
  20         )
  21 
  22         assert '<script src="/_static/{version}/sentry/dist/locale/fr.js"></script>' in result
  23 
  24     def test_unsupported_foreign_lang(self):
  25         result = self.TEMPLATE.render(
  26             request=MagicMock(LANGUAGE_CODE="ro")  # Romanian, not in locale/catalogs.json
  27         )
  28 
  29         assert result.strip() == ""
  30 
  31     def test_english(self):
  32         result = self.TEMPLATE.render(request=MagicMock(LANGUAGE_CODE="en"))
  33 
  34         assert result.strip() == ""
  35 
  36     def test_no_lang(self):
  37         result = self.TEMPLATE.render(request=MagicMock())
  38 
  39         assert result.strip() == ""
