   8 class FeaturesTest(TestCase):
   9     # get a backend-dependent Template, just like get_template in >= Django 1.8
  10     TEMPLATE = engines["django"].from_string(
  11         """
  12         {% load sentry_features %}
  13         {% feature auth:register %}
  14             <span>register</span>
  15         {% else %}
  16             <span>nope</span>
  17         {% endfeature %}
  18     """
  19     )
  20 
  21     def test_enabled(self):
  22         with self.feature("auth:register"):
  23             result = self.TEMPLATE.render()
  24             assert "<span>register</span>" in result
  25 
  26     def test_disabled(self):
  27         with self.feature({"auth:register": False}):

  29             assert "<span>nope</span>" in result
