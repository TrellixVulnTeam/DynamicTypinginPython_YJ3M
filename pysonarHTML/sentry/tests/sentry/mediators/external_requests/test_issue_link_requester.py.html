  12 class TestIssueLinkRequester(TestCase):
  13     def setUp(self):

  15 
  16         self.user = self.create_user(name="foo")

  19         self.group = self.create_group(project=self.project)

  29     @responses.activate
  30     def test_makes_request(self):
  31         fields = {"title": "An Issue", "description": "a bug was found", "assignee": "user-1"}
  32 
  33         responses.add(
  34             method=responses.POST,
  35             url="https://example.com/link-issue",
  36             json={
  37                 "project": "ProjectName",
  38                 "webUrl": "https://example.com/project/issue-id",
  39                 "identifier": "issue-1",
  40             },
  41             status=200,
  42             content_type="application/json",
  43         )
  44 
  45         result = IssueLinkRequester.run(
  46             install=self.install,
  47             project=self.project,
  48             group=self.group,
  49             uri="/link-issue",
  50             fields=fields,
  51             user=self.user,
  52             action="create",
  53         )
  54         assert result == {
  55             "project": "ProjectName",
  56             "webUrl": "https://example.com/project/issue-id",
  57             "identifier": "issue-1",
  58         }
  59 
  60         request = responses.calls[0].request
  61         assert request.headers["Sentry-App-Signature"]
  62         data = {
  63             "fields": {"title": "An Issue", "description": "a bug was found", "assignee": "user-1"},
  64             "issueId": self.group.id,
  65             "installationId": self.install.uuid,
  66             "webUrl": self.group.get_absolute_url(),
  67             "project": {"id": self.project.id, "slug": self.project.slug},
  68             "actor": {"type": "user", "id": self.user.id, "name": self.user.name},
  69         }
  70         payload = json.loads(request.body)
  71         assert payload == data
  72 
  73         buffer = SentryAppWebhookRequestsBuffer(self.sentry_app)
  74         requests = buffer.get_requests()
  75 
  76         assert len(requests) == 1
  77         assert requests[0]["response_code"] == 200
  78         assert requests[0]["event_type"] == "external_issue.created"
