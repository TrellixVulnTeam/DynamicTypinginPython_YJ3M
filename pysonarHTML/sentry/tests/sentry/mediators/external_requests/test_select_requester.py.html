  12 class TestSelectRequester(TestCase):
  13     def setUp(self):
  14         super(TestSelectRequester, self).setUp()
  15 
  16         self.user = self.create_user(name="foo")
  17         self.org = self.create_organization(owner=self.user)
  18         self.project = self.create_project(slug="boop", organization=self.org)
  19 
  20         self.sentry_app = self.create_sentry_app(
  21             name="foo", organization=self.org, webhook_url="https://example.com", scopes=()
  22         )
  23 
  24         self.install = self.create_sentry_app_installation(
  25             slug="foo", organization=self.org, user=self.user
  26         )
  27 
  28     @responses.activate
  29     def test_makes_request(self):
  30         options = [
  31             {"label": "An Issue", "value": "123", "default": True},
  32             {"label": "Another Issue", "value": "456"},
  33         ]
  34         responses.add(
  35             method=responses.GET,
  36             url=u"https://example.com/get-issues?installationId={}&projectSlug={}".format(
  37                 self.install.uuid, self.project.slug
  38             ),
  39             json=options,
  40             status=200,
  41             content_type="application/json",
  42         )
  43 
  44         result = SelectRequester.run(install=self.install, project=self.project, uri="/get-issues")
  45 
  46         assert result == {
  47             "choices": [["123", "An Issue"], ["456", "Another Issue"]],
  48             "defaultValue": "123",
  49         }
  50 
  51         request = responses.calls[0].request
  52         assert request.headers["Sentry-App-Signature"]
  53 
  54         buffer = SentryAppWebhookRequestsBuffer(self.sentry_app)
  55         requests = buffer.get_requests()
  56 
  57         assert len(requests) == 1
  58         assert requests[0]["response_code"] == 200
  59         assert requests[0]["event_type"] == "select_options.requested"
