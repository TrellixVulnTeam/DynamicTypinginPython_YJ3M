   9 class TestCreator(TestCase):
  10     def setUp(self):
  11         self.user = self.create_user()
  12         self.org = self.create_organization(owner=self.user)
  13         self.project = self.create_project(name="foo", organization=self.org)
  14         self.sentry_app = self.create_sentry_app(owner=self.org)
  15         self.creator = Creator(
  16             application=self.sentry_app.application,
  17             actor=self.sentry_app.proxy_user,
  18             organization=self.org,
  19             projects=[self.project],
  20             events=("event.created",),
  21             url=self.sentry_app.webhook_url,
  22         )
  23 
  24     def test_creates_service_hook(self):
  25         self.creator.call()
  26 
  27         service_hook = ServiceHook.objects.get(
  28             application=self.sentry_app.application,
  29             actor_id=self.sentry_app.proxy_user.id,
  30             project_id=self.project.id,
  31             url=self.sentry_app.webhook_url,
  32         )
  33 
  34         assert service_hook
  35         assert service_hook.events == ["event.created"]
  36         hook_project = ServiceHookProject.objects.get(project_id=self.project.id)
  37         assert hook_project.service_hook_id == service_hook.id
  38 
  39     def test_expands_resource_events_to_specific_events(self):
  40         self.creator.events = ["issue"]
  41         service_hook = self.creator.call()
  42 
  43         assert set(service_hook.events) == set(
  44             ["issue.created", "issue.resolved", "issue.ignored", "issue.assigned"]
  45         )
  46 
  47     def test_expand_events(self):
  48         assert expand_events(["issue"]) == set(
  49             ["issue.created", "issue.resolved", "issue.ignored", "issue.assigned"]
  50         )
  51 
  52     def test_consolidate_events(self):
  53         assert consolidate_events(["issue.created"]) == set(["issue"])
