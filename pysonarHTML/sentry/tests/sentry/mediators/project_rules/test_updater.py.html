   7 class TestUpdater(TestCase):
   8     def setUp(self):
   9         self.user = self.create_user()
  10         self.org = self.create_organization(name="bloop", owner=self.user)
  11         self.project = self.create_project(
  12             teams=[self.create_team()], name="foo", fire_project_created=True
  13         )

  17     def test_update_name(self):
