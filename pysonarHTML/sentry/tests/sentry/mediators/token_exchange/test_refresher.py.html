  11 class TestRefresher(TestCase):
  12     def setUp(self):
  13         self.install = self.create_sentry_app_installation()
  14         self.client_id = self.install.sentry_app.application.client_id
  15         self.user = self.install.sentry_app.proxy_user
  16 
  17         self.token = GrantExchanger.run(
  18             install=self.install,
  19             code=self.install.api_grant.code,
  20             client_id=self.client_id,
  21             user=self.user,
  22         )
  23 
  24         self.refresher = Refresher(
  25             install=self.install,
  26             client_id=self.client_id,
  27             refresh_token=self.token.refresh_token,
  28             user=self.user,
  29         )
  30 
  31     def test_happy_path(self):
  32         assert self.refresher.call()
  33 
  34     def test_adds_token_to_installation(self):
  35         token = self.refresher.call()
  36         assert SentryAppInstallation.objects.get(id=self.install.id).api_token == token
  37 
  38     def test_deletes_refreshed_token(self):
  39         self.refresher.call()
  40         assert not ApiToken.objects.filter(id=self.token.id).exists()

  50     def test_validates_token_belongs_to_sentry_app(self):
