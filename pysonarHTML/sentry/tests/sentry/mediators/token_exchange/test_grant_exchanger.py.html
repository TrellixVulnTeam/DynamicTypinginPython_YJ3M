  12 class TestGrantExchanger(TestCase):
  13     def setUp(self):
  14         self.install = self.create_sentry_app_installation()
  15         self.code = self.install.api_grant.code
  16         self.client_id = self.install.sentry_app.application.client_id
  17         self.user = self.install.sentry_app.proxy_user
  18 
  19         self.grant_exchanger = GrantExchanger(
  20             install=self.install, client_id=self.client_id, code=self.code, user=self.user
  21         )
  22 
  23     def test_happy_path(self):
  24         assert self.grant_exchanger.call()
  25 
  26     def test_adds_token_to_installation(self):
  27         token = self.grant_exchanger.call()
  28         assert SentryAppInstallation.objects.get(id=self.install.id).api_token == token
  29 
  30     @patch("sentry.mediators.token_exchange.Validator.run")
  31     def test_validate_generic_token_exchange_requirements(self, validator):
  32         self.grant_exchanger.call()
  33 
  34         validator.assert_called_once_with(
  35             install=self.install, client_id=self.client_id, user=self.user
  36         )
  37 
  38     def test_grant_must_belong_to_installations(self):
  39         other_install = self.create_sentry_app_installation()
  40         self.grant_exchanger.code = other_install.api_grant.code
  41 
  42         with self.assertRaises(APIUnauthorized):
  43             self.grant_exchanger.call()
  44 
  45     def test_request_user_owns_api_grant(self):
  46         self.grant_exchanger.user = self.create_user()
  47 
  48         with self.assertRaises(APIUnauthorized):
  49             self.grant_exchanger.call()
  50 
  51     def test_grant_must_be_active(self):
  52         self.install.api_grant.update(expires_at=(datetime.utcnow() - timedelta(hours=1)))
  53 
  54         with self.assertRaises(APIUnauthorized):
  55             self.grant_exchanger.call()
  56 
  57     def test_grant_must_exist(self):

  62

  67 
  68     @patch("sentry.models.ApiApplication.sentry_app", side_effect=SentryApp.DoesNotExist)
  69     def test_sentry_app_must_exist(self, _):

  72 
  73     def test_deletes_grant_on_successful_exchange(self):
  74         grant_id = self.install.api_grant_id
  75         self.grant_exchanger.call()
  76         assert not ApiGrant.objects.filter(id=grant_id)
  77
