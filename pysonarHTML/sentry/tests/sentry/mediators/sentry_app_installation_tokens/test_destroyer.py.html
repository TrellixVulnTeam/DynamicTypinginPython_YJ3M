   5 from sentry.mediators.sentry_app_installation_tokens import Destroyer
   6 from sentry.models import AuditLogEntry, ApiToken, SentryAppInstallationToken, SentryAppInstallation
   7 from sentry.testutils import TestCase

  10 class TestDestroyer(TestCase):
  11     def setUp(self):
  12         self.user = self.create_user()
  13         self.org = self.create_organization(owner=self.user)
  14         self.create_project(organization=self.org)
  15 
  16         self.sentry_app = self.create_internal_integration(name="nulldb", organization=self.org)
  17 
  18         self.sentry_app_installation = SentryAppInstallation.objects.get(sentry_app=self.sentry_app)
  19         self.api_token = self.create_internal_integration_token(
  20             install=self.sentry_app_installation, user=self.user
  21         )
  22 
  23     @patch("sentry.analytics.record")
  24     def test_delete_token_with_audit(self, record):
  25         request = self.make_request(user=self.user, method="DELETE")
  26         api_token = Destroyer.run(
  27             api_token=self.api_token, user=self.user, generate_audit=True, request=request
  28         )
  29 
  30         assert not ApiToken.objects.filter(id=api_token.id).exists()
  31         assert not SentryAppInstallationToken.objects.filter(api_token_id=api_token.id).exists()
  32 
  33         log = AuditLogEntry.objects.get(organization=self.org)
  34         assert log.get_note() == "revoked a token for internal integration nulldb"
  35         assert log.organization == self.org
  36         assert log.target_object == api_token.id
  37 
  38         record.assert_called_with(
  39             "sentry_app_installation_token.deleted",
  40             user_id=self.user.id,
  41             organization_id=self.org.id,
  42             sentry_app_installation_id=self.sentry_app_installation.id,
  43             sentry_app=self.sentry_app.slug,
  44         )
  45 
  46     @patch("sentry.utils.audit.create_audit_entry")
  47     @patch("sentry.analytics.record")
  48     def test_delete_token_without_audit(self, record, create_audit_entry):
  49         request = self.make_request(user=self.user, method="DELETE")
  50         api_token = Destroyer.run(api_token=self.api_token, user=self.user, request=request)
  51 
  52         assert not ApiToken.objects.filter(id=api_token.id).exists()
  53         assert not SentryAppInstallationToken.objects.filter(api_token_id=api_token.id).exists()
  54 
  55         assert not create_audit_entry.called
  56 
  57         record.assert_called_with(
  58             "sentry_app_installation_token.deleted",
  59             user_id=self.user.id,
  60             organization_id=self.org.id,
  61             sentry_app_installation_id=self.sentry_app_installation.id,

  63         )
