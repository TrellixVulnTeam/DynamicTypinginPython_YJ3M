  45     def test_updates_unpublished_app_scopes(self):
  46         # create both expired token and not expired tokens
  47         ApiToken.objects.create(
  48             application=self.sentry_app.application,
  49             user=self.sentry_app.proxy_user,
  50             scopes=self.sentry_app.scopes,
  51             scope_list=self.sentry_app.scope_list,
  52             expires_at=(timezone.now() + timedelta(hours=1)),
  53         )
  54         ApiToken.objects.create(
  55             application=self.sentry_app.application,
  56             user=self.sentry_app.proxy_user,
  57             scopes=self.sentry_app.scopes,
  58             scope_list=self.sentry_app.scope_list,
  59             expires_at=(timezone.now() - timedelta(hours=1)),
  60         )
  61         self.updater.scopes = ("project:read", "project:write")
  62         self.updater.call()
  63         assert self.sentry_app.get_scopes() == ["project:read", "project:write"]
  64         tokens = ApiToken.objects.filter(application=self.sentry_app.application).order_by(
  65             "expires_at"
  66         )
  67         assert tokens[0].get_scopes() == ["project:read"]
  68         assert tokens[1].get_scopes() == ["project:read", "project:write"]
