  19 class TestCreator(TestCase):
  20     def setUp(self):
  21         self.user = self.create_user()
  22         self.org = self.create_organization(owner=self.user)
  23         self.creator = Creator(
  24             name="nulldb",
  25             user=self.user,
  26             author="Sentry",
  27             organization=self.org,
  28             scopes=("project:read",),

  31             is_internal=False,

  49     def test_creates_sentry_app(self):
  50         self.creator.call()
  51 
  52         proxy = User.objects.get(username__contains="nulldb")
  53         app = ApiApplication.objects.get(owner=proxy)
  54 
  55         sentry_app = SentryApp.objects.get(
  56             name="nulldb", application=app, owner=self.org, proxy_user=proxy
  57         )
  58 
  59         assert sentry_app
  60         assert sentry_app.scope_list == ["project:read"]
