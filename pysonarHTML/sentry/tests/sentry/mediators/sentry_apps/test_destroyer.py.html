  19 class TestDestroyer(TestCase):
  20     def setUp(self):
  21         self.user = self.create_user()
  22         self.org = self.create_organization()
  23 
  24         self.sentry_app = self.create_sentry_app(
  25             name="blah", organization=self.org, scopes=("project:read",)
  26         )
  27 
  28         self.destroyer = Destroyer(sentry_app=self.sentry_app, user=self.user)
  29 
  30     @responses.activate
  31     def test_deletes_app_installations(self):
  32         responses.add(responses.POST, "https://example.com/webhook", status=200)
  33         install = self.create_sentry_app_installation(
  34             organization=self.org, slug=self.sentry_app.slug, user=self.user
  35         )
  36         self.destroyer.call()
  37         assert not SentryAppInstallation.objects.filter(pk=install.id).exists()
  38 
  39     @patch("sentry.mediators.sentry_app_installations.Destroyer.run")
  40     def test_passes_notify_false_if_app_internal(self, run):
  41         self.create_project(organization=self.org)
  42         internal = self.create_internal_integration(organization=self.org)
  43         Destroyer.run(sentry_app=internal, user=self.user)
  44         run.assert_called_with(
  45             install=internal.installations.first(), user=internal.proxy_user, notify=False
  46         )
  47 
  48     def test_deletes_api_application(self):
  49         application = self.sentry_app.application
  50 
  51         self.destroyer.call()
  52 
  53         assert not ApiApplication.objects.filter(pk=application.id).exists()
  54 
  55     def test_deletes_proxy_user(self):
  56         proxy_user = self.sentry_app.proxy_user
  57 
  58         self.destroyer.call()
  59 
  60         assert not User.objects.filter(pk=proxy_user.id).exists()
  61 
  62     def test_creates_audit_log_entry(self):
  63         request = self.make_request(user=self.user, method="GET")
  64         Destroyer.run(user=self.user, sentry_app=self.sentry_app, request=request)
  65         assert AuditLogEntry.objects.filter(event=AuditLogEntryEvent.SENTRY_APP_REMOVE).exists()
  66 
  67     def test_soft_deletes_sentry_app(self):
  68         self.destroyer.call()
  69 
  70         with self.assertRaises(SentryApp.DoesNotExist):
  71             SentryApp.objects.get(pk=self.sentry_app.id)
  72 
  73         # The QuerySet will automatically NOT include deleted installs, so we
  74         # use a raw sql query to ensure it still exists.
  75         c = connection.cursor()
  76         c.execute(
  77             "SELECT count(1) "
  78             "FROM sentry_sentryapp "
  79             "WHERE id = %s AND date_deleted IS NOT NULL",
  80             [self.sentry_app.id],
  81         )
  82 
  83         assert c.fetchone()[0] == 1
  84 
  85     @patch("sentry.analytics.record")

  91         )
  92 
  93         record.assert_called_with(
  94             "sentry_app.deleted",
  95             user_id=self.user.id,
  96             organization_id=self.org.id,
  97             sentry_app=self.sentry_app.slug,
  98         )
