  21 class TestDestroyer(TestCase):
  22     @responses.activate
  23     def setUp(self):
  24         self.user = self.create_user()
  25         self.org = self.create_organization()
  26         self.project = self.create_project(organization=self.org)
  27 
  28         responses.add(responses.POST, "https://example.com/webhook")
  29 
  30         self.sentry_app = self.create_sentry_app(
  31             name="nulldb",
  32             organization=self.org,
  33             scopes=("project:read", "event:read"),
  34             events=("issue",),
  35         )
  36 
  37         self.install = Creator.run(organization=self.org, slug="nulldb", user=self.user)
  38 
  39         self.destroyer = Destroyer(install=self.install, user=self.user)

  41     @responses.activate
  42     def test_deletes_grant(self):
  43         grant = self.install.api_grant
  44 
  45         responses.add(responses.POST, "https://example.com/webhook")

 110     @responses.activate
 111     def test_soft_deletes_installation(self):
 112         responses.add(responses.POST, "https://example.com/webhook")
 113         self.destroyer.call()
 114 
 115         with self.assertRaises(SentryAppInstallation.DoesNotExist):
 116             SentryAppInstallation.objects.get(pk=self.install.id)
 117 
 118         # The QuerySet will automatically NOT include deleted installs, so we
 119         # use a raw sql query to ensure it still exists.
 120         c = connection.cursor()
 121         c.execute(
 122             "SELECT COUNT(1) "
 123             "FROM sentry_sentryappinstallation "
 124             "WHERE id = %s AND date_deleted IS NOT NULL",
 125             [self.install.id],
 126         )
 127 
 128         assert c.fetchone()[0] == 1

 144     @responses.activate
 145     def test_fail_on_other_error(self):
 146         from sentry.constants import SentryAppStatus
 147 
 148         install = self.install
 149         self.sentry_app.update(status=SentryAppStatus.PUBLISHED)
 150         # we don't log errors for unpublished and internal apps
 151         try:
 152             responses.add(
 153                 responses.POST, "https://example.com/webhook", body=Exception("Other error")
 154             )
 155 
 156             self.destroyer.call()
 157         except Exception:
 158             assert SentryAppInstallation.objects.filter(pk=install.id).exists()
 159         else:
 160             assert False, "Should fail on other error"
