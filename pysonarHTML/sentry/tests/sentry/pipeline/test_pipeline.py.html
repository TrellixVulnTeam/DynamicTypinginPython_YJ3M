  33 class PipelineTestCase(TestCase):
  34     def test_simple_pipeline(self):
  35         org = self.create_organization()
  36         request = HttpRequest()
  37         request.session = {}
  38         request.user = self.user
  39 
  40         pipeline = DummyPipeline(request, "dummy", org, config={"some_config": True})
  41         pipeline.initialize()
  42 
  43         assert pipeline.is_valid()
  44 
  45         assert "some_config" in pipeline.provider.config
  46 
  47         # Test state
  48         pipeline.finished = False
  49         pipeline.dispatch_count = 0
  50 
  51         # Pipeline has two steps, ensure both steps compete. Usually the
  52         # dispatch itself would be the one calling the current_step and
  53         # next_step methods after it determines if it can move forward a step.
  54         pipeline.current_step()
  55         assert pipeline.dispatch_count == 1
  56         assert pipeline.fetch_state("some_state") == "value"
  57 
  58         pipeline.next_step()
  59         assert pipeline.dispatch_count == 2
  60 
  61         pipeline.next_step()
  62         assert pipeline.dispatch_count == 2
  63         assert pipeline.finished
  64 
  65         pipeline.clear_session()
  66         assert not pipeline.state.is_valid()

  68     def test_invalidated_pipeline(self):
