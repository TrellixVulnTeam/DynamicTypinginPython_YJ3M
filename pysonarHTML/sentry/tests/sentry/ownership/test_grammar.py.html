  19 def test_parse_rules():
  20     assert parse_rules(fixture_data) == [
  21         Rule(Matcher("path", "*.js"), [Owner("team", "frontend"), Owner("user", "m@robenolt.com")]),
  22         Rule(Matcher("url", "http://google.com/*"), [Owner("team", "backend")]),
  23         Rule(Matcher("path", "src/sentry/*"), [Owner("user", "david@sentry.io")]),
  24         Rule(Matcher("tags.foo", "bar"), [Owner("user", "tagperson@sentry.io")]),

  55 def test_matcher_test_url():
  56     data = {"request": {"url": "http://example.com/foo.js"}}
  57 
  58     assert Matcher("url", "*.js").test(data)
  59     assert Matcher("url", "http://*.com/foo.js").test(data)
  60     assert not Matcher("url", "*.py").test(data)
  61     assert not Matcher("url", "*.jsx").test(data)
  62     assert not Matcher("path", "*.js").test(data)
  63     assert not Matcher("url", "*.js").test({})
  64

  66 def test_matcher_test_none():
