 109     def test_find_stacktraces_skip_none(self):
 110         # This tests:
 111         #  1. exception is None
 112         #  2. stacktrace is None
 113         #  3. frames is None
 114         #  3. frames contains only None
 115         #  4. frame is None
 116         data = {
 117             "message": "hello",
 118             "platform": "javascript",
 119             "exception": {
 120                 "values": [
 121                     None,
 122                     {"type": "Error", "stacktrace": None},
 123                     {"type": "Error", "stacktrace": {"frames": None}},
 124                     {"type": "Error", "stacktrace": {"frames": [None]}},
 125                     {
 126                         "type": "Error",
 127                         "stacktrace": {
 128                             "frames": [
 129                                 None,
 130                                 {
 131                                     "abs_path": "http://example.com/foo.js",
 132                                     "filename": "foo.js",
 133                                     "lineno": 4,
 134                                     "colno": 0,
 135                                 },
 136                                 {
 137                                     "abs_path": "http://example.com/foo.js",
 138                                     "filename": "foo.js",
 139                                     "lineno": 1,
 140                                     "colno": 0,
 141                                 },
 142                             ]
 143                         },
 144                     },
 145                 ]
 146             },
 147         }
 148 
 149         infos = find_stacktraces_in_data(data, with_exceptions=True)
 150         assert len(infos) == 4
 151         assert sum(1 for x in infos if x.stacktrace) == 3
 152         assert sum(1 for x in infos if x.is_exception) == 4
 153         # XXX: The null frame is still part of this stack trace!
 154         assert len(infos[3].stacktrace["frames"]) == 3
 155 
 156         infos = find_stacktraces_in_data(data)
 157         assert len(infos) == 1
 158         # XXX: The null frame is still part of this stack trace!
 159         assert len(infos[0].stacktrace["frames"]) == 3

 214     def test_ios_package_in_app_detection(self):
 215         data = {
 216             "platform": "native",
 217             "stacktrace": {
 218                 "frames": [
 219                     {
 220                         "package": "/var/containers/Bundle/Application/B33C37A8-F933-4B6B-9FFA-152282BFDF13/SentryTest.app/SentryTest",
 221                         "instruction_addr": "0x1000",
 222                     },
 223                     {
 224                         "package": "/var/containers/Bundle/Application/B33C37A8-F933-4B6B-9FFA-152282BFDF13/SentryTest.app/Frameworks/foo.dylib",
 225                         "instruction_addr": "0x2000",
 226                     },
 227                     {
 228                         "package": "/var/containers/Bundle/Application/B33C37A8-F933-4B6B-9FFA-152282BFDF13/SentryTest.app/Frameworks/libswiftCore.dylib",
 229                         "instruction_addr": "0x3000",
 230                     },
 231                     {"package": "/usr/lib/whatever.dylib", "instruction_addr": "0x4000"},
 232                 ]
 233             },
 234         }
 235 
 236         config = load_grouping_config(get_default_grouping_config_dict())
 237         normalize_stacktraces_for_grouping(data, grouping_config=config)
 238 
 239         # App object should be in_app
 240         assert data["stacktrace"]["frames"][0]["in_app"] is True
 241         # Framework should be in app (but optional)
 242         assert data["stacktrace"]["frames"][1]["in_app"] is True
 243         # libswift should not be system
 244         assert data["stacktrace"]["frames"][2]["in_app"] is False
 245         # Unknown object should default to not in_app
 246         assert data["stacktrace"]["frames"][3]["in_app"] is False
