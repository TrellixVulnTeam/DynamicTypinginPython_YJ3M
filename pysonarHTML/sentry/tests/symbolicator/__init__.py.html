  12 def strip_frame(frame):
  13     if frame:
  14         frame = {
  15             "data": {
  16                 "symbolicator_status": get_path(frame, "data", "symbolicator_status"),
  17                 "orig_in_app": get_path(frame, "data", "orig_in_app"),
  18             },
  19             "function": frame.get("function"),
  20             "instruction_addr": frame.get("instruction_addr"),
  21             "symbol": frame.get("symbol"),
  22             "package": frame.get("package"),
  23             "lineno": frame.get("lineno"),
  24             "in_app": frame.get("in_app"),
  25             "trust": frame.get("trust"),
  26         }

  31 def strip_stacktrace(stacktrace):
  32     if stacktrace:
  33         stacktrace = dict(stacktrace)
  34         stacktrace["frames"] = [strip_frame(x) for x in stacktrace.get("frames") or ()]
  35 
  36     return stacktrace
  37

  39 def strip_stacktrace_container(container):
  40     if container:
  41         container = dict(container)
  42         container["stacktrace"] = strip_stacktrace(container.get("stacktrace"))
  43         container["raw_stacktrace"] = strip_stacktrace(container.get("raw_stacktrace"))
  44 
  45     return container

  48 def insta_snapshot_stacktrace_data(self, event):

  52         {
