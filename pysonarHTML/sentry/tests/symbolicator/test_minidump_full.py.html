  40     def upload_symbols(self):
  41         url = reverse(
  42             "sentry-api-0-dsym-files",
  43             kwargs={
  44                 "organization_slug": self.project.organization.slug,
  45                 "project_slug": self.project.slug,
  46             },
  47         )
  48 
  49         self.login_as(user=self.user)
  50 
  51         out = BytesIO()
  52         f = zipfile.ZipFile(out, "w")
  53         f.write(get_fixture_path("windows.sym"), "crash.sym")
  54         f.close()
  55 
  56         response = self.client.post(
  57             url,
  58             {
  59                 "file": SimpleUploadedFile(
  60                     "symbols.zip", out.getvalue(), content_type="application/zip"
  61                 )
  62             },
  63             format="multipart",
  64         )
  65         assert response.status_code == 201, response.content
  66         assert len(response.data) == 1

  68     def test_full_minidump(self):
  69         self.project.update_option("sentry:store_crash_reports", STORE_CRASH_REPORTS_ALL)
  70         self.upload_symbols()
  71 
  72         with self.feature("organizations:event-attachments"):
  73             attachment = BytesIO(b"Hello World!")
  74             attachment.name = "hello.txt"
  75             with open(get_fixture_path("windows.dmp"), "rb") as f:
  76                 resp = self._postMinidumpWithHeader(
  77                     f, {"sentry[logger]": "test-logger", "some_file": attachment}
  78                 )
  79                 assert resp.status_code == 200
  80                 event_id = resp.content
  81 
  82         event = eventstore.get_event_by_id(self.project.id, event_id)
  83         insta_snapshot_stacktrace_data(self, event.data)
  84         assert event.data.get("logger") == "test-logger"
  85         # assert event.data.get("extra") == {"foo": "bar"}
  86 
  87         attachments = sorted(
  88             EventAttachment.objects.filter(event_id=event.event_id), key=lambda x: x.name
  89         )
  90         hello, minidump = attachments
  91 
  92         assert hello.name == "hello.txt"
  93         assert hello.file.type == "event.attachment"
  94         assert hello.file.checksum == "2ef7bde608ce5404e97d5f042f95f89f1c232871"
  95 
  96         assert minidump.name == "windows.dmp"
  97         assert minidump.file.type == "event.minidump"
  98         assert minidump.file.checksum == "74bb01c850e8d65d3ffbc5bad5cabc4668fce247"
