  51     def upload_symbols(self):
  52         url = reverse(
  53             "sentry-api-0-dsym-files",
  54             kwargs={
  55                 "organization_slug": self.project.organization.slug,
  56                 "project_slug": self.project.slug,
  57             },
  58         )
  59 
  60         self.login_as(user=self.user)
  61 
  62         out = BytesIO()
  63         f = zipfile.ZipFile(out, "w")
  64         f.write(get_fixture_path("unreal_crash.sym"), "crash.sym")
  65         f.close()
  66 
  67         response = self.client.post(
  68             url,
  69             {
  70                 "file": SimpleUploadedFile(
  71                     "symbols.zip", out.getvalue(), content_type="application/zip"
  72                 )
  73             },
  74             format="multipart",
  75         )
  76         assert response.status_code == 201, response.content
  77         assert len(response.data) == 1
