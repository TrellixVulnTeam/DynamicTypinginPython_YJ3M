  36 def install_sentry_plugins():
  37     # Sentry's pytest plugin explicitly doesn't load plugins, so let's load all of them
  38     # and ignore the fact that we're not *just* testing our own
  39     # Note: We could manually register/configure INSTALLED_APPS by traversing our entry points
  40     # or package directories, but this is easier assuming Sentry doesn't change APIs.
  41     # Note: Order of operations matters here.
  42     from sentry.runner.importer import install_plugin_apps
  43     from django.conf import settings
  44 
  45     install_plugin_apps("sentry.apps", settings)
  46 
  47     from sentry.runner.initializer import register_plugins
  48 
  49     register_plugins(settings, raise_on_plugin_load_failure=True)

  61 def pytest_collection_modifyitems(config, items):
  62     for item in items:
  63         total_groups = int(os.environ.get("TOTAL_TEST_GROUPS", 1))
  64         # TODO(joshuarli): six 1.12.0 adds ensure_binary: six.ensure_binary(item.location[0])
  65         group_num = (
  66             int(md5(six.text_type(item.location[0]).encode("utf-8")).hexdigest(), 16) % total_groups
  67         )
  68         marker = "group_%s" % group_num
  69         config.addinivalue_line("markers", marker)
  70         item.add_marker(getattr(pytest.mark, marker))
