  49 class SentrySDistCommand(SDistCommand):

  60 class SentryBuildCommand(BuildCommand):
  61     def run(self):
  62         from distutils import log as distutils_log
  63 
  64         distutils_log.set_threshold(distutils_log.WARN)
  65 
  66         if not IS_LIGHT_BUILD:
  67             self.run_command("build_integration_docs")
  68             self.run_command("build_assets")
  69             self.run_command("build_js_sdk_registry")
  70         BuildCommand.run(self)

  73 class SentryDevelopCommand(DevelopCommand):
