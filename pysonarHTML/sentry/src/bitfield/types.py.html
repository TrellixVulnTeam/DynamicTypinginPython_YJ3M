  13 class Bit(object):
  14     def __init__(self, number, is_set=True):
  15         self.number = number
  16         self.is_set = bool(is_set)
  17         self.mask = 2 ** int(number)
  18         self.children = []
  19         if not self.is_set:
  20             self.mask = ~self.mask
  21 
  22     def __repr__(self):
  23         return "<%s: number=%d, is_set=%s>" % (self.__class__.__name__, self.number, self.is_set)
  24 
  25     def __int__(self):
  26         return self.mask
  27 
  28     def __bool__(self):
  29         return self.is_set
  30 
  31     __nonzero__ = __bool__
  32 
  33     def __eq__(self, value):
  34         if isinstance(value, Bit):
  35             return value.number == self.number and value.is_set == self.is_set
  36         elif isinstance(value, bool):
  37             return value == self.is_set
  38         elif isinstance(value, int):
  39             return value == self.mask
  40         return value == self.is_set
  41 
  42     def __ne__(self, value):
  43         return not self == value
  44 
  45     def __coerce__(self, value):
  46         return (self.is_set, bool(value))
  47 
  48     def __invert__(self):
  49         return self.__class__(self.number, not self.is_set)
  50 
  51     def __and__(self, value):
  52         if isinstance(value, Bit):
  53             value = value.mask
  54         return value & self.mask
  55 
  56     def __rand__(self, value):
  57         if isinstance(value, Bit):
  58             value = value.mask
  59         return self.mask & value
  60 
  61     def __or__(self, value):
  62         if isinstance(value, Bit):
  63             value = value.mask
  64         return value | self.mask
  65 
  66     def __ror__(self, value):
  67         if isinstance(value, Bit):
  68             value = value.mask
  69         return self.mask | value

  71     def __lshift__(self, value):
  72         if isinstance(value, Bit):

  74         return value << self.mask
  75 
  76     def __rlshift__(self, value):
  77         if isinstance(value, Bit):

  79         return self.mask << value
  80 
  81     def __rshift__(self, value):

 112 class BitHandler(object):
 113     """
 114     Represents an array of bits, each as a ``Bit`` object.
 115     """
 116 
 117     def __init__(self, value, keys, labels=None):
 118         # TODO: change to bitarray?
 119         if value:
 120             self._value = int(value)
 121         else:
 122             self._value = 0
 123         self._keys = keys
 124         self._labels = labels is not None and labels or keys
 125 
 126     def __eq__(self, other):
 127         if not isinstance(other, BitHandler):
 128             return False
 129         return self._value == other._value
 130 
 131     def __lt__(self, other):
 132         return int(self._value) < other
 133 
 134     def __le__(self, other):
 135         return int(self._value) <= other
 136 
 137     def __gt__(self, other):
 138         return int(self._value) > other
 139 
 140     def __ge__(self, other):
 141         return int(self._value) >= other
 142 
 143     def __cmp__(self, other):
 144         return cmp(self._value, other)
 145 
 146     def __repr__(self):
 147         return "<%s: %s>" % (
 148             self.__class__.__name__,
 149             ", ".join("%s=%s" % (k, self.get_bit(n).is_set) for n, k in enumerate(self._keys)),
 150         )
 151 
 152     def __str__(self):
 153         return six.text_type(self._value)
 154 
 155     def __int__(self):
 156         return self._value
 157 
 158     def __bool__(self):
 159         return bool(self._value)
 160 
 161     __nonzero__ = __bool__
 162 
 163     def __and__(self, value):
 164         return BitHandler(self._value & int(value), self._keys)
 165 
 166     def __or__(self, value):
 167         return BitHandler(self._value | int(value), self._keys)
 168 
 169     def __add__(self, value):
 170         return BitHandler(self._value + int(value), self._keys)
 171 
 172     def __sub__(self, value):
 173         return BitHandler(self._value - int(value), self._keys)
 174 
 175     def __lshift__(self, value):
 176         return BitHandler(self._value << int(value), self._keys)
 177 
 178     def __rshift__(self, value):
 179         return BitHandler(self._value >> int(value), self._keys)
 180 
 181     def __xor__(self, value):
 182         return BitHandler(self._value ^ int(value), self._keys)
 183 
 184     def __contains__(self, key):
 185         bit_number = self._keys.index(key)
 186         return bool(self.get_bit(bit_number))
 187 
 188     def __getattr__(self, key):
 189         if key.startswith("_"):
 190             return object.__getattribute__(self, key)
 191         if key not in self._keys:
 192             raise AttributeError("%s is not a valid flag" % key)
 193         return self.get_bit(self._keys.index(key))
 194 
 195     __getitem__ = __getattr__
 196 
 197     def __setattr__(self, key, value):
 198         if key.startswith("_"):
 199             return object.__setattr__(self, key, value)
 200         if key not in self._keys:
 201             raise AttributeError("%s is not a valid flag" % key)
 202         self.set_bit(self._keys.index(key), value)
 203 
 204     __setitem__ = __setattr__
 205 
 206     def __iter__(self):
 207         return self.iteritems()  # NOQA
 208 
 209     def __sentry__(self):
 210         return repr(self)
 211 
 212     def _get_mask(self):
 213         return self._value
 214 
 215     mask = property(_get_mask)
 216 
 217     def evaluate(self, evaluator, qn, connection):
 218         return self.mask, []
 219 
 220     def get_bit(self, bit_number):
 221         mask = 2 ** int(bit_number)
 222         return Bit(bit_number, self._value & mask != 0)
 223 
 224     def set_bit(self, bit_number, true_or_false):
 225         mask = 2 ** int(bit_number)
 226         if true_or_false:
 227             self._value |= mask
 228         else:
 229             self._value &= ~mask
 230         return Bit(bit_number, self._value & mask != 0)
 231 
 232     def keys(self):
 233         return self._keys
 234 
 235     def iterkeys(self):
 236         return iter(self._keys)
 237 
 238     def items(self):
 239         return list(self.iteritems())  # NOQA
 240 
 241     def iteritems(self):
 242         for k in self._keys:
 243             yield (k, getattr(self, k).is_set)
 244 
 245     def get_label(self, flag):
 246         if isinstance(flag, string_types):
 247             flag = self._keys.index(flag)
 248         if isinstance(flag, Bit):
 249             flag = flag.number
 250         return self._labels[flag]
