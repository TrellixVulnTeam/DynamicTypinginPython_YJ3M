  10 # Count binary capacity. Truncate "0b" prefix from binary form.
  11 # Twice faster than bin(i)[2:] or math.floor(math.log(i))
  12 MAX_FLAG_COUNT = int(len(bin(BigIntegerField.MAX_BIGINT)) - 2)

  15 class BitFieldFlags(object):
  16     def __init__(self, flags):
  17         if len(flags) > MAX_FLAG_COUNT:
  18             raise ValueError("Too many flags")
  19         self._flags = flags
  20 
  21     def __repr__(self):
  22         return repr(self._flags)
  23 
  24     def __iter__(self):
  25         for flag in self._flags:
  26             yield flag
  27 
  28     def __getattr__(self, key):
  29         if key not in self._flags:
  30             raise AttributeError
  31         return Bit(self._flags.index(key))
  32 
  33     __getitem__ = __getattr__
  34 
  35     def iteritems(self):
  36         for flag in self._flags:
  37             yield flag, Bit(self._flags.index(flag))
  38 
  39     def iterkeys(self):
  40         for flag in self._flags:
  41             yield flag
  42 
  43     def itervalues(self):
  44         for flag in self._flags:
  45             yield Bit(self._flags.index(flag))
  46 
  47     def items(self):
  48         return list(self.iteritems())  # NOQA
  49 
  50     def keys(self):
  51         return list(self.iterkeys())  # NOQA
  52 
  53     def values(self):
  54         return list(self.itervalues())  # NOQA

  57 class BitFieldCreator(object):
  58     """
  59     A placeholder class that provides a way to set the attribute on the model.
  60     Descriptor for BitFields.  Checks to make sure that all flags of the
  61     instance match the class.  This is to handle the case when caching
  62     an older version of the instance and a newer version of the class is
  63     available (usually during deploys).
  64     """
  65 
  66     def __init__(self, field):
  67         self.field = field
  68 
  69     def __set__(self, obj, value):
  70         obj.__dict__[self.field.name] = self.field.to_python(value)
  71 
  72     def __get__(self, obj, type=None):
  73         if obj is None:
  74             return BitFieldFlags(self.field.flags)
  75         retval = obj.__dict__[self.field.name]
  76         if self.field.__class__ is BitField:
  77             # Update flags from class in case they've changed.
  78             retval._keys = self.field.flags
  79         return retval

  82 class BitField(BigIntegerField):
  83     def contribute_to_class(self, cls, name, **kwargs):
  84         super(BitField, self).contribute_to_class(cls, name, **kwargs)
  85         setattr(cls, self.name, BitFieldCreator(self))
  86 
  87     def __init__(self, flags, default=None, *args, **kwargs):
  88         if isinstance(flags, dict):
  89             # Get only integer keys in correct range
  90             valid_keys = (
  91                 k for k in flags.keys() if isinstance(k, int) and (0 <= k < MAX_FLAG_COUNT)
  92             )
  93             if not valid_keys:
  94                 raise ValueError("Wrong keys or empty dictionary")
  95             # Fill list with values from dict or with empty values
  96             flags = [flags.get(i, "") for i in range(max(valid_keys) + 1)]
  97 
  98         if len(flags) > MAX_FLAG_COUNT:
  99             raise ValueError("Too many flags")
 100 
 101         self._arg_flags = flags
 102         flags = list(flags)
 103         labels = []
 104         for num, flag in enumerate(flags):
 105             if isinstance(flag, (tuple, list)):
 106                 flags[num] = flag[0]
 107                 labels.append(flag[1])
 108             else:
 109                 labels.append(flag)
 110 
 111         if isinstance(default, (list, tuple, set, frozenset)):
 112             new_value = 0
 113             for flag in default:
 114                 new_value |= Bit(flags.index(flag))
 115             default = new_value
 116 
 117         BigIntegerField.__init__(self, default=default, *args, **kwargs)
 118         self.flags = flags
 119         self.labels = labels
 120 
 121     def pre_save(self, instance, add):
 122         value = getattr(instance, self.attname)
 123         return value
 124 
 125     def get_prep_value(self, value):
 126         if value is None:
 127             return None
 128         if isinstance(value, (BitHandler, Bit)):
 129             value = value.mask
 130         return int(value)
 131 
 132     def to_python(self, value):
 133         if isinstance(value, Bit):
 134             value = value.mask
 135         if not isinstance(value, BitHandler):
 136             # Regression for #1425: fix bad data that was created resulting
 137             # in negative values for flags.  Compute the value that would
 138             # have been visible ot the application to preserve compatibility.
 139             if isinstance(value, six.integer_types) and value < 0:
 140                 new_value = 0
 141                 for bit_number, _ in enumerate(self.flags):
 142                     new_value |= value & (2 ** bit_number)
 143                 value = new_value
 144 
 145             value = BitHandler(value, self.flags, self.labels)
 146         else:
 147             # Ensure flags are consistent for unpickling
 148             value._keys = self.flags
 149         return value
 150 
 151     def deconstruct(self):
 152         name, path, args, kwargs = super(BitField, self).deconstruct()
 153         args.insert(0, self._arg_flags)
 154         return name, path, args, kwargs
