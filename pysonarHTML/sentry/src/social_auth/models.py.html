  40     def __unicode__(self):
  41         """Return associated user unicode representation"""
  42         return u"%s - %s" % (six.text_type(self.user), self.provider.title())

  44     def get_backend(self):
  45         # Make import here to avoid recursive imports :-/
  46         from social_auth.backends import get_backends
  47 
  48         return get_backends().get(self.provider)

  50     @property
  51     def tokens(self):
  52         """Return access_token stored in extra_data or None"""
  53         backend = self.get_backend()
  54         if backend:
  55             return backend.AUTH_BACKEND.tokens(self)
  56         else:
  57             return {}

  59     def revoke_token(self, drop_token=True):
  60         """Attempts to revoke permissions for provider."""
  61         if "access_token" in self.tokens:
  62             success = self.get_backend().revoke_token(self.tokens["access_token"], self.uid)
  63             if success and drop_token:
  64                 self.extra_data.pop("access_token", None)
  65                 self.save()
  66 
  67     def refresh_token(self):
  68         refresh_token = self.extra_data.get("refresh_token")
  69         if refresh_token:
  70             backend = self.get_backend()
  71             if hasattr(backend, "refresh_token"):
  72                 response = backend.refresh_token(refresh_token, self.provider)
  73                 new_access_token = response.get("access_token")
  74                 # We have not got a new access token, so don't lose the
  75                 # existing one.
  76                 if not new_access_token:
  77                     return
  78                 self.extra_data["access_token"] = new_access_token
  79                 # New refresh token might be given.
  80                 new_refresh_token = response.get("refresh_token")
  81                 if new_refresh_token:
  82                     self.extra_data["refresh_token"] = new_refresh_token
  83                 self.save()
  84 
  85     def expiration_datetime(self):
  86         """Return provider session live seconds. Returns a timedelta ready to
  87         use with session.set_expiry().
  88 
  89         If provider returns a timestamp instead of session seconds to live, the
  90         timedelta is inferred from current time (using UTC timezone). None is
  91         returned if there's no value stored or it's invalid.
  92         """
  93         if self.extra_data and "expires" in self.extra_data:
  94             try:
  95                 expires = int(self.extra_data["expires"])
  96             except (ValueError, TypeError):
  97                 return None
  98 
  99             now = datetime.utcnow()
 100 
 101             # Detect if expires is a timestamp
 102             if expires > time.mktime(now.timetuple()):
 103                 # expires is a datetime
 104                 return datetime.fromtimestamp(expires) - now
 105             else:
 106                 # expires is a timedelta
 107                 return timedelta(seconds=expires)

 113     @classmethod
 114     def user_username(cls, user):
 115         if hasattr(user, "USERNAME_FIELD"):
 116             # Django 1.5 custom user model, 'username' is just for internal
 117             # use, doesn't imply that the model should have an username field
 118             field_name = user.USERNAME_FIELD
 119         else:
 120             field_name = "username"
 121         return getattr(user, field_name)
 122 
 123     @classmethod
 124     def username_field(cls, values):
 125         user_model = cls.user_model()
 126         if hasattr(user_model, "USERNAME_FIELD"):
 127             # Django 1.5 custom user model, 'username' is just for internal
 128             # use, doesn't imply that the model should have an username field
 129             values[user_model.USERNAME_FIELD] = values.pop("username")
 130         return values
