  13 class JSONField(TextField):
  14     """Simple JSON field that stores python structures as JSON strings
  15     on database.
  16     """
  17 
  18     def contribute_to_class(self, cls, name):
  19         """
  20         Add a descriptor for backwards compatibility
  21         with previous Django behavior.
  22         """
  23         super(JSONField, self).contribute_to_class(cls, name)
  24         setattr(cls, name, Creator(self))
  25 
  26     def to_python(self, value):
  27         """
  28         Convert the input JSON value into python structures, raises
  29         django.core.exceptions.ValidationError if the data can't be converted.
  30         """
  31         if self.blank and not value:
  32             return None
  33         if isinstance(value, six.string_types):
  34             try:
  35                 return simplejson.loads(value)
  36             except Exception as e:
  37                 raise ValidationError(six.text_type(e))
  38         else:
  39             return value
  40 
  41     def validate(self, value, model_instance):
  42         """Check value is a valid JSON string, raise ValidationError on
  43         error."""
  44         if isinstance(value, six.string_types):
  45             super(JSONField, self).validate(value, model_instance)
  46             try:
  47                 simplejson.loads(value)
  48             except Exception as e:
  49                 raise ValidationError(six.text_type(e))
  50 
  51     def get_prep_value(self, value):
  52         """Convert value to JSON string before save"""
  53         try:
  54             return simplejson.dumps(value)
  55         except Exception as e:
  56             raise ValidationError(six.text_type(e))
  57 
  58     def value_to_string(self, obj):
  59         """Return value from object converted to string properly"""
  60         return smart_text(self.get_prep_value(self._get_val_from_obj(obj)))
  61 
  62     def value_from_object(self, obj):
  63         """Return value dumped to string."""
  64         return self.get_prep_value(self._get_val_from_obj(obj))
