  11 def dsa_view(redirect_name=None):
  12     """Decorate djangos-social-auth views. Will check and retrieve backend
  13     or return HttpResponseServerError if backend is not found.
  14 
  15         redirect_name parameter is used to build redirect URL used by backend.
  16     """
  17 
  18     def dec(func):
  19         @wraps(func)
  20         def wrapper(request, backend, *args, **kwargs):
  21             if redirect_name:
  22                 redirect = reverse(redirect_name, args=(backend,))
  23             else:
  24                 redirect = request.path
  25             request.social_auth_backend = get_backend(backend, request, redirect)
  26             if request.social_auth_backend is None:
  27                 raise WrongBackend(backend)
  28             return func(request, request.social_auth_backend, *args, **kwargs)
  29 
  30         return wrapper
  31 
  32     return dec
