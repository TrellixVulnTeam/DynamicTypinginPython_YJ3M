  26 def sanitize_log_data(secret, data=None, leave_characters=LEAVE_CHARS):
  27     """
  28     Clean private/secret data from log statements and other data.
  29 
  30     Assumes data and secret are strings. Replaces all but the first
  31     `leave_characters` of `secret`, as found in `data`, with '*'.
  32 
  33     If no data is given, all but the first `leave_characters` of secret
  34     are simply replaced and returned.
  35     """
  36     replace_secret = secret[:leave_characters] + (len(secret) - leave_characters) * "*"
  37 
  38     if data:
  39         return data.replace(secret, replace_secret)
  40 
  41     return replace_secret

  49 def backend_setting(backend, name, default=None):
  50     """
  51     Looks for setting value following these rules:
  52         1. Search for <backend_name> prefixed setting
  53         2. Search for setting given by name
  54         3. Return default
  55     """
  56     backend_name = get_backend_name(backend)
  57     setting_name = "%s_%s" % (backend_name.upper().replace("-", "_"), name)
  58     if hasattr(settings, setting_name):
  59         return setting(setting_name)
  60     elif hasattr(settings, name):
  61         return setting(name)
  62     else:
  63         return default

  82 def model_to_ctype(val):
  83     """Converts values that are instance of Model to a dictionary
  84     with enough information to retrieve the instance back later."""
  85     if isinstance(val, Model):
  86         val = {"pk": val.pk, "ctype": ContentType.objects.get_for_model(val).pk}
  87     return val
  88 
  89 
  90 def ctype_to_model(val):
  91     """Converts back the instance saved by model_to_ctype function."""
  92     if isinstance(val, dict) and "pk" in val and "ctype" in val:
  93         ctype = ContentType.objects.get_for_id(val["ctype"])
  94         ModelClass = ctype.model_class()
  95         val = ModelClass.objects.get(pk=val["pk"])
  96     return val

  99 def clean_partial_pipeline(request):
 100     """Cleans any data for partial pipeline."""
 101     name = setting("SOCIAL_AUTH_PARTIAL_PIPELINE_KEY", "partial_pipeline")
 102     # Check for key to avoid flagging the session as modified unnecessary
 103     if name in request.session:
 104         request.session.pop(name, None)

 107 def url_add_parameters(url, params):
 108     """Adds parameters to URL, parameter will be repeated if already present"""
 109     if params:
 110         fragments = list(urlparse(url))
 111         fragments[4] = urlencode(parse_qsl(fragments[4]) + params.items())
 112         url = urlunparse(fragments)
 113     return url

 116 def dsa_urlopen(*args, **kwargs):
 117     """Like urllib2.urlopen but sets a timeout defined by
 118     SOCIAL_AUTH_URLOPEN_TIMEOUT setting if defined (and not already in
 119     kwargs)."""
 120     timeout = setting("SOCIAL_AUTH_URLOPEN_TIMEOUT")
 121     if timeout and "timeout" not in kwargs:
 122         kwargs["timeout"] = timeout
 123     return urlopen(*args, **kwargs)

 136 def parse_qs(value):

 141 def drop_lists(value):
 142     return dict((key, val[0]) for key, val in six.iteritems(value))
