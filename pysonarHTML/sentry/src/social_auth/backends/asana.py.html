  40 class AsanaAuth(BaseOAuth2):
  41     """Asana OAuth authentication mechanism"""
  42 
  43     AUTHORIZATION_URL = ASANA_AUTHORIZATION_URL
  44     ACCESS_TOKEN_URL = ASANA_TOKEN_EXCHANGE_URL
  45     AUTH_BACKEND = AsanaBackend

  49 
  50     def user_data(self, access_token, *args, **kwargs):
  51         """Loads user data from service"""
  52         headers = {"Authorization": "Bearer %s" % access_token}
  53         try:
  54             resp = requests.get(ASANA_USER_DETAILS_URL, headers=headers)
  55             resp.raise_for_status()
  56             return resp.json()["data"]
  57         except ValueError:
  58             return None
  59 
  60     def auth_complete(self, *args, **kwargs):
  61         """Completes logging process, must return user instance"""
  62         self.process_error(self.data)
  63         params = self.auth_complete_params(self.validate_state())
  64 
  65         response = requests.post(self.ACCESS_TOKEN_URL, data=params, headers=self.auth_headers())
  66         if response.status_code == 400:
  67             raise AuthCanceled(self)
  68 
  69         response.raise_for_status()
  70 
  71         try:
  72             response = response.json()
  73         except (ValueError, KeyError):
  74             raise AuthUnknownError(self)
  75 
  76         response.pop("data")
  77         self.process_error(response)
  78         return self.do_auth(response["access_token"], response=response, *args, **kwargs)
