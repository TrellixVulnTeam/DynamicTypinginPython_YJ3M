  40 class VisualStudioAuth(BaseOAuth2):
  41     """Slack OAuth authentication mechanism"""
  42 
  43     AUTHORIZATION_URL = VISUALSTUDIO_AUTHORIZATION_URL
  44     ACCESS_TOKEN_URL = VISUALSTUDIO_TOKEN_EXCHANGE_URL
  45     AUTH_BACKEND = VisualStudioBackend
  46     SETTINGS_KEY_NAME = "VISUALSTUDIO_APP_ID"
  47     SETTINGS_SECRET_NAME = "VISUALSTUDIO_APP_SECRET"
  48     SETTINGS_CLIENT_SECRET_NAME = "VISUALSTUDIO_CLIENT_SECRET"
  49     REDIRECT_STATE = False
  50     DEFAULT_SCOPE = settings.VISUALSTUDIO_SCOPES
  51     RESPONSE_TYPE = "Assertion"
  52 
  53     def user_data(self, access_token, *args, **kwargs):
  54         """Loads user data from service"""
  55         resp = requests.get(
  56             VISUALSTUDIO_USER_DETAILS_URL,
  57             headers={"Authorization": u"Bearer {}".format(access_token)},
  58         )
  59         resp.raise_for_status()
  60         content = resp.json()
  61         return {
  62             "id": content["id"],
  63             "email": content["emailAddress"],
  64             "full_name": content["displayName"],
  65         }
  66 
  67     def auth_complete_params(self, state=None):
  68         secret = setting(self.SETTINGS_CLIENT_SECRET_NAME)
  69         return {
  70             "client_assertion_type": "urn:ietf:params:oauth:client-assertion-type:jwt-bearer",
  71             "client_assertion": secret,
  72             "grant_type": "urn:ietf:params:oauth:grant-type:jwt-bearer",
  73             "assertion": self.data.get("code", ""),
  74             "redirect_uri": self.get_redirect_uri(state),
  75         }
  76 
  77     @classmethod
  78     def refresh_token_params(cls, token, provider):
  79         secret = setting(cls.SETTINGS_CLIENT_SECRET_NAME)
  80 
  81         return {
  82             "client_assertion_type": "urn:ietf:params:oauth:client-assertion-type:jwt-bearer",
  83             "client_assertion": secret,
  84             "grant_type": "refresh_token",
  85             "redirect_uri": absolute_uri(reverse("socialauth_associate_complete", args=[provider])),
  86             "assertion": token,
  87         }
