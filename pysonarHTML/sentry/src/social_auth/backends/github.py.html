   1 """
   2 GitHub OAuth support.
   3 This contribution adds support for GitHub OAuth service. The settings
   4 GITHUB_APP_ID and GITHUB_API_SECRET must be defined with the values
   5 given by GitHub application registration process.
   6 GITHUB_ORGANIZATION is an optional setting that will allow you to constrain
   7 authentication to a given GitHub organization.
   8 Extended permissions are supported by defining GITHUB_EXTENDED_PERMISSIONS
   9 setting, it must be a list of values to request.
  10 By default account id and token expiration time are stored in extra_data
  11 field, check OAuthBackend class for details on how to extend it.
  12 """
  13 from __future__ import absolute_import
  14 
  15 import simplejson
  16 
  17 from django.conf import settings
  18 from six.moves.urllib.error import HTTPError
  19 from six.moves.urllib.request import Request
  20 from social_auth.utils import dsa_urlopen
  21 from social_auth.backends import BaseOAuth2, OAuthBackend
  22 from social_auth.exceptions import AuthFailed

  25 # GitHub configuration
  26 GITHUB_AUTHORIZATION_URL = "https://github.com/login/oauth/authorize"
  27 GITHUB_ACCESS_TOKEN_URL = "https://github.com/login/oauth/access_token"
  28 GITHUB_USER_DATA_URL = "https://api.github.com/user"
  29 
  30 # GitHub organization configuration
  31 GITHUB_ORGANIZATION_MEMBER_OF_URL = "https://api.github.com/orgs/{org}/members/{username}"
  32 
  33 GITHUB_SERVER = "github.com"
  34 
  35 
  36 class GithubBackend(OAuthBackend):
  37     """Github OAuth authentication backend"""
  38 
  39     name = "github"
  40     # Default extra data to store
  41     EXTRA_DATA = [("id", "id"), ("expires", "expires")]
  42 
  43     def _fetch_emails(self, access_token):
  44         """Fetch private emails from Github account"""
  45         req = Request(
  46             GITHUB_USER_DATA_URL + "/emails", headers={"Authorization": "token %s" % access_token}
  47         )
  48 
  49         try:
  50             data = simplejson.load(dsa_urlopen(req))
  51         except (ValueError, HTTPError):
  52             data = []
  53         return data
  54 
  55     def get_user_details(self, response):
  56         """Return user details from Github account"""
  57         name = response.get("name") or ""
  58         details = {"username": response.get("login")}
  59 
  60         try:
  61             email = self._fetch_emails(response.get("access_token"))[0]
  62         except IndexError:
  63             details["email"] = ""
  64         else:
  65             details["email"] = email
  66 
  67         try:
  68             # GitHub doesn't separate first and last names. Let's try.
  69             first_name, last_name = name.split(" ", 1)
  70         except ValueError:
  71             details["first_name"] = name
  72         else:
  73             details["first_name"] = first_name
  74             details["last_name"] = last_name
  75         return details

  78 class GithubAuth(BaseOAuth2):
  79     """Github OAuth2 mechanism"""
  80 
  81     AUTHORIZATION_URL = GITHUB_AUTHORIZATION_URL
  82     ACCESS_TOKEN_URL = GITHUB_ACCESS_TOKEN_URL
  83     AUTH_BACKEND = GithubBackend
  84     SETTINGS_KEY_NAME = "GITHUB_APP_ID"
  85     SETTINGS_SECRET_NAME = "GITHUB_API_SECRET"
  86     SCOPE_SEPARATOR = ","
  87     # Look at http://developer.github.com/v3/oauth/
  88     SCOPE_VAR_NAME = "GITHUB_EXTENDED_PERMISSIONS"
  89 
  90     GITHUB_ORGANIZATION = getattr(settings, "GITHUB_ORGANIZATION", None)
  91 
  92     def user_data(self, access_token, *args, **kwargs):
  93         """Loads user data from service"""
  94         req = Request(GITHUB_USER_DATA_URL, headers={"Authorization": "token %s" % access_token})
  95 
  96         try:
  97             data = simplejson.load(dsa_urlopen(req))
  98         except ValueError:
  99             data = None
 100 
 101         # if we have a github organization defined, test that the current users
 102         # is a member of that organization.
 103         if data and self.GITHUB_ORGANIZATION:
 104             req = Request(
 105                 GITHUB_ORGANIZATION_MEMBER_OF_URL.format(
 106                     org=self.GITHUB_ORGANIZATION, username=data.get("login")
 107                 ),
 108                 headers={"Authorization": "token %s" % access_token},
 109             )
 110 
 111             try:
 112                 response = dsa_urlopen(req)
 113             except HTTPError:
 114                 data = None
 115             else:
 116                 # if the user is a member of the organization, response code
 117                 # will be 204, see http://bit.ly/ZS6vFl
 118                 if response.code != 204:
 119                     raise AuthFailed("User doesn't belong to the " "organization")
 120         return data
