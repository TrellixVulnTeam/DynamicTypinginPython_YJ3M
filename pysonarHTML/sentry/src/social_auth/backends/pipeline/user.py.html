  17 def get_username(
  18     details, user=None, user_exists=UserSocialAuth.simple_user_exists, *args, **kwargs
  19 ):
  20     """Return an username for new user. Return current user username
  21     if user was given.
  22     """
  23     if user:
  24         return {"username": UserSocialAuth.user_username(user)}
  25 
  26     email_as_username = setting("SOCIAL_AUTH_USERNAME_IS_FULL_EMAIL", False)
  27     uuid_length = setting("SOCIAL_AUTH_UUID_LENGTH", 16)
  28     do_slugify = setting("SOCIAL_AUTH_SLUGIFY_USERNAMES", False)
  29 
  30     if email_as_username and details.get("email"):
  31         username = details["email"]
  32     elif details.get("username"):
  33         username = six.text_type(details["username"])
  34     else:
  35         username = uuid4().hex
  36 
  37     max_length = UserSocialAuth.username_max_length()
  38     short_username = username[: max_length - uuid_length]
  39     final_username = UserSocialAuth.clean_username(username[:max_length])
  40     if do_slugify:
  41         final_username = slugify(final_username)
  42 
  43     # Generate a unique username for current user using username
  44     # as base but adding a unique hash at the end. Original
  45     # username is cut to avoid any field max_length.
  46     while user_exists(username=final_username):
  47         username = short_username + uuid4().hex[:uuid_length]
  48         username = username[:max_length]
  49         final_username = UserSocialAuth.clean_username(username)
  50         if do_slugify:
  51             final_username = slugify(final_username)
  52     return {"username": final_username}

  55 def create_user(backend, details, response, uid, username, user=None, *args, **kwargs):
  56     """Create user. Depends on get_username pipeline."""
  57     if user:
  58         return {"user": user}
  59     if not username:
  60         return None
  61 
  62     # Avoid hitting field max length
  63     email = details.get("email")
  64     original_email = None
  65     if email and UserSocialAuth.email_max_length() < len(email):
  66         original_email = email
  67         email = ""
  68 
  69     return {
  70         "user": UserSocialAuth.create_user(username=username, email=email),
  71         "original_email": original_email,
  72         "is_new": True,
  73     }

  76 def _ignore_field(name, is_new=False):
  77     if name in ("username", "id", "pk"):
  78         return True
  79 
  80     if not is_new and name in setting("SOCIAL_AUTH_PROTECTED_USER_FIELDS", []):
  81         return True
  82     return False

  85 def django_orm_maxlength_truncate(backend, details, user=None, is_new=False, *args, **kwargs):
  86     """Truncate any value in details that corresponds with a field in the user
  87     model. Add this entry to the pipeline before update_user_details"""
  88     if user is None:
  89         return
  90     out = {}
  91     names = get_all_field_names(user)
  92     for name, value in six.iteritems(details):
  93         if name in names and not _ignore_field(name, is_new):
  94             max_length = user._meta.get_field(name).max_length
  95             try:
  96                 if max_length and len(value) > max_length:
  97                     value = value[:max_length]
  98             except TypeError:
  99                 pass
 100         out[name] = value
 101     return {"details": out}

 104 def update_user_details(backend, details, response, user=None, is_new=False, *args, **kwargs):
 105     """Update user details using data from provider."""
 106     if user is None:
 107         return
 108 
 109     changed = False  # flag to track changes
 110 
 111     for name, value in six.iteritems(details):
 112         # do not update username, it was already generated, do not update
 113         # configured fields if user already existed
 114         if not _ignore_field(name, is_new):
 115             if value and value != getattr(user, name, None):
 116                 setattr(user, name, value)
 117                 changed = True
 118 
 119     if changed:
 120         user.save()
