  18     if social_user or not user:

  32 def load_extra_data(backend, details, response, uid, user, social_user=None, *args, **kwargs):
  33     """Load extra data from provider and store it on current UserSocialAuth
  34     extra_data field.
  35     """
  36     social_user = social_user or UserSocialAuth.get_social_auth(backend.name, uid, user)
  37     if social_user:
  38         extra_data = backend.extra_data(user, uid, response, details)
  39         if kwargs.get("original_email") and "email" not in extra_data:
  40             extra_data["email"] = kwargs.get("original_email")
  41         if extra_data and social_user.extra_data != extra_data:
  42             if social_user.extra_data:
  43                 social_user.extra_data.update(extra_data)
  44             else:
  45                 social_user.extra_data = extra_data
  46             social_user.save()
  47         return {"social_user": social_user}
