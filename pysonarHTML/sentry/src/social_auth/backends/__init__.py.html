  68 class SocialAuthBackend(object):
  69     """A django.contrib.auth backend that authenticates the user based on
  70     a authentication provider response"""
  71 
  72     name = ""  # provider name, it's stored in database
  73     supports_inactive_user = False
  74 
  75     def authenticate(self, request, *args, **kwargs):
  76         """Authenticate user using social credentials
  77 
  78         Authentication is made if this is the correct backend, backend
  79         verification is made by kwargs inspection for current backend
  80         name presence.
  81         """
  82         # Validate backend and arguments. Require that the Social Auth
  83         # response be passed in as a keyword argument, to make sure we
  84         # don't match the username/password calling conventions of
  85         # authenticate.
  86         if not (self.name and kwargs.get(self.name) and "response" in kwargs):
  87             return None
  88 
  89         response = kwargs.get("response")
  90         pipeline = PIPELINE
  91         kwargs = kwargs.copy()
  92         kwargs["backend"] = self
  93 
  94         if "pipeline_index" in kwargs:
  95             pipeline = pipeline[kwargs["pipeline_index"] :]
  96         else:
  97             kwargs["details"] = self.get_user_details(response)
  98             kwargs["uid"] = self.get_user_id(kwargs["details"], response)
  99             kwargs["is_new"] = False
 100 
 101         out = self.pipeline(pipeline, request, *args, **kwargs)
 102         if not isinstance(out, dict):
 103             return out
 104 
 105         social_user = out.get("social_user")
 106         if social_user:
 107             # define user.social_user attribute to track current social
 108             # account
 109             user = social_user.user
 110             user.social_user = social_user
 111             user.is_new = out.get("is_new")
 112             return user
 113 
 114     def pipeline(self, pipeline, request, *args, **kwargs):
 115         """Pipeline"""
 116         out = kwargs.copy()
 117 
 118         if "pipeline_index" in kwargs:
 119             base_index = int(kwargs["pipeline_index"])
 120         else:
 121             base_index = 0
 122 
 123         for idx, name in enumerate(pipeline):
 124             out["pipeline_index"] = base_index + idx
 125             mod_name, func_name = name.rsplit(".", 1)
 126             mod = __import__(mod_name, {}, {}, [func_name])
 127             func = getattr(mod, func_name, None)
 128 
 129             try:
 130                 result = {}
 131                 if func_name == "save_status_to_session":
 132                     result = func(request, *args, **out) or {}
 133                 else:
 134                     result = func(*args, **out) or {}
 135             except StopPipeline:
 136                 # Clean partial pipeline on stop
 137                 if "request" in kwargs:
 138                     clean_partial_pipeline(kwargs["request"])
 139                 break
 140 
 141             if isinstance(result, dict):
 142                 out.update(result)
 143             else:
 144                 return result
 145 
 146         # clean the partial pipeline at the end of the process
 147         if "request" in kwargs:
 148             clean_partial_pipeline(kwargs["request"])
 149         return out

 190 class OAuthBackend(SocialAuthBackend):
 191     """OAuth authentication backend base class.
 192 
 193     EXTRA_DATA defines a set of name that will be stored in
 194                extra_data field. It must be a list of tuples with
 195                name and alias.
 196 
 197     Also settings will be inspected to get more values names that should be
 198     stored on extra_data field. Setting name is created from current backend
 199     name (all uppercase) plus _EXTRA_DATA.
 200 
 201     access_token is always stored.
 202     """
 203 
 204     EXTRA_DATA = None
 205     ID_KEY = "id"
 206 
 207     def get_user_id(self, details, response):
 208         """OAuth providers return an unique user id in response"""
 209         return response[self.ID_KEY]
 210 
 211     @classmethod
 212     def extra_data(cls, user, uid, response, details=None):
 213         """Return access_token and extra defined names to store in
 214         extra_data field"""
 215         data = {"access_token": response.get("access_token", "")}
 216         name = cls.name.replace("-", "_").upper()
 217         names = (cls.EXTRA_DATA or []) + setting(name + "_EXTRA_DATA", [])
 218 
 219         for entry in names:
 220             if isinstance(entry, six.string_types):
 221                 entry = (entry,)
 222 
 223             try:
 224                 if len(entry) == 3:
 225                     name, alias, discard = entry
 226                 elif len(entry) == 2:
 227                     (name, alias), discard = entry, False
 228                 elif len(entry) == 1:
 229                     (name,), (alias,), discard = entry, entry, False
 230                 else:
 231                     raise ValueError("invalid tuple for EXTRA_DATA entry" % entry)
 232 
 233                 value = response.get(name)
 234                 if discard and not value:
 235                     continue
 236                 data[alias] = value
 237 
 238             except (TypeError, ValueError):
 239                 raise BackendError("invalid entry: %s" % (entry,))
 240 
 241         return data

 273     def to_session_dict(self, next_idx, *args, **kwargs):
 274         """Returns dict to store on session for partial pipeline."""
 275         return {
 276             "next": next_idx,
 277             "backend": self.AUTH_BACKEND.name,
 278             "args": tuple(map(model_to_ctype, args)),
 279             "kwargs": dict((key, model_to_ctype(val)) for key, val in six.iteritems(kwargs)),
 280         }
 281 
 282     def from_session_dict(self, session_data, *args, **kwargs):
 283         """Takes session saved data to continue pipeline and merges with any
 284         new extra argument needed. Returns tuple with next pipeline index
 285         entry, arguments and keyword arguments to continue the process."""
 286         args = args[:] + tuple(map(ctype_to_model, session_data["args"]))
 287 
 288         kwargs = kwargs.copy()
 289         saved_kwargs = dict(
 290             (key, ctype_to_model(val)) for key, val in six.iteritems(session_data["kwargs"])
 291         )
 292         saved_kwargs.update((key, val) for key, val in six.iteritems(kwargs))
 293         return (session_data["next"], args, saved_kwargs)
 294 
 295     def continue_pipeline(self, *args, **kwargs):
 296         """Continue previous halted pipeline"""
 297         kwargs.update({"auth": self, self.AUTH_BACKEND.name: True})
 298         return authenticate(*args, **kwargs)

 300     def request_token_extra_arguments(self):
 301         """Return extra arguments needed on request-token process,
 302         setting is per backend and defined by:
 303             <backend name in uppercase>_REQUEST_TOKEN_EXTRA_ARGUMENTS.
 304         """
 305         backend_name = self.AUTH_BACKEND.name.upper().replace("-", "_")
 306         return setting(backend_name + "_REQUEST_TOKEN_EXTRA_ARGUMENTS", {})
 307 
 308     def auth_extra_arguments(self):
 309         """Return extra arguments needed on auth process, setting is per
 310         backend and defined by:
 311             <backend name in uppercase>_AUTH_EXTRA_ARGUMENTS.
 312         The defaults can be overridden by GET parameters.
 313         """
 314         backend_name = self.AUTH_BACKEND.name.upper().replace("-", "_")
 315         extra_arguments = setting(backend_name + "_AUTH_EXTRA_ARGUMENTS", {})
 316         for key, value in six.iteritems(extra_arguments):
 317             if key in self.data:
 318                 extra_arguments[key] = self.data[key]
 319             elif value:
 320                 extra_arguments[key] = value
 321         return extra_arguments

 334     def disconnect(self, user, association_id=None):
 335         """Deletes current backend from user if associated.
 336         Override if extra operations are needed.
 337         """
 338         name = self.AUTH_BACKEND.name
 339         do_revoke = setting("SOCIAL_AUTH_REVOKE_TOKENS_ON_DISCONNECT")
 340         filter_args = {}
 341 
 342         if association_id:
 343             filter_args["id"] = association_id
 344         else:
 345             filter_args["provider"] = name
 346         instances = UserSocialAuth.get_social_auth_for_user(user).filter(**filter_args)
 347 
 348         if do_revoke:
 349             for instance in instances:
 350                 instance.revoke_token(drop_token=False)
 351         instances.delete()
 352 
 353     def build_absolute_uri(self, path=None):
 354         """Build absolute URI for given path. Replace http:// schema with
 355         https:// if SOCIAL_AUTH_REDIRECT_IS_HTTPS is defined.
 356         """
 357         uri = self.request.build_absolute_uri(path)
 358         if setting("SOCIAL_AUTH_REDIRECT_IS_HTTPS"):
 359             uri = uri.replace("http://", "https://")
 360         return uri

 402         return param

 409 class BaseOAuth1(OAuthAuth):
 410     """Consumer based mechanism OAuth authentication, fill the needed
 411     parameters to communicate properly with authentication service.
 412 
 413         AUTHORIZATION_URL       Authorization service url
 414         REQUEST_TOKEN_URL       Request token URL
 415         ACCESS_TOKEN_URL        Access token URL
 416     """
 417 
 418     AUTHORIZATION_URL = ""
 419     REQUEST_TOKEN_URL = ""
 420     ACCESS_TOKEN_URL = ""
 421 
 422     def auth_url(self):
 423         """Return redirect url"""
 424         token = self.unauthorized_token()
 425         name = self.AUTH_BACKEND.name + "unauthorized_token_name"
 426         if not isinstance(self.request.session.get(name), list):
 427             self.request.session[name] = []
 428         self.request.session[name].append(token.to_string())
 429         self.request.session.modified = True
 430         return self.oauth_authorization_request(token)
 431 
 432     def auth_complete(self, *args, **kwargs):
 433         """Return user, might be logged in"""
 434         # Multiple unauthorized tokens are supported (see #521)
 435         name = self.AUTH_BACKEND.name + "unauthorized_token_name"
 436         token = None
 437         unauthed_tokens = self.request.session.get(name) or []
 438         if not unauthed_tokens:
 439             raise AuthTokenError(self, "Missing unauthorized token")
 440         for unauthed_token in unauthed_tokens:
 441             token = unauthed_token
 442             if not isinstance(unauthed_token, dict):
 443                 token = parse_qs(unauthed_token)
 444             if token.get("oauth_token") == self.data.get("oauth_token"):
 445                 unauthed_tokens = list(set(unauthed_tokens) - set([unauthed_token]))
 446                 self.request.session[name] = unauthed_tokens
 447                 self.request.session.modified = True
 448                 break
 449         else:
 450             raise AuthTokenError(self, "Incorrect tokens")
 451 
 452         try:
 453             access_token = self.access_token(token)
 454         except HTTPError as e:
 455             if e.code == 400:
 456                 raise AuthCanceled(self)
 457             else:
 458                 raise
 459         return self.do_auth(access_token, *args, **kwargs)
 460 
 461     def do_auth(self, access_token, *args, **kwargs):
 462         """Finish the auth process once the access_token was retrieved"""
 463         data = self.user_data(access_token)
 464         if data is not None:
 465             data["access_token"] = access_token.to_string()
 466 
 467         kwargs.update({"auth": self, "response": data, self.AUTH_BACKEND.name: True})
 468         return authenticate(*args, **kwargs)
 469 
 470     def unauthorized_token(self):
 471         """Return request for unauthorized token (first stage)"""
 472         params = self.request_token_extra_arguments()
 473         params.update(self.get_scope_argument())
 474         key, secret = self.get_key_and_secret()
 475         response = self.request(
 476             url=self.REQUEST_TOKEN_URL,
 477             params=params,
 478             auth=OAuth1(key, secret, callback_uri=self.redirect_uri),
 479         )
 480         return response.content
 481 
 482     def oauth_authorization_request(self, token):
 483         """Generate OAuth request to authorize token."""
 484         if not isinstance(token, dict):
 485             token = parse_qs(token)
 486         params = self.auth_extra_arguments() or {}
 487         params.update(self.get_scope_argument())
 488         params["oauth_token"] = token.get("oauth_token")
 489         params["redirect_uri"] = self.redirect_uri
 490         return self.AUTHORIZATION_URL + "?" + urlencode(params)
 491 
 492     def oauth_auth(self, token=None, oauth_verifier=None):
 493         key, secret = self.get_key_and_secret()
 494         oauth_verifier = oauth_verifier or self.data.get("oauth_verifier")
 495         token = token or {}
 496         return OAuth1(
 497             key,
 498             secret,
 499             resource_owner_key=token.get("oauth_token"),
 500             resource_owner_secret=token.get("oauth_token_secret"),
 501             callback_uri=self.redirect_uri,
 502             verifier=oauth_verifier,
 503         )
 504 
 505     def oauth_request(self, token, url, extra_params=None, method="GET"):
 506         """Generate OAuth request, setups callback url"""
 507         return self.request(url, auth=self.oauth_auth(token))
 508 
 509     def fetch_response(self, request):
 510         """Executes request and fetches service response"""
 511         response = dsa_urlopen(request.to_url())
 512         return "\n".join(response.readlines())
 513 
 514     def access_token(self, token):
 515         """Return request for access token value"""
 516         return self.get_querystring(self.ACCESS_TOKEN_URL, auth=self.oauth_auth(token))

 519 class BaseOAuth2(OAuthAuth):
 520     """Base class for OAuth2 providers.
 521 
 522     OAuth2 draft details at:
 523         http://tools.ietf.org/html/draft-ietf-oauth-v2-10
 524 
 525     Attributes:
 526         AUTHORIZATION_URL       Authorization service url
 527         ACCESS_TOKEN_URL        Token URL
 528     """
 529 
 530     AUTHORIZATION_URL = None
 531     ACCESS_TOKEN_URL = None
 532     REFRESH_TOKEN_URL = None
 533     REVOKE_TOKEN_URL = None
 534     REVOKE_TOKEN_METHOD = "POST"
 535     RESPONSE_TYPE = "code"
 536     REDIRECT_STATE = True
 537     STATE_PARAMETER = True
 538 
 539     def state_token(self):
 540         """Generate csrf token to include as state parameter."""
 541         return get_random_string(32)
 542 
 543     def get_redirect_uri(self, state=None):
 544         """Build redirect_uri with redirect_state parameter."""
 545         uri = self.redirect_uri
 546         if self.REDIRECT_STATE and state:
 547             uri = url_add_parameters(uri, {"redirect_state": state})
 548         return uri
 549 
 550     def auth_params(self, state=None):
 551         client_id, client_secret = self.get_key_and_secret()
 552         params = {"client_id": client_id, "redirect_uri": self.get_redirect_uri(state)}
 553         if self.STATE_PARAMETER and state:
 554             params["state"] = state
 555         if self.RESPONSE_TYPE:
 556             params["response_type"] = self.RESPONSE_TYPE
 557         return params
 558 
 559     def auth_url(self):
 560         """Return redirect url"""
 561         if self.STATE_PARAMETER or self.REDIRECT_STATE:
 562             # Store state in session for further request validation. The state
 563             # value is passed as state parameter (as specified in OAuth2 spec),
 564             # but also added to redirect_uri, that way we can still verify the
 565             # request if the provider doesn't implement the state parameter.
 566             # Reuse token if any.
 567             name = self.AUTH_BACKEND.name + "_state"
 568             state = self.request.session.get(name) or self.state_token()
 569             self.request.session[self.AUTH_BACKEND.name + "_state"] = state
 570         else:
 571             state = None
 572 
 573         params = self.auth_params(state)
 574         params.update(self.get_scope_argument())
 575         params.update(self.auth_extra_arguments())
 576 
 577         if self.request.META.get("QUERY_STRING"):
 578             query_string = "&" + self.request.META["QUERY_STRING"]
 579         else:
 580             query_string = ""
 581         return self.AUTHORIZATION_URL + "?" + urlencode(params) + query_string
 582 
 583     def validate_state(self):
 584         """Validate state value. Raises exception on error, returns state
 585         value if valid."""
 586         if not self.STATE_PARAMETER and not self.REDIRECT_STATE:
 587             return None
 588         state = self.request.session.get(self.AUTH_BACKEND.name + "_state")
 589         if state:
 590             request_state = self.data.get("state") or self.data.get("redirect_state")
 591             if not request_state:
 592                 raise AuthMissingParameter(self, "state")
 593             elif not state:
 594                 raise AuthStateMissing(self, "state")
 595             elif not constant_time_compare(request_state, state):
 596                 raise AuthStateForbidden(self)
 597         return state
 598 
 599     def process_error(self, data):
 600         error = data.get("error_description") or data.get("error")
 601         if error:
 602             raise AuthFailed(self, error)
 603 
 604     def auth_complete_params(self, state=None):
 605         client_id, client_secret = self.get_key_and_secret()
 606         return {
 607             "grant_type": "authorization_code",  # request auth code
 608             "code": self.data.get("code", ""),  # server response code
 609             "client_id": client_id,
 610             "client_secret": client_secret,
 611             "redirect_uri": self.get_redirect_uri(state),
 612         }
 613 
 614     @classmethod
 615     def auth_headers(cls):
 616         return {"Content-Type": "application/x-www-form-urlencoded", "Accept": "application/json"}
 617 
 618     def auth_complete(self, *args, **kwargs):
 619         """Completes logging process, must return user instance"""
 620         self.process_error(self.data)
 621         params = self.auth_complete_params(self.validate_state())
 622         request = Request(
 623             self.ACCESS_TOKEN_URL, data=urlencode(params), headers=self.auth_headers()
 624         )
 625 
 626         try:
 627             response = json.loads(dsa_urlopen(request).read())
 628         except HTTPError as e:
 629             logger.exception(
 630                 "plugins.auth.error",
 631                 extra={"class": type(self), "status_code": e.code, "response": e.read()[:128]},
 632             )
 633             raise AuthUnknownError(self)
 634         except (ValueError, KeyError):
 635             raise AuthUnknownError(self)
 636 
 637         self.process_error(response)
 638         return self.do_auth(response["access_token"], response=response, *args, **kwargs)
 639 
 640     @classmethod
 641     def refresh_token_params(cls, token, provider):
 642         client_id, client_secret = cls.get_key_and_secret()
 643         return {
 644             "refresh_token": token,
 645             "grant_type": "refresh_token",
 646             "client_id": client_id,
 647             "client_secret": client_secret,
 648         }
 649 
 650     @classmethod
 651     def refresh_token(cls, token, provider):
 652         params = cls.refresh_token_params(token, provider)
 653         response = requests.post(
 654             cls.REFRESH_TOKEN_URL or cls.ACCESS_TOKEN_URL, data=params, headers=cls.auth_headers()
 655         )
 656         response.raise_for_status()
 657         return response.json()
 658 
 659     @classmethod
 660     def revoke_token_params(cls, token, uid):
 661         return None
 662 
 663     @classmethod
 664     def revoke_token_headers(cls, token, uid):
 665         return None
 666 
 667     @classmethod
 668     def process_revoke_token_response(cls, response):
 669         return response.code == 200
 670 
 671     @classmethod
 672     def revoke_token(cls, token, uid):
 673         if not cls.REVOKE_TOKEN_URL:
 674             return
 675         url = cls.REVOKE_TOKEN_URL.format(token=token, uid=uid)
 676         params = cls.revoke_token_params(token, uid) or {}
 677         headers = cls.revoke_token_headers(token, uid) or {}
 678         data = None
 679 
 680         if cls.REVOKE_TOKEN_METHOD == "GET":
 681             url = u"{}?{}".format(url, urlencode(params))
 682         else:
 683             data = urlencode(params)
 684 
 685         request = Request(url, data=data, headers=headers)
 686         if cls.REVOKE_TOKEN_URL.lower() not in ("get", "post"):
 687             # Patch get_method to return the needed method
 688             request.get_method = lambda: cls.REVOKE_TOKEN_METHOD
 689         response = dsa_urlopen(request)
 690         return cls.process_revoke_token_response(response)
 691 
 692     def do_auth(self, access_token, *args, **kwargs):
 693         """Finish the auth process once the access_token was retrieved"""
 694         data = self.user_data(access_token, *args, **kwargs)
 695         response = kwargs.get("response") or {}
 696         response.update(data or {})
 697         kwargs.update({"auth": self, "response": response, self.AUTH_BACKEND.name: True})
 698         return authenticate(*args, **kwargs)

 701 # Cache for discovered backends.
 702 BACKENDSCACHE = {}
 703 
 704 _import_lock = threading.Lock()
 705 
 706 
 707 def get_backends(force_load=False):
 708     """
 709     Entry point to the BACKENDS cache. If BACKENDSCACHE hasn't been
 710     populated, each of the modules referenced in
 711     AUTHENTICATION_BACKENDS is imported and checked for a BACKENDS
 712     definition and if enabled, added to the cache.
 713 
 714     Previously all backends were attempted to be loaded at
 715     import time of this module, which meant that backends that subclass
 716     bases found in this module would not have the chance to be loaded
 717     by the time they were added to this module's BACKENDS dict. See:
 718     https://github.com/omab/django-social-auth/issues/204
 719 
 720     This new approach ensures that backends are allowed to subclass from
 721     bases in this module and still be picked up.
 722 
 723     A force_load boolean arg is also provided so that get_backend
 724     below can retry a requested backend that may not yet be discovered.
 725     """
 726     global BACKENDSCACHE
 727 
 728     if not BACKENDSCACHE or force_load:
 729         with _import_lock:
 730             for auth_backend in setting("AUTHENTICATION_BACKENDS"):
 731                 mod, cls_name = auth_backend.rsplit(".", 1)
 732                 module = __import__(mod, {}, {}, ["BACKENDS", cls_name])
 733                 backend = getattr(module, cls_name)
 734 
 735                 if issubclass(backend, SocialAuthBackend):
 736                     name = backend.name
 737                     backends = getattr(module, "BACKENDS", {})
 738                     if name in backends and backends[name].enabled():
 739                         BACKENDSCACHE[name] = backends[name]
 740     return BACKENDSCACHE
 741 
 742 
 743 def get_backend(name, *args, **kwargs):
 744     get_backends()
 745 
 746     try:
 747         # Cached backend which has previously been discovered.
 748         backend_cls = BACKENDSCACHE[name]
 749     except KeyError:
 750         return None
 751     else:
 752         return backend_cls(*args, **kwargs)
