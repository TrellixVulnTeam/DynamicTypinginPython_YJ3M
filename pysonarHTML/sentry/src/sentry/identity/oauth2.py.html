  26 class OAuth2Provider(Provider):
  27     """
  28     The OAuth2Provider is a generic way to implement an identity provider that
  29     uses the OAuth 2.0 protocol as a means for authenticating a user.
  30 
  31     OAuth scopes are configured through the oauth_scopes class property,
  32     however may be overridden using the ``config['oauth_scopes']`` object.
  33     """
  34 
  35     oauth_access_token_url = ""
  36     oauth_authorize_url = ""
  37     refresh_token_url = ""
  38 
  39     oauth_scopes = ()
  40 
  41     def _get_oauth_parameter(self, parameter_name):
  42         """
  43         Lookup an OAuth parameter for the provider. Depending on the context of the
  44         pipeline using the provider, the parameter may come from 1 of 3 places:
  45 
  46         1. Check the class property of the provider for the parameter.
  47 
  48         2. If the provider has the parameters made available within the ``config``.
  49 
  50         3. If provided, check the pipeline's ``provider_model`` for the oauth parameter
  51            in the config field.
  52 
  53         If the parameter cannot be found a KeyError will be raised.
  54         """
  55         try:
  56             prop = getattr(self, u"oauth_{}".format(parameter_name))
  57             if prop != "":
  58                 return prop
  59         except AttributeError:
  60             pass
  61 
  62         if self.config.get(parameter_name):
  63             return self.config.get(parameter_name)
  64 
  65         model = self.pipeline.provider_model
  66         if model and model.config.get(parameter_name) is not None:
  67             return model.config.get(parameter_name)
  68 
  69         raise KeyError(u'Unable to resolve OAuth parameter "{}"'.format(parameter_name))

 106     def get_refresh_token_params(self, refresh_token, *args, **kwargs):
 107         return {
 108             "client_id": self.get_client_id(),
 109             "client_secret": self.get_client_secret(),
 110             "grant_type": "refresh_token",
 111             "refresh_token": refresh_token,

 114     def get_oauth_data(self, payload):
 115         data = {"access_token": payload["access_token"]}
 116         if "expires_in" in payload:
 117             data["expires"] = int(time()) + int(payload["expires_in"])
 118         if "refresh_token" in payload:
 119             data["refresh_token"] = payload["refresh_token"]
 120         if "token_type" in payload:
 121             data["token_type"] = payload["token_type"]
 122 
 123         return data
 124 
 125     def handle_refresh_error(self, req, payload):
 126         error_name = "unknown_error"
 127         error_description = "no description available"
 128         for name_key in ["error", "Error"]:
 129             if name_key in payload:
 130                 error_name = payload.get(name_key)
 131                 break
 132 
 133         for desc_key in ["error_description", "ErrorDescription"]:
 134             if desc_key in payload:
 135                 error_description = payload.get(desc_key)
 136                 break
 137 
 138         formatted_error = u"HTTP {} ({}): {}".format(req.status_code, error_name, error_description)
 139 
 140         if req.status_code == 401:
 141             self.logger.info(
 142                 "identity.oauth.refresh.identity-not-valid-error",
 143                 extra={
 144                     "error_name": error_name,
 145                     "error_status_code": req.status_code,
 146                     "error_description": error_description,
 147                     "provider_key": self.key,
 148                 },
 149             )
 150             raise IdentityNotValid(formatted_error)
 151 
 152         if req.status_code == 400:
 153             # this may not be common, but at the very least Google will return
 154             # an invalid grant when a user is suspended
 155             if error_name == "invalid_grant":
 156                 self.logger.info(
 157                     "identity.oauth.refresh.identity-not-valid-error",
 158                     extra={
 159                         "error_name": error_name,
 160                         "error_status_code": req.status_code,
 161                         "error_description": error_description,
 162                         "provider_key": self.key,
 163                     },
 164                 )
 165                 raise IdentityNotValid(formatted_error)
 166 
 167         if req.status_code != 200:
 168             self.logger.info(
 169                 "identity.oauth.refresh.api-error",
 170                 extra={
 171                     "error_name": error_name,
 172                     "error_status_code": req.status_code,
 173                     "error_description": error_description,
 174                     "provider_key": self.key,
 175                 },
 176             )
 177             raise ApiError(formatted_error)
 178 
 179     def refresh_identity(self, identity, *args, **kwargs):
 180         refresh_token = identity.data.get("refresh_token")
 181 
 182         if not refresh_token:
 183             raise IdentityNotValid("Missing refresh token")
 184 
 185         data = self.get_refresh_token_params(refresh_token, *args, **kwargs)
 186 
 187         req = safe_urlopen(
 188             url=self.get_refresh_token_url(), headers=self.get_refresh_token_headers(), data=data
 189         )
 190 
 191         try:
 192             body = safe_urlread(req)
 193             payload = json.loads(body)
 194         except Exception:
 195             payload = {}
 196 
 197         self.handle_refresh_error(req, payload)
 198 
 199         identity.data.update(self.get_oauth_data(payload))
 200         return identity.update(data=identity.data)

 203 class OAuth2LoginView(PipelineView):
 204     authorize_url = None
 205     client_id = None
 206     scope = ""
 207 
 208     def __init__(self, authorize_url=None, client_id=None, scope=None, *args, **kwargs):
 209         super(OAuth2LoginView, self).__init__(*args, **kwargs)
 210         if authorize_url is not None:
 211             self.authorize_url = authorize_url
 212         if client_id is not None:
 213             self.client_id = client_id
 214         if scope is not None:
 215             self.scope = scope
 216 
 217     def get_scope(self):
 218         return self.scope
 219 
 220     def get_authorize_url(self):
 221         return self.authorize_url
 222 
 223     def get_authorize_params(self, state, redirect_uri):
 224         return {
 225             "client_id": self.client_id,
 226             "response_type": "code",
 227             "scope": self.get_scope(),
 228             "state": state,
 229             "redirect_uri": redirect_uri,
 230         }
 231 
 232     @csrf_exempt
 233     def dispatch(self, request, pipeline):
 234         for param in ("code", "error", "state"):
 235             if param in request.GET:
 236                 return pipeline.next_step()
 237 
 238         state = uuid4().hex
 239 
 240         params = self.get_authorize_params(
 241             state=state, redirect_uri=absolute_uri(pipeline.redirect_url())
 242         )
 243         redirect_uri = u"{}?{}".format(self.get_authorize_url(), urlencode(params))
 244 
 245         pipeline.bind_state("state", state)
 246 
 247         return self.redirect(redirect_uri)

 250 class OAuth2CallbackView(PipelineView):
 251     access_token_url = None
 252     client_id = None
 253     client_secret = None
 254 
 255     def __init__(self, access_token_url=None, client_id=None, client_secret=None, *args, **kwargs):
 256         super(OAuth2CallbackView, self).__init__(*args, **kwargs)
 257         if access_token_url is not None:
 258             self.access_token_url = access_token_url
 259         if client_id is not None:
 260             self.client_id = client_id
 261         if client_secret is not None:
 262             self.client_secret = client_secret
 263 
 264     def get_token_params(self, code, redirect_uri):
 265         return {
 266             "grant_type": "authorization_code",
 267             "code": code,
 268             "redirect_uri": redirect_uri,
 269             "client_id": self.client_id,
 270             "client_secret": self.client_secret,
 271         }
 272 
 273     def exchange_token(self, request, pipeline, code):
 274         # TODO: this needs the auth yet
 275         data = self.get_token_params(code=code, redirect_uri=absolute_uri(pipeline.redirect_url()))
 276         verify_ssl = pipeline.config.get("verify_ssl", True)
 277         try:
 278             req = safe_urlopen(self.access_token_url, data=data, verify_ssl=verify_ssl)
 279             body = safe_urlread(req)
 280             if req.headers.get("Content-Type", "").startswith("application/x-www-form-urlencoded"):
 281                 return dict(parse_qsl(body))
 282             return json.loads(body)
 283         except SSLError:
 284             logger.info(
 285                 "identity.oauth2.ssl-error",
 286                 extra={"url": self.access_token_url, "verify_ssl": verify_ssl},
 287             )
 288             url = self.access_token_url
 289             return {
 290                 "error": "Could not verify SSL certificate",
 291                 "error_description": u"Ensure that {} has a valid SSL certificate".format(url),
 292             }
 293         except JSONDecodeError:
 294             logger.info("identity.oauth2.json-error", extra={"url": self.access_token_url})
 295             return {
 296                 "error": "Could not decode a JSON Response",
 297                 "error_description": u"We were not able to parse a JSON response, please try again.",
 298             }
 299 
 300     def dispatch(self, request, pipeline):
 301         error = request.GET.get("error")
 302         state = request.GET.get("state")
 303         code = request.GET.get("code")
 304 
 305         if error:
 306             pipeline.logger.info("identity.token-exchange-error", extra={"error": error})
 307             return pipeline.error(error)
 308 
 309         if state != pipeline.fetch_state("state"):
 310             pipeline.logger.info(
 311                 "identity.token-exchange-error",
 312                 extra={
 313                     "error": "invalid_state",
 314                     "state": state,
 315                     "pipeline_state": pipeline.fetch_state("state"),
 316                     "code": code,
 317                 },
 318             )
 319             return pipeline.error(ERR_INVALID_STATE)
 320 
 321         data = self.exchange_token(request, pipeline, code)
 322 
 323         if "error_description" in data:
 324             error = data.get("error")
 325             pipeline.logger.info("identity.token-exchange-error", extra={"error": error})
 326             return pipeline.error(data["error_description"])
 327 
 328         if "error" in data:
 329             pipeline.logger.info("identity.token-exchange-error", extra={"error": data["error"]})
 330             return pipeline.error("Failed to retrieve token from the upstream service.")
 331 
 332         # we can either expect the API to be implicit and say "im looking for
 333         # blah within state data" or we need to pass implementation + call a
 334         # hook here
 335         pipeline.bind_state("data", data)
 336 
 337         return pipeline.next_step()
