   9 def get_user_info(access_token):
  10     session = http.build_session()
  11     resp = session.get(
  12         "https://app.vssps.visualstudio.com/_apis/profile/profiles/me?api-version=1.0",
  13         headers={"Accept": "application/json", "Authorization": "bearer %s" % access_token},
  14     )
  15     resp.raise_for_status()
  16     user = resp.json()
  17     user["uuid"] = user["id"]
  18 
  19     resp = session.get(
  20         "https://app.vssps.visualstudio.com/_apis/connectionData/",
  21         headers={"Accept": "application/json", "Authorization": "bearer %s" % access_token},
  22     )
  23     resp.raise_for_status()
  24     # NOTE (from Microsoft PM):
  25     # The "descriptor" is the universal identifier for a given user and is consistent across
  26     # all VSTS accounts (organizations). The "id" field for the same user can be different for
  27     # the same user in different places, so the "descriptor" is the best identifier for a user.
  28     # This is returned in most/all of the VSTS REST APIs at this point (except for the
  29     # profiles/me API above). To get the current user's descriptor, we call the "connection data"
  30     # REST API (this assumes we are authenticating with an access token issued to the user).
  31     # We will also see descriptors returned for every user in the "Get users" (Graph) REST API.
  32     user["id"] = resp.json()["authenticatedUser"]["subjectDescriptor"]
  33 
  34     return user

  37 class VSTSIdentityProvider(OAuth2Provider):
  38     key = "vsts"
  39     name = "Azure DevOps"
  40 
  41     oauth_access_token_url = "https://app.vssps.visualstudio.com/oauth2/token"
  42     oauth_authorize_url = "https://app.vssps.visualstudio.com/oauth2/authorize"
  43     oauth_scopes = ("vso.code", "vso.graph", "vso.serviceendpoint_manage", "vso.work_write")

  45     def get_oauth_client_id(self):
  46         return options.get("vsts.client-id")
  47 
  48     def get_oauth_client_secret(self):
  49         return options.get("vsts.client-secret")
  50 
  51     def get_refresh_token_url(self):
  52         return self.oauth_access_token_url
  53 
  54     def get_pipeline_views(self):
  55         return [
  56             OAuth2LoginView(
  57                 authorize_url=self.oauth_authorize_url,
  58                 client_id=self.get_oauth_client_id(),
  59                 scope=" ".join(self.get_oauth_scopes()),
  60             ),
  61             VSTSOAuth2CallbackView(
  62                 access_token_url=self.oauth_access_token_url,
  63                 client_id=self.get_oauth_client_id(),
  64                 client_secret=self.get_oauth_client_secret(),
  65             ),
  66         ]
  67 
  68     def get_refresh_token_headers(self):
  69         return {"Content-Type": "application/x-www-form-urlencoded", "Content-Length": "1654"}
  70 
  71     def get_refresh_token_params(self, refresh_token, *args, **kwargs):
  72         oauth_redirect_url = kwargs.get("redirect_url")
  73         if oauth_redirect_url is None:
  74             raise ValueError("VSTS requires oauth redirect url when refreshing identity")
  75         return {
  76             "client_assertion_type": "urn:ietf:params:oauth:client-assertion-type:jwt-bearer",
  77             "client_assertion": self.get_oauth_client_secret(),
  78             "grant_type": "refresh_token",
  79             "assertion": refresh_token,
  80             "redirect_uri": absolute_uri(oauth_redirect_url),
  81         }
  82 
  83     def build_identity(self, data):
  84         data = data["data"]
  85         user = get_user_info(data["access_token"])
  86 
  87         return {
  88             "type": "vsts",
  89             "id": user["id"],
  90             "email": user["emailAddress"],
  91             "email_verified": True,
  92             "name": user["displayName"],
  93             "scopes": sorted(self.oauth_scopes),
  94             "data": self.get_oauth_data(data),
  95         }

  98 class VSTSOAuth2CallbackView(OAuth2CallbackView):
  99     def exchange_token(self, request, pipeline, code):
 100         from sentry.http import safe_urlopen, safe_urlread
 101         from sentry.utils.http import absolute_uri
 102         from six.moves.urllib.parse import parse_qsl
 103         from sentry.utils import json
 104 
 105         req = safe_urlopen(
 106             url=self.access_token_url,
 107             headers={"Content-Type": "application/x-www-form-urlencoded", "Content-Length": "1322"},
 108             data={
 109                 "client_assertion_type": "urn:ietf:params:oauth:client-assertion-type:jwt-bearer",
 110                 "client_assertion": self.client_secret,
 111                 "grant_type": "urn:ietf:params:oauth:grant-type:jwt-bearer",
 112                 "assertion": code,
 113                 "redirect_uri": absolute_uri(pipeline.redirect_url()),
 114             },
 115         )
 116         body = safe_urlread(req)
 117         if req.headers["Content-Type"].startswith("application/x-www-form-urlencoded"):
 118             return dict(parse_qsl(body))
 119         return json.loads(body)
