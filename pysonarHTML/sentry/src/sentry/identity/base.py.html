  23     def build_identity(self, state):
  24         """
  25         Return a mapping containing the identity information.
  26 
  27         - ``state`` is the resulting data captured by the pipeline
  28 
  29         >>> {
  30         >>>     "id":     "foo@example.com",
  31         >>>     "email":  "foo@example.com",
  32         >>>     "name":   "Foo Bar",
  33         >>>     "scopes": ['email', ...],
  34         >>>     "data":   { ... },
  35         >>> }
  36 
  37         The ``id`` key is required.
  38 
  39         The ``id`` may be passed in as a ``MigratingIdentityId`` should the
  40         the id key be migrating from one value to another and have multiple
  41         lookup values.
  42 
  43         If the identity can not be constructed an ``IdentityNotValid`` error
  44         should be raised.
  45         """
  46         raise NotImplementedError

  48     def update_identity(self, new_data, current_data):
  49         """
  50         When re-authenticating with a provider, the identity data may need to
  51         be mutated based on the previous state. An example of this is Google,
  52         which will not return a `refresh_token` unless the user explicitly
  53         goes through an approval process.
  54 
  55         Return the new state which should be used for an identity.
  56         """
  57         return new_data
