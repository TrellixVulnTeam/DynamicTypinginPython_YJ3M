   7 class SlackIdentityProvider(OAuth2Provider):
   8     key = "slack"
   9     name = "Slack"
  10 
  11     # This identity provider is used for authorizing the Slack application
  12     # through their Bot token (or legacy Workspace Token if enabled) flow.
  13 
  14     oauth_scopes = ("identity.basic", "identity.email")
  15 
  16     # Only used during installation for Bot apps in order to request "links:read"
  17     # user_scope, needed for unfurling.
  18     user_scopes = ()
  19 
  20     def get_oauth_authorize_url(self):
  21         return "https://slack.com/oauth/v2/authorize"
  22 
  23     # XXX(epurkhiser): While workspace tokens _do_ support the oauth.access
  24     # endpoint, it will no include the authorizing_user, so we continue to use
  25     # the deprecated oauth.token endpoint until we are able to migrate to a bot
  26     # app which uses oauth.access.
  27     def get_oauth_access_token_url(self):
  28         return "https://slack.com/api/oauth.v2.access"
  29 
  30     def get_oauth_client_id(self):
  31         return options.get("slack-v2.client-id") or options.get("slack.client-id")
  32 
  33     def get_oauth_client_secret(self):
  34         return options.get("slack-v2.client-secret") or options.get("slack.client-secret")
  35 
  36     def get_user_scopes(self):
  37         return self.config.get("user_scopes", self.user_scopes)
  38 
  39     def get_pipeline_views(self):
  40         return [
  41             SlackOAuth2LoginView(
  42                 authorize_url=self.get_oauth_authorize_url(),
  43                 client_id=self.get_oauth_client_id(),
  44                 scope=" ".join(self.get_oauth_scopes()),
  45                 user_scope=" ".join(self.get_user_scopes()),
  46             ),
  47             OAuth2CallbackView(
  48                 access_token_url=self.get_oauth_access_token_url(),
  49                 client_id=self.get_oauth_client_id(),
  50                 client_secret=self.get_oauth_client_secret(),
  51             ),
  52         ]
  53 
  54     def get_oauth_data(self, payload):
  55         # TODO(epurkhiser): This flow isn't actually used right now in sentry.
  56         # In slack-bot world we would need to make an API call to the 'me'
  57         # endpoint to get their user ID here.
  58         return super(SlackIdentityProvider, self).get_oauth_data(self, payload)
  59 
  60     def build_identity(self, data):
  61         data = data["data"]
  62 
  63         return {
  64             "type": "slack",
  65             # TODO(epurkhiser): See note above
  66             "id": data["user"]["id"],
  67             "email": data["user"]["email"],
  68             "scopes": sorted(data["scope"].split(",")),
  69             "data": self.get_oauth_data(data),
  70         }

  73 class SlackOAuth2LoginView(OAuth2LoginView):
  74     """
  75     We need to customize the OAuth2LoginView in order to support passing through
  76     the `user_scope` param in the request.
  77 
  78     The `user_scope` param would not be used when requesting identity scopes.
  79     """
  80 
  81     user_scope = ""
  82 
  83     def __init__(
  84         self, authorize_url=None, client_id=None, scope=None, user_scope=None, *args, **kwargs
  85     ):
  86         super(SlackOAuth2LoginView, self).__init__(
  87             authorize_url=authorize_url, client_id=client_id, scope=scope, *args, **kwargs
  88         )
  89         if user_scope is not None:
  90             self.user_scope = user_scope
  91 
  92     def get_authorize_params(self, state, redirect_uri):
  93         data = super(SlackOAuth2LoginView, self).get_authorize_params(state, redirect_uri)
  94 
  95         # XXX(meredith): Bot apps must be added manually to channels, and link unfurling
  96         # only works for channels the bot is a part of, so in order to expand the usage
  97         # of unfurling to work in channels the bot is not a part of we must request the
  98         # `links:read` scope for the user. This way channels that the authorizing user
  99         # are in will support link unfurling
 100         #
 101         # The way we can request scopes for users is by using the `user_scope` param. This
 102         # will give us the user token...which I don't think we use, but requesting the scope
 103         # seems to be enough to get the unfurling to work.
 104         #
 105         # Resources: https://api.slack.com/authentication/oauth-v2#asking
 106         if self.user_scope:
 107             data["user_scope"] = self.user_scope
 108 
 109         return data
