   8 def get_user_info(access_token):
   9     session = http.build_session()
  10     resp = session.get(
  11         "https://api.github.com/user",
  12         headers={
  13             "Accept": "application/vnd.github.machine-man-preview+json",
  14             "Authorization": "token %s" % access_token,
  15         },
  16     )
  17     resp.raise_for_status()
  18     resp = resp.json()
  19 
  20     return resp

  23 # GitHub has 2 types of apps -- GitHub apps and OAuth apps. SSO is implemented
  24 # using OAuth App, but signup and integrations use the github app. When github
  25 # apps have API parity with OAuth apps, we should move SSO to it as well.
  26 # https://developer.github.com/apps/differences-between-apps/
  27 
  28 
  29 class GitHubIdentityProvider(OAuth2Provider):
  30     key = "github"
  31     name = "GitHub"
  32 
  33     oauth_access_token_url = "https://github.com/login/oauth/access_token"
  34     oauth_authorize_url = "https://github.com/login/oauth/authorize"
  35 
  36     oauth_scopes = ()
  37 
  38     def get_oauth_client_id(self):
  39         return options.get("github-app.client-id")
  40 
  41     def get_oauth_client_secret(self):
  42         return options.get("github-app.client-secret")
  43 
  44     def build_identity(self, data):
  45         data = data["data"]
  46         user = get_user_info(data["access_token"])
  47 
  48         return {
  49             "type": "github",
  50             "id": user["id"],
  51             "email": user["email"],
  52             "email_verified": bool(user["email"]),
  53             "login": user["login"],
  54             "name": user["name"],
  55             "company": user["company"],
  56             "scopes": [],  # GitHub apps do not have user scopes
  57             "data": self.get_oauth_data(data),
  58         }
