  42 class ZDecoder(io.RawIOBase):
  43     """
  44     Base class for HTTP content decoders based on zlib
  45     See: https://github.com/eBay/wextracto/blob/9c789b1c98d95a1e87dbedfd1541a8688d128f5c/wex/http_decoder.py
  46     """
  47 
  48     def __init__(self, fp, z=None):
  49         self.fp = fp
  50         self.z = z
  51         self.flushed = None
  52 
  53     def readable(self):
  54         return True
  55 
  56     def readinto(self, buf):
  57         if self.z is None:
  58             self.z = zlib.decompressobj()
  59             retry = True
  60         else:
  61             retry = False
  62 
  63         n = 0
  64         max_length = len(buf)
  65 
  66         while max_length > 0:
  67             if self.flushed is None:
  68                 chunk = self.fp.read(Z_CHUNK)
  69                 compressed = self.z.unconsumed_tail + chunk
  70                 try:
  71                     decompressed = self.z.decompress(compressed, max_length)
  72                 except zlib.error:
  73                     if not retry:
  74                         raise
  75                     self.z = zlib.decompressobj(-zlib.MAX_WBITS)
  76                     retry = False
  77                     decompressed = self.z.decompress(compressed, max_length)
  78 
  79                 if not chunk:
  80                     self.flushed = self.z.flush()
  81             else:
  82                 if not self.flushed:
  83                     return n
  84 
  85                 decompressed = self.flushed[:max_length]
  86                 self.flushed = self.flushed[max_length:]
  87 
  88             buf[n : n + len(decompressed)] = decompressed
  89             n += len(decompressed)
  90             max_length = len(buf) - n
  91 
  92         return n
  93

 110 class SetRemoteAddrFromForwardedFor(object):

 113             raise MiddlewareNotUsed

 115     def _remove_port_number(self, ip_address):
 116         if "[" in ip_address and "]" in ip_address:
 117             # IPv6 address with brackets, possibly with a port number
 118             return ip_address[ip_address.find("[") + 1 : ip_address.find("]")]
 119         if "." in ip_address and ip_address.rfind(":") > ip_address.rfind("."):
 120             # IPv4 address with port number
 121             # the last condition excludes IPv4-mapped IPv6 addresses
 122             return ip_address.rsplit(":", 1)[0]
 123         return ip_address
 124 
 125     def process_request(self, request):
 126         try:
 127             real_ip = request.META["HTTP_X_FORWARDED_FOR"]
 128         except KeyError:
 129             pass
 130         else:
 131             # HTTP_X_FORWARDED_FOR can be a comma-separated list of IPs.
 132             # Take just the first one.
 133             real_ip = real_ip.split(",")[0].strip()
 134             real_ip = self._remove_port_number(real_ip)
 135             request.META["REMOTE_ADDR"] = real_ip

 138 class ChunkedMiddleware(object):
 139     def __init__(self):
 140         if not has_uwsgi:
 141             raise MiddlewareNotUsed
 142 
 143     def process_request(self, request):
 144         # If we are dealing with chunked data and we have uwsgi we assume
 145         # that we can read to the end of the input stream so we can bypass
 146         # the default limited stream.  We set the content length reasonably
 147         # high so that the reads generally succeed.  This is ugly but with
 148         # Django 1.6 it seems to be the best we can easily do.
 149         if "HTTP_TRANSFER_ENCODING" not in request.META:
 150             return
 151 
 152         if request.META["HTTP_TRANSFER_ENCODING"].lower() == "chunked":
 153             request._stream = io.BufferedReader(UWsgiChunkedInput())
 154             request.META["CONTENT_LENGTH"] = "4294967295"  # 0xffffffff

 157 class DecompressBodyMiddleware(object):
 158     def process_request(self, request):
 159         decode = False
 160         encoding = request.META.get("HTTP_CONTENT_ENCODING", "").lower()
 161 
 162         if encoding == "gzip":
 163             request._stream = GzipDecoder(request._stream)
 164             decode = True
 165 
 166         if encoding == "deflate":
 167             request._stream = DeflateDecoder(request._stream)
 168             decode = True
 169 
 170         if decode:
 171             # Since we don't know the original content length ahead of time, we
 172             # need to set the content length reasonably high so read generally
 173             # succeeds. This seems to be the only easy way for Django 1.6.
 174             request.META["CONTENT_LENGTH"] = "4294967295"  # 0xffffffff
 175 
 176             # The original content encoding is no longer valid, so we have to
 177             # remove the header. Otherwise, LazyData will attempt to re-decode
 178             # the body.
 179             del request.META["HTTP_CONTENT_ENCODING"]
