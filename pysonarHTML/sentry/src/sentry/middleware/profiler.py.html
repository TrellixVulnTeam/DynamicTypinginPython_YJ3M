  27 class ProfileMiddleware(object):
  28     """
  29     Displays hotshot profiling for any view.
  30     http://yoursite.com/yourview/?prof
  31 
  32     Add the "prof" key to query string by appending ?prof (or &prof=)
  33     and you'll see the profiling results in your browser.
  34     It's set up to only be available in django's debug mode, is available for superuser otherwise,
  35     but you really shouldn't add this middleware to any production configuration.
  36 
  37     WARNING: It uses hotshot profiler which is not thread safe.
  38     """
  39 
  40     def can(self, request):
  41         if "prof" not in request.GET:
  42             return False
  43         if settings.DEBUG:
  44             return True
  45         if is_active_superuser(request):
  46             return True
  47         return False
  48 
  49     def process_view(self, request, callback, callback_args, callback_kwargs):
  50         if not self.can(request):
  51             return
  52         self.prof = cProfile.Profile()
  53 
  54         return self.prof.runcall(callback, request, *callback_args, **callback_kwargs)
  55 
  56     def get_group(self, filename):
  57         for g in group_prefix_re:
  58             name = g.findall(filename)
  59             if name:
  60                 return name[0]
  61 
  62     def get_summary(self, results_dict, total):
  63         results = [(item[1], item[0]) for item in six.iteritems(results_dict)]
  64         results.sort(reverse=True)
  65         results = results[:40]
  66 
  67         res = "      tottime\n"
  68         for item in results:
  69             res += "%4.1f%% %7.3f %s\n" % (100 * item[0] / total if total else 0, item[0], item[1])
  70 
  71         return res

  73     def normalize_paths(self, stats):
  74         import os.path
  75         from pstats import add_func_stats, func_std_string
  76 
  77         python_paths = sorted(sys.path, reverse=True)
  78 
  79         def rel_filename(filename):
  80             for path in python_paths:
  81                 if filename.startswith(path):
  82                     return filename[len(path) + 1 :]
  83             return os.path.basename(filename)
  84 
  85         def func_strip_path(func_name):
  86             filename, line, name = func_name
  87             return rel_filename(filename), line, name
  88 
  89         oldstats = stats.stats
  90         stats.stats = newstats = {}
  91         max_name_len = 0
  92         for func, (cc, nc, tt, ct, callers) in six.iteritems(oldstats):
  93             newfunc = func_strip_path(func)
  94             if len(func_std_string(newfunc)) > max_name_len:
  95                 max_name_len = len(func_std_string(newfunc))
  96             newcallers = {}
  97             for func2, caller in six.iteritems(callers):
  98                 newcallers[func_strip_path(func2)] = caller
  99 
 100             if newfunc in newstats:
 101                 newstats[newfunc] = add_func_stats(newstats[newfunc], (cc, nc, tt, ct, newcallers))
 102             else:
 103                 newstats[newfunc] = (cc, nc, tt, ct, newcallers)
 104         old_top = stats.top_level
 105         stats.top_level = new_top = {}
 106         for func in old_top:
 107             new_top[func_strip_path(func)] = None
 108 
 109         stats.max_name_len = max_name_len
 110 
 111         stats.fcn_list = None
 112         stats.all_callees = None
 113         return self

 115     def summary_for_files(self, stats_str):
 116         stats_str = stats_str.split("\n")[5:]
 117 
 118         mystats = {}
 119         mygroups = {}
 120 
 121         total = 0
 122 
 123         for s in stats_str:
 124             fields = words_re.split(s)
 125             if len(fields) == 7:
 126                 time = float(fields[2])
 127                 total += time
 128                 filename = fields[6].split(":")[0]
 129 
 130                 if filename not in mystats:
 131                     mystats[filename] = 0
 132                 mystats[filename] += time
 133 
 134                 group = self.get_group(filename)
 135                 if group not in mygroups:
 136                     mygroups[group] = 0
 137                 mygroups[group] += time
 138 
 139         return (
 140             "\n"
 141             + " ---- By file ----\n\n"
 142             + self.get_summary(mystats, total)
 143             + "\n"
 144             + " ---- By group ---\n\n"
 145             + self.get_summary(mygroups, total)
 146             + "\n"
 147         )

 149     def process_response(self, request, response):
 150         if not self.can(request):
 151             return response
 152 
 153         out = StringIO.StringIO()
 154         old_stdout = sys.stdout
 155         sys.stdout = out
 156 
 157         stats = pstats.Stats(self.prof)
 158         self.normalize_paths(stats)
 159         stats.sort_stats("time", "calls")
 160         stats.print_stats()
 161 
 162         sys.stdout = old_stdout
 163         stats_str = out.getvalue()
 164 
 165         content = "\n".join(stats_str.split("\n")[:40])
 166         content += "\n\n"
 167         content += self.summary_for_files(stats_str)
 168 
 169         return HttpResponse(content, "text/plain")
