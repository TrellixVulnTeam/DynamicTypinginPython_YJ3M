   8 class SuperuserMiddleware(object):
   9     def process_request(self, request):
  10         # This avoids touching user session, which means we avoid
  11         # setting `Vary: Cookie` as a response header which will
  12         # break HTTP caching entirely.
  13         self.__skip_caching = request.path_info.startswith(settings.ANONYMOUS_STATIC_PREFIXES)
  14 
  15         if self.__skip_caching:
  16             # XXX(dcramer): support legacy is_superuser calls for unauthenticated requests
  17             request.is_superuser = lambda: False
  18             return
  19 
  20         su = Superuser(request)
  21 
  22         request.superuser = su
  23         request.is_superuser = lambda: request.superuser.is_active
  24 
  25         if su.is_active:
  26             logger.info(
  27                 "superuser.request",
  28                 extra={
  29                     "url": request.build_absolute_uri(),
  30                     "method": request.method,
  31                     "ip_address": request.META["REMOTE_ADDR"],
  32                     "user_id": request.user.id,
  33                 },
  34             )
  35 
  36     def process_response(self, request, response):
  37         try:
  38             if self.__skip_caching:
  39                 return response
  40         except AttributeError:
  41             pass
  42         su = getattr(request, "superuser", None)
  43         if su:
  44             su.on_response(response)
  45         return response
