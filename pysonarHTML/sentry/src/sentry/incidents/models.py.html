  46 class IncidentManager(BaseManager):
  47     CACHE_KEY = "incidents:active:%s:%s"
  48 
  49     def fetch_for_organization(self, organization, projects):
  50         return self.filter(organization=organization, projects__in=projects).distinct()
  51 
  52     @classmethod
  53     def _build_active_incident_cache_key(self, alert_rule_id, project_id):
  54         return self.CACHE_KEY % (alert_rule_id, project_id)
  55 
  56     def get_active_incident(self, alert_rule, project):
  57         cache_key = self._build_active_incident_cache_key(alert_rule.id, project.id)
  58         incident = cache.get(cache_key)
  59         if incident is None:
  60             try:
  61                 incident = (
  62                     Incident.objects.filter(
  63                         type=IncidentType.ALERT_TRIGGERED.value,
  64                         alert_rule=alert_rule,
  65                         projects=project,
  66                     )
  67                     .exclude(status=IncidentStatus.CLOSED.value)
  68                     .order_by("-date_added")[0]
  69                 )
  70             except IndexError:
  71                 # Set this to False so that we can have a negative cache as well.
  72                 incident = False
  73             cache.set(cache_key, incident)
  74             if incident is False:
  75                 incident = None
  76         elif not incident:
  77             # If we had a falsey not None value in the cache, then we stored that there
  78             # are no current active incidents. Just set to None
  79             incident = None
  80 
  81         return incident
  82 
  83     @classmethod
  84     def clear_active_incident_cache(cls, instance, **kwargs):
  85         for project in instance.projects.all():
  86             cache.delete(cls._build_active_incident_cache_key(instance.alert_rule_id, project.id))
  87 
  88     @classmethod
  89     def clear_active_incident_project_cache(cls, instance, **kwargs):
  90         cache.delete(
  91             cls._build_active_incident_cache_key(
  92                 instance.incident.alert_rule_id, instance.project_id
  93             )
  94         )
  95 
  96     @TimedRetryPolicy.wrap(timeout=5, exceptions=(IntegrityError,))
  97     def create(self, organization, **kwargs):
  98         """
  99         Creates an Incident. Fetches the maximum identifier value for the org
 100         and increments it by one. If two incidents are created for the
 101         Organization at the same time then an integrity error will be thrown,
 102         and we'll retry again several times. I prefer to lock optimistically
 103         here since if we're creating multiple Incidents a second for an
 104         Organization then we're likely failing at making Incidents useful.
 105         """
 106         with transaction.atomic():
 107             result = self.filter(organization=organization).aggregate(models.Max("identifier"))
 108             identifier = result["identifier__max"]
 109             if identifier is None:
 110                 identifier = 1
 111             else:
 112                 identifier += 1
 113 
 114             return super(IncidentManager, self).create(
 115                 organization=organization, identifier=identifier, **kwargs
 116             )

 287 class AlertRuleManager(BaseManager):
 288     """
 289     A manager that excludes all rows that are snapshots.
 290     """
 291 
 292     CACHE_SUBSCRIPTION_KEY = "alert_rule:subscription:%s"
 293 
 294     def get_queryset(self):
 295         return (
 296             super(AlertRuleManager, self)
 297             .get_queryset()
 298             .exclude(status=AlertRuleStatus.SNAPSHOT.value)
 299         )
 300 
 301     def fetch_for_organization(self, organization, projects=None):
 302         queryset = self.filter(organization=organization)
 303         if projects is not None:
 304             queryset = queryset.filter(snuba_query__subscriptions__project__in=projects)
 305 
 306         return queryset
 307 
 308     def fetch_for_project(self, project):
 309         return self.filter(snuba_query__subscriptions__project=project)
 310 
 311     @classmethod
 312     def __build_subscription_cache_key(self, subscription_id):
 313         return self.CACHE_SUBSCRIPTION_KEY % subscription_id
 314 
 315     def get_for_subscription(self, subscription):
 316         """
 317         Fetches the AlertRule associated with a Subscription. Attempts to fetch from
 318         cache then hits the database
 319         """
 320         cache_key = self.__build_subscription_cache_key(subscription.id)
 321         alert_rule = cache.get(cache_key)
 322         if alert_rule is None:
 323             alert_rule = AlertRule.objects.get(snuba_query__subscriptions=subscription)
 324             cache.set(cache_key, alert_rule, 3600)
 325 
 326         return alert_rule
 327 
 328     @classmethod
 329     def clear_subscription_cache(cls, instance, **kwargs):
 330         cache.delete(cls.__build_subscription_cache_key(instance.id))
 331 
 332     @classmethod
 333     def clear_alert_rule_subscription_caches(cls, instance, **kwargs):
 334         subscription_ids = QuerySubscription.objects.filter(
 335             snuba_query=instance.snuba_query
 336         ).values_list("id", flat=True)
 337         if subscription_ids:
 338             cache.delete_many(
 339                 cls.__build_subscription_cache_key(sub_id) for sub_id in subscription_ids
 340             )

 504 class AlertRuleTriggerAction(Model):
 505     """
 506     This model represents an action that occurs when a trigger is fired. This is
 507     typically some sort of notification.
 508     """
 509 
 510     __core__ = True
 511 
 512     _type_registrations = {}
 513 
 514     # Which sort of action to take
 515     class Type(Enum):
 516         EMAIL = 0
 517         PAGERDUTY = 1
 518         SLACK = 2
 519 
 520     class TargetType(Enum):
 521         # A direct reference, like an email address, Slack channel or PagerDuty service
 522         SPECIFIC = 0
 523         # A specific user. This could be used to grab the user's email address.
 524         USER = 1
 525         # A specific team. This could be used to send an email to everyone associated
 526         # with a team.
 527         TEAM = 2
 528 
 529     TypeRegistration = namedtuple(
 530         "TypeRegistration",
 531         ["handler", "slug", "type", "supported_target_types", "integration_provider"],
 532     )
 533 
 534     alert_rule_trigger = FlexibleForeignKey("sentry.AlertRuleTrigger")
 535     integration = FlexibleForeignKey("sentry.Integration", null=True)
 536     type = models.SmallIntegerField()
 537     target_type = models.SmallIntegerField()
 538     # Identifier used to perform the action on a given target
 539     target_identifier = models.TextField(null=True)
 540     # Human readable name to display in the UI
 541     target_display = models.TextField(null=True)
 542     date_added = models.DateTimeField(default=timezone.now)
 543 
 544     class Meta:
 545         app_label = "sentry"
 546         db_table = "sentry_alertruletriggeraction"
 547 
 548     @property
 549     def target(self):
 550         if self.target_type == self.TargetType.USER.value:
 551             try:
 552                 return User.objects.get(id=int(self.target_identifier))
 553             except User.DoesNotExist:
 554                 pass
 555         elif self.target_type == self.TargetType.TEAM.value:
 556             try:
 557                 return Team.objects.get(id=int(self.target_identifier))
 558             except Team.DoesNotExist:
 559                 pass
 560         elif self.target_type == self.TargetType.SPECIFIC.value:
 561             # TODO: This is only for email. We should have a way of validating that it's
 562             # ok to contact this email.
 563             return self.target_identifier
 564 
 565     def build_handler(self, incident, project):
 566         type = AlertRuleTriggerAction.Type(self.type)
 567         if type in self._type_registrations:
 568             return self._type_registrations[type].handler(self, incident, project)
 569         else:
 570             metrics.incr("alert_rule_trigger.unhandled_type.{}".format(self.type))
 571 
 572     def fire(self, incident, project, metric_value):
 573         handler = self.build_handler(incident, project)
 574         if handler:
 575             return handler.fire(metric_value)
 576 
 577     def resolve(self, incident, project, metric_value):
 578         handler = self.build_handler(incident, project)
 579         if handler:
 580             return handler.resolve(metric_value)
 581 
 582     @classmethod
 583     def register_type(cls, slug, type, supported_target_types, integration_provider=None):
 584         """
 585         Registers a handler for a given type.
 586         :param slug: A string representing the name of this type registration
 587         :param type: The `Type` to handle.
 588         :param handler: A subclass of `ActionHandler` that accepts the
 589         `AlertRuleTriggerAction` and `Incident`.
 590         :param integration_provider: String representing the integration provider
 591         related to this type.
 592         """
 593 
 594         def inner(handler):
 595             if type not in cls._type_registrations:
 596                 cls._type_registrations[type] = cls.TypeRegistration(
 597                     handler, slug, type, frozenset(supported_target_types), integration_provider
 598                 )
 599             else:
 600                 raise Exception(u"Handler already registered for type %s" % type)
 601             return handler
 602 
 603         return inner
 604 
 605     @classmethod
 606     def get_registered_type(cls, type):
 607         return cls._type_registrations[type]
 608 
 609     @classmethod
 610     def get_registered_types(cls):
 611         return cls._type_registrations.values()
