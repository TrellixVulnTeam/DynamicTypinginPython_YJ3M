  40 class SubscriptionProcessor(object):
  41     """
  42     Class for processing subscription updates for an alert rule. Accepts a subscription
  43     and then can process one or more updates via `process_update`. Keeps track of how
  44     close an alert rule is to alerting, creates an incident, and auto resolves the
  45     incident if a resolve threshold is set and the threshold is triggered.
  46     """
  47 
  48     # Each entry is a tuple in format (<alert_operator>, <resolve_operator>)
  49     THRESHOLD_TYPE_OPERATORS = {
  50         AlertRuleThresholdType.ABOVE: (operator.gt, operator.lt),
  51         AlertRuleThresholdType.BELOW: (operator.lt, operator.gt),
  52     }

  54     def __init__(self, subscription):
  55         self.subscription = subscription
  56         try:
  57             self.alert_rule = AlertRule.objects.get_for_subscription(subscription)
  58         except AlertRule.DoesNotExist:
  59             return
  60 
  61         self.triggers = AlertRuleTrigger.objects.get_for_alert_rule(self.alert_rule)
  62         self.triggers.sort(key=lambda trigger: trigger.alert_threshold)
  63 
  64         (
  65             self.last_update,
  66             self.trigger_alert_counts,
  67             self.trigger_resolve_counts,
  68         ) = get_alert_rule_stats(self.alert_rule, self.subscription, self.triggers)
  69         self.orig_trigger_alert_counts = deepcopy(self.trigger_alert_counts)
  70         self.orig_trigger_resolve_counts = deepcopy(self.trigger_resolve_counts)

  98     def check_trigger_status(self, trigger, status):
  99         """
 100         Determines whether a trigger is currently at the specified status
 101         :param trigger: An `AlertRuleTrigger`
 102         :param status: A `TriggerStatus`
 103         :return: True if at the specified status, otherwise False
 104         """
 105         incident_trigger = self.incident_triggers.get(trigger.id)
 106         return incident_trigger is not None and incident_trigger.status == status.value

 108     def process_update(self, subscription_update):
 109         if not hasattr(self, "alert_rule"):
 110             # If the alert rule has been removed then just skip
 111             metrics.incr("incidents.alert_rules.no_alert_rule_for_subscription")
 112             logger.error(
 113                 "Received an update for a subscription, but no associated alert rule exists"
 114             )
 115             # TODO: Delete subscription here.
 116             return
 117 
 118         if subscription_update["timestamp"] <= self.last_update:
 119             metrics.incr("incidents.alert_rules.skipping_already_processed_update")
 120             return
 121 
 122         self.last_update = subscription_update["timestamp"]
 123 
 124         if len(subscription_update["values"]["data"]) > 1:
 125             logger.warning(
 126                 "Subscription returned more than 1 row of data",
 127                 extra={
 128                     "subscription_id": self.subscription.id,
 129                     "dataset": self.subscription.snuba_query.dataset,
 130                     "snuba_subscription_id": self.subscription.subscription_id,
 131                     "result": subscription_update,
 132                 },
 133             )
 134         aggregation_value = subscription_update["values"]["data"][0].values()[0]
 135 
 136         for trigger in self.triggers:
 137             alert_operator, resolve_operator = self.THRESHOLD_TYPE_OPERATORS[
 138                 AlertRuleThresholdType(trigger.threshold_type)
 139             ]
 140 
 141             if alert_operator(
 142                 aggregation_value, trigger.alert_threshold
 143             ) and not self.check_trigger_status(trigger, TriggerStatus.ACTIVE):
 144                 metrics.incr("incidents.alert_rules.threshold", tags={"type": "alert"})
 145                 with transaction.atomic():
 146                     self.trigger_alert_threshold(trigger, aggregation_value)
 147             elif (
 148                 trigger.resolve_threshold is not None
 149                 and resolve_operator(aggregation_value, trigger.resolve_threshold)
 150                 and self.check_trigger_status(trigger, TriggerStatus.ACTIVE)
 151             ):
 152                 metrics.incr("incidents.alert_rules.threshold", tags={"type": "resolve"})
 153                 with transaction.atomic():
 154                     self.trigger_resolve_threshold(trigger, aggregation_value)
 155             else:
 156                 self.trigger_alert_counts[trigger.id] = 0
 157                 self.trigger_resolve_counts[trigger.id] = 0
 158 
 159         # We update the rule stats here after we commit the transaction. This guarantees
 160         # that we'll never miss an update, since we'll never roll back if the process
 161         # is killed here. The trade-off is that we might process an update twice. Mostly
 162         # this will have no effect, but if someone manages to close a triggered incident
 163         # before the next one then we might alert twice.
 164         self.update_alert_rule_stats()
 165 
 166     def calculate_event_date_from_update_date(self, update_date):
 167         """
 168         Calculates the date that an event actually happened based on the date that we
 169         received the update. This takes into account time window and threshold period.
 170         :return:
 171         """
 172         # Subscriptions label buckets by the end of the bucket, whereas discover
 173         # labels them by the front. This causes us an off-by-one error with event dates,
 174         # so to prevent this we subtract a bucket off of the date.
 175         update_date -= timedelta(seconds=self.alert_rule.snuba_query.time_window)
 176         # We want to also subtract `frequency * (threshold_period - 1)` from the date.
 177         # This allows us to show the actual start of the event, rather than the date
 178         # of the last update that we received.
 179         return update_date - timedelta(
 180             seconds=(
 181                 self.alert_rule.snuba_query.resolution * (self.alert_rule.threshold_period - 1)
 182             )
 183         )

 185     def trigger_alert_threshold(self, trigger, metric_value):
 186         """
 187         Called when a subscription update exceeds the value defined in the
 188         `trigger.alert_threshold`, and the trigger hasn't already been activated.
 189         Increments the count of how many times we've consecutively exceeded the threshold, and if
 190         above the `threshold_period` defined in the alert rule then mark the trigger as
 191         activated, and create an incident if there isn't already one.
 192         :return:
 193         """
 194         self.trigger_alert_counts[trigger.id] += 1
 195         if self.trigger_alert_counts[trigger.id] >= self.alert_rule.threshold_period:
 196             metrics.incr("incidents.alert_rules.trigger", tags={"type": "fire"})
 197             # Only create a new incident if we don't already have an active one
 198             if not self.active_incident:
 199                 detected_at = self.calculate_event_date_from_update_date(self.last_update)
 200                 self.active_incident = create_incident(
 201                     self.alert_rule.organization,
 202                     IncidentType.ALERT_TRIGGERED,
 203                     # TODO: Include more info in name?
 204                     self.alert_rule.name,
 205                     alert_rule=self.alert_rule,
 206                     date_started=detected_at,
 207                     date_detected=detected_at,
 208                     projects=[self.subscription.project],
 209                 )
 210             # Now create (or update if it already exists) the incident trigger so that
 211             # we have a record of this trigger firing for this incident
 212             incident_trigger = self.incident_triggers.get(trigger.id)
 213             if incident_trigger:
 214                 incident_trigger.status = TriggerStatus.ACTIVE.value
 215                 incident_trigger.save()
 216             else:
 217                 incident_trigger = IncidentTrigger.objects.create(
 218                     incident=self.active_incident,
 219                     alert_rule_trigger=trigger,
 220                     status=TriggerStatus.ACTIVE.value,
 221                 )
 222             self.handle_incident_severity_update()
 223             self.handle_trigger_actions(incident_trigger, metric_value)
 224             self.incident_triggers[trigger.id] = incident_trigger
 225 
 226             # TODO: We should create an audit log, and maybe something that keeps
 227             # all of the details available for showing on the incident. Might be a json
 228             # blob or w/e? Or might be able to use the audit log
 229 
 230             # We now set this threshold to 0. We don't need to count it anymore
 231             # once we've triggered an incident.
 232             self.trigger_alert_counts[trigger.id] = 0
 233 
 234     def check_triggers_resolved(self):
 235         """
 236         Determines whether all triggers associated with the active incident are
 237         resolved. A trigger is considered resolved if it either has no resolve
 238         threshold, or is in the `TriggerStatus.Resolved` state.
 239         :return:
 240         """
 241         for incident_trigger in self.incident_triggers.values():
 242             if (
 243                 incident_trigger.alert_rule_trigger.resolve_threshold is not None
 244                 and incident_trigger.status != TriggerStatus.RESOLVED.value
 245             ):
 246                 return False
 247         return True
 248 
 249     def trigger_resolve_threshold(self, trigger, metric_value):
 250         """
 251         Called when a subscription update exceeds the value defined in
 252         `trigger.resolve_threshold` and the trigger is currently ACTIVE.
 253         :return:
 254         """
 255         self.trigger_resolve_counts[trigger.id] += 1
 256         if self.trigger_resolve_counts[trigger.id] >= self.alert_rule.threshold_period:
 257             metrics.incr("incidents.alert_rules.trigger", tags={"type": "resolve"})
 258             incident_trigger = self.incident_triggers[trigger.id]
 259             incident_trigger.status = TriggerStatus.RESOLVED.value
 260             incident_trigger.save()
 261             self.handle_trigger_actions(incident_trigger, metric_value)
 262             self.handle_incident_severity_update()
 263 
 264             if self.check_triggers_resolved():
 265                 update_incident_status(
 266                     self.active_incident,
 267                     IncidentStatus.CLOSED,
 268                     status_method=IncidentStatusMethod.RULE_TRIGGERED,
 269                     date_closed=self.calculate_event_date_from_update_date(self.last_update),
 270                 )
 271                 self.active_incident = None
 272                 self.incident_triggers.clear()
 273             self.trigger_resolve_counts[trigger.id] = 0
 274 
 275     def handle_trigger_actions(self, incident_trigger, metric_value):
 276         method = "fire" if incident_trigger.status == TriggerStatus.ACTIVE.value else "resolve"
 277 
 278         for action in incident_trigger.alert_rule_trigger.alertruletriggeraction_set.all():
 279             handle_trigger_action.apply_async(
 280                 kwargs={
 281                     "action_id": action.id,
 282                     "incident_id": incident_trigger.incident_id,
 283                     "project_id": self.subscription.project_id,
 284                     "metric_value": metric_value,
 285                     "method": method,
 286                 },
 287                 countdown=5,
 288             )

 290     def handle_incident_severity_update(self):
 291         if self.active_incident:
 292             active_incident_triggers = IncidentTrigger.objects.filter(
 293                 incident=self.active_incident, status=TriggerStatus.ACTIVE.value
 294             )
 295             severity = None
 296             for active_incident_trigger in active_incident_triggers:
 297                 trigger = active_incident_trigger.alert_rule_trigger
 298                 if trigger.label == CRITICAL_TRIGGER_LABEL:
 299                     severity = IncidentStatus.CRITICAL
 300                     break
 301                 elif trigger.label == WARNING_TRIGGER_LABEL:
 302                     severity = IncidentStatus.WARNING
 303 
 304             if severity:
 305                 update_incident_status(
 306                     self.active_incident,
 307                     severity,
 308                     status_method=IncidentStatusMethod.RULE_TRIGGERED,
 309                 )

 311     def update_alert_rule_stats(self):
 312         """
 313         Updates stats about the alert rule, if they're changed.
 314         :return:
 315         """
 316         updated_trigger_alert_counts = {
 317             trigger_id: alert_count
 318             for trigger_id, alert_count in self.trigger_alert_counts.items()
 319             if alert_count != self.orig_trigger_alert_counts[trigger_id]
 320         }
 321         updated_trigger_resolve_counts = {
 322             trigger_id: resolve_count
 323             for trigger_id, resolve_count in self.trigger_resolve_counts.items()
 324             if resolve_count != self.orig_trigger_resolve_counts[trigger_id]
 325         }
 326 
 327         update_alert_rule_stats(
 328             self.alert_rule,
 329             self.subscription,
 330             self.last_update,
 331             updated_trigger_alert_counts,
 332             updated_trigger_resolve_counts,
 333         )

 336 def build_alert_rule_stat_keys(alert_rule, subscription):
 337     """
 338     Builds keys for fetching stats about alert rules
 339     :return: A list containing the alert rule stat keys
 340     """
 341     key_base = ALERT_RULE_BASE_KEY % (alert_rule.id, subscription.project_id)
 342     return [ALERT_RULE_BASE_STAT_KEY % (key_base, stat_key) for stat_key in ALERT_RULE_STAT_KEYS]

 345 def build_trigger_stat_keys(alert_rule, subscription, triggers):
 346     """
 347     Builds keys for fetching stats about triggers
 348     :return: A list containing the alert rule trigger stat keys
 349     """
 350     return [
 351         build_alert_rule_trigger_stat_key(
 352             alert_rule.id, subscription.project_id, trigger.id, stat_key
 353         )
 354         for trigger in triggers
 355         for stat_key in ALERT_RULE_TRIGGER_STAT_KEYS
 356     ]

 364 def partition(iterable, n):
 365     """
 366     Partitions an iterable into tuples of size n. Expects the iterable length to be a
 367     multiple of n.
 368     partition('ABCDEF', 3) --> [('ABC', 'DEF')]
 369     """
 370     assert len(iterable) % n == 0
 371     args = [iter(iterable)] * n
 372     return zip(*args)

 375 def get_alert_rule_stats(alert_rule, subscription, triggers):
 376     """
 377     Fetches stats about the alert rule, specific to the current subscription
 378     :return: A tuple containing the stats about the alert rule and subscription.
 379      - last_update: Int representing the timestamp it was last updated
 380      - trigger_alert_counts: A dict of trigger alert counts, where the key is the
 381        trigger id, and the value is an int representing how many consecutive times we
 382        have triggered the alert threshold
 383      - trigger_resolve_counts: A dict of trigger resolve counts, where the key is the
 384        trigger id, and the value is an int representing how many consecutive times we
 385        have triggered the resolve threshold
 386     """
 387 
 388     alert_rule_keys = build_alert_rule_stat_keys(alert_rule, subscription)
 389     trigger_keys = build_trigger_stat_keys(alert_rule, subscription, triggers)
 390     results = get_redis_client().mget(alert_rule_keys + trigger_keys)
 391     results = tuple(0 if result is None else int(result) for result in results)
 392     last_update = to_datetime(results[0])
 393     trigger_results = results[1:]
 394     trigger_alert_counts = {}
 395     trigger_resolve_counts = {}
 396     for trigger, trigger_result in zip(
 397         triggers, partition(trigger_results, len(ALERT_RULE_TRIGGER_STAT_KEYS))
 398     ):
 399         trigger_alert_counts[trigger.id] = trigger_result[0]
 400         trigger_resolve_counts[trigger.id] = trigger_result[1]
 401 
 402     return last_update, trigger_alert_counts, trigger_resolve_counts

 405 def update_alert_rule_stats(alert_rule, subscription, last_update, alert_counts, resolve_counts):
 406     """
 407     Updates stats about the alert rule, subscription and triggers if they've changed.
 408     """
 409     pipeline = get_redis_client().pipeline()
 410 
 411     counts_with_stat_keys = zip(ALERT_RULE_TRIGGER_STAT_KEYS, (alert_counts, resolve_counts))
 412     for stat_key, trigger_counts in counts_with_stat_keys:
 413         for trigger_id, alert_count in trigger_counts.items():
 414             pipeline.set(
 415                 build_alert_rule_trigger_stat_key(
 416                     alert_rule.id, subscription.project_id, trigger_id, stat_key
 417                 ),
 418                 alert_count,
 419                 ex=REDIS_TTL,
 420             )
 421 
 422     last_update_key = build_alert_rule_stat_keys(alert_rule, subscription)[0]
 423     pipeline.set(last_update_key, int(to_timestamp(last_update)), ex=REDIS_TTL)
 424     pipeline.execute()

 427 def get_redis_client():
 428     cluster_key = getattr(settings, "SENTRY_INCIDENT_RULES_REDIS_CLUSTER", "default")
 429     return redis.redis_clusters.get(cluster_key)
