  49     def get_targets(self):
  50         target = self.action.target
  51         if not target:
  52             return []
  53         targets = []
  54         if self.action.target_type in (
  55             AlertRuleTriggerAction.TargetType.USER.value,
  56             AlertRuleTriggerAction.TargetType.TEAM.value,
  57         ):
  58             if self.action.target_type == AlertRuleTriggerAction.TargetType.USER.value:
  59                 targets = [(target.id, target.email)]
  60             elif self.action.target_type == AlertRuleTriggerAction.TargetType.TEAM.value:
  61                 users = self.project.filter_to_subscribed_users(
  62                     set(member.user for member in target.member_set)
  63                 )
  64                 targets = [(user.id, user.email) for user in users]
  65         # TODO: We need some sort of verification system to make sure we're not being
  66         # used as an email relay.
  67         # elif self.action.target_type == AlertRuleTriggerAction.TargetType.SPECIFIC.value:
  68         #     emails = [target]
  69         return targets

  77     def email_users(self, status):
  78         email_context = self.generate_email_context(status)
  79         for user_id, email in self.get_targets():
  80             self.build_message(email_context, status, user_id).send_async(to=[email])

  82     def build_message(self, context, status, user_id):
  83         display = self.status_display[status]
  84         return MessageBuilder(
  85             subject=u"[{}] {} - {}".format(
  86                 context["status"], context["incident_name"], self.project.slug
  87             ),
  88             template=u"sentry/emails/incidents/trigger.txt",
  89             html_template=u"sentry/emails/incidents/trigger.html",
  90             type="incident.alert_rule_{}".format(display.lower()),
  91             context=context,
  92         )

  94     def generate_email_context(self, status):
  95         trigger = self.action.alert_rule_trigger
  96         alert_rule = trigger.alert_rule
  97         snuba_query = alert_rule.snuba_query
  98         is_active = status == TriggerStatus.ACTIVE
  99         is_threshold_type_above = trigger.threshold_type == AlertRuleThresholdType.ABOVE.value
 100 
 101         # if alert threshold and threshold type is above then show '>'
 102         # if resolve threshold and threshold type is *BELOW* then show '>'
 103         # we can simplify this to be the below statement
 104         show_greater_than_string = is_active == is_threshold_type_above
 105         environment_string = snuba_query.environment.name if snuba_query.environment else "All"
 106         aggregate = alert_rule.snuba_query.aggregate
 107         return {
 108             "link": absolute_uri(
 109                 reverse(
 110                     "sentry-metric-alert",
 111                     kwargs={
 112                         "organization_slug": self.incident.organization.slug,
 113                         "incident_id": self.incident.identifier,
 114                     },
 115                 )
 116             ),
 117             "rule_link": absolute_uri(
 118                 reverse(
 119                     "sentry-alert-rule",
 120                     kwargs={
 121                         "organization_slug": self.incident.organization.slug,
 122                         "project_slug": self.project.slug,
 123                         "alert_rule_id": self.action.alert_rule_trigger.alert_rule_id,
 124                     },
 125                 )
 126             ),
 127             "incident_name": self.incident.title,
 128             "environment": environment_string,
 129             "time_window": format_duration(snuba_query.time_window / 60),
 130             "triggered_at": trigger.date_added,
 131             "aggregate": self.query_aggregates_display.get(aggregate, aggregate),
 132             "query": snuba_query.query,
 133             "threshold": trigger.alert_threshold if is_active else trigger.resolve_threshold,
 134             # if alert threshold and threshold type is above then show '>'
 135             # if resolve threshold and threshold type is *BELOW* then show '>'
 136             "threshold_direction_string": ">" if show_greater_than_string else "<",
 137             "status": INCIDENT_STATUS[IncidentStatus(self.incident.status)],
 138             "is_critical": self.incident.status == IncidentStatus.CRITICAL,
 139             "is_warning": self.incident.status == IncidentStatus.WARNING,
 140             "unsubscribe_link": None,
 141         }

 164 def format_duration(minutes):
 165     """
 166     Format minutes into a duration string
 167     """
 168 
 169     if minutes >= 1440:
 170         days = minutes / 1440
 171         return "{} day{}".format(days, pluralize(days))
 172 
 173     if minutes >= 60:
 174         hours = minutes / 60
 175         return "{} hour{}".format(hours, pluralize(hours))
 176 
 177     if minutes >= 1:
 178         return "{} minute{}".format(minutes, pluralize(minutes))
 179 
 180     seconds = minutes / 60
 181     return "{} second{}".format(seconds, pluralize(seconds))
