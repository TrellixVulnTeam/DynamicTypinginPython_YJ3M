  31 @instrumented_task(name="sentry.incidents.tasks.send_subscriber_notifications", queue="incidents")
  32 def send_subscriber_notifications(activity_id):
  33     from sentry.incidents.logic import get_incident_subscribers, unsubscribe_from_incident
  34 
  35     try:
  36         activity = IncidentActivity.objects.select_related(
  37             "incident", "user", "incident__organization"
  38         ).get(id=activity_id)
  39     except IncidentActivity.DoesNotExist:
  40         return
  41 
  42     # Only send notifications for specific activity types.
  43     if activity.type not in (
  44         IncidentActivityType.COMMENT.value,
  45         IncidentActivityType.STATUS_CHANGE.value,
  46     ):
  47         return
  48 
  49     # Check that the user still has access to at least one of the projects
  50     # related to the incident. If not then unsubscribe them.
  51     projects = list(activity.incident.projects.all())
  52     for subscriber in get_incident_subscribers(activity.incident).select_related("user"):
  53         user = subscriber.user
  54         access = from_user(user, activity.incident.organization)
  55         if not any(project for project in projects if access.has_project_access(project)):
  56             unsubscribe_from_incident(activity.incident, user)
  57         elif user != activity.user:
  58             msg = generate_incident_activity_email(activity, user)
  59             msg.send_async([user.email])

  73 def build_activity_context(activity, user):
  74     if activity.type == IncidentActivityType.COMMENT.value:
  75         action = "left a comment"
  76     else:
  77         action = "changed status from %s to %s" % (
  78             INCIDENT_STATUS[IncidentStatus(int(activity.previous_value))],
  79             INCIDENT_STATUS[IncidentStatus(int(activity.value))],
  80         )
  81     incident = activity.incident
  82 
  83     action = "%s on alert %s (#%s)" % (action, incident.title, incident.identifier)
  84 
  85     return {
  86         "user_name": activity.user.name if activity.user else "Sentry",
  87         "action": action,
  88         "link": absolute_uri(
  89             reverse(
  90                 "sentry-metric-alert",
  91                 kwargs={
  92                     "organization_slug": incident.organization.slug,
  93                     "incident_id": incident.identifier,
  94                 },
  95             )
  96         )
  97         + "?"
  98         + urlencode({"referrer": "incident_activity_email"}),
  99         "comment": activity.comment,
 100     }

 117 @instrumented_task(
 118     name="sentry.incidents.tasks.handle_trigger_action",
 119     queue="incidents",
 120     default_retry_delay=60,
 121     max_retries=5,
 122 )
 123 def handle_trigger_action(action_id, incident_id, project_id, method, metric_value=None, **kwargs):
 124     try:
 125         action = AlertRuleTriggerAction.objects.select_related(
 126             "alert_rule_trigger", "alert_rule_trigger__alert_rule"
 127         ).get(id=action_id)
 128     except AlertRuleTriggerAction.DoesNotExist:
 129         metrics.incr("incidents.alert_rules.action.skipping_missing_action")
 130         return
 131     try:
 132         incident = Incident.objects.select_related("organization").get(id=incident_id)
 133     except Incident.DoesNotExist:
 134         metrics.incr("incidents.alert_rules.action.skipping_missing_incident")
 135         return
 136 
 137     try:
 138         project = Project.objects.get(id=project_id)
 139     except Project.DoesNotExist:
 140         metrics.incr("incidents.alert_rules.action.skipping_missing_project")
 141         return
 142 
 143     metrics.incr(
 144         "incidents.alert_rules.action.{}.{}".format(
 145             AlertRuleTriggerAction.Type(action.type).name.lower(), method
 146         )
 147     )
 148     getattr(action, method)(incident, project, metric_value=metric_value)

 151 @instrumented_task(
 152     name="sentry.incidents.tasks.auto_resolve_snapshot_incidents",
 153     queue="incidents",
 154     default_retry_delay=60,
 155     max_retries=2,
 156 )
 157 def auto_resolve_snapshot_incidents(alert_rule_id, **kwargs):
 158     from sentry.incidents.models import AlertRule
 159     from sentry.incidents.logic import update_incident_status
 160 
 161     try:
 162         alert_rule = AlertRule.objects_with_snapshots.get(id=alert_rule_id)
 163     except AlertRule.DoesNotExist:
 164         return
 165 
 166     if alert_rule.status != AlertRuleStatus.SNAPSHOT.value:
 167         return
 168 
 169     batch_size = 50
 170     incidents = Incident.objects.filter(alert_rule=alert_rule).exclude(
 171         status=IncidentStatus.CLOSED.value
 172     )[: batch_size + 1]
 173     has_more = incidents.count() > batch_size
 174     if incidents:
 175         incidents = incidents[:batch_size]
 176         for incident in incidents:
 177             update_incident_status(
 178                 incident,
 179                 IncidentStatus.CLOSED,
 180                 comment="This alert has been auto-resolved because the rule that triggered it has been modified or deleted.",
 181                 status_method=IncidentStatusMethod.RULE_UPDATED,
 182             )
 183 
 184     if has_more:
 185         auto_resolve_snapshot_incidents.apply_async(
 186             kwargs={"alert_rule_id": alert_rule_id}, countdown=1
 187         )

 190 @instrumented_task(
 191     name="sentry.incidents.tasks.process_pending_incident_snapshots", queue="incident_snapshots"
 192 )
 193 def process_pending_incident_snapshots():
 194     """
 195     Processes PendingIncidentSnapshots and creates a snapshot for any snapshot that
 196     has passed it's target_run_date.
 197     """
 198     from sentry.incidents.logic import create_incident_snapshot
 199 
 200     batch_size = 50
 201 
 202     now = timezone.now()
 203     pending_snapshots = PendingIncidentSnapshot.objects.filter(
 204         target_run_date__lte=now
 205     ).select_related("incident")
 206 
 207     if not pending_snapshots:
 208         return
 209 
 210     for processed, pending_snapshot in enumerate(pending_snapshots):
 211         incident = pending_snapshot.incident
 212         if processed > batch_size:
 213             process_pending_incident_snapshots.apply_async(countdown=1)
 214             break
 215         else:
 216             with transaction.atomic():
 217                 if (
 218                     incident.status == IncidentStatus.CLOSED.value
 219                     and not IncidentSnapshot.objects.filter(incident=incident).exists()
 220                 ):
 221                     create_incident_snapshot(incident, windowed_stats=True)
 222                 pending_snapshot.delete()
