  28     def get(self, request, organization):

  30         Fetches actions that an alert rule can perform for an organization

  32         if not features.has("organizations:incidents", organization, actor=request.user):
  33             raise ResourceDoesNotExist
  34 
  35         actions = []
  36 
  37         integrations = get_available_action_integrations_for_org(organization).order_by("id")
  38         provider_integrations = defaultdict(list)
  39         for integration in integrations:
  40             provider_integrations[integration.provider].append(integration)
  41 
  42         registered_types = AlertRuleTriggerAction.get_registered_types()
  43         registered_types.sort(key=lambda x: x.slug)
  44 
  45         for registered_type in AlertRuleTriggerAction.get_registered_types():
  46             if registered_type.integration_provider:
  47                 for integration in provider_integrations[registered_type.integration_provider]:
  48                     actions.append(self.build_action_response(registered_type, integration))
  49             else:
  50                 actions.append(self.build_action_response(registered_type))
  51 
  52         return Response(actions, status=status.HTTP_200_OK)
