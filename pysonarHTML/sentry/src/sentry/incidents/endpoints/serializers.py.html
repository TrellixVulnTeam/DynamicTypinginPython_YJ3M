  64 class AlertRuleTriggerActionSerializer(CamelSnakeModelSerializer):
  65     """
  66     Serializer for creating/updating a trigger action. Required context:
  67      - `trigger`: The trigger related to this action.
  68      - `alert_rule`: The alert_rule related to this action.
  69      - `organization`: The organization related to this action.
  70      - `access`: An access object (from `request.access`)
  71     """
  72 
  73     id = serializers.IntegerField(required=False)
  74     type = serializers.CharField()
  75     target_type = serializers.CharField()
  76 
  77     class Meta:
  78         model = AlertRuleTriggerAction
  79         fields = ["id", "type", "target_type", "target_identifier", "integration"]
  80         extra_kwargs = {
  81             "target_identifier": {"required": True},
  82             "target_display": {"required": False},
  83             "integration": {"required": False, "allow_null": True},
  84         }
  85 
  86     def validate_type(self, type):
  87         if type not in string_to_action_type:
  88             raise serializers.ValidationError(
  89                 "Invalid type, valid values are [%s]" % ", ".join(string_to_action_type.keys())
  90             )
  91         return string_to_action_type[type]
  92 
  93     def validate_target_type(self, target_type):
  94         if target_type not in string_to_action_target_type:
  95             raise serializers.ValidationError(
  96                 "Invalid targetType, valid values are [%s]"
  97                 % ", ".join(string_to_action_target_type.keys())
  98             )
  99         return string_to_action_target_type[target_type]
 100 
 101     def validate(self, attrs):
 102         if ("type" in attrs) != ("target_type" in attrs) != ("target_identifier" in attrs):
 103             raise serializers.ValidationError(
 104                 "type, targetType and targetIdentifier must be passed together"
 105             )
 106         type = attrs.get("type")
 107         target_type = attrs.get("target_type")
 108         access = self.context["access"]
 109         identifier = attrs.get("target_identifier")
 110 
 111         if type is not None:
 112             type_info = AlertRuleTriggerAction.get_registered_type(type)
 113             if target_type not in type_info.supported_target_types:
 114                 allowed_target_types = ",".join(
 115                     [
 116                         action_target_type_to_string[type_name]
 117                         for type_name in type_info.supported_target_types
 118                     ]
 119                 )
 120                 raise serializers.ValidationError(
 121                     {
 122                         "target_type": "Invalid target type for %s. Valid types are [%s]"
 123                         % (type_info.slug, allowed_target_types)
 124                     }
 125                 )
 126 
 127         if attrs.get("type") == AlertRuleTriggerAction.Type.EMAIL:
 128             if target_type == AlertRuleTriggerAction.TargetType.TEAM:
 129                 try:
 130                     team = Team.objects.get(id=identifier)
 131                 except Team.DoesNotExist:
 132                     raise serializers.ValidationError("Team does not exist")
 133                 if not access.has_team(team):
 134                     raise serializers.ValidationError("Team does not exist")
 135             elif target_type == AlertRuleTriggerAction.TargetType.USER:
 136                 try:
 137                     user = User.objects.get(id=identifier)
 138                 except User.DoesNotExist:
 139                     raise serializers.ValidationError("User does not exist")
 140 
 141                 if not OrganizationMember.objects.filter(
 142                     organization=self.context["organization"], user=user
 143                 ).exists():
 144                     raise serializers.ValidationError("User does not belong to this organization")
 145         elif attrs.get("type") == AlertRuleTriggerAction.Type.SLACK:
 146             if not attrs.get("integration"):
 147                 raise serializers.ValidationError(
 148                     {"integration": "Integration must be provided for slack"}
 149                 )
 150 
 151         return attrs
 152 
 153     def create(self, validated_data):
 154         try:
 155             return create_alert_rule_trigger_action(
 156                 trigger=self.context["trigger"], **validated_data
 157             )
 158         except InvalidTriggerActionError as e:
 159             raise serializers.ValidationError(e.message)
 160 
 161     def update(self, instance, validated_data):
 162         if "id" in validated_data:
 163             validated_data.pop("id")
 164         try:
 165             return update_alert_rule_trigger_action(instance, **validated_data)
 166         except InvalidTriggerActionError as e:
 167             raise serializers.ValidationError(e.message)

 170 class AlertRuleTriggerSerializer(CamelSnakeModelSerializer):
 171     """
 172     Serializer for creating/updating an alert rule trigger. Required context:
 173      - `alert_rule`: The alert_rule related to this trigger.
 174      - `organization`: The organization related to this trigger.
 175      - `access`: An access object (from `request.access`)
 176     """
 177 
 178     id = serializers.IntegerField(required=False)
 179 
 180     # TODO: These might be slow for many projects, since it will query for each
 181     # individually. If we find this to be a problem then we can look into batching.
 182     excluded_projects = serializers.ListField(child=ProjectField(), required=False)
 183     actions = serializers.ListField(required=True)
 184 
 185     class Meta:
 186         model = AlertRuleTrigger
 187         fields = [
 188             "id",
 189             "label",
 190             "threshold_type",
 191             "alert_threshold",
 192             "resolve_threshold",
 193             "excluded_projects",
 194             "actions",
 195         ]
 196         extra_kwargs = {"label": {"min_length": 1, "max_length": 64}}

 198     def validate_threshold_type(self, threshold_type):
 199         try:
 200             return AlertRuleThresholdType(threshold_type)
 201         except ValueError:
 202             raise serializers.ValidationError(
 203                 "Invalid threshold type, valid values are %s"
 204                 % [item.value for item in AlertRuleThresholdType]
 205             )

 207     def create(self, validated_data):
 208         try:
 209             actions = validated_data.pop("actions")
 210             alert_rule_trigger = create_alert_rule_trigger(
 211                 alert_rule=self.context["alert_rule"], **validated_data
 212             )
 213             self._handle_actions(alert_rule_trigger, actions)
 214 
 215             return alert_rule_trigger
 216         except AlertRuleTriggerLabelAlreadyUsedError:
 217             raise serializers.ValidationError("This label is already in use for this alert rule")
 218 
 219     def update(self, instance, validated_data):
 220         actions = validated_data.pop("actions")
 221         if "id" in validated_data:
 222             validated_data.pop("id")
 223         try:
 224             alert_rule_trigger = update_alert_rule_trigger(instance, **validated_data)
 225             self._handle_actions(alert_rule_trigger, actions)
 226             return alert_rule_trigger
 227         except AlertRuleTriggerLabelAlreadyUsedError:
 228             raise serializers.ValidationError("This label is already in use for this alert rule")
 229 
 230     def _handle_actions(self, alert_rule_trigger, actions):
 231         if actions is not None:
 232             # Delete actions we don't have present in the updated data.
 233             action_ids = [x["id"] for x in actions if "id" in x]
 234             actions_to_delete = AlertRuleTriggerAction.objects.filter(
 235                 alert_rule_trigger=alert_rule_trigger
 236             ).exclude(id__in=action_ids)
 237             for action in actions_to_delete:
 238                 delete_alert_rule_trigger_action(action)
 239 
 240             for action_data in actions:
 241                 if "integration_id" in action_data:
 242                     action_data["integration"] = action_data.pop("integration_id")
 243 
 244                 if "id" in action_data:
 245                     action_instance = AlertRuleTriggerAction.objects.get(
 246                         alert_rule_trigger=alert_rule_trigger, id=action_data["id"]
 247                     )
 248                 else:
 249                     action_instance = None
 250 
 251                 action_serializer = AlertRuleTriggerActionSerializer(
 252                     context={
 253                         "alert_rule": alert_rule_trigger.alert_rule,
 254                         "trigger": alert_rule_trigger,
 255                         "organization": self.context["organization"],
 256                         "access": self.context["access"],
 257                     },
 258                     instance=action_instance,
 259                     data=action_data,
 260                 )
 261 
 262                 if action_serializer.is_valid():
 263                     action_serializer.save()
 264                 else:
 265                     raise serializers.ValidationError(action_serializer.errors)

 273 class AlertRuleSerializer(CamelSnakeModelSerializer):
 274     """
 275     Serializer for creating/updating an alert rule. Required context:
 276      - `organization`: The organization related to this alert rule.
 277      - `access`: An access object (from `request.access`)
 278     """
 279 
 280     environment = EnvironmentField(required=False, allow_null=True)
 281     # TODO: These might be slow for many projects, since it will query for each
 282     # individually. If we find this to be a problem then we can look into batching.
 283     projects = serializers.ListField(child=ProjectField(), required=False)
 284     excluded_projects = serializers.ListField(child=ProjectField(), required=False)
 285     triggers = serializers.ListField(required=True)
 286     dataset = serializers.CharField(required=False)
 287     query = serializers.CharField(required=True, allow_blank=True)
 288     time_window = serializers.IntegerField(
 289         required=True, min_value=1, max_value=int(timedelta(days=1).total_seconds() / 60)
 290     )
 291     threshold_period = serializers.IntegerField(default=1, min_value=1, max_value=20)
 292     aggregate = serializers.CharField(required=True, min_length=1)
 293 
 294     class Meta:
 295         model = AlertRule
 296         fields = [
 297             "name",
 298             "dataset",
 299             "query",
 300             "time_window",
 301             "environment",
 302             "threshold_period",
 303             "aggregate",
 304             "projects",
 305             "include_all_projects",
 306             "excluded_projects",
 307             "triggers",
 308         ]
 309         extra_kwargs = {
 310             "name": {"min_length": 1, "max_length": 64},
 311             "include_all_projects": {"default": False},
 312         }
 313 
 314     def validate_aggregate(self, aggregate):
 315         try:
 316             if not check_aggregate_column_support(aggregate):
 317                 raise serializers.ValidationError(
 318                     "Invalid Metric: We do not currently support this field."
 319                 )
 320         except InvalidSearchQuery as e:
 321             raise serializers.ValidationError("Invalid Metric: {}".format(e.message))
 322         return translate_aggregate_field(aggregate)
 323 
 324     def validate_dataset(self, dataset):
 325         try:
 326             return QueryDatasets(dataset)
 327         except ValueError:
 328             raise serializers.ValidationError(
 329                 "Invalid dataset, valid values are %s" % [item.value for item in QueryDatasets]
 330             )
 331 
 332     def validate(self, data):
 333         """Performs validation on an alert rule's data
 334         This includes ensuring there is either 1 or 2 triggers, which each have actions, and have proper thresholds set.
 335         The critical trigger should both alert and resolve 'after' the warning trigger (whether that means > or < the value depends on threshold type).
 336         """
 337         data.setdefault("dataset", QueryDatasets.EVENTS)
 338         project_id = data.get("projects")
 339         if not project_id:
 340             # We just need a valid project id from the org so that we can verify
 341             # the query. We don't use the returned data anywhere, so it doesn't
 342             # matter which.
 343             project_id = list(self.context["organization"].project_set.all()[:1])
 344         try:
 345             snuba_filter = build_snuba_filter(
 346                 data["dataset"],
 347                 data["query"],
 348                 data["aggregate"],
 349                 data.get("environment"),
 350                 params={
 351                     "project_id": [p.id for p in project_id],
 352                     "start": timezone.now() - timedelta(minutes=10),
 353                     "end": timezone.now(),
 354                 },
 355             )
 356         except (InvalidSearchQuery, ValueError) as e:
 357             raise serializers.ValidationError("Invalid Query or Metric: {}".format(e.message))
 358         else:
 359             if not snuba_filter.aggregations:
 360                 raise serializers.ValidationError(
 361                     "Invalid Metric: Please pass a valid function for aggregation"
 362                 )
 363 
 364             try:
 365                 raw_query(
 366                     aggregations=snuba_filter.aggregations,
 367                     start=snuba_filter.start,
 368                     end=snuba_filter.end,
 369                     conditions=snuba_filter.conditions,
 370                     filter_keys=snuba_filter.filter_keys,
 371                     having=snuba_filter.having,
 372                     dataset=Dataset(data["dataset"].value),
 373                     limit=1,
 374                     referrer="alertruleserializer.test_query",
 375                 )
 376             except Exception:
 377                 logger.exception("Error while validating snuba alert rule query")
 378                 raise serializers.ValidationError(
 379                     "Invalid Query or Metric: An error occurred while attempting "
 380                     "to run the query"
 381                 )
 382 
 383         triggers = data.get("triggers", [])
 384         if not triggers:
 385             raise serializers.ValidationError("Must include at least one trigger")
 386         if len(triggers) > 2:
 387             raise serializers.ValidationError(
 388                 "Must send 1 or 2 triggers - A critical trigger, and an optional warning trigger"
 389             )
 390 
 391         for i, (trigger, expected_label) in enumerate(
 392             zip(triggers, (CRITICAL_TRIGGER_LABEL, WARNING_TRIGGER_LABEL))
 393         ):
 394             if trigger.get("label", None) != expected_label:
 395                 raise serializers.ValidationError(
 396                     'Trigger {} must be labeled "{}"'.format(i + 1, expected_label)
 397                 )
 398         critical = triggers[0]
 399         self._validate_trigger_thresholds(critical)
 400 
 401         if len(triggers) == 2:
 402             warning = triggers[1]
 403             if critical["threshold_type"] != warning["threshold_type"]:
 404                 raise serializers.ValidationError(
 405                     "Must have matching threshold types (i.e. critical and warning "
 406                     "triggers must both be an upper or lower bound)"
 407                 )
 408             self._validate_trigger_thresholds(warning)
 409             self._validate_critical_warning_triggers(critical, warning)
 410 
 411         # Triggers have passed checks. Check that all triggers have at least one action now.
 412         for trigger in triggers:
 413             actions = trigger.get("actions")
 414             if not actions:
 415                 raise serializers.ValidationError(
 416                     '"' + trigger["label"] + '" trigger must have an action.'
 417                 )
 418 
 419         return data
 420 
 421     def _validate_trigger_thresholds(self, trigger):
 422         if trigger.get("resolve_threshold") is None:
 423             return
 424         # Since we're comparing non-inclusive thresholds here (>, <), we need
 425         # to modify the values when we compare. An example of why:
 426         # Alert > 0, resolve < 1. This means that we want to alert on values
 427         # of 1 or more, and resolve on values of 0 or less. This is valid, but
 428         # without modifying the values, this boundary case will fail.
 429         if trigger["threshold_type"] == AlertRuleThresholdType.ABOVE.value:
 430             alert_op, alert_add, resolve_add = operator.lt, 1, -1
 431         else:
 432             alert_op, alert_add, resolve_add = operator.gt, -1, 1
 433 
 434         if alert_op(
 435             trigger["alert_threshold"] + alert_add, trigger["resolve_threshold"] + resolve_add
 436         ):
 437             raise serializers.ValidationError(
 438                 "{} alert threshold must be above resolution threshold".format(trigger["label"])
 439             )
 440 
 441     def _validate_critical_warning_triggers(self, critical, warning):
 442         if critical["threshold_type"] == AlertRuleThresholdType.ABOVE.value:
 443             alert_op = operator.lt
 444             threshold_type = "above"
 445         elif critical["threshold_type"] == AlertRuleThresholdType.BELOW.value:
 446             alert_op = operator.gt
 447             threshold_type = "below"
 448 
 449         if alert_op(critical["alert_threshold"], warning["alert_threshold"]):
 450             raise serializers.ValidationError(
 451                 "Critical trigger must have an alert threshold {} warning trigger".format(
 452                     threshold_type
 453                 )
 454             )
 455         elif alert_op(critical["resolve_threshold"], warning["resolve_threshold"]):
 456             raise serializers.ValidationError(
 457                 "Critical trigger must have a resolution threshold {} (or equal to) "
 458                 "warning trigger".format(threshold_type)
 459             )
 460 
 461     def create(self, validated_data):

 473     def update(self, instance, validated_data):
 474         triggers = validated_data.pop("triggers")
 475         if "id" in validated_data:
 476             validated_data.pop("id")
 477         try:
 478             with transaction.atomic():
 479                 alert_rule = update_alert_rule(instance, **validated_data)
 480                 self._handle_triggers(alert_rule, triggers)
 481                 return alert_rule
 482         except AlertRuleNameAlreadyUsedError:
 483             raise serializers.ValidationError("This name is already in use for this organization")
 484 
 485     def _handle_triggers(self, alert_rule, triggers):
 486         if triggers is not None:
 487             # Delete triggers we don't have present in the incoming data
 488             trigger_ids = [x["id"] for x in triggers if "id" in x]
 489             triggers_to_delete = AlertRuleTrigger.objects.filter(alert_rule=alert_rule).exclude(
 490                 id__in=trigger_ids
 491             )
 492             for trigger in triggers_to_delete:
 493                 delete_alert_rule_trigger(trigger)
 494 
 495             for trigger_data in triggers:
 496                 if "id" in trigger_data:
 497                     trigger_instance = AlertRuleTrigger.objects.get(
 498                         alert_rule=alert_rule, id=trigger_data["id"]
 499                     )
 500                 else:
 501                     trigger_instance = None
 502 
 503                 trigger_serializer = AlertRuleTriggerSerializer(
 504                     context={
 505                         "alert_rule": alert_rule,
 506                         "organization": self.context["organization"],
 507                         "access": self.context["access"],
 508                     },
 509                     instance=trigger_instance,
 510                     data=trigger_data,
 511                 )
 512 
 513                 if trigger_serializer.is_valid():
 514                     trigger_serializer.save()
 515                 else:
 516                     raise serializers.ValidationError(trigger_serializer.errors)
