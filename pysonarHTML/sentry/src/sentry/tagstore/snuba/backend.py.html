  96         if environment_id:
  97             filters["environment"] = [environment_id]
  98         if group_id is not None:
  99             filters["group_id"] = [group_id]

 101         aggregations = kwargs.get("aggregations", [])

 103         conditions.append([tag, "!=", ""])
 104         aggregations += [
 105             ["uniq", tag, "values_seen"],
 106             ["count()", "", "count"],

 115             conditions=conditions,

 118             orderby="-count",
 119             limit=limit,

 121             referrer="tagstore.__get_tag_key_and_top_values",
 122         )

 173     def __get_tag_keys_for_projects(
 174         self,
 175         projects,
 176         group_id,
 177         environments,
 178         start,
 179         end,
 180         limit=1000,
 181         keys=None,
 182         include_values_seen=True,
 183         use_cache=False,
 184         **kwargs
 185     ):
 186         """ Query snuba for tag keys based on projects
 187 
 188             When use_cache is passed, we'll attempt to use the cache. There's an exception if group_id was passed
 189             which refines the query enough caching isn't required.
 190             The cache key is based on the filters being passed so that different queries don't hit the same cache, with
 191             exceptions for start and end dates. Since even a microsecond passing would result in a different caching
 192             key, which means always missing the cache.
 193             Instead, to keep the cache key the same for a short period we append the duration, and the end time rounded
 194             with a certain jitter to the cache key.
 195             This jitter is based on the hash of the key before duration/end time is added for consistency per query.
 196             The jitter's intent is to avoid a dogpile effect of many queries being invalidated at the same time.
 197             This is done by changing the rounding of the end key to a random offset. See snuba.quantize_time for
 198             further explanation of how that is done.
 199         """
 200         default_start, default_end = default_start_end_dates()
 201         if start is None:
 202             start = default_start
 203         if end is None:
 204             end = default_end
 205 
 206         filters = {"project_id": sorted(projects)}
 207         if environments:
 208             filters["environment"] = sorted(environments)
 209         if group_id is not None:
 210             filters["group_id"] = [group_id]
 211         if keys is not None:
 212             filters["tags_key"] = sorted(keys)
 213         aggregations = [["count()", "", "count"]]
 214 
 215         if include_values_seen:
 216             aggregations.append(["uniq", "tags_value", "values_seen"])
 217         conditions = []
 218 
 219         should_cache = use_cache and group_id is None
 220         result = None
 221 
 222         if should_cache:
 223             filtering_strings = [
 224                 u"{}={}".format(key, value) for key, value in six.iteritems(filters)
 225             ]
 226             cache_key = u"tagstore.__get_tag_keys:{}".format(
 227                 md5_text(*filtering_strings).hexdigest()
 228             )
 229             key_hash = hash(cache_key)
 230             should_cache = (key_hash % 1000) / 1000.0 <= options.get(
 231                 "snuba.tagstore.cache-tagkeys-rate"
 232             )
 233 
 234         # If we want to continue attempting to cache after checking against the cache rate
 235         if should_cache:
 236             # Needs to happen before creating the cache suffix otherwise rounding will cause different durations
 237             duration = (end - start).total_seconds()
 238             # Cause there's rounding to create this cache suffix, we want to update the query end so results match
 239             end = snuba.quantize_time(end, key_hash)
 240             cache_key += u":{}@{}".format(duration, end.isoformat())
 241             result = cache.get(cache_key, None)
 242             if result is not None:
 243                 metrics.incr("testing.tagstore.cache_tag_key.hit")
 244             else:
 245                 metrics.incr("testing.tagstore.cache_tag_key.miss")
 246 
 247         if result is None:
 248             result = snuba.query(
 249                 start=start,
 250                 end=end,
 251                 groupby=["tags_key"],
 252                 conditions=conditions,
 253                 filter_keys=filters,
 254                 aggregations=aggregations,
 255                 limit=limit,
 256                 orderby="-count",
 257                 referrer="tagstore.__get_tag_keys",
 258                 **kwargs
 259             )
 260             if should_cache:
 261                 cache.set(cache_key, result, 300)
 262                 metrics.incr("testing.tagstore.cache_tag_key.len", amount=len(result))
 263 
 264         if group_id is None:
 265             ctor = TagKey
 266         else:
 267             ctor = functools.partial(GroupTagKey, group_id=group_id)
 268 
 269         results = set()
 270         for key, data in six.iteritems(result):
 271             params = {"key": key}
 272             if include_values_seen:
 273                 params["values_seen"] = data["values_seen"]
 274                 params["count"] = data["count"]
 275             else:
 276                 # If only one aggregate is requested then data is just that raw
 277                 # aggregate value, rather than a dictionary of
 278                 # key:aggregate_value pairs
 279                 params["count"] = data
 280             results.add(ctor(**params))
 281         return results

 322     def get_tag_keys_for_projects(
 323         self, projects, environments, start, end, status=TagKeyStatus.VISIBLE, use_cache=False
 324     ):
 325         MAX_UNSAMPLED_PROJECTS = 50
 326         # We want to disable FINAL in the snuba query to reduce load.
 327         optimize_kwargs = {"turbo": True}
 328         # If we are fetching less than MAX_UNSAMPLED_PROJECTS, then disable
 329         # the sampling that turbo enables so that we get more accurate results.
 330         # We only want sampling when we have a large number of projects, so
 331         # that we don't cause performance issues for Snuba.
 332         if len(projects) <= MAX_UNSAMPLED_PROJECTS:
 333             optimize_kwargs["sample"] = 1
 334         return self.__get_tag_keys_for_projects(
 335             projects,
 336             None,
 337             environments,
 338             start,
 339             end,
 340             include_values_seen=False,
 341             use_cache=use_cache,
 342             **optimize_kwargs
 343         )

 456     def get_group_tag_keys_and_top_values(
 457         self,
 458         project_id,
 459         group_id,
 460         environment_ids,
 461         user=None,
 462         keys=None,
 463         value_limit=TOP_VALUES_DEFAULT_LIMIT,
 464         **kwargs
 465     ):
 466         # Similar to __get_tag_key_and_top_values except we get the top values
 467         # for all the keys provided. value_limit in this case means the number
 468         # of top values for each key, so the total rows returned should be
 469         # num_keys * limit.
 470 
 471         # First get totals and unique counts by key.
 472         keys_with_counts = self.get_group_tag_keys(project_id, group_id, environment_ids, keys=keys)
 473 
 474         # Then get the top values with first_seen/last_seen/count for each
 475         filters = {"project_id": get_project_list(project_id)}
 476         if environment_ids:
 477             filters["environment"] = environment_ids
 478         if keys is not None:
 479             filters["tags_key"] = keys
 480         if group_id is not None:
 481             filters["group_id"] = [group_id]
 482         conditions = kwargs.get("conditions", [])
 483         aggregations = kwargs.get("aggregations", [])
 484         aggregations += [
 485             ["count()", "", "count"],
 486             ["min", SEEN_COLUMN, "first_seen"],
 487             ["max", SEEN_COLUMN, "last_seen"],
 488         ]
 489 
 490         values_by_key = snuba.query(
 491             start=kwargs.get("start"),
 492             end=kwargs.get("end"),
 493             groupby=["tags_key", "tags_value"],
 494             conditions=conditions,
 495             filter_keys=filters,
 496             aggregations=aggregations,
 497             orderby="-count",
 498             limitby=[value_limit, "tags_key"],
 499             referrer="tagstore.__get_tag_keys_and_top_values",
 500         )
 501 
 502         # Then supplement the key objects with the top values for each.
 503         if group_id is None:
 504             value_ctor = TagValue
 505         else:
 506             value_ctor = functools.partial(GroupTagValue, group_id=group_id)
 507 
 508         for keyobj in keys_with_counts:
 509             key = keyobj.key
 510             values = values_by_key.get(key, [])
 511             keyobj.top_values = [
 512                 value_ctor(
 513                     key=keyobj.key,
 514                     value=value,
 515                     times_seen=data["count"],
 516                     first_seen=parse_datetime(data["first_seen"]),
 517                     last_seen=parse_datetime(data["last_seen"]),
 518                 )
 519                 for value, data in six.iteritems(values)
 520             ]
 521 
 522         return keys_with_counts

 669     def get_tag_value_paginator_for_projects(
 670         self, projects, environments, key, start=None, end=None, query=None, order_by="-last_seen"
 671     ):
 672         from sentry.api.paginator import SequencePaginator
 673 
 674         if not order_by == "-last_seen":
 675             raise ValueError("Unsupported order_by: %s" % order_by)
 676 
 677         dataset = Dataset.Events
 678         snuba_key = snuba.get_snuba_column_name(key)
 679         if snuba_key.startswith("tags["):
 680             snuba_key = snuba.get_snuba_column_name(key, dataset=Dataset.Discover)
 681             if not snuba_key.startswith("tags["):
 682                 dataset = Dataset.Discover
 683 
 684         conditions = []
 685 
 686         # transaction status needs a special case so that the user interacts with the names and not codes
 687         transaction_status = snuba_key == "transaction_status"
 688         if transaction_status:
 689             conditions.append(
 690                 [
 691                     snuba_key,
 692                     "IN",
 693                     # Here we want to use the status codes during filtering,
 694                     # but want to do this with names that include our query
 695                     [
 696                         span_key
 697                         for span_key, value in six.iteritems(SPAN_STATUS_CODE_TO_NAME)
 698                         if (query and query in value) or (not query)
 699                     ],
 700                 ]
 701             )
 702         elif key in FUZZY_NUMERIC_KEYS:
 703             converted_query = int(query) if query is not None and query.isdigit() else None
 704             if converted_query is not None:
 705                 conditions.append([snuba_key, ">=", converted_query - FUZZY_NUMERIC_DISTANCE])
 706                 conditions.append([snuba_key, "<=", converted_query + FUZZY_NUMERIC_DISTANCE])
 707         else:
 708             if snuba_key in BLACKLISTED_COLUMNS:
 709                 snuba_key = "tags[%s]" % (key,)
 710 
 711             if query:
 712                 conditions.append([snuba_key, "LIKE", u"%{}%".format(query)])
 713             else:
 714                 conditions.append([snuba_key, "!=", ""])
 715 
 716         filters = {"project_id": projects}
 717         if environments:
 718             filters["environment"] = environments
 719 
 720         results = snuba.query(
 721             dataset=dataset,
 722             start=start,
 723             end=end,
 724             groupby=[snuba_key],
 725             filter_keys=filters,
 726             aggregations=[
 727                 ["count()", "", "times_seen"],
 728                 ["min", "timestamp", "first_seen"],
 729                 ["max", "timestamp", "last_seen"],
 730             ],
 731             conditions=conditions,
 732             orderby=order_by,
 733             # TODO: This means they can't actually paginate all TagValues.
 734             limit=1000,
 735             arrayjoin=snuba.get_arrayjoin(snuba_key),
 736             referrer="tagstore.get_tag_value_paginator_for_projects",
 737         )
 738 
 739         # With transaction_status we need to map the ids back to their names
 740         if transaction_status:
 741             results = OrderedDict(
 742                 [
 743                     (SPAN_STATUS_CODE_TO_NAME[result_key], value)
 744                     for result_key, value in six.iteritems(results)
 745                 ]
 746             )
 747 
 748         tag_values = [
 749             TagValue(key=key, value=six.text_type(value), **fix_tag_value_data(data))
 750             for value, data in six.iteritems(results)
 751         ]
 752 
 753         desc = order_by.startswith("-")
 754         score_field = order_by.lstrip("-")
 755         return SequencePaginator(
 756             [(int(to_timestamp(getattr(tv, score_field)) * 1000), tv) for tv in tag_values],
 757             reverse=desc,
 758         )

 760     def get_group_tag_value_iter(
 761         self, project_id, group_id, environment_ids, key, callbacks=(), limit=1000, offset=0
 762     ):
 763         filters = {
 764             "project_id": get_project_list(project_id),
 765             "tags_key": [key],
 766             "group_id": [group_id],
 767         }
 768         if environment_ids:
 769             filters["environment"] = environment_ids
 770         results = snuba.query(
 771             groupby=["tags_value"],
 772             filter_keys=filters,
 773             aggregations=[
 774                 ["count()", "", "times_seen"],
 775                 ["min", "timestamp", "first_seen"],
 776                 ["max", "timestamp", "last_seen"],
 777             ],
 778             orderby="-first_seen",  # Closest thing to pre-existing `-id` order
 779             limit=limit,
 780             referrer="tagstore.get_group_tag_value_iter",
 781             offset=offset,
 782         )
 783 
 784         group_tag_values = [
 785             GroupTagValue(group_id=group_id, key=key, value=value, **fix_tag_value_data(data))
 786             for value, data in six.iteritems(results)
 787         ]
 788 
 789         for cb in callbacks:
 790             cb(group_tag_values)
 791 
 792         return group_tag_values

 794     def get_group_tag_value_paginator(
 795         self, project_id, group_id, environment_ids, key, order_by="-id"
 796     ):
 797         from sentry.api.paginator import SequencePaginator
 798 
 799         if order_by in ("-last_seen", "-first_seen"):
 800             pass
 801         elif order_by == "-id":
 802             # Snuba has no unique id per GroupTagValue so we'll substitute `-first_seen`
 803             order_by = "-first_seen"
 804         else:
 805             raise ValueError("Unsupported order_by: %s" % order_by)
 806 
 807         group_tag_values = self.get_group_tag_value_iter(project_id, group_id, environment_ids, key)
 808 
 809         desc = order_by.startswith("-")
 810         score_field = order_by.lstrip("-")
 811         return SequencePaginator(
 812             [
 813                 (int(to_timestamp(getattr(gtv, score_field)) * 1000), gtv)
 814                 for gtv in group_tag_values
 815             ],
 816             reverse=desc,
 817         )

 824     def get_group_event_filter(self, project_id, group_id, environment_ids, tags, start, end):
 825         filters = {"project_id": get_project_list(project_id), "group_id": [group_id]}
 826         if environment_ids:
 827             filters["environment"] = environment_ids
 828 
 829         conditions = []
 830         for tag_name, tag_val in tags.items():
 831             operator = "IN" if isinstance(tag_val, list) else "="
 832             conditions.append([u"tags[{}]".format(tag_name), operator, tag_val])
 833 
 834         result = snuba.raw_query(
 835             start=start,
 836             end=end,
 837             selected_columns=["event_id"],
 838             conditions=conditions,
 839             orderby="-timestamp",
 840             filter_keys=filters,
 841             limit=1000,
 842             referrer="tagstore.get_group_event_filter",
 843         )
 844 
 845         event_id_set = set(row["event_id"] for row in result["data"])
 846 
 847         if not event_id_set:
 848             return None
 849 
 850         return {"event_id__in": event_id_set}
